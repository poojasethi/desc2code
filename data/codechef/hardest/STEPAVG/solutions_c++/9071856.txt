#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<set>
#include<vector>
using namespace std;
#define PROB "stepavg"
#define MAXN 10100
typedef double real;
real a[MAXN];
pair<real,int> stack[MAXN];
int tops=-1;
vector<pair<int,int> > res;

int main()
{
		int nn;
		scanf("%d",&nn);
		while (nn--)
		{
				int n;
				real t;
				scanf("%d",&n);
				scanf("%lf",&t);
				real t0=t;
				//fprintf(stderr,"%0lf\n",t);
				set<pair<real,int> > S;
				res.clear();
				for (int i=1;i<=n;i++)
				{
						real x;
						scanf("%lf",&x);
						a[i]=x;
						S.insert(make_pair(x,i));
				}
				int cnt=n;
				real alpha=0.001;
				while (S.size()>1)
				{
						pair<real,int> x=*S.begin();
						pair<real,int> y=*(--S.end());
						real mid=(x.first+y.first)/2;
						if (mid>t*(1+alpha))
						{
								stack[++tops]=x;
								t=2*t-x.first;
								S.erase(S.begin());
						}else if (mid<t*(1-alpha))
						{
								stack[++tops]=y;
								t=2*t-y.first;
								S.erase(--S.end());
										--++y.second;
						}else
						{
								res.push_back(make_pair(x.second,y.second));
								S.erase(S.begin());
								S.erase(--S.end());
								S.insert(make_pair((x.first+y.first)/2,++cnt));
						}
				}
				stack[++tops]=*S.begin();
				cnt=n;
				for (int i=0;i<(int)res.size();i++)
				{
						cout<<res[i].first<<" "<<res[i].second<<endl;
						a[++cnt]=(a[res[i].first]+a[res[i].second])/2;
						a[res[i].first]=a[res[i].second]=-1e100;
				}
				res.push_back(make_pair(stack[tops--].second,stack[tops--].second));
				a[++cnt]=(a[res.back().first]+a[res.back().second])/2;
				cout<<res.back().first<<" "<<res.back().second<<endl;
				a[res.back().first]=a[res.back().second]=-1e100;
				while (~tops)
				{
						res.push_back(make_pair(cnt,stack[tops--].second));
						cout<<res.back().first<<" "<<res.back().second<<endl;
						a[++cnt]=(a[res.back().first]+a[res.back().second])/2;
						a[res.back().first]=a[res.back().second]=-1e100;
				}
				fprintf(stderr,"%0lf\n",a[cnt]-t0);
		}
}
