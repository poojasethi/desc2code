#include <bits/stdc++.h>
using namespace std;
char a[1003],b[1003],d[1003];
int z[1003],z1[1003];
int main()
{
    int t,l1,l2,g=0;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%d %s",&l1,a);
    scanf("%d %s",&l2,b);
    int c=0,i,j,t[l1+1][l2+1];
    memset(t,0,sizeof(t));
    for(i=1;i<=l1;i++)
    {
        for(j=1;j<=l2;j++)
        {
            if(a[i-1]==b[j-1])
                t[i][j]=t[i-1][j-1]+1;
            else
                t[i][j]=max(t[i-1][j],t[i][j-1]);
            if(t[i][j]>c)
                c=t[i][j];
        }
    }
    g++;
    if(c>1)
    {
        printf("case %d Y\n",g);
        printf("%d\n",c);
    int l=-1,x=l1,y=l2;
    while(1)
    {
        if(a[x-1]==b[y-1])
        {
            l++;
            //printf("%c",b[y-1]);
            d[l]=b[y-1];
            z[l]=x;
            z1[l]=y;
            x--;
            y--;
        }
        else if(t[x-1][y]>=t[x][y-1])
            x--;
        else
            y--;
        if(x==0||y==0)
            break;
    }
    for(i=l;i>=0;i--)
    printf("%c %d %d\n",d[i],z[i],z1[i]);
    }
    else
        printf("case %d N\n",g);
    }
    return 0;
}
