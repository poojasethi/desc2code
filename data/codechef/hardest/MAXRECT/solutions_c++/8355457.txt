#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <algorithm>
using namespace std;
 
int n,m,i,j,k,x,y,I;
int a[305][305],Max[605],id[605];
long long sr[305],sc[305],ans,sum,valx,valy;
bool ur[305],uc[305],ar[305],ac[305];
 
inline bool cmp(const int &a,const int &b){return Max[a]>Max[b];}
 
void R(int x)
{
	int i;ur[x]^=1;
	if(ur[x])for(i=1;i<=m;++i)sc[i]+=a[x][i];
	else for(i=1;i<=m;++i)sc[i]-=a[x][i];
}
 
void C(int y)
{
	int i;uc[y]^=1;
	if(uc[y])for(i=1;i<=n;++i)sr[i]+=a[i][y];
	else for(i=1;i<=n;++i)sr[i]-=a[i][y];
}
 
int main()
{
	scanf("%d%d",&n,&m);
	for(i=1;i<=n+m;++i)id[i]=i;
	for(i=1;i<=n;++i)
	for(j=1;j<=m;++j)
	{
		scanf("%d",&a[i][j]);
		if(a[i][j]>Max[i])Max[i]=a[i][j];
		if(a[i][j]>Max[j+n])Max[j+n]=a[i][j];
	}
	sort(id+1,id+n+m+1,cmp);
	
	for(I=1;I<=n+m&&Max[id[I]]>0&&clock()<400000;++I)
	{
		sum=0;
		for(i=1;i<=n;++i)sr[i]=0,ur[i]=false;
		for(i=1;i<=m;++i)sc[i]=0,uc[i]=false;
		if(id[I]<=n)R(id[I]);else C(id[I]-n);
		for(;clock()<400000;)
		{
			for(i=1;i<=n;++i)
			if(ur[i])
			{
				if(-sr[i]>valx||i==1)
				valx=-sr[i],x=i;
			}
			else
			{
				if(sr[i]>valx||i==1)
				valx=sr[i],x=i;
			}
			for(i=1;i<=m;++i)
			if(uc[i])
			{
				if(-sc[i]>valy||i==1)
				valy=-sc[i],y=i;
			}
			else
			{
				if(sc[i]>valy||i==1)
				valy=sc[i],y=i;
			}
			if(valx<=0&&valy<=0)break;
			if(valx>valy)R(x),sum+=valx;else C(y),sum+=valy;
		}
		if(sum>=ans)
		{
			ans=sum;
			for(i=1;i<=n;++i)ar[i]=ur[i];
			for(i=1;i<=m;++i)ac[i]=uc[i];
		}
	}
	x=y=0;
	for(i=1;i<=n;++i)if(ar[i])++x;
	for(i=1;i<=m;++i)if(ac[i])++y;
	printf("%d %d\n",x,y);
	for(i=1;i<=n;++i)if(ar[i])printf("%d ",i-1);printf("\n");
	for(i=1;i<=m;++i)if(ac[i])printf("%d ",i-1);printf("\n");
}
