#include<cstdio>
#include<cstring>
#include<iostream>
#include<map>
#include<ctime>
using namespace std;
multimap<int,int>mapa;
int n,br,brr,maxx,a[300100],minn,eee,eee2,rezniz[300100][5],rqw;
long long zbir,raz;
bool uradjen[300100];
clock_t now,then;
int main(){

  now=clock();
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
    scanf("%d",&n);
    minn=1000000000;
    maxx=-1000000000;
    for(int i=1;i<=n*3;i++){
        scanf("%d",&a[i]);
        mapa.insert(pair<int,int>(a[i],i));
       maxx=max(maxx,a[i]);
       minn=min(minn,a[i]);
    }
    zbir=(maxx+minn)/2*3;
    eee=n/2;
    eee2=n/2+1;

    for(int i1=1;i1<=eee;i1++){
      if((uradjen[i1]))continue;

        for(int i2=n;i2>=eee2;i2--){

         if((uradjen[i2]))continue;
         raz=zbir-a[i1]-a[i2];

         multimap<int,int>::iterator it=mapa.find(raz);

         if(it!=mapa.end()){

          if(uradjen[it->second])continue;
          if((it->second==i1)or(it->second==i2))continue;

         rezniz[++brr][1]=i1;
         rezniz[brr][2]=i2;
         rezniz[brr][3]=it->second;
         uradjen[i1]=1;
         uradjen[i2]=1;
         uradjen[it->second]=1;
         mapa.erase(it);
         break;
        }

      }
      then=clock();
      if ((float)(then-now)/CLOCKS_PER_SEC>4.8) break;
    }

    cout<<brr<<endl;

    //cout<<zbir<<endl;
    memset(uradjen,0,sizeof(uradjen));
    for(int i=1;i<=brr;i++){
       //if(zbir==rezniz[i][1]+rezniz[i][2]+rezniz[i][3])return 666;
       //if((uradjen[rezniz[i][1]])or(uradjen[rezniz[i][2]])or(uradjen[rezniz[i][3]]))cout<<1/rqw;
       cout<<rezniz[i][1]<<" "<<rezniz[i][2]<<" "<<rezniz[i][3]<<" ";
       uradjen[rezniz[i][1]]=1;
       uradjen[rezniz[i][2]]=1;
       uradjen[rezniz[i][3]]=1;
    }

return 0;
}
