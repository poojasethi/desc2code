#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int t,i,a[1000000],n;
	scanf("%li",&t);
	
	while(t--)
	{
		scanf("%li",&n);
		for(i=0;i<n;i++)
		{
			scanf("%li",&a[i]);
		}
		sort(a,a+n);
		long int sum=0;
		for(i=0;i<n;i++)
		{
			sum=a[i]+a[i+1];
			break;
		}
		printf("%li\n",sum);
	}
	return 0;	
}
