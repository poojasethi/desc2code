#include<stdio.h>
#include<stdlib.h>

int cmpfunc(const void *a, const void *b){
	return(*(int *)a - *(int *)b);
}

int main(){
	int t, n, i; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		int a[n];
		for(i = 0; i < n; scanf("%d", &a[i++]));
		qsort(a, n, sizeof(int), cmpfunc), printf("%d\n", a[0] + a[1]);
	}		
	return 0;
}
