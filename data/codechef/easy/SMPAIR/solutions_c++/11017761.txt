#include<iostream>

using namespace std;

int main()
{
	int T,i,j,k,N,s1,s2,y,q;

	//cout << "write total no. of test cases\n";
	cin >> T;

	for (i=0;i<T;i++)
	{
		//cout << "write the vale of N\n";
		cin >> N;

		int A[N];

		//cout << "fill the arraay\n";
		cin >> A[0];
		s1 = A[0];
		y = 0;
		for (j=1;j<N;j++)
		{
			cin >> A[j];	
			if (s1 > A[j])
			{
				s1 = A[j];
				y = j;
			}
		}	
		if  (y==0)
		{
			q = 1;
		}
		else
		{
			q = 0;
		}
		s2 = A[q];

		for (j=0;j<N;j++)
		{
			if (j==y)
			{
				j++;
				if (j>=N)
				{
					break;
				}
			}
			if (s2 > A[j])
			{
				s2 = A[j];
			}
		}
		cout << s1 + s2 << endl;
	}

	return 0;
}
