#include <iostream>
#include<vector>
#include<stdio.h>
#include<climits>
using namespace std;
vector <int> vi;
int main()
{
        int t, n, num, min1=INT_MAX, min2=INT_MAX;

        scanf("%d", &t);

        while(t--){
              scanf("%d", &n);

                for(int i = 0; i<n; i++){
                        scanf("%d", &num);
                        vi.push_back(num);
                }

                if (vi[0] >= vi[1]){
                        min1 = vi[1];
                        min2 = vi[0];
                }
                else{
                        min1 = vi[0];
                        min2 = vi[1];
                }
                for(int i =2; i<n; i++){

                        if(min1 > vi[i]){
                                min2 = min1;
                                min1 = vi[i];
                        }

                        else if(min2 > vi[i]){
                                min2 = vi[i];
                        }
                }
                printf("%d\n", min1+min2);
                vi.clear();

        }
    return 0;
}
