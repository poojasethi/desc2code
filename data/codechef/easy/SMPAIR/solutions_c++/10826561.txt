#include<bits/stdc++.h>
using namespace std;

typedef long long int lli;
#define loop(i,t) for(i=0;i<t;++i)

int main()
{
    lli t,i,j,n,temp,m1,m2;
    cin>>t;
    loop(i,t)
    {
        cin>>n;
        cin>>m1>>m2;
        if(m1>m2)
            swap(m1,m2);
        loop(j,n-2)
        {
            cin>>temp;
            if(temp<=m1)
            {
                m2=m1;
                m1=temp;
            }
            else if(temp<m2)
            {
                m2=temp;
            }
        }
        cout<<m1+m2<<"\n";
    }
}
