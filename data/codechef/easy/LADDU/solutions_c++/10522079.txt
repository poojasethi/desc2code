int main() {

	ll tc,activities,sum,i;
	cin >> tc;
	while (tc--) {
		ll score = 0;
		string origin,activity;
		cin >> activities >> origin ;
		for (i = 0; i < activities; i++) {
			cin >> activity;
			if (activity.compare("CONTEST_WON") == 0) {
				ll rank;
				cin >> rank;
				if(rank <= 20)
					score += (300 + 20 - rank);
				else
					score += 300 ;
			}else if(activity.compare("TOP_CONTRIBUTOR")==0)
				score += 300;
			else if (activity.compare("BUG_FOUND") == 0) {
				ll severity;
				cin >> severity;
				score += severity;
			}else if(activity.compare("CONTEST_HOSTED") == 0)
				score += 50;
		}
		if(origin.compare("INDIAN")== 0)
			cout << score/200 << endl;
		else if(origin.compare("NON_INDIAN")==0)
			cout << score/400 << endl;
	}
}
