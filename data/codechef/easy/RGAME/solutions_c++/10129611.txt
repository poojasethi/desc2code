#include <bits/stdc++.h>
using namespace std;

const long long maxn = 1e6;
long long ar[maxn];
long long suf[maxn];
long long mod = 1e9 + 7;

long long exp(long long a, long long b) {
  long long res = 1;
  while (b > 0) {
    if (b & 1) {
      res *= a;
      if (res > mod) res %= mod;
    }
    a *= a;
    if (a > mod) a %= mod;
    b /= 2;
  }
  return res;
}

int main() {
  ios::sync_with_stdio(0);
  cin.tie(0);

  long long t; cin >> t;
  while (t--) {
    long long n; cin >> n;
    for (long long i = 0; i <= n; ++i) cin >> ar[i];
    suf[n + 1] = 0;
    for (long long i = n; i + 1; --i) {
      suf[i] = suf[i + 1] + ar[i] * exp(2, n - i);
      if (suf[i] > mod) suf[i] %= mod;
    }
    long long ans = 0;
    for (long long i = 0; i <= n; ++i) {
      ans += ((exp(2, i - 1) * ar[i]) % mod) * suf[i + 1];
      if (ans > mod) ans %= mod;
    }
    cout << (2 * ans) % mod << "\n";
  } 

  return 0;
}
