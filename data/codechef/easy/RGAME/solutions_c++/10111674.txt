#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int main()
{
    int t,pow2[100001];
    scanf("%i",&t);
    pow2[0]=1;
    for(int i=1;i<100001;i++)
        pow2[i]=(pow2[i-1]*2)%1000000007;
    while(t--)
    {
        int n,a[100001];
        long long int c[100001],s[100001];
        c[0]=0;
        scanf("%i",&n);
        for(int i=0;i<=n;i++)
            scanf("%i",&a[i]);
        s[1]=((long long int)2*a[0])%1000000007;
        for(int i=1;i<=n;i++)
        {
            s[i+1]=(s[i]+((long long int)pow2[i]*a[i])%1000000007)%1000000007;
            c[i]=(s[i]*a[i])%1000000007;
            c[i]+=(2*c[i-1])%1000000007;
            c[i]%=1000000007;
        }
        c[n]%=1000000007;
        printf("%i\n",c[n]);
    }
	return 0;
}
