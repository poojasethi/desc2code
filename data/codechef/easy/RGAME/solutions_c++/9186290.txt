#include<stdio.h>
#define M 1000000007
int main(){
	long long int t;
	scanf("%lld",&t);
	while(t--){
		long long int n,i,sum;
		scanf("%lld",&n);
	
		long long int a[n+1];
		for(i=0;i<=n;i++){
			scanf("%lld",&a[i]);
			
		}
		sum=(2*(a[1]*a[0]))%M;
		long long int power=1,term;
		term=(a[0]+a[1])%M;
		for(i=2;i<=n;i++){
			power=(power*2)%M;
			sum=(2*(sum+a[i]*term))%M;
			term=(term+(a[i]*power))%M;
		}
		printf("%lld\n",sum);
	}
	return 0;	
}
