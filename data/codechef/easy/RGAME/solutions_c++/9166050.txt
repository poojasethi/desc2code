#include<stdio.h>
#include<iostream>
#include<algorithm>
#define M 100005
#define P 1000000007
using namespace std;
int A[M],B[M]; 
int main(){
	int cas,n,i,j,k; 
	B[0]=1;
	for(i=1;i<M;i++)B[i]=(B[i-1]<<1)%P;
	scanf("%d",&cas);
	while(cas--){
		scanf("%d",&n);
		for(i=0;i<=n;i++)
			scanf("%d",&A[i]);
		int ans=0,sum=2*A[0],way=1;
		for(i=1;i<=n;i++){ 
			ans=(ans+1LL*A[i]*sum%P*B[n-i])%P;
			way=(way<<1)%P;
			sum=(sum+1LL*way*A[i])%P;	
		}
		printf("%d\n",ans);
	}
	return 0;
}
