#include<cstdio>
#define MOD 1000000007
using namespace std;
long long int l[100005],a[100005],ts,tmp,pow;
int main()
{
    int t,n,i,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<=n;i++)
        {
            scanf("%lld",&a[i]);
        }
        l[0]=a[0];

        pow=1;
        i=1;


        while(i<=n)
        {
            tmp=pow*a[i];
            tmp%=MOD;
            l[i]=tmp;
            l[i]+=l[i-1];
            l[i]%=MOD;
            pow<<=1;
            pow%=MOD;


            i++;
        }

        ts=0;
        i=1;
        while(i<=n)
        {
            tmp=l[i-1]*a[i];
            tmp%=MOD;
            ts+=tmp;
            ts<<=1;
            ts%=MOD;
            i++;
        }
        printf("%lld\n",ts);
    }
    return 0;
}
