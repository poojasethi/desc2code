#include<cstdio>
#define mod 1000000007
long long s[100005],a[100005],s1[100005];
int main()
{
	long long t,i,n,ans;
	scanf("%lld",&t);
	while(t--)
	{
		ans=0;
		scanf("%lld",&n);
		for(i=0;i<=n;i++)
		{
			scanf("%lld",&a[i]);
		}
		s1[0]=0;
		s[0]=a[0]*2;
		long long p=2;
		for(i=1;i<=n;i++)
		{
			s1[i]=(s[i-1]*a[i]+2*s1[i-1])%mod;
			s[i]=(s[i-1]+(p*a[i])%mod)%mod;
			p*=2;
			p%=mod;
			
		}		
		ans=s1[n];
		printf("%lld\n",ans);
	}
	return 0;
}
