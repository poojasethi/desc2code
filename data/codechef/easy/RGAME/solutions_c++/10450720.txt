#include<iostream>
#include<cmath>
#define D 1000000007
using namespace std;
int pwr[100000];

int main(){
unsigned long long  T,N,A,E,S;
cin>>T;
pwr[0] = 1;
	for (int i = 1; i < 100000; i++)
		pwr[i] = (pwr[i - 1] << 1)%D;

while(T--)
{ cin>>N;
  cin>>A;
    E=2*A;
    S=0;
 for(int i=1;i<=N;i++){
     cin>>A;
     S=(2*S+A*E)%D;
     E=(E+pwr[i]*A)%D;
     }
  cout<<S<<"\n";
}
return 0;
}
