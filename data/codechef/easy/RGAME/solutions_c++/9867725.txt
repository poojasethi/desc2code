#include <bits/stdc++.h>
using namespace std;
const long long M = 1000000007;
int main(){
    int t,n,i;
    scanf("%d",&t);
    long long p2[100001];
    p2[1]=2;
    for(i=2;i<=100000;++i){
        p2[i]=(p2[i-1]<<1) % M;
    }
    while(t--){
        scanf("%d",&n);
        long long s=0, a[n+1], co[n];
        for(int i=0;i<=n;++i) scanf("%lld",&a[i]);
 
        co[0]=a[n];
        for(i=1;i<=n-1;++i){
            co[i] = co[i-1]+p2[i]*a[n-i];
            co[i] %= M;
        }

        s = a[0]*co[n-1] % M;
        s <<= 1;
        s %= M;
        for(i=1;i<=n-1;++i){
            s += co[n-i-1]*a[i] % M * p2[i] % M;
            s %= M;
        }
        printf("%lld\n",s);
    }
    return 0;
}
