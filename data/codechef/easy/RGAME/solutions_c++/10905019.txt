const int N = (int)(1e5+4);

ll dp[N];
ll dp2[N];
ll val[N];
ll solve(int n,int *arr){
    memset(dp,0,sizeof dp);
    dp[0] = dp[1] = 1;
    for(int i = 2 ;i<=n;i++)
        dp[i] = (dp[i-1]*2)%mod;
     ll p2 = 1LL;
     ll tot_val = 0LL;
     ll cur = 0;
     ll val = 0LL;
     ll fval = 0;
     for(int i = n-1;i>=0;i--){
            p2 = (p2*2)%mod;
            val = (val%mod+((p2%mod*arr[i+1]%mod)%mod))%mod;
            fval = (arr[i]%mod*dp[i]%mod*val%mod)%mod;
            tot_val = (tot_val%mod+fval%mod)%mod;
          //  cout << val << endl;
     }
    return tot_val;
}
