#include <iostream>
#define mod 1000000007
using namespace std;
 
 
int main() {
	long int t,n;
	long long int a,b,c,d,k;
	cin>>t;
	while(t--){
	    k=0;
	    cin>>n;
	    cin>>a;
	    a=(2*a)%mod;
	    c=2;
	    for(long int i=0;i<n;i++){
	        cin>>b;
	        k=(k*2+a*b)%mod;
	        a=(a+b*c)%mod;
	        c=(c*2)%mod;
	    }
	    cout<<k<<endl;
	}
	return 0;
}
