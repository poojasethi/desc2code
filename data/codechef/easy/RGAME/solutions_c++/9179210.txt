#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define f first
#define maxn 100005
#define s second
#define ll long long int
#define inf 1000000014
#define infl (ll)(1e18)
#define mod 1000000007
using namespace std;
typedef pair<int,int>p;
typedef pair<p,int> pp;
ll arr[maxn];
ll pow2[maxn];
ll myans[maxn];
int main(int argc, char const *argv[])
{
  int t,n;
  pow2[0]=1ll;
  for(int i=1;i<=100010;i++)
    pow2[i]=(2*pow2[i-1])%mod;
  cin>>t;
  while(t--)
  {ll fans=0;
    cin>>n;
    for (int i = 0; i <= n; ++i)
    {
      scanf("%d",arr+i);
      if(i)
      {
        myans[i] = (arr[i]*pow2[n+1-i])%mod;
      }
    }
    for(int i=n-1;i>0;i--)
    {
      myans[i]=(myans[i]+myans[i+1])%mod;
    }
    for(int i=0;i<n;i++)
    {
      fans = (fans+((arr[i]*myans[i+1])%mod*pow2[max(0,i-1)])%mod)%mod;
    }
    cout<<fans<<"\n";
  }
  return 0;
}
