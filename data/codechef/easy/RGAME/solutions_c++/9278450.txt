#include<bits/stdc++.h>

using namespace std;
#define MOD 1000000007
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,a,an=0,sum=0,mu,i,j;
		scanf("%lld",&n);
		n++;
		if(n>=1)
		{
			scanf("%lld",&a);
			an=(a)%MOD;
			sum=(a*2)%MOD;
		}
		if(n>=2)
		{
			scanf("%lld",&a);
			an=(a*2*an)%MOD;
			sum=(sum%MOD+(a*2)%MOD)%MOD;
			
		}
		mu=4;
		if(n>=3)
		{
			for(i=2;i<n;i++)
			{
				scanf("%lld",&a);
				j=an;
				an=((j*2)%MOD+(sum*a)%MOD)%MOD;
				sum=(sum%MOD+(a*mu)%MOD)%MOD;
				mu=(mu*2)%MOD;
			}
		}
		printf("%lld\n",an);
		
	}
	return 0;
}
