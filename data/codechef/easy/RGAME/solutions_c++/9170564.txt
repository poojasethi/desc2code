#include<bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll n,p=1,ans=0;
        scanf("%lld",&n);
        ll sum[n+1],a[n+1],cs[n+1];
        for(int i=0;i<=n;i++)
            scanf("%lld",&a[i]);
        ans=(2*a[0]*a[1])%mod;
        ll pre=a[0];
        for(int i=2;i<=n;i++)
        {
            pre+=(p*a[i-1])%mod;
            pre=pre%mod;
            ans+=(a[i]*pre)%mod;
            ans=ans%mod;
            p=(p*2)%mod;
            ans=(ans*2)%mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
