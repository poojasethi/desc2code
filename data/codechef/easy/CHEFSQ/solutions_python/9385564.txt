from sys import stdin
sequence = favorite = []
number_testes = input()
for k in range(number_testes):
	nr_sequence = input()
	sequence = map(int,stdin.readline().split())
	nr_favorite = input()
	favorite = map(int,stdin.readline().split())
	x = i = 0
	while i < nr_sequence and x < nr_favorite :
		if sequence[i] == favorite[x] :
			x += 1
		i += 1
	if x == nr_favorite:
		print ("Yes")
	else:
		print ("No")
