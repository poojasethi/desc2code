def sums_in_triangle():
	# SUMTRIAN, calculate largest sum along a path from top to bottom.
	# Move only down or down and right.
	num_cases = int(raw_input())	
	for case in range(num_cases):		
		num_lines = int(raw_input())
		array = [[0 for j in range(num_lines)] for i in range(num_lines)]		
		for i in range(num_lines):
			for j, v in enumerate(raw_input().split()):
				array[i][j] = int(v)
		for i in range(1, len(array)):
			for j in range(len(array[i])):				
				array[i][j] = array[i][j] + max(array[i-1][j], array[i-1][j-1])
		print max(array[-1])
					

sums_in_triangle()
