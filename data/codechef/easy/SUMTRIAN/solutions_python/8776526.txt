def sum_triangle():
	n = int(raw_input())
	triangle =[]
	for i in range(n):
		new = map(int, raw_input().split())
		triangle.append(new)
	for i in range(n-2, -1, -1):
		for j in range(len(triangle[i])):
			triangle[i][j] += max(triangle[i+1][j], triangle[i+1][j+1])
		triangle.remove(triangle[i+1])
	return triangle[0][0]

test = int(raw_input())
for i in range(test):
	print sum_triangle()
