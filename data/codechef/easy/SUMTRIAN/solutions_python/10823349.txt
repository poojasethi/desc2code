def solve(i, j):
  if i >= n:
    return 0
  if '{},{}'.format(i, j) in cache:
    return cache['{},{}'.format(i, j)]
  t1 = solve(i + 1, j)
  t2 = solve(i + 1, j + 1)
  t = max(t1, t2) + tri[i][j]
  cache['{},{}'.format(i, j)] = t
  return t

for t in xrange(int(raw_input())):
  n = int(raw_input())
  tri = []
  cache = {}
  for row in range(n):
    tri.append(map(int, raw_input().split()))
  print solve(0, 0)
