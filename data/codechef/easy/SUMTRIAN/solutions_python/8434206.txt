for _ in xrange(int(raw_input())):
	n = int(raw_input())
	a = []
	for _ in xrange(n):
		a.append(map(int,raw_input().split()))
	for i in xrange(1,n):
		for j in xrange(len(a[i])):
			x = 0
			y = 0
			try:
				if i >= 1 and j >= 0:
					x = a[i-1][j]
			except IndexError:
				pass
			try:
				if i >= 1 and j >= 1:
					y = a[i-1][j-1]
			except IndexError:
				pass
			a[i][j] += max(x,y)
	print max(a[n-1])
