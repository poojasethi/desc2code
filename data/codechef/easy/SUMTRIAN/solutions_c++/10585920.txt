#include <iostream>
#include<stdio.h>

using namespace std;

int findMax(int arr[100][100], int n){
        int q = 0, temp = 0, maxSum;

        for(int i = 1; i<n; i++){
                q++;
                for(int j = 0; j<=q; j++){
                        temp = 0;
                        if(j-1 >= 0)
                                temp = arr[i-1][j-1] + arr[i][j];

                        if(arr[i-1][j] != -1 && arr[i][j]+arr[i-1][j] > temp)
                                temp = arr[i][j]+arr[i-1][j];

                        arr[i][j] = temp;
                }
        }

        maxSum = arr[n-1][0];
        for(int j = 1; j<n; j++){
                maxSum = max(maxSum, arr[n-1][j]);
        }
        return maxSum ;
}

int main()
{
        int t, n, q, sum = 0;
        int arr[100][100];

        scanf("%d", &t);
        for(int i = 0; i<t; i++){
                scanf("%d", &n);

                //initializing array with -1
                for(int j = 0; j<n; j++){
                        for(int k=0; k<n; k++){
                                arr[j][k] = -1;
                        }
                }
                q = 0;
                for(int j = 0; j<n; j++){
                        q++;
                        for(int k=0; k<q; k++){
                                scanf("%d", &arr[j][k]);
                        }
                }

                sum = findMax(arr, n);
                printf("%d\n", sum);
        }
    return 0;
}
