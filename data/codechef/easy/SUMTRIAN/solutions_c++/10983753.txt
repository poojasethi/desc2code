#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int A[101][101];
        for(int i=1; i<=n; i++)
        {
            for(int j=1;j<=i;j++)
                cin>>A[i][j];
        }
        int B[101][101];
        for(int i=n;i>=1;i--)
        {
            for(int j=1;j<=i;j++)
            {
                if(i==n)
                B[i][j]=A[i][j];
                else
                B[i][j]=A[i][j]+max(B[i+1][j],B[i+1][j+1]);
            }
        }
        cout<<B[1][1]<<endl;
    }
    return 0;
}
