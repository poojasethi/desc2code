#include<bits/stdc++.h>
using namespace std;
int max(int a,int b)
{
	int m,i;
	m = a;
	if(b > a)
	{
		m = b;
	}
	return m;
}
int main()
{
	int t,i,j,n,z;
	scanf("%d",&t);
	for(z=0;z<t;z++)
	{
		scanf("%d",&n);
		int a[100][100];
		for(i=0;i<n;i++)
		{
			for(j=0;j<=i;j++)
			{
				scanf("%d",&a[i][j]);
			}
		}
		for(i=n-1;i>=0;i--)
		{
			for(j=0;j<=i;j++)
			{
				int m = a[i][j];
				if(a[i][j+1] > a[i][j])
				{
					m = a[i][j+1];
				}
				a[i-1][j] = a[i-1][j] + m;
			}
		}
		printf("%d\n",a[0][0]);
	}
	return 0;
}
