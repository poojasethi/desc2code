#include <cstdio>
#include <cstdlib>
 
using namespace std;
 
inline int inp()
{
	int n=0;
	char p=getchar_unlocked();
 
	while((p<'0'||p>'9')&&p!=EOF)
		p=getchar_unlocked();
 
	while(p>='0'&&p<='9') {
		n = (n<< 3) + (n<< 1) + (p - '0');
		p=getchar_unlocked();
	}
 
	return n;
}
 
int main() {
 
	int t, r=0;
	int a[100][100];
	
	scanf("%d", &t);
 
	if( !(t <= 1000) ) {
		exit(EXIT_FAILURE);
	}
 
	while( t-- > 0 ) {
 
		//input maximum
		scanf("%d", &r);
 
		//input triangle
		for (int i=0; i < r; ++i) {
			for (int j = 0; j < i+1; ++j) {
				a[i][j] = inp();
			}
		}
 
		//calculate sumtrian
		int i,j;
		for(i=r-1;i>=0;i--) {
			for(j=0;j<i;j++) {
				if(a[i][j]<a[i][j+1])
					a[i-1][j] += a[i][j+1];
				else
					a[i-1][j] += a[i][j];
			}
		}
		
		printf("%d\n",a[0][0]);
 
	}
 
	return 0;
}
