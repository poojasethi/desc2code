#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
#define mp make_pair
#define FILE_IO freopen("input.in","r",stdin);freopen("output.txt","w",stdout);
#define FAST_IO ios_base::sync_with_stdio(false);cin.tie(NULL);cout.tie(NULL);
ll mat[1000][1000],dp[1000][1000];
int main()
{
    ll t,n,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=i;j++)
            {
                cin>>mat[i][j];
            }
        }
        for(i=n;i>=1;i--)
        {
            dp[n][i]=mat[n][i];
        }
        for(i=n-1;i>=1;i--)
        {
            for(j=1;j<=i;j++)
            {
                dp[i][j]=mat[i][j]+max(dp[i+1][j],dp[i+1][j+1]);
            }
        }
        cout<<dp[1][1]<<"\n";
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=i;j++)
            {
                dp[i][j]=0;
            }
        }
    }
    return 0;
}
