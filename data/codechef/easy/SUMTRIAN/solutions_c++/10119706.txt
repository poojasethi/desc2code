#include <iostream>
#include <cstdio>
using namespace std;

int t,n,a;
int dp[101][101];

int main() {
	// your code goes here
	
	scanf("%d",&t);
	
	while(t--){
		scanf("%d",&n);
		
		for(int i=1;i<=n;i++){
			for(int j=1;j<=i;j++){
				scanf("%d",&a);
				dp[i][j] = a;
				if(i>1){
					dp[i][j] += dp[i-1][j];
					if(j>1){
						int temp = dp[i-1][j-1]+a;
						dp[i][j] = temp>dp[i][j] ? temp: dp[i][j];
					}
				}
			}
		}
		
		int maxx = -1;
		
		for(int i=1;i<=n;i++){
			if(maxx<dp[n][i])
				maxx = dp[n][i];
		}
		
		printf("%d\n",maxx);
		
	}
	
	return 0;
}
