#include <bits/stdc++.h>
using namespace std;
const int N = 111;
int n, a[N][N], F[N][N] = {0};

int main() {
    int t; cin >> t;
    while (t--) {
        cin >> n;
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= i; j++) scanf("%d", &a[i][j]);
        F[1][1] = a[1][1];
        int mx = F[1][1];
        for (int i = 2; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                F[i][j] = max(F[i-1][j], F[i-1][j-1]) + a[i][j];
                mx = max(mx, F[i][j]);
            }
        }
        cout << mx << endl;
    }
}
