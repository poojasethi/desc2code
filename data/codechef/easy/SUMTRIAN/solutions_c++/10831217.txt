#include<iostream>
#include<cstring>
using namespace std;

int cell[100][100];
int N;
int dp[100][100];

int func(int i, int j)
{
	if(i>N)
		return 0;
	
	if(dp[i][j]>0)
		return dp[i][j];

	int t1 = func(i+1, j);
	int t2 = func(i+1,j+1);
	int ans = std::max(t1, t2) + cell[i][j];
	dp[i][j] = ans;

	return ans;

}


int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		cin >> N;
		memset(dp, 0, sizeof(dp));
		memset(cell,0,sizeof(cell));
		for(int i=0;i<N;i++)
		{
			for(int j=0;j<=i;j++)
			{
				cin >> cell[i][j];
			
			}
		}
		cout << func(0,0) << endl;
	
	}

}
