#include<iostream>
using namespace std;
//int a[10010];
int b[110][110];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,max1=0,i=0,j;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            for(j=0;j<=i;j++)
            {
                int x;
                cin>>x;
                if(i==0&&j==0)
                {
                    b[i][j]=x;
                    max1=b[i][j];
                }
                else if(j==0)
                {
                    b[i][j]=b[i-1][j]+x;
                }
                else if(i==j)
                {
                    b[i][j]=b[i-1][j-1]+x;
                }
                else
                {
                    b[i][j]=max(b[i-1][j-1],b[i-1][j])+x;
                }
            }
        }
        i=n-1;
        for(j=0;j<n;j++)
        {
            if(max1<b[i][j])
            {
                max1=b[i][j];
            }
        }
        cout<<max1<<"\n";
    }
}
