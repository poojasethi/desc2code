
#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<cmath>
#include<algorithm>
#include<cstdlib>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,n1=0;
        scanf("%d",&n);
        int a[100][100];
        while(n1<n)
        {
            for(int i=0;i<=n1;i++)
            {
                scanf("%d",&a[n1][i]);
            }
            n1++;
        }
        for(int i=n-2;i>=0;i--)
        {
            for(int j=0;j<=i;j++)
             a[i][j] += a[i+1][j] > a[i+1][j+1] ? a[i+1][j] : a[i+1][j+1];        }
        printf("%d\n",a[0][0]);
    }
    return 0;
}
