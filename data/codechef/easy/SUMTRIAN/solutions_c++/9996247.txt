#include<cstdio>
using namespace std;
#define max(a,b) a>=b ? a : b ;
int main()
{
	int t,a,b,c;
	int A[100][100];
	scanf("%d",&t);
	while(t--)
	{
        scanf("%d",&a);
        for(b=0;b<a;b++)
        {
        	for(c=0;c<=(b);c++)
        	{
        		scanf("%d",&A[b][c]);
        	}
        }
        for(b=(a-2);b>=0;b--)
        {
        	for(c=0;c<=b;c++)
            {
            	A[b][c]+=max(A[b+1][c],A[b+1][c+1]);
            }
        }
        printf("%d\n",A[0][0]);
	}
	return 0;
}
