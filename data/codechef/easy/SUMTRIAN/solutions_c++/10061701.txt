#include <iostream>
using namespace std;

const int maxn = 110;
int ar[maxn][maxn];

int main() {
	int t; cin >> t;
	while (t--) {
	    int n; cin >> n;
        for (int i = 1; i <= n; ++i)
            for (int j = 1; j <= i; ++j)
                cin >> ar[i][j];
        for (int i = n - 1; i >= 1; --i) {
            for (int j = 1; j <= i; ++j) {
                ar[i][j] += max(ar[i + 1][j], ar[i + 1][j + 1]);
            }
        }
        cout << ar[1][1] << endl;
	}
	return 0;
}
