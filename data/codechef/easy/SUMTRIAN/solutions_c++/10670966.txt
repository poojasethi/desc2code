#include <iostream>
using namespace std;

int main(int argc, const char * argv[]) {
    int s,n,a[100][100],i,j;
    cin>>n;
    while(n!=0)
    {
    cin>>s;
    for( i=0;i<s;i++)
    {
        for( j=0;j<=i;j++)
            cin>>a[i][j];
        
    }
    for(i=s-2;i>=0;i--)
    {
        for(j=0;j<=i;j++)
             a[i][j]+= a[i+1][j]>a[i+1][j+1]?a[i+1][j]:a[i+1][j+1];
    }
   
        cout<<a[0][0]<<"\n";
        n--;
    }
        return 0;
}
