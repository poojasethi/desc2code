#include<bits/stdc++.h>
using namespace std;
int main()
{
int i,j,maxx,t,n;
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
int array[n+1][n+1];
memset(array,0,sizeof array);
for(i=1;i<=n;++i)
    for(j=1;j<=i;++j)
        scanf("%d",&array[i][j]);
maxx=array[1][1];
for(i=2;i<=n;++i)
    for(j=1;j<=i;++j)
    {
        array[i][j]=max(array[i][j]+array[i-1][j],array[i][j]+array[i-1][j-1]);
        maxx=max(array[i][j],maxx);
    }

    printf("%d\n",maxx);
}
return 0;
}
