#include<bits/stdc++.h>
using namespace std;

int A[100][100],t,n;
int B[100][100];

int solve(int a,int b)
{
    int t1,t2;
    if(a>=n)
        return 0;
    if(B[a+1][b]==-1)
        t1=solve(a+1,b);
    else
        t1=B[a+1][b];

    if(B[a+1][b+1]==-1)
        t2=solve(a+1,b+1);
    else
        t2=B[a+1][b+1];

    B[a][b]=max(t1,t2)+A[a][b];
    return B[a][b];

}

int main()
{
    int i,j,k;
    cin>>t;

    for(i=0;i<t;++i)
    {
        for(j=0;j<100;++j)
            for(k=0;k<=j;++k)
            B[j][k]=-1;
        cin>>n;
        //input
        for(j=0;j<n;++j)
            for(k=0;k<=j;++k)
                scanf("%d",&A[j][k]);

        //process
        cout<<solve(0,0)<<"\n";
    }
    return 0;
}
