#include<iostream>

#define MAX(x,y) ((x) > (y) ? (x) : (y))

int main(){
  int n ,t;
  std::cin >> t ;

  while(t--)
  {
    std::cin >> n ;
    	int  arr[n][n];
        for(int i=0; i<n;i++)
          for(int j=0; j < i + 1 ; j++)
            std::cin >> arr[i][j];

        for (int i= n-2; i >= 0 ; i--)
          for (int j = 0 ; j < i+1 ; j++)
            arr[i][j] += MAX(arr[i+1][j],arr[i+1][j+1]);

        std::cout << arr[0][0] << std::endl ;
   }

   return 0;
}
