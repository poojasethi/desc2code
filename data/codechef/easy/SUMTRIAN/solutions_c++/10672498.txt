#include<iostream>
using namespace std;
int main()
{
    int n,N,k,p,i=0,j=0;
    int A[100][100];
    cin>>n;
    for(p=0;p<n;p++)
    {
        cin>>N;
        for(k=0;k<N;k++)
        {

        for(j=0;j<=k;j++)
        {
            cin>>A[k][j];
        }
        }
        for(i=N-2;i>=0;i--)
        {
            for(j=0;j<=i+1;j++)
            {
                if(A[i+1][j]>=A[i+1][j+1])
                A[i][j]+=A[i+1][j];
                else
                    A[i][j]+=A[i+1][j+1];
            }
        }
        cout<<A[0][0]<<endl;
    }
}
