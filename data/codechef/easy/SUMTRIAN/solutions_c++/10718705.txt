#include<iostream>
#include<math.h>
using namespace std;

int main()
{
  int T , N;
  int a[100][100];
  int sum1= 0 , sum2 = 0;
  int max = -1;
  int k = 0;
  cin>>T;
  while(T--)
  {
    cin>>N;
    for(int i = 0 ; i<N ; i++)
    {
     for(int j = 0 ; j<=i ; j++)
     {
       cin>>a[i][j];
     }
    }
    for(int i = N-2 ; i>=0 ; i--)
    {
      for(int j = 0 ; j<=i ; j++)
      {
        sum1 = a[i][j]+a[i+1][j];
        sum2 = a[i][j]+a[i+1][j+1];
        if(sum1>sum2)
        {
            a[i][j] = sum1;
        }
        else
            a[i][j] = sum2;
      }
    }
    cout<<a[0][0]<<endl;
  }
}
