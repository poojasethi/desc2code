#include<bits/stdc++.h>
using namespace std;
#define f(i,x)for(int i=0;i<x;++i)
int t, n, a[101][101], dp[101][101];
int calc(int i, int j) {
	if (i == n || j > i)
		return 0;
	int &ret = dp[i][j];
	if (ret != -1)
		return ret;
	ret = a[i][j] + max(calc(i + 1, j), calc(i + 1, j + 1));
	return ret;
}
int main() {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif

	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		f(i,n)
			f(j,i+1)
				scanf("%d", &a[i][j]);
		memset(dp, -1, sizeof dp);
		printf("%d\n", calc(0, 0));
	}
	return 0;
}
