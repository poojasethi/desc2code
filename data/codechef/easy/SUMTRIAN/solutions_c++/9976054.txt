#include<iostream>

using namespace std;

int main()
{
	int n,t,i,j,r,c,max;
	
	cin>>n;
	
	while(n--)
   {

		cin>>t;
		
		int **arr = new int*[t];
		
		for(i=0;i<t;i++)
		arr[i] = new int[t];
		
		for(i=0;i<t;i++)
	     for(j=0;j<=i;j++)
	        cin>>arr[i][j];
	        
	      for(r=t-1;r>0;r--)
	       {
	       	 for(c=t-1;c>0;c--)
	       	  {
	       	  	 if(arr[r][c]>arr[r][c-1])
	       	  	   max = arr[r][c];
	       	  	   else
	       	  	   max = arr[r][c-1];
	       	  	   
	       	  	   arr[r-1][c-1] = arr[r-1][c-1] + max; 
	       	  }
	       }
	       
	       cout<<arr[0][0]<<endl;
	}
}
