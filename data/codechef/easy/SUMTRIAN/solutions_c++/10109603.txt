#include<stdio.h>
#include<iostream>
using namespace std;

int main(){
	
	int testc;
	cin>>testc;
	int l;
	int arr[100][100];
	int s[100][100];
	
	int path, sum=0;
	
	while(testc>0){
		testc--;
		sum=0;
		cin>>l;
		for(int i=0;i<l;i++){
			for (int j = 0; j <= i; j++){
				cin>>arr[i][j];
			}
		}
		
		
	for(int i=0;i<l;i++){
		for(int j=0;j<=i;j++){
			if(i==0 && j==0){
				s[i][j]=arr[i][j];
			}
			else if(j==0){
				s[i][j]=s[i-1][j] + arr[i][j];
			}
			else{
				s[i][j]=arr[i][j] + max(s[i-1][j],s[i-1][j-1]);
			}
		}
	}
	for(int j=0;j<l;j++){
		if(s[l-1][j]>sum){
			sum=s[l-1][j];
		}
	}
	
	cout<<sum<<endl;
	
		
	}
	
	return 0;
	
	
	
}
