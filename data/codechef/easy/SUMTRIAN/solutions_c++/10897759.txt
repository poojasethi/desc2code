#include<iostream>
using namespace std;
int main()
{
	int m[100][100];
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<=i;j++)
			cin>>m[i][j];
		}
		for(int i=n-1;i>0;i--)
		{
			for(int j=i;j>0;j--)
			{
				m[i-1][j-1]=m[i-1][j-1]+max(m[i][j-1],m[i][j]);
			}
		}
		cout<<m[0][0]<<endl;
	}
}
