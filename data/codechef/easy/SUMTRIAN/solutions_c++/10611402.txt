#include <iostream>
#include <cstdio>

using namespace std;

int recur(int row, int col, int a[][101], int n, int sum[][101])
{
        if(row == n-1){
                return a[row][col];
        }
	if(sum[row][col] != 0){
		return sum[row][col];
	}
      	sum[row][col] = max(a[row][col] + recur(row+1,col, a, n, sum),a[row][col]+recur(row+1,col+1, a, n, sum));
	return sum[row][col];
}


int main()
{       
        int t;
        scanf("%d",&t);
        while(t--){
                int n;
                int arr[101][101];
		int sum[101][101];
                scanf("%d",&n);
                for(int i=0; i<n; i++){
                        for(int j=0; j<=i; j++){
                                scanf("%d",&arr[i][j]);
				sum[i][j] = 0;
                        }
                }
                int max = recur(0, 0, arr, n, sum);
                printf("%d\n",max);
        }       
        return 0;
}
