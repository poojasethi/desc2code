#include<stdio.h>
#include<stdlib.h>

int main()
{
    int t,i,j,k,**a,n;
    scanf("%d",&t);
    while(t-->0)
    {
        scanf("%d",&n);
        a=(int **)malloc(n*sizeof(int *));
        for(i=0;i<n;i++)
        {
            a[i]=(int *)malloc((i+1)*sizeof(int));
            for(j=0;j<=i;j++)
                scanf("%d",&a[i][j]);

        }

        for(i=n-2,k=n-1;i>=0;i--,k--)
        {
            for(j=0;j<=i;j++)
            {
                if(a[k][j]>a[k][j+1])
                    a[i][j]+=a[k][j];
                else
                   a[i][j]+=a[k][j+1];
            }
        }
        printf("%d\n",a[0][0]);
    }
    return 0;
}
