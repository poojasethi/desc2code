#include<iostream>
#include<bits/stdc++.h>

using namespace std;


int main(){

int n;
scanf("%d",&n);

while(n--){

int ans=0;
int row=0;

scanf("%d",&row);
int a[100][100];

for(int i=0;i<row;i++){
	
	for(int j=0;j<=i;j++){
			
			scanf("%d",&a[i][j]);
	}
	
}

for(int i=row-2;i>=0;i--){
	
	for(int j=0;j<=i;j++){
		
		int sum1=0,sum2=0;
		
		sum1=a[i][j]+a[i+1][j];
		sum2=a[i][j]+a[i+1][j+1];
		
		if(sum1 > sum2)
			a[i][j]=sum1;
		else 
			a[i][j]=sum2;
	}
}

printf("%d\n",a[0][0]);


}


return 0;
}
