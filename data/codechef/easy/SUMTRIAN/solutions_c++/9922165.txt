#include<bits/stdc++.h>
using namespace std;int a[100][100];
int main(void)
{int t,n,i,j,k,sum;cin>>t;while(t--){cin>>n;sum=-1;for(i=0;i<n;i++)for(j=0;j<i+1;j++)cin>>a[i][j];
                 for(i=1;i<n;i++)a[i][0]+=a[i-1][0];for(i=1;i<n;i++){for(j=1;j<i+1;j++){if(i==j) a[i][j]+=a[i-1][j-1];
                   else a[i][j]+=max(a[i-1][j],a[i-1][j-1]);}}
                   for(i=0;i<n;i++)if(a[n-1][i]>sum)sum=a[n-1][i];cout<<sum<<"\n";}return 0;}
