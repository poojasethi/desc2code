import sys

data = sys.stdin.readlines()
t = int(data.pop(0))
output = []

for i in range(t):
    num = map(int, data.pop(0).split(' '))
    b = num[0]
    ls = num[1]
    
    rs_min = (ls**2 - b**2)**(0.5)
    rs_max = (ls**2 + b**2)**(0.5)

    output.append(' '.join(map(str,[rs_min, rs_max])))

for k in output:
    print k
