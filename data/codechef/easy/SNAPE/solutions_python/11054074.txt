testcases = raw_input()
 
def rightside(base, leftside):
	max_length = (base**2 + leftside**2)**0.5
	min_legth = (leftside**2 - base**2)**0.5
	return str(min_legth) + " " + str(max_length)
 
for i in range(int(testcases)):
	rawnums = raw_input()
	nums = rawnums.split()
	base = int(nums[0])
	leftside = int(nums[1])
	print rightside(base, leftside)
