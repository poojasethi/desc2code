import math
def findRS(B, LS):
    maxRS = math.sqrt(LS*LS + B*B)
    minRS = math.sqrt(LS*LS - B*B)
    print "%.4f %.4f" % (minRS,maxRS)

trials = int(raw_input())
sides = []

for counter in range(trials):
    inputs = map(int, raw_input().split())
    sides.append(inputs[0])
    sides.append(inputs[1])

for counter in range(0,trials*2,2):
    ground = sides[counter]
    leftSide = sides[counter+1]
    findRS(ground, leftSide)
