#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int b,ls;
        cin>>b>>ls;
        double maxm,minm;
        minm=sqrt(ls*ls-(b*b));
        maxm=sqrt(ls*ls + (b*b));
        cout<<minm<<" "<<maxm<<endl;
    }
    return 0;
}
