#include<iostream>
#include<math.h>

using namespace std;

int main ()
{
        int T,i;

        //cout << "write the total no. of test cases\n";
        cin >> T;

        for (i=0;i<T;i++)
        {
                //cout << "write the base and second side\n";
                int B,S;
                cin >> B >> S;

                cout << pow(S*S-B*B,.5) << " " << pow(S*S+B*B,.5) << endl;
        }

        return 0;
}
