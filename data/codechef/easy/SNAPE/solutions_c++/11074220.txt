#include<bits/stdc++.h>
using namespace std;
  int main(void)
   {
  int t;
  cin>>t;
  while(t--)
    {
        float b,ls,q,r;
        cin>>b>>ls;
        q=sqrt((b*b)+(ls*ls));
        r=sqrt((ls*ls)-(b*b));
       // cout<<r<<q<<endl;
       printf("%f\t%f",r,q);
       printf("\n");
    }
    return 0;
   }
