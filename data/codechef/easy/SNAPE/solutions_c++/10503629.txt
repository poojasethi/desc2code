#include<bits/stdc++.h>
#include<math.h>
using namespace std;
int main()
{
    long long int t,i,j;
    float rs;
    scanf("%lld",&t);
    while(t>0)
    {
        float ls,b;
        scanf("%f %f",&b,&ls);
        float max , min , m ,n;
        min = (ls*ls) - (b*b);
        m = sqrt(min);
        max = (ls*ls) + (b*b);
        n = sqrt(max);
        printf("%.6f %.6f",m,n);
        printf("\n");
        t--;
    }
    return 0;
}
