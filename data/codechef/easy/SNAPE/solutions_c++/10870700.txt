#include<iostream>
#include<math.h>
double minLength(int,int);
double maxLength(int,int);
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int b,ls;
        cin>>b>>ls;
        cout<<minLength(ls,b)<<" "<<maxLength(ls,b)<<endl;
    }
return 0;
}

double minLength(int a,int b){
    return sqrt(fabs(a*a-b*b));
}
double maxLength(int a,int b){
    return sqrt(a*a+b*b);
}
