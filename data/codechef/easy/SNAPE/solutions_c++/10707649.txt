
#include <iostream>
#include<math.h>

using namespace std;
int main(int argc, const char * argv[]) {
    int B,LS,T;
    cin>>T;
    while(T!=0)
    {
    float y,l,x,z;
    cin>>B>>LS;
    x=(LS*LS-B*B);
    y=sqrt(x);
    z=LS*LS+B*B;
    l=sqrt(z);
    cout<<y<<" "<<l<<"\n";
        T--;
    }
    return 0;
}
