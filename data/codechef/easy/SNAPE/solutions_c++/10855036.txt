#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    int t;
    int *b,*ls;
    float rs;
    cin>>t;
    b=new int[t];
    ls=new int[t];
    for(int i=0;i<t;i++)
    {
        cin>>b[i]>>ls[i];
    }
    for(int i=0;i<t;i++)
    {
        rs=sqrt(ls[i]*ls[i]-b[i]*b[i]);
        cout<<(float)rs<<" ";
        rs=sqrt(ls[i]*ls[i]+b[i]*b[i]);
        cout<<(float)rs<<endl;
    }
    delete b;
    delete ls;
    return 0;
}
