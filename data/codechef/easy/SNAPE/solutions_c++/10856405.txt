#include<iostream>
#include<cmath>
//#include<conio.h>
using namespace std;
int main()
{
	int t;
	double max_rs,min_rs,b ,ls;
	cin>>t;
	while(t--){
		cin>>b;
		cin>>ls;
		max_rs=pow((pow(b,2)+pow(ls,2)),0.5);
		min_rs=pow((pow(ls,2)-pow(b,2)),0.5);
		cout<<min_rs<<" "<<max_rs<<endl;
	}
//getch();
return 0;
}
