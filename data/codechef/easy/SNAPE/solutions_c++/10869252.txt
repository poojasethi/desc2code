#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    float b,ls,min,max;
    int t;
    cin>>t;
    while(t--)
    {
        cin>>b>>ls;
        max=sqrt((b*b)+(ls*ls));
        min=sqrt((ls*ls)-(b*b));
        cout.setf(ios::fixed);
        cout.precision(8);
        cout<<min<<" "<<max<<"\n";

    }
    return 0;
}
