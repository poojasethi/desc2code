#include<stdio.h>
#include<math.h>
#define max(a, b) (((a) >= (b)) ? (a) : (b))
#define min(a, b) (((a) <= (b)) ? (a) : (b))
#define s sqrt
int main(){
	int t; scanf("%d", &t); double b, l, mi, ma;
	while(t-- && scanf("%lf%lf", &b, &l)){
		mi = min(b,l), ma = max(b, l);
		printf("%lf %lf\n", s((ma * ma) - (mi * mi)), s((ma * ma) + (mi * mi)));
	}
	return 0;
}
