#include <iostream>
#include<math.h>

using namespace std;



int main(void){
    int arr[1000][2],i,T,LS[1000],B[1000];
    cin>>T;
    for(i=0;i<T;i++){
        cin>>B[i]>>LS[i];
    }
    for(i=0;i<T;i++){
        cout<<sqrt((LS[i]*LS[i]-B[i]*B[i]))<<" "<<sqrt((LS[i]*LS[i]+B[i]*B[i]))<<"\n";
    }
    return 0;
}
