#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,cnt=0;
    vector<int>v;
    while(cin>>n)
    {

           v.push_back(n);
            cnt++;

    }
    for(int i=0;i<cnt;i++)
    {

         if(v[i]!=42)
        {
            cout<<v[i]<<endl;

        }
        else
        {
            break;
        }

    }
}
