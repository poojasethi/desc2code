#include<iostream>
using namespace std;
class Test
{
    int a[100];
    int i,j,n;
public:
        void input()
        {
            j=0;
            n=100;
            for (i=0;i<n;i++)
            {
                cin>>a[i];
                j++;
                if (a[i]==42)
                    break;

            }
        }
        void print()
        {

            for (i=0;i<j-1;i++)
                    cout<<a[i]<<endl;
        }
};

int main()
{
    Test t;
    t.input();
    t.print();

}
