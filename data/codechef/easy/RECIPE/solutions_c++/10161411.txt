#include<iostream>
using namespace std;
int hcf(int a,int b)
{
	int t;
	while(a)
	{
		t=a;
		a=b%a;
		b=t;
		}
		return b;
	}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];	
		}
		int ans;
		if(n==1)
		ans=a[0];
		if(n>=2)
		ans=hcf(a[0],a[1]);
		for(int i=2;i<n;i++)
		{
			ans=hcf(ans,a[i]);
		}
		for(int i=0;i<n;i++)
		cout<<(a[i]/ans)<<" ";
		cout<<"\n";
		
	}
}
