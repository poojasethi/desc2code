#include <iostream>

using namespace std;

int main()
{
    int t,i,n,ingredients[50],smallest=1001,count,div;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>ingredients[i];
            if(smallest>ingredients[i])
            {
                smallest=ingredients[i];
            }
        }
        for(int i=1;i<=smallest;i++)
        {
            count=0;
            for(int j=0;j<n;j++)
            {
                if((ingredients[j]%i)==0)
                {
                    count++;
                }
            }
            if(count==n)
            {
                div=i;
            }
        }
        smallest=1001;
        for(i=0;i<n;i++)
        {
            cout<<ingredients[i]/div<<" ";
        }
        cout<<endl;

    }
    return 0;
}
