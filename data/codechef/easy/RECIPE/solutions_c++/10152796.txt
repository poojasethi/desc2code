#include <iostream>
using namespace std;
int findgcd(int n1,int n2 )
{
	int i,gcd;
	for(i=1;i<=n1 && i<=n2;i++)
	{
		if(n1%i==0 && n2%i==0)
		{
			gcd=i;
		}
	}
	return gcd;
}
int main ()
{
	int t;
	cin>>t;
	int a[50];
	while(t--)
	{
		int N,gcd;
		cin>>N;
		cin>>a[0];
		gcd=a[0];
		for(int i=1;i<N;i++)
		{
			cin>>a[i];
			gcd = findgcd(a[i],gcd);
		}
		for(int i=0;i<N;i++)
		{
			cout<<a[i]/gcd<<" ";
		}
		cout<<endl;

	}
return 0;
}
