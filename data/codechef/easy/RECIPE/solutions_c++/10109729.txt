#include <iostream>
using namespace std;
int findgcd(int a,int b)
{
    if(a%b==0)
    {
        return b;
    }
    else
    {
        return findgcd(b,a%b);
    }
}
int main() {
	// your code goes here
	int t,n;
	cin>>t;
	while(t--)
	{
	    cin>>n;
	    int a[n],i;
	    for(i=0;i<n;i++)
	    {
	        cin>>a[i];
	    }
	    int gcd=a[0];
	    for(i=1;i<n;i++)
	    {
	        if(gcd<a[i])
	        {
	            gcd=findgcd(a[i],gcd);
	        }
	        else
	        {
	            gcd=findgcd(gcd,a[i]);
	        }
	        if(gcd==1)
	        {
	            break;
	        }
	    }
	   // cout<<gcd;
	    for(i=0;i<n;i++)
	    {
	       // cout<<a[i]<<" ";
	        cout<<a[i]/gcd<<" ";
	    }
	    cout<<endl;
	}
	return 0;
}
