#include <bits/stdc++.h>

long long int t, n , a[10000], i, j, flag, b[10000], k, mult;

using namespace std;

int main()
{
	scanf("%lld",&t);
	while (t--) {
		flag = 1;
		scanf("%lld",&n);
		for (i = 0; i < n; i++) {
			scanf("%lld",&a[i]);
			b[i] = a[i];
		}
		sort(b, b+n);
		for (i = 0; i < n - 1; i++) {
			if (a[i] != a[i+1]) {
				flag = 0;
				break;
			}
		}
		if (flag == 1) {
			for (i = 0; i < n; i++) {
				printf("1 ");
			}
			printf("\n");
		}else {
			j = b[0];
			i = 2;
			
			while (j > 0 && i <= j) {
//				printf("i = %lld\n",i); 
				if (j % i == 0) {
					j = j / i;
					for (k = 0; k < n; k++) {
						if (a[k] % i != 0 ) {
							break;
						}
//						printf("a[k] = %lld, i = %lld \n",a[k],i);
					}
//					printf("k = %lld\n", k);
					if (k == n) {
						for (k = 0; k < n; k++) {
							a[k] = a[k] / i ;
//							printf("a[k] = %lld \n",a[k]);
						}
					}
					i = 2;
				}
				else {
					i++;
				}
			}
			for (i = 0; i < n; i++) {
				printf("%lld ",a[i]);
			}
			printf("\n");
		}
	}
		return 0;
}
