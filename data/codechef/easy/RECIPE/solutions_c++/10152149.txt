#include<bits/stdc++.h>
using namespace std;
 
long int gcd(long int a,long int b){
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
 
int main(){
    long int t,flag=0;
    cin>>t;
    while(t--){
        long int n;
        cin>>n;
        int arr[n],gcdresult;
        for(long int i=0;i<n;i++)
            cin>>arr[i];
            gcdresult=arr[0];
            for(long int i=1;i<n;i++)
                gcdresult = gcd(gcdresult,arr[i]);
            for(long int i=0;i<n;i++)
                cout<<arr[i]/gcdresult<<" ";
 
    }
    return 0;
}
