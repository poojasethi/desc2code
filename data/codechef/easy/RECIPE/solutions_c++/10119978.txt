//running GCD
#include<bits/stdc++.h>
using namespace std;
long gcd(long a, long b){
if(b == 0)
return a;
else return gcd(b,a%b);
}

int main()
{
    int t;
    scanf("%d",&t);
    int a[51]={0};
    while(t--)
    {
        int n,i;
        int gc=1;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        gc=gcd(a[0],a[1]);
        for(i=2;i<n;i++)
        {
            gc=gcd(gc,a[i]);
        }
        for(i=0;i<n;i++)
        {
            printf("%d ",a[i]/gc);
        }
        printf("\n");

    }
    return 0;
}
