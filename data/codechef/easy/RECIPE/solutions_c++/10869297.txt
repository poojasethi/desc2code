#include<iostream>
using namespace std;
int main()
{
    int t,n,m,i,j,flag,min;
    cin>>t;
    while(t--)
    {
    cin>>n;
    int a[n];
    m=n;
    for(i=0;i<n;i++)
        cin>>a[i];
    min=a[0];
    for(i=1;i<n;i++)
    {
        if(a[i]<min)
            min=a[i];
    }
    for(i=min;i>0;i--)
    { flag=0;
        for(j=0;j<n;j++)
        {
            if(a[j]%i!=0)
                    flag=1;
        }
    if(flag!=1)
   {
       for(j=0;j<m;j++)
       cout<<a[j]/i<<" ";
    break;
   }
    }
 cout<<"\n";
    }
    return 0;
}
