#include <iostream>

using namespace std;

int gcd(int a, int b)
{
	int gcd;
	if (a > b)
	{
		for (int i = 1; i <= b; i++)
		{
			if ((a%i == 0) && (b%i == 0))
				gcd = i;
		}
	}
	else
	{
		for (int i = 1; i <= a; i++)
		{
			if ((a%i == 0) && (b%i == 0))
				gcd = i;
		}
	}
	return gcd;
}

int main()
{
	int n, m, GCD;
	int A[50];
	cin >> n;
	while (n--)
	{
		cin >> m;
		cin >> GCD;
		A[0] = GCD;
		for (int i = 1; i < m; i++)
		{
			cin >> A[i];
			GCD = gcd(GCD, A[i]);
		}
		for (int i = 0; i < m; i++)
			A[i] /= GCD;
		cout << A[0];
		for (int i = 1; i < m; i++)
			cout << " " << A[i];
		cout << endl;
	}
	return 0;
}
