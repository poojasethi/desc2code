#include <iostream>
#include <stdio.h>

using namespace std;

int min(int a[], int n) {
	int i,min;
	min = a[0];
	for (i = 0; i < n; i++) {
		if(a[i]<min)
			min = a[i];
	}
	return min;
}

bool is_divide(int a[], int n, int m) {
	int i,j,k;
	bool status = true;
	for (i = 0; i < n; i++) {
		if(a[i] % m !=0)
			status = false;
	}
	return status;
}


int gcd(int a[], int n) {
	int i,j,k,max = 1;
	int min_element = min(a,n);
	for (i = 1; i <= min_element; i++) {
		if(is_divide(a,n,i)==true)
			max = i;
	}
	return max;
}

void solve(int a[], int n) {
	int i,j,k;
	int max_divisor = gcd(a,n);
	for (i = 0; i < n; i++) {
		cout << (a[i]/max_divisor)<< " ";
	}
	cout << endl;
}

int main() {
	int t,i,j,k,n,a[50];
	cin>>t;
	while (t--) {
		cin >> n;
		for(i = 0 ; i < n ; i++)
			cin >> a[i] ;
		solve(a,n);
	}
}
