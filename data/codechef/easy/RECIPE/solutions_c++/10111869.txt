#include<iostream>
#include<algorithm>
using namespace std;
int gcd(int x,int y){
    int x1,y1;
    if(x>y)
        x1=x;
    else
        x1=y;
    if(x<y)
        y1=x;
    else
        y1=y;
    if(y1==0)
        return x1;
    else
        return gcd(y1,x1%y1);
}
int main(){
    int t,p;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        int g= a[0];
        for(int i=1;i<n;i++){
                g=gcd(g,a[i]);
        }
        for(int i=0;i<n;i++){
            p=a[i]/g;
            cout<<p<<" ";
        }
}
}
