#include<bits/stdc++.h>
using namespace std;
typedef vector<int> vi;
#define loop(i,t) for(i=0;i<t;++i)
#define pb(temp) push_back(temp)
int hcf(int a,int b);

int main()
{
    vi A(100);
    int t,i,n,j,h,temp;
    cin>>t;
    loop(i,t)
    {
        cin>>n;
        cin>>h;
        A[0]=h;

        loop(j,n-1)
        {
            cin>>temp;
            A[j+1]=temp;
            h=hcf(temp,h);
        }

        loop(j,n)
        {
            cout<<(A[j]/h)<<" ";
        }
        cout<<"\n";
    }

	return 0;
}
int hcf(int a,int b)    //euclid
{
    int dividend=a;
    int divisor=b;
    int quotient;
    int rem;
    if(dividend==1||divisor==1)
        return 1;
    else if(dividend<divisor)
        return hcf(divisor,dividend);
    else
    {
        quotient=dividend/divisor;
        rem=dividend%divisor;
        if(rem==0)
        return divisor;
        dividend=divisor;
        divisor=rem;
        return hcf(dividend,divisor);
    }
}
