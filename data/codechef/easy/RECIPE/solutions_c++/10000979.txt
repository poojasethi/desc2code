#include <bits/stdc++.h>
using namespace std;

int main() {
	int t,n,gcd;
	int A[100];
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>A[i];
			if(i==0)
				gcd=A[0];
			else
				gcd=__gcd(A[i],gcd);
		}
		for(int i=0;i<n;i++){
			cout<<A[i]/gcd<<" " ;
		}
		cout<<endl;
	}
	return 0;
}
