#include <iostream>
using namespace std;

const int maxn = 1e6;
int ar[maxn];

int gcd(int a, int b) {
    return a == 0 ? b : gcd(b % a, a);
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    int t; cin >> t;
    while (t--) {
        int n; cin >> n;
        int g = 0;
        for (int i = 0; i < n; ++i) {
            cin >> ar[i];
            g = gcd(g, ar[i]);
        }
        for (int i = 0; i < n; ++i)
            cout << ar[i] / g << " ";
        cout << endl;
    }
        
	return 0;
}
