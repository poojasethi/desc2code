#include <iostream>
using namespace std;
int gcd(int a,int b)
{
if(a>b)
	{
	if(a%b==0)
	return b;
	else{
	a%=b;
	gcd(a,b);
	}
	}
	else
	{
	if(b%a==0)
	return a;
	else
	{
	b%=a;
	gcd(a,b);
	}
	}
}	
int main() {
int t,n,g,i,a[50];
cin>>t;
while(t--)
{
cin>>n;
for(i=0;i<n;i++)
cin>>a[i];
g=a[0];
for(i=1;i<n;i++)
g=gcd(g,a[i]);
for(i=0;i<n;i++)
cout<<a[i]/g<<" ";
cout<<"\n";
}
	return 0;
}
