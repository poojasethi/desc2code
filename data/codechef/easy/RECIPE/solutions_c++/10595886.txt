#include <iostream>
#include<stdio.h>
using namespace std;

int arr[50];

int gcd(int a, int b){

        if(a<b) swap(a, b);

        if(b==0) return a;
        if(b==1) return 1;

        gcd(b, a%b);
}

int main()
{
        int t, n, temp;

        scanf("%d", &t);

        while(t--){
                scanf("%d", &n);

                for(int i = 0; i<n; i++){
                        scanf("%d", &arr[i]);
                }

                temp = arr[0];

                for(int i = 1; i<n; i++){
                        temp = gcd(temp, arr[i]);
                        if (temp == 1) break;
                }

                for(int i = 0; i<n; i++)
                        printf("%d ", arr[i]/temp);

                printf("\n");

        }
    return 0;
}
