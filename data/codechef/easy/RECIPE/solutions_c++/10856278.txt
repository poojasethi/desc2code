#include<iostream>
//#include<conio.h>
using namespace std;
int main()
{
	int t,i,j ,num,k,min,gcd;
	cin>>t;//cout<<endl;
	while(t--)
	{	
		cin>>num;
		int arr[num];
		for(i=0;i<num;i++)
		{
			cin>>arr[i];
		}
		cout<<endl;
		min=arr[0];
		for(i=1;i<num;i++)
		{  if(arr[i]<min)
			min=arr[i];
		}
		
		for(j=min;j>=1;j--)
		{	k=0;
			for(i=0;i<num;i++)
			{  
				if(arr[i]%j==0)
				k++;			
			}
			if(k==num)
				{gcd=j;break;}		
		}
		for(i=0;i<num;i++)
		{
			arr[i]=arr[i]/gcd;
			cout<<arr[i]<<" ";
		}
		cout<<endl;
	}
	//getch();
	return 0;
}
