#include<iostream>

using namespace std;

int main ()
{
        int T,i;

        //cout << "write total no. of test cases\n";
        cin >> T;

        for (i=0;i<T;i++)
        {
                int j,k,N,d,l,t=0,y;

                //cout << "\nprint the N\n";
                cin >> N;

                int a[N];

                //cout << "\nprint the aaray\n";
                cin >> a[0];
                d = a[0];
                t = 0;
                for (j=1;j<N;j++)
                {
                        cin >> a[j];
                        if (d>a[j])
                        {
                                t = j;
                        }

                }
                j = 2;
                while (j<=a[t])
                {
                        l = 0;
                        for (k=0;k<N;k++)
                        {
                                if (a[k]%j != 0)
                                {
                                        l++;
                                }
                        }
                        if (l==0)
                        {
                                for (k=0;k<N;k++)
                                {
                                        a[k] = a[k]/j;
                                        //cout << a[k] << "done " << j;
                                }
                                j = 2;
                        }
                        else
                        {
                                j++;
                        }
                }
                for (j=0;j<N;j++)
                {
                        cout << a[j] << "\t";
                }
        }
        cout << endl;

        return 0;
}
