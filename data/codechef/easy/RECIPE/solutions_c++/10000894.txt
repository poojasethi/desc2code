#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin>>t;
	int hcf,size;
	int arr[51],i = 0;
	while(t--)
	{
		cin>>size;
		cin>>arr[0];
		cin>>arr[1];
		hcf = __gcd(arr[0],arr[1]);
		i = 2;
		while(i<size)
		{
			cin>>arr[i];
			hcf = __gcd(hcf,arr[i-1]);
			i++;
		}
		for(i = 0; i<size; i++)
			cout<<(arr[i]/hcf)<<' ';
		cout<<'\n';
	}
	return 0;
}
