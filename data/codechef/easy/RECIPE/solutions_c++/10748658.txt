#include <iostream>
#include<math.h>
using namespace std;

int main()
{
	int t,i;
	cin>>t;
	for(i=0;i<t;i++)
	{
		int n;
		cin>>n;
		int a[n],b[n],min=1001,gcd;
		for(int j=0;j<n;j++)
		{
			cin>>a[j];
			if(a[j]<min)
				min=a[j];
		}
		for(int j=1;j<=min;j++)
		{
			int flag=1;
			for(int k=0;k<n;k++)
			{
				if(a[k]%j!=0)
					flag=0;
			}
			if(flag==1)
				gcd=j;
		}
		for(int k=0;k<n;k++)
		{
			b[k]=a[k]/gcd;
		}
		for(int k=0;k<n;k++)
		{
			cout<<b[k]<<" ";
		}
		cout<<""<<endl;
		
	}
	return 0;
}
