#include<iostream>
#include<math.h>

using namespace std;
int mi;

int findsmallest(int x[] , int z)
{
  mi = x[0];
 for(int i = 0 ; i<z ; i++)
 {
   if(x[i]<mi)
      mi = x[i];

 }

 return mi;

}

int main()
{
 int Test;
 int a[55];
 int N , k , chk = 0 , div =1;

 cin>>Test;
 while(Test--)
 {
  cin>>N;
  for(int i = 0 ; i<N ; i++)
  {
   cin>>a[i];
  }
  k = findsmallest(a , N);

  for(int j = 1 ; j<=k ; j++)
  {
    for(int i = 0 ; i<N ; i++)
    {
      if(a[i]%j!=0)
        {
         chk++;
        }

    }
   if(chk==0)
   {
     div = j;
   }
    chk = 0;
  }

 for(int i = 0 ; i<N ; i++)
  cout<<a[i]/div <<" ";


 }



}
