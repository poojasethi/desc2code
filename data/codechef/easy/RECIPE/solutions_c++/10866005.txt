#include<stdio.h>
#define f for(i = 0; i < n; i++)

int gcd(int a, int b){
	if(a == 0)	return b;
	return gcd(b % a, a);
}

int main(){
	int t, n; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		int a[n], i, g;
		f	scanf("%d", &a[i]);
		g = a[0];
		for(i = 1; i < n; i++)	g = gcd(g, a[i]);
		f	printf("%d ", a[i] / g);
		printf("\n");
	}		
	return 0;
}
