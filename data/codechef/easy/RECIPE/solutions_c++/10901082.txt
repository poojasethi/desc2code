#include<iostream>

using namespace std;

int main(void){
    int t, n, hcf=1, temp, p, q;
    cin>>t;
    while(t--){
        cin>>n;
        int a[n];
        for(int i=0; i<n; i++)
            cin>>a[i];
        p = a[0];
        for(int i=1; i<n; i++){
            q = a[i];
            while(p != 0 && q !=0){
                if(p>q)
                    p = p % q;
                else
                    q = q % p;
            }
            if(p == 0)
                p = q;
        }
        hcf = p;
        for(int i=0; i<n; i++){
            cout<< a[i]/hcf<<" ";
        }
        cout<<"\n";
    }
}
