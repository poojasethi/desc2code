#include <bits/stdc++.h>
using namespace std;
const int N = 55;
int n, a[N];

int main() {
    int t; cin >> t;
    while (t--) {
        cin >> n >> a[1];
        int k = a[1];
        for (int i = 2; i <= n; i++) {
            cin >> a[i];
            k = __gcd(k, a[i]);
        }
        for (int i = 1; i <= n; i++) cout << a[i]/k << " ";
        cout << endl;
    }
}
