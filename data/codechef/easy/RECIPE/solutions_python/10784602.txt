for Cycle in xrange(int(raw_input().strip())):
	Ingredients = map(float, (raw_input().strip().split(" "))[1:])

	MaxFactor = int(max(Ingredients))

	ReductionFactor = 1
	for FactorReductor in xrange(1, (MaxFactor + 1)):
		BadFactor = False
		for Ingredient in Ingredients:
			if not (Ingredient/FactorReductor).is_integer():
				BadFactor = True
				break
		if not BadFactor:
			ReductionFactor = ReductionFactor if ReductionFactor > FactorReductor else FactorReductor

	XRange = len(Ingredients)
	PrintStr = str(int(Ingredients[0]/ReductionFactor))

	for Ingredient in Ingredients[1:]:
		PrintStr += " {0}".format(int(Ingredient/ReductionFactor))

	print PrintStr
