def gcd(a, b):
    if b == 0:
        return a
    return gcd(b, a%b)

if __name__ == "__main__":
    t = int(raw_input())
    while t:
        x = raw_input()
        a = [int(i) for i in x.split(" ")]
        a.pop(0)
        g = a[0]
        for i in a:
            g = gcd(g, i)
        print " ".join([str(i/g) for i in a])
        t -= 1
