def GCD(a, b):

    if b == 0:
        return a
    else:
        return GCD(b, a % b)

test = int(raw_input())
while(test):
	k = map(int,raw_input().split())
	k.remove(k[0])
	#print k
	ans=reduce(GCD, k)
	if(ans <= min(k)):
		newlist = [x / ans for x in k]
	print ' '.join(map(str, newlist))
	test-=1
