tests = int(raw_input())
for _ in range(tests):
  line = map(int, raw_input().split())
  unique_ingredients = line[0]
  factors = []
  for n in line[1:]:
    [factors.append(pf) for pf in range(1, n + 1) if not n % pf]
  count_of = {}
  for factor in sorted(factors, reverse=True):
    count_of[factor] = count_of.get(factor, 0) + 1
    if count_of[factor] == unique_ingredients:
      gcf = factor
      break
  print ' '.join([str(n / gcf) for n in line[1:]])
