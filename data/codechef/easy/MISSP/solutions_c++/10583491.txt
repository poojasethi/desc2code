#include <iostream>
using namespace std;
 
int main() {
	int *a,n,temp,i,l,j,t;
	cin>>t;
	a=new int[n];
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n-1;i++)
		{
		for(j=0;j<n-i-1;j++)
		{
			if(a[j]>a[j+1])
			{
				temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		}
	}
	for(l=0;l<n-1;l+=2)
	{
	if(a[l]==a[l+1])
	continue;
	else
	break;
	}
	cout<<a[l]<<endl;
	}
	return 0;
}
