#include <iostream>

using namespace std;

int main()
{
        int t, n, m, k, num;
        int track[100];
        cin>>t;

        while(t--){
               cin>>n>>m>>k;

                for(int i=1; i<=n; i++)
                        track[i]=0;

                 for(int i=0; i<m; i++){
                        cin>>num;
                        track[num]+=1;
                 }

                for(int i=0; i<k; i++){
                        cin>>num;
                        track[num]+=1;
                }

                int count1=0, count2=0;
                for(int i=1; i<=n; i++){
                        if(track[i]==2) count1++;
                        else if (track[i]==0) count2++;
                }

                cout<<count1<<" "<<count2<<endl;
        }
    return 0;
}
