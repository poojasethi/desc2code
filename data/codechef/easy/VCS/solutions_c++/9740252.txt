#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	int i;
	int a[101]={0},b[101]={0};
     int n,m,k,s;
     cin>>n>>m>>k;
     int ans=n-m-k;
     int count=0;
     for(int i=0;i<m;i++)
     {
     	cin>>s;
     	a[s]=1;
     	
	 }
	 for(int i=0;i<k;i++)
	 {
	 	cin>>s;
	 	b[s]=1;
	 	
	 }
	 for(int i=1;i<=100;i++)
	 {
	 	if(a[i]&&b[i])
        count++;
	 }
	 cout<<count<<" "<<ans+count<<"\n";
}
	
}
