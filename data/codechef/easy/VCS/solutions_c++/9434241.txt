#include<iostream>
using namespace std;

int t, iandt,niandnt, n, m, k, i,x;

struct file 
{
       int ig,tr;
       } f[100];

int main()
{
    cin>>t;
    while(t--)
    {
              iandt=0;
              niandnt=0;
              cout<<"\n";
              cin>>n>>m>>k;
              for(i=1;i<=n;i++)
              {
                               f[i].ig=f[i].tr=0;                 
              }
              cout<<"\n";
              for(i=1;i<=m;i++)
              {
                               cin>>x;
                               f[x].ig=1;
              }
              cout<<"\n";
              for(i=1;i<=k;i++)
              {
                               cin>>x;
                               f[x].tr=1;
              }
              for(i=1;i<=n;i++)
              {
                               if((f[i].ig==1)&&(f[i].tr==1))  iandt++;
                               if((f[i].ig==0)&&(f[i].tr==0))  niandnt++;         
              }
              cout<<"\n"<<iandt<<" "<<niandnt;
              
              
                        
    }
    return 0;
}
