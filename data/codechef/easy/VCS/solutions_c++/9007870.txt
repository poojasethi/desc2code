#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		
	
		long long int i,a[100000]={0},b[100000]={0},k,n,m,g,sum1=0,sum2=0;
		scanf("%lld%lld%lld",&n,&m,&k);
		for(i=0;i<m;i++)
		{
			scanf("%lld",&g);
			a[g]=1;
		}
		for(i=0;i<k;i++)
		{
			scanf("%lld",&g);
			b[g]=1;
		}
		for(i=0;i<100000;i++)
		{
			if(a[i]==1&&b[i]==1)
			sum1++;
			
		}
		for(i=1;i<=n;i++)
		{
			if(b[i]==0&&a[i]==0)
			{
				sum2++;
			}
		}
		printf("%lld %lld\n",sum1,sum2);
	}
	return 0;
}
