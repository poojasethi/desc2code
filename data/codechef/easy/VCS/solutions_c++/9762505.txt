#include<iostream>
using namespace std;
int main()
{
   int t,n,m,k,data,tr,un;
   bool a1[110]={0},b1[110]={0};
   cin>>t;
   while(t--)
   {
       cin>>n>>m>>k;
       un=tr=0;

       for(int i=0;i<m;i++)
       {
           cin>>data;
           a1[data]=true;
       }

       for(int i=0;i<k;i++)
       {
           cin>>data;
           b1[data]=true;
       }

      for(int i=1;i<=n;i++)
        if(a1[i]==b1[i] && b1[i]==true)
            tr++;
        else if(a1[i]==b1[i] && b1[i]==false)
            un++;
        cout<<tr<<" "<<un<<"\n";

      for(int i=1;i<=n;i++)
            a1[i]=b1[i]=false;

   }

}
