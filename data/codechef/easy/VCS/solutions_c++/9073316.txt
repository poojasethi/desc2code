#include<iostream>
using namespace std;
int T,N,M,K,i,tr[100],ig[100];
int ti,uu;
int main()
{
    cin>>T;
    while(T--)
    {
        cin>>N>>M>>K;
        ti=uu=0;
        int tot[101] = {0};
        for(i=0;i<M;i++)
        {
            cin>>ig[i];
            tot[ig[i]]++;
        }
        for(i=0;i<K;i++)
        {
            cin>>tr[i];
            tot[tr[i]]++;
        }
        for(i=1;i<=N;i++)
        {
            if(tot[i]==2)
                ti++;
            else if(tot[i]==0)
                uu++;
        }
        cout<<ti<<" "<<uu<<"\n";
    }
    return 0;
}
