#include<iostream>
using namespace std;
int main()
{
int t,n,id,td,tdid,_tdid,i,j;
// t=test cases; n=source files; id=ignored; td=tracked;
// tdid= total files that are tracked and ignored
// _tdid= total files that are untracked and unignored
// i,j= indexes to be used for arrays
cin>>t;
// input number of test cases
while(t--) // perform the whole process t times
{
cin>>n>>id>>td;
int ida[id],tda[td];
// ida= aray of ignored files
// tda= aray of tracked files
tdid=_tdid=0;
// initializing total number of files to be zero
for(i=0;i<id;i++)
cin>>ida[i];              // to input sequence of ignored files in the aray ida
for(i=0;i<td;i++)
cin>>tda[i];             // to inpute sequence of tracked files in the aray tda
for(i=0;i<id;i++)        // loops to compare ida aray to tda aray
{
for(j=0;j<td;j++)
{
if(ida[i]==tda[j])
tdid++;
}
}
_tdid=n-td-id+tdid;      // by set theroy of exclusion and inclusion
cout<<tdid<<" "<<_tdid<<endl;
}                         // ending while loop
return 0;
}                       // ending program
