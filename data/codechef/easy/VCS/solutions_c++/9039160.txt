#include<iostream>
using namespace std;

int main()
{
int t,m,n,k,a[200],b[100],i,j,f,s;
	cin>>t;
     while(t--)
     {
     	cin>>n>>m>>k;
     	for(i=1;i<=n;i++)
     	{
     		a[i]=0;
     		b[i]=0;
     	}
     	for(i=1;i<=m;i++)
     	{
     		cin>>j;
     		a[j]++;
     	}     	
     	for(i=1;i<=k;i++)
     	{
     		cin>>j;
     		b[j]++;
     	}
     	f=0;s=0;
     	for(i=1;i<=n;i++)
     	{
     		if(a[i]>0 && b[i]>0)
     		f++;
     		else if(a[i]==0 && b[i]==0)
     		s++;
     		
     	}
     	cout<<f<<" "<<s<<endl;
     }
}
