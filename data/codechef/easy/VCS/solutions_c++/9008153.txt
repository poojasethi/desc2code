#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,k,m,a[101],b[101],x,c1=0,c2=0;
    int i;
    for( i=0;i<101;i++)
    a[i]=b[i]=0;
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>k;
        c1=0;
        c2=0;
        for( i=0;i<101;i++)
        a[i]=b[i]=0;
        for(i=1;i<=m;i++)//ignored
        {
            cin>>x;
            a[x]++;
        }
        for(i=1;i<=k;i++)//tracked
        {
            cin>>x;
            b[x]++;
        }

        //ignored and tracked
        for(i=1;i<=n;i++)
        {
            if(a[i]!=0 && b[i]!=0)
            c1++;
        }
        //untracked and unignored
        for(i=1;i<=n;i++)
        {
            if(a[i]==0 && b[i]==0)
            c2++;
        }

        cout<<c1<<" "<<c2<<endl;
    }
    return 0;
}
