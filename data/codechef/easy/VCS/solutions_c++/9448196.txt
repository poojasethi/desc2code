#include<iostream>
using namespace std;
int main(void)
{
    ios::sync_with_stdio(false);
    int t; cin>>t;
    while(t--)
    {
        bool ignore[101]={false},track[1001]={false};
        
        int n,m,k;
        cin>>n>>m>>k;

        int x;
        for(int i=1;i<=m;i++)
        {
            cin>>x;
            ignore[x]=true;
        }
        for(int i=1;i<=k;i++)
        {
            cin>>x;
            track[x]=true;
        }
        int cnt1=0,cnt2=0;
        for(int i=1;i<=n;i++)
        {
            if(ignore[i]&&track[i])
                cnt1++;
            if(!ignore[i]&&!track[i])
                cnt2++;
        }
        cout<<cnt1<<" "<<cnt2<<"\n";

    }
}
