#include<iostream>
using namespace std;
int main()
{
	int t,n,m,k,a[100];
	cin>>t;
	while(t--)
	{
		cin>>n>>m>>k;
		int temp,c1=0,c2=0;
		for(int i=1;i<=n;i++)
		{
			a[i]=0;
		}
		for(int i=1;i<=m;i++)
		{
			cin>>temp;
			a[temp]++;
		}
		for(int i=1;i<=k;i++)
		{
			cin>>temp;
			a[temp]++;
		}
		for(int i=1;i<=n;i++)
		{
			if(a[i]==2)
			{
				c1++;
			}
			else if(a[i]==0)
			{
				c2++;
			}
			
		}
		cout<<c1<<" "<<c2<<endl;
		
		
		
	}
}
