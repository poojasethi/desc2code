#include <iostream>

using namespace std;

int main()
{
    int t,n,m,k,value;
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>k;
        int arr[100]={0};
        for(int i=0;i<m;i++)
        {
            cin>>value;
            arr[value-1]++;
        }
        for(int i=0;i<k;i++)
        {
            cin>>value;
            arr[value-1]++;
        }
        m=0;
        k=0;
        for(int i=0;i<n;i++)
        {
            if(arr[i]==0)
            {
                k++;
            }
            if(arr[i]==2)
            {
                m++;
            }
        }
        cout<<m<<" "<<k<<endl;
    }
    return 0;
}
