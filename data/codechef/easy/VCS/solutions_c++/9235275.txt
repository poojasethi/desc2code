#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int t,n,m,k,uu,it,i,a;
     scanf("%d",&t);
     while(t--)
     {
         uu=0;
         it=0;
          scanf("%d %d %d",&n,&m,&k);
          int tt[101]={0};
          for(i=0;i<m;i++)
            {
                scanf("%d",&a);
                tt[a]=1;


          }

          for(i=0;i<k;i++)
            {
                scanf("%d",&a);
                tt[a]++;


          }

          for(i=1;i<=n;i++)
          {
              if(tt[i]==0)
                uu++;
                if(tt[i]==2)
                    it++;
          }
         printf("%d %d\n",it,uu);
     }



}
