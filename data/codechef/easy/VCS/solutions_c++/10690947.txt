#include <iostream>
#include <algorithm> 
using namespace std;
        int main()
        {
            int t,n,m,k,i,j,c;
            cin>>t;
            while(t--)
            {
                    cin>>n>>m>>k;
                     int a[m],b[k];
                    for(i=0;i<m;i++)
                    cin>>a[i];
                    for(i=0;i<k;i++)
                    cin>>b[i];
                    c=0;
                    for(i=0;i<m;i++)
                    {
                    for(j=0;j<k;j++)
                    {
                    if(a[i]==b[j])
                    {
                    c++;
                    break;
                    }
                    }
                    }
                    printf("%d %d\n",c,n-(m+k-c));
            }
            return 0;
        }
