#include<iostream>
using namespace std;
int main()
{
    int i,j,t,n,fl,p,q,m,k,x,y,a[100],b[100],c[100];
    cin>>t;
    while(t--)
    {
        x=0;y=0;
        cin>>n>>m>>k;
        for(i=0;i<m;i++)
        cin>>a[i];
        for(i=0;i<k;i++)
        cin>>b[i];
        for(i=0;i<m;i++)
        {
            for(j=x;j<k;j++)
            {
                if(a[i]==b[j])
                {
                    x++;
                    break;
                }
            }
        }
        p=1;j=0;q=0;
        while(p<=n)
        {
            fl=0;
        for(i=q;i<m;i++)
        {
            if(a[i]==p)
            {
                fl=1;
                q++;
            }
        }
        if(fl==0)
        {
            c[j]=p;
            j++;
        }
        p++;
        }
        q=0;y=j;
        for(i=0;i<k;i++)
        {
            for(p=q;p<j;p++)
            {
                if(b[i]==c[p])
                {
                    y--;
                    q++;
                    break;
                }
            }
        }
        cout<<x<<" "<<y<<endl;

    }
    return 0;
}
