#include<bits/stdc++.h>
using namespace std;

void solve()
{
    int n,m,k;
    cin>>n>>m>>k;
    int a[n+1];
    for(int i=1;i<=n;i++)a[i]=0;
    while(m--)
    {
        int x;
        cin>>x;
        a[x]++;
    }
    while(k--)
    {
        int x;
        cin>>x;
        a[x]++;
    }
    int x=0,y=0;
    for(int i=1;i<=n;i++)
    {
        if(a[i]==2)x++;
        if(a[i]==0)y++;
    }
    cout<<x<<" "<<y<<endl;
}

int main()
{
    int t;
    cin>>t;
    while(t--)solve();
    return 0;
}
