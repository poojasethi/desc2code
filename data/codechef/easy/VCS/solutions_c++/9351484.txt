#include<bits/stdc++.h>

using namespace std;

int main(void)
{
   int t; scanf("%d",&t);
   while(t--)
   {
      int n,m,k; scanf("%d %d %d",&n,&m,&k);
      
      bool ignored[101]={false}, tracked[101]={false};
      
      for(int i=0;i<m;i++)
      {
         int temp; scanf("%d",&temp);
         ignored[temp]=true;
      }
      for(int i=0;i<k;i++)
      {
         int temp; scanf("%d",&temp);
         tracked[temp]=true;
      }
      
      int ti=0,uu=0;
      for(int i=1;i<=n;i++)
      {
         if(ignored[i]==true && tracked[i]==true)
           ti++;
         
         if(ignored[i]==false && tracked[i]==false)
            uu++;
      }
      cout<<ti<<" "<<uu<<endl;
   }
   return 0;
}
