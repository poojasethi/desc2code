#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n,m,k;
        cin >> n >> m >> k;
        int a[m+1];
        int b[k+1];
        for(int i=0;i<m;i++)
            cin >> a[i];
        for(int i=0;i<k;i++)
            cin >> b[i];
        int ans1=0;
        int ans2=0;
        for(int i=1;i<=n;i++)
        {
            int flag1=0;
            int flag2=0;
            for(int j=0;j<m;j++)
            {
                if(a[j]==i)
                {
                    flag1=1;
                    break;
                }
            }
            for(int j=0;j<k;j++)
            {
                if(b[j]==i)
                {
                    flag2=1;
                    break;
                }
            }
            if(flag1==0 && flag2==0)
                ans1++;
            if(flag1==1 && flag2==1)
                ans2++;
        }
        cout << ans2 << " " << ans1 << endl;
    }
    return 0;
}
