#include<stdio.h>
int main()
{
	int t,m,n,k;
	scanf("%d",&t);
	while(t--)
	{
		int count2=0,count1=0;
		scanf("%d",&n);
		int a[102]={0},i,x;
		scanf("%d%d",&m,&k);
		while(m--)
		{
			scanf("%d",&x);
			a[x-1]++;
		}
		while(k--)
		{
			scanf("%d",&x);
			a[x-1]++;
		}
		for(i=0;i<n;i++)
		{
			if(a[i]==2)
				count2++;
			if(a[i]==0)
				count1++;
			
		}
		printf("%d %d\n",count2,count1);
	}
}
