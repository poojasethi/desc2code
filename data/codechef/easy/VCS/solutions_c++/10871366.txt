#include<stdio.h>
int main(){
	int t, n, m, k; scanf("%d", &t);
	while(t-- && scanf("%d%d%d", &n, &m, &k)){
		int ha[n + 1], hb[n + 1], ti = 0, uu = 0, no, i;
		for(i = 1; i <= n; ha[i] = hb[i] = 0, i++);
		while(m-- && scanf("%d", &no))	ha[no] = 1;
		while(k-- && scanf("%d", &no))	hb[no] = 1;
		for(i = 1; i <= n; i++){
			if(ha[i] && hb[i]) ti++;
			if(!ha[i] && !hb[i])	uu++;
		}
		printf("%d %d\n", ti, uu);
	}
	return 0;
}
