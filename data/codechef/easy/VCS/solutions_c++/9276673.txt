#include "stdio.h"
#include <cstring>

int main()
{
	int t,i,j,l,z,counta=0,countb=0;
	int n,m,k,num;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&n);
		scanf("%d",&m);
		scanf("%d",&k);
		
		int ar[n+1],br[n+1];
		memset(ar,0,sizeof(ar));
		memset(br,0,sizeof(br));
		for(j=0;j<m;j++)
		{
			scanf("%d",&num);
			ar[num]=num;
		}
		
		for(l=0;l<k;l++)
		{
			scanf("%d",&num);
			br[num]=num;
		}
		
		
		
		for(z=0;z<=n;z++)
		{
			if(ar[z]==br[z] && ar[z]!=0)
			counta++;
			
			else if(ar[z]==br[z] && ar[z]==0)
			countb++;
		}
		
		printf("%d %d\n",counta,countb-1);
		counta=countb=0;
	}
	
}
