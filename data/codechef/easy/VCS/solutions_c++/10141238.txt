#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
       int T,n,m,k,i,flag,flag1,count,count1;
       scanf("%d",&T);
       while(T--)
       {
              scanf("%d%d%d",&n,&m,&k);
              int a[m],b[k];
              count=count1=0;
              for(i=0;i<m;i++)
                     scanf("%d",&a[i]);
              for(i=0;i<k;i++)
                     scanf("%d",&b[i]);
              while(n!=0)
              {
                     flag=0;flag1=0;
                     for(i=0;i<m;i++)
                     {
                            if(a[i]==n)
                                   {
                                          flag=1;
                                          break;
                                   }
                     }
                     for(i=0;i<k;i++)
                     {
                            if(b[i]==n)
                            {
                                   flag1=1;
                                   break;
                            }
                     }
                     if(flag==1&&flag1==1)
                            count++;
                     if(flag==0&&flag1==0)
                            count1++;
                            n--;
              }
              printf("%d %d\n",count,count1);
       }
       return 0;
}
