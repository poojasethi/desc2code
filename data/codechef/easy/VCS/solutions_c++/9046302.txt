#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int* arr;
        int n,m,k;
        int p;
        int tra=0,utra=0;
        cin>>n>>m>>k;
        arr = new int[n+1];
        for(int i=1;i<=n;i++)
            arr[i] =0;
        for(int i=1;i<=m;i++)
        {
            cin>>p;
            arr[p]++;
        }
        for(int i=1;i<=k;i++)
        {
            cin>>p;
            arr[p]++;
        }
        for(int i=1;i<=n;i++)
        {
            if(arr[i]==2)
                tra++;
            if(arr[i]==0)
                utra++;
        }
        cout<<tra<<" "<<utra<<endl;
    }
}
