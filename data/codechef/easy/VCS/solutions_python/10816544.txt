for t in xrange(int(raw_input())):
  n, m, k = map(int, raw_input().split())
  ma = map(int, raw_input().split())
  kb = map(int, raw_input().split())
  both, neither = 0, 0
  for i in range(1, n + 1):
    if i in ma and i in kb:
      both += 1
    elif i not in ma + kb:
      neither += 1
  print both, neither
