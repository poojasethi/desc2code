def difflist(nooffiles,all):
    sum = 0
    nosum = 0
    
    for i in range(1,nooffiles):
        c = all.count(i)
        if c==2:
            all.remove(i)
            sum += 1
        elif c == 0 :
            nosum +=1
    return sum,nosum
