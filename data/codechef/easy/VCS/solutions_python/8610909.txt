
t = int(raw_input()) #number of test cases
for e in range(t):
    n = int(raw_input().split()[0])
    m = map(int, raw_input().split()) #ignored 
    k = map(int, raw_input().split()) #tracked
    
    it = 0
    nit = 0
    
    for x in m:
        if x in k:
            it+=1
    
    for x in range(1, n+1):
        if (x not in k) and (x not in m):
            nit+=1
            
    print it, nit
