import sys
test_cases = int(sys.stdin.readline())
while(test_cases > 0):
    test_cases -= 1
    no_of_soruce_files, no_of_ignored_source_file, no_of_tracked_source_file = map(int, sys.stdin.readline().split(' '))
    ignored_source_file = set(map(int, sys.stdin.readline().split()))
    tracked_source_file = set(map(int, sys.stdin.readline().split()))
    all_files = set([x for x in range(1, no_of_soruce_files + 1)])
    tracked_and_ignored_source_file = ignored_source_file.intersection(tracked_source_file)
    untracked_files = all_files.difference(tracked_source_file)
    unignored_files = all_files.difference(ignored_source_file)
    untracked_and_unignored_files = untracked_files.intersection(unignored_files)
    print len(tracked_and_ignored_source_file), len(untracked_and_unignored_files)
