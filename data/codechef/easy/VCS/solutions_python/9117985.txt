test = int(raw_input())
while(test > 0):
	test -= 1
	n, m, k = map(int, raw_input().split())
	assert(n >=m and n >= k)
	A = map(int, raw_input().split())
	B = map(int, raw_input().split())
	both = 0
	if(len(A) < len(B)):
		for i in A:
			if(i in B):
				both += 1
	else:
		for i in B:
			if(i in A):
				both += 1
	print both, n-m-k+both
