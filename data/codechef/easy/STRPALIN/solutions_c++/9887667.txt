#include<iostream>
#include<string.h>
using namespace std ;

int main(){
	string A , B ; //consists only of lower case alphabets
	int T ;
	cin>>T ;
	for(int i = 0;i<T;i++){
		cin>>A>>B ;
		bool flag = false ;
		for(size_t p = 0;p<A.length();p++){
			for(size_t q = 0;q<B.length();q++)
				if(A[p] == B[q]){
					flag = true ;
					break ;
				}
		}
		if(flag)
			cout<<"Yes\n" ;
		else
			cout<<"No\n" ;		
	} 
}
