#include<stdio.h>
#include<string.h>
int main(){
	int t, i, la, lb, f; scanf("%d", &t); char a[1001], b[1001];
	while(t-- && scanf("%s%s", a, b)){
		f = 0, la = strlen(a), lb = strlen(b); int h[26] = {0};
		for(i = 0; i < la; i++)	h[a[i] - 97] = 1;
		for(i = 0; i < lb; i++)	if(h[b[i] - 97]){f = 1, printf("Yes\n"); break;}
		if(f == 0)	printf("No\n");
	}
	return 0;
}
