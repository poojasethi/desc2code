#include<stdio.h>
#include<iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string a,b;
		bool flag=0;
		cin>>a>>b;
		int i=0;
		while(a[i]!='\0')
		{
			int j=0;
			while(b[j]!='\0')
			{
				if(a[i]==b[j])
				{
					printf("Yes\n");
					flag=1;
					break;
				}
				j++;
			}
			if(flag)	break;
			i++;
		}
		if(!flag)	printf("No\n");
	}
	return 0;
}
