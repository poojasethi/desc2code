#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char A[1001];
        char B[1001];
        int s1[200]={0};
        int s2[200]={0};
        int i,l1,l2,fl=0;
        scanf("%s",A);
        scanf("%s",B);
        l1=strlen(A);
        l2=strlen(B);
        for(i=0;i<l1;i++)
        {
            s1[A[i]]=1;
        }
        for(i=0;i<l2;i++)
        {
            s2[B[i]]=1;
        }
        for(i=0;i<200;i++)
        {
            if(s1[i]==1 && s2[i]==1)
            {
                fl=1;break;
            }
        }
        if(fl==1)
        {
            cout<<"Yes\n";
        }
        else
        {
            cout<<"No"<<endl;
        }
    }
    return 0;
}
