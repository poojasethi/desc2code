#include<iostream>
using namespace std;
int main(){
	int t,count,l1,l2,i,j;
	string s1,s2;
	cin>>t;
	while(t--){
		cin>>s1>>s2;
		count=0;
		l1=s1.length(),l2=s2.length();
		for(i=0;i<l1;i++){
			for(j=0;j<l2;j++){
				if(s1[i]==s2[j]){
					++count;
					break;
				}
			}
		}
		if(count>0)
		cout<<"Yes";
		else
		cout<<"No";
		cout<<endl;
	}
	return 0;
}
