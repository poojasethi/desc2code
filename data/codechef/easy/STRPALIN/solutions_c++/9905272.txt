#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
   char a[1001],b[1001];
   int c[26];
   // std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
       cin>>a>>b;
       int flag=0;
       memset(c,0,sizeof(c));
       int x=strlen(a);
       int y=strlen(b);
       for(int i=0;i<x;i++)
       {
           c[a[i]-'a']++;
       }
       for(int i=0;i<y;i++)
       {
           if(c[b[i]-'a']>=1)
           {
              flag=1;
               break;
           }

       }
       if(flag==1)
        cout<<"Yes"<<endl;
       else
        cout<<"No"<<endl;


    }
}
