#include<iostream>
using namespace std;
char a[1000],b[1000];
void check()
{
	int t=0;
for (int i=0;a[i]!='\0'&&t!=1;i++)
{
	for (int j=0;b[j]!='\0'&&t!=1;j++)
	{
		if (a[i]==b[j]){
			cout<<"Yes"<<endl;
t++;
		}
		
}
}
if (t==0){
			
		cout<<"No"<<endl;
	}


}
int main()
{
	int input;
	cin>>input;
	for (int i=0;i<input;i++)
	{
	cin>>a;
	cin>>b;
	check();

	}

}
