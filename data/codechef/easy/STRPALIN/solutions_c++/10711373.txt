#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    int ch[26]={0};
    char A[1001];
    char B[1001];
    scanf("%s %s",A,B);
    int i;
    for(i=0;i<strlen(A);i++)
    {
    ch[A[i]-'a']++;
    }
    int flag=0;
    for(i=0;i<strlen(B);i++)
    {
    if(ch[B[i]-'a'])
    {
    flag=1;
    break;
    }
    }
    if(!flag)
    {
    printf("No\n");
    }
    else
    {
    printf("Yes\n");
    }
    }
    return 0;
}
