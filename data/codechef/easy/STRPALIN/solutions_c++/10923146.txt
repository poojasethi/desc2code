#include <iostream>
#include<string>

using namespace std;

int main()
{
        int arr[27];
        int t;
        cin>>t;

        while(t--){
                string str1, str2;
                cin>>str1>>str2;

                for(int i=0; i<26; i++){
                        arr[i]=0;
                }

                for(int i=0; i<str1.size(); i++){
                        arr[str1[i]-97]++;
                }

                int i;
                for(i=0; i<str2.size(); i++){
                        if(arr[str2[i]-97]>0){
                                cout<<"Yes"<<endl;
                                break;
                        }
                }
                if(i==str2.size()){
                        cout<<"No"<<endl;
                }
        }

        return 0;
}
