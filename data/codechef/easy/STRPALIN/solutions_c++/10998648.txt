#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
 
int main() {
	int t;
	cin>>t;
	while(t)
	{
	    string a,b;
	    cin >> a >> b;
	    int flag=1;
	    for(int i=0;i<a.length();i++){
	        for(int j=0;j<b.length();j++){
	            if(a[i]==b[j]){
	                flag=0;
	                break;
	            }
	        }
	        if(flag==0){
	            break;
	        }
	    }
	    if(flag==1){
	        cout << "No" <<endl;
	    }
	    else{
	        cout << "Yes" <<endl;
	    }
	   
    	t--;
	}
	return 0;
}
