#include<iostream>
using namespace std;

int main()
{
	int t,j,i,d;
	char b;
	cin>>t;
	while(t>0)
	{
		d=0;
		char a[1001],b[1001];
		cin>>a;
		cin>>b;
		for(i=0;a[i]!='\0';i++)
		{
			for(j=0;b[j]!='\0';j++)
			{
				if(a[i]==b[j])
				{
					d=1;
					break;
				}
			}
		}
				
		if(d==1)
		{
			cout<<"Yes";
		}
		else
		{
			cout<<"No";
		}
		t--;
		cout<<endl;
	}
	return 0;
}
