#include <iostream>
using namespace std;

int main() {
	int t,ans;
	string a,b;
	cin>>t;
	while(t--)
	{
		cin>>a>>b;
		ans=0;
		for(int i=0;i<a.size()&&ans==0;i++)
		{
			for(int j=0;j<b.size();j++)
			{
				if(a[i]==b[j]){
				ans=1;
				break;
			}	
		}
		}
		if(ans==1)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
	return 0;
}
