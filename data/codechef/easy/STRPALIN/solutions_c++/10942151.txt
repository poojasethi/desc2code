    #include <bits/stdc++.h>
    using namespace std;
     int main() {
    	int t;
    	cin>>t;
    	while(t--)
    	{
    		int x=0;
    		string a,b;
    		cin>>a>>b;
    		for(int i=0;i<a.length();i++)
    		{
    			for(int j=0;j<b.length();j++)
    			{
    				if(a[i]==b[j]){
    				x++;
    				break;
    			}	
    		}
    		}
    		if(x!=0)
    		cout<<"Yes"<<endl;
    		else
    		cout<<"No"<<endl;
    	}
    	return 0;
    }
