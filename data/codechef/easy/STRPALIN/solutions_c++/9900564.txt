    #include <iostream>
    #include<cstring>
    using namespace std;
     
    int main() {
    	int t;
    	cin>>t;
    	while(t--)
    	{
    	int i,j,flag=0,l1,l2;
    	string s1,s2;
    	cin>>s1;
    	cin>>s2;
    	l1=s1.length();
    	l2=s2.length();
    	for(i=0;i<l1;i++)
    	{
    	for(j=0;j<l2;j++)
    	if(s1[i]==s2[j])
    	{flag=1;}}
    	if(flag==1)
    	cout<<"Yes"<<endl;
    	else
    	cout<<"No"<<endl;
    	}
    	return 0;
    }
