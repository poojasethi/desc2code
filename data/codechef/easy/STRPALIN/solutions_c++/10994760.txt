#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin >> t;
	while(t--) {
	    string s, t;
	    cin >> s >> t;
	    int cnts[26], cntt[26];
	    memset(cnts, 0, sizeof(cnts));
	    memset(cntt, 0, sizeof(cntt));
	    for(int i = 0; i < s.size(); i++) cnts[s[i] - 'a'] = 1;
	    for(int i = 0; i < t.size(); i++) cntt[t[i] - 'a'] = 1;
	    int flag = 0;
	    for(int i = 0; i < 26; i++) if(cnts[i] == 1 && cntt[i] == 1) { flag = 1; break; }
	    (flag) ? puts("Yes") : puts("No");
	}
	return 0;
}
