#include <iostream>
using namespace std;
#include <cstring>
 
int main(){
    int t;
    cin >> t;
    while(t--){
        char a[1500], b[1500];
        int found = 0;
        cin >> a >> b;
        for(int i = 0 ; a[i] != '\0' ; i++){
            for(int j = 0 ; b[j] != '\0' ; j++){
                if(a[i] == b[j]){
                    found = 1;
                    break;
                }
                if(found == 1){
                break;
              }
            }
        }
        if(found == 1){
            cout << "Yes" << endl;
        }
        else{
            cout << "No" << endl;
        }
        found = 0;
    }
}
