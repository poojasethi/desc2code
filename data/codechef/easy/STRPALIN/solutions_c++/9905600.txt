#include<iostream>

using namespace std;

int isLast(string a, int i){
	char ch = char(i+97);
	for(long index = 0 ; index < a.length()-1 ; index++){
		if(a[index] == ch){
			return false;
		}
	}
	return true;
}

int main(){
	int t;
	cin>>t;
	while(t--){
		string a, b;
		cin>>a>>b;
		int helpA[26] = {0};
		int helpB[26] = {0};
		for(long i = 0 ; i < a.length() ; i++){
			helpA[int(a[i])-97] = 1;
		}
		for(long i = 0 ; i < b.length() ; i++){
			helpB[int(b[i])-97] = 1;
		}
		int isPalindrom = 0;
		for(int i = 0 ; i < 26 ; i++){
			if(helpA[i] && helpB[i]){
//				int isLastChar = isLast(a, i);
				isPalindrom = 1;
				break;
			}
		}
		if(isPalindrom){
			cout<<"Yes\n";
		}
		else{
			cout<<"No\n";
		}
	}
	
	return 0;
}
