#include<stdio.h>
typedef long long ll;

int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		char s1[1001],s2[1001];
		ll s_1[123]={0},s_2[123]={0};
		ll i,flag=0;
		scanf("%s%s",s1,s2);
		i=0;
		while(s1[i]!='\0'){
			s_1[s1[i]]++;
			i++;
		}	i=0;
		while(s2[i]!='\0'){
			s_2[s2[i]]++;
			i++;
		}
		for(i=97;i<=122;i++)
		if(s_1[i]>0&&s_2[i]>0)
	    flag=1;
	    if(flag==1)
			printf("Yes\n");
			
    else  if(flag==0)
       printf("No\n");
	}return 0;
}
