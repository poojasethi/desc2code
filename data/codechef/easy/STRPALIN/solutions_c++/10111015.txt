#include <iostream>
using namespace std;
#define ll long long

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   string a,b;
        cin>>a>>b;
        ll x[26]={0},y[26]={0},i,k=0;
        for(i=0;i<a.length();i++)
            ++x[a[i]-97];
        for(i=0;i<b.length();i++)
            ++y[b[i]-97];
        for(i=0;i<26;++i)
            if(x[i] && y[i])
            {   k=1;
                break;
            }
        if(k)
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
