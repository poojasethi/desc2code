def issub_palindrome(mx, mn):
  for x in range(len(mx)):
    for i in range(len(mx)-x):
      pred = mx[i:i+x+1]
      if pred[::-1] in mn:
        return 'Yes'
  return 'No'

for case in xrange(int(input())):
  print issub_palindrome(raw_input(), raw_input())
