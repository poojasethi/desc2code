def process(A,B):
    count = 0;
    for i in range(0,len(A)):
        for j in range(0,len(B)):
            if(A[i]==B[j]):
                count = count+1
    return count
def input1():
      
    A = raw_input("")
    B = raw_input("")
    c1=process(A,B)
    if(c1>0):
        print'Yes'
    else:
        print'No'
T = input("")
for k in range (0,T):
    input1()
