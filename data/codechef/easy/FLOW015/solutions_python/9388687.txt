#!/usr/bin/env python2

def isleapyear(y):
	if (y % 400 == 0) or (y % 4 == 0 and y % 100):
		return 1
	return 0

def first_day_of_year():
	x = int(raw_input())

	while x:
		y = int(raw_input())
		s = 1
		for i in range(1901, y + 1):
			if isleapyear(i - 1):
				s += 2
			else:
				s += 1
		s %= 7;
		if s == 0:
			print 'sunday'
		elif s == 1:
			print 'monday'
		elif s == 2:
			print 'tuesday'
		elif s == 3:
			print 'wednesday'
		elif s == 4:
			print 'thursday'
		elif s == 5:
			print 'friday'
		elif s == 6:
			print 'saturday'
		x -= 1

if __name__ == '__main__':
	first_day_of_year()
