for x in xrange(0, input()):
	first = raw_input()
	second = raw_input()
	matches = True
	if (len(first) == len(second)) :
		for x in xrange(0, len(first)):
			if ((first[x] != '?' and second[x] != '?')):
				if first[x] != second[x]:
					matches = False
	else:
		matches = False
	if matches:
		print("Yes")
	else:
		print("No")
