n = input()
for i in range(0, n):
	s1 = raw_input()
	s2 = raw_input()
	flag = 0
	for j in range(0, len(s1)):
		if (s1[j] != '?' and s2[j] == '?') or (s1[j] == '?' and s2[j] != '?') or (s1[j] == '?' and s2[j] == '?') or (s1[j] == s2[j]):
			flag = 1
			#continue
		else:
			flag = 0
			break
	if flag == 1:	
		print "Yes"
	else:
		print "No"
