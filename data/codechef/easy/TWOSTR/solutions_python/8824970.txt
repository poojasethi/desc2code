def chef_and_wildcard_matching():
    # TWOSTR, print YES if two strings can be matched.
    for _ in xrange(int(raw_input())):
        s1 = raw_input()
        s2 = raw_input()
        match = True
        for i in range(len(s1)):
            if s1[i] == s2[i] or s1[i] == '?' or s2[i] == '?':
                continue
            else:
                match = False
        print 'Yes' if match else 'No'

chef_and_wildcard_matching()
