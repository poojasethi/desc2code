import sys
test_cases = int(sys.stdin.readline())
while(test_cases > 0):
    test_cases -= 1
    str1 = sys.stdin.readline()
    str2 = sys.stdin.readline()
    len = str1.__len__()
    flag = 0
    for i in range(0, len):
        if(str1[i] != str2[i]):
            if(str1[i] != '?' and str2[i] != '?'):
                flag = 1
                break
    if(flag == 0): print "Yes"
    else: print "No"
