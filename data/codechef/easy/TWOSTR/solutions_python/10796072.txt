for _ in xrange(int(raw_input().strip())):
	Subject1 = raw_input()
	Subject2 = raw_input()
	SubjectLength = len(Subject1)

	LoopFucked = False
	for ChrIndx in xrange(SubjectLength):
		ChrOne = Subject1[ChrIndx]
		ChrTwo = Subject2[ChrIndx]

		if ChrOne == ChrTwo:
			continue
		elif ChrOne == "?" and ChrTwo.islower():
			continue
		elif ChrTwo == "?" and ChrOne.islower():
			continue
		elif ChrOne == ChrTwo == "?":
			continue
		else:
			print "No"
			LoopFucked = True
			break

	if not LoopFucked:
		print "Yes"
