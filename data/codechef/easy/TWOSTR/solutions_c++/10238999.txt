#include<iostream>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
	string a,b;
	cin>>a;
	cin>>b;
	int len=a.length(),flag=0;
	for(int i=0;i<len;i++)
	{
		if((a[i]!='?' && b[i]!='?') && (a[i]!=b[i]))
		{
			flag=1;
			break;
		}
	}
	if(flag==0)
	cout<<"Yes\n";
	else
	cout<<"No\n";
}
return 0;
}
