#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
    while(t--)
	{
	    char a[12], b[12];
	    int f=1;
	    scanf("%s%s",a,b);
	    int len = strlen(a);
	    for(int i=0;i<len;i++) {
	        if(a[i]=='?'||b[i]=='?')
	        continue;
	        else if(a[i]!=b[i]) {
	            f=0;
	            break;
	        }
	    }
	    if(f){
	        printf("Yes\n");
	    } else{
	        printf("No\n");
	    }

	}
	return 0;
}
