#include<iostream>
#include<string>

using namespace std;

int main(void){
    int t;
    string a, b;
    cin>>t;
    while(t--){
        int j=0;
        cin>>a>>b;
        for(int i=0; i<a.length(); i++){
            if(a[i]!=b[i] && a[i]!='?' && b[i]!='?'){
                j++;
                break;
            }
        }
        if(j== 0)
            cout<<"Yes"<<"\n";
        else
        cout<<"No"<<"\n";
    }
}
