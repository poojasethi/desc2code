#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string a,b;
        cin>>a>>b;
        int i=0,c=0;
        while(a[i]!='\0')
        {
            if(a[i]==b[i]||a[i]=='?'||b[i]=='?')
                ++c;
            ++i;
        }
        if(c==i)
            cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
}
