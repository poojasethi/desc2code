#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<vector>

using namespace std;

int main()
{
    int t;
    char str1[15],str2[15];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",&str1);
        scanf("%s",&str2);
    int n=strlen(str1);
    int count=0,curr=0;
    for(int i=0;i<n;i++)
    {
        if(str1[i]=='?' || str2[i]=='?')
            count++;
        else if(str1[i]==str2[i])
            curr++;
    }
    if(n-count-curr==0)
        printf("Yes\n");
    else
        printf("No\n");
   }
return 0;
}
