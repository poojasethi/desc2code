#include<string>
#include <iostream>


int main()
{

	int testcases;
	std::cin >> testcases;
	for (int x = 0; x < testcases; x++)
	{
		std::string a;
		std::string b;
		std::cin >> a;
		std::cin >> b;
		int count = 0;
		for (int y = 0; y < a.length(); y++)
		{
			if (a[y] == b[y]||a[y]=='?'&&b[y]!='?'||a[y]!='?'&&b[y]=='?')
				count = count + 1;
			else
				count = count;
		}
		if (count == a.length())
		{
			std::cout << "Yes" << std::endl;
		}
		else
		{
			std::cout << "No" << std::endl;
		}







	}
}
