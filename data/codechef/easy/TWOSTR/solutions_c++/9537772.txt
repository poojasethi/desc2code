#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t,c1,c2,l1,l2,i,j;
    cin>>t;
    string s1,s2;
    while(t--)
    {
        c1=1;
        cin>>s1>>s2;
        l1=s1.length();
        l2=s2.length();
       if(l1==l2)
       {
           for(i=0;i<l1;i++)
           {
               if(s1[i]==s2[i])
                continue;
              if(s1[i] != s2[i])
               {
                  if(( s1[i] == '?' ) || (s2[i] == '?'))
                        continue;
               }
                   c1=0;
                   break;
           }
           if(c1)
           cout<<"Yes\n";
           else
            cout<<"No\n";
       }
       else
        cout<<"No\n";

    }

}
