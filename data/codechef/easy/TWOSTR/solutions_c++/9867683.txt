#include<iostream>
using namespace std;
int main()
{
    int t,n,i,j,flag;
    cin>>t;
    string s1,s2;
    while(t--)
    {
       cin>>s1>>s2;flag=0;

         n=s1.length();
         i=0;
         while(i<n)
         {
             if((s1[i]==s2[i])||(s1[i]=='?')||(s2[i]=='?'))
                 i++;
             else
               {
                   
                flag=1;break;}

         }
         if(flag==1) cout<<"No"<<endl; else cout<<"Yes"<<endl;

    }
}
