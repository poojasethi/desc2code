#include<math.h>
#include<iostream>
#include<stdio.h>
using namespace std;

using namespace std;

int main()
{
 int Test , chk = 0;
 char str1[11];
 char str2[11];

 cin>>Test;

 while(Test--)
 {
  cin>>str1;
  cin>>str2;

  for(int i = 0 ; str1[i]!=0 ; i++)
  {
   if(str1[i]!='?'&&str2[i]!='?')
   {
    if(str1[i]!=str2[i])
    {
     chk++;
     break;
    }

   }

  }
  if(chk==0)
  {
   cout<<"Yes"<<endl;
   chk = 0;
  }
  else
  {
   cout<<"No"<<endl;
   chk = 0;
  }

 }
}
