#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string x,y;
		int flag=1;
		cin>>x>>y;
		
		for(int i=0;i<x.size();i++)
		{
			if(x[i]!=y[i])
			{
				if(x[i]!='?' && y[i]!='?')
				{
					flag=0;
					break;
				}
			}
		}
		if(flag==1)
		cout<<"Yes";
		else
		cout<<"No";
		cout<<endl;
	}
}
