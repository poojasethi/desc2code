#include<cstdlib>
#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        string a,b;
        cin>>a>>b;
         bool possible=true;
        if(a.length()==b.length())
        {
            for(int i=0;i<a.length();++i)
            {
                if(a[i]=='?'||b[i]=='?'||a[i]==b[i])
                    continue;
                else
                {
                    possible=false;
                    break;
                }
            }
        }
        else
        {
            possible=false;
        }
        if(possible)
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
