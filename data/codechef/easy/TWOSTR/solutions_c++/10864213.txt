#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
        int t;
        char str1[11], str2[11];
        scanf("%d", &t);

        while(t--){
                scanf("%s%s", &str1[0], &str2[0]);
                int i=0, j=0;

                while(str1[i]!='\0'){
                        if(str1[i]!=str2[j] && str1[i]!='?' && str2[j]!='?')
                                break;
                        i++, j++;
                }

                if(str1[i]=='\0')
                        printf("Yes\n");
                else printf("No\n");
        }
    return 0;
}
