#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
 
int f()
{
	char s[20],s1[20];
	int i,flag=0;
	
	scanf("%s",&s);
	scanf("%s",&s1);
 
	for(i=0;s[i]!='\0';i++)
	{
		if((s[i]!=s1[i])&&(s[i]!='?' && s1[i]!='?'))
		{
			flag = 1;
			break;
		}
	}
	if(flag)
		printf("No\n");
	else
		printf("Yes\n");
}
 
int main()
{
	int t;
 
	scanf("%d",&t);
	while(t--)
	{
		f();
	}
	
	return 0;
}
