#include<iostream>
#include<string>
using namespace std;
int main(){
	int t,count,l,l1,i;
	string st,st1;
	cin>>t;
	while(t--){
		cin>>st;
		cin>>st1;
		count=0;
		l=st.length();
		l1=st1.length();
		if(l==l1){
			for(i=0;i<l;i++){
				if(st[i]=='?'||st1[i]=='?')
				continue;
				if(st[i]!=st1[i]){
					count=1;
					break;
				}
			}
			if(count==1)
			cout<<"No";
			else
			cout<<"Yes";
		}
		else
		cout<<"No";
		cout<<endl;
	}
	return 0;
}
