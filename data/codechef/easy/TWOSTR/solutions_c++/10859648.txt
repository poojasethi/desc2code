#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t,match,i;
	string s,q;
	cin>>t;
	while(t--)
	{	match=0;
		cin>>s>>q;
		for(i=0;i<(s.length());i++)
		{
			if(s[i]==q[i]||s[i]=='?'||q[i]=='?')
				match+=1;
		}
		if(match==s.length())
		  cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}
    return 0 ;
}
