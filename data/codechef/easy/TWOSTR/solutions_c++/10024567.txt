#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>
 
int f(char *s,char *s1)
{
	int i,flag=1;
	for(i=0;s[i]!='\0';i++)
	{
		if((s[i]!=s1[i])&&(s[i]!='?' && s1[i]!='?'))
		{
			flag = 0;
			break;
		}
	}
	if(flag)
		printf("Yes\n");
	else
		printf("No\n");
}
 
int main()
{
	char s[1000],a[1000];
	int t,c;
 
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",&s);
		scanf("%s",&a);
		f(s,a);
	}
	
	return 0;
}
