#include <bits/stdc++.h>
using namespace std;
string s, t;

int main() {
    int n; cin >> n;
    while (n--) {
        cin >> s >> t;
        bool k = 1;
        for (int i = 0; i < s.size(); i++)
        if (s[i] != '?' && t[i] != '?' && s[i] != t[i]) {
            cout << "No" << endl;
            k = 0;
            break;
        }
        if (k) cout << "Yes" << endl;
    }
}
