#include <iostream>
#include <cstring>
using namespace std;

int main() {
int t;
cin>>t;
while(t--)
{
	char a[11],b[11];
	int len,i,flag=0;
	cin>>a;
	cin>>b;
	len=strlen(a);
	for(i=0;i<len;i++)
	{
		if(a[i]!=b[i] && a[i]!='?' && b[i]!='?')
		{
			flag=1;
			break;
		
		}
	}
	if(flag==0)
	cout<<"Yes"<<"\n";
	else
	cout<<"No"<<"\n";
}
	return 0;
}
