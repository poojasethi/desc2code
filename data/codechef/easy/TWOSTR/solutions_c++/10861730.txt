#include<stdio.h>
#include<string.h>
int main(){
	int t, i, la, f; scanf("%d", &t); char a[12], b[12];
	while(t-- && scanf("%s%s", a, b)){
		la = strlen(a), f = 0;
		for(i = 0; i < la; i++)	if((a[i] != '?' && b[i] != '?') && (a[i] != b[i])){f = 1, printf("No\n"); break;}
		if(f == 0)	printf("Yes\n");
	}
	return 0;
}
