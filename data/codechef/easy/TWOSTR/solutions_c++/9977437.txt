#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int flag=0;
		string a,b;
		cin>>a;
		cin>>b;
		for(int i=0;i<a.length();i++)
		{
			if(a[i]!=b[i] && a[i]!='?' && b[i]!='?')
				{
					flag=1;
					//cout<<"Flag is 1 now\n";
					goto label;
				}
		}
		label:;
	if(flag==0)
		cout<<"Yes\n";
	else
		cout<<"No\n";
	}
	return 0;
}
