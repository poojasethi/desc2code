#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char a[20],b[20];
        int i=0;
        cin>>a>>b;
        int n=strlen(a);
        int m=strlen(b);
        if(m!=n)
            cout<<"No"<<endl;
            else
            {
                while(i!=n)
                {
                    if(a[i]==b[i])
                        i++;
                    if(a[i]!=b[i])
                    {
                        if(a[i]=='?'||b[i]=='?')
                            i++;
                        else
                            break;
                    }
                }
                if(i==n)
                    cout<<"Yes"<<endl;
                else
                    cout<<"No"<<endl;

            }
    }
}
