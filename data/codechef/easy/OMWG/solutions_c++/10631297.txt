#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,m,i,j;
	cin >> t;
	while (t--)
	{
		cin >> n >> m;
		int c = 0;
		for (i=0;i<n;i++)
		{
			for (j=0;j<m;j++)
			{
				if (i==0)
					c = m-1;
				else if (j==0)
					c = c + 1;
				else
					c = c + 2;
		//		cout << c << "\n";
			}
		}
		cout << c << "\n";
	}
}
