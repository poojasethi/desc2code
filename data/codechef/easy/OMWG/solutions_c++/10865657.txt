#include<iostream>
using namespace std;
int main()
{
	int t,score=0,m,n;
	cin>>t;
	for(int k=0;k<t;k++)
	{	
		score=0;
		cin>>m>>n;
		for(int i=0;i<m;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(i==0 && j==0)
				score=score+0;
				else if(i==0||j==0)
				score=score+1;
				else
				score=score+2;	
			}
		}
		cout<<score<<endl;
	}
}
