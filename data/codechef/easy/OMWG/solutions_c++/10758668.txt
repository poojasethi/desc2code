#include<iostream>
#include<cmath>

using namespace std;

int main(){
int T;
cin>>T;
while(T--!=0){
int N, M;
cin>>N>>M;
int arr[1010][1010];
for(int i=0;i<N+2;i++){
for(int j=0;j<M+2;j++){
arr[i][j]=0;
}
}

int sum=0;

for(int i=1;i<N+1;i++){
for(int j=1;j<M+1;j++){
sum+=arr[i][j-1]+arr[i][j+1]+arr[i-1][j]+arr[i+1][j];
arr[i][j]=1;


}


}

cout<<sum<<endl;
}


return 0;

}
