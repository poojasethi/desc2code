int main()
{
	int x,i;
	scanf("%d",&x);
	int r[x],c[x],sum[x];
	for(i=0;i<x;++i)
	{

	sum[i]=r[i]+c[i]-2+2*(r[i]*c[i]-r[i]-c[i]+1);
}

	{
	printf("%d \n",sum[i]);
}

}
