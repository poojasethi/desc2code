#include<iostream>
using namespace std;
#include<cstdio>
#include<cmath>
#include<vector>
#include<algorithm>

int main()
{

int t,n,m,i;

scanf("%d",&t);
while(t--){
	long int ans;
	scanf("%d%d",&n,&m);
	if(n==m)
		ans=(n*n-n)*2;
	else {
		int temp;
		if(n>m){
			temp=n;
			n=m;
			m=temp;

		}
		ans=(n*n-n)*2+(m-n)+2*(m-n)*(n-1);
	}
printf("%ld\n",ans);

}
return 0;
}
