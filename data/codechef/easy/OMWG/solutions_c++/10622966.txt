using namespace std;
int m,n;

void xuly()
{
    scanf("%d%d", &n, &m);
    if(n>m) swap(n,m);
    int res = 2*(n*n - n) ;
    res = res + 2*(m-n)*(n-1) + m-n;
    printf("%d\n", res);
}

void doc()
{
    int t;
    scanf("%d", &t);
    for(int i=1; i<=t; i++) xuly();
}

int main()
{

}
