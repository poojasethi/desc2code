#include<bits/stdc++.h>
int main()
{
	int t,n,m,i,j;
	scanf("%d",&t);
	while(t--)
	{
		long long int sum=0,count=0;
		scanf("%d%d",&m,&n);
		int mat[m][n],col[m][n];
		for(i=0;i<m;i++)
		{
			for(j=0;j<n;j++)
			mat[i][j]=col[i][j]=0;
		}
		for(i=0;i<m;i++)
		{
			
			for(j=0;j<n;j++)
			{
				count=0;
				mat[i][j]=1;	
				if(i>0)
				{
					count++;
				}
				if(j>0)
				{
					count++;
				}
				col[i][j]=count;
			}
		}
		for(i=0;i<m;i++)
		{
			for(j=0;j<n;j++)
			{
				sum+=col[i][j];
			}
		}
		printf("%lld\n",sum);
	}
}
