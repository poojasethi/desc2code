def stick(B,n):
	A=[0]*1001
	for i in B:
		A[i]+=1
	for i in range(1000,-1,-1):
		if A[i]>=2:
			if A[i]>=4:
				return i*i
			else:
				for j in range(i-1,-1,-1):
					if A[j]>=2:
						return i*j
	return -1
