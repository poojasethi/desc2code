t = input()
while t:
    length = input()
    l = []
    i = 0
    string = map(int,raw_input().split())
    string.sort(reverse=True)
    while i<(length-1):
        if(string[i]==string[i+1]):
            l.append(string[i])
            i+=2
        else:
            i+=1

    if(len(l)<2):
        print "-1"
    else:
        print l[0]*l[1]
    t-=1
