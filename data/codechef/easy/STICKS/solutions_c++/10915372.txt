#include<bits/stdc++.h>
using namespace std;
 

 int main() {
 	long int i,j,n,r[2],t,a[1005];
 	
	cin >> t;
	while(t--) {
		cin >> n;
	
		for(i=0;i<n;i++)
			cin >> a[i];
		sort(a,a+n);

		i=n-1;j=0;
		while(i && j<2) {
			if(a[i]==a[i-1]) {
				r[j]=a[i];
				i--;j++;
			}
			i--;
		}
		if(j!=2)
			cout << "-1" << endl;
		else
			cout << r[0]*r[1] << endl;
	}
	return 0;
}
