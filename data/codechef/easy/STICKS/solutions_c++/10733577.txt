#include <stdio.h>
 
int main(void) {
  int t,n,a[1000],i,area,temp,j,count;
  scanf("%d",&t);
  while(t--)
  {  area=1;
    count=0;
      scanf("%d",&n);
      for(i=0;i<n;i++)
      {
          scanf("%d",&a[i]);
      }
     
      for(i=0;i<n-1;i++)
      {
        for(j=0;j<n-i-1;j++)
        {
            if(a[j]<a[j+1])
            {
                temp=a[j];
                a[j]=a[j+1];
                a[j+1]=temp;
            }
        }
      }
      
    
      for(i=0;i<n-1&&count<2;i++)
      {  
          if(a[i]==a[i+1])
          { 
              count++;
               area=area*a[i];
               i++;
         }
             
              
             
     }
    
          if(count<2)
          {
              printf("-1\n");
          }
          else
          {
              printf("%d\n",area);
          }
      
  }
	return 0;
}
