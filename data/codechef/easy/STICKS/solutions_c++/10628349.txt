#include <iostream>
using namespace std;

int main()
{
   int t;
   int ans,n;
   int i,j,a[1009];
   cin>>t;
   while(t--)
   {int b[1009]={0};
   for(int i=0;i<1002;i++)
   b[i]=0;
       cin>>n;
       for(i=0;i<n;i++)
       {
       cin>>a[i];
       b[a[i]]++;
       }
       int m=0;
       ans=0;
       for(j=1005;j>=0;j--)
       {
           if((b[j]>=4))
           {
               if(m==0)
               {ans=j*j;
               break;}
               else
               {
                   ans=m*j;
                   break;
               }
               
           }
           
           if(b[j]==3||b[j]==2)
           {
               ans=m*j;
               m=j;
               if(ans>0)
               break;
               
           }
           
           
           
       }
       if(ans==0)
       cout<<"-1\n";
       else
       cout<<ans<<"\n";
       
       
       
   }
    return 0;
}
