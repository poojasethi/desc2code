#include<iostream>
using namespace std;
int main()
{
 int t,n,a[1001],i,b[1001],j;
 cin>>t;
 while(t--)
 {
  cin>>n;
  for(i=0;i<n;i++)
  {cin>>a[i];b[i]=0;}
  for(i=0;i<n;i++)
  {
   for(j=i+1;j<n;j++)
   {
    if(a[i]<a[j])
    {
     int h=a[i];
     a[i]=a[j];
     a[j]=h;
    }
   }
  }
  j=0,i=0;
  while(i<n-1)
  {
   if(a[i]==a[i+1])
   {
    b[j]=a[i];
    i=i+2;j++;
   }
   else i++;
  }
  if(b[1]!=0)
  {cout<<b[0]*b[1]<<"\n";}
  else
  {cout<<"-1\n";}
 }
 return 0;
}
