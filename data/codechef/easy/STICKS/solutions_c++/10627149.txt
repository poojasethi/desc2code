#include<bits/stdc++.h>
using namespace std;
int main()
{
	int T,n,A[1001],i,j,k,a=1,count=0;
	
	cin>>T;
	while(T--)
	{
		cin>>n;
		a=1;
		count=0;
		int op=0;
		int x[1001]={0};
		for(i=0;i<n;i++)
		{
			cin>>A[i];
			x[A[i]]++;
		}
		
		for(i=1000;i>0 && count<2;i--)		
		{
			if(x[i]>3 && op==0)
			{
				a*=i*i;
				count=2;
			}
			else if(x[i]>1)
			{
				a*=i;
				count++;
				op=1;
			}
		}
		if(count==2)
		cout<<a<<endl;
		else
		cout<<"-1\n";
	}
	return 0;
}
