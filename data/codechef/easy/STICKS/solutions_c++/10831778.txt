#include <stdio.h>
#include <memory.h>
#include <algorithm>
using namespace std;

const int MAX = 1000;
int test, n, a, cnt[MAX + 1], res, tra;

int main(){
    //freopen("in.txt", "r", stdin);
    scanf("%d", &test);
    while (test--){
        scanf("%d", &n);
        memset(cnt, 0, sizeof(cnt));
        while (n--){
            scanf("%d", &a);
            ++cnt[a];
        }
        res = -1;
        tra = 0;
        for (int i = 1; i <= MAX; ++i)
        if (cnt[i] >= 2){
            if (cnt[i] >= 4)
                res = i * i;
            else
            if (tra > 0)
                res = max(res, i * tra);
            tra = i;
        }
        printf("%d\n", res);
    }
    return 0;
}
