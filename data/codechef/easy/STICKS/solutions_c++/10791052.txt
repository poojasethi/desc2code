#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n],temp[1005]={0};
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			temp[a[i]]++;
		}
		int flag=0,counter=0,ans=1;
		for(int i=1004;i>=1;i--)
		{
			if(temp[i]>=4)
			{
                if(counter==0)
                {
                    ans*=i*i;
                    counter+=2;
                }
                else
                {
                    ans*=i;
                    counter++;
                }
				
			}
			else if(temp[i]>=2)
			{
                ans*=i;
                counter++;
            }
			if(counter>=2)
				break;
		}
		if(counter<2)
			printf("-1\n");
		else
			printf("%d\n",ans);
	}
}
