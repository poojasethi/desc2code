using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int t,n;cin>>t;
    while(t-->0)
    {
        cin>>n;
        int arr[n+5];
        for(int i=1;i<=n;i++) cin>>arr[i];
        sort(arr+1,arr+n+1);
        int a=-1,b=-1;
        int ind1=0,ind2=0;
        for(int i=1;i<n;i++)
        {
            if(arr[i]==arr[i+1] )
            {
                ind1=i;
                a=arr[i];
            }
        }
        for(int i=1;i<ind1-1;i++)
        {
            if(arr[i]==arr[i+1] )
            {
                ind2=i;
                b=arr[i];
            }
        }
    if(ind1!=0&&ind2!=0)
        cout<<a*b<<endl;
    else cout<<-1<<endl;

    }
    return 0;
}
