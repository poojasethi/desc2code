#include <bits/stdc++.h>
using namespace std;
int main()
  {
      int t,n,i,x,y,z,area;
      scanf("%d",&t);
      while(t--)
      {   x=0;
          y=1;
          z=0;
          area=-1;
          scanf("%d",&n);
          int a[n];
          for(i=0;i<n;i++)
            scanf("%d",&a[i]);
          sort(a,a+n);
          for(int i=1;i<n;i++)
          {
              if(a[i]==a[i-1])
              {
                  if(z==0)
                  {
                     y++;
                     x=a[i];
                     z=1;
                  }
                  else
                  {
                      y++;
                      if(y%2==0)
                      {
                          area=max(area,a[i]*x);
                          x=a[i];
                      }

                  }

              }
              else
              {
                  y=1;
              }



          }
          if(area<0)
            printf("%d\n",-1);
          else printf("%d\n",area);
      }
  }
