#include <iostream>
#include<vector>
#include<algorithm>

using namespace std;

int main()
{
    int t,n,a[100100];
    vector<int> v;
    cin>>t;
    while(t--)
    {
        cin>>n;
        v.clear();
        for(int i=0; i<n; i++)
            cin>>a[i];
        sort(a,a+n);
        int c=0;
        int i=n-1;
        while(i>=1)
        {
            if(a[i]==a[i-1]&&c!=2)
            {
                --i;
                v.push_back(a[i]);
                c++;

            }
            --i;

        }
        if(c!=2)
            cout<<-1<<endl;
        else
            cout<<v[0]*v[1]<<endl;



    }
    return 0;
}
