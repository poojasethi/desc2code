#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
	long numOfTestCases;
	cin>>numOfTestCases;
	
	for(long tc =0; tc<numOfTestCases; tc++){
		
		
		long numOfSubTestCases;
		cin>>numOfSubTestCases;
		vector<long> vecArray;
		long tmp;
		
		for(long stc =0; stc<numOfSubTestCases; stc++){
			cin>>tmp;
			vecArray.push_back(tmp);			
		}
		
		sort(vecArray.begin(),vecArray.end());
		
		long side1, side2, finished =0;
		for(long stc =numOfSubTestCases; stc>0; stc--){
			if(vecArray[stc] == vecArray[stc-1]){
				side1 = vecArray[stc];
				for(long stc1 =stc-2; stc1>0; stc1--){
					if(vecArray[stc1] == vecArray[stc1-1]){
						side2 = vecArray[stc1];
						finished =1;
						stc1 = 0;
						stc =0; 
					}
				}
					
				
			}
			
		}
		if(finished ==1)
			cout<<side1*side2<<endl;
		else 
		cout<<"-1"<<endl;
		
		
		
	}
}
