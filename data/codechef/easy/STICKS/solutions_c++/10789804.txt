#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,l=0,b=0;
		scanf("%d",&n);
		int a[n];
		for (i=0;i<n;i++)
		scanf("%d",&a[i]);
		sort(a,a+n);
		for (i=1;i<n;i++)
		{
			if(a[i-1]==a[i]&&a[i-1]==a[i+1]&&a[i-1]==a[i+2])
			{
				b=a[i];
				l=a[i];
			}
			else{
					if(a[i]==a[i-1]&&a[i]>l)
			           {
			            	b=l;
				            l=a[i];
		            	}
			
		             else
			            {
		                	if(a[i]==a[i-1]&&a[i]<l&&a[i]>b)
		                	b=a[i];
		                }
			    }
		
		}
		if(l&&b)
        printf("%d\n",l*b);
        else
        printf("-1\n");
		
		
	}
}
