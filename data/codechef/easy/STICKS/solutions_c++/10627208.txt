#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
        long long i, n, a[10001], x;
        for(i=0;i<10001;i++)
            a[i]=0;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&x);
            a[x]++;
        }
        int p, q, r;
        p=0, q=0, r=0;
        for(i=0;i<10001;i++)
        {
            if(a[i]>=2)
            {
                p=q;
                q=i;
            }   
            if(a[i]>=4)
                r=i;
        }
        if((p==0)&&(r==0))
            printf("-1\n");
        else if((p*q)>=(r*r))
            printf("%lld\n",p*q);
        else
            printf("%lld\n",r*r);
    }
	return 0;
}
