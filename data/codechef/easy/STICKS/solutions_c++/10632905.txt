#include <cstdio>
int main()
{
	const int MAX_LENGTH = 1000;
	int trials; // # of trials
	int i; // index
	int stick_count[MAX_LENGTH + 1];
	int n; // number of sticks
	int s1; // longest side;
	int stick_length; // a stick length entry
	int outstanding; // outstanding rectangle sides count
	bool rectangle_formed;
	scanf("%d", &trials);
	while (trials--)
	{
		rectangle_formed = false;
		outstanding = 4;
		for (i = 1; i <= MAX_LENGTH; ++i)
			stick_count[i] = 0;
		scanf("%d", &n);
		for (i = 0; i < n; ++i)
		{
			scanf("%d", &stick_length);
			++stick_count[stick_length];
		}
		for (i = 1000; i > 0; --i)
		{
			if (stick_count[i] > 1)
			{
				if (outstanding == 4)
				{
					if (stick_count[i] > 3)
					{
						printf("%d\n", i*i);
						rectangle_formed = true;
						break;
					}
					else
					{
						s1 = i;
						outstanding = 2;
						continue;
					}
				}
				else
				{
					printf("%d\n", s1*i);
					rectangle_formed = true;
					break;
				}
			}
		}
		if (!rectangle_formed)
			printf("-1\n");
	}
}
