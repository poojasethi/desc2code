	int t,i,j,k;

		int n;si(n);
		for(i=0;i<n;i++) {
			si(a[i]);
		}
		sort(a,a+n);
		bool flag=0;
		int side1=-1,side2=-1;
		for(i=n-1;i>0;i--) {

				if(side1==-1) {
					side1=a[i];

				}
			}
		}
		if(flag) {
			printf("%d\n",side1*side2);
		} else {

		}
	}
