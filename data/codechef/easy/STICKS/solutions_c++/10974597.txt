#include<bits/stdc++.h>
using namespace std;
main(){
    int t;
    cin>>t;
    int n,i,j,k,flag=-1,cnt=0;
    for(k=0;k<t;k++){
        cin>>n;
        vector<int> v;
        vector<int> temp;
        for(i=0;i<n;i++){
            v.push_back(i);
            cin>>v[i];
        }
        for(i=0;i<n;i++){
            for(j=i+1;j<n;j++){
                if(v[i]==v[j]){
                    cnt++;
                    temp.push_back(v[i]);
                    v[i]=flag;
                    v[j]=flag;
                    flag--;
                }
            }
    }
        if(cnt>1){
            sort(temp.begin(),temp.end());
            cout<<((temp[temp.size()-1])*(temp[temp.size()-2]))<<endl;
            cnt=0;
        }
        else{
            char a[] = "-1";
            cout<<a<<endl;
            cnt=0;
        }
    }
}
