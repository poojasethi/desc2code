int findMaxRec(vector <pair <int, int> > nums) {
	sort (nums.rbegin(), nums.rend()); 

	int first = -1, second = -1;
	for (int i = 0; i < nums.size() && (first == -1 || second == -1); i++) {
		if (nums[i].second >= 4) {
			if (first == -1) {
				first = nums[i].first;
				second = nums[i].first;
			} else {
				second = nums[i].first;
			}
			continue;
		} 
		if (nums[i].second >= 2) {
			if (first == -1) {
				first = nums[i].first;
			} else {
				second = nums[i].first;
			}
			continue;
		}
	}
	return (first == -1 || second == -1) ? -1 : (first * second);
}

int main(){
	unsigned int T;
	cin >> T;

	while(T--){
		int n;
		scanf("%d",&n);

		vector <pair<int, int> > nums (1001, make_pair(0,0));
		for (int i = 0; i < n; i++) {
			int num;
			scanf("%d",&num);
			nums[num] = make_pair(num, nums[num].second+1);
		}

		printf("%d\n", findMaxRec(nums)); 
	}
	return 0;
}
