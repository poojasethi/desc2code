#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int i,c=0,area=1;
		int n;
		cin>>n;
		int a[n];
		for(i=0;i<n;i++){
			cin>>a[i];
		}
		sort(a,a+n);
		for(i=n-1;i>0;i--){
			if(a[i]==a[i-1] && c<2){
				area=area*a[i];
				c++;
				i--;
			}
		}
		if(c==2)
		cout<<area<<endl;
		else
		cout<<"-1"<<endl;
	}
	return 0;
}
