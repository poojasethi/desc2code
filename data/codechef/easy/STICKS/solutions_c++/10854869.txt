#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int i,a[n];
        for(i=0;i<n;i++)
        cin>>a[i];
        sort(a,a+n);
        int ans=1,p=0;

        for(i=n-2;i>=0;i--)
        {

            if(a[i]==a[i+1] && p<2)
            {
                ans = ans*a[i];
                i=i-1;
                p++;
            }
        }
        if(p==2)
        cout<<ans<<endl;
        else
        cout<<"-1\n";
    }
	return 0;
}
