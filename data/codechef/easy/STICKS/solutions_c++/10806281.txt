#include<iostream>
using namespace std;

int main()
{
	int a[1001],i,t,n,x,g,h;
	cin>>t;
	while(t>0)
	{
		
		g=0;
		h=0;
		for(i=0;i<1001;i++)
		{
			a[i]=0;
		}
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>x;
			a[x]++;
		}
		for(i=1000;i>=0;i--)
		{
			if(a[i]>1)
			{
				g=i;
				a[i]=a[i]-2;
				break;
			}
		}
		for(i=1000;i>=0;i--)
		{
			if(a[i]>1)
			{
				h=i;
				a[i]=a[i]-2;
				break;
			}
		}
		if(g>0&&h>0)
		{
			cout<<g*h;
		}
		else
		{
			cout<<"-1";
		}
		t--;
		cout<<endl;
	}
	return 0;
}
