#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	// your code goes here
	
	
	int t;
	cin>>t;
	while(t--)
	 {
	     int n;
	     cin>>n;
	     int a[n];
	     for(int i=0;i<n;i++)
	      {
	          cin>>a[i];
	      }
	      
	      sort(a,a+n);
	      
	      long long int sum=-1;
	      long long int max_sum=-1;
	      
	      int i=0;
	      int j=n-1;
	      while(i+1<j-1)
	      {  
	          if(a[i]==a[i+1]&&a[j-1]==a[j])
	           {
	               sum=a[i]*a[j];
	               i=i+2;
	               //j=j-2;
	           }
	          if(a[i]!=a[i+1])
	           {
	               i++;
	           }
	           if(a[j]!=a[j-1])
	            {
	                j--;
	            }
	          
	           
	           
	           if(sum>=max_sum)
	            {
	                max_sum=sum;
	            }
	           
	           
	       }
	      if(n<4)
	       {
	           cout<<"-1"<<endl;
	       }
	       else
	      
	      cout<<max_sum<<endl;
	      
	     
	     
	     
	     
	     
	 }
	
	
	
	return 0;
}
