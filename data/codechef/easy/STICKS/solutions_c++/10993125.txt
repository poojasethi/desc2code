#include <iostream>

using namespace std;

int main()
{
        int t, n, i, maxx;
        int arr[1000], index[1001];

        scanf("%d", &t);
        while(t--){

                scanf("%d", &n);
                for(int i=0; i<n; i++){
                        scanf("%d", &arr[i]);
                }

                for(i=0; i<=1000; i++){
                        index[i]=0;
                }
                for(i=0; i<n; i++){
                        index[arr[i]]++;
                }

                int first=0, second=0;

                for(i=1000; i>=0; i--){
                        if (index[i]>3 && !first){
                                first=second=i;
                                break;
                        }
                        else if(index[i]>=2 && !first){
                                first=i;
                        }

                        else if(index[i]>=2 && first){
                                second=i;
                                break;
                        }
                }

                //for(int i=0; i<=maxx; i++) cout<<index[i]<<" ";

               //cout<<first<<" "<<second<<endl;
               if(first&&second) printf("%d\n", first*second);
               else printf("-1\n");
        }
           return 0;
}
