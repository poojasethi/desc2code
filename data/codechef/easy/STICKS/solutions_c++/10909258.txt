#include <stdio.h>

using namespace std;

int main ()
{
    int t = 0, n = 0, a[1000], b[1010], i = 0, max = 0, smax = 0;
    scanf ("%d", &t);
    while (t--)
    {
         scanf ("%d", &n);
         for (i = 0; i < 1010; i++)
         {
              b[i] = 0;
         }
         for (i = 0; i < n; i++)
         {
              scanf ("%d", &a[i]);
              b[a[i]]++;
         }
         max = 0;
         smax = 0;
         for (i = 0; i < 1010; i++)
         {
              if (b[i] >= 4)
              {
                   max = i;
                   smax = i;
              }
              else if (b[i] >= 2)
              {
                   smax = max;
                   max = i;
              }
         }
         if ((smax * max) > 0)
         {
              printf ("%d\n", (smax * max));
         }
         else
         {
              printf ("-1\n");
         }
    }

    return 0;
}
