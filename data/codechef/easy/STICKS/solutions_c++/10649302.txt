#include<bits/stdc++.h>
using namespace std;
main()
{
	int k,t,n,p,q,cnt,i;
	vector<int> aux(10001);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		p=0;q=0;cnt=0;
		for(i=0;i<1001;i++)	
			aux[i]=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&k);
			aux[k]++;
		}
		for(i=1000;i>0;i--)
		{
			if(p==0)
			{	if(aux[i]>=2)
					p=i;
				if(aux[i]>=4)
					q=i;
				continue;
			}
			if(q==0)
			{
				if(aux[i]>=2)
					q=i;
			}
		}
		if(p==0 || q==0)	
			printf("-1\n");
		else
			printf("%d\n",p*q);
	}
	
}
