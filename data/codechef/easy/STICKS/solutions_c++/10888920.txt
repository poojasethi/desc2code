#include<iostream>
#include<algorithm>
using namespace std;

void cal_(int *a,int n){
    int count=0,mul=1;
           sort(a,a+n);
           if(n==1){
              cout<<-1<<endl;
           }
              else{
              for(int i=n-1;i>=0;i--){
                       if(a[i]==a[i-1]){
                                if(count==2){
                                    break;
                                }
                              mul=mul*a[i-1];
                              i-=1;
                              count++;
                       }

              }
                if(count==0||count==1){
                       cout<<-1<<endl;
                }
            else{
                cout<<mul<<endl;
            }
              }

}
int main(){
      int t;
        cin>>t;
          while(t--){
                 int n,a[10000];
                   cin>>n;
                       for(int i=0;i<n;i++){
                                cin>>a[i];
                       }
                       cal_(a,n);
          }


return 0;}
