#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main(){
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        vector<int> l (n);
        for(int i=0;i<n;i++){
            cin>>l[i];
        }
        int a=0,b=0;
        sort(l.rbegin(),l.rend());
        for(int i=0;i<n;i++){
            if(l[i]==l[i+1]){
                if(a==0){
                    a=l[i];
                }
                else if(b==0){
                    b=l[i];
                }
                i++;
            }
            if(a&&b) break;
        }
        if(a&&b) cout<<a*b<<"\n";
        else cout<<-1<<"\n";
    }
return 0;
}
