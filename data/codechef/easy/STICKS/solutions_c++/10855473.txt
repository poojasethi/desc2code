	int T;

	while(T--){
	    int N;
	    cin>>N;
	    int arr[1007];
	    map<int , int>M;
	    for(int i =0 ; i<N ;i++){
	        cin>>arr[i];
	        M[arr[i]]++;
	    }
	    map<int , int>::iterator it;
	    int max1 = INT_MIN , max2 = INT_MIN ;
	    for( it =M.begin(); it!=M.end();++it){
	        if(it->second >=2 && it->second < 4){
	            //cout <<it->first <<" "<<it->second<<endl;
	            if(max1 < it->first){
	                max2 = max1;
	                max1 = it->first;
	                
	            }
	        }else if(it->second >= 4){
	            if(max1 < it->first){
	                max1 = it->first;
	            }
	            if(max2 < it->first){
	                max2 = it->first;
	            }
	        }
	    }
	    long long area;

	        area = max1*max2;

}
