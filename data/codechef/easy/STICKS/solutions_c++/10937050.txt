#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
	int t,n,i=0,j=0;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n;
		int A[n],freq[1001],count=0,max=0,penmax=0;
		for(j=0;j<n;j++)
			A[j]=0;
		for(j=0;j<1001;j++)
			freq[j]=0;
		for(j=0;j<n;j++)
			cin>>A[j];
		for(j=0;j<n;j++)
		{
			freq[A[j]]+=1;
			//cout<<A[j]<<" "<<freq[A[j]]<<endl;
			if(freq[A[j]]==2)
			{
				count++;
				//cout<<"A[j]="<<A[j]<<endl;
				if(A[j]>max)
					max=A[j];
				//cout<<"max="<<max<<endl;
			}
		}
		if(freq[max]>=4)
			count=2;
		if(count<2)
			cout<<-1<<endl;
		else if (freq[max]>=4)
			cout<<max*max<<endl;
		else
		{
			for(j=0;j<n;j++)
			{
				if(freq[A[j]]>=2 && A[j]>penmax && A[j]!=max)
					penmax=A[j];
			}
			cout<<max*penmax<<endl;
		}
	}
}
