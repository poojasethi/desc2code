#include <bits/stdc++.h>
using namespace std;
long int k;
long long int f(int x)
{
    return ((long long)x*(x+1))/2;
}
long int bs(int lo,int hi)
{
    int mid = lo + (hi-lo)/2;
    if(lo<mid)
    {
       if(f(mid)<=k)
        return bs(mid,hi);
       return bs(lo,mid-1);
    }
    if(f(hi)<=k)
        return hi;
    if(f(lo)<=k)
        return lo;
    return -1;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>k;
        cout<<bs(1,1e5)<<endl;
    }
    return 0;
}
