#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	 {
	     int n;
	     cin>>n;
	     
	    long long  int ans=1;
	     
	     int l=1;
	     int r=n;
	     
	     while(l<=r)
	      {
	         long long  int mid=l-(l-r)/2;
	          
	          if(mid*(mid+1)/2==n)
	           {
	               ans=mid;
	               break;
	           }
	           
	           if(mid*(mid+1)/2>n)
	            {
	                r=mid-1;
	            }
	            else
	             { ans=mid;
	             l=mid+1;
	             }
	          
	          
	      }
	     
	     cout<<ans<<endl;
	     
	 }
	
	
	return 0;
}
