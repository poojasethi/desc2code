#include<iostream>
using namespace std;
int main()
{
    int t,n,h,tot;
    cin>>t;
    for(int i=1;i<=t;i++)
    {
        cin>>n;
        h=0;
        tot=0;
        while(tot<=n)
        {
            h++;
            tot=tot+h;
        }
        if(tot>n)h--;
        cout<<h<<endl;
    }
    return 0;
}
