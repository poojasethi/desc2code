#include<iostream>
using namespace std;
int main()
{
	int test,level;
	long long int n,low,mid,up;
	scanf("%d",&test);
	while(test){
		test--;
		scanf("%lld",&n);
		low=1;
		up=n;
	
		while(low<=up){
				mid=(low+up)/2;
			if(((mid)*(mid+1))/2<=n&&((mid+1)*(mid+2))/2>n){
				printf("%lld\n",mid);
				break;
			}
			else if(((mid)*(mid+1))/2>n){
				up=mid-1;
			}
			else{
				low=mid+1;
			}
		}
	}
	return 0;
}
