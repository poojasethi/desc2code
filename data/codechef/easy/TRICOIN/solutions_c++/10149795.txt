#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    long long int n,d,ans;
    cin>>t;
    while(t--)
    {
        cin>>n;
        d = 1+sqrt(1+8*n);
        d = d/2;
        ans = (d*(d+1))/2;
        if(n<ans)
            cout<<d-1<<endl;
        else
            cout<<d<<endl;
    }
    return 0;
}
