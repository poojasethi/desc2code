#include <iostream>
#include<math.h>
using namespace std;

int bs(int low, int high, int number){
        int mid = (high+low)/2;
        int l = mid*(mid+1)/2;
        int r = (mid+1)*(mid+2)/2;

        if(number>=l && number<r) return mid;
        else if(number<l) return bs(low, mid-1, number);
        else if(number>l) return bs(mid+1, high, number);

}

int main()
{
        int t, n, pos;
        cin>>t;

        while(t--){
                cin>>n;
                pos = bs(1, sqrt(n*2)+1, n);
                cout<<pos<<endl;
        }
    return 0;
}
