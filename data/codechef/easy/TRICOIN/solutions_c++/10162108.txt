#include <iostream>

using namespace std;

int main()
{
long long int n,s;
    int a[100],i,j;

    cin>>n;




    for(i=0;i<n;i++)
    {

       cin>>a[i];
    s=0;
    for(j=1;j<=a[i];j++)
    {
     s=s+j;
     if(s>=a[i])
     {

     break;
     }


    }
    if(s==a[i])
    cout<<j<<"\n";
    else
    cout<<j-1<<endl;
    }





    return 0;
}
