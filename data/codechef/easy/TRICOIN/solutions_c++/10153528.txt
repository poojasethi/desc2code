#include <iostream>

#include <cmath>

using namespace std;

bool check(int x,long long N){

	return (N>=(x*(x+1)/2.0))&&(N<(x+1)*(x+2)/2.0);

}

int height(long long N){

	int y=sqrt(2*N);

	for(int i=y-2;i<y+1;i++){

		if(check(i,N)) return i;

	}

}

int main(){

	//cout<<check(2,7)<<endl;

	int n;

	cin>>n;

	for(int i=0;i<n;i++){

		long long N;

		cin>>N;

		cout<<height(N)<<endl;

	}

}
