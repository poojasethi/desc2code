#include <iostream>
#include<bits/stdc++.h>
using namespace std;
 
int main()
{
int t;
cin>>t;
while(t--)
{
int n,ctr1=0,ctr2=0,ctr3=0,i,c=0,d=0;
cin>>n;
char a[n];
for(i=0;i<n;i++)
cin>>a[i];
 
for(i=0;i<n;i++)
{
if(a[i]=='R')
ctr1++;
else if(a[i]=='G')
ctr2++;
else
ctr3++;
}
 
c=max(ctr1,ctr2);
d=max(c,ctr3);
cout<<n-d<<endl;
}
return 0;
}
