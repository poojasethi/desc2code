t = int(raw_input())
for i in range(0,t):
    num = int(raw_input())
    val = [] 
    arr = []
    while num != 0:
        arr += [num%10]
        num /= 10
    for j in range(0,len(arr)):
        val += [arr[-1-j]]    
    flag = 0
    for k in range(0,len(arr)):
        if arr[k] != val[k]:
           flag = 1
           break
    if flag == 0:
        print("wins")
    else:
        print("losses")
