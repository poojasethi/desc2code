def BlockGame(n):
    answer = 0
    caseN = n
    while caseN >= 1:
        test = caseN
        counter = 0
        while test >= 1:
            test /= 10
            counter += 1
        answer += (caseN % 10) * 10 ** (counter - 1)
        caseN /= 10
    if answer == n:
        return 'wins'
    else:
        return 'losses'

cases = int(raw_input())
for i in range(0, cases):
    n = int(raw_input())
    print BlockGame(n)
