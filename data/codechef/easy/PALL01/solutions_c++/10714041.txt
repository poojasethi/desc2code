#include <iostream>
using namespace std;
int main(int argc, const char * argv[]) {
    int T,N,dig,rev,z;
    
    cin>>T;
    while(T!=0)
    {
        rev=0;
        cin>>N;
        z=N;
        while(N!=0)
        {
            dig=N%10;
            rev=rev*10+dig;
            N=N/10;
        }
        if(z==rev)
        cout<<"wins\n";
        else
            cout<<"losses"<<"\n";
        T--;
    }
}
