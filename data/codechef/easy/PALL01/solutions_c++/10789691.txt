#include <iostream>
using namespace std;

int main() {
    int t,n;
    cin>>t;
    while(t--)
    {
        int rem,orig;
        int s=0;
        cin>>n;
        orig=n;
        while(n)
        {
            rem=n%10;
            s=s*10+rem;
            n=n/10;
        }
        if(s==orig)
        cout<<"wins\n";
        else cout<<"losses\n";
    }
	return 0;
}
