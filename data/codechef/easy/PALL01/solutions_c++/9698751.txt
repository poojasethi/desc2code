#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int t;
	cin>>t;
	while(t--)
	{
		long int a,r,sum=0,temp;
		cin>>a;
		temp=a;
		while(a!=0)
		{
			r=a%10;
			sum=sum*10+r;
			a=a/10;
			
		}
		if(sum==temp)
		{
			cout<<"wins"<<endl;
			
		}
		else
		{
			cout<<"losses"<<endl;
			
		}
}
	return 0;
	
}
