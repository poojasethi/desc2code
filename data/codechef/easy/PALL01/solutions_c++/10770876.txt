#include <iostream>
#include <math.h>
using namespace std;

int main() 
{
	int t,i;
	cin>>t;
	for(i=0;i<t;i++)
	{
		int n,m,rev=0,l=0,temp;
		cin>>m;
		temp=m;
		n=m;
		while(temp>0)
		{
			l++;
			temp/=10;
		}
		for(int j=l-1;j>=0;j--)
		{
			rev+=(pow(10,j))*(n%10);
			n/=10;
		}
		if(rev==m)
			cout<<"wins"<<endl;
		else
			cout<<"losses"<<endl;
			
	}
	return 0;
}
