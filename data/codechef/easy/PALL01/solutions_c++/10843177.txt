#include<iostream>
using namespace std;
int main()
{
    int a,n,t,m,rev=0;
    cin>>a;
    while(a--)
    {
        cin>>n;
        m=n;
        while(n)
        {
            t=n%10;
            rev=rev+t;
            n=n/10;
            rev=rev*10;
        }
        rev=rev/10;
        //cout<<n<<"\t"<<rev<<"\t";
        if(rev==m)
        {
            cout<<"wins\n";
        }
        if(rev!=m)
        {
            cout<<"losses\n";
        }
        rev=0;
    }
    return 0;
}
