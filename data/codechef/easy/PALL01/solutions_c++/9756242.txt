#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int main()
{
	int t, r, x;
	cin>>t;
	long int n[t];
	for(int i = 0;i<t;i++)
	{
		
			cin>>n[i];	
	}
 
	for(int i = 0;i<t;i++)
	{
		x = n[i];
		int rev = 0;
		while(n[i]!=0)
		{
			r = n[i] % 10;
			rev = r + rev*10;
			n[i] = n[i]/10;
		}
		if(x==rev)
		cout<<"wins\n";
		else
		cout<<"losses\n";
		
	}
	return 0;
}
