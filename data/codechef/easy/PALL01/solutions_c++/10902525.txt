#include<cstdio>
using namespace std;
int main()
{
    int T,N,n,R;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&N);
        n=N;R=0;
        while (n>0)
        {
            R=(R*10)+n%10;
            n/=10;
        }
        if (R==N) printf("%s\n","wins");
        else printf("%s\n","losses");
    }
    return 0;
}
