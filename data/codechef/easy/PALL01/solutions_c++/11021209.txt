#include<iostream>
#include<math.h>

using namespace std;

int main ()
{
	int T,i,j,k,N,N1,q,p,N0,N2;

	//cout << "write T\n";
	cin >> T;

	for (i=0;i<T;i++)
	{
		//cout << "write N\n";
		cin >> N;
		N0 = N1 = N2 = N;
		
		q = 0;
		k = 0;		

		while (N0 > 0)
		{
			N0 /= 10;
			q++;
		}
		
		p = (q/2)-1;
		//cout << q << "\t"<< p << endl;
		for (j=0;j<=p;j++)
		{
			if ((N1/(int)pow(10,q-1-j)) != N2%10)
			{
				k++;
			}
			N1 %= (int)pow (10,q-1-j);
			N2 -= N2%10;
			N2 /= 10;
			//cout << N1 << "\t" << N2 << endl;
		}
		if (k == 0)
		{
			cout << "wins\n";
		}
		else 
		{
			cout << "losses\n";
		}
		//cout << k << endl;
	}

	return 0;
}
