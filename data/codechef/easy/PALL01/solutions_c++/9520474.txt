#include<iostream>
using namespace std;
int reve(int n)
{
    int rev=0;
    while(n)
    {
        rev=rev*(10)+n%10;
        n/=10;
    }
    return rev;
}
int main()
{
    int n;
    int t;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==reve(n))
        cout<<"wins\n";
        else
        cout<<"losses\n";
    }
    return 0;
}
