#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        int ans=0;
        int a=n;
        while(a>0)
        {
            ans=10*ans+a%10;
            a=a/10;
        }
        if(ans==n)
            cout << "wins" << endl;
        else
            cout << "losses" << endl;
    }
    return 0;
}
