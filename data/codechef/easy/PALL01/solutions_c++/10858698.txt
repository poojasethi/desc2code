#include<iostream>
using namespace std;
int main()
{
	int t,n,len,i,temp,flag;
	cin>>t;
	while(t--)
	{	cin>>n;len=0;temp=n;flag=1;
		while(n>0)
		{
			n=n/10;
			len++;
		}
		n=temp;
		int rev_num[len];
		i=0;
		while (n>0)
		{
			rev_num[i]=n%10;
			n=n/10;i++;
		}
		for(i=0;i<(len/2+1);i++)
		{	if(rev_num[i]!=rev_num[len-i-1])
				{flag=0;break;}
		}
		if(flag==1)
		   cout<<"wins"<<endl;
		else if(flag==0) cout<<"losses"<<endl;
	}
	return 0;
}
