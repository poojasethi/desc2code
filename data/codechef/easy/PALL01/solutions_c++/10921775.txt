#include<iostream>
#include<string>

using namespace std;

string compare(string a){
    for(int i=0; i<a.length()/2; i++){
        if(a[i]!=a[a.length()-i-1])
            return "losses";
    }
    return "wins";
}

int main(void){
    int t, n, k;
    string a;
    cin>>t;

}
