#include<stdio.h>
#include<stdlib.h>
#include<math.h>
 
int f(long long int n)
{
    long long int i,j=0;
    while(n)
    {
        i=n%10;
        n=n/10;
        j=j*10+i;
    }
    return j;
}
 
int main()
{
long long int n,l,r;
scanf("%lld",&n);
while(n)
{
	scanf("%lld",&l);
	r = f(l);
	if(r == l)
		printf("wins\n");
	else
		printf("losses\n");
	--n;
}
return 0;
}
