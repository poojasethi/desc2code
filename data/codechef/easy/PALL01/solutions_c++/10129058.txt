//palindrome
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        cin>>n;
        int palin=n;
        int rev=0;
        while(n!=0)
        {
            rev=(rev*10)+n%10;
            n/=10;
        }
        if(palin==rev)
        {
            cout<<"wins\n";
        }
        else
            cout<<"losses\n";
    }
    return 0;
}
