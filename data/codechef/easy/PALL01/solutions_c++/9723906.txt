#include<iostream>
#include<stdio.h>
#include<math.h>
void calc(int num){



     int n = num;
     int rev = 0;
     while (num > 0)
     {
          int dig = num % 10;
          rev = rev * 10 + dig;
          num = num / 10;
     }

    if(n == rev)
        printf("wins\n");
    else
        printf("losses\n");

}
int main(){
    int n,num;
    scanf("%d",&n);
    while(n--){
        scanf("%d",&num);
        calc(num);
    }
    return 0;
}
