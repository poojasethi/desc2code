#include<iostream>
using namespace std;
int main(){
	long numOfTestCase;
	cin>>numOfTestCase;
	
	for(long tc=0;tc<numOfTestCase;tc++){
		string input;
		cin>>input;
		
		long length = input.length();
		long i = 0,j = length-1, failed =0;
		
		while(i!=j){
			if(input[i]!=input[j]){
				failed = 1;
				cout<<"losses"<<endl;
				break;
			}
			i++;j--;
		}
		if(failed == 0)
		cout<<"wins"<<endl;
		
	}
	return 0;
}
