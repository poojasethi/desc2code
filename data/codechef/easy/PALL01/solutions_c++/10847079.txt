#include <bits/stdc++.h>
using namespace std;
string s;

int main() {
    int t; cin >> t;
    while (t--) {
        cin >> s;
        bool k = 1;
        for (int i = 0; i < s.size()/2; i++)
        if (s[i] != s[s.size()-i-1]) {
            k = 0;
            cout << "losses" << endl;
            break;
        }
        if (k) cout << "wins" << endl;
    }
}
