#include<bits/stdc++.h>
using namespace std;
int main() {
	int t;
	scanf("%d",&t);
    while(t--)
	{
	    int n, rev=0;
	    scanf("%d",&n);
	    int k = n;
	    while(k!=0) {
	        rev=(rev*10)+(k%10);
	        k=k/10;
	    }
	    if(rev==n) {
	        printf("wins\n");
	    } else {
	        printf("losses\n");
	    }
	}
	return 0;
}
