#include<cstdio>
using namespace std;

void merge(long data[],long f, long l, long mid)
{
	long start, k, i,b[1000000];
	start=f;
	k=mid+1;
	i=f;

	while(f<=mid && k<=l)
	{
		if(data[f]<=data[k])
		{
			b[i++]=data[f++];
		}
		else
		{
			b[i++]=data[k++];
		}

	}
		
	while(f<=mid)
	{
		b[i++]=data[f++];
	}

	while(k<=l)
	{
		b[i++]=data[k++];
	}
	
for(i=start;i<=l;i++)
{
	data[i]=b[i];
}

}

void mergesort(long data[],long first,long last)
{
	long mid;
	if(first>=last)
	{
		return;
	}

	mid=(first+last)/2;
	mergesort(data,first,mid);
	mergesort(data,mid+1,last);
	merge(data,first,last,mid);

}



int main()
{
	long a[1000000],t,i;
	
	scanf("%ld",&t);

	for(i=0;i<t;i++)
	{
		scanf("%ld",&a[i]);
	}

	mergesort(a,0,t-1);

	for(i=0;i<t;i++)
	{
		printf("%ld\n", a[i] );
	}
}
