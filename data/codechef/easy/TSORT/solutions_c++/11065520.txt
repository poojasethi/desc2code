#include<iostream>
#include<algorithm>

using namespace std;

int partition(int*A, int p, int r)
{
    int x = A[r];
    int j,i = p - 1;
    for (j=p; j<r; j++)
    {
        if (A[j]<=x)
        {
            i = i + 1;
            int temp=A[i];
            A[i]=A[j];
            A[j]=temp;
        }
    }
    int temp=A[i+1];
    A[i+1]=A[r];
    A[r]=temp;

    return i+1;
}

void q_sort(int*A, int p, int r)
{

       // qsort_comp++;
    if (p<r)
    {
        int q = partition(A, p, r);
        q_sort(A, p, q-1);
        q_sort(A, q+1, r);
    }
}
