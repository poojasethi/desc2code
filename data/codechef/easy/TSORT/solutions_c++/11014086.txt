/// basically it is counting sort because it is given that Aith <1000000
#include<bits/stdc++.h>
#include<stdio.h>
#include<string.h>
#define ML 1000001
int n,arr[ML];
int scani(){/// as given only +ve no need of sign check
	char c = getchar();
	while(c < '0' || c > '9') c = getchar();
	int ret = 0;
	while(c >= '0' && c <= '9')	ret = (ret<<3)+(ret<<1)+c-'0',c = getchar();
	return ret;
}
void printi(int no){/// 1234501000
	//if()
	if(no == 0){putchar_unlocked('0'); putchar_unlocked('\n'); return; }
	int l = 0;int rev = 0;
	while(no>0){
		l++;
		rev =rev*10 + no%10;
		no /= 10;
	}
	while(rev>0){ putchar_unlocked(rev%10+'0');l--; rev/=10;}
	while(l-- >0){	putchar_unlocked('0');}
	putchar('\n');
}
int main(){
    n = scani();
    for(int index =1; index <= n; index++)
        arr[index] = scani();///scanf("%d",arr+index);
    /// now sorting
    int crr[ML];
    for(int index =0; index < ML; index++)   crr[index] = 0;
    for(int index =1; index <= n; index++)  crr[(arr[index])]++; /// storing frequency
    for(int index =1; index < ML; index++)  crr[index] += crr[index-1];
    int brr[ML];
    for(int index =n; index> 0; index--)
        brr[(crr[(arr[index])])] = arr[index],crr[(arr[index])]--;
    for(int index =1; index <= n; index++)  printi(brr[index]);///printf("%d\n",brr[index]);
    return 0;
}
