#include <iostream>

using namespace std;

void mergeSort(int[], int, int);
void merge(int[], int, int, int);
int main()
{
    int t;
    cin>>t;
    int a[t];
    for (int i=0;i<t;i++)
        cin>>a[i];
    mergeSort(a, 0, t);
    for (int i=0;i<t;i++)
        cout<<a[i]<<" ";
    cout<<"\n";
    return 0;
}

void mergeSort(int a[], int beg, int end)
{
    int mid = (beg+end)/2;
    
    if (beg < end)
    {
        mergeSort(a, beg, mid);
        mergeSort(a, mid+1, end);
        merge(a,beg, mid, end);
    }
}

void merge(int a[], int beg, int mid, int end)
{
    int n1 = mid - beg + 1;
    int n2 = end - mid;
    
    int l[n1+1], r[n2+1];
    
    l[n1] = r[n2] = 738067713;
    
    for (int i=0; i<n1; i++)
        l[i] = a[beg + i];
        
    for (int i=0; i<n2; i++)
        r[i] = a[mid + i + 1];
        
    int i = 0, j = 0;
    
    for (int k = beg; k <= end; k++)
    {
        if (l[i] <= r[j])
            a[k] = l[i++];
        else
            a[k] = r[j++];
    }
}
