#include<iostream>
using namespace std;

void merge(int a[],int i,int j,int k)
{
    int ptr1=i;
    int ptr2=j+1;
    int temp[k+1-i];
    int s=0;
    while(ptr1<=j && ptr2<=k)
    {
        if(a[ptr1] < a[ptr2])
        {
            temp[s]=a[ptr1];
            s++;
            ptr1++;
        }
        else
        {
            temp[s]=a[ptr2];
            ptr2++;
            s++;
        }
    }
    while(ptr1<=j)
       temp[s++]=a[ptr1++];
    while(ptr2<=k)
        temp[s++]=a[ptr2++];
    for(int z=0;z<s;z++,i++)
        a[i]=temp[z];
    return;
}

void mergesort(int a[],int start,int end)
{
    if(start>=end)
        return;
    int mid= start+(end-start)/2;
    mergesort(a,start,mid);
    mergesort(a,mid+1,end);

    merge(a,start,mid,end);
}

int main()
{
    int n;
    cin>>n;
    int a[1000000];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    mergesort(a,0,n-1);
    for(int i=0;i<n;i++)
        cout<<a[i]<<endl;
}
