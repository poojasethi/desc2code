#include<iostream>
using namespace std;
int median(int a[],int d,int b,int c)
{
    if(a[d]<a[b])
    {
        if(a[d]>=a[c])
        return d;
        else if(a[b]<=a[c])
        return b;
    }
    else
    {
        if(a[d]<=a[c])
        return d;
        else if(a[b]>=a[c])
        return b;
    }
    return c;
}
void swaps(int &a,int &b)
{
    int temp=a;
    a=b;
    b=temp;
}
void quicksort(int a[],int low,int high)
{

    if(low<high)
    {    int pivot=median(a,low,high,(low+high)/2);
        swaps(a[pivot],a[high]);
        int i=low-1,j=high;
        while(i<j)
        {while(a[++i]<a[high] && i<=high);
        while(a[--j]>a[high] && j>=low);
        if(i<j)
        swaps(a[i],a[j]);
        }
        swaps(a[i],a[high]);
        quicksort(a,low,i-1);
        quicksort(a,i+1,high);
    }
    }
