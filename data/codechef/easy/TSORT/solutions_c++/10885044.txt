#include <iostream>
#define MAX 1000007

using namespace std;


int main() {
	// your code goes here
    int t, a[MAX], count=0;
    cin>>t; 
    for(int i=0; i<MAX; i++){
        a[i] = 0;
    }
    while(t--){
        int b;
        cin>>b;
        a[b]++;
    }
    for(int i=0; i<MAX && count != t; i++){
        while(a[i]>0){
            count++;
            cout<<i<<"\n";
            a[i]--;
        }
    }
	return 0;
}
