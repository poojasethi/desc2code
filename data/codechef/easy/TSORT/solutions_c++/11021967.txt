#include<bits/stdc++.h>
using namespace std;

int cmp(const void*a,const void*b){

    int l = *(const int*)a;
    int r = *(const int*)b;

    if(l<r)
        return -1;
    else return 1;

}

int main(){

    int n;
    cin>>n;
    int arr[n],i;
    for(i=0;i<n;i++)
        cin>>arr[i];
    qsort(arr,n,sizeof(int),cmp);
    for(i=0;i<n;i++)
        cout<<arr[i]<<endl;

}
