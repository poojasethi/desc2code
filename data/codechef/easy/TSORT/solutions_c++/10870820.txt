#include<iostream>
using namespace std;
int main()
{
    int num,t,i;
    int a[1000001]={0};
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>num;
        a[num]++;
    }
    for(i=0;i<1000001;i++)
      while(a[i]>0)
        {
            cout<<i<<"\n";
            a[i]--;
        }
    return 0;
}
