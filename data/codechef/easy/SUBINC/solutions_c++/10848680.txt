#include<bits/stdc++.h>

using namespace std;

typedef long long int ll;

ll pe,ce,max_good_suffix,n;
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		ll answer=0;
		cin>>n;
		for(ll i=0;i<n;i++)
		{
			cin>>ce;
			if(i==0)
				max_good_suffix = 1;
			else
				if(ce >= pe)
					max_good_suffix++;
				else
					max_good_suffix = 1;
			answer += max_good_suffix;
			pe = ce;
		}
		cout<<answer<<endl;
	}
	return 0;
}
