#include<iostream>
#include<stdio.h>

using namespace std;

#define ll long long int

int main()
{
	ll t,n,i,count,sum;

	scanf("%lld",&t);

	while(t--)
	{
		scanf("%lld",&n);

		ll arr[n];
		count=1;
		sum=0;

		for(i=0;i<n;i++)
			scanf("%lld",&arr[i]);

		for(i=0;i<n-1;i++)
		{
			if(arr[i]<=arr[i+1])
			{
				count++;
			}
			else
			{
				if(count%2==0)
					sum+=((count+1)*(count/2));
				else
					sum+=(((count+1)/2)*count);
				count=1;
			}
		}
		if(count%2==0)
			sum+=((count+1)*(count/2));
		else
			sum+=(((count+1)/2)*count);

		printf("%lld\n",sum);

	}

	return 0;
}
