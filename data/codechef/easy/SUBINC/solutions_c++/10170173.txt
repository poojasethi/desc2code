#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n,c=0,j,i;
        cin>>n;
        long long a[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        for( i=0;i<n;)
        {
            for( j=1;i+j<n;j++)
            {
                if(a[i+j]>=a[i+j-1])
                    c=c+j;
                else break;
            }
            if(i+j<n)
                i=i+j;
            else break;

        }
        cout<<c+n<<endl;
    }
}
