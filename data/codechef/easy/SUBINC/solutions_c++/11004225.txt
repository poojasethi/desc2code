#include <iostream>
#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main()
{
	ll t,n,ele;
	scanf("%lld",&t);
	while(t--)
	{
		ll currSum=0;
		scanf("%lld",&n);
		vector<ll> elements;
		for(ll i=0;i<n;i++)
		{
			scanf("%lld",&ele);
			elements.push_back(ele);
		}
		vector<ll> dp;
		dp.push_back(1);
		currSum+=1;
		for(ll i=1;i<n;i++)
		{
			if(elements[i]>=elements[i-1])
			{
				dp.push_back(1+dp[i-1]);
				currSum+=(1+dp[i-1]);
			}
			else
			{
				dp.push_back(1);
				currSum+=1;
			}
		}
		dp.clear();
		elements.clear();
		printf("%lld\n", currSum);
	}
	return 0;
}
