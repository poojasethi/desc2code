#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		long int n,i;
		cin>>n;
		long long int a[n],b[n],count=0;
		for(i=0;i<n;i++){
		cin>>a[i];
		b[i]=1;
	}
		for(i=n-2;i>=0;i--){
			if(a[i]<=a[i+1])
			b[i]=b[i+1]+1;
		}
		for(i=0;i<n;i++)
		count+=b[i];
		cout<<count<<endl;
	}
	return 0;
}
