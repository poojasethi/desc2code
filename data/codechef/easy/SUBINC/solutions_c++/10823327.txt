#include<iostream>
using namespace std;


int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		int N;
		cin >> N;
		int a[N+1], b[N+1];

		for(int i=0;i<N;i++)
		{
			cin >> a[i];

		}
		b[0] = 1;
		for(int i=1;i<N;i++)
		{
			if(a[i-1]<=a[i])
				b[i] = b[i-1]+1;
			else
				b[i] = 1;
		
		}
		long long sum=0;
		for(int i=0;i<N;i++)
			sum+=b[i];

		cout << sum << endl;
	
	}

}
