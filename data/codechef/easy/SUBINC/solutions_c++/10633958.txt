#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t,n,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long long arr[n];
        for(i=0;i<n;i++)
            cin>>arr[i];
            long long co=0;
            long long sum=0;
        for(i=0;i<n-1;i++)
        {
            if(arr[i]<=arr[i+1])
                co++;
            else
            {
                sum=sum+((co)*(co+1))/2;
                co=0;
            }
        }
        sum=sum+((co)*(co+1))/2;
        cout<<sum+n<<"\n";
    }
}
