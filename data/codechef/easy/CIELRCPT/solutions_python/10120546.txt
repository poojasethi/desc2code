def fun(p):
    n=0
    global ans
    if p==1 or p==0:
        ans+=p
        return ans
    else:
        while(p-2**n)>=(2**n):
            n = n + 1
            if (p-2**n<2**n) and (p!=0):
                ans = ans + 1
                fun(p-2**n)
    return ans
t=input()
p=[]
for j in range(t):
    p.append(input())
for i in p:
    ans=0
    while i/2048>0:
        ans+=1
        i=i-2048
    print fun(i)
