#include<iostream>

int main()
{
	double c = 0.0;
	int testcases;
	std::cin >> testcases;
	int *array = new int[testcases];
	int x;
	int cupcakes;


	for (x = 1; x <= testcases; x++)
	{
		std::cin >> cupcakes;

		if (cupcakes == 2)
		{
			c = 2;
		}

		if (cupcakes % 2 != 0)
		{
			c = (cupcakes / 2) + 1;
		}
		else
		{
			c = (cupcakes / 2) + 1;

		}


		array[x] =c ;

	}

	for (int y = 1; y <= testcases; y++)
	{
		std::cout << array[y] << std::endl;
	}










	return 0;
}
