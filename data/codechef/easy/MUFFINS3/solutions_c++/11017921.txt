#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    long *a=new long[t];
    for(int i=0;i<t;i++)
    {
            cin>>a[i];
    }
    for(int i=0;i<t;i++)
    {
        if(a[i]>2)
            cout<<a[i]/2+1<<endl;
        else
            cout<<a[i]<<endl;
    }
    delete a;
    return 0;
}
