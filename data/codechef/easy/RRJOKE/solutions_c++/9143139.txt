#include <stdio.h>
int main()
{
    int n;
    scanf("%d",&n);

    while(n--)
    {
        int a,ans=1;
        scanf("%d",&a);
        for(int i=0; i<a; i++)
        {
            int c,d;
            scanf("%d%d",&c,&d);
            if(i!=0)
                ans = ans^(i+1);
        }
        printf("\n%d",ans);
    }

    return 0;
}
