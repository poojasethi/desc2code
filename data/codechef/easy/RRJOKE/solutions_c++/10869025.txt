#include <iostream>

using namespace std;

int main()
{
        int t;
        cin>>t;
        while(t--){
                int num, x, y, point;
                cin>>num;

                for(int i=0; i<num; i++){
                        cin>>x>>y;
                }
                point=1;
                for(int i=2; i<=num; i++){
                        point^=i;
                }
                cout<<point<<endl;

        }
    return 0;
}
