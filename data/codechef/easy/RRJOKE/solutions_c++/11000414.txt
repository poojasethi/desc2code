#include <iostream>
using namespace std;
 
int main() {
	int t;
	cin>>t;
	while(t)
	{
	    int n, a, b, cnt, res=1;
	    cin >> n;
	    cnt = n;
	    while(cnt--)
	    {
	        cin >> a >> b;
	    }
	    
	    for(int i=2; i<=n; i++)
	    {
	        res= res ^ i;
	    }
	    cout << res <<endl;
    	t--;
	}
	return 0;
}
