#include<bits/stdc++.h>
using namespace std;
int min(int a,int b) {
    return (a<b)?a:b;
}
int dist(int ax,int ay, int bx, int by) {
    return min((abs(bx-ax)),(abs(by-ay)));
}
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    int n,x,y,ans=0;
	    scanf("%d",&n);
	    for(int i=1;i<=n;i++) {
	        scanf("%d %d",&x,&y);
	        ans=ans^i;
	    }
	    printf("%d\n",ans);
	}
	return 0;
}
