#include<bits/stdc++.h>
using namespace std;
int main()
{
        int n,t,x,y,i,ans=0;
                scanf("%d",&t);
        while(t--)
        {

        ans=0;
        for(i=1;i<=n;i++)
        {
        scanf("%d %d",&x,&y);
        ans^=i;
        }
        printf("%d\n",ans);
        }

}
