#include <bits/stdc++.h>
using namespace std;
 
int main() {	
	int t,n,ans,i,a,b;
	scanf("%d",&t);
	while(t--)
	{    scanf("%d",&n);
	    for(i=0;i<n;i++)
	    {
	        scanf("%d %d",&a,&b);
	    }
	    ans=1;	
	    for(i=2;i<=n;i++)
	    {
	        ans=ans^i;
	
	    }
	    printf("%d\n",ans);	}
	return 0;
}
