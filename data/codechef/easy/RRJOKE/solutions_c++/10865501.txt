#include <iostream>
using namespace std;
#define ll long long

int main()
{   ll t;
    scanf("%lld",&t);
    while(t--)
    {   ll n,a,b,x=0,i;
        scanf("%lld",&n);
        for(i=1;i<=n;i++)
        {   scanf("%lld%lld",&a,&b);
            x^=i;
        }
        printf("%lld\n",x);
    }
    return 0;
}
