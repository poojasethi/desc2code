#include<iostream>
using namespace std;
int main()
{
    int ips,range;
    cin>>ips;
    while(ips--)
    {
        range=3;
        int data[range];
        int min1=999999999,min2=999999999;
        for(int i=0;i<range;i++)
        {
            cin>>data[i];
            if(data[i]<min1)
            {
                min2=min1;
                min1=data[i];
            }
            else if(data[i]<min2)
                min2=data[i];
        }
        cout<<min2<<endl;
    }
}
