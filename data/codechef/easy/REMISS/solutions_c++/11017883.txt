#include <iostream>

using namespace std;

int main()
{
   long t;
   cin>>t;
   long *a,*b;
   a=new long[t];
   b=new long[t];
   for(int i=0;i<t;i++)
   {
       cin>>a[i]>>b[i];
   }
   for(int i=0;i<t;i++)
   {
       if(a[i]>b[i])
            cout<<a[i]<<"  ";
       else
            cout<<b[i]<<"  ";
       cout<<a[i]+b[i]<<endl;
   }
   delete a;
   delete b;
   return 0;
}
