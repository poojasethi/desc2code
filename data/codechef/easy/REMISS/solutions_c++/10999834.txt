#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int a,b;
        cin>>a;
        cin>>b;
        int m;
        if(a>b)
        {
            m=a;
        }
        else
        {
            m=b;
        }
        cout<<m<<" "<<a+b<<endl;
    }
    return 0;
}
