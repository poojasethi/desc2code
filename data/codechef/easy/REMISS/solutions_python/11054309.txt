testcases = raw_input()

def maxmincommon(num1, num2):
	mintimes = max(num1, num2)
	maxtimes = num1 + num2
	return str(mintimes) + " " + str(maxtimes)
 
for i in range(int(testcases)):
	rawnums = raw_input()
	nums = rawnums.split()
	num1 = int(nums[0])
	num2 = int(nums[1])
	print maxmincommon(num1, num2)
