import sys

data = sys.stdin.readlines()
t = int(data.pop(0))
output = []

for i in range(t):
    num = map(int, data.pop(0).split(' '))
    a = num[0]
    b = num[1]
    result = []
    result.append(max(a,b))
    result.append(a+b)
    output.append(' '.join(map(str,result)))

for k in output:
    print k
