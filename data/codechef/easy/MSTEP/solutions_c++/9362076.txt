#include <iostream>
#include<vector>

using namespace std;

int main()
{
    vector<int> x_coord,y_coord;

    int t,n,n_2,i,j,val,diff_x,diff_y;
    scanf("%d",&t);
    while(t>0)
    {
        scanf("%d",&n);
        n_2=n*n;
        x_coord.clear();
        x_coord.reserve(n_2);
        y_coord.clear();
        y_coord.reserve(n_2);

        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                scanf("%d",&val);
                x_coord[val-1]=i;
                y_coord[val-1]=j;



            }
        }
        val=0;
        for(i=0;i<n_2-1;i++)
        {
            diff_x=x_coord[i]-x_coord[i+1];
            if(diff_x<0){diff_x=0-diff_x;}
            val=val+diff_x;

          diff_y=y_coord[i]-y_coord[i+1];
          if(diff_y<0){diff_y=0-diff_y;}
          val=val+diff_y;

        }

        printf("%d\n",val);

        t--;
    }
    return 0;
}
