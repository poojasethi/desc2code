#include<iostream>
#include<stdlib.h>

using namespace std;
int main()
{int t;
cin>>t;
while(t--)
{
    int n,l;
    int x[250001],y[250001];
    cin>>n;
    int a[n][n];
    for(int i=0;i<n;i++)
    for(int j=0;j<n;j++)
    {cin>>a[i][j];
    l=a[i][j];
    x[l]=i;
    y[l]=j;

    }
    int steps=0;
    for(l=1;l<n*n;l++)
    {
        steps=steps+abs(x[l]-x[l+1])+abs(y[l]-y[l+1]);

    }

    cout<<steps<<endl;
 }   return 0;
}
