#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	    long long ans=0;
		int n;
		cin>>n;
		int a[n][n];
		class matrix
		{
		    public:
		    int i;
		    int j;
		    matrix()
		    {
		        i=j=0;
		    }
		}m[n*n+1];
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
            {
                cin>>a[i][j];
                m[a[i][j]].i=i;
                m[a[i][j]].j=j;
            }
		}
		for(int i=2;i<=n*n;i++)
        {
            ans+=abs(m[i].i-m[i-1].i)+abs(m[i].j-m[i-1].j);
        }
		cout<<ans<<"\n";
	}
}
