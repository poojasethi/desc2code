#include <bits/stdc++.h>
using namespace std;

int main()
{
#ifdef gsdt
    freopen("input.txt","r",stdin);
#endif // gsdt

    int T;
    scanf("%d",&T);
    struct cell
    {
        int x,y;
    };
    cell ls[500*500+5];
    for(int q=0; q<T; q++)
    {
        int n, ans=0; scanf("%d",&n);
        for(int i=1; i<=n; i++)
            for(int j=1; j<=n; j++)
        {
            int val;
            scanf("%d",&val);
            ls[val].x=i;
            ls[val].y=j;
        }
        for(int i=2; i<=n*n; i++)
        {
            ans+=abs(ls[i].x-ls[i-1].x);
            ans+=abs(ls[i].y-ls[i-1].y);
            //cout<<ans<<endl;
        }
        printf("%d\n",ans);
    }
    return 0;
}
