#include<bits/stdc++.h>
using namespace std;
int main()
{

    struct node
    {
        int row,column;

    };
    int t;
    scanf("%d",&t);
    while(t-->=1)
    {

        int n,minstep=0,r,c;
        scanf("%d",&n);

        node *arr1=new node[n*n];
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                int x;
                scanf("%d",&x);
                arr1[x-1].row=i;
                arr1[x-1].column=j;
            }
        }
        r=arr1[0].row;
        c=arr1[0].column;
        for(int i=1;i<n*n;i++)
        {
            minstep=minstep+(abs(arr1[i].row-r)+abs(arr1[i].column-c));
            r=arr1[i].row;
            c=arr1[i].column;
        }

        printf("%d\n",minstep);

    }
    return 0;
}
