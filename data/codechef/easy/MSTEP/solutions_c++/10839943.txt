#include <bits/stdc++.h>
using namespace std;
int main(){
      int t;
	  cin>>t;
	  while(t--){
	  	int n,i,j,num,count=0;
	  	cin>>n;
	  	int a[n][n],b[n*n],c[n*n];
	  	for(i=0;i<n;i++){
	  		for(j=0;j<n;j++){
	  			cin>>num;
	  			a[i][j]=num;
	  			b[num-1]=i;
	  			c[num-1]=j;
			  }
		  }
		  for(i=1;i<(n*n);i++){
		  	count+=abs(b[i]-b[i-1])+abs(c[i]-c[i-1]);
		  }
		  cout<<count<<endl;
	  }
	  return 0;
}
