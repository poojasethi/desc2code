#include<iostream>
using namespace std;
int abs(int x,int y)
{
	int diff=x-y;
	if(diff>=0)
		return(diff);
	else
		return((-1)*diff);
}
int main()
{
	int k=0,t;
	cin>>t;
	while(t--)
	{
		int n,temp,ans=0;
		cin>>n;
		int a[n*n+1];
		
		for(int i=0;i<n*n;i++)
		{
			cin>>temp;
			a[temp]=i;
		}
		for(int i=1;i<n*n;i++)
		{
			ans+=abs(a[i]/n,a[i+1]/n)+abs(a[i]%n,a[i+1]%n);
		}
		cout<<ans<<endl;
	}
}
