#include <iostream>
#include <bits/stdc++.h>

using namespace std;

pair<int,int>  arr[250009];

int main()
{
    int t;
    cin >> t;
    while(t--){
        int n,x;
        set<int> myset;
        cin >> n;
        for(int i=0; i<n; i++)
            for(int j=0; j<n; j++){
                cin >> x;
                arr[x].first = i; arr[x].second = j;
                myset.insert(x);
            }
        int ans=0;
        for(int i=1; i<n*n; i++){
            ans += abs(arr[i+1].second-arr[i].second) + abs(arr[i+1].first-arr[i].first);
        }
        cout << ans << endl;
    }
    return 0;
}
