#include<iostream>
#include<cstdlib>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long step=0;
        int l[250001], m[250001], n;
        cin>>n;
        for(int i=1; i<=n; i++)
        {
            for(int j=1; j<=n; j++)
            {
                int v;
                cin>>v;
                l[v]=i;
                m[v]=j;
            }
        }
        for(int k=1; k<n*n; k++)
        {
            step+=abs(l[k+1]-l[k]);
            step+=abs(m[k+1]-m[k]);
        }
        cout<<step<<"\n";
    }
}
