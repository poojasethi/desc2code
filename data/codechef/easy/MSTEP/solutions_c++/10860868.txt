#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int main()
{int t,j;
cin>>t;
while(t--)
{int n,p,s=0;
cin>>n;
int b[n*n],c[n*n];
for(int i=0;i<n;i++)
for(int j=0;j<n;j++)
{scanf("%d",&p);
b[p-1]=i;c[p-1]=j;
}
for(int i=0;i<n*n-1;i++)
{s=s+abs(b[i+1]-b[i])+abs(c[i+1]-c[i]);
}
cout<<s<<endl;
}
}
