#include <iostream>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    do
    {
        cin>>n;
        int win[n+1];
        int j=0,i;
        char ans[n],cans[n];
        for(i=0;i<n;i++)
        {
            cin>>cans[i];
        }
        for(i=0;i<n;i++)
        {
            cin>>ans[i];
        }
        for(i=0;i<=n;i++)
        {
            cin>>win[i];
        }
        for(i=0;i<n;i++)
        {
            if(cans[i]==ans[i])
                j++;
        }
        if(j==n)
            cout<<win[n]<<"\n";
        else
        {
            int large=win[0];
            for(i=1;i<=j;i++)
            {
                if(large<win[i])
                    large=win[i];
            }
            cout<<large<<"\n";
        }
        t--;
    }
    while(t>0);
    return 0;
}
