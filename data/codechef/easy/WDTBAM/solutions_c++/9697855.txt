#include<iostream>
using namespace std;

int main()
{
    int t;
    char ans_corr[1001],ans_chef[1001];
    long int w[1001];
    cin>>t;
    while(t--)
    {
        int n,ans,counter=0;
        cin>>n;

        cin>>ans_corr;
        cin>>ans_chef;

        for(int i=0;i<=n;i++)
            cin>>w[i];
        for(int i=0;i<n;i++)
        {
            if(ans_corr[i]==ans_chef[i])
                counter++;
        }
        if(counter==0)
            cout<<w[0]<<"\n";
        else if(counter==n)
        {
            cout<<w[n]<<"\n";
        }
        else
        {
            int maximum=-1;
            for(int i=0;i<=counter;i++)
            {
                if(w[i]>maximum)
                    maximum=w[i];
            }
            cout<<maximum<<"\n";
        }
    }
    return 0;
}
