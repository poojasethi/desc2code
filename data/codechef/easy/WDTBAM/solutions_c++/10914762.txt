#include <iostream>
#include<string>

using namespace std;

int main()
{
        int t, n, cont;
        int arr[1001];
        string str1, str2;

        cin>>t;
        while(t--){
                cin>>n;
                cin>>str1>>str2;

                for(int i=0; i<=n; i++){
                        cin>>arr[i];
                }
                cont=0;
                int high=0;

                for(int i=0; i<str1.size(); i++){
                        if(str1[i]==str2[i]) cont++;
                }

                if(cont==n) cout<<arr[n]<<endl;
                else{
                        for(int i=0; i<=cont; i++){
                                high = max(high, arr[i]);
                        }

                        cout<<high<<endl;
                        //cout<<arr[cont]<<endl;
                }

        }

    return 0;
}
