#include<iostream>
using namespace std;
int main()
{
   int t,n,i,j,c;
  long int w[1010],max;
  string s1;
       string s2;
   cin>>t;
   while(t--)
   {
       c=0;

       cin>>n>>s1>>s2;
       for(i=0;i<=n;i++)
        cin>>w[i];
       i=j=0;
       while(i<n)
       {
           if(s1[i]==s2[j])
            c++;
           i++;j++;
       }

        max=w[0];
       if(c==n)
        max=w[n];
       else
        for(int k=0;k<=c;k++)
        if(max<w[k])
           max=w[k];
       cout<<max<<"\n";

   }

}
