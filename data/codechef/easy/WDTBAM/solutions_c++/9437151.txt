#include<cstdio>
#include<iostream>

using namespace std;

int main()
{
    int t,i,n,counter[1002],j,correct,incorrect,winnings;
    char a[1001],b[1001];
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n;
        cin>>a;
        cin>>b;
        for(j=0;j<=n;j++)
        {
            cin>>counter[j];
        }
        correct=0;incorrect=0;
        for(j=0;j<n;j++)
        {
            if(a[j]==b[j])
            {
                correct++;
            }
            else
            {
                incorrect=1;
            }
        }
        if(incorrect==0)
        {
            cout<<counter[n]<<"\n";
        }
        else
        {
            winnings=counter[0];
            for(j=1;j<=correct;j++)
            {
                if(counter[j]>winnings)
                {
                    winnings=counter[j];
                }
            }
            cout<<winnings<<"\n";
        }
    }
    return 0;
}
