#include<bits/stdc++.h>

using namespace std;

#define pb push_back
int cequal(string &s1,string &s2)
{
    int temp=0;
    for(int i=0;i<s1.length();i++)
    {
        if(s1[i]==s2[i])
        {
           temp++;
        }
    }
    return temp;
}

int main(void)

{
    int t; scanf("%d",&t);
    while(t--)
    {
        int n; scanf("%d",&n);
        string correct,chef;
        cin>>correct;
        cin>>chef;
        
        vector<int> w;
        for(int i=0;i<=n;i++)
        {
            int temp; scanf("%d",&temp);
            w.pb(temp);
        }
        int ans;
        if(cequal(correct,chef)==0)
        {
           ans=w[0];
        }
       else if(cequal(correct,chef)==correct.length())
       {
           ans=w[n];
       }
       else
       {
           int qmax=cequal(correct,chef);
           
           int indmax=0;
           for(int i=0;i<=qmax;i++)
           {
              if(w[i]>=w[indmax])
              {
                indmax=i;
              }
           }
        ans=w[indmax];
       }
       printf("%d\n",ans);
   }
   
   
   return 0;
}
