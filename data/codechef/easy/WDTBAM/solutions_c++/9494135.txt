#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;scanf("%d",&t);
    while(t--)
    {
        int n,m;long long ans;scanf("%d",&n);
        char a[n],b[n];
        scanf("%s",a);
        scanf("%s",b);
        long long q[n+1];
        for(int i=0;i<=n;i++)
            scanf("%lld",&q[i]);
        int c=0;
        for(int i=0;i<n;i++)
        {
            if(a[i]==b[i])
                c++;
        }
        if(c==0)
            ans=q[0];
            else if(c==n)
                ans=q[n];
        else{
                 m=q[c];
        for(int i=c-1;i>=0;i--)
        {
            if(m<q[i])
            m=q[i];
        }
        ans=m;
    }
    printf("%lld\n",ans);
}
}
