#include<iostream>
using namespace std;
int main()
{
    int ips,count,data;
    cin>>ips;
    while(ips--)
    {
        count=0;
        cin>>data;
        if(data>=100)
        {
            count+=data/100;
            data=data%100;
        }
        if(data>=50)
        {
            count+=data/50;
            data=data%50;
        }
        if(data>=10)
        {
            count+=data/10;
            data=data%10;
        }
        if(data>=5)
        {
            count+=data/5;
            data=data%5;
        }
        if(data>=2)
        {
            count+=data/2;
            data=data%2;
        }
        if(data==1)
        {
            count+=1;
            data--;
        }
        cout<<count<<endl;
    }
}
