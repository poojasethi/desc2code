#include<iostream>
using namespace std;
int main()
{
    int k,t,m,x,y,h[10],i,l[2][10],cnt,j,a,temp;
    cin>>t;
    while(t--)
    {
        cin>>m>>x>>y;
        a=x*y;
        cnt=0;
        for(i=0;i<m;i++)
        {
            cin>>h[i];
            l[0][i]=h[i]-a;
            if(l[0][i]<1)
                 l[0][i]=1;
            l[1][i]=h[i]+a;
            if(l[1][i]>100)
                 l[1][i]=100;
        }
        k=0;
        while(k<2)
        {
            for(i=0;i<m-1;i++)
            {
                for(j=i;j<m;j++)
                {
                    if(l[k][i]>l[k][j])
                    {
                        temp=l[k][i];
                        l[k][i]=l[k][j];
                        l[k][j]=temp;
                    }
                }
            }
            k++;
        }
        for(i=0;i<m-1;i++)
        {
            if((l[0][i+1]-l[1][i])>0)
                cnt=cnt+l[0][i+1]-l[1][i]-1;
        }
        if(l[0][0]!=1)
        cnt=cnt+l[0][0]-1;
        if(l[1][m-1]!=100)
            cnt=cnt+100-l[1][m-1];
        cout<<cnt<<endl;
    }
    return 0;
}
