no_of_tests = int(raw_input())

for i in range(no_of_tests):
    n, k = map(int, raw_input().split())
    int_arr = map(float, raw_input().split())
    int_arr.sort()

    j = k 
    end_j = n - k 
    total = 0 

    while j < end_j:
        total += int_arr[j]
        j +=1 

    print total/(n - 2*k)
