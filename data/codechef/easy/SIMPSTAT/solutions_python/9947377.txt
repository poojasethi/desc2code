testcases = int(raw_input())

for i in range(0, testcases):
    metadataarray = [int(x) for x in raw_input().split()]
    dataarray = [int(x) for x in raw_input().split()]
    dataarray.sort()
    summ = 0
    if metadataarray[1] == 0:
        for number in dataarray:
            summ = summ + number
    else:
        for number in dataarray[metadataarray[1]:-metadataarray[1]]:
            summ = summ + number
    divisor = float(metadataarray[0] - 2*metadataarray[1])
    print "%.6f" % (summ/divisor)
