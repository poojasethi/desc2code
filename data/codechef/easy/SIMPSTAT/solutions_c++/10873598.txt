#include<stdio.h>
#include<stdlib.h>

int cmpfunc(const void *a, const void *b){
	return(*(int *)a - *(int *)b);
}

int main(){
	int t, i, n, k;; scanf("%d", &t);
	while(t-- && scanf("%d%d", &n, &k)){
		int a[n]; double s = 0;
		for(i = 0; i < n; scanf("%d", &a[i++]));
		qsort(a, n, sizeof(int), cmpfunc);
		for(i = k; i < n - k; s += (a[i++]));
		printf("%lf\n", s / (n - (2 * k)));
	}
	return 0;
}
