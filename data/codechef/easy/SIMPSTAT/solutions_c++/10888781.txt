#include <iostream>
#include<algorithm>
#include<iomanip>
using namespace std;

int main() {
	int t,n,k,*a,i;
	double sum;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		a=new int [n];
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		sum=0;
		for(i=k;i<n-k;i++){
		sum+=a[i];}
		cout<<fixed<<setprecision(6)<<sum/(n-2*k)<<endl;
		delete []a;
	}	
	return 0;
}
