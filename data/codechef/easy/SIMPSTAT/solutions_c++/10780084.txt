#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int* func(int);
void swap(int*, int*);
int partition(int*,int,int);
void quicksort(int*,int,int);
void swap(int* x,int* y){
	int test;
	test = *x;
	*x   = *y;
	*y   = test;
}
int partition(int* array,int low, int high){
	int i = low;
	int val = array[high];
	for(int j=low;j<high;j++){
		if(array[j]<val){
			swap(&array[j],&array[i]);
			i++;
		}	
	}
	swap(&array[high],&array[i]);
	return i;
}

void quicksort(int* array, int low,int high){
	if(low<high){	
		int part = partition( array, low, high);
		quicksort( array, low,part-1);
		quicksort(array, part+1,high);
	}
}

int* func(int a){
	int * array;
	array = new int [100000];
	for (int i = 0; i<a;i++){
		cin>>array[i];
	}
	return array;
}
void stats(){
	int* array;
	int N,M;
	double value = 0;
	cin>>N>>M;
	array = func(N);
	quicksort(array,0,N-1);
	for(int i = M;i<N-M;i++){
		value += array[i];
	}
	value = value/(N-2*M);
	cout<<setprecision(6)<<fixed<<value<<'\n';
}
int main(){
	int cases;
	cin>>cases;
	while(cases){
		stats();
		cases--;
	}	
	
}
