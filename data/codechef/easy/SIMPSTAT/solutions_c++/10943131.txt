#include <iostream>
#include<algorithm>
#include<iomanip>
using namespace std;

int main() 
{
    int t,i,n,k;
    long sum;
    double avg;
	cin>>t;
	while(t--)
	{
	    cin>>n;
	    cin>>k;
	    avg=sum=i=0;
	    int a[n];
	    for(i=0;i<n;i++)
	     cin>>a[i];
	    sort(a,a+n);
	    for(i=k;i<n-k;i++)
	     sum+=a[i];
	    avg=(double)sum/(n-2*k);
	    cout<<fixed<<setprecision(6)<<avg<<endl;
	}
	return 0;
}
