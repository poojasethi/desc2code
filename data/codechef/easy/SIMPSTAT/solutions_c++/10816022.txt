#include<bits/stdc++.h>
#include <iostream>
#include <algorithm>
using namespace std;
int main()
{ ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout << setprecision(6) << fixed;
int t=0,no=0,rem=0,k=0,in[100000],j=0;
cin>>t;
while(t>0)
{double avg=0,tot=0;
cin>>no>>rem;

for(k=0;k<no;k++)
{
cin>>in[k];
}
sort(in,in+no);
for(j=rem;j<(no-rem);j++)
{
tot=tot+in[j];
}

avg=tot/(no-(2*rem));
cout<<avg<<"\n";
tot=0;
t--;
}
return 0;
}
