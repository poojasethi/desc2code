#include<iostream>
#include<algorithm>
#include<iomanip>
using namespace std;
int main()
{
    long long  t;
    cin>>t;
    while(t--)
    {
        double sum=0.00;
        long long  n,k;
        cin>>n>>k;
        long long  a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
            sum+=a[i];
        }
        sort(a,a+n);
        if(k==0)
        {
            sum=sum;
        }
        else if(k>0)
        {
            for(int i=1;i<=k;i++)
            {
                sum=sum-a[i-1]-a[n-i];
            }
        }

        cout<<setprecision(6)<<fixed;
        cout<<sum/(n-(2*k))<<endl;
    }
}
