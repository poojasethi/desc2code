#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		long int i,n,k,sum=0;
		float r=0.0;
		cin>>n>>k;
		long long int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		for(i=k;i<n-k;i++)
		sum=sum+a[i];
		r=(float)sum/(n-(2*k));
		printf("%.6f",r);
		cout<<endl;
	}
	return 0;
}
