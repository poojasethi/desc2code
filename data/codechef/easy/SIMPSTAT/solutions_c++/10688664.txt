#include<stdio.h>
#include<algorithm>
#include <iostream>
using namespace std;

int main() {
long long int t,n,k,i,j,a[10000];
double s=0;
cin>>t;
while(t--)
{
cin>>n>>k;
for(i=0;i<n;i++)
	cin>>a[i];
	
sort(a,a+n);

if(k>0)
{
j=n-k;
}
else
{j=n;}

for(i=k;i<j;i++)
{
s+=a[i];
}





printf("%lf\n",s/(n-(2*k)));

s=0;

}







	return 0;
}
