#include <iostream>
#include<algorithm>
#include<iomanip>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	 {
	     int n,k;
	     cin>>n>>k;
	     int a[n];
	     
	     for(int i=0;i<n;i++)
	      {
	          cin>>a[i];
	      }
	      
	      
	      sort(a,a+n);
	      int i;
	      long long int sum=0;
	      for(i=k;i<n-k;i++)
	       {
	           sum+=a[i];
	       }
	      
	      
	      double avg=(double)sum/(n-2*k);
	      cout<<fixed<<setprecision(6);
	      
	      cout<<avg<<endl;
	      
	      
	 }
	
	
	return 0;
}
