#include<bits/stdc++.h>
#include <iostream>
#include <algorithm>
using namespace std;
int main()
{ ios_base::sync_with_stdio(0);
	cin.tie(0);
	cout << setprecision(6) << fixed;
int t=0,no,rem,in[100000],j=0,i=0,k;
cin>>t;

for(i=0;i<t;i++)
{
cin>>no>>rem;
for(k=0;k<no;k++)
{
cin>>in[k];
}
sort(in,in+no);
float tot=0;
for(j=rem;j<(no-rem);j++)
{
tot=tot+in[j];
}

cout<<tot/(no-(2*rem))<<"\n";
}
return 0;
}
