#include <iostream>

using namespace std;

int qs_util(int arr[], int low, int high){
        int i= low-1;

        for(int j=low; j<high; j++){

                if(arr[j]<=arr[high]){
                        i++;
                        swap(arr[i], arr[j]);
                }
        }

        swap(arr[i+1], arr[high]);

        return i+1;
}

void qs(int arr[], int low, int high){

        if(low<high){
                int pi = qs_util(arr, low, high);
                qs(arr, low, pi-1);
                qs(arr, pi+1, high);
        }

}

int main()
{
        int t, arr[10000], n, k;
        cin>>t;
        while(t--){
                cin>>n>>k;

                for(int i=0; i<n; i++){
                        cin>>arr[i];
                }
                qs(arr, 0, n-1);

                int sum=0;
                for(int i=k; i<n-k; i++){
                        sum+=arr[i];
                }

                printf("%.6lf\n", (double)sum/(n-2*k));
                /*for(int i=0; i<n; i++){
                        cout<<arr[i]<<" ";
                }*/

        }
    return 0;
}
