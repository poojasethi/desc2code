#include <bits/stdc++.h>
using namespace std;
int main() {
	int T,i,N,K,j;
	scanf("%d",&T);
	for(i=1;i<=T;i++)
	{
	    scanf("%d%d",&N,&K);
	    int arr[N];
	    for(j=0;j<N;j++)
	    scanf("%d",&arr[j]);
	    sort(arr,arr+N);
	    long long num=0;
	    for(j=K;j<N-K;j++)num+=arr[j];
	    printf("%lf\n",double(num)/(N-2*K));
	}
	return 0;
}
