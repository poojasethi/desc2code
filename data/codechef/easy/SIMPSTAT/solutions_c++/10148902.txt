#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int i,j,k,l,m,n;
        cin>>n>>k;
        int a[10001];
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        sort(a,a+n);
        j=0;m=0;
        for(i=k;i<n-k;i++)
        {
            j+=a[i];
            m++;
        }
        printf("%0.6f\n",(j*1.0)/m);
    }
    return 0;
}
