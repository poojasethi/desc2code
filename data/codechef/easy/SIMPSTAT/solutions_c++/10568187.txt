#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,i,sum,k,a[100005];
    float avg;
    cin>>t;
    while(t--)
    {
        sum=0;
        cin>>n>>k;
        for(i=0;i<n;i++)cin>>a[i];
        sort(a,a+n);
        for(i=k;i<n-k;i++)
            sum+=a[i];
        avg=(float)sum/(n-(2*k));
        printf("%f\n",avg);
    }
    return 0;
}
