#include<cstdio>
using namespace std;
long gcd(long A,long B)
{
    long R,temp;
    if (B>A)
    {
        temp=B;
        B=A;
        A=temp;
    }
    while (A%B!=0)
    {
        R=A%B;
        A=B;B=R;
    }
    return B;
}
int main()
{
    int T; long M,N,side;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%ld %ld",&M,&N);
        side=gcd(M,N);
        printf("%ld\n",(M*N)/(side*side));
    }
    return 0;
}
