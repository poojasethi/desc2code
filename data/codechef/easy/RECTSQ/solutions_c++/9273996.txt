#include<iostream>

using namespace std;

int gcd(int a,int b){
    if(b==0)
    return a;
    else
    return gcd(b,a%b);
}

int main(){
    int t,i,a,b;
	cin>>t;
    for(i=0;i<t;i++){
        cin>>a>>b;
        int temp=gcd(a,b);
        int sq=(a*b)/(temp*temp);
        cout<<sq<<endl;
    }
    return 0;
}
