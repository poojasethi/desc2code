#include<iostream>
#include<math.h>

using namespace std;
int findMin(int x , int y)
{
 if(x>y)
  return y;

  else
   return x;


}
int gcdFinder(int x , int y)
{
int gcd = 1;
 for(int i = 1 ; i<=findMin(x , y) ; i++)
 {
  if(x%i==0 && y%i==0)
  {
   gcd = i;

  }


 }

return gcd;

}

int main()
{
 long Test , l , b;
 cin>>Test;
 while(Test--)
 {
  cin>>l>>b;
  cout<<l/gcdFinder(l , b)*(b/gcdFinder(l , b))<<endl;


 }

}
