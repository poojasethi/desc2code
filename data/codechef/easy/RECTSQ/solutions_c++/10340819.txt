#include<bits/stdc++.h>
using namespace std;
int hcf(int a,int b) {
    int r;
    while(b>0) {
        r=a%b;
        a=b;
        b=r;
    }
    return a;
}
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    int l,b,hc;
	    scanf("%d%d",&l,&b);
	    if(l>b) {
	        hc=hcf(l,b);
	    } else {
	        hc=hcf(b,l);
	    }
	    int ans = (l*b)/(hc*hc);
	    printf("%d\n",ans);
	}
	return 0;
}
