#include<iostream>
using namespace std;
int main()
{
    int t,m,n,mini,i,p,ans;
    cin>>t;
    while(t--)
    {
        p=1;
        cin>>m>>n;
        if(m>n)
            {
                mini=n;
            }
        else
            mini=m;
        for(i=(mini/2);i>1;i--)
        {
            if(m%i==0 && n%i==0)
               {
                   p=i;
                   break;
               }
        }
        if(m%mini==0 && n%mini==0)
            p=mini;
        m=m/p;
        n=n/p;
        ans=m*n;
        cout<<ans<<endl;
    }
    return 0;
}
