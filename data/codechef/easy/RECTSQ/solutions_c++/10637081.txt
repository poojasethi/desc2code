#include <iostream>
#include <bits/stdc++.h>
using namespace std;
 
int main() 
{
	long long int t,m,n,v;
	cin>>t;
	while(t--)
	{
	    cin>>n>>m;
	    v=min(m,n);
	    long long int side,ans;
	    for(long long int i=v;i>=1;i--)
	    {
	        if(m%i==0 && n%i==0)
	        {
	            side=i;
	            break;
	        }
	    }
	    ans=(m*n)/(side*side);
	    cout<<ans<<"\n";
	}
	return 0;
}
