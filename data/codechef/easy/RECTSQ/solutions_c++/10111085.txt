#include <iostream>
using namespace std;
#define ll long long

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll a,b,i,k;
        cin>>a>>b;
        if(a<b)
            k=a;
        else
            k=b;
        for(i=k;i>=1;--i)
            if(a%i==0 && b%i==0)
                break;
        cout<<(a*b)/(i*i)<<"\n";
        
    }
    return 0;
}
