#include<bits/stdc++.h>
using namespace std;
int findgcd(int x,int y){
     while(x!=y){
          if(x>y)
              return findgcd(x-y,y);
          else
             return findgcd(x,y-x);
     }
     return x;
}
int main()
{
   int t,n,m,num;
   cin>>t;
   while(t--)
   {
        cin>>n>>m;
        num= findgcd(n,m);
        cout<<(n*m)/(num*num)<<endl;
    }
    return 0;
}
