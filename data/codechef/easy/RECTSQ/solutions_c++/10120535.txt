#include<iostream>
#include<cstdio>
#include<cmath>

using namespace std;

int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int l,b;
        scanf("%d %d",&l,&b);
        int n=gcd(l,b);
        l=l/n;
        b=b/n;
        printf("%d\n",l*b);
    }
    return 0;
}
