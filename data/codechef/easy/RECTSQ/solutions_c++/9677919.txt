#include<iostream>
using namespace std;
inline int min1(int a,int b)
{
   return ((a<b)?a:b);
}

inline int max1(int a,int b)
{
    return ((a>b)?a:b);
}
int main()
{
  int t,m,n,a,b,div;
  cin>>t;
  while(t--)
  {
     div=1;
     cin>>m>>n;
     a=min1(m,n);
     b=max1(m,n);
     for(m=2;m<=a;m++)
     {
         if(a%m==0 && b%m==0)
            div=m;
     }
     cout<<(a*b)/(div*div)<<"\n";
  }
}
