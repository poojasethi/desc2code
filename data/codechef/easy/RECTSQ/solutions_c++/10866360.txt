#include <iostream>

using namespace std;

int gcd(int n, int m){
        if(m==1) return 1;
        if(m==0) return n;

        if(n<m) swap(n,m);
        return gcd(m, n%m);

}

int main()
{
        int t;
        cin>>t;

        while(t--){
                int n, m;
                cin>>n>>m;
                int g = gcd(n,m);
                cout <<(n/g)*(m/g)<< endl;
        }
    return 0;
}
