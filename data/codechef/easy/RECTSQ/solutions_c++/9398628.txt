#include<iostream>
using namespace std;
int gcd(int, int);

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int g, m, n;
        cin>>m>>n;
        g=gcd(m,n);
        cout<<((m*n)/(g*g))<<"\n";
    }
}

int gcd(int a, int b)
{
    if(b==0)
        return a;
    else
        return gcd(b, a%b);
}
