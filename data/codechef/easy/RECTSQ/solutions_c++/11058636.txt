#include<iostream>

using namespace std;

int main ()
{
	int T,i,j,k,M,N,S;

	//cout << "write total no. of test cases\n";
	cin >> T;

	for (i=0;i<T;i++)
	{
		//cout << "write the length & breath\n";
		cin >> N >> M;

		if (N < M)
		{
			S = N;
		}
		else
		{
			S = M;
		}

		for (j=S;j>0;j--)
		{
			if (M%j==0 && N%j==0)
			{
				break;
			}
		}
		cout << (M*N)/(j*j) << endl;
	}

	return 0;
}
