int Hcf(int a, int b){
    if (a>b) {
        if (a%b==0) {
            return b;
        }
        else
            return Hcf(b, a%b);
    }
    else{
        if (b%a==0) {
            return a;
        }
        else
            return Hcf(a,b%a);
    }
