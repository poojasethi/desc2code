def gcd (a, b):

    if a > b:

        smaller = a

    else:

        smaller = b


    for i in range(1,smaller + 1):

        if((a % i == 0) and (b % i == 0)):

            hcf = i

    return hcf


test = int(input())

while test > 0:

    m, n = map(int, raw_input().split())

    test = test - 1

    g =  gcd(m, n) * gcd(m, n)

    no_of_sqrs = (m * n) / g

    print(no_of_sqrs)
