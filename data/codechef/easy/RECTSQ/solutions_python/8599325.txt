def hcf(num1,num2):
    if num1==num2:
        return num1
    hcf=min(num1,num2)
    if hcf==0:
        return max(num1,num2)
    
    while(hcf>=1):
        if(num1 % hcf==0 and num2 % hcf==0):
            ans=hcf
            break
        hcf=hcf-1
    return ans

for i in range(0,t):
    
    m,n=map(int,raw_input().split())
    #m=int(m)

    hc=hcf(m,n)
    area=hc*hc
