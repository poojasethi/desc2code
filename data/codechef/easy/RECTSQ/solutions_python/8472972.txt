#!/usr/local/bin/python

def gcd(x,y):

    if y == 0:
        return x
    else:
        return gcd(y, x % y)

test_case_count = int( raw_input() )
for i in range(test_case_count):
    input = raw_input().split()
    (a,b) = ( int( input[0] ), int( input[1] ) )
    GCD = gcd(a,b)
    area = a * b
    total = area / GCD**2
    print total
