import sys

def gcd(a,b):
    if(b == 0): return a
    elif(b > a): return gcd(b, a)
    else: return gcd(b, a%b)

test_cases = int(sys.stdin.readline())
while(test_cases > 0):
    test_cases -= 1
    length, breath = map(int, sys.stdin.readline().split(' '))
    square_side = gcd(length, breath)
    rectangle_area = length * breath
    square_area = square_side**2
    print rectangle_area/square_area
