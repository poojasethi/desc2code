def gcdfn(x, y) :
    if ( x == 0 ) :
        return y
    elif ( y == 0 ): 
        return x
    else :
        return gcdfn(y, x%y )

for testcases in xrange(int(raw_input() ) ) :
    m, n = map(int, raw_input().split() )
    gcd = gcdfn(m, n);
    print int(m*n/(gcd*gcd))
