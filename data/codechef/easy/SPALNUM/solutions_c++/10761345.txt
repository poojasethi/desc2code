

#include <iostream>

using namespace std;
int main(int argc, const char * argv[]) {
    int T,L,R,f,rev,dig,i,num;
  
    cin>>T;
    while(T!=0)
    {
        f=0;
        cin>>L>>R;
        for(i=L;i<=R;i++)
        {
            dig=0,rev=0;
            num=i;
            while(num!=0)
            {
                dig=num%10;
                rev=rev*10+dig;
                num=num/10;
            }
            if(i==rev)
                f=f+i;
        }
        cout<<f<<"\n";
        T--;
    }
    return 0;
}
