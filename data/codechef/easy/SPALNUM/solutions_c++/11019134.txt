#include<iostream>
using namespace std;
bool isPallim(int n)
{
	int temp=n,r=0;
	while(n>0)
	{
		r=r*10+n%10;
		n/=10;
	}
	if(temp==r)
	return true;
	return false;
}
int main()
{
	long long arr[100001]={0};
	int i=1;
	for(i=1;i<100001;i++)
	if(isPallim(i)==true)
	arr[i]=i;
	for(i=1;i<100001;i++)
	arr[i]+=arr[i-1];
	int l,r,t;
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		if(isPallim(l)==true)
		cout<<arr[r]-arr[l]+l<<"\n";
		else
		cout<<arr[r]-arr[l]<<"\n";
	}
	return(0);
}
