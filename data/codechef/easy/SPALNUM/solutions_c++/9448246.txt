#include<iostream>
using namespace std;
int arr[100001];
int palin(int a)
{
    int x=0;
    while(a>0)
    {
        x=(x<<1)+(x<<3)+a%10;
        a/=10;
    }
    return x;
}
int main(void)
{
    ios::sync_with_stdio(false);
    for(int i=1,x=0;i<=100001;i++)
    {
        x=palin(i);
        if(x==i)
            arr[i]=x;
        else
            arr[i]=0;
    }
    int t; cin>>t;
    while(t--)
    {
        int l,r;
        cin>>l>>r;
        int sum=0;
        for(int i=l;i<=r;i++)
        {
            sum+=arr[i];
        }
        cout<<sum<<"\n";
    }
}
