#include <bits/stdc++.h>
using namespace std;
int l, r;

int Re(int x) {
    int res = 0;
    while (x) {
        res = res*10 + x%10;
        x /= 10;
    }
    return res;
}

int main() {
    int t; cin >> t;
    while (t--) {
        cin >> l >> r;
        int res = 0;
        for (int i = l; i <= r; i++)
        if (i == Re(i)) res += i;
        cout << res << endl;
    }
}
