#include<stdio.h>
 
long long int a[100005];
 
int
palin(long n){
	int d=0, i=0, b[15];
	while(n){
		b[d]=n%10;
		n=n/10;
		d++;
	}
	while(i<d/2+1){
		if(b[i]!=b[d-i-1])
			return 0;
		i++;
	}
	return 1;
}
 
void pre(){
	int i;
	a[1]=1;
	a[0]=0;
	for(i=0; i<=100001; i++){
		if(palin(i))
			a[i]=i;
		else
			a[i]=0;
	}
	for(i=1; i<=100001; i++){
		a[i]+=a[i-1];
	}
}
int main(){
	int t, l, r;
	pre();
	scanf("%d",&t);
	while(t--){
		long long sum;
		scanf("%d %d",&l, &r);
		if(l>1)
			sum=a[r]-a[l-1];
		else
			sum=a[r];
		printf("%lld\n",sum);
	}
	return 0;
}
