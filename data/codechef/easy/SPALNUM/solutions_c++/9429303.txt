#include<iostream>
using namespace std;
int main()
{
 int test,a,num,b,digit;
 cin>>test;
 while(test--)
 {
 int sum=0;
  cin>>a>>b;
  for(int i=a;i<=b;i++)
  {
  num=i;
  int rev=0;
  do
     {
         digit = num%10;
         rev = (rev*10) + digit;
         num = num/10;
     }while (num!=0);
     if(rev==i)
     {
     sum=sum+i;
     }
  }
  cout<<sum<<endl;
 }
 return 0;
}
