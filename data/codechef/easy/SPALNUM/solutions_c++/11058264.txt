#include<iostream>

using namespace std;

int main ()
{
	int T,L,R,i,j,k,N0,N1,N2,l,q,p,z,sum=0;

	//cout << "write total no. of test case\n";
	cin >> T;

	for (i=0;i<T;i++)
	{
		//cout << "write the extreme ends\n";
		cin >> L >> R;
		sum = 0;

		for (j=L;j<=R;j++)
		{
			N0 = N1 = j;
			q = 0;
		
			while (N0 > 0)
			{
				q++;
				N0 /= 10;
			}
			
			int A[q];

			for (k=0;k<q;k++)
			{

				A[k] = N1%10;
				N1 /= 10;
			}
			
			p = 0;
			z = (q/2)-1;
			for (k=0;k<=z;k++)
			{
				if (A[k]!=A[q-1-k])
				{
					p++;
				}
			}
			
			if (p==0)
			{
				sum += j;
			}
		}
		cout << sum << endl;
	}

	return 0;
}
