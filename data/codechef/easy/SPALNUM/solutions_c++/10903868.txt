#include<cstdio>
using namespace std;
bool palindrome(long N)
{
    long R=0,n=N;
    while(N>0)
    {
        R=(R*10)+N%10;
        N/=10;
    }
    if (n==R) return 1;
    else return 0;
}
int main()
{
    int T; long L,R,sum=0;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%ld %ld",&L,&R);
        sum=0;
        for (L;L<=R;L++) if (palindrome(L)) sum+=L;
        printf("%ld\n",sum);
    }
    return 0;
}
