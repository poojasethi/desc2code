#include<iostream>
using namespace std;
int rev(int n)
{
  int rev=0;
  while(n>0)
  {
       rev=rev*10+n%10;
       n/=10;
  }
  return rev;
}

int checkpalin(int n)
{
   if(n<10)
        return 1;
   return (n==rev(n));
}
int main()
{
   int t,a,b;
   long int sum=0;
   cin>>t;
   while(t--)
   {
       sum=0;
       cin>>a>>b;
       for(int i=a;i<=b;i++)
        if(checkpalin(i))
            sum+=i;
        cout<<sum<<"\n";
   }
}
