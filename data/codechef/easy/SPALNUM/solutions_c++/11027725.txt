#include<iostream>
using namespace std;
int rev(long n)
{int rem;
long num=0;
while(n!=0)
{rem=n%10;
num=num*10+rem;
n=n/10;
}
return num;}


int main()
{int t;
long l[100],r[100],sum;
cin>>t;
for(int i=0;i<t;i++)
cin>>l[i]>>r[i];
for(int i=0;i<t;i++)
{sum=0;
for(long j=l[i];j<=r[i];j++)
if(rev(j)==j)
sum+=j;
cout<<sum<<"\n";
}
return 0;
}
