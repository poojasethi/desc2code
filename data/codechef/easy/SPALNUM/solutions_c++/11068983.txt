#include<iostream>

using namespace std;

int ispalindrome(int m){
    int rev=0, temp, n = m;
    while(n >0){
        temp = n % 10;
        n = n / 10;
        rev = rev *10 + temp;
    }
    if(rev == m)
        return 1;
    else
        return 0;
}

int main(void){
    int t, l, r, sum =0; 
    cin>>t;
    while(t--){
        sum =0;
        cin>>l>>r;
        for(int i=l; i<=r; i++){
            if(ispalindrome(i))
                sum += i;
        }
        cout<<sum<<"\n";
    }
}
