_Bool checkForPalindrome(int a){
    int x,reverse=0,b;
    b=a;
    while (a) {
        
        x=a%10;
        a=a/10;
        reverse=reverse*10+x;
    }
    if (b==reverse) {
        return true;
    }
    else
        return false;
    
    
}
