#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,rev=0,temp,temp1,i,sum=0;
		cin>>a;
		cin>>b;
		for(i=a;i<=b;i++)
		{
			rev=0;
			temp1=i;
			while(temp1!=0)
			{
				temp=temp1%10;
				rev=rev*10+temp;	
				temp1=temp1/10;
			}	
			if(rev==i)
			{
				sum=sum+i;
			}
		}
		cout<<sum<<endl;
	}
}
