#include <iostream>
#include<string>
using namespace std;

int main()
{
        int t, l, r;
        int arr[6];

        cin>>t;
        while(t--){
                cin>>l>>r;
                int sum = 0, num;
                for(int p=l; p<=r; p++){
                        num=p;
                        int i = 0, j=0;
                        while(num){
                               arr[i] = num%10;
                               num/=10;
                               i++;
                        }
                        i--;
                        while(arr[i]==arr[j]){
                                i--, j++;
                        }

                        if(j>=i){
                                sum+=p;
                        }
                }

                cout<<sum<<endl;
        }
    return 0;
}
