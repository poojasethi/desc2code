#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
using namespace std;

bool palindrome(long long int num)
{
    int  reverse = 0, temp;
    temp = num;
    while (temp != 0)
     {
       reverse = reverse * 10;
       reverse = reverse + temp%10;
       temp    = temp/10;
     }
  if (num == reverse)
    return 1;
  else
    return 0;
}

long long int check(long long int r,long long int k)
{
    long long int sum=0;
    for(long long int i=r;i<=k;i++)
    {
        if(palindrome(i))
            sum=sum+i;
    }
    return sum;
}
int main()
{
    int l,r,t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&l,&r);
        printf("%lld\n",check(l,r));
    }
    return 0;
}
