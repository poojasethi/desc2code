#include<iostream>
#include<math.h>
#include<stdio.h>

using namespace std;
int k =0 , pal = 0;

int palindromeCheck(int x)
{
 int N = x;
 while(N)
     {
     k = N%10;
     pal = pal*10 + k;
     N = N/10;
     }

 if(pal==x)
 {
    pal = 0;
  return 1;
 }
 else
 {
     pal = 0;
     return 0;
 }

}

int main()
{
 int L , R , Test , sum = 0;
 cin>>Test;
 while(Test--)
 {
  cin>>L>>R;

  for(int i = L ; i<=R ; i++)
  {
   if(palindromeCheck(i)==1)
   {
    sum = sum+i;
   }

  }
  cout<<sum<<endl;
  sum = 0;
 }



}
