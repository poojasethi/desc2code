#include<iostream>
using namespace std;
int main()
{ int t;
  long int l,r,rev,tmp,s;
  cin>>t;
  while(t--)
  { cin>>l>>r;s=0;
    for(int i=l;i<=r;i++)
    { tmp=i;rev=0;
      while(tmp>0)
      { rev=(rev*10)+(tmp%10);
        tmp/=10;
      }
      if(i==rev)
       s+=i;  
    }
    cout<<s<<"\n";
  }
  return 0;
}
