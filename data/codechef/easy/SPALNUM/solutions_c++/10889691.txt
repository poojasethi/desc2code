#include<iostream>
using namespace std;
int main()
{
    int n,a,b,temp;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        int sum=0;
        cin>>a>>b;
        for(int i=a;i<=b;i++)
        {
            int rev=0,rem;
            temp=i;
            while(temp!=0)
            {
                rem=temp%10;
                rev=rev*10+rem;
                temp=temp/10;
            }
            if(rev==i)
                sum=sum+i;

        }
        cout<<sum<<endl;
    }
}
