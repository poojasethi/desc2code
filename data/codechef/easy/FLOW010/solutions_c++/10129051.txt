#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char a;
        cin>>a;
        if(a=='b'||a=='B')
            cout<<"BattleShip\n";
        if(a=='c'||a=='C')
            cout<<"Cruiser\n";
        if(a=='d'||a=='D')
            cout<<"Destroyer\n";
        if(a=='f'||a=='F')
            cout<<"Frigate\n";
    }
    return 0;
}
