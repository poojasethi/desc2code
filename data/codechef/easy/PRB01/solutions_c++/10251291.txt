#include<iostream>
#include<math.h>
using namespace std;
int main()
{
int i,j,n,p,t,T;
cin>>T;
for(t=1;t<=T;t++)
{
p=2;
cin>>n;
j=sqrt(n);
for(i=1;i<=j;i++)
 {
  if(j==1)
  {
    cout<<"yes\n";
  }
  else  
  {
   i=p;
   p++;
   if(n%i==0)
    {
      cout<<"no\n";
      break;
    }
   else
    {
     if(i==j)
     {
       cout<<"yes\n";
       break;
     }
    }
  }
 }
}
return 0;
}
