#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int ips,count,data;
    cin>>ips;
    while(ips--)
    {
        bool flag=true;
        cin>>data;
        if(data%2==0&&data>2)
            flag=false;
        else
        {
        for(int i=3;i<sqrt(data);i=i+2)
        {
            if(data%i==0)
                flag=false;
        }
    }
        if(flag==false||data==1)
        {
            cout<<"no\n";
        }
        else
        {
            cout<<"yes\n";
        }
    }
}
