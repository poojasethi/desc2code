#include <bits/stdc++.h>
using namespace std;
const int N = 1e6 + 4;
int n;
bool prime[N] = {0};

int init() {
    prime[0] = prime[1] = 1;
    for (int i = 2; i*i < N; i++)
    if (!prime[i]) {
        for (int k = i*i; k < N; k += i) prime[k] = 1;
    }
}

int main() {
    init();
    int t; cin >> t;
    while (t--) {
        cin >> n;
        if (prime[n]) cout << "no" << endl;
        else cout << "yes" << endl;
    }
}
