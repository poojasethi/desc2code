#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,j;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		int count = 0;
		scanf("%d",&n);
		for(j=2;j<=n;j++)
		{
			if(n % j == 0)
			{
				count++;
			}
		}
		if(count == 1)
		{
			printf("yes\n");
		}
		else
		{
			printf("no\n");
		}
	}
	return 0;
}
