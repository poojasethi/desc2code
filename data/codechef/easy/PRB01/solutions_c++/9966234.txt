#include <bits/stdc++.h>
using namespace std;

void checkprime(int n)
{
	if(n%2==0 && n!=2 )
		cout<<"no"<<endl;
	else 
	{
		for(int i=3;i<=sqrt(n);i++)
		{
			if(n%i==0)
				{
					cout<<"no"<<endl;
					return ;
				}
		}
		cout<<"yes"<<endl;
	}
}

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		if(n!=2 || n!=3 || n!=1)
			checkprime(n);
		else
			cout<<"yes"<<endl;
	}
	return 0;
}
