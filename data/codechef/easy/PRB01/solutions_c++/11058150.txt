#include <iostream>

using namespace std;

int main()
{
    int t,i,j,num,prime_no;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>num;
        prime_no=num;
        for(j=2;j<num/2;j++)
        {
            if(num%j==0)
            {
                prime_no=j;
                break;
            }
        }
        if(prime_no==num)
            cout<<"yes"<<endl;
        else
            cout<<"no"<<endl;
    }
    return 0;

}
