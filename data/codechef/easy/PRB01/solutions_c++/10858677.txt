#include<iostream>
using namespace std;
int main()
{
	int t,i,m,prime;
	cin>>t;
	while(t--)
	{	cin>>m;
	        prime=0;
	        for(i=2;i<(m/2+1);i++)
		{  if(m%i==0)
			{ prime=1;
			 break;}
		}
		if(m==1 ||m==2)
	           prime=0;
		if(prime==1)
		 cout<<"no"<<endl;
                else if(prime==0)
                cout <<"yes"<<endl;	
	}
	return 0;
}
