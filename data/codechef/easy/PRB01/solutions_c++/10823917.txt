#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,x=0,i;
	cin >> t;
	while(t--)
	{
		x=0;
	    cin >> n;
	    if(n==1)
	    	x=1;
	    for(i=2;i<=(n/2);++i)
	    {
	    	if(n%i == 0)
	    		{
	    			x=1;
	    	    break;
	    		}
	    }
	    if(x==0)
	    	cout << "yes" << endl;
	    else
	    	cout << "no" << endl;

	}
	return 0;
}
