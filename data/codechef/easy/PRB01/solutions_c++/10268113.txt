#include<bits/stdc++.h>

using namespace std ;

bool isprime(long long int n )
{
    if(n<=1)
        return false;
    else if(n<=3)
        return true;
    else if((n%2==0)||(n%3==0))
        return false;
    long long int i=5;
    while(i*i<=n)
    {
        if((n%i==0)||(n%(i+2))==0)
            return false;
        i=i+6;

    }
    return true;
}
int main()
{
    long long int n,T;
    cin>>T;
    while(T--){
    scanf("%lld",&n);
    if(isprime(n))
        printf("yes\n");
    else
        printf("no\n");
    }

}
