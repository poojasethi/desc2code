#include <bits/stdc++.h>
using namespace std;

int p[(int)1e6];

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    
    for (int i = 2; i < 1e5; ++i) {
        if (p[i]) continue;
        long long j = i * 1LL * i;
        while (j < 1e5) {
            p[j] = 1;
            j += i;
        }
    }
    
    int t; cin >> t;
    while (t--) {
        int n; cin >> n;
        cout << (p[n] ? "no" : "yes") << endl;
    }
    
    return 0;
}
