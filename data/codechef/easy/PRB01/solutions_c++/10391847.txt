ll fast_exp(l base,l exp)
 {
     if(exp==0)
        return 1;
     else
     {
         if(exp%2==0)
         {
             ll temp=fast_exp(base,exp/2);
             temp=temp*temp;
             if(temp>=MOD)
                return temp%MOD;
             else
                return temp;
         }
         else
         {
             ll temp=fast_exp(base,(exp-1)/2);
             temp=base*temp*temp;
             if(temp>=MOD)
                return temp%MOD;
             else
                return temp;

         }
     }
 }

int setbit(ll N)
{

bool is_prime[100005];
void sieve()//For precalculations
{
    l i,j;
    MEM(is_prime,1);//Assuming all are primes
    is_prime[1]=0;// 1 is not prime
    for(i=2;i<=100000;++i)

        if(is_prime[i])
        {
            for(j=i*2;j<=100000;j+=i)
                is_prime[j]=0;

        }

}
