

#include<bits/stdc++.h>
using namespace std;

bool prime( int x ){
	
		if(x<2)
                       return false;
                for( int i=2; i<=sqrt(x); i++ ){
                        if((x%i) == 0)
                                return false;
                }
                return true;

}

int main(void){
	
	int t,x;
	cin>>t;
	while(t--){
		cin>>x;
		if(prime(x))
			cout<<"yes"<<endl;
		else
			cout<<"no"<<endl;
	}
}
