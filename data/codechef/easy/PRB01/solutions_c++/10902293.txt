#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
    long T,N;bool flag;
    scanf("%ld",&T);
    while(T--)
    {
        scanf("%ld",&N);
        flag=1;
        for(long i=2;i<sqrt(N);i++)
        {
            if (N%i==0) {flag=0;break;}
        }
        if (flag==1) printf("%s\n","yes");
        else printf("%s\n","no");
    }
    return 0;
}
