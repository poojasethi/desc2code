#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<vector>

using namespace std;

int  main()
{
    int t;
    long long int num;
    scanf("%d",&t);
    while(t--)
    {
        int count=0;
        scanf("%lld",&num);
        for(int i=2;i<=num/2;i++)
        {
            if(num%i==0)
            {
             count++;
             break;
            }
         }
         if(count!=0)
            printf("no\n");
         else
            printf("yes\n");
    }
    return 0;
}
