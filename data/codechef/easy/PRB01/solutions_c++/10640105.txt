#include <iostream>
#include<vector>
#include<stdio.h>
using namespace std;
vector<int> prime;

void createPrimeArr(){
        int j;
        prime.push_back(2);

        for(int i = 3; i<100000; i++){
                for(j = 0; j<prime.size(); j++){
                        if (i%prime[j] == 0) break;
                }
                if (j == prime.size())
                        prime.push_back(i);
        }
}

int main()
{
        int t, n, i;

        scanf("%d", &t);
        createPrimeArr();

        while(t--){
                scanf("%d", &n);
                i = 0;
                while(prime[i] < n)
                        i++;
                if (prime[i] == n)
                        printf("yes\n");
                else printf("no\n");

        }
    return 0;
}
