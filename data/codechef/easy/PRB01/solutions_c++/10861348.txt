#include<bits/stdc++.h>
using namespace std;
 
int check_prime(int a)
{
   int c;
 
   for ( c = 2 ; c <= a - 1 ; c++ )
   { 
      if ( a%c == 0 )
     return 0;
   }
   if ( c == a )
      return 1;
}



int main()
{   
 
 
    
 
      int t;
    
    
    
    scanf("%d", &t);
    
    while(t--)
        
    {
        
        int n, result;
 
   
   scanf("%d",&n);
 
   result = check_prime(n);
 
   if ( result == 1 )
   { printf("yes");
        printf("\n");}
   else
       
   {
      printf("no");
       printf("\n");
   }
    } 
 
        
        return 0;
}
