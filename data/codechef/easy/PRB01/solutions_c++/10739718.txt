#include<iostream>
#include<math.h>
#include<string.h>
#include<stdio.h>
using namespace std;

int main()
{
 int Test;
 int Number , chk = 0;
 cin>>Test;
 while(Test--)
 {
  cin>>Number;
  for(int i = 2 ; i<=Number/2 ; i++)
  {
   if(Number%i==0)
   {
    chk++;
    break;
   }


  }
  if(chk==0)
   cout<<"yes"<<endl;
   else
    cout<<"no"<<endl;
  chk = 0;

 }



}
