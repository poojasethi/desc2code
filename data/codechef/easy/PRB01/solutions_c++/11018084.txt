#include<iostream>

using namespace std;

int main ()
{
	int T,N,j,k,i;

	//cout << "write T\n";
	cin >> T;

	for (i=0;i<T;i++)
	{
		//cout << "write N\n";
		cin >> N;
		
		k = 0;
		for (j=2;j<N;j++)
		{
			if (N%j == 0)
			{
				k++;
			}
		}
		if (k == 0)
		{
			cout << "yes\n";
		}
		else 
		{
			cout << "no\n";
		}
	}

	return 0;
}
