#include<bits/stdc++.h>
using namespace std;
bool isPrime(int n) {
    for(int i=2;i<n;i++) {
        if(n%i==0) {
            return false;
        }
    }
    return true;
}
int main() {
	int t;
	scanf("%d",&t);
    while(t--)
	{
	    int n;
	    scanf("%d",&n);
	    if(isPrime(n)) {
	        printf("yes\n");
	    } else {
	        printf("no\n");
	    }
	}
	return 0;
}
