#include <iostream>

using namespace std;

string prime(int n)
{
    if(n==1)
    {
        return "yes";
    }
    for(int i=2;i<=n/2;i++)
    {
        if(n%i==0)
        {
            return "no";
        }
    }
    return "yes";
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<prime(n)<<endl;
    }
    return 0;
}
