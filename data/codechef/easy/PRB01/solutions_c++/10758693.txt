

#include <iostream>

using namespace std;
int main(int argc, const char * argv[]) {
    int T,i,f,k,N;
  
    cin>>T;
    while(T!=0)
    {
        
        f=0;
        cin>>N;
     
        for(i=2;i<=N/2;i++)
        {
            if(N%i==0)
            
                f=1;
            
        }
        if(f==0)
            cout<<"yes\n";
        
        else
            cout<<"no\n";
        T--;
    }
    return 0;
}
