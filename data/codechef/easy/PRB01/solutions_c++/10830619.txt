

#include <cstdlib>
#include <iostream>
#include <math.h>
using namespace std;


int main(void) 
{
    int X;
    cin>>X;
    if(X<1 || X>20)
        return 0;
    while(X--)
    {
        int flag=1;
    int Q,R;
    cin>>Q;
    if(Q>=1 && Q<=pow(10,5))
    {
    R=sqrt(Q);
    for(int i=2;i<=R;i++)
    {
       if((Q%i)==0)
       {
           flag=0;
       }   
    }
    if(flag==1)
    cout<<"yes"<<endl;
    else if(flag==0)
        cout<<"no"<<endl;
    }
  }
    return 0;
}
