//prime
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        cin>>n;
        int prime=0,i;
        if(n==1)
        {
            cout<<"no\n";
            continue;
        }
        if(n==2)
        {
            cout<<"yes\n";
            continue;
        }
        int m=sqrt(n);
        for(i=2;i<=m;i++)
        {
            if(n%i==0)
                {prime=1;
                break;}
        }
        if(prime==0)
            cout<<"yes\n";
        else
            cout<<"no\n";
    }
    return 0;
}
