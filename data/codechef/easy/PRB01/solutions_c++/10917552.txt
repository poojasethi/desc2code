#include<iostream>
#include<math.h>

using namespace std;

int main(void){
    int t, a;
    cin>>t;
    
    while(t--){
        cin>>a;
        
        if(a == 1)
            cout<<"no";
        else if(a == 2 || a == 3)
            cout<<"yes";
        else{
            int i;
            for(i=2; i<(int)sqrt(a); i++){
                if(a % i == 0){
                    cout<<"no";
                    break;
                }
               
            }if(a % i != 0){
                    cout<<"yes";
            }
        }
        cout<<"\n";
    }
}
