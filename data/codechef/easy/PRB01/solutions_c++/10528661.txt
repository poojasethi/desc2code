#include <iostream>
#include<cstdlib>
using namespace std;

int main() {
	int t,n,i,found;
	cin>>t;
	while(t--)
	{
		found=0;
		cin>>n;
		if(n==1)
		cout<<"no";
		else
		{
		for(i=2;i<n;i++)
		{
			if(n%i==0)
			{
				found++;
				cout<<"no\n";
				break;
			}
		}
		}
		if(found==0)
		cout<<"yes\n";
    }			
	return 0;
}
