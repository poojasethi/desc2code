def sieve(m,n):
     "Return all primes <= n."
     np1 = n + 1
     s = list(range(np1))
     s[1] = 0
     sqrtn = int(round(n**0.5))
     for i in xrange(m, sqrtn + 1):
         if s[i]:
             s[i*i: np1: i] = [0] * len(xrange(i*i, np1, i))
     return filter(None, s)


for _ in range(input()):
    inp = input()
    if (inp in sieve(0,inp)):
        print 'yes'
    else:
        print 'no'
