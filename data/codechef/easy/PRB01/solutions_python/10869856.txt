def prime(n):
    if n==2 or n==3:
        return True
    if n%2==0 or n%3==0:
        return False
    i=5
    skip=2
    while i**2<n:
        if n%i==0:
            return False
        i+=skip
        skip=6-skip
    return True

T=int(raw_input())
for t in range(T):
    if prime(int(raw_input())):
        print "yes"
    else:
        print "no"
