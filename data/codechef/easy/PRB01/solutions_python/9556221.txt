t = int(raw_input())
for i in range(0,t):
    flag = 0
    num = int(raw_input())
    if num == 1:
        print("no")
    else:
        for j in range(2,num):
            if num % j == 0:
                flag = 1 
        if flag == 0:
            print("yes")
        else:
            print("no")
