class NumberMirror{

    void output()
    {
        int number;
        Scanner scanner=new Scanner(System.in);
        number=scanner.nextInt();

        System.out.println(number);
    }

    public static void main(String args[])
    {
        NumberMirror obj=new NumberMirror();
        obj.output();
    }
}
