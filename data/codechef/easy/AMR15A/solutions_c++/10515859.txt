#include <iostream>
#include<stdio.h>


int main(){
    int n,nm,cnt_e=0, cnt_o=0;
    scanf("%d",&n);
    while(n--)
    {
        scanf("%d", &nm);
        if(nm %2 == 0)
            cnt_e++;
        else
            cnt_o++;
    }
    if(cnt_e > cnt_o)
        printf("READY FOR BATTLE");
    else
        printf("NOT READY");
    return 0;
}
