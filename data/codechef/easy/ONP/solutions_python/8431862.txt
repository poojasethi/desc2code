inp = int(raw_input())

opr = ["(", "+", "-", "*", "/", "^"]

for j in range(inp):
    num = ""
    sym = []
    res = ""
    arr = raw_input()
    for k in arr:
        if k == ")":
            res += num + sym.pop()
            num = ""
            sym.pop()
            
        elif (k in opr):
            sym.append(k)
        elif (k not in opr):
            num += k

    print res
