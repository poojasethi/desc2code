#!/usr/bin/python
cases = int(raw_input())
result = []
operators = []
for i in range(cases):
	expression = str(raw_input())
	

	for c in expression:

		if c.isalpha():
			result.append(c)
		elif c is ')':
			result.append(operators.pop())
		elif c is not '(':
			operators.append(c)

	print ''.join(result)
	result = []
	operators = []
