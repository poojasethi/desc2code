k=int(raw_input())
while k>0:
    y=raw_input()
    m=""
    n=[]
    x=[]
    i=0
    while i<len(y):

        if y[i]=='(' :
            x.append(y[i])
    
        elif y[i]==')':
            j=len(n)
            m+=n[j-1]
            n.pop()
            x.pop()

        else:
            if y[i].isalnum():
                m+=y[i]
            else :
                n.append(y[i])
        i=i+1

    print m

    k=k-1
