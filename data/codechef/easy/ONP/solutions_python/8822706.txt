no_tc=int(raw_input())
while no_tc>0:
    no_tc-=1
    tc=raw_input()
    operators=[""]
    chars=""
    for i in tc:
        if i=="(":
            operators.append("")
        elif ((i>="a" and i<="z") or (i>="A" and i<="Z")):
            chars+=i
        elif i==")":
            chars+=operators.pop()
        elif i==" ":
            pass
        else:
            operators[-1]+=i
    if len(operators)>0:
        chars+=operators[0]
    print chars
