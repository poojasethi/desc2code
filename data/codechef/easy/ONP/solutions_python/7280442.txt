t = input()
for i in xrange(t):
    s = []
    ans =""
    ex = raw_input()
    for ch in ex:
        if ch == '(':
            continue
        elif ch == ')':
            ans += s.pop()
        elif ch == '+' or ch == '-' or ch == '*' or ch== '^' or ch =='/':
            s.append(ch)
        else:
            ans += ch
    print ans
