t=int(raw_input())
for asd in range(t):
	s=raw_input()
	l=[]
	ans=""
	for c in s:
		if ((ord(c)>=ord('A') and ord(c)<=ord('Z')) or (ord(c)>=ord('a') and ord(c)<=ord('z'))):
			ans+=c
		else:
			if (c!=')'):
				l.append(c)
			else:
				while (l[-1]!='('):
					x=l.pop()
					ans+=x
				l.pop()
	print ans
