cases=int(raw_input())
operators=[]
result=[]
list_oper=['+','-','*','/','^']
while cases>0:
    output=[]
    exp=list(raw_input())
    for j in range(len(exp)):
        if exp[j]==')':
            output.append(operators.pop())
        elif exp[j] in list_oper:
            operators.append(exp[j])
        elif exp[j]=='(':
            continue
        else:
            output.append(exp[j])
    result.append(''.join(output))
    cases-=1
for a in result:
    print a
