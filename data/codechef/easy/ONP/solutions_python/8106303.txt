
t = int(raw_input())

c=0
iden = ['+','-','*','/','^']
oper = []
brc = []
tt =[]
while c<t:

    exp = list(raw_input())
    out =[]

    for i in exp:
        if i== '(':
            brc.append(i)
        elif i in iden:
            oper.append(i)
        elif i == ')':
            brc.pop()
            z = oper.pop()
            out.append(z)
        else:
            out.append(i)

    c+=1
    tt.append(''.join(out))

for k in range(t):
    print tt[k]
