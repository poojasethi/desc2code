#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        stack <char> st;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]==')')
            {
                cout<<st.top();
                st.pop();
            }
            else if(s[i]>='a'&& s[i]<='z')
            {
                cout<<s[i];
            }
            else if(s[i]!='(')
            {
                st.push(s[i]);
            }
        }
        cout<<endl;
    }
}
