#include<iostream>
#include<string>
#include<stack>
using namespace std;
int main()
{
int n,i;
string s;
stack <char> operators;
stack <char> bracket;
cin>>n;
while(n--)
{
cin>>s;
for(i=0;i<s.size();i++)
{
if(s[i]=='+' || s[i]=='-' || s[i]=='*' || s[i]=='/' || s[i]=='^' || s[i]=='%')operators.push(s[i]);
else if(s[i]==')') 
{
cout<<operators.top();
operators.pop();
}
else if(s[i]=='(')
{
bracket.push(s[i]);
}
else cout<<s[i];
}
cout<<endl;
}
return 0;
 
}
