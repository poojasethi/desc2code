#include <bits/stdc++.h>
using namespace std;
string s;

int main() {
    int t; cin >> t;
    while (t--) {
        cin >> s;
        stack <char> S;
        for (int i = 0; i < s.size(); i++) {
            if (s[i] == '(') continue;
            if (s[i] >= 'a' && s[i] <= 'z') cout << s[i];
            else if (s[i] == ')') {
                cout << S.top();
                S.pop();
            }
            else S.push(s[i]);
        }
        cout << endl;
    }
}
