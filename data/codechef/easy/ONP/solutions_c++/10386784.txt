#include<bits/stdc++.h>

using namespace std;
queue<char> q;
stack<char> s;
int main()
{
        int t;
    cin>>t;
    while(t--)
    {
        string str;
        cin>>str;
        int len=str.length();
        for(int i=0;i<len;i++)
        {
            if(str[i]!=')'&&str[i]!='(')
            {
                char ch=str[i];
                if(ch>=97&&ch<=122)
                    q.push(ch);
                else
                    s.push(ch);
            }
            if(str[i]==')')
            {
                q.push(s.top());
                s.pop();
            }
        }
        while(!q.empty())
        {
            cout<<q.front();
            q.pop();
        }
        cout<<"\n";
    }
}
