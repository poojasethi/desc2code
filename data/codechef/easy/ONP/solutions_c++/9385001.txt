void convert(char ch[400],char rpn[400])
 {   int i=0,j=0;
    char c;
     c=cin.get();
    while(1)
    { if(c=='\n'||c==' ')
        break;
        if(isalpha(c))
            rpn[j++]=c;
       else if(c!=')')
        {
         ch[i++]=c;
        }
       else if(c==')')
        {
            i--;
            while(ch[i]!='(')
             rpn[j++]=ch[i--];

        }
       c=cin.get();

    }
    rpn[j]=c;
    for(i=0;rpn[i]!='\n';i++)
    {
        cout<<rpn[i];
    }
    cout<<endl;
    }
int main()
{
