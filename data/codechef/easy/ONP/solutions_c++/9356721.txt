#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;cin>>t;
    while(t--)
    {
        stack<char>q;
        string s;cin>>s;int i=0;string t;
        while(i<s.size())
        {
            if((s[i]=='(')||(s[i]=='+')||(s[i]=='-')||(s[i]=='*')||(s[i]=='/')||(s[i]=='^'))
                q.push(s[i]);
            if(s[i]>=97 && s[i]<=122)
                t+=s[i];
            if(s[i]==')')
            {
                while(q.top()!='(')
                {
                   t+=q.top();
                   q.pop();
                }
                q.pop();
            }
            i++;
        }
        cout<<t<<endl;
        t.clear();
    }
}
