#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s[10000];
		cin>>s;
		stack <char> st;
		for(int i=0;i<strlen(s);i++)
		{
			if(s[i]-'a'>=0 and s[i]-'a'<=25)
				cout<<s[i];
			else if(s[i]==')')
				{
					cout<<st.top();
					st.pop();
				}
			else if(s[i]!='(')
				st.push(s[i]);
		}
		cout<<endl;
	}
	return 0;
}
