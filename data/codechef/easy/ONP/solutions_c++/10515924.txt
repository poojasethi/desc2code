#include<stdio.h>
#include<iostream>
#include<vector>


int main () {

	int t,  k;
	char stack[512];
	char str[512];
	scanf("%d",&t);
	
	
	while(t--) {
		scanf(" %s",str);
		k = 0;
		for(int i=0; str[i]; i++) {
			if(str[i] >= 'a' && str[i] <= 'z')
				printf("%c",str[i]);
			else if(str[i] != ')') {
				stack[k++] = str[i];
			}
			else if(str[i] == ')') {
				k--;
				while(stack[k] != '(') {
					printf("%c",stack[k--]);
				}
				//k--;
			}
		}
		
		printf("\n");
	
	}

	return 0;
}
