#include<iostream>
#include<string>
using namespace std;
int main()
{
    char stack[400],exprssn[400];
    int ips,top;
    cin>>ips;
    string data;
    while(ips--)
    {
        cin>>data;
        int j=0;
        top=-1;
        for(int i=0;data[i]!='\0';i++)
        {
            if(data[i]=='(')
                continue;
            else if(data[i]>=97&&data[i]<=122)
            {
                exprssn[j++]=data[i];
            }
            else if(data[i]=='*'||data[i]=='+'||data[i]=='-'||data[i]=='/'||data[i]=='^')
            {
                stack[++top]=data[i];
            }
            else if(data[i]==')')
            {
                exprssn[j++]=stack[top--];
            }
        }
        exprssn[j]='\0';
        for(int k=0;exprssn[k]!='\0';k++)
        {
            cout<<exprssn[k];
        }
        cout<<endl;
    }
    
}
