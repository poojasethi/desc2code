#include <cstdio>
#include <cctype>
 
using namespace std;
 
int main()
{
    int t;
    static char rpn[400], op[400], c;
    int rpn_pos, op_pos;
    scanf("%d\n", &t);
    for (int i = 0; i < t; ++i)
    {
        rpn_pos = 0;
        op_pos = -1;
        while ((c = getchar()) != '\n')
        {
            if (c == ')')
            {
                while (op[op_pos] != '(')
                    rpn[rpn_pos++] = op[op_pos--];
                --op_pos;
            }
            else if (isalpha(c))
                rpn[rpn_pos++] = c;
            else
                op[++op_pos] = c;
        }
        rpn[rpn_pos] = '\0';
        printf("%s\n", rpn);
    }
    return 0;
}
