#include<iostream>
#include<stack>
using namespace std;		
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		char exp[400],ch;
		int count=1,t=0,flag=0,i;
		stack< char> s;
		
		while(count!=0)
		{
			if(flag==0)
			{
				count--;
				flag++;
			}
			cin>>ch;
			if(ch=='(')
			count++;
			if(ch==')')
			count--;
			exp[t++]=ch;
		}
		for(i=0;i<t;i++)	
		{
			ch=exp[i];
			if(ch=='+' || ch=='-' || ch=='*' || ch=='/' || ch=='%' || ch=='^' || ch=='(')
			s.push(ch);
			else if(ch==')')
			{
				while(s.top()!='(')
				{
					cout<<s.top();
					s.pop();
				}
				s.pop();
			}
			else
			cout<<ch;
		}
		cout<<"\n";
	}
	return 0;
}
