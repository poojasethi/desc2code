# include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i;
    std::ios_base::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        stack<char> st;
        string s,str;
        str="";
        cin>>s;
        for(i=0;i<s.length();i++)
        {
            if((s[i]>='0' && s[i]<='9') || (s[i]>='a' && s[i]<='z') || (s[i]>='A' && s[i]<='Z'))
            {
                str=str+s[i];
            }
            else
            {
                if(st.empty())
                {
                    st.push(s[i]);
                }
                else
                {
                    if(s[i]=='(')
                    {
                        st.push(s[i]);
                    }
                    else if(s[i]==')')
                    {
                        while(st.top()!='(')
                              {
                                  str=str+st.top();
                                  st.pop();
                              }
                              st.pop();
                    }
                    else
                    {
                        if(st.top()=='(')
                        {
                            st.push(s[i]);
                        }
                        else
                        {
                            str=str+st.top();
                            st.pop();
                        }
                    }
                }
            }
        }
        cout<<str<<endl;
    }
}
