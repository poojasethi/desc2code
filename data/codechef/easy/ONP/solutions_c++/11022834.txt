        #include<stdio.h>
        #include<string.h>
        int main()
        {
        	int t,i,c;
        	char a[400];
        	char b[400];
        	scanf("%d",&t);
        	while(t--)
        	{
        		scanf("%s",a);
        		c=0;
        		for(i=0;i<strlen(a);i++)
        		{
        			if(a[i]=='(')
        				continue;
        			else if(a[i]=='+'||a[i]=='-'||a[i]=='^'||a[i]=='*'||a[i]=='/')
        			{
        				c++;
        				b[c]=a[i];
        			}
        			else if(a[i]==')')
        			{
        				printf("%c",b[c]);
        				c--;
        			}
        			else
        				printf("%c",a[i]);
        		}
        		printf("\n");
        	}
        	return 0;
        }
