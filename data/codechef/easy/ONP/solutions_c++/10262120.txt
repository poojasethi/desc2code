#include <bits/stdc++.h>

long long int t, i, j;
char str[401],res[401];
using namespace std;
int precedence(char c)
{
	switch(c){
	case '(':
	 return 0;
	 
	case '+':
		return 1;
	case '-':
	 return 2;
	case '*':
		return 3;
	case '/':
	 return 4;
	case '^':
		return 5;
	default:
	 return 0;	
        }
}

int main()
{
	stack<char>s;
	scanf("%lld",&t);
	while (t--) {
		j = 0;
		scanf(" %s",str);
		i = 0;
		while(str[i] != '\0') {
			if (str[i] == '(') {
				s.push(str[i]);
			} else if (str[i] == ')') {
				while(s.top() != '(') {
					printf("%c",s.top());
					s.pop();
				}
				s.pop();
			} else if (str[i] == '+' || str[i] == '-' || str[i] == '*' || str[i] == '/' || str[i] == '%' || str[i] == '^') {
				while (!s.empty() && precedence(str[i]) < precedence(s.top())) {
					printf("%c", s.top());
					s.pop();
				}
				s.push(str[i]);
			} else {
				printf("%c",str[i]);
			}
			i++;
		}
		while (! s.empty()) {
			printf("%c", s.top());
			s.pop();
		}
		printf("\n");
	}
	return 0;
}
