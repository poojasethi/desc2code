#include<iostream>
#include<string.h>
//#include<conio.h>	
using namespace std;
int main()
{
	int t,i,j,k,count;
	cin>>t;
	char a[405],b[405],stack[405];
	while(t)
	{	t--;i=0;j=0;k=0;count=0;
	    memset(&a[0], 0, sizeof(a));
	    memset(&b[0], 0, sizeof(b));
	    memset(&stack[0], 0, sizeof(stack));
		cin>>a;//cout<<endl;
		
		while(a[i]=='(')
			{count++;i++;}
		while (count!=0)
		{	if(a[i]=='(')
			count++;
			else if(a[i]=='+'||a[i]=='-'||a[i]=='*' || a[i]=='/' ||a[i]=='%' ||a[i]=='^')
				{stack[j]=a[i];j++;}
			else if(a[i]==')')
				{count--;j--;b[k]=stack[j];k++;}
			else
				{b[k]=a[i];k++;}
			i++;
		}
		
		cout<<b<<endl;
	}
	//getch();
	return 0 ;
}
