#include <iostream>
#include<stack>
#include<deque>
#include<string>
#include<stdio.h>

using namespace std;

stack<char> brace;
deque<string> expression;

void rpn(char str[]){

        int i = 0;
        string first, second, third;

        while(str[i]!='\0'){

                while(str[i] != ')' || str[i] == '('){

                        if (str[i] == '('){
                                brace.push('(');
                        }
                        else{
                                expression.push_front(string (1, str[i]));
                        }
                        i++;
                }

                if(str[i] == ')'){
                        third.assign(expression.front());
                        expression.pop_front();
                        second.assign(expression.front());
                        expression.pop_front();
                        first.assign(expression.front());
                        expression.pop_front();

                        expression.push_front(first+third+second);
                        brace.pop();
                        i++;
                }
        }

        cout<< expression.front() << "\n";
        return;
}

int main()
{
        int t;
        char str[400];

        scanf("%d", &t);
        while(t--){
                scanf("%s", str);
                rpn(str);
        }
    return 0;
}
