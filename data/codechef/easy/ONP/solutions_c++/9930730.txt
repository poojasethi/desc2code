#include<iostream>
 
using namespace std;
 
char stack[500];
int top=-1;
void push(char a)
{
    stack[++top]=a;
}
 
char pop()
{
 
    if(top!=-1)
	   return (stack[top--]);
}
int main()
{
  int t;
