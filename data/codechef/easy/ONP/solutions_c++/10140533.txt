#include<stdio.h>

int main()
{
    char expr[400];
    int test,k;
    char s[400];
    int top;
    scanf("%d",&test);
    while(test-->0)
    {
        scanf("%s",&expr);
        k=0;
        top=-1;
        for(int i=0;expr[i];i++)
        {
            if(expr[i]=='(')
                continue;
            else if(expr[i]=='+'||expr[i]=='-'||expr[i]=='*'||expr[i]=='/'||expr[i]=='^')
                s[++top]=expr[i];
            else if(expr[i]==')')
                printf("%c",s[top--]);
            else
                printf("%c",expr[i]);
        }
        printf("\n");
    }
    return 0;
}
