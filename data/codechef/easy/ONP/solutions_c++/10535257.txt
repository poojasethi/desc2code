#include<iostream>
#include<math.h>
using namespace std;

int main() {
	// your code goes here
	int T;
	char expr[401],stack[401];
	cin>>T;
	while(T--){
	    int top=-1,brackets=0;
	    cin>>expr;
	    for(int i=0;expr[i]!='\0';i++){
	        if(expr[i]=='('){
	            brackets++;
	        }else if(expr[i]==')'){
	            brackets--;
	            cout<<stack[top--];
	        }else if(expr[i]>='a'&&expr[i]<='z'){
	            cout<<expr[i];
	        }else
	            stack[++top]=expr[i];
	    }
	    cout<<endl;
	    
	}
	return 0;
}
