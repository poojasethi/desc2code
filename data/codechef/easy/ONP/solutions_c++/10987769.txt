#include <iostream>
#include <math.h>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <string>
#include <vector>

using namespace std;


int main ()
{ 
	int t;
	cin >> t;
	string exp;
	while(t!=0)
	{	
		cin >> exp;
		string::const_iterator i;
		vector<char> stack1(402);
		vector<char> stack2(402);
		
		for (i=exp.begin();i!=exp.end();i++)
		{
			if(*i>=97 && *i<=122)
				cout << *i;
			else if(*i=='(')
				stack1.push_back(*i);
			else if(*i==')')
				{	stack1.pop_back();
					cout << stack2.back();
					stack2.pop_back();
				}
			else 
			{	if(stack1.back()=='(')
					stack2.push_back(*i);
			}

		}
		printf("\n");
		t--;
	}
}
