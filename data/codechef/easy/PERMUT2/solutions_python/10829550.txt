#!/usr/bin/python




maxLengthList = 1
while maxLengthList!=0:
  count=0
  iList=[]
  maxLengthList = input()
  if maxLengthList==0:
    break
  shopList = [None]*maxLengthList
  s = raw_input()
  iList = map(int, s.split())
  var=len(iList)
  while var > count:
     var2=iList[count]-1	
     shopList[var2]=count+1
     count=count+1

  if shopList==iList:
     print "ambiguous" 
  else:
     print "not ambiguous"
