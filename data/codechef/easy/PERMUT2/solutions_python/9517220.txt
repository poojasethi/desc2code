
def is_ambiguous(lst):
    for idx, el in enumerate(lst):
        other = lst[el - 1]
        if idx + 1 != other:
            return False
    return True

num_chars = int(input())
while num_chars != 0:
    if is_ambiguous(map(int, raw_input().split())):
        print "ambiguous"
    else:
        print "not ambiguous"
    num_chars = int(input())
