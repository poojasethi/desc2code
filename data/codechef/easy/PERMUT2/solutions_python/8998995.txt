permutation = []
inverse = []

def inv(p):
    inv_list = [0] * len(p)
    for i, x in enumerate(p):
        inv_list[x-1] = i+1
    return inv_list

while True:
    t = int(input())
    if t == 0:
        break
    else:
        n = t
        permutation = list(map(int, raw_input().split()))
        inverse = inv(permutation)
        if permutation == inverse:
            print "ambiguous"
        else:
            print "not ambiguous"
