#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int i,n,x,cnt;
    while(1)
    {
        cin>>n;
        int arr[n+2];
        int arr1[n+2];
        if(n==0)
            return 0;
        for(i=1;i<=n;i++)
            cin>>arr[i];
        for(i=1;i<=n;i++)
        {
            x=arr[i];
            arr1[x]=i;//1 4 3 2
        }
        cnt=0;
        for(i=1;i<=n;i++)
        {
            if(arr[i]==arr1[i])
                cnt++;
        }
        if(cnt==n)
            cout<<"ambiguous"<<endl;
        else
            cout<<"not ambiguous"<<endl;
    }
    return 0;
}
