#include<iostream>

using namespace std; 

int main ()
{
	int N,i;
	
	//cout << "write the string length \n";
	cin >> N;

	while (N!=0)
	{
		int a[N],j=0;
		
		//cout << "write the string\n";
		for (i=0;i<N;i++)
		{
			cin >> a[i];
		}
		
		for (i=0;i<N;i++)
		{
			if (i+1 == a[a[i]-1])
			{
				j++;
				//cout << j;
			}
		}
		if (j == N)
		{
			cout << "ambiguous\n";
		}
		else 
		{
			cout << "not ambiguous\n";
		}

		cin >> N;		
	}
	
	return 0;
}
