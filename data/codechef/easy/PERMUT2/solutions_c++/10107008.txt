#include<stdio.h>

int main()
{
    int a[100000],n,i;
    scanf("%d",&n);
    while(n!=0)
    {
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        for(i=0;i<n;i++)
        {
            if(a[a[i]-1]!=i+1)
                break;
        }
        if(i==n)
            printf("ambiguous\n");
        else
            printf("not ambiguous\n");
        scanf("%d",&n);
    }
    return 0;
}
