#include<stdio.h>
#include<stdlib.h>

int main()
{
    int *a,n,i;
    scanf("%d",&n);
    while(n!=0)
    {
        a=(int *)malloc((n+1)*sizeof(int));
        for(i=1;i<=n;i++)
            scanf("%d",&a[i]);
        for(i=1;i<=n;i++)
        {
            if(a[a[i]]!=i)
                break;
        }
        if(i==n+1)
            printf("ambiguous\n");
        else
            printf("not ambiguous\n");
        scanf("%d",&n);
    }
    return 0;
}
