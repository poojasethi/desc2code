#include <iostream>
using namespace std;

int main() {
	// your code goes here
	while(true){
	    int n;
	    cin>>n;
	    int a[n],b[n];
	    if(n==0)
	    break;
	    for(int i=0;i<n;i++){
	        cin>>a[i];
	        b[a[i]-1]=i+1;
	    }
	    for(int i=0;i<n;i++){
	        if(a[i]!=b[i]){
	            cout<<"not ambiguous"<<endl;
	            break;
	        }
	        if(i==n-1){
	            cout<<"ambiguous"<<endl;
	        }
	    }
	    
	}
	return 0;
}
