#include<iostream>
using namespace std;
 
 
int main()
{
    int n,i,j,flag;
    one:
    cin>>n;
    if(n==0)
	return 0;
    else
    {
	int p[n];
	for(i=0;i<n;i++)
	    cin>>p[i];
	for(i=0;i<n;i++)
	{
	    j=p[i];
		if(p[j-1]==i+1)
	      {
		  flag=1;
	      }
		else
		{
		    flag=0;
		    break;
		}
	      }
	    }
	if(flag==1)
	    cout<<"ambiguous"<<"\n";
	else
	    cout<<"not ambiguous"<<"\n";
 
    goto one;
}
