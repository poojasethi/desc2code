#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
	int n;
	scanf("%d",&n);
	while(n){
		int a[n+1];
		int b[n+1];
		for(int i=1; i<=n; i++){
			scanf("%d", &a[i]);
			b[a[i]] = i;
		}
		
		int flag = 1;
		for(int j=1; j<=n; j++){
			if(a[j] != b[j]){
				flag = 0;
			}
		}

		if(flag){
			printf("ambiguous\n");
		}else{
			printf("not ambiguous\n");
		}
		scanf("%d",&n);
	}
	return 0;
}
