#include<bits/stdc++.h>
using namespace std;
int main()
{
    std::ios_base::sync_with_stdio(false);
	int j,n,flag=0;
	cin>>n;
	while(n){
		int arr[n],location[n+1];
	location[0]=0;

	for(j=0;j<n;j++)
	cin>>arr[j];
	for(j=0;j<n;j++)
	location[arr[j]]=j+1;
	for(j=0;j<n;j++){
	if(location[j+1]!=arr[j])
	flag=1;
	}
	if(flag)
	cout<<"not ambiguous\n";
	else
	cout<<"ambiguous\n";
	cin>>n;
	flag=0;
}
	return 0;
}
