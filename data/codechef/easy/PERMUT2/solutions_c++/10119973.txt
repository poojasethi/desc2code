#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a[100000];
    int b[100000];
    int t,f,i;
    scanf("%d",&t);
    while(t!=0)
    {
        for(i=0;i<t;i++)
        {
            scanf("%d",&a[i]);
            b[a[i]-1]=i+1;
        }
        f=0;
        for(i=0;i<t;i++)
        {
            if(a[i]!=b[i])
            {f=1; break;}
        }
        if(f==0)
        {
            printf("ambiguous\n");
        }
        else
        {
            printf("not ambiguous\n");
        }
        scanf("%d",&t);
    }
    return 0;
}
