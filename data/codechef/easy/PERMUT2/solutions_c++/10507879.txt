#include<iostream>
#include<cstring>
#include<cmath>
#include<vector>
#include<cstdio>
#include<cstdlib>

using namespace std;

int main()

{
    int t;
    scanf("%d",&t);
    while(t!=0)
    {
        int arr[t],flag=0;

        for(int i=0;i<t;i++)
            scanf("%d",&arr[i]);
        for(int i=0;i<t;i++)
        {
            if(arr[arr[i]-1]!=i+1)
                flag=1;
        }
        if(!flag)
            printf("ambiguous\n");
        else
            printf("not ambiguous\n");

            scanf("%d",&t);
    }
    return 0;
}
