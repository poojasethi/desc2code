#include<iostream>
#include<vector>
#include<string>

using namespace std;

int n;

string solve(vector<int> A)
{
    vector<int> B(100002);
    B[0]=0;
    int i;
    for(i=1;i<=n;++i)
        B[A[i]]=i;
    if(A==B)
        return "ambiguous\n";
    else
        return "not ambiguous\n";
}

int main()
{
    int i;
    cin>>n;
    while(n)
    {
        vector<int> A(100002);
        A[0]=0;
        for(i=1;i<=n;++i)
            cin>>A[i];

        cout<<solve(A);
        cin>>n;
    }
}
