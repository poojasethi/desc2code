#include<iostream>
using namespace std;

int main()
{
    int *a,n,i;
    cin>>n;
    while(n!=0)
    {
        a=new int [n+1];
      //  a[0]=0;
        for(i=1;i<=n;i++)
            cin>>a[i];
     //   for(i=1;i<=n;i++)
       //     cout<<a[i];
        for(i=1;i<=n;i++)
        {
            if(a[a[i]]!=i)
                break;
        }
        if(i==n+1)
            cout<<"ambiguous\n";
        else
            cout<<"not ambiguous\n";
        cin>>n;
    }
    return 0;
}
