#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
        int arr[100001], n;

        while(1){
                scanf("%d", &n);
                if(n==0) break;

                for(int i = 0; i<=n; i++)
                        arr[i] = 0;

                for(int i = 1; i<=n; i++){
                        scanf("%d", &arr[i]);
                }
                int i;
                for(i = 1; i<=n; i++){
                        if(arr[arr[i]] != i)
                                break;
                }
                if(i == n+1)
                        printf("ambiguous\n");
                else
                        printf("not ambiguous\n");
        }
    return 0;
}
