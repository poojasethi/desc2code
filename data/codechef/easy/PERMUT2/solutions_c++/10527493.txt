#include<iostream>
using namespace std;
long int n,i,a[1000000],flag;
int main()
{
	while(true)
	{
		for(i=0;i<1000000;i++)
		{
			a[i]='\0';	
		}	
		flag=0;
		cin>>n;
		if(n==0)
			break;
		for(i=0;i<n;i++)
		{
			cin>>a[i];	
		}	
		for(i=0;i<n;i++)
		{
			if(a[a[i]-1]!=i+1)	
				flag=1;
		}
		if(flag)
			cout<<"not ambiguous";
		else
			cout<<"ambiguous";
		cout<<endl;
	}
}
