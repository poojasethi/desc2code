#include<iostream>
using namespace std;
int main(){
	int T;float hs,cc,ts;
	cin>>T;
	int s[T];
	for(int i=0;i<T;i++){
		cin>>hs;cin>>cc;cin>>ts;
	    if((hs>50)&(cc<0.7)&(ts>5600)){
	    	s[i]=10;
	    }
	    else if((hs>50)&(cc<0.7)&(ts<=5600)){
	    	s[i]=9;
	    }
	    else if((hs<=50)&(cc<0.7)&(ts>5600)){
	    	s[i]=8;
	    }
	    else if((hs>50)&(cc>=0.7)&(ts>5600)){
	    	s[i]=7;
	    }
	    else if((hs<=50)&(cc>=0.7)&(ts<=5600)){
	    	s[i]=5;
	    }
	    else{
	    	s[i]=6;
	    }
	}
	for(int i=0;i<T;i++){
		cout<<s[i]<<endl;
	}
}
