#include <iostream>

using namespace std;

int main()
{
        int t, a, b, c;
        scanf("%d", &t);

        while(t--){
                scanf("%d%d%d", &a, &b,&c);

                if (a+b+c == 180)
                        printf("YES\n");
                else printf("NO\n");
        }
    return 0;
}
