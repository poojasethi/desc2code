n=input()
for i in range(n):
    t=input()
    a=map(int,raw_input().split())
    b=map(int,raw_input().split())
    max=a[0]*b[0]
    p=0
    for j in range(1,t):
        if ((a[j]*b[j])>max):
            max=a[j]*b[j]
            p=j
        elif (max==a[j]*b[j]) and (b[j] > b[p]):
                p=j
    print p+1
