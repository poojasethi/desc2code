No_tests = int(raw_input())

for i in range(No_tests):
    no_movies = int(raw_input())
    L = map(int, raw_input().split())
    R = map(int, raw_input().split())
    ind = 0 
    prod = 0 
    for j in range(no_movies):
        int_prod = L[j] * R[j]
        if int_prod > prod:
            prod = int_prod
            ind = j
        elif int_prod == prod:
            if R[j] > R[ind]:
                ind = j
        else:
            continue
    print (ind + 1)
