for i in range(int(raw_input())):
	n = int(raw_input())
	l = map(int, raw_input().split(' '))
	r = map(int, raw_input().split(' '))
	lr = []
	for counter in range(n):
		lr.append(l[counter] * r[counter])
	
	
	max_lr = max(lr)
	index_max = [i for i,x in enumerate(lr) if x == max_lr]

	maximum = 0
	for x in index_max:
		maximum = max(maximum,r[x])
		
	print r.index(maximum) + 1
