for i in range(t):
    n=input()
    l=map(int,raw_input().split())
    r=map(int,raw_input().split())
    max=l[0]*r[0]
    pos=1
    maxr=r[0]
    for j in range(1,n):
        w=l[j]*r[j]
        if w>max:
            max=w
            pos=j+1
            maxr=r[j]
            continue
        if w==max:
            if r[j]>maxr:
                pos=j+1
                maxr=r[j]
                continue
