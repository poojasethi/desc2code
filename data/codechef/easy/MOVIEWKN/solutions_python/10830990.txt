for t in xrange(int(raw_input())):
  n = int(raw_input())
  L = map(int, raw_input().split())
  R = map(int, raw_input().split())
  max_prod = 0
  max_Ri = 0
  min_i = n
  for i in xrange(n - 1, -1, -1):
    if L[i] * R[i] >= max_prod:
      max_prod = L[i] * R[i]
      if R[i] >= max_Ri:
        max_Ri = R[i]
        if i <= min_i:
          min_i = i
  print min_i + 1
