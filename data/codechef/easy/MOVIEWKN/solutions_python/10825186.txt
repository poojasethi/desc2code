testcase=int(input())
while(testcase>0):
	testcase-=1
	n=int(input())
	L=map(int,raw_input().split())
	R=map(int,raw_input().split())
	LR=[L[i]*R[i] for i in xrange(n)]
	maxi=max(LR)
	ind=[i for i in xrange(n) if LR[i]==maxi]
	if ind.__len__()==1:
		print(ind[0]+1)
	else:
		maxiR=max([R[i] for i in ind])
		secR=[i for i in ind if R[i]==maxiR]
		print(secR[0]+1)
