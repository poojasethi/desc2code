T = int(raw_input())
for i in range(T):
    n = int(raw_input())
    L = map(int,raw_input().split())
    R = map(int,raw_input().split())
    prod = 0
    ind = 0
    for j in range(n):
        if L[j]*R[j]>prod:
            ind = j
            prod = L[j]*R[j]
        else:
            if prod == L[j]*R[j] and R[j]>R[ind]:
                ind = j
    print (ind+1)
