#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int t,n,i,r,pos,p;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int a[n],b[n],max=0,pos=0;
        fflush(stdin);
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(i=0;i<n;i++)
        {
            cin>>b[i];
        }
        fflush(stdin);
        for(i=0;i<n;i++)
        {
            if((a[i]*b[i])>max)
                {
                    max=a[i]*b[i];
                    r=b[i];
                    pos=i+1;
                }
            else if((a[i]*b[i])==max)
                   if(b[i]>r)
                   { r=b[i];
                       pos=i+1;
                   }
        }

        cout<<pos<<"\n";
    }
    return 0;
}
