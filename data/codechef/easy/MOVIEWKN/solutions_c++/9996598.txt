#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <vector>
#include <cstdio>
 
using namespace std;
 
 
int main(){
	int t,n;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		int L[n],R[n];
		for(int i = 0;i < n;i++) scanf("%d",&L[i]);
		for(int i = 0;i < n;i++) scanf("%d",&R[i]);
		int maxi = -1;
		int pos = -1;
		int auxr = -1;
		for(int i = 0;i < n;i++){
			if(L[i]*R[i] > maxi){
				maxi = L[i]*R[i];
				auxr = R[i];
				pos = i; 
			}
			else if(L[i]*R[i] == maxi){
				if(R[i] > auxr){
					auxr = R[i];
					pos = i;
				}
			}
		}
		printf("%d\n",pos + 1);
	}
	return 0;
}
