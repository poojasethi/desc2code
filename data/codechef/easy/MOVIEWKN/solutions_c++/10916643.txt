#include <iostream>

using namespace std;

int main()
{
        int t, n;
        int l[101];
        int r[101];

        cin>>t;

        while(t--){
                cin>>n;

                for(int i=1; i<=n; i++){
                        cin>>l[i];
                }
                for(int i=1; i<=n; i++){
                        cin>>r[i];
                }

                int max_prod=0, rating=0, index=0;

                for(int i=1; i<=n; i++){
                        if(l[i]*r[i]>max_prod){
                                max_prod=l[i]*r[i];
                                rating = r[i];
                                index = i;
                        }
                        else if(l[i]*r[i]==max_prod && r[i]>rating){
                                rating = r[i];
                                index = i;
                        }
                }
                cout<<index<<endl;

        }
    return 0;
}
