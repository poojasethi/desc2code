#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t, n, l[100], r[100], max = -1, vt = -1;
	vector<long> v;
	cin >> t;
	for(int i = 0 ; i < t; i++)
	{
		cin >> n;
		max = -1; 
		vt = -1;
		for(int j = 1 ; j <= n ; j++)
		{
			cin >> l[j];
		}
		for(int j = 1 ; j <= n ; j++)
		{
			cin >> r[j];
		}
		for(int j = 1 ; j <= n ; j++)
		{
			if(max < l[j]*r[j])
			{
				vt = j;
				max = l[j]*r[j];
			}
			if(max == l[j]*r[j])
			{
				if(r[j] > r[vt])
				{
					vt = j;
				}
			}
		}
		v.push_back(vt);
	}
	for(vector<long>::iterator it = v.begin() ; it != v.end() ; it++)
		cout << *it << endl;
	return 0;
}
