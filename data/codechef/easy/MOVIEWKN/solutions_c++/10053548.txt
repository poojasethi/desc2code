#include<iostream>
using namespace std;
int main()
{
int T;
cin>>T;
int *n=new int[T];
int *Li;
int *Ri;
int j;
int *LiRi;
int max,ans;
for(int i=0;i<T;i++)
{
cin>>n[i];
Li=new int[n[i]];
Ri=new int[n[i]];
LiRi=new int[n[i]];

for(j=0;j<n[i];j++)
{
cin>>Li[j];
}

for(j=0;j<n[i];j++)
{
cin>>Ri[j];
LiRi[j]=Li[j]*Ri[j];
}
max=LiRi[0];
ans=1;
for(j=1;j<n[i];j++)
{
if(LiRi[j]>max)
{
max=LiRi[j];
ans=j+1;
}
else if(LiRi[j]==max)
{
if(Ri[j]>Ri[ans-1])
ans=j+1;
}
}
cout<<ans<<"\n";
}
 
 
return 0;
}
