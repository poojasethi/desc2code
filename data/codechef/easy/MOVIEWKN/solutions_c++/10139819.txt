#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,j,ans,l[105],r[105],p,max;
	cin>>t;
	while(t--)
	{	p=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>l[i];
		}
		for(i=0;i<n;i++)
		{
			cin>>r[i];
			max=l[i]*r[i];
			if(max>p)
			p=max;
		}
		max=0;
		for(i=0;i<n;i++)
		{
			if((l[i]*r[i]==p)&&(r[i]>max))
			{
				max=r[i];
				ans=i;
			}
		}
		cout<<ans+1<<endl;
	}
	
	return 0;
}
