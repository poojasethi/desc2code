#include<bits/stdc++.h>
using namespace std;

int main(){
    int L[101],R[101];
    int n,t;
    cin>>t;
    while(t--){
        cin>>n;
        for(int i=0;i<n;i++){
                cin>>L[i];
        }
        for(int i=0;i<n;i++){
                cin>>R[i];
        }
        int resI = 1;
        int pdtCheck = L[0]*R[0];
        int rCheck=R[0];
        for(int i=1;i<n;i++){
                int thisPdt = L[i]*R[i];
                if(thisPdt>pdtCheck){
                    resI = i+1;
                    pdtCheck = thisPdt;
                    rCheck = R[i];
                }
                else if(thisPdt==pdtCheck){
                    if(R[i]>rCheck){
                        resI = i+1;
                        rCheck = R[i];
                    }
                }
        }      
        cout<<resI<<"\n";
    }
    return 0;
    }
