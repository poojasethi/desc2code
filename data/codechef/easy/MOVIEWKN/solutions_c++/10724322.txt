#include <iostream>
using namespace std;

int main() {
	int t,l[100]={0},r[100]={0},i,max,max_r,index,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>l[i];
		for(i=0;i<n;i++)
		cin>>r[i];
		max=l[0]*r[0];
		index=0,max_r=r[0];//rememberto initialise them
		for(i=1;i<n;i++)
		{
			if(l[i]*r[i]>max)
			{
				max_r=r[i];
				max=l[i]*r[i];//max value should be updated
				index=i;
			}
			else if(l[i]*r[i]==max)
			{
				if(max_r<r[i])
				{
					max_r=r[i];
					index=i;
				}	
			}
		}
		cout<<index+1<<endl;
		}
	return 0;
}
