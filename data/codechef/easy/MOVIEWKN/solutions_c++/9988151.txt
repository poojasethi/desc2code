#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
//#include<>

using namespace std;

int main(){
	
	int t;
	scanf("%d",&t);
	
	while(t--){
		int n;
		scanf("%d",&n);
		
		int l[200]={0};
		int r[200]={0};
		int arr[200]={0};
		for(int i=1;i<=n;i++){
			scanf("%d",&l[i]);
		}
		for(int i=1;i<=n;i++){
			scanf("%d",&r[i]);
			arr[i-1]=l[i]*r[i];
		}
		sort(arr,arr+n);
		int m=arr[n-1];
		int maxofR=0;
		for(int i=0;i<=n;i++)
		{
			if(l[i]*r[i]==m){
				if(maxofR<r[i])
				maxofR=r[i];
			}
		}
		int i=1;
		for(i=1;i<=n;i++){
			if(maxofR==r[i])
			break;
		}
	printf("%d",i);	
	printf("\n");
	}
	return 0;
}
