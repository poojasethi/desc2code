#include <iostream>
using namespace std;

int main() {
	int t,n,*l,*r,i,j,max,answer;
	cin>>t;
	while(t--)
	{
		cin>>n;
		l=new int[n];
		r=new int[n];
		for(i=0;i<n;i++)
		cin>>l[i];
		for(i=0;i<n;i++)
		cin>>r[i];
		max=l[0]*r[0];
		for(i=1;i<n;i++)
		{
			if(max<=l[i]*r[i])
				max=l[i]*r[i];
		}	
		int max_rating=0;
		for(j=0;j<n;j++)
		{
			if(r[j]*l[j]==max)
			{
				if(r[j]>max_rating)
				{
					max_rating=r[j];
					answer=j;
				}	
			}
		}	
		cout<<answer+1<<endl;
	}	
	return 0;
}
