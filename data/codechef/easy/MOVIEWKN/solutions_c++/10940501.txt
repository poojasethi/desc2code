#include <iostream>
using namespace std;

int main() {
	// your code goes here 
	int t;
	cin>>t;
	while(t--)
	 {
	     int n;
	     cin>>n;
	     int a[n];
	     int b[n];
	     for(int i=0;i<n;i++)
	      {
	          cin>>a[i];
	      }
	      
	      for(int i=0;i<n;i++)
	      {
	          cin>>b[i];
	      }
	      
	      int c[n];
	      
	      for(int i=0;i<n;i++)
	       {
	           c[i]=a[i]*b[i];
	       }
	      int max=-1;
	      
	      int ans;
	      
	      for(int i=0;i<n;i++)
	       {
	           if(c[i]>max)
	            {
	                max=c[i];
	                ans=i+1;
	            }
	           else if(c[i]==max)
	             {
	                 if(b[i]>b[ans-1])
	                  {
	                      ans=i+1;
	                  }
	             }
	         }
	      
	      
	      cout<<ans<<endl;
	      
	      
	      
	 }
	
	
	return 0;
}
