#include<bits/stdc++.h>
using namespace std;
int rate(int r[],int n)
{
    int maxx=0,index;
    for(int i=1;i<=n;i++)
    {
        if(r[i]>maxx){
           maxx=r[i];
           index=i;}
    }
    return index;
}
int main()
{
  int t,i;
  cin>>t;
  while(t--)
  {
      int n;
      cin>>n;
      int l[n],r[n];
      for(i=1;i<=n;i++)
        cin>>l[i];
      for(i=1;i<=n;i++)
        cin>>r[i];
        int maxx=0,index,m,x=0;
        for(i=1;i<=n;i++)
        {
            m=l[i]*r[i];
            if(m>=maxx){
                   if(m==maxx)
                   {
                    index=rate(r,i);
                    continue;
                   }
                maxx=m; index=i;
            }
        }
        cout<<index<<endl;
  }
  return 0;
}
