#include <iostream>
using namespace std;
#include <vector>
#include <math.h>
#define ll long long


int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll n,q,i;
        cin>>n;
        vector<ll> x,y;
        for(i=0;i<n;i++)
        {   cin>>q;
            x.push_back(q);
        }
        for(i=0;i<n;i++)
        {   cin>>q;
            y.push_back(q);
        }
        ll s,m=0,ans;
        for(i=0;i<n;i++)
        {   if(x[i]*y[i]>m)
            {   m=x[i]*y[i];
                s=y[i];
                ans=i;
            }
            else if(x[i]*y[i]==m)
            {   if(y[i]>s)
                {    s=y[i];
                    ans=i;
                }
            }
        }
        cout<<ans+1<<"\n";
    }
    return 0;
}
