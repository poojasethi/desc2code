#include<iostream>
#define cout std::cout
#define cin std::cin
int main()
{
    int t;
    cin>>t;
    while(t--)
    {

        int n,i,max=0,in,rn;
        cin>>n;
        int l[n],r[n],c[n];
        for(i=0;i<n;i++)cin>>l[i];
        for(i=0;i<n;i++)cin>>r[i];

        for(i=0;i<n;i++)
    {
        c[i]=l[i]*r[i];
        if(c[i]>max)
        {
            max=c[i];
            rn=r[i];
            in=i;
        }
        else if(c[i]==max)
        {

            if(r[i]>rn)
            {
                 max=c[i];
                 rn=r[i];
                 in=i;

            }
            else if(r[i]==rn)
            {

                in=in;
            }
        }
    }
   cout<<"\n"<<in+1;
}
    return 0;
}
