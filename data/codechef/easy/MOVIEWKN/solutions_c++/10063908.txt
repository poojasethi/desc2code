#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,t,n,max,ans;
	int*a,*b,*c;
	cin>>t;
	while(t--)
	{
		cin>>n;
		a=new int[n];
		b=new int[n];
		c=new int[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		cin>>b[i];
		for(i=0;i<n;i++)
		c[i]=a[i] * b[i];
		max=c[0];
		ans=0;
		for(i=0;i<n;i++)
		{
			if(c[i]>max)
			{
			max=c[i];
			ans=i;
			}
		}
		for(i=0;i<n;i++)
		{
			if(c[i]==max)
			{
				if(b[i]>b[ans])
				ans=i;
			}
		}
		delete[] a;
		delete[] b;
		delete[] c;
		cout<<(ans+1)<<endl;
	}
}
