#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
       int n,i,p=0,w=0;
       cin>>n;
       int l[n],r[n];

        for(i=0;i<n;i++)
        cin>>l[i];

       for(i=0;i<n;i++)
        cin>>r[i];

    w=l[0]*r[0];

        for(i=1;i<n;i++)
        {
            if(l[i]*r[i]>w)
            {
                w=l[i]*r[i];
                p=i;

            }

          if(l[i]*r[i]==w && r[i]>r[p])
          {
              p=i;
          }



        }

   cout<<++p<<endl;

   }
    return 0;
}
