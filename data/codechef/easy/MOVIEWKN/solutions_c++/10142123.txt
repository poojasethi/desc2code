#include<bits/stdc++.h>
#include<string>
#define ll long long int
using namespace std;
int main()
{
    int t,n,l[100],r[100],mlr,mr;
    cin >> t;
    while(t--)
    {
        cin >> n;
        for(int i=0;i<n;i++)
        {
            cin >> l[i];
        }
        for(int i=0;i<n;i++)
        {
            cin >> r[i];
        }
        mlr=0;
        for(int i=0;i<n;i++)
        {
            if(l[i]*r[i] > mlr)
            {
                mlr=l[i]*r[i];
            }
        }
        mr=0;
        for(int i=0;i<n;i++)
        {
            if(l[i]*r[i] == mlr)
            {
                if(r[i]>mr)
                    mr=r[i];
            }
        }
        for(int i=0;i<n;i++)
        {
            if(((r[i]*l[i])==mlr) && (r[i]==mr) )
            {
                cout << i+1 << "\n";
                break;
                
            }
        }
       
    }
}
