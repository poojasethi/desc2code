#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
	int t;
	cin >> t;
	while (t--) {
		int n;
		cin >> n;
		int l[n + 1], r[n + 1];
		for (int i = 0; i < n; ++i)
		{
			cin >> l[i];
		}
		for (int i = 0; i < n; ++i)
		{
			cin >> r[i];
		}
		int max_prod = -1;
		int max_r = -1;
		int min_index = -1;
		for (int i = 0; i < n; ++i)
		{
			if ( (l[i] * r[i]) > max_prod) {
				max_prod = l[i] * r[i];
				max_r = r[i];
				min_index = i;
			} else if (((l[i] * r[i]) == max_prod)) {
				if (r[i] > max_r) {
					max_prod = l[i] * r[i];
					max_r = r[i];
					min_index = i;
				}
			}
		}

		cout << min_index + 1 << endl;
	}	
}
