#include<iostream>
using namespace std ;

int main()
{
    int test_case ,n ;
    cin>>test_case ;
    while(test_case--){
        cin>>n ;
        int L[n], R[n], prod=0, ind=0 ;
        for(int i=0; i<n; i++)
            cin>>L[i] ;
        for(int i=0; i<n; i++)
            cin>>R[i] ;
        for(int i=0; i<n; i++)
            if(L[i]*R[i]>=prod && R[i]>R[ind]){
                prod = L[i]*R[i] ;
                ind = i ;
            }
        cout<<ind+1<<endl ;
    }
}
