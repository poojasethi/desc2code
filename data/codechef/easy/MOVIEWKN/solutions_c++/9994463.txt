#include <cstdio>
#include <algorithm>
#include <iostream>

using namespace std;

struct node
{
	int index,l,r,mul;
}k[100];

bool comp(struct node a, struct node b)
{
	return (a.mul==b.mul)?((a.r==b.r)?(a.index<b.index):(a.r>b.r)):(a.mul>b.mul);
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t--) 
	{
		int n,l[100],r[100];
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			scanf("%d",&k[i].l);
		for(int i=0;i<n;i++)
			scanf("%d",&k[i].r);
		for(int i=0;i<n;i++)
		{
			k[i].index = i;
			k[i].mul = k[i].l*k[i].r;
		}

		sort(k,k+n,comp);

		printf("%d\n",k[0].index+1 );
	}
	return 0;
}
