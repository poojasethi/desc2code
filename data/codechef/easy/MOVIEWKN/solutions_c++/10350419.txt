#include<bits/stdc++.h>

using namespace std;

int main(void)
{
	 int t; scanf("%d",&t);
	 while(t--)
	 {
	 	int n; scanf("%d",&n);
	 	int l[n],r[n];
	 	for(int i=0;i<n;i++)
	 	{
	 		scanf("%d",&l[i]);
	 		
	 	}
	 	for(int i=0;i<n;i++)
	 	{
	 		scanf("%d",&r[i]);
	 	}
	 	int mrl=-1;
	 	for(int i=0;i<n;i++)
	 	   if(r[i]*l[i]>mrl)
	 	     mrl=r[i]*l[i];
	 	
	 	int rval=-1,ind=0;
	 	for(int i=0;i<n;i++)
	 	{
	 		if(r[i]*l[i]==mrl)
	 		{
	 			if(r[i]>rval)
	 			{
	 				rval=r[i];
	 				ind=i+1;
	 			}
	 		}
	 	}
	 	printf("%d\n",ind);
	 }
	 
}
