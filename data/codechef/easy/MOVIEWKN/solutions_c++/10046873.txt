#include <stdio.h>
#include <math.h>
#include<iostream>
using namespace std;
int main ()
{
    int t;
    int i,N,maxLR;
    int a[2][101];
    int LR[101];
    cin>>t;
    int index=-1;
    while(t--)
    {
        cin>>N;
        for(i=0;i<N;i++)
            cin>>a[0][i];
        for(i=0;i<N;i++)
            cin>>a[1][i];
        maxLR= LR[0]=a[0][0] * a[1][0];
        for(i=1;i<N;i++)
        {
            LR[i]= a[0][i] * a[1][i];
            
            if(LR[i]>maxLR)
                maxLR= LR[i];
        }
        for(i=0;i<N;i++)
        {
            if(LR[i]== maxLR || index==-1)
            {
                if(a[1][i] >a[1][index])
                    index=i;
                else if(a[1][i]== a[1][index])
                {
                    if(i<index)
                        index=i;
                }
            }
        }
        cout<<index+1<<endl;
    }
     return 0;
}
