#include <stdio.h>

const int MAX = 110;
int n, test, l[MAX], r[MAX], res;

int main(){
    //freopen("in.txt", "r", stdin);
    scanf("%d", &test);
    while (test--){
        scanf("%d", &n);
        for (int i = 1; i <= n; ++i)
            scanf("%d", l + i);
        for (int i = 1; i <= n; ++i)
            scanf("%d", r + i);

        res = 1;
        for (int i = 2; i <= n; ++i)
        if (l[i] * r[i] > l[res] * r[res] || (l[i] * r[i] == l[res] * r[res] && r[i] > r[res]))
            res = i;
        printf("%d\n", res);
    }
    return 0;
}
