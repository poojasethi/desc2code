#include<bits/stdc++.h>
#include <iostream>
using namespace std;
 
int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	while(t--)
	{
	    int n,m;
	    cin>>n>>m;
	    string v[n];
        for(int i=0;i<n;i++)
	    {
            cin>>v[i];
	    }
	    int collisions=0;
	    for(int i=0;i<n;i++)
	    {
	        for(int j=i+1;j<n;j++)
	        {
	            for(int k=0;k<m;k++)
	            {
	                if(v[i][k]=='1' && v[j][k]=='1')
	                    collisions+=1;
	            }
	        }
	    }
	    cout<<collisions<<endl;
	}
	return 0;
}
