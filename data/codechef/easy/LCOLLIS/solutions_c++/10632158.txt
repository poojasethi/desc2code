#include<iostream>
using namespace std;
int main()
{
	long long int t,k,i,j,n,m,count,ans;
	char a[10][10];
	cin>>t;
	for(k=0;k<t;k++)
	{
		ans=0;
		cin>>n>>m;
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
				cin>>a[i][j];
		}
		for(j=0;j<m;j++)
		{
			count=0;
			for(i=0;i<n;i++)
			{
				if(a[i][j]=='1')
					count++;
			}
			ans=ans+((count*(count-1))/2);
		}
		cout<<ans<<endl;
	}
	
}
