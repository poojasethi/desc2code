rounds=int(raw_input())
p1=p2=0
maxlead=0
while rounds>0:
    rounds=rounds-1
    player1,player2=raw_input().split()
    p1=p1+int(player1)
    p2=p2+int(player2)
    if p1>p2:
        leadsc=p1-p2
        leadp=1
        if leadsc>maxlead:
            maxlead=leadsc
            leadp=1
    else:
        leadsc=p2-p1
        
        if leadsc>maxlead:
            maxlead=leadsc
            leadp=2

print leadp,maxlead
