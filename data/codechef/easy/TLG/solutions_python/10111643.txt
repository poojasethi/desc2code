tc = int(raw_input())
ascore, bscore, winner, answer = 0, 0, 0, 0
while tc > 0:
    tc = tc - 1
    a, b = map(int, raw_input().split())
    ascore = ascore + a
    bscore = bscore + b
    if abs  ( ascore - bscore ) > answer:
        answer = abs ( ascore - bscore )
        winner = 1 if ascore > bscore else 2
print winner, answer
