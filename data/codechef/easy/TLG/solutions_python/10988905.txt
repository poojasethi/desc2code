n = int(input())
score1 = 0
score2 = 0
winner = 0
max_lead = 0
for i in range(n):
    line = raw_input().split()
    score1 += int(line[0])
    score2 += int(line[1])
    winner = abs(score1 - score2)
    if winner > max_lead:
        max_lead = winner
        if score1 > score2:
            contestant = 1
        else:
            contestant = 2
print ("%d %d" %(contestant,max_lead))
