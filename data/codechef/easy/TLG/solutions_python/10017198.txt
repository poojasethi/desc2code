t = int(raw_input())
lead=[]
leader=[]
player1 = 0
player2 = 0
while t:
    num = raw_input().split()
    player1 += int(num[0])
    player2 += int(num[1])
    
    leader.append(1 if(player1>player2) else 2)
    lead.append(abs(player1-player2))
    t -= 1
max_lead = max(lead)
winner = leader[lead.index(max_lead)]
print winner,max_lead
