N=int(raw_input())
ts_p1=ts_p2=0
lead_score=[]
for element in range(N):
    p1_score, p2_score=map(int, raw_input().split())
    ts_p1+=p1_score
    ts_p2+=p2_score
    lead_score.append(ts_p1-ts_p2)
max_p1_lead=max(lead_score)
max_p2_lead=min(lead_score)
if max_p1_lead>abs(max_p2_lead):
    W=1
    L=max_p1_lead
else:
    W=2
    L=abs(max_p2_lead)
print W, L
