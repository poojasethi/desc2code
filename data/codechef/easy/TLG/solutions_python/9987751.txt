#!/usr/bin/python
#The Lead Game
#https://www.codechef.com/problems/TLG
t=input()
p1=0
p2=0
ll1=0
ll2=0
while t!=0:
	t-=1
	a,b = map(int,raw_input().split())
	p1+=a
	p2+=b
	# print p1, p2
	if p1>p2:
		lead1=p1-p2
		if lead1>ll1:
			ll1=lead1
	else:
		lead2=p2-p1
		if lead2>ll2:
			ll2=lead2
	# print lead1, lead2

if max(ll1,ll2)==ll1:
	print "1 "+str(ll1)
else:
	print "2 "+str(ll2)
