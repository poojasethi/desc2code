p1=p2=0
maxlead = 0
for i in range(int(raw_input())):
    a,b = map(int,raw_input().split())
    p1+=a
    p2+=b
    if p1>p2:
        lead = p1-p2
        if lead>maxlead:
            maxlead = lead
            maxp = 1
    else:
        lead = p2-p1
        if lead>maxlead:
            maxlead = lead
            maxp = 2
print maxp,maxlead
