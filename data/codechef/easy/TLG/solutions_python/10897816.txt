x = int(raw_input())
n = 0
suma = 0
sumb = 0
diff1 = 0
diff2 = 0
tempdiff1 = 0
tempdiff2 = 0
while (n<x):
    a,b = (int(x) for x in raw_input('').split(' '))
    suma = suma + a
    sumb = sumb + b
    if suma > sumb :
        diff1 = suma - sumb
    else:
        diff2 = sumb - suma
    if(diff1 > tempdiff1):
        tempdiff1 = diff1
    if(diff2 > tempdiff2):
        tempdiff2 = diff2
    n=n+1
if tempdiff1 > tempdiff2 :
    print "1" ,tempdiff1
else :
    print "2" ,tempdiff2
