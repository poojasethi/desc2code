#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,i;
	cin>>n;
	int p1[n],p2[n],p[n],l[n],winner,lead,a=0,b=0;
	for(i=0;i<n;i++){
		cin>>p1[i];
		cin>>p2[i];
	     
	    a+=p1[i];                            // we input score in each round and add to total score and then calculate the lead
	    b+=p2[i];
		if(a>b){
		l[i]=a-b;
		p[i]=1;}
		else
		{
		l[i]=b-a;
		p[i]=2;
	    }
	}
    lead=l[0];
    winner=p[0];
    for(i=1;i<n;i++){
    	if(l[i]>lead){
    		lead=l[i];
    		winner=p[i];
		}
	}
	cout<<winner<<" "<<lead<<endl;
     return 0;
}
