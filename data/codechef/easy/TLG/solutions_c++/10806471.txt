#include <iostream>

using namespace std;

int main()
{
    int n,lead=0,score1,score2,flag=0,lead1=0,lead2=0;
    cin>>n;
    while(n--)
    {
        cin>>score1>>score2;
        lead1=lead1+score1;
        lead2=lead2+score2;
        if(lead1>lead2)
        {
            if((lead1-lead2)>lead)
            {
                lead=lead1-lead2;
                flag=1;
            }
        }
        else
        {
             if((lead2-lead1)>lead)
             {
                lead=lead2-lead1;
                flag=2;
             }
        }
    }
    cout<<flag<<" "<<lead<<endl;
    return 0;
}
