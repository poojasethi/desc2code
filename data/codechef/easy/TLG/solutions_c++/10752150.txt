#include<iostream>
#include<bits/stdc++.h>

using namespace std;

int main(){

int n;
scanf("%d",&n);

int a,b,i=0,p=0,l=0,c=0,d=0;

for(int i=0;i<n;i++){

	scanf("%d",&a);
	scanf("%d",&b);
	
	c+=a;
	d+=b;
	
	
	if((c-d) >=0)
	{
	if((c-d) >l){
	
		l=c-d;
		p=1;
	
	}
	
	}
	else{
		
		if((d-c) > l){
		l=d-c;
		p=2;
		}
		
	}
	
	
}




printf("%d %d",p,l);
return 0;

}
