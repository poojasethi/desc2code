#include <iostream>
#include <stdio.h>
//#include <climits>
using namespace std;

int main()
{
	long int n,lead=0,win,c,d,a=0,b=0;
	scanf("%ld",&n);
	while(n--)
	{
		scanf("%ld%ld",&c,&d);
		a+=c;
		b+=d;

		if((a-b)>=0)
		{
			if((a-b)>lead)
			{
				lead=a-b;
				win=1;
			}
		}
		else
		{
			if((b-a)>lead)
			{
				lead=b-a;
				win=2;
			}
		}
	}
	
	cout<<win<<" "<<lead<<endl;

	return 0;
}
