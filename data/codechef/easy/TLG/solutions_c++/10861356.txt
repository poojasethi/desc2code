#include <iostream>


using namespace std;

int main() {

	int n,p1,p2,a1=0,b=0;
	cin>>n;

	int a[n][2];

	for(int i=0;i<n;i++){
        cin>>p1>>p2;
        a1=a1+p1;
        b=b+p2;
        if(a1>b){
            a[i][0]=1;
            a[i][1]=a1-b;
        }
        else if(a1<b){
            a[i][0]=2;
            a[i][1]=b-a1;
        }
	}
	int maxi=0;
	for(int i=0;i<n;i++){
        if(a[i][1]>a[maxi][1]){
            maxi=i;
        }
	}

	cout<<a[maxi][0]<<" "<<a[maxi][1];

	return 0;
}
