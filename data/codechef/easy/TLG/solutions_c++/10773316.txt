#include<iostream>




int main()
{
	int a = 0;
	int b = 0;
	int w = 0;
	int l = 0;
	int temp1 = 0;
	int temp2 = 0;
	long long int size;
	std::cin >> size;
	int x;
	 int** array = new int*[size];
	for (int i = 0; i<size; i++) {
		array[i] = new int[2];
	}

	for ( x = 0; x < size; x++)
	{
	


		std::cin >> array[x][0];
		std::cin >> array[x][1];
		

		a = a + array[x][0];
		b = b + array[x][1];

		


		if (b > a)
			w = b - a;
		if (w>temp2)
			temp2 = w;
		else
			l = a - b;
		if (l > temp1)
			temp1 = l;

	


	}

	if (temp1 >= temp2)
		std::cout <<"1"<<" "<< temp1;
	else
		std::cout <<"2"<<" "<< temp2;





}
