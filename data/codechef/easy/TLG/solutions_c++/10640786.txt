#include <iostream>
using namespace std;

int main(int argc, const char * argv[]) {
    int n,m,s1=0,s2=0,t,lead=0,e=0;
    cin>>t;
    
   while(t!=0)
    {
        cin>>n>>m;
        s1=s1+n;
        s2=s2+m;
        if(lead<s1-s2)
        {
            
            lead=s1-s2;
            e=1;
        }
        else
        {
            if(lead<s2-s1)
            {
                lead=s2-s1;
                e=2;
            }
        }
        t--;
   }
    cout<<e<<"\t"<<lead<<"\t";
    
    
    return 0;
}
