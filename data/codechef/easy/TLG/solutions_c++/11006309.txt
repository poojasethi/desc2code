#include <iostream>
using namespace std;

int main() {
	int n, i, s, t;
    cin>>n;
    int a[n], b[n];
	for(i=0;i<n;i++)
	{cin>>a[i]>>b[i];}
	int w,l;
	s=0;t=0,l=0;
	for(i=0;i<n;i++)
	{
	    s=s+a[i];
	    t=t+b[i];
	    if(s>t && (s-t)>l)
	    {   l=s-t;
	        w=1;
	    }
	    else if ((t-s)>l)
	    {
	        l=t-s;
	        w=2;
	    }
	}cout<<w<<" "<<l;
	return 0;
}
