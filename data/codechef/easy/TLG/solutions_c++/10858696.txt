#include <iostream>
using namespace std;

int main() 
{
	int N; // number of rounds
	int p1=0, p2=0, max=0, winner, t1, t2;

	cin >> N;
	for (int i = 0; i < N; i++)
	{
		cin >> t1;
		cin >> t2;
		p1+=t1;
		p2+=t2;
		if (p1 - p2 > max)
		{
			max = p1 - p2;
			winner = 1;
		}
		else if (p2 - p1 > max)
		{
			max = p2 - p1;
			winner = 2;
		}
	}
	cout << winner << " " << max;
	return 0;
}
