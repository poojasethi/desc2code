#include <iostream>
using namespace std;
#include<stdio.h>

int main()
{
	int i,j,x=0,y=0,nor,max1,max2;
	scanf("%d",&nor);
	max1=0;
	max2=0;
	int p,q;
	for(i=0;i<nor;i++)
   {
   	scanf("%d%d",&p,&q);
   	x+=p;
   	y+=q;
   	//cout<<x<<" "<<y<<endl;
   	if(x>y)
   	{
   		if((x-y)>max1)
   		{
   		max1=x-y;
   //	cout<<1<<endl;
   		}
   	}
   	else
   	  {
   	  	if((y-x)>max2)
   	  	{
   	  	max2=y-x;
   	  	//cout<<2<<endl;
   	  	}
   	  }
   	 // cout<<max1<<" "<<max2<<endl;
   }
   
   if(max1>max2)
   cout<<1<<" "<<max1<<endl;
   else
   cout<<2<<" "<<max2<<endl;
    	  return 0;
}
