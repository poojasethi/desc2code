#include<iostream>
using namespace std;
int main()
{   int test,a,b,lead=0,temp,winner=0,score1=0,score2=0;
    cin>>test;
    for(int i=0;i<test;i++)
    {
        cin>>a;
        cin>>b;
        score1=score1+a;
        score2=score2+b;
        if(score1>score2)
        {   temp=score1-score2;
            if(temp>lead)
            {winner=1;
            lead=temp;
            }
        }
        else if(score2>score1)
        {   temp=score2-score1;
            if(temp>lead)
            {
            winner=2;
            lead=temp;
            }
        }
    }
    cout<<winner<<" "<<lead<<endl;
    return 0;

}
