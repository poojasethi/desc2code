    #include <iostream>
    #include <climits>
    using namespace std;
     
    int main()
    {
    	int n, a, b, max_a = INT_MIN, max_b = INT_MIN, sum_a, sum_b;
    	sum_a = sum_b = 0;
    	cin>>n;
    	while(n--)
    	{
    		cin>>a>>b;
    		sum_a += a;
    		sum_b += b;
    		if(sum_a > sum_b)
    		{
    			if(sum_a-sum_b > max_a)
    			{
    				max_a = sum_a-sum_b;
    			}
    		}
    		else
    		{
    			if(sum_b-sum_a > max_b)
    			{
    				max_b = sum_b-sum_a;
    			}
    		}
    	}
    	
    	if(max_a > max_b)
    	{
    		cout<<"1 "<<max_a<<endl;
    	}
    	else
    	{
    		cout<<"2 "<<max_b<<endl;
    	}
    return 0;
    }
