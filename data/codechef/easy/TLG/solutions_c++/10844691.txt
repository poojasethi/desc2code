#include<iostream>
#include<cmath>
#include<algorithm>


using namespace std;

int main()
{
	int t,s1,s2,w=0,l=0,p1=0,p2=0;
	cin>>t;

	while(t--){
	
		cin>>s1>>s2;
	
		p1=p1+s1;
		p2=p2+s2;
	        if(abs(p1-p2)>l){

	       		if(p1-p2>0){
				l=p1-p2;
				w=1;
			}else{
				l=p2-p1;
				w=2;
			}
           	 }
	}
		cout<<w<<" "<<l<<endl;


return 0;

}
