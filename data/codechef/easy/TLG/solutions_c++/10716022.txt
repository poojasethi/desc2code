#include<iostream>
#include<math.h>
using namespace std;

int main()
{

 int p1[10001];
 int p2[10001];
 int lead[10001];
 int rounds , temp[10001];
 cin>>rounds;
 int chk;
 int sum1 = 0;
 int sum2 = 0;
  for(int i = 0 ; i<rounds ; i++)
  {
   cin>>p1[i]>>p2[i];
  }

  int sum = 0;

  for(int i = 0 ; i<rounds ; i++)
  {
     sum1 = sum1+p1[i];
     sum2 = sum2+p2[i];
    if(sum1>sum2)
    {
     lead[i] = sum1 - sum2;
     temp[i] = 1;
    }
    else
    {
      lead[i] = sum2 - sum1;
      temp[i] = 2;
    }

  }
  int max = -1;

  for(int i = 0 ; i<rounds ; i++)
  {

    if(lead[i]>max)
   { max =lead[i];
     chk = i;
     }
  }

  cout<<temp[chk] <<" " << max<<endl;
}
