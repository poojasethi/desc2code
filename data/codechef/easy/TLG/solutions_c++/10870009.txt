#include<cstdio>
using namespace std;
int main()
{
    int N,S=0,s=0,T=0,t=0,W,L=0,i;
    scanf("%d",&N);
    for (i=0;i<N;i++)
    {
        scanf("%d %d",&s,&t);
        S+=s;T+=t;
        if (S-T>L)
        {
            L=S-T;
            W=1;
        }
        else if (T-S>L)
        {
            L=T-S;
            W=2;
        }
    }
    printf("%d %d\n",W,L);
    return 0;
}
