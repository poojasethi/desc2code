#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t,sum1=0,sum2=0,win,lead=0;
	cin>>t;
	for(int i =0;i<t;i++){
	    int x1,x2,temp;
	    cin>>x1>>x2;
	    sum1+=x1;
	    sum2+=x2;
	    if(sum1>sum2){
	        temp = sum1-sum2;
	        if(temp>lead){
	            win=1;
	            lead=temp;
	        }
	    }
	    else{
    	    temp=sum2-sum1;
    	    if(temp>lead){
	            win=2;
	            lead=temp;
	        }
	    }
	}
	cout<<win<<" "<<lead;
	return 0;
}
