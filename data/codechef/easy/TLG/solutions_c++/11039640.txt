
#include <iostream>
#include <cmath>
using namespace std;



int main()
{
    
long long int a , b,t,a1=0 ,b1=0;
 cin >> t;
 long long int win[t] , lead[t] ;
   
for(long long int i=0; i<t; i++)
{
	
  
  cin >> a  >> b;
  
  a1 = a1+a;
  b1 = b1+b;
  
   if(a1>b1)
   {
       win[i] = 1;
       lead[i] = a1-b1;
   }
  else  if(b1>a1)
   {
       win[i] = 2;
       lead[i] = b1-a1;
   }
   
   else {
       
       win[i] =0;
       lead[i] =0;
   }
   
  }
  
  
  long long int max ,w=0;
  max = lead[0];
  for(long long int j=1; j<t; j++)
  {
   
   if(max<lead[j])
   {
       max = lead[j];
       w=j;
   }
  }
  
  cout << win[w]<< " "<< max << endl; 
    
  return 0;   
}
