#include<bits/stdc++.h>
#define ll  int
using namespace std;


int main()
{
   ll n,m,o,p,q,s1=0,s2=0;

  cin>>n;
  q=-100000000;
  map<int,int>k;
  while(n--)
  {
       cin>>m>>o;

           s1+=m;
           s2+=o;
           q=max(q,abs(s1-s2));

           if(s1>s2)
            k[abs(s1-s2)]=1;
           else
            k[abs(s1-s2)]=2;


  }

  cout<<k[q]<<" "<<q<<endl;







}
