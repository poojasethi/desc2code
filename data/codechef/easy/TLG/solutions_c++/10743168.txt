#include <iostream>
using namespace std;

int main ()
{
	int n, max=0, player=0, sum1=0, sum2=0;
	
	cin >> n;
	if (n>10000)
		goto end;
	int p[n], l[n];
	
	for (int i=0; i<n; i++)
	{
		int s1, s2;
		cin >> s1 >> s2;
		sum1 += s1;
		sum2 += s2;
		if (s1>1000 || s2>1000)
			goto end;
		if (sum1 > sum2)
		{
			p[i] = 1;
			l[i] = sum1 - sum2;
		}
		else
		{
			p[i] = 2;
			l[i] = sum2 - sum1;
		}	
	}
	
	for (int i=0; i<n; i++)
		if (max < l[i])
		{
			max = l[i];
			player = p[i];
		}
		
	cout << player << " " << max << endl;
	end:
	return 0;
}
