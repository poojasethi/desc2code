#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    int t1=0,t2=0,s1,s2,x=0,y;
    for(int i=0;i<=t;i++)
    {
        cin>>s1>>s2;
        t1=t1+s1;
        t2=t2+s2;
        if(t1-t2>x)
        {
            y=1;
            x=t1-t2;
        }
         if(t2-t1>x)
        {
            y=2;
            x=t2-t1;
        }
    }
    cout<<y<<"\t"<<x<<endl;
    return 0;
    }
