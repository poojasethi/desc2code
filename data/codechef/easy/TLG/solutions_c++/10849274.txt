#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int p[n],q[n];
    int a=0,b=0;
    for(int i=0;i<n;i++)
    {
        int sc1,sc2;
        cin>>sc1>>sc2;
        a+=sc1;
        b+=sc2;
        if(a>b)
        {
            p[i]=(a-b);
            q[i]=1;

        }
        else
        {
            p[i]=(b-a);
            q[i]=2;
        }

    }

    int max=p[0],y,tt=0;
        for( y=0;y<n;y++)
           {
            if(p[y]>max)
            {max=p[y];
            }}
            for(int i=0;i<n;i++)
            {if(p[i]==max)
            {tt=q[i];
            break;}}


    cout<<tt<<" "<<max<<endl;

}
