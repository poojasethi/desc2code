#include<iostream>

using namespace std;
int main()
{
int t,s1,s2,p1=0,p2=0,i,j,max=0,ans;
cin>>t;
while(t--)
{
    cin>>s1>>s2;
    p1=p1+s1;
    p2=p2+s2;
    if(p1>p2)
    {
        i=p1-p2;
        j=1;
        
    }
    else 
    {
        i=p2-p1;
        j=2;
    }
	if(i>max)
	{
		max=i;
		ans=j;
	}
    
}
cout<<ans<<" "<<max;
return 0;
}
