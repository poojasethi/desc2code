# include <iostream>
# include <stdlib.h>

using namespace std;

int main(){

int size,x,y,i,x1,y1,lead;
std::cin>>size;

//int *a = new int[size];
//int *b = new int[size];

cin>>x>>y;
lead=x-y;

for( i=1 ; i<size ; ++i ) {
    cin>>x1>>y1;
    x+=x1;
    y+=y1;
    if(abs(x-y)>abs(lead))
        lead=x-y;
}

if(lead>0)
    cout<<"1 "<<lead;
else
    cout<<"2 "<<abs(lead);

return 0;

}
