#include <iostream>

using namespace std;

int main()
{
    int t,b,count,j=1;
    cin>>t;
    while(t--)
    {
        cin>>b;
        j=1;
        count=0;
        for(int i=4;i<=b;i+=2)
        {
            count=count+j;
            j++;
        }
        cout<<count<<endl;
    }
    return 0;
}
