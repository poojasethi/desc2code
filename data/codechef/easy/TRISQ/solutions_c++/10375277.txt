#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	int b;
	for(int i=0;i<t;i++)
	{
	    cin>>b;
	    int flag=0;
	    if(b<4)
	    {cout<<"0\n";flag=1;}
	    else if(b%2!=0)
	    {
	        b=b--;
	    }
	    if(flag==0)
	    {
	        int n=((b-4)/2)+1;
	        int add=1;
	        for(int j=2;j<=n;j++)
	        {
	            add=add+j;
	        }
	        cout<<add<<"\n";
	    }
	}
	return 0;
	
}
