


#include <iostream>
using namespace std;
int main(int argc, const char * argv[]) {
    int T,B,n,s,l;
    cin>>T;
    while(T!=0)
    {
        n=1,l=1;
        cin>>B;
        if(B<4)
        
            cout<<"0\n";
        else
        {
        
        s=(B-4)/2;
        while(s)
        {
            n++;
            l=l+n;
            s--;
        }
        cout<<l<<"\n";
        }
            T--;
    }
    return 0;
}
