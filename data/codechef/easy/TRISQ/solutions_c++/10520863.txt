#include <iostream>
#include <cstdio>
#include <math.h>

int main() {
  int t, b;
  long long n;
  scanf("%d", &t);

  for (int i = 0; i < t; i++) {
    n = 0;
    scanf("%d", &b);

    while (b > 2) {
      int x = (b - 2) / 2;
      if (x <= 0) break;
      n += 2 * x - 1;
      b -= 4;
    }

    printf("%lld\n", n);
  }
  return 0;
}
