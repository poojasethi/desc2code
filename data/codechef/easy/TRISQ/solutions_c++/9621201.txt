#include <iostream>
#include <cstdlib>
using namespace std;
 
int main()
{
  int ans[10000];
  int reached;
  int T, B;
  ans[0] = 0;
  ans[1] = 0;
  ans[2] = 0;
  ans[3] = 0;
  reached = 3;
  cin >> T;
  for (int i = 0; i < T; i++)
  {
    cin >> B;
    if(reached < B){
      for(int j = reached+1; j <= B; j++){
        ans[j] = ans[j-2] + (j-2)/2;
      }
      reached = B;
    }
    cout << ans[B] <<endl;
  }
}
