#include<iostream>

using namespace std;

int NoOfSquare(int b){
    if(b == 4)
        return 1;
    else if(b<4)
        return 0;
    else
        return 1 + NoOfSquare(b-2);
}

int main(void){
    int t, b, n, count;
    cin>>t;
    
    while(t--){
        cin>>b;
        //number of square in one row
        n = NoOfSquare(b);
        
        cout<< n*(n+1)/2<<"\n";
    }
}
