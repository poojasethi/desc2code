#include <iostream>
 
using namespace std;
 
int main()
{
	long long int T;
	long long int B;
	long long int i;
	long long int sum;
	long long int N;
	
	cin >> T;
	
	while (T--) {
		cin >> B;
		
		sum = 0;
		
		if (B < 4) {
		    sum = 0;
		} else {
		    sum = 1;
		    for (i = 6; i <= B; i = i + 2) {
			    sum = sum + ((i)/2) - 1;
		    }
		}
		
		cout << sum << endl;
	}
}
