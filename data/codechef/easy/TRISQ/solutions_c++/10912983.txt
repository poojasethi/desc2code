#include <iostream>
using namespace std;

int T = 0;
int *B;
int n=0;
int temp = 0;

int main(){
	cin>>T;
	B = new int[T];
	for (int i = 0; i < T; ++i)
	{
		cin>>B[i];
	}

	for (int i = 0; i < T; ++i)
	{
		if(B[i] <= 3) {
			cout<<0<<endl;
		}

		else {
			if(B[i]%2 == 0){
				n = ((B[i] - 4)/2) + 1;
			}

			else{
				temp = B[i]	- 1;
				n = ((temp - 4)/2) + 1;
			}

			cout<<(n*(n+1))/2<<endl;
		}
	}
	return 0;
}
