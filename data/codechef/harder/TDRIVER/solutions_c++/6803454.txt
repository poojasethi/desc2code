#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
ll X[200005],Y[200005];
int main(){
 ll t,n,a,b,x,y;
 cin >> t;
 while(t--){
	cin >> n >> a >> b;
   for(ll i=0;i<n;i++){
		cin >> x >> y;
      X[i] = a*x + b*y;
      Y[i] = a*x - b*y;
   }
   sort(X,X+n); sort(Y,Y+n);
   ll ans = 0;
   ll md;
   md = X[0];
   for(ll i=1;i<n;i++){
		ans += (i*X[i]-md);
		md += X[i];
   }
   md = Y[0];
   for(int i=1;i<n;i++){
		ans += (i*Y[i]-md);
		md += Y[i];
   }
   cout << ans/2 << endl;
 }
 return 0;
}
