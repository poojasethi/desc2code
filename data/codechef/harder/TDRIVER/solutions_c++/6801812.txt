using namespace std; 

int r[N],c[N];
ll s[N];
int main()
{
    int T,i,j,k,m,n,ca=0;
    scanf("%d",&T);
    while(T--){
        int a,b;
        scanf("%d%d%d",&n,&a,&b);
        for(i=0;i<n;i++){
            scanf("%d%d",&j,&k);
            j*=a,k*=b;
            r[i]=j+k,c[i]=j-k;
        }
        sort(r,r+n),sort(c,c+n);
        s[0]=0;
        for(i=0;i<n;i++)s[i+1]=s[i]+r[i];
        ll ans=0;
        for(i=0;i<n;i++){
            ans+=1LL*i*r[i]-s[i];
        }
        for(i=0;i<n;i++)s[i+1]=s[i]+c[i];
        for(i=0;i<n;i++){
            ans+=1LL*i*c[i]-s[i];
        }
        cout<<ans/2<<"\n";
    }
    return 0;
}
