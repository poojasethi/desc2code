#include <iostream>
#include <vector>
#include <algorithm>
#include <stdio.h>
using namespace std;
int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int	N;
		scanf("%d",&N);
		int a,b;
		scanf("%d %d",&a,&b);
		long long x[N],y[N];
		for (int i = 0; i < N; ++i)
		{
			scanf("%lld %lld",&x[i],&y[i]);
			x[i]=a*x[i];
			y[i]=b*y[i];
			long long t1,t2;
			t1=(x[i]+y[i]);
			t2=(x[i]-y[i]);
			x[i]=t1;
			y[i]=t2;
		}
		sort(&x[0],&x[N]);
		long long sumx=0,tmp=x[0];
		for (int i = 1; i < N; ++i)
		{
			sumx=sumx+i*x[i]-tmp;
			tmp=tmp+x[i];
		}
		sort(&y[0],&y[N]);
		long long sumy=0,tmp2=y[0];
		for (int i = 1; i < N; ++i)
		{
			sumy=sumy+i*y[i]-tmp2;
			tmp2=tmp2+y[i];
		}
		printf("%lld\n",(sumx+sumy)/2);
	}
	return 0;
}
