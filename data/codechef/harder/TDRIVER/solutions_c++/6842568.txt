#include <iostream>
#include <algorithm>

using namespace std; 

long long x[200000], y[200000]; 
int main(){
	int t; 
	cin >> t; 
	while(t--){
		int n; 
		cin >> n; 
		int a, b; 
		cin >> a >> b; 
		int p, q; 
		for(int i=0;i<n;i++){

			cin >> p >> q; 
			x[i] = a*p + b*q; 
			y[i] = a*p - b*q;
			// cout << x[i] << " " << y[i] << endl;
		}
		sort(x, x+n);
		sort(y, y+n);
		long long ans = 0;
		long long temp = 0; 
		for(int i=1;i<n;i++){
			temp = x[i] - x[i-1];
			ans += temp * i * (n-i);
		}
		temp = 0; 
		for(int i=1;i<n;i++){
			temp = y[i] - y[i-1];
			ans += temp * i * (n-i);
		}
		ans >>= 1; 
		cout << ans << endl;
	}
}
