/*
  Topic:- Given a set of N points and constants a and b, calculate the sum of the distances between each pair of points. However, the distance between 2 points is defined as max(a × |x1 - x2|, b × |y1 - y2|).

  Method:- 
  ----Transform each coordinate (x, y) into (a∗x+b∗y2, a∗x−b∗y2).
  ----Calculate the distance between all pairs of points using the Manhattan distance on the new coordinates.
  
  // Taxi Driver
  //Medium-Hard Problem
  //Link to Editorial:- http://discuss.codechef.com/questions/68102/tdriver-editorial
  
  // Author: Constantine Sokol
  // Complexity : O(N log N)
  ----We sort the coordinates and then process them in linear time. Hence, the total time complexity is O(N log N).
  
  // Expected Verdict: AC
  
  */

#include <iostream>
#include <cstdio>
#include <algorithm> //To use Sort function

using namespace std;

const int MAXN = 200200;

int n;
int a, b;
int x[MAXN];
int y[MAXN];

long long process(int n, int x[]) 
{
  /* cout << "Initially x[i]\n" ;
  for( int i = 1 ; i <= n ; i++ )
    {
      cout << x[i] << " " ;
    }
    cout << "\n" ;*/
  sort(x + 1, x + n + 1); //Sort array from x[1] to x[n] O(nlogn) time
  /*cout << "Display upto n + 1 \n" ;
  for( int i = 1 ; i <= n ; i++ )
    {
      cout << x[i] << " " ;
    }
    cout << "\n" ;*/
  long long result = 0;
  long long sum_prev = 0;
  for (int i = 1; i <= n; i++) 
    {
      sum_prev += x[i];
      result += (long long)x[i] * i - sum_prev;
    }
  return result;
}

void solve() 
{
  scanf("%d", &n);
  scanf("%d%d", &a, &b);
  for (int i = 1; i <= n; i++) 
    {
      int xs, ys;
      scanf("%d%d", &xs, &ys);
      x[i] = a * xs + b * ys;         //Transforming into new co-ordinate system
      y[i] = a * xs - b * ys;
    }
  long long answer_x = process(n, x);          //Finding the difference between x-co-ordinates for all points 
  long long answer_y = process(n, y);          //Finding the difference between y-co-ordinates for all points
  printf("%lld\n", (answer_x + answer_y) / 2); //Adding the x and y difference and div by 2 becoz in new transormation we have ( ax + by ) / 2 
}

int main() 
{
  int t; 
  scanf("%d", &t);
  for (int i = 0; i < t; i++) 
    {
      solve();
    }
  return 0;
}
