#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
	int t;
	scanf("%d", &t);
	while(t--) {
		int n;
		ll a, b;
		scanf("%d", &n);
		scanf("%lld%lld", &a, &b);
		vector<ll> x, y;
		for(int i=0;i<n;i++) {
			ll c, d;
			scanf("%lld%lld", &c, &d);
			x.push_back(a*c + b*d);
			y.push_back(a*c - b*d);
		}
		sort(x.begin(), x.end());
		sort(y.begin(), y.end());
		ll ans = 0, psum = x[0];
		for(int i=1;i<x.size();i++) {
			ans += (i*x[i]-psum);
			psum += x[i];
		}
		psum = y[0];
		for(int i=1;i<y.size();i++) {
			ans += (i*y[i]-psum);
			psum += y[i];
		}
		ans /= 2;
		printf("%lld\n", ans);
	}
	return 0;
}
