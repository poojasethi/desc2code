#include <bits/stdc++.h>    
using namespace std;
typedef long long ll;    
ll go(vector<int>& p)
{
    sort(p.begin(), p.end());
    ll res = 0, diff = 0;
    for (int i = 1; i < (int) p.size(); i++)
    {
        diff += ll(i) * (p[i] - p[i - 1]);
        res += diff;
    }
        return res;
}
void solve(int test_num)
{
    int N, A, B;
    cin >> N >> A >> B;
    vector<int> x(N), y(N);
    for (int i = 0; i < N; i++) cin >> x[i] >> y[i];
    for (int i = 0; i < N; i++)
    {
        x[i] *= A;
        y[i] *= B;
    }
    vector<int> u(N), v(N);
    for (int i = 0; i < N; i++)
    {
        u[i] = x[i] + y[i];
        v[i] = x[i] - y[i];
    }
    const ll res = (go(u) + go(v)) >> 1;
    cout << res << endl;
} 
void solve()
{  
    ios::sync_with_stdio(false);
    int T = 1;
    cin >> T;
    for (int t = 1; t <= T; t++) solve(t);
}    
int main()
{
    solve();
    return 0;
}
