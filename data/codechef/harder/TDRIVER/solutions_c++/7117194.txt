#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int t,n,a,b,i,xt,yt;
	long long ans;
	long long x[200200],y[200200];
	cin>>t;
	while(t--){
		cin>>n>>a>>b;
		for(i=0;i<n;i++){
			cin>>xt>>yt;
			x[i] = (a*xt)+(b*yt);
			y[i] = (a*xt)-(b*yt);
		}
		sort(x,x+n);sort(y,y+n);
		ans = 0;
		for(i=0;i<n;i++){
			ans += (((long long)(2*i-n+1))*(x[i]+y[i]));
		}
		cout<<(ans)/2<<endl;
	}
}
