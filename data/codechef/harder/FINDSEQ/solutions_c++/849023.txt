#include <cstdio>
#include <algorithm>
using namespace std;

typedef pair<int,int> PII;

const int maxN = 1000;
PII b[maxN+1];
int a[maxN+1];
int s[maxN+1][maxN+1];

int sum(int i1,int i2,int j1,int j2)
{
	if(i1>i2 || j1>j2) return 0;
	i1--;j1--;

	return s[i2][j2]-s[i2][j1]-s[i1][j2]+s[i1][j1];
}

int main()
{
	int TST;
	scanf("%d",&TST);
	for(int tst=0;tst<TST;tst++)
	{
		int n,i,j,k;
		char w[6];
		scanf("%d %s",&n,w);
		int p[5];
		for(i=0;i<5;i++)
			p[i]=w[i]-'1';
		for(i=1;i<=n;i++)
		{
			scanf("%d",&b[i].first);
			if(p[1]>p[3]) b[i].first*=-1;
			b[i].second=i;
		}
		if(p[1]>p[3]) for(i=0;i<5;i++) p[i]=4-p[i];
		sort(b+1,b+1+n);
		int m=0;
		for(i=1;i<=n;i=j)
		{
			m++;
			for(j=i;j<=n && b[i].first==b[j].first;j++)
				a[b[j].second]=m;
		}
		for(i=1;i<=n;i++)
			for(j=1;j<=m;j++)
				s[i][j]=(a[i]==j)+s[i-1][j]+s[i][j-1]-s[i-1][j-1];
		PII x[3]={PII(p[0],0),PII(p[2],2),PII(p[4],4)};
		sort(x,x+3);
		int mini=x[0].second;
		int midi=x[1].second;
		int maxi=x[2].second;
		bool found=false;
		for(i=2;i<n && !found;i++)
		{
			for(j=i+2;j<n && !found;j++)
				if(a[i]<a[j])
				{
					int l[5],r[5],L[5],R[5];
					for(k=0;k<5;k+=2)
					{
						if(k==0) l[k]=1,r[k]=i-1; else
						if(k==2) l[k]=i+1,r[k]=j-1; else
							l[k]=j+1,r[k]=n;
						if(p[k]<p[1]) L[k]=1,R[k]=a[i]-1; else
						if(p[k]<p[3]) L[k]=a[i]+1,R[k]=a[j]-1; else
							L[k]=a[j]+1,R[k]=m;
						if(sum(l[k],r[k],L[k],R[k])==0) break;
					}
					if(k<5) continue;

					int lo=L[mini]-1,hi=R[mini];
					while(lo+1<hi)
					{
						int md=(lo+hi)/2;
						if(sum(l[mini],r[mini],L[mini],md)>0) hi=md; else lo=md;
					}
					int val_min=hi;
					if(L[midi]<hi+1) L[midi]=hi+1;
					if(sum(l[midi],r[midi],L[midi],R[midi])==0) continue;

					lo=L[maxi],hi=R[maxi]+1;
					while(lo+1<hi)
					{
						int md=(lo+hi)/2;
						if(sum(l[maxi],r[maxi],md,R[maxi])) lo=md; else hi=md;
					}
					int val_max=lo;
					if(R[midi]>lo-1) R[midi]=lo-1;
					if(sum(l[midi],r[midi],L[midi],R[midi])==0) continue;

					found=true;
					int ans[5]={-1,i,-1,j,-1};

					for(k=l[mini];a[k]!=val_min;k++);
					ans[mini]=k;

					for(k=l[maxi];a[k]!=val_max;k++);
					ans[maxi]=k;

					for(k=l[midi];!(L[midi]<=a[k] && a[k]<=R[midi]);k++);
					ans[midi]=k;

					for(k=0;k<5;k++)
						printf("%d%c",ans[k]-1,k<4?' ':'\n');
				}
		}
		if(!found) puts("-1");
	}
	return 0;
}
