#define y1 Y1
#define x1 X1
#define rep(i,j,k) for(int i=(int)j;i<=(int)k;i++)
#define per(i,j,k) for(int i=(int)j;i>=(int)k;i--)
using namespace std;
typedef long long LL;
typedef double db;
const int N=105;
struct po{
	double x,y;
	inline po(double _x=0.0,double _y=0.0){
		x=_x;y=_y;
	}
}a[105];int n;
inline po operator +(const po &a,const po &b){
	return po(a.x+b.x,a.y+b.y);
}
inline po operator -(const po &a,const po &b){
	return po(a.x-b.x,a.y-b.y);
}
inline double operator *(const po &a,const po &b){
	return a.x*b.y-a.y*b.x;
}
inline double operator /(const po &a,const po &b){
	return a.x*b.x+a.y*b.y;
}
int id[N];
po d1,d2;
inline bool cmp(const int &x,const int &y){
	return (a[x]-d1)/(d2-d1)<(a[y]-d1)/(d2-d1);
}
double xy1,xx1,yy1,x1,y1,s1;
double xy2,xx2,yy2,x2,y2,s2;
inline void jia1(const po &a){
	xy1+=a.x*a.y;
	xx1+=a.x*a.x;
	yy1+=a.y*a.y;
	x1+=a.x;
	y1+=a.y;
	s1+=1.0;
}
inline void jia2(const po &a){
	xy2+=a.x*a.y;
	xx2+=a.x*a.x;
	yy2+=a.y*a.y;
	x2+=a.x;
	y2+=a.y;
	s2+=1.0;
}
inline void jian1(const po &a){
	xy1-=a.x*a.y;
	xx1-=a.x*a.x;
	yy1-=a.y*a.y;
	x1-=a.x;
	y1-=a.y;
	s1-=1.0;
}
inline void jian2(const po &a){
	xy2-=a.x*a.y;
	xx2-=a.x*a.x;
	yy2-=a.y*a.y;
	x2-=a.x;
	y2-=a.y;
	s2-=1.0;
}
double calc1(){
	if (s1==0) return 0;
	double xa=x1/s1;
	double ya=y1/s1;
	double A=xx1-s1*xa*xa;
	double B=2*s1*xa*ya-2*xy1;
	double C=yy1-s1*ya*ya;
	double k1=4;
	double k2=-4*(A+C);
	double k3=4*A*C-B*B;
	return (-k2-sqrt(k2*k2-4*k1*k3))/(k1*2);
}
double calc2(){
	if (s2==0) return 0;
	double xa=x2/s2;
	double ya=y2/s2;
	double A=xx2-s2*xa*xa;
	double B=2*s2*xa*ya-2*xy2;
	double C=yy2-s2*ya*ya;
	double k1=4;
	double k2=-4*(A+C);
	double k3=4*A*C-B*B;
	return (-k2-sqrt(k2*k2-4*k1*k3))/(k1*2);
}
double work(int u,int v){
	rep(i,1,n)id[i]=i;
	d1=a[u];d2=a[v];
	sort(id+1,id+1+n,cmp);
	
	xy1=xx1=yy1=x1=y1=s1=0;
	xy2=xx2=yy2=x2=y2=s2=0;
	rep(i,1,n)if(i==u||((d2-d1)*(a[i]-d1)>(1e-8)))jia1(a[i]);
	else jia2(a[i]);
	
	double ret=calc1()+calc2();
	rep(i,1,n){
		if(id[i]==u||((d2-d1)*(a[id[i]]-d1)>(1e-8))){
			jian1(a[id[i]]);
			jia2(a[id[i]]);
		}
		else{
			jian2(a[id[i]]);
			jia1(a[id[i]]);
		}
		double cc=calc1()+calc2();
		if(cc<ret)ret=cc;
	}
	return ret;
}
int main(){
	scanf("%d",&n);
	rep(i,1,n)scanf("%lf%lf",&a[i].x,&a[i].y);
	double ans=1e18;
	rep(i,1,n)rep(j,1,n)if(i^j){
		double ret=work(i,j);
		if(ret<ans)ans=ret;
	}
	printf("%.10lf\n",ans/n);
	return 0;
}
