#include<cstdio>
#include<algorithm>
using namespace std;

const int maxlen=1400;
const int mod=1000000007;
int pow[8][maxlen+10];
pair<int,int> A[1500000];
int S[1500000][7];
int n,m,t;

int ginv(int x)
{
	int res=1;
	for (int tmp=x,o=mod-2;o;o/=2)
	{
		if (o&1) res=(long long)res*tmp%mod;
		tmp=(long long)tmp*tmp%mod;
	}
	return res;
}

int main()
{
	//freopen("input.txt","r",stdin);

	for (int i=2;i<=7;i++)
	{
		pow[i][0]=1;
		for (int j=1;j<=maxlen;j++)
			pow[i][j]=(long long)pow[i][j-1]*i%mod;
	}
	for (int i=0;i/3<=700;i++)
	for (int j=0;i/3+j/2<=700;j++)
		A[++m]=make_pair((long long)pow[2][i]*pow[3][j]%mod,(i<<15)+j);
	sort(A+1,A+1+m);
	int _m=m;m=1;
	for (int i=2;i<=_m;i++)
	{
		int u=(A[i].second>>15);
		int v=(A[i].second&32767);
		int i8=u/3,i9=v/2;
		int i6=min(u-i8*3,v-i9*2);
		int i4=(u-i8*3-i6)/2;
		int i2=u-i8*3-i4*2-i6,i3=v-i9*2-i6;
		int len=i2+i3+i4+i6+i8+i9;

		if (A[i].first!=A[m].first) 
		{
			A[++m]=A[i];
			S[m][0]=i2;S[m][1]=i3;
			S[m][2]=i4;S[m][3]=i6;
			S[m][4]=i8;S[m][5]=i9;
			S[m][6]=len;
		} else 
		{
			if (len<S[m][6] || (len==S[m][6] &&
				((i2>S[m][0]) || (i2==S[m][0] && i3>S[m][1]) ||
				(i2==S[m][0] && i3==S[m][1] && i4>S[m][2]) ||
				(i2==S[m][0] && i3==S[m][1] && i4==S[m][2] && i6>S[m][3]) ||
				(i2==S[m][0] && i3==S[m][1] && i4==S[m][2] && i6==S[m][3] && i8>S[m][4]))))
			{
				S[m][0]=i2;S[m][1]=i3;
				S[m][2]=i4;S[m][3]=i6;
				S[m][4]=i8;S[m][5]=i9;
				S[m][6]=len;
			}	
		}
	}

	for (scanf("%d",&t);t--;)
	{
		scanf("%d",&n);
		
		if (n==0) {printf("10\n");continue;}
		if (n<10) {printf("%d\n",n);continue;}
		
		int r2,r3,r4,r5,r6,r7,r8,r9,len=maxlen;
		r2=r3=r4=r5=r6=r7=r8=r9=1000;
		
		for (int i=0;i<=len;i++)
		for (int j=0;i+j<=len;j++)
		{
			int key=(long long)ginv((long long)pow[5][i]*pow[7][j]%mod)*n%mod;
			int l=1,r=m;
			while (l+1<r)
			if (A[(l+r)>>1].first<=key) l=((l+r)>>1);
			else r=((l+r)>>1);
			if (A[l].first!=key) continue;
			
			int i2=S[l][0],i3=S[l][1],i4=S[l][2],i5=i,i6=S[l][3],i7=j,i8=S[l][4],i9=S[l][5];
			int _len=i2+i3+i4+i5+i6+i7+i8+i9;
			if (_len>len) continue;
			if (_len<len || (i2>r2) || (i2==r2 && i3>r3) || (i2==r2 && i3==r3 && i4>r4) ||
				(i2==r2 && i3==r3 && i4==r4 && i5>r5) || (i2==r2 && i3==r3 && i4==r4 && i5==r5 && i6>r6) ||
				(i2==r2 && i3==r3 && i4==r4 && i5==r5 && i6==r6 && i7>r7) ||
				(i2==r2 && i3==r3 && i4==r4 && i5==r5 && i6==r6 && i7==r7&& i8>r8))
			{
				len=_len;
				r2=i2;r3=i3;r4=i4;r5=i5;r6=i6;r7=i7;r8=i8;r9=i9;
			}
		}
		
		//printf("%d\n",len);
		for (int i=0;i<r2;i++) printf("2");
		for (int i=0;i<r3;i++) printf("3");
		for (int i=0;i<r4;i++) printf("4");
		for (int i=0;i<r5;i++) printf("5");
		for (int i=0;i<r6;i++) printf("6");
		for (int i=0;i<r7;i++) printf("7");
		for (int i=0;i<r8;i++) printf("8");
		for (int i=0;i<r9;i++) printf("9");
		printf("\n");
	}
}
