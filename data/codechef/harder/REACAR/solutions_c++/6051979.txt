#include <cstdio>
#include <cstring>
#include <map>

using namespace std;
long idx, temp, n, tree[100005], ans;
map <int, int> mapping;
void update(int x) {
  while (x<=n) {
    tree[x]+=1;
    x+=(x&(-x));
  }
}
int read(int x) {
  int sum=0;
  while (x>0) {
    sum+=tree[x];
    x-=(x&(-x));
  }
  return sum;
}

int main(){
    scanf("%d",&n);
    for (int i=1; i<=n; i++) {
        scanf("%d",&temp);
        mapping[temp]=i;
    }
    for (int i=1; i<=n; i++) {
        scanf("%d",&temp);
        idx=mapping[temp];
        ans+=read(n)-read(idx);
        update(idx);
    }
    printf("%d\n", ans);
    return 0;
}
