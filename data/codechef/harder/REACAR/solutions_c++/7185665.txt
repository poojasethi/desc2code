#include<bits/stdc++.h>
using namespace std;

const int N=123456;

int a[N], b[N], bit[N];
map<int, int> m;

void update(int idx, int val){

    while(idx<=N){

        bit[idx]+=val;
        idx+=(idx & -idx);
    }
}

int get(int idx){

    int sum=0;
    while(idx>0){

        sum+=bit[idx];
        idx-=(idx & -idx);
    }
    return sum;
}

int main(){

    int n;
    scanf("%d", &n);
    for(int i=1;i<=n;i++){

        scanf("%d", &a[i]);
    }
    for(int i=1;i<=n;i++){

        scanf("%d", &b[i]);
        m[b[i]]=i;
    }
    for(int i=1;i<=n;i++){

        a[i]=m[a[i]];
        update(i, 1);
    }
    int ans=0;
    for(int i=1;i<=n;i++){

        ans+=get(a[i]-1);
        update(a[i], -1);
    }
    printf("%d", ans);
}
