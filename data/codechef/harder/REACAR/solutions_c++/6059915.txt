#include<bits/stdc++.h>
#include<stdio.h>
#include<stdlib.h>
#include "map"

#define ll long long
#define l long

using namespace std;


ll bit[100006];
int a[100005],b[100005],n,temp;

map<int,int> mp;
map<int,int>::iterator it;


ll read(int i)
{
    ll res=0;
    while(i>0)
    {
        res+=bit[i];
        i=i-(i&-i);
    }
    return res;
}
void update(int i,int val)
{
    while(i<=n)
    {
        bit[i]+=val;
        i=i+(i&-i);
    }
}

int main()
{
    
    int i,x;
    ll res;
    scanf("%d",&n);
    for(i=0;i<n;i++)  scanf("%d",&a[i]);
    for(i=0;i<n;i++)
    {
       scanf("%d",&temp);
       mp.insert(pair<int,int>(temp,i+1));
    }
    for(i=0;i<n;i++)
    {
        it=mp.find(a[i]);
        
        b[i]=it->second;

    }
    res=0;
    for(i=n-1;i>=0;i--)
    {
        res+=read(b[i]-1);
        update(b[i],1);
    }
    printf("%lld",res);
    return 0;
}
