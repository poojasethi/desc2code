#include<iostream>
using namespace std;
int search(int a[],int n,int x)
{
	int i;
	for(i=0;i<n;i++)
	if(a[i]==x)
		break;
		return i;
}
int main()
{
	int n,*a,*b,k,i,x,j,count;
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin>>n;
	a=new int[n];
	b=new int[n];
	for(i=0;i<n;i++) cin>>a[i];
	for(i=0;i<n;i++) cin>>b[i];
	count=0;
	for(i=0;i<n;i++)
	{
		x=b[i];
		j=search(a,n,x);
		if(i!=j)
		{
			for(k=j-1;k>=i;k--)
			{
				swap(a[k],a[k+1]);
				count++;
			}
		}
	}
	cout<<count;
}
