#include<iostream>
#include<algorithm>
using namespace std;
long long int ans;
int pos[10000000];
void func1(int *a,int n1,int n2,int n)
{
int i=0,j=0,k=0,w=0;
int *p=a,*q=&a[n/2],temp[n];
for(;;)
{
while(p[i]<=q[j]&&i<n1)
{
temp[k++]=p[i];
i++;
}
if(i==n1||j==n2)
break;
while(q[j]<p[i]&&j<n2)
{
temp[k++]=q[j];
j++;
w++;
}
ans=ans+w*(n1-i);
w=0;
if(i==n1||j==n2)
break;
}
if(i!=n1)
{
while(i<n1)
{
temp[k++]=p[i];
i++;
}
}
if(j!=n2)
{
while(j<n2)
{
temp[k++]=q[j];
j++;
}
}
for(int m=0;m<k;m++)
a[m]=temp[m];

}
void func(int *a,int n)
{
if(n==1)
return;
if(n==2)
{
if(a[0]>a[1])
{
swap(a[0],a[1]);
ans++;
}
return;
}
if(n%2==0)
{
func(a,n/2);
func(&a[n/2],n/2);
func1(a,n/2,n/2,n);
}
else
{
func(a,n/2);
func(&a[n/2],n/2+1);
func1(a,n/2,n/2+1,n);
}
}
int main()
{
ios_base::sync_with_stdio(false);
int n;
cin>>n;
int a[n],temp[n],b[n];
for(int i=0;i<n;i++)
cin>>a[i];
for(int i=0;i<n;i++)
{
cin>>b[i];
pos[b[i]]=i;
}
for(int i=0;i<n;i++)
temp[i]=pos[a[i]];
func(temp,n);
cout<<ans<<endl;
return 0;
}
