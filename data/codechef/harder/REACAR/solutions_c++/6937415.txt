typedef long long ll;
int tree[1000005];int n;
inline void update(int idx) {
    while(idx<=n && idx>0) {
        tree[idx]+=1;
        idx += (idx&-idx);
    }
}
inline ll read(int idx) {
     ll sum=0;
     while(idx>0) {
        sum+=tree[idx];
        idx -= (idx&-idx);
    }
    return sum;
}
int main()
{
     sci(n);
     int a[n+2];
     map<int ,int> m;
     for(int i=1;i<=n;i++) {
        sci(a[i]);
        m[a[i]]=i;
     }
     long long ans=0;
     int r[n+2];
     for(int i=1;i<=n;i++) {
        sci(r[i]);
        int pos=m[r[i]];
        //cout<<read(n)<<" "<<read(pos)<<endl;
        ans+=read(n)-read(pos);
        update(pos);
    }
    printf("%lld\n",ans);
}
