#include<bits/stdc++.h>
#include<stdio.h>
#include<stdlib.h>
#include "map"

#define ll long long
#define l long

using namespace std;


ll bit[100006];
int ar[100005],b[100005],n;

int up[100005],org[100005];

ll read(int i)
{
    ll res=0;
    while(i>0)
    {
        res+=bit[i];
        i=i-(i&-i);
    }
    return res;
}
void update(int i,int val)
{
    while(i<=n)
    {
        bit[i]+=val;
        i=i+(i&-i);
    }
}
map<int,int> mp;
map<int,int>::iterator it1;
int main()
{
    
    int i,c,x;
    ll res;
    scanf("%d",&n);
    for(i=0;i<n;i++)  scanf("%d",&up[i]);
    for(i=0;i<n;i++)
    {
        scanf("%d",&org[i]);

       mp.insert(pair<int,int>(org[i],i+1));
    }
    for(i=0;i<n;i++)
    {
        it1=mp.find(up[i]);
        
        ar[i]=it1->second;

    }
    res=0;
    for(i=n-1;i>=0;i--)
    {
        res+=read(ar[i]-1);
        update(ar[i],1);
    }
    printf("%lld",res);
    return 0;
}
