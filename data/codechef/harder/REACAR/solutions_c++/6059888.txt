ll bit[100006];

ll read(int i)
{
    ll res=0;
    while(i>0)
    {
        res+=bit[i];
        i=i-(i&-i);
    }
    return res;
}
void update(int i,int val)
{
    while(i<=n)
    {
        bit[i]+=val;
        i=i+(i&-i);
    }
}

map<int,int>::iterator it1;

       mp.insert(pair<int,int>(org[i],i+1));
