long long int merg(int x[],int start,int end,long long int count)
{
int mid;
if(start<end)
{
mid=(start+end)/2;
count=merg(x,start,mid,count);
count=merg(x,mid+1,end,count);
int temp[100001],i,j,k;
i=start;
j=mid+1;
k=0;
int w=1;
while(i<=mid&&j<=end)
{
if(x[i]<x[j])
temp[k++]=x[i++];
else
{
temp[k++]=x[j++];
count+=j-i-w;
w++;
}
}
//printf("%dc",count);
//nl;
while(i<=mid)
temp[k++]=x[i++];
while(j<=end)
temp[k++]=x[j++];
for(i=start,j=0;i<=end;i++,j++)
x[i]=temp[j];
//for(int z=0;z<i;z++)
//printf("%d ",x[z]);
//nl;
}
return count;
}
