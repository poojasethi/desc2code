#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<map>

using namespace std;

#define maxn 1000009

typedef long long ll;

ll arr[maxn], tree[maxn];
map<ll,int> hash;
void update(int idx, int val)
{
   while (idx <= maxn)
   {
       tree[idx] += val;
       idx += (idx & -idx);
   }
}
      
ll query(int idx)
{
        ll sum=0;
        while (idx>0)
        {
              sum += tree[idx];
              idx -= (idx & -idx);
        }
        return sum;
}

int main()
{
	int n;
	scanf("%d",&n);
	ll a;
	for(int i=1;i<=n;i++)
	{
		scanf("%lld",&a);
		hash[a] = i ; 
	}
	int c1=1;
	for(int i = 0;i<n;i++)
	{
		scanf("%d",&a);
		arr[hash[a]] = c1;
		c1++;
	}
	ll sum = 0;
	for(int i=n;i>0;i--)
	{
		sum = sum + query(arr[i]-1);
		update(arr[i],1);		
	}	
	printf("%lld\n",sum);
}
