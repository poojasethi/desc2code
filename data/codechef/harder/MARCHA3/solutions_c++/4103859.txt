#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
typedef pair<int,int> ii;
#define M 1000000000

int main()
{
    //freopen("test.inp","r",stdin);
    int t,m,i,n,k,ma;
    vector<ii> v;
    char c[2],s[5];
    scanf("%d",&t);
    while(t--)
    {
        v.clear();
        scanf("%d",&m);
        n=m;k=0;ma=0;
        while(m--)
        {
            scanf("%s%d%s",&c,&i,&s);
            if(s[0]=='Y')
            {
                if(c[0]=='>')
                {
                    if(i==M)
                    continue;
                    v.push_back(ii(i+1,-1));
                    v.push_back(ii(M,1));
                }
                else if(c[0]=='<')
                {
                    if(i==1)
                    continue;
                    v.push_back(ii(1,-1));
                    v.push_back(ii(i-1,1));
                }
                else
                {
                    v.push_back(ii(i,-1));
                    v.push_back(ii(i,1));
                }
            }
            else
            {
                if(c[0]=='<')
                {
                    v.push_back(ii(i,-1));
                    v.push_back(ii(M,1));
                }
                else if(c[0]=='>')
                {
                    v.push_back(ii(1,-1));
                    v.push_back(ii(i,1));
                }
                else
                {
                    if(i>1)
                    {
                        v.push_back(ii(1,-1));
                        v.push_back(ii(i-1,1));
                    }
                    if(i<M)
                    {
                        v.push_back(ii(i+1,-1));
                        v.push_back(ii(M,1));
                    }
                }
            }
        }
        sort(v.begin(),v.end());
        for(i=0;i<v.size();++i)
        {
            k+=v[i].second;
            if(v[i].second==-1)
            ma=min(ma,k);
        }
        printf("%d\n",n+ma);
    }
}
