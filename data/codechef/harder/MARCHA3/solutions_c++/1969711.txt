#define inf 1000000000

#define N 100002
int g[N],l[N],e[N];
int main()
{
	//freopen("in.txt","r",stdin);
	int T,i,j,q,k,m,n,num;char op,s[10];
	scanf("%d",&T);
	while(T--)
	{
	 scanf("%d",&n);
	 int gsz=0,esz=0,lsz=0;
	 for(i=0;i<n;i++)
	 {
	  scanf(" %c%d%s",&op,&num,s);
	  if(s[0]=='N')
	  {
	   if(op=='<')g[gsz++]=num-1;
	   else if(op=='>')l[lsz++]=num+1;
	   else
	   {
	    if(num>1)l[lsz++]=num;
	    if(num<inf)g[gsz++]=num;
	   }
	  }
	  else
	  {
	   if(op=='<'){if(num>1)l[lsz++]=num;}
	   else if(op=='>'){if(num<inf)g[gsz++]=num;}
	   else e[esz++]=num;
	  }
	 }
	 e[esz]=inf+10,l[lsz]=inf+10,g[gsz]=-10;
	 sort(l,l+lsz+1);sort(g,g+gsz+1);sort(e,e+esz+1);
	 int gc=0,ec=0,lc=0,m=0,cnt;
	 while(1)
	 {
	  if(g[gc]+1<l[lc])
	  {
	   	cnt=gc+lsz-lc;
	   	int ani=0;
	   	while(e[ec]<=g[gc])ec++;
	   	for(i=ec;e[i]<l[lc];)
	   	{
	   	 int mc=1;
	   	 for(j=i++;e[j]==e[i];i++)mc++;
	   	 if(ani<mc)ani=mc;
		}
		cnt+=ani;
		if(cnt>m)m=cnt;
		if(gc>=gsz)break;
	  }
	  if(gc<gsz&&l[lc]>g[gc+1])gc++;
	  else lc++;
	 }
	 printf("%d\n",n-m);
	}
	return 0;
}
