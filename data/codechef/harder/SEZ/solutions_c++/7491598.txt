#include <bits/stdc++.h>
#define N 202
using namespace std;

vector<int> a[N];
int n, m;

bool visited[N];
int match[N];

int kuhn(int v) {
	if(visited[v]) {
		return 0;
	}
	
	visited[v] =  true;
	
	int sz = a[v].size();
	
	for(int u=0; u<sz  ; u++) {
		int t = a[v][u];
		
		if(match[t] == -1 || kuhn(match[t])) {
			match[t] = v;
			return 1;
		}
	}
	
	return 0;
}

int main() {
	scanf("%d %d ",&n, &m);
	int u, v;
	
	for(int i=0; i<m;i++) {
		scanf("%d %d ",&u, &v);
		a[u].push_back(v);
		a[v].push_back(u);
	}
	
	memset(match, -1, sizeof(match));
	
	int res = 0;
	for(int i = 1; i<=n; i++) {
		memset(visited, 0, sizeof(visited));
		res += kuhn(i);
	}
	
	cout<<(n-res)<<"\n";
}
