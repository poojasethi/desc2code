#define MAXN 205

using namespace std;
int adj[MAXN][MAXN];
int match[MAXN];
bool visit[MAXN];
int n,m;
bool dfs(int u)
{

    for(int i=1;i<=n;i++)
    {
        if(adj[u][i] && !visit[i])
        {
            int t = match[i];
            visit[i] = true;
            if(t==-1 || dfs(t))
            {
                match[i] = u;
                return true;
            }
        }

    }
    return false;
}
