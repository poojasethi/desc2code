const int maxn=1e6+10;
const int mod=1e9+7;
int n, m;
char sa[maxn], sb[maxn];
char ta[maxn], tb[maxn];
int p[maxn];

int dpa[maxn], dpb[maxn];
int tsf[2][maxn];

int faste(int a, int p){int x=1; for (; p; p>>=1, a=1ll*a*a%mod) if (p&1) x=1ll*x*a%mod; return x;}



int main(){
    int tn; cin>>tn;
    REP(ti,tn){
        scanf("%d %s", &n,sa);
        char c;
        for (int i=0; i<n; ++i){
            if (i%5==0){
                c=sa[i/5];
                if (c>='a') c-='a';
                else c-='A'-26;
            }
            ta[i]=c>>(4-i%5)&1;
        }
        scanf("%d %s", &m,sb);
        for (int i=0; i<m; ++i){
            if (i%5==0){
                c=sb[i/5];
                if (c>='a') c-='a';
                else c-='A'-26;
            }
            tb[i]=c>>(4-i%5)&1;
        }

        p[0]=-1;
        REP(i,n){
            int px=p[i];
            while(px>=0 && ta[px]!=ta[i]) px=p[px];
            p[i+1]=px+1;
        }

        dpa[0]=1;
        tsf[ta[0]][0]=1;
        tsf[1^ta[0]][0]=0;

        REP(a,n){

            char cx=ta[a];
            tsf[cx][a]=a+1;
            tsf[cx^1][a]=tsf[cx^1][p[a]];
            int u=tsf[1^cx][a];

            int x=(2*dpa[a]-dpa[u])%mod;
            while(x>=mod) x-=mod;
            while(x<0) x+=mod;
            dpa[a+1]=x;

            x=(2*dpb[a]-dpb[u]-2)%mod;
            while(x>=mod) x-=mod;
            while(x<0) x+=mod;
            dpb[a+1]=x;
        }
        int pos=0;
        int fd=0;
        REP(i,m){
            pos=tsf[tb[i]][pos];
            if (pos==n){ fd=1; puts("0"); break;}
        }
        if (fd) continue;

        assert(dpa[n]%mod!=0);
        int eo=1ll*(mod-dpb[n])*faste(dpa[n],mod-2)%mod;
        int u=pos;
        int res=(1ll*dpa[u]*eo+dpb[u])%mod;
        printf("%d\n", res);
    }
    return 0;
}
