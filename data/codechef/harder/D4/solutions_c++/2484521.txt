#define  maxn  10000001
#define  maxm  10005
#define  ll    long long
#define  inf   1e9
#define  pb    push_back
using namespace std;
int a[maxn],b[maxn],c[maxn],f[maxn],s[maxn],x,y,t;
void init(){
    for(int i=2;i<maxn;i++)if(!a[i])
    for(int j=i;j<maxn;j+=i){
        a[j]++;
        if((j/i)&&(j/i)%i==0){
            b[j]++;
            if((j/i/i)&&(j/i/i)%i==0)c[j]++;
        }
    }
    for(int i=2;i<maxn;i++){
        if(c[i])f[i]=0;
        else if(b[i]>1)f[i]=0;
        else if(b[i]==1)f[i]=(a[i]&1)?-1:1;
        //else if(a[i]==1)f[i]=1;
        else f[i]=(a[i]&1)?a[i]:-a[i];
        s[i]=s[i-1]+f[i];
    }
}
int main(){
    init();
    cin>>t;
    while(t--){
        scanf("%d%d",&x,&y);
        ll ans=0;
        int k=min(x,y);
        for(int i=1;i<=k;i++){
            int m=min(x/(x/i),y/(y/i))-i;
            ans+=(ll)(s[i+m]-s[i-1])*(x/i)*(y/i);
            i+=m;
            //ans+=(ll)f[i]*(x/i)*(y/i);
        }
        printf("%lld\n",ans);
    }
}
