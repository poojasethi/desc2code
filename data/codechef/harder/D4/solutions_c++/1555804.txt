#include<stdio.h>
#include<iostream>
using namespace std;
int t,n,m,N=10000000;
int p[10000020],pp;
int v[10000020];
int s[10000020];
long long g[10000020];
int main()
{
	s[1]=1;
	for(int i=2;i<=N;i++)
	{
		if(!v[i])
			v[i]=p[pp++]=i,s[i]=-1,g[i]=1;
		for(int j=0;p[j]*i<=N;j++)
		{
			v[p[j]*i]=p[j];
			if(v[i]==p[j])
			{
				g[p[j]*i]=s[i];
				break;
			}
			s[p[j]*i]=-s[i];
			g[p[j]*i]=-g[i]+s[i];
		}
	}
	for(int i=2;i<=N;i++)
		g[i]+=g[i-1];
	for(scanf("%d",&t);t--;)
	{
		scanf("%d %d",&n,&m);
		long long _=0;
		int t1,t2,j;
		for(int i=1;i<=min(n,m);i=j+1)
		{
			t1=n/i;
			t2=m/i;
			j=min(n/t1,m/t2);
			_+=(g[j]-g[i-1])*t1*t2;
		}
		printf("%lld\n",_);
	}
}
