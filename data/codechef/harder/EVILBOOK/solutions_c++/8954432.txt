#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n	15
#define DB	double
#define For(i,a,b)	for(int i=a;i<=b;i++)

int		T,N,V[n];
DB		Ans,X,SumC,SumM,C[n],M[n];

void	DFS(int k,int s,DB div,DB m,DB c){
		if	(c>=Ans)	return;
		if	(m>=666)	{Ans=c;return;}
		if	(k*X>m)		return;
		if	(m+SumM/div<666)	return;

		For(i,s,N)	if	(!V[i]&&k*X<M[i]/div){
			V[i]=1;	SumM-=M[i];
			DFS(k,i+1,div,m-k*X+M[i]/div,c+C[i]/div);
			V[i]=0;	SumM+=M[i];
		}
		DFS(k+1,1,div*3,m,c);
}

int		main(){
		scanf("%d",&T);
		for	(;T--;SumC=SumM=0)	{
			scanf("%d%lf",&N,&X);
			For(i,1,N)	scanf("%lf%lf",&C[i],&M[i]),SumC+=C[i],SumM+=M[i];
			if	(SumM<666)	puts("impossible");
				else	Ans=SumC,DFS(0,1,1,0,0),printf("%.0lf\n",Ans);
		}
}
