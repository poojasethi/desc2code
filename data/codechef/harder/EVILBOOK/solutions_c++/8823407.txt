#include <stdio.h>
#include <stdlib.h>
using namespace std;

int T,n,X,i,j,k;
int C[15],M[15];
double sumc,sum,ans;
bool use[15];

void dfs(int x,int s,double div,double magic,double cost)
{
	if(cost>=ans)return;
	if(magic>=666){ans=cost;return;}
	if(x*X>magic)return;
	if(magic+sum/div<666)return;
	for(int i=s;i<=n;++i)
	if(!use[i]&&x*X<M[i]/div)
	{
		use[i]=true;sum-=M[i];
		dfs(x,i+1,div,magic+M[i]/div-x*X,cost+C[i]/div);
		use[i]=false;sum+=M[i];
	}
	dfs(x+1,1,div*3,magic,cost);
}

int main()
{
	scanf("%d",&T);
	for(;T;--T)
	{
		scanf("%d%d",&n,&X);sum=sumc=0;
		for(i=1;i<=n;++i)scanf("%d%d",&C[i],&M[i]),sumc+=C[i],sum+=M[i];
		ans=sumc+sumc;
		dfs(0,1,1,0,0);
		if(ans>sumc)printf("impossible\n");
		else printf("%lld\n",(long long)(ans+0.5));
	}
}
