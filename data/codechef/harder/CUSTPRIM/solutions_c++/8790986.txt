#define ll long long
ll ksc(ll bas,ll k,ll p)
{
  if(log(bas)+log(k)<log(p))return bas*k;
  ll c=floor(exp(log(bas)+log(k)-log(p))),ans=bas*k-c*p;
  while(ans<0)ans+=p;
  while(ans>=p)ans-=p;
  return ans;
}
ll ksm(ll bas,ll k,ll p)
{
  if(k==0)return 1;
  if(k==1)return bas%p;
  ll ans=ksm(bas,k>>1,p);
  ans=ksc(ans,ans,p);
  if(k&1==1)ans=ksc(ans,bas,p);
  return ans;
}
bool m_r(ll x)
{
  if(x==2||x==3)return true;
  if(x<2||x%2==0)return false;
  ll s=x-1;
  int t=0;
  while(s%2==0)
    t++,s/=2;
  for(int i=1;i<=20;i++)
  {
    ll now=rand()%(x-1)+1;
    now=ksm(now,s,x);
    for(int j=1;j<=t;j++)
    {
      ll last=now;
      now=ksc(now,now,x);
      if(now==1&&last!=1&&last!=x-1)return false;
    }
    if(now!=1)return false;
  }
  return true;
}
int main()
{
  int t,a,b,c;
  scanf("%d",&t);
  srand(time(0));
  while(t--)
  {
    scanf("%d%d%d",&a,&b,&c);
    int x=33-2*a-c,y=b-a;
    if(y==0)
    {
      x=abs(x);
      if(x==2||m_r(x)&&ksm((x*11-11)%x,(x-1)/2,x)==x-1)printf("PRIME\n");
        else printf("NOT PRIME\n");
    }
      else if(m_r((ll)x*x+(ll)x*y+(ll)3*y*y))printf("PRIME\n");
        else printf("NOT PRIME\n");
  }
  return 0;
}
