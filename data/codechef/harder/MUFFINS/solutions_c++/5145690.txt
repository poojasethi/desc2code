int cc[1000];
int v[1004][1004];
bool g[1001][1001];
int n, m, k;
int del[10001];
int kk;
int vis[2001];
int v1[20006], v2[20006];

int dfs(int cur)
{
    bool poss = false;
    fori(i,0,n) {
        fori(j,0,cc[i]) {
            if(g[i][v[i][j]]) poss = true;
        }
    }
    if(!poss) return true;
    if(cur == 0) return false;
    poss = false;
    fori(i,0,n) {
        kk = 0;
        fori(j,0,cc[i]) {
            if(g[i][v[i][j]]) del[kk++] = v[i][j];
            g[i][v[i][j]] = g[v[i][j]][i] = 0;
        }
        if(kk) poss |= dfs(cur-1);
        fori(j,0,kk) {
            g[i][del[j]] = g[del[j]][i] = 1;
        }
        if(poss) return true;
    }
    return false;
}

bool solve(int pos, int cur)
{
    if(pos == m) return true;
    if(vis[v1[pos]] || vis[v2[pos]]) return solve(pos+1, cur);
    if(cur == 0) return false;

    if(!vis[v1[pos]]) {
        vis[v1[pos]] = 1;
        if(solve(pos+1, cur-1)) return true;
        vis[v1[pos]] = 0;
    }
    if(!vis[v2[pos]]) {
        vis[v2[pos]] = 1;
        if(solve(pos+1, cur-1)) return true;
        vis[v2[pos]] = 0;
    }
    return false;
}

    int t;

    while(t--) {
        S(n), S(m), S(k);

        fori(i,0,n) vis[i] = 0;
        int x, y;
        fori(i,0,m) {
            S(x), S(y);
            v1[i] = x, v2[i] = y;
        }

        if(solve(0,n-k)) {
            puts("Possible");//<< endl;
        }
        else {
            puts("Impossible");// << endl;
        }
    }
    return 0;
}
