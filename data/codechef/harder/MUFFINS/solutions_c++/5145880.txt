#include<bits/stdc++.h>
#define s(x) scanf("%d",&x)

using namespace std;

int n,m,gi,arr[1010],a[1010];

struct node {
    bool a,b;
}e[20010];

bool check(int x,int g)
{
    if(x == m) return 1;
    if(arr[e[x].a] || arr[e[x].b]) return check(x+1,g);
    if(g == 0) return 0;
    arr[e[x].a] = 1;
    bool f = check(x+1,g-1);
    arr[e[x].a] = 0;
    if(f) return 1;
    arr[e[x].b] = 1;
    f = check(x+1,g-1);
    arr[e[x].b] = 0;
    return f;
}

int main()
{
    int t;
    s(t);
    while(t--) {
        s(n); s(m); s(gi);
        for(int i=0;i<m;i++) { s(e[i].a); s(e[i].b); }

        memset(arr,0,sizeof arr);

        if(check(0,n-gi)) cout << "Possible";
        else cout << "Impossible";
        cout << "\n";
    }
}
