#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxN=1010,maxM=20100;
int n,m,e[maxM][2];
bool v[maxN];
void search(int limit) {
	int pos=0;
	for(int i=1;i<=m;i++)
	    if(!v[e[i][0]]&&!v[e[i][1]]) {
			pos=i;break;
		}
	if(pos==0)
	    throw 1;
	if(limit==0)
	    return;

    v[e[pos][0]]=true;
    search(limit-1);
    v[e[pos][0]]=false;
    v[e[pos][1]]=true;
    search(limit-1);
    v[e[pos][1]]=false;
}
int main() {
	int T;
	for(scanf("%d",&T);T;T--){
		int limit;
		scanf("%d%d%d",&n,&m,&limit);

		for(int i=1;i<=m;i++) {
			int a,b;
			scanf("%d%d",&a,&b);
			a++,b++;

			e[i][0]=a,e[i][1]=b;
		}

		memset(v,0,sizeof(v));
		bool sol=false;
		try {
			search(n-limit);
		} catch(int) {
			sol=true;
		}
		puts(sol?"Possible":"Impossible");
	}
	scanf("%*s");
}
