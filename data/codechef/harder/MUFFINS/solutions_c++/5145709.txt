int n, m , g;
int s[20001], d[20001];
int vis[20001];

bool f(int pos, int left){
    if(pos == m)
        return true;
    if(vis[s[pos]] || vis[d[pos]]){
        return f(pos+1, left);
    }
    bool ans = 0;
    if(left == 0)
        return false;
    vis[s[pos]] = 1;
    ans |= f(pos+1, left-1);
    vis[s[pos]] = 0;

    vis[d[pos]] = 1;
    ans |= f(pos+1, left-1);
    vis[d[pos]] = 0;

    return ans;
}

int main(){
    int x, y, t;
    dec;
    cin >> t;
    while(t--){
        cin >> n >> m >> g;
        mst(vis, 0);
        frl(i, 0, m){
            cin >> x >> y;
            s[i] = x, d[i] = y;
        }
        if(f(0, n-g)){
            cout << "Possible\n";
        } else{
            cout << "Impossible\n";
        }
    }
}
