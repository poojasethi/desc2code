int main()
{
    ios_base::sync_with_stdio(false);
    int n, k;
    cin>>n>>k;
    vector<int> v;
    REP(i, n)
    {
        int p; cin>>p;
        v.push_back(p);
    }
    sort(v.begin(), v.end());

    //cout<<pretemp<<endl;

    }
