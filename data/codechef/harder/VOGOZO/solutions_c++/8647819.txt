#include <bits/stdc++.h>
using namespace std;
int a[1000001];
int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;++i)
        scanf("%d",&a[i]);
    sort(a,a+n);
    int cnt=1;
    int take=a[0];
    for(int i=1;i<n;++i)
    {
        if(a[i]-take>=k)
        {
            ++cnt;
            take=a[i];
        }
    }
    cout<<cnt;
}
