#include<iostream>
#include<algorithm>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main()
{	std::ios::sync_with_stdio(false);
	int n,k;
	n=scan();
	k=scan();
	int a[n];
	for(int i=0;i<n;i++)
	a[i]=scan();
	sort(a,a+n);
	int count=1;
	int strt=a[0];
	for(int i=1;i<n;i++)
	{
		if(a[i]-strt>=k)
		{
			count++;
			strt=a[i];
		}
	}
	cout<<count<<"\n";
		
		
		
	
}
