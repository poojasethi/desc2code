#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int a[1000000 + 10];
int main()
{
	int n, k, i, ans;
	scanf("%d %d", &n, &k);
	for (i = 0; i < n; i++) scanf("%d", &a[i]);
	sort(a, a + n);
	if (n == 0){
		cout << 0 << endl;
		return 0;
	}
	ans = 1;
	int j = a[0];
	for (i = 1; i < n; i++)
		if (a[i] >= j + k){
			ans++;
			j = a[i];
		}
	cout << ans << endl;
	return 0;
 }
