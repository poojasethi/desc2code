#include<iostream>
#define MAX_SIZE 10000006
bool dr[MAX_SIZE];
using namespace std;

int main()
{

	long n,k;
	long i,j;

	for(i=0;i<MAX_SIZE;i++)
		dr[i]=false;
	cin>>n>>k;
	for(i=0;i<n;i++)
	{
		
		cin>>j;
		if(j>=MAX_SIZE)
			continue;
		dr[j]=true;
	}

	for(i=j=0;i<MAX_SIZE;i++)
	{
		if(!dr[i])
			continue;
		j++;
		i+=k-1;
	}
	cout<<j<<endl;
}
