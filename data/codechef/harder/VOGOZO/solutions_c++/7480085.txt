#include<iostream>
using namespace std;
int partition(int *a,int start,int end)
{
	int i;
	int pindex=start;
	int pivot=a[end];
	for(i=start;i<end;i++)
	{
		if(a[i]<=pivot)
		{
		swap(a[i],a[pindex]);
		pindex++;
		}
	}
	swap(a[pindex],a[end]);
	return pindex;
}
void quicksort(int *a,int start,int end)
{
	if(start<end)
	{
		int pindex=partition(a,start,end);
		quicksort(a,start,pindex-1);
		quicksort(a,pindex+1,end);
	}
}
int main()
{
	int c,i,t,n,j;
	cin>>n>>t;
	int a[n];
	for(i=0;i<n;i++)
	cin>>a[i];
	quicksort(a,0,n-1);
	c=1;
	for(i=0;i<n;i++)
	{
	    for(j=i+1;j<n;j++)
	    {
	    if((a[j]-a[i])>=t)
	    {
	        c++;
	        i=j;
	    }
	    }
	    
	}
	cout<<c<<"\n";
	return 0;
}
