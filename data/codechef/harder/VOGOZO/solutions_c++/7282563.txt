#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int *a,n,k,i,count,m;
    cin>>n>>k;
    a=new int[n];
    for(i=0;i<n;i++)
    cin>>a[i];
    sort(a,a+n);
    count=a[0];
    m=1;
    for(i=1;i<n;i++)
    {
    	if((count+k)<=a[i])
    	{
    		m++;
    		count=a[i];
    	}
    }
    cout<<m;
}
