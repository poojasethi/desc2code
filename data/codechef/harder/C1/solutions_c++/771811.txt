const int MaxN=120000;
int a[MaxN],b[MaxN],val[MaxN];

void getnum(int &tmp)
{
	char ch;

	do ch=getchar(); while(ch<'0' || ch>'9');
	tmp=0;
	while(ch>='0' && ch<='9')
	{
		tmp=tmp*10+ch-'0';
		ch=getchar();
	}
}

void putnum(int tmp)
{
	int base=1;

	do base*=10; while(base<=tmp);
	base/=10;while(base)
	{
		putchar('0'+tmp/base);
		tmp%=base;base/=10;
	}
}

void merge(int ll,int rr)
{
	if(ll==rr) return;

	int mm=(ll+rr)>>1;
	int i=mm,j=rr,pos=rr,pmax=-1;

	merge(ll,mm);merge(mm+1,rr);

	for(int i=mm+1;i<=rr;i++)
	{
		pmax=max(pmax,a[i]);
		val[i]=pmax;
	}

	while(i>=ll && j>mm)
	{
		if(a[i]>val[j]+1) b[pos--]=a[i--];
		else b[pos--]=a[j--];
	}
	while(i>=ll) b[pos--]=a[i--];
	while(j>mm) b[pos--]=a[j--];
	for(i=ll;i<=rr;i++) a[i]=b[i];
}

void solve()
{
	int n;

	getnum(n);
	for(int i=0;i<n;i++) getnum(a[i]);
	merge(0,n-1);
	for(int i=0;i<n;i++) 
	{
		putnum(a[i]);
		putchar(i==n-1?'\n':' ');
	}
}

int main()
{
	int test;

	getnum(test);
	for(int num=0;num<test;num++)
		solve();
	return 0;
}
