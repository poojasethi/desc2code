#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>
#include <cstdio>

using namespace std;

int A[100000];
int B[100000];
pair<int, int> seg[100000];

void srt(int a, int b) {
  if(a + 50 >= b) {
    for(int i = a + 1; i < b; i++) {
      for(int j = i; j > a && A[j] + 1 < A[j - 1]; j--) {
        swap(A[j], A[j - 1]);
      }
    }
    return;
  }

  if(a + 1 >= b) return;
  int m = a + (b - a) / 2;
  srt(a, m);
  srt(m, b);

  int mn = 0x7FFFFFFF;
  int segst = b - a;
  for(int i = m - 1; i >= a; i--) {
    if(A[i] < mn) {
      mn = A[i];
      seg[--segst] = make_pair(mn, i + 1);
    }
  }
  seg[--segst] = make_pair(-1, a);

  int bsz = 0;
  int aa = a;
  int bb = segst + 1;
  for(int i = m; i < b; i++) {
    while(bb < b - a && A[i] + 1 >= seg[bb].first) bb++;
    int pos = seg[bb - 1].second;

    while(aa < pos) B[bsz++] = A[aa++];
    B[bsz++] = A[i];
  }

  memcpy(A + b - (m - aa), A + aa, (m - aa) * sizeof(int));
  memcpy(A + a, B, sizeof(int) * bsz);
}

int main() {
  int T; scanf("%d", &T);
  for(int t = 1; t <= T; t++) {
    int N; scanf("%d", &N);
    for(int i = 0; i < N; i++) scanf("%d", A + i);
    srt(0, N);
    for(int i = 0; i < N; i++) {
      if(i) printf(" ");
      printf("%d", A[i]);
    }
    printf("\n");
  }
}
