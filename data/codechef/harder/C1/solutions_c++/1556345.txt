#include<stdio.h>
#include<iostream>
#include<vector>
using namespace std;
int t[100020];
int a[100020];
int b[100020];
int o,n;
void F(int l,int r)
{
	if(r-l<2)
		return;
	int m=l+r>>1;
	F(l,m),F(m,r);
	t[m-1]=a[m-1];
	for(int i=m-2;i>=l;i--)
		t[i]=min(a[i],t[i+1]);
	int p=l,p1=l,p2=m;
	while(p<r)
		if(p1==m||p2<r&&a[p2]<t[p1]-1)
			b[p++]=a[p2++];
		else
			b[p++]=a[p1++];
	for(int i=l;i<r;i++)
		a[i]=b[i];
}
int main()
{
	for(scanf("%d",&o);o--;)
	{
		scanf("%d",&n);
		for(int i=0;i<n;i++)
			scanf("%d",a+i);
		F(0,n);
		for(int i=0;i<n;i++)
			printf("%d ",a[i]);
		puts("");
	}
}
