int a[N],q[N],t[N];
void merge(int *p,int l,int m,int r)
{
 t[m]=p[m];
 for(int i=m-1;i>=l;i--)t[i]=min(t[i+1],p[i]);
 if(p[m]<=p[m+1]+1)return;
 int x=l,y=m+1,k=l;
 while(x<=m&&y<=r)
 {
  if(t[x]>p[y]+1)q[k++]=p[y++];
  else q[k++]=p[x++];
 }
 while(x<=m)q[k++]=p[x++];
 while(y<=r)q[k++]=p[y++];
 for(k=l;k<=r;k++)p[k]=q[k];
}
void merge_sort(int *a,int s,int e)
{
 if(e-s<=0)return;
 if(e-s==1)
 {
  if(a[s]>a[e]+1)swap(a[s],a[e]);
  return;
 }
 int m=(s+e)>>1;
 merge_sort(a,s,m);
 merge_sort(a,m+1,e);
 merge(a,s,m,e);
}
