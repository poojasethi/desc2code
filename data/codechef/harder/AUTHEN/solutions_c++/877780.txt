#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int n,k,last[200],mod=1009419529;
long long f[10110][110];
char ch[10110];
void solve(){
	scanf("%d%d",&n,&k);n+=k;
	scanf("%s",ch+1);
	memset(last,0,sizeof(last));
	f[0][0]=1;
	for(int i=1;i<=n;++i){
		f[i][0]=1;
		for(int j=1;j<=k;++j){
			f[i][j]=f[i-1][j-1]+f[i-1][j];
			if(last[ch[i]]&&i-last[ch[i]]<=j)f[i][j]-=f[last[ch[i]]-1][j-i+last[ch[i]]];
			f[i][j]%=mod;
		}
		last[ch[i]]=i;
	}
	printf("%lld\n",(f[n][k]-1+mod)%mod);
}
int main(){
	int T;scanf("%d",&T);
	for(int i=1;i<=T;++i)solve();
	return 0;
}
