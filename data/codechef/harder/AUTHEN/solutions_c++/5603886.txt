#include <cstdio>
#include <iostream>

using namespace std;

typedef long long ll;

const ll mod= 1009419529;
ll n, k;
char a[10105];
ll next[30][10105];
ll f[10105][105];

ll add(ll x, ll y){
    if(x+y>=mod) return (x+y-mod);
    else return (x+y);
}

ll sub(ll x, ll y){
    if(x-y<0) return (x-y+mod);
    else return x-y;
}

int main(){
    ios_base::sync_with_stdio(false);
    ll test;
    cin >> test;
    while(test--){
        cin >> n >> k;
        cin >> (a+1);
        ll i, j, t;
        for(i=1;i<=26;i++){
            next[i][n+k]= 1000000000;
            for(j=n+k;j>=1;j--){
                if(a[j]-'a'+1==i) next[i][j-1]= j;
                else next[i][j-1]= next[i][j];
            }
        }
        for(i=0;i<=n+k;i++) next[27][i]= n+k+1;

        for(i=n+k;i>=0;i--){
            for(j=0;j<=k;j++){
                if(i==n+k){
                    if(j==0) f[i][j]= 1LL;
                    else f[i][j]= 0LL;
                }
                else{
                    if(j==0) f[i][j]= 1LL;
                    else{
                        f[i][j]= f[i+1][j-1];
                        if(next[a[i+1]-'a'+1][i+1]-(i+1)-1<=j-1)
                            f[i][j]= sub(f[i][j],f[next[a[i+1]-'a'+1][i+1]][j-1-(next[a[i+1]-'a'+1][i+1]-(i+1)-1)]);
                        if(next[a[i+1]-'a'+1][i]-i-1<=j)
                            f[i][j]= add(f[i][j],f[next[a[i+1]-'a'+1][i]][j-(next[a[i+1]-'a'+1][i]-i-1)]);
                    }
                }
            }
        }

        cout << (f[0][k]+mod-1)%mod << "\n";
    }
    return 0;
}
