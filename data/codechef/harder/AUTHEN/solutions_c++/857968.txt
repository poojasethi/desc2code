#include <stdio.h>
#include <string.h>
#include <iostream>
#include <string>
using namespace std;

#define MOD 1009419529

int tests,N,K;
string s;

int f[10101][101];
int g[10101];
int h[30][10101];

int main() {
	cin >> tests;
	while (tests--) {
		cin >> N >> K;
		cin >> s;
		memset(f,0,sizeof(f));
		memset(g,0,sizeof(g));
		memset(h,0,sizeof(h));
		f[0][0]=1;
		g[0]=1;
		int r=0;
		for (int n=1;n<=N+K;n++) {
			int ch=s[n-1]-'a';
			for (int k=0;k<=min(n-1,K);k++) {
				f[n][k]=(g[n-k-1]-h[ch][n-k])%MOD;
				if (n-k==N) r=(r+f[n][k])%MOD;
				g[n-k]=(g[n-k]+f[n][k])%MOD;
				h[ch][n-k]=(h[ch][n-k]+f[n][k])%MOD;
			}
		}
		printf("%d\n",(r-1+MOD)%MOD);
	}
	return 0;
}
