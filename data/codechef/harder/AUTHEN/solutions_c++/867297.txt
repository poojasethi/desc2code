#include <stdio.h>
#define MOD 1009419529
#define MAXN 10000
#define MAXK 100

int f[MAXN+MAXK+2][MAXK+2];
int g[MAXN+MAXK+2][MAXK+2];
int last [256];
char w[MAXK+MAXN+3];
int main(){
	int t;
	int n;
	int k;
	int lim;
	scanf("%d", &t);
	while(t--){
		scanf("%d %d", &n, &k);
		scanf("%s", w+1);
		lim=n+k;
		for(int i = 0; i <=lim; i++){
			for(int j = 0; j <= k; j++){
				f[i][j] = 0;
				g[i][j] = 0;
			}
		}
		for(int i = 0; i<=255; i++)
			last[i]=-1;
		g[0][0]=1;
		f[0][0]=1;
		for(int i = 1; i<=lim;i++){
			for(int j=0;j<=i && j<=k;j++){
				f[i][j]=g[i-1][j];
				if(last[w[i]]>=(i-j)){
					f[i][j]-=g[last[w[i]]-1][j-(i-last[w[i]])];
					if(f[i][j]<0)
						f[i][j]+=MOD;
				}
				g[i][j]=f[i][j];
				if(j>0){
					g[i][j]+=g[i-1][j-1];
					if(g[i][j]>=MOD)
						g[i][j]-=MOD;
				}
			}
			last[w[i]]=i;
		}
		int ans=MOD-1;
		for(int i=n;i<=lim;i++){
			ans+=f[i][i-n];
			if(ans>=MOD)
				ans-=MOD;
		}
		printf("%d\n", ans);
	}
	return 0;
}
