#include <ctime>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n	2000005
#define I64	long long
#define For(i,a,b)	for(int i=a;i<=b;i++)

I64		S[n];
int		N,K,A[n],B[n],H[n],G[n],ID[n];

bool	Cmp(int a,int b)	{return	G[a]<G[b];}

int		main(){
	//	freopen("1.in","r",stdin);
	//	freopen("1.out","w",stdout);

		int pp=clock();

		scanf("%d%d",&N,&K);
		For(i,1,N)	scanf("%d",&G[i]),ID[i]=i;
		sort(ID+1,ID+N+1,Cmp);
		For(i,1,N)	A[i]=G[ID[i]],S[i]=S[i-1]+A[i];

		For(i,0,(1<<2*K-1)-1)	{
			H[i]=H[i>>1]+(i&1);
			if	(H[i]==K-1)	B[++*B]=i;
		}

		For(i,1,N-K+1)	For(j,i+K,N-K+1)
		if	(S[i+K-2]-S[i-1]>A[i+K-1]&&S[j+K-2]-S[j-1]>A[j+K-1]){
			puts("Yes");
			For(t,i,i+K-1)	printf("%d ",ID[t]);
			For(t,j,j+K-1)	printf("%d ",ID[t]);	return	0;
		}

		For(i,1,N-2*K+1)	if	(S[i+2*K-2]-S[i-1]>A[i+2*K-1])
		For(j,1,*B){
			int s=B[j],z=0;	I64 x=0,y=0;
			For(k,1,2*K-1)	if	((1<<k-1)&s)	x+=A[i+k-1];
				else	y+=A[i+k-1],z=max(z,A[i+k-1]);
			if	(x>A[i+2*K-1]&&y-z>z)	{
				puts("Yes");
				For(t,i,i+2*K-1)	if	((1<<t-i)&s)	printf("%d ",ID[t]);	printf("%d ",ID[i+2*K-1]);
				For(t,i,i+2*K-1)	if	(!((1<<t-i)&s))	printf("%d ",ID[t]);
				return	0;
			}
		}
		puts("No");
}
