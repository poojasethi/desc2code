#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<vector>
using namespace std;
struct data{
	long long int val;
	int ind;
};
bool func(struct data a,struct data b){
	return (a.val<b.val);
}
int main(){
	int n,k,i,j,l;
	scanf("%d%d",&n,&k);
	vector<data> a(1111);
	for (i=1;i<=n;i++){
		scanf("%lld",&a[i].val);
		a[i].ind=i;
	}
	sort(a.begin()+1,a.begin()+n+1,func);
	vector<int> s;
	int f_last=0;
	for (i=k;i<=n;i++){
		long long int sum=0;
		for (j=i-k+1;j<i;j++) sum=sum+a[j].val;
		if (sum>a[i].val){
			for (j=i-k+1;j<=i;j++) s.push_back(a[j].ind);
			for (l=0;l<k;l++) a.erase(a.begin()+(i-k+1));
			f_last=1;
			break;
		}
	}
	if (f_last!=0){
		for (i=k;i<=n-k;i++){
			long long int sum=0;
			for (j=i-k+1;j<i;j++) sum=sum+a[j].val;
			if (sum>a[i].val){
				for (j=i-k+1;j<=i;j++) s.push_back(a[j].ind);
				break;
			}
		}
	}
	if ((int)s.size()==2*k){
		sort(s.begin(),s.begin()+k);
		sort(s.begin()+k,s.end());
		printf("Yes\n");
		for (i=0;i<2*k;i++){
			if (i==(2*k-1)) printf("%d\n",s[i]);
			else printf("%d ",s[i]);
		}
	}else printf("No\n");
	return 0;
}
