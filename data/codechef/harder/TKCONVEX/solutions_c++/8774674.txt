#include<cstdio>
#include<cstring>
#include<algorithm>
#include<set>
#include<map>
#include<vector>
#define rep(i,j,k) for(int i=(int)j;i<=(int)k;i++)
#define per(i,j,k) for(int i=(int)j;i>=(int)k;i--)
using namespace std;
typedef long long LL;
typedef double db;
const int N=1005;
int a[N];
int id[N];
int n,k;
LL t1[N];int b1=0;
LL t2[N];int b2=0;
int d1[N],d2[N];
inline LL sum(int l,int r){
	LL ret=0;
	rep(i,l,r)ret+=1ll*a[i];
	return ret;
}
inline bool cmp(const int &x,const int &y){return a[x]<a[y];}
inline void check(){
	if((t1[0]>t1[b1]*2ll)&&(t2[0]>t2[b2]*2ll)){
		printf("Yes\n");
		rep(i,1,k)printf("%d ",d1[i]);
		rep(i,1,k)printf("%d ",d2[i]);
		exit(0);
	}
}
void dfs(int l,int r){
	if(l>r){
		check();
		return;
	}
	if(b1<k){
		t1[++b1]=a[l];
		t1[0]+=1ll*t1[b1];
		d1[b1]=id[l];
		dfs(l+1,r);
		t1[0]-=1ll*t1[b1];
		--b1;
	}
	if(b2<k){
		t2[++b2]=a[l];
		t2[0]+=1ll*t2[b2];
		d2[b2]=id[l];
		dfs(l+1,r);
		t2[0]-=1ll*t2[b2];
		--b2;
	}
}
int main(){
	scanf("%d%d",&n,&k);
	rep(i,1,n)scanf("%d",&a[i]);
	rep(i,1,n)id[i]=i;
	
	sort(id+1,id+1+n,cmp);
	sort(a+1,a+1+n);
	
	int l1=n+5;
	rep(i,k,n)if(sum(i-k+1,i-1)>a[i]){l1=i;break;}
	int r1=-5;
	per(i,n,k)if(sum(i-k+1,i-1)>a[i]){r1=i-k+1;break;}
	if(r1>l1){
		printf("Yes\n");
		rep(i,l1-k+1,l1)printf("%d ",i);
		rep(i,r1,r1+k-1)printf("%d ",i);
		return 0;
	}
	
	
	rep(i,2*k,n){
		t1[0]=b1=t2[0]=b2=0;
		dfs(i-2*k+1,i);
	}
	printf("No\n");
	return 0;
}
