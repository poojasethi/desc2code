#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;

int n,k,i,j,l;
int a[1005],b[1005],c[1005],d[1005];
long long sum[1005];

inline bool cmp(const int &x,const int &y){return b[x]<b[y];}

void dfs(int x,int nA,int nB,long long sumA,long long sumB,int maxA,int maxB)
{
	if(nA>k||nB>k)return;
	if(x>l)
	{
		if(sumA-maxA>maxA&&sumB-maxB>maxB)
		{
			printf("Yes\n");
			for(i=x-k-k;i<x;++i)if(d[i]==1)printf("%d ",c[i]);
			for(i=x-k-k;i<x;++i)if(d[i]==2)printf("%d ",c[i]);
			printf("\n");
			exit(0);
		}
		return;
	}
	d[x]=1;dfs(x+1,nA+1,nB,sumA+a[x],sumB,a[x],maxB);
	d[x]=2;dfs(x+1,nA,nB+1,sumA,sumB+a[x],maxA,a[x]);
}

int main()
{
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;++i)scanf("%d",&b[i]),c[i]=i;
	sort(c+1,c+n+1,cmp);
	for(i=1;i<=n;++i)a[i]=b[c[i]];
	for(i=1;i<=n;++i)sum[i]=sum[i-1]+a[i];
	for(i=k;i<=n;++i)
	if(a[i]<sum[i-1]-sum[i-k])
	break;
	if(i>n)
	{
		printf("No\n");
		return 0;
	}
	for(l=i,i+=k;i<=n;++i)
	if(a[i]<sum[i-1]-sum[i-k])
	{
		printf("Yes\n");
		for(j=l-k+1;j<=l;++j)printf("%d ",c[j]);
		for(j=i-k+1;j<=i;++j)printf("%d ",c[j]);
		printf("\n");
		return 0;
	}
	for(i=k+k;i<=n;++i)
	{
		l=i;
		d[i-k-k+1]=1;
		dfs(i-k-k+2,1,0,a[i-k-k+1],0,a[i-k-k+1],0);
	}
	printf("No\n");
}
