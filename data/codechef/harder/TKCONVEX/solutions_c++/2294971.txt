#include<bits/stdc++.h>
#define f(i,x,y) for (int i = int(x); i < int(y); i++)
#define fd(i,x,y) for(int i = int(x); i>= int(y); i--)
#define FOR(it,A) for(typeof A.begin() it = A.begin(); it!=A.end(); it++)
#define all(v) (v).begin(), (v).end()
#define rall(v) (v).rbegin(), (v).rend()
#define vint vector<int>
#define pll pair<ll,ll>
#define ll long long
#define clr(A,x) memset(A, x, sizeof A)
#define pb push_back
#define pii pair<int,int>
#define fst first
#define snd second
#define ones(x) __builtin_popcount(x)
#define cua(x) (x)*(x)
#define oo 10000000
#define MAX 1000009
#define MOD 1000000007

using namespace std;



int main(){
//    freopen("out.c","w",stdout);
    int n,x,k;
    scanf("%d%d",&n,&k);
    vector<pii> save;
    f(i,0,n){
        scanf("%d",&x);
        save.pb(pii(x,i+1));
    }
    sort(all(save));
    bool cmp[1009];
    clr(cmp,false);
    f(i,k-1,n){
        ll cont = 0;
        fd(j,i-1,i-k+1) cont+=save[j].fst;
        if(cont>(ll)save[i].fst)    cmp[i] = true;
    }
    int izq = -1,der = -1;
    f(i,0,n) if(cmp[i]==true){
            izq = i;
            break;
    }

    fd(i,n-1,0) if(cmp[i]==true && izq!=i){
            der = i;
            break;
    }
    //cout<<izq<<"--"<<der<<endl;
    vint ans;
    if(der!=-1 && izq!=-1){
        if(der-izq>=k){//la solucion 1
            fd(i,der,der-k+1){
                ans.pb(save[i].snd);
            }
            fd(i,izq,izq-k+1){
                ans.pb(save[i].snd);
            }
            cout<<"Yes"<<endl;
            f(i,0,ans.size()){
                if(i!=0) printf(" ");
                printf("%d",ans[i]);
            }
            return 0;
        }else{//la solucion 2
            vector<pii> temp;
            if((der-(2*k)+1)>=0){
                f(i,(der-(2*k)+1),der+1){
                    temp.pb(save[i]);
                }
            }else{
                cout<<"No"<<endl;
                return 0;
            }
            f(i,0,(1<<(2*k))){
                vector<pii> v1,v2;
                f(j,0,2*k){
                    if(i&(1<<j))   v1.pb(temp[j]);
                    else    v2.pb(temp[j]);
                }
                if(v2.size()==k && v1.size()==k){
                    ll cont1 = 0,cont2 = 0;
                    f(i,0,k-1){
                        cont1+=v1[i].fst;
                    }
                    f(i,0,k-1){
                        cont2+=v2[i].fst;
                    }
                    if(cont1>(ll)v1[v1.size()-1].fst && cont2>(ll)v2[v2.size()-1].fst){
                        f(i,0,k)    ans.pb(v1[i].snd);
                        f(i,0,k)    ans.pb(v2[i].snd);
                        cout<<"Yes"<<endl;
                        f(i,0,ans.size()){
                            if(i!=0) printf(" ");
                            printf("%d",ans[i]);
                        }
                        return 0;
                    }

                }

            }
        }
    }
    cout<<"No"<<endl;



    return 0;
}
