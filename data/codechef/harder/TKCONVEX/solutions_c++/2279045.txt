#include<iostream>
#include<vector>
#include<cstdio>
#include<algorithm>

using namespace std;

#define ll long long int

vector<int> first;
vector<int> second;

int n,k;
int flag=0;

void solve(vector <pair<ll,int> > len)
{

    for(int i=0;i<len.size();i++)
    {
        int j;
        ll sum=0;

        if(len.size()>=(i+k))
        {
            for(j=i;j<(i+k);j++)
                sum+=len[j].first;

            if(len[j-1].first<(sum-len[j-1].first))
            {
                flag=1;

                while(first.size()!=k)
                {
                    first.push_back(len[j-1].second);
                    len.erase(len.begin()+(j-1));
                    j--;
                }
                break;
            }
        }
        else
            break;
    }

    if(!flag)
        return;


    for(int i=0;i<len.size();i++)
    {
        int j;
        ll sum=0;
        if(len.size()>=(i+k))
        {
            for(j=i;j<(i+k);j++)
                sum+=len[j].first;

            if(len[j-1].first<(sum-len[j-1].first))
            {
                while(second.size()!=k)
                {
                    second.push_back(len[j-1].second);
                    len.erase(len.begin()+(j-1));
                    j--;
                }
                break;
            }
        }
        else
            break;
    }

}


int main()
{
    cin>>n>>k;

    vector<pair<ll,int> > v;

    for(int i=0;i<n;i++)
    {
        ll temp;
        cin>>temp;
        v.push_back(make_pair(temp,i+1));
    }

    sort(v.begin(),v.end());

    solve(v);

    if(first.size()==0||second.size()==0)
        cout<<"No\n";

    else
    {
        cout<<"Yes\n";

        for(int i=0;i<first.size();i++)
            cout<<first[i]<<" ";
        for(int i=0;i<second.size();i++)
            cout<<second[i]<<" ";
    }


    return 0;
}
