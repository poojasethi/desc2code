#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
const int n=1005,m=100005;
int N,K,i,j,k,T,flag,X,Cnt;
long long Sum,sum;
int W[m],f[n];
long long S[n];
pair<int,int> A[n];
int Count(int i)
{
	int sum=0;
	for(;i;i-=i&-i) ++sum;
	return sum;
}
int main()
{
	scanf("%d%d",&N,&K);
	for(i=1;i<=N;++i) scanf("%d",&X),A[i]=make_pair(X,i);
	sort(A+1,A+N+1);
	for(i=1;i<=N;++i) S[i]=S[i-1]+(long long)A[i].first;
	for(i=K;i<=N;++i)
		if(S[i-1]-S[i-K]>A[i].first)
			f[i]=1;
	for(i=K;i<=N;++i)
		for(j=i+K;j<=N;++j)
			if(f[i]&&f[j])
			{
				printf("Yes\n");
				for(k=i-K+1;k<=i;++k) printf("%d ",A[k].second);
				for(k=j-K+1;k<j;++k) printf("%d ",A[k].second);
				printf("%d\n",A[j].second);
				return 0;
			}
	for(i=0;i<1<<2*K-1;++i)
		if(Count(i)==K)
			W[++Cnt]=i;
	for(i=2*K;i<=N;++i)
		if(f[i])
		{
			Sum=0;
			for(j=i-2*K+1;j<i;++j)
				Sum+=(long long)A[j].first;
			for(j=1;j<Cnt;++j)
			{
				sum=T=0;
				for(k=0;k<2*K-1;++k)
					if(W[j]&(1<<k))
					{
						++T;
						if(T==K)
						{
							if(A[i-2*K+1+k].first<sum) flag=1;
							else flag=0;
							sum+=(long long)A[i-2*K+1+k].first;
							break;
						}
						sum+=(long long)A[i-2*K+1+k].first;
					}
				if(Sum-sum>A[i].first&&flag)
				{
					printf("Yes\n");
					for(k=0;k<2*K-1;++k)
						if(W[j]&(1<<k))
							printf("%d ",A[i-2*K+1+k].second);
					for(k=0;k<2*K-1;++k)
						if(!(W[j]&(1<<k)))
							printf("%d ",A[i-2*K+1+k].second);
					printf("%d\n",A[i].second);
					return 0;
				}
			}
		}
	printf("No\n");
	return 0;
}
