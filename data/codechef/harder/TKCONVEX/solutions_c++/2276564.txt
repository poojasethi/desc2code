#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

pair<long long int,int>a[1005],b[1005];
long long int sum[1005];
int f[1005];
int main()
{
    int n,k,i,j,flag,l,pos;
    long long int s,cnt;
    scanf("%d%d",&n,&k);
    sum[0]=0;
    for(i=1;i<=n;i++)
    {
        scanf("%lld",&a[i].first);
        a[i].second=i;
    }
    //cout<<"e";
    sort(a,a+n+1);
    for(i=1;i<=n;i++)
    sum[i]=sum[i-1]+a[i].first;
    for(i=k;i<=n;i++)
    {
        s=a[i].first;
        cnt=k-1;
        if(sum[i-1]-sum[i-1-cnt]>s)
        {
            f[i]=1;
            flag=1;
        }
        else
        continue;
        for(j=i-1;j>=0;j--)
        {
            if(sum[j]-sum[j-cnt]>s)
            {
                if(j-1>=1&&j-1-cnt>=0)
                {
                if(sum[j-1]-sum[j-1-cnt]>s)
                continue;
                else
                {
                    f[j]=1;
                    s=s-a[j].first;
                    cnt--;
                    if(cnt==0)
                    break;
                }
                }
                else
                {
                    for(l=1;l<=j;l++)
                    f[l]=1;
                    break;
                }
            }
            else
            break;
        }
        if(flag==1)
        break;
    }
    if(flag==1)
    {
    l=1;
    sum[0]=0;
    for(i=1;i<=n;i++)
    {
        if(f[i]==0)
        {
            b[l].first=a[i].first;
            b[l].second=a[i].second;
            sum[l]=sum[l-1]+b[l].first;
            l++;
        }
    }
    for(i=k;i<l;i++)
    {
        if(sum[i-1]-sum[i-1-(k-1)]>b[i].first)
        {
            flag=2;
            pos=i;
            break;
        }
    }
    }
    if(flag!=2)
    printf("No");
    else
    {
        printf("Yes\n");
        for(i=1;i<=n;i++)
        {
            if(f[i]==1)
            printf("%d ",a[i].second);
        }
        for(i=pos-(k-1);i<=pos;i++)
        {
            printf("%d ",b[i].second);
        }
    }
    return 0;
}
