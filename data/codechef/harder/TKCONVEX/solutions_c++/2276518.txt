#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
 
using namespace std;
 
struct node {
       long long x;
       long long y;
};
 
int main()
{
    long long n,k,p1=-1,j,p2=-1,i,sum=0,c[1002];
    vector <node> v;
    pair<long long,long long> a[1002];
    c[0] = 0;
    scanf("%lld%lld", &n, &k);
    a[0].first = -100000000;
    a[0].second = -2;
    
    for ( i = 1; i <= n; i++ ) {
        scanf("%lld", &a[i].first);
        a[i].second = i;
    }
    
    sort(a,a+n);
    for ( i = 1; i <= n; i++ ) {
        c[i] = c[i-1] + a[i].first;
    }
    
    for ( i = k; i <= n; i++ ) {
        sum = c[i]-c[i-k];
        if ( sum > 2*a[i].first ) {
             p1 = i;
             break;
        }
    }
    
    if ( p1 == -1 ) {
         printf("No\n");
         return 0;
    }
    
 
    long long lim = 1;
    c[0] = 0;
    
    node t;
    t.x = t.y = 0;
    v.push_back(t);
    
    for ( i = 1; i <= n; i++ ) {
        if ( i >= p1-k+1 && i <= p1 ) continue;
        else {
             t.x = a[i].first;
             t.y = a[i].second;
             v.push_back(t);
             c[lim] = c[lim-1] + a[i].first;
             lim++;
        }
    }   
 
    for ( i = k; i < lim; i++ ) {
        sum = c[i]-c[i-k];
        if ( sum > 2*v[i].x ) {
             p2 = i;
             break;
        }
    }
 
    if ( p2 == -1 ) {
         printf("No\n");
         return 0;
    }    
    
    printf("Yes\n");
    for ( i = p1-k+1; i <= p1; i++ ) printf("%lld ", a[i].second);
             
    for ( i = p2; i >= p2-k+1; i-- ) printf("%lld ", v[i].y);
    printf("\n");
    
    return 0;
}
