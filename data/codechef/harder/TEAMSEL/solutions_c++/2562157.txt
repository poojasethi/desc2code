#include <cstdio>

#define MAX 50000

typedef unsigned long long ll;

int main(){
	int T;
	scanf("%d", &T);

	while(T--){
		int N;
		scanf("%d", &N);

		int skill[N+1], sum = 0;
		for(int i = 1; i <= N; i++) scanf("%d", &skill[i]), sum += skill[i];
		
		if (N == 1) 
			printf("0 %d\n", sum);
		else{
			int sum_req = sum/2;

			ll mat[MAX] = {};
			for(int i = 1; i <= N; i++){
				for(int j = sum_req; j >= 0; j--)
					if (j - skill[i] >= 0) mat[j] |= (mat[j-skill[i]] << 1);
				mat[skill[i]] |= 1;
			}

			ll max_card = ((ll)1 << (N/2-1));
		       	if (N&1) max_card |= max_card << 1;
			int j;
			for(j = sum_req; j >= 0 && !(mat[j]&max_card); j--);
			printf("%d %d\n", j, sum-j);
		}
		if (T) printf("\n");
	}

	return 0;
}
