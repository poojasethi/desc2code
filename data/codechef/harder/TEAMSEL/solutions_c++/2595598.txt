int main()
{
    int t;
    cin>>t;
    while(t--)
    {
	int n,a[100],s=0;
	cin>>n;
	for(int i=0;i<n;i++)
	{
	    cin>>a[i];
	    s+=a[i];
	}
	ull b[45001]={0};
	b[0]=1;
	int ma=0;
	for(int i=0;i<n;i++)
	{
	    for(int j=ma;j>=0;j--)
	    {
		b[j+a[i]]|=(b[j]<<1);
		ma=max(ma,j+a[i]);
	    }
	}
	int ansi=-1,ansm=500000;
	for(int i=0;i<=s;i++)
	{
	    if((b[i]>>(n/2))&1)
	    {
		if(ansm>abs(s-2*i))
		{
		    ansm=abs(s-2*i);
		    ansi=i;
		}
	    }
	    if((b[i]>>((n+1)/2))&1)
	    {
		if(ansm>abs(s-2*i))
		{
		    ansm=abs(s-2*i);
		    ansi=i;
		}
	    }
	}
	cout<<min(ansi,s-ansi)<<" "<<max(ansi,s-ansi)<<endl<<endl;
    }
    return 0;
}
