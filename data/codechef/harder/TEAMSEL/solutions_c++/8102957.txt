#include<bits/stdc++.h>

using namespace std;

#ifndef ONLINE_JUDGE
    #define gc getchar
    #define pc putchar
#else
    #define gc getchar_unlocked
    #define pc putchar_unlocked
#endif 

inline void fr(int &x) {
    register int c = gc();
	x = 0; 
    for(; (c<48 || c>57); c = gc());
    for(; c>47 && c<58 ; c = gc()) {
        x = (x<<1)+(x<<3) + c - 48;
}}
inline void fp(int n){
    if (n / 10 != 0)
        fp(n / 10);
    pc((n % 10) + '0');
} 

int a[105],tym=0,dp[25001][101][51],parallel[25001][101][51];

bool ans(int sum,int pos,int left){
	if(sum<0||pos<0)return 0;
	if(parallel[sum][pos][left]==tym)return dp[sum][pos][left];
	parallel[sum][pos][left]=tym;
	if(left==0)return dp[sum][pos][left]=(sum==0);
	if(pos<left)return dp[sum][pos][left]=0;
	return dp[sum][pos][left]=ans(sum,pos-1,left)||ans(sum-a[pos],pos-1,left-1);
}

int main(){
	int t,n,temp,sum;
	fr(t);
	while(t--){
		++tym;
		fr(n);
		sum=0;
		for(int i=1;i<=n;++i)fr(a[i]),sum+=a[i];
		for(temp=(sum>>1);1;--temp){
			if(ans(temp,n,(n>>1))||ans(temp,n,((n+1)>>1)))break;
		}
		printf("%d %d\n",temp,sum-temp);
		if(t)pc('\n');
	}
	return 0;	
}
