#include <bits/stdc++.h>

using namespace std;

//#define top 0xFFFFFFFFFF

void solve(int*arr, int n) {
    int mini = 100000000;
    for(int i=0;i<n;++i)
        mini = min(mini,arr[i]);

    bool flag = 0;
    if(mini < 0) {
        flag = 1;
        mini = abs(mini);
        for(int i=0;i<n;++i)
            arr[i]+=mini;
    }
    else
        mini = 0;

    int sum2 = (n&1)?mini:0;
    for(int i=0;i<n;++i)
        sum2+=arr[i];

    int sum = ceil(sum2/(2*1.0));
    int noe = ceil(n/(2*1.0));
    long long x[sum+1];
    memset(x,0,sizeof(x));
    x[0] = 1;
    for(int i=0;i<n;++i) {
        for(int j=sum;j>=arr[i];--j) {
            x[j] = (x[j-arr[i]]<<1)|x[j];
            //x[j] = x[j]&top;
        }
    }
    if(n%2) {
        for(int j=sum;j>=0;--j) {
            if(j-mini>=0)
            {
                x[j] = (x[j-mini]<<1)|x[j];
                //x[j] = x[j]&top;
            }
        }
    }

    for(int i=sum;i>=0;--i) {
        if(x[i]&(1LL<<(noe))) {
                if(flag)
                {
                    int x1 = i-(mini*noe);
                    int y1 = sum2-i-(mini*(noe));
                    if(x1>y1)
                        printf("%d %d\n",y1,x1);
                    else
                        printf("%d %d\n",x1,y1);
                }
                else
                {
                    if(i<sum2-i)
                        printf("%d %d\n",i,sum2-i);
                    else
                        printf("%d %d\n",sum2-i,i);
                }
            break;
        }
    }
}

int main() {
    int t;
    cin>>t;
    while(t--) {
        int n;
        cin>>n;
        int *arr = new int[n];
        for(int i=0;i<n;++i)
            cin>>arr[i];
        /*int result = */solve(arr,n);
    }
}
