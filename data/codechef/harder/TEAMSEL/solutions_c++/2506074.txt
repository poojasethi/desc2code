#include<bits/stdc++.h>
using namespace std;
#define s(n) scanf("%d",&n);
unsigned long long dp[100000];
//dp[cardinity][sum];
int main()
{
    //freopen("in","r",stdin);
    //freopen("out","w",stdout);

    int N,test;
    s(test);
    while(test--){
    s(N);
    vector<int> v(N);
    int sum=0;
    for(int i=0;i<N;i++)
    {
        s(v[i]);
        sum+=v[i];
    }
    for(int i=0;i<=sum;i++)
            dp[i]=0;

    for(int i=0;i<N;i++)
    {
        for(int j=sum/2;j>=0;j--)
        {
            if(j>=v[i])
            {
                dp[j]|=(dp[j-v[i]]<<1);
            }
        }
        dp[v[i]]|=1;
    }

    int fst=N/2;
    int scd=N-fst; 
    int diff=INT_MAX;
    int ans;

    for(int i=0;i<=sum;i++)
    {
        if(dp[i]&(1LL<<(fst-1))||dp[i]&(1LL<<(scd-1)))
        {
            if(abs(sum-2*i)<diff)
            {
                ans=i;
                diff=abs(sum-2*i);
            }
        }
    }
    //cout<<"ans "<<" "<<ans<<endl;

    printf("%d %d\n",min(ans,sum-ans),max(ans,sum-ans));
    }
    return 0;
}
