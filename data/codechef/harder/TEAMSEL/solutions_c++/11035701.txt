#include <bits/stdc++.h>
#define maxn 101

using namespace std;

int n, m, a[maxn];
int t[maxn][maxn];
int ans;

void read_in() {
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    m=0;
    for(int i=1;i<=n;i++) m+=a[i];
}

int calc(int lo,int hi,int cnt,int tong, int gtmax) {
    if (cnt==0)
        if (tong>gtmax) return 0; else return tong;
    if (hi-lo+1<cnt) return 0;
    if (t[n-cnt+1][n]+tong>gtmax) return 0;
    int kq=0;
    for(int i=lo;i<=hi;i++) {
        if (kq>0 && t[i][i+cnt-1]+tong<=kq) return kq;
        kq=max(kq,calc(i+1,hi,cnt-1,tong+a[i],gtmax));
        if (kq==gtmax) return kq;
    }
    return kq;
}

void solve() {
    sort(a+1,a+n+1); reverse(a+1,a+n+1);
    for(int i=1;i<=n;i++) {
        t[i][i]=a[i];
        for(int j=i+1;j<=n;j++) t[i][j]=t[i][j-1]+a[j];
    }
    ans=0;
    if (n%2) ans=calc(1,n,(n+1)/2,0,m/2);
    ans=max(ans,calc(1,n,n/2,0,m/2));
    printf("%d %d\n",ans,m-ans);
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    int T; scanf("%d",&T);
    for(int t=1;t<=T;t++) {
        read_in();
        solve();
        if (t<T) printf("\n");
    }
}
