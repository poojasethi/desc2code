#include <bits/stdc++.h>
using namespace std;
long long sum[45455], val;
vector<int> vec;
long long calculate(){
    sum[0] = 1;
    for(int i = 0; i < vec.size(); i++){
        for(int j = val; j - vec[i]>= 0; j--){
            sum[j] |= (sum[j - vec[i]] << 1);
        }
    }
    int ans = val, tmp = vec.size() / 2, mn = 1000000;
    for(int i = 1; i <= val; i++){
        if(sum[i] & (1LL << (tmp))){
            int tmp1 = i;
            int tmp2 = val - i;
             if(tmp1 < tmp2)
                swap(tmp1, tmp2);
            if(mn > tmp1 - tmp2){
                mn = tmp1 - tmp2;
                ans = tmp1;
            }
        }
    }
    return ans;
}
int main(){
    int t, n;
    scanf("%d", &t);
    for(;t--;){
        scanf("%d", &n);
        vec.resize(n);
        val = 0;
        memset(sum, 0, sizeof(sum));
        for(int i = 0; i < n; i++){
            scanf("%d", &vec[i]);
            val += vec[i];
        }
        int ans = calculate();
        printf("%lld %d\n", val - ans, ans);
    }
}
