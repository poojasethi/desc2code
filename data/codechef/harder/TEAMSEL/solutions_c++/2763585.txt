#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
 
typedef long long ll;
 
int sum ( int arr[], int n)
{
	int sum = 0;
	int i;
	for ( i=0; i< n; i++)
		 sum += arr[i];
	
	return sum ;
 
}
 
int subsetSum( int arr[], int n, int sum1)
{
int MAXSUM = sum1/2;
ll sum[450*100+1];
memset( sum , 0, sizeof(ll)* ( 450*100+1));
 
 
int i, j;
for (i=0; i< n; i++)
{
	for (j= MAXSUM; j>=0; j--)
	{
		if( sum[j] && arr[i] + j <= MAXSUM)
		{
			sum[j + arr[i] ] |= (sum[j]<<1);	
		}
	
		
	}
	sum[arr[i]]|=1;
 
	
}
 
for ( j= MAXSUM; j>=0 ; j--)
{
	
	if( n%2 == 0 && (sum[j] & 1ll<<(n/2-1)))
	{
		break;
	}
	else if ( n%2 != 0 &&( (sum[j] & 1ll<<(n/2-1)) || (sum[j] & 1ll<<(n/2))))
	{
	
			break;
	}
}
 
return j;
}
