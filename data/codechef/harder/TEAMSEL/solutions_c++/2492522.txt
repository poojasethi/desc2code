#define MAX_SUM 22500

long long int dp[MAX_SUM+5];
int n,arr[105],sum;

void cal_subset()
{
    int i,j;
    
    for(i=sum/2;i>=0;i--)
        dp[i]=0;
    
    for(i=0;i<n;i++)
    {
        for(j=sum/2;j>0;j--)
        {
            if(dp[j] && j+arr[i]<=sum/2)
            {
                dp[j+arr[i]]|=(dp[j]<<1);
            }
        }
        dp[arr[i]]|=1;
    }
}

int main()
{
    //READ();
    int t,i,d1,d2;
    
    S(t);
    while(t--)
    {
        sum=0;
        S(n);
        for(i=0;i<n;i++)
        {
            S(arr[i]);
            sum+=arr[i];
        }
        cal_subset();
        //printf("dello %lld ",dp[190]);
        for(i=sum/2;i>=1;i--)
        {
            if(n%2 && ((dp[i] & (1ll<<n/2-1)) || (dp[i] & (1ll<<n/2))))
            {
                d1=i;
                d2=sum-i;
                break;
            }
            if(n%2==0 && (dp[i] & (1ll<<n/2-1)))
            {
                d1=i;
                d2=sum-i;
                break;
            }
            
        }
        
        printf("%d %d\n",d1,d2);
        if(t)
            printf("\n");
    }
    return 0;
}
