#include<cstdio>
#include<cstring>
#include<algorithm>
#define ll long long
using namespace std;
ll A[105],dp[45005];
int main()
{
  ll T,N,su,i,max1,j,val1,sum;
  scanf("%lld",&T);
  while(T--)
    {
      scanf("%lld",&N);
      sum=0;
      for(i=0;i<N;i++)
	{
	  scanf("%lld",&A[i]);
	  sum+=A[i];
	}
      sort(A,A+N);
      memset(dp,0,sizeof(dp));
      max1=(1LL<<((N+1)/2+1))-1;
      /* dp[k] & (1<<i)==1 means k can be formed with i players */
      dp[0]=1;
      for(i=0;i<N && A[i]<=sum/2;i++)
	{
	  for(j=sum/2-A[i];j>=0;j--)
	    {
	      dp[j+A[i]]|=(dp[j]<<1);
	      dp[j+A[i]]&=max1;
	    }
	}
      val1=0;
      for(i=sum/2;i>=0;i--)
	{
	  if((dp[i]&(1LL<<((N+1)/2))) || (dp[i]&(1LL<<(N/2))))
	    {
	      val1=i;
	      break;
	    }
	}
      printf("%lld %lld\n",val1,sum-val1);
      if(T!=0)printf("\n");
    }
  return 0;
}
