#include <bits/stdc++.h>
using namespace std;
#define LL long long
int A[110];
LL N;
LL dp[450*100 + 10];
int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		cin>>N;
		int Sum = 0;
		for(int i = 0;i<N;i++){
			cin>>A[i];
			Sum+=A[i];
		}
		memset(dp,0,sizeof(dp));
		dp[0] = 1;
		int S = 0;
		for(int i = 0;i<N;i++)
		{
			S+=A[i];
			for(int j = S;j>=A[i];j--)
			{
				dp[j]|=(dp[j-A[i]]<<1);
			}
		}
		int Req = N/2;
		int Ans = 1000;
		for(int i = 1;i<=Sum;i++)
		{	
			if(dp[i]&(1LL<<Req))
				Ans = min(Ans,abs(i - (Sum - i)));
			if(dp[i]&(1LL<<(N - Req)))
				Ans = min(Ans,abs(i - (Sum - i)));
		}
		int ans1 = (Ans + Sum)/2;
		int ans2 = (ans1 - Ans);
		cout<<ans2<<" "<<ans1<<endl;
		if(T>0)
			cout<<endl;
	}
	return 0;

}
