#include <stdio.h>
#include <string.h>
int i,j,l,e,p,k,ii,x,y,z,n,m,g[255][32][22][8],r[255];
bool f[255][255];
char s[255],t[32][22];
int main() {
  scanf("%s",s); n=strlen(s); scanf("%d",&m);
  for (i=0; i<m; i++) scanf("%s",t[i]);
  for (i=n-1; i>=0; i--) {
    for (j=0; j<m; j++) {
      e=strlen(t[j]);
      if (t[j][e-1]==s[i]) {
        x=i>>5; y=i&31;
        g[i][j][e-1][x]|=(1<<y);
      }
      for (k=e-2; k>=0; k--) if (t[j][k]==s[i]) {
        if (t[j][k+1]==s[i+1]) for (ii=0; ii<8; ii++) g[i][j][k][ii]|=g[i+1][j][k+1][ii];
        for (p=0,z=i+2; z<n; p++,z++) if (f[i+1][p] && t[j][k+1]==s[z]) for (ii=0; ii<8; ii++) g[i][j][k][ii]|=g[z][j][k+1][ii];
      }
      for (p=i; p<n; p++) {
        x=p>>5; y=p&31;
        if (g[i][j][0][x]&(1<<y)) f[i][p-i]=true;
      }
    }
    for (l=0; i+l<n; l++) for (ii=0; ii<l; ii++) if (f[i][ii] && f[i+ii+1][l-ii-1]) f[i][l]=true;
  }
  for (i=0; i<=n; i++) r[i]=i;
  for (i=0; i<n; i++) {
    if (i && r[i-1]+1<r[i]) r[i]=r[i-1]+1;
    for (k=0; i+k<n; k++) if (f[i][k] && r[i]<r[i+k+1]) r[i+k+1]=r[i];
  }
  if (r[n-1]+1<r[n]) r[n]=r[n-1]+1;
  printf("%d\n",r[n]);
  return 0;
}
