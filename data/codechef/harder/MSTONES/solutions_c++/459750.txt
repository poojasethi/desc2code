#include<cstdlib>
#include<cstdio>
#define INF (1<<31)
int P[10000][2];
int main()
{
	int X;
	scanf("%d",&X);
	srand (69);
	while(X--)
	{
		int n;
		scanf("%d",&n);
		
		for(int i=0;i<n;i++)scanf("%d%d",&P[i][0],&P[i][1]);
		if(n==1)
		{
			puts("1");
			continue;
		}
		int t1,t2;
		int T=200;
		int res=-1,temp=-1;
		int x,y,x1,y1,x2,y2;
		while(T--)
		{
			temp=2;
			t1= rand() % n;
			t2= rand() % n;
			if(t1==t2){T++;continue;}
			for(int i=0;i<n;i++)
			{
				if(i==t1 || i==t2)continue;
				x1=P[t1][0];
				x2=P[t2][0];

				y1=P[t1][1];
				y2=P[t2][1];
				
				x=P[i][0];
				y=P[i][1];
				if((y-y1)*(x2-x1)==(y2-y1)*(x-x1))temp++;
			}
			if(temp>res)res=temp;
		}
		printf("%d\n",res);

	}


}
