#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn=10000+10;
int x[maxn],y[maxn];
int n,t;

int main()
{
	for (scanf("%d",&t);t--;)
	{
		scanf("%d",&n);
		for (int i=0;i<n;i++) scanf("%d%d",&x[i],&y[i]);
		int res=1;
		if (n>1)
		for (int tt=0;tt<300;tt++)
		{
			int u,v;
			do {u=rand()%n;v=rand()%n;} while (u==v);
			int cc=0;
			int xx=x[v]-x[u],yy=y[v]-y[u];
			for (int k=0;k<n;k++)
			if ((x[k]-x[u])*yy-(y[k]-y[u])*xx==0)
				cc++;
			res=max(res,cc);
		}
		printf("%d\n",res);
	}
}
