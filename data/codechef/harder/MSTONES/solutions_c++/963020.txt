#include<stdio.h>
#include<stdlib.h>
#include<string.h>
using namespace std;
#define s(n) scanf("%d",&n)
#define fill(A,c) memset(A,c,sizeof(A))

int solve(){
	int x[10001],y[10001];
	int N,i,k,x1,y1,x2,y2,a,b;
	s(N);
       
	int max=1,ans;
	for(i=0;i<N;++i)
		{
		s(x[i]);
		s(y[i]);
		}
        if(N==1)
        return 1;
        if(N==2)
        return 2;
for(i=0;i<200;++i)
		{
                 a=rand()%N;
                 b=rand()%N;

                 if(a==b)
                 continue;
		x2=x[b]-x[a];
                y2=y[b]-y[a];
                ans=0;
                                  for(k=0;k<N;++k)
                                         {
                                          x1=x[a]-x[k];
                                          y1=y[a]-y[k];
                                          if(x1*y2-x2*y1==0)
                                          ans++;
                                          }
                if(ans>max)
                 max=ans;
		}
      
return (max);
}

int main(){

int T,i;
s(T);
for(i=0;i<T;++i)
	{
	int m=solve();
	printf("%d\n",m);
	}
return 0;
}
