#include<bits/stdc++.h>
using namespace std;
const int booster=262144;
int tree[2*booster];
void przygotuj(int n)
{
   for(int i=1+booster;i<=n+booster;i++)
      tree[i]=1;
   for(int i=booster-1;i>=1;i--)
      tree[i]=tree[i*2]+tree[2*i+1];
}

int szukaj(int ile)
{
   int x=1;
   while(2*x<2*booster)
   {
      if(tree[2*x+1]<ile)
      {
	 x*=2;
	 ile-=tree[x+1];
      }
      else
	 x=2*x+1;
   }
   return x-booster;
}

void usun(int x)
{
   x+=booster;
   while(x)
   {
      tree[x]--;
      x/=2;
   }
}
int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
      int n;
      scanf("%d",&n);
      
      przygotuj(n);
      vector<int>wejscie,odp;
      wejscie.resize(n);
      odp.resize(n);
      for(int i=0;i<n;i++)
	 scanf("%d",&wejscie[i]);
      
      for(int i=n-1;i>=0;i--)
      {
	 odp[i]=szukaj(wejscie[i]+1);
	 usun(odp[i]);
	 
      }
      
      for(int i=0;i<n;i++)
	 printf("%d ",odp[i]);
      puts("");
   }
   
   return 0;
}
