# include <bits/stdc++.h>
# define mp make_pair
# define pb push_back
using namespace std;
typedef long long ll;
int a[200005],b[200005],tr[200005<<2|1];

void build (int root, int a, int b) {
	if (a==b) {
		tr[root] = 1;
		return;
	}
	int m = (a+b)/2;
	build (root<<1,a,m);
	build (root<<1|1,m+1,b);
	tr[root]=tr[root<<1]+tr[root<<1|1];
}

int query(int root, int a,int b, int val) {
	if (a==b) {
		return a;
	}
	int m = (a+b)/2;
	if (tr[root<<1] >= val)
		return query(root<<1,a,m,val);
	return query(root<<1|1,m+1,b,val-tr[root<<1]);	
}                

void update(int root, int a,int b, int val) {
	if (a==b) {
		tr[root] = 0;
		return;
	}
	int m = (a+b)/2;
	if (val <= m)
		update(root<<1,a,m,val);
	else 
		update(root<<1|1,m+1,b,val);
	tr[root]=tr[root<<1]+tr[root<<1|1];
}

main () {
	int t,n,i,j,k;
	scanf("%d",&t);
	while (t--) {
		scanf("%d",&n);
		build(1,0,n-1);
		for (i=0 ; i < n ; i++) scanf("%d",a+i);
		for (i=n-1; i >= 0; i--) {
			b[i] = query(1,0,n-1,i+1-a[i]);
			update(1,0,n-1,b[i]);
		}
		for (i=0 ; i < n ; i++) {
			printf("%d ",b[i]+1);
		}
		printf("\n");
	}
	return 0;
}
