#include<bits/stdc++.h>
using namespace std;
#define sc(x) scanf("%d",&x)
int arr1[10000000],arr2[10000000];
int main()
{
	int t;
	sc(t);
	
	while(t--)
	{
		int n;
		sc(n);
		for(int i=0;i<n;i++)
		{
		scanf("%d",&arr1[i]);
		arr2[i]=i+1;
	    }
		
		for(int i=n-1;i>=0;i--)
		{
			int temp=arr2[i-arr1[i]];
		     
		     int count=i-arr1[i];
		     for(;count<i;)
		      	{
		      	arr2[count]=arr2[count+1];
		      	count++;
		      }
		      arr2[i]=temp;
		}
		
		for(int i=0;i<n;i++)
		printf("%d ",arr2[i]);
		printf("\n");
		
	}
	
}
