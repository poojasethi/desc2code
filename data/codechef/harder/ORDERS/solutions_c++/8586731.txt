#define M 10010
#define N 212345

int bit[N];
int arr[N];
int ans[N];
bool done[N];

void update(int idx,int x)
{
	while(idx<N)
	{
		bit[idx]+=x;
		idx+=idx&(-idx);
	}
}

int query(int idx)
{
	int sum=0;
	while(idx)
	{
		sum+=bit[idx];

		//printf("%d %d %d\n",idx,sum,bit[idx]);

		idx-=idx&(-idx);
	}
	return sum;
}

void solve()
{
	int n,temp,start,end,mid;
	scanf("%d",&n);
	Fill(bit,0);
	Fill(arr,0);
	Fill(ans,0);
	Fill(done,0);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&arr[i]);
		//ans[i]=i;
	}
	/*for(int i=0;i<=n;i++)
		query(i);*/
	//printf("\n");
	for(int i=n;i>0;i--)
	{
		start=1;
		end=n;
		while(start<=end)
		{
			mid = (start+end)/2;
			temp=i-arr[i]+query(mid-1);
			//printf("checking %d\n",mid);
			if(temp==mid)
			{	
				if(done[mid])
				{
					//printf("done %d\n",mid);
					start=mid+1;
				}
				else
				{
					//printf("ans %d\n",mid);
					break;
				}
			}
			else if(temp>mid)
			{
				//printf("start %d\n",mid);
				start=mid+1;
			}
			else
			{
				//printf("end %d\n",mid);
				end=mid-1;
			}
		}
		ans[i]=mid;
		done[mid]=true;
		//ans[i]=i-arr[i];
		//printf("%d %d\n",query(i),query(n));
		update(ans[i],1);
	}

	for(int i=1;i<n;i++)
	{
		printf("%d ",ans[i]);
	}
	printf("%d\n",ans[n]);
}

int main()
{
	//freopen("in2.txt","r",stdin);
	//freopen("out2.txt","w",stdout);
	int t=1;
	scanf("%d",&t);
	for(int i=1;i<=t;i++)
	{
		//printf("Case #%d:\n",i);
		solve();
	}
}
