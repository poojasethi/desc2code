const int MOD = 1000000000 + 7 ;
const int MAXN = 100000 + 9 ;

int n;
vector<int> tree(8e5+10);

void build(int idx,int st,int ed){
	if(st==ed) tree[idx]=1;
	else{
		int mid=(st+ed)/2;
		build(cllft);
		build(clrgt);
		tree[idx]=tree[lft]+tree[rgt];
	}
}

void update(int idx,int st,int ed,int pos){
	if(st == ed) tree[idx]=0;
	else{
		int mid=(st+ed)/2;
		if(pos <= mid) update(cllft,pos);
		else update(clrgt,pos);
		tree[idx]=tree[lft]+tree[rgt];
	}
}

int query(int idx,int st,int ed,int pos){
	if(st == ed) return st;
	else{
		int mid=(st+ed)/2;
		if(tree[rgt] >= pos) return query(clrgt,pos);
		else return query(cllft,pos-tree[rgt]);
	}
}

void solve(vector<int> &vec){
	build(1,1,n);
	vector<int> ans(n);
	for(int i=n-1;i>=0;i--){
		ans[i]=query(1,1,n,vec[i]+1);
		update(1,1,n,ans[i]);
	}
	for(int i=0;i<n;i++){
		cout << ans[i] << " ";
	}
	cout << "\n";
}

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t=1;
	cin >> t;
	while(t--){
		cin >> n;
		vector<int> vec(n);
		for(int i=0;i<n;i++){
			cin >> vec[i];
		}
		solve(vec);
	}
	return 0;
}
