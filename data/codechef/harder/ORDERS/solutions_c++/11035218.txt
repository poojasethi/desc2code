#include <bits/stdc++.h>
#define maxn 200001

using namespace std;

int n, a[maxn], d[maxn];
int bit[maxn], res[maxn];

void read_in() {
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",a+i);
}

void update(int u,int val) {
    while (u<=n) {
        bit[u]+=val;
        u+=u&(-u);
    }
}

int get(int u) {
    int kq=0;
    while (u>0) {
        kq+=bit[u];
        u&=(u-1);
    }
    return kq;
}

int tim(int r) {
    int lo=0, hi=n;
    while (hi-lo>1) {
        int mid=(lo+hi)/2;
        if (get(mid)>=r) hi=mid; else lo=mid;
    }
    return hi;
}

void solve() {
    for(int i=1;i<=n;i++) bit[i]=0;
    for(int i=1;i<=n;i++) update(i,1);
    for(int i=n;i>=1;i--) {
        int u=tim(i-a[i]);
        res[i]=u;
        update(u,-1);
    }
}

void write_out() {
    for(int i=1;i<=n;i++) printf("%d ",res[i]);
    printf("\n");
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    int T; scanf("%d",&T);
    while (T--) {
        read_in();
        solve();
        write_out();
    }
}
