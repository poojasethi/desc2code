const int N=210000;
int sz[N*4];int n;int t;
inline void read(int &x){
	x=0;char p=getchar();
	while(!(p<='9'&&p>='0'))p=getchar();
	while(p<='9'&&p>='0')x*=10,x+=p-48,p=getchar();
}
int a[N];
void build(int me,int l,int r){
	sz[me]=r-l+1;
	if(l==r)return;
	int mid=(l+r)>>1;
	build(me*2,l,mid);
	build(me*2+1,mid+1,r);
}
inline int kth(int k){
	int me=1;int l=1;int r=n;
	while(l^r){
		int mid=(l+r)>>1;
		if(sz[me*2]<k){
			k-=sz[me*2];me=me*2+1;l=mid+1;
		}
		else{
			me*=2;r=mid;
		}
	}
	return l;
}
void del(int me,int l,int r,int x){
	if(l==r){sz[me]=0;return;}
	int mid=(l+r)>>1;
	if(x<=mid)del(me*2,l,mid,x);
	else del(me*2+1,mid+1,r,x);
	sz[me]=sz[me*2]+sz[me*2+1];
}
int main(){
	read(t);
	while(t--){
		read(n);
		rep(i,1,n)read(a[i]);
		build(1,1,n);
		per(i,n,1){
			a[i]=kth(sz[1]-a[i]);
			del(1,1,n,a[i]);
		}
		rep(i,1,n)printf("%d ",a[i]);printf("\n");
	}
	return 0;
}
