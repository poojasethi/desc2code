#include<bits/stdc++.h>
using namespace std;
void fast()
{
  ios_base::sync_with_stdio(0);
  cin.tie(0);
  cout.tie(0);
}
int main()
{
  fast();
  int t;
  cin>>t;
  while(t--)
  {
    int n,a[200002],i,ans[200002];
    cin>>n;
    for(i=1;i<=n;i++)
      cin>>a[i];
    for(i=1;i<=n;i++)
    {
      for(int j=1;j<=a[i];j++)
      {
        ans[i-j+1]=ans[i-j];
      }
      ans[i-a[i]]=i;
    }
    for(i=1;i<=n;i++)
      a[ans[i]]=i;
    for(i=1;i<=n;i++)
      cout<<a[i]<<" ";
    cout<<endl;
  }
  return 0;
}
