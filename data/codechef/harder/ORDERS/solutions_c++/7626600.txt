#include<iostream>
#include<cstring>


using namespace std;
int out[200002];

int main() {
	ios::sync_with_stdio(false);
	int t,n,in[200002],no[200002];
	for(int i=0;i<200002;i++) no[i]=i;
	int p,p1;
	cin>>t;
	while(t--) {
		cin>>n;
		for(int i=1;i<=n;i++)	cin>>in[i];		
		memcpy(out,no,4*n+4);		
		for(p=n;p;p--) {
			if(!in[p]) continue;
			p1=out[p-in[p]];
			memmove(out+(p-in[p]),out+(p-in[p]+1),in[p]*4);			
			out[p]=p1;		
		}
		for(int i=1;i<=n;i++)	cout<<out[i]<<" ";
		cout<<"\n";			
	
	}
	
	return 0;
}
