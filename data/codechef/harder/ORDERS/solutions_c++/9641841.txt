#include <bits/stdc++.h>
#define ll long long int
#define MAX 200000
using namespace std;

int stree[4*MAX];

void create(int index,int ss,int se)
{
	if(ss==se)
	{
		stree[index]=1;
		return;
	}
	if(ss<se)
	{	
		int mid=ss+(se-ss)/2;									//remember
		create(2*index,ss,mid);
		create(2*index+1,mid+1,se);
		stree[index]=stree[2*index]+stree[2*index+1];
	}
}

int find(int index,int ss,int se,int val)
{
	//cout<<"in"<<endl;
	int ans;
	if(ss==se)
	{
		if(val==stree[index])
		{	
			//cout<<"p"<<endl;
			stree[index]=0;
			return ss;
		}
	}
	int mid=(ss+se)/2;
	if(ss<se)
	{
		if(stree[2*index]>=val)
		{
			//cout<<"l"<<endl;
			ans=find(2*index,ss,mid,val);
		}
		else
		{
			//cout<<"r"<<endl;
			ans=find(2*index+1,mid+1,se,val-stree[2*index]);
		}
		stree[index]=stree[2*index]+stree[2*index+1];
	}
	return ans;
}

int main()
{
	std::ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		cin>>n;
		int arr[n+1];
		for(i=1;i<=n;i++)
			cin>>arr[i];
		create(1,1,n);
		for(i=n;i>0;i--)
		{
			arr[i]=find(1,1,n,i-arr[i]);
		}
		for(i=1;i<=n;i++)
			cout<<arr[i]<<" ";
		cout<<endl;
	}
	return 0;
}
