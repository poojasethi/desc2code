#include<stdio.h>
#include<iostream>
#include<vector>

using namespace std;

int main()
{
    int t,n;
    cin>>t;

    while(t--)
    {
        cin>>n;
        int a[n];
        for(int j=0;j<n;j++)
        {
            cin>>a[j];
        }



        vector<int> b;

        for(int k=1;k<=n;k++)
        {
            b.push_back(k);
        }
        int c[n],l;

        for(int i=n-1;i>=0;i--)
        {
            l=b.size();
            c[i]=b[l-1-a[i]];
            b.erase(b.begin()+(l-1-a[i]));
        }


        for(int i=0;i<n;i++)
        {
            cout<<c[i]<<" ";
        }

        cout<<"\n";

    }

}
