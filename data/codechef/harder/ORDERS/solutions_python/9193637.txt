num_cases = int(raw_input())
for case in range(num_cases):
    num_soldiers = int(raw_input())
    sequence = raw_input().split()
    sequence = [int(i) for i in sequence]
    positions = range(1,num_soldiers+1,1)
    for idx in range(num_soldiers-1,-1,-1):
        soldier = positions.pop(idx-sequence[idx])
        positions.insert(idx,soldier)
    #print final positions
    positions = [str(i) for i in positions]
    print " ".join(positions)
