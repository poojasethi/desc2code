int addmod(int v1, int v2) {
    int v3 = v1+v2;
    if(v3 >= MOD) v3 -= MOD;
    return v3;
}

int cc[100];
int p[10005], r[10001], v[10001], e[10001];

int get(int pp)
{
    if(p[pp] == pp) return pp;
    return p[pp] = get(p[pp]);
}

int main()
{
//    freopen("input.in", "r", stdin);
//    freopen("output.in", "w", stdout);
    int t;
    S(t);

    while(t--) {
        int n, m;
        S(n), S(m);
        fori(i,0,n) p[i] = i, r[i] = -1, v[i] = 1, e[i] = 0;

        V(ppi) edges;
        int x, y, val;
        fori(i,0,m) {
            S(x), S(y), S(val);
            x--, y--;
            edges.pb(ppi(val,pii(x,y)));
        }

        sort(allr(edges));
        int ans = 0;
        int cc = 0;
        fori(i,0,m) {
            x = edges[i].se.fr;
            y = edges[i].se.se;
            int cost = edges[i].fr;

            int p1, p2;
            p1 = get(x);
            p2 = get(y);

            if(p1 == p2) {
                if(v[p1] > e[p1]) {
                    ans += cost;
                    e[p1]++;
                    cc++;
                }
            }
            else {
                if(v[p1] == e[p1] && v[p2] == e[p2]) continue;
                if(r[p1] == r[p2]) r[p1] = r[p2]+1;
                if(r[p1] > r[p2]) {
                    p[p2] = p1;
                    e[p1] += e[p2] + 1;
                    v[p1] += v[p2];
                }
                else if(r[p1] < r[p2]) {
                    p[p1] = p2;
                    e[p2] += e[p1] + 1;
                    v[p2] += v[p1];
                }
                ans += cost;
                cc++;
            }
        }

        if(cc == n) {
            cout << ans << endl;
        }
        else {
            cout << "impossible" << endl;
        }
    }
    return 0;
}
