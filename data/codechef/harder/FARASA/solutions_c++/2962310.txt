#include<iostream>
#include<cmath>
#include<cassert>
#include<cstdio>

#include<vector>
#include<ctime>
using namespace std;
#define MAXN 2500
#define ll long long
#define PI 3.141592653589793238462
#define getcx getchar_unlocked

int inp()
{
    int n=0, ch=getcx();
    while(ch>'9'||ch<'0') ch=getcx();
    while(ch<='9'&&ch>='0') {n=(n<<3)+(n<<1)+(ch-'0');ch=getcx();}
    return n;
}

vector <ll> sum;
vector <ll> a;
map < ll , bool > m; 

bool hash[20000009];

int main()
{
    int n,i,j,x,typ;    
    n=inp();
    if(n<=MAXN)
    {
           a.resize(n+2);
           for(i=0;i<n;i++) a[i]=inp();
           for(i=0;i<n;i++)
           {
               ll sum=0;
               for(j=i;j<n;j++) {sum+=a[j];m[sum]=true;}
           }   
           printf("%d\n",m.size()-1);
    }
    else
    {        
           ll val;                    
           sum.resize(n+2);
           sum[0]=0;        
           for(i=0;i<n;i++)
           {
               val = inp();
               sum[i+1] = val + sum[i];
               hash[sum[i+1]]=true;   
           }
           ll sm=sum[n],cnt=0;
           for(i=1;i<=sm;i++)if(!hash[i])
           {
                for(j=1;j<=n;j++)if(hash[sum[j]+i]){hash[i]=true; break;}
           }
           for(i=1;i<=sm;i++)if(hash[i])cnt++;
           printf("%lld\n",cnt-1);
    }
    return 0;
}
