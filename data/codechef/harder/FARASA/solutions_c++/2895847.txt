#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <set>

using namespace std;
typedef long long LL;

int n, done;
LL a[200005], sum[200005], answer;
LL value;

bool sumFlag[20000005] = {0};

int main()
{
  scanf("%d", &n);
  a[0] = -1;
  for(int i = 1; i <= n; i++)
   scanf("%lld", &a[i]);

  // Cumulative sums
  sum[0] = 0;
  for(int i = 1; i <= n; i++)
   sum[i] = sum[i - 1] + a[i];

  if(n < 2000)
  {
    set<LL> distinctSums;
    for(int i = 1; i <= n; i++)
    {
      distinctSums.insert(sum[i]);
      distinctSums.insert(a[i]);
    }

    for(int i = 1; i <= n; i++)
     for(int j = 1; j <= i; j++)
      distinctSums.insert(sum[i] - sum[j - 1]);

    answer = distinctSums.size();
  }
  else
  {
    answer = 0;
    for(int i = 1; i <= n; i++)
     sumFlag[sum[i]] = 1;

    int maxSum = sum[n];
    for(int i = 1; i <= maxSum; i++)
    {
      if(!sumFlag[i])
      {
        for(int j = 1; j <= n; j++) 
         if(sumFlag[sum[j] + i])
         {
           sumFlag[i] = 1;
           break;
         }
      }
    }

    for(int i = 1; i <= sum[n]; i++)
     answer += sumFlag[i];
  }

  printf("%lld\n", answer - 1); 

  return 0;
}
