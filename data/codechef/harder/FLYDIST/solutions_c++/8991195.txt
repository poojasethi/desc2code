#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#define eps 0.0000001
using namespace std;

int N,M,n,m,m_,i,j,k,x,y,z,u,v,c;
int U[105],V[105],W[105],inc[105],dec[105],id[105][105],h[105][105];
double A[10005][205],B[10005],C[205],a[205],b,ans,tmp;
bool vis[105];

void Pivot(int u,int v)
{
	int i,j;
	B[u]/=A[u][v];
	for(i=1;i<=n;++i)if(i!=v)A[u][i]/=A[u][v];
	A[u][v]=1/A[u][v];
	for(i=1;i<=m;++i)
	if(i!=u&&fabs(A[i][v])>0)
	{
		B[i]-=A[i][v]*B[u];
		for(j=1;j<=n;++j)if(j!=v)A[i][j]-=A[i][v]*A[u][j];
		A[i][v]=-A[i][v]*A[u][v];
	}
	ans+=C[v]*B[u];
	for(i=1;i<=n;++i)if(i!=v)C[i]-=C[v]*A[u][i];
	C[v]=-C[v]*A[u][v];
}

void Simplex()
{
	int i,j,k;double tmp;
	for(;;)
	{
		for(i=1;i<=n;++i)if(C[i]>0)break;
		if(i>n)return;
		tmp=1e10;j=i;k=-1;
		for(i=1;i<=m;++i)
		if(A[i][j]>eps&&B[i]/A[i][j]<tmp)
		tmp=B[i]/A[i][j],k=i;
		if(k==-1){ans=1e10;return;}
		Pivot(k,j);
	}
}

int main()
{
	scanf("%d%d",&N,&M);
	for(i=1;i<=N;++i)
	for(j=1;j<=N;++j)
	h[i][j]=10000;
	for(i=1;i<=M;++i)
	{
		scanf("%d%d%d",&U[i],&V[i],&W[i]);
		++U[i];++V[i];
		h[U[i]][V[i]]=h[V[i]][U[i]]=1;
	}
	for(i=1;i<=N;++i)h[i][i]=0;
	for(k=1;k<=N;++k)
	for(i=1;i<=N;++i)
	for(j=1;j<=N;++j)
	if(h[i][k]+h[k][j]<h[i][j])
	h[i][j]=h[i][k]+h[k][j];
	
	for(i=1;i<=M;++i)inc[i]=++n;
	for(i=1;i<=M;++i)dec[i]=++n;
	for(i=1;i<=N;++i)
	for(j=i+1;j<=N;++j)
	id[i][j]=id[j][i]=++n;
	
	for(i=1;i<=N;++i)
	for(j=i+1;j<=N;++j)
	if(h[i][j]<N)
	for(k=1;k<=N;++k)
	if(h[i][k]<N&&i!=k&&j!=k)
	{
		++m;
		A[m][id[i][k]]=1;
		A[m][id[k][j]]=1;
		A[m][id[i][j]]=-1;
		B[m]=20*(h[i][k]+h[k][j]-h[i][j]);
	}
	for(i=1;i<=M;++i)
	{
		++m;
		A[m][id[U[i]][V[i]]]=1;
		A[m][inc[i]]=-1;
		A[m][dec[i]]=1;
		++m;
		A[m][id[U[i]][V[i]]]=-1;
		A[m][inc[i]]=1;
		A[m][dec[i]]=-1;
	}
	for(i=1;i<=M;++i)
	{
		C[inc[i]]=C[dec[i]]=1;ans-=18+W[i];
		++m;A[m][inc[i]]=1;B[m]=19;
		++m;A[m][dec[i]]=1;B[m]=W[i]-1;
	}
	
	Simplex();ans=-ans;
	for(i=1;;++i)
	{
		tmp=ans*i;
		if(fabs(tmp-(int)(tmp+0.5))<eps){printf("%d/%d\n",(int)(tmp+0.5),i);return 0;}	
	}	
}
