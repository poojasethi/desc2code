#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;



int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int N,M;
		vector<int> gr[1001],gr2[1001];
		vector< pair<int,int> > gr3;
		scanf("%d%d",&N,&M);
		for(int i=0;i<M;i++)
		{
			int n1,n2;
			scanf("%d%d",&n1,&n2);
			n1--;n2--;
			gr[n1].push_back(i);
			gr[n2].push_back(i);
		}
		for(int i=0;i<N;i++)
			for(int j=0;j<gr[i].size();j++)
				for(int k=j+1;k<(int)gr[i].size();k++)
				{
					gr2[gr[i][j]].push_back(gr[i][k]);
					gr2[gr[i][k]].push_back(gr[i][j]);
					gr3.push_back(make_pair(gr[i][j],gr[i][k]));
				}
		long long int edg=0,ver=0;
		//first calculate number of vertices
		for(int i=0;i<M;i++)
		{
			ver+= gr2[i].size()*(gr2[i].size()-1)/2;
		}
		//now calculate no of edges
		for(int i=0;i<gr3.size();i++)
		{
			int k= gr2[gr3[i].first].size()+ gr2[gr3[i].second].size()-2;
			edg+= k*(k-1)/2;
		}
		cout<<ver<<" "<<edg<<"\n";
	}
	return 0;
}
