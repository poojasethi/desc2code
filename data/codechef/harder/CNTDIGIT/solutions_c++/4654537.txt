int cnt[1024],r[1024],f[105][1024][10][10][2][2];
char s[105];
void add(int &a,int b){a+=b;while(a>=mod)a-=mod;}
int main()
{
	int T,k,i,j,ca=0,n,m;
	for(i=1;i<1024;i++)cnt[i]=cnt[i>>1]+(i&1);
	scanf("%s",s);
	n=strlen(s);
	for(i=0;i<n;i++)s[i]-='0';
	f[0][0][0][0][1][1]=1;// pos,mask,x000000y,zero,<=
	int a=0,b,v;
	for(i=0;i<n;i++){
	a=(a*10LL+s[i])%mod;
	for(int mask=0;mask<1024;mask++)
	for(int pl=0;pl<10;pl++)
	for(int last=0;last<10;last++)
	for(int z=0;z<2;z++)
	for(int eq=0;eq<2;eq++)
	if(v=f[i][mask][pl][last][z][eq])
	{
		for(int p=0;p<10;p++)
		{
			if(eq&&p>s[i])break;
			int neq=eq&&(p==s[i]);
			int nz=z&(p==0);
			int npl=last==0?pl:last;
			int nmask=mask|(nz?0:(1<<p));
			add(f[i+1][nmask][npl][p][nz][neq],v);
		}
	}
	}
	b=a+1;
	if(a%2)b/=2;else a/=2;
	a=(ll)a*b%mod;
	for(int mask=0;mask<1023;mask++)
	for(int pl=0;pl<10;pl++)
	for(int last=0;last<10;last++)
	{
		int cur=(f[n][mask][pl][last][0][0]+f[n][mask][pl][last][0][1])%mod;
		if(!cur)continue;
		int nmask=mask;
		for(j=last;;j--)
		{
			if(j==0&&pl==0)break;
			if(j<0)nmask|=(1<<pl-1),j=9;
			nmask|=1<<j;
			if(nmask==1023)break;
			add(r[cnt[nmask]],cur);
			add(a,mod-cur);
		}
	}
	int res=a*10LL%mod;
	for(i=1;i<10;i++)add(res,r[i]*(ll)i%mod);
	printf("%d\n",res);
    return 0;
}
