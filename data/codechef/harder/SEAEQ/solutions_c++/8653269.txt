#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
#include <vector>
using namespace std;
typedef unsigned long long u64;
const int Maxn=505,Maxe=130005,MaxT=10005;
const int mod=1000000007;
int T,n[MaxT],e[MaxT],N=0;
int ans[MaxT];
int C[Maxn][Maxn],f[Maxe],fac[Maxn];

void init() {
	C[0][0]=1;
	for (int i=1;i<=N;i++) {
		C[i][0]=1;
		for (int j=1;j<=i;j++) {
			C[i][j]=C[i-1][j-1]+C[i-1][j];
			if (C[i][j]>=mod) C[i][j]-=mod;
		}
	}
	for (int i=1;i<=N;i++)
		for (int j=0;j<=i;j++)
			C[i][j]=u64(C[i][j])*C[i][j]%mod;
	fac[0]=1;
	for (int i=1;i<=N;i++) fac[i]=u64(fac[i-1])*i%mod;
	for (int i=1;i<=N;i++) fac[i]=u64(fac[i])*fac[i]%mod;
}

int main() {
	scanf("%d",&T);
	for (int i=1;i<=T;i++) {
		scanf("%d%d",n+i,e+i);
		N=max(N,n[i]);
	}
	init();
	f[0]=1;
	for (int i=1;i<=N;i++) {
		for (int j=i*(i-1)/2;j>=i;j--) {
			f[j]-=f[j-i];
			if (f[j]<0) f[j]+=mod;
		}
		for (int j=1;j<=i*(i+1)/2;j++)
			if ((f[j]+=f[j-1])>=mod)
				f[j]-=mod;
		for (int j=1;j<=T;j++) {
			if (n[j]<i) continue;
			ans[j]=(ans[j]+u64(C[n[j]][i])*f[min(i*(i-1)/2,e[j])]%mod*fac[n[j]-i]%mod*(n[j]-i+1))%mod;
		}
	}
	for (int i=1;i<=T;i++) printf("%d\n",ans[i]);
	return 0;
}
