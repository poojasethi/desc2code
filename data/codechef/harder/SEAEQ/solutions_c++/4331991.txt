void inv()
{
    inversion[1][0] = 1;
    cum_inversion[1][0] = 1;
    for(int i = 2; i <= 500; i++)
    {
        long int x = ((i-1)*(i-2))/2;
        int k = i;
        long long int cum_sum = 0;
        for(long int y = 0; y <= x; y++)
        {
            cum_sum = cum_sum + inversion[i-1][y];
            if(cum_sum >= 1000000007)
                cum_sum -= 1000000007;
            //cum_sum = cum_sum % 1000000007;
            if((y-k) >= 0)
            {
                cum_sum = cum_sum - inversion[i-1][y-k];
                if(cum_sum < 0)
                {
                    cum_sum  = cum_sum + 1000000007;
                }
                else
                {
                   if(cum_sum >= 1000000007)
                        cum_sum -= 1000000007;
                }
                    //cum_sum = cum_sum % 1000000007;
            }
            inversion[i][y] = cum_sum;
            if(y != 0)
            {
                cum_inversion[i][y] = cum_inversion[i][y-1] + inversion[i][y];
                if(cum_inversion[i][y] >= 1000000007)
                    cum_inversion[i][y] -= 1000000007;
                //cum_inversion[i][y] = cum_inversion[i][y] % 1000000007;
            }
            else
            {
                cum_inversion[i][0] = 1;
            }
        }
        long int x1 = x + i - 1;
        for(long int y = x + 1; y <= x1; y++)
        {
            inversion[i][y] = inversion[i][x1-y];
            cum_inversion[i][y] = cum_inversion[i][y-1] + inversion[i][y];
            if(cum_inversion[i][y] >= 1000000007)
                cum_inversion[i][y] -= 1000000007;
            //cum_inversion[i][y] = cum_inversion[i][y] % 1000000007;
        }
    }
}
int main()
{
    //fact();
    inv();
    long int t;
    scanf("%ld",&t);
    while(t--)
    {
        int n;
        long int e;
        long long int ans = 0;
        scanf("%d%ld",&n,&e);
        long long int y = 1;
        long long int x;
        for(int i = n; i >= 1; i--)
        {
            if(i != n)
                y = y*(i+1)*(i+1);
            y = y % 1000000007;
            x = (n-i+1)*y;
            x = x % 1000000007;
            if (e <= (i*(i-1))/2)
                x = x * cum_inversion[i][e];
            else
                x = x * cum_inversion[i][(i*(i-1))/2];
            x = x % 1000000007;
            ans = ans + x;
            ans = ans % 1000000007;
            //cout<<ans<<"\n";
        }
        printf("%ld\n",ans);
    }
}
