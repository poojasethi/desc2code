#include<cstdio>
#include<cstring>
#include<algorithm>
#include<set>
#include<map>
#include<vector>
#define rep(i,j,k) for(int i=(int)j;i<=(int)k;i++)
#define per(i,j,k) for(int i=(int)j;i>=(int)k;i--)
using namespace std;
typedef long long LL;
typedef double db;
const int P=1000000007;
const int N=505;
int C[N][N];
int fac[N];
int f[N][N*N/2];
int n,E;
int main(){
	n=500;
	int up=n*(n-1)/2;
	rep(i,0,up)f[1][i]=1;
	rep(i,2,n)rep(j,0,up){
		if(j>(i-1))f[i][j]=(f[i-1][j]+P-f[i-1][j-(i-1)-1])%P;
		else f[i][j]=f[i-1][j];
		
		if(j>(i)*(i-1)/2)f[i][j]=0;
		
		if(j){
			f[i][j]+=f[i][j-1];
			if(f[i][j]>=P)f[i][j]-=P;
		}
	}
	rep(i,0,n){
		C[i][0]=C[i][i]=1;
		rep(j,1,i-1)C[i][j]=(C[i-1][j-1]+C[i-1][j])%P;
	}
	fac[0]=1;rep(i,1,n)fac[i]=fac[i-1]*1ll*i%P;
	
	int T;scanf("%d",&T);
	while(T--){
		scanf("%d%d",&n,&E);
		if(E>=n*(n-1)/2)E=n*(n-1)/2;
		int ans=0;
		rep(i,1,n){
			int val=(n-i+1)*1ll*fac[n-i]%P;
			val=val*1ll*fac[n-i]%P;
			
			val=val*1ll*C[n][i]%P;
			val=val*1ll*C[n][i]%P;
			val=val*1ll*f[i][min(E,i*(i-1)/2)]%P;
			ans+=val;if(ans>=P)ans-=P;
		}
		printf("%d\n",ans);
	}
	return 0;
}
