#include <stdio.h>
#include <stdlib.h>
#define p 1000000007
using namespace std;

int T,n,m,E,i,j,k,ans;
int f[505][125005],C[505][505],fac[505];

int main()
{
	f[0][0]=1;
	for(i=1;i<=500;++i)
	{
		m=(i-1)*(i-2)/2;
		for(j=0;j<=m;++j)
		{
			f[i][j]+=f[i-1][j];if(f[i][j]>=p)f[i][j]-=p;
			f[i][j+i]-=f[i-1][j];if(f[i][j+i]<0)f[i][j+i]+=p;
		}
		m=i*(i-1)/2;
		for(j=1;j<=m;++j)
		{
			f[i][j]+=f[i][j-1];
			if(f[i][j]>=p)f[i][j]-=p;
		}
	}
	for(i=1;i<=500;++i)
	{
		m=i*(i-1)/2;
		for(j=1;j<=m;++j)
		{
			f[i][j]+=f[i][j-1];
			if(f[i][j]>=p)f[i][j]-=p;
		}
	}
	C[0][0]=1;
	for(i=0;i<=500;++i)
	for(j=0;j<=i;++j)
	{
		if(C[i][j]>=p)C[i][j]-=p;
		C[i+1][j]+=C[i][j];
		C[i+1][j+1]+=C[i][j];
	}
	fac[0]=1;
	for(i=1;i<=500;++i)fac[i]=(long long)fac[i-1]*i%p;
	
	scanf("%d",&T);
	for(;T;--T)
	{
		ans=0;
		scanf("%d%d",&n,&E);
		for(i=1;i<=n;++i)
		{
			m=i*(i-1)/2;if(E<m)m=E;
			ans=(ans+(long long)f[i][m]*C[n][i]%p*C[n][i]%p*fac[n-i]%p*fac[n-i]%p*(n-i+1))%p;
		}
		printf("%d\n",ans);
	}
}
