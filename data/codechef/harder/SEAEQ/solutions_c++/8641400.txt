#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n	505
#define P	1000000007
#define For(i,a,b)	for(int i=a,lim=b;i<=lim;i++)

int		T,N,M,L[n],C[n][n],F[n][n*n/2];

void	Add(int&a,int b)	{
		if	((a+=b)>=P)	a-=P;
}

int		main(){
		For(i,0,500){
			C[i][0]=1;
			For(j,1,i)	C[i][j]=(C[i-1][j-1]+C[i-1][j])%P;
		}
		L[0]=1;
		For(i,1,500)	L[i]=1ll*L[i-1]*i%P;

		F[0][0]=1;
		For(i,1,500){
			For(j,0,(i-1)*(i-2)/2)	Add(F[i][j],F[i-1][j]),Add(F[i][j+i],P-F[i-1][j]);
			For(j,1,i*(i-1)/2)	Add(F[i][j],F[i][j-1]);
		}
		For(i,1,500)	For(j,1,i*(i-1)/2)	Add(F[i][j],F[i][j-1]);

		scanf("%d",&T);
		for (;T--;)	{
			scanf("%d%d",&N,&M);	int	Ans=0;
			For(i,1,N)	Add(Ans,1ll*F[i][min(M,i*(i-1)/2)]*C[N][i]%P*C[N][i]%P*(N-i+1)%P*L[N-i]%P*L[N-i]%P);
			printf("%d\n",Ans);
		}
}
