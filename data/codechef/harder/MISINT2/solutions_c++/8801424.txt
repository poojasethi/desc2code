#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
using namespace std;

int T,n,m,i,j,k,w;
int ord[100005],lp[100005],p[100005],phi[100005],prime[100005],tot;
long long ord2[100005][30],a[150005],b[150005],ll,l,r,u,v,c,ans;
long long Pi[150005][45],pi[45],pord[45];
int Pk[150005][45],pk[45],Tot[150005];
bool flag;

long long mult(long long a,long long b,long long p)
{
	long long ans=(long long)((long double)a*b/p);
	ans=a*b-ans*p;
	if(ans>=p)ans-=p;if(ans<0)ans+=p;
	return ans;
}

long long Power2(long long a,long long b,long long p)
{
	long long ans=1;
	for(;b;b>>=1,a=mult(a,a,p))if(b&1)ans=mult(ans,a,p);
	return ans;
}

int Power(int a,int b,int p)
{
	int ans=1;
	for(;b;b>>=1,a=(long long)a*a%p)if(b&1)ans=(long long)ans*a%p;
	return ans;
}

int Ord(int x)
{
	int i,k,q=phi[x];
	for(i=phi[x];i>1;i=lp[i])
	for(k=p[i];q%k==0&&Power(2,q/k,x)==1;q/=k);
	return q;
}

long long ORD(long long p)
{
	long long k,x=p-1,q=p-1;
	for(k=2;k*k<=x;++k)
	if(x%k==0)
	{
		for(;x%k==0;x/=k);
		for(;q%k==0&&Power2(2,q/k,p)==1;q/=k);
	}
	if(x>1)for(k=x;q%k==0&&Power2(2,q/k,p)==1;q/=k);
	return q;
}

long long lcm(long long a,long long b){return a/__gcd(a,b)*b;}

void dfs(int x,long long y,long long z)
{
	if(x>tot)
	{
		ans+=y/z;
		return;
	}
	dfs(x+1,y,z);
	long long P=1,p=pi[x],q=pord[x];
	y*=p-1;
	for(int i=1;i<=pk[x];++i)
	{
		P*=p;
		if(p<=n)
		{
			if(i<=ord2[p][0])q=ord2[p][i];
			else
			{
				for(;Power2(2,q,P)!=1;q*=p);
				++ord2[p][0];
				ord2[p][i]=q;
			}
		}
		dfs(x+1,y,lcm(z,q));
		y*=p;
	}
}

int main()
{
	n=100000;
	for(i=2;i<=n;++i)
	{
		if(!p[i])prime[++tot]=p[i]=i;
		for(j=1;j<=tot&&prime[j]<=p[i]&&prime[j]<=n/i;++j)p[i*prime[j]]=prime[j];
	}
	phi[1]=lp[1]=1;
	for(i=2;i<=n;++i)
	if(p[i]==p[i/p[i]])phi[i]=phi[i/p[i]]*p[i],lp[i]=lp[i/p[i]];
	else phi[i]=phi[i/p[i]]*(p[i]-1),lp[i]=i/p[i];
	for(i=3;i<=n;i+=2)if(lp[i]==1)ord[i]=Ord(i);else ord[i]=lcm(ord[lp[i]],ord[i/lp[i]]);
	
	scanf("%d",&T);
	for(;T;--T)
	{
		scanf("%lld%lld",&ll,&r);
		if(r%2==0)++r,flag=true;else flag=false;
		l=max(1ll,ll-100000);m=r-l;
		for(i=0;i<=m;++i)a[i]=l+i,Tot[i]=0;
		for(i=2;i<=n;++i)if(p[i]==i)
		{
			u=l/i*i;if(u<l)u+=i;
			for(;u<=r;u+=i)
			{
				for(v=0;a[u-l]%i==0;++v)a[u-l]/=i;
				++Tot[u-l];Pi[u-l][Tot[u-l]]=i;Pk[u-l][Tot[u-l]]=v;
			}
		}
		for(i=0;i<=m;++i)if(a[i]>1)++Tot[i],Pi[i][Tot[i]]=a[i],Pk[i][Tot[i]]=1;
		for(i=ll-l;i<=m;++i)if((l+i)&1)
		{
			tot=Tot[i];ans=0;
			for(j=1;j<=tot;++j)
			{
				pi[j]=Pi[i][j];
				pk[j]=Pk[i][j];
				if(pi[j]<=n)pord[j]=ord[pi[j]];
				else
				{
					k=(l+i)/pi[j]*(pi[j]-1)-l;
					u=pi[j]-1;
					for(w=1;w<=Tot[k];++w)
					{
						v=Pi[k][w];
						for(;u%v==0&&Power2(2,u/v,pi[j])==1;u/=v);
					}
					for(;u%2==0&&Power2(2,u/2,pi[j])==1;u/=2);
					pord[j]=u;
				}
			}
			dfs(1,1,1);
			b[i]=ans;
		}
		for(i=0;i<=m;++i)if(((l+i)&1)==0)b[i]=b[i+1]-1;
		if(flag)--m;
		ans=0;
		for(i=ll-l;i<=m;++i)ans+=Power(26,b[i],1000000007);
		printf("%lld\n",ans%1000000007);
	}
}
