#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#define p 1000000007
using namespace std;

int N,L,X,K,i,j,k,w,l,q,u,v,ans;
int L_[15],X_[15];
int f[35][11],g[35][55][10],t[35][55][10];

void add(int &a,int b){a+=b;if(a>=p)a-=p;}

void work()
{
	if(L<=4)
	{
		for(i=0;i<=K*(X>=1);++i)
		for(j=0;j<=K*(X>=2);++j)
		for(k=0;k<=K*(X>=3);++k)
		if(i+2*j+3*k>L&&i+j+k==5)
		add(ans,1);
		return;
	}
	for(i=L-5,L_[0]=0;i;i/=10)L_[++L_[0]]=i%10;
	for(i=X-1,X_[0]=0;i;i/=10)X_[++X_[0]]=i%10;
	memset(f,0,sizeof(f));
	for(i=L_[0],k=0;i>X_[0];--i)k=k*10+L_[i];
	if(k>=5)return;
	f[1+2+4+8+16][k+1]=1;
	for(i=X_[0];i;--i)
	{
		memset(g,0,sizeof(g));
		for(j=0;j<=31;++j)
		for(k=0;k<=5;++k)
		if(f[j][k])
		{
			if(k)l=(k-1)*10+L_[i]+1;else l=0;
			add(g[j][l][X_[i]],f[j][k]);
		}
		for(l=4;l>=0;--l)
		{
			memset(t,0,sizeof(t));
			for(j=0;j<=31;++j)
			for(k=0;k<=50;++k)
			for(w=0;w<=9;++w)
			if(g[j][k][w])
			{
				if(j&(1<<l))q=w;else q=9;
				for(u=j;q>=0;--q,u&=31-(1<<l))
				{
					v=k-q;if(v<0)v=0;
					add(t[u][v][q],g[j][k][w]);
				}
			}
			memcpy(g,t,sizeof(g));
		}
		memset(f,0,sizeof(f));
		for(j=0;j<=31;++j)
		for(k=0;k<=6;++k)
		for(w=0;w<=9;++w)
		if(g[j][k][w])
		add(f[j][k],g[j][k][w]);
	}
	for(i=0;i<=31;++i)
	{
		l=k=0;
		for(j=4;j>=0;--j)
		{
			if(i&(1<<j))++k;else k=1;
			if(k>l)l=k;
		}
		if(l<=K)add(ans,f[i][0]);
	}
}

int main()
{
	scanf("%d%d%d%d",&N,&L,&X,&K);
	for(;L<=N;++L)work();
	printf("%d\n",ans);
}
