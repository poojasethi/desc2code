#include <stdio.h>
#include <stdlib.h>
#define p 1000000007
using namespace std;

int T,n,m,i,j,k;
int fac[1005],fac_inv[1005],f[1005],g[1005];

int C(int n,int m){return (long long)fac[n]*fac_inv[n-m]%p*fac_inv[m]%p;}
int Power(int a,int b)
{
	int ans=1;
	for(;b;b>>=1,a=(long long)a*a%p)if(b&1)ans=(long long)ans*a%p;
	return ans;
}

int calc(int n,int m)
{
	int ans=0,tmp;
	f[0]=g[0]=1;
	for(i=1;i<=n;++i)f[i]=(long long)f[i-1]*m%(p-1);
	for(i=1;i<=n;++i)g[i]=(long long)g[i-1]*(m-1)%(p-1);
	for(i=0;i<=n;++i)
	{
		tmp=(long long)C(n,i)*Power(2,(long long)f[n-i]*g[i]%(p-1))%p;
		if(i&1)ans=(ans+p-tmp)%p;else ans=(ans+tmp)%p;
	}
	return ans;
}

int main()
{
	fac[0]=1;
	for(i=1;i<=1000;++i)fac[i]=(long long)fac[i-1]*i%p;
	for(i=0;i<=1000;++i)fac_inv[i]=Power(fac[i],p-2);
	scanf("%d",&T);
	for(;T;--T)
	{
		scanf("%d%d",&n,&m);
		printf("%d\n",(calc(n,m+1)-calc(n,m)+p)%p);
	}
}
