using namespace std;
const int max_n=1005;

typedef long long ll;

const ll M=1000000007;
int t,n;
int a[max_n];
ll C[max_n][max_n];

ll pwr(ll x, ll y, ll mod)
{
	if(y==0ll)
		return 1ll;
	ll ret=pwr(x,y/2,mod);
	ret*=ret;
	ret%=mod;
	if(y%2)
		ret*=x;
	ret%=mod;
	return ret;
}

ll func(int d)
{
	ll res=0ll;
	for(int k=0; k<=n; k++)
	{
		ll ch=(pwr(d,k,M-1)*pwr((ll)(d+1),(ll)(n-k),M-1))%(M-1);
		ll tmp=(C[n][k]*pwr(2ll,ch,M))%M;
		if(k%2)
			tmp=-tmp;
		res+=M+tmp;
		res%=M;
	}
	return res;
}

int main()
{
	int d;
	for(int i=1; i<max_n; i++)
	{
		C[i][0]=1;
		C[i][i]=1;
		for(int j=1; j<i; j++)
		{
			C[i][j]=C[i-1][j-1]+C[i-1][j];
			C[i][j]%=M;
		}
	}
    scanf("%d",&t);
	for(int z=0; z<t; z++)
	{
		scanf("%d%d",&n,&d);
		
		ll res=(M+func(d)-func(d-1))%M;
		cout<<res<<"\n";
    }
    return 0;
}
