#include <iostream>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;
#define N 1050
#define P 1000000007
#define ll long long
ll Inv[N],InJ[N],JC[N],ans;
int t,n,m;
inline int Read()
 {
 	int x=0;char y;
 	do y=getchar(); while (y<'0'||y>'9');
 	do x=x*10+y-'0',y=getchar(); while (y>='0'&&y<='9');
 	return x;
 }
ll Q_Power(ll x,int y,int p)
 {
 	ll z=true;
 	while (y) z=y&1?z*x%p:z,x=x*x%p,y >>= true;
 	return z;
 }
inline ll C(int x,int y)
 {return JC[x]*InJ[y]%P*InJ[x-y]%P;}
void Pretreat()
 {
 	JC[0]=Inv[0]=InJ[0]=JC[1]=Inv[1]=InJ[1]=true;
 	for (int i=2;i<N;i++)
 	  JC[i]=JC[i-1]*i%P,Inv[i]=Inv[P%i]*(P-P/i)%P,
 	  InJ[i]=InJ[i-1]*Inv[i]%P;
 	return;
 }
int GetAns(int x,int y)
 {
 	ans = false;
 	for (int i=0;i<=x;i++)
 	  ans=(ans+(i&1?-1:1)*C(x,i)*Q_Power(2,Q_Power(y+1,x-i,P-1)*
 	  	Q_Power(y,i,P-1)%(P-1),P))%P,ans=(ans+P)%P;
 	return ans;
 }
int main()
 {
 	//freopen("input.txt","r",stdin);
 	t=Read();Pretreat();
 	while (t--)
 	 {
 	 	n=Read();m=Read();
 	 	printf("%d\n",(GetAns(n,m)-GetAns(n,m-1)+P)%P);
 	 }
 	return 0;
 }
