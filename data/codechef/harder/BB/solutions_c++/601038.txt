ll mpow(ll a, ll b)// (a^b) % MOD
{
    a=a%MOD;
    ll w=1;
    while(b)
    {
        if (b%2) w=(w*a)%MOD;
        a=(a*a)%MOD;
        b/=2;
    }
    return w;
}

ll inv(ll x)
{
    return mpow(x,MOD-2);
}
