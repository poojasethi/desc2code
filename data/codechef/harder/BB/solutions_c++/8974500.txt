#include <stdio.h>
#include <stdlib.h>
#define p 1000000007
using namespace std;

int T,n,m,K,i,j,k;

int Power(int a,int b)
{
	int ans=1;
	for(;b;b>>=1,a=(long long)a*a%p)if(b&1)ans=(long long)ans*a%p;
	return ans;
}

int calc(int n,int m,int r)
{
	if(r<n)return 0;
	if(!n||!m)return 1;
	int mul=1,div=1;
	for(i=1;i<=r-n&&i<=m;++i)
	for(j=1;j<=n;++j)
	mul=(long long)mul*(m-i+1+r-j)%p,
	div=(long long)div*(i+j-1)%p;
	return (long long)mul*Power(div,p-2)%p;
}

int main()
{
	scanf("%d",&T);
	for(;T;--T)
	{
		scanf("%d%d%d",&n,&m,&K);
		if(n%m<=m-K)printf("%d\n",calc(K,n/m,m-n%m));
		else printf("%d\n",calc(K-(m-n%m),n/m+1,n%m));
	}
}
