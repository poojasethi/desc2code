using namespace std;
const ll N=100005;
ll n,m,i,j,x,y,cnt[N],cur,u,h[N],ele;
bool f[N];
vector<ll> g[N];
ll ans,sum;
ULL e,R[N];

ULL ran(){
   return (((ULL)rand()<<48)|((ULL)rand()<<32)|((ULL)rand()<<16)|(ULL)rand());
}

void go(ll u,ll p){
     ll i,to;
     f[u]=1;
     h[u]=h[p]+1;

     for(i=0;i<g[u].size();i++){
        to=g[u][i];
        if(to==p)continue;
        if(f[to]){
           if(h[to]<h[u]){
             cnt[u]++;
             cnt[to]--;
             e=ran();
             R[u]^=e;
             R[to]^=e;
           }
           continue;
        }
        go(to,u);
        cnt[u]+=cnt[to];
        R[u]^=R[to];
     }
}

int main()
{//freopen("ele.txt","r",stdin);
   // freopen("out.txt","w",stdout);
 scanf("%lld%lld",&n,&m);
 for(i=1;i<=m;i++){
    scanf("%lld%lld",&x,&y);
    g[x].pb(y);
    g[y].pb(x);
 }
 u=m-(n-1);
 if(u<0)assert(0);
 ans+=(ll)u*(u-1)/2;
 //cout<<m*(m-1)/2-ans<<endl;

 go(1,0);
 for(i=1;i<=n;i++)
 if(!f[i]){
    cout<<m*(m-1)/2<<endl;return 0;
 }

 for(i=1;i<=n;i++)
    if(cnt[i]==1)ans+=(u-1);else
    if(cnt[i]>1)ans+=u;


 //cout<<m*(m-1)/2-ans<<endl;
 //for(i=1;i<=n;i++)cout<<cnt[i]<<' ';cout<<endl;system("pause");
 //for(i=1;i<=n;i++)cout<<R[i]<<' ';cout<<endl;system("pause");
 sort(R+1,R+n+1);
// for(i=1;i<=n;i++)cout<<R[i]<<' ';cout<<endl;system("pause");

 cur=0;
 for(i=2;i<=n;i++)if(R[i]==0)ele++;
 for(i=2;i<=n;i++){
    if(R[i]==0)continue;
    if(R[i]!=R[i-1]){

       sum+=(ll)(n-1-ele-cur)*cur;
       cur=1;
    }else cur++;
 }

 sum+=(ll)(n-1-ele-cur)*cur;
 //cout<<"sum "<<sum<<endl;
 ans+=sum/2;
 ans=m*(m-1)/2-ans;
 cout<<ans<<endl;
}
