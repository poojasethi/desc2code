#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n	5000005
#define For(i,a,b)	for(int i=a;i<=b;i++)
#define Rep(i,a,b)	for(int i=a;i>=b;i--)

int		N,M,D,Q[n],ID[n],Tim[n],Fail[n],A[n][63];
char	s[n];

int		Trans(int x){
		if	(x>='a'&&x<='z')	return	x-'a';
		if	(x>='A'&&x<='Z')	return	x-'A'+26;
		if	(x>='0'&&x<='9')	return	x-'0'+52;
		return	62;
}

int		main(){
		scanf("%d\n",&N);
		For(i,1,N)	{
			gets(s+1);	int L=strlen(s+1),x=0;
			for	(int j=1,t;j<=L;x=A[x][t],j++)
				if	(!A[x][t=Trans(s[j])])	A[x][t]=++D;	ID[i]=x;
		}

		for	(int l=0,r=1;l<r;){
			int u=Q[++l],v;
			For(i,0,62)	if	(v=A[u][i]){
				Q[++r]=v;
				if	(u)	Fail[v]=A[Fail[u]][i];
			}	else	A[u][i]=A[Fail[u]][i];
		}

		scanf("%d\n",&M);
		for	(;M--;){
			gets(s+1);	int L=strlen(s+1),x=0;
			for	(int j=1;j<=L;Tim[x]++,j++)	x=A[x][Trans(s[j])];
		}

		Rep(i,D+1,1)Tim[Fail[Q[i]]]+=Tim[Q[i]];
		For(i,1,N)	printf("%d\n",Tim[ID[i]]);
}
