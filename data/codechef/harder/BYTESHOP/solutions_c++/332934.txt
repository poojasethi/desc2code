#include <stdio.h>
#include <time.h>
int n,m,r,i,j,k,t,l,l1,l2,r1,r2,res,d,c[444][444],pp[44444],pr[44444],md=1000000007;
long long cur,e;
long long pow(int a, int b) {
  if (b==0) return 1;
  if (b&1) return (a*pow(a,b-1))%md;
  long long p=pow(a,b/2);
  return (p*p)%md;
}
int main() {
//freopen("a.in","r",stdin);
  for (i=0; i<444; i++) {
    c[i][0]=1;
    for (j=1; j<=i; j++) {
      c[i][j]=c[i-1][j-1]+c[i-1][j];
      if (c[i][j]>=md) c[i][j]-=md;
    }
  }
//  for (i=2; i<=40000; i++) for (j=2; j*j<=i; j++) if (i%j==0) { a[i]=j; break; }
//  for (i=0; i<44444; i++) pp[i]=pow(i,md-2);
  pp[0]=1; pr[0]=1;
  for (long long ii=1; ii<44444; ii++) {
    pp[ii]=(pp[ii-1]*ii)%md;
    pr[ii]=pow(pp[ii],md-2);
  }
  scanf("%d",&t);
  while (t--) {
    scanf("%d%d%d",&n,&m,&r); res=0;
    if (r<n || r<m) { puts("0"); continue; }
    for (i=0; i<n; i++) for (j=0; j<m; j++) if ((n-i)*(m-j)>=r) {
      if (i+j) {
        if (j) {
          l1=(n-i)*(m-j+1)-r;
          r1=(n-i)*(m-j+1);
          l2=(n-i)*(m-j)-r;
          r2=(n-i)*(m-j);
//          if (r2+1>l1) l1=r2+1;
//          if (l-1<r2) r2=l-1;
          e=(e*pp[r2])%md;
          e=(e*pr[l2])%md;
          e=(e*pr[r1])%md;
          e=(e*pp[l1])%md;
//          for (k=l2; k<=r2; k++) e=(e*k)%md;
//          for (k=l1; k<=r1; k++) e=(e*pp[k])%md;
        } else {
          e=d;
          l1=(n-i+1)*m-r;
          r1=(n-i+1)*m;
          l2=(n-i)*m-r;
          r2=(n-i)*m;
//          if (r2+1>l1) l1=r2+1;
//          if (l-1<r2) r2=l-1;
          e=(e*pp[r2])%md;
          e=(e*pr[l2])%md;
          e=(e*pr[r1])%md;
          e=(e*pp[l1])%md;
//          for (k=l2; k<=r2; k++) e=(e*k)%md;
//          for (k=l1; k<=r1; k++) e=(e*pp[k])%md;
          d=e;
        }
      } else {
        e=pp[n*m];
//        printf("%d\n",int(e));
        e=(e*pr[n*m-r])%md;
//        printf("%d\n",int(e));
        e=(e*pr[r])%md;
//        printf("%d\n",int(e));
        d=e;
      }
      cur=(e*c[n][i])%md;
      cur=(cur*c[m][j])%md;
//      if (i+j>1) cur=(cur*(i+j-1))%md;
//      printf("%d %d: %d %d %d %d %d\n",i,j,int(cur),int(e),int(c[n][i]),int(c[m][j]),int(i+j-1));
      if ((i+j)&1) res+=md-cur; else res+=cur;
      if (res>=md) res-=md;
    }
    printf("%d\n",res%md);
  }
  return 0;
}
