#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
#define mod 1000000007
long long fac[40001];
long long invfac[40001];
long long power(long long a,long long b){
	long long c=1;
	while(b){
		if (b&1) c=(c*a)%mod;
		a=(a*a)%mod;
		b>>=1;
	}
	return c;
}
long long inv(long long x){
	return power(x,mod-2);
}
long long binom(int n,int m){
	if (m>n) return 0;
	long long res=fac[n];
	res=(res*invfac[m])%mod;
	res=(res*invfac[n-m])%mod;
	return res;
}
int main(){
	fac[0]=1;
	for (int i=1;i<=40000;i++)
		fac[i]=(fac[i-1]*i)%mod;
	for (int i=0;i<=40000;i++)
		invfac[i]=inv(fac[i]);
	int T;
	scanf("%d",&T);
	while(T--){
		int n,m,r;
		scanf("%d%d%d",&n,&m,&r);
		if (r<max(n,m)) { printf("0\n"); continue; }
		long long res=0;
		for (int i=0;i<=n;i++)
			for (int j=0;j<=m;j++){
				long long way;
				way=binom((n-i)*(m-j),r)*binom(n,i)%mod*binom(m,j)%mod;
				if (i+j&1) res=(res+mod-way)%mod;
				else res=(res+way)%mod;
			}
		printf("%d\n",(int)res);
	}
//	system("pause");
	return 0;
}
