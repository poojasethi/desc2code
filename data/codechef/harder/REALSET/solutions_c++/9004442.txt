#include <stdio.h>
#include <stdlib.h>
using namespace std;

int T,n,d,i,j,k,p;
int a[30005],b[30005],c[30005];
int pi[15],tot;

bool check()
{
	for(i=1;i<=tot;++i)
	if(d%pi[i]==0)
	{
		p=pi[i];
		for(j=0;j<d;++j)c[j]=b[(j+d-d/p)%d]-b[j];
		for(j=0;j<d;++j)b[j]=c[j];
	}
	for(i=0;i<d;++i)if(b[i])return false;
	return true;
}

int main()
{
	scanf("%d",&T);
	for(;T;--T)
	{
		scanf("%d",&n);
		for(i=0;i<n;++i)scanf("%d",&a[i]);
		tot=0;
		for(k=n,i=2;i<=k;++i)
		if(k%i==0)
		{
			pi[++tot]=i;
			for(;k%i==0;k/=i);
		}
		for(d=1;d<=n;++d)
		if(n%d==0)
		{
			for(i=0;i<d;++i)b[i]=0;
			for(i=0;i<n;++i)b[i%d]+=a[i];
			if(check())break;
		}
		if(d<=n)printf("YES\n");
		else printf("NO\n");
	}
}
