#include <cstdio>
#include <cmath>
#include <memory.h>
#define rep(i,x,y) for(int i=x;i<=y;i++)
using namespace std;
int a[30010],b[30010],c[30010],p[30],n;
bool check(int d)
{
	rep(i,0,d-1)
		b[i]=0;
	rep(i,0,n-1)
		b[i%d]+=a[i];
	int m=0;
	for(int tem=d,j=2;tem>1;j++)
		if(tem%j==0)
		{
			while(tem%j==0)tem/=j;
			p[++m]=j;
		}
	rep(i,1,m)
	{
		rep(j,0,d-1)
			c[j]=0;
		rep(j,0,d-1)
		{
			c[j]-=b[j];
			c[(j+d/p[i])%d]+=b[j];
		}
		rep(j,0,d-1)
			b[j]=c[j];
	}
	rep(i,0,d-1)
		if(b[i]!=0)return false;
	return true;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		rep(i,0,n-1)
			scanf("%d",&a[i]);
		bool tf=true;
		rep(i,1,n)
			if(n%i==0&&check(i))
			{
				tf=false;
				printf("YES\n");
				break;
			}
		if(tf)printf("NO\n");
	}
	return 0;
}
