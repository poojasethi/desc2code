#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define n	30005
#define For(i,a,b)	for(int i=a;i<=b;i++)

int		T,N,M,A[n],F[n],G[n],H[n];

bool	check(int d){
		For(i,0,d)	F[i]=0;
		For(i,0,N-1)F[i%d]+=A[i];
		For(t,1,*G)	if	(d%G[t]==0){
			For(i,0,d)	H[i]=F[i],F[i]=0;
			For(i,0,d)	F[(i+d/G[t])%d]+=H[i],F[i]-=H[i];
		}
		For(i,0,d)	if	(F[i])	return	0;
		return	1;
}

int		main(){
		scanf("%d",&T);
		for	(;T--;)	{
			scanf("%d",&N);
			For(i,0,N-1)scanf("%d",&A[i]);

			int M=N;	*G=0;
			for	(int i=2;i*i<=M;i++)
			if	(M%i==0)	for	(G[++*G]=i;M%i==0;M/=i);
			if	(M^1)	G[++*G]=M;

			int flag=0;
			for	(int i=1;i*i<=N;i++)
			if	(N%i==0)	{
				if	(check(i))	{flag=1;break;}
				if	(check(N/i)){flag=1;break;}
			}
			puts(flag?"YES":"NO");
		}
}
