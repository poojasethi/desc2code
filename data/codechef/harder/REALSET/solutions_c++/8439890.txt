int n;
int a[333333];
int b[333333];
int pr[333333];
int test(int d){
	for (int i=0;i<n;i++)b[i]=a[i];
	for (int i=n-1;i>=d;i--){
		b[i-d]+=b[i];
		b[i]=0;
	}
	for (int i=1;i<d;i++)if(d%i==0 &&!pr[d/i]){
		for (int j=d-1;j>=0;j--)b[j+i]-=b[j];
		for (int j=d-1+i;j>=d;j--){
			b[j-d]+=b[j];
			b[j]=0;
		}
	}
	for (int i=0;i<d;i++)if(b[i])return 0;
	return 1;
}

int main()
{
	for (int i=2;i<=30000;i++)if(!pr[i])
		for (int j=i+i;j<=30000;j+=i)pr[j]=1;
	int tes;scanf("%d",&tes);
	while(tes--){
		memset(b,0,sizeof(b));
		scanf("%d",&n);
		for (int i=0;i<n;i++)scanf("%d",&a[i]);
		int bo=0;
		for (int d=1;d<=n;d++)if(n%d==0){
			if(test(d)){
				bo=1;
				break;
			}
		}
		if(bo)printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
