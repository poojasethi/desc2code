#include<cstring>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<iostream>
using namespace std;
#define mo 1000000007
int n;
int qp(int a,int b){int ans=1;do{if(b&1)ans=1ll*ans*a%mo;a=1ll*a*a%mo;}while(b>>=1);return ans;}
char C[105];
int cmod(int p){
	int ans=0;
	for (int i=0;C[i];i++)ans=(1ll*ans*10+C[i]-'0')%p;
	return ans;
}
int d[55];
int b[55][505];
int tmp[505],tmp2[505];
void calcb(int I){
	int D=d[I];
	for (int j=0;j<D;j++)tmp[j]=0;
	tmp[0]=1;
	for (int i=1;i<=n;i++){
		int j=(i==I)?1:d[i];
		int sua=0;
		for (int l=0;l<D;l++)sua=(sua+tmp[l])%mo;
		tmp2[0]=0;for (int l=0;l<D;l++)tmp2[0]=(tmp2[0]+1ll*l*tmp[((-l*j)%D+D)%D])%mo;
		for (int l=1;l<D;l++)tmp2[l*j%D]=(tmp2[(l-1)*j%D]-1ll*D*tmp[l*j%D]+sua)%mo;
		int x=qp(D,mo-2);
		for (int l=0;l<D;l++)tmp[l]=-1ll*tmp2[l]*x%mo;
	}
	for (int i=0;i<=D-2;i++)tmp[i]=(tmp[i]-tmp[D-1])%mo;
	tmp[D-1]=0;
	for (int i=0;i<=D-1;i++)b[I][i]=(tmp[i]+mo)%mo;
}
int dp[55];
void work(){
	scanf("%d",&n);
	scanf("%s",C);
	for (int i=1;i<=n;i++)scanf("%d",&d[i]);
	for (int i=1;i<=n;i++)calcb(i);
	memset(dp,0,sizeof(dp));	
	dp[0]=1;
	for (int j=1;j<=n;j++)
		for (int i=0;i<=n-1;i++)if(i>=d[j])dp[i]=(dp[i-d[j]]+dp[i])%mo;
	for (int i=0;i<=n-1;i++)
		for (int j=1;j<=n;j++){
			int id=i%d[j];
			dp[i]=(dp[i]-1ll*(id==0?b[j][id]:b[j][id]-b[j][id-1]))%mo;
		}
	int sa=0;
	int c=cmod(mo);
	for (int i=0;i<=n-1;i++){
		int tmp=1;
		for (int j=0;j<=n-1;j++)if(j!=i){
			tmp=1ll*tmp*(c-j)%mo;
			tmp=1ll*tmp*qp(i-j,mo-2)%mo;
		}
		tmp=1ll*tmp*dp[i]%mo;
		sa=(sa+tmp)%mo;
	}
	int sb=0;
	for (int j=1;j<=n;j++){
		int id=cmod(d[j]);
		sb=(sb+1ll*(id==0?b[j][id]:b[j][id]-b[j][id-1]))%mo;
	}
	int ret=((sa+sb)%mo+mo)%mo;
	printf("%d\n",ret);
}
int main()
{
	int tes;
	scanf("%d",&tes);
	while(tes--)work();
	return 0;
}
