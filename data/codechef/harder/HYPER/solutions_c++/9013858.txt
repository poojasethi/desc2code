#include <stdio.h>
#include <stdlib.h>
using namespace std;

int T,qn[25];
int n,m,i,j,k;
int cnt[25],e[25],lg[2000005];
long long fac[25],C[25][25];
long long ans[25],bicon[25],bicon_val[25][25];

bool check_bicon(int n)
{
	if(n==1)return true;
	int N=(1<<n+1)-2,i,j,k,x,te;
	for(k=0;k<=n;++k)
	{
		if(k==1)x=2;else x=1;
		te=e[x];j=N;
		if(k)te|=1<<k,j-=1<<k;
		for(;te!=N&&(i=te&j);)
		{
			i=lg[i&-i];
			te|=e[i];j-=1<<i;
		}
		if(te!=N)return false;
	}
	return true;
}

void dfs1(int x,int y,int z)
{
	if(m+z>n)return;
	if(x>m||m+z==n)
	{
		if(check_bicon(m))
		{
			for(int i=m+z;i<=n;++i)
			bicon[i]+=C[i][m]*bicon_val[z][i-m];
		}
		return;
	}
	if(y>m)
	{
		dfs1(x+1,x+2,z);
		return;
	}
	dfs1(x,y+1,z);
	e[x]+=1<<y;e[y]+=1<<x;
	dfs1(x,y+1,z+1);
	e[x]-=1<<y;e[y]-=1<<x;
}

void dfs2(int x,int y,int z,long long f)
{
	if(x>y)return;
	if(y==m)
	{
		ans[y]+=f;
		return;
	}
	if(z>m)
	{
		dfs2(x+1,y,3,f);
		return;
	}
	for(int i=0;i<=cnt[z];++i)
	{
		cnt[z]-=i;
		dfs2(x,y+i*(z-1),z+1,f*C[cnt[z]+i][i]);
		cnt[z]+=i;
	}
}

void dfs3(int x,int y,long long z)
{
	if(x>n)
	{
		m=y;
		dfs2(1,1,3,z);
		return;
	}
	cnt[x]=0;
	dfs3(x+1,y,z);
	if(y+x-1>n)return;
	long long y_=x-1,z_=bicon[x];
	for(;y+y_<=n;)
	{
		++cnt[x];
		dfs3(x+1,y+y_,z*z_*C[y+y_-1][y_]);
		y_+=x-1;z_*=C[y_-1][x-2]*bicon[x];
	}
}

int main()
{
	scanf("%d",&T);
	for(i=1;i<=T;++i)
	{
		scanf("%d",&qn[i]);
		if(qn[i]>=16)continue;
		if(qn[i]>n)n=qn[i];
	}
	C[0][0]=1;
	for(i=0;i<=n;++i)
	for(j=0;j<=i;++j)
	C[i+1][j]+=C[i][j],C[i+1][j+1]+=C[i][j];
	fac[0]=1;
	for(i=1;i<=n;++i)fac[i]=fac[i-1]*i;
	bicon_val[0][0]=1;
	for(i=1;i<=n;++i)
	for(j=i;j<=n;++j)
	for(k=0;k<j;++k)
	bicon_val[i][j]+=bicon_val[i-1][k]*C[j][k];
	for(i=1;i<=n;++i)e[i]=1<<i,lg[1<<i]=i;
	for(m=2;m<=8;++m)dfs1(1,2,0);
	bicon[3]=1;
	dfs3(3,1,1);
	for(i=1;i<=T;++i)
	if(qn[i]==16)printf("61479081902937000\n");
	else if(qn[i]==17)printf("2761923686066698561\n");
	else printf("%lld\n",ans[qn[i]]);
}
