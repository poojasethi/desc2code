#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_v) (int)(_v).size()
#define all(_v) (_v).begin(),(_v).end()
#define pii pair<int,int>
typedef long long ll;
const int INF = 1000000000;
const ll MOD = 1e9 + 7;

const int maxn = 1000010;
ll fact[maxn];

ll fast_expo(ll a,ll b) {
	if(b == 0) return 1;
	ll p;
	if(b % 2 == 0) {
		p = fast_expo(a,b / 2);
		p = (p * p) % MOD;
	}
	else p = (a * fast_expo(a,b - 1)) % MOD;
	return p;
}

int main() {
    int t; cin >> t;
	
	fact[0] = 1;
	for (int i = 1; i <= maxn - 4; i ++) fact[i] = (fact[i - 1] * i) % MOD;

	while (t --) {
		ll n,m;
		cin >> n >> m;
		ll ans = fact[m - n];
		ll k = n + n - m;
		ans = (ans * fast_expo(m - n + 1,(k + 1) / 2) + MOD) % MOD;
		ans = (ans * fast_expo(m - n,k / 2) + MOD) % MOD;
		cout << ans << "\n";
	}
	return 0;
}
