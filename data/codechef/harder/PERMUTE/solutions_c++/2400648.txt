#include<stdio.h>
#define MOD 1000000007
typedef long long int ll;
int fact[1000005];
int expp(int a,int n){
	int ans = 1,p = a;
	while(n){
		if(n%2 == 1)
			ans = (ll(ans)*p)%MOD ;
		p = (ll(p)*p)%MOD ;
		n = n/2 ;
	}
	return ans ;
}
int main(){
	int i,n,m,t,k,ans;
	fact[0] = 1 ;

		fact[i] = ll(i)*fact[i-1]% MOD ;
    scanf("%d",&t);
	while(t--){
		scanf("%d%d",&n,&m);
		k = 2*n - m ;
		/*printf("%d\n",fact[n-k]);
		printf("%d\n",expp(n-k+1,(k+1)/2));
		printf("%d\n",expp(n-k,k/2));*/
		ans = ll(ll(fact[n-k])*expp(n-k+1,(k+1)/2)%MOD)*expp(n-k,k/2)%MOD;
		printf("%d\n",ans);
	}
	return 0;
}
