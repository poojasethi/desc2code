#include <iostream>

using namespace std;
typedef long long ll;
const int maxn = 1e6 + 5;

ll n,m;
ll mod = 1e9 + 7;
ll fact[maxn];

void factor()
{
	fact[0] = fact[1] = 1;
	for(int i = 2; i < maxn; i++)
	{
		fact[i] = fact[i - 1]*(ll)i;
		fact[i] %= mod;
	}
}

ll power(ll a, int pot)
{
	if(pot == 1)return a;
	if(pot == 0)return 1;

	ll res  = power(a,pot/2);
	res = (res*res)%mod;
	if(pot & 1)
		res *= a;
	res %= mod;
	return res;
}

int main()
{
	int t;
	cin >> t;
	factor();
	while(t--)
	{
		cin >> n >> m;
		ll k = m - n;

		ll t = (n - k)/2;

		//if(n - 2*t > k)
			//t++;
		//cout << t << " " << k << " " << n-2*t << endl;

		cout << (power((k*(k + 1))%mod,t)*fact[n - 2*t])%mod << endl;
	}
	return 0;
}
