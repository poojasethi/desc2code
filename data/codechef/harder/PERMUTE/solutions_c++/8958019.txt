# define limit 1000000L
# define prime 1000000007LL
using namespace std;
long long int fact[limit+1L],x,k;
long int N,M,i,T;
void initiallise()
{
	fact[0]=1LL;
	for(i=1;i<=limit;i++)
		fact[i]=(fact[i-1LL]*i)%prime;
}
long long int power(long long int base,long long int exp)
{
	if(exp==0)
		return 1LL;
	if(exp==1)
		return base%prime;
	long long int y=power(base,exp/2);
	y=(y*y)%prime;
	if(exp%2)
		y*=base;
	return y%prime;
}
int main()
{
	initiallise();
    cin>>T;
    while(T-->0)
    {
        cin>>N>>M;
        k=(M-N);
        x=(k*(k+1))%prime;
        //cout<<x<<"\n";
        x=power(x,N-(M+1)/2);
        //cout<<x<<"\n";
        x=(x*fact[(M%2)?(k+1):k])%prime;
        while(x<0)
			x+=prime;
		cout<<x<<"\n";
    }
    return 0;
}
