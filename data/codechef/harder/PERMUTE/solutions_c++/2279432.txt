using namespace std;
#define ll long long
 
ll n,m;
ll mod;
ll pow(ll x, ll y) 
{
	if(y==0) return 1;
	ll ans = pow(x,y/2)%mod;
	ans*=ans; ans%=mod;
	if(y%2) {
		ans*=x;
		ans%=mod;
	}
	return ans%mod;
}
ll fact[1000001];
int t;
 
int main()
 {
	mod = 1000000007;
	scanf("%d",&t);
	fact[0] = 1;
	for(int i=1; i<=1000000; i++) {
		fact[i] = fact[i-1]*i;
		fact[i]%=mod;
	}
	while(t--) {
		ll n,m;
		scanf("%lld %lld",&n,&m);
		ll k=m-n;
		ll ans = pow(k,(2*n-m)/2)*pow(m-n+1,(2*n-m-1)/2);
		ans %= mod;
		ans*= fact[k+1];
		ans %=mod;
		printf("%lld\n",ans);
	}
	return 0;
}
