#include<iostream>
#include<cstdio>
#include<math.h>

#define  MOD 1000000007
#define ull long long

using namespace std;

      ull fact[1000001];

        void facto()
        {
            fact[0]=1;
            for(ull i=1;i<=1000000;i++)
                fact[i]=(fact[i-1]*i)%MOD;
        }


        ull pow(ull base,ull power)
        {
            ull ans=1; ull sum=base;

            while(power>0)
            {
                if(power%2==1) //(power&1)
                ans=(ans*sum)%MOD;

                sum=(sum*sum)%MOD;
                 power>>=1;
            }
                return ans%MOD;
        }

        int main()
        {
             //freopen("C:\\Users\\Sameer Barha\\Desktop\\in.txt","r",stdin);
             //freopen("C:\\Users\\Sameer Barha\\Desktop\\out.txt","w",stdout);

            facto();

            int T;
            ull n,m,r,ans,c,k;
            scanf("%d",&T);
            while(T--)
            {
                scanf("%lld %lld",&n,&m);

                r=((n<<1)-m)>>1;
                k=m-n;
                c=k*(k+1);
                ans=(pow(c%MOD,r)*fact[n-2*r])%MOD;

               // cout<<"pow= "<<pow(c,r)<<endl;
                //cout<<"Fact= "<<fact[n-2*r]<<endl;
                printf("%lld\n",ans);
            }
            return 0;
        }
