typedef long long LL;

long long int pow1(long long int a,long long int b)
{
	long long int res=1,mod=1000000007;
	while(b>0)
	{
		if(b%2!=0)
		res*=a;
		a*=a;a%=mod;
		res%=mod;
		b/=2;
	}
	return res;
}


 
LL modexp(LL a,LL b)
{
	if(b==0) return 1;
	if(b%2) return ((a%M)*modexp(a,b-1))%M;
	LL q=modexp(a,b/2);
	return (q*q)%M;
}
