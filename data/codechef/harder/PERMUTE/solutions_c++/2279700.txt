#include <cstdio>
using namespace std;
#define M 1000000007

static long long f[1000010];
long long T, N, L;

inline void fact(){
    f[0]=f[1]=1;
    for(int i=2; i<1000010; ++i)
        f[i]=(f[i-1]*i)%M;
}
inline long long modexp(long long x, long long p){
    long long RET=1;
    while(p>0){
        if(p%2)RET=(RET*x)%M;
        p/=2; x=(x*x)%M;
    }
    return RET;
}

int main(){
    fact(); scanf("%lld", &T);
    while(T--){
        scanf("%lld %lld", &N, &L);
        long long K=2*N-L;
        long long tmp1=f[N-K], tmp2=(N-K+1);
        tmp2=(modexp(tmp2, (K+1)/2)*modexp(tmp2-1, K/2))%M;
        printf("%lld\n", (tmp1*tmp2)%M);
    }
    return 0;
}
