# define limit 1000000L
# define prime 1000000007L
using namespace std;
long int fact[limit+1L],x;
long int N,M,i,T,k;
void initiallise()
{
	fact[0]=1;
	for(i=1;i<=limit;i++)
		fact[i]=((long long)fact[i-1]*i)%prime;
}
long int power(long int base,long  int exp)
{
	if(exp==0)
		return 1;
	if(exp==1)
		return base%prime;
	long int y=power(base,exp/2);
	y=((long long)y*y)%prime;
	if(exp%2)
		y=((long long)y*base)%prime;
	return y;
}
int main()
{
	initiallise();
    cin>>T;
    while(T-->0)
    {
        cin>>N>>M;
        k=(M-N);
        x=((long long)k*(k+1))%prime;
        //cout<<x<<"\n";
        x=power(x,N-(M+1)/2);
        //cout<<x<<"\n";
        x=((long long)x*fact[(M%2)?(k+1):k])%prime;
        while(x<0)
			x+=prime;
		cout<<x<<"\n";
    }
    return 0;
}
