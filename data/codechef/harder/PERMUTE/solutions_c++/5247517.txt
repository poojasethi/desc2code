#include <bits/stdc++.h>
using namespace std;

#define		pii		pair<int,int>
#define     vi     	vector<int>
#define     mp   	make_pair
#define		pb		push_back
#define     X 		first
#define     Y    	second
#define 	ill		int long long
#define MOD 1000000007

ill factorial[1000005];

ill power(ill val, ill b)
{
	ill x,y;
	x = 1; y = val%MOD;
	while(b != 0)
	{
		if(b&1) x = (x*y)%MOD;
		y = (y*y)%MOD;
		b/=2;
	}
	return x;
}
void solve()
{
	ill ans;
	ill n,m;
	cin>>n>>m;
	m = m - n;
	if((n-m)&1)//n-m%2 == 1
	{
		ans = (factorial[m+1] * power(m*(m+1), (n-m-1)>>1))%MOD;
	}
	else if((n-m+1)&1)//n-m%2 == 0
	{
		ans = (((factorial[m+1]*m)%MOD) * power(m*(m+1), (n-m-2)>>1))%MOD;
	}
	cout<<ans<<endl;
}

void precompute()
{
	factorial[0] = 1;
	for(int i=1;i<1000005;i++)
		factorial[i] = (factorial[i-1]*i)%MOD;
}
int main()
{
	int t;
	precompute();
	cin>>t;
	while(t--)
		solve();
	return 0;
}
