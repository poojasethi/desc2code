#include <cstdio>
#define LL long long
#define MOD 1000000007

using namespace std;

int sil[1000001];
int S;


int silnia(int x) 
{
  if(x <= S) return sil[x];
  while(S < x){ ++S; sil[S] = (LL)sil[S - 1] * S % MOD; }
  return sil[x];
}

LL powerMod(LL x, int n) 
{
  LL w = 1, p = x;
  
  while(n) 
  {
    if(n % 2) w = w * p % MOD;
    p = p * p % MOD;
    n /= 2;
  }

  return w;
}

int t, n, m;

int main() 
{
  sil[0] = 1;

  scanf("%d", &t);
  while(t--) 
  {
    scanf("%d %d", &n, &m);

    LL x = m - n + 1;
    int d = n - x;
    int k = x * (x - 1) % MOD;
    LL w = silnia(x);

    if(d % 2)
     w = w * powerMod(k, d / 2) % MOD * (x - 1) % MOD;
    else
     w = w * powerMod(k, d / 2) % MOD;

    printf("%lld\n",w);
  }

  return 0;
}
