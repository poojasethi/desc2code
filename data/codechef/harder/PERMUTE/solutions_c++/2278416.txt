#include<iostream>
using namespace std;
#define mod 1000000007
#define Mod 1000000007
#define MOD 1000000007
long long int a[1000005];
long long int poww(long long int X, long long int Y)
{

    long long int Product=1;

    for( ; Y; Y >>= 1)

    {

        if(Y & 1) Product = (Product * X) %Mod;

        X = (X * X) % Mod;

    }

    return Product;

}

void gen()

{a[0]=1;

     for(int i=1;i<1000001;++i)

     a[i]=(i*a[i-1])%mod;

}
int main()
{
    int t,n,m;
    gen();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&m);
        long long diff,x,y,ans,temp,p;
        diff=m-n;
        p=n-diff;
        ans=a[diff+1];
        if(p%2)
        {
            temp=p/2;
            x=poww(diff,temp);
            y=poww(diff+1,temp);
        }
        else
        {
            temp=p/2;
            x=poww(diff,temp);
            y=poww(diff+1,temp-1);
        }
        ans=(((ans*x)%MOD)*(y))%mod;
        printf("%lld\n",ans);
    }

    return 0;

}
