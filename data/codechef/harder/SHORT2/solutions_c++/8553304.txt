#include <cstdio>
#include <cmath>
using namespace std;
long long p,b,d,s;
int t,sum,tem,a,t2,i;
int main()
{
  scanf("%d",&t);
  while(t--)
  {
    scanf("%lld",&p);
    sum=0;
    tem=(int)(sqrt(p+1))+1;
    tem=(int)(sqrt(p+tem));
    for(i=1;i<=tem;i++)
    {
      t2=i-p%i;
      d=i;
      a=t2;
      b=(a+p)/d;
      s=a%p;
      if(s>0&&(d+1)%a==0&&a<b)sum++;
      if(a<=1&&(a+d)%p>0&&(d+1)%a==0&&a+d<b+1)sum++;
      d=b;
      b=i;
      if(a==b){a=0;s=0;}
      if(d<=tem)continue;
      if(s>0&&(d+1)%a==0)sum++;
    }
    printf("%d\n",sum*6+8);
  }
  return 0;
}
