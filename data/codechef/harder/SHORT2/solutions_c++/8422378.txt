	db w=1+sqrt(p+1.0);

	for (int d=1;d<=xx;d++){
		int a=(-p)%d+d;
		while(a<=d+1){
			if((d+1)%a==0 && (a+p)/d>a && a%p)an++;
			a+=d;
		}
	}
	for (int b=1;b<=xx;b++){
		int a=(-p)%b;if(a<0)a+=b;
		ll d=(a+p)/b;
		if(a>=1 && d>xx && (d+1)%a==0 && a%p)an++;
	}
	an=an*2+1;

}
