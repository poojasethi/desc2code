#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <algorithm>
#include <cmath>

#define MAXN 11000

using namespace std;

typedef long long int LL;

LL p;

int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%lld",&p);
        int ans=0;
        int upperBound=(int)sqrt((double)(1+p)+sqrt((double)(1+p)));
        for(int d=1;d<=upperBound;d++)
        {
            int a=(-p)%d;
            if(a<0) a+=d;
            while(a<=d+1)
            {
                LL b=(a+p)/d;
                if(a>=1&&(d+1)%a==0&&a<b&&a%p&&b%p)
                    ans++;
                a+=d;
            }
        }
        for(int b=1;b<=upperBound;b++)
        {
            int a=(-p)%b;
            if(a<0) a+=b;
            LL d=(a+p)/b;
            while(a<b)
            {
                if(a>=1&&(d+1)%a==0&&a%p&&b%p&&d>upperBound)
                    ans++;
                a+=d;
            }
        }
        printf("%d\n",(ans*2+1)*3+5);
    }
    return 0;
}
