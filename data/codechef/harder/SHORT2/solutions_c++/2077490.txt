#include <stdio.h>
#include <math.h>
#include <memory.h>
 
long long p,u;
int t,d,a,ans,w,dd;
 
int main() {
  scanf("%d",&t);
  while (t--) {
    scanf("%lld",&p);
    w = (int)(sqrt(1.0+p)+1);
    dd = (int)(sqrt(1.0+p+w)+1);
    ans = 1;
    for (d=1;d<=dd;d++) {
      u = p/d+1; a = u*d-p;
      if (d <= u && (d+1) % a == 0 && u > a) ans += 2;
      if (a == 1 && p > 2 && u > d) ans += 2;
      if (d < u && (u+1) % a == 0 && d > a) ans += 2;
    }
    printf("%d\n",3*ans+5);
  }
  
}
