	while(i < N)
	{
		j = i + 1;
		while(j < N && V[j].f == V[i].f)
			++j;
		
		if(V[i].s > prev)
			prev = V[j-1].s;
		else
		{
			++cnt;
			int k = i;
			while(k < j && V[k].s < prev)
				++k;
			
			--k;
			prev = V[k].s;
		}
		i = j;
	}	
	
	for(;cnt > 1;)
	{
		++rez;
		cnt = (cnt + 1) / 2;
	}

}
