void dfs(long v)
{
 sub[v]=c[v];
 for (int i=0;i<g[v].size();i++)
 {
  long q=g[v][i];
  dfs(q);
  sub[v]+=sub[q];
 }
}
void add_item(long w)
{
 for (int i=s;i>=w;--i)
  ways[i]+=ways[i-w];
}

void del_item(long w)
{
 for (int i=w;i<=s;i++)
  ways[i]-=ways[i-w];
}

void solve(long v,long rp)
{
 rp+=c[v];
 //cout<<v<<"  ^^ "<<rp<<endl;
 for (int i=0;i<=s;i++)
 if (ways[i])can[i+rp]=1;
 
 /*for (int i=0;i<=s;i++)
  if (i+rp==12)
   cout<<v<<"   "<<rp<<" "<<ways[i]<<" "<<v<<endl;
   */
 for (int i=0;i<g[v].size();i++)
 {
  add_item(sub[g[v][i]]);
 }
 for (int i=0;i<g[v].size();i++)
 {
  del_item(sub[g[v][i]]);
  solve(g[v][i],rp);
  add_item(sub[g[v][i]]);
 }
 
 for (int i=0;i<g[v].size();i++)
 {
  del_item(sub[g[v][i]]);
 }
}
