#include <cstdio>

typedef long long LL;

const int maxN = 10000000;
const int MOD = 1000000007;

int f[maxN+1];

int main() {
  f[1] = f[2] = 0;
  f[3] = 1;
  int half = (MOD - 1) / 2;
  int b = half;
  int a = (MOD + 1) / 2 + 1;
  int t = 3;
  int s = a;
  for (int n = 3; n < maxN; ++n) {
    t += n;         if (t >= MOD) t -= MOD;
    s += n + half;  if (s >= MOD) s -= MOD;
    a = LL(a) * t % MOD;
    b = LL(b) * s % MOD;
    f[n+1] = (LL(t) * f[n] + a + b) % MOD;
  }
  int T;
  scanf("%d", &T);
  for(int t=0;t<T;t++) {
    int n;
    scanf("%d", &n);
    printf("%d\n",f[n]);
  }
  return 0;
}
