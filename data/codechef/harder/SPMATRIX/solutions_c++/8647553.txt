const int mo=1000000007;
int qp(int a,ll b){int ans=1;do{if(b&1)ans=1ll*ans*a%mo;a=1ll*a*a%mo;}while(b>>=1);return ans;}
int gcd(int a,int b){return b?gcd(b,a%b):a;}

int fac[11111111],an[11111111];
int main()
{
	fac[0]=1,an[0]=0;
	for (int i=1;i<=10000000;i++){
		fac[i]=1ll*fac[i-1]*i%mo;
		an[i]=(1ll*i*an[i-1]+fac[i-1])%mo;
	}
	int tes;
	scanf("%d",&tes);
	while(tes--){
		int n;gn(n);
		int ann=1ll*fac[n]*((1ll*fac[n]*3*(mo+1>>1)-2ll*fac[n-1]-an[n-1])%mo)%mo;
		ann=1ll*ann*qp(3,mo-2)%mo*qp(2,mo-1-(n-1))%mo;
		ann=(ann+mo)%mo;

	}
