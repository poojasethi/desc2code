#include <cstdio>
#include <cmath>
#include <memory.h>
using namespace std;
#define ll long long
const int p=1000000007;
ll f[10000001],sum[10000001];
ll ksm(int bas,int k)
{
	if(k==0)return 1;
	if(k==1)return bas%p;
	ll ans=ksm(bas,k>>1);
	ans=(ans*ans)%p;
	if(k&1==1)ans=(ans*bas)%p;
	return ans;
}
int main()
{	
	f[0]=1;
	for(int i=1;i<=10000000;i++)
		{
			f[i]=(f[i-1]*i)%p;
			sum[i]=(sum[i-1]*i+f[i-1])%p;
		}
	ll rev2=ksm(2,p-2),rev3=ksm(3,p-2),d23=(2*rev3)%p;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		ll tem=(f[n]*ksm(rev2,n-1))%p;
		ll tem2=f[n]*rev2-f[n-1]*d23-sum[n-1]*rev3;
		if(tem2<0)tem2=p-((-tem2)%p);
		tem2%=p;
		printf("%lld\n",(tem*tem2)%p);
	}
	return 0;
}
