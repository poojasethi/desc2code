const int N=111111,M=333;
int l,m,n,t,C,xtot,x,y,z,d;
vector<pii>a[N];
int f[N],len[N],sz[N],su[N],dfsx[N],bz[N][22],dep[N],dlen[N],suf[N];
int sml[N][M*2],lar[N][M*2],lcnt[N][M*2];
int BLO,STEP;
void dfs(int x){
	sz[x]=1;dfsx[++xtot]=x;
	rep(i,0,SZ(a[x])-1)if(a[x][i].X!=f[x]){
		dep[a[x][i].X]=dep[x]+1;f[a[x][i].X]=x;len[a[x][i].X]=a[x][i].Y;dlen[a[x][i].X]=dlen[x]+a[x][i].Y;
		dfs(a[x][i].X);
		sz[x]+=sz[a[x][i].X];
	}
}
int lca(int x,int y){
	if(dep[x]>dep[y])swap(x,y);
	dep(i,20,0)if(bz[y][i]&&dep[bz[y][i]]>=dep[x])y=bz[y][i];
	if(x==y)return x;
	dep(i,20,0)if(bz[x][i]!=bz[y][i])x=bz[x][i],y=bz[y][i];
	return f[x];
}
pii work(int x){
	int st=0,rem=0;
	while(x!=d){
		if(su[x]&&dep[x]-dep[d]>BLO){
			if(rem>=dlen[x]-dlen[suf[x]])rem-=dlen[x]-dlen[suf[x]];
			else{
				int p=sml[x][rem];rem=0;
				if(dlen[p]-dlen[suf[x]]<=z){
					rem=z-(dlen[p]-dlen[suf[x]]);st++;
				}else{
					st+=lcnt[p][z];p=lar[p][z];
					if(p!=suf[x])st++,rem=z-(dlen[p]-dlen[suf[x]]);
				}
			}x=suf[x];
		}else{
			if(rem>=len[x])rem-=len[x];else st++,rem=z-len[x];x=f[x];
		}
	}return mp(st,rem);
}

	dfs(1);
	BLO=int(sqrt(n));STEP=0;
	rep(i,1,n)if(dep[i]%BLO==0)su[i]=1;su[1]=1;
	rep(i,2,n){
		int p=i,cnt=0;while(1){
			cnt+=len[p];p=f[p];sml[i][cnt]=p;if(su[p])break;
		}sml[i][0]=i;rep(j,1,cnt)if(!sml[i][j])sml[i][j]=sml[i][j-1];suf[i]=p;
		MAX(STEP,cnt);
	}
	rep(_,1,n){
		int i=dfsx[_];
		rep(j,1,STEP){
			if(!sml[i][j]){rep(k,j,STEP)lar[i][k]=i;break;}
			if(su[sml[i][j]])lar[i][j]=sml[i][j],lcnt[i][j]=1;
			else lar[i][j]=lar[sml[i][j]][j],lcnt[i][j]=lcnt[sml[i][j]][j]+1;
		}
	}
	rep(i,1,n)bz[i][0]=f[i];rep(j,1,20)rep(i,1,n)bz[i][j]=bz[bz[i][j-1]][j-1];
	scanf("%d",&C);rep(_,1,C){
		scanf("%d%d%d",&x,&y,&z);d=lca(x,y);
