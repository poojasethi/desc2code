#include <stdio.h>
#include <stdlib.h>
#include <math.h>
using namespace std;

int T,n,m,i,j,k,l;
double X,ans[65][65],power[65][65],tmp[65][65];

int main()
{
	scanf("%d",&T);
	for(;T;--T)
	{
		scanf("%d%d%lf",&m,&n,&X);l=m+m;
		for(i=1;i<=l;++i)
		for(j=1;j<=l;++j)
		ans[i][j]=power[i][j]=0;
		for(i=1;i<=m;++i)
		{
			if(i>1)power[i][i-1]=sin(X),power[i+m][i-1]=cos(X);
			power[i][i]=cos(X);power[i][i+m]=sin(X);
			power[i+m][i]=-sin(X);power[i+m][i+m]=cos(X);
		}
		for(i=1;i<=l;++i)ans[i][i]=1;
		for(--n;n;n>>=1)
		{
			if(n&1)
			{
				for(i=1;i<=l;++i)
				for(j=1;j<=l;++j)
				tmp[i][j]=0;
				for(i=1;i<=l;++i)
				for(j=1;j<=l;++j)
				for(k=1;k<=l;++k)
				tmp[i][k]+=ans[i][j]*power[j][k];
				for(i=1;i<=l;++i)
				for(j=1;j<=l;++j)
				ans[i][j]=tmp[i][j];
			}
			for(i=1;i<=l;++i)
			for(j=1;j<=l;++j)
			tmp[i][j]=0;
			for(i=1;i<=l;++i)
			for(j=1;j<=l;++j)
			for(k=1;k<=l;++k)
			tmp[i][k]+=power[i][j]*power[j][k];
			for(i=1;i<=l;++i)
			for(j=1;j<=l;++j)
			power[i][j]=tmp[i][j];
		}
		printf("%.10lf\n",sin(X)*ans[m][1]+cos(X)*ans[m][m+1]);
	}
}
