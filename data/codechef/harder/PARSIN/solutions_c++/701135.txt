#include<cmath>
#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int T; scanf("%d",&T);
	while(T--){
		int m,n;
		double x; scanf("%d%d%lf",&m,&n,&x);

		double A[60][60]={};
		rep(i,m){
			A[i][m+i]=1;
			A[i+m][i]=-1;
			A[i+m][i+m]=2*cos(x);
			if(i>0) A[i+m][i+m-1]=sin(x);
		}

		double v[60]={};
		v[m]=sin(x);

		rep(l,30){
			if(n&(1<<l)){
				double v2[60]={};
				rep(i,2*m) rep(j,2*m) v2[i]+=A[i][j]*v[j];
				rep(i,2*m) v[i]=v2[i];
			}

			double A2[60][60]={};
			rep(i,2*m) rep(j,2*m) rep(k,2*m) A2[i][j]+=A[i][k]*A[k][j];
			rep(i,2*m) rep(j,2*m) A[i][j]=A2[i][j];
		}

		printf("%f\n",v[m-1]);
	}

	return 0;
}
