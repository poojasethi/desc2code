#include <cstdio>
#include <cmath>
#include <complex>
using namespace std;

const double PI=acos(-1.0);

typedef complex<double> CD;

CD expi(double x) { return CD(cos(x),sin(x)); }

#define M 34
double C[M][M];

double solve(int m,int n,double x)
{
	if(x<0.005 || n<m) return 0.;
	CD pw[M];
	double q=2*sin(x);
	double qq=1;
	int j,k;
	for(k=0;k<M;k++)
	{
		pw[k]=expi(k*(-PI/2-x))/qq;
		qq*=q;
	}
	CD a[M];
	for(k=0;k<m;k++)
	{
		a[k]=(k%2?-1.:1.)*C[m][k];
		for(j=0;j<k;j++)
			a[k]-=a[j]*pw[k-j]*C[m][k-j];
	}
	double real=0.,imag=0.;
	for(j=m;j;j--)
	{
		double coef=1.*n/j+1;
		real=real*coef+a[m-j].real();
		imag=imag*coef+a[m-j].imag();
	}
	double res;
	if(m%2==0)
		res=cos(n*x)*real-sin(n*x)*imag;
	else
		res=cos(n*x)*imag+sin(n*x)*real;
	res*=(m/2%2?-1.:1.)/(1<<(m-1));
	return res;
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("in.txt","rt",stdin);
#endif
	int n,m;
	for(n=0;n<M;n++)
	{
		C[n][0]=1;
		for(m=1;m<=n;m++)
			C[n][m]=C[n-1][m]+C[n-1][m-1];
	}
	int T;
	for(scanf("%d",&T);T--;)
	{
		double x;
		scanf("%d%d%lf",&m,&n,&x);
		double ans=solve(m,n,x);
		printf("%.10e\n",ans);
	}
	return 0;
}
