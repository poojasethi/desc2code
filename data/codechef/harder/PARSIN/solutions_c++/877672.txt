const int MaxM=36;

struct matrix
{
	double g[MaxM*2][MaxM*2];

	void set(int i,int j,double x) { g[i][j]=x; }
}a;

void mul(matrix &a,int M)
{
	matrix c;

	for(int i=0;i<M;i++) for(int j=0;j<M;j++)
		c.set(i,j,0);

	for(int i=0;i<M;i++) for(int j=0;j<M;j++)
		for(int k=0;k<M;k++) c.g[i][j]+=a.g[i][k]*a.g[k][j];

	a=c;
}

void build(int m,double x)
{
	double sinX=sin(x),cosX=cos(x);

	for(int i=0;i<2*m;i++) for(int j=0;j<2*m;j++)
		a.set(i,j,0);
	for(int i=0;i<m;i++)
	{
		if(i>0)
		{
			 a.set(2*i,2*i-2,sinX);
			 a.set(2*i+1,2*i-2,cosX);
		}
		a.set(2*i,2*i,cosX);
		a.set(2*i,2*i+1,sinX);
		a.set(2*i+1,2*i,-sinX);
		a.set(2*i+1,2*i+1,cosX);
	}
}

void calc(int n,int M)
{
	double res[MaxM*2],tmp[MaxM*2];

	for(int i=0;i<M;i++) res[i]=0;
	res[1]=1;
	for(;n;n/=2)
	{
		if(n&1)
		{
			for(int i=0;i<M;i++) tmp[i]=0;
			for(int i=0;i<M;i++) for(int j=0;j<M;j++)
				tmp[i]+=a.g[i][j]*res[j];
			for(int i=0;i<M;i++) res[i]=tmp[i];
		}
		mul(a,M);
	}
	printf("%.10f\n",res[M-2]);
}

void init(int &m,int &n,double &x)
{
	cin>>m>>n>>x;
}

int main()
{
	int test;

	for(cin>>test;test;test--)
	{
		int m,n;
		double x;

		init(m,n,x);
		build(m,x);
		calc(n,2*m);
	}
	return 0;
}
