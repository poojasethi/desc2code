#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#define fortodo(i,a,b) for(int i=(a);i<=(b);i++)
using namespace std;
const int maxn=61;int n,m;double X;
struct matrix{
	double A[maxn][maxn];
	int n,m;
}ans,tmp;

inline matrix operator *(const matrix &A,const matrix &B){
	matrix ans;ans.n=A.n;ans.m=B.m;memset(ans.A,0,sizeof(ans.A));
	fortodo(i,1,A.n) fortodo(j,1,B.m) fortodo(k,1,A.m) 
		ans.A[i][j]+=A.A[i][k]*B.A[k][j];
	return ans;
}
inline matrix pow(matrix a,int k){
	matrix tmp;tmp.n=a.n;tmp.m=a.m;memset(tmp.A,0,sizeof(tmp.A));
	fortodo(i,1,tmp.n) tmp.A[i][i]=1;
	while(k){
		if (k&1) tmp=tmp*a;
		a=a*a;k>>=1;
	}
	return tmp;
}
int main(){
	int T;cin>>T;
	while(T--){
		scanf("%d%d%lf",&m,&n,&X);
		memset(ans.A,0,sizeof(ans.A));
		memset(tmp.A,0,sizeof(tmp.A));
		ans.n=1;ans.m=2*m;
		ans.A[1][1]=sin(X);ans.A[1][1+m]=cos(X);
		tmp.n=2*m;tmp.m=2*m;
		fortodo(i,1,m){
			tmp.A[i][i]=cos(X);tmp.A[i+m][i]=sin(X);if (i!=1) tmp.A[i-1][i]=sin(X);
		}
		fortodo(i,m+1,2*m){
			tmp.A[i][i]=cos(X);tmp.A[i-m][i]=-sin(X);if (i>m+1) tmp.A[i-m-1][i]=cos(X);
		}
		double tot=(ans*pow(tmp,n-1)).A[1][m];
		printf("%.10f\n",tot);
	}
}
