#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;

#define n	65
#define DB	double
#define For(i,a,b)	for(int i=a;i<=b;i++)

DB		X;
int		T,M,N,D;

struct	Mat{
		DB	A[n][n];

		void	clear()	{
			For(i,1,D)	For(j,1,D)	A[i][j]=0;
		}
		void	Init(){
			DB	Sx=sin(X),Cx=cos(X);	clear();
			For(i,1,M)	{
				A[i-1+1][i+1]=Sx;	A[i-1+1][i+M+2]=Cx;
				A[i+1][i+1]=Cx;		A[i+M+2][i+1]=Sx;
				A[i+1][i+M+2]=-Sx;	A[i+M+2][i+M+2]=Cx;
			}
		}

		Mat	operator *	(const Mat&a){
			Mat	c;	c.clear();
			For(i,1,D)	For(j,1,D)	For(k,1,D)
				c.A[i][j]+=A[i][k]*a.A[k][j];
			return	c;
		}
		Mat	operator ^	(int N){
			Mat	c,a=*this;	c.clear();
			For(i,1,D)	c.A[i][i]=1;

			for	(;N;N>>=1,a=a*a)
				if	(N&1)	c=c*a;
			return	c;
		}
}A;

int		main(){
		cin>>T;
		for	(;T--;)	{
			cin>>M>>N>>X;	D=(M+1)<<1;
			A.Init();	A=A^N;
			printf("%.12lf\n",A.A[1][M+1]+A.A[M+2][M+1]);
		}
}
