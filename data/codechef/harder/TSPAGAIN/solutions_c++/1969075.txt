	int T,i,j,q,k,m,n;

	 for(q=0;q<k;q++)

	  scanf("%d%d",&i,&j);
	  d[i][j]++;
	 }
	 for(j=0;j<n;j++)
	 {
	  int prev=0;
	  for(i=n-1;i>=0;i--)
	  {
	   d[i][j]+=prev;
	   prev=d[i][j];
	  }
	 }
	 for(i=0;i<n;i++)
	 {
	  int prev=0;
	  for(j=0;j<n;j++)
	  {

	   prev=d[i][j];
	  }
	 }

	 to[0]=come[n-1]=0;
	 for(i=1;i<n;i++)
	 to[i]=to[i-1]+d[i-1][i];
	 for(i=n-2;i>=0;i--)
	 come[i]=come[i+1]+d[i+1][i];
	 f1[1]=d[0][1];f0[1]=d[1][0];
	 for(i=2;i<n;i++)

	  int x1=inf,x2=inf;

	   x1=min(x1,f1[j]+to[i-1]-to[j]+d[i][j-1]);
	   x2=min(x2,f0[j]+come[j]-come[i-1]+d[j-1][i]);
	  }
	  f0[i]=x1;f1[i]=x2;
	 }
	 printf("%d\n",min(f0[n-1]+d[n-2][n-1],f1[n-1]+d[n-1][n-2]));
	}

}
