//int d[][2]={{-1.0},{1,0},{0,-1},{0,1}};
int n;
int k;
int cost[1001][1001];
int dp[1001][1001];
void calcCost(){
	int a, b, c, i, j;
	for(i=0; i<n; i++){
		for(j=n-1; j>=0; j--){
			if(i-1>=0){a=cost[i-1][j];}
			else{a=0;}
			if(j+1<n){b=cost[i][j+1];}
			else{b=0;}
			if(i-1>=0 && j+1<n){c=cost[i-1][j+1];}
			else{c=0;}
			cost[i][j]+=a+b-c;
		}
	}
}
 
int dist( int i, int j ){
	if( i==j) return 0;
	return cost[j][i];
}
int solve(){
	int i, j, index;
	for(i=n-1; i>=0; i--){
		for(j=n-1; j>=0; j--){
			dp[i][j]=1000000000;
			index=max(i, j)+1;
			if(index>=n){
				dp[i][j]=dist(i, n-1)+dist(n-1, j);
			}
			else{
				dp[i][j]=min(dist(i, index)+dp[index][j], dist(index, j)+dp[i][index]);
			}
		}
	}
}
int main(){
	int x,y;
	int t; S("%d", &t );
	while( t-- ){
		for(int i=0; i<n; i++){
			for(int j=0; j<n; j++){
				cost[i][j]=0;
			}
		}
		S("%d%d", &n, &k );
		for(int i=0; i<k; i++){
			S("%d%d", &x, &y );
			cost[y][x] += 1;
		}
		calcCost();
		P("%d\n", solve() );
	}
	return 0;
}
