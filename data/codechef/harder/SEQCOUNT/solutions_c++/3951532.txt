using namespace std;

const int MOD = 1000000007;

int N, K, D, dp[100005], next[100005];
int main(){
    scanf("%d %d %d", &N, &K, &D);
    
    for(int i = 1; i <= K; i++)
        N -= i;
    if(N < 0){
        printf("0\n");
        return 0;
    }
    
    dp[0] = 1;
    for(int i = 0; i < K; i++){
        memset(next, 0, sizeof next);
        
        int v = K - i;
        for(int m = 0; m < v; m++){
            int c = 0;
            for(int d = 0; v * d <= N; d++){
                c = (c + dp[v * d + m]) % MOD;
                if(v * d - v * D + m >= 0)c = (c - dp[v * d - v * D + m] + MOD) % MOD;
                
                next[v * d + m] = c;
            }
        }
        
        for(int j = 0; j <= N; j++)
            dp[j] = next[j];
    }
    printf("%d\n", dp[N]);
    
    return 0;
}
