#include <iostream>
#include <cstdio>
using namespace std;
int n,k,d,dp[610][100010],mod=1000000007;
int main()
{
    //freopen("r.txt","r",stdin);
    scanf("%d %d %d",&n,&k,&d);
    if(n<(k*(k+1))/2)
    {
        printf("0");
        return 0;
    }
    dp[0][0]=1;
    for(int length=1;length<=k;length++)
    {
        for(int sum=1;sum<=n;sum++)
        {
            if(sum>=length)
            {
                dp[length][sum] = dp[length][sum-length]%mod ;
                dp[length][sum] += dp[length-1][sum-length]%mod ;///elwa7da elna2sa
            }
            dp[length][sum]%=mod;
            if( sum>=length*(d+1) )
            {
                dp[length][sum] -= dp[length-1][sum-(length*(d+1))]%mod ;///elwa7da elzeyada
            }
            dp[length][sum]+=mod , dp[length][sum]%=mod;
        }
    }
    printf("%d",dp[k][n]%mod);

    return 0;
}
