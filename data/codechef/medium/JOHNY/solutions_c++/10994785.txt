#include<bits/stdc++.h>
using namespace std ;

int main()
{  
int tc ;
cin>>tc ;
while(tc--)
{
	int n,k  ;
	cin>>n ;
	int arr[n] ;
	for(int i=0;i<n;i++)
	{
		cin>>arr[i]; 
	}
	cin>> k ;
	int l=arr[k-1]  ;
	sort(arr,arr+n) ;
	int ans ;
	for(int i=0;i<n;i++)
	{ if(arr[i]==l)
	    {
		ans=i+1 ;
		break ;
	}
	}
	cout<<ans<<endl; 
}
  
}
