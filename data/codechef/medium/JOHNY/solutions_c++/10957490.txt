#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
	int t,n,k;
	cin>>t;
	for(int i=0;i<t;i++){
		cin>>n;
		vector<long int> v;
		long int h;
		for(int j=0;j<n;j++){
			cin>>h;
			v.push_back(h);
		}	
		cin>>k;
		long int hh = v[k-1];
		sort(v.begin(),v.end());
		std::vector<long int>:: iterator u;
		u = find(v.begin(),v.end(),hh);
		cout<<u-v.begin()+1<<"\n";
	}
	return 0;
}
