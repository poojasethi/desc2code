#include <iostream>
#include <iomanip>
using namespace std;

int main() {
	// your code goes here
	int p[101],t,s,k,pos,fav,i;
	cin >> t;
	while(t--) {
		pos=1;
		cin >>s;
		for(i =0;i<s;i++) {
			cin >> p[i];
		}
		cin >> k;
		fav = p[k-1];
		for (i =0; i< s; i++) {
			if(p[i] < fav)
				pos++;
		}
		cout << pos << endl;
	}
	return 0;
}
