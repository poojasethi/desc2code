#include<stdio.h>
#include<stdlib.h>
#define ll long long int

int cmpfunc(const void *a, const void *b){
	return(*(ll *)a - *(ll *)b);
}

int main(){
	ll t, n, k; scanf("%lld", &t);
	while(t-- && scanf("%lld", &n)){
		ll a[n], i, d;
		for(i = 0; i < n; scanf("%lld", &a[i++]));
		scanf("%lld", &k), d = a[k - 1], qsort(a, n, sizeof(ll), cmpfunc);
		for(i = 0; i < n; i++)	if(a[i] == d){printf("%lld\n", i + 1); break;}
	}
	return 0;
}
