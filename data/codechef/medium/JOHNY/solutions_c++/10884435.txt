#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n;
		scanf("%d",&n);
		vector<pair<int,int> > a(n);
		for (int i = 0; i < n; i++) {
 			scanf("%d",&a[i].first);
			a[i].second = i+1;	
		}
		sort(a.begin(),a.end());
		int k;
		scanf("%d",&k);
		for (int i = 0; i < n; i++)
		{
			if (a[i].second == k) {
 				printf("%d\n",i+1);
				break;
 			}
		}
	}
	return 0;
}
