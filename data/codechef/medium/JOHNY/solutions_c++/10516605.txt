#include<cstdio>
using namespace std;
int main()
{
	int t,n,i,k,count;
	long int a[100];
	scanf("%d",&t); //no of test cases
	while(t--)
	{ 
	  count=0;
	  scanf("%d",&n); //no of songs in playlist
	  for(i=0;i<n;i++)
	  {
	    scanf("%d ",&a[i]);  //length of each song
      }
    
      scanf("%d",&k);  //position of uncle johnu in playlist
      k-=1;
      
      for(i=0;i<n;i++)
	  {  
	    if(a[i]<a[k])
	    {
	    count++;
	    }
      }

      printf("%d\n",count+1);

	}
	return 0;
}
