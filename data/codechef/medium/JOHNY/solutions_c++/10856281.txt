#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        /*unsigned long long*/ int n,*a,i,k;
        scanf("%d",&n);
        a=(int*)malloc(sizeof(int)*n);
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        scanf("%d",&k);
        int res=a[k-1];
        sort(a,a+n);
        for(i=0;i<n;i++)
        {
            if(a[i]==res)
            {
                printf("%d\n",i+1);
                break;
            }
        }
    }
    return 0;
}
