#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,k;
		cin>>n;
		long int a[n+1],l;
		for(i=1;i<n+1;i++)
			cin>>a[i];
		cin>>k;
		l=a[k];
		sort(a+1,a+n+1);
		for(i=1;i<n+1;i++)
		{
			if(a[i]==l)
			{
				cout<<i<<endl;
				break;
			}
		}
	}
	return 0;
}
