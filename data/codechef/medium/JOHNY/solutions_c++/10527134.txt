#include<stdio.h>
#include<limits.h>
#include<algorithm>
#define p(n) printf("%d\n",n);
#define s(n) scanf("%d",&n);
#define S(n) scanf("%li",&n);
using namespace std;
int main()
{
    int t;
    s(t)
    while(t--)
    {
        int n,k;
        s(n)
        long int a[n];
        long int q;
        for(int i=0;i<n;i++)
            S(a[i])
            s(k)
            q=a[k-1];
            sort(a,a+n);
            int low=0;
            int temp;
            int high=n-1;
            while(low<=high)
            {
              int mid=(low+high)/2;
              if(q==a[mid])
              {
                  temp=mid+1;
                  break;
              }
              else if(q>a[mid])
              {
                  low=mid+1;
              }
              else
                high=mid-1;
            }
            p(temp)
    }
    return 0;
}
