#include <iostream>
#include <bits/stdc++.h>
#include <limits.h>
typedef long long int ll;
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		vector<ll>v(n);
		for(int i=0;i<n;i++)
		{
			ll a;
			cin>>a;
			v[i]=a;
		}
		int k;
		cin>>k;
		ll t= v[k-1];
		sort(v.begin(),v.end());
		for(int i=0;i<n;i++)
		{
			if(v[i]==t)
			{
				cout<<i+1<<endl;
				break;
			}
		}
	}
	return 0;
}
