#include<iostream>
#include<algorithm>
using namespace std;

int main(){
	int m,testCase,n,k;
	cin>>testCase;
	
	for(int c=0;c<testCase;c++){
		cin>>n;
		int a[n],b[n];
	for(int i=0;i<n;i++){
		cin>>a[i];
		b[i]=a[i];
	}
	cin>>k;
	sort(a,a+n);
	m=b[k-1];
	
	for(int i=0;i<=n;i++){
		if(m==a[i]){
			cout<<i+1<<endl;	
		}
	}
 }
}
