#include <iostream>
using namespace std;

int main() {
	int T;
	cin>>T;
	int *songs;
	while(T>0)
	{
	 int n;
	 cin>>n;
	 songs = new int[n];
	 for(int i=0;i<n;i++)
	 {
	 	cin>>songs[i];
	 }
	 int k;
	 cin>>k;
	 int fav=songs[k-1];
	 int count=0;
	 for(int i=0;i<n;i++)
	 {  
	 	if(songs[i]<=fav)
	 	{
	 	count++;
	 	}
	 }
	 cout<<count<<endl;
	T--;
	
	}
	
	return 0;
}
