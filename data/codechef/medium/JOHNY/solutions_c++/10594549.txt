#include<iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main()
{
    int T;
    cin>>T;
    int N;
    int *A;
    int K;
    int result;
    for(int i=0;i<T;i++)
    {

        cin>>N;
        A=new int[N];

        for(int j=0;j<N;j++)
            cin>>A[j];
        cin>>K;
        result=1;
        for(int j=0;j<N;j++)
        {
            if(A[j]<A[K-1])
                result++;
        }
        cout<<result<<"\n";
    }
    return 0;
}
