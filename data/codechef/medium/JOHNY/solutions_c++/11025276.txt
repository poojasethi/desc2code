#include<stdio.h>
    int main()
    {
        int t,n,k,i;
        scanf("%d",&t);
        while(t--)
        {
            scanf("%d",&n);
            int a[n+1];
            for(i = 1;i <= n;i++)
                scanf("%d",&a[i]);
            scanf("%d",&k);
            int s = a[k];
            int c = 0;
            for(i = 1;i <= n;i++)
            {
                if(a[i] < s)
                    c++;
            }
            printf("%d\n",c+1);
        }
        return 0;
    }
