#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
 {
  vector<long long int>v;
  long long int t,n,k,i,kn,x;
  cin>>t;
  while(t!=0)
   {
    cin>>n;
    v.clear();
    for(i=0;i<n;i++)
     {
      cin>>x;
      v.push_back(x);
     }
     cin>>k;
     long int kk;
     kk=v[k-1];
     sort(v.begin(),v.end());
    for(i=0;i<v.size();i++)
    {
        if(v[i]==kk)
        {
            kn=i+1;
            break;
        }
    }
    cout<<kn<<"\n";
    t--;
   }
 }
