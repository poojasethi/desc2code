#include<iostream>
using namespace std;
int main()
{
   int t,n,l,k,a[100000],i=0,o,j,m;
   cin>>t;
   while(t--)
   {
     cin>>n;
     for(i=0;i<n;i++)
     cin>>a[i];
     cin>>k;
     l=a[k-1];
      for(j=n-1;j>=0;j--)   
       for(o=0;o<j;o++)
         if(a[o]>a[o+1])
         { m=a[o+1]; a[o+1]=a[o]; a[o]=m;}
      for(int z=0;z<n;z++)
      if(l==a[z])
      cout<<z+1<<"\n";
   }      
     return 0;
  }
