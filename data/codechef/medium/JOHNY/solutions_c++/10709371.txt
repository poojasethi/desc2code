#include<stdio.h>
int main()
{
	int t,n,i,j,k,a,m;
	int desvalue,desposition;
	scanf("%d",&t);
	for( i = 0; i < t; i++)
	{
		scanf("%d",&n);
		int arr[n];
		printf("\n");
		for( j = 0; j < n; j++)
		{
			scanf("%d",&a);
			arr[j] = a;	
		}
		printf("\n");
		scanf("%d",&k);
		desvalue = arr[k-1];
		
		desposition = 1;
		for( m = 0; m < n; m++)
		{
			if(arr[m] < desvalue)
			{
				desposition++;
			}
		}
		printf("\n%d",desposition);
		printf("\n");
	}
	return 0;
}
