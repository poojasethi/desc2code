#include <iostream>
#include <stdlib.h>
#include <algorithm>
#include <string.h>
using namespace std;


int main ()
{
	int t;
	short int k, n;
	
	cin>>t;
	while(t--)
	{
		cin>>n;
	    long *arr = (long *)malloc (sizeof (long) * n);
		for (int i = 0; i < n; i++)
		{
			cin>>arr[i];
		}
		cin>>k;
		int pos = arr[k-1];
		
		sort (arr, arr+n);
		int i;
		for ( i = 0; i < n; i++)
		{
			if (arr[i] == pos)
			  break;
		}
		cout<<i+1<<"\n";
	}
}
