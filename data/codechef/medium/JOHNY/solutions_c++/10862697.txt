#include<cstdio>
#include<algorithm>
using namespace std;
int search_index(int x,int arr[],int N){
    int l=0,r=N-1;
    while(l<=r){
        int mid=(l+r)/2;
        if(arr[mid]>x){
            r=mid-1;
        }
        else if(arr[mid]==x)
        {
            return mid+1;
        }
        else{
            l=mid+1;
        }
    }
    return -1;
}
int main(){
    
    int T;
    scanf("%d",&T);
    while(T--){
    int N,K;
    scanf("%d",&N);
    int arr[N];
    for(int i=0;i<N;i++){
        scanf("%d",&arr[i]);
    }
    scanf("%d",&K);
    int x=arr[K-1];
    sort(arr,arr+N);
    printf("%d\n",search_index(x,arr,N));
    }
}
