#include<iostream>
using namespace std;
 
int main()
{
   int t,n,p,i,k,count;
   long long int a[5002];
   cin>>t;
   while(t--)
    {  count=1;
       cin>>n;
       for(i=0;i<n;i++)
        cin>>a[i];
        cin>>k;
        p=a[k-1];
        for(i=0;i<n;i++)
        { if(a[i]<p)
          count++;
        }
        cout<<count<<endl;
   }
 
    return 0;
}
