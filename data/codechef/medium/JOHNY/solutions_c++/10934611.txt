#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
int t,n,k;
scanf("%d",&t);
while(t--)
{
scanf("%d",&n);
int i,a[n],flag,pos=1;
for(i=1;i<=n;i++)
scanf("%d",&a[i]);
scanf("%d",&k);
flag=a[k];

for(i=1;i<=n;i++)
{

    if(a[i]<flag)
        pos++;
}
printf("%d\n",pos);
}
return 0;
}
