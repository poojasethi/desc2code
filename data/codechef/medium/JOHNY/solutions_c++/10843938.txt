#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%i",&t);
    while(t--)
    {
        int n,a[100],k,x,c=1;
        scanf("%i",&n);
        for(int i=0;i<n;i++)
            scanf("%i",&a[i]);
        scanf("%i",&k);
        x=a[k-1];
        for(int i=0;i<n;i++)
        {
            if(a[i]<x)
                c++;
        }
        printf("%i\n",c);
    }
	return 0;
}
