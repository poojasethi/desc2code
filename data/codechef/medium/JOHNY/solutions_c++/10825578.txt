#include <iostream>
#include<algorithm>
using namespace std;


int main() {
	int t,n,k,i;
	long long int *songs,johny;
	cin>>t;
	while(t--)
	{
		cin>>n;
		songs=new long long int[n];
		for(i=0;i<n;i++)
		cin>>songs[i];
		cin>>k;
		johny=songs[k-1];
		sort(songs,songs+n);
		int i=0;
		while(i<n)
		{
			if(songs[i]==johny)
			{
				cout<<i+1<<endl;
				break;
			}
			i++;
		}	
	}
	return 0;
}
