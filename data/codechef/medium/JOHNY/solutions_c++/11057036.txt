#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        int a[n+1],pos,i;
        for(i=1;i<n+1;i++)
            cin >> a[i];
        cin >> pos;
        int temp= a[pos];
        sort(a+1,a+n+1);
        for(i=1;i<n+1;i++)
        {
            if(a[i]==temp)
                break;
        }
        cout << i << endl;
    }
    return 0;
}
