#include<iostream>
using namespace std;
 
void bubbleSort(int a[100], int n){
    for(int i=0;i<n;i++){
        for(int j=0;j<n-1;j++){
             if(a[j]>a[j+1]){
                 int t=a[j];
                 a[j]=a[j+1];
                 a[j+1]=t;} } }   }  
 
 
int main(){
      int numbers;
      cin>>numbers;
      int s=0;
      while(s<numbers){
          int n;
          cin>>n;
          int a[n];
          for(int i=0;i<n;i++){
               cin>>a[i];}
          int k;
          cin>>k;
          int fav=a[k-1];
          bubbleSort(a,n);
          int index;
          for(int j=0;j<n;j++){
              if(a[j]==fav){
                  index=j+1;
                  break;}}  
          cout<<index<<endl;
          s=s+1;}  }
