
#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <stdbool.h>

using namespace std;

void swap(int *a,int *b)
{
    int temp = *a;*a = *b;*b = temp;
}
int part(int*arr,int l,int r)
{
    int pindex = l;
    int pivot = *(arr+r);
    int i;
    for(i=l;i<r;i++)
    {
        if(pivot>*(arr+i))
        {
            swap(arr+i,arr+pindex);
            pindex++;
        }
    }
    swap(arr+pindex,arr+r);
    return pindex;
}

void quicksort(int*arr,int l,int r)
{
    if(l>=r)
        return;
    int p = part(arr,l,r);
    quicksort(arr,l,p-1);
    quicksort(arr,p+1,r);
}
int main()
{
	int n,k,t,i;
	int arr[101];//(int*)malloc(sizeof(int));
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",arr+i);
		}
		scanf("%d",&k);
		int num = *(arr+k-1);
		quicksort(arr,0,n-1);
		for (int i = 0; i < n; ++i)
		{
			if(num == *(arr+i)){
				printf("%d\n",i+1);
				break;
			}

		}
		
	
	}
	return 0;
}
