#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k,i;
        scanf("%d",&n);
        long int x;
        vector<long int> a;
        for(i=1;i<=n;i++)
            {
               scanf("%ld",&x);
                a.push_back(x);
            }

        scanf("%d",&k);
        x=a[k-1];
        sort(a.begin(),a.end());

        for(i=0;i<n;i++)
            {
               if(a[i]==x)
               {
                 printf("%ld\n",i+1);
                 break;
               }

            }
    }
    return 0;
}
