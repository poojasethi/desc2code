t = input()
while t:
    no_songs = input()
    for i in range(0,no_songs):
        list = map(int,raw_input().split())
        pos = input()
        pos-=1
        num = list[pos]
        list.sort()
        for j in range(0,len(list)):
            if(list[j]==num):
                print (j+1)
                break
        break
    t-=1
