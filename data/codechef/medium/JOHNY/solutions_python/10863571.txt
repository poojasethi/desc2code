def main():
    t = int(input())
    while t > 0:
        t -= 1
        input()
        numbers = map(int, raw_input().split())
        position = int(input()) - 1
        number = numbers[position]
        numbers.sort()
        index = numbers.index(number) + 1
        print index


if __name__ == "__main__":
    main()
