t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    a = [int(x) for x in raw_input().split()]
    K = int(raw_input())
    query = a[K - 1]
    for j in range(n):
        for k in range(n - j - 1):
            if a[k] > a[k + 1]:
                tmp = a[k]
                a[k] = a[k + 1]
                a[k + 1] = tmp
    for j in range(n):
        if a[j] == query:
            print j + 1
            break
