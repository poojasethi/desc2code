#include<stdio.h>
#include<algorithm>
#include<math.h>
using namespace std;

int main()
{
    long int t,i,j,h,m;
    double a,cala;
    scanf("%ld",&t);
    while(t--)
    {
        scanf("%lf",&a);
        for(i=0;i<12;i++)
        {
            for(j=0;j<60;j++)
            {
                cala=fabs(0.5*(60*i-11*j));
                if(cala>180)
                    cala=360-cala;
                if(cala==a||fabs(cala-a)<=0.00833333)
                {
                    if(i<10&&j<10)printf("0%d:0%d\n",i,j);
                    else if(i<10&&j>=10)printf("0%d:%d\n",i,j);
                    else if(i>=10&&j<10)printf("%d:0%d\n",i,j);
                    else printf("%d:%d\n",i,j);
                }
            }
        }
    }
    return 0;
}
