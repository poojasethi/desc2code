#include <cstdio>
#include <cmath>
double times[2459];

void init_times() {
    double init_angle_hour = 0;
    for (int i = 0; i < 12; i++) {
        double init_angle_minute = 0;
        for (int j = 0; j < 60; j++) {
            times[i * 100 + j] = fabs(init_angle_minute - init_angle_hour);
            if (times[i * 100 + j] > 180)
                times[i * 100 + j] = 360 - times[i * 100 + j];
            init_angle_minute += 6;
            init_angle_hour += 0.5;
        }
    }

}

int main() {
    int t;
    scanf("%d", &t);
    init_times();
    while (t--) {
        double a;
        double b = 1.0 / 120.0;
        scanf("%lf", &a);
        for (int i = 0; i < 12; i++) {
            for (int j = 0; j < 60; j++) {
                if (fabs(a - times[i * 100 + j]) < b)
                    printf("%02d:%02d\n", i, j);
            }
        }
    }
    return 0;
}
