#include<iostream>
#include<iomanip>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        double a,temp,lb,ub;
        cin>>a;
        lb=a-0.0083;
        ub=a+0.0083;
        //cout<<"ub  lb"<<ub<<" "<<lb<<"\n";
        int hh,mm;
        double angle=0;
        if((a>=-0.0083) && (a<=0.0083))
        {
            hh=mm=0;
            cout<<setfill('0')<<setw(2)<<hh<<":"<<setfill('0')<<setw(2)<<mm<<"\n";
        }

        for(int i=1;i<720;i++)
        {
            angle+=5.5;

            if(angle>=360)
                angle=angle-360;

            if(angle>180)
            {
                temp=360-angle;
                if((temp>=lb) && (temp<=ub))
                {
                    hh=i/60;
                    mm=i%60;
                    cout<<setfill('0')<<setw(2)<<hh<<":"<<setfill('0')<<setw(2)<<mm<<"\n";
                }
            }
            else
            {
                if((angle>=lb) && (angle<=ub))
                {
                    hh=i/60;
                    mm=i%60;
                    cout<<setfill('0')<<setw(2)<<hh<<":"<<setfill('0')<<setw(2)<<mm<<"\n";
                }
            }
        }
    }
    return 0;
}
