#include<iostream>

#include<stdio.h>
#include<math.h>
#include<ctype.h>
using namespace std;

long long n,k,p,a,b,i;
vector< pair<long long, long long> > x;
vector<long long> max_dist;
int main()
{
    cin>>n>>k>>p;
    x.resize(n);
    max_dist.resize(n+1);
    for(i=0;i<n;i++)
    {
        cin>>x[i].first;
        x[i].second=i+1;
    }
    sort(x.begin(),x.end(),std::greater_equal< pair<long long, long long> >());
    max_dist[x[0].second]=x[0].first+k;
    for(i=1;i<n;i++)
    {
        if(x[i-1].first-x[i].first<=k)
            max_dist[x[i].second]=max_dist[x[i-1].second];
        else
            max_dist[x[i].second]=x[i].first+k;
    }
    for(i=0;i<p;i++)
    {
        cin>>a>>b;
        if(max_dist[a]==max_dist[b])
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
}
