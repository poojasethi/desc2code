#include <bits/stdc++.h>
using namespace std;

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cout.tie(NULL);

	int n, k, p; cin >> n >> k >> p;
	vector<pair<int, int> > vec(n);
	map<int, int> fmap;
	map<int, int> mmap;
	for(int i=0; i<n; i++){
		cin >> vec[i].first; vec[i].second = i;
		fmap[i] = vec[i].first;
	}
	sort(vec.begin(), vec.end());
	vector<int> maxdist(n, 0);
	mmap[vec[n-1].second] = 0;
	for(int i = n-2; i>=0; i--){
		if(abs(vec[i].first-vec[i+1].first)<=k){
			maxdist[i] = (abs(vec[i].first-vec[i+1].first)+maxdist[i+1]);
		}
		else{
			maxdist[i] = 0;
		}
		mmap[vec[i].second] = maxdist[i];
	}
	/*
	for(auto it=mmap.begin(); it!=mmap.end(); ++it){
		cout << it->first << " " << it->second << endl;
	}
	*/

	int a, b, x, y;
	for(int i=0; i<p; i++){
		cin >> a >> b;
		x = fmap[a-1];
		y = fmap[b-1];
		if(x+mmap[a-1]==y+mmap[b-1]){
			cout << "Yes" << '\n';
		}else{
			cout << "No" << '\n';
		}
	}
}
