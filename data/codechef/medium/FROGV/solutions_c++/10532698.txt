#include<bits/stdc++.h>
using namespace std;
vector <pair<int,int> >frog;
int main()
{
int n, k, p,chef,i;
cin>>n>>k>>p;
for(i=1;i<=n;++i)
{
    cin>>chef;
    frog.push_back(make_pair(chef,i));
}
sort(frog.begin(),frog.end());
int sz=frog.size(),dist[n+1],expand=0;
dist[frog[0].second]=expand;
for(i=1;i<sz;++i)
{
    if(frog[i].first-frog[i-1].first<=k)
    {
        dist[frog[i].second]=expand;
    }
    else
        dist[frog[i].second]=++expand;
}
 while(p--)
  {int A,B;
   scanf("%d%d",&A,&B);
   if(dist[A]==dist[B])
    printf("Yes\n");
    else printf("No\n");
  }
	return 0;
}
