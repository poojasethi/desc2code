#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mp make_pair
#define pb push_back
vector <pair<ll,ll> > v;
ll prev[100005],pos[100005];

int main()
{
ll n,k,p,i,num,x,y,prevx,prevnum,temp;
cin>>n>>k>>p;

for(i=1;i<=n;i++)
{
    cin>>x;
    pos[i]=x;
    v.pb(mp(x,i));
}
sort(v.begin(),v.end());

x=v[0].first;
num=v[0].second;
prev[num]=num;

for(i=1;i<v.size();i++)
{
    x=v[i].first;
    num=v[i].second;
    prevx=v[i-1].first;
    prevnum=v[i-1].second;
    if(prevx>=x-k)
        prev[num]=prev[prevnum];
    else
        prev[num]=num;
}

for(i=0;i<p;i++)
{
    cin>>x>>y;
    if(pos[x]>pos[y]){
    temp=x;
    x=y;
    y=temp;
    }
    if(pos[prev[y]]>pos[x])
        cout<<"No\n";
    else
        cout<<"Yes\n";
}

return 0;
}
