#include<bits/stdc++.h>
#define N 110000
using namespace std;
vector<pair<int,int> >v;
int cn[N];
int main(){
//	freopen("in.txt","r",stdin);
	int n,k,p;
	cin>>n>>k>>p;
	for(int i=1; i<=n; i++){
		int x; cin>>x;
		v.push_back(make_pair(x,i));
	}
	sort(v.begin(),v.end());
	cn[v[0].second]=1;
	for(int i=1; i<v.size(); i++){
		if(v[i].first-v[i-1].first>k){
			cn[v[i].second]=cn[v[i-1].second]+1;
		}
		else cn[v[i].second]=cn[v[i-1].second];
	}
	while(p--){
		int x,y; cin>>x>>y;
		if(cn[x]==cn[y]) cout<<"Yes\n";
		else cout<<"No\n";
	}
}
