#include <bits/stdc++.h>
using namespace std;
#define MAX 100001
pair<int, int>  arr[MAX];
int disj[MAX];
int cnt = 0;
 
int main()
{
	int n, k, p, x, y;  scanf("%d%d%d", &n, &k, &p);
	for (int i = 0; i < n; i++)
	{
		scanf("%d", &arr[i].first); arr[i].second = i;
	}
	sort(arr, arr + n);
	disj[0] = cnt;
	for (int i = 1; i < n; i++)
	{
		if (arr[i].first - arr[i-1].first <= k)
			disj[arr[i].second] = cnt;
		else
			disj[arr[i].second] = ++cnt;
	}
	for (int i = 0; i < p; i++)
	{
		scanf("%d%d", &x, &y); x--; y--;
		if (disj[x] == disj[y])
			printf("Yes\n");
		else
			printf("No\n");
	}
	return (0);
 
}
