#include <iostream>
#include<bits/stdc++.h>
#define ll long long int
#define pb push_back
#define mp make_pair
#define f first
#define s second
using namespace std;
typedef pair<int,int> p;
vector< p > v;
vector< int > g[100100];
int papa [100100];
int vis[100100];
void dfs(int s,int par)
{
    vis[s]=1;
    papa[s] = par;
    for(int i=0;i<g[s].size();i++)
    {
        if(!vis[g[s][i]])
            dfs(g[s][i],par);
    }
}
using namespace std;
int arr[100100];
int main()
{int n,k,p,val,a,b;
cin>>n>>k>>p;
for(int i=0;i<n;i++)
{
    scanf("%d",&val);
    v.pb(mp(val,i+1));
}
v.pb(mp(v[n-1].f+1234567899,n));
sort(v.begin(),v.end());
for(int i=0;i<n;i++)
{
    if(v[i+1].f-v[i].f<=k)
     {
      g[v[i+1].s].pb(v[i].s);
      g[v[i].s].pb(v[i+1].s);
     }

}
for(int i=1;i<=n;i++)
{
    if(!vis[i])
        dfs(i,i);
}
for(int i=0;i<p;i++)
{
    scanf("%d%d",&a,&b);
    if(papa[a]==papa[b])
        puts("Yes");
    else
        puts("No");
}
    return 0;
}
