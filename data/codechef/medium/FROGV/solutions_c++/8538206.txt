#define FOR(i,a,b) for (int i=(a),_b=(b);i<_b;i++)
#define FORE(i,a,b) for (int i=(a),_b=(b);i<=_b;i++)
#define FORD(i,a,b) for (int i=(a),_b=(b);i>=_b;i--)
#define ri(n)({\
    int neg=0;\
    n=0;\
    char ch;\
    for(ch=getchar(); ch<'0' || ch>'9'; ch=getchar()) if (ch=='-') neg=1-neg;\
    n=ch-48;\
    for(ch=getchar(); ch>='0' && ch<='9'; ch=getchar()) n=(n<<3)+(n<<1)+ch-48;\
    if (neg) n=-n;\
})
 
typedef long long ll;
typedef pair<int,int> II;
const int INF=1000000000+7;
const double esp=1e-13;
const double pi=3.141592653589;
 
struct fa
{
    int gt,vt;
};
 
bool cmp(fa a, fa b)
{
    return (a.gt<b.gt || (a.gt==b.gt && a.vt<b.vt));
}
 
fa a[100000+10];
int q,n,k,s[100000+10],x,y,f[100000+10];
 
int main()
{
 //   freopen("FROGV.inp", "r", stdin);
 //   freopen("FROGV.out", "w", stdout);
    ri(n); ri(k); ri(q);
    FORE(i,1,n)
    {
        ri(a[i].gt);
        a[i].vt=i;
    }
    sort(a+1,a+n+1,cmp);
    FORE(i,1,n) s[a[i].vt]=i;
    memset(f,0,sizeof(f));
    FORE(i,2,n)
    if (a[i].gt-a[i-1].gt<=k) f[i]=f[i-1];
    else f[i]=f[i-1]+1;
    FORE(i,1,q)
    {
        ri(x); ri(y);
        x=s[x];
        y=s[y];
        if (f[y]-f[x]==0) cout << "Yes" << "\n"; else cout << "No" << "\n";
    }
}
