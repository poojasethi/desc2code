#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll dp[1000000],a[1000000];
map<ll,ll> m,rm,nrm;
int main()
{
    ll n,k,p,i,x,y,nx,ny;
    cin>>n>>k>>p;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        rm[a[i]]=i;
        m[i]=a[i];
    }
    sort(a+1,a+n+1);
    for(i=1;i<=n;i++)
    {
        nrm[a[i]]=i;
    }
    dp[n]=0;
    for(i=n-1;i>=1;i--)
    {
        if(abs(a[i+1]-a[i])<=k)
        {
            dp[i]=dp[i+1]+1;
        }
        else
        {
            dp[i]=0;
        }
    }
    for(i=1;i<=p;i++)
    {
        cin>>x>>y;
        nx=nrm[m[x]];
        ny=nrm[m[y]];
        if(ny<nx)
            swap(nx,ny);
        if(nx+dp[nx]>=ny)
        {
            cout<<"Yes\n";
        }
        else
        {
            cout<<"No\n";
        }
    }
    return 0;
}
