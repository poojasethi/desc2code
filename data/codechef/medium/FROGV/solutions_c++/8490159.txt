#include<iostream>
#include<cmath>
#include<algorithm>
#include<vector>
#include<cstdio>
#define ll long long 
#define sc(c) scanf("%lld",&c) 
#define pb push_back 
#define pf(c) printf("%lld",c) 
#define mp make_pair
using namespace std;
ll maxi[100001];
bool comp( pair <ll,ll> v1,pair <ll,ll> v2)
{
	if(v1.first >v2.first) return true;
	else return false;
}
int main()
{
	ll n,k,p,temp,a,b;
	vector <pair <ll,ll> > v;
	scanf("%lld%lld%lld",&n,&k,&p);
	
	for(ll i=0;i<n;i++)
	{
		sc(temp);
		v.pb(mp(temp,i+1));
	}
	sort(v.begin(),v.end(),comp);
	maxi[v[0].second]=v[0].first+k;
	for(int i=1;i<n;i++)
	{
		if(v[i-1].first-v[i].first <=k)
		maxi[v[i].second]=maxi[v[i-1].second];
		else
		{
			maxi[v[i].second]=v[i].first+k;
		}
	}
	while(p--)
	{
		scanf("%lld%lld",&a,&b);
		if(maxi[a]==maxi[b])
		printf("Yes\n");
		else
		printf("No\n");		
	}
	return 0;
}
