#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
bool comp(ll i,ll j)
{
    return j>i;
}
int main()
{
    ll n,k,p,i,u,v,x;
    cin>>n>>k>>p;
    vector<pair<ll,ll> > a(n);
    ll maxd[n];
    for(i=0;i<n;i++)
    {
        cin>>x;
        a[i].first=x;
        a[i].second=i;
    }
    sort(a.rbegin(),a.rend());
    maxd[a[0].second]=a[0].first+k;
    for(i=1;i<n;i++)
    {
        if(a[i-1].first-a[i].first<=k)
        {
            maxd[a[i].second]=maxd[a[i-1].second];
        }
        else
        {
            maxd[a[i].second]=a[i].first+k;
        }
    }
    while(p--)
    {
        cin>>u>>v;
        if(maxd[u-1]==maxd[v-1])
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
