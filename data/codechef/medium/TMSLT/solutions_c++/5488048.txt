#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000
long long int dp[3000009],q[3000009];
int main() 
{
	long long int i,j,k,l,n,m,t,a,b,c,d,vr,vt,vm,total,sum;
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld %lld %lld %lld %lld",&n,&a,&b,&c,&d);
		for(i=0;i<1000000;i++)
		{	
			q[i]=0;
		}
		dp[0]=d;
		q[dp[0]]=1;
		for(i=1;i<n;i++)
		{
			dp[i]=((a*dp[i-1]*dp[i-1])+(b*dp[i-1])+c)%MOD;
			if(q[dp[i]])
				q[dp[i]]=0;
			else
				q[dp[i]]=1;
		}
		total=0;
		for(i=1;i<1000000;i++)
		{
			if(q[i]>0)
			{
				if(total%2==0)
					sum+=i;
				else
					sum-=i;
				total++;
			}
		}
		printf("%lld\n",abs(sum));
	}
	return 0;
}
