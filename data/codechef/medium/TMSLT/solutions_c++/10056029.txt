#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL MOD = 1e6;
const int MAXN = 3e6 + 10;
LL A[MAXN];
LL B[MAXN];
int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        int N, a, b, c, d;
        scanf("%d%d%d%d%d", &N, &a, &b, &c, &d);
        LL aSum = 0;
        LL bSum = 0;
        LL s = d;
        for(int i = 0;i < MOD;++i){
            A[i] = 0;
        }
        for(int i = 1;i <= N;++i){
            ++A[s];
            s = (a * s * s + b * s + c) % MOD;
        }
        int _j = 0;
        for(int i = MOD - 1;i >= 0;--i){
            while(A[i] > 0){
                B[_j] = i;
                ++_j;
                --A[i];
            }
        }
        for(int i = 0;i < N;++i){
            if(i&1){
                aSum += B[i];
            } else {
                bSum += B[i];
            }
        }
        LL ans = aSum - bSum;
        if(ans < 0) ans = -ans;
        printf("%lld\n", ans);
    }
    return 0;
}
