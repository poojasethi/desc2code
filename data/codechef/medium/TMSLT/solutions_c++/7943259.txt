#include<bits/stdc++.h>
#define MOD 1000000
#define SIZE 1000005
#define lli long long int
int ount[SIZE];
using namespace std;
lli ABS(lli n)
{
	if(n>0)
		return n;
	else
		return -n;
}
int main()
{
	lli i,j,k,m,n,sum;
	lli t,a,b,c,d,flag,s;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld%lld%lld%lld",&n,&a,&b,&c,&d);
		for(i=0;i<MOD+5;i++)
                       ount[i]=0;

		ount[d]++;
		s=d;
		for(i=1;i<n;i++)
			ount[s=(a*s*s + b*s + c)%MOD]++;
		flag=0;
		sum=0;
		for(i=0;i<MOD+5;i++)
		{
			if(flag%2==0)
				sum+=(ount[i]%2)*i;

			else
				sum-=(ount[i]%2)*i;
			flag += ount[i];
		}
		printf("%lld\n",ABS(sum));
	}
}
