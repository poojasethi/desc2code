#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;

int br[1000000];
int ar[3000002];

int main()
{
    ios_base::sync_with_stdio(false);
    ll t,n,a,b,c,d,put,i,m=1000000;
    cin>>t;
    while(t--)
    {
        cin>>n>>a>>b>>c>>d;
        for(i=0;i<1000000;i++)
            br[i]=0;
        br[d]++;
        for(i=1;i<n;i++)
        {
            put = ((a*d+b)*d+c)%m;
            d = put;
            br[put]++;
        }
        put = 0;
        d = 0;
        for(i=999999;i>=0;i--)
        {
            while(br[i])
            {
                ar[put] = i;
                br[i]--;
                put++;
            }
        }
        for(i=1;i<n;i=i+2)
            d += ar[i-1]-ar[i];
        if(n%2!=0)
            d += ar[i-1];
        cout<<d<<endl;
    }
    return 0;
}
