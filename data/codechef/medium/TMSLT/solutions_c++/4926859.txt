#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
long long int s[3000005];
int p[1000001];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int sum1=0,sum2=0;
		int a,i,b,c,d,n,f=1;
		scanf("%d %d %d %d %d",&n,&a,&b,&c,&d);
		s[0]=d;
		p[s[0]]++;
		for(i=1;i<n;i++)
	    {
		   s[i]=(a*s[i-1]*s[i-1]+b*s[i-1]+c)%1000000;
		   p[s[i]]++;
	    }
	    for(i=1000000;i>=0;i--)
	    {
	    	
	    	while(p[i]!=0)
	    	{
	    		
	    		if(f==1)
	    		{sum1+=i;f=0;}
	    		else
	    		{sum2+=i;f=1;}
	    		p[i]--;
	    	}
	    }
	    printf("%lld\n",sum1-sum2);
	    for(i=0;i<=1000000;i++)
	    p[i]=0;
	}
	return 0;
}
