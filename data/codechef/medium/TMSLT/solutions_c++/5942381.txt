   for (int i = 1; i < n; ++i) {

      ++order[d];
      order[d] %= 2;
   }

   int answer = 0;
   bool ok = true;
   for (int i = 1000000; i > 0; --i)
      if (order[i]) {
         if (ok) answer -= i;
         else answer += i;
         ok = !ok;
      }
   printf("%d\n", abs(answer));
}
