    #include<iostream>
    #include<bits/stdc++.h>
    using namespace std;
    int arr[1000000]={0};
    int main()
    {
    long long int tc,n,a,b,c,d,count1=0,count2=0,temp,flag1=0,flag2=0;
    cin>>tc;
    for(int i=0;i<tc;i++)
    {
    	cin>>n>>a>>b>>c>>d;
    	temp=d;
    	
    	for(int i=0;i<n;i++)
    	{
    		
    		arr[temp]++;
    		temp=((a*temp*temp)+(b*temp)+c)%1000000;
    		
    	}
    	for(int i=1;i<1000000;i++)
    	{
    		if(arr[i]>0&&flag1==0)
    		{	count1+=i;
    			arr[i]--;
    			flag2=0;
    			flag1=1;
    		}
    		if(arr[i]>0&&flag2==0)
    		{
    			count2+=i;
    			arr[i]--;
    			flag1=0;
    			flag2=1;
    		}
    		if(arr[i]>0)
    		i--;
    	}
    	
    	cout<<abs(count2-count1);
    	cout<<"\n";
    	count2=0;
    	count1=0;
    	memset(arr,0,sizeof arr);
    }
    return 0;
    }
