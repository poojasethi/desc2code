#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;
#define ll long long
#define mod 1000000
int t, n, a, b, c, d;
int W[mod];
int main() {
	scanf("%d",&t);
	while(t--)
	{
	    scanf("%d %d %d %d %d", &n, &a, &b, &c, &d);
	    memset(W, 0, sizeof(W));
	    ll prev=d,ans=0;
	    int count=0;
	    W[prev]++;
	    for (int i=2; i<=n; ++i) {
	        prev = (prev * (prev * a + b) + c) % mod;
	        W[prev]++;
	    }
		for(int i=0;i<=mod;i++)
		{
			while(W[i]--)
	    	{
  				count++;
  				if(count%2==0)
	    		ans-=i;
	    		else
	    		ans+=i;
  			}
  			if(count==n)
  			break;
  		}
  	   	printf("%lld\n",abs(ans));
	}
    return 0;
}
