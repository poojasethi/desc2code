#include<iostream>
#include<math.h>                      
#include<stdlib.h>

#include<cstring>
#include<algorithm>
using namespace std; 



int main()
{    long long a,b,c,d,s1,s;   int t,n,now,i;
     	 
 scanf("%d",&t);
 while(t--)
 { s1=0; 
   scanf("%d%lld%lld%lld%lld",&n,&a,&b,&c,&d);   
   int *count=new int[1000006];
   memset(count,0,sizeof(count));
   
       count[d]++;   s=d;
   for(i=1;i<n;i++)
      count[s=( (a*s*s)+(b*s)+c )%1000000 ]++;
   
      now=0;
      for(i=1;i<1000000;i++)
      {
		if(now%2==0)     s1+= (count[i]%2)*i;      
		  else         s1-= (count[i]%2)*i;   
		  now+=count[i];
	  }		  
  printf("%lld\n",abs(s1)); 
 }
 return 0;
}
