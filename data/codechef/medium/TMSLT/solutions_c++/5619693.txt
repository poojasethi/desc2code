#include<cstdio>
#include<algorithm>
#define M 1000000
using namespace std;
int z[3000000],y[1000000];
int main()
{
	long long int t,n,a,b,c,d,i,s;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld%lld%lld%lld",&n,&a,&b,&c,&d);
		for(i=0;i<1000000;i++)
		y[i]=0;
		s=0;
		y[d]++;
		s=d;
		for(i=1;i<n;i++)
		{
			y[s = (a * s * s + b * s + c) % 1000000]++;
		}
		s=0;
		int now=0;
		for(i=0;i<1000000;i++)
		{
			if(now%2==0)
			s+=(y[i]%2)*i;
			else
			s-=(y[i]%2)*i;
			now+=y[i];
		}
		printf("%lld\n",abs(s));
		
	}
	return 0;
}
