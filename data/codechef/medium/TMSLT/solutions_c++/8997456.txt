#define MOD 1000000

bool arr[1000005];
int main(){
	int t;
	scan(t);
	while(t--){
		memset(arr,false,sizeof(arr));
		ll i,n,a,b,c,d;
		scanll(n);
		scanll(a);
		scanll(b);
		scanll(c);
		scanll(d);
		arr[d] = true;
		ll s=d;
		rep(i,1,n){
			s = ((a*s*s)+(b*s)+c)%MOD;
			arr[s] = !arr[s];
		}
		ll odd=0,even=0;
		ll ctr=0;
		repr(i,999999,0){
			if(arr[i]){
				ctr++;
				if(ctr%2==0){
					odd+=i;
				}
				else{
					even+=i;
				}
			}
		}
		printll(llabs(odd-even));
	}
	return 0;
}
