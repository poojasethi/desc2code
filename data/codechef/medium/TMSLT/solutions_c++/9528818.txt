#include<iostream>
#include <algorithm>
#include <cassert>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iomanip>
using namespace std;
int n,arr[1000001],a,b,c,d;
long long int ne,pre,now;
#define mod 1000000
int main()
{
    int t;
    cin>>t;
    while(t--){
    scanf("%d%d%d%d%d",&n,&a,&b,&c,&d);
    memset(arr,0,sizeof(arr));
    arr[d]+=1;
    now = d;
    long long int maxv = d;
    for(int i=1;i<n;i++)
    {
        ne = (a*now+b)*now+c;
        if(ne>=mod)
        ne %= mod;
        arr[ne]++;
        now = ne;
        if(ne>maxv)
        maxv=ne;
    }
    long long int ans = 0;
    int flag=1;
    for(int i=maxv;i>0;i--)
    {
        if(arr[i]%2==0)
        continue;
        if(flag)
        ans += i;
        else 
        ans -= i;
        flag = (flag+1)%2;
    }
    printf("%d\n",ans);
    }
}
