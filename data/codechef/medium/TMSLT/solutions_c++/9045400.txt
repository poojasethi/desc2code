#include<bits/stdc++.h>
using namespace std;
#define mod 1000000
#define ll long long
int main()
{
	int i,a,b,c,d,t,n,turn;
	ll temp,cal,mini,maxi,sum1,sum2;
	vector<int> cnum(1000001);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d%d%d",&n,&a,&b,&c,&d);
		cnum[d]++;
		temp=d;
		mini=d;
		maxi=d;
		for(i=1;i<n;i++)
		{
			cal=(a*temp*temp+b*temp+c)%mod;
			cnum[(int)cal]++;
			temp=cal;
			mini=min(mini,cal);
			maxi=max(maxi,cal);
		}
		sum1=sum2=0;
		turn=0;
		for(i=mini;i<=maxi;i++)
		{
				if((cnum[i]&1))
				{
					turn+=cnum[i];
					if(turn&1)
						sum1+=i;
					else
						sum2+=i;	
				}
				cnum[i]=0;	
		}
		if(sum1-sum2<0)
			printf("%lld\n",sum2-sum1);
		else
			printf("%lld\n",sum1-sum2);
	}
	return 0;
}
