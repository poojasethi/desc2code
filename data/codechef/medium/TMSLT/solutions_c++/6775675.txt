long long int ra()
{
    char c=gc;
    long long int r=0;
    while(c<'0' || c>'9')
        c=gc;
    while(c>='0' &&c<='9')
    {
        r=(r<<3)+(r<<1)+c-'0';
        c=gc;
    }
    return r;
}
void fastwrite(long long int a)
{
    char num[20];
    int i=0;
    do{
        num[i++]=a%10+'0';
        a /=10;
    }while(a!=0);
    i=i-1;
    while(i>=0)
    {
        putchar(num[i--]);
    }
    putchar('\n');

}
