import math

def nCr(n,r):
    f = math.factorial
    return f(n) / f(r) / f(n-r)



x,n=map(int,raw_input().strip().split(" "))
ans=0
for i in xrange(n):
	a=str(raw_input().strip())
	b=a[::-1]
	count=0
	#for (i,j) in[(i,j) for i in range(0,36,4) for j in range(0,18,2)]:
	i=0
	j=0
	while i<36 and j<18:
		cpt=a[i:i+4].count('0')+b[j:j+2].count('0')
		if cpt>=x:
		    count=count+nCr(cpt,x)
		i=i+4
		j=j+2
	ans=ans+count
print ans
