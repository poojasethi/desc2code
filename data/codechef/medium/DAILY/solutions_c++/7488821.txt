#include<stdio.h>
char arr[54];
int combinations(int n, int k)
{
    if(k>n)
        return 0;
    int ans = 1;
    int i,j;
    for(i=n;i>n-k;i--)
        ans*=i;
    for(j=k;j>0;j--)
        ans/=j;
    return ans;
}
int main(){
    int x, n, i, j, ans = 0;
    scanf("%d%d",&x,&n);
    for(i=0;i<n;i++)
    {
       
        for(j=0;j<54;j++)
            scanf(" %c",&arr[j]);

        int count[9] = {0};
        int in = 0;
        for(j=0;j<36;j=j+4)
        {
            int ctr = 0;
            if(arr[j] == '0')
                ctr++;
            if(arr[j+1]=='0')
                ctr++;
            if(arr[j+2]=='0')
                ctr++;
            if(arr[j+3]=='0')
                ctr++;
            count[in++] = ctr;
        }

        in = 0;
        for(j=53;j>=36;j=j-2)
        {
            int ctr = 0;
             if(arr[j] == '0')
                ctr++;
            if(arr[j-1]=='0')
                ctr++;
            count[in++]+= ctr;
        }

        for(j=0;j<9;j++){
            //printf("count: %d\n", count[j]);
            ans = ans + combinations(count[j], x);
        }
    }
    printf("%d\n", ans);
    return 0;
}
