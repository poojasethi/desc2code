#include<stdio.h>
#include<iostream>
using namespace std;
int fact(int n)
{
    int f=1;
    while(n>0)
    {
        f=f*n;
        n--;
    }
    return f;
}
int main()
{
  int x,n;
  cin>>x>>n;
  int ways=0;
  while(n--)
  {   char a[56];
      cin>>a;
      fflush(stdin);
      for(int i=1;i<=9;i++)
      {
          int count=0;
          int c=(i-1)*4;
            if(a[c]=='0')
                count++;
            if(a[++c]=='0')
                count++;
            if(a[++c]=='0')
                count++;
            if(a[++c]=='0')
                count++;
            c=54-((i-1)*2);
            if(a[--c]=='0')
                count++;
            if(a[--c]=='0')
                count++;
            if(count>=x)
                ways+= (fact(count)/(fact(x)*fact(count-x)));
      }
}
       cout<<ways<<endl;
     return 0;
}
