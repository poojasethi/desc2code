inline int C( int x, int y )
{
	if( x == 0 || y > x )
		return 0;
	if( x == y || y == 0 )
		return 1;
	return C( x-1, y ) + C( x-1, y-1 );
}

int main()
{
	int x, n, good, ans = 0;
	string s[16];
	int a[6] = { 1,2,3,4,53,54 };
	int d[6] = { 4,4,4,4,-2,-2 };
	cin >> x >> n;
	for( int i = 0 ; i < n ; i ++ )
	{
		cin >> s[i];
		s[i] = "#" + s[i];
	}
	for( int i = 0 ; i < n ; i ++ )
		for( int j = 0 ; j < 9 ; j ++ )
		{
			good = 6;
			for( int k = 0 ; k < 6 ; k ++ )
				good -= s[i][ a[k]+d[k]*j ] == '1';
			if( good >= x )
				ans += C( good, x );
		}
	cout << ans << endl;
	return 0;
}
