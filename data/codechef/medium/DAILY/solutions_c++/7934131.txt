#include<iostream>
using namespace std;
#include<cstdio>
#include<cstring>
int main(){
	int t,n,x,i;
	char s[56];
	scanf("%d %d%*cB",&x,&n);
	//cout<<x<<n<<endl;
	int fact[]={1,1,2,6,24,120,720};
	int a[9];
	long int count=0;
	while(n--){
		scanf("%s",&s);
		for(i=0;i<9;i++)
			a[i]=0;
		for(i=1;i<=54;i++){
			if(i<37){
				if(s[i-1]=='0')
					a[(i-1)/4]++;
			}
			else{
				if(s[i-1]=='0')
					a[8-((i-37)/2)]++;
			}
		}
		//cout<<s<<endl;
		for(i=0;i<9;i++){
			if(a[i]>=x)
				count+=fact[a[i]]/(fact[x]*fact[a[i]-x]);
				//cout<<"count is "<<count<<endl;
		}
		//cout<<count<<endl;
	}
	printf("%ld\n",count);

}
