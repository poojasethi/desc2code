# include <stdio.h>
int ncr(int b,int x)
{
    switch(x)
    {
        case 1: return b;
        case 2:
            switch(b)
          {
            case 3:
               return 3;
            case 4:
                return 6;
            case 5:
                return 10;
            case 6:
                 return 15;
          }
        case 3:
              switch(b)
              {case 4:
                return 4;
              case 5:
                 return 10;
              case 6:
                 return 20;}
        case 4:
              return ncr(b,b-4);
         case 5:
              return ncr(b,b-5);
    }
}
 
int main()
{
    int x,n,i,j,sum=0;
    char a[55];int b[10];
   scanf("%d%d",&x,&n);
    while(n--)
    {
        scanf("%s",a);
        for(i=0;i<9;i++)
         {
             b[i] = 0;
           for(j=4*i;j<4*(i+1);j++)
              if(a[j]== '0') b[i]++;
            if(a[53 - (2*i)]== '0') b[i]++;
            if(a[52 - (2*i)] == '0') b[i]++;
            if(b[i] == x)
                sum++;
            if(b[i] > x)
                    sum += ncr(b[i],x);
         }
    }
   // for(i=6;i>=6;i--)
     //   printf("%d ",ncr(i,5));
    printf("%d\n",sum);
    return 0;
}
