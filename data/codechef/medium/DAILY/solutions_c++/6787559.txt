#include <iostream>
#include <math.h>
#include <map>
using namespace std;
typedef long long LL;

map<LL,LL> ma;

LL ncr(LL n, LL r) {
    if(r > n / 2) r = n - r;
    LL ans = 1;
    LL i;
    
    for(i = 1; i <= r; i++) {
        ans *= n - r + i;
        ans /= i;
    }
    
    return ans;
}

int main() {
    LL i=0,r=0;
    for(i=1;i<=36;i++) {
        r = ceil(double((double)(i)/(double)(4)));
        ma[i] = r;
    }
    for(i=37;i<=54;i++) {
        r = (54 - i)/2;
        ma[i] = r+1;
    }

    LL x=0,n=0,j=0,sum=0;
    cin>>x>>n;
    string str;
    for(i=0;i<n;i++) {
        cin>>str;
        LL res[10] = {0};
        for(j=0;j<str.length();j++) {
            if(str[j]=='0') {
                res[ma[j+1]]++;
            }
        }
        
        for(j=1;j<=9;j++) {
            if(res[j]>=x)
                sum+=(ncr(res[j],x));
        }
    }
    
    cout<<sum<<endl;
    return 0;
}
