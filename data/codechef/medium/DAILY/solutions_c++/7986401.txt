#include <iostream>
#include <cstdio>
using namespace std;
int fact(int x)
 {
    if (x==0 || x==1)
     return 1;
   return (x*fact(x-1));
 }
int main()
{
   int x,n;
  int count,count1,i,j,k,l;
   long int ans;
  char s[100];
   scanf ("%d %d",&x,&n);
  ans=0;
    for (i=0;i<n;i++)
     {
        scanf ("%s",s);
        count1=0;
        for (j=0;j<9;j++)
         {
           k=j*4;
           count=0;
            for (l=0;l<4;l++)
             if (s[k+l]=='0')
              count++;
           k=26-j;
            if (s[k*2]=='0')
             count++;
            if (s[k*2+1]=='0')
              count++;
            if (count>=x)
           {
            count1+=(fact(count)/(fact(x)*fact(count-x)));
           }
         }
     ans+=count1;
     }
   printf ("%ld\n",ans);
 
  return 0;
}
