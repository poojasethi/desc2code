ll fact (ll n)
{
	if(n==1 || n==0) return  1;
	else return n*fact(n-1);       
}

int main()
{ 
  //ios_base::sync_with_stdio(false);
  //cin.tie(NULL);
    string str[15];
    ll x,n,ways=0;
    cin>>x>>n;
    for(int i=0;i<n;i++)
     cin>>str[i];
    for(int i=0;i<n;i++)
    {
	  for(int j=0,k=53;j<36,k>=36;j+=4,k-=2)
	  {
	  	int c=0;
	    if(str[i][j]=='0') c++;
	    if(str[i][j+1]=='0') c++;
	    if(str[i][j+2]=='0') c++;
	    if(str[i][j+3]=='0') c++;
	    if(str[i][k]=='0') c++;
	    if(str[i][k-1]=='0') c++;
	    
	   // cout<<c<<"\n";
	    
	    if(c>=x)
	     ways=ways+fact(c)/(fact(x)*fact(c-x));
	  }	
	}
	cout<<ways<<"\n";
   	return 0;
}
