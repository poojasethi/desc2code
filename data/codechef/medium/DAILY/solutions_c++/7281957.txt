#include <iostream>
#include <map>
#include <string>
using namespace std;

int C(int n, int r)
{
	int num = 1, den = 1;
	for(int i = 1; i <= r; i++) {
		num *= (n - i + 1);
		den *= i;
	}
	return num / den;
}

int main(int argc, char const *argv[])
{
	int x, n;
	char s[60];
	int ans = 0;

	cin >> x >> n;
	while(n--) {
		cin >> s;
		for(int i = 0; i < 9; i++) {
			int cnt = 0;
			if(s[4 * i] == '0') cnt++;
			if(s[4 * i + 1] == '0') cnt++;
			if(s[4 * i + 2] == '0') cnt++;
			if(s[4 * i + 3] == '0') cnt++;
			if(s[34 + 2 * (9 - i)] == '0') cnt++;
			if(s[34 + 2 * (9 - i) + 1] == '0') cnt++;
			ans += (cnt >= x) ? C(cnt, x): 0;
		}
	}
	cout << ans << endl;
	return 0;
}
