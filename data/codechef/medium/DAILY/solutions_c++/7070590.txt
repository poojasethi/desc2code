#include <iostream>
#include <vector>
#include <string>
using namespace std;
typedef long long ll;
typedef vector<ll> vll;
vll factsave(7,0);
ll fact(ll n)
{
	ll i,ans=1;
	if(factsave[n]!=0)
		return factsave[n];
	else
	{
		for(i=2;i<=n;i++)
		{
			ans=ans*i;
		}
		factsave[n]=ans;
		return ans;
	}
}
ll c(ll n,ll r)
{
	ll ans;
	if(r>n) return 0;
	ans=fact(n)/(fact(r)*fact(n-r));
	return ans;
}
int main()
{
	ll n,ans,i,temp,x;
	string status;
	ans=0;
	scanf("%lld",&x);
	scanf("%lld",&n);
	while(n--)
	{
		cin>>status;
		//cout<<status<<endl;
		vll compnos(9,0);
		for(i=0;i<=35;i++)
		{
			if(status[i]=='0')
			{
				compnos[i/4]++;
			}
		}
		//cout<<"here"<<endl;
		for(i=53;i>=36;i--)
		{
			temp=i-53;
			if(temp<0) temp=-temp;
			if(status[i]=='0')
			{
				compnos[temp/2]++;
			}
		}

		for(i=0;i<9;i++)
		{
			//cout<<compnos[i]<<" ";
			ans+=c(compnos[i],x);
		}
		//cout<<endl;
		compnos.clear();
	}
	printf("%lld",ans);
	return 0;
}
