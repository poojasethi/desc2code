#include<iostream>
#include<cstdio>
long long int comb(int count,int x)
{long long int prod=1,fact=1,i;
 for(i=count;i>(count-x);i--)
 prod=prod*i;
 for(i=1;i<=x;i++)
 fact=fact*i;
 return prod/fact;
} 
int main()
{
 
 long long int n,x,ans=0,i,j,count;
 char a[60];
 scanf("%lld",&x);
 scanf("%lld",&n);
 while(n--)
 {
  scanf("%s",a);

  
 for(i=0,j=53;i<=32;i+=4,j-=2)
 { count=0;
   if(a[i]=='0')
   count+=1;
   if(a[i+1]=='0')
   count+=1;
   if(a[i+2]=='0')
   count+=1;
   if(a[i+3]=='0')
   count+=1;
   if(a[j]=='0')
   count+=1;
   if(a[j-1]=='0')
   count+=1;
   if(count>=x)
   ans+=comb(count,x);
 }
}
  printf("%lld\n",ans);

return 0;
}
