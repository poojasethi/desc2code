int factorial(int n)
{
	int product = 1;
	for(; n > 1; n--)
	{
		product *= n;
	}
	return product;
}
int calculation(int n, int k)
{
	int calc;
	calc = factorial(n)/(factorial(k)*(factorial(n-k)));

	return calc;
}
