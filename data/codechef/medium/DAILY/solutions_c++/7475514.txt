#include <stdio.h>
long long int comb(int n, int k) 
{ 
     if (k > n) 
         return 0; 
     long long int r = 1; 
     for (long long int d = 1; d <= k; ++d) 
     { 
         r *= n--; 
         r /= d; 
     } 
     return r; 
} 
int main()
{
	int x,n,i,j;
	int b[10][10]={};
	int a[11],i_ans=0;
	long long int ans=0;
	char ch[100];

	scanf("%d",&n);

	while(n--)
	{
		scanf("%s",ch);
		for(i=1;i<=10;i++)
			a[i]=0;
		
		for(j=0;j<54;j++)
		{
			
			i=j+1;
			if(ch[j]=='0')
			{
		
				if((i>=1 && i<=4)||(i==54)||(i==53))
					a[1]++;
				else if((i>=5 && i<=8)||(i==52)||(i==51))
					a[2]++;
				else if((i>=9 && i<=12)||(i==50)||(i==49))
					a[3]++;
				else if((i>=13 && i<=16)||(i==48)||(i==47))
					a[4]++;
				else if((i>=17 && i<=20)||(i==46)||(i==45))
					a[5]++;
				else if((i>=21 && i<=24)||(i==44)||(i==43))
					a[6]++;
				else if((i>=25 && i<=28)||(i==42)||(i==41))
					a[7]++;
				else if((i>=29 && i<=32)||(i==40)||(i==39))
					a[8]++;
				else if((i>=33 && i<=36)||(i==38)||(i==37))
					a[9]++;

			}
		}

		for(i=1;i<=10;i++)
		{
			if(a[i]==x)
				ans++;

			else if(a[i]>x)
			{
				if(b[a[i]][x]==0)
				{
					b[a[i]][x]=comb(a[i],x);
					ans+=b[a[i]][x];
				}
				else
					ans+=b[a[i]][x];

			}

		}

	}

	printf("%lld\n",ans);

}
