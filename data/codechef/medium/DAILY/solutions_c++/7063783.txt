#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	int x,n,i,j,sum=0,s=0,l,k;
	scanf("%d%d",&x,&n);
	int a[7]={1,1,2,6,24,120,720};
	char c[55];
	for(i=0;i<n;i++)
	{
		scanf("%s",c);
		s=0;
		for(j=0,k=strlen(c)-1;j<33;j+=4,k-=2)
		{
			if(c[j]=='0')
				s+=1;
			if(c[j+1]=='0')
				s+=1;
			if(c[j+2]=='0')
				s+=1;
			if(c[j+3]=='0')
				s+=1;
			if(c[k]=='0')
				s+=1;
			if(c[k-1]=='0')
				s+=1;
			if(s>=x)
				sum+=a[s]/(a[x]*a[s-x]);
			s=0;
		}
	}
	printf("%d\n",sum);
return 0;
}
