#include<iostream>
#include<string.h>

using namespace std;

int freeSeats(int i, char *a)
{
    int j, free = 0;

    for(j = i*4-4; j < i*4; j++)
    {
        if(a[j] == '0')
        {
            free++;
        }
    }

    j = (2*(9-i)) + 36;
    if(a[j] == '0')
        free++;

    if(a[j+1] == '0')
        free++;

    return free;
}

int combination(int n, int r)
{
    if(n < r)
        return 0;
    else
    {
        int f = 1, f1 = 1, f2 = 1;
        int i;
        for(i = 2; i <= n; i++)
        {
            f = f*i;
        }

        for(i = 2; i <= (n-r); i++)
        {
            f1 = f1*i;
        }

        for(i = 2; i <= r; i++)
        {
            f2 = f2*i;
        }

        return (f/(f1*f2));
    }
}

int main()
{
    int x, n, i, j;
    char car[10][55];
    int ans = 0, free;
    //cout<<combination(6, 3)<<"\n";
    cin>>x;
    cin>>n;

    for(i = 0; i < n; i++)
    {
        cin>>car[i];

        for(j = 1; j < 10; j++)
        {
            free = freeSeats(j, car[i]);
            //cout<<free<<"\n";
            ans = ans + combination(free, x);
        }
    }

    cout<<ans<<"\n";
    return 0;

}
