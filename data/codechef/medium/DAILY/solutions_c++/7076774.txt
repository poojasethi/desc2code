#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{	
	long comb[7][7],i,j,n,x,c; string s;
	
	for(i=0;i<=6;i++)
	for(j=0;j<=i;j++)
	{
		if(j==0 || i==j)
		comb[i][j]=1;
		else
		comb[i][j]=comb[i-1][j] + comb[i-1][j-1];
	}
	
	cin>>x>>n;
	c=0;
	while(n--)
	{
		cin>>s;
		int t;
		for(i=1;i<=9;i++)
		{
			t=0;
			for(j=4*(i-1);j<=4*i-1;j++)
			if(s[j]=='0')
			t++;
			
			for(j=55-2*i;j>=54-2*i;j--)
			if(s[j]=='0')
			t++;
			
			if(t>=x)
			c+=comb[t][x];
		}
	}
	cout<<c<<endl;
	return 0;
}
