int cr(int n, int r) {
	int ret = 1;
	for (int i = 2; i <= n; i++) {
		ret *= i;
	}
	for (int i = 2; i <= r; i++) {
		ret /= i;
	}
	for (int i = 2; i <= (n - r); i++) {
		ret /= i;
	}
	return ret;
}
