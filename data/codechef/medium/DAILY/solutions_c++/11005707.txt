#include<bits/stdc++.h>
using namespace std;
long long int sum=0;
long factorial(int n) {
   int c;
   long result = 1;
 
   for (c = 1; c <= n; c++)
      result = result*c;
 
   return result;
}
long find_ncr(int n, int r) {
   long result;
 
   result = factorial(n)/(factorial(r)*factorial(n-r));
 
   return result;
}
int main()
{
	int x,y,k,j,count;
	cin>>x>>y;
	while(y--)
	{
	char a[55];
	scanf("%s",a);
 
	for(k=0,j=53;k<=35,j>=36;k=k+4,j=j-2)
	{
            count=0;
			if(a[k]=='0')
			count++;
			if(a[k+1]=='0')
			count++;
			if(a[k+2]=='0')
			count++;
			if(a[k+3]=='0')
			count++;
			if(a[j]=='0')
			count++;
			if(a[j-1]=='0')
			count=count+1;
	if(count>=x)
	sum=sum+find_ncr(count,x);
	}
}
cout<<sum<<endl;
 
}
