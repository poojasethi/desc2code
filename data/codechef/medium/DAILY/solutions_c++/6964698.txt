#include<iostream>
#include<string.h>
#include<stdio.h>

using namespace std;

int fact(int n) {
	int i, ans=1;
	for(i=1; i<=n; ++i)
		ans *= i;
	return ans;
}

int main() {
	int X, N, i, j, v, ans=0;
	char seat[55];
	scanf("%d %d", &X, &N);
	while(N--) {
		scanf("%s", seat);
		for(i=0; i<36; i+=4) {
			v=0;
			for(j=i; j<i+4; ++j)
				if(seat[j]=='0')
					++v;
			if(seat[53-(i/4)*2]=='0')
				++v;
			if(seat[52-(i/4)*2]=='0')
				++v;
			if(v>=X)
				ans+=fact(v)/(fact(X)*fact(v-X));
		}
			
	}
	printf("%d\n", ans);
}
