#include<stdio.h>
#include<iostream>
int fact(int n)
{ if(n==0)
    return 1;
  int sum=1;
  for(int i=n;i>1;i--)
     sum*=i;
  return sum;   
}
int main()
{ int x,n,total=0;
  scanf("%d%d",&x,&n);
  char s[55];
  for(int i=0;i<n;i++)
     { scanf("%s",&s);
       for(int i=0,j=53;i<35;i+=4,j-=2)
          { int count=0;
            if(s[i]=='0') count++;
            if(s[i+1]=='0') count++;
            if(s[i+2]=='0') count++;
            if(s[i+3]=='0') count++;
            if(s[j]=='0') count++;
            if(s[j-1]=='0') count++;
            if(count>=x)
              total+=(fact(count))/((fact(count-x))*fact(x));
          }   
     }
  printf("%d",total);   
}
