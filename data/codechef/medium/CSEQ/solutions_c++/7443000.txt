#include<iostream>
#include<cstdio>
using namespace std;
#define ll long long int
#define mod 1000003
ll fact[1000001];
 
ll fast_exp(ll base, ll exp)
{
ll res=1;
while(exp>0)
{
if(exp%2==1) res= (res*base)%mod;
base= (base*base)%mod;
exp/=2;
}
return res;
}
 
ll fermat(ll n, ll p)
{
ll res;
if(p>n)
return 0;
else if(p==n) return 1;
else
{
ll res1 = (fast_exp(fact[p], mod-2));
ll res2 = (fast_exp(fact[n-p], mod-2));
res= (res1*res2)%mod;
res= (res*fact[n])%mod;
}
return res;
}
 
int main()
{
int t;
ll i;
fact[0]=fact[1]=1;
for(i=2; i<1000001; i++)
fact[i]= (i*fact[i-1])%mod;
 
ll n, l, r;
cin>>t;
while(t--)
{
cin>>n>>l>>r;
ll x= r-l+1;
ll ans= fermat((n+x)%mod, x%mod);
ans= ans*fermat((n+x)/mod, x/mod);
ans= (ans+mod-1)%mod ;
cout<<ans<<endl;
}
return 0;
}
