#include <bits/stdc++.h>
#define ll long long
#define MOD 1000003

using namespace std;
#define MAX 1000005
ll f[MAX],invf[MAX];
ll mod_expo(ll base,ll expo)
{
    if(expo==0)
        return 1;
    if(expo%2==0)
    {
        ll y=mod_expo(base,expo/2);
        return (y*y)%MOD;
    }
    return (base*mod_expo(base,expo-1))%MOD;
}

ll choose(ll n,ll r)
{
    if(r>n)
        return 0;
    return (((f[n]*invf[r])%MOD)*invf[n-r])%MOD;
}

ll lucas(ll n,ll r)
{
    ll ret=1;
    while(n)
    {
        ret*=choose(n%MOD,r%MOD);
        ret%=MOD;
        n/=MOD;
        r/=MOD;
    }
    return ret;
}

void pre()
{
    f[0]=1;invf[0]=1;
    for(int i=1;i<MAX;i++){
        f[i]=(i*f[i-1])%MOD;
        invf[i]=(mod_expo(i,MOD-2)*invf[i-1])%MOD;
    }
}

int main()
{
    int t;
    scanf("%d",&t);
    pre();
    while(t--)
    {
        ll n,l,r;
        scanf("%lld%lld%lld",&n,&l,&r);
        printf("%lld\n",(lucas(n+r-l+1,n)-1+MOD)%MOD);
    }
    return 0;
}
