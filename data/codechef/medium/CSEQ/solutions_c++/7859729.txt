#include<bits/stdc++.h>
using namespace std;
long long int f[2011100];
long long int mod=1e6+3;
long long int poww(long long int x,long long int y)
{
    long long int s=1;
    while(y)
    {
        if(y&1) s=(s*x)%mod;
        x=(x*x)%mod;
        y>>=1;
    }
    return s;
}
long long int cal(long long int x,long long int y)
{
    if(x>y) return 0;
    return (f[y]*poww((f[x]*f[y-x])%mod,mod-2))%mod;
}
long long int c(long long int x,long long int y)
{
    if(x<mod && y<mod) return cal(x,y);
    return (c(x/mod,y/mod)*c(x%mod,y%mod))%mod;
}
int main()
{
    f[0]=1;
    //freopen("test.inp","r",stdin);
    long long int i,n,l,r,t;
    scanf("%lld",&t);
    for(i=1;i<=2*mod;i++)
    {
        f[i]=(f[i-1]*i)%mod;
    }
    while(t--)
    {
    scanf("%lld %lld %lld",&n,&l,&r);
    printf("%lld\n",(c(r-l+1,r-l+n+1)+mod-1)%mod);
    }
}
