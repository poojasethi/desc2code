#include <bits/stdc++.h>
#define M 1000003
#define lli long long

using namespace std;

lli L,R,K,N;
lli fact[1000005];

lli fast_pow(lli a, lli b)
{
	lli res = 1;
	while ( b > 0 ) {
		if ( b&1 ) res = (res*a)%M;
		a = (a*a)%M;
		b >>= 1;
	}
	return res;
}

lli C(lli a, lli b)
{
	if ( a == b ) return 1;
	lli ans = 1;
	while ( a > 0 || b > 0 ) {
		lli a1 = a % M;
		lli b1 = b % M;
		if ( b1 > a1 ) {
			ans = 0;
			break;
		}
		ans = (ans*fact[a1])%M;
		ans = (ans*fast_pow(fact[b1], M-2))%M;
		ans = (ans*fast_pow(fact[a1-b1], M-2))%M;
		a /= M;
		b /= M;
	}
	return ans;
}

int main()
{
	fact[0] = 1LL;
	for ( lli i = 1; i <= 1000002; i++ ) fact[i] = (fact[i-1]*i)%M;
	int t;
	cin >> t;
	while ( t-- ) {
		cin >> N >> L >> R;
		K = R-L+1;
		lli ans = C(N+K,K);
		ans--;
		if ( ans < 0 ) ans += M;
		cout << ans << endl;
	}
	return 0;
}
