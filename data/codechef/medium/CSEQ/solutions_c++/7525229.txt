#define ll long long
ll mod=1000003;
ll pow(ll a,ll b)
{
	if(b==0)return 1;
	 ll ans=pow(a,b/2);
	ans*=ans;
	ans%=mod;
	if(b%2==1){ans*=a;ans%=mod;}
	return ans;
} 
ll fact[1000007],infact[1000007];


ll ncr(ll n,ll r)
{ll ans;
	ans=(((fact[n]*infact[r])%mod)*infact[n-r])%mod;
	return ans;
}

ll lucas(ll n,ll r)
{
	if(n==0&&r==0) return 1;
	ll ni=n%mod,ri=r%mod;
	if(ri>ni) return 0;
	else 
	return (lucas(n/mod,r/mod)*ncr(ni,ri))%mod;
}
