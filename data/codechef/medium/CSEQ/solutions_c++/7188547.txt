#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000003
ll fact[mod+5];
ll invfact[mod+5];
ll pow1(ll x,ll y)
{ll temp;
    if(y==0)
        return 1;
    temp=pow1(x,y/2);
    if(y%2==0)
        return (temp*temp)%mod;
    else
        return (temp*(temp*x)%mod)%mod;
}
ll func(ll n, ll r)
{
    ll a=fact[n];
    ll b=invfact[r];
    ll c =invfact[n-r];
    return ((a*(b*c)%mod)%mod);
}
void ini()
{
     fact[0]=1;
     fact[1]=1;
     invfact[0]=pow1(fact[0],mod-2)%mod;
     invfact[1]=pow1(fact[1],mod-2)%mod;
     ll i;
     for(i=2;i<(mod+5);i++)
     {
         fact[i]=(fact[i-1]*i)%mod;
         invfact[i]=pow1(fact[i],mod-2)%mod;
     }
}
ll calc(ll n,ll r)
{
    ll a,b;
    if(n==0&&r==0)
        return 1;
    a=n%mod;
    b=r%mod;
    if(a<b)
        return 0;
    return (calc(n/mod,r/mod)*func(a,b));
}

int main()
{
    ll t;
    scanf("%lld",&t);
    ini();
    while(t--)
    {ll n,l,r;
        scanf(" %lld %lld %lld",&n,&l,&r);
        ll ans=(calc(n+r-l+1,n)-1)%mod;
        if (ans<0) ans+=mod;
        printf("%lld\n",ans);
    }
    return 0;
}
