void mulmat(int A[105][105], int B[105][105])
{
    memset(Rmat, 0, sizeof(Rmat));
    for (int i = 0; i < LIM; i++)
        for (int j = 0; j < LIM; j++){
            ll x = 0;
            for (int t = 0; t < LIM; t++)
                x += 1ll*A[i][t] * B[t][j];
            if(x >= MOD2) x %= MOD2;
            Rmat[i][j] = x;
            }
}

void pow_matrix(int p)
{
    memset(Cmat, 0, sizeof(Cmat));
    for (int i = 0; i < LIM; ++i)
        Cmat[i][i] = 1;

    while (p > 0)
    {
        if (p%2 == 1){
             mulmat(Cmat, Amat);
             memcpy(Cmat, Rmat, sizeof(Rmat));
        }
        p /= 2;
        mulmat(Amat, Amat);
        memcpy(Amat, Rmat, sizeof(Rmat));
    }
    //pmatrix(Cmat,k);
    //Final answer is stored in Cmat
}

int powmod(ll a,ll P){
    ll r = 1;
    for(;P>0;P>>=1)
    {
        if(P&1) r = (r*a)%MOD;
        a = (a*a)%MOD;
    }
    return r;
}
