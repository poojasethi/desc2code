typedef long long ll;
typedef pair<int,int> pii;

const int mod = 1e8 + 7, mod2 = mod - 1;

ll tax[1<<10];

struct M {
  int a[111][111],d;
  M() {
    memset(a,0,sizeof a);
  }
  M operator*(M m) const {
    M ret;
    ret.d = m.d;
    rep(i,d) rep(j,d) {
      ll x = 0;
      rep(k,d) x += a[i][k] * (ll)m.a[k][j];
      ret.a[i][j] = x % mod2;
    }
    return ret;
  }
};

M fpow(M m, ll e) {
 	M x; x.d = m.d;
	rep(i,x.d) x.a[i][i] = 1;
	while(e){
		if(e&1) x = x*m;
		if(e>>=1) m=m*m; else break;
	}
	return x;
}

ll fpow(ll m, ll e) {
	ll x = 1;
	while(e){
		if(e&1) x=x*m%mod;
		if(e>>=1) m=m*m%mod; else break;
	}
	return x;
}
int main() {
  int T;
  for(cin>>T;T;T--){
    int initTax, slot1, slot2, K, N;
    cin >> initTax >> slot1 >> slot2 >> K >> N;
    int n = 1;
    tax[n] = initTax % mod;
    rep(i,slot1) tax[n+1] = (tax[n]+1) % mod, n++;
    rep(i,slot2) tax[n+1] = (tax[n]*2) % mod, n++;
    if(n >= N) {
      printf("%lld\n", tax[N]);
      continue;
    }
    vector<ll> v;
    rep(i,K) v.push_back(tax[n-i]);
    reverse(v.begin(),v.end());
    M m; m.d = K;
    rep(i,K-1) m.a[i][i+1] = 1;
    rep(i,K) m.a[K-1][i] = 1;
    m = fpow(m,N - n);
    ll x = 1;
    rep(i,K) x = x * fpow(v[i], m.a[K-1][i]) % mod;
    printf("%lld\n",x);
  }
  return 0;
}
