#include<cstdio>
#include<cctype>
#include<queue>
using namespace std;
#define min1(a,b) (a<b ? a:b)
const int sta=1<<14;
const int INF=0x3f3f3f3f;
char s[17][17];
int d[sta][17][17];
int x=0,y=0,co,n;
int bfs()
{
    queue<int> q;
    int posx[]= {0,0,-1,1},posy[]= {1,-1,0,0};
    int xi,yi,zi=0,i,xf,yf,zf,dist=INF;
    d[zi][x][y]=0;
    q.push(x);
    q.push(y);
    q.push(zi);
    while(!q.empty())
    {
        xi=q.front();
        q.pop();
        yi=q.front();
        q.pop();
        zi=q.front();
        q.pop();
        for(i=0; i<4; i++)
        {
            xf=xi+posx[i];
            yf=yi+posy[i];
            zf=zi;
            if(xf>=0 && xf <n && yf>=0 && yf<n && s[xf][yf]!='#')
            {
                if(isupper(s[xf][yf]))
                    zf|=(1<<(s[xf][yf]-'A'));
                if(d[zf][xf][yf]>d[zi][xi][yi]+1)
                {
                    d[zf][xf][yf]=d[zi][xi][yi]+1;
                    if(zf==((1<<co)-1) && xf==n-1 && yf==n-1)
                        dist=min1(dist,d[zf][xf][yf]);
                    q.push(xf);
                    q.push(yf);
                    q.push(zf);
                }
            }
        }
    }
        return ((dist < INF) ? dist : -1);
}
int main()
{
    int i,j,k,t;
    scanf("%d",&t);
    while(t--)
    {
        co=0;
        scanf("%d",&n);
        for(i=0; i<n; i++)
        {
            scanf("%s",s[i]);
            for(j=0; j<n; j++)
            {
                if(s[i][j]=='*')
                {
                    s[i][j]='A'+ co++;
                }
            }
        }
        for(k=0; k<(1<<co); k++)
            for(i=0; i<n; i++)
                for(j=0; j<n; j++)
                    d[k][i][j]=INF;
        printf("%d\n",bfs());
    }
    return 0;
}
