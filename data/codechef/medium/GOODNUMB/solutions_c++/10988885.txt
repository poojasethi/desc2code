#include<vector>
#include<math.h>
#define max 403201
using namespace std;
vector <bool> a(max,true);long b[max]={};long c[max]={};long long int d[max]={},i,j;
int main(){
	for(i=2;i<=sqrt(100000);i++){
		for(j=i*i;j<=100000;j+=i*i) a[j]=false;
	}
	for(i=1;i<=max;i++){
		for(j=i;j<=max;j=j+i){
			c[j]=c[j]+i;
			if(c[i]==i+1)
			b[j]++;
		}
	}
	for(i=1;i<=100000;i++){
		if(a[i]==true&&c[b[c[i]]]==b[c[i]]+1)
		d[i]=d[i-1]+c[i];
		else d[i]=d[i-1];
	}
	int test;cin>>test;
	while(test--){
		int inp1,inp2;
		cin>>inp1>>inp2;
		cout<<(d[inp2]-d[inp1-1])<<endl;
	}
}
