#include<cstdio>
#include<iostream>
#include<map>
using namespace std;
 
int main() {
int n, a[105];
 
scanf("%d", &n);
for(int i=0; i<n; i++)
scanf("%d", &a[i]);
 
int ans=0;
map<int, int> hash;
for(int i=0; i<n; i++)
{
for(int j=i+1; j<n; j++)
 {
int t=a[i]+a[j];hash[t]++;
if(hash[t]>ans) ans=hash[t];
}
}
 
cout<<2*ans;
}
