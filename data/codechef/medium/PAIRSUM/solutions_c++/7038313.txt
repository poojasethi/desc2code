#include<iostream>
#include<cstdio>
#include<string.h>
#include<stdlib.h>
#include<cmath>
#include<bits/stdc++.h>
#define S scanf 
#define P printf 
using namespace std;
inline int maxi(int a,int b)
{
	return a>b?a:b;
}
int main()
{
	int j,k,n,i,arr[1000],l=0;
	S("%d",&n);
	for(i=0;i<n;i++)
	{
		S("%d",&arr[i]);
	}
	sort(arr,arr+n);
	i=arr[0]+arr[1]; // i is minimum
	int max=arr[n-1]+arr[n-2]; 
	while(i<max)    //iterating from minimum pair sum to maximum pair sum
	{
		j=0;
		k=n-1;
		int y=0;
		while(j<k)
		{
			if(arr[j]+arr[k]==i)
			{
				k--;
				j++;
				y++;
			}
			else if(i<arr[j]+arr[k])
			{
				k--;
			}
			else
			{
				j++;
			}
		}	
		l=maxi(l,y);
		i++;
	}
	P("%d",2*l);
	return 0;
}
