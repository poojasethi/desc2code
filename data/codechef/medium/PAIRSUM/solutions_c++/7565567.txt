#include<iostream>
#include<map>


using namespace std;
int main()
{
    int a[105];
    int n,i,ans=0,j,sum;
    cin>>n;
    map<int,int> m;

    for(i=0;i<n;i++)
        cin>>a[i];
       for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            sum=a[i]+a[j];
            m[sum]++;
            if(m[sum]>ans)
                ans=m[sum];
        }
    }
    cout<<2*ans;
    return 0;
}
