vector<int>arr;
int countpairs(int L,int R,int sum)
{
    int i=L,j=R,ans=2;
    while(i<j)
    {
        if(arr[i]+arr[j]==sum)
        {
            ++i;
            --j;
            ans+=2;
        }
        elif(arr[i]+arr[j]<sum)
        {
            ++i;
        }
        else
        {
            --j;
        }
    }
    return ans;
}
int solve(int L,int R)
{
    int sum=0,ans=0,i,j;
    FORAB(i,L,R)
    {
        FORR(j,R+1)
        {
            sum=arr[i]+arr[j];
            ans=MAX(ans,countpairs(i+1,j-1,sum));
        }
    }
    return ans;
}
