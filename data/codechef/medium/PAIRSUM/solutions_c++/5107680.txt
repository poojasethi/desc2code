#include<stdio.h>
#include<algorithm>
using namespace std;

int main(){
	int n,ans,i,j,k,l,sum,temp;
	int arr[110];
	scanf("%d",&n);
	//input
	for(i=0;i<n;i++){
					 scanf("%d",&arr[i]);
					 
					 }
	sort(arr,arr+n);
	
	// finding the ans
	ans=2;
	for(i=0;i<n;i++){
		for(j=n-1;j>i+2;j--){
					sum=arr[i]+arr[j];		 
					 temp=2;
					 // finding the matches
					 k=i+1; 
					 while (true){
					 	   l=j-1;
					 	   while(arr[k]+arr[l] > sum){
						   				l--;	   
						   				if(k>=l )
										     break;	   
												  }
							if(k==j-1)
							 break;
							if(arr[k]+arr[l] == sum)
							 temp+=2; 					  
					 	   k++;
							
							}
							
							 
							 if(temp>ans)
							   ans=temp;
							 
							 
							 }			 
					 
					 }
	
	printf("%d\n",ans);

	}
