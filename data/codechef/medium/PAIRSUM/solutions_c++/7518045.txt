#include<stdio.h>
#include<iostream>
#include<map>
using namespace std;
int main()
{
    int n,i,j,t,ans=0;
    scanf("%d",&n);
    int arr[n];
    for(i=0;i<n;i++)
        scanf("%d",&arr[i]);
    map<int , int> m;
    for(i=0;i<n;i++)
        for( j=i+1;j<n;j++)
        {
            t=arr[i]+arr[j];
            m[t]++;
            if(m[t]>ans)
                ans=m[t];
        }
    printf("%d",2*ans);
}
