#include <iostream>
#include<algorithm>
using namespace std;
typedef long long int ll;

int main()
{
	ll arr[100], A[10000], cnt=0, N, i, j;
	ll arr2[100][100];
	cin>>N;
	for( i=0;i<N;i++)
	cin>>arr[i];

	for( i=0;i<N;i++)
        for( j=0;j<N;j++)
            arr2[i][j]=arr[i]+arr[j];

	for( i=0;i<N;i++)
    {
        for( j=i+1;j<N;j++)
            A[cnt++]=arr2[i][j];
    }
    sort(A,A+cnt);
    //finding mode and frequency
    ll num=A[0], count=1, cntmode=1, mode=A[0];
    for ( i=0; i<cnt; i++)
    {
        if (A[i]==num)
            count++;
        else
        {
            if (cntmode<count)
            {
                cntmode=count;
                mode=num;
            }
            count=1;
            num=A[i];
        }
    }
    cout<<2*cntmode<<endl;
	return 0;
}
