#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
const char fi[] = "PAIRSUM.INP";
const int MAXN = 101;
int a[MAXN];
 
int main()
{
    int n;
    int i,j,k,l;
    int sum;
    int upper;
    int cur;
    int val;
    int pos;
    //freopen(fi,"r", stdin);
    scanf("%d", &n);
    for (i=0;i<n;i++) scanf("%d", &a[i]);
 
    sort(a, a+n);
    int best = n>=2?2:0;
    for (i=0;i<n;i++)
    for (j=n-1;j>i;j--)
    {
        sum = a[i] + a[j];
        upper = j-1;
        cur = 2;
        for (k=i+1;k<upper;k++)
        {
            for (l=upper;l>k;l--)
            {
                if (a[k]+a[l] == sum)
                {
                    cur+=2;
                    upper = l-1;
                    break;
                }
            }
        }
        best = max(best, cur);
    }
    printf("%d\n", best);
    return 0;
}
