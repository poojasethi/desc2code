#include <iostream>
#include <algorithm>
#include <set>
using namespace std;
 
int main()
{
int n;
cin>>n;
 
int a[n];
for(int i=0;i<n;++i)
{
	cin>>a[i];
}
 
 
 
sort(a,a+n);
 
set<int> sum;
for(int i=0;i<n;++i)
{
	for(int k=i+1;k<n;++k)
	{
		sum.insert(a[k]+a[i]);
		
	}	
 
}
int ans=0;
for(set<int>::iterator it=sum.begin();it!=sum.end();++it)
{
 
	int s= *it;
	int c=0;
	for(int i=0,j=n-1;i<j;)
	{
		int d = a[i]+a[j];
 
		if(d==s)
		{
			++c; ++i; --j;
		}
		else if(d>s){  --j;   }
		else ++i;
	}
 
	ans = max(ans,2*c);
}
 
cout<<ans<<endl;
 
return 0;
 
}
