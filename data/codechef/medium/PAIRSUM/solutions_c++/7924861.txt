#include<iostream>
#include<map>
using namespace std;
 
int main() {
    int n, a[105];
 
    cin>>n;
    for(int i=0; i<n; i++) cin>>a[i];
 
    int ans=0;
    map<int, int> h;
    for(int i=0; i<n; i++)
    for(int j=i+1; j<n; j++) {
        h[a[i]+a[j]]++;
        if(h[a[i]+a[j]]>ans) 
        ans=h[a[i]+a[j]];
    }
 
    cout<<2*ans;
}
