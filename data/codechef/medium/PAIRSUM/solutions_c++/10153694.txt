#include<bits/stdc++.h>
using namespace std;
int A[200];
int main(){
	int N;
	scanf("%d", &N);
	map<int, int> mp;
	for(int i = 1;i <= N;++i){
		scanf("%d", &A[i]);
		mp[A[i]] += 1;
	}
	int ans = 0;
	for(int i = 1;i <= N;++i){
		for(int j = i + 1;j <= N;++j){
			int cur = A[i] + A[j];
			int curAns = 0;
			for(map<int, int> :: iterator it = mp.begin();it != mp.end();++it){
				int a = it->first;
				int b = cur - a;
				int cnt = it->second;
				if(a < b){
					curAns += 2 * min(cnt, mp[b]);
				} else if(a == b){
					curAns += ((cnt&1)?cnt - 1:cnt);
				} else break;
			}
			ans = max(ans, curAns);
		}
	}
	printf("%d\n", ans);

	return 0;
}
