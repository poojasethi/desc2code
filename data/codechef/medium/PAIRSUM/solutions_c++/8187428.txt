#include<vector>
#include<iostream>
#include<cstdio>
#include<map>
#include<cstdlib>
using namespace std;
int main()
{
	int n,i,j,temp=0,max=0;;
	cin>>n;
	int a[102];
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	map<int,int>s;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			temp=a[i]+a[j];
			s[temp]++;
			if(s[temp] > max)
			max=s[temp];
		}
	}
	cout<<max*2;
}
