#include<iostream>
using namespace std;
int main()
{
	int n,i;
	cin>>n;
	int *arr = new int[n];
	for(i=0;i<n;i++)
	cin>>arr[i];
	for(int j=1;j<n;j++)
	{
		int temp = arr[j];
		for(i=j-1;i>=0 && arr[i] > temp;i--)
		arr[i+1] = arr[i];
		arr[i+1] = temp;
	}
	int s,k,l,m,maxlen=1,sum,len=1;
	for(s=0;s<n;s++)
	{
		for(k=n-1;k>s;k--)
		{
			sum = arr[s] + arr[k];
			m = k-1;
			l = s+1;
			len = 1;	
			while(l<m)
			{
				if(arr[l] + arr[m] == sum)
				{
					len++;
					l++;
					m--;		
				}
				else if((arr[l] + arr[m]) > sum)
				m--;
				else
				l++;
			}
			if(len > maxlen)
			maxlen = len;
		}
	}
	cout<<2*maxlen;
}
