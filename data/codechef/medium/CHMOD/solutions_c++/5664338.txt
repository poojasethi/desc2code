#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
#include <map>

#define MAX 100005
typedef long long ll;

using namespace std;

int a[MAX][25];
int prime[25];
map<int, int> m;

int f(int a, int b, int c)
{
    if(b==0)
        return 1;

    ll ans=1, y=a;

    while (b>0) {
        if(b&1)
            ans=(ans*y)%c;
        y=(y*y)%c;
        b/=2;
    }
    return ans%c;
}

int main()
{
    memset(a, 0, sizeof(a));

    int k=0;
    for (int i=2; i<=100; i++) {
        int flag=0;
        for (int j=2; j*j<=i; j++) {
            if (i%j==0)
                flag=1;
        }
        if(!flag) {
            prime[k]=i;
            m[i]=k++;
        }
    }

    int n, x;
    scanf("%d", &n);

    for (int i=0; i<n; i++) {
        scanf("%d", &x);

        int cnt;
        for (int j=0; j<25 && x>0; j++) {
            cnt=0;
            if (x%prime[j]==0) {
                while (x%prime[j]==0) {
                    x/=prime[j];
                    cnt++;
                }
            }
            if (i==0)
                a[0][j] = cnt;
            else
                a[i][j] = cnt + a[i-1][j];
        }
    }

    int q, l, r, m;

    scanf("%d", &q);
    while (q--) {
        scanf("%d %d %d", &l, &r, &m);
        ll ans=1;
        if(l==1) {
            for (int i=0; i<25; i++) {
                ll u=a[r-1][i];
                ll z=f(prime[i], u, m);
                ans = (ans*z)%m;
            }
        } else {
            for (int i=0; i<25; i++) {
                int u=a[r-1][i] - a[l-2][i];
                int z=f(prime[i], u, m);
                ans = (ans*z)%m;
            }
        }
        printf("%d\n", ans);
    }
    return 0;
}
