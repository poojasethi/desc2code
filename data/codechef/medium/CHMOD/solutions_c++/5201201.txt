long long power(long long b,long long p,long long m)
{
if(b==0)
return 1;
long long int x = 1, y = b;
while(p > 0) {
if(p%2 == 1) {
x=(x*y);
if(x>m) x%=m;
}
y = (y*y);
if(y>m) y%=m;
p/= 2;
}
return x;
}
