#include <iostream>
using namespace std;

int main(void) {
	int  t,m,n,*x,y,i,c;
	cin>>t;
	while(t--)
	{
		cin>>m>>n;
		x=new int[m+1];
		for(i=0;i<n;i++)
		{
			cin>>y;
			x[y]=-1;
		}
		c=1;
		for(i=1;i<=m;i++)
		{
			if(x[i]!=-1)
			{
				c++;
				if(c==2)
				{
					cout<<i<<" ";
					x[i]=-1;
					c=0;
				}	
			}
		}
		cout<<endl;
		for(i=1;i<=m;i++)
		{
			if(x[i]!=-1)
			{
				cout<<i<<" ";
			}
		}
		cout<<endl;
	}
	return 0;
}
