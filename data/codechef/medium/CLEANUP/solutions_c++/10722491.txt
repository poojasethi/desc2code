#include <iostream>

using namespace std;

int main()
{
	int T, n, m, arr[1000], i, j, num;
	cin >> T;
	if(T > 50)
		return 1;
	for(i=0; i<T; i++)
	{
		cin >> n >> m;
		if(m > n || n > 1000 || m < 0)
			return 1;
		for(j=0; j<n; j++)
			arr[j] = 0;
		for(j=0; j<m; j++)
		{
			cin >> num;
			arr[num-1] = -1;
		}
		num = 0;
		for(j=0; j<n; j++)
			if(arr[j] == 0)
				if(num == 1)
				{
					arr[j] = 1;
					num = 0;
				}
				else
					num = 1;
		for(j=0; j<n; j++)
			if(arr[j] == 0)
				cout << j+1 <<" ";
		cout << endl;
		for(j=0; j<n; j++)
			if(arr[j] == 1)
				cout << j+1 <<" ";
		cout << endl;
	}
	return 0;
}
