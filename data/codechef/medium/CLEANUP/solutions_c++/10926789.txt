#include<iostream>
using namespace std;
int main()
{
	int t,n,m,q=0;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int o=-1,l=-1,flag=0;
	        cin>>n>>m;
       		int a[m],b[n-m],c[n-m];
		for(int j=0;j<m;j++)
		{
	                cin>>a[j];
		}
		for(int k=1;k<=n;k++)
		{
			q=0;
			for(int j=0;j<m;j++)
			{
				if(k==a[j])
				{
				q=1;
				}
			}
			if(q==0 && flag==0)
			{
			    ++l;
				c[l]=k;
				flag=1;
 
			}
			else
			if(q==0 && flag==1)
			{
              			  ++o;
				b[o]=k;
				flag=0;
 
			}
 
		}
		for(int z=0;z<l+1;z++)
		{
			cout<<c[z]<<" ";
		}
		cout<<endl;
		for(int z=0;z<o+1;z++)
		{
			cout<<b[z]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
