#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
 
using namespace std;
 
int main()
{
    int t;
    scanf("%d",&t);
    int n,m;
    int jobs[1001];
    for(int j=0;j<1001;j++)
    {
        jobs[j]=0;
    }
    for(int i=0;i<t;i++)
    {
        scanf("%d %d",&n,&m);
        int a;
        for(int j=0;j<m;j++)
        {
            scanf("%d",&a);
            jobs[a] = 1;
        }
        int chef=1;
        for(int k=1;k<n+1;k++)
        {
            if(chef && (jobs[k]==0))
            {
                printf("%d ",k);
                chef=0;
                jobs[k]=1;
            }
            else if(!chef && jobs[k]==0)
            {
                chef=1;
 
            }
        }
        printf("\n");
        for(int k=1;k<n+1;k++)
        {
            if(jobs[k]!=1)
            {
                printf("%d ",k);
            }
            else
            {
                jobs[k]=0;
            }
        }
        printf("\n");
    }
    return 0;
}
