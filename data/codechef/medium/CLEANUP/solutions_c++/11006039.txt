#include<iostream>
#include<vector>
using namespace std;

int main(){
	int t; cin>>t;
	int num;
	int n,m;
	while(t--){
		cin>>n>>m;
		bool a[n+1];
		vector<int> ve;
		for(int i=0;i<n+1;i++)
			a[i]=false;
		for(int i=0;i<m;i++){
			cin>>num;
			a[num]=true;}
		num=-1;
		for(int i=1;i<n+1;i++){
			if(a[i]==false){
				num = i;
				break;}
		}
		if(num!=-1){
			cout<<num<<" ";
			int count=2;
			for(int i=num+1;i<n+1;i++){
				if(a[i]==false){
				    if(count%2)
					 cout<<i<<" ";
					else
					 ve.push_back(i);
					count++;
		        }
			}
			cout<<endl;
			num = ve.size();
			if(num!=0){
			    for(int i=0;i<num;i++){
			        cout<<ve[i]<<" ";
			    }
		    }
		 
		}
		else{
		cout<<endl;
		cout<<endl;
		}
		cout<<endl;
	}
	return 0;
}
