#include<iostream>
#include<malloc.h>
using namespace std;
int main()
{
    int t,m,n;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>n>>m;
        int *arr=(int *)malloc(m*sizeof(int));
        int j=0;
        while(j<m)
        {
            cin>>arr[j];
            j++;
        }
        int *arr1=(int *)malloc(n*sizeof(int));
        for(j=0;j<n;j++)
        	arr1[j]=j+1;
        for(int j=0;j<n;j++)
        {
        	for(int k=0;k<m;k++)
        	{
        		if(arr[k]==arr1[j])
        		arr1[j]=0;
        	}
        }
        int flag=0,k=0;
        while(k<n)
        {
            if(arr1[k]!=0)
            {
                if(flag%2==0)
                cout<<arr1[k]<<" ";
                flag++;
            }

            k++;
        }
        cout<<endl;
        flag=1;
        k=0;
        while(k<n)
        {
            if(arr1[k]!=0)
            {
                if(flag%2==0)
                cout<<arr1[k]<<" ";
                flag++;
            }
            k++;
        }
        cout<<endl;
    }
    return 0;
}
