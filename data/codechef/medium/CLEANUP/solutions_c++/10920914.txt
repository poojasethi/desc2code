#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
int t,n,m;
scanf("%d",&t);
while(t--)
{
scanf("%d%d",&n,&m);
int i,f[m],j[n],flag[n],k=1;
for(i=0;i<n;i++)
{j[i]=(1+i);flag[i]=0;
}
for(i=0;i<m;i++)
{
scanf("%d",&f[i]);
flag[f[i]-1]=-1;
}

for(i=0;i<n;i++)
{
if(flag[i]==0)
{flag[i]=k;
k++;
}
}int c=0;
for(i=0;i<n;i++)
{
if(flag[i]!=-1)
{
if(flag[i]%2==1)
printf("%d ",j[i]);c++;
}
}
if(c==0)
printf("\n");
printf("\n");
c=0;
for(i=0;i<n;i++)
{
if(flag[i]!=-1)
{
if(flag[i]%2==0)
printf("%d ",j[i]);c++;
}
}
if(c==0)
printf("\n");
printf("\n");
}
return 0;
}
