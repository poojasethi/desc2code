#include<iostream>
using namespace std;
int main()
{
    int T;
    cin>>T;
    int n,m;
    int *comp;
    int jobs[1000];
    int chef[1000];
    int assist[1000];
    int ch_no,ass_no;
    for(int i=0;i<T;i++)
    {

        cin>>n>>m;
        comp=new int[m];
        for(int k=0;k<n;k++)
            jobs[k]=0;
        for(int j=0;j<m;j++)
        {
            cin>>comp[j];
            jobs[comp[j]-1]=1;
        }
        ch_no=0;
        ass_no=0;

        for(int k=0;k<n;k++)
        {
            if(jobs[k]==0)
            {
                if(ch_no==ass_no)
                    chef[ch_no++]=k+1;
                else
                    assist[ass_no++]=k+1;
            }
        }
        for(int k=0;k<ch_no;k++)
            cout<<chef[k]<<" ";
        cout<<"\n";
        for(int k=0;k<ass_no;k++)
            cout<<assist[k]<<" ";
        cout<<"\n";


    }
    return 0;
}
