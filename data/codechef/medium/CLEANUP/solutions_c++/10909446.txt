#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int t;
cin>>t;
while(t-->0)
{
int n,m;
cin>>n>>m;
int *job=new int[m];
for(int i=0;i<m;i++)
{
cin>>job[i];
}
sort(job,job+m);
int j=0;
int done=0;
for(int i=1;i<=n;i++)
{
if(i==job[j])
j++;
else
{
if(done%2==0)
{
cout<<i<<" ";
}
done++;
}
}
cout<<endl;
done=0;
j=0;
for(int i=1;i<=n;i++)
{
if(i==job[j])
j++;
else
{
if(done%2==1)
{
cout<<i<<" ";
}
done++;
}
}
cout<<endl;
}
return 0;
}
