#include <bits/stdc++.h>
using namespace std;

const int N = 1001;

int main(){
	int t, n, m, i, j, k, comp[N], ch[N], as[N];
	bool flag;
	scanf("%d", &t);
	while(t--){
		scanf("%d %d", &n, &m);
		fill(comp, comp + n + 1, 0);
		for(i = 0; i < m; ++i){
			scanf("%d", &j);
			comp[j] = 1;
		}
		flag = false;
		for(i = 1, j = 0, k = 0; i <= n; ++i){
			if(!flag && comp[i] == 0){
				ch[j++] = i;
				flag = true;
			}
			else if(flag && comp[i] == 0){
				as[k++] = i;
				flag = false;
			}
		}
		for(i = 0; i < j; ++i)
			printf("%d ", ch[i]);
		printf("\n");
		for(i = 0; i < k; ++i)
			printf("%d ", as[i]);
		printf("\n");
	}
	return 0;
}
