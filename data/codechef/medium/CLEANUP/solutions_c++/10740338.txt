#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int cup(){
    int n,m;cin >> n >> m;
    vector<int> j(n+1,0);
    vector<int> c;
    vector<int> a;
    int t;
    for(int i=0;i<m;i++){
        cin >> t;
        j[t]=1;
    }
    t=0;
    for(int i=1;i<=n;i++){
        if(!j[i]){
            if(t==0){
                c.push_back(i);
                t=1;
            }
            else{
                a.push_back(i);
                t=0;
            }
        }
    }
    for(int i=0;i<c.size();i++){
        cout << c[i] << " ";
    }
    cout << "\n";
    for(int i=0;i<a.size();i++){
        cout << a[i] << " ";
    }
    cout << "\n";
    return 0;
}



int main() {
	// your code goes here
	int t ;
	cin >> t;
	for(int i=0;i<t;i++){
	    cup();
	}
	return 0;
}
