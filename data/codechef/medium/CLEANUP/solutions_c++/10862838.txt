#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int m,n;
        cin>>n>>m;
        bool arr[n+1];
        for(int j=1;j<=n;j++)
            arr[j]=true;
        for(int j=0;j<m;j++)
        {
            int k;
            cin>>k;
            arr[k]=false;
        }
        int flag=0;
        for(int j=1;j<=n;j++)
        {
            if(arr[j])
            {
                flag=~flag;
                if(flag)
                {
                    cout<<j<<" ";
                    arr[j]=false;
                }
            }
        }
        cout<<endl;
        for(int j=1;j<=n;j++)
        {
            if(arr[j])
                cout<<j<<" ";
        }
        cout<<endl;
    }
    return 0;
}
