#include<stdio.h>
int main()
{
	int t;
	int n,m,k;
	int i,temp;
	scanf("%d",&t);
	while(t>0)
	{
		scanf("%d",&n);
		scanf("%d",&m);
		int arr[n+1];
		for(i=0;i<=n;i++)
		{
			arr[i]=1;
		}
		
		for(i=1;i<=m;i++)
		{
			scanf("%d",&k);
			arr[k]=0;
		}
		int cnt=1;
		int cnt1=1;
		for(i=1;i<=n;i++)
		{
			if(arr[i]==1 &&cnt%2)
			{
				printf("%d ",i);
				cnt++;
				
			}
			else if(arr[i]==1)
			{
				cnt++;
			}
		}
		printf("\n");
		for(i=1;i<=n;i++)
		{
			if(arr[i]==1 &&cnt1%2==0)
			{
				printf("%d ",i);
				cnt1++;
			}
			else if(arr[i]==1)
			{
				cnt1++;
			}
		}
		printf("\n");
		
	
		t--;
	}
	

}
