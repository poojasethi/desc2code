#include<iostream>
#include <vector>
#include<algorithm>
using namespace std;

bool search(int a[],int n,int size)
{
  for(int i=0;i<size;i++)
  {
   if(a[i]==n)
     return true;
  }
  return false;
}

int main()
{
  int t,i;
  cin>>t;
  for(i=0;i<t;i++)
  {
    int n,m,j=0;
    cin>>n>>m;
    int mi[m],ni[n-m];
    for(j=0;j<m;j++)
     cin>>mi[j];
    sort(mi,mi+m);
    
	int c=0;
    for(j=0;j<n;j++)
    {
      if(!search(mi,j+1,m))
      {
      	ni[c]=j+1;
        c++;
      } 
    }
    for(int k=0;k<c;k=k+2)
      cout<<ni[k]<<" ";
    cout<<endl;  
    for(int k=1;k<c;k=k+2)
      cout<<ni[k]<<" ";
     
  }
  
  return 0;
}
