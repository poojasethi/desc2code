#include<iostream>
using namespace std;
int main()
{
int i;
cin>>i;
for(int j=0;j<i;j++)
{
int n,m,con,p=0;
cin>>n>>m;
int a[n-m],b[m];
for(int y=0;y<m;y++)
cin>>b[y];
for(int q=1;q<=n;q++)
{
con=0;
for(int w=0;w<m;w++)
if(q==b[w]){
con=con+1;}
if(con==0)
{
a[p]=q;
p++;
}
}

for(int z=0;z<p;z+=2)
cout<<a[z]<<" "; cout<<"\n";
for(int z=1;z<p;z+=2)
cout<<a[z]<<" "; cout<<"\n";
}
return 0;
}
