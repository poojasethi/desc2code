#include <iostream>
#include <vector>

using namespace std;

int main()
{

    int t, n, m;

    int a[1001];
    int temp, input;
    vector<int> v1, v2;
    bool turn;

    cin>>t;
    while(t)
    {
        turn=true;
        cin>>n;
        cin>>m;

        //initializing it to 0
        for(int i=0; i<1001; i++)
        {
            a[i]=0;
        }

        for(int i=0; i<m; i++)
        {
            cin>>input;
            a[input]=1;
        }

        for(int i=1; i<=n; i++)
        {
            if(a[i]==0)
            {
               if(turn)
                    v1.push_back(i);
               else
                    v2.push_back(i);
               turn=!turn;
            }
        }
        for(int i=0; i<v1.size(); i++)
            cout<<v1[i]<<" ";
        cout<<endl;
        for(int i=0; i<v2.size(); i++)
            cout<<v2[i]<<" ";

        if(v1.size()==0)
            cout<<endl;
        if(v2.size()==0)
            cout<<endl;

        v1.clear();
        v2.clear();
        t--;
    }

    return 0;
}
