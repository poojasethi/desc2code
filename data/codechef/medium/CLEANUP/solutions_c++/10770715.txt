#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,t,n,m,ch,as,k,r;
    cin>>t;
    while(t--)
    {
        k=0,as=0,ch=0,r=2;
        cin>>n>>m;
        int ar[m];
        for(i=0;i<m;i++)
        cin>>ar[i];
        int c[n-m],a[n-m];
        sort(ar,ar+m);
        for(i=1;i<=n;i++)
        {
            if(i==ar[k])
            k++;
            else if(r%2==0)
            {
                c[ch]=i;
                ch++;
                r++;
            }
            else
            {
                a[as]=i;
                as++;
                r++;
            }
        }
        for(i=0;i<ch;i++)
        cout<<c[i]<<" ";
        cout<<endl;
        for(i=0;i<as;i++)
        cout<<a[i]<<" ";
        cout<<endl;
    }
    return 0;
}
