#include <stdio.h>
#include <vector>
#include <cstring>
using namespace std;

int main() {

	int T,n,m,iCompletedJobIndex;

	scanf("%d",&T);

	while(T--) {

		scanf("%d %d",&n,&m);
		bool bArray[n+1];
		memset(bArray,false,sizeof(bool)*(n+1));

		for(int i=0;i<m;i++) {
			scanf("%d",&iCompletedJobIndex);
			bArray[iCompletedJobIndex] = true;
		}

		vector<int> vChefJobs,vAssistantJobs;
		vector<int> *pvCur = &vChefJobs, *pvNext = &vAssistantJobs,*pvTmp;
		bool bAssignToChef = true;

		for(int i=1;i<=n;i++) {
			
			if(bArray[i] == true) {
				continue;
			}

			pvCur->push_back(i);

			pvTmp = pvCur;
			pvCur = pvNext;
			pvNext = pvTmp;
			
		}

		int iNumChefJobs = vChefJobs.size(),iNumAssistantJobs = vAssistantJobs.size();

		for(int i=0;i<iNumChefJobs;i++) {
			printf("%d ",vChefJobs[i]);
		}
		printf("\n");
		for(int i=0;i<iNumAssistantJobs;i++) {
			printf("%d ",vAssistantJobs[i]);
		}
		printf("\n");



	}

	return 0;
	
}
