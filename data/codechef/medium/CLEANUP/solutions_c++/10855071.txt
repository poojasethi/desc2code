#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int display(int* line, int size);


int main() {
	int array[1001],line1[1001],line2[1001];
	int t, n,m,v,flag,l1,l2;
	scanf("%d",&t);
	int i;

	
	while(t){
		scanf("%d %d",&n,&m);	
		memset(array,'\0',sizeof(int)*(n+1));
		
		//printf("%d %d %d\n",t,n,m);
		while(m){
			scanf("%d",&v);
			array[v] = 1;
			--m;	
		}
		l1 = l2 =0;	
		flag = 1;	
		for(i=1;i<=n;i++){
			if(array[i]==0){
				if(flag){
					line1[l1++] = i;
					flag = 0;
				}else{
					line2[l2++] = i;
					flag = 1;
				}
			}
		
		}
		display(line1,l1);
		display(line2,l2);
		--t;
	}
	   
	return 0;
}

int display(int* line, int size){
	for(int i=0;i<size;i++){
		printf("%d ",line[i]);
	}
	printf("\n");
}
