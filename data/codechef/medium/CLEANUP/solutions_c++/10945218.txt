#include<iostream>
using namespace std;
#define MAX 1001
int main()
{
	int t,n,m;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		int arr[MAX]={0};
		for(int i=0;i<m;i++)
		{
			int t; cin>>t;
			arr[t]=1;
		}
		bool f=true;
		for(int i=1;i<=n;i++)
		{
			if(arr[i]==0) 
			{
				if(f) cout<<i<<" ";
				f=!f;
			}
		}
		cout<<endl;
		f=false;
		for(int i=1;i<=n;i++)
		{
			if(arr[i]==0)
			{
				if(f) cout<<i<<" ";
				f=!f;
			}
		}
	}
}
