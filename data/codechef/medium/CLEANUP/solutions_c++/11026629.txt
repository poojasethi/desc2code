#include<bits/stdc++.h>
using namespace std;
int flag=1;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int m,n,j=0,k=0;
		cin>>m>>n;
		long int chef[(m/2)+1],ass[(m/2)+1];
		long int a[m],b[n];
		for(int ran=0;ran<m;ran++)
		a[ran]=0;
		for(int i=0;i<n;i++)
		{
			cin>>b[i];
	
			a[(b[i]-1)]=1;
		}
		for(int i=0;i<m;i++)
		{
			if(flag==1)
			{
			if(a[i]!=1)
			{
				chef[j]=i+1;
				j++;		
				flag=0;
			}
			}
			else if(flag==0)
			{
			if(a[i]!=1)
			{
				ass[k]=i+1;
				k++;
				flag=1;
			}
			}
		}
		for(int p=0;p<j;p++)
		cout<<chef[p]<<" ";
		cout<<"\n";
		for(int q=0;q<k;q++)
		cout<<ass[q]<<" ";
		cout<<"\n";
	flag=1;
	}
}
