#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
	int test;
	cin>>test;
	while(test--){
		int n,m;
		cin>>n>>m; // 6 3
		// 2 1 4
		
		vector<int> vec;
		vector<int> chef;
		vector<int> asnt;
		for(int loop=0;loop<m;loop++){ 
			int a;
			cin>>a;
			vec.push_back(a);	
		}
		
		int a=0;
		for(int i=0;i<n;i++){
			int x=i+1;
			if(find(vec.begin(),vec.end(),x)==vec.end()){
				if(a%2==0){
				chef.push_back(x);
				a++;	
				}else{
					asnt.push_back(x);
					a++;
				}
			}		
		}
		
		for(int i=0;i<chef.size();i++){
			cout<<chef[i]<<" ";
		}

		cout<<endl;
		if(asnt.size()==0){
			cout<<endl;
		}
			
		for(int i=0;i<asnt.size();i++){
			cout<<asnt[i]<<" ";
		}
		
	}
	
	
	
	return 0;
}
