#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	long int m,n,t,i,a[1000],c[1000],as[1000],k,ci,asi;
	cin>>t;
	while(t--)
    {
	    cin>>n>>m;
	    for(i=0;i<m;i++)
	    cin>>a[i];
	    sort(a,a+m);
	    k=0;
	    ci=0;
	    asi=0;
	    for(i=1;i<=n;i++)
	    {
	        if(k<m)
	        {
	        if(i!=a[k])
	        {
	            if(ci==asi)
	            c[ci++]=i;
	            else
	            as[asi++]=i;
	        }
	        else
	        k++;
	        }
	        else
	        {
	            if(ci==asi)
	            c[ci++]=i;
	            else
	            as[asi++]=i;
	        }
	    }
	    for(i=0;i<ci;i++)
	    cout<<c[i]<<" ";
	    cout<<endl;
	    for(i=0;i<asi;i++)
	    cout<<as[i]<<" ";
	    cout<<endl;
    }
	
	return 0;
}
