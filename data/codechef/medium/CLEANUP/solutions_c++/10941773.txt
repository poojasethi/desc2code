#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
	int t,n,m;
	cin>>t;
	for(int i=0;i<t;i++){
		cin>>n>>m;
		vector<int> v;int l;
		for(int j=1;j<=n;j++){
			v.push_back(j);
		}
		for(int j=0;j<m;j++){
			cin>>l;
			v.erase(std::remove(v.begin(),v.end(),l),v.end());
		}
		int siz = v.size();
		if(siz>=1){
			for(int ii=0;ii<siz;ii+=2){
				cout<<v[ii]<<" ";
			}
		}
		cout<<"\n";
		if(siz>1){
			for(int ii=1;ii<siz;ii+=2){
				cout<<v[ii]<<" ";
			}
		}
		cout<<"\n";
	}
}
