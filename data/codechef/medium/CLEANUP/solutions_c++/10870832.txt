#include<iostream>
using namespace std;
int main()
  { int t,m,n,k,flag=1,arr[1000];
    cin>>t;
    while (t--)
    { cin>>m>>n;
      for (int j=0;j<1000;j++)
        arr[j]=1;
      for (int i=0;i<n;i++)
         { cin>>k;
           arr[k-1]=0;
         }
      k=0; flag=1;
      while (k<m)
      { if (arr[k]!=0)
           { if (flag%2!=0)
               { cout<<k+1<<" ";
                 arr[k]=0;
               }
             ++flag;
           }
        ++k;
      }
      cout<<endl;
      k=0;
      while (k<m)
      { if (arr[k]!=0)
           { cout<<k+1<<" ";
             arr[k]=0;
           }
        ++k;
      }
      cout<<endl;
    }
    return 0;
  }
