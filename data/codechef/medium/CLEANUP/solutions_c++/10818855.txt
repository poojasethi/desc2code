#include <iostream>
#include <stdlib.h>
#include <map>
#include <string.h>
using namespace std;


int main ()
{
	short int t, flag = 1;
	int n, m, j;
	cin>>t;
	
	while (t--)
	{
		cin>>n>>m;
		int arr[n+1];
		memset (arr, 0, sizeof (arr));
		
		for (int i = 1; i <= m; i++)
		{
			cin>>j;
			arr[j] = 1;
		}
		for (int i = 1; i <= n; i++)
		{
			if (arr[i] == 0 && flag == 1)
			{
				cout<<i<<" ";
				arr[i] = 1;
				flag = 0;
			}
			else if (arr[i] == 0)
			  flag = 1;
		}
		cout<<"\n"; flag = 1;
		for (int i = 1; i <= n; i++)
		{
			if (arr[i] == 0)
			  cout<<i<<" ";
		}
		cout<<"\n";
	}
}
