#include <cstdio>
#include <cstring>

int main()
{
    int t, n, m, i;
    bool a[1001];
    scanf("%d", &t);

    while (t--)
    {
        scanf("%d %d", &n, &m);
        memset(a, 0, sizeof(a));
        while (m--)
        {
            scanf("%d", &i);
            a[i] = true;
        }
        m = 1;
        for (i = 1; i <= n; i++)
        {
            if (!a[i])
            {
                if (m)
                {
                    printf("%d ", i);
                    m = 0;
                    a[i] = true;
                }
                else
                    m = 1;
            }
        }
        printf("\n");
        for (i = 1; i <= n; i++)
            if (!a[i])
                printf("%d ", i);
        printf("\n");
    }
    return 0;
}
