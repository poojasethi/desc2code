#include <iostream>
using namespace std;

int main() {
	int m, n, x;
	int t;
	cin >> t;
	while(t--) {
		cin >> n >> m;
		int jobs[n+1];
		for(int i=1; i<n+1; jobs[i]=0, i++);
		if(m != n || m != 0) {
			for(int i=0; i<m; i++) {
				cin >> x;
				jobs[x] = 1;
			}
		}
		int control = 1;
		for(int i=1; i<n+1; i++) {
			if(!jobs[i]) {
				control++;
				if(!(control%2)) {
					cout << i << " ";
					jobs[i] = 1;
				}
			}
		}
		cout << endl;
		for(int i=1; i<n+1; i++)
			if(jobs[i] == 0)
				cout << i << " ";
		cout << endl; 
	}
	return 0;
}
