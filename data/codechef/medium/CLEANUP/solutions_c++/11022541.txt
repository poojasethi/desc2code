#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    int i, n, m, t, j;
    cin >> t;
    for(i=0; i<t; i++){
        cin >> n;
        cin >> m;
        int arr[1000] = {0};
        int index;
        for(j=0; j<m; j++){
            cin >> index;
            arr[index-1] = 1;
        }
        int c1=0, c2=0, c=0;
        int chef[500] = {0};
        int assis[500] = {0};
        for(j=0; j<n; j++){
            if(arr[j]==0){
                if(c%2==0){
                    chef[c1] = j+1;
                    c1++;
                    c++;
                }
                else{
                    assis[c2] = j+1;
                    c2++;
                    c++;
                }
            }
        }
        for(j=0; j<500; j++){
            if(chef[j]!=0){
                printf("%d ", chef[j]);
            }
        }
        cout << endl;
        for(j=0; j<500; j++){
            if(assis[j]!=0){
                printf("%d ", assis[j]);
            }
        }
        cout << endl;
    }

    return 0;
}
