#include <bits/stdc++.h>

using namespace std;

int main(){
	int t, n, m;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d", &m, &n);
		int A[1007] = {0}, tmp;
		for(int i=0;i<n;i++){
			scanf("%d", &tmp);
			A[tmp] = 1;
		}
		int flag = 0;
		int chef[1007],in0 = 0, ass[1007], in1 = 0;
		for(int i=1;i<=m;i++){
			if(flag == 0 && A[i] == 0){
				chef[in0++] = i;
				flag = 1;
			}
			else
			if(flag == 1 && A[i] == 0){
				ass[in1++] = i;
				flag = 0;
			}
		}

		for(int i=0;i<in0 ;i++){
			printf("%d ",chef[i]);
		}
		printf("\n");

		for(int i=0;i<in1; i++){
			printf("%d ",ass[i]);
		}
		printf("\n");
	}

	return 0;
}
