import sys





if __name__ == '__main__':
    T = int(sys.stdin.readline())

    for i in range(T):
        n, m = [int(x) for x in sys.stdin.readline().split()]
        completed = [int(x) for x in sys.stdin.readline().split()]

        jobs = set(range(1, n + 1))
        remaining = sorted(list(jobs - set(completed)))
        
        print " ".join(str(x) for x in remaining[::2])
        print " ".join(str(x) for x in remaining[1::2])
