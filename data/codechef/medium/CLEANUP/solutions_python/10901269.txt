t = input()
while(t):
    dim  = map(int, raw_input().split())
    n = dim[0]
    m = dim[1]
    cooksList = map(int, raw_input().split())
    chef = True
    chefList = []
    assistantList = []
    for i in range(1,n+1):
        if i not in cooksList:
            if chef : 
                chefList.append(i)
                chef = False
            else:     
                assistantList.append(i)
                chef = True
    print ' '.join(map(str, chefList))
    print ' '.join(map(str,assistantList))
    t -= 1
