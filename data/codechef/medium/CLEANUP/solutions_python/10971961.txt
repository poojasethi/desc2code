def order(arr):
	noSwap = True
	while noSwap:
		noSwap = False
		for num in xrange(0,len(arr)-1):
			if arr[num]>arr[num+1]:
				noSwap = True
				numSwap = arr[num+1]
				arr[num+1] = arr[num]
				arr[num] = numSwap

def AssingJobs(numJobs, numDone, doneJobs):

	order(doneJobs)

	firstJob = 1
	doneJobsIndex = 0
	while doneJobsIndex<numDone and firstJob==doneJobs[doneJobsIndex]:
		firstJob += 1
		doneJobsIndex += 1

	cheffJobs = []
	helperJobs = []
	doneJobsIndex = 0
	tick = True

	for job in xrange(firstJob,numJobs+1):

		isDone = False
		
		while doneJobsIndex<numDone:
			if doneJobs[doneJobsIndex]==job:
				isDone = True			
			if job<doneJobs[doneJobsIndex]:
				break
			doneJobsIndex += 1

		if isDone:
			continue

		if tick:
			cheffJobs.append(job)
		else:
			helperJobs.append(job)

		tick = not tick
		

	# print "Total of jobs available %d" % numJobs
	# print "Total of jobs already done %d" % numDone
	# print "Especific jobs done %s" % doneJobs
	print "%s" % " ".join(map(str,cheffJobs))
	print "%s" % " ".join(map(str,helperJobs))
