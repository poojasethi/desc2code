t = int(raw_input())
while t > 0 :
    t -= 1
    n,m = map(int,raw_input().split())
    a = map(int,raw_input().split())
    chef = ''
    part = ''
    c = True
    for i in xrange(1,n+1) :
        if i not in a :
            if c :
                chef += str(i) + ' '
                c = False
            else :
                part += str(i) + ' '
                c = True
    print chef
    print part
