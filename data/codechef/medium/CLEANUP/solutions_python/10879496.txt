from __future__ import print_function
t = input();
while(t > 0):
	n, m = map(int, raw_input().split()); a = [];
	for i in range(1, n + 1):	a.append(i);
	b = [int(i) for i in raw_input().split()];
	for i in range(0, m):	a.remove(b[i]);
	la = len(a);
	for i in range(0, la):	
		if(i & 1 == 0):	print(a[i], end = " ");
	print(end = "\n");
	for i in range(0, la):	
		if(i & 1 != 0):	print(a[i], end = " ");
	print(end = "\n");
	t -= 1;
