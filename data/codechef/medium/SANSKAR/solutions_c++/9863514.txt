#include<iostream>
#include<math.h>
#include<cstdio>
#include<string>
#include<algorithm>
#define ll long long

using namespace std;

ll a[24],b[24],n,k;
ll solverec(ll csum,ll i,ll rsum,ll cnt){
if(i<n){
    if(!b[i]){
       if(csum+a[i]==rsum){
        if(cnt==k-1) return 1;
        b[i]=1;
        return solverec(0,0,rsum,cnt+1);
       }
       else if(csum+a[i]<rsum){
        b[i]=1;
        ll x=solverec(csum+a[i],i+1,rsum,cnt);
        b[i]=0;
        ll y=solverec(csum,i+1,rsum,cnt);
        return (x|y);
       }
       else return solverec(csum,i+1,rsum,cnt);
    }
    else return solverec(csum,i+1,rsum,cnt);
}
return 0;
}

int main(){
    int t;
    cin>>t;
    while(t--){
        int i,j;
        cin>>n>>k;

        ll sum=0;
        for(i=0;i<n;i++){
            cin>>a[i];
            b[i]=0;
            sum+=a[i];
        }
        if(sum%k!=0) cout<<"no"<<endl;
        else if(solverec(0,0,sum/k,0)) cout<<"yes"<<endl;
        else cout<<"no"<<endl;
    }
    return 0;
}
