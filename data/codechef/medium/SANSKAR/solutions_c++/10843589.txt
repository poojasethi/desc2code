#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k,j;
        cin>>n>>k;
        long long a[n],total=0,z=0;
        int used[n];
        memset(used,0,sizeof(used));

        for(int i=0;i<n;i++)
        {
            cin>>a[i];
            total+=a[i];
            if(a[i]==0)
                z++;
        }
        if(k>n)
        {
            cout<<"no"<<endl;
           // cout<<"k>n"<<endl;
            continue;
        }
        if(z==n)
        {
            cout<<"yes"<<endl;
          //  cout<<"z==n"<<endl;
            continue;
        }
        if(total%k!=0)
        {
            cout<<"no"<<endl;
           // cout<<"total%k"<<endl;
            continue;
        }
        long long avg=total/k,ans=0;
        for(int i=0;i<(1<<n);i++)
        {
            total=0;
            for(j=0;j<n;j++)
            {
                if((i&(1<<j))&&used[j]==0)
                {
                    total+=a[j];
                }
            }

            if(total==avg)
            {
                ans++;
                for( j=0;j<n;j++)
            {
                if(((1<<j)&i)&&used[j]==0)
                {
                    used[j]=1;
                }
            }
           // for(int i=0;i<n;i++)
             //   cout<<used[i]<<" ";

            }
        }
     //   cout<<ans<<endl;


        if(ans==k)
        {
            cout<<"yes"<<endl;
        }
        else
            cout<<"no"<<endl;


    }
}
