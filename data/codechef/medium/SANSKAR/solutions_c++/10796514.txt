ll powmod(ll a,ll b)
{
    if(b==0)return 1;
    ll x=powmod(a,b/2);
    ll y=(x*x)%MOD;
    if(b%2)
	return (a*y)%MOD;
    return y%MOD;
}

bool mark[N];
ll a[N];
int main()
{
    int tc;
    si(tc);
    while(tc--)
    {
	ll n,k;
	sll(n);
	sll(k);
	ll total=0;
	memset(mark,false,sizeof(mark));
	REP(i,0,n)
	{
	    sll(a[i]);
	    total+=a[i];
	}
	if(total%k!=0 || n<k) 
	{
	    printf("no\n");
	    continue;
	}
	if(total==0)
	{
	    printf("yes\n");
	    continue;
	}
	int curr=0;
	for(int mask=1;mask<(1<<n);mask++)
	{
	    ll sum=0;
	    if(curr==k)
		break;
	    for(int i=0;i<n;i++)
	    {
		if(!(mask&(1<<i)))
		    continue;
		if(mark[i])
		    continue;
		sum+=a[i];
	    }
	    if(sum==total/k)
	    {
		for(int i=0;i<n;i++)
		{
		    if(!(mask&(1<<i)))
			continue;
		    if(mark[i])
			continue;
		    mark[i]=true;
		}
		curr++;
	    }
	}
	if(curr==k)
	    printf("yes\n");
	else
	    printf("no\n");
    }
    return 0;
}
