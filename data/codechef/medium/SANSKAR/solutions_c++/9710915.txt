#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define INF 0x3f3f3f3f

using namespace std;
int dp[1<<22][10];
ll a[25];
int solve(int n,int K,ll x)
{
    memset(dp,0,sizeof(dp));
    int y=1<<n;
    dp[0][0]=1;
    for(int mask=0;mask<y;mask++)
    {
        for(int k=0;k<K;k++)
        {
            if(dp[mask][k]==0)
                continue;
            ll sum=0;
            for(int i=0;i<n;i++)
            {
                if(mask&(1<<i))
                    sum+=a[i];
            }
            sum-=k*x;
            for(int i=0;i<n;i++)
            {
                if(!(mask&(1<<i)))
                {
                    if(sum+a[i]==x)
                        dp[mask|(1<<i)][k+1]=1;
                    else if(sum+a[i]<x)
                        dp[mask|(1<<i)][k]=1;
                }
            }
        }
    }
    return dp[(1<<n)-1][K];
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k;
        scanf("%d%d",&n,&k);
        ll sum=0;
        for(int i=0;i<n;i++)
        {
            scanf("%lld",&a[i]);
            sum+=a[i];
        }
        if(sum%k!=0)
        {
            printf("no\n");
            continue;
        }
        if(k>n){
            printf("no\n");
            continue;
        }
        if(sum==0)
        {
            printf("yes\n");
            continue;
        }
        int ans=solve(n,k,sum/k);
        if(ans)
            printf("yes\n");
        else
            printf("no\n");
    }
    return 0;
}
