#include<bits/stdc++.h>
using namespace std;
#define LL int long long
LL A[22];
const int N=2097152;
int main()
{
	int i,j,k,comb;
	int n,K;
    int t;
    scanf("%d",&t);
    while(t--)
    {
      scanf("%d %d",&n,&K);
      LL tot=0;
      for(i=0;i<n;i++)
      {
       scanf("%lld",&A[i]);    
       tot+=A[i];
      }
           if(tot%K)
            {cout<<"no\n";
            continue;}
            if(tot==0&&n>K)
            {
                cout<<"yes\n";
                continue;
            }
            if(tot==0&&n<K)
            {
               cout<<"no\n";
               continue;
            }
      LL sub=tot/K;
      LL cnt=0;
      LL limi=(1<<n);
      for(comb=1;comb<limi;comb++)
      {
      	LL s=0;
      	for(j=0;j<n;j++)
      	{
      		if(comb&(1<<j))
      		{
              s+=A[j];
      		}
      	}
      	if(s==sub)
      	{
           cnt++;
           for(j=0;j<n;j++)
           	{   if(comb&(1<<j))
           		{
           			A[j]=0;
           		}
           	}
      	}
      	if(cnt==K)
      	{
      		break;
      	}
      }
      LL check=0;
      for(i=0;i<n;i++)
      {
        check+=A[i];
      }
      if(cnt==K && !check)
      {
      	printf("yes\n");
      }
      else
      {
      	printf("no\n");
      }
    }
    return 0;
}
