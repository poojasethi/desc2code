#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define INF 0x3f3f3f3f

using namespace std;
int n;
int dp[1<<22][10];
ll a[25],x;
int solve(int mask,int k,ll sum)
{
    if(mask==0)
    {
        if(k==0)
            return 1;
        return 0;
    }
    if(dp[mask][k]!=-1)
        return dp[mask][k];
    int ret=0;
    for(int i=0;i<n;i++)
    {
        if((mask&(1<<i)))
        {
            if(sum-a[i]==(k-1)*x)
                ret|=solve(mask^(1<<i),k-1,sum-a[i]);
            else if(sum-a[i]>(k-1)*x)
                ret|=solve(mask^(1<<i),k,sum-a[i]);
        }
    }
    return dp[mask][k]=ret;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int k;
        scanf("%d%d",&n,&k);
        ll sum=0;
        for(int i=0;i<n;i++){
            scanf("%lld",&a[i]);
            sum+=a[i];
        }
        if(sum%k!=0)
        {
            printf("no\n");
            continue;
        }
        if(sum==0)
        {
            if(k>n)
                printf("no\n");
            else
                printf("yes\n");
            continue;
        }
        if(k>n)
        {
            printf("no\n");
            continue;
        }
        x=sum/k;
        memset(dp,-1,sizeof(dp));
        int ans=solve((1<<n)-1,k,sum);
        if(ans==0)
            printf("no\n");
        else
            printf("yes\n");
    }
    return 0;
}
