#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int dp[10][(1<<21)+3];
int k;
int mx;
ll x;
ll a[22];
int n;
int solve(int pp,int msk,ll sum)
{
//	   cout<<pp<<" "<<msk<<" "<<sum<<" "<<k<<endl;
	   if(pp==k-1 && mx==msk && sum==x )
	   {
	   	return 1;
//	   	cout<<"here "<<endl;
	   	    
	   }  
	   if(sum>x)
	   return 0;
	   if(dp[pp][msk]!=-1)
	   return dp[pp][msk];
	   int ret=0;
	   for(int i=0;i<n;i++)
	   {
	   	        if((msk>>i&1)==0)
	   	        {
	   	        	     if(sum==x)
	   	        	     {
	   	        	       ret=ret|solve(pp+1,msk,0);
	   	        	      }
							else 
	   	        	     ret=ret|solve(pp,msk|(1<<i),a[i]+sum);
				}
	   }
	   dp[pp][msk]=ret;
	   return ret;
	   
}
int main()
{
	  int t;
	  cin>>t;
	  while(t--)
	  {
	  
	  cin>>n;
	   cin>>k;
	  ll sum=0;
	  memset(dp,-1,sizeof(dp));
	  for(int i=0;i<n;i++)
	  {
	  	  cin>>a[i];
	  	  sum+=a[i];
	  }
	  if(sum%k!=0)
	  {
	  	  printf("no\n");
	  }
	  else if(sum==0 && k<=n)
	  { 
	     printf("yes\n");
	  }
	  else if(k>n)
	  {
	  	  printf("no\n");
	  }
	  else
	  {
	  	   x=sum/k;
	  //	   cout<<x<<endl;
	  	   mx=(1<<n)-1;
	  	//   cout<<mx<<endl;
	  	   int res=solve(0,0,0);
	  	   if(res)
	  	   {
	  	   	  printf("yes\n");
	  	   	  
		   }
			 else printf("no\n");
	  }
   
   }
   return 0;
}
