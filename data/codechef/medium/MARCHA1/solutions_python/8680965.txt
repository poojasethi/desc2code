def calc(idx, total, A, visited):
    if total == 0:
        return True
    if idx >= len(A):
        return False
    key = (idx, total)
    if key in visited:
        return visited[key]
    result = calc(idx + 1, total - A[idx], A, visited) or calc(idx + 1, total, A, visited)
    visited[key] = result
    return result

def main():
    n, total = map(int, raw_input().split())
    A = []
    for i in range(n):
        A.append(int(raw_input()))
    visited = {}
    print 'Yes' if calc(0, total, A, visited) else 'No'
    
if __name__ == '__main__':
    testCase = int(raw_input())
    for test in range(testCase):
        main()
