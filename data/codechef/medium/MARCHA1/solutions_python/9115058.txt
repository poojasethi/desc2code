from itertools import combinations
for _ in xrange(int(raw_input())):
    n, m = map(int, raw_input().split())
    d = [int(raw_input()) for _ in range(n)]
    d = [i for i in d if i <= m]
    ans = "No"
    found = 0
    for i in range(1, n + 1):
        for c in combinations(d, i):
            if sum(c) == m:
                ans = "Yes"
                found = 1
                break
        if found: break
    print ans
