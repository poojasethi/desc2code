def isSum(i, noOfNotes, notes, total, moneyAsked):
	for x in range(i, noOfNotes):
		if total + notes[x] == moneyAsked:
			return 'Yes'
		elif total + notes[x] > moneyAsked:
			return 'No'
		else:
			if isSum(x+1, noOfNotes, notes, total + notes[x], moneyAsked) == 'Yes':
				return 'Yes'
	return 'No'
