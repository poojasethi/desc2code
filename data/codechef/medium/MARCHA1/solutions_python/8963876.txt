#!/usr/bin/env python
import sys

def intToIndicies(n):
    i = 0
    ret = []

    while n != 0:
        if n % 2 == 1:
            ret.append(i)
        i += 1
        n = n / 2

    return ret

def sumValues(vals, indicies):
    ret = 0
    for i in indicies:
        ret += vals[i]
    return ret

def go(amount, notes):
    # skip all the notes that are larger than the amount asked for
    # if any of them are equal to the amount asked for, this obvious just works
    validNotes = []
    for note in notes:
        if note == amount:
            return True
        if note < amount:
            validNotes.append(note)

    for n in xrange(1, 2 ** len(validNotes)):
        if sumValues(validNotes, intToIndicies(n)) == amount:
            return True
    return False

if __name__ == '__main__':
    T = int(sys.stdin.readline().strip())
    for t in xrange(0, T):
        N, M = map(int, sys.stdin.readline().strip().split())
        A = [0] * N
        for n in xrange(0, N):
            A[n] = int(sys.stdin.readline().strip())
        if go(M, A):
            print "Yes"
        else:
            print "No"
