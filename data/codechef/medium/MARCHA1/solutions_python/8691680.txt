t=input()
for i in range(0,t):
    b=0
    a=raw_input().split()
    n,m=int(a[0]),int(a[1])
    notes=[]
    for j in range(0,n):
        notes=notes+[input()]
    for j in range(1,(2**n)):
        a=str(bin(j))[2:]
        a=('0'*(n-len(a)))+a
        sum1=0
        for k in range(0,n):
            if(a[k]=='1'):
                sum1=sum1+notes[k]
        if sum1==m:
            print "Yes"
            b=1
            break
    if (b==0):
        print "No"
