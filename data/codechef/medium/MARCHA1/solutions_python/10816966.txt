for t in range(input()):
    n,m = map(int,raw_input().split())
    arr = []
    for i in range(n):
        x = input()
        if x<=m:
            arr.append(x)
    val = [False]*(m+1)
    val[0] = True
    for i in arr:
        val2 = []
        for j in val:
            val2.append(j)
        #print val,val2
        for j in range(i,m+1):
            if val2[j-i]==True:
                val[j]=True
#        print val
    if val[-1]==True:
        print 'Yes'
    else:
        print 'No'
