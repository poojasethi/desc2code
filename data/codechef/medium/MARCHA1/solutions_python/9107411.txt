import itertools
for testcases in xrange(int(raw_input())):
    n, tot = map(int, raw_input().split() )
    L = []
    for i in xrange(n) :
        L.append(int(raw_input()))
    L = sorted(L)
    purse = []
    for item in L:
        if item > tot:
            break
        purse.append(item)
    flag = 0
    for i in xrange(1, len(purse)+1):
        if flag == 0:
            for Arr in itertools.combinations(purse, i) :
                if sum(Arr) == tot :
                    flag = 1
                    break
    print 'Yes' if flag == 1 else 'No'
