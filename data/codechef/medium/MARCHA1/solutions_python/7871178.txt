for t in range(int(raw_input())):
    (n, m) = (int(x) for x in raw_input().split())
    coins = [int(raw_input()) for i in range(n)]
    u = [0] * (m + 1)
    u[0] = 1
    for coin in coins:
        for x in range(m, 0, -1):
            if x - coin >= 0 and u[x - coin] == 1:
                u[x] = 1
    print 'Yes' if u[m] == 1 else 'No'
