#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int a[21];
int fin(int d,int b,int c){
	int flag=0,sum;
	for(int i=d;i<c;i++){
		sum = b-a[i];
		if(sum==0)
			return 1;
		else if(sum>0){
			flag = fin(i+1,sum,c);
			if(flag)
				return 1;
		}
	}
	return 0;
}
int main(){
	int t,m,n;
	cin>>t;
	for(int i=0;i<t;i++){
		cin>>m>>n;
		for(int j=0;j<m;j++){
			cin>>a[j];
		}
		sort(a,a+m);
		int flag=0;
		flag = fin(0,n,m);
		if(flag)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	return 0;
}
