#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m,sum,i;
        cin>>n>>m;
        vector<int> A(n);
        for(int i=0;i<n;i++)
            cin>>A[i];
        for(i=0;i<pow(2,n);i++)
        {
            sum=0;
           for(int j=0;j<n;j++)
           {
               if(i&(1<<j))
               {
                   sum+=A[j];


               }
           }
           if(sum==m)
            {cout<<"Yes"<<endl;
           break;}
        }
        if(i==pow(2,n))
            cout<<"No"<<endl;
    }
    return 0;
}
