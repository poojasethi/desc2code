#include<iostream>
#include<stdio.h>
using namespace std;

bool isSubset(int *a, int n, int sum){

    bool subset[sum+1][n+1];
    for(int i=0;i<=n;i++)
    {
        subset[0][i] = true;

    }
    for(int i=1;i<=sum;i++)
    {
        subset[i][0] = false;
    }
    for(int i=1;i<=sum;i++){
        for(int j=1;j<=n;j++){
            subset[i][j] = subset[i][j-1];
            if(i>=a[j-1]){
                subset[i][j] = subset[i][j] || subset[i-a[j-1]][j-1];
            }

        }
    }
    return subset[sum][n];

}

int main()
{

    int t;
    cin>>t;
    while(t--)
    {
        int n,m;
        cin>>n>>m;
        int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        if(isSubset(a,n,m)){
            cout<<"Yes\n";
        }
        else {
            cout<<"No\n";
        }




    }


}
