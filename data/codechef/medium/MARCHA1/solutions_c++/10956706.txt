#include <iostream>
#include <math.h>
using namespace std;
int main()
{
	int t,n,m;
	cin>>t;
	for(int x=0;x<t;x++) 
        { 		
        	cin>>n>>m; int a[n]; int num=0;
        	for(int i=0;i<n;i++) cin>>a[i];
	
	for(int j=1;j<pow(2,n);j++)
			{
					int p=j;int sum=0;
						int l=1;
						 for(int y=0;y<n;y++)
								{
									int k=p%2;
										sum+=a[n-l]*k;
									p/=2;
								l++;
							}
				
			if(sum==m) num++;
		}
				if(num>0) cout<<"Yes"<<endl;
						else cout<<"No"<<endl;
		}
	return 0;
}
