#include <stdio.h>
#include <string>
#include <vector>
using namespace std;
 
int main(){
	int t;
	for(scanf("%d",&t);t>0;t--){
		int n,m,b;
		scanf("%d%d",&n,&m);
		vector<bool> v(20001,false);
		v[0]=true;
		for(int i=0;i<n;i++){
			scanf("%d",&b);
			for(int j=m;j>=0;j--){
				if(v[j])v[j+b]=true;
			}
		}
		if(v[m])printf("Yes\n");
		else printf("No\n");
	}
	return 0;
}
