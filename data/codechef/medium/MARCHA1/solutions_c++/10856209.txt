#include <iostream>
#include <bits/stdc++.h>
using namespace std;
 
int main() {
    
    int t;
    cin>>t;
    while(t--) {
        int n,m,a[100000],i,j,sum,till,flag=0;
        cin>>n>>m;
        for(i=0;i<n;i++) cin>>a[i];
        till = pow(2,n);
        for(i=1; i < till; i++) 
        {
            sum=0;
            for(j=0;j<n;j++)
            {
                if((i & (1<<j))!=0 )
                sum=sum+a[j];
            }
            
            if(sum==m)
            {flag=1; break;}
        }
        
        if(flag==1) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
        
    }
            
	return 0;
}
