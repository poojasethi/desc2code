#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n,m;
		scanf("%d%d",&n,&m);
		int a[n];
		for (int i = 0; i < n; i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		int dp[m+1];
		memset(dp,0,sizeof(dp));
		dp[0] = 1;
		for (int i = 0; i < n; i++)
		{
			for (int j = m; j >= 0; j--)
			{
				if (dp[j] == 1)
				{
					if ((j+a[i]) <= m)
						dp[j+a[i]] = 1;
				}
			}
		}
		if (dp[m] == 1)
			cout << "Yes\n";
		else
			cout << "No\n";
	}
	return 0;
}
