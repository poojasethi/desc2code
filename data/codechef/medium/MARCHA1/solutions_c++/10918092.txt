#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int t,n,m,i,j,d[29],sum,c;
    cin>>t;
    while(t--) { c=0;
        cin>>n>>m;
        for(i=0;i<n;i++) cin>>d[i];
        for(i=1;i<pow(2,n);i++) {
            sum=0;
            for(j=0;j<n;j++) {
                if(i & (1<<j)) {
                   sum = sum + d[j] ; } }
            if(sum==m) {
                c=1; break;
            }
        }
        if(c==1) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
    return 0;
}
