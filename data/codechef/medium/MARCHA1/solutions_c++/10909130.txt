#include<bits/stdc++.h>
using namespace std;
#define ll long long
int a[100], m, n;

int main(){
	int i;
//	for(i=0; i<1000000000; i++);
//		cout<<"hurray\n";
	int t;
	cin>>t;
	while(t--){
		ll i, j, k, f=0, sum=0;
		memset(a, 0, sizeof a);
		cin>>n>>m;
		for(i=0; i<n; i++){
			cin>>a[i];
		}
		for(i=0; i<(1<<n); i++){
			sum=0;
			for(j=0; j<n; j++){
				if(i&(1<<j)){
				sum+=a[j];}
			}
			if(sum==m){
				f=1;break;
			}
		}
		if(f)
		cout<<"Yes\n";else cout<<"No\n";
	}
}
