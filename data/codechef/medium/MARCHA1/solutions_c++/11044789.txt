#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

bool isSubsetSum(int Set[],int n,int sum)
{
    if(sum==0)
        return true;
    if(n==0 && sum!=0)
        return false;
    if(Set[n-1]>sum)
        return (isSubsetSum(Set,n-1,sum));
    return (isSubsetSum(Set,n-1,sum)||isSubsetSum(Set,n-1,sum-Set[n-1]));
}

int main()
{
        int t,i;
        cin >> t;
        while(t>0)
        {
            int n,m;
            cin >> n >>  m;
            int Set[n];
            for(i=0;i<n;i++)
                cin >> Set[i];
            if(isSubsetSum(Set,n,m)==true)
                printf("Yes\n");
            else
                printf("No\n");
            t--;
        }
    return 0;
}
