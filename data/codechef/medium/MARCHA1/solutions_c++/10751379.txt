#include <stdio.h>
#include <iostream>

using namespace std;

bool isSubsetSum(int notes[], int k, int M){
	if(M<0)
		return false;
	if(M==0)
		return true;
	if(k==0)
		return false;
	return isSubsetSum(notes,k-1,M-notes[k-1]) || isSubsetSum(notes,k-1,M);
}

int main(){
	int T;
	cin >> T;
	for(int t=0;t<T;t++){
		int N,M;
		cin >> N >> M;
		int notes[N];
		for(int n=0;n<N;n++)
			cin >> notes[n];
		if(isSubsetSum(notes,N,M))
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
	return 0;
	
}
