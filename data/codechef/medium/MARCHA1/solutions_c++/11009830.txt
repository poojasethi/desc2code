#include<iostream>
using namespace std;
 
bool isSub(int arr[],int n, int k){
	if(k==0) return true;
	if(n==0 && k!=0) return false;
	if(arr[n-1]>k)
		return isSub(arr,n-1,k);
	return isSub(arr,n-1,k) || isSub(arr,n-1,k-arr[n-1]);
}
int main(){
	int t,n,m; cin>>t;
	while(t--){
		cin>>n>>m;
		int a[n];
		for(int i=0;i<n;i++) cin>>a[i];
		if(isSub(a,n,m)) cout<<"Yes\n";
		else cout<<"No\n";
	}
	return 0;
}
