    int main(){
    	int t,n,m,i,notes[21];
    	scanf("%d",&t);
    	while(t--){
    		scanf("%d%d",&n,&m);
    		for(i=0;i<n;i++)
    			scanf("%d",&notes[i]);
    		notes[n]=-1;
    		printf(payup(notes,m)==1 ? "Yes\n" : "No\n");
    	}
    	return 0;
    }
    int payup(int *a, int m){
    	for(;*a!=-1;a++){
    		if(*a==m) return 1;
    		else if(*a<m) return payup(a+1,m-*a) || payup(a+1,m);
    	}
    	return 0;
    }
