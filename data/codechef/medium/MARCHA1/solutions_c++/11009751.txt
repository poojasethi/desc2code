#include<iostream>
using namespace std;

bool isSub(int arr[],int n, int k){
	bool dp[k+1][n+1];
	for(int i=0;i<=n;i++)
	    dp[0][i]=true;
	for(int i=1;i<=k;i++)
	    dp[i][0]=false;
	for(int i=1;i<=k;i++){
	    for(int j=1;j<=n;j++){
	        dp[i][j]=dp[i][j-1];
	        if(i>=arr[j-1])
	            dp[i][j] = dp[i][j] || dp[i-arr[j-1]][j-1];
	    }
	}
	return dp[k][n];
}
int main(){
	int t,n,m; cin>>t;
	while(t--){
		cin>>n>>m;
		int a[n];
		for(int i=0;i<n;i++) cin>>a[i];
		if(isSub(a,n,m)) cout<<"Yes\n";
		else cout<<"No\n";
	}
	return 0;
}
