#include<bits/stdc++.h>
using namespace std;
int main()
{
    int m,n,t,i,x,j,sum;
    cin>>t;
    while(t--)
    {
        x=0;
        cin>>n>>m;
        int ar[n];
        for(i=0;i<n;i++)
        cin>>ar[i];
        for(i=0;i<pow(2,n);i++)
        {
            sum=0;
            for(j=0;j<n;j++)
            {
                if(i & (1<<j))
                sum+=ar[j];
            }
            if(sum==m)
            {
                x=1;
                break;
            }
        }
        if(x)
        cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
}
