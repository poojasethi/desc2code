#include <iostream>
using namespace std;

bool mat[1005][1005];
int a[100000];
void func(int n,int sum)
{
    for(int i=0;i<=n;i++)mat[i][0]=1;
    for(int j=1;j<=sum;j++)mat[0][j]=0;
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=sum;j++)
        {
            //cout<<"dd"<<" ";
            mat[i][j]=mat[i-1][j];
            //cout<<a[i-1]<<" "<<j<<"\n";
            if(a[i-1]<=j)
            {
                mat[i][j]=(mat[i-1][j]||mat[i-1][j-a[i-1]]);
               // cout<<"polo\n";
            }
        }
        //cout<<"\n";
    }
    if(mat[n][sum]==true)
        cout<<"Yes\n";
    else
        cout<<"No\n";
}
int main()
{
    int t,n,sum;
    
    cin>>t;
    while(t--){
    cin>>n>>sum;
    for(int i=0;i<n;i++)cin>>a[i];
    func(n,sum);    
    }
    
}
