#include<bits/stdc++.h>
using namespace std;

bool isSumSubset(int a[],int n,int sum)
{
    bool subset[sum+1][n+1];
    for(int i=0;i<n;i++)
        subset[0][i]=true;
    for(int i=1;i<=sum;i++)
        subset[i][0]=false;
    for(int i=1;i<=sum;i++)
    {
        for(int j=1;j<=n;j++)
        {
            subset[i][j]=subset[i][j-1];
            if(i>=a[j-1])
               subset[i][j]=subset[i][j] || subset[i-a[j-1]][j-1];
        }
    }
    return subset[sum][n];
}

int main()
{
   int t;
   cin>>t;
   for(int i=0;i<t;i++)
   {
       int n,amt;
       cin>>n>>amt;
       int a[n];
       int sum=0;
       for(int i=0;i<n;i++)
       {
          cin>>a[i];
       }
       if(isSumSubset(a,n,amt))
          cout<<"Yes";
       else
        cout<<"No";
       cout<<endl;
   }

}
