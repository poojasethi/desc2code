#include <iostream>
using namespace std;
bool subset(int arr[],int n,int m)
{
	if(m==0) return true;
	if(n==0 && m!=0) return false;
	if(arr[n-1]>m) return subset(arr,n-1,m);
	else return subset(arr,n-1,m) || subset(arr,n-1,m-arr[n-1]);
}
int main()
{
	int n,m,t;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		int arr[n],i=0;
		for(;i<n;i++) cin>>arr[i];
		if(subset(arr,n,m)) cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}
	return 0;
}
