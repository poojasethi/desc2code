#include<iostream>
#include<algorithm>
using namespace std;

void calculate(int arr[],int notes,int money){
    bool dp[notes][money+1];
    for(int i=0;i<notes;++i)
        for(int j=0;j<=money;++j){
                if(j==0)
                {
                    dp[i][j]=true;
                }
                else if(i==0){
                    if(arr[i]==j)
                        dp[i][j]=true;
                    else
                        dp[i][j]=false;
                }
                else if(arr[i]>j){
                    dp[i][j]=dp[i-1][j];
                }
                else if(arr[i]<=j){
                    dp[i][j]=(dp[i-1][j]||dp[i-1][j-arr[i]]);
                }
                if(j==money&&dp[i][j]==true){
                    cout<<"Yes\n";
                    return;
                }
        }
        if(dp[notes-1][money]==true)
            cout<<"Yes\n";
        else
            cout<<"No\n";
}

int main(){
    int test;
    cin>>test;
    while(test--){
        int notes,money;
        cin>>notes>>money;
        int arr[notes];
        for(int i=0;i,i<notes;++i)
            cin>>arr[i];
        sort(arr,arr+notes);
        calculate(arr,notes,money);
    }
    return 0;
}
