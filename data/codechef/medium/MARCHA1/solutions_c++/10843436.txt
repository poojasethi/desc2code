#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int a[20],f=0;
void mug(int m,int pos);
int main()
{
    int t;
    scanf("%i",&t);
    while(t--)
    {
        f=0;
        int n,m;
        scanf("%i%i",&n,&m);
        for(int i=0;i<n;i++)
        {
            scanf("%i",&a[i]);
        }
        sort(a,a+n);
        for(int i=n-1;i>=0;i--)
        {
            if(f==1)
                break;
            mug(m,n-1);
        }
        if(f==1)
            printf("Yes\n");
        else
            printf("No\n");
    }
	return 0;
}
void mug(int m,int pos)
{
    for(int i=pos;i>=0;i--)
    {
        if(a[i]<m)
            mug(m-a[i],i-1);
        else if(a[i]==m)
        {
            f=1;
            break;
        }
    }
}
