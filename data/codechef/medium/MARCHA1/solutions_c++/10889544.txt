#include<bits/stdc++.h>
using namespace std;
int main()
{
	
	int t;
	cin>>t;
	while(t--)
	{
		int a[21];
		int i,j;
		long long m,n,k;
		cin>>n;
		cin>>m;
		int fl=0;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		
		long long sum=0;
		k=pow(2,n);
		for(i=0;i<k;i++)
		{
			sum=0;
			for(j=0;j<n;j++)
			{
				if(((i>>j)&1)==1)
				{
					sum+=a[j];
				}
			}
			if(sum==m)
			{
				cout<<"Yes\n";
				fl=1;
				break;
			}
		}
		if(fl==0)
		{
			cout<<"No\n";
		}
	}
	return 0;
}
