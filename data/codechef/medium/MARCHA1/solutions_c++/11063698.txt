#include<iostream>
#include<cmath>
using namespace std;
typedef long long int lli;
int main(){
	int t;
	cin >>t;
	while(t){
		lli x,y;
		cin >>x;
		cin>>y;
		lli p[x];
		for(int i=0;i<x;i++){
			cin >>p[i];
		}
		int bit=0;
		for(int i=1;i<=pow(2,x);i++){
			int sum=0;
			
			for(int j=0;j<x;j++){
				int tmp=i;
				if((tmp>>j)%2==1){
					sum+=p[j];
				}
				if(sum==y){
					bit=1;
					break;
				}
			}
		}
		if(bit==0){
			cout <<"No"<<endl;
		}else{
            cout<<"Yes"<<endl;
        }
		bit=0;
		t--;
	}
}
