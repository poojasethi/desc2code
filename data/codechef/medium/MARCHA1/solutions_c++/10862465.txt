#include<cstdio>
#include<algorithm>
using namespace std;
void subset(int index,int sum,int m,int& result,int n,int arr[]){
    if(sum<=m){
        if(index!=n-1){
            subset(index+1,sum+arr[index],m,result,n,arr);
            subset(index+1,sum,m,result,n,arr);
        }else{
            if(sum+arr[index]==m){
                result=1;
            }
            if(sum==m){
                result=1;
            }
        }
    }
}
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,m;
        scanf("%d %d",&n,&m);
        int arr[n];
        for(int i=0;i<n;i++){
            scanf("%d",&arr[i]);
        }
        sort(arr,arr+n);
        int result=0;
        subset(0,0,m,result,n,arr);
        if(result){
            printf("Yes\n");
        }
        else
        {
            printf("No\n");
        }
    }
}
