#include<iostream>
using namespace std;
int n,note[25];
int calc(int sum,int start)
{
	if(sum<0)
		return 0;

	if(sum==0)
		return 1;

	int flag=0,i;
	for(i=start;i<n;i++)
	{
		flag=calc(sum-note[i],i+1);
		if(flag==1)
			return 1;
	}
	return 0;
}
int main()
{
	int test,m;
	cin>>test;
	while(test!=0)
	{
		cin>>n>>m;
		int i;
		for(i=0;i<n;i++)
			cin>>note[i];
		int flag=0;
		for(i=0;i<n;i++)
		{
			flag=calc(m,i);
			if(flag==1)
				break;
		}
		if(flag==1)
			cout<<"Yes\n";

		else
			cout<<"No\n";
		test--;
	}
	return 0;
}
