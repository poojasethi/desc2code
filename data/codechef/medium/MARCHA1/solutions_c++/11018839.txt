#include<bits/stdc++.h>
using namespace std;
int a[1005];
bool solve(int a[],int n,int rs)
{
	int sum;
	int i,j;
	for(i=1;i<pow(2,n);i++)
	{
		sum=0;
		for(j=0;j<n;j++)
		{
			if(i&(1<<j))
			  sum+=a[j];
		}
		if(sum==rs)
		{
			return true;
		}
		
	}
	return false;
}
int main()
{
	int t,n,m,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		for(i=0;i<n;i++)cin>>a[i];
		bool ans=solve(a,n,m);
		if(ans)cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
		
	}
	
}
