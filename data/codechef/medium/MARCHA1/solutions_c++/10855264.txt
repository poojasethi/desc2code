#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

int pay(int, int);

int note[20];
int flag;

int main()
{
	int t, n, z, m, count, val;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		count=0;
		flag=1;
		for(int i=0;i<n;i++)
		{
			cin>>z;
			if(z<=m)
			{
				note[count]=z;
				count++;
			}
		}
		sort(note, note + count);
		count--;
		val=pay(m, count);
		if(val)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	return 0;
}

int pay(int x, int y)
{
//	cout<<"VAL REM:\t"<<x<<"\n";
//	cout<<"RANGE:\t"<<y<<"\n";
	int j, ret;
	if(y==0)
	{
//		cout<<"INSIDE 0\n";
		if(note[0]==x)
		{
//			cout<<"1 RET from 0\n";
			return 1;
		}
		else
			return 0;
	}
	for(int i=y;i>=0;i--)
	{
		if(note[i]==x)
		{
//			cout<<"VAL IS:\t"<<x<<"\n";
			return 1;
		}
		else
		{
//			cout<<"X:\t"<<x<<"\n";
//			cout<<"NOTE[i]:\t"<<note[i]<<"\n";
//			cout<<"REM SHOULD BE:\t"<<x-note[i]<<"\n";
			for(j=i-1;j>=0;j--)
			{
				ret=10;
				if(note[j]<=(x-note[i]))
				{
					ret=pay(x-note[i], j);
				}
				if(ret==1)
					return 1;
			}
		}
	}
	return 0;
}
