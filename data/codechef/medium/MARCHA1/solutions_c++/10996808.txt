//subset sum problem

#include <bits/stdc++.h>

using namespace std;


bool sub_sum(int A[], int arr[1000], int n,int m){
	arr[0] = 1;

	for(int i=0;i<n;i++){
		for(int j=m;j>=A[i];j--){
			arr[j] |= arr[j-A[i]];
		}
	}
	if(arr[m] == 1)
		return true;
	return false;
}



int main(){
	int t, n, m;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d",&n, &m);
		int A[n];
		for(int i=0;i<n;i++)
			scanf("%d",&A[i]);
		int arr[1007] = {0};
		if(sub_sum(A,arr,n,m))
			printf("Yes\n");
		else
			printf("No\n");
	}
}
