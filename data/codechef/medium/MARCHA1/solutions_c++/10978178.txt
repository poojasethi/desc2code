#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,m,i,len,x,sum,j;
    cin>>t;
    while(t--)
    {
        int flag=0;
        cin>>n>>m;
        int a[n];
        for(i=0; i<n; i++)
        {
            cin>>a[i];
        }
        len=pow(2,n);
        //cout<<len<<endl;
        for(i=0; i<len; i++)
        {
            if(flag==0)
            {
                x=1;
                sum=0;
                for(j=n-1; j>=0; j--)
                {
                    if((i&x)>0)
                    {
                        sum=sum+a[j];
                    }
                    x=x<<1;
                }
                if(sum==m)
                {
                flag=1;
                }
                }//flag=0
        }
         if(flag==1)
         {
         cout<<"Yes"<<endl;
         }
         else
         cout<<"No"<<endl;

    }//test case ends
}//main ends
