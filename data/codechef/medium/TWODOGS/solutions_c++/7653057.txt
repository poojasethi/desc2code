#include<bits/stdc++.h>
using namespace std;
#define maxi 1000010
#define inf 1000000
typedef long long ll;

int a[maxi]={inf};

int main()
{
	ios::sync_with_stdio(false);
	for(ll i=0;i<maxi;i++)
	{
		a[i]=inf;
	}
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++)
	{
		ll j;
		cin>>j;
		if(j<k)
		{
			a[j]=min(a[j],min(i+1,n-i));
		}
	}
	int ans=inf;
	for(int i=0;i<k/2;i++)
	{
		ans=min(ans,max(a[i],a[k-i]));
	}
	if(k&1)
	{
		ans=min(ans,max(a[k/2],a[k/2+1]));
	}
	if(ans==inf)
	{
		ans=-1;
	}
	cout<<ans<<"\n";
	return 0;
}
