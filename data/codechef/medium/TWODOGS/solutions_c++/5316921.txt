using namespace std;

#define ll long long
#define vt vector

int main()
{
	ll n,k,tmp;
	cin>>n>>k;
	vector<ll> vals(1e6+1,INT_MAX);
	for(ll i=1;i<=n;i++)
	{
		scanf("%lld",&tmp);
		vals[tmp]=min(vals[tmp],min(i,n-i+1));
	}
	ll ans=INT_MAX;
	for(ll i=1;i<=k/2;i++)
		if(i!=k-i)
		{
			if(vals[i]!=INT_MAX && vals[k-i]!=INT_MAX)
			{
				ans = min(ans,max(vals[i],vals[k-i]));
			}
		}
	if(ans==INT_MAX)
		cout << -1 << endl;
	else
		cout << ans << endl;
	return 0;
}
