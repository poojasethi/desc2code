#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define MAX 10000005
#define M 1000000007
using namespace std;

int d[1000005],a[500005];
bool b[1000005];
int main ()
{
	int n,k,d1,dmin,dmin1;
	sc(n);
	sc(k);
	for(int i=1;i<=1000005;i++)
	{
		d[i]=MAX;
	}
	dmin=MAX+1;
	for(int i=1;i<=n;i++)
	{
		sc(a[i]);
		b[a[i]]=1;
		d1=min(i,n+1-i);
		if(d1<d[a[i]])
		{
			d[a[i]]=d1;	
		}			
	//	cout<<d[a[i]]<<" ";
	}
	for(int i=1;i<=k/2+1;i++)
	{
		if(b[i]&&b[k-i]&&i!=k-i)
		{
			dmin1=max(d[i],d[k-i]);
			
			if(dmin1<dmin)
			{
				dmin=dmin1;
			}
		//	cout<<endl<<i<<" "<<k-i<<" "<<dmin<<endl;
		}
	}
	if(dmin!=MAX+1)
	{
		pf(dmin);
	}
	else
	{
		pf(-1);
	}
	return 0;
}
