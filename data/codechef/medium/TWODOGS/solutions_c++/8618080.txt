#include "bits/stdc++.h"
using namespace std;
typedef long long int ll;
const int SIZE = 5e5+5;
int N,K;
ll sum;
int arr[SIZE];
map<int,pair<pair<int,int>,int> > mtp;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int tt;
    int x;
    scanf("%d%d",&N,&K);
    for(int i = 0;i<N;i++){
            scanf("%d",&arr[i]);
            if(!mtp[arr[i]].second){
                    mtp[arr[i]].first.first = i;
                    mtp[arr[i]].first.second = i;
                    mtp[arr[i]].second+=1;
            }
            else{
                    mtp[arr[i]].first.second = i;
                    mtp[arr[i]].second+=1;
            }
    }
    int mini = INT_MAX;
    for(int i = 0;i<N;i++){
            if(arr[i]<K&&arr[i]!=(K/2)){
                    if(mtp[K-arr[i]].second){
                            mini = min(mini,max(min(i+1,N-i),min(mtp[K-arr[i]].first.first+1,N-mtp[K-arr[i]].first.second)));}
            }
    }
    mini==INT_MAX?printf("-1\n"):printf("%d\n",mini);
    mtp.clear();
return 0;
}
