using namespace std;
ll dis[1000001];
main()
{
	ll val,n,k,a,ans;
	scanf("%lld%lld",&n,&k);
	for(int i=0;i<=1000000;++i)
	dis[i]=10000000;
	for(ll i=0;i<n;++i)
	{
		scanf("%lld",&a);
		dis[a]=min(dis[a],i+1);
		dis[a]=min(dis[a],n-i);
	}
	ans=10000000;
	for(ll i=0;i<=k;++i)
	{
		if(i!=k-i)
		{
			val=max(dis[i],dis[k-i]);
			ans=min(ans,val);
		}
		//cout<<ans<<" "<<i<<endl;
	}
	if(ans==10000000)
	cout<<"-1";
	else
	cout<<ans;
}
