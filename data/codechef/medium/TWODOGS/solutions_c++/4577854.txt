#include <bits/stdc++.h>
using namespace std;
#define MAX 1000000
int inp[MAX+1];
vector <int> aux (MAX + 1, INT_MAX);
int main()
{
    int N, K, ans = INT_MAX;
    scanf ("%d %d", &N, &K);

    for (int i = 0; i < N; ++i)
    {
        scanf ("%d", &inp[i]);
        aux [inp [i]] = min (aux[inp[i]], min (1 + i, N - i));
    }

    for (int i = 0; i < N; ++i)
    {
        if (inp[i] != K - inp[i])
        {
            if (K - inp[i] >= 0 && aux [K - inp [i]] != INT_MAX)
            {
                ans = min (ans, max (aux [inp[i]], aux [K - inp[i]]));
            }
        }
    }

    if (ans == INT_MAX) printf ("-1\n");
    else printf ("%d\n", ans);
    return 0;
}
