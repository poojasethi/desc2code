#include <iostream>
#include<limits.h>
#include<map>
using namespace std;

int main() {
int m1=0,m2=INT_MAX,a[600000],i,n,k;
map<int,int>m;
cin>>n>>k;
for(i=0;i<n;++i)
{
 cin>>a[i];
 if(a[i]>=k)continue;
 if((a[i]==k/2)&&(k-a[i])==k/2)continue;
 m[a[i]]=m[a[i]]?min(m[a[i]],min(1+i,n-i)):min(1+i,n-i);
 if(m[k-a[i]])
 {
  m1=max(m[a[i]],m[k-a[i]]);
  m2=min(m1,m2);
 }
}
 m2=m2!=INT_MAX?m2:-1;
 cout<<m2<<endl;
return 0;
}
