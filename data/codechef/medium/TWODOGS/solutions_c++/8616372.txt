#include "bits/stdc++.h"
using namespace std;
typedef long long int ll;
const int SIZE = 5e5+5;
int N,K;
ll sum;
int arr[SIZE];
map<int,pair<int,int> > mtp;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int tt;
    int x;
    cin>>N>>K;
    for(int i = 0;i<N;i++){
            cin>>arr[i];
            if(!mtp[arr[i]].second){
                    mtp[arr[i]].first = min(i+1,N-i);
                    mtp[arr[i]].second+=1;
            }
            else{
                    mtp[arr[i]].first = min(min(i+1,N-i),mtp[arr[i]].first);
                    mtp[arr[i]].second+=1;
            }
    }
    int mini = INT_MAX;
    for(int i = 0;i<N;i++){
            if(arr[i]<K&&arr[i]!=K-arr[i]){
                    if(mtp[K-arr[i]].second){
                            mini = min(mini,max(mtp[arr[i]].first,mtp[K-arr[i]].first));}
            }
    }
    if(mini==INT_MAX)
    printf("-1\n");
    else
    printf("%d\n",mini);
    mtp.clear();
return 0;
}
