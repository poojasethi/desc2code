#include<iostream>
#include<stdlib.h>
#include<limits.h>
using namespace std;

struct apple
{
	int type;
	int index;
};
int comp(const void *a,const void *b)
{
	struct apple *x=(struct apple *)a;
	struct apple *y=(struct apple *)b;
	if(x->type>y->type)
	return 1;
	else
	if(x->type<y->type)
	return -1;
	else
	if(x->index>y->index)
	return 1;
	else
	return -1;
}
int max(int a,int b);
int main()
{
	int n,k,temp,i,j;
	
	cin>>n>>k;
	struct apple a[n+1];
	for(i=1;i<=n;i++)
	{
		cin>>a[i].type;
		a[i].index=min(i,n-i+1);
	}	
	
	qsort(a+1,n,sizeof(struct apple),comp);
	//for(i=1;i<=n;i++)
	//{
	//	cout<<a[i].type<<" ";
		//a[i].index=min(i,n-i+1);
	//}	
	//cout<<endl;
	i=1;
	j=n;
	int minn=INT_MAX;
	while(j>i)
	{
		if(a[j].type!=a[i].type && a[j].type+a[i].type==k)
		{
			while(a[j].type==a[j-1].type && j-1>i)
			j--;
			temp=max(a[i].index,a[j].index);
			if(temp<minn)
			minn=temp;
			
			while(a[i].type==a[i+1].type)
			i++;
			
			i++;
			j--;
		}
		else
		if(a[i].type==a[j].type)
		break;
		else
		if(a[i].type+a[j].type > k)
		{
			j--;
		}
		else
		if(a[i].type+a[j].type < k)
		{
			i++;
		}
	}
	if(minn==INT_MAX)
	cout<<"-1\n";
	else
	cout<<minn<<"\n";
}
int max(int a,int b)
{
	return a<b?b:a;
}
