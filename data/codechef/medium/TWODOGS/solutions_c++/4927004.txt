#include <iostream>
#include <stdio.h>
#include <string.h>

#define INT_MAX 0x70000000
using namespace std;
int Hash[1000000];
int data[500000];

int FindMinTime(int i,int j,int n)
{
	int min1 = min(i,n-i+1);
	int min2 = min(j,n-j+1);
	return max(min1 , min2) ;
}

int FindApples(int k,int n)
{
	int minTime = INT_MAX ;
	for(int i = 1 ;i <= n ;i++)
	{
		//if both elements are equal or the number can't contribute to sum.
		if( data[i] == (k - data[i]) || (k - data[i]) < 0)
			continue ;
		
		if( Hash[ data[i] ] == -1)
		{
			Hash[ data[i] ] = i ;
		}
		else
		{
			int t1 = min( i , n -i+1);
			int t2 = min( Hash[data[i]] ,n - Hash[data[i]]+1);
			if(t1 < t2)
				Hash[data[i]] = i ;
		}
		if( Hash[k-data[i]] != -1)
				minTime = min( minTime, FindMinTime(i,Hash[k-data[i]],n));
	}
	 if(minTime == INT_MAX)
		return -1;
	return minTime;
}

int main()
{
	//freopen("input.txt","r",stdin);
	int N, K;
	cin >> N;
	cin >> K ;
	Hash[0] = -1;
	memset(Hash,-1,sizeof(int)*1000000);
	for(int i=1;i<=N;i++)
	{
		//cin >> data[i];
		scanf("%d",&(data[i]));
		//Hash[data[i]] = -1;
	}
	int minTime = FindApples(K,N);
	cout << minTime << endl;
}
