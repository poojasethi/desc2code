inline int min(int a,int b){
	return a<b?a:b;
}
inline int max(int a,int b){
	return a>b?a:b;
}
int b[1000006];
int t[1000006];
int main(){
	int n,k;
	int i=0;
	scan(n);
	scan(k);
	memset(b,1000007,sizeof(b));
	rep(i,0,n){
		scan(t[i]);
	}
	rep(i,0,n){
		int val = t[i];
		int j = min(i+1,n-i);
		b[val] = min(j,b[val]);
	}
	int ans = 1000007;
	rep(i,0,n){
		int val = t[i];
		if(val != k - val && val<k){
			ans = min(ans,max(b[val],b[(k-val)])); 
		}
	}
	ans==1000007?printf("-1"):printf("%d",ans);
	return 0;
}
