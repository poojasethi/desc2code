#include<iostream>
#include<algorithm>
#include<cstdio>

#define INF 1e9
#define MAX 1000000

using namespace std;

// Input macros
#define s(n)               scanf("%d",&n)
#define sl(n)              scanf("%lld",&n)

int A[MAX + 1];
int B[MAX + 1];

int main()
{
   int N, K, ans = INF;
   s(N);s(K);

   for(int i=0; i < MAX + 1; i++)
        B[i] = INF;

   for(int i=0; i < N ; i++)
   {
       s(A[i]);
       int x = A[i];
       int d = min( i+1, N-i );
       B[x] = min( B[x], d );
   }

   /*for(int i=0; i < *max_element(A,A+N) + 1; i++)
      cout<<i<<" "<<B[i]<<endl;
    cout<<endl<<endl;*/

   for(int i=0; i < N ; i++)
   {
       //cout<<"i:"<<i<<" ";
       int x = A[i];
       //cout<<"x:"<<x<<" ";
       //cout<<B[K-x]<<" ";
       //cout<<B[x]<<" ";
       if( (x != K-x) && (x < K) )
          ans = min( ans, max( B[x], B[K-x] ) );
        //cout<<"ans:"<<ans<<endl;
   }

   if(ans != INF)
        cout<<ans<<endl;
   else
        cout<<-1<<endl;
   return 0;
}
