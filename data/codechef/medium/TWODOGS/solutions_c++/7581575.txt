#include<stdio.h>

using namespace std;
int A[1000000], B[1000000];
int mini(int a, int b)
{
    if (a<b)
        return a;
    else
        return b;
}

int maxi(int a, int b)
{
    if (a>b)
        return a;
    else
        return b;
}

int main()
{
    int N, K, step1, step2, time, minim=-1, found=0;
    scanf("%d%d", &N, &K);
    for(int i=0; i<N; i++)
        scanf("%d", &A[i]);

    for (int i=0; i<1000000; i++)
        B[i]=1000000000;

    for (int i=0; i<N; i++)
    {
        int x=A[i];
        int d=mini(1+i, N-i);
        B[x]=mini(B[x], d);
    }
    int ans=1000000000;
    for (int i=0; i<N; i++)
    {
        int x=A[i];
        if (x!=K-x && (K-x>=0))
            ans=mini(ans, maxi(B[x], B[K-x]));

    }
    if (ans==1000000000)
        printf("-1\n");
    else
        printf("%d\n", ans);
    return 0;
}
