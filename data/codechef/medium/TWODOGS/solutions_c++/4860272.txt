#include <iostream>
using namespace std;
# include<algorithm>

struct dog
{
    int apple;
    int time;
}arr[500001];

int b[1000002];

bool cmp(struct dog x,struct dog y)
{
    return (x.time<y.time);
}


int main()
{
    int n,k;
    cin>>n>>k;
    for(int i=0;i<n;++i)
    {
        cin>>arr[i].apple;
        if(i+1<n-i)
            arr[i].time=i+1;
        else
            arr[i].time=n-i;
    }
    sort(arr,arr+n,cmp);
    for(int i=0;i<n;++i)
        {
        if(b[arr[i].apple]==0&&k>arr[i].apple&&k-arr[i].apple!=arr[i].apple)
        {
        b[k-arr[i].apple]++;
        }
        else if(b[arr[i].apple]!=0&&k-arr[i].apple!=arr[i].apple)
        {
                cout<<arr[i].time;
                return 0;
        }
        }
    cout<<"-1";
    return 0;
}
