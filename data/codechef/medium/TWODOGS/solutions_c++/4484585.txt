#include<bits/stdc++.h>

using namespace std;

int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    map<int,int> map1;
    for(int i=0;i<n;i++)
    {
        int temp;
        scanf("%d",&temp);
        if(map1.find(temp)!=map1.end())
            map1[temp]=min(min(i+1,n-i),map1[temp]);
        else
            map1[temp]=min(i+1,n-i);
    }
    int min1=100000000;
    for(map<int,int>::iterator it=map1.begin();;it++)
    {
        map<int,int>::iterator it1=it,it2;
        it1++;
        if(it1==map1.end())
            break;
        int a1=it->first;
        int a2=k-a1;
        it2=map1.find(a2);
        if(it2!=map1.end() && it2->first != it->first )
        {
            int sum1=max((it->second) , it2->second);
            if(sum1<min1)
                min1=sum1;
        }

    }
    if(min1 < 100000000)
        printf("%d",min1);
    else
        printf("%d",-1);
    return 0;
}
