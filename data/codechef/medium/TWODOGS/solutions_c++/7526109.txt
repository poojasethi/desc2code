//nice question
#include"iostream"
#define MAX 1000000000
using namespace std;
int b[10000000],a[10000000];
int main()
{
		int i,k,n,d,ans,cur;	
		cin>>n>>k;
		for(i=0;i<n;++i)
		cin>>a[i];
		for(i=0;i<10000000;++i)
		b[i]=MAX;		
		for(i=0;i<n;++i)
		{
			d=min(i+1,n-i);
			if(d < b[a[i]])
			b[a[i]]=d;
		}
		ans=MAX;		
		for(i=0;i<n;++i)
		{
			if(a[i]!=k-a[i] && (k-a[i]>=0))
			{
				cur=min(ans,max(b[a[i]],b[k-a[i]]));				
				if(cur<ans)
				ans=cur;
			}
		}
		if(ans!=MAX)
		cout<<ans<<endl;
		else
		cout<<"-1\n";
	
	return 0;
}
