#include<stdio.h>
    #include<stdlib.h>
    int fastread()
    {
        int k;
        scanf("%d",&k);
        return k;
    }
    int main()
    {
    int flag=0,out=-1,n,k,i,r,*a,*count;
    n = fastread();
    k = fastread();
    a=(int *)malloc(sizeof(int)*n + 1);
    count=(int *)malloc(sizeof(int)*1000000);
    for(i=0;i<1000000;i++)
    count[i]=0;
    for(i=0;i<n;i++)
    a[i]=fastread();
    if(k==1||k==2)
    {
    printf("-1\n");
    return 0;
    }
    if(n%2==0)
    r=n/2;
    else
    r=n/2+1;
    for(i=0;i<r;i++)
    {
    if(a[i]<k)
    {
    if(count[k-a[i]]>0 && k!=2*a[i])
    {
    out=i+1;
    flag=1;
    }
    else count[a[i]]++;
    }
    if(a[n-i-1]<k)
    {
    if(count[k-a[n-1-i]] > 0 && k != 2*a[n-1-i])
    {
    out=i+1;
    flag=1;
    }
    else count[a[n-1-i]]++;
    }
    if(flag==1)
    break;
    }
    printf("%d\n",out);
    return 0;
    }
