inline ll scn()
{
	ll n = 0, c = getchar_unlocked(),t=0;
	while(c < '0' || c > '9')
	{
		if(c==45)t=1;
		c = getchar_unlocked();
	}
	while(c >= '0' && c <= '9')n = (n<<3) + (n<<1) + c - '0', c = getchar_unlocked();

	if(t==0)return n;
	else return -n;
}

struct firestyle
{
    l Mi;
    l Ma;
    l C;
};


firestyle lol[1000001];
main()
 {
   coolreshab;
   l N,K,i,temp,ans=LONG_MAX;
   N=scn();K=scn();
   FORN(i,N)
   {
        temp=scn();
       (lol[temp].C)++;
       if(lol[temp].C==1)
       {
           lol[temp].Mi=i+1;
           lol[temp].Ma=i+1;
       }
       else
       {
           lol[temp].Ma=i+1;

       }

   }

       //cout<<lol[i].C<<" "<<lol[i].Mi<<" "<<lol[i].Ma<<" "<<endl;

   FORAB(i,1,K/2)
   {
       if(i==(K-i))
       {
           continue;
       }

       if(lol[i].C and lol[K-i].C)
       {
           ans=MIN(ans, MAX( MIN(lol[i].Mi,N-lol[i].Ma+1),MIN(lol[K-i].Mi,N-lol[K-i].Ma+1) ) );
       }
   }
   if(ans==LONG_MAX)
    ans=-1;

   pl(ans);

 }
