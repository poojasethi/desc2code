#include<vector>
#define size 1000001
#define s(a) scanf("%lld",&a)
#define p(a) printf("%lld ",a)
#define ull long long int
#define pb push_back
#define FOR(i,i1,i2) for(i=i1;i<i2;i++)
#define INF 100000000
using namespace std;
ull min(ull a,ull b)
{
if(a<b)
return a;
else
return b;
}
ull max(ull a,ull b)
{
if(a>b)
return a;
else
return b;
}
int main()
{
vector<ull> v;
ull t,i,n,hash[size],ans1=1000001,ans2=1000001,ans=INF,check,k,num,flag=0;
s(n);
s(k);
FOR(i,0,size)
hash[i]=INF;
FOR(i,0,n)
{
s(num);
v.pb(num);
//if(hash[num]==0)
check=min(i+1,n-i);
//printf("check: %lld\n",check);
hash[num]=min(hash[num],check);
//p(hash[num]);
}
//FOR(i,0,k+1)
//p(hash[i]);
FOR(i,0,n)
{
if(v[i]<k)
{
if(hash[k-v[i]]!=INF && v[i]!=k-v[i])
{
// printf("%lld that\n",i);
ans1=min(hash[v[i]],n-hash[v[i]]+1);
//printf("ans %lld:%lld\n",i,ans1);
ans2=min(hash[k-v[i]],n-hash[k-v[i]]+1);

}
