#include<bits/stdc++.h>
#include<vector>
#define size 1000005
#define s(a) scanf("%lld",&a)
#define p(a) printf("%lld ",a)
#define ull long long int
#define pb push_back
#define FOR(i,i1,i2) for(i=i1;i<i2;i++)
#define INF 100000000
using namespace std;
ull min(ull a,ull b)
{
if(a<b)
return a;
else
return b;
}
ull max(ull a,ull b)
{
if(a>b)
return a;
else
return b;
}
int main()
{
vector<ull> v;
ull t,i,n,hash[size],ans=size,k,num,flag=0,check;
s(n);
s(k);
FOR(i,0,size)
hash[i]=INF;
FOR(i,0,n)
{
s(num);
v.pb(num);
if(hash[num]==INF)
hash[num]=min(i+1,n-i);
else
{
check=min(i+1,n-i);
hash[num]=min(hash[num],check);
}
}
/*FOR(i,0,k+1)
p(hash[i]);*/
FOR(i,0,n)
{
if(v[i]<k)
{
if(hash[k-v[i]]!=INF && v[i]!=k-v[i])
{
ans=min(ans,max(hash[v[i]],hash[k-v[i]]));
flag=1;
}
}
}
if(flag==0)
printf("-1");
else
p(ans);
return 0;
}
