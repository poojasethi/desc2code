#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main()
{
    ll x,y,k,n,t,i,p,c,s;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld%lld%lld",&x,&y,&k,&n);
        x=x-y;
        s=0;
        for(i=0;i<n;i++)
        {
            scanf("%lld%lld",&p,&c);
            if(p>=x && c<=k)
                s=1;
        }
        if(s==1)
            printf("LuckyChef\n");
        else
            printf("UnluckyChef\n");
    }
    return 0;
}
