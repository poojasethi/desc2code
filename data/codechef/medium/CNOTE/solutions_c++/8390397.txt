#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,x,y,k,n,p[100004],c[100004];
	scanf("%d",&t);
	while(t--){
		scanf("%d %d %d %d",&x,&y,&k,&n);
		for(int i=0;i<n;i++){
			scanf("%d %d",&p[i],&c[i]);
			
		}
		int i;
		for(i=0;i<n;i++){
			if(c[i]<=k && p[i]>=(x-y)){
				printf("LuckyChef\n");
				break;
			}
		}
		if(i==n){
			printf("UnluckyChef\n");
		}
	}
}
