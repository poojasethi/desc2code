#include <iostream>
using namespace std;

int main() {
	int t,x,y,n,k,i,flag=0;
	scanf("%d",&t);
	while(t--)
	{
	flag=0;
	scanf("%d %d %d %d",&x,&y,&k,&n);
	int a,b;
	for(i=0;i<n;i++)
	{
	scanf("%d %d",&a,&b);
	if((x-y<=a)&&(b<=k))
	{
	flag=1;
	}
	}
	if(flag==1)
	{
	printf("LuckyChef\n");
	}
	else
	{
	printf("UnluckyChef\n");
	}
	}
	return 0;
}
