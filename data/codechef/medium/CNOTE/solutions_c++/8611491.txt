#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int x,y,k,n,p[100001],c[100001],count=0;
		scanf("%d%d%d%d",&x,&y,&k,&n);
		for (int i = 0; i < n; ++i)
		{
			scanf("%d%d",&p[i],&c[i]);
		}
		for (int i = 0; i < n; ++i)
		{
			if (c[i]<=k && p[i]>=x-y)
			{
			count+=1;
			break;
			}
		}
		if (count>=1)
		{
			printf("LuckyChef\n");
		}
		else
			printf("UnluckyChef\n");
	}
	return 0;
}
