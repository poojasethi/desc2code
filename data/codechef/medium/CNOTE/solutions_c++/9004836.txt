#include<bits/stdc++.h>
using namespace std;
#define getchar_unlocked() getchar()
#define debug(x) cout << "[DEBUG] " << #x << " = " << x << "\n"
inline int inp()
{
    register int r=0,c;
    for(c=getchar_unlocked(); c<=32; c=getchar_unlocked());
    if(c=='-')
        return -inp();
    for(; c>32; r=(r<<1)+(r<<3)+c-'0',c=getchar_unlocked());
    return r;
}
int main()
{
	int t,x,y,k,n,diff,f,p,c;
	cin>>t;
	while(t--)
	{
	x=inp();
	y=inp();
	k=inp();
	n=inp();
		diff=x-y;
		f=0;
		while(n--)
		{
			p=inp();
			c=inp();
			if(diff<=p && k>=c)
			f=1;
		}
		if(f==1)
		cout<<"LuckyChef"<<endl;
		else
		cout<<"UnluckyChef"<<endl;
	}
	return 0;
	}
