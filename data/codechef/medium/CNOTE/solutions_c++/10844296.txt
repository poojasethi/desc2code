#include <iostream>
#include <bits/stdc++.h>

typedef long long int ll;
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int x,y,k,n,f=0;
		cin>>x>>y>>k>>n;
		int diff = x-y;
		for(int i=0;i<n;i++)
		{
			int a,b;
			cin>>a>>b;
			if(!f && b<=k&&a>=diff)
				f = 1;
		}
		if(f)
			cout<<"LuckyChef"<<endl;
		else
			cout<<"UnluckyChef"<<endl;
	}	
	return 0;
}
