#include<iostream>
#include<stdio.h>
using namespace std;
 int t,l,y,x,k,n,p[100005],c[100005],i;
int main()
{
   
    scanf("%d",&t);
    while(t--)
    {
        l=0;
        scanf("%d%d%d%d",&x,&y,&k,&n);
        for(i=0;i<n;i++)
        {
           scanf("%d%d",&p[i],&c[i]);
           
        }
        for(i=0;i<n;i++)
         if((x-y)<=p[i]&&k>=c[i]&&l==0)
            {
                printf("LuckyChef\n");
                l++;
                 break;
            }
        if(l==0)
            printf("UnluckyChef\n");
    }
    return 0;
}
