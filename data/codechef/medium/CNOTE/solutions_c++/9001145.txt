#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int x,y,n,k,i,a,b,z,flag=1;
		scanf("%lld%lld%lld%lld",&x,&y,&k,&n);
		z=x-y;
		for(i=0;i<n;i++)
		{
			scanf("%lld%lld",&a,&b);
			if(a>=z&&k>=b)
			{
				flag=0;
				
			}
		}
		if(flag==0)
		{
			printf("LuckyChef\n");
			flag=1;
		}
		else
		printf("UnluckyChef\n");

		
	}
	return 0;
}
