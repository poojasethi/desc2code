#include<stdio.h>
int main()
{
   long t;
   scanf("%ld",&t);
   while(t--)
   {
       int x,y,k;
       int flag=0;
       long n,i;
       int p[110000],c[110000];
       scanf("%d %d %d %ld",&x,&y,&k,&n);
       for(i=0;i<n;i++)
       {
           scanf("%d %d",&p[i],&c[i]);
        }
         for(i=0;i<n;i++)
       {
          if(p[i]>= x-y && c[i]<=k)
           {
               flag=1;
               break;
           }
       }
       if(flag==1)
        printf("\nLuckyChef");
       else
        printf("\nUnluckyChef");
 
 
   }
   return 0;
}
