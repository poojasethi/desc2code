#include <iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
 
using namespace std;
 
int main()
{   
    int t,x,y,n,k;
    cin>>t;
    while(t--)
    {
    scanf("%d%d%d%d",&x,&y,&k,&n);
    int c[n],p[n],flag=1;
    int amt=x-y;
    

    for(int i=0;i<n;i++)
   { scanf("%d%d",&p[i],&c[i]);
   if(p[i]>=amt&&c[i]<=k)
   flag=0;
   }
   
    if(flag==1)
    cout<<"UnluckyChef"<<endl;
    else cout<<"LuckyChef"<<endl;
    
    }	
	return 0;
}
