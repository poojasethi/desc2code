	#include<iostream>
	#include<stdio.h>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			int x,y,k,n;
			cin>>x>>y>>k>>n;
			int pagerequire=x-y;
			int flag=0;
			int num1,num2;
			for(int i=0;i<n;i++)
			{
				scanf("%d%d",&num1,&num2);
				if(num1>=pagerequire && num2<=k)
					flag=1;
			}	
			if(flag==1)
				printf("LuckyChef\n");
			else
				printf("UnluckyChef\n");
			t--;

		}	
	}
