#!/usr/bin/python
import sys
t=int(raw_input())
for i in range(0,t):
 x,y,k,n=map(int,sys.stdin.readline().split())
 p=[]
 c=[]
 for j in range(0,n):
  q,r=map(int,sys.stdin.readline().split())
  p.append(q)
  c.append(r)
 flag=0
 if y>=x:
  flag=1
 else:
  j=0
  sum=0
  cost=0
  for m in range(0,n):
   if y+p[m] >= x and c[m]<=k:
    flag=1
    break
 if flag==1:
  print "LuckyChef"
 else:
  print "UnluckyChef"
