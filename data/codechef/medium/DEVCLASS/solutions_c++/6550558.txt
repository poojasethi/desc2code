#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;
typedef long long int lli;
char str[100000];
vector<int> boyodd,boyeven,girlodd,girleven;
int type;
lli boyFirst()
{
	int i;
	if(type==0)	return (lli)boyodd.size();
	else if(type==1 || type==2)
	{
		lli ans=0;
		for(i=0;i<boyodd.size();i++)
			ans+=(lli)(abs(boyodd[i]-girleven[i]));
		return ans;		
	}
}
 
lli girlFirst()
{
	int i;
	if(type==0)	return (lli)boyeven.size();
	else if(type==1||type==2)
	{
		lli ans=0;
		for(i=0;i<boyeven.size();i++)
			ans+=(lli)(abs(boyeven[i]-girlodd[i]));
		return ans;		
	}
}
 
 
int main()
{
	int t,n,i,j,girlcount,boycount;
	cin>>t;
	lli ans;
	char str[100000];
	while(t--)
	{
		scanf("%d",&type);
		getchar();
		scanf("%s",str);
		n=strlen(str);
		girlcount=0;
		boycount=0;
		for(i=0;i<n;i++)
		{
			if(str[i]=='B')
			{
				boycount++; 
				if(i%2==0) boyeven.push_back(i);
				else boyodd.push_back(i); 
			}
			else
			{
				girlcount++; 
				if(i%2==0) girleven.push_back(i);
				else girlodd.push_back(i);
			}
		}
		if(boycount<n/2 || girlcount<n/2)  printf("-1\n"); 
		else
		{
			ans=-1;
			if(boycount>=girlcount) ans=boyFirst();
			if(girlcount>=boycount) 
			{
				if(ans!=-1) ans=min(ans,girlFirst());
				else ans=girlFirst();
			}
			printf("%lld\n",ans);
		}
		boyeven.clear();
		boyodd.clear();
		girleven.clear();
		girlodd.clear();
		//cout<<boyeven.size()<<" "<<boyodd.size()<<" "<<girleven.size()<<" "<<girlodd.size()<<endl;
	}
}
