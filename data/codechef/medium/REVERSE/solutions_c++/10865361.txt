int n,m;
vector<vpi > graph(100005);
int dijstra(int s)
{
	priority_queue<pi,vpi, greater<pi > > pq;
	vi dis(n+1, MAX);
	
	pq.push(mp(0,s));
	dis[s]=0;
	int i=1;
	while(i <= n && !pq.empty())
	{
		int u = pq.top().second;
		pq.pop();
		
		vector<pi > :: iterator p;
		for(p = graph[u].begin(); p!= graph[u].end();p++)
		{
			int v = p->second, w = p->first;
			if(w + dis[u] < dis[v])
			{
				dis[v] = dis[u] + w;
				pq.push(mp(dis[v], v));
			}
		}
	}
	if(dis[n] == MAX)
	return -1;
	else
	return dis[n];
}
