#include<bits/stdc++.h>
using namespace std;
#define maxs 999999
#define REP(i,a,b) for(i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)
#define INF 1000000000
int i,j,k,si,n,a,node,b,m,dist[100001];
vector <pair<int ,int> > chef[100001];
queue<int>q;
int solve()
{
    q.push(1);
    while(!q.empty())
    {   node=q.front();
        q.pop();
        si=chef[node].size();
        for(i=0;i<si;++i)
        {
            if(dist[chef[node][i].first]>dist[node] +chef[node][i].second )
            {
                dist[chef[node][i].first]=dist[node]+chef[node][i].second;
                q.push(chef[node][i].first);
            }
        }
    }
   return (dist[n]==maxs?-1:dist[n]);
}
int main()
{
    cin>>n>>m;
    for(i=1;i<=n;++i)dist[i]=maxs;
    while(m--)
    {
        scanf("%d%d",&a,&b);
        chef[a].push_back(make_pair(b,0));
        chef[b].push_back(make_pair(a,1));
    }
    dist[1]=0;
    printf("%d",solve());
}
