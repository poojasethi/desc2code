#include<iostream>
#include<vector>
#include<queue>
#include<cstdio>
using namespace std;
queue<int>q1,q2;
vector<int>a[100005],b[100005];
int dist[100005],visited[100005];

int main()
{



int j,s,t;
int m,n,i,x,y;
scanf("%d %d",&n,&m);

for(i=0;i<m;i++)
{

    scanf("%d %d",&x,&y);
    x--;
    y--;
    a[x].push_back(y);
    b[y].push_back(x);

}

for(i=0;i<n;i++)
{
    visited[i]=0;
    dist[i]=999999;
}



q1.push(0);
dist[0]=0;

while(!q1.empty() || !q2.empty())
{

    if(q1.empty() && q2.empty()) break;


   if(q1.empty() && !q2.empty())
 {
     s=q2.front();
    q2.pop();
 }
 else
{

 s=q1.front();
    q1.pop();

}
if(visited[s]==0){
        visited[s]=1;
    for(i=0;i<a[s].size();i++)
    {
        j=a[s][i];


        if(dist[j]>dist[s])
            {
            dist[j]=dist[s];q1.push(j);
            }
            }

    for(i=0;i<b[s].size();i++)
    {

        j=b[s][i];

            if(dist[j]>dist[s]+1)
           {

            dist[j]=dist[s]+1; q2.push(j);
            }

    }


}
}
    if(dist[n-1]==999999)
        dist[n-1]=-1;

    printf("%d\n",dist[n-1]);
return 0;
    }
