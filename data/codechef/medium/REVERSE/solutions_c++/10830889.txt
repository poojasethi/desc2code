
#include <iostream>
#include <stdio.h>
#include <utility>
#include <algorithm>
#include <vector>
#include <queue>
#include <stack>
#include <set>
#include <math.h>
#include <string.h>
#define ll long long
#define pii pair<int,int>
using namespace std;

ll read(){
	ll input;
	char c=0;
	while (c<33) c=getchar_unlocked();
	input=0;
	while (c>33){
		input=input*10+c-'0';
		c=getchar_unlocked();
	}
	return input;
}

struct comp
{
	bool operator()(const pii &a, const pii &b)
	{
			return a.second>b.second;
	}
};
int main()
{
	int n,m,u,v;
	int D[100001]={0};
	n=read();
	m=read();
	for(int x=0;x<n+1;x++)
		D[x]=-1;
	priority_queue<pii, vector < pii >, comp> Q;
	vector < pii > G[100001];
	for(int x=0;x<m;x++)
	{
		u=read();
		v=read();
		G[u].push_back(make_pair(v,0));
		G[v].push_back(make_pair(u,1));
	}
	Q.push(make_pair(1,0));
	D[1]=0;
	while(!Q.empty())
	{
		int w;
		pii top=Q.top();
		Q.pop();
		u=top.first;
		int s=G[u].size();
		for(int x=0;x<s;x++)
		{
			v=G[u][x].first;
			w=G[u][x].second;
			if(D[u]+w<D[v] || D[v]<0)
			{
				D[v]=D[u]+w;
				//cout<<v<<" "<<D[v]<<endl;
				Q.push(make_pair(v,D[v]));
			}
		}
	}
	printf("%d\n",D[n]);
	return 0;
}
