#include<bits/stdc++.h>
using namespace std;
#define pii pair<int,int>
#define pb push_back
#define mp make_pair
list<pii > *adj;
int main()
{
	int n,m,x,y;
	cin>>n>>m;
	adj = new list<pii >[n+1];
	while(m--)
	{
		cin>>x>>y;
		adj[x].pb(mp(y,0));
		adj[y].pb(mp(x,1));
	}
	set<pii > s;
	int dist[n+1];
	for(int i=1;i<=n;i++)
		dist[i] = INT_MAX;
	dist[1] = 0;
	s.insert(mp(0,1));
	while(!s.empty())
	{
		pii temp = *(s.begin());
		s.erase(s.begin());
		int u = temp.second;
		list<pii >::iterator it;
		for(it=adj[u].begin();it!=adj[u].end();it++)
		{
			int v = (*it).first;
			int w = (*it).second;
			
			if(dist[v] > dist[u] + w)
			{
				if(dist[v]!=INT_MAX)
					s.erase(s.find(mp(dist[v],v)));
				dist[v] = dist[u] + w;
				s.insert(mp(dist[v],v));
			}
		}
	}
	if(dist[n]==INT_MAX)
		cout<<"-1";
	else
		cout<<dist[n];
	cout<<endl;
	return 0;
}
