# include <bits/stdc++.h>
# define MP make_pair
# define ii pair< int ,int >
# define iii pair<int , ii >
# define MAX 100005


using namespace std ;
vector <ii > arr[MAX];
int dis[MAX];
int dij(int st,int end);
int main()
{
	int n,m,u,v,i;

	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++)
	{
		scanf("%d%d",&u,&v);
		arr[u].push_back(MP(0,v));
		arr[v].push_back(MP(1,u));
	}

	for(i=0;i<=n;i++)
		dis[i]=INT_MAX ;



	int ans=dij(1,n);
	if(ans==INT_MAX)
		ans=-1;
	printf("%d\n",ans);

	return 0;
}
int dij(int st,int end)
{
	priority_queue<ii,vector < ii > , greater<ii> > q;

	dis[st]=0;
	q.push(MP(0,st));
	int d,idx,cost,vertex;
	ii x;
	while(!q.empty())
	{
		x=q.top();
		q.pop();

		d=x.first;
		idx=x.second;

		if(idx==end)
			break;

		if(d<=dis[idx]){
			dis[idx]=d;
			int size=arr[idx].size();
			for(int j=0;j<size;j++)
			{
				cost=arr[idx][j].first;
				vertex=arr[idx][j].second ;

				if(d!=INT_MAX && d+cost<dis[vertex])
				{
					dis[vertex]=d+cost;
					q.push(MP(dis[vertex],vertex));
				}
			}
		}

	}

	return dis[end];

}
