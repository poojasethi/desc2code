#include<iostream>
#include<string>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<vector>
using namespace std;

string s;
int n;
vector<string> ans;

void com(string a)
{
    int i,x,y,z;
    string b=a;
    for(i=2;i<n;i++)
    {
        x=b[i-1]-'0';
        y=b[i-2]-'0';
        z=s[i-1]-'0';
        if((x+y+z)%2==0) b+='0';
        else b+='1';
    }

    x=b[n-2]-'0';
    y=b[n-1]-'0';
    z=b[0]-'0';
    if((x+y+z+s[n-1]-'0')%2) return ;
    x=b[n-1]-'0';
    y=b[0]-'0';
    z=b[1]-'0';
    if((x+y+z+s[0]-'0')%2==0) ans.push_back(b);
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        cin>>s;
        n=s.length();
        ans.resize(0);
        com("00");
        com("01");
        com("10");
        com("11");
        if(ans.size()==0)  cout<<"No solution\n";
        else if(ans.size()>1) cout<<"Multiple solutions\n";
        else cout<<ans[0]<<endl;
    }
    return 0;
}
