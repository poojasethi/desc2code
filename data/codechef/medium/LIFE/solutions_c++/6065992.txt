#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std ;
int main()
{
    int i,j,k,n,t,ans = 0 ,p1,p2,p3;
    int sans[60] ;
    cin >> t ;
    char a[60] ;
    int  a1[60] , b[60] ;
    while(t--) {
        ans = 0 ;
        cin >> a ;
        n = strlen(a);
        for(i=0;i<n;i++) a1[i] = a[i] - '0' ;
        for(i=0;i<=1;i++) {
            for(j=0;j<=1;j++) {
                b[0] = i  ;
                b[1] = j  ;
                for(k=1;k<n-1;k++) {
                    b[k+1] = b[k]^a1[k]^b[k-1] ;
                }
                p1 = b[n-1]^a1[n-1]^b[n-2] ;
                p2 = b[0]^a1[0]^b[n-1] ;
                if((p1 == b[0])&&(p2 == b[1])){
                    ans++ ;
                    for(p3=0;p3<n;p3++) sans[p3] = b[p3] ;
                }
                if(ans > 1) break ;

            }
        }
      //  cout << ans << endl ;
        if(ans == 0) {
            cout << "No solution\n" ;

        }
        else if(ans == 1) {
            for(i=0;i<n;i++) cout << sans[i] ;
            cout << endl ;

        }
        else cout << "Multiple solutions\n" ;

       // cout << t << endl ;
    }
    return 0;
}
