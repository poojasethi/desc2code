#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

char str[51],ans[51], res[51];
int len,count;

bool isValid(){
    for(int i = 0; i < len; i++){
        if(str[i] != (ans[(i+len-1)%len] ^ ans[(i+len)%len] ^ ans[(i+len+1)%len]))
        return false;
    }
    for(int i = 0; i < len; i++){
        res[i] = ans[i];
    }
    return true;
}

void solve(){
    for(int i = 0; i < 4; i++){
        if(i&1) ans[1] = '1';
        else ans[1] = '0';
        if(i&2) ans[0] = '1';
        else ans[0] = '0';

        for(int j = 2; j < len; j++){
            ans[j] = ans[j-2]^ans[j-1]^str[j-1];
        }
        if(isValid()){
            count++;
        }
    }
}

int main(){
    int T;
    scanf("%d", &T);

    while(T--){
        count = 0;
        scanf("%s", str);
        len = strlen(str);

        solve();
        res[len] = 0;

        if(count > 1) printf("Multiple solutions\n");
        else if(count == 0) printf("No solution\n");
        else printf("%s\n", res);
    }
}
