#include<stdio.h>
#include<string.h>
int main()
{
	char input[51],output[51],ans[51];
	int i,n,t,count,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",input);
		n=strlen(input);
		count=0;
		for(k=1;k<=4 && count<=1;k++)
		{
			if(k==1)
				strcpy(output,"00");
			else if(k==2)
				strcpy(output,"01");
			else if(k==3)
				strcpy(output,"10");
			else
				strcpy(output,"11");
			for(i=2;i<n;i++)
				output[i]='0'+((output[i-1]-'0')^(output[i-2]-'0')^(input[i-1]-'0'));
			output[i]='\0';
			
			if((output[0]-'0')^(output[1]-'0')^(input[0]-'0')^(output[n-1]-'0')==0 && (output[n-2]-'0')^(output[n-1]-'0')^(input[n-1]-'0')^(output[0]-'0')==0)
			{
				strcpy(ans,output);
				count++;
			}
		}
		if(!count)
			printf("No solution\n");
		else if(count==1)
			printf("%s\n",ans);
		else
			printf("Multiple solutions\n");
	}
	return 0;
}
