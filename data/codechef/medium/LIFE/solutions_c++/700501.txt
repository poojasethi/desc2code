#include<stdio.h>
#include<string.h>

int check(int org[],int cur[],int n)
{
   int buffer[55]={0};
   int i;
   for(i=0;i<n;i++)
   {
        if(i==0)
        {    
            if(org[n-1]^org[(i+1)%n])
                buffer[i]=!org[i];
            else
                buffer[i]=org[i];
        }
        else
        {    
            if(org[i-1]^org[(i+1)%n])
                 buffer[i]=!org[i];
            else
                buffer[i]=org[i];
        }
   }
   for(i=0;i<n;i++)
     if(buffer[i]!=cur[i])
     return 0;
   return 1;
}

int main()
{
  int t,i,j,k,count,n;
  int cur[55],org[55],buffer[55];
  char str[55];
  
  scanf("%d",&t);
  while(t--)
  {
    scanf("%s",&str);
    count=0;
    n=strlen(str);
    for(i=0;i<n;i++)
      cur[i]=str[i]-48;

    for(i=0;i<=1;i++)
    {
       for(j=0;j<=1;j++)
       {
           org[0]=i;
           org[1]=j;
           for(k=2;k<n;k++)
              org[k]=org[k-1]^org[k-2]^cur[k-1];
           if( check(org,cur,n) )
           {  
              count++;
              for(k=0;k<n;k++)
                buffer[k]=org[k];
           }
       }
    }
    if(count==0)
      printf("No solution\n");
    else if(count==1)
    {  
      for(i=0;i<n;i++)
        printf("%d",buffer[i]);
      printf("\n");
    }
    else if(count > 1)
      printf("Multiple solutions\n");
  }
  return 0;
}
