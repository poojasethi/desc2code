#include<stdio.h>
#include<string.h>
int main()
{
	int t,i,ctr,flag,j,l,k;
	char a[51],b[4][51],c[4][51];
	scanf("%d",&t);
	while (t--)
	{
		scanf("%s",a);
		l=strlen(a);
		ctr=0;
		for (j=0;j<4;j++)
		{
			flag=0;
			if (j==0)
				b[j][0]=b[j][1]=48;
			else if (j==1)
				{
				b[j][0]=48;
				b[j][1]=49;
				}
			else if (j==2)
				{
					b[j][0]=49;
					b[j][1]=48;
				}
			else
				b[j][0]=b[j][1]=49;
			for (i=2;a[i];i++)
				b[j][i]=b[j][i-2]^b[j][i-1]^a[i-1];
			b[j][i]=0;
			c[j][0]=b[j][l-1]^b[j][1]^b[j][0];
			for (i=1;i<l-1;i++)
				c[j][i]=b[j][i-1]^b[j][i+1]^b[j][i];
			c[j][l-1]=b[j][l-2]^b[j][0]^b[j][l-1];
			c[j][l]=0;
			for (i=0;a[i];i++)
				if (a[i]!=c[j][i])
					{
						flag=1;
						break;
					}
			if (!flag)
			{
				k=j;
				ctr++;
			}
		//	printf("%s\n%s\n",b[j],c[j]);
		}
		if (ctr==0)
			printf("No solution\n");
		else if (ctr==1)
			printf("%s\n",b[k]);
		else
			printf("Multiple solutions\n");
	}
	return 0;
}
