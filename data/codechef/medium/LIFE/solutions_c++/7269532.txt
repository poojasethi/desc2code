#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
              char str[51],cp[51],count=0;
              scanf("%s",str);
              int i,j,k,len=strlen(str);


              for(j=0;j<=1;j++)
              {
                               for(k=0;k<=1;k++)
                               {
                                                char gn[51]={'\0'};
                                                gn[0]=j+48;
                                                gn[len-1]=k+48;

                                                for(i=1;i<len-2;i++)
                                                gn[i]=gn[i-1]^str[i-1]^gn[(i-2+len)%len] ;

                                                if(((gn[i-1]^str[i]^gn[i+1])==gn[(i-2+len)%len]^str[i-1]^gn[i-1]))
                                                {
                                                     gn[i]=gn[i-1]^str[i]^gn[i+1];
                                                     if(((gn[i]^str[i+1]^gn[0])==gn[i+1]))
                                                {
                                                                                //printf("-->%s\n",gn);
                                                                                 if(!count)
                                                                                 strcpy(cp,gn);
                                                                                 count++;
                                                }
                                                }

                                                }

                                                }
              if(count==1)
              printf("%s\n",cp);
              else if(count>1)
              printf("Multiple solutions\n");
              else
              printf("No solution\n");




              }
    return 0;
}
