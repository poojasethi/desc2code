#include<stdio.h>
#include<string.h>
#define BUG 0
char single[60];

int transform(char in[],char fi[],int length){
	char temp[60];
	if(in[1]==in[length-1])
	{
		temp[0]=in[0];
	}
	else
	{
		if(in[0]=='0') temp[0]='1';
		else temp[0]='0';
	}
	if(in[length-2]==in[0])
	{
		temp[length-1]=in[length-1];
	}
	else
	{
		if(in[length-1]=='0') temp[length-1]='1';
		else temp[length-1]='0';
	}
	int i,j,k;
	for(i=1;i<length-1;i++)
	{
		if(in[i-1]==in[i+1])
		{
			temp[i]=in[i];
		}
		else
		{
			if(in[i]=='0') temp[i]='1';
			else temp[i]='0';
		}
	}
	temp[length]='\0';
	if(BUG) printf("temp: %s final: %s initial: %s\n",temp,fi,in);
	if(strcmp(temp,fi)==0)
	{
		strcpy(single,in); 
		return 1;
	}	
	else return 0;


}

int main(){
	int T,i,j,k,l;
	scanf("%d",&T);
	while(T--){
		char final[60],init[60];
		scanf("%s",final);	
		int possible=0,length=strlen(final);
		final[length]='\0';
		for(i=0;i<4;i++)
		{
			if(i==0)
			{
				init[0]='0';
				init[1]='0';
			}
			if(i==1)
			{
				init[0]='0';
				init[1]='1';
			}
			if(i==2)
			{
				init[0]='1';
				init[1]='0';
			}
			if(i==3)
			{
				init[0]='1';
				init[1]='1';
			}
			
			if(init[0]==final[0])
			{
				init[length-1]=init[1];
			}
			else
			{
				if(init[1]=='0') init[length-1]='1';
				else init[length-1]='0';
			}		
			for(j=1;j<length-2;j++)
			{
				if(init[j]==final[j])
				{
					init[j+1]=init[j-1];
				}
				else
				{
					if(init[j-1]=='0') init[j+1]='1';
					else init[j+1]='0';
				}
			}
			init[length]='\0';
			possible+=transform(init,final,length);
		}
		if(possible==0)
		{
			printf("No solution\n");
		}
		else if(possible==1)
		{
			printf("%s\n",single);
		}
		else
		{
			printf("Multiple solutions\n");
		}
	}
	
	return 0;
}
