#include<cstdio>
 
int a[100],b[100],c[100];
int len;
int ans;
 
void geta() {
len=0;
char c;
 
scanf("%c",&c);
while(c==' ' || c=='\n')
	scanf("%c",&c);
	
while(c!='\n') {
		a[len++]=c-'0';
		scanf("%c",&c);
	}
	//for(int i=0;i<len;i++) {
		//printf("%d",a[i]);
	//}
}
 
 
int opp(int a) {
if(a) return 0;
else return 1;
}
 
 
 
void check(int l) {
 
for(int i=1;i<len-1;i++) {
	if(a[i]==b[i]) {
		b[i+1] = b[i-1];
	} else {
		b[i+1] = opp(b[i-1]);
	}
}
 
int change;
 
if(b[len-2]==b[0] ) change = b[len-1];
else change = opp(b[len-1]);
 
if(b[len-1] == l && change==a[len-1] ) {
		ans++;
		for(int i=0;i<len;i++)
			c[i]=b[i];
			
			//printf("\n");
	//for(int i=0;i<len;i++)
		//printf("%d",b[i]);
	}
	
	
}
 
void solve() {
ans=0;
 
b[0]=a[0];
b[1]=1;
b[len-1]=1;
check(b[len-1]);
 
b[0]=a[0];
b[1]=0;
b[len-1]=0;
check(b[len-1]);
 
b[0]=opp(a[0]);
b[1]=1;
b[len-1]=0;
check(b[len-1]);
 
b[0]=opp(a[0]);
b[1]=0;
b[len-1]=1;
check(b[len-1]);
 
if(ans==0) {
printf("No solution\n");
} else if(ans==1) {
	for(int i=0;i<len;i++) {
		printf("%d",c[i]);
	}
	printf("\n");
} else {
printf("Multiple solutions\n");
 
}
 
}
 
int main(int jesus , char *loves[] ) {
int tc;
scanf("%d",&tc);
while(tc--) {
geta();
solve();
}
return 0;
}
