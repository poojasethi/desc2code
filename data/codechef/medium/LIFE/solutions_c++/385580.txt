#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
#include<string.h>
using namespace std;
int main()
{
    int testcase;
    cin>>testcase;
    char c[4][50];
    char ans[50];
    int rez[4];
    while(testcase--)
    {
       int totalans=0;
       int oneans;
       cin>>ans;
       //cout<<" ans : "<<ans;
       int len=strlen(ans);
       c[0][0]='0';
       c[0][len-1]='0';
       c[1][0]='0';
       c[1][len-1]='1';
       c[2][0]='1';
       c[2][len-1]='0';
       c[3][0]='1';
       c[3][len-1]='1';
       
       for(int k=0;k<4;k++)
       {
              rez[k]=1;
              // puts(c[k]);
       for(int i=1;i<len;i++)
       {
         if(i==1)
         {
        if(ans[i-1]==c[k][i-1])        
        //no change
         c[k][i]=c[k][len-1];                 
        else
        c[k][i]=c[k][len-1]=='0'? '1' : '0';
         }      
        else if(i==len-1)
        {
        
        char temp=c[k][i];
        if(ans[i-1]==c[k][i-1])        
        //no change
         c[k][i]=c[k][i-2];                 
        else
        c[k][i]=c[k][i-2]=='0'? '1' : '0';  
            
            if(c[k][i] != temp)
            {
            rez[k]=0;
            //cout<<"\n from inside String :  "<<k<<" is not a solution ! ";
            }
        }
        else
        {
        if(ans[i-1]==c[k][i-1])        
        //no change
         c[k][i]=c[k][i-2];                 
        else
        c[k][i]=c[k][i-2]=='0'? '1' : '0';    
            
        }         
             
       }
       
       //checking cond for last bit
       if(ans[len-1]==c[k][len-1])
       {
       if(c[k][len-2] != c[k][0])                           
        {//cout<<"\n String :  "<<k<<" is not a solution ! ";
        rez[k]=0;
        }
        }
       else
        {
       if(c[k][len-2] == c[k][0])                           
       {   //     cout<<"\n String :  "<<k<<" is not a solution ! ";
       rez[k]=0;
       }
        }
          }
       /*for(int i=0;i<4;i++)
       {for(int k=0;k<len;k++)
       cout<<c[i][k];      
       cout<<"\n";
       }
       */
       for(int i=0;i<4;i++)
       {
       if(rez[i]==1)
       {totalans++;
       oneans=i;
       }
       }
       //cout<<"\n total ans : "<<totalans;
       if(totalans==1)
       {for(int k=0;k<len;k++)
       cout<<c[oneans][k];      
       cout<<"\n";
        }
       else if(totalans==0)
       cout<<"No solution\n";
       else
       cout<<"Multiple solutions\n";
       //cout<<" c1 : "<<c[0]<<" c2 : "<<c[1]<<" c3: "<<c[2]<<" c4 : "<<c[3];
           }
    //system("pause");
return 0;
}
