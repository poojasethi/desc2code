#include<cstdio>
#include<cstring>
char input[51];
char attempt[51];
char sol[51];
bool check(char val,char val2);
bool validate(int lhs,int mhsIndex,int rhs);
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%s",input);
		int val = check('0','0');
		val += check('0','1');
		if(val < 2)
			val += check('1','0');
		if(val < 2)
			val += check('1','1');
		if(val == 2)
			printf("Multiple solutions\n");
		else if(val == 1)
			printf("%s\n",sol);
		else
			printf("No solution\n");
	}
}
bool check(char val,char val2)
{
	attempt[0] = val;
	attempt[1] = val2;
	int neighbour = (val == '1');
	int i;
	for(i=1;input[i+1];i++)
	{
		if(attempt[i] == input[i])
			attempt[i+1] = neighbour?'1':'0';
		else
			attempt[i+1] = neighbour?'0':'1';
		neighbour = attempt[i] - '0';
	}
	attempt[i+1] = 0;
	if(validate(neighbour+'0',i,attempt[0]) && validate(attempt[i],0,attempt[1]))
	{
		strcpy(sol,attempt);
		return true;
	}
	return false;
}
bool validate(int lhs,int mhsIndex,int rhs)
{
	if(attempt[mhsIndex] == input[mhsIndex])
		return lhs == rhs;
	else
		return lhs != rhs;
}
