#include<bits/stdc++.h>
using namespace std;
void plotnext(int a[],int b[],int i)
{
	if(b[i-1]==1 && b[i]!=a[i])
	{
		b[i+1]=0;
	}
	if(b[i-1]==1 && a[i]==b[i])
	{
		b[i+1]=1;
	}
	if(b[i-1]==0 && a[i]!=b[i])
	{
		b[i+1]=1;
	}
	if(b[i-1]==0 && a[i]==b[i])
	{
		b[i+1]=0;
	}
	
}
int plotprev(int a[],int b[])
{
	if((b[1]==0 && b[0]==a[0])||(b[1]==1 && b[0]!=a[0]))
	{
		return 0;
	}
	else if ((b[1]=0 && b[0]!=a[0]) || (b[1]=1 && a[0]==b[0]))
	return 1;
}
int main()
{
	int t,i,j,k,bb,cc,dd,ee;

	char arr[52];
	int a[52],b[52],c[52],d[52],e[52];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",&arr);
		k=strlen(arr);
			bb=cc=dd=ee=0;
		for(i=0;i<k;i++)
		{
			a[i]=int(arr[i]-'0');
		}
		a[k]=a[0];
		int count=0;
		// case 1:
		b[0]=0;
		b[1]=0;
		for(i=1;i<=k;i++)
		{
			plotnext(a,b,i);
		}
		if(b[k]==b[0] && b[k+1]==b[1])
		{
			count++;
			bb=1;
		//	cout<<"00 succss\n";
		}
		// case 2:
		c[0]=0;
		c[1]=1;
		for(i=1;i<=k;i++)
		{
			plotnext(a,c,i);
		}
		
		if(c[k]==c[0] && c[k+1]==c[1])
		{
			count++;
			cc=1;
		//	cout<<"01 succss\n";
		}
		//case 3:
		d[0]=1;
		d[1]=0;
		for(i=1;i<=k;i++)
		{
			plotnext(a,d,i);
		}
		if(d[k]==d[0] && d[k+1]==d[1])
		{
			count++;
			dd=1;
		//	cout<<"10 succss\n";
		}
		// case 4:
		e[0]=1;
		e[1]=1;
		for(i=1;i<=k;i++)
		{
			plotnext(a,e,i);
		}
		if(e[k]==e[0] && e[k+1]==e[1])
		{
			count++;
			ee=1;
		//	cout<<"11 succss\n";
		}
		if(count==0)
		{
			printf("No solution\n");

		else if(count>1)
		{
			printf("Multiple solutions\n");

		else
		{
			if(bb==1)
