#include <cstdio>
#include <cstring>

using namespace std;

char s[51];
int len;

int orig[51];
int sol[51];
bool tryit(int n)
{ 
  orig[0] = (n&1)==1;
  orig[1] = (n&2)==2;
  
  for(int i = 2; i < len; i++)
    orig[i] = (s[i-1]^orig[i-1])?(!orig[i-2]):orig[i-2];
  
  if((orig[1] == (s[0]^orig[0])?(!orig[len-1]):orig[len-1])
   &&(orig[0] == (s[len-1]^orig[len-1])?(!orig[len-2]):orig[len-2]))
  {
    for(int i = 0; i < len; i++)
      sol[i] = orig[i];
    return true;
  }
  else
    return false;
}
  
int main()
{
  int t;
  scanf("%d", &t);
    
  while(t--)
  {
    scanf("%s", s);
    len = strlen(s);
              
    for(int i = 0; i < len; i++)
      s[i] = (s[i]=='1')?1:0;
      
    int sols = 0;
    
    for(int i = 0; i < 4; i++)
      if(tryit(i))
        sols++;
        
    if(sols == 1)
    {
      for(int i = 0; i < len; i++)
        printf("%d", sol[i]);
      printf("\n");
    }
    else if(sols > 1)
      printf("Multiple solutions\n");
    else
      printf("No solution\n");
  }
}
