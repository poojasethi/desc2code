#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

char in[55];
char solution[55];
int a[55];
int b[55];
int st[4][2] = {{0,0}, {0,1}, {1,0}, {1,1}};
int n, t;

int main()
{
    scanf("%d", &t);
    while (t--) {
        scanf("%s", in);
        n = strlen(in);
        for (int i=0; i<n; i++) {
            if (in[i]=='0')
                a[i] = 0;
            else
                a[i] = 1;
        }
        int cnt = 0;
        for (int k=0; k<4; k++) {
            b[0] = st[k][0];
            b[1] = st[k][1];
            b[n-1] = (a[0]==b[0])?b[1]:!b[1];
            for (int i=2; i<n-1; i++) {
                b[i] = (a[i-1]==b[i-1])?b[i-2]:!b[i-2];
            }
            //for (int i=0; i<n; i++) {
            //    printf("%d", b[i]);
            //}
            //printf("\n");

            bool ok;
            if (a[n-2]==b[n-2]) {
                ok = (b[n-1]==b[n-3]);
            } else {
                ok = (b[n-1]!=b[n-3]);
            }
            //cout << ok << endl;
            if (ok) {
                if (b[n-1]==a[n-1]) {
                    ok = (b[0]==b[n-2]);
                } else {
                    ok = (b[0]!=b[n-2]);
                }
                //cout << ok << endl;
            }
            if (ok) {
                cnt++;
                for (int i=0; i<n; i++) {
                    solution[i] = '0'+b[i];
                }
                solution[n] = '\0';
            }
        }
        if (cnt==0)
            printf("No solution\n");
        else if (cnt==1)
            printf("%s\n", solution);
        else
            printf("Multiple solutions\n");
    }
    return 0;
}
