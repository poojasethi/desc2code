#include<iostream>
using namespace std;

void f(char z[],string s,int i)
{
int k=i-1;
if(k==-1)
k=s.length()-1;
if(z[i]==s[i])
{
z[i+1]=z[k];
}
else
{
if(z[k]=='0')
z[i+1]='1';
else
z[i+1]='0';
}
}

void check(char z[],string s,int &cnt)
{
int l=s.length();
int c=0;
if(z[l-2]==s[l-2])
{
if(z[l-3]==z[l-1])
c++;
}
else
{
if(z[l-3]!=z[l-1])
c++;
}

if(z[l-1]==s[l-1])
{
if(z[l-2]==z[0])
c++;
}
else
{
if(z[l-2]!=z[0])
c++;
}
if(c==2)
cnt++;
}

void print(char z[],int l)
{
for(int i=0;i<l;i++)
{
cout<<z[i];
}
cout<<endl;
}

int main()
{
int t,l;
string s;
char z1[52],z2[52],z3[52],z4[52];
cin>>t;
for(int i=0;i<t;i++)
{

l=s.length();
z1[0]='0';
z2[0]='0';
z3[0]='1';
z4[0]='1';
z1[l-1]='0';
z2[l-1]='1';
z3[l-1]='0';
z4[l-1]='1';
for(int i=0;i<l-2;i++)
{
f(z1,s,i);
f(z2,s,i);
f(z3,s,i);

}
//print(z1,l);

int bit=0,cnt=0;
check(z1,s,cnt);
if(cnt==1 && bit==0)
bit=1;
check(z2,s,cnt);
if(cnt==1 && bit==0)
bit=2;
check(z3,s,cnt);
if(cnt==1 && bit==0)
bit=3;
check(z4,s,cnt);
if(cnt==1 && bit==0)
bit=4;
if(cnt>1)
cout<<"Multiple solutions"<<endl;
else if(cnt==0)

else

if(bit==1)
