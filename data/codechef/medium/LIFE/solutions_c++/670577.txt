#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const char fi[] = "LIFE.INP";
typedef long long ll;

int n;
ll rot(ll x, int amt)
{
    return ((x<<amt)|(x>>(n-amt)))&((1ll<<n)-1);
}

int main()
{
    char s[55];
    int T;
    int i;
    ll cur, org;
    //freopen(fi,"r", stdin);
    scanf("%d", &T);
    while (T--)
    {
        scanf("%s", s);
        n = strlen(s);
        cur = org = 0;
        for (i=0;i<n;i++) cur |=  ll(s[i]-'0')<<i;
        for (i=0;i<n;i++)
            if ((i%3) != (2*n%3))
                org ^= rot(cur, i);
        if (n%3==0)
        {
            if (org == 0) puts("Multiple solutions");
            else puts("No solution");
        }
        else
        {
            for (i=0;i<n;i++)
                putchar('0'+((org>>i)&1));
            putchar('\n');
        }
    }
    return 0;
}
