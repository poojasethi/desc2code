#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t,i,j,k,l,ans;
    char a[100],x[100],f[100];
    cin>>t;
    while(t--)
    {
        cin>>a;
        l=strlen(a);
        ans=0;
        x[l]='\0';
        for(i=0;i<=1;i++)
            for(j=0;j<=1;j++)
                {
                    x[0]=i+'0';
                    x[1]=j+'0';
                    for(k=2;k<l;k++)
                    {
                       x[k]=x[k-1]^x[k-2]^a[k-1];
                    }
                    if(x[k-1]==x[0]^x[k-2]^a[k-1] && x[0]==x[k-1]^x[1]^a[0])
                    {
                        ans++;
                        strcpy(f,x);
                    }
//cout<<i<<"   "<<j<<"   "<<x<<"  "<<ans<<endl;
                }
                if(ans==1)
                    cout<<f<<endl;
                else if(ans>1)
                    cout<<"Multiple solutions\n";
                else
                    cout<<"No solution\n";

    }
    return 0;
}
