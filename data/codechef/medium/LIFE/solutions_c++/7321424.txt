int cnt=0;
string s,ans;
char Xor(char a, char b) {
    return (((a-'0')^(b-'0'))+'0');
}
bool isvalid(string res) {
    int n=res.size();
    if(Xor(res[1],res[n-1])=='1') {
        if(s[0]==res[0])
            return 0;
    }
    if(Xor(res[1],res[n-1])=='0') {
        if(s[0]!=res[0])
            return 0;
    }
    if(Xor(res[n-2],res[0])=='1') {
        if(s[n-1]==res[n-1])
            return 0;
    }
    if(Xor(res[n-2],res[0])=='0') {
        if(s[n-1]!=res[n-1])
            return 0;
    }
    F(i,1,n-1) {
        if(Xor(res[i-1],res[i+1])=='1') {
            if(s[i]==res[i])
                return 0;
        }
        if(Xor(res[i-1],res[i+1])=='0') {
            if(s[i]!=res[i])
                return 0;
        }
    }
    return 1;
}
inline void genperm(int x,int y) {
    int n=s.size();
    string res="";
    res+=(x+'0');res+=(y+'0');
    F(i,1,n-1) {
        if(res[i]==s[i])
            res+=res[i-1];
        else
            res+=((res[i-1]-'0')^1)+'0';
    }
  //  cout<<res<<endl;
    if(isvalid(res)) {
        ans=res;
        cnt++;
    }
}
int main() {
    int t;sci(t);
    while(t--) {
        cin>>s;cnt=0;
        genperm(0,0);
        genperm(0,1);
        genperm(1,0);
        genperm(1,1);
//      cout<<cnt<<endl;
        if(cnt==1){
            cout<<ans<<endl;
        }
        if(cnt>1){
            cout<<"Multiple solutions"<<endl;
        }
        if(!cnt)
            cout<<"No solution"<<endl;
    }
    return 0;
}
