#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int check(int c1,int c2,int c3,int c4,int o1,int o2){
    if((c1^c2^o1)==c3 && (c3^c4^o2)==c1) return 1;
    return 0;
}
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--){
                   string given;
                   cin>>given;
                   int len=given.length();
                   int org[len];
                   for(int i=0;i<len;i++) org[i]=given[i]-48;
                   int buffer[4][len];
                   int curr[len];
                   buffer[0][0]=curr[0]=0;
                   buffer[0][1]=curr[1]=0;
                   int count=0;
                   int k;
                   for(int i=2;i<len;i++) buffer[0][i]=curr[i]=curr[i-1]^curr[i-2]^org[i-1];
                   if(check(curr[0],curr[1],curr[len-1],curr[len-2],org[0],org[len-1])){count++;k=0;}
                   buffer[1][1]=curr[1]=1;
                   buffer[1][0]=0;
                   for(int i=2;i<len;i++) buffer[1][i]=curr[i]=curr[i-1]^curr[i-2]^org[i-1];
                   if(check(curr[0],curr[1],curr[len-1],curr[len-2],org[0],org[len-1])){count++;k=1;}
                   buffer[2][0]=curr[0]=1;
                   buffer[2][1]=1;
                   for(int i=2;i<len;i++) buffer[2][i]=curr[i]=curr[i-1]^curr[i-2]^org[i-1];
                   if(check(curr[0],curr[1],curr[len-1],curr[len-2],org[0],org[len-1])){count++;k=2;}
                   buffer[3][1]=curr[1]=0;
                   buffer[3][0]=1;
                   for(int i=2;i<len;i++) buffer[3][i]=curr[i]=curr[i-1]^curr[i-2]^org[i-1];
                   if(check(curr[0],curr[1],curr[len-1],curr[len-2],org[0],org[len-1])){count++;k=3;}
                   if(count==0) printf("No solution\n");
                   if(count>1) printf("Multiple solutions\n");
                   if(count==1){
                                for(int j=0;j<len;j++) cout<<buffer[k][j];
                                printf("\n");
                   }
    }
    return 0;
}
