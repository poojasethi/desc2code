#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    int  max_count=0;
    int count=0;
    for(int i=0;i<n;i++)
    {
        if(a[i]==0)
            count=0;
        else
            count++;
        if(count>max_count)
            max_count=count;
    }
    cout<<max_count<<endl;

}
