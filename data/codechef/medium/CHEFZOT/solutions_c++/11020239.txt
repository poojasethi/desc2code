#include <bits/stdc++.h>

using namespace std;
long long int M=10000009;
int main()
{
    long long int n;
    cin>>n;
    long long int a[n],i;
    for(i=0;i<n;++i)
    cin>>a[i];
    long long int ctr=0;
    long long int prod=1;
    long long int max=0;
    for(i=0;i<n;++i)
    {
        prod*=a[i];
        if(prod==0)
        {
            if(ctr>max)
            max=ctr;
           ctr=0;
           prod=1;
        }
        else
        {
            ++ctr;
            if(ctr>max)
            max=ctr;
        }


    }
    cout<<max<<"\n";
    return 0;
}
