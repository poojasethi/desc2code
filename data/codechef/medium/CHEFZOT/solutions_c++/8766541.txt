    #include <iostream>
    using namespace std;
     
    int main() {
    	int n,i,flag=0,count,max,k=0;
    	scanf("%d",&n);
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    	scanf("%d",&a[i]);
    	}
    	count=0;
    	max=0;
    	for(i=0;i<n;i++)
    	{
    	if((a[i]!=0)&&(flag==0))
    	{
    	flag=1;
    	count++;
    	}
    	else if((a[i]!=0)&&(flag==1))
    	{
    	count++;
    	}
    	else if((a[i]==0)&&(flag==1))
    	{
    	if(count>max)
    	{
    	max=count;
    	}
    	flag=0;
    	count=0;
   		k=1;
    	}
    	}
    	if(k==1)
    	{
    	printf("%d\n",max);
    	}
    	else
    	{
    	printf("%d\n",count);
    	}
    	return 0;
    }
