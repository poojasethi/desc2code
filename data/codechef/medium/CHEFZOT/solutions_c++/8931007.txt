#include<iostream>
#include<stdio.h>
#include<string.h>

using namespace std;

int main(){

int n;
scanf("%d",&n);
int *a = new int[n];
for(int i=0;i<n;i++)
    scanf("%d",&a[i]);
int maxvalue = 0;
int count = 0;

for(int i=0;i<n;i++){
if(a[i]!=0){
    count++;
}
if(maxvalue < count){
    maxvalue = count;
}
if(a[i]==0)
    count = 0;
}

printf("%d\n",maxvalue);

return 0;
}
