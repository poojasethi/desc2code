#include <iostream>
#include <stdlib.h>
using namespace std;
 
int main() {
	for(int t=0;t<1;t++){
	int n,c=0,e=0;
	cin>>n;
	int a[n];
    
    for(int i=0;i<n;i++)
    {
    	cin>>a[i];
    	if(a[i]==0)
    	c++;
    }
    
    if(n==1 && c==0)
    {
    	cout<<"1";
    	continue;
    }
   if(n==1 && c!=0)
   {
   	cout<<"0";
   	continue;
   }
    int b[c];
   
    for(int i=0;i<n;i++)
    {
    	if(a[i]==0)
    	{
    		b[e]=i;
    	    e++;
    	}
   }
   	if(c==1)
   	{
   		if(n-b[0]>b[0])
   	   {
   	   	cout<<n-b[0]-1;
   		continue;
   		}
   		else
   		{
   			cout<<b[0];
   		continue;
   		}
   	}
   	int max;
   	
   	max=b[1]-b[0];
	for(int i=1;i<c-1;i++)
	{
		int q = b[i+1]-b[i];
		if(q>max)
		max=q;
	}
	
	if(b[0]>=max)
	{cout<<b[0];
	continue;
	}
	cout<<max-1;
	}
	return 0;
}
