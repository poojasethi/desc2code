#include<bits/c++io.h>
#include<iostream>
using namespace std;
#include<string.h>
int main()
{
    int i,n,a[100020],count=0,max=0;
    cin>>n;
    for(i=0;i<n;i++)
    cin>>a[i];
    for(i=0;i<n;i++)
    {
        if(a[i]==0)
        {
        count=0;
        }
        else
        {
        count++;
        if(count>max)
        max=count;
        }
        //printf("%d %d\n",count,max);
    }
    cout<<max<<endl;
    max=count=0;
}
