#include<iostream>
using namespace std;
int main()
{
     int n,c=0,i,maxi=0;
        cin>>n;
        int a[n];
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        for(i=0;i<n;i++)
        {
               if(a[i]!=0)
                c++;
               
                if(maxi<c)
                    maxi=c;
                    if(a[i]==0)
                     c=0;
        }
       cout<<maxi<<endl;
    return 0;
}
