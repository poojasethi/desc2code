#include<bits/stdc++.h>
using namespace std;

int n,a;

int gcd(int a,int b){
	return((b==0)?a:gcd(b,a%b));
}

int main (){
	while(scanf("%d",&n) == 1){
		int cnt = 0, ans =0;
	
		while(n--){
			scanf("%d",&a);
			if(a)
				++cnt;
			else{
				ans = max(ans,cnt);
				cnt = 0;
			}
		}
	
		printf("%d\n",max(ans,cnt));
	}
	return 0;
}
