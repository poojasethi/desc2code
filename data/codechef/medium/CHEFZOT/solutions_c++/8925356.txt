#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    int n,arr[100010],max=-1,count=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    scanf("%d",&arr[i]);

    for(int i=0;i<n;i++)
    {
        if(arr[i]==0)
        {
            if(count>max)
            max=count;
            count=0;
        }
        else
        count++;
    }
    if(count>max)
    printf("%d",count);
    else
    printf("%d",max);
    return 0;
}
