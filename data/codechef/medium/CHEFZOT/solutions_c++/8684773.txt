#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
using namespace std;

unsigned long long int gcd(unsigned long long int x,unsigned long long int y){
    if(y==0){
        return x;
    }
    else{
        return gcd(y,x%y);
    }
}
int main()
{
    int n;
    scanf("%d\n",&n);
    int array[n];
    for(int i=0;i<n;i++){
        scanf("%d",&array[i]);
    }
    int max=0;
    int trouble=0;
    for(int i=0;i<n;i++){
        
        if(array[i]!=0){
            trouble++;
        }
        if(trouble>max){
            max=trouble;
        }
        if(array[i]==0){
            trouble=0;
        }
    }
    
    
    printf("%d\n",max);

    
    
    
    
    
    
    
return 0;
}
