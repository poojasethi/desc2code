#include<cstdio>
#include<algorithm>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main()
{
    int n;
    scanf("%d",&n);
    int pre=-1,length=0;
    for(int i=0;i<n;i++){
        int tmp;
        scanf("%d",&tmp);
        if(tmp==0){
            pre=i;
        }
        length=max(i-pre,length);
    }
    printf("%d\n",length);
}
