#include <iostream>

using namespace std;

int main()
{
    long int n,cur=0,maxm=0;
    cin>>n;
    int a;
    for(int i=1;i<=n;i++)
    {
        cin>>a;
        if(a!=0)
            cur++;
        else
        {
            if (cur>maxm)
                maxm=cur;
            cur=0;
        }
    }
    if (cur>maxm)
        maxm=cur;
    cout<<maxm<<"\n";
    return 0;
}
