#include<iostream>
using namespace std;
int main()
{
 long int N,A[100005];
 cin>>N;
 for(long int i=0;i<N;i++)
 {
 	cin>>A[i];
 }
 long int count=0,max=0;
 A[N]=0;
  for(long int j=0;j<=N;j++)
     {
     	if(A[j]!=0)
     	count++;
 	    if(A[j]==0)
		{
			if(count>max)
			max=count;
			count=0;
		} 
	}
	cout<<max<<endl;
}
