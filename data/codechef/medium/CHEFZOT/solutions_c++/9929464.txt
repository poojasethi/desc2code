#include<iostream>
using namespace std;
int main()
{
    int a[100000], n, j, cnt, maxcnt;
    j = cnt = maxcnt = 0;
    cin>>n;
 
    for(int i=0;i<n;i++)
        cin>>a[i];
 
    for(int k=0;k<n;k++)
    {
        if(a[k]==0)
        {
            j++;
            continue;
        }
 
        cnt = 0;
 
        while((a[j]!=0) && (j<n))
        {
            cnt++;
            j++;
        }
 
        k=j-1;
        if(cnt > maxcnt) maxcnt=cnt;
    }
 
    cout<<"\n"<<maxcnt;
    return 0;
}
