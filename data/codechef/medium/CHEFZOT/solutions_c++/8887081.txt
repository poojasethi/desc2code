#include <iostream>
using namespace std;

int main() {
	// your code goes here
	
	int a[100000],n,i,max=0;
	int count=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
	    cin>>a[i];
	}
	for(i=0;i<n;i++)
	{
	   if(a[i]==0)
	   {
	       count=0; 
	      
	   }
	   else
	   {
	        count++;
	       if(count>max)
	       {
	           max=count;
	       }
	      
	   }
	}
	cout<<max;
	return 0;
}
