#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n;
    cin >> n;

    int a[n],i,j,cur;
    for(i = 1;i <= n;i++) {
        cin >> a[i];
    }
    int ans = 0,last = 0;
    for(i = 1;i <= n;i++) {
        if(a[i] == 0) {
            last = i;
        }
        cur = i - last;
        ans = max(ans,cur);
    }
    cout << ans << "\n";
    return 0;
}
