#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	
		int n,a,pro=1,t=0;
		cin>>n;
		int count[n];
		for(int i=0;i<n;i++)
		  count[i]=0;
		int x=0;
		
		for(int j=0;j<n;j++)
		{
			cin>>a;
			pro=pro*a;
			if(pro==0)
			 {
			 	count[x]=t;
			 	x++;
			 	pro=1;
			 	t=0;
			 }
			else 
			{
				t++;
				count[x]=t;
			}
		}
		sort(count,count+n);
		cout<<count[n-1]<<endl;

	return 0;
}
