#include <bits/stdc++.h>
using namespace std;

int A[10004][2], B[10004];
double X[10004][2];

int main()
{
    int T;
    scanf("%d", &T);
    
    while(T--)
    {
        int N, M, P;
        scanf("%d %d %d", &N, &M, &P);
        
        for(int i=1; i<=M; i++)
        {
            scanf("%d", &B[i]);
            X[i][0] = 1;
        }
        
        sort(B+1, B+1+M);
        
        for(int i=1; i<=M; i++)
            A[i][0] = B[i];
        
        int tmp = N, k = 0;
        double ans = 0;
        
        while(tmp)
            tmp/=2, k++;
        
        for(int j=0; j<k-1; j++)
        {
            int m = 0;
            
            for(int i=1; i<=M; i++)
            {
                if(i<M && A[i][j%2]%2 && A[i+1][j%2]==A[i][j%2]+1)
                {
                    X[++m][(j+1)%2] = X[i][j%2]*X[(i+1)][j%2] + X[i][j%2]*(1-X[(i+1)][j%2])*P/100 + (1-X[i][j%2])*X[(i+1)][j%2]*P/100;
                    A[m][(j+1)%2] = (A[i][j%2]+1)/2;
                    i++;
                }
                else
                {
                    X[++m][(j+1)%2] = X[i][j%2]*P/100;
                    A[m][(j+1)%2] = (A[i][j%2]+1)/2;
                }
            }
            
            M = m;
        }
        
        for(int i=1; i<=M; i++)
            ans += X[i][(k+1)%2];
        
        printf("%.9lf\n", ans*100);
    }
 
	return 0;
}
