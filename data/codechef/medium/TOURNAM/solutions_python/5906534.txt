#include<stdio.h>
#include<string.h>
void merge(int *a,int s,int mid,int end)
{
        int i=s,j=s,k=mid+1,b[end+1];
        while(i<=mid&&k<=end)
        {
                if(a[i]<a[k])
                b[j++]=a[i++];
                else
                b[j++]=a[k++];
        }
        while(i<=mid)
        b[j++]=a[i++];
        while(k<=end)
        b[j++]=a[k++];
        for(i=s;i<=end;i++)
        a[i]=b[i];
        return;
}
void divide(int *a,int s,int l)
{
        if(s<l)
        {
                int mid=(s+l)/2;
                divide(a,s,mid);
                divide(a,mid+1,l);
                merge(a,s,mid,l);
        }
}
int main()
{
        int t,n,k,i,j;
        double p;
        scanf("%d",&t);
        int a[10000];
        double b[10000];
        while(t--)
        {
                scanf("%d%d%lf",&n,&k,&p);
                p=p/100.0;
                for(i=0;i<k;i++)
                scanf("%d",&a[i]);
                divide(a,0,k-1);
                for(i=0;i<k;i++)
                b[i]=1;
                while(n>1)
                {
                        j=0;
                        for(i=0;i<k;i++)
                        {
                                if((i!=(k-1))&&(a[i]%2)&&(a[i+1]==a[i]+1))
                                {
                                        a[j]=(a[i]+1)/2;
                                        b[j++]=b[i]*b[i+1]+(b[i]*(1-b[i+1])+b[i+1]*(1-b[i]))*p;
                                        i++;
                                }
                                else
                                {
                                        a[j]=(a[i]+1)/2;
                                        b[j++]=b[i]*p;
                                }
                        }
                        k=j;
                        n/=2;
                }
                printf("%.8lf\n",100*b[0]);
        }

}
