def main():
		t = int(raw_input())
		while t>0 :
				s = raw_input().strip()
				t-=1
				print lastNode(s,len(s))

def lastNode(s,n):
		track = 1
		mod = 1000000007
		for i in range(n):
				if s[i] == "l":
						if i%2==1:
								track = track*2 -1
						else :
								track = track*2
				else:
						if i%2 :
								track = track*2 + 1
						else :
								track = track*2+2
		track%=mod
		return track

if __name__ == "__main__":
		main()
