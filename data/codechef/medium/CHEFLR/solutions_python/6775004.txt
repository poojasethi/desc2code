T = int(raw_input())
for i in range(T):
	le = raw_input()
	odd = False
	count = 1
	for l in le:
		if(l == 'l'):
	   		if(odd == False):
				count = count * 2
	   		else:
	 			count = (count * 2)-1
		elif(l == 'r'):
			if(odd == False):
				count = (count * 2) + 2
			else:
				count = (count * 2) + 1
		odd = not odd
 
	print count % 1000000007
