#include<stdio.h>
#include<string.h>
#define gc getchar_unlocked
#define mod 1000000007
inline void fastRead_string(char *str)
	{
	    register char c = 0;
	    register int i = 0;
	 
	    while (c < 33)
	        c = gc();
	 
	    while (c != '\n') {
	        str[i] = c;
	        c = gc();
	        i = i + 1;
	    }
	 
	    str[i] = '\0';
    }
void fastread(long long *a)
{
	char c=0;
	while(c<33)
	c=gc();
	*a=0;
	while(c>33)
	{
		*a=*a*10+c-'0';
		c=gc();
	}
}    
int main()
{
	long long t,level,num,length;
	char s[100001];
	fastread(&t);
	while(t--)
	{
	 	fastRead_string(s);
		length=strlen(s);
		level=1;
		num=1;
		for(int i=0;i<length;i++)
		{
			if(level%2==1)
			{
				if(s[i]=='l')
				num*=2;
				else if(s[i]=='r')
				num=num*2+2;
			}
			else
			{
				if(s[i]=='l')
				num=num*2-1;
				else if(s[i]=='r')
				num=num*2+1;
			}
			level++;
			num=num%mod;
		}
		printf("%lld\n",num);
	}
}
