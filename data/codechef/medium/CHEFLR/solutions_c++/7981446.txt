#include<stdio.h>
#include<stdlib.h>

#define max 1000000007
int main()
{
	int t,level,i,j,k;
	
	char str[100005];
	scanf("%d",&t);
	while(t--)
	{
		level=1;
		scanf("%s",str);
		j=1;
		
		for(i=0;str[i]!='\0';i++)
		{
			if(str[i]=='l')
			{
				if(level%2==0)
				{
					j=(j*2)-1;
				}
				else
				{
					j=(j*2);
				}
				
				level++;
			}
			else
			{
				if(level%2==0)
				{
					j=(2*j)+1;
				}
				else
				{
					j=(2*j)+2;
				}
				
				level++;
			}
			
			if(j>=max) j=j%max;
			
		}
		
		printf("%ld\n",j);
		
		
	}
	
	
	
}
