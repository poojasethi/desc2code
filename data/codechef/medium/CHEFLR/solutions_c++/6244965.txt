#include<iostream>
using namespace std;
#include<string.h>
void foo(string s)
{
   long long unsigned int i=0,ans=1,k=1;
   for(i=0;i<s.size();i++)
   {
       if(k%2==1)
       {
       if(s[i]=='l')
       {
           ans=(ans*2)%1000000007;
       }
       else{
        ans=(ans*2+2)%1000000007;

       }
       k++;
       }
       else
       {
           if(s[i]=='l')
           {

               ans=(ans*2-1)%1000000007;
           }
           else
           {

               ans=(ans*2+1)%1000000007;
           }
           k++;
       }
   }

    cout<<ans%1000000007<<endl;

}
int main()
{
    string s;
long long unsigned int t;
cin>>t;
while(t--)
{
  cin>>s;
  foo(s);
}

}
