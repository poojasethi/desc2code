#include <iostream>
#include <cstdio>

using namespace std;

#define mod 1000000007
#define ll long long int

int main()
{
    ll t, ans, i;
    char a[111111];
    scanf("%lld", &t);
    while (t--) {
        ans = 1;
        scanf("%s", a);
        for (i = 0; a[i] != '\0'; i++) {
            if (i % 2 == 0) {
                if (a[i] == 'l')
                    ans = (2*ans)%mod;
                else
                    ans = (2*ans+2)%mod;
            }
            else {
                if (a[i] == 'l')
                    ans = (2*ans-1)%mod;
                else
                    ans = (2*ans+1)%mod;
            }
        }
        if (ans < 0)
            ans += mod;
        printf("%lld\n", ans);
    }
    return 0;
}
