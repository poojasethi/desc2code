#include<iostream>
#include<string.h>
using namespace std;
#define M 1000000007
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s[100005];
		cin>>s;
		long int n=0,i=0;
		long long int x=1;
		while(n!=strlen(s))
		{
			if(n%2==0)
			{
				if(s[i++]=='l')
				{
					x=2*x;
				}
				else
				{
					x=2*x+2;
				}	
				n++;
			}
			else
			{
				if(s[i++]=='l')
				{
					x=2*x-1;
				}
				else
				{
					x=2*x+1;
				}	
				n++;
			}
			x=x%M;
		}
		cout<<x%M<<endl;
	}
	return 0;
}
