#include<iostream>
using namespace std;
#define mod 1000000007
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        char a[100001];
        cin>>a;
        long long int num=1;
        for(int i=0;a[i]!='\0';i++)
        {
            if(i%2==0)
            {
                if(a[i]=='l')
                    num=(2*num)%mod;
                else
                    num=(2*num+2)%mod;
            }
            else
            {
                if(a[i]=='l')
                    num=(2*num-1)%mod;
                else
                    num=(2*num+1)%mod;
            }
        }
        cout<<num%mod<<endl;
    }
    return 0;
}
