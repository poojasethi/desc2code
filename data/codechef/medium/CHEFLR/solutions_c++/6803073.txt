#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    while(n--)
    {
        char arr[100001];
        cin>>arr;
        int p=0;
        long long int ans=1;
        for(int i=0;i<strlen(arr);i++)
        {
            if(p%2==0)
            {
                if(arr[i]=='l')
                {
                  ans=(ans*2)%1000000007;
                }
                else
                {
                  ans=(ans*2+2)%1000000007;
                }
            }
            else
            {
                if(arr[i]=='l')
                {
                  ans=(ans*2-1)%1000000007;
                }
                else
                {
                    ans=(ans*2+1)%1000000007;
                }
            }
            p++;
        }
        cout<<ans%1000000007<<endl;
    }
    return 0;
}
