#include <iostream>
using namespace std;

#define mod 1000000007

int main() {
    string s;
    int t, n, l;
    cin >> t;
    while(t--) {
        cin >> s;
        n = 1;
        l = s.length();
        for(int i=0;i<l;++i) {
            if (i&1) {
                n = ((2 * n) % mod - 1 + mod) % mod;
                if (s[i] == 'r')
                    n = (n + 2) % mod;
            } else {
                n = (2 * n) % mod;
                if (s[i] == 'r')
                    n = (n + 2) % mod;
            }
        }
        cout << n << endl;
    }
    return 0;
}
