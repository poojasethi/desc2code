#include<iostream>
#include<cstdio>
using namespace std;

int mod = 1000000007;
int main(){
	int t;
		char c;

		scanf("%d",&t);
			getchar();

	while(t--){	
			long long res=1;
			int level=1;

			scanf("%s",str);
			int i=0;		
				while(str[i]){
			
			if(str[i]=='l'){
				if((level%2)==0)
						res=2L*res-1L;
				else
						res=2L*res;
						}

  				if((level%2)==0)

return 0;
