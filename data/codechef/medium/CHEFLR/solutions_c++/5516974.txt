#include<iostream>

#define mod 1000000007

using namespace std;

int main()
{
    unsigned long long t,l,i,n;
    string s;

    cin>>t;
    while(t--)
    {
        cin>>s;
        l=s.size();
        n=1;
        for(i=0;i<l;i++)
        {
            if(i%2==0)
            {
                if(s[i]=='l')
                {
                    n=(2*n);
                }
                else
                {
                    n=(2*n+2);
                }
            }
            else
            {
                if(s[i]=='l')
                {
                    n=(2*n-1);
                }
                else
                {
                    n=(2*n+1);
                }
            }
            n=n%mod;
        }
        cout<<n<<endl;
    }
    return 0;
}
