#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
main()
{
int t;
scanf("%d",&t);
while(t--)
{
char str[100009];
scanf("%s",str);
int ans=1;
for(int i=0;str[i]!='\0';i++)
{
if(i%2==0)
{
if(str[i]=='l')
ans=(ans*2)%1000000007;
else
ans=(ans*2+2)%1000000007;
}
else
{
if(str[i]=='l')
ans=(ans*2-1)%1000000007;
else
ans=(ans*2+1)%1000000007;
}
}
printf("%d\n",ans);
}
return 0;
}
