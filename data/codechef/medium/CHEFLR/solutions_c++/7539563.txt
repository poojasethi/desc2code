    #include<iostream>
    #include<cstdio>
    #include<algorithm>
    #include<string>
    #define mod 1000000007
    using namespace std;
    typedef long long int ll;
    int main()
    {
        ll t,n,i,x,r;
        cin>>t;
        string s;
        while(t--)
        {
            cin>>s;
            n=s.length();
            for(i=0,r=1;i<n;i++)
            {
                if(i%2==0)
                {
                    if(s[i]=='l')
                        x=(r*2)%mod;
                    else
                        x=((r*2)%mod+2)%mod;
                    r=x;
                }
                else
                {
                    if(s[i]=='l')
                        x=((r*2)%mod-1)%mod;
                    else
                        x=((r*2)%mod+1)%mod;
                    r=x;
                }
            }
            cout<<x<<'\n';
        }
        return 0;
    }
