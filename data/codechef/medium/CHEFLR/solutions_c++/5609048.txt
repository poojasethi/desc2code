#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[100005];
#define MOD 1000000007
int main()
{
    int t,odd,n,i;
    long long int ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",&s);
        n=strlen(s);
        odd=1;
        ans=1;
        for(i=0;i<n;i++)
        {
            if(odd)
            {
                if(s[i]=='l')
                   ans=(ans<<1)%MOD;
                else if(s[i]=='r')
                    ans=((ans<<1)%MOD+2)%MOD;
                odd=0;
            }
            else
            {
                if(s[i]=='l')
                   ans=((ans<<1)%MOD-1)%MOD;
                else if(s[i]=='r')
                    ans=((ans<<1)%MOD+1)%MOD;
                odd=1;
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
