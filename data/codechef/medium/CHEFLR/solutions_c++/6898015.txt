#include <bits/stdc++.h>
using namespace std;

#define sd(var) scanf("%d", &var)
#define slld(var) scanf("%lld", &var)
#define sllu(var) scanf("%llu", &var)
#define lld long long int
#define llu unsigned long long int
#define MOD 1000000007
#define print_arr(a, n) for (int I = 0; I < n; I++) { cout << a[I] << " "; } cout << endl;

int main() {
	lld tc, i, j, k, n, mn, a, b, p, gc, sum, t, len, ans;
	string s;
	
	slld(tc);

	while (tc--) {
		cin >> s;
		
		len = s.length();
		
		ans = 1;
		t = 1;
		
		for (i = 0; i < len; i++) {
			if (t == 1) {
				if (s[i] == 'l') {
					ans = (ans * 2) % MOD;
				} else {
					ans = (ans * 2 + 2) % MOD;
				}
				t = 0;
			} else {
				if (s[i] == 'l') {
					ans = (ans * 2 - 1) % MOD;
				} else {
					ans = (ans * 2 + 1) % MOD;
				}
				t = 1;
			}
		}
		
		cout << ans << endl;
		
	}
	
	return 0;
}
