#include<iostream>
#include<cstdio>
#include<string.h>
#define MOD 1000000007
using namespace std;
int main()
{
   int t;
   char s[100001];
   scanf("%d",&t);
   while(t--)
   {
       scanf("%s",s);
       int len = strlen(s);
       long long int cnt = 1, flg = 1, pos = 1, odd = 0, even = 0;
       for(int i=0; i<len; i++)
       {
           if(s[i] == 'l')
           {
               pos = pos*2 - 1;
               if(pos < 0)
                 pos += MOD;
               if(pos >= MOD)
               pos %= MOD;

           }
           else
           {
               pos *= 2;
               if(pos >= MOD)
                pos %= MOD;
           }
           if(flg)
           {
               if(odd == 0)
                odd = 1;
               else
               {
                   odd += cnt*2;
                   if(odd >= MOD)
                    odd %= MOD;
               }
           }
           else
           {
               even += cnt*2;
               if(even >= MOD)
                even %= MOD;
           }
           cnt *= 2;
           if(cnt >= MOD)
            cnt %= MOD;
           flg = (flg + 1)%2;
       }
       if(flg == 0)
       {
           even += 2*pos;
           if(even >= MOD)
            even %= MOD;
           printf("%lld\n",even);
       }
       else
       {
           odd += 2*pos;
           if(odd >= MOD)
            odd %= MOD;
           printf("%lld\n",odd);
       }
   }
   return 0;
}
