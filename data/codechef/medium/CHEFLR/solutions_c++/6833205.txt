#include<iostream>
using namespace std;
const int num = 100001;
const int mod = 1000000007;
int main()
{
    int T;
    cin >> T;
    long long  ans = 1;
    char str[num];
    while(T--)
    {
        cin >> str;
        ans = 1;
        for(int i = 0 ; str[i] ; i++)
        {
            if(i%2 == 0)
            {
               if(str[i] == 'l') {ans = (ans * 2 ) % mod;}
               if(str[i] == 'r') {ans = (ans * 2 + 2) % mod ;}
            }
            if(i%2 )
            {
               if(str[i] == 'l')
               {ans = ((ans * 2)  - 1 + mod) % mod;}
               if(str[i] == 'r')
               {ans = (ans * 2) % mod + 1;}
            }
        }
        cout << ans << endl;
 
    }
}
