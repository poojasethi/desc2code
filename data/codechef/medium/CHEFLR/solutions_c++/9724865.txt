#include <bits/stdc++.h>
#define ll long long
#define mod 1000000007
using namespace std;
ll power[200015];
char s[100005];
ll pows(ll a, ll b)
{
    ll ans=1;
    a=a%mod;
    while(b>0){
        if(b%2!=0)
            ans=(ans*a)%mod;
        a=(a*a)%mod;
        b/=2;
    }
    return ans;
}
int main()
{
    int i,j;
    power[0]=1;
    for(i=1;i<=200014;i++)
        power[i]=(power[i-1]*2)%mod;
    int t;
    scanf("%d",&t);
    while(t--){
        scanf("%s",s);
        int l=strlen(s);
        ll ans=0;
        for(i=0;i<l-1;i++){
            ll p=power[l-i-1];
            if(s[i]=='r')
                ans=(ans+p)%mod;
        }
        if(s[l-1]=='l')
            ans=(ans+1)%mod;
        else
            ans=(ans+2)%mod;
        if(l==1){
            if(s[0]=='l')
                cout<<"2\n";
            else cout<<"4\n";
            continue;
        }
        ll three=pows(3,mod-2);
        if((l&1)){
            ll x=(l-2-1)/2+1;
            x=power[2*x]-1;
            if(x<0)
                x+=mod;
            x=(x*three)%mod;
            x=(x*2)%mod;
            //cout<<x<<"  "<<ans<<endl;
            ans+=x;
            ans--;
            if(ans<0)
                ans+=mod;
            ans=(2+((ans*2)%mod))%mod;
        }
        else if(l%2==0){
            ll x=(l-2)/2+1;
            x=power[2*x]-1;
            if(x<0)
                x+=mod;
            x=(x*three)%mod;
            //cout<<x<<"  "<<ans<<endl;
            ans+=x;
            ans--;
            if(ans<0)
                ans+=mod;
            ans=(1+((ans*2)%mod))%mod;
        }
        cout<<ans<<endl;
    }
    return 0;
}
