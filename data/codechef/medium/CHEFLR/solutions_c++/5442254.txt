#include <iostream>
#define mod 1000000007
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	string s;
	while(t--){
		cin>>s;
		unsigned long ans=1;
		for(int i=0;s[i]!='\0';i++){
			if(i%2==0){
				if(s[i]=='l')ans=(ans*2)%mod;
				else ans=(ans*2+2)%mod;
			}
			else {
				if(s[i]=='l')ans=(ans*2-1)%mod;
				else ans=(ans*2+1)%mod;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
