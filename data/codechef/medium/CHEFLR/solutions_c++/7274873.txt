#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define MOD 1000000007
int main()
{
    //freopen("in.txt","r",stdin);
    int TC;
    scanf("%d",&TC);
    while(TC--){
        string s;
        cin >> s;
        long long int val=1;
        for(int i=0;i<s.size();i++){
            if(i%2==1){
                if(s[i]=='l') val=(val*2 - 1)%MOD;
                else val=(val*2 + 1)%MOD;
            }
            else{
                if(s[i]=='l') val=(val*2)%MOD;
                else val=(val*2 + 2)%MOD;
            }
        }
        printf("%lld\n",val);
    }
}
