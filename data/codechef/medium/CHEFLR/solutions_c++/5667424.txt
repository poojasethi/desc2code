#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main()
{
    int t;cin>>t;
    while(t--)
    {
        char a[100001]={0};
	scanf("%s",a);
	int n=strlen(a);
	long long ans=1;
	for(int i=0;i<n;i++)
	{
	    if(i%2==0)
	    {
		if(a[i]=='l')
		{
		    ans=ans*2;
		    ans%=mod;
//		    cout<<ans;
		}
	        else
		    ans=ans*2+2;
			ans%=mod;

	    }
	    else
	    {
		if(a[i]=='l')
		{
		    ans=ans*2-1;
			ans%=mod;
		}
		else
		{    ans=ans*2;
		     ans+=1;
//		     cout<<ans;
			ans%=mod;

		}
	    }
	 }
	 cout<<ans<<endl;
     }
     return 0;
}
