#include <iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
    int t,count;
    long long int i,len=0,ind=1;
    char s[100005];
    scanf("%d",&t);
    while(t--){
        ind=1;
        count=0;
        scanf("%s",s);
        len=strlen(s);
        for(i=0;i<len;i++){
            if(count==0){
                if(s[i]=='l'){
                    ind=(ind%(1000000007))*2;
                }
                if(s[i]=='r'){
                    ind=(ind%(1000000007))*2+2;
                }
            }
             if(count==1){
                if(s[i]=='l'){
                    ind=(ind%(1000000007))*2-1;
                }
                if(s[i]=='r'){
                    ind=(ind%(1000000007))*2+1;
                }
            }
        count=count^1;

        }
        ind=ind%(1000000007);
        printf("%lld\n",ind);
    }
    return 0;
}
