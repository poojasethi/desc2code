#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
char str[100000];
int main()
{
	int t,ans=1,l,i;
	cin>>t;
	while(t--)
	{
		scanf("%s",str);
		l=strlen(str);
		for(i=0;i<l;i++)
		{
			if(i%2==0)
			{
				if(str[i]=='l')
				{
				   ans=((2%1000000007)*(ans%1000000007))%1000000007;
				}
				else
				{
					ans=((2%1000000007)*(ans%1000000007))%1000000007+2;
				}
			}
			else
			{
				if(str[i]=='l')
				{
					ans=((2%1000000007)*(ans%1000000007))%1000000007-1;
					//i++;
				}
				else
				{
					ans=((2%1000000007)*(ans%1000000007))%1000000007+1;
					//i++;
				}
			}
		}
		cout<<ans<<"\n";
		ans=1;
	}
	return 0;
}
