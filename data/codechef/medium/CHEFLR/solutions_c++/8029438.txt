#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,t,i;
    char s[100001];
    long long int sum=1;
    cin>>t;
    while(t--)
    {
        scanf("%s",s);
        n=strlen(s);
        for(i=1;i<n;i+=2)
        {
            if(s[i-1]=='l')
                {sum=(sum*2)%1000000007;

                }
            else
                {sum=((sum*2)+2)%1000000007;
                }
            if(s[i]=='l')
                {sum=((sum*2)-1)%1000000007;
                }
            else
                {sum=((sum*2)+1)%1000000007;
                }
        }
        if(i-n==0)
        {
            if(s[n-1]=='l')
            sum=(sum*2)%1000000007;
            else
            sum=((sum*2)+2)%1000000007;
        }
        cout<<sum%1000000007<<"\n";
        sum=1;
    }
    return 0;
}
