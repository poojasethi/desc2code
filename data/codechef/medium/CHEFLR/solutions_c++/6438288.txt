#include<stdio.h>
#include<string.h>
int main()
{
	long long int sum;
	int t,i,levl;
	char ch[100001];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",ch);
		sum=1,levl=2;
		i=0;
		while(ch[i]!=0)
		{
			if(levl%2==1)
			{
				if(ch[i]=='l')
					sum=(sum*2-1)%1000000007;
				else
					sum=(sum*2+1)%1000000007;
			}
			else
			{
				if(ch[i]=='l')
					sum=sum*2%1000000007;
				else
					sum=(sum*2+2)%1000000007;
			}
			levl++;
			i++;
		}
		printf("%lld\n",sum);
	}
}
