#include<iostream>
using namespace std;
string s;
long long int i,v,t,k,mod=1000000007;
int main()
{
    cin>>v;
while(v--)
{
    cin>>s;
    t=1;
    k=1;
    i=0;
    while(s[i]!='\0')
    {
        if(t%2==0)
        {
            if(s[i]=='l')
                k=(2*k-1)%mod;
            else
                k=(2*k+1)%mod;
        }
    else
    {
            if(s[i]=='l')
                k=(2*k)%mod;
            else
                k=(2*k+2)%mod;

    }
    i++;
    t++;
    }

cout<<k<<endl;
}

return 0;
}
