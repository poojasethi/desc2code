#include<bits/stdc++.h>
using namespace std;
#define q 1000000007
void last(string s)
{
long long int val=1;
 
for(long int i=1;i<=s.size();i++)
{
val=(val*2)%q;
if(i&1)
{
if(s[i-1]=='r')
{
val+=2;
val=val%q;
}
}
else
{
if(s[i-1]=='r')
{
val=(val+1)%q;
}
else
val=val-1;
}
}
 
cout<<val;
}
 
int main()
{
int t=0;
cin>>t;
while(t--)
{
string s;
cin>>s;
last(s);
cout<<"\n";
}
 
return 0;
}
