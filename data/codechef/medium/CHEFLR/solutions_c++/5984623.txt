#include <iostream>
#include <string>
#define mod 1000000007
using namespace std;

int main()
{
	
	int t;
	
	cin >> t;
	
	while (t--)
	{
		
		long long x = 1;
		string s;
		cin>> s;
		int Slength = s.length();

		for (int i = 1; i <= Slength; i++){
			if (i % 2 == 0){
				if (s[i - 1] == 'l'){
					x = x * 2 - 1;
				}
				else{
					x = x * 2 + 1;
				}
				x = (x) % mod;
			}
			else{
				if (s[i - 1] == 'l'){
					x *= 2;
				}
				else{
					x = x * 2 + 2;
				}
				x = (x) % mod;
			}
		}
		cout<< x << endl;
	}
}
