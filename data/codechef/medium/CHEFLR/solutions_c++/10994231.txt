#include<iostream>
#include<string>
using namespace std;
int main(){
	int i,t,ans;
	string s;
	cin>>t;
	while(t--){
		cin>>s;
		ans=1;
		i=0;
		while(i < s.size()){
			if(s[i] == 'l' && (i+1)%2!=0){
				ans = (ans*2)%1000000007;
			}
			else if(s[i]=='r' && (i+1)%2!=0){
				ans = ((ans+1)*2)%1000000007;
			}
			else if(s[i]=='l' && (i+1)%2==0){
				ans = (ans*2 -1)%1000000007;
			}
			else if(s[i]=='r' && (i+1)%2==0){
				ans = (ans*2 +1)%1000000007;
			}
			i++;
		}
		cout<<ans<<"\n";
	}
	return 0;
}
