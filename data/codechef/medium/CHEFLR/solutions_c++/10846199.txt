#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int t,i,n;
long long ans;
char s[100005];
int main()
{
    scanf("%d",&t);
    while(t--)
    {
        ans=1;
        scanf("%s",s);
        n=strlen(s);
        for(i=0;i<n;i++)
        {
            ans*=2;
            if(s[i]=='r'&&i%2==0)
                ans+=2;
            else if(s[i]=='l'&&i%2!=0)
                ans-=1;
            else if(s[i]=='r'&&i%2!=0)
                ans+=1;
            ans=ans%1000000007;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
