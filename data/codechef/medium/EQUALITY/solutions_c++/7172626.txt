#include<iostream>
using namespace std;
typedef unsigned long long ull;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	ull t,n,a[100001];
	cin>>t;
	
	while(t--)
	{
		ull sum=0;
		cin>>n;
		for(ull i=0;i<n;i++)
		 {
		 	cin>>a[i];
		 	sum=sum+a[i];
		 }
		 
		sum=sum/(n-1);
		
		for(ull i=0;i<n;i++)
		 cout<<sum-a[i]<<" ";
		cout<<"\n";  
		 
	}
	return 0;
}
