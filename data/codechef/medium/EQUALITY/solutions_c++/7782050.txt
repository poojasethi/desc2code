    #include <stdio.h>
    #include <stdlib.h>
     
    int main()
    {
    	long long int arr[60000];
    	long long int j,sum,i,x;
    	
    	int testcases,number;
    	
    	scanf("%d",&testcases);
    	while(testcases--)
    	{
    	//printf("enter number\n");
    	scanf("%d",&number);
    	
    	
    	for(i=0,j=0,sum=0;i<number;i++)
    	{
    		//printf("enter number\n");
    		scanf("%lld",&x);
    		//printf("%d",x);
    		sum = sum + x;
    		arr[j++]=x;
    	}
    	
    	//printf("%d\n",sum);
    	
    	sum = sum/(number-1);
    	for(i=0;i<j;i++)
    	{
    		printf("%lld\n",sum-arr[i]);
    	}
    	
    	}
    	return 0;
    }
