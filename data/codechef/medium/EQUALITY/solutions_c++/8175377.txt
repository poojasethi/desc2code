#include<stdio.h>
int main()
	{
		long t,n;
		long long a[50000],sum;
		scanf("%ld",&t);
		while(t--)
		{
			scanf("%ld",&n);
			long i;
			sum=0;
			for(i=0;i<n;++i)
			{
				scanf("%lld",&a[i]);
				sum+=a[i];
			}
			sum/=n-1;
			for(i=0;i<n;++i)
				printf("%lld ",sum-a[i]);
			printf("\n");
		}
		return 0;
	}
