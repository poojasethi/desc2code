#include <iostream>
#include <cstdio>
using namespace std;

#define MAX_SIZE 50050


int main()
{
    int n,i,t;
    long long a[MAX_SIZE], sum;

    scanf("%d",&t);

    while(t--){
        scanf("%d",&n);

        sum=0;
        for(i=0;i<n;i++){
            scanf("%lld",&a[i]);
            sum += a[i];
        }
        sum /= (n-1);

        for(i=0;i<n;i++){
            printf("%lld ",sum-a[i]);
        }
        printf("\n");

    }
    return 0;
}
