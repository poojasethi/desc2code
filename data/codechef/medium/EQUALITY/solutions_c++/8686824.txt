#include <iostream>
using namespace std;
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int n,sum=0;
		cin>>n;
		long long int a[n];
		for (long long int i = 0; i < n; ++i)
		{	
			cin>>a[i];
			sum+=a[i];
		}
		sum=sum/(n-1);
		for (long long int i = 0; i < n; ++i)
		{
			cout<<sum-a[i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
