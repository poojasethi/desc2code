#include<bits/stdc++.h>
using namespace std;
#define lint long long
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        lint n;
        scanf("%lld",&n);
        lint arr[n];
        lint i=0;
        lint sum=0;
        while(i<n)
        {
            scanf("%lld",&arr[i]);
            sum+=arr[i];
            i++;
        }
        sum/=(n-1);
        i=0;
        while(i<n)
        {
            printf("%lld ",sum-arr[i]);
            i++;
        }
        printf("\n");
    }
}
