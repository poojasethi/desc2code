#include <cstdio>
using namespace std;

main() {
  int t, n;
  long long ans = 0;
  long long a[50001];
  scanf("%d", &t);
  for(int i=0;i<t;++i) {
    ans = 0;
    scanf("%d", &n);
    for(int j=0; j<n; ++j) {
      scanf("%lld", &a[j]);
      ans= ans + a[j];
    }
    ans = ans/(n-1);
    for(int j=0;j<n;++j) {
      printf("%lld ", (ans - a[j]));
    }
    
    printf("\n");
  }
}
