#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int MAXN = 1e6 + 10;
LL A[MAXN];
int main(){
    int t;
    cin >> t;
    while(t--){
        int N;
        scanf("%d", &N);
        LL sum = 0;
        for(int i = 1;i <= N;++i){
            scanf("%lld", &A[i]);
            sum += A[i];
        }
        sum /= (N - 1);
        for(int i = 1;i <= N;++i){
            printf("%lld ", sum - A[i]);
        }
        printf("\n");
    }
    return 0;
}
