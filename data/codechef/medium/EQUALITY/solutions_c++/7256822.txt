#include<iostream>
#include<algorithm>
#include<cmath>
#include<vector>
#include<set>
#include<queue>
#include<deque>
#include<stack>
#define f(i, n) for (int i = 0; i < n; i++)
#define fi(i, n) for (; i < n; i++)
#define d(i, n) for (int i = n; i >= 0; --i)
#define ff(i, m, n) for (int i = m; i < n; i++)
#define dd(i, m, n) for (int i = m; i > n; --i)
#define ull unsigned long long
#define mp(i, j) make_pair(i, j)
#define pb(v, a) v.push_back(a)
#define MAXN 50001
using namespace std;
int n, t;
ull a[MAXN], l[MAXN], r[MAXN];
int main() {
  cin >> t;
  while(t--) {
    cin >> n; f(i, n) cin >> a[i];
    if (n == 2) { cout << a[1] << " " << a[0] << endl; continue; }
    f(i, n) r[i] = l[i] = 0;
    f(i, n) if (i) l[i] = a[i-1]+l[i-1];
    d(i, n-2) r[i] = a[i+1]+r[i+1];
    f(i, n) cout << (r[i]+l[i]-(n-2)*a[i])/(n-1) << " "; cout << endl;
  }
}
