#include<stdio.h>
int main(){
    int t,n,i;
    long long sum,a[50001];
    scanf("%d",&t);
    while(t--){
     scanf("%d",&n);
     for(sum=i=0;i<n;i++){
      scanf("%lld",&a[i]);
      sum+=a[i];
     }
     sum/=n-1;
     //printf("sum = %lld, n= %d",sum,n);
     for(i=0;i<n;i++)
      printf("%lld ",sum-a[i]);
     putchar('\n');
    }
    return 0;
}
