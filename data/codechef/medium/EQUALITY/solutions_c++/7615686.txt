# include <bits/stdc++.h>
using namespace std;
int main () {
	ios::sync_with_stdio(0);
	int t; cin >> t;
	while (t--) {
		long long int i, n, sum = 0, A[100005]; cin >> n;
		for (i = 0; i< n; i++) {
			cin >> A[i]; sum += A[i];
		}
		//cout << sum << endl; 
		sum /= (n-1);
		for (i = 0; i < n; i++) {
			cout << sum-A[i] << " " ;
			//printf ("%lld ", A[i]);
		}
		cout << endl;
	}
}
