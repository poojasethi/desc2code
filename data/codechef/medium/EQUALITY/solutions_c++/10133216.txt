#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 50005
#define pb(a) push_back(a)
lli arr[MAX];
int main()
{
	lli i,j,k,m,n,t,sum;
	slli(t);
	while(t--)
	{
		slli(n);
		sum=0;

		f(i,0,n)
		slli(arr[i]);

		f(i,0,n)
		sum+=arr[i];

		sum=sum/(n-1);

		f(i,0,n)
		printf("%lld ",sum-arr[i]);
		printf("\n");
	}
}
