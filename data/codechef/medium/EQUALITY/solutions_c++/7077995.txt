#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,b;
	long long int count;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		long long int a[n+5];
		count=0;
		for(i=1;i<=n;i++)
		{
			scanf("%lld",&a[i]);
			count+=a[i];
		}
		count=count/(n-1);
		for(i=1;i<=n;i++)
		{
			b=count-a[i];
			printf("%d ",b);
		}
		printf("\n");
	}
}
