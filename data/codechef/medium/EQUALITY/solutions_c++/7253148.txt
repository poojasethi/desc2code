#include<stdio.h>
#include<math.h>

using namespace std;
int main()
{int tc;
 long int n;
scanf("%d\n",&tc);

while(tc--)
{
long long int sum=0;
scanf("%ld",&n);
long long int arr[n];
for(long int i=0;i<n;++i)
{    scanf("%lld",&arr[i]);
sum+=arr[i];
}

sum=(sum/(n-1));

for(long int i=0;i<n;++i)
    printf("%lld ",(sum-arr[i]));
    printf("\n");

}





return 0;
}
