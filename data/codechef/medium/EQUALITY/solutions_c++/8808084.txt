#include<stdio.h>
#include<stdlib.h>

long long int a[50010];

int main()
{
	int i,j,k,l,t,n;
	long long int sum,sum2;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		sum=sum2=0;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			sum+=a[i];
		}
		k= n-1;
		for(i=1;i<n;i++)
		{
			a[i]=sum-(k*a[i]);
			a[i]/=k;
			sum2+=a[i];
		}
		a[0]= (sum/k)-sum2;
		for(i=0;i<n;i++)
		printf("%lld ",a[i]);
		printf("\n");
	}
	return 0;
	
}
