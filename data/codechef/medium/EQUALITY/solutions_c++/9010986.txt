#include <iostream>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while(t--)
    {
        int n , i;
        cin >> n;

        long long a[n] , sum = 0;
        for(i = 0 ; i < n ; ++i)
        {
            cin >> a[i];
            sum+=a[i];
        }
        for(i = 0 ; i < n ; ++i)
        {
            cout << sum/(n-1) - a[i] << " ";
        }
        cout << '\n';
    }
    return 0;
}
