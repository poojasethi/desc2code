#include<iostream>
#include<cstdio>
using namespace std;
main()
{
	long long n,t,i,j,k,l,sum;
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld",&n);
		long long a[n];
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			sum+=a[i];
		}
		sum=sum/(n-1);
		for(i=0;i<n;i++)
			printf("%lld ",sum-a[i]);
		printf("\n");
	}
}
