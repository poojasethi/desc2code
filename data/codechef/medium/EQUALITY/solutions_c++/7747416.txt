#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <math.h>

using namespace std;

long long input()
{
    long long n=0;
    char c=getchar_unlocked();
    while(c < '0' || c >'9') {c=getchar_unlocked();}
    while(c>='0' && c<='9')
    {
        n=(n<<3)+(n<<1)+c-'0';
        c=getchar_unlocked();
    }
    return n;
}

int main(){
	long long temp, sum = 0;
	int N, T;
	long long* A;
	T = input();
	while(T--){
		N = input();
		sum = 0;
		A = new long long[N];
		for(int i = 0; i < N; i++){
			A[i] = input();
			sum += A[i];
		}
			sum /= (N-1);
		for(int i = 0; i < N; i++){
			cout << (long long)(sum-A[i]) << " ";
		}
		cout << endl;
		delete(A);
	}
}
