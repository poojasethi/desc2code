#include <iostream>
#include<set>
#include<cstring>

#define mod 1000000007

using namespace std;

set<unsigned long long int> myset;

unsigned long long int i,fib[10000];

int main() {
	
	fib[0]=0;
	fib[1]=1;
	myset.insert(1);
	myset.insert(0);
	
	for(i=2;i<=10000;i++)
	{
		fib[i]=fib[i-1]+fib[i-2];
		fib[i]=fib[i]%mod;
		myset.insert(fib[i]);
	}
	
	unsigned long long int ten=10,temp,number=0,t;
	char n[2000];
	cin>>t;
	while(t--)
	{
		number=0;
		cin>>n;
	    	for(i=0;i<strlen(n);i++)
	    	{
	    		char c=n[i];
	    		temp=c-'0';
	    		number=number*ten+temp;
	    		number=number%mod;
	    	}
	    	
	    	if(myset.find(number)==myset.end()) cout<<"NO"<<endl;
	    	else cout<<"YES"<<endl;
	}
	
	return 0;
}
