#include<iostream>
#include<string>
using namespace std;
string cvt(int i)
{
	if(i==1)
	return "2(0)";
	int k=0;
	while((1<<k)<=i)
	k++;
	k--;
	string s="2";
	if(k>1)
	s+="("+cvt(k)+")";
	if((i-(1<<k))>=1)
	s+="+"+cvt(i-(1<<k));
	return s;
}
int main()
{
int arr[]={137, 1315, 73, 136, 255, 1384, 16385};	
for(int i=0;i<7;i++)
cout<<arr[i]<<"="<<	cvt(arr[i])<<endl;
	return 0;
}
