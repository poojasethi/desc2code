#include<iostream>
using namespace std;
string cnt(int n)
{
    string s;
    if(n==1)
        return "2(0)";
    int i=0;
    while((1<<i)<=n)
        i++;
    i--;
    s="2";
    if(i>1)
    s=s+"("+cnt(i)+")";
    if(n-(1<<i))
        s=s+"+"+cnt(n-(1<<i));
    return s;
}
int main(void)
{
    int a[]={137, 1315, 73, 136, 255, 1384, 16385};
    for(int i=0;i<6;i++)
    {
        cout<<a[i]<<"="<<cnt(a[i])<<endl;
    }
    cout<<a[6]<<"="<<cnt(a[6]);
}
