#include<stdio.h>

int dp[78][78];
int min[78][78][78][78];

inline void MIN(int a, int &b){ if(b>a) b=a;}
inline void MAX(int &a, int b){ if(b>a) a=b;}

int main()
{
	int r,c,i,j,m,n,ans,si,sj;

	while(scanf("%d%d",&r,&c)!=EOF)
	{
		if(!r && !c) break;

		for(i=1;i<=r;i++) for(j=1;j<=c;j++)
		{
			scanf("%d",&dp[i][j]);
			dp[i][j]=dp[i][j-1]+dp[i-1][j]+dp[i][j]-dp[i-1][j-1];
		}

		for(si=1;si<=r;si++)
			for(sj=1;sj<=c;sj++)
				for(m=1;m<=r;m++)
					for(n=1;n<=c;n++)
					{
						i=m-si+1;
						j=n-sj+1;

						if(i<1 || j<1) continue;

						min[i][j][m][n]=dp[m][n] - dp[m][j-1] - dp[i-1][n] + dp[i-1][j-1];

						if(i!=m) MIN(min[i+1][j][m][n],min[i][j][m][n]);
						if(j!=n) MIN(min[i][j+1][m][n],min[i][j][m][n]);
						if(i!=m) MIN(min[i][j][m-1][n],min[i][j][m][n]);
						if(j!=n) MIN(min[i][j][m][n-1],min[i][j][m][n]);
					}

		ans = -2000000000;
		for(i=1;i<=r;i++)
			for(j=1;j<=c;j++)
				for(m=i+2;m<=r;m++)
					for(n=j+2;n<=c;n++)
						MAX(ans,dp[m][n] - dp[m][j-1] - dp[i-1][n] + dp[i-1][j-1]-min[i+1][j+1][m-1][n-1]);

		printf("%d\n",ans);
	}

	return 0;
}
