#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
char S[1000001];
main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n;
		int map[130];
		memset(map, 0, sizeof(map));
		scanf("%d", &n);
		while(n--)
		{
			char a, b;
			cin>>a>>b;
			//scanf("%c", &b);
			map[a]=b;
		}
		scanf("%s", S);
		bool hasDot=false;
		int di;
		for(int i=0 ; S[i]!='\0'; i++)
		{
			if(map[S[i]]!=0)
				S[i]=map[S[i]];
			if(S[i]=='.')
			{
				di=i;
				hasDot=true;
			}
		}
		char *p=S;
		while(1)
		{
			if(*p=='.' || *p==0)
				break;
			if(*p!='0')
				break;
			p++;
		}
		int b=0;
		while(1)
		{
			if(*p=='.' || *p==0)
				break;
			printf("%c", *p);
			p++;
			b++;
		}
		if(*p==0 && b==0)
			printf("0");
		if(hasDot)
		{
			char *q= S+strlen(S)-1;
			while(*q!='.')
			{	
				if(*q!='0')
					break;
				q--;
			}
		//	char *p
			if(p==q)
			{
				if(b==0)
					printf("0");
				goto jump;
			}
			while(p<=q)
			{
				printf("%c", *p);
				p++;	
			}
			jump:
				;
		}
		printf("\n");
	//	system("pause");
	}
}
