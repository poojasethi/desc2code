#include<bits/stdc++.h>
using namespace std;
 
#define ll long long int
 
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
 
    ll t,n,i,j,k,flag;
    char str[1000001];
    char v1,v2;
    map<char,char> map;
 
    cin>>t;
    while(t--)
    {
        map.clear();
        flag=0;
        cin>>n;
        while(n--)
        {
            cin>>v1>>v2;
            map[v1]=v2;
        }
        cin>>str;
        for(i=0;str[i]!='\0';i++)
        {
            if(map.count(str[i]))
            {
                str[i]=map[str[i]];
            }
            if(str[i]=='.')
                flag=1;
        }
        i=0;
        j=strlen(str)-1;
        if(flag)
        {
            while(str[j]=='0')
                j--;
        }
        if(str[j]=='.')
            j--;
        if(j<0)
            cout<<"0";
        else
        {
            while(str[i]=='0' && i<j)
                i++;
            for(k=i;k<=j;k++)
                cout<<str[k];
        }
        cout<<"\n";
    }
	return 0;
}
