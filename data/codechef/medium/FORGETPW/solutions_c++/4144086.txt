#include<iostream>
#include<cstdio>
#include<cstring>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        char a[94],str[1000001];
        scanf("%d",&n);
        for(int i=0;i<94;i++)
            a[i]=i+33;
        for(int i=0;i<n;i++)
        {
            char c,p;

            cin>>c>>p;
            a[(int)c-33]=p;
        }
      //  for(int i=0;i<94;i++)
         //   cout<<i<<" "<<a[i]<<" "<<(char)(i+33)<<endl;
        cin>>str;

        int len=strlen(str);
        for(int i=0;i<len;i++)
          {
             str[i]=a[(int)(str[i])-33];

          }

        int start=0,fin=len-1;

        for(start=0;start<len;start++)
        {
            if(str[start]!='0')
                break;
        }

        for(fin=len-1;fin>=0;fin--)
        {
            if(str[fin]!='0')
                break;
        }

        int decimal=-1;
        for(int i=0;i<len;i++)
        {
            if(str[i]=='.')
              {
                  decimal=i;
                  break;
              }
        }


        if(decimal==-1)
        {
            for(int i=start;i<len;i++)
                cout<<str[i];
            if(start>len-1)
                cout<<"0";


        }
        else
        {
            if(fin==decimal)
                fin=fin-1;
            for(int i=start;i<=fin;i++)
                cout<<str[i];
            if(start>fin)

                cout<<"0";

        }
        cout<<"\n";



    }
}
