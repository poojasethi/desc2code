#define sc scanf
#define si(a) sc("%d",&a)
#define sd(a) sc("%lf",&a)
#define sl(a) sc("%lld",&a)
#define pf printf
#define pi(a) pf("%d\n",a)
#define pd(a) pf("%lf\n",a)
#define pl(a) pf("%lld\n",a)

#define pb push_back
#define PI 2*acos(0.0)
#define LL long long
#define ULL unsigned long long
#define gcd(a,b) __gcd(a,b)
#define lcm(a,b) (a*b)/gcd(a,b)
#define CLR(a) memset(a,0,sizeof(a))

int main()
{
    char a[1000005],x[105],y[105];
    bool vis[1000005],f;
    int i,j,k,l,m,n,t;
    sc("%d",&t);
    while(t--)
    {
        sc("%d",&n);
        for(i=0;i<n;i++)
            cin>>x[i]>>y[i];
        sc("%s",a);

        CLR(vis);
        l=strlen(a);
        for(i=0;i<n;i++)
        {
            for(j=0;j<l;j++)
            {
                if(a[j]==x[i] && vis[j]==0)
                {
                    vis[j]=1;
                    a[j]=y[i];
                }
            }
        }
        f=0;
        for(i=0;i<l;i++)
            if(a[i]=='.')
                f=1;
        if(f)
        {
            i=l-1;
            while(a[i]=='0')
            {
                a[i]='\0';
                l--;i--;
            }
        }
        if(a[l-1]=='.')
            a[l-1]='\0',l--;
        i=0;
        while(a[i]=='0')
        {
            i++;
        }
        if(a[i]=='\0')
            pf("0");
        else
        for(j=i;j<l;j++)
            pf("%c",a[j]);
        pf("\n");

    }
    return 0;
}
