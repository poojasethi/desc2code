void calc(char S[], int s, char res[]){
	bool hasDecimal = false;
	for(int i = 0; i< s; i++)
		if(S[i] == '.') hasDecimal = true;
	if(hasDecimal) {
		int start = 0, end = s - 1;
		for(int i = 0; i < s; i++)
			if(S[i] != '0'){
				start = i;
				break;
			}
		for(int i = s - 1; i >= 0; i--)
			if(S[i] != '0' ) {
				if( S[i] == '.')
				end = i - 1;
				else
					end = i;
				break;
			}
		int c = 0;
		if(end < start) res[c++] = '0';
		else {
			for(int i = start; i <= end; i++) {
				res[c++] = S[i];
			}

		}
		res[c++] = '\0';
	}
	else
	{
		int start = -1, end = s - 1;
		for(int i = 0; i < s; i++)
			if(S[i] != '0'){
				start = i;
				break;
			}
		int c = 0;
		if(start == -1) {
			res[c++] = '0';
		}
		else {
			for(int i = start; i <= end; i++) {
				res[c++] = S[i];
			}

		}
		res[c++] = '\0';
	}


    return;
}
