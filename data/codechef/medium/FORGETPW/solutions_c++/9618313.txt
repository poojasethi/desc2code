#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int t,n,i,j,b,fl,c,p;
    char ch;
    scanf("%d",&t);
    string s;
    while(t--)
    {
        scanf("%d",&n);
        getchar();
         char a[n],b[n];
        for(i=0;i<n;i++)
            {
               ch=getchar();
               getchar();
               a[i]=ch;
               ch=getchar();
               getchar();
               b[i]=ch;

            }

        cin>>s;
        for(i=0;s[i]!='\0';i++)
        {
            for(j=0;j<n;j++)
                if(s[i]==a[j])
                {
                    s[i]=b[j];
                    break;

                }


        }
        c=0;
        fl=0;
        for(i=0;s[i]!='\0';i++)
        {
            if(s[i]!='0')
                fl=1;
            if(fl==0 && s[i]=='0')
                c++;
            if(s[i]=='.')
            {

                break;

            }


        }
        if(c>0)
          s.erase(0,c);
        fl=0;
        c=0;
        p=0;
        for(i=s.length()-1;i>=0;i--)
        {
            if(s[i]!='0')
                fl=1;
            if(fl==0 && s[i]=='0')
                c++;
            if(s[i]=='.')
                {
                    p=1;
                    break;
                }


        }
        n=s.length();
        if(p==1 && c>0)
        s.erase(n-c,c);
        n=n-c;
        if (s[n-1]=='.')
            s.erase(n-1,1);
        if(s=="")
            s='0';

         cout<<s<<endl;


    }
}
