#include<bits/stdc++.h>
using namespace std;

char con[256];
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		for(int i = 0;i < 256;++i)
			con[i] = i;
		int Q;
		scanf("%d", &Q);
		while(Q--){
			char src, des;
			scanf(" %c %c", &src, &des);
			con[src] = des;
		}
		string s;
		cin >> s;
		int N = s.size();
		for(int i = 0;i < N;++i)
			s[i] = con[s[i]];
		string ans = "";
		bool haveDot = false;
		int cntZeroes = 0;
		for(int i = 0;i < N;++i){
			haveDot |= (s[i] == '.');
			cntZeroes += (s[i] == '0'?1:0);
		}
		if((haveDot && cntZeroes == N - 1) || ((!haveDot) && cntZeroes == N)){
			ans = "0";
		} else if(haveDot){
			int _i = 0;
			while(s[_i] == '0')
				++_i;
			while(s[N - 1] == '0'){
				s.erase(N - 1);
				--N;
			}
			if(s[N - 1] == '.'){
				s.erase(N - 1);
				--N;
			}
			ans = s.substr(_i);
		} else {
			int _i = 0;
			while(s[_i] == '0')
				++_i;
			ans = s.substr(_i);
		}
		cout << ans << "\n";
	}
	return 0;
}
