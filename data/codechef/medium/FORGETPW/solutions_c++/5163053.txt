#include <iostream>
#include <map>
#include <string>
using namespace std;

int main() {
	int t,n;
	char c,p;
	string s;
	map<char,char> rep;
	map<char,char>::iterator it;
	
	cin >> t;
	
	while(t--)
	{
		rep.clear();
		
		cin >> n;
		
		while(n--)
		{
			cin >> c >> p;
			rep[c] = p;
		}
		
		cin >> s;
		int l = s.length();
		bool dot = false;
		for(int i=0; i<l; i++)
		{
			it = rep.find(s[i]);
			
			if(it != rep.end())
				s[i] = it->second;
		
			if(s[i] == '.')
				dot = true;
		}
		
		//Remove leading and trailing 0's
		int start = l, end = l-1;
		for(int i=0; i<l; i++)
		{
			if(s[i]!='0')
			{
				start = i;
				break;
			}
		}
		
		if(dot)
		{
			for(int i=l-1; i>=0; i--)
			{	
				if(s[i]!='0')
				{
					end = i;
					break;
				}
			}
		}
		
		if(s[end]=='.')
			end--;
			
		if(start>end)
			cout << 0 << "\n";
		else
		{
			for(int i=start; i<=end; i++)
				cout << s[i];
			cout << "\n";
		}
	}
	
	return 0;
}
