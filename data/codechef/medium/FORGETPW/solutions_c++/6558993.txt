#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;

int fast_integer() {
    int set = 0;
    char c = getchar();
    while(c>='0' && c<='9') {
    	set = (set<<3) + (set<<1)+c-'0';
    	c = getchar();
    }
    return set;
}

int main() {
    int i, N, started, decimal, decimalStarted, countDecimalZero, printed;
    char c, decoded;
    int T = fast_integer();
    while (T--) {
        N = fast_integer();
        started = 0, decimal = 0, decimalStarted = 0, countDecimalZero = 0, printed = 0;
        char A[94];
        for (i=0; i<94; i++) {
            A[i] = ' ';
        }
        while (N--) {
            c = getchar();
            getchar();
            A[c-'!'] = getchar();
            getchar();
        }
        c = getchar();
        while(c>='!' && c<='~') {
            decoded = A[c-'!'];
            if (decoded == ' ') {
                decoded = c;
            }
            if (started == 0) {
                if (decoded == '0') {
                    c = getchar();
                    continue;
                } else {
                    started = 1;
                }
            }
            if (decoded == '.') {
                decimal = 1;
                c = getchar();
                continue;
            }
            if (decimal == 1) {
                if (decoded == '0') {
                    countDecimalZero++;
                    c = getchar();
                    continue;
                } else {
                    if (decimalStarted ==0 ) {
                        printf(".");
                        decimalStarted = 1;
                    }
                    for (i=0; i<countDecimalZero; i++) {
                        printf("0");
                    }
                    countDecimalZero = 0;
                }
            }
        	printf("%c", decoded);
        	printed = 1;
        	c = getchar();
        }
        if (printed == 0) {
            printf("0");
        }
        printf("\n");
    }
}
