#include<iostream>
#include<vector>
 
using namespace std;
 
char c[100],p[100];
 
int main(){
    int t,i,j;
    cin>>t;
    while(t--){
        int n;
        string s;
        cin>>n;
        
        for(i=0;i<n;i++){
            cin>>c[i]>>p[i];
        }
        cin>>s;
        int f = 0;
        vector <int> done(s.size(),0);
        for(i=0;i<n;i++){
            for(j=0;j<s.size();j++){
                if(s[j] == c[i] && !done[j]){
                    s[j] = p[i];
                    done[j] = 1;
                }
            }
        }
        for(i=0;i<s.size();i++)
            if(s[i] == '.'){
            f = 1;
            break;
        }
        string ans = "";
        i = 0;
        while(i < s.size() && s[i] == '0')
            i++;
        j = s.size()-1;
        while(f && j >= 0 && s[j] == '0' )
            j--;
        while(i<=j)
            ans += s[i],i++;
        if(ans[ans.size()-1] == '.')
            ans = ans.substr(0,ans.size()-1);
        if(ans == "")
            ans = "0";
        cout<<ans<<endl;
    }
}
