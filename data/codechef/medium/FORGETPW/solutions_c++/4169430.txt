#include<stdio.h>
#include<string.h>
int main(){
	int t,i;
	scanf("%d",&t);
	//printf("value of t %d\n",t);
	for(i=1;i<=t;i++){
		int n;
		scanf("%d",&n);
		//printf("value of n %d\n",n);
		char rules[127];
		int j;
		for(j=33;j<127;j++)rules[j]=' ';
		char ci;
		getchar();
		for(j=1;j<=n;j++){
			ci=getchar();
			getchar();
			rules[ci]=getchar();
			getchar();
		}
		char cipher[1000000];
		gets(cipher);
		/*printf("%s",cipher);*/
		int len=strlen(cipher);
		if(len==1 && cipher[0]=='0'){
			printf("%c\n",cipher[0]);
			continue;
		}

		for(j=0;j<len;j++)if(rules[cipher[j]] !=' ')cipher[j]=rules[cipher[j]];
		j=0;
		int flag=0;
		while(cipher[j]=='0')j++;
		for(;j<len;j++)
		if(cipher[j]=='.')break;
		else {
			printf("%c",cipher[j]);
			flag=1;
		}
		/*printf("\nindex of decimal point= %d\n", j);*/
		int end=len-1;
		while(cipher[end]=='0')end--;
		j++;
		/*printf("j= %d",j);*/
		if(j<=end)printf(".");
		for(;j<=end;j++){
			printf("%c",cipher[j]);
			flag=1;
		}
		if(flag==1)
			printf("\n");
		else
			printf("0\n");
	}
	return 0;
}
