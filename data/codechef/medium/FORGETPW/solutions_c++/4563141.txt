#include<iostream>
#include<cstring>
#include<map>
#include<cstdio>
using namespace std;
char arr[1000006]={'\0'};
map<char,char> first;
main()
{
    int t,special;
    cin>>t;
    while(t)
    {
        t--;
        int n,pos;
        special=0;
        int sign=0;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            char a,b;
            cin>>a>>b;
            first[a]=b;
        }
        scanf("%s",arr);
        int len=strlen(arr);
        for(int i=0;i<len;i++)
        {
            if(first.count(arr[i])>0)
            arr[i]=first[arr[i]];
        }

        //printf("%s\n",arr);
        for(int i=0;i<len;i++)
        if(arr[i]=='.')
        {
            sign=1;
            pos=i;
            break;
        }
        if(!sign)
        {
            int i=0;
            while(arr[i]=='0')
            i++;
            for(int j=i;j<len;j++)
            {
            special=1;
            printf("%c",arr[j]);
            }
        }
        if(sign)
        {
            int i=0;
            while(arr[i]=='0')
            i++;
            for(int j=i;j<pos;j++)
            {
            special=1;
            printf("%c",arr[j]);
            }
            i=len-1;
            while(arr[i]=='0')
            i--;
            if(i!=pos)
            for(int j=pos;j<=i;j++)
            {
             special=1;
             printf("%c",arr[j]);
            }
        }
        if(!special)
        printf("0");
        printf("\n");
        first.clear();
        for(int i=0;i<len;i++)
        arr[i]='\0';
    }
    return 0;
}
