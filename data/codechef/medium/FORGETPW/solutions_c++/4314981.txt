#include <iostream>
#include <vector>
#include <cstdio>

using namespace std;

int main()
{
    int t;
    int n;
    char a;
    char b;
    int c = 0;
    string s;
    string s1;
    int flag = 0;
    int flag1 = 0;
    vector<pair<char, char> > p;

    scanf("%d", &t);
    while (t--) {
        flag = 0;
        flag1 = 0;
        scanf("%d\n", &n);
        for (int i = 0; i < n; i++) {
            scanf("%c %c\n", &a, &b);
            p.push_back(make_pair(a, b));
        }
        cin >> s;
        s1 = s;
        for (int j = 0; j < n; j++) {
            for (unsigned int i = 0; i < s.size(); i++) {
                if (s[i] == p[j].first && s[i] == s1[i]) {
                    s[i] = p[j].second;
                }
            }
        }
        //cout << s << endl;
        int i = 0;
        int sz = s.size();
        while (i < sz && s[i] == '0') {
        	i++;
        }
        if (i < sz) {
        	while (i < sz && s[i] != '.') {
        		printf("%c", s[i]);
        		flag = 1;
        		i++;
        	}
        	if (i < sz) {
        		i++;
        		while (i < sz) {
        			c = 0;
        			while (i < sz && s[i] == '0') {
        				c++;
        				i++;
        			}
        			if (i < sz) {
        				if (!flag1) {
        					printf("%c", '.');
        					flag = 1;
        					flag1 = 1;
        				}
        				for (int k = 0; k < c; k++) {
        					printf("%c", '0');
        				}
        				c = 0;
        				printf("%c", s[i]);
        				i++;
        			}
        		}
        	}
        }
        if (!flag) {
            printf("%c\n", '0');
        }
        p.clear();
        printf("\n");
    }

    return 0;
}
