using namespace std;
char s[N], c[5], p[5];

int main() {
    int T;
    cin >> T;
    while (T --) {
        int n;
        cin >> n;
        char hash[1005];

        memset(hash, '\0', sizeof(hash));

        while (n --) {
            scanf("%s%s", c, p);
            hash[c[0]] = p[0];
        }

        scanf("%s", s);
        int len = strlen(s);
        for (int i = 0; i < len; i++) {
            if (hash[s[i]] > 0) s[i] = hash[s[i]];
        }
//        puts(s);
        int dot = 0;
        for (int i = 0; i < len; i++) {
            if (s[i] == '.') dot = 1;
        }
        int head = len - 1, end = len -1;
        for (int i = 0; i < len; i++) {
            if (s[i] != '0') {
                head = i;
                break;
            }
        }
        if (dot) {
            while (s[end] == '0') end --;
        }
        if (s[end] == '.') end --;

        if (head > end) puts("0");
        else {
            for (int i = head; i <= end; i++) putchar(s[i]);
            puts("");
        }
    }
    return 0;
}
