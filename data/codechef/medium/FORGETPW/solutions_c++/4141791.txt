#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t,n,a=0,len;
    char ci[100],pi[100];
    static char enc_pass[1001000],dec_pass[1001000];
    cin>>t;
    for(int i=0; i<t; i++)
    {
        cin>>n;
        for(int b=0; b<n; b++)
        {
            cin>>ci[b]>>pi[b];
        }
        int ch;
        ch=getchar();
        a=0;
        while((ch=getchar())!='\n')
        {
            enc_pass[a]=dec_pass[a]=ch;
            a++;
        }
            enc_pass[a]=dec_pass[a]='\0';
            len=a;
                int j=0;
                while(j<a)
                {
                    int k=0;
                    while(k<n)
                    {
                        if(enc_pass[j]==ci[k])
                        {
                            dec_pass[j]=pi[k];
                             break;
                        }
                            k++;
                    }
                    j++;
                }

         int count_l=0,count_r=0,flag=0;

         for(int j=0; j<len; j++)
             if(dec_pass[j]=='.')
             {
                 flag=1;
                 break;
             }

        for(int j=0; j<len; j++)
            if (dec_pass[j]=='0')
                 count_l++;
            else
                break;

        for(int j=len-1; j>=0&&flag==1; j--)
            if(dec_pass[j]=='0')
                 count_r++;

            else if(dec_pass[j]=='.')
                    {
                        count_r++;
                        break;
                    }
            else
                break;

            int spcl=1;
            for(int j=count_l; j<len-count_r; j++)
                {
                        spcl=0;
                        cout<<dec_pass[j];
                }

                if(spcl==1)
                    cout<<"0";

                    cout<<"\n";
      }
}
