using namespace std;
char rule[127];
char S[1000002];

int main(){
    int T, N;
    scanf("%d", &T);
    for(int t = 0; t < T; t++){
        scanf("%d", &N);
        memset(rule, 0, sizeof(rule));
        for(int j = 0; j < N; j++) {
        	char c1, c2;
        	scanf(" %c %c", &c1, &c2);
        	rule[c1] = c2;
        }


        scanf("%s", S);

        N = strlen(S);
        for(int j = 0; j < N; j++){
        	if((rule[S[j]] != 0)) {
        		S[j] = rule[ S[j]];
        	}
        }
//        calc(S, strlen(S), res);
//        cout << res << endl;
		int i = 0;
		int j = N - 1;
		int k;

		while (i <= j && S[i] == '0') i++;

		if (i <= j) {
			for (k = i; k <= j; k++)
				if (S[k] == '.') break;
			if (k <= j) {
				while (i <= j && S[j] == '0') j--;
			}
		}

		if (i > j) S[j = i] = '0';
		else if (i == j && S[i] == '.') S[i] = '0';
		else if (S[j] == '.') j--;

		S[j + 1] = 0;
		printf("%s\n", S + i);
    }
    return 0;
}
