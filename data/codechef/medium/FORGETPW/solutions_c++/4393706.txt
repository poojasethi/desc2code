#include <stdio.h>
#include <string.h>

using namespace std;
char rule[127];
char S[1000002];
char c1, c2;
int i, j , k , t;
int main(){
    int T, N;
    scanf("%d", &T);
    for(t = 0; t < T; t++){
    	memset(rule, 0, sizeof(rule));
    	scanf("%d", &N);

        for(j = 0; j < N; j++) {

        	scanf(" %c %c", &c1, &c2);
        	rule[c1] = c2;
        }


        scanf("%s", S);
        N = strlen(S);

        for(j = 0; j < N; j++){
        	if((rule[S[j]] != 0)) {
        		S[j] = rule[ S[j]];
        	}
        }
//        calc(S, strlen(S), res);
//        cout << res << endl;
		i = 0;
		j = N - 1;


		while (i <= j && S[i] == '0') i++;

		if (i <= j) {
			for (k = i; k <= j; k++)
				if (S[k] == '.') break;
			if (k <= j) {
				while (i <= j && S[j] == '0') j--;
			}
		}

		if (i > j) S[j = i] = '0';
		else if (i == j && S[i] == '.') S[i] = '0';
		else if (S[j] == '.') j--;

		S[j + 1] = 0;
		printf("%s\n", S + i);
    }
    return 0;
}
