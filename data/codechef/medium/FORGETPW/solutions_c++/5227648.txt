#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int i,j,t,n,r[127],len,dot;
    char c,p,s[1000009];
    scanf("%d",&t);
    while(t--)
    {
              scanf("%d",&n);
              getchar();
              for(i=0;i<127;i++)
              r[i]=i;
              for(i=0;i<n;i++){
              scanf("%c %c",&c,&p);
              getchar();
              r[c]=p;
              }
              gets(s);
              dot=0;
              for(i=0;s[i]!='\0';i++){
              s[i]=r[s[i]];
              if(s[i]=='.')
              dot=1;}
              len=--i;
              for(i=0;s[i]=='0' && i<=len;i++);
              j=len;
              if(dot==1)
              for(;s[j]=='0' && j>=i;j--);
              if((s[i]=='.' && s[j]=='.')||(j<i))
              printf("0\n");
              else if(s[i]!='.' && s[j]=='.'){
                   s[j]='\0';
                   printf("%s\n",s+i);}
              else{
                   s[++j]='\0';
                   printf("%s\n",s+i);}
    }
    return 0;
}
