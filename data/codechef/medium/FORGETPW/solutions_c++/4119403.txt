#include<iostream>
#include<map>
#include<string>

using namespace std;

int main()
{
    long int t, n;
    char c, p;
    string s;
    map<char, char> m;

    cin >> t;

    while (t--) {
        cin >> n;

        while (n--) {
            cin >> c >> p;
            m.insert(pair<char, char>(c, p));
        }

        cin >> s;

        long int sz = s.length();

        for (int i = 0; i < sz; i++) {
            if (m[s[i]])
                s[i] = m[s[i]];
        }

        int f = 0;

        for (int i = 0; i < sz; i++) {
            if (s[i] == '.') {
                f = 1;
                break;
            }
        }
   long int k  = 0;
        while(s[k] == '0') {
            k++;
            //s.erase(0, 1);
        }
        s.erase(0,k);
        k = s.length() - 1;
        long int d = 0;
        if (f) {
            while(s[k] == '0') {
                k--;d++;
                //s.erase(s.length()-1, 1);
            }
            //cout << "d = " << d << endl;
            s.erase(s.begin() + s.length() - d, s.end());

            if (s[s.length()-1] == '.') {
                s.erase(s.length()-1, 1);
            }
        }

        if (!s.length()) {
            s.insert(s.begin(), '0');
        }

        cout << s << endl;

        m.clear();
    }

    return 0;
}
