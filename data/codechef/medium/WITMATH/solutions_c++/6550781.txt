typedef long long ll;
long long mulmod(ll a, ll  b , ll mod)
{
    a=a%mod;
    b=b%mod;
    long double  res=a;
    res=res*b;
    a=a*b;
    ll c=res/mod;
    a=a-c*mod;
    if(a<c)
       a+=mod;
    return a;
}
ll pwr(ll a ,ll b, ll n)
{
    ll ans=1;
    if(b==0)
        return 1;
    else
    {
        while(b)
        {
            if(b&1)
            ans=mulmod(ans,a,n);
            b/=2;
            a=mulmod(a,a,n);
        }
    }
    return ans%n;
}
bool test(long long num)
{
    if(num<2)
        return 0 ;
    if(num%2==0 && num!=2)
        return 0;
    long long temp=num-1;
    while(!(temp&1)) {
        temp/=2;
    }
    int iter=2;
    while(iter--)
    {
        ll a=((rand())%(num-1))+1;
        ll d=temp;
        ll x=pwr(a,d,num);
        while(d!=num-1 && x!=1 && x!=num-1)
        {
            x=mulmod(x,x,num);
            d*=2;
        }
        if(x!=num-1 && d%2==0)
            return 0;
    }
    return 1;
}
