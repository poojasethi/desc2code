#include<iostream>
#include<cstdio>
#include<cstdlib>
#define lld long long int
using namespace std;

lld big_mult(lld a,lld b,lld mod)
{
lld c;
long double ans;
a=a%mod;
b=b%mod;
ans=(long double)a*b;
a=a*b;
c=(lld)(ans/mod);
a=a-c*mod;
a=a%mod;
if(a<0)
a=a+mod;
return a;
}


lld fast_mod(lld b,lld e,lld mod)
{
	lld r=1;
	while(e>0)
    {
    	if(e&1)
    	 r=big_mult(r,b,mod);
    	 e=e>>1;
    	 b=big_mult(b,b,mod);
    	 
	}
	return r;
}

bool MRT(lld n,int k)
{
	if(n<2)
	 return false;
	else if(n==2)
	 return true;
	else
	{
		int j,s=0;
		lld p;
		p=n-1;
		//cout<<p<<endl;
		while(p%2==0)
		{p=p/2;s++;}
		int y,flag;
		for(j=0;j<k;j++)
		{
		    flag=0;	
			lld a=rand()%(n-1)+1;
			//cout<<a<<endl;
			lld x=fast_mod(a,p,n);
		    //printf("%lld\n",x);
		    if(x==1 || x==n-1) {continue;}
		    
		    for(y=0;y<s-1;y++)
		    {
		    	x=fast_mod(x,2,n);
		    	if(x==1) return false;
		    	if(x==n-1) 
				{
				flag=1;break;}
		    	
			}
			if(flag==1)
			 {flag=0;continue;}
			return false;
			
			
		}
		return true;
	}
}
