#include<stdlib.h>
#define ll long long
ll mulmod(ll a,ll b,ll m)
{
    	a%=m;
b%=m;
long double res = a;
res *= b;
long long c = (long long)(res / m);
a *= b;
a -= c * m;
a %= m;
if (a < 0) a += m;
return a;
}
ll mod(ll a,ll b,ll m)
{
    ll x=1,y=a%m;
    while(b>0)
    {
        if(b%2!=0)
        {
            x=mulmod(x,y,m);
        }
        y=mulmod(y,y,m);
        b=b/2;
    }
    return x;
}
ll millerrabin(ll n)
{ if((n%2==0 || n<2) && n!=2) return 0;
  ll s=n-1;
  ll count=0;
  while(s%2==0)
       { count++;
         s/=2;
       }
  for(int i=0;i<18;i++)
     { ll a=rand()%(n-1)+1,temp=s;
       ll ans=mod(a,s,n);
       if(ans==1||ans==-1)
        { 
            continue;
        }
        while(temp!=(n-1)&&ans!=(n-1))
        {
            ans=mulmod(ans,ans,n);
            temp=temp*2;
        }
        if(ans!=(n-1))
        {

     }     
  return 1;   
}
