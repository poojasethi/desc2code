lli mod(lli a,lli b,lli c)
{
	lli x=1,y=a;
	while(b>0)
	{
		if(b%2==1)
			x=mulmod(x,y,c);

		y=mulmod(y,y,c);
		b=b/2;
	}
	return (x%c);
}

lli mulmod(lli a,lli b,lli c)
{
	lli x=0,y=a%c;
	while(b>0)
	{
		if(b%2==1)
			x=(x+y)%c;

		y=(y*2)%c;
		b=b/2;
	}
	return (x%c);
}
