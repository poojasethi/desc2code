ull mul(ull x, ull y, ull m){
    ull sum=0;
    while (y){
        if (y%2) sum=(sum+x)%m;
        x=x*2%m;
        y/=2;
    }
    return sum;
}

ull power(ull a, ull n, ull m){
    ull ans=1;
    while (n){
        if (n%2) ans=mul(ans,a,m);
        a=mul(a,a,m);
        n/=2;
    }
    return ans;
}
