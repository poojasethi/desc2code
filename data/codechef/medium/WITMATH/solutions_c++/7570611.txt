#include<bits/stdc++.h>
#define lli unsigned long long int
using namespace std;
lli mulmod(lli ,lli ,lli );
lli modulo(lli a,lli b,lli c)
{
	lli y=a,x=1;
	while(b>0)
	{
		if(b&1)
		{
			x=mulmod(x,y,c);
		}
		y=mulmod(y,y,c);
		b >>= 1;
	}
	return x;
}
lli mulmod(lli a,lli b,lli c)
{
	lli y=a%c,x=0;
	while(b>0)
	{
		if(b&1)
			x=(x+y)%c;
		y=(y<<1)%c;
		b >>= 1;
	}
	return x;
}
bool Miller(lli p,int it){
	if(p<2) return false;
	if(p==2) return true;
	if((p&1)==0) return false;

	lli s = p-1;
	while(s%2==0) s >>= 1;

	while(it--){
		lli a = rand()%(p-1)+1,temp = s;
		lli mod = modulo(a,temp,p);

		if(mod==-1 || mod==1) continue;

		while(temp!=p-1 && mod!=p-1){
			mod = mulmod(mod,mod,p);
			temp <<= 1;
		}

		if(mod!=p-1) return false;
	}

	return true;
}
int main()
{
	lli p,a,b,c,i,j,k,m,n,t;
	//srand(time(0));
	scanf("%llu",&t);
	while(t--)
	{
		scanf("%llu",&p);

		while(1)
		{
			if(Miller(p,5)!=1)
				p--;
			else
				break;
		}
		printf("%llu\n",p);

	}
}
