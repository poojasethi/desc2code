#include<iostream>
#include<cstdlib>
using namespace std;

long long modu(long long int a,long long int b,long long int c)
{
long long int x=0,y=a%c;
while(b>0)
{
if(b%2==1)
{
x=(x+y)%c;
}
y=(y+y)%c;
b=b/2;
}
return x%c;
}

long long int modulo(long long int a,long long int b,long long int c)
{
long long int x=1,y=a;
while(b>0)
{
if(b%2==1)
{
x= modu(x,y,c);
}
y= modu(y,y,c);
b=b/2;
}
return x%c;
}
bool check(long long p,int iteration)
{
    if(p<2)
    return false;
    if(p!=2&&p%2==0)
        return false;
    long long s=p-1;
    while(s%2==0)
    {
        s=s/2;
    }
    for(int i=0;i<iteration;i++)
    {
        long long a=rand()%(p-1)+1,temp=s;
        long long mod=modulo(a,temp,p);
        while(temp!=p-1&&mod!=1&&mod!=p-1)
        {
            mod=modu(mod,mod,p);
            temp *= 2;
        }
        if(mod!=p-1&&temp%2==0){
            return false;
        }
    }
    return true;
}

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        long long int n;
        cin>>n;
        for(long long int i=n;;i--)
        {
            if(check(i,3))
            {
                cout<<i<<endl;
                break;
            }
        }
    }
    return 0;
}
