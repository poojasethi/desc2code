#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll mulmod(ll a,ll b,ll m)
{
    	a%=m;
b%=m;
long double res = a;
res *= b;
long long c = (long long)(res / m);
a *= b;
a -= c * m;
a %= m;
if (a < 0) a += m;
return a;
}
ll modulo(ll a,ll b,ll m)
{
    ll x=1,y=a%m;
    while(b>0)
    {
        if(b%2!=0)
        {
            x=mulmod(x,y,m);
        }
        y=mulmod(y,y,m);
        b=b/2;
    }
    return x;
}
bool millerrabin(ll p)
{
    ll i,a,temp,mod;
    if(p<2)
    {
        return false;
    }
    if(p==2)
    {
        return true;
    }
    if(p%2==0)
    {
        return false;
    }
    ll s=p-1;
    while(s%2==0)
    {
        s=s/2;
    }
    for(i=0;i<18;i++)
    {
        a=rand()%(p-1)+1;
        temp=s;
        mod=modulo(a,temp,p);
        if(mod==1||mod==-1)
        {
            continue;
        }
        while(temp!=(p-1)&&mod!=(p-1))
        {
            mod=mulmod(mod,mod,p);
            temp=temp*2;
        }
        if(mod!=(p-1))
        {
            return false;
        }
    }
    return true;
}
int main()
{
    ios_base::sync_with_stdio(false);
    ll t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        while(1)
        {
            if(millerrabin(n)==true)
            {
                cout<<n<<"\n";
                break;
            }
            --n;
        }
    }
    return 0;
}
