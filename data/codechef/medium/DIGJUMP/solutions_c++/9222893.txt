#include<bits/stdc++.h>
using namespace std;
int visited[100005];
int dist[100005];
vector<int>myvector[10];
int main()
{
    int i;
    string s;
    cin>>s;
    for(i=0;i<s.size();i++)
    {
        int val=s[i]-'0';
        myvector[val].push_back(i);
    }

    visited[0]=1;
    queue<int>myqueue;
    myqueue.push(0);
    while(!myqueue.empty())
    {
        int idx=myqueue.front();
        if(idx==(s.size()-1))
            break;

        myqueue.pop();

        int val=s[idx]-'0';

        for(i=0;i<myvector[val].size();i++)
        {
            int nidx=myvector[val].at(i);
            if(!visited[nidx])
            {
                dist[nidx]=dist[idx]+1;
                myqueue.push(nidx);
                visited[nidx]=1;
            }
        }
        myvector[val].clear();
        if(idx-1>=0 && !visited[idx-1])
        {
            dist[idx-1]=dist[idx]+1;
            visited[idx-1]=1;
            myqueue.push(idx-1);
        }
        if(idx+1<s.size() && !visited[idx+1])
        {
            dist[idx+1]=dist[idx]+1;
            visited[idx+1]=1;
            myqueue.push(idx+1);
        }
    }
    cout<<dist[s.size()-1];
    return 0;
}
