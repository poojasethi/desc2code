#include<iostream>
#include<stdio.h>
#include<map>
#include<string.h>
#include<math.h>
#include<vector>
#include<queue>
#include<iomanip>
#include<algorithm>
#define FOR(i,a,b) for(int i=(a);i<=(b);++i)
#define NFOR(i,a,b) for(int i=(a);i>=(b);--i)
#define pb push_back
#define ll long long
using namespace std;
char s[100006];
int l;
int visited[100006]={0};
int v[10]={0};
int bfs(int x)
{
    queue<int>q;
    q.push(x);
    visited[x]=0;
    int r;
    while(!q.empty())
    {
        r=q.front();
        q.pop();
        if(!v[s[r]])
        {
            v[s[r]]=1;
        FOR(i,0,l-1)
        {
            if(s[i]==s[r]&&!visited[i])
            {
                q.push(i);
                visited[i]=visited[r]+1;
            }
        }
        }
        if((r+1)<l &&!visited[r+1])
            {visited[r+1]=visited[r]+1;q.push(r+1);}
        if(r-1>=0&&!visited[r-1])
            {visited[r-1]=visited[r]+1;q.push(r-1);}
        if(r==l-1)
            return visited[r];
    }

}
int main()
{
    scanf("%s",s);
    l=strlen(s);
    printf("%d\n",(bfs(0)-1));
    return 0;
}
