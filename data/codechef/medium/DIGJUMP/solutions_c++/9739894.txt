#include<bits/stdc++.h>
using namespace std;
 bool visited[100001];
 bool arr1[10];
 char s[100001];
int main()
{
    int l,i,min1;
    scanf("%s",s);
    min1=strlen(s)+1;
    l=strlen(s);
    set<int> arr[10];
    for(i=0;i<l;i++)
   arr[s[i]-48].insert(i);
    queue <pair<int,int> > q;
    q.push(make_pair(0,0));
    visited[0]=1;
    while(!q.empty())
    {

        i=q.front().first;
        if(i==l-1&&q.front().second<min1)
        {
           min1=q.front().second;

        }
        if(!arr1[s[i]-48])
        {
            arr1[s[i]-48]=1;
            //cout<< s[i]-48<<" hlo\n";
       set<int> :: iterator it;
       for(it=arr[s[i]-48].begin();it!=arr[s[i]-48].end();it++)
       {
        if(!visited[*it])
        q.push(make_pair(*it,q.front().second+1));
       }
        }
        if(i-1>=0&&(!visited[i-1]))
        {

                q.push(make_pair(i-1,q.front().second+1));

                visited[i-1]=1;

        }
         if(i+1<l&&(!visited[i+1]))
        {

                q.push(make_pair(i+1,q.front().second+1));
                visited[i+1]=1;


        }
        q.pop();
    }
    printf("%d",min1);
    return 0;
}
