#include<iostream>
#include<cstring>
#include<string>
using namespace std;
int mi(int a, int b)
{
	if (a<b)
		return a;
	else
		return b;
}
int main()
{
	string s;
	int dist[100005], a[10];
	cin.clear();
	cin >> s;
	dist[0] = 0;
	for (int i = 1; i<s.size() + 1; i++)
	{
		dist[i] = 100005;
	}
	for (int i = 0; i<10; i++)
	{
		a[i] = -1;
	}
	a[s[0] - '0'] = 0;
	for (int i = 1; i<s.size(); i++)
	{
		if (a[s[i] - '0'] == -1)
		{
			dist[i] = dist[i - 1] + 1;
			a[s[i] - '0'] = dist[i];
		}
		else
		{
			dist[i] = mi(dist[i - 1], a[s[i] - '0']) + 1;
			if (dist[i]<a[s[i] - '0'])
			{
				a[s[i] - '0'] = dist[i];
			}


		}
	}

	bool b = true;
	for (int k = 1; k<s.size() && b; k++)
	{
		b = false;
		for (int i = 1; i<s.size(); i++)
		{
			if (dist[i]>mi(dist[i - 1], dist[i + 1]) + 1)
			{
				dist[i] = mi(dist[i - 1], dist[i + 1]) + 1;
				if (dist[i]<a[s[i] - '0'])
					a[s[i] - '0'] = dist[i];
				b = true;
			}
			if (dist[i]>a[s[i] - '0'] + 1)
			{
				dist[i] = a[s[i] - '0'] + 1;
				b = true;
			}

		}
	}
	cout << endl;

	cout << dist[s.size() - 1];

}
