#include<stdio.h>
#include<queue>
#include<string.h>
#include<vector>
using namespace std;
typedef pair<int,int> ii;
char str[100010];
vector<int>V[10];
int visited[100010]={0};
void bfs()
{
	int n=strlen(str);
	int indx,c;
	ii p;
	queue< pair<int,int> >Q;
	Q.push(ii(0,0));
	visited[0]=1;
	while(!Q.empty())
	{
		p=Q.front();
		indx=p.first,c=p.second;
		Q.pop();
		//printf("pop %c\n",str[indx]);
		visited[indx]=1;
		if(indx==n-1)
		{
			printf("%d\n",c);
			return;
		}
		
		int xx;
		int mm=str[indx]-'0';
		int k=V[mm].size();
		for(xx=0;xx<k;xx++)
		{
		int indxx=V[mm][xx];
		if(visited[indxx]==0)
		{
		Q.push(ii(indxx,c+1));
		visited[indxx]=1;
		//for(xx=0;xx<k;xx++)	
		}
        }
        V[mm].clear();
		if(indx-1>=0&&visited[indx-1]==0)
		{
			Q.push(ii(indx-1,c+1));
			visited[indx-1]=1;
		}
		if(indx<n&&visited[indx+1]==0)
		{
			Q.push(ii(indx+1,c+1));
			visited[indx+1]=1;
		}
				
	}
}
int main()
{
	int i,k;
	
	scanf("%s",str);
	//int i=0;
	i=0;
	while(str[i]!='\0')
	{
	 k=str[i]-'0';	
	 V[k].push_back(i);
	 i++;
	}
//	printf("hi");
	bfs();
}
