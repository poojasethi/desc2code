import sys

def main():
  for (r, s) in testcases():
    if pseudo_granama(r, s):
      print "NO"
    else:
      print "YES"

def testcases(cin = sys.stdin):
  nc = int(cin.next())
  for _ in xrange(nc):
    r, s = cin.next().split()
    yield (r, s)

def pseudo_granama(r, s):
  r, s = list(r), list(s)
  r.sort(); s.sort()

  return (r != s) and (set(r) == set(s))

if __name__ == "__main__": sys.exit(main())
