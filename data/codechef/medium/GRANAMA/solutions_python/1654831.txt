import sys
t = int(sys.stdin.readline())
for i in range(t) :
  (a,b) = sys.stdin.readline().rstrip().split()
  seta = set([]);
  setb = set([]);
  for c in a : seta.add(c)
  for c in b : setb.add(c)
  lista = list(a)
  listb = list(b)
  lista.sort()
  listb.sort()
  if   seta != setb   : print "YES"
  elif lista != listb : print "NO"
  else                : print "YES"
