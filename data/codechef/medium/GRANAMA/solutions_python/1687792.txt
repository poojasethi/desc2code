def main():
    import sys
    tokenizedInput = sys.stdin.read().split()
    k=int(tokenizedInput[0])
    readAt = 1
    for n in range(k):
        recipe1, recipe2 = tokenizedInput[readAt:readAt+2]
        sRecipe1, sRecipe2 = set(recipe1), set(recipe2)
        if sRecipe1 == sRecipe2:
            if sorted(recipe1) == sorted(recipe2):
                print 'YES'
            else:
                print 'NO'
        else:
            print 'YES'
        readAt = readAt + 2

main()
