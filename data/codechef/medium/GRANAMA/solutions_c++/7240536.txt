#include <iostream>
#include <string.h>
using namespace std;
//	long int a[400000000];
int main() {
    int t;
    cin>>t;
    char a[1001],b[1001];
    for(int i=0;i<t;i++){
        int c[150]={0},d[150]={0};
    cin>>a>>b;
    for(int j=0;a[j]!='\0';j++){
        c[a[j]]++;
    }
    
    for(int j=0;b[j]!='\0';j++){
        d[b[j]]++;
    }
    int count=0,count2=0;
    for(int j=0;j<150;j++){
        if(c[j]==0&&d[j]!=0)
        {count++;
        break;
        }
        if(c[j]!=0&&d[j]==0)
        {count++;
        break;
        }
    }
        for(int j=0;j<150;j++){
        if(c[j]!=d[j])
        {count2++;
        break;
        }
    }
    if(count==1)
    cout<<"YES"<<endl;
    else if(count2==1)
    cout<<"NO"<<endl;
    else
    cout<<"YES"<<endl;
    }
}
