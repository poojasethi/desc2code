#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;

    char str1[1001],str2[1001];
    cin>>t;
    while(t--)
    {
        int flag=1;
        int p1[26]= {0};
        int p2[26]= {0};
        cin>>str1>>str2;
        for(int i=0; str1[i]!='\0'; i++)
            p1[str1[i]-'a']++;
        for(int i=0; str2[i]!='\0'; i++)
            p2[str2[i]-'a']++;
        for(int i=0; i<26; i++)
        {
            if(p1[i]==p2[i])
                continue;
                if((p1[i]==0 && p2[i]!=0) || (p2[i]==0 && p1[i]!=0))
                {
                    flag=2;
                    break;
                }
                else
                {
                    flag=0;

                }
            }
          if(flag)

                cout<<"YES"<<endl;
                else
                    cout<<"NO"<<endl;



        }
        return 0;
    }
