#include<iostream>
#include<stdio.h>
#include<string.h>
#include<string>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char s1[1005],s2[1005];
        scanf("%s%s",s1,s2);
        int check[26][2];
        int i;
        for(i=0;i<26;i++)
            check[i][0]=check[i][1]=0;
        i=0;
        while(s1[i]!='\0')
            check[s1[i++]-97][0]++;
        i=0;
        while(s2[i]!='\0')
            check[s2[i++]-97][1]++;
        int flag=0;
        for(i=0;i<26;i++)
        {
            if(check[i][0]&&check[i][1]&&check[i][0]!=check[i][1])
            {
                flag=1;
            }
            if((check[i][0]&&!check[i][1])||(check[i][1]&&!check[i][0]))
                break;
        }
        if(i==26)
        {
            if(flag)
                printf("NO\n");
            else
                printf("YES\n");
        }
        else
            printf("YES\n");
    }
    return 0;
}
