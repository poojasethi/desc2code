bool chefcheck(int a, int b)
{
	if(a == 0 && b > 0)
		return false;
	else if(b == 0 && a > 0)
		return false;
	return true;
}

int main()
{
	#ifndef ONLINE_JUDGE
	freopen("GRANAMA.txt", "r", stdin);
	#endif
	int t, arr1[26], arr2[26];
	bool chef, actual;
	cin >> t;
	string r, s;
	while(t--)
	{
		cin >> r >> s;
		chef = true;
		actual = true;
		for(int i=0; i<26; i++)
			arr1[i] = arr2[i] = 0;
		for(int i=0; i<r.length(); i++)
			arr1[r[i]-'a']++;
		for(int i=0; i<s.length(); i++)
			arr2[s[i]-'a']++;
		for(int i=0; i<26; i++)
		{
			if(chefcheck(arr1[i], arr2[i]))
			{
				if(arr1[i] != arr2[i])
					actual = false;
			}
			else
			{
				chef = false;
				break;
			} 
		}
		if(!chef)
			cout << "YES" << endl;
		else if(chef && actual)
			cout << "YES" << endl;
		else 
			cout << "NO" << endl;
	}
	return 0;
}
