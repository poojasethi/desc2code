#include <bits/stdc++.h>
using namespace std;
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    string s,s1;
    cin>>s>>s1;
    int a[1111]={0},b[1111]={0};
    for (int i = 0; i <s.size(); ++i)
    {
      a[s[i]-'a']++;
    }
    for (int i = 0; i <s1.size(); ++i)
    {
      b[s1[i]-'a']++;
    }
    bool find=false;
    for (int i = 0; i <26; ++i)
    {
        if(a[i]!=b[i])find=true;
    }
    for (int i = 0; i <26; ++i)
    {
        if((!a[i]) && (b[i]) || ((a[i])&&(!b[i])) )
          find=false;
    }
    if(find)puts("NO");
    else puts("YES");
  }
  return 0;
}
