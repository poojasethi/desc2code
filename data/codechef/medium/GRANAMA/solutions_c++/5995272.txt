#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>

using namespace std;

int main()
{
    int t;
    bool granama = true,chefclaim = true;
    scanf("%d",&t);
    char R[1002],S[1002];
    int lr,ls;
    int Rwala[26],Swala[26];
    while(t--)
    {
      granama = true,chefclaim = true;
      scanf("%s %s",&R,&S);
      for(int i=0;i<26;i++)
      {
          Rwala[i]=0;Swala[i]=0;
      }
      lr = strlen(R);ls = strlen(S);
      for(int i=0;i<lr;i++)
          Rwala[(int)R[i]-'a']++;
      for(int i=0;i<ls;i++)
          Swala[(int)S[i]-'a']++;

     /* for(int i=0;i<26;i++)
        printf("%d ",Rwala[i]);
      printf("\n");
      for(int i=0;i<26;i++)
        printf("%d ",Swala[i]);*/

      for(int i=0;i<26;i++)
      {
        if(!((Rwala[i]==0 && Swala[i]==0) || (Rwala[i] != 0 && Swala[i] != 0)))
            {
                chefclaim = false;
                break;
            }
      }
      for(int i=0;i<26;i++)
      {
          if(Rwala[i] != Swala[i])
          {
              granama = false;
              break;
          }
      }
      if(chefclaim)
      {
          if(granama)
            printf("YES\n");
          else printf("NO\n");
      }
      else
      {
          if(granama) printf("NO\n");
          else printf("YES\n");
      }
    }
    return 0;
}
