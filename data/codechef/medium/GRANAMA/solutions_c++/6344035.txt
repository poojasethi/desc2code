#include <iostream>
#include <cstring>
using namespace std;

int main() {
    int tc, i, j, k;
    string r, s;
    int countr[26], counts[26];
    bool chef, truth;

    cin >> tc;

    while (tc--) {
        cin >> r >> s;

        chef = true; truth = true;

        memset(counts, 0, sizeof counts);
        memset(countr, 0, sizeof countr);

        for (i = 0; i < s.length(); i++) {
            counts[s[i] - 97]++;
        }
        for (i = 0; i < r.length(); i++) {
            countr[r[i] - 97]++;
        }

        for (i = 0; i < 26; i++) {
            if ((counts[i] > 0 && countr[i] == 0) || (counts[i] == 0 && countr[i] > 0)) {
                chef = false;
            }
        }

        for (i = 0; i < 26; i++) {
            if (counts[i] != countr[i]) {
                truth = false;
            }
        }

        if (chef == truth) {
            cout << "YES\n";
        } else {
            cout << "NO\n";
        }

    }

    return 0;
}
