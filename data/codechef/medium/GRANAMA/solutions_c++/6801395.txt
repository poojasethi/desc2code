    #include <cstdio>
    #include <iostream>
    #include <string>
    #include <cstdlib>
    #include <cstring>
     
    using namespace std;
     
     
    int main()
    {
     
    int t, i;char str1[1003]; char str2[1003];int freq1[27]; int freq2[27];
    int l1, l2;int chef , us;
    cin >> t;
    while(t--)
    {
    chef = 0 ; us = 0 ;
    memset(freq1 , 0 , 27 * sizeof(int));
    memset(freq2 , 0 , 27 * sizeof(int));
    scanf("%s" , str1);
    scanf("%s" , str2);
    l1 = strlen(str1);
    l2 = strlen(str2);
    for(i = 0 ; i < l1 ; i++)
    {
    freq1[str1[i] - 'a']++;
    }
     
    for(i = 0 ; i < l2 ; i++)
    {
    freq2[str2[i] - 'a']++;
    }
     
    for(i = 0 ; i < 26 ; i++)
    {
    if(freq1[i] > 0 || freq2[i] > 0)
    {
    if(!(freq1[i] > 0 && freq2[i] > 0))
    chef = 1;
    if(!(freq1[i] == freq2[i]))
    us = 1;
    }
    }
     
    if(chef != us)
    cout << "NO" << endl;
    else
    cout << "YES" << endl;
    }
    return 0;
    }
