#include<stdio.h>
char a[1001],b[1001];
int c[26],d[26];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s%s",a,b);
        int i,k=0,f=0;
        for(i=0;i<26;i++)
        {
            c[i]=0;
            d[i]=0;
        }
        for(i=0;a[i]!='\0';i++)
            c[a[i]-97]++;
        for(i=0;b[i]!='\0';i++)
            d[b[i]-97]++;

        for(i=0;i<26;i++)
        {
            if((c[i]==0&&d[i]>0)||(c[i]>0&&d[i]==0))
                f=1;
            else
                if((c[i]>0&&d[i]>0)&&(c[i]!=d[i]))
                     k=1;
        }
        if(f==1)
            printf("YES\n");
        else
            if(k==1)
            printf("NO\n");
        else
            printf("YES\n");
    }
    return 0;
}
