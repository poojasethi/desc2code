#include<bits/stdc++.h>
using namespace std;
int main(){
	int t, check[27], check2[27],flag;
	char a[1005], b[1005];
	cin>>t;
	while(t){
		cin>>a>>b;
		for(int i = 0; i < 27; i++){
			check[i] = 0;
			check2[i] = 0;
		}
		for(int i = 0; a[i] != '\0'; i++ ) {
			int c = a[i] - 97;
			check[c]++;
		}
		flag = 1;
		for(int i = 0; b[i] != '\0'; i++ ) {
			int c = b[i] - 97;
			check2[c]++;
		}
		for(int i = 0; i < 27; i++){
			if (check[i] != 0 && check2[i] != 0){
				continue;
			} else if ( check[i] == 0 && check2[i] == 0 ){
				continue;
			} else {
				flag = 0;
				break;
			}
		}
		if(flag == 0 ){
			cout<<"YES\n";
		} else {
			for(int i = 0; i < 27; i++){
				if (check[i] == check2[i]){
					continue;
				} else {
					flag = 2;
					break;
				}
			}
		}
		if (flag == 1){
			cout<<"YES\n";
		} else if(flag == 2)
			cout<<"NO\n";
		t--;
	}
return 0;
}
