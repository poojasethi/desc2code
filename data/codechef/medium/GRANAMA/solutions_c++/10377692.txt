#include<iostream>
using namespace std;
int main()
{
	int t,k,r,i,arr[26];
	char str1[1001],str2[1001];
	cin>>t;
	while(t--)
	{
		k=0;
		cin>>str1;
		cin>>str2;
		for(i=0;i<26;i++)
		arr[i]=0;
		for(i=0;str1[i]!='\0';i++)
		arr[str1[i]-'a']++;
		for(i=0;str2[i]!='\0';i++)
		{
			arr[str2[i]-'a']--;
			if(arr[str2[i]-'a']<0)
			{
				k=1;
				break;
			}
		}
		for(i=0;i<26;i++)
		{
			if(arr[i]!=0)
			{
				k=1;
				break;
			}
		}
		r=0;
		for(i=0;i<26;i++)
		arr[i]=0;
		for(i=0;str1[i]!='\0';i++)
		arr[str1[i]-'a']=1;
		for(i=0;str2[i]!='\0';i++)
		{
			if(arr[str2[i]-'a']==0)
			{
				r=1;
				break;
			}
		}
		if(r==0)
		{
			for(i=0;str2[i]!='\0';i++)
			arr[str2[i]-'a']=0;
			for(i=0;i<26;i++)
			{
				if(arr[i]==1)
				{
					r=1;
					break;
				}
			}
		}
		if(k==r)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	}
	return 0;
}
