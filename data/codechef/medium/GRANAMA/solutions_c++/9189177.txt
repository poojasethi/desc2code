#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int t,i,f,g;
    string a,b;
    scanf("%d",&t);

    while(t--)
    {
        int al1[26]={0},al2[26]={0};
        cin>>a>>b;
        for(i=0;i<a.length();i++)
        {
            al1[a[i]-97]++;


        }
                for(i=0;i<b.length();i++)
                {
                    al2[b[i]-97]++;

                }

        g=1;
        f=1;
        for(i=0;i<26;i++)
        {
            if(al1[i]==0 && al2[i]!=0 || al2[i]==0 && al1[i]!=0  )
                g=0;
            if(al1[i]!=al2[i])
                f=0;

        }
         if(g==f)
            printf("YES\n");
        else
            printf("NO\n");
    }

}
