#include<iostream>
#include<cstring>
using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		char *s,*r;
		s=new char[1000];
		r=new char[1000];
		cin>>s;
		cin>>r;
		int *s1,*s2,*r1,*r2;
		s1=new int[26];
		s2=new int[26];
		r1=new int[26];
		r2=new int[26];

		for(int i=0;i<strlen(s);i++){
			s1[s[i]-'a']++;
			s2[s[i]-'a']=1;
		}

		for(int i=0;i<strlen(r);i++){
			r1[r[i]-'a']++;
			r2[r[i]-'a']=1;
		}

		bool a1=true,a2=true;
		for(int i=0;i<26;i++)
		if(r1[i]!=s1[i]){
			a1=false;
			break;
		}

		for(int i=0;i<26;i++)
		if(r2[i]!=s2[i]){
			a2=false;
			break;
		}

		if(a1!=a2)
		cout<<"NO"<<endl;

		else
		cout<<"YES"<<endl;
	}
	return 0;
}
