#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
	int t,flag1,flag2,i;
	int arr1[125], arr2[125];
	char r[1005], s[1005];
	scanf("%d", &t);
	while(t--)
	{
		scanf("%s %s", r, s);
		flag1=flag2=1;
		
		for(i=0;i<125;i++)
		{
			arr1[i]=0;
			arr2[i]=0;
		}
		for(i=0;i<strlen(r);i++)
			arr1[r[i]]++;
		for(i=0;i<strlen(s);i++)
			arr2[s[i]]++;
//		for(i=97;i<=123;i++)
//		{
//			printf("%d%d", arr1[i], arr2[i]);
//		}	
//		printf("\n");
		for(i=97;i<=123;i++)
		{
			if(arr1[i]==0 && arr2[i]!=0)
			{
				flag1=0;
				break;
			}
			if(arr1[i]!=0 && arr2[i]==0)
			{
				flag1=0;
				break;
			}
		}
		for(i=97;i<=123;i++)
		{
			if(arr1[i]!=arr2[i])
			{
				flag2=0;
				break;
			}
		}
		if(flag1==flag2)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
