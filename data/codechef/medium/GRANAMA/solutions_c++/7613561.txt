#include<iostream>
using namespace std;

int main()
{
	int t,flag,fl;
	string s1,s2;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int hash1[256]={0},hash11[256]={0};
		int hash2[256]={0},hash22[256]={0};
		flag=0,fl=0;
		cin>>s1>>s2;
		
		for(int j=0;s1[j]!='\0';j++)
		{
			hash1[s1[j]]++;
			hash11[s1[j]]=1;
		}
		
		for(int j=0;s2[j]!='\0';j++)
		{
				hash2[s2[j]]++;
				hash22[s2[j]]=1;
		}
		for(int j=0;j<256;j++)
		{
			if(hash1[j]!=hash2[j])
			{
				flag=1;
				break;
			}
		}
		
		for(int j=0;j<256;j++)
		{
			if(hash11[j]!=hash22[j])
			{
				fl=1;
				break;
			}
			
			
		}
		
		if(flag==fl)
		cout<<"YES\n";
		else
		cout<<"NO\n";
		
	}
	
	
	return 0;
}
