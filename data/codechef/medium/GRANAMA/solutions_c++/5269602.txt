#include <iostream>
#include <cstdio>
#include <string>

using namespace std;

int main()
{
	int a[128],b[128];
	char r[1005],s[1005];
	int flag,t;

	scanf("%d",&t);
	for (int k = 0; k < t; k++) {
 		scanf("%s%s",&r,&s);
		for (int i = 0; i < 125; i++) {
			a[i] = 0;
			b[i] = 0;
		}
		for (int i = 0; r[i] !='\0'; i++) {
			a[r[i]]++;
		}
		for (int i = 0; s[i] != '\0'; i++) {
			b[s[i]]++;
		}
		flag = 0;
		for (int i = 97; i < 123; i++) {
			if ((a[i] == 0 && b[i] != 0)||(a[i] != 0 && b[i] == 0)) {
				flag = 0;
				break;
			}
			if(a[i] != b[i] && a[i]!=0 && b[i]!=0)
			{
				flag = 1;
			}
		}
		if (flag == 0)
			printf("YES\n");
		else
			printf("NO\n");
	}
}
