#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int main()
{
    int t,i,flag1[26],flag2[26];
    char s1[1001],s2[1001];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s %s",s1,s2);
        int l1=strlen(s1);
        int l2=strlen(s2);
        for(i=0;i<26;i++)
        {flag1[i]=0;flag2[i]=0;}
        for(i=0;i<l1;i++)
            flag1[s1[i]-'a']++;
        for(i=0;i<l2;i++)
            flag2[s2[i]-'a']++;
        bool ok=true;
        for(i=0;i<26;i++)
        {
            if((flag1[i]==0&&flag2[i]!=0)||(flag1[i]!=0&&flag2[i]==0))
            {
                ok=false;
                break;
            }

        }
        if(!ok)
        {
            printf("YES\n");
            continue;
        }
        int f=1;
        for(i=0;i<26&&f;i++)
        {
            if(flag1[i]!=flag2[i])
                f=0;
        }
        if(f)
        {
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }

        return 0;

    }
