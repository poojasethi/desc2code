#include<iostream>
#include<cstdio>
using namespace std;

int match(string a,string b)
{
int ar[26],ar1[26];
int i;
for(i=0;i<26;i++)
{ar[i]=0;ar1[i]=0;}
for(i=0;i<a.length();i++)
ar[a[i]-'a']++;
for(i=0;i<b.length();i++)
ar1[b[i]-'a']++;
int v1=1,v2=1;
for(i=0;i<26;i++)
{
	if(ar[i]==0 && ar1[i]!=0)
	v1=0;
	if(ar1[i]==0 && ar[i]!=0)
	v1=0;
	if(ar1[i]!=ar[i])
	v2=0;
}
if(v1==v2)
return 1;
return 0;
}
int main()
{
int T;
string a,b;
scanf("%d",&T);
while(T>0)
{
	cin>>a>>b;
	if(match(a,b))
	printf("YES\n");
	else
	printf("NO\n");
	T--;
}
}
