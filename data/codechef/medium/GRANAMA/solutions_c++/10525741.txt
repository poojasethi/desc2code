#include<bits/stdc++.h>
using namespace std;
char r[1001], s[1001];
int correct(char r[], char s[])
{
	int r_ctr[26] = {0}, s_ctr[26] = {0};
	int i;
	for(i=0;r[i]!='\0';i++)
		r_ctr[r[i]-97]++;
	for(i=0;s[i]!='\0';i++)
		s_ctr[s[i]-97]++;
	int ans = 1;
	for(i=0;i<26;i++)
		if(r_ctr[i]!=s_ctr[i])
			ans = 0;
	return ans;
}
int wrong(char r[], char s[])
{
	int r_ctr[26] = {0}, s_ctr[26] = {0};
	int i;
	for(i=0;r[i]!='\0';i++)
		r_ctr[r[i]-97]=1;
	for(i=0;s[i]!='\0';i++)
		s_ctr[s[i]-97]=1;
	int ans = 1;
	for(i=0;i<26;i++)
		if(r_ctr[i]!=s_ctr[i])
			ans = 0;
	return ans;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int r_ctr[26] = {0}, s_ctr[26] = {0};
		cin>>r>>s;
		int i;
		int x = correct(r,s);
		int y = wrong(r,s);
		if(x==y)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
