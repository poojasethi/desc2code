#include<bits/stdc++.h>
using namespace std;
int main() {
	int t,n1,n2;
	string s1,s2;
	cin >> t;
	while ( t-- ) 
	{		
		int i;
		cin >> s1 >> s2;
		vector<bool> v1(26,false);
		vector<bool> v2(26,false);
		n1 = s1.size();
		n2 = s2.size();
		for ( i = 0; i < n1; ++i ) {
			v1[(s1[i]-'a')] = true;
		}
		for ( i = 0; i < n2; ++i ) {
			v2[(s2[i]-'a')] = true;
		}
		sort(s1.begin(), s1.end());
		sort(s2.begin(), s2.end());
		if( v1 == v2 ) 
		{
			if ( s1 == s2 )
				cout << "YES";
			else
				cout << "NO";
		} 
		else
			cout << "YES";
		cout << endl;
	}
	return 0;
}
