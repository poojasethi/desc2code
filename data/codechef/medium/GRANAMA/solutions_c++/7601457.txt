#include<iostream>
#include <list>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int sum,i,j=0,k=0,l=0,aa[26]={0},bb[26]={0};
		char a[1001],b[1001];
		scanf("%s %s",a,b);
		for(i=0;i<strlen(a);i++)
		{
			aa[a[i]-'a']++;
		}
		for(i=0;i<strlen(b);i++)
		{
			bb[b[i]-'a']++;
		}
		for(i=0;i<26;i++)
		{
			if((aa[i]==0 && bb[i]!=0) || (aa[i]!=0 && bb[i]==0))
			{
				k=1;
				break;
			}
			if(aa[i]==bb[i])
			{
				l++;
			}
			
		}
	//	printf("k=%d j=%d l=%d",k,j,l);
		if(k==1)
		printf("YES\n");
		else if(l==26 && k==0)
		printf("YES\n");
		else if(l!=26 && k==0)
		printf("NO\n");
	}
}
