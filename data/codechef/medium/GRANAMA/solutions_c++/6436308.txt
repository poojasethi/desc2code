#include <vector>
#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

using namespace std; 

int main(){
	int T;
	scanint(T);
	string str1, str2;
	int j = 0;
	int arr1[26];
	int arr2[26];
	bool result = true;
	
	for(int i = 0; i < T; i++){
		cin >> str1 >> str2;
		result = true;
		for(j = 0; j < 26; j++){
			arr1[j] = 0;
			arr2[j] = 0;
		}
		for(j = 0; j < str1.length(); j++){
			arr1[str1[j]-'a'] += 1;
		}
		for(j = 0; j < str2.length(); j++){
			arr2[str2[j]-'a'] += 1;
		}
		for(j = 0; j < 26; j++){
			if((arr1[j] == 0 && arr2[j] != 0) || (arr1[j] != 0 && arr2[j] == 0)){
				result = true;
				break;
			}
			else if(arr1[j] != 0 && arr2[j] != 0 && arr1[j] != arr2[j]){
				result = false;				
			}
		}
		if(result)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
}
