#include<iostream>
#include<cmath>
#include<climits>
#include<cstring>
using namespace std;
int main(){
	int t,i,flag;
	string s1,s2;
	cin>>t;
	while(t--){
		int h1[26]={0},h2[26]={0};
		cin>>s1>>s2;
		for(i=0;i<s1.length();i++)
			h1[s1[i]-'a']++;
		for(i=0;i<s2.length();i++)
			h2[s2[i]-'a']++;
		flag=0;
		for(i=0;i<26;i++){
			if((!h1[i]&&h2[i])||(h1[i]&&!h2[i])){
				flag=1;
				break;
			}
		}
		if(flag==0){
			for(i=0;i<26;i++){
				if(h1[i]!=h2[i]){
					flag=1;
					break;
				}
			}
			if(flag==1)
				cout<<"NO\n";
			else
				cout<<"YES\n";
		}
		else if(flag==1)
			cout<<"YES\n";
	}
}
