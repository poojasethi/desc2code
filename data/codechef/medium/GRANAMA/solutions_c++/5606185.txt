#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    string s,k;
	int a[27],b[27],found=1,i,n;
	scanf("%d",&n);
	while(n--)
	{
		found=1;
		cin>>s;
		cin>>k;
		for(i=0;i<27;i++)
		{
			a[i]=0;
		    b[i]=0;
		}
		for(i=0;i<s.length();i++)
			a[s[i]-'a']++;
		for(i=0;i<k.length();i++)
			b[k[i]-'a']++;
		for(i=0;i<27;i++)
		{
			if(a[i]!=b[i])
			{
				if(a[i]&&b[i])
				found=0;
				else
				{
					found=1;
					break;
				}
			}
		}	
		if(found)
		printf("YES \n");
		else
		printf("NO \n");
	}
	return 0;
}
