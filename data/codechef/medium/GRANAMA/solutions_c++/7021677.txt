#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,i;
    string a,b;
    cin>>t;
    while(t--)
    {
        int arr[26]={0},arr2[26]={0};
        cin>>a>>b;
        for(i=0;i<a.size();i++)
        {
            arr[a[i]-97]++;
        }
        for(i=0;i<b.size();i++)
        {
            arr2[b[i]-97]++;
        }
        int flag=0,flag2=0;
        for(i=0;i<26;i++)
        {
            if(arr[i]!=arr2[i])
            {
                flag=1;
            }
            if((arr[i]==0&&arr2[i]!=0)||(arr[i]!=0&&arr2[i]==0))
            {
                flag2=1;
            }
        }
        if(flag==1 &&flag2==0)
        {
            cout<<"NO"<<endl;
        }
        else
        {
            cout<<"YES"<<endl;
        }
    }
    return 0;
}
