#include<iostream>
#include<string>
#include<memory.h>
using namespace std;

int main() {
	int t;
	cin >> t;
	int N;
	int rArr[26];
	int sArr[26];
	string R;
	string S;
	while(t--) {
		cin >> R >> S;
		memset (rArr, 0, sizeof(rArr));
		memset (sArr, 0, sizeof(sArr));
		for(int i=0; i<R.size(); i++) {
			rArr[(int)R[i]-97]++; 
		}
		for(int i=0; i<S.size(); i++) {
			sArr[(int)S[i]-97]++; 
		}
		int chef = 1, actual = 1;
		for(int i=0;i <26;i++) {
			if(actual && rArr[i] != sArr[i])
				actual = 0;
			if(chef) {
				if((rArr[i] == 0 && sArr[i]>0) || (sArr[i] == 0 && rArr[i]>0))
					chef = 0;
			}
		}
		//cout << actual << " " << chef << endl;
		if(chef == actual)
			cout << "YES" << endl;
		else 
			cout << "NO" << endl;
	}
}
