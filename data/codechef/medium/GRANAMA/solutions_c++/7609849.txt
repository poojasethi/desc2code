#include<stdio.h>
#include<string.h>

int main()
{
    int tc,i,hash_r[26],hash_s[26],hash_rr[26],hash_ss[26],len1,len2,flag1,flag2;
    char r[1001],s[1001];

    scanf("%d\n",&tc);

    while(tc--)
    {
        scanf("%s%s",r,s);

        for(i=0;i<=25;++i)
            hash_r[i]=hash_s[i]=hash_rr[i]=hash_ss[i]  = 0;
        len1 = strlen(r);
        len2 = strlen(s);
        for(i=0;i<len1;i++)
        {
            hash_rr[r[i]-'a'] = 1;
            hash_r[r[i]-'a']++;
        }
        for(i=0;i<len2;++i)
        {
            hash_ss[s[i]-'a'] = 1;
            hash_s[s[i]-'a']++;
        }
        /*for(i=0;i<26;i++)
            printf("%d",hash_r[i]);
        printf("\n");
        for(i=0;i<26;i++)
            printf("%d",hash_s[i]);
        printf("\n");*/
        flag1=flag2 = 1;
        for(i=0;i<26;++i)
        {
            if(hash_r[i] != hash_s[i])
                flag1 = 0;
            if(hash_rr[i] != hash_ss[i])
                flag2  = 0;
        }
        if(flag1 == flag2)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
