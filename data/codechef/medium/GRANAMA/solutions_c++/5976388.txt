		vector<bool> v1(26,false);
		vector<bool> v2(26,false);
		int n1 = s1.size();
		int n2 = s2.size();
		for ( i = 0; i < n1; ++i ) {
			v1[(s1[i]-'a')] = true;
		}
		for ( i = 0; i < n2; ++i ) {
			v2[(s2[i]-'a')] = true;
		}

		sort(s2.begin(), s2.end());

			if ( s1 == s2 )
