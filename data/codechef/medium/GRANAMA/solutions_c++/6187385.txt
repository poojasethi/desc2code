#include<iostream>
using namespace std;
int correct(char a[],char b[])
{
	int c[26]={0};
	int d[26]={0};
	int i,flag=1;
	for(i=0;a[i]!='\0';++i)
	c[a[i]-'a']++;
	for(i=0;b[i]!='\0';++i)
	d[b[i]-'a']++;
	
	for(i=0;i<26;++i)
	{
		if(c[i]!=d[i])
		{
			flag=0; break;
		}
	}
	return flag;
}
int chef(char a[],char b[])
{
	int c[26]={0};
	int d[26]={0};
	int i,flag=1;
	for(i=0;a[i]!='\0';++i)
	c[a[i]-'a']=1;
	for(i=0;b[i]!='\0';++i)
	d[b[i]-'a']=1;	
	
	for(i=0;i<26;++i)
	{
		if(c[i]!=d[i])
		{
			flag=0; break;
		}
	}
	return flag;
}
int main()
{
	int test,i,flag,k1,k2;
	char r[1001],s[1001];
	cin>>test;
	while(test--)
	{
		cin>>r>>s;
		k1=correct(r,s);
		k2=chef(r,s);
		
		//cout<<k1<<" "<<k2<<" ";
		if(k1==k2 )
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	return 0;
}
