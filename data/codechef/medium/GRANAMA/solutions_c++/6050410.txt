#include<stdio.h>
#include<iostream>
#include<math.h>
int main()
{ int t;
  char s[10000],p[10000];
  scanf("%d",&t);
  while(t--)
       { int n1[200],n2[200],count1=0,count2=0;
         for(int i=0;i<=199;i++)
            { n1[i]=0;
              n2[i]=0;
            }
	     scanf("%s%s",&s,&p);
         for(int i=0;s[i]!='\0';i++)
            n1[s[i]-0]++;
         for(int i=0;p[i]!='\0';i++)   
            n2[p[i]-0]++;
         for(int i=0;i<=199;i++)
		    { if(n1[i]==0 && n2[i]==0)
		        continue;
			  if(n1[i]!=0 && n2[i]!=0 && n2[i]-n1[i]!=0)
			    count1=1;
			  if((n1[i]==0 && n2[i]!=0) || (n1[i]!=0 && n2[i]==0))
			    count2=1;  
		    }
		 if(count2==1) printf("YES\n");
		 else 
		     if(count1==1)
			   printf("NO\n");
			 else printf("YES\n");  	     
       }
}
