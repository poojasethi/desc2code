#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
using namespace std;
int main()
{
    int t,len1,len2,i,flag1,flag2;
    char str1[1001],str2[1001];
    scanf("%d",&t);
    while(t--)
    {
        flag1=0;
        flag2=1;
        scanf("%s",str1);
        scanf("%s",str2);
        len1=strlen(str1);
        len2=strlen(str2);
        vector<int> v1(26);
        vector<int> v2(26);
        vector<int> v3(26);
        vector<int> v4(26);
        for(i=0;i<len1;i++)
        {
            v1[str1[i]-97]=1;
            v2[str1[i]-97]++;
        }
        for(i=0;i<len2;i++)
        {
            v3[str2[i]-97]=1;
            v4[str2[i]-97]++;
        }
        for(i=0;i<26;i++)
        {
            if(v1[i]!=v3[i])
            {
                flag1=1;
                break;
            }
        }
        if(flag1)
        {
            printf("YES\n");
        }
        else
        {
            for(i=0;i<26;i++)
            {
                if(v2[i]!=v4[i])
                {
                    flag2=0;
                    break;
                }
            }
            if(flag2)
                printf("YES\n");
            else
                printf("NO\n");
        }
    }
    return 0;
}
