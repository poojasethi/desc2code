#include <cstdio>
#include <cctype>

using namespace std;

const int BUFSIZE = 4096;

char g_buffer[BUFSIZE];
int g_bufpos = BUFSIZE;

int readint()
{
    int n;
    for (; g_bufpos < BUFSIZE && !isdigit(g_buffer[g_bufpos]); g_bufpos++);
    if (g_bufpos == BUFSIZE)
    {
        g_bufpos = 0;
        fread(g_buffer, 1, BUFSIZE, stdin);
        for (; g_bufpos < BUFSIZE && !isdigit(g_buffer[g_bufpos]); g_bufpos++);
    }
    n = 0;
    for (; g_bufpos < BUFSIZE && isdigit(g_buffer[g_bufpos]); g_bufpos++)
        n = 10 * n + g_buffer[g_bufpos] - '0';
    if (g_bufpos == BUFSIZE)
    {
        g_bufpos = 0;
        fread(g_buffer, 1, BUFSIZE, stdin);
        for (; g_bufpos < BUFSIZE && isdigit(g_buffer[g_bufpos]); g_bufpos++)
            n = 10 * n + g_buffer[g_bufpos] - '0';
    }
    return n;
}

int main()
{
    int t; // test cases
    int tp; // total page cnt
    int F; // number of pages
    int tl; // tored leaves
    double pagesum;
    t = readint();
    for (; t > 0; t--)
    {
        tp = readint();
        pagesum = tp * (tp + 1) / 2;
        if (tp % 2)
            tp += 1;
        F = readint();
        for (int i=0; i<F; i++)
            pagesum -= readint();
        tl = readint();
        printf("%.4f\n", (tp == 0) ? 0.0 : pagesum - pagesum * 2 * tl / tp);
    }
	return 0;
}
