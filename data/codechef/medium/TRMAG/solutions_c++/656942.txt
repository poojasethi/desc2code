#include<stdio.h>
main() {
   int t,n,x,tore;
   int sum;
   float ans;
   scanf("%d",&t);
   while(t--) {
		scanf("%d",&n);
		sum = n*(n+1)/2;
		scanf("%d",&tore);
		while(tore--) {
			scanf("%d",&x);
			sum -= x;
        }

		scanf("%d",&tore);
				
		ans = (float) sum;
		int v;
		if(n%2==0)
		v = n/2;
		else
		v = n/2 + 1;

		ans = ans - (ans*tore)/v;

		printf("%.4f\n",ans);
   }
   return 0;
}
