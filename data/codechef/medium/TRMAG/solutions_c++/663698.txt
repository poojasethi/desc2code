#include <iostream>

using namespace std;

int main ()
{
    int T;
    cin >> T;

    while (T--)
    {
        int N;
        cin >> N;

        double Sum = N * (N + 1) / 2;

        int M;
        cin >> M;

        for (int i = 0; i < M; i++)
        {
            int P;
            cin >> P;
            Sum -= P;
        }

        int T;
        cin >> T;

        Sum *= (1.0 - (double)T / ((N + 1) / 2));

        cout.setf(ios::fixed, ios::floatfield);
        cout.precision(4);
        cout << Sum << endl;
    }

    return 0;
}
