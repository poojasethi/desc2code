#include <iostream>
#include <cstdio>
using namespace std;

int main() {
  int t, n, b, p;
  cin >> t;
  while(t--) {
    cin >> n;
    double ans = (n*(n+1.0))/2.0;
    int tot = (n+1)/2;
    cin >> b;
    for(int i=0;i<b;++i) {
      cin >> p;
      ans -= 1.0*p;
    }
    cin >> p;
    printf("%.4f\n",ans*(tot-p)/tot);
  }
  return 0;
}
