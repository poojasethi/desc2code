#include <cstdio>

using namespace std;

int main()
{
    int t; // test cases
    int tp; // total page cnt
    int F; // number of pages
    int p;
    int tl; // tored leaves
    double pagesum;
    scanf("%d", &t);
    for (; t > 0; t--)
    {
        scanf("%d", &tp);
        pagesum = tp * (tp + 1) / 2;
        if (tp % 2)
            tp += 1;
        scanf("%d", &F);
        for (int i=0; i<F; i++)
        {
            scanf("%d", &p);
            pagesum -= p;
        }
        scanf("%d", &tl);
        printf("%.4f\n", (tp == 0) ? 0.0 : pagesum - pagesum * 2 * tl / tp);
    }
	return 0;
}
