#include <stdio.h>

int bin[2048][2048];

int main()
{
	for(int i = 0; i < 2048; i++)
		for(int j = 0; j <= i; j++)
			if(j == 0 || j == i) bin[i][j] = 1;
			else bin[i][j] = bin[i-1][j] + bin[i-1][j-1];

	int T;
	scanf("%d", &T);
	while(T--) {
		int n, m, t;
		double sol;
		scanf("%d%d", &n, &m);
		
		sol = n*(n+1)/2;
		for(int i = 0; i < m; i++) {
			int p;
			scanf("%d", &p);
			sol -= p;
		}
		scanf("%d", &t);
		
		printf("%.4lf\n", sol*(1-(double)t/((n+1)/2)));
	}
	return 0;
}
