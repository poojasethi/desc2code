#include <iostream>
#include <cstdlib>
#include <algorithm>
#include <vector>
#include <stdio.h>

using namespace std;

int main()
{
int t,n,f,tor,sum,temp,sum2;
float avg,ans,mux;
cin>>t;
while(t--)
{
sum=0; sum2=0 ;ans=0; mux=0;
cin>>n>>f;
for(int i=0;i<f;i++){ cin>>temp; sum +=temp;}
cin>>tor;
avg = (float)((n*(n+1))/2 - sum);
if(n%2==0)mux =(float) (n-2*tor)/n;
else mux = (float)(n+1-2*tor)/(n+1);
ans = (float)avg*mux;
//cout<<ans<<endl;
printf("%.4f\n",ans);
}
return 0;
}
