#include<stdio.h>
main() {
   int t,n,te,x,tore;
   long long sum;
   double ans;
   scanf("%d",&t);
   while(t--) {
		scanf("%d",&n);
		sum = (long long) n*(n+1)/2;
		scanf("%d",&te);
		while(te--) {
			scanf("%d",&x);
			sum -= x;
        }

		scanf("%d",&tore);
		
		
		ans = (double) sum;
		int v;
		if(n%2==0)
		v = n/2;
		else
		v = n/2 + 1;

		ans = ans - (ans*tore)/v;

		printf("%.4lf\n",ans);
   }
   return 0;
}
