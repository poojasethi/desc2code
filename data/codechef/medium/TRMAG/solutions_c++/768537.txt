#define DEBUG 0

#include <stdio.h>
#include <string.h>

int main(void) {
  int tc, n, f, i, x, l, ans;

  scanf("%d", &tc);
  while(tc--) {
    scanf("%d%d", &n, &f);
    ans=n*(n+1)/2;
    for(i=0;i<f;i++) {
      scanf("%d", &x);
      ans-=x;
    }
    scanf("%d", &l);
    n=n/2+n%2;
    printf("%.4lf\n", (n-l)*(double)ans/n);
  }
  return 0;
}
