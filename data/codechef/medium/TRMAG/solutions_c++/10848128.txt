#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int t;
    cin>>t;
    while(t--)
    {
        int n,pa,b,x,p;
        cin>>n;
        double res = (double)(n*(n+1))/2;
        pa = (n+1)/2;
        cin>>b;
        for(int i=0;i<b;i++)
        {
            cin>>x;
            res -= x;
        }
        cin>>p;
        cout<<fixed<<setprecision(4)<<res*(pa-p)/pa<<endl;
    }
    return 0;
}
