#include<iostream>
#include<stdio.h>
#include<limits.h>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cmath>
#include<vector>
#include<map>
#define MOD 1000000007
using namespace std;
int main(){
	int t,n,a[3000],avg,sum,cnt,i,min,max;
	cin>>t;
	while(t--){
		cin>>n;
		sum=0;cnt=0;
		for(i=0;i<n;i++){
			cin>>a[i];
			sum+=a[i];
		}
		if(sum%n)cout<<"-1";
		else{
			while(1){
				min=0,max=0;
				for(int i=1;i<n;i++){
					if(a[i]<a[min])min=i;
					else if(a[i]>a[max]) max=i;
				}
				if(a[max]==a[min]){
					cout<<cnt;
					break;
				}
				else{
					avg=(a[max]-a[min])/2;
					a[max]-=avg;
					a[min]+=avg;
				}
				cnt++;
			}
		}
		cout<<endl;
	}
	return 0;
}
