#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
 int n,T,r,i,j,tmp,max,min,sum,x,y,*a;
scanf("%d",&T);
while(T--)
{
//printf("r");
sum=0;
scanf("%d",&n);
a=( int*)malloc(n*sizeof( int));
for(i=0;i<n;i++)
{
scanf("%d",&a[i]);
sum=sum+a[i];
}
if(sum%n!=0)
printf("-1\n");
else{
         r = 0;

        while(1)
        {
                int mn = n+1, mx = -1, mni = -1, mxi = -1;
                for(int i=0;i<n;i++)
                {
                        if(a[i]<mn) mn = a[i], mni = i;
                        if(a[i]>mx) mx = a[i], mxi = i;
                }
                if(mn==mx) break;
                ++r;
                int d = (mx-mn+1)/2;
                a[mni]+=d; a[mxi]-=d;
        }
        printf("%lld\n",r);
}
}
return 0;

}
