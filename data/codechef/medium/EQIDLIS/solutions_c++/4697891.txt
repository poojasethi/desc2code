#include <cstdio>
#include <iostream>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;

#define SI ({int x;scanf("%d",&x);x;})
#define MX 1000010

int n, a[MX];

int solve()
{
	int sum = 0;
	for(int i=0;i<n;i++) sum+=a[i];
	if(sum%n!=0) return -1;

	int r = 0;

	while(1)
	{
		int mn = n+1, mx = -1, mni = -1, mxi = -1;
		for(int i=0;i<n;i++)
		{
			if(a[i]<mn) mn = a[i], mni = i;
			if(a[i]>mx) mx = a[i], mxi = i;
		}
		if(mn==mx) break;
		++r;
		int d = (mx-mn+1)/2;
		a[mni]+=d; a[mxi]-=d;
	}
	return r;
}


int main()
{
	int kases=SI, last;
	
	for(int kase=1;kase<=kases;kase++)
	{
		n = SI;
		for(int i=0;i<n;i++) a[i] = SI;
		last = a[n-1];
		printf("%d\n",solve());
	}

	return 0;
}
