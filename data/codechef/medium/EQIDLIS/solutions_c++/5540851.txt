#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

int main()
{
	int mx,mn;
	int last,first;
	int a[3001];
	int n,r;
	int t,count,flag;
	
	scanf("%d",&t);
	for (int i = 0; i < t; i++) {
		scanf("%d",&n);
		for (int i = 0; i < n; i++) {
			scanf("%d",&a[i]);
		}
		count = 0;
		flag = 0;
		mn = a[0];
		first = 0;
		mx = a[0];
		last = 0;
		for (int i = 0; i < n; i++) {
			if (a[i] < mn) {
				mn = a[i];
				first = i;
			}
			if (a[i] > mx) {
				mx = a[i];
				last = i;
			}
		}
		while (a[last] - a[first] > 1) {
			r = (a[last] - a[first])/2;
			a[last] = a[last] - r;
			a[first] = a[first] + r;
			count++;
			mn = a[0];
			first = 0;
			mx = a[0];
			last = 0;
			for (int i = 0; i < n; i++) {
				if (a[i] < mn) {
					mn = a[i];
					first = i;
				}
				if (a[i] > mx) {
					mx = a[i];
					last = i;
				}
			}
		}
		for (int i = 0; i < n-1; i++) {
			if (a[i] != a[i+1]) {
				flag = -1;
				printf("-1\n");
				break;
			}
		}
		if (flag == 0)
			printf("%d\n",count);
	}
}
