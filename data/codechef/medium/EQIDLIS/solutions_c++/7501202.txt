#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,m,n,t,sum,ans,count,max_i,min_i,Max,Min,a;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int arr[n];
		sum=0;

		for(i=0;i<n;i++)
		{
			scanf("%d",&arr[i]);
			sum+=arr[i];
		}
		count=0;
		ans=0;
		if(sum%n!=0)
			printf("-1\n");

		else
		{
			while(1)
			{
				Max=0;
				Min=4000;
				for(i=0;i<n;i++)
				{
					if(arr[i]>Max)
					{
						Max=arr[i];
						max_i=i;
					}
					if(arr[i]<Min)
					{
						Min=arr[i];
						min_i=i;
					}
				}
				if(Max==Min)
					break;

				a=(Max-Min)/2;
				arr[max_i]-=a;
				arr[min_i]+=a;
				count++;
			}
			printf("%d\n",count);
		}
	}
}
