#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		int arr[n], sum = 0, count = 0, avg, tempind = 0, moves = 0;
		for(int i = 0; i < n; i++)
		{
			scanf("%d", &arr[i]);
			sum += arr[i];
		}
		if(sum % n)
		{
			printf("-1\n");
			continue;
		}
		sort(arr, arr+n);
		avg = sum / n;
		while(arr[tempind] != avg)
		{
			tempind++;
		}
		while(arr[tempind] == avg)
		{
			count++;
			tempind++;
		}
		while(count < n)
		{
			moves++;
			int dif = (arr[n-1] - arr[0]) / 2;
			arr[n-1] -= dif;
			arr[0] += dif;
			if(arr[0] == avg)
			{
				count++;
			}
			if(arr[n-1] == avg)
			{
				count++;
			}
			int ind1 = 0, ind2 = n-1;
			while(arr[ind1] > arr[ind1+1])
			{
				int temp = arr[ind1];
				arr[ind1] = arr[ind1+1];
				arr[ind1+1] = temp;
				ind1++;
			}
			while(arr[ind2] < arr[ind2-1])
			{
				int temp = arr[ind2];
				arr[ind2] = arr[ind2-1];
				arr[ind2-1] = temp;
				ind2--;
			}
		}
		printf("%d\n", moves);
	}
    return 0;
}
