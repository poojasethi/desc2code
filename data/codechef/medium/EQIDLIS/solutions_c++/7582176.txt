#include<bits/stdc++.h>
using namespace std;
int a[3005];
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int sum=0;
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			sum+=a[i];
		}
		if(sum%n!=0)
		printf("-1\n");
		else
	{
		int ans=0;
		int min=0,max=1;
		while(min<max)
		{
			int minp=0,maxp=0;
			max=INT_MIN;
			min=INT_MAX;
		   for(int i=0;i<n;i++)
		   {
		   	if(a[i]>max)
		    {
		    	max=a[i];
		    	maxp=i;
		    }
		    if(a[i]<min)
		    {
		    	min=a[i];
		    	minp=i;
		    }
		   }
			int change;
			int x=a[minp],y=a[maxp];
			if((y-x)%2==0)
			 change=(y-x)/2;
		    else
		    change=(y-x)/2+1;
			a[minp]+=change;
			a[maxp]-=change;
			ans++;
		}
		printf("%d\n",ans-1);
	}
	}
}
