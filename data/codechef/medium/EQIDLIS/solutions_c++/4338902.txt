#include <iostream>
#include <string.h>
#include <algorithm>
#include <vector>

using namespace std;


int main(){
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        int t;scanf("%d",&t);
        vector<int> vec;
        int sum=0;
        for(int j=0;j<t;j++){
            int num;
            scanf("%d",&num);
            vec.push_back(num);
            sum+=num;
        }
        if(sum%t!=0){
            cout<<"-1"<<endl;
            continue;
        }
        int cnt=0;
        vector<int>::iterator minit=min_element(vec.begin(),vec.end());
        vector<int>::iterator maxit=max_element(vec.begin(),vec.end());
        while((*minit)!=(*maxit)){
                int temp=((*maxit)-(*minit)+1)/2;
                (*minit)=(*minit)+temp;
                (*maxit)=(*maxit)-temp;
                cnt++;
                minit=min_element(vec.begin(),vec.end());
                maxit=max_element(vec.begin(),vec.end());
        }
        printf("%d\n",cnt);
    }
}
