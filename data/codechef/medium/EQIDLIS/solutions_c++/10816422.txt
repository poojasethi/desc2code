int main()
{
    int t;si(t);
    while(t--){
        int n;si(n);
        int i,a[n+1],j,x,sum=0;
        memset(a,0,sizeof(a));
        iter(i,0,n){
            si(x);
            sum+=x;
            a[x]+=1;
        }
        int ans=0,r,k=0;
        if(sum%n!=0){
            cout<<"-1"<<endl;

        }
        for(i=0,j=n;i<j;){
            while(i<j && a[i]==0)i++;
            while(j>i && a[j]==0)j--;
            if(i>=j)break;
            r=ceil((j-i)/2);
            a[j]--;a[i]--;
            a[j-r]++;a[i+r]++;
            ans++;
        }
        cout<<ans<<endl;
    }
