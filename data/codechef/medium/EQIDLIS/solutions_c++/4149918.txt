#include<stdio.h>
int main()
{
          int t;
          scanf("%d",&t);
          while(t--)
          {
                    int s,a[3008],n,i,min,max,x,y,count=0,sum=0;
                    scanf("%d",&n);
                    for(i=0;i<n;i++){
                    scanf("%d",&a[i]);
                    sum=sum+a[i];}
                    if(sum%n!=0)
                    printf("-1\n");
                    else{
              do
              {
                    min=a[0];max=a[0];x=0;y=0;
                    for(i=1;i<n;i++)
                    {
                                    if(min>a[i]){
                                    min=a[i];
                                    x=i;}
                                  if(max<a[i]){
                                  max=a[i];
                                  y=i;}  
                    }
                    s=(a[y]-a[x]+1)/2;
                    a[x]+=s;
                    a[y]-=s;
                    count++;
              }while(min!=max);
             printf("%d\n",count-1); }      
          }       
          return 0;
}
