#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cmath>
#include <queue>
#include <set>

using namespace std;

int main()
{
	int t,n;scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int a[n];
		int sum=0,count=0,min=0,max=0;
		for(int i=0;i<n;i++){scanf("%d",&a[i]);sum+=a[i];
			if(a[min]>a[i])min=i;
			if(a[max]<a[i])max=i;}
		if(sum%n!=0){cout<<-1<<endl;continue;}
		while(a[max]!=a[min])
		{
			count++;
			int q = ceil((a[max]-a[min])/2);
			a[min]+=q;
			a[max]-=q;
			for(int i=0;i<n;i++)
			{
				if(a[min]>a[i])min=i;
			if(a[max]<a[i])max=i;}
		}
				
		
		cout<<count<<endl;
	}
}
