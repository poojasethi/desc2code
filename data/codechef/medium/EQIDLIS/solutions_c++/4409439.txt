#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define mod 1000000007
using namespace std;
main()
{   std:ios_base::sync_with_stdio(false);
    int t,n,i,tmp;
    long int tot=0,each,ans=0,max,min,indmx,indmn;
    cin>>t;
    while(t--)
    {
    	max=-1;
		min=3002;
    	ans=0;
    	tot=0;
    	cin>>n;
    	int a[n];
    	for(i=0;i<n;i++)
    	{
    	  cin>>a[i];
		  tot+=a[i];	
    	}
    	if((tot%n))
    	printf("-1\n");
    	else
    	{ each=tot/n;
    	  while(1)
    	  {
    	  	max=-1;
    	    min=3002;
    	  	for(i=0;i<n;i++)
    	  	{
    	  		if(a[i]>max)
				{	 max=a[i]; indmx=i;  } 
    	  		if(a[i]<min) 
				{    min=a[i]; indmn=i;  }
    	  	}
    	  	if(max==min)
    	  	break;
    	  	tmp=(max-min+1)/2;
    	  	a[indmx]-=tmp;
    	  	a[indmn]+=tmp;
    	  	ans++;
    	  }
    	  printf("%ld\n",ans);
    	}
    }
	return 0;
}
