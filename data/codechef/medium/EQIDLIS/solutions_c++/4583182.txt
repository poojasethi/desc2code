#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main()
{
    int t,i,n;
    long sum = 0,turns = 0;
    scanf("%d",&t);
    int a[3000],transfer,flag = 0,max1 = -1,min1 = 90000,indmax = 0,indmin = 0;

while(t--){
    sum = 0,turns = 0,flag = 0,max1 = -1,min1 = 90000,indmax = 0,indmin = 0;

    scanf("%d",&n);

    for(i =  0 ; i < n ; i++){
        scanf("%d",&a[i]);
        sum += a[i];
    }

    if(sum % n != 0)
        printf("-1\n");
    else{
      while(flag == 0){
        for(i = 0 ; i < n ; i++){
                if(a[i] > max1){
                    max1 = a[i];
                    indmax = i;
                }
                if(a[i] < min1){
                    min1 = a[i];
                    indmin = i;
                }
        }
        if(max1 - min1 == 1){
            printf("-1\n");
            flag = 1;
            break;
        }
        else if(max1 == min1){
            printf("%ld\n",turns);
            flag = 1;
            break;
        }
        else{
            if((max1 - min1) % 2 == 0){
                transfer = (max1 - min1)/2;
            }
            else
                transfer = (max1 - min1)/2 + 1;

            a[indmax] -=  transfer;
            a[indmin] += transfer;
            turns++;
            max1 = 0,min1 = 90000;
        }

    }

 }


}

	return 0;
}
