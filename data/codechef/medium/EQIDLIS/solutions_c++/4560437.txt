#include <cstdio>
using namespace std;

int main()
{
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		int temp, sum = 0, out = 0, min = 3001, max = -1, arr[3001] = {0};
		for(int i = 0; i < n; i++)
		{
			scanf("%d", &temp);
			arr[temp]++;
			sum += temp;
			if(temp < min)
			{
				min = temp;
			}
			if(temp > max)
			{
				max = temp;
			}
		}
		if(sum % n)
		{
			printf("-1\n");
			continue;
		}
		sum /= n;
		while(arr[sum] < n)
		{
			out++;
			temp = (max - min)/2;
			arr[min + temp]++;
			arr[max - temp]++;
			arr[min]--;
			arr[max]--;
			while(!arr[min])
			{
				min++;
			}
			while(!arr[max])
			{
				max--;
			}
		}
		printf("%d\n", out);
	}
    return 0;
}
