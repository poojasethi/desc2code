#include <cstdio>

int main(void)
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i;
		scanf("%d",&n);
		int arr[n],max,min,diff=-1,count=0;
		max = 0;
		min = 0;
		long long int sum = 0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&arr[i]);
			if(arr[i]<arr[min])
			   min = i;
			else if(arr[i]>arr[max])
			   max = i;  
			sum = sum + arr[i];
		}
		if(sum%n==0)
		{
			while(diff!=0)
			{
				diff = (arr[max]-arr[min]+1)/2;
				arr[max] = arr[max] - diff;
				arr[min] = arr[min] + diff;
				max = 0;
				min=  0;
				for(i=0;i<n;i++)
				{
					if(arr[i]<arr[min])
			           min = i;
		        	else if(arr[i]>arr[max])
			           max = i;  
					
				}
				count++;
				
			}
			printf("%d\n",count-1);
			
		}
		else
		  printf("-1\n");
		
	}
	
	
	return 0;
}
