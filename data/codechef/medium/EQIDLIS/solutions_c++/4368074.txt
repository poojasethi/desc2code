#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int rd() {
	char c = getchar_unlocked();
	while(c<'0' || c>'9') c = getchar_unlocked();
	int ret = 0;
	while(c>='0' && c<='9') {
		ret = 10 * ret + c - 48;
		c = getchar_unlocked();
	}
	return ret;
}

int main(){
	int t,n;
	t=rd();
	while(t--){
		n=rd();
		int a[n];
		int sum=0;
		for(int i=0; i<n; i++){
			a[i]=rd();
			sum+=a[i];
		}

		if(sum%n!=0){
			cout<<"-1"<<endl;
			continue;
		}

		int Mi,mi;
		int steps=0;
		while(1){
			Mi=0; mi=0;
			for(int i=0; i<n; i++){
				if(a[i]<a[mi])
					mi=i;
				if(a[i]>a[Mi])
					Mi=i;
			}
			if(a[mi]==a[Mi])
				break;
			steps++;
			int d = (a[Mi]-a[mi]+1)/2;
			a[Mi]-=d;
			a[mi]+=d;
		}
		cout<<steps<<endl;
	}
	return 0;
}
