#include<stdio.h>
#include<limits.h>
int findmax(int *a,int n)
{
    int i,max=0;
    for(i=0;i<n;i++)
    {
    if(a[max]<a[i])
    {
    max=i;
    }               
    }
     return max;
    }
    int findmin(int *a,int n)
{
    int i,min=0;
    for(i=0;i<n;i++)
    {
    if(a[min]>a[i])
    {
    min=i;
    }               
    }
     return min;
    }
