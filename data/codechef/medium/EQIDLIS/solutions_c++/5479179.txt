#include<iostream>
#include<malloc.h>
#include<stdio.h>
using namespace std;
int main()
{
    int t,n,*a,i,j,s,c,z,r,maxm,minm;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
        scanf("%d",&n);
        a=(int*)malloc(n*sizeof(int));
        for(j=0;j<n;j++)
            scanf("%d",&a[j]);
        s=0;
        for(j=0;j<n;j++)
            s=s+a[j];
        z=(s%n);
        if(z!=0)
            printf("-1\n");
        else
        {
            c=0;
            while(1)
            {
                maxm=0;
                minm=0;
                for(j=0;j<n;j++)
                {
                    if(a[j]<a[minm]) minm=j;
                    if(a[j]>a[maxm]) maxm=j;
                }
                if(maxm==minm) break;
                else
                {
                    c++;
                    r=(a[maxm]-a[minm]+1)/2;
                    a[maxm]-=r;
                    a[minm]+=r;
                }
            }
            printf("%d\n",c);
        }
    }
    free(a);
    return 0;
}
