#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_v) (int)(_v).size()
#define all(_v) (_v).begin(),(_v).end()
#define pii pair<int,int>
typedef long long ll;
const int INF = 1000000000;
const ll MOD = 1e9 + 7;

const int maxn = 100010;
map < string,string > cntry;
map < string,int > cntchef;
map < string,int > cntcntry;

int main() {
	ios_base::sync_with_stdio(false);

	int n,m; cin >> n >> m;
	for (int i = 0; i < n; i ++) {
		string s,ss; cin >> s >> ss;
		cntry[s] = ss;
	}

	while (m --) {
		string s; cin >> s;
		cntchef[s] ++;
		cntcntry[cntry[s]] ++;
	}

	int mx = -1;
	string ans;
	for (map < string,int > :: iterator it = cntcntry.begin(); it != cntcntry.end(); it ++) {
		string s = it -> first;
		int num = it -> second;
		if(num == mx) {
			ans = min(ans,s);
		} 
		if(num > mx) {
			ans = s;
			mx = num;
		}
	}
	
	cout << ans << "\n";

	ans = "";
	mx = -1;
	for (map < string,int > :: iterator it = cntchef.begin(); it != cntchef.end(); it ++) {
		string s = it -> first;
		int num = it -> second;
		if(num == mx) {
			ans = min(ans,s);
		}
		if(num > mx) {
			ans = s;
			mx = num;
		}
	}

	cout << ans << "\n";
	return 0;
}
