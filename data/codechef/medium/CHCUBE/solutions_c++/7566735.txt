#include <iostream>
#include <fstream>
#include <string>

using namespace std;

int T, i;

int main()
{
	//ifstream cin ("main.in");

	int adj[8][3] = { {0, 3, 4}, {0, 4, 2}, {0, 2, 5}, {0, 5, 3}, {1, 3, 4}, {1, 4, 2}, {1, 2, 5}, {1, 5, 3} };
	string str[6];

	cin >> T;
	while (T--)
	{
		i = 0;
		while (i < 6)
		{
			cin >> str[i];

			i++;
		}

		i = 0;
		while (i < 8)
		{
			if (str[adj[i][0]] == str[adj[i][1]] && str[adj[i][1]] == str[adj[i][2]])
				break;

			i++;
		}

		if (i != 8) 
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}

	return 0;
}
