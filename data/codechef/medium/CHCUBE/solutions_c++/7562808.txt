#include <iostream>
#include <sstream>
using namespace std;

string arr[10];

int main(){
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	stringstream ss;
	while(t--){
		for(int i=0;i<6;i++) cin>>arr[i];
		bool found = false;
		for(int i=0;i<2;i++)
			for(int j=0;j<2;j++)
				for(int k=0;k<2;k++)
					if (arr[i]==arr[j+2] && arr[j+2]==arr[k+4])
						found = true;
		if (found) ss<<"YES\n";
		else ss<<"NO\n";
	}
	cout<<ss.str()<<endl;
}
