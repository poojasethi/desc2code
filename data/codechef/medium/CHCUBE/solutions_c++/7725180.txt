#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;
int main()
{

   int t;
   string s;
   cin>>t;
   getchar();
   while(t--){
     getline(cin,s);
     
     string str[10];
    
     int i=0;
     int j=0;
     while(s[i])
     {
           string ss;
          while(s[i]!=' ' && s[i]!='\0')
          {
               ss=ss+s[i];
               i++;
          }
          str[j++]=ss;
          if(s[i]==' ')
          i++;
     }
     int flag=0;
     int k;
     for(i=0;i<6;i++)
     {
         if(i%2==0)
         j=2+i;
         else
         j=1+i;
          for(;j<6;j++)
          {
              if(str[i]==str[j])
              {
                  if(j%2==0)
                  k=j+2;
                  else
                  k=j+1;
              
                    for( ;k<6;k++)
                    {
                        if(str[k]==str[j])
                        flag=1;
                    
                    }
              }
          
          }
     
     }
     if(flag==1)
     cout<<"YES\n";
     else
     cout<<"NO\n";
    }
   return 0;
}
