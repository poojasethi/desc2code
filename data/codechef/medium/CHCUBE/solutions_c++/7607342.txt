#include<iostream>
#include<string>
using namespace std;
int main()
{
	int adj[8][3] = {
			{0, 2, 4},
			{0, 3, 4},
			{0, 2, 5},
			{0, 3, 5},
			{1, 2, 4},
			{1, 3, 4},
			{1, 2, 5},
			{1, 3, 5},
		};
	int t;
	cin>>t;
	while(t--)
	{
		string ans = "NO";
		string s[6];
		for(int i = 0; i < 6 ; i++)
			cin>>s[i];
		for(int i = 0; i < 8; i++)
		{
			if(s[adj[i][0]].compare(s[adj[i][1]]) == 0 && s[adj[i][0]].compare(s[adj[i][2]]) == 0) 
				ans = "YES";
		}
		cout<<ans<<endl;
	}
	return 0;
}
