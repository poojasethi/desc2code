#include<iostream>
using namespace std;
int cmp[8][3] ={
    {0,2,4},
    {0,2,5},
    {0,3,4},
    {0,3,5},
    {1,2,4},
    {1,2,5},
    {1,3,4},
    {1,3,5}
};

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
        string face[6];
        for(int i=0;i<6;i++)
        {
            cin >> face[i];
        }
        int i;
        for( i=0;i<8;i++)
        {
            int a,b,c;
            a=cmp[i][0];
            b=cmp[i][1];
            c=cmp[i][2];
            
            if(face[a]==face[b] && face[b]==face[c])
            {
                cout << "YES" << endl;
                break;
            }
            
        }
        if(i==8)
        cout << "NO" << endl;
        
    }
    
    return 0;
}
