#include<iostream>
#include<cstring>
#include<string>
#include<stdio.h>

using namespace std;

int main()
{
    int tc;
    char f[7], ba[7], l[7], r[7], t[7], bo[7];
    cin>>tc;
    while(tc--)
    {
        scanf("%s%s%s%s%s%s",f,ba,l,r,t,bo);
        if((!strcmp(f,l)||!strcmp(f,r))&&(!strcmp(f,t)||!strcmp(f,bo)))
        {
            cout<<"YES\n";
            continue;
        }
        if((!strcmp(ba,l)||!strcmp(ba,r))&&(!strcmp(ba,t)||!strcmp(ba,bo)))
        {
            cout<<"YES\n";
            continue;
        }
        cout<<"NO\n";
    }

}
