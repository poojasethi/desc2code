#include<iostream>
#include<string.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;

	while(t--)
	{	bool flag=0;
		char ar[6][10];
		for(int i=0;i<6;i++)
		{
			cin>>ar[i];
			
		}
			
		if(!strcmp(ar[2],ar[0])||!strcmp(ar[3],ar[0]))
		{
		
			if(!strcmp(ar[4],ar[0])||!strcmp(ar[5],ar[0]))
			flag=1;
			
		}
		if(!strcmp(ar[2],ar[1])||!strcmp(ar[3],ar[1]))
		{
		
			if(!strcmp(ar[4],ar[1])||!strcmp(ar[5],ar[1]))
			flag=1;
			
		}
		if(flag)
		cout<<"YES\n";
		else
		cout<<"NO\n";
		
		
	}
	return 0;
}
