#include<iostream>
#include<cstdio>
#include<map>

using namespace std;

map<string, int> mapColor;

void MapColor()
{
    string colors[] = {"black", "blue", "red", "green", "yellow", "orange"};
    for(int i = 0; i < 6; i++)
    {
        mapColor[colors[i]] = i;
        //cout << colors[i] << " " << mapColor[colors[i]] << endl;
    }
}

int color[7];

void clearAll()
{
    for(int i = 0; i < 7; i++)
    {
        color[i] = 0;
    }
}

string solve()
{
    int tc = color[1];
    for(int i = 2; i <= 5; i++)
    {
        if(i == 5)
        {
            if(color[i] == color[2] && color[i] == tc)return "YES\n";
        }
        else if(color[i] == color[i+1] && color[i] == tc)return "YES\n";
    }
    int bc = color[6];
    for(int i = 2; i <= 5; i++)
    {
        if(i == 5)
        {
            if(color[i] == color[2] && color[i] == bc)return "YES\n";
        }
        else if(color[i] == color[i+1] && color[i] == bc)return "YES\n";
    }
    return "NO\n";
}

int main()
{
    MapColor();
    int t;
    scanf("%d", &t);
    while(t--)
    {
        clearAll();
        for(int i = 1; i <= 6; i++)
        {
            string inp;
            cin >> inp;
            int ixColor = mapColor[inp];

            if(i == 1)color[1] = ixColor;
            if(i == 2)color[6] = ixColor;
            if(i == 3)color[2] = ixColor;
            if(i == 4)color[4] = ixColor;
            if(i == 5)color[3] = ixColor;
            if(i == 6)color[5] = ixColor;

        }
        cout << solve() << endl;
    }
}
