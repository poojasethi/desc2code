#include <iostream>
#include <cstring>
using namespace std;

int main(){
    int T,side[6],flag,count[6];
    char color[7];
    scanf("%d",&T);
    for(int i=0;i<T;i++){
        flag=0;
        for(int k=0;k<6;k++)
        count[k]=0;
        for(int j=0;j<6;j++){
            scanf("%s",&color);
            if(strcmp("black",color)==0)
            side[j]=0;     
            if(strcmp("blue",color)==0)
            side[j]=1;
            if(strcmp("red",color)==0)
            side[j]=2;
            if(strcmp("green",color)==0)
            side[j]=3;
            if(strcmp("yellow",color)==0)
            side[j]=4;
            if(strcmp("orange",color)==0)
            side[j]=5;
            count[side[j]]++;
        }
        for(int j=0;j<6;j++){
            if(count[j]>2 && count[j]<5){
                if(side[0]==j || side[1]==j)
                    if(side[2]==j || side[3]==j)
                        if(side[4]==j || side[5]==j){
                                      flag=1;
                                      break;
                        }
            }
            if(count[j]>4){
                flag=1;
                break;
            }
        }
        if(flag==1)
        printf("%s\n","YES");
        else
        printf("%s\n","NO");
    }
    return 0;
}
