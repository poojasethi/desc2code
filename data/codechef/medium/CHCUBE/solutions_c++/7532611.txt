#include<bits/stdc++.h>
using namespace std;
 
map<string, int> mp;
string str[6];
 
int main() {
    int test;
    cin>>test;
    while(test--) {
        for(int i=0;i<6;i++) {
            cin>>str[i];
            mp[str[i]]++;
        }
        string save;
        int cnt=0;
        bool ok=false;
        map<string, int> :: iterator it;
        for(it=mp.begin();it!=mp.end();it++) {
            if(it->second>=3) {
                ok=true;
                cnt=it->second;
                save=it->first;
                break;
            }
        }
        if(ok) {
            bool ok1=(str[0]==save) || (str[1]==save);
            bool ok2=(str[2]==save) || (str[3]==save);
            bool ok3=(str[4]==save) || (str[5]==save);
            if(ok1 && ok2 && ok3) {
                puts("YES");
            } else {
                puts("NO");
            }
        } else {
            puts("NO");
        }
        mp.clear();
    }
}
