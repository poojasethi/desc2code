#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string arr[7];
	while(n--)
	{
	cin>>arr[1]>>arr[2]>>arr[3]>>arr[5]>>arr[4]>>arr[6];
	int ans = 0;
	for(int i=1;i<=2;i++) 
	{
		for(int j=3;j<=5;j++)
		{     //  cout<<"\n"<<arr[i]<<arr[j]<<arr[j+1];
			if(arr[i]==arr[j]&&arr[j]==arr[j+1])
			{
				ans = 1;
				break;
			}
		}
		if(arr[i]==arr[6]&&arr[6]==arr[3])
		{
			ans =1;
		}
	}
	if(ans) cout<<"YES\n";
	else cout<<"NO\n";
	}
}
