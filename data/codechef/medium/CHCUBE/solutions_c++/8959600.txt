	#include<iostream>
	#include<string.h>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		int check[8][3]={{0,3,4},{0,3,5},{0,2,4},{0,2,5},{1,3,4},{1,3,5},{1,2,4},{1,2,5}};

		while(t>0)
		{
				char a[6][100];
				for(int i=0;i<6;i++)
				{
					cin>>a[i];
				}	
				int flag=0;
				for(int i=0;i<8;i++)
				{
					if(strcmp(a[check[i][0]],a[check[i][1]])==0 && strcmp(a[check[i][1]],a[check[i][2]])==0 )
					{
						flag=1;
						break;
					}	
				}	
				if(flag==1)
					cout<<"YES\n";
				else
					cout<<"NO\n";
				t--;

		}	
	}
