#include <bits/stdc++.h>
#include <stdio.h>
using namespace std;
int main(){
	string a[7];
	int test,count=0,flag =0;
	cin >> test;
	while(test--)
	{
		for (int i = 1; i < 7; ++i)
			cin>> a[i];
		for(int i = 1 ; i< 7 ; i++)
		{	
			if ((i==1 || i==2) && (a[i]==a[3] || a[i]==a[4]) && (a[i]==a[5] || a[i]==a[6]))
			{
				flag=1;
				break;
			}
			if ((i==3 || i==4) && (a[i]==a[1] || a[i]==a[2]) && (a[i]==a[5] || a[i]==a[6]))
			{
				flag=1;
				break;
			}
			if ((i==5 || i==6) && (a[i]==a[1] || a[i]==a[2]) && (a[i]==a[3] || a[i]==a[4]))
			{
				flag=1;
				break;
			}
		}
		if (flag==1)
		
			cout<<"YES"<<endl;
		else
			cout<< "NO"<<endl;
		flag=0;
	}
}
