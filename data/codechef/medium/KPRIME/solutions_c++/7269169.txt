#include<stdio.h>
int p[100001];
int table[100001][6];
void sieve()
{
   int i,j;
   for(i=0;i<=100000;i++)
       p[i]=0;
   for(i=1;i<=5;i++)
      table[1][i]=0; 
   for(i=2;i<=100000;i++)
   {
      if(p[i]==0)
         for(j=i;j<=100000;j+=i)
           p[j]++;
   }
   
   for(i=2;i<=100000;i++)
     for(j=1;j<=5;j++)
      {  table[i][j]=table[i-1][j];
         if(p[i]==j)
           table[i][j]++;
      }
}

int main()
{
  int t,a,b,k,j,temp;
  scanf("%d",&t);
  sieve();
  
  while(t--)
  { scanf("%d%d%d",&a,&b,&k);

    printf("%d\n",table[b][k]-table[a-1][k]);
  }
 
 
  return 0;  
  
}
