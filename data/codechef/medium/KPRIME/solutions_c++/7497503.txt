#include <bits/stdc++.h>
 using namespace std;
 
int num[100110];
int sum[6][100110];
bitset<100110> bs;
void sieve(){
	bs.set();
	bs[0] = bs[1] = 0;
	for(long long i=2 ; i<=100010 ; i++) if(bs[i]){
		for(long long j=i*2 ; j<=100010 ; j+=i) {bs[j] = 0; num[j]++;}
		num[i] = 1;
	}
}
int main(){
	sieve();
	for(int i=1 ; i<=5 ; i++)
		for(int j=2 ; j<100010 ; j++)
			sum[i][j] = sum[i][j-1] + (num[j]==i);
	int t; cin >> t;
	while(t--){
		int a,b,k; cin >> a >> b >> k;
		cout << sum[k][b] - sum[k][a-1] << endl;
	}
	return 0;
}
