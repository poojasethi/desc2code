#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	int a[100001];
	memset(a,0,sizeof(a));
	a[0] = 1;
	a[1] = 1;
	for (int i = 2; i <= 100000; i++)
	{
		if (a[i] == 0)
		{
			for (int j = 2; j*i <= 100000; j++)
			{
				a[j*i]++;
			}
			a[i] = 1;
		}
	}
	while (t--)
	{
		int x,y,k;
		scanf("%d%d%d",&x,&y,&k);
		int ctr = 0;
		for (int i = x; i <= y; i++)
		{
			if (a[i] == k)
				ctr++;
		}
		cout << ctr << endl;
	}	
	return 0;
}
