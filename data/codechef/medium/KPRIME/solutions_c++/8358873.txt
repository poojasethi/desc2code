int fact[100005];
int primeFactors(int n)
{
    int cnt=0;
    int sum=0;
    if(n%2==0)
    {
      cnt++;
      sum=sum+2;
    while (n%2 == 0)
     {
        //printf("%d ", 2);
        n = n/2;
     }
    }


    // n must be odd at this point.  So we can skip one element (Note i = i +2)
    for (int i = 3; i*i <= n; i = i+2)
    {
        // While i divides n, print i and divide n
        if(n%i==0)
        {
            sum+=i;
            cnt++;
        while (n%i == 0)
        {
            //printf("%d ", i);
            n = n/i;
        }
        }

    }

    // This condition is to handle the case whien n is a prime number
    // greater than 2
    if (n > 2)
        {  cnt++;
        sum+=n;
           // printf ("%d ", n);
        }

        //printf("\nno of distinct prime fact is %d",cnt);
         //printf("\nsum of distinct prime fact is %d",sum);
         return cnt;
}

void calc()
{
    int i;
     for(i=1;i<=100005;i++)
        {
            fact[i]=primeFactors(i);

        }
}
int main()
{

    calc();
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,l,r,k,ans=0,i;

        scanf("%d %d %d",&l,&r,&k);
        for(i=l;i<=r;i++)
        {
            if(fact[i]==k)
                ans++;
        }

        printf("%d\n",ans);

    }
    return 0;
}
