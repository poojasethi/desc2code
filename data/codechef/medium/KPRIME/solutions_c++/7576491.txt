int arr[100000];
int AB[100000];
// marks all mutiples of 'a' ( > 'a' but <= 'n') as 1.
void markMultiples(int A[], int a, int n)
{
    int i = 2, num;
    while ( (num = i*a) <= n )
    {
        arr[ num-1 ]++; // indexing starts from 0.
        ++i;
    }
}

// A function to print all prime numbers smaller than n but larger than m
void SieveOfEratosthenes(int n)
{
    int k=0;
    memset(arr, 0, n);
        /* arr[i] == 0 means i + 1 is prime
           arr[i] == 1 means i + 1 is not prime */
        for (int i=1; i<n; ++i)
        {
            if ( arr[i] == 0 )
            {
                //(i+1) is prime, save it and mark its multiples
                AB[k++]=i+1;
                markMultiples(arr, i+1, n);
            }
        }

}
