#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e5 + 10;
int cnt[MAXN];
int ans[6][MAXN];
int main(){
	memset(cnt, 0, sizeof cnt);
	for(int i = 2;i < MAXN;++i){
		if(!cnt[i]){
			for(int j = i;j < MAXN;j += i){
				cnt[j] += 1;
			}
		}
	}
	memset(ans, 0, sizeof ans);
	for(int i = 1;i <= 5;++i){
		for(int j = 1;j < MAXN;++j){
			ans[i][j] = ans[i][j - 1] + (cnt[j] == i);
		}
	}
	// for(int i = 1;i <= 20;++i){
	// 	cout << i << " " << ans[1][i] << "\n";
	// }
	int t;
	scanf("%d",&t);
	while(t--){
		int A, B, K;
		scanf("%d%d%d",&A,&B,&K);
		printf("%d\n",  ans[K][B] - ans[K][A - 1]);
	}
	return 0;
}
