#include <cstdio>
#include <cmath>
#include <algorithm>
#include <bitset>
#include<cstring>
#include<iostream>
#define MAX 100009
using namespace std;
int primes[100005];
bool isprime[100005];
int answer[5][100002]={0};
void sieve()	
{
	memset(isprime,true,sizeof isprime);
	for(int i=0;i<=100005;i++)primes[i]=0;
	for(int i=2;i<=(100005);i++)
	{
		if(isprime[i])
		{
			primes[i]=1;
			for(int j=(2*i);j<=100005;j+=i)
			{
				isprime[j]=false;
				primes[j]+=1;
			}
		}
	}
	for (int i=2; i<100002; i++) 
	{
        	for (int j=1; j<=5; j++) 
		{
            		answer[j-1][i] = answer[j-1][i-1];
        	}
        	if (primes[i] > 0 && primes[i] <= 5) 
		{
            	answer[primes[i]-1][i]++;
        	}
    	}
}
int main()
{
	sieve();
	int t,c=0;
	cin>>t;
	while(t--)
	{
		int a,b,k,j=0,c=0;
		cin>>a>>b>>k;
		cout<<answer[k-1][b] - answer[k-1][a-1]<<endl;
	}
}
