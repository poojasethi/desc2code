#include<stdio.h>
#include<math.h>

int main()
{
    int tc,k;
    long int a,b,i,j,sieve[100001]={0},ans;

    for(i=2;i<=100000;i++)
    {
        if(sieve[i] == 0)
        {
            sieve[i] = 1;
            for(j=2;i*j<=100000;++j)
                sieve[i*j]++;
        }
    }

    scanf("%d",&tc);

    while(tc--)
    {
        scanf("%ld%ld%d",&a,&b,&k);

        ans = 0;
        for(i=a;i<=b;++i)
            if(sieve[i] == k)
                ans++;
        printf("%ld\n",ans);
    }
    return 0;
}
