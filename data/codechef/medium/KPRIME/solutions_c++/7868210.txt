#include<iostream>
int n[100005]={0};
int cnt[100005][6];

using namespace std;

int main()
{
	int i,j;
	long long int k,c,a,b;
	n[0]=-1;
	n[1]=-1;
	for(i=2;i<=100000;i++)
	{
		if(!n[i])
		{
			for(j=1;i*j<=100000;j++)
			{
				n[i*j]+=1;
			}
		}
	}
	for(i=2;i<=100000;i++)
		for(int j=1;j<=5;j++)
			if(n[i]==j)
				cnt[i][j]=cnt[i-1][j]+1;
			else
				cnt[i][j]=cnt[i-1][j];
	/*
	for(k=0;k<=30;k++)
	cout<<k<<","<<n[k]<<"  ";
	*/
	int t;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>k;
		cout<<cnt[b][k]-cnt[a-1][k]<<"\n";
	}
	return 0;
	
}
