// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define s(x) scanf("%d", &x)

int dp[100011][6];
int a[100011];
vector <int> p;

void sieve()
{
    int i, j;
    for (i = 2; i <= 320; i++) {
        if (a[i] == 0) {
            p.pb(i);
            for (j = 2; i*j <= 100000; j++)
                a[i*j] = 1;
        }
    }
}

int factors(int i)
{
    int j, cnt = 0;
    for (j = 0; i > 1 && j < p.size(); j++) {
        if (i % p[j] == 0) {
            cnt++;
            while (i % p[j] == 0)
                i /= p[j];
        }
    }
    if (i > 1)
        cnt++;
    return cnt;
}

void compute()
{
    int i, cnt;
    for (i = 2; i <= 100000; i++) {
        cnt = factors(i);
        if (cnt <= 5)
            dp[i][cnt] = 1;
    }
    int j;
    for (i = 1; i <= 100000; i++) {
        for (j = 1; j <= 5; j++)
            dp[i][j] += dp[i-1][j];
    }
}

int main()
{
    sieve();
    compute();
    int t, a, b, k;
    s(t);
    while (t--) {
        s(a);
        s(b);
        s(k);
        printf("%d\n", dp[b][k]-dp[a-1][k]);
    }
    return 0;
}
