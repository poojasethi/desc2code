#include<stdio.h>
#include<math.h>
int sieve[100001];
int total[5][100001];
//#define ll long long
int main()
{
    //printf("CEHCK2\n");
    int t,i,j;
    for(int i=0;i<100001;i++)
        sieve[i]=0;
    //
    for(int i=2;i<100001;i++)
    {
        if(sieve[i]==0)
        {
            for(int j=i;j<100001;j=j+i)
                sieve[j]++;
        }
    }
    int no1[100001];
    for(j=0;j<100001;j++)
        no1[j]=0;
    for(j=2;j<100001;j++)
        if(sieve[j]==1)
            no1[j]=no1[j-1]+1;
        else
            no1[j]=no1[j-1];

    for(i=0;i<5;i++)
        total[i][0]=total[i][1]=0;
    for(i=2;i<100001;i++)
    {
        if(sieve[i]==1)
            total[0][i]=total[0][i-1]+1;
        else
            total[0][i]=total[0][i-1];
        if(sieve[i]==2)
            total[1][i]=total[1][i-1]+1;
        else
            total[1][i]=total[1][i-1];
        if(sieve[i]==3)
            total[2][i]=total[2][i-1]+1;
        else
            total[2][i]=total[2][i-1];
        if(sieve[i]==4)
            total[3][i]=total[3][i-1]+1;
        else
            total[3][i]=total[3][i-1];
        if(sieve[i]==5)
            total[4][i]=total[4][i-1]+1;
        else
            total[4][i]=total[4][i-1];
    }

    scanf("%d",&t);
    while(t--)
    {
        int a,b,k,cnt=0;
        scanf("%d%d%d",&a,&b,&k);
        printf("%d\n",total[k-1][b]-total[k-1][a-1]);
       // printf("%d\n",cnt);
    }
    return 0;
}
