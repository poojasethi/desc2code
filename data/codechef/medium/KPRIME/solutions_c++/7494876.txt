/**************************Code****************************/

int lp[100010], dp[6][100010];

int main()
{
	for( int i = 2 ; i < 100010 ; i ++ )
		if( !lp[i] )
			for( int j = i ; j < 100010 ; j += i )
				lp[j] = i;
	for( int i = 1 ; i < 100010 ; i ++ )
	{
		int x = i, cnt = 0;
		while( x != 1 )
		{
			int p = lp[x];
			while( x % p == 0 )
				x /= p;
			cnt ++;
		}
		if( cnt <= 5 )
			dp[cnt][i] = 1;
	}
	for( int i = 0 ; i < 6 ; i ++ )
		for( int j = 1 ; j < 100010 ; j ++ )
			dp[i][j] += dp[i][j-1];
	int t, a, b, k;
	scanf( "%d", &t );
	while( t -- )
	{
		scanf( "%d%d%d", &a, &b, &k );
		printf( "%d\n", dp[k][b] - dp[k][a-1] );
	}
	return 0;
}
