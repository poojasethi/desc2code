#include<bits/stdc++.h>
using namespace std;
int ctr[100005]={0};
void sieve(int MAX)
{
	int i,j;
	for(i=2;i<MAX;i++)
		if(ctr[i] == 0)
			for(j=i;j<MAX;j+=i)
				ctr[j]++;
}
 
int main()
{
	int t,count,k,a,b,i;
	cin>>t;
	sieve(100005);
	while(t--)
	{
		count = 0;
		cin>>a>>b>>k;
		for(i=a;i<=b;i++)
			if(ctr[i]==k)
				count++;
		cout<<count<<"\n";
	}
	return 0;
}
