#include<iostream>
#include<stdio.h>
#define max 100000

using namespace std;
int arr[100000];
// int count[500004];

int main()
 {
    int i,j,t,a,b,k=1,l;
    int ans=0;
    for(i=2;i<=max;i+=2)
    { arr[i]++;
    }

    i=3;
    while(i<=max)
     {
       if(arr[i]!=0)
         { i=i+2; }
        else
        {
         for(j=i;j<=max;j+=i)
           {
             arr[j]++;
           }
        }
     }

   scanf("%d",&t);
   while(t--)
    { ans=0;
      scanf("%d %d %d",&a,&b,&l);
      for(i=a;i<=b;i++)
      {
          if(arr[i]==l)
            ans++;
      }
   //   segtree(1,b,0,l);

    // ans=query(1,b,a,b,0);
     printf("%d\n",ans);
    }

  return 0;
 }
