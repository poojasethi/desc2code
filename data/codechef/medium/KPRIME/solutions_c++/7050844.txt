#include <iostream>

using namespace std;

int main()
{
    int t, i, j, a, b, k, count;
    int kp[100001] = {0};
    kp[1] = -1;
    cin>>t;
    for(i=2; i<100001; i++)
    {
        if(kp[i] == 0)
        {
            for(j=1; j<=(100000/i); j++)
            {
                kp[i*j]++;
            }
        }
    }
    while(t--)
    {
        cin>>a>>b>>k;
        count = 0;
        for(i=a; i<=b; i++)
        {
            if(kp[i]==k)
                count++;
        }
        cout<<count<<endl;
    }
    return 0;
}
