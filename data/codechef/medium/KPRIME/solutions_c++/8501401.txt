#include<stdio.h>
#include<iostream>

long long int temp;

int arr[100001]={0};
int arr1[100001]={0};

void sieve(int arr[],int a,int b)
{
   int i=1,num;
   while((num=(i*a))<=b)
   {
   arr1[num]++;

     arr[num]=1;
     i++;
   }
}

int main()
{

for(int i=2;i<=100000;i++)
  {
     if(arr[i]==0)
      sieve(arr,i,100000);
  }

int r,a,b,t;
scanf("%d",&t);
while(t--)
{
  scanf("%d %d %d",&a,&b,&r);
  long int count=0;
  for(long int k=a;k<=b;k++)
   {
      if(arr1[k]==r)
       count++;

   }
   printf("%ld\n",count);
}


return 0;
}
