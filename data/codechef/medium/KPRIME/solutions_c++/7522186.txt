#include<bits/stdc++.h>
using namespace std;
#define maxi 100010

int comp[maxi];

void pre()
{
	comp[1]=0;
	for(int i=2;i<maxi/2;i++)
	{
		if(comp[i])
		{
			continue;
		}
		for(int j=i*2;j<maxi;j+=i)
		{
			comp[j]++;
		}
	}
	for(int i=2;i<maxi;i++)
	{
		if(comp[i]==0)
		{
			comp[i]=1;
		}
	}
}


int main()
{
	ios::sync_with_stdio(false);
	pre();
	// cout<<comp[5]<<" "<<comp[100]<<"\n";
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,k;
		cin>>a>>b>>k;
		int count=0;
		for(int i=a;i<=b;i++)
		{
			if(comp[i]==k)
			{
				count++;
			}
		}
		cout<<count<<"\n";
	}
	return 0;
}
