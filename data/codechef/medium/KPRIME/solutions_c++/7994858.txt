#include<iostream>
#include<string.h>
using namespace std;
void sieve(int a[], int k, int n)
{
    int i=0;
    while(++i)
    {
        if((k*i)<=n)
            {a[k*i]++;}
        else
            break;
    }
}
int main(){
    int a[100001];int t, l, b, k, counter;
    memset(a,0,sizeof(a));
    for(int i=2;i<=100000;i++)
    {
        if(a[i]==0)
            sieve(a,i,100000);
    }
    cin>>t;
    while(t--)
    {
        cin>>l>>b>>k;
        counter=0;
        for(int i=l;i<=b;i++)
        {
            if(a[i]==k)
                counter++;
        }
        cout<<counter<<endl;
    }
}
