#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	int a[100001],mark[100001],i,j,a1,b,k,count=0;
	for(i=0;i<100001;i++)
	{
		a[i]=0;
		mark[i]=0;
	}

	for(int i=2;i<100001;i++){
		if(a[i]!=-1){
			mark[i]++;
			for(j=2*i;j<100001;j=j+i)
			{
				a[j]=-1;
				mark[j]++;
			}
		}
	}

	
	while(t--){
		count=0;
		scanf("%d %d %d",&a1,&b,&k);
		for(i=a1;i<=b;i++){
			if(mark[i]==k)
				count++;
		}
		printf("%d\n",count);
	}	

}
