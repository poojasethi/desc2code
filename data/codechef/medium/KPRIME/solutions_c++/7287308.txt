#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int p[100001]={0};
int h[100000][6];
void kpo()
{
  for(int i=2;i<100001;i++)
 
    if(p[i]==0)
      for(int j=i;j<100001;j=j+i)
      p[j]++;

  for(int i=2;i<100001;i++)
     for(int j=1;j<6;j++)
     {
      if(p[i]==j)
           h[i][j]=h[i-1][j]+1;
      else
           h[i][j]=h[i-1][j];
     } 

}


int main()
{
  kpo();
  int t;
  cin>>t;
  while(t--)
  {
    int a,b;
    cin>>a>>b;
    int k;
    cin>>k;
    int ans;
    ans=h[b][k]-h[a-1][k];
    cout<<ans<<"\n";
  }
}
