#include<bits/stdc++.h>
using namespace std;
int prime[1000001];
int main()
{
    int i,j;
    memset(prime,0,sizeof prime);
    for(i=2;i<=1000000;i++)
    {
        if(prime[i]==0)
        {
        for(j=i;j<=1000000;j=j+i)
        {
prime[j]++;
        }
        }
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int a,b,k;
        cin>>a>>b>>k;
        int count=0;
        for(i=a;i<=b;i++)
        {
            if(prime[i]==(k))
                count++;

        }
        cout<<count<<endl;


    }



}
