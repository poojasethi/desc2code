#include<bits/stdc++.h>
using namespace std;
int factors[100001];
void prework()
{
    for(int i=0; i<100001; i++)
        factors[i]=0;
    bool p[100001];
    for(int i=0; i<100001; i++)
        p[i]=true;
    p[0]=false;
    p[1]=false;
    for(int i=2; i<100001; i++){
        if(p[i]==true){
            factors[i]=1;
            for(int j=i+i; j<100001; j+=i){
                p[j]=false;
                factors[j]++;
            }
        }
    }
}
int main(void)
{
    int t;
    prework();
    for(scanf("%d", &t); t--; )
    {
        int a, b, k;
        scanf("%d %d %d", &a, &b, &k);
        int cnt=0;
        for(int i=a; i<=b; i++){
            if(factors[i]==k)
                cnt++;
        }
        printf("%d\n", cnt);
    }
    return 0;
}
