#include<bits/stdc++.h>
using namespace std;

int main()

{

std::ios::sync_with_stdio(false);
int P;
int T;
cin>>T;
int i, j;
int kprime[100100];
int count;

for(i=2; i<=100000; i++)
{
count=0;
P=i;
if(P%2==0)
{
count++;
while(P%2==0)
P=P/2;
}

for(j=3; j<=sqrt(P); j+=2)
{
if(P%j==0)
{
count++;
while(P%j==0)
P=P/j;
}
}
if(P>2)
count++;

kprime[i]=count;
}
				    
while(T--)
{
int A, B, K;
cin>>A>>B>>K;
count=0;
for(i=A; i<=B; i++)
{
if(kprime[i]==K)
count++;
}
cout<<count<<"\n";
}

return 0;
}
