#include<bits/stdc++.h>
using namespace std;

#define MAX 100001
int prime[MAX]={0};
int factor[MAX]={0};

void seive()
{
	for(int i=2;i<=MAX;i++)
	{
		if(prime[i]==0)
		{
			for(int j=2*i;j<=MAX;j+=i)
			{
				prime[j]=1;
				factor[j]++;
			}
		}
	}
	for(int i=2;i<=MAX;i++)
	if(factor[i]==0)
	factor[i]++;
	return;
}

int main()
{
	seive();
	int t,a,b,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d",&a,&b,&k);
		int  ans = 0;
		for(int i=a;i<=b;i++)
		{
			if(factor[i]==k)
			ans++;
		}
		printf("%d\n",ans);
	}
	return 0;
}
