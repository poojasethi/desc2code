inline void get(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int a[1000005];

int main()
{
	int T;
	get(T);
	for (int __rep = 1; __rep <=T; __rep++)
	{
		int n;
		get(n);
		F(i,n)
			get(a[i]);
		F(i,n-1)
		{
			if (a[i] <= a[i+1])
				continue;
			swap(a[i], a[i+1]);
			i++;
		}
		bool flag = true;
		F(i,n-1)
		{
			if (a[i] > a[i+1])
			{
				flag = false;
				printf("NO\n");
				break;
			}
		}
		if (flag)
			printf("YES\n");
	}
	return 0;
}
