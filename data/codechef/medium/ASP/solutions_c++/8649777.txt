ll a[1000010],b[1000010];

int main()
{   
    ios::sync_with_stdio(false);
    ll t;
    cin>>t;
    while(t--){
        ll n,i,j,k;
        cin>>n;
        for(i=0;i<n;i++)
            cin>>a[i];
        i=0;
        while(i<n)
        {
            if((i+1)>=n)
            {
                b[i]=a[i];
                break;
            }
            else{
                if(a[i]<=a[i+1])
                {
                    b[i]=a[i];
                    i++;
                }
                else{
                    b[i]=a[i+1];
                    b[i+1]=a[i];
                    i+=2;
                }
            }
        }
        for(i=0;i<(n-1);i++)
            if(b[i]>b[i+1])
                break;
        if(i<(n-1))
            cout<<"NO\n";
        else cout<<"YES\n";
    }
    return 0;
}
