#include<iostream>
#include<cstdio>
 
using namespace std;
 
int check(long long int tarx,long long int tary,long long int ax,long long int ay,long long int bx,long long int by,long long int cx,long long int cy)
{
     long long   int a1,a2,b1,b2,d1,d2,valx,valy,det;
        a1=-1*2*(ax-cx);
        a2=-1*2*(ax-bx);
        b1=2*(cy-ay);
        b2=2*(by-ay);
        d1=cy*cy+cx*cx-ax*ax-ay*ay;
        d2=by*by+bx*bx-ax*ax-ay*ay;
        det=a1*b2-a2*b1;
//cout<<det<<"-<";
        //cout<<det<<endl;
        if(det==0)
                return 0;
        else
                {
                valx=d1*b2-d2*b1;
                valy=a1*d2-a2*d1;
                //cout<<centrex<<" "<<centrey<<" "<<tarx<<" "<<tary<<endl;
                if((tarx*det-valx)*(tarx*det-valx)+(tary*det-valy)*(tary*det-valy)<=(ax*det-valx)*(ax*det-valx)+(ay*det-valy)*(ay*det-valy))
                        return 1;
                else
                        return 0;               
                }
 
}
int main()
{
long long int t;
scanf("%lld",&t);
 
while(t--)
{
        long long int i,j,k,l,n,count=0,x[32]={0},y[32]={0};
       scanf("%lld",&n);
        for(i=0;i<n;i++)
                {
                        scanf("%lld %lld",&x[i],&y[i]);
                }
        for(i=0;i<n;i++)
                for(j=i+1;j<n;j++)
                        for(k=j+1;k<n;k++)
                                for(l=k+1;l<n;l++)
                                        {
                                                if(check(x[i],y[i],x[j],y[j],x[k],y[k],x[l],y[l]))
                                                        count++;
                                                if(check(x[j],y[j],x[i],y[i],x[k],y[k],x[l],y[l]))
                                                        count++;
                                                if(check(x[k],y[k],x[j],y[j],x[i],y[i],x[l],y[l]))
                                                        count++;
                                                if(check(x[l],y[l],x[j],y[j],x[k],y[k],x[i],y[i]))
                                                        count++;
 
                                        }
long long int total=(n*(n-1)*(n-2)*(n-3))/6;
double prob=(double)count/total;
printf("%.7lf\n",prob);
//cout<<check(0,9,0,1,0,0,0,-1);                                
}
 return 0;
}
