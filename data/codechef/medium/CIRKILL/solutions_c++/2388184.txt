#include<iostream>
#include<cstdio>
using namespace std;

struct cordinate
{
   int x;
    int y;
};


int det(int a[][3])
{
return (a[0][0]*((a[1][1]*a[2][2]) - (a[2][1]*a[1][2])) -a[0][1]*(a[1][0]*a[2][2] - a[2][0]*a[1][2]) + a[0][2]*(a[1][0]*a[2][1] - a[2][0]*a[1][1]));
}
int aa,da,ea,fa;

int chk(int x,int y)
{
if(aa>0){
if((aa*(x*x+y*y)+da*x+ea*y+fa)<=0)
return 1;
return 0;}
else{
if((aa*(x*x+y*y)+da*x+ea*y+fa)>=0)
return 1;
return 0;
}
}


int main()
{
    int t,i,j,k;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        struct cordinate tile[n];
        int i;
        for(i=0;i<n;i++)
        {scanf("%d %d",&tile[i].x,&tile[i].y);}


   int ax,bx,cx,ay,by,cy;
   int ycount=0,total=(n*(n-1)*(n-2)*(n-3))/6;
   double prob;

for(i=0;i<n-2;i++)
{
 for(j=i+1;j<n-1;j++)
  {
   for(k=j+1;k<n;k++)
   {
    ax=tile[i].x;bx=tile[j].x;cx=tile[k].x;
   ay=tile[i].y;by=tile[j].y;cy=tile[k].y;

 int a1[3][3]={{ax,ay,1},{bx,by,1},{cx,cy,1}};
      aa=det(a1);

  if(aa)
   {
      int a[3][3]={ax*ax+ay*ay,ay,1,bx*bx+by*by,by,1,cx*cx+cy*cy,cy,1};
      da=-1*det(a);
      a[0][1]=ax;a[1][1]=bx;a[2][1]=cx;
      ea=det(a);
      a[0][2]=ay;a[1][2]=by;a[2][2]=cy;
      fa=-1*det(a);

       for(int i=0;i<n;i++)
           {
              long double e,f;
              e=tile[i].x;f=tile[i].y;
              if(!( ((e==ax)&&(f==ay)) || ((e==bx)&&(f==by)) || ((e==cx)&&(f==cy))))
                {
                    if(chk(e,f))
                    {ycount++;
                    }
                }
                }
      }

}
}
}
prob=(ycount)/(long double)total;

printf("\n %.20g \n",prob);
}
return 0;
}
