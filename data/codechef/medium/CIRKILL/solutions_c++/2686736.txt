#include<stdio.h>
#include<math.h>
#define EPS 1e-007
  int x[100],y[100];
  long double a1,a2,a3,b1,b2,b3,c1,c2,c3,g1,g2,g3,r1,p1,p2;
  int i,j,k;
void getdet1()
{
a1=x[i]*x[i]+y[i]*y[i];
a2=y[i];
a3=1;
b1=x[j]*x[j]+y[j]*y[j];b2=y[j];b3=1;
c1=x[k]*x[k]+y[k]*y[k];c2=y[k];c3=1;

g1=a1*(b2*c3-b3*c2)-a2*(b1*c3 -b3*c1)+a3*(b1*c2-b2*c1);
}

void getdet2()
{
a1=x[i];
a2=x[i]*x[i]+y[i]*y[i];
a3=1;

b1=x[j];
b2=x[j]*x[j]+y[j]*y[j];
b3=1;

c1=x[k];
c2=x[k]*x[k]+y[k]*y[k];
c3=1;

g2=a1*(b2*c3-b3*c2)-a2*(b1*c3 -b3*c1)+a3*(b1*c2-b2*c1);
}
void getdet3()
{
a1=x[i];a2=y[i];a3=1;b1=x[j];b2=y[j];b3=1;c1=x[k];c2=y[k];c3=1;
g3=a1*(b2*c3-b3*c2)-a2*(b1*c3 -b3*c1)+a3*(b1*c2-b2*c1);

return ;

}

int main()
{
     int t,w,n;
long double count;
double nc3;
     scanf("%d",&t);
     while(t--)
     {
     count=0;
         scanf("%d",&n);
         nc3=n*(n-1)*(n-2)/6;
          for(i=0;i<n;i++)
          {
         scanf("%d",&x[i]);
         scanf("%d",&y[i]);
          }

          for(i=0;i<n;i++)
          {
               for(j=i+1;j<n;j++)
               {
                    for(k=j+1;k<n;k++)
                    {
                      getdet3();
                       if(g3==0){continue;}
                       getdet1();
                       getdet2();
                     g3=g3*2;
                       p1=g1;
                       p2=g2;
                      long double rad=sqrt((x[j]*g3-p1)*(x[j]*g3-p1) +(y[j]*g3-p2)*(y[j]*g3-p2));
                       for(w=0;w<n;w++)
                       {
                       
                           if(w==i||w==j||w==k){continue;}
                          r1=sqrt((x[w]*g3-p1)*(x[w]*g3-p1) +(y[w]*g3-p2)*(y[w]*g3-p2));
                           if(r1<=rad){count++;}

                       }
                    }
               }
          }
         ;
      printf("%.10llf\n",count/((n-3)*nc3));
     }
     return 0;
}
