#include<iostream>
#include<cstdio>
#define ll long long 
 
using namespace std;
 
int check(ll int tarx,ll int tary,ll int ax,ll int ay,ll int bx,ll int by,ll int cx,ll int cy)
{
     ll int centrex,centrey;
    ll int a1,a2,b1,b2,d1,d2,valx,valy,det;
    a1=-1*2*(ax-cx);
	a2=-1*2*(ax-bx);
	b1=2*(cy-ay);
	b2=2*(by-ay);
	d1=cy*cy+cx*cx-ax*ax-ay*ay;
	d2=by*by+bx*bx-ax*ax-ay*ay;
    //d1=a1*(cx+ax)+b1*(cy+ay);
    //d2=a2*(bx+ax)+b2*(by+ay);
	det=a1*b2-a2*b1;
    //d1=d1/2;
    //d2=d2/2;
	//cout<<det<<endl;
	if(det==0)
		return 0;
	else
		{
		valx=d1*b2-d2*b1;
		valy=a1*d2-a2*d1;
		centrex=valx;
		centrey=valy;
		//cout<<centrex<<" "<<centrey<<" "<<tarx<<" "<<tary<<endl;
		if((tarx*det-centrex)*(tarx*det-centrex)+(tary*det-centrey)*(tary*det-centrey)<=(ax*det-centrex)*(ax*det-centrex)+(ay*det-centrey)*(ay*det-centrey))
			return 1;
		else
			return 0;		
		}
 
}
int main()
{
ll int t;
scanf("%lld",&t);
 
while(t--)
{
        ll int i,j,k,l,n,count=0,x[30]={0},y[30]={0};
       scanf("%lld",&n);

                {
                        scanf("%lld %lld",&x[i],&y[i]);
                }

                for(j=i+1;j<n;j++)
                        for(k=j+1;k<n;k++)
                                for(l=k+1;l<n;l++)
                                        {
                                                if(check(x[i],y[i],x[j],y[j],x[k],y[k],x[l],y[l]))
                                                        count++;
                                                if(check(x[j],y[j],x[i],y[i],x[k],y[k],x[l],y[l]))
                                                        count++;
                                                if(check(x[k],y[k],x[j],y[j],x[i],y[i],x[l],y[l]))
                                                        count++;
                                                if(check(x[l],y[l],x[j],y[j],x[k],y[k],x[i],y[i]))
                                                        count++;
 
                                        }
ll int total=(n*(n-1)*(n-2)*(n-3))/6;
