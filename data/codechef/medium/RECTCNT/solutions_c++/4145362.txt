#include <iostream>
#include <map>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;


int main()
{
    int n,x,y;
    for ( ;; ) {
        scanf("%d", &n);
        if ( !n ) break;
        vector < pair<int,int> > v;
        for ( int i = 0; i < n; i++ ) {
            scanf("%d%d", &x, &y);
            v.push_back(make_pair(x,y));
        }
        sort(v.begin(),v.end());
        map < pair<int,int>, int > m;
        long long ans = 0;
        for ( int i = 0; i < v.size(); i++ ) {
            if ( i+1 < v.size() ) {
                 if ( v[i].first == v[i+1].first ) {
                    ans += m[make_pair(v[i].second,v[i+1].second)];  
                    m[make_pair(v[i].second,v[i+1].second)]++;
                    i++;
                 } 
            }       
        }
        printf("%lld\n", ans);
    }
    return 0;
}
