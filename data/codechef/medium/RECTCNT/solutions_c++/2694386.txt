#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
 
pair<int,int>p[100005];
pair<int,int>a[100005];
int main()
{
    int n,i,k,j;
    long long int ans,c;
    while(1)
    {
        scanf("%d",&n);
        if(n==0)
        break;
        for(i=0;i<n;i++)
        scanf("%d%d",&p[i].first,&p[i].second);
        sort(p,p+n);
        k=0;
        for(i=1;i<n;i++)
        {
            if(p[i].first==p[i-1].first)
            {
                a[k].first=p[i].second;
                a[k++].second=p[i].second-p[i-1].second;
            }
        }
        sort(a,a+k);
        i=0;ans=0;
        while(i<k-1)
        {
            j=i+1;
            c=1;
            while(a[j]==a[j-1])
            {
                c++;
                j++;
            }
            ans=ans+(c*(c-1))/2;
            i=j;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
