#include<stdio.h>
#include<algorithm>
using namespace std;
pair<int,int> p[100001];
pair<int,int> l[100001];
int main()
{
    int n,i,j,k;
    long long int ans,c;
    scanf("%d",&n);
    while(n!=0)
    {
        for(i=0;i<n;i++)
        scanf("%d%d",&p[i].first,&p[i].second);
        sort(p,p+n);
        k=0;
        for(i=1;i<n;i++)
        {
            if(p[i].first==p[i-1].first)
            {
                l[k].first=p[i].second;
                l[k++].second=p[i].second-p[i-1].second;
            }
        }
        sort(l,l+k);
        ans=0;
        for(i=0;i<k-1;)
        {
            c=1;
            j=i+1;
            while(l[j]==l[j-1])
            {
                c++;
                j++;
            }
            i=j;
            ans+=(c*(c-1))/2;
        }
        printf("%lld\n",ans);
        scanf("%d",&n);
    }
    return 0;
}
