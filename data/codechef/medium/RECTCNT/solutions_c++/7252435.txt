#include<bits/stdc++.h>
using namespace std;
main()
{


    pair<int, int>a[100001];
    pair<int, int>b[100001];
    int i,j,n,k;
    long long c;
    long long ans=0;
    while(1)
    {

    scanf("%d",&n);
    if(n==0)
        break;
    for(i=0;i<n;i++)
    {
        scanf("%d%d",&a[i].first,&a[i].second);

    }
    sort(a,a+n);
     k=0;
    for(i=0;i<n-1;i++)
    {
        if(a[i].first==a[i+1].first)
        {
            b[k].first=a[i].second;
            b[k++].second=a[i+1].second;
        }
    }
    sort(b,b+k);
    i=0;

    ans=0;
    while(i<k-1)
    {
        j=i+1;
        c=1;
        while(b[j].first==b[j-1].first && b[j].second==b[j-1].second)
        {
            c++;
            j++;
        }
        ans+=(c*(c-1))/2;
        i=j;
    }
    printf("%lld\n",ans);
}
}
