#include<cstdio>
#include<algorithm>
#include<vector>
#include<algorithm>
#include<utility>

using namespace std;

vector< pair<int, int> >v, sides;

pair<int, int> m;
int N;
int x, y;


int main()
{


   while(1)
   {
          scanf("%d", &N);
          if(N == 0 )break;
       
        for(int i=0; i<N; i++)
        {
            scanf("%d%d", &x, &y);
            m = make_pair(x, y);
            v.push_back(m);
        }


        sort(v.begin(), v.end());
        

        for(int i=1; i<N; i++)
        {
            if(v[i].first == v[i-1].first)
            {
                m= make_pair(v[i].second, v[i].second - v[i-1].second);
                sides.push_back(m);
            }
        }
        sort(sides.begin(), sides.end());


        long long cnt = 0;
        //for(int i=0; i<sides.size(); i++)
        //printf("%d\n", sides[i]);
        if(sides.size())
        {
            long long k =1;
        for(int j=1;j<sides.size(); j++)
        {
          if(sides[j]== sides[j-1])
          {
              k++;
          }
          else
          {

              cnt+= k*(k-1)/2;
              k = 1;

          }
        }


          cnt+= k*(k-1)/2;
        }
        printf("%lld\n", cnt);
         v.clear();
          sides.clear();

}

}
