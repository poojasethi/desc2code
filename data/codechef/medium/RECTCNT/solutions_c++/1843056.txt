#include<algorithm>
#include<iostream>
#include<vector>

using namespace std;

#define fi first
#define se second

typedef pair<int, int> PII;

long n;
vector<PII> p, seg;

int main() 
{
    long x, y;
    do {
        scanf("%ld",&n);
		if (!n) break;
        p.clear();

        for (int i = 0; i < n; i++) 
		{
            scanf("%ld%ld",&x,&y);
            p.push_back(PII(x, y));
        }
        sort(p.begin(), p.end());
        seg.clear();
        for (int i = 1; i < n; i++) 
		{
            if (p[i].fi == p[i - 1].fi) 
			{
                seg.push_back(PII(p[i].se, p[i].se - p[i - 1].se));
            }
        }
        sort(seg.begin(), seg.end());
        long long res = 0;
        if (seg.size()) 
		{
            PII last = seg[0];
            long long cnt = 1;
            for (int i = 1; i < seg.size(); i++) 
			{
                if (seg[i] == seg[i - 1]) 
					cnt++;
                else 
				{
                    res += cnt * (cnt - 1) / 2;
                    cnt = 1;
                }
            }
            res += cnt * (cnt - 1) / 2;
        }

        printf("%lld\n",res);
    } while (1);
    return 0;
}
