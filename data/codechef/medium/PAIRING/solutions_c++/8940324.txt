#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
#include<map>
using namespace std;

vector<pair<int,int> > a;
vector<int> ans;

int main(){
    int n,t,i,m,x,y;
    scanf("%d",&t);
    map<int,int> ma;

    while(t){
        scanf("%d %d",&n,&m);

        a.resize(m);

        for(i = 0 ; i < m ; i++){
            scanf("%d",&x);
            scanf("%d",&y);

            a[i] = make_pair(x,y);
        }

        for(i = m-1 ; i >= 0 ; i--){

            if(ma.find(a[i].first) == ma.end() && ma.find(a[i].second) == ma.end()){
                ma[a[i].first] = 1;
                ma[a[i].second] = 1;

                ans.push_back(i);
            }

        }

        for(i = ans.size() - 1 ; i >= 0 ; i--)
            printf("%d ",ans[i]);

        printf("\n");
        ma.clear();
        ans.clear();

        t--;
    }

}
