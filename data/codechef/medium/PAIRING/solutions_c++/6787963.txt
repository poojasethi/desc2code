#include<stdio.h>
char arr[1000];
int gt[10000][2];
void fun(int i)
{
     if(i<0)
     return;
     
     if(arr[gt[i][0]]=='0' && arr[gt[i][1]]=='0')
     {
                       arr[gt[i][0]]='1';
                       arr[gt[i][1]]='1';
                       fun(i-1);
                       printf("%d ",i);
                       }
     else
     fun(i-1);
     }
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
              int n,m;
              scanf("%d %d",&n,&m);
              int i;
                          for(i=0;i<n;i++)
                          arr[i]='0';
              for(i=0;i<m;i++)
              scanf("%d %d",&gt[i][0],&gt[i][1]); 
              
              fun(m-1);
              printf("\n");
              }
    return 0;
}
