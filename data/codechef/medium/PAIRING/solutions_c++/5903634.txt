#include<iostream>
#include<algorithm>
#include<vector>
#include<stdlib.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n >>m;
		vector<pair<int, int> > chef;
		vector<pair<int, int> > :: reverse_iterator it;
		vector<int>:: iterator it1;
		vector<int> mark(n);
		vector<int> ans;
		for(int i=0; i < m; ++i)
		{
			int a,b;
			cin>>a >>b;
			chef.push_back(make_pair(a,b));
		}
		m--;
		for(it=chef.rbegin(); it!=chef.rend(); ++it)
		{
			if((mark[it->first]==0 && mark[it->second]==0))
			{
				mark[it->first]=1; 
				mark[it->second]=1;
				ans.push_back(m);
			}
			--m;
		}
		sort(ans.begin(),ans.end());
		for(it1=ans.begin(); it1!=ans.end(); ++it1)
			cout<<*it1<<" ";
		cout<<endl;
	}
}
