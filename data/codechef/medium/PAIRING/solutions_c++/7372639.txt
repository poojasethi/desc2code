#include<cstdio>
#include<algorithm>
#include<string.h>
#define gc() getchar_unlocked()
using namespace std;
struct employee{
int x;
int y;
};
bool used[1001];
struct employee e[10001];
 
static inline int fastio()
{
    int x=0;
    char c;
    c=gc();
    while(c<'0'||c>'9')
        c=gc();
    while(c>='0'&&c<='9')
    {
        x=(x<<1)+(x<<3)+c-'0';
        c=gc();
    }
    return x;
}
 
int main()
{
    int t,i,n,pairs,a[1001];
    //cin>>t;
    t=fastio();
    while(t--)
    {
        //cin>>n>>pairs;
        n=fastio();
        pairs=fastio();
        memset(used,false,n);
        for(i=0;i<pairs;i++)
           {
              e[i].x=fastio();
              e[i].y=fastio();
            //cin>>e[i].x>>e[i].y;
           }
        int c=0;
        for(i=pairs-1;i>=0;i--)
        {
            if(!used[e[i].x]&&!used[e[i].y])
            {
                used[e[i].x]=1;
                used[e[i].y]=1;
                a[c]=i;
                c++;
            }
 
        }
        for(i=c-1;i>=0;i--)
        printf("%d ",a[i]);
        printf("\n");
    }
    return 0;
}
