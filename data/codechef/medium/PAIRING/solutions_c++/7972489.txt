#include<iostream>
using namespace std;
int main()
{
	int t;
	long int k,m,n,i,a[10000],b[10000],arr[1000],ar[1000];
	cin>>t;
	while(t--)
	{
		cin>>m;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			cin>>b[i];
		}
		for(i=0;i<m;i++)
		arr[i]=0;
		k=0;
		for(i=n-1;i>=0;i--)
		{
			if(arr[a[i]]==0&&arr[b[i]]==0)
			{
				arr[a[i]]=1;
				arr[b[i]]=1;
		//		cout<<i<<" ";
				ar[k]=i;
				//cout<<ar[k]<<endl;
				k++;
			}
		}
		for(i=k-1;i>=0;i--)
		cout<<ar[i]<<" ";
		cout<<"\n";
	}
	return 0;
}
