#include <stdio.h>
#define MAX 10000
 
inline int readInt(){
	int number = 0;
	char c = getchar_unlocked();
	while(c<'0')
		c = getchar_unlocked();
	while(c>='0' && c<='9'){
		number = (number<<3)+(number<<1)+c-'0';
		c = getchar_unlocked();
	}
	return number;
}
 
int main(){
    short int t,n,m,i,j,a[MAX][2],pos[1000],ind[500];
    t = readInt();
    
    while(t--){
         n = readInt();
         m = readInt();
         
         for(i = 0;i<n;i++)
          pos[i] = 0;
         
         for(i = 0;i<m;i++){
               a[i][0] = readInt();
               a[i][1] = readInt();
         }
         
         j = 0;
         for(i = m-1;i>=0;i--)
           if(pos[a[i][0]] == 0 && pos[a[i][1]] == 0){
            pos[a[i][0]] = pos[a[i][1]] = 1;
            ind[j++] = i;
           }
         
         for(i = j-1;i>0;i--)
          printf("%d ",ind[i]);
          
         printf("%d\n",ind[i]);
    }
    
    return 0;
}
