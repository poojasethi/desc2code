#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int ip[10000][2],ans[1000];
short int pairs[1000];
inline void fastread(int *a)
{
register char c=0;
while (c<33) c=getchar_unlocked();
*a=0;
while (c>33)
{
*a=*a*10+c-'0';
c=getchar_unlocked();
}
}
inline void fastwrite(int a)
{
char snum[20];
int i=0;
do
{
snum[i++]=a%10+48;
a=a/10;
}while(a!=0);
i=i-1;
while(i>=0)
putchar_unlocked(snum[i--]);
putchar_unlocked('\n');
}
int main()
{
	int t,n,m,cnt;
	cin >> t;
	while(t--)
	{
		fastread(&n);
		fastread(&m);
		cnt = 0;
		memset(pairs,0,sizeof(pairs));
		for (int i = 0 ; i < m ; i ++)
			cin >> ip[i][0] >> ip[i][1];
		for(int i = m -1 ; i >= 0 ; i--)
		{
			if(pairs[ip[i][0]] == 0 && pairs[ip[i][1]] == 0)
			{
				pairs[ip[i][0]] = 1;
				pairs[ip[i][1]] = 1;
				ans[cnt++]=i;
			}
		}
		for (int i = cnt - 1 ; i >=0 ; i--)
			cout << ans[i] <<" ";
		cout << endl;
	}
	return 0;
}
