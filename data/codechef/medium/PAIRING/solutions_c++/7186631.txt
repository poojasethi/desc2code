#include<bits/stdc++.h>
#define ll long long
using namespace std;

struct par
{
	ll a,b;
};

int main()
{	
	ll t,n,m,i;par *p;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld",&n,&m);
		p=new par[m];
		for(i=0;i<m;i++)
		{
			scanf("%lld %lld",&p[i].a,&p[i].b);
		}
		map<ll,int> ma;map<ll,int> chk;
		for(i=m-1;i>=0;i--)
		{
			if(chk[p[i].a]==0 && chk[p[i].b]==0)
			{
				ma[i]=1;
				chk[p[i].a]=1;
				chk[p[i].b]=1;
			}
		}
		for(i=0;i<m;i++)
		if(ma[i]==1)
		printf("%lld ",i);
		printf("\n");
	}
	return 0;
}
