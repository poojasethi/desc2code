typedef pair< int , int > pii;

#define fr first

/**************************Code****************************/

int mark[1024], x[10010], y[10010];

int main()
{
	int t, n, m;
	scanf( "%d", &t );
	while( t -- )
	{
		string sp = "";
		memset( mark, 0, sizeof mark );
		stack < int > ans;
		scanf( "%d%d", &n, &m );
		for( int i = 0 ; i < m ; i ++ )
			cin >> x[i] >> y[i];
		for( int i = m-1 ; i >= 0 ; i -- )
			if( !mark[ x[i] ] && !mark[ y[i] ] )
				mark[ x[i] ] = mark[ y[i] ] = 1, ans.push( i );
		while( !ans.empty() )
			cout << sp << ans.top(), sp = " ", ans.pop();
		cout << endl;
	}
	return 0;
}
