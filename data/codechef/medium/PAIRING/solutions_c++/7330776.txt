#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,m,p[10000][2],i;
    bool take[10000];
    cin>>t;
    while(t--)
    {
        memset(take,false,sizeof(take));
        cin>>n>>m;
        for(i=0;i<m;i++)
            cin>>p[i][0]>>p[i][1];
        for(i=m-1;i>=0;i--)
            if(!take[p[i][0]]&&!take[p[i][1]])
                take[p[i][0]]=take[p[i][1]]=true;
            else
                p[i][0]=-1;
        for(i=0;i<m;i++)
        {
            if(p[i][0]!=-1)
                cout<<i<<" ";
        }
        cout<<endl;
    }
}
