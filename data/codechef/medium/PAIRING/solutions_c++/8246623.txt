#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,t,n,m,a,b;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        vector<int> v(n);
        set<int> res;
        vector<pair<int,int> > p(m);
        for(i=0;i<m;i++)
        {
            scanf("%d%d",&a,&b);
            p[i].first=a;
            p[i].second=b;
        }
        for(i=m-1;i>=0;i--)
        {
            if((v[p[i].second]==0)&&(v[p[i].first]==0))
            {
                v[p[i].second]=1;
                v[p[i].first]=1;
                res.insert(i);
            }
        }
        set<int>::iterator it;
        for(it=res.begin();it!=res.end();it++)
            printf("%d ",*it);
        printf("\n");
    }
    return 0;
}
