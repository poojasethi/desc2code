#include<bits/stdc++.h>

using namespace std;

#define mp  make_pair
#define pb push_back

int main(void)
{
     int t; scanf("%d",&t);
     while(t--)
     {
        int n,m; scanf("%d %d",&n,&m);
        vector<pair<int,int> > uv;
        for(int i=0;i<m;i++)
        {
            int u,v; scanf("%d %d",&u,&v);
            uv.pb(mp(u,v));
        }
        
        bool arr[10001]={false};
        vector<int> ans;
        for(int i=m-1;i>=0;i--)
        {
            if( arr[uv[i].first] ==false &&
                arr[uv[i].second]== false)
            {
                arr[uv[i].first]=true;
                arr[uv[i].second]=true;
                ans.pb(i);
            }
        }
        
        for(int i=ans.size()-1;i>=0;i--)
        {
           printf("%d ",ans[i]);
        }
        printf("\n");
        
   }
   return 0;
}
