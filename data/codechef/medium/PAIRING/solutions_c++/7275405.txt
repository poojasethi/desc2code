#include<iostream>
using namespace std;

int main(){
    int w;
    cin>>w;
    while(w--){
        int n,m,i,j;
        cin>>n>>m;
        int x[m],y[m];
        for(i=0;i<m;i++)
            cin>>x[i]>>y[i];
        int hasOccured[n],ifPrint[m];
        for(i=0;i<n;i++)
            hasOccured[i]=0;

         for(i=0;i<m;i++)
            ifPrint[i]=0;

        for(i=m-1;i>=0;i--){
            if(hasOccured[x[i]]==0&&hasOccured[y[i]]==0){
                hasOccured[x[i]]=1;
                hasOccured[y[i]]=1;
                ifPrint[i]=1;
                //cout<<"x[i] "<<x[i]<<" y[i] "<<y[i]<<" hasOccured[x[i]] "<<hasOccured[x[i]]<<" hasOccurred[y[i]] "<<hasOccured[y[i]]<<endl;
            }
            //else continue;
        }
        for(i=0;i<m;i++){
            if(ifPrint[i]==1)
                cout<<i<<' ';
        }
        cout<<endl;
    }
        return 0;
}
