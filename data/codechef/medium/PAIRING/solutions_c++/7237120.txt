#include<stdio.h>
#include<math.h>
#include<algorithm>

using namespace std;

int main()
{
    int tc,m,n;

{
scanf("%d %d",&n,&m);

long int arr1[m];
long int arr2[m];
long int arr3[n];
long int arr[m];
for(int i=0;i<m;++i)
scanf("%d %d",&arr1[i],&arr2[i]);

    arr3[i]=0;
    int c=0;

 return 0;




}
