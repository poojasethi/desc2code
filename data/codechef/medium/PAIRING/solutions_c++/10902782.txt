#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n,m;
		scanf("%d%d",&n,&m);
		int a[n+1];
		memset(a,0,sizeof(a));
		int b[m];
		int c[m];
		for (int i = 0;i < m; i++)
			scanf("%d%d",&b[i],&c[i]);
		vector<int> ans;
		for (int i = m-1; i >= 0; i--)
		{
			if (a[b[i]] == 0 && a[c[i]] == 0)
			{
				ans.push_back(i);
				a[b[i]] = 1;
				a[c[i]] = 1;
			}
		}
		sort(ans.begin(),ans.end());
		int len = ans.size();
		for (int i = 0; i < len; i++)
			cout << ans[i] << endl;
		cout << endl;
	}
	return 0;
}
