#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int h,x,y,k,j,i,n,m;
		scanf("%d %d",&n,&m);
		int g[m];
		int a[m];
		int b[m];
		int c[2][n];
		for(i=0;i<n;i++)
		{
			c[0][i]=i;
			c[1][i]=1;
		}
		for(i=0;i<m;i++)
		{
			scanf("%d %d",&a[i],&b[i]);
		}
	h=0;
	for(i=m-1;i>=0;i--)
		{
			x=0;
			y=0;
				for(j=0;j<n;j++)
				{
					if(a[i]==c[0][j] && c[1][j]!=0)
					{
						x=1;
						break;
					}
				}
				for(k=0;k<n;k++)
				{
					if(b[i]==c[0][k] && c[1][k]!=0)
					{
						y=1;
						break;
					}
				}
				if(x==1 && y==1)
				{
					c[1][j]=0;
					c[1][k]=0;
					g[h]=i;
					h++;
				}
		}
		for(i=h-1;i>=0;i--)
		{
			printf("%d ",g[i]);
		}
		if(t>=1)
		{
			printf("\n");
		}
	}
	return 0;
}
