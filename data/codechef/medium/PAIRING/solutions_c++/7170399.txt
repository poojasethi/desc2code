#include<iostream>
using namespace std;
#include<bits/stdc++.h>
int main()
 {
 	int t;
 	 cin>>t;
 	 while(t--)
 	  {
 	  		int n,m;
 	  		 cin>>n>>m;
 	  		 int has[n+10];
 	  		  for(int i=0;i<=n;i++)has[i]=0;
 	  		 
 	        vector<pair<int,int> > v;
 	        for(int i=0;i<m;i++)
 	         {
 	         	int a,b;
 	         	 cin>>a>>b;
 	         	 v.push_back(make_pair(a,b));
 	         	 
 	         }
 	        int ans[10000];
 	        int p=0;
 	        for(int i=m-1;i>=0;i--)
 	         {
 	         	
 	         	int a=v[i].first;
 	         	int b=v[i].second;
 	         	if(has[a]==0 && has[b]==0)
 	         	 {
 	         	 	has[a]=1;
 	         	 	has[b]=1;
 	         	 	ans[p++]=i;
 	         	 	
 	         	 }
 	         }
 	          for(int i=p-1;i>=0;i--)
 	           {
 	           	cout<<ans[i]<<" ";
 	           }
 	            cout<<endl;
 	  }
 
 }
