#include <cstdio>
using namespace std;

int main() {
	int t;
	int n,m;
	scanf("%d",&t);
	while(t--)
	{
	    scanf("%d%d",&n,&m);
	    int one[m],two[m],visited[10001]={0},arr[m],count=0;
	    for(int i=0;i<m;i++)
	        scanf("%d%d",&one[i],&two[i]);
	    for(int i=(m-1);i>=0;i--)
	        if(!visited[two[i]] && !visited[one[i]])
	        {
	            visited[two[i]]=visited[one[i]]=1;
	            arr[count++]=i;
	        }
	    for(int i=count-1;i>=0;i--)
	        printf("%d ",arr[i]);
	   printf("\n");
	}
	return 0;
}
