#include<iostream>
#include<stdio.h>
#include<utility>
#include<vector>
using namespace std;
int main()
{
    int t,n,m,a,b;
   cin>>t;
   while(t--)
   {

       cin>>n>>m;
       pair<int ,int> p[m];
       int pos[n];
       vector<int >ind;
       vector<int >::iterator it=ind.begin();
       for(int i=0;i<m;i++)
         {
            pos[i]=0;
         }
       for(int i=0;i<m;i++)
       {
           cin>>a>>b;
           p[i]=make_pair(a,b);
       }
     int j=0;
 for(int i=m-1;i>=0;i--)
 {
     if(pos[p[i].first]==0 && pos[p[i].second]==0)
        {
            pos[p[i].first] = pos[p[i].second]=1;
            ind.push_back(i);
        }
 }

while(!ind.empty())
{
    cout<<ind.back()<<" ";
    ind.pop_back();
}
 cout<<endl;
}
return 0;
}
