#include<stdio.h>


int main()
{ 
  int t;
  scanf("%d",&t);
  
  while(t--)
  { int n,m;
     scanf("%d%d",&n,&m);
     int i,k;
    int u[m],v[m];
    for(i=0;i<m;i++)
     scanf("%d%d",&u[i],&v[i]);
     
    int c[n],f[m];
    for(i=0;i<n;i++)
     c[i]=0;
    
    for(i=0;i<m;i++)
     f[i]=0;
     
    for(i=m-1;i>=0;i--)
    { if(c[u[i]]==0&&c[v[i]]==0)
       { f[i]=1;
         c[u[i]]=1;
         c[v[i]]=1;
       }
    }
    for(i=0;i<m;i++)
    if(f[i])
     printf("%d ",i);
    printf("\n");
  }
  return 0;
}
