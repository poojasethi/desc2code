#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string.h>
using namespace std;
struct employee{
int x;
int y;
};
bool used[1001];
struct employee e[10001];
int main()
{
    int t,i,n,pairs,a[1001];
    cin>>t;
    while(t--)
    {
        cin>>n>>pairs;
        memset(used,false,n);
        for(i=0;i<pairs;i++)
        {
            cin>>e[i].x>>e[i].y;

        }
        int c=0;
        for(i=pairs-1;i>=0;i--)
        {
            if(!used[e[i].x]&&!used[e[i].y])
            {
                used[e[i].x]=1;
                used[e[i].y]=1;
                a[c]=i;
                c++;
            }

        }
        for(i=c-1;i>=0;i--)
        printf("%d ",a[i]);
        printf("\n");
    }
    return 0;
}
