#include <iostream>
#include <map>

using namespace std;
typedef long long  LL;
LL a[10001] = {0}, b[10001] = {0};
LL ans[10001] = {0};
int main() {
    LL t=0;
    cin>>t;
    while (t--) {
        LL n=0,m=0,i=0,k=0;
        map<LL,int> ma;

        cin>>n>>m;
        for(i=0;i<m;i++) {
            cin>>a[i]>>b[i];
        }
        
        map<LL,int>::iterator iter1, iter2;
        for(i=m-1;i>=0;i--) {
            iter1 = ma.find(a[i]);
            iter2 = ma.find(b[i]);
            if(iter1 == ma.end() && iter2 == ma.end()){
                ans[k++] = i;
                ma[a[i]] = 1;
                ma[b[i]] = 1;
            }
            
        }
        
        for(i=k-1;i>=0;i--) {
            cout<<ans[i]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
