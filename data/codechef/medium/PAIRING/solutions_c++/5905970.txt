#include<bits/stdc++.h>
using namespace std;
int fastread()
{int t=0;
char ch=getchar_unlocked();
while(ch<33)
ch=getchar_unlocked();
while(ch>33)
{t=t*10+ch-'0';
ch=getchar_unlocked();
}
return t;
}
int main()
{
    int a[2][10001],t,n,i,j,k;
    int b[1000];
    t=fastread();
    while(t--)
    {
        int c[1001]={0};
        n=fastread();
        k=fastread();
        for(i=0;i<k;i++)
        {
            a[0][i]=fastread();
            a[1][i]=fastread();
        }
        for(i=k-1,j=0;i>=0;i--)
            if(c[a[0][i]]==0&&c[a[1][i]]==0)
                {b[j++]=i;
                    c[a[0][i]]=1;
                    c[a[1][i]]=1;
                }
            while(j--)
                printf("%d ",b[j]);
        printf("\n");
    }
    return 0;
}
