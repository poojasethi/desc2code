#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,m;
	int pr[10000][2];
	int dp[1000];
	int chosen[10000];
	scanf("%d",&t);
	while(t--){
		int k=0;
		scanf("%d %d",&n,&m);
		for(int i=0;i<n;i++){
			dp[i]=0;
		}
		for(int i=0;i<m;i++){
			scanf("%d %d",&pr[i][0],&pr[i][1]);
		}
		for(int i=m-1;i>=0;i--){
			if(!dp[pr[i][0]] && !dp[pr[i][1]]){
				chosen[k++]=i;
				dp[pr[i][0]]=1;
				dp[pr[i][1]]=1;
			}
		}
		for(int i=k-1;i>=0;i--){
			printf("%d ",chosen[i]);
		}
		printf("\n");
	}
	return 0;
}
