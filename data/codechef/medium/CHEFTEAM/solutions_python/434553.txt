#!/user/bin/python

def  gcd(a,b):
	if a % b == 0 :
		return b
	else :
	    return gcd(b,a%b)


def  ncr(n,r):
	if n < r :
		return 0
	tr = n-r
	if r > tr :
		r = tr
        c = 1;
        d = 1;
	
	for num in range(1,r+1):
		p = n - num + 1
		q = num
		gd = gcd(p,q)
		p = p / gd
		q = q / gd
		gd = gcd(p,d)
		p = p / gd
		d = d / gd
		gd = gcd(q,c)
		q = q / gd
		c = c / gd
		c = c * p
		d = d * q
	return c / d
	  

x = int(raw_input())
for i in range(0,x):
	s = raw_input()
	n,r = [int(j) for j in s.split(' ')]
	print ncr(n,r)
