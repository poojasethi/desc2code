import decimal
def ncr(n,k):
  result = decimal.Decimal("1.0")
  for i in range (0,k):
	a = decimal.Decimal(n - i)
	b = decimal.Decimal(i+1)
	result = result*decimal.Decimal(a/b)	
  return result + decimal.Decimal("0.5")

t = int(raw_input())
for j in range(t):
  temp = raw_input().split(' ')
  n = int(temp[0])  
  k = int(temp[1])  
  if(n<k):
    print '0'
    continue
  if(k > n-k):
    k = n-k
  print int(ncr(n,k))
