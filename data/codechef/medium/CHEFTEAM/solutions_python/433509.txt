t=raw_input()
for j in range(int(t)):
    s=raw_input()
    q=s.partition(" ")
    n=int(q[0])
    k=int(q[2])
    if(n<k):
        print 0;
        continue
    num=1
    den=1
    if(k<n-k):
        k=n-k
    for i in range(k+1,n+1):
        num=num*i
    for l in range(1,n-k+1):
        den=den*l
    print (num/den)
