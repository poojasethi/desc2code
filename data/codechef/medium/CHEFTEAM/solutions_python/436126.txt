def NchooseK(N,K):
    if K > N:
        return 0
    K = min(N-K,K)

    ans = 1;
    for i in range(1,K+1):
        ans *= (N-i+1)
        ans /= i
    return int(ans)

t = int(raw_input())
for i in range(0,t):
    N,K = map(int,raw_input().split())
    print(NchooseK(N,K))
