allnum = int(tmp)
for h in range(1,allnum+1):
	tmp = raw_input()
	tmp = tmp.split(' ')
	n = int(tmp[0])
	k = int(tmp[1])
	if n < k:
		print 0
		continue
	c1 = n - k
	c2 = k
	if c1 < c2:
		cc = c1
		c1 = c2
		c2 = cc
	ans = 1
	for i in range(c1+1,n+1):
		ans = ans * i
	for i in range(1,c2+1):
		ans = ans / i
	print ans
