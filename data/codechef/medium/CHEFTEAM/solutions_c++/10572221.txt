#include <iostream>
#include <cmath>
#define ll unsigned long long
using namespace std;
ll gcd(ll a, ll b)
{
	if (b == 0)
		return a;
	return gcd(b, a % b);
}
ll ncr(ll n, ll k)
{
	if (n < k)
		return 0;
	ll num = 1, den = 1, nm, dn, g, res = 1;
	if (k > (n / 2))
		k = n - k;
	for (int i = 1; i <= k; i++)
	{
		g = gcd(i,res);
		res /= g;
		res *= (n - i + 1) / (i / g);
	}
	return res;
}
