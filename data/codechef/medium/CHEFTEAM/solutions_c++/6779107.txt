# include <iostream>
using namespace std;

unsigned long long gcd(unsigned long long a,unsigned long long b){
    if(b==0) return a;
    return gcd(b,a%b);
}

int main(){
	int t;
	unsigned long long num,m,n,k,tn,td,r;
	cin>>t;
	while(t-->0){
		cin>>n;
		cin>>k;
		if(k>n) {
			cout<<"0\n";continue;
		}
		k=(n-k>k)?k:n-k;
		n=n-k+1;num=1;
		for(r=1;r<=k;r++,n++){
			m=gcd(n,r);
			tn=n/m;
			td=r/m;
			num/=td;
			num*=tn;
		}
		cout<<num<<"\n";
	}
	return 0;
}
