#include<iostream>
using namespace std;
#define ULL unsigned long long int
ULL gcd(ULL a,ULL b)
{
    if(a>=b)
    {
         if(a%b==0) return b;
         else return gcd(b,a%b);
    }
    else
    {
        if(b%a==0) return b;
        else return gcd(a,b%a);
    }
}
ULL PNC(ULL n, ULL k)
{
    k = min(n-k,k);
    ULL ans=1;
    for(ULL i=1;i<=k;i++)
    {
        ULL D =gcd(ans,i);
        ans/=D;
        ans*=(n-i+1) /(i/D);
    }
    return ans;
}
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
      ULL N,K;
      cin>>N>>K;
      if(K>N) cout<<0<<endl;
      else cout<<PNC(N,K)<<endl;
  }
}
