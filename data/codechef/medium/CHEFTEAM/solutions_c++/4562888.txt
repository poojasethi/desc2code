# include <cstdio>

unsigned long long gcd(unsigned long long a,unsigned long long b)
{
	if(b==0)return a;
	if(b==1)return 1;
	return gcd(b,a%b);
}
int main()
{
	int T;
	scanf("%d",&T);

	for(int t=0;t<T;t++)
	{
		unsigned long long N,K;
		scanf("%Lu%Lu",&N,&K);

		if(K>N)
		{
			printf("0\n");
			continue;
		}
		else if(K==0||K==N)
		{
			printf("1\n");
			continue;
		}
		if(K>N/2)K=N-K;

		unsigned long long ret=1;
		for(unsigned long long i=0;i<K;i++)
		{
			unsigned long long a=i+1,b=gcd(ret,a);
			ret/=b;
			a/=b;
			ret*=(N-i)/a;
		}
		printf("%Lu\n",ret);
	}
	return 0;
}
