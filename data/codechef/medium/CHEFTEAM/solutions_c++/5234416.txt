#include<iostream>
#include<cstdio>
#include<vector>
#define ull unsigned long long

using namespace std;

ull gcd(ull a, ull b)
{
    if(b % a == 0)return a;

    else return gcd(b, a%b);
}


//vector<ull> nck[1000001];

ull nck(ull n, ull k)
{
    ull ans = 1;
    for(ull i = 1; i <= k; i++)
    {
       // ans = (ans * (n-i)) / i;
       ull g1 = gcd(ans, i), x = i;

       ans /= (g1);
       x /= g1;
       ull g2 = gcd(n-i+1, x);
       ull y = (n-i+1)/g2;
       x /= g2;
       ans = ans * y;
      // cout << ans << endl;
    }
    return ans;

}



int main()
{
    //cout << gcd(24, 18) << endl;
    int t;
    cin >> t;
    while(t--)
    {
        ull n, k;
        cin >> n >> k;
        ull x = (n - k);
        if(n < k)printf("0\n");
        else if(x > k)
        {
            cout << nck(n, k) << endl;
        }
        else
        cout << nck(n, x) << endl;;


    }
