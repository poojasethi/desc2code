#include <cstdio>
using namespace std;

unsigned long long gcd(unsigned long long a, unsigned long long b)
{
	if(!a)
	{
		return b;
	}
	else
	{
		return gcd(b % a, a);
	}
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		unsigned long long n, k;
		scanf("%llu %llu", &n, &k);
		if(k > n)
		{
			printf("0\n");
			continue;
		}
		else if(k == n || !k)
		{
			printf("1\n");
			continue;
		}
		if(k > n/2)
		{
			k = n-k;
		}
		unsigned long long arr[k], out = 1;
		for(unsigned long long i = 0; i < k; i++)
		{
			arr[i] = n - i;
		}
		for(unsigned long long i = 2; i <= k; i++)
		{
			unsigned long long temp = i;
			for(unsigned long long j = 0; j < k; j++)
			{
				unsigned long long temp2 = gcd(temp, arr[j]);
				arr[j] /= temp2;
				temp /= temp2;
				if(temp == 1)
				{
					break;
				}
			}
		}
		for(unsigned long long i = 0; i < k; i++)
		{
			out *= arr[i];
		}
		printf("%llu\n", out);
	}
    return 0;
}
