#include <iostream>
using namespace std;

unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
	unsigned long long int c;
	while(b>0) c=a%b, a=b, b=c;
	return a;
}

int main()
{
    unsigned long long int T,N,K;
    cin>>T;
    while(T--)
    {
        cin>>N>>K;
        if(K > (N/2))
        {
            K = N-K;
        }
        unsigned long long int ans = 1;
        if(N<K)
        {
            cout<<0<<endl;
            continue;
        }
        else if(N==K)
        {
            cout<<1<<endl;
            continue;
        }
        else
        {
            for(int i=1;i<=K;i++)
            {
                unsigned long long int temp = gcd(ans,i);
                ans/= temp;
                ans*= ((N-i+1)/(i/temp));
            }
            cout<<ans<<endl;
        }
    }
    return 0;
}
