ull gcd(ull a,ull b)
{
    if(a>=b)
    {
        if(a%b==0) return b;
        else return gcd(b,a%b);
    }
    else
    {
        if(b%a==0) return a;
        else return gcd(a,b%a);
    }
}


ull comb( ull n, ull k)
{
	ull i;
	if (k>n)
		return 0;
	k=min(n-k,k);
	ull nk=1,g;
	
	for ( i=1;i<=k;++i)
	{
		g=gcd(nk, i);
		nk/=g;
		nk *= (n-i+1)/(i/g);
	}
	return nk;
}

int main()
{
	int num;
	ull n,k,ans;
	cin>>num;
	while(num--)
	{
		cin>>n>>k;
		ans=comb(n,k);
		cout<<ans<<endl;
	}
}
