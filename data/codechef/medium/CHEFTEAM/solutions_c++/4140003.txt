#include<cstdio>
#include<cstdlib>
#include<iostream>
unsigned long long  gcd(unsigned long long a, unsigned long long b)
{
unsigned long long x;
if(b>a)
return gcd(b,a); 
while(b)
{
x = a%b;
a = b;
b = x;
}
return a;
}
 int main()
{
unsigned long long int T, i,j,k,l,g;
scanf("%llu",&T);
while(T--)
{
scanf("%llu %llu",&j,&k);
if(k>j)
printf("0\n");
else
{
l=1;
//m=1;
if(k>j/2)
k=j-k;
for(i=1;i<=k;i++)
{
g=gcd(l,i);
l/=g;

}

}
}
return 0;
}
