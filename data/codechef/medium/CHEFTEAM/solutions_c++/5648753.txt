#include<iostream>

using namespace std;

unsigned long long int gcd(unsigned long long int a,unsigned long long int b){
	int temp=0;
	while(a%b>0){
		temp=b;
		b=a%b;
		a=temp;
	}
return b;
}



unsigned long long int ncr(unsigned long long int n,unsigned long long int k){
	if(n<k) return 0;
	if(k>n-k) k=n-k;
	unsigned long long int p=1;
	for(unsigned long long int i=1;i<=k;i++)
	{
		unsigned long long int d=gcd(p,i);
		p/=d;
		p*=(n-i+1)/(i/d);
	}
	return p;
}


int main(){
	unsigned long long int t;
	cin>>t;
	unsigned long long int n,r;
	
	while(t--){
		cin>>n>>r;
		
		cout<<ncr(n,r)<<"\n";
	}
}
