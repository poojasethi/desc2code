#include<cstdio>
unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{unsigned long long int t=0;
while(b>0)
{t=a%b;
a=b;
b=t;
}          
return a;    
}
 
int main()
{
	int t;
	unsigned long long int n,k,i;
	unsigned long long int d,c,a,b,tt;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%llu %llu",&n,&k);
		
		if(k>n){printf("0\n");
		continue;
		}
		
		if(k>n-k)k=n-k;
		
		c=1;
		for(i=1;i<=k;i++)
		{
	
		
		d=gcd(c,i);
			
			
			c=c/d;
			c*=(n-i+1)/(i/d);
		
		}
			printf("%llu\n",c);
	}
 
	return 0;
}
