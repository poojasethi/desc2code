#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

typedef unsigned long long ULL;
ULL gcd(ULL a,ULL b)
{
	if(b==0) return a;
	else return gcd(b,a%b);
}
ULL comb(ULL n,ULL k)
{
	if(k>n) return 0;
	k=min(k,n-k);
	ULL i,ans=1,d;
	for(i=1;i<=k;i++)
	{
	 d=gcd(i,ans);
	 ans/=d;
	 ans*=(n-i+1)/(i/d);
	}
	return ans;
	
}
main()
{   std:ios_base::sync_with_stdio(false);
    int t;
    ULL n,k;
    cin>>t;
    while(t--)
    {
     cin>>n>>k;
	 cout<<comb(n,k)<<"\n";
    }
	return 0;
}
