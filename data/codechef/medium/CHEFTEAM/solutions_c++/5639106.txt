#include<bits/stdc++.h>
using namespace std;

typedef unsigned long long int ll;

int main()
{
    ios_base::sync_with_stdio(false);
    ll t,i,n,k,c,d;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        k = min(k, n-k);
        if(k==0)
        {
            cout<<1<<endl;
            continue;
        }
        else if(k==1)
        {
            cout<<n<<endl;
            continue;
        }
        else if(k>n)
        {
            cout<<0<<endl;
            continue;
        }
        c = 1;
        for(i=1;i<=k;i++)
        {
            d = __gcd(c,i);
            c /= d;
            c *= (n-i+1)/(i/d);
        }
        cout<<c<<endl;
    }
    return 0;
}
