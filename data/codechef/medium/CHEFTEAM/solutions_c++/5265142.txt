#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;

unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
    if(b==0) return a;
    if(b>a) return gcd(b,a);
    return gcd(b,a%b);
}

int main()
{
    int t;
    cin>>t;
    while(t--){
        unsigned long long int n,k,c=1,j,d;
        cin>>n>>k;
        if(n<k) c=0;
        else{
        if(k>n-k) k=n-k;
        for(j=1;j<=k;j++){
            d=gcd(c,j);
            c/=d;
            c*=(n-j+1)/(j/d);
        }
        }
        cout<<c<<endl;
    }
    return 0;
}
