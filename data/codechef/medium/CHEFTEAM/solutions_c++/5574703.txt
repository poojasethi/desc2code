#include<stdio.h>
#define ll unsigned long long

unsigned long long gcd(unsigned long long a,unsigned long long b)
{
	unsigned long long c;
	while(b>0)
		c=a%b,a=b,b=c;
	return a;
}
 
 
unsigned long long nck(unsigned long long n,unsigned long long k)
{
	unsigned long long result=1;
	unsigned long long i;
	
	if(n<k) return 0;
	if(k>n-k) k=n-k;
 
	
	for(i=1;i<=k;i++)
	{
		unsigned long long l=gcd(result,i);
		result/=l;
		result*=(n-i+1)/(i/l);
	}
	return result;
}


int main()
{
	ll test;
	scanf("%llu",&test);
	while(test--)
	{
		ll n,r;
		scanf("%llu%llu",&n,&r);
		printf("%llu\n",nck(n,r));
	}
	return 0;
}
