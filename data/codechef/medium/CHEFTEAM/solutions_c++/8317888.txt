	#include<iostream>
	#include<stdio.h>
	#include<algorithm>
	using namespace std;
	unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
	{
		if(b>a)
		{
			unsigned long long int temp=a;
			a=b;
			b=temp;		
		}	

		if(b==0)
			return a;
		else
			return gcd(b,a%b);
	}	
	int main()
	{
		int t;
		scanf("%d",&t);
		while(t>0)
		{
			unsigned long long int  n,k,ans=1,temp;
			scanf("%llu%llu",&n,&k);	
			if(k<=n)
			{
				if(k>n-k)
					k=n-k;
				for(int i=1;i<=k;++i)
				{
					temp=gcd(ans,i);
					ans=ans/temp;
					ans=ans*((n+1-i)/(i/temp));
				
				}
			}
			else
				ans=0;	

			printf("%llu\n",ans);
			t--;
		}	
	}
