#include <cstdio>
unsigned long long int gcd(unsigned long long int ,unsigned long long );
inline unsigned long long int Combinations( unsigned long long int n,  unsigned long long int k)
{unsigned long long int d,p;
     if (k > n)
         return 0;
     unsigned long long r = 1;
     for ( d = 1; d <= k; ++d)
     {
         p = gcd(r,d);
         r /= p;
         r*=(n-d+1)/(d/p);
     }
     return r;
}

unsigned long long int gcd(unsigned long long int a,unsigned long long b)
{
	unsigned long long int c;
	while(b>0) c=a%b, a=b, b=c;
	return a;
}
int main() {
    unsigned long long int t,n,k,x,y;
    unsigned long long m;
    scanf("%llu",&t);
    while(t--)
    {
        scanf("%llu%llu",&n,&k);
        if(n==0)
        {
            if(k!=0)
                printf("0\n");
            else
                printf("1\n");
            continue;
        }
        if(k<=n)
        {

            x=n;
            y=k;
            if(y<x-y)
                m=Combinations(x,y);
            else
                m=Combinations(x,x-y);
            printf("%llu\n",m);
        }
        else
            printf("0\n");
    }
    return 0;
}
