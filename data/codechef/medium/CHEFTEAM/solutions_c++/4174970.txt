#include<stdio.h>
unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main()
{
 unsigned long long int n,k,ans,div,i,j;
 int t;
 scanf("%d",&t);
 while(t--)
 {
  ans=1;
  scanf("%llu %llu",&n,&k);
  if(k>n)
  {
	printf("0\n");
  }
  else
  {
	k=(k<n-k?k:n-k);
	for(i=1;i<=k;i++)
	{
	 div=gcd(ans,i);
	 ans/=div;
	 j=i/div;
	 ans*=((n-i+1)/j);
	}
	printf("%llu\n",ans);
  }
 }
 return 0;
}
