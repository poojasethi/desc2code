#include<stdio.h>
#include<iostream>
using namespace std;
unsigned long long gcd(unsigned long long a ,unsigned long long b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
main()
{
	int t;
	cin>>t;
	unsigned  long long n,k,ans,i,d,temp,temp1;
	while(t--)
	{
		cin>>n>>k;
		if(n<k)
		{
			cout<<"0"<<endl;
			continue;
		}
		ans=1;
		if(k>n-k)
			k=n-k;
		for(i=1;i<=k;++i)
		{
			if(ans>=i)
				d=gcd(ans,i);
			else
				d=gcd(i,ans);
			ans=ans/d;
			ans=ans*((n-i+1)/(i/d));
 
		}
               cout<<ans<<endl;
 
	}
 
}
