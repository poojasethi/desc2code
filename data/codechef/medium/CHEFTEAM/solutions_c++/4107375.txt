#include <iostream>
using namespace std;
#define ULL unsigned long long int
ULL gcd(ULL a,ULL b)
{
    if(a>=b)
    {
        if(a%b==0) return b;
        else return gcd(b,a%b);
    }
    else
    {
        if(b%a==0) return a;
        else return gcd(a,b%a);
    }
}
ULL pNc(ULL n,ULL k)
{
    if(k>n) return 0;

    k = min(k,n-k);
    ULL NK=1;
    for(ULL i=1;i<=k;i++)
    {
        ULL g = gcd(NK,i);
        NK/=g;
        NK*=(n-i+1)/(i/g);
    }
    return NK;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ULL N,K;
        cin>>N>>K;
        cout<<pNc(N,K)<<endl;
    }
    return 0;
}
