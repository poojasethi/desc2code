#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i;
    long long int rem;
    long int k;
    scanf("%d",&t);
    while(t--)
    {
        long long int bread=0;
        cin>>n>>k;
        int l[n];
        for(i=0;i<n;i++)
            scanf("%d",&l[i]);
        for(i=0;i<n;i++)
        {
            bread+=l[i];
            if(bread%k!=0)
                bread+=1;
        }
        rem=(bread%k);
        if((bread%k)!=0)
            bread+=(k-rem);
        cout<<(bread/k)<<endl;
    }
    return 0;
}
