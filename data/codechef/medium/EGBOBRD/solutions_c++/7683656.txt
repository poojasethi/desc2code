#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    long long int n,k;
    long long int a[100002];
    cin>>t;
    while(t--)
    {
        long long int j=0,p=0;
        cin>>n>>k;
        for(long long int i=0;i<n;i++)
        {
            cin>>a[i];
        }
         for(long long int i=0;i<n;i++)
        {
            if(a[i]<j)
            {
                j-=a[i];
                j--;
            }
            else
            {
                if(j>1)
                    a[i]=a[i]-j+1;
                p=p+a[i]/k;
                a[i]=a[i]%k;
                if(a[i]%k!=0)
                {
                    p=p+1;
                    j=k-a[i];
                }
                else
                    j=0;

            }
        }
        cout<<p<<"\n";
    }
    return 0;
}
