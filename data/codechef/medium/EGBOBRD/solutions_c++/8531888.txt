#include <stdio.h>

int main() {
	int T, N, i, x;
	long long K, left, ans;
	scanf("%d", &T);
	while (T--) {
		scanf("%d %lld", &N, &K);
		for (i = 0, ans = left = 0; i < N; i++) {
			scanf("%d", &x);
			if (left < x) {
				ans += (x - left) / K;
				left += ((x - left) / K) * K;
				if (left < x) {
					left += K;
					ans++;
				}
			}
			left -= x;
			if (left > 0) left--;
		}
		printf("%lld\n", ans);
	}
	return 0;
}
