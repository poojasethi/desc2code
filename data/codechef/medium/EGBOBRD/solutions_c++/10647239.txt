#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		long long k;
		scanf("%d %lld",&n,&k);
		long long x;
		long long packets = 1;
		long long curr = k;
		for(int i=0;i<n;i++)
		{
			scanf("%lld",&x);
			if(x>curr)
			{
				x-=curr;
				packets+=(x/k);
				curr = 0;
				if(x%k!=0)
				{
					packets++;
					curr = k - (x%k)-1;
				}
			}
			else
			{
				curr-=x;
				curr--;
			}
			//cout<<"Packets : "<<packets<<endl;
			curr = max(0LL,curr);
		}
		printf("%lld\n",packets);
	}
}
