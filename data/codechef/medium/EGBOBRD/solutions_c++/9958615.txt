#include <iostream>

using namespace std;


int main(){
	std::ios_base::sync_with_stdio(false);
	cin.tie(0);
	
	int t;
	cin >> t;
	
	long long int n,p,a,i;
	long long int pkgs, rem, rems,inc;
	
	while(t--){
		cin >> n >> p;
		pkgs = 1, rem = p;
		for(i=0;i<n;i++){
			cin >> a;
			rems = a-rem;
			inc = 0;
			if(rems > 0){
				inc = rems/p;
				if(rems%p){
					inc+= 1;
				}
			}
			rem = rem + inc*p - a;
			if(rem)
				rem--;
			pkgs+= inc;
		}
		cout << pkgs << '\n';
	}
	
	return 0;
}
