#include<bits/stdc++.h>
using namespace std;
int main()
{  int t;
   scanf("%d",&t);
   while(t--)
   { long long int n,k,a[100000],i,pack,left=0,sum=0; 
   	 scanf("%lld%lld",&n,&k);
   	 
   	 for(i=0;i<n;i++)
   	{  scanf("%lld",&a[i]);
   	   
	   if(left<a[i])
	   { pack=((a[i]-left)%k==0?(a[i]-left)/k:(a[i]-left)/k+1);
   	     left=pack*k+left-a[i];
       }
       else
        {  left=left-a[i];
           pack=0;
        }
   	    
		if(left>0)
   		--left;
   		
   	    sum+=pack;	
   	}
	   printf("%lld\n",sum);
   }
	return 0;
}
