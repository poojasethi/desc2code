#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	for(int w=0;w<t;w++){
	long long int n,k;
	cin>>n>>k;
	long long int ans=0,sum=0;
	for(int i=0;i<n;i++){
	int bread;
	cin>>bread;
	ans+=bread/k;
	sum+=bread%k;
	if(sum%k==0){
	ans+=sum/k;
	sum=0;
	}else{
	sum++;
	ans+=sum/k;
	sum=sum%k;
	}
	if(i==n-1&&sum!=0){
	ans++;
	}
	}
	cout<<ans<<"\n";
	}
	
	return 0;

}
