#include<iostream>
using namespace std;
int main()
{
	long long int sum=0;
	int t=0,n=0,k=0,left=0,curr=0,num=0,i=0;
	cin>>t;
	while(t--)
	{
		sum = 0;
		curr = 0;
		left = 0;
		num = 0;
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>curr;
			if(left > curr)
			{
				left = left - curr -1;
				num = 0;
			}
			else if(left == curr)
			{
				left = 0;
				num = 0; 
			}
			else
			{
				num = (curr-left)/k + ((curr-left)%k > 0);
				if(num < 0)
				num = 0;
				left = left + (num * k - curr);
				if(left > 0)
				left--; 
			}
			sum =sum + num;
		}
		cout<<sum<<endl;
		sum = 0; 
	}
}
