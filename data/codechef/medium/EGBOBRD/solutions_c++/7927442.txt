     #include<bits/stdc++.h>
    #define ll long long int
    #define sc(a) scanf("%lld",&a)
    #define pr(a) printf("%lld\n",a)
     
    ll a,b;
    using namespace std;
    #define gc getchar_unlocked
    inline int scan() {
    char c = gc();
    while(c<'0' || c>'9') c = gc();
    int ret = 0;
    while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
    }
    return ret;
    }
    void go()
    {
    ll i;
    ll bch=0,jaw=0;
    ll cur;
    for(i=0;i<a;i++)
    {
    int cur=scan();
    if(cur==bch)
    {bch=0;cur=0;}
    else if(cur<bch){bch=bch-cur-1;cur=0;if(bch<0)bch=0;}
    else if(cur>bch)
    {cur-=bch;bch=0;}
    if(cur==b&&cur!=0){jaw+=1;}
    else if(cur<b&&cur!=0){jaw+=1;bch+=b-cur-1;if(bch<0)bch=0;}
    else if(cur>b&&cur!=0){jaw=jaw+ceil(cur/(b*1.0));ll ss=ceil(cur/(b*1.0));bch+=ss*b-cur;bch-=1;if(bch<0)bch=0;}
    }
    pr(jaw);
    }
    int main()
    {
    ll t;
    sc(t);
    while(t--)
    { a=scan();
    b=scan();
    go();
    }
