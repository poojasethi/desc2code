#include<stdio.h>
long long int a[100005];
int main()
{
	long long int t, n, k, res, rem, i;
	scanf("%lld", &t);
	while (t--)
	{
		rem = res = 0;
		scanf("%lld%lld", &n, &k);
		for (i = 0; i < n; i++)
		{
			scanf("%lld", &a[i]);
		}
		for (i = 0; i < n; i++)
		{
			if (rem)
			{
				if (rem <= a[i])
				{
					a[i] -= rem;
				}
				else
				{
					rem -= a[i];
					rem--;
					continue;
				}

			}
			rem = a[i] % k;
			if (rem)
			{
				res = res+(a[i] / k) + 1;
				rem = k - rem;
				rem--;
			}
			else
			{
				res += (a[i] / k);
			}
		}
		printf("%lld\n", res);
	}
	getchar();
	getchar();
	return 0;
}
