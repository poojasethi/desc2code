#include<iostream>
#include<string.h>
#include<vector>
#include<stdio.h>
using namespace std;

#define gc getchar

int read()
{
	char c = gc();
	while(c<'0' || c>'9')
		c = gc();
	int ret = 0;
	while(c>='0' && c<='9')
	{
		ret = 10 * ret + c - '0';
		c = gc();
	}
    return ret;
}

int main(){
	
	int test;
	long long int n,k;
	
	scanf("%d",&test);
	
	while(test--){
	
	n = read();
	k = read();
	
	long long int a[n+1],i;
	
	for(i=0;i<n;i++)
		a[i] = read();
		
	long long int pckg=0;
	long long int s=0;
	
		for(i=0;i<n;i++){
			s+= a[i];
			
			// s denoting how many breads have been utilized
			
			if(s%k != 0){ // i.e. current req. is nt satisfied by the packet
				s++; 
			}
			
			if(s>=k){
				// we need a new packet
				int pck = s/k;
				s = s - (pck*k);
				pckg+=pck;
			}
		}
		if(s>0)
			pckg++;
	
		cout<<pckg<<endl;
	}
	
	return 0;
}
