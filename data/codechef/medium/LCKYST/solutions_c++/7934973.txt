#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
ios_base::sync_with_stdio(false);
int T;
cin >> T;
while(T--) {
int n;
cin >> n;
int m=n,cnt=0,c=0;
while(m%5==0){cnt++;m/=5;}
while(m%2==0){c++;m/=2;}
ll ans=(ll)n;
if(cnt>c) {
cnt-=c;
cnt=(cnt+1)/2;
ans=(ll)n;
while(cnt--)ans*=4ll;
}
printf("%lld\n",ans);
}
return 0;
}
