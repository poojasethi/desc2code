#include <iostream>
using namespace std;

int main() {
	// your code goes here
	long long n, x, y, i, a, b;
	cin>>n;
	for(i=0; i<n; i++){
		cin>>a;
		x=0;
		y=0;b=a;
		while(b%2==0){
			b/=2;x++;
		}
		while(b%5==0){
			b/=5;y++;
		}
		y-=x;
		while(y>0){
			a*=4;
			y-=2;
		}
		cout<<a<<endl;
	}
	return 0;
}
