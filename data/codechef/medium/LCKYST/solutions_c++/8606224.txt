#include<bits/stdc++.h>
using namespace std;
int main()
{  long long int val,n,c,i,mul,count,count2;
   scanf("%lld",&n);
   
   for(i=0;i<n;i++)
   {  scanf("%lld",&c);
      val=c;
     
	  count=0;	  
	  while(c%5==0)
   	  { ++count;
   	  	c/=5;
   	  }
   	  
	  c=val;
   	  count2=0;
	  while(c%2==0)
	  { ++count2;
	  	c/=2;
	  }
	  
	  if(count2<count)
	  { mul=count-count2;
	  	val=val*pow(4,(mul%2==0)?mul/2:mul/2+1);
	  }
	  
	  printf("%lld\n",val);
   }
	  return 0;
}
