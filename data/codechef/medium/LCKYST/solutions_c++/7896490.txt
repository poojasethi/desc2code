#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

typedef unsigned long long int ull;

ull find(ull n)
{
	int cnt = 0;
	ull k = n;

	while(k % 10 == 0)
	{
		k /= 10;
		cnt++;
	}

	if(k % 10 == 5)
	{
		return find((k<<2)*pow(10, cnt));
	}

	return n;
}

int main()
{
	int N;
	ull a;

	for(cin >> N; --N;)
	{
		cin >> a;
		cout << find(a) << " ";
	}

	cin >> a;
	cout << find(a) << endl;

	return 0;
}
