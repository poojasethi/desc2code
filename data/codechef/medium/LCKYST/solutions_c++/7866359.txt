#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
    long int n,i,c,a,j,no,dig;
    scanf("%d",&n);
    long long int l[n],k;
    for(i=0;i<n;i++)
        cin>>l[i];
    for(i=0;i<n;i++)
    {
        c=0;
        no=-1;
        k=l[i];
        dig=0;
        while(dig==0)
        {
            dig=k%10;
            k/=10;
            no++;
        }
        k = l[i];
        while(k%5==0)
        {
            c++;
            k/=5;
        }
        c-=no;
        if(c%2==0)
            a=c/2;
        else
            a=(c+1)/2;
        for(j=0;j<a;j++)
            l[i]*=4;
        cout<<l[i]<<endl;
    }
    return 0;
}
