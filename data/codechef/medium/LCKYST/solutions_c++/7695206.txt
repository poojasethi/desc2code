#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		long long  int k,c2=0,c5=0;
		cin>>k;
		long long int t=k;
		while((t%2)==0)
		{
			c2++;
			t=t/2;
		}
		long long int t1=k;
		while((t1%5)==0)
		{
			c5++;
			t1=t1/5;
		}
		long long int as=k;
		while(c2<c5)
		{
			as=as*4;
			c2=c2+2;
		}
		cout<<as<<endl;
	}
	return 0;
}
