#include <bits/stdc++.h>
using namespace std;
#define ll long long int
ll maxi=0 ;
vector<vector<ll> > v ;
ll vis[100004] = {0} ;
int main()
{
    ll i,j,k,n,m,p,t;
    cin >> n ;
    ll a[100001] ;
    for(i=0;i<n;i++) cin >> a[i] ;

    for(i=0;i<n;i++) {
        t = a[i] ;
        p = 0 ;
        while(t%5 == 0 && t > 0) {
            t = t/5 ;
            p++ ;
        }
        t = a[i] ;
        while(t%2 == 0 && t > 0) {
            t = t/2 ;
            p-- ;
        }

        if(p <= 0) cout << a[i] << endl ;
        else {
            if(p%2 == 0) p = p/2 ;
            else p = (p+1)/2 ;
            t = a[i] ;
            for(j=0;j<p;j++) t *= 4 ;
            cout << t << endl ;
        }
    }

    return 0 ;
}
