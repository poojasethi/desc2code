#include <bits/stdc++.h>
using namespace std;
int main() 
{
int n,i,cnt2,cnt5;
cin>>n;
long long int a,temp,ans;
for(i=0;i<n;i++)
{
	cnt2=cnt5=0;
	cin>>a;
	temp=a;
	while(temp%2==0)
	{
    cnt2++;
    temp/=2;
		
	}
	temp=a;
	while(temp%5==0)
	{
		cnt5++;
		temp=temp/5;
	}
	ans=a;
	while(cnt2<cnt5)
	{
		ans=ans*4;
		cnt2+=2;
	}
	cout<<ans<<"\n";
}
	return 0;
}
