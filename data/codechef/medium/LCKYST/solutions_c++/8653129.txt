#include<bits/stdc++.h>
using namespace std;
int main()
{
    long int n,m,c=0,j,g=0,x;
    long long int t;
    scanf("%ld",&n);
    while(n--)
    {
        scanf("%ld",&m);
        t=m;
        x=m;
        while(x%2==0)
        {
            g++;
            x=x/2;
        }

        while(m%5==0)
        {
            c++;
            m=m/5;
        }
        if(c<=g)
           printf("%lld\n",t);
        else
        {
            c=c-g;
            for(j=0;j<(c+1)/2;j++)
            {
                t=t*4;
            }
            printf("%lld\n",t);
        }
        g=0;
        c=0;
    }
    return 0;
}
