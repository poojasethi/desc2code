#define gc getchar_unlocked

using namespace std;
vector<int> v[50005];
int a[50005],l[50005],path[50005];
int visited[50005],visited1[50005],final[50005];

long long int ans=0;


int read_int() {
  char c = gc();
  int sign = 1;
  while((c<'0' || c>'9') && c!='-') c = gc();
  if(c=='-') {
    sign = -1;
    c = gc();
  }
  int ret = 0;
  while(c>='0' && c<='9' ) {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret*sign;
}

int dfs(int cur,int root,int count)
{
	int i,c=0;
	visited[cur]=root;
	l[cur]+=a[cur];
	for(i=0;i<v[cur].size();i++)
	{
		if(visited[v[cur][i]]==-1)
		{
			c+=dfs(v[cur][i],root,count);
			l[cur]+=l[v[cur][i]];
		}
	}
	c+=a[cur];
	return count+c;
}

void check(int cur,int root)
{
	int i,c=0,f=0,g=0;
	visited1[cur]=root;
	for(i=0;i<v[cur].size();i++)
	{
		if(visited1[v[cur][i]]==-1)
		{
			check(v[cur][i],root);
			if(l[v[cur][i]]>=1)
				f++;
		}
	}
	
	if(l[cur]>0 && l[cur]<l[root])
//		final[cur]=1;
		++ans;
	else if(l[cur]>1 && a[cur]==1)
//		final[cur]=1;
		++ans;
	else if(f>=2)
//		final[cur]=1;
		++ans;
}
