#include<bits/stdc++.h>
using namespace std;
int arr[100005];
int visited[100005];
vector<int>v[100005];
long long co,ans1,ans2;
void dfs(int ind)
{
	visited[ind]=1;
	if(arr[ind]==1)
	co++;
	for(int i=0;i<v[ind].size();i++)
	{
		if(!visited[v[ind][i]])
		dfs(v[ind][i]);
	}
}
int dfs1(int ind)
{
	int flag1=0;
	int flag2=0;
	visited[ind]=1;
	if(arr[ind]==1)
	flag1=1;
	for(int i=0;i<v[ind].size();i++)
	{
		if(!visited[v[ind][i]])
		{
			int temp=dfs1(v[ind][i]);
			flag2=flag2||temp;
		}
	}
	if(flag1||flag2)
	{
		ans2++;
		return 1;
	}
	return 0;
}
int main()
{
	long long n,m;
	scanf("%lld %lld",&n,&m);
	int i,j;
	for(i=0;i<n;i++)
	scanf("%d",&arr[i]);
	for(i=0;i<n;i++)
	visited[i]=0;
	int x,y;
	for(i=0;i<m;i++)
	{
		scanf("%d %d",&x,&y);
		x--;y--;
		v[x].push_back(y);
		v[y].push_back(x);
	}
	ans1=0;
	ans2=0;
	for(i=0;i<n;i++)
	{
		if(!visited[i])
		{
			co=0;
			dfs(i);
			ans1+=(co*(co-1))/2;
		}
	}
	for(i=0;i<n;i++)
	visited[i]=0;
	for(i=0;i<n;i++)
	{
		int temp=ans2;
		if(!visited[i]&&arr[i])
		{
			dfs1(i);
			if(temp+1==ans2)
			ans2-=1;
		}
	}
	printf("%lld %lld\n",ans1,ans2);
	return 0;
}
