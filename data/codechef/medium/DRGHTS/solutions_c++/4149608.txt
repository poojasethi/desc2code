#include<iostream>
#include<vector>
using namespace std;
int N,M;
vector<bool> f,s;
vector<vector<int> > e;
long long int ans1,ans2;
int fun(int c)
{
	int i,j,k=0;
	if(f[c])k++;
	s[c]=true;
	for(i=0;i<e[c].size();i++)
	{
		if(!s[e[c][i]])
		{
			k+=fun(e[c][i]);
		}
	}
	if(k)ans2++;
	return k;
}
int main()
{
	int i,j,k;

	cin>>N>>M;
	f.resize(N);
	e.resize(N);
	s.resize(N);
	for(i=0;i<N;i++)
	{
		cin>>k;
		if(k)f[i]=true;
		else f[i]=false;
		s[i]=false;
	}
	for(i=0;i<M;i++)
	{
		cin>>j>>k;
		e[j-1].push_back(k-1);
		e[k-1].push_back(j-1);
	}
	
	for(i=0;i<N;i++)
	{
		if(f[i]&&(!s[i]))
		{
			k=fun(i);
			if(k==1)ans2--;
			ans1+=k*((long long)k-1)/2;
		}
	}
	cout<<ans1<<" "<<ans2<<endl;

	return 0;
}
