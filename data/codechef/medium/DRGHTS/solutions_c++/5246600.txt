#include<bits/stdc++.h>

#define mod 1000000007
typedef set<string> ss;

vector<long long int> v[50004];
long long int cnt1,cnt2,cnt;
bool vis[50004],w[50004];
void dfs(long long int x)
{
	long long int s=v[x].size();
	vis[x]=1;
	for(long long int i=0;i<s;i++)
	{
		if(!vis[v[x][i]] && w[v[x][i]])
		{
			cnt1++;
			cnt2++;
			cnt2+=cnt;
			cnt=0;
			dfs(v[x][i]);
		}
		else if(!vis[v[x][i]] && !w[v[x][i]])
		{
			cnt++;
			dfs(v[x][i]);
		}
	}
	if(cnt>0)
	cnt--;
}
 
long long int ans1,ans2,x,y,i,m,n;
int main()
{
	//freopen("i.in", "r", stdin);
  	//freopen("o.out", "w", stdout);
  	cin>>n>>m;
  	for(i=1;i<=n;i++)
  	{
  		cin>>w[i];
  		vis[i]=0;
  	}
  	while(m--)
  	{
  		cin>>x>>y;
  		v[x].push_back(y);
  		v[y].push_back(x);	
  	}
  	ans1=0; ans2=0;
  	for(i=1;i<=n;i++)
  	{
  		if(!vis[i] && w[i])
  		{
  			cnt1=1;
  			cnt2=1;
  			cnt=0;
  			dfs(i);
  			ans1+=cnt1*(cnt1-1)/2;
  			if(cnt2>1)
  			ans2+=cnt2;
  		}
  	}
  	cout<<ans1<<" "<<ans2<<"\n";
	return 0;
}
