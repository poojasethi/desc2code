#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> v[100005];
int a[100005],l[100005],path[100005];
int visited[100005],visited1[100005],final[100005];

int dfs(int cur,int root,int count)
{
	int i,c=0;
	visited[cur]=root;
	l[cur]+=a[cur];
	for(i=0;i<v[cur].size();i++)
	{
		if(visited[v[cur][i]]==-1)
		{
			c+=dfs(v[cur][i],root,count);
			l[cur]+=l[v[cur][i]];
		}
	}
	c+=a[cur];
	return count+c;
}

void check(int cur,int root)
{
	int i,c=0,f=0,g=0;
	visited1[cur]=root;
	for(i=0;i<v[cur].size();i++)
	{
		if(visited1[v[cur][i]]==-1)
		{
			check(v[cur][i],root);
			if(l[v[cur][i]]>=1)
				f++;
		}
	}
	
	if(l[cur]>0 && l[cur]<l[root])
		final[cur]=1;
	else if(l[cur]>1 && a[cur]==1)
		final[cur]=1;
	else if(f>=2)
		final[cur]=1;
}

int main()
{
   #ifndef ONLINE_JUDGE

    #endif // ONLINE_JUDGE
	

	int n,m,i,j;
	long long int f,r,c,d;
	scanf("%d %d",&n,&m);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		visited[i]=-1;
		visited1[i]=-1;
		l[i]=0;
		path[i]=0;
		final[i]=0;
	}
	while(m--)
	{
		scanf("%d %d",&i,&j);
		v[i].push_back(j);
		v[j].push_back(i);
	}
	f=0;
	for(i=1;i<=n;i++)
	{
		c=0;
		if(visited[i]==-1)
			c+=dfs(i,i,0);
		f+=((long long int)c*((long long int)c-1))/2;
	}
	
/*	for(i=1;i<=n;i++)
		printf("%d ",l[i]);
	printf("\n");
*/

	d=0;
	for(i=1;i<=n;i++)
		if(visited1[i]==-1)
			check(i,i);
	
	for(i=1;i<=n;i++)
		d+=final[i];
	
	
	
	printf("%lld %lld\n",f,d);
	
	return 0;
}
