
t = int(raw_input())
for i in range(t):
    inp = raw_input()
    inp = inp.split()
    ds = int(inp[0])
    dt = int(inp[1])
    d = int(inp[2])
    if d >= (ds + dt):
        print (d - ds - dt)
    elif ds >= (d + dt):
        print (ds - d - dt)
    elif dt >= (d + ds):
        print (dt - d - ds)
    else:
        print 0
