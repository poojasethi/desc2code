#include<iostream>
#include<cstdio>
#define MAX 1000
using namespace std;
int main()
{
    float d1,d2,D;
    int count1;
    scanf("%d",&count1);
    while(count1>0)
    {
        scanf("%f %f %f",&d1,&d2,&D);
        if(d1+d2<D)
            printf("%f\n",D-d1-d2);
        else if(d1>d2+D)
            printf("%f\n",d1-d2-D);
        else if(d2>d1+D)
            printf("%f\n",d2-d1-D);
        else
            printf("0\n");
        
        count1--;

    }
}
