#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		float ds,dt,d,i,k;
		scanf("%f %f %f",&ds,&dt,&d);
		if(ds+dt<d)
		printf("%0.5f\n",(d-ds-dt));
		else if(ds+dt==d)
		printf("0.00000\n");
		else if(ds+dt>d)
		{
			if(ds>dt)
			{
				if(ds>d+dt)
				printf("%0.5f\n",ds-d-dt);
				else
				{
					printf("0.00000\n");
				}
			}
			else if(ds<dt)
			{
				if(ds<-d+dt)
				printf("%0.5f\n",-ds-d+dt);
				else
				{
					printf("0.00000\n");
				}
			}
			else
			printf("0.00000\n");
			
		}
	}
return 0;
}
