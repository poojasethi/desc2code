#include<bits/stdc++.h>
using namespace std;

int main()
{
std::ios::sync_with_stdio(false);

int T;
cin>>T;

while(T--)
{
int Ds, Dt, D;
cin>>Ds>>Dt>>D;

if(Ds+Dt<=D)
cout<<(D-(Ds+Dt));

else if(D+Ds<=Dt)
cout<<(Dt-Ds-D);

else if(D+Dt<=Ds)
cout<<(Ds-D-Dt);

else
cout<<"0";

cout<<endl;
}

return 0;
}
