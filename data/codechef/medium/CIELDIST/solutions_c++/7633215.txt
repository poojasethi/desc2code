	while(t--){
		int ds;
		int dt;
		int d;
		scanf("%d",&ds);
		scanf("%d",&dt);
		scanf("%d",&d);
		float ans;
		if(d>=(ds+dt))
			cout << (d-ds-dt) << endl;

	}

}
