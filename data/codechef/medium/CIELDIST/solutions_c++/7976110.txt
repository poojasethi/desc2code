
#include<bits/stdc++.h>
#define mod 1000000007
#define gc getchar
typedef long long int ll;
using namespace std;
int main()
{
    ll t;
    double ds,dt,d;
    cin>>t;
    while(t--)
    {
        cin>>ds>>dt>>d;
        if((ds+dt)>=d && (d+ds)>=dt && (d+dt)>=ds)
            printf("0.0\n");
        else
        {
            if(d>(ds+dt))
                printf("%.6lf\n",d-ds-dt);
            else if(ds>(d+dt))
                printf("%.6lf\n",ds-d-dt);
            else
                printf("%.6lf\n",dt-ds-d);
        }
 
    }
    return 0;
}
