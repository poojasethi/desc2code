#include <iostream>
#include <algorithm>
#include <cstdio>
#include <algorithm>
#include <cmath>

using namespace std;

double dt, ds, d;

int main()
{
	int tc;
	cin >> tc;
	while(tc--)
	{
		cin >> ds >> dt >> d;
	//	double k1 = 0.0, k2 = d - ds - dt, k3 = dt - ds - d, k4 = ds - dt - d;
	//cout << k1 << " " << k2 << " " << k3 << " " << k4 << endl;
		double m1 = max(0.0, d - ds - dt);
		double m2 = max(dt - ds - d, ds - dt - d);
		printf("%.6lf\n", max(m1, m2));
	}
	return 0;
}
