#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int t;
	float ds,dt,d;
	cin>>t;
	while(t--)
	{
		cin>>ds>>dt>>d;
		float dis;
		if(ds+dt < d)
		dis = d-ds-dt;
		else if(ds > d+dt)
		dis=ds-d-dt;
		else if(dt > d+ds)
		dis = dt-d-ds;
		else
		dis = 0.0;
		printf("%.6f\n",dis);
	}
	return 0;
}
