#include<cstdio>
using namespace std;
typedef long long int ll;
int max(int a,int b){
    if(a>b){
        return a;
    }
    return b;
}
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int Ds,Dt,D;
        scanf("%d %d %d",&Ds,&Dt,&D);
        if(D>=Ds+Dt)
        printf("%d.000000\n",D-Ds-Dt);
        else if(Dt>=D+Ds){
            printf("%d.000000\n",Dt-D-Ds);
        }else if(Ds>=D+Dt){
            printf("%d.000000\n",Ds-D-Dt);
        }
        else
        {
            printf("0.000000\n");
        }
    }
}
