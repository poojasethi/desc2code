#include <stdio.h>
 
int main() 
{
	int t, ds, dt, d;
	float diff;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d %d %d",&ds,&dt,&d );
		if((ds+dt)<d)
		 diff=d-ds-dt;
		else if((ds+dt)>=d&&ds<=(d+dt)&&dt<=(d+ds))
		 diff=0;
		else if(ds>(d+dt))
		 diff=ds-d-dt;
		else if(dt>(d+ds))
		 diff=dt-d-ds;
		printf("%.6f\n", diff);
	}
	
	return 0;
}
