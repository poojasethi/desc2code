#include <iostream>
#include<cmath>
using namespace std;


int main() {
	int t;
	cin>>t;
	int ds,dt,d;
	for(int i=0;i<t;i++)
	{
	cin>>ds>>dt>>d;
	if(ds+dt>d&&d+dt>ds&&d+ds>dt)
    cout<<"0\n";
	else if(d>=ds+dt)
    cout<<d-ds-dt<<"\n";
    else if(ds>=d+dt)
    cout<<ds-d-dt<<"\n";
    else
    cout<<dt-ds-d<<"\n";
	}
   
    return 0;
}
