#include<bits/stdc++.h>
using namespace std;
#define si(n) scanf("%d",&n)
#define pi(n) printf("%d\n",n)
#define pd(n) printf("%.5lf\n",n)
int main(){
	int t,ds,dt,d;
	
	scanf("%d",&t);
		while(t--){
			scanf("%d %d %d",&ds,&dt,&d);
			if((ds+dt)<d){
				printf("%.6lf\n",(double)(d-ds-dt));
			}
			else if(ds>=dt+d)
				printf("%.6lf\n",(double)(ds-d-dt));
			else if(dt>=d+ds)
				printf("%.6lf\n",(double)(dt-ds-d));
			else
				printf("0.000000\n");
		}
}
