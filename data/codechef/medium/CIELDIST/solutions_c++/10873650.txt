#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
       int m,n,d;
       scanf("%d%d%d",&m,&n,&d);
     //  printf("%f %f %f\n",(double)m,(double)n,(double)d);
        if(m+n<=d)
        printf("%f\n",(double)(d-m-n));
        else if(m+n>d&&m+d>n&&n+d>m)
        {
            printf("0.0\n");
        }
        else if(m>=n+d)
            printf("%f\n",(double)(m-n-d));
        else
            printf("%f\n",(double)(n-m-d));
        }
    return 0;
}
