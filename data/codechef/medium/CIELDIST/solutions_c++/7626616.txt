#include<stdio.h>
using namespace std;
int main()
{
    int t;
    float ds,dt,d,temp;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%f%f%f",&ds,&dt,&d);
        if(dt<ds)
        {
            temp=ds;
            ds=dt;
            dt=temp;
        }
        if(d>(ds+dt))
            printf("%f\n",d-ds-dt);
        else if(dt>(d+ds))
            printf("%f\n",dt-d-ds);
        else
            printf("0.0\n");
    }
    return 0;
}
