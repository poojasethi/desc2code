#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;scanf("%d",&t);
	while(t--)
	{
		int ds,dt,d;
		scanf("%d%d%d",&ds,&dt,&d);
        if(ds >= d+dt )
          printf("%d\n",ds-d-dt);
        else if(dt >= d + ds)
          printf("%d\n",dt-d-ds);
        else if(ds+dt<=d)
          printf("%d\n",d-ds-dt);
        else
          printf("0\n");
    }
	return 0;
}
