#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
int main(){
	int t;
	cin >> t;
	while(t--){
		LL D, Ds, Dt;
		cin >> Ds >> Dt >> D;
		if(Ds >= D + Dt){
			cout << Ds - (D + Dt) << "\n";
		} else if(Dt >= D + Ds){
			cout << Dt - (D + Ds) << "\n";
		} else if(D >= Ds + Dt){
			cout << D - (Ds + Dt) << "\n";
		} else cout << "0\n";
	}
	return 0;
}
