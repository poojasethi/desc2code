#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		double d,ds,dt,ans;
		cin>>ds;cin>>dt;cin>>d;
        if(d>ds+dt)
        ans=d-dt-ds;
        else if(ds>d+dt)
        ans=ds-d-dt;
        else if(dt>d+ds)
        ans=dt-d-ds;
        else if(d<=ds+dt)
        {
        	 ans=0;
        }
       printf("%f\n",ans);
	}
	return 0;
}
