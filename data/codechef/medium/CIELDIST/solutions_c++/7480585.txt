#include <iostream>
using namespace std;

int main() {
	int r1,r2,ds,dt,d,t;
	cin>>t;
	while(t--)
	{
		cin>>ds>>dt>>d;
		r1=max(ds,dt);
		r2=min(ds,dt);
	    if(r1+r2<d)
	     {
	     	int ans=d-r1-r2;
	     	cout<<ans<<endl;
	     }
	     else if (r1+r2==d)
	     cout<<"0"<<endl;
	     else if(r1-r2<=d)
		 cout<<"0"<<endl;
		 else if(r1-r2>d)
		  {
		  	int ans=r1-d-r2;
		  	cout<<ans<<endl;
		  }
	}
	return 0;
}
