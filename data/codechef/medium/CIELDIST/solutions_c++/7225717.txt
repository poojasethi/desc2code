#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>


int main()
{
  int T;
  scanf("%d\n", &T);
  while(T--)
  {
    int Ds, Dt, D;
    scanf("%d %d %d\n", &Ds, &Dt, &D);

    int res = 0;
    if(D < Dt + Ds)
    {
      if((D + Ds) < Dt)
        res = Dt - (D + Ds);
      else if((D + Dt) < Ds)
        res = Ds - (D + Dt);
    }
    else
    {
      res = D - Dt - Ds;
    }

    printf("%d\n", res);
  }

  return 0;
}
