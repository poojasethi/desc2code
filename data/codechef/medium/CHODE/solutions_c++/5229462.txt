#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
struct code
{
    char ch;
    int count;
    char rep;
    code()
    {
        count=0;
    }

};
bool comp(code c1,code c2)
{
    if(c1.count<c2.count)
        return false;
        else if(c1.count>c2.count)
        return true;
    else
    return (c1.ch>c2.ch);
}
int main()
{
    ios_base::sync_with_stdio(false);
    int t,i,k;
    char f[28],c[150010];
    cin>>t;
    cin.ignore();
    code * D;
    while(t>0)
    {
        t--;
        cin.getline(f,28);
        cin.getline(c,150010);
        D=new code[26];
        i=0;
        while(c[i]!='\0')
        {
            if(isalpha(c[i]))
            {
                if(isupper(c[i]))
                {
                    D[c[i]-65].count++;
                    D[c[i]-65].ch=c[i]+32;
                }
                else
                   {
                    D[c[i]-97].count++;
                    D[c[i]-97].ch=c[i];
                   }
            }
            i++;
        }
        sort(D,D+26,comp);
        k=25;
        i=0;
        while(D[i].count!=0&&k>=0)
        {
            D[D[i].ch-97].rep=f[k];
            i++;
            k--;
        }
        i=0;
        while(c[i]!='\0')
        {
         if(isalpha(c[i]))
            {
                if(isupper(c[i]))
                {
                    c[i]=D[c[i]-65].rep-32;
                }
                else
                    c[i]=D[c[i]-97].rep;
            }
            i++;
        }
        cout<<c<<endl;
    }
}
