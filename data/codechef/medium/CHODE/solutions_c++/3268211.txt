#include<bits/stdc++.h>

using namespace std;

int main()
{
	int cases , p , asc , i , len1 , len2 , l , get;
	char rescue;
	string str1 , str2;
	vector < pair < int , char > > dalo;
	pair < int , char > x;
	int arr[26] , check[26];
	
	scanf("%d", &cases);
	get = getchar();
	
	for(l = 0; l < cases; l++)
	{
		getline(cin , str1);
		getline(cin , str2);
		
		len1 = str1.size ();
		len2 = str2.size ();
		
		for(i = 0; i < 26; i++)
		{
			arr[i] = 0;
			check[i] = -1;
		}
		
		for(i = 0; i < len2; i++)
		{
			if(str2[i] >= 'a' && str2[i] <= 'z')
			{
				p = str2[i];
				arr[p - 97] ++;
			}
			
			if(str2[i] >= 'A' && str2[i] <= 'Z')
			{
				p = str2[i];
				arr[p - 65] ++;
			}
		}
		
		for(i = 0; i < 26; i++)
		{
			if(arr[i] != 0)
			{
				x.first = arr[i];
				x.second = char(i + 97);
				
				dalo.push_back (x);
				
			}
		}
		
		sort(dalo.begin () , dalo.end ());
		
		for(i = dalo.size () - 1; i >= 0 ; i --)
		{
			p = dalo[i].second;
			check[p - 97] = dalo.size () - i;
		}
			
			
		
		for(i = 0; i < len2;  i++)
		{
			p = str2[i];
		
			if(str2[i] >= 'a' && str2[i] <= 'z')
			{
				if(check[p - 97] != -1)
					printf("%c", str1[len1 - check[p - 97]]);	
							
			}
			
			else if(str2[i] >= 'A' && str2[i] <= 'Z')
			{
				if(check[p - 65] != -1)
				{
					asc = str1[len1 - check[p - 65]];
					rescue = char(asc - 32);
					
					printf("%c", rescue);
				}
			}
			
			else
				printf("%c", str2[i]);
		}
		
		dalo.clear ();
		
		printf("\n");
		
	}
	
	return 0;
}
