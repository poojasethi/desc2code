#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int t;cin>>t;
    while(t--)
    {
        char a[26];int c[26]={0};
        for(int i=0;i<26;i++)cin>>a[i];
        char b[150005];int ind=0;
        getchar();
        while(1)
        {
            char p;scanf("%c",&p);if(p=='\n')break;
            b[ind++]=p;
            if(int(p)>=65&&int(p)<=90)c[p-'A']++;
            else if(int(p)>=97&&int(p)<=122) c[p-'a']++;
        }
        int cnt=0;
        for(int i=0;i<26;i++)if(c[i])cnt++;
        int maxi=0,indi=0;
        char d[26];int k=25;
        for(int j=0;j<cnt;j++)
        {
            maxi=0;
            for(int i=0;i<26;i++){if(c[i]>=maxi){maxi=c[i];indi=i;}}
            c[indi]=0;
            d[indi]=a[k--];
        }
        for(int i=0;i<ind;i++)
        {
            if(int(b[i])>=65&&int(b[i])<=90)
            cout<<char(int(d[int(b[i])-'A'])-32);
            else if(int(b[i])>=97&&int(b[i])<=122)
            cout<<d[b[i]-'a'];
            else cout<<b[i];
        }
        cout<<endl;
    }
    return 0;
}
