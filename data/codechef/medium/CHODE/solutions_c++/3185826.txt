#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
int t ;
scanf("%d",&t);
while(t--)
{
char str[150010];
char fr[26];
char dup[150010];
int vv ;
map<char,char> mp ;
vector<pair<int,char> > v ;
vector<pair<int,char> > :: iterator it ;
for(it = v.begin() , vv = 97 ; vv <= 122; it++)
v.push_back(make_pair(0,char(vv++)) ) ;
scanf("%s\n",fr);
scanf("%[^\n]s",str);
int l = strlen(str);

for(int i = 0 ; i < l ; i++)
{
dup[i] = str[i];
str[i] = tolower(str[i]);
if(str[i] >= 'a' and str[i] <= 'z')
v[str[i] - 97].first++ ;
}
vv = 0;
sort(v.begin(),v.end());
for(it = v.begin() ; it != v.end() ; it++)
{
mp[(*it).second] = fr[vv++];
}
for(int i = 0 ; i< l ; i++)
{
if(dup[i] >= 'a' and dup[i] <= 'z' || dup[i] >= 'A' and dup[i] <= 'Z')
{
if(dup[i] >= 'A' and dup[i] <= 'Z')
{
printf("%c",toupper(mp[tolower(dup[i])]));
}
else
{
printf("%c",mp[dup[i]]);
}
}
else
{
printf("%c",dup[i]);
}
}
printf("\n");
}
return 0;
}
