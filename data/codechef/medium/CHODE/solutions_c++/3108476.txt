#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
char input[150001];
struct alphabet
{
	int index;
	int val;
};

bool comparer(alphabet a1,alphabet a2)
{
	if(a1.val==a2.val)
		return a1.index<a2.index;
	else
		return a1.val<a2.val;
}

int main()
{
	int t;
	scanf("%d",&t);
	char str[27];
	char replace[27];
	

	alphabet arr[26];
	char c;
	
	while(t--)
	{		
		scanf("%s",str);

		
		int l = 0;
		getchar();
		c = getchar();
		while(c!='\n' && c!=EOF)
		{
			input[l++] = c;
			c=getchar();
		}
		input[l]='\0';
		
		for(int i=0;i<26;i++)
		{
			arr[i].index = i;
			arr[i].val = 0;
		}
		
		for(int i=0;i<l;i++)
		{
			if(input[i]<='Z' && input[i]>='A')
				arr[input[i]-'A'].val++;
			if(input[i]<='z' && input[i]>='a')
				arr[input[i]-'a'].val++;
		}
		
		sort(arr,arr+26,comparer);
		
		int len = 25;
		
		for(int i=25;i>=0;i--)
		{
			if(arr[i].val==0)
				break;
			else
				replace[arr[i].index]=str[len--];
		}
		
		for(int i=0;i<l;i++)
		{
			if(input[i]<='Z' && input[i]>='A')
				input[i] = replace[input[i]-'A']-'a'+'A';
			if(input[i]<='z' && input[i]>='a')
				input[i] = replace[input[i]-'a'];		
		}
		
		printf("%s\n",input);

	}
	return 0;
}
