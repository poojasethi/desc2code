{

{
	return (a / gcd<T>(a, b) * b);
}

{
	return (a < b ? a : a % b);
}

inline V ifast(I &x)
{
	register I c = getchar_unlocked();
	x = 0;
	I neg = 0;
	for(; ((c<48 || c>57) && c != '-'); c = getchar_unlocked());
	if(c=='-')
	{
		neg = 1;
		c = getchar_unlocked();
	}
	for(; c>47 && c<58 ; c = getchar_unlocked())
	{
		x = (x<<1) + (x<<3) + c - 48;
	}
	if(neg)
		x = -x;
}
