#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define MAX 150005
using namespace std;
struct track
{
    char ch;
    int cnt;
}arr[MAX];
typedef struct track t;
bool fn(t a,t b)
{
    if(a.cnt<b.cnt) return true;
    else if(a.cnt==b.cnt)
    {
        if(a.ch<b.ch) return true;
        else return false;
    }
    else return false;
}
char ans[130];
int main()
{
    int t,i,l,p,z;
    char r[30];
    char s[MAX];
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        gets(r);gets(s);
        int c[130]={0};
        char lo[MAX];
        l=strlen(s);
        for(i=0;i<l;i++)
        {
            if((s[i]>='A')&&(s[i]<='Z'))
            lo[i]=(char)(s[i]+32);
            else lo[i]=s[i];
        }
        for(i=0;i<l;i++)
        if((lo[i]>='a')&&(lo[i]<='z')) c[lo[i]]++;
        p=0;z=25;
        for(i=97;i<=122;i++)
        {
            if(c[i]>0)
            {
                arr[p].ch=(char)i;
                arr[p].cnt=c[i];
                p++;
            }
        }
        sort(arr,arr+p,fn);
        for(i=p-1;i>=0;i--)
        ans[arr[i].ch]=r[z--];
        for(i=0;i<l;i++)
        {
            if((s[i]>='A')&&(s[i]<='Z'))
            printf("%c",(char)(ans[lo[i]]-32));
            else if((s[i]>='a')&&(s[i]<='z'))
            printf("%c",ans[lo[i]]);
            else printf("%c",lo[i]);
        }
        printf("\n");
        memset(lo,'\0',sizeof(lo));
    }
    return 0;
}
