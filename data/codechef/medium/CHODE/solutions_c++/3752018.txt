#include<stdio.h>
#include<stdlib.h>
char freq[26],str[150002];
typedef struct nod{
    long long freqen;
    char a;
}element;
int compare(const void *x, const void *y)
{
  return (((element*)x)->freqen -((element*)y)->freqen);
}
element ele[26];
int pos[26];
int main()
{
    long long i,j;
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",freq);
        getchar();
        gets(str);
        for(i=0;i<26;i++)
        {
            ele[i].a=(97+i);
            ele[i].freqen=0;
        }
        for(i=0;str[i]!='\0';i++)
        {
            if(str[i]>96 && str[i]<123)
                ele[str[i]-97].freqen++;
            else if(str[i]>64 && str[i]<91)
                ele[str[i]-65].freqen++;
        }
        qsort(ele,26,sizeof(element),compare);
        for(i=0;i<26;i++)
        {
            pos[ele[i].a-97]=i;
        }
        for(i=0;str[i]!='\0';i++)
        {
            if(str[i]>96 && str[i]<123)
                printf("%c",freq[pos[str[i]-97]]);
            else if(str[i]>64 && str[i]<91)
                printf("%c",(freq[pos[str[i]-65]]-32));
            else
                printf("%c",str[i]);
        }
        printf("\n");
    }
    return 0;
}
