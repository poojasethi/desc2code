#include<bits/stdc++.h>
using namespace std;
inline bool comp(const pair<char,int> &a,const pair<char,int> &b) {
	return a.second == b.second ? a.first > b.first : a.second > b.second;
}
int main() {
	int t,i;
	scanf("%d",&t);
	while(t--) {
		string s,v;
		cin >> s;
		getchar();
		getline(cin,v);
//		cout << s <<endl;
//		cout << v <<endl;
		pair<char,int>a[26];
		int b[26];
		for(i=0;i<26;i++){
			a[i].first = i+97;;
			a[i].second = 0;
		}
		for(i=0;i<v.size();i++) {
			if(v[i]>=97 && v[i]<=122)
			a[v[i]-97].second++;
			if(v[i]>=65 && v[i]<=90)
			a[v[i]-65].second++;
		}
		sort(a,a+26,comp);
//		for(i=0;i<26;i++) {
//			printf("%c %d\n",a[i].first,a[i].second);
//		}
		for(i=s.size()-1;i>=0;i--) {
			b[a[25-i].first-97] = s[i];
		}
//		for(i=0;i<26;i++) {
//			printf("b[%d]=%c\n",i,b[i]);
//		}
		for(i=0;i<v.size();i++) {
			if(v[i]>=97 && v[i]<=122) {
//				printf("ss= %d %c\n",v[i]-97,b[v[i]-97]);
				printf("%c",b[v[i]-97]);
			} else if(v[i] >=65 && v[i]<=90) {
//				printf("ss= %d %c\n",v[i]-65,b[v[i]-65]);
				printf("%c",b[v[i]-65]-32);
			}else
			printf("%c",v[i]);
		}
		printf("\n");
	}
	return 0;
}
