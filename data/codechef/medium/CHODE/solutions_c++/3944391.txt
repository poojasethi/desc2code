#include <cstdio>
#include <algorithm>
#include <cctype>

using namespace std;

struct chars
{
	char c;
	int count;
};

bool comp(chars A, chars B)
{
	if (A.count == B.count) return A.c < B.c;
	return A.count < B.count;
}

int main()
{	
	int T;
	scanf("%d",&T);
	
	char refstr[40];
	char str[150010];

	chars temp[26];
	int i;
	getchar();

	while(T--)
	{
		gets(refstr);
		gets(str);

		char ans[150010];
		int k = 0;

		for(i = 0; i < 26; i++)
		{
			temp[i].c = 'a' + i;
			temp[i].count = 0;
		}

		i = 0;
		while(str[i] != '\0')
		{
			if(str[i] >= 'a' && str[i] <= 'z')
				temp[str[i] - 'a'].count++;
			else if(str[i] >= 'A' && str[i] <= 'Z')
				temp[str[i] - 'A'].count++;

			i++;
		}

		sort(temp, temp + 26, comp);

		/*for(i = 0; i < 26; i++)
			printf("%c %d\n", temp[i].c, temp[i].count);*/

		int finder[26];
		for(i = 0; i < 26; i++)
			finder[temp[i].c - 'a'] = i;

		for(i = 0; str[i] != '\0'; i++)
		{
			if(str[i] >= 'a' && str[i] <= 'z')
				ans[i] = refstr[finder[str[i] - 'a']];
			else if(str[i] >= 'A' && str[i] <= 'Z')
				ans[i] = refstr[finder[str[i] - 'A']] - 32;
			else ans[i] = str[i];
		}

		ans[i] = '\0';
		puts(ans);
	}
	return 0;
}
