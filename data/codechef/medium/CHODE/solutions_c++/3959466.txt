#include <iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	char str[150002],str1[150002],*b,*e;
	int cnt,a[27],c[27],r[27],i,j,g;
	scanf("%s",str);
	scanf("\n%[^\n]s",str1);
	g=strlen(str1);
	for(i=0;i<=25;i++)
	{
		a[i]=0;
		c[i]=0;
	}
	for(b=str1,e=&str1[g-1];b<=e;b++)
	{
		if(((int)*b>=65)&&((int)*b<=90))
		{
			a[(int)*b%65]++;
			c[(int)*b%65]++;
		}	
		else if(((int)*b>=97)&&((int)*b<=122))
		{
			a[(int)*b%97]++;
			c[(int)*b%97]++;
		}
	}
	sort(c,c+26);
	cnt=0;
	for(i=25;i>=0;i--)
	{
		if(c[i]==0)
		{
			break;
		}
		for(j=25;j>=0;j--)
		{
			if(c[i]==a[j])
			{
				r[j]=cnt;
				cnt++;
				a[j]=-1;
				break;
			}
		}
	}
	for(b=str1,e=&str1[g-1];b<=e;b++)
	{
		if(((int)*b>=65)&&((int)*b<=90))
			printf("%c",toupper(str[25-r[(int)*b%65]]));
		else if(((int)*b>=97)&&((int)*b<=122))
			printf("%c",str[25-r[(int)*b%97]]);
		else
			printf("%c",*b);
	}
	printf("\n");
	}
	return 0;
}
