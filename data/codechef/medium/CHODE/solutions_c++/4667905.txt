void ins_sort(int arr[][26],int n)
{
	int key1,key2,i,j;
	
	for(j=1;j<n;j++)
	{
		
		key1=arr[1][j];
		key2=arr[0][j];
		i=j-1;
		while(i>=0&&key1<arr[1][i])
		{
		arr[1][i+1]=arr[1][i];
		arr[0][i+1]=arr[0][i];
		i--;
		}
		arr[0][i+1]=key2;
		arr[1][i+1]=key1;
		
		
	} 
}
