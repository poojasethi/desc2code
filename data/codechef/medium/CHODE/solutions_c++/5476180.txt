#include<iostream>
using namespace std;
#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
 
    int t,count=0,max,j,k;
    char cipher[150001];
    char text[27];
 
    long len,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",text);
        scanf(" %[^\n]s",cipher);
        i=0;
        long freq[26]={0};
        count=0;
        len=0;
        while(cipher[i]!='\0')
        {
            len++;
            if(isalpha(cipher[i]))
            {
                if(cipher[i]>='a' && cipher[i]<='z')
                {
                    if(freq[cipher[i]-'a']==0)
                        count++;
                    freq[cipher[i]-'a']++;
                }
                else
                {
                    if(freq[cipher[i]-'A']==0)
                        count++;
                    freq[cipher[i]-'A']++;
                }
            }
            i++;
        }
        char freqseqcipher[26],freqseqtext[26];
        int l=0;
        for(j=0;j<count;j++)
        {
            max=0;
            for(k=0;k<26;k++)
            {
                if(freq[k]>=freq[max])
                {
                    max=k;
                }
 
            }
 
            freqseqcipher[j]=max+'a';
            freq[max]=0;
        }
 
        for(j=0;j<count;j++)
        {
            freqseqtext[freqseqcipher[j]-'a']=text[25-j];
        }
        for(i=0;i<len;i++)
        {
            if(isalpha(cipher[i]))
            {
                if(cipher[i]>='a' && cipher[i]<='z')
                {
                    cipher[i]=freqseqtext[cipher[i]-'a'];
                }
                else
                {
                    cipher[i]=freqseqtext[cipher[i]-'A']-32;
                }
            }
 
        }
 
        printf("%s\n",cipher);
    }
    return 0;
}
