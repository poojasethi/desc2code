#define MOD 1000000007
using namespace std;
int main(){   
    long long T,N,G,nbit,K,i;
    static long  long f[1100000];
    f[1]=1,f[2]=2;
    for(i=3;i<1100000;i++)
      f[i]=(f[i-1]%MOD+f[i-2]%MOD)%MOD;
    scanf("%lld",&T);
    while(T--){
    	nbit=0;
    	scanf("%lld%lld",&N,&G);
    	K=f[N];
    	while(K){
    		nbit+=K%2;
    		K/=2;
    	}
    	if(nbit==G)
    	   printf("CORRECT\n");
    	else
		   printf("INCORRECT\n");   
    }
	return 0;
}
