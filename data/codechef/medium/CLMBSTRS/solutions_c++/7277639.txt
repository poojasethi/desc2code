#include<iostream>
#include<string.h>
using namespace std;
#define ll long long
#define MAX 1000000007

int res[1000001];

int find(ll n,ll m){
	ll ans,count=0;
	ans=res[n-1];
	while(ans){
		ans=ans & (ans-1);
		count++;
	}
	if(count==m)
		cout<<"CORRECT\n";
	else
		cout<<"INCORRECT\n";		
}

int main(){
	ll t,i;
	res[0]=1;res[1]=2;
	for(i=2;i<1000001;i++){
		res[i]=0;
		res[i]+=(res[i-1]+res[i-2])%1000000007;
	}
	cin>>t;
	while(t--){
		ll n,m;
		cin>>n>>m;
		find(n,m);
	}
	return 0;
}
