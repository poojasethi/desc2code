#include <iostream>
using namespace std;
long int a[1000001];
int main() {
	// your code goes here
	long int t,n,g;
	cin>>t;
	a[0]=1;
	a[1]=1;
	for(long int j=2;j<=1000001;j++){
	    a[j]=(a[j-1]+a[j-2])%1000000007;
	}
	for(long int k=2;k<=1000001;k++){
	    long int sum=0;
	    long int h=a[k];
	    while(h>0){
	        if(h%2!=0)
	        sum++;
	        h=h/2;
	    }
	    a[k]=sum;
	}
	/*for(long int i=0;i<8;i++){
	    cout<<a[i]<<endl;
	}*/
	for(long int i=0;i<t;i++){
	    cin>>n>>g;
	    if(a[n]==g)
	    cout<<"CORRECT"<<endl;
	    else
	    cout<<"INCORRECT"<<endl;
	}
	return 0;
}
