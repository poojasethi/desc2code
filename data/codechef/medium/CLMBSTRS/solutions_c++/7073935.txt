#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<assert.h>
#define REP(i,a,b) for(i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)
 
#define M 1000000007
 
int count_bit(int i){
  int res = 0;
  while(i){
    res += i%2;
    i /= 2;
  }
  return res;
}
 
int main(){
  int T, N, G;
 
  int i;
  static int fib[1100000];
 
  assert( scanf("%d",&T)==1 );
  assert( T<=100000 );
 
  fib[0] = fib[1] = 1;
  REP(i,2,1100000) fib[i] = (fib[i-1]+fib[i-2])%M;
 
  while(T--){
    assert( scanf("%d%d",&N,&G)==2 );
    assert( 1<=N && N<=1000000 );
 
    if(count_bit(fib[N])==G) puts("CORRECT");
    else                     puts("INCORRECT");
  }
 
  return 0;
}
