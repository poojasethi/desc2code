#define mod 1000000007
using namespace std;
map<ll,ll> m;
ll fibo(ll b)
{
	if(b==1)
		return 1;
	else if(b==0)
		return 1;
	else if(m.count(b))
		return m[b];
	else
	{
		ll x=fibo(b/2);
		ll z=fibo(b/2-1);
		ll an,tmp1,tmp2;
		tmp1=(x*x)%mod;
		if(b%2==1)
		{
			tmp2=(x*z)%mod;
			an=(tmp1+2*tmp2)%mod;
		}
		else
		{
			tmp2=(z*z)%mod;
			an=(tmp1+tmp2)%mod;
		}
		m[b]=an;
		return an;
	}
}
int main()
{
	int t;
	si(t);
	while(t--)
	{
		int n,g;
		si(n);si(g);
		ll ans=fibo(n);
		//pll(ans);
		int cnt=0;
		while(ans!=0)
		{
			if(ans%2==1)
				cnt++;
			ans/=2;
		}
		if(g==cnt)
			printf("CORRECT\n");
		else
			printf("INCORRECT\n");
	}
	return 0;
}
