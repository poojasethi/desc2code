#include<bits/stdc++.h>
#define m 1000000007
using namespace std;
long long int A[1000001];
int main()
{
    int i;
    A[1]=1;
    A[2]=2;
    for(i=3;i<=1000000;i++)
    {
        A[i]=(A[i-1]+A[i-2])%m;
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k;
        scanf("%d %d",&n,&k);
        bitset<30> emi(A[n]);
        if(emi.count()==k)
            cout<<"CORRECT"<<endl;
        else
            cout<<"INCORRECT"<<endl;



    }




}
