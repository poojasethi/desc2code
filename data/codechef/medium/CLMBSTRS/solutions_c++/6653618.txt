#include<iostream>
using namespace std;
#define M 1000000007
#define MAX 1000000
int main()
{
   int T,N,i,G;
   int fibbonicci[MAX];
   fibbonicci[0]=1;
   fibbonicci[1]=2;
   for(i=2;i<MAX;i++)
   fibbonicci[i]=(fibbonicci[i-1]+fibbonicci[i-2])%M;
   cin>>T;
   while(T--)
   {
      cin>>N>>G;
      N=fibbonicci[N-1];
      for(i=0;N;N/=2)
      i+=N%2;
      if(i==G)
         cout<<"CORRECT"<<endl;
      else
         cout<<"INCORRECT"<<endl;
   }
   return 0;
}
