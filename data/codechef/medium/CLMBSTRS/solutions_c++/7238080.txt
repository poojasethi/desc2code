#include <iostream>
#include <vector>
using namespace std;

#define MAX 1000006
#define MOD 1000000007

long long f[MAX];

int count(long long n)
{
	int c = 0;
	while(n) {
		c += (n & 1) ? 1 : 0;
		n = n >> 1;
	}
	return c;
}

int main() {

	f[0] = 1, f[1] = 1;
	for(int i = 2; i < MAX; i++) {
		f[i] = (f[i - 1] + f[i - 2]) % MOD;
	}

	int t, n, g, m;
	cin >> t;
	while(t--) {
		cin >> n >> g;
		m = count(f[n]);
		if(m == g)
			cout << "CORRECT" << endl;
		else
			cout << "INCORRECT" << endl;
	}
	return 0;
}
