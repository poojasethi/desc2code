#include<iostream>
using namespace std;
#define m 1000000007
int test,n,a[1000003],i,ans,g,w;
int main()
{	
	cin>>test;	
	a[0]=1;
	a[1]=1;
	for(i=2;i<1000003;++i)
	a[i]=(a[i-1]+a[i-2])%m;
	while(test--)
	{
		cin>>n>>g;
		
		ans=0;
		w=a[n];		
		while(w>0)
		{
			if(w%2==1)
			ans++;
			w=w/2;
		}		
		if(ans==g)
    	printf("CORRECT\n");
    	else
    	printf("INCORRECT\n");		
	}
	return 0;
}
