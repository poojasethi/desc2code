	#include<iostream>
	#include<stdio.h>
	#define mod 1000000007
	#define get getchar_unlocked
	#define put putchar_unlocked
 	using namespace std;

 	long long int a[1000005];
	inline int inp()
	{
	
	int n=0, c=get(),f=1;
	while(c!='-'&&(c<'0'||c>'9')) c=get();
	if(c=='-'){f=-1;c=get();}
	 
	while(c>='0'&&c<='9'){n=(n<<3)+(n<<1)+c-'0',c=get();}
	return n*f;
	 
	}
	int main()
	{
		int t;
		cin>>t;
		
		a[1]=1;
		a[2]=2;
		for(int i=2;i<=1000004;i++)
			a[i]=(a[i-1]+a[i-2])%mod;
		while(t>0)
		{
			int n,g;
			n=inp();
			g=inp();
			long long int temp=a[n+1];
			int count=0;
			while(temp>0)
			{
				if(temp%2==1)
					count++;
				temp=temp>>1;
			}	
			if(count==g)
				cout<<"CORRECT\n";
			else
				cout<<"INCORRECT\n";
			t--;
		}	
	}
