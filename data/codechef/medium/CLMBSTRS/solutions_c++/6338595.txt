#include<iostream>
#define MOD 1000000007
using namespace std;
int main()
{
    long long int *arr= new long long int [1000001];
    arr[0]=0;
    arr[1]=1;
    arr[2]=2;

    int i;
    for(i=3;i<1000001;i++)
    {
        arr[i]=(arr[i-1]+arr[i-2])%MOD;
    }

    int t;
    cin>>t;
    while(t--)
    {
        int n,g;
        cin>>n>>g;

        int number=arr[n];

        int countone=0;

        while(number!=0)
        {
            countone=countone+number%2;
            number=number/2;
        }

        if (countone==g)
            cout<<"CORRECT"<<'\n';
        else
            cout<<"INCORRECT"<<'\n';


    }
    return 0;
}
