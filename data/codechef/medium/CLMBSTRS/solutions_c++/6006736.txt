#include<iostream>
#include<math.h>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<limits.h>
#define gc getchar_unlocked
#define mod 1000000007
using namespace std;

int pc[1000001];
void set() 
{
	pc[1]=1;
	pc[2]=2;
	for(int i=3;i<=1000000;i++) pc[i]=((pc[i-1]%mod)+(pc[i-2]%mod))%mod;	
}
void fastin(int &x)
{
	register int a=gc();
	x=0;
	for(;a<48 || a>57;a=gc());
	for(;a>47 && a<58;a=gc())
		x=(x<<1)+(x<<3)+a-48;		
}


int main()
{
	set();
	int t,n,g,count_one,ways;
	fastin(t);
	while(t--)
	{
		fastin(n);fastin(g);
		ways=pc[n];
		count_one=0;
		while(ways>0)
		{
			if(ways%2) count_one++;
			(ways>>=1);
		}
		if(count_one==g) puts("CORRECT");
		else puts("INCORRECT");
	}
	return 0;
}
