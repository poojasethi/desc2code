#include <bits/stdc++.h>
using namespace std;
#define bitcnt __builtin_popcount

#define N 1000000+25
#define M 1000000007
int ans[N];
	
void preprocess() 
{
	long long a,b,c;
	ans[1] = a = 1;
	ans[2] = 1; b = 2;
	for(int i=3; i<N; i++) {
		c = a + b;
		c %= M;
		ans[i] = bitcnt((int)c);
		a = b;
		b = c;
	}
}
	

int main()
{
	int t,n,g;
	preprocess();
	scanf("%d",&t);
	while(t--) {
		scanf("%d%d",&n,&g);
		printf("%s\n",ans[n]==g? "CORRECT": "INCORRECT");
	}
	return 0;
}
