#include<iostream>
using namespace std;
#define mod 1000000007
#define max_size 1000009 
int dp[max_size];
void pre(){
  dp[1]=1;
  dp[2]=2;
 for(int i=3;i<max_size;i++){
  dp[i]=(dp[i-1]+dp[i-2])%mod;
  }
}
int main(){
 int t,n,g;
 pre();
 cin>>t;
 while(t--){
 cin>>n>>g;
 int ans=dp[n];
 while(ans>0){
   if(ans&1) g--;
   ans=ans>>1; 
 }
 if(g==0) cout<<"CORRECT"<<endl;
 else cout<<"INCORRECT"<<endl;
 }
 return 0;
}
