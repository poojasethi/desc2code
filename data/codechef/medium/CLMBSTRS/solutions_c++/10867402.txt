#include <iostream>

#define MAX_FIB 	1000000007
#define MAX_FIB_ARR_LEN 1000001

int fib[MAX_FIB_ARR_LEN];

int countBits(long int n)
{
	int count=0;
	while (n) {
		if (n & 01)
			count++;
		n >>= 1;
	}
	return count;
}

void calculateFib()
{
	fib[0] = 0;
	fib[1] = 1;

	int i;
	for (i=2; i<MAX_FIB_ARR_LEN; i++) {
		fib[i] = fib[i-1] + fib[i-2];
		if (fib[i] >= MAX_FIB)
			fib[i] -= MAX_FIB;
	}

}

int main()
{
	calculateFib();

	int t, n, g;
	std :: cin >> t;

	while (t--) {
		std :: cin >> n >> g;
		if (g == countBits(fib[n+1]))
			std::cout << "CORRECT" << std::endl;
		else
			std::cout << "INCORRECT" << std::endl;
	}

	return 0;
}
