#include <stdio.h>
#define gc getchar_unlocked
inline int getn(){
	int n = 0, c = gc();
	while(c < '0' || c > '9') c = gc();
	while(c >= '0' && c <= '9')
		n = (n<<3) + (n<<1) + c - '0', c = gc();
	return n;
}
#define MOD 1000000007
#define MAX 1000001
int fs[MAX];
int main(){
	int T,G, i, n,c,r;
	T = getn();
	fs[1] = 1, fs[2] = 2;
	for(i = 3; i < MAX; i++){
		fs[i] = fs[i-1] + fs[i-2];
		if(fs[i] >= MOD)
			fs[i] -= MOD;
	}
	while(T--){
		n = fs[getn()], G = getn();
		c = n - ((n >> 1) & 033333333333) - ((n >> 2) & 011111111111);
		r = ((c + (c >> 3)) & 030707070707) % 63;
		printf(r==G?"CORRECT\n":"INCORRECT\n");
	}
	return 0;
}
