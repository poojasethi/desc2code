#!/usr/bin/env python

import re

def process(REGEXP, R, C, M):
    M += ' ' + ''.join([M[c::C] for c in xrange(C)])
    if REGEXP.search(M):
       return 'There is a spoon!'
    else:
       return 'There is indeed no spoon!'

def main():
    REGEXP = re.compile(r'spoon', re.I)
    T = int(raw_input())
    for t in xrange(T):
        R, C = map(int, raw_input().strip().split()[:2])
        M = ''
        for r in xrange(R):
            M += raw_input()[:C]
        print process(REGEXP, R, C, M)

main()
