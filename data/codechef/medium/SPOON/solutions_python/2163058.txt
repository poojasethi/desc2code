def doCase():
    RC = raw_input();
    RC = RC.split();
    R = int(RC[0])
    C = int(RC[1])

    spoonFound = False

    rows = []
    for _ in range(0,R):
        row = raw_input().lower()
        rows.append(row)
        
    spoonFound = True in map(lambda x: "spoon" in x, rows)
    
    if not spoonFound:
        #search columns
        for i in range(0,C):
            string = ""
            for row in rows:
                string = string + row[i]
            if "spoon" in string:
                spoonFound = True
                break
    if spoonFound:
        print "There is a spoon!"
    else:
        print "There is indeed no spoon!"
            
        
        

T= input();
for _ in range(0,T):
    doCase();
