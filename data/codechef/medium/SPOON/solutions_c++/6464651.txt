#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#include<math.h>
#define gc getchar_unlocked // for linux
//#define gc getchar        // for windows
using namespace std;

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
void scanlong(long long &x)
{
    register long long c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	int t,r,c,i,j;
	scanint(t);
	while(t--)
	{
		scanint(r);
		scanint(c);
		char matrix[r][c];
		for(i=0;i<r;i++)
		{
			scanf("%s", matrix[i]);
		}
		int flag=0;
		for(i=0;i<r;i++)
		{
			if(flag==0)
			{
			for(j=0;j<c;j++)
			{
				if(flag==0)
				{
				if(matrix[i][j]=='s' || matrix[i][j]=='S')
				{
//					printf("%d %d\n", i, j);
					if(j+5<=c)
					{
//						printf("%d %d\n", i, j);
						if(matrix[i][j+1]=='p' || matrix[i][j+1]=='P')
						{
							if(matrix[i][j+2]=='o' || matrix[i][j+2]=='O')
							{
								if(matrix[i][j+3]=='o' || matrix[i][j+3]=='O')
								{
									if(matrix[i][j+4]=='n' || matrix[i][j+4]=='N')
									{
//										printf("%d %d\n", i, j);
										flag=1;
										break;
									}
								}
							}
						}
					}
					if(i+5<=r)
					{
//						printf("%d %d\n", i, j);
						if(matrix[i+1][j]=='p' || matrix[i+1][j]=='P')
						{
//							printf("%d %d\n", i, j);

//								printf("%d %d\n", i, j);

//									printf("%d %d\n", i, j);

//										printf("%d %d\n", i, j);
										flag=1;
										break;
									}
								}
							}
						}

		}
		if(flag==1)
			printf("There is a spoon!\n");

	}
