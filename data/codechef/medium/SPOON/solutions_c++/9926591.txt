#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,r,c,i,j;

    cin>>t;
    while(t--)
    {
        cin>>r>>c;
        char mat[r][c],ch;
        for(i=0;i<r;++i)
        {
            for(j=0;j<c;++j)
            {
                cin>>ch;
                ch=tolower(ch);
                mat[i][j]=ch;
            }
        }
        int flag=0;
        for(i=0;i<r;++i)
        {
            for(j=0;j<c;++j)
            {
                if(mat[i][j]=='s')
                {
                   if(i<=r-5)
                   {
                       if(mat[i+1][j]=='p' && mat[i+2][j]=='o' && mat[i+3][j]=='o' && mat[i+4][j]=='n')
                       {
                           flag=1;
                           break;
                       }
                   }
                   if(j<=c-5)
                   {
                       if(mat[i][j+1]=='p' && mat[i][j+2]=='o' && mat[i][j+3]=='o' && mat[i][j+4]=='n')
                       {

                           flag=1;
                           break;
                       }
                   }
                }
            }
            if(flag==1)
            break;
        }
        if(flag==1)
        cout<<"There is a spoon!\n";
        else cout<<"There is indeed no spoon!\n";
    }
    return 0;
}
