#include <stdio.h>
#include <string.h>
#include <ctype.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
              int r,c;
              scanf("%d %d\n",&r,&c);
              char arr[r][c];
              char arr1[c][r];
              int i,j;
              for(i=0;i<r;i++)
              {
              for(j=0;j<c;j++)
              {
              scanf("%c",&arr[i][j]);
              arr[i][j]=tolower(arr[i][j]);
              arr1[j][i]=arr[i][j];
              }
              getchar();
              }
              char *pat="spoon";
              int k=0;
              for(i=0;i<r && k<5;i++)
              {
                              k=0;
                         for(j=0;j<c && k<5;j++)
                         {
                                         if(arr[i][j]==pat[k])
                                         k++;
                                         else
                                         k=0;                     
                                         }
                      }
              if(k==5)
              printf("There is a spoon!\n");
              else
              {
                  k=0;
                  for(i=0;i<c && k<5;i++)
              {
                              k=0;
                         for(j=0;j<r && k<5;j++)
                         {
                                         if(arr1[i][j]==pat[k])
                                         k++;
                                         else
                                         k=0;                     
                                         }
                      }
              if(k==5)
              printf("There is a spoon!\n");
              else
              printf("There is indeed no spoon!\n");
              }
              
              }
    return 0;
    }
