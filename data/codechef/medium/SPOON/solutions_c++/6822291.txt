using namespace std;

#define MOD 500009

int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int R,C,i,j,k;
		cin>>R>>C;
		
		vector<string> V;
		string S;
		bool found =false;
		
		for(i=0;i<R;i++)
		{
			cin>>S;
			V.push_back(S);
		}
		
		for(i=0;i<R && !found;i++)
		{
			for(j=0;j<C;j++)
			{
				if(V[i][j]=='S' || V[i][j]=='s') 
				{
					if(j<C-4 && (V[i][j+1]=='p' ||V[i][j+1]=='P')  && (V[i][j+2]=='o' ||V[i][j+2]=='O') && (V[i][j+3]=='O' ||V[i][j+3]=='o') && (V[i][j+4]=='n' ||V[i][j+4]=='N'))
					{
						found=true;
						break;
					}
					if(i<R-4 &&(V[i+1][j]=='p' ||V[i+1][j]=='P')  && (V[i+2][j]=='o' ||V[i+2][j]=='O') && (V[i+3][j]=='O' ||V[i+3][j]=='o') && (V[i+4][j]=='n' ||V[i+4][j]=='N'))
					{
						found=true;
						break;
					}
				}
			}
		}
		
		if(found)
			cout<<"There is a spoon!\n";
		else
			cout<<"There is indeed no spoon!\n";
	}

	return 0;
}
