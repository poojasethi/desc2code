#include <iostream>
#include <string>
#include <cctype>

using namespace std;

char matrix[100][100];
int soln[100][100];

int main() {
	int t, r, c;
	string str;
	cin >> t;
	while(t --) {
		cin >> r >> c;
		for(int i = 0; i < r; i ++) {
                        cin >> str;
			for(int j = 0; j < c; j ++) {
				matrix[i][j] = tolower(str[j]);
				soln[i][j] = 0;
			}
		}
		int i, j;
		for(i = 0; i < r; i ++) {
			for(j = 0; j < c; j ++) {
				switch(matrix[i][j]) {
                                        case 's': soln[i][j] = 1; break;
                                        case 'p': if((i > 0 && matrix[i - 1][j] == 's') || (j > 0 && matrix[i][j - 1] == 's')) {
                                                        soln[i][j] = 2;
                                                  }
                                                  break;
                                        case 'o': if((i > 1 && matrix[i - 1][j] == 'p' && matrix[i - 2][j] == 's') ||
						     (j > 1 && matrix[i][j - 1] == 'p' && matrix[i][j - 2] == 's')){
							soln[i][j] = 3;
						     }
						  else if(
						     (i > 2 && matrix[i - 1][j] == 'o' && matrix[i - 2][j] == 'p' && matrix[i - 3][j] == 's') ||
						     (j > 2 && matrix[i][j - 1] == 'o' && matrix[i][j - 2] == 'p' && matrix[i][j - 3] == 's')) {
						     	  soln[i][j] = 4;
                                                     }
						     break;
					case 'n': if((i > 3 && matrix[i - 1][j] == 'o' && matrix[i - 2][j] == 'o' && matrix[i - 3][j] == 'p' && matrix[i - 4][j] == 's') ||
                                                     (j > 3 && matrix[i][j - 1] == 'o' && matrix[i][j - 2] == 'o' && matrix[i][j - 3] == 'p' && matrix[i][j - 4] == 's')) {
                                                          soln[i][j] = 5;
                             			     }
						     break;
                                }
				if(soln[i][j] == 5) {
					break;
				}
			}
			if(soln[i][j] == 5) {
				break;
			}
		}

		if(i < r && j < c) {
			cout << "There is a spoon!" << endl;
		} else {
			cout << "There is indeed no spoon!" << endl;
		}
	}

	return 0;
}
