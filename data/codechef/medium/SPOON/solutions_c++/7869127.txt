#include<stdio.h>
#include<stdlib.h>
#include<string.h>

char a[210][210];
typedef struct
{
	int row;
	int col;
}pair;
pair p[100000];

int main()
{
	int t,r,co,i,j,k,l,m,flag;
	char c;
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		k=0;
		scanf("%d%d",&r,&co);
		c=getchar();
		for(i=0;i<r;i++)
		{
		    scanf("%s",a[i]);
			for(j=0;j<co;j++)
			{
				if(a[i][j]=='s'|| a[i][j]=='S')
				{
					p[k].row=i;
					p[k++].col=j;
					
				}
			}
		}
		for(i=0;i<k;i++)
		{
			l=p[i].row;
			m= p[i].col;	
			if(m+4<co && (a[l][m+1]=='P' || a[l][m+1]=='p') && (a[l][m+2]=='O'  || a[l][m+2]=='o')&&
				 (a[l][m+3]=='O'  || a[l][m+3]=='o' )&& (a[l][m+4]=='N' || a[l][m+4]=='n'))
			{
				flag=1;
				break;
			}
			else if(l+4<r && (a[l+1][m]=='P' || a[l+1][m]=='p') && (a[l+2][m]=='O'  || a[l+2][m]=='o')&&
				 (a[l+3][m]=='O'  || a[l+3][m]=='o' )&& (a[l+4][m]=='N' || a[l+4][m]=='n'))
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		printf("There is a spoon!\n");
		else
		printf("There is indeed no spoon!\n");
		
	}
}
