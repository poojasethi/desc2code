#include <bits/stdc++.h>
using namespace std;

int main(){
    int t;
    cin >> t;

    while(t--) {
        int r,c,i,j;
        cin >> r >> c;
        char l;
        char s[101][102];
        for(i = 0;i < r;i++) {
            for(j = 0;j < c;j++) {
                cin >> l;
                l = tolower(l);
                s[i][j] = l;
            }
        }

        int flag = 0;
        for(i = 0;i < r && (flag == 0);i++) {
            for(j = 0;j <= c-5 && (flag == 0);j++) {
                   if(s[i][j] == 's' && s[i][j+1] == 'p' && s[i][j+2] == 'o' && s[i][j+3] == 'o' && s[i][j+4] == 'n') {
                            flag = 1;
                            break;
                }
                if(flag == 1)
                    break;
            }
            if(flag == 1)
                    break;
        }

        for(i = 0;i <= r-5;i++) {
            for(j = 0;j < c;j++) {
                   if(s[i][j] == 's' && s[i+1][j] == 'p' && s[i+2][j] == 'o' && s[i+3][j] == 'o' && s[i+4][j] == 'n') {
                            flag = 1;
                            break;
                    }
                    if(flag == 1)
                    break;
            }
            if(flag == 1)
                    break;
        }

        if(flag == 1)
            cout << "There is a spoon!" << "\n";
        else
            cout << "There is indeed no spoon!" << "\n";
    }
    return 0;
}
