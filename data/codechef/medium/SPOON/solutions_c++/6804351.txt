#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main() 
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m;
        cin>>n>>m;
        char mat[101][101];
        for (int i = 1; i <=n; ++i)
        {
            for (int j = 1; j<=m; ++j)
            {
                cin>>mat[i][j];
            }
        }bool harsh=false;
        for (int i = 1; i <=n; ++i)
        {
            for (int j = 1; j<=m-4; ++j)
            {
                if(  (mat[i][j]=='s'||mat[i][j]=='S')  && (mat[i][j+1]=='P'||mat[i][j+1]=='p') &&(mat[i][j+2]=='o'||mat[i][j+2]=='O') &&(mat[i][j+3]=='o'||mat[i][j+3]=='O')&&(mat[i][j+4]=='n'||mat[i][j+4]=='N') )harsh=true;
            }
        }
        for (int i= 1; i<=m; ++i)
        {
            for (int j= 1; j<=n-4; ++j)
            {
                if(  (mat[j][i]=='s'||mat[j][i]=='S')  && (mat[j+1][i]=='P'||mat[j+1][i]=='p') &&(mat[j+2][i]=='o'||mat[j+2][i]=='O') &&(mat[j+3][i]=='o'||mat[j+3][i]=='O')&&(mat[j+4][i]=='n'||mat[j+4][i]=='N') )harsh=true;
         
            }
        }

                
        if(harsh)printf("There is a spoon!\n");
        else printf("There is indeed no spoon!\n");

    }
    return 0;
}
