#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,m,n,i,j,flag=0;
    cin>>t;
    while(t--)
    {
        flag=0;
        cin>>m>>n;
        char arr[m][n];
        for(i=0;i<m;i++)
        {
            cin>>arr[i];
        }
        for(i=0;i<m;i++)
        {
            for(j=0;j<n;j++)
            {
                if((j+4<n)&&((arr[i][j]=='s')||(arr[i][j]=='S'))&&((arr[i][j+1]=='p')||(arr[i][j+1]=='P'))&&((arr[i][j+2]=='o')||(arr[i][j+2]=='O'))&&((arr[i][j+3]=='o')||(arr[i][j+3]=='O'))&&((arr[i][j+4]=='n')||(arr[i][j+4]=='N')))
                {
                    flag=1;
                    break;
                }
                if((i+4<m)&&((arr[i][j]=='s')||(arr[i][j]=='S'))&&((arr[i+1][j]=='p')||(arr[i+1][j]=='P'))&&((arr[i+2][j]=='o')||(arr[i+2][j]=='O'))&&((arr[i+3][j]=='o')||(arr[i+3][j]=='O'))&&((arr[i+4][j]=='n')||(arr[i+4][j]=='N')))
                {
                    flag=1;
                    break;
                }
            }
        }
        if(flag==1)
        {
            cout<<"There is a spoon!"<<endl;
        }
        else
            cout<<"There is indeed no spoon!"<<endl;
    }
    return 0;
}
