#include<cstdio>
#include<iostream>
#include<cstring>

using namespace std;
main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int r,c;
        cin>>r>>c;
        char s[r][c+1];
        for(int i=0;i<r;i++)
        {
            cin>>s[i];
        }
        for(int i=0;i<r;i++)
        {
            for(int j=0;j<c;j++)
            {

                if(s[i][j]=='s' || s[i][j]=='S')
                {
 //                   cout<<"s seen"<<endl;
                    //check col
                    for(int k=i+1;k<(i+5) && k<r;k++)
                    {
                        if(k==(i+1))
                        {
                            if(s[k][j]!='p' && s[k][j]!='P')
                            break;
                        }else if(k==(i+2))
                        {
                            if(s[k][j]!='o' && s[k][j]!='O')
                            break;
                        }else if(k==(i+3) )
                        {
                            if( s[k][j]!='o' && s[k][j]!='O')
                            break;
                        }else if(k==(i+4) )
                        {
                            if(s[k][j]=='n' || s[k][j]=='N')
                            {
                                cout<<"There is a spoon!"<<endl;
                                goto label;
                            }
                        }
                    }
   //                 cout<<"checking row"<<endl;

                    //check row
                    for(int k=j+1;k<(j+5) && k<c;k++)
                    {
                        if(k==(j+1))
                        {
                            if(s[i][k]!='p' && s[i][k]!='P')
                            break;
                        }else if(k==(j+2))
                        {
                            if(s[i][k]!='o' && s[i][k]!='O')
                            break;
                        }else if(k==(j+3) )
                        {
                            if( s[i][k]!='o' && s[i][k]!='O')
                            break;
                        }else if(k==(j+4) )
                        {
                            if(s[i][k]=='n' || s[i][k]=='N')
                            {
                                cout<<"There is a spoon!"<<endl;
                                goto label;
                            }
                        }
                    }
                }
            }
        }
        cout<<"There is indeed no spoon!"<<endl;
        label:
            ;
    }
}
