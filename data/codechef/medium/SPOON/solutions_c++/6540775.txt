#include <bits/stdc++.h>
#include <vector>
#include <cstdio>
#include <map>
#include <set>
#define MOD 1000000007

using namespace std;

int F[1000002] = {0};

int main()
{
	int t, r, c;
	
	scanf("%d", &t);
	while (t--) {
		scanf("%d%d", &r, &c);
		char s[200][200];
		
		for (int i = 0; i < r; i++) {
			scanf("%s", s[i]);
		}
		for (int i = 0; i < r; i++) {
			for (int j = 0; j < c; j++) {
				s[i][j] = tolower(s[i][j]);
			}
		}
		int flag = 0;
		for (int i = 0; i < r; i++) {
			for (int j = 0; j < c; j++) {
				if (s[i][j] == 's') {
					if (j + 4 < c && s[i][j + 1] == 'p' && s[i][j + 2] == 'o' && s[i][j + 3] == 'o' && s[i][j + 4] == 'n') {
						flag = 1;
						break;
					} else {
						if (i + 4 < r && s[i + 1][j] == 'p' && s[i + 2][j] == 'o' && s[i + 3][j] == 'o' && s[i + 4][j] == 'n') {
							flag = 1;
							break;
						}
					}
				}
				
			}
			if (flag) {
				break;
			}
		}
		if (flag) {
			printf("There is a spoon!\n");
		} else {
			printf("There is indeed no spoon!\n");
			
		}
	}
	return 0;
}
