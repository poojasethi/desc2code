#include<bits/stdc++.h>
#define ll long long
using namespace std;

bool chk(char a[102][102],ll r,ll c)
{
	ll i,j;
	for(i=0;i<r;i++)
	{
		for(j=0;j<c-4;j++)
		if(a[i][j]=='s')
		{
			if(j+1<c && a[i][j+1]=='p' && j+2<c && a[i][j+2]=='o' && j+3<c && a[i][j+3]=='o' && j+4<c && a[i][j+4]=='n')
			return true;
		}
	}
	for(i=0;i<c;i++)
	{
		for(j=0;j<r-4;j++)
		if(a[j][i]=='s')
		{
			if(j+1<r && a[j+1][i]=='p' && j+2<r && a[j+2][i]=='o' && j+3<r && a[j+3][i]=='o' && j+4<r && a[j+4][i]=='n')
			return true;
		}
	}
	return false;
}

int main()
{	
	ll t,r,c,i,j;
	cin>>t;
	while(t--)
	{
		cin>>r>>c;
		char a[102][102];
		for(i=0;i<r;i++)
		for(j=0;j<c;j++)
		{
			cin>>a[i][j];
			a[i][j]=tolower(a[i][j]);
		}
		if(chk(a,r,c))
		cout<<"There is a spoon!\n";
		else
		cout<<"There is indeed no spoon!\n";
	}
	return 0;
}
