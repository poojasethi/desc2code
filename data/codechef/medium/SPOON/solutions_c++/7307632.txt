#include<bits/stdc++.h>
using namespace std;
char matrix[105][105];
int main()
{
	int t,r,c;
	cin>>t;
	while(t--)
	{
		cin>>r>>c;
		for(int i=0;i<r;i++)
		{
			for(int j=0;j<c;j++)
			{
				cin>>matrix[i][j];
			}
		}
		bool flag=0;
		char s1[6],s2[6];
		strcpy(s1,"     ");
		strcpy(s2,"     ");
		for(int i=0;i<r;i++)
		{
			for(int j=0;j<c;j++)
			{
		 if(matrix[i][j]=='s'||matrix[i][j]=='S');
		 {
		 	int k=0;
		 	while(k<=4)
		 	{
		 		if(i<=r-5)
		 		s1[k]=matrix[i+k][j];
		 		if(j<=c-5)
		 		s2[k]=matrix[i][j+k];
		 		k++;
		 	}
		 	s1[5]='\0';
		 	s2[5]='\0';
		 	if(strcasecmp(s1,"spoon")==0||strcasecmp(s2,"spoon")==0)
		 	{
		 		flag=1;
		 		break;
		 	}
		 }   
		}
		if(flag==1)
		break;
	}
		if(flag==1)
		cout<<"There is a spoon!"<<endl;
		else
		cout<<"There is indeed no spoon!"<<endl;
	}
}
