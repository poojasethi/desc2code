#include<cstdio>
#define INF 1000000000
using namespace std;
typedef long long int ll;
char str[]="spoon";
char mat[106][106];
int R,C;
bool check_col(int x,int y){
    //column constant
    if(R<x+5){
        return false;
    }
    for(int i=x;i<x+5;i++){
        if(str[i-x]!=mat[i][y] && (str[i-x]-32)!=mat[i][y]){
            return false;
        }
    }
    return true;
}
bool check_row(int x,int y){
    //column constant
    if(C<y+5){
        return false;
    }
    for(int i=y;i<y+5;i++){
        if(str[i-y]!=mat[x][i] && (str[i-y]-32)!=mat[x][i]){
            return false;
        }
    }
    return true;
}
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d",&R,&C);
        for(int i=0;i<R;i++){
            scanf("%s",&mat[i]);
        }
        int found=0;
        for(int i=0;i<R && !found;i++){
            for(int j=0;j<C && !found;j++){
                if(check_col(i,j) || check_row(i,j)){
                 found=1;   
                }
            }
        }
        if(found){
            printf("There is a spoon!\n");
        }
        else
        {
            printf("There is indeed no spoon!\n");
        }
    }
}
