#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>

char s[105];
int CC[105];

int main()
{
  int T;
  scanf("%d\n", &T);
  while(T--)
  {
    int R, C;
    scanf("%d %d\n", &R, &C);
    bool isSpoon = false;

    memset(CC, 0, sizeof(CC));

    while(!isSpoon && (R--))
    {
      gets(s);
      int *cc = CC;
      int rr = 0;
      for(char *p = s; *p != '\0'; ++p, ++cc)
      {
        if(*p == 's' || *p == 'S')
        {
          *cc = 1;
          rr = 1;
        }
        else if(*p == 'p' || *p == 'P')
        {
          if(*cc == 1) ++(*cc); else *cc = 0;
          if(rr == 1) ++(rr); else rr = 0;
        }
        else if(*p == 'o' || *p == 'O')
        {
          if((*cc == 2) || (*cc == 3)) ++(*cc); else *cc = 0;
          if((rr == 2) || (rr == 3)) ++(rr); else rr = 0;
        }
        else if(*p == 'n' || *p == 'N')
        {
          if(*cc == 4) { isSpoon = true; break; } else *cc = 0;
          if(rr == 4) { isSpoon = true; break; } else rr = 0;
        }
        else
        {
          *cc = 0;
          rr = 0;
        }
      }
    }

    if(R > 0)
    {
      while(R--) // No check - just read the rest of the data
        gets(s);
    }

    printf(isSpoon ? "There is a spoon!\n" : "There is indeed no spoon!\n");
  }

  return 0;
}
