#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
#define mod 1000000007
using namespace std;
int main()
{
    int i,t,r,c,j,flag;
    char ch;
    scanf("%d",&t);
    while(t--)
    {
        flag=0;
        scanf("%d%d",&r,&c);
        scanf("%c",&ch);
        char v[r+1][c+1];
        for(i=0;i<r;i++)
        {
            for(j=0;j<c;j++)
            {
                scanf("%c",&(v[i][j]));
                if((v[i][j]>=65)&&(v[i][j]<=90))
                    v[i][j]+=32;
            }
            scanf("%c",&ch);
        }
        if(c>=5)
      {
        for(i=0;i<r;i++)
        {
            for(j=0;j<=c-5;j++)
            {
                if((v[i][j]=='s')&&(v[i][j+1]=='p')&&(v[i][j+2]=='o')&&(v[i][j+3]=='o')&&(v[i][j+4]=='n'))
                {
                    flag=1;
                    break;
                }
            }
            if(flag)
                break;
        }
      }
        if(!flag)
        {
          if(r>=5)
          {
            for(i=0;i<c;i++)
            {
                for(j=0;j<=r-5;j++)
                {
                    if((v[j][i]=='s')&&(v[j+1][i]=='p')&&(v[j+2][i]=='o')&&(v[j+3][i]=='o')&&(v[j+4][i]=='n'))
                    {
                        flag=1;
                        break;
                    }
                }
                if(flag)
                    break;
            }
           }
        }
        if(flag)
            printf("There is a spoon!\n");
        else
            printf("There is indeed no spoon!\n");
    }
    return 0;
}
