#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,R,C,i,j;
	scanf("%d",&t);
	while(t--)
	{
		char *ar[101];
		scanf("%d %d",&R,&C);
        for(i=0;i<R;i++)
        {
        	ar[i]=(char*)malloc((C+1)*sizeof(char));
        	scanf("%s",ar[i]);
        }
        if(R<5 && C<5)
        	{printf("There is indeed no spoon!\n");continue;}
        char temp[101];
        int flag=0;

        for(i=0;i<R;i++)
        {
        	for(j=0;j<C;j++)
        	{
        		if(ar[i][j]>=65 && ar[i][j]<=90)
        		   temp[j]=ar[i][j]+32;
        		else
        			temp[j]=ar[i][j];
        	}
          temp[j]='\0';
          //printf("%s\n",temp);
          if(strstr(temp,"spoon")!=NULL)
          {
          	printf("There is a spoon!\n");flag=1;break;
          }
        }
        if (flag==1) continue;
        
        for(i=0;i<C;i++)
        {
        	for(j=0;j<R;j++)
        	{
        		if(ar[j][i]>=65 && ar[j][i]<=90)
        		   temp[j]=ar[j][i]+32;
        		else
        			temp[j]=ar[j][i];
        	}
          temp[j]='\0';
          //printf("%s\n",temp);
          if(strstr(temp,"spoon")!=NULL)
          {
          	printf("There is a spoon!\n");flag=1;break;
          }
        }

        if(flag==1)
        	continue;
        else
        	printf("There is indeed no spoon!\n");
	}
	return 0;
}
