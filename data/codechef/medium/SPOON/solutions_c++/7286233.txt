#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int t,r,c,flag=0,i,j;
	scanf("%d",&t);
	char a[101][101];
	while(t--)
	{
		flag=0;
		scanf("%d%d",&r,&c);
		 gets(a[0]);
    	for(i=0;i<r;i++)
    	{
    		gets(a[i]);
    	}
		for(i=0;i<r;i++)
		{
			for(j=0;j<c;j++)
			{
				if(a[i][j]=='s'||a[i][j]=='S')
				{
					if(i+4<r)
					{
						if((a[i+1][j]=='p'||a[i+1][j]=='P')&&(a[i+2][j]=='o'||a[i+2][j]=='O')&&(a[i+3][j]=='o'||a[i+3][j]=='O')&&(a[i+4][j]=='n'||a[i+4][j]=='N'))
						{
							flag=1;
							break;
						}
					}
					if(j+4<c)
					{
						if((a[i][j+1]=='p'||a[i][j+1]=='P')&&(a[i][j+2]=='o'||a[i][j+2]=='O')&&(a[i][j+3]=='o'||a[i][j+3]=='O')&&(a[i][j+4]=='n'||a[i][j+4]=='N'))
						{
							flag=1;
							break;
						}
					}
				}
			}
			if(flag==1)
			{
				break;
			}
		}
		if(flag==1)
		{
			printf("There is a spoon!\n");
		}
		else
		{
			printf("There is indeed no spoon!\n");
		}
	}
	return 0;
}
