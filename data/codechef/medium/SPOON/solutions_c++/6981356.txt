#include <iostream>
#include <cctype>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t --) {
        int row, col;
        cin >> row >> col;
        char ch;
        char mat[row + 1][col + 1];
        for(int i = 0; i < row; i++)
            for(int j = 0 ; j < col ; j++){
               cin >> ch;
               mat[i][j] = tolower(ch);
            }
        int flag = 0;
        for(int i = 0 ; i < row; i ++) {
            for(int j = 0; j < col ; j ++) {
                if(mat[i][j] == 's') {
                    if( i + 4 < row && mat[i+1][j] == 'p' && mat[i+2][j] == 'o' && mat[i+3][j] == 'o' && mat[i+4][j] == 'n')
                        flag = 1;
                    else if (j + 4 < col && mat[i][j+1] == 'p' && mat[i][j+2] == 'o' && mat[i][j+3] == 'o' && mat[i][j+4] == 'n')
                        flag = 1;
                }
                if(flag == 1)
                    break;
            }
        }
      if(flag)
         cout << "There is a spoon!" << endl;
      else
        cout << "There is indeed no spoon!" << endl;

    }
 return 0;
}
