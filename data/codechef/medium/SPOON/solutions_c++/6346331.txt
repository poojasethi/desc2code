#include <iostream>
using namespace std;
#include<cctype>
#include<cstdio>
int main() 
{
	short t;
	cin>>t;
	while(t-->0)
	{
		short r,c,i,j;
		cin>>r>>c;
		char str[r][c];
		for(i=0;i<r;i++)
		{
			for(j=0;j<c;j++)
			{
				cin>>str[i][j];
				str[i][j]=toupper(str[i][j]);
			}
		}
		bool f=false;
		for(i=0;i<r;i++)
		{
			for(j=0;j+4<c;j++)
			{
				if(str[i][j]=='S'&&str[i][j+1]=='P'&&str[i][j+2]=='O'&&str[i][j+3]=='O'&&str[i][j+4]=='N')
				 f=true;
			}
		}
		for(i=0;i+4<r;i++)
		{
			for(j=0;j<c;j++)
			{
				if(str[i][j]=='S'&&str[i+1][j]=='P'&&str[i+2][j]=='O'&&str[i+3][j]=='O'&&str[i+4][j]=='N')
				 f=true;
			}
		}
		if(f) printf("There is a spoon!\n");
		else printf("There is indeed no spoon!\n");
	}
	// your code goes here
	return 0;
}
