#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int r,c,i,j,flag=0,k;
		scanf("%d %d",&r,&c);
		char a[r][c];
		for(i=0;i<r;i++)
		{
			scanf("%s",&a[i]);
		}
		for(i=0;i<r;i++)
		{
			for(j=0;j<c;j++)
			{
				if((a[i][j]=='S' || a[i][j]=='s') && j<=c-5)
				{
					if(a[i][j+1]=='P' || a[i][j+1]=='p')
					if(a[i][j+2]=='O' || a[i][j+2]=='o')
					if(a[i][j+3]=='O' || a[i][j+3]=='o')
					if(a[i][j+4]=='N' || a[i][j+4]=='n')
					{
						flag=1;
						break;
					}
				}
				if((a[i][j]=='S' || a[i][j]=='s') && i<=r-5)
				{
					if(a[i+1][j]=='P' || a[i+1][j]=='p')
					if(a[i+2][j]=='O' || a[i+2][j]=='o')
					if(a[i+3][j]=='O' || a[i+3][j]=='o')
					if(a[i+4][j]=='N' || a[i+4][j]=='n')
					{
						flag=1;
						break;
					}
				}
			}
		}
		if(flag)
		printf("There is a spoon!\n");
		else
		printf("There is indeed no spoon!\n");
	}
return 0;
}
