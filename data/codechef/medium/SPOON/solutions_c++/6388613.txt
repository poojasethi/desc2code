#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s[100][101];
		int n,m;
		bool yes=false;
		cin>>n>>m;
		for(int i=0;i<n;i++)
		{
			scanf("%s",s[i]);
			for(int j=0;j<m;j++)
				s[i][j]=tolower(s[i][j]);
		}
		for(int i=0;i<=n-5 && !yes;i++)
			for(int j=0;j<m && !yes;j++)
				if(s[i][j]=='s' && s[i+1][j]=='p' && s[i+2][j]=='o' && s[i+3][j]=='o' && s[i+4][j]=='n')
					yes=true;
		for(int i=0;i<n && !yes;i++)
			for(int j=0;j<=m-5 && !yes;j++)
				if(s[i][j]=='s' && s[i][j+1]=='p' && s[i][j+2]=='o' && s[i][j+3]=='o' && s[i][j+4]=='n')
					yes=true;
		if(yes)
		{
			cout<<"There is a spoon!"<<endl; 
		}
		else
		{
			cout<<"There is indeed no spoon!"<<endl;
		}

		
	}
	return 0;
}
