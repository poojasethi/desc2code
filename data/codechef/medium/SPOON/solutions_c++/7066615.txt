#include<bits/stdc++.h>
using namespace std;

string mat[1010];

int main()

{
	int T;
	cin>>T;
	
	while(T--)
	{
		int R, C, i, j, flag=0;
		cin>>R>>C;
		//const int slen=5;
		//string mat[R];
		
		for(i=0; i<R; i++)
		cin>>mat[i];
		
		for(i=0; i<R; i++)
		{
			for(j=0; j<C; j++)
			{
				if(mat[i][j]=='S' || mat[i][j]=='s')
				{
					if(j<=C-5 || i<=R-5)
					{
						if((mat[i][j+1]=='P' || mat[i][j+1]=='p')&&(mat[i][j+2]=='O' || mat[i][j+2]=='o')&&(mat[i][j+3]=='O' || mat[i][j+3]=='o')&&(mat[i][j+4]=='N' || mat[i][j+4]=='n'))
						{
							flag=1;
							break;
						}
						
						if((mat[i+1][j]=='P' || mat[i+1][j]=='p')&&(mat[i+2][j]=='O' || mat[i+2][j]=='o')&&(mat[i+3][j]=='O' || mat[i+3][j]=='o')&&(mat[i+4][j]=='N' || mat[i+4][j]=='n'))
						{
							flag=1;
							break;
						}
					}
				}
			}
		}
		
		if(flag==1)
		cout<<"There is a spoon!";
		
		else
		cout<<"There is indeed no spoon!";
		
		cout<<"\n";
	}
	return 0;
}
