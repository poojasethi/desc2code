#include<cstdio>
#include<iostream>
using namespace std;
#define p 1000000007
int gcd(int a, int b) {
	int r = b % a;
	while(r) {
		b = a;
		a = r;
		r = b % a;
	}
	return a;
}


int main()
{
    int t,n,i,j,l;
    long long int g;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        long long int a[n+1],visited[n+1],k=0,res=1,cycle[n+1];

        for(i=1;i<=n;i++)
            {
                scanf("%lld",&a[i]);
                visited[i]=0;
            }
        for(i=1;i<=n;i++)
        {
            l=1;
            if(visited[i]==0)
            {
                {
                visited[i]=1;
                j=a[i];
                if(j==i)
                    continue;


              }

            while(j!=i)
            {
                l++;
                visited[j]=1;
                //cout<<"kkk";
                j=a[j];
            }
           // cout<<"llll";
            if(l>1)
             cycle[++k]=l;



            }


        }

         //cout<<"bieee";
        if(k==0)
            printf("1\n");
        else if(k==1)
            printf("%lld\n",cycle[1]%p);
        else
        {
            for(i=1;i<k;i++)
                {
                    for(j=i+1;j<=k;j++)
                    {
                        g=gcd(cycle[i],cycle[j]);
                        cycle[j]=cycle[j]/g;

                    }


                }
            for(i=1;i<=k;i++)
            {
                res*=cycle[i];
                res=res%p;


            }
            printf("%lld\n",res%p);

        }




    }
}
