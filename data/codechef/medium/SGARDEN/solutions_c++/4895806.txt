#include<iostream>
#include<cstring>
#include<stdio.h>
#include<algorithm>
#include<cmath>
#include<vector>
#include<queue>
#include<stack>
#include<map>
#include<set>
#define mp make_pair
#define pb push_back
#define MAX(a,b) ((a)>(b)?(a):(b))
#define MIN(a,b) ((a)<(b)?(a):(b))
#define F first
#define S second
#define ll long long
#define pp pair<int,int>
#define INF 2000000000
#define P 1000000007ll
using namespace std;
const int N=100005;
int a[N],i,j,T,n,x,y,op[N],f[N],lp[N],cnt;
vector<int> v,pr;
ll ans;

int main()
{n=100000;
 for(i=2;i<=n;i++){
    if(!lp[i])lp[i]=i,pr.pb(i);
    for(j=0;j<pr.size() && pr[j]<=lp[i] && i*pr[j]<=n;j++)
        lp[i*pr[j]]=pr[j];
 }
 scanf("%d",&T);
 while(T--){
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%d",&a[i]);
    for(i=1;i<=n;i++)
    if(!f[i]){
        cnt=0;
        x=i;
        while(!f[x]){
            f[x]=1;
            cnt++;
            x=a[x];
        }
        x=cnt;
        while(x>1){
            y=lp[x];
            cnt=0;
            while(x>1 && lp[x]==y){
                  x/=y;
                  cnt++;
            }
            op[y]=MAX(op[y],cnt);
        }
    }
    ans=1;
    for(i=2;i<=n;i++)
        for(j=1;j<=op[i];j++)
        ans=(ans*i)%P;
    printf("%d\n",ans);

    for(i=1;i<=n;i++)f[i]=op[i]=0;
    v.clear();
 }


 return 0;
}
