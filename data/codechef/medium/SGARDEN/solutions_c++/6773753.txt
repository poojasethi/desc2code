int main() {
	long long t,n;
	scanf("%lld",&t);
	while(t--) {
		scanf("%lld",&n);
		long long a[n+1],s,k=0,length[n+1],pos[n+1],visited[n+1],len=1;
		for(long long i=0;i<=n;i++) {
			visited[i]=0;
		}
		for(long long i=1;i<=n;i++) {
			scanf("%lld",&pos[i]);	
		}
		for(long long i=1;i<=n;i++) {
			if(visited[i]==0) {
				length[k++]=len;
				len=1;
				s=pos[i];
				visited[i]=1;
				while(s!=i) {
					len++;
					visited[s]=1;
					s=pos[s];
				}
			}
			if(i==n && length[k-1]!=len){
				length[k++]=len;
			}
		}
		long long ans=1;
		//find lcm of lengths
		for(int i=0;i<k;i++) {
			if(length[i]!=1) {
			for(int j=i+1;j<k;j++) {
				length[j]/=__gcd(length[j],length[i]);
			}
		}
			ans=(ans*length[i])%mod;
		}
		printf("%lld\n",ans);
}
	return 0;
}
