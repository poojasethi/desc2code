#include<iostream>
#include<cstdio>
using namespace std;
#define MOD 1000000007
bool prime[100005]= {0},vis[100005];
int pow[100005],a[100005];
int main()
{
    int i,j,c,s,t,n,l;
    long long int res;
    prime[1]=1;
    for(i=2; i<=317; i++)
    {
        if(!prime[i])
        {
            for(j=i*i; j<=100001; j+=i)
            {
                prime[j]=1;
            }
        }
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=1;i<=n+2;i++)
        {
            pow[i]=0;vis[i]=0;
        }
        for(i=1;i<=n;i++)
            scanf("%d",&a[i]);
        for(i=1;i<=n;i++)
        {
            if(vis[i]==0)
            {
                l=1;
                s=a[i];
                vis[i]=1;
                while(s!=i)
                {
                    vis[s]=1;
                    l++;
                    s=a[s];
                }
                if(prime[l]!=0)
                {
                    for(j=2;l>1;j++)
                    {
                        c=0;
                        if(prime[j]==0)
                        {
                            while(l%j==0)
                            {
                                c++;
                                l/=j;
                            }
                            if(pow[j]<c)
                                pow[j]=c;
                        }
                    }
                }
                else if(pow[l]==0)
                {
                    pow[l]=1;
                }
            }
        }
        res=1;
        for(i=2;i<=n;i++)
        {
            for(j=1;j<=pow[i];j++)
            {
                res=(res*i)%MOD;
            }
        }
        printf("%lld\n",res);
    }
    return 0;
}
