#include<bits/stdc++.h>
using namespace std;

#define sd(x) scanf("%d",&x)
#define ss(x) scanf("%s",x)
#define lld long long int
#define slld(x) scanf("%lld",&x)

lld power(lld a, lld b,lld m)
{
    lld ans=1;
    while(b!=0)
    {
        if(b%2==1)
        ans=(a*ans)%m;
        a=(a*a)%m;
        b=b/2;
    }
    return ans;


}

int main()
{
    lld t,n,i,w[100010],a[100010],visited[100010],ans[100010],x,m=1000000007,c,ansre,j,k,l;
	slld(t);
	while(t--)
	{
        slld(n);
        c=0;
        ansre=1;
        fill(a,a+n+2,1);
        fill(visited,visited+n+2,0);
        for(i=1;i<=n;i++)
            slld(w[i]);
        for(i=1;i<=n;i++)
        {
        if(visited[i]==0){
            x=w[i];
            visited[i]=1;

            while(x!=i)
            {
                visited[x]=1;
                x=w[x];
                a[c]++;
            }
            c++;
            }

        }



        fill(ans,ans+n,0);
        for(j=0;j<c;j++)
        {
            k=a[j];
            for(i=2;i*i<=k;i++)
            {
                l=0;
                while(k%i==0)
                {
                    k=k/i;
                    l++;
                }
                ans[i]=max(l,ans[i]);
            }
            if(k!=1)
            ans[k]=max(ans[k],1LL);
        }

       for(i=0;i<=n;i++)
       {
            if(ans[i]!=0)
            {
                ansre=(ansre*power(i,ans[i],m))%m;
            }
       }
        printf("%lld\n",ansre%m);
	}

	return 0;
}
