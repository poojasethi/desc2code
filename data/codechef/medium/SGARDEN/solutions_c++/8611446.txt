inline void scan(string &ret){
    register int a=getchar_unlocked();
    ret="";
    while(a==' '||a=='\n'||a==EOF)
        a=getchar_unlocked();
    while(a!=' '&&a!='\n'&&a!=EOF){
        ret+=(char)a;
        a=getchar_unlocked();
    }
}
 
 
void solve_test(){
    int n = scan_d();
    vector<int> p(n);
    for(int i=0;i<n;i++){
        p[i] = scan_d() - 1;
    }
    map<int, int> maxpw;
    vector<int> u(n);
    for(int i=0;i<n;i++){
        if(u[i])
            continue;
        int leng = 0;
        for(int j = i; !u[j]; j = p[j]){
            leng++;
            u[j] = true;
        }
//        cerr << leng << endl;
        for(int d = 2; d*d <= leng; d++){
            int st = 0;
            while(leng % d == 0){
                st++;
                leng /= d;
            }
            if(st){
                maxpw[d] = max(maxpw[d], st);
            }
        }
        if(leng != 1)
            maxpw[leng] = max(maxpw[leng], 1);
    }
    int res = 1;
    for(map<int, int>::iterator it = maxpw.begin(); it != maxpw.end(); it++){
        for(int i=0;i< it->second;i++){
            res = (res * 1LL * it->first) % MOD;
        }
    }
    printf("%d\n", res);
}
 
int main(){
 
    int T = scan_d();
    while(T--){
        solve_test();
    }
    return 0;
}
