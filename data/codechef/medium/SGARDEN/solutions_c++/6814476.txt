#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;
#define M 1000000007
long long gcd(long long a,long long b)
{
    if(b==0)
        return a;
    return(gcd(b,a%b));
}
int main()
{
    int i,j,n,t,countm,hcf,temp,res;
    long long ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        vector<int> v(n);
        vector<long long> flag(n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&(v[i]));
        }
        for(i=0;i<n;i++)
        {
            if(flag[i]==0)
            {
                map<int,int> m;
                map<int,int>::iterator it;
                countm=1;
                m[i]=i;
                temp=v[i]-1;
                while(temp!=i)
                {
                    m[temp]=temp;
                    countm++;
                    temp=v[temp]-1;
                }
                for(it=m.begin();it!=m.end();it++)
                {
                    flag[it->second]=countm;
                }
            }
        }
      /*  for(i=0;i<n;i++)
        {
            printf("%d\t",flag[i]);
        }
        printf("\n");*/
        ans=1;
        for(i=0;i<n;i++)
        {
            if(flag[i]!=1)
            {
                for(int k=i+1;k<n;k++)
                flag[k]/=gcd(flag[k],flag[i]);
            }
            ans=(ans*flag[i])%M;
        }
        res=ans;
        printf("%d\n",res);
    }
    return 0;
}
