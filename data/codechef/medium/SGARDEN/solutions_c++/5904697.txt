#include <iostream>
#include <algorithm>
#include <string.h>
#include <stdio.h>
#include <vector>
#include <math.h>
using namespace std;
#define M 1000000007
typedef long long ll;
ll vis[100005];
ll a[100005];
ll cy[100005];

int main(){
    ll t,n,j,ans;
    cin >> t;

    while(t--){
            ans = 1;
            memset(vis,0,sizeof(vis));
            memset(cy,0,sizeof(cy));
        cin >> n;
        for(ll i=1; i<=n; i++){
            cin >> a[i];
        }
        ll length=0;
        ll s=0;
        ll k = 0;
        for(ll i=1; i<=n; i++){
                s = 0;
                j = i;
            while(!vis[j]){
                s++;
                vis[j] = 1;
               j = a[j];
            }
        if(s!=0){
            ll power = 0;
            ll num = s;
            for(ll b=2; b<=num; b++){
                power = 0;
                while(s%b == 0){
                    power++;
                    s /= b;
            }
            cy[b] = max(cy[b],power);
        }
        }
        }
        for(ll i=2; i<=n; i++){
            for(j=1; j<=cy[i]; j++){
                ans = (ans*i)%M;
            }
        }
        cout << ans << "\n";

    }

    return 0;
}
