#include <stdio.h>
int main()
{
     int T,A,B;
     scanf("%d\n",&T);
     while(T>0)
     {
        scanf("%d %d\n",&A,&B);
         if(A%2 !=0 && B%2!=0)
         {
            printf("Vanka\n");
         }
         else
         {
            printf("Tuzik\n");
         }
       T--;
     }
 return 0;
}
