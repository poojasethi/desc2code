#include<bits/stdc++.h>
using namespace std;
char c[2];
int main()
{
    int i,j,k;
    int T,t,a;
    int sum=0;
    int n;

    while(scanf("%d",&n)!=EOF)
    {
        if(n==0)
            exit(0);
        int mc=0;
        int b=0;
        for(i=0;i<n;i++)
        {
            scanf("%s %d",&c,&a);
            if(c[0]=='P')
            {
                b=a-1;
            }
            else if(c[0]=='M')
            {
                b=a+1;
            }
            b=abs(b);
            if(b>=mc)
                mc=max(mc,b);
            else
            {
                if(b%2==mc%2)
                    mc=max(mc,b);
                else
                    mc=max(mc,b)+1;
            }

        }
        printf("%d\n",mc);
    }
}
