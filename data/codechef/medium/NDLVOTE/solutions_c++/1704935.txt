#include <iostream>
#include <cstdio>

using namespace std;

inline int my_abs(int x)
{
    int s = x >> 31;
    return (x ^ s) - s;
}

int n;

int main()
{
    cin.sync_with_stdio(0);

    while (true)
    {
        int Ans = -1;

        cin >> n;
        if (n == 0) return 0;

        while (n--)
        {
            char vote;
            int num;
            cin >> vote >> num;

            if (vote == 'P') num--;
            else num++;

            num = my_abs(num);
            Ans = max(Ans, num);

            if (Ans > num && (Ans - num) % 2 == 1) Ans++;
        }

        printf("%d\n", Ans);
    }
}
