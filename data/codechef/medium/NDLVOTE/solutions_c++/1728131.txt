#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int Get()
{
  char c;
  while(c=getchar(),c!='P'&&c!='M');
  int X;
  scanf("%d",&X);
  if(c=='P')
    return X-1;
  else
    return X+1;
}

int main()
{
  int N;
  while(scanf("%d",&N),N)
    {
      int Ans=0;
      while(N--)
        {
          int X=abs(Get());
          if(X>Ans)
            Ans=X;
          else if((X^Ans)&1)
            Ans++;
        }
      printf("%d\n",Ans);
    }
  return 0;
}
