#include<iostream>
#include<cstdio>
using namespace std;
int abs(int n)
{
    if(n<0)
    return (-n);
    else
    return n;
}
int main()
{
    while(1)
    {
        int n,max=0;
        scanf("%d",&n);
        if(n==0)
        break;
        while(n--)
        {
            getchar();
            char c;
            int m;
            c=getchar();
            scanf("%d",&m);
            if(c=='M')
            m+=1;
            if(c=='P')
            m-=1;
            if(abs(m)>max)
            max=abs(m);
            else if((m%2) && !(max%2) || !(m%2) && (max%2))
            max+=1;
        }
        cout<<max<<endl;
    }
    return 0;
}
