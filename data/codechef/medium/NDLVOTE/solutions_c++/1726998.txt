#include<iostream>
#include<cstdio>
using namespace std;

int N;
int v;
int user;
int s;


char c[2];
int abs(int a){return a>0?a:-a;}
int main()
{


    while(1)
    {
        user = 0;

        scanf("%d", &N);
        if(N== 0)break;
        for(int i=0; i<N; i++)
        {

            scanf("%s", c);

            scanf("%d", &v);

            if(c[0] == 'P')s = abs(v-1);
            else s = abs(v+1);

            if(s>=user)user += (s-user);
            else
            {
              if((user - s)%2 == 1)user++;
            }

        }
        printf("%d\n", user);
    }
}
