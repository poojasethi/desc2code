#include <cstdio>
#include <cstdlib>
int main(int argc, char **argv)
{
	while(1){
		int N;
		long long users = 0;
		scanf("%d",&N);
		if(!N)	break;
		while(N--){
			char po;
			long long score;
			scanf(" %c %lld",&po,&score);
			if(po == 'P' )	score--;
			else 	score++;
			score = abs (score);
			if(score>users)	
				users = score;
			else if( (users - score)%2)	
				users++;
		}
		printf("%lld\n",users);
	}
	return 0;
}
