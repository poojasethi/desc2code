#include<cstdio>
#include<iostream>
using namespace std;

#define DEBUG(x) cout<<#x<<"="<<x<<endl

inline int mod(int n) {
    return n>0 ? n:-n;
}

int main() {
    int n;
    while(scanf("%d", &n) && n) {
        char S[1000];
        int P[1000];

        for(int i=0; i<n; i++) {
            scanf(" %c %d", S+i, P+i);
            if(S[i]=='P') P[i]--;
            else P[i]++;
        }

        int ans=mod(P[0]);
        for(int i=1; i<n; i++) {
            if(mod(P[i])>=ans)
                ans=mod(P[i]);
            else if((ans-mod(P[i]))&1)
                    ans++;
//            DEBUG(ans);
        }

        printf("%d\n", ans);

    }
}
