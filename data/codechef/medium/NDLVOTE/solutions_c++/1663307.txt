#include <stdio.h>
using namespace std;

int main()
{
  int n, s, c; 
  char vote;
  while (scanf("%d",&n) && n>0) {
    c=0;
    while (n--) {
      scanf(" %c %d", &vote, &s);
      s += (vote=='P') ? -1 : +1;
      s *= (s<0) ? -1 : 1;
      if(s>c) c += (s-c);
      else c += (c-s)&1;
    }
    printf("%d\n", c);
  }
}
