#include<iostream>
#include<cmath>

using namespace std;

int main()
{
  int n;
  cin>>n;
  while(n)
  {
    char a;
    int x,users=0;
    while(n--)
    { 
      cin>>a>>x;
      if(a=='P')
	x--;
      else
	x++;
      x=abs(double(x));
      if(x>users)
	users=x;
      else
      {
	if((users-x)%2!=0)
	  users++;
      }
    }
    cout<<users<<"\n";
    cin>>n;
  }
  return 0;
}
