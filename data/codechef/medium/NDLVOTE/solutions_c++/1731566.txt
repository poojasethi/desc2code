#include <cstdio>

int main()
{
#ifndef ONLINE_JUDGE
    freopen("in", "r", stdin);
#endif
    int n, pv, nv, v, d;
    char s[8];
    scanf("%d", &n);
    while (n)
    {
        pv = nv = 0;
        while (n--)
        {
            scanf("%s %d", s, &v);
            if (s[0] == 'P')
                v--;
            else
                v++;
            d = v + nv - pv;
            if (d >= 0)
            {
                d = (d + 1)/2;
                pv += d;
                if (pv < v)
                    pv = v;
                nv = pv - v;
            }
            else
            {
                d = (1 - d)/2;
                nv += d;
                if (nv < -v)
                    nv = -v;
                pv = nv + v;
            }
        }
        printf("%d\n", pv + nv);
        scanf("%d", &n);
    }
    return 0;
}
