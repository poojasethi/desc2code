#include<iostream>
#include<cstdlib>

using namespace std;

int main()
{
    int currNoUsers,timesPoVote,currPoVote,score;
    char ch;
    while(1)
    {
        cin >> timesPoVote;
        currNoUsers = 0;
        currPoVote = 0;

        if(timesPoVote == 0)
            return 0;

        while(timesPoVote--)
        {
            cin >> ch >> score;

            if(ch == 'P')
            {
                score = score - 1;
            }

            if(ch == 'M')
            {
                score = score + 1;
            }

            score = abs(score);
            if(score - currNoUsers > 0)
            {
                currNoUsers += score - currNoUsers;
            }
            else if(currNoUsers - score > 0)
            {
                if( (currNoUsers - score)%2 != 0 )
                        currNoUsers++;
            }
        }

        cout << currNoUsers << endl;
    }
}
