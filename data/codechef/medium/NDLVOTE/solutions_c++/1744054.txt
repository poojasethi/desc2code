#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int n,max;
    while(1)
    {
        max=0;
        scanf("%d",&n);
        if(n==0)
        break;
        while(n--)
        {
            getchar();
            char c;
            int m;
            c=getchar();
            scanf("%d",&m);
            if(c=='M')
            m+=1;
            if(c=='P')
            m-=1;
            if(m<0)
            m=-m;
            if(m>max)
            max=m;
            else if((max-m)%2!=0)//if a persons changes his vote then total count of vote changes by 2;
            max+=1;
        }
        cout<<max<<endl;
    }
    return 0;
}
