int f (char ch, int x, int sum)
{
	if (ch == 'P') {
		x--;
	} else {
		x++;
	}
	if (sum == -1) {
		return abs(x);
	}
	x = abs(x);
	if (x > sum) {
		return x;
	}
	if (sum%2 ==  x%2) {
		return sum;
	}
	
	return sum+1;
}
