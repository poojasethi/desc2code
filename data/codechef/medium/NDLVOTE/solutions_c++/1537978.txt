#include<stdio.h>
#include<iostream>
#include<math.h>
#include<string.h>

#include<vector>
#include<algorithm>

#define in(i) scanf("%d",&i);
#define lin(i) scanf("%lld",&i);

using namespace std;



int main(){
  long long ans,c;
  int n;
  char v;
  in(n);
  
  while(n){
    ans=0;
    while(n--){
      getchar();
      scanf("%c ",&v);
      lin(c);
      if(v=='P')
	c--;
      else
	c++;
      c=abs(c);
      if(ans<c)
	ans=c;
      else if((ans-c)%2)
	ans++;
    }
    printf("%lld\n",ans);
    in(n);
  }
  return 0;
}
