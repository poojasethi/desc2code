#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long ll;

int main()
{
    //freopen("in.txt", "r", stdin);
    int n;
    char c;
    ll ans;
    ll clicks;
    while (cin >> n) {
        if (n==0) break;
        ans = 0;
        for (int i=0; i<n; i++) {
            cin >> c >> clicks;
            if (c == 'P') clicks--;
            else if (c=='M')clicks++;
            else return 1;
            clicks = abs(clicks);
            if (clicks >= ans)
                ans = clicks;
            else if ((ans - clicks)%2==1)
                ans++;
            //ans = max(ans, abs(clicks));
        }
        //printf("%d\n", ans);
        cout << ans << endl;
    }
    return 0;
}
