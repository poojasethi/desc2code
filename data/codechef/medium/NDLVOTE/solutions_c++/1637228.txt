using namespace std;

int main()
{
	int N;
	scanf("%d", &N);
	while (N != 0)
	{
		int userP = 0, userM = 0;
		string ch;
		int last, temp;
		cin >> ch;
		scanf("%d", &temp);
		if (ch[0] == 'P')
			temp--;
		else 
			temp++;
		if (temp >= 0)
			userP = temp;
		else
			userM = -temp;
		last = temp;
		//printf("%d %d\n", userM, userP);
		for (int i = 1; i < N; i++)
		{
			cin >> ch;
			scanf("%d", &temp);
			if (ch[0] == 'P')
				temp--;
			else
				temp++;
			if (temp >= last)
			{
				int m = (temp-last)/2;
				int mm = min(userM, m);
				userM -= mm;
				userP += mm;
				mm = temp-last-2*mm;
				userP += mm;			
			}
			else
			{
				int m = (last-temp)/2;
				int mm = min(userP, m);
				userP -= mm;
				userM += mm;
				mm = last-temp-2*mm;
				userM += mm; 
			}
			last = temp;
			//printf("%d %d\n", userM, userP);
		}
		printf("%d\n", userM+userP);
		scanf("%d", &N);
	}
	return 0;
}
