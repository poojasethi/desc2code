#include <bits/stdc++.h>
#define abs(a)(a>0?a:-a)
int main()
{
    int n;
    long long int score,t,count;
    char c;
    while(1)
    {
            scanf("%d",&n);
            if(n==0)
            break;
            count=0;
            while(n--)
            {
            		getchar();                      
                      scanf("%c %lld",&c,&score);
                      if(c=='P')
                      score--;
                      else
                      score++;
                      t= abs(score);
                      if(t>count)
                      count+=(t-count);
                       else
                       count= count+ (count-t)%2;
                      
                      
            }
           printf("%lld\n",count);
    }
    
  
    return 0;
}
