#include<stdio.h>
//#include<conio.h>
inline long modulo(long v)
{
	if(v<0)
	  return -v;
	  else
	  return v;
}
main()
{
	int n;
	long votes;
	long max;
	char po;
	while(1)
	{
		scanf("%d",&n);
		if(n==0)
		   break;
		max=0;
	
		while(n--)
		{
		getchar();
		po=getchar();
		scanf("%ld",&votes);
		if(po=='P')
		  votes-=1;
		else
		  votes+=1;
	        votes=modulo(votes);
		if(max==0)
			max=votes;
		else  if(votes>max)
			    max=votes;
		else if((votes+max)%2==1)
			    max=max+1;
	        }      
		printf("%ld\n",max); 
		getchar();     
	}
//getch();
}
