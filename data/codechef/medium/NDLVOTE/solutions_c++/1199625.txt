void inout_vote_for_the_noodle_soup() {
	int n;
	char ch;
	int min_user;
	int score;

	do {
		cin >> n;
		if (n == 0) {
			break;
		}

		min_user = 0;
		for (int i = 0; i < n; ++i) {
			cin >> ch >> score;
			cin.get();
			if (ch == 'P') {
				score--;
			}
			else {
				score++;
			}

			score = abs(score);
			if (score >= min_user) {
				min_user = score;
			}
			else if ((min_user - score) % 2) {
				min_user++;
			}
		}

		cout << min_user << endl;
	} while (true);
}

int main() {
	inout_vote_for_the_noodle_soup();
	return 0;
}
