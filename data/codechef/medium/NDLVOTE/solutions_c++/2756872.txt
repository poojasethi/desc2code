#define FOR(i,x,y) for(int i=(x);i<(y);i++)
#define RFOR(i,x,y) for(int i=(x);i>=(y);i--)
#define MIN(x,y) ((x)<(y)?(x):(y))

typedef long long int lli;
typedef long double ld;

using namespace std;


int main()
{
    //freopen("inp.txt","r",stdin);
    int n,k;
    lli m;
    char c;
    lli ma;
    scanf("%d",&n);
    while(n!=0)
    {
    	ma=0;
    	for(int i=0;i<n;i++)
    	{
    		getchar();
    		c=getchar();
    		scanf("%lld",&m);
    		k=(c=='P')?ABS(m-1):ABS(m+1);
    		if(k>=ma) ma=k;
    		else if((k-ma)%2) ma++;
    	}
    	printf("%lld\n",ma);
    	scanf("%d",&n);
    }
    return 0;
    
}
