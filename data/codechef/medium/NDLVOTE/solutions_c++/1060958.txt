#include <iostream>

using namespace std;



int main()
{
	while (1)
	{
		int score=0, click=0, users=0;
		char a[4];
		cin>> click;
		if (click== 0) break;
		for(click>0; click--;)
		{
			scanf("%s %d",a, &score);
			if (a[0]== 'P')score--; else score++;
			score=abs(score);
			if(score>users)
            users = score;
            else if( (users - score)%2)
            users++;
			
		}
	printf("%d\n",users);
	
	}
	
}
