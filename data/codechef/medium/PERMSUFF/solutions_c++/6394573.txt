#include<bits/stdc++.h>
 
#define gc getchar_unlocked
int read_int() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
 
using namespace std;
int main()
{
    int n,m,p,q,arr[100005];
    int a[100005];
    int tc,j,i;
    tc=read_int();
    while(tc--)
    {
        cin>>n>>m;
        for(i=1;i<=n;i++)
          arr[i]=read_int();
 
        for(i=0;i<=n;i++) a[i]=0;
 
        while(m--)
        {
            p=read_int();q=read_int();
            a[p]+=1;a[q]-=1;
        }
        j=1;int sum=0;
        for(i=1;i<=n;i++)
        {
            sum+=a[i];
            if(sum==0)
              a[i]=j++;
            else
              a[i]=j;
        }
        for(i=1;i<=n;i++)
         if(a[arr[i]]!=a[i])
            break;
        if(i>n) printf("Possible\n");
         else printf("Impossible\n");
    }
    return 0;
}
