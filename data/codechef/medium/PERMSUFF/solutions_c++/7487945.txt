#include<bits/stdc++.h>
using namespace std;
 
int a[100005],v[100005];
#define getcx getchar_unlocked
//long int arr[10005],n;
 
inline void inp(int &n)
{
   n=0;
   int ch=getcx();
			int sign=1;
   while(ch<'0'||ch>'9')
			{
			  if(ch=='-')  
					   sign=-1; 
					ch=getcx();
			}
   while(ch>='0'&& ch<='9')
   {
   	  n = (n<<3)+(n<<1) + ch-'0';
						ch=getcx();
   }
   n=n*sign;
}
 
int main()
{
	int t;
	inp(t);
	while(t--)
	{
		int n,m,i,j,x,y;
		inp(n);
		inp(m);
		pair<int,int>pa[100005];
		for(i=1;i<=n;i++)
		inp(a[i]);
		for(i=0;i<100005;i++)
		v[i]=-1;
		for(i=0;i<m;i++)
		{
			inp(x);
			inp(y);
			pa[i].first=x;
			pa[i].second=y;
		}
		sort(pa,pa+m);
		j=0;
		for(i=1;i<m;i++)
		{
			if(pa[j].second>=pa[i].first)
			{
				pa[j].second=max(pa[j].second,pa[i].second);
				pa[i].first=-1;
				pa[i].second=-1;
			}
			else
			j=i;
		}
//		for(i=0;i<m;i++)
//		cout<<pa[i].first<<" "<<pa[i].second<<"\n";
        int cnt=0;
        for(i=0;i<m;i++)
        {
        	if(pa[i].first!=-1)
        	{
        		for(j=pa[i].first;j<=pa[i].second;j++)
        		v[j]=cnt;
        		cnt++;
        	}	
        }
        int flag=0;
        for(i=1;i<=n;i++)
        {
        	if(v[i]!=-1)
        	{
        		if(v[i]!=v[a[i]])
        		{
        			flag=1;
        			break;
        		}
        	}
        	else
        	{
        		if(i!=a[i])
        		{
        			flag=1;
        			break;
        		}
        	}
        }
        if(flag==1)
        printf("Impossible\n");
        else
        printf("Possible\n");
		}
	return 0;
}
