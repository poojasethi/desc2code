#include<bits/stdc++.h>
using namespace std;
inline int max(int a,int b){	if(a>=b)return a;else return b;
}
int main()
{	int t;
	scanf("%i",&t);
	while(t-->0)
	{	int n,m;
		scanf("%i%i",&n,&m);
		int a[n];
		for(int i=0;i<n;i++)
		{	scanf("%i",&a[i]);
		}
		pair<int,int> range[m];
		for(int i=0;i<m;i++)
		{	int k1,k2;
			scanf("%i%i",&k1,&k2);
			range[i]=make_pair(k1,k2);
		}
		sort(range,range+m);
		int l=range[0].first;int r=range[0].second;
		for(int i=1;i<m;i++)
		{	if(range[i].first>r)
			{	sort(a+l-1,a+r);
//				cout<<1<<l<<" "<<r<<endl;
				l=range[i].first;
				r=range[i].second;
			}
			else
			{	r = max(r,range[i].second);
			}
		}
		sort(a+l-1,a+r);
//		for(int i=0;i<n;i++)
//		{	cout<<a[i]<<" ";
//		}
//		cout<<endl;
		bool res = 1;
		for (int i=0;i<n;i++)
		{	if (a[i] != i+1)
				res = 0;
//			cout<<a[i]<<" ";
		}
		printf(res == 1 ? "Possible\n" : "Impossible\n");
	}
	}
