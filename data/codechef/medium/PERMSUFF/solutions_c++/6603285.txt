#include <stdio.h>
#include <algorithm>
#define maxn 100000+1
using namespace std;

int p[maxn];
pair<int, int> pairs[maxn];

int n,m;
int x, y;

int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	bool flag;
	int l,r;
	while(t--) {
	    flag = true;
	    scanf("%d %d", &n, &m);
	    for(int i=1;i<=n; i++) {
	        scanf("%d",&p[i]);
	    }
	    for(int i=0;i<m; i++) {
	        scanf("%d %d",&x, &y);
	        pairs[i] = make_pair(x, y);
	    }
	    
	    sort(pairs, pairs+m);
	    
	    l = pairs[0].first;
	    r = pairs[0].second;
	    
	    for(int i=1; i<m; i++) {
	        if(r >= pairs[i].first) {
	            r = max( r, pairs[i].second );
	        } else {
	            sort(p+l, p+r+1);
	            l = pairs[i].first;
	            r = pairs[i].second;
	        }
	    }
	    sort(p+l, p+r+1);
	    for(int i=1;i<=n;i++) {
	        if( i != p[i]) {
	            flag = false;
	            break;
	        }
	    }
	    
	    if(flag) {
	        printf("Possible\n");
	    } else {
	        printf("Impossible\n");
	    }
	}
	return 0;
}
