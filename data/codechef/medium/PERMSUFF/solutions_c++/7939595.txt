#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
bool fun(pii a,pii b)
{
	if (a.first!=b.first)
		return a.first<b.first;
	else
		return a.second<b.second;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m;
		scanf("%d%d",&n,&m);
		int p[n+1];
		pii a[m];
		int i,tp,tq;
		for(i=1;i<=n;i++)
		{
			scanf("%d",&tp);
			p[tp]=i;
		}
		for(i=0;i<m;i++)
			scanf("%d%d",&a[i].first,&a[i].second);
		sort(a,a+m,fun);
		pii tr;
		vector<pii> nr;
		tr=a[0];
		for(i=1;i<m;i++)
		{
			if(tr.second>=a[i].first)
				tr.second=max(tr.second,a[i].second);
			else
			{
				nr.push_back(tr);
				tr=a[i];
			}
		}
		nr.push_back(tr);
		int grp[100007]={},j,cnt=1,flag=0;
		for(i=0;i<nr.size();i++)
		{
			for(j=(nr[i]).first;j<=(nr[i]).second;j++)
				grp[j]=cnt;
			cnt++;
			for(j=nr[i].second+1;i+1<nr.size() && j<nr[i+1].first;j++)
				grp[j]=cnt++;
		}
		for(i=1;i<=n;i++)
		{
			if(grp[i]!=grp[p[i]])
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
			printf("Possible\n");
		else
			printf("Impossible\n");
	}
	return 0;
}
