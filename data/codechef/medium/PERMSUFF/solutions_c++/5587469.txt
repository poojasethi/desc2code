#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t-->0)
	{
		int n,m;
		scanf("%d %d",&n,&m);
		int inp[n];
		for(int i=0;i<n;i++)
		{scanf("%d",&inp[i]);
		inp[i]--;
		}
		pair<int,int> a[m];
		for(int i=1;i<=m;i++)
		{
			int l,r;
			scanf("%d %d",&l,&r);
			l--;
			r--;
			a[i-1]=make_pair(l,r);
		}
		sort(a,a+m);
		int l,r;
		l=a[0].first;
		r=a[0].second;
		for(int i=0;i<m;i++)
		{
			if(r>=a[i].first)
			{
				r=max(a[i].second,r);
			}
			else
			{
				sort(inp+l,inp+r+1);
				l=a[i].first;
				r=a[i].second;
				
			}	
		}
		sort(inp+l,inp+r+1);
		int done=-1;
		for(int i=0;i<n;i++)
		{
			if(inp[i]!=i)
			{
				printf("Impossible\n");
				done=1;
				break;
			}
		}
		if(done==-1)
		printf("Possible\n");
	}
	
}
