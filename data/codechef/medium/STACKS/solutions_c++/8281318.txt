#include<bits/stdc++.h>
using namespace std;

int main()
{
	std::ios::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--){
		long long int n;
		cin >> n;
		vector<long long int> v;
		for(long long int i=0;i<n;i++){
			long long int a,flag=0;
			cin >> a;
			
			vector<long long int> :: iterator it=upper_bound(v.begin(),v.end(),a);
			if(it==v.end()){
				v.push_back(a);
			}else{
				*it=a;
			}
		}
		cout << v.size() << " ";
		for(int i=0;i<v.size();i++){
			cout << v[i] << " ";
		}
		cout << "\n";
	}
	return 0;
}
