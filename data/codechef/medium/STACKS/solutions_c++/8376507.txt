#include <bits/stdc++.h>
using namespace std;


int main(){
  int t;
  scanf("%d",&t);
  for(int it=0;it<t;it++){    
      int n,v;
      scanf("%d",&n);
      multiset<int> s;
      scanf("%d",&v);
      s.insert(v);

      for(int i=1;i<n;i++){
	  
	  scanf("%d",&v);
	  multiset<int>::iterator it=s.insert(v);
	  while(it != s.end()){
	      if(*it != v){
		  s.erase(it);
		  break;
	      }
	      ++it;
	  }
	  
 
      }
      printf("%d ",(int)s.size());
      for(multiset<int>::iterator it=s.begin();it!=s.end();++it)
	  printf("%d ",*it);
      printf("\n");


  }
  

  return 0;
}
