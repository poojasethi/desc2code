#include<bits/stdc++.h>
#define ll long long int
#define si(x) scanf("%d",&x);
#define sl(x) scanf("%lld",&x);
#define caset int t; si(t); while(t--)

using namespace std;

vector<int> v;

void bs(int x)
{
    int l=0,r=v.size()-1;
    bool flag=0;
    while(l<=r)
    {
        int mid=(l+r)/2;
        if(v[mid]>x&&(mid==0||v[mid-1]<=x))
            v[mid]=x,flag=1;
        else if(v[mid]<x)
            l=mid+1;
        else
            r=mid-1;
    }
    if(flag==0)
    {
        int i=v.size()-1;
        while(i>=0&&v[i]>x)
            i--;
        v[i+1]=x;
    }

}

int main()
{
    //freopen("in.txt","r",stdin);
    caset{
        int n,i,x;
        si(n);
        v.clear();
        for(i=0;i<n;i++)
        {
            si(x);
            if(v.size()==0)
                v.push_back(x);
            else if(v[v.size()-1]<=x)
                v.push_back(x);
            else
                bs(x);
        }
        printf("%d ",v.size());
        for(i=0;i<v.size();i++)
            printf("%d ",v[i]);
        printf("\n");
    }
    return 0;
}
