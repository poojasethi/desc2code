#include<bits/stdc++.h>
using namespace std;
int main(){
	long x,t,n,*it;
	scanf("%ld",&t);
	long arr[100000];
	while(t--){
		scanf("%ld",&n);
		long k=0;
		for(long i=0;i<n;i++){
			scanf("%ld",&x);
			if(k!=0){
//				cout<<"k= "<<k<<endl;
			it=upper_bound(arr,arr+k,x);
//			cout<<"it= "<<it-arr<<endl;
			arr[it-arr]=x;
			if(k<=(it-arr)){
				k=(it-arr)+1;
//				cout<<"final k= "<<k<<endl;
			}
			}
			else{
				arr[0]=x;
				k++;
			}
		}
		printf("%ld ",k);
		for(long i=0;i<k;i++){
			printf("%ld ",arr[i]);
		}
		printf("\n");
	}
	return 0;
}
