#include<iostream>
#include<cstdio>
using namespace std;

typedef unsigned long long ull;

int bs(ull top[], int n, ull a)
{
	int l=0, r=n-1;
	int ans = n;
	while(l<=r)
	{
		ull m = l + (r-l)/2;
		if( top[m]>a )
		{
			ans = m;
			r = m-1;
		}
		else
			l = m+1;
	}
	return ans;
}
