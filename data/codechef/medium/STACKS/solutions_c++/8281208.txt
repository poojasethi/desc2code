#include<bits/stdc++.h>

using namespace std;

#define ll long long int
#define print(a) cout<<a
#define T ll t; cin>>t ; while(t--)
#define fin freopen("input.txt","r",stdin)
#define fout freopen("op.txt","w",stdout)

int main(){

    T {
        ll n;
        cin>>n;

        vector<ll> v;

        for(ll i=0;i<n;i++) {
            ll  x;
            cin>>x;

            vector<ll>::iterator it=upper_bound(v.begin(),v.end(),x);

            if(it==v.end()){
                v.push_back(x);
            }else {
                *it=x;
            }
        }

        ll l=v.size();
        printf("%lld ",l);
        for(ll i=0;i<v.size();i++){
            printf("%lld ",v[i]);
        }
        printf("\n");
    }

    return 0;
}
