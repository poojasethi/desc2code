#include <iostream>
#include <stdio.h>
#include <map>
using namespace std;
#define MAXN 100005
int A[MAXN], B[MAXN];
void solve() {
    int N;
    cin >> N;
    for(int i = 0; i < N; i++) cin >> A[i];
    map<int, int> mp; 
    map<int, int> :: iterator it;
    for(int i = 0; i < N; i++) {
        it = mp.upper_bound(A[i]);
        if(it != mp.end()) {
            int num = it -> first;
            mp[num]--;
            if(mp[num] == 0) mp.erase(num);
            mp[A[i]]++;
        } else {
            mp[A[i]]++;
        }
    }
    it = mp.begin();
    int total = 0;
    for(; it != mp.end(); it++) {
        int num = it -> second;
        for(int j = 0; j < num; j++) {
            B[total++] = it -> first;
        }
    }
   cout << total;
   for(int i = 0; i < total; i++) cout << " " << B[i];
   cout << endl;
}
int main() {
    int t;
    cin >> t;
    while(t--) solve();
}
