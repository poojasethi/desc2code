#include<bits/stdc++.h>
#define gc getchar
typedef long long int ll;
using namespace std;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
ll a[100001];
int binarySearch(ll siz,ll val)
{
    ll l=0, h=siz - 1;
    ll p = siz;
    while (l<=h) {
        ll mid = (l+h)/2;
        if (a[mid] > val) 
        {
            p=mid;
            h=mid-1;
        }
        else {
            l=mid+1;
        }
    }
    return p;
}
int main()
{
    ll t,n,i,k;
    t=read_int();
    while(t--)
    {
        n=read_int();
        ll x,maxi,j=0;
        for(i=0;i<n;i++)
        {
            x=read_int();
            if(i==0)
            {
                a[j]=x;
                maxi=x;
                j++;
            }
            else
            {
                if(x>=maxi)
                {
                    a[j]=x;
                    j++;
                    maxi=x;
                }
                else if(x<maxi)
                {
                    k=binarySearch(j,x);
                    a[k]=x;
                    if(k==j-1)
                        maxi=x;
                }
            }
        }
        printf("%lld ",j);
        for(i=0;i<j;i++)
        {
            printf("%lld ",a[i]);
        }
        printf("\n");
    }
    return 0;
}
