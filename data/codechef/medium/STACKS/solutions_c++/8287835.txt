#include <bits/stdc++.h>

using namespace std;
vector<int> v;
vector<int>::iterator low;
int main()
{
    long long int t,n,a,i;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        while(n--)
        {
            scanf("%lld",&a);
            low=lower_bound(v.begin(),v.end(),a+1);
            if(low!=v.end())
                *low=a;
            else
                v.push_back(a);
        }
        n=v.size();
        if(n!=0)
            printf("%d ",n);
        else
            printf("%d",n);
        for(i=0;i<n-1;i++)
            printf("%d ",v[i]);
        printf("%d\n",v[i]);
        v.clear();
    }
    return 0;
}
