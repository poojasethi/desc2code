#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ios::sync_with_stdio(false);
    int t,n,i;
    cin >> t;
    ll a[100001];
    while(t--)
    {
        cin >> n;
        for(i=0;i<n;++i)cin >> a[i];
        set < pair<ll,int> > s;
        set < pair<ll,int> >::iterator it;
        for(i=0;i<n;++i)
        {
            it=s.upper_bound(make_pair(a[i]+1,-1));
            if(it == s.end())//nt found
            {
                s.insert(make_pair(a[i],i));
            }
            else
            {
                s.erase(it);
                s.insert(make_pair(a[i],i));
            }
        }
        cout << s.size() <<" ";
        for(it=s.begin();it!=s.end();it++)
        cout << it->first <<" ";
        cout << endl;
    }
    return 0;
}
