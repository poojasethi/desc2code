#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ios::sync_with_stdio(false);
    int t,n,i;
    cin >> t;
    
    while(t--)
    {
        cin >> n;
        multiset < ll > ms;
        multiset < ll >::iterator it;
        ll x;
        for(i=0;i<n;++i)
        {
            cin >> x;
            it=ms.upper_bound(x);
            if(it == ms.end())//nt found
            {
                ms.insert(x);
            }
            else
            {
                ms.erase(it);
                ms.insert(x);
            }
        }
        cout << ms.size() <<" ";
        for(it=ms.begin();it!=ms.end();it++)
        cout << *it <<" ";
        cout << endl;
    }
    return 0;
}
