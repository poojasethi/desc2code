#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,c=0;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        vector<int>v;
        v.push_back(a[0]);
        for(int i=1;i<n;i++)
        {
            if(a[i]>=v.back())
                v.push_back(a[i]);
                else
                {
                    vector< int >::iterator it;
                    it=upper_bound(v.begin(),v.end(),a[i]);
                    *it=a[i];
                }
        }
        cout<<v.size()<<" ";
        for(int i=0;i<v.size();i++)
        {

                cout<<v[i]<<" ";
        }
        cout<<endl;
    }
    return 0;
}
