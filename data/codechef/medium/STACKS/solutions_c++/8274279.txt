#include<iostream>
#include<vector>
#define s(t) scanf("%d", &t)
#define p(t) printf("%d", t)
using namespace std;

int stk[100009];
int stkptr;
//return index of ceil or -1 if largest 
int ceil( int x)
{
	if(stkptr==0)return -1;
	if(x>=stk[stkptr-1] )return -1;
	int l= 0;
	int r= stkptr-1;
	
	while(l<r)
	{
	int mid= (l+r)/2;
	if(stk[mid]<= x)
		{
		l= mid+1;
		}
	else
		{
		r=mid;
		}
	}
	return l;
	
}


int main() {
	// your code goes here
	
	int t;
	s(t);
	int n;
	
	while(t--)
	{
	stkptr=0;
	s(n);
	int temp;
	for(int i= 0;i<n;i++ )
		{
		s(temp);
		int index= ceil(temp);
		if(index==-1){stk[stkptr]=temp;stkptr++;}
		else 
			{//replace at index
			stk[index]=temp;
			}
		}
	
	p(stkptr);
	printf(" ");
	for(int i=0;i<stkptr;i++){p(stk[i]);printf(" ");}
	printf("\n");
	}
	
	return 0;
}
