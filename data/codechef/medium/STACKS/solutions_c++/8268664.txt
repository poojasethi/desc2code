#include<iostream>
#include<map>
#include<vector>
using namespace std;
map<int,int> table;
map<int,int>::iterator it;
vector<int> listAns;
int main()
{
    int t,n,input[100005],i,j,ans;
    cin>>t;
    while(t--)
    {
        cin>>n;
        listAns.clear();
        for(i=0;i<n;i++)
            cin>>input[i];
        table.clear();
        for(i=0;i<n;i++)
        {
            it=table.upper_bound(input[i]);
            if(it==table.end())
            {
               // cout<<input[i]<<" yo\n";
                table[input[i]]++;
            }
            else
            {
                //cout<<input[i]<<" nnno\n";
                table[it->first]--;
                if(table[it->first]==0)
                {
                    table.erase(it->first);
                }
                table[input[i]]++;
            }
        }
        ans=0;
        for(it=table.begin();it!=table.end();it++)
        {
            ans+=(it->second);
           // cout<<it->first<<endl;
            for(i=0;i<it->second;i++)
                listAns.push_back(it->first);
        }
        cout<<ans<<" ";
        for(i=0;i<listAns.size();i++)
            cout<<listAns[i]<<" ";
        cout<<"\n";
    }
    return 0;
}
