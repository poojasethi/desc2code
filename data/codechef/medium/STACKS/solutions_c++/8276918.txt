#include<bits/stdc++.h>

using namespace std;

int A[100005];
vector<int> T(100005);

int main(){
int i,j,t,n,l,len,r,mid;
vector<int>::iterator itr;
cin>>t;
while(t--){
    cin>>n;
    len = 0;
    for(i=0;i<n;++i){
        cin>>A[i];
    }
    T[len] = A[0];
    for(i=1;i<n;++i){
        if(A[i]>=T[len]){
            len++;
            T[len] = A[i];
        }
        else{
            itr = upper_bound(T.begin(),T.begin()+len,A[i]);
            T[itr-T.begin()] = A[i];
        }
    }
    cout<<(len+1)<<" ";
    for(i=0;i<=len;++i){
        cout<<T[i]<<" ";
    }
    cout<<endl;

}



return 0;
}
