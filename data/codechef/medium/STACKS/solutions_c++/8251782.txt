#include<bits/stdc++.h>

using namespace std;
multiset<int> st;
	multiset<int> :: iterator it,it2;
int a[1000007];
int main()
{
	int t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		{
			it=st.upper_bound(a[i]);
			if(it==st.end())
			{
				st.insert(a[i]);
			}
			else
			{
				st.erase(it);
				st.insert(a[i]);
			}
		}
		it=st.begin();
		cout<<st.size()<<" ";
		while(it!=st.end())
		{
			cout<<*it<<" ";
			it++;
		}
		cout<<"\n";
		st.clear();
	}
    
	return 0;
}
