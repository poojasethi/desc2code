import bisect
t=input()
for x in range(t):
    m=input()
    n=map(int,raw_input().split())
    l=[n[0]]
    for z in xrange(1,m):
        x=n[z]
        if x>=l[-1]:
            l.append(x)
        else:
            pos=bisect.bisect(l,x)
            l[pos]=x
    print len(l),
    for p in l:
        print p,
    print ""
