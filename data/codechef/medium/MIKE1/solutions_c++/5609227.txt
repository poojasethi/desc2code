#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll n, m, i, j, e1=0, e2=0;
    scanf("%lld %lld", &n, &m);
    ll arr[n+1][m+1];
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=m;j++)
        {
            scanf("%lld", &arr[i][j]);
        }
    }
    ll q;
    scanf("%lld", &q);
    while(q--)
    {
        scanf("%lld %lld", &i, &j);
        if(i<=n && j<=m && e1!=-1)
            e1+=arr[i][j];
        else
            e1=-1;
        if(i<=m && j<=n && e2!=-1)
            e2+=arr[j][i];
        else
            e2=-1;
    }
    printf("%lld", max(e1, e2));
    return 0;
}
