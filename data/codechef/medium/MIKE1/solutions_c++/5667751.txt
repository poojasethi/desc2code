#include<iostream>
#include<stdio.h>
using namespace std;
long int ar[510][510];
long long int e1,e2;
int p[250010][2];
int main()
{  int m,n,i,j,l,flag1=0,flag2=0;
   cin>>n>>m;
   for(i=1;i<=n;i++)
   { for(j=1;j<=m;j++)
     {  cin>>ar[i][j]; }
   }
   cin>>l;
   for(i=1;i<=l;i++)
   {  cin>>p[i][0]>>p[i][1];
      if((p[i][0]<=n)&&(p[i][1]<=m)&&(flag1==0))
      {  e1=e1+ar[p[i][0]][p[i][1]]; }
      else
      { flag1=1; 
        e1=-1;
      }
      if((p[i][0]<=m)&&(p[i][1]<=n)&&(flag2==0))
      {  e2=e2+ar[p[i][1]][p[i][0]]; }
      else
      { flag2=1; 
        e2=-1;
      }
    }  
    cout<<max(e1,e2)<<endl;
    return 0;
}
