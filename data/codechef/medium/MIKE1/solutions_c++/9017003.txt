int main(){
	int m,n;
	scan(m);
	scan(n);
	ll arr[501][501] = {-1};
	int i,j;
	rep(i,0,m){
		rep(j,0,n){
			scanll(arr[i][j]);
		}
	}
	int l;
	scan(l);
	ll sum1=0,sum2=0;
	while(l--){
		int x,y;
		scan(x);
		scan(y);
		if(sum1!=-1&&x<=m&&y<=n){
			sum1+=arr[x-1][y-1];
		}
		else{
			sum1 = -1;
		}
		if(sum2!=-1&&y<=m&&x<=n){
			sum2+=arr[y-1][x-1];
		}
		else{
			sum2 = -1;
		}
	}
	ll ans = sum1>sum2?sum1:sum2;
	printll(ans);
	return 0;
}
