#include<cstdio>
#include<algorithm>
using namespace std;
int N,M,L,A[505][505],Kx,Ky;
long long E1,E2,pom;
bool pok1,pok2;
int main(){
    scanf("%d %d",&N,&M);
    for(int i=1;i<=N;i++)
        for(int j=1;j<=M;j++)
            scanf("%d",&A[i][j]);
    scanf("%d",&L);
    pok1=true;
    pok2=true;
    while(L--){
        scanf("%d %d",&Kx,&Ky);
        if(Kx>N||Ky>M)
            pok1=false;
        if(Kx>M||Ky>N)
            pok2=false;
        E1+=A[Kx][Ky];
        E2+=A[Ky][Kx];
    }
    pom=-1;
    if(pok1==true&&pok2==true)
        printf("%lld",max(E1,E2));
    else if(pok1==false&&pok2==true)
            printf("%lld",max(pom,E2));
         else if(pok1==true&&pok2==false)
                printf("%lld",max(E1,pom));
              else printf("-1");
    return 0;
}
