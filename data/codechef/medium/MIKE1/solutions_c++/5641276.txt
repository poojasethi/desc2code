#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
	long long sum1=0,sum2=0;
	int r,c,i,j,k,x,y;
	cin>>r;
	cin>>c;
	int a[r][c];
	for(i=0;i<r;i++)
	{
		for(j=0;j<c;j++)
		cin>>a[i][j];
	}
	int n;
	cin>>n;
	for(k=1;k<=n;k++)
	{
		cin>>x;
		cin>>y;
		if(x>r||y>c)
		sum1=-1;
		if(x>c||y>r)
		sum2=-1;
		if(x<=r&&y<=c&&sum1!=-1)
		sum1+=a[x-1][y-1];
		if(x<=c&&y<=r&&sum2!=-1)
		sum2+=a[y-1][x-1];
	}
		

	cout<<max(sum1,sum2);	
	
}
