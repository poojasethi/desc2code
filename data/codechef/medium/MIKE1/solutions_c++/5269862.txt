#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<cmath>
#define LL long long int
using namespace std;
int main()
{
    int m,n,i,j,l,flag1=0,flag2=0;
    LL sum1,sum2;
    cin>>n>>m;
    int A[n][m];
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
            cin>>A[i][j];
    cin>>l;
    int que[l][2];
    for(i=0;i<l;i++)
    {
        cin>>que[i][0]>>que[i][1];
        que[i][0]--;
        que[i][1]--;
    }
    sum1=0;
    sum2=0;
    for(i=0;i<l;i++)
    {
        if(que[i][1]>=m || que[i][0]>=n)
        {
            flag1=-1;
        }
        else
            sum1+=A[que[i][0]][que[i][1]];
        if(que[i][0]>=m || que[i][1]>=n)
        {
            flag2=-1;
        }
        else
            sum2+=A[que[i][1]][que[i][0]];
    }
    if(flag1==-1)
        sum1=flag1;
    if(flag2==-1)
        sum2=flag2;
    if(sum1>sum2)
        cout<<sum1;
    else
        cout<<sum2;
    return 0;
}
