#include<bits/stdc++.h>
using namespace std;
int a[600][600];
int flag[2]={0};
int n,m;
inline long long  in(int x,int y,int j)
{
	if(x<n&&y<m)
		return (long long)a[x][y];
	flag[j]=1;
	return 0;
}
int main()
{
	int i,j,k;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;++i)
		for(j=0;j<m;++j)
			scanf("%d",&a[i][j]);
	int x,y;
	long long ans=0,ans1=0;
	int l;
	scanf("%d",&l);
	for(i=0;i<l;++i)
	{
		scanf("%d%d",&x,&y);
		ans+=in(x-1,y-1,0);
		ans1+=in(y-1,x-1,1);
	}
	if(flag[0])
		ans=-1;
	if(flag[1])
		ans1=-1;
	printf("%lld\n",max(ans,ans1));
	return 0;
}
