#include<iostream>
#include<math.h>
#include<vector>
using namespace std;
int main()
{
int n,m;
cin>>n>>m;
vector<vector<int > >a(n,vector<int > (m));
for (int i=0;i<n;i++){
for (int j=0;j<m;j++)
cin>>a[i][j];
}
int l,x,y,s1=0,s2=0;
cin>>l;
signed long long int sum1=0,sum2=0;
for (int i=0;i<l;i++){
cin>>x>>y;
if (x>n || y>m || sum1<0) sum1=-1;
else sum1 += a[x-1][y-1];
if (y>n || x>m || sum2<0) sum2=-1;
else sum2 += a[y-1][x-1];
}
cout<<max(sum1,sum2);
cin.get();
cin.ignore();
return 0;
}
