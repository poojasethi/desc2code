#include <cstdio>
using namespace std;
int n,m,a[505][505],x,z,i,j,l,pom1,pom2;
long long e1,e2;
int main (){

 scanf("%d %d",&n,&m);
 for (i=1;i<=n;i++){
  for (j=1;j<=m;j++){
   scanf("%d",&a[i][j]);
  }
 }

 scanf("%d",&l);
 for (i=1;i<=l;i++){
  scanf("%d %d",&x,&z);
  e1+=a[x][z];
  e2+=a[z][x];
  if (x>n or z>m)
   pom1=-1;
  if (x>m or z>n)
   pom2=-1;
 }
 if (pom1==-1)
  e1=-1;
 if (pom2==-1)
  e2=-1;
 if (e1>=e2)
  printf("%lld",e1);
  else printf("%lld",e2);

return 0;
}
