#include <iostream>
#include <string>
using namespace std;
int main()
{
    long long i,j,sum1=0,sum2=0,c1=0,c2=0;
    cin>>i>>j;
    long long c[i][j];
    for(int a=0;a<i;a++)
    {
        for(int b=0;b<j;b++)
        {
            cin>>c[a][b];
        }
    }
    int d;
    cin>>d;
    long long e[d][2];
    for(int a=0;a<d;a++)
    {
        for(int b=0;b<2;b++)
        {
            cin>>e[a][b];
        }
        if(e[a][0]<=i&&e[a][1]<=j&&c1==0)
            sum1+=c[e[a][0]-1][e[a][1]-1];
        else
        {
            sum1=-1;
            c1=1;
        }
        if(e[a][0]<=j&&e[a][1]<=i&&c2==0)
            sum2+=c[e[a][1]-1][e[a][0]-1];
        else
        {
            sum2=-1;
            c2=1;
        }

    }
    if(sum1>sum2)
        cout<<sum1<<endl;
    else
        cout<<sum2<<endl;
    return 0;
}
