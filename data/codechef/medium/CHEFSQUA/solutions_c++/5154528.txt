#include<cstdio>
#include<iostream>
#include<limits.h>
#include<bits/stdc++.h>
using namespace std;

#define MAXI 2001
#define inp(n) scanf("%d",&n)
#define FOX(i,n) for(int i=0;i<n;i++)
#define mp make_pair
#define pb push_back
typedef long long ll;
#define F first
#define S second
map<pair<int,int >,int > mymap;
int x[MAXI],y[MAXI],n,x3,y3,x4,y4,res;
int main()
{

	inp(n);
	FOX(i,n)
	{
		inp(x[i]),inp(y[i]);
		mymap[mp(x[i],y[i])]=1;
	}
	if(n==0)	
		printf("4\n");
	else if(n==1)
		printf("3\n");
	else if(n==2)
		printf("2\n");
	else{
		res=2;
	
		FOX(i,n)
			{
				for(int j=i+1;j<n;j++)
				{
					x3=x[i]+y[j]-y[i];	
					y3=y[i]+x[i]-x[j];
					x4=x[j]+y[j]-y[i];
					y4=y[j]+x[i]-x[j];
					if(mymap.count(mp(x3,y3))&&mymap.count(mp(x4,y4)))
					{
						printf("0\n");	
						return 0;	//exit(1);					
					}
					else if(mymap.count(mp(x3,y3))||mymap.count(mp(x4,y4))){
						res=1;					
					}			
				
				}
			}
		printf("%d\n",res);


	}
	return 0;
}
