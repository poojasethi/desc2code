#include <iostream>
#include <map>
#include <set>
#include <utility>

using namespace std;
int min(int a,int b)
{
	if(a>b)
		return b;
	else
		return a;
}
int main()
{
	ios_base::sync_with_stdio(false);
	int n;
	cin>>n;
	int size=n;
	 int a,b;
	 set< pair<int,int> > ss;
	while(size--)
	{
		cin>>a>>b;
		ss.insert(make_pair(a,b));
  	}
  	if(n==0 ||n==1)
  	{
  		cout<<4-n<<endl;
  		return 0;
  	}
  	set< pair<int,int> >::iterator it1=ss.begin();
  	set< pair<int,int> >::iterator it2;
  	int dx,dy;
  	int ans=4;
  	for(;it1!=ss.end();it1++)
  	{
  		it2=it1;
  		it2++;
  		for(;it2!=ss.end();it2++)
  		{
  			int req=2;
  			dx=(*it2).first-(*it1).first;
  			dy=(*it2).second-(*it1).second;

  			if(ss.find(make_pair((*it2).first-dy,(*it2).second+dx) )!=ss.end())
  				req--;

  			if(ss.find(make_pair((*it1).first-dy,(*it1).second+dx) )!=ss.end())
  				req--;

  			ans=min(req,ans);

  			 req=2;
  			
  			dx=-dx;

  			if(ss.find(make_pair((*it2).first+dy,(*it2).second+dx) )!=ss.end())
  				req--;

  			if(ss.find(make_pair((*it1).first+dy,(*it1).second+dx) )!=ss.end())
  				req--;


  			ans=min(req,ans);
  			if(ans==0)
  			{
  				cout<<ans<<endl;
  				return 0;
  			}
  		}
  	}
  	 cout<<ans<<endl;
	return 0;
}
