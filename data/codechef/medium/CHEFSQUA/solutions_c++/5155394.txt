#include<iostream>
#include<map>
#include<vector>
#include<cstring>
#define mp make_pair


using namespace std;
int main()

{  vector< pair<int,int> >a;
map< pair<int,int>,int >m;
	int  n,i,j,ans,x,y;
	pair<int ,int >p1,p2,p3,p4;
	cin>>n;
    a.resize(n);
	bool f1,f2,f3,f4;

	for(i=0;i<n;i++)
	{
	cin>>a[i].first>>a[i].second;
	

	m[mp(a[i].first,a[i].second)]=1;
	}
	if(n == 0)
	{
	cout<<"4"<<endl;
		return 0;
	}
	if(n == 1)
	{
		cout<<"3";
		return 0;
	}
	ans=2;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			p1 = mp(a[i].first + (a[j].second-a[i].second), a[i].second - (a[j].first-a[i].first));
			p2 = mp(a[i].first - (a[j].second-a[i].second), a[i].second + (a[j].first-a[i].first));
			p3 = mp(a[j].first + (a[i].second-a[j].second), a[j].second - (a[i].first-a[j].first));
			p4 = mp(a[j].first - (a[i].second-a[j].second), a[j].second + (a[i].first-a[j].first));
			f1 = m.find(p1) != m.end();
			f2 = m.find(p2) != m.end();
			f3 = m.find(p3) != m.end();
			f4 = m.find(p4) != m.end();
		
	
	if(f1 || f2 || f3 || f4)
				ans = 1;
			if(f1 && f4)
			{
				cout<<"0"<<endl;
				return 0;
			}
			if(f2 && f3)
			{
			cout<<"0"<<endl;
				return 0;
			}
		}
	}
	cout<<ans<<endl;
	return 0;
}
