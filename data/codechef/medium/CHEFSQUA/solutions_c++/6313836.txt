#include <bits/stdc++.h>
using namespace std;
int main() {
	int ans = 2;
	long long X[2001], Y[2001];
	set< pair< int , int > >SET;
	int n;
	cin >> n;
	for( int i = 0 ; i < n ; i++ ){ 
		int x , y;
		cin >> x;
		X[ i ] = 4 * x;
		cin >> y;
		Y[ i ] = 4 * y;   
		SET.insert( make_pair( X[ i ] , Y[ i ] ) );
	}
	if( n == 0 ) {
		cout << 4 << '\n';
		return 0;
	}
	if( n == 1 ){
		cout << 3 << '\n';
		return 0;
	}
	for( int i = 0 ; i < n ; i++ )
		for(int j = i + 1 ; j < n ; j++ ) {     
			int lx = ( X[ i ] + X[ j ] +Y[ j ] - Y[ i ] )/2;    
			int ly = ( X[ i ] - X[ j ] +Y[ j ] + Y[ i ] )/2;
			int hx = ( X[ i ] + X[ j ] -Y[ j ] + Y[ i ] )/2;
			int hy = (-X[ i ] + X[ j ] +Y[ j ] + Y[ i ] )/2; 
			int cont = 0;
			if( SET.count( make_pair( lx , ly ) ) ) cont++;
			if( SET.count( make_pair( hx , hy ) ) ) cont++;
			ans = min( ans , 2 - cont );
			//cout << cont << " - > " << endl;
		}
	cout << ans << '\n';
}
