#include<bits/stdc++.h>
using namespace std;
using std::sort;
using std::string;
vector<pair<int,int> >v;
pair<int,int>pair1,pair2;
int a[2001][3];
int main()
{
  ios::sync_with_stdio(0);
  int num,i,j,k,l,e,f,c,d,q1,q2,p1,p2;
  int ans,ret;
 cin>>num;
  for(i=0;i<num;i++)
  {
    for(j=0;j<2;j++)
    cin>>a[i][j];
    v.push_back(make_pair(a[i][0],a[i][1]));
  }
 if(num<=2)
 {
 cout<<4-num<<endl;
 return 0;
 }
 sort(v.begin(),v.end());
 int min=2;
 for(i=0;i<num;i++)
 {
   e=a[i][0];
   f=a[i][1];
   for(j=i+1;j<num;j++)
   {
    c=a[j][0];
    d=a[j][1];
    p1=e-(d-f);
    p2=f+(c-e);
    pair1=make_pair(p1,p2);
    q1=c-(d-f);
    q2=d+(c-e);
    pair2=make_pair(q1,q2);
    
    ans=binary_search(v.begin(),v.end(),pair1)+binary_search(v.begin(),v.end(),pair2);
    if(ans==2)
    {
      cout<<"0"<<endl;
      return 0;
    }
    else
    ret=2-ans;
    if(ret==1)
    min=1;  
    }
    
    
   }
   cout<<min<<endl;
   return 0;
 
 }
