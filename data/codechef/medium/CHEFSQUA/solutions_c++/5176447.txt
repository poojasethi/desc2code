    #include <bits/stdc++.h>
    using namespace std;

    typedef pair<int ,int > PII;

    int n;
    int ans = 2;
    int arrx[2014],arry[2014];
    set< PII > arr;

    int main()
    {
    PII p;
    scanf("%d",&n);
    for(int i=0; i<n; i++){
        scanf("%d %d",&p.first,&p.second);
        arrx[i] = p.first;
        arry[i] = p.second;
        arr.insert(p);
    }
    if(n < 3)
        printf("%d\n",4-n);
    else
    {
    for(int i=0; i<n-1; i++)
    {
        for(int j=i+1; j<n; j++)
        {
            int x = arrx[j] - arrx[i];
            int y = arry[j] - arry[i];
            PII p1(arrx[i] - y, arry[i] + x);
            PII p2(arrx[j] - y, arry[j] + x);
            PII n1(arrx[i] + y, arry[i] - x);
            PII n2(arrx[j] + y, arry[j] - x);

   
            int flag = 2;
            if(arr.count(p1))
                flag--;
            if(arr.count(p2))
                flag--;
            if(flag < ans)
                ans = flag;
            if(flag == 0)
                break;

            flag = 2;
            if(arr.count(n1))
                flag--;
            if(arr.count(n2))
                flag--;
            if(flag < ans)
                ans = flag;
            if(flag == 0)
                break;
        }
        if(ans == 0)
        break;
    }
    printf("%d\n",ans);
    }
    return 0;
    }
