#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>

using namespace std;
int n,s,f,br,mi;
pair <int, int> niz[2005], asd,dsa;


int main(){
    mi=2;
    scanf("%d",&n);
    if (n==0){
        printf("%d",4);
        return 0;
    }
    if (n==1){
        printf("%d",3);
        return 0;
    }
    if (n==2){
        printf("%d",2);
        return 0;
    }


    for(int i=1;i<=n;i++){
        scanf("%d%d",&s,&f);
        niz[i]=make_pair(s,f);
    }

    sort(niz+1, niz+n+1);

    for (int i=1;i<n;i++){
        for (int j=i+1;j<=n;j++){
            br=0;
            int a=niz[j].first-niz[i].first;
            int b=niz[i].second-niz[j].second;
            int x=niz[i].first+b;
            int y=niz[i].second+a;
            int x1=niz[j].first+b;
            int y1=niz[j].second+a;
            asd=make_pair(x,y);
            dsa=make_pair(x1,y1);
            if (!(binary_search(niz+1,niz+n+1,asd)))br++;
            if (!(binary_search(niz+1,niz+n+1,dsa)))br++;
            if (mi>br) mi=br;
        }

    }
    printf("%d",mi);
    return 0;
}
