#include <bits/stdc++.h>
using namespace std;

pair<int ,int> p;
vector<pair<int,int> > vec;
set<pair<int,int> >  s;
int main() {
	// your code goes here
	int n;
	int a,b;
	int ans;
	scanf("%d",&n);
	for(int i=0; i < n; i++) {
	    scanf("%d %d",&a, &b);
	    p = make_pair(a,b);
	    vec.push_back(p);
	    s.insert(p);
	}
	if(n <= 2) {
	    ans = 4-n;
	    printf("%d\n",ans);
	} else {
	    int x1,y1;
	    int x2,y2;
	    int ans = 5;
	    
	    for(int i=0;i<n-1;i++) {
	        int cnt = 0 ;
	        x1 = vec[i].first;
	        y1 = vec[i].second;
	        for(int j=i+1;j<n;j++) {
	            x2 = vec[j].first;
	            y2 = vec[j].second;
	            
	            pair<int,int> A,B;
	            A = make_pair(x1-(y2-y1), y1+(x2-x1));
	            B = make_pair(x2-(y2-y1), y2+(x2-x1));
	            

	            if(s.find(A) == s.end()) {
	                cnt++;
	            }
	            if(s.find(B) == s.end()) {
	                cnt++;
	            }
	            ans = min(ans, cnt);
	            cnt = 0;
	            A = make_pair(x1+(y2-y1), y1-(x2-x1));
	            B = make_pair(x2+(y2-y1), y2-(x2-x1));
	            
	            if(s.find(A) == s.end()) {
	                cnt++;
	            }
	            if(s.find(B) == s.end()) {
	                cnt++;
	            }
	            ans = min(ans, cnt);
	            cnt= 0;
	        }
	    }
	    printf("%d\n",ans);
	}
	return 0;
}
