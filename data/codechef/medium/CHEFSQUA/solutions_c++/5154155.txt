# include <stdio.h>
# include <math.h>
# include <iostream>
# include <string.h>
# include <stdlib.h>

#define N 2222
#define rep(i,a,b) for(int i=a;i<b;i++)
#define drep(i,b,a) for(int i=b-1;i>=a;i--)
#define sort1(d,n) qsort(d,n,sizeof(cor),comp);

struct cor{
	int x,y;
};
int n;
cor d[N];

int comp(void const *a,void const *b)
{
	cor fi=*(cor*)a;
	cor se=*(cor*)b;
	if(fi.x-se.x==0)
	return fi.y-se.y;
	return fi.x-se.x;
}
int bin(int x1,int y1)
{
	int a=0,b=n;
	while(a<=b)
	{
		if(d[(a+b)/2].x==x1)
		{
			if(d[(a+b)/2].y==y1)
			{return 1;}
			else if(d[(a+b)/2].y<y1){
				a=(a+b)/2+1;continue;}
			else{
				b=(a+b)/2-1; continue;}
		}
		else if(d[(a+b)/2].x<x1){
			a=(a+b)/2+1; continue;}
		else{
			b=(a+b)/2-1; continue;}
	}
	
	return 0;
}

int cevirx(int a1,int b1,int a2,int b2)
{
	return (a2+b1-b2);
}
int ceviry(int a1,int b1,int a2,int b2)
{
	return (b2+a2-a1);
}

int main()
{
	scanf("%d",&n);
	rep(i,0,n) scanf("%d %d",&d[i].x,&d[i].y);
	
	if(n==0){
	printf("4"); return 0;}
	if(n==1){printf("3"); return 0;}
	
	int min=11;
	sort1(d,n);
	rep(i,0,n-1)
	rep(j,0,n)
	{if(i==j)continue;
		int a=d[i].x,b=d[i].y;
		int a1=d[j].x,b1=d[j].y;
		int ans=2;
		int a2=cevirx(a,b,a1,b1);
		int b2=ceviry(a,b,a1,b1);
		ans =ans- bin(a2,b2);
		ans =ans- bin(cevirx(a1,b1,a2,b2),ceviry(a1,b1,a2,b2));
		if(ans<min) min=ans;
	}
	printf("%d",min);
	getchar();
	getchar();
return 0;
}
