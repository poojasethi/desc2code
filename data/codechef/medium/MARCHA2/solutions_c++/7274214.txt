#include<bits/stdc++.h>
using namespace std;
int arr[1000001];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int i=0;
        bool ans=true;
        int stem=1,leaves;
        while(i<n-1)
        {
            scanf("%d",&leaves);
            if(leaves>=stem)
            {
                ans=false;
            }
            stem=2*(stem-leaves);
            i++;
        }
        scanf("%d",&leaves);
        if(ans&&leaves==stem)
        {
            printf("Yes\n");
        }
        else
        {
            printf("No\n");
        }
    }
    return 0;
}
