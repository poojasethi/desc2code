#include<bits/stdc++.h>
using namespace std;
inline int getno()
{
	int no=0;
	char ch;
	ch=getchar();
	while(ch<'0')
	ch=getchar();
	while(ch>='0' && ch<='9')
	{
	  no=(no<<3)+(no<<1)+ch-'0';
	  ch=getchar();	
	}
	return no;
}
int main()
{
	long int t,n,temp,a,flag;
	scanf("%ld",&t);
	while(t--)
	{flag=0;
		scanf("%ld",&n);
		temp=1;
		while(n--)
		{
		    a=getno();
			temp=temp-a;
			if(temp<0)
			{
				flag=1;
				break;
			}
			temp+=temp;
		}
		if(temp!=0)
		{
			flag=1;
		}
		if(flag==0)
		printf("Yes\n");
		else
		printf("No\n");
	}
	return 0;
}
