#include<bits/stdc++.h>
using namespace std;
typedef long long ll;


int main()
{
	ios::sync_with_stdio(false);
	int t;cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		ll a[n+1];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		ll stem=1;
		bool flag=true;
		for(int i=0;i<n;i++)
		{
			stem-=a[i];
			if(stem<0)
			{
				flag=false;
				break;
			}
			stem*=2;
		}
		if(flag)
		{
			if(stem==0)
			{
				cout<<"Yes\n";
			}
			else
			{
				cout<<"No\n";
			}
		}
		else
		{
			cout<<"No\n";
		}
	}
	return 0;
}
