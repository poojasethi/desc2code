	#include<iostream>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			int n;
			cin>>n;
			int stem=1;
			int leaves,temp;
			int flag=0;
			for(int i=0;i<n;i++)
			{
				cin>>temp;
				if(temp<=stem && flag==0)
				{
					stem=stem-temp;
					stem=stem*2;
				}	
				else
				{
					flag=1;
				}	
			}	
			if(stem==0 && flag==0)
				cout<<"Yes\n";
			else
				cout<<"No\n";
			t--;
		}	
	}
