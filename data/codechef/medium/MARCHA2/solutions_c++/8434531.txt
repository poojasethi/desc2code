bool isPowerTwo(int n)
{
	while (n!=1)
	{
		if (n%2 != 0)
			return false;
		n = n/2;
	}
	return true;
}

int main()
{
	int t;
	int k;
	int i;
	cin >> t;
	while (t--)
	{
		cin >> k;
		int arr[k];

		// cin >> arr[0];
		// int leaves = arr[0];
		int internals = 0;
		int expected = 1;
		bool flag = 0;

		for (i=0;i<k;i++)
		{
			cin >> arr[i];
			if (arr[i] > expected)
			{
				flag = 1;
				break;
			}
			// leaves += arr[i];
			internals = expected - arr[i];

		}

		else
			cout << "Yes\n";
	}
	return 0;
}
