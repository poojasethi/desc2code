#include <iostream>

using namespace std;

int main()
{
    int b=1,f=0,k,t,a;
    cin>>t;
    while (t--)
    {
        cin>>k;
        while(k--)
    {
        cin>>a;
        if(a>b)
        {
            f++;
            break;
        }
        b=(b-a)*2;
    }
        if(f||b)
        {
            cout<<"No"<<"\n";
        }
        else
        {
            cout<<"Yes"<<"\n";
        }
        b=1;
        f=0;
    }

    return 0;
}
