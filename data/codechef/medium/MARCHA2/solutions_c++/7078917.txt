#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;

int main(){
    //freopen( "input.txt", "r", stdin );
    //freopen( "output.txt", "w", stdout );
    lli t,k,leaf,level;
    cin>>t;
    while(t--){
        cin>>k;
        lli stem=1;
        for(lli i=0;i<k;i++){
            cin>>leaf;
            if(stem<0) break;
            stem=stem-leaf;
            stem*=2;
        }

        if(stem==0) cout<<"Yes"<<endl;
        else cout<<"No"<<endl;
    }
return 0;
}
