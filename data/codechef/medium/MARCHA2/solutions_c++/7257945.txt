#include<iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[n];
		int flag=0;
		int stem=1;
		int leaves;
		for(i=0;i<n;i++)
		{
			
			cin>>leaves;
			if(leaves>stem)
			{
				flag=1;
				break;
			}
			stem=(stem-leaves)*2;
		}
		if(flag||stem)
		{
			cout<<"No"<<endl;
		}
		else
		{
			cout<<"Yes"<<endl;
		}
		
	}
	
		
		return 0;
	
}
