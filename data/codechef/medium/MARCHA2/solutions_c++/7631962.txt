#include<stdio.h>
#include<stdlib.h>

int main()
{
	long signed int i,j,k,t;
	long long signed int x,y,diff;
	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%ld",&k);
		scanf("%lld",&y);
		diff = 0;
		int flag=0;
		diff = 1-y;
		if((y==1 && k!=1) || y>1 || (y==0 && k==1))
		flag=1;
		for(i=1;i<k;i++)
			{
			scanf("%lld",&x);
			//printf("diff = %d\n",diff);
			if(i==(k-1) && x==(diff*2) && flag!=1)
			break;
			else if(i<(k-1) && x<=(diff*2) && flag!=1)
			diff = diff*2 - x;
			else
			flag=1;
			}
			if(flag==0)
			printf("Yes\n");
			else
			printf("No\n");
	}
	return 0;
	
}
