#include <iostream>
using namespace std;
int main()
{
	int t;
	int k,a,s,i;
	int flag;
	cin >> t;
	while (t--) {
		s = 1;
		flag = 0;
		cin >> k;
        for (i = 0; i < k; i++)
        {
			cin >> a;
			if (a > s)
			{
				flag = 1;
				break;
			}
            else
            {
				s = (s - a)*2 ;
			}
		}

		if (s != 0)
        {
			flag = 1;
		}
		if (flag == 0)
		{
			cout << "Yes" << "\n";
		}
		else
		{
			cout << "No" << "\n";
		}
    }
    return 0;
}
