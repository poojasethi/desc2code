#include<cstdio>
#include<algorithm>
#include<iostream>
#include<string.h>
#include<math.h>
#include<vector>
using namespace std;
int main()
{
 int t;
 scanf("%d",&t);
 while(t--)
 {
  long k,stems=1,i;
  scanf("%ld",&k);
  long a[k+1];
  for(i=1;i<=k;i++)
  scanf("%ld",&a[i]);
  for(i=1;i<k;i++)
   {
    if(a[i]>=stems)
       {cout<<"No"<<endl;
           break;}
     else{
       stems=stems-a[i];
       stems*=2;
     }
   }
    if(i==k)
    {
        if(a[i]==stems)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
  }


   return 0;

  }
