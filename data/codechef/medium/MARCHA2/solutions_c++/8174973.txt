#include<bits/stdc++.h>
using namespace std;
typedef long long lli;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        lli k,i,tmp,f=0,max_val=1;

        scanf("%lld",&k);

        for(i=1;i<=k;i++)
        {
            scanf("%lld",&tmp);
            if(tmp>max_val)
            {
                f=1;
                break;
            }
            else
            {
                max_val=2*(max_val-tmp);
            }
        }
        if(!f&&max_val==0)printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}
