#include<iostream>
using namespace std;
int main()
{
    int t,x;ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        cin>>x;
        int arr[x];
        for(int i=0;i<x;i++)
            cin>>arr[i];
        int nextlevel = (1-arr[0])*2;
        int flag = 1;
        for(int i=1;i<x-1;i++)
        {
            nextlevel = (nextlevel-arr[i])*2;
            if(nextlevel<0)
            {
                flag = 0;
                break;
            }
        }
        if(flag == 1&&nextlevel==arr[x-1])
            cout<<"Yes"<<endl;
        else if(x==1&&arr[0]==1)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
}
