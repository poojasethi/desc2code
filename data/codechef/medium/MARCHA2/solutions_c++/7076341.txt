#include <iostream>
using namespace std;

int main() {
	long int T,l,i,N,*L,*S;
	std::cin>>T;
	for(i=0;i<T;i++)
	{
	    std::cin>>N;
	    L=new long int[N+1];L[0]=1;
	    S=new long int[N+1];S[0]=0;
	    S[1]=1;std::cin>>L[1];
	    for(l=2;l<N+1;l++)
	    {
	        std::cin>>L[l];
	        S[l]=2*(S[l-1]-L[l-1]);
	       // std::cout<<S[l]<<L[l]<<"\n"<<S[l-1]<<L[l-1]<<"\n";
	        if(S[l]<0) break;
	        
	    }
	    if(l==N+1) {
	        if(2*S[N]-2*L[N]==0) std::cout<<"Yes\n";
	        else std::cout<<"No\n";
	    }
	    else std::cout<<"No\n";
	}
	return 0;
}
