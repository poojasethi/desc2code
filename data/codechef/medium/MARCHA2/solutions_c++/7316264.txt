#include <iostream>
using namespace std;

#define MAX 1000006

int a[MAX];

int main(int argc, char const *argv[])
{
	int t, n;
	cin >> t;
	while(t--) {
		cin >> n;
		for(int i = 0; i < n; i++) {
			cin >> a[i];
		}
		
		long long c = 1;
		bool flag = true;
		for(int i = 0; i < n; i++) {
			if(c < a[i]) {
				flag = false;
				break;
			}
			c -= a[i];
			c = c << 1;
		}

		if(c == 0 && flag)
			cout << "Yes" << endl;
		else
			cout << "No" << endl;	
	}
	return 0;
}
