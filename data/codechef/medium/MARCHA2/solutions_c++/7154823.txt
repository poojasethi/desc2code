#include <iostream>
#include <stdio.h>
using namespace std;

int main()
{
    int i,t,stem,leaves,temp,k;
    scanf("%d",&t);
    while(t--)
    {
        int flag=0;
        scanf("%d",&k);
        int a[k];
        for(i=0;i<k;i++)
            scanf("%d",&a[i]);
        temp=1,leaves=1,stem=0;
        for(i=0;i<k;i++)
        {
            if(a[i]>leaves)
            {
                printf("No\n");
                flag=1;
                break;
            }
            if(i==k-1&&a[i]!=leaves)
            {
                printf("No\n");
                flag=1;
                break;
            }
            stem=stem*2+a[i];
            leaves=(temp-stem)*2;
            temp=temp*2;
        }
        if(flag==0)
            printf("Yes\n");

    }
}
