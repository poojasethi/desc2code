#include <iostream>
using namespace std;
int f(long l[], long k)
	{
	long level;
	long long stem=1;
	for (level=0; level<k; level++)
		if(l[level]>stem)
			return 0;
		else
			stem=(stem-l[level])*2;
	if(stem==0)
		return 1;
	else
		return 0;
	}
	
int main() 
{
	long i,t,k;
	cin>>t;
	while(t--)
	{
		cin>>k;
		long l[k];
		for (i=0; i<k; i++)
		{
			cin>>l[i];
		}
		if (f(l,k)==1)
			cout<<"Yes\n";
		else
			cout<<"No\n";
	}
	return 0;
}
