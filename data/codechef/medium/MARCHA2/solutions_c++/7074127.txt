#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#ifdef USE_DATA_TXT
static FILE *stream = NULL;
#endif

int main()
{
#ifdef USE_DATA_TXT
  freopen_s(&stream, "c:\\data.txt", "r", stdin);
#endif

  int T;
  scanf("%d\n", &T);
  
  while(T--)
  {
    int K;
    scanf("%d\n", &K);

    bool isYes = false;

    int S = 1; // Number of stems on the next level
    int i = 1;
    bool isValid = true;
    for(; i <= K; ++i)
    {
      int Li;
      scanf("%d\n", &Li);

      S = (S - Li) * 2;

      if(S < 0)
        isValid = false;
    }

    printf((isValid && (S == 0)) ? "Yes\n" : "No\n");
  }

#ifdef USE_DATA_TXT
  if(stream)
  {
    fclose(stream);
    stream = NULL;
  }
#endif

  return 0;
}
