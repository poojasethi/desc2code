#include<iostream>
using namespace std;
int main()
{
	int t,k;
	cin>>t;
	while(t--)
	{
		cin>>k;
		int l,stem=1,flag=0;
		while(k--)
		{
			cin>>l;
			if(l>stem)
			{
				flag=1;
				break;
			}
			stem=(stem-l)*2;
			
		}
		if(flag||stem)
		cout<<"No"<<endl;
		else
		cout<<"Yes"<<endl;
	}
	return 0;
}
