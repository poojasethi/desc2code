#include<stdio.h>

long int a[1000000];
int main()
{
    int tc;
    long int k,i;
    long long int branches;

    scanf("%d",&tc);

    while(tc--)
    {
        scanf("%ld",&k);
        for(i=0;i<k;i++)
            scanf("%ld",&a[i]);

        branches = 1;
        if(a[0] > 1)
        {
            printf("No\n");
            goto label;
        }
        else
        {
            branches = 2*(branches-a[0]);
        }
        for(i=1;i<k;i++)
        {
            if(a[i] > branches)
            {
                printf("No\n");
                goto label;
            }
            else
            {
                branches = 2*(branches-a[i]);
            }
        }
        if(branches == 0)
            printf("Yes\n");
        else
            printf("No\n");
        label:;
    }
    return 0;
}
