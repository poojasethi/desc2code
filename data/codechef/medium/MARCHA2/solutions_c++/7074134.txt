#include <stdio.h>
#include <string.h>
#include <stdlib.h>

#ifdef USE_DATA_TXT
static FILE *stream = NULL;
#endif

int main()
{
#ifdef USE_DATA_TXT
  freopen_s(&stream, "c:\\data.txt", "r", stdin);
#endif

  int T;
  scanf("%d\n", &T);
  
  while(T--)
  {
    int K;
    scanf("%d\n", &K);

    int S = 1; // Number of stems on the next level
    for(int i = 1; i <= K; ++i)
    {
      int Li;
      scanf("%d\n", &Li);

      if(S < 0)
        continue;

      S = (S - Li) * 2;
    }

    printf((S == 0) ? "Yes\n" : "No\n");
  }

#ifdef USE_DATA_TXT
  if(stream)
  {
    fclose(stream);
    stream = NULL;
  }
#endif

  return 0;
}
