#include<stdio.h>
int main()
{
	long long int t,i,num,k,count,max_level;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&k);
		max_level=1;
		for(i=1;i<=k;i++)
		{
			scanf("%lld",&num);
			if(num>max_level)
			{
				max_level=1;
				break;
			}
			else
			{
				max_level=max_level-num;
				max_level*=2;
			}
		}
		if(max_level==0)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
