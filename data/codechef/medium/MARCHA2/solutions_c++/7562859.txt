#include<stdio.h>
using namespace std;
int main()
{
    bool f;
    int t;
    long int k,i,a,br;
    scanf("%d",&t);
    while(t--)
    {
        f=0;
        scanf("%ld",&k);
        br=1;
        for(i=1;i<=k;++i)
        {
            scanf("%ld",&a);
            if(a>br || br<0)
                f=1;
            br=(br-a)*2;     //br is the number of branches in the next level
        }
        if(br)
            f=1;
        if(f)
            printf("No\n");
        else
            printf("Yes\n");
    }
    return 0;
}
