#include<iostream>
#include<cstdio>

using namespace std;

int main(){
    int t;
    long long n;
    scanf("%d",&t);
    while(t--){
               scanf("%lld",&n);
               long long curr;
               int flag;
               if(n==1){
                        scanf("%lld",&curr);
                        if(curr==1)
                        printf("Yes\n");
                        else
                        printf("No\n");
                        }
                        else{
                             long long a[n];
                             for(long long i=0;i<n;i++)
                             scanf("%lld",&a[i]);
                             if(a[0]!=0)
                             printf("No\n");
                             else{
                                  curr=2;
                                  flag=1;
                                  for(long long i=1;i<n;i++){
                                           if(i==(n-1)){
                                                    if(curr!=a[i])
                                                    flag=-1;
                                                }
                                           else{
                                           if(a[i]<=curr){
                                                          curr=(curr-a[i])*2;
                                                          }
                                                          else{
                                                               flag=-1;
                                                               break;
                                                              }
                                               }
                                           }
                                           if(flag==1)
                                           printf("Yes\n");
                                           else
                                           printf("No\n");
                             
                                 }
                             }
               
               }
}
