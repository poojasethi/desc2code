#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long k,i;
		cin>>k;
		long a[k];
		for( i=0;i<k;i++)
		{
			cin>>a[i];
		}
		if(k==1)
		{
			if(a[0]==1) cout<<"Yes\n";
			else cout<<"No\n";
			continue;
		}
		 long p=1;
		for(i=1;i<k;i++)
		{
			 p=2*(p-a[i-1]);
		//	 cout<<"i:"<<p<<endl;
			if(a[i]>p) {cout<<"No"<<endl; break;
			}
			else if(a[i]==p && i==k-1) { cout<<"Yes"<<endl; break; }
		}
		if(i==k) cout<<"No"<<endl;
		}
	
	return 0;
}
