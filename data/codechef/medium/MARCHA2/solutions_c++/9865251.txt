#include <iostream>
#include <math.h>
using namespace std;
 
int main() {
	// your code goes here
	int t;
	cin >> t;
	while(t--) {
		long long int n, i, m = 1, r;
		cin >> n;
		long long int a[n];
		for(i = 0; i < n; i++)
			cin >> a[i];
		for(i = 0; i < n - 1; i++) {
			if(a[i] > m) {
				cout << "No\n";
				break;
			}
			r = m - a[i];
			m = r*2;
		}
		if(i == n - 1) {
			if(m == a[i])
				cout << "Yes\n";
			else
				cout << "No\n";
		}
	}
	return 0;
}
