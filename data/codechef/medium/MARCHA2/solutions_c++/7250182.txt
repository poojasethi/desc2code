#include<bits/stdc++.h>
using namespace std;

const int N=1234567;
int a[N];

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n;
        scanf("%d", &n);
        for(int i=0;i<n;i++) {
            scanf("%d", &a[i]);
        }
        bool ok=true;
        int steam=1;
        for(int i=0;i<n;i++) {
            steam=2*(steam-a[i]);
            //cout<<steam<<endl;
            if((steam==0 && i<n-1) || steam<0 || (steam>0 && i==n-1)) {
                ok=false;
                break;
            }
        }
        if(ok) {
            puts("Yes");
        } else {
            puts("No");
        }
    }
}
