#!/usr/bin/python

# read the number of testcases
testcases = int(raw_input())

# for each testcase
for i in range(0, testcases):
	# read the number of levels
	beanlevel = int(raw_input())
	# read the leaf count
	leafcount = map(int, str(raw_input()).split())
	# sanity check
	if len(leafcount) > 1 and leafcount[0] > 0:
		print 'No'
		continue
	# reverse leafcount
	leafcount.reverse()
	# set noflag to false
	noflag = False
	# initialize stems to zero
	stems = 0
	# verify if john counted correctly
	for j in leafcount:
		stems = int(stems / 2)
		stems = stems + j
		if stems % 2 and stems > 1:
			noflag = True
			break
		
	# check if the last level matches
	if noflag == False and stems == 1:
		print 'Yes'
	else:
		print 'No'
