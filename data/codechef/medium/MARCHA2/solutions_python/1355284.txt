import sys

n = int(sys.stdin.readline())
for i in xrange(n):
    lvl = int(sys.stdin.readline())
    leaves_at_each_lvl = map(int, sys.stdin.readline().split())
    leaves_at_each_lvl.reverse()
    stems = 0
    for lvs in leaves_at_each_lvl:
        leaves_plus_stems = lvs + stems
        if lvl == 1:
            if leaves_plus_stems == 1:
                print "Yes"
            else:
                print "No"
            break
        if leaves_plus_stems % 2 != 0:
            print "No"
            break
        else:
            stems = leaves_plus_stems / 2
            lvl -= 1
