    length=input()
    flag=1
    l=[int(x) for x in raw_input().split(" ")]
    total=l[length-1]
    i=length-1
    for k in range(length):
        if (i>0 and total%2!=0) or (i==0 and total!=1):
            flag=0
            break
        i-=1
