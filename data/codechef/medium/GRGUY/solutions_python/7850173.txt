#!/usr/bin/python

T = int(raw_input())

def test():

	L1 = raw_input()
	L2 = raw_input()

	d1 = [ch == '#' for ch in L1]
	d2 = [ch == '#' for ch in L2]

	data = zip(d1, d2)
	possible = (True, True) not in data

	print 'Yes' if possible else 'No'

	if not possible:
		return

	# Filter out cells where both slots are open
	data = filter(lambda pair: pair != (False, False), data)

	count = 0
	if len(data):
		last = data[0]
		for cell in data:
			if cell != last:
				count += 1
				last = cell

	print count

for x in range(0, T):
	test()
