#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t;
	scanf("%lld",&t);
	getchar();
	while(t--)
	{
		char a[1000001],b[1000001];
		long long int i,count=0,max1=0,max2=0,flag=1,fbd=1,ca=0,db=0;
		gets(a);
		gets(b);
	
		for(i=0;a[i]!='\0';i++)
		{
			 if(a[i]=='#'&&b[i]=='#')
			{
				flag=0;
				break;
			}
			if(a[i]=='.'&&b[i]=='#'&&db==0)
			{
				max1++;
				db=1;
			}
			if(b[i]=='.'&&a[i]=='#'&&db==0)
			{
				max2++;
				db=1;
			}
		
			else
			{
				max1++;
				max2++;
			}
			
		}
		if(a[strlen(a)-1]=='#'&&b[strlen(b)-1]=='#')
		{
			flag=0;
		}
		if(max1>max2&&flag==1)
		{
			for(i=0;i<strlen(a);i++)
			{
				
			
				if(a[i]=='#'&&fbd==1)
				{
					if(b[i]!='#')
					{
						count++;
						fbd=0;
						
					}
				}
				else if(fbd==0&&b[i]=='#')
				{
					if(a[i]!='#')
					{
						count++;
						fbd=1;
					
					}
				}
			}
		}
		else if(max2>max1&&flag==1)
		{
			for(i=0;i<strlen(a);i++)
			{
				
			
				if(a[i]=='#'&&ca==1)
				{
					if(b[i]!='#')
					{
						count++;
						ca=0;
						
					}
				}
				else if(ca==0&&b[i]=='#')
				{
					if(a[i]!='#')
					{
						count++;
						ca=1;
						
					}
				}
			}
		}
		if(flag==1)
		printf("Yes\n%lld\n",count);
		else
		printf("No\n");
	}
	return 0;
}
