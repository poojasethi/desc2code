#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	string s,o,s1,s2;
	scanf("%d",&t);
	while(t--){
		cin>>s1>>s2;
		bool flag=true;
		int acnt=0,bcnt=0,i;
		s=s1;o=s2;
		for(i=0;i<s.length();i++){
			if(s[i]=='#'){
				if(o[i]=='#'){
					flag=false;
					break;
				}
				else{
					acnt++;
					swap(s,o);
				}
			}
		}
		if(flag){
			s=s1;o=s2;
		for(i=0;i<s.length();i++){
			if(o[i]=='#'){
				if(s[i]=='#'){
					flag=false;
					break;
				}
				else{
					bcnt++;
					swap(s,o);
				}
			}
		}
	}
		if(flag==false){
			printf("No\n");
		}
		else{
		int ans=min(acnt,bcnt);
			printf("Yes\n%d\n",ans);
		}
	}

	return 0;
}
