	#include<iostream>
	#include<string.h>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			char a[200005],b[200005];
			cin>>a>>b;
			int l=strlen(a);
			int d1[200005],d2[200005];
			d1[0]=d2[0]=0;
			int flag=0;
			for(int i=1;i<=l;i++)
			{
				if(a[i-1]==b[i-1] && b[i-1]=='#')
					flag=1;
				if(a[i-1]=='.')
				{
					d1[i]=min(d1[i-1],d2[i-1]+1);
				}	
				else if(a[i-1]=='#')
				{
					d1[i]=d2[i-1]+1;
				}
				if(b[i-1]=='.')
				{
					d2[i]=min(d2[i-1],d1[i-1]+1);
				}	
				else if(b[i-1]=='#')
				{
					d2[i]=d1[i-1]+1;
				}	
			}	
			if(flag==0)
				cout<<"Yes\n"<<min(d1[l],d2[l])<<endl;
			else
				cout<<"No\n";
			t--;
		}	
	}
