#include <bits/stdc++.h>
using namespace std;

int take(string st, string b){
    string cur_str = st, other = b;
    int gr = 0;
    for(int i = 0; i < cur_str.length()-1; i++){
        if(cur_str[i+1]=='#'){
            if(other[i+1]=='.'){
               gr++;
                swap(cur_str,other);
            }else{
                return -1;
            }
        }
    }
    return gr;
}

int main(){
    int t;
    cin >> t;
    while(t--){
        string a,b;
        int ans=INT_MAX;
        cin >> a >> b;
        if(a[0]=='.'){
            int x = take(a,b);
            if(x!= -1) ans = min(x,ans);
        }
        if(b[0]=='.'){
            int x = take(b,a);
            if(x!= -1) ans = min(x,ans);
        }
        if(ans == INT_MAX) cout << "No" << endl;
        else{cout << "Yes" << endl; cout << ans << endl;}
    }
}
