#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define sl(l) scanf("%lld",&l)
#define MAXZ 10000000
char lane[2][200005];

ll count_steps(ll len,ll flag)
{
	ll jump = 0;
	ll i;
	for(i=0;i<len-1;i++)
	{
		if(lane[flag][i+1] == '#' && lane[!flag][i+1] == '.')
		{
			jump++;
			flag=!flag;
		}

		if(lane[flag][i+1] == '#' && lane[!flag][i+1] == '#')
			break;
	}
	if(i<len-1)return MAXZ;
	return jump;
}
int main()
{
	ll t;
	sl(t);
	while(t--)
	{
		ll flag = 0;
		scanf("%s",lane[0]);scanf("%s",lane[1]);
		ll len = strlen(lane[0]);
		ll ans=0;
		if(lane[0][0] == '.' && lane[1][0] == '.')ans = min(count_steps(len,0),count_steps(len,1));
		else if(lane[0][0] == '.')ans = count_steps(len,0);
		else if(lane[1][0] == '.')ans = count_steps(len,1);
		else ans = MAXZ;
		if(ans == MAXZ)printf("No\n");
		else printf("Yes\n%lld\n",ans);
		
	}	
	return 0;

}
