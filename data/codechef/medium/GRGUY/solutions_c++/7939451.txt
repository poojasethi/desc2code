#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
int t;
cin>>t;
char *a=new char[200005];
char *b=new char[200005];
while(t--)
{
	int len,i,flag=0,initial,jump=0;
	cin>>a;
	cin>>b;
	len=strlen(a);
	int *arr= new int[len];
	for(i=0;i<len;i++)
	{
		if(a[i]=='#' && b[i]=='#')
		{
			flag=-1;
			break;		
		}
	if(a[i]=='#' && b[i]=='.')
		arr[i]=1;
	else if(a[i]=='.' && b[i]=='#')
		arr[i]=-1;
	else
		arr[i]=0;
	}
	if(flag==0)
	cout<<"Yes"<<endl;
	else
	{
	cout<<"No"<<endl;
	continue;
	}
	for(i=0;i<len;i++)
	{
		if(arr[i]==1||arr[i]==-1)
		{
		initial=arr[i];
		break;
		}
	}
	for(i=0;i<len;i++)
	{
		if(arr[i]!=initial && arr[i]!=0)
		{
			jump++;
			initial=arr[i];
		}
	}

cout<<jump<<endl;
}
return 0;
}
