//Philosophy is common sense with big words. -James Madison
#include<bits/stdc++.h>

using namespace std;

char l1[200005];
char l2[200005];

int main()
{

//    freopen( "A-small-practice-1.in" , "r" , stdin );
//    freopen( "Output.txt" , "w" , stdout );

//    long long cases;
//    cin >> cases;

    int t;
    cin >> t;
    int i;
    int len;
    int curr;
    int shift;
    int p1, p2;

    while (t--)
    {
        cin >> l1;
        cin >> l2;

        len = strlen(l1);

        curr = 1;
        shift = 0;
        p1 = 0;
        p2 = 0;

        while ( l1[p1] == '.' )
        {
            p1++;
        }

        while ( l2[p2] == '.' )
        {
            p2++;
        }

        if ( p1 == 0 && p2 == 0 )
        {
            cout << "No\n";
            continue;
        }

        else if ( p1 <= p2 )
        {
            curr = 2;
        }

        else if ( p2 < p1 )
        {
            curr = 1;
        }


        for ( i = 0 ; i <= len - 2 ; i++ )
        {
            if ( curr == 1 )
            {
                if ( l1[i+1] == '.' )
                {
                    continue;
                }
                else if ( l2[i+1] == '.' )
                {
                    shift++;
                    curr = 2;
                    continue;
                }
                else
                {
                    cout << "No\n";
                    break;
                }
            }
            else if ( curr == 2 )
            {
                if ( l2[i+1] == '.' )
                {
                    continue;
                }
                else if ( l1[i+1] == '.' )
                {
                    shift++;
                    curr = 1;
                    continue;
                }
                else
                {
                    cout << "No\n";
                    break;
                }
            }
        }

        if ( i == len - 1 )
        {
            cout << "Yes\n" << shift << endl;
        }
    }










    return 0;
}
