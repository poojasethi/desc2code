ll ca(ll l,ll in,ll n)
{
    if(in==n) return 0;
    if(s[l][in]=='#') return inf;

    m1=ca(l,in+1,n);
    m2=ca((l+1)%2,in+1,n)+1;
    return arri[l][in]=min(m1,m2);
}
