#include <bits/stdc++.h>

using namespace std;
typedef long long int ll;

ll INF = 1<<30;

int main() 
{
	int t;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		ll n = s1.length(),ans=0;
		int curr = 0;
		for(unsigned i=0;i<n;i++)
		{
			if(s1[i]==s2[i]&&s1[i]=='#')  ans = INF;
			if(s1[i]=='#')
			{
				if(curr == 1)
					ans ++;
				curr = 2;
			}
			if(s2[i]=='#')
			{
				if(curr==2)
					ans ++;
				curr = 1;
			}
		}
		if(ans >= INF)
			cout<<"No"<<endl;
		else
			{cout<<"Yes"<<endl; cout<<ans<<endl;}
		
	}
	return 0;
}
