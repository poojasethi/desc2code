#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
#define LIM 200011
#define INF (LIM<<3)

char L[2][LIM];
int D[2];
int nD[2];
int main() {
    int z;
    scanf("%d", &z);
    while (z--) {
        scanf("%s%s", L[0], L[1]);
        int n = strlen(L[0]);
        D[0] = D[1] = 0;        
        for (int i = 0; i < n; i++) {
            nD[0] = L[0][i] == '#' ? INF : min(D[0], D[1] + 1);
            nD[1] = L[1][i] == '#' ? INF : min(D[1], D[0] + 1);
            D[0] = nD[0];
            D[1] = nD[1];
        }
        int ans = min(D[0], D[1]);
        if (ans >= INF) {
            printf("No\n");
        } else {
            printf("Yes\n%d\n", ans);
        }
    }
}
