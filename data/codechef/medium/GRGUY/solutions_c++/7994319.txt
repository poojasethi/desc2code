#include<iostream>
#include<string>
using namespace std;

string s1,s2;
bool f_jumps(int lan, int &y){
	int i=-1;
	y=0;
	while(i!=s1.length()-1){
		if(s1[i+1]=='#' && s2[i+1]=='#'){
			return false;
		}
		if(lan==1 && s1[i+1]=='#'){
				y++;
				lan=2;
		}
		else if(lan==2 && s2[i+1]=='#'){
				y++;
				lan=1;
		}
		i++;
	}
	return true;
}

int min(int a,int b){
	if(a>b) return b;
	else return a;
}

int main(){
	int t;
	int jump,lane,pos;
	
	cin>>t;
	while(t--){
		cin>>s1>>s2;
		jump=0;
		int temp=0;
		pos=f_jumps(1,temp);
		pos=f_jumps(2,jump);
		//cout<<"jump from lane 1: "<<temp<<"\nfrom lane 2: "<<jump<<endl;
		jump=min(temp,jump);
		if(pos){
			cout<<"Yes\n"<<jump<<endl;
		}else{
			cout<<"No\n";
		}
	}
}
