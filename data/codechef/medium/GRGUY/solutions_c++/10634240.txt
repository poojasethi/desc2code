#include<bits/stdc++.h>
using namespace std;
char l1[1000010],l2[1000010];
int dp[1000010][2] , n;
int main()
{
    int T , i ;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%s%s",l1,l2);
        n = strlen(l1);
       // memset(dp,-1,sizeof(dp));
        int ans = INT_MAX/2;
        dp[n][0] = dp[n][1] =0;
        bool f1 , f2;
        for(i=n-1;i>=0;i--)
      {
        f1 = f2 = false;
        if(l1[i] == '#')
        {
            f1 = true;
            dp[i][0] = INT_MAX /2;
        }
        if(l2[i] == '#')
        {
            f2 = true;
            dp[i][1] = INT_MAX/2;
        }
        if(f1 && f2)
        {
            ans = INT_MAX/2;
           // break;
        }
        if(!f1)
        {
            dp[i][0] = min(dp[i+1][0] , 1 + dp[i+1][1]);
        }
        if(!f2)
        {
            dp[i][1] = min(dp[i+1][1] , 1 + dp[i+1][0]);
        }
      }
       // if(!(f1 && f2))
        ans = min(dp[0][1] , dp[0][0]);
        if(ans == INT_MAX/2)
            printf("No\n");
        else {
            printf("Yes\n");
            printf("%d\n",ans);
        }
    }
    return 0;
}
