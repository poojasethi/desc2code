#include<iostream>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
		//initialize arrays
		char L1[200005], L2[200005];
		L1[0]='0';
		L2[0]='0';
		//array input
		cin>>L1>>L2;
		//initialize pointers
		char *p,*q;
		p=L1;
		q=L2;
		
		int flag=1;
		//decide which array lasts longer initially
		while(*p&&*q)
		{
			if(*p=='.'&&*q=='.')
			{
				p++;
				q++;
			}else
			{
				if(*p=='#')
				{
					flag=0;
					//cout<<"start with q\n";
				}
				else
				{
					flag=1;
					//cout<<"start with p\n";
				}
				break;
			}			
		}
		//decide which array to start with
		char *r,*s;
		if(flag)
		{
			r=p;
			s=q;
		}
		else
		{
			r=q;
			s=p;
		}
		
		char *v;
		v=r;
		int flip=0;
		int cnt=0;
		int ans=0;
		flag=0;
		//count number of flips
		while(*v)
		{
			if(*v=='.')
			{
				v++;
				cnt++;
			}else
			{
				if(flip==0)//checking r presently 
				{
					if(*(s+cnt)=='#')//if other array is also dirty then No
					flag=1;
					else
					{
						v=s+cnt+1;
						//cout<<"v points to "<<*v<<endl;
						ans++;
						flip=1;//flipped to s
						//cout<<"flipped to s\n";
						cnt++;
					}
				}else//checking s presently
				{
					if(*(r+cnt)=='#')//if other array is also dirty then No
					flag=1;
					else
					{
						v=r+cnt+1;
						//cout<<"v points to "<<*v<<endl;
						ans++;
						flip=0;//flipped to r
						//cout<<"flipped to r\n";
						cnt++;
					}
				}
			}
			//cout<<"cnt : "<<cnt<<endl;	
			if(flag)//ans is No
			break;	
		}
		if(flag)
		cout<<"No\n";
		else
		cout<<"Yes\n"<<ans<<"\n";	
	}
	//system("pause");
	return 0;
}
