#include<iostream>
#include<algorithm>
using namespace std;
long long int check(string s1,string s2)
{
	long long int i,count=0,f=0;
	for(i=0;s1[i]!='\0';++i)
	{
		if(s1[i]=='#' && f==0)
		{
			f=1;
			count++;
			continue;
		}
		else if(s2[i]=='#' && f==1)
		{
			f=0;
			count++;
			continue;
		}
	}
	return count;
}
int main()
{
	long long int t,k,r=0;
	string s1,s2;
	cin>>t;
	while(t--)
	{
		r=0;
		cin>>s1;
		cin>>s2;
		for(k=0;s1[k]!='\0';++k)
		{
			if(s1[k]=='#' && s2[k]=='#')
			{
			cout<<"No"<<endl;
			r=1;
			break;
			}
		}
		if(r==0)
		{
			if(s2[0]=='#')
			{
			cout<<"Yes"<<endl<<check(s1,s2)<<endl;
			}
			else if(s1[0]=='#')
			{
			cout<<"Yes"<<endl<<check(s2,s1)<<endl;
			}
			else if(s1[0]=='.' && s2[0]=='.')
			{
				cout<<"Yes"<<endl<<min(check(s1,s2),check(s2,s1))<<endl;
			}
		}
	}
}
