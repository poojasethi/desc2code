#include<stdio.h>
#include<limits.h>
#include<string.h>
int main()
{
	long long int flag,sw1,sw2,sw,lane,i,t;
	char a[200003], b[200003];
	scanf("%lld", &t);
	while (t--)
	{
		scanf("%s", a);
		scanf("%s", b);
		i = 0;
		flag = 0;
		sw1 = sw2 = INT_MAX;
		while (a[i] != '\0')
		{
			if (a[i] == '#'&&b[i] == '#')
			{
				flag = 1;
				break;
			}
			i++;
		}
		if (flag == 0)
		{
			
			if (a[0] == '.')
			{
				sw1 = 0;
				lane = 1;
				i = 1;
				while (i < strlen(a))
				{
					if (lane == 1 && a[i] == '.')
						i++;
					else if (lane == 2 && b[i] == '.')
						i++;
					else
					{
						sw1++;
						lane = 3 - lane;
						i++;
					}
				}
			}
			if (b[0] == '.')
			{
				sw2 = 0;
				lane = 2;
				i = 1;
				while (i < strlen(a))
				{
					if (lane == 1 && a[i] == '.')
						i++;
					else if (lane == 2 && b[i] == '.')
						i++;
					else
					{
						sw2++;
						lane = 3 - lane;
						i++;
					}
				}
			}
		}

		if (sw2 < sw1)
			sw = sw2;
		else
			sw = sw1;
		if (flag == 1)
			printf("No\n");
		else
		{
			printf("Yes\n%lld\n", sw);
		}
	}
	getchar();
	getchar();
	return 0;
}
