long long unsigned ncr[1001][1001];

long long unsigned comb(int n, int k) {
  if(n<k)
    return 0;
  return ncr[n][k];
}

void precal(void) {
  int i, j;
  for(i=0;i<=1000;i++)
    for(j=0;j<=i;j++) {
      if(j==0||j==i)
        ncr[i][j]=1;
      else
        ncr[i][j]=(ncr[i-1][j-1]+ncr[i-1][j]);
      if(ncr[i][j]>=MOD)
        ncr[i][j]-=MOD;
    }
}

int main(void) {
  char g[1000][4];
  int tc, Y, X, wy, wx, x, y, c, x2, x3;
  scanf("%d", &tc);
  precal();
 while(tc--) {
  long long unsigned ans=0;
  scanf("%d%d%d%d", &Y, &X, &wy, &wx);
  for(y=0;y<Y;y++) {
    scanf("%s", &g[y]);
    for(x=0;x<X;x++)
      g[y][x]-='0';
  }
  if(wx==1)
    for(x=0;x<X;x++) {
      for(c=0,y=0;y<Y;y++)
        c+=g[y][x];
      ans=(ans+comb(c,wy))%MOD;
    }
  else if(wx==2)
    for(x=0;x<X-1;x++)
      for(x2=x+1;x2<X;x2++) {
        for(c=0,y=0;y<Y;y++)
          c+=(g[y][x]&&g[y][x2]);
        ans=(ans+comb(c,wy))%MOD;
      }
  else if(wx==3&&X==3) {    
          for(c=0,y=0;y<Y;y++)
            c+=(g[y][0]&&g[y][1]&&g[y][2]);
          ans=comb(c,wy)%MOD;
        }
  printf("%llu\n", ans);
 }
}
