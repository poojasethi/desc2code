#include<iostream>
#include<cstdio>
#include<vector>
#define MODULO 1000000080798150871LL
using namespace std;
typedef long long unsigned num;
typedef vector<num> vec;
char matrix[1001][3];
vector<vec> C(1001);
void solve()
{
	int n,m,l,k,i,set,j,cnt,x;
	int comb[3][3][3]={
			{{1}},
			{{2,1},{3}},
			{{1,2,4},{3,6,5},{7}}};
	int size[3][3]={
			{1},
			{2,1},
			{3,3,1}
	};
	num res=0;
	scanf("%d %d %d %d",&n,&m,&l,&k);
	//cin>>n>>m>>l>>k;
	for(i=1;i<=n;i++)
		scanf("%s",matrix[i]);
	if(l>n||k>m)
	{
		cout<<'0'<<endl;
		return;
	}
	for(i=0;i<size[m-1][k-1];i++)
	{
		set=comb[m-1][k-1][i];
		cnt=0;
		for(j=1;j<=n;j++)
		{
			for(x=0;x<m;x++)
			{
				if(set&(1<<(x)))
					if(matrix[j][x]!='1')
						break;
			}
			if(x==m)
				cnt++;
		}
		if(cnt >= l){
			res=(res+C[cnt][l])%MODULO;
		}
 
	}
	printf("%llu\n",res);
}
int main()
{
	int t,j;
	cin>>t;
	C[0].push_back(1);
	for(int i=1;i<=1000;i++)
	{
		C[i].push_back(1);
		for(j=1;j<i;j++)
		{
			C[i].push_back((C[i-1][j] + C[i-1][j-1])%MODULO);
		}
		C[i].push_back(1);
	}
	while((t--)>0)
		solve();
	return 0;
}
