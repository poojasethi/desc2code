#include<cstdio>

typedef long long LL;
const int N=1000;
const int M=3;
const LL MOD=1000000080798150871LL;

LL arr[N+1][N+1];

void precalculation()
{
	int i,j;
	for(i=0;i<=N;i++)
	{
		arr[i][0]=1;
		arr[i][i]=1;
		for(j=1;j<i;j++)
		{
			arr[i][j]=(arr[i-1][j-1] + arr[i-1][j])%MOD;
		}
	}
}

inline int bit(int n)
{
	return (1 << n);
}

LL calculate(char a[N+1][M+1],int n,int m,int l,int k)
{
	LL ans=0;
	int mask=0,i,j;
	if(l>n || k>m) return 0;
	int bValue=bit(m);
	for(mask=0;mask<bValue;mask++)
	{
		//printf("here %d\n",mask);
		int cnt=0;
		for(j=0;j<m;j++)
		{
			if(mask & bit(j))
				cnt++;
		}
		if(cnt != k)
			continue;
		//printf("hello\n");
		int b[N];
		//printf("aaa\n");
		for(j=0;j<n;j++)
			b[j]=1;
		//printf("www\n");
		for(j=0;j<m;j++)
		{
			//printf("bbb\n");
			if(mask & bit(j))
			{
				for(i=0;i<n;i++)
					b[i]&=a[i][j];
			}
		}
		//printf("yha\n");
		int num=0;
		for(i=0;i<n;i++)
			num+=b[i];
		ans=(ans+arr[num][l])%MOD;
	}
	return ans;
}

int main()
{
	int t,n,m,l,k,i,j;
	
	char a[N+1][M+1];
	
	precalculation();
	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d%d",&n,&m,&l,&k);
		for(i=0;i<n;i++)
		{
			scanf("%s",a[i]);
			for(j=0;j<m;j++)
			{
				a[i][j]=a[i][j]-'0';
			}
		}
		LL ans=calculate(a,n,m,l,k);
		printf("%lld\n",ans);
	}
	
}
