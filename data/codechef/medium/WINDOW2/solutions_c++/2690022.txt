typedef long long int lli;
typedef long double ld;
 
using namespace std;
 
lli bin[1010][1010];
int m[]={1,0,0,0,0,0,0,0,8,0,7,9,8,1,5,0,8,7,1};
lli MOD;
void init()
{
	lli p=1;
	MOD=0;
	for(int i=0;i<=18;i++) MOD=MOD*10+m[i];
	//cout<<MOD<<endl;
	mset(bin,0);
	for(int i=0;i<=1000;i++) bin[i][0]=1,bin[i][i]=1;
	for(int i=1;i<=1000;i++)
	for(int j=1;j<i;j++)
		bin[i][j]=(bin[i-1][j-1]+bin[i-1][j])%MOD;
}
 
int arr[1010][3];
int main()
{
    //freopen("inp.txt","r",stdin);
    init();
    int t;
    int n,m,k,l;
    int c1,c2,c3;
    lli sum;
    char str[10];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d%d%d%d",&n,&m,&l,&k);
    	
    	mset(arr,0);
    	for(int i=0;i<n;i++)
    		{
    			scanf("%s",str);
    			for(int j=0;j<m;j++) arr[i][j]=str[j]-'0';
    		}
    	if(k==1)
    	{
    		c1=c2=c3=0;
    		for(int i=0;i<n;i++)
    			{
    				if(arr[i][0]) c1++;
    				if(arr[i][1]) c2++;
    				if(arr[i][2]) c3++;
    			}
    		//cout<<MOD<<endl;
    		sum=(bin[c1][l]+bin[c2][l]+bin[c3][l])%MOD;
    	}
    	if(k==2)
    	{
    		c1=c2=c3=0;
    		for(int i=0;i<n;i++)
    			{
    				if(arr[i][0]&&arr[i][1]) c1++;
    				if(arr[i][1]&&arr[i][2]) c2++;
    				if(arr[i][2]&&arr[i][0]) c3++;
    			}
    		sum=(bin[c1][l]+bin[c2][l]+bin[c3][l])%MOD;
    	}
    	if(k==3)
    	{
    		c1=0;
    		for(int i=0;i<n;i++) if(arr[i][0]&&arr[i][1]&&arr[i][2]) c1++;
    		sum=bin[c1][l];
    	}
    	printf("%lld\n",sum);
    }
    return 0;
    
}
