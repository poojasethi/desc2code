#include <cstdio>
#include <iostream>
using namespace std;

typedef unsigned long long ll;

const ll num = 1000000080798150871ULL;
const int Maxn = 1001, Maxm = 3;

ll t, n, m, l, k, C[Maxn][Maxn];
char one[Maxn][Maxm];

ll Answer1()
{
    ll ans[Maxm], sum = 0;
    for (int j = 0; j < m; j++) ans[j] = 0;
    for (int i = 0; i < n; i++)
       for (int j = 0; j < m; j++)
          if (one[i][j] == '1') ans[j]++;
    for (int j = 0; j < m; j++)
       sum = (sum + C[ans[j]][l]) % num;
    return sum;
}

ll Answer2()
{
    ll ans[Maxm][Maxm], sum = 0;
    for (int j = 0; j < m; j++)
       for (int j2 = j+1; j2 < m; j2++) ans[j][j2] = 0;
    for (int i = 0; i < n; i++)
       for (int j = 0; j < m; j++)
          for (int j2 = j+1; j2 < m; j2++)
             if (one[i][j] == '1' && one[i][j2] == '1') ans[j][j2]++;
    for (int j = 0; j < m; j++)
       for (int j2 = j+1; j2 < m; j2++)
          sum = (sum + C[ans[j][j2]][l]) % num;
    return sum;
}

ll Answer3()
{
    ll ans = 0, sum = 0;
    for (int i = 0; i < n; i++)
       if (one[i][0] == '1' && one[i][1] == '1' && one[i][2] == '1')
          ans++;
    return C[ans][l];
}

int main()
{
    for (int i = 0; i < Maxn; i++) {
        C[i][0] = 1;
        for (int j = 1; j <= i; j++)
           C[i][j] = (C[i-1][j-1] + C[i-1][j]) % num;
    }
    scanf("%d", &t);
    while (t--) {
          scanf("%d %d %d %d", &n, &m, &l, &k);
          for (int i = 0; i < n; i++) scanf("%s", one[i]);
          if (l > n || k > m) { cout << "0\n"; continue; }
          switch (k) {
                 case 1: cout << Answer1() << endl; break;
                 case 2: cout << Answer2() << endl; break;
                 case 3: cout << Answer3() << endl; break;
          }
    }
    return 0;
}
