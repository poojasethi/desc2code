#include <cstdio>
#include <iostream>

typedef long long ll;
#define MOD 1000000080798150871LL

using namespace std;

int N, M, L, K;
char W[1010][4];

ll comb[1010][1010];
int bc[1<<3];

void bit_count(){
    for(int i = 0; i < (1<<3); i++){
        bc[i] = bc[i>>1];
        if(i & 1)bc[i]++;
    }
}

void precal(){
    for(int i = 0; i <= 1000; i++)comb[i][0] = 1;

    for(int i = 1; i <= 1000; i++){
        for(int j = 1; j <= i; j++){
            comb[i][j] = (comb[i - 1][j] + comb[i - 1][j - 1]) % MOD;
        }
    }
}

void solve(){
    ll ans = 0;

    for(int mask = 0; mask < (1<<M); mask++){
        if(bc[mask] != K) continue;

        int r[N];
        for(int i = 0; i < N; i++)r[i] = 1;
        for(int i = 0; i < N; i++)
            for(int j = 0; j < M && r[i]; j++)
                if(mask & (1<<j)) r[i] &= W[i][j];

        int cnt = 0;
        for(int i = 0; i < N; i++)if(r[i])cnt++;
        ans = (ans + comb[cnt][L]) % MOD;
    }

    cout << ans << endl;
}

int main(){
    precal();
    bit_count();

    int T;
    cin >> T;
    while(T--){
        cin >> N >> M >> L >> K;
        for(int i = 0; i < N; i++){
            cin >> W[i];
            for(int j = 0; j < M; j++)
                W[i][j] -= '0';
        }
        solve();
    }

    return 0;
}
