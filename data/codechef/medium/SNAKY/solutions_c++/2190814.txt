#include<stdio.h>
#include <memory.h>
char s[1000001];
int t[1001][1001];
int main(){
	int t_cases;
	scanf("%d",&t_cases);	
	while(t_cases--){
		int n,m,x,y,L,x1,y1,i,j,k;
		scanf("%d%d%d%d%d\n",&n,&m,&x,&y,&L);
		gets(s);
		x--;
		y--;
		
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
				t[i][j]=0;
		t[x][y]=1;
		for(i=0;i<L-1;i++)
		{
			x1=0;
			y1=0;
			switch(s[i]){
				case 'L': x1=-1;
							break;
				case 'R': x1=1;
							break;
				case 'U': y1=1;
							break;
				case 'D': y1=-1;
							break;			
				}
			x=x+x1;
			y=y+y1;
			t[x][y]=i+2;
		}

		k=1;
		while(1)
		{
			x=x+x1;
			y=y+y1;
			if(!(0<=x && x<n && 0<=y && y<m))
				 break;
			if(t[x][y]>k)
				 break;
			k++;
		}
		printf("%s %d\n",0<=x && x<n && 0<=y && y<m?"BODY":"WALL",k-1);
	}
	return 0;
}
