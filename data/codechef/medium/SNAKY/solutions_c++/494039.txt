#include <stdio.h>
#include <memory.h>

const int N = 1000;
char s[N*N+1];
int t[N][N];

int main()
{
	int TST;
	scanf("%d",&TST);
	for(int tst=0;tst<TST;tst++)
	{
		int n,m,x,y,L;
		scanf("%d%d%d%d%d\n",&n,&m,&x,&y,&L);
		x--;y--;
		gets(s);
		int i,k;
		for(i=0;i<n;i++)
			memset(t[i],0,4*m);
		int dx,dy;
		t[x][y]=1;
		for(i=0;i<L-1;i++)
		{
			dx=0,dy=0;
			if(s[i]=='L') dx=-1; else
			if(s[i]=='R') dx=+1; else
			if(s[i]=='D') dy=-1; else dy=+1;
			x+=dx;
			y+=dy;
			t[x][y]=i+2;
		}
		for(k=1;;k++)
		{
			x+=dx;
			y+=dy;
			if(!(0<=x && x<n && 0<=y && y<m)) break;
			if(t[x][y]>k) break;
		}
		printf("%s %d\n",0<=x && x<n && 0<=y && y<m?"BODY":"WALL",k-1);
	}
	return 0;
}
