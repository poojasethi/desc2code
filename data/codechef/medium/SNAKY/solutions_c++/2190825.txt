#include<stdio.h>
#include <memory.h>
char s[1000001];
int t[1001][1001];
inline void fastRead_int(int &x) {
register int c = getchar_unlocked();
x = 0;
int neg = 0;
for(; ((c<48 || c>57) && c != '-'); 
c = getchar_unlocked());
if(c=='-') {
neg = 1;
c = getchar_unlocked();
}
for(; c>47 && c<58 ; c = getchar_unlocked()) {
	x = (x<<1) + (x<<3) + c - 48;
    }
if(neg)
x = -x;
}

inline void fastRead_string(char *str)
{
register char c = 0;
register int i = 0;

while (c < 33)
	c = getchar_unlocked();

while (c != '\n') {
	str[i] = c;
    c = getchar_unlocked();
	i = i + 1;
}

str[i] = '\0';
}



int main(){
	int t_cases;
	//scanf("%d",&t_cases);	
	fastRead_int(t_cases);
	while(t_cases--){
		int n,m,x,y,L,x1,y1,i,j,k;
		//scanf("%d%d%d%d%d\n",&n,&m,&x,&y,&L);
		fastRead_int(n);
		fastRead_int(m);
		fastRead_int(x);
		fastRead_int(y);
		fastRead_int(L);	
		//gets(s);
		fastRead_string(s);
		x--;
		y--;
		
		for(i=0;i<n;i++)
			for(j=0;j<m;j++)
				t[i][j]=0;
		t[x][y]=1;
		for(i=0;i<L-1;i++)
		{
			x1=0;
			y1=0;
			switch(s[i]){
				case 'L': x1=-1;
							break;
				case 'R': x1=1;
							break;
				case 'U': y1=1;
							break;
				case 'D': y1=-1;
							break;			
				}
			x=x+x1;
			y=y+y1;
			t[x][y]=i+2;
		}
	
		k=1;
		while(1)
		{
			x=x+x1;
			y=y+y1;
			if(!(0<=x && x<n && 0<=y && y<m))
				 break;
			if(t[x][y]>k)
				 break;
			k++;
		}
		printf("%s %d\n",0<=x && x<n && 0<=y && y<m?"BODY":"WALL",k-1);
	}
	return 0;
}
