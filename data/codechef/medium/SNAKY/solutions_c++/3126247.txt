#include<stdio.h>
int move(long int map[1001][1001],long int l,char ch,long int sx,long int sy,long int cord[][2],long int &status,long int &stat)
{
long int temp=0;
long int &c=stat;
map[cord[c][0]][cord[c][1]]=0;
if(c==0)
temp=l-1;
else
temp=c-1;
if(ch=='L')
{
cord[c][0]=cord[temp][0]-1;
cord[c][1]=cord[temp][1];
}
if(ch=='R')
{
cord[c][0]=cord[temp][0]+1;
cord[c][1]=cord[temp][1];
}
if(ch=='D')
{
cord[c][0]=cord[temp][0];
cord[c][1]=cord[temp][1]-1;
}
if(ch=='U')
{
cord[c][0]=cord[temp][0];
cord[c][1]=cord[temp][1]+1;
}
if(cord[c][0]<1 || cord[c][0]>sx || cord[c][1]<1 || cord[c][1]>sy)
{
status=1;return 0;
}
else if(map[cord[c][0]][cord[c][1]]==-1)
{status=0;return 0;
}
else
map[cord[c][0]][cord[c][1]]=-1;
c++;
return 1;
}
long int cord[1000000][2]={0,0},map[1001][1001]={0};
char pos[1000000]={'\0'};
int main()
{
int t;
scanf("%d",&t);
while(t)
{
for(int i=0;i<1001;i++)
for(int j=0;j<1001;j++)
map[i][j]=0;
long int sx,sy,tx,ty,l,status=0,count=0,stat=0;
char ch;
status=0;count=0;stat=0;
scanf("%ld%ld%ld%ld%ld",&sx,&sy,&tx,&ty,&l);
scanf("%s",pos);
map[tx][ty]=-1;
cord[0][0]=tx;
cord[0][1]=ty;
for(long int i=0;pos[i]!='\0';i++)
{
if(pos[i]=='L')
{
   map[tx-1][ty]=-1;tx--;
   cord[i+1][0]=tx;
   cord[i+1][1]=ty;
}
if(pos[i]=='R')
{
map[tx+1][ty]=-1;tx++;
cord[i+1][0]=tx;
cord[i+1][1]=ty;
}
if(pos[i]=='D')
{
map[tx][ty-1]=-1;ty--;
cord[i+1][0]=tx;
cord[i+1][1]=ty;
}
if(pos[i]=='U')
{
 map[tx][ty+1]=-1;ty++;
 cord[i+1][0]=tx;
 cord[i+1][1]=ty;
}
ch=pos[i];
}
count=0;
stat=0;
while(move(map,l,ch,sx,sy,cord,status,stat))
{
    count++;
	stat=stat%l;
}
if(status==1)
printf("WALL %ld\n",count);
if(status==0)
printf("BODY %ld\n",count);
t--;
}
}
