#include<iostream>
#include<queue>
using namespace std;
int area[1009][1009];
char str[1009*1009];
main()
{
      int t,x,y,m,n,l,i,j,len,count,sx,sy,temp;
      scanf("%d",&t);
      while(t--)
      {
            scanf("%d%d%d%d%d",&n,&m,&x,&y,&l);
            count=0;sx=x;sy=y;
            for(i=1;i<=n;i++)
            {
              for(j=1;j<=m;j++)
              area[i][j]=0;
            }
            scanf("%s",str);
            len=l-1;
            area[x][y]=1;queue<char> Q;
            for(i=0;i<len;i++)
            {
                if(str[i]=='U')y++;else if(str[i]=='D')y--;else if(str[i]=='R')x++;else x--;
                area[x][y]=1;
                Q.push(str[i]);
            }
            while(true)
            {
                area[sx][sy]=0;
                if(Q.front()=='U')sy++;else if(Q.front()=='D')sy--;else if(Q.front()=='R')sx++;else sx--;
                Q.pop();
                Q.push(str[len-1]);       
                if(str[len-1]=='U')y++;else if(str[len-1]=='D')y--;else if(str[len-1]=='R')x++;else x--;
                if(x>n||y>m||x<1||y<1){printf("WALL %d\n",count);break;}else if(area[x][y]==1){printf("BODY %d\n",count);break;}else{area[x][y]=1;count++;}
            }
      }
}
