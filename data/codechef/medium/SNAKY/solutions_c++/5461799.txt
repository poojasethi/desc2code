#include <stdio.h>

using namespace std;

const int MAXN = 1010;

int g[MAXN][MAXN], dx[256], dy[256];
char str[MAXN*MAXN+1];

void move(int& x, int& y, char c) {
  x += dx[c];
  y += dy[c];//printf("x%dy%d",x,y);
}

int main() {
  dx['U'] = dx['D'] = dy['L'] = dy['R'] = 0;
  dx['R'] = dy['U'] = 1;
  dx['L'] = dy['D'] = -1;

  int t, sim = 0;
  scanf("%d", &t);
  while (t--) {
    int n, m, x, y, l;
    scanf("%d%d%d%d%d", &n, &m, &x, &y, &l);
    scanf("%s", str);
    g[x][y] = ++sim;//printf("sim%d",sim);
    for (int i = 0; str[i]; ++i) {
      move(x,y,str[i]);
      g[x][y] = ++sim;//printf("sim%d",sim);
    }
    int start = sim;
    while (true) {
      move(x,y,str[l-2]);
      if (x < 1 || x > n || y < 1 || y > m) {
	printf("WALL %d\n", sim-start);
	break;
      }
      if (g[x][y] + l > sim + 1) {
	printf("BODY %d\n", sim-start);
	break;
      }
      ++sim;
    }
  }
}
