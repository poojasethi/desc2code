using namespace std;

#define f(i, a, b) for(int i = a; i < b; i++)
#define rep(i, n) f(i, 0, n)

const int N = 1000000;

int T;
int rd[] = {-1, 0, 1, 0};
int cd[] = {0, 1, 0, -1};

char s[N + 1];
int occ[1000][1000];

int main(){

    scanf("%d", &T); while(T--) {

        memset(occ, 0, sizeof(occ));

        int R, C, r, c, l;
        scanf("%d %d %d %d %d", &C, &R, &c, &r, &l);
        scanf("%s", s);

        r = R - r, c = c - 1;
        occ[r][c] = 1;

        int dir;
        f(i, 0, l - 1) {

            if(s[i] == 'U') dir = 0;
            if(s[i] == 'R') dir = 1;
            if(s[i] == 'D') dir = 2;
            if(s[i] == 'L') dir = 3;

            r += rd[dir];
            c += cd[dir];
            occ[r][c] = i + 2;

        }

        int steps = 0;
        while(1) {

            r += rd[dir];
            c += cd[dir];

            if(r < 0 || r >= R || c < 0 || c >= C) {
                printf("WALL ");
                break;
            }
            if(occ[r][c] > steps + 1) {
                printf("BODY ");
                break;
            }

            steps++;

        }

        printf("%d\n", steps);
    }
}
