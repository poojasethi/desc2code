#include<iostream>
#include<cstring>
using namespace std;
int arr[1010][1010];
int main()
{
    //freopen("C:\\Users\\bhavini\\Documents\\input.txt","r",stdin);
    //freopen("C:\\Users\\bhavini\\Documents\\output.txt","w",stdout);
    int t;
    cin>>t;
    while(t--)
    {
              
              int N,M,x,y,len,i,l,lstx,lsty,c=0,j,lst=0;
              string s;
              cin>>N>>M>>y>>x>>l;
              for(i=0;i<=M;i++)
              {for(j=0;j<=N;j++)
              arr[i][j]=0;
              }
              char move;
              cin.ignore();
              arr[x][y]=1;
              lst=1;
              //cout<<l<<endl;
              cin>>s;
              
              len=s.length();
              //cout<<l<<endl;
              for(i=0;i<len;i++)
              {
                             lst+=1;
                             if(s[i]=='U')
                             {x+=1;arr[x][y]=lst;}
                             if(s[i]=='L')
                             {y-=1;arr[x][y]=lst;}
                             if(s[i]=='R')
                             {y+=1;arr[x][y]=lst;}
                             if(s[i]=='D')
                             {x-=1;arr[x][y]=lst;}
                             if(i==len-1)
                             {
                                       lstx=x;
                                       lsty=y;
                                       move=s[i];
                    
                             }
              }
              
              //cout<<move<<" "<<lstx<<" "<<lsty<<endl;
              if(move=='U')
              {
                           for(i=lstx+1;i<=M;i++)
                           {lst+=1;if(arr[i][lsty]>(lst-l))
                           {c=1;cout<<"BODY"<<" "<<(i-lstx-1)<<endl;break;}
                           else {arr[i][lsty]=lst;
                                }}
                           if(c==0)
                           cout<<"WALL "<<(M-lstx)<<endl;
              }
              else if(move=='D')
              {
                     for(i=lstx-1;i>0;i--)
                           {lst+=1;if(arr[i][lsty]>(lst-l))
                           {c=1;cout<<"BODY"<<" "<<(lstx-i-1)<<endl;break;}
                           else{arr[i][lsty]=lst;
                                }
                           } 
                           if(c==0)
                           cout<<"WALL "<<(lstx-1)<<endl;
              }
              else if(move=='L')
              {
                     for(i=lsty-1;i>0;i--)
                           {lst+=1;if(arr[lstx][i]>(lst-l))
                           {c=1;cout<<"BODY"<<" "<<(lsty-i-1)<<endl;break;}
                           else{arr[lstx][i]=lst;
                                }
                           } 
                           if(c==0)
                           cout<<"WALL "<<(lsty-1)<<endl;
              }
              else if(move=='R')
              {
                           for(i=lsty+1;i<=N;i++)
                           {lst+=1;if(arr[lstx][i]>(lst-l))
                           {c=1;cout<<"BODY"<<" "<<(i-lsty-1)<<endl;break;}
                           else{arr[lstx][i]=lst;
                                }
                           }
                           if(c==0)
                           cout<<"WALL "<<(N-lsty)<<endl;
              }
              /*for(i=1;i<=M;i++)
              {for(j=1;j<=N;j++)
              cout<<arr[i][j]<<" ";
              cout<<endl;
              }*/
    }
    return 0;
}
