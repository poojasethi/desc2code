#include<iostream>
#include<cstring>
using namespace std;
char len[1000002];
int dp[1002][1002];
int main()
{
      int t,n,m,x,y,l,i,j,mov;
    
      scanf("%d",&t);
      while(t--)
      {       
              mov=0;
              scanf("%d %d %d %d %d",&n,&m,&x,&y,&l);
              scanf("%s",len);
              for(i=0;i<=n+1;i++)
              {
                                 for(j=0;j<=m+1;j++)
                                 dp[i][j]=0;
              }
              for(i=0;i<l-1;i++)
              {
                             dp[x][y]=i+1;   
                                if(len[i]=='R')
                                x++;
                                else if(len[i]=='L')
                                x--;
                                else if(len[i]=='U')
                                y++;
                                else if(len[i]='D')
                                y--;
              }
             // dp[m-y][x-1]=0;
              while(dp[x][y]<=mov && (x-1)>=0 && (x-1)<=(n-1) && (m-y)>=0 && (m-y)<=(m-1))
              {
                         
                                if(len[l-2]=='R')
                                x++;
                                else if(len[l-2]=='L')
                                x--;
                                else if(len[l-2]=='U')
                                y++;
                                else if(len[l-2]='D')
                                y--;
                         mov++;
                         
                         
                        /* for(i=0;i<m;i++)
                         {

                                         printf("%d",dp[i][j]);
                                         printf("\n");
                         }*/
                        
              }
              if((x-1)>=0 && (x-1)<=(n-1) && (m-y)>=0 && (m-y)<=(m-1))
              printf("BODY "); 
              else
              printf("WALL ");
              mov--;
             printf("%d\n",mov);
      }
      return 0;
}
