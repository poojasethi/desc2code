#define DEBUG 0

#include <stdio.h>
#include <string.h>
using namespace std;

int g[1000][1000];

int main(void) {
  int tc, X, Y, x, y, l, i, c;

  scanf("%d", &tc);
  while(tc--) {
    scanf("%d%d%d%d%d", &X, &Y, &x, &y, &l);
    memset(g, 0, sizeof(g));
    x--,y--;
    g[x][y]=1;
    if(DEBUG)
      printf("%d,%d=%d\n", x+1, y+1, g[x][y]);
    while(getchar()!='\n');
    for(i=2;i<l+1;i++) {
      c=getchar();
      if(c=='U')
        y++;
      else if(c=='R')
        x++;
      else if(c=='L')
        x--;
      else if(c=='D')
        y--;
      g[x][y]=i;
      if(DEBUG)
	printf("%d,%d=%d\n", x+1, y+1, g[x][y]);
    }
    for(;;i++) {
      if(c=='U')
        y++;
      else if(c=='R')
        x++;
      else if(c=='L')
        x--;
      else if(c=='D')
        y--;
      if(x<0 || y<0 || x>=X || y>=Y)
        break;
      if(g[x][y]>i-l)
        break;
      g[x][y]=i;
      if(DEBUG)
	printf("%d,%d=%d\n", x+1, y+1, g[x][y]);
    }
    if(x<0 || y<0 || x>=X || y>=Y)
      printf("WALL %d\n", i-l-1);
    else if(g[x][y]>i-l)
      printf("BODY %d\n", i-l-1);
  }
  return 0;
}
