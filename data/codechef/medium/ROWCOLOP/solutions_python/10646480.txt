#include <iostream>
#include<stdio.h>
#include<string.h>
#include<ctype.h>
using namespace std;


int main()
{
    long int i,j,x,n,q,r,c,maxrow=0,maxcol=0;
    char s[10];
    scanf("%ld%ld",&n,&q);
    long int row[n+1],col[n+1];
        for(j=0;j<n;j++)
        {
            row[j]=0;
            col[j]=0;
        }
    for(j=0;j<q;j++)
    {
        cin>>s;
        if(!strcmp(s,"RowAdd"))
        {
            scanf("%ld",&r);
            scanf("%ld",&x);
                row[r-1]=row[r-1]+x;
                if(maxrow<row[r-1])
                {
                    maxrow=row[r-1];
                }
        }
        else
        {
            scanf("%ld",&c);
            scanf("%ld",&x);
                col[c-1]=col[c-1]+x;
                if(maxcol<col[c-1])
                {
                    maxcol=col[c-1];
                }
        }
    }
    printf("%ld\n",maxrow+maxcol);
    return 0;
}
