#include<stdio.h>
#include<string.h>
#define max 314160
int main(){
	int n ,q ,i;
	char s[7];
	scanf("%d%d" ,&n ,&q);
	int row[n] ,col[n];
	memset(row ,0 ,sizeof(row));
	memset(col ,0 ,sizeof(col));
	while(q--){
		int x ,y;
		scanf("%s %d %d" ,s ,&x ,&y);
		if(strcmp(s ,"RowAdd")==0)	row[x] +=y;
		else if(strcmp(s ,"ColAdd")==0)	col[x] +=y;
	}
	int maxR =0 ,maxC=0;
	for(i=0;i<n;i++){
		if(row[i]>maxR)	maxR = row[i];
		if(col[i]>maxC)	maxC = col[i];
	}
	printf("%d" ,maxR+maxC);
	return 0;
}
