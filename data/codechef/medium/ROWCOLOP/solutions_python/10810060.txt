#include<iostream>
#include<climits>
using namespace std;
int main()
{int N=0,q=0,x=0,add=0;
int RMax=0,CMax=0;
cin>>N>>q;
string a;
int Rar[N];
int Car[N];
for (int i = 0; i < N; i++)
  {for (int j = 0; j < N; j++)
    {Rar[i] = 0;
    Car[i] = 0;}}
while(q--)
{cin>>a;
if(a=="RowAdd")
{cin>>x;
cin>>add;
Rar[x-1]=Rar[x-1]+add;
if(Rar[x-1]>=RMax)
RMax=Rar[x-1];}
else if(a=="ColAdd")
{cin>>x;
cin>>add;
Car[x-1]=Car[x-1]+add;
if(Car[x-1]>CMax)
CMax=Car[x-1];}
}
cout<<RMax+CMax;
return 0;
}
