#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define l long
#define pf printf
#define sf scanf
int main()
{
    int n,q,x,r,c,i;
    sf("%d%d",&n,&q);
    char s[7];
    int row[n],col[n];
    for(i=0;i<n;i++){row[i]=0;col[i]=0;}
    while(q--)
    {
        sf("%s",s);
        if(s[0]=='R')
        {
            sf("%d%d",&r,&x);
            row[r-1]+=x;
        }
        else
        {
            sf("%d%d",&c,&x);
            col[c-1]+=x;
        }
    }
    sort(row,row+n);
    sort(col,col+n);
    pf("%d\n",row[n-1]+col[n-1]);

    return 0;
}
