#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,q,i;
    char s[10];
    scanf("%d%d",&n,&q);
    long long int r[n+1],c[n+1],x,y,ans,ans1;
    for(i=1;i<=n;i++)
    {
        r[i]=0;
        c[i]=0;
    }
    while(q--)
    {
        scanf("%s",s);
        scanf("%lld%lld",&x,&y);
        if(s[0]=='R')
            r[x]+=y;
        else
            c[x]+=y;
    }
    ans=0;
    ans1=0;
    for(i=1;i<=n;i++)
        ans=max(ans,r[i]);
    for(i=1;i<=n;i++)
        ans1=max(ans1,c[i]);
    printf("%lld",ans+ans1);
    return 0;
}
