#include<cstdio>
#include<cstring>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
typedef struct point{
    int x,y;
}point;

int main()
{
    int n,q;
    scanf("%d %d",&n,&q);
    int row[n],col[n];
    for(int i=0;i<n;i++){
        row[i]=0;
        col[i]=0;
    }
    for(int i=0;i<q;i++){
        char str[15];
        int in,X;
        scanf("%s %d %d",&str,&in,&X);
        in--;
        if(strcmp(str,"RowAdd")==0){
            row[in]+=X;
        }else
        {
            col[in]+=X;
        }
    }
    int mr=0,mc=0;
    for(int i=0;i<n;i++){
        if(mr<row[i]){
            mr=row[i];
        }
        if(mc<col[i]){
            mc=col[i];
        }
    }
    printf("%d\n",mr+mc);
}
