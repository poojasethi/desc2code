#include <iostream>
using namespace std;
int main(){
        long long int n,q,r,x;
        cin>>n>>q;
        long long int row[n],col[n],maxr=0,maxc=0,i=0;
        string s;
        while(i<n){
            row[i]=col[i]=0;
            i++;
        }
        while(q--){
            cin>>s>>r>>x;
            if(s=="RowAdd")
                row[r]+=x;
            else
                col[r]+=x;
        }
    i=0;
    while(i<n){
        if(row[i]>=maxr)
            maxr=row[i];
        if(col[i]>=maxc)
            maxc=col[i];
        i++;
    }
    cout<<maxr+maxc<<"\n";
}
