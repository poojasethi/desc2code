#include <bits/stdc++.h>
using namespace std;
typedef long long int LL;
const LL N=314180;
int t,i,ans,y,x,r[N],c[N],n;
string s;
int main()
{
   cin>>n;
   cin>>t;
   for(i=1;i<=n;i++)
    {
        c[i]=r[i]=0;
    }
   while(t--)
   {
       cin>>s>>y>>x;
       if(s=="RowAdd")
       {
           r[y]=r[y]+x;
       }
       else
       {
           c[y]=c[y]+x;
       }
    }
       sort(c+1,c+n+1);
       ans=0;
       for(i=1;i<=n;i++)
       {
           ans=max(r[i]+c[n],ans);
       }
    cout<<ans<<endl;
    return 0;
}
