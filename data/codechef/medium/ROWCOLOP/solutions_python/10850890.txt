#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define l long
#define pf printf
#define sf scanf
int main()
{
    int n,q,x,r,c,i;
    sf("%d%d",&n,&q);
    char s[7];
    int row[314159]={0},col[314159]={0};
    while(q--)
    {
        sf("%s",s);
        if(s[0]=='R')
        {
            sf("%d%d",&r,&x);
            row[r-1]+=x;
        }
        else
        {
            sf("%d%d",&c,&x);
            col[c-1]+=x;
        }
    }
    sort(row,row+314159);
    sort(col,col+314159);
    pf("%d\n",row[314158]+col[314158]);

    return 0;
}
