#include<bits/stdc++.h>
#include<string>
using namespace std;
typedef long long int ll;
ll r[320000]={0};
ll c[320000]={0};

int main() {
    ll n,q,m1=0,m2=0,a,b;
    string str;
    cin>>n>>q;
    
    while(q--) {
        cin>>str;
        if(str[0]=='R') {
            cin>>a>>b;
            r[a]+=b;
            if(r[a]>m1)
                m1=r[a];
        }
        else {
            cin>>a>>b;
            c[a]+=b;
            if(c[a]>m2)
                m2=c[a];
        }
    }
    
    if(m1==0)
        cout<<m2;
    else if(m2==0)
        cout<<m1;
    else
        cout<<m1+m2;
return 0;
}
