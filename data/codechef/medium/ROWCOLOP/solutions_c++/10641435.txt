#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main()
{
    long int n, x, q, y, maxR = 0, maxC = 0, i;
    string op;
    ios::sync_with_stdio(0);
    cin>>n>>q;
    vector<long int> r(n+1), c(n+1);
    while(q--)
    {
        cin>>op>>y>>x;
        if (op == "RowAdd")
        {
            r[y] += x;
        }
        else
        {
            c[y] += x;
        }
    }
    for (i = 0; i <= n; i++)
    {
        if (r[i] >= maxR)
        {
            maxR = r[i];
        }
    }
    for (i = 0; i <= n; i++)
    {
        if (c[i] >= maxC)
        {
            maxC = c[i];
        }
    }
    cout<<maxR+maxC<<"\n";
}
