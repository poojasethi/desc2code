#include <bits/stdc++.h>
using namespace std;
int main()
{
	
	int ROW[314160]={0},COL[314160]={0},i,j,k;
	int n,q,r,c,x;
	scanf("%d %d",&n,&q);
	while(q--)
	{
		string s;
		cin>>s;
		if(s=="RowAdd")
			{
				scanf("%d %d",&r,&x);
				ROW[r-1]+=x;
			}
		else
			{
				scanf("%d %d",&c,&x);
				COL[c-1]+=x;
			}
	}
	sort(ROW,ROW+n);
	sort(COL,COL+n);
	printf("%d\n",ROW[n-1]+COL[n-1]);

	return 0;
}
