#include<bits/stdc++.h>
using namespace std;

int n,m,k1,k2,r[100][100];

int main()
{
	long int n,q,r,c,x; char a[30];
	cin>>n>>q; long int max1=0,max2=0;
	long int row[n],col[n];
	for (int i = 0 ;i<n;i++)
	{
		row[i]=0; col[i]=0;
	}
	while (q--)
	{
		cin>>a;
		if (strcmp(a,"RowAdd")==0)
		{
			cin>>r>>x; row[r-1]+=x; 
			max1 = max(row[r-1],max1);
		}
		if (strcmp(a,"ColAdd")==0)
		{
			cin>>c>>x; col[c-1]+=x;
			max2=max(col[c-1],max2);
		}
	}
	cout<<max1+max2<<endl;
	return 0;
}
