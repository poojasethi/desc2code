#include <stdio.h>
#include <string>
#include <algorithm>
#include <iostream>
using namespace std;

int main() {
    long long int n,q;
    scanf ("%lld %lld",&n,&q);
    long long int row[n]={0},column[n]={0};
    while (q--)
    {
        string s;
        long long int num,x;
        cin>>s>>num>>x;
        if (s=="RowAdd")
            row[num-1]=row[num-1]+x;
        else column[num-1]=column[num-1]+x;
    }
    sort(row,row+n);
    sort(column,column+n);
    printf ("%lld",row[n-1]+column[n-1]);
	return 0;
}
