#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,q,p1,q1;
    cin>>n>>q;

    int row[n+1],col[n+1];
    for(int i=0;i<=n;i++)
    {
        row[i]=0;
        col[i]=0;
    }

    while(q--)
    {


        string s;
       cin>>s;
        cin>>p1>>q1;

        if(s[0]=='R')
        {

            row[p1]+=q1;
        }
        if(s[0]=='C')
        {
             col[p1]+=q1;
        }

    }


    cout<<(*max_element(row+1,row+n+1))+(*max_element(col+1,col+n+1))<<endl;
    return 0;
}
