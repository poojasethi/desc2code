#include <bits/stdc++.h>
using namespace std;

int main() {
	int N, Q, i, X, totalRow[400000] = {0}, totalCol[400000] = {0};
	long int maxRow = 0, maxCol = 0;
	char op[100], c;
	cin>>N>>Q;
	while(Q--) {
		cin>>op>>i>>X;
		if(!strcmp(op, "RowAdd")) {
			totalRow[i-1] += X;
			if(totalRow[i-1] > maxRow) {
				maxRow = totalRow[i-1];
			}
		}else{
			totalCol[i-1] += X;
			if(totalCol[i-1] > maxCol) {
				maxCol = totalCol[i-1];
			}
		}
	}
	cout<<(long int)(maxRow+maxCol)<<endl;
	return 0;
}
