#include<iostream>
#include<string.h>
using namespace std;

int main()
{
    int n,q;
    cin>>n>>q;
    int row[n]={0},col[n]={0};
    while(q--)
    {
        int rc,in;
        char str[20];
        cin>>str>>rc>>in;
        int x=strcmp("RowAdd",str);
        if(x==0)
        {
            row[rc]+=in;
        }
        else
        {
            col[rc]+=in;
        }
    }
    int maxr=row[0],maxc=col[0];
    for(int i=1;i<n;i++)
    {
        if(row[i]>maxr)
        maxr=row[i];
        if(col[i]>maxc)
        maxc=col[i];
    }
    cout<<maxr+maxc;
}
