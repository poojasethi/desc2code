#include <bits/stdc++.h>
using namespace std;
long long int row[314159];
long long int column[314159];
int main() {
	long long int n , q;
	cin >> n >> q;
	long long int rmax = 0, cmax = 0;
	for(int i=0;i<q;i++) {
		string s;
		long long int a, b;
		cin >> s >> a >> b;
		if(s=="RowAdd") {
			row[a-1] += b;
			if(row[a-1] > rmax)
				rmax = row[a-1];
		} 
		else {
			column[a-1] += b;
			if(column[a-1] > cmax)
				cmax = column[a-1];
		}
	}
	cout << rmax + cmax << endl;
	return 0;
}
