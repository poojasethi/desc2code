import java.util.*;
import java.lang.*;
import java.io.*;
//import java.math.BigInteger;

/* Name of the class has to be "Main" only if the class is public. */
class Codechef
{
	public static void main (String[] args) throws java.lang.Exception
	{
		// your code goes here
		Scanner sc = new Scanner(System.in);
		String x[]=sc.nextLine().split(" ");
		int n=Integer.parseInt(x[0]);
		int q=Integer.parseInt(x[1]);
		long row[] = new long[n+1];
		long col[] = new long[n+1];
		Arrays.fill(row,0);
		Arrays.fill(col,0);
		while(q>0){
		    String operation[]=sc.nextLine().split(" ");
		    if(operation[0].equals("RowAdd")){
		        int r=Integer.parseInt(operation[1]);
		        long num=Long.parseLong(operation[2]);
		        row[r]+=num;
		    }
		    else{
		        int c=Integer.parseInt(operation[1]);
		        long num=Long.parseLong(operation[2]);
		        col[c]+=num;
		    }q--;
		}Arrays.sort(row);
		Arrays.sort(col);
		long ans=row[n]+col[n];
		System.out.println(ans);
	}
}
