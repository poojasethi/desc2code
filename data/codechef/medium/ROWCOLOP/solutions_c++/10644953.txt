#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,b) for(int i=(a);i<(b);++i)

typedef long long int ll;

int main() {
	int n,q;
	string s;
	int i, val;
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	cin>>n>>q;
	int R[n+1]={}, C[n+1]={};
	while(q--){
		cin>> s >> i >> val;
		if(s == "RowAdd")
			R[i] += val;
		else C[i] += val;
	}
	ll ans = *max_element(R,R+n) + *max_element(C,C+n);
	cout << ans << "\n";
	return 0;
}
