    #include <stdio.h>
    char str1[100002], str2[100002];
    int arr1[100001][26], arr2[100001][26];
    int main()
    {
    	int t;
    	scanf ("%d", &t);
    	while (t--)
    	{
    		long long int i, j, k, l, d, n, flag=0;
    		scanf ("%lld %lld", &n, &d);
    		scanf ("%s", str1);
    		scanf ("%s", str2);
    		if (d>n)
    			d=n+1;
    		for (i=0; i<d; i++)
    		{
    			for (j=0; j<26; j++)
    			{
    				arr1[i][j]=arr2[i][j]=0;
    			}
    		}
    		for (i=0; i<n; i++)
    		{
    			arr1[i%d][str1[i]-97]++;
    			arr2[i%d][str2[i]-97]++;
    		}
    		for (i=0; i<d; i++)
    		{
    			for (j=0; j<26; j++)
    			{
    				if (arr1[i][j]!=arr2[i][j])
    				{
    					flag=1;
    					break;
    				}
    			}
    			if (flag==1)
    			{
    				break;
    			}
    		}
    		if (flag==1)
    		{
    			printf("No\n");
    		}
    		else
    		{
    			printf("Yes\n");
    		}
    	}
    }
