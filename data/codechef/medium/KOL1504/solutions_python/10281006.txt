#include<stdio.h>
#include<string.h>
int main()
{
int N,D,i,j,l,z=-1,t,T;          char a[100001],b[100001],temp;
scanf("%d",&T);
for(t=1;t<=T;t++)
{
scanf("%d%d%s%s",&N,&D,&a,&b);
l=strlen(a);
for(i=0;a[i]!='\0';)
{
  if(i==z)
    {
      break;
    }
  if(a[i]==b[i])
  {
    if(a[i+1]=='\0')
    {
       printf("Yes \n");
       break;
    }
    i++;
  }
  else
  {
    for(j=(i+D);;)
    {
      if(a[i]==b[j])
      {
        temp=b[i];
        b[i]=b[j];
        b[j]=temp;
        i++;
        break;
      }
      else
       {
        if(((j+D)>(l-1))||(b[j+1]=='\0'))
        {
          printf("No \n");
          z=i;
          break;
        }
          j=j+D;
      }
    }
  }
}
z=-1;
}
return 0;
}
