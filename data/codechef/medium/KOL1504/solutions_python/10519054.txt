#include <iostream>
using namespace std;

int main() {
	int testCase;
	cin >> testCase;
	for (int test = 0; test < testCase; test++) {
	int n, d;
	cin >> n >> d;
	string s1,s2;
	cin >> s1 >> s2;
	//cout << s1 << " " << s2 << endl;
	for (int i = 0; i < n; i++) {
		//cout << "a";
		if (s1[i] != s2[i]) {
			//cout << "a";
			for (int j = i+d; j < n; j += d) {
				if (s2[j] == s1[i]) {
					char tmp = s2[i];
					s2[i] = s2[j];
					s2[j] = tmp;
					//cout << "a";
					break;
				}
			}
		}
	}
	bool flag = true;
	for (int i = 0; i < n; i++) {
		if (s1[i] != s2[i]) {
			cout << "No";
			flag = false;
			break;
		}
	}
	if (flag) cout << "Yes";
	cout << "\n";
	}
	return 0;
}
