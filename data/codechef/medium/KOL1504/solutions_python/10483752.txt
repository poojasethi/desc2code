#include <iostream>
using namespace std;
int main()
	{
		int t;
		long int n,d;
		bool k;
		cin>>t;
		while (t--)
			{
				cin>>n>>d;
				char a[100001],b[100001];
				cin>>a>>b;
				for (long int i=0;i<n;i++)
					{
						k=0;
						for (long int j=i%d;j<n;j+=d)
							{
								if (a[i]==b[j])
									{
										b[j]='-';
										k=1;
										break;
									}
							}
						if (!k) break;
					}
				//for (long int i=0;i<n;i++) cout<<b[i];
				if (k) cout<<"Yes\n";
				else cout<<"No\n";
			}
	}
