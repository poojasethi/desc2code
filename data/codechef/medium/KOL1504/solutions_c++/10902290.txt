    #include<iostream>
     
    using namespace std;
     
    long int T;
    long int N;
    long int D;
    char str1[100001];
    char str2[100001];
     
    char ab[26];
    void set()
    {
    	for(int i=0;i<26;i++)
    	{
    		ab[i] = 0;
    	}
    }
     
    int check()
    {
    	for(int i=0;i<26;i++)
    	{
    		if( ab[i] != 0)
    		{
    			return -1;
    		}
    	}	
    	return 0;
    }
     
    int main()
    {
    	cin >> T;
    	for(long int i=0;i<T;i++)
    	{
    		int flag = 1;
    		cin >> N;
    		cin >> D;
    		cin >> str1;
    		cin >> str2;
    		for(long int j=0;j<N&&j<D;j++)
    		{
    			set();
    			long int k = 0;
    			while((j+k)<N)
    			{
    				int id1 = str1[j+k] - 'a';
    				int id2 = str2[j+k] - 'a';
    				ab[id1]++;
    				ab[id2]--;
    				k+=D;
    			}
    			if( check() != 0)
    			{
    				flag = 0;
    				break;
    			}
    		}
    		if(flag==1)
    		{
    			cout << "Yes"  << endl;
    		}
    		else
    		{
    			cout << "No" << endl;
    		}
    	}
    	return 0;	
    }
