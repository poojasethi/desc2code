#include<stdio.h>
#include<string.h>
int main()
{
   char s[100001],m[100001],temp;
   int t,i,j,flag=0;
   long d,n;
   scanf("%d",&t);
   while(t--)
   {
	scanf("%ld",&n);
	scanf("%ld",&d);
	scanf("%c",&temp);
	scanf("%s",s);
	scanf("%s",m);
	
	for(i=0;i<n;i++)
	{
	    flag=0;
	    for(j=i%d;j<n;j+=d)
              {
                if(s[i]==m[j])
		  {
			m[j]='0';
			flag=1;
			break;
       		   }
	       }
	       
	     if(flag==0)
		{
		   break;
		}
		
          }
          
	if(flag==0)
	printf("No\n");
	else
	printf("Yes\n");

    }
	return 0;
}
