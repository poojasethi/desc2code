#include <stdio.h>
using namespace std;
 
typedef long long ll;
 
int main(){
	int t;
	scanf("%d",&t);
	for (int i = 0; i < t; i ++){
		int n, d;
		char s1 [100005], s2 [100005];
		scanf("%d%d",&n,&d);
		scanf("%s",s1);
		scanf("%s",s2);
		bool result = true;
		for (int k = 0; k < n && k < d; k ++){
			int cnt [26];
			for (int j = 0; j < 26; j ++)
				cnt[j] = 0;
			for (int j = k; j < n; j += d){
				cnt[s1[j] - 'a'] ++;
				cnt[s2[j] - 'a'] --;
			}
			for (int j = 0; j < 26; j ++)
				if (cnt[j] != 0){
					result = false;
					break;
				}
		}
		if (result)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
