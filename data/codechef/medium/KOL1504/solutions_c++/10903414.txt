#include<iostream>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin >> t;
	while(t--)
	{
		int n,d,flag=0;
		char str1[100005],str2[100005];
		cin >> n >> d;
		cin >> str1 >> str2;
		if(d<=n)
		{
		for(int i=0;i<d;i++)
		{
			int c[30]={0},c1[30]={0};
			for(int j=i;j<n;j+=d)
			{
			c[str1[j]-'a']++;
			c1[str2[j]-'a']++;
			}
			for(int j=i;j<n;j+=d)
			{
				if(c[str1[j]-'a']!=c1[str1[j]-'a'])
				{
					flag=1;
					break;
				}
			}
			if(flag)
			break;
		}
		}
		else
		{
			for(int i=0;i<n;i++)
			{
				if(str1[i]!=str2[i])
					{
						flag=1;
						break;
					}
			}
		}
		if(flag)
		cout << "No\n";
		else
		cout << "Yes\n";
	}
	return 0;
}
