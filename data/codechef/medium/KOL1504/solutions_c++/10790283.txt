int checkalphabate(string A,string B,int n)
{
   int arr1[26];
   for(int i=0;i<26;i++)
        arr1[i]=0;

   int arr2[26];
   for(int i=0;i<26;i++)
       arr2[i]=0;

   for(int i=0;i<n;i++)
   {
     arr1[A[i]-97]++;
     arr2[B[i]-97]++;
   }

   for(int i=0;i<26;i++)
    {
       if(arr1[i]!=arr2[i])
       {
         return 0;
       }
    }
    return 1;
}
