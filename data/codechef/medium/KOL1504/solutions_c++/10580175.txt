void scanint(int &N)
{
    int ch,sign;
    N=0;
    while((ch<'0'||ch>'9') && ch!='-'&& ch!=EOF)
        ch=getchar_unlocked();
    if(ch=='-')
        sign=-1,ch=getchar_unlocked();
    else sign=1;
    do
        N=(N<<3)+(N<<1)+(ch-'0');
    while((ch=getchar_unlocked())>='0' && ch<='9');
    N*=sign;
    return;
}
