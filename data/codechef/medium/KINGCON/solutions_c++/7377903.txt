int find2(vector<int> v[],int vis[],int par[],int dis[],int low[],int ans[],int a)
{
	static int time=0;
	vis[a]=1;
	low[a]=dis[a]=++time;
	int chi=0;
	int i;
	for(i=0;i<v[a].size();i++)
	{
		if(!vis[v[a][i]])
		{
			chi++;
			par[v[a][i]]=a;
			find2(v,vis,par,dis,low,ans,v[a][i]);
			low[a]=min(low[a],low[v[a][i]]);
			if(par[a]==-1&&chi>1)
			ans[a]=1;
			if(par[a]!=-1&&low[v[a][i]]>=dis[a])
			ans[a]=1;
		}
		else if(v[a][i]!=par[a])
		{
			low[a]=min(low[a],dis[v[a][i]]);
		}
	}
}
int find(vector<int> v[],int n)
{
	int vis[3005]={0};
	int i;
	int par[3005],low[3005],ans[3005],dis[3005];
	for(i=0;i<n;i++)
	{
		par[i]=-1;
		dis[i]=0;
		low[i]=0;
		ans[i]=0;
	}
	//par[0]=-1;
	int sum=0;
	par[0]=-1;
	find2(v,vis,par,dis,low,ans,0);
	for(i=0;i<n;i++)
	{
		if(ans[i]==1)
		sum++;
	}
	return sum;
}
int main()
{
	int t,n,k,a,b;
	long long int m,i;
	vector<int> v[3005];
	scanf("%d",&t);
	while(t--)

		for(i=0;i<3005;i++)
		v[i].clear();
		scanf("%d %lld %d",&n,&m,&k);

		{
			scanf("%d %d",&a,&b);
			v[a].push_back(b);
			v[b].push_back(a);
		}
		int ans=find(v,n);
		long long int ans2=ans*k;

}
