vector<int> G[3002];
int ti,ans,v[3002],st[3002],anc[3002];
void dfs(int n,int par)
{
	int i,l=G[n].size(),c=0;
	bool f=false;
	v[n]=1;
	st[n]=anc[n]=++ti;
	REP(i,l)
	{
		int x=G[n][i];
		if(x==par)
			continue;
		if(!v[x])
		{
			dfs(x,n);
			c++;
			anc[n]=min(anc[n],anc[x]);
			if(anc[x]>=st[n])
				f=true;
		}
		else
			anc[n]=min(anc[n],st[x]);
	}
	if(n==0 && c>1)
			ans++;
	else if(f && n!=0)
		ans++;
}
int main()
{
	int t;
	si(t);
	while(t--)
	{
		int n,m,k,i,j;
		si2(n,m);
		si(k);
		REP(i,n)
		{
			G[i].clear();
			v[i]=0;
			st[i]=-1;
		}
		ti=0,ans=0;
		REP(i,m)
		{
			int x,y;
			si2(x,y);
			G[x].pb(y);
			G[y].pb(x);
		}
		dfs(0,-1);
		cout<<ans*k<<endl;
	}
	return 0;
}
