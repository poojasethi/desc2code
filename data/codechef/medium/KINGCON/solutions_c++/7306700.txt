#include<iostream>
#include<stdio.h>
#include<climits>
#include<cstring>
#include<math.h>
#include<algorithm>
#include<map>
#include<vector>
 
using namespace std;
 
vector<int> v[3005];
 
int counter;
int dfn[3005]={0},c,l[3005],co;
 
int recur(int n)
{
    int flag=0;
    dfn[n]=counter;
    l[n]=counter;
    //printf("%d\n",n);
    counter++;
    int temp;
    for(int i=0; i<v[n].size(); i++)
    {
        if(!dfn[v[n][i]])
        {
            if(n==0)
                co++;
            temp=recur(v[n][i]);
            if(l[n]>temp)
                l[n]=temp;
            if(temp>=dfn[n])
                flag=1;
        }
        else if(dfn[v[n][i]]<l[n])
            l[n]=dfn[v[n][i]];
    }
    if(flag==1)
        {
            c++;
            //printf("%d\n",n);
        }
    return l[n];
}
 
int main()
{
    int t,n,m,k,v1,v2,temp;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d",&n,&m,&k);
        counter=1;
        c=0;
        co=0;
        for(int i=0; i<m; i++)
        {
            scanf("%d %d",&v1,&v2);
            v[v1].push_back(v2);
            v[v2].push_back(v1);
        }
        temp=recur(0);
        if(co==1)
            c--;
        printf("%d\n",c*k);
        for(int i=0;i<n;i++)
        {
            v[i].clear();
            dfn[i]=0;
        }
    }
    return 0;
}
