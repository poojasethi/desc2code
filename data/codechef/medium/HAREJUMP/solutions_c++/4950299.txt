#include <bits/stdc++.h>
using namespace std;
long long MOD=1000000007;
int i, j, k,m=15;
void mult(long long a[][15],long long b[][15],long long c[][15]){
    for(i=0;i<m;i++){
    	for(j=0;j<m;j++){
			c[i][j]=0LL;
			for(k=0;k<m;k++)
				c[i][j]=(c[i][j]+a[i][k]*b[k][j])%MOD;
		}
	}
	return;
}
void fib_fn(long long n,long long mat[][15],long long ans[][15])
{
	long long temp[m][15];
	while(n)
	{
		if(n&1)
		{
			mult(ans,mat,temp);
			for(i=0;i<m;i++)
				for(j=0;j<m;j++)ans[i][j]=temp[i][j];
		}
		mult(mat,mat,temp);
		for(i=0;i<m;i++)
			for(j=0;j<m;j++)mat[i][j]=temp[i][j];
		n/=2;	
	}
	return ;
}
int main(){
	long long int t,n,te,i,j,k,arr[15][15],ans[15][15];
	cin>>t;
	for(te=0;te<t;te++){
		cin>>n>>k;
		for(i=0;i<15;i++)
			for(j=0;j<15;j++)arr[i][j]=ans[i][j]=0;
		for(i=1;i<15;i++)arr[i][i-1]=ans[i][i]=1;
        ans[0][0]=1;
		for(i=0;i<k;i++){
			cin>>j;
			arr[0][j-1]=1;		
		}
        fib_fn(n,arr,ans);
		cout<<ans[0][0]<<"\n";
	}
	return 0;
}
