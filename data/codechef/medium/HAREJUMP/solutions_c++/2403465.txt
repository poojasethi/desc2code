#include<stdio.h>

#define LLI long long int
#define MOD 1000000007
LLI c[16][16], b[16][16];

void clear(LLI a[16][16])
{
    for(int i =0;i < 16;i++ )
        for(int j =0;j < 16;j++)a[i][j] = 0;
}

void mul(LLI A[16][16], LLI B[16][16], LLI C[16][16])
{
    clear(C);
    for (size_t i = 0; i < 16; i++)
        for (size_t j = 0; j < 16; j++)
            for (size_t k = 0; k < 16; k++)
                C[i][j] = (C[i][j] + A[i][k] * B[k][j]) % MOD;
}

void nth_pow(LLI a[16][16], LLI l)
{
    if(l == 0)
    {
        clear(c);
        for(int i =0;i < 16;i++)c[i][i] = 1;
        return;
    }
    nth_pow(a,l/2);
    mul(c , c, b);
    for(int i = 0;i < 16;i++)
        for(int k = 0;k < 16; k++)c[i][k] = b[i][k];
    if(l & 1)
        mul(c, a, b);
    for(int i = 0;i < 16;i++)
        for(int k = 0;k < 16; k++)c[i][k] = b[i][k];
}

int main()
{
    int t, n, i, k;
    LLI l;
    scanf("%d",&t);
    while(t--)
    {
        LLI arr[16][16],ans[16][16];
        clear(arr);
        for(i = 0 ;i < 15; i++)arr[i+1][i] = 1;
        scanf("%lld",&l);
        scanf("%d",&k);
        for(i = 0;i < k;i++)
        {
            scanf("%d",&n);
            arr[0][n-1]++;
        }
        nth_pow(arr, l);
        printf("%lld\n",c[0][0]);
    }
    return 0;
}
