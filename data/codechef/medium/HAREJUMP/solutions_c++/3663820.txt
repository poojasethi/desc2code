#include<stdio.h>
#include<iostream>
using namespace std;
long long int MOD=1000000007;
long long int a[16][16];
long long int c[16][16];
void multi(long long int a[16][16],long long int b[16][16])
{
    int i,j,k;
    for(i=0;i<16;i++)
    {
        for(j=0;j<16;j++)
            c[i][j]=0;
    }
    for(i=0;i<16;i++)
    {
        for(j=0;j<16;j++)
        {
            for(k=0;k<16;k++)
                c[i][j]=(c[i][j]+a[i][k]*b[k][j])%MOD;
        }
    }
    for(i=0;i<16;i++)
    {
        for(j=0;j<16;j++)
            a[i][j]=c[i][j];
    }
    /*for(i=0;i<16;i++)
    {
        for(j=0;j<16;j++)
            cout<<a[i][j]<<"  ";
            cout<<endl;
    }
    cout<<endl;*/
}
void powerr(long long int a[16][16],long long int n)
{//cout<<"\n##"<<n<<"\n";
    if((n==0)||(n==1))
        return ;
    if((n%2)==0)
    {
        //cout<<"n="<<n<<"\n";
        multi(a,a);
        powerr(a,n/2);
    }
    else
    {
        long long int z[16][16];
        for(int i=0;i<16;i++)
        {
            for(int j=0;j<16;j++)
                z[i][j]=a[i][j];
        }
        powerr(a,n-1);
       // cout<<"n="<<n<<"\n";
        multi(a,z);
    }
    return ;
}

int main()
{
    int t;
    cin>>t;
    long long int n;
    int i,j,k,p;
    while(t>0)
    {
        t-=1;
        scanf("%lld",&n);
        scanf("%d",&k);
        for(i=0;i<16;i++)
        {
            for(j=0;j<16;j++)
                a[i][j]=0;
        }
        for(i=1;i<16;i++)
            a[i][i-1]=1;
        for(i=0;i<k;i++)
        {
            scanf("%d",&p);
            a[0][p-1]++;
        }
        powerr(a,n);
        printf("%d\n",a[0][0]);
    }
    return 0;
}
