#include<iostream>
using namespace std;

int absolute(int n){
	if (n < 0){
		n = n * (-1);
		return n;
	}
	else{
		return n;
	}
}

int gcd(int a, int b){
	if (a % b == 0){
		return b;
	}
	else{
		return gcd(b, a % b);
	}
}

int main(){
	int T, temp, i;
scanf("%d", & T);

	for (i = 0; i < T; i++){
		int w1, w2, n = 1, count = 0, rel_speed;
		//scanf("%d %d", &w1, &w2);
		cin >> w1 >> w2;

		rel_speed = absolute(w2 - w1);
		w1 = absolute(w1);
		w2 = absolute(w2);
		
		if (w1 > w2){
			temp = gcd(w1, w2);
			//count = temp / rel_speed;
		}
		else{
			temp = gcd(w2, w1);
		//	count = rel_speed / temp;
		}
		count = rel_speed / temp;
		//printf("%d\n", count);
		cout << count << endl;
	}
	return 0;
}
