#include<stdio.h>
int gcd(int a,int b)
{int temp=0;
		if(a<0)a=(-1)*a;
		if(b<0)b=(-1)*b;
		if(a<b){temp=a;a=b;b=temp;}
		if(b==0)return a;
		else return (gcd(b,a%b));
 
}
int main()
{
		int t=0;int w1,w2;int d=0;
		scanf("%d",&t);
		while(t--)
		{
				scanf("%d %d",&w1,&w2);
			d=(w1-w2)/gcd(w1,w2);
			d>0?printf("%d\n",d):printf("%d\n",(-1)*d);
		}	
		return 0;
}
