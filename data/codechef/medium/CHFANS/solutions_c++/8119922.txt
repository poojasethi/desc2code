    #include <bits/stdc++.h>
    using namespace std;
     
    int gcd(int a, int b) {
    if(b == 0) return a;
    return gcd(b, a%b);
    }
     
    int main() {
    int t;
    cin>>t;
    while(t--) {
    int a, b;
    scanf("%d%d", &a, &b);
    int g = gcd(abs(a), abs(b));
    a /= g;
    b /= g;
    printf("%d\n", abs(a-b));
    }
    return 0;
    }
