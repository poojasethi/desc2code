#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define sl(l) scanf("%lld",&l)

ll gcd(ll a,ll b)
{
	if(b==0)return a;
	else return gcd(b,a%b);
}


ll swap(ll &a,ll &b)
{
	a^=b^=a^=b;
}
int main()
{
	ll t;
	sl(t);
	while(t--)
	{
		ll a,b;
		sl(a);
		sl(b);
		ll c = abs(a);
		ll d = abs(b);
		if(d>c)swap(c,d);
		ll g = gcd(c,d);
		c/=g;
		d/=g;
	//	printf("%lld %lld\n",c,d);
		if(a*b<0)printf("%lld\n",c+d);
		else printf("%lld\n",abs(c-d));
	}
	return 0;
}
