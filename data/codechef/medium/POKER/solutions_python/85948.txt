#! /usr/bin/python

import sys

num = int(sys.stdin.readline())

def convert(char):
    if char == 'A':
        return 1
    elif char == 'T':
        return 10
    elif char == 'J':
        return 11
    elif char == 'Q':
        return 12
    elif char == 'K':
        return 13
    else:
        return int(char)

def royal(counts):
    if counts[10] and counts[11] and counts[12] and counts[13] and counts[1]:
        return True
    else:
        return False

def straight(counts):
    for i in range(1, 10):
        for j in range(5):
            if counts[i+j]:
                if j == 4:
                    return True
                else:
                    continue
            else:
                break
    return False

def hand(counts, ccounts, sameSuit):
    if royal(counts) and sameSuit:
        return 'royal flush'
    elif straight(counts) and sameSuit:
        return 'straight flush'
    elif ccounts[4] == 1:
        return 'four of a kind'
    elif ccounts[3] == 1 and ccounts[2] == 1:
        return 'full house'
    elif sameSuit:
        return 'flush'
    elif straight(counts) or royal(counts):
        return 'straight'
    elif ccounts[3] == 1:
        return 'three of a kind'
    elif ccounts[2] == 2:
        return 'two pairs'
    elif ccounts[2] == 1:
        return 'pair'
    else:
        return 'high card'
    

output = []
for i in range(num):
    input = sys.stdin.readline()
    input = input.split(' ')
    ccounts = [0]*5
    sameSuit = True
    counts = [0]*14
    result = ''
    
    counts[convert(input[0][0])] += 1
    for j in range(1, 5):
        if input[j-1][1] != input[j][1]:
            sameSuit = False

        counts[convert(input[j][0])] += 1
        ccounts[counts[convert(input[j][0])]] += 1
        ccounts[max(counts[convert(input[j][0])] - 1, 0)] -= 1

    output.append(hand(counts, ccounts, sameSuit));

for i in range(num):
    print output[i]
