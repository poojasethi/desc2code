def is_flush_pair(c1, c2):
   return c1[1] == c2[1]

def is_straight_pair(c1, c2):
   return c2[0] - c1[0] == 1

def is_match_pair(c1, c2):
   return c1[0] == c2[0]

def determine_hand(h):
   flush_count = 0
   straight_count = 0
   match_count = 0
   
   pair = False
   three = False
   four = False
   two_pairs = False
   
   c_p = h[0]
   for i in xrange(1, 5):
      c_c = h[i]
      
      if is_flush_pair(c_p, c_c):
         flush_count += 1
      if is_straight_pair(c_p, c_c):
         straight_count += 1
      if is_match_pair(c_p, c_c):
         match_count += 1
      else:
         if match_count == 1:
            if pair:
               two_pairs = True
            else:
               pair = True
         elif match_count == 2:
            three = True
         elif match_count == 3:
            four = True
         match_count = 0
      c_p = c_c
      
   if match_count != 0:
      if match_count == 1:
         if pair:
            two_pairs = True
         else:
            pair = True
      elif match_count == 2:
         three = True
      elif match_count == 3:
         four = True
   
   if flush_count == 4:
      if straight_count == 4:
         if h[0][0] == 0:
            return "royal flush"
         else:
            return "straight flush"
      elif straight_count == 3:
         if h[0][0] == 0 and h[1][0]==9:
            return "straight flush"
   if four:
      return "four of a kind"
   if pair and three:
      return "full house"
   if flush_count == 4:
      return "flush"
   if straight_count == 4:
      return "straight"
   elif straight_count == 3 and h[0][0] == 0 and h[1][0]==9:
      return "straight"
   if three:
      return "three of a kind"
   if two_pairs:
      return "two pairs"
   if pair:
      return "pair"
   return "high card"      
   

inp = sys.stdin.readline
