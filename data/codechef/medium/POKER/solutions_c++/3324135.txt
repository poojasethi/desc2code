void merge(int a[],int low,int mid,int high);
void mergesort(int a[],int low,int high){
if(low<high){
int mid=(low+high)/2;
mergesort(a,low,mid);
mergesort(a,mid+1,high);
merge(a,low,mid,high);
}
}
void merge(int a[],int low,int mid,int high){
int b[10];
int i=low,j=mid+1,k=low;
while(i<=mid && j<=high){
if(a[i]<=a[j]){
b[k]=a[i];
i++;
}
else{
b[k]=a[j];
j++;
}
k++;
}
if(i>mid){
for(int x=j;x<=high;x++){
b[k]=a[x];
k++;
}
}
else{
for(int x=i;x<=mid;x++){
b[k]=a[x];
k++;
}
}
for(int p=low;p<=high;p++){
a[p]=b[p];
}
}
