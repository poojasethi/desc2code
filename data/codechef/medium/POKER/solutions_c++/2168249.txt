#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
              string s[5];
              int a[20]={0}, suit[5]={0};
              for(int i=0;i<5;i++)
              {
                      cin>>s[i];
                      switch(s[i][0])
                      {
                                    case 'A':
                                         a[1]++;
                                         a[14]++;
                                         break;
                                    case 'K':
                                         a[13]++;
                                         break;
                                    case 'Q':
                                         a[12]++;
                                         break;
                                    case 'J':
                                         a[11]++;
                                         break;
                                    case 'T':
                                         a[10]++;
                                         break;
                                    default:
                                         a[s[i][0]-'0']++;
                      }
                      switch(s[i][1])
                      {
                                     case 'S':
                                          suit[1]++;
                                          break;
                                     case 'H':
                                          suit[2]++;
                                          break;
                                     case 'D':
                                          suit[3]++;
                                          break;
                                     case 'C':
                                          suit[4]++;
                                          break;
                      }
              }
              string ans[10]={"high card", "pair", "two pairs", "three of a kind", "straight", "flush", "full house", "four of a kind", "straight flush", "royal flush"};
              int i,flag=0,j;
              for(j=1;j<5;j++)
                               {
                                               if(suit[j]==5)
                                               {flag=5;
                                               break;}
                               }
              //cout<<s[flag]<<endl;
              for(i=1;i<15;i++)
              {
                               if(a[i]==2)
                               {
                                          flag=1;
                                          for(j=1;j<14;j++)
                                          {
                                                             if(a[j]==2&&i!=j)
                                                             {flag=2;
                                                             break;}
                                                             if(a[j]==3)
                                                             {flag=6;
                                                             break;}
                                          }
                                          
                               }
                               if(a[i]==3)
                               {
                                          if(flag!=6)
                                          flag=3;
                               }
                               if(a[i]==4)
                               flag=7;
                               
                                               
                               if(a[i]==1&&a[i+1]==1&&a[i+2]==1&&a[i+3]==1&&a[i+4]==1)
                               {
                                       flag=4;
                                       for(j=1;j<=4;j++)
                                       {
                                                        if(suit[j]==5)
                                                        {
                                                                      flag=8;
                                                                      if(i==10)
                                                                      flag=9;
                                                        }
                                       }
                               }
                               //cout<<flag<<endl;
              }
              cout<<ans[flag]<<endl;
    }
    return 0;
}
