#include <iostream>

using namespace std;

int main()
{
	int numTest = 0;
	scanf("%d",&numTest);
	while(numTest > 0)
	{
		numTest--;
		int card[5], suit[5];
		for(int i = 0; i < 5; i++)
		{
			char c = '0';
			char s = '0';
			cin>>c>>s;

			if(c == 'A')
				card[i] = 14;
			else if(c == 'K')
				card[i] = 13;
			else if(c == 'Q')
				card[i] = 12;
			else if(c == 'J')
				card[i] = 11;
			else if(c == 'T')
				card[i] = 10;
			else
				card[i] = c - 48;


			if(s == 'H')
				suit[i] = 1;
			else if(s == 'S')
				suit[i] = 2;
			else if(s == 'D')
				suit[i] = 3;
			else
				suit[i] = 4;		
		}
		 
		int pair = 0;

		int arrC[15] = {0};
		int arrS[15][5] = {0};
		int color = 0;
		for(int i = 0; i < 5; i++)
		{
			arrC[card[i]]++;
			arrS[card[i]][arrC[card[i]]-1]=suit[i];			
		}
		int j = 0;
		for(int i = 0; i < 15; i++)
		{
			int f = 0;
			while(arrC[i]>0)
			{
				card[j] = i;
				suit[j] = arrS[i][f];
				f++;
				j++;
				arrC[i]--;
			}
		}
		
		for(int i = 0; i < 4; i++)
			if(card[i] == card[i+1])
				pair++;

		if( (card[0] == card[1] && card[1] == card[2] && card[2] == card[3] ) || (card[1] == card[2] && card[2] == card[3] && card[3] == card[4] ))
		{
			printf("four of a kind\n");
			continue;
		}

		if( pair == 3 )
		{
			printf("full house\n");
			continue;
		}

		bool IsFlush = false;
		bool IsStraight = false;

		if( suit[0] == suit[1] && suit[1] == suit[2] && suit[2] == suit[3] && suit[3] == suit[4] )
			IsFlush = true;

		if( card[1]==card[0]+1 && card[2]==card[0]+2 && card[3]==card[0]+3 && card[4]==card[0]+4)
			IsStraight = true;

		else if (  card[0] == 2 && card[1] == 3 && card[2] == 4 && card[3] == 5 && card[4] == 14)
			IsStraight = true;

		if( IsFlush == true && IsStraight== true && card[3] == 13)
		{
			printf("royal flush\n");
			continue;
		}

		if( IsFlush == true && IsStraight== true )
		{
			printf("straight flush\n");
			continue;
		}

		if( IsFlush == true )
		{
			printf("flush\n");
			continue;
		}

		if( IsStraight== true )
		{
			printf("straight\n");
			continue;
		}

		if((card[0] == card[1]&& card[1] == card[2]) || (card[1] == card[2] && card[2] == card[3]) || (card[2] == card[3]&& card[3] == card[4]) )
		{
			printf("three of a kind\n");
			continue;
		}

		if( pair == 2 )
		{
			printf("two pairs\n");
			continue;
		}

		if( pair == 1 )
		{
			printf("pair\n");
			continue;
		}

		printf("high card\n");
	}
	return 0;
}
