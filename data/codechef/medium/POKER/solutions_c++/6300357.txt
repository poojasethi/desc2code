#include <cstdlib>
#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
#define gc getchar_unlocked
#define C 0
#define D 1
#define H 2
#define S 3

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

using namespace std; 

int main(){
	int T;
	scanint(T);
	string str[5];
	int suit[4] = {0};
	int rank[5] = {0};
	bool cont = true;
	bool seq = true;
	int count[14] = {0};
	int dir;
	char s;
	for(int t = 0; t < T; t++){
		for(int i = 0; i < 4; i++)
			suit[i] = 0;
		for(int i = 0; i < 5; i++)
			rank[i] = 0;
		for(int i = 0; i < 14; i++)
			count[i] = 0;
		cont = true;
		seq = true;
		for(int i = 0; i < 5; i++){
			cin >> str[i];
			if(i == 0)
				s = str[i][1];
			else if(cont && s == str[i][1])
				cont = true;
			else
				cont = false;
			if(str[i][str[i].length()-1] == 'C')
				suit[C]++;
			else if(str[i][str[i].length()-1] == 'D'){
				suit[D]++;
			}
			else if(str[i][str[i].length()-1] == 'H'){
				suit[H]++;
			}
			else
				suit[S]++;
			if(str[i][0] == 'K'){
				count[13]++;
				rank[i] = 13;
			}
			else if(str[i][0] == 'Q'){
				count[12]++;
				rank[i] = 12;
			}
			else if(str[i][0] == 'J'){
				count[11]++;
				rank[i] = 11;
			}
			else if(str[i][0] == 'A'){
				count[0]++;
				rank[i] = 0;
			}
			else if(str[i][0] == 'T'){
				count[10]++;
				rank[i] = 10;
			}
			else{
				count[str[i][0]-48]++;	
				rank[i] = str[i][0]-48;
			}
		}
		vector<int> myvector (rank, rank+5);     
  		sort(myvector.begin(), myvector.begin()+5);
		for(int i = 0; i < 5; i++){
			rank[i] = myvector[i];
		}
		dir = rank[1]-rank[0];
		if(dir != 1 && dir != 10 && dir != 2)
			seq = false;
		if(dir == 2 && rank[0] != 0)
			seq = false;
		else if(dir == 10 && rank[0] != 0)
			seq = false;
		int temp = 0;
		if(seq){
			for(int i = 2; i < 5; i++){
				temp = rank[i]-rank[i-1];
				if(temp != 1)
				{
					seq = false;
					break;
				}
			}
		}
		bool flush = false;
		bool three = false;
		bool two = false;
		bool brek = false;
		bool high = false;
		if(cont && count[0] && count[10] && count[11] && count[12] && count[13]){
			cout << "royal flush\n";
			continue;
		}
		else if(cont && seq){
			cout << "straight flush\n";
			continue;
		}
		else if(seq){
			cout << "straight\n";
			continue;
		}
		else{
			if(cont)
				flush = true;
			for(int i = 0; i < 14; i++){
				if(count[i] == 5){
					high = true;
				}
				else if(count[i] == 4){
					cout << "four of a kind\n";
					brek = true;
					break;
				}
				else if(count[i] == 2){
					if(three){
						cout << "full house\n";
						brek = true;
						break;
					}
					else if(two){
						if(flush){
							cout << "flush\n";
							brek = true;
							break;						
						}
						cout << "two pairs\n";
						brek = true;
						break;
					}
					else
						two = true;
				}
				else if(count[i] == 3){
					if(two){
						cout << "full house\n";
						brek = true;
						break;
					}
					else{
						three = true;
					}
				}
			}
		}
		if(brek)
			continue;
		if(high)
			cout << "high card\n";
		else if(flush){
			cout << "flush\n";
		}	
		else if(two){
		cout << "pair\n";
		}
		else if(three){
			cout << "three of a kind\n";			
		}
		else{
			cout << "high card\n";
		}
	}		
}
