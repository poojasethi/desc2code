#include <iostream>
#include <map>
using namespace std;
int main()
{
    int t, i, j, c[5], stat, inf2;
    bool flag, f2, f3;
    char temp[4];
    char ft[] = {'S', 'H', 'D', 'C'};
    char sd[] = {'A', '2', '3', '4', '5', '6', '7', '8', '9', 'T', 'J', 'Q', 'K'};
    cin >> t;
    while ( t-- )
    {
        map<char, int> r;
        map<char, int> s;
        stat = 0;
        for ( i=0; i<5; ++i )
        {
            cin >> temp;
            r[temp[0]]++;
            s[temp[1]]++;
        }
        if ( s['S']==5 || s['H']==5 || s['D']==5 || s['C']==5 )
        {
            if ( r['A'] && r['K'] && r['Q'] && r['J'] && r['T'] )
                stat = 1;
            else
            {
                flag = true;
                for ( i=0; ((i<8) && flag); ++i )
                {
                    for ( j=0; ((j<5) && flag); ++j )
                    {
                        if ( !r[sd[i+j]] ) break;
                        if ( j==4 ) flag = false;
                    }
                }
                for ( j=9; j<13; ++j )
                {
                    if ( !r[sd[j]] ) break;
                    if ( j==12 && r['A'] ) flag = false;
                }
                if ( !flag ) stat = 2;
                else stat = 5;
            }
        }
        else
        {
            for ( i=0; i<13; ++i )
            {
                if ( r[sd[i]]==4 )
                {
                    stat = 3;
                    break;
                }
            }
            if ( !stat )
            {
                f2 = f3 = false;
                for ( i=0; i<13; ++i )
                {
                    if ( r[sd[i]]==2 )
                    {
                        inf2 = i;
                        f2 = true;
                    }
                    else if ( r[sd[i]]==3 ) f3 = true;
                }
                if ( f3 )
                {
                    if ( f2 ) stat = 4;
                    else stat = 7;
                }
                else if ( f2 )
                {
                    f2 = false;
                    for ( i=0; i<13; ++i )
                    {
                        if ( r[sd[i]]==2 && i!=inf2 )
                        {
                            f2 = true;
                            break;
                        }
                    }
                    if ( f2 ) stat = 8;
                    else stat = 9;
                }
                else
                {
                    flag = true;
                    for ( i=0; ((i<8) && flag); ++i )
                    {
                        for ( j=0; ((j<5) && flag); ++j )
                        {
                            if ( !r[sd[i+j]] ) break;
                            if ( j==4 ) flag = false;
                        }
                    }
                    for ( j=9; j<13; ++j )
                    {
                        if ( !r[sd[j]] ) break;
                        if ( j==12 && r['A'] ) flag = false;
                    }
                    if ( !flag ) stat = 6;
                    else stat = 10;
                }
            }
        }
        switch ( stat )
        {
        case 1:
            cout << "royal flush" << endl;
            break;
        case 2:
            cout << "straight flush" << endl;
            break;
        case 3:
            cout << "four of a kind" << endl;
            break;
        case 4:
            cout << "full house" << endl;
            break;
        case 5:
            cout << "flush" << endl;
            break;
        case 6:
            cout << "straight" << endl;
            break;
        case 7:
            cout << "three of a kind" << endl;
            break;
        case 8:
            cout << "two pairs" << endl;
            break;
        case 9:
            cout << "pair" << endl;
            break;
        case 10:
            cout << "high card" << endl;
            break;
        }
    }
    return 0;
}
