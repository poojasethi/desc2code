#include <stdio.h>

int main()
{
	int T;
	char caractere;
	int i;
	bool mainTrouvee;
	int j;
	int valeurs[20];
	int couleurs[20];
	int k;

	scanf("%d",&T);
	scanf("%c",&caractere);

	for(i=0;i<T;i++)
	{
		mainTrouvee = false;
		for(j=0;j<20;j++)
		{
			valeurs[j] = 0;
			couleurs[j] = 0;
		}

		for(j=0;j<5;j++)
		{
			scanf("%c",&caractere);
			if(caractere > 49 && caractere < 58)
			{
				valeurs[caractere - 48]++;
			}
			else if(caractere=='T')
			{
				valeurs[10]++;
			}
			else if(caractere=='J')
			{
				valeurs[11]++;
			}
			else if(caractere=='Q')
			{
				valeurs[12]++;
			}
			else if(caractere=='K')
			{
				valeurs[13]++;
			}
			else
			{
				valeurs[14]++;
			}

			scanf("%c",&caractere);
			if(caractere == 'C')
			{
				couleurs[0]++;
			}
			else if(caractere == 'D')
			{
				couleurs[1]++;
			}
			else if(caractere == 'H')
			{
				couleurs[2]++;
			}
			else
			{
				couleurs[3]++;
			}

			scanf("%c",&caractere);
		}

		for(j=0;j<4;j++)
		{
			if(couleurs[j]==5)
			{
				if(valeurs[10]==1 && valeurs[11]==1 && valeurs[12]==1 && valeurs[13]==1 && valeurs[14]==1)
				{
					mainTrouvee = true;
					printf("royal flush\n");
					break;
				}
			}
		}

		if(!mainTrouvee)
		{
			for(j=0;j<4;j++)
			{
				if(couleurs[j]==5)
				{
					for(k=14;k>=6;k--)
					{
						if(valeurs[k]==1 && valeurs[k-1]==1 && valeurs[k-2]==1 && valeurs[k-3]==1 && valeurs[k-4]==1)
						{
							mainTrouvee = true;
							printf("straight flush\n");
							break;
						}
					}
					if(valeurs[14]==1 && valeurs[2]==1 && valeurs[3]==1 && valeurs[4]==1 && valeurs[5]==1)
					{
						mainTrouvee = true;
						printf("straight flush\n");
						break;
					}
				}
			}
		}

		if(!mainTrouvee)
		{
			for(j=14;j>=2;j--)
			{
				if(valeurs[j]==4)
				{
					mainTrouvee=  true;
					printf("four of a kind\n");
					break;
				}
			}
		}

		if(!mainTrouvee)
		{
			for(j=14;j>=2;j--)
			{
				if(valeurs[j]==3)
				{
					for(k=14;k>=2;k--)
					{
						if(valeurs[k]==2)
						{
							mainTrouvee = true;
							printf("full house\n");
							break;
						}
					}
				}
			}
		}

		if(!mainTrouvee)
		{
			for(j=0;j<4;j++)
			{
				if(couleurs[j]==5)
				{
					mainTrouvee = true;
					printf("flush\n");
					break;
				}
			}
		}

		if(!mainTrouvee)
		{
			for(j=14;j>=6;j--)
			{
				if(valeurs[j]==1 && valeurs[j-1]==1 && valeurs[j-2]==1 && valeurs[j-3]==1 && valeurs[j-4]==1)
				{
					mainTrouvee = true;
					printf("straight\n");
					break;
				}
				if(valeurs[14]==1 && valeurs[2]==1 && valeurs[3]==1 && valeurs[4]==1 && valeurs[5]==1)
				{
					mainTrouvee = true;
					printf("straight\n");
					break;
				}
			}
		}

		if(!mainTrouvee)
		{
			for(j=14;j>=2;j--)
			{
				if(valeurs[j]==3)
				{
					mainTrouvee = true;
					printf("three of a kind\n");
					break;
				}
			}
		}

		if(!mainTrouvee)
		{
			for(j=14;j>=2;j--)
			{
				if(valeurs[j]==2)
				{
					for(k=j-1;k>=2;k--)
					{
						if(valeurs[k]==2)
						{
							mainTrouvee = true;
							printf("two pairs\n");
							break;
						}
					}
					if(!mainTrouvee)
					{
						mainTrouvee = true;
						printf("pair\n");
						break;
					}
				}
			}
		}

		if(!mainTrouvee)
		{
			printf("high card\n");
		}
	}
}
