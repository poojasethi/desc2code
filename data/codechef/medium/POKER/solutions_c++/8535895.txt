int check_royal_flush(int *suits, int *_rank)
{
    if((_rank[1] == 1) &&(_rank[10] == 1) && (_rank[11] == 1) && (_rank[12] == 1) && (_rank[13] == 1))
    {
        for(int i=0; i<4; i++)
            if(suits[i] == 5)
            return 1;
    }
    return 0;
}
int check_straight_flush(int *suits, int *_rank)
{
    int i,j;
    for(i=0; i<4 ;i++)
        if(suits[i] == 5)
          break;
    if(i == 4)
        return 0;
    for(i=1;i<=13; i++)
        if(_rank[i] == 1)
          break;
    for(j = i+1; j<=i+4; j++)
        if(_rank[j] == 0)
          break;
    if(j == i+5)
        return 1;
    return 0;
}
