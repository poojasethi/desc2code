#include<iostream>
#include<string>
using namespace std;
char s[20];
int a[20];
void result()
{
     string ans;
     for(int i=0;i<=13;i++)  
    {            
     switch(s[i])
     {
         case 'T': a[i]=10;
                   break;
         case 'J': a[i]=11;
                   break;    
         case 'Q': a[i]=12;
                   break;
         case 'K': a[i]=13;
                   break;
         case 'A': a[i]=1;
                   break;
     }
     if(s[i]!='A' && s[i]!='T' && s[i]!='J' && s[i]!='Q' && s[i]!='K' )
      a[i]=int(s[i])-48;
    }
    
    for(int i=0;i<=13;i=i+3)        
            for(int j=i+3;j<=13;j=j+3)
                    if(a[j]<a[i])
                    {
                                 int temp=a[i];
                                 a[i]=a[j];
                                 a[j]=temp;
                    }
    
    if(a[1]==a[4] && a[4]==a[7] && a[7]==a[10] && a[10]==a[13])
    {    
         int temp=a[0];
        if(a[0]==1 && a[3]==10 && a[6]==11 && a[9]==12 && a[12]==13)                                                                         
                   ans="royal flush";
                   
        else if(a[3]==temp+1 && a[6]==temp+2 && a[9]==temp+3 && a[12]==temp+4)
                   ans="straight flush";
        else
                   ans="flush";
    }
    else
    {
        int temp=a[0],temp1=a[12],temp2=a[6],temp3=a[9],temp4=a[3];
        
        if(a[3]==temp+1 && a[6]==temp+2 && a[9]==temp+3 && a[12]==temp+4)
                        ans="straight";
                        
        else if((a[3]==temp && a[6]==temp && a[9]==temp) || (a[3]==temp1 && a[6]==temp1 && a[9]==temp1 ))
                         ans="four of a kind";   
        
        else if((a[3]==temp && a[9]==temp2 && a[12]==temp2) || (a[3]==temp && a[6]==temp && a[12]==temp3 ))
                            ans="full house";
                            
        else if((a[3]==temp && a[6]==temp) || (a[6]==temp4 && a[9]==temp4) || (a[9]==temp2 && a[12]==temp2))
                            ans="three of a kind";
        else if((a[6]==temp4 &&  a[12]==temp3) || (a[3]==temp &&  a[12]==temp3) || (a[3]==temp &&  a[9]==temp2))
                             ans="two pairs";
        
        else if(a[12]==temp3 || a[9]==temp2 || a[6]==temp4 || a[3]==temp )
                             ans="pair";
                             
        else 
                         ans="high card";        
    }
    cout<<ans<<"\n";
}
 
 
int main()
{
    int t;
    scanf("%d",&t);
    for(int i=1;i<=t;i++)
    {    
         cin.ignore();
         cin.get(s,18);    
         result();    
    }
    return 0;
}
