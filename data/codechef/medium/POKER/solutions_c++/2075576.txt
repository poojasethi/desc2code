#include <iostream>
#include <fstream>
#include <cstdio>
#include <map>
#include <vector>
#include <set>
#include <string>
#include <cstring>
#include <algorithm>
#include <cmath>

using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		char cardno[5][3];
		scanf("%s %s %s %s %s",cardno[0],cardno[1], cardno[2], cardno[3], cardno[4]);
		int cards[13] = {0};
		int types[4]  = {0};
		for(int i = 0; i < 5; i++)
		{
			switch (cardno[i][0])
			{
			case 'A':
				cards[0]++;
				break;
			case 'T':
				cards[9]++;
				break;
			case 'J':
				cards[10]++;
				break;
			case 'Q':
				cards[11]++;
				break;
			case 'K':
				cards[12]++;
			default:
				cards[cardno[i][0]-'0' -1]++;
				break;
			}
		}
		for(int i = 0; i < 5; i++)
		{
			switch (cardno[i][1])
			{
			case 'S':
				types[0]++;
				break;
			case 'H':
				types[1]++;
				break;
			case 'D':
				types[2]++;
				break;
			case 'C':
				types[3]++;
				break;
			default:
				break;
			}
		}
		bool royal = false;
		int sameType=0;
		for(int i = 0; i < 4; i++)
		{
			if (types[i]==5)
				sameType++;
		}

		if (((cards[0]&&cards[9]&&cards[10]&&cards[11]&&cards[12])) && sameType==1)
			royal = true;
		if (royal)
		{
			cout << "royal flush\n";
			continue;
		}
		bool sflush = false;
		for(int i=0; i <= 13-5; i++)
		{
			int count = 0;
			for(int j = i; j < 5+i; j++)
			{
				if (cards[j]==1)
					count++;
			}
			if (count == 5)
			{
				sflush = true;
			}
		}
		if (sameType == 1 && sflush)
		{
			cout << "straight flush\n";
			continue;
		}
		int pair4=0, pair2=0, pair3=0;
		for(int i = 0; i < 13; i++)
		{
			if (cards[i]==4)
				pair4++;
			if (cards[i]==2)
				pair2++;
			if(cards[i]==3)
				pair3++;
		}
		if(pair4 == 1)
		{
			cout << "four of a kind\n";
			continue;
		}
		if(pair3==1&&pair2==1)
		{
			cout << "full house\n";
			continue;
		}
		if(sameType)
		{
			cout << "flush\n";
			continue;
		}
		if (sflush || (cards[0]&&cards[9]&&cards[10]&&cards[11]&&cards[12]))
		{
			cout << "straight\n";
			continue;
		}
		if (pair3)
		{
			cout << "three of a kind\n";
			continue;
		}
		if (pair2==2)
		{
			cout << "two pairs\n";
			continue;
		}
		if (pair2==1)
		{
			cout << "pair\n";
			continue;
		}
		else
		{
			cout << "high card\n";
		}
	}
	return 0;
}
