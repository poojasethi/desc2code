        return 1;
    else if(card[0] == '3' and card[1] == '4' and card[2] == '5' and card[3] == '6' and card[4] == '7')
        return 1;
    else if(card[0] == '4' and card[1] == '5' and card[2] == '6' and card[3] == '7' and card[4] == '8')
        return 1;
    else if(card[0] == '5' and card[1] == '6' and card[2] == '7' and card[3] == '8' and card[4] == '9')
        return 1;
    else if(card[0] == '6' and card[1] == '7' and card[2] == '8' and card[3] == '9' and card[4] == 'T')
