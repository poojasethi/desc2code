#include <iostream>
#include <bits/stdc++.h>

using namespace std;
typedef long long int ll;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,sum=0;
		cin>>a>>b;
		sum = a + b;int cnt=0;
		while(1)
		{
			sum++; cnt++;
			int f = 0;
			for(int j=2;j<=sqrt(sum);j++)
			{
				if(sum%j==0) f = 1;
			}
			if(!f)
			{
				cout<<cnt<<endl;
				break;
			}
		}
	}
	return 0;
}
