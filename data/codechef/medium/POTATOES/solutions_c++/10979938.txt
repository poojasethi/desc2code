#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x,y,sum,p=0,i;
		cin>>x>>y;
		sum=x+y+1;
		while(p==0)
		{
			for(i=2;i<sum;i++)
			{
				if(sum%i==0){
					p=0;
					sum++;
					break;
				}
				else
				p=1;
				}
			}
			cout<<sum-(x+y)<<endl;	
	}
	return 0;
}
