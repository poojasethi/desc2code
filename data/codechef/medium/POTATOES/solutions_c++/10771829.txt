#include <iostream>
#include <math.h>

using namespace std;
int primes[3001];

int main(){
	int i=2;
	while(i<sqrt(3000)){
		while(primes[i]&&i<3001)
			i++;
		int n=2*i;
		while(n<3001){
			primes[n]=1;
			n+=i;
		}
		i++;
		
	}
	int T,X,Y;
	cin >> T;
	for(int t=0;t<T;t++){
		cin >> X >> Y;
		int XY = X + Y;
		int j = XY+1;
		while(j<3000 && primes[j])
			j++;
		cout << j - XY << endl;
	}

	return 0;
}
