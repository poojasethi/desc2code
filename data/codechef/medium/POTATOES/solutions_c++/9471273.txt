#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;

void sieve(bool arr[])
{
    int i,j;
    for(i=2;i*i<=2003;i++)
    {
        if(arr[i]==true)
        {
            for(j=i*2;j<=2003;j+=i)
                arr[j]=false;
        }
    }
}

int main()
{
    //freopen("inp.txt","r",stdin);
    int t,a,b,i;
    bool arr[2004];
    memset(arr,true,sizeof(arr));
    sieve(arr);
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&a,&b);
        for(i=a+b+1;i<=2003;i++)
            if(arr[i]==true)
            {
                printf("%d\n",i-a-b);
                break;
            }
    }
    return 0;
}
