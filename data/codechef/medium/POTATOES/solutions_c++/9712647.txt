#include<bits/stdc++.h>
using namespace std;
int primes[10001];
void seive()
{
	int i,j;
	for(i=2;i<=10000;i++)
		primes[i] = 0;
	for(i=2;i<=10000;i++)
			for(j=i;j*i<=10000;j++)
				primes[i*j] = 1;
}

int main()
{
	int sum,i,t,x,y;
	cin>>t;
	seive();
	while(t--)
	{
		cin>>x>>y;
		sum = x+y;
		for(i=sum+1;i<=10000;i++)
			if(primes[i] == 0)
				break;
		cout<<i-sum<<endl;
	}
	return 0;
}
