#include<stdio.h>
#include<iostream>
#include<math.h>

using namespace std;

bool isPrime(int p){
	int lim = sqrtf(p);
	for(int i=3; i<=lim; i+=2){
		if(p%i == 0)
			return false;
	}
	return true;
}

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int x,y;
		scanf("%d %d",&x,&y);
		int ans=2;
		if( (x+y)%2 == 0 )
			ans=1;
		while(!isPrime(x+y+ans))
			ans+=2;
		printf("%d\n",ans);
	}
}
