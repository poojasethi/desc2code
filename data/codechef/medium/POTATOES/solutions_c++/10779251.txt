#include<bits/stdc++.h>
using namespace std;
int main()
{
  int t,x,y,n,i,j,z;
  cin>>t;
  while(t>0)
    {
    cin>>x>>y;
    
    for(i=1;i<=1000;i++)
       {
       z=0;
       n=x+y+i;
         for(j=2;j<=n/2;j++)
             {
                if(n%j==0)
                    {
                    z=1;
                    break;
                    }
             }
           if(z==0)
              {
              cout<<i<<endl;
              break;
              }  
       }
    t--;
    }
}
