#include<iostream>
using namespace std;
int main(){
	int t,a,b,s,i,j,count;
	cin>>t;
	while(t--){
		cin>>a>>b;
		s=a+b;
		for(i=1;;i++){
			s=s+1;
			for(j=2,count=0;j<s;j++){
				if(s%j==0)
				++count;
				if(count==2)
				break;
			}
			if(count==0)
			break;
		}
		cout<<(s-a-b)<<endl;
	}
	return 0;
}
