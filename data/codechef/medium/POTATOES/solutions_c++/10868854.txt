#include<cstdio>
#include<cmath>
using namespace std;
bool prime[2101];
int Next_prime(int X){
    for(int i=X+1;i<=2100;i++){
        if(prime[i]){
            return i;
        }
    }
    return -1;
}
int main(){
    for(int i=0;i<2100;i++){
        prime[i]=true;
    }
    for(int i=2;i<=sqrt(2100);i++){
        if(prime[i])
            for(int j=i*i;j<2100;j+=i){
                prime[j]=false;
            }
    }
    prime[0]=false;
    prime[1]=false;
    int T;
    scanf("%d",&T);
    while(T--){
        int A,B;
        scanf("%d %d",&A,&B);
        printf("%d\n",Next_prime(A+B)-A-B);
    }
}
