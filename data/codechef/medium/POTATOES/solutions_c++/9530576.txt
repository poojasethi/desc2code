#include<iostream>
using namespace std;
int prime(int n)
{
	int i=0;
	for(i=2;i<=n/2;i++)
	{
		if((n%i)==0)
		return 0;
	}
	
	return 1;
}
int main()
{
	int t,x,y,sum=0,i;
	cin>>t;
	while(t--)
	{
		cin>>x>>y;
		
		for(i=1;i<=1000;i++)
		{
			sum=x+y+i;
			if(prime(sum)==1)
			{
				cout<<i<<endl;
				break;
			}
		}		
	}
}
