#include <iostream>


int main() {
	int x,y,t,s,j,flag;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&x,&y);
		flag=0;
		for(j=1;j<=1000;j++)
		{
			flag=0;
			s=x+y+j;
			for(int i=2;i<=s/2;i++)
			{
				if(s%i==0)
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
				break;
		}
		printf("%d\n",j);
	}
	return 0;
}
