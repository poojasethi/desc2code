int prime(int num)
{
	int j,count=0;
	for(j=2;j<num;j++)
	{
		if(num%j==0)
		{
			count=0;
			break;
		}
		else
		count=1;
	}
	if(count==1)
	return(num+1);
	else
	return(0);
