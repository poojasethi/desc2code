#include<iostream>
using namespace std;
int isprime(int n);
int main()
{
  int t,x,y,i,j=1,flag;
  cin>>t;
  while(t--)
  {
   flag=1;
   cin>>x>>y;
   i=x+y+1;
   while(flag!=0)
   {
       j=isprime(i);
       if(j==1)
       flag=0;
       else
       i++;
    }
    cout<<i-(x+y)<<"\n";
  }
  return 0;
 }
 int isprime(int n)
 {
   int i,flag=0;
   for(i=2;i<=n/2;i++)
   {
	if(n%i==0)
	{
	flag=1;
	break;
	}
     }
     if(flag==1)
     return 0;
     else
     return 1;
 }
