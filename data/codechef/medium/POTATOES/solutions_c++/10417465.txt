#include<iostream>
//#include<conio.h>
#include<cmath>
using namespace std;
int main()
{
//clrscr();
int T,x,y,count=0,c,g=0;
cin>>T;
int *result=new int[T];
for(int m=0;m<T;m++)
{
    cin>>x>>y;

    count=0; g=0;c=0;
    for(int i=x+y+1;;i++)
    { c=sqrt(x+y+i);
      for(int j=2;j<=c;j++)
      {
	if(i%j==0&&(i!=j))
	  break;
	else if(i==1);
	else
	{
	 if(j==c)
	  {count=i;
	  g=1;
	  break;}
	 else
	  continue;
	}

      }
      if(g==1)
	 break;
    }
result[m]=count-(x+y);
}
for(int j=0;j<T;j++)
 cout<<"\n"<<result[j];
//getch();
return 0;
}
