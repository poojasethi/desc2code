#include <iostream>
#include <math.h>

using namespace std;

bool isprime(int n)
{
    for(int i=2;i<=sqrt(n);i++)
    {
        if(n%i==0)
            return false;
    }
    return true;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int sum=0;
        int r;
        for(int i=0;i<2;i++){
            cin>>r;
            sum += r;
        }
        for(int i=sum+1;i<=2027;i++)
        {
            if(isprime(i))
            {
                cout<<i-sum<<endl;
                break;
            }
        }
    }
}
