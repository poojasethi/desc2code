#include <iostream>
using namespace std;
int isprime(int n){
	int f=1,j;
	for(j=2;j<=(n/2);j++){
		if(n%j==0)
			f=0;
	}
	if(f==1)
		return 1;
	else 
		return 0;
}
 
int main() {
	int t;
	cin>>t;
	while(t--){
		int m,n;
		cin>>m>>n;
		for(int i=(m+n+1);;i++)
			if(isprime(i)){
				cout<<i-(m+n)<<"\n";
				break;
			}
	}
	return 0;
}
