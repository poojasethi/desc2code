#include<bits/stdc++.h>
using namespace std;
int arr[5000]={0};
void fun()
{

   int i,j,k,n=5000;
   arr[0]=0;
   arr[1]=0;
     for(i=2;i<=n;i++)
    {
        if(arr[i]==0)
        {
        for(j=i*i;j<=n;j=j+i)
        {
                arr[j]=1;
        }
        }
    }


}
int main()
{
    int t,x,y,i;
    cin>>t;
    fun();
    while(t!=0)
    {
     cin>>x>>y;
     i=x+y;
     while(1)
        {
            i++;
            if(arr[i]==0)
                break;


        }

        t--;
    }

}
