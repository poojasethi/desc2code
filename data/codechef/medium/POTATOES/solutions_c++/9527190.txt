#include<iostream>
using namespace std;
int sieve(int num)
{
int i,primes[10002],elem,j;
for(i=2;i<=10000;i++)
primes[i]=1;
for(i=2;i<=10000;i++)
{
for(j=i;j*i<=10000;j++)
primes[j*i]=0;
}
int z=0;
for(i=2;i<=10000;i++)
{
if(primes[i])
{
primes[z]=i;
z++;
}
}
for(i=0;i<=10000;i++)
{
if(primes[i]>num)
{
elem=primes[i];
break;
}
}
return elem;
}
int main()
{
int t;
cin>>t;
while(t--)
{
int x,y;
cin>>x>>y;
int z;
z=sieve(x+y);
cout<<z-x-y<<endl;
}
return 0;
}
