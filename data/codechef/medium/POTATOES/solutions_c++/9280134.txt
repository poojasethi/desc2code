#include <iostream>
using namespace std;

int main() {
	int x,y,t,s,j,flag;
	cin>>t;
	while(t--)
	{
		scanf("%d %d",&x,&y);
		flag=0;
		for(j=1;j<=1000;j++)
		{
			flag=0;
			s=x+y+j;
			for(int i=2;i<=s/2;i++)
			{
				if(s%i==0)
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
				break;
		}
		cout<<j<<endl;
	}
	return 0;
}
