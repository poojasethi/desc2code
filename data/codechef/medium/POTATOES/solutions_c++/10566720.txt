#include <stdio.h>
#include <stdlib.h>
int q=0;
void prime(int a)
{int i;
    for(i=2;i<a;i++)
    {
        if(a%i==0)
        {
            q=0;
            break;
        }
        else
            q=1;
    }
}
 
 
int main()
{
   int n;
   scanf("%d",&n);
   while(n--)
   {q=0;
       int a,b,c,d=0;
       scanf("%d %d",&a,&b);
       c=a+b;
 
       while(1)
       {
        if(q==1)
        {
            printf("%d\n",d);
            break;
        }
        else
        {   c++;
              d++;
            prime(c);}
       }
   }
}
