#include "stdio.h"
#include "math.h"

bool isPrime(int number)
{
	int s = sqrt(number);

	for (int i = 2; i <= s; i++)
	{
		if (number%i == 0)
			return false;
	}

	return true;
}

int nextPrime(int number)
{
	number += number % 2 ? 0 : 1;
	do
	{
		if (isPrime(number))
		{
			return number;
		}

	} while (number += 2);

	return number;
}


int main()
{
	int tests,N,K,L;
	scanf("%d", &tests);


	while (tests--)
	{
		int x, y;
		scanf("%d %d", &x, &y);

		printf("%d\n", nextPrime(x + y + 1) - x - y);
	}

	return 0;
}
