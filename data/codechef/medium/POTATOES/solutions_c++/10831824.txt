#include<iostream>
using namespace std;
int arr[4001];
#define m 4001
void sieve()
{
    arr[0]=arr[1]=1;
    for(int i=4;i<m;i+=2)
        arr[i]=1;
    for(int i=3;i*i<m;i+=2)
    {
        if(arr[i]==0)
        {
            for(int j=i*2;j<m;j+=i)
                arr[j]=1;
        }
    }
}
int main()
{
    sieve();
    //for(int i=0;i<100;i++)
      //  cout<<arr[i]<<' ';
    //
    cout<<endl;
    int t;ios::sync_with_stdio(false);
    cin>>t;int x,y;
    while(t--)
    {
        cin>>x>>y;
        int i=x+y+1;
        while(i<m)
        {
            if(arr[i]==0)
            {
                break;
            }
            i++;
        }
        cout<<i-(x+y)<<endl;
    }
}
