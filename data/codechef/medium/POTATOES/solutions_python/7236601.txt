import sys

sieve = [False]*3000

def Sieve() :
    for i in range(2,3000) :
        j = 2
        while i*j < 3000 :
            if not sieve[i*j] :
                sieve[i*j] = True
            j += 1
        
Sieve()
for __ in range(input()) :
    k = sum(map(int,sys.stdin.readline().split()))
    i = k
    while i<3000 :
        if not sieve[i] and i > k :
            break
        i += 1
    print i-k
