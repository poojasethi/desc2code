import sys,math
def next_prime(p):
    
    while True:
        
        if prime(p):
            return p
        p+=1    
def prime(p):
    i=2
    while i<=math.ceil(math.sqrt(p)):
        
        if not p%i:
            return False
        i+=1    
    return True
