import math
def is_prime(l):
    s = int(math.sqrt(l))
    for i in range(2,s+2):
        if l%i == 0:
            return False
    return True
t = input()
for tests in xrange(t):
    x,y = map(int, raw_input().split())
    n = x+y+1
    if n%2 ==0:
        n +=1
    while not is_prime(n):
        n += 2
    print n-x-y
