import math
t=input()
assert t>=1 and t<=1000
for i in range(t):
    x,y=map(int,raw_input().strip().split())
    assert x >= 1 and x <= 1000
    assert y >= 1 and y <= 1000
    p=x+y
    if p%2==0:
        r=p+1
    else:
        r=p+2
    for i in range(r,2*p,2):
        for j in range(3,int(math.sqrt(r))+2,2):
            if i%j==0:
                break
        else:
            print i-(x+y)
            break
