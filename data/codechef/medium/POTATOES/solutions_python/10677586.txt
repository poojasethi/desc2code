#!/usr/bin/env python 3.1
t=input()
while t>0:
    t=t-1
    a=raw_input()
    a=a.split(' ')
    x=int(a[0])
    y=int(a[1])
    sum=x+y
    f=1
    j=0
    while f==1:
        j=j+1
        count=0
        i=1
        while i<=sum+j:
            if (sum+j)%i==0:
                count=count+1
            i=i+1
        if count==2:
            print(j)
            f=0
