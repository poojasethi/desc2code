from bisect import bisect_right as br
def primes(n):
    sieve = [True] * n
    for i in xrange(3,int(n**0.5)+1,2):
        if sieve[i]:
            sieve[i*i::2*i]=[False]*((n-i*i-1)/(2*i)+1)
    return [2] + [i for i in xrange(3,n,2) if sieve[i]]
li=primes(10010)
for _ in xrange(input()):
    a,b=map(int,raw_input().split())
    s=a+b
    ind=br(li,s)
    print li[ind]-s
