#include<iostream>
#include<cstdio>
#define MOD 1000000007
using namespace std;
int main()
{
   long long int t,left,right;
   scanf("%lld",&t);
   while(t--)
   {
       scanf("%lld%lld",&left,&right);
       long long int ans = 0;
       long long int tmp = 10;
       for(int i=1; i<=10; i++)
       {
           long long int cnt = min(tmp - 1, right) - max(left, tmp/10) + 1;
           if(cnt <= 0)
            {
                tmp*=10;
                continue;
            }
           long long int start = max(left,tmp/10);
           long long int diff = 1;
           long long int sm = 2*start + cnt - 1;
           sm *= cnt;
           sm /= 2;
           if(sm >= MOD)
            sm %= MOD;
           sm *= i;
           if(sm >= MOD)
            sm %= MOD;
           ans += sm;
           if(ans >= MOD)
            ans %= MOD;
           tmp *= 10;
       }
       printf("%lld\n",ans);
   }
   return 0;
}
