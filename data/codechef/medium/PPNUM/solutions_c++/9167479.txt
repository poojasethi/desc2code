#include<iostream>
using namespace std;
#define mod 1000000007;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        unsigned long long int l,r;
        cin>>l>>r;

        unsigned long long int m,n,temp;
        unsigned long long int sum=0;

        unsigned long long int digm=0;
        unsigned long long int dign=0;
        while(1)
        {
            digm=0;
            dign=0;
             m=l;
             n=r;
            while(m){
                m=m/10;
                digm++;
            }
            while(n){
                n=n/10;
                dign++;
            }

            if (digm==dign)
            {
                temp=(digm*(l+r)*(r-l+1))/2;
                temp=temp%mod;
                sum=(sum+temp)%mod;

                break;
            }
            else
            {
                temp=0;
                for(int i=0;i<digm;i++)
                {
                    temp=temp*10;
                    temp=temp+9;
                }

                dign=temp+1;
                temp=(digm*(l+temp)*(temp-l+1))/2;
                l=dign;
                temp=temp%mod;
                sum=(sum+temp)%mod;


            }

        }


        cout<<sum<<'\n';
    }

}
