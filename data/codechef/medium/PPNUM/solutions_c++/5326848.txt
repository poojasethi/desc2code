typedef unsigned long long int llu;

#define sc(n) scanf("%lld",&n)

#define rep(i,n) for(int i=0; i<(int)n; i++)
#define repv(i,n) for(int i=n-1; i>=0; i--)
#define repl(i,n) for(int i=1; i<=(int)n; i++)
#define replv(i,n) for(int i=n; i>=1; i--)
#define mod 1000000007

ll dig[]={0,9,99,999,9999,99999,999999,9999999,99999999,999999999};

int main()
{
	//freopen("input.txt","r",stdin);
	ll t,L,H,sum,n,i;
	scanf("%lld",&t);
	while(t--){
	scanf("%lld %lld",&L,&H);
	sum=0;
	for(i=1;i<=9;i++)
	{
		if(dig[i]>H)
		break;
		if(dig[i]>=L)
		{
			n=(dig[i]-L)+1;
			if(n%2==0)
				sum+=( (i*(n%mod)*(L%mod))%mod+(i*((n/2)%mod)*((n-1)%mod))%mod)%mod;
			else sum+=( (i*(n%mod)*(L%mod))%mod+(i*(n%mod)*(((n-1)/2)%mod))%mod)%mod;
			L=dig[i]+1;
		}
	}
	if(dig[i]>H&&H!=dig[i-1]){
	n=(H-L)+1;
	if(n%2==0)
	sum+=( (i*(n%mod)*(L%mod))%mod+(i*((n/2)%mod)*((n-1)%mod))%mod)%mod;
	else sum+=( (i*(n%mod)*(L%mod))%mod+(i*(n%mod)*(((n-1)/2)%mod))%mod)%mod;
	}
	else if(dig[i]<H)
		sum+=((H%mod)*10)%mod;

    sum=sum%mod;
	printf("%lld\n",sum);
	}
	return 0;
}
