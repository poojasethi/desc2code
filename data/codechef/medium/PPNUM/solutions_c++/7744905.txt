typedef long long LL;
typedef unsigned long long ULL;

using namespace std;

LL mod = 1000000007;

LL cum(LL a,LL b)
{
    LL mx = b*(b+1)/2;
    LL mn = a*(a-1)/2;
    return (mx - mn) % mod;
}

LL fun(LL x)
{
    LL prev = 0,cur = 9,tot = 0,mul = 1;
    while(1)
    {
        LL my = min(cur,x);
        LL sum = cum(prev,my);
        tot += ( sum * mul );
        tot %= mod;
        if(cur >= x)break;
        prev = cur+1;
        cur = (cur + 1) * 10 - 1;
        mul++;
    }
    return tot;
}

int main()
{
    LL t,T,a,b;
    sl(T);
    rep(t,T)
    {
        sll(a,b);
        LL mx = fun(b);
        LL mn = fun(a-1);
        //printf("%lld %lld\n",mx,mn);
        LL ans = mx - mn;
        if(ans < 0)ans += mod;
        pl(ans);NL;
    }
    return 0;
}
