#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int minimum(int a,int b)
{
    if(a<b)
        return a;
    else
        return b;
}
int no_of_digits(int num)
{
    int c=0;
    while(num!=0)
    {
        num=num/10;
        c++;
    }
    return c;
}
int pow(int a,int num)
{
    int res=1,i;
    for(i=1;i<=num;i++)
        res=res*a;
    return res;
}
int main()
{
    int t,l,r,u,v;
    long long int res;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&l,&r);
        u=l;res=0;
        while(u<=r)
        {
            v=minimum(pow(10,no_of_digits(u))-1,r);
            res=(res+((1LL*no_of_digits(u)*(u+v)*(v-u+1))/2))%mod;
            u=v+1;
        }
        printf("%lld\n",res);
    }
    return 0;
}
