#include <bits/stdc++.h>
using namespace std;
#define ll long long int
 int main()
 {
 	int t,i;
 	scanf("%d",&t);
 	for(i=0;i<t;i++)
 	{
 		ll sum=0,j,p,l,r,q,s;
 		scanf("%lld%lld",&l,&r);
 		p=log10(l);
		p+=1;
		for(j=l-1 ; j<=r;)
 		{
			if(pow(10,p)-1<r)
				s=pow(10,p)-1;
			else
				s=r;
			sum+=((s*(s+1)/2)-(j*(j+1)/2))*(p);
			sum=sum%1000000007;
			j=pow(10,p)-1;
			p+=1;
 		}
 		printf("%lld\n",sum%1000000007);
 	}
 return 0;
 }
