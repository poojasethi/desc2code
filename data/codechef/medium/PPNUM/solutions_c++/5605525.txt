LL P[12] , S[12] , tc , L , R;

LL sum(LL a, LL b)
{
    LL n = b-a+1;
    return (n*(a+b)/2)%MOD;
}

LL F(LL bil)
{
   if (bil==0) return 0;
   
   int i = 11;
   while (1)
   {
         if (P[i]<=bil) break;
         i--;
   }
   return (S[i-1] + i*sum(P[i] , bil))%MOD;
}
int main(void)
{
    P[1] = 1;
    for (int i = 2; i<=11; i++) P[i] = 10*P[i-1];
    S[0] = 0;
    for (int i = 1; i<=10; i++) 
    {
        S[i] = S[i-1] + i*sum(P[i],P[i+1]-1);
        S[i] %= MOD;
    }
    
    scanf("%lld",&tc);
    for (int cases = 1 ; cases<=tc; cases++)
    {
        scanf("%lld %lld",&L, &R);
        printf("%lld\n", (F(R) - F(L-1) + MOD) % MOD);
    }


}
