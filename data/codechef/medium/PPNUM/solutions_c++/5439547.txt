#include <iostream>
#include <cmath>
#include <cstdio>
#define mod 1000000007
#define ull unsigned long long int
using namespace std;
int count(ull n)
{
	int c = 0;
	while (n > 0) {
		n = n/10;
		c++;
	}
	return c;
}
int main()
{
	int t,d1,d2;
	ull l,r,temp,ans,n;
	
	scanf("%d",&t);
	for (int k = 0; k < t; k++) {
		scanf("%llu%llu",&l,&r);
		d1 = count(l);
		d2 = count(r);
		ans = 0;
		while (d1 <= d2) {
			if (d1 == d2) {
				n = r-l+1;
				ans = (ans + (d1*(((n*(2*l+n-1)))/2))%mod)%mod;
				d1++;
			}
			else {
				temp = pow(10,d1) - 1;
				n = temp-l+1;
				ans = (ans + (d1*(((n*(2*l+n-1)))/2))%mod)%mod;
				l = temp+1;
				d1++;
			}
		}
		printf("%llu\n",ans);
	}
}
