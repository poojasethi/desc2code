#include <iostream>
#include <stdio.h>
#include <math.h>
using namespace std;

long int fast_integer() {
    long int set = 0;
    char c = getchar();
    while(c>='0' && c<='9') {
    	set = (set<<3) + (set<<1)+c-'0';
    	c = getchar();
    }
    return set;
}

int main() {
    long long int i, L, R, sum, sumL, sumR, end, mod = 1000000007;
    char c;
    int T = fast_integer(), digitsL, digitsR;
    while (T--) {
        L = 0, R = 0, sum = 0;
        digitsL = 0, digitsR = 0;
        c = getchar();
        while(c>='0' && c<='9') {
            digitsL++;
        	L = (L<<3) + (L<<1)+c-'0';
        	c = getchar();
        }
        c = getchar();
        while(c>='0' && c<='9') {
            digitsR++;
        	R = (R<<3) + (R<<1)+c-'0';
        	c = getchar();
        }
        sumL = (L*(L-1))/2;
        while (digitsL <= digitsR) {
            end = (pow(10, digitsL)) - 1;
            if (end > R) {
                end = R;
            }
            sumR = (end * (end + 1))/2;
            sum += ((sumR - sumL) * digitsL);
            if (sum > mod) {
                sum = sum % mod;
            }
            sumL = sumR;
            L = end + 1;
            digitsL++;
        }
        printf("%lld\n", sum);
    }
}
