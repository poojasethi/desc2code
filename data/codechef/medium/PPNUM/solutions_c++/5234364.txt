#include <iostream>
#include <cmath>
#define NUM 1000000007
using namespace std;
int check(int k)
{
    if(k<=9) return 1;
    else if(k<=99) return 2;
    else if(k<=999) return 3;
    else if(k<=9999) return 4;
    else if(k<=99999) return 5;
    else if(k<=999999) return 6;
    else if(k<=9999999) return 7;
    else if(k<=99999999) return 8;
    else if(k<=999999999) return 9;
    return 10;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long x,y,start,end,n,ans = 0;
		cin>>x>>y;
        start = check(x);
		end = check(y);
		if(start==end)
		{
		    n = y-x+1;
		    ans = (n*(2*x+(n-1))*start/2)%NUM;
		    cout<<ans<<"\n";
		    continue;
		}
		for(int i=start;i<=end;i++)
		{
			long long mul = pow(10.0,i-1);
			if(i==start)
			{
				n = pow(10.0,i)-x;
				if(n==1) ans = (x*start)%NUM;
				else ans = (ans+n*(2*x+(n-1))*i/2)%NUM;
			}
			else if(i==end)
			{
				if(end!=10)
				{
					n = y-mul+1;
					if(n==1) ans = (ans+y)%NUM;
					else ans = (ans+n*(2*mul+(n-1))*i/2)%NUM;
				}
				else
					ans = (ans+999999937)%NUM;
			}
			else
			{
				n = 9*mul;
				ans = (ans+n*(2*mul+(n-1))*i/2)%NUM;
			}
		}
		cout<<ans<<"\n";
    }
    return 0;
}
