#include<stdio.h>
#define LL long long
#define mod 1000000007
int main()
{
    LL l,r,n,ans;
    int t,ll,lr;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld %lld",&l,&r);
        n=l;
        ll=lr=0;
        while(n)
        {
            ll++;
            n=n/10;
        }
        n=r;
        while(n)
        {
            lr++;
            n=n/10;
        }
        ans=((((r*(r+1)/2)%mod)*lr)-((l*(l-1)/2)%mod)*ll)%mod;
        int i,j;
        for(i=ll;i<lr;i++)
        {
            LL temp=1;
            for(j=1;j<=i;j++)
            {
                temp=temp*10;
            }
            ans=(ans-(temp*(temp-1)/2)%mod)%mod;
        }
        if(ans<0)
            ans=ans+mod;
        printf("%lld\n",ans);
    }
}
