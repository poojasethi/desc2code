#include<bits/stdc++.h>
#define llu unsigned long long int
#define MOD 1000000007
using namespace std;

int num_digits(llu num)
{
    int counts=0;
    while(num!=0)
    {
        counts++;
        num = num/10;
    }
    return counts;
}

int main()
{
    int cases;
    llu l,r,sum;
    cin>>cases;
    for(int i=0;i<cases;i++)
    {
        cin>>l>>r;
        int len = max(num_digits(l),num_digits(r));
        llu a=1,b=9;
        sum=0;
        for(int i=1;i<=len;i++)
        {
            llu ll = max(a,l);
            llu rr = min(b,r);
            if(ll <= rr)
            {
                llu ans = (rr-ll + 1)*(ll + rr)/2;
                sum = (sum + i*(ans%MOD))%MOD;
            }
            a = a*10;
            b = b*10 + 9;
        }
        cout<<sum<<endl;
    }
    return 0;
}
