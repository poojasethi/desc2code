#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
#define mod 1000000007
using namespace std;
vector<long long> v(10);
int main()
{
    long long i,n=9,b=1,e=9,num,j,n1,n2,l,r,sum,temp,t,ans;
    v[9]=10;
    for(i=0;i<9;i++)
    {
        num=0;
        b=1;
        for(j=1;j<=(i+1);j++)
        {
            num=num*10+1;
        }
        for(j=0;j<i;j++)
        {
            b=b*10;
        }
        e=num*9;
        n=e-b+1;
        v[i]=((((e+b))*n)/2)%mod;
        v[i]=(v[i]*(i+1))%mod;
        //printf("%lld\n",v[i]);
    }
    scanf("%lld",&t);
    while(t--)
    {
        sum=0;
        scanf("%lld%lld",&l,&r);
        temp=l;
        n1=n2=0;
        ans=0;
        while(temp>0)
        {
            n1++;
            temp/=10;
        }
        temp=r;
        while(temp>0)
        {
            n2++;
            temp/=10;
        }
        if(n1==n2)
        {
            n=r-l+1;
            ans=(((l+r)*n)/2)%mod;
            ans=(ans*n1)%mod;
        }
        else
        {
            for(i=n1;i<=n2;i++)
            {
                if((i==n1)||(i==n2))
                {
                    num=0;
                    b=1;
                    for(j=1;j<=i;j++)
                    {
                        num=num*10+1;
                    }
                    for(j=0;j<i-1;j++)
                    {
                        b=b*10;
                    }
                    if(i==n1)
                    {
                    e=num*9;
                    n=e-l+1;
                    sum=(((e+l)*n)/2)%mod;
                    }
                    else
                    {
                        n=r-b+1;
                        sum=(((r+b)*n)/2)%mod;
                    }
                    sum=(sum*i)%mod;
                    ans=(ans+sum)%mod;
                }
                else
                    ans=(ans+v[i-1])%mod;
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
