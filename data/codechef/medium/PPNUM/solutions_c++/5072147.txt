long long unsigned sum(long long unsigned a, long long unsigned b, int d)
{
	long long unsigned r1,r2;
	r1=(b-a+1);
	r2=(b+a);
	if(r1%2==0)
		r1=r1/2;
	else
		r2=r2/2;
	return ((r1%MOD)*(r2%MOD))%MOD;
}

int nod(int n)
{
	int i=0;
	while(n!=0)
	{
		n=n/10; 
		i++;
	}
	return i;
}

int main()
{
	int t;

	while(t--)
	{
		int r,l;

		int n = r-l+1;
		int d1 = nod(l);
		int d2 = nod(r);
		//cout << d1<< " " << d2;
		long long unsigned ans;

		if(d1 == d2)
		{
			ans = ( sum(l,r,d1)*d1 ) % MOD;
		}
		else
		{
			int power = d1;

			long long unsigned large1 =  pow(10,power);
			ans = ( sum(l,large1-1,d1)*power)%MOD;
			long long unsigned large2 = large1*10;
			//cout << ans << endl;
			//cout << large1 << " " << large2 << endl;
			
			for(int i=1;i<d2-d1;i++)
			{
				power++;

				ans=ans%MOD;
