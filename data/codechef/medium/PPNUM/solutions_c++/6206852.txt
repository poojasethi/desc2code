#include<iostream>
#include<cstdio>
#define  MOD  1000000007
using namespace std;
int main(){
	long long int t,n,i,st,en,ans,temp;
	long long int arr[]={0,9,99,999,9999,99999,999999,9999999,99999999,999999999,1000000000};
	scanf("%lld",&t);
	n=11;
	while(t--){
			   scanf("%lld%lld",&st,&en);
			   ans=0;
			   for(i=0;i<n;i++){
			   					if(arr[i]>=st)
			   					  break;
								   
								   }
			   for(;i<n;i++){
			   				 if(en<= arr[i])
			   				   break;
			   				temp=0; 
							if(st&1){
							   temp-= (((st-1)/2)*st);}
							 else
							  temp-= ((st/2)*(st-1));
							temp+= (arr[i]*((arr[i]+1)/2));
							temp%=MOD;
							temp= (temp*i)%MOD;
							ans+=temp;
							ans%=MOD;
							 st=arr[i]+1;
			   				 //cout<<" ans is for st "<<ans<<" "<<st<<"\n";
								
								}
								temp=0;
								if(st&1){
			   							temp-= (( st*(((st-1)/2)))*i);}
			    				else
								        temp-=(((st/2)*(st-1))*i);		
			   //cout<<" ans is for st "<<temp<<" "<<st<<"\n";
			   
			   			  if(en&1){
							temp+= (((en*((en+1)/2)))*i);}
							else
							   temp+=((((en/2)*(en+1)))*i);
				//			   cout<<" temp without MOD is "<<temp;
						temp%=MOD;	   
					//		cout<<" ans is for st "<<temp<<" "<<en<<"\n";
									ans+=temp;
									ans%=MOD;			   
			   printf("%lld\n",ans);
			   }
	
	
	}
