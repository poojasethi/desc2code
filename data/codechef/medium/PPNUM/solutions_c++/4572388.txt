#include<bits/stdc++.h>
using namespace std;
#define INV 500000004
#define NUM 1000000007
int dig(long long int N)
{
	int c=0;
	while(N>0)
	{
		c++;
		N/=10;
	}
	return c;
}
main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		long long int L,R,sum;scanf("%lld%lld",&L,&R);
		long long int n1=dig(L),n2=dig(R),i;
		if(n1==n2)
		{sum=(((n1*((R*(R+1)%NUM-L*(L-1)%NUM+NUM))%NUM)%NUM)*INV)%NUM;
		}
		else{
		long long int b1=(long long int)pow(10,n1),b2=(long long int)pow(10,n2-1);
		sum=(n1*(((((b1*(b1-1))%NUM-(L*(L-1))%NUM+NUM)%NUM)*INV)%NUM))%NUM;//printf("%lld\t",sum);
		sum=(sum+(((n2*(((R*(R+1))%NUM-(b2*(b2-1))%NUM+NUM)%NUM))%NUM)*INV)%NUM)%NUM;//printf("%lld\n",sum);
		for(i=n1+1;i<n2;i++)
		{
		b1=(long long int)pow(10,i-1);b2=(long long int)pow(10,i);
		sum=(sum+((i*((INV*((NUM-b1*(b1-1)%NUM+b2*(b2-1)%NUM)%NUM))%NUM))%NUM))%NUM;
		}}
		printf("%lld\n",sum);
	}
	return 0;
}
