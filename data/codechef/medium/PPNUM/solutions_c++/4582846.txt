#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;

int main(){
    unsigned long long int sum = 0,endno,l,r,x;
    int t,start = 0,end1 = 0,digit;
    scanf("%d",&t);

 while(t--){
    sum = 0,start = 0,end1 = 0;
    scanf("%lld %lld",&l,&r);

    x = l;
    while(x > 0){
        digit = x % 10;
        start++;
        x = x/10;
    }
    x = r;
    while(x > 0){
        digit = x % 10;
        end1++;
        x = x/10;
    }

    x = l;
    while(start < end1){
        endno = pow(10,start) - 1;
        sum += (start*(endno + x)*(endno - x + 1))/2;
        sum %= 1000000007;
        x = endno + 1;
        start++;
    }

    sum += (start*(r + x)*(r - x + 1))/2;
    sum %= 1000000007;

    printf("%lld\n",sum);


 }

return 0;
}
