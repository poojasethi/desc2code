#include<stdio.h>
#include<math.h>
int main()
{
long long unsigned int t,L,R,i,j,k,m,sum,c,p,ans,s,d;
s=pow(10,9)+7;
scanf("%llu",&t);
while(t--)
{
 sum=0;k=0;p=0;k=0;m=0;
 scanf("%llu %llu",&L,&R);
 j=L;
 while(j>0)
 {
  j=j/10;k++;
 }
 c=R;
 while(c>0)
 {
  c=c/10;p++;
           }
 for(i=k;i<=p;i++)
 {
  d=pow(10,i);
  if(L<=R)
  {
   if(R<d)
   {
     m=R-L+1;
     sum=(sum+(m*(R+L)*i)/2);
          }
      else
      {
       m=d-L;
       sum=(sum+((L+d-1)*m*i)/2);
       L=d;
          }
          }
 }
  s=pow(10,9)+7;
 ans=(sum)%s;
 printf("%llu\n",ans);
}
return 0;
}
