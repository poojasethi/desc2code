#include<stdio.h>
#include<math.h>
#define V 1000000007
#include<iostream>
using namespace std;

int main()
{
    int T,l,r;
    long long int L,R,i,j,k,ans;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%lld %lld",&L,&R);
        ans=0;
        i=L;
        j=R;
        for(l=0;i;i/=10,l++);
        for(r=0;j;j/=10,r++);
        if(L%2)
        {
            ans=l*L%V;
            L++;
        }
        if(R%2==0)
        {
            ans=(ans+r*R%V)%V;
            R--;
        }
        if(L>=R);
        else if(l==r)
        ans=(ans+(R-L+1)/2*l%V*(L+R)%V)%V;
        else
        {
            r--;
            i=pow(10,r);
            ans=(ans+((R-i+1)/2%V)*(R+i)%V*(r+1)%V)%V;
            i=pow(10,l);
            ans=(ans+((i-L)/2)%V*(i-1+L)%V*l%V)%V;
            for(l++;l<=r;l++,i*=10)
            ans=(ans+i/2*9%V*(11*i-1)%V*l%V)%V;

        }
        printf("%lld\n",ans);
    }
    return 0;
}
