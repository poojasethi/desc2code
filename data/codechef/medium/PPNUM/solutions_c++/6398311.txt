#include <iostream>
#define LL long long

using namespace std;

const int mod = 1e9 + 7;

LL solve(LL x) {
LL ans = 0, i, j, k;
for (i = 1, k = 1; i <= x; i *= 10, ++k) {
j = i * 10;
if (j > x) j = x + 1;
ans += k * (i + j - 1) * (j - i) / 2;
}
return ans % mod;
}

int main() {
int t;
LL L, R;
cin >> t;
while (t--) {
cin >> L >> R;
LL ans = solve(R) - solve(L - 1);
ans = (ans % mod + mod) % mod;
cout << ans << endl;
}
return 0;
}
