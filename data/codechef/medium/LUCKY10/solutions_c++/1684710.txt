#include <string>
#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

const int MaxDigit = 10;

int t;
string s1, s2;
int cnt1[MaxDigit], cnt2[MaxDigit];

int main()
{
    cin.sync_with_stdio(0);
    cin >> t;

    while (t--)
    {
        memset(cnt1, 0, sizeof(cnt1));
        memset(cnt2, 0, sizeof(cnt2));

        cin >> s1 >> s2;

        for (int i = 0; i < s1.size(); i++)
        {
            cnt1[s1[i] - '0']++;
            cnt2[s2[i] - '0']++;
        }

        string Ans = "";

        while (cnt1[7] > 0)
        {
            bool found = false;

            for (int i = 6; i >= 0; i--) if (cnt2[i] && i != 4)
            {
                cnt1[7]--;
                cnt2[i]--;
                Ans.push_back('7');
                found = true;
                break;
            }

            if (!found) break;
        }

        while (cnt1[7] && cnt2[4])
        {
            cnt1[7]--;
            cnt2[4]--;
            Ans.push_back('7');
        }

        while (cnt2[7] > 0)
        {
            bool found = false;

            for (int i = 6; i >= 0; i--) if (cnt1[i] && i != 4)
            {
                found = true;
                cnt2[7]--;
                cnt1[i]--;
                Ans.push_back('7');
                break;
            }

            if (!found) break;
        }

        while (cnt2[7] && cnt1[4])
        {
            cnt2[7]--;
            cnt1[4]--;
            Ans.push_back('7');
        }

        while (cnt1[7] && cnt2[7])
        {
            cnt1[7]--;
            cnt2[7]--;
            Ans.push_back('7');
        }

        while (cnt1[4] > 0)
        {
            bool found = false;

            for (int i = 3; i >= 0; i--) if (cnt2[i])
            {
                found = true;
                cnt1[4]--;
                cnt2[i]--;
                Ans.push_back('4');
                break;
            }

            if (!found) break;
        }

        while (cnt2[4] > 0)
        {
            bool found = false;

            for (int i = 3; i >= 0; i--) if (cnt1[i])
            {
                found = true;
                cnt2[4]--;
                cnt1[i]--;
                Ans.push_back('4');
                break;
            }

            if (!found) break;
        }

        while (cnt1[4] && cnt2[4])
        {
            Ans.push_back('4');
            cnt1[4]--;
            cnt2[4]--;
        }

        puts(Ans.c_str());
    }

    return 0;
}
