#include <iostream>
using namespace std;

int main() {
	int t, la, lb;
	cin>>t;
	string a, b;
	int count4, count7;
	bool visited;
	char ch;
	int idx, i, j;
	while(t>0) {
		count4=0;
		count7=0;
		cin>>a;
		cin>>b;
		la=a.length();
		lb=b.length();
		int countsa[10]={0};
		int countsb[10]={0};
		for(i=0;i<la;i++) {
			ch=a[i];
			countsa[ch-'0']++;
		}
		for(i=0;i<lb;i++) {
			ch=b[i];
			countsb[ch-'0']++;
		}
		idx=6;
		visited=false;
		while(countsa[7]>0) {
			if(idx==-1)
				break;
			if(countsb[idx]==0) {
				idx--;
				if(idx==4 && !visited)
					idx--;
				else if(idx==-1 && !visited) {
					idx=4;
					visited=true;
				}
				else if(idx==4 && visited)
					idx=-1;
			}
			else {
				countsb[idx]--;
				countsa[7]--;
				count7++;
			}
		}
		idx=6;
		visited=false;
		while(countsb[7]>0) {
			if(idx==-1)
				break;
			if(countsa[idx]==0) {
				idx--;
				if(idx==4 && !visited)
					idx--;
				else if(idx==-1 && !visited) {
					idx=4;
					visited=true;
				}
				else if(idx==4 && visited)
					idx=-1;
			}
			else {
				countsa[idx]--;
				countsb[7]--;
				count7++;
			}
		}
		count7+= ((countsa[7]<countsb[7]) ? countsa[7]:countsb[7]);
		idx=3;
		while(countsa[4]>0) {
			if(idx==-1)
				break;
			if(countsb[idx]==0)
				idx--;
			else {
				countsb[idx]--;
				countsa[4]--;
				count4++;
			}
		}
		idx=3;
		while(countsb[4]>0) {
			if(idx==-1)
				break;
			if(countsa[idx]==0)
				idx--;
			else {
				countsa[idx]--;
				countsb[4]--;
				count4++;
			}
		}
		count4+= ((countsa[4]<countsb[4]) ? countsa[4]:countsb[4]);
		while(count7>0) {
			cout<<"7";
			count7--;
		}
		while(count4>0) {
			cout<<"4";
			count4--;
		}
		cout<<"\n";
		t--;
	}
	return 0;
}
