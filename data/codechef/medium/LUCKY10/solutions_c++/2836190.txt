#include <algorithm>
#include <cstdio>
#include <cstring>

#define MAXL 20000

int st[2][5];
char a[MAXL+5], b[MAXL+5];

int stind(char c) {
  if(c>'7') return 0;
  if(c=='7') return 1;
  if(c>'4') return 2;
  if(c=='4') return 3;
  return 4;
}

void partition(char a[], int in, int l) {
  for(int i=0;i<5;++i) st[in][i] = 0;
  for(int i=0;i<l;++i) ++st[in][stind(a[i])];
}

int main() {
  int t;
  scanf("%d",&t);
  while(t--) {
    scanf("%s%s",a,b);
    int l = strlen(a);
    partition(a, 0, l);
    partition(b, 1, l);
    int in, st1, st2, mn;
    int cs=0, cf=0;
    for(int i=0;i<2;++i) {
      st1 = i;
      st2 = st1^1;
      in = mn = 2;
      while(st[st2][1] && !(in==1 && !mn)) {
	if(!st[st1][in]) {
	  switch(in) {
	    case 2:
	      in = 4;
	      break;
	    case 4:
	      in = 3;
	      break;
	    default:
	      in = 1;
	  }
	}
	mn = std::min(st[st2][1],st[st1][in]);
	cs += mn;
	st[st2][1] -= mn;
	st[st1][in] -= mn;
      }
    }
    for(int i=0;i<2;++i) {
      st1 = i;
      st2 = st1^1;
      in = mn = 4;
      while(st[st2][3] && !(in==3 && !mn)) {
	if(!st[st1][in]) in = 3;
	mn = std::min(st[st2][3],st[st1][in]);
	cf += mn;
	st[st2][3] -= mn;
	st[st1][in] -= mn;
      }
    }
    for(int i=0;i<cs;++i) printf("7");
    for(int i=0;i<cf;++i) printf("4");
    printf("\n");
  }
  return 0;
}
