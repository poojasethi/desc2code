#include<iostream>
#include<stdio.h>
#include<string>
#include<stdlib.h>
#include<ctype.h>
#include<cmath>
#include<algorithm>
#include<fstream>
#include<vector>

using namespace std;

typedef long long ll;

int main()
{
    int t;
    cin>>t;
    string s1, s2;
    int a7, a4, b7, b4, pa, qa, ra, pb, qb, rb, x, i, temp;

    int n7a, n7b, n4;
    while(t--)
    {
        cin>>s1>>s2;
        vector<int> a, b;
        for( i = 0; i < s1.length(); i++)
        {
            x = s1[i] - 48;
            a.push_back(x);
            x = s2[i] - 48;
            b.push_back(x);
        }
        a4 = b4 = a7 = b7 = pa = pb = qa = qb = ra = rb = 0;
        for(i = 0; i < a.size(); i++)                              // calculate p, q, r for a & b
        {
            if(a[i]>7)
                pa++;
            else if(a[i] == 7)
                a7++;
            else if(a[i]>4)
                qa++;
            else if(a[i] == 4)
                a4++;
            else
                ra++;
            if(b[i]>7)
                pb++;
            else if(b[i] == 7)
                b7++;
            else if(b[i]>4)
                qb++;
            else if(b[i] == 4)
                b4++;
            else
                rb++;
        }                                                           // p, q, r calculated for both a & b
        n7a = n7b = 0;
        while(a7 > 0)
        {
            if(qb > 0)
                qb--;
            else if(rb > 0)
                rb--;
            else if(b4 > 0)
                b4--;
            else if(b7 > 0)
                b7--;
            else
                break;
            n7a++;
            a7--;
        }
        while(b7 > 0)
        {
            if(qa > 0)
                qa--;
            else if(ra > 0)
                ra--;
            else if(a4 > 0)
                a4--;
            else if(a7 > 0)
                a7--;
            else
                break;
            n7b++;
            b7--;
        }
        n4 = 0;
        while(a4 > 0)
        {
            if(rb > 0)
                rb--;
            else if(b4 > 0)
                b4--;
            else
                break;
            a4--;
            n4++;
        }
        while(b4 > 0)
        {
            if(ra > 0)
                ra--;
            else if(a4 > 0)
                a4--;
            else
                break;
            b4--;
            n4++;
        }
        for(i = 0; i < (n7a+n7b); i++)
            cout<<"7";
        for(i = 0; i<n4; i++)
            cout<<"4";
        cout<<endl;
    }
    return 0;
}
