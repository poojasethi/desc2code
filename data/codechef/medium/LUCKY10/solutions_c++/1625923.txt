#include<cstdio>
#include<iostream>
#include<string>
#include<cmath>
using namespace std;

void fill(string A, int cA[])
{
	for (int i = 0; i < A.size(); i++)
	{
		switch((int)A[i]-'0')
		{
			case 0:
			case 1:
			case 2:
			case 3: cA[0]++; break;
			case 4: cA[1]++; break;
			case 5:
			case 6: cA[2]++; break;
			case 7: cA[3]++; break;
			default: break;
		}
	}
}

int seven;
int four;

int match(int *a, int *b)
{
	int m = min(*a, *b);
	*a -= m;
	*b -= m;
	return m;
}

int main()
{
	int T;
	scanf("%d", &T);
	for (int t = 0 ; t < T; t++)
	{
		seven = 0;
		four = 0;
		string A, B;
		cin >> A;
		cin >> B;
		//0-{0,1,2,3} | 1-{4} | 2-{5,6} | 3-{7}
		int cA[4];
		int cB[4];
		for (int i = 0; i < 4; i++)
		{
			cA[i] = 0;
			cB[i] = 0;
		}
		fill(A, cA);
		fill(B, cB);
		//priority-[7,5],[5,7],[7,0],[0,7],[7,4],[4,7],[7,7],[4,0],[0,4],[4,4]
		//actualpr-[3,2],[2,3],[3,0],[0,3],[3,1],[1,3],[3,3],[1,0],[0,1],[1,1]
		seven += match(cA+3, cB+2);
		seven += match(cA+2, cB+3);
		seven += match(cA+3, cB+0);
		seven += match(cA+0, cB+3);
		seven += match(cA+3, cB+1);
		seven += match(cA+1, cB+3);
		seven += match(cA+3, cB+3);
		four += match(cA+1, cB+0);
		four += match(cA+0, cB+1);
		four += match(cA+1, cB+1);
		for (int i = 0; i < seven; i++)
			printf("7");
		for (int i = 0; i < four; i++)
			printf("4");
		printf("\n");
	}
	return 0;
}
