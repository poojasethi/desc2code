#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int main()
{
  int Test;
  scanf("%d",&Test);
  while(Test--)
    {
      static int A[10],B[10];
      memset(A,0,sizeof(A));
      memset(B,0,sizeof(B));
      char c;
      while(c=getchar(),c<'0'||c>'9');
      while(c>='0'&&c<='9')
        {
          A[c-48]++;
          c=getchar();
        }
      while(c=getchar(),c<'0'||c>'9');
      while(c>='0'&&c<='9')
        {
          B[c-48]++;
          c=getchar();
        }
      for(int i=6;i>=0;i--)
        while(A[7]&&B[i]&&i!=4)
          {
            putchar('7');
            A[7]--;
            B[i]--;
          }
      for(int i=6;i>=0;i--)
        while(A[i]&&B[7]&&i!=4)
          {
            putchar('7');
            A[i]--;
            B[7]--;
          }
      while(A[7]&&B[4])
        {
          putchar('7');
          A[7]--;
          B[4]--;
        }
      while(A[4]&&B[7])
        {
          putchar('7');
          A[4]--;
          B[7]--;
        }
      while(A[7]&&B[7])
        {
          putchar('7');
          A[7]--;
          B[7]--;
        }
      for(int i=3;i>=0;i--)
        while(A[4]&&B[i])
          {
            putchar('4');
            A[4]--;
            B[i]--;
          }
      for(int i=4;i>=0;i--)
        while(A[i]&&B[4])
          {
            putchar('4');
            A[i]--;
            B[4]--;
          }
      putchar('\n');
    }
  return 0;
}
