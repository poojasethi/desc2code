#include <cstdio>
#include <cassert>
#include <cstring>

#define MAX 20010

using namespace std;

int seq[5][6] = {
	{1, 0},
	{2, 0, 1},
	{3, 2, 0, 1},
	{4, 2, 0, 1, 3},
	{5, 4, 2, 0, 1, 3}
};

void parse(char *str, int *a){
	while (*str != '\0'){
		int digit = *str-'0';
		if (digit < 4) a[0]++;
		else if (digit == 4) a[1]++;
		else if (digit < 7) a[2]++;
		else if (digit == 7) a[3]++;
		else a[4]++;
		++str;
	}
}

void annihilate(int *a, int *b, int index){
	for (int i = 1; a[index] && i <= seq[index][0]; i++)
		if (a[index] <= b[seq[index][i]]) b[seq[index][i]] -= a[index], a[index] = 0;
		else a[index]-= b[seq[index][i]], b[seq[index][i]] = 0;
}

int main(){
	int T;
	scanf("%d", &T);

	while(T--){
		char A[MAX], B[MAX];
		scanf("%s %s", A, B);

		int a[5] = {}, b[5] = {};
		parse(A, a);
		parse(B, b);
		
		int count7 = 0, count4 = 0;
		
		annihilate(a, b, 4);
		annihilate(b, a, 4);

		count7 += a[3];
		annihilate(a, b, 3);
		count7 += b[3];
		annihilate(b, a, 3);

		annihilate(a, b, 2);
		annihilate(b, a, 2);

		count4 += a[1];
		annihilate(a, b, 1);
		count4 += b[1];
		annihilate(b, a, 1);
		
		assert(a[0] == b[0]);

		for (int i = 0; i < count7; i++) printf("7");
		for (int i = 0; i < count4; i++) printf("4");
		printf("\n");
	}
}
