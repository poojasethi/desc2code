#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cmath>
#include<string.h>
#define intin(a) scanf("%d",&a)
#define MAX(a,b) (a>b?a:b)
#define MIN(a,b) (a<b?a:b)
using namespace std;
int cnt7,cnt4,a[5],b[5],c[5];
void apart(char s[])
{

    int i,l=strlen(s);

    for(i=0;i<5;i++)c[i]=0;

    for(i=0;i<l;i++)
    {
        if(s[i]=='9' || s[i]=='8')
            c[0]++;
        else if(s[i]=='7')
            c[1]++;
        else if(s[i]=='6' || s[i]=='5')
            c[2]++;
        else if(s[i]=='4')
            c[3]++;
        else if(s[i]=='3' || s[i]=='2' || s[i]=='1' || s[i]=='0')
            c[4]++;
    }
}
void calc7(int *a,int *b)
{
    if(a[1]>b[2])
    {
        cnt7+=b[2];
        a[1]-=b[2];b[2]=0;
        if(a[1]>b[4])
        {
            cnt7+=b[4];
            a[1]-=b[4];b[4]=0;

            if(a[1]>b[3])
            {
                cnt7+=b[3];
                a[1]-=b[3];b[3]=0;

                if(a[1]>b[1])
                {
                    cnt7+=b[1];
                    a[1]-=b[1];b[1]=0;
                    b[0]-=a[1];a[1]=0;
                }
                else
                {
                    cnt7+=a[1];
                    b[1]-=a[1];a[1]=0;
                }
            }
            else
            {
                cnt7+=a[1];
                b[3]-=a[1];a[1]=0;
            }
        }
        else
        {
            cnt7+=a[1];
            b[4]-=a[1];a[1]=0;
        }
    }
    else
    {
        cnt7+=a[1];
        b[2]-=a[1];a[1]=0;
    }
}
void calc4(int *a,int *b)
{
    if(a[3]>b[4])
    {
        cnt4+=b[4];
        a[3]-=b[4];b[4]=0;
        if(a[3]>b[3])
        {
            cnt4+=b[3];
            a[3]-=b[3];b[3]=0;
            a[3]=0;
        }
        else
        {
            cnt4+=a[3];
            b[3]-=a[3];a[3]=0;
        }
    }
    else
    {
        cnt4+=a[3];
        b[4]-=a[3];a[3]=0;
    }
}
int main()
{
    //freopen("lucky10.txt","r",stdin);
    int i,cases;
    char s1[20001],s2[20001];
    intin(cases);
    while(cases--)
    {
        scanf("%s%s",s1,s2);
        apart(s1);
        for(i=0;i<5;i++)a[i]=c[i];
        apart(s2);
        for(i=0;i<5;i++)b[i]=c[i];

        //cout<<a[0]<<" "<<a[1]<<" "<<a[2]<<" "<<a[3]<<" "<<a[4]<<"\n";
        //cout<<b[0]<<" "<<b[1]<<" "<<b[2]<<" "<<b[3]<<" "<<b[4]<<"\n";

        cnt7=0;cnt4=0;
        calc7(a,b);
        calc7(b,a);

        calc4(a,b);
        calc4(b,a);

        for(i=0;i<cnt7;i++)cout<<7;
        for(i=0;i<cnt4;i++)cout<<4;
        cout<<"\n";

    }
}
