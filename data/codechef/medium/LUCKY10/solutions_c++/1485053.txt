#include <iostream>
#include <vector>
#include <cstdio>
#include <string>

using namespace std;

int priority[11][2]={{7, 5}, {5, 7},{7, 0}, {0, 7},{7, 4}, {4, 7},{7, 7},{4, 0}, {0, 4},{4, 4}};

vector<int> readstring()
{
    string temp;
    vector<int> count(10,0);
    cin>>temp;
    for(int i=0;i<temp.length();i++)
    {
        if(temp[i]-'0'<4)
            count[0]++;
        else if(temp[i]-'0'==4)
            count[4]++;
        else if(temp[i]-'0'<7)
            count[5]++;
        else if(temp[i]-'0'==7)
            count[7]++;
    }
    return count;
}
int main()
{
    vector<int> a,b;

    int d1,d2,k,d,t;
    cin>>t;
    while(t--)
    {
        a=readstring();
        b=readstring();
        vector<int> c(10,0);

        for(int i=0;i<=10;i++)
        {
            d1=priority[i][0];
            d2=priority[i][1];

            k=min(a[d1],b[d2]);
            d=max(d1,d2);

            c[d]+=k;
            a[d1]-=k;
            b[d2]-=k;

        }

        cout << string(c[7],'7')+string(c[4],'4')<<"\n";

    }

    return 0;
}
