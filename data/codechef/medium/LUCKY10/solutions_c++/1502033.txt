char A[MAXN+1], B[MAXN+1];
const int chcs[][5] = {3, 1, 4, 2, 0,
                       0, 3, 1, 4, 2,
                       3, 4, 2, 0, 1,
                       2, 0, 1, 3, 4,
                       4, 2, 0, 1, 3
                      };
char hsh(char c)
{
    if(c > '7')return '4';
    else if(c == '7') return '3';
    else if(c >'4')return '2';
    else if(c == '4')return '1';
    return '0';
}
void solve()
{
    int l = strlen(A);
    int cnta[5], cntb[5];
    memset(cnta, 0, sizeof(cnta));
    memset(cntb, 0, sizeof(cntb));
    for(int i=0; i<l; ++i)
    {
        A[i] = hsh(A[i]);
        B[i] = hsh(B[i]);
        cnta[A[i]-'0']++;
        cntb[B[i]-'0']++;
    }
    int svn=0, frr=0;
    for(int i=4; i>=0; --i)
    {
        for(int k=0; k<cnta[i]; ++k)
        {
            for(int j=0; j<5; ++j)
            {
                if(!cntb[chcs[i][j]])continue;
                cntb[chcs[i][j]]--;
                int a = max(chcs[i][j], i);
                if(a == 1)frr++;
                else if(a == 3)svn++;
                break;
            }

        }

    }
    for(int i=0; i<svn; ++i)
        printf("7");
    for(int i=0; i<frr; ++i)
        printf("4");
    printf("\n");
}

int main()
{
    int testcases;
    scanf("%d", &testcases);
    for(int caseno=1; caseno<=testcases; caseno++)
    {
        scanf("%s%s", A, B);
        solve();
    }
    return 0;
}
