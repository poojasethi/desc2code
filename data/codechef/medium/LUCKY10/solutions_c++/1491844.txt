#include<stdio.h>
#include<string.h>
int main()
{
char A[20001];
char B[20001];
char c[20001];
int t, l, i, j, nA7g = 0, nAm = 0, nB7g = 0, nBm = 0, nC7 = 0, nC4 = 0;
scanf("%d",&t);
for (j = 0; j < t; j++)
{
scanf("%s",&A);
scanf("%s",&B);
nA7g = nAm = nB7g = nBm = nC7 = nC4 = 0;
l = strlen(A);
for (i = 0; i < l; i++)
{
if (A[i] > 55)
nA7g++;
else if (A[i] == 55)
{
nC7++;
nBm--;
}
else if (A[i] > 52)
nAm++;
else if (A[i] == 52)
nC4++;
if (B[i] > 55)
nB7g++;
else if (B[i] == 55)
{
nC7++;
nAm--;
}
else if (B[i] > 52)
nBm++;
else if (B[i] == 52)
nC4++;
}
l -= (nA7g > nB7g ? nA7g : nB7g);
nC7 = nC7 < l ? nC7 : l;
l -= nC7;
if (nA7g > nB7g)
nBm -= nA7g - nB7g;
else nAm -= nB7g - nA7g;
nBm = (nBm > nAm ? nBm : nAm);
if (nBm > 0)
l -= nBm;
nC4 = nC4 < l ? nC4 : l;
nC4 = nC4 > 0 ? nC4 : 0;
for (i = 0; i < nC4 + nC7; i++)
{
if (i < nC7)
putchar('7');
else putchar('4');
}
printf("\n");
}
return 0;
}
