#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
int n;
int k;
cin>>n;
cin>>k;
int N=n;
int cnt=0;
while(n--)
{
int e;
cin>>e;
if(e%2==0)
cnt++;
}
if(k==0 && cnt==N)
cout<<"NO"<<endl;
else if(cnt>=k)
{
cout<<"YES"<<endl;
}
else
{
cout<<"NO"<<endl;;
}
}
return 0;
}
