#include <iostream>

using namespace std;

int main()
{
    int t,n,k,i;
    cin>>t;

    while(t--){
    int sum=0,su=0,tmp;
    cin>>n>>k;

    for(i=0;i<n;++i)
    {
    cin>>tmp;
    if(tmp%2==0)
    sum++;
    else
    su++;
    }
    if(k==0)
    {if(su>0)
    cout<<"YES\n";
    else
    cout<<"NO\n";
    }else if(sum>=k)
    cout<<"YES\n";
    else
    cout<<"NO\n";

    }
    return 0;
}
