#include<cstdio>
#include<cstring>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k;
        scanf("%d %d",&n,&k);
        int count=0;
        for(int i=0;i<n;i++){
            int tmp;
            scanf("%d",&tmp);
            if(tmp%2==0){
                count++;
            }
        }
        if(count>=k && (k!=0 || count!=n)){
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }
}
