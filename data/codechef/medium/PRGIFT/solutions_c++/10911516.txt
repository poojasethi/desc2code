#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int n,k;
		cin>>n>>k;
		int a[n];
		int count=0;
		for(int j=0;j<n;j++)
		{
			cin>>a[j];
			if(a[j]%2==0)
			count++;
		}
		if((count>=k &&k!=0) || ((count>=k&&  count<n)))
		cout<<"YES"<<"\n";
		else
		cout<<"NO"<<"\n";
	}
}
