#include<stdio.h>
inline int scan()

{

        int noRead=0;

        char p=getchar();

        for(; p<33;) {

                p=getchar();

        };

        while(p>32) {

                noRead = (noRead << 3) + (noRead << 1) + (p - '0');

                p=getchar();

        }

        return noRead;

};
int main()
{
	 int n , k , test , i  , ans , flag  ; 
	 test = scan() ;

					 } 
			   }
		  }
		
		  	
		
		  
		 if( flag ==1 )

		  {
