#include <iostream>
using namespace std;

int main(int argc, char const *argv[])
{
	int testcase, n, k, array[100];
	int i, j, even, flag;
	cin >> testcase;
	while(testcase --)
	{
		cin >> n >> k;
		for(i = 0; i < n; i ++)
			cin >> array[i];
		flag = 0;
		for(i = 0; i < n; i ++)
		{
			even = 0;
			for(j = i; j < n; j ++)
			{
				if(array[j] % 2 == 0)
					even ++;
				if(even == k)
				{
					flag = 1;
					cout << "YES" << endl;  
					break;
				}
			}
			if(flag == 1)
				break;
		}
		if(flag == 0)
			cout << "NO" << endl;
	}
	return 0;
}
