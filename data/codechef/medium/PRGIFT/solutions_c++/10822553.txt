#include<stdio.h>
int main()
{
    int arr[100],t,n,k,i,flag=1;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        flag=0;
        if(!k)
            flag=1;
        for(i=0;i<n;i++)
        {
            scanf("%d",arr+i);
            if(arr[i]%2==0)
                k--;
        }
        if(k>0||k==(-n)&&flag)
            printf("NO\n");
        else
            printf("YES\n");
    }
    return 0;
}
