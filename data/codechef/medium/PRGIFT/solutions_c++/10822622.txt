#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		int arr[n];
		int cnt=0;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i]%2==0)
				cnt++;
		}

	   if(k==0)
	   {
	       if(cnt==n)
	       cout<<"NO"<<endl;
	       else
	       cout<<"YES"<<endl;
	   }
	   
	   else if(cnt>=k)
	   cout<<"YES"<<endl;
	   else
	   cout<<"NO"<<endl;
	}

	return 0;
}
