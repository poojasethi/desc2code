#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k,j,flag,count,a[100];
		scanf("%d%d",&n,&k);
		j = count = 0;
		while(j<n)
		{
			scanf("%d",&a[j]);
			if(a[j]%2==0)
				count++;
			j++;
		}
		if(k==0)
		{
			if(count==n)
				flag = 0;
			else
				flag = 1;
		}
		else
		{
			if(count>=k)
				flag = 1;
			else
				flag = 0;
		}
		if(flag==1)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
