#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k,i,a[50],count=0;
        scanf("%d %d",&n,&k);
        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
            if(a[i]%2==0)
                count++;
        }
        if(k==0)
        {
            if(count==n)
               printf("NO\n");
            else
                printf("YES\n");
            //continue;
        }
        else if(count>=k && k!=0)
        {
            printf("YES\n");
            //continue;
        }
        else
            printf("NO\n");
    }
    return 0;
}
