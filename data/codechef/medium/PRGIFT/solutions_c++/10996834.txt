#include <bits/stdc++.h>
using namespace std;

int main() {
	int T, N, K, i, count, a;
	cin>>T;
	while(T--) {
		count = 0;
		cin>>N>>K;
		for(i=0; i<N; i++) {
			cin>>a;
			if(a%2 == 0) {
				count++;
			}
		}
		if(K==0 && count < N) {
			cout<<"YES\n";
		}else if(K == 0 && count == N) {
			cout<<"NO\n";
		}else if(count >= K) {
			cout<<"YES\n";
		}else{
			cout<<"NO\n";
		}
	}
	return 0;
}
