#include <iostream>
using namespace std;

int main()
{
    int k,m,t;
    cin>>t;
    while(t--)
    {
    cin>>m>>k;
        int num,c=0,n;
        n=m;
    while(n--)
    {
        cin>>num;
        if(num%2==0)
        {
            c++;
        }
    }
    if(k==0&&c==m)
    {
        cout<<"NO"<<endl;
    }
    else if(k<=c)
        cout<<"YES"<<endl;
    else
        cout<<"NO"<<endl;
    }
    return 0;
}
