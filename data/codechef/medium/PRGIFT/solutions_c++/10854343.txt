#include<iostream>
using namespace std;
int main()
{
	int t, n, k, arr[10001];
	int ce;
	cin >> t;
	while (t--)
	{
		ce=0;
		cin>>n>>k;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i]%2==0)
				ce++;
         }
		if(ce<k||(k==0&&ce==n))
			cout<<"NO\n";
		else
			cout<<"YES\n";
		}
return 0;
}
