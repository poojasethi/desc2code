#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    long long int t,n,k,i,a[100005],count;
    cin>>t;
    while(t--)
    {
        count=0;
        cin>>n>>k;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if(a[i]%2==0)
            count++;
        }
        
        if(count<k || (count==n && k==0))
        cout<<"NO"<<endl;
        else
        cout<<"YES"<<endl;
    }
    return 0;
}
