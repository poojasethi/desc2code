#include<stdio.h>
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n, k, count=0, i, num;
		scanf("%d%d", &n, &k);
		for(i=0; i<n; i++)
		{
			scanf("%d", &num);
			if((num%2)==0)
				count++;
		}
		if(k==0 && count==n)
			printf("NO\n");
		else 
		{
			if(count>=k)
				printf("YES\n");
			else
				printf("NO\n");
		}
	}
	return 0;
}
