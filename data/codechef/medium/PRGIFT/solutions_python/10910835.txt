    #include <stdio.h>
     
    #define READ_INT(_x) \
    {	\
    int _tmp = 0; \
    register int _ch = getchar_unlocked(); \
    while (_ch < 48 || _ch > 57)	\
    _ch = getchar_unlocked(); \
    while (48 <= _ch && _ch <= 57)	\
    {	\
    _tmp = (_tmp << 3) + (_tmp << 1) + _ch - 48; \
    _ch = getchar_unlocked(); \
    }	\
    (_x) = _tmp; \
    }	\
     
    int main(void)
    {
    int k, n, t, x;
     
    READ_INT(t);
    while (t--)
    {
    READ_INT(n);
    READ_INT(k);
     
    if (k == 0)
    {
    for (; n > 0; n--)
    {
    READ_INT(x);
    if ((x & 0x1) == 1)
    {
    for (n--; n > 0; n--)
    READ_INT(x);
    k = 1;
    }
    }
    fputs_unlocked((k == 1) ? "YES\n" : "NO\n", stdout);
    }
    else
    {
    for (; n > 0; n--)
    {
    READ_INT(x);
    if ((x & 0x1) == 0)
    if (--k <= 0)
    for (n--; n > 0; n--)
    READ_INT(x);
    }
    fputs_unlocked((k <= 0) ? "YES\n" : "NO\n", stdout);
    }
    }
     
    return 0;
    }
