#include<stdio.h>

int main()
{
int t, n, k, i, j;
scanf("%d", &t);
for(j=0;j<t;j++)
{
scanf("%d %d", &n, &k);
int a[n];
int ans=0;
for(i=0;i<n;i++)
{
scanf("%d", &a[i]);
if(a[i]%2==0)
ans++;
}
if(k==0&&ans==n)
{
printf("NO\n");
continue;
}
if(ans>=k)
printf("YES\n");
else
printf("NO\n");

}

return 0;
}
