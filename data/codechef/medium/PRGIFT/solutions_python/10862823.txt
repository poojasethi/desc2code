#include <iostream>
using namespace std;

int main() 
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k,i;
        int e=0,o=0;
        cin>>n>>k;
        int a[n];
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            if(a[i]%2==0)
            {
                e++;
            }
            else
            {
                o++;
            }
        }
        if(e<k || (k==0 && e==n))
        {
            cout<<"NO\n";
        }
        else
        {
            cout<<"YES\n";
        }
    }
	return 0;
}
