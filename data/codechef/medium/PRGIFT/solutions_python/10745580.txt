#include<stdio.h>
int main(){
	int t,u,e;
	scanf("%d",&t);
	for(u=1;u<=t;u++){
		int n,k,tot=0,bo=0;
		scanf("%d %d",&n,&k);
		int a[n];
		for(e=0;e<n;e++){
		scanf("%d",&a[e]);
		if(a[e]%2==0){
	    if(e==0||e==n-1)bo++;
		tot++;
		}
	}
		if(k==0){
            if(tot==n)printf("NO\n");
			else printf("YES\n");
		}
		else {
		if(tot>=k)printf("YES\n");
		else printf("NO\n");
	}
	}
	return 0;
}
