#include<stdio.h>
int main()
{
  int t;
  scanf("%d",&t);
  while(t)
  {
  int n,w;
  scanf("%d %d",&n,&w);
  int i,a[n],count=0;
  for(i=0;i<n;i++){
  scanf("%d",&a[i]);
  if(a[i]%2==0 )
   count++;
   }
   if(w==0 && count==n)
   count=-1;
   if(count>=w)
   printf("YES\n");
   else
   printf("NO\n");
    t--;
  }
  return 0;
}
