#include <iostream>
#include <bits/stdc++.h>
 
 
using namespace std;
 
typedef long long int ll;
int main()
{
	std::ios::sync_with_stdio(false);
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n,k,ce=0;
    	cin>>n>>k;
    	vector<ll>v(n);
    	for(int i=0;i<n;i++)
    	{
    		ll a;
    		cin>>a;
    		v[i] = a;
		}
		int ok = false;
		for (int i = 0; i < n; i++) {
			for (int j = i; j < n; j++) {
				int cnt = 0;
				for (int k1 = i; k1 <= j; k1++) {
					if (v[k1] % 2 == 0) {
						cnt ++;
					}
				}
				if (cnt == k) {
					ok = true;
				}
			}
		}
		if (ok) {
			cout << "YES" << endl;
		} else {
			cout << "NO" << endl;
		}
	}
	return 0;
}
