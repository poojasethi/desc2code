#include<bits/stdc++.h>
using namespace std;
 
 
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int a[n];
        int count=0;
        int count1=0;
        for(int i=0;i<n;i++)
        {
              cin>>a[i];
              if(a[i]%2==0)
              {
                   count++;
              }
              else
              {
                   count1++;
              }
 
        }
        if(k==0)
        {
             if(count1>=1)
             {
                  cout<<"YES"<<endl;
             }
             else
             {
                   cout<<"NO"<<endl;
             }
        }
      else{
           if(count>=k)
           {
                cout<<"YES"<<endl;
           }
           else
           {
                cout<<"NO"<<endl;
           }
      }
 
    }
 
    return 0;
 
}
