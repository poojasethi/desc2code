#include<iostream>
using namespace std;
int main()
{
	int t, n, temp, k, arr[10001], max,curr;
	int co, ce;
	cin >> t;
	while (t--)
	{
		co=ce=temp=max=0;
		curr=0;
		cin>>n>>k;
		for(int i=0;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i]%2==0)
				ce++;
			else
				co++;
		}
		if(ce<k||(k==0&&ce==n))
			cout<<"NO\n";
		else
			cout<<"YES\n";
		}
return 0;
}
