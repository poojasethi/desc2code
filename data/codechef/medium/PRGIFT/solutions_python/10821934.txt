#include<stdio.h>
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k,i,p=0;
        scanf("%d%d",&n,&k);
        int a[n];
        for( i=0;i<n;++i){
            scanf("%d",&a[i]);
            if(a[i]%2==0)
                p++;}
                if(k==0){
                    if(p==n) printf("NO\n");
                    else printf("YES\n");
                }
                else if(p>=k)
                    printf("YES\n");
                else printf("NO\n");
            }
return 0;}
