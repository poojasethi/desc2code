#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t ; cin>>t ;
	while(t--)
	{
		long long int n , k , a , even = 0 ;
		cin>>n>>k;
		long long int nn = n ;
		while(nn--)
		{
			cin>>a;
			if(a%2==0)++even;
		}
		if(k==0 && even==n) cout<<"NO\n";
		else if(k<=even)cout<<"YES\n";
		else cout<<"NO\n";
	}
	return 0;
}
