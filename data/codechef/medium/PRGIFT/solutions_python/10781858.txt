#include<stdio.h>
int main()
{
    int t,i;
    scanf("%d",&t);
    for(i=1;i<=t;i++){
        int n,p,k;
        scanf("%d%d",&n,&k);
        int ar[n];
        for(p=0;p<n;p++){
            scanf("%d",&ar[p]);
        }
        int x,count=0;
        for(p=0;p<n;p++){
            x=ar[p]%2;
            if(x==0)
                count++;
        }
        if(k==0&&count!=n){
            printf("YES\n");
        }
        else
              if(count>=k && k!=0){
                    printf("YES\n");
              }
        else
            printf("NO\n");
    }
    return 0;
}
