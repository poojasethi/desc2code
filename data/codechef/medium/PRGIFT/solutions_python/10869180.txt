#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
   int t;
   scanf("%d",&t);
  while(t--)
  {
      unsigned int n,k;
      unsigned int flag= 0;
      scanf("%d %d",&n,&k);
      unsigned int arr[100];
      for(int i=0;i<n;++i)
      {
          scanf("%d",&arr[i]);
      }

      for(int i=0;i<n;i++)
      {
          if(arr[i]%2==0)
            {
                flag++;
                }
      }
       if((k==0)&&(flag==n))
        {
            cout<<"NO"<<endl;
        }


         else if(flag>=k)
            cout<<"YES"<<endl;
          else
            cout<<"NO"<<endl;


  }

}
