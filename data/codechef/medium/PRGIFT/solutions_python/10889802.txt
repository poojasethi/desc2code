#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,flag=0,c=0,item;
		cin>>n>>k;
		for(int i=0;i<n;i++)
		{
			cin>>item;
			if(item%2==0)
				c++;
		}
		if(k==0 && c==n)
			cout<<"NO"<<endl;
		else if(c>=k)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
		
	}
	return 0;
}
