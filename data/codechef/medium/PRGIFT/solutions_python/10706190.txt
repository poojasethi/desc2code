#include <stdio.h>
#define N 100
int a[N];
int main() {
    int t, n, k, i, j, k1, x, c;
     scanf("%d", &t);
    while(t--) {
        scanf("%d%d", &n, &k);
          x = 0;
          for(i=0; i<n; i++) {
            scanf("%d", &a[i]);
        }
        
        for (i = 0; i < n; i++) {
			for (j = i; j < n; j++) {
				c = 0;
				for (k1 = i; k1 <= j; k1++) {
					if (a[k1] % 2 == 0) {
						c++;
					}
				}
				if (c == k) {
					x = 1;
				}
			}
		}
        
        if(x)
            printf("YES\n");
        else
            printf("NO\n");
        
    }
    
    
	return 0;
}
