#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main() 
{
	long long int t,n,ar[100000],k,even;
	cin>>t;
	while(t--)
	{
	    even=0;
	    cin>>n>>k;
	    for(int i=0;i<n;i++)
	    {
	        cin>>ar[i];
	        if(ar[i]%2==0)
	        {
	            even++;
	        }
	    }
	    if(even<k || (even==n && k==0))
	    {
	        cout<<"NO\n";
	    }
	    else
	    {
	        cout<<"YES\n";
	    }
	}
	return 0;
}
