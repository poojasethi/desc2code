#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t>0){
        int n,k,i,a,count=0;
        cin>>n>>k;
        for(i=0;i<n;i++){
            cin>>a;
            if(a%2==0){
                count++;}
        }
        if (k==0){
            if(count==n){
                cout<<"NO"<<endl;}
            else{
                cout<<"YES"<<endl;}}
        else{
            if(count>=k){
                cout<<"YES"<<endl;}
            else{
                cout<<"NO"<<endl;}}
        t--;}
    return 0;}
