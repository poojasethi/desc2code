#include<stdio.h>
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k,i,j,p,h;
        scanf("%d%d",&n,&k);
        int a[n];
        for( i=0;i<n;++i)
            scanf("%d",&a[i]);
            int t=0;
            for(i=0;i<n;++i){
                if(t) break;
                    t=0;
                for( j=i;j<n;++j){
                    h=0;
                    for( p=i;p<=j;++p){
                        if(a[p]%2==0) h++;
                    }
                    if(h==k){
                        t=1; printf("YES\n"); break;
                    }}}

         if(t==0) printf("NO\n");
        }
return 0;}
