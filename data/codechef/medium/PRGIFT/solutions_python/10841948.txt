#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,count=0;
		cin>>n>>k;
		int i,a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		{
			if(a[i]%2==0)
			count++;
		}
		if(k==0)
		{
			if(count==n)
			cout<<"NO"<<"\n";
			else
			cout<<"YES"<<"\n";
		}
		else if(count>=k)
		cout<<"YES"<<"\n";
		else
		cout<<"NO"<<"\n";
	}
	return 0;
}
