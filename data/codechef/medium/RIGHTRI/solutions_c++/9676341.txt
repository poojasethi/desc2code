#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int n,count=0;
    cin>>n;
    while(n--)
    {
        int x1,y1,x2,y2,x3,y3;
        cin>>x1>>y1>>x2>>y2>>x3>>y3;
        int m1,m2,m3;
        
        
        m1= (x3-x2)*(x3-x2) + (y3-y2)*(y3-y2);
        
        m2= (x2-x1)*(x2-x1) + (y2-y1)*(y2-y1);
        
        m3 = (x1-x3)*(x1-x3) + (y1-y3)*(y1-y3);
             
        if(m1==m2+m3 || m2==m1+m3 || m3==m1+m2)
            count++;
        
    }
    cout<<count<<endl;
    return 0;
    
}
