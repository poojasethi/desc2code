#include <iostream>
#include<math.h>
using namespace std;

int main() {
	double a[3],b[3],s[3];
	int i,n;
	int count=0;
	cin>>n;
	while(n--)
	{
	
	  for(i=0;i<3;i++)
	  {
	    cin>>a[i];
	    cin>>b[i];
	  }
	  for(i=0;i<2;i++)
	  {
	    s[i]=pow((a[i+1]-a[i]),2)+(pow((b[i+1]-b[i]),2));
	  }
	   s[2]=pow((a[0]-a[2]),2)+(pow((b[2]-b[0]),2));
	  if((s[0]+s[1])==s[2] || (s[1]+s[2])==s[0] || (s[0]+s[2])==s[1])
	  count++;
	  
	}
	cout<<count;
	return 0;
}
