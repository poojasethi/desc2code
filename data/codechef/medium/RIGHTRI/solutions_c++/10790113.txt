#include<iostream>
using namespace std;
int main(){
	int n,x1,x2,x3,y1,y2,y3,count=0,i,a,b,c;
	cin>>n;
	for(i=0,count=0;i<n;i++){
		cin>>x1>>y1>>x2>>y2>>x3>>y3;
		a=(x2-x1)*(x2-x1)+(y2-y1)*(y2-y1);
		b=(x2-x3)*(x2-x3)+(y2-y3)*(y2-y3);
		c=(x3-x1)*(x3-x1)+(y3-y1)*(y3-y1);
		if(a==b+c||b==c+a||c==a+b)
		++count;
	}
	cout<<count;
	return 0;
}
