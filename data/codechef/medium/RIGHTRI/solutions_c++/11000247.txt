#include <iostream>
#include<cmath>
using namespace std;

int main() {
	int t,x1,x2,x3,y1,y2,y3,count=0;
	long long int d1,d2,d3;
	cin>>t;
	while(t--)
	{
		cin>>x1>>y1>>x2>>y2>>x3>>y3;
		d1=pow(x1-x2,2)+pow(y1-y2,2);
		//cout<<d1<<endl;
		d2=pow(x2-x3,2)+pow(y2-y3,2);
		//cout<<d2<<endl;
		d3=pow(x3-x1,2)+pow(y3-y1,2);
		//cout<<d3<<endl;
		if((d1+d2)==d3||(d2+d3)==d1||(d1+d3)==d2)
		{
			count++;
			//cout<<count<<" "<<t<<endl;
		}
	}
	cout<<count<<endl;
	return 0;
}
