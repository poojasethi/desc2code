#include <iostream>
#include<math.h>
#include<stdio.h>
using namespace std;

int main() 
{
    long long t, count=0;
    scanf("%lld",&t);
    
    while(t--)
    {
        long long x1, y1, x2, y2, x3, y3;
        long long a,b,c;
        scanf("%lld",&x1);
        scanf("%lld",&y1);
        scanf("%lld",&x2);
        scanf("%lld",&y2);
        scanf("%lld",&x3);
        scanf("%lld",&y3);
        a=pow(x1-x2,2)+pow(y1-y2,2);
        b=pow(x3-x2,2)+pow(y3-y2,2);
        c=pow(x1-x3,2)+pow(y1-y3,2);
        //printf("%lld %lld %lld\n",a,b,c);
        if((a==(b+c))||(b==a+c)||(c==a+b))
            count++;
    }
    
    printf("%lld",count);
    
	return 0;
}
