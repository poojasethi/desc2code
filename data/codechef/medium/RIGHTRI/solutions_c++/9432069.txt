#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    int s=0;
    while(t--)
    {
        int x1,y1,x2,y2,x3,y3,X1,Y1,X2,Y2,X3,Y3;
        cin>>x1>>y1>>x2>>y2>>x3>>y3;
        X1 = x2-x1;
        Y1 = y2-y1;

        X2 = (x3-x1);
        Y2 = (y3-y1);

        X3 = (x3-x2);
        Y3 = (y3-y2);

        if(Y1*Y2 + X1*X2 ==0)
            s++;
        else if(Y3*Y2 + X3*X2 ==0)
            s++;
        else if(Y1*Y3 + X1*X3 ==0)
            s++;
    }
    cout<<s<<endl;
}
