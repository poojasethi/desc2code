#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int a,b,c;
	cin>>a;
	b=0;
	while(a--)
	{
		int p=0;
      int x1,y1,x2,y2,x3,y3,l,m,n;
      cin>>x1>>y1>>x2>>y2>>x3>>y3;
      l=pow((x1-x2),2)+pow((y1-y2),2);
      l>p ? p=l : p ;
      m=pow((x2-x3),2)+pow((y2-y3),2);
      m>p ? p=m : p ;
      n=pow((x1-x3),2)+pow((y1-y3),2);
      n>p ? p=n : p ;
      if(p==n && p==(l+m))
      {
      	b++;
      }
      else if(p==m && p==(n+l))
      {
      	b++;
      }
      else if(p==l && p==(n+m))
      {
      	b++;
      }
	}
	cout<<b<<endl;
	return 0;
}
