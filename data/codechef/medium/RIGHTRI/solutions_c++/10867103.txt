#include<cstdio>
#include<cstring>
using namespace std;
int main(){
    int T,count=0;
    scanf("%d",&T);
    while(T--){
        int x1,y1,x2,y2,x3,y3;
        scanf("%d %d %d %d %d %d",&x1,&y1,&x2,&y2,&x3,&y3);
        int vx1,vy1,vx2,vy2,vx3,vy3;
        vx1=x1-x2;
        vy1=y1-y2;
        vx2=x1-x3;
        vy2=y1-y3;
        vx3=x3-x2;
        vy3=y3-y2;
        if(((vx1*vx2)+(vy1*vy2)==0)||((vx3*vx2)+(vy3*vy2)==0)||((vx1*vx3)+(vy1*vy3)==0)){
            count++;
        }
    }
    printf("%d\n",count);
}
