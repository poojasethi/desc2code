#include<cstdio>
int main()
{
 int n,i,count=0;
 scanf("%d",&n);
 for(i=0;i<n;i++)
 {
  int x1,x2,x3,y1,y2,y3;
  scanf("%d%d%d%d%d%d",&x1,&y1,&x2,&y2,&x3,&y3);
  int a,b,c;
  a=(y2-y1)*(y2-y1)+(x2-x1)*(x2-x1);
  b=(y3-y2)*(y3-y2)+(x3-x2)*(x3-x2);
  c=(y1-y3)*(y1-y3)+(x1-x3)*(x1-x3);

 if(a>b)
 {
     if(a>c)
     {
         if(a==b+c)
            count++;

     }
     else
     {
         if(c==a+b)
            count++;

     }

 }
 else
 {
     if(b>c)
     {
         if(b==a+c)
            count++;

     }
     else
     {
         if(c==a+b)
            count++;
     }

 }

 }
 printf("%d",count);

}
