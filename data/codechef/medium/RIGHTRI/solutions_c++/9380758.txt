#include<bits/stdc++.h>

using namespace std;

int sqr(int x)
{
   return x*x;
}

int main(void)
{
   int n; scanf("%d",&n);
   int tri=0;
   
   while(n--)
   {
      int x1,x2,x3,y1,y2,y3;
      scanf("%d %d %d %d %d %d",&x1,&y1,&x2,&y2,&x3,&y3);
      
      int l1=sqr(x1-x2)+sqr(y1-y2),
          l2=sqr(x2-x3)+sqr(y2-y3),
          l3=sqr(x3-x1)+sqr(y3-y1);
      
      if( l1+l2==l3 ||
          l2+l3==l1 ||
          l1+l3==l2)
          tri++;
    }
    printf("%d",tri);
}
