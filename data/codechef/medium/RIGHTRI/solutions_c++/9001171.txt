#include <stdio.h>
#include <math.h>
#include <iostream>

using namespace std;

int dist(int x1, int y1, int x2, int y2)
{
	int a,b; int c;
	a=(x1-x2)*(x1-x2);
	b=(y1-y2)*(y1-y2);
	c=a+b;
	return c;
}

int main()
{
	int ar[6]; int d1,d2,d3,c=0,n;
   cin>>n;
   while(n--)
   {
	for(int i=0;i<6;i++)
		cin>>ar[i];
	d1= dist(ar[0],ar[1],ar[2],ar[3]);
	d2= dist(ar[0],ar[1],ar[4],ar[5]);
	d3= dist(ar[2],ar[3],ar[4],ar[5]);

	if(d1==d2+d3 || d2==d1+d3 || d3==d1+d2) c++;
   }
	printf("%d",c);

}
