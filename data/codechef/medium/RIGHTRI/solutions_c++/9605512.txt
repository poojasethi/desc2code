#include<iostream>
#include<cstdio>
#include<cmath>

using namespace std;

int main()
{
	int N;
	int x[3],y[3];
	int a,b,c,count=0;
	scanf("%d",&N);
	
	while(N--)
	{
		scanf("%d%d%d%d%d%d",&x[0],&y[0],&x[1],&y[1],&x[2],&y[2]);
		a=pow(x[0]-x[1],2) + pow(y[0]-y[1],2);
		b=pow(x[1]-x[2],2) + pow(y[1]-y[2],2);
		c=pow(x[2]-x[0],2) + pow(y[0]-y[2],2);
		
		if((a+b==c) || (b+c==a) || (c+a==b))
			count++;
		
	}
	
	printf("%d",count);
	
	return 0;
}
