#include <iostream>
#include<cmath>
#include <cstdio>
#define dist(x1,x2,y1,y2,d) d=pow(x1-x2,2)+pow(y1-y2,2)
using namespace std;

int main() {
	cin.sync_with_stdio(false);
	cout.sync_with_stdio(false);
	int t,x1,x2,x3,y1,y2,y3,count=0;
	long long int d1,d2,d3;
	cin>>t;
	while(t--)
	{
		cin>>x1>>y1>>x2>>y2>>x3>>y3;
		dist(x1,x2,y1,y2,d1);
		//cout<<d1<<endl;
		dist(x2,x3,y2,y3,d2);
		//cout<<d2<<endl;
		dist(x3,x1,y3,y1,d3);
		//cout<<d3<<endl;
		if((d1+d2)==d3||(d2+d3)==d1||(d1+d3)==d2)
		{
			count++;
			//cout<<count<<" "<<t<<endl;
		}
	}
	cout<<count<<endl;
	return 0;
}
