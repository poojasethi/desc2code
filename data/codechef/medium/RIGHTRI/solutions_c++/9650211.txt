#include<iostream>
#include<string.h>
 
using namespace std;
 
int main()
{
    int t,ans=0, x1,y1,x2,y2,x3,y3;
    cin>>t;
    while(t--)
    {
 
       cin>>x1>>y1>>x2>>y2>>x3>>y3;
       if((x3-x2)*(x2-x1)+(y3-y2)*(y2-y1)==0||(x2-x1)*(x1-x3)+(y2-y1)*(y1-y3)==0||(x2-x3)*(x3-x1)+(y2-y3)*(y3-y1)==0)
       {
           ans++;
       }
      }
      cout<<ans;
    return 0;
}
