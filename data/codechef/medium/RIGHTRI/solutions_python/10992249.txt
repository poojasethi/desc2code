import math
t=input()
total=0
for i in range(t):
	x1,y1,x2,y2,x3,y3 = map(int, raw_input().split())
	d1=math.pow(x1-x2,2)+math.pow(y1-y2,2)
	d2=math.pow(x1-x3,2)+math.pow(y1-y3,2)
	d3=math.pow(x2-x3,2)+math.pow(y2-y3,2)
	

	
	l1=[d1,d2,d3]
	
	m=sorted(l1)
	


	if m[0]+m[1]==m[2]:
		total+=1
print total
