t=(int)(raw_input())
coun=0
for i in range(t):
    arr=[int(x) for x in raw_input().split()]
    l1=(arr[0]-arr[2])**2+(arr[1]-arr[3])**2
    l2=(arr[0]-arr[4])**2+(arr[1]-arr[5])**2
    l3=(arr[2]-arr[4])**2+(arr[3]-arr[5])**2
    if(l1+l2==l3)|(l2+l3==l1)|(l3+l1==l2):
        coun=coun+1
print coun
