def findlengthsquare(x1,y1,x2,y2):
	a = x2 - x1
	b = y2 - y1
	return a*a + b*b
n = int(raw_input())
count = 0
while n > 0:
	x1,y1,x2,y2,x3,y3 = map(int,raw_input().split())
	l1 = findlengthsquare(x1,y1,x2,y2)
	l2 = findlengthsquare(x2,y2,x3,y3)
	l3 = findlengthsquare(x3,y3,x1,y1)
	lst=[]
	lst = [l1,l2,l3]
	lst = sorted(lst)
	if lst[1] + lst[0] == lst[2] :
		count += 1
	n -= 1	
print count
