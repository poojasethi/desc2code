def ritri(a):
	b=((a[2]-a[0])**2)+((a[3]-a[1])**2)
	c=((a[4]-a[2])**2)+((a[5]-a[3])**2)
	d=((a[4]-a[0])**2)+((a[5]-a[1])**2)
	if b+c==d or b+d==c or c+d==b:
		return 1
	else:
		return 0
