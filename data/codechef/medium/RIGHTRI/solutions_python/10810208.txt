n=input();
i=0
count=0
while (i<n):
	x1,y1,x2,y2,x3,y3=[int(x) for x in raw_input().split()]
	d1=x1-x2
	d2=x2-x3
	d3=x3-x1
	e1=y1-y2
	e2=y2-y3
	e3=y3-y1
	dis1=((d1**2)+(e1**2))
	dis2=((d2**2)+(e2**2))
	dis3=((d3**2)+(e3**2))
	if ((dis1+dis2==dis3) or (dis2+dis3==dis1) or (dis1+dis3==dis2)):
		count+=1
	i+=1
print count
