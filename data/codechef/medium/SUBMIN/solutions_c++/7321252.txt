#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
using namespace std;
int main()
{
    int i,n,mini,c,k,q,j;
    scanf("%d",&n);
    vector<int> v(n);
    for(i=0;i<n;i++)
    {
        scanf("%d",&(v[i]));
    }
    scanf("%d",&q);
    while(q--)
    {
        c=0;
        scanf("%d",&k);
        for(i=0;i<n;i++)
        {
            mini=v[i];
            if(mini<k)
            {
                continue;
            }
            else
            {
                if(mini==k)
                {    c++;
                }
                for(j=i+1;j<n;j++)
                {
                    if(v[j]<mini)
                    {
                        mini=v[j];
                        if(mini<k)
                        {
                            break;
                        }
                        if(mini==k)
                        {
                            c++;
                        }
                    }
                    else
                    {
                        if(mini==k)
                            c++;
                    }
                }
            }
        }
         printf("%d\n",c);
    }
    return 0;
}
