#include<bits/stdc++.h>
using namespace std;
int rmq[7][52];
int idx[55],diff[55];
int a[55];
inline void pre()
{
    int cnt=0;
    int k=1;
    for(int i=1;i<52;i++) {
        diff[i]=k;
        idx[i]=cnt;
        if((i&(i+1))==0)
            cnt++,k=k<<1;
    }
}
inline void build(int n) {
    for(int j=0;j<n;j++) {
        rmq[0][j]=a[j];
    }
    for(int i=1;i<6;i++) {
        int len=1<<(i-1);
        for(int j=0;j<n;j++) {
            if(j+len>n)
                rmq[i][j]=min(rmq[i-1][j],10000000);
            else
                rmq[i][j]=min(rmq[i-1][j],rmq[i-1][j+len]);
       }
    }
}
inline int query(int l, int r) {
    int p=idx[r-l+1];
    return min(rmq[p][l],rmq[p][r-diff[r-l+1]+1]);
}
int main()
{
    pre();
    int n;cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    build(n);
    int q;cin>>q;
    while(q--) {
        int val;
        cin>>val;
        long long ans=0;
        for(int i=0;i<n;i++)
        {
            for(int j=i;j<n;j++)
            {
                if(query(i,j)==val)
                    ans++;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
