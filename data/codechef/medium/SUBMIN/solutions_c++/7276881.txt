#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<map>
#include<set>
#include<cmath>
using namespace std;
int n,q,k,A[52];
int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d",&A[i]);
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d",&k);
		int flag=1,res=0;
		for(int i=0;i<n;i++)
		if(A[i]==k)
		{
			int l=1,r=0,j=i;
			while(j<n&&A[j]>=k){r++,j++;}

			if(flag==1){	
				j=i-1;
				while(j>=0&&A[j]>=k){l++,j--;}
			}
			else{
				j=i-1;
				while(j>=0&&A[j]>k){j--,l++;}
			}
			res+=l*r;
			flag=0;
		}
		cout<<res<<"\n";
	}
	return 0;
}
