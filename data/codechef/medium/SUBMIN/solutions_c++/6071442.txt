#include<iostream>
using namespace std;
int main()
{
	int n,q,k,i,j;
	int a[51];
	int p[51][51];
	int c[1000001];
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	cin>>q;
	for(i=0;i<=1000000;i++)
	c[i]=0;
	for(i=0;i<n;i++)
	{
	p[i][i]=a[i];
	c[a[i]]++;
	}
	
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[j]<p[i][j-1])
			p[i][j]=a[j];
			else
			p[i][j]=p[i][j-1];
			c[p[i][j]]++;
		}
	}
	while(q--)
	{
		cin>>k;
		cout<<c[k]<<"\n";
	}
	return 0;
}
