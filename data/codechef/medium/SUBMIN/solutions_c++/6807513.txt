#include<bits/stdc++.h>
#define MAX 52
using namespace std;

int main()
{
    int t,a[MAX],q,i,j,k;
    cin>>t;
    for(i=0;i<t;i++)
        cin>>a[i];
    cin>>q;
    while(q--)
    {
        int i,j,min,ans=0,flag,yes=0;
        cin>>min;
        for(i=0;i<t;i++)
        {
            flag=0;
            yes=0;
            for(j=i;j<t;j++)
            {
                if(min==a[j]&&flag==0)
                {
                    flag=1;
                    yes=j-i;
                }
                if(min>a[j])
                   break;
            }
            if(flag)
            {
                ans +=(j-i)-yes;
            }
            //cout<<ans<<"   ";
        }
        cout<<ans<<endl;

    }
    return 0;
}
