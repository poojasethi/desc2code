
#include <iostream>

int main()
{
  long long int n ;
  
  cin >> n; 
  long long int a[n];
  for(long long int j=0; j<n; j++)
  {
  	cin >> a[j];
   } 
   
   
   long long int t ,e;
   cin >> t;
   
   for(long long int g=0 ; g<t ;g++)
   {
   	
   	cin >> e;
   	
   	long long int count=0 ;
   	
   	for(long long int k=0; k<n; k++)
   	{
   		
   		for(long long int l=k; l<n ; l++)
   		{
   		  long long int x = foo(k , l , a);
			 
			 	
			 	if(x==e)
			 	{
			 		
			 		count++;
				 }
				 	
   			
		   }
	   }
	   
	   cout  << count << endl;
   }

  return 0;   
}
 



 int foo(long long int p , long long int q , long long int arr[])
 {
 long long int min ;
 min = arr[p];
 for(long long int i=p; i<=q; i++)
{
	if(min > arr[i])
	{
		min = arr[i];
	}
}

return min;
 	
 }
