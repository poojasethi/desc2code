#include<bits/stdc++.h>
using namespace std;
int arr[100],front[100]={0},back[100]={0};
int main()
{
	int n,q;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&arr[i]);
	}
	back[0]=0;
	int counter=0,maxi=-1,k;
	for(int i=1;i<n;i++)
	{
		for(int j=i-1;j>=0;j--)
		{
			if(arr[i]<arr[j])
				counter++;
			else
				break;
		}
		back[i]=counter;
		counter=0;
	}
	front[n-1]=0;
	maxi=-1;
	for(int i=n-1;i>=0;i--)
	{
		for(int j=i+1;j<n;j++)
		{
			if(arr[i]<=arr[j])
				counter++;
			else
				break;
		}
		front[i]=counter;
		counter=0;
	}
/*	for(int i=0;i<n;i++)
		printf("%d %d\n",back[i],front[i]);*/
	scanf("%d",&q);
	long long int ans=0;
	while(q--){
		ans=0;
		scanf("%d",&k);
		for(int i=0;i<n;i++)
		{
			if(arr[i]==k){
				ans=ans+(front[i]+1)+(front[i]+1)*back[i];				
			}			
		}
		printf("%lld\n",ans);


	}
	return 0;
}
