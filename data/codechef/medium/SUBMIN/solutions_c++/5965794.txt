#include<bits/stdc++.h>
#define ll long long int
#define g1 getchar
#define MAX 1000001

using namespace std;

inline ll inp()
{
	ll x=0,sign=1;
	char c;
	while(((c=g1())<'0'||c>'9')&&c!='-');
	if(c=='-')	sign=-1,c=g1();
	while(c>='0'&&c<='9')
	x=(x<<3)+(x<<1)+c-48,c=g1();
	return sign*x;
}

ll min(ll a , ll b)
{
    return(a < b)?a : b;
}

ll b[MAX] = {0};

int main()
{
   ll N;

   N = inp();

   ll a[N];

   for(int i = 0 ; i < N ; i++)
     a[i] = inp();

    for(int i = 0 ; i < N ; i++)
    {
        int y = a[i];

        for(int j = i ; j < N ; j++)
        {
            y = min(y , a[j]);

            ++b[y];
        }
    }

    ll k , g;

    k = inp();

    while(k--)
    {
        g = inp();

        cout << b[g] << endl;
    }

    return 0;
}
