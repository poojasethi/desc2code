#include <iostream>

using namespace std;

int main()
{

int n;
cin>>n;
int a[n];
for(int i=0;i<n;i++)cin>>a[i];

int q;
cin>>q;
int ans[q];

for(int i=0;i<q;i++)
{
int x;
cin>>x;

int count =0;
for(int j=0	;j<n;j++)
{	if(a[j]<x)j++;

	for(int k=j;k<n;k++)
	{ 
		bool havesmall = false;
		bool have = false;
		
		for(int l=j;l<=k;l++)
		{	if(a[l]<x){havesmall=true;break;}
			if(a[l]==x){have=true;}
		}
		if(!havesmall && have)count++;
	}
}

ans[i]=count;
//cout<<count<<endl;
}


for(int i=0;i<q;i++)cout<<ans[i]<<endl;

}
