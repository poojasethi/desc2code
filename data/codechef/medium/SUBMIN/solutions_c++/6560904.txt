typedef long long int               ll;
 
using namespace std;

int min(int N[], int i, int j) {
	if (i == j) {
		return N[i];
	}
	int mn = N[i];
	forall(k, i + 1, j + 1) {
		if (N[k] < mn) {
			mn = N[k];
		}
	}
	return mn;
}

int main()
{
	int n, q, k;
	s(n);
	ll cnt;

	}
	
	s(q);
	w(q) {
		s(k);
		vector<int> ind;
		cnt = 0;
		
		forall(i, 0, n) {
			forall(j, i, n) {
				if (k == min(N, i, j)) {
					cnt++;
				}	
			}
		}
			
		prnt_l(cnt);
	}
	
	return 0;
}
