#include<iostream>
using namespace std;
int main()
{
int n,i,j;
cin>>n;
long a[n];
for(i=0;i<n;i++)
cin>>a[i];
int k;
cin>>k;
while(k--)
{
long m;
long sum=0;
int arr[n];
for(i=0;i<n;i++)
arr[i]=-1;
int k=0;
cin>>m;
for(i=0;i<n;i++)
{
if(a[i]==m)
{
arr[k]=i;
k++;
}
}
i=0;
while(arr[i]!=-1&&i!=n)
{
j=arr[i];
int y=1;
int left=0;
while(j!=0)
{
if(a[arr[i]-y]>m)
{
left++;
y++;
}
j--;
}
j=arr[i];
y=1;
int right=0;
while(j!=n-1)
{
if(a[arr[i]+y]>=m)
{
right++;
y++;
}
j++;
}
sum=sum+left*(right+1)+right+1;
i++;
}
cout<<sum<<"\n";
}
}
