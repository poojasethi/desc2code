	#include<iostream>
	using namespace std;
	int main()
	{
			int n;
			cin>>n;
			int a[100];
			int b[1000001]={0};
			int min;
			for(int i=0;i<n;i++)
				cin>>a[i];
			for(int i=0;i<n;i++)
			{
				min=a[i];
				for(int j=i;j<n;j++)
				{
					if(min>a[j])
						min=a[j];
					b[min]++;
				}	
			}	
			int q;
			cin>>q;
			int temp;
			for(int i=0;i<q;i++)
			{
				cin>>temp;
				if(b[temp]>0)
					cout<<b[temp]<<endl;
				else
					cout<<"0\n";
			}	
	}
