#include<iostream>
using namespace std;
int solve(int first,int idx,int last)
{
   return (last-idx+1)*(idx-first+1);
}
int main()
{
   int N,Q,K,i,first,ans,last,newfirst,newi,arr[51];
   cin>>N;
   for(i=0;i<N;i++)
   cin>>arr[i];
   cin>>Q;
   while(Q--)
   {
      cin>>K;
      for(ans=i=first=0;i<N;i++)
      {
         if(arr[i]==K)
         {
            newfirst=first;
            for(last=i+1;last<N&&arr[last]>K;last++);
            last--;
            ans+=solve(first,i,last);
            if(arr[last]<K)
            first=last+1;
            i=last;
         }
         else if(arr[i]<K)
         first=i+1;
      }
      cout<<ans<<endl;
   }
return 0;
}
