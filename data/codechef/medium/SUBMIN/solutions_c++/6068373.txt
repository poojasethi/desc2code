#include<stdio.h>
#include<iostream>
#include<math.h>                      
#include<stdlib.h>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std; 

int main()
{
	 int n,t,i,j;
	 int *count = new int[1000006];
	 memset(count,0,sizeof(count));
	 long arr[60],mini,k;
  cin>>n;
  for(i=0;i<n;i++)  scanf("%ld",&arr[i]);
  
  for(i=0;i<n;i++)
  {   mini=arr[i];
    for(j=i;j<n;j++)
	 {
	   mini=min(mini,arr[j]);    count[mini]++;
	 }
  }
  
  cin>>t;
  while(t--)
  { 
	  scanf("%ld",&k);
	  cout<<count[k]<<endl;	  
  }
	  return 0;
}
