# include <bits/stdc++.h>
using namespace std;
typedef long long ll;

int main() {
	int n,a[55],i,j,k,q;
	scanf("%d",&n);
	int m[55][55]={0};
	for (i=0 ; i < n ; m[i][i]=a[i],i++)
		scanf("%d",a+i);
	for (i=0 ; i < n ; i++) {
		for (j=i+1 ; j < n ; j++) {
			m[i][j]=1e7;
			for (k=i ; k <= j ; k++) {
				m[i][j] = min(m[i][j],a[k]);
			}
		}
	}
	scanf("%d",&q);
	while (q--) {
		int ans=0;
		scanf("%d",&k);
		for (i=0 ; i < n ; i++) {
			for (j=i ; j < n ; j++) {
				if(m[i][j]==k)
					ans++;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
