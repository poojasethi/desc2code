#include <stdio.h>
 
#define min(X,Y) ((X)<(Y)?(X):(Y))
 
int main(void) {
	// your code goes here
	int n,q,arr[51][51],i,j,k,count;
	
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&arr[0][i]);
		// printf("%d ",arr[0][i]);
	}
	// printf("\n");
	
	for(j=1;j<n;j++){
		for(i=0;i<n-j;i++){
			arr[j][i] = min(arr[j-1][i],arr[j-1][i+1]);
			// printf("%d ",arr[j][i]);
		}
		// printf("\n");
	}
	
	scanf("%d",&q);
	while(q--){
		scanf("%d",&k);
		count=0;
		
		for(j=0;j<n;j++){
			for(i=0;i<n-j;i++){
				if(arr[j][i] == k){
					count++;
				}
			}
		}
		printf("%d\n",count);
	}
	return 0;
}
