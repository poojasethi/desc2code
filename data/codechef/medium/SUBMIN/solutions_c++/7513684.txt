const int N=51;
int dp[N][N];
int a[N];
int f[1000001];
int main() {
    M(dp,-1);
    int n;sci(n);
    F(i,0,n) sci(a[i]);
    F(i,0,n) {
        dp[i][i]=i;
        f[a[i]]++;
    }
    F(i,0,n) {
        F(j,i+1,n) {
           if(a[dp[i][j-1]]<a[j]) {
               dp[i][j]=dp[i][j-1];
               f[a[dp[i][j]]]++;
           }
           else {
               dp[i][j]=j;
               f[a[dp[i][j]]]++;
           }
        }
    }
    int q;sci(q);
    while(q--) {
        int m;sci(m);
        cout<<f[m]<<endl;
    }
}
