#include<bits/stdc++.h>
#define in(t) scanf("%d",&t)
#define llin(t) scanf("%lld",&t)
#define lin(t) scanf("%ld",&t)
#define out(t) printf("%d\n",t);
#define lout(t) printf("%ld\n",t);
#define for(i,j,n) for(i=j;i<n;i++)
using namespace std;
long int a[50];
int main()
{
	int n,q,i,j,cnt1,cnt2;
	long int k,ans;
	in(n);
	for(i,0,n)
	{
		lin(a[i]);
	}
	in(q);
	while(q--)
	{
		lin(k);
		ans=0;
		for(i,0,n)
		{
			cnt1=1;
			cnt2=1;
			if(a[i]==k)
			{
				//cout<<"found.........."<<i<<"     \n\n";
				j=i-1;
				while(j>=0 && a[j]>k)
				{
					j--;
					cnt1++;	
				}
			//	cout<<cnt1<<"\n\n";
				j=i+1;
				while(j<n && a[j]>=k)
				{
					j++;
					cnt2++;
				}
			
				ans=ans+(cnt1*cnt2);
			}
			
			
		}
		lout(ans);
	}
	
}
