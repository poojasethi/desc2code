#include <iostream>
using namespace std;

int main() {
	int min,q,n,k,ar[50],index;
	long int sum=0;
	cin>>n;
	for(int i=0;i<n;i++)cin>>ar[i];
	cin>>q;
	while(q--) {
		cin>>k;
		min=k,index=0;
		sum=0;
	for(int j=0;j<n;j++) {
		while(ar[j]< min && j< n) j++;
		index=j;
		while(ar[j]>=min && j<n){
			j++;
		}
		int flag=0;
		for(int p=index;p<j;p++){
			flag=0;
			for(int q=p;q<j;q++){
				if(ar[q]==min) flag++;
				if(flag) sum++;
			}
		}
	}
	cout<<sum<<endl;
}
return 0;
}
