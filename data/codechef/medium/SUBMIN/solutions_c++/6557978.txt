#include<iostream>

using namespace std;

int minm(int arr[] , int i , int j)
{
	if(i==j){
		return arr[i];
	}
	int k = 0;
	int mn = arr[i];
	for(k=i+1;k<=j;k++){
		if(mn>arr[k]){
			mn=arr[k];
		}
	}
	return mn;
}

int main()
{
	int t;
	int n , i , j;
	cin>>n;
	int arr[n];
	int q; 
	int cou=0;
	for(i = 0 ; i < n ; i++){
		cin>>arr[i];
	}
	cin>>q;
	int k;
	while(q--){
		cou=0;
		cin>>k;
		for(i=0;i<n;i++){
			for(j=i;j<n;j++){
				if(k==minm(arr,i,j)){
					cou++;
				}
			}
		}
		cout<<cou<<endl;
	}
}
