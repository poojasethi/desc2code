#include<bits/stdc++.h>
#define ARR_SIZE(arr) (sizeof(arr)/sizeof(arr[0]))
#define MOD 1000000007

using namespace std;
typedef long long int ll;

int main(){
	int n, q, a[52], b[52], k, i, j, left=0, right=0;
	ll sum=0;
	scanf("%d", &n);
	for(i=0; i<n; i++)
		scanf("%d", &b[i]);
	scanf("%d", &q);
	while(q--){
		sum=0;
		for(i=0; i<n; i++)
			a[i]=b[i];
		scanf("%d", &k);
		for(i=0; i<n; i++)
			if(a[i]==k){
				left=0;
				right=0;
				for(j=i; j>=0 && a[j]>=k; j--)
					left++;
				for(j=i; j<n && a[j]>=k; j++)
					right++;
				//cout<<left<<" "<<right<<endl;
				sum+=left*right;
				a[i]=-1;
			}
			printf("%lld\n", sum);
		}
	return 0;
}
