#include<iostream>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
        cin>>a[i];
    int q;
    cin>>q;

    while(q--)
    {
        int total=0;
        int k;
        cin>>k;
        for(int i=0;i<n;i++)
        {
            if(a[i]==k)
                total++;
            if(a[i]>=k)
            {
                int min=a[i];
                for(int j=i+1;j<n;j++)
                {
                    if(a[j]<min)
                        min=a[j];
                    if(min==k)
                        total++;
                    if(min<k)
                        break;
                }
            }
        }
        cout<<total<<endl;
    }
    return 0;
}
