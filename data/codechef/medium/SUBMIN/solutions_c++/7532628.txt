#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int main()
{
	lli i,j,k,m,n,t,c,c1,c2,ans,q,flag;
	scanf("%lld",&n);
	lli arr[n];

	for(i=0;i<n;i++)
		scanf("%lld",&arr[i]);

	scanf("%lld",&q);
	while(q--)
	{
		scanf("%lld",&m);
		ans=0;
		flag=0;
		for(i=0;i<n;i++)
		{
			if(arr[i]<m)
				flag=0;

			if(arr[i]==m)
			{
				if(flag==0)
				{
					j=i;
					c1=0;
					while(arr[j]>=m && j>=0)
					{
						j--;
						c1++;
					}
					c2=0;
					j=i;
					while(arr[j]>=m && j<n)
					{
						j++;
						c2++;
					}
					c=c1*c2;
					ans+=c;
					flag=1;
				}
				else if(flag==1)
				{
					j=i;
					c1=0;
					while(arr[j]>=m && j<n)
					{
						j++;
						c1++;
					}
					j=i-1;
					c2=0;
					while(arr[j]>m && j>=0)
					{
						j--;
						c2++;
					}
                                        c2+=1;
					c=c1*c2;
                                        ans+=c;
				}
			}
		}
		printf("%lld\n",ans);
	}
}
