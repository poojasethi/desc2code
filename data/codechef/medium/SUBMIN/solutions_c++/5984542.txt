#include <cstdio>
#include <algorithm>

using namespace std;

int n, x[55], q, k;

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &x[i]);
    scanf("%d", &q);
    while (q--) {
        scanf("%d", &k);
        int l = 0, r = 0, sol = 0;
        for (int i = 0; i < n; i++) {
            if (x[i] < k) l = i + 1;
            if (x[i] != k) continue;
            r = max(r, i);
            while (r + 1 < n && x[r + 1] >= x[i]) r++;
            sol += (i - l + 1) * (r - i + 1);
            l = i + 1;
        }
        printf("%d\n", sol);
    }
    return 0;
}
