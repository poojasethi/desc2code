#include<iostream>
#include<algorithm>
using namespace std;
int Subcount(int arr[],int n,int k){
   int ans=0;
   for(int i=0;i<n;i++){
     int min_el=arr[i];
     if(min_el==k) ans++;
    for(int j=i+1;j<n;j++){
       min_el=min(min_el,arr[j]);
       if(min_el==k) ans++;
     } 
   }
   return ans;
}
int main(){
 ios::sync_with_stdio(0);
 int n,q,k;
 cin>>n;
 int arr[n];
 for(int i=0;i<n;i++){
  cin>>arr[i];
 }
 cin>>q;
 while(q--){
  cin>>k;
  cout<<Subcount(arr,n,k)<<endl;
 }
 return 0;
}
