#include<bits/stdc++.h>
using namespace std;
const int MAXN = 50 + 10;
const int INF = 1e9;
int A[MAXN];
int main(){
    int N;
    cin >> N;
    for(int i = 1;i <= N;++i)
        cin >> A[i];
    int Q;
    cin >> Q;
    while(Q--){
        int x;
        cin >> x;
        int ans = 0;
        for(int i = 1;i <= N;++i){
            int mini = INF;
            for(int j = i;j <= N;++j){
                mini = min(mini, A[j]);
                if(mini == x) ++ans;
            }
        }
        printf("%d\n", ans);
    }
    return 0;
}
