#include<stdio.h>
#include<math.h>

int min(long int a[],long int y, int i, int j)
{
	for(int k=i;k<=j;k++)
	{
		if(a[k]<y)
		y=a[k];
	}
	return y;
}

int b[1000000];

int main()
{
	int n,i,j,q,flag;
	long int a[55],y,t,ans;
	
	
	scanf("%d",&n);
	
	
	for(i=0;i<n;i++)
	{
		scanf("%ld",&a[i]);
		b[a[i]]=0;
	}
	
	for(i=0;i<n;i++)
	{
		y=a[i];
		
		for(j=i;j<n;j++)
		{
			y=min(a,y,i,j);
			++b[y];
		}
	}
	
	
	scanf("%d",&q);
	
	
	while(q--)
	{
		scanf("%ld",&t);
		
		for(i=0;i<n;i++)
		{
			if(a[i]==t)
			flag=1;
		}
		
		if(flag==1)
		{
		 ans=b[t];
	    }
	    else
	    ans=0;
		printf("%ld\n",ans);
	}
	return 0;
}
