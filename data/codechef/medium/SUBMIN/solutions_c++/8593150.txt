#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
   int n,arr[100],q,k;
   scanf("%d",&n);
   for(int i=0; i<n; i++)
    scanf("%d",&arr[i]);
   scanf("%d",&q);
   while(q--)
   {
       scanf("%d",&k);
       int cnt = 0;
       for(int i=0; i<n; i++)
       {
           int mn = arr[i];
           for(int j=i; j<n; j++)
           {
              mn = min(mn, arr[j]);
              if(mn == k)
                cnt++;
           }
       }
       printf("%d\n",cnt);
   }
   return 0;
}
