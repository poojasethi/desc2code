#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	int n;
	cin>>n;
	long int ar[n];
	for(int i=0;i<n;i++)
	{
		cin>>ar[i];
	}
	
	int t;
	cin>>t;
	while(t--)
	{
		int q,c[n],j=0,h=0,length=0;
		cin>>q;
		int s=0;
		for(int i=0,j=0;i<n;i++)
		{
			if(ar[i]==q) {c[j]=i+1;j++;h++;}
		}
	
		
		
			for(j=0;j<h;j++)
			{
					
					int p=0,l=0;
					for(int i=(c[j]-1);i<(n-1);i++) 
					{
				
						if(ar[i+1]>=ar[c[j]-1]) {++l;}
						else {break;}
					}
				
					for(int i=(c[j]-1);i>0;i--)
					{
						if(ar[c[j]-1]<ar[i-1]) 
						{
							++p;
						}
						else {break;}
					}
					length=l+(p*l)+1+p;
					s+=length;
				
			
			}
			
			cout<<s<<endl;
		
		
	}
	return 0;
}
