
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int

int ans[1000001]={0};

int main()
{
    int n;
    si(n);
    int a[n];
    for(int i=0;i<n;++i) {
	si(a[i]);
    }
    for(int i=0;i<n;++i) {
	int mn=a[i];
	for(int j=i;j<n;++j) {
	    mn=min(mn,a[j]);
	    ans[mn]++;
	}
    }
    int q; si(q);
    while(q--) {
	int k;
	si(k);
	pri(ans[k]); nl;
    }
    return 0;
}
