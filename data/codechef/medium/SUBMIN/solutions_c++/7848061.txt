#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef long long int ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll n,k,q,i,j,mini;
    n=read_int();
    vector<ll> a(n,0);
    vector<ll> b(1000005,0);
    for(i=0;i<n;i++)
    {
        a[i]=read_int();
    }
    for(i=0;i<n;i++)
    {
        mini=a[i];
        for(j=i;j<n;j++)
        {
            mini=min(mini,a[j]);
            b[mini]++;
        }
    }
    q=read_int();
    while(q--)
    {
        k=read_int();
        printf("%lld\n",b[k]);
    }
    return 0;
}
