using namespace std;
    void inp( long &n )
    {
    n=0;
    int ch=getcx();int sign=1;
        while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

         while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
    n=n*sign;
    }

long a[50],b[50][50],ctr[1000005]={0};
int main()
{
    long n,i,j,k,q;
    inp(n);
    for(i=0;i<n;i++)
        inp(a[i]);
    for(i=0;i<n;i++)
    {
        b[i][i]= a[i];
        ctr[b[i][i]]++;
        for(j=i+1;j<n;j++)
        {
            b[i][j]= (b[i][j-1] < a[j] ? b[i][j-1] : a[j]);
            ctr[b[i][j]]++;
        }
    }
    inp(q);
    while(q--)
    {
        inp(k);
        printf("%ld\n",ctr[k]);
    }
    return 0;
}
