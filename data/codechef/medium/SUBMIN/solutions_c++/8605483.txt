#include <bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,n,k,mn,q,num,ans;

	scanf("%d", &n);

	int arr[n];

	for(i=0;i<n;i++) {
		scanf("%d", &arr[i]);
	}

	scanf("%d", &q);

	while(q--) {
		scanf("%d", &num);

		ans=0;
		for(i=0;i<n;i++) {
			for(j=i;j<n;j++) {
				mn = INT_MAX;
				for(k=i;k<=j;k++) {
					mn = min(mn,arr[k]);
				}
				if(mn==num) ans++;
			}
		}
		printf("%d\n", ans);
	}

	return 0;
}
