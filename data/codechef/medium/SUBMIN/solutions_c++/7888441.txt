/*
https://www.codechef.com/problems/SUBMIN
TC - O(n(n+q))
*/
#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	int a[n],b[n],c[n],i,j,q,k;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	b[0]=0;
	for(i=0;i<n;i++)
	{
		j=i;
		while(j>=0 && a[j]>=a[i])
			j--;
		b[i]=i-j;
		j=i+1;
		c[i]=1;
		while(j<n && a[j]>=a[i])
		{
			if(a[j]!=a[i])
				c[i]++;
			j++;
		}
	}
	// for(i=0;i<n;i++)
	// {
	// 	printf("%d %d %d\n", a[i],b[i],c[i]);
	// }
	scanf("%d",&q);
	for(i=0;i<q;i++)
	{
		scanf("%d",&k);
		int ans=0;
		for(j=0;j<n;j++)
		{
			if(a[j]==k)
				ans+=b[j]*c[j];
		}
		printf("%d\n", ans);	
	}
	return 0;
}
