#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
	//	cout<<n<<endl;
		char s[n];
	//	cout<<strlen(s);
		int f[n];
		for(int i=0;i<n;i++)
		f[i]=0;
		int a,b;
		for(int i=0;i<k;i++)
		{
			scanf("%d%d",&a,&b);
			for(int j=a;j<=b;j+=2)
			{
				s[j]='(';
				s[j+1]=')';
				f[j]=f[j+1]=1;
			}
		}
		for(int i=0;i<n;i++)
		{
			if(f[i]==0)
			{
				s[i]='(';
				f[i]=1;
			}
		}
		for(int i=0;i<n;i++)
		printf("%c",s[i]);
		printf("\n");
	}
}
