#include<bits/stdc++.h>
using namespace std;
struct node{
	int x,y,d;
};
bool cmp(node a,node b)
{
	if(a.d!=b.d)
	return a.d<b.d;
	return a.x<b.x;
}
int main()
{
  int tp;
  scanf("%d",&tp);
  while(tp--)
  {
  	int n,k,i,j,l;
  	scanf("%d%d",&n,&k);
  	int x1,y1,z1;
  	node t;
  	vector<node>v;
  	for(i=0;i<k;i++)
  	{
  		scanf("%d%d",&x1,&y1);
  		t.x=x1;t.y=y1,t.d=y1-x1;
  		v.push_back(t);
  	}
  	sort(v.begin(),v.end(),cmp);
  //	for(i=0;i<k;i++)
  //	cout<<v[i].x<<" "<<v[i].y<<" "<<v[i].d<<endl;
  	int r,c;
  	int a[2005];
  	memset(a,-1,sizeof(a));
  	for(i=0;i<k;i++)
  	{
  		l=v[i].x; r=v[i].y; c=0;
  		for(j=l;j<n && j<=r;j++)
  		{if(a[j]==-1)
  		{	a[j]=c;
  	//	cout<<j<<" "<<c<<endl;
  		c^=1;
  		}
  		}
  		
  	}
  	for(i=0;i<n;i++)
  	{
  		if(a[i]==0) printf("(");
  		else if(a[i]==1)printf(")");
  		else printf("(");
  	}
  	printf("\n");
  }
}
