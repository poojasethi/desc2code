#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int x,y;
        char c[n+1];
        for(int i=0;i<=n;i++)
        {
            c[i]='\0';
        }
        while(k--)
        {
            cin>>x;
            cin>>y;
            for(int i=x;i<y;i+=2)
            {
                c[i]='(';
                c[i+1]=')';
            }
        }
        for(int i=0;i<n;i++)
        {
            if(c[i]=='\0')
            {
                c[i]='(';
            }
        }
        for(int i=0;i<n;i++)
        {
            cout<<c[i];
        }
        cout<<endl;
    }
    return(0);


}
