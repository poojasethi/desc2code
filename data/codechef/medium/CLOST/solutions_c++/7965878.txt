#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
struct stu
{
	int arr[2000];
	int brr[2000];
	int crr[2000];
}s;
int main()
{
	int t,n,k,i,j,flag=0;
	char str[2000];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		for(i=0;i<n;i++)
		str[i]='m';
		for(i=0;i<k;i++)
		{
			scanf("%d%d",&s.arr[i],&s.brr[i]);
			str[s.arr[i]]='(';
			str[s.brr[i]]=')';
			s.crr[i]=s.brr[i]-s.arr[i];
		}
		sort(s.crr,s.crr+k);
		for(i=0;i<k;i++)
		{
			j=s.arr[i];
			flag=0;
			while(1)
			{
				if(j>s.brr[i])
				break;
				if(flag==0)
				{
				str[j]='(';
				flag=1;
			   }
			   else
			   if(flag==1)
			   {
			   	str[j]=')';
			   	flag=0;
			   }
			j++;	
			}
		}
	  for(i=0;i<n;i++)
	  {
	  	if(str[i]=='m')
	  	str[i]='(';
	  }
      for(i=0;i<n;i++)
      cout<<str[i];
      cout<<"\n";
      flag=0;
      j=0;
	}
	return 0;
}
