#include<bits/stdc++.h>
#define lld long long int
#define inp(x) scanf("%lld",&x)
const int len=2010;
 
int main()
{
    lld t,left,right,p,K,n,j,i;
    inp(t);
    char s[len];
    while(t--)
    {
        inp(n);
        inp(K);
        memset(s,' ',sizeof(s));
        for(i=0;i<K;i++)
        {
            inp(left);
            inp(right);
            for(p=left+1,j=left;p<=right;j+=2,p+=2)
            {
                s[j]='(';
                s[p]=')';
            }
 
        }
        for(i=0;i<n;i++)
        {
            if(s[i]==' ')
            {
                s[i]=')';
            }
        }
 
        for(int i=0;i<n;i++) printf("%c",s[i]);
        printf("\n");
 
    }
    return 0;
}
