    #include <bits/stdc++.h>
    using namespace std;
    int main()
    {
      int t,i,n,m,c,start,end;
      cin>>t;
      while(t--)
      {
        int a[3000]={0};
        cin>>n>>m;
        for(i=0;i<n;i++)
        {
            if(i%2==1)
               a[i]=1;
        }
        while(m--)
        { 
            cin>>start>>end;
            c=0;
            for(i=start;i<=end;i++)
            {
                if(c%2==0)
                  a[i]=0;
                else 
                  a[i]=1;
                c++;
            }
        }
        for(i=0;i<n;i++)
        {
            if(a[i]==0)
                cout<<'(';
            else
                cout<<')';
        }
        cout<<endl;
      }
      return 0;
    }
