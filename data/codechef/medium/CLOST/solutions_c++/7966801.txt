#include<stdio.h>
#include<string.h>
 
char A[2001];
int main () {
    int t;
    scanf("%d",&t);
	while(t--) {
		int n, k, cont, i, j, a, b;
		scanf("%d %d",&n,&k);
		cont=0;
		for(i=0;i<n;i++) {
            if(cont==0) {
                A[i]='(';
                cont=1;
            }
            else if (cont==1) {
                A[i]=')';
                cont=0;
            }
        }
		while(k--) {
			cont=0;
			scanf("%d %d",&a,&b);
			while(a<=b) {
                if(cont==0) {
                    A[a]='(';
                    cont=1;
                    a++;
                }
                else if (cont==1) {
                    A[a]=')';
                    a++;
                    cont=0;
                }
            }
		}
		A[n] = 0;
		printf("%s\n",A);
	}
    return 0;}
