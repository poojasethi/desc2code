#include<iostream>
using namespace std;
int main()
{
    int t,i,j;
     int min,max;
    char arr[3000];
    for(i=0;i<3000;i++)
        arr[i]='(';
    int n,k,k1,k2;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cin>>k;

        for(j=0;j<k;j++)
        {
             //cout<<i;
            cin>>k1>>k2;
             min=k1<k2?k1:k2;
             max=k1>k2?k1:k2;

             for(i=min;i<=max-1;i=i+2)
            {
                arr[i]='(';
                arr[i+1]=')';
            }
        }

        for(i=0;i<n;i++)
            cout<<arr[i];
        cout<<endl;
    }
}
