#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)   {
        int n,k;
        cin >> n >> k;
        char a[n];
        for(int i=0;i<n;i++)
            a[i]='(';
        int x,y;
        while(k--)  {
            cin >> x >> y;
          while(x < y)  {
            a[x+1]=')';
            a[x]='(';
            x+=2;
          }
        }
     for(int i=0;i<n;i++)  {
        cout << a[i];
     }
     cout << endl;
    }
}
