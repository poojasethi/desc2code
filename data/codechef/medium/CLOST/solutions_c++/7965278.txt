#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int>PII;
typedef pair<int,PII>PIII;
typedef long long ll;


int main()
{
	int test,l,r;
	scanf("%d",&test);
	while(test--)
	{
		int N, K;
		scanf("%d%d",&N,&K);
		char s[N+1];

		for(int i = 0 ; i < N ; i++)
			s[i] = '(';

		for(int i = 0 ; i < K ; i++)
		{
			scanf("%d%d",&l,&r);
			int j = l;

			for(j ; j <= r ;)
			{
				s[j] = '(';
				s[j+1] = ')';
				j += 2;
			}
		}

		s[N] = '\0';
		printf("%s\n",s);
	}
	return 0;
}
