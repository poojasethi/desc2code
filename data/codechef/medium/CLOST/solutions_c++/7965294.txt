#include<iostream>
#include<string>
using namespace std;



int main()
{
	int T=0;

	int n,k;
	
	bool flag = false;
	cin>>T;

	while(T--)
	{
		flag = false;
		int x=0,y=0;
		cin>>n>>k;

		string strO(n,'(');
		for(int i=0;i<k;i++)
		{
			cin>>x>>y;

			for(int j=x;j<=y;j++)
			{
				if(!flag){
				strO[j] ='(';
				flag = true;
				}
				else if(flag)
				{
					strO[j] = ')';
					flag = false;
				}
			}


		}

	  cout<<strO<<"\n";
	}

	return 0;
}
