#include <iostream>
#include <algorithm>

using namespace std;
int main(){
int t;
cin>>t;
while(t--){
	int n,k,t1,t2;
	cin>>n>>k;
	int a[2010]={0};
	int b[2010]={-1};
	while(k--){
		cin>>t1>>t2;
		a[t1]=1;
		a[t2]=-1;
		b[t1]=t2;
	}
	for(int i=n-1;i>=0;i--){
		if(a[i]==1){
			int t1=1,t2=0;
			for(int j=i+1;j<b[i];j++){
				if(a[j]==1){
					t1++;
				}
				if(a[j]==-1){
					t2++;
				}
				if(a[j]==0){
					if(t1<=t2) {a[j]=1;t1++;}
					else {a[j]=-1;t2++;}
				}
			}
		}
	}
	cout<<"(";
	for(int i=1;i<n;i++) {
		//cout<<a[i];
		if(a[i]==1) cout<<"(";
		else cout<<")";
	}
	cout<<endl;}
}
