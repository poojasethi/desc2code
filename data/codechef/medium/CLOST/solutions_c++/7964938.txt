#include<stdio.h>
using namespace std;
int main()
{
    int i,t,n,k,mid,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        scanf("%d",&k);
        char str[n+1];
        int l,h;
        for(i=0;i<n;i++)
        {
            if(i%2==0)
                str[i]='(';
            else
                str[i]=')';
        }
        for(i=0;i<k;i++)
        {
            scanf("%d",&l);
            scanf("%d",&h);
            //mid=(l+h)/2;
            for(j=l;j<=h;j++)
            {
                if((j-l)%2==0)
                    str[j]='(';
                else
                    str[j]=')';
            }
        }
        str[n]='\0';
        printf("%s\n",str);
    }
    return 0;
}
