{
	int t;
	sc(t);

	while(t--)
	{
		int n,k;
		sc(n); sc(k);

		char s[n+3];
		for(int i=0;i<n;i++) s[i]='a';
		for(int i=0;i<k;i++)

			int st,end;
			sc(st); sc(end);

			s[st]='(';
			s[end]=')';
			
			for(int j=st;j<=end;j+=2)
			{
				s[j]='(';
				s[j+1]=')';
			}
		}

		for(int i=0;i<n;i++)
