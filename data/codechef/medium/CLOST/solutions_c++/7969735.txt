#include <stdio.h>
#include <string.h>
#include <utility>
#include <algorithm>
using namespace std;
#define MAXN 2005
char A[MAXN], F[MAXN];
int X[50], Y[50];
pair<int, int> H[50];
void solve() {
    int N, K;
    scanf("%d %d", &N, &K);
    int x, y;
    for(int i = 0; i < K; i++) {
        scanf("%d %d",&x ,&y);X[i] = x, Y[i] = y;
        H[i].first = y - x + 1;
        H[i].second = i;
    }
    sort(H, H + K);
    for(int i = 0; i < N; i++) A[i] = '(';
    for(int i = 0; i < N; i++) F[i] = 0;
    for(int i = 0 ;i < K; i++) {
        int k = H[i].second;
        int j;
        int flag = 0;
        for(j = X[k]; j <= Y[k]; j++) {
            if(F[j]) continue;
            if(!flag) A[j] = '(';
            else A[j] = ')';
            flag ^= 1;
            F[j] = 1;
        }
    }
    A[N] = 0;
    printf("%s\n", A);
}
int main() {
    int t;
    scanf("%d", &t);
    while(t--) solve();
}
