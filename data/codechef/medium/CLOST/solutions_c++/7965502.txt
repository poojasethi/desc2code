#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<limits.h>
#include<stack>
using namespace std;

int main()
{
	int n,q,t,st,en,i,j,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&q);
		//scanf("%s\n",ch);
		char ch[n+5];
		for(i=0;i<n;i++)
		{
			ch[i]='(';
		}
		while(q--)	
		{
			scanf("%d %d",&st,&en);
			int count=0;
			for(i=st;i<=en;i++)
			{
				
				if(count%2==0)
				{
					ch[i]='(';
					count++;
				}
				else
				{
					ch[i]=')';
					count++;
				}
				
			}
			
		}
	// printing the final string here..	
		for(i=0;i<n;i++)
		{
			putchar_unlocked(ch[i]);
		}
		printf("\n");
	}
return 0;
}
