#include <iostream>
#include <bits/stdc++.h>

using namespace std;
typedef long long int ll;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		ll ans=0,c[4]={0};
		int a;
		memset(c,0,sizeof(c));
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a);
			if(a<=2)
				c[a]++;
			else 
				c[3]++;
		}
		ans = c[2] * c[3] + (c[3] * (c[3]-1)) / 2;
		printf("%lld\n",ans);
	}
	return 0;
}
