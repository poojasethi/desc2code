#include<iostream>
#include<cstdio>
#include<cstring>
#define CLR(a) memset(a, 0, sizeof(a))
using namespace std;
int main()
{
  int t;
  cin>>t;
  long long int ans,curr[4];
  int n,num;
  for(int k=0;k<t;k++)
  {

      scanf("%d",&n);
      CLR(curr);
      for(int i=0;i<n;i++)
      {
          scanf("%d",&num);
          if(num <= 2)curr[num]++;
            else curr[3]++;
      }
      ans = (curr[3] * (curr[3]-1)) / 2 + curr[2]*curr[3];
      printf("%lld\n",ans);
  }
  return 0;
}
