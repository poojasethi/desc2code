#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    for(scanf("%d", &t); t--; ){
        int n, a;
        scanf("%d", &n);
        long long c2=0, c=0;
        for(int i=0; i<n; i++){
            scanf("%d", &a);
            if(a==2)
                c2++;
            else if(a>2)
                c++;
        }
        long long res = c2*c + (c*(c-1)/2);
        printf("%lld\n", res);
    }
    return 0;
}
