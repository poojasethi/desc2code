#include<bits/stdc++.h>
using namespace std;
void printPairs();

int main() {
	int n, i;

	scanf("%d",&n);

	for (i = 0; i < n; i++) {
		printPairs();
	}

	return 0;
}

void printPairs() {
	int i, n,temp;
	long long pairs, numTwo, num;
	scanf("%d",&n);
	numTwo = num = 0;

	for (i = 0; i < n; ++i) {
		scanf("%d",&temp);

		if (temp>2)
        num++;
        else if (temp == 2)
        numTwo++;


	}

	pairs = ((num*(num-1))>>1) + num*numTwo;

	printf("%lld\n",pairs);
}
