#include <iostream>
#include <stdio.h>

using namespace std;

inline int readInt()
{
    int t=0;
    char c;
    c=getchar();
    while(c<'0' || c>'9')
        c=getchar();
    while(c>='0' && c<='9')
    {
        t=(t<<3)+(t<<1)+c-'0';
        c=getchar();
    }
return(t);
}

int main()
{

  long long i,g,two,x,t,n;
  t=readInt();
  while(t--)
  {
      g=0;two=0;
      n=readInt();
      for(i=0;i<n;i++)
      {
          x=readInt();
          if(x==2)
          { g++;two++; }
          else if(x>2)
            g++;
      }
      x=((g*(g-1))/2)-((two*(two-1))/2);
      printf("%lld\n",x);
  }
}
