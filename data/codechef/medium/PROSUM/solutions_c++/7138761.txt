#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
    long long i,t,n,cal;
    long long res,countm;
    scanf("%lld",&t);
    while(t--)
    {
        countm=cal=0;
        scanf("%lld",&n);
        res=(n*(n-1))/2;
        vector<long long> v(n);
        vector<long long> store;
        for(i=0;i<n;i++)
        {
            scanf("%lld",&(v[i]));
        }
        sort(v.begin(),v.end());
        for(i=0;i<n;i++)
        {
            if((v[i]==0)||(v[i]==1))
            {
                cal+=n-(i+1);
            }
            else if(v[i]==2)
            {
                countm++;
            }
        }
        countm=((countm-1)*countm)/2;
        res=res-cal-countm;
        printf("%lld\n",res);
    }
    return 0;
}
