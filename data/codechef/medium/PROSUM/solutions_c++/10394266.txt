#include <stdio.h>
#include <stdlib.h>

#define getcx getchar

inline int inp ( ) { //fast input function
	int n=0;
	int ch=getcx();//int sign=1;

	while( (ch < '0') || (ch > '9') ) {
		ch=getcx();
	}

	while( (ch >= '0') && (ch <= '9') ) {
		n = (n<<3) + (ch - '0')+(n<<1);
		ch = getcx();
	}
	//n=n*sign;
	return n;
}
void printPairs();

int main() {
	int n, i;

	n = inp();

	for (i = 0; i < n; i++) {
		printPairs();
	}

	return 0;
}

void printPairs() {
	int i, n;
	long long pairs, numTwo, num, temp;
	n = inp();
	numTwo = num = 0;

	for (i = 0; i < n; ++i) {
		temp = inp();

		if (temp>2)
        num++;
        else if (temp == 2)
        numTwo++;


	}

	pairs = ((num*(num-1))>>1) + num*numTwo;

	printf("%lld\n", pairs);
}
