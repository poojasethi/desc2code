#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,x,t;
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
		scanf("%d",&n);
		int k=0,t=0;
		for(int j=0;j<n;j++)
		{
			scanf("%d",&x);
			if(x>2)
			t++;
			if(x==2)
			k++;
		}
		long long ans=1LL*t*(t-1)/2+1LL*t*k;
		printf("%lld\n",ans);
	}
}
