#include <cstdio>

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long n,lt = 0,t = 0,o = 0;
        scanf("%lld",&n);
        int a[n];
        for(int i = 0; i<n; ++i)
        {
            scanf("%d",a+i);
            if(a[i] < 2)
                ++lt;
            if(a[i] == 2)
                ++t;
            if(a[i] > 1)
                ++o;
        }
        printf("%lld\n",(((n-1)*(n))/2) - (((t-1)*(t))/2) - ((lt*(lt-1))/2) - (lt)*o);
    }
    return 0;
}
