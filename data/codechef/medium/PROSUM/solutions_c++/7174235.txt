#include<stdio.h>
#include<iostream>
using namespace std;
int main()
 {
    long long t,n,i,count,two,tmp;
    scanf("%lld", &t);

    while(t--)
     {
        scanf("%lld", &n);

        i=0;
        count=two=0;
        while(i++<n)
         {
            scanf("%lld",&tmp);

            if(tmp>1)
              count++;
            if (tmp==2)
              two++;
         }
       printf("%lld\n",(count*(count-1))/2 - (two*(two-1))/2);
     }
 }
