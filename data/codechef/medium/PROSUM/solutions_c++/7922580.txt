#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;

int main() {
	int T;
	long long int N;
	int i;
	scanf("%d", &T);
	while(T--) {
		scanf("%llu", &N);
		long long int A, count1, count2;
		
		count1 = count2 = 0;
		for(i = 0; i < N; ++i) {
			scanf("%llu", &A);
			if(A == 2)
				++count2;
			else if(A > 2)
				++count1;	
		}	
		
		count1 += count2;
		printf("%llu\n", ((count1*(count1-1))/2) - ((count2*(count2-1))/2) );
	}
}
