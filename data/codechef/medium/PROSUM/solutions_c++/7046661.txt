#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
long long int n,count1=0,count2=0,count3=0;
scanf("%lld",&n);
long long int a[n+1];
for(long long int i=0;i<n;i++)
scanf("%lld",&a[i]);
long long int ans;
for(long long int i=0;i<n;i++)
{
if(a[i]<2)
count1++;
else 
{
if(a[i]==2)
count2++;
count3++;
}
}
//cout<<count1<<" "<<count2<<" "<<count3<<endl;
ans=((count3*(count3-1))/2)-((count2*(count2-1))/2);
printf("%lld\n",ans);
}
return 0;
}
