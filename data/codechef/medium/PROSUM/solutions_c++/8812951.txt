#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int main() {
    long long t,n,i,x,count,s,l;
    scanf("%lld",&t);
    while(t--){
          scanf("%lld",&n);
          s=l=0;
          for(i=0;i<n;++i){
                scanf("%lld",&x);
                if(x!=0&&x!=1){
                    s++;
                    if(x==2) l++;
                }
          }
          s=(s*(s-1))/2;
          s-=((l-1)*l)/2;
          cout<<s<<"\n";
    }
    return 0;
}
