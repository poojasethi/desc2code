#include<stdio.h>
using namespace std;
int main()
{
    int t;
    long long int n,ans,temp,ctr,ctr_two,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld",&n);
        ans=n*(n-1)/2;
        ctr=ctr_two=0;
        for(i=1;i<=n;++i)
        {
            scanf("%lld",&temp);
            if(temp==0 || temp==1)
            {
                ++ctr;
                ans-=n-ctr;
            }
            if(temp==2)
                ++ctr_two;
        }
        ans=ans-(ctr_two*(ctr_two-1)/2);
        printf("%lld\n",ans);
    }
    return 0;
}
