using namespace std;

#define ll long long

int arr[100001];

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin >> t; 
	while (t--)
	{
		int n;
		cin >> n;
		int twos = 0, other = 0;
		for (int i = 0; i < n; i++)
		{
			cin >> arr[i];
			if (arr[i] == 2)
				twos++;
			else if (arr[i] > 2)
				other++;
		}
		ll ans = 0;
		for (int i = 0; i < n; i++)
			if (arr[i] == 2)
			{
				twos--;
				ans += other;
			}
			else if (arr[i] > 2)
			{
				other--;
				ans += twos + other;
			}
		cout << ans << '\n';
	}
	return 0;
}
