#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
	long long int n, t, i;
	long long int arr, count, count2;
	scanf("%lld",&t);
	while(t--){
		scanf("%lld",&n);
		count = 0;
		count2 = 0;
		for(i=0;i<n;i++){
			scanf("%lld",&arr);
			if((arr==0 || arr==1))
				count++;
			if(arr==2)
				count2++;
		}
		n -= count;
		count = (n*(n-1))/2;
		count -= (count2*(count2-1))/2;
		printf("%lld\n",count);
	}
	return 0;
}
