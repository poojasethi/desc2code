#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
int fastread()
{
 
    long long int t=0;
    char ch=getchar_unlocked();
    while(ch<33)
        ch=getchar_unlocked();
    while(ch>33)
    {
        t=t*10+ch-'0';
        ch=getchar_unlocked();
    }
    return t;
}
int main()
{
    int t;
    t=fastread();
    while(t--)
    {
    	long long int i,n,m,sum=0,two=0;
    	n=fastread();
    	for(i=0;i<n;i++)
    	{
    		m=fastread();
    		if(m!=1 && m!=0)
    		sum++;
    		if(m==2)
    		two++;
		}
		sum--;
		two--;
		sum=sum*(sum+1)/2;
		two=two*(two+1)/2;
		printf("%lld\n",sum-two);
    }
    return 0;
}
