#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	long long int i,n,m,sum=0,two=0;
    	scanf("%lld",&n);
    	for(i=0;i<n;i++)
    	{
    		scanf("%lld",&m);
    		if(m!=1 && m!=0)
    		sum++;
    		if(m==2)
    		two++;
		}
		sum--;
		two--;
		sum=sum*(sum+1)/2;
		two=two*(two+1)/2;
		printf("%lld\n",sum-two);
    }
    return 0;
}
