#include<cstdio>
#include<cstring>
using namespace std;
typedef long long int ll;
int main(){
    int t,k;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        ll tmp,c2=0,c3=0;
        for(int i=0;i<n;i++){
            scanf("%lld",&tmp);
            if(tmp==2){
                c2++;
            }
            else if(tmp>2){
                c3++;
            }
        }
        ll ans=(c2*c3) + (c3*(c3-1))/2;
        printf("%lld\n",ans);
        
    }
}
