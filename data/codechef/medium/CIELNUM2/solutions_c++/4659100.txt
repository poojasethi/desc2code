#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
int main()
{
    int count = 0;
    char str[256];
    //cout << "Reached";
    int cases;
    cin >> cases;
    //cout << cases;
    for(int k = 0 ; k <= cases; k ++)
    {
 
 
        gets(str);
        //cout << "Reached";
        int count_8 = 0;
        int count_5 = 0;
        int count_3 = 0;
        for (int i = strlen(str)-1 ;i >= 0 ; i --)
        {
            //cout << "Reached";
            if(str[i] == ' ')
            {
                if(count_8 >= count_5 && count_5 >=count_3)
                    count ++;
                break;
            }
            if(str[i] == '0'||str[i] == '1'||str[i] == '2'||str[i] == '4'||str[i] == '6'||str[i] == '7'||str[i] == '9')
                break;
            if(str[i] == '8')
                count_8++;
            if(str[i] == '5')
                count_5++;
            if(str[i] == '3')
                count_3++;
        }
 
    }
    cout << count;
}
