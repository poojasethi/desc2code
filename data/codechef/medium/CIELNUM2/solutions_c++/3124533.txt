#include <iostream>
#include <vector>
#include <sstream>
using namespace std;

bool checkceil(int n){
	vector<int> freq(3,0);
	while(n>0){
		int temp = n%10;
		if(temp!=8 && temp!=5 && temp!=3)
			return false;
		if(temp==8)
			++freq[0];
		else if(temp==5)
			++freq[1];
		else if(temp==3)
			++freq[2];
		n/=10;
	}
	if(freq[0]>=freq[1] && freq[1]>=freq[2])
		return true;
	return false;
}

int main(){
	int t;
	cin>>t;
	char c;
	cin>>c;
	int count=0;
	while(t--){
		string test;
		int n;
		getline(cin,test);
		int l = test.length();
		string num;
		for(int i=l-1;i>=0;i--){
			if(test[i]==' ')
				break;
			num = test[i]+num;
		}
		stringstream temp;
		temp<<num;
		temp>>n;
		if(checkceil(n))
			++count;
	}
	cout<<count<<"\n";
	return 0;
}
