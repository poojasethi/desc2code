#include<stdio.h>
#include<string.h>

int getprice(char s[111]);

int cielnum(int);

int main()
{	
	int n,i,price,cnt=0;
	char menu[111];
	gets(menu);
	n=getprice(menu);
	
	while(n--)
	{	
		gets(menu);
		price=getprice(menu);
		if(cielnum(price))
			cnt++;
	}
	printf("%d",cnt);
	return 0;
}

int getprice(char s[111])
{	
	int p=0,i,m=1,len;
	len=strlen(s)-1;
	while(s[len]<48 || s[len]>57)
		len--; 
	for(i=len;s[i]!=' '&&i>=0;i--)
	{	
		p+=m*(s[i]-'0');
		m*=10;
	}
	return p;
}

int cielnum(int p)
{	
	int eight=0,five=0,three=0,m;
	
	while(p)
	{	
		m=p%10;
		if(m==8)
			eight++;
		else if(m==5)
			five++;
		else if(m==3)
			three++;
		else
			return 0;
		p/=10;
	}
	if(three<=five && five<=eight)
		return 1;
	return 0;
}
