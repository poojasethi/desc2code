#include <stdio.h>
#include <string.h>
int main()
{
	int N,count3=0,count5=0,count8=0,ans=0,l;
	char A[110],ch;
	scanf("%d",&N);
		gets(A);
	while(N--)
	{
		count3=0;count5=0;count8=0;
		gets(A);
		l=strlen(A);
		while(A[l]<48||A[l]>57)
		--l;
		while(A[l]!=' ')
		{
			if(A[l]=='8')
				count8++;
			else if(A[l]=='5')
				count5++;
			else if(A[l]=='3')
				count3++;
			else break;
			l--;
			 
		}
		if(A[l]==' '&&count8>=count5&&count5>=count3)
			ans++;
	}
	printf("%d\n",ans);
	return 0;
}
