#include<bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
     int n,k=0,l,b,i,n3,n5,n8;
    char a[110];
    cin>>n;
    gets(a);

    while(n--)
    {

             gets(a);
             l=strlen(a);
             b=1;
             n3=n5=n8=0;


             for(i=l-1;b&&i>=0&&a[i]!=' ';i--)
             {


                             if(a[i]=='3')n3++;
                             else if(a[i]=='5')n5++;
                             else if(a[i]=='8')n8++;
                             else if(a[i]>='0'&&a[i]<='9') b=0;
             }
             if(b && n8>=n5 && n5>=n3)k++;

   }



   cout<<k<<endl;
    //getchar();
    return 0;

}
