#include <iostream>
#include<string.h>
using namespace std;

int main() {
	int n,i,di[10],j,l,d,q,f,r=0,k;
	char s[150];
	cin>>n;
	
	gets(s);
	
	
	q=0;
	for(i=0;i<n;i++)
	{
		gets(s);
		l=1;
		d=0;
		f=0;
		r=0;
		for(k=0;k<10;k++)
			di[k]=0;
		for(j=strlen(s)-1;j>=0;j--)
		{
			if(s[j]>='0' && s[j]<='9')
			{
				di[s[j]-'0']++;
				f=1;
			}
			else if(f==1)
			{
				break;
			}
		}
		for(k=0;k<10;k++)
		{
			if(k!=8 && k!=3 && k!=5)
			{
				if(di[k]!=0)
				{
					r=1;
					break;
				}
			}
			
		}
		if(r==0)
		{	
			if(di[8]>=di[5] && di[5]>=di[3])
				q++;
		}		
		
	}
	cout<<q<<endl;
	return 0;
}
