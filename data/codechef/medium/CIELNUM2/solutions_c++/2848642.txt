#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <cstdlib>
using namespace std;

bool is_ciel(int N)
{
    int Arr[11];
    int dig;

    memset(Arr,0,sizeof(Arr));

    while(N)
    {
        dig=N%10;
        if(dig!=8 && dig!=5 && dig!=3){return false;}
        Arr[dig]++;

        N=N/10;
    }

    if(Arr[3]<=Arr[5] && Arr[5]<=Arr[8])
    {
        return true;
    }
    else
    {
        return false;
    }
}

int main()
{
    int N;
    char buf[1000];

    int res=0;

    gets(buf);
    N=atoi(buf);

    while(N--)
    {
        gets(buf);

        int len=strlen(buf);

        int i=0;
        for(i=len-1;;i--){if(buf[i]==' '){break;}}

        int price=atoi(buf+i+1);

        if(is_ciel(price)){res++;}
    }

    printf("%d\n",res);
    return 0;
}
