#include <stdio.h>
#include <string.h>

int N,res=0;
char S[101];
void checkNum()
{
	int count[10]={0};
	gets(S);
	int length=strlen(S);
	while (S[length-1]<'0'||S[length-1]>'9')
	{
		length--;
	}
	for (int i = length - 1; i >= 0 ; i--)
	{
		char c=S[i];
		if(c!=' ')
		{
			if(c=='8'||c=='5'||c=='3') count[c-'0']++;
			else return;
		}
		else break;
	}
	if(count[8]>=count[5]&&count[5]>=count[3]) res++;
}
int main()
{
	scanf("%d\n",&N);
	while (N--)
	{
		checkNum();
	}
	printf("%d\n",res);
}
