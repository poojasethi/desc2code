#include <iostream>
#include <cstdio>
#include <string.h>
#define MAX 1000
#define FOR(i,m,n) for(i=m;i<=n;i++)

using namespace std;

bool check_ciel(char *s)
{
    int i,d[3];d[0]=d[1]=d[2]=0;i=strlen(s)-1;

    while(s[i]==' ')
            i--;

    while(s[i]!=' ')
    {
            if(s[i]=='3') d[0]++;
            else if(s[i]=='5') d[1]++;
            else if(s[i]=='8') d[2]++;
            else if(s[i]>='0' && s[i]<='9') return false;
            i--;
    }

    return d[0]<=d[1] && d[1]<=d[2];
}

int main()
{
    int i,n,counter=0;char s[MAX+10];

     scanf("%d",&n);
     gets(s);

    FOR(i,1,n)
    {
        gets(s);
        if(check_ciel(s)) counter++;
    }

    printf("%d\n",counter);

    return 0;
}
