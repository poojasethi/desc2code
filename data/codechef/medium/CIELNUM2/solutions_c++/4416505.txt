int isciel(int n){
int c1=0,c2=0,c3=0;
while(n){
    if(n%10==8) c1++;
    else if(n%10==5) c2++;
    else if(n%10==3) c3++;
    else return 0;
    n=n/10;
}
if(c1>=c2&&c2>=c3) return 1;
else return 0;
}
