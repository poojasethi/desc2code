int main()
{
	int T;
	scanf("%d\n",&T);

	int count = 0;

	while(T--)
	{
		char s[120];
		//cin >> s;
		//cin.ignore();
		gets(s);
		int l = strlen(s);

	//	cout << s <<"\n";

		int t=0,f=0,e=0;

		bool is_ciel =true;

		l--;
		while( s[l] < 48 || s[l] > 57)
			l--;

		while(l)
		{
			//cout << s[l] <<" ";

			if (s[l] == ' ')
				break;
			if( s[l] == '3')
				t++;
			else if( s[l] == '5')
				f++;
			else if ( s[l] == '8')
				e++;
			else
				is_ciel = false;

			l--;
		}

		//cout << is_ciel << " " << e << " " << f << " " << t << "\n";
		if ( is_ciel )
		{
			if (e >= f && f >= t )
			{
				count++;
	//			cout << "Y\n";
			}
		}


	}

	cout << count << "\n";
	return 0;
}
