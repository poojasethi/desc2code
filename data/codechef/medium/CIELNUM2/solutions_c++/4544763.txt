#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    int n, count = 0;
    scanf("%d", &n);
    char c = getchar_unlocked();
    while(c != '\n')
    {
    	c = getchar_unlocked();
    }
    while(n--)
    {
    	char str[200];
    	int index, num[10] = {0};
    	for(int i = 0; i < 200; i++)
    	{
    		str[i] = getchar_unlocked();
    		if(str[i] == '\n' || str[i] == EOF)
    		{
    			index = i-1;
    			break;
    		}
    	}
    	while(true)
    	{
    		if(!(str[index] >= '0' && str[index] <= '9'))
    		{
    			index--;
    		}
    		else
    		{
    			break;
    		}
    	}
    	bool test = true;
    	while(str[index] != ' ')
    	{
    		char asdf = str[index];
    		if(asdf == '3' || asdf == '5' || asdf == '8')
    		{
    			num[asdf-'0']++;
    		}
    		else
    		{
    			test = false;
    			break;
    		}
    		index--;
    	}
    	if(test && (num[8] >= num[5]) && (num[5] >= num[3]))
    	{
    		count++;
    	}
    }
    printf("%d\n", count);
    return 0;
}
