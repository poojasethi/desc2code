#include<iostream>
#include<cstdio>
#include<cstring>
#include<ctype.h>
#include<vector>
using namespace std;
int main()
{
int n;
scanf("%d \n",&n);
int ans=0;
char b[200];
while(n--)
{
int i,j=0;
vector<int>a(10,0);
gets(b);
i=strlen(b)-1;
while(i>=0&&!(isdigit(b[i])))
//cout<<b[i];
i--;
 
for(;b[i]!=' '&&i>=0;i--)
{
j++;
a[(int)(b[i]-'0')]++;
}
if(((a[3]+a[8]+a[5])==j)&&(a[8]>=a[5]&&a[5]>=a[3]))
ans++;
}
cout<<ans;
}
