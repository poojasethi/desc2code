#include <iostream>
#include <string>
using namespace std;
main()
{
	int i,j,k,l,n;
	cin>>n;
	string s;
	int a[10];
	getline(cin,s);
	l=0;
	while(n--)
	{
		for(i=0;i<10;i++)
			a[i]=0;
		getline(cin,s);
		for(j=s.length()-1;j>=0;j--)
		{
			if(s[j]==' ') break;
			a[s[j]-48]++;
		}
		k=0;
		for(i=0;i<10;i++)
		{
			if(i!=8 && i!=5 && i!=3)
			{
				if(a[i]>0) k=1;
			}
		}
		if(k!=1 && a[8]>=a[5] && a[5]>=a[3])
			l++;
	}
	cout<<l<<endl;
}
