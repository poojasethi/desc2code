#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
using namespace std;
bool check_ceil(int P)
{
    int t=0,f=0,e=0;
    while(P!=0)
    {
        if(P%10==3) t++;
        else if(P%10==5) f++;
        else if(P%10==8) e++;
        else return false;
        P/=10;
    }
    if(e>=f and f>=t) return true;
    else return false;

}

int main()
{
    char A[1000];
    int N;
    gets(A);
    N=atoi(A);
    int i;
    int count=0;
    while(N--)
    {
        gets(A);
        //cout<<A<<endl;
        int len=strlen(A);
        for( i=len-1;i>=0;i--) if(A[i]==' ') break;
        int P=atoi(A+i+1);
        if(check_ceil(P)) count++;
    }
    cout<<count<<endl;
}
