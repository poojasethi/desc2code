#include<stdio.h>
#include<string.h>
int main(){
	int n, f,c8, c5, c3,count,p,i;
	char s[120];
	scanf("%d", &n);
	count = 0;
	gets(s);
	while(n--){
	
		gets(s);
		p = strlen(s);
		i = p-1;
		c8=0;
		c5=0;
		c3=0;
		f=0;
		while(1){
			if(s[i]=='8')
			c8++;
			else if(s[i]=='5')
			c5++;
			else if(s[i]=='3')
			c3++;
			else if(s[i]=='2'||s[i]=='1'||s[i]=='4'||s[i]=='6'||s[i]=='7'||s[i]=='9'||s[i]=='0'){
				f=1;
				break;
			}
			else if(s[i]==' ')
			break;
			i--;
		}
		if(f==0){
			if(c8>=c5 && c5 >= c3)
			count++;
		}	
	}
	printf("%d\n", count);
	return 0;
}
