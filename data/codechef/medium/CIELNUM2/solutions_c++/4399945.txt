#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;

int is_ciel(int n)
{
	int n3=0,n5=0,n8=0;
	while(n)
	{
		int k=n%10;
		if(k==3) n3++;
		else if(k==5) n5++;
		else if(k==8) n8++;
		else return 0;
		n/=10;
	}
	if(n3<=n5 && n5<=n8) return 1;
	return 0;
}


int main()
{   std:ios_base::sync_with_stdio(false);
    int i,len,n,price,res=0;
    char buf[1000];
    gets(buf);
    n=atoi(buf);
    while(n--)
    {
    	gets(buf);
    	len=strlen(buf);
    	for(i=len-1;;i--)
    	if(buf[i]==' ') break;
    	price= atoi(buf+i+1);
    	if(is_ciel(price))res++;
    }
    printf("%d\n",res);
	return 0;
}
