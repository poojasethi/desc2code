#include<stdio.h>
#include<string.h>

char s[1000];
 
int main()
{
	int n, ans = 0;
	scanf( "%d", &n );
	gets(s);
	while( n -- )
	{
		int c3 = 0, c5 = 0, c8 = 0, bad = 0;
		gets(s);
		int i = strlen(s) -1;
		while( i > 0 && s[i] != ' ' )
		{
			if( s[i] == '8' )
				c8 ++;
			else if( s[i] == '5' )
				c5 ++;
			else if( s[i] == '3' )
				c3 ++;
			else if( '0' <= s[i] && s[i] <= '9' )
				bad = 1;
			i --;
		}
		if( !bad && c8 >= c5 && c5 >= c3 )
			ans ++;
	}
	printf( "%d\n", ans );
	return 0;
}
