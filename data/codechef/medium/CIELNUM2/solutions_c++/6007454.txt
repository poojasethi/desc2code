bool checkLuckyNumber(int number){
    bool flag=true,flag1=false;
    int eCount=0,fCount=0,tCount=0,rem;
    while(number>0){
        rem = number%10;
        number = number/10;
        if(rem==8)
            eCount++;
        else if(rem==5)
            fCount++;
        else if(rem==3)
            tCount++;
        else{
            flag1 = true;
            flag = false;
            break;
        }
    }
    if(flag1)
        return flag;
    else if(eCount >= fCount && fCount >=tCount)
        return true;
    else
        return false;
}
