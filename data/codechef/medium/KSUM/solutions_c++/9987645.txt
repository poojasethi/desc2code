#include <bits/stdc++.h>
using namespace std;

#define pb push_back
#define mp make_pair
#define pii pair<int,int>

typedef long long ll;
typedef pair<ll,ll>pll;
const int N = 100005;

ll inp[N];

int main(){
    int n,k; cin >> n >> k;
    ll cnt = 0;
    for(int i=1;i<=n;i++){
        cin >> inp[i];
        cnt += inp[i];
    }

    set < pair < ll , pii> > S;
    S.insert(mp(cnt,mp(1,n)));

    while(k--){
        pair< ll , pii > u = *S.rbegin();
        cout << u.first << " ";
        S.erase(u);
        int l = u.second.first, r = u.second.second;
        if(l != r){
            S.insert(mp(u.first - inp[l],mp(l+1,r)));
            S.insert(mp(u.first - inp[r],mp(l,r-1)));
        }
    }
    puts("");

    return 0;
}
