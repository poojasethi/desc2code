#include <cstdio>
#include <cstdlib>
#include <iostream>


using namespace std;

 
int main() {
	int N;
	scanf("%d\n", &N);
	char A[N][N];
 
    int i,j, k, l;

 for(int i = 0; i < N; i++){
        cin.getline(A[i],1002);
		
	}	

	int first = 1;
    for (i=0; i<N; i++) {
        for (j=0; j<N; j++) {
            if (A[i][j] == '.') {
				if(!first)
					putchar(' ');
				else
					first = 0;
				putchar('0');
			} else {
                int h = 1;
                for (k=j+1; k<N; k++)
                    if (A[i][k] == 'X')
                        h++;
                    else
                        break;
 
                for (k=j-1; k>-1; k--)
                    if (A[i][k] == 'X')
                        h++;
                    else
                        break;
 
 
                int v = 1;
                for (k=i+1; k<N; k++)
                    if (A[k][j] == 'X')
                        v++;
                    else
                        break;
 
                for (k=i-1; k>-1; k--)
                    if (A[k][j] == 'X')
                        v++;
                    else
                        break;
 
                int d1 = 1;
                for (k=i+1, l=j+1; k<N && l<N; k++, l++)
                    if (A[k][l] == 'X')
                        d1++;
                    else
                        break;
 
                for (k=i-1, l=j-1; k>-1 && l>-1; k--, l--)
                    if (A[k][l] == 'X')
                        d1++;
                    else
                        break;
 
                int d2 = 1;
                for (k=i+1, l=j-1; k<N && l>-1; k++, l--)
                    if (A[k][l] == 'X')
                        d2++;
                    else
                        break;
 
                for (k=i-1, l=j+1; k>-1 && l<N; k--, l++)
                    if (A[k][l] == 'X')
                        d2++;
                    else
                        break;
 
                if(h>v && h>d1 && h>d2)
                    h = h;
                else if (v>d1 && v>d2)
                    h = v;
                else if (d1 > d2)
                    h = d1;
                else
                    h = d2;
				
				if(!first)
					putchar(' ');
				else
					first = 0;
				printf("%d", h);
            }
        }
		putchar('\n');
		first = 1;
    }
	
    return (EXIT_SUCCESS);
}
