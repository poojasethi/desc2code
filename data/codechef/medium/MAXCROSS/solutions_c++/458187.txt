#include<iostream>

using namespace std;
int main()
{
    int n,i,j,k,y,countd2,countv,countd1,counth;
    char a[1001][1001];
    //freopen("input.txt","r",stdin);
    scanf("%d",&n);
    //fflush(stdin);
    gets(a[0]);
	for(i=0;i<n;i++)
	{
        //printf("%d\n",i);
		gets(a[i]);
	}
    /*

    {

        scanf("%s",a[i]);
        puts(a[i]);

        //gets(a[i]);

    //cout << endl;
      
    for(i=0;i<n;i++)
    {
       //printf("%s\n",a[i]);
       for(j=0;j<n;j++)
       {
          counth = 0;
          countv = 0;
          countd1 = 0;
          countd2 = 0;
          if(a[i][j] == '.')
          {
            printf("0 ");
            continue;
          }
          k=j+1;
          while(k<n && a[i][k] != '.')
          {
            k++;
            counth++;
          }
          k = j-1;
          while(k>=0 && a[i][k] != '.')
          {
            k--;
            counth++;
          }
          counth++;
          k = i+1;
          while(k<n && a[k][j] != '.')
          {
             k++;
             countv++;
          }
          k = i-1;
          while(k>=0 && a[k][j] != '.')
          {
             k--;
             countv++;
          }
          countv++;
          k = i+1;
          y = j+1;
          while(k<n && y<n && a[k][y] != '.')
          {
              k++;
              y++;
              countd1++;
          }
          k = i-1;
          y = j-1;
          while(k>=0 && y>=0 && a[k][y] != '.')
          {
              k--;
              y--;
              countd1++;
          }
          countd1++;
          k = i-1;
          y = j+1;
          while(k>=0 && y<n && a[k][y] != '.')
          {
              k--;
              y++;
              countd2++;
          }
          k = i+1;
          y = j-1;
          while(k<n && y>=0 && a[k][y] != '.')
          {
              k++;
              y--;
              countd2++;
          }
          countd2++;
          if(countd2 > countd1)
            countd1 = countd2;
          printf("%d ",(counth>countv) ? (counth>countd1?counth:countd1) : (countv>countd1?countv:countd1));
       }
       printf("\n");
    }
    //getch();

}
