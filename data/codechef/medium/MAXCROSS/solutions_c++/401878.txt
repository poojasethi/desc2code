#include<stdio.h>
#define goTop side1=0;tempJ=j; \
while(tempJ>-1 && A[i][tempJ]=='X'){tempJ--;side1++;}

#define goBot side2=0;tempJ=j; \
while(tempJ<n && A[i][tempJ]=='X'){tempJ++;side2++;}

#define goLeft side1=0;tempI=i; \
while(tempJ>-1 && A[tempI][j]=='X'){tempI--;side1++;}

#define goRight side2=0;tempI=i; \
while(tempI<n && A[tempI][j]=='X'){tempI++;side2++;}

#define goDiagonalyUp1 side1=0;tempI=i;tempJ=j; \
while(tempI<n && tempJ>-1 && A[tempI][tempJ]=='X') \
{tempI++;tempJ--;side1++;}

#define goDiagonalyDown1 side2=0;tempI=i;tempJ=j; \
while(tempI>-1 && tempJ<n && A[tempI][tempJ]=='X') \
{tempI--;tempJ++;side2++;}

#define goDiagonalyUp2 side1=0;tempI=i;tempJ=j; \
while(tempI>-1 && tempJ>-1 && A[tempI][tempJ]=='X') \
{tempI--;tempJ--;side1++;}

#define goDiagonalyDown2 side2=0;tempI=i;tempJ=j; \
while(tempI<n && tempJ<n && A[tempI][tempJ]=='X') \
{tempI++;tempJ++;side2++;}

#define computeMax if((side1+side2)>max){max=side1+side2;}

char A[1300][1300];
int main()
{
    int i,j,n,side1,side2,max,tempI,tempJ;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%s",A[i]);
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            max = 0;
            if(A[i][j] == 'X')
            {
                goTop
                goBot
                computeMax
                goLeft
                goRight
                computeMax
                goDiagonalyUp1
                goDiagonalyDown1
                computeMax
                goDiagonalyUp2
                goDiagonalyDown2
                computeMax
                max-=1;
            }
            printf("%d ",max);
        }
        printf("\n");
    }
    return 0;
}
