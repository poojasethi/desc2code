#include<cstdio>
#include<iostream>

using namespace std;

int main(){
int i,n,j,k,hval[1002][1002],vval[1002][1002];
char str[1002][1002],temp;


scanf("%d",&n);

for(i=1;i<=n;i++){
		scanf("%s",&str[i][1]);
	
}

for(i=0;i<n+2;i++)
	for(j=0;j<n+2;j++)
		hval[i][j]=vval[i][j]=0;//Mdval[i][j]=0;//mdval[i][j]=0;

for(i=1;i<(n+1);i++){
	for(j=1;j<(n+1);j++)
		if(str[i][j]=='X')
			hval[i][j]=hval[i][j-1]+1;

	for(j=n-1;j>0;j--)
		if(str[i][j]=='X' && str[i][j+1]=='X')
			hval[i][j]=hval[i][j+1];

}

for(j=1;j<=n;j++){
	for(i=1;i<=n;i++)
		if(str[i][j]=='X')
			vval[i][j]=vval[i-1][j]+1;

	for(i=n-1;i>0;i--)
		if(str[i][j]=='X' && str[i+1][j]=='X')
			vval[i][j]=vval[i+1][j];

}

for(i=0;i<n+2;i++)
	for(j=0;j<n+2;j++){
		hval[i][j]=((hval[i][j] > vval[i][j]) ? hval[i][j]:vval[i][j]);
		vval[i][j]=0;	
	}

for(i=n;i>0;i--){
	for(j=i,k=1;j<=n && k<=n;j++,k++)
		if(str[j][k]=='X')
			vval[j][k]=vval[j-1][k-1]+1;

	for(--j,--k;j>0 && k >0;j--,k--)
		if(str[j][k]=='X' && str[j+1][k+1]=='X')
			vval[j][k]=vval[j+1][k+1];
}

for(i=n;i>1;i--){
	for(j=i,k=1;j<=n && k<=n;j++,k++)
		if(str[k][j]=='X')
			vval[k][j]=vval[k-1][j-1]+1;

	for(--j,--k;j>0 && k >0;j--,k--)
		if(str[k][j]=='X' && str[k+1][j+1]=='X')
			vval[k][j]=vval[k+1][j+1];
}

for(i=0;i<n+2;i++)
	for(j=0;j<n+2;j++){
		hval[i][j]=((hval[i][j] > vval[i][j]) ? hval[i][j]:vval[i][j]);
		vval[i][j]=0;	
	}


for(i=n;i>0;i--){
	for(j=i,k=1;j>0 && k<=n;j--,k++)
		if(str[j][k]=='X')
			vval[j][k]=vval[j+1][k-1]+1;

	for(++j,--k;j<=n && k >0;j++,k--)
		if(str[j][k]=='X' && str[j-1][k+1]=='X')
			vval[j][k]=vval[j-1][k+1];
}

for(i=n-1;i>0;i--){
	for(j=i,k=1;j>0 && k<=n;j--,k++)
		if(str[n-j+1][n-k+1]=='X')
			vval[n-j+1][n-k+1]=vval[n-(j+1)+1][n-(k-1)+1]+1;

	for(++j,--k;j<=n && k >0;j++,k--)
		if(str[n-j+1][n-k+1]=='X' && str[n-(j-1)+1][n-(k+1)+1]=='X')
			vval[n-j+1][n-k+1]=vval[n-(j-1)+1][n-(k+1)+1];
}

for(i=0;i<n+2;i++)
	for(j=0;j<n+2;j++){
		hval[i][j]=((hval[i][j] > vval[i][j]) ? hval[i][j]:vval[i][j]);
		vval[i][j]=0;	
	}
/*
for(i=1;i<=n;i++){
	for(j=1;j<=n;j++)
		printf("%C",str[i][j]);
	printf("\n");

}
*/
for(i=1;i<=n;i++){
	for(j=1;j<=n;j++)
		printf("%d ",hval[i][j]);
	printf("\n");

}

}
