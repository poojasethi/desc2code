#include <iostream>

using namespace std;

int main()
{
    
   int n;
   scanf("%d",&n);
   char str[1001][1001];
   
   for(int i=0;i<n;i++)
     {
         scanf("%s",&str[i]);
     }
     
   for(int i=0;i<n;i++)
     {
        for(int j=0;j<n;j++)
          {   
              int max=0;
            
           //   printf("%c ",str[i][j]);
              if(str[i][j]=='X')
                {
                  
                    int h1=1;
                    
                    for(int k=j-1;k>=0;k--)
                      {
                         if(str[i][k]=='X')
                           h1++;
                           
                         else 
                           break;
                      }
                     
                     
                   int h2=0;
                      
                   for(int k=j+1;k<n;k++)
                     {
                        if(str[i][k]=='X')
                          h2++;
                          
                        else 
                          break;
                     }                               
                           
                           
                   int v1=1;
                   
                   for(int k=i-1;k>=0;k--)
                     {
                         if(str[k][j]=='X')
                           v1++;
                           
                         else
                           break;
                     }
                     
                   int v2=0;
                   
                   for(int k=i+1;k<n;k++)
                     {
                         if(str[k][j]=='X')
                            v2++;
                         else
                           break;
                     }
                     
                   int d1=1;
                   int k=j-1;
                   for(int d=i-1;d>=0;k--,d--)
                     {
                         if(str[d][k]=='X')
                           d1++;
                           
                         else
                           break;
                     }
                   
                  int d2=0;
                  k=j+1;
                  
                   for(int d=i+1;k<n;k++,d++)
                     {
                         if(str[d][k]=='X')
                           d2++;
                           
                         else
                           break;
                     }           
                 
                 
                 int dd1=1;
                 
                 for(int d=i-1,k=j+1;d>=0&&k<n;d--,k++)
                   {
                       if(str[d][k]=='X')
                         dd1++;
                         
                       else 
                         break;
                   }
                   
                 int dd2=0;
                 
                 for(int d=i+1,k=j-1;d<n&&k>=0;d++,k--)
                   {
                         if(str[d][k]=='X')
                            dd2++;
                            
                         else
                           break;
                   }         
                 
                                
                 d1+=d2;
                 h1+=h2;
                 v1+=v2;
                 dd1+=dd2;
                 
                 
                 max=d1;
                  
                  if(h1>max)
                    max=h1;
                    
                  if(v1>max)
                    max=v1;
                    
                  if(dd1>max)
                    max=dd1;
              }
                if(j<n-1)        
                 printf("%d ",max);
               
                else
                 printf("%d\n",max);                 
          }
    }    
    
//system("pause");

return 0;
}
