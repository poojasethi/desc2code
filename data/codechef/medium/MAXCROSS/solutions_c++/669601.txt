#include <iostream>
#include <cstdio>
using namespace std;
const char fi[] = "MAXCROSS.INP";
const int MAXN = 1000+5;

char c[MAXN][MAXN];
int b[MAXN][MAXN];
int n;

int main()
{
    int i,j,k, l;
    int dem;
   // freopen(fi,"r", stdin);
    scanf("%d", &n);
    for (i=0;i<n;i++)
    {
        scanf("%s", c[i]);
        //printf("%s\n", c[i]);
    }

    for (i=0;i<n;i++)
    for (j=0;j<n;j++)
    //i = 0; j=8;
    {
        if (c[i][j] == '.')
        {
            b[i][j] = 0;
        }
        else
        {
            dem = 0;
            k = 1;
            while (j+k < n && c[i][j+k] == 'X') k++;
            l = -1;
            while (j+l >= 0 && c[i][j+l] == 'X') l--;
            if (k-l-1 > dem) dem = k-l-1;

            k = 1;
            while (i+k < n && c[i+k][j] == 'X') k++;
            l = -1;
            while (i+l >= 0 && c[i+l][j] == 'X') l--;
            if (k-l-1 > dem) dem = k-l-1;

            k = 1;
            while (i+k < n && j+k < n && c[i+k][j+k] == 'X') k++;
            l = -1;
            while (i+l >= 0 && j+l >= 0 && c[i+l][j+l] == 'X') l--;
            if (k-l-1 > dem) dem = k-l-1;

            k = 1;
            while (i+k < n && j-k >= 0 && c[i+k][j-k] == 'X') k++;
            l = -1;
            while (i+l >= 0 && j-l < n && c[i+l][j-l] == 'X') l--;
            if (k-l-1 > dem) dem = k-l-1;
            //cout << k << " " << l << endl;

            b[i][j] = dem;
            //cout << i << " " << j << " " << dem << " " << c[i][j] << endl;
        }
    }
    for (i=0;i<n;i++)
    {
        for (j=0;j<n;j++)
            printf("%d ", b[i][j]);
        printf("\n");
    }
    return 0;
}
