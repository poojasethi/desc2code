#include<stdio.h>
char a[1005][1005];
int b[1005][1005];
int main()
{
int n;
scanf("%d",&n);

int i,j;
for(i=0;i<=n+2;i++)
for(j=0;j<=n+2;j++){
a[i][j] = '.';
}
for(i=1;i<=n;i++)
scanf("%s",&a[i][1]);

for(i=1;i<=n;i++)
for(j=1;j<=n;j++)
if(a[i][j]!='X')b[i][j] = 0;
else {
int ma =0;
int cur;
int x,y;
x=i;y=j;cur=0;
while(a[x][y] == 'X'){ cur++; y++;}
x=i;y=j;
while(a[x][y] == 'X'){ cur++; y--;}
cur--;
if(cur>ma)ma=cur;


x=i;y=j;cur=0;
while(a[x][y] == 'X'){ cur++; x++;}
x=i;y=j;
while(a[x][y] == 'X'){ cur++; x--;}
cur--;
if(cur>ma)ma=cur;


x=i;y=j;cur=0;
while(a[x][y] == 'X'){ cur++; y++;x++;}
x=i;y=j;
while(a[x][y] == 'X'){ cur++; y--;x--;}
cur--;
if(cur>ma)ma=cur;



x=i;y=j;cur=0;
while(a[x][y] == 'X'){ cur++; y++;x--;}
x=i;y=j;
while(a[x][y] == 'X'){ cur++; y--;x++;}
cur--;
if(cur>ma)ma=cur;

b[i][j] = ma;

}

for(i=1;i<=n;i++)
{
printf("%d",b[i][1]);
for(j=2;j<=n;j++)
printf(" %d",b[i][j]);
printf("\n");
}

return 0;
}
