char ar[1005][1005];
int dp[1005][1005][8];

int main()
{

	int i,j,n;
	scanf("%d",&n);
	
	for(i=1;i<=n;i++) 
		scanf("%s",ar[i]+1);

	for(i=n;i>=1;i--){
		for(j=n;j>=1;j--){
			if(ar[i][j]=='X'){
				dp[i][j][0]=1+dp[i][j+1][0];
				dp[i][j][1]=1+dp[i+1][j+1][1];
				dp[i][j][2]=1+dp[i+1][j][2];
				dp[i][j][3]=1+dp[i+1][j-1][3];
			}
		}
	}

	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++){
			if(ar[i][j]=='X'){
				dp[i][j][4]=1+dp[i][j-1][4];
				dp[i][j][5]=1+dp[i-1][j-1][5];
				dp[i][j][6]=1+dp[i-1][j][6];
				dp[i][j][7]=1+dp[i-1][j+1][7];
			}
		}
	}

	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++){
			int ans=0;
			if(ar[i][j]=='X'){
				ans = dp[i][j][1]+dp[i][j][5]-1;
				ans = max(ans, dp[i][j][2]+dp[i][j][6]-1);
				ans = max(ans, dp[i][j][3]+dp[i][j][7]-1);
				ans = max(ans, dp[i][j][4]+dp[i][j][0]-1);
			}
			printf("%d",ans);
			j<n+1?printf(" "):1;
		}
		printf("\n");
	}

	return 0;
}
