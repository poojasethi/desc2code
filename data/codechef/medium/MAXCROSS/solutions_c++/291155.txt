                                                                     
                                                                     
                                                                     
                                             
#include<stdio.h>
#define N 1500
   int hori[N][N];
   int vert[N][N];
   int diag[N][N];
   int diag2[N][N];
   char arr[N][N];
   int print;
   int dim;

    int horizontal(int i,int j)
   {
      int t=0;
      while(1)
      {
         if(i!=dim&&arr[i][j]=='X')
         {
            t++;
            i++;
         }
         else
            break;
      }
      return t;
   }
    int vertical(int i,int j)
   {
      int t=0;
      while(1)
      {
         if(j!=dim&&arr[i][j]=='X')
         {
            t++;
            j++;
         }
         else 
            break;
      }
      return t;
   }
    int diagonal(int i,int j)
   {
      int t=0;
      while(1)
      {
         if(j!=dim&&i!=dim&&arr[i][j]=='X')
         {
            t++;
            j++;
            i++;
         }
         else
            break;
      }
      return t;
   }
    int diagonal2(int i,int j)
   {
      int t=0;
      while(1)
      {
         if(i!=dim&&j!=-1&&arr[i][j]=='X')
         {
            t++;
            i++;
            j--;
         }
         else
            break;
      }
      return t;
   }
    int main()
   {
      scanf("%d",&dim);
      int i;
      int j;
      for(i=0;i<dim;i++)
         scanf("%s",&arr[i][0]);
      for(i=0;i<dim;i++)
      {
         for(j=0;j<dim;j++)
         {
            print=0;
            if(arr[i][j]=='X')
            {
               if(i-1==-1||hori[i-1][j]==0)
                  hori[i][j]=horizontal(i,j);
               else
                  hori[i][j]=hori[i-1][j];
               if(j-1==-1||vert[i][j-1]==0)
                  vert[i][j]=vertical(i,j);
               else
                  vert[i][j]=vert[i][j-1];
               if(j-1==-1||i-1==-1||vert[i-1][j-1]==0)
                  diag[i][j]=diagonal(i,j);
               else
                  diag[i][j]=diag[i-1][j-1];
               if(i-1==-1||j+1==dim||vert[i-1][j+1]==0)
                  diag2[i][j]=diagonal2(i,j);
               else
                  diag2[i][j]=diag2[i-1][j+1];
               if(hori[i][j]>print)print=hori[i][j];
               if(vert[i][j]>print)print=vert[i][j];
               if(diag[i][j]>print)print=diag[i][j];
               if(diag2[i][j]>print)print=diag2[i][j];
            }
            else
            {
               hori[i][j]=0;
               vert[i][j]=0;
               diag[i][j]=0;
               diag2[i][j]=0;
            }
            if(j<dim-1||i<dim-1)
               printf("%d ",print);
            else
               printf("%d",print);
         }
         printf("\n");
      }
      return 0;
   }
