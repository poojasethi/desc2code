#include<iostream>
int b[1001][1001];
char array[1001][1001];
main()
{
      int n,i,j,k,ch,cv,cd1,cd2,t,max;
      scanf("%d",&n);
      for(i=0;i<n;i++)
      {
             scanf("%s",array[i]);
      }
      for(i=0;i<n;i++)
      {
             for(j=0;j<n;j++)
             {
                 if(array[i][j]=='.')
                 b[i][j]=0;
                 else
                 {
                       /////horizontal////
                       k=j;ch=0;
                       while(k<n)
                       {
                             if(array[i][k]=='X')
                             ch++;
                             else
                             break;
                             k++;
                       }
                       k=j-1;
                       while(k>=0)
                       {
                             if(array[i][k]=='X')
                             ch++;
                             else 
                             break;
                             k--;
                       }
                       ///////vertical////
                       k=i;cv=0;
                       while(k>=0)
                       {
                             if(array[k][j]=='X')
                             cv++;
                             else
                             break;
                             k--;
                       }
                       k=i+1;
                       while(k<n)
                       {
                             if(array[k][j]=='X')
                             cv++;
                             else 
                             break;
                             k++;
                       }
                       //////////diagonal////
                       k=i;t=j;cd1=0;
                       while(k<n&&t<n)
                       {
                             if(array[k][t]=='X')
                             cd1++;
                             else 
                             break;
                             k++;t++;
                       }
                       k=i-1;t=j-1;
                       while(k>=0&&t>=0)
                       {
                             if(array[k][t]=='X')
                             cd1++;
                             else
                             break;
                             k--;t--;
                       }
                       //////////diagonal2/////
                       k=i;t=j;cd2=0;
                       while(k<n&&t>=0)
                       {
                            if(array[k][t]=='X')
                            cd2++;
                            else 
                            break;
                            k++;t--;
                       }
                       k=i-1;t=j+1;
                       while(k>=0&&t<n)
                       {
                            if(array[k][t]=='X')
                            cd2++;
                            else
                            break;
                            k--;
                            t++;
                       }                      
                       if(cv>ch)
                       max=cv;
                       else
                       max=ch;
                       if(max<cd1)
                       max=cd1;
                       if(max<cd2)
                       max=cd2;
                       b[i][j]=max;
                 }
             }
      }
      for(i=0;i<n;i++)
      {
           for(j=0;j<n;j++)
           {
                   printf("%d ",b[i][j]);
           }
           printf("\n");
      }
      //system("pause");
}
