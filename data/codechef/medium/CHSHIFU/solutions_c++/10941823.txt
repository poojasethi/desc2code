#include<iostream>
#include<stdlib.h>
using namespace std;
int main(){
	int test,a[21],co,i,n,m,u,v,min,max_co,ref;cin>>test;
	while(test--){
          cin>>n>>m;
		for(i=1;i<=n;i++) a[i]=1<<(i-1);
		while(m--){
			cin>>u>>v;
			a[u]|=(1<<(v-1));
			a[v]|=(1<<(u-1));
		}
		min=n;
		for(max_co=1;max_co<=((1<<n)-1);max_co++){
			ref=0;co=0;
			for(i=1;i<=n;i++){
				if(max_co&(1<<(i-1))){
					ref|=a[i];co++;
				}
			}
			if((ref+1)==(1<<n))	min=co<min?co:min;
		}
		cout<<min<<endl;
	}
}
