#include<iostream>
#include<cstdio>
using namespace std;
inline void scanint(int &x)
{
    register int c = getchar();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = getchar());
    if(c=='-') {neg=1;c=getchar();}
    for(;c>47 && c<58;c = getchar()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
int main()
{
	int t;
	scanint(t);
	while(t--)
	{
		int a[21]={0};
		int n,m,x,y;
		scanint(n);scanint(m);
		for(int i=0;i<n;i++)
		a[i]=1<<i;
		
		while(m--)
		{
			scanint(x);scanint(y);
			x--;y--;
			a[x]|=1<<y;
			a[y]|=1<<x;
		}
		int mn=22;
		for(int i=0;i<1<<n;i++)
		{
			int avail=0,e=0;
			for(int j=0;j<n;j++)
			{
				if(i&1<<j)
				{avail|=a[j];
				e++;}
			}
			if(avail==(1<<n)-1)
			mn=min(mn,e);	
		}
		printf("%d\n",mn);
	}
}
