ll gcd(ll m, ll n){
	
	if(m%n == 0)return n;
	else return gcd(n, m%n);
}

int main(){
	int T;
	cin >> T;
	while(T--){
		ll N,M;
		cin >> N;
		cin >> M;
		if(M == 0 && N != 1){
			cout << "No 1\n";
			continue;
		}else if(M == 0){
			cout << "Yes\n";
			continue;
		}
		if(gcd(N,M) == 1){
			cout << "Yes\n";
		}else cout << "No " << N/gcd(N,M) << "\n";
	}
}
