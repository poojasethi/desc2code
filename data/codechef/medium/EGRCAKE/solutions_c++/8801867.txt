#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int gcd(long long int,long long int);
int main()
{
    long long int m,n,s;
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld%lld",&n,&m);
        s=n/gcd(m,n);
        if(s==n)
            printf("Yes\n");
        else
            printf("No %lld\n",s);
    }
    return 0;
}

int gcd(long long int a,long long int b)
{
  if ( a==0 ) return b;
  return gcd ( b%a, a );
}
