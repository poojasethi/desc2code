#include <iostream>
#define ll long long
using namespace std;
ll gcd(ll a,ll b)
{
    if(b==0)
    return a;
    else
    return gcd(b,a%b);
}
int main() {
	// your code goes here
	ll t;
	cin>>t;
	while(t--)
	{
	    ll n,m;
	    cin>>n>>m;
	    ll k=gcd(n,m);
	    if(k==1)
	    cout<<"Yes"<<endl;
	    else
	    cout<<"No "<<n/k<<endl;
	}
	return 0;
}
