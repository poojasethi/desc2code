#include <iostream>
#include <algorithm>
using namespace std;

int gcd(int a, int b)
{
	if(b == 0)
		return a;
	return gcd(b, a%b);
}

int main()
{
	int tc;
	cin>>tc;
	while(tc--)
	{
		int n, x;
		cin>>n>>x;

		int sub = (x < n/2 ? x : n - x);
		int vis = n/gcd(n,sub);
		if(n == vis)
			cout<<"Yes\n";
		else
			cout<<"No "<<vis<<endl;
	}
}
