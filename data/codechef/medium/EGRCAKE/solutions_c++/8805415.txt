int gcd(int a,int b){
    while(b)b ^= a ^= b ^= a %= b;
    return a;
}

int main(){
#ifdef DEBUG
    fstream fin("EGRCAKE.inp");
#endif
    int count,n,m,t,gcd1;
    cin>>t;
    while(t--){
        cin>>n>>m;
        gcd1 = gcd(n,m);
        if(gcd1 == 1)cout<<"Yes\n";
        else cout<<"No "<<(n/gcd1)<<endl;
    }

}
