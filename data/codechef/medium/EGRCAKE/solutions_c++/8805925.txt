#include <cstdio>
using namespace std;
int t,i,n,m;
int gcd(int a,int b){
    while(b!=0){
        int c=b;
        b=a%b;
        a=c;
    }
    return a;
}
int main(){
        scanf("%d", &t);
        for(i=1;i<=t;i++){

            scanf("%d %d", &n,&m);
            if(gcd(n,m)==1) printf("Yes\n");
            else printf("No %d\n", n/gcd(n,m));
        }
    return 0;
}
