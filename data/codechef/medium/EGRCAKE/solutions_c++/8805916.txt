#include<cstdio>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<string>
using namespace std;
int gcd(int a,int b) {
    while (b!=0) {
        int s=a%b;
        a=b;
        b=s;
    }
    return a;
}
int t,x,y,q;
int main () {
    cin>>t;
    while(t--){
        cin>>x>>y;
        q=gcd(x,y);
        if(q==1) cout<<"Yes"<<endl;
        else if(q==x) cout<<"No 1"<<endl;
             else cout<<"No "<<x/q<<endl;
    }
return 0;
}
