#include<iostream>
#include<cstdio>
#include<algorithm>
int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int m,n;
        scanf("%d%d",&m,&n);
        int a=gcd(m,n);
        if(a==1)
            printf("Yes\n");
        else
            printf("No %d\n",m/a);
    }
}
