#include <bits/stdc++.h>
using namespace std;
int gcd(int a , int b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main() 
{
	int i,j,k,l,n,m,t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		int l = gcd(n,m);
		if(l==1)
			printf("Yes\n");
		else
			printf("No %d\n",n/l);
	}
	return 0;
}
