#include<iostream>
using namespace std;
int gcd(int n,int m)
{
    if(m==0) return n;
    return gcd(m,n%m);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
              long long int N , M;
              cin>>N>>M;
              if(M==0)
              {
               if(N==1) cout<<"Yes"<<endl;
               else   
              cout<<"No"<<" "<<"1"<<endl;
              }
              else
              {
                  int g=gcd(N,M);
                  if(g>1)
                  {
                         cout<<"No"<<" "<<N/g<<endl;
                         }
                  else
                  cout<<"Yes"<<endl;
              
              
              }
              }
    return 0;
}
