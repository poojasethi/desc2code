#include <iostream>
#include <cstdio>
using namespace std;

long long int gcd(long long int a , long long int b);

int main()
{
    int t;
    long long int n,m,a,ans,temp;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        temp = n;
        if(m != 0)
        {
            a = gcd(n,m);
            n = n/a;
            m = m/a;
            ans = n;
            if(ans != temp)
            {
                cout<<"No "<<ans<<endl;
            }
            else
            {
                cout<<"Yes\n";
            }
        }
        else
        {
            if(n != 1)
            {
                cout<<"No 1\n";
            }
            else
            {
                cout<<"Yes\n";
            }
        }
    }
    return 0;
}

long long int gcd(long long int a , long long int b)
{
    return b==0 ? a : gcd(b,a%b);
}
