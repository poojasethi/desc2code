#include <iostream>
#include<cstdio>

using namespace std;
int gcd(int,int);

int main()
{
    int t;
    int n,m;
    scanf("%d",&t);
    while(t--)
    {
        int gcd1;
        scanf("%d%d",&n,&m);
        if(m==0 && n!=1)
            printf("No 1\n");
        else if(m==0)
            cout<< "Yes" << endl;
        else{
            gcd1=gcd(n,m);
            if(gcd1==1)
                cout<< "Yes" << endl;
            else
                cout<<"No "<< n/gcd1 << endl;
        }

    }
    return 0;
}
int gcd(int a,int b)
{

    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
