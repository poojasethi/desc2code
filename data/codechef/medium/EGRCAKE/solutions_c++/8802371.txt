#include<iostream>
using namespace std;
int gcd(int a, int b)
{
   if(b==0)
     return a;
   else 
     return gcd(b,a%b);
}

int main()
{
    int test;
    long int n,m;
    cin>>test;
    while(test--)
   {
       cin>>n>>m;
       int t=gcd(n,m);
       if(t==1)
       {
          cout<<"Yes\n";   
        } 
       else
       {
           cout<<"No "<<n/t<<"\n"; 
       }   
   }
}
