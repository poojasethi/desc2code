#include <cstdio>

using namespace std;

unsigned int gcd (unsigned int n1, unsigned int n2) {
    unsigned int tmp;
    while (n2 != 0) {
        tmp = n1;
        n1 = n2;
        n2 = tmp % n2;
    }
    return n1;
}

int main(){
int tc,n,m,g;
scanf("%d",&tc);
while(tc--){
scanf("%d %d",&n,&m);
g = n/gcd(n,m);
if(g==n)
    printf("Yes\n");
else
    printf("No %d\n",g);


}
return 0;
}
