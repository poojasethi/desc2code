#include<bits/stdc++.h>
using namespace std;

typedef int ll;

ll gcd(ll a, ll b) {

	if ( b==0 )
	return a;
	else
	return gcd(b,a%b);
}


int main () {

    ll test;

    cin>>test;

    while ( test-- ) {

        ll n,m;

        cin>>n>>m;

	ll g = gcd(n,m);
	ll cnt = n/g;
        
        if ( g != 1 ) {
            cout << "No " << cnt << endl;
        }
        else
            cout << "Yes\n";

    }

    return 0;
}
