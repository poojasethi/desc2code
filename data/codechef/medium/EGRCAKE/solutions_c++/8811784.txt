#include <iostream> 
#include <vector>

using namespace std;

int gcd (int n, int m)
{
    return m == 0 ? n : gcd(m, n % m);
}
void findnumrobots(int n, int m)
{
    int g = gcd(n, m);
    if (g == 1) {
        printf ("Yes\n");
    } else {
        printf ("No %d\n", n/g);
    }    
}

int main()
{
   int t;
   cin >> t;
   for (int i = 0; i < t; ++i) {
      int n, m;
      cin >> n;
      cin >> m;
      findnumrobots(n, m);
   }
}
