#define lli long long int

#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);

lli gcd(lli a,lli b)
{
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main()
{
	lli i,j,k,m,n,t;
	slli(t);
	while(t--)
	{
		slli(n);

		if(gcd(n,m)==1)
			printf("Yes\n");

		else
			printf("No %lld\n",n/gcd(n,m));
	}
}
