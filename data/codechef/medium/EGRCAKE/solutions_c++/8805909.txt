    #include<cstdio>
    using namespace std;
    int gcd(int a,int b) {
    while (b!=0) {
    int r=a%b;
    a=b;
    b=r;
    }
    return a;
    }
    int t,m,n,x;
    int main () {
    scanf ("%d",&t);
    while(t--){
    scanf ("%d %d",&n,&m);
    x=gcd(n,m);
    if(x==1) printf("Yes\n");
    else if(x==n) printf("No 1\n");
    else printf("No %d\n",n/x);
    }
    return 0;
    }
