#include<cstdio>
#include<cstring>
using namespace std;
int n,m,t;
int nzd(int a,int b){
 if(b==0)
  return a;
  else return(nzd(b,a % b));
}
int main(){


 scanf("%d",&t);
 while(t--){
  scanf("%d %d",&n,&m);
  //printf("nzd=%d\n",nzd(m,n));
  if(nzd(n,m)==1)
   printf("Yes\n");
   else printf("No %d\n",n/nzd(n,m));
 }



return 0;
}
