#include<bits/stdc++.h>
using namespace std;
vector<int>v[100005];
int visited[100005];
long long cost[100005];
long long mini;
long long mincost[100005];
void dfs(int ind)
{
	visited[ind]=1;
	if(cost[ind]<0)
	{
		
	}
	else
	{
		mini=min(mini,cost[ind]);
	}
	for(int i=0;i<v[ind].size();i++)
	{
		if(!visited[v[ind][i]])
		{
			dfs(v[ind][i]);
		}
	}
}
int h=0;
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	int i,j;
	int x,y;
	for(i=0;i<m;i++)
	{
		scanf("%d %d",&x,&y);
		v[x].push_back(y);
		v[y].push_back(x);
	}
	for(i=1;i<=n;i++)
	scanf("%lld",&cost[i]);
	for(i=0;i<=n;i++)
	visited[i]=0;
	int coo=0;
	for(i=1;i<=n;i++)
	{
		if(!visited[i])
		{
			mini=INT_MAX;
			dfs(i);
			//printf("%lld\n",mini);
			if(mini==INT_MAX)
			{
				coo=1;
			}
			mincost[h++]=mini;
		}
	}
	if(h==1)
	{
		printf("0\n");
		return 0;
	}
	if(coo==1)
	{
		printf("-1\n");
		return 0;
	}
	long long sum=0;
	mini=INT_MAX;
	for(i=0;i<h;i++)
	{
		if(mincost[i]<mini)
		{
			mini=mincost[i];
		}
		sum+=mincost[i];
	}
	long long ans=sum+mini*(long long)(h-2);
	printf("%lld\n",ans);
	return 0;
}
