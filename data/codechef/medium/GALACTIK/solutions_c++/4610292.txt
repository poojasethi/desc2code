#include<bits/stdc++.h>

using namespace std;
 
#define FOR(i,a,b) for(int i=a;i<b;i++)
#define For(i,a,b) for(int i=a-1;i>=0;i--)
#define K 10000000
#define L 400
#define ll long long
#define s1(a) scanf("%d",&a);
#define s2(a) scanf("%lld",&a);
#define s3(a,b) scanf("%d%d",&a,&b);
#define s4(a,b,c) scanf("%d%d%d",&a,&b,&c);
#define pb push_back
#define mp make_pair
#define F first
#define S second 

ll powr(int s,int p)
{
	if(p==0)
		return 1;	
 
	if(p%2==1)
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);
		q=(q*s);	
		return ( q );
	}
 
	else
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);	
		return (q);
	}
}

/*******************************MAIN CODE STARTS*******************************/

int s[100001],n,m,c[100001],k[2][100001];

inline void create()
{
	FOR(i,1,n+1)
		s[i]=i;
}

inline int find(int a)
{
	int x,y;
	y=a;
	while(1)
	{
		x=s[y];
		if(x==y)
			break;
		y=x;
	}
	s[a]=x;
	return x;
}

inline void merge(int a,int b)
{
	int x=find(a);
	int y=find(b);
	if(x!=y)
	{
		if(x>y)
		{
			s[a]=s[x]=y;
		}
		else
		{
			s[b]=s[y]=x;
		}
	}
	return;
}

inline void scan()
{
	int a,b;
	s3(n,m)
	create();
	FOR(i,0,m)
	{
		s3(a,b)
		merge(a,b);
	}
	FOR(i,0,n)
		s1(c[i+1])
}

inline void out()
{
	FOR(i,1,n+1)
		k[0][i]=0;
	FOR(i,1,n+1)
	{
		int x=find(i);
		if(k[0][x]!=1)
		{
			k[0][x]=1;
			if(c[i]>=0)
				k[1][x]=c[i];
			else
				k[1][x]=-1;
		}
		else
		{
			if(c[i]>=0 && k[1][x]==-1)
			{
				k[1][x]=c[i];
			}
			else if(c[i]>=0 && k[1][x]>=0 && c[i]<k[1][x])
			{
				k[1][x]=c[i];
			}
		}
	}
	ll ans=0,mn,no=0;
	FOR(i,1,n+1)
	{
		if(k[0][i]==1)
		{
			if(k[1][i]!=-1)
			{
				if(no==0)
				{
					mn=k[1][i];
					ans+=k[1][i];
					++no;
				}
				else
				{
					//cout<<ans<<'\n';
					ans+=k[1][i];
					if(k[1][i]<mn)
						mn=k[1][i];
					++no;
				}
			}
			else
			{
				ans=-1;
				break;
			}
		}
	}
	//cout<<no<<'\n';
	if(ans==-1)
	{
		int f=0;
		FOR(i,1,n+1)
		{
			if(k[0][i]==1 && f==0)
				f=1;
			else if(k[0][i]==1 && f==1)
			{
				f=0;
				break;
			}
		}
		if(f==0)
			cout<<-1<<'\n';
		else
			cout<<0<<'\n';
	}
	else
	{
		ans=ans+(ll)((no-2)*mn);
		cout<<ans<<'\n';
	}
}

int main()
{
	scan();
	out();
	return 0;
}
