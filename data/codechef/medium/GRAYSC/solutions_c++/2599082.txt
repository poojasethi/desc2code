#include<iostream>
#include<cstdio>
#define loop(i,j,n) for(int i=j;i<n;i++)

using namespace std;

unsigned long long arr[100001];
int main()
{
    int n,i,j,k,l;
    scanf("%d",&n);

    loop(i,0,n)
        scanf("%llu",&arr[i]);



    loop(i,0,n)
        {   loop(j,i+1,n)
            {   loop(k,j+1,n)
                {  loop(l,k+1,n)
                   {
                      if((arr[i]^arr[j]^arr[k]^arr[l])==0)
                         { printf("Yes\n"); return 0; }
                   }
                }
            }
        }
    printf("No\n");

    return 0;

}
