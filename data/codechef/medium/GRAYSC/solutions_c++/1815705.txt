#include <stdio.h>
#include <iostream>
using namespace std;
typedef unsigned long long int LL;
int main()
{
	int n;
	cin>>n;
	if(n>=130)
	{
		printf("Yes");
		return 0;
	}
	LL p[n];
	LL t1;
	for(int i=0; i<n; i++)
		cin>>p[i];
	
	for(int i=0; i<n; i++)
	{
		for(int j=i+1; j<n; j++)
		{
			for(int k=j+1; k<n; k++)
			{
				t1=p[i]^p[j]^p[k];
				for(int l=k+1; l<n; l++)
				{
					if(t1==p[l])
					{
						printf("Yes");
						return 0;
					}
				}
			}
		}
	}					 
	printf("No");
	return 0;
}
