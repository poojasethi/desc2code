#include<iostream>
#include<cstdio>

using namespace std;

unsigned long long arr[100001];
int main()
{
    int n,i,j,k,l;
    scanf("%d",&n);

    for(i=0;i<n;i++)
        scanf("%llu",&arr[i]);



    for(i=0;i<n;i++)
        {   for(j=i+1;j<n;j++)
            {   for(k=j+1;k<n;k++)
                {  for(l=k+1;l<n;l++)
                   {
                      if((arr[i]^arr[j]^arr[k]^arr[l])==0)
                         { printf("Yes\n"); return 0; }
                   }
                }
            }
        }
    printf("No\n");

    return 0;

}
