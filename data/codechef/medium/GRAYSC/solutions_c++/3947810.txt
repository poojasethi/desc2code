#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<string.h>

#define INF 1000000000
#define get getchar_unlocked()

using namespace std;

inline int read ()
{
        char c;
        int n = 0;

        while ((c = get) < 48);
        n += (c - '0');
        while ((c = get) >= 48)
        n = n * 10 + (c - '0');
        return n;
}

int main(){

	int n = read();
	unsigned long long int arr[138];
	
	if(n >= 138){
		printf("Yes\n");
		return 0;
	}

	for(int i = 0 ; i < n ;i++ ){
		cin >> arr[i];
	}		

	for(int j = 0 ; j < n ; j++ ){
		for(int k = j+1 ; k < n ; k++ ){
			for(int l = k + 1 ; l < n ; l++ ){
				for(int m = l +1 ; m < n ; m++ ){
						if((arr[j]  ^ arr[k] ^ arr[l] ^ arr[m]) == 0){
							printf("Yes\n");
							return 0;
						}
				}
			}
		} 
	}
	printf("No\n");

	return 0;
}
