#include <cstdio>
#include <algorithm>
using namespace std;

typedef unsigned long long ull;

ull A[100005];

int main()  {
    int N, i, j, k;
    bool searching = true;

    scanf("%d", &N);
    for (i = 0; i < N; i++)
        scanf("%llu", &A[i]);

    if (N >= 130)   {
        printf("Yes\n");
        return 0;
    }

    sort (A, A + N);

    for (i = 0; searching && i < N; i++)
        for (j = i + 1; searching && j < N; j++)
            for (k = j + 1; k < N; k++)
                if (binary_search (A + k + 1, A + N, A[i] ^ A[j] ^ A[k]))   {
                    searching = false;
                    break;
                }

    if (searching)
        printf("No\n");
    else
        printf("Yes\n");

    return 0;
}
