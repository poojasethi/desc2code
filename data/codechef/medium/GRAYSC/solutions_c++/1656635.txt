#include <iostream>
#include <cstdio>

using namespace std;
typedef unsigned long long ll;

const int MaxN = 100224;

ll a[MaxN];
int n;

int main()
{
    cin.sync_with_stdio(0);

    cin >> n;
    for (int i = 0; i < n; i++) cin >> a[i];

    if (n > 64 * 2 + 15) {puts("Yes"); return 0;}

    for (int i = 0; i < n - 3; i++)
    for (int j = i + 1; j < n - 2; j++)
    for (int k = j + 1; k < n - 1; k++)
    for (int l = k + 1; l < n; l++)
    {
        ll tmp = a[i] ^ a[j] ^ a[k] ^ a[l];
        if (tmp == 0) {puts("Yes"); return 0;}
    }

    puts("No");

    return 0;
}
