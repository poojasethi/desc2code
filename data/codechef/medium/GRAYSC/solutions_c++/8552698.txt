#include <cstdio>
 
using namespace std;
 
typedef unsigned long long ULong;
 
#define NMAX 100005
 
int N;
ULong A[NMAX];
 
bool isPossible() {
  for(int i=0; i<N-3; i++) {
    for(int j=i+1; j<N-2; j++) {
      for(int k=j+1; k<N-1; k++) {
        for(int m=k+1; m<N; m++)
          if((A[i] ^ A[j] ^ A[k] ^ A[m]) == 0)
            return true;
      }
    }
  }
  return false;
}
 
int main() {
  scanf("%d",&N);
  if(N < 130) {
    for(int i=0; i<N; i++)
      scanf("%llu",&A[i]);
    if(isPossible()) printf("Yes\n");
    else printf("No\n");
  }else {
    printf("Yes\n");
  }
  return 0;
}
