#include<stdio.h>
#include<iostream>
#include<math.h>
#include<string.h>

#include<vector>
#include<algorithm>

#define in(i) scanf("%d",&i);

using namespace std;


int main(){
  int n;
  unsigned long long l[68];
  
  
  
    in(n);
    if(n>67)
      printf("Yes\n");
    else
    {
      for(int i=0;i<n;i++)
	scanf("%llu",&l[i]);
      
      for(int a=0;a<n;a++)
	for(int b=a+1;b<n;b++)
	  for(int c=b+1;c<n;c++)
	    for(int d=c+1;d<n;d++)
	      if((l[a]^l[b]^l[c]^l[d])==0){
		printf("Yes\n");
		return 0;		
	      }
      printf("No\n");      
    }
  
  return 0;
}
