    #include <stdio.h>
    int n;
    unsigned long long a[128];



    int main() {

        scanf("%d", &n);

        if (n >= 130) {
            printf("Yes\n");
            return 0;
         }


            int i = 0;

        for (; i < n; ++i) {
         scanf("%llu", &a[i]);
        }


            int i1, i2, i3, i4;


            for (i1 = 0; i1 < n; ++i1) {
                for (i2 = i1 + 1; i2 < n; ++i2) {
                    for (i3 = i2 + 1; i3 < n; ++i3) {
                        for (i4 = i3 + 1; i4 < n; ++i4) {
                                if ((a[i1] ^ a[i2] ^ a[i3] ^ a[i4]) == 0) {
                                        printf("Yes\n");
                                        return 0;
                                        }
                        }
                    }
                }
            }
    printf("No\n");
    return 0;
    }
