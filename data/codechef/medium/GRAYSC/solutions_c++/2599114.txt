#include<iostream>
#include<cstdio>
#define loop(i,j,n) for(int i=j;i<n;i++)


#define gc getchar_unlocked
#define ull unsigned long long

    using namespace std;

    inline ull getn(){
        int c = gc();
            ull n = 0;
            while(c < '0' || c > '9') c = gc();
            while(c >= '0' && c <= '9')
                n = (n<<3) + (n<<1) + c - '0', c = gc();
                return n;
            }


    ull arr[130];
int main()
{
    int n,i,j,k,l;
    scanf("%d",&n);


    if(n>=130)
    {
        printf("Yes\n");
        return 0;
    }

    loop(i,0,n)
        arr[i]=getn();



    loop(i,0,n)
        {   loop(j,i+1,n)
            {   loop(k,j+1,n)
                {  loop(l,k+1,n)
                   {
                      if((arr[i]^arr[j]^arr[k]^arr[l])==0)
                         { printf("Yes\n"); return 0; }
                   }
                }
            }
        }
    printf("No\n");

    return 0;

}
