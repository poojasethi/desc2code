#include <stdio.h>

typedef unsigned long long ull;

int main()
{
    int n, i, j, k, l;
    ull A[70];
    scanf("%d", &n);
    if(n >= 70) {
        printf("Yes\n");
        return 0;
    }
    for(i=0; i<n; i++) scanf("%llu", &A[i]);

    for(i=0; i<n; i++)
        for(j=i+1; j<n; j++)
            for(k=j+1; k<n; k++)
                for(l=k+1; l<n; l++) if((A[i]^A[j]^A[k]^A[l]) == 0) {
                printf("Yes\n");
                return 0;
            }

    printf("No\n");
    return 0;
}
