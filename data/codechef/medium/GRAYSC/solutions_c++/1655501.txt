#include <cstdio>
#include <map>

typedef unsigned long long ull;

int main()
{
    int n;
    scanf("%d", &n);
    if(n >= 70) {
        printf("Yes\n");
        return 0;
    }
    std::map<ull, int> inp;
    ull A[70];
    for(int i=0; i<n; i++) {
        scanf("%llu", &A[i]);
        inp[A[i]] = i;
    }

    for(int i=0; i<n; i++)
        for(int j=i+1; j<n; j++)
            for(int k=j+1; k<n; k++) if( inp[A[i]^A[j]^A[k]] > k ) {
                printf("Yes\n");
                return 0;
            }

    printf("No\n");
    return 0;
}
