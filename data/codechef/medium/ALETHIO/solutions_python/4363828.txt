string = raw_input()


def max_age(string):
  str_len = len(string)
  temp = 0
  for i in range(str_len):
    flag=2
    store = ""
    j=i
    while flag>0 and j< str_len:
      if string[j].isupper():
        flag-=1
        if flag ==1:
          store = store + "9"
      else:
        store = store + string[j]
      j+=1
    if int(store)>temp:
      temp = int(store)
  return temp

print max_age(string)
