#include <string>
#include <iostream>
using namespace std;

int T, N;
string turn[100], road[100];

int main(){
	cin>>T;
	while(T--){
		//string line;
		cin>>N;
		//getline(cin, line);
		for(int i=0; i<N; i++){
			cin>>turn[i];
			getline(cin, road[i]);
		}
		for(int i=0; i<N; i++){
			if(i==0)
				cout<<"Begin";
			else if(turn[N-i]=="Left")
				cout<<"Right";
			else
				cout<<"Left";
			cout<<road[N-1-i]<<endl;
		}
		cout<<endl;
	}
	return 0;
}
