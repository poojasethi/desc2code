#include<bits/stdc++.h>
using namespace std;

int main()

{
	int T;
	cin>>T;
	
	while(T--)
	{
		int N, i, j, counter=0, mark[50];
		cin>>N;
		char ch;
		ch=getchar();
		char S[50][100];
		for(i=0; i<N; i++)
		{
			scanf("%[^\n]s", S[i]);
			ch=getchar();
			
			j=0;
			while(counter!=2)
			{
				if(S[i][j]==' ')
				counter++;
				j++;
			}
			mark[i]=j;
			counter=0;
		}
		j=mark[N-1];
		printf("Begin on ");
		while(S[N-1][j]!='\0')
		{
			printf("%c",S[N-1][j]);
			j++;
		}
		printf("\n");
		
		for(i=N-1; i>0; i--)
		{
			if(S[i][0]=='L')
			printf("Right on ");
			
			else
			printf("Left on ");
			
				j=mark[i-1];
				while(S[i-1][j]!='\0')
				{
					printf("%c",S[i-1][j]);
					j++;
				}
			cout<<"\n";
		}
		
	}
	return 0;
}
