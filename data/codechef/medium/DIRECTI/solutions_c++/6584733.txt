#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		char s[41][100],route[41][100],on[10];
		for(int i=0;i<n;i++)
		{
			//getline(s[i]);
			scanf("%s %s",route[i],on);
			getchar();
			gets(s[i]);
		}
		printf("%s %s %s\n",route[0],on,s[n-1]);
		for(int i=n-2;i>=0;i--)
		{
			if(route[i+1][0]=='L')
				printf("Right %s %s\n",on,s[i]);
			if(route[i+1][0]=='R')
				printf("Left %s %s\n",on,s[i]);
		}
	}
return 0;
}
