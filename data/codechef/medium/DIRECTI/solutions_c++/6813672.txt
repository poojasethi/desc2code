#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		string dir[100];
		string roads[100];
		int n;
		cin>>n;
		for (int i =1; i <= n; ++i)
		{
			cin>>dir[i];
			getline(cin,roads[i]);
		}
		reverse(dir+2,dir+n+1);
		for (int i = 2; i <=n; ++i)
		{
			if(dir[i]=="Left")dir[i]="Right";
			else if(dir[i]=="Right")dir[i]="Left";
		}
		reverse(roads+1,roads+n+1);
		for (int i = 1; i <=n; ++i)
		{
			cout<<dir[i]<<roads[i]<<endl;
		}
	}
	return 0;
}
