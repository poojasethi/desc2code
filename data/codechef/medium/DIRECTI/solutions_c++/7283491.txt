#include <iostream>
#include <string.h>
#include <cstdlib>
#include <cstring>
#include <stdio.h>
using namespace std;

int main()
{
    int t,n,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        getchar();
        char *suffix[n];
        char pref[n];
        string tmp;
        for(i=0;i<n;i++)
        {
            //cin>>tmp;
            getline(std::cin,tmp);
            //cout<<tmp[0]<<endl;
            pref[i] = tmp[0];
            j=0;
            while(tmp[j]!=' ')
                j++;
            //cout<<j<<endl;
            suffix[i] = (char*)malloc(50);
            strcpy(suffix[i] , (tmp.substr(j,tmp.length()-1)).c_str());
            //cout<<suffix[i]<<endl;
            //cout<<tmp.substr(j,tmp.length()-1)<<endl;
        }

        for(i=n-1;i>=0;i--)
        {
            if(i==n-1)
                cout<<"Begin";
           else if(pref[i+1]=='R')
                cout<<"Left";
           else if(pref[i+1]=='L')
            cout<<"Right";


           cout<<suffix[i]<<endl;
        }
        cout<<endl;
    }
    return 0;
}
