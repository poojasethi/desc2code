// REV DIRECTION 

#include <iostream>
#include <cstring>
using namespace std;
int tc, n;

int main()
{
	cin >> tc;
	while(tc--)
	{
		cin >> n;
		string dir[44], rest[44];
		for(int i = 0; i < n; ++i)
		{
			cin >> dir[i];
			getline(cin, rest[i]);
		}
		cout << dir[0] + rest[n - 1] << endl;
		for(int i = n - 2; i >= 0; --i)
		{
			if(dir[i + 1] == "Left")
				cout << "Right" << rest[i] << endl;
			if(dir[i + 1] == "Right")
				cout << "Left" << rest[i] << endl;
		}
	} 
	return 0;
}
