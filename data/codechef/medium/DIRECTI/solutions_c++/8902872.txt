#include <bits/stdc++.h>
using namespace std;

string turn[1000],road[1000];
int main()
{
    int t,n;
    cin >> t;

    while(t--) {

        cin >> n;
        //cout << "\n";
        int i;

        for(i = 0;i < n;i++) {
            cin >> turn[i];
            getline(cin,road[i]);
        }

        for(i = 0;i < n;i++) {
            if(i == 0) {
                cout << "Begin";
            }
            else if(turn[n-i] == "Left") {
                cout << "Right";
            }

            else {
                cout << "Left";
            }
            cout << road[n-1-i] << "\n";

        }
    }
    return 0;
}
