#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{	
	ll t,n,i; string s[50];
	cin>>t;
	while(t--)
	{
		scanf("%lld",&n);
		cin.get();
		for(i=0;i<n;i++)
		getline(cin,s[i]);
		for(i=n-1;i>=0;i--)
		{
			if(i==n-1)
			{
				if(s[i].substr(0,5)=="Right")
				cout<<"Begin on"<<s[i].substr(8)<<endl;
				else
				cout<<"Begin on"<<s[i].substr(7)<<endl;
			}
			else
			{
				if(s[i+1].substr(0,5)=="Right")
				{
					if(i==0)
					cout<<"Left"<<s[i].substr(5)<<endl;
					else if(s[i].substr(0,5)=="Right")
					cout<<"Left"<<s[i].substr(5)<<endl;
					else
					cout<<"Left"<<s[i].substr(4)<<endl;
				}
				else
			    {
			    	if(i==0)
					cout<<"Right"<<s[i].substr(5)<<endl;
			    	else if(s[i].substr(0,5)=="Right")
					cout<<"Right"<<s[i].substr(5)<<endl;
					else
					cout<<"Right"<<s[i].substr(4)<<endl;
			    }
			}
		}
	}
	return 0;
}
