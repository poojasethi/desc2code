#include<cstdio>
#include<cstring>
using namespace std;
typedef long long int ll;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        char str[n][56];
        int dir[n];
        for(int i=0;i<n;i++){
            char d[10],waste[5];
            scanf("%s %s %[^\n]s",&d,&waste,&str[i]);
            if(strcmp(d,"Begin")==0){
                dir[i]=0;
            }else if(strcmp(d,"Right")==0){
                dir[i]=1;
            }else{
                dir[i]=2;
            }
        }
        for(int i=0;i<n-1;i++){
            dir[i]=dir[i+1];
        }
        dir[n-1]=0;
        char dir_str[][10]={"Begin","Left","Right"};
        for(int i=n-1;i>=0;i--){
            printf("%s on %s\n",dir_str[dir[i]],str[i]);
        }
        printf("\n");
    }
}
