#include<iostream>
using namespace std;
int main(){
	int t,n,i,j;
	cin>>t;
	while(t--){
		cin>>n;
		string s[n];
		cin.ignore(1,'\n');
		for(i=0;i<n;i++){
			getline(cin,s[i]);
		}
		for(i=n-1;i>=0;i--){
			if(i==n-1)
			cout<<"Begin on";
			if(s[i][0]=='l'||s[i][0]=='L'){
				for(j=7;j<s[i].length();j++)
				cout<<s[i][j];
				cout<<endl<<"Right on";
			}
			else if(s[i][0]=='r'||s[i][0]=='R'){
				for(j=8;j<s[i].length();j++)
				cout<<s[i][j];
				cout<<endl<<"Left on";
			}
			else if(s[i][0]=='B'||s[i][0]=='b'){
				for(j=8;j<s[i].length();j++)
				cout<<s[i][j];
			}
		}
		cout<<endl<<endl;
	}
	return 0;
}
