def rdir(t):
	for i in range(t):
		n,lc,lr = input(),[],[]
		for j in range(n):
			c,r = map(str,raw_input().split(' on '))
			lc.append(c)
			lr.append(r)
		print 'Begin on %s'%(lr[n-1])
		for j in range(n-2,-1,-1):
			if lc[j+1] == 'Right':
				print 'Left on %s'%(lr[j])
			else:
				print 'Right on %s'%(lr[j])
		print ''
rdir(input())
