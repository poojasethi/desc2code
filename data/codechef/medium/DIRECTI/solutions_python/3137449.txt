while(T>0):
	l = []
	N = input()
	for i in range(N) :
		l.append((raw_input()).split())
	print "\nBegin on",
	for words in l[-1][2:] :
		print words,
	i = 2
	while(i<=N) :
		if(l[-(i-1)][0] == "Left") :
			print "\nRight on",
			for words in l[-i][2:] :
				print words,
			print "\n"
		elif (l[-(i-1)][0] == "Right") :
			print "\nLeft on",
			for words in l[-i][2:] :
				print words,
		i += 1
	T -= 1
