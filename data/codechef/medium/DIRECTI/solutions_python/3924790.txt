import sys

num = int(sys.stdin.readline().rstrip())
while num != 0:
	num_instructions = int(sys.stdin.readline().rstrip())
	num_ins = num_instructions
	directions = []
	paths = []
	while num_instructions != 0:
		cur_path = sys.stdin.readline().rstrip().split(' on ')
		cur_dir = cur_path[0]
		paths.append(cur_dir)
		directions.append('on ' + cur_path[1])
		num_instructions -= 1
	paths.append("Begin")
	for i in xrange(0,num_ins):
		path = paths.pop()
		if path == "Left" :
			path = "Right"
		elif path == "Right":
			path = "Left"
		print path, directions.pop()
	print '\n'
	num -= 1
