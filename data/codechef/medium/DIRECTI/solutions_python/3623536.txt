def processTestCase():
    instructions_count = int(raw_input())
    instructions_stack = []
    for i in xrange(instructions_count):
        instructions_stack.append(raw_input())

    previous_direction = None
    for i in xrange(instructions_count):
        direction, rest = instructions_stack.pop().split(' ', 1)
        new_direction = 'Begin' if not previous_direction else 'Left' if previous_direction == 'Right' else 'Right'
        print ' '.join([new_direction, rest])
        previous_direction = direction

def main():
    test_case_count = int(raw_input())
    for i in xrange(test_case_count):
        processTestCase()

if __name__ == "__main__":
    main()
