
t = int(raw_input())

complement = {'Left': 'Right', 'Right': 'Left', 'Begin': 'Begin'}

while t > 0:
    t = t - 1
    n = int(raw_input())
    direction, road = [], []
    for i in range(n):
        ins = raw_input()
        direction.append(ins.partition(" ")[0])
        road.append(ins.partition(" ")[-1])
    direction[1:] = direction[-1:0:-1]
    for d, r in zip(direction, road[::-1]):
        print complement[d], r
