#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,t,colr[100001],flag=0;;
	cin>>t;
	while(t--)
	{
		flag=0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>colr[i];
		}
		for(int i=0;i<n-2;i++)
		{
			if(colr[i]==colr[i+1]&&colr[i+1]==colr[i+2])
			{
				flag=1;
				break;
			}
			else
			flag=0;
		}
		if(flag)
		cout<<"Yes\n";
		else
		cout<<"No\n";
	}
	return 0;
}
