#include <cstdio>
#include <algorithm>

int main()
{
	long int T, N, *color, i, color_value;
	bool isCorrect=false;
	scanf("%ld", &T);
	
	while(T--)
	{
		scanf("%ld", &N);
		isCorrect=false;
		color = new long int[N];
		for(i=0; i<N; i++)
		{
			scanf("%ld", &color[i]);
		}

		for(i=0; i<(N-2); i++)
		{
			if(color[i]==color[i+1] && color[i+1]==color[i+2]){
				isCorrect=true;
				break;
			}
		}
		
		if(isCorrect)
			printf("Yes\n");
		else
			printf("No\n");
	}

	return 0;
}
