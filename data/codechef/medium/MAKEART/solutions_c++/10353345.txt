#include<iostream>
typedef long long int ll;
using namespace std;
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll N,i;
		cin>>N;
		ll A[N];
		for(i=0;i<N;i++)
			cin>>A[i];
		bool flag=false;
		for(i=1;i<N-1;i++)
		{
			if(A[i-1]== A[i] && A[i]==A[i+1])
			{
				flag=true;
				break;
			}
		}
		if(flag)
			cout<<"Yes\n";
		else 
			cout<<"No\n";
	}
	return 0;
}
