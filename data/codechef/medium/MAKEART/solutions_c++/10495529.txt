#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int N;
		scanf("%d",&N);
		int* arr = new int[N];
		for(int i = 0; i < N; i++)
			scanf("%d",&arr[i]);
		bool ok = false;
		for(int i = 0; i < N - 2; i++)
		{
			if(arr[i] == arr[i + 1] && arr[i] == arr[i + 2])
			{
				ok = true;
				break;
			}
		}
		if(ok)
			printf("Yes\n");
		else
			printf("No\n");
	}
	return 0;
}
