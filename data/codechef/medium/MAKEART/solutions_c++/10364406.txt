#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<cmath>

using namespace std;

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int i, c[n+1];
        int flag=0;


        for(i=0;i<n;i++)
        {
            scanf("%d",&c[i]);

             if(i>=2)
             {
                 if(c[i]==c[i-1]&&c[i]==c[i-2])
                      flag=1;
             }
        }



        if(flag)
            printf("Yes\n");

        else
          printf("No\n");

    }
    return 0;

}
