#include<stdio.h>
int main()
{
    int t,n;
    int a,b,c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        c=b=a=-1;
        char trip='f';
        for(int i=0;i<n;i++)
        {
            c=b;b=a;
            scanf("%d",&a);
            if(i>=2)
            {
                if(a==b && a==c)
                {
                    trip='t';
                }
            }
        }
        if(trip=='t')
            printf("Yes\n");
        else
            printf("No\n");
    }
    return 0;
}
