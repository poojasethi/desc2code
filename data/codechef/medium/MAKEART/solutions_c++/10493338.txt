using namespace std;
#include<stdio.h>
#include<vector>

vector<int> colours(100005);
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        int N;
        scanf("%d",&N);

        for(int i=0;i<N;i++)
            scanf("%d",&colours[i]);

        bool flag = false;

        for(int i=0;i<N;i++)
        {
            int k=1;
            for(int j=1;j<3&&i+j<N;j++)
            {
                if(colours[i] == colours[i+j])
                    k++;
                else
                    break;
            }
            if(k==3)
            {
                flag = true;
                break;
            }
        }

        if(flag)
            printf("Yes\n");
        else
            printf("No\n");
    }
return 0;
}
