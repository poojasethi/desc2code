#include<iostream>
using namespace std;
int main(){
	int t,n,i,count;
	cin>>t;
	while(t--){
		cin>>n;
		int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0,count=0;i<n-2;i++){
			if(a[i]==a[i+1]&&a[i+1]==a[i+2]){
				count=10;
				break;
			}
		}
		if(count==10)
		cout<<"Yes";
		else
		cout<<"No";
		cout<<endl;
	}
	return 0;
}
