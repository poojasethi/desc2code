int arr[100000];
int main()
{
	int i,n,t,flag=0;
	cin>>t;
	while(t--)
	{
		flag=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>arr[i];
	}
	if(n>=3)
	{
		for(i=0;i<n-2;i++)
		{
			if(arr[i]==arr[i+1] && arr[i+1]==arr[i+2])
			{
				flag=1;
				break;
			}
		}
		if(flag==1)
		cout<<"Yes\n";
		else
		cout<<"No\n";
	}
	else
	cout<<"No\n";
}
	return 0;
}
