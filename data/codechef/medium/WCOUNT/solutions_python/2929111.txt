def wper(t):
	for i in range(t):
		s = raw_input()
		d = {}
		for j in range(len(s)):	
			if s[j] in d.keys():
				d[s[j]] += 1
			else:
				d[s[j]] = 1
		p = 1
		for j in d.keys():
			p = p * fact(d[j])
		print ((fact(len(s)))/p)%1000000007

def fact(a):
	p = 1
	for i in range(1,a+1):
		p = p*i
	return p

wper(input())
