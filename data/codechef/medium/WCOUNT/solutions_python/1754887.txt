def vroom(x):
    a = 1
    for n in range(1,x+1):
        a *= n
    return a
for T in range(int(input())):
    same = 0
    s = {}
    S = list(raw_input())
    for l in S:
        if s.has_key(l):
            s[l] +=1
        else:
            s[l] = 1
    ans = vroom(len(S))
    for key,val in s.items():
        if val != 1:
            ans /= vroom(val)
