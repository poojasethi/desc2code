def main():
	tc=int(raw_input().strip());
	for x in range(tc):
		s=str(raw_input().strip());
		l=len(s);
		d=dict();
		for y in range (0,l,+1):
			ch=s[y];
			if d.has_key(ch):
				v=d[ch];
				d[ch]=v+1;
			else:
				d[ch]=1;
		lisv=d.values();
		nf=fact(l);
		df=1;
		for z in lisv:
			df=df*fact(z);
		ans=nf/df;
		ans=ans%1000000007;
		print ans;
def fact(n):
	if n<=1:
		return 1;
	return n*fact(n-1);		

if __name__=="__main__":main();
