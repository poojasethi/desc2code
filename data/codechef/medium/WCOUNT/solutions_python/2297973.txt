import math
cases = int(raw_input())
for i in range(cases):
    strin = raw_input()
    length = len(strin)
    total = math.factorial(length)
    fact = 1
    ans = 1
    for j in range(length):
        char = strin[j]
        if(char == '0'):
            continue
        fact = strin.count(char)
        strin = strin.replace(char , '0')
        if(fact > 1):
            ans *= math.factorial(fact)
        fact = 1
    print (total/ans)%1000000007
