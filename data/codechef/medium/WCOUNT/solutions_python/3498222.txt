from collections import Counter
import math

modulus = (10 ** 9) + 7

cases = int(raw_input())

for _ in xrange(cases):
    S = raw_input()
    counter = Counter(S)
    nominator = math.factorial(len(S))
    denominator = 1
    for char in counter:
        times = counter[char]
        denominator *= math.factorial(times)

    print (nominator / denominator) % modulus
