from sys import stdin, stdout
MOD = 10**9 + 7
fact = [1]
for i in range(1, 503):
	fact.append((fact[-1]*i)%MOD)
t = int(stdin.readline())
while t:
	t -= 1
	s = stdin.readline().strip()
	n = len(s)
	count = {}
	for i in s:
		if i in count:
			count[i] += 1
		else:
			count[i] = 1
	den = 1
	for i in count:
		den = (den * fact[count[i]]) % MOD
	ans = (fact[n] * pow(den, MOD-2, MOD) ) % MOD
	stdout.write(str(ans)+'\n')
