import math
t=int(raw_input())
while t:
    s=raw_input()
    a={}
    for j in s:
        if a.has_key(j):
            a[j]+=1
        else:
            a[j]=1
    l=math.factorial(len(s))
    for j in a:
        if a.get(j)>1:
            l/=math.factorial(a.get(j))
    print l%(10**9+7)
    t-=1
