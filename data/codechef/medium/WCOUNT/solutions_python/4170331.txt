from math import factorial
f = factorial
test = int(raw_input().strip())
while test:
    s= raw_input().strip()
    d = {}
    for i in xrange(len(s)):
        if s[i] in d:
            d[s[i]]+=1
        else:
            d[s[i]]=1
    t = f(len(s))
    m = 1
    for key in d:
        m=m*f(d[key])
    print (t/m)%1000000007
    test-=1
