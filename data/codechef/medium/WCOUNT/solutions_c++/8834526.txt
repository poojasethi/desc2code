#include<stdio.h>
#include<stdlib.h>
#define MOD 1000000007;
int a[100];
char str[510];

int main()
{
	int i,j,t,l;
	long long int res,res2,b;
	long long int fact[11]={1,1,500000004,166666668,41666667,808333339,301388891,900198419,487524805,831947206,283194722};
	scanf("%d",&t);
	while(t--)
	{
		res=res2=1;
		scanf("%s",str);
		for(i=0;str[i]!='\0';i++)
		{
			if(str[i]>=97 && str[i]<=122)
			a[str[i]-97]++;
			else
			a[str[i]-39]++;
			res= (res*(i+1))%MOD;
		}
		//printf("%lld ",res);
		for(i=0;i<52;i++)
		{
			res2=(res2*(fact[a[i]]))%MOD;
			//printf("%lld ",res2);
			a[i]=0;
		}
		res=(res*res2)%MOD;
		printf("%lld\n",res);
	}
	return 0;
}
