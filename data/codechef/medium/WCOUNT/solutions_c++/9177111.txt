#include<iostream>
#define mod 1000000007
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        unsigned long long int length=s.length();
        unsigned long long int i;
        int carr[128];
        for(i=0;i<128;i++)
            carr[i]=0;

        for(i=0;i<length;i++)
            carr[s[i]]++;

        int arr[60];
        for(i=0;i<60;i++)
            arr[i]=0;
        unsigned long long int k=0;

        for(i=0;i<128;i++)
        {
            if (carr[i]!=0)
                {arr[k]=carr[i];
                k++;}
        }
        k=k-1;

        unsigned long long int num=1,den=1;

        while(length!=0)
        {
            num=(num*length)%mod;
            length--;
        }

        i=0;
        while(arr[i]!=0)
        {


            unsigned long long int m=arr[i];
            while(m!=0)
            {
                den=(den*m)%mod;
                m--;
            }
            i++;
        }

        unsigned long long int base = den;
        unsigned long long int result=1;
        unsigned long long int power= 1000000005;

        while(power!=0)
        {
            if (power%2==0)
            {
                base=(base*base)%mod;
                power=power/2;
            }
            else
            {
                result=(result*base)%mod;
                power=power-1;
            }
        }

        cout<<(num*result)%mod<<'\n';


    }
    return 0;

}
