#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<map>
using namespace std;
vector<int> v(501,1);
const long long mod=1000000007;
int power(long long a,long long b)
{
    long long temp;
    int res;
    if(a==1)
        return 1;
    if(b==0)
        return 1;
    if(b==1)
        return a;
    temp=power(a,b/2);
    temp=(temp*temp)%mod;
    if(b%2)
        temp=(temp*a)%mod;
    res=temp;
    return res;
}
int fact(int n)
{
    if((n==0)||(n==1))
        return 1;
    if(v[n]==1)
    {
        int i;
        long long fact=1;
        for(i=1;i<=n;i++)
        {
            fact=(fact*i)%mod;
        }
        v[n]=fact;
    }
    return v[n];
}
int calc(long long a,long long b)
{
    long long store,temp;
    int res;
    store=power(b,mod-2);
    temp=(a*store)%mod;
    res=temp;
    return res;
}
int main()
{
    int t,i,len,res;
    long long store1,store2;
        char str[501];
    scanf("%d",&t);
    while(t--)
    {
        store2=1;
        map<char,long long> m;
        scanf("%s",str);
        len=strlen(str);
        for(i=0;i<len;i++)
        {
            m[str[i]]++;
        }
        store1=fact(len);
        map<char,long long>::iterator it;
        for(it=m.begin();it!=m.end();it++)
        {
            store2=(store2*fact(it->second))%mod;
        }
        res=calc(store1,store2);
        printf("%d\n",res);
    }
    return 0;
}
