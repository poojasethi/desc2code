long long factorial(long long n)
{
  return ((n == 1 || n == 0) ? 1 : (factorial(n - 1)%1000000007  * n%1000000007)%1000000007);
}

long long fact(long long n)
{
  return (n == 1 || n == 0) ? 1 : (fact(n - 1) * n);
}


long long pow(long long b,long long p )
{
    long long res=1;
    b=b%1000000007;
    while(p>0)
    {
        if(p%2==1)
            res=(res*b)%1000000007;
        p=p>>1;
        b=(b*b)%1000000007;
    }
    return res;
}

long long modinv(long long a)
{
    if(a==0)return 1;
    else
    return pow(a,1000000005)%1000000007;
}

int main()
{
	int N,i,j;
	string str,temp;
	int A[26]={0},a[26]={0};
	long long len, cnt=0;
	cin>>N;
	getline(cin,str);
	for(i=0;i<N;i++)
	{
		getline(cin,str);
		for(j=0;str[j]!='\0';j++)
		{
			if(str[j]>='a' && str[j]<='z')a[str[j]-'a']++;
			else A[str[j]-'A']++;
		}
		len=str.length();
		cnt=factorial(len);
		for(j=0;j<26;j++){//cout<<A[j]<<"\t"<<a[j]<<endl;
		cnt=(cnt*modinv(fact(a[j])))%1000000007;
		a[j]=0;
		//cout<<cnt<<"\t";
        cnt=(cnt*modinv(fact(A[j])))%1000000007;
        A[j]=0;
        //cout<<cnt<<endl;
        }
		cout<<cnt<<endl;
		cnt=1;//%1000000007<<endl;
	}
	return 0;
}
