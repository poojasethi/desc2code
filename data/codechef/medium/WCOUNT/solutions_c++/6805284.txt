#define MOD 1000000007

long long int fact(int n)
{
	long long int ans=1;
	for(int i=2;i<=n;i++)
	{
		ans=ans*i;
		ans%=MOD;
	}
	return ans;
}

long long int inv(int n)
{
	long long int y,x,i;
	i=MOD-2;
	
	y=1;
	x=n;
	while(i)
	{
		if(i%2)
			y*=x;
		
		x*=x;
		i/=2;
		x%=MOD;
		y%=MOD;
	}
	return y;
}
