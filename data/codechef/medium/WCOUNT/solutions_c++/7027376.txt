#include<bits/stdc++.h>

typedef unsigned long long u64;
u64 modl=1000000007;
u64 factv[505],down=1,inv,ans;

	u64 pows(u64 a, u64 b, u64 MOD) {
	u64 x = 1, y = a;
	    while(b > 0) {
	        if(b%2 == 1) {
	            x=(x*y);
	            if(x>MOD) x%=MOD;
	        }
	        y = (y*y);
	        if(y>MOD) y%=MOD;
	        b /= 2;
	    }
	    return x;
	}
	 
	u64 modInverse(u64 a, u64 m) {
	    return pows(a,m-2,m);
	}
