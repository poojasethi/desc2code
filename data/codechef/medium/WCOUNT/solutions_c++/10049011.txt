#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL MOD = 1e9 + 7;
LL fac[600];
LL power(LL A, LL B){
    LL ans = 1;
    while(B > 0){
        if(B & 1) ans = (ans * A) % MOD;
        A = (A * A) % MOD;
        B >>= 1;
    }
    return ans;
}
int main(){
    fac[0] = 1;
    for(int i = 1;i < 600;++i){
        fac[i] = (fac[i - 1] * i) % MOD;
    }
    int t;
    cin >> t;
    while(t--){
        string s;
        cin >> s;
        int aCnt[26] = {0};
        int ACnt[26] = {0};
        int sz = s.size();
        for(int i = 0;i < sz;++i){
            if(s[i] >= 'a')
                aCnt[s[i] - 'a'] += 1;
            else ACnt[s[i] - 'A'] += 1;
        }
        LL ans = fac[s.size()];
        for(int i = 0;i < 26;++i){
            ans = (ans * power(fac[aCnt[i]], MOD - 2)) % MOD;
        }
        for(int i = 0;i < 26;++i){
            ans = (ans * power(fac[ACnt[i]], MOD - 2)) % MOD;
        }
        printf("%lld\n", ans);
    }
    return 0;
}
