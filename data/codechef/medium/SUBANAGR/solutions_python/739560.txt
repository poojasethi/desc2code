def id(c): return ord(c)-ord('a')

n = int(raw_input())
S = []
for i in range(n): S.append(raw_input())

cnt = [[] for i in range(26)]

for s in S:
    cnt2 = [0]*26
    for c in s: cnt2[id(c)]+=1
    for i in range(26): cnt[i].append(cnt2[i])
res = []
for i in range(26):
    tmp = min(cnt[i])
    if tmp>0: res.extend([chr(i+ord('a'))]*tmp)
if res:print ''.join(res)
else: print 'no such string'
