ans =''
flag=1
while n:
    n-=1
    if flag==1:
        ans = raw_input()
        ans = ''.join(sorted(ans))
        flag=0
    else:
        m = ''.join(sorted(raw_input()))
        temp =''

                temp += l
                ans = ans[:ans.find(l)] + ans[ans.find(l)+1:]
        ans = temp
