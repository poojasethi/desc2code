#include<stdio.h>
#include<string.h>
int main()
{
    int t,i,num[105][26]={0},j,min;
    char s[105];
    scanf("%d",&t);
    for(j=0;j<t;j++)
    {
        scanf("%s",s);
        for(i=0;s[i]!='\0';i++)
            num[j][s[i]-97]++;
    }
    int flag=0;
    for(i=0;i<26;i++)
    {
        min=num[0][i];
        for(j=1;j<t;j++)
            if(min>num[j][i])
                min=num[j][i];
        for(j=0;j<min;j++){
            flag=1;
            printf("%c",i+97);}
    }
    if(flag==0)
            printf("no such string\n");
    else
        printf("\n");
    return 0;
}
