#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j,arr[26]={0},arr2[26]={0},flag=0;
    string x;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>x;
        for(j=0;j<x.size();j++)
        {
            arr[x[j]-97]++;
        }
        if(i==0)
        {
            for(j=0;j<26;j++)
            {
                arr2[j]=arr[j];
            }
        }
        else
        {
            for(j=0;j<26;j++)
            {
                if(arr[j]>0&&arr2[j]>0)
                {
                    arr2[j]=arr[j]>arr2[j]?arr2[j]:arr[j];
                }
                else
                    arr2[j]=0;
            }
        }
        for(j=0;j<26;j++)
            {
                arr[j]=0;
            }
    }
    for(i=0;i<26;i++)
    {
        for(j=0;j<arr2[i];j++)
        {
            flag=1;
            printf("%c",i+97);
        }
    }
    if(flag==0)
        cout<<"no such string";
    cout<<endl;
    return 0;
}
