#include<stdio.h>
#include<stdlib.h>

int a[30],final[30];
char str[110];
int main()
{
	int i,j,n,flag;
	scanf("%d",&n);
	scanf("%s",str);
	for(i=0;str[i]!='\0';i++)
		{
			final[str[i]-'a']++;
		}
		n--;
		
	while(n--)
	{
		flag=0;
		scanf("%s",str);
		for(i=0;str[i]!='\0';i++)
		{
			a[str[i]-'a']++;
		}
		for(i=0;i<26;i++)
		{
			if(final[i]>0)
			{
				if(final[i]>=a[i])
				final[i]=a[i];
			}
			a[i]=0;
		}
		
	}
	for(i=0;i<26;i++)
	{
		if(final[i]>0)
		{
			while(final[i]-- )
			printf("%c",i+'a');
			flag=1;
		}
	}
	if(flag==1)
	printf("\n");
	else
	printf("no such string\n");
	return 0;
}
