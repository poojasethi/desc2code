#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{	
	int n,i;string s;
	scanf("%d",&n);
	map<char,int> final;
	cin>>s;
	for(i=0;i<s.size();i++)
	final[s[i]]++;
	
	n--;
	while(n--)
	{
		map<char,int> tmp;
		cin>>s;
		for(i=0;i<s.size();i++)
		tmp[s[i]]++;
		
		for(char a='a';a<='z';a++)
		final[a]=min(final[a],tmp[a]);
	}
	
	int sw=0;
	for(char a='a';a<='z';a++)
	{
		for(i=1;i<=final[a];i++)
		printf("%c",a);
		if(final[a]>0)
		sw=1;
	}
	
	if(sw==0)
	printf("no such string");
	
	return 0;
}
