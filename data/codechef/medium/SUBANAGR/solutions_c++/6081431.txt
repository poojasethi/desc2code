#include<bits/stdc++.h>
using namespace std;
int a[26][1005];

int minimum(int n , int x )
{
	int ans=10000;
	for(int i=0 ; i<n ; i++)
	ans=min(ans , a[x][i]);
	return ans;
}
int main()
{
	int n , ans=0 , flag=0;
	char s[105];

	scanf("%d" , &n);
	
		memset(a , 0 , sizeof(a));
		for(int i=0 ; i<n ; i++)
		{
			scanf("%s" , &s);
			for(int j=0 ; j<strlen(s) ; j++)
				a[s[j]-'a'][i]++;
		}
		for(int i=0 ; i<26 ; i++)
		{
			int y=minimum(n , i);
			if(y) flag=1;
			for(int j=0 ; j<y; j++) printf("%c" , i+97);
	    }
	   
		if(flag==0) printf("no such string\n");
		else printf("\n");
	
	return 0;
}
