#include<iostream>
using namespace std;
int main()
{
	long int n,i,j,flag,n1,flag1=0;
	char s[105],c;
	cin>>n;
	long int a[n][26];
	for(i=0;i<n;i++)
	 for(j=0;j<26;j++)
	  a[i][j]=0;n1=n;j=0;
	while(j<n)
	{
		cin>>s;
		for(i=0;s[i]!='\0';i++)
		 a[j][s[i]-97]++;
		j++;
	}
	for(i=0;i<26;i++)
	{
		flag=0;
		for(j=0;j<n1;j++)
		 if(a[j][i])
		  flag++;
		if(flag==n1)
		{
			flag1++;
			n=a[0][i];
			for(j=0;j<n1;j++)
			 if(n>a[j][i])
			  n=a[j][i];
			c=i+97;
			for(j=0;j<n;j++)
			 cout<<c;
		}
	}
	if(!flag1)
		cout<<"no such string";
		cout<<endl;
	return 0;
}
