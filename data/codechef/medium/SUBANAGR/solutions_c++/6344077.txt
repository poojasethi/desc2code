#include <iostream>
#include <cstring>
using namespace std;

int main() {
    int tc, i, j, k;
    string r, s;
    int countr[26], counts[26];
    bool zero;

    memset(counts, 0, sizeof counts);

    cin >> tc >> s;
    tc--;
    for (i = 0; i < s.length(); i++) {
        counts[s[i] - 97]++;
    }

    while (tc--) {
        cin >> r;

        memset(countr, 0, sizeof countr);
        for (i = 0; i < r.length(); i++) {
            countr[r[i] - 97]++;
        }

        for (i = 0; i < 26; i++) {
            counts[i] = min(counts[i], countr[i]);
        }
    }

    zero = true;
    for (i = 0; i < 26; i++) {
        while (counts[i]--) {
            cout << (char)(i + 97);
            zero = false;
        }
    }

    if (zero) {
        cout << "no such string";
    }
    cout << endl;

    return 0;
}
