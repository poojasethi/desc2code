    #include <iostream>
    #include <cstdio>
    using namespace std;
    char s[101];
    int Freq[26];
    int ans[26];
    int main()
    {
        int n;
        for(int i = 0 ; i < 26 ; i++)
        {
            Freq[i]=0;
        }
        for(int i = 0 ; i < 26 ; i++)
        {
            ans[i]=0;
        }
        cin >> n >> s;
        for(int i=0; s[i]!=0; i++)
        {
            ans[ s[i]-'a' ]++;
        }
        n--;
        while( n-- )
        {
            cin >> s;
            for(int i = 0; i < 26 ; i++)
            {
                Freq[i]=0;
            }
            for(int i = 0 ; s[i] != 0 ; i++)
            {
                Freq[ s[i]-'a' ]++;
            }
            for(int i = 0; i < 26 ; i++)
            {
                ans[i] = min(ans[i], Freq[i]);
            }
        }
        bool b=false;
        for(int i=0; i<26; i++)
        {
            for(int j=0; j<ans[i]; j++)
            {
                b=true;
                cout <<  (char)(i+'a');
            }
        }
        if(!b) cout << "no such string";
        return 0;
    }
