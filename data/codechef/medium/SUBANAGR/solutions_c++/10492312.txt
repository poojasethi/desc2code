#include<bits/stdc++.h>
using namespace std;

int main()
{
int num,i,j,minsum,count;
bool flag=true,execute=false;
string str;
scanf("%d",&num);
int array[num][26];
memset(array,0,sizeof(array));
for(i=0;i<num;++i)
{
    cin>>str;
    for(j=0;str[j]!='\0';++j){
        ++array[i][str[j]-97];
    }
}
for(i=0;i<26;++i)
{   flag=true;
    minsum=101;
 for(j=0;j<num;++j)
 {      count=0;
     if(array[j][i]){
    while(array[j][i]--)
    ++count;
    }
    else
    {
        flag=false;
        break;
    }
    minsum=min(count,minsum);
    //printf("( %d %d)",minsum,count);
 }
 while(minsum-- && flag)
 {
     printf("%c",i+97);
     execute=true;
 }

}
if(!execute)
    printf("no such string");
return 0;
}
