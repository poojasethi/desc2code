#include <cstdlib>
#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <vector>
#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

using namespace std; 

typedef struct Item{
	int id;
	int prev_count;
	int curr_count;
}item;

int main(){
	int T;
	scanint(T);
	item a[26];
	string str;
	int temp;
	for(int i = 0; i < 26; i++){
		a[i].id = 0;
		a[i].prev_count = 101;
		a[i].curr_count = 0;
	}
	for(int t = 0; t < T; t++){
		cin >> str;
		for(int i = 0; i < 26; i++){
			a[i].curr_count = 0;	
		}
		for(int i = 0; i < str.length(); i++){
			temp = str[i]-97;
			if(a[temp].id == t){
				a[temp].id = t+1;
				a[temp].curr_count = 1;
			}
			else if(a[temp].id == t+1){
				if(a[temp].curr_count < a[temp].prev_count){
					a[temp].curr_count++;
				}
			}
		}
		for(int i = 0; i < 26; i++){
			a[i].prev_count = a[i].curr_count;
		}
	}
	bool flag = false;
	for(int i = 0; i < 26; i++){
		if(a[i].id == T){
			flag = true;
			for(int j = 0; j < a[i].curr_count; j++)
				cout << (char)(i+97);
		}
	}	
	if(!flag){
		cout << "no such string";
	}
	cout << endl;	
}
