#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int j,t,i,a[101][26],b[26]={0},min,p;
	char str[101];
	cin>>t;
	
	for(j=0;j<t;j++)
	{
		for(i=0;i<26;i++)
	a[j][i]=0;
		cin>>str;
		for(i=0;str[i]!='\0';i++)
		{
			a[j][str[i]-'a']++;
		}
		
	}
	for(i=0;i<26;i++)
	{
		min=a[0][i];
		for(j=1;j<t;j++)
		{
			//cout<<"a :"<<a[j][i]<<endl;
			if(a[j][i]<min)
			min=a[j][i];
		}
		//cout<<"j : "<<j<<endl;
		if(min)
		b[i]=min;
	}
	p=1;
	for(i=0;i<26;i++)
	{
		for(j=0;j<b[i];j++)
		{
			cout<<char(i+97);
			p=0;
		}
	}
	if(p)
	cout<<"no such string";
	//cout<<"\n";
	return 0;
}
