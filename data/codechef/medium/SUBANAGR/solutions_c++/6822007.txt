#include <iostream>
#include <map>
#include <vector>
using namespace std;
typedef long long LL;

LL has[200]={0};
LL min(LL a, LL b) {
    return a<b ? a : b;
}
int main() {
    LL n=0,i=0,j=0;
    string str;
    cin>>n;
    cin>>str;
    for(j=0;j<str.length();j++) {
        has[str[j]]++;
    }
    
    for(i=1;i<n;i++) {
        cin>>str;
        LL temphas[200]={0};
        for(j=0;j<str.length();j++) {
            temphas[str[j]]++;
        }
        
        for(j=97;j<=122;j++) {
            has[j]=min(has[j], temphas[j]);
        }
    }
    LL flag=0;
    for(j=97;j<=122;j++) {
        if(has[j]>0) {
            for(i=1;i<=has[j];i++) {
                printf("%c",j);
            }
            flag=1;
        }
    }
    if(flag==0) {
        cout<<"no such string";
    }
    cout<<endl;
    return 0;
}
