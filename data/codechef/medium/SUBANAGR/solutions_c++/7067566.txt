#include <iostream>
#include <vector>
#include <string>
using namespace std;
typedef long long ll;
ll findno(string name, char c)
{
	ll i;
	ll cnt=0;
	for(i=0;i<name.length();i++)
	{
		if(name[i]==c) cnt++;
	}
	return cnt;
}
int main()
{
	ll n,i,no,temp;
	scanf("%lld",&n);
	string word;
	vector<string> list;
	while(n--)
	{
		cin>>word;
		list.push_back(word);
	}
	string ans;
	for(char c=97;c<=122;c++)
	{
		no=100000;
		for(i=0;i<list.size();i++)
		{
			temp=findno(list[i],c);
			if(temp<no) no=temp;
		}
		for(i=0;i<no;i++)
		{
			ans=ans+c;
		}
	}
	if(ans!="")
		cout<<ans;
	else
		printf("no such string");
	return 0;
}
