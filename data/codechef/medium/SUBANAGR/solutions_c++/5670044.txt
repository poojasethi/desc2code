#include<bits/stdc++.h>
using namespace std;
int main()
 {
     char a[101];
     int n,i,flag=0;
     scanf("%d",&n);
     int b1[26]={},b2[26];
     for(i=0;i<26;i++)
     {
         b2[i]=100;
     }
     while(n--)
     {
         scanf(" %s",a);

          for(i=0;i<strlen(a);i++)
          {
              b1[a[i]-'a']+=1;
          }
         for(i=0;i<26;i++)
             {

              b2[i]=min(b1[i],b2[i]);
              //cout<<b2[i]<<" ";
             }
         for(i=0;i<26;i++)
            b1[i]=0;

     }
     for(i=0;i<26;i++)
     {
         while(b2[i]!=0)
         {
            flag=1;
            printf("%c",(i+'a'));
             b2[i]--;
         }
     }
     if(flag==0)
        printf("no such string");

 return 0;
 }
