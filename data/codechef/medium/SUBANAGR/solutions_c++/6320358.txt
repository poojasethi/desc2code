#include<bits/stdc++.h>

using namespace std;

char str[102][102];
int freq[102][27];
int minfreq[27];
int main()
{
	long long int n,i,j,k,val;
	scanf("%lld\n",&n);
	for(i=0;i<n;i++)
	{
		scanf("%s",str[i]);
	}
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<26;j++)
		{
			freq[i][j]=0;
		}
	}

	for(i=0;i<n;i++)
	{
		for(j=0;j<strlen(str[i]);j++)
		{
			val=(int)str[i][j];
			freq[i][val-97]++;
		}
	}

	long long int minval;
	long long int flag=0;
	for(i=0;i<26;i++)
	{
		minval=200;

		for(j=0;j<n;j++)
		{
			if(freq[j][i]<minval)
				minval=freq[j][i];
		}
		if(minval>0)
			flag=1;
		for(j=0;j<minval;j++)
		{
			printf("%c",(char)(i+97));
		}
	}
	if(flag==0)
		printf("no such string");
	
	printf("\n");
		return 0;
}
