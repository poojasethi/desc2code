#include <bits/stdc++.h>
 using namespace std;

int num[26];
int main(){
    int n; cin >> n;
	for(int i=0 ; i<26 ; i++) num[i] = 1000000000;
	for(int i=0 ; i<n ; i++){
		string s; cin >> s;
		int temp[26]={0};
		for(int j=0 ; j<s.size() ; j++) temp[s[j]-'a']++;
		for(int j=0 ; j<26 ; j++) num[j] = min(num[j],temp[j]);
	}
	bool flag = true;
	for(int i=0 ; i<26 ; i++)
		if(num[i]>0) for(int j=0 ; j<num[i] ; j++) {cout << (char)('a'+i); flag = false;}
	if(flag) cout << "no such string";
	cout << endl;
	return 0;
}
