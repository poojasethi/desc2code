#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    string s;
    int n,k=0,i,m,flag,j;
    int a[200][27];
    scanf("%d",&n);
    for(i=0;i<n;i++)

    {
        cin>>s;
    for(j=0;j<s.length();j++)
    {
        a[k][s[j]-97]++;


    }
    k++;


    }
      string o="";
    for(i=0;i<=25;i++)
    {
        flag=1,
        m=999999999;
       for(j=0;j<k;j++)
       {
            if(a[j][i]<=0)
              flag=0;
            else if(m>a[j][i])
               m=a[j][i];
       }
       if(flag==1)
       {
           o+=string(m, 97+i);
       }
    }
    if(o!="")
    cout<<o<<endl;
    else

}
