#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
using namespace std;
int main()
{
    int i,n,l,j,mini;
    char ch;
    char s[101];
    scanf("%d",&n);
    vector<vector<int> > arr(n,vector<int>(26));
    vector<int> v(26);
    for(i=0;i<n;i++)
    {
        scanf("%c",&ch);
        scanf("%s",s);
        l=strlen(s);
        for(j=0;j<l;j++)
        {
           arr[i][s[j]-97]++;
        }
    }
     for(j=0;j<26;j++)
     {
        mini=200;
        for(i=0;i<n;i++)
        {
            if(mini>arr[i][j])
                mini=arr[i][j];
        }
        v[j]=mini;
    }
    int flag=0;
    for(i=0;i<26;i++)
    {
        if(v[i]!=0)
        {
            flag=1;
            break;
        }
    }
    if(flag)
    {
        for(i=0;i<26;i++)
        {
            for(j=0;j<v[i];j++)
            {
                printf("%c",i+97);
            }
        }
    }
    else
    {
        printf("no such string");
    }
    return 0;
}
