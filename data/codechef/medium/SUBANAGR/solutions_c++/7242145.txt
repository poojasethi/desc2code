#include<iostream>
#include<cstring>
using namespace std;
int main() {
	int n;int a;
	cin>>n;
	int arr[101][26]={0};int min;
	int i;int j;
	for(i=0;i<n;i++)
	{
		j=0;
		char str[101];
		cin>>str;
		while(str[j]!='\0')
		{
		a=(int)str[j];
		arr[i][a-97]++;
		j++;
			
			
			
		}
	}
	char h[200];int k=0;
	for(i=0;i<26;i++)
	{
		min=arr[0][i];
		for(j=1;j<n;j++)
		{
			if(min>arr[j][i])
			min=arr[j][i];
			
		}
		int g;
		for(g=0;g<min;g++)
		{
			h[k]=(char)(i+97);
			k++;
		}
		
	}
	h[k]='\0';
	if(strlen(h)==0)
	cout<<"no such string";
	else
	cout<<h;
	return 0;
}
