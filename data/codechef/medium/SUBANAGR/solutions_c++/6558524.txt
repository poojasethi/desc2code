#include<iostream>
#include<string.h>

using namespace std;

int findMin(int a, int b)
{
    if(a <= b)
        return a;
    return b;
}

int main()
{
    int n, i, l, ans[26], j, c[26], flag = 0;
    char a[102];

    cin>>n;
    //memset(ans, 0, sizeof(int)*26);
    for(i = 0; i < 26; i++)
    {
        ans[i] = 101;
    }

    for(i = 0; i < n; i++)
    {
        cin>>a;
        l = strlen(a);
        memset(c, 0, sizeof(int)*26);

        for(j = 0; j < l; j++)
        {
            c[(int)a[j]-97]++;
            //cout<<c[(int)a[j]-97]<<"\n";
        }

        for(j = 0; j < 26; j++)
        {
            ans[j] = findMin(c[j], ans[j]);
        }
    }

    for(i = 0; i < 26; i++)
    {
        while(ans[i]--)
        {
            cout<<(char)(i+97);
            flag = 1;
        }
    }
    if(flag == 0)
    {
        cout<<"no such string";
    }
    cout<<"\n";
    return 0;
}
