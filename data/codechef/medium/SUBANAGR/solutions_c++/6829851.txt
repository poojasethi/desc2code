#include <bits/stdc++.h>

using namespace std;

int main()
{
    int c[100][26],n,i,l,j,p,k,d;
    string x;
    char s[101];
    cin>>n;
    memset(c,0,sizeof(c));
    for(int y=0;y<n;y++)
    {
        cin>>x;
        l=x.length();
        for(i=0;i<l;i++)
            c[y][x[i]-97]++;
    }
    k=0;
    for(i=0;i<26;i++)
    {
        p=32000;
        for(j=0;j<n;j++)
        {
            if(c[j][i]<p)
                p=c[j][i];
        }
        for(d=0;d<p;d++)
            s[k++]=(char)(i+97);
    }
    s[k]='\0';
    if(strlen(s)>0)
        cout<<s;
    else
        cout<<"no such string";
    return 0;
}
