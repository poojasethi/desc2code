#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
#define gc getchar_unlocked
void scanint(int &x);

int main()
{
    int nVal;
    scanint(nVal);
    char input[nVal][101];
    memset(input,'\0',nVal*101);
    for(int i=0;i<nVal;++i)
        scanf("%s",input[i]);
    int charMap[nVal][26];
    for(int i=0;i<nVal;++i)
    {
        for(int j=0;j<26;++j)
            charMap[i][j]=0;
    }
    for(int i=0;i<nVal;++i)
    {
        for(int j=0;j<strlen(input[i]);++j)
            charMap[i][input[i][j]-'a']++;
    }
    int displayList[26]={0};
    int flag=0;
    bool found=false;
    for(int i=0;i<26;++i)
    {
        flag=1;
        for(int j=0;j<nVal;++j)
        {
            if(charMap[j][i]==0){flag=0;break;}
        }
        if(!flag)continue;
        found=true;
        int maxVal=999;
        for(int j=0;j<nVal;++j)
                if(maxVal > charMap[j][i]) maxVal=charMap[j][i];
        displayList[i]=maxVal;       
    }
    if(!found)
        printf("no such string");
    else
    for(int i=0;i<26;++i)
    {
        if(displayList[i])
            for(int j=0;j<displayList[i];++j)
                printf("%c",i+'a');
    }
    
    return 0;
}

{
        int flag=0;

	for(;(c<48 || c>57);c = gc());
