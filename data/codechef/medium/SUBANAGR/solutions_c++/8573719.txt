inline int get_int()
{
  int n=0,neg=0,c=getchar_unlocked();
  while((c<'0'||c>'9')&&c!='-')c=getchar_unlocked();
  if(c=='-') {neg=1;c=getchar_unlocked();}
  while(c>='0'&&c<='9')n=(n<<3)+(n<<1)+c-'0',c=getchar_unlocked();
  if(neg) n=-n;
  return n;
}
int a[26],b[26];
int main()
{
  pre();
  int n,i;
  cin>>n;
  for(i=0;i<26;i++)
    a[i]=103;
  while(n--)
  {
      string str;
      cin>>str;
      memset(b,0,sizeof(b));
      for(i=0;str[i]!='\0';i++)
      {
          b[str[i]-'a']++;
      }
      for(i=0;i<26;i++)
        if(b[i]<a[i])
          a[i]=b[i];
  }
  int index=0;
  for(i=0;i<26;i++)

    for(int j=0;j<a[i];j++)
      {

      }
  }

  else cout<<endl;

}
