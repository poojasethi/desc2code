string f( string a, string b )
{
	int i = 0, j = 0;
	string ret = "";
	while( i < a.size() && j < b.size() )
		if( a[i] == b[j] )
			ret += a[i], i ++, j ++;
		else if( a[i] < b[j] )
			i ++;
		else
			j ++;
	return ret;
}

int main()
{
	int n;
	string s, ans = "";
	cin >> n;
	for( int i = 0 ; i < n ; i ++ )
	{
		cin >> s;
		sort( s.begin(), s.end() );
		if( i )
			ans = f( ans, s );
		else
			ans = s;
	}
	if( ans == "" )
		ans = "no such string";
	cout << ans << endl;
	return 0;
}
