#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j;
    cin>>n;
    vector<int> v[n];
    char s[101];
    for(i=0;i<n;i++)
    {
        for(j=0;j<26;j++)
            v[i].push_back(0);
    }
    for(j=0;j<n;j++)
    {
        cin>>s;
        for(int k=0;s[k]!='\0';k++)
        {
            int x=s[k]-'a';
            v[j][x]++;
        }
    }
    int flag=0;
    for(j=0;j<26;j++)
    {
        int minn=v[0][j];
        for(i=1;i<n;i++)
        {
              if(v[i][j]<minn)
                minn=v[i][j];
        }
        char p='a'+j;
        for(int l=0;l<minn;l++)
        {
            cout<<p;
            flag=1;
        }
    }
    if(flag==0)
        cout<<"no such string"<<endl;
    cout<<endl;
    return 0;
}
