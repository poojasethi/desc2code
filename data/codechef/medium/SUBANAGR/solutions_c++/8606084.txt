#include <bits/stdc++.h>

using namespace std;

int main()
{
	int i,j,k,n;
	cin >> n;
	string a[n];
	string op;
	
	int d[n][26];
	
	for(i=0; i<n; i++)
	{
		for(j =0; j<26;j++)
			{
				d[i][j] =0;
			}
	}
	
	for(j = 0; j < n; j++)
	{	
		cin >> a[j];
		
		i =0;
		while(a[j][i]!='\0')
		{	
			d[j][a[j][i] - 'a']++;
			i++;
		}
	}
	
	/*for(i=0; i<n; i++)
	{
		for(j =0; j<26;j++)
			cout << d[i][j] <<" ";
		
		cout << endl;
	}
	*/
	k =0;
	for(i =0; i< 26;i++)
	{
		int count =0;
		int min = 100000000;
		
		for(j=0;j<n;j++)
		{
			if(d[j][i] == 0)
				break;
				
			else
			{
				if(min>d[j][i])
					min = d[j][i];
					
				count++;
			}
		}
		
		if(count == n)
		{
			while(min--){
				op[k] = 'a'+i;
				cout << op[k];
				k++;
			}
		}
	}
	
	if(op[0]=='\0')
		cout << "no such string";
}
