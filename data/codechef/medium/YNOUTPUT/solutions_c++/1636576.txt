bool cmp(bool *n, bool *m, int t)
{
     //printf("Entered cmp\n");
     int i=0;
     for(;i<t;i++)
     {
                  if(n[i] != m[i])
                          break;
     }
     //printf("exited cmp\n");
     if(i==t)
             return 0;
     else
         return 1;
}
