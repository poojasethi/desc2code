#include      <string>
#include      <iostream>
#define endl '\n'
using namespace std;
string states[101];
int main(int argc, char **argv) {
	ios_base::sync_with_stdio(false), cin.tie(0);
	int T, i, j;
	cin >> T;
	string strTmp;
	for (i = 0; i < T; i += 1) {
		for (j = 0; j < T; j += 1) { cin >> strTmp; states[i].push_back(strTmp[0]); }
	}
	bool ok = false;
	for (i = 0; i < T && !ok; i += 1) { // my
		ok = true;
		for (j = 0; j < T; j += 1) ok = ok && (states[i][j] == 'Y' ? states[i] == states[j] : states[i] != states[j]);
	}
	if (ok) {
		i -= 1;
		for (j = 0; j < T; j += 1) cout << (states[i][j] == 'Y' ? "YES" : "NO") << endl;
	} else for (j = 0; j < T; j += 1) cout << "NO" << endl;
	return 0;
}
