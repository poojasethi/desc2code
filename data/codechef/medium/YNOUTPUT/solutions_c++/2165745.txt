#include<iostream>
#include<cstring>

using namespace std;

int main()
{
	int t,i,j,k,temp,count1,count2;
	char mat[10005][105];
	char op[105][105];
	scanf("%d",&t);
	temp=t*t;
	count1=0;
	for(i=1;i<=temp;i++)
		scanf("%s",&mat[i]);
	for(i=1;i<=t;i++)
	{
		count2=0;
		for(j=1;j<=t;j++)
		{
			if(strcmp(mat[j+count1],"YES")==0)
			{	op[j][0]='Y';op[j][1]='E';op[j][2]='S';op[j][3]='\0';}
			else
			{	op[j][0]='N';op[j][1]='O';op[j][2]='\0';}
		}
		for(j=1;j<=t;j++)
		{
			for(k=1;k<=t;k++)
			{
				if(strcmp(op[k],mat[k+count2])!=0)
					break;
			}
			if(k>t)
			{
				if(strcmp(op[j],"YES")!=0)
					break;
			}
			else
			{
				if(strcmp(op[j],"NO")!=0)
					break;
			}
			count2+=t;
		}
		count1+=t;
		if(j>t)
			break;
	}
	if(i>t)
	{
		for(i=1;i<=t;i++)
		printf("NO\n");
	}
	else
	{
		for(i=1;i<=t;i++)
			printf("%s\n",op[i]);
	}
	return 0;
}
