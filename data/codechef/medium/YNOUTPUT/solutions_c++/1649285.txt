using namespace std;

int main()
{
	int T;
	scanf("%d", &T);
	string arr[T][T];
	for (int i = 0; i < T; i++)
	{
		for (int j = 0; j < T; j++)
			cin >> arr[i][j];
	}
	int pos = -1;
	for (int i = 0; i < T; i++)
	{
		bool c = true;//this is a solution
		if (arr[i][i] == "YES")
		{
			for (int j = 0; j < T; j++)
			{
				if (j == i)
					continue;
				bool b = true;//this matches
				for (int k = 0; k < T; k++)
				{
					if (arr[i][k] != arr[j][k])
					{
						b = false;
						break;
					}
				}
				if (b && arr[i][j] == "YES")
					;
				else if (!b && arr[i][j] == "NO")
					;
				else
				{
					c = false;
					break;
				}
			}
		}
		else
			c = false;
		if (c)
			pos = i;
	}
	if (pos == -1)
	{
		for (int i = 0; i < T; i++)
			printf("NO\n");
	}
	else
	{
		for (int i = 0; i < T; i++)
			cout << arr[pos][i] << endl;
	}
	return 0;
}
