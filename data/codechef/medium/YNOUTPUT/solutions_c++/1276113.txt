#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
       int t;
       cin>>t;
       char c[t][t];
       for(int i=0;i<t;i++)
       {      for(int j=0;j<t;j++)
       		  		  {     char d[5];
       		  		        cin>>d;
					  	 	if((d[0]=='Y')||(d[0]=='y'))
                                  c[i][j]='Y';
					  	 	else  c[i][j]='N';
  			          }
        }
        int f=0;
        for(int i=0;i<t;i++)
        {    if(c[i][i]=='Y')
             {       f=1;
			 		 for(int j=0;j<t;j++)
                     {       if(f==1)
                             {    int y=0;
					   		     for(int k=0;k<t;k++)
								 {     if(c[i][k]!=c[j][k])
								              {  y=1;
   			                                  }
							     }
							     if(((y==1)&&(c[i][j]=='Y'))||((y==0)&&(c[i][j]=='N')))
							     {            f=0;
							                  j=t;
								 }            
							}        
	   		         }       
             }
             if(f==1)
             {         for(int k=0;k<t;k++)
                        {   if(c[i][k]=='Y')
						            cout<<"YES\n";
	                        else cout<<"NO\n";
		 	           }
			   		      i=t;
			   
			 }
		}
		if(f==0)
		    for(int i=0;i<t;i++)
		         cout<<"NO\n";
  
	  // system("pause");
	    return 0;
        
}
