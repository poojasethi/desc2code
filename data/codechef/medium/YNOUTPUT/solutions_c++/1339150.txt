#include <iostream>
#include <vector>
#include <cmath>
using namespace std;

int main()
{
  int T; cin >> T;
  vector< vector<int> > v(T, vector<int>(T, 0));
  string s;
  for (int i = 0; i < T; ++i)
    for (int j = 0; j < T; ++j)
    {
      cin >> s;
      v[i][j] = (s == "YES");
    }
  
  vector<int> ret(T, 0);
  for (int i = 0; i < T; ++i)
  {
    bool ok = 1;
    for (int j = 0; j < T; ++j)
      if ((v[i][j] == 1 and v[i] != v[j]) or
          (v[i][j] == 0 and v[i] == v[j]))
      {
        ok = 0; break;
      }
    if (ok)
    {
      ret = v[i]; break;
    }
  }
  
  for (int i = 0; i < T; ++i)
    if (ret[i]) cout << "YES" << endl;
    else cout << "NO" << endl;
}
