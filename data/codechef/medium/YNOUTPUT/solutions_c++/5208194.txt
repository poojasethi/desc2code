#include <bits/stdc++.h>
using namespace std;
int main(){
	int t,te,i,j,k;
	cin>>t;
	string s,arr[t];
	for(i=0;i<t;i++)arr[i]="";
	for(i=0;i<t;i++)
		for(j=0;j<t;j++){
			cin>>s;
			if(s=="YES")arr[i]+="1";
			else arr[i]+="0";
		}
	vector<int>v[t],nv[t];
	for(i=0;i<t;i++){
		for(j=0;j<t;j++){
			if(arr[i]==arr[j])v[i].push_back(j);
			else nv[i].push_back(j);
		}
	}
	for(i=0;i<t;i++){
		if(arr[i][i]=='0')continue;
		for(j=0;j<v[i].size();j++)if(arr[i][v[i][j]]=='0')break;
		if(j<v[i].size())continue;
		for(j=0;j<nv[i].size();j++)if(arr[i][nv[i][j]]=='1')break;
		if(j<nv[i].size())continue;
		break;
	}
	if(i<t){
		for(j=0;j<t;j++)if(arr[i][j]=='1')cout<<"YES\n";
		else cout<<"NO\n";
	}
	else
		for(j=0;j<t;j++)cout<<"NO\n";
	return 0;
}
