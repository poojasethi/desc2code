#include<iostream>
#include<stdio.h>
#include<string.h>

int equals(int a[100][100],int p,int q,int t)
{
    int i;
    for(i=0;i<t;i++)
    {
        if(a[p][i]!=a[q][i])
        {
            return 0;
        }
    }
    return 1;
}

int main()
{
    char array[100][100];
    int t,i,j,k,flag=0;
    char x[5];
    scanf("%d",&t);

    for(i=0;i<t;i++)
    {
        for(j=0;j<t;j++)
        {
            scanf("%s",x);
            if(x[0]=='Y'||x[0]=='y')
            {
                array[i][j]='1';
            }
            else
            {
                array[i][j]='0';
            }
        }
        array[i][j]='\0';
    }


    for(i=0;i<t;i++)
    {
        flag = 1;
        for(j=0;j<t;j++)
        {
            if(!strcmp(array[i],array[j])&&array[i][j]=='1')
            {
                flag = 1;
            }
            else if(strcmp(array[i],array[j])&&array[i][j]=='0')
            {
                flag = 1;
            }
            else
            {
                flag = 0;
                break;
            }
        }
        if(flag==1)
        {
            if(array[i][i]=='1')
                {
                    printf("YES\n");
                }
                else
                {
                    printf("NO\n");
                }
        }
        else
        {
            printf("NO\n");
        }
    }

    return 0;
}
