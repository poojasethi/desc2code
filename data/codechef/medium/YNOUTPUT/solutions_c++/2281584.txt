#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
	int t,i,j,k;
	char s[20];
	scanf("%d",&t);
	bool b[101],ans[101],a[101][101]={0},ch=0;
	for(i=0;i<t;i++)
	{
		for(j=0;j<t;j++)	
		{
			scanf("%s",s);
			if(strcmp(s,"NO")==0) continue;
			else a[i][j]=1;
		}
	}
	for(i=0;i<t;i++)
	{
		memset(ans,0,sizeof(ans));		
		for(j=0;j<t;j++)
		{
			for(k=0;k<t;k++)
			{
				if(a[i][k]==a[j][k]) ans[j]=1;
				else{ans[j]=0;break;} 
			}
		}
		ch=0;
		for(j=0;j<t;j++) 
		{
			if(a[i][j]!=ans[j]) {ch=1;break;}
		}
		if(!ch) break;
	}
	if(!ch)
	{
		for(i=0;i<t;i++)
		{
			if(ans[i]) printf("YES\n");
			else printf("NO\n");
		}
	}
	else for(i=0;i<t;i++) printf("NO\n");
	return 0;
}
