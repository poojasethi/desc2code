char s[101][101];

char compare_func(int i,int j)
{
    for(int r=0;r<T;r++)
       if(s[i][r] != s[j][r])            
            return '0';
   return '1';           
}

void print(int i)
{
    for(int r=0;r<T;r++)
      if(s[i][r]=='0')
        cout<<"NO"<<endl;
      else cout<<"YES"<<endl; 
}

int main()
{
    
    scanf("%d",&T);
    for(int i=0;i<T;i++)
     for(int j=0;j<T;j++)

         if(temp.compare("YES")==0)
           s[i][j]='1';
         else s[i][j]='0';

      }
