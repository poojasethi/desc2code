#include<stdio.h>
int main()
{
     char temp;
     int ans,t,i,j,k;
     bool inp[100][100],op[100]={0};
     scanf("%d",&t);
     scanf("%c",&temp);
     for(i=0;i<t;i++)
     {
          for(j=0;j<t;j++)
          {
               scanf("%c",&temp);
               if(temp=='N'){
                    inp[i][j]=0;
                    scanf("%c",&temp);
                    scanf("%c",&temp);}
               else{
                    inp[i][j]=1;
                    scanf("%c",&temp);
                    scanf("%c",&temp);
                    scanf("%c",&temp);}
               if(i==j && inp[i][j]==1)
                    op[i]=1;
          }
     }
     for(k=0;k<t;k++)
     {
          bool f2 = 1;
          if(op[k])
          {
               bool f1 = 1;
               for(i=0;i<t;i++)
               {
                    if(inp[k][i])
                    {
                         bool f=1;
                         for(j=0;j<t;j++)
                         {
                              if(inp[k][j] != inp[i][j])
                              {
                                   f=0;
                                   break;
                              }
                         }
                         if (!f)
                         {
                              f1=0;
                              break;
                         }
                    }
                    else
                    {
                         bool f=1;
                         for(j=0;j<t;j++)
                         {
                              if(inp[k][j] != inp[i][j])
                              {
                                   f=0;
                                   break;
                              }
                         }
                         if (f)
                         {
                              f1=0;
                              break;
                         }
                    }
               }
               if(!f1)
               {
                    f2=0;
               }
          }
          if(f2 && op[k])
          {
               for(j=0;j<t;j++)
               {
                    if(inp[k][j])
                         printf("YES\n");
                    else
                         printf("NO\n");
               }
               return 0;
          }
     }
     for(j=0;j<t;j++)
     {
          printf("NO\n");
     }
     return 0;
}
