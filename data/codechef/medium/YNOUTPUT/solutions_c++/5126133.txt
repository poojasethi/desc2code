#include<stdio.h>
#include<string.h>
#include<stdio.h>
int s[105][105];
char STR[6];
int main()
{
    int t,i,j,k;
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
        for(j=0;j<t;j++)
        {
            scanf("%s",STR);
            if(STR[0]=='Y')
            s[i][j]=1;
            else
            s[i][j]=0;
        }
    }
    for(i=0;i<t;i++)
    {
        for(j=0;j<t;j++)
        {


                for(k=0;k<t;k++)        //compare s[i] s[j]
                if(s[i][k]!=s[j][k])
                break;
                if(k==t&&s[i][j]==0)        //if equal
                break;
                if(k!=t&&s[i][j]==1)        //if not equal
                break;

        }
        if(j==t)
        break;
    }
    //printf("%d",i);
    if(i<t)             //IF ANSWER FOUND
    {
        for(j=0;j<t;j++)
        {
            if(s[i][j]==1)
            printf("YES\n");
            else
            printf("NO\n");
        }

    }                   //IF NOT FOUND
      else
        {
            for(j=0;j<t;j++)
        {

            printf("NO\n");
        }
        }
    return 0;
}
