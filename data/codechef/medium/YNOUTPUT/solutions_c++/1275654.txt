#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 1001

int n,i,j,k;
char str[10];
bool A[lim][lim];

int main()
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    {
      scanf("%s",str);
      A[i][j]=str[0]=='Y';
    }
  for(i=0;i<n;i++)
  {
    for(j=0;j<n;j++)
    {
      for(k=0;k<n && A[j][k]==A[i][k];k++);
      if(k==n!=A[i][j])
        break;
    }
    if(j==n)
      break;
  }
  for(j=0;j<n;j++)
    printf("%s\n",A[i][j]?"YES":"NO");
  return 0;
}
