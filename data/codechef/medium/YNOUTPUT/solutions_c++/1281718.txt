#include<stdio.h>

int solve(int i,int arr[102],int input[102][102],int test_cases,int ouput[102] );

int main()
{
	int test_cases;
	
	scanf("%d",&test_cases);
	int input[102][102];
	char in[4];
	int output[102];
	for(int i=0;i<test_cases;i++)
	{
		for(int j=0;j<test_cases;j++)
		{
			scanf("%s",in);
			in[0]=='Y'?input[i][j]=1:input[i][j]=0;	
		}
	}

	int flag=0;
	int arr[102];
	for(int i=0;i<test_cases;i++)
	{	
		for(int j=0;j<test_cases;j++)
		{	
			arr[j]=input[i][j];
		}
		if(solve(i,arr,input,test_cases,output))
		{
			flag=1;break;
		}
	}
	for(int i=0;i<test_cases;i++)
		//printf("%s\n",arr[i]);
		printf("%s\n",flag?arr[i]?"YES":"NO":output[i]?"YES":"NO");
}

int solve(int i,int arr[102],int input[102][102],int test_cases,int output[102])
{ 
	
	for(int j=0;j<test_cases;j++)
	{
		if(j==i)
		{
			if(!(arr[j]==1))
			{
				output[i]=0;
				return 0;
			}
				
		}
		else
		{
			int flag=0;
			for(int k=0;k<test_cases;k++)
				if(!(input[j][k]==arr[k]))
				{
					flag=1;
					break;
				}
				
			if(flag && arr[j]==0)
				continue;
			else if(!flag && arr[j]==1)
				continue;
			output[i]=0;
			return 0;
		
		}
	}
	return 1;
}
