#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int T; scanf("%d",&T);
	char s[100][100][4];
	rep(i,T) rep(j,T) scanf("%s",s[i][j]);

	int ans=-1;
	rep(i,T){
		bool ok=true;
		rep(j,T){
			bool match=true;
			rep(k,T) if(s[j][k][0]!=s[i][k][0]) match=false;
			if(match!=(s[i][j][0]=='Y')) ok=false;
		}
		if(ok) ans=i;
	}

	if(ans!=-1){
		rep(i,T) puts(s[ans][i]);
	}
	else{
		rep(i,T) puts("NO");
	}

	return 0;
}
