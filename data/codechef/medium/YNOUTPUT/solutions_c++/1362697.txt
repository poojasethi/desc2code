#include<iostream>
#include<cstdio>

using namespace std;

int main(){
	int T;
	scanf("%d",&T);
	int t=0;
	bool testcases[T][T];
	bool output[T];
	for(int i=0; i<T; i++) output[i]=true;
	char str[4];
	while(t<T){
		for(int i=0; i<T; i++){
			scanf("%s",str);
			if(str[0]=='Y') testcases[t][i]=true;
			else testcases[t][i]=false;
			if(i==t) output[T]=testcases[i][i];
		}
		t++;
	}
	bool matches;
	int index = -1;
	bool correct_choice;
	
	for(int i=0; i<T; i++){
		//Assume that testcases[i] is the output
		correct_choice = true;
		for(int j=0; j<T; j++){
			if(testcases[i][j]==true){	//Therefore, testcases[i] should matche testcases[j]
				matches = true;
				for(int k=0; k<T; k++){
					if(testcases[i][k]!=testcases[j][k]){
						matches = false;
						break;
					}
				}
				if(!matches){
					correct_choice = false;
					break;
				}
			}
			else{
				matches = true;
				for(int k=0; k<T; k++){
					if(testcases[i][k]!=testcases[j][k]){
						matches = false;
						break;
					}
				}
				if(matches){
					correct_choice = false;
					break;
				}
			}
		}
		if(correct_choice){
			index = i;
			break;
		}
	}
	if(index==-1){
		for(int i=0; i<T; i++){
			printf("NO\n");
		}
	}
	else{
		for(int i=0; i<T; i++){
			if(testcases[index][i]==true) printf("YES\n");
			else printf("NO\n");
		}
	}
	return 0;
}
