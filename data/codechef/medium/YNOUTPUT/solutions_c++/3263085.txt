#include <stdio.h>
#include <sstream>
#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
class YNOUTPUT
{
public:
	void solve(){
		int t;
		string testcases[101];
		string str1;
		string temp;
		scanf("%d",&t);
		for(int i=0;i<t;i++){
			//testcases[i]="";
			for(int j=0;j<t;j++){
				cin>>temp;
				testcases[i]=testcases[i]+temp;
			}

		}
		for(int i=0;i<t;i++){
			string str1 = testcases[i];
			string str2="";
			for(int j=0;j<t;j++){
				if((string)testcases[j]==(string)(str1))
					str2=str2+"YES";
				else
					str2=str2+"NO";
			}
			if((string)str1==(string)str2)
				printf("YES\n");
			else
				printf("NO\n");
		}

	}
};
int main(int argc, char const *argv[])
{
	YNOUTPUT obj;
	obj.solve();
	return 0;
}
