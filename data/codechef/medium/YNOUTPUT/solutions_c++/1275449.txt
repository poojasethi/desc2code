#include <cstdio>
#include <string>
using namespace std;
int n,i,j,k;
string a[101][101];
char s[7];
int main() {
  scanf("%d",&n);
  for (i=0; i<n; i++) for (j=0; j<n; j++) {
    scanf("%s",s);
    a[i][j]=s;
  }
  for (i=0; i<n; i++) if (a[i][i]=="YES") {
    for (j=0; j<n; j++) {
	  for (k=0; k<n; k++) if (a[i][k]!=a[j][k]) break;
	  if ((k<n)^(a[i][j]=="NO")) break;
	}
	if (j>=n) break;
  }
  if (i<n) for (j=0; j<n; j++) puts(a[i][j].c_str()); else for (j=0; j<n; j++) puts("NO");
  return 0;
}
