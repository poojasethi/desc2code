# include<iostream>

using namespace std;

char T[100][100];

bool matching(int t, int pos1, int pos2)
{
	int i;

	for(i=0 ; i<t ; i++)
		if(T[pos1][i] != T[pos2][i])
			return false;
	return true;
}
bool checkForYes(int t, int pos)
{
	int i,j;
	bool matched;

	for(i=0 ; i<t ; i++)
	{
		matched = matching(t,i,pos);

		if(!(T[pos][i]=='Y' && matched) && !(T[pos][i]=='N' && !matched))
			return false;
	}
	return true;
}

int main()
{
	int t, i,j;
	char temp[4];

	scanf("%d",&t);
	

	for(i=0 ; i<t ; i++)
	{
		for(j=0 ; j<t ; j++)
		{
			scanf("%s",temp);
			T[i][j]=temp[0];
		}
	}

	for(i=0 ; i<t ; i++)
	{
		if(checkForYes(t,i))
			printf("YES\n");
		else
			printf("NO\n");
	}

	return 0;
}
