#include<iostream>
#include<cstring>

using namespace std;

int main(void)
{
 int tc,i,j,x;
 char str[105][105],inp[10];

 cin>>tc;


for(i=0;i<tc;i++)
 {
   for(j=0;j<tc;j++)
   {
    cin>>inp;

   if(strcmp(inp,"NO")==0)
    str[i][j]='0';
    else
    str[i][j]='1';
  }
 str[i][j]='\0';
 }

 for(i=0;i<tc;i++)
 {
  x=0;
  for(j=0;j<tc;j++)
  {
   if(str[i][j]=='0'&&(strcmp(str[i],str[j])==0))
    {
    x=1;
    break;
   }

  if(str[i][j]=='1'&&!(strcmp(str[i],str[j])==0))
   {
    x=1;
    break;
   }

}
 if(x==0)
 break;
 }

if(x==0)
{
 for(j=0;j<tc;j++)
{
 if(str[i][j]=='0')
 cout<<"NO"<<endl;
 else
 cout<<"YES"<<endl;
}
}
 else
 for(j=0;j<tc;j++)
 cout<<"NO"<<endl;

return 0;
}
