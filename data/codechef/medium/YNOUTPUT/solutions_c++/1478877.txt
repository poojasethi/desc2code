bool data[101][101];
int main() {
	int t;
	scanf("%d",&t);
	string s;
	getline(cin,s);
	for(int i=1; i<=t; i++) {
		for(int j=1; j<=t; j++) {
			getline(cin,s);
			if(s=="YES") data[i][j] = 1;
		}
	}
	bool found = false;
	for(int i=1; i<=t; i++) {
		bool match = true;
		for(int j=1; j<=t; j++) {
			bool mc = true;
			for(int k=1;k<=t; k++) {
				if(data[j][k]!=data[i][k]) { mc = false; break; }
			}
			if(data[i][j]!=mc) { match = false; break; }
		}
		if(match) { 
			for(int j=1; j<=t; j++) {
				if(data[i][j]==1) printf("YES\n");
				else printf("NO\n");
			}
			found = true;
			break;
		}
	}
	if(!found) {
		for(int i=0; i<t; i++) {
			printf("NO\n");
		}
	}
	return 0;
}
