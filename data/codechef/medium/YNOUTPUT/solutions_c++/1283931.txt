#include <iostream>
#include <string>
using namespace std;

string str[10005];

bool cmp(int i,int j,int N)
{
	bool flag=true;
	for(int k=0;k<N;++k)flag = flag && (str[i*N+k]==str[j*N+k]);
	return flag;
}
int main()
{
	int N;cin>>N;
	int power=N*N;
	for(int i=0;i<power;++i)cin>>str[i];
	for(int i=0;i<N;++i)
	{
		bool flag=true;
		if(str[i*N+i]=="NO")continue;
		for(int j=0;j<N;++j)
		{
			if(str[i*N+j]=="YES")
			{
				for(int k=0;k<N;++k)flag = flag && (str[i*N+k]==str[j*N+k]);
			}
			else if(str[i*N+j]=="NO" && (cmp(i,j,N)==true))flag=false;
		}
		if(flag)
		{
			for(int j=0;j<N;++j)cout<<str[i*N+j]<<endl;
			return 0;
		}
	}
	for(int i=0;i<N;++i)cout<<"NO"<<endl;
	return 0;
}
