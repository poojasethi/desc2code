#include<cstdio>
using namespace std;
int abs(int a){
    if(a<0){
        return -a;
    }
    return a;
}
int main(){
  int T;
    scanf("%d",&T);
    while(T--){
        int N,A,B;
        scanf("%d %d %d",&N,&A,&B);
        int c1=0,c2=0;
        while(A){
            if(A%2){
                c1++;
            }
            A=A/2;
        }
        while(B){
            if(B%2){
                c2++;
            }
            B=B/2;
        }
        int zero=abs(N-c1-c2);
        int one=N-zero;
        int ans=1;
        for(int i=0;i<one;i++){
            ans*=2;
        }
        ans--;
        for(int i=0;i<zero;i++){
            ans*=2;
        }
        printf("%d\n",ans);
        
    }
}
