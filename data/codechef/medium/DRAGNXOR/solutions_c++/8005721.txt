// DRAGONXOR
// Open the Dragon Scroll

#include <iostream>
#include <cstdio>
 
int min(int a, int b)
{
	if(a < b)
		return a;
	else
		return b;
}

using namespace std;
int main()
{
	int t,n,a,b, one_a, one_b, zero_a, zero_b, ans_zero, ans_one, temp;
	long long int ans;
	cin>>t;
	while (t--)
	{
		scanf("%d %d %d", &n, &a, &b);
		
		zero_a = 0;
		one_a= 0;
		zero_b= 0;
		one_b= 0;
 
		temp=n;
		while(temp--)
		{
			if (a & 1)
				one_a++;
			else
				zero_a++;
			a>>= 1;
		}
		
		temp = n;
		while (temp--)
		{
			if (b & 1)
				one_b++;
			else
				zero_b++;
			b >>= 1;
		}
 
		ans_one = min(one_a, zero_b) + min(one_b, zero_a);
		ans_zero = n - ans_one;
		ans = 0;
 
		while (ans_one)
		{
			n--;
			ans += (1 << n);
			ans_one--;
		}
 
		printf("%llu\n", ans);
	}
	return 0;
}
