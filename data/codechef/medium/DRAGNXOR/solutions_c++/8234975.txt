	#include<iostream>
	using namespace std;
	int count_one(int a)
	{
		int count=0;
		while(a>0)
		{
			if(a%2==1)
				count++;
			a/=2;
			
		}	
		return count;
	}	
	int main()
	{

		int t;
		cin>>t;
		while(t>0)
		{
			int a, b, n;
			cin>>n>>a>>b;
			int count1=count_one(a);
			int count2=count_one(b);
			int zeros;	
			if(count1+count2<=n)
				zeros=n-count1-count2;		
			else
				zeros=count1+count2-n;
			long long int base=1;
			long long int number=0;
			for(int i=1;i<=n-zeros;i++)
			{
				number=number+base;
				base=base*2;
			}
			number=number<<zeros;
			cout<<number<<endl;
			t--;
		}	

	}
