#include<bits/stdc++.h>
using namespace std;
int nsetBits(int n){
	int ans=0;
	while(n){
		if(n&1){
			ans++;
		}
		n=n>>1;
	}
	return ans;
}
int main(){
	int t,n,a,b;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d %d",&n,&a,&b);
		int na=nsetBits(a);
		int nb=nsetBits(b);
//		cout<<"na= "<<na<<" nb= "<<nb<<endl;
		int l;
		if(n>=(na+nb)){
			l=na+nb;
		}
		else{
			int p=na+nb-n;
			l=n-p;
		}
		n--;
		int ans=0;
		while(l--){
			ans+=pow(2,n);
			n--;
		}
		printf("%d\n",ans);
	}
	return 0;
}
