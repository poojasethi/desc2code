#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
#include <vector>
#include <set>
#include <iterator>
using namespace std;

int main(void){
    int t;
    cin>>t;
    while (t--) {
        int n, a, b, temp=1;
        cin>>n;
        cin>>a;
        cin>>b;
        short counter=0;
        for(int i=0; i<n; i++){
            if (a&1) {
                counter++;
            }
            if (b&1) {
                counter++;
            }
            a>>=1;
            b>>=1;
            if(i!=0)
            temp<<=1;
        }
        
        int i=0;
        b=0;
        while (i<n&&counter>0) {
                b|=temp;
                counter--;
                temp>>=1;
                i++;
        }
        
        temp=1;
        while(counter){
            b&=(~temp);
            temp<<=1;
            counter--;
        }
        cout<<b<<endl;
    }

    return 0;
}
