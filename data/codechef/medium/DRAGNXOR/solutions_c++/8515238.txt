#include<iostream>
#include<math.h>
using namespace std;

int countset(long long a){
	int count=0;
	while(a){
		a=a&(a-1);
		count++;
	}
	return count;
}

int main(){
	int t;
	cin>>t;
	while(t--){
		long long n,a,b;
		cin>>n>>a>>b;
		long long res=1;
		res=res<<n;
		res-=1;
		int a1=countset(a);
		int b1=countset(b);
		int n1;
		if(n>=(a1+b1))
			n1=(a1+b1);
		else
			n1=n-((a1+b1)-n);
		
		long long res1=res>>n1;
		res=res^res1;
		cout<<res<<endl;
		}
	return 0;
}
