#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int count(int n)
{
	int cnt=0;
	long p;
	while( n>0 )
	{
		if( n%2==1 )
			cnt++;
		n /= 2;
	}
	return cnt;
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n, a, b;
		scanf("%d %d %d", &n, &a, &b);
		int x = count(a);
		int y = count(b);
		int z = min(x, n-y) + min(y, n-x);
		printf("%d \n", ((1<<z)-1) << (n-z) );
	}
	return 0;
}
