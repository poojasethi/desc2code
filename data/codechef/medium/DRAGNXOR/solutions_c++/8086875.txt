#include <cstdio>
using namespace std;

int nbits(int k){
	int count=0;
	while(k!=0){
		k=k&(k-1);
		count++;
	}
	// printf("nbits %d\n",count);
	return count;
}

int get_num(int n, int k){
	// printf("%d %d\n",n,k);
	return ((1<<(n)) - (1<<(n-k)));
}

int main(){
	int T;
	scanf("%d",&T);
	while(T-->0){
		int N, A, B;
		scanf("%d %d %d",&N,&A,&B);
		int a_bits=nbits(A);
		int b_bits=nbits(B);
		if(a_bits+b_bits<=N)
			printf("%d\n",get_num(N, a_bits+b_bits));
		else
			printf("%d\n",get_num(N, 2*N - (a_bits+b_bits)));
	}
}
