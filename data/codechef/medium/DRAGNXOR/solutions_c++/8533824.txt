#include <iostream>
#include<string.h>
#include<math.h>
using namespace std;
string convert(int length,int number)
{
    char result[length];
    string result_1;
    for(int i=0;i<length;i++)
    {
        if(number%2==1)
        {
                result[length-1-i]='1';

                        }
        else{result[length-1-i]='0';}

        number/=2;
    }
    for (int i=0;i<length;i++)
    {
        result_1=result_1+result[i];
    }

    return result_1;
}
int ret(int length,int number)
{
    int y=0;
    for(int i=length-1;i>=length-number;i--)
    {
        y+=pow(2,i);
    }
    return y;
}
int main()
{
    int tc;
    cin>>tc;
    string input_1;
    string input_2;
    int number_1,number_2,length;



     while(tc--)
     {
            int count1=0;
            cin>>length;
            cin>>number_1;
            cin>>number_2;
            input_1=convert(length,number_1);
            input_2=convert(length,number_2);
            //cout<<input_1<<"  "<<input_2<<endl;

            //count no.of 1s
            for (int i=0;i<length;i++)
            {
                if (input_1[i]=='1')
                {
                    count1++;
                }
                if (input_2[i]=='1')
                {
                    count1++;
                }
            }
            if (count1<length)
            {
             cout<< ret(length,count1)<<endl;
            }
            else{
             cout<<ret(length,length*2-count1)<<endl;

            }


     }
}
