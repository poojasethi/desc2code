#include <iostream>
#include<cmath>
using namespace std;
int main() 
{
    int t,n,total,a,b,as,bs;
    cin>>t;
    while(t--)
    {
        cin>>n>>a>>b;
        as=0;
        bs=0;
        while(a>0)
            {
                if(a%2==1)
                    as++;
                a/=2;
            }
        while(b>0)
            {
                if(b%2==1)
                    bs++;
                b/=2;
            }
        total=as+bs;
        int maxno=0;
        if(total<=n)
            {
                for(int i=n-1;i>=n-total;i--)
                {
                    maxno+=pow(2,i);
                }
            }
        else
            {
                maxno=pow(2,n)-1;
                int sub=0;
                for(int i=0;i<total-n;i++)
                    sub+=pow(2,i);
                maxno-=sub;
            }
        cout<<maxno<<endl;
    }
	return 0;
}
