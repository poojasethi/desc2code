
#include<cstdio>
using namespace std;
long long int power_2[50];
void initialise()
{
	power_2[0]=1;power_2[1]=2;
	for(int i=2;i<40;i++)
	{
		power_2[i]=2*power_2[i-1];
	}
}
long long int calculate(int bit,long long int n)
{
	n--;
	long long int sum=0;
	for(int i=bit;i>=1;i--)
	{
		sum+=power_2[n--];
	}
	return sum;
}
int count_bits(long long int a)
{
	int count=0;
	while(a)
	{
		if(a&1)
		count++;
		a=a>>1;
	}
	return count;
}
int main()
{
	initialise();
	int tc,x,y,bits=0;
	long long int n,a,b,ans;
	scanf("%d",&tc);
	while(tc--)
	{
		scanf("%lld %lld %lld",&n,&a,&b);
		x=count_bits(a);
		y=count_bits(b);
		if(x+y<n)
		bits=x+y;
		else if(x+y==n)
		bits=x+y;
		else
		{
		  bits=(n-x)+(n-y);
		}
		//bits=(x+y)-n;
		//bits=n-bits;
		ans=calculate(bits,n);
		printf("%lld\n",ans);
	}
	return 0;
	}
