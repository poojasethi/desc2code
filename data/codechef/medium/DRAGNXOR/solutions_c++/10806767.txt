#include<iostream>
#include<cmath>
using namespace std;
int main(){
    int t,n,a,b,o,o1,i,j,count,x,y,min;
    cin>>t;
    while(t--){
        cin>>n>>a>>b;
        o=0,o1=0;
        while(a){
            a=a&(a-1);
            ++o;
        }
        while(b){
            b=b&(b-1);
            ++o1;
        }
        int c[n],d[n],e[n],f[n];
        for(i=0;i<n;i++){
            if(i<o)
            c[i]=1;
            if(i<o1)
            d[i]=1;
            if(i>=o)
            c[i]=0;
            if(i>=o1)
            d[i]=0;
        }
        for(i=0;i<n;i++){
            count=0,x=0,y=0;
            while(count!=(n)){
                x=x+c[(count+i)%n]*pow(2,n-count-1);
                y=y+d[(count+i)%n]*pow(2,n-count-1);
                ++count;
            }
            e[i]=x;
            f[i]=y;
        }
        min=e[0]^f[0];
        for(i=0;i<n;i++){
            for(j=0;j<n;j++){
                if(min<(e[i]^f[j]))
                min=e[i]^f[j];
            }
        }
        cout<<min<<endl;
    }
    return 0;
}
