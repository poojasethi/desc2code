#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t,n,a,b;
	cin>>t;
	while(t--)
	{
		cin>>n>>a>>b;
		int temp,i,a1=0,b1=0,res;
		temp=a;
		while(temp>0)
		{
			if(temp%2!=0)
			a1++;
			temp=temp/2;
		}
			temp=b;
		while(temp>0)
		{
			if(temp%2!=0)
			b1++;
			temp=temp/2;
		}
		if(a1+b1<=n)
		res=pow(2,n-a1-b1)*(pow(2,a1+b1)-1);
		else
		res=pow(2,a1+b1-n)*(pow(2,2*n-a1-b1)-1);
		cout<<res<<endl;
	}
	return 0;
}
