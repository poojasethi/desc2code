#include<bits/stdc++.h>
using namespace std;
int ones(int x)
{
	if(x==0)
	return 0;
	return 1+ones(x & x-1);
}
int main()
{
	int t,n,a,b;
	cin>>t;
	while(t--)
	{
		cin>>n>>a>>b;
		int a1,a0,b1,b0,p;
		a1=ones(a);
		a0=n-a1;
		b1=ones(b);
		b0=n-b1;
		p=min(a1,b0)+min(a0,b1);
		cout<<((1<<p)-1 << n-p)<<endl;
	}
}
