#include<iostream>
#include<math.h>
using namespace std;
int main()
{
 long long t,n,a,b,i,j,x=0,y=0,p;
 cin>>t;
 while(t--)
 { x=0;y=0;
   cin>>n>>a>>b;
   while(a!=0)
   {
     i=a%2;
     a=a/2;
     if(i==1)
     x++;
   }
   while(b!=0)
   {
     i=b%2;
     b=b/2;
     if(i==1)
     y++;
   }
   i=0;
   if(x+y<=n)
   {
     for(p=0;p<x+y;p++)
     {
	n=n-1;
	i+=pow(2,n);
      }
    }
    else
    {
    j=(2*n)-(x+y);
    for(p=0;p<j;p++)
    {
    n=n-1;
    i+=pow(2,n);
    }
  }
    cout<<i<<"\n";
  }
  return 0;
  }
