#include <iostream>
#include <cmath>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		int n,a,b,r,s;
		int p=0;int q=0;
		cin>>n>>a>>b;
		while(a!=0)
	  {
	    if(a%2==1)
	    p++;
	    a=a/2;
	  }
	  while(b!=0)
	  {
	    if(b%2==1)
	    q++;
	    b=b/2;
	  }
	  r=p+q;
	  if(r>n)
	  r=n-(r-n);
	  s=pow(2,n)-1-(pow(2,n-r)-1);
	  cout<<s<<endl;
	}
	return 0;
}
