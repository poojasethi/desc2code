#include<bits/stdc++.h>
using namespace std;
int func(int n,int a)
{
    int temp=0;
    for(int i=0;i<a;i++)
        temp=temp+pow(2,n-i);
    return temp;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,a,b,n0_a=0,n0_b=0,n1_b=0,n1_a=0,temp;
        cin>>n>>a>>b;
        do
        {
            temp=a%2;
            a=a/2;
            if(temp)
                n1_a++;
            else
                n0_a++;
        }while(a!=0);
        if(n1_a+n0_a<n)
            n0_a=n0_a+(n-(n0_a+n1_a));
        do
        {
            temp=b%2;
            b=b/2;
            if(temp)
                n1_b++;
            else
                n0_b++;
        }while(b!=0);
        if(n1_b+n0_b<n)
            n0_b=n0_b+(n-(n0_b+n1_b));

        if(n0_a==n1_b)
            temp=pow(2,n)-1;
        else if(n0_a<n1_b)
            temp=func(n-1,n0_a+n0_b);
        else
            temp=func(n-1,n1_a+n1_b);
        cout<<temp<<"\n";
    }
}
