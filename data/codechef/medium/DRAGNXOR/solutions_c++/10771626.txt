#include <iostream>
using namespace std;
int numOnes(int n){
	int count=0;
	while(n){
		if(n&1)
			count++;
		n=n>>1;
	}
	return count;
}

int main(){
	int T;
	cin >> T;
	int N,A,B;
	int A0,A1,B0,B1;
	for(int t=0;t<T;t++){
		cin >> N >> A >> B;
		A1 = numOnes(A);
		B1 = numOnes(B);
		A0 = N - A1;
		B0 = N - B1;
		int val=0;
		while(A1 && B0){
			val = val*2 + 1;
			A1--;B0--;
		}
		while(A0 && B1){
			val = val*2 + 1;
			A0--;B1--;
		}
		while(A0 && B0){
			val = val*2 + 0;
			A0--;B0--;
		}
		while(A1 && B1){
			val = val*2 + 0;
			A1--;B1--;
		}
		cout << val << endl;
	}
	return 0;
}
