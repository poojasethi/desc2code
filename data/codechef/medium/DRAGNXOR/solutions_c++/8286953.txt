void binary(int x, int y[])
{
    int i = 0, remainder;
    while(x > 1)
    {
        remainder = x % 2;
        x /= 2;
        y[i] = remainder;
        i++;
    }
    if(x <= 1)
    {
        y[i] = x;
    }
    return;
}
