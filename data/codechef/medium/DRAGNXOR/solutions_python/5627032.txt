import collections as cc
t=int(raw_input())
for ii in range(t):
    fil=map(int,raw_input().split())
    a=fil[1]
    b=fil[2]
    fil=fil[0]
    b=bin(b)[2:]
    a=bin(a)[2:]
    a=str(a).zfill(fil)
    b=str(b).zfill(fil)
    aa=cc.Counter(a)
    bb=cc.Counter(b)
    x=min(aa['1'],bb['0'])
    y=min(aa['0'],bb['1'])
    p=x+y
    ans=str(1)*p
    ad=fil-p
    ans=str(ans)+str(0)*ad
    ans=int(ans,2)
    print ans
