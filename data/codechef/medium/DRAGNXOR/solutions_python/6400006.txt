t=int(raw_input())
while t>0:
	t-=1
	n,a,b=map(int,raw_input().split())
	l1,l2=list(bin(a)),list(bin(b))
	l1=l1[-1:1:-1]
	l2=l2[-1:1:-1]
	if len(l1)<n:
		l1=l1[:]+[0]*(n-len(l1))
	else:
		l1=l1[:n]
	if len(l2)<n:
		l2=l2[:]+[0]*(n-len(l2))
	else:
		l2=l2[:n]
	#print l1,l2
	a=l1.count('1')
	b=l2.count('1')
	if a+b==n:
		ans=[1]*n
	elif a+b<n:
		ans=[0]*(n-a-b)+[1]*(a+b)
	else:
		ans=[0]*(a+b-n)+[1]*(2*n-a-b)
	s=0
	for i in xrange(n):
		s+=ans[i]*(2**i)
	print s
