def maxNum(N, A, B):
	#중요한 키 포인트는 하나는 1을 앞에서부터 채우고, 하나는 1을 뒤에서부터 채운 후 
	#xor을 하면 최대 값을 얻을 수 있다!
	A = bin(A)[2:]
	B = bin(B)[2:]
	num1inA = A.count('1')
	num1inB = B.count('1')
	number = 0
	if N >= num1inA+num1inB:
		for i in range(1, num1inA+num1inB+1):
			number += 2**(N-i)
	else:
		for i in range(1, N-(num1inA+num1inB-N)+1):
			number += 2**(N-i)
	return number

T = int(raw_input())
for _ in range(T):
	N, A, B = map(int, raw_input().split())
	print maxNum(N, A, B)
