def fun(n,a,b):
    l=bin(a)
    l=l[2:]
    m=bin(b)
    m=m[2:]
    w=l.count('1')+m.count('1')
    if w<=n:
        a='1'*w+'0'*(n-w)
        return int(a,2)
    else:
        a='1'*(2*n-w)+'0'*(w-n)
        return int(a,2)
x=input()
e=list()
for i in range(0,x,1):
    r=raw_input()
    s=r.rsplit()
    t=fun(int(s[0]),int(s[1]),int(s[2]))
    e.append(t)
for i in e:
    print i
