for _ in xrange(int(raw_input())):
	n,a,b  = map(int,raw_input().split())
	x = bin(a)[2:]
	y = bin(b)[2:]
	c1x = 0
	c1y = 0
	for i in xrange(len(x)):
		if x[i]=='1':
			c1x+=1
	for i in xrange(len(y)):
		if y[i]=='1':
			c1y+=1
	st=""
	#print c1x, c1y
	if c1x + c1y <= n:
		st+=('1'*(c1x+c1y))
		st+=('0'*(n-c1x-c1y))
	else:
		st+=('1'*(n-(c1x+c1y-n)))
		st+=('0'*(c1x+c1y-n))
	#print st
	print int(st,2)
