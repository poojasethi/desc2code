import math
t=input()
while t:
    n,a,b=map(int,raw_input().split())
    i=bin(a).count("1")
    j=bin(b).count("1")
    if i+j>=n:
        p=2*n-i-j
    else :
        p=i+j
    k=1
    xor=0
    while k<=p:
        n-=1
        xor+=math.pow(2,n)
        k+=1
    xor=int(xor)
    print(xor)
    t-=1
