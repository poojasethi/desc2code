def setBits(n):
	count = 0
  	while(n):
  		count += n & 1
  		n >>= 1
  	return count

t=int(raw_input())
while(t>0):
	t-=1
	n, a, b = map(int, raw_input().split())
	s_a=setBits(a)
	s_b=setBits(b)
	if((s_a+s_b)>n):
		s_r=2*n-(s_a+s_b)
	else:
		s_r=(s_a+s_b)
	ans=0
	n-=1
	while(s_r>0):
		ans+=pow(2, n)
		n-=1
		s_r-=1
