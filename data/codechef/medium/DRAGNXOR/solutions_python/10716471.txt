t=input()
for i in range(int(t)):
    n,a,b=map(int,raw_input().strip().split())
    assert 1<=n and n<=30
    assert 0<=a and 2**n>a
    assert 0<=b and 2**n>b
    c='0'+str(n)+'b'
    a=format(a,c)
    b=format(b,c)
    x=min(a.count('1'),b.count('0'))
    y=min(a.count('0'),b.count('1'))
    p=x+y
    res='1'*p+'0'*(n-p)
    print int(res,2)
