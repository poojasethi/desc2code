def count1s(n, x):
	count = 0
	for i in xrange(n):
		if x & (1 << i):
			count += 1
	return count
		
cases = int(raw_input())
for _dummy in range(cases):
	n, a, b = map(int, raw_input().split())
	#print n, a, bin(a), b, bin(b)
	a = count1s(n, a)
	b = count1s(n, b)
	numzeros = abs(n - (a+b))
	ans = (2**n)-1 - ((2**numzeros)-1)
	print ans
