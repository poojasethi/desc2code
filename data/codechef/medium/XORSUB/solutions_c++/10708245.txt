#include<bits/stdc++.h>
using namespace std;

#define tr(c,i) for(auto i=(c).begin(); i != (c).end(); i++)

void solve()
{
	int n,k;
	scanf("%d %d",&n,&k);
    set <int> s,w;
    int i;
    for(i=0;i<n;i++)
    {
    int x;
    scanf("%d",&x);
    s.insert(x);
    }
    w.insert(k);
    int ans=k;
    set <int>::iterator a;
    set <int>::iterator b;
    for(a=s.begin();a!=s.end();a++)
    {
    for(b=w.begin();b!=w.end();b++)
    {
    int x=(*a)^(*b);
    ans=max(ans,x);
    w.insert(x);
    }
    }
    printf("%d\n",ans);
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
      solve();
	}
	return 0;
}
