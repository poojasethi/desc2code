#include<cstdio>
#include<algorithm>
#include<vector>
#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	 scanf("%d",&t);
	 while(t--)
	  {
	  	
	  
    int n;
    
    scanf("%d",&n);
    long long  k;
      scanf("%lld",&k);
    
    vector<long long> v;
    long long x;
    for(int i=0;i<n;i++)
    {
            scanf("%lld",&x);
           
           
            /*for(int j=0;j<=32;j++)
             {
             	 if((x&(1<<j)) && (k &(1<<j)))
             	   {
             	        x&=~(1<<j);	
				   }
			 }*/
			  v.push_back(x);
    }
    
    sort(v.rbegin(),v.rend());
    
    long long int sig=1;
    
    
    //  calculating maximum no of bit on 
    while(sig<=v[0])
    sig<<=1;
    
    sig>>=1;
    
    for(int t=0;sig>=1;sig>>=1)
    {
            int i=t;
            
            while(i<n && (v[i]&sig)==0)
            i++;
            
            if(i>=n)
            continue;
            
            swap(v[t],v[i]);
            
            for(int j=0;j<n;++j)
            {
                    if(j!=t && (v[j]&sig)!=0)
                    v[j]=v[j]^v[t];
            }
            
            t++;
            
    }
    
    long long  result=k;
    //cout<<" res "<<result<<endl;
    
    for(int q=0;q<n;q++)
    {
    //	cout<<" result "<<result<<" "<<(result^v[q])<<endl;
    	if((result^v[q])>result)
            result=result^v[q];
    }
    
    printf("%lld\n",result);
    
}
    //system ("pause");a
    
    return 0;
}
