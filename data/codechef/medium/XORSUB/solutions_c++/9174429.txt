#include<iostream>
#include<cstdio>
#include<vector>

#define vi vector<int>
using namespace std;

int a[1005],storage[65];

int lencnt(int x){
    int cnt=0;
    while(x){
        cnt+=1;
        x>>=1;
    }
    return cnt;
}

int main(){
    int t,n,k,ans,i,j,index,len,tmp;
    scanf("%d",&t);
    while(t--){
        scanf("%d%d",&n,&k);
        for(i=0;i<n;i++){
            scanf("%d",&a[i]);
        }
        vi buckets[65];
        for(i=0;i<n;i++){
           len=lencnt(a[i]);
           buckets[len].push_back(a[i]);
        }
        index=0;
        for(i=64;i>0;i--){
            if(buckets[i].size()){
                storage[index]=buckets[i][0];
                index+=1;
                for(j=1;j<buckets[i].size();j++){
                   tmp=buckets[i][0]^buckets[i][j];
                   len=lencnt(tmp);
                   buckets[len].push_back(tmp);
                }
            }
        }
        ans=k;
        for(i=0;i<index;i++){
            if(ans<(ans^storage[i])){
                ans=(ans^storage[i]);
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
