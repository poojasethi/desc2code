#include <bits/stdc++.h>

using namespace std;

int dp[1001][1024];

int main() {
    int t, n, k;
    int array[1001];
    cin >> t;
    while(t --) {
        cin >> n >> k;
        for(int i = 1; i <= n; i ++) {
            cin >> array[i];
        }
        for(int i = 0; i <= n; i ++) {
            for(int j = 0; j <= 1023; j ++) {
                dp[i][j] = 0;
            }
        }
        dp[0][0] = 1;
        for(int i = 1; i <= n; i ++) {
            for(int j = 0; j <= 1023; j ++) {
                dp[i][j] = dp[i - 1][j] | dp[i - 1][j ^ array[i]];
            }
        }
        int max = INT_MIN;
        for(int i = 0; i <= 1023; i ++) {
            if((dp[n][i] * (i ^ k)) > max) {
                max = (dp[n][i] * (i ^ k));
            }
        }
        cout << max << endl;
    }
    return 0;
}
