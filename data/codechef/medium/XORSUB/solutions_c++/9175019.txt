#include<bits/stdc++.h>
using namespace std;
int dp[1<<10][1<<10];
void init()
{
  int k,n,a[1004];
  cin>>n>>k;
  for(int i=1;i<=n;i++)
  cin>>a[i];
  dp[0][0] = 1; // with no elements 0 can be achieved
  for(int i=1;i<1024;i++)
  dp[0][i] = 0;

  for(int i=1;i<=n;i++)
  for(int j=0;j<1024;j++)
  dp[i][j] = dp[i-1][j]|dp[i-1][j^a[i]];

  int mx = 0;

  //dp[n][j] : with all the n elements can we j
  for(int i=0;i<1024;i++)
  mx = max(mx,dp[n][i]*(k^i));
  cout<<mx<<endl;
}

int main()
{
  int t;
  cin>>t;
  while(t--)
  init();
  return 0;
}
