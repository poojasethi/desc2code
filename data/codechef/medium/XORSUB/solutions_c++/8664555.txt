#include <bits/stdc++.h>
using namespace std;

// for fast integer input
int nextInt() 
{
    int x = 0, p = 1;
    char c;
    do {
        c = getchar();
    } while (c <= 32);
    if (c == '-') {
        p = -1;
        c = getchar();
    }
    while (c >= '0' && c <= '9') {
        x = x * 10 + c - '0';
        c = getchar();
    }
    return x * p;
}

//for loop from zero till n
#define for0(i,n) for(__typeof(n) i = 0; i < n ; i++)
