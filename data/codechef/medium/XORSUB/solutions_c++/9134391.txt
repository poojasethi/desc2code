#include<bits/stdc++.h>
using namespace std;
int main()
{
std::ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
int n,k,x;
vector<int>a(1025,0);
cin>>n>>k;
for(int i=0;i<n;i++)
{
cin>>x;
for(int j=0;j<1024;j++)
{
if(a[j])
a[j^x]=1;
}
a[x]=1;
}
a[0]=1;
long long int l=0;
for(int j=0;j<1024;j++)
{
if(a[j])
{
if(l<(k^j))
l=k^j;
}
}
cout<<l<<endl;
}
return(0);
}
