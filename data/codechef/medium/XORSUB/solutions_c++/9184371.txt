#include <bits/stdc++.h>
using namespace std;

int maxor(int *a, int n, int k)
{
    int ans = 0;
    int dp[n+1][1024];
    dp[0][0] = 1;
    memset(dp, 0, sizeof(dp));
    dp[0][0] = 1;
    for(int i = 1; i<=n; ++i) {
        for(int j = 0; j<1024; ++j) {
            dp[i][j] = dp[i-1][j] | dp[i-1][j^a[i-1]];
            if(dp[i][j]) {
                ans = max(ans, j^k);
            }
        }
    }
    return ans;
}

int main()
{
    std :: ios :: sync_with_stdio(false);
    int t;
    cin >> t;
    while(t--) {
        int n, k;
        cin >> n >> k;
        int a[n];
        for(int i = 0; i<n; ++i)
            cin >> a[i];
        cout << maxor(a, n, k) << endl;
    }
    return 0;
}
