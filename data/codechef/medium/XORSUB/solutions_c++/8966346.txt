#include<iostream>
#include<cstring>
#define max(a,b) a>b?a:b
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		int dp[1024];
		memset(dp,0,sizeof dp);
		dp[0]=1;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<1024;j++)
			{
				dp[j^a[i]]|=dp[j];
			}
		}
		int ans=0;
		for(int i=0;i<1024;i++)
		{
			if(dp[i])	
			{
			//cout<<(k^i)<<endl;	
			ans=max(ans,(k^i));
			}
		}
		cout<<ans<<endl;
	}
}
