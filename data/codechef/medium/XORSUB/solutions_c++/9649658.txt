#include<bits/stdc++.h>
using namespace std;
const int MAXN = 2e3 + 10;
bool isThere[MAXN];
void insert(int K){
	for(int i = 0;i <= 1024;++i){
		if(isThere[i]){
			isThere[i ^ K] = true;
		}
	}
	isThere[K] = true;
}
int main(){
	int t;
	cin >> t;
	while(t--){
		int N, K;
		cin >> N >> K;
		memset(isThere, 0, sizeof(isThere));
		for(int i = 1;i <= N;++i){
			int x;
			cin >> x;
			insert(x);
		}
		int ans = K;
		for(int i = 1024;i >= 0;--i){
			if(isThere[i]){
				ans = max(ans, K^i);
			}
		}
		cout << ans << "\n";
	}
	return 0;
}
