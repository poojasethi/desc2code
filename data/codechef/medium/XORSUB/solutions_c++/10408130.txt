#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<cmath>

using namespace std;

int main()

{
    int t,n,k;
    scanf("%d",&t);
    while(t--)
    {
        int dp[1024]={0};
        scanf("%d %d",&n,&k);
        int num;
        for(int i=0;i<n;i++)
        {
             scanf("%d",&num);
             for(int j=0;j<1024;j++)
             {
                 if(dp[j])
                    dp[j^num]=1;
             }
             dp[num]=1;
        }
        int maxi=k;
        for(int i=0;i<1024;i++)
        {
            if(dp[i])
                maxi=max(maxi,i^k);
        }
        printf("%d\n",maxi);
    }
    return 0;
}
