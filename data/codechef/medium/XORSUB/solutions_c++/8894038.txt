#include <iostream>

using namespace std;
int dp[1025];
int main()
{
int test;
cin>>test;
while(test--)
{
    int a,b,ab;
    cin>>a>>b;
    		for(int i=0; i<=1024; i++)
			dp[i]=0;
    for(int i=1;i<=a;i++)
    {
        cin>>ab;
        for(int j=0;j<1024;j++)
        {
            if(dp[j])
                dp[j^ab]=1;
        }
        dp[ab]=1;
    }
    int max=b;
    for(int i=0;i<1024;i++)
    {
        if(dp[i])
        {
            if((i^b)>max)
                max=i^b;
        }
    }
    cout<<max<<endl;
}
    return 0;
}
