		memset(dp, 0, sizeof dp);
		
		dp[0][0] = 1;
		for (i = 1; i < n + 1; i++) {
			dp[i][a[i]] |= 1;					//including the i'th element when none have been include before
			for (j = 0; j < 1024; j++) {
				dp[i][j] |= dp[i - 1][j];			//preserving the previous row / not including the i'th element
				dp[i][j] |= dp[i - 1][j ^ a[i]];	//including the i'th element
			}
		}
		
		maxi = 0;
		for (i = 0; i < 1024; i++) {
			if (dp[n][i]) {
				maxi = max(maxi, i ^ k);
			}
		}
		cout << maxi << endl;

	}
