#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		int a[n];
		for (int i = 0; i < n; i++)
			scanf("%d",&a[i]);
		vector <int> dp(1024,0);
		dp[0] = 1;
		vector<int> temp(1024,0);
		for (int i = 0; i < n; i++)
		{
			temp = dp;
			for (int j = 0; j < 1024; j++)
			{
				if (dp[j] == 1)
				{
					int temp1 = j^a[i];
					if (temp1 < 1024)
						temp[temp1] = 1;
				}
			}
			dp = temp;
		}
		int ans = 0;
		for (int i = 0; i < 1024; i++)
		{
			if (dp[i] == 1)
			{
				ans = max(ans,k^i);
			}
		}
		cout << ans << endl;
	}
	return 0;
}
