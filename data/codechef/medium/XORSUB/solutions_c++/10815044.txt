#include<bits/stdc++.h>
typedef long long ll;
using namespace std;
#define SIZE 2000
int dp[SIZE];
int main(){
	int t;
	cin >> t;
	while(t--){
		int n,k;
		cin >> n >> k;
		vector<int> a(n);
		memset(dp, 0, sizeof(dp));
		for(int i=0; i<n;i++){
			cin >> a[i];
		}
		dp[0] = 1;
		for(int i=0; i<n; i++){
			for(int j=0; j<SIZE; j++){
				dp[a[i] ^ j] += dp[j];
			}
		}
		int mx = -1;
		for(int i=0; i<SIZE; i++)
			if(dp[i]) mx = max(mx, i^k); 
		
		cout << mx << endl;
	}
	return 0;
}
