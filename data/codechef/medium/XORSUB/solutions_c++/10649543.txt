ll GaussEli(vector<ll> V,int n,int K){
	sort(V.rbegin(),V.rend());
	ll bit = 1;
	while(bit<=V[0]){
		bit<<=1;
	}
	bit>>=1;
	for(int i=0;bit>=1;bit>>=1){
		int j=i;
		while(j<n && (V[j]&bit)==0)
			j++;
		if(j>=n)
			continue;
		swap(V[i],V[j]);
		for(int j=0;j<n;j++){
			if(j!=i && (V[j]&bit)!=0)
				V[j]=V[j]^V[i];
		}
		i++;
	}
	ull res=0,ans=K;
	Rep(i,n){
		ans = max(ans,ans^V[i]);
	}
	return ans;
}
