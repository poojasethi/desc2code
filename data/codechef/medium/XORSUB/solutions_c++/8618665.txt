#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,j,k,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        int a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        int dp[n+1][1024];
        memset(dp,0,sizeof(dp));
        dp[0][0]=1;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<=1023;j++)
                dp[i][j]=dp[i-1][j]|dp[i-1][j^a[i-1]];
        }
        int ans=-1;
        for(i=0;i<=1023;i++)
        {
            int x=dp[n][i]*(i^k);
            if(x>ans)
                ans=x;
        }
        cout<<ans<<endl;
    }
    return 0;
}
