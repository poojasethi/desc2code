int readint()
{
	register char cc=getc(stdin);
	for(;cc<'0' && cc>'9';)
	cc=getc(stdin);
	int res=0;
	for(;cc>='0' && cc<='9';)
	{
		res=(res<<3)+(res<<1)+ cc - '0';
		cc=getc(stdin);
	}
	return res;
}

ll euler(ll x)
{
	ll result=x;
	for(ll i=2;i*i<=x;++i)
	{
		if(x%i==0)
		{
			while(x%i==0)
				x/=i;
			result-=result/i;
		}
	}
	if(x>1)
		result-=result/x;

	return result;
}
