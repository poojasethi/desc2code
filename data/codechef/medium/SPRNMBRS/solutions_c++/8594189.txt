#include <iostream>
#include <cstdio>

using namespace std;

int main() {
  int T, ans;
  long long l, r, x, y;
  scanf("%d", &T);
  while(T--) {
    x = y = 1;
    ans = 0;
    scanf("%lld%lld", &l, &r);
    if(1>=l && 1<=r) ans ++;
    for(int i=1;i<64&&x<=r&&x>0;i++) {
      if(r/2+2<x) break;
      x = x*2ll;
      y = x;
      for(int j=0;j<40&&y<=r&&y>0;j++) {
	if(y>=l&&y<=r) {
	  ans ++;
	}
	if(r/3+2<y) break;
	y = y*3ll;
      }
    }
    printf("%d\n", ans);
  }
}
