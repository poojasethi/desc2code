#include<iostream>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    unsigned long long int t,l,r,ans,a,b;
    cin>>t;
    while(t--)
    {
        cin>>l>>r;
        ans=0,a=2,b=1;
        if(l==1)ans++;
        while((a*b)<=r)
        {
            while(a*b<=r)
            {
                if((a*b)>=l && (a*b)<=r)
                    ans++;
                b*=3;
            }
            b=1,a*=2;
        }
        cout<<ans<<"\n";
    }
    return 0;
}
