#include<iostream>
using namespace std;
int main()
{
	long long t,l,r,ans,value,current;
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		ans=0;
		value=2;
		while(value<=r)
			{
				current=value;
				while(current<=r)
				{
					if(l<=current && current<=r)
					{
						ans++;
					}
					current=current*3;
				}
				value=value*2;
			}
		if(l<=1 )
			ans++;
		cout<<ans<<"\n";
	}
	return 0;
}
