#include<bits/stdc++.h>

using namespace std;
long long int a[1000007];
int main()
{
	long long int t,l,r,first,sec,top,var,ans,i;
	cin>>t;
	while(t--)
	{
		cin>>l>>r;
		/*first=ceil(log2(l));
		first=pow(2,first);
		sec=log2(r);
		sec=pow(2,sec);
		cout<<first<<" "<<sec<<"\n";*/
		first=2;
		sec=r;
		top=0;
		while(first<=sec)
		{
			a[top]=first;
			first=first*2;
			top++;
		}
		ans=0;
		for(i=0;i<top;i++)
		{
		    first=a[i];
		    while(first <=r)
		    {
		    	if(first >=l && first <=r)
		    	ans++;
		    	first=first*3;
			}
		}
		if(l==1)
		ans++;
		cout<<ans<<"\n";
	}
	return 0;
}
