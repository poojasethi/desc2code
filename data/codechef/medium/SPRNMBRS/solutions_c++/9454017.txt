#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_v) (int)(_v).size()
#define all(_v) (_v).begin(),(_v).end()
#define pii pair<int,int>
typedef long long ll;
const int INF = 1000000000;
const ll MOD = 1e9 + 7;

const int maxn = 100010;
vector < ll > v;

void calc() {
	ll num = 1;
	v.pb(num);
	while (true) {
		num *= 2ll;
		if(num > 1e18) break;
		ll p = 1;
		while (true) {
			ll nn = num * p;
			if(nn > 1e18) break;
			v.pb(nn);
			p *= 3ll;
		}
	}
	sort(all(v));
}

int main() {
	ios_base::sync_with_stdio(false);

	int t; cin >> t;
	calc();
	while (t --) {
		ll left,right; cin >> left >> right;
		ll posl = lower_bound(all(v),left) - v.begin();
		ll posr = upper_bound(all(v),right) - v.begin();
		cout << posr - posl << "\n";
	}
	
	return 0;
}
