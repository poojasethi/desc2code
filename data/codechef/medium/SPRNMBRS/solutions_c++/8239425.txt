inline long long fastread() {
	long long i=0;
	char c=getchar_unlocked();
	while(c<'0')
		c = getchar_unlocked();
	while(c>='0' && c<='9') {
		i = (i<<3)+(i<<1)+c-'0';
		c = getchar_unlocked();
	}
	return i;
}

long long calculate(long long l, long long r) {
	
	long long val=2,count=0,curr;
	while(val<=r) {
		curr=val;
		while(curr<=r) {
			if(l<=curr && r>=curr)
				count++;
			curr*=3;
		}
		val*=2;
	}
	if(l<=1 && r>=1) count++;
	return count;
}
	
int main() {
	int t=fastread();
	while(t--) {
		long long l=fastread(), r=fastread();
		cout<<calculate(l,r)<<endl;
	}
	return 0;
}
