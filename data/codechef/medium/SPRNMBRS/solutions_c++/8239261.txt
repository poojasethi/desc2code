#include<bits/stdc++.h>
using namespace std;
 
#define ll long long
 
#define sc1(x) scanf("%d",&x)
#define sc2(x,y) scanf("%lld%lld",&x,&y)
#define sc3(x,y,z) scanf("%d%d%d",&x,&y,&z)
 
 
#define pr1(x) printf("%d\n",x)
#define pr2(x,y) printf("%d %d\n",x,y)
#define pr3(x,y,z) printf("%d %d %d\n"x,y,z)
int main()
{
    vector<ll> v;
    ll x=1,y;
    v.push_back(1);
    while(1)
    {
        x=x*2;
        v.push_back(x);
        if(x>1000000000000000000LL)
            break;
        y=x;
        while(1)
        {
            y=y*3;
            v.push_back(y);
            if(y>1000000000000000000LL)
                break;
        }
 
    }
    sort(v.begin(),v.end());
    int  t;
    scanf("%d",&t);
    while(t--)
    {
        ll l,r;
        scanf("%lld%lld",&l,&r);
        if(l>r)
            swap(l,r);
        cout<<upper_bound(v.begin(),v.end(),r) - lower_bound(v.begin(),v.end(),l)<<"\n";
    }
}
