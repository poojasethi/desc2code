#include <bits/stdc++.h>
// super numbers - can be only multiples of 2 and/or 3 (divisibility rule)  q1*q2*q3*.../(q1-1)(q2-2)(q3-3)...
using namespace std;

int main(){
	int t; cin >> t;
	while(t--){
		long long l, r; cin >> l >> r;
		long long temp = 1;
		int count = 0;
		while(temp <= r){
			temp *= 2;
			long long tp = temp;
			while(tp <= r){
				if(tp >= l && tp <= r)
					count += 1;
				tp *= 3;
			}
		}
		if( l != 1) cout << count << endl;
		else cout << count + 1 << endl;
	}
	return 0;
}
