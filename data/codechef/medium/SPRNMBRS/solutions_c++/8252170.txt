#include<iostream>
using namespace std;
int main()
{
	int t; long long int L,R,value,curr,ans;
	
	cin>>t;
	
	while(t--)
	{
		cin>>L>>R;
		
		value=2,ans=0;
		
		while(value<=R)
		{
			curr=value;
			
			while(curr<=R)
			{
				if( L <=curr && curr<=R )
				   ans++;
				
				 curr*=3;
			}
	        
		   value*=2;	
		}
		
		if(L==1)
		  ans++;
		
		cout<<ans<<endl;
	}
return 0;
}
