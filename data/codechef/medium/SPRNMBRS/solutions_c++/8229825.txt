#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
        long long l,r;
        cin>>l>>r;
        long long ans=0;
        long long curr=2;
        while(curr <= r){
            long long val = curr;
            if(curr <= r && curr >= l){
                ans++;
            }
            while(val <= r){
                val *= 3;
                if(val <= r && val >= l){
                    ans++;
                }
            }
            curr *= 2;
        }
        if(l == 1)  ans++;
        cout<<ans<<"\n";
    }
    return 0;
}
