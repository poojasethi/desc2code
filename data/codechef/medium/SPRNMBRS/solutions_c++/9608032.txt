#include <bits/stdc++.h>
using namespace std;
#define llu unsigned long long
int main() {
	llu val,i,t,l,r;
	set<llu> s;
	val=2;
	while(val<=1e18)
	{
		s.insert(val);
		val*=2;
	}
	set<llu>::iterator it;
	for(it=s.begin();it!=s.end();it++)
	{
		val=*it;
		for(i=3;(i*val)<=1e18;i*=3)
		{
			s.insert(i*val);
		}
	}
	s.insert(1);
	vector<llu> v;
	vector<llu>::iterator it1,it2;
	for(it=s.begin();it!=s.end();it++)
		v.push_back(*it);
		llu ans;
	scanf("%llu",&t);
	while(t--)
	{
		ans=0;
		scanf("%llu%llu",&l,&r);
		it1=lower_bound(v.begin(),v.end(),l);
		if((it1!=v.end())&&((*it1)<=r))
		{
			it2=lower_bound(v.begin(),v.end(),r);
			if((it2==v.end()||((*it2)>r)))
			it2--;
			ans=it2-it1+1;
		}
		printf("%llu\n",ans);
	}
	return 0;
}
