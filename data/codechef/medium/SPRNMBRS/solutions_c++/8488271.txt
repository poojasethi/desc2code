
#include<bits/stdc++.h>

using namespace std;

#define ll long long

ll fun(ll l)
{

    ll ans=0;

    if(l>=1) ans++;

    ll temp=2;

    while(temp<=l)
    {

        ans++;

        ll num=temp*3;

        while(num<=l)
        {

            ans++;
            num=num*3;

        }

        temp=temp*2;

    }

    return ans;

}

int main()
{

    int t;
    ll l,r;

    scanf("%d",&t);

    while(t--)
    {

        scanf("%lld%lld",&l,&r);

        printf("%lld\n",fun(r)-fun(l-1));

    }

    return 0;

}
