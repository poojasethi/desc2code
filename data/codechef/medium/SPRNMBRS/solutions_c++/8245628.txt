#include<bits/stdc++.h>
using namespace std;


typedef long long ll;



void solve()
{
	ll L, R;
	cin>>L >>R;

	ll two = 2ll;
	ll x;

	ll ans = 0;

	while(two <= R)
	{
		x = two;
		while(x <= R)
		{
			if(L <= x && x <= R)
				ans++;
			x *= 3ll;
		}
		two *= 2ll;
	}

	if(L <= 1 && 1 <= R)
		ans++;
	cout<<ans<<endl;
}

int main()
{
	ios_base::sync_with_stdio(false);

    int test;
    cin>>test;

    while(test--)
        solve();

	return 0;
}
