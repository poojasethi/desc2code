#include<bits/stdc++.h>
#define lld long long int
using namespace std;

inline lld F(lld R)
{
    lld p=2,q,ctr=0;
    while(p<=R)
    {
        ctr++;
        q=3;
        while(p*q<=R)
        {
            q*=3;
            ctr++;
        }
        p<<=1;
    }
    return ctr;
}

int main()
{
    int t;
    lld L,R;
    cin>>t;
    while(t--)
    {
        int flag=0;
        cin>>L>>R;
        if(L==1) flag=1;
        cout<<F(R)-F(L-1)+flag<<endl;
    }
    return 0;
}
