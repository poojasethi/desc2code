#include<bits/stdc++.h>

using namespace std;
#define ll long long 
int main(){
	ll l,r;
	int t;
	cin>>t;
	while(t--){
		int ans=0;
		cin>>l>>r;
		ll cur=2;
		while(cur<=r){
			ll  val=cur;
			while(val<=r){
				if(val>=l&&val<=r)
					++ans;
				val*=3;
			}
			cur*=2;
		}
		if(l==1)
			++ans;
		cout<<ans<<endl;
	}
	return 0;
}
