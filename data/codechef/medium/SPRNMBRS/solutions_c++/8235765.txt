	return (e&1) == 0?expo((b*b)%m, e>>1, m): (b*expo((b*b)%m, e>>1, m))%m;}

vll res;

void precomp()
{
	res.PB(1);
	for(LL i = 2; i < (1ll<<60); i *= 2){
		for(LL j = 1; j < (1ll<<60) && j*i < (1ll<<60); j *= 3){
			res.PB(i*j);  
		}
	}
	sort(res.begin(), res.end());
}

int main()
{
	ios_base::sync_with_stdio(false);
	precomp();
	int t;
	cin>>t;
	while(t--){
		LL l, r;
		cin>>l>>r;
		// Find first element greater than curr element
		LL left_lmt = (LL)(upper_bound(res.begin(), res.end(), l-1)-res.begin());
		LL right_lmt = (LL)(upper_bound(res.begin(), res.end(), r)-res.begin());
		cout<<right_lmt-left_lmt<<endl;
	}
	return 0;
}
