#include <bits/stdc++.h>

using namespace std;

#define ll long long int

ll test,l,r;

int main() 
{
	// your code goes here
   cin>>test;
   while(test--)
    {
        cin>>l>>r;
        ll ans = 0;
        if(l<=1 && r>=1)
         ans++;
        
        for(ll i=1;;i++) // power of 2
         {
             ll x = pow(2,i);
             if(x > r)
              break;
             
             for(ll j=0;;j++) // power of 3
              {
                  ll y = pow(3,j);
                  if(x*y>r)
                   break;
                  
                  if(x*y>=l) 
                   ans++;
                   
              }
              
         }
        
        cout<<ans<<endl;
    }

	return 0;
}
