/** Super Numbers **/

#include <bits/stdc++.h>
#define LL long long int

using namespace std;

int main()
{
	
	int T;
	scanf("%d",&T);
	LL L,R,value,current,ans = 0;
	while(T--)
	{
		scanf("%lld %lld",&L,&R);
		value = 2;
		ans = 0;
		while(value <= R)
		{
			current = value;
			while(current <= R)
			{
				if(L<= current && current <= R)
				{
					ans++;
				}
				current*= 3;
			}
			value*= 2;
		}
		
		if(L <= 1)
		{
			ans++;
		}
		
		printf("%lld\n",ans);
	}
	
	return 0;
}
