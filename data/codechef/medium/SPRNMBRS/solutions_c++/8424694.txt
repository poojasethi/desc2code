#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		ll l,r;
		cin>>l>>r;
		ll ans=0;
		if(l==1)
		{
			ans=1;
		}
		ll val=2;
		ll curr=0;
		// ll ans=0;
		while(val<=r)
		{
			curr=val;
			while(curr<=r)
			{
				if(curr>=l)
				{
					ans++;
				}
				curr*=3;
			}
			val*=2;
		}
		cout<<ans<<"\n";
	}
	return 0;
}
