#include <cstdio>

#define ull unsigned long long

ull gcd(ull a,ull b)
{
	ull temp;
	while(b){
		temp = b;
		b = a%b;
		a = temp;
	}
	return a;
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t--){
		ull a,b,c,d,k;
		scanf("%llu%llu%llu%llu%llu",&a,&b,&c,&d,&k);
		ull t1 = gcd(a,b);
		ull t2 = gcd(c,d);
		ull t3 = gcd(t1,t2);
		ull t4 = k/(t1/t3);
		printf("%llu\n",1 + 2 * (t4/t2));	
	}
	return 0;
}
