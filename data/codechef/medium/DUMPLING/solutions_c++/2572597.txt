#include <iostream>
using namespace std;

long long gcd(long long a, long long b) {
  long long temp;
  while(b) {
    temp = b;
    b = a%temp;
    a = temp;
  }
  return a;
}

int main() {
  long long a, b, c, d, k, x, y, ans;
  int t;
  cin >> t;
  while(t--) {
    cin >> a >> b >> c >> d >> k;
    x = gcd(a,b);
    y = gcd(c,d);
    x = x/gcd(x,y);
    x = k/x;
    y = x/y;
    ans = 2*y+1;
    cout << ans << endl;
  }
  return 0;
}
