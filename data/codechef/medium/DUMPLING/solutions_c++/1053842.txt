#include<stdio.h>
#include<assert.h>

#define ll long long

ll GCD(ll a,ll b){ll r; while(a){r=b; b=a; a=r%a;} return b;}

int main(){
  int T;
  ll A, B, C, D, K;
  ll g, res;

  assert( scanf("%d",&T) == 1 );
  assert( 1 <= T && T <= 50000 );

  while(T--){
    assert( scanf("%lld%lld%lld%lld%lld",&A,&B,&C,&D,&K)==5 );
    assert( 1LL <= A && A <= 1000000000000000000LL );
    assert( 1LL <= B && B <= 1000000000000000000LL );
    assert( 1LL <= C && C <= 1000000000000000000LL );
    assert( 1LL <= D && D <= 1000000000000000000LL );
    assert( 1LL <= K && K <= 1000000000000000000LL );

    A = GCD(A, B);
    B = GCD(C, D);
    g = GCD(A, B);
    if( (double)A*B/g > K+100 ) res = 1;
    else                        res = 2 * (K / (A/g*B)) + 1;
    printf("%lld\n",res);
  }

  return 0;
}
