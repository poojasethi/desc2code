#include<iostream>
#include<cstdio>
#include<cmath>
using namespace  std;
#define read(x) scanf("%d",&x)
#define write(x) printf("%d\n",x)
typedef unsigned long long int ull;
 
 
ull gcd(ull a,ull b)
{ 
if(!b)return a;
return gcd(b,a%b);
}
 
 
int main()
{ ull a,b,c,d,k,x,y,z;
  
  int t;
  cin>>t;
 
while(t--)
{
                        cin>>a>>b>>c>>d>>k;
                        x=gcd(a,b);
                        y=gcd(c,d);
                        z=(k/(x/gcd(x,y)))/y;
                        cout<<(ull)(2*z+1)<<endl;
}  
 
//system("pause");
return 0;
}
