#include<iostream>

using namespace std;

long long unsigned gcd(long long unsigned a, long long unsigned b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main(void)
{
 int t;
 long long unsigned a, b, c, d, k, gd1, gd2, gd3, gd, i, count;
 scanf("%d", &t);
 while(t--)
 {
           count=1;
           scanf("%llu %llu %llu %llu %llu", &a, &b, &c, &d, &k);
           gd1=gcd(a, b);
           gd2=gcd(c, d);
           gd3=gd1/gcd(gd1, gd2);
           gd=(k/gd3)/gd2;
           count=gd*2 + 1;
           printf("%llu\n", count);
           }
 return 0;
}
