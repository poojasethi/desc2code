#include<iostream>
#include<algorithm>
using namespace std;
typedef unsigned long long LL;
int T;
LL A,B,C,D,K;

LL gcd(LL a,LL b)
{
	if(b>a)
	return gcd(b,a);
	else
	return b==0?a:gcd(b,a%b);
}

int main()
{
	cin>>T;
	for(int i=0;i<T;i++)
	{
		cin>>A>>B>>C>>D>>K;
		LL n1=gcd(A,B);
		LL n2=gcd(C,D);
		LL den=gcd(n1,n2);
		LL S=n1/den;
		LL R=K/S;
		LL n4=R/n2;
		cout<<(2*n4)+1<<endl;
	}
	return 0;
}
