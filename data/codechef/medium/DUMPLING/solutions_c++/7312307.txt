
#include <iostream>

using namespace std;

long long a,b,c,d,k,t;

long long hcf(long long m,long long n)
{
    if(m==n)
        return n;
    else if(m<n)
    {
        t=m;
        m=n;
        n=t;
    }
    t=m%n;
    while(t>0)
    {
        m=n;
        n=t;
        t=m%n;
    }
    return n;
}

int main()
{
    int test;
    cin >> test;
    while(test--)
    {
        cin >> a;
        cin >> b;
        cin >> c;
        cin >> d;
        cin >> k;
        a=hcf(a,b);
        c=hcf(c,d);
        d=hcf(a,c);
        d=a/d;
        if(d<=k/c)
        {
            d*=c;
            d=k/d;
            d=2*d+1;
        }
        else
        {
            d=1;
        }
        cout << d << "\n";
    }
}
