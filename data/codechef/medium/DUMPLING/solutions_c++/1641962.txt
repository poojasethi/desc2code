#include<cstdio>
long long int gcd(long long int x,long long int y)
{
	while(1)
	{
		if(x>y)
			x=x%y;
		else
			y=y%x;
		if(x==0)
			return y;
		else if(y==0)
			return x;
	}
}
int main()
{
	int t;
	long long int a,b,c,d,k,p,q,ans,lcm,r;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld%lld%lld%lld%lld",&a,&b,&c,&d,&k);
		p=gcd(a,b);
		q=gcd(c,d);
		r=gcd(p,q);
		p=p/r;
		ans=k/q;
		ans/=p;
		ans=2*(ans)+1;
		printf("%lld\n",ans);
	}
	return 0;
}
