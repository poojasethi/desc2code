#include <iostream>
#include <stdio.h>
using namespace std;



long long gcd(long long a,long long b)
{
	return b==0 ? a : gcd(b,a%b);
}

long long lcm(long long a,long long b)
{
	return (a*b)/gcd(a,b);
}

int main(){
	long long t,a,b,c,d,k,g1,g2;
	scanf("%lld",&t);
	if(b>a)
		a=a+b -(b=a) ;
	if(d>c)
		c=c+d -(d=c) ;
	while(t--)
	{
		scanf("%lld %lld %lld %lld %lld",&a,&b,&c,&d ,&k);
		g1=gcd(a,b);
		g2=gcd(c,d);
		if(g2>g1)
			g1=g2+g1 -(g2=g1);
		k=k/g1;
		k*=gcd(g1,g2);
		k/=g2;
		printf("%lld\n",2*k +1);
		//printf("%lld\n", 2*(k/lcm(gcd(a,b),gcd(c,d)))  +1  );
	}
	return 0;
}
