#include<iostream>
#define ll long long 
ll gcd(ll a,ll b){if(b==0)return a;else return gcd(b,a%b);} 
ll lcm(ll a,ll b,ll k){ll r,s;s=(a/gcd(a,b));r=k/s;return r/b;}
main()
{
      ll t,a,b,c,d,k,shi,po,l;
      scanf("%lld",&t);
      while(t--)
      {
         scanf("%lld%lld%lld%lld%lld",&a,&b,&c,&d,&k);
         shi=gcd(a,b);po=gcd(c,d);l=lcm(shi,po,k);
         printf("%lld\n",(2*(l))+1);
      }
      //system("pause");
}
