//Basic Maths :
LL gcd(LL n, LL m) {return m==0? n : gcd(m,n%m);}
LL lcm(LL n, LL m) {return n==0 && m==0? 0 : n*(m/gcd(n,m));}

main()
{
    int t;
    LL a,b,c,d,k,m,n;
    scanf("%d",&t);
    while(t--)
    {
          scanf("%lld %lld %lld %lld %lld",&a,&b,&c,&d,&k);
          m=gcd(a,b);
          n=gcd(c,d);
          k/=(m/gcd(m,n));
          k/=n;
          printf("%lld\n",2*k+1);
    }
    return 0;
}
