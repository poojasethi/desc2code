#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

typedef long long ll;

ll a,b,c,d,k;

ll gcd(ll n1, ll n2) {return n2==0? abs(n1) : gcd(n2,n1%n2);}
ll lcm(ll n1, ll n2) {return n1==0 && n2==0? 0 : abs(n1*n2)/gcd(n1,n2);}

int main()
{
    int t;
    scanf("%d", &t);
    while (t--) {
        scanf("%lld%lld%lld%lld%lld", &a, &b, &c, &d, &k);
        ll g1 = gcd(a,b);
        ll g2 = gcd(c,d);
        ll s = g1/gcd(g1, g2);
        ll ans = k/s;
        ans = ans/g2;
        ans*=2;
        ans++;
        printf("%lld\n", ans);
    }
    return 0;
}
