#include<iostream>
#include<cstdio>
#include<cassert>
#define SI(x) scanf("%d",&x)
typedef long long LL;
using namespace std;

#define MXT 50000
LL a,b,c,d;
LL ans,k;
int t;

LL gcd(LL a,LL b)
{
    if(b==0) return a;
    return gcd(b,a%b);
}

LL lcm(LL a,LL b)
{
    LL ret = 1ll*a*b;
    LL g = gcd(a,b);
    ret/=(1ll*g);
    return ret;
}

int main()
{
    SI(t);
    while(t-- >0)
    {
        scanf("%lld %lld %lld %lld",&a,&b,&c,&d);
        scanf("%lld",&k);

        LL g1 = gcd(a,b);
        LL g2 = gcd(c,d);
        LL g = gcd(g1,g2);
        
        g1/=g;

        ans = (k/g1);
        ans = ans/g2;
        ans = 2ll*ans + 1;
        
        printf("%lld\n",ans);
    }
    return 0;  
}
