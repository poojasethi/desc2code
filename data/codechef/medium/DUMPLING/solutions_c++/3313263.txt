#include<iostream>
using namespace std;
unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
    if(a==0)
        return b;
    else if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
int main()
{
    unsigned long long int t,a,b,c,d,k,temp1,temp2,temp3,lcm,ans,temp;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c>>d>>k;
        temp1=gcd(a,b);
        temp2=gcd(c,d);
        temp3=gcd(temp1,temp2);
        if(temp1<temp2)
        {
            temp=temp1;
            temp1=temp2;
            temp2=temp;
        }
        ans=k;
        ans*=temp3;
        ans/=temp1;
        ans/=temp2;
        ans*=2;
        ans++;
        cout<<ans<<endl;
    }
}
