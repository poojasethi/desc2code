ll gcd(ll a, ll b)
{
    while(a&&b)
    {
        if(a>b) a%=b;
        else b%=a;
    }
    return a+b;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll a,b,x,y,k;
        scanf("%lld%lld%lld%lld%lld",&a,&b,&x,&y,&k);
        x=gcd(x,y);
        y=gcd(a,b);
        ll ans=x/gcd(x,y);
        ans=(k/ans)/y;
        printf("%lld\n",ans<<1^1);
    }
	return 0;
}
