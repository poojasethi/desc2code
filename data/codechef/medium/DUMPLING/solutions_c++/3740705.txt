#include <cstdio>
using namespace std;
long long GCD(long long a,long long b)
{
    if(b==0)
    	return a;
    return GCD(b,a%b);
}

int main()
{
	int t;
	long long a,b,c,d,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld %lld %lld %lld %lld",&a,&b,&c,&d,&k);
		long long g1 = GCD(a,b);
		long long g2 = GCD(c,d);
		g1 = g1 / GCD(g1,g2);
		long long ans = (k / g1);
		ans = ans / g2;
		ans = (2ll * ans) + 1;
		printf("%lld\n",ans);
	}
	return 0;
}
