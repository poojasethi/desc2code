#include<iostream>
using namespace std;
long long gcd(long long a,long long b)
{ if(b==0)
return a;
 else
    return gcd(b,a%b);
  }

  int main()
  { long long r,s,p,q,g1,g2,x,y,l,k,t,count=1;
   cin>>t;
   while(t--)
   { cin>>x>>y;
    g1=gcd(x,y);
    cin>>p>>q;
    g2=gcd(p,q);
    s=gcd(g1,g2);
    s=g1/s;
    cin>>k;

    r=k/s;
    r=r/g2;
    count=1+r*2;
    cout<<count<<endl;
    }
    }
