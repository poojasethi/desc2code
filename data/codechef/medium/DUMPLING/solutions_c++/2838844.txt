long long int GetHCF(long long int firstNum,long long int secondNum)
{
     while (firstNum != 0 && secondNum != 0)
     {
         if (firstNum > secondNum)
         {
            firstNum %= secondNum;
         }
         else
         {
            secondNum %= firstNum;
         }
     }

     if (firstNum == 0)
     {
         return secondNum;
     }
     else
     {
         return firstNum;
     }
}

int main()
{
	long long int t , n , m , i , j ,ans,c,d,a,b,k,mod=1000000000;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld%lld%lld%lld",&a,&b,&c,&d,&k);
		i=GetHCF(a,b);
		j=GetHCF(c,d);
		if(i==1&&j==1)ans=2*k+1;
		else if(i==1&&j!=1)ans=2*(k/j)+1;
		else if(i!=1&&j==1)ans=2*(k/i)+1;
		else 
		{
			n=GetHCF(i,j);
			a=(j/n)/mod;b=(j/n)%mod;
			c=i/mod;d=i%mod;
			if(a*c>1||(a*d+b*c>mod))ans=1;
			else if(a*c*mod+a*d+b*c>mod)ans=1;
			else
				{
					n=i*(j/n);
					ans=2*(k/n)+1;
				}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
