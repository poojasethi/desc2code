using namespace std;

#define i64 long long int

i64 gcd(i64 a, i64 b) {
	i64 t;
	while (b != 0) {
		t = b;
		b = a % b;
		a = t;
	}
	return a;
}

int main() {
	i64 T, A, B, C, D, K;
	scanf("%lld\n", &T);
	while (T--) {
		scanf("%lld %lld %lld %lld %lld", &A, &B, &C, &D, &K);
		i64 gcd1 = gcd(A, B);
		i64 gcd2 = gcd(C, D);
		i64 gcd3 = gcd(gcd1, gcd2);
		i64 R = gcd1 / gcd3;
		i64 Q = K/R;
		i64 F = Q/gcd2;
		printf("%lld\n", F*2+1);
	}
}
