unsigned long long int gcd(unsigned long long int a,unsigned long long int b)
{
      if(a<b)
          return gcd(b,a);
      if(b == 0)
          return a;
      return gcd(b, a%b);
}
main()
{
     int tc;
     cin>>tc;
     while(tc--)
     {
          unsigned long long int a,b,c,d,k,x,y,lcm,ans;
          scanf("%llu %llu %llu %llu %llu", &a, &b, &c, &d, &k);
          x = gcd(a,b);
          y = gcd(c,d);
          ans = (k/(x/gcd(x,y)))/y;
          //ans = k/lcm;
          printf("%llu\n", 1+2*ans);
     }
}
