#define MOD	       	    1000007
#define LU             	long long unsigned int
#define LL             	long long int

#define REP(i,n)        for(int i = 0; i < n; i++)
#define FOR(i,a,b,k)    for(int i = a; i < b; i+=k)

using namespace std;

int ni(){
    int v;
    scanf("%d", &v);
    return v;
}

int ways[21][201][10];

void preProcess(){
    FOR(i,1,10,1){
        ways[1][0][i] = 1;
    }

    FOR(n,2,21,1){
        FOR(w,0,201,1){
            FOR(d,0,10,1){
                FOR(d1,0,10,1){
                    int inc = abs(d1-d);
                    ways[n][w+inc][d] += ways[n-1][w][d1];
                    ways[n][w+inc][d] %= MOD;
                }
            }
        }
    }
}

int main(){
//    freopen("input.txt", "r", stdin);
//    freopen("output.txt", "w", stdout);
    preProcess();
    int t = ni();

    while(t--){
        int n = ni();
        int w = ni();
        int count  = 0;
        for(int i = 0; i < 10; i++){
            count = (count + ways[n][w][i]) % MOD;
        }
        printf("%d\n", count);
    }
}
