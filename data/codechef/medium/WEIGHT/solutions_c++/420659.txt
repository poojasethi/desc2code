#include <iostream>
#include <string.h>
#include <string>
#include <cmath>

#define mod 1000007

using namespace std;

int myabs(int a){
    if(a<0){return -1*a;}
    return a;
}

void init(int count[][201][10]){
    
    
     
     for(int i=2; i<=20; i++){
             for(int j=0; j<=200; j++){
                     for(int append=0; append<10; append++){
                             for(int prev=0; prev<10; prev++){
                                         int diff=myabs(append-prev);
                                         if(diff<=j){
                                                     count[i][j][append]+=count[i-1][j-diff][prev];
                                                     count[i][j][append]%=mod;
                                                     }
                                                     }
                                                     }
                                                     }
                                                     }
             }
                                                     
    

int main()
{int count[21][201][10]; 
     memset(count,0,sizeof(count));
     for(int i=1; i<10; i++){count[1][0][i]=1;}
    init(count);
    int tc;
    int n,w;
    scanf("%d",&tc);
    while(tc--){
                scanf("%d%d",&n,&w);
                int ans=0;
                for(int i=0; i<10; i++){
                        ans+=count[n][w][i];
                        ans%=mod;
                        }
                printf("%d\r\n",ans);
                }
//system("pause");
return 0;
}
