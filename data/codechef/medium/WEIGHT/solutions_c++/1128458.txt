#define FOR(i,a,b) for (long i(a); i < (b); i++)
#define REP(i,n) FOR(i,0,n)
#define SORT(v) sort((v).begin(),(v).end())
#define UN(v) sort((v).begin(),(v).end()),v.erase(unique(v.begin(),v.end()),v.end())
#define CL(a,b) memset(a,b,sizeof(a))
#define pb push_back

long long W[21][10][201];

void calc() {
	FOR(i, 1, 10) {
		W[1][i][0] = 1;
	}
	FOR(i, 2, 21) {
		REP(j, 10) {
			REP(l, 201) {
				REP(k, 10) {
					int diff = abs(j - k);
					W[i][j][l] += l >= diff ? W[i - 1][k][l - diff] : 0;
					W[i][j][l] %= 1000007;
				}
			}
		}
	}
}

long long ans(int n, int w) {
	long long a = 0;
	REP(i, 10) {
		a += W[n][i][w];
	}
	return a % 1000007;
}

int main() {
	int t, n, w;
	calc();
	cin >> t;
	REP(i, t) {
		cin >> n >> w;
		cout << ans(n, w) << endl;
	}
}
