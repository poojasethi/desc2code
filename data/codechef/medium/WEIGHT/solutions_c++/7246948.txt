using namespace std;

int debug=1;
//const int LIM=250;

LL a[22][210][10];

void solve()
	{
	LL i,j,k,l,t;
	
	memset(a,0,sizeof(a));
	
	
	F(i,1,10)
		{
		REP(j,10)
			{
			t=abs(i-j);
			a[2][t][j]++;
			}
		}
		
	F(i,2,20)
		{
		REP(j,202)
			{
			REP(k,10)
				{
				REP(l,10)
					{
					t=abs(k-l);
					a[i+1][j+t][l]=(a[i+1][j+t][l]+a[i][j][k])%MOD;
					}
				}
			}
		}
	
	}
	

int main()
	{
	ios_base::sync_with_stdio(0);
	//cin.tie(0);
	
	solve();
	LL T,n,i,j,k,l,m,x,y,z,s,w;
	
	cin>>T;
	while(T--)
		{
		cin>>n>>w;
		LL ans=0;
		REP(i,10) ans=(ans+a[n][w][i])%MOD;
		cout<<ans<<endl;
		}
	return 0;
	}
