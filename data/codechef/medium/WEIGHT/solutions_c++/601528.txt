#include <iostream>
#include <vector>

using namespace std;

int mem[21][201][10];

int abs(int a) {
    return a<0?-a:a;    
}

int dfs(int n,int w, int l) {

    if (n==0)
        return w==0;
    if (w<0)
        return 0;
        int& ref = mem[n][w][l];
    if (ref!=-1)
        return ref;
    //  cout<<n<<" "<<w<<" "<<l<<endl;
    int cnt=0;
    for (int i=0;i<=9;++i)
        cnt=(cnt+dfs(n-1,w-abs(l-i),i))%1000007;
    return (ref=cnt);
}

int main () {
    int t, n, w, cnt,tt;
    cin >> t;

    for (int i=0;i<21;++i)
        for (int j=0;j<201;++j)
            for (int k=0;k<10;++k)
                mem[i][j][k]=-1;
                            
    while (t--) {
        cin>>n>>w;
        cnt=0;
        for (int i=1;i<=9;++i)
            cnt=(cnt+dfs(n-1,w,i))%1000007;
        cout<<cnt<<endl;
    }
    return 0;
}
