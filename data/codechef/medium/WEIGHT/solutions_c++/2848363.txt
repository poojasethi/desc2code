#include <stdio.h>
#include <algorithm>
#define ll long long
#define mod 1000007
ll A[25][205][11];
int main()
{
int i,j,mul=1,k,x;
int n=2;
for(i=1;i<=9*mul;i++)
{
for(j=0;j<=9;j++)
{
if((j+i)<=9)
A[n][i][j]++;
if(j-i>=1)
A[n][i][j]++;
}
}
for(i=2;i<=20;i++)
{
for(j=1;j<=9;j++)
A[i][0][j]=1;
}
for(n=3;n<=20;n++)
{
mul++;
for(i=1;i<=9*mul;i++)
{
for(j=i;j>=(i-9) && j>=0;j--)
{
x=i-j;
for(k=0;k<=9;k++)
{
if(k+x<=9)
A[n][i][k]=(A[n][i][k]+A[n-1][j][k+x])%mod;
if((k-x)>=0 && (k-x)!=(k+x))
A[n][i][k]=(A[n][i][k]+A[n-1][j][k-x])%mod;
}
}
}
}
int a,b,t;
ll s=0;
scanf("%d",&t);
while(t--)
{
s=0;
scanf("%d %d",&a,&b);
for(i=0;i<=9;i++)
s=(s+A[a][b][i])%mod;
printf("%lld\n",s);
}
return 0;
}
