#include<iostream>
#include<stdio.h>
#include<math.h>
#include<string.h>
#include<vector>
#include<algorithm>
#include<stdlib.h>
#include<set>
#include<map>
#define mod 1000007
using namespace std;

int dp[22][2000][10];

int main()
{
          
          memset(dp,0,sizeof(dp));

          for(int i=1;i<=20;i++)
                  for(int wt=0;wt<=9*(i-1);wt++)
                          for(int k=0;k<=9;k++)
                          for(int l=0;l<=9;l++)
                          {
                                  if(i==1)
                                  {
                                          if(wt==0 and k!=0)dp[i][wt][k]=1;
                                          //else dp[i][wt][k]=0;
                                  }        
                                  
                                  //if(i==2)
                                  dp[i+1][wt+abs(k-l)][l]=(dp[i+1][wt+abs(k-l)][l] + dp[i][wt][k])%mod;
                                  
                                  
                                  //dp[i][wt+abs(k-l)][l]=dp[i][wt+abs(k-l)][l] + dp[i][wt][k];
                          }
          
          
          int t;
          cin>>t;
          while(t--)
          {
          int n,w;
          cin>>n>>w;
          
          int sum=0;
          for(int k=0;k<=9;k++)
                  sum=(sum+dp[n][w][k])%mod;
          
          cout<<sum%mod<<"\n";
          }
          scanf("\n");
          return 0;
}
