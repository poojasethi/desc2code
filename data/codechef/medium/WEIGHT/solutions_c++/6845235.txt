#include <bits/stdc++.h>
using namespace std;
#define gi(x) scanf("%d",&x) 
#define pb push_back
#define mp make_pair
#define all(p) p.begin(),p.end()
int dp[21][201][10];
int mod=1000007;
int main(){
	for(int i=0;i<21;i++){
		for(int j=0;j<201;j++){
			for(int k=0;k<10;k++){
				dp[i][j][k]=0;
			}
		}
	}	
	for(int i=0;i<10;i++){
		for(int j=1;j<10;j++){
			int w=abs(i-j);
			dp[2][w][i]+=1;
		}
	}
	for(int i=3;i<21;i++){
		for(int j=0;j<201;j++){
			for(int k=0;k<10;k++){
				for(int l=0;l<10;l++){
					int w=abs(k-l);
					if(j-w>=0)
						dp[i][j][k]+=dp[i-1][j-w][l];
					dp[i][j][k]%=mod;
				}
			}
		}
	}
	int t,n,w;
	gi(t);
	while(t--){
		gi(n);gi(w);
		int ans=0;
		for(int i=0;i<10;i++){
			ans+=dp[n][w][i];
		}
		ans%=mod;
		printf("%d\n",ans);
	}
}
