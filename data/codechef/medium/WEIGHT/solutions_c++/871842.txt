#include<iostream>
#define mod 1000007ll
using namespace std;

int abs(int a,int b)
{
if(a-b>0)
return a-b;
else
return b-a;
}

int main()
{
int t,n,w,bit;
int a[25][205][11];
long long int ans;
cin>>t;
for(int i=0;i<t;i++)
{
cin>>n>>w;

for(int i=0;i<=n;i++)
{
for(int j=0;j<=w;j++)
{
for(int d=0;d<=9;d++)
{
a[i][j][d]=0;
}
}
}


for(int i=1;i<=n;i++)
{
for(int d=0;d<=9;d++)
{
a[i][0][d]=1;
}
}

for(int i=1;i<=w;i++)
{
for(int d=0;d<=9;d++)
{
bit=0;
if(d+i<=9)
bit++;
if(d-i>=0)
bit++;
a[1][i][d]=bit;
}
}





for(int i=2;i<=n-1;i++)
{
for(int j=1;j<=w;j++)
{
for(int d=0;d<=9;d++)
{
for(int nd=0;nd<=9;nd++)
{
if((j-abs(nd,d))>=0)
{
a[i][j][d]=(a[i][j][d]+a[i-1][j-abs(d,nd)][nd])%mod;
}
//a[i][j][d]=a[i-1][j-abs(nd,d)][nd];
/*if(d-nd>=0)
a[i][j][d]=a[i-1][j-abs(nd,d)][nd];*/
}
}
}
}

{

}
}
}*/


//cout<<a[1][9][0]<<" dcnfsd "<<endl;
ans=0;
for(int i=1;i<=9;i++)
{
ans=(ans+a[n-1][w][i])%mod;
}
cout<<ans<<endl;
}
return 0;
}
