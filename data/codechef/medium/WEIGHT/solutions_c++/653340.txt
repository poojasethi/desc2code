#include <cstdio>

const int MAX=1000007;
int sols[19][201][10];

int main()
{
    for (int i=0; i<=9; i++)
        for(int j=0; j<=9; j++)
        {
            if (i == 0)
                sols[0][i][j] = 1;
            else
            {
                if (j-i >= 0)
                    sols[0][i][j] += 1;
                if (j+i <= 9)
                    sols[0][i][j] += 1;
            }
        }


    for(int n=3; n<=20; n++)
        for(int w=0; w<=9*(n-1); w++)
            for(int k=0; k<=9; k++)
                for(int j=0; j<=9; j++)
                    if ((w-j)>= 0 && (w-j)<=9*(n-2))
                    {
                        if (k-j >= 0)
                        {
                            sols[n-2][w][k] += sols[n-3][w-j][k-j];
                            sols[n-2][w][k] %= MAX;
                        }
                        if (j!=0 && k+j <= 9)
                        {
                            sols[n-2][w][k] += sols[n-3][w-j][k+j];
                            sols[n-2][w][k] %= MAX;
                        }
                    }

    int t, n, w;
    scanf("%d", &t);
    while(t-- > 0)
    {
        scanf("%d %d", &n, &w);
        int cnt = 0;
        for(int i=1; i<=9; i++)
        {
            cnt += sols[n-2][w][i];
            cnt %= MAX;
        }
        printf("%d\n", cnt);
    }
    return 0;
}
