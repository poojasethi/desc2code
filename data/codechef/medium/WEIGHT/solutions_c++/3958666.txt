//weight of numbers code: WEIGHT

#include<stdio.h>
#include<math.h>

long d[20][201][10]={0};
int done=0;

void make()
{
  int w,k;
  for(w=0;w<10;w++)
  {
	for(k=0;k<10;k++)
	{
	 if(k-w>=0 && k-w<=9)
	 d[1][w][k]++;

	 if(k+w>=0 && k+w<=9 && k+w!=k-w)
	 d[1][w][k]++;
	}
  }
 done=1;
}

void func1(int n,int w)
{int i,j,k,s;
 long sum=0;
 if(n<=done)
 {
  for(k=1;k<10;k++)
  sum=(sum+d[n][w][k])%1000007;

  printf("%ld\n",sum);
 }
 else
 {
	for(i=done+1;i<=n;i++)
	{
	 for(j=0;j<=200;j++)
	 {
	  for(k=0;k<10;k++)
	  {
		sum=0;
		for(s=0;s<10;s++)
		{
		    int dif;
		    if(s>=k)
		 dif=(s-k);
		 else dif=(k-s);
		 if(j>=dif)
		 sum=(sum+d[i-1][j-dif][s])%1000007;
}
		d[i][j][k]=sum;
	  }
	 }
	}

	done=n;
	sum=0;
	for(k=1;k<10;k++)
	sum=(sum+d[n][w][k])%1000007;

	printf("%ld\n",sum);
 }
}

int main()
{
 make();
 int t;
 scanf("%d",&t);
 while(t--)
 {
  int n,w;
  scanf("%d",&n);
  scanf("%d",&w);

  func1(n-1,w);
 }
return 0;
}
