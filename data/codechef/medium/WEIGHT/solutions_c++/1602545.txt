#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int MOD = 1000007;
int a[22][11][222];

int main()
{
    memset(a, 0, sizeof a);
    for (int i=1; i<10; i++) {
        a[1][i][0] = 1;
    }
    for (int len=2; len<=20; len++) {
        for (int digit = 0; digit <= 9; digit++) {
            for (int last_digit = 0; last_digit <= 9; last_digit++) {
                for (int diff=0; diff<=200; diff++) {
                    int last_diff = diff-abs(last_digit-digit);
                    if (last_diff>=0) {
                        a[len][digit][diff] += a[len-1][last_digit][last_diff];
                        if (a[len][digit][diff]>=MOD) a[len][digit][diff] -= MOD;
                    }
                }
            }
        }
    }
    int t, n, w;
    scanf("%d", &t);
    while (t--) {
        scanf("%d%d", &n, &w);
        int ans = 0;
        for (int i=0; i<10; i++) {
            ans += a[n][i][w];
            if (ans>=MOD) ans -= MOD;
        }
        printf("%d\n", ans);
    }
    return 0;
}
