#define md 1000007

int dp[21][201][10];

int ans(int n, int w, int d){
    if(n<1 || w<0) return 0;
    if(dp[n][w][d] != -1) return dp[n][w][d];
    int i;
    dp[n][w][d] = 0;
    FOR(i, 0, 9){
        dp[n][w][d] = ((ll) dp[n][w][d] + ans(n-1, w - abs(d-i), i)) % md;
    }
    return dp[n][w][d];
}

int main(){
    int i, j, k, l, t, n, w;
    S(t);
    FOR(i, 0, 20) FOR(j, 0, 200) FOR(k, 0, 9) dp[i][j][k] = -1;
    FOR(j, 0, 200) FOR(k, 0, 9) dp[1][j][k] = (j == 0);
    while(t--){
        S(n); S(w);
        int sum = 0;
        FOR(i, 1, 9){
            sum = ((ll) sum + ans(n, w, i)) % md;
        }
        P(sum);
    }
    SP;
    return 0;
}
