int main()
{
 
  int dp[22][202][10];
  memset(dp,0,22*202*10*sizeof(int));
 for(int j=1;j<=9;j++)
{
dp[1][0][j]=1;
for(int i=1;i<=200;i++)
dp[1][i][j]=0;
}
dp[1][0][0]=0;
  for(int i=2;i<=20;i++)
  {
   for(int j=0;j<=200;j++)
   {
    for(int k=0;k<=9;k++)
    {
      for(int m=0;m<=9;m++)
      {
       int temp=abs(k-m);
       if(j-temp>=0)
        dp[i][j][k]=(dp[i][j][k]+dp[i-1][j-temp][m])%1000007;
      }
    }
   }
  }
  int t;
 scanf("%d",&t);
 while(t--)
 {
  int n,w,ans=0;
  scanf("%d%d",&n,&w);
  for(int i=0;i<=9;i++)
  ans=(ans+dp[n][w][i])%1000007;
  printf("%d\n",ans);
 }
}
