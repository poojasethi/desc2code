#include<iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
#include<cstring>

using namespace std;
#define MOD 1000007

int n,t,w,dp[21][10][201];

int solve()
{
    int i,last,next,diff,W;
    memset(dp,0,sizeof(dp));
    for(i=1;i<=9;i++)
        dp[1][i][0]=1;
    for(i=0;i<n;i++)
    {
        for(W=0;W<=w;W++)
        {
            for(last=0;last<=9;last++)
            {
                for(next=0;next<=9;next++)
                {
                    diff=abs(next-last);
                    if(W+diff<=w)
                    dp[i+1][next][W+diff]=(dp[i+1][next][W+diff]+dp[i][last][W])%MOD;

                }
            }
        }
    }
    int sum=0;
    for(i=0;i<=9;i++)
        sum=(sum+dp[n][i][w])%MOD;
    return sum;
}

int main()
{
    cin>>t;
    while(t--)
    {
        scanf("%d%d",&n,&w);
        printf("%d\n",solve());
    }
}
