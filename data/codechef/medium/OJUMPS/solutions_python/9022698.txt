#!/usr/bin/env python
import sys
import unittest

def solve(N):
    if N == 0:
        return True
    N = N % 6
    if N in [0,1,3,6]:
        return True
    return False

if __name__ == '__main__':
    A = int(sys.stdin.readline().strip())
    if solve(A):
        print "yes"
    else:
        print "no"

class Tests(unittest.TestCase):
    def test1(self):
        pointsReached = [0, 1, 3, 6]
        while len(pointsReached) < 10000:
            base = pointsReached[-1]
            pointsReached.append(base+1)
            pointsReached.append(base+3)
            pointsReached.append(base+6)

        for i in xrange(max(pointsReached)):
            print i
            if i in pointsReached:
                self.assertTrue(solve(i))
            else:
                self.assertFalse(solve(i))
