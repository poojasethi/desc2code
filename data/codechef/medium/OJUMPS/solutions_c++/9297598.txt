#include<bits/stdc++.h>
using namespace std;
int main()
{
   int num[3]={6,3,1};
   long long n;
   cin>>n;
   while(1)
   {
       if(n==num[0]||n==num[1]||n==num[2]||n==0)
       {
           cout<<"yes";
           break;
       }
       else
       {
           if(n==2||n==4||n==5)
           {
               cout<<"no";
               break;
           }
           else
            n%=num[0];
       }
   }
   return 0;
}
