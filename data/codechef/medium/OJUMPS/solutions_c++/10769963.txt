#include <iostream>
#include <string>

using namespace std;

int div(string a, int n){
	int size = a.size();
	int val=0;
	for(int i=0;i<size;i++){
		val = val * 10 + a[i];
		val = val % n;
	}
	return val;
}

int main(){
	string a;
	cin >> a;
	int rem = div(a,6); 
	if(rem == 0 || rem == 1 || rem == 3)
		cout << "yes";
	else
		cout << "no";
	return 0;
}
