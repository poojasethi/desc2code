using namespace std;
typedef long long ll;
bool fxn(ll a)
{
	if(a%6==0)
	{
		return 1;
	}
	else
	{
		return 0;
	}
}
typedef long long ll;
int main()
{
	ll a;
	cin>>a;
	int x = 0,y = 0,z = 0;
	x = fxn(a);
	y = fxn(a-1);
	z = fxn(a-3);
