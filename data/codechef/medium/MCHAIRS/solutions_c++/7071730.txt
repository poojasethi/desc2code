#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long int y;
long long int modular(long long int a,long long int b)
{
    if(b==0)
        return 1;
    else if(b%2==0)
    {
        y=modular(a,b/2);
        return (y*y)%mod;
    }
    else
    {
        return ((a%mod)*modular(a,b-1))%mod;
    }
}
int main()
{
    long long int t,n,p;
    cin>>t;
    while(t--)
    {
        cin>>n;
        p=modular(2,n);
        cout<<(p-1)%mod<<endl;
    }
    return 0;
}
