#include<iostream>
using namespace std;

long long int modexp(long long int a,long long int b,long long int n)
	{
	  long long int	r = 1;
    while (true)
    {
        if (b % 2 == 1)
            r = r * a % n;
        b /= 2;
        if (b == 0)
            break;
        a = a * a % n;
	}
    return r;
	}
    

int main()
{
	int t;
	cin>>t;
	while(t)
	{
		long long int n,sum=1;
		cin>>n;
		sum=modexp(2,n,1000000007);
		cout<<sum-1<<endl;
		t--;
	}
	return 0;
}
