#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
long long int fun(long long int a,long long int power)
{
	long long int p,q;
	if(power==1)
		return a;
	if(power%2==0)
	{
		p=pow(fun(a,power/2),2);
		return p%1000000007;
	}
	//return int(pow(fun(a,power/2),2))%1000000007;
	else
	{
		q=a*pow(fun(a,power/2),2);
		return q%1000000007;
	}
	//return int(a*pow(fun(a,power/2),2))%1000000007;
}
int main()
{
	long long int n,i,j,k,a,power,t;
	scanf("%lld",&t);
	for(i=0;i<t;i++)
	{
		scanf("%lld",&n);
		printf("%lld\n",fun(2,n)-1);
	}
}
