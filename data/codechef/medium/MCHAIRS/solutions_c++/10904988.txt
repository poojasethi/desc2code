#include <bits/stdc++.h>

using namespace std;

#define mod 1000000007LL
long long int modexp(long long int a,long long int b)
{
	if (a == 0)
		return 1;
	long long int ans = 1;
	while (a)
	{
		if (a%2 == 1)
		{
			ans = (ans*b)%mod;
		}
		b = (b*b)%mod;
		a = a/2;
	}
	return ans;
}
int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{	
		long long int n;
		scanf("%lld",&n);
		long long int temp = modexp(n,2)-1;
		temp = temp+mod;
		temp = temp%mod;
		printf("%lld\n",temp);
	}
	return 0;
}
