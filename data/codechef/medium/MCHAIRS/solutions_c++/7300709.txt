#define all(c) c.begin(),c.end()
#define mod 1000000007
using namespace std;
long long power(long long a,long long b)
{
    if(b==0)
        return 1;
    if(b==1)
        return a;
    long long temp=power(a,b/2);
    temp=(temp*temp)%mod;
    if(b%2)
        temp=(temp*a)%mod;
    return temp;
}
int main()
{
    long long t,n,ans;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        ans=power(2,n);
        printf("%lld\n",ans-1);
    }
    return 0;
}
