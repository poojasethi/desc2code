#include <iostream>
#include <math.h>

using namespace std;

#define mod 1000000007

long long int fast_exp(int base, int exp)
{
    if(exp==1)
    return base;
    else
    {
        if(exp%2 == 0)
        {
            long long int base1 = pow(fast_exp(base, exp/2),2);
            if(base1 >= mod)
            return base1%mod;
            else
            return base1;
        }
        else
        {
            long long int ans = (base*  pow(fast_exp(base,(exp-1)/2),2));
            if(ans >= mod)
            return ans%mod;
            else
            return ans;
        }
    }
}

int main()
{
    int t;
    long long n;
    cin>>t;

    while(t--)
    {
    cin>>n;
    cout<<(fast_exp(2,n) - 1)%mod<<'\n';
    }

    return 0;
}
