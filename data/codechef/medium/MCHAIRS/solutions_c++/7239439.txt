#include<stdio.h>
#include<iostream>
#define MOD 1000000007
typedef long long int ll;
 using namespace std;
ll modexp(ll a,ll b)
{
ll x=1;
while(b>0)
{
if(b%2!=0)
x=(x*a)%MOD;
a=(a*a)%MOD;
b/=2;
}
return x%MOD;
}
 
 
int main()
{
int tt,n,ans;
scanf("%d",&tt);
while(tt--)
{
           scanf("%d",&n);
           ans = modexp(2,n);
           printf("%d\n",(ans-1));
           }
           return 0;
}
