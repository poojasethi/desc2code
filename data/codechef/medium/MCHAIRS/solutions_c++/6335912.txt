#include <bits/stdc++.h>

using namespace std;

int m = 1000000007;

long long f(int n)
{
    if(n==0)
        return 1;
    long long ans = f(n/2);
    if(n%2==0)
        return (ans*ans)%m;
    else return (ans*ans*2)%m;
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        cout<<f(n)-1<<endl;
    }
    return 0;
}
