#include <iostream>
#include <algorithm>
using namespace std;
long long int M=1000000007;
long long int fast_pow(long long int a,long long int b)
{
    long long int result=1;
    while(b>0)
    {
        if(b%2==1)
        {
            result=(result*a)%M;
        }
        b=b>>1;
        a=(a*a)%M;
    }
    return result;
}
int main()
{
    long long int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long long int x = fast_pow(2,n);
        cout<<x-1<<"\n";
    }
    return 0;
}
