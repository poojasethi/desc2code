#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define F first
#define S second
#define maxn 1005
#define M 1000000007
using namespace std;



int main ()
{
	int t,n;
	LL ans,b;
	sc(t);
	while(t--)
	{
		sc(n);
		ans=1;
		b=2;
		while(n)
		{
			b=b%M;
			if(n&1)
			{
				ans=(ans*b)%M;
			}
			n=n>>1;
			b=(b*b)%M;
		}
		ans=ans-1;
		if(ans<0)
		{
			ans+=M;
		}
		pl(ans%M);
	}
	return 0;
}
