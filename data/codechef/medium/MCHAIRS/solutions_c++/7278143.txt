#include<bits/stdc++.h>
using namespace std;

const int mod=1000000007;

inline void add(int &a, int b) {
    a+=b;
    if(a>mod) {
        a-=mod;
    }
}

inline int mul(int a, int b) {
    return (long long)a*b%mod;
}

int power(int a, int n) {
    int ans=1;
    while(n) {
        if(n & 1) {
            ans=mul(ans, a);
        }
        a=mul(a, a);
        n>>=1;
    }
    return ans;
}

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n;
        scanf("%d", &n);
        printf("%d\n", power(2, n)-1);
    }
}
