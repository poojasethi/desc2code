#include<iostream>
#include<vector>
using namespace std;
long long int MOD = 1000000007;
long long pow(int a, long long  b,long long  MOD)
{
	long long x=1,y=a; 
	while(b > 0)
	{
		if(b%2 == 1)
		{
			x=(x*y);
			if(x>MOD) x%=MOD;
		}
		y = (y*y);
		if(y>MOD) y%=MOD; 
		b /= 2;
	}
	return x;
}

int main()
{
    long long t;
    cin >> t;
    for(;t>0;t--)
{
    long long int n;
    cin >> n;
  
   long long ans = pow(2,n,MOD);
   cout << ans-1 << endl;
}   
return 0;
}
