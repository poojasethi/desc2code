#include<iostream>
using namespace std;
#define MOD	 1000000007
long long maadu(int n) {
	if (n < 25) {
		return ((long long) (1 << n)) % MOD;
	}
	long long temp = maadu(n / 2) , t=temp;
	if (n & 1)
		t <<= 1;
	t %= MOD;
	return (t * temp) % MOD;
}

int main(int argc, char **argv) {
	int t, n;
	cin >> t;
	while (t--) {
		cin >> n;
		cout << maadu(n) - 1 << endl;
	}
	return 0;
}
