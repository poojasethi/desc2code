#define gc getchar_unlocked
static  long int get_i()
{
    char c=gc();
    while(c<'0'||c>'9')
        c=gc();
    long int ret = 0;
    while(c>='0'&&c<='9')
    {
       ret = ret * 10 + c - '0';
       c=gc();
    }
   
    return ret;
}
