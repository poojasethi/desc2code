#include<bits/stdc++.h>
using namespace std;
#define ll long long

map<ll,ll> m;

ll pwr(ll a)
{
	if(a==1)
	return 2;
	
	if(m[a]!=0)
	return m[a];
	
	if(a%2==0)
	m[a]=( (pwr(a/2)%1000000007) * (pwr(a/2)%1000000007) )%1000000007;
	else
	m[a]=( (pwr(a/2)%1000000007) * (pwr(a/2)%1000000007) * 2 )%1000000007;
	
	return m[a];
}

int main()
{
    ll t,n;
    scanf("%lld",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	printf("%lld\n",pwr(n)-1);
    }
    return 0;
}
