#include<iostream>
#include<cmath>
#include<cstdio>

long long power(long long x, long long y)
{
long long temp;
if( y == 0)
return 1;
temp = power(x, y/2);
if (y%2 == 0)
return (temp*temp) % 1000000007 ;
else
return (((x*temp)%1000000007) * temp) % 1000000007;
}

using namespace std;

int main()
{
	int t;
	long long n;
	
	scanf("%d",&t);
	
	while(t--)
	{
		scanf("%lld",&n);
		long long ans=power(2,n);
		ans=(ans-1)%1000000007;
		printf("%lld\n",ans);	
	}
return 0;
}
