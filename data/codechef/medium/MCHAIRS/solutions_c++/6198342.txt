#include <iostream>
using namespace std;

#define lld long long int

lld mod_pow(lld base, lld exp, lld mod) {
	lld result = 1;
	base = base % mod;
	
	while (exp > 0) {
		if (exp % 2 == 1) {
			result = (result * base) % mod;
		}
		exp >>= 1;
		base = (base * base) % mod;
	}
	
	return result;
}

int main() {
	int tc;
	lld base, exp, mod;
	
	cin >> tc;
	
	while (tc--) {
		base = 2;
//		cin >> base >> exp >> mod;
		cin >> exp;
		mod = 1000000007;
		
		cout << mod_pow(base, exp, mod) - 1 << endl;
		
	}
	return 0;
}
