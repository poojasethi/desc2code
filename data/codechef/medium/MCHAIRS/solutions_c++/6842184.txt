#include <bits/stdc++.h>

using namespace std;
#define MOD 1000000007 ;
long long exp(long long  base, long long  exp) {
    long long res=1;
    while(exp>0) {
       if(exp%2==1) res=(res*base)%MOD;
       base=(base*base)%MOD;
       exp/=2;
    }
    return res%MOD;
}
