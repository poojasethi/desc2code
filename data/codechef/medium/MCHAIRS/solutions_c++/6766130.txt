#include<stdio.h>
#include<iostream>
#define M 1000000007
using namespace std;

long long int powM(long long int x,long long int n)
{
  long long int result = 1, a=x%M, b=n;
  while (b)
  {
    if (b%2==1)
      result = (result*a)%M;
    b /= 2;
    a = (a*a)%M;
  }
  return result;
}

int main(){
    long long t,n;
    scanf("%lld", &t);
    while(t--){
          scanf("%lld", &n);
          printf("%lld\n", powM(2,n)-1);           
    }

    return 0;    
}
