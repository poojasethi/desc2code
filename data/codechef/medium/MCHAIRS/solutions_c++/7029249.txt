#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
long long int power(int a,int b)
{
if(b==0)
return 1;
if(b==1)
return a;
long long int temp=power(a,b/2);
if(b%2==0)
return (temp*temp)%1000000007;
else
return (((temp*temp)%1000000007)*a)%1000000007;
}
main()
{
int t;
scanf("%d",&t);
while(t--)
{
int n;
scanf("%d",&n);
long long int ans=power(2,n)-1;
printf("%lld\n",ans);
}
return 0;
}
