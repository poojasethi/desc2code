ll int pow(ll int a, ll int b)
{
	if(b==0)
	return 1;
	ll int temp=pow(a,b/2)%mod;
	if(b%2==0)
	return ((temp)%mod*(temp)%mod)%mod;
	else
	return (((temp)%mod*(temp)%mod)%mod*(a)%mod)%mod;
}
