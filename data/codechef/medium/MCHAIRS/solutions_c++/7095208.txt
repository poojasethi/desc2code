#include<bits/stdc++.h>
using namespace std;
int main() {
	long long t,n;
	scanf("%lld",&t);
	while(t--) {
		scanf("%lld",&n);
		long long mod = 1000000007;
		long long base = 2,ans=1;
		while(n != 0) {
			if(n%2 != 0)
			ans = (ans * base) % mod;
			base = (base * base) % mod;
			n = n/2;
		}
		ans -= 1;
		printf("%lld\n",ans);
	}
	return 0;
}
