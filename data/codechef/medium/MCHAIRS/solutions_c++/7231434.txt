#include<iostream>
using namespace std;
#define m 1000000007
void cal(long long int base,long long int e)
{
	long long int result=1;
	while(e>0)
	{
		if(e%2==1)
		result=(result*base)% m;
		base=(base*base)%m;
		e/=2;
	}
	cout<<(result%m)-1<<"\n";
}
int main()
{
	long long int test,n;
	cin>>test;
	while(test--)
	{
		cin>>n;
		cal(2,n);
	}
	return 0;
}
