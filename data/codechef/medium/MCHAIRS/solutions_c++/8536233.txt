#include<iostream>
#include<cstdio>
#define MOD 1000000007
using namespace std;
long long int get_exponentiation(int n)
{
    long long int ans = 1;
    long long int tmp = 2;
    while(n != 0)
    {
        if(n % 2 == 1)
            ans *= tmp;
        if(ans >= MOD)
            ans %= MOD;
        tmp *= tmp;
        //printf("%lld %lld\n",tmp,ans);
        if(tmp >= MOD)
            tmp %= MOD;
        n /= 2;
    }
    return ans;
}

int main()
{
  int n,t;
  scanf("%d",&t);
  while(t--)
  {
      scanf("%d",&n);
      long long int ans = get_exponentiation(n);
      ans -= 1;
      if(ans < 0)
        ans += MOD;
      printf("%lld\n",ans);
  }
  return 0;
}
