#include<iostream>
#include<vector>
using namespace std;
typedef long long int ll;
const ll MOD=1e9+7;
ll v[100000005];
ll powr(ll base,ll power,ll M)
{
        ll result=1;
        while (power>0)
        {
                if (power%2==1)
                        result = (result*base)%M;
                base = (base*base)%M;
                power/=2;
        }
        return result;
}
int main()
{
    ll t,n,k;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ll ans=powr(2,n,MOD);
        cout<<ans-1<<'\n';
    }
    return 0;
}
