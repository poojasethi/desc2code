#include<bits/stdc++.h>
#define M 1000000007
using namespace std;
typedef long long ll;
ll powmod(ll a,ll b){
   ll x=1;
   while(b){
      if(b & 1)
         x=(x*a)%M;
      a=(a*a)%M;
      b>>=1;
   }
   return x%M;   
}   
int main(){
  ll t,n,i,d; 
  scanf("%lld",&t);
  while(t--){
     scanf("%lld",&n);
     d=(powmod(2,n))-1;
     if(d<=0)
       d+=M;
     printf("%lld\n",d);
  }
  return 0;
}
