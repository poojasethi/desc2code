#include <stdio.h>
#include <math.h>
long long int fast_exp(long long int base, long long int exp)
{
    if(exp==1)
    return base;
    else
    {
        if(exp%2 == 0)
        {
            long long int base1 = pow(fast_exp(base, exp/2),2);
            if(base1 >= 1000000007)
            return base1%1000000007;
            else
            return base1;
        }
        else
        {
            long long int ans = (base*  pow(fast_exp(base,(exp-1)/2),2));
            if(ans >= 1000000007)
            return ans%1000000007;
            else
            return ans;
        }
    }
}
int main(void) {
	int t;
	long long int n,ans;
	scanf("%d",&t);
	while(t--){
		scanf("%lld",&n);
		ans=fast_exp(2,n);
		printf("%lld\n",ans-1);
	}
	return 0;
}
