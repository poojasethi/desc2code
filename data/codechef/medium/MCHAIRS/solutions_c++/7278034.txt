#include<bits/stdc++.h>
using namespace std;

const int mod=1000000007;

long long power(long long a, long long n) {
    long long ans=1;
    while(n) {
        if(n & 1) {
            ans=(ans*a)%mod;
        }
        a=(a*a)%mod;
        n>>=1;
    }
    return ans;
}

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        long long n;
        scanf("%lld", &n);
        printf("%lld\n", power(2, n)-1);
    }
}
