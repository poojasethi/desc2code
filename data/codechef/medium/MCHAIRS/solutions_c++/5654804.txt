#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;

long long mod_exp(long long a, long long b)
{
	if(b == 0)
                return 1;
	long long x = mod_exp(a, b/2);
	x=(x*x)%MOD;
	if(b%2==1)
                x=(x*a)%MOD;
	return x;
}

int main()
{
        int t;
        long long n;

        cin >> t;
        while(t--)
        {
                cin >> n;
                cout << (mod_exp(2,n)-1+MOD)%MOD << endl;
        }
        return 0;
}
