#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL MOD = 1e9 + 7;
LL power(LL a, LL b){
	LL ans = 1;
	while(b>0){
		if(b&1) ans = (ans * a) % MOD;
		b >>= 1;
		a = (a * a) % MOD;
	}
	return ans;
}
int main(){
	int t;
	cin >> t;
	while(t--){
		int N;
		cin >> N;
		cout << (power(2, N) + MOD - 1) % MOD<< "\n";
	}
	return 0;
}
