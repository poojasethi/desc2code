#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<vector>
#include<algorithm>
#include<limits>
#include <stdio.h>
#include<set>
#include<string.h>
#define ll long long
#define NUM 10000000
int p[NUM+1];
int cnt,n;
using namespace std;
int a[351][351];
void prime()
{ll k=0,i,j;
memset(p,0,sizeof(p));
p[0]=p[1]=-1;
for(i=2;i<NUM+1;++i)
 {if(p[i]==0)
    p[i]=k++;
   for(j=i*i;j<NUM+1;j=j+i)
    p[j]=-1;
 }
}
void dfs(int i,int j)
{ if(i<n && i>=0 && j<n && j>=0)
  {if(a[i][j]!=-1)
    {if(a[i][j]%2==0 and p[a[i][j]]==-1)
      {
        a[i][j]=-1;
        dfs(i-1,j);
        dfs(i+1,j);
        dfs(i,j+1);
        dfs(i,j-1);
      }
    }
  }
}
void dfs1(int i,int j)
{ if(i<n && i>=0 && j<n && j>=0)
   {if(a[i][j]!=-1)
    {if(a[i][j]%2==1 && p[a[i][j]]==-1)
     {
        a[i][j]=-1;

        dfs1(i-1,j);

        dfs1(i+1,j);

        dfs1(i,j+1);

        dfs1(i,j-1);
     }
    }
   }
}
int main()
{
    int t,i,j,k,l;
    prime();
    scanf("%d",&t);
    while(t--)
    {ll res=0;
        scanf("%d",&n);
        
        for(i=0;i<n;++i)
            for(j=0;j<n;++j)
             scanf("%d",&a[i][j]);
             int k=0;
        for(i=0;i<n;++i)
            {
             for(j=0;j<n;++j)
               { if(a[i][j]!=-1)
                  {if(a[i][j]%2==0 && p[a[i][j]]==-1 )
                   {
                       res=res+a[i][j]/2;
                       dfs(i,j);
                    }
                   else if(a[i][j]%2 && p[a[i][j]]==-1)
                   {
                       res+=2+a[i][j]/2;
                      dfs1(i,j);
                    }
               else if(p[a[i][j]]!=-1)
                   {
                     res=res+p[a[i][j]];
                     a[i][j]=-1;
                   }
               
                  }
               }
            }
          printf("%lld\n",res);
      }
}
