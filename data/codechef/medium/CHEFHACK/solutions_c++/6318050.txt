#include <iostream>
using namespace std;

#define MAX 10000000


int prime[MAX+1]={0};
int a[352][352];

void sieve()
{
long long i,j,k=0;

prime[0]=prime[1]=-1;

for(i=2;i<=MAX;i++)
   if(prime[i] == 0)
     {
     prime[i]=k++;
     for(j=i*i;j<=MAX;j+=i)
        prime[j]=-1;
     }
}

void connect(int i,int j,int n,int k) 
{
if(i<n && i>=0 && j<n && j>=0) 
  if(a[i][j]!=-1) 
    if(a[i][j]%2==k && prime[a[i][j]]==-1)
      {
      a[i][j]=-1;
      connect(i-1,j,n,k); 
      connect(i+1,j,n,k);
      connect(i,j-1,n,k);
      connect(i,j+1,n,k);
      }
}

int main() 
{
int t;
scanf("%d",&t);
sieve();
while(t--)
 {
 int n,i,j;
 long long int result=0;
 scanf("%d",&n);
 for(i=0;i<n;i++)
    for(j=0;j<n;j++)
        scanf("%d",&a[i][j]); 
  for(i=0;i<n;i++)
     {
     for(j=0;j<n;j++)
        {
        if(a[i][j]!=-1) 
          {
          if(prime[a[i][j]]==-1) 
            {
            if(a[i][j]%2==0)
              {
              result+=a[i][j]/2;
              connect(i,j,n,0);
              }
            else
              {
              result+=2+a[i][j]/2;
              connect(i,j,n,1);
              }
            }
          else  
           {
           result+=prime[a[i][j]]; 
           a[i][j]=-1; 
           }
          }
        }
    }
 printf("%lld\n",result);
 }
return 0;
}
