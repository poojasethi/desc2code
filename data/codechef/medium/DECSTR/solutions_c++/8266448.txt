#include <iostream>
#include <cmath>
#include <string.h>
#include <stack>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	long t;
	int n;
	cin >> t;
	int i,j;
	char arr[] = {'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
	while(t--)
	{
		cin >> n;
		if (n<26)
		{
			for (i=n;i>=0;i--)
				cout << arr[i];
			cout << endl;
		}
		else
		{
			int cycles = n/25;
			int extra  = n%25;
			if (extra != 0)
			{
				for (i=extra; i>=0; i--)
					cout << arr[i];
			}
			for (j=0;j<cycles;j++)
			{
				for (i=25;i>=0;i--)
					cout << arr[i];
			}
			cout << endl;
		}
	}
	return 0;
}
