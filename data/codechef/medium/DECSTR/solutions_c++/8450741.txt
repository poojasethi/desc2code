#include<iostream>
#include<string>
using namespace std;

int main(){

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int w,i;
    cin>>w;

    while(w--){

        int n;
        cin>>n;

        if(n<=25){
            for(i=n;i>=0;i--)
                cout<<((char)(i+97));

        }
        else if(n>25&&n<=50){
            for(i=n;i>=25;i--)
                cout<<((char)(i+97-25));
            for(i=25;i>=0;i--)
                cout<<((char)(i+97));

        }
        else if(n>50&&n<=75){
            for(i=n;i>=50;i--)
                cout<<((char)(i+97-50));
            for(i=25;i>=0;i--)
                cout<<((char)(i+97));
            for(i=25;i>=0;i--)
                cout<<((char)(i+97));

        }

        else if(n>75&&n<=100){
            for(i=n;i>=75;i--)
                cout<<((char)(i+97-75));
            for(i=25;i>=0;i--)
                cout<<((char)(i+97));
            for(i=25;i>=0;i--)
                cout<<((char)(i+97));
            for(i=25;i>=0;i--)
                cout<<((char)(i+97));

        }
        cout<<endl;
    }
    return 0;
}
