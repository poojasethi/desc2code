#include <iostream>
using namespace std;

int main() {
	int t,k,i,quo,rem;
	cin>>t;
	while(t--)
	{
		cin>>k;
		quo=k/25;
		rem=k%25;
		if(rem!=0)
		{
			for(i=rem;i>=0;i--)
			cout<<(char)(i+97);
		}
		while(quo!=0)
		{
			for(i=25;i>=0;i--)
			cout<<(char)(i+97);
			quo--;
		}
		cout<<endl;
	}	
	return 0;
}
