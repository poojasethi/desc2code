#include <iostream>
#include <cmath>
#include <algorithm>
#include <vector>
#include <cstdio>
#include <cstring>

using namespace std;

int main(){
    int t;cin>>t;
    while(t--){
        int n;cin>>n;
        n = n+(n-1)/25;
        string str = "";
        for(int i=0;i<=n;i++){
            int j = i%26;
            char a = 97+j;
            str = a+str;
        }
        cout<<str<<endl;
    }
}
