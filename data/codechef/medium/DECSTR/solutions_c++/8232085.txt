#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,k;
	scanf("%d",&t);
	
	while(t--){
	char s[120];
	int index=0,x;
		scanf("%d",&k);
		while(k>0){
			if(k>25){
				x=25;
				k=k-25;
			}
			else{
				x=k;
				k=0;
			}
			for(int i='a';i<='a'+x;i++){
				s[index++]=(char)i;
			}
			
		}
		s[index]='\0';
		reverse(s,s+index);
		printf("%s\n",s);
	}
	return 0;
}
