using namespace std;
char arr[26]={'a','b','c','d','e','f','g','h','i','j','k','l','m','n','o','p','q','r','s','t','u','v','w','x','y','z'};
int main()
{
    int t,n,g,q,i,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        g=n/25;
        q=n%25;
        if(g==0)
        {for(i=q;i>=0;i--)
            printf("%c",arr[i]);
        }
        else
        {   if(g!=0 && q!=0)
            for(i=q;i>=0;i--)
            printf("%c",arr[i]);
        }
        for(i=g;i>0;i--)
        {
            for(j=25;j>=0;j--)
                printf("%c",arr[j]);
        }
        printf("\n");
    }
    return 0;
}
