#include <iostream>
using namespace std;

string genString(int k){
		int i; string ans;
		for(i=0;i<k;i++)
				ans+=('a'+k-i);
		ans+='a';
		return ans;
}


int main() {
	// your code goes here
	int t,k;
	cin>>t;
	
	while(t--)
	{
		string ans;
		cin>>k;
		
		while(k>0){
			if(k>=26)
				ans=genString(25)+ans;
			else ans=genString(k)+ans;
			k-=25;
		}
		cout<<ans<<endl;
	
	}
	return 0;
}
