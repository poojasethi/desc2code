#include<bits/stdc++.h>
int main()
{
	int t,k,len,i,pos;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&k);
		if(k%25==0)
		len=(k/25)*26;
		else
		len=((k/25)*26)+(k%25)+1;
		char a[len+1];
		a[len]='\0';
		i=len-1; pos=97;
		while(i>=0)
		{
			a[i]=pos;
			i--;
			if(pos==122)
			pos=97;
			else
			pos++;
		}
		printf("%s\n",a);
	}
}
