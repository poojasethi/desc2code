#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>> t;
    while(t--)
    {
        int k, ext, j = 0;
        cin>> k;
        ext = (k - 1)/25 + 1;
        char str[k + ext + 1];
        for(int i = k + ext - 1; i >= 0; i--)
        {
            str[i] = j + 97;
            j++;
            if(j >= 26)
            {
                j = 0;
            }
        }
        str[k + ext] = '\0';
        cout<< str<< endl;
    }
    return 0;
}
