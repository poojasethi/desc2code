
#include<iostream>

int main()
{
	int T;
	std::cin >> T;
	while(T--)
	{
		int N;
		std::cin >> N;
		int nReminder = N % 25;
		int nQuotient = N / 25;
		if(nReminder > 0)
		{
			char c = 'a' + nReminder;
			while(nReminder >= 0)
			{
				std::cout << c;
				c--;
				nReminder--;
			}
		}
		for(int i = 0; i < nQuotient; i++)
		{
			for(char c = 'z'; c >= 'a'; c--)
			{
				std::cout << c;
			}
		}
		std::cout << std::endl;
	}
	return 0;
}
