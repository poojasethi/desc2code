#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int t,k,i,q,r;
    string s="";
    for(i=25;i>=0;i--)
        s+=char(i+'a');
    cin>>t;
    while(t--) {
        cin>>k;
        q=k/25;
        r=k%25;
        if(r>0)
        for(i=25-r;i<=25;i++) {
            cout<<s[i];
        }
        for(i=0;i<q;i++) cout<<s;
        cout<<"\n";
    }
    return 0;
}
