#include<iostream>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,k;
	
		scanf("%d",&n);
		k=n;
		n=26*(n/25);
		if(k%25!=0)
		n=n+k%25+1;
	//	printf("%d",n);
	    char a[n];
		for(i=0;i<n;i++)
		{
			a[i]=i%26+97;
		}
		for(i=n-1;i>=0;i--)
		printf("%c",a[i]);
		printf("\n");
	}
}
