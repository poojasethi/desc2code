from sys import stdin

def suchString(K):
	numLoop = K / 25
	leftover = K % 25
	reverseString = 'zyxwvutsrqponmlkjihgfedcba'
	if leftover != 0:
		return reverseString[25-leftover:] + reverseString*numLoop 
	else:
		return reverseString*numLoop

T = int(stdin.readline())
for _ in range(T):
	K = int(stdin.readline())
	print suchString(K)
