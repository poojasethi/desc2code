def solve(n):
    counter = 0
    temp = ord('a') + n
    string = ''
    while counter <= n:
        string += chr(temp)
        temp -= 1
        counter += 1
    return string

t = int(raw_input())
for case in range(t):
    n = int(raw_input())
    if (n % 25 == 0):
        print solve(25) * (n / 25)
    else:
        print solve(n % 25) + (solve(25) * (n / 25))
