#include<iostream>
#include<cstdio>
using namespace std;
int main()
{

    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++)
    {
         long long int n,k;
        scanf("%lld %lld",&n,&k);
        if(n!=0 && k!=0)
        {
        printf("%lld %lld\n",n/k,n%k);
        }
        else if(n>0 && k==0)
            printf("%lld %lld\n",0ll,n);
        else
            printf("%lld %lld\n",0ll,0ll);
    }
    return 0;
}
