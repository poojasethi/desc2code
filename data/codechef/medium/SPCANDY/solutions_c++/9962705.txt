#include <iostream>


using namespace std;

int main(int argc, char const *argv[])
{
    int t;
    long long n, k, candiesPerChild, leftOvercandies;

    cin>>t;

    while(t--)
    {
        cin>>n>>k;

        if(k > 0)
        {
            candiesPerChild = n/k;
            leftOvercandies = n%k;
        }
        else
        {
            candiesPerChild = 0;
            leftOvercandies = n;
        }

        cout<<candiesPerChild<<" "<<leftOvercandies<<"\n";
    }

    return 0;
}
