#include<bits/stdc++.h>

using namespace std;
typedef long long int lli;
int main(void)
{
    int t; scanf("%d",&t);
    while(t--)
    {
        lli n,k;
        scanf("%lld %lld",&n,&k);
        
        lli stud,teach;
        if(n==0)
        {
           stud=0;
           teach=0;
        }
        else if(n!=0 && k==0)
        {
           stud=0;
           teach=n;
        }
        
       else if(n!=0 && k!=0)
       {
           stud=n/k;
           teach=n%k;
       }
       
       printf("%lld %lld\n",stud,teach);
    }
    
    return 0;
}
