#define p(n) printf("%d\n",n)
#define LLI long long int
#define R return 0
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		LLI n,k,flag,count; S(n); S(k);
		if(n==0)
		{
			flag=0;
			count=0;
		}
		else if(k==0)
		{			
			flag=0;
