#include<bits/stdc++.h>
using namespace std;
int main()
 {
 long long int n,k,tc,sc;
 int t;
 cin>>t;
 while(t>0)
     {
          
     cin>>n>>k;
     if(k==0)
       {
       sc=0;
       tc=n;
       cout<<sc<<" "<<tc<<endl;
       }
     else
     {  
     sc=n/k;
     tc=n%k;
     cout<<sc<<" "<<tc<<endl;
     }
     t--;
     }
 }
