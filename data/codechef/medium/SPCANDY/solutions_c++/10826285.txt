    #include<iostream>
    using namespace std;
    int main(){
    	int t;
    	cin>>t;
    	unsigned long long int n,k;
    	while(t--){
    		cin>>n>>k;
    		if(n==0&&k==0)
    		cout<<0<<" "<<0;
    		else if(k==0)
    		cout<<0<<" "<<n;
    		else if(n==0)
    		cout<<0<<" "<<0;
    		else
    		cout<<n/k<<" "<<n%k;
    		cout<<endl;
    	}
    	return 0;
    }
