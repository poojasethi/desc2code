
#include<stdio.h>

int main(void)
{
    
    unsigned long long int N;
    unsigned long long int K;
    
    unsigned long long int Q;
    unsigned long long int R;    
    
    int inputs;
    
    scanf("%d", &inputs);
    
    for( int i = 1; i <= inputs; i++ )
    {
        
        scanf("%llu", &N);
        scanf("%llu", &K);
        
        if(K == 0)
        {
            printf("0 %llu\n", N);
        }
        else
        {
            Q = N / K;
            R = N % K;
            
            printf("%llu %llu\n", Q, R);         
        }
        
    }
	
    return 0;
	
}
