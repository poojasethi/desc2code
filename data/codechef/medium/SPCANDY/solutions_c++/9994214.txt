#include <cstdio>
#include <iostream>

using namespace std;

int main()
{
    long long int t,n,k,s,te;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld",&n,&k);
        s = n/k;
        te = n%k;
        if(k != 0)
            printf("%lld %lld\n",s,te);
        else
            printf("0 %lld\n",n);
    }
}
