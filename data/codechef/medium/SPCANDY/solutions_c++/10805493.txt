#include <iostream>
using namespace std;
int main() 
{
    long long int n,k;
    int t;
    cin>>t;
    while(t>0)
    {
        cin>>n>>k;
        if(k==0)
        {
            cout<<"0 "<<n<<"\n";
        }
        else
        {
            cout<<n/k<<" "<<n%k<<"\n";
        }
        t--;
    }
	return 0;
}
