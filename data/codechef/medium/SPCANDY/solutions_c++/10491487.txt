#include<bits/stdc++.h>
using namespace std;

int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int p,q;
		cin>>p>>q;
		if(q==0&&p==0)
		{
			cout<<"0"<<" "<<"0"<<endl;
			
		}
	 else if( q==0 && p!=0)
        printf("0 %lld\n",p);
        else if (q!=0 && p==0)
            printf("0 0\n");
		else	{
	long long int r=0,s=0;
		r=p/q;
		if(p%q==0)
		{
			s=0;
		}
		else
		{
			s=p-r*q;
		}
		cout<<r<<" "<<s<<endl;}
	}
}
