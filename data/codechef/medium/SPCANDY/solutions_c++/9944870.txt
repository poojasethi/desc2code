#include <iostream>
#include<stdio.h>

using namespace std;

int main() {
	long long n,k,t,t2;
	int ts;
	scanf("%d",&ts);
	while(ts--){
		scanf("%lld %lld",&n,&k);
		if(n==0)
		printf("0 0\n");
		else if(k==0 && n!=0)
		printf("0 %lld\n",n);
		else
		{
			t=n%k;
			t2=n/k;
			printf("%lld %lld\n",t2,t);
		}
	}
	return 0;
}
