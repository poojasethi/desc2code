#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t,n,m,count;
	string j,s; 
	cin>>t;
	while(t--){
		cin>>j>>s;
		n = j.size();
		m = s.size();
		count = 0;
		for(int i=0;i<m;i++){
			for(int k=0;k<n;k++){
				if(s[i]==j[k]){
					count++;
					break;
				}
			}
		}
		cout<<count<<"\n";
	}
	return 0;
}
