#include<iostream>
using namespace std;
int main()
{
	int t,i,c,arr[100];
	char str1[101],str2[101];
	cin>>t;
	while(t--)
	{
		for(i=0;i<52;i++)
		arr[i]=0;
		cin>>str1;
		for(i=0;str1[i]!='\0';i++)
		{
			if(str1[i]<='Z'&&str1[i]>='A')
			arr[str1[i]-65]=1;
			else if(str1[i]<='z'&&str1[i]>='a')
			arr[str1[i]-97+26]=1;
		}
		
		cin>>str2;
		c=0;
		for(i=0;str2[i]!='\0';i++)
		{
			if(str2[i]<='Z'&&str2[i]>='A'&&arr[str2[i]-65]==1)
			c++;
			else if(str2[i]<='z'&&str2[i]>='a'&&arr[str2[i]-97+26]==1)
			c++;
		}
		cout<<c<<endl;
	}
	return 0;
}
