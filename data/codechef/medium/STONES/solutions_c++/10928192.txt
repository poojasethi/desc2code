#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,j,x;
    string s,ss;
    cin>>t;
    while(t--)
    {
        cin>>s>>ss;
        int a[100],b[100];
        for(i=0;i<100;i++)
        {
            a[i]=0;
            b[i]=0;
        }
        for(i=0;i<s.length();i++)
        a[s[i]-'A']++;
        for(i=0;i<ss.length();i++)
        b[ss[i]-'A']++;
        x=0;
        for(i=0;i<100;i++)
        {
            if(b[i]!=0&&a[i]!=0)
            x+=b[i];
        }
        cout<<x<<endl;
    }
    return 0;
}
