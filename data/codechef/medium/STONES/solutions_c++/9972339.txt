#include<iostream>
using namespace std;
int main(){
	int t,i,count,c,j;
	char ch;
	string k,s;
	cin>>t;
	while(t--){
		cin>>k>>s;
		for(i=0,c=0;i<s.length();i++){
			ch=s[i];
			for(j=0,count=0;j<k.length();j++){
				if(ch==k[j]){
					c=c+1;
					break;
				}
			}
		}
		cout<<c<<endl;
	}
	return 0;
}
