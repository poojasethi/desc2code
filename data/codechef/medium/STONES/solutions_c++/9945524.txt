#include<iostream>
using namespace std;
int main()
{
  int t,count;
  cin>>t;
  string j,s;
  while(t--)
  {
      count=0;
      bool a[26]={0};
      bool A[26]={0};
      cin>>j>>s;
      for(int i=0;j[i]!='\0';i++)
      {
          if(j[i]>='a' && j[i]<='z')
            a[j[i]-97]=1;
           if(j[i]>='A' && j[i]<='Z')
            A[j[i]-65]=1;
      }

      for(int i=0;s[i]!='\0';i++)
      {
           if((s[i]>='a' && s[i]<='z' && a[s[i]-97])|| (s[i]>='A' && s[i]<='Z' && A[s[i]-65]))
            count++;
      }
    cout<<count<<"\n";
  }
}
