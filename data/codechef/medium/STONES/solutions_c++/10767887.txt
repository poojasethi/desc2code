#include <iostream>
#include <string>
#include <string.h>
using namespace std;
int main(){
	int T;
	cin >> T;
	string J,S;
	int count[256];
	for(int t=0;t<T;t++){
		cin >> J >> S;
		memset(count,0,sizeof(count));
		int sJ = J.size();
		int sS = S.size();
		int ans = 0;
		for(int i=0; i<sJ; i++)
			count[J[i]]++;
		for(int i=0; i<sS; i++){
			if(count[S[i]]){
				//count[A[i]]--;
				ans++;
			}
		}
		cout << ans <<endl;
	}
}
