#include<iostream>
#include<stdio.h>
#include<string>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        string s1,s2;
        cin>>s1>>s2;
        int i,j,count=0;
        for(i=0;i<s1.length();i++)
            for(j=0;j<s2.length();j++)
            if(s1[i]==s2[j])
        {
            count++;
            s2[j]='!';
        }
        cout<<count<<"\n";
    }
    return 0;
}
