#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin >> t;

	for(int i = 0; i < t; i++){
		char a[1000];
		char b[1000];
		cin >> a;
		cin >> b;
		sort(a, a+strlen(a));
		sort(b, b+strlen(b));
		int j = 0;
		int k = 0;
		int count = 0;
		while( j < strlen(a) && k < strlen(b)){
			if(a[j] == b[k]){
				k++,count++;
			}
			else if(a[j] < b[k]){
				j++;
			}
			else{
				k++;
			}
		}
		cout << count << "\n";
	}
	return 0;
}
