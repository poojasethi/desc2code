#include<iostream>
using namespace std;

int main()
{
	int T;
	cin >> T;
	while(T--)
	{
		char J[101], F[101];
		int count = 0;
		cin >> J;
		cin >> F;
		for(int i =0; J[i]!='\0';i++)
		{
			for(int j =0; F[j]!='\0'; j++)
			{
				if(F[j]==J[i])
				{
					count++;
					F[j] = '1';
				}
			}
		}
		cout << count << endl;
	}
	return 0;
}
