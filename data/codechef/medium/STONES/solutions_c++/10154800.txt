#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<vector>

using namespace std;

int main()
{
    int t;
    char j[110];
    char s[110];
    scanf("%d",&t);
    while(t--)
    {
        int count=0;
        scanf("%s",&j);
        scanf("%s",&s);
        for(int i=0;i<strlen(s);i++)
        {
            for(int k=0;k<strlen(j);k++)
            {
                if(s[i]==j[k])
                {
                    count++;
                    break;
                }
            }
        }
        printf("%d\n",count);
    }
    return 0;
}
