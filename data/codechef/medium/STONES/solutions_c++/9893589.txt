#include <bits/stdc++.h>
using namespace std ;
int main ()
{
    int t;
    cin>>t;
    while(t--)
    {
      string s1,s2;
      cin>>s1;
      cin>>s2;
      int count = 0;
      for(int i = 0;i<s2.length();i++)
      {
          for(int  j = 0 ;j<s1.length();j++)
          {
              if(s2[i]==s1[j])
              {
                count++;
                break;
              }
              else
              {
                  continue;
              }
          }
      }
      cout<<count<<endl;
    }
}
