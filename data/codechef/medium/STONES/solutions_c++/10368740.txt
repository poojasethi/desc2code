#include<iostream>
#include<string>
using namespace std;

#define ll long long;

int main()
{
	int t;
	cin>>t;
	while(t--){
		string j,s;
		cin>>j>>s;
		int count=0;
		for (int i=0;i<s.length();i++){
			for (int k=0;k<j.length();k++){
				if (s[i]==j[k]){
					count++;
					break;
				}
			}
		}
		cout<<count<<'\n';
	}
	return 0;
}
