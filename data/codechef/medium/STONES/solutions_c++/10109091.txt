#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		char str1[1005], str2[1005],count=0,i,j;
		scanf("%s", str1);
		scanf("%s", str2);
		int n= strlen(str1);
		int m = strlen(str2);
		for(i=0; i<m; i++){
			for(j=0; j<n; j++){
				if(str2[i]==str1[j]){
					count++;
					//printf("%c ", str1[i] );
					break;
				}
			}
		}
		printf("%d\n", count );
	}	
}
