#include<stdio.h>
int main()
{   long long int t;
	scanf("%lld",&t);
	while(t--)
	{   char str1[1001],str2[1001];
	    long long int i,j,k=0;
		scanf("%s",str1);
		scanf("%s",str2);
		for(i=0;str1[i]!='\0';i++)
		{
			for(j=0;str2[j]!='\0';j++)
			{
				if(str1[i]==str2[j])
				{  str2[j]='@';
					k++;
				}
			}
		}
		printf("%lld\n",k);
	}
	return 0;
}
