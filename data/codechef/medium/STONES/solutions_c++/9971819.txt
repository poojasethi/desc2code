            #include<bits/stdc++.h>
            int main()
            {
                int t;
                 scanf("%d",&t);
                int i=0;
                for(i=0;i<t;i++)
                {
                    char j[1000];
                    char s[1000];
                    scanf("%s",j);
                    scanf("%s",s);
                    int l1=strlen(j);
                    int l2=strlen(s);
                    int k=0;
                    int l=0;
                    int sum=0;
                    for(k=0;k<l2;k++)
                    {
                        for(l=0;l<l1;l++)
                        {
                            if(s[k]==j[l])
                            {
                                sum=sum+1;
                                break;
                            }
                        }
                    }
                    printf("%d\n",sum);
                }
                return 0;
            }
