#include <iostream>
#include<algorithm>
#include<string.h>
using namespace std;
 
int main() {
	int k,i,n,count,m;
	string s;
	char j[100];
	cin>>n;
	while(n--)
	{
		count=0;
	  cin>>j>>s;
	  m=strlen(j);
	  for(i=0;i<s.size();i++)
	  {
	  	for(k=0;k<m;k++)
	  	{
	  		if(s[i]==j[k])
	  		{
	  		 count++;
	  		 break;
	  		}
	  	}
	  }
	  cout<<count<<endl;
	}
	return 0;
}
