#include <iostream>

#include<string>

using namespace std;

int main()
{
int T;
cin>>T;
cin.ignore();
while(T--)
{
	int count=0;
	bool jay[300]={0};
	string J,S;
	int i;
	cin>>J>>S;
	for(i=0;i<J.length();i++)
	jay[J[i]]=true;
	
	for(i=0;i<S.length();i++)
	if(jay[S[i]]==true)
	count++;
	
	cout<<count<<endl;
}
	return 0;
}
