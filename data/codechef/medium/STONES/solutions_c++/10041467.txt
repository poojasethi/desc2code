#include <iostream>
#include <string>

using namespace std;

int main(int argc, char const *argv[])
{
    int t, i, len_s, count;

    cin>>t;

    while(t--)
    {
        string j,s;

        cin>>j>>s;

        len_s = s.length();
        count = 0;

        for(i = 0; i < len_s; ++i)
        {
            if(j.find(s[i]) != -1)
                ++count;
        }

        cout<<count<<"\n";
    }
    return 0;
}
