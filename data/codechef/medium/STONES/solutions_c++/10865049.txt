#include<cstdio>
#include<cstring>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        char J[105],S[105];
        int arr[130]={0};
        scanf("%s %s",&J,&S);
        int Jn=strlen(J);
        int Sn=strlen(S);
        for(int i=0;i<Jn;i++){
            arr[(int)J[i]]=1;
        }
        int count=0;
        for(int i=0;i<Sn;i++){
            if(arr[(int)S[i]]){
                count++;
            }
        }
        printf("%d\n",count);
    }
}
