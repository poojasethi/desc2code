#include <iostream>
#include<stdio.h>
using namespace std;

int main() {
	int t,c;
	char a[101],b[101],e[2];//take array size one extra for '\0'
	cin>>t;
	gets(e);
	while(t--)
	{
		gets(a);
		gets(b);
		c=0;
		for(int i=0;b[i]!='\0';i++)
		{
			for(int j=0;a[j]!='\0';j++)
			{
				if(b[i]==a[j])
				{
					c++;
					break;
				}
			}
		}	
		cout<<c<<endl;	
	}
	return 0;
}
