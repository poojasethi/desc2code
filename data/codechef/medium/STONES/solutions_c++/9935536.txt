#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string j,s;
        cin>>j>>s;
        int hash1[256]={0};
        int hash2[256]={0};
        for(int i=0;i<j.length();i++)
            hash1[j[i]]++;
        for(int i=0;i<s.length();i++)
            hash2[s[i]]++;
        int res=0;
        for(int i=0;i<256;i++)
        {
            if(hash2[i]>0 && hash1[i]>0)
                res+= hash2[i];
        }
        cout<<res<<endl;
    }
    return 0;
}
