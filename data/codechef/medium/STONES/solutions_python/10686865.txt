def test():
    #Jewels and Stones
    matches = input()
    for _ in range(matches):
        jewels = raw_input()
        mines = raw_input()

        dict_jewel = {e:0 for e in jewels}
        for e in mines:
            if e in dict_jewel.keys():
                dict_jewel[e] += 1

        print sum(dict_jewel.values())
 
test()
