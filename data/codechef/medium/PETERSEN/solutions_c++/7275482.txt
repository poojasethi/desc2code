/* MAIN PROGRAM STARTS HERE */

map < pair<char,char>,LL > M;
 
void solve()
	{
	M[mp('A','B')]=0;
	M[mp('A','C')]=5;
	M[mp('A','D')]=5;
	M[mp('A','E')]=0;
	M[mp('B','A')]=1;
	M[mp('B','C')]=1;
	M[mp('B','D')]=6;
	M[mp('B','E')]=6;
	M[mp('C','A')]=7;
	M[mp('C','B')]=2;
	M[mp('C','D')]=2;
	M[mp('C','E')]=7;
	M[mp('D','A')]=8;
	M[mp('D','B')]=8;
	M[mp('D','C')]=3;
	M[mp('D','E')]=3;
	M[mp('E','A')]=4;
	M[mp('E','B')]=9;
	M[mp('E','C')]=9;
	M[mp('E','D')]=4;
	}
	
int main()
	{
	//freopen("TC1.txt","r",stdin);
	//freopen("output1","w",stdout);
	
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	
	LL T,n,i,j,k,l,m,x,y,z,p;
	solve();
	cin>>T;
	while(T--)
		{
		string s;
		
		cin>>s;	
		n=s.length();
		
		if (n==1)
			{
			cout<<s[0]-'A'<<"\n";
			continue;
			}
		y=-1;
		REP(i,n-1)
			{
			if (s[i]!=s[i+1]) 
				{
				y=i;
				break;
				} 
			}
		
		if (y==-1)
			{
			x=s[0]-'A';
			cout<<x;
			F(i,1,n)
				{
				if (x<5) x=x+5;
				else x-=5;
				cout<<x;
				}
			cout<<"\n";
			continue;
			}
		//trace1(y);
		
		LL ans[n];
		bool flag=false;
		ans[y]=M[mp(s[y],s[y+1])];
		
		for (i=y+1; i<n; i++)
			{
			x=s[i]-'A';
			if (ans[i-1]<5)
				{
				if (s[i]!=s[i-1]) ans[i]=x;
				else ans[i]=x+5;
				}
			else
				{
				if (s[i]!=s[i-1]) ans[i]=x+5;
				else ans[i]=x;
				}
			//trace3(i,ans[i-1],ans[i]);
			if (M[mp(s[i-1],s[i])]!=ans[i-1] && s[i]!=s[i-1])
				{
				flag=true;
				break;
				}
			}
				
		if (flag)
			{
			cout<<-1<<"\n";
			continue;
			}
		
		for (i=y-1; i>=0; i--)
			{
			if (ans[i+1]<5) ans[i]=ans[i+1]+5;
			else ans[i]=ans[i+1]-5;
			}
				
		REP(i,n)
			{
			cout<<ans[i];
			}
		cout<<"\n";
		
		}	
	
	
	
	return 0;
	}
