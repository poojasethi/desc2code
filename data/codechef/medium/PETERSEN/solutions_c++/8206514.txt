#include<bits/stdc++.h>
using namespace std;
int a[10][10];
int make(int level,string s)
{
string str="";
int l=s.size();
int v=s[0]-65,i;
if(level==0)
str=str+char(v+48);
else if(level==1)
{
v+=5;
str=str+char(v+48);
}
for(i=1;i<l;i++)
{
if(a[v][s[i]-65])
{
	str+=char(s[i]-65+48);
	v=s[i]-65;
}
else if(a[v][s[i]-60])
{
	str+=char(s[i]-60+48);
	v=s[i]-60;
}
else
return 0;	
}
cout<<str<<endl;
return 1;
}
int main()
{
	int l,t;
	for(int i=0;i<10;i++)
	{
		for(int j=0;j<10;j++)
		a[i][j]=0;
	}
    a[0][5]=a[5][0]=1;
    a[0][4]=a[4][0]=1;
    a[0][1]=a[1][0]=1;
    a[4][9]=a[9][4]=1;
    a[1][6]=a[6][1]=1;
    a[5][8]=a[8][5]=1;
    a[5][7]=a[7][5]=1;
    a[9][6]=a[6][9]=1;
    a[9][7]=a[7][9]=1;
    a[6][8]=a[8][6]=1;
    a[4][3]=a[3][4]=1;
    a[1][2]=a[2][1]=1;
    a[3][2]=a[2][3]=1;
    a[8][3]=a[3][8]=1;
    a[2][7]=a[7][2]=1;
    cin>>t;
    string s;
    while(t--)
    {
    cin>>s;
    if(!make(0,s))
    {
    	if(!make(1,s))
    	{
    		printf("-1\n");
    	}
    }
    }
	return 0;
}
