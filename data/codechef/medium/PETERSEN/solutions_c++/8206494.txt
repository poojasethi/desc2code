#include<bits/stdc++.h>
using namespace std;
int a[10][10];
char str[100000+5],s[100000+5];
int traverse(int level)
{
    int i, vert, v;
    vert = s[0] - 'A';
    if (level == 1)
        vert += 5;
    str[0] = '0' + vert;
    
    for (i = 1; s[i]; ++i)
    {
        v = s[i] - 'A';
        if (a[vert][v] || a[vert][v+=5])
        {
            str[i] = '0' + v;
            vert = v;
        }
        else
        {
            return 0;
        }
    }
    str[i] = '\0';
    printf("%s\n", str);
    return 1;
}
 
int main()
{
	int l,t;
	for(int i=0;i<10;i++)
	{
		for(int j=0;j<10;j++)
		a[i][j]=0;
	}
    a[0][5]=a[5][0]=1;
    a[0][4]=a[4][0]=1;
    a[0][1]=a[1][0]=1;
    a[4][9]=a[9][4]=1;
    a[1][6]=a[6][1]=1;
    a[5][8]=a[8][5]=1;
    a[5][7]=a[7][5]=1;
    a[9][6]=a[6][9]=1;
    a[9][7]=a[7][9]=1;
    a[6][8]=a[8][6]=1;
    a[4][3]=a[3][4]=1;
    a[1][2]=a[2][1]=1;
    a[3][2]=a[2][3]=1;
    a[8][3]=a[3][8]=1;
    a[2][7]=a[7][2]=1;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%s",s);
    if(!traverse(0))
    {
    	if(!traverse(1))
    	{
    		printf("-1\n");
    	}
    }
    }
	return 0;
}
