#include <iostream>
#include <string>

using namespace std;

int main() {
	int t, i;
	cin>>t;
	while(t--) 
	{
		string str;
		cin>>str;
		bool st_array[100001];
		int connection[10][3] = {{1, 5, 4},
								{0, 6, 2},
								{1, 7, 3},
								{2, 8, 4},
								{0, 9, 3},
								{0, 7, 8},
								{1, 8, 9},
								{2, 9, 5},
								{3, 5, 6},
								{4, 6, 7}};
		st_array[0] = true;
		int count = 1;
		int index = str[0] - 65;
		while(count < str.length()) {
			for(i = 0; i < 3; i++) {
				if((connection[index][i] % 5) == (str[count] - 65))
				{
					index = connection[index][i];
					if(index > 4)
						st_array[count] = false;
					else
						st_array[count] = true;
					break;
				}
			}
			if(i == 3) {
				if(st_array[0] == true) {
					st_array[0] = false;
					count = 0;
					index = str[0] - 60;
				}
				else {
					break;
				}
			}
			count++;
		}
		if(count == str.length())
		{
			for(i = 0; i < str.length(); i++) {
				if(st_array[i] == true)
					cout<<str[i] - 65;
				else
					cout<<str[i] - 60;
			}
		}
		else
			cout<<-1;
		cout<<endl;
	}
	return 0;
}
