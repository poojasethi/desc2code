#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

pair<int,int>ma[10];
int graph[10][10];
vector<int>v;
int main()
{
    int test,i,in,x,y;
    string s;
    scanf("%d",&test);
    for(i=0;i<=9;i++)
    {
        if(i<=4)
            ma[i%5].first=i;
        else
            ma[i%5].second=i;
    }
    graph[0][1]=1;graph[0][5]=1;graph[0][4]=1;
    graph[1][2]=1;graph[1][6]=1;graph[1][0]=1;
    graph[2][1]=1;graph[2][3]=1;graph[2][7]=1;
    graph[3][2]=1;graph[3][4]=1;graph[3][8]=1;
    graph[4][3]=1;graph[4][0]=1;graph[4][9]=1;
    graph[5][0]=1;graph[5][8]=1;graph[5][7]=1;
    graph[6][1]=1;graph[6][9]=1;graph[6][8]=1;
    graph[7][2]=1;graph[7][5]=1;graph[7][9]=1;
    graph[8][3]=1;graph[8][5]=1;graph[8][6]=1;
    graph[9][4]=1;graph[9][6]=1;graph[9][7]=1;
    while(test--)
    {
        cin>>s;
        in=ma[s[0]-'A'].first;
        v.clear();
        v.push_back(in);
        int flag;
        flag=0;
        for(i=1;i<s.size();i++)
        {
            x=ma[s[i]-'A'].first;
            y=ma[s[i]-'A'].second;
            if(graph[in][x]==1)
            {
                v.push_back(x);
                in=x;
            }
            else if(graph[in][y]==1)
            {
                v.push_back(y);
                in=y;
            }
            else
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
        {
            //cout<<"e";
            in=ma[s[0]-'A'].second;
            v.clear();
            v.push_back(in);
            flag=0;
        for(i=1;i<s.size();i++)
        {
            x=ma[s[i]-'A'].first;
            y=ma[s[i]-'A'].second;
            if(graph[in][x]==1)
            {
                v.push_back(x);
                in=x;
            }
            else if(graph[in][y]==1)
            {
                v.push_back(y);
                in=y;
            }
            else
            {
                flag=1;
                break;
            }
        }
        }
        if(flag==1)
            printf("-1\n");
        else
        {
            for(i=0;i<v.size();i++)
                printf("%d",v[i]);
            printf("\n");
        }
    }
    return 0;
}
