#include<cstdio>
#include<cstring>
using namespace std;
int a[10][10],ans[100001];
char s[100001];
void set(){
	int i,j;
	for(i=0;i<10;i++)
		for(j=0;j<10;j++)
			a[i][j]=0;
	a[0][1]=a[0][4]=a[0][5]=a[1][0]=a[1][2]=a[1][6]=a[2][1]=a[2][3]=a[2][7]=a[3][2]=a[3][4]=a[3][8]=a[4][3]=a[4][0]=a[4][9]=a[5][0]=a[5][7]=a[5][8]=a[6][1]=a[6][8]=a[6][9]=a[7][2]=a[7][5]=a[7][9]=a[8][3]=a[8][5]=a[8][6]=a[9][4]=a[9][6]=a[9][7]=1;
}
int check(int strt,int l){
	int i;
	ans[0]=strt;
	for(i=1;i<l;i++){
		if(a[ans[i-1]][s[i]-65]==1)
			ans[i]=s[i]-65;
		else 	if(a[ans[i-1]][s[i]-60]==1)
					ans[i]=s[i]-60;
				else
					return 0;
	}
	return 1;
}
int main(){
	set();
	int t;
	scanf("%d",&t);
	while(t--){
		scanf("%s",s);
		int i,l=strlen(s);
		if(check(s[0]-65,l))
			for(i=0;i<l;i++)
				printf("%d",ans[i]);
		else
			if(check(s[0]-60,l))
				for(i=0;i<l;i++)
					printf("%d",ans[i]);
			else
				printf("-1");
		printf("\n");
	}
	return 0;
}
