int main()
{
 int t;
 cin>>t;
 int g[10][3];
 	g[0][0]=5;
 	g[0][1]=4;
 	g[0][2]=1;
 	g[1][0]=2;
 	g[1][1]=0;
 	g[1][2]=6;

 	g[3][2]=4;

 	g[4][1]=0;
 	g[4][2]=9;
 	g[5][0]=0;
 	g[5][1]=8;
 	g[5][2]=7;
 	g[6][0]=1;
 	g[6][1]=9;
 	g[6][2]=8;

 	g[8][2]=3;
 	g[9][0]=4;

 	g[9][2]=7;
 while(t--)
{
 	char s[100005];
 	cin>>s;
 	int len_s=strlen(s);
 	int graphs=2;
 	int temp=0;
 	int ans[len_s],i,j=0;
 	bool is_ans=true;
 	while(graphs--)
 	{
 		int ch=int(s[0])-65;
 		j=0;
 		ans[j++]=ch+temp;

 		for(i=1;i<len_s;i++)
 		{
 			if((g[ch][0]+60==int(s[i]))||(g[ch][0]+65==int(s[i])))

 		        ch=g[ch][0];	
 			}

 				ans[j++]=g[ch][2];
 		        ch=g[ch][2];	
 			}
 			else
 			{
 			 is_ans=false;
 			 break;
 		    }
 		}
 		if(is_ans==true)
		 break; 
		else
		 {
		 	temp=5;
		 	if(graphs!=0)
		 	 is_ans=true;
		 }
 	}
 	if(is_ans==false)
 	 cout<<"-1";
 	else 
	 {
	 	for(i=0;i<len_s;i++)
	 	 cout<<ans[i];
	 }
	 cout<<endl;
}
 	return 0;
 }
