#include <iostream>
#include<bits/stdc++.h>
using namespace std;
#define pb push_back
int main() {

    int t; cin>>t;

    int arr[10][3]={{1,4,5},{0,2,6},{1,3,7},{2,4,8},{0,3,9},{0,7,8},{1,8,9},{2,5,9},{3,5,6},{4,6,7}};


    while(t--){

        string s; cin>>s;
        int flag=0;
        int n = s.size();
        int brr[n];
        if(n==1) {cout<<s[0]-'A'<<"\n";}

        else{

            brr[0]=s[0]-'A';

            for(int i=1;i<n;i++){

                if((arr[brr[i-1]][0]==s[i]-'A') || (arr[brr[i-1]][1]==s[i]-'A') || (arr[brr[i-1]][2]==s[i]-'A')){

                    brr[i]=s[i]-'A';
                }
                else if((arr[brr[i-1]][0]==s[i]-'A'+5) || (arr[brr[i-1]][1]==s[i]-'A'+5) || (arr[brr[i-1]][2]==s[i]-'A'+5)){

                    brr[i]=s[i]-'A'+5;
                }
                else{
                //cout<<s[i]<<" ";
                    flag=1;
                    break;
                }
            }

            if(flag==1){

                brr[0]=s[0]-'A'+5;

                flag=0;

                for(int i=1;i<n;i++){

                if((arr[brr[i-1]][0]==s[i]-'A') || (arr[brr[i-1]][1]==s[i]-'A') || (arr[brr[i-1]][2]==s[i]-'A')){

                    brr[i]=s[i]-'A';
                }
                else if((arr[brr[i-1]][0]==s[i]-'A'+5) || (arr[brr[i-1]][1]==s[i]-'A'+5) || (arr[brr[i-1]][2]==s[i]-'A'+5)){

                    brr[i]=s[i]-'A'+5;
                }
                else{
                //cout<<s[i]<<" ";
                    flag=1;
                    break;
                }
                }

                if(flag==1) cout<<"-1\n";

                else{

                    for(int i=0;i<n;i++) cout<<brr[i];
                cout<<"\n";
                }
            }
            else{

                for(int i=0;i<n;i++) cout<<brr[i];
                cout<<"\n";
            }
        }

    }
}
