typedef long long ll;
float abs(float x)
{
	return	x<0 ? -x:x;
}
void sim(pair< pair<float,float>,char >& x)
{
	if(x.second=='U')
		x.first.second+=1;
	if(x.second=='D')
		x.first.second-=1;
	if(x.second=='L')
		x.first.first-=1;
	if(x.second=='R')
		x.first.first+=1;
}

		}
