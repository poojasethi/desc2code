#include <iostream>
#include <algorithm>
#include <map>
#include <cstdio>
using namespace std;
int main() {
  map<char,int> dx, dy;
  dx['L'] = -1;
  dx['R'] = 1;
  dy['U'] = 1;
  dy['D'] = -1;
  int t, n, xe, ye, xa, ya;
  char de, da;
  float time, tx, ty;
  cin >> t;
  while ( t-- ) {
    cin >> xe >> ye >> de;
    cin >> n;
    time = 400.0;
    while ( n-- ) {
      cin >> xa >> ya >> da;
      if ( dx[da]==dx[de] ) {
	if ( xe!=xa ) continue;
	ty = 1.0*(ye-ya)/(dy[da]-dy[de]);
	if ( ty>=0 ) time = min(ty,time);
      } else if ( dy[da]==dy[de] ) {
	if ( ye!=ya ) continue;
	tx = 1.0*(xe-xa)/(dx[da]-dx[de]);
	if ( tx>=0 ) time = min(tx,time);
      } else {
	tx = dx[de]?1.0*(xa-xe)/dx[de]:1.0*(ya-ye)/dy[de];
	ty = dx[da]?1.0*(xe-xa)/dx[da]:1.0*(ye-ya)/dy[da];
	if ( tx==ty && tx>=0.0 ) time = min(tx,time);
      }
    }
    if ( time<400.0 ) printf("%.1f\n",time);
    else cout << "SAFE" << endl;
  }
  return 0;
}
