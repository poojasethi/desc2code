#include<bits/stdc++.h>
#include<stdint.h>
#include<sstream>
using namespace std;
double intersect(int ex,int ey,char er,int x,int y,char c){
    //cout<<ex<<" "<<ey<<" "<<er<<" "<<x<<" "<<y<<" "<<c<<"\n";
    if(er == c) return -1;
    double ans = -1;
    if(er == 'R' && c == 'L' && (x > ex) && y == ey){
        ans = abs(x-ex);
        ans/=2;
    }
    else if(er == 'L' && c == 'R' && (x < ex) && y == ey){
        ans = abs(ex-x);
        ans/=2;
    }
    else if(er == 'U' && c == 'D' && y > ey && x == ex){
        ans = (y-ey);
        ans/=2;
    }
    else if(er == 'D' && c == 'U' && y < ey && x == ex){
        ans = (ey-y);
        ans/=2;
    }
    else if(er == 'R' && x > ex && ((c == 'U' && y<ey) || (c == 'D' && y > ey))){
        int xu = abs(y-ey);
        int iu = abs(x-ex);
        if(xu == iu) ans = xu;
       // cout<<0000<<"\n";
    }
    else if(er == 'L' && x < ex && ((c == 'U' && y<ey) || (c == 'D' && y > ey))){
        int xu = abs(y-ey);
        int iu = abs(x-ex);
        if(xu == iu) ans = xu;
    }
    else if(er == 'U' && y > ey && ((c == 'L' && x > ex) || (c == 'R' && x < ex))){
       int xu = abs(y-ey);
        int iu = abs(x-ex);
        if(xu == iu) ans = xu;
    }
    else if(er == 'D' && y < ey && ((c == 'L' && x > ex) || (c == 'R' && x < ex))){
        int xu = abs(y-ey);
        int iu = abs(x-ex);
        if(xu == iu) ans = xu;
    }
        return ans;
    
}

int main()
{
  int t;
  cin>>t;
  while(t--){
      int n; int ex,ey;
      char er;
      cin>>ex>>ey>>er;
      cin>>n;
     
      double ans = 999999;
      for(int i=0;i<n;i++){
          int x,y;
          char c;
          cin>>x>>y>>c;
          double t = intersect(ex,ey,er,x,y,c);
          if(t != -1){
              ans = min(ans,t);
          }
      }
      cout << std::fixed;
      if(ans == 999999){
          cout<<"SAFE\n";
      }
      else
      cout<<setprecision(1)<<ans<<"\n";
  }
  
}
