#include<iostream>
using namespace std;
int main()
{
	int sieve[10001];
	int i,t,n;
	int k=0;
	int prime[5000];
	for(i=0;i<=10000;i++)
	sieve[i]=1;
	sieve[0]=0;
	sieve[1]=0;
	for(i=2;i<=10000;i++)
	{
		if(sieve[i])
		{
			int j=2;
			while(i*j<=10000)
			{
				sieve[i*j]=0;
				j++;
			}
			prime[k++]=i;
		}
	}
	int count;
	cin>>t;
	while(t--)
	{
		count=0;
		cin>>n;
		for(i=0;i<k&&i<n;i++)
		{
			/*if((n-prime[i])%2==0)
			{
				if(sieve[(n-prime[i])/2])
				count++;
				
			}*/
			if(sieve[n-2*prime[i]] && 2*prime[i]<n)
			count++;
			
		}
		cout<<count<<endl;
	}
}
