// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

vector <int> p;
int isP[10011];
int mrk[10011];

void sieve()
{
    int i, j, sum;
    for (i = 2; i <= 100; i++) {
        if (isP[i])
            continue;
        for (j = 2; i*j <= 10000; j++)
            isP[i*j] = 1;
    }
    for (i = 2; i < 10000; i++) {
        if (!isP[i])
            p.push_back(i);
    }

    for (i = 0; i < p.size(); i++) {
        for (j = 0; j < p.size(); j++) {
            sum = p[i] + 2*p[j];
            if (sum <= 10000)
                mrk[sum]++;
        }
    }

    //cout << mrk[2] << endl;

}

int main()
{
    sieve();
    int t, n;
    scanf("%d", &t);
    while (t--) {
        scanf("%d", &n);
        printf("%d\n", mrk[n]);
    }
    return 0;
}
