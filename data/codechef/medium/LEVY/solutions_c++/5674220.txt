#include<stdio.h>
int sieve[10001]={0},prime[1230],ctr=0;
void ini()
{
    sieve[0]=1;
    sieve[1]=1;
    int i,j;
    for(i=2;i<=10000;i++)
    {
        if(sieve[i]==0)
        {
            prime[ctr++]=i;
            j=2;
            while(i*j<=10000)
            {
                sieve[i*j]=1;
                j++;
            }
        }
    }
}
main()
{
    ini();
    int test,i;
    scanf("%d",&test);
    for(i=0;i<test;i++)
    {
        int num,i,j,ctr1=0;
        scanf("%d",&num);
        for(i=0;i<ctr,prime[i]<=num/2;i++)
        {
            if(sieve[num-2*prime[i]]==0)
                {
                    ctr1++;

                }

        }
        printf("%d\n",ctr1);
    }
    return 0;
}
