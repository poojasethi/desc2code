#include <cstdlib>
#include <iostream>
#include <stdio.h>
#include <queue>
#include <stack>
#include <vector>
#include <algorithm>
#include <math.h>
#include <ctype.h>

using namespace std;

bool prost (int n)
{
    for(int i=3;i<=(int)sqrt(n);i++)
        if(n%i==0)
          return(false);
    return(true);
}

int main()
{
    vector <int> prosti;
    prosti.push_back(2);

    for(int i=3;i<10000;i+=2)
    {
        if(prost(i))
            prosti.push_back(i);
    }

   /* for(int i=0;i<100;i++)
        printf("%d\n",prosti[i]);*/

    int sol[10005],pom,n;
    for(int i=1;i<=10000;i++)
        sol[i] = 0;

    for(int i1=0;i1<prosti.size();i1++)
        for(int i2=0;i2<prosti.size();i2++)
        {
            pom = prosti[i1] + 2*prosti[i2];
            if(pom<=10000)
                sol[pom]++;
        }

    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&pom);
        printf("%d\n",sol[pom]);
    }




    return 0;
}
