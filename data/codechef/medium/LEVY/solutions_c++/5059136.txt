#include<iostream>
#include<algorithm>
#define MAX 10002

using namespace std;

bool prime[MAX] = {false};

void seive()
{
    prime[0] = prime[1] = true;
    int i, j;

    for(i=2; i*i<MAX; i++)
    {
        if(!prime[i])
        {
            for(j=i*i; j<MAX; j=j+i)
                prime[j] = true;
        }
    }
}

int main()
{
    seive();

    int t, n, i, j, res[MAX]={0};

    for(i=2; i<MAX; i++)
    {
        if(!prime[i])
        {
            for(j=2; j<MAX; j++)
            {
                if(i+2*j >= MAX)
                    break;
                if(!prime[j])
                    res[i + 2*j]++;
            }
        }
    }

    cin>>t;

    while(t--)
    {
        cin>>n;

        cout<<res[n]<<"\n";

    }

    return 0;
}
