#include <iostream>
#include <cstring>
#define MAX 10010

using namespace std;

bool a[MAX];
int p[10005];

void prime()
{
    int i, j, k = 0;

    memset(a, 0, sizeof(a));

    a[0] = 1;
    a[1] = 1;
    p[k++] = 2;
    for (i = 4; i < MAX; i += 2)
        a[i] = 1;

    for (i = 3; i * i < MAX; i++) {
        if (a[i] == 0) {
            p[k++] = i;
            for (j = i + i; j < MAX; j += i)
                a[j] = 1;
        }
    }

    for (; i < MAX; i++)
        if (a[i] == 0)
            p[k++] = i;

    //for (i = 0; i < 20; i++)
       // cout << p[i] << endl;

}

int main()
{
    int t, k;

    cin >> t;

    prime();

    while (t--) {
        int n, i, t1, count;

        cin >> n;

        if (n % 2) {
            k = 1;
            count = 0;
            while (p[k] <= n) {
                t1 = (n - p[k]) / 2;

                if (a[t1] == 0)
                    count++;

                k++;
            }

            cout << count << endl;
        }

        else {
            t1 = (n - 2) / 2;

            if (a[t1] == 0)
                cout << "1\n";

            else
                cout << "0\n";
        }
    }

    return 0;
}
