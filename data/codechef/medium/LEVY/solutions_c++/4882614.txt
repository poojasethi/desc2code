const double pi = acos(-1.0);
int flag[10050];
vi prime;
int main()
{
	int t,n,i,j,sum;
	M(flag,0);
	flag[0]=flag[1]=1;
	for (i = 2; i <= 10048; i++) {
		if (flag[i] == 0) {
			for (j = i*i; j <= 10048; j+=i) {
				flag[j] = 1;
			}
			prime.pb(i);
		}
	}
	S(t);
	while(t--) {
		sum = 0;
		S(n);
		if (n == 6) printf("1\n");
		else if (n <= 5) {
			printf("0\n");
		}
		else if (n%2 == 0) {
			 if (flag[(n-2)/2] == 0) printf("1\n");
			 else printf("0\n");
		}
		else {
			for(i = 1; prime[i] < n; i++) {
				if (flag[(n-prime[i])/2] == 0) {
				//	cout<<i<<" "<<(n-i)/2<<endl;
					sum++;
				}
			}
			printf("%d\n", sum);
		}
	}
	return 0;
}
