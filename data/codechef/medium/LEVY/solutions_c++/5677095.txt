#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t,n,i,j,ctr,ctr2;
    ctr=0;
    int prime[1230];
    int sieve[10001]={0};
        sieve[0]=1;
        sieve[1]=1;
        for(i=2;i<=10000;i++)
        {
            if(!sieve[i])
            {
                prime[ctr++]=i;
                for(j=2;i*j<=10000;j++)
                    sieve[i*j]=1;
            }
        }

        cin>>t;
    while(t--)
    {
        ctr2=0;
        cin>>n;
        for(i=0;i<ctr,prime[i]<=n/2;i++)
            if(!sieve[n-2*prime[i]])
               {
                 ctr2++;
               }
        cout<<ctr2<<endl;
    }

    return 0;
}
