#include<iostream>
#include<cstdio>
#include<vector>

#include<algorithm>
#define ll long long
#define all(c) c.begin(),c.end()
using namespace std;
vector<int> v(10001);
vector<int> st(1300);
map<int,int> m;
int main()
{
    int i,j,cal1,cal2,n,t,in=0,l;
    for(i=2;i<=10000;i++)
    {
        if(v[i]==0)
        {
            st[in]=i;
            in++;
            for(j=2*i;j<=10000;j=j+i)
            {
                v[j]=1;
            }
        }
    }
    l=in;
    //printf("%d\n",l*l);
    in=0;
    for(i=0;i<l;i++)
    {
        for(j=0;j<l;j++)
        {
            in++;
            m[st[i]+2*st[j]]++;
        }
    }
    //printf("%d\n",in);
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%d\n",m[n]);
    }
    return 0;
}
