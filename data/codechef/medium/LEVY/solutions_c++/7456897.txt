#include<iostream>
using namespace std;
int x[10005],w=0;
void find(int n)
{
	int i,j,a[10005];
	for(i=1;i<=n;++i)
	a[i]=1;
	
	for(i=2;i<=n;++i)
	{
		if(a[i])     //sieve
		x[w++]=i;		
		for(j=i*i;j<=n;j+=i)
		a[j]=0;
	}
	//for(i=0;i<n;++i)
	//cout<<x[i]<<" ";
}
int main()
{
	int n,i,j,test,p,q;
	cin>>test;
	find(10001);
	int count[10005]={0};
		for(i=0;i<w;++i)
		{
			p=x[i];
			for(j=0;j<w;++j)
			{
				q=x[j];
				if(p+2*q>10004)
				break;
				count[p+2*q]++;
			}
		}
	while(test--)
	{
		cin>>n;
		
		cout<<count[n]<<"\n";
	}
	
	return 0;
}
