#define gc getchar_unlocked

void inline inp(int &n)
{
char ch;
n=0;
while(ch<'0'||ch>'9')
ch=gc();
while(ch>='0'&&ch<='9')
{
n=(n<<3)+(n<<1)+(ch-'0');
ch=gc();
}
}
inline void op(int a)
{
register char c;
char snum[20];
int i=0;
do
{
snum[i++]=a%10+48;
a=a/10;
}while(a!=0);
i=i-1;
while(i>=0)
putchar_unlocked(snum[i--]);
putchar_unlocked('\n');
}

	
int main()
{
	int t,j,n,i,p,q;
	inp(t);
	vector<int>isprime(10001,1);
	isprime[0]=0;isprime[1]=0;
	for(i=2;i*i<10001;i++)
	if(isprime[i])
	for(j=i*i;j<10001;j+=i)
	isprime[j]=0;
	
	vector<int>cnt(10001,0);

	for(p=0;p<=10000;p++)
	if(isprime[p])
	for(q=1;(p+(2*q))<=10000;q++)
	if(isprime[q])
	cnt[p+(2*q)]++;
	
	while(t--)
	{
	inp(n);
	
	
	
	
	op(cnt[n]);
	}
	
	return 0;
	}
