#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 10005
#define pb(a) push_back(a)
int arr[MAX];
void sieve()
{

	int i,j,k,m,n,t,l;

	for(i=0;i<MAX;i++)
		arr[i]=i+2;

	for(i=0;i<MAX;i++)
	{
		if(arr[i]!=0)
		{
			m=1;
			while(m*arr[i]+i<=MAX-2)
			{
				arr[m*arr[i]+i]=0;
				m++;
			}
		}
	}

}
int main()
{
	int p,q,t,i,j,k,count=0,m,n,brr[1230],crr[MAX];
	sieve();

	for(i=0;i<MAX;i++)
	{
		if(arr[i]!=0)
			brr[count++]=arr[i];
	}


	for(j=0;j<1230;j++)
	{
		for(k=0;k<1230;k++)
		{
			p=brr[j];
			q=brr[k];
			if(p+2*q<MAX)
				crr[p+2*q]++;
			else
				break;
		}
	}



	si(t);
	while(t--)
	{
		si(m);
		printf("%d\n",crr[m]);
	}
}
