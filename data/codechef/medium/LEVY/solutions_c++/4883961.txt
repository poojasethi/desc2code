#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

int p[10000];
int a[10000];

int main()
{
    int i,q,j,t,n,coun=0,in=0,k=0,c;
    p[0] = p[1] = 1;
    for (i = 2; i <= 10000; i++) {
        if (p[i] == 0) {
           a[k] = i;
           k++;
           for (j = i*i; j <= 10000; j = j + i)
               p[j] = 1;
        }
    }
    a[0] = 2;

    cin >> t;
    while (t--) {
          cin >> n;
          if (n < 6)//|| (n > 6 && n%2 == 0))
             cout << "0\n";
          else if (n == 6)
               cout << "1\n";
          else {
               coun = 0;
               j = 0;
               for (j = 0; a[j] < n && j < k; j++) {
                         q = n-a[j];
                         if (q % 2 == 0) {
                          q = q/2;
                          if (p[q] == 0) {
                             coun++;
                             }
                         }     
               }
               cout << coun << endl;
          }
    }
    
    return 0;
}
