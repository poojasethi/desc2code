#include<bits/stdc++.h>
#define mp(x,y) make_pair(x,y)
#define F first
#define S second
using namespace std;
int n,p[5000],np,m[40000];
void prime()
{
    int i,j;
    p[0]=2;
    p[1]=3;
    np=2;
    for(i=5;i<10000;i++)
    {
        for(j=0;j<np;j++)if(i%p[j]==0)j=10000;
        if(j==np)p[np]=i,np++;
    }
    for(i=0;i<np;i++)for(j=0;j<np;j++)m[p[i]+2*p[j]]++;
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int t;
    prime();
    cin>>t;
    while(t--)
    {
        cin>>n;
        cout<<m[n]<<endl;
    }
}
