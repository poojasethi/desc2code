#include<stdio.h>
#include<math.h>
int prime[10000],chk[10005],use[10005];
int main()
{
    int t,n,i,j,idx=0,k,ck,ll;
    for(i=2;i<=10000;i++)
    {
        if(!chk[i])
        {
            prime[idx++]=i;
            for(j=i+i;j<=10000;j+=i)
                chk[j]=1;
        }
    }
    for(j=0;j<idx;j++)
    {
        if(prime[j]>10000)
        {
            break;
        }    
        for(k=0;k<idx;k++)
        {
            ll=prime[j]+2*prime[k];
            if(prime[j]+2*prime[k]>10000)
            {
                break;
            }
            use[ll]++;
        }
    }
    scanf("%d",&t);
    for(i=0;i<t;i++)
    {
        scanf("%d",&n);
        printf("%d\n",use[n]);
    }
    scanf(" ");
}
