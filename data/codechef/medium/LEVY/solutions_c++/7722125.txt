#include<cstdio>
#include<set>
using namespace std;
int check[10005];
int prime[10005];
int m=1;
int res[10005]={0};
void seive()
{
	int i,j;
	for(int i=2;i*i<=10000;i++)
	{
		if(check[i]==0)
		{
			for(j=i*i;j<=10000;j+=i)
			check[j]=1;
		}
	}
prime[0]=2;
for(int i=3;i<=10000;i+=2){

if(check[i]==0)
{
prime[m++]=i;
}
}}

int main()
{	seive();
	int t;
	scanf("%d",&t);
		for(int i=0;i<m;i++)
		{
			for(int j=0;j<m;j++)
			{	int x=prime[i]+2*prime[j];
				if(x>10005)
				break;
				else res[x]++;	
				//if(x==11) printf("%d %d",i,j);
			}
		}
	while(t--)
	{
		int n;
		scanf("%d",&n);
		
	
	printf("%d\n",res[n]);	
	}
return 0;	
}
