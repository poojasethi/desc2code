#include<iostream>
using namespace std;
int k,ar[10011],prime[10000];
void sieve()
{
    ar[0]=1;
    ar[1]=1;
    for(int i=2;i<=10011;i++)
    {
        for(int j=2;j*i<=10011;j++)
        {
            ar[j*i]=1;
        }
    }
    for(int i=2;i<=10011;i++)
    {
        if(ar[i]==0)
            {prime[k++]=i;//cout<<i<<endl;
            }
    }
}

int func(int n)
{
    int c=0;
    for(int i=0;prime[i]<n;i++)
    {
        if((n-prime[i])%2==0)
        {
            if(ar[(n-prime[i])/2]==0)
                c++;
        }

    }
    return c;
}
int main()
{
    ios_base::sync_with_stdio(false);
    sieve();
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;

        cout<<func(n)<<endl;
    }
    return 0;
}
