#include<stdio.h>
#include<iostream>
#include<math.h>
int main()
{ int t,flag1=2,flag2=0,s[10000]={2,3},p[10001]={0};
  for(int i=4;i<10000;i++)
     { int flag=0;
	   for(int j=2;j<=sqrt(i);j++)
          { if(i%j==0)
              { flag=1;
                break;
              }
          } 
       if(flag==0)
	     s[flag1++]=i;
	 }	 	 	 
  for(int i=0;i<flag1;i++)
     { int j=0;
       while(s[i]+2*s[j]<=10000 && j<flag1)
            { p[s[i]+2*s[j]]++;
			  j++;
		    }
     }   
  scanf("%d",&t);
  while(t--)
       { int n;
         scanf("%d",&n);
         printf("%d\n",p[n]);   
       }
}
