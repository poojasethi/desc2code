#include <iostream>
#include <vector>
using namespace std;

int main() {
	int n,t,a[10009],b[10009],i,j,sz,temp;
	vector<int> p;
	for(i=0;i<10009;i++)
	{
		a[i]=0;
		b[i]=0;
	}
	a[0]=1;
	a[1]=1;
	for(i=2;i<10009;i++)
	{
		if(a[i]==0)
		{
			for(j=2*i;j<10009;j+=i)
			{
				a[j]=1;
			}
		}
	}
	for(i=0;i<10009;i++)
	{
		if(a[i]==0)
			p.push_back(i);
	}
	sz=p.size();
	for(i=0;i<sz;i++)
	{
		for(j=0;j<sz;j++)
		{
			temp=p[i]+(2*p[j]);
			if(temp<10009)
				b[temp]++;
		}
	}
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		printf("%d\n",b[n]);
	}
	return 0;
}
