#include <cstdio>
using namespace std;

int main()
{
	bool primes[10001] = {0, 0, 1};
	int prime[1229] = {2}, size = 1, primecount[5000] = {0, 0, 0};
	for(int i = 3; i < 10000; i += 2)
	{
		bool test = true;
		for(int j = 0; j < size; j++)
		{
			if(!(i%prime[j]))
			{
				test = false;
				break;
			}
		}
		if(test)
		{
			prime[size] = i;
			size++;
			primes[i] = 1;
		}
	}
	for(int i = 3; i < 5000; i++)
	{
		int index = 1;
		while(prime[index] < 2*i+1 && index < 1229)
		{
			if(primes[(2*i+1-prime[index])/2])
			{
				primecount[i]++;
			}
			index++;
		}
	}
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		if(n&1)
		{
			printf("%d\n", primecount[(n-1)/2]);
		}
		else
		{
			if(primes[n/2-1])
			{
				printf("1\n");
			}
			else
			{
				printf("0\n");
			}
		}
	}
    return 0;
}
