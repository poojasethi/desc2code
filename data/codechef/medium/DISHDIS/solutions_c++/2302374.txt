//HMM

#include<iostream>
#define M 1000000007
using namespace std;
int main()
{
    long long int t,n,m,a[101],b[101],i,j,k,s;
    cin>>t;
    while(t--)
    {
              cin>>n>>m;
              s=0;
              for(i=1;i<=m;i++){cin>>a[i]>>b[i];s+=a[i];b[i]=b[i]-a[i];}
              int dp[101][101]={0};
              for(i=0;i<=n;i++)dp[0][i]=1;
              if(n<s){cout<<0<<endl;continue;}
              else if(n==s){cout<<1<<endl;continue;}
              else
              {
                  n=n-s;
                  for(i=1;i<=n;i++)
                  {
                                   for(j=1;j<=m;j++)
                                   {
                                               for(k=0;k<=b[j];k++)
                                               {
                                                         if(k>i) break;
                                                         dp[i][j]=(dp[i][j]+dp[i-k][j-1])%M;
                                               }
                                   }
                  }                    
              }
              cout<<dp[n][m]<<endl;
    }
}
