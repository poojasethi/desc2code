#include<cstdio>
#include<iostream>
//#include<conio.h>
using namespace std;
#define MOD 1000000007
int t,n,m,x[105],y[105],i,j,k;
int main()
{
 scanf("%d",&t);
 while(t--)
 {
  scanf("%d%d",&n,&m);
  for(i=1;i<=m;i++)
  scanf("%d%d",&x[i],&y[i]);
  int dp[105][105]={0};
  for(i=x[1];i<=y[1];i++)
  dp[1][i]=1;
  for(i=2;i<=m;i++)
  {
   for(j=0;j<=n;j++)
   {
    for(k=x[i];k<=y[i];k++)
    {
     if((j-k)>=0)
     {
      dp[i][j]+=dp[i-1][j-k];
      if(dp[i][j]>=MOD)
      dp[i][j]-=MOD;
     }
    }
   }
  }
  printf("%d\n",dp[m][n]);
 }
 //getch();
}
