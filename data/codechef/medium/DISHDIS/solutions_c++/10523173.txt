#include<cstdio>
using namespace std;
# define MOD 1000000007LL
int main()
{
   long long int t,n,m,dp[105][105],i,j,k,c;
    scanf("%lld",&t);
    while(t--)
    {

         scanf("%lld %lld",&n,&m);
         long long int x[m+1],y[m+1];
         for(i=1;i<=m;i++)
         {
             scanf("%lld %lld",&x[i],&y[i]);

         }


         for(i=0;i<=n;i++)
         {

             dp[1][i]=(i<=y[1] && i>=x[1])?1:0;


         }
         for(i=2;i<=m;i++)
         {
             for(j=0;j<=n;j++)
             {
                 dp[i][j]=0;
                  for(k=x[i];k<=y[i] && (j-k>=0);k++)
                  {
                      dp[i][j]+=(dp[i-1][j-k]);

                  }
                  dp[i][j] %= MOD;

             }

         }
        printf("%lld\n",dp[m][n]%MOD);


    }
}
