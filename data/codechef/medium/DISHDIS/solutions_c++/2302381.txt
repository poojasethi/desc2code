#include<iostream>
#define M 1000000007
using namespace std;
int main()
{
long long int t,n,m,a[101],b[101],i,j,k,s;
scanf("%lld",&t);
while(t--)
{
scanf("%lld %lld",&n,&m);
s=0;
for(i=1;i<=m;i++){cin>>a[i]>>b[i];s+=a[i];b[i]=b[i]-a[i];}
int dp[101][101]={0};
for(i=0;i<=n;i++)dp[0][i]=1;
if(n<s){printf("0\n");continue;}
else if(n==s){printf("1\n");continue;}
else
{
n=n-s;
for(i=1;i<=n;i++)
{
for(j=1;j<=m;j++)
{
for(k=0;k<=b[j];k++)
{
if(k>i) break;
dp[i][j]=(dp[i][j]+dp[i-k][j-1])%M;
}
}
}
}
printf("%d\n",dp[n][m]);
}
}
