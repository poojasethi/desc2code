#include <cstdio>
#define p 1000000007

int b[103];
int e[103];

int n, m;

int dp[103][103];

int main(){
	int tc;
	scanf("%d", &tc);
	while (tc--){
		scanf("%d%d", &n, &m);
		for (int i=0; i<m; i++){
			scanf("%d%d", b+i, e+i);
		}
		for (int i=0; i<=n; i++){
			dp[m-1][i] = b[m-1]<=i && i<=e[m-1];
		}
		for (int i=m-2; i>=0; i--){
			for (int j=0; j<=n; j++){
				int ans = 0;
				for (int k=b[i]; k<=e[i]; k++){
					if (j-k>=0){
						ans = (ans+dp[i+1][j-k])%p;
					}
				}
				dp[i][j] = ans;
			}
		}
		printf("%d\n", dp[0][n]);
	}
	return 0;
}
