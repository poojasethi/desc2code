#include <cstdio>

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
        const int size = 101;
        const int mod = 1000000007;
        int numCooksDishes[size][size] = {};
        int n, m;
        scanf("%d %d", &n, &m);
        int x, y;
        scanf("%d %d", &x, &y);
        for(int i=x; i<=y; i++) numCooksDishes[1][i]++;
        for(int cook=2; cook<=m; cook++) {
            scanf("%d %d", &x, &y);
            for(int j=0; j<=n; j++) if(numCooksDishes[cook-1][j])
                    for(int i=x; i<=y && i+j<=n; i++) {
                        numCooksDishes[cook][i+j] += numCooksDishes[cook-1][j];
                        numCooksDishes[cook][i+j] %= mod;
                    }
        }

        printf("%d\n", numCooksDishes[m][n]);
    }
}
