#include<stdio.h>
    #define MOD 1000000007
     
    int main()
    {
	    int t,i,m,n,j,k;
	    scanf("%d",&t);
	    while(t--)
	    {
		    scanf("%d%d",&n,&m);
		    int x[105],y[105],dp[105][105]={0};
		    for(i=0;i<m;i++)
		    scanf("%d%d",&x[i],&y[i]);
		    for(i=x[0];i<=y[0];i++)
		    dp[0][i]=1;
		    for(i=1;i<m;i++)
		    	for(j=0;j<=n;j++)
		    		for(k=x[i];k<=y[i];k++)
		    			if((j-k)>=0)
		    			{
		    				dp[i][j]+=dp[i-1][j-k];
		    				if(dp[i][j]>=MOD)
		    				dp[i][j]%=MOD;
		    			}
		     
		    printf("%d\n",dp[m-1][n]);
	    }
	    return 0;
    }
