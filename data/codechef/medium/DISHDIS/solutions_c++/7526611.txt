#include <bits/stdc++.h>
#define MOD 1000000007
#define MAXN 110

using namespace std;

int mod(int x)
{
	while(x > MOD)
		x -= MOD;
	while (x < 0)
		x += MOD;
	return x;	
}

int main() 
{
	int T;
	cin >> T;
	while (T--)
	{
		int DP[MAXN][MAXN] = {0}, A[MAXN][2], N, M;
		cin >> N >> M;
		for (int i = 1; i <= M; i++)
			cin >> A[i][0] >> A[i][1];
		// Cooks and dishes
		for(int i = A[1][0]; i <= A[1][1]; i++)
			DP[1][i] = 1;
		for (int i = 2; i <= M; i++)
			for (int j = 0; j <= N; j++)
				for (int k = A[i][0]; k <= A[i][1]; k++)
					if(j - k >= 0)
						DP[i][j] = mod(DP[i][j] + DP[i - 1][j - k]);
		cout << DP[M][N] << '\n';	
	}
	return 0;
}
