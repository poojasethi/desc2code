#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;

int main()
{
    int t,i,j,ans,m,n,dp[200][200],x[300][2],k;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        for(i=0;i<m;i++)
            scanf("%d%d",&x[i][0],&x[i][1]);
        for(i=0;i<200;i++)
            for(j=0;j<200;j++)
                dp[i][j]=0;

        for(i=x[0][0];i<=x[0][1];i++)
            dp[0][i]=1;

        for(i=1;i<m;i++)
        {
            ans=0;
            for(j=0;j<=n;j++)
            {

                if(j>=x[i][0])
                {
                    ans+=dp[i-1][j-x[i][0]];
                    ans%=mod;
                }
                if(j>=x[i][1]+1)
                {
                    ans=(ans-dp[i-1][j-x[i][1]-1]+mod)%mod;

                }
                dp[i][j]=ans;
            }
        }
        /*for(i=0;i<m;i++)
        {
            for(j=0;j<=n;j++)
                cout<<dp[i][j]<<" ";
            cout<<endl;
        }*/
        //cout<<dp[m-1][n]%mod<<endl;
        printf("%d\n",dp[m-1][n]%mod);
    }
    return 0;
}
