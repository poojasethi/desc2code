#include <iostream>
#include <stdio.h>
#include <math.h>
#include <vector>
#include <algorithm>
#include <queue>


using namespace std;
int t1,t2,n,m,t3,ans,tests,last;
vector<pair<int,pair<int,int> > > v;


int main()
{
scanf("%d",&tests);
    while(tests--)
    {
        scanf("%d %d",&n,&m);
        v.clear();
        for(int i=0;i<n;i++)
        {
            scanf("%d %d %d",&t1,&t2,&t3);
            v.push_back(make_pair(t3,make_pair(t2,t1)));
        }
        sort(v.begin(),v.end());
        ans=0;
        for(int i=0;i<v.size();i++)
        {
            if(i==0 || v[i].first!=v[i-1].first)
            {
                last = v[i].second.first;
                ans++;
            } else
            if(v[i].second.second>=last)
            {
                last=v[i].second.first;
                ans++;
            }
        }
        printf("%d\n",ans);

    }
    return 0;
}
