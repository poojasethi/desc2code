#include <bits/stdc++.h>
using namespace std;
struct pree
{
	int x,y,z;
};
bool compare(pree s,pree r)
{
	if(s.z==r.z)
	{
		return s.y<r.y;
	}
	return s.z<r.z;
}
vector<pair<int,int> > v[100001];
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		if(n==0)
		{
			cout<<"0\n";
			continue;
		}
		pree a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i].x>>a[i].y>>a[i].z;
		}
		sort(a,a+n,compare);
		int j=0;
		v[0].clear();
		v[0].push_back(make_pair(a[0].x,a[0].y));
		for(int i=0;i<n-1;i++)
		{
			if(a[i].z==a[i+1].z)
			{
				v[j].push_back(make_pair(a[i+1].x,a[i+1].y));
			}
			else
			{
				j++;
				v[j].clear();
				v[j].push_back(make_pair(a[i+1].x,a[i+1].y));
			}
		}
		int answer=0;
		for(int i=0;i<=j;i++)
		{
			if(v[i].size()==1)
			{
				answer++;
			}
			else
			{
				answer++;
				int l=v[i][0].second;
				for(int k=1;k<v[i].size();k++)
				{
					if(v[i][k].first>=l)
					{
						answer++;
						l=v[i][k].second;
					}
				}
			}
		}
		cout<<answer<<"\n";
	}
}
