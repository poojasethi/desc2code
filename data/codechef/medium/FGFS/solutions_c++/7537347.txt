#include <bits/stdc++.h>
#define pb push_back
#define vi vector<long long>
#define ss second
#define ff first
#define pii pair<long long, long long>

using namespace std;

pair<int, pair<int,int> > p[100005];

int main()
{
    ios::sync_with_stdio(false);
    int T; cin>>T;
    while(T--)
    {
        int n,k; cin>>n>>k;
        for(int i=0; i<n; i++)
            cin>>p[i].ss.ff>>p[i].ss.ss>>p[i].ff;
        sort(p,p+n);
        long long ans=0;
        for(int i=0; i<n; i++)
        {
            ans+=1;
            while(i<n-1&&p[i].ff==p[i+1].ff)
            {
                if(p[i+1].ss.ff<p[i].ss.ss)
                {
                    if(p[i+1].ss.ss>p[i].ss.ss)
                        p[i+1]=p[i];
                }
                else
                    ans++;
                i++;
            }
        }
        cout<<ans<<endl;
    }
}
