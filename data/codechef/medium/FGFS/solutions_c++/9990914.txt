#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_v) (int)(_v).size()
#define all(_v) (_v).begin(),(_v).end()
#define pii pair<int,int>
typedef long long ll;
const int INF = 1000000000;
const ll MOD = 1e9 + 7;

const int MAXN = 100010;
vector < pii > vec[MAXN];
map < int,int > mm;

int activity_selection(int x) {
	int fn = -111;
	int ret = 0;

	for (int i = 0; i < sz(vec[x]); i ++) {
		if(vec[x][i].second >= fn) {
			ret ++;
			fn = vec[x][i].first;
		}
	}

	return ret;
}

int main() {

    int t; scanf("%d",&t);
	while (t --) {
		int n,k; scanf("%d%d",&n,&k);
		int num = 0;
		for (int i = 0; i < n; i ++) {
			int st,fn,p; scanf("%d%d%d",&st,&fn,&p);
			if(mm.find(p) == mm.end()) {
				mm[p] = num ++;
			}
			vec[mm[p]].pb(mp(fn,st));
		}

		int ans = 0;
		for (int i = 0; i < num; i ++) {
			sort(all(vec[i]));
			ans += activity_selection(i);
			vec[i].clear();
		}

		printf("%d\n",ans);
		mm.clear();
	}
	return 0;
}
