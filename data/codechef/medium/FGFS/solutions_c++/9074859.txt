#include<map>
#include <utility>
#include <vector>
#include <algorithm>
#include <stdio.h>
# define its (it->second)
int T,N,K,s,f,p,counter,singlecount;
using namespace std;
map< int , vector<pair<int,int> > > cus;
typedef map< int , vector<pair<int,int> > >::iterator mit;
//typedef vector<pair<int,int> >::iterator vit;
int main()
{
	int i,j;
	scanf("%d",&T);
	while(T--)
	{
		counter=0;
		cus.clear();
		scanf("%d%d",&N,&K);
		for(i=0;i<N;i++)
		{
			scanf("%d%d%d",&s,&f,&p);
			cus[p].push_back(make_pair(f,s));
		}
		for(mit it=cus.begin();it!=cus.end();it++)
		{
			sort(its.begin(),its.end());
			singlecount=1;
			for(i=0;i<its.size();)
			{
				for(j=i+1;j<its.size()&&its[j].second<its[i].first;j++);
				if(j<its.size())
				{
					//printf("i = %d: %d TO %d : j = %d :%d TO %d\n",i,its[i].second,its[i].first,j,its[j].second,its[j].first);  
					i=j;
					//printf("prev scount is %d\n",singlecount);
					singlecount++;
					//printf("new scount is %d\n",singlecount);
				}
				else
				{
					i=its.size();
					break;
				}
			}
			//printf("singlecout is %d\n",singlecount);
			counter+=singlecount;
		}
		printf("%d\n",counter);
	}
	return 0;
}
