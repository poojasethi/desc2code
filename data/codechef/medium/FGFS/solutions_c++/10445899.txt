void mergeSort(ll *a1,ll *a2,ll *a3,ll s,ll f)
{
	ll mid=s+(f-s)/2;
	if(f-s>0)
	{
		mergeSort(a1,a2,a3,s,mid);
		mergeSort(a1,a2,a3,mid+1,f);	
		merge(a1,a2,a3,s,mid+1,mid,f);
	}
}
