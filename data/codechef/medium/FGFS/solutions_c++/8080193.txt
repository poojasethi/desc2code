#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef long long int ll;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
struct customer
{
    ll s,f,p;
};
bool operator<(customer const &a,customer const &b)
{
    if(a.p!=b.p)
    {
        return (a.p<b.p);
    }
    else
    {
        return (a.f<b.f);
    }
}
int main()
{
    ll t,n,k,i;
    t=read_int();
    while(t--)
    {
        n=read_int();
        k=read_int();
        customer a[n];
        for(i=0;i<n;i++)
        {
            a[i].s=read_int();
            a[i].f=read_int();
            a[i].p=read_int();
        }
        if(n==0||n==1)
            printf("%lld\n",n);
        else
        {
            sort(a,a+n);
            ll x=0,c=1;
            for(i=1;i<n;i++)
            {
                if(a[i].p!=a[i-1].p || (a[i].p==a[i-1].p && a[i].s>=a[x].f) )
                {
                    x=i;
                    c++;
                }
            }
            printf("%lld\n",c);
        }
    }
    return 0;
}
