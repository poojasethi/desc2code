#include<iostream>
#include<cstdio>
#include<vector>
#include<map>
#include<algorithm>
using namespace std;
vector< pair<int,int> >v[1000005];
map<int,int>m;
int fun(vector< pair<int,int> >arr)
{
    int ans=1,i,j;
    i=0;
    if(arr.size()==0)
	return 0;
	sort(arr.begin(), arr.end());
    for(j=1;j<arr.size();j++)
        if(arr[j].second>=arr[i].first)
    {

        ans++;
        i=j;
    }
    return ans;

}

int main()
{
    int t,n,k,i,s,f,p,ans;
    scanf("%d",&t);
    while(t--)
    {
        m.clear();
        int pos=0;
        ans=0;
        scanf("%d %d",&n,&k);

        for(int i=0;i<=n;i++)
	    v[i].clear();
        for(i=0;i<n;i++)
        {

            scanf("%d %d %d",&s,&f,&p);
            p--;
            if(m.count(p)==0)
            {
                m[p]=pos++;
            }
            v[m[p]].push_back(make_pair(f,s));
        }
        for(i=0;i<pos;i++)
            ans+=fun(v[i]);


        printf("%d\n",ans);
    }



}
