#include <cstdio>
#include <algorithm>
struct Cust
{
	int s, f, p;
};
 
bool operator <(Cust c1, Cust c2)
{
	if(c1.p<c2.p || (c1.p==c2.p && c1.f<c2.f))
		return true;
	return false;
}
 
int main()
{
	int T;
	scanf("%d", &T);
 
	while(T--)
	{
		int N, K;
		scanf("%d%d", &N, &K);
 
		Cust c[100000];
		for(int i = 0; i < N; ++i)
			scanf("%d%d%d", &c[i].s, &c[i].f, &c[i].p);
 
		if(N <= 1)
			printf("%d\n", N);
		else
		{
			std::sort(c, c+N);
 
			int ans = 1, last = c[0].f;
			for(int i = 1; i < N; ++i)
			{
				if(c[i].p==c[i-1].p)
				{
					if(c[i].s>=last)
					{
						last = c[i].f;
						++ans;
					}
				}
				else
				{
					last = c[i].f;
					++ans;
				}
			}
 
			printf("%d\n", ans);
		}
	}
 
	return 0;
}
