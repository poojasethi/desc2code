#include <iostream>
#include<algorithm>
using namespace std;

const int inf = 1e9;
const int maxh = 1e5;

pair< long long, long long > ct[maxh + 5];

int main()
{
    long long n, m;
    int h;
    cin >> n >> m >> h;
    for (int i = 0; i < h; ++i)
        cin >> ct[i].second >> ct[i].first;

    sort(ct, ct + h);
    long long cnt = 0;
    long long ans = 0;
    for (int i = 0; i < h; ++i)
    {
        if (cnt + ct[i].second < n * m)
            {
            ans += ct[i].second * ct[i].first;
            cnt += ct[i].second;
             }
        else
        {
            ans += (n * m - cnt) * ct[i].first;
            cout << ans << "\n";
            return 0;
        }
    }
    cout << "Impossible\n";
    return 0;
}
