#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

vector<pair<long long, long long> > ct;

int main() {
    long long t, c, r, cur_t=0, cur_tot_t, cur_c=0, tot;
    long long n, m, h;
    cin >> n >> m >> h;
    tot = n * m;
    while(h--) {
        cin >> t >> c;
        ct.push_back(make_pair(c, t));
    }
    sort(ct.begin(), ct.end());
    vector<pair<long long, long long> >::iterator it;
    for ( it = ct.begin(); it != ct.end(); it++) {
        c = it->first;
        t = it->second;
        cur_tot_t = min(cur_t + t, tot);
        cur_c += (cur_tot_t - cur_t) * c;
        cur_t = cur_tot_t;
        if (cur_t == tot)
            break;
    }

    if (cur_t < tot)
        cout << "Impossible" << endl;
    else
        cout << cur_c << endl;
    return 0;
}
