#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

pair<long long int,long long int>a[100005];
int main()
{
    long long int n,m,sum,tot,ans;
    int h,i;
    scanf("%lld%lld%d",&n,&m,&h);
    for(i=0;i<h;i++)
    scanf("%lld%lld",&a[i].second,&a[i].first);
    sum=0;
    sort(a,a+h);
    tot=n*m;
    ans=0;
    for(i=0;i<h;i++)
    {
        if(sum+a[i].second<tot)
        {
            ans=ans+(a[i].second*a[i].first);
            sum=sum+a[i].second;
        }
        else
        {
            ans=ans+((tot-sum)*a[i].first);
            sum=tot;
            break;
        }
    }
    if(sum==tot)
    printf("%lld\n",ans);
    else
        printf("Impossible\n");
    return 0;
}
