using namespace std;
int main()
{
	long long n,m,h,i=0,h1;
	priority_queue<pii,vector<pii>, greater<pii> > q; 
	scanf("%lld %lld %lld",&n,&m,&h);
	h1=h;
	long long tot=0,sum=0,cost=0,t,c;
	while(h1--)
	{
	 scanf("%lld %lld",&t,&c);	
	 q.push(pii(c,t));   
	 tot+=t;
    }
    if(tot<n*m)
     {
     	printf("Impossible");
     	return 0;
     }
     tot=n*m;
    while(!q.empty())
    {
    //	cout<<q.top().first;
	   if(tot>=q.top().second)
	    {
	    	cost+=q.top().second*q.top().first;
	    	tot-=q.top().second;
	    	q.pop();
	    }  
	    else
	     {
	     //	cout<<q.top().first;
	     	cost+=q.top().first*tot;
	     //	cout<<cost;
	     	break;
	     }
    }
    printf("%lld",cost);
	return 0; 
}
