#include <bits/stdc++.h>
#define ll long long int
using namespace std;

bool compare(pair<ll,ll> p,pair<ll,ll> q)
{
    if(p.second>q.second)
        return false;
    return true;
}
int main()
{
    vector<pair<ll,ll> >x;
    ll h,n,m,i,a,b,s;
    cin>>n>>m>>h;
    s=0;
    for(i=0;i<h;i++)
    {
        cin>>a>>b;
        s+=a;
        x.push_back(make_pair(a,b));
    }
    n=n*m;
    if(s<n)
        cout<<"Impossible\n";
    else
    {
        sort(x.begin(),x.end(),compare);
        a=0;
        for(i=0;i<h;i++)
        {
            if(n<x[i].first)
                {a+=(n*x[i].second);break;}
            else
                a+=(x[i].first*x[i].second);
            n=n-x[i].first;
        }
        cout<<a<<endl;
    }
    return 0;
}
