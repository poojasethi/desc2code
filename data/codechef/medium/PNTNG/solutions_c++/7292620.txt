#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
using namespace std;
int main()
{
    long long i,n,m,h,flag,req,c,cost;
    scanf("%lld%lld%lld",&n,&m,&h);
    vector<pair<long long,long long> > v(h);
    for(i=0;i<h;i++)
    {
        scanf("%lld%lld",&(v[i].second),&(v[i].first));
    }
    sort(all(v));
    req=n*m;
    c=cost=flag=0;
    for(i=0;i<h;i++)
    {
        if(v[i].second!=0)
        {
            if(v[i].second<req)
            {
                req=req-v[i].second;
                cost+=v[i].second*v[i].first;
            }
            else
            {
                cost+=req*v[i].first;
                flag=1;
                break;
            }
        }
    }
    if(flag)
        printf("%lld\n",cost);
    else
        printf("Impossible\n");
    return 0;
}
