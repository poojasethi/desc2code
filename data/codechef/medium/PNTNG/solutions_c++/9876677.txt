#include <iostream>
#include <utility>
#include <algorithm>
using namespace std;

int main() {
	
	pair<long long int, long long int> p[100009];
	long long int m,n,cur=0,temp,cost=0,temp2,ans;
	long long int h,i;
	scanf("%lld %lld %lld",&n,&m,&h);
	cur=m*n;
	ans=0;
	for(i=0;i<h;i++)
	{
		scanf("%lld %lld",&p[i].second,&p[i].first);
	}
	sort(p,p+h);
	for(i=0;i<h;i++)
	{
		if(cur==0)
			break;
		else 
		{
	 		if(p[i].second>cur)
	 		{
	 			ans+=(p[i].first*cur);
	 			cur=0;
	 			break;
	 		}
	 		else
	 		{
	 			ans+=(p[i].first*p[i].second);
	 			cur=cur-p[i].second;
	 		}
		}
	}
	if(cur>0)
		printf("Impossible\n");
	else
		printf("%lld\n",ans);
	return 0;
}
