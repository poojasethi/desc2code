    inline void readL(long long &a)
    {
    register int c;
    a = 0;
    do c = getchar_unlocked(); while(c < '0');
    do{
    a = (a << 1) + (a << 3) + c - '0';
    c = getchar_unlocked();
    }while(c >= '0');
    }
     
    inline void printL(long long a)
    {
    int s[22],t = -1;
    do
    {
    s[++t] = a % 10 + '0';
    a /= 10;
    }while(a > 0);
     
    while(t >= 0)putchar_unlocked(s[t--]);
    putchar_unlocked('\n');
    }
