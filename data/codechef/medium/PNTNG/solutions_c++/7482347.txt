#include<iostream>
#include<map>
using namespace std;
typedef long long ll;
int main()
{
    ll n,m,h,t,c,i,clls=0;
    cin>>n>>m>>h;
    clls=n*m;
    //ll a[h][2];
    ll sum=0,cost=0;
    multimap<ll,ll> coll;
    multimap<ll,ll>::iterator it;
    for(i=0;i<h;i++)
    {
        cin>>t>>c;
        //a[i][0]=t;
        //a[i][1]=c;
        coll.insert(make_pair(c,t));
        sum+=t;
    }
    /*for(it=coll.begin();it!=coll.end();++it)
    {
        cout<<it->first<<" "<<it->second<<'\n';
    }*/
    if(sum<clls)
        cout<<"Impossible\n";
    else
    {
        ll x=0,y=0,p=0;
        for(it=coll.begin();it!=coll.end();++it)
        {
            if(x==clls)
                break;
            p=clls-x;
            if(it->second<=p)
            {
                x+=it->second;
                y=it->second;
            }
            else
            {
                x+=p;
                y=p;
            }
            cost+=(it->first)*y;
        }
        cout<<cost<<'\n';
    }
    return 0;
}
