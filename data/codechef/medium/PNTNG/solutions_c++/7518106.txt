#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
 
int main(){
  long long c,cost = 0,temp,i,n,m,h,tk,ck,sum = 0,a[100001] = {0};
  scanf("%lld",&n);
  scanf("%lld",&m);
  scanf("%lld",&h);
 
 
  for(i = 0 ; i < h ; i++){
      scanf("%lld",&tk);
      scanf("%lld",&ck);
 
      a[ck] += tk;
      sum += tk;
  }
  temp = n*m;
  if(sum < temp){
    printf("Impossible\n");
  }
  else{
      sum = 0;
      for(i = 0 ; i < 100001 ; i++){
        c =  min(temp,a[i]);
        temp = temp - c;
        sum = sum - c;
 
        cost = cost + i*c;
 
        if(temp <= 0){
            printf("%lld",cost);
             break;
        }
 
    }
  }
  return 0;
}
