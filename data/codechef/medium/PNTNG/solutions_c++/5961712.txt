3 4 2
6 1
6 2
*/

#include<iostream>
#include<stdio.h>
#define ll long long
#include<algorithm>
using namespace std;

struct node
{
	ll t,c;
};

bool bycost(node a,node b)
{
	return(a.c<b.c);
}

int main()
{
	ll n,m,h;
	scanf("%lld%lld%lld",&n,&m,&h);
	ll i,sum=0;
	ll total=n*m;
	node a[h];
	for(i=0;i<h;i++)
	{
		scanf("%lld%lld",&a[i].t,&a[i].c);
		if(a[i].t>total)
			a[i].t=total;
		sum+=a[i].t;
	}
	if(sum<total)
		printf("Impossible");
	else
	{
		ll count=0;
		i=0;
		sort(a,a+h,bycost);
		sum=0;
		while(count<total)
		{
			if(a[i].t<=total-count)
			{
				sum+=a[i].t*a[i].c;
				count+=a[i].t;
			}
			else
			{
				sum+=(total-count)*a[i].c;
				count=total;
			}
			i++;
		}
		printf("%lld",sum);
	}
	return 0;
}
