#include <stdio.h>
#include <stdlib.h>
#define gc getchar_unlocked
long long unsigned read()
{
        long long unsigned num=0;
        char ch=gc();
        while(ch<'0' || ch>'9')
                ch=gc();
        while(ch>='0' && ch<='9')
        {
                num=(num<<1)+(num<<3)+(ch-48);
                ch=gc();
        }
        return num;
}

	// your code goes here
