#include <bits/stdc++.h>
#define ll long long 
using namespace std;
int main()
{
	ll n,m,h,t,c,val,i;
	scanf("%lld %lld %lld",&n,&m,&h);
	vector<pair<ll,ll> > a;
	val=0;
	for(i=0;i<h;i++){
		scanf("%lld%lld",&t,&c);
		a.push_back(make_pair(c,t));
		val+=t;
	}
	sort(a.begin(),a.end());
	if(val>=n*m){
		val=c=0;
		for(i=0;i<a.size() && val<n*m;i++){
			if(val+a[i].second > n*m)
				c+=(n*m-val)*a[i].first;
				
			else
				c+=a[i].second*a[i].first;
			val+=a[i].second;
		}
		printf("%lld\n",c);
	}
	else
		printf("Impossible\n");
	return 0;
}
