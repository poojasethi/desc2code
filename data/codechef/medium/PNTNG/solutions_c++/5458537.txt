int main()
{
	long long n,m,h,i=0,h1;
	priority_queue<pii,vector<pii>, greater<pii> > q; 
	scanf("%lld %lld %lld",&n,&m,&h);

	long long tot=0,sum=0,cost=0,t,c;
	while(h1--)
	{
	 scanf("%lld %lld",&t,&c);	
	 q.push(pii(c,t));   
	 tot+=t;
    }
    if(tot<n*m)
     {
     	printf("Impossible");
     	return 0;
     }
    while(!q.empty())
    {
    	sum+=q.top().second;
    	if(sum<=n*m)
    	{
    	 cost+=q.top().first*q.top().second;
        q.pop();
       }
    	else
		{
		 cost+=(q.top().second-(sum-(n*m)))*q.top().first;
		 break;
	    }   
    }
    printf("%lld",cost);
	return 0;
