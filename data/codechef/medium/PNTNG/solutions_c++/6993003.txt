#include <iostream>
#include <algorithm>
using namespace std;

pair<long long int,long long int>ct[100001];
int main(){
	long long int n,m,h,cnt,i,cost=0;
	cin>>n>>m>>h;
	for(i=0;i<h;++i){
		cin>>ct[i].second>>ct[i].first;
	}
	sort(ct,ct+h);
	cnt=n*m;
	i=0;
	while(cnt && i<h){
		if(cnt>ct[i].second)
		{cost+=ct[i].first*ct[i].second;
		cnt-=ct[i].second;}
		else
		 {
		  cost+=ct[i].first*cnt;
		cnt=0;
		 }
		i++;
	}
	if(cnt)
		cout<<"Impossible";
	else
		cout<<cost;
	return 0;
	}
