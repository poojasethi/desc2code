ll gcd(ll a,ll b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
ll lcm(ll a,ll b)
{
    return (a*b/gcd(a,b));
}

struct cost
{
    ll t,c;
};
bool comp(struct cost a,struct cost b)
{
    if(a.c==b.c)
        return a.t<b.t;
    return a.c<b.c;
}

   ll n,m,h,i,total=0,ans=0;
   n=Rll();

   h=Rll();
   struct cost arr[h];
   for(i=0;i<h;i++)
   {
       arr[i].t=Rll();
       arr[i].c=Rll();
       total+=arr[i].t;
   }
   if(total<n*m)
   {
       ps("Impossible");
       return 0;
   }
   else
   {
       ll req=n*m;
       sort(arr,arr+h,comp);
       for(i=0;i<h&&req>0;i++)
       {
           if(req>=arr[i].t)
           {
               ans+=arr[i].c*arr[i].t;
               req-=arr[i].t;

           else
           {
               ans+=arr[i].c*req;
               req-=req;
           }
       }
