#include<bits/stdc++.h>
#define lli long long int
using namespace std;
struct node{
lli max_cells,cost;
};
int fun(struct node x,struct node y)
{
	if(x.cost<y.cost)
		return 1;
	else
		return 0;
}
int main()
{
	lli cells,current,i,j,k,amount,m,n,t,h;
	scanf("%lld%lld%lld",&n,&m,&h);
	cells=n*m;
	struct node arr[h];

	for(i=0;i<h;i++)
		scanf("%lld%lld",&arr[i].max_cells,&arr[i].cost);

	sort(arr,arr+h,fun);
	current=cells;
	amount=0;
	for(i=0;i<h;i++)
	{
		if(current==0)
			break;
		if(current>=arr[i].max_cells)
		{
			current=current-arr[i].max_cells;
			amount=amount+arr[i].max_cells*arr[i].cost;
		}
		else
		{
			amount+=arr[i].cost*(current);
			current=0;
		}
	}
	if(current==0)
		printf("%lld\n",amount);
	else
		printf("Impossible\n");
}
