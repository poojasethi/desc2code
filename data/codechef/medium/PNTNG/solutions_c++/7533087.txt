#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

struct layer
{
	ll no;
	ll cost;
};

bool comp(layer a,layer b)
{
	return a.cost<b.cost;
}

int main()
{
	ll n,m;
	int h;
	cin>>n>>m>>h;
	ll tot=n*m;
	layer l[h];
	for(int i=0;i<h;i++)
	{
		cin>>l[i].no>>l[i].cost;
	}
	sort(l,l+h,comp);
	ll sum=0;
	ll ans=0;
	int i=0;
	while(i<h && (sum+l[i].no<tot))
	{
		sum+=l[i].no;
		ans+=(l[i].cost*l[i].no);
		i++;
	}
	if(i==h)
	{
		cout<<"Impossible\n";
	}
	else
	{
		ans+=(l[i].cost*(tot-sum));
		cout<<ans<<"\n";
	}
	return 0;
}
