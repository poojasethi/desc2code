#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

#define ll long long
#define pb push_back

vector<pair<ll,ll> > vec;

int main()
{
ll n,m,h,i=0,t,c,total;
cin>>n>>m>>h;
total = n*m;
while(h--)
 {
 cin>>t>>c;
 vec.pb(make_pair(c,t));
 i+=t;
 }
if(i < total)
  cout<<"Impossible"<<endl;
else
  {
  c=0;
  sort(vec.begin(),vec.end());
  vector<pair<ll,ll> >::iterator it;
  for(it = vec.begin();it != vec.end();it++)
     {
     t = min(total,it->second);
     c += t*(it->first);
     total -= t;
     }
  cout<<c<<endl;
  }
return 0;
}
