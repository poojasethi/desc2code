#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
struct paint
{
    long long n,cost;
};
bool cmp(paint a,paint b)
{
    return a.cost<b.cost;
}
int main()
{
    long long r,c,h,sum=0,ans=0,x=0;
    cin>>r>>c>>h;
    paint p[h+5];
    for(long long i=0;i<h;i++)
    {
        cin>>p[i].n>>p[i].cost;
        sum+=p[i].n;
    }
    if(sum<(r*c))
    {
        cout<<"Impossible\n";
    }
    else
    {
        sort(p,p+h,cmp);
        long long i=0;
        long long tot=r*c;
        while(tot>0)
        {

            if(p[i].n>tot)
            {
                  ans+=(tot*p[i].cost);
                  tot=0;
            }
            else
            {
                 ans+=p[i].n*p[i].cost;
                 tot-=p[i].n;
            }
            i++;
        }
        cout<<ans<<endl;
    }
}
