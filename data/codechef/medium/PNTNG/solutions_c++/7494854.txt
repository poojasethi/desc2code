#include <bits/stdc++.h>
using namespace std;

typedef long long LL;
typedef pair < LL, LL > pii;
#define fr first
#define se second

bool cmp( pii a, pii b )
{
	return( a.se == b.se ? a.fr < b.fr : a.se < b.se );
}

pii a[1000010];

int main()
{
	LL n, m, h;
	cin >> n >> m >> h;
	for( int i = 0 ; i < h ; i ++ )
		cin >> a[i].fr >> a[i].se;
	sort( a, a+h, cmp );
	n *= m;
	int i = -1;
	LL ans = 0;
	while( ++i < h )
	{
		if( a[i].fr <= n )
			ans += a[i].fr * a[i].se, n -= a[i].fr;
		else
			ans += n * a[i].se, n = 0;
	}
	if( n > 0 )
		cout << "Impossible" << endl;
	else
		cout << ans << endl;
	return 0;
}
