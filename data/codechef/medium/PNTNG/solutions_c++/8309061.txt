	#include<iostream>
	#include<map>
	using namespace std;
	int main()
	{
	
			multimap< long long int ,long long  int > myset;
			long long  int n,m,h;
			cin>>n>>m>>h;
			 long long int tk,ck;
			for(int i=0;i<h;i++)
			{
				cin>>tk;
				cin>>ck;
				myset.insert(make_pair(ck,tk));
			}	
			long long int cost=0;
			long long int max=n*m;
			long long int itemsadded=0;
			for(multimap<long long int , long long int >::iterator it=myset.begin();it!=myset.end();it++)
			{
				tk=it->second;
				ck=it->first;
				if(tk+itemsadded>=max)
				{

					cost=cost+ck*(max-itemsadded);
					itemsadded=max;
					break;
				}	
				else
				{
					cost=cost+tk*ck;
					itemsadded+=tk;
				}	
			}	
			if(itemsadded<max)
				cout<<"Impossible\n";
			else
				cout<<cost<<endl;
	
	}
