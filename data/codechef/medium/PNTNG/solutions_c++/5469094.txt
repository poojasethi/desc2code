#include <iostream>
#include <algorithm>
#include <cmath>

using namespace std;

struct st{
    long long t;
    long long c;
    
    bool operator<(const st &s) const {
        return c < s.c;
    }
};

int main()
{
    long long N, M, H;
    cin >> N >> M >> H;
    st p[H];
    long long poss = 0;
    for(int i = 0; i < H; i++){
        long long t, c;
        cin >> t >> c;
        st a;
        a.t = t;
        a.c = c;
        p[i] = a;
        poss += t;
    }
    sort(p, p + H);
    if(poss >= N*M){
        long long rem = N*M;
        long long res = 0;
        int index = 0;
        while(rem > 0){
            if(p[index].t >= rem){
                res += rem*p[index].c;
            }
            else{
                res += p[index].c*p[index].t;
            }
            rem -= p[index].t;
            index++;
        }
        cout << res << endl;
    }
    else{
        cout << "Impossible" << endl;
    }
}
