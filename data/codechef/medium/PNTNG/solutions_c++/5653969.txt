#include<bits/stdc++.h>
using namespace std;

bool compare(const pair<long long,long long>& a, const pair<long long,long long>& b)
{
        return a.second < b.second;
}

int main()
{
        long long n,m,h,i,j,t,c;
        long long sum=0, cost;
        cin >> n >> m >> h;

        vector< pair<long long,long long> > p;

        for(i=0;i<h;i++)
        {
                cin >> t >> c;
                p.push_back(make_pair(t,c));
        }

        for(i=0;i<h;i++)
        sum += p[i].first;

        if(sum < n*m)
                cout << "Impossible\n";
        else
        {
                sort(p.begin(), p.end(), compare);

                //cout << endl;
                //for(i=0;i<h;i++)
                //cout << p[i].first << " " << p[i].second << endl;

                sum = cost = 0;
                for(i=0;i<h;i++)
                {
                        if(sum + p[i].first >= n*m)
                        {
                                cost += (n*m - sum)*p[i].second;
                                break;
                        }
                        else
                        {
                                sum += p[i].first;
                                cost += p[i].first * p[i].second;
                        }
                }
                cout << cost;
        }
        return 0;
}
