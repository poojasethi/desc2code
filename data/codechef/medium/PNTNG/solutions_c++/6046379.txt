#include<iostream>
#include<utility>
#include<algorithm>
using namespace std;
int main()
{
    long long  ans,sum,money,n,m,h,i;;
        cin>>n>>m>>h;
        pair<long long ,long long > store[h];
        for(i=0;i<h;i++)
                cin>>store[i].second>>store[i].first;
            sort(store,store+h);
            sum=n*m;
            money=0;
            for(i=0;i<h;i++)
            {
                ans=min(sum,store[i].second);
                money+=ans*store[i].first;
                sum-=ans;
            }
            if(sum>0)
                cout<<"Impossible"<<endl;
            else
            cout<<money<<endl;

    return 0;
}
