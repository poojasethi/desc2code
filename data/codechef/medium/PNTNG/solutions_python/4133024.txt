#!/usr/bin/env python2.7

"""painting.py: solution to http://www.codechef.com/problems/PNTNG"""

import sys

dimensions = sys.stdin.readline().split(" ")
n, m, h = int(dimensions[0]), int(dimensions[1]), int(dimensions[2])
a = n*m
layers = []
maxsquares = 0

for i in xrange(0, h):
    layer = sys.stdin.readline().split(" ")
    Tk, Ck = int(layer[0]), int(layer[1])
    layers.append((Ck, Tk))
    maxsquares += Tk

if maxsquares < a:
    print "Impossible"
else:
    layers.sort(reverse=True)
    cost = 0
    while a > 0:
        cheapest = layers.pop()
        cost += cheapest[0] * (a if cheapest[1] > a else cheapest[1])
        a -= cheapest[1]

    print cost
