n,m,h=map(int,raw_input().split())
v=[]
for i in range(h):
	a,b=map(int,raw_input().split())
	v+=[(b,a)]
v.sort()
x=n*m
i=0
cost=0
f=0
while(x>=1):
	if(i<h):
		if(x>=v[i][1]):
			x-=v[i][1]
			cost+=v[i][0]*v[i][1]
		else:
		#	t=x-v[i][1]
			cost+=v[i][0]*x
			x=0
	else:
		print "Impossible"
		f=1
		break
	i+=1
if(f==0):
	print cost
