n,m,h=map(int,raw_input().split())
arr=[]
for i in range(h):
	t,c=map(int,raw_input().split())
	arr+=[(c,t)]
arr.sort()
prod=n*m
i=0
cost=0
f=0
while(prod>=1):
	if(i<h):
		if(prod>=arr[i][1]):
			prod=prod-arr[i][1]
			cost+=arr[i][1]*arr[i][0]
		else:
			cost+=arr[i][0]*prod
			prod=0
	else:
		print "Impossible"
		f=1
		break
	i+=1
if(f==0):
	print cost
