#include<bits/stdc++.h>

using namespace std;

long long int MOD = 10000007;

int main(){
    long long int n,m,i,j,a[50][50],ans=1;
    long long int s[40]={0};
    //scanf("%lld%lld",&n,&m);
    cin>>n>>m;
    for(i=0;i<n;i++){
        for(j=0;j<m;j++){
            cin>>a[i][j];
            //scanf("%lld",&a[i][j]);
        }
    }
    for(j=0;j<m;j++){
        for(i=0;i<n;i++){
            s[j]=(s[j]+a[i][j]%MOD)%MOD;
        }
    }
    //for(i=0;i<m;i++) cout<<s[i]<<" ";
    //cout<<"\n";
    for(i=0;i<m;i++){
        ans=((ans%MOD)*((s[i])%MOD))%MOD;
    }
    ans=(ans+MOD)%MOD;
    cout<<ans<<"\n";
return 0;
}
