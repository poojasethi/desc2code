#include <bits/stdc++.h>
using namespace std;
#define MOD 10000007
typedef long long ll;
int main()
{
	int n,m,i,j,column_sum[40]={},temp;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			scanf("%d",&temp);
			column_sum[j]=(column_sum[j]+temp+MOD)%MOD;
		}
	}
	int ans=1;
	for(j=0;j<m;j++)
		ans=(((ll)ans)*column_sum[j])%MOD;
	printf("%d\n", ans);
	return 0;
}
