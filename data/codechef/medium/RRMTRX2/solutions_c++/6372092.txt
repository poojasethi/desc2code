#include <iostream>
using namespace std;

#define ll long long int
#define mod 10000007
int main() {
	// your code goes here
    
     ll n,m;
     cin>>n>>m;
     ll a[n+1][m+1];
     for(ll i=1;i<=n;i++)
      for(ll j=1;j<=m;j++)
       {
           cin>>a[i][j];
       }
       
     ll ans=1;
     for(ll j=1;j<=m;j++)
      {
          ll sum = 0;
          for(ll i=1;i<=n;i++)
           sum+=a[i][j];
          ans = (ans*sum)%mod; 
      }
       
       cout<<(ans+mod)%mod<<endl;

	return 0;
}
