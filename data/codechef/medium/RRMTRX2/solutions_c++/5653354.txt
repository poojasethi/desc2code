#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
void calculatesum(int arr[50][50], int n , int m ,int64_t &sum)
{
    int64_t mullocal;
    for(int j=0;j<m;j++)
    {
        mullocal=0;
        for(int i=0;i<n;i++)
            {
                mullocal+=arr[i][j];
            }
            if(mullocal<0)
                mullocal+=10000007;
            sum=(sum*mullocal)%10000007;
    }
}
int main()
{
        int n , m;
        int arr[50][50];
        cin>>n>>m;
        int i , j;
        int64_t sum=1;
        for(i=0;i<n;i++)
            for(j=0;j<m;j++)
                cin>>arr[i][j];
        calculatesum(arr,n,m,sum);
        cout<<sum<<endl;
        return 0;
}
