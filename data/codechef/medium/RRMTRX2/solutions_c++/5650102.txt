#include<bits/stdc++.h>
using namespace std;
int a[45][45];
int main()
{
	int n,m;
		scanf("%d%d",&n,&m);
		for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		scanf("%d",&a[i][j]);
		long long k=1;
		for(int i=0;i<m;i++)
		{
			long long s=0;
			for(int j=0;j<n;j++)
			s+=a[j][i];
			k=(k*s)%10000007;
		}
		k=(k+10000007)%(10000007);
		printf("%lld\n",k);
	return 0;
}
