    #include <stdio.h>
     
    #define MOD 10000007
    #define lld long long int
     
    int main()
    {
    	int n,m,i,j;
    	scanf("%d%d",&n,&m);
    	int a[n][m];
    	lld ar[m];
    	for(i=0;i<m;i++)
    		ar[i]=0;
    	for(i=0;i<n;i++)
    		for(j=0;j<m;j++)
    			scanf("%d",&a[i][j]);
    	for(j=0;j<m;j++)
    	{
    		for(i=0;i<n;i++)
    		{
    			ar[j]+=a[i][j];
    		}
    	}
    	lld temp=1;
    	for(i=0;i<m;i++)
    		temp=((temp%MOD)*(ar[i]%MOD))%MOD;
    	if(temp<0)
    		temp+=MOD;
    	printf("%lld\n",temp%MOD);
    	return 0;
    }
