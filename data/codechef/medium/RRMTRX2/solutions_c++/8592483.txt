typedef long long ll;

template <typename X> void swap( X &a, X &b) {
    X tp;tp = a;a = b;b = tp;
}

ll a[maxn][maxn];
ll S[maxn];
int main()
{
	int n,m;
	cin>>n>>m;
	ll ans;
	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++) cin>>a[i][j];
	for(int i=0;i<m;i++) S[i]=0;
	for(int i=0;i<m;i++)
	{
		for(int j=0;j<n;j++)
			S[i]=(S[i]+a[j][i]+mod)%mod;
	}
	for(int i=0;i<m;i++) assert(S[i]>=0);
	ans=1;
	for(int i=0;i<m;i++)
	{
		ans=(ans*S[i]+mod)%mod;
	}
	cout<<ans<<"\n";
	return 0;
}
