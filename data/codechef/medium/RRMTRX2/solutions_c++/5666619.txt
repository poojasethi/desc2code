#include<iostream>
using namespace std;
#define mod 10000007
int main()
{
    ios_base::sync_with_stdio(false);
        int n,m;
        cin>>n>>m;
        int a[n][m];
        long long int sum=0,ans=1;
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
                cin>>a[i][j];
        }
        for(int i=0;i<m;i++)
        {
            sum=0;
            for(int j=0;j<n;j++)
            {
                sum=sum+a[j][i];
            }
            ans=(ans*sum)%mod;
        }
        if(ans<0)
            ans=ans+mod;
        cout<<ans%mod<<endl;

    return 0;
}
