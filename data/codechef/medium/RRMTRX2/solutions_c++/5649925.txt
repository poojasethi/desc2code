#include<bits/stdc++.h>
using namespace	std;

#define rep(i,a,b) for(int i = a; i < b; i++)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)
 
typedef long long int LL;
const int mod = 10000007;

int main(){
	int n,m,A[100][100];
	S(n);S(m);
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			S(A[i][j]);}}
	
	long long ans=1;
	for(int i=0;i<m;i++){
		int temp =0;
		for(int j=0;j<n;j++)
			temp = (temp+A[j][i]);
		ans = (ans*(LL)temp)%mod;	
	}
	if(ans < 0) ans+= mod;
	printf("%lld\n",ans);
	
	return 0;
}
