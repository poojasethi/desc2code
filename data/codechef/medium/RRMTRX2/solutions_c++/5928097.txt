#include<bits/stdc++.h>
#define ll long long
#define S(n) scanf("%d",&(n))
#define P(n) printf("%d",(n))
#define SL(n) scanf("%lld",&(n))
#define PL(n) printf("%lld",(n))
#define NL printf("\n")
#define SP printf(" ")
#define Debug cout<<"Hello"
#define N 100000
#define M 10000007
using namespace std;
int main()
{
	
	//freopen("input.txt","r",stdin);
	
	

	int t;
	
	//S(t);
	
	t = 1;
	
	while(t--)
	{
		
		int n;int m;
		ll mat[50][50];
		
		S(n);S(m);
		
		for(int i=1;i<=n;i++)
		{
			
			for(int j=1;j<=m;j++)
			{
				
				SL(mat[i][j]);
				
			}
			
		}
		
		
		ll ans=1;
		
		for(int j=1;j<=m;j++)
		{
			ll tmp=0;
			for(int i=1;i<=n;i++)
			{
				tmp+=mat[i][j];tmp%=M;
			}
			ans = ans*tmp%M;
		}
		
		if(ans<0)ans+=M;
		PL(ans);NL;
		
		
	}
	
	
	return 0;
	
}
