#include<bits/stdc++.h>
#define MOD 10000007
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	long long int a[100][100];
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			cin>>a[i][j];
		}
	}
	long long int ans=1;
	for(int j=0;j<m;j++)
	{
		long long int sum=0;
		for(int i=0;i<n;i++)
		{
			sum+=a[i][j];
		}
		ans=(ans%MOD)*(sum%MOD);
		ans=ans%MOD;
	}
	if(ans<0)
	  ans=ans+MOD;
	cout<<ans<<endl;
	return 0;
}
