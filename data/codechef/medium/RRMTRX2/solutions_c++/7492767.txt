#include<bits/stdc++.h>
#define lli long long int
#define MOD 10000007
using namespace std;
int main()
{
        lli i,j,k,m,n,t,sum,ans;
        scanf("%lld%lld",&n,&m);
        lli arr[n][m];

        for(i=0;i<n;i++)
                for(j=0;j<m;j++)
                        scanf("%lld",&arr[i][j]);

        ans=1;
        for(j=0;j<m;j++)
        {
                sum=0;
                for(i=0;i<n;i++)
                {
                        sum+=arr[i][j];
                        sum=sum%MOD;
                }
                ans=ans*sum;
                ans=ans%MOD;
        }
        if(ans<0)
                ans=MOD+ans;
        printf("%lld\n",ans);
}
