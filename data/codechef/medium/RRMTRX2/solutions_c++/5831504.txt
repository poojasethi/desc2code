	#include<iostream>
	using namespace std;
	int main()
	{	
		int n,m;
		cin>>n;
		cin>>m;
		int a[40]={0};
		int temp;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				cin>>temp;
				a[j]=a[j]+temp;
			
			}
		}
		long long int pro=1;
		for(int i=0;i<m;i++)
		{
			pro=(pro*a[i])%10000007;
		
		}
		cout<<(pro+10000007)%10000007<<endl;
	}
