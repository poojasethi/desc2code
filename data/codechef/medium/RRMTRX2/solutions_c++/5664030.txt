#include <bits/stdc++.h>

using namespace std;

#define ll long long
int n,m;
ll a[55][55];
ll mo = (int)1e7+7;

ll modp(ll a, ll b)
{
    if(!b) return 1;
    if(b==1) return a%mo;
    ll c= modp(a,b/2) ;
    c*=c;
    c%=mo;
    if(b&1) c*=a;
    c%=mo;
    return c;
}

ll dp[55];
ll solve(int prod, int idx)
{
    //cout << prod <<" "<<idx<<endl;
    prod%=mo;
    if(idx == m)
    {
        return prod;
    }
    if(dp[idx]!=-1)return dp[idx];
    ll s = 0;
    for(int i=0;i<n;i++)
    {
        s += ((a[i][idx]+mo)%mo )* solve(prod, idx +1);
        s %=  mo;
    }
    dp[idx] = s;
//    dp[make_pair(prod,idx)] = s;
    return s;
}
int main()
{

    cin>>n>>m;

    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++) cin>>a[i][j];
    }

    ll p =  1;
//    dp.clear();
memset(dp, -1, sizeof(dp));
   p = solve(1, 0);
    cout<<p<<endl;
    return 0;
}
