#include <iostream>
using namespace std;

#define maxn 47
#define maxm 38
#define mod 10000007

int a[maxn][maxm];

int main() {
    long long sum, product;
    int n, m;
    cin >> n >> m;

    for(int i=0;i<n;++i) {
        for(int j=0;j<m;++j) {
            cin >> a[i][j];
        }
    }

    product = 1;
    for(int j=0;j<m;++j) {
        sum = 0;
        for(int i=0;i<n;++i) {
            sum = (sum + a[i][j] + mod) % mod;
        }
        product = (product * sum) % mod;
    }

    cout << product << endl;
    return 0;
}
