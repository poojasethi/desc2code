#include<iostream>
#include<vector>
#include<utility>
#include<algorithm>
#define mod 10000007
using namespace std;
long long arr[50][50];
long long dp[100],n,m,i,j,sum;    
int main()
{
        cin>>n>>m;
        for(i=0;i<n;i++)
        {
        
                for(j=0;j<m;j++)
                {
                        cin>>arr[i][j];
                }
        }
        for(i=0;i<m;i++)
        {
               sum=0;
                for(j=0;j<n;j++)
                 sum=(sum+arr[j][i])%mod;
                 dp[i]=sum;
        }
        sum=1;
        for(i=0;i<m;i++)
        sum=(sum*dp[i])%mod;
        cout<<(sum+mod)%mod<<"\n";
	 return 0;
}
