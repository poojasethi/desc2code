#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define mod 10000007

int main()
{
	ios::sync_with_stdio(false);
	int n,m;
	cin>>n>>m;
	int a[n][m];
	ll sum=0;
	ll prod=1;
	for(int i=0;i<n;i++)
	{
		sum=0;
		for(int j=0;j<m;j++)
		{
			cin>>a[i][j];
		}
	}
	for(int i=0;i<m;i++)
	{
		sum=0;
		for(int j=0;j<n;j++)
		{
			sum+=a[j][i];
		}
		prod=(prod*sum)%mod;
		if(prod<0)
		{
			prod+=mod;
		}
	}
	cout<<prod<<"\n";
	return 0;
}
