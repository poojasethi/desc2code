#include<cstdio>
using namespace std;
#define M 10000007
int main()
{
    int n,m,x;
    scanf("%d %d",&n,&m);
    int sum[40]={0};
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            scanf("%d",&x);
            sum[j]+=x;
        }
    }
    long long res=1;
    for(int i=0;i<m;i++)res=(res*sum[i])%M;
    printf("%lld",(res+M)%M);
    return 0;
}
