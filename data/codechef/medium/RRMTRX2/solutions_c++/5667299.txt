#include<iostream>
#include<cstdio>
#define mod 10000007
using namespace std;
int main()
{
    int a[50][40],n,m;
    int b[50];
    long long ans=1;
    cin>>n>>m;
    //memset(b,1,n);
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            cin>>a[i][j];

    for(int i=0;i<m;i++)
    {
        b[i]=0;
        for(int j=0;j<n;j++)
        {
            b[i]=b[i]+a[j][i];
        }
    }
    for(int i=0;i<m;i++)
        ans=((ans%mod)*(b[i]%mod))%mod;
    printf("%lld\n", ans >= 0 ? ans: mod+ans);
}
