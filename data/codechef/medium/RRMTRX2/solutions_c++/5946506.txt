#include <iostream>
#include <cstdio>

const long long int MOD= 10000000+7;

using namespace std;

int main(){

	int n, m;
	scanf("%d %d",&n, &m);
	
	//creating a 2-d array

	
	long long int **a = new long long int*[n];
           
	
	for(int i=0; i < n; i++)
		a[i] = new long long int[m];


        for(int r=0; r < n; r++)
		for(int c=0; c < m; c++)
			scanf("%lld",&a[r][c]);
         
	long long int sum =0 ;
	
	long long int ans =1;
	for(int c=0; c < m; c++){
		sum=0;
		for(int r=0; r <n; r++){
                         sum += a[r][c];
		}


		ans =(ans * sum) % MOD;
	//	ans %= MOD;

	}

	ans = (ans + MOD)% MOD;

	printf("%lld \n", ans);

       //cleanup code
	for(int i=0; i < n; i++)
		delete [] a[i];
	delete [] a;
			
	return 0;

}
