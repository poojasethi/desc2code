#define trace6(a,b,c,d,e,f) cerr << #a << ": " << a << " | " << #b << ": " << c << " | " << #c << ": " << c << " | " << #d << ": " << d  << " | " << #e << ": " << e << " | " << #f << ": " << f << "\n";

int a[50][50];
int s[50];

int main() {
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++) for(int j=0;j<m;j++) scanf("%d",&a[i][j]);
	for(int j=0;j<m;j++) for(int i=0;i<n;i++) s[j]+=a[i][j];	
	LL prod=s[0];
	for(int i=1;i<m;i++) {
		prod = (prod*s[i])%MOD;
	}
	if(prod<0) prod+=MOD;
	printf("%lld\n",prod%MOD);
	return 0;
}
