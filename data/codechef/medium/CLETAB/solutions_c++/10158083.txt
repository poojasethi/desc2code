#include<bits/stdc++.h>
using namespace std;
int orders[500];
vector<vector<int> > lst(500);
bool occ[500];
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		int N, M;
		scanf("%d%d", &N, &M);
		for(int i = 1;i <= M;++i){
			scanf("%d", &orders[i]);
		}
		for(int i = M;i >= 1;--i){
			lst[orders[i]].push_back(i);
		}
		memset(occ, 0, sizeof occ);
		int cus = 0;
		int ans = 0;
		for(int i = 1;i <= M;++i){
			if(!occ[orders[i]]){
				if(cus < N){
					++cus;
					++ans;
				} else{
					int maxVal = 0;
					int maxId = 0;
					for(int j = 1;j <= 400;++j){
						if(occ[j]){
							if(lst[j].empty()){
								maxId = j;
								break;
							} else if(lst[j][lst[j].size() - 1] > maxVal){
								maxVal = lst[j][lst[j].size() - 1];
								maxId = j;
							}
						}
					}
					occ[maxId] = false;
					++ans;
				}
				occ[orders[i]] = true;
			}
			lst[orders[i]].pop_back();
		}
		printf("%d\n", ans);
	}
	return 0;
}
