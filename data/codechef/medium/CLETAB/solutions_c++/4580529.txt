#include<iostream>
using namespace std;
int main(){
	int n,m,t,v,count,fill;
	//int cust[205]={0};//gives the customer's first position
	int table[205];
	int order[402];
	cin>>t;
	while(t){
		t--;
		cin>>n>>m;
		fill=0;
		count=0;
		
		for(int i=0;i<m;i++){
			cin>>order[i];
		}
		
		for(int i=0;i<m;i++){
			int check=0;//check(=0)gives the customer is not already there
			
			 for(int k=0;k<fill;k++){
					if(table[k]==order[i]){
						check=1;
						break;
					}
				}
			if(!check){
			
				if(fill<n){
					table[fill]=order[i];
					count++;
					fill++;
				}
				
				else if(fill == n){
					
					int found;//to break if for a customer no furter orders are found
					int max=-1,imax=-1;//max is val of order farthest, imax is the table no.
					for(int j=0;j<n;j++){
						
						
						found=0;
						for(int k=i+1;k<m;k++){
							if(table[j]==order[k]){
								if(max<k){
									max=k;
									imax=j;
									
								}
								found=1;
								break;
							}
						}
						if(found==0){
							imax=j;
							break;
						}
					}
					table[imax]=order[i];
					
					count++;
					
				}
				
			}
			
		}
		
		cout<<count<<"\n";
	}
}
