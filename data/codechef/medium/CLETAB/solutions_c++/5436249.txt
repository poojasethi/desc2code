#include <cstdio>
#include<vector>
#include<iostream>
using namespace std;

int main() {
	int test;
	scanf("%d",&test);
	while(test--)
	{
		int n,m;
		scanf("%d%d",&n,&m);
		int tables[n],odr[m];
		bool flag[401];
		vector<int>v[401];
	
		for(int i=0;i<n;i++)
		{
			tables[i]=0;
		}
		for(int i=0;i<m;i++)
		{
			scanf("%d",&odr[i]);
			v[odr[i]].push_back(i);
			flag[odr[i]]=false;
		}
		int k=0,no=0;
		for(int i=0;i<m;i++)
		{
		    if(flag[odr[i]]) continue;
			if(k<n)
			{
			if(tables[k]==0)
			{
				tables[k]=odr[i];
				flag[odr[i]]=true;
				k++;
				no++;
				}
			}
			else
			{
			   int in=0,ind=0,inde=0,s;
			   for(int j=0;j<n;j++)
			   {
			   	     in=-1;
			   	     s=tables[j];
			   	     for(int t=0;t<(int)v[s].size();t++)
			   	     {
			   	     	if(i<v[s][t])
			   	     	{
			   	     		in=v[s][t];
			   	     		break;
			   	     	}
			   	     }
			   	     	if(in==-1)
			   	     	{
			   	     		inde=j;
			   	     		break;
			   	     	}
			   	     
			   	     if(ind<in)
			   	     		{
			   	     			ind=in;
			   	     			inde=j;
			   	     		}
			   	     
			   	     
			   }
			   	     flag[tables[inde]]=false;
			   	     flag[odr[i]]=true;
			   	     tables[inde]=odr[i];
			   	     no++;
			}
		}
		
		printf("%d\n",no);
		
	}

}
