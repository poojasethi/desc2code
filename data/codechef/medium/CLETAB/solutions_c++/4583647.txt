    #include<stdio.h>
    #include<iostream>
     
    using namespace std;
     
    int main()
    {
    int t;
    cin >> t;
    while(t--){
    int fr[402];
    int p[402];
    int n,m;
    cin >> n >> m;
    for(int i = 0 ; i < m ; i++){
    cin >> p[i];
    }
    int i,j,fs[402];
    int max,found=0,lg[402],index,k,l,flag1=0,flag2=0,pf=0,frsize=n;
    for(i=0;i<=402;i++)
    {
    fr[i]=-1;
    }
    long clean = 0;
    for(j=0;j<m;j++)
    {
    flag1=0;
    flag2=0;
    for(i=0;i<n;i++)
    {
    if(fr[i]==p[j])
    {
    flag1=1;
    flag2=1;
    break;
    }
    }
    if(flag1==0)
    {
    for(i=0;i<n;i++)
    {
    if(fr[i]==-1)
    {
    fr[i]=p[j];
    clean++;
    flag2=1;
    break;
    }
    }
    }
     
    if(flag2==0)
    {
    for(i=0;i<n;i++)
    lg[i]=0;
    for(i=0;i<frsize;i++)
    {
    for(k=j+1;k<m;k++)
    {
    if(fr[i]==p[k])
    {
    lg[i]=k-j;
    break;
    }
    }
    }
    found=0;
    for(i=0;i<frsize;i++)
    {
    if(lg[i]==0)
    {
    index=i;
    found=1;
    break;
    }
    }
    if(found==0)
    {
    max=lg[0];
    index=0;
    for(i=1;i<frsize;i++)
    {
    if(max<lg[i])
    {
    max=lg[i];
    index=i;
    }
    }
    }
    fr[index]=p[j];
    clean++;
    pf++;
    }
    }
    cout << clean << "\n";
    }
    }
