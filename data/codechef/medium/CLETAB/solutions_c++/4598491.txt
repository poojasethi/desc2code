#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<vector>
#include <cstdio>

using namespace std;

int main()
{
    long n, m , i, j, k, t, arr[200000];
    scanf("%ld", &t);
    while(t--)
    {
    vector<long>v[410];
    long max, cnt=0, rem, order[410]={0}, occ=0, table[210]={0}, given[410]={0};
    scanf("%ld%ld", &n, &m);
    for(i=0; i<m; i++)
    {
        scanf("%ld", &arr[i]);
        order[arr[i]]=i;
        v[arr[i]].push_back(i);
    }
    for(i=0; i<m; i++)
    {
        if(given[arr[i]]) continue;
        if(occ != n) {
            cnt++;
            table[occ]=arr[i];
            given[arr[i]]=occ+1;
            occ++;
        } else {
            cnt++;
            max=rem=0;
            for(j=0; j<n; j++) {
                if(order[table[j]] < i)
                {
                    rem=j;
                    break;
                }
                for(k=0; k<v[table[j]].size(); k++)
                    if(v[table[j]][k] > i)
                    break;
                if(max<v[table[j]][k])
                {
                    max=v[table[j]][k];
                    rem=j;
                }
            }
                given[arr[i]]=rem+1;
                given[table[rem]]=0;
                table[rem]=arr[i];
            }
        }

    printf("%ld\n", cnt);

    }
        return 0;
}
