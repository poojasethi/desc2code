#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define F first
#define S second
#define MAX 405
#define M 1000000007
using namespace std;

int main ()
{
 

int t;

	sc(t);
	while(t--)
	{
		vector<set<int> > g(405);
			int n,m,x,i,s,min,max;
				set<int>e;
	set<int>::iterator it,it1;
		int c[MAX],a[MAX];
		s=0;
		sc(n);
		sc(m);
		for(i=1;i<=400;i++)
		{
			c[i]=0;
			a[i]=0;
		}
		for(i=1;i<=m;i++)
		{
			sc(a[i]);
			g[a[i]].insert(i);
			c[a[i]]++;
		}
		{
			for(i=1;i<=m;i++)
			{
				if(e.size()<n)
				{
					it=e.find(a[i]);
					if(it!=e.end())
					{
						c[a[i]]--;
					}
					else
					{
					s++;
					c[a[i]]--;
					e.insert(a[i]);
					}
				}
				else if(e.size()==n)
				{
					it=e.find(a[i]);
					if(it!=e.end())
					{
						c[a[i]]--;
					}
					else
					{
						min=1000;
						for(it=e.begin();it!=e.end();it++)
						{
							if(c[*it]<min)
							{
								min=c[*it];
								it1=it;
							}
						}
						if(min==0)
						{
							e.erase(it1);
							e.insert(a[i]);
							c[a[i]]--;
							s++;
					    }
					    else
					    {
					    	max=0;
					    	for(it=e.begin();it!=e.end();it++)
					    	{
					    		if(*g[*it].lower_bound(i) >max)
					    		{
					    			max=*g[*it].lower_bound(i);
					    			it1=it;
					    		}
					    	}
					    		e.erase(it1);
							e.insert(a[i]);
							c[a[i]]--;
							s++;
					    }
					}					
				}
			}
			pf(s);
			e.clear();
		}
	}
	return 0;
}
