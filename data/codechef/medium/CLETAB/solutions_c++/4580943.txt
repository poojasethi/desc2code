#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int a[512],b[512],c[512];
	int t;	//p for pivot
	scanf("%d",&t);
	while(t--)
	{
			//a to store sequence, b to store the nearest same customer, c stands for status of the tables
		int i,j,n,m,k;
		scanf("%d%d",&n,&m);
		
		for(i=0;i<m;i++)
		{
			scanf("%d",&a[i]);	//input
			b[i]=m;		//assuming no repeat customers
		}
		
		for(i=0;i<m;i++)
			for(j=i+1;j<m;j++)
				if(a[i]==a[j])
				{
					b[i]=j;		//finding the nearest same customers
					break;
				}
		for(i=0;i<n;i++)
		c[i]=-1;			//starting status of tables
		int ans=0;		
		for(i=0;i<m;i++)
		{
			int p=-1;		//for greedy algo
			for(j=0;j<n;j++)
				if(c[j]>=0 && a[c[j]]==a[i])		// checking whether the same customer is already eating
				{
					p=j;
					c[p]=i;
					break;
				}
			if(p>=0)		// if yes then no need for cleaning
			{
				continue;
			}
			ans++;		//else increment the value
			
			for(j=0;j<n;j++)	// if a table is vacant then accomodate that first
			{
				if(c[j]<0)
				{
					p=j;
					break;
				}
			}
		
			if(p<0)		// else remove the customer who will order again the last among the seated customers
				for(j=0;j<n;j++)
					if(p<0 || b[c[j]]>b[c[p]])
					{
						p=j;
							
					}
			c[p]=i;
		}
		printf("%d\n",ans);
	}
	return 0;
}
