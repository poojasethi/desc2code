#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
	int i,j,k,l,T,N,M,ft,max,x,count,top,flag;
	scanf("%d", &T);
	for(i=1;i<=T;i++) {
		scanf("%d%d", &N,&M);
		int arr[M+1],stat[M+1];
		count=0;
		ft=0;
		for(j=1;j<=M;j++) {
			scanf("%d", &arr[j]);
		}
		top=1;
		for(j=1;j<=M;j++) {
			flag = 0;
			for(k = 1;k<top;k++) {
				if(stat[k]==arr[j]) {
					flag = 1;
					break;
				}
			}
			if(flag==0) {
				if(ft<N) {
					count++;
					stat[top++]=arr[j];
					ft++;
				} else {
					max = -1;
					for(k=1;k<top;k++) {
						for(l=j+1;l<=M;l++) {
							if(arr[l]==stat[k]) break;
						}
						if(max<l) {
							max=l;
							x = k;
						}
					}
					if(max==-1) {
						x = top-1;
					}
					for(k=x;k<=(top-2);k++) {
						stat[k]=stat[k+1];
					}
					top = top-1;
					stat[top++]=arr[j];
					count++;
				}
			}
		}
		printf("%d\n", count);
	}
	return 0;
}
