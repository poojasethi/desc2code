#include<iostream>
#include<algorithm>
#include<limits.h>

const int INF=10000000;
using namespace std;
struct Comp {
    bool operator () (const pair<int,int> &a, const pair<int,int> &b)
    {
        if (a.first == b.first)
            return a.second < b.second;

        return a.first < b.first;
    }
};

int main()
{
   int t;
   cin>>t;
   while(t--)
   {
             
    int n,m,i,cnt=0,j,min,occ,k,rem,pos;
    int a[500],h[500],set[500]={0};
    
     for(i=0;i<500;i++)
      set[i]=0;
 
    cin>>n>>m;
    
    for(i=0;i<m;i++)
    cin>>a[i];
   
   i=0;  
   cnt=0;
  while(cnt<n && i<m)
  {
    if(set[a[i]]==0)
    {
       set[a[i]]=1;
       h[cnt]=a[i];
       cnt++;
       }
       i++;
    }
   
       for(;i<m;i++)
       {
            if(set[a[i]]==0)
            {
                min=-1;
              
               for(j=0;j<n;j++)
               {
                    occ=0;            
                   for(k=i+1;k<m;k++)
                   {
                     occ++;              
                     if(h[j]==a[k])              
                      break; 
                      }
                      if(k==m)
                      occ=INT_MAX;
                    if(occ > min)
                    {
                       min=occ;
                       pos=j;
                       } 
                   }    
                   
                    //cout<<pos;
                    set[h[pos]]=0;
                    set[a[i]]=1;
                    h[pos]=a[i];
                     cnt++;         
                   }
                  
             }           
                             
       cout<<cnt<<"\n";      
       }
        //getchar();
        return 0;
        }
