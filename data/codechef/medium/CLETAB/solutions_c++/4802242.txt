//tutorial
#include<cstdio>
#include<cstring>
#include<vector>
#include<iostream>
using namespace std;
int main()
{
    int T,n,m,cus[401],last_seen[401],last_use[401];bool present[401];
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&n,&m);
        for(int i=0;i<m;i++){scanf("%d",&cus[i]);last_use[cus[i]]=i;}
        memset(last_seen,0,sizeof(int)*401);
        memset(present,false,sizeof(bool)*401);
        int ord=0,ans=0;
        while(ord<m && (ans<n || present[cus[ord]]))
        {
            if(!present[cus[ord]])
            {
                ans++;present[cus[ord]]=true;
            }
            ord++;
        }
        while(ord<m)
        {
            if(!present[cus[ord]])
            {
                int victim=-1;
                for(int i=ord-1;i>=0;i--)
                {
                    if(present[cus[i]] && last_use[cus[i]]==i)
                    {
                        victim=cus[i];break;
                    }
                }
                if(victim==-1)
                {
                    for(int i=ord+1;i<m;i++)
                    {
                        if(present[cus[i]] && last_seen[cus[i]]!=ord)
                        {
                            victim=cus[i];
                            last_seen[cus[i]]=ord;
                        }
                    }
                }
            present[cus[ord]]=true;
            present[victim]=false;
            ans++;
            }
        ord++;
        }
        printf("%d\n",ans);
    }
    return 0;
}
