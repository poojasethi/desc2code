#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <vector>

using namespace std;

int main() {
   int i, j, k, t;

    int maxtab, maxcust;
     cin >> t;

     while (t--) {
        cin >> maxtab >> maxcust;
        int a[405];

        int table[405] = {0};
        int maxindex[405] = {0};
        int flag[405] = {0};
        vector <int> v[405];

        for (i = 0; i < maxcust; ++i) {
            cin >> a[i];
            v[a[i]].push_back(i);
            maxindex[a[i]] = i;
        }
        int cnt = 0, ans = 0;
        for (i = 0; i < maxcust; ++i) {
            if (flag[a[i]]) continue;




            if (cnt != maxtab) {
                    ans++;
                table[cnt] = a[i];
                flag[a[i]] = cnt + 1;
                cnt++;
            } else {
                ans++;
                int index = 0, l = 0;
                for (j = 0; j < maxtab; ++j) {
                    if (maxindex[table[j]] < i) {
                        index = j;
                        break;
                    }
                    for (k = 0; k < v[table[j]].size(); ++k) {
                        if (v[table[j]][k] > i) {
                            break;
                        }
                    }
                    if (l < v[table[j]][k] ) {
                        l = v[table[j]][k];
                        index = j;
                    }
                }

                flag[a[i]] = index + 1;
                flag[table[index]] = 0;
                table[index] = a[i];
            }

        }
        cout << ans << endl;
     }
     return 0;
}
