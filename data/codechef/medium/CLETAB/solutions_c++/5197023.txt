#include<stdio.h>
main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		int a[550],i,prsnt[500]={0},np=0,b[500],ap[500],seat[500]={0},rr,fr,ans=0;
		int m,n;
		scanf("%d%d",&n,&m);
		for(i=1;i<=m;i++)
		scanf("%d",&a[i]);
		for(i=m;i>=1;i--)
		{if(prsnt[a[i]]==0)
		{prsnt[a[i]]=i;ap[i]=9999;
		}
		else
		{ap[i]=prsnt[a[i]];prsnt[a[i]]=i;
		}
		}
		/*for(i=1;i<=m;i++)
		printf("%d  ",ap[i]);
		printf("\n");*/
		rr=0;//fr=rr+n;
		/*for(i=fr;i>=1;i--)
		frq[a[i]]++;
		for(i=1;i<=fr;i++)
		ap[i]=frq[a[i]]--;*/
		rr++;
		while(rr<=m)
		{
			 if(seat[a[rr]]==1)
			{
				for(i=0;i<np;i++)
				{
				if(a[b[i]]==a[rr])
				{
				b[i]=rr;break;
				}
				}
			}
			else if(seat[a[rr]]==0 && np<n)
			{
				ans++;//printf("when rr=%d np=%d in if esle 1\n",rr,np);
				seat[a[rr]]=1;b[np++]=rr;
			}
			else if(seat[a[rr]]==0 && np>=n)
			{
				int max=-99,pos;
				for(i=0;i<np;i++)
				if(max<ap[b[i]])
				{max=ap[b[i]];
				pos=i;
				}
				seat[a[b[pos]]]=0;
				b[pos]=rr;
				ans++;
				seat[a[rr]]=1;
			}
			rr++;
		}
		printf("%d\n",ans);
	}
}
