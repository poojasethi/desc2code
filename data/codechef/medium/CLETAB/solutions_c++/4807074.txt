#include <bits/stdc++.h>
using namespace std;
int a[567],f[567];
int ans=0;
bool seat[567];
bool seated[567];
	
int main() {
	int t;
	cin>>t;
	while(t--)
	{
		ans=0;
		memset(seat, false, sizeof(seat));
		memset(f, false, sizeof(f));
	int n,m;
	
	cin>>n>>m;
	int vacant=n;
	for(int i=0;i<m;i++)
	{
		cin>>a[i];
		f[a[i]]++;
	}
	for(int i=0;i<m;i++)
	{
		if(!seat[a[i]])
		{
			if(vacant!=0)
			{
				seat[a[i]]=true;
				vacant--;
			}
			else
			{
				int last;
				memcpy(seated, seat, sizeof(seat));
				for(int j=i+1;j<m;j++)
				{
					if(seated[a[j]])
					{
						last=a[j];
						seated[a[j]]=false;
					}
				}
				seat[a[i]]=true;
				seat[last]=false;
			}
			ans++;
		}
		f[a[i]]--;
		if(f[a[i]]==0)
		{
			vacant++;
			seat[a[i]]=false;
		}
		
	}
	cout<<ans<<endl;
}
	// your code goes here
	return 0;
}
