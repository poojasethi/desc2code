#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int t,n,m,i,j,w,sum,flag,l,c[400],q[200],pos[200],flag1=0;
    scanf("%d",&t);
    for(w=0;w<t;w++)
    {
        scanf("%d%d",&n,&m);
        for(i=0;i<n;i++)
         q[i]=0;
        for(i=0;i<m;i++)
            scanf("%d",&c[i]);
            sum=0;
        for(i=0;i<m;i++)
        {
            flag=0;
             for(j=0;j<n;j++)
            {
               if(c[i]==q[j]){
                  flag=1;
                  break;
               }
            }
            if(flag==1)
              continue;
                sum++;
           for(l=0;l<n;l++)
           {
               pos[l]=401;
               for(j=i+1;j<m;j++){
               if(c[j]==q[l])
               {
                   pos[l]=j;
                   break;
               }
           }
           if(pos[l]==401)
           {
               flag1=l;
            break;
           }
         }
         if(pos[flag1]==401)
            q[flag1]=c[i];
         else{
         flag=pos[0];
         l=0;
         for(j=1;j<n;j++)
         {
             if(flag<pos[j]){
                flag=pos[j];
                l=j;}
         }
          q[l]=c[i];
        }
        }

       printf("%d\n",sum);
    }
    return 0;
}
