t=input()
n=[]
while t>=1:
    n=[]
    p=raw_input()
    p=p.split()      
    for k in p:
        n.append(int(k))
    for i in range(len(n)-1):
        for j in range(len(n)-i-1):
            if n[j]>n[j+1]:
                temp=n[j]
                n[j]=n[j+1]
                n[j+1]=temp
    ways=(n[2]*n[1]-n[2]-2*n[1]+2)*n[0]
    ans=ways%1000000007
    print ans
    t=t-1
