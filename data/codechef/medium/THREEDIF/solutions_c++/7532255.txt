#include <iostream>
#include<algorithm>
using namespace std;
 
int main() {
	int t;
	long long int n[3];
	cin>>t;
	while(t--)
	{
		cin>>n[0]>>n[1]>>n[2];
		sort(n,n+3);
		long long int ans=((n[0]%1000000007)*((n[1]-1)%1000000007)%1000000007);
		ans=(ans*((n[2]-2)%1000000007)%1000000007);
		cout<<ans;
		if(t!=0)
		 cout<<endl;
	}
	return 0;
}
