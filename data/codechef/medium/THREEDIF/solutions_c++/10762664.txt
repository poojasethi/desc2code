#include <stdio.h>
#include <algorithm>


const int mod = (int)1e9 + 7;

int main(){
	int t, i;
	long long int a[3], ans;

	scanf("%d",&t);
	while(t--){
		for(i=0;i<3;++i)
			scanf("%lld",&a[i]);

		std::sort(a,a+3);
		ans=1;
		for(i=0;i<3;++i){
			ans *= ((a[i]-i) % mod);
			ans %= mod;
		}

		printf("%lld\n", ans);
	}
	return 0;
}
