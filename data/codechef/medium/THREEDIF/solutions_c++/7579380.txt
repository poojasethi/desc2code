#include<stdio.h>
#include<algorithm>
#define mod 1000000007

int main()
{int t,i,j;
 unsigned long long int a[3],ways;

 scanf("%d",&t);

 while(t--)
    {for(j=0;j<3;j++)
            scanf("%lld",&a[j]);
        std::sort(a,a+3);
        ways = 1;
        for(i=0;i<3;++i)
        {
            ways = (ways*((a[i]-i)%mod))%mod;
        }
        printf("%lld\n",ways);
    }
 return 0;
}
