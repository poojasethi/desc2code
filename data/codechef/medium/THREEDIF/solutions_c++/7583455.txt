#include <iostream>
#include <algorithm>

#define MODULO 1000000007

using namespace std;

unsigned long long mulmod(unsigned long long a, unsigned long long b) {
    unsigned long long result = 0;
    while(a != 0) {
        if(a & 1) {
            result = (result + b) % MODULO;
        }
        a >>= 1;
        b = (b << 1) % MODULO;
    }
    return result;
}

int main() {
    int t;
    int n;
    unsigned long long array[3];
    cin >> n;
    for(int i = 0; i < n; i ++) {
        cin >> array[0] >> array[1] >> array[2];
        sort(array, array + 3);
        cout << mulmod(array[0], mulmod(array[1] - 1, array[2] - 2)) << endl;
    }
    return 0;
}
