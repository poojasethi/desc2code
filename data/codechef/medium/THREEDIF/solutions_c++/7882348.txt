#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
#define M 1000000007
int main()
{
    int t; scanf("%d",&t);
    while(t--)
    {
    	long long arr[3]; int i;
    	for(i =0;i<3;i++) scanf("%lld",&arr[i]);
    	sort(arr,arr+3);
    	long long ans = (arr[0]%M)*((arr[1]-1)%M)%M*((arr[2]-2)%M)%M;
    	printf("%lld\n",ans);
    }
    return 0;
}
