#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define ll long long int
#define m 1000000007

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        ll arr[3],s;
        scanf("%lld",&arr[0]);
        scanf("%lld",&arr[1]);
        scanf("%lld",&arr[2]);
        sort(arr,arr+3);
        ll x = arr[0],y = arr[1],z = arr[2];
        s = ((((x%m) * ((y-1)%m))%m) * ((z-2)%m))%m;
        printf("%lld\n",s);
    }
    return 0;
}
