#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main()
{
	int t;
	scanf("%d", &t);
	while(t--){
		long long a[5];
		scanf("%lld %lld %lld", &a[0], &a[1], &a[2]);
		sort(a, a+3);
		long long res;
		res = (a[0]%mod) * ((a[1]-1)%mod) %mod * ((a[2]-2)%mod) %mod;
		printf("%lld\n", res%mod);
	}
	return 0;
}
