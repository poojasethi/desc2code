# include <bits/stdc++.h>
using namespace std;
# define mod 1000000007
typedef long long ll;

main () {
	int t;
	scanf("%d",&t);
	while (t--) {
		ll n[3];
		cin >> n[0] >> n[1] >> n[2];
		sort(n,n+3);
		ll x = (n[0]%mod);
		ll y = (n[1]-1+mod)%mod;
		ll z = (n[2]-2+mod)%mod;
		printf("%lld\n",(((x*y)%mod)*z)%mod);
	}	
	return 0;
}
