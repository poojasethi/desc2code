#include <iostream>
#include <algorithm>
using namespace std;


const int MOD = 1000000007;

const int K = 3;

int main() {

	int T;
	cin >> T;

	while(T--) {


		long long N[K];
		for (int i = 0; i < K; ++i) {
			cin >> N[i];
		}
		sort(N, N + K);

		int ans = 1;
		for (int i = 0; i < K; ++i) {

			ans = (N[i] - i) % MOD * ans % MOD;

		}
		cout << ans << endl;
	}
}
