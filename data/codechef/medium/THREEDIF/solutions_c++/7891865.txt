#include<stdio.h>
#include<algorithm>
#define mod 1000000007
using namespace std;
int main()
{
    int t,i;
    long long int a[3],ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld%lld%lld",&a[0],&a[1],&a[2]);
        sort(a,a+3);
        ans=(a[0]%mod) * ((a[1]-1)%mod) % mod * ((a[2]-2)%mod) % mod;
        printf("%lld\n",ans);
    }
    return 0;
}
