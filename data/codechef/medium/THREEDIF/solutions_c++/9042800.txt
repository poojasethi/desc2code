#include<cstdio>
#include<algorithm>
using namespace std;
int t;
long long a[3],rez;
int main(){
 scanf("%d",&t);
 for(int i=1;i<=t;i++){
  scanf("%lld %lld %lld",&a[0],&a[1],&a[2]);
  sort(a,a+3);
  rez=a[0]%1000000007;
  rez=rez*((a[1]-1)%1000000007)%1000000007;
  rez=rez*((a[2]-2)%1000000007)%1000000007;
  printf("%lld\n",rez);
 }
return 0;
}
