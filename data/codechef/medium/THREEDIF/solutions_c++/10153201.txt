#include <algorithm>
#include <stdio.h>
#define  mod	   1000000007
using	 namespace std;
typedef  long long LL;

int T;
LL n[3];

int main() {
	
	scanf( "%d" , &T );
	for( int tc = 1 ; tc <= T ; tc++ ) {
		for( int i = 0 ; i < 3 ; i++ )
			scanf( "%lld" , &n[i] );
		sort( n , n + 3 );
		LL ans = 1;
		for( int i = 0 ; i < 3 ; i++ )
			ans = ( ans * ( (n[i] - i + mod) % mod ) ) % mod;
		printf( "%lld\n" , ans );
	}
	
}
