#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	long long int product,a[3];
	scanf("%d",&t);
	while(t--)
	{
		product=0;
		scanf("%lld%lld%lld",&a[0],&a[1],&a[2]);
		sort(a,a+3);
		
		a[1]--;
		a[2]-=2;
		a[0]%=1000000007;
		a[1]%=1000000007;
		a[2]%=1000000007;
		product=(a[0]*a[1])%1000000007;
		product= (product*a[2])%1000000007;
		printf("%lld\n",product);
		
	}
	return 0;
}
