#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t,i;
    unsigned long long x[3],m=1000000007,ans=1;
    cin>>t;
    while(t--)
	{
		for(i=0;i<3;i++)
            cin>>x[i];
		sort(x,x+3);
		for(i=0;i<3;i++)
			x[i]=x[i]-i;
		ans=((((x[0]%m)*(x[1]%m))%m)*(x[2]%m))%m;
		cout<<ans<<endl;
	}
    return 0;
}
