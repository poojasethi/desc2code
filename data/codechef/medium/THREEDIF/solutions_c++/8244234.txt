	#include<iostream>
	#include<vector>
	#include<algorithm>
	#define mod 1000000007
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			vector<long long int > v;;
			long long int temp;
			
			for(int i=0;i<3;i++)
			{
				cin>>temp;
				v.push_back(temp);
			}	

			sort(v.begin(),v.end());
			long long int pro=1;
			for(int i=0;i<3;i++)
			{
				if(v[i]-i>0)
				{
					pro=(pro*((v[i]-i)%mod))%mod;
				}	
				else
					pro=0;
			}	
			cout<<pro<<endl;
			t--;
		}	
	}
