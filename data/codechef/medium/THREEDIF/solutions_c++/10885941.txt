#include<cstdio>
#include<algorithm>
#define MOD 1000000007
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        ll arr[3];
        for(int i=0;i<3;i++){
            scanf("%lld",&arr[i]);
        }
        sort(arr,arr+3);
        for(int i=0;i<3;i++){
            arr[i]=(arr[i]-i)%MOD;
        }
        ll ans=arr[0];
        for(int i=1;i<3;i++){
            ans=(ans*arr[i])%MOD;
        }
        printf("%lld\n",ans);
    }
}
