#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007

int main()
{
	ios_base::sync_with_stdio(false);
	int t; cin>>t;
	while(t--)
	{
      unsigned long long a[3];
      for(int i=0;i<3;i++)
      {
      	cin>>a[i];
      }
      sort(a,a+3);
      unsigned long long ans=1;
      for(int i=0;i<3;i++)
      {
      	ans*=(a[i]-i)%mod;
      	ans%=mod;
      }
     printf("%llu\n",ans);
    }
	return 0;
}
