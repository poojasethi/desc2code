#include <bits/stdc++.h>
using namespace std;

const int MOD = 1000000007;
const int k = 3;
int main() {
	int t;
	cin >> t;

	while(t--) {
		long long int n[k];
		int i;

		for(i = 0;i < k;i++) {
			cin >> n[i];
		}

		sort(n,n+k);
		int ans = 1;

		for(i = 0;i < k;i++) {
			ans = (n[i] - i) % MOD * ans % MOD;
		}
		cout << ans << "\n";
	}
	return 0;
}
