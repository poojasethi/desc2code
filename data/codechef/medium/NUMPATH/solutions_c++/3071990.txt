int b,f[N],dp[N],a[N];
void add(int x,int v)
{
	while(x>0)f[x]+=v,f[x]%=mod,x-=x&-x;
}
int query(int x)
{
	int ans=0;
	while(x<=b)ans+=f[x],ans%=mod,x+=x&-x;
	return ans;
}
int query(int a,int b)
{
	return query(a)-query(b+1);
}
int main()
{
	int i,j,ca=0,m,k,T,n;
	scanf("%d%d",&n,&b);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	memset(f,0,sizeof(f));
	scanf("%d",&m);
	dp[b]=1;add(b,1);
	for(i=b-1;i>=1;i--)
	{
		dp[i]=query(i+1,i+a[i]);
		if(dp[i]<0)dp[i]+=mod;
		add(i,dp[i]);
	}
	while(m--)
	{
		scanf("%d",&j);
		printf("%d\n",dp[j]);
	}
	return 0;
}
