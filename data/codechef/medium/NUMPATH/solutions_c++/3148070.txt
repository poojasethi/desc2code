#define MAXN 200011
#define MOD 1000000007

int N, B, Q;
int n[MAXN];
LL sum[MAXN], ans[MAXN];

int min(int a, int b)
{
    return a<b? a:b;
}

void solve()
{
    scanf("%d%d", &N, &B);
    for(int i=1; i<=N; i++)
    {
        scanf("%d", &n[i]);
    }
    int x, q;
    sum[B] =1, sum[B+1]=0;
    for(int i=B-1; i>=1; i--)
    {
       if(n[i] == 0)
       {
           ans[i] = 0;
       }
       else
       {
           x = min(i+n[i], B);
           ans[i] = sum[i+1] - sum[x+1];
           if(ans[i]<0LL)ans[i] += MOD;
       }
        sum[i] = sum[i+1] + ans[i];
        if(sum[i]>=MOD)sum[i] -= MOD;
    }
    scanf("%d", &Q);
    for(int i=0; i<Q; i++)
    {
        scanf("%d", &q);
        printf("%lld\n", ans[q]);
    }
}
