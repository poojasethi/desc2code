#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

const int mx = 200000+10;
const int mod = 1000000007;
int d[mx],f[mx],F[mx];
int main(){
//    freopen("data.in","r",stdin);
    int N,B,Q,u;
    cin>>N>>B;
    B--;
    for(int i=0;i<N;i++)
        cin>>d[i];    
    f[B] = 1;
    F[B] = 1;
    F[B+1] = 0;
    for(int i=B-1;i>=0;i--){
        int di = d[i];
        f[i] = F[i+1]-F[min(i+di+1,B+1)];
        f[i] = (f[i]+mod)%mod;
        F[i] = (F[i+1]+f[i])%mod;
    }
    cin>>Q;
    while(Q--){
        cin>>u;
        cout<<f[u-1]<<endl;
    }
    return 0;
}
