#include<iostream>
using namespace std;
#define ll long long int
int a[200010],n,b,x,ans[200010],q,s[200010];
#define M 1000000007
int main()
{
    cin>>n>>b;
    //cout<<n<<endl;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
    }
    s[b]=1,ans[b]=1;
    for(int i=b-1;i>=1;i--)
    {
        ans[i]=((s[i+1]-s[i+a[i]+1])%M+M)%M;
        s[i]=(ans[i]+s[i+1])%M;
    }
   // for(int i=1;i<=n;i++)cout<<ans[i]<<" ";
    //cout<<endl;
    cin>>q;
    while(q--)cin>>x,cout<<ans[x]<<endl;
}
