#include <iostream>
#include <cstdio>
#include <memory.h>

using namespace std;

const int md = 1000000007;
const int maxn = 200333;
int n, f[maxn], a[maxn], i, ans[maxn];

void add(int x, int d){
	while (x <= n){
		f[x] += d;
		if (f[x] >= md) f[x] -= md;
		x = x | (x + 1);
	}
}

int get(int x){
	int res = 0;
	while (x > 0){
		res += f[x];
		if (res >= md) res -= md;
		x = (x & (x + 1)) - 1;
	}
	return res;
}


int main(){
	int t;
	scanf("%d%d", &n, &t);
	for (i = 1; i <= n; i++)
		scanf("%d", &a[i]);
	add(t, 1);
	for (i = t - 1; i > 0; i--){
		int r = min(n, i + a[i]);
		ans[i] = get(r);
		add(i, ans[i]);
	}
	scanf("%d", &n);
	for (i = 1; i <= n; i++){
		int x;
		scanf("%d", &x);
		printf("%d\n", ans[x]);
	}
}
