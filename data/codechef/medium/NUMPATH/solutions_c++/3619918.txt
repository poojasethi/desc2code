#define mod 1000000007
#define ll long long
#define init ll i,j,k,l,m,n,test

#define inf 1000000000

long long disp[200009];
ll ans[200009];
int main()
{
	init;
	ll b,q;
	
	scan("%lld %lld",&n,&b);
	for(i=1;i<=n;i++)
		scan("%lld",&disp[i]);
	scan("%lld",&q);
	//compute all paths from i to b for all i
	//start from b-1
	disp[b+1]=0;
	disp[b]=1;

	for(i=b-1;i>0;i--)
	{
		j=disp[i];
		k=disp[i+1]-disp[min(j+i+1,b+1)];
		disp[i]=disp[i+1]+(k+mod)%mod;
		disp[i]%=mod;
		ans[i]=(k+mod)%mod;



	}
	for(i=0;i<q;i++)
	{
		scan("%lld",&j);
		printf("%lld\n",ans[j]);

	}

	return 0;
}
