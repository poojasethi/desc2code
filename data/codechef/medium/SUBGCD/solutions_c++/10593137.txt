#include <iostream>
#include <cstdio>
#include <algorithm>
#define gc getchar_unlocked
#define pc putchar_unlocked
using namespace std;
inline unsigned long int fastread()
    {
      unsigned long int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
    int gcd(int a, int b)
{
	if(b==0)
		return a;
	else
	{
		return gcd(b,a%b);
	}
}

 int main()
 {
     int t;
     t=fastread();//scanf("%d",&t);
     while(t--)
     {
          int n;
         n=fastread();//scanf("%d",&n);
         int num,temp;
         num=fastread();//scanf("%d",&num);
         temp=num;
         for(int i=1;i<n;i++)
            {
                num=fastread();//scanf("%d",&num);
                temp=gcd(temp,num);
            }
        if(temp==1)
            printf("%d\n",n);
            else
                printf("-1\n");
     }
     return 0;
 }
