#include<iostream>
#include<stdio.h>
using namespace std;
int gcd(int a,int b)
{
    if(a>=b)
        {
        if(a%b==0)
            return b;
        else
            return gcd(b,a%b);
        }
        else
            return gcd(b,a);
}
int main()
{
    int t,n,a[100000],m;
    cin>>t;
    while(t--)
    {
 
        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        m=a[0];
        for(int i=1;i<n;i++)
            m=gcd(m,a[i]);
            if(m==1)
                cout<<n<<"\n";
            else
                cout<<-1<<"\n";
    }
}
