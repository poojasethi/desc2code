#include <iostream>
using namespace std;

int gcd (int a, int b){
	int u;
	if(b==0)return a;
	u=a%b;
	a=b;
	b=u;
	
	return gcd(a, b);
}

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
		int n, i, x, g;
		cin>>n;
		cin>>g;
		for(i=1; i<n; i++){
			cin>>x;
			g=gcd(x, g);
		}
		if(g==1)cout<<n<<endl; else cout<<-1<<endl;
	}
	return 0;
}
