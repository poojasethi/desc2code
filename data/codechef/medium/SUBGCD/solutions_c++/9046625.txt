#include<stdio.h>
long int GCD(int a,int b)
{
	 if(b==0)
	   return a;
	   else
	   return GCD(b,a%b);
}

{

	while(t--)
	{
		long int n,a;
		scanf("%ld",&n);
		scanf("%ld",&a);
		long int gcd=a;
		int flag=0;
		for(long int i=1;i<n;i++)
		{
			scanf("%ld",&a);
			 gcd=GCD(gcd,a);
		}

          printf("%ld\n",n);

	}

}
