#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,i;
	cin>>t;
	while(t--)
	{
		int n,a[100005],k;
		int f=0;
		cin>>n;
		for(i=0;i<n;i++)
		{cin>>a[i];
		}
		k=a[0];
		for(i=0;i<n-1;i++)
		{
		k=	__gcd(k,a[i+1]);
		if(k==1)
		{f=i+2;}
		}
		if(f==0)
		{cout<<-1<<endl;}
		else
		cout<<f<<endl;
	}


}
