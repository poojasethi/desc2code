#include<iostream>
#include<stdio.h>
using namespace std;
long gcd(long a,long b)
 { long t=b;
    while(b>0)
   {
      b=a%b;
      a=t;
      t=b;
   }
  return a;
 }
long arr[100001];
int main()
{
  int t;
  long n,i;
  cin>>t;
   while(t--)
  {
     cin>>n;
      for(i=0;i<n;i++)
      {
        scanf("%ld",arr+i);
      }
     long hcf=arr[0];
      for(i=1;i<n;i++)
       {
          hcf=gcd(arr[i],hcf);
       }
       if(hcf==1)
         cout<<n<<endl;
       else
         cout<<-1<<endl;
  }
 return 0;
}
