#include <iostream>
#include <cstdio>

using namespace std;

int hcf(int a, int b)
{
    if(b > a)
    {
        int temp = a;
        a = b;
        b = temp;
    }

    if(b == 0)
    {
        return a;
    }
    else
    {
        return hcf(b, a%b);
    }

}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n, gcd;
        scanf("%d", &n);
        int arr[n];
        for(int i = 0; i < n; i++)
        {
            scanf("%d", &arr[i]);
        }

        gcd = arr[0];
        for(int i = 1; i < n; i++)
        {
            gcd = hcf(gcd, arr[i]);
        }

        if(gcd == 1)
        {
            printf("%d\n", n);
        }
        else
        {
            printf("-1\n");
        }
    }
    return 0;
}
