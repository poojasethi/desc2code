#include<bits/stdc++.h>
using namespace std;

int t,n,a,g;

int gcd(int a,int b){
	return((b==0)?a:gcd(b,a%b));
}

int main (){
	scanf("%d",&t);
	
	while(t--){
		scanf("%d",&n);
		scanf("%d",&g);
		
		for(int i=1;i<n;++i){
			scanf("%d",&a);
			g = gcd(a,g);
		}
		
		printf("%d\n",(g==1)?n:-1);
		
	}
	
	return 0;
}
