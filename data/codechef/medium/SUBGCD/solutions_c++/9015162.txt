#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <string>
#include <cstdlib>
#include <cstring>
#include <cstdio>
#include <cmath>
using namespace std;

int gcd(int a, int b)
{
	int c;
	while(a) {
		c = a;
		a = b % a;
		b = c;
	}
	return b;
}

int main(int argc, char const *argv[])
{
	int t;
	scanf("%d", &t);
	while(t--) {
		int n;
		scanf("%d", &n);

		int a, ans;
		for(int i = 0; i < n; i++) {
			scanf("%d", &a);
			if(i == 0)
				ans = a;
			else
				ans = gcd(ans, a);
		}

		if(ans == 1) {
			printf("%d\n", n);
		} else {
			printf("-1\n");
		}
	}
	return 0;
}
