#include<bits/stdc++.h>
using namespace std;
#define gc getchar_unlocked
int read_int()
{
	char c=gc();
 
	while(c<'0'||c>'9') 
		{
			c=gc();
		}
		int ret=0;
		while(c>='0' && c<='9')
		{
			ret=ret*10+c-48;
			c=gc();
		}
		return ret;
 
}
int main()
{
	int t,i;
	t=read_int();
	while(t--)
	{
		int n,a[100005],k;
		int f=0;
		n=read_int();
		for(i=0;i<n;i++)
		{cin>>a[i];
		}
		k=a[0];
		for(i=0;i<n-1;i++)
		{
		k=	__gcd(k,a[i+1]);
		if(k==1)
		{f=i+2;}
		}
		if(f==0)
		{cout<<-1<<endl;}
		else
		cout<<f<<endl;
	}


}
