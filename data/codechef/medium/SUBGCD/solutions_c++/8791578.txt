#include <iostream>
using namespace std;
#include<stdio.h>
inline int scan()
{
	int n=0,c=getchar_unlocked();
	 while(c<'0'||c>'9')
       c=getchar_unlocked();
     while(c>='0'&&c<='9')
       n=(n<<3)+(n<<1)+c-'0',c=getchar_unlocked();
    return n;
}
int main() 
{
	int t=scan();
	while(t--)
	{
		int n=scan();
		int a[n],i,cnt=0;
		for(i=0;i<n;i++)
		{
			a[i]=scan();
			if(a[i]%2==0)
			 cnt++;
		}
		printf("%d\n",cnt==n?-1:n);
	}
	
	return 0;
}
