#include <bits/stdc++.h>

using namespace std;

int gcd(int a,int b)
{
	if (b == 0)
		return a;
	else if (a == 0)
		return b;
	while (a%b != 0)
	{
		int c = a%b;
		a = b;
		b = c;
	}
	return b;
}
int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n;
		scanf("%d",&n);
		int a[n];
		for (int i = 0; i < n; i++)
			scanf("%d",&a[i]);
		int ans = gcd(a[0],a[1]);
		for (int i = 2; i < n; i++)
			ans = gcd(ans,a[i]);
		if (ans == 1)
			cout << n << endl;
		else
			cout << "-1" << endl;
	}
}
