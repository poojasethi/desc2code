#include<iostream>
using namespace std;
int gcd(int a,int b)
{
	int tmp;
	while(a%b)
	{
		tmp=a%b;
		a=b;
		b=tmp;
	}
	return b;
}
int main()
{
	int t,n,i,g;
	int a[100009];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		g=a[0];
		for(i=1;i<n;i++)
		{
			g=gcd(g,a[i]);
		}
		if(g!=1)
		printf("-1\n");
		else
		printf("%d\n",n);
	}
	return 0;
}
