#include<bits/stdc++.h>
using namespace std ;

typedef long long int ll ;

int main()
{
    int test ;
    scanf("%d",&test) ;
    while(test--) {
        int n ;
        scanf("%d",&n) ;
        int input[n] ;
        for(int i=0 ; i<n ; i++) {
            scanf("%d",&input[i]) ;
        }
        int res=input[0] ;
        for(int i=1 ; i<n ; i++) {
            res= __gcd(res,input[i]) ;
        }
        if(res==1) {
            printf("%d\n",n) ;
        }
        else {
            printf("-1\n") ;
        }
    }
}
