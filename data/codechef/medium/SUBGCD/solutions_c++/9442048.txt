#include<iostream>
#include<string.h>
#include<algorithm>
#include<cmath>
#include<stdio.h>
using namespace std;
long long int gcd(long long int a,long long int b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    long long int t,n,a[100005];
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        for(int i=0;i<n;i++)scanf("%d",&a[i]);
        long long int hcf=a[0];
        for(int i=1;i<n;i++)
        {
            hcf=gcd(hcf,a[i]);
        }
        //cout<<hcf<<" ";
        if(hcf==1)
            printf("%lld\n",n);
        else

            printf("-1\n");


    }
    return 0;
}
