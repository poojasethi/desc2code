#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,m,i,ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        m=n;
        n--;
        scanf("%d",&i);
        ans=i;
        while(n--)
        {
            scanf("%d",&i);
            ans=__gcd(ans,i);
        }

        if(ans==1)
            printf("%d\n",m);
        else
            printf("-1\n");
    }
    return 0;
}
