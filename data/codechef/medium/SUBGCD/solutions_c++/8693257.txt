#include <iostream>
#include <vector>
#include <algorithm>
#include <stdio.h>

using namespace std;

long gcd(long a, long b)
{
    if (a>b)
        return (b==0)?a:(gcd(b,a%b));
    else
        return (a==0)?b:(gcd(a,b%a));
}

int main()
{
    short int t;
    cin>>t;
    while (t--)
    {
        long n,a,GCD;
        scanf("%ld", &n);
        scanf("%ld", &GCD);
        for (long i=1;i<n;i++)
        {
            scanf("%ld", &a);
            GCD = gcd(a,GCD);
        }
        if (GCD==1)
            printf("%ld\n", n);
        else
            printf("-1\n");

    }

return 0;
}
