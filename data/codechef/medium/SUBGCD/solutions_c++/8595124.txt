#include<bits/stdc++.h>
using namespace std;
long long int gcd(long long int a,long long int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
int main()
{
    long long int t;
    scanf("%lld",&t);
    while(t--)
    {
        long long int n;
        scanf("%lld",&n);
        long long int i,A[n],res,flag=0;
        for(i=0;i<n;i++)
            scanf("%lld",&A[i]);
            res=A[0];
        for(i=1;i<n;i++)
        {
            res=gcd(A[i],res);
            if(res==1)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            printf("%lld\n",n);
        else
            printf("-1\n");
    }
return 0;
}
