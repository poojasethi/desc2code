#include<iostream>
#include<math.h>
using namespace std;

long gcd(long a, long b)
{
    // base case
    if (a == b)
        return a;

    // a is greater
    if (a > b)
        return gcd(a-b, b);
    return gcd(a, b-a);
}

int main()
{
    long t,n,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        long len=0;
        long ans=a[0];
        for(i=1;i<n;i++)
        {
            ans=gcd(ans,a[i]);
            if(ans==1)
                break;
        }
        if(ans==1)
            len=n;
        if(len==0)
            cout<<"-1"<<endl;
        else
            cout<<len<<endl;
    }
}
