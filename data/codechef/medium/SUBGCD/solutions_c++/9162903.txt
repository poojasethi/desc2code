#include <iostream>

using namespace std;

int gcd(int a , int b)
{
    if(b == 0)
        return a;
    else 
        return gcd(b , a%b);
}

int main()
{
    int t , n , i;
    
    cin >> t;
    
    while(t--)
    {
        cin >> n;
        int a[n];
        
        cin >> a[0];
        int hcf = a[0];
        
        for(i = 1 ; i < n ; ++i)
        {
            cin >> a[i];
            hcf = gcd(hcf , a[i]);
        }
        if(hcf == 1)
            cout << n  <<'\n';
        else 
            cout << -1 <<'\n';
    }
    return 0;
}
