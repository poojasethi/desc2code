#include <iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
int main()
{
    int t;
   scanf("%d",&t);

    while(t--)
    {
        int n;
        scanf("%d",&n);
        int g;
        scanf("%d",&g);
        for(int i=1;i<n;i++)
        {
           int b;
           scanf("%d",&b);
           g=gcd(g,b);
        }
           if(g==1)
            printf("%d\n",n);
           else
              printf("-1\n");

      }


  return 0;
}
