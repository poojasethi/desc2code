#include<iostream>
using namespace std;

int a[100005];

int gcd(int a, int b){
	if(a==0) return b;
	else return gcd(b%a, a);
}

int main(){
	int t;
	int n;
	cin>>t;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		int g=a[0];
		for(int i=1;i<n;i++){
			g=gcd(g, a[i]);
		}
		if(g==1) cout<<n<<endl;
		else cout<<"-1\n";
	}
}
