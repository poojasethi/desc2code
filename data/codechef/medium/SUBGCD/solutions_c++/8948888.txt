#include<iostream>
#include<algorithm>
using namespace std;
main()
{
long int n,t,z,ar[100001];
cin>>t;
for(long int z=0;z<t;z++)
{
cin>>n;
for(int i=0;i<n;i++)
cin>>ar[i];
sort(ar,ar+n);
long int a=0;
for(long int i=1;i<n;i++)
{
   if(ar[i]%ar[0]!=0)
    {
        a=1;
        break;
    }
}
if((a==0)&&(ar[0]!=1))
cout<<"-1"<<endl;
else
cout<<n<<endl;
}
}
