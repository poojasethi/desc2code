#include <stdio.h>
inline int read() {
    register int t=0;
    char c=getchar();
    while(c>'9' || c<'0')
        c=getchar();
    while(c<='9' && c>='0') {
    	t=(t<<1)+(t<<3)+c-'0';
    	c=getchar();
    }
    return t;
}

inline int gcd(int a, int b) {
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int main(void) {
    int t;
    t=read();
    while(t--) {
        int n,i,a[100000],ans;
        n=read();
        a[0]=read();
        ans=a[0];
        for(i=1;i<n;i++) {
            a[i]=read();
            if(ans!=1)
                ans=gcd(ans,a[i]);
        }
        if(ans>1)
            printf("-1\n");
        else {
            printf("%d\n",n);
        }
    }
	return 0;
}
