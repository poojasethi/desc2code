#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	int c=b%a;
	while(c)
	{
		b =  a;
		a= c;
		c = b%a;
	}return a;
}
int main()
{
     int t;
     scanf("%d",&t);
     while(t--)
     {
     	int n,i,j,b;
     	scanf("%d",&n);
     	int a[n];
     	for(i = 0; i < n; i++) {
		 scanf("%d",&a[i]);}
		 b= a[0];
		 for(i =1; i < n; i++)
		 {
		 	b = gcd(b,a[i]);
		 }
		 if(b!= 1) printf("-1\n");
		 else printf("%d\n",n);
     	
     }
    return 0;
}
