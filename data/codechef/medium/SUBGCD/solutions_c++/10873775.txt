#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long int ll;
int gcd(int a,int b){
    if(a<b){
        int tmp=a;
        a=b;
        b=tmp;
    }
    while(b){
        int tmp=a%b;
        a=b;
        b=tmp;
    }
    return a;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    int n;
    scanf("%d",&n);
    int arr[n];
    scanf("%d",&arr[0]);
    int gc=arr[0];
    for(int i=1;i<n;i++){
        scanf("%d",&arr[i]);
        gc=gcd(gc,arr[i]);
    }
    if(n==1 || gc!=1){
        printf("-1\n");
        continue;
    }
    printf("%d\n",n);
    }
}
