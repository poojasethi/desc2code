#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int t,f,n,g,i,a[100001];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        f=0;
        g=a[0];
        if(g==1)
            f=1;
        for(i=1;i<n;i++)
        {
            g=(__gcd(g,a[i]));
            if(g==1)
            {
                f=1;
                break;
            }
        }
        if(f)
            printf("%d\n",n);
        else
            printf("-1\n");
    }
    return 0;
}
