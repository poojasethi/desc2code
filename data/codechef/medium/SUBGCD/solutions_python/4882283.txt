import fractions
from sys import stdin
t=int(stdin.readline())
for i in xrange(t):
        n=int(stdin.readline())
        line=map(int,stdin.readline().split())
        c=0
        for j in range(1,n):
                if fractions.gcd(line[0],line[j])==1:
                                c=1
                                break
        if c<>1: print -1
        else: print n
