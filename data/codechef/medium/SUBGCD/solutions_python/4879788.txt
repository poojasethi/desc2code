def gcd(a, b):
    if(b==0): return a
    if(b>a): return gcd(b,a)
    return gcd(b, a%b)

t = int(raw_input())
for _ in xrange(t):
    n = int(raw_input())
    a = map(int, raw_input().split())
    g = gcd(a[0], a[1])
    if g==1:
        print n
        continue
    for i in xrange(2, n):
        g = gcd(g, a[i])
        if g==1: break
    if g==1:
        print n
    else:
        print -1
