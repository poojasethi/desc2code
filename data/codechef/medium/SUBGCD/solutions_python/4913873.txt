import sys,fractions

def main(*args,**kwargs) :
    for __ in range(input()) :
        n=int(raw_input())
        g=0
        lists=list(map(int,sys.stdin.readline().split()))
        for i in range(n) :
            g=fractions.gcd(g,lists[i])
        if g==1 :
            print n
        else :
            print "-1"

if __name__ == "__main__":
    main()
