using namespace std;

#define LL long long
#define mod 1000000007

bool iden(int x)
{
    int d = x%10;
    while(x)
    {
        if(d != (x%10))
            return 0;
        x /= 10;
    }
    return 1;
}

void solve()
{
    int h,m,c=0;

    scanf("%d%d",&h,&m);
    for(int i=0;i<h;i++)
        if(iden(i))
            for(int j=0;j<m;j++)
                if(i%10 == j%10 && iden(j))
                    c++;
    printf("%d\n",c);
}

int main()
{
    int t;

    scanf("%d",&t);
    while(t--)
        solve();
    return 0;
}
