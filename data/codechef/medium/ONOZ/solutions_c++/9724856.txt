#include<bits/stdc++.h>

using namespace std;

bool check(int h,int m)
{
	int digit,temp;
	digit = h%10;
	h = h/10;
	while(h>0)
	{
		temp =  h%10;
		h=h/10;
		if(temp != digit)
		return 0;
	}
	temp=m%10;
	if(temp!=digit)
	return 0;
	m=m/10;
	while(m>0)
	{
		temp = m%10;
		m=m/10;
		if(temp!=digit)
		return 0;
	}
	return 1;
}
int main()
{
	int t,h,m,cnt,i,j;
	cin>>t;
	while(t--)
	{
		cin>>h>>m;
		cnt = 0;
		for(i=0;i<h;i++)
		{
			for(j=0;j<m;j++)
			{
				if(check(i,j))
				{
				//cout<<i<<" "<<j<<"\n";
				cnt++;
				}
			}
		}
		cout<<cnt<<"\n";
	}
	return 0;
}
