#include <bits/stdc++.h>
using namespace std;

int get_len(int x){
    int ans = 0;
    while(x > 0){
        ans++;
        x/=10;
    }
    return ans;
}

int get_num(int dig, int times){
    int ans = 0, x = 1;
    for(int i = 1; i <= times; i++){
        ans+=dig*x;
        x*=10;
    }
    return ans;
}

int main(){
    int t;
    cin >> t;
    while(t--){
        int n1,n2;
        cin >> n1 >> n2;
        int len1 = get_len(n1),len2 = get_len(n2);
        int ans = 0;
        for(int i = 1; i <= len1; i++){
            for(int j = 1; j <= len2; j++){
                for(int k = 1; k <= 9; k++){
                    if(get_num(k,i) < n1 && get_num(k,j) < n2){
                        //cout << get_num(k,i) << ": " << get_num(k,j) << endl;
                        ans++;
                    }
                }
            }
        }
        cout << ans+1 << endl;
    }
}
