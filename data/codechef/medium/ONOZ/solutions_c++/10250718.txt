#include<iostream>
using namespace std;
int main(){
	int i,j,k,c,h,m,t,count,x,y;
	cin>>t;
	while(t--){
		cin>>h>>m;
		count=0;
		for(i=0;i<h;i++){
		x=i%10;
	    if(i/10>0&&x!=i/10)
	    continue;
	    else{
	    	for(j=0;j<m;j++){
	    	y=j%10;
	     if(j/10>0&&y!=j/10){
	     	continue;
		 }
		 else{
		 	if(x==y){	
		 	count++;
			 }	
		 	else
		 	continue;
		 }
		}
	}
}
cout<<count<<" \n";
}
	return 0;
}
