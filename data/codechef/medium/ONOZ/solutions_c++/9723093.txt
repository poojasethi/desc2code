#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
    int t,h,m,c=1,val,i;
    scanf("%d",&t);
    while(t--)
    {
        c=1;
        val=11;
        scanf("%d%d",&h,&m);
        for(i=1;i<=9;i++)
        {
            if(i<h && i<m )
                c++;
            if( i<h && val<m)
                c++;
            if( i<m && val<h)
                c++;
            if( val<m && val<h)
                c++;
            val=val+11;
        }
        cout<<c<<"\n";


    }
    return 0;

}
