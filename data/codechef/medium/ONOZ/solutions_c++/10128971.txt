#include <iostream>
using namespace std;

int mat(int x)
{   int d=x%10;
    while(x)
    {   if(d!=x%10)
            return 0;
        x/=10;
    }
    return 1;
}

int match(int x,int y)
{   if(x%10==y%10)
        return 1;
    return 0;
}

int main()
{   int t;
    cin>>t;
    while(t--)
    {   int i,j,a,b,s=0;
        cin>>a>>b;
        for(i=0;i<a;++i)
            for(j=0;j<b;j++)
                if(mat(i) && mat(j))
                    if(match(i,j))
                        s++;
        cout<<s<<"\n";
    }
    return 0;
}
