#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int h,m,i,j,total=0;
        cin>>h>>m;
        for(i=0;i<h;i++)
            for(j=0;j<m;j++)
                {
                    int a=i,b=j,digit=-1,flag=1,tempdigit;
                    if (i==0){
                                digit=0;
                             }
                    else {
                            digit=i%10;
                            i=i/10;

                            while(i!=0)
                            {
                                tempdigit=i%10;
                                i=i/10;
                                if (tempdigit!=digit)
                                    flag=0;

                            }
                         }
                    if (j==0&&digit!=0)
                        {
                            flag=0;
                        }
                    while(j!=0)
                        {
                            tempdigit=j%10;
                            j=j/10;
                            if (tempdigit!=digit)
                                    flag=0;
                        }
                    if (flag==1)
                        {total++;}
                    i=a;
                    j=b;
                }
                cout<<total<<'\n';
    }

    return 0;
}
