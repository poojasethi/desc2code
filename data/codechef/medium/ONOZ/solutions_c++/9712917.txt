#include <bits/stdc++.h>
using namespace std;

int main() {
	
	long long int h,m,t;
	cin>>t;
	
	int arr[100][100],matrix[10][10][10][10];
	
	while(t--)
	{
		long long int i,j,x=0;
		cin>>h>>m;
		
		memset(matrix,0,sizeof(matrix));
		
		for(i=0;i<h;i++)
		{
			int q=i/10;
			int r=i%10;
			
			 for(j=0;j<m;j++)
			 {
				
				int q1=j/10;
				int r1=j%10;
				
				//cout<<q<<" "<<r<<" "<<q1<<" "<<r1;
				
				if(q==0)
				{
			     if((q1==r1)&&(r1==r)||(q1==0&&r1==r))
				  matrix[q][r][q1][r1]=1;//cout<<"yes1";
				}
				else if(q==r)
				{
					if((q1==0&&r==r1)||(q1==r&&q1==r1))
				  matrix[q][r][q1][r1]=1;//cout<<"yes2";
				}
				
				
				//cout<<endl;
			}
		}
		
	//	cout<<k;
		
		
		for(int i=0;i<10;i++)
		{
			for(int j=0;j<10;j++)
			{
				for(int k=0;k<10;k++)
				{
					for(int l=0;l<10;l++)
					{
					  if(matrix[i][j][k][l]==1)
					   x++;
					}
				}
			}
		}
		
		cout<<x<<endl;
		
	}
	
	return 0;
}
