#include <bits/stdc++.h>
using namespace std;
int sg(long long x){
	if(x>0) return 1;
	else return -1;
}
int main(){
	ios::sync_with_stdio(0);
	int t;
	cin >> t;
	while(t--){
		int h,m,hh,mm,ans;
		cin >> h >> m;
		ans=0;
		for(hh=mm=0;hh<h && mm<m && hh<=9 && mm<=9; ++hh,++mm)
			ans++;
		for(hh=11,mm=1;hh<h && mm<m && hh<=99 && mm<=9; hh+=11,++mm)
			ans++;
		for(hh=11,mm=11;hh<h && mm<m && hh<=99 && mm<=99; hh+=11,mm+=11)
			ans++;
		for(hh=1,mm=11;hh<h && mm<m && hh<=9 && mm<=99; hh+=1,mm+=11)
			ans++;
		cout << ans << '\n';
	}
	return 0;
}
