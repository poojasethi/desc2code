#include<bits/stdc++.h>
using namespace std;
inline bool is(int x, int y){
	if(x == 0 && y != 0) return false;
	if(x != 0 && y == 0) return false;
	int dig = x % 10;
	while(x > 0){
		if(x % 10 != dig) return false;
		x /= 10;
	}
	while(y > 0){
		if(y % 10 != dig) return false;
		y /= 10;
	}
	return true;
}
int main(){
	int t;
	cin >> t;
	while(t--){
		int h, m;
		cin >> h >> m;
		int cnt = 0;
		for(int i = 0;i < h;++i){
			for(int j = 0;j < m;++j){
				if(is(i, j)){
					//cout << i << " " << j << "\n";
					++cnt;
				}
			}
		}
		cout << cnt << "\n";
	}


	return 0;
}
