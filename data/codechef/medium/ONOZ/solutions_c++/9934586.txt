#include<iostream>
#include<set>
#include<cstring>
#include<sstream>
using namespace std;

string toString(int x) {
	stringstream ss;
	ss << x;
	string str = ss.str();
	return str;
}

bool hasIdenticalDigits(int h, int m) {
	
	set<char> s;
	string k1 = toString(h);
	string k2 = toString(m);
	for (int i = 0; i < k1.length(); i++) s.insert(k1[i]) ;
	for (int i = 0; i < k2.length(); i++) s.insert(k2[i]) ;
	return s.size() == 1;
}


int main() {
	int t;
	cin >> t;
	while(t--) {
		int h,m;
		cin >> h >> m;
		int ans = 0;
		for (int i = 0; i < h; i++) {
			for (int j = 0; j < m; j++) {
				ans += hasIdenticalDigits(i, j);
			}	
		}
		cout << ans << "\n";
	}
	return 0;
	
}
