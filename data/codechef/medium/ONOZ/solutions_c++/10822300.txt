#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int t,m,h,temp,t1,t2;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&h,&m);
        if(h>m)
        {
            temp=m;
            m=h;
            h=temp;
        }
        t1=h/11;
        t2=m/11;
        if(h%11==0)
            t1--;
        if(m%11==0)
            t2--;
        if(m<=9)
            printf("%d\n",h);
        else if(h<=9&&h>t2)
            printf("%d\n",(t2*2)+h-t2);
        else if(h<=9)
            printf("%d\n",(h*2)-1);
        else
            printf("%d\n",((t1+t2)*2)+9-t2+1);
    }
    return 0;
}
