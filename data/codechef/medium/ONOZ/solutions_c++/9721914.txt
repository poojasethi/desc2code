#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int h,m,c=0;
		scanf("%d %d",&h,&m);
		c=1;
		for(int i=1;i<=9;i++)
		{
		//for minutes
			if(i < h && i < m)	c++;
			if(i < h && (i*10+i) < m)	c++;
			//for hours
			if((i*10+i) < h && i < m)	c++;
			if((i*10+i) < h && (i*10+i) < m) c++;
		}
	cout<<c<<"\n";
	}	
}
