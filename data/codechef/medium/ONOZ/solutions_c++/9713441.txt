using namespace std;
int main()
{ int t;
  cin>>t;
  while(t--)
  { int h,m,i,j,val,count=0,p;
    cin>>h>>m;
    for(i=1;i<h;i++)
    for(j=1;j<m;j++)
    { p=0;
	  val=i;
      while(val>0)
      { if(val%10!=i%10)
      	p=1;
      	val/=10;
	  }
	  
	  val=j;
      while(val>0)
      { if(val%10!=i%10)
      	p=1;
      	val/=10;
	  }
    	 
      if(p==0)
	  { //cout<<i<<":"<<j<<"\n";
	   ++count;	 
      }
	}
	
	 cout<<++count<<"\n";
  }
	 return 0;
}
