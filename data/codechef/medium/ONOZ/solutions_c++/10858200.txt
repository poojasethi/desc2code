#include <bits/stdc++.h>

using namespace std;

#define boost ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0)

typedef long long ll;

int main(){
	boost;
	ll T,i,j,H,M;
	ll count = 0;
	cin >> T;
	while(T--){
		count = 0;
		ll hcount,mcount;
		cin >> H >> M;
		count++;
		for(i = 1 ; i <= 9 ; i++){
			hcount = mcount = 0;
			if(i < H)
				hcount++;
			if((10*i+i)<H)
				hcount++;
			if(i < M)
				mcount++;
			if((10*i+i)<M)
				mcount++;
			count += (hcount * mcount);
		}
		cout << count << endl;
	}
}
