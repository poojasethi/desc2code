#define mod 1000000007
 
ll repeat(int a,int b)
{   
    ll x= 1 , y = a;
    while(b > 0)
    {
        if(b%2 == 1)
        { 
           x =( x * y ) % mod;
        } 
        y = (y * y) % mod;
        b /= 2;
    }
    return x;
}
int main()

		ll prev,ans;
		cin>>n>>k;
		prev=0;
		for(int i=2;i<=n;i++) {
			ans=(repeat(k,i-1)-prev+mod)%mod;
			prev=ans;
		}
		cout<<ans<<endl;
	}
