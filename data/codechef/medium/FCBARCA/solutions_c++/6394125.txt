#include<iostream>
using namespace std;
#define MOD 1000000007
long long int arr[11][1001];
int main()
{
    //int arr[11][1001];

   // long long int *arr=new long long int [11][1001];

    int i,j;

    for(i=1;i<=10;i++)
        {arr[i][1]=0;
         arr[i][0]=1;}

    for(i=2;i<=1000;i++)
    {
        if (i%2==0)
            arr[1][i]=1;
        else
            arr[1][i]=0;
    }

    for(i=2;i<=10;i++)
        for(j=2;j<=1000;j++)
        arr[i][j]=(((arr[i][j-1])*(i-1))%MOD+(arr[i][j-2]*i)%MOD)%MOD;

    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        cout<<arr[k][n]<<'\n';
    }
    return 0;
}
