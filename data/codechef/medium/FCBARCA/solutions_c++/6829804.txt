#include<cstdio>
#include<cstring>
using namespace std;
long long int A[1010][1010];
long long int n,m;
long long int rec(long long int p,long long int q)
{
    if(A[p][q]!=-1) return A[p][q];
    if(q==0){
        if(p==0) return 1;
        else return 0;
    }
    long long int temp=0;
    for(int i=0;i<=m;i++){
        if(i==p) continue;
        temp+=rec(i,q-1)%1000000007;
    }
    return A[p][q]=temp%1000000007;
}
int main()
{
    int TC;
    scanf("%d",&TC);
    while(TC--){
        scanf("%lld %lld",&n,&m);
        memset(A,-1,sizeof A);
        printf("%lld\n",rec(0,n));
    }
}
