#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int main()
{
    long long int t,n,k,i,g[1005];
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld %lld",&n,&k);
        g[0]=1;
        g[1]=0;
        for(i=2;i<=n;i++)
        {
            g[i]=(((k-1)*g[i-1])%mod + (k*g[i-2])%mod)%mod;
        }
        printf("%lld\n",g[n]);
    }
    return 0;
}
