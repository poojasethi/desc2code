#include<iostream>
#define ll long long int
#define mod 1000000007
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll i,j,k,l,m,n,dp[1111][3]={0};
        cin>>n>>k;
        dp[1][1]=k;
        dp[1][2]=0;
        for(i=2;i<=n;i++)
        {
            dp[i][1]=((k-1)*dp[i-1][1]+k*dp[i-1][2])%mod;
            dp[i][2]=dp[i-1][1];
        }
        cout<<dp[n][2]<<"\n";
    }
    return 0;
}
