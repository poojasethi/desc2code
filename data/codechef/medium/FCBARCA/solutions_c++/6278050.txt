#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define pf printf
#define sf scanf

int main()
{
    ll dp[1001][2],t,n,k,i,mod=1000000007;
    sf("%lld",&t);
    while(t--)
    {
        sf("%lld %lld",&n,&k);
        memset(dp,0,sizeof(dp));
        dp[0][0]=1;
        dp[1][0]=0;
        for(i=1;i<=n;i++)
        {
            dp[i][0]=((ll)k*dp[i-1][1])%mod;
            dp[i][1]=(dp[i-1][0]+(ll)dp[i-1][1]*(k-1))%mod;
        }
        pf("%lld\n",dp[n][0]);
    }
    return 0;
}
