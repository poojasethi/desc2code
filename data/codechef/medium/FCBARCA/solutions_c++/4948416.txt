#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long int dp[2000][2],i,j,ans,n,k,t;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(i=0;i<2000;i++)
            for(j=0;j<=1;j++)
                dp[i][j]=0;
        dp[0][0]=1;
        dp[0][1]=0;
        for(i=1;i<=n;i++)
        {
            dp[i][0]=(dp[i-1][1]*k)%1000000007;
            dp[i][1]=(dp[i-1][0]+dp[i-1][1]*(k-1))%1000000007;
        }
        cout<<dp[n][0]<<endl;
    }
    return 0;
}
