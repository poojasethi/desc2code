#include<cstdio>
using namespace std;
#define mod 1000000007;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k;
        scanf("%d %d",&n,&k);

        int nonmessi = 0;
        int nonmessip=0;
        int messi = 1;
        while(--n){
            nonmessip = nonmessi;
            nonmessi = ((long long)messi*k+ (long long)nonmessi*(k-1))%mod;
            messi = ((long long)nonmessip)%mod;
        }
        printf("%d\n",nonmessi);

    }
}
