#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std ;
long long int dp[1001][11] = {0};
long long int fun(int,int,int,int);
int main()
{
int t,n,k;
long long int count ;
scanf("%d",&t);
while(t--) {

	scanf("%d %d",&n,&k);
	long long int count = fun(k+1,1,n,k);
	count = count%(1000000007);
	printf("%lld\n",count);
	memset(dp,0,sizeof(dp));
	}
return 0;
}

long long int fun(int pastp,int passno , int n ,int k) 
{
long long int arr[12] = {0} ;
int i ;
if(passno == n) {
	if(pastp != k+1) return 1 ;
	else return 0;
	}
else {
	for(i=1;i<=k+1;i++) {
		if(i == pastp) continue ;
		else {
			if(dp[passno+1][i] == 0) {
				dp[passno+1][i] = fun(i,passno+1,n,k) ;
				arr[i] = dp[passno+1][i] ;
				}
			else arr[i] = dp[passno+1][i] ;
			}
		}
	long long int sum = 0;
	for(i=1;i<=11;i++) sum = (sum + arr[i])%(1000000007) ;
	return sum ;
	}
}
