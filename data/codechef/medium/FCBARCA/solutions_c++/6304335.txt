#include<bits/stdc++.h>

#define ll long long int
#define mod 1000000007

using namespace std;
long long f[1005];
ll pw(ll a, ll b,ll M){
  ll r;
  if(b==0) return 1;
  r = pw(a,b/2,M);
  r = (r*r)%M;
  if(b%2) r = (r*a)%M;
  return r;
}

int main()
{
	int t,n,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&k);
		f[1]=0;
		//f[2]=1;

		for(int i=2;i<=n;i++)
		{
			f[i]=(pw(k,i-1,mod)-f[i-1]%mod+mod)%mod;
		}
		printf("%lld\n",f[n]);

	}
}
