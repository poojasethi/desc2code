inline void inp( ll &n )
{
n=0;
ll ch=inchar();ll sign=1;
while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=inchar();}
while( ch >= '0' && ch <= '9' )
n = (n<<3)+(n<<1) + ch-'0', ch=inchar();
n=n*sign;
}

ll dp[1010][3];

int main() {

     ll test,n,k;
     inp(test);
     while(test--)

        inp(n); inp(k);
        dp[1][1] = 0; dp[1][2] = k;
