#include<iostream>
#include<algorithm>
#include<cstring>
#include<cmath>
#define N 12
#define P 1000000007
using namespace std;

int dp[1005][12];

int main(){
	int t,n,k,tot=0;
	cin>>t;
	while(t--){
		memset(dp,0,sizeof(dp));
		cin>>n>>k;
		k++;
		dp[1][0]=0;
		for(int i=1;i<k;i++)	dp[1][i]=1;
		for(int i=2;i<n;i++){
			for(int j=0;j<k;j++){
				for(int m=0;m<k;m++){
					if(j!=m)	dp[i][j]+=dp[i-1][m];
					dp[i][j]%=P;
				}
			}
		}
		int tot=0;
		for(int i=1;i<k;i++){
			tot+=dp[n-1][i];
			tot%=P;
		}
		cout<<tot<<endl;
	}
	return 0;
}
