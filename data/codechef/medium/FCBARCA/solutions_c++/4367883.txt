#include <iostream>
#include <cstdlib>
#include <cstdio>

#define LL long long int
#define sld(x) scanf("%d", &x)
#define sd(x) scanf("%lld", &x)
#define MOD 1000000007
#define MAXN 1000

using namespace std;

LL dp[MAXN + 2][11];

LL PowerMod(LL x, LL y){
    if(y == 0){
        return 1;
    }
    LL ans = PowerMod(x, y / 2);
    ans %= MOD;
    ans *= ans;
    ans %= MOD;
    if(y % 2){
        ans *= x;
        ans %= MOD;
    }
    return ans;
}

inline void Solve(){
    int n, k, i, j, index;
    cin>>n>>k;
    for(i = 0; i <= n; i++){
        for(j = 0; j <= 10; j++){
            dp[i][j] = 0;
        }
    }

    for(j = 1; j <= 10; j++){
        dp[1][j] = 1;
    }


    for(i = 2; i <= n; i++){
        for(j = 0; j <= k; j++){
            for(index = 0; index <= k; index++){
                if(j == index){
                    continue;
                }
                dp[i][j] += dp[i - 1][index];
                dp[i][j] %= MOD;
            }
        }
    }

    LL ans = dp[n][0];
    if(ans < 0){
        ans %= MOD;
    }
    cout<<ans<<endl;
    return;
}

int main(){
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
    #endif
    LL t;
    sd(t);
    while(t--){
        Solve();
    }
    return 0;
}
