#include<cstdio>
const int N=1000,MOD=1000000007;
class DP{
    public:
        long long messi,other;
};
DP dp[N+1];
int t,n,k;
void files(){
    freopen("std.in","r",stdin);
    freopen("std.out","w",stdout);
}
void scan(){
    scanf("%d",&t);
}
void init(){
    scan();
}
void scanTest(){
    scanf("%d%d",&n,&k);
}
void initTest(){
    scanTest();
}
void solveTest(){
    int i;
    dp[0].messi=1;
    for(i=1;i<=n;i++){
        dp[i].messi=dp[i-1].other;
        dp[i].other=((dp[i-1].messi*k)%MOD+(dp[i-1].other*(k-1))%MOD)%MOD;
    }
    printf("%I64d\n",dp[n].messi);
}
int main(){
    //files();
    init();
    while(t!=0){
        t--;
        initTest();
        solveTest();
    }
    return 0;
}
