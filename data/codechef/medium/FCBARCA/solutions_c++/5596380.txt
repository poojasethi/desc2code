#include <iostream>
#include<stdio.h>
#include<malloc.h>
#include<string>

#define MOD 1000000007

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    int n,k;
    cin>>n>>k;

    long long dp[1005]={0};
    long long mdp[1005]={0};

    dp[1]=k;
    mdp[1]=0;

    for(int i=2;i<=n;++i)
    {
    dp[i]=((dp[i-1]*(k-1))%MOD+(mdp[i-1]*k)%MOD)%MOD;
    mdp[i]=dp[i-1]%MOD;
    }

    cout<<mdp[n]<<endl;

    }
    return 0;

}
