#include<bits/stdc++.h>
using namespace std;
int n,k;
#define mod 1000000007
int ans(int pass,int player)
{
    int i;
    if(pass==n)
        return 1;
    int sum=0;
    if(player==100)
    {
        for(i=1;i<=k;i++)
            sum+=ans(pass+1,i);
    }
    else
    {
        for(i=1;i<=k;i++)
        {
            if(player==i)
                continue;
            else
                sum+=ans(pass+1,i);
        }
        if(pass!=n-1)
            sum+=ans(pass+1,100);
    }
    return sum;
}
int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        long long int dp[n+1];
        dp[0]=1;
        dp[1]=0;
        for(i=2;i<=n;i++)
        {
            dp[i]=((long long int)((k-1)*dp[i-1])%mod+(long long int )(k*dp[i-2])%mod)%mod;
        }
        cout<<dp[n]%mod<<endl;
    }
    return 0;
}
