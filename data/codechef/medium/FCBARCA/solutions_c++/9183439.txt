#define N 1000000007

int main(){

	int t; cin>>t;

	while(t--){

		int n,k; cin>>n>>k;

		int dp[n+1][k+1];

		memset(dp,0,sizeof(dp));

		for(int i=1;i<k+1;i++) dp[1][i]=1;

		for(int i=2;i<=n;i++){

			for(int j=0;j<k+1;j++){

				for(int l=0;l<k+1;l++){

					if(j!=l) dp[i][j]=(dp[i][j]+dp[i-1][l])%N;
				}
			}
		}

		cout<<dp[n][0]<<"\n";
	}
}
