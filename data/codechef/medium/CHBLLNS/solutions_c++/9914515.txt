/*Chef and Ballons*/
 
#include<stdio.h>
 
int main()
{
	int B, G, K, R, T;
	long long count;
	scanf("%d", &T);
	while (T--)
	{
		scanf("%d %d %d", &R, &G, &B);
		scanf("%d", &K);
		count = 1;
		count += (R < K) ? R : (K - 1);
		count += (G < K) ? G : (K - 1);
		count += (B < K) ? B : (K - 1);
		printf("%lld\n", count);
	}
	return 0;
}
