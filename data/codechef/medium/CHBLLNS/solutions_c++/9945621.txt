#include<cstdio>
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;
int t;
long long int r,g,b,k,resenje;
int main(){
    scanf("%d",&t);
    for(int i=1;i<=t;i++){
        scanf("%lld %lld %lld",&r,&g,&b);
        scanf("%lld",&k);
        resenje=0;
        if(r<=k-1) resenje=resenje+r;
        else resenje=resenje+k-1;
        if(b<=k-1) resenje=resenje+b;
        else resenje=resenje+k-1;
        if(g<=k-1) resenje=resenje+g;
        else resenje=resenje+k-1;
        printf("%lld\n",resenje+1);
    }
return 0;
}
