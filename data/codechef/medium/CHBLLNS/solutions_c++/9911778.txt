#include <cstdio>

#include <cmath>

#include <cstring>

#include <iostream>

#include <algorithm>

#include <queue>

#include <vector>

#include <map>



using namespace std;



typedef long long LL;


int T;
LL a,b,c,k;
LL ans;

int main()

{
	scanf("%d",&T);
	while (T--)
	{
		ans=0;
		cin>>a>>b>>c;
		cin>>k;
		if (a<k) ans+=a;else ans+=k-1;
		if (b<k) ans+=b;else ans+=k-1;
		if (c<k) ans+=c;else ans+=k-1;
		ans++;
		cout<<ans<<endl;
	}

	return 0;

}
