#include <stdio.h>
int main(void)
{
	int i,t;
	unsigned int R, G, B,K,n;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		R = G = B = n = 0;
		scanf("%d",&R);
		scanf("%d",&G);
		scanf("%d",&B);
		scanf("%d",&K);
		if ( K <= R )
			n += (K-1);
		else
			n += R;
		if ( K <= G )
			n += (K-1);
		else
			n += G;
		if ( K <= B )
			n += (K-1);
		else
			n += B;
		n++;
		//        printf("%d\n",n);
		if ( K <= R && K <= G && K <= B )
			printf("%u\n",3*(K-1)+1);
		else
			printf("%u\n",n);
	}
	return 0;
 
}
