#include <iostream>

int main()
{long long int t,k,r,g,b,min;
 scanf("%lld",&t);
 while(t--)
 {min=0;
  scanf("%lld %lld %lld",&r,&g,&b);
  scanf("%lld",&k);
  if(r<=(k-1))
  min+=r;
  else
  min+=(k-1);
  if(g<=(k-1))
  min+=g;
  else
  min+=(k-1);
  if(b<=(k-1))
  min+=b;
  else
  min+=(k-1);
  
  min+=1;
  printf("%lld\n",min);
 }
 return 0;
}
