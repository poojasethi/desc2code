#include<bits/stdc++.h>
using namespace std;
#define ll long long

int main()
{
ll t,k,index,r,g,b,a[4],i,sum;
cin>>t;
while(t--)
{
    cin>>r>>g>>b>>k;
    sum=0;
    a[0]=r;
    a[1]=g;
    a[2]=b;
    sort(a,a+3);
    if(a[0]>=k)
        index=0;
    else if(a[1]>=k)
        index=1;
    else
        index=2;
    for(i=0;i<index;i++)
        sum+=a[i];
    sum+=(k-1)*(2-index+1);
    sum++;
    cout<<sum<<"\n";
}

return 0;
}
