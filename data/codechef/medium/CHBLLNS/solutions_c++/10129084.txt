#include <iostream>
using namespace std;
#define ll long long

ll minim(ll a,ll b)
{   if(a<b)
        return a;
    return b;
}

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll a,b,c,k;
        cin>>a>>b>>c>>k;
        cout<<(minim(a,k-1)+minim(b,k-1)+minim(c,k-1)+1)<<"\n";
    }
    return 0;
}
