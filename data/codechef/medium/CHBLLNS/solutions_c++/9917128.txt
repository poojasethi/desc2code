int main()
{
	int T;
	scanf("%d", &T);

	while(T--) {
		long long Arr[3];
		long long K;
		scanf("%lld%lld%lld%lld", &Arr[0], &Arr[1], &Arr[2], &K);
		sort(Arr, Arr + 3);
		if(Arr[0] >= K) {
			printf("%lld\n", (3 * (K - 1)) + 1);
		} else {
			long long Temp = Arr[0];
			long long Ans = (Temp * 3);
			K -= Temp;
			for(int i = 0; i < 3; ++i) Arr[i] -= Temp;
			if(Arr[1] >= K) {
				Ans += (2 * (K - 1))+1;
			} else {
				Ans += (2 * Arr[1]) + (K - Arr[1]);
			}
			printf("%lld\n", Ans);
		}
	}
    return 0;
}
