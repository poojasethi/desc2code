#include<iostream>
#include<stdlib.h>
using namespace std;

int compare(const void* a,const void* b)
{
    return(*(int*)a-*(int*)b);
}
void fun(int a[],int k)
{
    qsort(a,3,sizeof(int),compare);
    long long int count=0;
    for(int i=0;i<3;i++)
    {
        if(a[i]>=k)
        {
            count+=k-1;
        }
        else if(a[i]<k)
        {
            count+=a[i];
        }
    }
    cout<<count+1<<endl;
}
int main()
{
    int t;
    cin>>t;
    int a[3];
    for(int i=0;i<t;i++)
    {
      for(int j=0;j<3;j++)
        cin>>a[j];
    int k;
    cin>>k;
    fun(a,k);
    }
    return 0;


}
