#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    ll r,g,b,ans,k,t;
    cin>>t;
    while(t--)
    {
        cin>>r>>g>>b;
        cin>>k;
        if(max(max(r,g),b)==r)
        {
            ans=min(k-1,g)+min(k-1,b)+k;
        }
        else if(max(max(r,g),b)==g)
        {
            ans=min(k-1,r)+min(k-1,b)+k;
        }
        else
        {
            ans=min(k-1,g)+min(k-1,r)+k;
        }
        cout<<ans<<"\n";
    }
    return 0;
}
