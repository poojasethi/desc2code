//Driver program->
int main()
{
	int t;
	sf(t);
	while(t--)
	{
		ll r,g,b;
		sfl(r);sfl(g);sfl(b);
		
		ll k;
		sfl(k);
		ll ans=min(r,k-1)+min(g,k-1)+min(b,k-1)+1;
		pfl(ans);pn;
	}
	return 0;
}
