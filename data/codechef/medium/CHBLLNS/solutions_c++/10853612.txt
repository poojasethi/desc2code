#include<bits/stdc++.h>
#include <cassert>

using namespace std;
typedef long long int ll;
int main() {
    ios::sync_with_stdio(0);
    int t;
    cin >> t;
    while (t --) 
	{
    	ll r,g,b;
		cin>>r>>g>>b;
		ll k;
		cin>>k;
		ll ans = min(r,k-1) + min(g,k-1) + min(b,k-1)+1;
		cout<<ans<<endl;
    }
    return 0;
}
