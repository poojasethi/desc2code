#include <iostream>
#include <algorithm>
#include <cmath>
#include <string>
#include <cstring>
#include <cstdio>
#include <map>
using namespace std;
#define maxn 200010
#define mod 1000000007
typedef long long LL;
LL Pow(LL a,LL b){
    LL ans=1;
    while(b){
        if(b&1) b--,ans=(ans*a)%mod;
        else b>>=1,a=(a*a)%mod;
    }
    return ans;
}
int dp[110][210];
int main(){
    int T,n,x;
    scanf("%d",&T);
    while(T--){
        scanf("%d",&n);
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++){
            scanf("%d",&x);
            for(int j=-100;j<=100;j++){
                int tmp=x+j;
                if(tmp>=0 && tmp<=100){
                    dp[x][j+100]+=dp[tmp][j+100];
                    dp[x][j+100]%=mod;
                }
                dp[x][j+100]++;
            }
        }
        LL ans=0;
        for(int i=1;i<=100;i++){
            for(int j=-100;j<=100;j++){
                ans+=dp[i][j+100];
                ans%=mod;
            }
        }
        ans=Pow(2,n)-1-ans+200*n;
        ans%=mod;
        if(ans<0) ans+=mod;
        printf("%I64d\n",ans);
    }
    return 0;
}
