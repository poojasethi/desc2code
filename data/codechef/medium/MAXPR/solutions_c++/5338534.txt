#include<bits/stdc++.h>
using namespace std;
#define getcx getchar
#define maxn 200005
#define F(i,p,n) for(i=p;i<n;i++)
#define I(i,p,q) for(i=p;i>=q;i--)
#define Ss(x) scanf("%s",x)
#define P(x) printf("%d\n",x)
#define Ps(x) printf("%d ",x)
#define LL long long int
int mod=1e9+7;
#define pii pair<int,int>
inline void S(int &n)
{
    n=0; int ch = getcx(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getcx(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getcx(); }
    n = n * sign;
}
int cal[maxn];
void pre()
{
    int i,j;
    cal[0]=1;
    F(i,1,maxn)
    {
        cal[i]=(2*cal[i-1]);
        if(cal[i]>=mod)
            cal[i]-=mod;
    }
}
int main()
{
    int i,j,k,m,n,s[105],dp[205][105],p,q,t;
    pre();
    S(t);
    while(t--)
    {
        int ans=0;
        memset(dp,0,sizeof(dp));
        memset(s,0,sizeof(s));
        S(n);
        F(i,0,n)
        {
            S(p);
            F(j,0,101)
            {
                if(s[j])
                {
                    dp[p-j+100][p]=(dp[p-j+100][p]+dp[p-j+100][j]+s[j])%mod;
                    //printf("dp[%d][%d]=%d\n",p-j+100,p,dp[p-j+100][p]);
                }
            }
            s[p]++;
        }
        F(i,0,101)
        {
            F(j,0,201)
            {
                ans=(ans+dp[j][i]);
                if(ans>=mod)
                    ans-=mod;
            }
        }
       // printf("ans=%d\n",ans);
        ans=((cal[n]-(ans+1+n)%mod)+mod)%mod;
        P(ans);
    }
    return 0;
}
