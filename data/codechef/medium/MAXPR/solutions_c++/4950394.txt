#include <iostream>
#include <cstdio>
using namespace std;
int main(){
static long long n,m,i,j,k,t,te, arr[101][101],brr[101][101],pow[200002];
long long ans=0,mod=1000000007;
pow[0]=1;
for(i=1;i<200002;i++)pow[i]=(2*pow[i-1])%mod;
cin>>t;
for(te=0;te<t;te++){
ans=0;
for(i=0;i<101;i++)
for(j=0;j<101;j++)arr[i][j]=brr[i][j]=0;
cin>>n;
for(i=0;i<n;i++){
scanf("%lld",&j);
arr[j][0]++;
for(k=1;k<101&&k<j;k++)arr[j][k]+=(arr[j-k][k]+arr[j-k][0])%mod;
for(k=1;(j+k)<101;k++)brr[j][k]+=(arr[j+k][0]+brr[j+k][k])%mod;
}
for(i=1;i<101;i++)ans=(ans+pow[arr[i][0]]-1)%mod;
for(i=1;i<101;i++){
for(j=0;j<101;j++)ans=(ans+arr[j][i]+brr[j][i])%mod;
}
i=pow[n]-1;
cout<<(i+mod-ans)%mod<<"\n";
}
return 0;
}
