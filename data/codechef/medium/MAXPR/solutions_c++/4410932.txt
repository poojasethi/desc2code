#include <cstdio>
#include <cstdlib>

#define mod 1000000007 

int main(void)
{
	int T,n,p;
	scanf("%d",&T);
	long long b[101];
	long long A[101][200];
	long long ser,ans;
	for(int i =0;i<T;i++)
	{
		scanf("%d",&n); 
		for(int t=1;t<101;t++) b[t] = 0;
		
		for(int t=0;t<101;t++)
		{
			for(int j=1;j<=199;j++)
			{
				A[t][j] = 0;
			}
		}
		
		for(int t=0;t<n;t++)
		{
			scanf("%d",&p);
			for(int k=1;k<=100;k++)
			{
			
				A[p][p-k+100]+= A[k][p-k+100];
				A[p][p-k+100]+= b[k];
				A[p][p-k+100]%= mod;
			}
			b[p]++;
	    }
		ser =0;
		
		for(int t=1;t<=100;t++)
		{
			for(int q=1;q<=199;q++)
			{
				ser+=A[t][q];
				ser%= mod ;
			}
		}
		
	    ans =1;
		for (int t=0;t<n;t++){
			ans *= 2;
			ans %= mod;
		}
	
	    ans = ans- ser-n-1;
	    while(ans<0)
	    {
	    	ans = ans+mod;
	    }
	printf("%lld\n",ans);
	}
}
