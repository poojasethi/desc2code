#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#define lli int
#define gc getchar_unlocked
#define MOD 1000000007
using namespace std;
 
lli add(lli &x, lli y)
{   x += y;
    if (x >= MOD)
        x -= MOD;
    return x;
}
 
void scaninp(lli &a)
{
	register int b;
	a = 0;
	
	b = gc();
	
	for (; ((b < 48) || (b > 57)); b = gc());
	for (; ((b > 47) && (b < 58)); b = gc()) {
		a = (a << 1) + (a << 3) + b-48;
	}
}
