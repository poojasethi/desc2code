#define mod 1000000007

using namespace std;
typedef long long int LL;
typedef int I;
typedef char C;
 
LL power(LL a,LL b)
{   LL ans=1,c=mod;
	if(a==0)
		return 0;
	if(b==0)
		return 1;
	a=a%c;
	while(b>0)
	{ if(b%2==1)
		{ ans=(ans*a)%c;
		}
		b=b>>1;
		a=(a*a)%c;
	}
	return ans%c;
}
vector< vector<LL> > dp;
vector<LL> cu_dp;
int main()
{  LL T,i,N,ans,j,x,y;
//	cin>>T;
	scanf("%lld",&T);
	while(T--)
	{ 	//cin>>N;
		scanf("%lld",&N);
		dp.clear(); 
		cu_dp.clear();
		dp.resize(105); 
		cu_dp.resize(105,0);
		for(i=0;i<105;i=i+1)
		{ dp[i].resize(210,0);
		}
		ans=N+1;
		for(i=0;i<N;i=i+1)
		{ //cin>>x;
			scanf("%lld",&x);
			for(j=1;j<=100;j=j+1)
			{ 	if(cu_dp[j]==0)
				{ continue;
				}
				else
				{ y=100-(x-j);
				  dp[x][y]+=dp[j][y]+cu_dp[j];
				  dp[x][y]%=mod;
				}
			}
			cu_dp[x]+=1;
		}
		for(i=1;i<=100;i=i+1)
		{ for(j=1;j<200;j=j+1)
			{ ans+=dp[i][j];
				ans%=mod;
			}
		}
		ans=power(2,N)-ans;
		ans%=mod;
		if(ans<0)
		{ ans=(ans+mod)%mod;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
