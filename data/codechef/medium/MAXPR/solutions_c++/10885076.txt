#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long int P(int n)
{
    long long int ans = 1;
    if(n == 0)
        return 1;
    if(n == 1)
        return 2;
    ans  = P(n/2);
    if(n%2 == 0)
    {
        return (ans*ans)%mod;
    }
    else return (((ans*ans)%mod)*2)%mod;
}
int main()
{
    long int a[200002]  , i , j , T  , n;
    long int dp[200002] , d , curr , ans;
    scanf("%ld",&T);
    while(T--)
    {
        scanf("%ld",&n);
        for(i=1;i<=n;i++)
            scanf("%ld",&a[i]);
        ans = curr = 0;
        for(d=-100;d<=100;d++)
        {
            long int sum[102] = {0};
            curr = 0;
            for(i=1;i<=n;i++)
            {
                dp[i] = 1;
                if(a[i] - d >= 1 && a[i] - d <= 100)
                dp[i] = (sum[a[i] - d] + 1)%mod;
                sum[a[i]] = (sum[a[i]] + dp[i])%mod;
                curr = (curr + dp[i])%mod;
            }
            curr = (curr - n + mod)%mod;
            ans = (ans + curr)%mod;
        }
        ans = (ans + n)%mod;
        printf("%ld\n",(P(n) - ans - 1 + mod)%mod );
    }
    return 0;
}
