#include<bits/stdc++.h>
#define ll long long int
using namespace std;

const int mod=1000000007;
ll pow_(ll a, ll b)
{
    ll ans = 1;
    while(b)
    {
        if(b&1) ans = ans * a % mod;
        b >>= 1;
        a = a * a % mod;
    }
    return ans;
}
// dp[i][j]=no. of subsequence ending at number i and having common diff j
void solve()
{
      int N, x, dp[101][201];
    scanf("%d",&N);
    memset(dp, false, sizeof dp);
    for(int i=0; i<N; ++i)
    {
        scanf("%d",&x);
        for(int d=-100; d<=100; ++d)
        {
            if(x-d>=0 && x-d<=100) dp[x][d+100] += dp[x-d][d+100];
            if(dp[x][d+100]>=mod) dp[x][d+100] -= mod;
            dp[x][d+100] += 1;
        }
    }

    int ans = 0;
    for(int i=1; i<101; ++i)
        for(int j=0; j<201; ++j)
        {
            ans += dp[i][j];
            if(ans >= mod) ans -= mod;
        }
    ans = ans - N * 200 + 1;
    if(ans < 0) ans += mod;
    ans = pow_(2,N) - ans;
    if(ans < 0) ans += mod;
    printf("%d\n",ans);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
       solve();
    }
    return 0;
}
