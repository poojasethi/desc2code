const int N = 1000005;
const int M = 1000005;
const int inf = 1 << 28;
const int mod = 1e9 + 7;
const double eps = 1e-8;
using namespace std;
int n;
int dp[105][215];

int main() {
    int T;
    cin >> T;
    while (T --) {
        scanf("%d", &n);
        memset(dp, 0, sizeof(dp));
        int x, tot = 1;
        for (int i = 1; i <= n; i++) {
            tot <<= 1;
            tot %= mod;
            scanf("%d", &x);
            for (int j = -100; j <= 100; j++) {
                if (x + j <= 0 || x + j > 100) continue;
                dp[x][j + 101] += dp[x + j][j + 101];
                dp[x][j + 101] %= mod;
                dp[x][j + 101] += dp[x + j][204];
                dp[x][j + 101] %= mod;
            }
            dp[x][204] ++;
            dp[x][204] %= mod;
        }

        int res = 0;
        for (int i = 1; i <= 100; i++) {
            res += dp[i][204];
            res %= mod;
            for (int j = -100; j <= 101; j++) {
                res += dp[i][j + 101];
                res %= mod;
            }
        }

        res = (res + 1) % mod;
        res = (tot - res + mod) % mod;
        cout << res << endl;
    }
    return 0;
}
