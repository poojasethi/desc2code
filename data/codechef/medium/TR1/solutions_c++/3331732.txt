#include <iostream>
#include <cstdio>
#include <vector>
#define MOD 1000000007
using namespace std;
vector<vector<int> > vec;
long long child[10002];
long long lm[10002];

 void dfs(int vertex,int from)
  {
    long long total=0;
    int len=vec[vertex].size();
    for(int i=0;i<len;i++)
    {
       int temp=vec[vertex][i];
       if(temp==from)
        continue;
         dfs(temp,vertex);
         total+=child[temp];
    }
      child[vertex]=total+1;
      lm[vertex]=child[vertex];
      for(int i=0;i<len;i++)
      {
        int temp=vec[vertex][i];
        if(temp==from)
            continue;
         lm[vertex]+=((child[vertex]-child[temp])*child[temp]);
      }
  }
int main()
{
    int t,n,a,b;
    scanf("%d",&t);
    while(t--)
    {
      long long res=0;
      scanf("%d",&n);
      vec.clear();
      vec.resize(n+1);
      for(int i=1;i<=n;i++)
      {
          child[i]=0;
          lm[i]=0;
      }
      for(int i=1;i<n;i++)
      {
          scanf("%d%d",&a,&b);
          vec[a].push_back(b);
          vec[b].push_back(a);
      }
       dfs(1,0);
       for(int i=1;i<=n;i++)
        {
            res=(res+((i*lm[i])%MOD))%MOD;
        }
        printf("%lld\n",res);
    }
    return 0;
}
