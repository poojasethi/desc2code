#include <iostream>
#include <vector>

#define VVI vector< vector<int> >
using namespace std;
 
 
bool vis[10001];
vector< vector<int> > T(10001,vector<int>());
unsigned long long val;
unsigned long long mod = 1000000007ULL;
 
unsigned long long traverse(int n)
{
	vis[n] = 1;	
	unsigned long long cnt=1;
 
	for(int i=0 ; i<T[n].size() ; i++)
		if(!vis[T[n][i]])
		{
			unsigned long long tmp = traverse(T[n][i]);
			val += (2*n*(cnt-1)*tmp);
			val %= mod;
			cnt += tmp;
		}
 
	val += ((2*cnt-1) % mod * n);
	val %= mod;
	return cnt;
}
 
int main()
{
	int te;
	cin >> te;
	while(te--)
	{
		int n;
		cin >> n; 
 
		for(int i=1 ; i<=n ; i++) T[i].clear();
 
		for(int i=1 ; i<=n-1 ; i++)
		{
			int s,d;
			cin >> s >> d;
			T[s].push_back(d);
			T[d].push_back(s);
			vis[i] = 0;
		}
		vis[n] = 0;
 
		val=0;
		traverse(1);
		cout << val << endl;
	}
}
