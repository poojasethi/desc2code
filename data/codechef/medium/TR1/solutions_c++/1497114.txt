#include <vector>
#include <cstdio>
#define LL long long
using namespace std;
int N,tmp1,tmp2;
vector<int> adj_list[10005]; 
LL Im[10005];
LL kids[10005];
void make(int vertex,int from)
{
    int totalkids=0;
	int size=(int)adj_list[vertex].size();
	for(int i=0;i<size;i++)
	{
		int tmp=adj_list[vertex][i];
		if(tmp==from)continue;
		make(tmp,vertex);
		totalkids+=kids[tmp];
	}
	kids[vertex]=totalkids+1;
	Im[vertex]=kids[vertex];
	for(int i=0;i<size;i++)
	{
		int tmp=adj_list[vertex][i];
		if(tmp==from)continue;
		Im[vertex]+=kids[tmp]*(kids[vertex]-kids[tmp]);
	}
}
void init()
{
	for(int i=1;i<=N;i++)
	{
		adj_list[i].clear();
		Im[i]=0;
		kids[i]=0;
	}
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&N);
		init();
		for(int i=0;i<N-1;i++)
		{
			scanf("%d",&tmp1);
			scanf("%d",&tmp2);
			adj_list[tmp1].push_back(tmp2);
			adj_list[tmp2].push_back(tmp1);	
		}
        LL res=0;
		make(1,0);
		for(int i=1;i<=N;i++)
		{
			res+=i*Im[i];
			res%=1000000007;
		}
		printf("%lld\n",res);
	}
}
