#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define in(a) scanf("%d", &a)
#define out(a) printf("%d\n", a);
#define loop(n) for(int i=0; i<n; i++)
#define pb(u) push_back(u)

#define Max 10001
#define mod 1000000007

vector<int> adj[Max];
bool isVisit[Max];
ll size[Max];

ll ans;

void dfs(int root){
	int ch;
	isVisit[root] = true;
	size[root] = 1;
	ll sum = 0;
	ll temp=0;
	loop(adj[root].size()){
		ch = adj[root][i];
		if(!isVisit[ch]) {
			dfs(ch);
			temp = (temp + size[ch]*sum)%mod;
			size[root] += size[ch];
			sum += size[ch];
		}
	}
	
	temp = (2*temp + 2*sum + 1)%mod;
	ans = (ans + temp*root)%mod;
}

	int t,n,u,v;

	in(t);

		}

			in(u);

			adj[u].pb(v);
			adj[v].pb(u);
		}

		dfs(1);
		out(ans);
	}

}
