#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
 
#define maxn 444444
 
int i, n, m, x[maxn], y[maxn], z[maxn], pr[maxn], X, Y, cc[maxn], gx, gy, tqn, tqi;
char dd[maxn];
pair<int, int> a[maxn];
long long ans;
 
int fs(int k){
	if (k != pr[k]) pr[k] = fs(pr[k]);
	return pr[k];
}
 
int main(){
	scanf("%d", &tqn);
	for(tqi = 0; tqi<tqn; tqi++){
		ans = 0;
		scanf("%d %d", &n, &m);
		while (n>100000) ;
		while (m>400000) ;
		gets(dd); gets(dd);
		for(i = 1; i<=m; i++){
			scanf("%d %d %d", &x[i], &y[i], &z[i]);
			while(x[i]==y[i]);
			while(z[i]<-20000 || z[i]>20000) ;
		}
		for(i = 1; i<=m; i++) a[i].first = z[i], a[i].second = i;
		sort(a+1, a+m+1);
		for(i = 1; i<=n; i++){
			pr[i] = i;
			cc[i] = (dd[i-1] == 'R');
		}
		for(i = 1; i<=m; i++){
			X = x[a[i].second]; Y = y[a[i].second];
			gx = fs(X); gy = fs(Y);
			if ((cc[gx] & cc[gy]) == 0 && gx != gy){
			pr[gx] = gy;
				cc[gy] |= cc[gx];
				cc[gx] |= cc[gy];
				ans += a[i].first;
			}else if (a[i].first<0) ans += a[i].first;
		}
		cout << ans << endl;
	}
	return 0;
}
