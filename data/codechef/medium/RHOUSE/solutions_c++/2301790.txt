using namespace std;

int n,m,g,i;
char c;
long long ans;
int x[400005],y[400005],z[400005],t[400005],p[400005];

bool comp(int i, int j){
    return z[i]<z[j];
    }

int find(int x){
    if(p[x]==x)return x;
    else{
         p[x]=find(p[x]);
         return p[x];
         }
    }

void unite(int x, int y, int z){
    x=find(x);y=find(y);
    if(x!=y||z<0)ans+=1ll*z;
    if(x==y)return;
    p[x]=y;
    return;
    }

int main(){
           // freopen("input.txt","r",stdin);
           // freopen("output.txt","w",stdout);
            scanf("%d",&g);
            while(g--){
                    scanf("%d%d\n",&n,&m);
                    for(i=0; i<=n; i++)p[i]=i;
                    for(i=1; i<=n; i++){
                            c=getchar();
                            if(c=='R')unite(i,0,0);
                            }
                    ans=0;
                    for(i=0; i<m; i++){
                            scanf("%d%d%d",&x[i],&y[i],&z[i]);
                            t[i]=i;
                            }
                    sort(t,t+m,comp);
                    for(i=0; i<m; i++)
                            unite(x[t[i]],y[t[i]],z[t[i]]);
                    printf("%lld\n",ans);
                    }
            return 0;
}
