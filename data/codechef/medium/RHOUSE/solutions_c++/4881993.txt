#include<bits/stdc++.h>
#define pii pair<int,pair<int,int> >
using namespace std;
vector <pii> e;

int parent[100001];
int n,m;

int find(int x) {
	if (x!=parent[x]) {
		parent[x]=find(parent[x]);
	}
	return parent[x];
}

void ans() {
	sort(e.begin(),e.end());
	int p,q,i,j,k,x,y;
	long long a=0;
	for (i=0;i<e.size();i++) {
		p=e[i].second.first;
		q=e[i].second.second;
		j=find(p);
		k=find(q);
		if (j!=k) {
			parent[k]=j;
			a+=e[i].first;
		}
		else {
			if (e[i].first<=0) {
				a+=e[i].first;
			}
		}
	}
	printf("%lld\n",a);
}

int main() {
	int t,n,m,i,j,k,u,v,x,y;
	scanf("%d",&t);
	char s[100001];
	while (t--) {
		scanf("%d %d",&n,&m);
		e.clear();
		for (i=1;i<=n;i++) parent[i]=i;
		scanf("%s",s+1);
		vector <int> rest;
		for (i=1;i<=n;i++) {
			if (s[i]=='R') {
				rest.push_back(i);
			}
		}
		for (i=1;i<rest.size();i++) {
			parent[rest[i]]=parent[rest[0]];
		}
		while (m--) {
			scanf("%d %d %d",&u,&v,&i);
		//	if (s[u]==s[v] && s[u]=='R') continue;
			e.push_back(make_pair(i,make_pair(u,v)));
		}
		ans();
	}
	return 0;
}
