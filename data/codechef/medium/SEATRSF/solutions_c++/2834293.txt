using namespace std;

LL bigmod(LL b,LL p){
    if(p==0ll) return 1ll;
    if(p&1ll) return (b*bigmod(b,p-1ll))%MOD;
    LL x=bigmod(b,p/2ll)%MOD;
    return (x*x)%MOD;
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie();

    cin>>test;
    for(LL kase=1;kase<=test;kase++){
        LL n,m,q,k;
        cin>>n>>m>>q>>k;
        if(m<=q) cout<<0<<endl;
        else cout<<((m-q)*(bigmod(q+1,n)-(2ll*bigmod(q,n)-bigmod(q-1,n)+MOD)%MOD+MOD)%MOD)%MOD<<endl;
    }
    return 0;
}
