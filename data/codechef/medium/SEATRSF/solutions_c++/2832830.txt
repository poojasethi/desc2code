LLI powx(LLI a, int n)
{
	LLI ans = 1;
	while(n > 0)
	{
		if(n&1)
			ans = (ans * a) % M;
		a = (a * a) % M;
		n >>= 1;
	}
	return ans;
}

int main()
{
	int t, n, m, q, k;

	for(si(t); t--; )

		scanf("%d%d%d%d", &n, &m, &q, &k);
		if(n == 1)
		{
			if(q != 0)
				puts("0");
			else
				printf("%d\n", m);
		}
		else
		{
			int num_interval = max(0, m - q);
			LLI ans = (powx(q+1, n) + powx(q-1, n) - 2*powx(q, n) + 4LL*M) % M;
			ans = (ans * num_interval) % M;
			printf("%lld\n", ans);
		}
	}
	return 0;
}
