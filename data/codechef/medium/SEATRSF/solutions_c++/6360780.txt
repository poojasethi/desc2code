#include <iostream>
#include <cmath>
#include <cstdio>
#include <algorithm>

#define Mod 1000000007

using namespace std;

long long expo(long long base , long long exp)
{
    long long ans = 1;
    while(exp)
    {
        if(exp & 1)
        {
            ans *= base;
            ans %= Mod;
        }
        base *= base;
        base %= Mod;
        exp >>= 1;
    }
    return ans;
}

int main()
{
    int T;
    scanf("%d" , &T);
    while(T--)
    {
        long long n , m , q , k;
        scanf("%lld%lld%lld%lld" , &n , &m , &q , &k);
        if(q > m)
        {
            printf("0\n");
            continue;
        }
        long long ans = (((((expo(q + 1 , n) - ((expo(q , n) << 1) % Mod) + expo(q - 1 , n)) % Mod) + Mod) % Mod) * (m - q)) % Mod;
        printf("%lld\n" , ans);
    }
    return 0;
}
