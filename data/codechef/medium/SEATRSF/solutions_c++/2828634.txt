#include<cstdio>
#include<iostream>
using namespace std;
long long int MAX=1000000007;
long long int power(long long int a,long long int d)
{
	long long int x=1,y=a;
	while(d)
	{
		if(d%2==1)
			x=(x*y)%MAX;
		y=(y*y)%MAX;
		d/=2;
	}
	return x;
}
int main(){
	int tc;
	scanf("%d",&tc);
	while(tc--){
		long long int n,m,q,p;
		scanf("%lld %lld %lld %lld",&n,&m,&q,&p);
		if(n==1){
			printf("0\n");
		}
		else{
			long long int k,l;
			k=1+q;
			if(k > m){
				printf("0\n");
			}
			else{
				l=m-k+1;
				long long int ans=((power(k,n)-2*power(k-1,n)+power(k-2,n))%MAX+MAX)%MAX;
				ans=(long long int)(ans*l)%MAX;
//				long long int y=(long long int)(n*(n-1))%MAX;
			//	ans=(ans*y)%MAX;
				printf("%lld\n",ans);
			}
		}
	}
	
	return 0;
}
