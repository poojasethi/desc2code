#define gc getchar_unlocked
#define INF 1000000007
# define LL long long 
using namespace std;
void scanint(LL &x)
{
register LL c = gc();
x = 0;
LL neg = 0;
for(;((c<48 || c>57) && c != '-');c = gc());
if(c=='-') {neg=1;c=gc();}
for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
if(neg) x=-x;
} 


LL modpow(LL n ,LL k,LL mod)
{LL ans=1;
while(k>0)
{if(k&1)
ans=(ans*n)%mod;
k>>=1;
n=(n*n)%mod;
}
return ans%mod;}
