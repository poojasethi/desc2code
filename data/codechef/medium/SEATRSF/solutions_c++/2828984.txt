#define MOD 1000000007
long long poww(long long n,long long  m) 
{ 
  	if(m==0) return (1)%MOD;
   	long long x=poww (n,m/2);
    if(m%2==0) return (x*x)%MOD; 
	else return (((x*x)%MOD)*n) %MOD;
}

{

long long t;

{

	if(m<q)
{
	cout<<0<<endl;

}

	
	a=poww(q+1,n);
