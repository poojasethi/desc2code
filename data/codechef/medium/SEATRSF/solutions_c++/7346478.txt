#include <cstdio>
#include <algorithm>
using namespace std;

const int MOD = 1000000007;

inline void add(int &a, int b, int mod)
{
    a += b;
    if (a >= mod) {
        a -= mod;
    }
}

inline int powmod(int x, int t, int mod) // x ^ t % mod
{
    if (t == 0) {
        return 1 % mod;
    }
    int y = powmod(x, t >> 1, mod);
    y = (long long)y * y % mod;
    if (t & 1) {
        y = (long long)y * x % mod;
    }
    return y;
}

int main()
{
    int T;
    for (scanf("%d", &T);T --;) {
        int n, m, q, k;
        scanf("%d%d%d%d", &n, &m, &q, &k);
        int x = max(m - q, 0);
        int y = 0;
        if (q == 0) {
            y = 1;
        } else {
            add(y, powmod(q + 1, n, MOD), MOD);
            add(y, (MOD - powmod(q, n, MOD) * 2 % MOD) % MOD, MOD);
            add(y, powmod(q - 1, n, MOD), MOD);
        }
        printf("%d\n", (long long)x * y % MOD);
    }
}
