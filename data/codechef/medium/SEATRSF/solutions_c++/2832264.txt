#include <iostream>
#include <cmath>
#define MOD 1000000007
using namespace std;

typedef long long int ll;

ll mod_exp(ll base, ll exp, ll mod){
	ll result = 1;
	while (exp > 0){
		if ((exp & 1) == 1)
			result = (result * base) % mod;
		exp /= 2;
		base = (base * base) % mod;
	}
	return result;
}

int main(){

    int ts;
    cin >> ts;
    while(ts--){
        ll n, m, diff, k;
        cin >> n >> m >> diff >> k;
        if (n == 1 or m == 1 or diff > m - 1)
            cout << 0 << endl;
        else {
            ll ans = mod_exp(diff+1, n, MOD);
            ans += mod_exp(diff-1, n, MOD);
            ans %= MOD;
            ans -= mod_exp(diff, n, MOD);
            if (ans < 0) ans = MOD + ans;
            ans %= MOD;
            ans -= mod_exp(diff, n, MOD);
            if (ans < 0) ans = MOD + ans;
            ans %= MOD;
            ans = ans * (m - diff);
            ans %= MOD;
            cout << ans << endl;
        }
    }

    return 0;
}
