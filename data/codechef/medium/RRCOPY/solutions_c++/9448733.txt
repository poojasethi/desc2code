#include <iostream>
#include <set>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,r;
        cin>>n;
        set<int> a;
        for(int i=0;i<n;i++)
        {
            cin>>r;
            a.insert(r);
        }
        cout<<a.size()<<endl;
    }
}
