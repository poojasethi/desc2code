#include<bits/stdc++.h>
#include<tr1/unordered_set>
using namespace std;
int main()
{
	
	int t;
	cin>>t;
	while(t--)
	{
	vector<int> v;
	int i,n,m;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>m;
		v.push_back(m);
	}
//	sort(v.begin(),v.end());
//	v.erase(unique(v.begin(),v.end()),v.end());
	std::tr1::unordered_set<int> s(v.begin(),v.end());
	v.assign(s.begin(),s.end());
	i=v.size();
	cout<<i<<endl;
	}
	return 0;
}
