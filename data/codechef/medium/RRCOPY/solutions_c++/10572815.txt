#include<bits/stdc++.h>
#define S(n) scanf("%li",&n);
#define s(n) scanf("%d",&n);
#define p(n) printf("%d\n",n);
using namespace std;
int main()
{
int t;
s(t)
while(t--)
{
    int n;
    s(n)
    int a[n];
    for(int i=0;i<n;i++)
        s(a[i])
        int c=0;
        sort(a,a+n);
        for(int i=0;i<n-1;i++)
        {
            if(a[i+1]==a[i])
                c++;
        }
    p(n-c);
}
return 0;
}
