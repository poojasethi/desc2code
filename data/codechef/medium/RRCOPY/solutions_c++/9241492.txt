#include<stdio.h>
#include<algorithm>
int main()
{
	int t;
	scanf("%d" ,&t);
	while(t--)
	{
		int n,i,arr[100005],count=0;
		scanf("%d" ,&n);
		for(i=0;i<n;i++)
		{
		  scanf("%d" ,&arr[i]);
		}
		std ::sort(arr,arr+n);
		if(arr[0]==arr[n-1])
		{
		  printf("%d\n" ,1);
		  continue;}
		for(i=0;i<n;i++)
		{
			if(arr[i]^arr[i+1])
			count++;
		}
		printf("%d\n" ,count);
	}
	return 0;
}
