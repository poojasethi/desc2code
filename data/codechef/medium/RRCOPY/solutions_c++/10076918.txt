#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int num;
        int dp[100000];
        for(int i=0;i<100000;i++){
            dp[i]=0;
        }
        int sum=0;
        for(int i=0;i<n;i++){
            cin>>num;
            if(dp[num-1]==0){
                dp[num-1]=1;
                sum++;
            }
        }
        cout<<sum<<endl;
        
        
    }
}
