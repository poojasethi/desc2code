#include<iostream>
using namespace std;

int main()
{

    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[100000] = {0};
        int i,j,count=0;
        for(i=0;i<n;i++)
        {
            cin>>j;
            if(a[j]==0)
            {
                a[j] = 1;
                count++;
            }
        }
        cout<<count<<endl;
    }
}
