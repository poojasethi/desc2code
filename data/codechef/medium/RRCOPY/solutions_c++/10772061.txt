#include <iostream>
#include <string.h>
using namespace std;
int main(){
	int T;
	cin >> T;
	int visited[100001];
	for(int t=0;t<T;t++){	
		int N;
		cin >> N;
		memset(visited,0,sizeof(visited));
		int count = 0;
		int a;
		for(int i=0;i<N;i++){
			cin >> a;
			if(!visited[a]){
				count++;
				visited[a]=1;
			}
			
		}
		cout << count << endl;
	}
	return 0;
	
}
