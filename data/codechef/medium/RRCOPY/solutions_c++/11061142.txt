#include<stdio.h>
int main(){
    int t,n,x;
    scanf("%d",&t);
    while(t--){
        int s[100001]={0};
        int count = 0;
        scanf("%d",&n);
        for(int i =0;i<n;i++){
            scanf("%d",&x);
            s[x]=1;
        }
        for(int i=0;i<100001;i++){
            if(s[i]==1){
                count++;
            }
        }
        printf("%d\n",count);
    }
}
