#include<iostream>
#include<string.h>
using namespace std;
int main()
{ 
 long int t,a[100010],b[100010],n,i,j,max=0;
 cin>>t;
 while(t--)
 {
  memset(b,0,sizeof(b));
   cin>>n;
   for(i=0;i<n;i++)
   {
     cin>>a[i];
     b[a[i]]++;
     if(max<a[i])
     max=a[i];
   }
   for(i=0;i<=max;i++)
   {
     if(b[i]>1)
     n=n-(b[i]-1);
    }
    cout<<n<<"\n";
   }
 return 0;
 }
