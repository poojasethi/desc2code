#include<bits/stdc++.h>
using namespace std;
set<int> s;
int main () {
    int t;
    scanf ( "%d" , &t );
    while ( t-- ) {
        int n;
        scanf ( "%d" , &n );
        int num ;
        for ( int i = 0 ; i < n ; i++ ) {
            scanf ( "%d" , &num );
            s.insert ( num );
        }
        printf ( "%d\n" , s.size() );
        s.clear();
    }
    return 0;
}
