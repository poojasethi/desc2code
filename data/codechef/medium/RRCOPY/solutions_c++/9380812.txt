#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    long int n,t;
    cin>>t;
    while(t--){
        cin>>n;
        int arr[100000],count = 0;
        for(int i = 0; i < n ; i++ ){
            cin>>arr[i];
        }
        sort(arr,arr+n);
        for(int i= 0; i<n; i++){
            if(arr[i]!=arr[i+1]){
                count = count +1;

            }
        }
        cout<<count<<endl;

        }

    return 0;
}
