#include<iostream>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		long long int a[100090]={0},b[100090]={0},i,n,c;
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0;i<n;i++)
		{
			if(b[a[i]]==0)
			b[a[i]]=1;
		}
		c=0;
		for(i=0;i<=100090;i++)
		{
		if(b[i]==1)
		c++;
		}
	cout<<c<<endl;
	}
	return 0;
}
