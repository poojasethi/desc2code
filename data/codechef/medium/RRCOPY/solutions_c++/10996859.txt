#include<stdio.h>
#include<iostream>
#include<math.h>
#include<stdlib.h>
#include<cstring>


using namespace std;
int  arr[100001];
bool map[100001];

int main(){
	int t,n;
	scanf("%d",&t);
	while(t--){
		memset(map,false,sizeof(map));
		scanf("%d",&n);
		int ans = 0;
		for(int i=0; i<n; i++){
			scanf("%d",&arr[i]);
			if(!map[arr[i]])
				ans++;
			map[arr[i]] = true;
		}
		cout<<ans<<endl;
	}
}
