#include <iostream>
using namespace std;
 
int main() {
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		long arr[n],nmbrs[100000],i,c=0;
		for(i=0;i<n;i++){
			cin>>arr[i];
			nmbrs[i]=0;
		}
		for(;i<100000;i++)
			nmbrs[i]=0;
		for(i=0;i<n;i++){
			nmbrs[arr[i]-1]++;
		}
		for(i=0;i<100000;i++){
			if(nmbrs[i]>0)
				c++;
		}
		cout<<c<<"\n";
	}
	return 0;
}
