#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int t,i,j;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        sort(a,a+n);
        int count=1;
        for(int i=1;i<n;i++){
            if(a[i]!=a[i-1])
                count++;
        }
        cout<<count<<endl;

    }

}
