#include<cstdio>
#include<map>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int n;
        scanf("%d",&n);
        map<int,int> m;
        int count=0;
        for(int i=0;i<n;i++){
        int tmp;
        scanf("%d",&tmp);
        if(!m[tmp]){
            count++;
            m[tmp]=1;
        }
        }
        printf("%d\n",count);
    }
}
