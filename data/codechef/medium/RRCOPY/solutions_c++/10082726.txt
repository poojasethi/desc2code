#include<iostream>
#include<stdio.h>
#include<stdlib.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long int n,*a,i,c,b[100001]={0};
        scanf("%ld",&n);
        c=0;
        a=(long int*)malloc(n*sizeof(long int));
        for(i=0;i<n;i++)
            scanf("%ld",&a[i]);
        for(i=0;i<n;i++)
           {

           if(b[a[i]]==0)
            b[a[i]]++;
            else
                a[i]=-1;
           }
           for(i=0;i<n;i++)
            if(a[i]!=-1)
            c++;
            printf("%ld\n",c);

    }
    return 0;
}
