#include <iostream>

using namespace std;

int main()
{
    int n,count=0,temp,t;
    cin>>t;
    while(t--){
    int b[100001]={0};
        count=0;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        for(int i=0;i<n;i++){
            temp=a[i];
            b[temp]++;
        }
        for(int i=0;i<100001;i++){
            if(b[i]>0)
                count++;
        }
        cout<<count<<endl;
    }
    return 0;
}
