#include<iostream>
#include<cstdlib>
using namespace std;
int main()
{
	 int t;
	cin >>t;
	while(t--)
	{
		long long int x,y,ans;
		cin>>x>>y;
		x=abs(x);
		y=abs(y);
		if(x>y)
		{
		    if((x-y)%2==1)
		      ans=2*x+1;
		     else
		      ans=2*x;
		}
		else
		{
		    
		    if((y-x)%2==1)
		     ans=2*y-1;
		    else
		     ans=2*y;
		}
		cout << ans << endl;
	}
	return 0;
}
