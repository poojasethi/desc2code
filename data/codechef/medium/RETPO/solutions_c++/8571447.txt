#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
   long long int t,x,y;
   scanf("%lld",&t);
   while(t--)
   {
      scanf("%lld%lld",&x,&y);
      if(x < 0)
        x *= (-1);
      if(y < 0)
        y *= (-1);
      if(x == y)
        printf("%lld\n",x+y);
      else if(y > x)
      {
          long long int ans = 2*x;
          y -= x;
          if( (y % 2 == 1) )
            ans += 2*y - 1;
          else
            ans += 2*y;
          printf("%lld\n",ans);
      }
      else
      {
          long long int ans = 2*y;
          x -= y;
          if( (x % 2 == 1))
            ans += 2*x + 1;
          else
            ans += 2*x;
          printf("%lld\n",ans);
      }
   }
    return 0;
}
