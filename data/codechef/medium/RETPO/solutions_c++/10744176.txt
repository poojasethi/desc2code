#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll x,y;
        scanf("%lld%lld",&x,&y);
        x=abs(x);
        y=abs(y);
        ll ans,diff;
        ans= 2*min(x,y);
        
        if(x>y)
        {
        diff= max(x,y)-min(x,y);
        if(diff%2==0)
         ans+=2*diff;
        else
         ans=ans+2*diff+1;
        }
        else
        if(y>x)
        {
           diff= max(x,y)-min(x,y);
         if(diff%2==0)
         ans+=2*diff;
          else
         ans=ans+2*diff-1; 
        }
        printf("%lld\n",ans);
    }
    return 0;
}
