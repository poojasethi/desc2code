#include<cstdio>
#include<iostream>
#include<algorithm>

using namespace std;
int main(){
long int tst,x,y,ans;
scanf("%ld",&tst);
while(tst--)
{    ans=0;
    scanf("%ld%ld",&x,&y);
    if((x+y)%2==0){
        ans+=2*max(abs(x),abs(y));
    }
    else
    {   long int z=abs(y)-1;
        ans+=2*max(abs(x),abs(z))+1;
    }
    printf("%ld\n",ans);
}

return 0;

}
