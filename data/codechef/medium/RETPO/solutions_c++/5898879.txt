#include<iostream>
using namespace std;
long long int che(long long int a)
{
    if(a>=0)
        return a;
    else
        return -a;
}
long long int check(long long int x,long long int y)
{
    long long int cost;
    if(x == 0){
                    if(y%2==0){
                cost = 2*che(y);}
                    else{ cost =  2*che(y) - 1;}
            }
            else if(y == 0){
                if(x%2==0){
                cost =  2*che(x);}
                    else{ cost =  2*che(x) + 1;}
            }
            else if(x == 0 && y == 0){
                cost =  0;
            }
         return cost;
}
int main()
{
    long long int t,x,y,z,cost;
    cin >> t;
    while(t--){
            cost = 0;
            cin >> x >> y;
            x = che(x);
            y = che(y);
            if(x !=0 && y !=0){
                z = min(x,y);
                cost = cost + 2*z;
                x = x - z;
                y = y - z;
                cost+= check(x,y);
            }
            else{
                cost = check(x,y);
            }
            cout << cost << endl;
        }

    return 0;
}
