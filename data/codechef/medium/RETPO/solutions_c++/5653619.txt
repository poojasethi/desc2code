int scanint(){
	register int c = getchar();
	int x=0;
	bool neg=false;
	for(;(c<48 || c>57) && c!=45 ;c=getchar());
	if(c==45){
		neg=true;
		c=getchar();
	}
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	if(neg==true)
	x=x*(-1);
	return x;
}

long long scanlong()
{
	register int c = getchar();
	long long x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

typedef long long ll;

int main(){
	int t,x,y;
	ll mini , ans ;
	t = scanint();
	while(t--){
		x = scanint();
		y = scanint();
		if(x<0)
			x = -x ;
		if(y<0)
			y = -y ;
		mini = std::min(x,y);
		ans = 2*mini ;
		if(x == y){
			cout<<ans<<endl;
			continue;
		}
		x -= mini ;
		y -= mini ;
		if(y){
			ans += (4 * (y/2));
			if(y&1)
				ans++;
		}else{
			ans += (4 * (x/2));
			if(x & 1)
				ans += 3 ;
		}
		cout<<ans<<endl;
	}
}
