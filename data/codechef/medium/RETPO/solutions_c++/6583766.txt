#include <bits/stdc++.h>
#define lld long long int 

lld min(lld a,lld b)
{
    return ((a<b)?(a):(b));
}

lld value(lld a,lld b)
{
    return ((a&1)?(2*a+1):(2*a))+((b&1)?(2*b-1):(2*b));
}

int main()
{
	int t;
	scanf("%d",&t);
	long long int x,y,z,ans;
	while(t--)
	{   
	   ans=0;
		scanf("%lld%lld",&x,&y);
		x=abs(x);y=abs(y);
		z=min(x,y);
	    ans=2*z;	
	
	    x-=z;y-=z;
	    
	    ans+=value(x,y);
		printf("%lld\n",ans);	

	}
return 0;	
}
