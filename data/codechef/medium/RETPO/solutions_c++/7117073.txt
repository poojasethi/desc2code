#include<iostream>
using namespace std;

int main()
{
    std::ios::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        int x,y;
        cin>>x>>y;
        if(x<0){x=-x;}
        if(y<0){y=-y;}
        int ans=0;
        if(x<y)
            {
                ans = 2*x;
                int temp= y-x;
                if(temp%2==0){ans+= (2*temp);}
                else{ans+=(2*temp)-1;}
                cout<<ans<<"\n";
            }
        else if(x>y)
            {
                ans = 2*y;
                int temp= x-y;
                if(temp%2==0){ans+= 2*temp;}
                else{ans+= 2*temp +1;}
                cout<<ans<<"\n";
            }
        else {cout<<2*x<<"\n";}
    }

    return 0;
}
