#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int main()
{
	lli ans,i,j,k,m,n,t,x,y;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&x,&y);

		if(x<0)
			x=-x;
		if(y<0)
			y=-y;

		if(x>y)
		{
			if((x+y)%2==0)
				ans=2*x;
			else
				ans=2*x+1;
		}

		else if(y>x)
		{
			if((x+y)%2==0)
				ans=2*y;

			else
				ans=2*y-1;
		}

		else
			ans=2*x;

		printf("%lld\n",ans);
	}
}
