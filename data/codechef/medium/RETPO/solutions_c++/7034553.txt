#include<iostream>
#include<cstdlib>
using namespace std;
int main()
{
	 int t;
	cin >>t;
	while(t--)
	{
		long long int x,y,a=0,b=0,count=0,max1,i,j,k;
		cin>>x>>y;
		x=abs(x);
		y=abs(y);
		if((x+y)%2==0)
		{
			max1=max(x,y);
			count=2*max1;
		}
		else
		{
			k=y-1;
			k=abs(k);
			i=max(x,k);
			j=max(x,y+1);
			if(i>j)
			count=1+2*j;
			else
			count=1+2*i;
		}
		cout << count << endl;
	}
	return 0;
}
