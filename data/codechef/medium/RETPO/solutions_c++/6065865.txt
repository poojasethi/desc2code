#include<iostream>
#include<stdio.h>
#include<iomanip>

using namespace std;
typedef  long long ll;

ll calculate_steps(ll x,ll y)
{
    ll minimum=min(x,y);
    ll steps=2*minimum;
    x-=minimum;
    y-=minimum;
    if(x==0)
    {
        if(y%2==0)
            steps+=2*y;
        else
            steps+=((2*y)-1);
    }
    else if(y==0)
    {
        if(x%2==0)
            steps+=2*x;
        else
            steps+=((2*x)+1);
    }
    return steps;
}

int main()
{
   ll t;
   cin>>t;
   while(t--)
   {
      ll x,y;
      cin>>x>>y;
      if(x<0)
        x*=(-1);
      if(y<0)
        y*=(-1);
      cout<<calculate_steps(x,y)<<endl;
   }
}
