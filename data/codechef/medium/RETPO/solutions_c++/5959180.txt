#include <iostream>
#include <cstdlib>
#include <cmath>
using namespace std;
 
int main()
{
    int test;
 
    cin>>test;
 
    while(test--)
    {
        long long int X,Y;
        
    	cin>>X>>Y;
    	
        if(abs(X)==abs(Y))
            cout<<2*abs(X)<<endl;
 
        else if(abs(X)> abs(Y))
        {
            if((abs(X)-abs(Y))%2)
            	cout<<2*abs(X)+1<<endl;
 
            else
            	cout<<2*abs(X)<<endl;
        }
 
        else if(abs(Y)> abs(X))
        {
            if((abs(Y)-abs(X))%2)
            	cout<<2*abs(Y)-1<<endl;
 
            else
            	cout<<2*abs(Y)<<endl;
        }
 
    }
 
    return 0;
}
