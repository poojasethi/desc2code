#include<stdio.h>
#include<iostream>
#include<math.h>
#include<algorithm>
using namespace std;
int main()
{  long t,x,y,ans;
   cin>>t;
   while(t--)
   { 
     cin>>x>>y;
	 if(x<0) x=-x;       if(y<0)  y=-y;
	 
	 if(x==y) ans=2*x;
	 else if(x<y)
	 {  if(x%2==0)  {   if(y%2==0)  ans=2*y;
	                    else        ans=2*y-1;
	                }
		else        {   if(y%2!=0)  ans=2*y;
	                    else        ans=2*y-1;
	                }
	 }

     else 
	 {  if(x%2==0)  {   if(y%2==0)  ans=2*x;
	                    else        ans=2*x+1;
	                }
		else        {   if(y%2!=0)  ans=2*x;
	                    else        ans=2*x+1;
	                }
	 }	 
	 cout<<ans<<endl;
   }
   return 0;
}
