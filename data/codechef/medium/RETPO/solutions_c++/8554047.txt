#include <bits/stdc++.h>

using namespace std;

int main()
{
    long int t;
    scanf("%ld",&t);
    while(t--)
    {
        long long int x,y,temp,var,res,shiftx,shifty;
        scanf("%lld %lld",&x,&y);
        if(abs(x)==abs(y))
            printf("%lld\n",abs(x)+abs(y));
        else
        {
            if(abs(x)<abs(y))
            {
                temp=abs(x)+abs(x);
                var=abs(x);
            }
            else
            {
                temp=abs(y)+abs(y);
                var=abs(y);
            }
            shiftx=abs(x)-var;
            shifty=abs(y)-var;
            if(shiftx%2==0&&shiftx!=0)
            {
                res=shiftx*2;
            }
            else if(shifty%2==0&&shifty!=0)
            {
                res=shifty*2;
            }
            else if(shiftx%2!=0)
            {
                res=2*shiftx+1;
            }
            else if(shifty%2!=0)
            {
                res=2*shifty-1;
            }
            printf("%lld\n",temp+res);
        }
    }
    return 0;
}
