#include<bits/stdc++.h>
using namespace std;
typedef long long int lld;
int main()
{
	ios::sync_with_stdio(false);
	lld t;
	cin>>t;
	while(t--)
	{
		lld x,y;
		cin>>x>>y;
		x=abs(x);
		y=abs(y);
		if(x==y)
			cout<<2*abs(x)<<endl;
		else if(x>y)
		{
			if((abs(x)-abs(y))%2==0)
				cout<<2*abs(x)<<endl;
			else
				cout<<2*abs(x)+1<<endl;
		}
		else
		{
			if((abs(x)-abs(y))%2==0)
				cout<<2*y<<endl;
			else
				cout<<2*y-1<<endl;
		}
	}
	return 0;
}
