#include<stdio.h>
long int abs(long int n)
{ if(n<0)
    return (-1)*n;
  else
    return n;
}
long int min(long int a,long int b)
{ if(a<b)
    return a;
  else
    return b;
}
int main()
{ int t;
  scanf("%d",&t);
  while(t--)
  { long int x,y,w;
    scanf("%ld%ld",&x,&y);
    x=abs(x);
    y= abs(y);
    long long int ans = 0;
    
    w = min(x,y);
    ans = 2*w;
    x = x-w;
    y = y-w;
    
    if(x==0&&y!=0)
    {  if(y%2==0)
         ans += y*2;
       else
         ans += y*2 -1;
    }
    else
     if(y==0&&x!=0)
     {  if(x%2==0)
          ans += x*2;
        else
          ans += x*2+1;
     }
   
    printf("%lld\n",ans);
    
  }
  
  return 0;
}
