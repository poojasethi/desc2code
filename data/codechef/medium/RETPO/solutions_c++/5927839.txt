#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,x,y;

    scanf("%d",&t);

    while(t--)
    {
        scanf("%d%d",&x,&y);
        if(abs(x)==abs(y))
            printf("\n%d",2*abs(x));

        else if(abs(x)> abs(y))
        {
            if((abs(x)-abs(y))%2)
            printf("\n%d",2*abs(x)+1);

            else
            printf("\n%d",2*abs(x));
        }

        else if(abs(y)> abs(x))
        {
            if((abs(y)-abs(x))%2)
            printf("\n%d",2*abs(y)-1);

            else
            printf("\n%d",2*abs(y));
        }

    }

    return 0;
}
