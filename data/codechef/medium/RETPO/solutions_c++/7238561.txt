#include<bits/stdc++.h>
//#include<math.h>
using namespace std;
long int min(long int a,long int b)
{
	if(a>b)
	return b;
	else
	return a;
}
int main()
{
	long int test,x,y;
	cin>>test;
	while(test--)
	{
		long int w,ans=0;
		cin>>x>>y;
		x=labs(x);
		y=labs(y);
		w=min(x,y);
		ans=2*w;
		x=x-w;
		y=y-w;
		if(y==0 && x!=0)
		{
			if(x%2==1)
			ans+=(2*x)+1;
			else
			ans+=(2*x);
		}
		else if(x==0 && y!=0)
		{
			if(y%2==1)
			ans+=(2*y)-1;
			else
			ans+=(2*y);
		}
		cout<<ans<<"\n";
	}
	return 0;
}
