#include<stdio.h>

int axis_move(int x,int y)
{
int ans;
if(y==0)
ans=x&1?((x<<1)+1):(x<<1);
else if(x==0)
ans=y&1?((y<<1)-1):(y<<1);
//printf("AXIS MOVE ANS= %d\n",ans);
return ans;
}

int main()
{
int i,j,k,t,x,y,ans;
scanf("%d",&t);
while(t--)
{
scanf("%d %d",&x,&y);
x=x<0?-x:x;
y=y<0?-y:y;
ans=x<y?((x<<1)+axis_move(0,y-x)):((y<<1)+axis_move(x-y,0));
printf("%d\n",ans);
}

return 0;
}
