#include<bits/stdc++.h>
using namespace std;

#define sd(x) scanf("%d",&x)
#define ss(x) scanf("%s",x)
#define lld long long int
#define slld(x) scanf("%lld",&x)

int main()
{
    lld t,x,y,ans,k;
	slld(t);
	while(t--)
	{
       slld(x);slld(y);
       x=abs(x);y=abs(y);
       k=min(x,y);
       ans=2*k;
       x=x-k;
       y=y-k;
       if(x==0)
       {
            if(y%2==0)
                ans+=2*y;
            else
                ans+=2*y-1;
       }
       else
       {
            if(x%2==0)
                ans+=2*x;
            else
                ans+=2*x+1;
       }
       cout<<ans<<endl;
	}


	return 0;
}
