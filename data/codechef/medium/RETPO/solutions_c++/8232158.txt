#include <iostream>
#include <stdlib.h>
#define ll long long int 
#define min(a,b) (a>b)?b:a
using namespace std;

int main() 
{
	int t;
	ll x,y,count,diff;
	cin>>t;
	while(t--)
	{
		count=0;
		cin>>x>>y;
		x=abs(x);
		y=abs(y);
		count+=2*(min(x,y));
		diff=abs(x-y);
		if(y>x)
	{	if(diff%2==0)
			count+=2*diff;
		else
			count+=2*diff-1;}
		else
		{
				if(diff%2==0)
			count+=2*diff;
		else
			count+=2*diff+1;
		}
		cout<<count<<endl;
	}
	return 0;
}
