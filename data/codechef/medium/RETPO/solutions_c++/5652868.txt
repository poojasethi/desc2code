bool comp(int a,int b)

int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
int main(void) {
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int x,y;
        long long int r=0;
        scanf("%d %d",&x,&y);
        r=2*min(abs(x),abs(y));
        if(abs(x)-r/2)
        {
            if((abs(x)-r/2)%2)
                r+=2*(abs(x)-r/2)+1;
            else
                r+=2*(abs(x)-r/2);
        }
        else if(abs(y)-r/2)
        {
            if((abs(y)-r/2)%2)
                r+=2*(abs(y)-r/2)-1;
            else
                r+=2*(abs(y)-r/2);
        }
        printf("%lld\n",r);
    }
	return 0;
}
