#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int t;
    long long x,y;
    scanf("%d",&t);
    while(t--)
    {
        cin>>x>>y;
        if(x<0)
            x*=(-1);
        if(y<0)
            y*=(-1);
        if(y<x)
        {
            if((x-y)%2==0)
            {
                cout<<((2*y)+2*(x-y))<<endl;
            }
            else
                cout<<((2*y)+2*(x-y)+1)<<endl;
        }
        else
        {
            if((y-x)%2==0)
                cout<<((2*x)+2*(y-x))<<endl;
            else
                cout<<((2*x)+2*(y-x))-1<<endl;
        }
    }
    return 0;
}
