#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
		long long int x,y;
		cin>>x>>y;
		if(x<0)
		x=x*(-1);
		if(y<0)
		y=y*(-1);
		if(x<y)
		{
			long long int r=x*2;
			long long int e=y-x;
			if(e%2==0)
			r+=e*2;
			else
			r+=(e*2)-1;
			cout<<r<<endl;
		}
		else if(x>y)
		{
			long long int r=y*2;
			long long int e=x-y;
			if(e%2==0)
			r+=(e*2);
			else
			r+=(e*2)+1;
			cout<<r<<endl;
		}
		else
		{
			long long int r=y*2;
			cout<<r<<endl;
		}
	}
	return 0;
}
