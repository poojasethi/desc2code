#include<bits/stdc++.h>
using namespace std;

int main()
{
std::ios::sync_with_stdio(false);

int T;
cin>>T;

while(T--)
{
int x, y;
cin>>x>>y;

x=abs(x);
y=abs(y);

long long int ans=0;

if(x==y)
ans=2*x;

else if(x>y)
{
ans=2*y;

if((x-y)%2==0)
ans+=((x-y)*2);

else
ans+=(((x-y+1)*2)-1);
}

else
{
ans=2*x;

if((y-x)%2==0)
ans+=((y-x)*2);

else
ans+=(((y-x)*2)-1);
}

cout<<ans<<endl;
}
return 0;
}
