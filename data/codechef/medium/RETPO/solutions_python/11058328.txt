t=int(raw_input())
for i in range(t):
    x,y=map(int,raw_input().split())
    x,y=abs(x),abs(y)
    if x==y:
        print x+y
    elif x<y:
        if (x+y)%2==0:
            print 2*y
        else:
            print 2*y-1
    else:
        if (x+y)%2==0:
            print 2*x
        else:
            print 2*x+1
