#include    <bits/stdc++.h>
using namespace std;
struct res{
  int m,e;
  char age;
}rest[100004];
char txt[400005],pat[4005];
void read()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        int x;
        scanf("%d",&x);
        rest[i].age=x;
        rest[i].m=rest[i].e=0;
    }
    for(int i=0;i<n-1;i++)
    {
        int a,b;
        char c;
        scanf("%d%d",&a,&b);
        cin>>c;
        if(c=='M')
            rest[a].m=b;
        else
            rest[a].e=b;
    }
}
void process(int root,char *buffer,int &i)
{
    //cout<<j<<endl;
    //j++;
    if(root)
    {
        buffer[i++]=rest[root].age;
        process(rest[root].m,buffer,i);
        process(rest[root].e,buffer,i);
    }
    buffer[i++]=')';
}
int main()
{
    int n,l;
    read();
     n=0;
    process(1,txt,n);
    txt[n]='\0';
    int q;
    scanf("%d",&q);
    while(q--)
    {
        read();
        //cout<<1;
        l=0;
        process(1,pat,l);
       pat[l]='\0';
        //cout<<2;
        if(strstr(txt,pat)==NULL)
            printf("NO\n");
        else
            printf("YES\n");
    }
}
