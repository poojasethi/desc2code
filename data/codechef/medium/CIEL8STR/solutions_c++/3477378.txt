#include <iostream>
#include <stdio.h>
#include <cmath>
#include <string.h>
#include <limits.h>
#include <stdlib.h>
#include <algorithm>
#define ll long long
using namespace std;

int main()
{
    char str[1000005];
    scanf("%s",str);
    ll cnt=0;
    ll len=strlen(str);
    if(len==1)
    {
        if(((str[0]=='0')||str[0]=='8'))
        printf("1\n");
        else
        printf("0\n");
        return 0;
    }
    if(len==2)
    {
        if(str[0]=='0'||str[0]=='8')
        {
            cnt++;
        }
        if(str[1]=='0'||str[1]=='8')
        {
            cnt++;
        }
        if((str[0]*10+str[1])%8==0&&str[0]!='0')
        {
            cnt++;
        }
        printf("%lld\n",cnt);
        return 0;
    }
    if((str[len-1]-48)%8==0)
    {
        cnt++;
    }
    if((str[len-2]-48)%8==0)
    {
        cnt++;
    }
    if(((str[len-1]-48)+(str[len-2]-48)*10)%8==0&&str[len-2]!='0')
    {
        cnt++;
    }
    //cout<<cnt<<endl;
    ll zero=0;
    if(len>=3)
    for(ll int i=0;i<=len-3;i++)
    {
        if(str[i]=='8'||str[i]=='0')
        {
            //cout<<"hii";
            cnt++;
        }
        if(str[i]!='0')
        {
            if(((str[i]-48)*10+(str[i+1]-48))%8==0)
            {
                //cout<<"hii";
                //cout<<i<<" "<<i+1<<"\n";
                cnt++;
            }
            //zero=0;
        }
        else
        {
            zero++;
        }
        if(((str[i]-48)*100+(str[i+1]-48)*10+(str[i+2]-48))%8==0)
        {
            //cout<<"hii";
            if((((str[i]-48)*100+(str[i+1]-48)*10+(str[i+2]-48))!=0)&&(str[i]!='0'))
            {
                //cout<<str[i]<<endl;
                //cout<<i-zero<<"\n";
                cnt=cnt+i-zero+1;
            }
            else if(((str[i]-48)*100+(str[i+1]-48)*10+(str[i+2]-48))!=0&&str[i]=='0')
            {
                //cout<<i<<endl;
                //cout<<zero<<endl;
                //cout<<i-zero+1<<"\n";
                cnt=cnt+i-zero+1;
            }
            else
            {
                //cout<<i<<" "<<zero<<endl;
                //cout<<i-zero+1<<"\n";
                cnt=cnt+i-zero+1;
            }
        }
    }
    printf("%lld\n",cnt);
    return 0;
}
