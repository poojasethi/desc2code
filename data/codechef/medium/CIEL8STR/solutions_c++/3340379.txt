#include <stdio.h>
const int MAX=1000001;
long long res;
char s[MAX];
bool ref_chars[10][10][10]={0};
void init()
{
	for (int i = 0; i < 1000; i+=8)
	{
		int j=i,a,b;
		a=j%10;j/=10;
		b=j%10;j/=10;
		ref_chars[a][b][j]=true;
	}
}
int main()
{
	init();
	int i=0,j=0,a,b,c;
	res=0;
	gets(s);

	s[i]-='0';
	if(ref_chars[s[i++]][0][0]) res++;

	if(s[i]!=0) 
	{
		s[i]-='0';
		if(ref_chars[s[i]][0][0]) res++;
		if(s[i-1]!=0&&ref_chars[s[i]][s[i-1]][0]) res++;
		i++;
	}

	while (s[i]!=0)
	{
		s[i]-='0';
		a=s[i];
		b=s[i-1];
		c=s[i-2];
		if(c!=0)j++;
		if((a&1)==0)
		{
			if(ref_chars[a][0][0]) res++;
			if(b!=0&&ref_chars[a][b][0]) res++;
			if(ref_chars[a][b][c]) res+=j;
		}
		i++;
	}
	printf("%lld\n",res);
}
