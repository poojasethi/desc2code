void sint(int &x)
{
	register int c = gc();
	x = 0;
	for(;(c<48 || c>57);c = gc());
	for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

void sll(LL &x)
{
	register LL c = gc();
	x = 0;
	for(;(c<48 || c>57);c = gc());
	for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

LL dp[2][10];

VI mp[10];

char s[1000100];

int main()
{
	int t,prv,cur,l;
	LL sum=0;
	for(int j=0;j<8;++j) mp[ (j*10)%8].pb(j);
	scanf("%s",s);
	l = strlen(s);
	CLR(dp);
	t=0;
	dp[t][(s[0]-48)%8]=1;
	sum += dp[t][0];
	t=1;
	for(int i=1;i<l;++i){
		cur = (s[i]-48)%8;
		dp[t][cur]=1;
		for(int j=0;j<8;++j) {
			prv = (j - cur + 8)%8;
			for(int k=0;k<mp[prv].size();++k){
				dp[t][j] += dp[1-t][mp[prv][k]];
			}
		}
		if ( s[i-1]=='0') dp[t][(s[i]-48)%8] --;
		sum += dp[t][0];
		t = !t;
		CLR(dp[t]);
	}
	pl(sum);
	
	return 0;
}
