#include<cstdio>
#include<cmath>
#include<cstring>
#include <iostream>
using namespace std;
#define N 10000005

char s[N];

int strlenn(char *ss)
{
	int i=0;
	while(s[i++]);
	return i-1;

}

int main()
{
	int i,j,k,n;
	scanf("%s",s);
	n=strlenn(s);
	long long res=0;

	//calculate answer for the single digit.
	for(i=0;i<n;i++)
	{
		k=s[i]-'0';
		if(k%8==0)
			res++;
	}
	//calculate answer for the double digit.
	for(i=1;i<n;i++)
	{
		k=(s[i-1]-'0')*10+(s[i]-'0');
		if(k%8==0&&s[i-1]!='0')
			res++;
	}

	//calculate for three digits or more.
	int no_of_nonzero=0;
	for(i=2;i<n;i++)
	{
		k=(s[i-2]-'0')*100+(s[i-1]-'0')*10+(s[i]-'0');
		if(s[i-2]!='0') no_of_nonzero++;
		if(k%8==0) res+=no_of_nonzero;
	}
	printf("%lld\n",res);
    return 0;
}
