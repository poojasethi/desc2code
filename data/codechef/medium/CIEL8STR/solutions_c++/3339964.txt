#include <stdio.h>
const int MAX=1000001;
long long res;
char s[MAX];
bool ref_chars[10][10][10]={0};
void init()
{
	for (int i = 0; i < 1000; i+=8)
	{
		int j=i,a,b;
		a=j%10;j/=10;
		b=j%10;j/=10;
		ref_chars[a][b][j]=true;
	}
}
int main()
{
	init();
	int i=0,j=0,k,a,b,c;
	res=0;
	gets(s);
	while (s[i]!=0)
	{
		s[i]-='0';
		i++;
	}
	int length=i;
	for (i = 0; i < length; i++)
	{
		a=s[i];
		if(a!=0) j++;
		if((a&1)==0)
		{
			k=j;
			if(a!=0) k--;
			if(ref_chars[a][0][0]) res++;
			if(i>0)
			{
				b=s[i-1];
				if(b!=0)
				{
					k--;
					if(ref_chars[a][b][0]) res++;
				}
				if(i>1)
				{
					c=s[i-2];
					if(ref_chars[a][b][c])
					{
						res+=k;
					}
				}
			}
		}
	}
	printf("%lld\n",res);
}
