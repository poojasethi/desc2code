int main()
{
    ios_base::sync_with_stdio(false);
    int len, num;
    lli count = 0;
    char str[1000010];
    cin >> str;
    len = strlen(str);
    FOR(i, 0, len)
    {
        num = str[i] - '0';
        if(num % 8 == 0)
            count++;
    }
    FOR(i, 1, len)
    {
        if(str[i-1] == '0')
            continue;
        num = (str[i-1] - '0') * 10 + (str[i] - '0');
        if(num % 8 == 0)
            count++;
    }
    lli nonzero = 0;
    FOR(i, 2, len)
    {
        num = (str[i-2] - '0') * 100 + (str[i-1] - '0') * 10 + (str[i] - '0');
        if(str[i-2] != '0')
            nonzero++;
        if(num % 8 == 0)
            count += nonzero;
    }
    cout << count << endl;
    return 0;
}
