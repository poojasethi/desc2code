#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e6 + 10;
typedef long long LL;
LL ways[MAXN];
int main(){
	string s;
	cin >> s;
	int N = s.size();
	s = "@" + s;
	ways[0] = 0;
	LL ans = 0;
	for(int i = 1;i <= N;++i){
		ways[i] = ways[i - 1];
		if(s[i] > '0')
			ways[i] += 1;
		int num = s[i] - '0';
		if(num % 8 == 0) ++ans;
		if(i - 1 > 0){
			num = (s[i - 1] - '0') * 10 + num;
			if(num % 8 == 0 && s[i - 1] != '0') ++ans;
		}
		if(i - 2 > 0){
			num = (s[i - 2] - '0') * 100 + num;
			if(num % 8 == 0) {
				if(s[i - 2] != '0')
					++ans;
				ans += ways[i - 3];
			}
		}
		//cout << i << " " << ans << " " << ways[i] << "\n";
	}
	cout << ans << "\n";


	return 0;
}
