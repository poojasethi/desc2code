#include<cstdio>
#include<cstring>
char str[1000005];
int zerosbef[1000005]={0};
int main()
{
    
    scanf("%s",str);
    int len=strlen(str),num,k=0;
    long long int count=0;
    for(int i=0;i<len;i++)
    {
            zerosbef[i]=k;
            if(str[i]=='0')
            k++;
            
    }
    //printf("%d\n",k);
    for(int i=len-1;i>=0;--i)
    {
            
            if( (str[i]-'0')%8==0 ) 
              count++;
            
            
            //Check 2 digits
            if(i+1<len)
            {
                       num=(str[i]-'0')*10+str[i+1]-'0';
                       if(num%8==0 && num!=0 && str[i]!='0')
                       count++;
                       
            }
            //CHeck 3 digits
            if(i+2<len)
            {
                       num=(str[i]-'0')*100+(str[i+1]-'0')*10+str[i+2]-'0';
                       if(num%8==0 && num!=0 && str[i]!='0')
                       {
                                   count++;
                       }
                       if(num%8==0)
                       {
                                   count=count+i-zerosbef[i];
                                  // printf("0bfri %d\n",zerosbef[i]);
                       }
            }
                       
    }
    
    
    printf("%lld\n",count);
   //scanf("%d",&num);
}
