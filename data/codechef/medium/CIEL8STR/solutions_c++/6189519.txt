#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int i,z=0;
	char s[1000005];
	cin>>s;
	unsigned long long count=0;
	long long d=0;
	for(i=0;i<strlen(s);i++)
	{
		if(s[i]-'0'==8 || s[i]=='0')
		count++;
		if(i<strlen(s)-1)
		{
			if(s[i]!='0')
			{
				if(((s[i]-48)*10+(s[i+1]-48))%8 == 0)
				count++;
			}
		}
		if(i<strlen(s)-2)
		if(((s[i]-'0')*100+(s[i+1]-'0')*10+(s[i+2]-'0'))%8==0)
		{
			if(s[i]-48 !=0)
			count++;
			count+=d;
		}
		if(s[i]-48!=0)
		d++;
	}
	cout<<count<<"\n";
	return 0;
}
