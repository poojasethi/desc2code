#include <stdio.h>
 
char s[1000001];
int main(){
	unsigned long long r = 0;
	int i = 0, j, n, d = 0;
	gets(s);
	while(s[i] != '\0'){
		if(s[i] == '8' || s[i] == '0')
			r++;
		if(i > 0){
			n = (s[i]-48) + (s[i-1]-48)*10;
			if(n >= 10 && n % 8 == 0)
				r++;
			if(i > 1){
				if(i > 2 && s[i-3] != '0')
					d++;
				n += (s[i-2]-48)*100;
				if(n % 8 == 0){
					if(n >= 100)
						r++;
					r += d;
				}
			}
		}
		i++;
	}
	printf("%llu\n",r);
	return 0;
}
