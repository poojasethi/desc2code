#include<iostream>
using namespace std;
#include<cstdio>
#include<cstring>
int main()
{char str[1000001];
int num;
scanf("%s",str);
int length=strlen(str);
unsigned long long answer=0;
for(int i=0;i<length;i++)
  {
  	num=str[i]-'0';
  	if(num%8==0) answer++;
  	
  }
  for(int i=1;i<length;i++)
  {
  	if(str[i-1]-'0'==0) continue;
  	num=(str[i-1]-'0')*10+str[i]-'0';
  	if(num%8==0) answer++;
  }
  int notzero=0;
  for(int i=2;i<length;i++)
  {
  	if(str[i-2]-'0'!=0) notzero++;
  	num=(str[i-2]-'0')*100+(str[i-1]-'0')*10+(str[i]-'0');
  	if(num%8==0) answer+=notzero;
  	
  }
  printf("%llu\n",answer);
  
	
}
