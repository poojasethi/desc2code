typedef long long LL;

#define reset(a,b) memset(a, b, sizeof a)

string s;
LL ans, occ[1000003],satu , dua, tiga;

int main()
{
    cin>>s;
    reset(occ,0);
    
    occ[0] = (s[0]=='0'?0:1);
    for (int i = 1; i<s.length(); i++) occ[i] = occ[i-1] + (s[i]=='0'?0:1);
    
    ans = 0;
    for (int i = 0; i<s.length(); i++)
    {
        //cek 1 angka;
        satu = (s[i]-'0');
        if (satu%8==0) 
        {
                       //cout<<s[i]<<endl;
                       ans++;
        }
        
        //cek 2 angka;
        if (i+1 <s.length())
        {
                dua = (s[i]-'0')*10 + (s[i+1]-'0');
                if (dua>9 && dua%8==0) 
                {
                          //cout<<s[i]<<s[i+1]<<endl;
                          ans++;
                }
        }
        
        //cek 3 angka;
        if (i+2<s.length())
        {
                tiga = (s[i]-'0')*100 + (s[i+1]-'0')*10 + (s[i+2]-'0');
                if (tiga>99 && tiga%8==0) 
                {
                            //cout<<s[i]<<s[i+1]<<s[i+2]<<endl;
                            ans++;
                }
                
                if (tiga%8==0)  ans += (i==0?0:occ[i-1]);
        }
    }
    cout<<ans<<endl;
    
    //system("pause");
    return 0;
}
