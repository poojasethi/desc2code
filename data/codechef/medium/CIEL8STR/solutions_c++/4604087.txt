#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
	char str[1000001];
	scanf("%s", str);
	int length = strlen(str), zeros = 0;
	long long answer = 0;
	for(int i = 0; i < length; i++)
	{
		if(str[i] == '0' || str[i] == '8')
		{
			answer++;
			if(str[i] == '0')
			{
				zeros++;
			}
		}
		if(i < length-1 && str[i] != '0')
		{
			int temp = 10*(str[i]-'0') + (str[i+1]-'0');
			if(!(temp%8))
			{
				answer++;
			}
		}
		if(i < length-2)
		{
			int temp = 100*(str[i]-'0') + 10*(str[i+1]-'0') + (str[i+2]-'0');
			if(!(temp%8))
			{
				answer += i+1-zeros;
			}
		}
	}
	printf("%lld\n", answer);
    return 0;
}
