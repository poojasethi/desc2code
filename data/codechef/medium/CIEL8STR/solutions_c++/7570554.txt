#include<stdio.h>
#include<string.h>
char s[1000005];
long long int a[1000005];

int main()
{
	long long int count=0,l,n,k=0;
	int i,j;
	scanf("%s",s);
	l=strlen(s);
	
	for(i=0;i<l;i++)
		if(s[i]=='0' || s[i]=='8')
			count++;
	
	for(i=0;i<l-1;i++)
	{
		if((((s[i]-'0')*10)+(s[i+1]-'0'))%8==0 && s[i]!='0')
			count++;
	}
	
	for(i=0;i<l-2;i++)
	{
		n=((s[i]-'0')*100)+((s[i+1]-'0')*10)+(s[i+2]-'0');
		if(s[i]!='0')
			k++;
		if(n%8==0)
			count+=k;
	}
	
	printf("%lld\n",count);
	return 0;
}
