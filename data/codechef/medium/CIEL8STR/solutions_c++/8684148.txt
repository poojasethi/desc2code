#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define MX 1000009
int main()
{
	ll ans=0;
	char s[MX];
	scanf("%s",s);
	int i,num_zr=0,n=strlen(s);
	if(n>1)
	{
		for(i=0;i<n-2;i++)
		{
			if(s[i]=='0')
				num_zr++;
			char c[3];
			strncpy(c,s+i,3);
			int m=atoi(c);
			if(m%8==0)
				ans+=i-num_zr+1;
			m=m/10;
			if(s[i]!='0' && m%8==0)
				ans++;
			m=m/10;
			if(m%8==0)
				ans++;
		}
		char c[2];
		strncpy(c,s+i,2);
		int m=atoi(c);
		if(s[i]!='0' && m%8==0)
			ans++;
		if((m/10)%8==0)
			ans++;
		if((m%10)%8==0)
			ans++;
	}
	else if(s[0]=='0' || s[0]=='8')
		ans=1;
	printf("%lld\n", ans);
	return 0;
}
