#include<iostream>
#include<string.h>
using namespace std;
int arr[1000005];
int main(void)
{
    char s[1000005];

    cin>>s;
    int k,j,l,one=0,two=0,i,len=strlen(s),zero=0;
    long long int ans,three=0;

    for(i=0;i<len;++i)
    {
        if(s[i]=='0')
        {
            zero++;
            one++;
            arr[i]=zero;
            continue;
        }
        arr[i]=zero;

        if(s[i]=='8')
            one++;

        if(s[i]!='0' && i<len-1)
        {
            k=s[i]-'0';
            j=s[i+1]-'0';
            k=k*10+j;
            if(k%8==0)
                two++;
        }
    }

    for(i=len-1;i>=2;--i)
    {
        l=s[i]-'0';
        k=s[i-1]-'0';
        j=s[i-2]-'0';

        j=j*100+k*10+l;
        if(j%8==0)
            three+=i-1-arr[i-2];
    }

    ans=one+two+three;
    cout<<ans<<endl;
    return 0;
}
