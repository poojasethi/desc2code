#include <cstdio>
#include <cstring>
#include <cmath>

#include <vector>

using namespace std;

int dp[110][110], n, m, p, w, cost[110], Time[110];

int cal( int i, int total)
{
    if(i==n) return 0;
    else if( dp[i][total]!=-1) return dp[i][total];

    int pft1=0, pft2=0;
    if(Time[i]<=total)
        pft1 = cost[i] + cal(i+1, total-Time[i]);
    pft2 = cal( i+1, total);
    return dp[i][total]=max( pft1, pft2);
}
int main()
{
    int i, j, k, t, max2;
    scanf("%d", &t);
    while(t--)
    {
        scanf("%d %d", &n, &w);
        for( i=0; i<n; i++)
        {
            scanf("%d %d %d", &m, &p, &Time[i]);
            cost[i]=m*p;
        }
        memset( dp, -1, sizeof dp);
        max2=cal(0, w);
        printf("%d\n", max2);
    }
    return 0;
}
