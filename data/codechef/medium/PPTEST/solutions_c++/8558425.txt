#include<stdio.h>
int max(int a,int b)
{
	if(a>b)
	return a;
	else
	return b;
}
int main()
{
	int T;
	scanf("%d",&T);
    while(T--)
{
    int n,w,a,b,i,j;
    int f[101][101]={0};
    int t[101];
    int p[101];
    scanf("%d%d",&n,&w);
    for(i=1;i<=n;i++)
    {
    	scanf("%d%d%d",&a,&b,&t[i]);
    	p[i]=a*b;
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=w;j++)
		{
			if(t[i]<=j)
			f[i][j]=max(f[i-1][j],f[i-1][j-t[i]]+p[i]);
			else
			f[i][j]=f[i-1][j];
		}
	}
	printf("%d\n",f[n][w]);
}
	return 0;
}
