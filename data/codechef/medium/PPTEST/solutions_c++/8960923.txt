#include<bits/stdc++.h>
using namespace std;
int maxim(int a, int b)
{
	return (a<b)?b:a;
}
int main(void)
{
	int tt;
	scanf("%d", &tt);
	while(tt--)
	{
		int n, w;
		scanf("%d %d", &n, &w);
		int c[101], p[101], t[101];
		for(int i=0; i<n; i++)
			scanf("%d %d %d", &c[i], &p[i], &t[i]);
		int dp[w+1][n+1];
		dp[0][0]=0;
		for(int i=1; i<=n; i++)
			dp[0][i]=0;
		for(int i=1; i<=w; i++)
			dp[i][0]=0;
		for(int i=1; i<=w; i++){
			for(int j=1; j<=n; j++){
				dp[i][j]=(i>=t[j-1])?(maxim(dp[i-t[j-1]][j-1]+(c[j-1]*p[j-1]), dp[i][j-1])):(dp[i][j-1]);
				//printf("%d ", dp[i][j]);
			}
			//cout<<endl;
		}
		printf("%d\n", dp[w][n]);
	}
	return 0;
}
