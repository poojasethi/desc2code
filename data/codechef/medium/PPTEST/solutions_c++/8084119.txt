#include <bits/stdc++.h>

using namespace std ;

int dp[110][110], n, a[110], b[110], c[110], w ;

int call(int i, int s)
{
    if( i == n )
        return 0 ;
    if( dp[i][s] != -1 )
        return dp[i][s] ;
    int fst = 0, scnd = 0 ;
    if( s + c[i] <= w )
        fst = ( a[i] * b[i] ) + call(i + 1, s + c[i]) ;
    scnd = call(i + 1, s) ;
    return dp[i][s] = max(fst, scnd) ;
 }


int main()
{
//    freopen("in.txt", "r", stdin) ;
    int i, j, k, l, tst ;
    scanf("%d", &tst) ;
    while(tst--)
    {
        scanf("%d %d", &n, &w) ;
        for( i = 0; i < n; i++ )
            scanf("%d %d %d", &a[i], &b[i], &c[i]) ;
        memset(dp, -1, sizeof dp) ;
        printf("%d\n", call(0,0) ) ;
    }
    return 0;
}
