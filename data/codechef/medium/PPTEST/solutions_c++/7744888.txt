#define Max 105

typedef long long LL;
typedef unsigned long long ULL;

using namespace std;

struct node
{
    int c,p,t;
};
node ara[Max];
int n,w;
int visit[Max][Max],dp[Max][Max],CS;
int fun(int pos,int sum)
{
    if(pos > n)return 0;

    if(visit[pos][sum] == CS)return dp[pos][sum];

    int p1 = 0,p2 = 0;
    if(sum + ara[pos].t <= w)p1 = ara[pos].c * ara[pos].p + fun(pos+1,sum+ara[pos].t);
    p2 = fun(pos+1,sum);

    visit[pos][sum] = CS;
    return dp[pos][sum] = max(p1,p2);
}

int main()
{
    int t,T,i;
    si(T);
    rep(t,T)
    {
        sii(n,w);
        rep(i,n)
        {
            siii(ara[i].c,ara[i].p,ara[i].t);
        }
        CS++;
        int ans = fun(1,0);
        pi(ans);NL;
    }
    return 0;
}
