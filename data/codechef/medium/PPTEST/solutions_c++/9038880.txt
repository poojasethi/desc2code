#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,w;
		scanf("%d%d",&n,&w);
		int c[101],p[101],t[101],i,dp[101][101],j;
		for(i=1;i<=n;i++)
		{
			scanf("%d%d%d",&c[i],&p[i],&t[i]);
			p[i]=p[i]*c[i];
		}
		for(i=0;i<=max(n,w);i++)
		{
			dp[i][0]=0;
			dp[0][i]=0;
		}
		for(i=1;i<=n;i++)
		{
			for(j=0;j<=w;j++)
			{
				if(t[i]<=j)
				dp[j][i]=max((p[i]+dp[j-t[i]][i-1]),dp[j][i-1]);
				else
				dp[j][i]=dp[j][i-1];
			}
		}
		printf("%d\n",dp[w][n]);
	}
	return 0;
}
