#include <iostream>

using namespace std;

int main()
{
    long long int t,n,W;
    cin>>t;
    while(t--)
    {

        cin>>n>>W;
        long long int c[n],p[n],t[n],i,j;
        for(i=0;i<n;++i)
        cin>>c[i]>>p[i]>>t[i];
        long long int dp[n+1][W+1];
        for(i=0;i<=n;++i)
        {
            for(j=0;j<=W;++j)
            {
                if(i==0 || j==0)
                {
                    dp[i][j]=0;
                }
                else if(t[i-1]>j)
                {
                    dp[i][j]=dp[i-1][j];
                }
                else
                {
                    dp[i][j]=max(dp[i-1][j],dp[i-1][j-t[i-1]]+p[i-1]*c[i-1]);
                }
            }
        }
        cout<<dp[n][W]<<"\n";
    }
    return 0;
}
