#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,w,p,q,i,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&w);
        int a[n],b[n],t[n+1][w+1];
        for(i=1;i<=n;i++)
        {
            scanf("%d%d%d",&p,&q,&a[i]);
            b[i]=p*q;
        }
        for(i=0;i<=w;i++)
            t[0][i]=0;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<=w;j++)
            {
                if(j>=a[i])
                    t[i][j]=max(t[i-1][j-a[i]]+b[i],t[i-1][j]);
                else
                    t[i][j]=t[i-1][j];
            }
        }
        printf("%d\n",t[n][w]);
    }
    return 0;
}
