#include <bits/stdc++.h>
 using namespace std;

int c[110],t[110], dp[110][110];
int main(){
    int tc; cin >> tc;
	while(tc--){
		int n,w; cin >> n >> w;
		memset(dp,0,sizeof dp);
		for(int i=1 ; i<=n ; i++){
			int x,y; cin >> x >> y >> t[i];
			c[i] = x*y;
		}
		for(int i=1 ; i<=n ; i++){
			for(int j=0 ; j<=w ; j++){
				if(j<t[i]) dp[i][j] = dp[i-1][j];
				else dp[i][j] = max(dp[i-1][j], dp[i-1][j-t[i]] + c[i]);
			}
		}
		cout << dp[n][w] << endl;
	}
	return 0;
}
