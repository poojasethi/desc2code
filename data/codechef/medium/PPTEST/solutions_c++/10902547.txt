#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n,w;
		scanf("%d%d",&n,&w);
		int time[n+1];
		int weight[n+1];
		for (int i = 1; i <= n; i++)
		{
			int x,y,z;
			scanf("%d%d%d",&x,&y,&z);
			weight[i] = x*y;
			time[i] = z;
		}
		int dp[n+1][w+1];
		memset(dp,-1,sizeof(dp));
		int ans = 0;
		for (int i = 0; i <= n; i++)
		{
			for (int j = 0; j <= w; j++)
			{
				if (i == 0 || j == 0)
					dp[i][j] = 0;
				else
				{
					if (time[i] <= j)
					{
						dp[i][j] = max(dp[i-1][j-time[i]]+weight[i],dp[i-1][j]);
						ans = max(ans,dp[i][j]);
					}
					else
						dp[i][j] = dp[i-1][j];
				}
			}
		}
		cout << ans << endl;
	}
	return 0;
}
