#include<bits/stdc++.h>
using namespace std;
int c[110],p[110],t[110];
int dp[110][110];
int main()
{
	int test;
	cin>>test;
	while(test--)
	{
		int n;
		cin>>n;
		int w;
		cin>>w;
		memset(dp,0,sizeof(dp));
		for(int i=1;i<=n;i++)
		{
			cin>>c[i]>>p[i]>>t[i];
		}
		for(int i=1;i<=n;i++)
		{
			  for(int j=1;j<=w;j++)
			  {
			  	dp[i][j]=dp[i-1][j];
			  	   if(t[i]<=j)
			  	   {
			  	   	      dp[i][j]=max(dp[i-1][j],dp[i-1][j-t[i]]+p[i]*c[i]);
				   }
			  }
		}
		int ans=0;
		//cout<<"w si "<<w<<endl;
		for(int  i=1;i<=w;i++)
		{
		//	cout<<"done "<<endl;
		//	cout<<dp[n][i]<<endl;
			ans=max(ans,dp[n][i]);
		}
		cout<<ans<<endl;
	}
	return 0;
}
