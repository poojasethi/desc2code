#include<iostream>
#include<cstring>
using namespace std;

int main(){

    ios_base::sync_with_stdio(false);
    cin.tie(0);

    int w;
    cin>>w;
    while(w--){
        int n,pot,i,j;
        cin>>n>>pot;
        int tests[n],pointsArray[n],timeArray[n];
        for(i=0;i<n;i++)
            cin>>tests[i]>>pointsArray[i]>>timeArray[i];
        int dp[n][pot+1];

        memset(dp,0,sizeof dp);

        for(i=0;i<=pot;i++){
            if(i>=timeArray[0])
                dp[0][i]=pointsArray[0]*tests[0];
        }

        for(i=1;i<n;i++){
            for(j=0;j<=pot;j++){
                if(timeArray[i]>j)
                    dp[i][j]=dp[i-1][j];
                else
                    dp[i][j]=max((dp[i-1][j-timeArray[i]]+tests[i]*pointsArray[i]),dp[i-1][j]);

            }
        }
        cout<<dp[n-1][pot]<<endl;
   }

    return 0;
}
