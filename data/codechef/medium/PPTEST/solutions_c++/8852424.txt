int main(int argc, char const *argv[])
{
    int t;
    sc(t);
    while(t--) {
        int n,m;
        sc(n); sc(m);
        int c[n+3], p[n+3], t[n+3];
        for (int i = 1; i <= n; ++i)
        {
            sc(c[i]); sc(p[i]); sc(t[i]);
        }

        int dp[105][105];
        fill(dp,0);
        for (int i = 1; i <= m; ++i)
        {
            for (int j = 1; j <= n; ++j)
            {
                if(i >= t[j]) {
                    dp[i][j] = max(dp[i][j-1] , (c[j]*p[j])+dp[i-t[j]][j-1]);
                }
                else dp[i][j] = dp[i][j-1];
            }
        }

        pf(dp[m][n]); pn;
    }
    return 0;
}
