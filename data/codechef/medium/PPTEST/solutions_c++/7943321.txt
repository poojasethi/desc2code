#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,w,i,j;
        scanf("%d %d",&n,&w);
        int c[n],p[n],T[n],knapsack[n+1][w+1];
        for(i=0;i<n;i++)
            scanf("%d %d %d",&c[i],&p[i],&T[i]);
        for(i=0;i<w+1;i++)
            knapsack[0][i]=0;
        for(i=1;i<n+1;i++)
        {
            for(j=0;j<w+1;j++)
            {
                if(j<T[i-1])
                    knapsack[i][j]=knapsack[i-1][j];
                else
                    knapsack[i][j]=max(((c[i-1]*p[i-1])+knapsack[i-1][j-T[i-1]]),knapsack[i-1][j]);
            }
        }
        printf("%d\n",knapsack[n][w]);
    }
    return 0;
}
