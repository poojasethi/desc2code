#include <bits/stdc++.h>
using namespace std;

int maxmarks(int marks[], int time[], int n, int total)
{
	int i,t,dp[n+1][total+1];
	for(i=0;i<=n;i++)
	{
		for(t=0;t<=total;t++)
		{
			if(i==0||t==0)
				dp[i][t]=0;
			else if(t<time[i-1])
				dp[i][t]=dp[i-1][t];
			else
				dp[i][t]=max(dp[i-1][t], marks[i-1]+dp[i-1][t-time[i-1]]);
		}
	}
	return dp[n][total];
}

int main() 
{
	int t,n,w;
	cin>>t;
	while(t--)
	{
		cin>>n>>w;
		int c[n],p[n],t[n];
		for(int i=0;i<n;i++){cin>>c[i]>>p[i]>>t[i];p[i]*=c[i];}
		cout<<maxmarks(p,t,n,w)<<endl;
	}
	return 0;
}
