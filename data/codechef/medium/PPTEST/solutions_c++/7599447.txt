#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    int x,n,a[102],b[102],t[102],w,i,j,k,d[102][102];
    sd(x);
    while(x--)
    {
        sd(n);
        sd(w);
        for(i=0;i<n;i++)
        {
            sd(a[i]);
            sd(b[i]);
            sd(t[i]);
            a[i]*=b[i];
        }
        for(i=0;i<=n;i++)
        {
            for(j=0;j<=w;j++)
            {
                if(i==0||j==0)
                    d[i][j]=0;
                else if(t[i-1]<=j)
                    d[i][j]=max(d[i-1][j],a[i-1]+d[i-1][j-t[i-1]]);
                else
                    d[i][j]=d[i-1][j];
            }
        }
        printf("%d\n",d[n][w]);
    }
    return 0;
}
