using namespace std;
int dp[110][110];
int c[110],p[110],tt[110];
int main(int argc, char const *argv[])
{
int t;int n,w;
cin>>t;
while(t--)
{memset(dp,0,sizeof(dp));
cin>>n>>w;
 for (int i = 1; i <= n; ++i)
 {
   cin>>c[i]>>p[i]>>tt[i];
   c[i]*=p[i];
 }
 
 for (int i = 1; i <= n; ++i)
 {  
 	for (int j = 1; j <= w; ++j)
 	{  dp[i][j] = dp[i-1][j];
 		if(j-tt[i]>=0)
 			dp[i][j] = max(dp[i-1][j-tt[i]]+c[i],dp[i][j]);
 	}
 }
 cout<<dp[n][w]<<"\n";
}
}
