using namespace std;

#define MEM(a,b) memset(a,(b),sizeof(a))
#define MAX(a,b) ((a) > (b) ? (a) : (b))
#define MIN(a,b) ((a) < (b) ? (a) : (b))
#define MP make_pair
#define pb push_back


#define inf 1000000007
#define maxn 100


typedef long long  LL;

int best[105];

int main()
{
   int i,j,k,T;
   scanf("%d",&T);
   assert(T>=1 && T<=maxn);

   while(T--)
   {
      int n,w;
      scanf("%d%d",&n,&w);
      assert(n>=1 && n<=maxn && w>=1 && w<=maxn);

      for(i=1;i<=w;i++) best[i]=0;
      best[0]=0;

      for(i=0;i<n;i++)
      {
         int t,c,p;
         scanf("%d%d%d",&c,&p,&t);
         assert(c>=1 && c<=maxn);
         assert(p>=1 && p<=maxn);
         assert(t>=1 && t<=maxn);

         for(j=w;j>=t;j--)
            if(best[j-t]>=0)
               best[j]=MAX(best[j],best[j-t]+p*c);
      }

      int ans=0;
      for(i=1;i<=w;i++) ans=MAX(ans,best[i]);
      printf("%d\n",ans);
   }

	return 0;
}
