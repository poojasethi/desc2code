#include<stdio.h>

int main()
{
    int tc,n,w,c,p,t,i,j;
    long int a[101],hi;

    scanf("%d",&tc);

    while(tc--)
    {
        scanf("%d%d",&n,&w);

        for(i=0;i<=w;i++)
            a[i] = -100;
        a[0] = 0;
        for(i=0;i<n;i++)
        {
            scanf("%d%d%d",&c,&p,&t);
            for(j=w;j>=t;j--)
                if(a[j-t] >=0)
                    a[j] = a[j]>=a[j-t]+c*p?a[j]:a[j-t]+c*p;
        }
        hi = 0;
        for(i=0;i<=w;++i)
            if(a[i] > hi)
                hi = a[i];
        printf("%ld\n",hi);
    }
    return 0;
}
