#include<iostream>
#include<cstdio>
#include<map>
using namespace std;
int knapSack(int W, int wt[], int val[], int n)
{
   int i, w;
   int K[n+1][W+1];

   for (i = 0; i <= n; i++)
   {
       for (w = 0; w <= W; w++)
       {
           if (i==0 || w==0)
               K[i][w] = 0;
           else if (wt[i-1] <= w)
                 K[i][w] = max(val[i-1] + K[i-1][w-wt[i-1]],  K[i-1][w]);
           else
                 K[i][w] = K[i-1][w];
       }
   }
 
   return K[n][W];
}
int main(){
int t,n,c,p,w;
scanf("%d",&t);
while(t--){
scanf("%d %d",&n,&w);
int val[102],t[102];
for(int i=0;i<n;i++){
scanf("%d %d %d",&c,&p,&t[i]);
val[i]=c*p;
}
printf("%d\n",knapSack(w,t,val,n));
}
}
