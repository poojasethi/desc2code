#include <bits/stdc++.h>

using namespace std;

int n,w;
int c[102];
int p[102];
int t[102];
int dp[102][102];

int rec(int index, int sum)
{
	if(index==n)
	return 0;
	if(dp[index][sum]!=-1)
	return dp[index][sum];
	
	int ans=INT_MIN;
	if(sum+t[index] <= w)
	ans= rec(index+1,sum+t[index])+(c[index]*p[index]);
	ans= max(ans,rec(index+1,sum));
	return dp[index][sum]=ans;
}

int main()
{
int T;
cin>>T;
for(int i=0;i<T;i++)
{
	cin>>n;
	cin>>w;
	for(int j=0;j<n;j++)
	cin>>c[j]>>p[j]>>t[j];
	memset(dp,-1,sizeof(dp));
	cout<<rec(0,0)<<endl;
}
	return 0;
}
