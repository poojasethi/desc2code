void mul(ll mat[2][2],ll temp[2][2])
{
ll mat1,mat2,mat3,mat4;
mat1=mat[0][0]*temp[0][0]+mat[0][1]*temp[1][0];
mat2=mat[0][0]*temp[0][1]+mat[0][1]*temp[1][1];
mat3=mat[1][0]*temp[0][0]+mat[1][1]*temp[1][0];
mat4=mat[1][0]*temp[0][1]+mat[1][1]*temp[1][1];
mat[0][0]=mat1%1000000007;
mat[0][1]=mat2%1000000007;
mat[1][0]=mat3%1000000007;
mat[1][1]=mat4%1000000007;
}
void matexp(ll mat[2][2],int n)
{
if(n==1 || n==0)
return;
ll temp[2][2]={{2,2},{1,0}};
matexp(mat,n/2);
mul(mat,mat);
if(n%2!=0)
mul(mat,temp);
}
int main()
{
int t;
si(t);
while(t--)
{
ll n;
sll(n);
if(n==1)
{
printf("1\n");

ll mat[2][2]={{2,2},{1,0}};
matexp(mat,n-1);

nl;
}
return 0;
}
