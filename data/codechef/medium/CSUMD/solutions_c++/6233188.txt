#include<cstdio>
#include<iostream>
using namespace std;
#define ll long long
#define rl(x) scanf("%lld",&x)
#define MOD (ll)1000000007
void pow(ll a[][2],ll b[][2])
{
	ll c[2][2];
	for(int i=0;i<2;i++)
	    for(int j=0;j<2;j++)
	    {	
		c[i][j]=0;
		for(int k=0;k<2;k++)
			c[i][j]+=(a[i][k]*b[k][j]);
	    }
	for(int i=0;i<2;i++)
	    for(int j=0;j<2;j++)
		a[i][j]=c[i][j]%MOD;
}
ll solve(ll n)
{
	ll f[2][2]={{2,2},{1,0}};
	ll id[2][2]={{1,0},{0,1}};
	ll u,v,x,y;
	while(n)
	{
		if(n&1)
		   pow(id,f);
		pow(f,f);
		n>>=1;
	}
	return ((id[0][0]*3)%MOD+(id[0][1])%MOD)%MOD;
}
int main()
{
	ll t;
	rl(t);
	while(t--)
	{
		ll n;
		rl(n);
		if(n==1)
		{
		printf("1\n");
		}
		else if(n==2)
		{
		printf("3\n");
		}
		else
		printf("%lld\n",solve(n-2));
	}
}
