#define DOW(i,b,a) for(int i=(b),_a=(a); i>=_a ;i--)

struct mat{
    ll f[2][2];
    mat()
    {
        memset(f,0,sizeof(f));
    }
};

mat operator*(mat a, mat b)
{
    mat c;
    FOR(i,0,1)
    FOR(j,0,1)
    FOR(k,0,1)
    {
        c.f[i][j]+=a.f[i][k]*b.f[k][j];
        c.f[i][j]=c.f[i][j]%modul;
    }
    return c;
}

mat c;
mat calc(int n)
{
    if (n==1) return c;
    mat p=calc(n/2);
    if (n%2==1) return p*p*c;
    else return p*p;
}
