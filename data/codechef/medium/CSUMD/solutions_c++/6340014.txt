     #include<cstdio>
    #include<iostream>
    using namespace std;
    #define ll long long
    #define rl(x) scanf("%lld",&x)
    #define MOD (ll)1000000007
    ll solve(ll n)
    {
    ll f[2][2]={{2,2},{1,0}};
    ll id[2][2]={{1,0},{0,1}};
    ll u,v,x,y;
    while(n)
    {
    if(n&1)
    {
    u=(f[0][0]*id[0][0]+f[0][1]*id[1][0])%MOD;
    v=(f[0][0]*id[0][1]+f[0][1]*id[1][1])%MOD;
    x=(f[1][0]*id[0][0]+f[1][1]*id[1][0])%MOD;
    y=(f[1][0]*id[0][1]+f[1][1]*id[1][1])%MOD;
    id[0][0]=u,id[0][1]=v,id[1][0]=x,id[1][1]=y;
    }
    u=(f[0][0]*f[0][0]+f[0][1]*f[1][0])%MOD;
    v=(f[0][0]*f[0][1]+f[0][1]*f[1][1])%MOD;
    x=(f[1][0]*f[0][0]+f[1][1]*f[1][0])%MOD;
    y=(f[1][0]*f[0][1]+f[1][1]*f[1][1])%MOD;
    f[0][0]=u,f[0][1]=v,f[1][0]=x,f[1][1]=y;
    n>>=1;
    }
    return ((id[0][0]*3)%MOD+(id[0][1])%MOD)%MOD;
    }
    int main()
    {
    ll t;
    rl(t);
    while(t--)
    {
    ll n;
    rl(n);
    if(n==1)
    {
    printf("1\n");
    //continue;
    }
    else if(n==2)
    {
    printf("3\n");
    //continue;
    }
    else
    printf("%lld\n",solve(n-2));
    }
    }
