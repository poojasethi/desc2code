#include<bits/stdc++.h>
using namespace std;
typedef long long int lli;
#define MOD 1000000007
void multiply(lli a[2][2],lli b[2][2],lli c[2][2])
{
	int i,j,k;
	for(i=0;i<2;i++)
	{
		for(j=0;j<2;j++)
		{
			c[i][j]=0;
			for(k=0;k<2;k++)
				c[i][j] += (a[i][k]*b[k][j])%MOD;
		}
	}
}

void power(lli a[2][2],lli n,lli rslt[2][2])
{
	rslt[0][0]=1;
	rslt[0][1]=0;
	rslt[1][0]=0;
	rslt[1][1]=1;
	lli temp[2][2];
	int i,j;
	while(n>0)
	{
		if(n&1)
		{
			multiply(rslt,a,temp);
			for(i=0;i<2;i++)
				for(j=0;j<2;j++)
					rslt[i][j] = temp[i][j];
		}
		multiply(a,a,temp);
		for(i=0;i<2;i++)
			for(j=0;j<2;j++)
				a[i][j] = temp[i][j];
		n=n/2;
	}
	return;
}

lli arrangements(lli n)
{
	if(n==1)
		return 1;
	else if(n==2)
		return 3;

	lli ans = (temp[0][0]*3 + temp[0][1]*1)%MOD;

}
