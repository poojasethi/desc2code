#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
struct mat
{
	long long int arr[2][2];
};
mat mat_mul(mat a,mat b)
{
	mat c;
	c.arr[0][0]=0;
	c.arr[0][1]=0;
	c.arr[1][0]=0;
	c.arr[1][1]=0;
	for(int i=0;i<2;i++)
	{
		for(int j=0;j<2;j++)
		{
			for(int k=0;k<2;k++)
			c.arr[i][k]=(c.arr[i][k]+a.arr[i][j]*b.arr[j][k])%mod;
		}
	}
	return c;
}
mat fast_expo(mat a,int n)
{
	mat a1;
	if(n==1)
	return a;
	else
	{
		if(n%2==0)
		{
			a1=fast_expo(a,n/2);
			 return mat_mul(a1,a1);
			
			
		}
		else
		{
			return mat_mul(a,fast_expo(a,n-1));
			
		}
	}
}

{
	long long int ans;

}
