void mul(int a[2][2],int b[2][2],int siz){
	int c[siz][siz];
	lli sum = 0;
	
	for(int i=0;i<siz;i++){
		for(int j=0;j<siz;j++){
			sum=0;
			for(int k=0;k<siz;k++){
				sum += ( ((long long)a[i][k])*( (long long)b[k][j] ) );
				sum = sum%mod;	
			}
			c[i][j]=sum;
		}	
	}
	
	for(int i=0;i<siz;i++){
		for(int j=0;j<siz;j++)
			a[i][j]=c[i][j];
	}
	return;
}

void solve(int n){
	int m[2][2];
	m[0][0]=2;
	m[0][1]=2;
	m[1][0]=1;
	m[1][1]=0;
	
	int id[2][2]={{1,0},{0,1}};
	n=n-2;
	while(n>0){
		if(n&1)
			mul(id,m,2);
		mul(m,m,2);
		n = n>>1;
	}
	lli ans = ((3LL*id[0][0])+id[0][1])%mod;
	printf("%lld\n",ans);
}

int main(){
	int t,n;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		if(n==1){
			printf("1\n");
			continue;
		}
		if(n==2){
			printf("3\n");
			continue;
		}
		solve(n);
	}
	
}
