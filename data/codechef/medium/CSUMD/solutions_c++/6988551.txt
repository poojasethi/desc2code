#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#define MODBY 1000000007
typedef long long int lld;
struct mat{
   lld a00,a01;
   lld a10,a11;
}matpower[34];
void printmat(struct mat a)
{
   printf("\t%lld",a.a00);
   printf("\t%lld\n",a.a01);
   printf("\t%lld",a.a10);
   printf("\t%lld\n",a.a11);
   printf("\n\n");
}
struct mat multiply(struct mat &x,struct mat &y)
{//x*y
   struct mat temp;
   temp.a00=(x.a00*y.a00+x.a01*y.a10)%MODBY;
   temp.a01=(x.a00*y.a01+x.a01*y.a11)%MODBY;
   temp.a10=(x.a10*y.a00+x.a11*y.a10)%MODBY;
   temp.a11=(x.a10*y.a01+x.a11*y.a11)%MODBY;
   return temp;
}
void init()
{
   int i;
   matpower[0].a00=matpower[0].a11=1;//identity matrix
   matpower[0].a10=matpower[0].a01=0;
   matpower[1].a00=matpower[1].a01=2;
   matpower[1].a10=1;
   matpower[1].a11=0;
   for(i=2;i<34;++i) matpower[i]=multiply(matpower[i-1],matpower[i-1]);
   //printmat(matpower[0]);
}
int main()
{
   int cases;
   int n;
   lld h1,h2;
   int power,cnt;
   struct mat calc;
   init();
   for(scanf("%d",&cases);cases;cases--){
      scanf("%d",&n);
      if(n==1){ printf("1\n");continue;}
      power=n-2;
      for(cnt=1,calc=matpower[0];power;power>>=1,++cnt)
         if(power&1) calc=multiply(calc,matpower[cnt]);
      h1=2*calc.a00+calc.a01;
      h2=2*calc.a10+calc.a11;
      printf("%lld\n",(h1+h2)%MODBY);
   }
   return 0;
}
