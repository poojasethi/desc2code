#include<iostream>
#define MOD 1000000007
using namespace std;
long long res[3][3],tmul[3][3],tpow[3][3],a[3][3];
void mat_mul(long long res[3][3],long long a[3][3],long long b[3][3])
{
	long long i,j,k,x;
	
	for(i=1;i<=2;i++)
	{
		for(j=1;j<=2;j++)
		{
			x=0;
			for(k=1;k<=2;k++)
			{
				x=x+(1ll*a[i][k]*b[k][j]);
				if(x>=MOD)
				x%=MOD;
			}
			
			tmul[i][j]=x;
		}
	}
	
	for(i=1;i<=2;i++)
	{
		for(j=1;j<=2;j++)
		{
			res[i][j]=tmul[i][j];
		}
	}
}
void mat_pow(long long res[3][3],long long a[3][3],long long p)
{
	int i,j;
	
	for(i=1;i<=2;i++)
	{
		for(j=1;j<=2;j++)
		{
			res[i][j]=(i==j);
			tpow[i][j]=a[i][j];
		}
	}
	
	while(p)
	{
		if(p&1)
		{
			mat_mul(res,res,tpow);
		}
		if(p>>=1)
		{
			mat_mul(tpow,tpow,tpow);
		}
	}
}
int main()
{
	long long t,n;
	
	cin>>t;
	
	while(t--)
	{
		cin>>n;
		
		if(n==1)
		cout<<1<<endl;
		else if(n==2) cout<<3<<endl;
		
		else
		{
			a[1][1]=0;
			a[1][2]=1;
			a[2][1]=2;
			a[2][2]=2;
			
			
			mat_pow(res,a,n-1);
			
			cout<<(res[1][1]+(res[1][2]*3)%MOD)%MOD<<endl;
		}
	}
	
	return 0;
}
