#include<iostream>
using namespace std;
#define mod 1000000007
void expo(unsigned long long int a[2][2],int exp)
{
    if(exp==1)
        return;
    else if(exp%2==0)
    {
        unsigned long long int p=a[0][0];
        unsigned long long int q=a[1][0];
        unsigned long long int r=a[0][1];
        unsigned long long int s=a[1][1];
        a[0][0]=(((p*p)%mod)+((r*q)%mod))%mod;
        a[1][0]=(((p*q)%mod)+((s*q)%mod))%mod;
        a[0][1]=(((p*r)%mod)+((r*s)%mod))%mod;
        a[1][1]=(((s*s)%mod)+((q*r)%mod))%mod;
        expo(a,exp/2);
    }
    else
    {
        unsigned long long int p=a[0][0];
        unsigned long long int q=a[1][0];
        unsigned long long int r=a[0][1];
        unsigned long long int s=a[1][1];
        a[0][0]=(((p*p)%mod)+((r*q)%mod))%mod;
        a[1][0]=(((p*q)%mod)+((s*q)%mod))%mod;
        a[0][1]=(((p*r)%mod)+((r*s)%mod))%mod;
        a[1][1]=(((s*s)%mod)+((q*r)%mod))%mod;
        expo(a,exp/2);
        unsigned long long int w=a[0][0]%mod;
        unsigned long long int y=a[1][0]%mod;
        unsigned long long int x=a[0][1]%mod;
        unsigned long long int z=a[1][1]%mod;
        a[0][0]=(((p*w)%mod)+((q*x)%mod))%mod;
        a[1][0]=(((p*y)%mod)+((z*q)%mod))%mod;
        a[0][1]=(((w*r)%mod)+((x*s)%mod))%mod;
        a[1][1]=(((z*s)%mod)+((y*r)%mod))%mod;
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        unsigned long long int a[2][2]={{2,1},{2,0}};
        cin>>n;
        if(n==1)
        {
            cout<<"1"<<endl;
            continue;
        }
        if(n==2)
        {
            cout<<"3"<<endl;
            continue;
        }
        expo(a,n-2);
        unsigned long long int m=3*a[0][0]+a[1][0];
        cout<<m%mod<<endl;
    }
    return 0;
}
