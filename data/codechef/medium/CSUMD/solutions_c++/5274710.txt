#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define MOD 1000000007
#define K 2
void matrix(ll a[2][2],ll b[2][2])
{
int i,j,k;
ll c[2][2];
for(i=0;i<2;i++)
{
for(j=0;j<2;j++)
{
c[i][j] = 0;
for(k=0;k<2;k++)
{
c[i][j] = (c[i][j]+(a[i][k]*b[k][j])%MOD)%MOD;
}

}
}
for(i=0;i<2;i++)
{
for(j=0;j<2;j++)
{
a[i][j] = c[i][j];
}
}
}
ll series(int n)
{
int i,j;
ll iden[2][2] = {{1,0},{0,1}};
ll z[2][2] = {{2,2},{1,0}};
n=n-2;
while(n)
{
if(n&1)
matrix(iden,z);
matrix(z,z);
n =n/2;
}
return (3*iden[0][0] + 1*iden[0][1] )%MOD;
}
int main()
{
int t,n;

 if(n == 1)
