#include <cstdio>

using namespace std;
FILE *f,*g;
long long v[12000];
struct cp{long long st,dr;};
cp a[12000];
double x,y;
long long t,mn,dr;
int n,i,q,k,p;
long long cat,rest,ax;

inline long long maxx(long long x, long long y)
{
    if (x>y) return x;
    return y;
}

int ok(long long x)
{
    a[1].st=maxx(0,v[1]-x);


    for (i=2;i<=n;i++)
    {
        if ((long long)a[i-1].st+t>(long long)v[i]+x)
            return 0;
        a[i].st=maxx(a[i-1].st+t,v[i]-x);
    }
    return 1;
}

void caut(long long i, long long j)
{
    long long m;
    if (i<=j)
    {
        m=(long long)(i+j)/2;
        if (ok(m))
        {
            if (m<mn || mn==-1)
                mn=m;
            caut(i,m-1);
        }
        else
            caut(m+1,j);
    }
}

inline int cif(char c)
{
    if (c>='0' && c<='9') return 1;
    return 0;
}

char c;

void read(long long &t)
{
    int i;
    scanf("%c",&c);
    while (c==' ') scanf("%c",&c);
    t=0;
    while (cif(c))
    {
        t=(long long)t*10+(c-'0');
        scanf("%c",&c);
    }

    if (c=='.')
        scanf("%c",&c);
    for (i=1;cif(c);i++)
        {
            t=(long long)t*10+(c-'0');
            scanf("%c",&c);
        }
    for (;i<=4;i++)
        t=(long long)t*10;

    while (c!='\n') scanf("%c",&c);
}

int main()
{

    for (i=1,dr=1;i<=18;i++)
        dr=(long long)dr*10;

    scanf("%d",&k);
    for (q=1;q<=k;q++)
    {
        scanf("%d",&n);
        read(t);

        for (i=1;i<=n;i++)
        {
            read(v[i]);
        }

        mn=-1;
        caut(0,dr);

        cat=mn/10000;
        rest=mn%10000;
        printf("%lld.",cat);
        for (p=0,ax=rest;ax;)
        {
          p++;
          ax/=10;
        }
        if (p==0) p=1;
        for (i=1;i<=4-p;i++) printf("0");
        printf("%lld\n",rest);

    }



	return 0;
}
