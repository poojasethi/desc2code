#include <cstdio>
#include <cmath>

using namespace std;

const double EP=0.00005;

const int MAX=10001;
double dist[MAX];

inline bool f(double len, double T, int N)
{
    double pos = dist[N-1] + len;
    //printf("pos=%f\n", pos);
    for(int i=N-2; i>=0; i--)
    {
        pos -= T;
        if (pos > dist[i] + len)
            pos = dist[i] + len;
        else if (pos < dist[i] - len)
            return false;
    }
    if (pos < 0)
        return false;
    return true;
}

int main()
{
    int K, N;
    double T;

    scanf("%d", &K);
    for(; K>0; K--)
    {
        double left, right, mid;
        scanf("%d %lf", &N, &T);
        for(int i=0; i<N; i++)
            scanf("%lf", &dist[i]);
        left = 0.0, right = 10000 * 1000001;
        //left = 0.0, right = 10.0;
        while( right - left >= EP )
        {
            //printf("left=%f, right=%f\n", left, right);
            if ( f(mid = ((right + left) / 2), T, N) )
                right = mid;
            else
                left = mid;
            
        }
        printf("%.4f\n", mid);
    }
    return 0;
}
