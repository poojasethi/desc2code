#define DEBUG 0

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <algorithm>
using namespace std;

int n;
double b, p[10000];

int eval(double d) {
  int i;
  double last=max(0.0,p[0]-d);
  for(i=1;i<n;i++) {
    if(last+b>p[i]+d)
      return 0;
    last=max(last+b,p[i]-d);
  }
  return 1;
}

double bSearch(double left, double right) {
  double mid=(left+right)/2.0;
  if(right-left<1e-5)
    return mid;
  if(DEBUG)
    printf("left=%lf mid=%lf right=%lf\n", left, mid, right);
  if(eval(mid))
    return bSearch(left, mid);
  else
    return bSearch(mid, right);
}

int main(void) {
  int tc, i;

  scanf("%d", &tc);
  while(tc--) {
    scanf("%d%lf", &n, &b);
    for(i=0;i<n;i++)
      scanf("%lf", &p[i]);
    printf("%.4lf\n", bSearch(0.0,10000000000.0));
  }
}
