const int maxn = (int)1e4 + 10;
int n;
double t,x[maxn];
bool solve(double m) {
	double last = 0.0;
	for (int i = 0; i < n; i++) {
		double next = max(last,x[i] - m);
		if (fabs(next - x[i]) > m) {
			return false;
		}
		last = next + t;
	}
	return true;
}
