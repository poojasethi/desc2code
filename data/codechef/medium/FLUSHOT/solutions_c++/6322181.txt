#include <cstdio>
#include <algorithm>
 
using namespace std;
 
const int MAXN = 10000;
const double EPS = 1e-8;
 
double a[MAXN];
 
bool solvable(int n, double t, double d) {
  double x = max(a[0]-d,0.0);
  
  for (int i = 1; i < n; ++i) {
    if (a[i]+d<x+t) return false;
    x = max(a[i]-d,x+t);
  }
  return true;
}
 
int main() {
  int cnum;
  scanf("%d", &cnum);
  while (cnum--) {
    int n;
    double t;
    scanf("%d%lf", &n, &t);
    for (int i = 0; i < n; ++i) scanf("%lf", &a[i]);
    sort(a,a+n);
    double lo = 0, hi = 1;
    while (!solvable(n, t, hi)) { lo=hi; hi *= 2;}
    while (lo+EPS<hi) {
      double mid = (lo+hi)/2;
      if (solvable(n,t,mid)) hi = mid;
      else lo = mid;
    }
    printf("%.4lf\n", lo);
  }
}
