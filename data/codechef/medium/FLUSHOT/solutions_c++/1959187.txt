#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
const double ep=1e-8;
bool mind(double*x,double t,int n,double d)
{
    double a=max(x[0]-d,0.0);
    for(int i=1;i<n;i++){
		if(x[i]+d-a<t)return false;
		a=max(x[i]-d,a+t);
    }
    return true;

}
int main()
{
	int k;
	scanf("%d",&k);
	while(k--){
		int n;
		double t;
		scanf("%d%lf",&n,&t);
		double* a=new double [n];
		for(int i=0;i<n;i++)scanf("%lf",&a[i]);
		sort(a,a+n);
        double lb=0.0,ub=1;
        while(!mind(a,t,n,ub))ub*=2;
        while(lb+ep<ub){
				double mid=(lb+ub)/2;
				if(mind(a,t,n,mid))ub=mid;
				else lb=mid;
        }
        printf("%.4lf\n",lb);
	}
	return 0;
}
