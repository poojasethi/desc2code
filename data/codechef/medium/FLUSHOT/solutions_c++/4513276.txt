#define gcd __gcd
#define e 1e-8
using namespace std;
double a[10005],t,d;
int n,i;
bool check(double d)
{
    double x;
    x=max((double)0.0,a[0]-d);
    for(i=1;i<n;i++)
    {
        if(a[i]+d<x+t)
        return false;
        x=max(a[i]-d,x+t);
    }
    return true;
}
int main()
{
//freopen("input.txt","r",stdin);
//freopen("output.txt","w",stdout);
int test;
scanf("%d",&test);
while(test--)
{
    cin>>n>>t;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }
    sort(a,a+n);
    double low=0.0,mid,high=1;
    while(!check(high))
    high*=2;
    while(low+e<high)
    {
        mid=(low+high)/2;
        if(check(mid) )
        {
            high=mid;
        }
        else
        low=mid;
    }
    printf("%0.4lf\n",low);
}
 
return 0;
}
