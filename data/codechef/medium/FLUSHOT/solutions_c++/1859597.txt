typedef long long lint;typedef vector<int> vi;typedef vector<vi> vii;typedef pair<int, int> p;
class node{public :    int x;    int y;    int cost;};
bool operator < ( const node & first, const node & second){    return first.cost<second.cost;}
double pi = 4 * std::atan(1);

double x[10001];

int main()
{
    int t,n;
    double k;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        for(int i=0;i<n;i++)
            cin>>x[i];
        double lo=0,mid,hi=1000000000;
        for(int i=0;i<=200;i++)
        {
            mid=(lo+hi)/2;
            double y[n];
            for(int j=0;j<n;j++) y[j]=x[j];
            y[0]=max(0.0,y[0]-mid);
            int fl=0;
            for(int j=1;j<n;j++)
            {
                double dist=y[j]-y[j-1];
                if(dist>k)
                {
                    double extra=dist-k;
                    y[j]-=min(extra,mid);
                }
                else
                {
                    double need=k-dist;
                    if(need>mid) {fl=1; break;}
                    else y[j]+=need;
                }
            }
            if(fl) //not satisfied then increase the distance
                lo=mid;
            else hi=mid;
        }
        printf("%.4lf\n",mid);
    }
    return 0;
}
