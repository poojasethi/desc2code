#define scanfloat(n) scanf("%lf", &n)
#define FOR(i,n) for (int i = 0; i < n; ++i)
using namespace std;

int N;
double pos[10000], T;

double minMove(){
	double move = 0, p = pos[0];
	int h = 0;
	for(int i = 1; i < N; ++i){
		p += T;
		if(p - pos[i] > move){
			move = p - pos[i];
			h = i;	// this one has to move the highest
		}
		p = max(p, pos[i]);
		pos[i] = p;
	}
/*
1
8 99999.999
0 99999.999 999999.1 999999.2 999999.3 999999.4 999999.5 999999.6 999999.7
*/
/*	FOR(i,N){
		cout<<pos[i]<<" ";
	}*/
	move /= 2;
	double res = move;
	if(res > pos[0]){ // first person cannot move the half to the left
		p = 0;
		double diff;
		FOR(i,h+1){
			if(pos[i] - p >= res){
				res = 0;
				break;
			}else{
				diff = pos[i] - p;
				res -= diff;
				for(int j = i; j <= h; ++j)
					pos[j] = pos[j] - diff;
				p = pos[i] + T;
			}
		}
		if(res > 0)
			move += res;
	}

	return move;
}

int main(){

	int cases;
	scanf("%d", &cases);
	while(cases--){
		scanf("%d %lf", &N, &T);
		for(int i = 0; i < N; ++i)
			scanfloat(pos[i]);
		printf("%.4lf\n", minMove());
	}
	return 0;
}
