#include<cstdio>
#include<algorithm>
#include<cstdlib>
#include<cmath>
using namespace std;

#define forall(a,b) for(int i=0;i<(a);i+=(b))

#define N 10001

int n;
double people[N],t;

bool isValid(double d) {
	double pos = max(people[0]-d,0.0);

	for(int i=1;i<n;i++) {
		if(people[i]+d<pos+t) return false;
		pos = max(people[i]-d,pos+t);
	}

	return true;
}

void solve() {
	scanf("%d%lf",&n,&t);
	forall(n,1)
		scanf("%lf",people+i);

	double high=1,low=0,mid;
	while(!isValid(high)) high*=2;

	while(fabs(high-low)>1e-9) {
		mid = (high+low)/2;
		if(isValid(mid))
			high = mid;
		else
			low = mid;
	}
	printf("%.4lf\n",high);

}

int main(){
	int k;
	scanf("%d",&k);
	forall(k,1)
		solve();
	return 0;
}
