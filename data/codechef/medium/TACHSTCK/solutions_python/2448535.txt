n, d = [int(i) for i in raw_input().strip().split()]
c = []
for i in range(n):
    c.append(int(raw_input().strip()))
c.sort()
total = 0
i = 0
while i < len(c)-1:
    if c[i+1] - c[i] <= d:
        total += 1
        i += 2
    else:
        i += 1
print total
