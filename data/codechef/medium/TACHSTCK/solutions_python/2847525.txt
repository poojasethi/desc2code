import sys



def main():
	firstLine = map(int,sys.stdin.readline().split())
	n = firstLine[0]
	d = firstLine[1]

	sticks = []
	for i in range(n):
		sticks.append(int(sys.stdin.readline()))

	sticks.sort()

	pair = 0
	i = 0
	while i < n-1:
		if sticks[i+1] - sticks[i] <= d:
			i = i+2
			pair = pair + 1
		else:
			i = i+1
	print pair


if __name__ == '__main__':
	main()
