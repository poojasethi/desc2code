import sys
import math

n,d = map(int,raw_input().split(' '))
chops = []
for i in range(n):
    chops.append(input())
chops.sort()
iter = 0
count = 0
while iter<=n-2:
    if chops[iter+1]-chops[iter]<=d:
        count+=1
        iter += 2
    else:
        iter += 1
print(count)
