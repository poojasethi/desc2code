def takeInput():
    N, D = raw_input().split()
    N = int(N)
    D = int(D)
    L = []
    for i in range(N):
        L.append(int(raw_input()))
    return N, D, L
    
def findMaxPairs(N, D, L):
    L = sorted(L)
    L.reverse()
    ans = 0
    i = 1
    while (i <= N - 1):
        if (L[i - 1] - L[i] <= D):
            ans += 1
            i += 2
        else:
            i += 1
    return ans

N, D, L = takeInput()
ans = findMaxPairs(N, D, L)
print ans
