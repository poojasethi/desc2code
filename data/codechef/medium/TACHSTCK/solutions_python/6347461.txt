import sys
 
def main():
	f = map(int,sys.stdin.readline().split())
	n = f[0];
	d = f[1];
	l = [];
	mark = [];
	for i in range (n) :
		l.append(int(sys.stdin.readline()))
		mark.append(0)
	l.sort()
	ans = 0
	j=0
	while(j < len(l)-1):
		if(mark[j] == 0):
			if(l[j+1] - l[j] <= d):
				ans += 1
				mark[j] = 1
				mark[j+1] = 1
		j += 1
	return ans
 
print main()
