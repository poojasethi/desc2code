import sys

[num,d] = [int(s) for s in sys.stdin.readline().rstrip().split()]
chopsticks = []
n = num
while num != 0:
	chopsticks.append(int(sys.stdin.readline().rstrip()))
	num -= 1

chopsticks.sort()
num_pairs = 0
i = 0
while i < n-1:
	if chopsticks[i] >= chopsticks[i+1] - d:
		num_pairs += 1
		i += 2
	else:
		i += 1
	
print num_pairs
