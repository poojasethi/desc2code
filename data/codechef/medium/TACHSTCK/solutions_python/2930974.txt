#!/usr/bin/env python

# Task: Chopsticks

def solve(d, values):
    pairs = 0
    i = 0
    values.sort()
    while i < len(values) - 1:
        if values[i+1] - values[i] <= d:
            pairs += 1
            i += 2
        else:
            i += 1
    return pairs


def main():
    n, d = map(int, raw_input().split())
    values = []
    for _ in range(n):
        values.append(int(raw_input()))
    print("%s" % solve(d, values))


if __name__ == "__main__":
    main()
