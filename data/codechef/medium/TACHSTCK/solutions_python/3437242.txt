
stickCount, D = map(int, raw_input().split())

sticks = []
for _ in xrange(stickCount):
    sticks.append(int(raw_input()))

sticks.sort()

i = 1
count = 0
while i < stickCount:
    diff = sticks[i] - sticks[i - 1]
    if diff <= D:
        count += 1
        i += 2
    else:
        i += 1

print count
