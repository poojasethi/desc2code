using namespace std;
 inline unsigned int fastread()
    {
      unsigned int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
int main(void)
{
	int t,d,i=0,count=0;
	//scanf("%d%d",&t,&d);
	t=fastread();
	d=fastread();

	int a[t+1];
	while(i<t)
   a[i++]=fastread();
	sort(a,a+t);
	for(i=0;i<t-1;i++)
	if(a[i+1]-a[i]<=d)
	{
		count++;
		i=i+1;
	}
	printf("%d\n",count);
	return 0;
}
