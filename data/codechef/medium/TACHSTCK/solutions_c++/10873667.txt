#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long int ll;
int main()
{
    int n,d;
    scanf("%d %d",&n,&d);
    int arr[n];
    for(int i=0;i<n;i++)
    scanf("%d",&arr[i]);
    sort(arr,arr+n);
    int count=0,i=0;
    while(i<n-1){
        if(arr[i+1]-arr[i]<=d){
            count++;
            i++;
        }
        i++;
    }
    printf("%d\n",count);
}
