#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_v) (int)(_v).size()
#define all(_v) (_v).begin(),(_v).end()
#define pii pair<int,int>
typedef long long ll;
const int INF = 1000000000;
const ll MOD = 1e9 + 7;

const int maxn = 100010;
int arr[maxn];

int main() {
	ios_base::sync_with_stdio(false);

    int n,d; cin >> n >> d;
	for (int i = 0; i < n; i ++) cin >> arr[i];
	sort(arr,arr + n);

	int ans = 0;
	for (int i = 0; i < n - 1; i ++) {
		if(arr[i + 1] - arr[i] <= d) {
			ans ++;
			i ++;
		}
	}

	cout << ans << "\n";
	return 0;
}
