#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,d;
	cin>>n>>d;
	int a[n],b[n],i,c=0;
	for(i=0;i<n;i++)
	cin>>a[i];
	sort(a,a+n);
	b[0]=0;
	for(i=1;i<n;i++)
	{
		if(a[i]-a[i-1]<=d)
		{
			if(i==1)
			b[i]=1;
			else
			b[i]=b[i-2]+1;
		}
		else
		b[i]=b[i-1];
	}
	cout<<b[n-1]<<endl;
	return 0;
}
