#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int n, count = 0;
    long long int d;
    scanf("%d %lld", &n, &d);
    long long int arr[n];
    for(int i = 0; i < n; i++)
    {
        scanf("%lld", &arr[i]);
    }
    sort(arr, arr + n);
    for(int i = 0; i < n - 1; i++)
    {
        if(arr[i + 1] - arr[i] <= d)
        {
            count++;
            i++;
        }
    }
    printf("%d\n", count);
    return 0;
}
