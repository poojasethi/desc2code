#include<bits/stdc++.h>
using namespace std;
int main()
{
	
	
		long long int n,m,pair=0,i;	
		scanf("%lld%lld",&n,&m);
		long long int a[n];
		for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		sort(a,a+n);
		for(i=0;i<n;)
		{
			if(abs(a[i]-a[i+1])<=m)
			{
				pair++;
				i+=2;
			}
			else
			{
				i++;	
			}
		}
			
	
		printf("%lld\n",pair);
	
	
}
