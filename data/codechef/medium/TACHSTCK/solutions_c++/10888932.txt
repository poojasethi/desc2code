#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n,d;
	scanf("%d%d",&n,&d);
	int a[n];
	for (int i = 0; i < n; i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	int ctr = 0;
	for (int i = 0; i < n-1;)
	{
		if ((a[i+1]-a[i]) <= d)
		{
			ctr++;
			i = i+2;
		}
		else
		{
			i++;
		}
	}
	cout << ctr << endl;
}
