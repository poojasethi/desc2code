#include <iostream>
#include <bits/stdc++.h>
typedef long long int ll;

using namespace std;

int main()
{
	ll t,d;
	cin>>t>>d;
	vector<ll>v;
	for(ll i =0 ;i<t;i++)
	{
		ll a;
		cin>>a;
		v.push_back(a);
	}
	sort(v.begin(),v.end());
	ll cnt =0;
	for(ll i=0;i<v.size()-1;i++)
	{
		if(abs(v[i]-v[i+1])<=d)
		{
			cnt++;
			i++;
		}
	}
	cout<<cnt<<endl;
	return 0;
}
