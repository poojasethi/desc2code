#include<bits/stdc++.h>
using namespace std;
int main()
{
	long n,d,a;
	cin>>n>>d;
	std::vector<long> foo;
	for(long i=0;i<n;i++)
	{
		cin>>a;
		foo.push_back(a);
	}
	sort(foo.begin(),foo.end());
	int c=0;
	long i=1;
	while(i<n)
	{
		if(foo[i]-foo[i-1]<=d)
		{
			c++;
			i+=2;
		}
		else
			i++;
	}
	cout<<c<<endl;
}
