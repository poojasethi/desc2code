#include<bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("in","r",stdin);
	long long n,d;
	cin>>n>>d;
	long long a[n];
    for(int i=0;i<n;i++)
    	cin>>a[i];
    sort(a,a+n);
    long diff;
    long c=0;
    long i=0;
    while(i<n-1)
    {
     diff=a[i+1]-a[i];
     if(diff<=d)
     {
     	c++;
     	i+=2;
     }
     else
     	i++;
    }
    cout<<c<<endl;
}
