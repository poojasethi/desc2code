#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	int n,d;
	cin>>n>>d;
	 int a[n];
	for(int i=0;i<n;i++)
	{cin>>a[i];}
	sort(a,a+n);
    int j,count=0;
    for(j=1;j<n;j++)
    {
    	if(a[j]-a[j-1]<=d)
    	{j++;
    	 count++;
    	}
    	
    }
    
cout<<count<<"\n";
return 0;
}
