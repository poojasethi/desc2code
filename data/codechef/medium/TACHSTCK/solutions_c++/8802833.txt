#include <iostream>
#include <cstdio>
#include<algorithm>
using namespace std;
inline int next_int() {
	int n = 0;
	char c = getchar_unlocked();
	while (!('0' <= c && c <= '9')) {
	c = getchar_unlocked();
	}
	while ('0' <= c && c <= '9') {
  n = n * 10 + c - '0';
  c = getchar_unlocked();
 }
 return n;
}
int main(void) {
    	int t1,d1,i=0,c=0;
	t1=next_int();
	d1=next_int();
	int a[t1+1];
    	while(i<t1)
	a[i++]=next_int();
	sort(a,a+t1);
	for(i=0;i<t1-1;i++)
	if(a[i+1]-a[i]<=d1)
	{
		c++;
		i=i+1;
	}
	printf("%d\n",c);
	return 0;
}
