#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define endll "\n";
int main()
{
	ios_base::sync_with_stdio(false);
  long long int n,d;
  cin>>n;
  cin>>d;
  vector<long long int> lengths;
  long long int result=0;
  for(int i=0;i<n;i++){
    long long int tmp;
    cin>>tmp;
    lengths.push_back(tmp);
  }
  sort(lengths.begin(), lengths.end());
  // cout<<"length: "<<lengths.size()<<endll;
  for(long long int i=0;i<n-1;i++){
    // cout<<"a"<<endll;
    if(lengths[i+1]-lengths[i]<=d){
      // cout<<"lol "<<lengths[i]<<endll;
        result++;
      i++;
    }
  }
  cout<<result<<endll;
	return 0;
}
