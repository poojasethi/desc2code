#include <cstdio>
#include <algorithm>

using namespace std;

int N, D, A[122222];

int main() {
    scanf("%d%d", &N, &D);

    for (int i = 1; i <= N; i++)
        scanf("%d", A + i);

    sort (A+ 1, A + 1 + N);

    int j=1,count=0;

    while (j < N) {
        if (abs(A[j] - A[j+1]) <= D) {
            count++;
            j += 2;
        }
        else j++;
    }

    printf("%d\n", count);

    return 0;
}
