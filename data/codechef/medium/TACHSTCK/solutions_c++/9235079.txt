#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	long long n, d;
	vector<long long> chops;
	scanf("%lld %lld", &n, &d);
	while (n--)
	{
		long long temp;
		scanf("%lld", &temp);
		chops.push_back(temp);
	}
	sort(chops.begin(), chops.end());
	long long count = 0;
	for (long long i = 0; i < chops.size() - 1; ++i)
	{
		if (chops[i + 1] - chops[i] <= d)
		{
			++count;
			++i;
		}
	}
	printf("%lld", count);
}
