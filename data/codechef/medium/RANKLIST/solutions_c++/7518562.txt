#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
    int t;cin>>t;
    while(t--){
        long long n,s;
        cin>>n>>s;
    long long k;
    for(k=n;k>0;k--)
    {
        long long l=((k*(k+1))/2)+n-k;
        long long r=((k*(k+1))/2)+((n-k)*k);
        if(s<=r&&s>=l)
            break;
    }
    long long ans=k;
    cout<<n-ans<<endl;
    }
}
