#include<iostream>
using namespace std;

int main()
{  
   ios_base::sync_with_stdio(false);
   cin.tie(NULL);
    unsigned long long t,n,n2,s;
	cin>>t;
	while(t--)
{   
    cin>>n>>s;
    n2=n;
	if(s==n*(n+1)/2)
	  cout<<"0"<<endl;
	 else
	  if(n==s)
	   cout<<n-1<<endl; 
	else
	{
		for(unsigned long long i=2;i<=n*(n+1)/2;i++)
		{
			if(s<=n+i)
			 {
			 	cout<<n2-i<<endl;
			 	break;
		    }
			n=n+i;
		}
  }
}	return 0;
}
