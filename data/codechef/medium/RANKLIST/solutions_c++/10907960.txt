#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

const int maxn = (int)1e5;
int a[maxn];

int main()
{
	std::ios::sync_with_stdio(false);;
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		//cin>>n>>k;
		ll s;
		cin>>n>>s;
		int ans = 0;
		for(int i=0,j=1;i< n;i++)
		{
			if((n-i-1)+j<=s)
			{
				a[i]=j;
				s -= j;
				++j;
			}
			else
			{
				ans = n - i;
				break;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
