
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

long long n,sum,pr,ans;

int main()
{
    int test;
    cin >> test;
    while(test--)
    {
        int i;
        cin >> n;
        cin >> sum;
        pr=(n*(n+1))/2;
        if(sum==pr)
            ans=0;
        else if(sum>n && sum<pr)
        {
            sum-=n;
            for(int i=0;i<n;i++)
            {
                sum-=i;
                if(sum==0)
                {
                    ans=n-i-1;
                    break;
                }
                else if(sum<0)
                {
                    ans=n-i;
                    break;
                }
            }
        }
        else if(sum==n)
            ans=n-1;
        cout << ans << "\n";
    }
}
