#include<iostream>
using namespace std;
int main(){
	int i,t,n,ans,flag=0;
	long long int s;
	cin>>t;
	while(t--){
		ans = 0;
		i=0;
		flag = 0;
		cin>>n>>s;
		if(n==1){
			if(s==1){
				cout<<0<<"\n";
			}
			else{
				cout<<1<<"\n";
			}
		}
		else{
			s = s - n;
			for(i=1;i<n;i++){
				//temp  =  s;
				s = s-i;
				if(s < 0){
					//j = i-1;
					flag = 1;
					ans = n-i;
					cout<<ans<<"\n";
					break;
				}
				else if(s==0){
					flag = 1;
					ans = n-1-i;
					cout<<ans<<"\n";
				}
			}
			if(flag==0){
				cout<<1<<"\n";
			}
		}
	}
	return 0;
}
