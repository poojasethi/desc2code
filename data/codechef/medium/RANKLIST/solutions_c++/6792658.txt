#include <stdio.h>

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int ans=0;
        long long int currsum,s,n;
        scanf("%lld %lld",&n,&s);
        currsum = (n*(n+1))/2;
        while(currsum!=s)
        {
            //printf("%d ",currsum);
            if (n-1>=currsum-s) currsum=s;
            else
            {
                n--;
                currsum-=n;
            }
            ans++;
        }
        printf("%d\n",ans);
    }
}
