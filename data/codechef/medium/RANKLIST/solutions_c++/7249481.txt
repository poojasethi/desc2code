#include<stdio.h>
using namespace std;
int main()
{
	long long int t,n,s,r;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&n,&s);
		if((n*(n+1))/2==s)
		printf("0\n");
		else
		if((n*(n+1))/2-s==1)
		printf("1\n");
		else
		{
			s=s-n;
			r=1;
			while(s>=r)
			{
				s-=r;
				r++;
			}
			printf("%lld\n",n-r);
		}
	}
}
