#include <bits/stdc++.h>
using namespace std;
int main(){
    int t;
    long long n,s;
    long long diff;
    long long ans;
    long long i,j;
    scanf("%d",&t);
    while(t--){
        scanf("%lld%lld",&n,&s);
        diff = ((n+1)*n)/2 - s;
        i = 0; j = 0;
        while(i < diff)
        {i += n-1-j; j++;}
        ans = j;
        cout<<ans<<"\n";
    }
    return 0;
}
