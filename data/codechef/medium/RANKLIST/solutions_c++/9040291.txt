#include<bits/stdc++.h>
using namespace std;

#define ll long long int

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		ll n,s,i,count=0,diff=0,sum=0;
		scanf("%lld %lld",&n,&s);
		diff=(n*(n+1)/2)-s;
		for(i=n-1;i>=0;i--)
		{
			if(sum<diff)
			{
				sum=sum+i;
				count++;	
			}
			else
			break;
		}
		printf("%lld\n",count);
	}
	return 0;
}
