#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define ll long long
#define all(c) c.begin(),c.end()
using namespace std;
int main()
{
    ll i,t,n,s,cal,ans;
    cin>>t;
    while(t--)
    {
        scanf("%lld%lld",&n,&s);
        cal=((n+1)*n)/2;
        if(cal==s)
            printf("0\n");
        else
        {
            cal=s;
            cal-=n;
            ans=n-1;
            for(i=2;i<=n;i++)
            {
                if(cal>=(i-1))
                {
                    cal=cal-(i-1);
                    ans--;
                }
            }
            cout<<ans<<endl;
        }
    }
    return 0;
}
