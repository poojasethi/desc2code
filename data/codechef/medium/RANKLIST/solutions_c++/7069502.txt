#include <iostream>
using namespace std;

int main() 
{
	int t;
	cin >> t;
	for(;t>0;t--)
	{
		long long int n,s;
		cin >> n;
		cin >> s;
		if( (n*(n+1))/2 ==s)
			cout<<"0"<<endl;
		else
		{
			long long int i;
			for(i=n-1;i>0;i--)
			{
				if( s >= (i*(i+1))/2 + n-i )
					break;
			}
			cout<< n-i <<endl;
		}
	}
	return 0;
}
