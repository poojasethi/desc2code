#include<iostream>
using namespace std;
int main()
{
int t;
long long a,b,i,j,sum;
cin>>t;
while(t--)
{
cin>>a>>b;
long long arr[a];
long long brr[a];
for(i=0;i<a;i++)
{
arr[i]=i+1;
}
sum=(a*(a+1))/2;
if(b==(a*(a+1))/2)
cout<<0<<"\n";
else
{
    j=a-2;
for(i=0;i<a-1;i++)
{
brr[i]=sum;
sum=sum-arr[j];
j--;
}
for(i=0;i<a-1;i++)
{
if(b>=brr[i])
break;
}
cout<<i<<"\n";
}
}
}
