#include <iostream>

using namespace std;

int main()
{
    int t,n,i,ans,count;
    unsigned long long int s,sum;
    cin>>t;
    while(t--){
        cin>>n>>s;
        sum=n; i=1;count=1;
        if(n==s){
            ans= n-1;
        }else{
            while(sum<s){
                sum+=i;
                count++;
                if(sum<s){
                    i++;
                }
                if(sum>s){
                    count--;
                }
            }

            ans=n-count;
        }

        cout<<ans<<"\n";
    }
    return 0;
}
