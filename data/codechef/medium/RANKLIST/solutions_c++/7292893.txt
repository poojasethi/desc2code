using namespace std;

inline int scan()
{
	int n = 0, sign = 1, ch = get();

	while (ch < 48 || ch > 57) {
		if (ch == '-') {
			sign = -1;
		}

		ch = get();
	}

	while (ch >= 48 && ch <= 57) {
		n = (n << 3) + (n << 1) + ch - 48;
		ch = get();
	}

	return n * sign;
}

int main()
{
	int t;
	ll s, n, count, i;
	bool flag;

	t = scan();

	while (t--) {
		slld(n);
		slld(s);

		int a[n];
		count = i = 0;
		flag = 0;

		if (s == n * (n + 1) / 2) {
            pdn(0);
            continue;
		}

		for (i = 0; i < n; i++) {
            a[i] = 1;
		}

		s -= n;
		i = 1;

        while(s) {
            if (s >= i) {
                a[i] += i;
                s -= i;
                i++;
            } else {
                a[i] += s;
                s = 0;
                flag = 1;
            }
        }

        if (flag) {
            count++;
            i++;
        }

        //if (i != n - 1) {
            count += n - i;
        //}

        plldn(count);
	}
}
