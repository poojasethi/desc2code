#include <iostream>
#include <math.h>
using namespace std;

int main()
{
	long long N,S,maxsum,sq,diff;
	int t,j,i,flag=0;
	cin>>t;
	while(t--)
	{
		cin>>N>>S;
		maxsum=(N*(N+1))/2;
		if(S==N)
		cout<<N-1<<endl;
		else if(maxsum==S)
		cout<<"0"<<endl;
		else
		{
			diff=S-N;
			sq=(1+sqrt(1+8*diff))/2;
			cout<<N-sq<<endl;
		}
		
		
		
	}	
	return 0;
}
