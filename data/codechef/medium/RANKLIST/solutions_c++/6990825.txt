// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%lld", &x)

int main()
{
    ll t, n, s, ans, i;
    s(t);
    while (t--) {
        s(n);
        s(s);
        s = (n*(n+1))/2 - s;
        ans = 0;
        for (i = n; s > 0; i--) {
            if ((s-i+1) >= 0) {
                ans++;
                s -= (i-1);
            }
            else {
                ans++;
                break;
            }
        }
        printf("%lld\n", ans);
    }
    return 0;
}
