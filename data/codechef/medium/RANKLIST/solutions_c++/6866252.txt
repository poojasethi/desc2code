#include <stdio.h>
 
int main()
{
    int t;
    long long n,s,i,sum;
    scanf("%d",&t);
 
    while(t--)
    {
 
        scanf("%lld%lld",&n,&s);
 
        if(s==(n*(n+1))/2)
          printf("0\n");
 
        else if(s==n)
            printf("%lld\n",n-1);
 
        else
        {
            sum=0;
            for(i=1;i<n;i++)
              {
                  sum+=i;
                  if(s-sum<=n-i)
                   break;
              }
 
                printf("%lld\n",n-i+1);
        }
 
    }
	return 0;
}
