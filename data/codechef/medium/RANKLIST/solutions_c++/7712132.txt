#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		ll n,s;
		cin>>n>>s;
		ll sum=(n*(n+1))/2;
		ll count=0;
		while(sum!=s)
		{
			if(n-1>=sum-s)
			{
				sum=s;
			}
			else
			{
				n--;
				sum-=n;
			}
			count++;
		}
		cout<<count<<"\n";
	}
	return 0;
}
