#include <stdio.h>
int main(int argc, char const *argv[])
{
	long long int i , j, testcases, x, y, n, s;
    scanf("%lld",&testcases);
	for (i = 0; i < testcases; i++)
	{
		scanf("%lld %lld", &n, &s);
		if ((n*(n+1))/2 == s)
			printf("%d\n", 0);
		else
		{
    		for (j = n-1; j > 0; j--)
    		{
    			x = (j*(j+1))/2; 
    			if ((s - x)/(n - j) > 0)
    				break;
    		}
    		printf("%lld\n", n - j);
	    }
	}
	return 0;
}
