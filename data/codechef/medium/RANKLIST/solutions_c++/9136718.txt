#include<stdio.h>
int main()
{
	long long int t, sum, count, i, temp, n, s;
	scanf("%lld", &t);
	while (t--)
	{
		scanf("%lld%lld", &n, &s);
		temp = (n*(n + 1)) / 2;
		temp -= s;
		sum = 0;
		count = 0;
		if (temp)
		{
			for (i = n - 1; i >= 1; i--)
			{
				sum += i;
				count++;
				if (temp <= sum)
					break;
			}
		}
		printf("%lld\n", count);
	}
	getchar();
	getchar();
	return 0;
}
