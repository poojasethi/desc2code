#include<iostream>
#include<cmath>
using namespace std;

int main()
{
	int t;
	long long n, operations;
	cin>>t;
	while(t--)
	{
		long long sum;
		cin>>n>>sum;
		if( n==sum )
			operations = n-1;
		else if( (n*(n+1))/2 == sum )
			operations = 0;
		else
		{
			operations = (1.0+sqrt(1+(8*(sum-n))))/2;
			operations = n-operations;
		}
		cout<<operations<<endl;
	}
	return 0;
}
