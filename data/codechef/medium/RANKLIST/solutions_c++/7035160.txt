#include <iostream>
#include<stdio.h>

using namespace std;

int main()
{
    int t;

    long long int n,c,s,sum,d,i;
    scanf("%d",&t);
    while(t-->0)
    {
        sum=0;
        d=0;
        scanf("%lld %lld",&n,&s);
        for(i=1;i<=n;i++)
        {

            sum=sum+i;
        }
        if(sum==s)
        {
            c=0;
        }
        else{
            c=1;
            for(i=n;i>1;i--)
            {
                d=d+i-1;
                if(sum-d<=s)
                {

                    break;
                }
            else{
                c++;
                }
            }
        }
        printf("%lld\n",c);

    }
    return 0;
}
