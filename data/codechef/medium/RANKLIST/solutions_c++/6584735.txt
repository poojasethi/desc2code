#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int t, i, num, oper;
    long long int n, s;
   cin>>t;
    while(t--)
    {
        num=2;
        oper=0;
        cin>>n>>s;
        s-=n;
        i=1;
        while(s)
        {
            if(s<i)
                break;
            s-=i;
            i++;
        }
        if(i==n)
           cout<<"0"<<endl;
        else if(i<n)
           cout<<n-i<<endl;
    }
    return 0;
}
