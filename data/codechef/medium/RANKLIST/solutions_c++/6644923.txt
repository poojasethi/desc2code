#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int n,s,res=0;
        scanf("%lld%lld",&n,&s);
        res=(n*(n+1))/2;
        if(s==res)
        {
            printf("0\n");
            continue;
        }
        long long int i=1,sum=0,flag=0;
        for(i=1;i<=n;i++)
        {
            sum=sum+i;
            if(s-sum==n-i)
            {
                flag=1;
                break;
            }
            if(s-sum<n-i)
                break;
        }
 
        if(flag)
        {
            long long int k=n-i;
            printf("%lld\n",k);
            continue;
        }
        long long k1=n-i+1;
        printf("%lld\n",k1);
    }
    return 0;
}
