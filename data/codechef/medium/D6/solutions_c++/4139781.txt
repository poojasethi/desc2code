#include<bits/stdc++.h>
using namespace std;
struct point{
    int x;
    int y;
};
int main(){
    int plane[500][500];
    vector<point> data(500);
    int t;
    cin>>t;
    while(t--){
        for(int i=0;i<110;i++){
            for(int j=0;j<110;j++){
                plane[i][j]=0;
            }
        }

        int n;
        cin>>n;
        for(int i=0;i<n;i++){
            cin>>data[i].x>>data[i].y;
            data[i].x+=50;
            data[i].y+=50;
            plane[data[i].x][data[i].y]=1;
        }

        int sol=0;
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                int X1=data[i].x;
                int Y1=data[i].y;
                int X2=data[j].x;
                int Y2=data[j].y;

                if(X1==X2 || (float)(Y2-Y1)/(float)(X2-X1)<0)
                {
                    if(plane[X1+abs(Y1-Y2)][Y1+abs(X1-X2)]==1 && plane[X2+abs(Y1-Y2)][Y2+abs(X1-X2)]==1)
                    sol++;
                }
            }
        }
        cout<<sol<<"\n";
    }
}
