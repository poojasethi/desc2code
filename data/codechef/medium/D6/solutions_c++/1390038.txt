#include<stdio.h>

int main()
{
    int t;
    scanf("%d",&t);
    int** points = new int*[101];
    for(int i=0; i<101; ++i)
    {
        points[i] = new int[101];
        for(int j=0; j<101; ++j) points[i][j] = -1;
    }
    for(int i=0; i<t; ++i)
    {
        int N;
        scanf("%d",&N);
        int* X = new int[N];
        int* Y = new int[N];
        for(int j=0; j<N; ++j)
        {
            scanf("%d",&X[j]);
            scanf("%d",&Y[j]);
            points[X[j]+50][Y[j]+50] = i;
        }
        int count = 0;
        int x1,y1,x2,y2;
        for(int j=0; j<N-1; ++j)
        {
            x1 = X[j];
            y1 = Y[j];
            for(int k=j+1; k<N; ++k)
            {
                x2 = X[k];
                y2 = Y[k];

                // first set

                int x = x2+y1-y2;
                int y = y2+x2-x1;

                if(x>=-50 && x<=50 && y>=-50 && y<=50)
                {
                    if(points[x+50][y+50] == i)
                    {
                        x = x1+y1-y2;
                        y = y1+x2-x1;
                        if(x>=-50 && x<=50 && y>=-50 && y<=50) if(points[x+50][y+50] == i) count++;
                    }
                }

                // second set

                x = x2+y2-y1;
                y = y2+x1-x2;

                if(x>=-50 && x<=50 && y>=-50 && y<=50)
                {
                    if(points[x+50][y+50] == i)
                    {
                        x = x1+y2-y1;
                        y = y1+x1-x2;
                        if(x>=-50 && x<=50 && y>=-50 && y<=50) if(points[x+50][y+50] == i) count++;
                    }
                }
            }
        }

        delete X,Y;

        // get rid of double counting

        count /= 4;

        printf("%d\n",count);
    }
    for(int i=0; i<101; ++i) delete points[i];

    return 0;
}
