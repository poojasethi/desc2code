#include<iostream>
using namespace std;
int main()
{
    int count;
    int n;
    int t;
    int x;
    int y;
    int sx1, sx2, sy1, sy2, dx1, dx2, dy1, dy2;
    int array1[500][2];
    
    cin>>t;
   for(int k=0;k<t;k++)
  {
           bool array2[101][101] = {};
           count = 0;
           cin>>n;
           for(int i=0;i<n;i++)
           {
                   cin>>x;
                   cin>>y;
                   array1[i][0] = x;
                   array1[i][1] = y;
                   array2[x+50][y+50] = 1;
           }
           for(int i=0;i<n;i++)
           {
                   
                   for(int j=i+1;j<n;j++)
                   {
                           x=array1[j][0] - array1[i][0];
                           y = array1[j][1] - array1[i][1];
                           sx1 = -y + array1[i][0] + 50;
                           sy1 = x + array1[i][1] + 50;
                           sx2 = y + array1[i][0] + 50 ;
                           sy2 = -x + array1[i][1] + 50;
                           dx1 = x -y + array1[i][0] + 50;
                           dy1 = x+y + array1[i][1] + 50;
                           dx2 = x + y + array1[i][0] + 50;
                           dy2 = y - x + array1[i][1] + 50;
                           if(!((sx1<0 || sx1>100) || (sy1<0 || sy1>100) || (dx1<0 || dx1 >100) || (dy1<0 || dy1>100)))                           
                           if(array2[sx1][sy1] && array2[dx1][dy1])
                           count++;
                           if(!((sx2<0 || sx2>100) || (sy2<0 || sy2>100) || (dx2<0 || dx2 >100) || (dy2<0 || dy2>100)))                           
                           if(array2[sx2][sy2] && array2[dx2][dy2])
                           count++;
                   }
           }
           count = count/4;
           cout<<count<<endl;
  }
 //  system("pause");
   return 0;
}
