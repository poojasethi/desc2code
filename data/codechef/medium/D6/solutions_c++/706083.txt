#include <iostream>

using namespace std;
int n;
int pt[500][2] = {{0}};
int pos[101][101] = {{0}};
bool exist(int a,int b)
{
	if(a>=0 && a<=100)
		if(b>=0 && b<=100)
			return true;
		else
			return false;
	else
		return false;
}
int main()
{
	int t;
	cin >>t;
	while(t--)
	{
		cin >> n;
		int i,j,count = 0;
		for(i=0;i<=100;i++)
			for(j=0;j<=100;j++)
				pos[i][j] = 0;
		for(i=0;i<n;i++)
		{
			cin >> pt[i][0] >> pt[i][1];
			pos[pt[i][0]+50][pt[i][1]+50]++;
		}
		int x,y,a,b;
		for(i=0;i<n-1;i++)
			for(j=i+1;j<n;j++)
			{
				x = pt[j][0] - pt[i][0];
				y = pt[j][1] - pt[i][1];
				if((x+y)%2==0 && (y-x)%2 == 0)
				{
					a = (x+y)/2;
					b = (y-x)/2;
					int x1 = pt[i][0] + 50 + a;
					int y1 = pt[i][1] + 50 + b;
					int x2 = pt[i][0] + 50 - b;
					int y2 = pt[i][1] + 50 + a;

					if(exist(x1,y1) && exist(x2,y2))
						if(pos[x1][y1] > 0 && pos[x2][y2] > 0)
							count++;
				}
			}
		cout << count/2 << endl;
	}
	return 0;
}
