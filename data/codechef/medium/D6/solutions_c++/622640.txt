#include<iostream>
#include<utility>
#include<list>
#include<vector>
#include<cmath>
#include<algorithm>
#define x first
#define y second
using namespace std;
bool bp[200][200];
vector<pair<int,int> >p;

bool square(int i,int j)
{
   int x0,x1,y0,y1;
   if(p[i].y>p[j].y)
   {
    x0=p[i].x+(p[i].y-p[j].y);
    y0=p[i].y+(p[j].x-p[i].x);
    x1=p[j].x+(p[i].y-p[j].y);
    y1=p[j].y+(p[j].x-p[i].x);              
   }
   else
   {
    x0=p[i].x+(p[j].y-p[i].y);
    y0=p[i].y+(p[i].x-p[j].x);
    x1=p[j].x+(p[j].y-p[i].y);
    y1=p[j].y+(p[i].x-p[j].x);
   }
   if((bp[x0][y0])&&(bp[x1][y1]))
    return true;
   else
    return false;
}

int main()
{
   int t,i,j;
   scanf("%d",&t);
   while(t--)
   {
      pair<int,int>a;
      int n;
      scanf("%d",&n);
      for(i=0;i<200;i++)
       for(j=0;j<200;j++)
        bp[i][j]=false;
      for(i=0;i<n;i++)
      {
       scanf("%d%d",&a.x,&a.y);
       a.x+=50;
       a.y+=50;
       bp[a.x][a.y]=true;
       p.push_back(a);
       }
       sort(p.begin(),p.end());
       int c=0;
       for(i=0;i<n-1;i++)
        for(j=i+1;j<n;j++)
        {
          if(square(i,j))
            c++;
          }
        c/=2;
        cout<<c<<endl;
        p.clear();
     }
     return 0;
}
