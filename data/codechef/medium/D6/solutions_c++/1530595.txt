#include<stdio.h>
#include<iostream>
#include<math.h>
#include<string.h>

#include<vector>
#include<algorithm>

#define in(i) scanf("%d",&i);
#define uin(i) scanf("%u",&i);

using namespace std;



int main(){
  int tc,n,m,a[101][101],x,y,ans,x1,y1,dx,dy;
  in(tc);
  
  vector <pair<int,int> > l;
  vector <pair<int,int> > ::iterator it;
  
  while(tc--){
    memset(a,0,sizeof(a));
    ans=0;
    in(n);
    
    m=n;
    while(m--){
      in(x);
      in(y);
      a[x+50][y+50]=1;
      l.push_back(make_pair(x,y));
    }
    
    while(l.size()>=4){
      x=l.back().first;
      y=l.back().second;
      a[x+50][y+50]=0;
      l.pop_back();
      
      for(it=l.begin();it!=l.end();it++){
	x1=(*it).first;
	y1=(*it).second;
	dx=x1-x;
	dy=y1-y;
	if(x+dy+50>=0 && x+dy+50<=100 && y-dx+50>=0 && y-dx+50<=100 && a[x+dy+50][y-dx+50] && x1+dy+50>=0 && x1+dy+50<=100 && y1-dx+50>=0 && y1-dx+50<=100 &&a[x1+dy+50][y1-dx+50])
	  ans++;
      }
    }
    printf("%d\n",ans);
    while(l.size()!=0)
      l.pop_back();
  }
  return 0;
}
