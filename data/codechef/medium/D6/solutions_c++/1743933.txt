#include<iostream>
using namespace std;
int binarysearch(int a[][2],int x,int y,int lb,int ub)
{
    while(lb<=ub)
    {
        int mid=(lb+ub)/2;
        if(a[mid][0]==x && a[mid][1]==y)
        return 1;
        else if(a[mid][0]<x || (a[mid][0]==x && a[mid][1]<y))
        lb=mid+1;
        else
        ub=mid-1;
    }
    return 0;
}
void selection(int a[][2],int n)
{
    for(int i=0;i<n-1;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if((a[i][0]>a[j][0]) || ((a[i][0]==a[j][0]) && (a[i][1]>a[j][1])))
            {
                int t;
                t=a[i][0];
                a[i][0]=a[j][0];
                a[j][0]=t;
                t=a[i][1];
                a[i][1]=a[j][1];
                a[j][1]=t;
            }
        }
    }
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,count=0;
        cin>>n;
        int a[n][2];
        for(int i=0;i<n;i++)
        cin>>a[i][0]>>a[i][1];
        selection(a,n);
        for(int i=0;i<n-1;i++)
        {
            for(int j=i+1;j<n;j++)
            {
                int xm,ym,x1,y1,x2,y2,x3,y3,x4,y4;
                x1=a[i][0];x2=a[j][0];y1=a[i][1];y2=a[j][1];
                xm=(x1+x2);ym=(y1+y2);
                x3=xm-2*y2+ym;
                y3=ym+2*x2-xm;
                if(x3%2==0 && y3%2==0)
                {
                    x3/=2;y3/=2;
                    x4=x1+x2-x3;
                    y4=y1+y2-y3;
                    if((binarysearch(a,x3,y3,0,n-1)) && (binarysearch(a,x4,y4,0,n-1)))
                    count++;
                }
            }
        }
        cout<<(count/2)<<"\n";
    }
    return 0;
}
