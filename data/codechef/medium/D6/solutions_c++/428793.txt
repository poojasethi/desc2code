#include<iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<stdlib.h>
#include<malloc.h>

#define min(a,b) (a>b?b:a)
#define max(a,b) (a>b?a:b)
#define correct(a) ((a<=100 && a>=0)?1:0)
typedef long long LL;
using namespace std;

int main()
{
    //freopen("input.txt","r",stdin);
    int t;
    cin>>t;
    while(t--){
        int p[101][101]={};
        vector< pair<int,int> > points;
        int n;
        cin>>n;
        for(int i=0;i<n;i++){
            int a,b;
            cin>>a>>b;
            points.push_back(make_pair(a+50,b+50));
            p[a+50][b+50]=1;
        }
        sort(points.begin(),points.end());
        int x1,x2,y1,y2,x3,y3,x4,y4;
        int cnt=0;
        for(int i=0;i<n;i++){
            x1=points[i].first;
            y1=points[i].second;
            for(int j=i+1;j<n;j++){
                x2=points[j].first;
                y2=points[j].second;
                if(y2>y1)
                    continue;
                x3=x1+y1-y2;y3=y1+x2-x1;
                x4=x2+y1-y2;y4=y2+x2-x1;
                if(correct(x3) && correct(x4) && correct(y3) && correct(y4)){
                    if(p[x3][y3] && p[x4][y4])
                        cnt++;
                }
            }
        }
        cout<<cnt<<endl;
    }

}
