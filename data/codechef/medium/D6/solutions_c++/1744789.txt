#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,count=0;
        cin>>n;
        int a[n][2],x[102][102];
        for(int i=0;i<102;i++)
        for(int j=0;j<102;j++)
        x[i][j]=0;
        for(int i=0;i<n;i++)
        {
            cin>>a[i][0]>>a[i][1];
            a[i][0]=a[i][0]+50;
            a[i][1]=a[i][1]+50;
            x[a[i][0]][a[i][1]]=1;
        }
        for(int i=0;i<n-1;i++)
        {
            for(int j=i+1;j<n;j++)
            {
                int xm,ym,x1,y1,x2,y2,x3,y3,x4,y4;
                x1=a[i][0];x2=a[j][0];y1=a[i][1];y2=a[j][1];
                xm=(x1+x2);ym=(y1+y2);
                x3=xm-2*y2+ym;
                y3=ym+2*x2-xm;
                if(x3%2==0 && y3%2==0)
                {
                    x3/=2;y3/=2;
                    x4=x1+x2-x3;
                    y4=y1+y2-y3;
                    if(x3<=100&&x3>=0 && x4<=100&&x4>=0 && y3<=100&&y3>=0 && y4<=100&&y4>=0)
                    {
                        if(x[x3][y3] && x[x4][y4])
                        count++;
                    }
                }
            }
        }
        cout<<(count/2)<<"\n";
    }
    return 0;
}
