#include<cstdio>
#include<iostream>
#include<algorithm>
#include<set>
using namespace std;
int T;
int N,X[1000],Y[1000];
set< pair<int,int> > S;
int main()
{
	cin>>T;
	while(T--)
	{
		cin>>N;
		S=set<pair<int,int> > ();
		for(int i=0;i<N;i++)
		{
			cin>>X[i]>>Y[i];
			S.insert(make_pair(X[i],Y[i]));
		}
		int res=0;
		for(int i=0;i<N;i++)
			for(int j=0;j<N;j++) if(i!=j)
			{
				int dx=X[j]-X[i];
				int dy=Y[j]-Y[i];
				if(dx==0)
					continue;
				double m=(double)dy/dx;
				if(m<0) continue;
				if(X[i]>X[j]||Y[i]>Y[j]) continue;
				if(S.count(make_pair(X[i]-dy,Y[i]+dx))&&S.count(make_pair(X[j]-dy,Y[j]+dx)))
					res++;
			}
		cout<<res<<endl;
	}
	return 0;
}
