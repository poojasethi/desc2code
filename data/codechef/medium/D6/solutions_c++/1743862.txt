#include<iostream>
using namespace std;
int binarysearch(int a[][2],int x,int y,int lb,int ub)
{
    while(lb<=ub)
    {
        int mid=(lb+ub)/2;
        if(a[mid][0]==x && a[mid][1]==y)
        return 1;
        else if(a[mid][0]<x || (a[mid][0]==x && a[mid][1]<y))
        lb=mid+1;
        else
        ub=mid-1;
    }
    return 0;
}
void selection(int a[][2],int n)
{
    for(int i=0;i<n-1;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if((a[i][0]>a[j][0]) || ((a[i][0]==a[j][0]) && (a[i][1]>a[j][1])))
            {
                int t;
                t=a[i][0];
                a[i][0]=a[j][0];
                a[j][0]=t;
                t=a[i][1];
                a[i][1]=a[j][1];
                a[j][1]=t;
            }
        }
    }
}
