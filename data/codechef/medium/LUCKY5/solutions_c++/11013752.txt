#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
        string str;
        cin>>str;
        long long int ctr=0;
        for(long long int i=0;str[i]!='\0';++i)
        {
            if(str[i]!='4' && str[i]!='7')
            ++ctr;
        }
        cout<<ctr<<"\n";
    }
    return 0;
}
