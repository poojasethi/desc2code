#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define endll "\n";
int main()
{
	ios_base::sync_with_stdio(false);
  int t;
  string input;
  cin>>t;
  while(t--){
    cin>>input;
    long long int result=0;
    for(long long int i=0;i<input.size();i++){
      if(input[i]!='4'&&input[i]!='7'){
        result++;
      }
    }
    cout<<result<<endll;
  }
	return 0;
}
