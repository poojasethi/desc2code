#include <iostream>
#include <string>
#include <cstdlib>
#include <cmath>
using namespace std;

int main(){
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        long int len=s.length();

        long int x=0;
        for(int i=0;i<len;i++){
            if(s[i]!='4' && s[i]!='7'){
                x++;
            }
        }
        cout<<x<<endl;
    }

    return 0;

}
