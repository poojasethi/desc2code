#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        char n[100005];
        scanf("%s", n);
        int l = strlen(n), i, c = 0;
        for(i=0;i<l;i++)
        {
            if(n[i] != '4' && n[i] != '7')
            {
                c++;
            }
        }
        printf("%d\n", c);
    }
    return 0;
}
