#include<stdio.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char ch=getchar_unlocked();
        while(ch<'0'||ch>'9')
            ch=getchar_unlocked();
        long int count=0;
        while(ch>='0'&&ch<='9')
        {
            if((ch!='4')&&(ch!='7'))
                count++;
            ch=getchar_unlocked();
        }
        printf("%d\n",count);
    }
    return 0;
}
