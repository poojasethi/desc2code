#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int tc;
    scanf("%d",&tc);
    while(tc--)
    {
        long int i,count=0;
        char str[100000]={'\0'};
        scanf("%s",str);
        for(i=0;str[i]!='\0';i++)
        {
            if(str[i]!='4' && str[i]!='7')
                count++;
        }
        cout<<count<<endl;
        //printf("%d\n",count);
    }
    return 0;
}
