#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
	int t;
	scanf("%d", &t);
	char a[100001];
	while(t--)
	{
		scanf("%s", a);
		int n=0;
		int l = strlen(a);
		for(int i=0; i<l; i++)
		{
			if( a[i]!='4' && a[i]!='7' )
				n++;
		}
		printf("%d \n", n);
	}
	return 0;
}
