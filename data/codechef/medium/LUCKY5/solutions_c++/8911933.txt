#include<iostream>
#include<cstdio>
#include<cstring>
#define gi(n) scanf("%d",&n);
#define pi(n) printf("%d",n);
using namespace std;
int main()
{
    int t,count;
    char s[100001];
    gi(t);

    for(int j=0;j<t;j++)
    {
          scanf("%s",&s);
         count=0;
        for(int j=0;j<strlen(s);j++)
        {
         if(s[j] != '4' && s[j] !='7')
            count++;
        }
    cout<<count<<endl;
    }
    return 0;
}
