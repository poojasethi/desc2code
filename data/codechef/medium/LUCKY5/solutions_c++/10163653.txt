#include<iostream>
#include<string.h>
using namespace std;
int main()
{
 int t,i,j=0,x=0,y=0,p=0,l;
 char c[100010];
 cin>>t;
 while(t--)
 {
   cin>>c;
   j=0;
   x=4;
   l=strlen(c);
   for(i=0;i<l;i++)
   {
    if(c[i]!='4'&&c[i]!='7')
    j++;
   }
   if(j==2)
   {
     if((c[l-1]!='4'&&c[l-1]!='7')&&c[l-2]!='4'&&c[l-2]!='7')
     {
	if((c[l-2])<'7')
	{
	if(c[l-2]<'4')
	y=4;
	else
	y=7;
	p=(y*10+x)-(int (c[l-2]-48)*10+int (c[l-1]-48));
	if(p<=9)
	j=1;
	}
     }
   }
   cout<<j<<"\n";
   }
   return 0;
   }
