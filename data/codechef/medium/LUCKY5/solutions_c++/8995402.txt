#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    long int t;
    char num[1000001];
    cin>>t;
    while(t--)
    {
        scanf("%s",num);
        int i, digits=0;
        for(i=0; num[i]!='\0'; i++)
        {
            if(num[i]!='4'&&num[i]!='7')
                digits++;
        }
        cout<<digits<<endl;

    }
    return 0;
}
