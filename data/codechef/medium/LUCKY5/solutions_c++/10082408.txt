#include<iostream>
#include<stdio.h>
#include<string>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        int i,c=0;
        for(i=0;i<s.length();i++)
            if(s[i]!='4'&&s[i]!='7')
            c++;
        cout<<c<<"\n";
}
return 0;
}
