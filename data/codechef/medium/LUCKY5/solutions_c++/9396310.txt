#include<iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        char s[100001];
        long long int count = 0;
            scanf("%s",s);
            for(int i=0;s[i]!='\0';i++)
            {
            if(s[i]!='4' && s[i]!='7')
            {
                count++;
            }
            }
        printf("%lld\n",count);
    }
    return 0;
}
