#include<iostream>
#include<cstring>
#include<vector>
using namespace std;
int main()
{
int test,i;
string s;
cin>>test;
while(test--)
	{
	    int count=0;
	    cin>>s;
	    for(i=0;i<s.length();i++)
	    {
         if(s.at(i)=='4' || s.at(i)=='7')
            continue;
         else
            count++;
	    }
	    cout<<count<<endl;
	}
return 0;
}
