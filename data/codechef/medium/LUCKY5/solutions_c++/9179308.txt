#include<iostream>
#include<cstring>
using namespace std;
int main()
{
	int t;
	char A[100001];
	cin>>t;
	while(t--)
	{
       cin>>A;
       int c=strlen(A);
       int i=0,j;
       for(j=0;j<c;j++)
       {
       	if((A[j]-48)!=4 && (A[j]-48)!=7)
       	{
       		i++;
       	}
       }
       cout<<i<<endl;
	}
	return 0;
}
