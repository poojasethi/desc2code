def leperms():
    
    t = int(raw_input())
    
    while (t > 0):
        t -= 1
        
        lengthOfPerm = int(raw_input())
        permList = map(int, raw_input().split())
        
        localInversionCount = 0
        
        if lengthOfPerm == 1:
            print "YES"
            continue
        
        for x in xrange(1,len(permList)):
            if permList[x] < permList[x-1]:
                localInversionCount+=1
        
        inversionCount = 0
        localGreaterThanInversion = False
        
        for i in xrange(len(permList)):
            for j in xrange(i, len(permList)):
                if permList[j] < permList[i]:
                    inversionCount +=1
                    if inversionCount > localInversionCount:
                        localGreaterThanInversion = True
                if localGreaterThanInversion:
                    break
            if localGreaterThanInversion:
                break
        
        if localGreaterThanInversion:
            print "NO"
            continue
        if inversionCount < localInversionCount:
            print "NO"
            continue
        if inversionCount == localInversionCount:
            print "YES"
            continue
    
    pass

leperms()
