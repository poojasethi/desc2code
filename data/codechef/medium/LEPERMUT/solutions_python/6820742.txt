t=input()

for _ in range (t):
    size=input()
    perm=map(int,raw_input().split())
    inv,local_inv=0,0        
    for i in range(size-1):
        if perm[i]>perm[i+1]:
            local_inv+=1
        for j in range(i+1,size):
            if perm[i]>perm[j]:
                inv+=1
            
    print 'YES' if inv==local_inv else 'NO'
