for _ in xrange(int(raw_input())):
	n=int(raw_input())
	a=[eval(x) for x in raw_input().split()]
	d=0
	e=0
	if n==1:
		print "YES"
		continue
	for i in xrange(0,n):
		for j in xrange(i+1,n):
			if a[i]>a[j]:
				d=d+1
	for i in xrange(0,n-1):
		if a[i]>a[i+1]:
			e=e+1
	if d==e:
		print "YES"
	else:
		print "NO"
