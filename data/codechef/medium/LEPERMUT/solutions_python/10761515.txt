t = int(raw_input())
while t > 0 :
	n = int(raw_input())
	lst1 = raw_input()
	lst = lst1.split()
	inv = 0
	linv = 0
	for i in range(n):
		for j in range(i,n):
			if int(lst[i]) > int(lst[j]) :
				inv = inv + 1
	for k in range(n-1):
		if int(lst[k]) > int(lst[k+1]) :
			linv +=1
	if inv == linv :
		print "YES"
	else :
		print "NO"
	t = t - 1
