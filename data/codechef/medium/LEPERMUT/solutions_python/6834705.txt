t = int(raw_input())
for case in range(t):
    n = int(raw_input())
    perm = map(int, raw_input().split())
    inversions = 0
    localInv = 0
    for i in range(0, len(perm) - 1):
        for x in range(i, len(perm)):
            if perm[i] > perm[x]:
                inversions += 1
    for i in range(0, len(perm) - 1):
        if perm[i] > perm[i + 1]:
            localInv += 1
    if inversions == localInv:
        print "YES"
    else:
        print "NO"
