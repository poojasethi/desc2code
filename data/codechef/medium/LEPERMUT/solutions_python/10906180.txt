t = input();
while(t > 0):
	n = input(); a = [int(i) for i in raw_input().split()]; noi = noli = 0;
	for i in range(0, n):
		for j in range(i + 1, n):	
			if(a[i] > a[j]):	noi += 1;
	for i in range(0, n - 1):
		if(a[i] > a[i + 1]):	noli += 1;
	if(noi == noli):	print "YES";
	else:	print "NO";
	t -= 1;
