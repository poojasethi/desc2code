t=(int)(raw_input())
re=['NO' for i in range(t)]
for i in range(t):
    n=(int)(raw_input())
    coun,coun1=0,0
    arr=[int(x) for x in raw_input().split()]
    for j in range(n-1):
        for d in range(j+1,n):
            if(arr[d]<arr[j]):
                coun=coun+1
    for j in range(n-1):
        if(arr[j]>arr[j+1]):
            coun1=coun1+1
    if(coun==coun1):
        re[i]='YES'
for i in range(t):
    print re[i]
