    n = input()
    invrsns=0
    lclinvrsns=0
    s=raw_input()
    arrlist=map(int,s.split())
    for j in range(0,n-1):
        for k in range(j+1,n):
            if arrlist[j]>arrlist[k]:
                invrsns+=1
    for l in range(0,n-1):
