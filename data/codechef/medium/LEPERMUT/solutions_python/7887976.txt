for _ in range(int(raw_input())):
    N = int(raw_input())
    a = map(int, str(raw_input()).split())
    a1, a2 = 0, 0
    for i in range(N-1):
        if a[i] > a[i+1]:
            a1 += 1
        for j in range(N-i):
            if a[i] > a[i+j]:
                a2 += 1
    if a1 == a2:
        print "YES"
    else:
        print "NO"
