def fun(n):
    count=0
    d=0
    for i in range(0,len(n),1):
        for j in range(0,len(n),1):
            if j>i and int(n[i])>int(n[j]):
                d+=1
    for i in range(0,len(n)-1,1):
        if int(n[i+1])<int(n[i]):
            count+=1
    if d==count:
        return "YES"
    else:
        return "NO"
