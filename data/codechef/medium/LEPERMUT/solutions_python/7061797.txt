for z in range(int(raw_input())):
    inv=0
    l_inv=0
    
    N=int(raw_input())
    A=map(int,raw_input().split())
    for i in range(N-1):
        if A[i]>A[i+1]:
            l_inv+=1
    for i in range(N):
        for j in range(i+1,N):
            if A[i]>A[j]:
                inv+=1
    if l_inv==inv:
        print "YES"
    else:
        print "NO"
