import sys
 
for __ in range(input()) :
    n = input()
    lists = [int(i) for i in sys.stdin.readline().split()]
    loc, inv = 0, 0
    for i in range(n) :
        for j in range(i+1,n) :
            if lists[i] > lists[j] :
                inv += 1
    
    for i in range(n-1) :
        if lists[i] > lists[i+1] :
            inv -= 1
 
    if inv == 0 :
        print "YES"
    else :
        print "NO"
