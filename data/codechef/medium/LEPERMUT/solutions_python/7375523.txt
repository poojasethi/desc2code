t=int(raw_input())
for asd in range(t):
	n=int(raw_input())
	a=map(int,raw_input().split())
	cg=0
	cl=0
	for i in range (0,n-1):
		for j in range (0,i):
			if a[i]<a[j] :
				cg+=1
		if (a[i]>a[i+1]):
			cl+=1
	for j in range (0,n-1):
		if a[n-1]<a[j] :
			cg+=1
	if cg==cl:
		print "YES"
	else:
		print "NO"
