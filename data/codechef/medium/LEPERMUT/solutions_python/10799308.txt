t=int(raw_input())
def inversions(data):
	count=0
	for i in range(len(data)):
		for j in range(i+1, len(data)):
			if data[i]>data[j]:
				count+=1
	return count
def l_inversions(data):
	count=0
	for i in range(len(data)-1):
		if data[i]>data[i+1]:
			count+=1
	return count

	per=[]
	for number in input1:
		per.append(int(number))#made a number list of the permutation
	if inversions(per)==l_inversions(per):
		print "YES"
