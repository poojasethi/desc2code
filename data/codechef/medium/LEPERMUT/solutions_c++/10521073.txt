#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

int main() {
  int t, n, v;
  vector<int> val;
  scanf("%d", &t);

  for (int i = 0; i < t; ++i) {
    scanf("%d", &n);
    val.resize(n);
    
    for (int j = 0; j < n-1; j++) {
      scanf("%d ", &v);
      val[j] = v;
    }
    scanf("%d", &v);
    val[n-1] = v;

    int local = 0, global = 0;
    for (int i = 0; i < n - 1; ++i) {
      for (int j = i + 1; j <  n; ++j) {
	if (val[i] > val[j]) {
	  global++;
	  if (j == i + 1) local++;
	}
      }
    }

    if (local == global) printf("YES\n");
    else printf("NO\n");
  }
  return 0;
}
