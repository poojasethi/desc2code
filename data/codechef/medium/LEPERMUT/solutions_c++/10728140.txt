#include<iostream>
using namespace std;
#define size 100
int getinversion(int* a,int n)
{
	int i,c=0,j;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i]>a[j])
			c++;
		}
	}
	return c;
}
int getlocalinversion(int* a,int n)
{
	int i,j,c=0;
	for(i=0;i<n-1;i++)
	{
		if(a[i]>a[i+1])
		c++;
	}
	return c;	
}
int main()
{
	int t,i,n,a[size],c1,c2;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		if(n==1)
		cout<<"YES"<<"\n";
		else
		{
			c1=getinversion(a,n);
			c2=getlocalinversion(a,n);	
			if(c1==c2)
			cout<<"YES"<<"\n";
			else
			cout<<"NO"<<"\n";
		}
		

	}
	return 0;
}
