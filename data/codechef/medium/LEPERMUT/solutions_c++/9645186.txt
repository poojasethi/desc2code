#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,arr[1000],inversion=0,localin=0,i,j;
    cin>>t;
    while(t!=0)
    {
        cin>>n;
        inversion=0;
        localin=0;
        for(i=0;i<n;i++)
            cin>>arr[i];
        for(i=0;i<n-1;i++)
        {

            for(j=i+1;j<n;j++)
            {
                if(arr[j]<arr[i])
                {
                    inversion++;
                }

            }
        }
        for(j=0;j<n-1;j++)
        {
            if(arr[j+1]<arr[j])
                localin++;
        }


        if(localin==inversion)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;






        t--;
    }

return 0;
}
