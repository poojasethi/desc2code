#include <iostream>
using namespace std;

int main(){
    int test;
    cin >> test;
    for(int i = 1; i <= test; i++){
        int n;
        cin >> n;
        int A[105][105] = {0};
        int a[105];
        for(int j = 1; j <= n; j++)
            cin >> a[j];
        if( n == 1){
                cout << "YES\n";
        }else{
            int count1 = 0, count2 = 0;
            for(int j = 1; j <= n; j++){
                for(int k = j + 1; k <= n; k++){
                    if(a[j] > a[k]){
                        count1++;
                    }
                }
            }
            for(int j = 1; j < n; j++){
                if(a[j] > a[j+1]){
                    count2++;
                }
            }
            if(count1 == count2){
                cout << "YES\n";
            }
            else{
                cout << "NO\n";
            }
        }
    }
return 0;
}
