#include<stdio.h>
int main(){
	int t, n, i, j; scanf("%d", &t);
	while(t-- && scanf("%d", &n)){
		int a[n], noi = 0, noli = 0;
		for(i = 0; i < n; scanf("%d", &a[i++]));
		for(i = 0; i < n; i++)	for(j = i + 1; j < n; j++)	if(a[i] > a[j])	noi++;
		for(i = 0; i < n - 1; i++)	if(a[i] > a[i + 1])	noli++;
		(noi == noli) ? printf("YES\n") : printf("NO\n");
	}		
	return 0;
}
