# include<iostream>
#include <stdlib.h>

using  namespace std;
int A[100];

int main(){
    int t;
    cin >> t;
    while(t--){

        int n;
        cin >> n;


        int local = 0, inv = 0;

        for(int i = 0; i < n; i++){
            cin >> A[i];
        }

        for(int i = 0; i < n; i++){
         //   cout << A[i];
            if((A[i] > A[i+1]) && ( i + 1 <= n - 1))
                local = local + 1;
            for(int j = i + 1 ; j < n; j++){
                if((A[i] > A[j]) && ( j <= n-1))
                    inv = inv + 1;
            }
        }
        //cout << local << inv;
        if(local == inv)
            cout << "YES\n";
        else cout << "NO\n";
    }

    return 0;
}
