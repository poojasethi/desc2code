#include<iostream>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
int n;
cin>>n;
int a[n];
for(int i=0;i<n;i++)
cin>>a[i];
int res=0,ans=0;
if(n==1)
cout<<"YES"<<endl;
else
{
for(int i=0;i<n;i++)
{
for(int j=i+1;j<n;j++)
{
if(a[i]>a[j])
res++;
}
}
for(int i=0;i<n-1;i++)
{
if(a[i]>a[i+1])
ans++;
}
if(res==ans)
cout<<"YES"<<endl;
else
cout<<"NO"<<endl;
}
}
return 0;
}
