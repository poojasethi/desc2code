#include<cstdio>
#include<cstring>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int N;
        scanf("%d",&N);
        int arr[N];
        for(int i=0;i<N;i++){
            scanf("%d",&arr[i]);
        }
        int global_inversion=0,local_inversion=0;
        for(int i=0;i<N;i++){
            for(int j=i+1;j<N;j++){
                if(arr[j]<arr[i]){
                    global_inversion++;
                }
            }
        }
        for(int i=0;i<N-1;i++){
         if(arr[i]>arr[i+1]){
             local_inversion++;
        }
        }
        if(local_inversion==global_inversion){
            printf("YES\n");
        }
        else{
            printf("NO\n");
        }
    }
}
