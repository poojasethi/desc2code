#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,loc_inv=0,inv=0;
        cin>>n;
        int a[n+1];
        for(int i=1;i<=n;i++)
            cin>>a[i];
        for(int i=1;i<n;i++)
        {
            for(int j=i+1;j<=n;j++)
            {
                if(a[i]>a[j])
                    inv++;
            }
            if(a[i]>a[i+1])
                loc_inv++;
        }
        if(loc_inv==inv)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
}
