int main() {
	int t;
	S(t);
	while(t--) {
        int n, i , j, c1 = 0, c2 = 0;
        S(n);
        int a[n];
        for (i = 0; i < n; i++)
        {
            S(a[i]);
        }
        for (i = 0; i < n - 1; i++)
        {
            if (a[i] > a[i + 1]) {
                c1++;
            }
        }
        for (i = 0; i < n - 1; i++)
        {
            for ( j = i + 1; j < n; j++)
            {
                if (a[i] > a[j])
                {
                    c2++;
                }
            }
        }
        if (c1 == c2)
        {
            cout << "YES" << endl;
        }
        else {
            cout << "NO" << endl;
        }
	}
    return 0;
}
