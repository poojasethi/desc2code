#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++){
			cin>>a[i];
		}
		int count=0,counti=0;
		for(int i=0;i<n;i++)
		{
			for(int j=i+1;j<n;j++){
				if(a[i]>a[j])count++;
			}
		}
		for(int i=0;i<n-1;i++){
			if(a[i]>a[i+1])counti++;
		}
		if(count==counti)cout<<"YES\n";
		else cout<<"NO\n";
	}
	return 0;
}
