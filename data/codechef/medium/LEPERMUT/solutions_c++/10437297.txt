#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t,n,inv,linv;
	cin>>t;
	while(t--){
		cin>>n;
		int arr[n];
		inv=0;
		linv=0;
		for(int i=0; i<n;i++){
			cin>>arr[i];
		}
		for(int i=0;i<n-1;i++){
			for(int j=i+1;j<n;j++){
				if(arr[i]>arr[j]){
					inv++;
				}
			}
		}
		for(int i=0;i<n-1;i++){
			if(arr[i]>arr[i+1]){
				linv++;
			}
		}
		if(inv==linv) cout<<"YES\n";
		else cout<<"NO\n";
	}
	return 0;
}
