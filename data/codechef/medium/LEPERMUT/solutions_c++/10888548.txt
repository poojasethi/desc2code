#include<iostream>
using namespace std;
int main(){
	int t,n,i,j,c,c1;
	cin>>t;
	while(t--){
		cin>>n;
		int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		for(i=0,c=0,c1=0;i<n;i++){
			for(j=i+1;j<n;j++){
				if(a[i]>a[j])
				++c;
				if(j==i+1&&a[i]>a[j])
				++c1;
			}
		}
		if(c==c1)
		cout<<"YES";
		else
		cout<<"NO";
		cout<<endl;
	}
	return 0;
}
