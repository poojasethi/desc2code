#include<iostream>
using namespace std;

int main()
{
	int n,t,lcount,scount;
	cin>>n;
	
	for(int i=0;i<n;i++)
	{
		lcount=0;
		scount=0;
		cin>>t;
		
		int arr[t];
		for(int j=0;j<t;j++)
		{
			cin>>arr[j];
		}
		for(int j=0;j<t-1;j++)
		{
			for(int k=j+1;k<t;k++)
			{
				if(arr[j]>arr[k])
					scount++;
			}
		}
		for(int j=0;j<t-1;j++)
		{
			if(arr[j]>arr[j+1])
				lcount++;
		}
		if(lcount==scount)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;	
	}
	
	return 0;
}
