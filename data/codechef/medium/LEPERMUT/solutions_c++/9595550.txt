#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define in(t) scanf("%lld", &t)
#define rep(i,n) for(i=1;i<=n;i++)

int main()
{
	ll t,i,j,n;
	in(t);
	while(t--)
	{
		in(n);
		ll a[105];
		rep(i,n)
		{
			in(j);
			a[i]=j;
		}
		int flag =0;
		for(i=1;i<=n;i++)
		{
			for(j=i+2;j<=n;j++)
			{
				if(a[i]>a[j])
				{
					flag=1;
					goto en;
				}
			}
		}
		en:
		(flag==1)?cout<<"NO\n":cout<<"YES\n";
	}
	return 0;
}
