#include<iostream>
#include<math.h>
#include<cstdio>

using namespace std;

int main()
{
    int t,n,i,j,a[101],k,ans;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>n;
        for(j=0;j<n;j++)
        {
            cin>>a[j];
        }
        ans=1;
        for(j=0;j<n-2;j++)
        {
            for(k=j+2;k<n;k++)
            {
                if(a[j]>a[k])
                {
                    ans=0;
                    break;
                }
            }
            if(ans==0)
            {
                cout<<"NO\n";
                break;
            }
        }
        if(ans==1)
        {
            cout<<"YES\n";
        }

    }
    return 0;
}
