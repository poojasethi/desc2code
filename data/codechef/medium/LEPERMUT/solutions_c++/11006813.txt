#include<iostream>
using namespace std;

int main(){
   int numbers;
   cin>>numbers;
   int s=0;
   while(s<numbers){
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++){
           cin>>a[i];}
        int localInv=0;
        for(int j=0;j<n-1;j++){
            if(a[j]>a[j+1]){
                localInv+=1;}}

        int inversions=0;
        for(int p=0;p<n-1;p++){
           for(int q=p+1;q<n;q++){
                if(a[p]>a[q]){
                    inversions+=1;}}}

       if(inversions==localInv){
             cout<<"YES"<<endl;}
       else{
             cout<<"NO"<<endl;}
       s+=1;}}
