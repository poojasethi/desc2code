# include<iostream>
using namespace std;
 
int main()
{
  int t,i,j,k,x=0,y=0,n;
  cin>>t;
  for(k=1;k<=t;k++)
  {
    cin>>n;
    int a[n+1];
    for(i=1;i<=n;i++)
      cin>>a[i];
    
    for(i=1;i<n;i++)
    {
      for(j=i+1;j<=n;j++)
      {
        if(a[i]>a[j])
          x++;
      }
    }
    
    for(i=1;i<n;i++)
    {
      if(a[i]>a[i+1])
        y++;
    }
    if(x==y)
      cout<<"YES"<<endl;
    else
      cout<<"NO"<<endl;
    
    x=0;
    y=0;
  }
  return 0;
}
