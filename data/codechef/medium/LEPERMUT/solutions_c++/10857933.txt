#include<iostream>
using namespace std ;
int main(){
	int testCase,n;
	cin>>testCase;
	for(int i=0;i<testCase;i++){
		cin>>n;
		int a[n],co=0,ct=0;
		for(int j=0;j<n;j++){
			cin>>a[j];
		}
		for(int i=0;i<n;i++){
			for(int j=i+1;j<n;j++){
				if(a[i]>a[j]){
					co++;
				}
			}
		}
		for(int i=0;i<n-1;i++){
			if(a[i]>a[i+1]){
				ct++;
			}
		}
		if(co==ct){
			cout<<"YES"<<endl;
		}
		else
		cout<<"NO"<<endl;
	}
}
