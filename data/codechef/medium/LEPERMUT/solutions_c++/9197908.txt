#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t, n, a[100], i, j, inv, linv;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		inv=0;
		linv=0;
		for(i=0;i<n-1;i++)
		for(j=i+1;j<n;j++)
		if(a[i]>a[j])
		inv++;
		for(i=0;i<n-1;i++)
		if(a[i]>a[i+1])
		linv++;
		if(inv==linv)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
	 } 
}
