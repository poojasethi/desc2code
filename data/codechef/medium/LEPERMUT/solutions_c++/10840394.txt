#include <iostream>
using namespace std;

int main() {
	int local,general,t,n,*per,j,i;
	cin>>t;
	while(t--)
	{
		cin>>n;
		per=new int[n];
		j=0,local=0,general=0;
		cin>>per[0];
		if(n==1)
		cout<<"YES"<<endl;
		else
		{
		for(i=1;i<n;i++)
		{
			cin>>per[i];
			if(per[j++]>per[i])
			local++;
		}
		for(i=0;i<n;i++)
		{
			j=i+1;
			while(j<n)
			{
				if(per[i]>per[j++])
				general++;
			}
		}	
		if(local==general)
		cout<<"YES"<<endl;
		else
		cout<<"NO"<<endl;
		}
		delete []per;
        }	
	return 0;
}
