#include<iostream>
using namespace std;
int main()
{
   int t;
   int n;
   int a[110];
   int inv;
   int l_inv;
   cin>>t;
   while(t--)
   {
      cin>>n;
      for(int i=0;i<n;i++)
      cin>>a[i];
      inv=l_inv=0;
      for(int i=0;i<n;i++)
      {
         if(i<n-1 && a[i]>a[i+1])
         l_inv++;
         for(int j=i+1;j<n;j++)
         {
           if(a[i]>a[j])
           inv++;
         }
      }

      if(inv==l_inv)
        cout<<"YES\n";
      else
        cout<<"NO\n";
   }

}
