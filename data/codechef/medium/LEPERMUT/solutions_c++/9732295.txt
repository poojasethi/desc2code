#include <iostream>

using namespace std;
int a[100];
int main()
{
    int t,i,j,n,in,lin;
    cin>>t;
    while(t--)
    {
        cin>>n;
        in=lin=0;
        for(i=0;i<n;i++)
        cin>>a[i];
        for(i=0;i<n-1;i++)
        {
            for(j=i+1;j<n;j++)
                if(a[i]>a[j])
                in++;
        }
        for(i=0;i<n-1;i++)
        {
            if(a[i]>a[i+1])
                lin++;
        }
        if(in==lin)
            cout<<"YES\n";
        else
            cout<<"NO\n";

    }
    return 0;
}
