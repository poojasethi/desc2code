#include <iostream>
#include <cstdio>

using namespace std;

int main(int argc, char const *argv[])
{
    int t, n, i, j, inversionCount, localInversionCount;

    scanf("%d", &t);

    while(t--)
    {
        localInversionCount = 1;
        inversionCount = 1;

        scanf("%d", &n);

        int *ptr = new int [n];

        for(i = 0; i < n; ++i)
        {
            scanf("%d", ptr+i);
        }

        for(i = 1; i < n; ++i)
        {
            if(ptr[i-1] > ptr[i])
                ++localInversionCount;
        }

        for(i = 0; i < n-1; ++i)
        {
            for(j = i+1; j < n; ++j)
            {
                if(ptr[i] > ptr[j])
                    ++inversionCount;
            }
        }

        if(inversionCount == localInversionCount)
            printf("YES\n");
        else
            printf("NO\n");

        delete ptr;
    }
    return 0;
}
