#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int a[n],f=1;
		for(int i=0;i<n;i++)
		cin>>a[i];
		if(n==1)
		cout<<"YES"<<endl;
		else
		{
			for(int i=0;i<n-1;i++)
			{
				if(a[i]>a[i+1])
				{
					swap(a[i],a[i+1]);
					i++;
				}
			}
			for(int i=0;i<n-1;i++)
			{
				if(a[i]>=a[i+1])
                		{
                    			f=0;
                    			break;
                		}
			}
			if(f==1)
                	cout<<"YES"<<endl;
            		else
                	cout<<"NO"<<endl;
		}
	}
	return 0;
}
