#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,j,l,g;
    cin>>t;
    while(t--)
    {
        l=0,g=0;
        cin>>n;
        int ar[n];
        cin>>ar[0];
        for(i=1;i<n;i++)
        {
            cin>>ar[i];
            if(ar[i]<ar[i-1])
            l++;
        }
        for(i=0;i<n;i++)
        {
            for(j=i+1;j<n;j++)
            {
                if(ar[i]>ar[j])
                g++;
            }
        }
        if(l==g)
        cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
    return 0;
}
