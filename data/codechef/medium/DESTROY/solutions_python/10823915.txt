import operator
t=int(raw_input())
for i in xrange(t):
    dic={}
    n=int(raw_input())
    x=list(map(int,raw_input().split()))
    for j in xrange(n):
        if x[j] in dic:
            dic[x[j]]+=1
        else:
            dic[x[j]]=1
    p=max(dic.iteritems(),key=operator.itemgetter(1))[1]
    if n%2!=0:
        if p>(n/2):
            print p
        else:
            print (n+1)/2
    else:
        if p>=(n/2):
            print p
        else:
            print n/2
