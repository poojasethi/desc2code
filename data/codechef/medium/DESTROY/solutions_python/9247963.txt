from sys import stdin, stdout
t = int(stdin.readline())
while t:
	t -= 1
	n = int(stdin.readline())
	a = map(int, stdin.readline().strip().split(' '))
	c = {}
	for i in a:
		if i in c:
			c[i] += 1
		else:
			c[i] = 1
	maxi = max(c.values())
	if maxi > n //2 :
		val = n - maxi
		val += 2*maxi - n
	else:
		val = n / 2
		if n%2:
			val += 1
	stdout.write(str(val)+"\n")
