#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		int flag=0;
		map<long long int,long long int> m;
		long long int n;
		cin>>n;
		long long int num=0,val=0;
		map<long long int,long long int>::iterator ii;
		for(long long int i=0;i<n;i++)
		{
			cin>>num;
			if(m.find(num)!=m.end())
			{
				ii=m.find(num);
				(*ii).second++;
				if((*ii).second>=(n+1)/2)
				{
					flag=1;
					val=num;
				}
			}
			else
			m[num]=1;
			
		}
		if(flag==1)
		{
			ii=m.find(val);
			cout<<(*ii).second<<"\n";
		}
		else
		cout<<(n+1)/2<<"\n";
	}
}
