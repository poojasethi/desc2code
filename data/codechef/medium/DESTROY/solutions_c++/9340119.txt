#include<bits/stdc++.h>
#include<map>
#include<iterator>
using namespace std;

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,ans;
		scanf("%d",&n);
		int arr[n],i;
		map<int,int> m;
		map<int,int>::iterator itr;
		for(i=0;i<n;i++)
		{
		scanf("%d",&arr[i]);
		m[arr[i]]++;
		}
		int maxi=0;
		for(itr=m.begin();itr!=m.end();itr++)
		maxi=max(maxi,itr->second);
		ans=max(maxi,(n+1)/2);
		printf("%d\n",ans);
	}
	return 0;
}
