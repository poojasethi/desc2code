#include<iostream>
#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
    int ar[50005];
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            scanf("%d",&ar[i]);
        }
        sort(ar,ar+n);
        int ans=0;
        for(int i=0;i<n/2;i++)
        {
            ans++;
            ans+=!(ar[i]-ar[(n+1)/2+i]);
        }
        ans+=n&1;
        cout<<ans<<"\n";
    }
    return 0;
}
