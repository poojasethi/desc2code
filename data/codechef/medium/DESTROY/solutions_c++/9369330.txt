#include <bits/stdc++.h>

using namespace std;
int arr[50004];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int n,i;
        priority_queue<int>P;
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%d",&arr[i]);
        sort(arr,arr+n);
        int c;
        for(i=0;i<n;i++)
        {
            int j=arr[i];
            c=0;
            while(i<n && arr[i]==j){
                c++;
                i++;
            }
            P.push(c);
            if(i<n)
                i--;
        }
        //cout<<endl;
        int sum=0;
        while(!P.empty()){
            int x=P.top();
            P.pop();
            if(P.empty()){
                sum+=x;
                break;
            }
            int y=P.top();
            P.pop();
            sum++;
            x--;
            y--;
            if(x>0)
                P.push(x);
            if(y>0)
                P.push(y);
        }
        printf("%d\n",sum);
    }
    return 0;
}
