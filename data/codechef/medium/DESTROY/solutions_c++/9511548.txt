#include<bits/stdc++.h>
 
using namespace std;
 
const int MX = 500000;
 
int a[MX], c[MX];
priority_queue<int> q;
 
int main() {
	int t;
	scanf("%d", &t);
	while (t--) {
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; i++) scanf("%d", a + i);
		
		sort(a, a + n);
		
		for (int i = 1, j = 1; i <= n; i++) {
			if (i == n || a[i] != a[i - 1]) {
				q.push(j);
				
				j = 0;
			}
			
			j++;
		}
		
		int ans = 0;
		while (q.size() > 1) {
			int x = q.top();
			q.pop();
			int y = q.top();
			q.pop();
			
			ans++;
			
			if (x > 1) q.push(x - 1);
			if (y > 1) q.push(y - 1);
		}
		
		if (q.size() == 1) {
			ans += q.top();
			q.pop();
		}
		
		printf("%d\n", ans);
	}
	
	return 0;
}
