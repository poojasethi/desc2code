#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
 
int main()
{
	long t;
	scanf("%ld", &t);
	while(t--)
	{
		long n;
		scanf("%ld", &n);
		long a[n];
		long cnt=0;
		for(int i=0; i<n; i++)
			scanf("%ld", &a[i]);
		sort(a, a+n);
		if( n%2 )
		{
			for(long i=0; i<n/2; i++)
			{
				if( a[i]!=a[n/2+i+1] )
					cnt++;
				else
					cnt += 2;
			}
			cnt++;
		}
		else
			for(long i=0; i<n/2; i++)
			{
				if( a[i]!=a[n/2+i] )
					cnt++;
				else
					cnt += 2;
			}
		printf("%ld \n", cnt);
	}
	return 0;
}
