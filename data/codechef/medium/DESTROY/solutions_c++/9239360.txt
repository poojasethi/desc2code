#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
using namespace std;
#define MAX 50000

vector<int> vec(MAX);

int main(){
  int t,n,i,j,ans,max_freq;
  scanf("%d",&t);

  while(t--){
    scanf("%d",&n);

    for(i=0;i<n;++i)
      scanf("%d",&vec[i]);

    sort(vec.begin(),vec.begin()+n);

    i=max_freq=0;
    while(i<n){
      j=i+1;
      while(j<n && vec[i] == vec[j])
        ++j;
      if((j-i)>max_freq)
        max_freq=j-i;
      i=j;
    }
    ans = max(max_freq,(n+1)/2);
    printf("%d\n",ans);
  }
  return 0;
}
