int destroy(long long *arr,int n)
{
int max=1,k=1;
for(int i=1;i<n;i++)
{
if(arr[i]==arr[i-1])
k++;
else
{
if(k>max)max=k;
k=1;
}
}
if(k>max)max=k;
if(max < ((n+1)/2))return ((n+1)/2);
else
return max;
}
