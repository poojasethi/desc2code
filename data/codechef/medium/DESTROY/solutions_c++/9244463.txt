#include <iostream>
#include <string>
#include <map>
#include <set>
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <vector>
#include <queue>

using namespace std;

int main() {
	int t;
	scanf("%d", &t);
	while(t--) {
		int n,sum=0;
		map<long long int, int> counts;
		long long int tmp;
		cin>>n;
		for (int ctr=0; ctr < n; ctr++) {
			cin>>tmp;
			counts[tmp]++;
		}
		map<long long int, int>::iterator itr = counts.begin();
		priority_queue<int, vector<int> > pq;
		while(itr != counts.end()) {
			pq.push((itr++)->second);
		}
		while(pq.size() > 1) {
			int num1=pq.top();pq.pop();
			int num2=pq.top();pq.pop();
			num1--; num2--;
			sum++;
			if(num1>0) pq.push(num1);
			if(num2>0) pq.push(num2);
		}
		std::cout<<sum+(pq.empty()?0:pq.top())<<"\n";
	}
	return 0;
}
