#include<stdio.h>
#include<algorithm>
using namespace std;
long int max(long int a,long int b)
{

if(a>b)
return a;
return b;
}
int main()
{
   long	int t,n;
	long int arr[50001]={0};
   scanf("%ld",&t);
   while(t)
   {
   	scanf("%ld",&n);
   	for(int i=0;i<n;i++)
   	scanf("%ld",&arr[i]);
   	sort(arr,arr+n);
   	long int freq=1,max_freq=0;
   	long int last=arr[0];
    for(int i=1;i<n;i++)
    {
    	if(last==arr[i])
    	freq++;
    	else
    	{freq=1;last=arr[i];
		}
	    if(max_freq<freq)
	    {
	    	max_freq=freq;
		}
	}
    long int mx=max((n+1)/2,max_freq);
    printf("%ld\n",mx);
   t--;
   }
}
