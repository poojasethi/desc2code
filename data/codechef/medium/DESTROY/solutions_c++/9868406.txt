#include <bits/stdc++.h>
using namespace std;
int main(){
	long long t,n,i,j,temp;
	scanf("%lld",&t);
	while(t--){
		vector <long long> a;
		scanf("%lld",&n);
		for(i=0;i<n;++i){
			scanf("%lld",&j);
			a.push_back(j);
		}
		sort(a.begin(),a.end());
		temp=i=0;
		while(i<n){
			j=i+1;
			while(j<n and a[i]==a[j]) j++;
			temp=max(temp,j-i);
			i=j;
		}
		if(temp>=(n+1)>>1) printf("%lld\n",temp);
		else printf("%lld\n",(n+1)>>1);
	}
	return 0;
}
