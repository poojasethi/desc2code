int main()
{
	lld t,n,i,j,temp;
	t=scan_lld();
	while(t--)
	{
		vlld a;
		n=scan_lld();
		fr(i,n)
		{
			j=scan_lld();
			a.pb(j);
		}
		sort(all(a));
		temp=i=0;
		while(i<n)
		{
			j=i+1;
			while(j<n and a[i]==a[j])
				j++;
			temp=max(temp,j-i);
			i=j;
		}
		if(temp>=(n+1)/2)
			println_lld(temp);
		else
			println_lld((n+1)/2);
	}
	return 0;
}
