#include <bits/stdc++.h>
using namespace std;
 
int main() {
	// your code goes here
	int t;
	cin >> t;
	while(t--)  {
	  int n;
	  cin>>n;
	  int a[n];
		for(int i=0;i<n;i++)
		   cin>>a[i];
		sort(a,a+n);
		int c=1;
		int ans=(n+1)/2;
		for(int i=1;i<n;i++)
		{
			if(a[i]==a[i-1])
			   c++;
			else
			{
				ans=max(ans,c);
				c=1;
			}
		}
		cout<< max(ans,c) << endl;
   }
	return 0;
}
