# include <iostream>
# include <cstdio>
# include <algorithm>
# include <cmath>

using namespace std ;

int main ( ) {
	ios_base::sync_with_stdio ( false ) ;
	int tc ; cin >> tc ;
	while ( tc-- ) {
		int n ; cin >> n ;
		int a[n+1] , i , j , max = 1 ;
		for ( i = 0 ; i < n ; i += 1 ) 
			cin >> a[i] ;

		sort ( a , a+n ) ;

		i = 0 ;
		while ( i < n ) {
			j = i+1 ;
			while ( j < n && a[i] == a[j] )
				++j ;

			if ( max < j-i )
				max = j-i ;
			i = j ;
		}

		if ( max <= n/2 )
			cout << ( n+1 ) / 2 << endl ;
		else
			cout << max << endl ;

	}

	return 0 ;

}
