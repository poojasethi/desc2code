#include <bits/stdc++.h>
using namespace std;
const int N = 5e5+1;
int a[N];
priority_queue < int, vector <int> > pq;
vector<int>ct;
int main(){
    int t;
    cin >> t;
    while(t--){
        int n,ans=0;
        ct.clear();
        cin >> n;
        for(int i = 0; i < n; i++){
            cin >> a[i];
        }
        while(!pq.empty()) pq.pop();
        sort(a,a+n);
        int i = 0;
        while(i < n){
            int cur = a[i], x = 0;
            while(a[i]==cur && i < n){
                x++; i++;
            }
            ct.push_back(x);
        }
        for(int i = 0; i < ct.size(); i++) pq.push(ct[i]);
        while(!pq.empty()){
            if(pq.size()==1){
                ans+=pq.top();
                pq.pop();
            }else{
                //while(pq.top()==0 && !pq.empty()) pq.pop();
                if(!pq.empty()){
                    int cur = pq.top();
                    pq.pop();
                    int c2 = pq.top();
                    pq.pop();
                    cur--;
                    c2--;
                    if(cur>0)pq.push(cur);
                    if(c2>0)pq.push(c2);
                    ans++;
                }
            }
        }
        cout << ans << endl;
    }
}
