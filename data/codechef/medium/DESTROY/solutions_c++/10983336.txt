    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
        int t;
        cin >> t;
        while(t--)
        {
            int n;
            cin >> n;
            int arr[n];
            for(int i = 0; i<n; i++)
                cin >> arr[i];
            sort(arr,arr+n);
            priority_queue<int> v;
            int i = 0;
            while(i<n)
            {
                int j = i+1;
                while(j<n && arr[j]==arr[i])
                    j++;
                v.push(j-i);
                i = j;
            }
            cout << max(v.top(),(n+1)/2) << "\n";
        }



        return 0;
    }
