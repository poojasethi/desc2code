#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e6 + 10;
int A[MAXN];
vector<int> B;
int main(){
    int t;
    cin >> t;
    while(t--){
        int N;
        scanf("%d", &N);
        for(int i = 1;i <= N;++i){
            scanf("%d", &A[i]);
        }
        sort(A + 1, A + 1 + N);
        int cnt = 1;
        int ele = A[1];
        B.clear();
        for(int i = 2;i <= N;++i){
            if(A[i] != ele){
                B.push_back(cnt);
                cnt = 1;
                ele = A[i];
            } else {
                ++cnt;
            }
        }
        int ans = 0;
        B.push_back(cnt);
        make_heap(B.begin(), B.end());
        while(!B.empty()){
            int x = B.front();
            pop_heap(B.begin(), B.end());
            B.pop_back();
            if(B.empty()){
                ans += x;
            } else {
                int y = B.front();
                pop_heap(B.begin(), B.end());
                B.pop_back();

                if(x - 1 > 0){
                    B.push_back(x - 1);
                    push_heap(B.begin(), B.end());
                }

                if(y - 1 > 0){
                    B.push_back(y - 1);
                    push_heap(B.begin(), B.end());
                }
                ++ans;
            }
        }
        cout << ans << "\n";
    }
    return 0;
}
