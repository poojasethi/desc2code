#include<stdio.h>
#include<algorithm>
int main()
{
	long int t;
	scanf("%ld" ,&t);
	while(t--)
	{
	long long int n,i,j,count=0,arr[50005];
		scanf("%lld" ,&n);
		for(i=0;i<n;i++)
		{
			scanf("%lld" ,&arr[i]);
		}
		std :: sort(arr,arr+n);
		if(n%2)
		{
			count++;
		 	for(i=0;i<n/2;i++)
			{
					if(arr[i]!=arr[n/2+i+1])
					count++;
					else
					count+=2;
 
			}
		}
		else
		{
			for(i=0;i<n/2;i++)
			{
				if(arr[i]!=arr[n/2+i])
				count++;
				else
				count+=2;
			}
		}
		printf("%lld\n" ,count);
	}
	return 0;
}
