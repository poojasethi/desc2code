#include<iostream>
#include<algorithm>
using namespace std;
int func(long int a,long int b)
{
    return(a<b);
}
int main()
{
    int t,i,j,n,count,f;
    long int a[50000];
    cin>>t;
    while(t--)
    {
        cin>>n;
        count=0;
        int maxf=0;
        for(i=0;i<n;i++)
        cin>>a[i];
        sort(a,a+n,func);
       for(i=0;i<n;i++)
       {
           f=1;
           j=i+1;
           while(a[j]==a[i]&&j<n)
           {
               f++;
               j++;
           }
           if(f>maxf)
            maxf=f;
           i=j-1;
       }
       if(maxf>n/2)
        count=maxf;
       else
       {
           count=n/2 + n%2;
       }
       cout<<count<<"\n";
    }
    return 0;
}
