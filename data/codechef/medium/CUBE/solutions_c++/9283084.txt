#include<stdio.h>
 
int main()
{
	int t,n,p,i,j,k;
	char c;
	
	scanf("%d",&t);
	
	while(t--)
	{
		scanf("%d%d",&n,&p);
		
		int ar[n+1][n+1][n+1];
		char str[n*n*n+1];
		int dp[n+1][n+1][n+1];
		
		scanf("%d",&c);
	
		scanf("%s",str);
		
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		for(k=0;k<n;k++)
		ar[i+1][j+1][k+1]=str[n*n*i+n*j+k];
		
		scanf("%d",&c);
	
		scanf("%s",str);
		
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		for(k=0;k<n;k++)
		ar[i+1][j+1][k+1]=ar[i+1][j+1][k+1]-str[n*n*i+n*j+k];
		
		for(i=0;i<=n;i++)
		for(j=0;j<=n;j++)
		for(k=0;k<=n;k++)
		dp[i][j][k]=0;
		
		for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
		for(k=1;k<=n;k++)
		{
			dp[i][j][k]=dp[i-1][j][k]+dp[i][j-1][k]+dp[i][j][k-1]-dp[i-1][j-1][k]-dp[i-1][j][k-1]-dp[i][j-1][k-1]+dp[i-1][j-1][k-1];
			
			if(ar[i][j][k]==0)
			dp[i][j][k]+=1;
		}
		
		int m;
		int ans,num=0;
		
		for(m=n;m>=1;m--)
		{
			for(i=m;i<=n;i++)
			for(j=m;j<=n;j++)
			for(k=m;k<=n;k++)
			{
				ans=dp[i][j][k]-dp[i-m][j][k]-dp[i][j-m][k]-dp[i][j][k-m]+dp[i-m][j-m][k]+dp[i-m][j][k-m]+dp[i][j-m][k-m]-dp[i-m][j-m][k-m];
				
				if(ans>=float((float(p)/100.0)*m*m*m))
				num++;
			}
			
			if(num>0)
			break;
		}
		
		if(num>0)
		printf("%d %d\n",m,num);
		
		else
		printf("-1\n");
	}
	
	return 0;
}
