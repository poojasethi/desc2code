#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cstring>
#include<cmath>
double EPS=1e-8;
using namespace std;
int T,n,p,dp[41][41][41];
#ifndef ONLINE_JUDGE
inline int getchar_unlocked()
{
	return getchar();
}
#endif
inline int input()
{
	int n = 0, c = getchar_unlocked(),t=0;
	while(c < '0' || c > '9')
	{
		if(c==45)t=1;
		c = getchar_unlocked();
	}
	while(c >= '0' && c <= '9')n = (n<<3) + (n<<1) + c - '0', c = getchar_unlocked();
	
	if(t==0)return n;
	else return -n;
}
int main()
{
	char s1[41*41*41],s2[41*41*41];
	T=input();
	while(T--)
	{
		n=input();p=input();
		scanf("\n%[^\n]\n%[^\n]",s1,s2);
		if(p==0)printf("%d %d\n",n,1);
		else
		{
			for(int i=1;i<=n;i++)for(int j=1;j<=n;j++)
			{
				int d=(i-1)*n*n+(j-1)*n;
				for(int k=1;k<=n;k++)
				dp[i][j][k]=dp[i][j-1][k]+dp[i][j][k-1]-dp[i][j-1][k-1]+dp[i-1][j][k]-dp[i-1][j-1][k]-dp[i-1][j][k-1]+dp[i-1][j-1][k-1]+(s1[d+k-1]==s2[d+k-1]);
			}
			
			int size,x,cmn,p1,count=0;
			for(size=n;size>=1&&count==0;size--)
			{
				x=size*size*size;
				for(int i=1;i+size-1<=n;i++)
				{
					for(int j=1;j+size-1<=n;j++)
					{
						for(int k=1;k+size-1<=n;k++)
						{
							cmn=dp[i+size-1][j+size-1][k+size-1]-dp[i+size-1][j-1][k+size-1]-dp[i+size-1][j+size-1][k-1]+dp[i+size-1][j-1][k-1]-dp[i-1][j+size-1][k+size-1]+dp[i-1][j-1][k+size-1]+dp[i-1][j+size-1][k-1]-dp[i-1][j-1][k-1];
							p1=ceil((double)x*((double)p/100.0));
							if(cmn>=p1)count++;
						}
					}
				}
			}
			if(count==0)printf("-1\n");
			else printf("%d %d\n",size+1,count);			
		}
	}
	return 0;
}
