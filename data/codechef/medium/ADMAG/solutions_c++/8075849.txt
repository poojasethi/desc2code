#include<iostream>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	long long int a[100];
	a[0]=0;
	a[1]=1;
	int i=2;
	//cout<<a[0]<<" "<<a[1]<<" ";
	while(i<=90)
	{
		a[i]=a[i-1]+a[i-2];
		//cout<<a[i]<<" ";
		i++;
	}
	//cout<<endl;
	for(int i=0;i<t;i++)
	{
		long long int n;
		scanf("%lld",&n);
		long long int cnt=0;
		int j=0;
		while(cnt<n)
		cnt+=a[j++];
		cout<<j-1<<endl;		
	}
	//system("pause");
	return 0;
}
