#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,a1,a2,b1,b2,c,p,q,e;
    scanf("%d",&t);
    while(t--)
    {
        double ans1=0.0,ans2=0.0;
        scanf("%d:%d",&a1,&a2);
        scanf("%d:%d",&b1,&b2);
        scanf("%d",&c);
        p=((a1-b1)*60)+(a2-b2);
        e=c;
        ans1+=p+c;
        ans2+=p;
        q=p-c;
        c-=q;
        if(p<=2*e)
            {
                ans2+=c/2.0;
            }
        printf("%0.1f %0.1f\n",ans1,ans2);
    }
    return 0;
}
