using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int x, y, d;
		double t1, t2, ans[2] = { 0, 0 };
		scanf("%d:%d\n", &x, &y);
		t1 = x * 60 + y;
		scanf("%d:%d\n", &x, &y);
		t2 = x * 60 + y;
		cin >> d;
		ans[0] = t1 - t2 + d;
		if (t1 - t2 >= d * 2)
			ans[1] = t1 - t2;
		else
			ans[1] = (t1 - t2 + 2 * d) / 2;
		printf("%.1f %.1f\n", ans[0], ans[1]);
	}
	return 0;
}
