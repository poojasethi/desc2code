#include<bits/stdc++.h>
using namespace std;
int main()
{
	char t1[6],t2[6];
	int dist;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s%s",t1,t2);
		scanf("%d",&dist);
		int dom=(((t1[0]-'0')*10+(t1[1]-'0'))*60+((t1[3]-'0')*10+(t1[4]-'0')))-(((t2[0]-'0')*10+(t2[1]-'0'))*60+((t2[3]-'0')*10+(t2[4]-'0')));
		if(2*dist>dom)
		{
			printf("%d.0 %.1f\n",dom+dist,(float)dom/2.0+dist);
		}
		else
		printf("%d.0 %d.0\n",dom+dist,dom);
	}
}
