#include<bits/stdc++.h>
int main()
{
    int t,time1,time2,dist,diff,h,m;
    char t1[6],t2[6];
    double ans;
    scanf("%d",&t);
    while(t--)
    {
        h=m=diff=0;
        scanf("%s %s %d",t1,t2,&dist);
        h=(t1[0]-'0')*10+(t1[1]-'0');
        m=(t1[3]-'0')*10+(t1[4]-'0');
        time1=(h*60)+m;
        
        h=(t2[0]-'0')*10+(t2[1]-'0');
        m=(t2[3]-'0')*10+(t2[4]-'0');
        time2=(h*60)+m;
        diff=time1-time2;
        if(2*dist<=diff)
            ans=diff*1.0;
        else
           ans=dist+(diff/2.0);
        printf("%d.0  %.1f\n",diff+dist,ans);
    }
   return 0; 
}
