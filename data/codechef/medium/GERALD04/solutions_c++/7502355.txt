    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
    	int t; scanf("%d",&t);
    	while(t--)
    	{
    		int t11,t12,t21,t22,time,dist;
    		float result2;
    		scanf("%d:%d",&t11,&t12);
    		scanf("%d:%d",&t21,&t22);
    		scanf("%d",&dist);
    		time=(t11-t21)*60+(t12-t22);
    		printf("%d.0 ",time+dist);
    		if((2*dist)<=time) printf("%d.0\n",time);
    		else 
    		{
    			result2=0.5*(time)+dist;
    			printf("%.1f\n",result2);
    		}
    	}
    	return 0;
    }
