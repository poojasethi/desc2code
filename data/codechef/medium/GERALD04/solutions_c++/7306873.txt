#include<stdio.h>
#include<stdlib.h>
int main()
{
	char t1[6],t2[6];
	int t,dist,time1,time2;
	double answer;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s%s",t1,t2);
		scanf("%d",&dist);
		time1=((t1[0]-'0')*10+(t1[1]-'0'))*60+(t1[3]-'0')*10+t1[4]-'0';
		time2=((t2[0]-'0')*10+(t2[1]-'0'))*60+(t2[3]-'0')*10+t2[4]-'0';
		if(2*dist<=time1-time2)
			answer=time1-time2;
		else
		{
			answer=(time1-time2)/2.0+dist;
		}
		printf("%d.0 %0.1f\n",time1-time2+dist,answer);
	}
	return 0;
}
