#include <iostream>
#include<stdio.h>
#include<iomanip>
using namespace std;

int main() {
	int t,h1,m1,h2,m2,dist;
	float t1,t2;
	cin>>t;
	while(t--)
	{
		scanf("%d:%d",&h1,&m1);
		scanf("%d:%d",&h2,&m2);
		//cout<<h1<<":"<<m1<<"\n";
		//cout<<h2<<":"<<m2<<"\n";
		cin>>dist;
		t1=h1*60+m1;
		t2=h2*60+m2;
		cout<<setprecision(1)<<fixed<<t1-t2+dist<<" ";
		if(t1-t2 > 2*dist)
			cout<<setprecision(1)<<fixed<<t1-t2<<"\n";
		else
			cout<<setprecision(1)<<fixed<<(t1-t2)/2.0+dist<<"\n";
			
	}
	return 0;
}
