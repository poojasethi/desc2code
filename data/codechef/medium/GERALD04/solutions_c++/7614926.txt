#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;
int main()
{
	int t,a,b,c,d,a1,b1,c1,d1,dist,y,y1,x,x1,yd,xd;
	float final,final2,fg;
	char t1[10],t2[10];
	int tmp1,tmp2,ty,t1y,t2y,t12y;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",t1);
		scanf("%s",t2);
		scanf("%d",&dist);
		a=(int)t1[0];
		b=(int)t1[1];
		c=(int)t1[3];
		d=(int)t1[4];
		x=10*a+b;
		y=10*c+d;
		a1=(int)t2[0];
		b1=(int)t2[1];
		c1=(int)t2[3];
		d1=(int)t2[4];
		x1=10*a1+b1;
		y1=10*c1+d1;
		if(y>y1)
		{
		yd=y-y1;
		xd=x-x1;
	    }
	    else
	    {
	    	yd=60-(y1-y);
	    	xd=x-x1-1;
	    }
	    final=xd*60+yd+dist;
	    final2=xd*60+yd;
	    if(2*dist>final2)
	    {
	    	fg=dist+(final2)/2;
	    }
	    else
	    fg=final2;
	    tmp1=10*final;
	    tmp2=10*fg;
	    ty=tmp1%10;
	    t2y=tmp2%10;
	    t1y=tmp1/10;
	    t12y=tmp2/10;
	    printf("%d.%d ",t1y,ty);
	    printf("%d.%d ",t12y,t2y);
	    printf("\n");
	    
	}
	return 0;
}
