#include <iostream>
using namespace std;
typedef long long  LL;
int main() {
    LL t=0,hh1=0,mm1=0,hh2=0,mm2=0,dist;
    cin>>t;
    while (t--) {
        scanf("%lld:%lld",&hh1,&mm1);
        scanf("%lld:%lld",&hh2,&mm2);
        scanf("%lld", &dist);
        LL t1 = hh1*60+mm1;
        LL t2 = hh2*60+mm2;
        
        double ans1 = (t1 - t2) + dist;
        double ans2 = (dist);
        if(t2 + 2*dist <= t1) {
            ans2 = ans2+dist;
            ans2 = ans2 + (t1 - (t2 + 2*dist));
        } else {
            LL rem = t1 - t2;
            ans2 = ans2 + (double)((double)(rem)/(double)(2));
        }
        
        printf("%.1lf %.1lf\n", ans1, ans2);
    }
    return 0;
}
