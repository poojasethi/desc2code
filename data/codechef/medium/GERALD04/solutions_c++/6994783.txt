#include <iostream>

typedef long long ll;
float convert(string s)
{
	//cout<<s<<endl;
	int len=s.length();
	float val=0;
	for(ll i=0;i<len;i++)
	{
		val=val*10+(s[i]-48);
	}
	return val;
}
