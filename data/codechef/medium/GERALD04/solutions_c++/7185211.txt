#include<bits/stdc++.h>
#define ll long long
using namespace std;

int main()
{	
	ll t;double h1,h2,m1,m2,dist,an1,an2; string s1,s2;
	cin>>t;
	while(t--)
	{
		cin>>s1;
		cin>>s2;
		cin>>dist;
		h1=(s1[0]-'0')*10+(s1[1]-'0');
		m1=(s1[3]-'0')*10+(s1[4]-'0');
		h2=(s2[0]-'0')*10+(s2[1]-'0');
		m2=(s2[3]-'0')*10+(s2[4]-'0');
		
		cout.precision(1);
		cout<<fixed;
		an1=(h1*60+m1)-(h2*60+m2)+dist;
		
		double diff=(h1*60+m1)-(h2*60+m2);
		
		if(diff>=dist)
		{
			if(2*dist>=diff)
			an2=diff + (2*dist-diff)/2;
			else
			an2=diff;
		}
		
		else
		an2=dist + diff/2;
		
		cout<<an1<<" "<<an2<<endl;
	}
	return 0;
}
