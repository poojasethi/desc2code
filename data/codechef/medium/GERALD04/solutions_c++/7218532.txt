#include <iostream>
#include<stdio.h>

using namespace std;

int main()
{
    int T,t1a,t1b,t2a,t2b,dist;
    double c=0.0,t=0.00;
    scanf("%d",&T);
    while(T--)
    {
    scanf("%d:%d",&t1a,&t1b);
    scanf("%d:%d",&t2a,&t2b);
    cin>>dist;

    c=60*(t1a-t2a)+(t1b-t2b);

    printf("%0.1f ",c+dist);
    if(2*dist<c)
        t=c;
    else
        t=dist+c/2;
    printf("%0.1f\n",t);
    }
    return 0;
}
