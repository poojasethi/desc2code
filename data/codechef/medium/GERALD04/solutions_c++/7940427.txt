#define max(a,b) ((a)>(b)?(a):(b))
int main()
{
    int t,h1,m1,h2,m2,dist;
    double ans1,ans2;
    scanf("%d",&t);
    while(t--)
    {
        char ch;
        scanf("%d:%d",&h1,&m1); // girlfriend time t1
         scanf("%d:%d",&h2,&m2); // chef time t2
       scanf("%d",&dist);
        int td=h1*60+m1-h2*60-m2;
        ans1=td+dist;
        if(dist<td)
        {
            int ntd=max(2*dist-td,0);
            ans2=2*td+ntd;

        }

        {
            int ntd=2*dist-td;
            ans2=2*td+ntd;
            ans2/=2.0;
        }

    }

}
