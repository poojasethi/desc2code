int t1, t2, dist;

int f( int t )
{
	if( t >= dist + dist )
		return 0;
	if( t >= dist )
		return dist - t + dist;
	else
		return t;
}

int main()
{
	int t;
	cin >> t;
	while( t -- )
	{
		int h, m;
		scanf( "%d:%d", &h, &m );
		t1 = h * 60 + m;
		scanf( "%d:%d", &h, &m );
		t2 = h * 60 + m;
		cin >> dist;
		int diff = t1 - t2;
		int x = f( diff ), d;
		if( diff < dist )
			d = dist - x + dist;
		else
			d = x;
		printf( "%d.0 %d.%d\n", diff + dist, diff + d / 2, d % 2 ? 5 : 0 );
	}
	return 0;
}
