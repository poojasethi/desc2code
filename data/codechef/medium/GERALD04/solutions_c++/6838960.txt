#include<bits/stdc++.h>
#include <iomanip>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string c1,c2;
		cin>>c1>>c2;
		int dist;
		cin>>dist;
		int t1,t2;
		t1=((c1[0]-'0')*10+(c1[1]-'0'))*60+((c1[3]-'0')*10+(c1[4]-'0'));
		t2=((c2[0]-'0')*10+(c2[1]-'0'))*60+((c2[3]-'0')*10+(c2[4]-'0'));
		//cout<<t1<<"   "<<t2<<"\n";
			float wait=(t1-t2+dist);
			cout<<wait<<".0"<<" ";
			if(2*dist<=t1-t2)
			{
				wait=t1-t2;
			}
			else
			{
				wait=(t1-t2)/2.0+dist;
			}
			printf("%0.1f\n",wait);
			//cout<<std::setprecision(1)<<wait<<"\n";
		
	}
	return 0;
}
