#include<iostream>
#include<string.h>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
inline int getmin(char *t){
	int temp=0;
	temp=((t[0]-'0')*10+t[1]-'0')*60+t[3]*10+t[4];
	return temp;
}
int main(){
	int t;
	char t1[6],t2[6];
		int  dist,diff;
		double ans;
	cin>>t;
	while(t--){
		
		scanf("%s",t1);
		scanf("%s",t2);
		scanf("%d",&dist);
		diff=getmin(t1)-getmin(t2);
		if(2*dist<=diff)
		ans=diff;
		else
		ans=dist+diff/2.0;
		
		printf("%d.0 %0.1f\n",diff+dist,ans);
	}
	return 0;
}
