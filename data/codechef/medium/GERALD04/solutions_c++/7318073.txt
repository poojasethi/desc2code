#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int g1,g2,b1,b2;
		cin>>g1;
		scanf(":");
		cin>>g2;
		cin>>b1;
		scanf(":");
		cin>>b2;
		double gtime=g1*60.0 + g2;
		double btime=b1*60.0 + b2;
		double dist;
		cin>>dist;
		double ans1=double(gtime - btime + dist);
		printf("%0.1lf ",ans1);
		if(gtime - btime >= 2*dist)
		{
			ans1=double(gtime-btime);
		}
		else
		{
			ans1=gtime -btime + ((2*dist)-(gtime-btime))/2.0;
		}
		printf("%0.1lf\n",ans1);


			// cout<<g1<<" "<<g2<<" "<<b1<<" "<<b2<<endl;
	}
	


	// cout<<a<<endl<<b<<endl;

}
