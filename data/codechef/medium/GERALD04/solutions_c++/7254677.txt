#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test,a,c,d;
	char b;
	double t1,t2;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%d%c%d",&a,&b,&c);
		t2=a*60+c;
		scanf("%d%c%d",&a,&b,&c);
		scanf("%d",&d);
		t1=a*60+c;
		double y=0;
		if(t1+2*d<t2)
			printf("%.1lf %.1lf\n",t2-t1+d,t2-t1+y);
		else
		{
			y=(2*d+(t1-t2))/2;
			printf("%.1lf %.1lf\n",t2-t1+d,t2-t1+y);
		}
	}
	return 0;
}
