#include<iostream>
#include<iomanip>// std::cout, std::fixed
using namespace std;
int main()
{
	char str[10],str1[10];
	double f,f1;
	int i,j,k,t;
	cin>>t;
	while(t--)
	{
		cin>>str>>str1;
		cin>>i;
		j=((str[0]-'0')*10+(str[1]-'0'))*60+(str[3]-'0')*10+(str[4]-'0');
		k=((str1[0]-'0')*10+(str1[1]-'0'))*60+(str1[3]-'0')*10+(str1[4]-'0');
		if(2*i<=(j-k))
		f=j-k;
		else
		{
			f=(j-k)/2.0+i;
		}
		f1=j-k+i;
		cout << std::fixed;
		cout<<std::setprecision(1)<<f1<<" ";
		cout<<std::setprecision(1)<<f<<"\n";
		
	}
	return 0;
}
