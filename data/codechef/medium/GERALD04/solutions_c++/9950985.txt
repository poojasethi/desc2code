int main(){
	ll int t;
	Sl(t);
	while(t--){
		ll int min, h1, m1, h2, m2;
		scanf("%lld:%lld",&h1,&m1);
		scanf("%lld:%lld",&h2,&m2);
		Sl(min);
		ll int time;
		time = 60 * (h1-h2)+ m1 - m2;
		printf("%.1f ",(1.0) * (time + min));
		if (time >= 2 * min)
            printf("%.1f\n",(1.0) * time);
		else
            printf("%.1f\n",(1.0) * (min) + time / 2.0);
		}
	return 0;
	}
