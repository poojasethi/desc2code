void fastint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}



int main(){
int t,h1,m1,h2,m2,d;
float ans;
char s[6];
fastint(t);
while(t--){
scan(s);
h1=s[0]*10+s[1];
m1=s[3]*10+s[4];

scan(s);
h2=s[0]*10+s[1];

fastint(d);

}
else{

}

}
