#include <bits/stdc++.h>

typedef pair< int , int > pii;
typedef long long LL;

/**************************Code****************************/

int a[128], b[128];

int main()
{
	int t, n, m, c, l;
	scanf( "%d", &t );
	while( t -- )
	{
		memset( a, 0, sizeof a );
		memset( b, 0, sizeof b );
		scanf( "%d%d", &n, &m );
		for( int i = 0 ; i < n ; i ++ )
		{
			scanf( "%d%d", &c, &l );
			a[l] += c;
		}
		for( int i = 0 ; i < m ; i ++ )
		{
			scanf( "%d%d", &c, &l );
			b[l] += c;
		}
		int ans = 0;
		for( int k = 1 ; k < 101 ; k ++ )
			ans += max( 0, b[k] - a[k] );
		cout << ans << endl;
	}
	return 0;
}
