#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m,i,j,sum=0,l,h,x,y;
		int a[105]={0};
		int b[105]={0};
		cin>>n>>m;
		for(i=0;i<n;i++)
		{
			cin>>l>>h;
			a[h]=a[h]+l;
		}
		for(j=0;j<m;j++)
		{
			cin>>x>>y;
			b[y]=b[y]+x;
		}
		for(i=0;i<105;i++)
		{
			if(b[i]>=a[i])
			{
				b[i]=b[i]-a[i];
			}
			else
			{
				b[i]=0;
			}
			sum=sum+b[i];
		}
		cout<<sum<<endl;
	}
	return 0;
}
