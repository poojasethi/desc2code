#include<iostream>
#include<algorithm>
#include<set>
#define mod 1000000007
using namespace std;
typedef long long int ll;
int main()
{
    ll t,n,m,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        ll soints[n][2],sofloats[m][2],a=0,b=0,p=0;
        set<ll> s;
        set<ll>::iterator it;
        for(i=0;i<n;i++)
        {
            cin>>soints[i][0]>>soints[i][1];
            s.insert(soints[i][1]);
        }
        for(i=0;i<m;i++)
        {
            cin>>sofloats[i][0]>>sofloats[i][1];
        }
        for(it=s.begin();it!=s.end();++it)
        {
            for(i=0;i<n;i++)
            {
                if(soints[i][1]==*it)
                    a+=soints[i][0];
            }
            for(i=0;i<m;i++)
            {
                if(sofloats[i][1]==*it)
                    b+=sofloats[i][0];
            }
            if(a<b)
                p+=b-a;
            a=0;b=0;
        }
        cout<<p<<'\n';
    }
    return 0;
}
