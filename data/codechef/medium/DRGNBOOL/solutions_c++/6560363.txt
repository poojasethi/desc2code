
#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    
    while(t)
    {
            int arr[101]={};int count=0;
            int n,m;
            cin>>n>>m;
            for(int i=1;i<=n;i++)
            {
                    int no;
                    cin>>no;
                    int temp;
                    cin>>temp;
                    arr[temp]+=no;
            }
            for(int i=1;i<=m;i++)
            {
                    int no;
                    cin>>no;
                    int temp;
                    cin>>temp;
                    arr[temp]-=no;
            }
            for(int i=1;i<=100;i++)
            {
                    if(arr[i]<0)
                    count-=arr[i];
                    
            }
            cout<<count<<"\n";
            t--;
    }
 
            return 0;
}
