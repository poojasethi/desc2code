#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int soint[101]={0};
		int n,m,c,l;
		cin>>n>>m;
		for (int i = 0; i <n; ++i)
		{
			cin>>c>>l;
			soint[l]+=c;
		}
		long long int sum=0;
		for (int i = 0; i < m; ++i)
		{
			cin>>c>>l;
			soint[l]-=c;
		}
		for (int i = 0; i <101; ++i)
		{
			if(soint[i]<0)
				sum-=soint[i];
		}
		cout<<sum<<endl;
	}
	return 0;
}
