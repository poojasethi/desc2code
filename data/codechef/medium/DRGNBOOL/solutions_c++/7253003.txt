#include<stdio.h>
#include<math.h>

using namespace std;
int main()
{int tc;
int m,n,x,y;
scanf("%d\n",&tc);

while(tc--)
{

int counter1[101];
int counter2[101];
int sum=0;
scanf("%d %d",&n,&m);
for(int i=0;i<=100;++i)
{
 counter1[i]=0;
 counter2[i]=0;
}
for(int i=0;i<n;++i)
{
scanf("%d %d",&x,&y);
counter1[y]+=x;
}

for(int i=0;i<m;++i)
{
scanf("%d %d",&x,&y);
counter2[y]+=x;
}

for(int i=0;i<=101;++i)
    if(counter2[i]>counter1[i])
        sum+=counter2[i]-counter1[i];
printf("%d\n",sum);
}





return 0;
}
