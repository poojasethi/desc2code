#include<iostream>
using namespace std;
int main()
{
int t;
scanf("%d",&t);
long long int a[101]={0};
long long int b[101]={0};
while(t--)
{
int n,m;
scanf("%d%d",&n,&m);
int c,l;
long long int res=0;
for(int i=0;i<n;i++)
{
scanf("%d%d",&c,&l);
a[l]+=c;
}
for(int j=0;j<m;j++)
{
scanf("%d%d",&c,&l);
b[l]+=c;
}
for(int i=0;i<101;i++)
{
if(b[i]>a[i])
{
res+=(b[i]-a[i]);
}
}
printf("%lld\n",res);
for(int i=0;i<101;i++)
{
a[i]=b[i]=0;
}
}
return 0;
}
