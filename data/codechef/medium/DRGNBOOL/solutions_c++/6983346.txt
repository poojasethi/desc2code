#include<iostream>
using namespace std;
class Battle{
	int m,n;
	int p[101],s[101];
	public:
		Battle()
		{
			int i;
			int x,y;
			for(i=0;i<=100;i++)
			p[i]=0;
			for(i=0;i<=100;i++)
			s[i]=0;
			cin>>n>>m;
			for(i=0;i<n;i++)
			{
				cin>>x>>y;
				p[y]+=x;
			}
			for(i=0;i<m;i++)
			{
				cin>>x>>y;
				s[y]+=x;
			}
		}
		void calculate()
		{
			int ans=0;
			int i;
			for(i=0;i<=100;i++)
			{
				p[i]-=s[i];
				if(p[i]<0)
					ans+=-p[i];
			}
			cout<<ans<<endl;
		}
};
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		Battle b;
		b.calculate();
	}
	return 0;
}
