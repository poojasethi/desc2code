#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n,m,i,j,sum=0;
		cin>>n>>m;
		pair<int,int>a[105];
		pair<int,int>b[105];
		for(i=0;i<n;i++)
		cin>>a[i].first>>a[i].second;
		for(i=0;i<m;i++)
		cin>>b[i].first>>b[i].second;
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				if(a[i].second==b[j].second&&a[i].first<=b[j].first)
				{
					b[j].first=b[j].first-a[i].first;
					a[i].first=0;
				}
				else if(a[i].second==b[j].second&&a[i].first>b[j].first)
				{
					a[i].first=a[i].first-b[j].first;
					b[j].first=0;
				}
			}
		}
		for(i=0;i<m;i++)
		{
			sum=sum+b[i].first;
		}
		cout<<sum<<endl;
	}
	return 0;
}
