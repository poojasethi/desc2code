#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int arr[101];
int str[101];

int main()
{
    int t,n,m,sum,l,c;
    cin>>t;
    while(t--)
    {
              cin>>n>>m;
              for(int i=0;i<n;i++)
              {
                      cin>>c>>l;
                      arr[l] += c;
                      }
              for(int i=0;i<m;i++)
              {
                      cin>>c>>l;
                      str[l] += c;
                      }
              sum = 0;
              for(int i=0;i<101;i++)
              {
                      c = str[i]-arr[i];
                      if(c>0)sum += c; 
                      str[i] = arr[i] = 0;
                      }
              cout<<sum<<endl;
              }
    return 0;
    }
