#include <bits/stdc++.h>
using namespace std;

int main() {
	int t,n,m,a,b,soint[105],sofloat[105],ans;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		ans=0;
		for(int i=1;i<=100;i++)
		{
			soint[i]=0;
			sofloat[i]=0;
		}
		for(int i=1;i<=n;i++)
		{
			cin>>a>>b;
			soint[b]=soint[b]+a;
		}
		
		for(int i=1;i<=m;i++)
		{
			cin>>a>>b;
			sofloat[b]=sofloat[b]+a;
		}
		
		for(int i=1;i<=100;i++)
		{
			if(sofloat[i]>soint[i])
			ans=ans+(sofloat[i]-soint[i]);
		}
		cout<<ans<<endl;
	}
	return 0;
}
