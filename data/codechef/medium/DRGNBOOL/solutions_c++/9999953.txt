#include <bits/stdc++.h>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
    	int n,m,a[101]={0};
    	long long sum=0;
    	cin>>n>>m;
    	while(n--)
    	{
    		int c,l;
    		cin>>c>>l;
    		a[l]+=c;
    	}
    	while(m--)
    	{
    		int c,l;
    		cin>>c>>l;
    		a[l]-=c;
    	}
    for(int i=1;i<101;i++)
    	if(a[i]<0)
    	sum+=abs(a[i]);
    cout<<sum<<"\n";
    }
    cin.tie(NULL);
    return 0;
}
