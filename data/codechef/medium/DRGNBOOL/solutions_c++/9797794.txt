#include <iostream>
using namespace std;

int main() {
	int t,n,m,a[105],b[105],i,l,c,temp,ans;
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		for(i=0;i<105;i++)
		{
			a[i]=0;
			b[i]=0;
		}
		scanf("%d %d",&n,&m);
		for(i=0;i<n;i++)
		{
			scanf("%d %d",&c,&l);
			a[l]+=c;
		}
		for(i=0;i<m;i++)
		{
			scanf("%d %d",&c,&l);
			b[l]+=c;
		}
		for(i=0;i<105;i++)
		{
			if((a[i]!=0)&&(b[i]!=0))
			{
				if(a[i]<=b[i])
				{
					ans+=(b[i]-a[i]);
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
