#include<bits/stdc++.h>
using namespace std;
int so[100];
int sf[100];
int main(){
    int t;
    cin >> t;
    while(t--){
        int N, M;
        scanf("%d%d", &N, &M);
        for(int i = 1;i <= 100;++i){
            so[i] = 0;
            sf[i] = 0;
        }
        while(N--){
            int C, L;
            scanf("%d%d", &C, &L);
            so[L] += C;
        }
        while(M--){
            int C, L;
            scanf("%d%d", &C, &L);
            sf[L] += C;
        }
        int ans = 0;
        for(int i = 1;i <= 100;++i){
            if(sf[i] > so[i])
                ans += sf[i] - so[i];
        }
        printf("%d\n", ans);
    }
    return 0;
}
