#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long int t,n,m,i,j,sum, p, l;
	   long long int soint[102],sofloat[102];

    scanf("%lld", &t);
    while(t--){
           sum = 0;
           memset(soint,0,sizeof(soint));
           memset(sofloat,0,sizeof(sofloat));
       cin >> n;
       cin >> m;

    for(i = 1; i <= n; i++){
       cin >> p >> l;
       soint[l] += p;
    }

    for(i = 1; i <= m; i++){
        cin >> p >> l;
        sofloat[l] += p;
    }

    for (i = 1; i <= 100; i++) {
            if(sofloat[i] > soint[i])
                    sum = sum + sofloat[i] - soint[i];
    }

    printf("%lld\n", sum);
    }
    return 0;
}
