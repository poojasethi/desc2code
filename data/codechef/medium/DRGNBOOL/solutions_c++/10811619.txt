#include<stdio.h>
typedef long long ll;
int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n,m,s=0;
		scanf("%lld%lld",&n,&m);
		ll w[101]={0};
		ll L[101]={0},l1=-1;
		ll c,l;
		for(ll i=0;i<n;i++)
		{
			scanf("%lld%lld",&c,&l);
			w[l]+=c;
			if (l>l1)
			l1=l;
		}
			for(ll i=0;i<m;i++)
		{
			scanf("%lld%lld",&c,&l);
			L[l]+=c;
		}
			for(ll i=1;i<=l1;i++)
	        if((w[i]-L[i])<0)
		    s+=(L[i]-w[i]);
		
		printf("%lld\n",s);
	}
}
