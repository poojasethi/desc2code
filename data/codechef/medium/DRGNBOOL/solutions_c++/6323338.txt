#include<stdio.h>
#include<iostream>
int main()
{ int t;
  scanf("%d",&t);
  while(t--)
       { int n,m,s[101]={0},p[101]={0},count=0;
         scanf("%d%d",&n,&m);
         for(int i=0;i<n;i++)
            { int x,y;
              scanf("%d%d",&x,&y);
              s[y]+=x;
            }
         for(int i=0;i<m;i++)
		    { int x,y;
		      scanf("%d%d",&x,&y);
		      p[y]+=x;
		    }   
		 for(int i=1;i<=100;i++)
		    if(s[i]<p[i])
		      count+=p[i]-s[i];
		 printf("%d\n",count);	  	   
       }
}
