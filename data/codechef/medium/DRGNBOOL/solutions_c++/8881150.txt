 
#include<stdio.h>
#include<string.h>
 
int count[101];
 
int main()
{
	int C,i,L,M,N,sum,T;
	scanf("%d",&T);
	while(T--)
	{
		memset(count,0,sizeof(int)*101);
		scanf("%d %d",&N,&M);
		while(N--)
		{
			scanf("%d %d",&C,&L);
			count[L]+=C;
		}
		while(M--)
		{
			scanf("%d %d",&C,&L);
			count[L]-=C;
		}
		sum=0;
		for(i=1;i<=100;i++)
			if(count[i]<0)
				sum-=count[i];
		printf("%d\n",sum);
	}
	return 0;
}
