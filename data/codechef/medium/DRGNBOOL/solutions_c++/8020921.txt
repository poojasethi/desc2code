#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,m,j,k,i,sum=0,a,b;
    cin>>t;
    while(t--)
    {
        int cl[101]={0};
        cin>>n>>m;
        j=n;
        k=m;
        while(j--)
        {
            cin>>a>>b;
            cl[b]+=a;
        }
        while(k--)
        {
            cin>>a>>b;
            cl[b]-=a;
        }
        for(i=0;i<101;i++)
        {
            if(cl[i]<0)
                sum+=cl[i];
        }
        sum=abs(sum);
        cout<<sum<<"\n";
        sum=0;
    }
    return 0;
}
