#include <iostream>
#include<vector>
#include<cstdio>
#include<cmath>
#include<algorithm>

using namespace std;

int main()
{
    //freopen("in.txt","r",stdin);
    int t,i,j,k,x,n,m,y,result;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        vector<int> soints(100,0);

        for(i=0;i<n;i++)
        {
            scanf("%d%d",&x,&y);
            soints[y-1]+=x;
        }

        for(i=0;i<m;i++)
        {
            scanf("%d%d",&x,&y);
            soints[y-1]-=x;
        }

        result=0;
        for(i=0;i<100;i++)
        {
            if(soints[i]<0)
                result-=soints[i];

        }

        printf("%d\n",result);
    }

    return 0;
}
