#include<stdio.h>
#define gc getchar_unlocked
void fastread(int *a)
 {
 	char c=0;
 	while(c<33){
 		c=gc();
	 }
	 *a=0;
	while(c>33){
		*a=*a*10+c-'0';
		c=gc();
	} 
 }
 int main()
 {
 	int m,n,t,chakra,level;
 	fastread(&t);
 	while(t--)
 	{
 	    int a[105]={0},b[105]={0};
 		fastread(&n); fastread(&m);
 		while(n--)
 		{
 			fastread(&chakra); fastread(&level);
 			a[level]+=chakra;
		}
		while(m--)
		{
			fastread(&chakra); fastread(&level);
 			b[level]+=chakra;
 	    }
 	    	int ans=0;
 	    for(int i=1;i<=100;i++)
 	    {
 	    	if(a[i]>0)
 	    	{
			   	if(b[i]>a[i])
 	    	     ans=ans+b[i]-a[i];
 	        }
 	    	     
		 }
		printf("%d\n",ans); 
	 }
 }
