#include <iostream>
using namespace std;
int c1[101],c2[101],l1[101],l2[101];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		int hash1[101]={0},hash2[101]={0};
		cin>>n>>m;
		for (int i = 0; i < n; ++i)
		{
			cin>>c1[i]>>l1[i];
			hash1[l1[i]]+=c1[i];
		}
 
		for (int i = 0; i < m; ++i)
		{
			cin>>c2[i]>>l2[i];
	
			hash2[l2[i]]+=c2[i];
		}
		int ans=0;
		for (int i = 1; i < 101; ++i)
		{
			if((hash2[i]-hash1[i])>0)
			{
				ans+=hash2[i]-hash1[i];
				// cout<<ans<<endl;
			}
		}
		cout<<ans<<endl;


	}
}
