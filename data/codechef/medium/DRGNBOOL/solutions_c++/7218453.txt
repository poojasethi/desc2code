#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
	int t;
	for(scanf("%d", &t); t--; )
	{
		int n, m;
		int a[101];
		scanf("%d %d", &n, &m);
		memset(a, 0, sizeof(a));
		while(n--)
		{
			int c, l;
			scanf("%d %d", &c, &l);
			a[l-1]+=c;
		}
		while(m--)
		{
			int c, l;
			scanf("%d %d", &c, &l);
			a[l-1]-=c;
		}
		long int sum=0;
		for(int i=0; i<101; i++){
			if(a[i]<0)
				sum+=(-1*a[i]);
		}
		printf("%ld\n", sum);
	}
	return 0;
}
