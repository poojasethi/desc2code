#include<stdio.h>

int main()
{
    int test;
    scanf("%d",&test);
    while(test!=0)
    {
        int n,m;
        scanf("%d%d",&n,&m);
        int a[100][2],b[100][2],temp,temp1;
         for(int j=0;j<100;j++)
        {
            a[j][0]=0;
            b[j][0]=0;
            a[j][1]=j+1;
            b[j][1]=j+1;
        }
        for(int j=0;j<n;j++)
        {
            scanf("%d%d",&temp,&temp1);
            a[temp1-1][0]+=temp;
        }
         for(int j=0;j<m;j++)
        {
             scanf("%d%d",&temp,&temp1);
            b[temp1-1][0]+=temp;
        }
        temp=0;
         for(int j=0;j<100;j++)
         {
             if(b[j][0]>a[j][0])
             {
                 temp+=(b[j][0]-a[j][0]);
             }
         }
        printf("%d\n",temp);

        test--;
    }

    return 0;
}
