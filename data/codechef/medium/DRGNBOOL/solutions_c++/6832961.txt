#include<stdio.h>
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
 
        int l1[102]={0},l2[102]={0};
 
        int n,m,i,max=0,min=0;
        int c,l;
        scanf("%d %d",&n,&m);
 
        for(i=0;i<n;i++)
          {
              scanf("%d %d",&c,&l);
              if(max<l)
                 max=l;
              if(min>l)
                min=l;
 
              l1[l]=l1[l]+c;
          }
 
          for(i=0;i<m;i++)
          {
              scanf("%d %d",&c,&l);
 
              l2[l]=l2[l]+c;
          }
 
          int sum=0;
          for(i=min;i<=max;i++)
             if(l2[i]>l1[i])
                sum+=(l2[i]-l1[i]);
 
              printf("%d\n",sum);
    }
    return 0;
}
