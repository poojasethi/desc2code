#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
using namespace std;
int main()
{
    int i,t,n,m,e1,e2,diff,ans;
    scanf("%d",&t);
    while(t--)
    {
        map<int,int> m1,m2;
        scanf("%d%d",&n,&m);
        for(i=0;i<n;i++)
        {
            scanf("%d%d",&e1,&e2);
            m1[e2]+=e1;
        }
        for(i=0;i<m;i++)
        {
            scanf("%d%d",&e1,&e2);
            m2[e2]+=e1;
        }
        ans=0;
        map<int,int>::iterator it1,it2;
        for(it1=m1.begin(),it2=m2.begin();(it1!=m1.end());it1++,it2++)
        {
            diff=it1->second-it2->second;
            if(diff<0)
            {
                diff*=-1;
                ans+=diff;
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
