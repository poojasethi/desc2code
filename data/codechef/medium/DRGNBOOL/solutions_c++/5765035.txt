#include<iostream>
//#include<fstream>
#include<cstring>
#include<cmath>
using namespace std;

int siArr[100];
int sfArr[100];

int main() {
	int t;
	//ifstream cin;
	//cin.open("input.txt");
	//ofstream cout;
	//cout.open("output.txt");
	cin >> t;
	int N, M;
	int C, L;
	int count;
	while(t--) {
		memset (siArr, 0, sizeof(siArr));
		memset (sfArr, 0, sizeof(sfArr));
		cin >> N >> M;
		count = 0;
		for(int i=0;i<N;i++) {
			cin >> C >> L;
			siArr[L-1] = siArr[L-1] + C; 
		}
		for(int i=0;i<M;i++) {
			cin >> C >> L;
			sfArr[L-1] = sfArr[L-1] + C;
		}
		int diff;
		for(int i=0;i<100;i++) {
			diff = siArr[i] - sfArr[i];
			if(diff<0) {
				count = count + diff;
			}
		}
		cout << count*(-1) << endl;
	}
	return 0;
}
