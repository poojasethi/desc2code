#include<iostream>
#include<stdio.h>

using namespace std ;

int main()
{
    int test;
    cin>>test;
    for(int i=0;i<test;i++)
    {
      int no_int,no_float;
      cin>>no_int;
      cin>>no_float;
      int array_int[100],array_float[100];
      for(int x=0;x<100;x++)
      {
          array_int[x]=0;
          array_float[x]=0;
      }
      int power,level;
      for(int x=0;x<no_int ;x++)
      {
          cin>>power;
          cin>>level;
          array_int[level-1]+=power;
      }
      for(int x=0;x<no_float ;x++)
      {
          cin>>power;
          cin>>level;
          array_float[level-1]+=power;
      }
      int ans=0;
      for(int i=0;i<100;i++)
      {
          int difference=array_int[i]-array_float[i];
          if(difference<0)
          {
              ans+=(difference*(-1));
          }
      }
      cout<<ans<<endl;
    }
}
