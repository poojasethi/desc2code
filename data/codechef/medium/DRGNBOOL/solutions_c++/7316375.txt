#include<bits/stdc++.h>
using namespace std;
int main()
{
        int power,level,count,i,j,k,m,n,t,arr[105],brr[105];
        cin>>t;
        while(t--)
        {
                cin>>n>>m;
                for(i=0;i<105;i++)
                {
                        arr[i]=0;
                        brr[i]=0;
                }

                for(i=0;i<n;i++)
                {
                        cin>>power>>level;
                        arr[level]+=power;
                }

                for(i=0;i<m;i++)
                {
                        cin>>power>>level;
                        brr[level]+=power;
                }
                count=0;
                for(i=0;i<105;i++)
                {
                        if(arr[i]<brr[i])
                                count+=brr[i]-arr[i];
                }
                cout<<count<<endl;
        }
}
