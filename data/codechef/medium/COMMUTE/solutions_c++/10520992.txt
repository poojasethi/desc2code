#include <stdio.h>
#include <algorithm>

int main()
{
	int t, n, i, j;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int x[n], l[n], f[n], arrive, leave;
		for(i=0; i<n; i++)
        {
            scanf("%d%d%d",&x[i],&l[i],&f[i]);
        }
        arrive=0;
        leave=x[0];
        for(i=1; i<n; i++)
        {
            arrive=leave+l[i-1];
            while(x[i]<arrive)
            {
                x[i]+=f[i];
            }
            leave=x[i];
        }
        printf("%d\n",leave+l[i-1]);


    }
	return 0;
}
