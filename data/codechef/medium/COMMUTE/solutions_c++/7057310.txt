#include<bits/stdc++.h>
#define ll long long
using namespace std;



int main()
{	
	ll tc,n,i,x,l,f,t,d;
	cin>>tc;
	while(tc--)
	{
			cin>>n;
			t=0;
			while(n--)
			{
				cin>>x>>l>>f;
				d=(t-x)/f;
				if(x>=t)
				t=x+l;
				else if(d*f + x==t)
				t=t+l;
				else
				t=x+(d+1)*f+l;
			}
			cout<<t<<endl;
	}
	return 0;
}
