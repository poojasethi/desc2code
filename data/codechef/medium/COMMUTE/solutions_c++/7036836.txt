#include<bits/stdc++.h> 
# define ll long long
using namespace std;
int main()
{
	int n,t,i,j;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int x[n],l[n],f[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&x[i]);
			scanf("%d",&l[i]);
			scanf("%d",&f[i]);
		}
		int time=0;
		for(i=0;i<n;i++)
		{
			j=0;
			while((x[i]+j*f[i])<time)
			{
				j++;
			}
			time+=(x[i]+j*f[i])-time;
			time+=l[i];
		}
		printf("%d\n",time);
	}
}
