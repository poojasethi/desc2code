    #include<iostream>
    #include<stdio.h>
    using namespace std;
     
    int main()
    {
    	int t,n,x,tm,f,sum;
    	scanf("%d",&t);
    	while(t--)
    	{
    		sum=0;
    		scanf("%d",&n);
    		//scanf("%d %d %d",&x,&tm,&f);
    		
    		while(n--)
    		{
    			scanf("%d %d %d",&x,&tm,&f);
    			if(sum<=x)
    				sum=x+tm;
    			else
    			{//while(x<sum)
    			//x+=f;
    			//cout<<"sum="<<sum<<" x="<<x<<" time="<<tm<<endl;
    			sum=x+(((sum-x)+f-1)/f)*f+tm;}
    		}
    		printf("%d\n",sum);
    	}
    }
