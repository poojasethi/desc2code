#include<iostream>
using namespace std;
int main()
{
	int t,n;
	int x,l,f;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int ans=0;
		for(int i=1;i<=n;i++)
		{
		cin>>x>>l>>f;
			if(x==ans)
		{
			ans=ans+l;
		}
		else if(x>ans)
		{
			ans=x+l;
		}
		else
		{
			int j=0;
			while(x+j*f<ans)
			{
				j++;
			}
			ans=x+j*f+l;
		}
	}
	cout<<ans<<endl;
	}
}
