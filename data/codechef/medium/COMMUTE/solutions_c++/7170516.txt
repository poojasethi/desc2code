using namespace std;
 
int main()
{
    int t,i,n,x,l,f,curr,a,b;
    cin>>t;
    while(t--)
    {
        scanf("%d",&n);
        curr=0;
        for(i=0;i<n;i++)
        {
            scanf("%d%d%d",&x,&l,&f);
            if(curr<x)
                curr=x;
            if(curr==x)
            {
                curr+=l;
                continue;
            }
            a=curr-x;
            if(a%f==0)
                b=0;
            else
                b=f-a%f;
            curr+=(b+l);
 
        }
        printf("%d\n",curr);
    }
    return 0;
}
