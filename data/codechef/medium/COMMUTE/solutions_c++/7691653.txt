#include <iostream>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        cin >> n;
        int x[n],l[n],f[n],i;
        for(i = 0;i < n;++i)
            cin >> x[i] >> l[i] >> f[i];
        int total = 0;
        for(i =0;i < n;++i)
        {
            if(x[i] > total)
                total = total + x[i] - total + l[i];
            else
            {
                while(x[i] < total)
                x[i] = x[i] + f[i];
                total = total + x[i] - total + l[i];
            }
        }
        cout << total << "\n";
    }
    return 0;
}
