#include<bits/stdc++.h>
using namespace std;

int main()

{

int T;
cin>>T;

while(T--)
{
int N;
cin>>N;

int i;

int X[N], L[N], F[N];

for(i=0; i<N;i++)
cin>>X[i]>>L[i]>>F[i];

int ans=X[0]+L[0];

for(i=1; i<N; i++)
{
if(ans<=X[i])
ans=X[i];

else
{
while(ans>X[i])
{
X[i]=X[i]+F[i];
}

ans=X[i];
}

ans=ans+L[i];
}

cout<<ans<<endl;
}
return 0;
}
