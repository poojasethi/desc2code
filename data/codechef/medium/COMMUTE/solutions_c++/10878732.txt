#include<cstdio>
#include<algorithm>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        int X[n],L[n],F[n];
        for(int i=0;i<n;i++){
            scanf("%d %d %d",&X[i],&L[i],&F[i]);
        }
        int time=0;
        for(int i=0;i<n;i++){
            int k=(time-X[i])/F[i];
            if(k<0){
                k=0;
            }
            while(X[i]+(k*F[i])<time){
                k++;
            }
            time=X[i]+(k*F[i])+L[i];
        }
        printf("%d\n",time);
    }
    
}
