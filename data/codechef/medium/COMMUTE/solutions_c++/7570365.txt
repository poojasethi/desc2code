#include<stdio.h>

int main()
{
    int tc,l,x,f,n;
    long long int ans;

    scanf("%d",&tc);

    while(tc--)
    {
        scanf("%d",&n);

        ans = 0;
        while(n--)
        {
            scanf("%d%d%d",&x,&l,&f);
            if(ans <= x)
                ans = x;
            else if((ans-x)%f == 0)
                ans = x+(((ans-x)/f))*f;
            else
                ans = x+(((ans-x)/f)+1)*f;
            ans = ans+l;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
