#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <math.h>
#include <string.h>
#include <limits.h>
#include <stdlib.h>

using namespace std;

int main()
{
    int test,x,l,f,t,n;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d",&n);
        t=0;
        while(n--)
        {
            scanf("%d %d %d",&x,&l,&f);
            if(x>t)
                t=x;
            t+=(x-(t%f)+f)%f;
            t+=l;
        }
        printf("%d\n",t);
    }
}
