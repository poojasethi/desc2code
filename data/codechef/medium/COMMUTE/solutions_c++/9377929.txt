#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n, time = 0;
        scanf("%d", &n);
        int x[n], l[n], f[n];
        for(int i = 0; i < n; i++)
        {
            scanf("%d %d %d", &x[i], &l[i], &f[i]);
            while(time > x[i])
            {
                x[i] += f[i];
            }
            time = x[i] + l[i];
        }
        printf("%d\n", time);
    }
    return 0;
}
