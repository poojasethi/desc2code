#include<iostream>
using namespace std;
int main()
{
long int t,n,x[1000],l[1000],f[1000],i,j,d,k;
cin>>t;
while(t--)
{k=0;
 cin>>n;
 for(i=0;i<n;i++)
 cin>>x[i]>>l[i]>>f[i];
 k=x[0]+l[0];
 for(i=1;i<n;i++)
 {
 if(k<=x[i])
 k+=x[i]-k+l[i];
 else
 {
  d=x[i];
  while(1)
  { d+=f[i];
    if(k<=d)
    {
    k+=d-k+l[i];
    break;
    }
 }
}
 }
 cout<<k<<"\n";
 }
 return 0;
 }
