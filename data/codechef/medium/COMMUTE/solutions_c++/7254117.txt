#include<stdio.h>
#include<math.h>

using namespace std;
int main()
{int tc;
  int n;
scanf("%d\n",&tc);

while(tc--)
{
int xi[1001];
int li[1001];
int fi[1001];
int time=0;
for(int i=0;i<1001;++i)
    xi[i]=li[i]=fi[i]=0;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
        scanf("%d%d%d",&xi[i],&li[i],&fi[i]);

        for(int i=0;i<n;++i)
        {
            if(time<=xi[i])
				time = xi[i];
			else
				time = time + fi[i] - ((time-xi[i]-1)%fi[i])-1;
			time = time + li[i];
        }

printf("%d\n",time);


}





return 0;
}
