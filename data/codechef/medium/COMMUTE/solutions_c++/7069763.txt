#include <iostream>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--){
        int n,x,l,f,s = 0;
        cin >> n;
        while(n--){
            cin >> x >> l >> f;
            if(x>s) s = x;
            s += (f-(s%f)+x)%f;
            s += l;
        }
        cout << s << endl;
    }
    return 0;
}
