#include<stdio.h>
#include<stdlib.h>

int main()
{
	int i,j,k,start,delay,tt,t,n,sum;
	
	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		sum=0;
		while(n--)
		{
			scanf("%d%d%d",&start,&tt,&delay);
			if(sum>start)
			while(sum>start)
			start+=delay;
			sum=start+tt;
		}
		printf("%d\n",sum);
	}
	return 0;
}
