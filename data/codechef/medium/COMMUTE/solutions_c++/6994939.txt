using namespace std;
#define max3(a, b, c) max(a, b) > max(b, c) ? max(a, b) : max(b, c)
#define min3(a, b, c) min(a, b) < min(b, c) ? min(a, b) : min(b, c)
#define digit(c) (c - '0')
#define SET( ARRAY , VALUE ) memset( ARRAY , VALUE , sizeof(ARRAY) )
#define Inf 2147483647

#define _ ios_base::sync_with_stdio(0);cin.tie(0);
int main(){
	int t,n,i,num,amt;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		int arr[n][3];
		for(i=0;i<n;i++)
			scanf("%d%d%d",&arr[i][0],&arr[i][1],&arr[i][2]);
		amt=0;
		for(i=0;i<n;i++){
			if(amt<=arr[i][0]){
				amt=amt+(arr[i][0]-amt)+arr[i][1];
			}
			else if(amt>arr[i][0]){
				num=amt/arr[i][2];
				if(amt%arr[i][2]!=0)
					num++;
				num=num*arr[i][2];
				num+=arr[i][0];
				amt=amt+(num-amt)%arr[i][2]+arr[i][1];
			}
		}
		printf("%d\n",amt);
	}
	return 0;
}
