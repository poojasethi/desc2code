#include<bits/stdc++.h>
using namespace std;
int main()
{
    int i,n,t;
    long long int sum=0;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int a[n][3];
        if(n==1 || n>1)
        {
            cin>>a[0][0]>>a[0][1]>>a[0][2];
            sum=sum+a[0][0]+a[0][1];
        }
        if(n>1)
        {
            for(i=1;i<n;i++)
            {
                cin>>a[i][0]>>a[i][1]>>a[i][2];
                if(sum<=a[i][0])
                sum=a[i][0]+a[i][1];
                else
                {
                    if(((sum-a[i][0])%a[i][2])!=0)
                    sum=sum+a[i][2]-(sum-a[i][0])%a[i][2];

                    sum=sum+a[i][1];
                }
            }
        }
        cout<<sum<<"\n";
        sum=0;
    }
    return 0;
}
