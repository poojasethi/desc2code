#include<bits/stdc++.h>
using namespace std;
int main(){
  int t,n,x,l,f,ans;
  cin>>t;
  while(t--)
  {
    cin>>n;
    cin>>x>>l>>f;
    ans=x+l;
    for(int i=2;i<=n;i++)
    {
      cin>>x>>l>>f;
      if(ans>x && ans<=(x+f))
      {
        ans=x+l+f;
      }
      else if(ans<=x)
      {
        ans=x+l;
      }
      else if(ans>(x+f))
      {
        int temp=x;
        while(temp<ans)
        {
          temp+=f;
        }
        ans=temp+l;
      }
    }
    cout<<ans<<endl;
   }
   return 0;
}
