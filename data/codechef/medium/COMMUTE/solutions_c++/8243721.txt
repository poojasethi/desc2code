#include<bits/stdc++.h>
using namespace std;
typedef struct journey{
	int x,l,f;
};
int nxt(int a,int d,int t){
	while(t>a){
		a+=d;
	}
	return a;
}
int main(){
	int t,n;
	journey j[1005];
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			scanf("%d %d %d",&j[i].x,&j[i].l,&j[i].f);
		}
		int time=j[0].x+j[0].l;
		int i=1;
		while(i<n){
			int p=nxt(j[i].x,j[i].f,time);
			time+=(p-time);
			time+=j[i].l;
			i++;
		}
		printf("%d\n",time);
	} 
	return 0;
}
