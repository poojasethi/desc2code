#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	cin>>test;
	int t;int n;int i;
	for(t=0;t<test;t++)
	{
		cin>>n;
		int arr[n][3];
		for(i=0;i<n;i++)
		{
			cin>>arr[i][0]>>arr[i][1]>>arr[i][2];
		}
		int time=0;
		for(i=0;i<n;i++)
		{
			if (arr[i][0]>=time)
			{
				time=arr[i][0]+arr[i][1];
			}
			else
			{
				int temp=arr[i][0];
				while(temp<time)
				{
					temp+=arr[i][2];
				}
				time=temp+arr[i][1];
			}
			
		}//i
		cout<<time<<"\n";
	}//test
	
	
	return 0;
	
}//main
