#include <iostream>
#include <vector>
using namespace std;
typedef long long ll;
typedef vector<ll> vll;
int main()
{
	ll t,x,l,f,n,i,currtime,j;
	scanf("%lld",&t);
	vll schedule[3];
	while(t--)
	{
		scanf("%lld",&n);
		for(i=0;i<3;i++)
		{
			for(j=0;j<n;j++)
			{
				schedule[i].push_back(0);
			}
		}
		for(i=0;i<n;i++)
		{
			currtime=0;
			scanf("%lld %lld %lld",&x,&l,&f);	
			schedule[0][i]=x;
			schedule[1][i]=l;
			schedule[2][i]=f;
		}
		for(i=0;i<n;i++)
		{
			if(currtime<=schedule[0][i])
			{
				currtime=schedule[0][i];
			}
			else
			{
				while(currtime>schedule[0][i])
				{
					schedule[0][i]+=schedule[2][i];
				}
				currtime=schedule[0][i];	
			}
			currtime+=schedule[1][i];
		}
		printf("%lld\n",currtime);
		for(i=0;i<3;i++)
		{
				schedule[i].clear();
		}
	}
	return 0;
}
