#include<iostream>
using namespace std;
long long int  n,m,a[100010],b[100010],s,i,k,an;
#define M 1000000007
int main()
{
    cin>>n>>m>>k;
    for(i=0;i<n;i++)
    {
                    cin>>a[i];
                    b[i]=(m-a[i])/k+1;
                    s+=b[i];
    }
    for(i=0;i<n;i++)
    {
                    if(s%2)
                    {
                           if(b[i]%2)
                           {
                                     an=max(b[i]/2+1,an);
                           }
                           else
                           {
                                     an=max(b[i]/2,an);
                           }
                    }
                    else
                    {
                           if(b[i]%2)
                           {
                                     an=max(b[i]/2,an);
                           }
                           else
                           {
                                     an=max(b[i]/2+1,an);
                           }
                    }
    }
    cout<<(an%M)<<endl;
    return 0;
}
