typedef pair<int,int> PII;
typedef vector<PII> VPII;
typedef vector<int> VI;
typedef vector<VI> VVI;
typedef long long LL;
LL a[100005];
LL m,k;
int main()
{
    int n,i,j;
    cin>>n>>m>>k;
    rep(i,n)
    {
        sll(a[i]);
        a[i] = (m-a[i])/k + 1;
    }
//    rep(i,n)pllw(a[i]);pn;
    LL rem=0;
    rep(i,n)rem+=a[i];
    rem = rem&1;
    LL ans=0;
    rep(i,n)
    {
        if(a[i]&1)
            ans = max(ans, (a[i]+1)/2);
        else
            ans = max(ans,1-rem +  ((a[i])/2));
    }
    plln(ans % mod);
    return 0;
}
