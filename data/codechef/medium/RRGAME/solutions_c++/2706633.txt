#include <iostream>
#include <stdio.h>
#include <string.h>
#include  <cmath>
#include  <algorithm>
#define LL long long 
using namespace std;
 
LL brr[100001],arr[100001];
int main()
{
    LL n,m,k;
    cin>>n>>m>>k;
    for(int i=0;i<n;i++)
    cin>>arr[i];
    for(int i=0;i<n;i++)
    {
      brr[i]=(m-arr[i])/k +1;
     
    }
    sort(brr,brr+n);
    if(n==1){cout<<1<<endl;
    	return 0;
    }
    LL sum = 0,ans,temp;
    for(int i=0;i<n-2;i++)
    {
      sum+=brr[i];
    }
    //cout<<sum<<endl;
    if(sum>=brr[n-2])
    {
      temp = sum-brr[n-2];
      if(sum+brr[n-2]>=brr[n-1])
      {if(temp%2==0)
        ans = brr[n-1]/2 + 1;
      else
       ans = (brr[n-1]-1)/2 +1;
      }
      else
      {
        if(temp%2==0)
          ans = brr[n-1]/2 - (brr[n-1]-brr[n-2]-sum)/2+1;
        else
          ans = (brr[n-1]-1)/2 -(brr[n-1]-brr[n-2]-sum)/2+1;
      }
    }
    else
    {
        sum = sum+brr[n-2];
        if(sum>=brr[n-1])
        ans = (brr[n-1]-brr[n-2])/2 +1;
        else
        ans = (brr[n-1]-brr[n-2])/2 - (brr[n-1]-sum)/2 +1;
      
    }
    cout<<ans%1000000007<<endl;//cin>>ans;
    return 0;
}
