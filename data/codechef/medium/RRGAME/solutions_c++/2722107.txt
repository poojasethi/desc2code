const int mod = 1e9 + 7;
int main() {
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt","w", stdout);
	cin.sync_with_stdio();
	int n;
	i64 m, k;
	cin >> n >> m >> k;
	vector<i64> a(n);
	forn (i, n)
		cin >> a[i];
	vi64 b;
	forn (i, n) {
		i64 x = (m - a[i]) / k;
		if (a[i] <= m) x++;
		if (x)
			b.pb(x);
	}
	sort(b.begin(), b.end());
	i64 l = 0, r = b.back();
	i64 s = 0;
	forn (i, b.size() - 1)
		s += b[i];
	r = min(r, s);
	s = b[b.size() - 2];
	forn (i, b.size() - 2)
		s -= b[i];
	l = max(l, s);
	
	s = 0;
	forn (i, b.size() - 1)
		s += b[i];
	if (l % 2 != s % 2)
		l++;
	if (r % 2 != s % 2)
		r--;
	i64 ans = (r - l) / 2 + 1;
	cout << ans % mod;
	return 0;
}
