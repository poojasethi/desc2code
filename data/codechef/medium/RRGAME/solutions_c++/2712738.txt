using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    ll t,i,j,n,m,k;
    vector<ll> v;
    ll arr[100002],a[100002];
    ll sum=0;
    cin>>n>>m>>k;
    for(i=0;i<n;++i)
    {
        cin>>arr[i];
        a[i]=(m-arr[i])/k+1;
        v.pb(a[i]);
        sum+=a[i];
    }
    sort(all(v));
    ll mx1=v[n-1],mx2=v[n-2];
    ll sum1=sum-mx1;
    ll sum2=sum1-mx2;
    ll low,high;
    if(sum2<mx2)
    {
        low=mx2-sum2;
    }
    else
    {
        if(sum1%2)
            low=1;
        else
            low=0;
    }
    if(mx1>sum1)
    {
        high=sum1;
    }
    else
    {
        if(sum%2)
            high=mx1-1;
        else
            high=mx1;
    }
    cout<<((high-low)/2+1)%mod;
    return 0;
}
