#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define sl(l) scanf("%lld",&l)
#define pl(l) printf("%lld\n",l)
#define mod 1000000007
int main()
{
	ll t;
	sl(t);
	ll m,k,mx=0,sum=0;
	sl(m);
	sl(k);
	for(ll i=0;i<t;i++)
	{
		ll a;
		sl(a);
		a = (m-a)/k+1;
		sum+=a;
		mx = max(mx,a);
	}

	sum-=mx;
	mx = min(mx,sum);
	ll ans = mx>>1;
	if(!(sum&1))ans++;

	printf("%lld\n",ans%mod);
	return 0;
}
