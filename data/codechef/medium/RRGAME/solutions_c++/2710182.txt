const int N=1111111;
const int M=1111111;

int test;
int n;
ll m,k,ans;
const ll md=inf;
ll a[N];
ll s,max1,max2,cnt1,cnt2;

int main()
{
    //fileio_in_out();
    //fileio_txt();
    //ios::sync_with_stdio(false);
    
    scanf("%d%lld%lld",&n,&m,&k);
    for(int i=1;i<=n;i++)
    {
        scanf("%lld",&a[i]);
        a[i]=(m-a[i])/k+1;
        s+=a[i];
        maxe(max2,a[i]);
        if(max1<max2)
            swap(max1,max2);
    }
    
    cnt1=cnt2=(max1>s/2)?(s-max1):(s/2);
    for(int i=1;i<=n;i++)
    {
        ll ss=s-a[i],mm=(a[i]==max1)?max2:max1;
        ll t=max(ss%2,mm*2-ss);
        if(t<a[i])
            mine(cnt2,(t+ss)/2);
    }
    
    ans=mod(cnt1-cnt2+1,md);
    printf("%lld\n",ans);
    
    //system("pause");
    return 0;
}
