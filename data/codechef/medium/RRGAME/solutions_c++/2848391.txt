#include<iostream>
#include<cstdio>
#define MOD 1000000007
using namespace std;
int main()
{
	int n;
	long long m,k,a[100005],bmax1=0,bmax2=0,temp,s=0,s1,s2;
	scanf("%d",&n);
	scanf("%lld",&m);
	scanf("%lld",&k);
	for(int i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		temp = 1+ ((m-a[i])/k);
		s = s+temp;
		if(bmax1<temp)
		{	
			bmax2=bmax1;
			bmax1 = temp;
		}
		else if(bmax2<temp)
		{
			bmax2=temp;
		}
	}
	s1 = ((s-bmax1)+1)/2;
	/*if((s-bmax1)%2==0)
	{
		s1 = (s-bmax1)/2;
	}
	else
	{	
		s1 = (s-bmax1)/2+1;
	}*/
	s2 = s/2;
	if(bmax2>s1)
	{
		s1 = bmax2;
	}
	if(s2<bmax1)
	{
		s2 = s - bmax1;
	}
	long long ans = (s2-s1+1)%MOD;
	printf("%lld\n",ans);
	return 0;
}
