#include <vector>

#include <string>

#include <cmath>

using namespace std;

#define FOR(i, a, b) for(int i = a; i < b; i++)
#define REP(i, n) FOR(i, 0, n)
#define ll long long
#define mp make_pair
#define pb push_back

#define MOD 1000000007LL

int n;
ll m, k;
ll a[100005], b[100005];

int main()
{
    cin >> n >> m >> k;
    REP(i, n) cin >> a[i];
    
    REP(i, n)
    {
        ll z = m - a[i];
        ll x = z / k;
        a[i] = x + 1;
    }
    
    if(n == 1 or n == 2)
    {
        cout << 1 << endl;
        return 0;
    }
    
    ll mini = 0;
    ll sum = 0;
    
    sort(a, a+n);
    
    REP(i, n-2) sum += a[i];
    if(sum <= a[n-2])
    {
        mini = sum;
        mini += (a[n-2] - sum);
        mini %= MOD;
    }
    else
    {
        mini = (sum + a[n-2]) / 2;
        if((sum + a[n-2]) & 1) mini++;
        mini %= MOD;
    }
    
    ll maxi = 0;
    sort(a, a+n);
    
    sum = 0;
    REP(i, n-1) sum += a[i];
    if(sum <= a[n-1])
    {
        maxi = sum % MOD;
    }
    else
    {
        sum += a[n-1];
        sum /= 2;
        maxi = sum % MOD;
    }
    
    ll ans = maxi + 1;
    ans += MOD - mini;
    ans %= MOD;
    cout << ans << endl;
    return 0;
}
