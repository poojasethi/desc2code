#define MOD (1000000007)
#define MAXINT 1e9

using namespace std;
typedef long long int ll;

ll a[5]={4,2,1};//*0,01,11.
ll ans[5][5];
ll m[5][5]={{1,1,1},{1,0,0},{0,1,0}};

void mult(ll x[5][5],ll y[5][5])
{
	int i,j,k;
	ll t,temp[5][5];
	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)
		{
			t=0;
			for(k=0;k<3;k++)
			{
				t=(t+(x[i][k]*y[k][j])%MOD)%MOD;
			}
			temp[i][j]=t;
		}
	}
	for(i=0;i<3;i++)
	{
		for(j=0;j<3;j++)	ans[i][j]=temp[i][j];
	}
}

void power(ll n)
{
	if(n==0)	return;
	power(n>>1);
	mult(ans,ans);
	if(n&1)	mult(ans,m);
}

ll cal(ll a,ll b)
{
	if(b==0)	return 1;
	ll x=cal(a,b>>1);
	x=(x*x)%MOD;
	if(b&1)	x=(x<<1)%MOD;
	return x;	
}

int main()
{
	int test,i,j;
	ll n,k,q;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%lld",&n);
		if(n==1)	{printf("0\n");continue;}
		else if(n==2)	{printf("0\n");continue;}
		for(i=0;i<5;i++)	{for(j=0;j<5;j++)	ans[i][j]=0;	ans[i][i]=1;}
		power(n-3);
		k=0;
		for(i=0;i<3;i++)
		{
			for(j=0;j<3;j++)
			{
				k=(k+ans[i][j]*a[j])%MOD;
			}
		}
		//cout<<k<<endl;
		q=cal(2,n);
		printf("%lld\n",(q+MOD-k)%MOD);
	}
	return 0;
}
