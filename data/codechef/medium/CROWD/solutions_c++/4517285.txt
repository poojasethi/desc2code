#include<iostream>
using namespace std;
#define MOD 1000000007
#define ll long long int
class mat
{
public:
	ll x11,x12,x13,x21,x22,x23,x31,x32,x33;
};
mat pro(mat x,mat y)
{
	mat z;
	ll i,j,k;
	z.x11=(x.x11*y.x11)%MOD+(x.x12*y.x21)%MOD+(x.x13*y.x31)%MOD;
	z.x12=(x.x11*y.x12)%MOD+(x.x12*y.x22)%MOD+(x.x13*y.x32)%MOD;
	z.x13=(x.x11*y.x13)%MOD+(x.x12*y.x23)%MOD+(x.x13*y.x33)%MOD;
	z.x21=(x.x21*y.x11)%MOD+(x.x22*y.x21)%MOD+(x.x23*y.x31)%MOD;
	z.x22=(x.x21*y.x12)%MOD+(x.x22*y.x22)%MOD+(x.x23*y.x32)%MOD;
	z.x23=(x.x21*y.x13)%MOD+(x.x22*y.x23)%MOD+(x.x23*y.x33)%MOD;
	z.x31=(x.x31*y.x11)%MOD+(x.x32*y.x21)%MOD+(x.x33*y.x31)%MOD;
	z.x32=(x.x31*y.x12)%MOD+(x.x32*y.x22)%MOD+(x.x33*y.x32)%MOD;
	z.x33=(x.x31*y.x13)%MOD+(x.x32*y.x23)%MOD+(x.x33*y.x33)%MOD;
	return z;
}
ll po(ll n)
{
	if(n==0)return 1;
	if(n==1)return 2;
	ll x=po(n/2);
	if(n%2==0)
		return (x*x)%MOD;
	return (2*x*x)%MOD;
}
mat pow(mat a,ll n)
{
	if(n==1)
		return a;
	if(n==0)
	{
		
		a.x11=1;
		a.x12=0;
		a.x13=0;
		a.x21=0;
		a.x22=1;
		a.x23=0;
		a.x31=0;
		a.x32=0;
		a.x33=1;
		return a;
	}
	mat x;
	x=pow(a,n/2);
	if(n%2==0)
		return pro(x,x);
	return pro(a,pro(x,x));
}
int main()
{
	mat a;
	ll two;
	ll t,n;
	ll ans;
	cin>>t;
	while(t--)
	{
		a.x11=1;
		a.x12=1;
		a.x13=0;
		a.x21=1;
		a.x22=0;
		a.x23=1;
		a.x31=1;
		a.x32=0;
		a.x33=0;
		
		
		cin>>n;
		if(n==1||n==2)
		{
			cout<<0<<endl;
			continue;
		}
		two=po(n);
		a=pow(a,n-2);
				
		ans=a.x11;
		ans%=MOD;
		ans+=a.x12;
		ans%=MOD;
		ans+=a.x13;
		ans%=MOD;
		ans+=a.x11;
		ans%=MOD;
		ans+=a.x12;
		ans%=MOD;
		ans+=a.x13;
		ans%=MOD;
		ans+=a.x21;
		ans%=MOD;
		ans+=a.x22;
		ans%=MOD;
		ans+=a.x23;
		ans%=MOD;
		ans+=a.x31;
		ans%=MOD;
		ans+=a.x32;
		ans%=MOD;
		ans+=a.x33;
		ans%=MOD;
		ans=two-ans;
		if(ans<0)ans+=MOD;
		cout<<ans<<endl;
	}
	return 0;
	
}
