#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#define fr(i,n)    for((i)=0;(i)<n;(i)++)
typedef long long int ll;

struct mat
{ ll p[3][3];
};
mat mul(mat a, mat b)
{    int i,j,k;
    mat c;
    fr(i,3) fr(j,3) 
    c.p[i][j]=0;
    fr(i,3)  {  fr(j,3)  {  fr(k,3){
    c.p[i][j]+=(a.p[i][k]*b.p[k][j])%1000000007;
    c.p[i][j]%=1000000007;}}}
    return c;
 
}
mat power(mat m, ll n) {
    if(n == 1) return m;
    mat u = mul(m, m);
    u = power(u, n/2);
    if(n&1) u = mul(u,m);
    return u;
}
ll pwr(ll a, ll n) {
    if(n == 1) return a%1000000007;;
    ll u = (a*a)%1000000007;
    u = pwr(u, n/2);
    if(n&1) u = u*a;
    return u%1000000007;
}
int main()
{
    int t;
    ll n,ans;
    mat z,a;
    a.p[0][0]=1; a.p[0][1]=1; a.p[1][0]=a.p[2][1]=a.p[0][2]=1;
    a.p[1][1]=a.p[1][2]=a.p[2][0]=a.p[2][2]=0;
    scanf("%d",&t);
    while(t--){ scanf("%lld",&n);
                if(n==1)    ans=2;
                else if(n==2) ans=4;
                else{
                    z=power(a,n-2);
                    ans=4*z.p[0][0] + 2*z.p[0][1]+ z.p[0][2];
                    ans%=1000000007;
                    }
                ans=pwr(2,n)-ans;
                ans%=1000000007;
                if(ans<0)    ans+=1000000007;
                printf("%lld\n",ans);
                }
    return 0;
}
