struct mat{
    ll f[3][3];
    mat()
    {
        memset(f,0,sizeof(f));
    }
};

mat operator*(mat a, mat b)
{
    mat c;
    FOR(i,0,2)
    FOR(j,0,2)
    FOR(k,0,2)
    {
        c.f[i][j]+=a.f[i][k]*b.f[k][j];
        c.f[i][j]=c.f[i][j]%modul;
    }
    return c;
}

mat c;

mat calc(ll n)
{
    if (n==1) return c;
    mat p=calc(n/2);
    if (n%2==1) return p*p*c;
    else return p*p;
}
