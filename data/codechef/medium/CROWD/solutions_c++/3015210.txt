lli** mult(lli** a,lli** b)
{
	int i,j,k;
	lli** ans;
	ans = (lli**)calloc(3,sizeof(lli*));

	for(i=0;i<3;i++)
		ans[i] = (lli*)calloc(3,sizeof(lli));

	for(i=0;i<3;i++)
		for(j=0;j<3;j++)
	
		{
			ans[i][j] = 0;
			for(k=0;k<3;k++)
			{
				ans[i][j] += a[i][k]*b[k][j];
				ans[i][j] = ans[i][j] % mod;
			}
		}

	return ans;


}

lli pwr(lli b,lli p)
{

	lli ans=1;
	while(p >= 1)
	{
		if (p%2==1)
		{
			ans = ((ans * b) % mod);
			p -= 1;
		}
		if (p<1)
			break;
		b = (b*b)%mod;
		p = p /2;
	}

	return ans;

}

lli mtx_pwr(lli** b,lli p)
{

	int i,j,k;
	lli** ans;
	ans = (lli**)calloc(3,sizeof(lli*));

	for(i=0;i<3;i++)
		ans[i] = (lli*)calloc(3,sizeof(lli));

	for(i=0;i<3;i++)
		for(j=0;j<3;j++)
		{
			if (i==j)
				ans[i][j] = 1;
			else
				ans[i][j] = 0;
		}

	while(p >= 1)
	{
		if (p%2==1)
		{
			ans = mult(ans,b);
			p -=1;
		}
		if(p<1)
			break;
		b = mult(b,b);
		p = p /2;
	}

	return ans[0][2]*4+ans[1][2]*2+ans[2][2];

}
