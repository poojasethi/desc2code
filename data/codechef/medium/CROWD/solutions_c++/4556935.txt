#include<iostream>
#define ll long long int
#define mod 1000000007
using namespace std;


ll pow_2(ll b)
{
    if(b==0)
        return 1;
    ll z;
    z=pow_2(b/2)%mod;
    z=(z*z)%mod;
    if(b&1)
    {
        z=(z*2)%mod;
    }
    return z;
}

ll multiply(ll a[3][3],ll b[3][3])
{
    ll c[3][3]={0},sum=0;
    int i,j,k;
    for(i=0;i<3;i++)
    {
        for(j=0;j<3;j++)
        {
            sum=0;
            for(k=0;k<3;k++)
            {
                sum=(sum+((a[i][k])*(b[k][j]))%mod)%mod;
            }
            c[i][j]=sum;
        }
    }
    for(i=0;i<3;i++)
    {
        for(j=0;j<3;j++)
        {
            a[i][j]=c[i][j];
        }
    }
}



ll tribonacci(ll a[3][3],ll b[3][3],ll n)
{
    if(n==1)
    {
        return 0;
    }
    tribonacci(a,b,n/2);
    multiply(a,a);
    if(n&1)
    {
        multiply(a,b);
    }
    return 0;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n;
        ll a[3][3]={1,1,1,1,0,0,0,1,0};
        ll b[3][3]={1,1,1,1,0,0,0,1,0};
        cin>>n;
        if(n==1)
            cout<<"0\n";
        else if(n==2)
            cout<<"0\n";
        else if(n==3)
            cout<<"1\n";
        else
        {
            ll ans=pow_2(n);
            tribonacci(a,b,n-3);
            ll temp=0;
            temp=((((a[0][0]*7)%mod+(a[0][1]*4)%mod)%mod)+(a[0][2]*2)%mod)%mod;
            cout<<(ans-temp+mod)%mod<<"\n";
        }
    }
    return 0;
}
