ll c[3][3];
ll exp_by_sqr(ll a,ll n)
{
ll x=1,y=a;
while(n>0)
{
if(n&1)
{
x=(x*y)%MOD;
}
y=(y*y)%MOD;
n=n/2;
}
return x;
}

void matrix(ll a[3][3],ll b[3][3])
{
ll i,j,k;

for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
c[i][j] = 0;
for(k=0;k<3;k++)
{
c[i][j] = (c[i][j]+(a[i][k]*b[k][j])%MOD)%MOD;
}

}
}
for(i=0;i<3;i++)
{
for(j=0;j<3;j++)
{
a[i][j] = c[i][j];
}
}
}
