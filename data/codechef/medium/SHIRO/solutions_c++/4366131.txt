#include<iostream>
#include<cstdio>
using namespace std;
double dp[105][10002]={0};
int main()
{
    int t; cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int flag[n],prob[n];
        int total=0;
        for(int i=1;i<=n;i++)
        {
            cin>>flag[i];
            total=total+flag[i];
        }
        for(int i=1;i<=n;i++)
            cin>>prob[i];
        dp[0][0]=1;
        for(int i=1;i<=n;i++)
        {
            for(int j=0;j<=total;j++)
            {
                dp[i][j]=dp[i-1][j]*(100-prob[i]);
                if(j>=flag[i])
                    dp[i][j]+=dp[i-1][j-flag[i]]*prob[i];
                dp[i][j]/=100;
            }
        }
        double ans=0;
        for(int i=(total+1)/2;i<=total;i++)
            ans+=dp[n][i];
        printf("%.6lf\n",ans);
    }
    return 0;
}
