    using namespace std;
    int n;
    double dp[MAXN][MAXN*MAXN];
    int a[MAXN];
    double p[MAXN];
    ll sum=0;
    double ans = 0.0;
    void read(void)
    {
        sum = 0;
        ans = 0.0;
        cin>>n;
        f(i,0,n)
        {
            scanf("%d",&a[i]);
            sum+=a[i];
        }
        f(i,0,n)
        {
            cin>>p[i];
            p[i]*=0.01;
            //cout<<p[i]<<endl;
        }
    }
    
    void solve()
    {
        int sum = accumulate(a, a+n, 0);
        memset(dp,0,sizeof(dp));
        dp[0][0] = 1.0;
        f(i,0,n)
        {
            f(j,0,sum+1)
            {
                dp[i+1][j+a[i]]+= dp[i][j]*p[i];
                dp[i+1][j]+= dp[i][j]*(1.0-p[i]);
            }
        }
        if(!(sum%2))
        {
            f(j,sum/2,sum+1)
            {
                //cout<<dp[n][j]<<endl;
                ans+=dp[n][j];
            }
        }
        else
        {
            f(j,sum/2+1,sum+1)
            {
                //cout<<dp[n][j]<<endl;
                ans+=dp[n][j];
            }
        }
        printf("%.7lf\n",ans);
    }
    int main()
    {
        int test  = 1;
        cin>>test;
        while(test--)
        {
            read();
            solve();
        }
        return 0;
    }
