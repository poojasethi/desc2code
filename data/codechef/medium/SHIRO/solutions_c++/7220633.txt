using namespace std;

int debug=1;
const int LIM=1e4+5;


/* MAIN PROGRAM STARTS HERE */


int main()
	{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	
	LL T,n,i,j,k,l,m,x,y,z;
	
	cin>>T;
	cout<<fixed<<setprecision(7);
	while(T--)
		{
		cin>>n;
		LL A[n]; double P[n];
		LL s=0;
		REP(i,n) {cin>>A[i]; s+=A[i];}
		REP(i,n) cin>>P[i];
		
		double DP[LIM];
		memset(DP,0.0,sizeof(DP));
		DP[0]=1.0;
		REP(i,n)
			{
			double x=P[i]/100.0;
			for (j=LIM-1-A[i]; j>-1; j--)
				{
				DP[j+A[i]]=DP[j]*(x)+DP[j+A[i]]*(1.0-(x));
				}
			for (j=0; j<A[i]; j++) DP[j]=DP[j]*(1-x);
			}
		LL p=ceil(s/2.0);
		double ans=0.0;
		F(i,p,LIM)
			{
			ans+=DP[i];
			}
		cout<<ans<<endl;
		}
	return 0;
	}
