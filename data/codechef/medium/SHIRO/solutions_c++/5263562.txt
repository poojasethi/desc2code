#include<iostream>

using namespace std;

int main(void) {
	int T,N;
	

	std::cin >> T;
	for (int i=0; i<T; i++)
	{
		std::cin >> N;
		int A[104] = {0};
		double P[104] = {0};
		int totalFlags = 0;
		for (int j=0; j<N; j++)
		{
			std::cin >> A[j];
			totalFlags += A[j];
		}
		//cout << "total flags: " << totalFlags << endl;

		for(int j=0; j<N; j++)
		{
			int a_j;
			std::cin >> a_j;
			P[j] = a_j/100.0;
			//cout << j << "  " << a_j << " " << P[j] << " ";
		}

		double dp[104][10004] = {0};
		dp[0][A[0]] = P[0];
		dp[0][0] = 1-P[0];

		for (int i=1; i<N; i++)
		{
			for (int t=0; t <= totalFlags; t++)
			{
				if (t - A[i] >= 0)
				{
					dp[i][t] = dp[i-1][t-A[i]] * P[i] + dp[i-1][t]*(1-P[i]);
				}
				else
				{
					dp[i][t] = dp[i-1][t]*(1-P[i]);
				}
			}
		}

		int threshold = totalFlags % 2 == 1 ? (totalFlags+1)/2 : totalFlags/2;
		//cout << "threshold: " << threshold << endl;
		double trust = 0;
		for(int i=threshold; i<=totalFlags;i++)
		{
			trust += dp[N-1][i];
		}
		//std::cout << trust << std::endl;
		printf("%.7f\n",trust);
	}

}
