    #include<stdio.h>
    #include<stdlib.h>
    int main(){
    int test;
    scanf("%d",&test);
    while(test){
     
    int noofgroups;
    scanf("%d",&noofgroups);
    int noofflags[noofgroups];
    double prob[noofgroups];
     
    for (int i = 0; i < noofgroups; ++i)
    {
    scanf("%d",&noofflags[i]);
    }
     
    for (int i = 0; i < noofgroups; ++i)
    {
    scanf("%lf",&prob[i]);
    prob[i]=prob[i]/100.0;
    }
     
    double possibleflags[10000];
     
    for (int i = 0; i < 10000; ++i)
    {
    possibleflags[i]=0.0;
    }
    possibleflags[0]=1.0;
     
    int maxflags=0;
    int i,j;
     
    for (i = 0; i < noofgroups; ++i)
    {
    for (j = maxflags+noofflags[i]; j >= 0; j--)
    {
    //printf("%d\n", j-noofflags[i]);
    if(j-noofflags[i]>=0){
    possibleflags[j]=(1-prob[i])*possibleflags[j]+prob[i]*possibleflags[j-noofflags[i]];
    }
    else{
    possibleflags[j]=(1-prob[i])*possibleflags[j];
    }
    }
    maxflags=maxflags+noofflags[i];
    }
     
    double finalans=0.0;
    int start=0;
    if(maxflags%2==0){
    start=maxflags/2;
    }
    else{
    start=maxflags/2+1;
    }
    for (int i = start; i <= maxflags; ++i)
    {
    //printf("%lf ",possibleflags[i]);
    finalans+=possibleflags[i];
    }
    //printf("\n");
    printf("%lf\n",finalans);
     
    test--;
    }
    return 0;
    }
