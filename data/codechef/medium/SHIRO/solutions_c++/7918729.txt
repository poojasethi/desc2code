#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

double dp[10005],p[105];

void solve(int a[], int n){
    int tot=0;
    for(int i=1;i<=n;i++){
        tot += a[i-1];
        for(int j=tot;j>=0;j--){
            if(j-a[i-1]>=0)
                dp[j] = (double)dp[j-a[i-1]]*p[i-1]+(double)dp[j]*(1.0-p[i-1]);
            else
                dp[j] = dp[j]*(double)(1.0-p[i-1]);

            //cout<<i<<" "<<j<<" "<<(double)dp[j]<<" "<<p[i-1]<<"\n";
        }
    }
}

int main()
{
	int t;
	cin>>t;
	while(t--){
        int n;
        cin>>n;
        int a[n],to=0;
        //double p[n];
        for(int i=0;i<n;i++){
            cin>>a[i];
            to += a[i];
        }
        for(int i=0;i<n;i++){
            cin>>p[i];
            p[i] = p[i]/(double)100.0;
            //cout<<p[i]<<" ";
        }
        //cout<<"\n";
        for(int i=1;i<=to;i++){
            dp[i]=0;
        }
        double ans = 0.0;
        dp[0] = 1.0;
        solve(a,n);
        int k = (to+1)/2;
        for(int i=k;i<=to;i++){
            ans+=dp[i];
        }
        printf("%.8lf\n",ans);
	}
	return 0;
}
