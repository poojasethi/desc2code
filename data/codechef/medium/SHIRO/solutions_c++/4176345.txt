#include<iostream>
#include<stdio.h>
using namespace std;
double dp[10005][105];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,j,sum=0,flag[105],prob,temp;
		double p[105],ans;
		cin>>n;
		for(i=1;i<=n;i++)
		{
			cin>>flag[i];
			sum+=flag[i];
		}
		for(i=1;i<=n;i++)
		{
			cin>>prob;
			p[i]=(double)prob/100;
		}
		dp[0][0]=1;
		for(i=1;i<=n;i++)
			dp[i][0]=0;
		for(j=1;j<=n;j++)
			dp[0][j]=dp[0][j-1]*(1-p[j]);
		
		for(i=1;i<=sum;i++)
		{
			for(j=1;j<=n;j++)
			{
				if((i-flag[j])<0)
					dp[i][j]=dp[i][j-1]*(1-p[j]);
				else
					dp[i][j]=dp[i][j-1]*(1-p[j])+dp[i-flag[j]][j-1]*p[j];
			}
		}	
		ans=0.00;
		temp=(sum+1)/2;
		for(i=temp;i<=sum;i++)
			ans+=dp[i][n];		
		printf("%0.7f\n",ans);	
	}
	return 0;
}
