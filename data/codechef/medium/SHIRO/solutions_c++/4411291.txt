#include<bits/stdc++.h>
using namespace std;
float dp[102][10002];
int main()
{
    int t,i,j,n,sum,a[102],d,z;float prob[102],res,s;
    scanf("%d",&t);
    while(t--)
    {
        sum=0;res=0;
        scanf("%d",&n);

        for(i=1;i<=n;i++)
           {
               scanf("%d",&d);
               sum=sum+d;
               a[i]=d;
           }

            for(i=1;i<=n;i++)
            {
                scanf("%f",&s);
                prob[i]=s/100;


            }

            dp[0][0]=1;
            for(i=1;i<=10000;i++)
            {
                dp[0][i]=0;

            }
            for(i=1;i<=n;i++)
            {
                for(j=0;j<=10000;j++)
                {
                    if(j-a[i]>=0)
                    dp[i][j]=dp[i-1][j-a[i]]*prob[i]+dp[i-1][j]*(1-prob[i]);
                    else
                        dp[i][j]=dp[i-1][j]*(1-prob[i]);
                }
            }
            z=(sum%2==0)?(sum/2):(sum+1)/2;
            for(i=z;i<=sum;i++)
                res=res+dp[n][i];
            printf("%f\n",res);
    }
    return 0;
}
