#define sd(x) scanf("%d", &x)
#define sld(x) scanf("%lld", &x)

float dp[101][10000], p[101];
int a[101];

inline void Solve(){
    int n, i, j, x, sum = 0;
    sd(n);
    for(i = 1; i <= n; i++){
        sd(a[i]);
        sum += a[i];
    }
    for(i = 1; i <= n; i++){
        sd(x);
        p[i] = x / 100.0;
    }
    for(i = 1; i <= sum; i++){
        dp[0][i] = 0;
    }
    dp[0][0] = 1.0;
    for(i = 1; i <= n; i++){
        for(j = 0; j <= sum; j++){
            dp[i][j] = (1 - p[i]) * dp[i - 1][j];
            if(j >= a[i])
                dp[i][j] += p[i] * dp[i - 1][j - a[i]];
        }
    }
    float ans = 0.0;
    for(i = sum - sum / 2; i <= sum; i++){
        ans += dp[n][i];
    }
    printf("%f\n", ans);
    return;
}

int main(){
    //freopen("input.txt", "r", stdin);
    int t = 1;
    cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
