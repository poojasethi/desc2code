#include <cstring>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstdio>
using namespace std;
#define mod 1000000007
#define maxn 10010
double dp[maxn][2],p[110];
int flag[110];
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int n;
        scanf("%d",&n);
        memset(dp,0,sizeof(dp));
        int total=0;
        for(int i=0;i<n;i++){
            scanf("%d",&flag[i]);
            total+=flag[i];
        }
        for(int i=0;i<n;i++){
            scanf("%lf",&p[i]);
            p[i]/=100.0;
        }
        dp[0][0]=1;
        int now=1;
        for(int i=0;i<n;i++){
            for(int j=total;j>=0;j--){
                dp[j][now]=dp[j][1-now]*(1-p[i]);
            }
            for(int j=total;j>=flag[i];j--){
                dp[j][now]+=dp[j-flag[i]][1-now]*p[i];
            }
            now=1-now;
        }
        double ans=0;
        for(int i=total;i>=(total+1)/2;i--){
            ans+=dp[i][1-now];
        }
        printf("%.12lf\n",ans);
    }
    return 0;
}
