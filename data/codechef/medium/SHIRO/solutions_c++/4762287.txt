#include <iostream>
#include <cstdio>
using namespace std;
double prob[105][10005];
int main()
{
    int t,n,a[105],sum,p,i,j;
    double ans;
    scanf("%d",&t);
    while(t--)
    {
        ans=sum=0;
        scanf("%d",&n);
        for(i=0;i<=10000;i++)
            for(j=0;j<=n;j++) prob[j][i]=0;
        prob[0][0]=1;
        for(i=1;i<=n;i++)
        {
            scanf("%d",&a[i]);
            sum+=a[i];
        }
        for(i=1;i<=n;i++)
        {
            scanf("%d",&p);
            for(j=0;j<=10000;j++)
            {
                prob[i][j]+=( ((100-p)*prob[i-1][j])/100 ) ;
                prob[i][j+a[i]] += ( (p*prob[i-1][j])/100 ) ;
            }
        }
        sum=(sum+1)/2;
        for(sum=sum;sum<=10000;sum++)
            ans+=prob[n][sum];
        printf("%.6lf\n",ans);

    }
    return 0;
}
