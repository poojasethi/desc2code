double dp[10001];
int tot,n;
int p[100],a[100];

ll powr(int s,int p)
{
	if(p==0)
		return 1;	
 
	if(p%2==1)
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);
		q=(q*s);	
		return ( q );
	}
 
	else
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);	
		return (q);
	}
}

inline void scan()
{
	s1(n)
	//cout<<n<<'\n';
	tot=0;
	FOR(i,0,n)
	{
		s1(a[i])
		tot+=a[i];
	}
	FOR(i,0,n)
		s1(p[i])
}

inline void out()
{
	//cout<<tot<<'\n';
	for(int i=0;i<=tot;i++)
	{		
		dp[i]=0;
	}
	dp[0]=1.0;
	//cout<<tot<<'\n';
	FOR(i,0,n)
	{
		//cout<<tot<<'\n';
		for(int j=tot-a[i];j>=0;j--)
		{
			if(dp[j]>0)
			{
				dp[j+a[i]]+=dp[j]*((double)p[i]/100);
				dp[j]=dp[j]*((double)(100-p[i])/100);
			}
		}
	}
	double ans=0.0;
	//cout<<n<<'\n';;
	/*FOR(i,0,tot+1)
		cout<<dp[i]<<'\n';*/
	FOR(i,0,tot+1)
	{
		if(dp[i]>0)
			if((tot-i)<=i)	ans+=dp[i];
	}
	printf("%.7lf\n",ans);
}

int main()
{
	int t;
	s1(t)
	//cout<<t<<'\n';
	while(t--)
	{
		scan();
		out();
	}
	return 0;
}
