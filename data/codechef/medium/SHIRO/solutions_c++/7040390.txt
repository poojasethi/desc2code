int mod = 1000000007;
lli modpow(lli a, lli b, lli mod){lli res=1;while(b>0){if(b&1)res=(res*a)%mod;a=(a*a)%mod;b=b/2;}return res%mod;} 
int a[101],cum[101];
double p[101],dp[101][10001];

int main()
{
    int t,n,i,j,x;
    sc(t);
    while(t--)
    {
        sc(n);
        for(i=1;i<=n;i++)
            sc(a[i]),cum[i]=cum[i-1]+a[i];
        for(i=1;i<=n;i++)
            sc(x),p[i]=x/100.0;
        memset(dp,0.0,sizeof dp);
        dp[1][a[1]]=p[1];
        dp[1][0]=1-p[1];
        for(i=2;i<=n;i++)
            for(j=0;j<=cum[n];j++)
            {
                dp[i][j]=dp[i-1][j]*(1-p[i]);
                if(j>=a[i]) dp[i][j]+=dp[i-1][j-a[i]]*p[i];
            }
        double ans=0;
        for(j=cum[n]-cum[n]/2;j<=cum[n];j++)
            ans+=dp[n][j];
        printf("%.8lf\n",ans);
    }
}
