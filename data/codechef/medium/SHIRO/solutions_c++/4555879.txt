#include<iostream>
#include <string.h>
using namespace std;
double table[110][10010];
int main()
{
	int T,a[110],n,sum,p,x=0;double s=0.0,total=0.0,pro[110];
	cin>>T;
	while(T--)
	{
		sum=0;total=0;

		memset(table,0,sizeof(table));

		cin>>n;

		int nn=n;
		for(int i=1;i<n+1;i++)
		{
			scanf("%d",&a[i]);
			sum+=a[i];
		}
		nn=n;
		for(int i=1;i<n+1;i++)
		{
			scanf("%d",&x);
			pro[i]=(x*1.0)/100;
		}

		table[0][0]=1;
		for(int i=1;i<n+1;i++)
		{
			for(int j=0;j<sum+1;j++)
			{
				if(j-a[i]>=0)
					table[i][j]+=table[i-1][j-a[i]]*pro[i];

				table[i][j] += (table[i-1][j]*(1-pro[i]));
			}
		}

		for(int i=(sum + 1)/2;i<sum+1;i++)
		{
			total+=table[n][i];
		}
		cout<<total<<endl;
	}
}
