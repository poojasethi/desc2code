#include<stdio.h>
double dp[101][10001];
int main()
{ int t;
  scanf("%d",&t);
  while(t--)
  {
      int n,i,j,sum=0;
      scanf("%d",&n);
      int a[n+1],p[n+1];
      
      for(i=1;i<=n;i++)
       {  scanf("%d",&a[i]);
          sum+=a[i];
       }
      for(i=1;i<=n;i++)
         scanf("%d",&p[i]);
       for(i=0;i<=n;i++)
		{
			for(j=0;j<=sum;j++)
			{
				dp[i][j]=0;
			}
		}
		dp[0][0]=1.0;
		for(i=1;i<=n;i++)
		{
			for(j=0;j<=sum;j++)
			{
				if(a[i]>j)
				{
					dp[i][j]=((dp[i-1][j]*(100-p[i]))/100);
				}
				else
				{
					dp[i][j]=((dp[i-1][j]*(100-p[i]))/100)+((dp[i-1][j-a[i]]*p[i])/100);
				}
			}
		}
	    double ans=0;
		for(j=sum;(j*2)>=sum;j--)
		{
			ans+=dp[n][j];
		}
		printf("%lf\n",ans);
      
  }
  return 0;
  
}
