#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;
	scanf("%d",&test);
	while(test--)
	{
		vector<int> flags;
		vector<float> prob;
		int n,sum=0,abra=0;
		scanf("%d",&n);
		flags.push_back(0);
		for(int i=0;i<n;i++)
		{
			int tmp;
			scanf("%d",&tmp);
			flags.push_back(tmp);
			sum+=tmp;
		}
		prob.push_back(0);
		for(int i=0;i<n;i++)
		{
			int tmp;
			scanf("%d",&tmp);
			prob.push_back(tmp*1.0/100);
		}
		double dp[n+1][sum+1];
		for(int i=0;i<=n;i++)
		{
			for(int j=0;j<=sum;j++)
			{
				dp[i][j]=0;
			}
		}
		//for(int i=0;i<=n;i++)dp[i][0]=1;
		dp[0][0]=1;
		for(int i=1;i<=n;i++)
		{
			for(int j=0;j<=sum;j++)
			{
				if(j-flags[i]>=0)
				dp[i][j]=prob[i]*dp[i-1][j-flags[i]]+(1.0-prob[i])*(dp[i-1][j]);
				else
				dp[i][j]=(1.0-prob[i])*(dp[i-1][j]);
			}
		}
		int x=(sum+1)/2;
		double ans=0;
		for(int i=x;i<=sum;i++)
		{
			ans+=dp[n][i];
		}
		printf("%0.6lf\n",ans);
	}
	return 0;
}
