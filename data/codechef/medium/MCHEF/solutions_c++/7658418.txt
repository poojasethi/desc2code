const int MAXN=100010;
int main()
{
	int t;
	cin>>t;
	while(t--){
		int n,k,m,i,j,l,r,cost[MAXN];
		siii(n,k,m);
		ll sum=0;
		vector<int> neg;
		vector<ll> a(MAXN);
		for(i=0;i<n;i++)
            {
                scanf("%lld",&a[i]);
        sum+=a[i];
        cost[i]=10e6;
        if(a[i]<0)neg.pb(i);
        }
		vector<ll> w;vector<ll> ncost;
		vector< pair<int,pii> > c(MAXN);
		for(i=0;i<m;i++){
			int p,q,r;siii(p,q,r);
			p--;
			q--;
			c[i].se.fi=p;
			c[i].se.se=q;
			c[i].fi=r;
		}
		sort(c.begin(),c.begin()+m);
		for(i=0;i<neg.size();i++){
			for(j=0;j<m;j++){
				if(c[j].se.fi<=neg[i] && c[j].se.se>=neg[i])
                    {cost[neg[i]]=c[j].fi;
                    break;}
			}
			w.pb(a[neg[i]]*-1);
			ncost.pb(cost[neg[i]]);
		}
		ll s[510];
		for(i=0;i<=k;i++)s[i]=0;
		for(j=0;j<w.size();j++){
			for(i=k;i>=1;i--){
				if(ncost[j]<=i && (s[i-ncost[j]]+w[j])>s[i])s[i]=s[i-ncost[j]]+w[j];
			}
		}
		printf("%lld\n",sum+s[k]);
	}
	return 0;
}
