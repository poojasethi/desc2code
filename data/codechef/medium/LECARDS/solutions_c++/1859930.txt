#include<cstdio>
#include<iostream>
#include<algorithm>
#define MOD 1000000007
#define MAX 1002
using namespace std;
int C[MAX][MAX];
int A[MAX];
int main()
{
   int testcase,n;
   scanf("%d",&testcase);
   for(int i=0;i<MAX;i++)
   {
       C[i][0]=1;
   }
   for(int i=0;i<MAX;i++)
   {
       for(int j=1;j<=i;j++)
       {
           C[i][j]=(C[i-1][j]+C[i-1][j-1])%MOD;
       }
   }

   while(testcase--)
   {
       long int res=0;
       scanf("%d",&n);
       for(int i=0;i<n;i++)
       {
           scanf("%d",&A[i]);
       }
       for(int i=0;i<n;i++)
       {

       if(i<n-i)
       {
           res+=C[n][i];
           res%=MOD;
       }
       }
       printf("%ld\n",res);

   }
    return 0;
}
