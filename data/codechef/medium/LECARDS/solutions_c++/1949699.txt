#include<iostream>
#include<stdio.h>
#define M 1000000007
#define siz 1100
#include<math.h>
using namespace std;
int main()
{
static int ar[siz][siz];
static int C[siz];
int n,r,cases;
for(r=0;r<siz;r++)ar[0][r]=0;
    for(n=1;n<siz;n++)
    {
        for(r=0;r<siz;r++)
        {
            if(r>n)
            break;
            if(r==0)
            {
                ar[n][0]=1;
            }
            else if(r==1)
            {
                ar[n][r]=n;
            }
            else
            {
                ar[n][r]=(ar[n-1][r-1] + ar[n-1][r])%M;
            }
        }
    }
    int res,num;
    scanf("%d",&cases);
    for(n=0;n<cases;n++)
    {
        res=0;
        scanf("%d",&num);
        for(r=0;r<num;r++)
        {
            scanf("%d",&C[r]);
        }
        for(r=num/2+1;r<=num;r++)
        {
            res=(res+ar[num][r])%M;
        }
        printf("%d\n",res);
    }

}
