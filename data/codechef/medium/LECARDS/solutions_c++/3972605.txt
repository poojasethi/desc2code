#include<cstdio>
#define MOD 1000000007
	     		
#define INVERSE 500000004
			


int main()
{
int T,C,N;
long long a[1002],perm[1001][1001],temp;
a[0]=1;
for(int i=1;i<=1000;i++)
{
	a[i]=a[i-1]*2%MOD;
}
perm[0][0]=1;
for(int i=1;i<=1000;i++)
{
	perm[i][0]=1;
	perm[i][1]=i;
	perm[i][i]=1;
}
for(int i=3;i<=1000;i++)
	for(int j=2;j<i && j<=500;j++)
		perm[i][j]=(perm[i-1][j]+perm[i-1][j-1])%MOD;


scanf("%d",&T);
while(T-->0)
{
	scanf("%d",&N);
	for(int i=0;i<N;i++)
		scanf("%d",&C);
	if(N%2==1)
		printf("%lld\n",a[N-1]);
	else
	{
		temp=perm[N][N/2];
//		printf("%lld\t%lld\n",temp,a[N]);
		if(a[N]>temp)
			temp=a[N]-temp;
		else
			temp=a[N]-temp+MOD;
		temp=(temp*INVERSE)%MOD;
		printf("%lld\n",temp);		
	}				
}
}
