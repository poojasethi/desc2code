#include<iostream>
//#include<conio.h>
using namespace std;
unsigned long long md=1000000007;
unsigned long long  ar[1010][1010],sum;
int main()
{
	int t,n,j,i,k;
	cin>>t;
	for(i=0;i<=1000;i++)
		{
			ar[i][0]=1;
		//	sum=1;
			//cout<<ar[i][0]<<"  ";
		//	if(i==0){cout<<endl;getch();}
			for(j=1;j<=i/2;j++)
			ar[i][j]=(ar[i-1][j-1]+ar[i-1][j])%md;//sum=(sum+ar[i][j])%md;}//cout<<ar[i][j]<<"  ";} 
			if(i%2)
			ar[i][j]=ar[i][j-1];
			//else
			//cout<<ar[i][j];}//cout<<"\n";getch();
		}
	while(t--)
	{
		cin>>n;
	//for(i=0;i<n;i++)cin>>j;
		sum=0;
		if(n%2)k=n/2;
		else
		k=n/2-1;
	for(i=0;i<n;i++){if(i<=k)sum=(sum+ar[n][i])%md;cin>>j;}
		cout<<sum<<endl;
	}
	return 0;
}
