#include<iostream>
#define MOD 1000000007
using namespace std;
int arr[1001][1001];
int main()
{


    int i,j;

    for(i=1;i<1001;i++)
        for(j=0;j<=i;j++)
    {
        if (j==0)
            arr[i][j]=1;
        else if (j==1)
            arr[i][j]=i;
        else if (i==j)
            arr[i][j]=1;
        else
            arr[i][j]=(arr[i-1][j]%MOD+arr[i-1][j-1]%MOD)%MOD;
    }

    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;

        int number;
        for(i=0;i<n;i++)
            cin>>number;

        int lowerbound=n/2+1;

        long long int result=0;
        for(i=lowerbound;i<=n;i++)
            result=(result+(arr[n][i])%MOD)%MOD;

        cout<<result<<'\n';
    }

    return 0;
}
