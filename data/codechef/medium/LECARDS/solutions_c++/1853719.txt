#include <iostream>   
using namespace std;  
#define FOR(i,a,b) for(int (i) = (a); (i) < (b); ++(i))  
#define MAX 1024
#define MOD 1000000007
int T, n;
int A[MAX];
int C[MAX][MAX];
typedef long long Int;
int main()
{

	// precalculating C(n, k)

	FOR (i,0,MAX)
		C[i][0] = 1;
	FOR (i,1,MAX)
		FOR (j,1,i+1)
			C[i][j] = (C[i-1][j] + C[i-1][j-1]) % MOD;
			


	// reading the number of test cases
	cin >> T;
	// iterating through all test cases
	FOR (t,0,T)
	{
		// reading n
		cin >> n;
		// reading all colors of the cards
		FOR (i,0,n)
			scanf("%d", &A[i]);
		Int res = 0;

		// calculating the result
		FOR (i,0,n)
			{
                  if (i < (n-i))
			{
				res += C[n][i];
				res %= MOD;
			}
			else break;
                  }
			
		// the result output
		cout << res << endl;

	}
	return 0;
}
