#include<cstdio>
//#include<conio.h>
#define p 1000000007
int main()
{
    int a[1001][501],i,j,n,no,r,ans,t;
    for(i=0;i<501;i++)
    a[i][i]=1;
    for(i=0;i<1001;i++)
    a[i][0]=1;
      
    for(i=2;i<1001;i++)
    {
      for(j=1;j<i && j<501 ;j++)
      {
         a[i][j]=(a[i-1][j]+a[i-1][j-1])%p;
      }
    }
    
    scanf("%d",&t);
    while(t--)
    {
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&no);
    if(n%2==0)
    r=n/2;
    else
    r=n/2+1;
    ans=0;
    for(i=0;i<r;i++)
    ans=(ans+a[n][i])%p;
    
    printf("%d\n",ans);
    }
    //getch();
    return 0;
}
