#define M (1000000000+7)


using namespace std;
 static long val[1001][1001];
long int nCr(int n,int r)
{
    if(val[n][r]!=0)
        return val[n][r];

    if(r==0 || r==n)
        return 1;

    long int a,b;
    val[n-1][r]=nCr(n-1,r)%M;
    val[n-1][r-1]=nCr(n-1,r-1)%M;
    return (val[n-1][r]+val[n-1][r-1])%M;
}

int main()
{
    int t,n;

    long c,sum;


   scanf("%d",&t);
   while(t--)
   {   sum=0;
       scanf("%d",&n);
       for(int i=0;i<n;i++)
       {
          scanf("%d",&c);
       }

       for(int i=(n+2)/2;i<=n;i++)
       {
           sum=(sum+nCr(n,i))%M;
       }
       printf("%ld\n",sum);

   }


   return 0;


}
