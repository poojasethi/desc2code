#include <bits/stdc++.h>
#define msg(x) cout << #x << " = " << x << endl
using namespace std;

const int maxN = 1007;
const int mod = 1e9 + 7;

int memo[maxN];
int c[maxN][maxN];
int t, n, x;

void pascal() {
    for (int i = 0; i < maxN; i++) {
        c[i][0] = 1;
    }
    for (int i = 1; i < maxN; i++) {
        for (int j = 1; j <= i; j++) {
            c[i][j] = (c[i - 1][j] + c[i - 1][j - 1]) % mod;
        }
    }
}

inline void add(int& x, int& val) {
    x += val;
    if (x >= mod) x -= mod;
}

int main() {
    cin.sync_with_stdio(0); cin.tie(0);
    pascal();
    cin >> t;
    while (t--) {
        cin >> n;
        for (int i = 0; i < n; i++) {
            cin >> x;
        }
        int ans = 0;
        if (memo[n]) {
            cout << memo[n] << "\n";
            continue;
        }
        for (int i = n / 2 + 1; i <= n; i++) {
            add(ans, c[n][i]);
        }
        memo[n] = ans;
        cout << ans << "\n";
    }
    return 0;
}
