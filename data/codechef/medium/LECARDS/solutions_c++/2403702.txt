#include <iostream>

#include <cmath>

#define mod 1000000007

using namespace std;

long long fact[1005];

void precompute()
{
	fact[0]=fact[1]=1;
	for(int i=2;i<1005;i++)
		fact[i]=(i*fact[i-1])%mod;
}

long long pow(long long a,int b){ // fast exponentation
	if(!b)return 1;else
	if(b==1)return a;else{
		long long tmp=pow(a,b/2);
		tmp=(tmp*tmp)%mod;
		if(b&1)return(tmp*a)%mod;else return tmp;
	}
}

long long divide(long long a,long long b)
{
	return (a*pow(b,mod-2))%mod;
} // division modulo prime number

long long ncr(int n,int r)
{
	return divide(fact[n],(fact[r]*fact[n-r])%mod);
}
