#define sqr(a) (a)*(a)
#define MOD 1000000007
#define MAX 1010
int dp[MAX][MAX];
inline void precompute()
{
    FOR(i,0,MAX)
     dp[i][0]=1;
    FOR(i,1,MAX)
    {
      FOR(j,0,i+1)
        dp[i][j]=(dp[i-1][j]+dp[i-1][j-1])%MOD;          
    }   
}
int main()
{
    precompute();
    int tc;
    input(tc);
    int n,c;
    LL ans;
    while(tc--)
    {
      input(n);
      FOR(i,0,n)
       input(c);
      ans=0;
      int lim=(n+1)/2;
      FOR(i,0,lim)
      {
        ans =(ans + dp[n][i])%MOD;          
      }
      printf("%lld\n",ans);         
    }
    //getch();
    return 0;
}
