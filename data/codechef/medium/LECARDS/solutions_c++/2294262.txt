#include<iostream>
#include<cstdio>
using namespace std;
long long c[1001][1001], res;
int t, n, bak;
void init(){
    int i, j;
    for(i=1; i<1001; ++i){
        c[i][1] = i;
    }
    for(i=2; i<1001; ++i){
        for(j=2; j<=i; ++j)
            c[i][j] = (c[i-1][j]+c[i-1][j-1])%1000000007;
    }
}
int main(){
    //freopen("input.txt", "r", stdin);
    init();
    int i;
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        for(i=0; i<n; ++i)
            scanf("%d", &bak);
        res = 0;
        for(i=n/2+1; i<=n; ++i){
            res += c[n][i];
            res %= 1000000007;
        }
        printf("%lld\n", res);
    }
    return 0;
}
