#include<stdio.h>
#include<iostream>
using namespace std;
int pascal[1001][501];
int main()
{
int i,j,t;
for(i=0;i<1001;i++)
{
for(j=0;j<=i;j++)
{
if(j==0 || j==i)
pascal[i][j]=1;
else
pascal[i][j]= ((pascal[i-1][j-1]) +  (pascal[i-1][j]))%1000000007;
if(j==500)
break;
}
}
cin>>t;
while(t--)
{
          int n,c,ans=0;;
          cin>>n;
          for(int k=0;k<n;k++)
          {
                  cin>>c;
          }
          int m= n/2;
          m= m+1;
          for(;m<=n;m++)
          {
                 ans=(ans+(pascal[n][n-m]))%1000000007;
          }
          
          cout<<ans<<endl;
}
return 0;
}
