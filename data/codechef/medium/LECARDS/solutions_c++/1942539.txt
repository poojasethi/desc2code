#include <iostream>
#include <cstdio>
#define M 1000000007
using namespace std;

int main ()
{
        int a[1001][1001];
        int t;
        int n;
        int p, i, j;
        long long int s;
        for (i = 0; i < 1001; ++i) {
                for (j = 0; j <= i; j++) {
                        if (j) {
                                a[i][j] = (a[i-1][j] + a[i-1][j-1]) % M;
                        } else {
                                a[i][j] = 1;
                        }
                }
        }

        scanf ("%d", &t);
        while (t--) {
                scanf ("%d", &n);
                for (i = 0; i < n; ++i) {
                        scanf ("%d", &p);
                }
                s = 0;
                i = 0;
                while (i < n - i) {
                        s = (s + a[n][i]) % M;
                        i++;
                }

                printf ("%lld\n", s);
        }

        return 0;
}
