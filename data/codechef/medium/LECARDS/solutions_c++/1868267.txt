#include<iostream>
#define MAX 1001
#define MOD 1000000007
using namespace std;
int c[MAX][MAX];
int main()
{
    int t,n,i,j,a;
    long long res;
    
    scanf("%d",&t);
    
    for(i=0;i<MAX;i++)
        c[i][0]=1;
    for(i=1;i<MAX;i++)
        for(j=0;j<i+1;j++)
            c[i][j]=(c[i-1][j]+c[i-1][j-1])%MOD;
    
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%d",&a);
        res=0;
        for(i=n/2+1;i<=n;i++)
        {
            res+=c[n][i];
            res%=MOD;
        }
        cout<<res<<endl;        
    }
    return 0;
}
