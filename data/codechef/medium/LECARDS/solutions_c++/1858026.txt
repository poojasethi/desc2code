#include <iostream>
#include <cmath>
#define mod 1000000007
long a[1001][1001];
using namespace std;
 
long ini()
{
for(int i=0;i<=1000;i++)
{
a[i][i]=1;
a[i][0]=1;
}
 
for(int i=2;i<=1000;i++)
{
for(int j=1;j<i;j++)
{
a[i][j]=(a[i-1][j-1]+a[i-1][j])%mod;
}
}
 
}
int main()
{
ini();
int t,x,temp,temp2;
long n,ans;
cin>>t;
while(t--)
{
ans=0;
cin>>n;
temp=n;
while(temp--)
cin>>temp2;
x=floor(n/2.0)+1;
for(int i=x ;i<=n;i++)
{
ans=(ans+a[n][i])%mod;
}
cout<<ans<<endl;
}
 
}
