#define RFOR(i,x,y) for(int i=(x);i>=(y);i--)

#define MOD 1000000007
#define INV 500000004
typedef long long int lli;
typedef long double ld;

using namespace std;

int bin[1010][1010];

void init()
{
	mset(bin,-1);
	for(int i=0;i<1010;i++) bin[i][0]=bin[i][i]=1;
	for(int i=1;i<1010;i++)
		for(int j=1;j<i;j++) bin[i][j]=(bin[i-1][j-1]+bin[i-1][j])%MOD;
}

lli exp(int n, int p)
{
	if(p==0) return 1;
	if(p==1) return n;
	lli temp=exp(n,p/2);
	temp=(temp*temp)%MOD;
	if(p%2) return (temp*n)%MOD;
	return temp;
}

int main()
{
   // freopen("inp.txt","r",stdin);
    init();
    int t,n;
    lli sum;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	for(int i=0;i<n;i++) scanf("%lld",&sum);
    	if(n%2) printf("%lld\n",exp(2,n-1));
    	else
    	{
    		sum=exp(2,n);
    		sum-=bin[n][n/2];
    		sum%=MOD;
    		sum=(sum*INV)%MOD;
    		if(sum<0) sum+=MOD;
    		printf("%lld\n",sum);
    	}
    }
    return 0;
    
}
