//FILE *fin = freopen("in","r",stdin);

const int N = 2600;
VII g[N];
LL dist[N];
II T[N];
int vis[N];
typedef pair<LL,int> state;
const LL INF = LL(2e18);
int main()
{
	DRT()
	{
		int n;si(n);
		for(int i=0;i<n;i++)
		{
			si(T[i].F);si(T[i].S);
		}
		int s = n,d=n+1;
		si(T[n].F);si(T[n].S);
		si(T[n+1].F);si(T[n+1].S);
		n+=2;
		for(int i=0;i<n;i++)
			for(int j=i+1;j<n;j++)
			{
				int dd = (T[i].F-T[j].F)*(T[i].F-T[j].F) + (T[i].S-T[j].S)*(T[i].S-T[j].S);
				g[i].PB(MP(j,dd));
				g[j].PB(MP(i,dd));
			}
		set<state> S;
		S.insert(MP(0ll,s));
		for(int i=0;i<N;i++)
			dist[i]=INF,vis[i]=0;
		dist[s]=0ll;
		while(!S.empty())
		{
			state top = *S.begin();
			S.erase(S.begin());
			int u = top.S;
			if(vis[u])continue;
			vis[u]=1;
			for(int i=0;i<SZ(g[u]);i++)
			{
				int w = g[u][i].F;
				int weight = g[u][i].S;
				if(dist[u]+weight < dist[w])
				{
					dist[w]=dist[u]+weight;
					S.insert(MP(dist[w],w));
				}
			}
		}
		lldout(dist[d]);
		for(int i=0;i<n;i++)
			g[i].clear();
	}
	return 0;
}
