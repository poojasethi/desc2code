#include<stdio.h>
#define inf 1000000000
int a[2505][2505],x[2505],y[2505],v[2505],mark[2505];
int dist(int x1,int y1,int x2,int y2)
{
	return (((x2-x1)*(x2-x1))+((y2-y1)*(y2-y1)));
}
int main()
{
	#ifndef ONLINE_JUDGE
		freopen("in.txt","r",stdin);
	#endif
	int t,n,i,j,min,mini;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n+2;i++)
		{
			scanf("%d %d",&x[i],&y[i]);
			v[i]=inf;
			a[i][i]=inf;
			mark[i]=0;
		}
		for(i=0;i<n+2;i++)
		{
			for(j=0;j<i;j++)
			{
				a[i][j]=dist(x[i],y[i],x[j],y[j]);
				a[j][i]=dist(x[i],y[i],x[j],y[j]);
			}
		}
		
//		for(i=0;i<n+2;i++)
//		{
//			for(j=0;j<n+2;j++)
//			{
//				printf("%d ",a[i][j]);
//			}
//			printf("\n");
//		}
		
		
		v[n]=0;
		mark[n]=1;
		mini=n;
		for(i=0;i<n+2;i++)
		{
			min=inf;
			for(j=0;j<n+2;j++)
			{
				if(mark[j]==0 && v[j]<min)
				{
					min=v[j];
					mini=j;
				}
			}
			
			if(mini==n+1)
				break;
			
			mark[mini]=1;
			
			for(j=0;j<n+2;j++)
			{
				if(mark[j]==0 && a[j][mini]+v[mini]<v[j])
				{
					v[j]=a[j][mini]+v[mini];
				}
			}
		}
		printf("%d\n",v[n+1]);
	}
	return 0;
}
