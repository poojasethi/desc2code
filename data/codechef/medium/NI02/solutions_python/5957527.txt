#include<stdio.h>
#include<limits.h>
int t,n,a[2550][2],i,ans=0,s[2550][2550],j;
int dijsktra(int cost[][2550],int source,int target)
{
    int dist[n],prev[n],selected[n],i,m,min,start,d,j;
    for(i=0;i<n;i++)
        selected[i]=0;
    for(i=1;i<n;i++)
    {
        dist[i] = INT_MAX;
        prev[i] = -1;
    }
    start = source;
    selected[start]=1;
    dist[start] = 0;
    while(selected[target] ==0)
    {
        min = INT_MAX;
        m = 0;
        for(i=1;i< n;i++)
        {
            d = dist[start] +cost[start][i];
            if(d< dist[i]&&selected[i]==0)
            {
                dist[i] = d;
                prev[i] = start;
            }
            if(min>dist[i] && selected[i]==0)
            {
                min = dist[i];
                m = i;
            }
        }
        start = m;
        selected[start] = 1;
    }
    start = target;
    return dist[target];
}
 
int main()
{
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        scanf("%d",&n);
        for(i=1;i<=n;i++)
            scanf("%d%d",&a[i][0],&a[i][1]);
        scanf("%d%d",&a[0][0],&a[0][1]);
        scanf("%d%d",&a[i][0],&a[i][1]);
 
        for(i=0;i<n+2;i++)
            for(j=0;j<n+2;j++)
                s[i][j]=((a[i][0]-a[j][0])*(a[i][0]-a[j][0]))+((a[i][1]-a[j][1])*(a[i][1]-a[j][1]));
 
        n+=2;
        ans=dijsktra(s, 0 ,n-1);
        printf("%d\n",ans);
    }
}
