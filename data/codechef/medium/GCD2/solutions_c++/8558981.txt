#include <bits/stdc++.h>
using namespace std;

int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main(){
	int T;
	cin>>T;
	for(int t=0;t<T;t++){
	int n;
	string s;
	cin>>n>>s;
	int m=0;
	int d;
	if(n==0){
		for(int i=0;i<s.size();i++){
			cout<<s[i];
			}
		cout<<endl;
		}
	else{
	for(int i=0;i<s.size();i++){
		m += s[i] - '0';
		d = m%n;
		m = 10*d;
		}
	cout<<gcd(d,n)<<endl;
	}
	}
}
