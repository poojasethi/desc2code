#include <iostream>
#include <stdio.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while ( t-- ) {
		int A, B;
		char b[300];
		cin>>A;
		cin.ignore();
		gets( b );
		if ( A == 0 ) {
			puts( b );
			continue;
		}
		B = 0;
		for ( int i = 0; b[i] != '\0'; i++ ) {
			B = (( B*10 ) + ( b[i] - '0' )) % A;
		}
		int temp;
		while ( B != 0) {
			temp = B;
			B = A % B;
			A = temp;
		}
		cout<<A<<endl;
	}
	return 0;
}
