#include <bits/stdc++.h>
using namespace std;

int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}

int main ()
{
    int t,len;
    cin>>t;
    int a;
    char b[252];
    int bNum;

    while(t--)
    {
        scanf ("%d",&a);
        scanf("%s",b);
        len = strlen(b);
        bNum = 0;
        if (a==0)
        {
            cout<<b<<endl;
        }
        else
        {
            for (int i =0;i<len;i++)
            {
                bNum = (bNum*10 + b[i] - '0')%a;
                //cout<<i<<"   "<<bNum<<endl;
            }

            cout<<gcd(bNum,a)<<endl;
        }



    }


    return 0;
}
