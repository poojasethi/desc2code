#include<cstdio>
#include<bits/stdc++.h>
#include <iostream>
#include<string.h>
#include<stdio.h>
using namespace std;

int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int main()
{
    int t,a,n,temp,k,c;
    char s[251];
    cin>>t;
    while(t--)
    {
        scanf("%d",&a);
        scanf("%s",s);

        n=strlen(s);
        int i=0;
        temp=0;
        if(!a)
        {
           cout<<s<<endl;
           continue; 
        }    
        
        while(i<n)
        {
            temp=temp*10+s[i]-'0';
            temp=temp%a;
            i++;
        }
        k=gcd(a,temp);
        printf("%d\n",k);
        
       
    }

        return 0;
}
