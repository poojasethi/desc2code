    #include<iostream>
    #include<stdio.h>
    #include<bits/stdc++.h>
    using namespace std;

    int main() {
	int test,a,B,temp;
	char b[300];
	cin>>test;
	for(int i=0;i<test; ++i) {
		cin>>a;
           cin.ignore();
		gets(b);
		if(a==0) {
			puts(b);
			continue;
		}
		B=0;
		for(int j=0; b[j]!='\0'; ++j)
			B=((B*10)+(b[j]-'0'))%a;

		while(B!=0){
			temp=B;

			B=a%B;

			a=temp;
		}
	        cout<<a<<endl;
	}
    }
