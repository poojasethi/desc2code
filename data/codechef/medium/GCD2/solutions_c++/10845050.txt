#include<cstdio>
#include<iostream>
 
using namespace std;
 
int gcd(int a,int b)
{
	if (b==0)
	return a;
	else
	return gcd(b,a%b);
}
 
int main()
{
	int i,T,A,r;
	char ch,B[251];
	
	cin>>T;
	
	while(T--)
	{
		cin>>A;
		scanf("%s",B);
		
		if (A==0)
		{
			printf("%s\n",B);
			continue;
		}
		
		r=0;
		i=0;
		while (B[i] != '\0')
		{
			r = (r *10+ (B[i]-'0'))%A;
			i++;
		}
		cout<<gcd(A,r)<<"\n";
		
	}
	
	return 0;
}
