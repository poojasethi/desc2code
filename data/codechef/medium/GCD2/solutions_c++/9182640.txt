#include<stdio.h>
#include<string.h>
#include<math.h>
int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main()
{
      int i,j,k,a,b,c;
      char st[255];
      scanf("%d",&a);
      
      while(a--)
      {
            scanf("%d%s",&b,st);
            if(b==0)
            {
                  printf("%s\n",st);
                  continue;
            }
           j=0;k=0;
            for(i=0;i<strlen(st);i++)
            {
                  
                 j*=10;
                  j+=(st[i]-48);
                  //printf("sfj%d\n",j);
                  j=j%b;
          
            }
            //printf("%d %dhhh\n",b,j);
            printf("%d\n",gcd(b,j));
      }
}
