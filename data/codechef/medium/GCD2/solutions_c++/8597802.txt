#include <iostream>
using namespace std;

int getgcd(int a,int b)
{
	if(b==0)
		return a;
	else
		return getgcd(b,a%b);
}

int main() {
	
	int a,t;
	char b[251];
	cin>>t;
	while(t--){int temp=0;
		cin>>a;
		cin>>b;
	if(a!=0)
	{
		char *p=b;
		while(*p!='\0')
		{
			temp*=10;
			temp+=(*p++)-'0';
			temp=temp%a;
		}
		cout<<getgcd(a,temp)<<"\n";
	}
	else
		cout<<b<<"\n";
	}
	
	// your code goes here
	return 0;
}
