#include<iostream>
using namespace std;
#include<stdio.h>
#include<string.h>
int isDiv(int b[], int n, int s)
{
		int r[s];
		int i;
		for(i=0; i<s; i++)
		{
			if(i==0)
				r[i]=b[0]%n;
			else
				r[i]=(r[i-1]*10+b[i])%n;
		}
		if(r[i-1]==0)
			return 1;
		return 0;
}
int gcd(int a, int b[], int s)
{
    int G=1;
    for(int i=2; i<=a; i++)
		{
			if(a%i==0&&(isDiv(b, i, s)==1))
				G=i;
		}
		return G;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int a;
        char str[251];
        scanf("%d%s",&a, str);
        if(a==0)
            printf("%s\n", str);
        else if(a==1)
            printf("%d\n",1);
        else{
            int s=strlen(str), b[s];
            for(int i=0; i<s; i++)
            {
                b[i]=(int)str[i]-(int)'0';
            }
            cout<<gcd(a, b, s)<<endl;
        }
    }
    return 0;
}
