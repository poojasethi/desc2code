#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main()
{
	char b[251];
	int l,i,t;
	int x,y=0,a;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&a);
		scanf("%s",b);
		l=strlen(b);
		if(a==0)
		printf("%s\n",b);
		else
		{
		for(i=0;i<l;i++)
		{
		  x=(b[i]-'0')+y;
		  y=x%a;
		  y=y*10; 	
		}
		y=y/10;
		a=gcd(y,a);
		printf("%d\n",a);
		y=0;
	    }
	}
	return 0;
}
