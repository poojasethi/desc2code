#include <bits/stdc++.h>

using namespace std;
int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int main()
{
    //cout << "Hello world!" << endl;
    int t;
    cin>>t;
    while(t--)
    {
        int temp=0,a;
        char c[10000];
        cin>>a>>c;
        int l=strlen(c);
        if(a==0)
        {
            cout<<c<<endl;
            continue;
        }

        for(int i=0;i<l;i++)
        {
            temp=(temp*10+(c[i]-'0'))%a;
//cout<<temp<<" ";
        }
//cout<<endl;
        cout<<gcd(a,temp)<<endl;
    }
    return 0;
}
