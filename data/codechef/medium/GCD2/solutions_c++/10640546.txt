#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int main()
{
    int t,a,n,m,i,c,temp;
    char b[250];
    scanf("%d",&t);
    while(t--)
    {
        m=0;
        cin>>a;
        cin>>b;
        temp=a;
        if(a==0)
            cout<<b<<endl;
        else
        {
            while(temp!=0)
        {
            m++;
            temp/=10;
        }
        for(i=0;b[i]!='\0';i++);
        n=i;
        c=0;
        for(i=0;i<m;i++)
        {
            c+=(int)(b[i]-'0')*pow(10,(m-i-1));
        }
        c=c%a;
        while(i<n)
        {
            if(i==n)
                break;
            c*=10;
            c+=(int)(b[i]-'0');
            c=c%a;
            i++;
        }
        cout<<gcd(a,c)<<endl;
        }
    }
    return 0;
}
