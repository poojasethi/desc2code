#include<iostream>
using namespace std;
long gcd(long a,long b)
{
	if(b==0) return a;
	else return gcd(b,a%b);
}
long rem(long a,char *b)
{
	long r=0;
	for(long i=0;b[i]!='\0';i++)
	{
		long c = b[i]-'0';
		r=(c+r*10)%a;
	}
	return r;
}
int main()
{
	long t; cin>>t;
	while(t--)
	{
		long a;
		char b[250]; cin>>a>>b;
		if(a==0) cout<<b<<endl;
		else
		{
			long r=rem(a,b);
			cout<<gcd(a,r)<<endl;
		}
	}
}
