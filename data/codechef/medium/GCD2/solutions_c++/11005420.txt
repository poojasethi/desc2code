#include<bits/stdc++.h>
using namespace std;
 
int gcd(int a,int b){
    if(b==0)
        return a;
    gcd(b,a%b);    
}
int main(void)
{
	int t;
	cin>>t;
	while(t--)
	{
		int n1;
		string str;
		cin>>n1;
		cin>>str;
		int len=str.length();
		int i;
		int temp,ptemp=0;
		if(n1==0)
			cout<<str<<endl;
		else{	
		for(i=0;i<len;i++)
		{
			temp=ptemp*10+str[i]-'0';
			ptemp=temp%n1;
		}
		
		int res=gcd(n1,ptemp);
		cout<<res<<endl;
		}
		
	}
		return 0;
	}
