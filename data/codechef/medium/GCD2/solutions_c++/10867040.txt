#include<cstdio>
using namespace std;
int gcd(int a,int b){
    if(a<b){
        int tmp=a;
        a=b;
        b=a;
    }
    while(b){
        int tmp=a%b;
        a=b;
        b=tmp;
    }
    return a;
}
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int A;
        char B_str[255];
        scanf("%d %s",&A,&B_str);
        if(A==0){
            printf("%s\n",B_str);
            continue;
        }
        else
        {
            int B=0;
            for(int i=0;B_str[i]!='\0';i++){
                B=((B*10)+(B_str[i]-'0'))%A;
            }
            printf("%d\n",gcd(A,B));
        }
    }
}
