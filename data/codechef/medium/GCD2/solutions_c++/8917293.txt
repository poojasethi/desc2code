#include<iostream>
#include<cstdio>
#include<string>
#include<cstdlib>
using namespace std;
 int gcd(int a,int b)
 {
 	if(b==0)
 	return a;
 	return gcd(b,a%b);
 }

int find_mod(string s,int a)
{
	int sz=s.size(),i;
	int sum=0;
	for(i=0;i<sz;i++)
	{
		sum=(10*sum+(s[i]-'0'));
		sum=sum%a;
	}
	return sum;
}

int main()
{
	int a;
	string b;
	int t;
	cin>>t;
	while(t--)
	{
		cin>>a;
		cin>>b;
		if(a==0)
		{
		cout<<b<<endl;
		continue;
		}
		
		cout<<gcd(a,find_mod(b,a))<<endl;
	}
	
	
	return 0;
}
