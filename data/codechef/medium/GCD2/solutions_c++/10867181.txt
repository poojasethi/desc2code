#include <bits/stdc++.h>
#include<math.h>
#include<iostream>
using namespace std;

#define S(x) scanf("%d",&x)
#define SS(x) scanf("%s",x)
#define FOR(i,s,e) for(i=(s);i<(e);i++)
#define P(x) printf("%d",(x))
#define NL printf("\n")
int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main() {
	int t,a,l,g,i;
	char b[255];
	S(t);
	while(t--)
	{
	    S(a);
	    SS(b);
	    if(a==0)
	    {
	        printf("%s\n",b);
	        continue;
	    }
	    int l=strlen(b);
	    g=b[0]-'0';
	    FOR(i,1,l)
	    g=(g*10+b[i]-'0')%a;
	    g=gcd(g,a);
	    P(g);
	    NL;
	    
	    
	}
	return 0;
}
