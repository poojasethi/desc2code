#include <stdio.h>
#include <string>
#include <iostream>
using namespace std;
int getSimpleNumber(int num1,string num2){
	int index = 0;
	int new_num2 = 0;
	while(num2[index] != '\0'){
		new_num2 = new_num2*10+(num2[index]-'0');
		new_num2 %= num1;
		index++;
	}
	return new_num2;
}
int gcd(int a,int b){
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main(){
	int t;
	cin>>t;
	int num1;
	string num2;
	while(t--){
		cin>>num1>>num2;
		if(num1 == 0)
			cout<<num2<<endl;
		else
			cout<<gcd(num1,getSimpleNumber(num1,num2))<<endl;
	}
	return 0;
}
