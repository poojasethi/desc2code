long long int gcd(long long int  a,long long int b)
{
	if(a==0)
	return b;
	return gcd(b%a,a);
}
int main()
{
     long long int i,j,k,l,a,final,rem,temp,t;
     scanf("%lld",&t);
     while(t--)
     {
     scanf("%lld",&a);
     char str[300];
     scanf("%s",str);
     if(a==0)
     {
     printf("%s\n",str);
     continue;
     }
     //we know that gcd(a,b)=gcd(b,a%b)    "euclid's algorithm
    temp=str[0]-48;
    l=strlen(str);
    rem=temp%a;
    //printf("temp=%lld rem==%lld\n",temp,rem);
     for(i=1;i<l;i++)
     	{
     	temp=str[i]-48;
		 rem=(rem*10+temp)%a;
		 //printf("rem==%lld\n",rem);
	}
    printf("%lld\n",gcd(rem,a));
    }
return 0;
}
