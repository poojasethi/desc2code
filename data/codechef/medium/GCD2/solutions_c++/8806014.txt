#include<cstdio>
#include<string>
#include<iostream>
using namespace std;
int a,i,t,d,p,x;
string b;
int gcd(int a,int b){
    while(b!=0){
        int c=b;
        b=a%b;
        a=c;
    }
    return a;
}
int main(){
        scanf("%d", &t);
        for(i=1;i<=t;i++){
            p=0;
            cin>>a>>b;
            if(a==0) {cout<<b<<endl; continue;}
            x=b.size();
            for(d=0;d<x;d++){
                p=p*10+b[d]-48;
                p=p%a;
            }
            printf("%d\n", gcd(a,p));
        }

    return 0;
}
