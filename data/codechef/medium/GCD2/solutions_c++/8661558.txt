#include <iostream>
#include <cstdio>
#include <string.h>
#include <stdlib.h>

using namespace std;

int hcf(long int a,long int b)
{
	if (b==0)
		return a;
	else
		return hcf(b,a%b);
}

int main()
{
    int tc;
    scanf("%d", &tc);
    while(tc--)
    {
        long int x, a = 0;
        int n;
        char str[251];
        scanf("%ld", &x);
        scanf("%s", str);
        if(x == 0)
        {
            printf("%s\n", str);
            continue;
        }
        n = strlen(str);
        for(int i = 0; i < n; i++)
        {
            //cout<< a<<" "<< x<<" ";
            a = a*10 + (str[i] - '0');
           // cout<< a<<" ";
            if(a >= x)
            {
                a %= x;
            }
            //cout<<str[i]<<" "<<a<<endl;
        }

        printf("%ld\n", hcf(x, a));
    }
    return 0;
}
