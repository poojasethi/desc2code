#include<iostream>
using namespace std;
int gcd(int a, int b)
{
	if (b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main(){
int t,num1;
string num2;
cin>>t;
while(t--){
    cin>>num1>>num2;
    if(num1==0) cout<<num2<<endl;
    else{
        int px=0,y=0;
        for(int i=0;num2[i]!='\0';i++){
            int x=num2[i]-'0';
            px=y*10+x;
            //cout<<"px="<<px<<endl;
			y=px%num1;
			//cout<<"y="<<y<<endl;
        }
    cout<<gcd(y,num1)<<endl;
    }
}
}
