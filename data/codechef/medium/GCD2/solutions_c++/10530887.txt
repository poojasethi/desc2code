#include<stdio.h>
#include<string.h>
int longgcd(int a,char arr[])
{
    int n=strlen(arr);

    for(int i=0;i<n;i++)
    {
        arr[i]=arr[i]-'0';
    }

    int rem=0;

    for(int i=0;i<n;i++)
    {
        rem=rem*10+arr[i];

        rem=rem%a;
    }

    return rem;
}
int gcd(int a, int b)
{
	if (b==0)
    return a;

	else
    return gcd(b,a%b);
}
int main()
{
    int t,m;
    char arr[255];

    scanf("%d",&t);

    while(t--)
    {
        scanf("%d %s",&m,&arr);

        if(arr[0]-'0'==0&&m==0)
        {
            printf("0\n");
            continue;
        }

        if(m==0)
        {
            printf("%s\n",arr);
            continue;
        }

        int rem=longgcd(m,arr);

        printf("%d\n",gcd(m,rem));
    }
    return 0;
}
