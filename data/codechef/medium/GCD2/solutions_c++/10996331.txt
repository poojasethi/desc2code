#include<bits/stdc++.h>
using namespace std;
int mod(char s[],int d)
{
    int r = 0;
    int i;
    for(i=0;s[i];++i)
    {
      r=10*r +(s[i] - 48);
      r = r % d;
    }
    return r;
}
int gcd(int a,int b){
	if(b == 0)
		return a;
	else
	        gcd(b,a%b);
}
int main() {
	int t;
	cin>>t;
	while(t--) {
	  int a;
	  cin>>a;
	  char b[257];
	  cin>>b;
	  if(a == 0 ) {  cout<<b<<"\n"; }
	  else {
	  	cout<<gcd(a,mod(b,a))<<"\n";
	  }
	}
}
