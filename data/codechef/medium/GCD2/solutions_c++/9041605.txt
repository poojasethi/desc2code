#include<iostream>
#include<stdio.h>
using namespace std;
 
int main() {
	int T,t=0,A,B,temp;
	char b[300];
	scanf("%d", &T);
	for(int i=0;i<T; ++i) {
		cin>>A;
		cin.ignore();
		gets(b);
		if(A==0) {
			puts(b);
			continue;
		}
		B=0;
		for(int j=0; b[j]!='\0'; ++j) 
			B=((B*10)+(b[j]-'0'))%A;
		while(B!=0){
			temp=B;
			B=A%B;
			A=temp;
		}
	        printf("%d\n", A);
	}
}
