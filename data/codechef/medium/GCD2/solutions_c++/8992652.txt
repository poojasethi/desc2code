#include<iostream>
#include<cstring>
using namespace std;
int gcd(int x, int y)
{
    if(y==0)    return x;
    return gcd(y, x%y);
}
int main()
{
    int N;
    cin>>N;

    int i, j, k, ans, a;
    char b[300];

    for(i=0;i<N;i++) {
        cin>>a;
        cin>>b;
        if(a==0) {
            cout<<b<<endl;
            continue;
        }

        int m, n, d;
        m=0;
        for(j=0;j<strlen(b);j++) {
            m=10*m + (int)b[j] - 48;
            m=m%a;
        }
        int x, y;
        x=a;
        y=m;
        if(y==0)
            cout<<a<<endl;
        else        cout<<gcd(x,y)<<endl;
    }
}
