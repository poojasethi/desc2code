#include <iostream>
#include <cstring>
using namespace std;

int gcd(int a, int b)
{
if(b==0)
return a;
else
return gcd(b,a%b);
}

int main() {
	int t,i,len,ans,a,rem,k,flag;
	char b[300];
	scanf("%d",&t);
	while(t--)
	{
	flag=0;
	rem=0;
	k=1;
	scanf("%d %s",&a,&b);
	len=strlen(b);
	//put 0 0, 0 n, n 0 conditions!
	if((b[0]=='0')&&(a==0))
	{
	ans=0;
	flag=1;
	}
	else if(b[0]=='0')
	{
	ans=a;
	flag=1;
	}
	else if(a==0)
	{
	for(i=0;i<len;i++)
	{
	printf("%c",b[i]);
	}
	printf("\n");
	flag=2;
	}
	else if(len<4)
	{
	for(i=len-1;i>=0;i--)
	{
	rem=rem+k*(b[i]-'0');
	k=k*10;
	}
	}
	else 
	{
	k=1000;
	rem=0;
	for(i=0;i<4;i++)
	{
	rem=rem+k*(b[i]-'0');
	k=k/10;
	}
	rem=rem%a;
	k=10;
	for(i=4;i<len;i++)
	{
	rem=rem*10;
	rem=rem+(b[i]-'0');
	rem=rem%a;
	}
	}
	if(flag==0)
	{
	ans=gcd(rem,a);
	}
	if(flag!=2)
	{
	printf("%d\n",ans);
	}
	}
	return 0;
}
