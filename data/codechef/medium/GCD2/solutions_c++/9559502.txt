#include <iostream>
using namespace std;

int gcd(int a, int b)
{
	if (b == 0) return a;
	else return gcd(b, a % b);
}

int main()
{
	int t, a, tmp, i;
	char str[3000];
	cin >> t;

	for (; t; t--) {
		cin >> a >> str;
		i = 0; tmp = 0;
		if (a == 0) {
			cout << str << endl;
		} else {
			while (str[i] != '\0') {
				while ((str[i] != '\0') && (tmp < a)) {
					tmp = tmp * 10 + (str[i++] - '0');
				}
				tmp = tmp % a;
			}
			cout << gcd(a, tmp) << endl;
		}
	}

	return 0;
}
