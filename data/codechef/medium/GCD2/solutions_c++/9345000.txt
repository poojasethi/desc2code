#include <bits/stdc++.h>
#define L long long int
using namespace std;

L gcd(L a, L b) {
	while(b) {
		L temp = a;
		a = b;
		b = temp % b;
	}
	return a;
}

int main() {
	L t, a, len, x, c, new_b; // x will store the new value of b after first modulus is done, c stores char when converted to int
	char b[500];
	scanf("%lld", &t);
	while (t--) {
		scanf("%lld%s", &a, b);
		if (a == 0) { printf("%s\n", b); continue; }
		if (b[0] == 0) { printf("%lld\n", a); continue; }
		// first modulus to convert string to int i.e b % a
		len = strlen(b);
		x = 0;
		for (L i = 0; i < len; i++) {
			c = b[i] - '0'; // char to int conversion
			x = (x*10 + c) % a;
		}
		new_b = x; // a = a % b
		printf("%lld\n", gcd(a, new_b));
	}
	return 0;
}
