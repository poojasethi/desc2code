import sys
import math

def main():
	t = int(sys.stdin.readline())

	for i in range(0, t):
		n = int(sys.stdin.readline())

		sqrtN = int(math.sqrt(n))

		minDiff = 100000001

		for j in range(1, sqrtN + 1):
			if (n % j == 0):
				diff = abs(j- n / j)
				if (diff <= minDiff):
					minDiff = diff
		print minDiff			
main()
