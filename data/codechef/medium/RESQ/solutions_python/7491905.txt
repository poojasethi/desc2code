T = int(raw_input())
while(T>0):
    T=T-1
    N = int(raw_input())
    mindiff = None
    j = int(N**0.5)
    for i in range(j, 0, -1):
        if(N%i==0):
            j = N/i
            diff = abs(i-j)
            if(mindiff!=None):
                if(diff<mindiff):
                    mindiff=diff
            else:
                mindiff = diff


    print mindiff
