#include <stdio.h>
#include <math.h>
int read()
{
    int x=0;
    char c;
    while(c=getchar_unlocked())
    {
        if(c==' '||c=='\n') break;
        x=x*10+c-48;
    }
    return x;
}
int main( int argc, char *argv[])
{
    int n=0,i;
    n=read();
    while(n--)
    {
        int sq=read();
        for (i = (int) sqrt(sq) ; i > 0 ; i--)
        if(sq%i==0)
        {
             printf("%i\n",sq/i - i);
             break;
        }
    }
    return 0;
}
