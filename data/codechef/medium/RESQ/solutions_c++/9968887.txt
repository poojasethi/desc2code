#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    long long int N;

    cin>>t;

    while(t--)
    {
        cin>>N;

        int mini=INT_MAX;
        for(int i=1;i<=sqrt(N);i++)
        {

            if(N%i==0)
            {
                int p= N/i;
                if(p-i<mini)
                {
                    mini=p-i;
                }
            }

        }
        cout<<mini<<endl;
    }
    return 0;
}
