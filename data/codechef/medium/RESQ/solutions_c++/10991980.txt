#include<stdio.h>
#include<math.h>
#define ll long long int

ll isprime(ll n){
	ll i;
	for(i = 2; i * i <= n; i++)	if(!(n % i))	return 0;
	return 1;
}

int main(){
	ll t, n, i, m, j; scanf("%lld", &t);
	while(t-- && scanf("%lld", &n)){
		if(isprime(n))	printf("%lld\n", n - 1);
			else{
				i = sqrt(n);
				if(n % i){
					i--;
					while(n % i) i--;
				}
				printf("%lld\n", (n / i) - i);
			}
	}
	return 0;
}
