#include <iostream>
#include <climits>
#include <cstdlib>
using namespace std;

int main() {
	int t,n,diff,i,a,b,temp;
	scanf("%d",&t);
	while(t--)
	{
		diff=INT_MAX;
		temp=0;
		scanf("%d",&n);
		for(i=1;i*i<=n;i++)
		{
			if(n%i==0)
			{
				a=i;
				b=n/i;
				temp=abs(a-b);
				if(temp<diff)
				{
					diff=temp;
				}
			}
		}
		printf("%d\n",diff);
	}
	return 0;
}
