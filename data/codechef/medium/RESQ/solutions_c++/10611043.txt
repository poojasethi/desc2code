#include<bits/c++io.h>
#include<iostream>
using namespace std;
#include<math.h>
int main()
{
    int t,min,i;
    cin>>t;
    while(t--)
    {
    int n;
    cin>>n;
    for(i=1;i<=sqrt(n);i++)
    {
        if(n%i==0)
        {
        int o=n/i;
        int tw=i;
        //printf("%d %d ",o,tw);
        int diff=o>tw?o-tw:tw-o;
        //printf("%d ",diff);
        if(i==1)
        min=diff;
        if(diff<=min)
        min=diff;
        //printf("%d\n",min);
        }
    }
    cout<<min<<endl;
    }
}
