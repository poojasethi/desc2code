#include <iostream>
#include <cmath>

using namespace std;

int main(int argc, char const *argv[])
{
    short t;
    long n, l, b, diff, currDiff, limit;

    cin>>t;

    while(t--)
    {
        cin>>n;

        diff = n;
        limit = (long)sqrt((double)n);
        for(l = 1; l <= limit; ++l)
        {
            if(n%l == 0)
            {
                b = n/l;

                currDiff = (long)abs((double)(l-b));
                if(currDiff < diff)
                    diff = abs((double)(l-b));
            }
        }

        cout<<diff<<"\n";
    }
    return 0;
}
