#include<stdio.h>
int min(int a,int b)
{
	if(a<b)
		return a;
	else if(a==b)
		return b;
	else
		return b;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,min1 = 100000000,temp;
		scanf("%d",&n);
		for(i=1;i*i<=n;i++)
		{
			if(n%i==0)
			{
				temp = (n/i-i);
				if(temp<0)
					temp = -temp;
				min1 = min(temp,min1);
			}
		}
		printf("%d\n",min1);
	}
	return 0;
}
