#include <bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		int n,i;
		scanf("%d", &n);
		int ans= INT_MAX, diff = INT_MAX;
		for (i=1; i<=sqrt(n); i++){
			if(n%i == 0){
				diff = (n/i) - i;
			}
			if(diff < ans){
				ans = diff;
			}
		}
		printf("%d\n", ans );
	}
	return 0;
}
