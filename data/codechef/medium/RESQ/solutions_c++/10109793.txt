#include <iostream>
#include <stdio.h>
#include <stdlib.h>
 
using namespace std;
 
int main() {
    int x;
    cin >> x;
    for (int j=0;j<x;j++)
    {
        int N;
        cin >> N;
        int mini=N-1;;
        int b;
        for (int l=1;l*l<=N;l++)
            if (N%l==0)
        {
            b=N/l;
            if (b-l<=mini)
            {
                mini=b-l;
            }
        }
        cout << mini << "\n";
    }
 
	return 0;
}
