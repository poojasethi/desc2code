#include<iostream>
#include<cmath>
#include<cstdio>

using namespace std;

int main()
{
    int t;

    scanf("%d",&t);

    while(t--)
    {
        int n,i;

        scanf("%d",&n);

        for(i=(int)sqrt(n);i>=1;i--)
        {
            if(n%i==0)
            {
                printf("%d\n",n/i-i);

                break;

            }
        }
    }

    return(0);
}
