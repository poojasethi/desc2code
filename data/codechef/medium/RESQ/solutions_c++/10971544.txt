#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{
    int T,N,i,j,min,temp;
    cin>>T;
    while(T--)
    {
        cin>>N;
        min=N;
        for(i=1;i<=sqrt(N);i++)
        {
            
            if(N % i==0)
            {
                j=N/i;
                temp=abs(j-i); 
                if(min>temp)
                min=temp;
                
            }
        }
        cout<<min<<endl;
    }
    return 0;
}
