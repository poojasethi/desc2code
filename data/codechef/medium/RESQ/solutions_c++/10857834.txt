#include <iostream>
#include <vector>
#include "math.h"
using namespace std;

int main() {
	// your code goes here
	int T;
	cin>>T;
	while(T--)
	{
	    int N;
	    cin>>N;
	    int sq = (int) sqrt(N);
	    int mindiff = N-1;
	    for(int i=sq;i>=1;i--)
	    {   
	        if(N%i==0)
	        {
	        if(N/i - i < mindiff)
	        mindiff = N/i -i;
	        }
	    }
	    cout<<mindiff<<endl;
	    
	}
	return 0;
}
