#include <iostream>
#include<math.h>
using namespace std;

int main()
{
    int i,a,b,min,t;
    cin>>t;
    long long int N;
    while(t--)
    {
    cin>>N;
    min=N;
    for(i=1;i<=sqrt(N);i++)
    {
        if(N%i!=0)
            continue;
        else
        {
            a=N/i;
            //cout<<a<<" "<<i<<endl;
            if(a>i)
            {
                b=a-i;
                if(min>b)
                min=b;
            }
            else
            {
                b=i-a;
                if(min>b)
                min=b;
            }
        }

    }
    cout<<min<<endl;
    }
    return 0;
}
