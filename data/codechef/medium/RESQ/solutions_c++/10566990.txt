#include<stdio.h>
#include<string>
#include<limits.h>
#include<math.h>
#define S(n) scanf("%li",&n);
#include<iostream>
#define s(n) scanf("%d",&n);
#define p(n) printf("%d\n",n);
using namespace std;
int main()
{
int t;
s(t)
while(t--)
{
    int min=INT_MAX;
    long int n;
    S(n)
    for(int i=1;i*i<=n;i++)
    {
        if(n%i==0)
        {
         int k=n/i;
         if(k-i<min)
            min=k-i;
        }
    }
    p(min);
}
return 0;
}
