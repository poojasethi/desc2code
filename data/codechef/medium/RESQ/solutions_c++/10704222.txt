#include <iostream>
#include<stdlib.h>
using namespace std;
int main() 
{
	int t,n,min,i,d;
	cin>>t;
	while(t--)
	{
        cin>>n;
        min=n;
        for(i=1;i<=n/i;i++)
        {
            if(n%i==0)
            {
                d=abs((n/i)-i);
                if(d<min)
                {
                    min=d;
                }
            }
        }
        cout<<min<<"\n";
	}
	return 0;
}
