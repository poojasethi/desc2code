#include <cmath>
#include <cstring>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main() 
{
   int t,n,diff,min,i;
    cin>>t;
     while(t--)
         {
         cin>>n;
         diff=0;
         min=0;
         for(i=1;i<=sqrt(n);i++)
             {
             if(n%i==0)
                 {diff=(n/i)-i;}
             min=diff;
             if(min>=diff)
                 {min=diff;}
             }
         cout<<min<<endl;
     }
   return 0;
}
