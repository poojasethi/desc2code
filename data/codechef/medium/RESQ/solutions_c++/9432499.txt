#include <iostream>
#include <map>
#include <numeric>
#include <algorithm>
#include<limits.h>
#include <math.h>
using namespace std;


int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int small = INT_MAX;
        int x,y;

        for(int i=1;i<=sqrt(n);i++)
        {
            if(n%i==0)
            {
                x = i;
                y = n/i;
                small = min(small,y-x);
            }

        }
        cout<<small<<endl;
    }
}
