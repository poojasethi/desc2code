#include<cstdio>
#include<cmath>
using namespace std;

int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int n;
        scanf("%d",&n);
        int ans=n-1;
        for(int i=(int)sqrt(n);i>=1;i--){
            if(n%i==0){
                ans=(n/i)-(i);
                break;
            }
        }
        printf("%d\n",ans);
    }
    
}
