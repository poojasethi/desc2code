#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
int main(){

    int t,w,n,i,k,g,q;

    cin>>t;

    while(t--){

        cin>>n;

        for(i=sqrt(n);i>=1;--i){
            if(n%i==0){
                w=n/i-i;
                break;
            }
        }
        cout<<w<<"\n";

    }

    return 0;
}
