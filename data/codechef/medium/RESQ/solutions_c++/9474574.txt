#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstdio>
#define ll long long int

using namespace std;

int main()
{
    int t;

    scanf("%d",&t);

    while(t--)
    {
        int n,min;

        scanf("%d",&n);

        for(int i=(int)sqrt(n);i>=1;i--)
        {
            if(n%i==0)
            {
                int a=n/i;

                min=abs((double)a-(double)i);

                break;

            }
        }

        printf("%d\n",min);
    }

    return(0);
}
