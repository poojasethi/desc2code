#include <iostream>
#include <map>
#include <string>
using namespace std;

int main(){
	int n, q;
	cin>>n>>q;
	map<string, string> types;
	while(n--){
		string s1, s2;
		cin>>s1>>s2;
		types[s1]=s2;
	}
	while(q--){
		string fstr;
		cin>>fstr;
		int pos=fstr.rfind('.');
		if(pos==string::npos){
			cout<<"unknown"<<endl;
		}else{
			string ext=fstr.substr(pos+1);
			if(types.count(ext))
				cout<<types[ext]<<endl;
			else
				cout<<"unknown"<<endl;
		}
	}
	return 0;
}
