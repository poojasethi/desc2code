using namespace std;
typedef long long int ll;

map<string,string> extentions;
char str[102], str1[102];

int main(){
    int n, q, i;
    scanf("%d%d", &n, &q);
    for( i=0; i<n; i++){
        scanf("%s%s", str, str1);
        extentions[str] = str1;
    }
    int j,len;
    for( i=0; i<q; i++){
        scanf("%s", str);
        len = strlen(str);
        for( j=len-1; j>=0 && str[j] != '.'; j--);
        if( j < 0 )
            printf("unknown\n");
        else if( extentions.find(str+j+1) != extentions.end())
            printf("%s\n", extentions[str+j+1].c_str());
        else
            printf("unknown\n");
    }
    return 0;
}
