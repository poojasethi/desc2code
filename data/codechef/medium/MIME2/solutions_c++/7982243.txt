#include<iostream>
#include<string.h>
using namespace std;
int main ()
{
    int n,q;
    int i,j,k,l,m;
    int flag;
    cin>>n>>q;
    char a[n][11],b[n][51];
    char c[51];
    char d[51];
    for(i=0;i<n;i++)
    {
        cin>>a[i]>>b[i];
    }
    while(q--)
    {
        cin>>c;
        l=strlen(c);
        flag=0;
        for(i=l-1;i>=0;i--)
        {
            if(c[i]=='.')
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
        {
            i++;
            j=0;
            while(c[i]!='\0')
            {
                d[j]=c[i];
                j++;
                i++;
            }
            d[j]='\0';
            for(i=0;i<n;i++)
            {
                if(strcmp(d,a[i])==0)
                {
                    cout<<b[i]<<endl;
                    break;
                }
            }
            if(i==n)
            {
                cout<<"unknown\n";
            }
        }
        else
        {
            cout<<"unknown\n";
        }
    }
}
