#include<bits/stdc++.h>
using namespace std;
map<string,string> m;
int main(){
    int n,q,p;
    string k,s;
    cin>>n>>q;
    while(n--){
            s.clear();k.clear();
            cin>>s>>k;
    m[s]=k;
    }
    while(q--){
            s.clear();k.clear();
            cin>>s;
            p=s.find_last_of('.');
            if(p>=s.length()){
                    cout<<"unknown"<<endl;
            }
    else{
        k=s.assign(s.begin()+p+1,s.end());
    if(m.count(k)==0){
                    cout<<"unknown"<<endl;
            }else{
        cout<<m[k]<<endl;}
    }
    }
    return 0;
}
