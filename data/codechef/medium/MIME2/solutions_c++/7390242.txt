map < string , string > mp;

int main()
{
	int n, q;
	string a, b;
	scanf( "%d%d", &n, &q );
	for( int i = 0 ; i < n ; i ++ )
		cin >> a >> b, mp[a] = b;
	for( int i = 0 ; i < q ; i ++ )
	{
		cin >> a;
		int p = a.rfind( "." );
		if( p == a.npos || !mp.count( a.substr( p+1 ) ) )
			puts( "unknown" );
		else
			cout << mp[ a.substr( p+1 ) ] << "\n";
	}
	return 0;
}
