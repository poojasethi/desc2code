#include<stdio.h>
#include<string.h>
int n,m,i,l,j,k,x;
char e[100][10],t[100][50],s[50];
int main()
{
    scanf("%d %d",&n,&m);
    for(i=0;i<n;i++)
        scanf("%s %s",e[i],t[i]);
    for(x=0;x<m;x++)
    {
       char p[50]={};
       scanf("%s",s);
       l=strlen(s);
       j=l-1;
       while(s[j]!='.'&&j>=0)
        j--;
       if(j>=0)
       {
        j=j+1;
        k=0;
        for(i=j;i<l;i++)
        {
         p[k]=s[i];
         k++;
        }
        int f=0;
        for(i=0;i<n;i++)
        {
            if(strcmp(p,e[i])==0)
            {
                printf("%s\n",t[i]);
                f=1;
                break;
            }
        }
        if(f==0)
        printf("unknown\n");
       }
       else
        printf("unknown\n");
    }
    return 0;
}
