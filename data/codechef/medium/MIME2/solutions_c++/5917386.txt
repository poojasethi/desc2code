#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <utility>
#include <string>
#include <vector>
#include <map>

using namespace std;

int main()
{
    int n,q;
    map<string,string> mediatypes;
    scanf("%d%d",&n,&q);
    string s1,s2;
    for(int i=0;i<n;i++)
    {
        cin >> s1 >> s2;
        mediatypes[s1]=s2;
    }
    string files[q];
    for(int i=0;i<q;i++)
    {
     cin >> files[i];
    }
    for(int i=0;i<q;i++)
    {
        int index = files[i].find_last_of(".");
        if(index==string::npos)
        {
            printf("unknown\n");
        }
        else{
            string ext = files[i].substr(index+1);
            if(!mediatypes.count(ext))
               printf("unknown\n");
            else
              cout << mediatypes[ext] << endl;
        }
    }
    return 0;
}
