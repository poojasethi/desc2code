#include <iostream>
#include <map>
#include <cstring>
 
using namespace std;
 
int main()
{
    int n, q, i, j;
 
    cin >> n >> q;
 
    string s1, s2;
    map <string, string> m;
 
    for (i = 0; i < n; i++) {
        cin >> s1 >> s2;
 
        m[s1] = s2;
    }
 
    for (j = 0; j < q; j++) {
        char str[55];
 
        cin >> str;
 
        s1 = "";
        for (i = strlen(str); i >= 0; i--) {
            if (str[i] == '.') {
                s1 = &str[i+1];
                break;
            }
        }
 
        if (m.count(s1) > 0)
            cout << m[s1] << endl;
 
        else
            cout << "unknown\n";
    }
 
    return 0;
}
