#include <iostream>
#include <string>
#include <map>
using namespace std;

int main() {
	int n, q;
	int pos;
	
	cin >> n >> q;
	string a, b, s;
	map<string, string> relation;
	
	while (n--) {
		cin >> a >> b;
		relation[a] = b;
	}
	
	while (q--) {
		cin >> s;
		pos = s.rfind("."); 
		if (pos == string::npos) {
			cout << "unknown" << endl;
			continue;
		}
		s = s.substr(pos + 1);
		
		if (relation.find(s) != relation.end()) {
			cout << relation[s] << endl;
		} else {
			cout << "unknown\n";
		}
	}
//	map<string, string>::iterator it = relation.begin();
//	for (; it != relation.end(); it++) {
//		cout << it -> first << " " << it -> second << endl;
//	}
	
	return 0;
}
