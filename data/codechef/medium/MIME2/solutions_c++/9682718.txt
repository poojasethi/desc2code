    #include<iostream>
    #include<map>
    using namespace std;
    int main()
    {
    int n,m1,i,j;
    string s1,s2;
    map <string,string> m;
    cin>>n>>m1;
    for(i=0;i<n;i++)
    {
    cin>>s1>>s2;
    m.insert(make_pair(s1,s2));
    }
    for(i=0;i<m1;i++)
    {
    string str;
    cin>>str;
    int pos=str.rfind('.');
    if(pos==string::npos)
    {
    cout<<"unknown"<<endl;
    }
    else{
    string str1=str.substr(pos+1);
    if(m.find(str1)==m.end())
    {
    cout<<"unknown"<<endl;
    }
    else{
    cout<<m[str1]<<endl;
    }
    }
     
    }
    }
