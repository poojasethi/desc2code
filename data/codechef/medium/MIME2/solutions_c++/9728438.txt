#include<bits/stdc++.h>
using namespace std;
int main(void)
{
    int n, q;
    scanf("%d %d", &n, &q);
    map<string, string> type;
    for(int i=0; i<n; i++){
        char a[101], b[101];
        scanf("%s %s", a, b);
        string aa, bb;
        for(int j=0; a[j]!='\0'; j++)
            aa+=a[j];
        for(int j=0; b[j]!='\0'; j++)
            bb+=b[j];
        type[aa]=bb;
    }
    for(int i=0; i<q; i++){
        char p[101];
        scanf("%s", p);
        string ques;
        int j=strlen(p)-1;
        while(p[j]!='.'){
            j--;
        }
        if(p<0){
            printf("unknown\n");
            continue;
        }
        j++;
        for(; p[j]!='\0'; j++)
            ques+=p[j];
        if(type.find(ques)==type.end())
            printf("unknown\n");
        else
            printf("%s\n", type[ques].c_str());
    }
    return 0;
}
