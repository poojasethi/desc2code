#include<bits/stdc++.h>
using namespace std;
map<string,string> ::iterator it;
map<string,string> data;
int main()
{   int a,b;
string temp, str1,str2,extension;
    cin>>a>>b;
    while(a--){
        cin>>str1>>str2;
        data[str1]=str2;
    }
    while(b--){
        cin>>extension;
       size_t found=extension.rfind(".");
        if(found!=string::npos)
        {
        temp=extension.substr(found+1);
            it=data.find(temp);
        if(it!=data.end())
            cout<<it->second<<endl;
        else
            cout<<"unknown\n";
        }
        else
            cout<<"unknown\n";
    }
return 0;
}
