#include<iostream>
#include<map>
using namespace std;

int main()
{
    map<string,string>fileType;

    int n,q;
    cin>>n>>q;

    string a,b;             //entring fileext & type
    for(int i=0;i<n;i++)
        {
            cin>>a>>b;
            fileType[a] = b;
        }

        while(q--)
        {
            string f , e;  //fileName == f //e == after dot extension
            cin>>f;
            int p = f.rfind('.');
            if(p==string::npos)    //no position in string ie file name without dot
                cout<<"unknown"<<endl;
            else
            {
                e = f.substr(p+1);
                if(fileType.count(e))
                    cout<<fileType[e]<<endl;
                else
                    cout<<"unknown"<<endl;
            }
        }
}
