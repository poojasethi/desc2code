#include<iostream>
#include<stdio.h>
#include<string.h>

using namespace std;

int main() {
	int N, Q;
	cin>>N>>Q;
	char ext[N][20], type[N][60], file[60];
	for(int i=0; i<N; ++i)
		scanf("%s %s", ext[i], type[i]);
	for(int i=0; i<Q; ++i) {
		scanf("%s", file);
		int pos=0, j=0, I=0, f=0;
		while(file[j]!='\0') {
			if(file[j]=='.')
				pos=j;
			++j;
		}
		++pos;
		for(j=0; j<N; ++j) {
			if(!strcmp(&file[pos], ext[j])) {
				printf("%s\n", type[j]);
				f=1;
				break;
			}
		}	
		if(f==0)
			printf("unknown\n");		
	}
}
