#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int i, j,n, q,k, len;
    scanf("%d %d",&n,&q);
    char ext[n][11], type[n][51], ques[51];
    for(i=0; i<n; i++)
    {
        cin>>ext[i]>>type[i];
    }
    while(q--)
    {
        cin>>ques;
        int flag=0;
        len=strlen(ques);
        char comp[len];
        for(i=(len-1); i>=0; i--)
        {
            if(ques[i]=='.')
            {
                for(j=i+1, k=0; j<len; j++, k++)
                {
                    comp[k]=ques[j];
                }
				comp[k]='\0';
                break;
            }
        }
        for(i=0; i<n; i++)
        {
            if(strcmp(ext[i],comp)==0)
            {
                flag=1;
                break;
            }
        }
        if(flag==1)
            printf("%s\n", type[i]);
        else printf("unknown\n");
    }

    return 0;
}
