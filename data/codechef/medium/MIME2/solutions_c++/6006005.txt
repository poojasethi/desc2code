#include <iostream>
#include <vector>
#include <string>
#include <map>
using namespace std;

int main()
{
	int n,q;
	cin>>n>>q;
	map<string,string> smap;
	for(int i=0;i<n;++i)
	{
		string s1,s2;
		cin>>s1>>s2;
		smap[s1]=s2;
	}
	for(int i=0;i<q;++i)
	{
		string s1,s2;
		cin>>s1;
		int a=s1.find_last_of(".");
		if(a>-1)
		{
			s2=s1.substr(a+1);
			if(smap[s2]!=""){
			cout<<smap[s2]<<endl;}
			else
			{
				cout<<"unknown"<<endl;
			}
		}
		else
		{
			cout<<"unknown"<<endl;
		}
	}
	return 0;
}
