#include<iostream>
#include<map>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<string>
using namespace std;
int main()
{
	int n,q,l,index;
	cin>>n>>q;
	map <string,string > ht;
	string s1,s2,s3,s4;
	for(int i=0;i<n;i++)
	{
		cin>>s1>>s2;
		ht[s1]=s2;
	}
	for(int i=0;i<q;i++)
	{
		cin>>s3;
		l=s3.length();
		s4="";
		int count=0;
		for(int j=0;j<l;j++)
		if(s3[j]=='.')
		{
		index=j;
		count++;
		}
		if(count>0)
		{
		for(int j=index+1;j<l;j++)
		s4+=s3[j];
		}
		if(ht.find(s4)==ht.end())
		cout<<"unknown\n";
		else
		cout<<ht[s4]<<"\n";
	}
}
