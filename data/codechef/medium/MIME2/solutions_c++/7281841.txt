#include <iostream>
#include <map>
#include <string>
using namespace std;

int main(int argc, char const *argv[])
{
	string e, t;
	map<string, string> m;
	int n, q;

	cin >> n >> q;
	for(int i = 0; i < n; i++) {
		cin >> e >> t;
		m[e] = t;
	}

	string s, r;
	for(int i = 0; i < q; i++) {
		cin >> s;
		r = s.substr(s.find_last_of(".") + 1);
		if(s.find(".") != string::npos && m.find(r) != m.end()) {
			cout << m.at(r) << endl;
		} else {
			cout << "unknown" << endl;
		}
	}
	return 0;
}
