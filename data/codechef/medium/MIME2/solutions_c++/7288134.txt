#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
int main()
{
    int n,q,i;
    cin>>n>>q;
    char f_type[n][51],m_type[n][51],t_type[51],*temp=NULL;
    for(i=0;i<n;i++)
    {
        cin>>f_type[i]>>m_type[i];
    }
    while(q--)
    {
        cin>>t_type;
        for(i=0;t_type[i]!='\0';i++)
        {
            if(t_type[i]=='.')
            {
                temp=&t_type[i+1];
            }
        }
        if(temp==NULL||*temp=='\0')
        {
            cout<<"unknown\n";

        }

        else
        {  //cout<<t_type<<"\n";
        for(i=0;i<n;i++)
        {
            if(strcmp(temp,f_type[i])==0)
            {
                cout<<m_type[i]<<"\n";
                break;
            }
        }
        if(i==n)
            cout<<"unknown\n";
        }
    }
    return 0;
}
