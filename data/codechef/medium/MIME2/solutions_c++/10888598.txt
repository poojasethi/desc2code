#include<cstdio>
#include<map>
#include<iostream>
using namespace std;
map<string,string> m;
int main()
{
	int q,n,i,j;
	string s,a,b,temp,t1;
	cin>>n>>q;
	while(n--)
	{
		cin>>a>>b;
		m[a]=b;
	}
	for(i=0;i<q;i++)
	{
		cin>>s;
		t1="";
		temp="";
		int f=0;
		for(j=s.size()-1;j>=0;j--)
		{
			if(s[j]=='.')
			{
				f=1;
				break;
			}
			t1+=s[j];
		}
		if(f==0)
		t1="";
		for(j=0;j<t1.size();j++)
		{
			temp+=(t1[t1.size()-j-1]);
		}
		if(m.find(temp)!=m.end())
		{
			cout<<m[temp]<<endl;
		}
		else
		{
			cout<<"unknown"<<endl;
		}
	}
	return 0;
}
