#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,q,i,len,j,k,z;
    map<string,string>exe;
    string s1,s2,s,sub;
    char temp;
    cin>>n>>q;
    while(n--)
    {
        cin>>s1>>s2;
        exe[s1]=s2;
    }
    while(q--)
    {
        char str[50]="\0";
        cin>>s;
        i=s.length()-1;
        k=0;
        while(i>=0&&s[i]!='.')
        {
            str[k++]=s[i];
            i--;
        }
        if(s[i]!='.')
        {
            cout<<"unknown"<<endl;
        }
        else if(s[s.length()-1]=='.')
        {
            cout<<"unknown"<<endl;
        }
        else
        {
            for(j=0;j<k/2;j++)
            {
                temp=str[j];
                str[j]=str[k-1-j];
                str[k-1-j]=temp;
            }
            sub=string(str);
            if(exe[sub]=="\0")
            {
                cout<<"unknown"<<endl;
            }
            else
            {
                cout<<exe[sub]<<endl;
            }
        }
    }
    return 0;
}
