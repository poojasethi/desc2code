#include <iostream>
#include <map>
#include <string>
using namespace std;
typedef long long ll;
ll find_last(string a,char c)
{
	ll i;
	for (i=a.length()-1;i>=0;i--)
	{
		if(a[i]==c)
			break;
	}
	return i;
}
	
int main()
{
	ll n,q,i,pos,posw;
	string ext,type,fname;
	map<string,string> extype;
	scanf("%lld %lld",&n,&q);
	for(i=0;i<n;i++)
	{
		cin>>ext;
		cin>>type;
		extype[ext]=type;
	}	
	//getline(cin,fname);
	for(i=0;i<q;i++)
	{
		cin>>fname;
		//getline(cin,fname);
		//posw=fname.find_first_of(' ');
		//fname=fname.substr(0,posw);
		//cout<<fname<<endl;
		pos=find_last(fname,'.');
		//cout<<pos<<endl;
		if(pos==-1)
		{
			printf("unknown\n");
			continue;
		}
		ext=fname.substr(pos+1,fname.length()-pos-1);
		if(extype.count(ext)==0)
		{
			printf("unknown\n");
			continue;	
		}
		else
		{
			cout<<extype[ext]<<endl;
		}
	}
	return 0;
}
