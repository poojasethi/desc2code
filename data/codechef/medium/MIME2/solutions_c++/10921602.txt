#include <iostream>
#include<map>
#include<string>
#include<cstdio>
using namespace std;
map<string,string>m;

int main()
{int n,q,i,j,l,l2,flag;
 string a,b,c,s,copy; 
 cin>>n>>q;
 
 for(i=0;i<n;i++)
 {cin>>a>>b;
  m[a]=b;
  
 }
 for(i=0;i<q;i++)
 {cin>>s;
  l=s.size();
  flag=0;
  c="";
  copy="";
  for(j=l-1;j>=0;j--)
  {if(s[j]=='.')
   {flag=1;
    break;
   }
   c+=s[j];
  }
  if(flag==0)
  {c="";
  }
    
    l2=c.size();
    for(j=0;j<l2;j++)
    { copy+=c[l2-j-1];
    }
    
    
    
    if(m.find(copy)!=m.end())
    { cout<<m[copy]<<endl;
    }
    else
    {cout<<"unknown"<<endl;
    }
  }
 return 0;
}
