#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{	
	map<string,string> m;
	ll n,q,i; string k,t,s;
	cin>>n>>q;
	
	while(n--)
	{
		cin>>k>>t;
		m[k]=t;
	}
	
	while(q--)
	{
		cin>>s;
		string tmp="";
		for(i=s.size()-1;i>=0;i--)
		{
			if(s[i]=='.')
			break;
			else
			tmp=(char)s[i] + tmp;
		}
		if(i<0)
		cout<<"unknown\n";
		else if(m[tmp]=="")
		cout<<"unknown\n";
		else
		cout<<m[tmp]<<endl;
	}
	
	return 0;
}
