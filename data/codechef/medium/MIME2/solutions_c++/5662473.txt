#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{   char ext[110][60],type[110][60],name[110][60],file[60];
    int len,k,i,n,q,j,flag,x;
    cin>>n>>q;
    for(i=0;i<n;i++)
    {  cin>>ext[i]>>type[i]; }
    for(i=0;i<q;i++)
    {  cin>>name[i];  }
    for(i=0;i<q;i++) 
    {  len=strlen(name[i]);
       len--;
       flag=0;
       x=0;
       while((name[i][len]!='.')&&len>=0)
       {  len--;   }
       len++;
       if(len!=0)
       { for(k=0;name[i][len]!='\0';k++)
         {  file[k]=name[i][len];
            len++;
         }
         file[k++]='\0';
         for(j=0;j<n;j++)
         {  if(!strcmp(ext[j],file))
            {  flag=1;
               break;   
            }
         }
       }  
       if(flag==1&&len!=0)
       {  cout<<type[j]<<endl; }
       else
       {  cout<<"unknown"<<endl; }
   }         
    return 0;
}
