import sys
testCases = sys.stdin.readline().strip().split()
mediaTypes = int(testCases[0])
fileNames = int(testCases[1])
mediaTable = {}
for i in range(mediaTypes):
	mediaData = sys.stdin.readline().strip().split()
	mediaTable[mediaData[0]] = mediaData[1]
for i in range(fileNames):
	try:
		extension = sys.stdin.readline().strip().rsplit(".", 1)[1]
		if extension in mediaTable.keys():
			print mediaTable[extension]
		else:
			print "unknown"
	except IndexError: 
		print "unknown"
