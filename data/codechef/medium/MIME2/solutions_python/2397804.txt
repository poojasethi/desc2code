from sys import stdin

N,Q = map(int,stdin.readline().split())
d = {}
for i in xrange(N):
    extension, mediaType = stdin.readline().strip().split()
    d[extension] = mediaType
for i in xrange(Q):
    line = stdin.readline().strip()
    loc = line.rfind(".")
    type = "unknown"
    if loc > -1:
        key = line[loc+1:]
        if key in d:
            type = d[key]
    print type
