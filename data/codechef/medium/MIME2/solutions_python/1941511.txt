tip = raw_input().split(' ')
N = int(tip[0])
Q = int(tip[1])
media = {}
for i in range(N):
  ip = raw_input().split(' ')
  media[ip[0]] = ip[1]
  
for i in range(Q):
  temp = raw_input()
  temp = temp.split()
  temp = temp[0]
  if temp.find('.') != -1:
    temp1 = temp.split('.')
    key = temp1[len(temp1)-1]
    if key in media:
      print media[key]
    else:
       print "unknown"
  else:
    print "unknown"
