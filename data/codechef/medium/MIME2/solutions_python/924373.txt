#!/usr/bin/env python
import sys
h = {}

if __name__ == '__main__':
    (N,Q) = map(lambda x: int(x), sys.stdin.readline().split())
    while N:
        (ext, mime) = sys.stdin.readline().strip().split()
        h[ext] = mime
        N-=1
    while Q:
        name = sys.stdin.readline().strip()
        try:
            print h[name[name.rindex('.')+1:]]
        except:
            print 'unknown'
        Q-=1
