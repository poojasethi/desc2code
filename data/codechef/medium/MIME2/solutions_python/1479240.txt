inp = raw_input()
inp = inp.split()
n = int(inp[0])
q = int(inp[1])
lookups = {}
for i in range(0, n):
	x = raw_input()
	x = (x.strip()).split()
	lookups[x[0]] = x[1]
keyset = lookups.keys()
for i in range(0, q):
	f = raw_input()
	f = f.strip()
	idx = f.rfind('.')
	if idx==-1:
		print "unknown"
	else:
		ext = f[idx+1:]
		if ext in keyset:
			print lookups[ext]
		else:
			print "unknown"
