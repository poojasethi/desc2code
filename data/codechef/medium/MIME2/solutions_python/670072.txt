import sys

lines = sys.stdin.readlines()

n, q = map(int, lines.pop(0).split())

extTable = {}
for i in range(n):
    ext, media = lines.pop(0).split()
    extTable[ext] = media

#print extTable

for i in range(q):
    filename = lines.pop(0).strip().split(".")
    #print filename
    if len(filename)==1:
        print "unknown"
    else:
        ext = filename[-1]
        if ext in extTable.keys():
            print extTable[ext]
        else:
            print"unknown"
