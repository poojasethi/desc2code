#include<iostream>
using namespace std;

long long gcd(long long a,long long b)
{      long long comp=a%b;

       if(comp==0 )return b;
       else return gcd(b,comp);
}


int main()
{

    long long t,n,i,min,max,k;
    long long sum=0,g;
    cin>>t;

    while(t--)
    {  sum=0;
       cin>>n;
       for(k=1;k<=n;)
           {
               max=n/k;
               min=n/(k+1);

               sum += k * (max-min);
               if(min>0)
               k=n/min;
               else
               ++k;



           }
      //cout<<sum<<endl;
      g=gcd(n*n,sum);
       cout<<sum/g<<"/"<<(n*n)/g<<endl;

    }
}
