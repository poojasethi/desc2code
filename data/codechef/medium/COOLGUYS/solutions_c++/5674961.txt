#include<iostream>
using namespace std;
#include<stdio.h>
#include<math.h>
 long long gcd(long long a,long long b)
    {
    if(b==0)
    return a;
    else return gcd(b,a%b);
    }
int main()
{
int t,i;
long long ans=0,gc,d,n,x;
scanf("%d",&t);
while(t--)
{
ans=0;
scanf("%lld",&n);
x=sqrt(n);
for(i=1;i<=x;i++)
    ans=ans+n/i;
ans=ans<<1;
d=n*n;
ans=ans-(x*x);
gc=gcd(ans,d);
printf("%lld/%lld\n",ans/gc,d/gc);
}
return 0;
}
