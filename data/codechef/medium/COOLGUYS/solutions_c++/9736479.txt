#include"cstdio"
long long gcd(long long a, long long b) {
    return b==0?a:gcd(b, a%b);
}
int main () {
    int T;
    scanf("%d", &T);
    while(T--) {
        int n;
        scanf("%d", &n);
        long long tot = n*1ll*n, cnt = 0, g;
        for(int i=1; i<=n; ) {
            int val = n/i;
            int nxt = n/val+1; // the first guy with val <n/i
            cnt += (nxt-i)*val;
            i = nxt;
        }
        g = gcd(tot, cnt);
        printf("%lld/%lld\n", cnt/g, tot/g);
    }
}
