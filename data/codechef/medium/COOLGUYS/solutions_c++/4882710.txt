#include<stdio.h>
#include<math.h>
#define ll long long
#define gamma 0.577215

ll gcd(ll a,ll b)
{
	ll c;
	while(a!=0)
	{
		c = a;
		a = b%a;
		b = c;
	}
	return b;
}

int main()
{
	int t;
	ll a,b,n;
	scanf("%d", &t);
	while(t-->0)
	{
		scanf("%lld", &n);
		ll u=(ll)(sqrt(n));
		a=0ll;
		for(int i=1;i<=u;i++)
			a=(ll)a+n/i;
		a=(ll)2*a;
		a=(ll)a-u*u;
		b=(ll)n*n;
		ll g =gcd(a,b);
		printf("%lld/%lld\n", a/g,b/g);
	}
	return 0;
}
