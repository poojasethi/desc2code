#include<cstdio>
#include<cmath>
typedef long long ll;
using namespace std;
ll gcd(ll a,ll b){
	if(b==0)
		return a;
	else
		return gcd(b,a%b);
}
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		ll i,n,sq,ans=0,g;
		scanf("%lld",&n);
		sq=sqrt(n);
		for(i=1;i<=sq;i++){
			if(i<=n/(i+1))
				ans+=n/i;
		}
		for(i=1;i<=sq;i++){
			ans+=i*(n/i-n/(i+1));
		}
		g=gcd(ans,n*n);
		printf("%lld/%lld\n",ans/g,n*n/g);
	}
	return 0;
}
