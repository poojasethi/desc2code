#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL gcd(LL a, LL b){
	if(b == 0) return a;
	return gcd(b, a % b);

}
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		LL N;
		scanf("%lld", &N);
		LL ans = 0;
		LL u = sqrt(N);
		for(int i = 1;i <= u;++i)
			ans += N/i;
		ans = 2 * ans - u * u;
		LL a = ans;
		LL b = N * N;
		LL g = gcd(a, b);
		printf("%lld/%lld\n", a/g, b/g);
	}

	return 0;
}
