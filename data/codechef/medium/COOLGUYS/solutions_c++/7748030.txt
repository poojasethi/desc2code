    #include<stdio.h>
     
    long long gcd(long long a, long long b)
    {
    if(a>b) return gcd(b,a);
    if(a==0) return b;
    else return gcd(b%a,a);
    }
     
    int main()
    {
    int i,n,t;
    long long num,den,ans;
     
    scanf("%d",&t);
    while(t--)
    	{
    	scanf("%d",&n);
    	num=0;
    	for(i=1;i*i<=n;i++)	
    		num+= n/i;
    	num*=2;	
    	num -= (i-1)*(i-1);
    	
    	den=n;
    	den*=n;
    	ans=gcd(num,den);
    	printf("%lld/%lld\n",num/ans,den/ans);	
    	}
     
    return 0;
    }
