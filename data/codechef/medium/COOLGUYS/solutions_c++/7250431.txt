#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<vector>
#define mod 1000000007ll
using namespace std;
long long T,n;
long long gcd(long long a,long long b)
{
	if(b==0)return a;
	return gcd(b,a%b);
}
int main()
{
  	scanf("%lld",&T);
	while(T--)
	{
		scanf("%lld",&n);
		int sq=(int)sqrt(n);
		
		long long cur,res=n,pre=n;
		for(int i=2;i<=sq;i++)
		{
			cur=n/i;
			res+=cur+(pre-cur)*(i-1);
			pre=cur;
		}
		if(n/(sq+1)==sq){
			res+=(pre-sq)*sq;
		}
		n=n*n;
		long long d=gcd(n,res);
		cout<<(res/d)<<"/"<<(n/d)<<"\n";
	}
	return 0;
}
