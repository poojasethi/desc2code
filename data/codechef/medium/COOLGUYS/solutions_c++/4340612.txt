#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>

using namespace std;
typedef long long ll;
#define REP(i,a,b)\
    for(int(i)=int(a);i<=int(b);i++)

ll gcd(ll a,ll b){
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int main(){
    int T;
    ll N,sum,K,val,denom,upVal,downVal;
    scanf("%d",&T);
    while(T--){
        scanf("%lld",&N);
        K = N;
        sum = 0;
        upVal = 1;
        while(upVal <= N){
            downVal = N/K;
            sum += K*(downVal - upVal + 1);
            upVal = downVal + 1;
            K = N/upVal;
        }
        val = gcd(N*N,sum);
        sum = sum/val;
        denom = (N*N)/val;
        printf("%lld/%lld\n",sum,denom);
        
    }
    return 0;
}
