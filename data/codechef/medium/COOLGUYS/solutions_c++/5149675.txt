#include<cstdio>
#include<cmath>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
} 
int main()
{   
	int t,i;
	long long int n,sum,u;
	scanf("%d",&t);
	while(t--)
	{   sum=0;
		scanf("%lld",&n);
		u=sqrt(n);
		for(i=1;i<=u;i++)
		{
			sum=sum+(n/i);
		}
		sum=sum*2-u*u;
		long long int k=n*n;
		long long int m;
		m=gcd(sum,k);
		printf("%lld/%lld\n",sum/m,k/m);
		
	}
	return 0;
}
