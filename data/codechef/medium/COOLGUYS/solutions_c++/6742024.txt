#include<bits/stdc++.h>
#define gc getchar()
using namespace std;
typedef long long int ll;
ll ra()
{
    char c=gc;
    ll r=0;
    while(c<'0' || c>'9')
        c=gc;
    while(c>='0' && c<='9')
    {
        r=(r<<3)+(r<<1)+c-'0';
        c=gc;
    }

    return r;

}

ll gcd(ll a,ll b)
{
    if(b==0)
        return a;

    return gcd(b,a%b);
}


int main()
{
    ll t,n,i,j;
    t=ra();
    while(t--)
    {
        ll sum=0;
        n=ra();
        j=sqrt(n);
        for(i=1;i<=j;i++)
        {
            sum +=n/i;
        }

        sum =(sum*2)-(j*j);
        ll a=gcd(sum,n*n);
        printf("%lld/%lld\n",sum/a,n*n/a);
    }
    return 0;
}
