long long gcd(long long a, long long b){
    if(b == 0) return a;
    else return gcd(b, a%b);
}
int main() {
    int t;
    scanf("%d", &t);
    for(int i = 0; i < t; i++){
        int n;
        scanf("%d", &n);
        long long base = ((long long)n * n);
        long long num = 0;
        int sqr = sqrt(n);
        for(int j = 1; j <= sqr; j++)
                num += (n / j);

        num =num * 2 -  sqr * sqr;
        long long g = gcd(base, num);
        printf("%lld/%lld\n", num / g, base / g);
    }

    return 0;


}
