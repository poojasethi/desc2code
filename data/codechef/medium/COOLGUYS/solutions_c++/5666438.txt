#include<iostream>
using namespace std;

long long gcd(long long a,long long b){
	return b==0?a:gcd(b,a%b);
}

int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		long long total=n*1ll*n,answer=0,g;
		for(int i=1;i<=n;){
			int val=n/i;
			int next=n/val+1;
			answer+=(next-i)*val;
			i=next;
		}
		g=gcd(total,answer);
		cout<<answer/g<<"/"<<total/g<<endl;
	}
	return 0;
}
