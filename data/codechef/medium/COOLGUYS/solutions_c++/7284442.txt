using namespace std;
long long gcd(long long a,long long b){
	return (b==0?a:gcd(b,a%b));
}
int main(){   
    long long T,N,sum,i,root,g;
    scanf("%lld",&T);
    while(T--){
    	sum=0;
    	scanf("%lld",&N);
    	root=sqrt(N);
    	for(i=1;i<=root;i++){
    		sum+=N/i;
    	}  
		sum*=2;
		sum-=root*root;  
    	g=gcd(sum,N*N);
    	printf("%lld/%lld\n",(sum)/g,(N*N)/g);
    }
	return 0;
}
