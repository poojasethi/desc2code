#include<iostream>
#include<cstdio>
#include<cmath>
typedef long long int LL;
using namespace std;
LL gcd(LL a,LL b)
{
    if(b==0)
        return a;
    else
    gcd(b,a%b);
}
int main()
{
    LL t,n;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        LL a,b,k=n;
        LL ans=0;
        LL imin=1,imax;
        while(imin<=n)
        {
          imax=n/k;
          ans+=k*(imax-imin+1);
          imin=imax+1;
          k=n/imin;
        }
        b=n*n;
        a=gcd(ans,b);
        //printf("%d ",a);
        printf("%lld/%lld\n",(ans/a),(b/a));
    }
    return 0;
}
