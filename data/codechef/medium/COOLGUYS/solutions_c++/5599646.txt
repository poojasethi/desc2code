#include<iostream>
#include<cmath>
using namespace std;

long long int gcd(long long int a,long long int b)
{
    long long int r=a%b;
    if(r==0)
        return b;
    else
        return gcd(b,r);
}
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        long long int n;
        cin>>n;
        long long int sum=0;
        int sq=sqrt(n);
        for(int i=1;i*i<=n;i++)
        {
            sum=sum+n/i;
        }
        sum=(2*sum)-(sq*sq);
        long long int g=gcd(sum,n*n);
        cout<<sum/g<<"/"<<n*n/g<<endl;
    }
    return 0;
}
