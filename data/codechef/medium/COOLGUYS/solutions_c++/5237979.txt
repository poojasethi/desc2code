#include<iostream>
#include<math.h>
#include<cstdio>
#define gc getchar_unlocked
using namespace std;

void fastin(long long int &x)
{
	register int a=gc();
	x=0;
	for(;a<48 || a>57;a=gc());
	for(;a>47 && a<58;a=gc())
		x=(x<<1)+(x<<3)+a-48;	
}

long long int gcd(long long int a,long long int b)
{
	if(b==0) 
	return a;
	else return gcd(b,a%b);
}

int main()
{
	long long int n,t,i,j;
	long long int numer,denom,gc_nd,k;
	fastin(t);
	while(t--)
	{
		numer=0;
		fastin(n);
		denom=n*n;
		k=n;
		i=1;
		while(i<=n)
		{
			j=n/k;
			numer+=k*(j-i+1);
			i=j+1;
			k=n/i;
		}
		
		//cout<<numer<<" "<<denom;
		gc_nd=gcd(numer,denom);
		printf("%lld/%lld\n",numer/gc_nd,denom/gc_nd);
	}
	return 0;
}
