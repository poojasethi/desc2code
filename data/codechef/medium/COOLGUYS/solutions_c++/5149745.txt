#include<cstdio>
#include<cmath>
using namespace std;
long long int gcd(long long int a,long long int b)
{
	if(b==0)
	return a;
	return gcd(b,a%b);
}
int main()
{
	long long int t,n,c,i,d,u,sum;
	scanf("%lld",&t);
	while(t--)
	{
		sum=0;
		scanf("%lld",&n);
		u=sqrt(n);
		for(i=1;i<=u;i++)
		{
			sum=sum+(n/i);
		}
		sum=sum*2-u*u;
		d=n*n;
		c=gcd(sum,d);
		sum=sum/c;
		d=d/c;
		printf("%lld/%lld\n",sum,d);
	}
	return 0;
}
