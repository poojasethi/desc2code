#include <cstdio>
#include <iostream>
#include <map>

using namespace std;

long long dp[101][101], a[101];

long long modn = 1e9 + 7;

int main() {
	int n;
	cin >> n;
	for (int i = 1; i <= n; i++) 
		cin >> a[i];

	for (int i = 0; i < 101; i++)
		for (int j = 0; j < 101; j++)
			dp[i][j] = 1;

	for (int i = 1; i <= n; i++) {

		for (int j = 1; j <= i; j++)
			dp[j][i] = dp[j][i-1];

		if (a[i] > 0)
			for (int j = i; j >= 1; j--) 
				if (a[j] + a[i] == 0) 
					for (int k = j; k >= 1; k--)
						dp[k][i] = (dp[k][i] + (dp[k][j-1] * dp[j+1][i]) % modn) % modn;
				
	}

	cout << dp[1][n] << endl;

	return 0;
}
