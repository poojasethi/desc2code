#include <iostream>

#define MOD 1000000007;

using namespace std;

long long int dp[105][105];
long long int a[105];

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> a[i + 1];
		dp[i + 1][i + 1] = 0;
	}
	for (int l = 2; l <= n; ++l)
	{
		for (int i = 1; i <= n - l + 1; ++i)
		{
			int j = i + l - 1;
			if (a[j] <= 0){
				dp[i][j] = dp[i][j - 1] % MOD;
			}
			else {
				int x;
				dp[i][j] = dp[i][j - 1] % MOD;
				for (x = i; x < j; ++x)
				{
					if (a[x] == -a[j]) {
						long long n = dp[i][x - 1] + 1;
                    	long long m = dp[x + 1][j - 1] + 1;
                    	long long int d = n*m;
                    	d = d % MOD;
                    	dp[i][j] = (dp[i][j] + d) % MOD;
					}
				}
			}
		}
	}
	long long int ans = dp[1][n]+1;
	ans = ans % MOD;
	cout << ans << endl;
	return 0;
}
