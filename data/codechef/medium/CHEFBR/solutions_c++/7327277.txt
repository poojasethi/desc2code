#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<vector>
#define mod 1000000007ll
using namespace std;
typedef vector<long long> vll;
int T,n,A[103];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d",&A[i]);
	
	vector<vll> dp(n+1,vll(n+1,0));
	vector<vll> C(n+1,vll(n+1,0));
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j+i<=n;j++)
		{
			dp[j][i+j]=dp[j][i+j-1]+dp[j+1][i+j]-dp[j+1][j+i-1];
			C[j][i+j]=C[j+1][i+j];
			if(A[j]>0||A[i+j]<0)continue;

			if(A[j]==-A[j+i])
			{
				dp[j][i+j]+=dp[j+1][i+j-1]+1;
				C[j][i+j]+=dp[j+1][i+j-1]+1;
			}
	
			for(int k=j+1;k<i+j;k++)
			if(A[j]==-A[k])
			{
				long long x=(1+dp[j+1][k-1])*C[k+1][i+j]%mod;
				dp[j][i+j]+=x;
				C[j][j+i]+=x;
			}
			dp[j][i+j]%=mod;
			C[j][i+j]%=mod;
		}
	}
	/*for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)printf("%2lld ",dp[i][j]);
		cout<<endl;
	}*/
	cout<<(dp[1][n]+1)%mod<<"\n";
	return 0;
}
