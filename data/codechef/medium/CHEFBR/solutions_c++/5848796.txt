#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main()
{
int n;
long long int dp[120][120];
memset(dp,0,sizeof dp);
scanf("%d",&n);
long long int a[n];
for(int i=0;i<n;i++)
scanf("%lld",&a[i]);
long long int sum1,sum2;
for(int l=1;l<=n;l++)
{
for(int i=0;i<=n-l+1;i++)
{
int j=i+l-1;
dp[i][j]=dp[i+1][j];
if(a[i]<0)
{
for(int k=i+1;k<=j;k++)
{
if(a[i]==(-a[k]))
{
if(k>i+1)
sum1=(dp[i+1][k-1]+1)%MOD;
else
sum1=1;
if(k>=j)
sum2=0;
else
sum2=dp[k+1][j];
dp[i][j]=(dp[i][j]+sum1+((sum1*sum2)%MOD))%MOD;
}
}
}
}
}
printf("%lld",(dp[0][n-1]+1)%MOD);
return 0;
}
