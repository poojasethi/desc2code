    #include<iostream>
    #include<map>
    using namespace std;
     
    #define mod 1000000007
     
    typedef long long int ll;
     
    int main()
    {
    int n;
    cin>>n;
    ll i,j,k;
    ll a[n+1];
    ll dp[n+2][n+2];
    for(i=0;i<=n;i++)
    {
    	for(j=0;j<=n;j++)
    	 dp[i][j]=1;
    }
    
    for(i=0;i<n;i++)
    {
    cin>>a[i];
    //dp[i][i]=dp[i+1][i]=1;
    }
for(i=2;i<=n;i++)
{
for(j=0;j<=n-i;j++)
{
int ans=j+i-1;
dp[j][ans]=dp[j+1][ans];
if(a[j]<0)
{
for(int k=j+1;k<=ans;k++)
{
if(a[k]==-a[j])
{
dp[j][ans]+=(dp[j+1][k-1]*dp[k+1][ans]);
dp[j][ans]%=mod;
}
}
}
}
}cout<<dp[0][n-1]%mod<<endl;
    return 0;
    }
