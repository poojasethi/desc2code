using namespace std;
typedef unsigned long long ull;
#define mod 1000000007
int main()
{
	int n, i, j, k,l, dp[101][101];
	long long arr[101];
	cin >> n;
	for (i = 1; i <= n; i++)
		cin >> arr[i];
	for (i = 0; i <= n;i++)
	for (j = 0; j <= n; j++)
		dp[i][j] = 0;
	//memset(dp, 0, sizeof(dp));
	for (l = 2; l <= n; l++)
	{
		for (i = 1; i <= n; i++)
		{
			j = i + l - 1;
			if (j > n) break;
			dp[i][j] = dp[i][j - 1];
			for (k = i; k < j; k++)
			if (arr[j]>0 && arr[k] == -arr[j])
				dp[i][j] = (dp[i][j]+((1 + dp[k + 1][j - 1]) * (1 + dp[i][k - 1])%mod))%mod;		
			
		}
		
	}
	cout << (dp[1][n] + 1)%mod << endl;
}
