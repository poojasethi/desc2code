#include <bits/stdc++.h>

using namespace std;

#define mod 1000000007

int main()
{
    long long int n;
    cin >> n;
    vector <long long int> v(n+1);
    long long int i, j, k, x, y;
    for (i = 1; i <= n; i++)
        cin >> v[i];
    long long int dp[n+1][n+1];
    memset(dp, 0, sizeof(dp));
    for (j = 1; j <= n;j++) {
        for (i = 1; i < j; i++) {
            dp[i][j] += dp[i][j-1];
            if (v[j] > 0) {
                for (k = j-1; k >= i; k--) {
                    if (v[k] == -1*v[j]) {
                        if ((j-k) == 1)
                            x = 0;
                        else
                            x = dp[k+1][j-1];
                        if (k > 0)
                            y = dp[i][k-1];
                        else
                            y = 0;
                        dp[i][j] = (dp[i][j] + (x+1)*(y+1))%mod;
                    }
                }
            }
        }
    }
    printf("%lld\n", (dp[1][n]+1)%mod);
    return 0;
}
