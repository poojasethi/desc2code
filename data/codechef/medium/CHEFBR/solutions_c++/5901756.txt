#include<stdio.h>
#include<algorithm>
#define MAX 110
#define MOD 1000000007
#define ll long long int
#define print(i) printf("%lld ",i)
#define printn(n) printf("%lld\n",n)
#define print2(i,j) print(i),printn(j)
#define print3(i,j,k) print(i),print(j),printn(k)
#define print4(i,j,k,sum) print(i),print(j),print(k),printn(sum)
using namespace std;
ll a[MAX],n,dp[MAX][MAX];
ll fun(ll i,ll j){
 ll k,sum=0;
 if(i<0||j>=n) return 1;
 if(i+1==j&&(a[i]+a[j]==0))
    return 2;
 if(((i+1==j&&(a[i]+a[j]!=0))||(i>=j)))
    return 1;
 if(dp[i][j]!=-1)
    return dp[i][j];
 for(k=i+1;k<=j;k++)
 {
    if((a[i]+a[k])==0)
     {
         sum=(sum+(((fun(i+1,k-1))*(fun(k+1,j))))%MOD);//print4(i,j,k,sum);
     }
 }//printf("%d %d\n",k,sum);
sum=(sum+fun(i+1,j))%MOD;
 return dp[i][j]=sum;
}
int main(){
 ll i,flag=0,j,st,sum=0;
    scanf("%lld",&n);
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
         dp[i][j]=-1;
    for(i=0;i<n;i++)
      {
          scanf("%lld",&a[i]);
          if(a[i]<0&&flag==0)
            {st=i;flag=1;}
      }
   sum=fun(st,n-1);
  printf("%lld",sum);
 return 0;
}
