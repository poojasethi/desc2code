#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <vector>
#include <string>
#include <stack>
#include <queue>
#include <set>

#define pb push_back

using namespace std;
const int MAXN = 110;
const int mod = 1e9 + 7;
int n;
int arr[MAXN];
int dp[MAXN][MAXN];
int main()
{
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", &arr[i]);

    for (int k = 0; k < n; k++)
    {
        for (int i = 1; i <= n; i++)
        {
            int j = k + i;
            if (j > n) continue;
            if (i == j) { dp[i][j] = 0; continue; }
            dp[i][j] = dp[i][j - 1];
            for (int q = i; q < j; q++)
            {
                if (arr[j] > 0 && arr[q] == -arr[j])
                {
                    if (q == i)
                    {
                        dp[i][j] = (dp[i][j] + dp[q + 1][j - 1] + 1) % mod;
                    }
                    else
                    {
                        dp[i][j] = (dp[i][j] + ((1 + dp[q + 1][j - 1]) * (1 + dp[i][q - 1])) % mod) % mod;
                    }
                }
            }
        }
    }
    printf("%d\n", dp[1][n] + 1);
    return 0;
}
