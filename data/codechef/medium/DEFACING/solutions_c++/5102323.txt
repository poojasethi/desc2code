#include<bits/stdc++.h>
#define pb(a) push_back(a)
using namespace std;

char s[12];
char m[12];
char ans[12];
vector<int> d[12];
void init() {
	d[0].pb(0);
	d[0].pb(8);
	d[1].pb(0); d[1].pb(3); d[1].pb(4); d[1].pb(7); d[1].pb(8); d[1].pb(9);
	d[1].pb(1);
	d[2].pb(2); d[2].pb(8);
	d[3].pb(8); d[3].pb(3); d[3].pb(9);
	d[4].pb(4); d[4].pb(8);d[4].pb(9);
	d[5].pb(5);
	d[5].pb(6); d[5].pb(8); d[5].pb(9);
	d[6].pb(6); d[6].pb(8);
	d[7].pb(7); d[7].pb(0); d[7].pb(3); d[7].pb(8); d[7].pb(9);
	d[8].pb(8);
	d[9].pb(8); d[9].pb(9);
}

int check(int l,int l2,int l1) {
	int i,j,k,u,v,x,y;
	i=0;
	j=l;
	int last=-1;
	int lastnum=-1;
	int mx=0;
	int status;
	int overall=-1;
	ans[l2]='\0';
	//printf("L: %d %d\n",l1,l2);
	while (i<l1) {
		x=s[i]-'0';
		///printf("X: %d\n",x);
		y=m[j]-'0';
		mx=-1;
		status=0;
		for (k=0;k<d[x].size();k++) {
			if (overall==0 || d[x][k]<=y) {
				if (d[x][k]<y) {
					if (last!=i) lastnum=-1;
					last=i;
					lastnum=max(lastnum,d[x][k]);
				}
				mx=max(mx,d[x][k]);
				status=1;
			}
		}
		//printf("X: %d Y:%d : %d Last:%d %d\n",x,y,status,last,lastnum);
		if (status==0) {
			if (last>=0) {
				//printf("Entered %d %d %d\n",i,last,lastnum);
				ans[l+last]='0'+lastnum;
				for (u=last+1;u<=i;u++) {
					x=s[u]-'0';
					mx=-1;
					for (k=0;k<d[x].size();k++) {
						mx=max(mx,d[x][k]);
					}
					ans[l+u]='0'+mx;
				}
				overall=0;
			}
			else {
				//printf("Ywah\n");
				if (l==0) return -1;
				char temp[12];
				int z;
				for (i=0;i<l;i++) {
					temp[i]=m[i];
				}
				//printf("Heha %s\n",temp);
				temp[l]='\0';
				z=atoi(temp);
				z-=1;
				//printf("z:%d\n",z);
				sprintf(temp,"%d",z);
				//printf("%s\n",temp);
				int nl=strlen(temp);
				for (i=0;i<l-nl;i++) ans[i]='0';
				for (i=l-nl,j=0;i<l;i++,j++) ans[i]=temp[j];
				for (i=l;i<l+l1;i++) {
					x=s[i-l]-'0';
					mx=-1;
					//printf("X: %d\n",x);
					for (k=0;k<d[x].size();k++) {
						mx=max(mx,d[x][k]);
					}
					//printf("Max: %d\n",mx);
					ans[i]='0'+mx;
				}
				//printf("%s\n",ans);
				for (i=l+l1;i<l2;i++) ans[i]='9';
				return atoi(ans);
			}
		}
		else {
			if (mx<y) overall=0;
			ans[j]='0'+mx;
			//printf("%d\n",mx);
		}
		i++;
		j++;
	}
	//printf("Yeah,overall:%d\n",overall);
	for (i=0;i<l;i++) ans[i]=m[i];
	for (i=l+l1;i<l2;i++) {
		if (overall==-1) {
			ans[i]=m[i];
		}
		else {
			ans[i]='9';
		}
	}
	if (l==0 && overall==0 && ans[0]=='0') return -1;
	//printf("%s\n",ans);
	return atoi(ans);
}

int answer() {
	int l1,l2,l,i,j;
	l1=strlen(s);
	l2=strlen(m);
	int mx=atoi(s);
	for (j=l2-l1;j>=0;j--) {
		mx=max(mx,check(j,l2,l1));
		//printf("%d %d\n",mx,j);
	}
	return mx;
}

int main() {
	int t,i,j,k;
	scanf("%d",&t);
	init();
	while (t--) {
		scanf("%s%s",s,m);
		//printf("%s %s",s,m);
		printf("%d\n",answer());
	}
	return 0;
}
