void process()
{
    int j=strlen(tmp);
    int l=strlen(m);

    int i,k,p;
    for(i=0;i<l-j;i++)
    {
        p=0;
        for(k=0;k<j;k++)
        {
            if(tmp[k]!=s[i+k])
            {
                p=1;
                break;
            }
        }
        if(p==0)
        {
            int x=0;
            for(k=0;k<i+j;k++)
                if(s[i]<m[i])
                {
                    x=1;
                    break;
                }
            for(k=i+j;k<l;k++)
            {
                if(x==0)
                    s[k]=m[k];
                else
                    s[k]='9';
            }
            break;
        }
    }
}

void update(int l)
{
    int i,j;
    for(i=0;i<l;i++)
    {
        if(bst[i]>s[i])
            return;
        else if(bst[i]<s[i])
            break;
    }
    for(i=0;i<l;i++)
        bst[i]=s[i];
}

int f;
void func(int i,int l)
{
    int j,k,x;
    if(i==l)
    {
        f=1;
        update(l);
        return;
    }

    if(s[i]==58 || change[s[i]-48][m[i]-48])
    {
        char c=s[i];
        s[i]=m[i];
        func(i+1,l);
        s[i]=c;
    }

    if(f)
        return;
    for(j=m[i]-49;j>=0;j--)
    {
        //printf("%d %d %d %d\n",i,j,m[i],s[i]);
        if(j+48<=m[i] && change[s[i]-48][j])
        {
            s[i]=j+48;
            for(k=i+1;k<l;k++)
            {
                for(x=9;x>=0;x--)
                {
                    if(change[s[k]-48][x])
                    {
                        s[k]=x+48;
                        break;
                    }
                }
            }
            f=1;
            update(l);
            break;
        }
    }
}

            for(k=i;k<i+j;k++)
                s[k]=tmp[k-i];
            for(k=i+j;k<l;k++)
                s[k]=58;
            f=0;
            func(0,l);
        }

        k=0;
        while(k<l && bst[k]=='0')
