int solve(int k){

	for(int i=0;i<s.length()+k;i++){
		
		if(i<k){
		
			equal *= 10;
			int k1 = con(m.at(i));
			equal += k1;
			less = equal-1;
		} else {
			
			int p1 = con(s.at(i-k));
			int q1 = con(m.at(i));
			
			
			if(equal==-1){
			
				less *= 10;
				if(val[p1].at(9)=='1')
					less += 9;
				else

			} else {

					for(int j=0;j<q1;j++){
						if(val[p1].at(j)=='1')
							hh = j;
					}
					if(hh!=-1)
						less = 10*equal + hh;
					else{
						less *= 10;
						if(val[p1].at(9)=='1')
							less += 9;
						else
							less += 8;
					}
					equal = -1;
				} else {
					int tmp1 = equal;
					int tmp2 = less;
					equal *= 10;
					equal += q1;
					int ka = -1;
					for(int j=0;j<q1;j++){
						if(val[p1].at(j)=='1')
							ka = j;

						if(val[p1].at(9)=='1')
							less += 9;
						else
							less += 8;
					}

						less = -1;
				}
				
					
			}
		
		}

		return -1;
	if(equal>less)
		return toInt(m);

		for(int i=s.length()+k;i<m.length();i++){
			less *=10;
			less += 9;
		}

}
