#include<stdio.h>
char puzzle[1001][1001];
char s[1000001];
int n;
int kk=0;
int recursion(int i, int j,int index)
{

	if(s[index+1]=='\0')
		return 1;
	int numberofways=0;
	if(i!=n-1 && j!=n-1 && puzzle[i+1][j+1]==s[index+1])
		numberofways+=recursion(i+1,j+1,index+1);
	//printf("%d\n",numberofways );
	if(i && j!=n-1 && puzzle[i-1][j+1]==s[index+1])
		numberofways+=recursion(i-1,j+1,index+1);
	if(i && j && puzzle[i-1][j-1]==s[index+1])
		numberofways+=recursion(i-1,j-1,index+1);
	if(i!=n-1 && j && puzzle[i+1][j-1]==s[index+1])
		numberofways+=recursion(i+1,j-1,index+1);

return numberofways;
}
int main()
{
int i,j;
char te;
int answer=0;
scanf("%d ",&n);
//scanf("%c",&te);

for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
			scanf("%c ",&puzzle[i][j]);
			
	}
scanf("%s",s);
for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		if(puzzle[i][j]==s[0])
			answer+=recursion(i,j,0);
	}
printf("%d", answer);
		
return 0;
}
