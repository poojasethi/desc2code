#include<iostream>
#include<string>
using namespace std;
int getCount(char A[1100][1100],int i,int j,int n,string s,int index){
	if(A[i][j]=='X')
	return 0;	
	if(A[i][j]==s[index] && s[index+1]=='\0')
		return 1;
	if(A[i][j]!=s[index])
		return 0;
	int count=0;
	index++;
	if(i<n && j<n){
	count+=getCount(A,i+1,j+1,n,s,index);
	}
	if(i>0 && j>0){
	count+=getCount(A,i-1,j-1,n,s,index);
	}
	if(i>0 && j<n){
	count+=getCount(A,i-1,j+1,n,s,index);
	}
	if(i<n && j>0){
	count+=getCount(A,i+1,j-1,n,s,index);			
	}	
	return count;
}
int countPossible(char A[1100][1100],int n,string s){
		
	int count=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++){
			
			if(A[i][j]==s[0]){
				count+=getCount(A,i,j,n,s,0);			
			}		
					
		}
		
	}
			
	return count;
}
int main(){
	int n;
	string s;
	cin>>n;
	char A[1100][1100];
	for(int i=0;i<n;i++){
		for(int j=0;j<n;j++)
		cin>>A[i][j];
	}
	cin>>s;
	cout<<countPossible(A,n,s)<<"\n";
return 0;
}
