#include <stdio.h>
#include <stdlib.h>
#define MAX 1010
char mat[MAX][MAX];
char S[MAX*MAX];
int count=0;
int n;
 
void seed(int x,int y,int i)
{
 
    if( x>n || y>n || x<1 || y<1 ) return ;
 
    if(mat[x][y]==S[i])
    {
        if(S[i+1]=='\0')
        {
            count++;
            return ;
        }
 
        seed(x+1,y+1,i+1);
        seed(x+1,y-1,i+1);
        seed(x-1,y+1,i+1);
        seed(x-1,y-1,i+1);
    }
 
}
int main()
{
    int i,j;
    char x;
    scanf("%d",&n);
 
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            x='*';
            while( !((x>='A'&&x<='Z')||(x>='a'&&x<='z')) )
                x=getchar();
            mat[i][j]=x;
        }
    }
    scanf("%s",S);
 
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            seed(i,j,0);
        }
    }
    printf("%d\n",count);
    return 0;
}
