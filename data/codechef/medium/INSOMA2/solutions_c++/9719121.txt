#include<iostream>
using namespace std;

int findanswer(char arr[][1001], char search[], int n, int x, int y, int current){
    if(search[current] == '\0')
        return 1;
    if(x<=0 || y<=0 || x>n || y>n){
        return 0;
    }

    int moves[][2]={            
        {-1,-1},
        {-1,1},
        {1,-1},
        {1,1}
    };

    int answer = 0;

    for(int i=0; i<4; i++){
        if(arr[x+moves[i][0]][y+moves[i][1]] == search[current]){
            answer += findanswer(arr,search,n,x+moves[i][0],y+moves[i][1],current+1);
        }
    }
    return answer;
}

int main(){
    char arr[1001][1001]={0},search[1010];
    int n;
    cin>>n;
    for(int i=1; i<=n; i++){
        for(int j=1; j<=n; j++){
            cin>>arr[i][j];
        }
    }
    cin>>search;
    
    int answer = 0;
    for(int i=1; i<=n; i++){
        for(int j=1; j<=n; j++){
            if(arr[i][j]==search[0])
                answer += findanswer(arr,search,n,i,j,1);
        }
    }
    cout<<answer;
}
