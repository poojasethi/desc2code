#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int matrix[1005][1005]={0},n,len;
char str[5005];
long long int func(int i,int j,int k)
{
    if(i>=n || j>=n || i<=-1 || j<=-1 || k>=len)  return 0;
    long long int ans=0;
    if(matrix[i][j]==str[k])
    {
        if(k==len-1)  return 1;
        else
            ans+=func(i-1,j-1,k+1)+func(i-1,j+1,k+1)+func(i+1,j-1,k+1)+func(i+1,j+1,k+1);
    }
    return ans;
}
int main()
{
    int i,j,k;
    char ch;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            ch=getchar();
            if(ch<=122 && ch >=65)
                matrix[i][j]=ch;
            else
                j--;
        }
    }
    /*
    printf("\n\nMATRIX\n");
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
            printf("%c",matrix[i][j]);
        printf("\n");
    }
    printf("\n\n");*/
    
    scanf("%s",str);
    //printf("str:%s\n",str);
    len=strlen(str);
    //printf("len:%d\n",len);
    long long int ans=0;
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
            ans+=func(i,j,0);
    }
    printf("%lld\n",ans);
    return 0;
}
