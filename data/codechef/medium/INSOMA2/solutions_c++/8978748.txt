#include<iostream>
#include<climits>
using namespace std;

int count=0,n;
char mat[1001][2001],pat[2001];

void checkPattern(int i,int j,int k){
    
	if(!(i>=0 && j>=0 && i<n && j<n))
		return;

	if(mat[i][j]==pat[k]){
		if(!pat[k+1])
			count++;
		else{
			checkPattern(i-1,j-1,k+1);
			checkPattern(i-1,j+1,k+1);
			checkPattern(i+1,j-1,k+1);
			checkPattern(i+1,j+1,k+1);
		}
	}
}

int main(){
	int i,j,k;
	
	cin>>n;
    	cin.ignore(INT_MAX,'\n');
	for(i=0;i<n;i++)
		cin.getline(mat[i],2001);
	
	cin>>pat;
	for(i=0;i<n;i++){
		k=0,j=0;
		while(mat[i][j]){
			if(mat[i][j]!=' ')
				mat[i][k++]=mat[i][j];
			j++;
		}
		mat[i][k]='\0';
	}
	
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			checkPattern(i,j,0);
	cout<<count;
}
