#include <stdio.h>
#include <stdlib.h>
#include <string.h>
 
char **box;
int ctr;
 
void fun(char *str,int len,int k,int n,int i,int j)
{
if (k == len)
{
ctr++;
return;
}
if (i+1 < n && j+1 < n)
{
if (box[i+1][j+1] == str[k])
fun(str,len,k+1,n,i+1,j+1);
}
if (i+1 < n && j-1 >= 0)
{
if (box[i+1][j-1]==str[k])
fun(str,len,k+1,n,i+1,j-1);
}
if (i-1 >= 0 && j+1 < n)
{
if (box[i-1][j+1]==str[k])
fun(str,len,k+1,n,i-1,j+1);
}
if (i-1 >= 0 && j-1 >= 0)
{
if (box[i-1][j-1]==str[k])
fun(str,len,k+1,n,i-1,j-1);
}
}
void count(char *str, int len, int n)
{
int i,j,k=0;
for (i=0;i<n;i++)
for (j=0;j<n;j++)
if (box[i][j] == str[k])
fun(str,len,k+1,n,i,j);
}
int main()
{
int n;
scanf (" %d",&n);
box = (char **)calloc(n,sizeof(char *));
int i,j;
for (i=0;i<n;i++)
box[i] = (char *)calloc(n,sizeof(char));
for (i=0;i<n;i++)
for (j=0;j<n;j++)
scanf(" %c",&box[i][j]);
char *str = (char *)calloc(n*n,sizeof(char));
scanf (" %s",str);
ctr = 0;
count(str,strlen(str),n);
printf ("%d\n",ctr);
return 0;
}
