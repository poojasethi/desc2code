inline void inp(int *a) {
    register char c=0;
    while (c < 33)
		c = getchar_unlocked();
	*a = 0;	
	while (c > 33) {
		*a = *a * 10 + c - '0';
		c = getchar_unlocked();
	}
}

    if ((x+1<n)&&(y-1>=0)&&(str[k]==a[x+1][y-1]))
        check(a,str,x+1,y-1,k+1);
}
