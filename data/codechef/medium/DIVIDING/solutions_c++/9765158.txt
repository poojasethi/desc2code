#include <iostream>
using namespace std;
#define gt getchar_unlocked
void inp(long long int &n)
{
int c;	
n=0;
c=gt();
while(c<'0' || c>'9')
c=gt();
while(c>='0' && c<='9')
{
n=(n<<3)+(n<<1)+c-'0';
c=gt();
}
}
int main() {
long long int n,i,a;
long long int s=0;
cin>>n;
for(i=0;i<n;i++)
{
inp(a);
s+=a;
}
if(s==n*(n+1)/2)
cout<<"YES"<<endl;
else
cout<<"NO"<<endl;
return 0;
}
