#include <iostream>
using namespace std;

int main() {
	cin.sync_with_stdio(false);
	int i;
	long long int stamp,n,sum=0,res;
	cin>>n;
	res=(n*(n+1))/2;
	while(n--)
	{
		cin>>stamp;
		//cout<<stamp<<endl;
		sum+=stamp;
	}
	//cout<<sum<<" "<<res<<endl;
	if(sum==res)
	cout<<"YES";
	else
	cout<<"NO";
	return 0;
}
