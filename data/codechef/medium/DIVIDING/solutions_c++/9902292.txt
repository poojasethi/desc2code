#include<iostream>
using namespace std;
int main()
{
    unsigned long long int n;
    cin>>n;
    unsigned long long int arr[n];
   unsigned long long int sum=0;

    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
        sum+=arr[i];
    }
    if(sum==((n)*(n+1)/2))
       {
           cout<<"YES"<<endl;

       }
       else
        cout<<"NO"<<endl;
}
