#include <cmath>
#include <cstring>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main() 
{
   int n,sum,sum1,i,m;
    int ar[100000];
    cin>>n;
    sum=0;sum1=0;
    m=0;
     for(i=1;i<=n;i++)
         {
        cin>>ar[i]; 
        }
    for(i=1;i<=n;i++)
        {
        sum=sum+i;
        sum1=sum1+ar[i];
        if(sum1%sum==0)
            {m=1;}
        else
            {m=0;}
    }
    if(m==1)
        {cout<<"YES"<<endl;}
    else
        {cout<<"NO"<<endl;}
   return 0;
}
