#include <iostream>
using namespace std;

int main() {
    // insert code here...
    long int n;
    cin>>n;
    long long int a[n],sum=0,total=0;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
    }
    for(int i=0;i<n;i++)
    {
        sum+=a[i];
    }
    while(n>0)
    {
        total+=n;
        n--;
    }
    if(total==sum)
        cout<<"YES"<<"\n";
    else
        cout<<"NO"<<"\n";
    return 0;
}
