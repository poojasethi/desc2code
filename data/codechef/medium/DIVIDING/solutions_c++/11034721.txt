#include <iostream>
using namespace std;

int main() {
	cin.sync_with_stdio(false);
	int i;
	long long int stamp,n,sum=0,res; //res should be long long int as value lies in that range
	cin>>n;
	res=(n*(n+1))>>1;
	//cout<<res<<endl;
	while(n--)
	{
		cin>>stamp;
		//cout<<stamp<<endl;
		sum+=stamp;
	}
	//cout<<sum<<" "<<res<<endl;
	if(sum==res) //if we calculate res here i.e. res=(n*(n+1))/2 then n will reduce to 0 due to n-- and thus res=0
	cout<<"YES";
	else
	cout<<"NO";
	return 0;
}
