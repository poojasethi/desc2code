#include<cstdio>
using namespace std;
int main(){
        int N;
        scanf("%d",&N);
        long total=0;
        for(int i=0;i<N;i++){
            long tmp;
            scanf("%ld",&tmp);
            total+=tmp;
        }
        if(2*total==(N*(N+1))){
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }
