#include <iostream>

using namespace std;

int main(int argc, char const *argv[])
{
    int n, i;
    long totalStamps = 0, reqStamps, Ci;

    cin>>n;

    reqStamps = (n + 1) * n / 2;

    for(i = 0; i < n; ++i)
    {
        cin>>Ci;
        totalStamps += Ci;
    }

    if(totalStamps == reqStamps)
        cout<<"YES\n";
    else
        cout<<"NO\n";
    return 0;
}
