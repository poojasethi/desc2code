#include<stdio.h>
using namespace std;
int main()
{
	double high,low,p,pa[101];
	int t,a[101][3],n,i,k1,k2,k;
	scanf("%d",&t);
	while(t--)
	{
		high=1.0;
		low=0.0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i][0]);
			if(a[i][0]!=0)
			{
				scanf("%d %d",&a[i][1],&a[i][2]);
			}
		}
		k=0;
		while(k<=80)
		{
			k++;
			p=(high+low)/2;
			//printf("%0.5f p",p);
			for(i=0;i<n;i++)
			{
				if(a[i][0]==0)
				{
					pa[i]=p;
				}
				else if(a[i][0]==1)
				{
					k1=a[i][1];k2=a[i][2];
					pa[i]=1-(1-pa[k1-1])*(1-pa[k2-1]);
				}
				else
				{
					k1=a[i][1];k2=a[i][2];
					pa[i]=pa[k1-1]*pa[k2-1];
				}
			}
//k1=(int)(high*100000);
//			k2=(int)(low*100000);
//			if(k1==k2)break;
			if(pa[n-1]>0.5)
			{
				high=p;
			}
			else if(pa[n-1]<0.5)
			{
				low=p;
			}
			else 
			{
				break;
			}
			
			//printf("%0.5f %0.5f %0.5f\n",pa[n-1],high,low);
		}
		printf("%0.5f\n",p);
	}
	return 0;
}
