#include <stdio.h>
#include <iostream>
#include <time.h>
#include <math.h>

using namespace std;


float get_result(float a, float b, int c){
	if(c == 1){
		return (1-(1-a)*(1-b));	
	}
	else 
		return a*b;
}

int main(){
	int n, no_of_nodes; 
	int i,j,k;
	int g = 0;
	cin >> n;
	float min = 0, max = 1, p;
	p = 0.5;
	for(i = 0; i < n; i++){
		cin >> no_of_nodes;
		int a[no_of_nodes][3];
		float out[no_of_nodes];
		min = 0;
		max = 1;
		p = 0.5;
		g = 0;
		for(j = 0; j < no_of_nodes; j++){
		//	cout << "enter nodes : " << endl;
			cin >> a[j][0];
			if(a[j][0] != 0){
				if(g == no_of_nodes){
					g = j;
				}
				cin >> a[j][1];
				cin >> a[j][2];
			}
			else{
				out[j] = p;
				g = no_of_nodes;
			}
		}
		for(int t = 0; t < 70; t++){
			p = (min+max)/2;
			for(int x = 0; x < g; x++){
				out[x] = p;
			}
			for(k = g; k < no_of_nodes; k++){
				out[k] = get_result(out[a[k][1]-1], out[a[k][2]-1], a[k][0]);
			}
			if(out[no_of_nodes-1] == 0.5)
				break;
			else if(out[no_of_nodes-1] > 0.5){
				max = p;
				continue;
			}
			else{
				min = p;
				continue;
			}
		}
		printf("%.5f\n", p);
	}
	return 0;
}
