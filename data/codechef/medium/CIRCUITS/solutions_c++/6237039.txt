#include<iostream>
#include<iomanip>
#include<cstdio>

using namespace std;

int main()
{
    int testCases;
    cin >> testCases ;
    int nodes;
    int n[100][2], t[100];
    double p[100];
    while (testCases--) {
        int i;
        cin >> nodes;
        for (i = 0; i < nodes; i++) {
            cin >> t[i];
            if (t[i] != 0)
                cin >> n[i][0] >> n[i][1];
            --n[i][0];
            --n[i][1];
        }


    double ep = 1e-8, low=0, high=1;

    for (int iter = 0; iter < 30; iter++) {
        double mid = (low + high)/2;

        for (int j = 0; j< nodes; j++) {
            if (t[j] == 0)
                p[j] = mid;
            else if (t[j] == 1)
                p[j] = p[n[j][0]] + p[n[j][1]] - p[n[j][0]] * p[n[j][1]];
            else
                p[j] = p[n[j][0]] * p[n[j][1]];
        }

    if (p[nodes-1] > 0.5-ep)
        high = mid;
    else
        low = mid;
    }
        printf("%.5lf\n",low);
    }
    return 0;
}
