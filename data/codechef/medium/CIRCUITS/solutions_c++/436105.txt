double prob[200];
int gate[200][3]; //0-type , 1-a 2-b
int n;

double check(double p)
{
    int a,b;
    double c,d;
    for(int i=1;i<=n;i++){
        a=gate[i][1];b=gate[i][2];
        c=prob[a];d=prob[b];
        if(gate[i][0]==0){
            prob[i]=p;
        }
        else if(gate[i][0]==1){
            prob[i]=(c*(1-d))+((1-c)*d)+(c*d);
        }
        else if(gate[i][0]==2){
            prob[i]=c*d;
        }
    }
    return prob[n];
}

int main()
{
    //freopen("input.txt","r",stdin);
    int t;
    cin>>t;
    while(t--){
        cin>>n;
        int a,b;
        for(int i=1;i<=n;i++){
            int ty,a,b;
            cin>>ty;
            gate[i][0]=ty;
            if(ty!=0){
                cin>>a>>b;
                gate[i][1]=a;gate[i][2]=b;
            }
        }
        double lb,ub,mid,rt,ans;
        lb=0;ub=1.0;
        while(1){
            mid=(ub+lb)/2;
            rt=check(mid);
            if(fabs(rt-0.5)<10e-8){
                ans=mid;
                break;
            }
            else if(rt>0.5)
                ub=mid;
            else
                lb=mid;
        }
        printf("%.5f\n",ans);
    }
}
