#include<bits/stdc++.h>
using namespace std;
#define  mp make_pair
#define  pb push_back
#define  M 10000000
typedef long long int ll;
int n;
int a[10020];
int maxi;
int dp[1001][10002];
int fs()
{
    int x=0;
    char c=getchar_unlocked();
    while (!('0'<=c&&c<='9'))
        c=getchar_unlocked();
    while ('0'<=c&&c<='9')
    {
        x=(x<<3)+(x<<1)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}

int func(int i, int sum)
{
    if(i==n)
    {
        int taken=sum;
        int nottaken=maxi-sum;
        return abs(nottaken-taken);
    }
    int &ans=dp[i][sum];
    if(ans!=-1)
        return ans;

    return ans=(func(i+1,sum+a[i])+func(i+1,sum))%M;
}
int main()
{
    int t,i;
    scanf("%d",&t);
    while(t--)
    {   
        memset(dp,-1,sizeof(dp));
        scanf("%d",&n);
        maxi=0;
        for(i=0;i<n;i++)
        {
            a[i]=fs();
            maxi+=a[i];
        }
        int ans=func(0,0);
        printf("%d\n",ans);
    }       
    return 0;
}
