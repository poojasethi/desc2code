#include <bits/stdc++.h>
using namespace std;
long long dp[1010][10010],sum=0;
long long int a[100010],mod=10000000,n;
long long int f(long long i,long long s)
{
    if(i==n)
    {
        return abs(sum-(2*s))%mod;
    }
    if(dp[i][s]!=-1)
       return dp[i][s];
    return dp[i][s]=(f(i+1,s+a[i])%mod+f(i+1,s)%mod)%mod;
}
int main() {
	// your code goes here
	long long int t;
	scanf("%lld",&t);
	while(t--)
	{
	    scanf("%lld",&n);
	    
	    long long int i,j;
	    sum=0;
	    for(i=0;i<n;i++){
	        scanf("%lld",&a[i]);
	        sum=(sum+a[i])%mod;
	    }
	    for(i=0;i<=n;i++)
	    for(j=0;j<=sum;j++)
	        dp[i][j]=-1;
	    printf("%lld\n",f(0,0)%mod);
	}
	return 0;
}
