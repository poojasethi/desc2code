#include<bits/stdc++.h>
using namespace std;
#define mod 10000000
long long int a[100010];
long long int sum=0;
long long int n;
int dp[1010][10010];
long long int fun(long long int ind,long long int su)
{
    if(ind==n)
     {
        int s=sum-su;
        return abs(s-su)%mod;
     }
     if(dp[ind][su]!=-1)
     return dp[ind][su]%mod;
    return dp[ind][su]=(fun(ind+1,(su+a[ind])%mod)%mod+fun(ind+1,su)%mod)%mod;
    
}
int main()
{    
    long long int t,i;
    scanf("%lld",&t);
    
    while(t--)
    { sum=0;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
       { scanf("%lld",a+i);
       sum=(sum+a[i])%mod;
       }
       int i,j;
       for(i=0;i<=n;i++)
       for(j=0;j<=sum;j++)
       dp[i][j]=-1;
        long long int ans=fun(0,0)%mod;
        printf("%lld\n",ans);
        
    }
}
