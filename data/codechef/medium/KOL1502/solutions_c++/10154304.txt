#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int MAXN = 1e3 + 10;
typedef long long LL;
int A[MAXN];
LL dp[10*MAXN];
const LL MOD = 1e7;
int main(){
    int t;
    scanf("%d", &t);
    while(t--){
        memset(dp, 0, sizeof dp);
        int N;
        scanf("%d", &N);
        int sum = 0;
        for(int i = 1;i <= N;++i){
            scanf("%d", &A[i]);
            sum += A[i];
        }
        dp[0] = 1;
        for(int i = 1;i <= N;++i){
            for(int j = sum;j; --j){
                if(A[i] <= j){
                    dp[j] += dp[j - A[i]];
                }
                while(dp[j] >= MOD) dp[j] -= MOD;
            }
        }
        LL ans = 0;
        for(int i = 0;i <= sum;++i){
            ans = (ans + max(i - (sum - i), (sum - i) - i) * dp[i]) % MOD;
        }
        printf("%lld\n", ans);
    }
}
