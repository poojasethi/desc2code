void scanint(int &N)
{
    int ch,sign;
    N=0;
    while((ch<'0'||ch>'9') && ch!='-'&& ch!=EOF)
        ch=getchar_unlocked();
    if(ch=='-')
        sign=-1,ch=getchar_unlocked();
    else sign=1;
    do
        N=(N<<3)+(N<<1)+(ch-'0');
    while((ch=getchar_unlocked())>='0' && ch<='9');
    N*=sign;
    return;
}

const int mod = 1e7;
int arr[1005];
int sum,n;
vector<vector<int> >dp;
int pro(int ind, LLI in) {
	if(ind == n) {
		return max(2*in-sum, sum-2*in);
	}
	if(dp[ind][in] != -1) {return dp[ind][in];}
	LLI a = pro(ind+1, in+arr[ind]);
	LLI b = pro(ind+1, in);
	return dp[ind][in] = (a+b)%mod;
}
