#include <bits/stdc++.h>
using namespace std;

/* Note

*/	

long long knp[10005];

main(){
	long long t,i,j,n,m,sum,num,res;	
	
	scanf("%lld",&t);
	while(t--){
		sum = res = 0;
		memset(knp,0,sizeof knp);
		knp[0] = 1;
		scanf("%lld",&n);
		for(i = 1; i <= n; i++){
			scanf("%lld",&num);
			for(j = sum; j >= 0; j--){
				knp[j+num] += knp[j];
				knp[j+num] %= 10000000;
			}
			sum += num;
		}
		for(i = 0; i <= sum; i++){
			res += (abs((sum-i)-i)*knp[i])%10000000;
			res %= 10000000;
		}
		printf("%lld\n",res);
	}
	
	return 0;	
}
