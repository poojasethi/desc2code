#define LM 1007
#define mod 10000000
ll n,scr[LM],t=0;
ll dp[LM][30006];
ll vs[LM][30006];
ll sol(ll pos,ll cost)
{
    if(pos>=n)return abs(cost-15000);
    ll &ans = dp[pos][cost];
    if(vs[pos][cost]==t)return ans;
    vs[pos][cost] = t;
    ans = 0;
    ans = (ans+sol(pos+1,cost+scr[pos]))%mod;
    ans = (ans+sol(pos+1,cost-scr[pos]))%mod;
    return ans;
}

int main()
{
    ll T;
    cin>>T;
    while(T--)
    {
        t++;
        cin>>n;
        For(i,0,n-1)
        {
            cin>>scr[i];
        }
        cout<<sol(0,15000)<<endl;
    }
    return 0;
}
