LL call(int indx, int sum)
{
    if(indx < 0)
        return max(sum,tot-sum) - min(sum,tot-sum);
    LL &ret = dp[indx][sum];
    int &visit = visited[indx][sum];
    if(visit == kase)
        return ret;
    visit = kase;
    ret = call(indx-1, sum);
    ret += call(indx-1, sum+arr[indx]);
    ret %= mod;
    return ret;
}
