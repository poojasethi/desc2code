#include <iostream>
#include <algorithm>
#include <string>
#include <string.h>
#include <time.h>
#include <queue>
#include <stack>

#define MAX 1009
#define MOD (int)(1e7)
#define INF 987654321
//#define scanf scanf_s
using namespace std;

int N, sum;
int arr[MAX];
int dp[MAX][20001];

void init()
{
	sum = 0;
	scanf("%d", &N);
	for (int i = 0; i < N; ++i)
	{
		scanf("%d", &arr[i]);
		sum += arr[i];
	}
}

int func(int idx, int num)
{
	if (idx == N)
		return abs(num - sum);
	int&ret = dp[idx][num];
	if (ret != -1)
		return ret;
	ret = 0;

	ret += func(idx + 1, num + arr[idx]);
	ret %= MOD;
	ret += func(idx + 1, num - arr[idx]);
	ret %= MOD;
	return ret;
}

void solve()
{
	for (int i = 0; i < N; ++i)
		for (int j = 0; j <= sum + sum; ++j)
			dp[i][j] = -1;
	printf("%d\n", func(0, sum));
}

int main()
{
	int tc = 1;
	scanf("%d", &tc);

	for (int i = 0; i < tc; ++i)
	{
		init();
		solve();
	}
	return 0;
}
