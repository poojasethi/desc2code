#include <bits/stdc++.h>
 
#define max(a,b) a>b?a:b 
 
using namespace std;
 
int a[100][100];
 
int main()
{	
	int t,n;
	scanf("%d", &t);
	while(t--)
	{
		int ans=0;
		int colxor[100][100][100]={0};
		scanf("%d", &n);
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				scanf("%d", &a[i][j]);
			}
		}
 
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{	
				colxor[i][j][j]=a[i][j];
				for(int k=j+1;k<n;k++)
				{
					colxor[i][j][k] = colxor[i][j][k-1]^a[i][k];
				}
			}
		}
 
		int temp;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				for(int k=j;k<n;k++)
				{
					temp=colxor[i][j][k];
					ans=max(temp, ans);
					for(int l=i+1;l<n;l++)
					{
						temp^=colxor[l][j][k];
						ans=max(temp, ans);	
					}
				}
			}
		}
		printf("%d\n", ans);
	}
	return 0;
}
