#include <cstdio>
#include <algorithm>
#define MAX 70

using namespace std;

long long int xor_matrix[MAX+5][MAX+5];int a[MAX+5][MAX+5];

int main()
{
    int i,j,T,N,x1,x2,y1,y2;long long int cur,maxi;
    scanf("%i",&T);
    while(T--)
    {
        scanf("%i",&N);
        for(i=0;i<N;i++)
            for(j=0;j<N;j++)
                scanf("%i",&a[i][j]);
        maxi=xor_matrix[0][0]=a[0][0];
        for(i=1;i<N;i++) {
        xor_matrix[0][i]=xor_matrix[0][i-1]^a[0][i];
        xor_matrix[i][0]=xor_matrix[i-1][0]^a[i][0]; cur=max(xor_matrix[0][i],xor_matrix[i][0]);
        maxi=max(cur,maxi);
        }
        for(i=1;i<N;i++)
            for(j=1;j<N;j++)
            {xor_matrix[i][j]=xor_matrix[i-1][j]^xor_matrix[i][j-1]^xor_matrix[i-1][j-1]^a[i][j];maxi=max(xor_matrix[i][j],maxi);}
        for(x1=1;x1<N;x1++)
            for(y1=1;y1<N;y1++)
                for(x2=x1;x2<N;x2++)
                    for(y2=y1;y2<N;y2++)
                    {cur=xor_matrix[x1-1][y1-1]^xor_matrix[x1-1][y2]^xor_matrix[x2][y1-1]^xor_matrix[x2][y2];maxi=max(maxi,cur);}
        x1=0;
        for(y1=1;y1<N;y1++)
                for(x2=0;x2<N;x2++)
                    for(y2=y1;y2<N;y2++)
                    {cur=xor_matrix[x2][y1-1]^xor_matrix[x2][y2];maxi=max(maxi,cur);}
        y1=0;
        for(x1=1;x1<N;x1++)
                for(x2=x1;x2<N;x2++)
                    for(y2=0;y2<N;y2++)
                    {cur=xor_matrix[x1-1][y2]^xor_matrix[x2][y2];maxi=max(maxi,cur);}

        printf("%lld\n",maxi);
    }
    return 0;
}
