#include<iostream>
using namespace std;
int M[71][71];
int C[71][71];
int main()
{
	int t,n,ans,i,j,i1,j1;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		for(j=0;j<n;j++)
		scanf("%d",&M[i][j]);
		
		for(i=0;i<n;i++)
		{
			C[i][i]=M[i][i];
			if(i!=0)
			{
				C[i][i]=C[i][i]^C[i-1][i]^C[i][i-1]^C[i-1][i-1];
			}
			for(j=i+1;j<n;j++)
			{
				C[i][j]=M[i][j];
				C[j][i]=M[j][i];
				if(((i-1)>=0) && ((j-1)>=0))
				{
					C[i][j]=C[i][j]^C[i-1][j]^C[i][j-1]^C[i-1][j-1];
					C[j][i]=C[j][i]^C[j-1][i]^C[j][i-1]^C[j-1][i-1];
				}
				else if((i-1)>=0)
				{
					C[i][j]^=C[i-1][j];
					C[j][i]^=C[j][i-1];
				}
				else if((j-1)>=0)
				{
					C[i][j]^=C[i][j-1];
					C[j][i]^=C[j-1][i];
				}
				
			}
		}
		ans=0;int tmp;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				for(i1=i;i1<n;i1++)
				{
					for(j1=j;j1<n;j1++)
					{
						tmp=C[i1][j1];
						if(((i-1)>=0) && ((j-1)>=0))
						{
							tmp^=C[i-1][j-1]^C[i-1][j1]^C[i1][j-1];
						}
						else if((i-1)>=0)
						{
							tmp^=C[i-1][j1];
						}
						else if((j-1)>=0)
						{
							tmp^=C[i1][j-1];
						}
						if(tmp>ans)
						ans=tmp;
					}
				}
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
