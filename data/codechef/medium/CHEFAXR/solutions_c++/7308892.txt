#include <iostream>
#include <cstdio>
using namespace std;
int dp[70][70][70][70];

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        int arr[n][n];
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++)
                scanf("%d",&arr[i][j]);
        }
        int ans=0;
        for(int x1=0;x1<n;x1++){
            for(int y1=0;y1<n;y1++){
                for(int x2=x1;x2<n;x2++){
                    int hor[n];
                    hor[y1]=arr[x2][y1];
                    for(int i=y1+1;i<n;i++)
                        hor[i]=hor[i-1]^arr[x2][i];
                    for(int y2=y1;y2<n;y2++){
                        if(x1==x2&&y1==y2)
                            dp[x1][y1][x2][y2]=arr[x1][y1];
                        else if(x1==x2)
                            dp[x1][y1][x2][y2]=hor[y2];
                        else if(y1==y2)
                            dp[x1][y1][x2][y2]=dp[x1][y1][x2-1][y2]^arr[x2][y2];
                        else
                            dp[x1][y1][x2][y2]=dp[x1][y1][x2-1][y2]^hor[y2];
                        ans=max(ans,dp[x1][y1][x2][y2]);
                    }
                }
            }
        }
        printf("%d\n",ans);
    }
}
