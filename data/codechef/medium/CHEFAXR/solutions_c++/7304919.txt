#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define ll long long
#define fr(i,beg,end) for(i=beg;i<=end;i++)
#define PII pair<int,int>
using namespace std;
int dp[71][71];
int main()
{
	int i,j,p,q,t,n;
    int a[71][71];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        memset(dp,0,sizeof(dp));
        fr(i,1,n)
            fr(j,1,n)
                scanf("%d",&a[i][j]);
        fr(i,1,n)
            fr(j,1,n)
                dp[i][j]=a[i][j]^dp[i][j-1]^dp[i-1][j]^dp[i-1][j-1];
        int ans=0;
        fr(i,1,n)
        {
            fr(j,1,n)
            {
                int tans=0;
                fr(p,i,n)
                    fr(q,j,n)
                        tans=max(tans,(dp[i-1][j-1]^dp[i-1][q]^dp[p][q]^dp[p][j-1]));
                ans=max(ans,tans);
            }
        }
        printf("%d\n",ans);
    }
	return 0;
}
