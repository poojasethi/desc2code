#include <bits/stdc++.h>
#define maxn 72

using namespace std;

typedef long long ll;

ll A[maxn][maxn];
int n;

#define rep(x, a, b) for(int x=a; x<=b; x++)
int main() {
	int t;
	scanf("%d",&t);
	
	while(t--) {
		scanf("%d",&n);
		
		
		rep(i, 1, n) {
			rep(j, 1, n) {
				scanf("%lld",&A[i][j]);
				A[i][j] ^= A[i-1][j]^A[i][j-1]^A[i-1][j-1];
			}
		}
		
		ll ans = 0;
		rep(i, 1, n) {
			rep(j, 1, n) {

				ll temp = 0;
				rep(k , 1, i) {
					rep(l, 1, j) {
						temp = max(temp, A[i][l-1] ^ A[k-1][j] ^ A[i][j] ^ A[k-1][l-1]);
					}
				}
				
				ans = max(ans, temp );
			}
		}
		
		printf("%lld\n", ans);
	}
}
