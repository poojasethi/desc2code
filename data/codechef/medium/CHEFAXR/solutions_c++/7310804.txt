#include<bits/stdc++.h>
using namespace std;
 
int mat[100][100];
 
int main()
{
    int t,n,ans=0,tmp,i,j,c,r;
    scanf("%d",&t);
    while(t--)
    {
        
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
                scanf("%d",&mat[i][j]);
        }
        ans=mat[0][0];
        for(i=1;i<n;i++)
            mat[0][i]=mat[0][i]^mat[0][i-1];
        for(i=1;i<n;i++)
            mat[i][0]=mat[i][0]^mat[i-1][0];
         for(i=1;i<n;i++)
        {
            for(j=1;j<n;j++)
                {
                    mat[i][j]=mat[i][j]^mat[i-1][j]^mat[i][j-1]^mat[i-1][j-1];
                    if(mat[i][j]>ans)
                        ans=mat[i][j];
                }
        }
        for(j=1;j<n;j++)
        {
            for(r=0;r<n;r++)
                {
                    for(c=0;c+j<n;c++)
                    {
                        tmp=mat[r][c+j]^mat[r][j-1];
                        if(tmp>ans)
                            ans=tmp;
                    }
                }
        }
        for(i=1;i<n;i++)
        {
              for(r=0;r+i<n;r++)
                {
                    for(c=0;c<n;c++)
                    {
                        tmp=mat[r+i][c]^mat[i-1][c];
                        if(tmp>ans)
                            ans=tmp;
                    }
                }
        }
        for(i=1;i<n;i++)
        {
            for(j=1;j<n;j++)
            {
                for(r=0;r+i<n;r++)
                {
                    for(c=0;c+j<n;c++)
                    {
                        tmp=mat[r+i][c+j]^mat[i+r][j-1]^mat[i-1][c+j]^mat[i-1][j-1];
                        if(tmp>ans)
                            ans=tmp;
                    }
                }
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
