#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,j,k,l;
	long long int maxi,temp,count;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int a[n+5][n+5];
		long long int dp[n+5][n+5];
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				scanf("%d",&a[i][j]);	
			}
		}
		for(i=0;i<=n;i++)
		{
			dp[i][0]=0;
			dp[0][i]=0;
		}
		for(i=1;i<=n;i++)
		{
			dp[i][1]=a[i][1];
			for(j=2;j<=n;j++)
			{
				dp[i][j]=dp[i][j-1]^a[i][j];
			}
		}
		for(i=2;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				dp[i][j]=dp[i-1][j]^dp[i][j];
			}
		}
		maxi=0;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				for(k=0;k<j;k++)
				{
					count=dp[i][k];
					for(l=0;l<i;l++)
					{
						temp=count^dp[l][j]^dp[i][j]^dp[l][k];
						if(temp>maxi)
						maxi=temp;
					}
				}
			}
		}
		printf("%lld\n",maxi);
	}
}
