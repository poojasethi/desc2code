#include<bits/stdc++.h>
using namespace std;
#define sz 100000
#define pb(a) push_back(a)
#define ll long long
#define ull unsigned long long
#define fread freopen("input.txt","r",stdin)
#define fwrite freopen("output.txt","w",stdout)
#define inf (1<<29)
#define mem(abc,z) memset(abc,z,sizeof(abc))
#define PI acos(-1)
#define quick ios_base::sync_with_stdio(0)

int main(){
   int t,n;
   ll A[72][72],mxXor;
   cin>>t;
   while(t--){
     cin>>n;
     mem(A,0);
     for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            cin>>A[i][j];
            A[i][j]^=(A[i][j-1]^A[i-1][j]^A[i-1][j-1]);
        }
     }
     mxXor=0;
     for(int i=1;i<=n;i++){
        for(int j=1;j<=n;j++){
            for(int k=1;k<=i;k++){
                for(int l=1;l<=j;l++){
                    mxXor=max(mxXor,A[i][j]^A[i][l-1]^A[k-1][j]^A[k-1][l-1]);
                }
            }
        }
     }
     cout<<mxXor<<endl;
   }
   return 0;
}
