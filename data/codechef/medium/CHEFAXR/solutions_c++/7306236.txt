#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

ll inp[75][75];
ll temp[75];
ll ans;

ll solve(int n){
    ll sum = 0,maxSum = 0;
    for(int i=1;i<=n;i++){
        sum = 0;
        for(int j=i;j<=n;j++){
            sum ^= temp[j];
            if(sum > maxSum) maxSum = sum;
        }
    }
    for(int i=1;i<=n;i++){
        if(temp[i] > maxSum){
            maxSum = temp[i];
        }
    }
    return maxSum;
}

int main(){
    int test; scanf("%d",&test);
    while(test--){
        int n; scanf("%d",&n);
        ans = 0;
        for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++){
                scanf("%lld",&inp[i][j]);
                ans = max(ans,inp[i][j]);
            }
        }
        for(int i=1;i<=n;i++){
            for(int k=1;k<=n;k++) temp[k] = 0;
            for(int j=i;j<=n;j++){
                for(int k=1;k<=n;k++) temp[k] ^= inp[k][j];
                ans = max(ans,solve(n));
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
