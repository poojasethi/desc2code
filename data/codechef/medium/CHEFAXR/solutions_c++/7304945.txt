#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        long long prefix[75][75] = {};
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                long long temp;
                cin >> temp;
                prefix[i][j] = temp ^ prefix[i-1][j-1] ^ prefix[i-1][j] ^ prefix[i][j-1];
            }
        }
        long long maxXor = 0;
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                for (int i2 = 1; i2 <= i; i2++) {
                    for (int j2 = 1; j2 <= j; j2++) {
                        maxXor = max(maxXor, prefix[i][j] ^ prefix[i2-1][j2-1] ^ prefix[i2-1][j] ^ prefix[i][j2-1]);
                    }
                }
            }
        }
        cout << maxXor << endl;
    }

}
