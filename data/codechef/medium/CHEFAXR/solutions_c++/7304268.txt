#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		ll n,xor_max=0,k,res;
		cin>>n;
		ll a[n+1][n+1],ans[n+1][n+1];
		for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
		{
			cin>>a[i][j];
		}
		for(int i=0;i<=n;i++)
		ans[i][0]=0;
		for(int j=0;j<=n;j++)
		ans[0][j]=0;
		
		for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
		{
			ans[i][j]=a[i][j]^ans[i-1][j]^ans[i][j-1]^ans[i-1][j-1];
		}
		
		for(int x1=1;x1<=n;x1++)
		{
			for(int y1=1;y1<=n;y1++)
			{
				for(int x2=x1;x2<=n;x2++)
				{
				//	xor1=0;
					for(int y2=y1;y2<=n;y2++)
					{
						res=ans[x2][y2]^ans[x1-1][y2]^ans[x2][y1-1]^ans[x1-1][y1-1];
						xor_max=max(res,xor_max);
						
					}
				}
			}
		}
		cout<<xor_max<<endl;
	}
	return 0;
}
