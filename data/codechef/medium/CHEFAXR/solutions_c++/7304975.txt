#include <iostream>
#include <algorithm>
using namespace std;

const int MAX = 71;
int test, n, a[MAX][MAX], res, tmp;

int main(){
	ios_base::sync_with_stdio(0);
	cin >> test;
	while (test--){
		cin >> n;
		res = 0;
		for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= n; ++j){
			cin >> a[i][j];
			a[i][j] = a[i][j] ^ a[i - 1][j] ^ a[i][j - 1] ^ a[i - 1][j - 1];
			for (int k = 1; k <= i; ++k)
			for (int t = 1; t <= j; ++t)
				res = max(res, a[i][j] ^ a[k - 1][j] ^ a[i][t - 1] ^ a[k - 1][t - 1]);
		}
		cout << res << endl;
	}

	return 0;
}
