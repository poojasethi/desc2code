#include<bits/stdc++.h>
using namespace std;
int s[100][100],ar[100][100],t,ans,n;
int main()
{
    cin >> t;
    while(t--)  {
        cin >> n;
	for (int i=1;i<=n;i++)
	{
		for (int j=1;j<=n;j++)
		{
			cin>>ar[i][j];
		}
	}
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++)
			s[i][j]=(s[i-1][j]^s[i][j-1]^s[i-1][j-1]^ar[i][j]);

	 ans=0;
	for (int x1=1;x1<=n;x1++)
		for (int y1=1;y1<=n;y1++)
			for (int x2=x1;x2<=n;x2++)
				for (int y2=y1;y2<=n;y2++)
					ans=max(ans,s[x2][y2]^s[x1-1][y2]^s[x2][y1-1]^s[x1-1][y1-1]);
	cout << ans <<endl;

}
return 0;
    }
