#define scs(x) scanf("%s",x);
#define print(x) printf("%d",x);
#define nl printf("\n")

#define F(i,a,n) for(int i=a;i<n;i++)
#define INF 100000000000000000LL
#define LL long long
const int N=71;
int dp[N][N];
int main() {
    int t;sci(t);
    while(t--) {
        int n;sci(n);
        M(dp,0);
        F(i,1,n+1) {
            F(j,1,n+1) {
                sci(dp[i][j]);
                    dp[i][j]^=dp[i-1][j];
                    dp[i][j]^=dp[i][j-1];
                    dp[i][j]^=dp[i-1][j-1];
            }
        }
        int mx=0;
        int ans=0;
        F(i,1,n+1) {
            F(j,1,n+1) {
                F(k,i,n+1) {
                    F(l,j,n+1) {
                        mx=dp[k][l];
                            mx^=dp[i-1][l];
                            mx=mx^dp[k][j-1];
                            mx^=dp[i-1][j-1];
                        ans=max(mx,ans);
                    }
                }
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
