#include<stdio.h>
#include<string.h>
typedef long long ll;
int main()
{
	ll t,i,n,max;
	ll ar[75][75];
	ll dp[75][75];
	ll j;
	memset(ar,0,sizeof(ar));
	memset(dp,0,sizeof(dp));
	scanf("%lld",&t);
	while(t--)
	{
		max=-1;
		scanf("%lld",&n);
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				scanf("%lld",&ar[i][j]);
			}
		}
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
	            {
				dp[i][j]=dp[i][j-1]^dp[i-1][j]^dp[i-1][j-1]^ar[i][j];
			}
		}
		ll k,l;
		ll ans;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				for(k=1;k<=n;k++)
				{
					for(l=1;l<=n;l++)
					{
						ans=dp[k][l]^dp[k][j-1]^dp[i-1][j-1]^dp[i-1][l];
					if(ans>max)
					max=ans;
				}
				}
			}
		}
		printf("%lld\n",max);
	}
	return 0;
}
