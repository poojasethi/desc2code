
#include <iostream>
using namespace std;

#define llong long long

llong inp[71][71];
llong xro[71][71];

void fillxro(llong n){
	llong i=0,j;
	while(i<=n){
		xro[0][i]=0;
		xro[i][0]=0;
		i++;
	}
	i=1,j=1;
	while(i<=n){
		j=1;
		while(j<=n){
			xro[i][j]=(xro[i-1][j-1]^xro[i][j-1])^(xro[i-1][j]^inp[i][j]);
			j++;
		}
		i++;
	}
}

int main() {
	llong t,i=0,n,k,l;
	cin>>t;
	while(i<t){
		cin>>n;
		k=1,l=1;
		while(k<=n){
			l=1;
			while(l<=n){
				cin>>inp[k][l];
				l++;
			}
			k++;
		}
		fillxro(n);
		llong mx=0,startr=1,endr,startc=1,endc,temp;
		while(startr<=n){
			endr=startr;
			while(endr<=n){
				startc=1;
				while(startc<=n){
					endc=startc;
					while(endc<=n){
						temp=(xro[startr-1][startc-1]^xro[endr][endc])^(xro[startr-1][endc]^xro[endr][startc-1]);
						//cout<<" "<<temp;
						if (temp>mx){
							mx=temp;
						}
						endc++;
					}
					startc++;
				}
				endr++;
			}
			startr++;
		}
		cout<<"\n"<<mx;
		i++;

	}
	return 0;
}
