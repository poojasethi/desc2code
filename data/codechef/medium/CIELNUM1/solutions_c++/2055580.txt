using namespace std;

char digits[16],*ptr1,*ptr2,*ptrlast;

int main(){
  int i=0,threes=0,fives=1,eights=0;

  for(i=0;i<14;i++) digits[i]='3';
  digits[14]='5';
  digits[15]=0;
  ptr2=ptrlast=digits+14;
  i=0;
  while(i<50000){
    ptr1=ptrlast;
    while(true){
      if(*ptr1=='3'){
        *ptr1='5';
        threes--;
        fives++;
        break;
      }
      else if(*ptr1=='5'){
        *ptr1='8';
        fives--;
        eights++;
        break;
      }
      else{
        *ptr1--='3';
        eights--;
        threes++;
        if(ptr1<ptr2){
          ptr2=ptr1;
          threes++;
          break;
        }
      }
    }
    if(eights>=fives && fives>=threes){
      puts(ptr2);
      i++;
    }
  }
  return 0;
}
