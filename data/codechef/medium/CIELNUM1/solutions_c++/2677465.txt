#include <vector>
#include <algorithm>
#include <iostream>

typedef long long LL;

using namespace std;

int x,y,z;
LL n1;
vector<LL> v;

void pre(int x,int y,int z,LL n) {
  if(x>=y&&y>=z&&n<=85500000000LL) v.push_back(n);
  if(n>85500000000LL) return;
  LL n1=10*n + 8;
  pre(x+1,y,z,n1);
  n1=10*n + 5;
  pre(x,y+1,z,n1);
  n1=10*n + 3;
  pre(x,y,z+1,n1);
  return ;
}
 
 
int main() {
  pre(0,0,0,0);
  sort(v.begin(),v.end());
  // cout << v.size() << endl;
  // loop is from 1 because even 0 is inserted
  for(int i=1;i<50001;++i) cout<<v[i]<<"\n";
  return 0;
}
