#include<iostream>
using namespace std;
char s[100];
int i,count=0;
void gen(int n,int d3,int d5,int d8)
{
	if(count==50000)return;

	if(n==i)
	{       if(d8>=d5 && d5>=d3)
		{
			cout<<s<<"\n";
			count++;
		}
		return;
	}
	s[n]='3';
	gen(n+1,d3+1,d5,d8);
	s[n]='5';
	gen(n+1,d3,d5+1,d8);
	s[n]='8';
	gen(n+1,d3,d5,d8+1);
}
int main()
{
	for(i=1;i<100;i++)
	{
		gen(0,0,0,0);
	}
}
