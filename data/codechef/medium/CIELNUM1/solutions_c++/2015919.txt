#include<iostream>
#include<stdio.h>
using namespace std;
#define MAX 50000
char number[150];
int CNT = 0;
int Level = 0 ;
void tree(int n,int n3,int n5,int n8)
{
if(CNT == MAX) return;
if ( n == Level )
{	
if(n3 <= n5 && n5 <= n8 && CNT++ < MAX)
printf("%s\n", number);
return;
}
number[n] = '3'; tree(n+1, n3+1, n5, n8);
number[n] = '5'; tree(n+1, n3, n5+1, n8);
number[n] = '8'; tree(n+1, n3, n5, n8+1);
}
 
int main()
{
while(CNT != MAX )
{
Level++;
tree(0,0,0,0);
}
return(0);
}
