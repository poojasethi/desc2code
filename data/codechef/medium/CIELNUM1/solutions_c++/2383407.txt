#include <stdio.h>

int rem;
char cur[] = "00000000000005", inc[100], freq[100];
char * pr = cur+13;
inline void next(){
  do {
    int i = 13;
    while(cur[i] == '8') cur[i--] = '3';
    if (cur[i] == '0') pr--;
    freq[cur[i]]--;
    freq[inc[cur[i]]]++;
    freq['8'] -= 13 - i;
    freq['3'] += 13 - i;
    cur[i] = inc[cur[i++]];
  } while (freq['3'] > freq['5'] || freq['5'] > freq['8']); 
  puts(pr);
}

int main(){
  inc['0'] = '3';
  inc['3'] = '5';
  inc['5'] = '8';
  freq['5'] = 1;
  freq['3'] = 0;
  freq['8'] = 0;
  for (rem = 50000; rem; rem--) next();
  return 0;
}
