#include<iostream>
using namespace std;
char s[100];
int i,cnt=0;
void gen(int n,int n3,int n5,int n8)
{
	if(cnt==50000)return;
	
	if(n==i)
	{       if(n8>=n5 && n5>=n3)
		{
			cout<<s<<"\n";
			cnt++;
		}
		return;
	}	
	s[n]='3';
	gen(n+1,n3+1,n5,n8);	
	s[n]='5';
	gen(n+1,n3,n5+1,n8);
	s[n]='8';
	gen(n+1,n3,n5,n8+1);
}
main()
{
	for(i=1;i<100;i++)
	{
		gen(0,0,0,0);
	}
}
