#include <iostream>
#include <stdio.h>
#include <string.h>
#include <limits.h>
using namespace std;
#define getcx getchar_unlocked
inline void inp( int &n )//fast input function
{
   n=0;
   int ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   n=n*sign;
}

static int count =0;


bool recursive(char* s, int three, int five, int eight, int height, int req) {
    
    if(count >= 50000)
        return true;
    if(height == req) {
        if(eight>= five && five >=three) {
            s[height] = '\0';
            printf("%s\n",s);
            count++;
        }
        return true;
    }
    
    s[height] = '3';
    recursive(s, three+1, five, eight, height+1, req );
    s[height] = '5';
    recursive(s, three, five+1, eight, height+1, req);
    s[height] = '8';
    recursive(s, three, five, eight+1, height+1, req);
    
    return true;
}

int main()
{
    char s[25];
     for(int height =1; height < 25; height++) {
         recursive(s, 0, 0,0, 0, height);
         if(count >= 50000)
             break;
     }
    return 0;
}
