#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

const int LIM = 100000;

typedef long long ll;

ll ans[LIM];
int nn = 0;
int cnt=0;

void make_num(int n3, int n5, int n8) {
    vector<int> a;
    for (int i=0; i<n3; i++) a.push_back(3);
    for (int i=0; i<n5; i++) a.push_back(5);
    for (int i=0; i<n8; i++) a.push_back(8);
    ll n=0;
    for (int i=0; i<a.size(); i++) {
        n = n*10+a[i];
    }
    ans[nn++] = n;
    while (next_permutation(a.begin(), a.end())) {
        n=0;
        for (int i=0; i<a.size(); i++) {
            n = n*10+a[i];
        }
        ans[nn++] = n;
    }
}

void generate(int d) {
    for (int n3=d/3; n3>=0; n3--) {
        int rem = d-n3;
        for (int n5=rem/2; n5>=n3; n5--) {
            int n8 = d-n5-n3;
            make_num(n3, n5, n8);
        }
    }
}

int main()
{
    //freopen("ciel.txt", "w", stdout);
    for (int i=1; i<12; i++) {
        generate(i);
    }
    sort(ans, ans+nn);
    //printf("%d\n", nn);
    int TEST = 20;
    int SOLU = 50000;
    for (int i=0; i<SOLU; i++) printf("%lld\n", ans[i]);
    return 0;
}
