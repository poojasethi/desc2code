#include <cstdio>
using namespace std;
char s[15];
int i,cnt;
void ciel(int n,int a,int b,int c)
{
  if(cnt==50000) return;
  if(n==i)
  {
    if(c>=b && b>=a)
    {
      ++cnt;
      printf("%s\n",s);
    }
    return;
  }
  s[n]='3';
  ciel(n+1,a+1,b,c);
  s[n]='5';
  ciel(n+1,a,b+1,c);
  s[n]='8';
  ciel(n+1,a,b,c+1);
}
int main()
{
  for(i=1;i<15;i++)
  {
    ciel(0,0,0,0);
  }
  return 0;
}
