#define MAX 50000
int scount;

void recurse(string str, int len, int c3, int c5, int c8)
{
	if (scount >= MAX)
		return;
	if (len == 0)
	{
		if (c8 >= c5 && c5 >= c3)
		{
			printf("%s\n", str.c_str());
			scount++;
		}
	}
	else
	{
		recurse(str+"3", len-1, c3+1, c5, c8);
		recurse(str+"5", len-1, c3, c5+1, c8);
		recurse(str+"8", len-1, c3, c5, c8+1);
	}
}

int main()
{
	scount = 0;
	recurse("", 1, 0, 0, 0);

	//printf("%d\n", scount);
	return 0;
}
