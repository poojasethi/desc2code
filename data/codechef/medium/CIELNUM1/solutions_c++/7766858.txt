#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

#define N 50000
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int i,d3,d5,d8,cnt=0;
    queue < string > q;
    q.push("");
    while (!q.empty() && cnt < N) {
        string t = q.front();
        q.pop();
        d3=d5=d8=0;
        for(i=0;i<(int)t.size();i++) {
            if(t[i]=='3') d3++;
            if(t[i]=='5') d5++;
            if(t[i]=='8') d8++;
        }
        if(!(d3==0 && d5==0 && d8==0) && d8>=d5 && d5>=d3) {
            cnt++;
            cout<<t<<'\n';
        }
        q.push(t+"3");
        q.push(t+"5");
        q.push(t+"8");
    }
    return 0;
}
