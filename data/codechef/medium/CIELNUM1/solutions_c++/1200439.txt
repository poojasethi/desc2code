#include<stdio.h>
char c[26];
int cnt3=1,cnt5=0,cnt8=0;
int lim=24;

void ciel( )
{
     int i;
     harekrishna : for(i=25;i>=lim;i--)
                  {
                  if(c[i]=='3')
                  {
                   c[i]='5';
                   cnt3--;
                   cnt5++;
                   break;
                  }
                  if(c[i]=='5')
                  {
                   c[i]='8';
                   cnt5--;
                   cnt8++;
                   break;
                  }
                  if(c[i]=='8')
                  {
                   c[i]='3';
                   cnt8--;
                   cnt3++;
                  }
                  }
     if(i==(lim-1))
     {
     c[i]='3';
     cnt3++;
     lim--;              
     }
     if(!((cnt3<=cnt5)&&(cnt5<=cnt8)))
     goto harekrishna;
}
int main()
{
    int n=50000;
    c[lim]='3';
c[lim+1]='\0';
    while(n)
    {
            ciel();
            printf("%s\n",&c[lim]);
            n--; 
    }
    return 0;
}
