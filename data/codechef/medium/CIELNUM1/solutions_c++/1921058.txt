#include <iostream>
#include <cstdio>

using namespace std;

unsigned long long int res[1000001], n3[1000001], n5[1000001], n8[1000001], ar[1000001];

int main()
{
    unsigned long long int ress = 0, i;
    unsigned long long int cal = 0, val = 1;
    n3[0] = 0;
    n5[0] = 0;
    n8[0] = 0;
    ar[0] = 0;
    while(ress < 50000) {
        ar[val] = ar[cal] * 10 + 3;
        n3[val] = n3[cal] + 1;
        n5[val] = n5[cal];
        n8[val] = n8[cal];
        //cout << ar[val] << endl;
        if(n3[val] <= n5[val] && n5[val] <= n8[val]) {
            res[ress++] = ar[val];
        }
        val++;
        ar[val] = ar[cal] * 10 + 5;
        n3[val] = n3[cal];
        n5[val] = n5[cal] + 1;
        n8[val] = n8[cal];
      //  cout << ar[val] << endl;
        if(n3[val] <= n5[val] && n5[val] <= n8[val]) {
            res[ress++] = ar[val];
        }
        val++;
        ar[val] = ar[cal] * 10 + 8;
        n3[val] = n3[cal];
        n5[val] = n5[cal];
        n8[val] = n8[cal] + 1;
      //  cout << ar[val] << endl;
        if(n3[val] <= n5[val] && n5[val] <= n8[val]) {
            res[ress++] = ar[val];
        }
        val++;
        cal++;
    }

    for(i = 0; i < 50000; i++) {
        printf("%lld\n", res[i]);
    }
}
