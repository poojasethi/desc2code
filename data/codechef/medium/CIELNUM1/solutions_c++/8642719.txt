#include <bits/stdc++.h>
#define lli long long int

using namespace std;
bool check(lli a)
{
    int cnt3=0,cnt5=0,cnt8=0;

    while(a > 0)
    {
        int b=a%10;
        if(b==3)cnt3++;
        else if(b==5)cnt5++;
        else cnt8++;
        a/=10;
    }
    bool ok=false;
    if(cnt8 >= cnt3 && cnt5 >= cnt3 &&  cnt8>= cnt5)ok=true;

    if(ok)return true;
    else return false;
}

int main()
{
    lli t,i,j,k,p,q,a;
    vector<lli> vec;
    int ans=0;
    p=0;
    q=3;
    vec.push_back(3);
    vec.push_back(5);
    vec.push_back(8);
    while(1)
    {
        for(i=p; i<q; i++)
        {
            a=vec[i];
            vec.push_back(a*10 +3);
            vec.push_back(a*10 +5);
            vec.push_back(a*10 +8);
        }
        p=q;
        q=vec.size();
        if(vec.size() > 6*50000)break;
    }

    sort(vec.begin(),vec.end());
    for(i=0; i<vec.size() && ans<50000; i++)
    {
        if(check(vec[i]))
        {
            printf("%lld\n",vec[i]);
            ans++;
        }
    }
}
