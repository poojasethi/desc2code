#include<bits/stdc++.h>
using namespace std;
main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		long long x=0,ans,fds[100010]={0},a[100010],bks[100010]={0};
		int i,n,k;
		scanf("%d%d",&n,&k);
		for(i=1;i<=n;i++)
		scanf("%lld",&a[i]);
		fds[1]=a[1],bks[n]=a[n],x=a[1];
		for(i=2;i<=n;i++)
		{
			x=max(a[i],x+a[i]);
			fds[i]=max(fds[i-1],x);
		}
		x=a[n];
		for(i=n-1;i>=1;i--)
		{
			x=max(a[i],x+a[i]);
			bks[i]=max(bks[i+1],x);
		}
		ans=-1e18;
		for(i=k+2;i<=n;i++)
		{
			ans=max(ans,fds[i-k-1]+bks[i]);
		}
		printf("%lld\n",ans);
	}
}
