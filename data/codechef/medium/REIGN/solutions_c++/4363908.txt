#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#include<climits>
using namespace std;
int main()
{
   long long int p=-100000000;
   long long int q=100000000;
   long long int t,dpl[100010],dpr[100010],a[100010],sum,sumNv,n,k,temp,flag;
   long long int nvsf,pvsf;
   scanf("%lld",&t);
   while(t--)
   {
      temp=p*q;;
      sum=0;sumNv=0;flag=0;
      nvsf=p*q;
      scanf("%lld%lld",&n,&k);
      for(int i=1;i<=n;i++)
      scanf("%lld",&a[i]);
      for(int i=1;i<=n;i++)
      {
        if(sum+a[i]>nvsf)
        nvsf=sum+a[i];
        dpl[i]=nvsf;
        if(sum+a[i]>0)
        sum=sum+a[i];
        else
        sum=0;
      }
      sum=0;sumNv=0;flag=0;
      nvsf=p*q;
      for(int i=n;i>=1;i--)
      {
      if(sum+a[i]>nvsf)
        nvsf=sum+a[i];
        dpr[i]=nvsf;
        if(sum+a[i]>0)
        sum=sum+a[i];
        else
        sum=0;
      }
     for(int i=1;i<=(n-k-1);i++)
     {
       if(dpl[i]+dpr[i+k+1]>temp)
       temp=dpl[i]+dpr[i+k+1];
     }
     printf("%lld\n",temp);
   }
   return 0;
}
