/**
Input:
3
5 3
-1 1 2 3 -1
8 3
5 5 -1 -2 3 -1 2 -2
6 0
5 -1 5 0 -1 9

OUTPUT:
-2
12
18
*/

#include <iostream>
#include <cstdio>
#define N 100005
using namespace std;

int a[N];
long long dpl[N];
long long dpr[N];

long long solve(int a[],int n,int k)
{
	int i,j;
	long long maxx,cur;
	long long ans=0;
	for(i=0;i<n;i++){
		dpl[i]=0;
		dpr[i]=0;
	}
	maxx=a[0];
	cur=0;
	dpl[0]=maxx;
	for(i=0;i<n;i++)
	{
		cur+=a[i];
		if(cur>maxx)
			maxx=cur;
		dpl[i]=maxx;
		if(cur<0)
			cur=0;
	}
	maxx=a[n-1];
	cur=0;
	dpr[n-1]=maxx;
	for(i=n-1;i>=0;i--)
	{
		cur+=a[i];
		if(cur>maxx)
			maxx=cur;
		dpr[i]=maxx;
		if(cur<0)
			cur=0;
	}
	ans=dpl[n-k-2]+dpr[n-1];
	for(i=n-k-3;i>=0;i--)
	{
		if(dpl[i]+dpr[i+k+1]>ans)
			ans=dpl[i]+dpr[i+k+1];
	}
	return ans;
}

int main()
{
	int i,j,k,t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		printf("%lld\n",solve(a,n,k));
	}
	return 0;
}
