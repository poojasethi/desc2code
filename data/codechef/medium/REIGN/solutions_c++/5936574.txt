#include<iostream>
#include<cstdio>
#include<climits>
typedef long long int ll;
ll i,j,k,n,t,a[100500],b[100500],c[100500],s,ans;
 
ll max(ll p,ll q)
{
	if(p>q) return p;
	return q;
}
int main()
{
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld",&n,&k);
		for(i=0;i<n;i++) scanf("%lld",&a[i]);
		
		b[0]=s=a[0];
		for(i=1;i<n;i++)
		{
			if(s+a[i]>a[i])
			s=s+a[i];
			else 
			s=a[i];
			
			b[i]=max(b[i-1],s);
		}
		/*for(i=0;i<n;i++) printf("%lld ",b[i]);
		printf("\n");
		*/
		s=c[n-1]=a[n-1];
		for(i=n-2;i>=0;i--)
		{
			if(s+a[i]>a[i])
			s=s+a[i];
			else 
			s=a[i];
			
			c[i]=max(c[i+1],s);
		}
		/*for(i=n-1;i>=0;i--) printf("%lld ",c[i]);
		printf("\n");
		*/
		ans=LONG_MIN;
		//printf("ans is %lld\n",ans);
		for(i=0;i<n-k-1;i++)
		{
			//printf("b[i],c[i+k+1] %lld %lld\n",b[i],c[i+k+1]);
			ans=max(ans,b[i]+c[i+k+1]);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
