using namespace std;

/* MAIN PROGRAM STARTS HERE */

LL A[100005];

int main()
	{
	//freopen("TC1.txt","r",stdin);
	//freopen("output1","w",stdout);
	
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	
	LL T,n,i,j,k,l,m,x,y,z,p;
	
	cin>>T;
	
	while (T--)
		{
		cin>>n>>k;
		REP(i,n) cin>>A[i];
		
		x=0; y=-INF; bool flag=false;
		
		LL dp[n][2];
		
		REP(i,n)
			{
			x+=A[i];
			if (A[i]>=0) flag=true;
			if (x<0) x=0;
			if (!flag) y=max(y,A[i]);
			else y=max(x,y);
			dp[i][0]=y;
			}
			
		x=0; y=-INF; flag=false;
		for (i=n-1; i>=0; i--)
			{
			x+=A[i];
			if (A[i]>=0) flag=true;
			if (x<0) x=0;
			if (!flag) y=max(y,A[i]);
			else y=max(x,y);
			dp[i][1]=y;
			}
		
		LL ans=-INF;
		for (i=0; i+k+1<n; i++)
			{
			ans=max(ans,dp[i][0]+dp[i+k+1][1]);
			}
		cout<<ans<<"\n";
		
		}
			
	return 0;
	}
