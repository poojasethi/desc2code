#include <bits/stdc++.h>
using namespace std;
#define bitcnt __builtin_popcount
#define ln length()
#define rep(i, n) for(int i = 0; i < n; i++)
#define rrep(i, n) for(int i = n - 1; i >= 0; i--)
#define all(a)  a.begin(),a.end()
#define LL long long
#define limit 100005
#define sl(n) scanf("%lld", &n)
#define s(n)  scanf("%d", &n)
#define pl(n) printf("%lld", n)
#define pi(n)  printf("%d", n)

void print(LL *arr, LL n)
{
	rep(i, n) cout << arr[i] << " ";
	cout << endl;	
}

LL arr[limit];
LL forward[limit];
LL backward[limit];

void solve()
{
	LL n, k;
	sl(n); sl(k);
	rep(i, n) sl(arr[i]);
	LL finalmax = INT_MIN;
	LL curmax = 0;
	rep(i, n) {
		curmax = max(curmax + arr[i], arr[i]);
		finalmax = max(finalmax, curmax);
		forward[i] = finalmax;
	}	
	curmax = 0;
	finalmax = INT_MIN;
	rrep(i, n) {
		curmax = max(curmax + arr[i], arr[i]);
		finalmax = max(finalmax, curmax);
		backward[i] = finalmax;
	}
	LL ans = INT_MIN;
	rep(i, n - k - 1) {
		ans = max(ans, forward[i] + backward[i + k + 1]);
	}
	pl(ans);
	printf("\n");	
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		solve();
	}
	return 0;
}
