#include <bits/stdc++.h>

using namespace std;

#define pb push_back

int main()
{
    long long int t, n, a[111111], i, mx, sum, k;
    scanf("%lld", &t);
    while (t--) {
        scanf("%lld%lld", &n, &k);
        for (i = 1; i <= n; i++)
            scanf("%lld", &a[i]);
        vector <long long int> l, r;
        sum = 0;
        mx = LONG_LONG_MIN;
        for (i = 1; i <= (n-k-1); i++) {
            sum = (sum+a[i]) > a[i] ? (sum+a[i]) : a[i];
            mx = mx > sum ? mx : sum;
            l.pb(mx);
        }
        sum = 0;
        mx = LONG_LONG_MIN;
        for (i = n; i >= (k+2); i--) {
            sum = (sum+a[i]) > a[i] ? (sum+a[i]) : a[i];
            mx = mx > sum ? mx : sum;
            r.pb(mx);
        }
        mx = LONG_LONG_MIN;
        for (i = 0; i < l.size(); i++)
            mx = mx > (l[i]+r[r.size()-1-i]) ? mx : (l[i]+r[r.size()-1-i]);
        printf("%lld\n", mx);
    }
    return 0;
}
