/* Reference Editorial */
#include <bits/stdc++.h>

typedef long long lld;
typedef unsigned long long llu;
#define rep(i,x,y) for(i=x;i<y;i++)
#define rrep(i,x,y) for(i=x;i>=y;i--)
#define trv(y,x) for(typeof(x.begin())y=x.begin();y!=x.end();y++)
#define _max(a,b,c) (max(max(a,b),c))
using namespace std;

lld *A,*MaxLeft,*MaxRight,*MaxK;

int main()
{
    ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t,n,i,k;
	#ifndef ONLINE_JUDGE
        freopen("in.txt","r",stdin);
	#endif
	lld Ans,MAX;
	cin>>t;
	while(t--)
    {
        cin>>n>>k;
        A=new lld[n];
        MaxLeft=new lld[n];
        MaxRight=new lld[n];
        MaxK=new lld[n];

        rep(i,0,n) cin>>A[i];

        MaxLeft[0]=A[0];
        rep(i,1,n){
        MaxLeft[i]=max(MaxLeft[i-1] + A[i], A[i]);
        }

        MaxRight[n-1]=A[n-1];
        rrep(i,n-2,0){
        MaxRight[i]=max(MaxRight[i+1] + A[i], A[i]);
        }

        MaxK[n-1]=MAX=A[n-1];
        rrep(i,n-2,0){
        if(MaxRight[i]>MAX) MAX=MaxRight[i];
        MaxK[i] = MAX;
        }

        Ans=MaxLeft[0] + MaxK[k+1];
        rep(i,1,n-k-1){
        if(MaxLeft[i]+MaxK[i+k+1] > Ans)
            Ans=MaxLeft[i]+MaxK[i+k+1];
        }

        cout<<Ans<<"\n";
        delete[] A;
        delete[] MaxLeft;
        delete[] MaxRight;
        delete[] MaxK;
    }
	return 0;
}
