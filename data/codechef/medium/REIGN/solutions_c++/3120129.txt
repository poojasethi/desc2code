#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<string>
#include<cstring>
#include<vector>
using namespace std;
long long int ar[100010], left_arr[100010], right_arr[100010];
int main() {
	int t;
	scanf("%d", &t);
	while (t--) {
		long long int n, k;
		scanf("%lld %lld", &n, &k);
		for (int i = 0; i < n; i++)
			scanf("%lld", &ar[i]);
		{
			long long int sum_left = ar[0];
			long long int left_minim = (ar[0] < 0 ? ar[0] : 0);
			left_arr[0] = ar[0];
			for (int i = 1; i < n; i++) {
				sum_left += ar[i];
				left_arr[i] = sum_left - left_minim;
				left_minim = min(left_minim, sum_left);
			}
			for (int i = 1; i < n; i++) {
				left_arr[i] = max(left_arr[i - 1], left_arr[i]);
			}
		}

		{
			long long int sum_right = ar[n - 1];
			long long int right_minim = (ar[n - 1] < 0 ? ar[n - 1] : 0);
			right_arr[n - 1] = ar[n - 1];
			for (int i = n - 2; i >= 0; i--) {
				sum_right += ar[i];
				right_arr[i] = sum_right - right_minim;
				right_minim = min(right_minim, sum_right);
			}
			for (int i = n - 2; i >= 0; i--) {
				right_arr[i] = max(right_arr[i + 1], right_arr[i]);
			}
		}

		long long int ans = -2000000002;
		for (int i = 0; i < n - k - 1; i++) {
			ans = max(ans, left_arr[i] + right_arr[i + k + 1]);
		}

		cout << ans << "\n";

//		for (int i = 0; i < n; i++)
//			cout << left_arr[i] << ", ";
//		cout << "\n";
//		for (int i = 0; i < n; i++)
//			cout << right_arr[i] << ", ";
//		cout << "\n";
	}
	return 0;
}
