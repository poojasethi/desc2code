struct chance
{
	int num;
	string name;
};

bool compare(const chance& l, const chance& r)
{
	return l.num < r.num;
}

int main()
{
	int t;
	si(t);

	while (t--) {
		int n;
		si(n);

		vector<chance> v(n);

		F(i, 0, n)
			cin >> v[i].name >> v[i].num;

		sort(v.begin(), v.end(), compare);

		int i=0, flag = 0;
		string ans;
		while (i<n) {
			int j=i+1;
			while (j<n && (v[i].num==v[j].num)) {
				j++;
			}

			if (j == i+1) {
				flag=1;
				ans = v[i].name;
				break;
			}
			i = j;
		}

		if (flag == 1) 
			cout << ans << endl;
		else 
			cout << "Nobody wins." << endl;
	}

	return 0;
}
