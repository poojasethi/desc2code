#include<iostream>
#include<map>
#include<stack>
using namespace std;

typedef long long int ll;

int main(){
    int t,n;
    string str;
    ll no;
    map<ll,string> m;
    stack<ll> st;
    cin>>t;
    while(t--){
        cin>>n;
        m.clear();
        for(int i=0;i<n;i++){
            cin>>str;
            cin>>no;
            if(!m.count(no)){
                m.insert(make_pair(no,str));
            }else{
                st.push(no);
            }
        }

        while(!st.empty()){
            m.erase(st.top());
            st.pop();
        }

        if(!m.empty())cout<<m.begin()->second<<endl;
        else cout<<"Nobody wins."<<endl;
    }
    return 0;
}
