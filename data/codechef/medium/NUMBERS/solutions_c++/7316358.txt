#include<bits/stdc++.h>
using namespace std;

map<int, int> score;
map<int, string> save;

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n;
        scanf("%d", &n);
        for(int i=0;i<n;i++) {
            string str;
            int num;
            cin>>str;
            cin>>num;
            save[num]=str;
            score[num]++;
        }
        map<int, int> :: iterator it;
        bool ok=false;
        int res=0;
        for(it=score.begin();it!=score.end();it++) {
            if(it->second==1) {
                res=it->first;
                ok=true;
                break;
            }
        }
        if(ok) {
            cout<<save[res]<<endl;
        } else {
            puts("Nobody wins.");
        }
        save.clear();
        score.clear();
    }
}
