#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		map<int,char*> mp;
		int i;
		for(i=0;i<n;i++)
		{
			char *c=new char[100];
			int num;
			scanf("%s %d",c,&num);
			if(mp.find(num)==mp.end())
			{
				mp[num]=c;
			}
			else
				mp[num]=NULL;
		}
		map<int,char*>::iterator it;
		for (it=mp.begin(); it!=mp.end(); ++it)
		{
			if(it->second!=NULL)
				break;
		}
		if(it!=mp.end())
			printf("%s\n", it->second);
		else
			printf("Nobody wins.\n");
	}
	return 0;
}
