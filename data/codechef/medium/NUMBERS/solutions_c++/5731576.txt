#include<bits/stdc++.h>
using namespace std;

int main() {
     int tc;
     cin>>tc;
     while(tc--) {
          int n;
          cin>>n;
          map<int,string> name;
          map<int,int> count;
          string str;
          int a;
          for(int i=0;i<n;i++) {
               cin>>str>>a;
               name[a] = str;
               count[a]++;
          }
          int flag = 0;
          map<int,int>::iterator it;
          for(it=count.begin();it!=count.end();it++) {
               if(it->second == 1) {
                    cout<<name[it->first]<<endl;
                    flag = 1;
                    break;
               }
          }
          if(!flag)
               cout<<"Nobody wins."<<endl;
     }
     return 0;
}
