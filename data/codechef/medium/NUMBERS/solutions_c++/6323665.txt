    #define pc putchar

    #define gc getchar_unlocked
    #define pc putchar_unlocked

using namespace std;

void scanint(int &x) {
    register int c = gc();
    x = 0;
    for(; (c < 48 || c > 57); c = gc());
    for(; c > 47 && c < 58; c = gc()) {
        x = (x << 1) + (x << 3) + c - 48;
    }
}

int function(const void *a, const void *b) {
    return *(int *)a - *(int *)b;
}

int main() {
    int t, n, x;
    string s, res;
    pair<int, string> array[10000];
    scanint(t);
    while(t --) {
        scanint(n);
        for(int i = 0; i < n; i ++) {
            cin >> s;
            scanint(x);
            array[i] = make_pair(x, s);
        }
        sort(array, array + n);
        int i;
        if(n == 1) {
            res = array[0].second;
        } else if(n == 2) {
            if(array[0].first != array[1].first) {
                res = array[0].second;
            } else {
                res = "Nobody wins.";
            }
        } else if(n >= 3) {
            if(array[0].first != array[1].first) {
                res = array[0].second;
            } else {
                for(i = 1; i < n - 1; i ++) {
                    if(array[i].first != array[i - 1].first && array[i].first != array[i + 1].first) {
                        res = array[i].second;
                        break;
                    }
                }
                if(i == n - 1) {
                    if(array[n - 2].first != array[n - 1].first) {
                        res = array[n - 1].second;
                    } else {
                        res = "Nobody wins.";
                    }
                }
            }
        }
        cout << res << endl;
    }
    return 0;
}
