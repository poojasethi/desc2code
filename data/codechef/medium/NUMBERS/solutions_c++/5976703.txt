#include<bits/stdc++.h>
using namespace std;
typedef struct 
{
	int x;
	char s[10];
}player;
int cmp(const void* a,const void* b)
{
	return ((player*)a)->x-((player*)b)->x;
}
int main()
{
	int t;
	scanf("%d",&t);
	player pl[10001];
	int i,j,n;
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;++i)
			scanf("%s%d",pl[i].s,&pl[i].x);
		qsort(pl,n,sizeof(player),cmp);
		int c=0,p=0;
		int ind;
		for(i=0;i<n;++i)
		{
			if(pl[i].x!=p)
			{
				if(c==1)break;
				p=pl[i].x;
				c=1;
				ind=i;
			}
			else
			{
				++c;
			}

		}
		if(c==1)
			cout<<pl[ind].s<<"\n";
		else cout<<"Nobody wins.\n";
	}
	return 0;
}
