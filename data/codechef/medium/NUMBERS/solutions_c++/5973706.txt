#include <iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<map>
#define ll long long int
using namespace std;
char ch[10001][7];
ll in(){
    ll r=0,c;for(c=getchar_unlocked();
c<=32;c=getchar_unlocked());
if(c=='-')
    return -in();
for(;c>32;r=(r<<1)+(r<<3)+c-'0',c=getchar_unlocked());return r;}
int main() {
	ll t,n,i,a,b;
	t=in();
	int flag=0;
	map<ll,pair<ll,ll> >M;
	while(t>0)
    {
        t--;
        n=in();
        pair<ll,ll>temp;
        for(i=0;i<n;i++)
        {
            scanf("%s %lld",&ch[i],&a);
            temp=M[a];
            temp.first=temp.first+1;
            temp.second=i;
            M[a]=temp;
        }
        map<ll,pair<ll,ll> >::iterator it;
        for(it=M.begin();it!=M.end();it++)
        {
            if(it->second.first==1)
            {
                b=it->second.second;
                printf("%s\n",ch[b]);
                flag=1;
                break;
            }
        }
        if(flag)
            {
                flag=0;
            }
            else
                printf("Nobody wins.\n");
                M.clear();
    }

}
