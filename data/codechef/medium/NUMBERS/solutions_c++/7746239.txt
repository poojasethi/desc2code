#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		map<ll,ll>m1;
		map<ll,string>m2;
		vector<ll>v;
		for(int i=0;i<n;i++)
		{
			ll x;
			string s;
			cin>>s>>x;
			m1[x]++;
			m2[x]=s;
			v.push_back(x);
		}
		sort(v.begin(),v.end());
		bool flag=false;
		int i;
		for(i=0;i<v.size();i++)
		{
			if(m1[v[i]]==1)
			{
				flag=true;
				break;
			}
		}
		if(flag)
		{
			cout<<m2[v[i]]<<"\n";
		}
		else
		{
			cout<<"Nobody wins.\n";
		}
	}
	return 0;
}
