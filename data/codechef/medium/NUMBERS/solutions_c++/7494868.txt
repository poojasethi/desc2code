#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<utility>
#define gc getchar
typedef long long int ll;
using namespace std;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll t,n,no,i;
    cin>>t;
    string nm;
    while(t--)
    {
        cin>>n;
        string ans;
        pair<ll,string> a[n];
        for(i=0;i<n;i++)
        {
            cin>>nm>>no;
            a[i].first=no;
            a[i].second=nm;
        }
        sort(a,a+n);
        if(n==1)
            cout<<a[0].second<<'\n';
        else
        {
            for(i=0;i<n;i++)
            {
                if(i==0)
                {
                    if(a[i].first!=a[i+1].first)
                    {
                        ans=a[i].second;
                        break;
                    }
                }
                else if(i==(n-1))
                {
                    if(a[i].first!=a[i-1].first)
                    {
                        ans=a[i].second;
                        break;
                    }
                }
                else
                {
                    if(a[i].first!=a[i-1].first && a[i].first!=a[i+1].first)
                    {
                        ans=a[i].second;
                        break;
                    }
                }
            }
            if(ans.empty())
                cout<<"Nobody wins.\n";
            else
                cout<<ans<<'\n';
        }
    }
    return 0;
}
