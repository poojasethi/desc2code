#include<bits/stdc++.h>
using namespace std;
bool  cmp(pair<string,int > a,pair<string,int > b)
{
	return a.second<b.second;
}
int main()
{
	int t;
	scanf("%d",&t);
	vector<pair<string,int > > pl;
	int i,j,n;
	while(t--)
	{
		pl.clear();
		scanf("%d",&n);
		for(i=0;i<n;++i)
			{
				string s;
				cin>>s>>j;
				pl.push_back(pair<string,int> (s,j));
			}
		sort(pl.begin(),pl.end(),cmp);
		int c=0,p=0;
		int ind;
		for(i=0;i<n;++i)
		{
			if(pl[i].second!=p)
			{
				if(c==1)break;
				p=pl[i].second;
				c=1;
				ind=i;
			}
			else
			{
				++c;
			}

		}
		if(c==1)
			cout<<pl[ind].first<<"\n";
		else cout<<"Nobody wins.\n";
	}
	return 0;
}
