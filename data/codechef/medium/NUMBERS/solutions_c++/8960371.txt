#include<iostream>
#include<algorithm>
using namespace std;
class A
{
public:
string name;
long long a;
};
int main()
{
long t;
cin>>t;
while(t--)
{
int k=0;
long n;
cin>>n;
A arr[n];
long long b[n];
long long miny,i;
for(i=0;i<n;i++)
{
cin>>arr[i].name;
cin>>arr[i].a;
b[i]=arr[i].a;
}
sort(b,b+n);
if(b[0]!=b[1])
{
k=1;
miny=b[0];
}
if(k!=1)
{
for(i=1;i<n-1;i++)
{
if(b[i]!=b[i-1]&&b[i]!=b[i+1])
{
k=1;
miny=b[i];
break;
}
}
}
if(k!=1)
{
if(b[n-1]!=b[n-2])
{
k=1;
miny=b[n-1];
}
}
if(k==1)
{
for(i=0;i<n;i++)
{
if(arr[i].a==miny)
{
cout<<arr[i].name<<"\n";
break;
}
}
}
else
cout<<"Nobody wins."<<"\n";
}
}
