#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
#include<string.h>
#include<math.h>

using namespace std;

struct inp
{
    char name[6];
    unsigned long long int num;
    void input()
    {
        scanf("%s",name);
        scanf("%lld",&num);
    }
};
bool cmp(inp a,inp b) 
{
    if (a.num<b.num) return true;
    return false;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        inp obj[10000];
        int N,i,ctr=1;
        scanf("%d",&N);
        for (i=0;i<N;i++) obj[i].input();
        sort(obj,obj+N,cmp);
        
        for (i=0;i<N-1;i++)
        {
            if (obj[i].num==obj[i+1].num) ctr++;
            else
            {
                if (ctr==1) break;
                ctr = 1;
            }
        }
        if (i==N-1)
        {
            if (obj[N-1].num==obj[N-2].num) printf("Nobody wins.");
            else printf("%s",obj[N-1].name);
        }
        else printf("%s",obj[i].name);
        printf("\n");
    }
    return 0;
}
