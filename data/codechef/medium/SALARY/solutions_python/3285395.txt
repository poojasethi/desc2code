#!/usr/local/bin/python2.7

def mini(a,b):
  if a<b:
    return a
  else:
    return b

t=int(raw_input())
while t>0:
  n=int(raw_input())
  w=map(int,raw_input().strip().split(" "))
  su=0
  mins=10000
  for i in xrange(0,n):
    su+=w[i]
    mins=mini(w[i],mins)
  x=su - (n*mins)
  print x
  t-=1
