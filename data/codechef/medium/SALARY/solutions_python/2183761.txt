from sys import stdin
 
t=int(stdin.readline())
for i in range(0,t):
        n=input()
        b = map(int,stdin.readline().split())
        s=0
        m=b[0]
        for j in range(0,n):
                s=s+b[j]
                if m>b[j]:
                        m=b[j]
        l=s-n*m
        print l
