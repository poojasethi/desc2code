    n=raw_input()
    a=map(int,raw_input().split(' '))
    x=0
    for j in a:
        if j!=a[0]:
            x=1    
    count=0
    while x:
        x=0
        z=(max(a)-min(a))
        count+=z
        j=a.index(max(a))
        a=[i+z for i in a]
        a[j]-=z
        for j in a:
            if j!=a[0]:
                x=1
    print count
