#include <iostream>
#include <cstdio>
#include <string.h>

using namespace std;

int main()
{
    int tc;
    scanf("%d", &tc);
    while(tc--)
    {
        int n, min = 10000;
        long int sum = 0;
        scanf("%d", &n);
        int arr[n];
        for(int i = 0; i < n; i++)
        {
            scanf("%d", &arr[i]);
            sum += arr[i];
            if(arr[i] < min)
            {
                min = arr[i];
            }
        }
        printf("%ld\n", sum - (min*n));
    }
    return 0;
}
