#include<bits/stdc++.h>
using namespace std;
int equal(int arr[],int n)
{
	for(int i=0;i<n-1;i++)
	{
		if(arr[i]!=arr[i+1])
			return 0;
	}
	return 1;
	
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int count=0;
		int arr[n];
		for(int i=0;i<n;i++)
			cin>>arr[i];
		while(!equal(arr,n))
		{
			sort(arr,arr+n);
			int temp=arr[n-1];
			arr[n-1]=arr[n-1]-1;
			count++;
			for(int i=n-2;i>0;i--)
			{
				if(arr[i]==temp)
				{
					arr[i]=arr[i]-1;
					count++;
				}
				
			}
		}
		
		cout<<count<<endl;
	}
	return 0;
}
