#include<bits/stdc++.h>
#define ll long long int

using namespace std;

int diff[1000],a[1000];

int main(){
	int t,n,i,sum;

	cin>>t;
	while(t--){
		cin>>n;

		for(i=0;i<n;i++)
			cin>>a[i];

		sort(a,a+n);
        
        diff[0]=0;sum=0;
		for(i=1;i<n;i++){
           diff[i]=diff[i-1]+(a[i]-a[i-1]);
           sum+=diff[i];
		}

		cout<<sum<<endl;
	    
	}
	return 0;
}
