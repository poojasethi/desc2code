#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    long long t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int a[n],min=9999999,sum=0;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
            if(a[i]<min)
                min=a[i];
            sum+=a[i];
        }
        int w=sum-n*min;
        cout<<w<<endl;
    }
}
