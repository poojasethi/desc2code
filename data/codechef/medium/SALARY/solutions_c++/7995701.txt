#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,i,t,s,sum=0;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int a[n];

        for(i=0;i<n;i++)
        {
        cin>>a[i];

        }
        sort(a,a+n);
        s=a[0];
        for(i=1;i<n;i++)
        {
        sum=sum+(a[i]-s);
        }
        cout<<sum<<"\n";
        sum=0;
    }
    return 0;
}
