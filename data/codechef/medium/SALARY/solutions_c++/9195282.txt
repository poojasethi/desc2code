#include<bits/stdc++.h>
using namespace std;
int main(void){
	int t,n,sum,count,val,c,i,diff,min;
	cin>>t;
	while(t!=0){
		t--;
		count=0;
		sum=0;
		cin>>n;
		vector<int> a(n);
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<n;i++)
			sum+=a[i];
		c=0;
		val=a[0];
		for(i=1;i<n;i++){
			if(a[i]!=val){
				c=1;
				break;
			}
		}
		if(c==0)
			printf("0\n");
		else{
			min=999999;
			for(i=0;i<n;i++){
				if(a[i]<min)
					min=a[i];
			}
			count=sum-min*n;
			cout<<count<<endl;
		}			
	}
	return 0;
}
