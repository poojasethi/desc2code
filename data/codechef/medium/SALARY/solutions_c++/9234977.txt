#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int t, n;
	vector<int> W;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d", &n);
		while (n--)
		{
			int p;
			scanf("%d", &p);
			W.push_back(p);
		}
		sort(W.begin(), W.end());
		int op = 0;
		for (int i = 0; i < W.size(); ++i)
		{
			op += W[i] - W[0];
		}
		printf("%d\n", op);
		W.clear();
	}
}
