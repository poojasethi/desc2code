#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,a[101],ans=0,k;
		cin>>n;
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		for(;;)
		{
			if(a[0]==a[n-1])
			break;
			k=a[n-1]-a[0];
			ans+=k;
			for(i=0;i<n-1;i++)
			a[i]+=k;
			sort(a,a+n);
			
			
			
		}
		cout<<ans<<"\n";
		
	}
	// your code goes here
	return 0;
}
