#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,t;
	int w[10002];
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			scanf("%d",&w[i]);
		}
		int sum=0,min=INT_MAX;
		for(int i=0;i<n;i++){
			sum+=w[i];
			if(w[i]<min)
			min=w[i];
		}
		printf("%d\n",sum-(n*min));
	}
	return 0;
}
