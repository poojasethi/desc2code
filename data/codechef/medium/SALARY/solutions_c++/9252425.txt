#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t,i,s,n;
    scanf("%d",&t);
    while(t--)
    {
        s=0;
        scanf("%d",&n);
        int a[n];
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        sort(a,a+n);
        if(n>1)
        {
            for(i=1;i<n;i++)
            {
                s=s+a[i]-a[0];
            }
        }
        printf("%d\n",s);
    }
}
