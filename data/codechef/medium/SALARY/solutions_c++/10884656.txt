#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n;
		scanf("%d",&n);
		int a[n];
		for (int i = 0; i < n; i++)
		{
			scanf("%d",&a[i]);
		}
		int maxx = -1,minn = 987654321;
		int id = -1;
		int ctr = 0;
		while (1)
		{
			for (int i = 0; i < n; i++)
			{
				if (a[i] > maxx)
				{
					maxx = a[i];
					id = i;
				}
				minn = min(a[i],minn);
			}
			if (maxx == minn)
				break;
			ctr += (maxx-minn);
			for (int j = 0; j < n; j++)
			{
					if (j != id)
					a[j] += (maxx-minn);
			}
			maxx = -1;
			minn = 987654321;
			id = -1;  
		}
		cout << ctr << endl;
	}
	return 0;
}
