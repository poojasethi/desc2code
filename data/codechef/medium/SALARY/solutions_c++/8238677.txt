	#include<iostream>
	#include<vector>
	#include<algorithm>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			vector<int > v;
			int n;
			cin>>n;
			int temp;
			int sum=0;
			for(int i=0;i<n;i++)
			{	
				cin>>temp;
				v.push_back(temp);
				sum+=temp;
			}
			vector<int >::iterator it=min_element(v.begin(),v.end());	
			sum-=(*it)*n;
			cout<<sum<<endl;
			t--;
		}	
	}
