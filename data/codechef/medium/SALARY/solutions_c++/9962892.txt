#include<cstdio>
using namespace std;
int main()
{
    long long int tc;
    scanf("%lld",&tc);
    while(tc--)
    {
        long long int n,i,sum=0,max=1000000,a,ans;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&a);
            sum+=a;
            if(a<max)
                max=a;
        }
        ans=sum-max*n;
        printf("%lld\n",ans);
    }
    return 0;
}
