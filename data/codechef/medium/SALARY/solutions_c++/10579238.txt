   #include <vector>
    #include<algorithm>
    #define gc getchar_unlocked
    #define pc putchar_unlocked
    #define p(n) printf("%d\n",n);
    #define s(n) scanf("%d",&n);
    #define S(n) scanf("%li",&n);
     inline unsigned int fastread()
    {
      unsigned int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
    using namespace std;
    int main()
    {
        int t;
        t=fastread();
        while(t--)
        {
            int n;
            n=fastread();
            int a[n];
            int c=0;
            int min=10001;
            for(int i=0;i<n;i++){
                a[i]=fastread();
                if(min>a[i])
                    min=a[i];
                    c=c+a[i];
            }
            p(c-n*min)
        }
        return 0;
    }
