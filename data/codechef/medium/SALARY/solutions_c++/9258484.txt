#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n;
		scanf("%d", &n);
		int w, min=99999;
		long sum=0;
		for(int i=0; i<n; i++)
		{
			scanf("%d", &w);
			if( min>w )
				min=w;
			sum += w;
		}
		printf("%ld \n", sum-n*min);
	}
	return 0;
}
