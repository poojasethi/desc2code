using namespace std;

int main()
{
	int t;
	scanf("%d", &t);
	while (t--) {
		int n;
		scanf("%d", &n);
		int wt[10010];
		memset(wt, 0, sizeof(wt));
		for (int i = 0; i < n; i++) {
			int w;
			scanf("%d", &w);
			wt[w]++;
		}
		int moves = 0;
		for (int i = 10000; i >= 0; i--) {
			if (wt[i] == n)
				break;
			moves = moves + wt[i];
			wt[i-1] += wt[i];
			
		}
		printf("%d\n", moves);
	}
	return 0;
}
