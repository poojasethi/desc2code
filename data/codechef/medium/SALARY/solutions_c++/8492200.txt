#include <iostream>
#include <cmath>
using namespace std;
#include<algorithm>
int main() {
	int T,n,a[10001],y,ans;
	cin >> T;
	while(T--)
    {
        cin>>n;
        ans=0;
        for(int i=0;i<n;i++)
        cin>>a[i];
        y=*min_element(a,a+n);
        for(int i=0;i<n;i++)
        {
            if(a[i]>y)
                ans+=a[i]-y;
        }
           cout<<ans<<endl;
    }
}
