#include<iostream>
using namespace std;
int main()
{
 long int t,n,min=0,i,w[500],count=0;
 cin>>t;
 while(t--)
 {
  count=0;
  cin>>n;
  for(i=0;i<n;i++)
  {
   cin>>w[i];
   if(i==0)
   min=w[i];
   if(min>w[i])
   min=w[i];
 }
 for(i=0;i<n;i++)
 count+=w[i]-min;
 cout<<count<<"\n";
 }
 return 0;
 }
