#include<cstdio>
#define INF 1000000000
using namespace std;
int MIN(int a,int b){
    if(a<b){
        return a;
    }
    return b;
}
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int N;
        scanf("%d",&N);
        int arr[N],min=INF;
        for(int i=0;i<N;i++){
            scanf("%d",&arr[i]);
            min=MIN(min,arr[i]);
        }
        int ans=0;
        for(int i=0;i<N;i++){
            ans+=arr[i]-min;
        }
        printf("%d\n",ans);
    }
}
