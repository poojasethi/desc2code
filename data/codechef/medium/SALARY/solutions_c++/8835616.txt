#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int t,n,sum;
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
	sum=0;
	scanf("%d",&n);
	int *arr=(int *)malloc(sizeof(int)*n);
	for(int j=0;j<n;j++)
	{
	scanf("%d",arr+j);
	}
	sort(arr,arr+n);
	for(int j=1;j<n;j++)
	{
	sum=sum+arr[j]-arr[0];
	}
	printf("%d\n",sum);
	}
	return 0;
}
