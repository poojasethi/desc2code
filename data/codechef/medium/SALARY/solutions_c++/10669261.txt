#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main(){
    int t;
    cin>>t;
    while(t-- ){
        int n;;
        cin>>n;
        long long int lst[n];
        for(long long int i=0;i<n;i++){
            cin>>lst[i];
        }
        sort(lst,lst+n);
        long long result=0;
        for(long long int i=1;i<n;i++){
            result+=lst[i]-lst[0];
        }
        cout<<result<<endl;
    }

    return 0;
}
