#include <bits/stdc++.h>
using namespace std;
#define ll long long
#define pb push_back
int main() 
{
	vector<bool> v(1000001);
	v[0]=v[1]=true;
	ll i,j,l,r,t,lt=1000000,pz,cal,temp,val,c,ans;
	vector<ll> p;
	for(i=4;i<=lt;i=i+2)
		v[i]=true;
	for(i=9;i<=lt;i=i+6)
		v[i]=true;
	p.pb(2);
	p.pb(3);
	for(i=6;i<lt;i=i+6)
	{
		if(!v[i-1])
		{
			p.pb(i-1);
			if((i-1)*(i-1)<=lt)
			{
				for(j=(i-1)*(i-1);j<=lt;j=j+2*(i-1))
					v[j]=true;
			}
		}
		if(!v[i+1])
		{
			p.pb(i+1);
			if((i+1)*(i+1)<=lt)
			{
				for(j=(i+1)*(i+1);j<=lt;j=j+2*(i+1))
					v[j]=true;
			}
		}
	}
	pz=p.size();
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&l,&r);
		vector<ll> sv(r-l+1);
		for(i=0;(i<pz)&&(p[i]*p[i]<=r);i++)
		{
			if(l%(p[i])==0)
			{
				cal=l;
			}
			else
				cal=((l/p[i])+1)*p[i];
			for(j=max(p[i]*2,cal);j<=r;j=j+p[i])
			{
				temp=j;
				val=p[i];
				c=0;
				while((temp%val)==0)
				{
					temp/=val;
					c++;
				}
				if(temp==1)
				{
					if(!v[c+1])
					{
						sv[j-l]=0;
					}
					else
						sv[j-l]++;
				}
				else
					sv[j-l]++;
			}
		}
		ans=0;
		for(i=max((ll)2,l);i<=r;i++)
		{
			if(!sv[i-l])
			{
				//cout<<i<<endl;
				ans++;
			}
		}
		printf("%lld\n",ans);
		
	}
	return 0;
}
