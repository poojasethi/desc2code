ll powr(int s,int p)
{
	if(p==0)
		return 1;	
 
	if(p%2==1)
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);
		q=(q*s);	
		return ( q );
	}
 
	else
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);	
		return (q);
	}
}

ll l,r,f[1000005];
bool isPrime2[1000005];

void setup(ll m,ll n)
{
	unsigned long long i,j;
	for(i=0;i<=K;i++)
		isPrime2[i]=false;
	for(i=2;i*i<=K;i++)
		if(isPrime2[i]==false)
			for(j=i*i;j<=K;j+=i)
				isPrime2[j]=true;
	isPrime2[1]=true;
	return;
}

void scan()
{
	s3(l,r)
	setup(l,r);
}

void out()
{
	FOR(i,0,r-l+1)		f[i]=0;
	int ans=0;
	for(ll i=1;i*i<=r;i++)
	{
		if(l%i==0)
		{
			if(i*i<=l)
			{
			if(i*i==l)	f[0]+=1LL;
			else	f[0]+=2LL;
			}
		}
		ll u;
		if(l<i)		u=i;
		else
		{
			u=l+i-(l%i);
		}
		//cout<<u;
		for(ll j=u;j<=r;j+=i)
		{
			if(i*i>j)	continue;
			if(i*i==j)	f[j-l]+=1LL;
			else	f[j-l]+=2LL;
		}
	}
	FOR(i,0,r-l+1)
	{
		//cout<<f[i]<<'\n';
		if(!isPrime2[f[i]])	++ans;		
	}
	cout<<ans<<'\n';
}

int main()

		scan();
