/*Yet Another Cute Girl*/
#include<iostream>
#include<cmath>
using namespace std;
long long prime[1000005],divisor[1000005];
int main()
{
    prime[1]=1;
    long long i,j;
    for(i=2;i<=1000;i++)
    {   if(!prime[i])
        for(j=2*i;j<=1000000;j+=i)
           prime[j]=i;
    }
    //long long f=0;
    for(i=2;i<=1000000;i++)
        if(!prime[i])
        {   //f++;
            prime[i]=i;
        }
    //cout<<f<<endl;
    int t;
    cin>>t;
    while(t--)
    {   long long l,r;
        cin>>l>>r;
        long long ans=0,temp,s=sqrt(r)+1;
        for(i=0;i<1000004;i++)
            divisor[i]=0;
        for(i=2;i<=s;i++)
        {   temp=(l/i)*i;
            //cout<<temp<<endl;
            while(temp<=r)
            {   if(temp<l)
                {   temp+=i;
                    continue;
                }
                if(i*i>temp)
                {   temp+=i;
                    continue;
                }
                if(i*i==temp)
                    divisor[temp-l]++;
                else
                    divisor[temp-l]+=2;
                temp+=i;
            }
        }
        for(i=l;i<=r;i++)
        {   if(i==1)
                continue;
            divisor[i-l]+=2;
            if(prime[divisor[i-l]]==divisor[i-l])
                ans++;
        }
        cout<<ans<<endl;
    }
    return 0;
}
