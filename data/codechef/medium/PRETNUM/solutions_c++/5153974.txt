#include<cstdio>
using namespace std;
long long int prime[1000009]={0},a[1000009],b[1000009];
int main()
{  long long int t,i,j,m,ans,l,r,cn;
	prime[0]=1;
	prime[1]=1;
	for(i=2;i*i<=1000000;i++)
	{
		if(prime[i]==0)
		{
			for(j=i*2;j<=1000000;j=j+i)
			{
				prime[j]=1;
			}
		}
	}
	scanf("%lld",&t);
	while(t--)
	{   ans=0;
		scanf("%lld%lld",&l,&r);
		for(i=l;i<=r;i++)
		{
			a[i-l]=i;
			b[i-l]=1;
		}
		for(i=2;i<=1000009;i++)
		{
			if(prime[i]==0)
			{
				m=l/i;
				m=m*i;
				if(m<l)
				{
					m=m+i;
				}
				for(j=m;j<=r;j=j+i)
				{  long long int p;
				    cn=0;
					p=j-l;
					while(a[p]%i==0)
					{
						a[p]=a[p]/i;
						cn++;
					}
					b[p]=b[p]*(cn+1);
				}
			}
		}
		for(i=l;i<=r;i++)
		{
			if(a[i-l]!=1)
			b[i-l]=b[i-l]*2;
			if(prime[b[i-l]]==0)
			{
				ans++;
			}
		}
		printf("%lld\n",ans);
	}
}
