#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<string>
using namespace std;


long long int isP[1000010];

long long int rem[1000010];
long long int cnt[1000010];

int main() {
        long long int L, R,i,t,x,c,j,ans;

	memset(isP,1,8000010);
        isP[0]=0;
        isP[1]=0;
	for (i = 2; i * i < 1000010; ++i) {
		if (isP[i]) {
			for (j = i * i; j <1000010; j += i)
				isP[j] = 0;
		}
	}

	cin >> t;
	while (t--) {

		cin >> L >> R;

		for (i = L; i <= R; ++i) {
			rem[i - L] = i;
			cnt[i - L] = 1;
		}

		for (i = 2; i < 1000010; ++i)
			if (isP[i]) {
				for (j = ((L - 1) / i + 1) * i; j <= R; j += i) {
					x = j - L;
					c = 0;
					while (rem[x] % i == 0) {
						rem[x] /= i;
						++c;
					}
					cnt[x] *= c + 1;
				}
			}

		ans = 0;

		for (i = 0; i <= R - L; ++i) {
			if (rem[i] > 1) {
				cnt[i] *= 2;
			}
			if (isP[cnt[i]]) {
				++ans;
			}
		}

		cout << ans << "\n";
	}
}
