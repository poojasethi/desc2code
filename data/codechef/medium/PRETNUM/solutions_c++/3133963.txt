#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
using namespace std;
long long int a[1000005] = {0};
int main () {
	int b[10005];
	long long int l, r, i, j, ans = 0;
	for ( i = 0; i < 10005; i ++ )
		b[i] = 1;
	b[0] = 0;
	b[1] = 0;
	for ( i = 2; i < 10005; i ++ ) {
		if (b[i] == 1 ) {
			for ( j = i; j*i < 10005; j ++)
				b[j*i] = 0;
		}
	}
	
	int T;
	scanf ("%d", &T);
	while ( T -- ) {
		ans = 0;
		scanf ("%lld %lld", &l, &r);
		for ( i = l; i <= r; i ++ )
			a[i-l] = 0;
		for ( i = 1; i <= sqrt (r); i ++ ) {
			long long int s = i*i;
			for ( j = ((l-1)/i + 1) * i; j <= r; j += i ) {
				if ( j < s )
					continue;
				if ( j == s ) {
					//cout << 1 << " " << j << " " << j-l << " " << i << endl;
					a[j-l] ++;
				}
				else {
					//cout << 2 << " " << j << " " << j-l << " " << i << endl;
					a[j-l] += 2;
				}
			}
		}
		for ( i = l; i <= r; i ++ ) {
			if ( b[a[i-l]] ){
				//cout << i << endl;
				ans ++;
			}
		}
		printf ("%lld\n", ans );
	}
	return 0;
}
