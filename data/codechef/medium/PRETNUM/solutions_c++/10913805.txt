        using namespace std;
       ll p[100001];
        void gprime()
        {
        	FOR(i,0,100000)
        	p[i]=1;
        	p[0]=0;p[1]=0;
        	for(int i=2;i*i<=100000;++i)
        	{
        		if(p[i]){
        		for(int j=2*i;j<=100000;j+=i)
        		p[j]=0;}}
        }
        int main()
        {
        	ll t;
        	cin>>t;
        	gprime();
        	ll cd[1000005]={0};
        	while(t--)
        	{
        		ll l,r;
        		ll count=0;
        		cin>>l>>r;
        		FOR(i,l,r)
        		cd[i-l]=0;
        		for(ll i=1;i*i<=r;++i)
        		{
        	     for(ll j=((l-1)/i+1)*i;j<=r;j+=i)
        	     {
        	     	if(j<i*i);
        	     	else if(j==i*i)
        	     	cd[j-l]++;
        	     	else if(j>i*i)
        	     	cd[j-l]+=2;
        	     }
        		}
        		
        		FOR(i,l,r)
        			{if(p[cd[i-l]])
        			{++count;}
        		}
        		cout<<count<<endl;
        	}
        	return 0;
        }
