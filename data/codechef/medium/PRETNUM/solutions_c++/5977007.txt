void sieve()
{
	for(int i=4;i<MAX;i+=2)
		arr[i]=1;
	arr[0]=1;
	arr[1]=1;
	arr[2]=0;
	 
	for(long long int i=3;i<=sqrt(MAX);i+=2)
	{
	
		if(!arr[i])
		{
			
      		for(long long int j=i*i,k=i*2; j<MAX; j+=k)
      		{
      
				arr[j]=1;
			}
		}
	}
	
	primes[0] = 2;
	for(long long int i=3;i<=MAX;i=i+2)
	{
	
		if(!arr[i])
		{
			isprime[i]=1;
       		primes[l++] = i;
   		}
	}
 
		
}
