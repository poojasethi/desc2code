#include <bits/stdc++.h>
#define M 1000000007

using namespace std;

int n;
long long dp[1002][1010];

long long f(int idx, int mx)
{
	if ( idx == -1 ) return 1;
	long long &ans = dp[idx][mx];
	if ( ans != -1 ) return ans;
	ans = 0;
	ans += ((long long)(mx)*f(idx-1,mx))%M;
	ans %= M;
	ans += f(idx-1,mx+1)%M;
	ans %= M;
	return ans;
}

int main() {
	int t,n;
	cin >> t;
	memset(dp, -1, sizeof(dp));
	while ( t-- ) {
		cin >> n;
		long long ans = f(n-2,1)%M;
		cout << ans << endl;
	}
	return 0;
}
