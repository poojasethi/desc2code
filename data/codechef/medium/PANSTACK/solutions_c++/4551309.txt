#include<bits/stdc++.h>
#define ll unsigned long long
#define MOD 1000000007
using namespace std;
ll dp[1001][1001];
int main(){
int n,j,t,i,k;
dp[1][1]=1;
for(i=2;i<1001;i++){
for(j=1000;j>=1;j--){
dp[i][j]=(dp[i-1][j]*(j)+dp[i-1][j-1])%MOD;
}
}

cin>>t;
for(i=0;i<t;i++){
cin>>n;
ll ans=0;
for(j=1;j<=n;j++){
ans=(ans+dp[n][j])%MOD;
}
cout<<ans<<endl;

}
return 0;
}
