#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <vector>
#include <algorithm>
#include <string.h>
#include <cmath>
#include <limits.h>
#define M 1000000007
#define max 1001
#define ll long long
using namespace std;
ll a[max][max];
int main()
{

    a[0][0]=1;
    for(int i=1;i<max;i++)
    {
        a[i][0]=a[i-1][i-1];
        for(int j=1;j<=i;j++)
        {
            a[i][j]=a[i][j-1]+a[i-1][j-1];
            if(a[i][j]>=M)
            {
                a[i][j]=a[i][j]%M;
            }
        }
    }
    /*for(int i=0;i<7;i++)
    {
        for(int j=0;j<=i;j++)
        {
            printf("%d ",a[i][j]);
        }
        cout<<endl;
    }*/
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        int n;
        scanf("%d",&n);
        printf("%lld\n",a[n-1][n-1]);
    }
    return 0;
}
