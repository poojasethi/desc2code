#define INF 2000000000 // 2 billion
#define MOD 1000000007

int dp[1009][1009];

int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("output.txt","r",stdout);
    memset(dp,0,sizeof(dp));
    dp[1][1] = 1;
    REP(i,2,1000)
    {
        REP(j,1,i)
        {
            dp[i][j] += dp[i-1][j-1];
            if(dp[i][j] >= MOD)
                dp[i][j] -= MOD;
            dp[i][j] += (j*1ll*dp[i-1][j])%MOD;
            if(dp[i][j] >= MOD)
                dp[i][j] -= MOD;
        }
    }
    int t;s(t);
    while(t--)
    {
        int n;s(n);
        int ans = 0;
        REP(i,1,n)
        {
            ans += dp[n][i];
            if(ans >= MOD)
                ans -= MOD;
        }
        printf("%d\n",ans);
    }
    return 0;
}
