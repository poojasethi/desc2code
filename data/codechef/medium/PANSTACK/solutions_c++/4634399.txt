#include<bits/stdc++.h>

using namespace std;
 
#define FOR(i,a,b) for(ll i=a;i<b;i++)
#define For(i,a,b) for(int i=a-1;i>=0;i--)
#define K 10000000
#define L 400
#define ll long long
#define s1(a) scanf("%d",&a);
#define s2(a) scanf("%lld",&a);
#define s3(a,b) scanf("%d%d",&a,&b);
#define s4(a,b,c) scanf("%d%d%d",&a,&b,&c);
#define pb push_back
#define mp make_pair
#define F first
#define S second 
/*******************************MAIN CODE STARTS*******************************/

#define MOD 1000000007
ll dp[1001][1001]={0},ans[1001]={0};
int n;

void pre()
{
	FOR(i,1,1001)
		dp[i][1]=ans[i]=1LL;
	FOR(i,2,1001)
	{
		FOR(j,2,1001)
		{
			dp[i][j]=((j*dp[i-1][j])%MOD + dp[i-1][j-1]);
			if(dp[i][j]>MOD)
				dp[i][j]=dp[i][j]%MOD;
			ans[i]=(ans[i]+dp[i][j]);
			if(ans[i]>MOD)
				ans[i]=ans[i]%MOD;
			if(dp[i][j]==0)
				break;
		}
	}
}

void scan()
{
	s1(n)
}

void out()
{
	cout<<ans[n]<<'\n';
}

int main()
{
	pre();
	int t;
	s1(t)
	while(t--)
	{
		scan();
		out();
	}
	return 0;
}
