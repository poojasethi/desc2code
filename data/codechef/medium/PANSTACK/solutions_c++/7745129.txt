typedef long long LL;

using namespace std;

LL dp[Max][Max];
LL n, CS, ck[Max][Max];
LL mod = 1000000007;

LL dp_sol(LL pos,LL mx)
{
    if(pos==1) return 1;
    if(ck[pos][mx] == CS) return dp[pos][mx];

    LL ret = 0;
    /*for(int i=1; i<=mx; i++)
    {
        ret += dp_sol(pos-1, mx);
    }*/
    ret+=(mx*dp_sol(pos-1,mx));
    ret%=mod;
    ret += dp_sol(pos-1, mx+1);
    ret%=mod;
    ck[pos][mx] = CS;
    return dp[pos][mx] = ret;
}
