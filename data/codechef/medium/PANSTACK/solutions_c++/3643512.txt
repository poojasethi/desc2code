#include <stdio.h>
#define ull unsigned long long
#define maxn 1001
ull md = 1000000007;
ull dp[1001][1001];
ull ans[maxn];
ull mod(ull x)
{
	return x%md;
}
void pre()
{
	int h,l;
	for (l = 1; l < maxn; l++) {
		dp[1][l] = 1;
		ans[l] = 1;
	}
	for (h = 2; h < maxn; h++) {
		for (l = h; l < maxn; l++) {
			dp[h][l] = mod(dp[h][l-1]*h+dp[h-1][l-1]);
			ans[l] = mod(ans[l]+dp[h][l]);
		}
	}
}
int main()
{
	pre();
	int t;
	scanf("%d",&t);
	while (t--) {
		int n;
		scanf("%d",&n);
		printf("%llu\n",ans[n]);
	}
	return 0;
}
