using namespace std;

int main()
{
	static long long int t , n , i , j,ans, a[1003][1003],mod=1000000007;
	for(i=1;i<1003;i++)a[0][i]=0;
	for(i=1;i<1003;i++)
	{
		a[i][0]=0;
		a[i][1]=1;
		for(j=2;j<1003;j++)
		{
			a[i][j]=(a[i-1][j-1]+a[i-1][j]*j)%mod;
		}
	}
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ans=0;
		for(i=0;i<1002;i++)
		{
			ans=(ans+a[n][i])%mod;
		}
		printf("%lld\n",ans );
	}
	return 0;
}
