#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <time.h>
#include <vector>
#include <set>
#include <map>
#include <algorithm>
#include <queue>
using namespace std;
long long dp[1010][1010],ans[1010];
const int Mod = 1000000007;
int main()
{
    int T;
    dp[1][1]=1;
    for (int i=1;i<=1000;i++)
    {
        for (int j=1;j<=1000;j++)
        {
            if (dp[i][j]==0) continue;
            dp[i+1][j]+=dp[i][j]*j%Mod;
            dp[i+1][j+1]+=dp[i][j];
            if (dp[i+1][j]>=Mod) dp[i+1][j]-=Mod;
            if (dp[i+1][j+1]>=Mod) dp[i+1][j+1]-=Mod;
        }
        ans[i]=0;
        for (int j=1;j<=1000;j++)
            ans[i]=(ans[i]+dp[i][j])%Mod;
    }

    scanf("%d",&T);
    while (T--)
    {
        int n;
        scanf("%d",&n);
        printf("%lld\n",ans[n]);
    }
    return 0;
}
