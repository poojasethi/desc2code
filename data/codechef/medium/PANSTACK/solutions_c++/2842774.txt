#include<stdio.h>

long long comb[1001][1001];
long long bell[1001];
void init()
{
    for(int y=0;y<1001;y++)
    {
        comb[y][y]=1;
        comb[y][0]=1;
    }
    for(int i=2;i<1001;i++)
    {
        for(int j=1;j<i;j++)
            comb[i][j]=(comb[i-1][j]+comb[i-1][j-1])%1000000007;
    }

    bell[0]=1;bell[1]=1;
    for(int e=2;e<1001;e++)
    { bell[e]=0;
        for(int q=0;q<e;q++)
            bell[e]=(comb[e-1][q]*bell[q]+bell[e])%1000000007;
    }
}

int main()
{
    init();
    int T;
    scanf("%d",&T);
    while(T)
    {
        int N;
        scanf("%d",&N);
        printf("%lld\n",bell[N]);
        T--;
    }
}
