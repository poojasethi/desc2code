#include<iostream>
#include<stdio.h>
using namespace std;
#define max 1000000007
long long recursion(long long dp[][1005],int n,int i)
{
long long sum=0;

if(i>n)
return 0;
if(n<=0)
return 0;

if(dp[n][i]==-1)
{
dp[n][i]=(recursion(dp,n-1,i-1)+(recursion(dp,n-1,i)*i)%max)%max;

}
return dp[n][i];

}


int main()
{
int T,n=1001,i,j;
long long dp[1005][1005],ans;
for(i=1;i<=n;i++)
for(j=1;j<=n;j++)
dp[i][j]=-1;
dp[1][1]=1;
for(i=0;i<=n;i++)
{
dp[0][i]=0;
dp[i][0]=0;
}

scanf("%d",&T);
while(T--)
{
scanf("%d",&n);

ans=0;
for(i=1;i<=n;i++)
ans=(ans+recursion(dp,n,i))%max;
printf("%lld\n",ans);
}

}
