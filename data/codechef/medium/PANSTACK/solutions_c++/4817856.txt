#include<stdio.h>
#include<iostream>
using namespace std;
#define gc getchar_unlocked
#define md 1000000007
void fastin(int &x)
{
	register int a=gc();
	x=0;
	for(;a<48 || a>57;a=gc());
	for(;a>47 && a<58;a=gc())
		x=(x<<1)+(x<<3)+a-48;	
}
int main()
{
int t,no;
long long i,j,k=2;
long long pc[1001][1001]={};
long long n[1001],ans;
pc[1][1]=1;
n[1]=1;
for(i=2;i<=1000;i++)
{	ans=0;
	for(j=1;j<=1000;j++)
	{
	pc[i][j]=( (((pc[i-1][j]%md)*j)%md)+(pc[i-1][j-1])%md)%md;
	ans=((ans%md)+pc[i][j])%md;
	if(pc[i][j]==0)
	break;
	}
	n[k]=ans;
	k++;	
}

fastin(t);
while(t--)
{
	fastin(no);
	printf("%lld\n",n[no]);
}
return 0;
}
