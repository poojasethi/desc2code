#include <stdio.h>

#define N 1001
#define MOD 1000000007

typedef long long ll;

ll a[N][N];

void precompute()
{
    ll i, j;
    //Base Case
    for (i = 1; i < N; i++)
        a[i][1] = 1;

    for (i = 2; i < N; i++)
        for (j = 2; j <= i; j++)
            a[i][j] = (a[i-1][j-1] + j*a[i-1][j])%MOD;
}

int solve(int n)
{
    int i;
    ll sum = 0;
    for (i = 1; i <= n; i++)
        sum = (sum + a[n][i])%MOD;
    return sum;
}

int main(int argc, const char *argv[])
{
    int t, n;

    precompute();
    scanf("%d", &t);

    while (t--)
    {
        scanf("%d", &n);
        printf("%d\n", solve(n));
    }

    return 0;
}
