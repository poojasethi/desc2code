#include <cstdio>

#define MOD 1000000007
#define MAXSIZE 1010
#define ull unsigned long long

int main()
{
	ull arr[2][MAXSIZE];
	ull count[MAXSIZE] = {0};
	
	for(int index1 = 1;index1 < MAXSIZE;index1++)
	{
		int ii = 1&index1;
		arr[ii][1] = 1;
		for(int index2 = index1+1;index2 < MAXSIZE;index2++) arr[ii][index2] = 0;
		for(int index2 = 2;index2 <= index1;index2++){
			arr[ii][index2] = ((arr[1 - ii][index2] * index2)%MOD + arr[1 - ii][index2 - 1])%MOD;
		}
		for(int index2 = 1;index2 <= index1;index2++) count[index1] = (count[index1] + arr[ii][index2])%MOD;
	}
	
	//for(int index1 = 1;index1 < MAXSIZE; index1++) printf("%llu ",count[index1]);
	//printf("\n");
	
	int t,n;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		printf("%llu\n",count[n]);
	}
	
	return 0;
}
