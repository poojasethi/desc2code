#include<stdio.h>
#include<assert.h>
#define REP(i,a,b) for(i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)

#define M 1000000007

int dp[1001][1001];
int res[1001];

int main(){
  int T, N;
  int i, j;

  dp[0][0] = 1;
  for(i=0;i<1000;i++) for(j=0;j<=i;j++) if(dp[i][j]){
    dp[i+1][j] = (dp[i+1][j] + dp[i][j] * (long long)(j+1))%M;
    dp[i+1][j+1] = (dp[i+1][j+1] + dp[i][j])%M;
  }

  for(i=0;i<=1000;i++) for(j=0;j<=i;j++){
    res[i] = (res[i] + dp[i][j]) % M;
  }

  assert( scanf("%d",&T) == 1 );
  assert( 1 <= T && T <= 1000 );
  while(T--){
    assert( scanf("%d",&N) == 1 );
    assert( 1 <= N && N <= 1000 );
    printf("%d\n",res[N-1]);
  }

  return 0;
}
