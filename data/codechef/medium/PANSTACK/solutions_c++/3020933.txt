#include<stdio.h>
#include<stdlib.h>
#define BIG 1000000007
int main()
{
	 long long N[1001][1001],sum;
	 int i,n,j,t;

	 scanf("%d",&t);
  
     N[0][0]=0;
     N[1][0]=0;
     N[0][1]=0;
     N[1][1]=1;
     N[2][1]=1;
     
     
     for(i=2;i<=1000;++i)
     	N[1][i]=0;
     
     for(i=0;i<=1000;++i)
     	N[0][i]=0;

     for(i=0;i<=1000;++i)
     	N[i][0]=0;



     for(i=2;i<=1000;++i)
     {  //if(i<30) printf("lane no = %d\n",i);
     	 for(j=1;j<=1000;++j)
      	   { 
            N[i][j] = ( ( N[i-1][j] * j )% BIG + (N[i-1][j-1])%BIG)%BIG;
      	  //if(i<30 && j<30)  printf("%lld  ",N[i][j]);
      	   }
   ///  printf("%lld",N[30][25]);
 }
     

	 while(t--)
	{ 

	
	 	scanf("%d",&n);
	 	sum=0;
	 	for(i=1;i<=n;++i)
	 		{
	 			 sum += N[n][i];
	             sum = sum % BIG;
	        }     
	    printf("%lld\n",sum);
	 	
	 }

}
