#include<iostream>
#include<vector>

#define mod 1000000007

using namespace std;

vector< vector<long long int> > arr;

vector<int> ans;

int main()
{
	arr.resize(1000);
	for(int i=0; i<1000; ++i)
	{
		arr[0].push_back(0);
	}
	arr[0][0] = 1;
	ans.push_back(1);
	for(int i=1; i<1000; ++i)
	{
		for(int j=0; j<1000; ++j)
		{
			if(j == 0)
				arr[i].push_back(arr[i-1][j]);
			else
				arr[i].push_back((arr[i-1][j-1] + (j + 1) * arr[i-1][j]) % mod);
		}
		int count = 0;
		for(int j=0; j<1000; ++j)
			count = (count + arr[i][j]) % mod;
		ans.push_back(count);
	}
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		cout<<ans[n-1]<<endl;
	}
	return 0;
}
