#include<cstdio>
#define MOD 1000000007
long long int pre[1001][1001];
long long int find(int a,int b)
{
	if(a==1) return (b+1);
	if(pre[a-1][b]==-1) pre[a-1][b]=find(a-1,b)%MOD;
	if(pre[a-1][b+1]==-1) pre[a-1][b+1]=find(a-1,b+1)%MOD;
	return (pre[a-1][b]%MOD*b%MOD)+(pre[a-1][b+1]%MOD);
}
int main()
{
	long long int t,n,ans;
	scanf("%lld",&t);
	for(int i=0;i<1000;i++) for(int j=0;j<1000;j++) pre[i][j]=-1;
	while(t--)
	{
		scanf("%lld",&n);
		ans=0;
		if(n==1 || n==2) {printf("%lld\n",n); continue; }
		printf("%lld\n",find(n,0));
	}
	return 0;
}
