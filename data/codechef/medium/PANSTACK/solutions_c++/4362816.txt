#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cstring>
#define mod 1000000007
using namespace std;
int main()
{
  int t,n;
  long long int dp[1010][1010];
  long long int ans=0;
    dp[1][1]=1;
    for(int i=2;i<1010;i++)
    for(int j=1;j<1010;j++)
    {
    dp[i][j]=(dp[i-1][j]*j+dp[i-1][j-1])%mod;
    dp[i][j]=dp[i][j]%mod;
    }
  scanf("%d",&t);
  while(t--)
  {
    ans=0;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
    ans+=dp[n][i];
    ans=ans%mod;
    }
    printf("%lld\n",ans);
  }
  return 0;
}
