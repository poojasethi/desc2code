#include<iostream>
#include<stdio.h>
using namespace std;
#define MAX 1000000007
int main()
{
long long T,n,i,j;
long long dp[1001][1001],sum;
for(i=0;i<=1000;i++)
{
dp[0][i]=0;
dp[i][0]=0;
}
dp[1][1]=1;
for(i=1;i<=1000;i++)
for(j=1;j<=1000;j++)
if(i*j!=1)
dp[i][j]=(dp[i-1][j-1]+(dp[i][j-1]*i)%MAX)%MAX;
scanf("%lld",&T);
while(T--)
{
scanf("%lld",&n);
sum=0;
for(i=1;i<=n;i++)
sum=(sum+dp[i][n])%MAX;

printf("%lld\n",sum);
}
}
