#include <stdio.h>
typedef long long ll;
ll ans [1005][1005];
#define mod 1000000007
int main(){

	int T;
	scanf("%d",&T);

	int n;
	for(int i = 1 ; i<=1000; i++)
		ans[i][1] = 1;
	for(int i = 2 ; i<=1000 ; i++){
		for(int j = 2 ; j<=i ; j++){
			ans[i][j] = (ans[i-1][j-1] + (ans[i-1][j]*j)%mod)%mod;
		}
	}
	while(T--){
		scanf("%d",&n);
		ll fin = 0;
		for(int i = 1 ; i<= n ; i++)
			fin = (fin + ans[n][i])%mod;
		printf("%lld\n",fin);
	}
	return 0;
}
