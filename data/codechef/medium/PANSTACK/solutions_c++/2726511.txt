#include <iostream>
using namespace std;

long long dp[1001][1001];
#define MOD 1000000007

int main() {
  for(int j=0;j<2;++j) for(int i=0;i<1001;++i) dp[j][i] = 0;
  dp[1][1] = 1;
  for(int i=2;i<1001;++i) for(int j=1;j<1001;++j) dp[i][j]=(dp[i-1][j]*j+dp[i-1][j-1])%MOD;
  int t, n;
  cin >> t;
  while(t--) {
    cin >> n;
    long long sum=0;
    for(int i=1;i<=n;++i) sum=(sum+dp[n][i])%MOD;
    cout << sum << endl;
  }
  return 0;
}
