#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    long int t ;
    scanf("%ld",&t) ;
    while(t--)
    {
         double k,gaina,gainb,net=0.0,i=0.0;
            scanf("%lf",&k) ;
            if(k>1-k)
            {
                i=10000.00 ;
            }
            gaina=2*(1-k)*i+i;
            gainb=2*k*(10000-i)+(10000-i) ;
            net=k*gaina+(1-k)*gainb ;

         printf("%lf\n",net) ;
    }
    return 0;
}
