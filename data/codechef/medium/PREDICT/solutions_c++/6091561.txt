#include <iostream>
using namespace std;
#include<stdio.h>
#include<string>
#include<algorithm>
#define nik long long int 
inline nik ashu()
{
	 nik n=0,c=getchar_unlocked();
	 while(c<'0'||c>'9')c=getchar_unlocked();
     while(c>='0'&&c<='9')n=(n<<3)+(n<<1)+c-'0',c=getchar_unlocked();
     return n;
}
int main() 
{
	nik t=ashu();
	while(t--)
	{
		double Pa;
		double ans;
		scanf("%lf",&Pa);
		if(Pa>=0.5)
		{
			ans=10000+10000*(1-Pa)*(2*Pa-1);
		}
		else
		{
			ans=10000+10000*Pa*(1-2*Pa);
		}
		printf("%lf\n",ans);
	}
	// your code goes here
	return 0;
}
