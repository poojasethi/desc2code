#include<iostream>
#include<cstdio>
 using namespace std;
 
 
int main()
{
	int test;
	double pnum,ans;
	cin>>test;
	while(test>0)
	{
		cin>>pnum;
 
		if(pnum>=0.5){
			ans = 10000+10000*(2*pnum-1)*(1-pnum);			
		}else{
			ans = 10000+10000*(1-2*pnum)*pnum;
		}
 
 
		printf("%.9lf\n", ans);
	    test--;
	}
 
	return 0;
}
