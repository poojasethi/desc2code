#include<bits/stdc++.h>
int main()
{
    int t;
    double ans,p;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lf",&p);
        if(p>0.50)
            ans=(10000.0*(1.0+(2.0*(1.0-p))))*p;
        else if(p<0.50)
            ans=(10000.0*(1.0+(2.0*p)))*(1.0-p);
        else
            ans=10000.0;
        printf("%0.6lf\n",ans);
    }
    return 0;
}
