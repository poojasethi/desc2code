#define GETCHAR getchar()
int getInt()
{
char ch = GETCHAR;
char c;
if(ch=='-')
c = GETCHAR;
else
c = ch;
 
 
int t = 0;
while(c<='9' && c>='0')
{
t=t*10+c-'0';
c = GETCHAR;
}
 
if(ch=='-')
return (-t);
else return t;
}
double getDouble()
{
char c;
c = GETCHAR;
int t = 0;
while(c<='9' && c>='0')
{
t=t*10+c-'0';
c = GETCHAR;
}
int d=0,i=1;
if(c=='.')
{
c = GETCHAR;
while(c<='9' && c>='0')
{
i*=10;
d=d*10+c-'0';
c = GETCHAR;
 
}
}
 
return (double)t + ((double)d)/i;
}
