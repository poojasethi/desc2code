#include<stdio.h>
#include<iostream>
#include<stdlib.h>
#include<string.h>

#define INF 1000000000
#define get getchar_unlocked()

using namespace std;

inline int read ()
{
        char c;
        int n = 0;

        while ((c = get) < 48);
        n += (c - '0');
        while ((c = get) >= 48)
        n = n * 10 + (c - '0');
        return n;
}

int main(){

	int tc =read();
	while(tc--)
	{	
		double Pa,  output;
		scanf("%lf", &Pa);
		
		if(Pa >= 0.5){
			output = 10000 + 10000 * (2 * Pa  - 1)* (1 - Pa);			
		}else{
			output = 10000 + 10000 * (1 - 2 * Pa) * Pa;
		}


		printf("%.9lf\n", output);
	}
	return 0;
}
