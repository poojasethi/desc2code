#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    long int t ;
    scanf("%ld",&t) ;
    while(t--)
    {
         double k,gaina,gainb,i=0.0;
            scanf("%lf",&k) ;
            if(k>1-k)
            {
                i=10000 ;
            }
            gaina=2*(1-k)*i+i;
            gainb=2*k*(10000-i)+(10000-i) ;
            printf("%lf\n",(k*gaina+(1-k)*gainb) ) ;
    }
    return 0;
}
