#include<stdio.h>
using namespace std;
int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        double a;
        scanf("%lf",&a);
        if(a<0.5)
            a=1-a;

        double ans=10000*((3*a)-(2*a*a));
        printf("%0.6lf\n",ans);
        //cout<<setprecision(6)<<ans<<"\n";
    }
}
