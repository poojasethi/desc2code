using namespace std;
#define EPS 0.000001
#define REP(i,a,b) \
    for(int(i)=int(a);i<=int(b);i++)

int main(){
    int T;
    double pa,result;
    scanf("%d",&T);
    while(T--){
        scanf("%lf",&pa);
        if((1-2*pa) > EPS)
            pa = (1-pa);
        result = (2*(1-pa)*10000 + 10000)*pa;
        printf("%.12lf\n",result);
    }
    return 0;
}
