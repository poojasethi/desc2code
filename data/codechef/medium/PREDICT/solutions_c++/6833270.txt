#include<stdio.h>
#include<math.h>
 
int main()
{
 
    int t;
    scanf("%d",&t);
 
    while(t--)
    {
 
        double px;
        scanf("%lf",&px);
        double m=10000.0;
 
        double ans1,ans2;
 
        ans1=  m*(px + 1 - (2*px*px));
        ans2=  m*(3*px-(2*px*px)) ;
 
        if(ans1>ans2)
        printf("%.8lf\n",ans1);
        else
         printf("%.8lf\n",ans2);
 
    }
    return 0;
}
