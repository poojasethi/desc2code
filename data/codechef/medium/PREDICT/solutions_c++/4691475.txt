#include<iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include<stdlib.h>
#include<vector>
#include<algorithm>
using namespace std;


int main()
  {

    int t;
    
    cin>>t;
    while(t--)
       {
     double pa;
      scanf("%lf",&pa);
      
      
      if(pa>=0.5)
         printf("%lf\n",(10000+10000*(1-pa)*(2*pa-1)));
         else
          printf("%lf\n",(10000+10000*(pa)*(2*(1-pa)-1)));
       
       
       
              
       }

   //system("pause");
    return 0;

  }
