#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)   {
        double pa;
        cin >> pa;
        if(pa >= 0.50000)
            printf("%.10lf\n",2*10000*pa+10000.0*pa - 20000*pa*pa);
        else
            printf("%.10lf\n",10000+(10000.0*pa - 20000*pa*pa));
        }
}
