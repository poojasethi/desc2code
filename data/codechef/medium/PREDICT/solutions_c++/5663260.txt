#include <bits/stdc++.h>
using namespace std;

double prob(double p)
{
    return p * (2 * (1 - p) + 1);
}

int main() {
    int t;
    double p,a,b,res;
    cin >> t;
    while (t--)
    {
        cin >> p;
        a = prob(p);
        b = prob(1 - p);
        res = max(10000 * a, 10000 * b);
        printf("%.9lf\n", res);
    }
    return 0;
}
