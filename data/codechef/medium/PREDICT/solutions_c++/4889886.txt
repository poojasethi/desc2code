#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{
    int t;
    double pa,pb;
    double ans;
    cin>>t;
    while(t--)
    {
        scanf("%lf",&pa);
        pb=1-pa;
        if(pa>pb)
        {
            ans=pa*(10000+(2*pb)*10000);
            printf("%lf\n",ans);
        }
        else
        {
            ans=pb*(10000+(2*pa)*10000);
            printf("%lf\n",ans);

        }
    }

return 0;
}
