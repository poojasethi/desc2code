mod = 10 ** 9 + 7
def powmod(n, p):
    if p == 0:
    	return 1
    if p == 1:
        return n % mod
    if p % 2 == 1:
        return n * powmod(n, p - 1) % mod
    x = powmod(n, p / 2) % mod
    return x * x % mod
t = int(raw_input())
for i in xrange(t):
    n, k = map(int, raw_input().split())
    print k * powmod(k - 1, n - 1) % mod
