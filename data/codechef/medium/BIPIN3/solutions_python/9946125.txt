MOD=1000000007

def power(x , y):
    res=1
    while y>0:
        if y%2==0:
            x=x*x%MOD
            y/=2
        else:
            res=res*x%MOD
            y-=1
    return res


t=int(raw_input())
for i in xrange(t):
    n , k=map(int , raw_input().split())
    print k*power(k-1 , n-1)%MOD
