#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

const int MAXN = 1E7 + 5;
const ll oo = 1E9 + 7;

int N, K, Q;

ll A[MAXN], minK[MAXN];

deque <int> S;

void generateA() {
	ll a, b, c, d, e, f, r, s, t, m;
	cin >> a >> b >> c >> d >> e >> f >> r >> s >> t >> m >> A[1];
	ll tmp = (ll)t * t % s;	
	for (int i = 2; i <= N; ++i) {
		if (tmp <= r)	
			A[i] = ((ll)a * A[i - 1] % m * A[i - 1] + (ll)b * A[i - 1] + c) % m;
		else
			A[i] = ((ll)d * A[i - 1] % m * A[i - 1] + (ll)e * A[i - 1] + f) % m;
		tmp = (ll)tmp * t % s;

	}
}


int main() {
	// freopen("input.txt", "r", stdin);
	ios_base::sync_with_stdio(0);

	cin >> N >> K >> Q;
	generateA();
	for (int i = 1; i <= N; ++i) {
		while (S.size() && A[S.back()] >= A[i]) S.pop_back();
		S.push_back(i);
		if (i >= K) {
			minK[i - K + 1] = A[S.front()];
		}
		if (S.front() <= i - K + 1)
			S.pop_front();
	}

	ll res1 = 0, res2 = 1;
	int L1, La, Lc, Lm, D1, Da, Dc, Dm;
	cin >> L1 >> La >> Lc >> Lm >> D1 >> Da >> Dc >> Dm;
	for (int i = 1; i <= Q; ++i) {
		L1 = ((ll)La * L1 + Lc) % Lm;
		D1 = ((ll)Da * D1 + Dc) % Dm;
		int L = L1 + 1;
		int R = min(L + K - 1 + D1, N);
		ll val = min(minK[L], minK[R - K + 1]);
		res1 += val;
		res2 = res2 * val % oo;
	}
	cout << res1 <<" "<< res2 << endl;
}
