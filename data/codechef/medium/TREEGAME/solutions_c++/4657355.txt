#include <iostream>
#include<stdio.h>
#include<vector>
#include<string.h>
using namespace std;
 int lev[10004];
 vector<int> adj[10004];

void dfs(int n,int l,int p,int *maxl)
{
    lev[l]++;
    if(l>*maxl) *maxl=l;
    int i;
    for(i=0;i<adj[n].size();i++)
    {
        if(adj[n][i]!=p)
        {
            dfs(adj[n][i],l+1,n,maxl);
        }
    }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
       int n,a,b,i,j;

        scanf("%d",&n);
        for(i=0;i<n-1;i++)
        {
            scanf("%d%d",&a,&b);
            adj[a].push_back(b);
            adj[b].push_back(a);
        }
        int maxl=0;
        dfs(1,0,-1,&maxl);
        i=0,j=maxl;
        int cnt=0;
        while(i<=j)
        {
            i++;
            cnt++;
            if(i<=j)
            {
                if(lev[j]==1) j--;
                else lev[j]--;
                cnt++;
            }
        }
        printf("%d\n",cnt);
        for(i=1;i<=n;i++) adj[i].clear();
        memset(lev,0,sizeof(lev));
    }
    return 0;
}
