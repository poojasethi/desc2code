#include <iostream>
#include <vector>

using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
        int n, o, in, p, r, k;
        cin >> n >> o >> in >> p >> r >> k;
    int travel = 0;
    int avail = n;
    vector<int> vis(n);
    vector<int> val(n);
        for(int i = 1; i <= k; i++) {
            if(i%r == 0 && o) {
                in++;
                o--;
            }
            for(int j = 0; j < n; j++) {
                if(vis[j] && i-val[j] == p) {
                    val[j] = 0;
                    vis[j] = 0;
                    travel++;
                    avail++;
                }
                if(!vis[j] && in) {
                    vis[j] =1;
                    val[j] = i;
                    in--;
                    avail--;
                }
            }
        }
        cout << avail << " " << travel << " " << o << " " << in << endl;
    }
}
