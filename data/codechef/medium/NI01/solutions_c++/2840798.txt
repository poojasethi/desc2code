#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
int n,m,q,p,r,k;
const int N=100005;
int used[N];
int main(){
	int T;
	scanf("%d",&T);
	while(T--){
		scanf("%d%d%d%d%d%d",&n,&m,&q,&p,&r,&k);
		int x=min(n,q);
		n-=x;
		q-=x;
		memset(used,0,sizeof(used));
		used[0]=x;
		int cnt=0;
		for(int t=1;t<=k;t++){
			if(t-p>=0){
				n+=used[t-p];
				cnt+=used[t-p];
			}
			if(t%r==0&&m){
				q++;
				m--;
			}
			x=min(n,q);
			n-=x;
			q-=x;
			used[t]=x;
		}
		printf("%d %d %d %d\n",n,cnt,m,q);
	}
	return 0;
}
