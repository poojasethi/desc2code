inline int fr()
{
        int ip=0;
        char p=getchar_unlocked();
        for(; p<33;) {
                p=getchar_unlocked();
        }
        while(p>32) {
                ip = (ip << 3) + (ip << 1) + (p - '0');
                p=getchar_unlocked();
        }
        return ip;
}
