#include<cstdio>
#include<algorithm>
using namespace std;
int t,nc,nm,ng,p,r,ttime,counti,cars;
int gate[1000000],val[100007];
int i,j,sum;
int main()
{
scanf("%d",&t);
while(t--)
{
scanf("%d %d %d %d %d %d",&nc,&nm,&ng,&p,&r,&ttime);
sum=nm;
val[0]=0;
for(i=0;i<=ttime;i++)
{
    gate[i]=0;
    val[i]=0;



}




for(i=r;i<=ttime;i=i+r)
{
    if(sum>0)
    {

    sum--;
    gate[i]=1;} }
counti=0;
for(j=0;j<=ttime;j++)
{    //printf("hi          j===%d          %d             nc===%d nm===%d ng=====%d   counti===%d\n\n\n",j,val[j],nc,nm,ng,counti);
    nc=nc+val[j];
    if(val[j])
    counti=counti+val[j];


    if(gate[j]!=0)
{
    nm--;

    ng=ng+gate[j];}

if(ng>=nc)
{
    val[j+p]=nc;

    ng=ng-nc;nc=0;

}
else
{
    val[j+p]=ng;
    nc=nc-ng;
    ng=0;
}
//printf("j===%d          %d             nc===%d nm===%d ng=====%d   counti===%d\n",j,val[j],nc,nm,ng,counti);





}










printf("%d %d %d %d\n",nc,counti,nm,ng);
}
return 0;
}
