#include<iostream>
#include<algorithm>
#include<cstdio>
#include<vector>
#include<queue>
#include<deque>
#include<map>
#include<cstdlib>
#include<cmath>
#include<utility>
typedef long long int ll;
using namespace std;


int main()
{
    int t,car,pmus,p,r,k,pgate,totalpas,sel,i,cnt;
    cin>>t;
    while(t--)
    {
        cnt=0;
        queue<pair <int,int> > q;
        pair<int,int> dummy;
        cin>>car>>pmus>>pgate>>p>>r>>k;
        totalpas=pmus+pgate;
        for(i=0;i<=k;i++)
        {
            if(!q.empty())
            {
                dummy=q.front();
                if(dummy.first+p==i)
                {
                    car+=dummy.second;
                    q.pop();
                }
            }
            if(i!=0&&i%r==0&&pmus)
            {
                pmus--;
                pgate++;
            }
            sel=min(pgate,car);
            q.push(make_pair(i,sel));
            car-=sel;
            pgate-=sel;
            cnt+=sel;
        }
        ///totalpas-=(pmus+pgate);
        while(!q.empty())
        {
            dummy=q.front();
            cnt-=dummy.second;
            q.pop();
        }
        cout<<car<<" "<<cnt<<" "<<pmus<<" "<<pgate<<endl;
    }
    return 0;
}
