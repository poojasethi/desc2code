#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
	int t,n,p,k,r,nm,np,op1,op2,op3,op4,i,j;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d %d %d %d %d",&n,&nm,&np,&p,&r,&k);
		int cars[n],wc=0;
		op4=np;
		op2=0;
		op3=nm;
		fill(cars,cars+n,0);				
		for(i=0;i<=k;i++)
		{			
			if(i>0&&i%r==0)
			{
				if(op3>0)
				{
					op4++;
					op3--;
				}
			}			
			for(j=0;j<n;j++)
			{
				if(cars[j]==0)
				{
					if(op4>0)
					{
						cars[j]=p;
						op4--;
					}
				}
				else
				{
					cars[j]--;
					if(cars[j]==0) 
					{
						op2++;
						if(op4>0)
						{
							cars[j]=p;
							op4--;
						}
					}
				}
			}			
		}
		op1=0;
		for(i=0;i<n;i++)
		{						
			if(cars[i]==0)
				op1++;			
		}
		printf("%d %d %d %d\n",op1,op2,op3,op4);
	}
	return 0;
}
