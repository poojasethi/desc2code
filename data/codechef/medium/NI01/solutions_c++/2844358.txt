#include<iostream>
using namespace std;
int main()
{
    int te,a,b,c,p,r,k,i;
    cin>>te;
    while(te--)
    {
              cin>>a>>b>>c>>p>>r>>k;
              int a1=a,a2=0,a3=b,a4=c;
              int t[110003]={0};
              for(i=0;i<=k;i++)
              {
                                    if(i!=0 && i%r==0)
                                    {
                                            if(a3)
                                            {
                                                  a3--;
                                                  a4++;
                                            }
                                    } 
                                    if(t[i])
                                    {
                                           a1+=t[i];
                                           a2+=t[i];
                                    }   
                                    int k=min(a1,a4);
                                    a4-=k;
                                    a1-=k;
                                    t[p+i]+=k; 
              }
              cout<<a1<<" "<<a2<<" "<<a3<<" "<<a4<<endl;
    }
    return 0;
}
