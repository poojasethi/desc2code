#include<iostream>
#include<cstring>
using namespace std;
int gcd(int a,int b)
{
    if(b==0) return a;
    return gcd(b,a%b);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
              int n,m,q,p,r,k,cp=0;
              cin>>n>>m>>q>>p>>r>>k;
              int A[n];
              memset(A,-1,sizeof(A));
              int gc=gcd(p,r);
              int qq=0;
              for(int i=0;i<n;i++)
              {
                      if(q-qq==0) break;
                      if(A[i]==-1)
                      {
                                   A[i]=0;
                                   qq++;
                      }
              }
              q=q-qq;
              for(int i=gc;i<=k;i+=gc)
              {
                      for(int j=0;j<n;j++)
                      {
                              if(A[j]!=-1 && A[j]+p==i)
                              {
                                          A[j]=-1;
                                          cp++;
                              }
                      }
                      if(i%r==0)
                      {
                                if(m>0)
                                {
                                m--;
                                q++;
                                }
                      }
                      qq=0;
                      for(int j=0;j<n;j++)
                      {
                              if(q-qq==0) break;
                              if(A[j]==-1)
                              {
                                           A[j]=i;
                                           qq++;
                              }
                      }
                      q=q-qq;
                      /*cout<<m<<" "<<q<<" ";
                      for(int j=0;j<n;j++)
                              cout<<A[j]<<" ";
                      cout<<"\n";*/
              }
              int cc=0;
              for(int i=0;i<n;i++)
              {
                      if(A[i]==-1)
                                  cc++;
              }
              cout<<cc<<" "<<cp<<" "<<m<<" "<<q<<"\n";              
    }
    return 0;
}
