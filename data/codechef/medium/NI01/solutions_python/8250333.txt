#include<bits/stdc++.h>
using namespace std;

int main()
{
	int N;
	scanf("%d",&N);
	while (N--)
	{
		int n,m,wait,p,r,k;
		int i;
		queue<int>carq,waitq;
		scanf("%d%d%d%d%d%d",&n,&m,&wait,&p,&r,&k);
		for (i=0;i<wait;i++)
			waitq.push(0);
		int tot=0;
		// wait is number of people waiting 
		// and m is number of people in the meuseum !
		int car=n; // car left!
		for (i=1;i<=k;i++)
		{
			if (i%r==0&&m>0)
			{
				wait++;
				waitq.push(i);
				m--;
				//printf("For i:%d person arrived wait is :%d\n",i,wait);
			}
			
			while (!carq.empty()&&carq.front()+p==i)
			{
				tot++;
				car++;
				carq.pop();
				//printf("For i:%d a car returned !\n",i);
			}
			int z=min(car,wait),c;
			for (c=0;c<z;c++)
			{
				wait--;
				car--;
				carq.push(i);
				//printf("for i:%d taken a car for a ride !\n",i);
			}
		}
		printf("%d %d %d %d\n",car,tot,m,wait);
	}
	return 0;
}
