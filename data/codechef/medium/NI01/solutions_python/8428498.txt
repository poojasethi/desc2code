#include <bits/stdc++.h>
using namespace std;



int main()
{
	int t;
	scanf("%d",&t);
	//t=1;
	
	while(t--)
	{
		int n,tp,gp,p,r,k,cw,rc;
		scanf("%d%d%d%d%d%d",&n,&tp,&gp,&p,&r,&k);
		queue<pair<int,int> > gque;
		cw=n;rc=0;tp+=gp;
		for (int i = 0; i <= k; ++i)
		{
			if (!gque.empty()&&gque.front().second==i)
			{
				cw+=gque.front().first;
				rc+=gque.front().first;
				//printf("asfasfgasd%d\n",gque.front().first);
				gque.pop();
			}
			if (i!=0&&(i%r)==0&&(tp-rc-gp-(n-cw))>0)
			{
				gp++;
			}
			if (cw>0&&gp>0)
			{
				if (cw<=gp)
				{
					gp-=cw;
					gque.push(make_pair(cw,i+p));
					cw=0;
				}
				else
				{
					cw-=gp;
					gque.push(make_pair(gp,i+p));
					gp=0;
				}
			}
		}
		printf("%d %d %d %d\n",cw,rc,(tp-rc-gp-(n-cw)),gp );


	}
}
