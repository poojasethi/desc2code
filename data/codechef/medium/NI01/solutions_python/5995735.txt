#include<stdio.h>
int main()
{
	int N,cars,TP,PQ,p,r,k,time,PC,carp,i;
	scanf("%d",&N);
	while(N--)
	{
		scanf("%d%d%d%d%d%d",&cars,&TP,&PQ,&p,&r,&k);
		int car[51]={0};
		PC=0;
		for(time=1;time<=k;time++)
		{
			for(i=1;i<=cars;i++)
			{
				if(car[i]==0)
				{
					if(PQ>0)
						PQ--;
					else
						break;
					car[i]++;
				}
			} 			// 1 sec. completed
			if(time%r==0&&TP>0)
			{
				TP--;
				PQ++;
			}
			for(i=1;i<=cars;i++)
			{
				if(car[i]==p)
				{
					PC++;
					car[i]=0;
				}
				else if(car[i]!=0)
					car[i]++;
			}
		}
		carp=0;
		for(i=1;i<=cars;i++)
		{
			if(car[i]==0)
				carp++;
		}
		if(PQ>0&&carp>0) //At time k some passengers at queue will be seated in car.
		{
			if(PQ>=carp)
			{
				PQ-=carp;
				carp=0;
			}
			else
			{
				carp-=PQ;
				PQ=0;
			}
		}
		printf("%d %d %d %d\n",carp,PC,TP,PQ);
	}
	return 0;
}
