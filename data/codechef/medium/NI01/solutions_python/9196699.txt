#include <bits/stdc++.h>
using namespace std;

priority_queue<int, vector<int>, greater<int> > Q;

int main()
{
    int T;
    scanf("%d", &T);

    while(T--)
    {
        int safari, museum, gate, p, r, k;
        scanf("%d %d %d %d %d %d", &safari, &museum, &gate, &p, &r, &k);

        for(int i=1; i<=min(safari, gate); i++)
            Q.push(p);
        for(int i=min(safari, gate)+1; i<=safari; i++)
            Q.push(-1);

        gate-= min(safari, gate);
        int complete = 0, waiting = 0;

        for(int i=1; i<=k; i++)
        {
            if(i%r==0 && museum)
            {
                museum--;
                gate++;
            }

            if(!Q.empty() && Q.top()>i)
                continue;

            while(gate && !Q.empty() && Q.top()<=i)
            {
                int x = Q.top();

                if(x==-1)
                {
                    Q.pop();
                    Q.push(i+p);
                    gate--;
                    continue;
                }

                complete++;
                Q.pop();
                Q.push(i+p);
                gate--;
            }
        }

        while(!Q.empty())
        {
            int x = Q.top();
            Q.pop();

            if(x!=-1 && x<=k)
                complete++, waiting++;
        }

        printf("%d %d %d %d\n", waiting, complete, museum, gate);
    }

    return 0;
}
