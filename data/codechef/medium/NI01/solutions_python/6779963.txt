#include<stdio.h>
void main(){
short t,count,com,q,n,m,p,r,i;
int k,time;
scanf("%hd",&t);
while(t--){
scanf("%hd%hd%hd%hd%hd%d",&n,&m,&q,&p,&r,&k);
time=0;
short ride[n];
memset(ride,0,sizeof(ride));
com=0;count=0;
for(i=0;i<n;i++){
if(q!=0){
ride[i]=p;
q--;
}
}
while(1){
//printf("%d %hd %hd %hd %hd %hd%hd%hd\n",time,count,com,m,q,ride[0],ride[1],ride[2]);
time++;
if(time>k) break;
if(time%r==0 && m>0) {q++;m--;}
for(i=0;i<n;i++){
if(ride[i]!=0){
ride[i]--;
if(ride[i]==0) com++;
}
if(q!=0 && ride[i]==0){
ride[i]=p;
q--;
}
}
}
for(i=0;i<n;i++) if(ride[i]==0) count++;
printf("%hd %hd %hd %hd\n",count,com,m,q);
}
}
