x=input()
for i in range(x):
    n,k=map(int,raw_input().split())
    l=list(map(int,raw_input().split()))
    l.sort()
    sum1=0
    sum2=0
    for j in range(n):
        if j<k:
            sum1-=l[j]
        else:
            sum1+=l[j]
    l.sort(reverse=True)
    for j in range(n):
        if j<k:
            sum2+=l[j]
        else:
            sum2-=l[j]
    print max(sum1,sum2)
