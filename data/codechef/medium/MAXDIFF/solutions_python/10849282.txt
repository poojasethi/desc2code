t = int(raw_input())
while t > 0:
	n,k = map(int, raw_input().split())
	lst = map(int, raw_input().split())
	lst = sorted(lst)
	sum1 = abs(sum(lst[:k:]) - sum(lst[k::]))
	sum2 = abs(sum(lst[:n-k:]) - sum(lst[n-k::]))
	if sum1 > sum2 :
		print sum1
	else :
		print sum2
	t -= 1
