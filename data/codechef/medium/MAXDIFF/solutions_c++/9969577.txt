using namespace std;
 
long long get_min(long long a,long long b)
{
    return a>b?b:a;
}
 
int main()
{
    //freopen("practice.in","r",stdin);
    int t;
    scanf("%d",&t);
    //t=1;
    while(t--)
    {
        long long n,k,arr[100002],ans=0,child=0;
        scanf("%lld %lld",&n,&k);
        for(long long i=0;i<n;i++)
        {
            scanf("%lld",&arr[i]);
            ans += arr[i];
        }
        sort(arr,arr+n);
        k = get_min(k,n-k);
        for(int i=0;i<k;i++)
        {
            child += arr[i];
        }
        printf("%lld\n",ans - 2*child);
    }
    return 0;
}
