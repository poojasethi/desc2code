#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        int a[101];
        cin>>n>>k;
        int i,j;
        int diff=0,sum=0,tot_sum=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            tot_sum+=a[i];
        }

        sort(a,a+n);
        for(i=0;i<k;i++)
            diff+=a[i];
        for(i=0;i<k;i++)
            sum+=a[n-1-i];
        if(sum>tot_sum-diff)
            cout<<sum-(tot_sum-sum);
        else
            cout<<tot_sum-diff-diff;
        cout<<endl;

    }
    return 0;
}
