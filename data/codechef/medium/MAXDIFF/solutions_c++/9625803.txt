#include <iostream>
#include <algorithm>
#include <math.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin >> t;
	while(t--) {
		int n, i, k;
		long long int s = 0, s1 = 0;
		cin >> n >> k;
		long long int a[n];
		for(i = 0; i < n; i++) {
			cin >> a[i];
			s+=a[i];
		}
		sort(a, a+n);
		if(k <= n/2) {
			for(i = 0; i < k; i++)
			s1+=a[i];
		}
		else {
			for(i = 0; i < n-k ; i++)
			s1+=a[i];
		}
		cout << s-2*s1 << endl;
	}
	return 0;
}
