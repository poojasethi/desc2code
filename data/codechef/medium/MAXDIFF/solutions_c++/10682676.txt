#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int arr[1000];
	long int cases;
	long int len;
	long int num;
	long int max;
	long int index;
	long int sum;
	long int total;
	cin>>cases;
	for(int i = 0 ; i<cases ; i++)
	{
	    total = 0;
		sum = 0;
		cin>> len >> num; 
		num = min(num,len-num);
		
		for(int k = 0 ; k<len ; k++)
		{
	    	cin>>arr[k];
	    	total+=arr[k];
	    }
		
		for(int j = 0 ; j<len-num; j++)
		{
			max = 0;
			for(int k = 0 ; k<len;k++)
			{
				if(arr[k]>max)
				{
					max = arr[k];
					index = k;
				}
			}
			sum+=arr[index];
			arr[index] = 0;
	 }
	
		
		cout<<sum -(total - sum)<<endl;
		
	}
}
