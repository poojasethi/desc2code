#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,k,b;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		int a[1000000];
		for(int i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		int s=0,f=0;
		b=min(k,n-k);
		for(int i=0;i<b;i++)
		s=s+a[i];
		for(int i=b;i<n;i++)
		f=f+a[i];
		cout<<f-s<<endl;
	}
	
	
	return 0;
}
