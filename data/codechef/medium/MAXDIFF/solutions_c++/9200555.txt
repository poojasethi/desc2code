#include<cstdio>
#include<cstdlib>
using namespace std;
int compare(const void *a, const void*b)
{
	if (*(long int*)a > *(long int*)b)
		return 1;
	if (*(long int*)a == *(long int*)b)
		return 0;
	if (*(long int*)a < *(long int*)b)
		return -1;
}
int main()
{
	long int t, n, k;
	long int W[100];
	scanf("%ld", &t);
	while (t--)
	{
		scanf("%ld %ld", &n, &k);
		if (k > n - k)
			k = n - k;
		for (int i = 0; i < n; ++i)
		{
			scanf("%ld", &W[i]);
		}
		qsort(W, n, sizeof(int), compare);
		long long sum1 = 0, sum2 = 0;
		for (int i = 0; i < k; ++i)
			sum1 += W[i];
		for (int i = k; i < n; ++i)
			sum2 += W[i];
		printf("%lld\n", sum2 - sum1);
	}
	return 0;
}
