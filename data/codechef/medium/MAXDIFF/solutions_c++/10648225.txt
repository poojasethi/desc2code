#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k;
        scanf("%d%d",&n,&k);
        long long weights[n];
        long long sum=0;
        for(int i=0;i<n;i++){
            scanf("%lld",&weights[i]);
            sum+=weights[i];
        }
/*
        for(int i=0;i<n;i++){
            printf("%lld\t",weights[i]);
        }
*/
        sort(weights,weights+n);
        long long S1=0;
        long long S2=0;
        for(int i=0;i<k;i++){
            S1+=weights[i];
            S2+=weights[n-1-i];
        }
        long long maxdiff=max(abs((sum-S1)-S1),abs((sum-S2)-S2));
        printf("%lld\n",maxdiff);

    }
}
