#include<iostream>
#include<stdlib.h>
using namespace std;
int cmp1(const void *a,const void *b){
    return *(int *)b-*(int *)a;
}
int cmp2(const void *a,const void *b){
    return *(int *)a-*(int *)b;
}
int main(){
    int t;
    cin>>t;
    while(t--){
        int n,k;
        cin>>n>>k;
        long long int w[n],i;
        for (i=0;i<n;i++){
            cin>>w[i];
        }

        int sumc=0,sump=0;
        if (k>n-k){
            qsort(w,n,sizeof(long long int),cmp1);
            for (i=0;i<k;i++){
                sumc+=w[i];
            }
            for (i=k;i<n;i++){
                sump+=w[i];
            }
            cout<<sumc-sump<<endl;
        }
        else{
            qsort(w,n,sizeof(long long int),cmp2);
            for (i=0;i<k;i++){
                sump+=w[i];
            }
            for (i=k;i<n;i++){
                sumc+=w[i];
            }
            cout<<sumc-sump<<endl;
        }
    }
    return 0;
}
