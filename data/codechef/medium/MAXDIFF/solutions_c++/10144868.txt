#include <bits/stdc++.h>
using namespace std;
int comp(const void *p, const void *q){
	int l = *(const int *)p;
    int r = *(const int *)q;
    if(l<r) return -1;
    else return 1;
}

int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		int i, n,k,arr[100005],sum1=0, sum2=0;
		scanf("%d %d", &n, &k);
		for( i=0; i<n; i++){
			scanf("%d", &arr[i]);
		}
		qsort(arr,n,sizeof(int), comp);
		for(i=0; i<k; i++){
			sum1= sum1+ arr[i];
		}
		for(i=k; i<n; i++){
			sum2= sum2+arr[i];
		}
		int m1= sum2-sum1;
		int sum3=0,sum4=0;
		for(i=0; i<n-k; i++){
			sum3= sum3+ arr[i];
		}
		for(i=n-k; i<n; i++){
			sum4= sum4+arr[i];
		}
		int m2= sum4-sum3;
		if(m1<m2) printf("%d\n", m2 );
		else printf("%d\n", m1 );

	}
	return 0;
}
