#include<bits/stdc++.h>
#include<cmath>
using namespace std;

main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int a[105];
        int i,s1=0,s2=0,s=0;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
            s+=a[i];
        }

        sort(a,a+n);

        if(k<=n/2)
        {

        for(i=0;i<k;i++)
            s1 += a[i];

        for(;i<n;i++)
            s2 += a[i];
        cout<<s2-s1<<endl;
        }
        else
        {
         i=n-1,s1=0,s2=0;
        for(;i>n-k-1;i--)
            s1+=a[i];

        for(;i>=0;i--)
            s2 += a[i];

        cout<<abs(s2-s1)<<endl;

    }
}
return 0;
}
