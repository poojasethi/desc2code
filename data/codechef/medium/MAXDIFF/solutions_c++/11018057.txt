#include <iostream>
#include<cmath>
#include<algorithm>
using namespace std;

int main() {
	int t,n,k,*items,sum1,sum2,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		items=new int[n];
		for(i=0;i<n;i++)
		cin>>items[i];
		sort(items,items+n);
		sum1=0;
		sum2=0;
		if(n-k<k)
		{
			for(i=0;i<n-k;i++)
			sum1+=items[i];
		}
		else 
		{
			for(i=0;i<k;i++)
			sum1+=items[i];
		}	
		for(j=i;j<n;j++)
		sum2+=items[j];
		cout<<abs(sum1-sum2)<<endl;
	}	
	return 0;
}
