#include<iostream>
using namespace std;
void sort(int C[],int b)
{
	int i,key,j;
	for(i=1;i<b;i++)
	{
		key=C[i];
		j=i-1;
		while(key<C[j] && j>=0)
		{
			C[j+1]=C[j];
			j=j-1;
		}
		C[j+1]=key;
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
       int a,b,p,c,d,e,f,g,h;
       cin>>a>>b;
       int A[a];
       d=0;
       for(c=0;c<a;c++)
       {
       	cin>>A[c];
         d=d+A[c];
       }
       sort(A,a);
       e=0;
        for(c=(a-b);c<a;c++)
        {
        	e=e+A[c];
        }
        if(e>=(d-e))
        {
        	g=e;
        	h=d-e;
        }
        else
        {
        	g=d-e;
        	h=e;
        }
        f=g-h;
        e=0;
        for(c=0;c<b;c++)
        {
        	e=e+A[c];
        }
        if(e>=(d-e))
        {
        	g=e;
        	h=d-e;
        }
        else
        {

        (g-h)>f ? f=g-h : f ;
       cout<<f<<endl;
	}

}
