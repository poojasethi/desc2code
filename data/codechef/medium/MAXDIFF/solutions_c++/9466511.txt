#include<iostream>
#include<algorithm>
using namespace std;
int main(){

    int t,w,n,i,k;

    cin>>t;

    while(t--){

        cin>>n>>k;

        long long a[n];

        for(i=0;i<n;++i){
            cin>>a[i];
        }

        sort(a,a+n);
        long long sum=0;
        if(k<(n+1)/2){
            for(i=0;i<k;++i) sum+=a[i];
            for(i=k;i<n;++i) sum-=a[i];
        }else{
            for(i=n-1;i>=n-k;--i) sum+=a[i];
            for(i=0;i<n-k;++i) sum-=a[i];
        }
        if(sum<0) sum*=-1;
        cout<<sum<<"\n";

    }

    return 0;
}
