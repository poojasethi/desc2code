#include <iostream>
#include <algorithm>

using namespace std;
int main(){
	int T;
	cin >> T;
	for(int i=0;i<T;i++){
		int N,K;
		cin >> N >> K;
		long long arr[N];
		long long total=0;
		for(int j=0;j<N;j++){
			cin >> arr[j];
			total += arr[j];
		}
		sort(arr,arr+N);
		int k = K < N - K ? K : N - K;
		long long childTotal = 0;
		for(int j=0;j<k;j++){
			childTotal+= arr[j];
		}
		cout << total - childTotal - childTotal << endl;
	}
	return 0;
}
