#include<bits/stdc++.h>
using namespace std;

int main(){

    int t,n,k,i;
    long int weight[1001],sum1,sum2;
    cin>>t;
    while(t--){
    sum1=sum2=0;
    cin>>n>>k;

    for(i=0;i<n;i++)
    cin>>weight[i];

    sort(weight,weight+n);

    if(n-k < k)
    k=n-k;

    for(i=0;i<k;i++)
    sum1+=weight[i];

    for(i=k;i<n;i++)
    sum2+=weight[i];

    cout<<sum2-sum1<<endl;

    }

}
