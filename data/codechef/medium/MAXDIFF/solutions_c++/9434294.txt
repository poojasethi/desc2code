#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        vector<int> a(n);
        for(int i=0;i<n;i++)
            cin>>a[i];
        sort(a.begin(),a.end());
        if(k<=n/2)
        {
        int i=0,sc=0,sp=0;
        for(;i<k;i++)
            sc+=a[i];
        for(;i<n;i++)
            sp += a[i];
        cout<<sp-sc<<endl;
        }
        else
        {
            int i=n-1,sc=0,sp=0;
        for(;i>n-k-1;i--)
            sc+=a[i];
        for(;i>=0;i--)
            sp += a[i];
        cout<<abs(sp-sc)<<endl;
        }
    }
}
