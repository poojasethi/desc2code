#include<iostream>
#include<cstdlib>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
	    long long sum=0,sum1=0,sum2=0;
	    long long a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			sum=sum+a[i];
		}
		sort(a,a+n);
		for(int i=0;i<n;i++)
		{
		  if(i<k)
		   sum1=sum1+a[i];
		  if(i<n-k)
		    sum2=sum2+a[i]; 	
		}
		long long ans1=abs(sum-(2*sum1));
		long long ans2=abs(sum-(2*sum2));
		if(ans1>ans2)
		  cout<<ans1<<endl;
		else
		 cout<<ans2<<endl;  
	}
	return 0;
}
