#include<iostream>
#include<algorithm>
#include<cstdio>

using namespace std;

int main()
{
    int t;

    scanf("%d",&t);

    while(t--)
    {
        int n,k,i,asum=0,bsum=0;

        scanf("%d %d",&n,&k);

        int a[n];

        int b=min(n-k,k);

        for(i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }

        sort(a,a+n);

        for(i=0;i<b;i++)
            asum+=a[i];

        while(i<n)
        {
            bsum+=a[i++];
        }

        printf("%d\n",bsum-asum);

    }

    return(0);
}
