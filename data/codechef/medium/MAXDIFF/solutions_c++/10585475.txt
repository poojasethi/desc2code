#include<bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,k,b;
		cin>>n>>k;
		long long int a[n],sum1=0,sum2=0;
		for(b=0;b<n;b++)
		{
			cin>>a[b];
			sum1=sum1+a[b];
		}
		sort(a,a+n);
		if(k>n/2)
		k=n-k;
		for(int r=0;r<k;r++)
		{
			sum2=sum2+a[r];
		}
		sum1=sum1-sum2;
		cout<<sum1-sum2<<endl;
	
		
		
	}
}
