#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int i,n,k;
		scanf("%d %d",&n,&k);
		if(k>n/2)
		k=n-k;
		int a[n];
		for (i=0;i<n;i++)
		scanf("%d",&a[i]);
		sort(a,a+n);
		long long int sum=0,sum1=0;
		for (i=0;i<n;i++)
		{
			if(i<k)
			sum=sum+a[i];
			else
			sum1=sum1+a[i];
		}
		printf("%d\n",sum1-sum);
	}
}
