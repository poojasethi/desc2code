#include<algorithm>
#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t,n,k,i;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&k);
		int arr[n];
		
		for(i=0;i<n;i++)
		{
			scanf("%d",&arr[i]);
		}
		sort(arr,arr+n);
		int j,a=0,b=0,min;
		min=0;
		int key=0;
		while(key<n)
		{
		
		a=0;
		for(i=0;i<=key;i++)
		{	
			a=a+arr[i];
		}
		
		b=0;
		
		for(j=key+1;j<n;j++)
		{
			b=b+arr[j];	
		}
		
		if(abs(b-a)>=min && (key+1==k || n-key-1==k))
		{
		min=abs(b-a);
		}
		
		
		key++;
		
		
		}
		printf("%d\n",min);
	}
	return 0;
}
