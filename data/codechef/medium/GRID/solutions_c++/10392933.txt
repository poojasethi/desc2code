#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,num,i,j;
    cin>>t;
    while(t--){
        int count=0;
        cin>>num;
        short check_i[num+1],check_j[num+1];
        memset(check_i,0,(num+1)*sizeof(short));
        memset(check_j,0,(num+1)*sizeof(short));
        char str[num][num];
        for(i=0;i<num;++i){
        scanf("%s",str[i]);
        }
    for(j=num-1;j>=0;--j){
        for(i=num-1;i>=0;--i){
            if(str[i][j]=='#')
                check_i[i]=1,check_j[j]=1;
             else if(check_i[i]!=1&&check_j[j]!=1)
                ++count;

        }
    }
    printf("%d\n",count);

    }
return 0;
}
