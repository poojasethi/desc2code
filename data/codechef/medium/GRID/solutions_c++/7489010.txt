#include<bits/stdc++.h>
using namespace std;
//int counter;
int main()
{
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin>>t;
    for(int h=1;h<=t;h++)
    {
int n;
cin>>n;
bool ar[n][n];
bool a_r[n][n];
bool a_b[n][n];
for(int i=0;i<n;i++)
{
    for(int j=0;j<n;j++)
    {
        char c;
        cin>>c;
        if(c=='#')
        {
            ar[i][j]=false;
        }else
        {
            ar[i][j]=true;
        }
    }
}
int temp;
for(int j=0;j<n;j++)
{
    temp=-1;
    for(int i=n-1;i>=0;i--)
    {
        if(ar[i][j]==false)
        {
            a_b[i][j]=false;
            temp=i;
            break;
        }
        a_b[i][j]=true;
    }
    if((temp!=0)&&(temp!=-1))
    {
        for(int i=0;i<temp;i++)
        {
            a_b[i][j]=false;
        }
    }
}
for(int i=0;i<n;i++)
{
    temp=-1;
    for(int j=n-1;j>=0;j--)
    {
        if(ar[i][j]==false)
        {
            a_r[i][j]=false;
            temp=j;
            break;
        }
        a_r[i][j]=true;
    }
    if((temp!=0)&&(temp!=-1))
    {
        for(int j=0;j<temp;j++)
        {
            a_r[i][j]=false;
        }
    }
}
int counter=0;
for(int i=0;i<n;i++)
{
    for(int j=0;j<n;j++)
    {
        if((a_b[i][j]==true)&&(a_r[i][j]==true))
        {
            counter++;
        }
    }
    //cout<<"\n";
}
cout<<counter<<"\n";
}
return 0;
}
