#include<bits/stdc++.h>
using namespace std;
int mat[1001][1001], sum[1001][1001];
int main(void)
{
    int t;
    for(scanf("%d", &t); t--; ){
        int n;
        scanf("%d", &n);
        for(int i=0; i<n; i++)
        {
            char s[1001];
            scanf("%s", s);
            for(int j=0; j<n; j++)
                mat[i][j] = (s[j]=='.')?0:1;
        }
        for(int i=0; i<n; i++){
            sum[i][n-1]=mat[i][n-1];
            for(int j=n-2; j>=0; j--){
                sum[i][j] = sum[i][j+1] + mat[i][j];
            }
        }
        int ans=0;
        for(int j=0; j<n; j++){
            for(int i=n-1; i>=0 && mat[i][j]==0; i--){
                if(sum[i][j]==0)
                    ans++;
            }
        }
        printf("%d\n", ans);
    }
    return 0;
}
