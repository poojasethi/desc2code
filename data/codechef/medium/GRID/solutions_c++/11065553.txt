#include<iostream>
using namespace std;

int main(){
    int test;
    cin>>test;
    while(test--){
        int N;
        cin>>N;
        char arr[N][N];
        for(int i=0;i<N;++i)
            for(int j=0;j<N;++j)
                cin>>arr[i][j];
        for(int j=0;j<N;++j)
            for(int i=N-1;i>=0;--i){
                    if(arr[i][j]=='#')
                        break;
                    arr[i][j]='U';

            }
        int count=0;
        for(int i=N-1;i>=0;--i)
            for(int j=N-1;j>=0;--j){
                    if(arr[i][j]=='#')
                        break;
                    if(arr[i][j]=='U')
                        count++;
            }
        cout<<count<<"\n";
    }
}
