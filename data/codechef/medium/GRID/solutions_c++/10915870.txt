#include <iostream>
#include<stdio.h>
using namespace std;

int main() {
	int t,n,i;
	char a[1001];
	scanf("%d",&t);
	while(t--){
		int ans = 0;
		 int res[1000]={0};       
		scanf("%d",&n);
		for(i = 0; i < n; i++){
			for(int j =0 ; j < n; j++){
				cin>>a[j];
			}
			int block = 0;
			for(int k = n-1 ; k >= 0 ; k-- ){
				if(a[k] == '#'){
					block++;
					res[k] = 0;
					
				}else if(a[k] == '.' && block == 0){
					res[k]++;
				}
			}
		}
		for(i =0 ; i < n ; i++){
			ans += res[i];
		}
		printf("%d\n",ans);
	}
	return 0;
}
