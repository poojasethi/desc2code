#include <bits/stdc++.h>

using namespace std;

int main()
{
    char a[1010][1010];
    int track[1010];
    int t,n,i,j;
    scanf("%d",&t);
    while(t--)
    {
        int counter=0;
        scanf("%d",&n);
        memset(track,-1,sizeof(track));
        for(i=0;i<n;i++)
            scanf("%s",a[i]);
        for(i=0;i<n;i++)
            for(j=n-1;j>=0;j--)
                if(a[j][i]=='#')
                    {
                        track[i]=j;
                        break;
                    }
        for(i=n-1;i>=0;i--)
            for(j=n-1;j>=0;j--)
        {
            if(a[i][j]=='#')
                break;
            if(track[j]>=i)
                continue;
            counter++;
        }
        printf("%d\n",counter);
    }
}
