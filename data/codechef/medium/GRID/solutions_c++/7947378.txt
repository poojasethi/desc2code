#include<cstdio>
#include<vector>
using namespace std;

int main()
{
    int t,n,temp,sum,flag;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        sum=0;
        vector<vector<char> > vec(n,vector<char> (n));
        vector<int> spe(n,0);
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++) scanf(" %c",&vec[i][j]);
        }
        for(int i=n-1;i>=0;i--)
        {
            flag=0;
            for(int j=n-1;j>=0;j--)
            {
                if(vec[i][j]=='.')
                {
                    if(flag==0)
                    {
                    if(spe[j]==0)
                        sum++;
                    }
                }
                else
                {
                    flag=1;
                    spe[j]=1;
                }
            }
        }
        printf("%d\n",sum);
    }
    return 0;
}
