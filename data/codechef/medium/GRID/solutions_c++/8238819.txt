#include<iostream>
#include<vector>
#include<utility>
#include<algorithm>
using namespace std;
char arr[1005][1005];
long long t,rx[1005][1005],ry[1005][1005],n,i,m,j;        
int main()
{
        cin>>t;
        while(t--)
        {
                cin>>n;
                for(i=0;i<n;i++)
                {
                 
                      for(j=0;j<n;j++)
                      cin>>arr[i][j];
                }
                if(arr[n-1][n-1]=='.')
                rx[n-1][n-1]=ry[n-1][n-1]=1;
                else
                rx[n-1][n-1]=ry[n-1][n-1]=0;
                for(i=n-2;i>=0;i--)
                {
                      if(arr[i][n-1]=='.')
                      rx[i][n-1]=1;
                      else
                       rx[i][n-1]=0;
                
                }
                
                for(i=n-2;i>=0;i--)
                {
                        if(arr[n-1][i]=='.')
                        ry[n-1][i]=1;
                        else
                         ry[n-1][i]=0;
                }
                
               for(i=0;i<n;i++)
               {
                        for(j=n-2;j>=0;j--)
                        {
                               if(arr[i][j]=='.')
                               rx[i][j]=rx[i][j+1];
                               else
                               rx[i][j]=0;       
                        }
               
               }
               for(i=n-2;i>=0;i--)
               {
               
                        for(j=0;j<n;j++)
                        {
                                if(arr[i][j]=='.')
                                 ry[i][j]=ry[i+1][j];
                                 else
                                  ry[i][j]=0;
                        }
               }
               
                      int ans=0;
                      for(i=0;i<n;i++)
                      {
                                for(j=0;j<n;j++)
                                if(rx[i][j] && ry[i][j])
                                ans++;
                      }
                      cout<<ans<<"\n";
        }
	 return 0;
}
