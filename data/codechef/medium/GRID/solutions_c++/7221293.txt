using namespace std;

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		vector<string> g(n);
		vector< vector<int> > f(n, vector<int>(n, 1)), f1(n, vector<int>(n, 1));
		for (int i = 0; i < n; i++)
			cin >> g[i];
		for (int i = 0; i < n; i++)
			for (int j = n - 1; j >= 0; j--)
				if (j == n - 1)
				{
					if (g[i][j] == '.')
						f[i][j] = 0;
				}
				else
				{
					if (g[i][j] == '.' && f[i][j + 1] == 0)
						f[i][j] = 0;
				}
		for (int j = 0; j < n; j++)
			for (int i = n - 1; i >= 0; i--)
				if (i == n - 1)
				{
					if (g[i][j] == '.')
						f1[i][j] = 0;
				}
				else
				{
					if (g[i][j] == '.' && f1[i + 1][j] == 0)
						f1[i][j] = 0;
				}
		int ans = 0;
		for (int i = 0; i < n; i++)
			for (int j = 0; j < n; j++)
				if (f[i][j] == 0 && f1[i][j] == 0)
					ans++;
		cout << ans << '\n';
	}
	return 0;
}
