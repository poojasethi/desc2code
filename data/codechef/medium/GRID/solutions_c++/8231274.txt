#include<bits/stdc++.h>
using namespace std;
char s[1005][1005];
int dp[1005][1005][2];
int main()
{
    int t,n,i,j,c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%s",s[i]);
        for(i=0;i<n;i++)
            dp[i][n][0]=dp[i][n][1]=dp[n][i][0]=dp[n][i][1]=1;
        for(i=n-1;i>=0;i--)
            for(j=n-1;j>=0;j--)
        {
            if(dp[i][j+1][0]==1&&s[i][j]=='.')
                dp[i][j][0]=1;
            else
                dp[i][j][0]=0;
            if(dp[i+1][j][1]==1&&s[i][j]=='.')
                dp[i][j][1]=1;
            else
                dp[i][j][1]=0;
        }
        for(i=c=0;i<n;i++)
            for(j=0;j<n;j++)
            if(dp[i][j][0]==1&&dp[i][j][1]==1&&s[i][j]=='.')
            c++;
        printf("%d\n",c);
    }
}
