#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    	int n;
    	scanf("%d",&n);
    	int i,j,k,t,sum=0,row[1001]={0},col[1001]={0},flag;
    	char a[n][n];
    	for(i=0;i<n;i++)
    	scanf("%s",a[i]);
    	
    	for(i=n-1;i>=0;i--)
    	{
    		for(j=n-1;j>=0;j--)
    		{
    			if(a[i][j]=='.' )
    			{
    				if(row[i]==0 && col[j]==0)
    				sum++;
				}
				else
				{
					row[i]=1;
					col[j]=1;
				}
			}
		}
		printf("%d\n",sum);
    }
    return 0;
}
