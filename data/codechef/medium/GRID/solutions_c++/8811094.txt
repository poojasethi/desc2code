#include<iostream>

using namespace std;

int main()
{
	
	int t;
	
	cin>>t;
	
	while(t--)
	{
		int num;
		
		cin>>num;
		
		char a[num][num];
		
		int hor[num];
		
		int ver[num];
		
		for(int i=0;i<num;i++)
		{
			hor[i]=0;
			
			ver[i]=0;
		}
		
		
		
		for(int i=0;i<num;i++)
		{
			for(int j=0;j<num;j++)
			{
				cin>>a[i][j];
				
				
			}
		}
		
		
		
		int coun=0;
		
		for(int i=num-1;i>=0;i--)
		{
			for(int j=num-1;j>=0;j--)
			{
				if(a[i][j]=='#')
					{
						hor[i]=1;
						
						ver[j]=1;
					}
				else
				{
					if(hor[i]!=1 and ver[j]!=1)
				
						coun=coun+1;
				}
			}
		}
		
		cout<<coun<<"\n";
	}
}
