#include <iostream>
#include<cstdio>
using namespace std;
 
int main() {
	int count,k,t,n,i,j;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		char grid[n][n];
		int stone[n];
		for(i=0;i<n;i++)
		{
			stone[i]=-1;
			for(j=0;j<n;j++)
			{
				cin>>grid[i][j];
				if(grid[i][j]=='#')
				 {
				 	stone[i]=j;
				 }
				 
			}
		}
			count=0;
			i=0;
				for(j=0;j<n;j++)
				{
						for(k=n-1;k>=0,grid[k][j]=='.';k--)
						{
							if(stone[k]<j)
							{
								count++;
							}
						}
				}
	 cout<<count<<endl;
	}
	return 0;
}
