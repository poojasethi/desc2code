#include <iostream>
#include <vector>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
		int n,ct=0;
		cin>>n;
		vector<string> v(n);
		vector<vector<int> > x(n+1,vector<int>(n+1,1));
		vector<vector<int> > y(n+1,vector<int>(n+1,1));
		for(int i=0;i<n;i++){
			cin>>v[i];
		}
		for(int j=n-1;j>=0;j--){
			for(int i=n-1;i>=0;i--){
				if(v[i][j]=='.'&&x[i+1][j]==1){
					x[i][j]=1;
				}else{
					x[i][j]=0;
				}
			}
		}
		for(int i=n-1;i>=0;i--){
			for(int j=n-1;j>=0;j--){
				if(v[i][j]=='.'&&y[i][j+1]==1){
					y[i][j]=1;
				}else{
					y[i][j]=0;
				}
			}
		}
		for(int i=n-1;i>=0;i--){
			for(int j=n-1;j>=0;j--){
				if(x[i][j]==1&&y[i][j]==1){
					ct++;
				}
			}
		}
		cout<<ct<<"\n";
	}
	return 0;
}
