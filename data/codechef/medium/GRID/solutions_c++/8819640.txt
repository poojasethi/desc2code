#include<iostream>
using namespace std;

int a[1005][1005];
string s[1005];

int main(){
	std::ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	int n;
	while(t--){
		cin>>n;
		for(int i=0;i<n;i++){
			cin>>s[i];
		}
		for(int i=0;i<n;i++){
			bool blk=false;
			for(int j=n-1;j>=0;j--){
				if(s[i][j]=='#') blk=true;
				
				if(blk){
					a[i][j]=0;
				}else a[i][j]=1;
			}
		}
		
		int cnt=0;
		for(int j=0;j<n;j++){
			for(int i=n-1;i>=0;i--){
				if(s[i][j]=='#') break;
				if(a[i][j]!=0) cnt++;
			}
		}
		cout<<cnt<<"\n";
	}
}
