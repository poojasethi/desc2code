#include <iostream>
using namespace std;
int main() {
    int t;
    cin>>t;
    while(t--){
        int n,i,j,count=0,grid[1001][1001],hori[1001],vert[1001];
        cin>>n;
        for(i=0;i<n;i++){
            for(j=0;j<n;j++){
                grid[i][j]=1;
            }
            hori[i]=1;
            vert[i]=1;
        }
        char a[1001][1001];
        for(i=0;i<n;i++){
            cin>>a[i];
        }
        for(i=n-1;i>=0;i--){
            for(j=n-1;j>=0;j--){
                if(a[i][j]=='#'){
                    hori[i]=0;
                    vert[j]=0;
                }
                if(hori[i]==0||vert[j]==0){
                    grid[i][j]=0;
                }
            }
        }
         for(i=n-1;i>=0;i--){
            for(j=n-1;j>=0;j--){
               if(grid[i][j]==1){
                   count ++;
               } 
            }
         }
        cout<<count<<endl;
    }
}
