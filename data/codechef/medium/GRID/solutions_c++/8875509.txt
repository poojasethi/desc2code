		int left[n][n];
		
		for (i = 0; i < n; i++) {

			for (j = n - 1; j >= 0; j--) {

					stone = true;
				}
				if (stone) {
					left[i][j] = 0;

					left[i][j] = 1;

			}

					stone = true;
				}
				if (stone) {
					up[j][i] = 0;
				} else {
					up[j][i] = 1;
				}
			}
		}
		
		count = 0;
		for (i = 0; i < n; i++) {
			for (j = 0; j < n; j++) {
				if (up[i][j] == 1 && left[i][j] == 1) {
					count++;
				}
			}
		}
		
		printf("%d\n", count);
