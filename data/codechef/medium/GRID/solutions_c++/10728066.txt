#include<iostream>
#include<stdio.h>
using namespace std;
char a[1005][1005],b[1005][1005],c[1005][1005];
int main()
{
    int t,n,i,j,ans,p;

    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        cin>>n;
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
                cin>>a[i][j];
        }
        for(i=n-1;i>=0;i--)
        {
            p=0;
            for(j=n-1;j>=0;j--)
            {
                if(a[i][j]=='#'||p!=0)
                {
                    b[i][j]=0;
                    p++;
                }
                else
                    b[i][j]=a[i][j];
            }
        }
        for(j=n-1;j>=0;j--)
        {
            p=0;
            for(i=n-1;i>=0;i--)
            {
                if(a[i][j]=='#'||p!=0)
                {
                    c[i][j]=0;
                    p++;
                }
                else
                    c[i][j]=a[i][j];
            }
        }
        for(i=n-1;i>=0;i--)
        {
            for(j=n-1;j>=0;j--)
            {
                if(b[i][j]==c[i][j]&&b[i][j]!=0)
                    ans++;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
