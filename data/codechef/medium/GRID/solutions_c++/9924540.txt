#include<stdio.h>
#include<string.h>
typedef long long ll;
char ar[3002][3002];
ll dp1[3002][3002];
ll dp2[3002][3002];
int main()
{
	ll n,t,i,j;
	scanf("%lld",&t);
	ll count=0;
	while(t--)
	{
		memset(dp1,0,sizeof(dp1));
		memset(dp2,0,sizeof(dp2));
		count=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		scanf("%s",ar[i]);
		for(i=n-1;i>=0;i--)
		{
			for(j=n-1;j>=0;j--)
			{
				if(ar[i][j]=='#')
				continue;
				else
					dp1[i][j]=dp1[i][j+1]+1;
			}
		}
			for(j=0;j<n;j++)
			{
					for(i=n-1;i>=0;i--)
		{
				if(ar[i][j]=='#')
				continue;
				else
					dp2[i][j]=dp2[i+1][j]+1;
			}
		}
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				if(dp1[i][j]==n-j && dp2[i][j]==n-i && ar[i][j]!='#')
				count++;
			}
		}
		printf("%lld\n",count);
	}
	return 0;
}
