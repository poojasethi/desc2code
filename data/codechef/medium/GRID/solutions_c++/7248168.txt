#include<bits/stdc++.h>
using namespace std;

int main() {

    int t;cin>>t;
    while(t--) {
        int n;cin>>n;
        char a[n+2][n+2];
        int dp[n+2][n+2];
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++) {
            for(int j=0;j<n;j++) {
                cin>>a[i][j];
            }
        }
        for(int i=0;i<=n;i++)
            dp[n][i]=1;
        int ans=0;
        for(int j=0;j<n;j++) {
            for(int i=n-1;i>=0;i--) {
                if(a[i][j]=='#')
                    break;
                else
                    dp[i][j]=1;
            }
        }
        for(int i=0;i<n;i++) {
            for(int j=n-1;j>=0;j--) {
                if(a[i][j]=='#')
                     break;
                if(a[i][j]=='.') {
                    if(dp[i+1][j]==1)
                        ans++;
                }
            }
        }
        cout<<ans<<endl;
    }
}
