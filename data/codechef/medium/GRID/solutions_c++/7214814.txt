#include<bits/stdc++.h>
#define ll long long
using namespace std ;
int main()
{
	ios_base::sync_with_stdio(false) ;
	ll t , n ;
	cin >> t ;
	while(t--)
	{
		cin >> n ;
		char a[n][n] ;
		bool flag = false ;
		for( ll i = 0 ; i < n ; i++ )
			for( ll j = 0 ; j < n ; j++ )
				cin >> a[i][j] ;
		ll count = 0 ;
		for( ll i = n-1 ; i >= 0 ; i-- )
		{
			//cout << i << "---" ;
			for( ll j = n-1 ; j >= 0 ; j-- )
			{
				//cout << j << " " ;
				if( a[j][i] == '#' )
				{
					flag = true ;
					//cout << j << " " << i << endl ;
					for( ll k = i-1 ; k >= 0 ; k-- )
					{
						if( a[j][k] != '#' )
							a[j][k] = 'h' ;
						else break ;
					}
					//break ;
				}
				else if(a[j][i] == '.'&&(!flag))
					count++ ;
			}
			flag = false ;
		}
	/*	for( ll i = 0 ; i < n ; i++ )
			{
				for( ll j = 0 ; j < n ; j++ )
					cout <<  a[i][j] << " ";
				cout << endl ;
			}*/
		cout << count << endl ;
	}
}
