#include <stdio.h>

int main( int argc, const char* argv[] )
{
	long long numStations, gas, station;
	long long s, e;

	bool sum = false;
	numStations = gas = station = s = e = 0;

	scanf ("%d %d", &numStations, &gas);
	
	long long *stations = new long long[numStations];
	long long *ways = new long long[numStations];
	long long *stops = new long long[numStations];
	long long *waysLevel = new long long[numStations];
	
	scanf ("%d", &station);
	stations[e] = station;
	ways[e] = 1;
	stops[e] = 0;
	waysLevel[e] = 1;

	for( long i = 1; i < numStations; i++)
	{
		waysLevel[i] = 0;
		scanf ("%d", &station);
		
		while ( station - stations[s] > gas)
		{
			sum = true;
			waysLevel[stops[s]] -= ways[s];
			s++;
		}
		stations[++e] = station;
		if(!sum )
		{
			ways[e] = ways[e-1];
			stops[e] = stops[e-1];
		}
		else
		{
			sum = false;
			stops[e] = stops[s] + 1;
			ways[e] = waysLevel[stops[s]]%1000000007;
		}
		waysLevel[stops[e]] += ways[e];
	}
	printf("%lld %lld",stops[e], ways[e]);
}
