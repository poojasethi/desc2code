#include <cstdio>
#include <cstdlib>
#include <cctype>

using namespace std;

const int MODULO = 1000000007;
const int BUFSIZE = 4096;
int buf_pos;
char g_buf[BUFSIZE];

// 1. minimum path
// OPT1[n] : min(OPT1[n-1], OPT1[n-2], OPT1[n-3]) + 1

// 2. the number of mp
// OPT2[n] : sum of (OPT2[n-1], OPT2[n-2], OPT2[n-3])

int readint()
{
    int n;
    while(buf_pos < BUFSIZE && !isdigit(g_buf[buf_pos]))
        buf_pos++;
    // 1. buf_pos == BUF
    // 2. 1234
    //    ^
    if (buf_pos == BUFSIZE)
    {
        buf_pos = 0;
        fread(g_buf, 1, BUFSIZE, stdin);
        while (buf_pos < BUFSIZE && !isdigit(g_buf[buf_pos]))
            buf_pos++;
    }
    // make a number
    n = 0;
    while (buf_pos < BUFSIZE && isdigit(g_buf[buf_pos]))
        n =  10 * n + (g_buf[buf_pos++] - '0');
    if (buf_pos == BUFSIZE)
    {
        buf_pos = 0;
        fread(g_buf, 1, BUFSIZE, stdin);
        while(buf_pos < BUFSIZE && isdigit(g_buf[buf_pos]))
            n = 10 * n + (g_buf[buf_pos++] - '0');
    }
    return n;
}

int main()
{
    int n, m;
    int* station;
    int* opt1;
    int* opt2;
    long long sum;
    int pos1, pos2;

    // init
    n = readint();
    m = readint();
    station = (int*) malloc(sizeof(int) * n);
    for (int i=0; i<n; i++)
    {
        station[i] = readint();
    }
    opt1 = (int*) malloc(sizeof(int) * n);
    opt2 = (int*) malloc(sizeof(int) * n);
    opt1[0] = -1;
    opt2[0] = 1;

    sum = pos1 = pos2 = 0;
    // dp
    for (int i=1; i<n; i++)
    {
        while (station[i] - station[pos1] > m) 
            sum -= opt2[pos1++];
        opt1[i] = opt1[pos1] + 1;
        if(pos2 < pos1)
        {
            pos2 = pos1;
            sum = 0;
        }
        while(opt1[pos1] == opt1[pos2])
        {
            sum += opt2[pos2];
            pos2++;
        }
        opt2[i] = sum % MODULO;
    }

    // print result
    printf("%d %d\n", opt1[n-1], opt2[n-1]);

    return 0;
}
