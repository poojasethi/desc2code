#include <stdio.h>
 
using namespace std;
 
int main( int argc, const char* argv[] )
{
long long numStations, gas, station, sum, stop, counter;
long long s, e;
numStations = gas = station = s = e = stop = 0;
 
scanf ("%d %d", &numStations, &gas);
long long *stations = new long long[numStations];
long long *ways = new long long[numStations];
long long *stops = new long long[numStations];
long long *waysLevel = new long long[numStations];
scanf ("%d", &station);
stations[e] = station;
ways[e] = 1;
stops[e] = 0;
waysLevel[e] = 1;
 
counter = 0;
 
for( long i = 1; i < numStations; i++)
{
waysLevel[i] = 0;
scanf ("%d", &station);
sum = -1;
stop = 0;
int temp = 0;
counter = 0;
while ( station - stations[s] > gas)
{
if ( sum == -1 )
{
stop = stops[s] + 1;
sum = 1;
temp = s;
}
 
waysLevel[stops[s]] -= ways[s];
s++;
}
stations[++e] = station;
if( sum == -1 )
{
ways[e] = ways[e-1];
stops[e] = stops[e-1];
waysLevel[stops[e]] += ways[e];
}
else
{
stops[e] = stops[s] + 1;
ways[e] = waysLevel[stops[s]]%1000000007;
waysLevel[stops[e]] += ways[e];
}
}
printf("%lld %lld",stops[e], ways[e]);
}
