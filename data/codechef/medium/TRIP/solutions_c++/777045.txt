#define DEBUG 0
#define SIZE 1000000
#define MOD 1000000007

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int a[SIZE], b[SIZE], s[SIZE];

int main(void) {
  int n, m, i, j, k;
  long long temp;
//b is number of ways to get to that point in minimum steps
//s=min steps
  
  scanf("%d%d", &n, &m);
  for(i=0;i<n;i++)
    scanf("%d", &a[i]);
  memset(b, 0, sizeof(b));
  b[0]=1;
  s[0]=0;
  temp=0;
  for(i=1,j=0,k=0;i<n;i++) {
    while(a[i]-a[j]>m)
      temp-=b[j++];
    s[i]=s[j]+1;
    while(s[k]==s[j])
      temp+=b[k++];
    b[i]=temp%MOD;
  }
      
  printf("%d %d\n", s[n-1]-1, b[n-1]);

  return 0;
}
