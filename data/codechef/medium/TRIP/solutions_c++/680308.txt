#include<stdio.h>
#include<stdlib.h>
#define MOD 1000000007

int arr[1000000];
int minLength[1000000];
int noOfways[1000000];

int main()
{
	int m,n,i,j,k;
	long long count=0;

	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
		scanf("%d",arr+i);

	noOfways[0]=i=1;
	j=k=count=0;
	while(i<n)
	{
		while(arr[i]-arr[j] > m){
			count=count-noOfways[j];
			j++;
		}
		minLength[i]=minLength[j]+1;
		while(minLength[k]==minLength[j]){
			count=count+noOfways[k];
			k++;
		}
		noOfways[i]=count%MOD;
		i++;
	}
	printf("%d %d\n",(minLength[n-1]-1)%MOD,noOfways[n-1]);
	return 0;
}
