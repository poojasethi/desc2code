#include <cstdio>
using namespace std;

int N, M, gas[1000000], mod=1000000007, cnt[1000000];

main() {
	scanf("%d%d", &N, &M);
	for (int i=0; i<N; i++) {
		scanf("%d", gas+i);
	}
	
	int stops=0, from=0, count1=1, max1=0, count2=0, max2;
	cnt[0]=1;
	
	for (int st=1; st<N; st++) {
		while (gas[from]+M < gas[st]) {
			if (++from > max1) {
				// next stops
				++stops;
				count1=count2;
				count2=0;
				max1=max2;
				--st;
				goto nextst;
			} else {
				// skip station
				count1=(count1-cnt[from-1]+mod)%mod;
			}
		}
		// can stop
		cnt[st]=(cnt[st]+count1)%mod;
		count2=(count2+cnt[st])%mod;
		max2=st;
nextst: ;
	}
	
	printf("%d %d\n", stops, cnt[N-1]);
}
