#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const char fi[] = "TRIP.INP";
const int MAXN = 1000000;
const int MOD = 1000000007;

int a[MAXN];
int q[MAXN];
int best[MAXN];
int n,m;

int main()
{
    int i;
    //freopen(fi, "r", stdin);
    scanf("%d%d",&n, &m);
    for (i=0;i<n;i++) scanf("%d", &a[i]);
    q[0] = 1;
    int ip = 0;
    int in = 0;
    long long sum = 0;
    for (i=1;i<n;i++)
    {
        while (a[ip] < a[i] - m) sum -= q[ip++];
        if (in < ip) in = ip, sum = 0;
        best[i] = best[ip] + 1;
        while (best[in] == best[ip]) sum += q[in++];
        q[i] = sum % MOD;
    }
    /*
    for (i=0;i<n;i++) cout << " " << best[i];
    cout << endl;
    for (i=0;i<n;i++) cout << " " << q[i];
    cout << endl;
    */
    printf("%d %d\n", best[n-1]-1, q[n-1]);
    return 0;
}
