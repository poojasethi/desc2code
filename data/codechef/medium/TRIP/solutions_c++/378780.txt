#include <iostream>
#include <vector>
#include <cstdio>
#include <map>
using namespace std;

int arr[1000000];
int dp[1000000];
int best[1000000];
int N;
int M;
const int MOD = 1000000007;

int main()
{
	cin >> N >> M;

	for(int i=0 ; i<N ; i++)
		scanf("%d", &arr[i]);

	dp[0] = 1;
	int ip = 0;
	int in = 0;
	long long sum = 0;

	for(int i=1 ; i<N ; i++)
	{
		while(arr[ip] < arr[i] - M) sum -= dp[ip++];
		if(in < ip) in = ip, sum = 0;
		best[i] = best[ip] + 1;
		while(best[in] == best[ip]) sum += dp[in++];
		dp[i] = sum % MOD;
	}

	cout << best[N-1]-1 << " ";
	cout << dp[N-1] << endl;

	return 0;
}
