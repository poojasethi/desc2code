#include<iostream>
#include<cstdio>
using namespace std;
#define size 1000001
#define MOD 1000000007
int n, m;
int a[size];
int Dp[size];
int arr[size];
int stops;

void Solve() {
     long long count=0;
     Dp[0]=1;
     int j=0, k=0;
     
     for(int i=1;i<n;i++) {
                      while(a[i]-a[j] > m)
	                                      count-= Dp[j++];
                      
                      arr[i]= arr[j]+1;
                      
                      while(arr[k]==arr[j])
                                          count += Dp[k++];
                      
                      Dp[i]=count%MOD;
    
    }
    
    printf("%d %d\n",arr[n-1]-1, Dp[n-1]);

}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0; i<n; i++)
            scanf("%d",&a[i]);
    Solve();
    return 0;
}
