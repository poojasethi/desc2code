#include <cstdio>
 
#define MOD 1000000007
 
int loc[1000000], ways[1000000], minm[1000000];
 
int main(){
	int M, N, a;
	scanf("%d %d", &N, &M);
	for(int i=0; i<N; i++)
		scanf("%d", loc+i);
	int last=0;
	for(int i=1; i<N; i++){
		while(loc[i]-loc[last]>M)
			last++;
		minm[i]=minm[last]+1;
	}
	ways[0]=1;
	last=0;
	int sum=0, next=1;
	for(int i=1; i<N; i++){
		if(minm[i]!=minm[i-1]){
			sum=next;
			next=0;
			while(minm[last]<minm[i]-1)
				last++;
		}
		while(loc[i]-loc[last]>M){
			sum-=ways[last];
			if(sum<0)
				sum+=MOD;
			last++;
		}
		ways[i]=sum;
		next+=ways[i];
		if(next>=MOD)
			next-=MOD;
	}
	printf("%d %d\n", minm[N-1]-1, ways[N-1]);
	return 0;
}
