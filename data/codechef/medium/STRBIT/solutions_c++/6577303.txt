#include<bits/stdc++.h>
using namespace std;
typedef int ll;
char a[100005];
vector<int> b;
int n;
ll query(ll a)
{
    ll sum=0,tmp;
    while(a)
    {
        sum=(sum+b[a]);
        tmp=a-((a)&(-a));
        a=tmp;
    }
    return sum;
}
void update1(ll a,ll v)
{
    ll tmp;
    while(a<=n)
    {
        b[a]=(b[a]+v);
        tmp=a+((a)&(-a));
        a=tmp;
	}
}
void update(int x,int y,int v)
{
	update1(x,v);
	update1(y+1,-v);
}
int main()
{
	int t,k,x,cnt;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		scanf("%s",&a);
		b.assign(n+2,0);
		cnt=0;
		for(int i=1;i<=n;i++)
		{
			x=query(i);
			if(a[i-1]=='G')
			x+=1;
			if(x%2==0)
			{	
			cnt++;
			update(i,min(i+k-1,n),1);
			}
		}
		printf("%d\n",cnt);
	}
	return 0;
}
