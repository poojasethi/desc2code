#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t, n, k, i, tmp, ans;
    string s;
    cin >> t;
    while (t--) {
        cin >> n >> k >> s;
        int a[n];
        memset(a, 0, sizeof(a));
        ans = 0;
        for (i = 0; i < n; i++) {
            if (i)
                a[i] += a[i-1];
            if ((s[i] == 'R' && a[i]%2 == 0) || (s[i] == 'G' && a[i]%2)) {
                a[i]++;
                tmp = min(n,i+k);
                if (tmp < n)
                    a[tmp]--;
                ans++;
            }
        }
        cout << ans << endl;
    }
    return 0;
}
