const LL N = 1e5;
LL t[2*N];
LL n,k;

void build(){
    for(LL i = n-1;i > 0; --i){
        t[i] = t[i<<1] + t[i<<1|1];
    }
}
void update(LL l,LL r){
    for(l+=n,r+=n; l < r; l >>= 1, r >>= 1){
        if(l&1)t[l++] += 1;
        if(r&1)t[--r] += 1;
    }
}

LL get(LL po){
    LL ans = 0;
    for(po+=n;po > 0; po>>=1){
        ans += t[po];
    }
    return ans;
}

void push() {
  for (int i = 1; i < n; ++i) {
    t[i<<1] += t[i];
    t[i<<1|1] += t[i];
    t[i] = 0;
    }
}
