#define FOR(var,i,j) for(var = i ; var <= j ; ++var)
#define rep(var,n)   for(var=1;var<=n;++var)

typedef long long int ll;

using namespace std;

int main()
{
    ll t;
    scanf("%lld",&t);

    while(t--)
    {
        ll n,k,ans=0,i,sum=0,l,r;
        scanf("%lld %lld",&n,&k);

        string s;
        cin>>s;

        int p[100001]={0};

        for(i=0;i<n;i++)
        {
            int d=(s[i]=='R'?0:1);
            sum=sum^p[i];
            d=d^sum;
            if(d==0)   // 0 based index
            {
                sum^=1;
                p[min(n,i+k)]^=1;
                ans++;
            }
        }

        printf("%lld\n",ans);

    }

    return 0;
}
