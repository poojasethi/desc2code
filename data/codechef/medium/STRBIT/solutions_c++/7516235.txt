#include<iostream>
#include<stdio.h>
char arr[100005];
int count[100005];
int min(int a,int b)
{
    if(a<b)
        return a;
    else
        return b;
}
int main()
{
    int t,n,k,i,sum;
    scanf("%d",&t);
    while(t--)
    {
        sum=0;
        scanf("%d %d",&n,&k);
        scanf("%s",arr);
        for(i=0;i<n;i++)
            count[i]=0;
        for(i=0;i<n;i++)
        {
            if(arr[i]=='R')
                break;
        }
        count[i]++;
        count[min(n,i+k)]--;
        sum++, i++;

        for(;i<n;i++)
        {
            count[i]+=count[i-1];
            if((arr[i]=='R'&&count[i]%2==0)||(arr[i]=='G' && count[i]%2!=0))
            {
                count[i]++;
                count[min(n,i+k)]--;
                sum++;
            }

        }
        printf("%d\n",sum);
    }
    return 0;
}
