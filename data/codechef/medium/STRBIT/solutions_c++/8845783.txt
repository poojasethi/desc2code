#define problem "LATXU2"
#include <cstdio>
#include <iostream>
#include <memory.h>
using namespace std;
const int maxN(1e5 + 10);
typedef long long ll;
int N, K, IT[maxN * 4];;
char A[maxN];
ll S[maxN];

int main(){
    
    int test;
    scanf("%d\n", &test);
    while(test--){
        scanf("%d %d\n", &N, &K);
        gets(A + 1);
        memset(S, 0, sizeof(S));
        int ans(0);
        for(int i = 1; i <= N; i++) if(A[i] == 'G') S[i]++, S[i + 1]--;
        for(int i = 1; i <= N; i++){
            S[i] += S[i - 1];
            if(S[i] % 2 == 0){
                ans++;
                S[i]++;
                S[min(N, i + K - 1) + 1]--;
            }
        }
        printf("%d\n", ans);
    }
}
