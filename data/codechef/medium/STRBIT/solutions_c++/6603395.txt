#include<bits/stdc++.h>
#define gc getchar//_unlocked

int t,n,k,i,mark[1000010],flip,ans;
    
int main()
{
    char ch;
    scanf("%d\n",&t);
    while(t--)
    {
        scanf("%d %d\n",&n,&k);
        memset(mark,0,sizeof(mark));
        flip=ans=0;
        for(i=0;i<n;i++)
        {
            ch=gc();
            flip+=mark[i];
            if(((flip&1)==1 && ch=='G')||((flip&1)==0 && ch=='R'))
            {
                ans++;
                mark[i+1]++;
                mark[i+k]--;
            }
        }
        printf("%d\n",ans);
    }
}
