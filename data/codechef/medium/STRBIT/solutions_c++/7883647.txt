const int N=1e5+5;
int tree[N];
int a[N];
char s[N];
inline void update(int idx,int val) {
    while(idx<N) {
        tree[idx]+=val;
        idx+=idx&-idx;
    }
}
inline int query(int idx) {
    int sm=0;
    while(idx>0) {
        sm+=tree[idx];
        idx-=idx&-idx;
    }
    return sm;
}
int main() {
    int t;sci(t);
    while(t--) {
        M(tree,0);
        M(s,0);
        int n;sci(n);
        int k;sci(k);
        scanf("%s",s+1);
        for(int i=1;i<=n;i++) {
            if(s[i]=='R')
                a[i]=1;
            else
                a[i]=0;
        }
        int ans=0;
        for(int i=1;i<=n;i++) {
            int temp=query(i);
            if((a[i]+temp)%2) {
                ans++;
                update(i+1,1);
                update(min(i+1+k-1,n+1),-1);
            }
        }
        cout<<ans<<endl;
    }
}
