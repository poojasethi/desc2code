#include<bits/stdc++.h>
using namespace std;

#define pb push_back
#define mp make_pair
#define N 100001
int main(){
    int t;
    cin>>t;
    while(t--){
        int n,k;
        cin>>n>>k;
        char arr[100001];
        int mark[N<<1]={0};
        scanf("%s",arr);
        int cur=0;
        int ans=0;
        for(int i=0;i<n;i++){
            cur+=mark[i];
            if((arr[i]=='R'&&(cur%2)==0)||(arr[i]=='G'&&(cur%2)==1)){
                ans++;
                mark[i+1]++;
                mark[i+k]--;
            }
        }
        printf("%d\n",ans);
    }
}
