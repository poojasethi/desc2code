#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;
vector<int> vec;
char str[100005];
int main()
{
	int t,N,K,i,pos;
    scanf("%d",&t);
    while(t--)
    {
	vec.clear();
	scanf("%d %d",&N,&K);
	
	scanf("%s",str);
	
	for(i=0;str[i]!='\0';i++)
	{
	      pos=upper_bound(vec.begin(),vec.end(),i-K)-vec.begin();
	      
	      int ele=vec.size()-pos;
	      
	      if(str[i]=='R')
	      {
	      	if(ele%2==0)
	      	vec.push_back(i);
	      	
		  }
		  else
		  {
		  	if(ele%2!=0)
		  	vec.push_back(i);
		  }
	      
	      
		
	}
	printf("%d\n",vec.size());
	
	
    }
	return 0;
}
