#include <bits/stdc++.h>

using namespace std;

int main()
{
	int tc,N,K,ans;
	char s[100010];
	queue<int> f; 
	scanf("%d",&tc);
	while(tc--)
	{
		while(f.size()>0)	f.pop();
		ans=0;
		scanf("%d%d%s",&N,&K,s);
		for (int i=0;i<N;i++)
		{
			while(f.size()>0 && f.front()<=i-K) f.pop();
			if (f.size()%2==0 && s[i]=='R')
			{
				ans++;
				f.push(i);
			}
			else if (f.size()%2==1 && s[i]=='G')
			{
				ans++;
				f.push(i);
			}
		}
		printf("%d\n",ans);
	}
}
