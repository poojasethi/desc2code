//using prefix sum
//time complexity: O(n)
#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int prefsum[100005];
int main()
{
    int t,n,k,sum,x,i,y;
    char s[100005];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        scanf("%s",s);
        
        memset(prefsum,0,sizeof(prefsum));
        sum=0;
        for(i=0;i<n;i++)
        {    
            
             if(i!=0)
             {
                 prefsum[i]+=prefsum[i-1];
             }
             // printf("prefsum[%d]=%d\n",i,prefsum[i]);  
              if(s[i]=='G')
              {
                  x=prefsum[i];
                  if(x%2==0)
                    continue;
                  else
                  {
                      sum+=1;
                      prefsum[i]+=1;
                       y=min(n,i+k-1);
                    //  printf("y=%d\n",y);
                      prefsum[y+1]+=-1;
                  }
              }
            else
              {
                   x=prefsum[i];
                  if(x%2!=0)
                    continue;
                  else
                  {
                      sum+=1;
                      prefsum[i]+=1;
                      y=min(n,i+k-1);
                   //   printf("y=%d\n",y);
                      prefsum[y+1]+=-1;
                  }
                  
              }
        }
        printf("%d\n",sum);
        
    }
    return 0;
}
