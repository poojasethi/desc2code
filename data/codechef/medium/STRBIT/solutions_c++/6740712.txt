#include <iostream>
#include<cstdio>
#include<string>
#include<vector>
#include<algorithm>
#include<cstring>
#include<map>
#include<queue>
#include<set>
#include<math.h>
#include<stack>
#include<cstdlib>

using namespace std;

int a[100005];
int main()
{
    int test,n,k,i,sum,ans;
    string s;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d%d",&n,&k);
        cin>>s;
        memset(a,0,sizeof(a));
        sum=0;ans=0;
        for(i=0;i<n;i++)
        {
            sum=sum+a[i];
            if(s[i]=='R'&&sum%2==0)
            {
                a[i]++;
                a[min(i+k,n)]--;
                ans++;
                sum++;
            }
            else if(s[i]=='G'&&sum%2==1)
            {
                a[i]++;
                a[min(i+k,n)]--;
                ans++;
                sum++;
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
