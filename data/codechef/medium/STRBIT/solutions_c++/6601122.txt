#include <bits/stdc++.h>

using namespace std;

char str[100010];

int main() {
	// your code goes here
	
	int test;
	cin>>test;
	while(test--)
	 {
	     int n,k;
	     cin>>n>>k;
         scanf("%s",str+1);
         int a[100010]={0};
         int cum[100010] = {0};
         
         for(int i=1;i<=n;i++)
          {
              if(str[i] == 'R')
               {
                   a[i] += 1;
                   a[i+1] -= 1;
               }
          }
          
         
          
          int i=1,ans = 0;
          
         while(i<=n)
         {
          
          while(i<=n && (cum[i-1]+a[i]) % 2 == 0)
           {
               i++;
               cum[i] = cum[i-1] + a[i];
           }
           
           if(i<=n)
            {
                ans++;
                a[i] += 1;
                a[min(i+k,n+1)] -= 1;
                cum[i] = cum[i-1] + a[i];
                i++;
            }
         }
         
         cout<<ans<<endl;
         
	 }
	
	return 0;
}
