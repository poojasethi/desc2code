#include<bits/stdc++.h>

using namespace std;

char l[2]={'R','G'};
int main(){
int t;
cin>>t;
while(t--){
    int n,k,i;
    cin>>n>>k;
    string s;
    cin>>s;
    queue<int>q;
    int ans=0;
    int c=0;
    for(i=0;i<n;i++){
        if(s[i]==l[c]){
           ans++;
           c=1-c;
           q.push(i+k-1);
        }
        if(!q.empty() && q.front()==i){
            c=1-c;
            q.pop();
        }
    }
    cout<<ans<<endl;
}

return 0;
}
