#include<bits/stdc++.h>
using namespace std;
#define sz 100002
#define pb(a) push_back(a)
#define ll long long
#define ull unsigned long long
#define fread freopen("input.txt","r",stdin)
#define fwrite freopen("output.txt","w",stdout)
#define inf (1<<29)
#define mem(abc,z) memset(abc,z,sizeof(abc))
#define PI acos(-1)
//sliding window problem or cumulative sum
int main(){
   int t,n,k,ans,mark[sz],cnt;
   string c;
   cin>>t;
   while(t--){
      cin>>n>>k>>c;
      cnt=ans=0;
      mem(mark,0);
      for(int i=0;i<n;i++){
         cnt+=mark[i]; //cumulative sum
         if((c[i]=='R' && cnt%2==0)||(c[i]=='G' && cnt&1)){ //now red
            ans++;
            mark[i+1]++;
            mark[min(n,i+k)]--;
         }
      }
      cout<<ans<<endl;
   }
   return 0;
}
