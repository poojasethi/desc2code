#include<iostream>
#include<string>
using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		int n,k;
		cin>>n>>k;
		string s;
		cin>>s;
		int cnt=0;
		int a[n];
		for(int i=0;i<n;i++) a[i]=0;
		
		int x,y=0;
		for(int i=0;i<n;i++){
			if(s[i]=='R'){
				x=1;
			}else x=0;
			y+=a[i];
			x+=y;
			if(x%2==1){
				cnt++;
				a[i]++;
				if((i+k)<n) a[i+k]--;
				y++;
			}
		}
//		for(int i=0;i<n;i++) cout<<a[i]<<" ";
//		cout<<endl;
		cout<<cnt<<endl;
	}
}
