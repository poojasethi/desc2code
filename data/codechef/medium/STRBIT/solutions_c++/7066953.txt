#include<bits/stdc++.h>
using namespace std;
int main()
{
        int i,j,k,m,n,t;
        scanf("%d",&t);
        for(i=0;i<t;i++)
        {
                scanf("%d%d",&n,&k);
                char arr[n];
                scanf(" %s",arr);
                int mark[100001],ans=0,flips=0;
                for(j=0;j<n;j++)
                        mark[j]=0;

                for(j=0;j<n;j++)
                {
                        flips+=mark[j];
                        if(arr[j]=='R' && flips%2==0 || arr[j]=='G' && flips%2==1)
                        {
                                ans+=1;
                                mark[j+1]+=1;
                                mark[min(j+k,n)]-=1;
                        }
                }
                printf("%d\n",ans);
        }
}
