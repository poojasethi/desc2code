import sys

def isFamily(parents,children):
    previous_cases = {}
    for char in children:
        if char in previous_cases:
            continue
        count = parents.count(char)
        if count < children.count(char):
            return 'NO'
        previous_cases[char] = count
    return 'YES'
        
t = input()
for case in range(t):
    parents = ''.join(sys.stdin.readline().strip().split())
    family = input()
    children = ''
    for child in range(family):
        children += sys.stdin.readline().strip()
    print isFamily(parents,children)
