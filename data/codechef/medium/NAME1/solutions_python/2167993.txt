import sys

def count_letters(name):
    counted_char = {}
    for char in name:
        if char in counted_char:
            continue
        counted_char[char] = name.count(char)
    return counted_char

def isFamily(parents,children):
    countname_parents = count_letters(parents)
    countname_children = count_letters(children)
    for char in countname_children:
        if char not in countname_parents or countname_children[char] > countname_parents[char]:
            return 'NO'
    return 'YES'
        
t = input()
for case in range(t):
    parents = ''.join(sys.stdin.readline().strip().split())
    family = input()
    children = ''
    for child in range(family):
        children += sys.stdin.readline().strip()
    print isFamily(parents,children)
