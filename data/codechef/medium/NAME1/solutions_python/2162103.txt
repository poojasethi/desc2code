tc=int(raw_input())
while tc:
    tc-=1
    pname=raw_input()
    [f,m]=pname.split(' ')
    f=f+m
    pdic={}
    for letter in "abcdefghijklmnopqrstuvwxyz":
        pdic.update({letter:0})
    for letter in f:
        pdic[letter]+=1
    n=int(raw_input())
    child=[]
    while n:
        child.append(raw_input())
        n-=1
    total=''
    for name in child:
        total+=name
    cdic={}
    for letter in "abcdefghijklmnopqrstuvwxyz":
        cdic.update({letter:0})
    for letter in total:
        cdic[letter]+=1
    flag=0
    for key in cdic.keys():
        if pdic[key]-cdic[key]<0:
            flag=1
            break
    if flag==1:
        print "NO"
    else:
        print "YES"
