#include <iostream>
using namespace std;

int main() {
	char a[40009],b[40009],ch;
	int c[30]={0};
	int d[30]={0};
	int t,n,i,j,flag;
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		for(i=0;i<30;i++)
		{
			c[i]=0;
			d[i]=0;
		}
		scanf("%s %s",&a,&b);
		for(i=0;a[i]!='\0';i++)
		{
			ch=a[i];
			c[ch-97+1]++;
		}
		for(i=0;b[i]!='\0';i++)
		{
			ch=b[i];
			c[ch-97+1]++;
		}
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%s",&a);
			for(j=0;a[j]!='\0';j++)
			{
				ch=a[j];
				d[ch-97+1]++;
			}
		}
		for(i=1;i<=26;i++)
		{
			if(c[i]<d[i])
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
