#include <iostream>

using namespace std;

int main()
{
	int t,n;
	cin>>t;

	string temp,p1,p2="";
	long int len;

	int flag;
	while(t--)
	{
		flag=0;
		int a[125]={0};
		cin>>p1>>p2;
		cin>>n;

		len=p1.length();
		for(int i=0;i<len;i++)
			a[p1[i]]++;

		len=p2.length();
		for(int i=0;i<len;i++)
			a[p2[i]]++;

		string c="";
		for(int i=0;i<n;i++)
		{
			cin>>temp;
			c+=temp;
		}

		len=c.length();
		for(int i=0;i<len;i++)
		{
			a[c[i]]--;
			if(a[c[i]]<0)
			{
				flag=1;
				break;
			}
		}

		if(flag) cout<<"NO\n";
		else cout<<"YES\n";
	}
	return 0;
}
