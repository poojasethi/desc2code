#include <iostream>
#include <stdio.h>
#include <cmath>
#include <string.h>
#include <stack>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;


int main()
{
	int t;
	int i;
	
	cin >> t;
	string a,b,str;
	int n;

	while (t--)
	{
		cin >> a;
		cin >> b;
		int parent[26] = {0};
		int children[26] = {0};

		for (i=0;a[i]!='\0';i++)
			parent[a[i]-'a']++;

		for (i=0;b[i]!='\0';i++)
			parent[b[i]-'a']++;

		cin >> n;
		while (n--)
		{
			cin >> str;
			for (i=0;str[i]!='\0';i++)
				children[str[i]-'a']++;
		}


		for (i=0;i<26;i++)
		{
			if (children[i] > parent[i])
				break;
		}

		if (i == 26)
			cout << "YES\n";
		else
			cout << "NO\n";
		
	}
	return 0;

}
