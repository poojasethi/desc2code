#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t,i,n,z;
	char str[40001];
	cin>>t;
	
	while(t--)
	{
		int b[26];
		for(i=0;i<26;i++)
		b[i]=0;
		cin>>str;
		for(i=0;i<strlen(str);i++)
		b[str[i]-'a']++;
		cin>>str;
		for(i=0;i<strlen(str);i++)
		b[str[i]-'a']++;
		scanf("%d",&n);
	
		for(i=0;i<n;i++)
		{
			cin>>str;
			for(z=0;z<strlen(str);z++)
			b[str[z]-'a']--;
		}
		for(i=0;i<26;i++)
		{
			//cout<<"  "<<b[i]<<endl;
			
			if(b[i]<0)
			break;
		}
		if(i<26)
		cout<<"NO\n";
		else
		cout<<"YES\n";
	}
	return 0;
}
