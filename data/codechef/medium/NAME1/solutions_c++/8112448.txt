#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	cin.ignore();
	while(t--)
	{
		char A[40010],B[40010];
		int a[26]={0};
		int b[26]={0};
		int i,n,j;
		cin>>A;
		for(i=0;A[i];i++)
		a[A[i]-'a']++;
		cin>>A;
		for(i=0;A[i];i++)
		a[A[i]-'a']++;
		cin>>n;
		cin.ignore();
		for(i=0;i<n;i++)
		{
			cin>>B;
			for(j=0;B[j];j++)
			b[B[j]-'a']++;
		}		
		for(i=0;i<26;i++)
		{
			if(b[i]>a[i])
			{
				cout<<"NO"<<endl;
				break;
			}
		}
		if(i==26)
		cout<<"YES"<<endl;		
	}
	return 0;
}
