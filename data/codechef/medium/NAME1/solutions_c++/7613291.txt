#include<iostream>
using namespace std;

int main()
{
	int t,flag,n;
	string p,m;
	string c;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		int hash1[256]={0},hash2[256]={0};
		flag=0;
		
		cin>>p>>m;
		for(int j=0;p[j]!='\0';j++)
		hash1[p[j]]++;
		for(int j=0;m[j]!='\0';j++)
		hash1[m[j]]++;
		cin>>n;
		for(int j=0;j<n;j++)
		{
			cin>>c;
			for(int k=0;c[k]!='\0';k++)
			hash2[c[k]]++;
			
		}
		for(int j=0;j<256;j++)
		{
			if(hash2[j]>hash1[j])
			{
				flag=1;
				break;
			}
		}
		
		if(flag==0)
		cout<<"YES\n";
		else
		cout<<"NO\n";
		
	}
	
	
	return 0;
}
