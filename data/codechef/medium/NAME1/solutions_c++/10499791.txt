#include<bits/stdc++.h>
using namespace std;
int main()
{
	int test;cin>>test;int t;int i,j;
	int parents[26];
	int children[26];
	for(t=0;t<test;t++)
	{
		memset(parents,0,sizeof(parents));
		memset(children,0,sizeof(children));
		string a,b;
		cin>>a>>b;
		int n;cin>>n;
		int length=a.size()+b.size();
		for(i=0;i<a.size();i++)
		{
			parents[a[i]-'a']++;
		}
		for(i=0;i<b.size();i++)
		{
			parents[b[i]-'a']++;
		}
		string c;
		int lengthc=0;
		for(i=0;i<n;i++)
		{
			cin>>c;
			lengthc+=c.size();
			for(j=0;j<c.size();j++)
			{
				children[c[j]-'a']++;
			}
		}
		if (lengthc<=length)
		{
			int flag=1;
			for(i=0;i<26;i++)
			{
				if(parents[i]<children[i])
				{
					cout<<"NO\n";
					flag=0;
					break;
				}
			}
				if (flag==1)
				{
					cout<<"YES\n";
				}
			
		}
		else
		{
			cout<<"NO\n";
		}
		
		
	}//test
	
	
	
	return 0;
	
}//main
