#include <iostream>
#include <string.h>
#include <cstdio>
using namespace std;

void process()
{
    int t;
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        int h[26] = {0},flag = 1;
        int k[26] = {0};
        char a[40001],b[40001],c[40001];
        scanf("%s",a);
        scanf("%s",b);
        int n,i,j;
        cin >> n;
        for(i = 0; i<n; ++i)
        {
            scanf("%s",c);
            j = 0;
            while(c[j] != '\0')
            {
                ++k[c[j] - 'a'];
                ++j;
            }
        }
        i = 0;
        while(a[i] != '\0')
        {
            ++h[a[i] - 'a'];
            ++i;
        }
        i = 0;
        while(b[i] != '\0')
        {
            ++h[b[i] - 'a'];
            ++i;
        }
        //display(h);
        //display(k);
        for(i = 0; i<26; ++i)
        {
            if(k[i] > h[i])
            {
                printf("NO\n");
                flag = 0;
                break;
            }
        }
        if(flag)
        {
            printf("YES\n");
        }
    }
}

int main()
{
    process();
    return 0;
}
