#include<bits/stdc++.h>
using namespace std;
main()
{
	string a,b,p,child,c;
	int n,d,i;
	int t;
	cin>>t;
	while(t--)
	{
		int array[26]={};
		child="";
		cin>>a>>b;
		p=a+b;
		//cout<<p;
		cin>>n;
		while(n--)
		{
			cin>>c;
			child=child+c;
		}
		//cout<<child;
		for(i=0;i<p.size();i++)
		{
			array[p[i]-97]++;
		}
		
		// for(i=0;i<26;i++)
		// 	cout<<array[i]<<" ";
		// cout<<endl;
        d=0;

		for(i=0;i<child.size();i++)
		{
			array[child[i]-97]--;

			if(array[child[i]-97]<0)
			{   d=1;
				break;}
		}
		(d!=1)?cout<<"YES":cout<<"NO";
		cout<<endl;

	}
}
