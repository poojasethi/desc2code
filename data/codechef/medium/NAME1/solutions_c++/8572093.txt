#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    getchar();
    while(t--)
    {
    	string str;
    	int n,i=0,flag=1;
   		int arr[26]={0};
   		getline(cin,str);
   		while(str[i]!='\0')
   		{
   			++arr[str[i]-'a'];
   			i++;
   		}
   		cin>>n;
   		getchar();
   		while(n--)
    	{
    		string st;
    		getline(cin,st);
    		int j=0;
    		while(st[j]!='\0' && flag!=0)
    		{
    			--arr[st[j]-'a'];
    			if(arr[st[j]-'a']<0)
    				flag=0;
    			j++;
    		}
    	}
    	flag?cout<<"YES\n":cout<<"NO\n";
 	}
  	return 0;
}
