#include <bits/stdc++.h>
using namespace std;

char a[100000];
char b[100000];
char c[100000];
int main(){
	int t;
	cin >> t;
	while(t--){
		cin >> a;
		cin >> b;
		map < char,int > m;
		int i;
		for(i=0;a[i]!='\0';i++) m[a[i]]++;
		for(i=0;b[i]!='\0';i++) m[b[i]]++;
		int n,f = 0;
		cin >> n;
		while(n--){
			cin >> c;
			for(i=0;c[i]!='\0' && f!=1;i++){
				if(m[c[i]] == 0){
					f = 1;
					break;
				}else{
					m[c[i]]--;
				}
			}
		}
		if(f==1) cout << "NO\n";
		else cout << "YES\n";
	}
}
