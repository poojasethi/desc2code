#include <iostream>
#include<string.h>

int main() 
{  int t,n,i,flag,l1,l2,l3;
  char a[40002],b[40002],c[40008];
  scanf("%d",&t);
  while(t--)
  { flag=0;
    scanf("%s %s",a,b);
    l1=strlen(a);
	l2=strlen(b);
	int x[26]={0};
	
	for(i=0;i<l1;i++)
	{x[a[i]-'a']++;
	}
	for(i=0;i<l2;i++)
	{x[b[i]-'a']++;
	}
	scanf("%d",&n);
	while(n--)
	{
	 scanf("%s",c);
	 l3=strlen(c);
	 for(i=0;i<l3;i++)
	 x[c[i]-'a']--;
	 
	}
	for(i=0;i<26;i++)
	{if(x[i]<0)
	 {
	  flag=1;
	  break;
	 }
    }
	 if(flag==0)
	 printf("YES\n");
	 else
	 printf("NO\n");	

    
 } 
	return 0;
	
}
