#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		char a[40001],b[40001],c[40001];
		int ctr_c[26] = {0}, ctr_p[26] = {0};
		cin>>a>>b;
		for(i=0;a[i]!='\0';i++)
			ctr_p[a[i]-97]++;
		for(i=0;b[i]!='\0';i++)
			ctr_p[b[i]-97]++;
		cin>>n;
		while(n--)
		{
			cin>>c;
			for(i=0;c[i]!='\0';i++)
				ctr_c[c[i]-97]++;
		}
		int flag = 0;
		for(i=0;i<26;i++)
		{
			if(ctr_c[i]!=0 && ctr_c[i]>ctr_p[i])
			{
				flag = 1;
				break;
			}
			
		}
		if(flag==1)
			cout<<"NO";
		else
			cout<<"YES";
		cout<<endl;
	}
	return 0;
}
