#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    bool f;
    char a[40001];
    int t,n;
    long int l,ctr1[26]={0},ctr2[26]={0};
    scanf("%d",&t);
    while(t--)
    {
        f=1;
        scanf("%s",a);
        l=strlen(a);
        while(l--)
            ++ctr1[a[l]-'a'];
        scanf("%s",a);
        l=strlen(a);
        while(l--)
            ++ctr1[a[l]-'a'];
        scanf("%d",&n);
        while(n--)
        {
            scanf("%s",a);
            l=strlen(a);
            while(l--)
                ++ctr2[a[l]-'a'];
        }
        for(l=0;l<26;++l)
        {
            if(ctr2[l]>ctr1[l])
                f=0;
            ctr1[l]=0;
            ctr2[l]=0;
        }
        if(f)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
