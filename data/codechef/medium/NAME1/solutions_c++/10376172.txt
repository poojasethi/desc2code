#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int table1[30],table2[30];
char a[100000],b[100000],c[100000];
int main()
{
    int t,flag,i,j,l,n;
    scanf("%d",&t);
    while(t--)
    {
        for(i=0;i<=29;i++)
        {
            table1[i]=0;
            table2[i]=0;
        }
        scanf("%s %s",&a,&b);
        //printf("%s\n",a);
        //printf("%s\n",b);
        l=strlen(a);
        for(i=0;i<l;i++)
        {
            table1[a[i]-'a']++;
        }
        /*for(i=0;i<=25;i++)
        {
            printf("%d ",table1[i]);
        }
        printf("\n");*/
        l=strlen(b);
        for(i=0;i<l;i++)
        {
            table1[b[i]-'a']++;
        }
        /*for(i=0;i<=25;i++)
        {
            printf("%d ",table1[i]);
        }
        printf("\n");*/
        scanf("%d",&n);
        for(j=0;j<n;j++)
        {
            scanf("%s",&c);
            l=strlen(c);
            for(i=0;i<l;i++)
            {
                table2[c[i]-'a']++;
            }
        }
        flag=0;
        for(i=0;i<=25;i++)
        {
            if(table1[i]<table2[i])
            {
                flag=1;
                break;
            }
        }
        if(flag)
            printf("NO\n");
        else
            printf("YES\n");
    }
    return 0;
}
