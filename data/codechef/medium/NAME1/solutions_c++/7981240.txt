#include<iostream>
using namespace std;
#include<cstdio>
#include<cstring>
int main(){
	int t,n,i,l,temp;
	bool flag;
	scanf("%d",&t);
	char s1[40000],s2[40000],s3[40000];
	while(t--){
		scanf("%s %s",&s1,&s2);		
		scanf("%d%*c",&n);
		int a[26];
		for(i=0;i<26;i++)
			a[i]=0;
		temp=n;
		for(i=0;i<strlen(s1);i++)
				a[s1[i]-'a']++;
		for(i=0;i<strlen(s2);i++){
				a[s2[i]-'a']++;
			}
 
		while(temp--){
			scanf("%s",&s3);
			for(i=0;i<strlen(s3);i++)
				a[s3[i]-'a']--;
		}
		flag=0;
		for(i=0;i<26;i++)
		{	
			if(a[i]<0)
				flag=1;
		}
 
		if(flag==1)
			printf("NO\n");
		else
			printf("YES\n");
 
 
 
	}
}
