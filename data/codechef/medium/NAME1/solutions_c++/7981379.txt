#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    cin.get();
    while(t--)
    {
        int A[200]={0};
        char ch=cin.get();
        while(ch!='\n')
        {
            if(ch!=' ')
            {
                A[ch]++;
            }
            ch=cin.get();
        }
        int c;
        cin>>c;
        cin.get();
        while(c--)
        {
            ch=cin.get();
            while(ch!='\n')
            {
                A[ch]--;
                ch=cin.get();
            }
        }
        bool ans=1;
        for(int i=0;ans &&i<200;i++)
        {
            if(A[i]<0)
                ans=0;
        }
        if(ans)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
