#include<stdio.h>
#include<stdlib.h>

int a[30],b[30];
char str[80010];
int main()
{
	int i,j,t,n,flag;
	char c;
	scanf("%d",&t);
	c=getchar();
	while(t--)
	{
		flag=0;
		gets(str);
		for(i=0;i<27;i++)
		{
			a[i]=0;
			b[i]=0;
		}
		for(i=0;str[i]!='\0';i++)
			{
				if(str[i]>=97 && str[i]<=122)
				a[str[i]-97]++;
			}
		scanf("%d",&n);
			c=getchar();
		for(j=0;j<n;j++)
		{
			gets(str);
		for(i=0;str[i]!='\0';i++)
		{
			if(str[i]>=97 && str[i]<=122)
			b[str[i]-97]++;
			
		}
			
		}
		for(i=0;i<27;i++)
		{
			if(b[i]>a[i])
			{
				flag=1;
				break;
			}
				
		}
		if(flag==1)
		printf("NO\n");
		else
		printf("YES\n");
	}
	return 0;
}
