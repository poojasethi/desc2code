#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;

	while(t--) 
	{
		char A[40001],B[40001];
		cin >> A >> B;

		int c1[256]={0},i;	

		for(i = 0;i < strlen(A);i++) {
			++c1[A[i]];
		}

		for(i = 0;i < strlen(B);i++) {
			++c1[B[i]];
		}		

		int ch;
		cin >> ch;

		char ch_name[40001];
		char full_name[40001];
		full_name[0] = '\0';

	
		for(i = 0;i < ch;i++) {
			cin >> ch_name;
			strcat(full_name,ch_name);
		}

		int c2[256] = {0};
		for(i = 0;i < strlen(full_name);i++) {
			++c2[full_name[i]];
		}

		int flag = 0;
		for(i = 0;i < 256;i++) {
			if(c2[i] > c1[i]) {
				flag = 0;
				break;
			}
			if(c2[i] <= c1[i])
				flag = 1;
		}

		if(flag == 1)
			cout << "YES" << "\n";
		else
			cout << "NO" << "\n";

		/*for(i = 0;i < 256;i++) 
		{
			cout << "i = " << "c1 = " << c1[i] << " " << "c2 = " << c2[i] << "\n";
		}*/

	}
	return 0;
}
