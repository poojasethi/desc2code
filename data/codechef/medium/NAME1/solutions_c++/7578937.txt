#include <bits/stdc++.h>
using namespace std;
int main() {
    int T;
    cin>>T;
    while(T--)
    {
        int N,par[26]={0},ch[26]={0};
        string A,B,C;
        cin>>A>>B;
        cin>>N;
        for(int i=0;i<N;i++)
        {
            cin>>C;
            for(int j=0;j<C.length();j++)
            {
                ch[(int)C[j]-97]++;
            }
        }
        for(int j=0;j<A.length();j++)par[(int)A[j]-97]++;
        for(int j=0;j<B.length();j++)par[(int)B[j]-97]++;
        bool flag =true;
        for(int i=0;i<26;i++)
        {
            if(par[i]<ch[i])
            {
                flag=false;
                break;
            }
        }
        if(flag)cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
	return 0;
}
