#include<stdio.h>
#include<cstring>
using namespace std;
int main()
{
	int loop;
	char e[2];
	scanf("%d",&loop);
	gets(e);
	for(int i=loop;i>0;i--)
	{
		char parent[40001];
		int noc,i=0,flag=1;
		int arr[123]={0};
		gets(parent);
		while(parent[i]!='\0')
		{
			++arr[parent[i]];
			i++;
		}
		//for(int k=97;k<123;k++)
			//printf("%d ",arr[k]);
		scanf("%d",&noc);
		gets(e);
		while(noc--)
		{
			char child[40001];
			gets(child);
			int j=0;
			while(child[j]!='\0' && flag!=0)
			{
				--arr[child[j]];
				//printf("%c ",child[j]);
				if(arr[child[j]]<0)
					flag=0;
				j++;
			}
		}
		printf("%s\n",flag?"YES":"NO");
	}
	return 0;
}
