#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
int a[26],b[26],j;
char p1[40000],p2[40000],c1[40000],c2[40000];
int main()
{
	int n,t;
	cin >> t;
	while(t--)
	{
		for (int i = 0; i < 26; ++i)
		{
			a[i] = 0;
			b[i] = 0;
		}
		cin >> p1;
		cin >> p2;
		strcat(p1,p2);
		cin >> n;
		c1[0] = '\0';
		while(n--)
		{
			cin >> c2;
			strcat(c1,c2);
		}
// 		cout << p1 << endl;
// 		cout << c1 << endl;
		
		int h = strlen(p1);
		for (int i = 0; i < h; ++i)
		{
			a[p1[i] - 97]++;
		}
		h = strlen(c1);
		for (int i = 0; i < h; ++i)
		{
			b[c1[i] - 97]++;
		}
		for ( j = 0; j < 26; ++j)
		{
			if(a[j] < b[j])
			{
				cout << "NO\n";
				break;
			}
		}
		if(j == 26)
			cout << "YES\n";
	}
	return 0;
}
