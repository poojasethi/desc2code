#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long int M=1000000007;
int main()
{
   long long int t,n;
   cin>>t;
   while(t--)
   {
       //char str1[100001],str2[100001],str[100001],str3[100001];
       string str1,str2,str3,str;
       cin>>str1>>str2;
       cin>>n;
       str="";
       while(n--)
       {
           cin>>str3;
           str+=str3;
       }
       long long int p[26]={0},q[26]={0},i;
       str1+=str2;
       for(i=0;str1[i]!='\0';++i)
       {
           ++p[str1[i]-'a'];
       }
       for(i=0;str[i]!='\0';++i)
       {
           ++q[str[i]-'a'];
       }
       int flag=1;
       for(i=0;i<26;++i)
       {
           if(p[i]<q[i])
           {
               flag=0;
               break;
           }
       }
       if(flag==0)
       {
          cout<<"NO\n";
       }
       else{
       cout<<"YES\n";
       }
   }
    return 0;
}
