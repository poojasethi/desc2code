#include<cstdio>
#include<cstring>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k;
        char A[40005],B[40005];
        int count[26];
        for(int i=0;i<26;i++){
            count[i]=0;
        }
        scanf("%s %s",&A,&B);
        for(int i=0;i<strlen(A);i++){
            count[A[i]-'a']++;
        }
        for(int i=0;i<strlen(B);i++){
            count[B[i]-'a']++;
        }
        scanf("%d",&n);
        for(int i=0;i<n;i++){
            char child[40005];
            scanf("%s",&child);
            for(int j=0;j<strlen(child);j++){
                count[child[j]-'a']--;
            }
        }
        int fail=0;
        for(int i=0;i<26;i++){
            if(count[i]<0){
                fail=1;
            }
        }
        if(fail){
            printf("NO\n");
        }
        else
        {
            printf("YES\n");
        }
        
    }
}
