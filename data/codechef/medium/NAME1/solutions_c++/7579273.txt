#include<stdio.h>
#include<string.h>

char a[40001],b[40001],str[40001];

int main()
{
    int tc,n,len1,len2,len,flag;
    long int hashe[26],i;

    scanf("%d\n",&tc);

    while(tc--)
    {
        scanf("%s%s",a,b);
        scanf("%d\n",&n);

        len1 = strlen(a);
        len2 = strlen(b);
        for(i=0;i<26;++i)
            hashe[i] = 0;
        flag = 1;
        for(i=0;i<len1;++i)
            hashe[(int)a[i]-97]++;
        for(i=0;i<len2;++i)
            hashe[(int)b[i]-97]++;

        while(n--)
        {
            gets(str);
            if(flag)
            {
                len = strlen(str);
                for(i=0;i<len;i++)
                    if(hashe[(int)str[i]-97])
                        hashe[(int)str[i]-97]--;
                    else
                        flag = 0;
            }
        }
        if(flag)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
