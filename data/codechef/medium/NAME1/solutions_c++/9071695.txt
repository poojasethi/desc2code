#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,count,count1,c,n;
	int arr[130];
	scanf("%d",&t);
	while(t--)
	{
		for(int i=0;i<130;i++)
		arr[i]=0;
		count=0;
		count1=0;
		c=0;
		char s[40005],s1[40005];
		scanf("%s",s);
		scanf("%s",s1);
		for(int i=0;s[i]!='\0';i++)
		count++;
		for(int i=0;s1[i]!='\0';i++)
		count1++;
		for(int i=0;i<count;i++)
		arr[s[i]]++;
		for(int i=0;i<count1;i++)
		arr[s1[i]]++;
	//	cout<<s<<" "<<s1<<" "<<count<<" "<<count1<<" ";
		scanf("%d",&n);
	//	cout<<"n="<<n<<endl;;
		while(n--)
		{
		count1=0;
		char sn[40005];
		scanf("%s",sn);
		if(c!=1)
		{
		for(int i=0;sn[i]!='\0';i++)
		count1++;
		for(int i=0;i<count1;i++)
		if(arr[sn[i]]!=0)
		{
		arr[sn[i]]--;
		}
		else
		{
		c=1;
		break;
		}
		}
		}//while
		if(c==1)
		cout<<"NO\n";
		else
		cout<<"YES\n";
	}
	return 0;
}
