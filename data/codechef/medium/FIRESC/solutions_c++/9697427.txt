#include<iostream>
#include<string>
#include<cstdio>
#include<algorithm>
#include<cmath>
#include<string>
#include<string.h>
#include<vector>
using namespace std;
vector <int> v[100005];
int kp,dir[100005],x,n,m,a,b,pom,pom2;
long long xx[100005],rez;
int dfs(int x){
 dir[x]=kp;
 for(int i=0;i<v[x].size();i++){
  if(!dir[v[x][i]])
   dfs(v[x][i]);
 }
}
int t;
int main(){


 scanf("%d",&t);
 while(t--){
  kp=0;
  memset(dir,0,sizeof(dir));
  memset(xx,0,sizeof(xx));
  for(int i=1;i<=n;i++)
   v[i].clear();
  scanf("%d %d",&n,&m);
  for(int i=1;i<=m;i++){
   scanf("%d %d",&a,&b);
   v[a].push_back(b);
   v[b].push_back(a);
  }

  for(int i=1;i<=n;i++){
   if(!dir[i]){
     kp++;
     dfs(i);
    }
  }
  rez=1;
  for(int i=1;i<=n;i++) xx[dir[i]]++;
  for(int i=1;i<=kp;i++) rez=rez*xx[i]%1000000007;
  printf("%d %lld\n",kp,rez);
 }

return 0;
}
