#include <bits/stdc++.h>
#define MAX 100005
#define MOD 1000000007
using namespace std;
long long int size=0,ans=0;
bool visit[200010];
vector<int> v[200010];
void dfs(int n)
{
    int i;
    visit[n]=true;
    size++;
    for(i=0;i<v[n].size();i++)
        if(!visit[v[n][i]])
            dfs(v[n][i]);
    return;
}

int main(int argc, char const *argv[]) {
    int test,n,m,i,j,k;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d%d",&n,&m);
        for(i=0;i<=n;i++)
            v[i].clear();
        for(i=0;i<m;i++)
        {
            scanf("%d%d",&j,&k);
            v[j].push_back(k);
            v[k].push_back(j);
        }
        memset(visit,0,sizeof(visit));
        long long int mul=1;
        ans=0;
        for(i=1;i<=n;i++)
        {
            if(!visit[i])
            {
                size=0;
                ans++;
                dfs(i);
                mul*=size;
                mul%=MOD;
            }
        }
        printf("%lld %lld\n",ans,mul);
    }
    return 0;
}
