#include<bits/stdc++.h>
#define ll long long int
#define mp make_pair
#define pb push_back
#define si(x) scanf("%d",&x)
#define pi(x) printf("%d\n",x)
#define s(x) scanf("%lld",&x)
#define p(x) printf("%lld\n",x)
#define M 1000000007

using namespace std;

vector<bool> visited;
vector<vector<int> > graph;

int val=0;
void dfs(ll u){
val++;
visited[u]=true;
for(vector<int>::iterator it = graph[u].begin(); it != graph[u].end(); it++){
    if(!visited[*it]) dfs(*it);
}

}

int main(){
int t;
si(t);
while(t--){
        int n,m;
        si(n);si(m);
     graph = vector<vector<int> > (n);
        for(int i = 0; i < m; i++){
            int u,v;
            si(u);si(v);
            u--;
            v--;
            graph[u].push_back(v);
            graph[v].push_back(u);
        }
        int res = 0;
        int ways = 1;
        visited = vector<bool> (n, 0);
        for(int u = 0; u < n; u++){
            if(visited[u]==true)
                continue;

            res++;
            val=0;
            dfs(u);
            ways=(ll)val*ways%M;
        }
        printf("%d %d\n",res,ways);
}


return 0;
}
