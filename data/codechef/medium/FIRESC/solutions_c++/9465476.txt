#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define ll long long
void dfs(bool*visited,ll int u,ll int &master,list<ll int >*a){
    list<ll int>::iterator itr=a[u].begin();
    master++;
    while(itr!=a[u].end()){
        if(visited[*itr]==0){
            visited[*itr]=1;
            dfs(visited,*itr,master,a);
        }
        itr++;
    }
}

int main(){
    ll int t;
    bool visited[100001];
    cin>>t;
    while(t--){
        ll int n,m;
        scanf("%lld%lld",&n,&m);
        list<ll int>a[n+1];
        for(int i=1;i<=n;i++){
            visited[i]=0;
        }
        ll int x,y;
        for(int i=0;i<m;i++){
            scanf("%lld%lld",&x,&y);
            a[x].push_back(y);
            a[y].push_back(x);
        }
        long long int ans=1,z=0,master;
        for(int i=1;i<=n;i++){
            if(visited[i]==0){
                z++;
                visited[i]=true;
                master=0;
                dfs(visited,i,master,a);
                ans=(ans*master)%mod;
            }
        }
        printf("%lld %lld\n",z,ans);
    }
    return 0;
}
