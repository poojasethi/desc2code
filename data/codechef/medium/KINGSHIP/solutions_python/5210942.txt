t=input()
while t>0:
    t-=1
    n=input()
    a=raw_input()
    a=a.strip().split()
    v=[]
    m=int(a[0])
    pos=0
    v.append(m)
    for i in range(1,n):
        x=int(a[i])
        v.append(x)
        if(m>x):
            m=x
            pos=i
    s=0
    for i in range(n):
        if(i==pos):
            continue
        else:
            s=s+ m*v[i]
    print s
