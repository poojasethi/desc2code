for _ in xrange(int(raw_input())):
    raw_input()
    populations = map(int, raw_input().split())
    minIndex = 0
    minPopulation = populations[0]
    for i, p in enumerate(populations):
        if p < minPopulation:
            minPopulation = p
            minIndex = i
    costs = 0
    for i, p in enumerate(populations):
        if i == minIndex:
            continue
        costs += p * minPopulation

    print costs
