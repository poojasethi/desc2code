#include<iostream>
#include<cstdio>
using namespace std;
int main(){
    long long t,n,x,s,i,l;

    cin>>t;

    while(t--){

        cin>>n;
        s=0;
        cin>>x;

        for(i=1;i<n;++i){
            cin>>l;
            if(l<x){
                s+=x;
                x=l;
            }else s+=l;
        }
        cout<<s*x<<"\n";
    }

    return 0;
}
