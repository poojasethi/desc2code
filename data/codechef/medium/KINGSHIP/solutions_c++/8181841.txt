#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
	long long int sum=0,min=0;;
	int t=0,n=0,i=0;
	cin>>t;
	while(t--)
	{
		sum = 0;
		min = 1000000;
		cin>>n;
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(min > a[i])
            min = a[i];
		}
		for(i=0;i<n;i++)
		sum = sum+min * a[i];
		sum=sum - min *min;
		cout<<sum;
		cout<<endl;
		sum=0;
		n=0;
		min=0;
	}
}
