#include <iostream>
#include<limits.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--){
    long long int minm=INT_MAX,sum=0,index=-1;
    int n;
    cin>>n;
    long long int *p = new long long int[n];
    for(int i=0;i<n;i++)
    {
        cin>>p[i];
        if(minm>p[i])
        {
            minm=p[i];
            index=i;
        }
    }
    for(int i=0;i<n;i++)
    {
        if(i!=index)
        {
            sum=sum+(minm*p[i]);
        }
    }
    cout<<sum<<"\n";

    }
    return 0;
}
