#include<iostream>
#include<algorithm>
#include<cstdio>
#define ll long long int
using namespace std;
int main()
{
	ll t,i,k;
	scanf("%lld", &t);
	while(t--)
	{
		scanf("%lld", &k);
		ll a[k];
		for(i=0;i<k;i++)
		  scanf("%lld", &a[i]);
		sort(a, a+k);
		ll sum = 0;
		for(i=1;i<k;i++)
		{
			ll pdt = a[0]*a[i];
			sum += pdt;
		}
		printf("%lld\n", sum);
	}
}
