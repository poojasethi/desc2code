#include<bits/stdc++.h>
using namespace std;
#define ll long long int
void pre()
{
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  cout.tie(NULL);
}
inline int get_int()
{
  int n=0,neg=0,c=getchar_unlocked();
  while((c<'0'||c>'9')&&c!='-')c=getchar_unlocked();
  if(c=='-') {neg=1;c=getchar_unlocked();}
  while(c>='0'&&c<='9')n=(n<<3)+(n<<1)+c-'0',c=getchar_unlocked();
  if(neg) n=-n;
  return n;
}

int main()
{
  pre();
  ll t;
  cin>>t;
  while(t--)
  {
    ll N,i;
    cin>>N;
    ll P[N];
    for(i=0;i<N;i++)
      cin>>P[i];
    ll ans=0;
    sort(P,P+N);
    for(i=1;i<N;i++)
    {
      ans+=(P[0]*P[i]);
    }
    cout<<ans<<endl;
  }
  return 0;
}
