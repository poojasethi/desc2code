#include<bits/stdc++.h>
using namespace std;
#define ll long long

int t , n ;
ll p[100001];
ll total;
int main()
{
	ios_base::sync_with_stdio(0);
	cin>>t;
	while(t--)
	{
	cin>>n;
	ll minpath=1000001,minpathid;
	for(int i=0;i<n;i++)
	{
		cin>>p[i];
		if(minpath>p[i])
		{
			minpath=p[i];
			minpathid=i;
		}
	}
	total=0;
	for(int i=0;i<n;i++)
	{
		if(i!=minpathid)
		total+=(minpath*p[i]);
	}
		cout<<total<<"\n";
	}
}
