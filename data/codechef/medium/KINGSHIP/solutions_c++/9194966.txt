#include<bits/stdc++.h>
using namespace std;
long long a[1234567];
int main()
{
	//freopen("in","r",stdin);
	long t;
	cin>>t;
	while(t--)
	{
		long n;
		cin>>n;
		for(long i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		long long cost=0;
		for(long i=1;i<n;i++)
		{
			cost+=(a[i]*a[0]);
		}
		cout<<cost<<endl;
	}
}
