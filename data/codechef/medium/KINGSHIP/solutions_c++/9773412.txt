#include <iostream>
#include <cstdio>
#include <algorithm>
#include <math.h>
#include <string.h>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n;
        scanf("%d", &n);
        int a, min = 1000000;
        long long int count = 0, sum = 0;
        for(int i = 0; i < n; i++)
        {
            scanf("%d", &a);
            if(a < min)
            {
                min = a;
            }
            sum += a;
        }
        sum -= min;
        count = sum * min;

//        for(int i = 0; i < n; i++)
//        {
//            count += arr[i] * min;
//        }
//        count -= min * min;
        printf("%lld\n", count);
    }
    return 0;
}
