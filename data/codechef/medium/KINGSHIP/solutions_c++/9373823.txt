#include<cstdio>
#include<vector>
using namespace std;
int main()
{
	long long int cost = 0;
	int t, n;
	vector<int> input;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d", &n);
		while (n--)
		{
			int temp;
			scanf("%d", &temp);
			input.push_back(temp);
		}
		int min = input[0];
		for (int i = 0; i < input.size(); ++i)
		{
			if (min > input[i])
				min = input[i];
			cost += input[i];
		}
		cost -= min;
		cost *= min;
		printf("%lld\n", cost);
		cost = 0;
		input.clear();
	}
}
