#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int n;
        scanf("%lld",&n);
        long long int arr[n];
        for(int i=0;i<n;i++)
            scanf("%lld",&arr[i]);
        sort(arr,arr+n);
        long long int temp=arr[0];
        long long int sum=0;
        for(int i=1;i<n;i++)
        {
            sum+=temp*arr[i];
        }
        if(sum!=0)
            printf("%lld\n",sum);
        else
            printf("%lld\n",arr[0]);
    }
    return 0;
}
