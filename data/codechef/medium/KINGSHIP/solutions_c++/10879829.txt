#include<cstdio>
#include<algorithm>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        ll Min=INF,sum=0;
        for(int i=0;i<n;i++){
            ll tmp;
            scanf("%lld",&tmp);
            Min=min(tmp,Min);
            sum+=tmp;
        }
        ll ans=Min*(sum-Min);
        printf("%lld\n",ans);
    }
    
}
