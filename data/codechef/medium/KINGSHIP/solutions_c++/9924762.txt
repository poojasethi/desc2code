#include <iostream>

using namespace std;

int main()
{
    //cout << "Hello world!" << endl;
    long long int t,n;
    cin>>t;
    while(t--)
    {

        cin>>n;
        long long int p[n],i,small=100000000,sum=0;
        for(i=0;i<n;++i)
        {cin>>p[i];
        if(p[i]<small)
        small=p[i];
        sum+=p[i];
        }
        sum-=small;
        sum*=small;
        cout<<sum<<"\n";


    }
    return 0;
}
