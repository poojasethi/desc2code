#include<bits/stdc++.h>
using namespace std;
int main(){
	long long t,n,pop[100003],dis[100002];
	scanf("%lld",&t);
	while(t--){
		scanf("%lld",&n);
		for(long long i=0;i<n;i++){
			scanf("%lld",&pop[i]);
		}
		long long min=INT_MAX;
		for(long long  i=0;i<n;i++){
			if(pop[i]<min)
			min=pop[i];
		}
		long long k=0;
		for(long long i=0;i<n;i++){
			if(pop[i]!=min)
			dis[k++]=pop[i]*min;
		}	
		long long  ans=0;	
		for(long long i=0;i<(n-1);i++){
			ans+=dis[i];
		}
		printf("%lld\n",ans);
		
	}
	return 0;
}
