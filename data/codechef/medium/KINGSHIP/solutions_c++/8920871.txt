#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;
    assert(t >= 1 && t <= 10);
    while(t--) {
        int n;
        cin >> n;
        assert(n >= 1 && n <= 100000);
        int a[n],i;
        for(i = 0;i < n;i++) {
            cin >> a[i];
            assert(a[i] >= 1 && a[i] <= 1000000);
        }

        sort(a,a+n);
        long long int ans = 0;
        for(i = 1;i < n;i++) {
            ans += a[i];
        }
        cout << (ans*a[0]) << "\n";
    }
    return 0;
}
