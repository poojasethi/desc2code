#include<bits/stdc++.h>
#define F(i,x,y) for(i=x;i<y;i++)
#define S(n) scanf("%lld",&n)
#define s(n) scanf("%d",&n)
#define P(n) printf("%lld\n",n)
#define p(n) printf("%d\n",n)
#define LLI long long int
#define R return 0
using namespace std;
int main()
{
	int t; cin>>t;
	while(t--)
	{
		int n,i; cin>>n;
		LLI a[n];
		F(i,0,n) cin>>a[i];
		sort(a,a+n);
		LLI x=0;
		F(i,1,n) x=x+(a[i]*a[0]);
		cout<<x<<endl;
	}
	R;
}
