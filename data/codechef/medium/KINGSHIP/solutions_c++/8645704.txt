#include <bits/stdc++.h>

long long a[100000];
using namespace std;

int main(){
    long tt;
    //ifstream cin("a.txt");
    long n;
    cin>>tt;
    while(tt--){
        cin>>n;
        for(long i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);

        long long sum=0;
        for(long i=1;i<n;i++)
            sum+=(a[i]*a[0]);
        cout<<sum<<endl;
    }
}
