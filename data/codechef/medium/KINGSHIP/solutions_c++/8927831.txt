#include <iostream>
using namespace std;
int main()
{
    long long int t,i,n,j,s,mnii;
    cin>>t;
    for(i=1;i<=t;i++)
    {
        mnii=0;
        cin>>n;
        long long int a[n];
        for(j=0;j<n;j++)
        {
            cin>>a[j];
        }
        mnii=a[0];
        for(j=0;j<n;j++)
        {
            if(a[j]<mnii)
            {
                mnii=a[j];
            }
        }
        s=0;
        for(j=0;j<n;j++)
        {
            s+=a[j]*mnii;
        }
        s=s-(mnii*mnii);
        cout<<s<<"\n";

    }
    return 0;
}
