#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long
int main(){
  int t,n;
  cin>>t;
  while(t--){
   cin>>n;
   ll sum=0;
   ll p[n],min_p=9999999;
   for(int i=0;i<n;i++){
    cin>>p[i];
    min_p=min(min_p,p[i]); 
   }
   for(int i=0;i<n;i++){
    sum+=(min_p*p[i]);
   }
   sum-=(min_p*min_p);
   cout<<sum<<endl;
  }
  return 0;
}
