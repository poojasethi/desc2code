#include <iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;

int main()
{
  //  cout << "Hello world!" << endl;
    int tst;
    long long int ans=0,n,a[100001],i;
    scanf("%d",&tst);
    while(tst--)
    {
        ans=0;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
           scanf("%lld",&a[i]);
        sort(a,a+n);
        for(i=1;i<n;i++)
        {
            ans+=(a[i]*a[0]);
        }
       printf("%lld\n",ans);


    }



    return 0;
}
