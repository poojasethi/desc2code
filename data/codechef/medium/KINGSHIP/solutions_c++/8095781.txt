#include <stdio.h>
#include <stdlib.h>
#define ll long long
int cmpfunc (const void * a, const void * b)
{
   return ( *(int*)a - *(int*)b );
}
ll int pop[100005];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        ll int p, n, i, sum = 0;
        scanf("%lld",&n);
        for(i = 0; i < n; i++)
            scanf("%lld",&pop[i]);
        qsort(pop, n, sizeof(ll int), cmpfunc);
        //for(i = 0; i < n; i++)
            //printf("%d ",pop[i]);
        for(i = 1; i < n; i++)
        {
            sum = sum + (pop[0] * pop[i]);
        }
        printf("%lld\n",sum);
    }
}
