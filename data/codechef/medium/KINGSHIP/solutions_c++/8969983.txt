#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	for(scanf("%d", &t); t--; )
	{
		int n;
		long long p[100001];
		scanf("%d", &n);
		for(int i=0; i<n; i++)
			scanf("%lld", &p[i]);
		int idx=0;
		for(int i=1; i<n; i++)
			if(p[i]<p[idx])
				idx=i;
		long long res =0;
		for(int i=0; i<n; i++)
			res += (idx!=i)?p[i]*p[idx]:0;
		printf("%lld\n", res);
	}
	return 0;
}
