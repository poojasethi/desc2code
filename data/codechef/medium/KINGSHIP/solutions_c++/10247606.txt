#include<iostream>
using  namespace std;
int main()
{
 uint64_t t,a[100000],i,k,n,min;
 cin>>t;
 while(t--)
 {
  cin>>n;
  for(i=0;i<n;i++)
  {
   cin>>a[i];
   if(i==0)
   min=a[i];
   if(a[i]<min)
   min=a[i];
   }
   k=0;
   for(i=0;i<n;i++)
   {
    k+=a[i]*min;
    }
    cout<<(k-(min*min))<<"\n";
    }
    return 0;
    }
