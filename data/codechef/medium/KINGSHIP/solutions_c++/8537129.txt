#include <iostream>
#include <stdio.h>
#include <cmath>
#include <string.h>
#include <stack>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	int m,n;
	int i,j;
	cin >> t;
	
	while (t--)
	{
		cin >> n;
		long long arr[n];
		long long min = 1000000;
		int pos = 0;
		for (i=0;i<n;i++)
		{
			cin >> arr[i];
			if (arr[i] < min)
			{
				min = arr[i];
				pos = i;
			}
		}	

		long long ans = 0;
		for (i=0;i<n;i++)
		{
			if (pos!=i)
				ans += arr[i]*arr[pos];
		}
		cout << ans << endl;
	}
	return 0;

}
