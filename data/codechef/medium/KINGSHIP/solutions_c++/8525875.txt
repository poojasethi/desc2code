#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int T, n;
	cin>>T;
	while(T--)
	{
		cin>>n;
		long long int a[n], ans = 0;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}

		sort(a, a+n);

		for(int i=1;i<n;i++)
		{
			ans += a[i]*a[0];
		}

		cout<<ans<<"\n";

	}
	return 0;
}
