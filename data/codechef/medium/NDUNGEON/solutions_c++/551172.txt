#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
int m,n;
scanf("%d",&m);
scanf("%d",&n);
int a[m][n];
int flag[m][n];
int dist[m][n];
for(int i=0;i<m;i++)
{
for(int j=0;j<n;j++)
{
scanf("%d",&a[i][j]);
dist[i][j]=1000000000;
flag[i][j]=0;
}
}
int x,y,t;
scanf("%d",&x);
scanf("%d",&y);
scanf("%d",&t);
x--;
y--;
dist[0][0]=a[0][0];
int q=m*n;
int minx=0,miny=0;
while(q!=0)
{
if(dist[minx][miny]==1000000000)
break;
q--;
flag[minx][miny]=1;
for(int h=-1;h<=1;h=h+2)
{
int l=0;
if((minx+h)<m && (miny+l)<n && (minx+h)>=0 && (miny+l)>=0)
{
int alt=a[minx+h][miny+l]+dist[minx][miny];
if(alt<dist[minx+h][miny+l])
{
flag[minx+h][miny+l]=2;
dist[minx+h][miny+l]=alt;
}
}
}
for(int l=-1;l<=1;l=l+2)
{
int h=0;
if((minx+h)<m && (miny+l)<n && (minx+h)>=0 && (miny+l)>=0)
{
int alt=a[minx+h][miny+l]+dist[minx][miny];
if(alt<dist[minx+h][miny+l])
{
flag[minx+h][miny+l]=2;
dist[minx+h][miny+l]=alt;
}
}
}
int min=1000000000;
for(int c=0;c<m;c++)
{
for(int d=0;d<n;d++)
{
if(flag[c][d]==2)
{
if(dist[c][d]<min)
{
min=dist[c][d];
minx=c;
miny=d;
}
}
}
}
//cout<<minx<<" "<<miny<<endl;
}
//cout<<dist[x][y]<<endl;
//cout<<t<<endl;
if(dist[x][y]<=t)
{
printf("YES\n");
int ans=t-dist[x][y];
printf("%d\n",ans);
}
else
printf("NO\n");
return 0;
}
