#include <stdio.h>
#include<algorithm>
#include<vector>
#define MAX 999999999
using namespace std;
int time[75][75];

bool processed[75][75];
struct cell
{
    int x;
    int y;
    int time;
};
vector<cell>V;
int m,n,posx,posy,t;

bool cmp(cell A,cell B)
{
    if(A.time<B.time)
        return false;
    else 
        return true;
}
void push(cell C)
{
    V.push_back(C);
    push_heap(V.begin(),V.end(),cmp);
}
cell pop()
{
    cell C=V.front();
    pop_heap(V.begin(),V.end(),cmp);
    V.pop_back();
    return C;
}
int calc_cost()
{
    
    cell start,pres,next;
    start.x=0;
    start.y=0;
    start.time=time[0][0];
    V.push_back(start);
    int v,x,y,t;
    while(!V.empty())
    {
        pres=pop();
        processed[pres.x][pres.y]=true;
        if(pres.x==posx && pres.y==posy)
        {
          v=pres.time;//the position of the princess
          return v;
        }
        x=pres.x;
        y=pres.y;
        t=pres.time;
        /*
         pres.x-1 must be >=0
         pres.x+1 must be <m
         
         pres.y-1 must be >=0
         pres.y+1 must be <n
         */
        if(x-1>=0 && !processed[x-1][y])
        {
            next.x=x-1;
            next.y=y;
            next.time=t+time[x-1][y];
            push(next);
        }
        if(x+1<m && !processed[x+1][y])
        {
            next.x=x+1;
            next.y=y;
            next.time=t+time[x+1][y];
            push(next);
        }
        if(y-1>=0 && !processed[x][y-1])
        {
            next.x=x;
            next.y=y-1;
            next.time=t+time[x][y-1];
            push(next);
        }
        if(y+1<n && !processed[x][y+1])
        {
            next.x=x;
            next.y=y+1;
            next.time=t+time[x][y+1];
            push(next);
        }
    }
}
void initialize()
{
    int i,j;
    for(i=0;i<=m;i++)
    {
        for(j=0;j<=n;j++)
        {
            processed[i][j]=false;
            
        }
    }
}
/*
 * 
 */
int main() 
{
    int j,i;
    scanf("%d%d",&m,&n);
    for(i=0;i<m;i++)
    {
        for(j=0;j<n;j++)
        {
            scanf("%d",&time[i][j]);
        }
    }
    scanf("%d%d%d",&posx,&posy,&t);
    posx-=1;
    posy-=1;
    initialize();
    int res=calc_cost();
    if(t-res<0)
        printf("\nNO");
    else
    {
        printf("\nYES");
        printf("\n%d",t-res);
    }
    return 0;
}
