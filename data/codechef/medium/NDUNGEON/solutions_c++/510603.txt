#include<iostream>
#include<stdio.h>
#define max_dis 1000000000
using namespace std;
int main()
{
	int n,m;
	int timestore[101][101];
	int done[101][101];
	int dis[101][101];
	int end_x,end_y,blasttime;
	int i,j;
	while(scanf("%d",&n)!=EOF)
	{
		cin>>m;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			{
				done[i][j]=0;
				dis[i][j]=max_dis;
				cin>>timestore[i][j];
			}
		}
		cin>>end_x>>end_y>>blasttime;
		done[1][1]=1;
		dis[1][1]=timestore[1][1];
		int x_c,y_c;
		int min_x,min_y;
		int min;
		x_c=1;y_c=1;
		while(1)
		{
			if(x_c+1<=n&&done[x_c+1][y_c]==0)
			{
                              if(dis[x_c+1][y_c]>dis[x_c][y_c]+timestore[x_c+1][y_c])
				      dis[x_c+1][y_c]=dis[x_c][y_c]+timestore[x_c+1][y_c];
				 
			
			}
			if(x_c-1>=1&&done[x_c-1][y_c]==0)
			{
                              if(dis[x_c-1][y_c]>dis[x_c][y_c]+timestore[x_c-1][y_c])
				      dis[x_c-1][y_c]=dis[x_c][y_c]+timestore[x_c-1][y_c];

			}
			if(y_c+1<=n&&done[x_c][y_c+1]==0)
			{
				if(dis[x_c][y_c+1]>dis[x_c][y_c]+timestore[x_c][y_c+1])
					dis[x_c][y_c+1]=dis[x_c][y_c]+timestore[x_c][y_c+1];
			}
			if(y_c-1>=1&&done[x_c][y_c-1]==0)
			{
				if(dis[x_c][y_c-1]>dis[x_c][y_c]+timestore[x_c][y_c-1])
					dis[x_c][y_c-1]=dis[x_c][y_c]+timestore[x_c][y_c-1];
			
			}
			min=max_dis;
			for(i=1;i<=n;i++)
			{
				for(j=1;j<=m;j++)
				{
					if(done[i][j]==0)
					{
						if(dis[i][j]<min)
						{
							min=dis[i][j];
							min_x=i;
							min_y=j;
						}
					}
				}
			}
	         //	cout<<min_x<<" "<<min_y<<min<<endl;
			if(min_x==end_x&&end_y==min_y)
				break;
			done[min_x][min_y]=1;
			x_c=min_x;
			y_c=min_y;



                       
		}
		if(blasttime>dis[end_x][end_y])
		   cout<<"YES"<<endl<<blasttime-dis[end_x][end_y];
		else
			cout<<"NO"<<endl;
	}
	return 0;
}
