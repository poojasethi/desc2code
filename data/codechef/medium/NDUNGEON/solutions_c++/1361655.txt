#include <iostream>
#define l 1000000005
using namespace std;
int ar[75][75],c[75][75],d[75][75];
int main()
{
    int m,n,a,b,t,i,j,x,y,mini;
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
        for(j=0;j<m;j++)
        {
            cin>>ar[i][j];
            c[i][j]=l;
        }
    }
    cin>>a>>b>>t;
    a--;
    b--;
    c[0][0]=ar[0][0];
    while(1)
    {
        mini=l;
        x=-1;y=-1;
        for(i=0;i<n;i++)
        {
            for(j=0;j<m;j++)
            {
                if(c[i][j]<mini&&d[i][j]==0)
                {
                    mini=c[i][j];
                    x=i;y=j;
                }
            }
        }
        if(x==a&&y==b)
        break;
        if(x+1<n&&d[x+1][y]==0)
        c[x+1][y]=min(c[x+1][y],c[x][y]+ar[x+1][y]);
        if(y+1<m&&d[x][y+1]==0)
        c[x][y+1]=min(c[x][y+1],c[x][y]+ar[x][y+1]);
        if(x-1>=0&&d[x-1][y]==0)
        c[x-1][y]=min(c[x-1][y],c[x][y]+ar[x-1][y]);
        if(y-1>=0&&d[x][y-1]==0)
        c[x][y-1]=min(c[x][y-1],c[x][y]+ar[x][y-1]);
        d[x][y]=1;
    }
    if(c[a][b]<=t)
    {
        cout<<"YES"<<endl<<t-c[a][b]<<endl;
    }
    else
    cout<<"NO"<<endl;
    return 0;
}
