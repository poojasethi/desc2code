#include <iostream>
#include <cstdio>
#include <queue>
using namespace std;
const char fi[] = "NDUNGEON.INP";
const int MAXN = 71;
const int INF = 0x7fffffff;

int a[MAXN][MAXN];
int d[MAXN][MAXN];
int m,n;
queue< pair<int,int> > q;
const int dx[] = {0,0,1,-1};
const int dy[] = {1,-1,0,0};

int main()
{
    int px,py,t;
    int i,j;
    int u,v;
    int x,y;
    pair<int,int> cur;

    //freopen(fi,"r", stdin);
    scanf("%d%d", &m, &n);
    for (i=0;i<m;i++)
    for (j=0;j<n;j++)
    {
        scanf("%d", &a[i][j]);
        d[i][j] = INF;
    }
    scanf("%d%d%d", &px, &py, &t);
    px--; py--;

    q.push(make_pair(0,0));
    d[0][0] = a[0][0];
    while (!q.empty())
    {
        cur = q.front();
        q.pop();
        u = cur.first; v = cur.second;
        for (i=0;i<4;i++)
        {
            x = u+dx[i];
            y = v+dy[i];
            if (x >= 0 && x < m && y >= 0 && y < n)
            {
                if (d[x][y] > d[u][v] + a[x][y])
                {
                    d[x][y] = d[u][v] + a[x][y];
                    q.push(make_pair(x,y));
                }
            }
        }
    }
    if (d[px][py] > t) printf("NO\n");
    else printf("YES\n%d\n", t-d[px][py]);
    return 0;
}
