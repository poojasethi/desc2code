count = int( raw_input() )
for i in xrange( count ):
    s = list( raw_input() )
    for t1 in xrange( len(s) - 1, 2, -1 ):
        if  (s[t1]    =="F" or s[t1]    =="?") and \
            (s[t1 - 1]=="E" or s[t1 - 1]=="?") and \
            (s[t1 - 2]=="H" or s[t1 - 2]=="?") and \
            (s[t1 - 3]=="C" or s[t1 - 3]=="?"):
                s[t1-3:t1+1] = "CHEF"
    for t1 in xrange( len( s ) ):
        if s[t1]=="?": 
            s[t1]="A"           
    print "".join( s )
