#include<stdio.h>
#include<stdlib.h>
#include<string.h>
main()
{
	int T,i,j;
	char str[2014];
	scanf("%d",&T);
	while(T--)
	{
		scanf("%s",str);
		i=strlen(str)-1;
		while(i!=(-1))
		{
			if((str[i-3]=='?'||str[i-3]=='C')&& (str[i-2]=='?'||str[i-2]=='H')&&(str[i-1]=='?'||str[i-1]=='E')&&(str[i]=='?'||str[i]=='F'))
			{
				str[i-3]='C';
				str[i-2]='H';
				str[i-1]='E';
				str[i]='F';
				i=i-4;
			}
			else if(str[i]=='?')
			{
			str[i]='A';
			i--;}
			else
			i--;
		}
		printf("%s\n",str);
	}
	
}
