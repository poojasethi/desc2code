#include<iostream>
#include<cstring>
using namespace std;

int main()
{
	int t;
	cin>>t;
	for(int T=0;T<t;T++)
	{
		char s[2013];
		cin>>s;
		int l=strlen(s);
		for(int i=l-1;i>=3;i--)
		{
			if((s[i]=='F'||s[i]=='?') && (s[i-1]=='E'||s[i-1]=='?') && (s[i-2]=='H'||s[i-2]=='?') && (s[i-3]=='C'||s[i-3]=='?'))
				{
					s[i]='F';
					s[i-1]='E';
					s[i-2]='H';
					s[i-3]='C';
				}
		}
		for(int j=0;j<l;j++)
		{
			if(s[j]=='?')
			s[j]='A';
		}
		cout<<s<<endl;
	}
}
