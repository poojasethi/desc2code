#include<stdio.h>
#include<iostream>
#include<string.h>
int main()
{ int t;
  char s[3000];
  scanf("%d",&t);
  while(t--)
       { scanf("%s",&s);
         for(int i=strlen(s)-1;i>=0;i--)
            { if((s[i]=='?' || s[i]=='F') && (s[i-1]=='?' || s[i-1]=='E') && (s[i-2]=='?' || s[i-2]=='H') && (s[i-3]=='?' || s[i-3]=='C'))
                { s[i--]='F';
                  s[i--]='E';
                  s[i--]='H';
                  s[i]='C';
                }
              else 
				  if(s[i]=='?') s[i]='A';  
            }
         printf("%s\n",s);   
       }
}
