#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
char a[2015];
int main()
{
    int t,i,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",a);
        n=strlen(a);
        for(i=n-4;i>=0;i--)
        {
            if(!(a[i]=='?'||a[i]=='C'))
                continue;
            if(!(a[i+1]=='?'||a[i+1]=='H'))
                continue;
            if(!(a[i+2]=='?'||a[i+2]=='E'))
                continue;
            if(!(a[i+3]=='?'||a[i+3]=='F'))
                continue;

                a[i]='C';
                a[i+1]='H';
                a[i+2]='E';
                a[i+3]='F';

        }
        for(i=0;i<n;i++)


            if(a[i]=='?')
            {
                a[i]='A';

            }

        printf("%s\n",a);
    }
    return 0;

}
