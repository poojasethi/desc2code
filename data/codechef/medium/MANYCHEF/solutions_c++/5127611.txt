#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int t,i,flag=0;
    char str[2020];
    scanf(" %d",&t);
    while(t>0)
    {
        t--;
        scanf(" %s",str);
        flag=0;
        i=strlen(str);
            while(i>=0)
            {
                if(str[i]=='F'||str[i]=='?')
                {
                    if((i-1)>=0&&(str[i-1]=='E'||str[i-1]=='?'))
                    {
                        if((i-2)>=0&&(str[i-2]=='H'||str[i-2]=='?'))
                        {
                            if((i-3)>=0&&(str[i-3]=='?'||str[i-3]=='C'))
                            {
                                str[i-3]='C';
                            }
                            else
                                flag=1;
                                    if(!flag&&str[i-3]=='C')
                                        str[i-2]='H';
                        }
                        else
                            flag=1;
                            if(!flag&&str[i-2]=='H')
                            str[i-1]='E';
                    }
                    else
                        flag=1;
                        if(!(flag)&&str[i-1]=='E')
                            str[i]='F';
                            if(flag&&str[i]=='?')
                    str[i]='A';
                }
                i--;
                flag=0;
                }
        printf("%s\n",str);
    }
}
