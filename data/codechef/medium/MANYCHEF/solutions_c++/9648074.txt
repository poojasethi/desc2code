#include<bits/stdc++.h>
using namespace std;
int ans[2100];
int main(){
	int t;
	cin >> t;
	while(t--){
		string s;
		cin >> s;
		int N = s.size();
		ans[N] = 0;
		for(int i = N - 1;i >= 0;--i){
			if(i + 3 < N && (s[i] == 'C' || s[i] == '?')&&
				(s[i + 1] == 'H' || s[i + 1] == '?')&&
				(s[i + 2] == 'E' || s[i + 2] == '?')&&
				(s[i + 3] == 'F' || s[i + 3] == '?')){
				ans[i] = ans[i + 4] + 1;
			} else{
				ans[i] = ans[i + 1];
			}
		}
		for(int i = 0;i < N;){
			if(ans[i] == ans[i + 1]){
				if(s[i] == '?')
					s[i] = 'A';
				++i;
			} else{
				s[i + 0] = 'C';
				s[i + 1] = 'H';
				s[i + 2] = 'E';
				s[i + 3] = 'F';
				i += 4;
			}
		}
		cout << s << "\n";
	}
	return 0;
}
