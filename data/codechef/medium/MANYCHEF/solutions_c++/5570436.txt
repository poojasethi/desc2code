    #include <iostream>
    using namespace std;
     
    int main() {
    int t;
    string s;
    cin>>t;
    for(int k=1;k<=t;k++)
    {
    cin>>s;
    
    for(int i=s.size()-4;i>=0;i--)
    {
    
    
    if((s.at(i)=='?'||s.at(i)=='C')&&(s.at(i+1)=='H'||s.at(i+1)=='?')&&(s.at(i+2)=='E'||s.at(i+2)=='?')&&(s.at(i+3)=='F'||s.at(i+3)=='?'))
    {
    s[i]='C';
    s[i+1]='H';
    s[i+2]='E';
    s[i+3]='F';
    }
   
    
    }
    for(int i=0;i<s.size();i++)
    {
    	if(s[i]=='?')
    	s[i]='A';
    }
   
    cout<<s<<endl;
    }
    return 0;
    }
