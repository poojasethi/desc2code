#include <bits/stdc++.h>

using namespace std;

int main()
{
    //freopen("input.txt", "r", stdin);
    int t;
    string s;
    for(cin >> t;t--;)
    {
        cin >> s;
        for(int i = s.length() - 1;i >= 3;--i)
        {
            if(s[i-3] == '?' or s[i-3] == 'C')
                if(s[i-2] == '?' or s[i-2] == 'H')
                    if(s[i-1] == '?' or s[i-1] == 'E')
                        if(s[i] == '?' or s[i] == 'F')
                        {
                            s[i-3] = 'C';
                            s[i-2] = 'H';
                            s[i-1] = 'E';
                            s[i] = 'F';
                            i -= 3;
                        }
        }
        for(int i = 0;i < s.length();++i)
            if(s[i] == '?') s[i] = 'A';
        cout << s << endl;
    }
    return 0;
}
