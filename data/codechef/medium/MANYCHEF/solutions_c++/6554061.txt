#include<iostream>
#include<string.h>

using namespace std;

int main()
{
    int t, len, i;
    char str[2014];
    cin>>t;

    while(t--)
    {
        cin>>str;
        len = strlen(str);

        for(i = len-1; i >= 0; i--)
        {

            if((i-3 >= 0) && (str[i] == '?' || str[i] == 'F') && (str[i-1] == '?' || str[i-1] == 'E') && (str[i-2] == '?' || str[i-2] == 'H') && (str[i-3] == '?' || str[i-3] == 'C'))
            {
                str[i] = 'F';
                str[i-1] = 'E';
                str[i-2] = 'H';
                str[i-3] = 'C';
                i = i-3;
            }
            else if(str[i] == '?')
            {
                str[i] = 'A';
            }
        }
        cout<<str<<"\n";
    }
    return 0;
}
