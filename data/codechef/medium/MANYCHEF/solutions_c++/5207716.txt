#include<iostream>
#include<string.h>
using namespace std;

int main()
{
int t;
cin>>t;
while(t--)
{
char c[2014];
cin>>c;
int i;
for(i=strlen(c)-1;i>=0;)
   {
   if((c[i]=='F'||c[i]=='?')&&(c[i-1]=='E'||c[i-1]=='?')&&(c[i-2]=='H'||c[i-2]=='?')&&(c[i-3]=='C'||c[i-3]=='?'))
     {
     c[i-3]='C';
     c[i-2]='H';
     c[i-1]='E';
     c[i]='F';
     i-=4;
     }
   else if(c[i]=='?')
          c[i--]='A';
        else
          i--;
   }
 cout<<c<<endl;
 }
return 0;
}
