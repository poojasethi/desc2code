#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
 
using namespace std;
int main()
{
	int t;
	char s[2050];
	cin>>t;
	getchar();
	while(t--)
	{
		gets(s);
		int len=strlen(s);
		for(int i=len-1;i>=0;i--)
		{
			if(s[i]=='?'||s[i]=='F')
			{
				if(s[i-1]=='E' || s[i-1]=='?')
				{
					if(s[i-2]=='H' || s[i-2]=='?')
					{
						if(s[i-3]=='C' || s[i-3]=='?')
						{
							s[i]='F';
							s[i-1]='E';
							s[i-2]='H';
							s[i-3]='C';
							i-=3;
						}
					}
				}
			}
		}
		for(int i=0;i<len;i++) if(s[i]=='?') s[i]='A';
		printf("%s\n",s);
	}
	return 0;
}
