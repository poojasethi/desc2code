#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int test,i,n;
	char a[2014];
	cin>>test;
	while(test--)
	{
		cin>>a;
		n=strlen(a);
		for(i=n-4;i>=0;--i)
		{
			if((a[i]=='?'||a[i]=='C') && (a[i+1]=='?'||a[i+1]=='H') && (a[i+2]=='?'||a[i+2]=='E') && (a[i+3]=='?'||a[i+3]=='F'))
			{
				a[i]='C';
				a[i+1]='H';
				a[i+2]='E';
				a[i+3]='F';
			}			
		}
		for(i=0;i<n;++i)
		{
			if(a[i]=='?')
			a[i]='A';
		}
		cout<<a<<"\n";
	}
	return 0;
}
