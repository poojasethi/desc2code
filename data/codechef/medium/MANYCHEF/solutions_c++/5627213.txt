#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,len,f1;
    cin>>t;
    while(t--)
    {
        string s;
        cin>>s;
        len=s.length();
        f1=0;
        for(i=len-1;i>2;i--)
        {
            f1=0;
            if(s[i]=='?'||s[i]=='F')
            {
                if(s[i-1]=='?'||s[i-1]=='E')
                {
                   if(s[i-2]=='?'||s[i-2]=='H')
                   {
                       if(s[i-3]=='?'||s[i-3]=='C')
                       {
                           f1=1;
                       }
                   }
                }
            }
            if(f1==1)
            {
                    s[i]='F';
                    s[i-1]='E';
                    s[i-2]='H';
                    s[i-3]='C';
            }
        }
        for(i=0;i<len;i++)
        {
            if(s[i]=='?')
            {
                s[i]='A';
            }
        }
        cout<<s<<endl;
    }
    return 0;
}
