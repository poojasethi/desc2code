#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		string s;
		cin >> s;
		int n = s.size();
		int arr[n];
		memset(arr,0,sizeof(arr));
		if (s[n-1] == '?')
			arr[n-1] = 1;
		for (int i = n-2; i >= 0; i--)
		{
			if (s[i] == '?')
				arr[i] = 1+arr[i+1];
			else
				arr[i] = 0;
		}
		for (int i = n-1; i >= 0;)
		{
			if ((i-3) >= 0)
			{
				if ((s[i] == 'F' || s[i] == '?') && (s[i-1] == 'E' || s[i-1] == '?') && (s[i-2] == 'H' || s[i-2] == '?') && (s[i-3] == 'C' || s[i-3] == '?'))
				{
					s[i] = 'F';
					s[i-1] = 'E';
					s[i-2] = 'H';
					s[i-3] = 'C';
					i = i-4;
				}
				else {
					if (s[i] == '?')
					s[i] = 'A';
					i--;
				}
			}
			else
			{
				if (s[i] == '?')
					s[i] = 'A';
				i--;
			}
		}
		cout << s << endl;
	}
	return 0;
}
