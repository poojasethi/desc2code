#include<stdio.h>
#include<string.h>
#include<iostream>
#include<algorithm>
#include<math.h>
#define gc getchar_unlocked // for linux
//#define gc getchar        // for windows
#define LL long long
using namespace std;

int inp()   //Signed long long data type
{
	int n=0;
	int ch=gc();int sign=1;
	while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=gc();}
	while( ch >= '0' && ch <= '9' )
	n = (n<<3)+(n<<1) + ch-'0', ch=gc();
	return(n*sign);
}
void scanint(int &x)  // Unsigned int data type
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
void scanlong(long long &x)     //Unsigned long long data type
{
    register long long c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

bool check(char str[], int start)
{
	bool temp=false;
	if(str[start]=='?' || str[start]=='C')
	{
		if(str[start+1]=='?' || str[start+1]=='H')
		{
			if(str[start+2]=='?' || str[start+2]=='E')
			{
				if(str[start+3]=='?' || str[start+3]=='F')
				{
					temp=true;
				}
			}
		}
	}
	return temp;
}
int main()
{
	int t,len,i;
	char str[2020];
	scanint(t);
	while(t--)
	{
		scanf("%s", str);
		len=strlen(str);
		for(i=len-1-3;i>=0;i--)
		{
			if(check(str,i))
			{
				str[i]='C';
				str[i+1]='H';
				str[i+2]='E';
				str[i+3]='F';
			}
		}
		for(i=0;i<len;i++)
		{
			if(str[i]=='?')
				str[i]='A';
		}
		printf("%s\n", str);
	}
	return 0;
}
