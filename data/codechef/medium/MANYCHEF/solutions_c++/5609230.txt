#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;
char ch[2014];

int main()
{
    ios_base::sync_with_stdio(false);
    int t,i,l;
    string str;
    cin>>t;
    while(t--)
    {
       cin>>str;
       l = str.length();
       for(i=l-1;i>=0;i--)
       {
           if(str[i]=='F' || str[i]=='?')
           {
               if(str[i-1]=='E' || str[i-1]=='?')
               {
                   if(str[i-2]=='H' || str[i-2]=='?')
                   {
                       if(str[i-3]=='C' || str[i-3]=='?')
                       {
                           ch[i--] = 'F';
                           ch[i--] = 'E';
                           ch[i--] = 'H';
                           ch[i] = 'C';
                           continue;
                       }
                   }
               }
           }
           if(str[i]=='?') ch[i] = 'A';
           else ch[i] = str[i];
       }
       for(i=0;i<l;i++)
        cout<<ch[i];
       cout<<endl;
    }
}
