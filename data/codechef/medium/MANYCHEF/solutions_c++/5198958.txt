#include <iostream>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(false);
	int T,i;
	string S;
	cin>>T;
	while(T--)
	{
		cin>>S;
		for(i=S.size()-1;i>=0;i--)
		{
			if((S[i]=='F'||S[i]=='?')&&(S[i-1]=='E'||S[i-1]=='?')&&(S[i-2]=='H'||S[i-2]=='?')&&(S[i-3]=='C'||S[i-3]=='?'))
			{
				S[i]='F';
				S[i-1]='E';
				S[i-2]='H';
				S[i-3]='C';
			}
			else if(S[i]=='?')
				S[i]='A';
		}
		cout<<S<<"\n";
	}
	return 0;
}
