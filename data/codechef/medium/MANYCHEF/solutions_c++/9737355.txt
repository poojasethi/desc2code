#include<bits/stdc++.h>
using namespace std;
int main(void)
{
    int t;
    for(scanf("%d", &t); t--; )
    {
        char s[10001];
        scanf("%s", s);
        int n = strlen(s);
        int i;
        for(i=n-1; i>=3; i--){
            if((s[i-3]=='C' || s[i-3]=='?') && (s[i-2]=='H' || s[i-2]=='?') && (s[i-1]=='E' || s[i-1]=='?') && (s[i]=='F' || s[i]=='?'))
            {
                s[i-3]='C';
                s[i-2]='H';
                s[i-1]='E';
                s[i]='F';
                i-=3;
            }
            else if(s[i]=='?')
                s[i]='A';
        }
        while(i>=0){
            if(s[i]=='?')
                s[i]='A';
            i--;
        }
        printf("%s\n", s);
    }
    return 0;
}
