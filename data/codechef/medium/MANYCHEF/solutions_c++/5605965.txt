using namespace std;
typedef long long int ll;

char str[2015];

int main(){
    int t;
    for( scanf("%d", &t); t; t-- ){
        scanf("%s", str);
        int len = strlen(str), i;
        for( i=len-1; i>=0; i-- ){
            if( ( i-3 >=0 ) && 
               ( str[i] == '?' || str[i] == 'F' ) &&
               ( str[i-1] == '?' || str[i-1] == 'E' ) &&
               ( str[i-2] == '?' || str[i-2] == 'H' ) &&
               ( str[i-3] == '?' || str[i-3] == 'C' ) ){
                str[i] = 'F';
                str[i-1] = 'E';
                str[i-2] = 'H';
                str[i-3] = 'C';
            }else if( str[i] == '?' )
                str[i] = 'A';
        }
        printf("%s\n", str);
    }
    return 0;
}
