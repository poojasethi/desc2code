#include <cstdio>
#include<cstring>
using namespace std;
 
int main() {
	int t;
	char s[2014];
	scanf("%d",&t);
	while(t--)
	{
	    scanf("%s",s);
	    int i,len=strlen(s);
	    for(i=len-1;i>=0;i--)
	    {
	            if((s[i-3]=='?' || s[i-3]=='C') &&(s[i-2]=='H'||s[i-2]=='?')&&(s[i-1]=='E'||s[i-1]=='?')&& (s[i]=='F'||s[i]=='?'))
	            {
	                s[i-3]='C';
	                s[i-2]='H';
	                s[i-1]='E';
	                s[i]='F';
	                i-=3;
	            }
	            else if(s[i]=='?')s[i]='A';
	    }
	    printf("%s\n",s);
	}
	return 0;
}
