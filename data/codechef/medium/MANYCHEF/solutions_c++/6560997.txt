bool CHEF(char C, char H, char E, char F) {
	if ((C == '?' || C == 'C') && (H == '?' || H == 'H') && (E == '?' || E == 'E') && (F == '?' || F == 'F')) {
		return true;
	}
	return false;
}

int main()
{
	int t;
	char s[2020];
	
	s(t);
	
	w(t) {
		ss(s);
		int l = strlen(s);
		
		for (int i = l - 1; i >= 0; ) {
			if (i >= 3 && CHEF(s[i - 3], s[i - 2], s[i - 1], s[i])) {
				s[i - 3] = 'C';
				s[i - 2] = 'H';
				s[i - 1] = 'E';
				s[i] = 'F';
				i -= 4; 
			} else {
				if (s[i] == '?') {
					s[i] = 'A';
				}
				i--;
			}
		}		
		printf("%s\n", s);
	}
	
	return 0;
}
