typedef long long int ll;
using namespace std;

const ll mod=1e9+7;

int main()
{

    ll t;
    scanf("%lld",&t);

    while(t--) {
       char s[2020];
       ll i;
       cin>>s;

       ll l=strlen(s);
       for(i=l-1;i>=0;i--) {
            if((s[i-3]=='C' || s[i-3]=='?') && (s[i-2]=='H' || s[i-2]=='?') && (s[i-1]=='E' || s[i-1]=='?') &&
               (s[i]=='F' || s[i]=='?') ) {
                s[i-3]='C';
                s[i-2]='H';
                s[i-1]='E';
                s[i]='F';
            }
       }

       for(i=0;s[i]!='\0';i++) {
            if(s[i]=='?')
                s[i]='A';
       }

       cout<<s<<endl;
    }

    return 0;
}
