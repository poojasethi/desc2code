inline void get(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int main()
{
	int T;
	get(T);
	for (int __rep = 1; __rep <=T; __rep++)
	{
		int m;
		double p;
		get(m);
		cin >> p;

		// if (p<=0.5 && m>1)

		sec = (p*(1-p_m1))/(1+p);
		LL x = 1e9;
		printf("%lf %lf\n", x*chef, x*sec);
	}

}
