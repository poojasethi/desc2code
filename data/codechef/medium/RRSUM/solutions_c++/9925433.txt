#include<stdio.h>
typedef long long ll;
int main()
{
	ll n,m;
	scanf("%lld%lld",&n,&m);
	ll x;
	while(m--)
	{
		scanf("%lld",&x);
			if(x<=2*n+1)
			{
				if((x-n-1)<0)
				{
					printf("0\n");
					continue;
				}
				printf("%lld\n",x-n-1);
				continue;
			}
			else
			{
				if((n-(x-2*n)+1)<0)
				{
					printf("0\n");
					continue;
				}
				printf("%lld\n",n-(x-2*n)+1);
				continue;
			}
	}
	return 0;
}
