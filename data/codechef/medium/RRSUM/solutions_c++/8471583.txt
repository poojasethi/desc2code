#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,m,q,i,j,k;
	scanf("%lld%lld",&n,&m);
	for(i = 0; i < m; i++)
	{
		scanf("%lld",&q);
		if(q < n+2) printf("0\n");
		else
		{
			j = q-n-2;
			k = 3*n-q;
			if(j > k) printf("%lld\n",++k);
			else printf("%lld\n",++j);
	    }
	}
	//for(i = 0; i < k; i++)printf("%d ",c[i]);
    return 0;
}
