#include<bits/stdc++.h>
using namespace std;
#define AIN(a,n,b) assert(a<=n && n<=b)
#define i64 long long
#define MAX 1000000
#define pb push_back
#define all(a) a.begin(),a.end()

typedef pair<int,int> pii;

char a[1005][1005];
bool vis[1005];
int main(){
    double cl = clock();
    long long n,q;
    int m;

    scanf("%lld %d",&n,&m);
    AIN(1,n,1000000000);
    AIN(1,m,100000);
    while(m--){
        scanf("%lld",&q);
        AIN(1,q,3*n);
        if(q<n+2){
            printf("0\n");
            continue;
        }
        if(3*n - q < q - n-2){
            printf("%lld\n",3*n - q + 1);
        }
        else printf("%lld\n",q - n - 1);
    }
    cl = clock() - cl;
    fprintf(stderr,"Execution Time = %lf sec\n",cl / CLOCKS_PER_SEC);
    return 0;
}
