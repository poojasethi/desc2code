#include <iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;

int main()
{
    long long int n,t;
   scanf("%lld %lld",&n,&t);

    while(t--)
    {

          long long int q,c;
         scanf("%lld",&q);
         if(q<n+2)
            c=0;
         else
         {
             if(q<=(2*n+1))
             {
                 c=q-(n+1);
             }
             else if(q<=3*n)
             {
                 c=abs(q-3*n)+1;
             }
             else
             {
                 c=0;
             }

         }

       printf("%lld\n",c);

    }
  return 0;
}
