#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    long long int n, m, q, count, avg1, avg2;
    scanf("%lld %lld", &n, &m);
    while(m--)
    {
        scanf("%lld", &q);
        avg1 = (n + 2) / 2;
        avg2 = (3*n + 1) / 2;
        if(q < n + 2)
        {
            count = 0;
        }
        else if(q == avg1 + avg2)
        {
            count = n;
        }
        else if(q < avg1 + avg2)
        {
            count = q - (n + 2) + 1;
        }
        else if(q > avg1 + avg2)
        {
            count = 3*n - q + 1;
        }
        printf("%lld\n", count);
    }
    return 0;
}
