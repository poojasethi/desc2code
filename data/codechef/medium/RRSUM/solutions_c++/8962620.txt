inline int lcs(char *a,char *b,int m,int n)
{
    int my[m+1][n+1];
    for(int i=0;i<=m;i++)
    {
        for(int j=0;j<=n;j++)
        {
            if(i==0 || j==0)
                my[i][j]=0;
            else if(a[i-1]==b[j-1])
                my[i][j]=my[i-1][j-1]+1;
            else
                my[i][j]=max(my[i-1][j],my[i][j-1]);
            
        }
    }
    
    return my[m][n];
    
}

{

{
    if(l<=r)
    {
        int m=(l+r)/2;
        
        
        if(arr[m]==x)
            return m;
        if(arr[m]>x)
            return BinarySearch(arr, 0, m, x);
        return BinarySearch(arr,m+1, r, x);

    return -1;
    
}
