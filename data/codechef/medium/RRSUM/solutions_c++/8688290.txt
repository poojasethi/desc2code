#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    long long n,m,q;
    cin>>n>>m;
    while(m--)
    {
        long long count=0;
        cin>>q;
        if(q==2*n+1)
            count=n;
        else if(q<2*n+1&&q>=n+2)
            count=n-(2*n+1-q);
        else if(q>2*n+1&&q<=3*n)
            count=n-(q-(2*n+1));
        cout<<count<<endl;
    }
}
