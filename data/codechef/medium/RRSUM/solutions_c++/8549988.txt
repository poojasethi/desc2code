# include <iostream>
# include <cmath>
# include <cstdlib>
using namespace std;
int main()
{
   long long n,q,m,ans;
   cin>>n>>m;
   while(m--)
   {
      cin>>q;
      if(q<(n+1))
         ans=0;
      else if(q>2*n)
         ans=3*n-q+1;
      else
         ans=q-n-1;
      cout<<ans<<endl;                   
   }
   return 0;
}
