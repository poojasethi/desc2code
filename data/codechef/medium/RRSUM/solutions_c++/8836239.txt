#include <iostream>
#include <algorithm>
#include <map>
#include <set>

using namespace std;

const int mn = 1e5 + 100;
long long int n,m,i,x;
int main()
{
	cin>>n>>m;
	for(i=0;i<m;i++)
	{
		cin>>x;
		if(x <= n || x > 3*n)
		{
			cout<<"0\n";
			continue;
		}
		if(x-1 <= 2*n)
		{
			cout<<(x-n-1)<<endl;
		}
		else
		{
			if(x%n) cout<<n-(x%n)+1<<endl;
			else cout<<"1\n";
		}
	}
return 0;
}
