#include <iostream>
#include <algorithm>
#include <string>
using namespace std;

int main(){

        long long int n,m;
        cin>>n>>m;
        long long int x;
        long long int result=0;
        for(long long int i=0;i<m;i++){
            cin>>x;
            if(x<=n+1 || x>3*n){
                result=0;
            }
            else if(x<=2*n+1){
                    result=x-(n+1);
            }
            else{
                result=3*n-x+1;
            }
            cout<<result<<endl;
        }
    return 0;
}
