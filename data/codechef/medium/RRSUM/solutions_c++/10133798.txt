#include <bits/stdc++.h>
using namespace std;
 
int main() {
	long long int n,m,min,max,mid,a,ans;
	scanf("%lld%lld",&n,&m);
	min=n+2;
	max=3*n;
	mid=(min+max)/2;
	while(m)
	{
	scanf("%lld",&a);
	if(a<min||a>max)
	 printf("0\n");
	else
	 {
	 if(a>mid)
	  ans=n-(a-mid);
	 else
	  ans=n-(mid-a);
	 printf("%lld\n",ans);
 
	 }
	m--;
	}
	return 0;
}
