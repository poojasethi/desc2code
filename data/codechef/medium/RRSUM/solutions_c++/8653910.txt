#include <stdio.h>

int main(void) {
	// your code goes, here
	long long n,m,no,f,l,t;
	scanf("%lld %lld",&n,&m);
	while(m--)
	{
		scanf("%lld",&no);
		f=n+2;
		l=3*n;
		if(no<=2*n+1&&no>=f)
		{
			t=no-f;
			printf("%lld\n",t+1);
		}
		else if(no>2*n+1&&no<=l)
		{
			t=l-no;
			printf("%lld\n",t+1);
		}
		else
		 printf("0\n");
	}
	
	return 0;
}
