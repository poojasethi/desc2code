#include<iostream>
using namespace std;
int main(){
	long long int n,m,p;
	cin>>n>>m;
	p=2*n+1;
	while(m--){
		long long int q;
		cin>>q;
		if(q<=n+1||q>=3*n+1)
		cout<<"0"<<endl;
		else if(q==p)
		cout<<n<<endl;
		else if(q>p){
			q=q-p;
			cout<<n-q<<endl;
		}
		else{
			q=p-q;
			cout<<n-q<<endl;
		}
	}
}
