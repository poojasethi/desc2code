#include <iostream>
 
using namespace std;
int main(){
	long long N,M, num;
	cin >> N >> M;
	long long min = N+2;
	long long max = 3*N;
	long long mean = (min+max)/2;
	for(long long i=0;i<M;i++){
		cin >> num;
		if(num<min||num>max){
			cout << "0" << endl;
			continue;
		}
		if (num<=mean){
			cout << num-min+1<<endl;
		}else{	
			cout << max-num+1<<endl;
		}
		
	}
	return 0;
}
