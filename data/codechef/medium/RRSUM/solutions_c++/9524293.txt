#include<bits/stdc++.h>
    using  namespace std;
    int main()
    {
        long long n,m,q,l;
        cin>>n>>m;
        while(m--)
        {
            cin>>q;
            if(q<n+2)
            {
                cout<<"0"<<"\n";
                continue;
            }
            if(q<((2*n) +1))
                cout<<q - n -1<<"\n";
                else

                {
                    l=q-((2*n)+1);
                    q=(2*n) + 1 -l;
                    cout<<q-n-1<<"\n";
                }

            }

        return 0;
    }
