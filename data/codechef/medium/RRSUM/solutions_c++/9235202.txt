#include<cstdio>
using namespace std;
int main()
{
	long long n, m;
	scanf("%lld %lld", &n, &m);
	while (m--)
	{
		long long p;
		scanf("%lld", &p);
		if (p > 3 * n || p < n + 2)
			printf("0\n");
		else if (p < 2*n + 2)
		{
			printf("%lld\n", p - n - 1);
		}
		else
		{
			printf("%lld\n", 3 * n + 1 - p);
		}
	}
}
