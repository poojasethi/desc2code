     inline unsigned long int fastread()
    {
      unsigned int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
    using namespace std;
