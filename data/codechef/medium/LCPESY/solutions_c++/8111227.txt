#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	cin.ignore();
	while(t--)
	{
		char A[10001],B[10001];
		cin>>A>>B;
		int a[26]={0};
		int b[26]={0};
		int c[26]={0};
		int d[26]={0};
		int e[26]={0};
		int f[26]={0};
		int l1,l2,i,sum,sum1,sum2;
		l1=strlen(A);
		for(i=0;i<l1;i++)
	    {
	    	if(A[i]>=97&&A[i]<=122)
	        a[A[i]-'a']++;
	        else if(A[i]>=65&&A[i]<=90)
	        b[A[i]-'A']++;
	        else
	        c[A[i]-'0']++;
		}
		l2=strlen(B);
		for(i=0;i<l2;i++)
	    {
	    	if(B[i]>=97&&B[i]<=122)
	        d[B[i]-'a']++;
	        else if(B[i]>=65&&B[i]<=90)
	        e[B[i]-'A']++;
	        else
	        f[B[i]-'0']++;
		}
		for(i=0,sum=0;i<26;i++)
		{
			if(a[i]>=d[i])
			sum=sum+d[i];
			else
			sum=sum+a[i];
		}
		for(i=0,sum1=0;i<26;i++)
		{
			if(b[i]>=e[i])
			sum1=sum1+e[i];
			else
			sum1=sum1+b[i];
		}
		for(i=0,sum2=0;i<26;i++)
		{
			if(c[i]>=f[i])
			sum2=sum2+f[i];
			else
			sum2=sum2+c[i];
		}
		cout<<sum+sum1+sum2<<endl;
	}
}
