#include<iostream>
#include<cstring>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string a,b;
		cin>>a>>b;
		int ans=0;
		int ac[123]={0},bc[123]={0};
		for(int i=0;i<a.size();i++)
		ac[a[i]]++;
		for(int i=0;i<b.size();i++)
		bc[b[i]]++;
		
		/*for(int i=65;i<123;i++)
		cout<<ac[i]<<" ";
		cout<<"\n";
		for(int i=65;i<123;i++)
		cout<<bc[i]<<" ";*/
		
		for(int i=0;i<123;i++)
		ans+=min(ac[i],bc[i]);
		cout<<ans<<"\n";
	}	
}
