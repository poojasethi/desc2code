#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--) {
		string A,B;
		cin >> A >> B;

		int i;

		int c1[256] = {0};
		int c2[256] = {0};

		for(i = 0;i < A.size();i++) {
			c1[A[i]]++;
		}

		for(i = 0;i < B.size();i++) {
			c2[B[i]]++;
		}	

		int c = 0;

		int res;
		for(i = 0;i < 256;i++) {
			res = min(c1[i],c2[i]);
			if(res > 0) 
				c += res;
		}

		cout << c << "\n";
	}
	return 0; 
}
