#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
    char str1[10001],str2[10001];
    int t,a,b,c,d,e,f;
    cin>>t;
    while(t--)
    {
        cin>>str1;
        cin>>str2;
        a=strlen(str1);
        b=strlen(str2);
        int A[10001]={0}; 
        e=0;
        for(c=0;c<b;c++)
        {
            if(A[str2[c]]==0)
            {
            d=count(str1,str1+a,str2[c]);
            f=count(str2,str2+b,str2[c]);
            e=e+min(d,f);
            A[str2[c]]=1;
            }
        }
        cout<<e<<endl;
    }
    return 0;
}
