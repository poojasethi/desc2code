#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
int test;
cin>>test;
while(test--)
{

int arr[200],count=0;
for(int i=0;i<200;i++)
    arr[i]=0;

int arr2[200];
for(int i=0;i<200;i++)
    arr2[i]=0;
    string str,str2;
    cin>>str;
    cin>>str2;
    for(int i=0;i<str.length();i++)
    {
        arr[str[i]]+=1;

    }
    for(int i=0;i<str2.length();i++)
    {
        arr2[str2[i]]+=1;

    }
    for(int i=0;i<200;i++)
    {
      count+=min(arr[i],arr2[i]);
    }
    cout<<count<<endl;

}

    return 0;
}
