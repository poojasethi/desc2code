#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	cin.ignore();
	while(t--)
	{
		char A[10001],B[10001];
		cin>>A>>B;
		int i,sum;
		int a[123]={0};
		int b[123]={0};
		for(i=0;A[i];i++)
		a[A[i]]++;
		for(i=0;B[i];i++)
		b[B[i]]++;
		for(i=0,sum=0;i<123;i++)
		{
			if(a[i]>=b[i])
			sum=sum+b[i];
			else
			sum=sum+a[i];
		}
		cout<<sum<<endl;
	}
}
