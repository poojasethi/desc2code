#include<bits/stdc++.h>
using namespace std;
int main(void)
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		char a[10001], b[10001];
		scanf("%s", a);
		scanf("%s", b);
		int aalpha[256];
		int balpha[256];
		for(int i=0; i<256; i++)
		{
			aalpha[i] = balpha[i] = 0;
		}
		for(int i=0; i<strlen(a); i++){
			aalpha[(int)a[i]]++;
		}
		for(int i=0; i<strlen(b); i++){
			balpha[(int)b[i]]++;
		}
		int cnt=0;
		for(int i=0; i<256; i++)
			cnt+=(aalpha[i]<=balpha[i])?aalpha[i]:balpha[i];
		printf("%d\n", cnt);
	}
	return 0;
}
