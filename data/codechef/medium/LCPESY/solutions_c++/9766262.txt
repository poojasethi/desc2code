#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int t,i;
	int a1[150],b1[150],c,ans;
	char a[10009],b[10009],ch;
	scanf("%d",&t);
	while(t--)
	{
		for(i=0;i<150;i++)
		{
			a1[i]=0;
			b1[i]=0;
		}
		ans=0;
		scanf("%s %s",a,b);
		for(i=0;a[i]!='\0';i++)
		{
			ch=a[i];
			c=ch;
			a1[c]++;
		}
		for(i=0;b[i]!='\0';i++)
		{
			ch=b[i];
			c=ch;
			b1[c]++;
		}
		for(i=0;i<150;i++)
		{
			if((a1[i]!=0)&&(b1[i]!=0))
			{
				ans=ans+min(a1[i],b1[i]);
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
