#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char a[99999],b[99999];
		long long int i,c[300],d,count=0,e[300];
		for(i=0;i<300;i++)
		{
			c[i]=0;
			e[i]=0;
		}
		scanf("%s%s",a,b);
		for(i=0;a[i]!='\0';i++)
		{
			d=a[i];
			c[d]++;
			
		}
		for(i=0;b[i]!='\0';i++)
		{
			d=b[i];
			e[d]++;
			
		}
		for(i=0;i<300;i++)
		{
			count+=min(c[i],e[i]);
		}
	
		printf("%lld\n",count);
		
	}
	return 0;
}
