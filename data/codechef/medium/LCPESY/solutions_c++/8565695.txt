#include <iostream>
#include<set>
#include<string>
using namespace std;

string a,b;
int t,i,j;
int main()
{
    cin>>t;
    while(t--)
    {
            int count=0;
            int c[400]={0},d[400]={0};
            cin>>a>>b;
            for(i=0;i<a.length();i++)
            {
                c[a[i]]++;
            }
            for(i=0;i<b.length();i++)
            {
                d[b[i]]++;
            }
            for(i=0;i<260;i++)
            {
                count=count+min(c[i],d[i]);
            }
            cout<<count<<"\n";
    }


    return 0;
}
