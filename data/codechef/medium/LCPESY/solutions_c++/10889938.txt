#include <bits/stdc++.h>

using namespace std;

int  main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		string s,s1;
		cin >> s;
		cin >> s1;
		sort(s1.begin(),s1.end());
		sort(s.begin(),s.end());
		int n1 = s.size();
		int n2 = s1.size();
		int i = 0;
		int j = 0;
		int ctr = 0;
		while (i < n1 && j < n2)
		{
			if (s[i] == s1[j])
			{
				i++;
				j++;
				ctr++;
			}
			else if (s[i] < s1[j])
			{
				i++;
			}
			else if (s[i] > s1[j])
			{
				j++;
			}
		}
		cout << ctr << endl;
	}
	return 0;
}
