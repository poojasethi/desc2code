#include<bits/stdc++.h>
using namespace std;
int mi(int a,int b)
{

    if(a<b)
        return a;
    else return b;
}
main()
{
    int T;
    cin>>T;
    while(T-->0)
    {
        char s1[10002];
        char s2[10002];
        cin>>s1>>s2;
        int l1=strlen(s1);
        int l2=strlen(s2);
        int i;
        int sm1[26]={0};
        int sm2[26]={0};
        int lg1[26]={0};
        int lg2[26]={0};
        int dg1[10]={0};
        int dg2[10]={0};
        int count=0;
        for(i=0;i<l1;i++)
        {
            if(s1[i]>='a' && s1[i]<='z')
                sm1[s1[i]-'a']++;
            else if(s1[i] >='A' && s1[i]<='Z')
                lg1[s1[i]-'A']++;
                else
                    dg1[s1[i]-'0']++;
        }
        for(i=0;i<l2;i++)
        {
            if(s2[i]>='a' && s2[i]<='z')
                sm2[s2[i]-'a']++;
            else if(s2[i] >='A' && s2[i]<='Z')
                lg2[s2[i]-'A']++;
                else
                    dg2[s2[i]-'0']++;
        }
        for(i=0;i<26;i++)
        {
            if(sm2[i]!=0 && sm1[i]!=0)
            count+=mi(sm2[i],sm1[i]);
            if(lg2[i]!=0 && lg1[i]!=0)
            count+=mi(lg2[i],lg1[i]);
        }
        for(i=0;i<10;i++){
            if(dg2[i]!=0 && dg1[i]!=0)
                count+=mi(dg1[i],dg2[i]);
        }
        cout<<count<<endl;
    }

}
