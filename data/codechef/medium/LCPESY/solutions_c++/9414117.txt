#include<iostream>
#include<cstring>
#include<cstdio>
int main()
{
int T,n1,n2,i,count;
char s1[40000],s2[40000];
scanf("%d",&T);
while(T--)
{
       int a[75]={0};
       scanf("%s%s",s1,s2);
       n1=strlen(s1);
       n2=strlen(s2);
       count=0;
       for(i=0;i<n1;i++)
              a[s1[i]-48]++;
       for(i=0;i<n2;i++)
       {
              if(a[s2[i]-48]>0)
              {
                     count++;
                     a[s2[i]-48]--;
              }
       }
       printf("%d\n",count);
}
return 0;
}
