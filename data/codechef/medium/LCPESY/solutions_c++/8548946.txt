#include <iostream>
#include <stdio.h>
#include <cmath>
#include <string.h>
#include <stack>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;


int main()
{
	int t;
	int i;
	
	cin >> t;
	string a,b;
	int a1[26], a2[26];
	int b1[26], b2[26];
	int a3[10],b3[10];

	while (t--)
	{
		cin >> a;
		cin >> b;
		for (i=0;i<26;i++)
		{
			a1[i] = a2[i] = 0;
			b1[i] = b2[i] = 0;
			
		}

		for (i=0;i<10;i++)
			a3[i] = b3[i] = 0;

		for (i=0; a[i]!='\0';i++)
		{
			if (a[i]>='a' && a[i]<='z')
				a1[a[i]-'a']++;
			if (a[i]>='A' && a[i]<='Z')
				a2[a[i]-'A']++;
			else
				a3[a[i]-'0']++;
		}


		for (i=0; b[i]!='\0';i++)
		{
			if (b[i]>='a' && b[i]<='z')
				b1[b[i]-'a']++;
			if (b[i]>='A' && b[i]<='Z')
				b2[b[i]-'A']++;
			else
				b3[b[i]-'0']++;
		}

		int ans=0;
		for (i=0;i<26;i++)
		{
			if (a1[i]<b1[i])
				ans += a1[i];
			else
				ans += b1[i];

			if (a2[i]<b2[i])
				ans += a2[i];
			else
				ans += b2[i];

			
		}

		for (i=0;i<10;i++)
		{
			if (a3[i]<b3[i])
				ans += a3[i];
			else
				ans += b3[i];
		}
		cout << ans << endl;

	}
	return 0;

}
