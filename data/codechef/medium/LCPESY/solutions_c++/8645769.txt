#include <bits/stdc++.h>

using namespace std;

int a[62],b[62];

int main(){
    long tt;
    //ifstream cin("a.txt");
    string s1,s2;
    cin>>tt;
    while(tt--){
        cin>>s1>>s2;
        memset(a,0,sizeof a);
        memset(b,0,sizeof a);
        for(int i=0;i<s1.length();i++){
            if(s1[i]>='a'&&s1[i]<='z')
                a[s1[i]-97]++;
            else if(s1[i]>='A'&&s1[i]<='Z')
                a[s1[i]-65+26]++;
            else
                a[s1[i]-48+52]++;
        }
        for(int i=0;i<s2.length();i++){
            if(s2[i]>='a'&&s2[i]<='z')
                b[s2[i]-97]++;
            else if(s2[i]>='A'&&s2[i]<='Z')
                b[s2[i]-65+26]++;
            else
                b[s2[i]-48+52]++;
        }

        long sum=0;

        for(int i=0;i<62;i++)
            sum+=min(a[i],b[i]);
        cout<<sum<<endl;
    }
}
