#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;cin>>t;
    while(t--)
    {
       int c=0,ar[256];
       memset(ar,0,sizeof(ar));
       string s1,s2;
       cin>>s1>>s2;
       for(int i=0;s1[i]!='\0';i++)
       {
           ar[s1[i]]++;
       }
           for(int j=0;s2[j]!='\0';j++)
           {
               if(ar[s2[j]]>0)
               {
                  c++;
                  ar[s2[j]]--;
               }
           }
       cout<<c<<endl;
    }
    return 0;
}
