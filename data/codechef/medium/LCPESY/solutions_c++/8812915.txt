#include<iostream>
#include<ctype.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    int t,count,flag,i,j,x;
    char a[100000],b[100000];
   scanf("%d",&t);
    while(t--)
    {
        scanf("%s%s",a,b);
        count=0;
        int c[450],d[450];
        for(i=0;i<450;i++){
            d[i]=0;c[i]=0;
        }
        for(i=0;a[i]!='\0';i++)
        {
            c[a[i]]++;
        }
        for(i=0;b[i]!='\0';i++)
        {
            d[b[i]]++;
        }
        for(i=0;i<450;i++)
        {
            count+=min(c[i],d[i]);
        }
        printf("%d\n",count);
    }
    return 0;
}
