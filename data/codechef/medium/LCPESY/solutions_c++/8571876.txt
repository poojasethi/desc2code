#include<bits/stdc++.h>
using namespace std;
void pre()
{
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  cout.tie(NULL);
}
inline int get_int()
{
  int n=0,neg=0,c=getchar_unlocked();
  while((c<'0'||c>'9')&&c!='-')c=getchar_unlocked();
  if(c=='-') {neg=1;c=getchar_unlocked();}
  while(c>='0'&&c<='9')n=(n<<3)+(n<<1)+c-'0',c=getchar_unlocked();
  if(neg) n=-n;
  return n;
}

int main()
{
  pre();
  int t;
  cin>>t;
  while(t--)
  {
      int c1[26]={0},c2[26]={0},c3[26]={0},c4[26]={0},d1[10]={0},d2[10]={0};
      string A,B;
      cin>>A;
      cin>>B;
      for(int i=0;i<A.length();i++)
      {
        if(A[i]>='a'&&A[i]<='z')
          c1[A[i]-'a']++;
        else if(A[i]>='A'&&A[i]<='Z')
          c2[A[i]-'A']++;
         else if(A[i]>='0'&&A[i]<='9')
         	d1[A[i]-'0']++;
      }
      for(int j=0;j<B.length();j++)
      {
        if(B[j]>='a'&&B[j]<='z')
          c3[B[j]-'a']++;
        else if(B[j]>='A'&&B[j]<='Z')
          c4[B[j]-'A']++;
         else if(B[j]>='0'&&B[j]<='9')
         	d2[B[j]-'0']++;
      }
      int ans=0;
      for(int i=0;i<26;i++)
      {
        if((c1[i]!=0)&&(c3[i]!=0))
        {
          ans+=min(c1[i],c3[i]);
        }
        if((c2[i]!=0)&&(c4[i]!=0))
        {
          ans+=min(c2[i],c4[i]);
        }
        if((d1[i]!=0)&&(d2[i]!=0)&&(i<10))
        {
        	ans+=min(d1[i],d2[i]);
        }
        
      }
      cout<<ans<<endl;
  }
  return 0;
}
