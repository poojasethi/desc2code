#include <iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int arr[256]={};
        string a,b;
        cin>>a>>b;
        for(int i=0;i<a.length();i++)
        {
            int x=a[i];
            arr[x-1]++;
        }
        int cnt=0;
        for(int i=0;i<b.length();i++)
        {
            int x=b[i];
            if(arr[x-1]!=0)
            {
                cnt++;
                arr[x-1]--;
            }
        }
        cout<<cnt<<endl;
    }
    return 0;
}
