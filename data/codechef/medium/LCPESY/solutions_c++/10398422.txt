#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t,i,j,a1,b1,count;
    cin>>t;
    while(t>0)
    {
        char a[10005]={'\0'},b[10005]={'\0'};
        count=0;
        cin>>a>>b;
        for(i=0;a[i]!='\0';i++){}
            a1=i;
        for(i=0;b[i]!='\0';i++){}
            b1=i;
        sort(a,a+a1);
        sort(b,b+b1);
        i=0;j=0;
        while(i!=(a1)&&j!=(b1))
        {
            if(a[i]==b[j])
            {
                count++;
                i=i+1;
                j=j+1;
            }
            else if(a[i]>b[j])
                j++;
            else if(a[i]<b[j])
                i++;
        }
        cout<<count<<"\n";
        t--;
    }
    return 0;
}
