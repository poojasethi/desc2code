#include<bits/stdc++.h>
using namespace std;
char s1[10016],s2[10016];
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
memset(s1,0,sizeof(s1));
memset(s2,0,sizeof(s2));
cin>>s1>>s2;
int a1[26]={0},b1[26]={0},a2[26]={0},b2[26]={0},a[10]={0},b[10]={0};
for(int i=0;i<strlen(s1);i++)
{
if(s1[i]>=97&&s1[i]<=122)
a1[s1[i]-97]++;
else if(s1[i]>=65&&s1[i]<=90)
     a2[s1[i]-65]++;
     else
     a[s1[i]-48]++;
}
for(int i=0;i<strlen(s2);i++)
{
     if(s2[i]>=97&&s2[i]<=122)
b1[s2[i]-97]++;
else if(s2[i]>=65&&s2[i]<=90)
     b2[s2[i]-65]++;
     else
     b[s2[i]-48]++;
}
int ans=0;
for(int i=0;i<26;i++)
ans=ans+min(a1[i],b1[i])+min(a2[i],b2[i]);
for(int i=0;i<10;i++)
ans=ans+min(a[i],b[i]);
printf("%d\n",ans);
}
return 0;
}
