#include<iostream>
using namespace std;
int main()
{
	char p[10001],q[10001];
	long int z,b[62],c[62];
	int t,i;
	cin>>t;
	while(t--)
	{
		//long int b[62]={0},c[62]={0};
		for(i=0;i<62;i++)
		{
			b[i]=0;
			c[i]=0;
		}
		cin>>p;
		cin>>q;
		for(i=0;p[i]!='\0';i++)
		{
			if(p[i]<='z'&&p[i]>='a')
			b[p[i]-'a'+10]++;
			else if(p[i]<='Z'&&p[i]>='A')
			b[p[i]-'A'+36]++;
			else
			b[p[i]-'0']++;
		}
		for(i=0;q[i]!='\0';i++)
		{
			if(q[i]<='z'&&q[i]>='a')
			c[q[i]-'a'+10]++;
			else if(q[i]<='Z'&&q[i]>='A')
			c[q[i]-'A'+36]++;
			else
			c[q[i]-'0']++;
		}
		z=0;
		for(i=0;i<62;i++)
		{
			if(b[i]<=c[i])
			z=z+b[i];
			else
			z=z+c[i];
		}
		cout<<z<<endl;
	}
	return 0;
}
