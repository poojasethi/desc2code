#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,count,count1,c;
	int arr[130];
	scanf("%d",&t);
	while(t--)
	{
		for(int i=0;i<130;i++)
		arr[i]=0;
		count=0;
		count1=0;
		c=0;
		char s[10005],s1[10005];
		scanf("%s",s);
		scanf("%s",s1);
		for(int i=0;s[i]!='\0';i++)
		count++;
		for(int i=0;s1[i]!='\0';i++)
		count1++;
		for(int i=0;i<count;i++)
		arr[s[i]]++;
		for(int i=0;i<count1;i++)
		if(arr[s1[i]]!=0)
		{
		arr[s1[i]]--;
		c++;
		}
		printf("%d\n",c);
	}
	return 0;
}
