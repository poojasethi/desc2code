#include<cstdio>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int abs(int a){
    if(a<0){
        return -a;
    }
    return a;
}
int MIN(int a,int b){
    if(a<b){
        return a;
    }
    return b;
}
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int c1[130],c2[130];
        for(int i=0;i<130;i++){
            c1[i]=0;
            c2[i]=0;
        }
        char s1[10005],s2[10005];
        scanf("%s %s",&s1,&s2);
        int i=0;
        while(s1[i]!='\0'){
            c1[s1[i]]++;
            i++;
        }
        i=0;
        while(s2[i]!='\0'){
            c2[s2[i]]++;
            i++;
        }
        int count=0;
        for(int i=0;i<130;i++){
            count+=MIN(c1[i],c2[i]);
        }
        printf("%d\n",count);
    }
}
