#include <iostream>
#include<string.h>
using namespace std;
int c[256],d[256];
int main()
{ int t,i,count,l1,l2;
 char a[10002],b[10002];
 cin>>t;
 while(t--)
 { memset(c,0,sizeof(c));
  memset(d,0,sizeof(d));
   count=0;
  cin>>a>>b;
   l1=strlen(a);
   l2=strlen(b);
   for(i=0;i<l1;i++)
    { c[a[i]]+=1;
    }
    for(i=0;i<l2;i++)
    { d[b[i]]+=1;
    }
    for(i=0;i<256;i++)
    {
      count+=min(c[i],d[i]);
      
    }
   cout<<count<<endl;
  }
   
 
 return 0;
}
