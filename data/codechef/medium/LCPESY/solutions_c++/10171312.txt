#include<bits/stdc++.h>
using namespace std;
map<char,int>count1;
map<char,int>count2;
int main()
{
	int test;int i;
	cin>>test;int t;
	string str1,str2;
	for(t=0;t<test;t++)
	{
		cin>>str1;
		cin>>str2;
		count1.clear();
		count2.clear();
		for(i=0;i<str1.size();i++)
		{
			count1[str1[i]]++;
		}
		for(i=0;i<str2.size();i++)
		{
			count2[str2[i]]++;
		}
		map<char,int>::iterator it;
		int result=0;
		for(it=count1.begin();it!=count1.end();it++)
		{
			result+=min(it->second,count2[it->first]);
		}
		cout<<result<<"\n";
	}
	
	return 0;
}//main
