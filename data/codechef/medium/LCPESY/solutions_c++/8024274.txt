#include <iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
 
int main() {
	int t;
	char a[10001],b[10001];
	cin>>t;
	for(int m=0;m<t;m++)
	{ scanf("%s",&a);
	  scanf("%s",&b);
 
	sort(a,a+strlen(a));
	sort(b,b+strlen(b));
	int i=0,j=0,count=0;
//cout<<a<<'\n';
//cout<<b<<'\n';

while(a[i]!=NULL && b[j]!=NULL)
	{ 
		if(a[i]==b[j])
		{ 
			count++;
		    i++;
		    j++;
		    //cout<<count<<'\n';
	    }
	  else if(a[i]<b[j])
	  {i++;}
	  else
	  {j++;}
	  }
	  cout<<count<<'\n';
	 }
	return 0;
}
