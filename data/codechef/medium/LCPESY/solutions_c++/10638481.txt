#include<stdio.h>
#include<bits/stdc++.h>
#include<string>
using namespace std;
int main()
{
    int t;

    scanf("%d",&t);
    while(t--)
    {
        int arr[123];
        int brr[123];
       for(int i=1;i<123;i++)
     {
        arr[i]=0;
       brr[i]=0;
    }
       // cout<<arr[100]<<endl;
        string a,b;
        cin>>a>>b;
        int c=0;
        for(int i=0;i<a.size();i++)
            arr[(int)a[i]]++;
            for(int i=0;i<b.size();i++)
                brr[(int)b[i]]++;
        for(int i=48;i<=57;i++)
        {
            if(arr[i]<=brr[i]&&arr[i]!=0)
            c=c+arr[i];
            if(brr[i]<arr[i]&&brr[i]!=0)
                c=c+brr[i];
        }
         for(int i=65;i<=122;i++)
        {
            if(arr[i]<=brr[i]&&arr[i]!=0)
            c=c+arr[i];
            if(brr[i]<arr[i]&&brr[i]!=0)
                c=c+brr[i];
        }
        printf("%d\n",c);
    }
    return 0;
}
