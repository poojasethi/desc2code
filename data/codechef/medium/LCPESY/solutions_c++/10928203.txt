#include<iostream>
#include<string.h>
using namespace std;
int main(){
	int t,i,x;
	cin>>t;
	string s1,s2;
	int a[62],b[62];
	while(t--){
		memset(a, 0,sizeof(a));
		memset(b, 0,sizeof(b));
		cin>>s1;
		for(i=0;i<s1.length();i++){
			if(s1[i]>=65&&s1[i]<=90)
				++a[(int)s1[i]-65];
			else if(s1[i]>=97&&s1[i]<=122)
			++a[(int)s1[i]-97+26];
			else
			++a[(int)s1[i]-48+52];
		}
		cin>>s2;
			for(i=0;i<s2.length();i++){
			if(s2[i]>=65&&s2[i]<=90)
				++b[(int)s2[i]-65];
			else if(s2[i]>=97&&s2[i]<=122)
			++b[(int)s2[i]-97+26];
			else
			++b[(int)s2[i]-48+52];
		}
		for(i=0,x=0;i<62;i++){
			if(a[i]>0&&b[i]>0){
				if(a[i]>b[i])
				x=x+b[i];
				else if(a[i]==b[i])
				x=x+a[i];
				else
				x=x+a[i];
			}
		}
		cout<<x<<endl;
	}
	return 0;
}
