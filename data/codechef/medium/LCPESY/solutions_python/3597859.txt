import sys
import string

for x in xrange(int(sys.stdin.readline())):
    s1 = sys.stdin.readline().strip()
    s2 = sys.stdin.readline().strip()
    c = 0
    l = []
    for i in s1:
        if i not in l:
            l.append(i)
            c += min(s1.count(i),s2.count(i))
    for i in s2:
        if i not in l:
            l.append(i)
            c += min(s1.count(i),s2.count(i))
    print c
