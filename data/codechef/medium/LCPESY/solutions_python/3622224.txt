def common(a,b):
	c=0
	da={}
	db={}
	for i in a:
		if i in da:
			da[i]+=1
		else:
			da[i]=1	
		
	for j in b:
		if j in db:
			db[j]+=1
		else:
			db[j]=1
			
	for k in da:
		if k in db:
			c=c+min(da[k],db[k])
			
	return c
			
			
for i in range(input()):
	a=raw_input()
	b=raw_input()
	print common(a,b)
