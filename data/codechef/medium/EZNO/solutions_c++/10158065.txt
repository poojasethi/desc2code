bool mod(ll base,string n){
	ll l = n.length();
	ll w = 0;
	for(ll i=0;i<l;i++){
		char b = n.at(i);
		ll g = m[b];
		w *= base;
		w += g;
		w %= l;

	}
	if(w==0)
		return true;
	return false;

}

ll recurse(ll base, string current,char ch[]){

	string next;
	ll u = 1;
	bool f = false;
	for(ll i=0;i<base;i++){

		next = current + ch[i];

		if(mod(base,next) && next.length()<42){
		f = true;
		//cout << "next"<<next<<endl;
			u += recurse(base,next,ch);
		}
	}

	dp[base][current.length()] += 1;
	return u;
}
