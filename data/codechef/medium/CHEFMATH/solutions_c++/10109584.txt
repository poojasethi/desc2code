#include<stdio.h>
#include<iostream>
#define mod 1000000007

using namespace std;

long long arr[50];
int k;

long long fun(long long x, long long k, long long n) {
    if (k==0) {
        if (x==0) {
            return 1;
        } else {
            return 0;
        }
    }
    if (x > k*arr[n]) {
        return 0;
    } else {
        long long ans = 0;
        if (n>0) {
            ans = fun(x,k,n-1);
        }
        if (x >= arr[n]) {
            ans = (ans + fun(x-arr[n],k-1,n))%mod;
        }
        return ans;
    }
}

int main() {
    int q,x,k;
    scanf("%d",&q);
    arr[0] = 1;
    arr[1] = 2;
    for(int i=2; i<43;i++) {
        arr[i] = arr[i-1] + arr[i-2];
    }
    while(q--) {
        scanf("%d%d",&x,&k);
        long long ans = fun(x, k, 42);
        printf("%lld\n",ans);
    }
    return 0;
}
