#include <bits/stdc++.h>
using namespace std;

#define mod 1000000007
#define mp make_pair
#define ll int

const int N = 100005;

int f[100];
int x,k;

map < pair < pair <int,int> , int >, int > mymap;

int go(int rem,int cur,int cnt){
   if(cnt == 0 && rem == 0) return 1;
   if(cnt == 0 || rem == 0) return 0;
   if(1LL * f[cur] * 1LL * cnt < 1LL * rem) return 0;
   if(cur == 0) return 0;
   if(mymap.find(mp(mp(rem,cnt),cur)) != mymap.end()) return mymap[mp(mp(rem,cnt),cur)];
   int ret = 0;
   if(rem >= f[cur]) ret += go(rem - f[cur], cur, cnt - 1) % mod;
   ret = (ret + go(rem ,cur - 1, cnt) % mod ) % mod;
   return mymap[mp(mp(rem,cnt),cur)] = ret;
}

int main(){
    f[1] = 1; f[2] = 2;
    for(int i=3;i<=50;i++){
        f[i] = f[i-1] + f[i-2];
    }
    int test; scanf("%d",&test);
    while(test--){
        scanf("%d %d",&x,&k);
        mymap.clear();
        printf("%d\n",go(x,45,k) % mod);
    }
    return 0;
}
