#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define INF 0x3f3f3f3f

using namespace std;
ll f[55];
void pre()
{
    f[0]=1;
    f[1]=2;
    for(int i=2;i<=50;i++)
        f[i]=f[i-1]+f[i-2];
}

ll solve(int sum,int i,int k)
{
    if(k==0)
    {
        if(sum==0)
            return 1;
        else
            return 0;
    }
    if(i<0||sum<0)
        return 0;
    ll ways=0;
    ways+=solve(sum,i-1,k);
    ways%=MOD;
    if(f[i]<=sum&&k*f[i]>=sum)
        ways+=solve(sum-f[i],i,k-1);
    ways%=MOD;
    return ways;

}
int main()
{
    pre();
    int q;
    scanf("%d",&q);
    //memset(dp,-1,sizeof(dp));
    while(q--)
    {
        int x,k;
        scanf("%d%d",&x,&k);
        printf("%lld\n",solve(x,50,k));
    }
    return 0;
}
