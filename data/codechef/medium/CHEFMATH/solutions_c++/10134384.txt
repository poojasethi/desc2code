#include <bits/stdc++.h>
 
using namespace std;
#define lli long long int
#define M 1000000007
 
vector<lli> chefoncs;
 
lli getcnt(lli x, lli k, lli mini)
{
	if(mini<0) return 0;
	if(k==0)
	{
		if(x==0) return 1;
		else return 0;
	} 
	else if(x<=0) return 0;
	// printf("%lld\t",mini);
	lli count= getcnt(x,k,mini -1);
	

	if(x/k <= chefoncs[mini-1])
		count = (count + getcnt(x-chefoncs[mini-1] , k-1,mini))%M;
	return count;
}
 
int main()
{
	lli q;
	scanf("%lld",&q);
	lli num1 = 1;
	lli num2 = 2;
	chefoncs.push_back(num1);
	chefoncs.push_back(num2);
	while(num2<pow(10,9)+1)
	{
		lli tmp = num1;
		num1 = num2;
		num2 += tmp;
		chefoncs.push_back(num2);
	}
	// printf("%d\n",chefoncs.size());
 
	for(int i=0;i<q;i++)
	{
		lli x,k;
		scanf("%lld%lld",&x,&k);
		printf("%lld\n",getcnt(x,k,(long long int)(chefoncs.size()-1)));

		
 
	}
}
