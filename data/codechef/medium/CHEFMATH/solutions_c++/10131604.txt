#include<bits/stdc++.h>
#define ll long long int
using namespace std;

const int size = 44;
ll mod = 1e9+7;
ll fib[size];

void pre()
{
	fib[0] = 1;
	fib[1] = 2;
	//int i= 1;
	for(int i=2; i<size; i++)
	{
		
		fib[i] = fib[i-1] + fib[i-2];
	}	
}

ll solve(ll x, int k, int idx)
{
	if(x == 0 && k == 0)
		return 1;
	else if(k<=0 || x<=0 || idx<0)
		return 0;
	ll c =0;
		c += solve(x, k, idx-1) % mod;
		
		if(x >= fib[idx] && x<= k*fib[idx])
			c += solve(x - fib[idx], k-1, idx) % mod;	 
		c %= mod;
	return c;	
}

int main()
{
	pre();
	int q;
	scanf("%d",&q);
	while(q--)
	{
		ll x;
		int k;
		scanf("%lld%d",&x, &k);
		//
		//for(int i=0; i<50; i++)
		//	printf("fib[%d] = %lld\n", i, fib[i]);
		ll ans;
		ans = solve(x,k, 43);
		printf("%lld\n", ans % mod);
	}
	return 0;
}
