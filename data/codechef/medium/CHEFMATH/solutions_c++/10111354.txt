#include<bits/stdc++.h>
using namespace std ;
#define ll long long
#define mod 1000000007
ll fac[12] ,chefno[50],q,x,k ; ll ans ;
void solve(ll a,ll b,int ind)
{
    if(ind<0||a<0) return ; if(b*chefno[ind]<a) return ;
  	if(b==0){if(a==0) ans++ ; return ;}
    if(chefno[ind]<=a) solve(a-chefno[ind],b-1,ind) ;
    solve(a,b,ind-1) ;
}
int main()
{
	fac[0] = 1 ;
	for(int i = 1;i<=10;i++) fac[i] = fac[i-1]*i ;
    chefno[0]  = 1 ; chefno[1] = 2 ;
    for(int i =2;i<50&&chefno[i-1]<1000000000 ;i++) chefno[i] = chefno[i-1]+chefno[i-2] ;
    cin>>q ;
    while(q--)
    {
        ans = 0;
    	cin>>x>>k ;
    	solve(x,k,43) ;
    	cout<<ans<<endl ;
    }
}
