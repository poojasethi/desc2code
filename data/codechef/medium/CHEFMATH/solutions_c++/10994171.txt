#include <bits/stdc++.h>

using namespace std;

#define ll				long long int
#define pb				push_back
#define mp				make_pair
#define MOD				1000000007
#define F				first
#define S				second

ll pow_mod(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a);
		a = (a * a);
		b >>= 1;
	}
	return res;
}

vector<ll> fib;
map< pair<ll, ll> , ll > memo[43];

ll solve(ll x, ll k, ll lim){

	if(memo[lim].find(mp(x,k)) != memo[lim].end())
		return memo[lim][mp(x,k)];
	else if(lim == 0)
		return (k == x);
	else if(k == 0 || x > k*fib[lim])
		return 0;
	else{
		ll ans = 0;
		if(x - fib[lim] > 0){
			ans = (ans + solve(x - fib[lim], k-1, lim))%MOD;
		}
		ans = (ans + solve(x, k, lim-1))%MOD;
		
		memo[lim][mp(x,k)] = ans;
		return ans;
	}

}

int main(){

	ios_base::sync_with_stdio(0);


	fib.pb(1); fib.pb(2);
	memo[0][mp(fib[0], 1)] = 1;
	memo[1][mp(fib[1], 1)] = 1;

	int i = 2;
	ll num = fib[i-1] + fib[i-2];
	while(num <= 1e9){
		fib.pb(num);
		memo[i][mp(num, 1)] = 1;
		i++;
		num = fib[i-1] + fib[i-2];
	}

	ll q;
	cin >> q;

	while(q--){
		ll x,k;
		cin >> x >> k;
		solve(x,k,42);
		cout << memo[42][mp(x,k)] << "\n";
	}

	return 0;
}
