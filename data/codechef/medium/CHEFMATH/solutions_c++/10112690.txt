#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
#define pb push_back
#define mp make_pair

ll f[1005];
int sz=0;

void calc() {
    f[0] = 1; f[1] = 2; sz = 2;
    for(int i=2;;++i) {
	f[i] = f[i-1]+f[i-2];
	sz++;
	if(f[i]>1000000000) break;
    }
}

ll solve(ll x, int k, int i) {
    if(k==0) {
	return (x==0);
    }
    if(x<=0 || i==0) {
	return 0;
    }
    if(x>k*f[i-1]) return 0;
    ll ans = 0;
    ans += solve(x,k,i-1);
    ans += solve(x-f[i-1],k-1,i);
    return ans;
}

int main()
{
    calc();
    int q;
    si(q);
    while(q--) {
	ll x; sll(x);
	int k; si(k);
	if(k==1) {
	    bool flg = false;
	    for(int i=0;i<sz;++i) {
		if(f[i]==x) {
		    flg = true;
		}
	    }
	    if(flg) {
		cout<<1<<endl;
	    }
	    else {
		cout<<0<<endl;
	    }
	    continue;
	}
	ll ans = solve(x,k,sz);
	pll(ans); nl;
    }
    return 0;
}
