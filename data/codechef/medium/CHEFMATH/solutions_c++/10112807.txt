#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define f first
#define s second
#define F(i,a,b) for(ll i = (ll)(a); i < (ll)(b); i++)
#define RF(i,a,b) for(ll i = (ll)(a); i >= (ll)(b); i--)
#define inf LLONG_MAX
#define mod 1000000007
#define MAXN 100005
#define pb(x) push_back(x)

long long int fib[50],temp,i,t,a,b;

long long int ans(int j,long long int x,int k2)
{
    if(x<0)
     return 0;
    if(k2==0&&x==0)
         return 1;

    if(j>0&&x>0&&k2>0&& (x-k2*fib[j-1])<=0 )
         return ans(j-1,x,k2)+ans(j,x-fib[j-1],k2-1);

    return 0; ///if any strange condition
}

int main()
{
	ios_base::sync_with_stdio(false);cin.tie(0);

	fib[0]=1;
	fib[1]=2;

	for(i=2;;i++)
    {
        temp=fib[i-1]+fib[i-2];

        if(temp<=1000000000)
            fib[i]=fib[i-1]+fib[i-2];
        else
             break;
    }

    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        cout<<ans(i,a,b)<<endl;
    }


	return 0;
}
