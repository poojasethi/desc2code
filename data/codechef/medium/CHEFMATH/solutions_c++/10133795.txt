
#include<bits/stdc++.h>

typedef long long LL;

const int M = 1000000007;
LL a;
LL v[45];
LL cnt=0;

LL dp(LL n,LL k,LL s)
{LL i,A=0,j;
cnt++;
//cout<<n<<' '<<k<<' '<<s<<endl;

//cout<<n<<' '<<k<<endl;

if(n<0 || k<0 || s<0)
	return 0;


//cout<<n<<' '<<k<<endl;

if(s==0 && n==k)
	return 1;
else if(s==0)
		return 0;


//cout<<n<<' '<<k<<endl;
//cout<<n<<' '<<v[s]<<endl;

if(n>v[s]*k)
	return 0;	


//cout<<n<<' '<<k<<endl;

//cout<<"Yo"<<endl;

for(i=n,j=0;i>=0 && j<=k;i-=v[s],j++)
		{//cout<<"YP"<<endl;
		A=(A+dp(i,k-j,s-1))%M;
		}

return A;

}
