#include<bits/stdc++.h>
#define mod 1000000000
#define ll long long
 
using namespace std;
ll gcd(ll a,ll b) {if(b==0) return a; return gcd(b,a%b);}
 
ll power(ll b,ll exp,ll m)
 {ll ans=1;
  b%=m;
  while(exp)
   {if(exp&1)
     ans=(ans*b)%m;
    exp>>=1;
	b=(b*b)%m;
   }
  return ans;
 }
 ll a[]={1,10,100,1000,10000,100000,1000000,10000000,100000000,1000000000LL};
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
        long long n,k;
        cin>>n>>k;
        long double x=((double)n)*log10l(n);
        long double ans=pow(10,x-floor(x)+k-1);
        ll Kdigit=a[k];
		ll second=power(n,n,a[k]);
        cout<<(int)ans<<" ";
        while(Kdigit/10){
 
			Kdigit/=10;
			cout<<second/Kdigit;
			second%=Kdigit;
		}
		cout<<endl;
    }
return 0;
}
