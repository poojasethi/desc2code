inline int Calc(int x,int y,int mad)
{
int tmp=x,ans=1;
for (;y;y>>=1)
{
if (y&1) ans=((long long)ans*tmp)%mad;
tmp=((long long)tmp*tmp)%mad;
}
return ans;
}
