#include <cstdio>
#include <algorithm>
 
using namespace std;
 
int t, n, m, i, r1, r2, f[1000];
 
int l[1000];
 
pair<long long,long long> r[1000];
 
int main() {
	scanf("%d", &t);
	while(t--) {
		scanf("%d", &n);
		for (i=0; i<n; i++) 
			scanf("%d", &l[i]);
		
		sort(l, &l[n]);
	
		scanf("%d", &m);
		for (i=0; i<m; i++) {
			scanf("%d%d",&r1,&r2);
 
			f[i]=-1;
			
			r[i]=make_pair((long long)11*r2,(long long)11*r1);
		}
		sort(r, &r[m]);
		
		int re=0;
		for (i=0; i<n; i++) {
			long long p=l[i]*(long long)7;
			for (int rx=0; rx<m; rx++) if (f[rx]<0) {
				if (r[rx].second<=p && p<=r[rx].first) {
					f[rx]=2;
					re++;
					break;
				}
			}
		}
		printf("%d\n", re);
		
	}
}
