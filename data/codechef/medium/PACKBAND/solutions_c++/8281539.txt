#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<cstring>
using namespace std;
typedef long long ll;
int len[1001];
bool occu[1001];;
pair<int,int> radius[1001]; 
int main()
{
    int co=0,t,i,m,n,r1,r2;
	 scanf("%d",&t);
	 while(t--)
	 {
	 	co=0;
        scanf("%d", &n);
        for(i=0;i<n;i++)
		{
            scanf("%d",&len[i]);
            occu[i]=0;
        }
        scanf("%d",&m);
        for(i=0;i<m;i++)
		{
            scanf("%d%d",&r1,&r2);
            radius[i]=pair<int,int>(r2,r1);
        }
        
		sort(radius,radius+m);
        sort(len,len+n);
        
		for(int i=0;i<m;i++)
		{
            r1=radius[i].second;
            r2=radius[i].first;
            for(int j=0;j<n;j++)
            {     
		        if(!occu[j]&&11*r1<=7*len[j]&&7*len[j]<=11*r2)
				{
                    occu[j]=true;
                    co++;
                    break;
                }
            }
    }
	    printf("%d\n",co);
    }
}
