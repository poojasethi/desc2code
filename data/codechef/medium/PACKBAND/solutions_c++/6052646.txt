#include <cstdio>
#include <algorithm>
 
using namespace std;
 
int t, n, m, s;
long  r1, r2, l[1000];
pair<long ,long > r[1000];
 
int main() {
	scanf("%d", &t);
	while(t--) {
		scanf("%d", &n);
		for (int i=0; i<n; i++){ 
			scanf("%ld", &l[i]);
			l[i]*=7;
        }
		
		sort(l, &l[n]);
	
		scanf("%d", &m);
		for (int i=0; i<m; i++) {
			scanf("%ld%ld",&r1,&r2);
			r[i]=make_pair(11*r2,11*r1);
		}
		sort(r, &r[m]);
		
		s=0;
		for (int i=0; i<n; i++)
			for (int j=0; j<m; j++)
				if (r[j].first>=l[i] && l[i]>=r[j].second) {
					r[j].first=0;
					s++;
					break;
		}
		printf("%d\n", s);
	}
}
