#include<bits/stdc++.h>
#define ll long long
using namespace std;

bool chk(int a[],int div,int n)
{
	for(int i=0;i<n;i++)
	if(a[i]>div)
	return false;
	
	return true;
}

int main()
{	
	int r,c,i,j; ll a[102][102]; ll rmin[102],cmax[102],an;
	scanf("%d %d",&r,&c);
	
	for(i=0;i<r;i++)
	for(j=0;j<c;j++)
	{
		scanf("%lld",&a[i][j]);
		if(i==0)
		cmax[j]=a[i][j];
		if(j==0)
		rmin[i]=a[i][j];
		
		rmin[i]=min(rmin[i],a[i][j]);
		cmax[j]=max(cmax[j],a[i][j]);
	}
	
	an=-1;
	for(i=0;i<r;i++)
	for(j=0;j<c;j++)
	{
		if(a[i][j]==rmin[i] && a[i][j]==cmax[j])
		{
			an=a[i][j];
			break;
		}
	}
	if(an==-1)
	printf("GUESS\n");
	else
	printf("%lld\n",an);
	return 0;
}
