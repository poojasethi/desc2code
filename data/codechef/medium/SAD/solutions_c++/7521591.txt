#include<stdio.h>
# define size 100
long read()
{
    char c;
    long d=0;
    c=getchar();
    while(c>='0'&&c<='9')
    {
        d=(d<<3)+(d<<1)+c-'0';
        c=getchar();
    }
return d;
}
int main()
{
    int k,r,c,i,j,p;
    long s,val,ans;
    long m[size][size];
r=read();
c=read();
ans=-1;
for(i=0;i<r;++i)
for(j=0;j<c;++j)
m[i][j]=read();
 
for(i=0;i<r;++i)
{
    val=0;
    s=m[i][0];
 
        for(j=0;j<c;++j)
        if(m[i][j]<=s)
    {
        s=m[i][j];
        k=j;
        val=1;
    for(p=0;p<r;++p)
    if(m[p][k]>s)
    {val=0;break;}

    }

}
if(ans>0&&ans==s)
