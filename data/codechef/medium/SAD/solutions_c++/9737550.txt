#include<bits/stdc++.h>
using namespace std;
int main(void)
{
    int r, c;
    scanf("%d %d", &r, &c);
    long mat[111][111];
    long row[111], column[111];
    for(int i=0; i<r; i++)
        for(int j=0; j<c; j++)
            scanf("%ld", &mat[i][j]);
    for(int i=0; i<r; i++){
        int minim=0;
        for(int j=1; j<c; j++)
            if(mat[i][j]<mat[i][minim])
                minim=j;
        row[i]=mat[i][minim];
    }
    for(int j=0; j<c; j++){
        int maxim=0;
        for(int i=1; i<r; i++){
            if(mat[i][j]>mat[maxim][j])
                maxim=i;
        }
        column[j]=mat[maxim][j];
    }
    bool found=false;
    for(int i=0; i<r; i++){
        for(int j=0; j<c; j++)
            if(mat[i][j]==row[i] && mat[i][j]==column[j]){
                found=true;
                printf("%ld\n", mat[i][j]);
                break;
            }
        if(found)
            break;
    }
    if(!found)
        printf("GUESS\n");
    return 0;
}
