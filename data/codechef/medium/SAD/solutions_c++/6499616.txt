#include <bits/stdc++.h>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)
int main()
{
	int r,c;
	cin>>r>>c;
	int mat[101][101];
	int mx[101];
	int mn[101];
	for (int i = 0; i <r; ++i)
	{
		for (int j= 0; j<c; ++j)
		{
			cin>>mat[i][j];
		}
	}
	for (int i = 0; i <r ; ++i)
	{
		mn[i]=mat[i][0];
	}
	for (int i = 0; i <c; ++i)
	{
		mx[i]=mat[0][i];
	}
	for (int i = 0; i <r; ++i)
	{
		for (int j=0; j <c; ++j)
		{
			if(mn[i]>mat[i][j])mn[i]=mat[i][j];
			if(mx[j]<mat[i][j])mx[j]=mat[i][j];
		}
	}
	int flag=0;
	for (int i = 0; i <r; ++i)
	{
		for (int j = 0; j <c; ++j)
		{
			if((mat[i][j]==mn[i]) && (mat[i][j]==mx[j]))flag =mat[i][j];
		}
	}
	if(flag) printf("%d\n",flag); else puts("GUESS");


	return 0;
}
