#include <iostream>
#include <cstdio>
#include <bits/stdc++.h>
using namespace std;

int mat[101][101];

class problem
{
private:
    int n, m;
public:
    problem();
    void solve();
};

problem :: problem()
{
    scanf("%d %d",&n,&m);
}

void problem :: solve()
{
    int maxcol[m];
    memset(maxcol, 0, sizeof(maxcol));
    for(int i = 0; i<n; ++i)
    {
        for(int j = 0; j<m; ++j)
        {
            scanf("%d",&mat[i][j]);
            if(maxcol[j] < mat[i][j])
                maxcol[j] = mat[i][j];
        }
    }
    int min;
    for(int i = 0; i<n; ++i)
    {
        min = 100000000;
        for(int j = 0; j<m; ++j)
        {
            if(min > mat[i][j])
                min = mat[i][j];
        }
        int flag = 0;
        for(int k = 0; k<m; ++k) {
            if(min == maxcol[k]) {
                printf("%d\n",min);
                return;
            }
        }
    }
    printf("GUESS\n");
}

int main()
{
    problem p;
    p.solve();
    return 0;
}
