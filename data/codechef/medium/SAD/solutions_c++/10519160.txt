#include<bits/stdc++.h>
using namespace std;
int main()
{
//while(true)
    {
int row,col,j,i,flag=1;
cin>>row>>col;
int arr[row][col],minrow[row],maxcol[col];
for(i=0;i<row;++i)
{minrow[i]=1e9;
for(j=0;j<col;++j)
{
    cin>>arr[i][j];
    minrow[i]=min(arr[i][j],minrow[i]);

}
}

for(j=0;j<col;++j)
{maxcol[j]=-1;
for(i=0;i<row;++i)
{
    maxcol[j]=max(arr[i][j],maxcol[j]);
}
}
for(i=0;i<row;++i)
{
    for(j=0;j<col;++j)
    {
        if(maxcol[j]==minrow[i])
        {
            cout<<maxcol[j];
            flag=0;
            break;
        }
    }
    if(!flag)
        break;
}

if(flag)
    printf("GUESS\n");
}return 0;
}
