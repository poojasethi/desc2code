#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
	long int temp,m,n;
	int b_1,c_1;
	long int b[101],c1[101];
	long int a[101][101];
	int r;
	int c;
	for (int i = 0; i < 101; i++) {
		b[i] = 0;
		c1[i] = 0;
	}
	scanf("%ld%ld",&r,&c);
	for (int i = 0; i < r; i++) {
		for (int j = 0; j < c; j++) {
			scanf("%ld",&a[i][j]);
		}
	}
	for (int i = 0; i < r; i++) {
		m = a[i][0];
		for (int j = 0; j < c; j++) {
			m = min(m,a[i][j]);
		}
		b[i] = m;
	}
	for (int i = 0; i < c; i++) {
		m = a[0][i];
		for (int j = 0; j < r; j++) {
			m = max(m,a[j][i]);
		}
		c1[i] = m;
	}
	int flag = 0;
	for (int i = 0; i < r; i++) {
		for (int j = 0; j < c; j++) {
			if (b[i] == c1[j]) {
				printf("%ld\n",b[i]);
				flag = 1;
				break;
			}
		}
		if (flag == 1)
			break;
	}
	if (flag == 0)
		printf("GUESS\n");
}
