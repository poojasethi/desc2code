#include <iostream>
using namespace std;
typedef long long LL;

LL a[101][101]={0};
LL maxcol[101]={0};
LL minrow[101]={0};
int main() {
    LL r=0,c=0,i=0,j=0,minr=0,maxc=0,flag=0;
    cin>>r>>c;
    for(i=0;i<r;i++) {
        for(j=0;j<c;j++) {
            cin>>a[i][j];
        }
    }
    
    for(i=0;i<r;i++) {
        j = 0;
        minr = a[i][j];
        for(j=0;j<c;j++) {
            if(a[i][j]<minr) {
                minr = a[i][j];
            }
        }
        minrow[i] = minr;
    }
    
    for(j=0;j<c;j++) {
        i=0;
        maxc = a[i][j];
        for(i=0;i<r;i++) {
            if(a[i][j]>maxc) {
                maxc = a[i][j];
            }
        }
        maxcol[j] = maxc;
    }
    
    for(i=0;i<r;i++) {
        for(j=0;j<c;j++) {
            if(a[i][j]==minrow[i] && a[i][j]==maxcol[j]) {
                flag=1;
                cout<<a[i][j]<<endl;
                break;
            }
        }
        if(flag) { break; }
    }
    
    if(!flag) {
        cout<<"GUESS"<<endl;
    }
    return 0;
}
