#include<iostream>
#include<fstream>
#include<string.h>
#include<stdio.h>
#include<math.h>
#include<algorithm>
#include<string>
#include<vector>

using namespace std;

int A[100][100];
int row[100][100];
int col[100][100];


int max(int a,int b)
{
	return (a>b)?a:b;
}

int min(int a,int b)
{
	return (a<b)?a:b;
}

int main()
{
	ios_base::sync_with_stdio(false);
	int r,c,maxi,mini;
	cin>>r>>c;
	
	for(int i=0;i<r;i++)
		for(int j=0;j<c;j++)
			cin>>A[i][j];
	
	memset(row,0,sizeof(row));
	memset(col,0,sizeof(col));
	
	for(int i=0;i<r;i++)
	{
		mini=1000000000;
		for(int j=0;j<c;j++)
				mini=min(mini,A[i][j]);
		
		for(int j=0;j<c;j++)
			if(A[i][j]==mini)
				row[i][j]=1;
	}
	
	for(int j=0;j<c;j++)
	{
		maxi=-100;
		for(int i=0;i<r;i++)
			maxi=max(maxi,A[i][j]);
		
		for(int i=0;i<r;i++)
			if(A[i][j]==maxi)
				col[i][j]=1;
	}
	
	for(int i=0;i<r;i++)
		for(int j=0;j<c;j++)
			if(row[i][j] && col[i][j])
				{
					cout<<A[i][j]<<endl;
					return 0;
				}
	cout<<"GUESS"<<endl;
	return 0;
}
