#include <iostream>
 
using namespace std;
int r[105],c[105],a[105][105];
int main()
{
    int x,y,i,j;
    cin>>x>>y;
    for(i=0;i<x;i++)
    {
        for(j=0;j<y;j++)
        {
            cin>>a[i][j];
            if(r[i]==0)
            r[i]=a[i][j];
            else if(a[i][j]<r[i])
            r[i]=a[i][j];
            if(c[j]==0)
            c[j]=a[i][j];
            else if(a[i][j]>c[j])
            c[j]=a[i][j];
        }
    }
    for(i=0;i<x;i++)
    {
        for(j=0;j<y;j++)
        {
            if(r[i]==c[j]&&a[i][j]==r[i])
            {
                cout<<a[i][j];
                return 0;
            }
        }
    }
    cout<<"GUESS";
    return 0;
}
