#include<bits/stdc++.h>
using namespace std;

#define sd(x) scanf("%d",&x)
#define ss(x) scanf("%s",x)
#define lld long long int
#define slld(x) scanf("%lld",&x)

int main()
{
    int r,c,i,j,ans,mxi[105],a[105][105],mini[105],flag=0;
    sd(r);sd(c);
    for(i=0;i<r;i++)
    for(j=0;j<c;j++)
        sd(a[i][j]);
    fill(mini,mini+104,INT_MAX);
    fill(mxi,mxi+104,INT_MIN);
    for(i=0;i<r;i++)
    {
        for(j=0;j<c;j++)
        {
            if(a[i][j]<mini[i])
                mini[i]=a[i][j];
        }
    }
    for(i=0;i<c;i++)
    {
        for(j=0;j<r;j++)
        {
            if(a[j][i]>mxi[i])
                mxi[i]=a[j][i];
        }
    }
    sort(mini,mini+r);
    sort(mxi,mxi+c);
    /*for(i=0;i<r;i++)
    cout<<mini[i]<<" ";
    for(i=0;i<c;i++)
    cout<<mxi[i]<<" ";*/
    i=0;
    j=0;
    while(i<c && j<r)
    {
        if(mxi[i]==mini[j])
        {
            ans=mxi[i];
            flag=1;
            break;
        }
        else if(mxi[i]>mini[j])
        {
            j++;
        }
        else
        {
            i++;
        }
    }
    if(flag==1)
    printf("%d\n",ans);
    else
    printf("GUESS\n");
	return 0;
}
