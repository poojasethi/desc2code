#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long int r,c;
    cin>>r>>c;
    long long int arr[r][c],row[r],mini,maxi,col[c],i,j;
    for(i=0;i<r;i++)
    {
        mini=0;
        for(j=0;j<c;j++)
        {
            cin>>arr[i][j];
            if(j==0)
                mini=arr[i][j];
            if(arr[i][j]<mini)
                mini=arr[i][j];
        }
        row[i]=mini;
    }
    for(i=0;i<c;i++)
    {
        maxi=0;
        for(j=0;j<r;j++)
        {
            if(arr[j][i]>maxi)
                maxi=arr[j][i];
        }
        col[i]=maxi;
    }
    long long int flag=0;
    for(i=0;i<r;i++)
    {
        for(j=0;j<c;j++)
        {
            if((arr[i][j]==row[i])&&(arr[i][j]==col[j]))
            {
                cout<<arr[i][j]<<endl;
                flag=1;
                break;
            }
        }
        if(flag==1)
            break;
    }
    if(flag==0)
        cout<<"GUESS"<<endl;
    return 0;
}
