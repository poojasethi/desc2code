#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
using namespace std;
int main()
{
    int i,r,c,flag,elem,j,maxi;
    scanf("%d%d",&r,&c);
    int arr[r][c];
    vector<pair<int,int> > s;
    vector<int> mini(r),maxim(c);
    for(i=0;i<r;i++)
    {
        for(j=0;j<c;j++)
        {
            scanf("%d",&(arr[i][j]));
        }
    }
    for(i=0;i<r;i++)
    {
        mini[i]=arr[i][0];
        for(j=1;j<c;j++)
        {
            if(mini[i]>arr[i][j])
                mini[i]=arr[i][j];
        }
    }
    for(i=0;i<r;i++)
    {
        for(j=0;j<c;j++)
        {
            if(mini[i]==arr[i][j])
            {
                s.push_back(make_pair(i,j));
            }
        }
    }
    for(i=0;i<c;i++)
    {
        maxi=arr[0][i];
        for(j=0;j<r;j++)
        {
            if(arr[j][i]>maxi)
                maxi=arr[j][i];
        }
        maxim[i]=maxi;
    }

    flag=0;
    for(i=0;i<s.size();i++)
    {
        elem=arr[s[i].first][s[i].second];
        if(elem==maxim[s[i].second])
        {
            flag=1;
            break;
        }
    }
    if(flag)
    {
        printf("%d\n",elem);
    }
    else
    {
        printf("GUESS\n");
    }
    return 0;
}
