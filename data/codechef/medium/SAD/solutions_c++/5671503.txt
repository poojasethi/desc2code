#include<bits/stdc++.h>
using namespace std;
int main()
{
    int r,c,i,j,f;
    long int a[100][100],ans,minm[105],maxm[105],minma,maxma;
    cin>>r>>c;
    for(i=0;i<r;i++)
    {
        for(j=0;j<c;j++)
        {
            cin>>a[i][j];
        }
    }
    maxma=-1;
    f=0;
    for(i=0;i<r;i++)
    {
       minma=100000005;
       for(j=0;j<c;j++)
       {
           if(a[i][j]<minma)
           {
               minma=a[i][j];
           }
       }
       minm[i]=minma;
    }
       for(i=0;i<c;i++)
       {
           maxma=-1;
          for(j=0;j<r;j++)
          {
              if(a[j][i]>maxma)
              {
                  maxma=a[j][i];
              }
          }
          maxm[i]=maxma;
       }
       for(i=0;i<r;i++)
       {
           for(j=0;j<c;j++)
           {
               if(a[i][j]==minm[i]&&a[i][j]==maxm[j])
               {
                   ans=minm[i];
                   f=1;
                   break;
               }
           }
           if(f==1)
           {
               break;
           }
       }
    if(f==0)
    {
        cout<<"GUESS"<<endl;
    }
    else
    {
        cout<<ans<<endl;
    }
    return 0;
}
