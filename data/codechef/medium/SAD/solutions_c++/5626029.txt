#include<iostream>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
        int r,c;
        cin>>r>>c;
        long long int a[r][c],ans,mini[r],maxi[c],flag=0;
        for(int i=0;i<r;i++)
        {
            cin>>a[i][0];
            mini[i]=a[i][0];
            for(int j=1;j<c;j++)
                {
                    cin>>a[i][j];
                    if(mini[i]>a[i][j])
                    mini[i]=a[i][j];
                }
        }
        for(int j=0;j<c;j++)
        {
            maxi[j]=a[0][j];
            for(int i=1;i<r;i++)
            {
                if(maxi[j]<a[i][j])
                    maxi[j]=a[i][j];
            }
        }
        for(int i=0;i<r;i++)
        {
            for(int j=0;j<c;j++)
            {
                if(a[i][j]==mini[i])
                {
                    if(a[i][j]==maxi[j])
                    {
                        ans=a[i][j];
                        flag=1;
                        break;
                    }
                }

            }
            if(flag==1)
                break;
        }
        if(flag==0)
            cout<<"GUESS"<<endl;
        else
            cout<<ans<<endl;

    return 0;
}
