#include<bits/stdc++.h>
using namespace std;
int main()
{
    int r,c,i,j,s,b;
    int val=-1;
    int small[101],big[101],a[101][101];
    cin>>r>>c;
    for(i=0;i<r;i++)
    {
        for(j=0;j<c;j++)
        {
            cin>>a[i][j];
        }
    }
    for(i=0;i<r;i++)
    {   s=a[i][0];
        for(j=1;j<c;j++)
        {
            if(s>a[i][j])
                s=a[i][j];
        }
        small[i]=s;
    }


    for(j=0;j<c;j++)
    {   b=a[0][j];
        for(i=0;i<r;i++)
        {
            if(b<a[i][j])
                b=a[i][j];
        }
        big[j]=b;
    }
    for(i=0;i<r;i++)
    {

        for(j=0;j<c;j++)
        {
            if(small[i]==big[j])
            {
                val=small[i];
                break;
            }
        }
    }
    if(val!=-1)
        cout<<val<<endl;
    else
        cout<<"GUESS"<<endl;
}
