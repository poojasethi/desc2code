#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main()
{

	int r , c, i, j, temp; 
	int row_min[101];
	int col_max[101];
	int matrix[110][110]; 
	int col_max_pos[101];
	int flag = 0, ans; 

	for(i = 1 ; i <= 100 ; i++)
	{
		row_min[i] = 100000001;
		col_max[i] = -1;
	} 
	cin >> r >> c;
	
	for(i = 1 ; i <= r ; i++)
	{
		for(j = 1 ; j <= c ; j++)
		{
			cin >> temp;
			matrix[i][j] = temp;

			if(temp <= row_min[i])
			{
				row_min[i] = temp;
				
			}
			if(temp >= col_max[j])
			{
				col_max[j] = temp;
			}
					
	
		}

	}

	for(i = 1 ; i <= r ; i++)
	{
		for(j = 1 ; j <= c ; j++)
		{
			if(matrix[i][j] == row_min[i])
			{
				if(matrix[i][j] == col_max[j])
				{
					flag = 1;
					ans = matrix[i][j];
					break;
				}
			}
		}						

	}

	if(flag == 1)
	{
		cout << ans << endl;
	}
	else
		cout << "GUESS" << endl; 
 


return 0;
}
