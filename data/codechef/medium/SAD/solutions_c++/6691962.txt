#include <cstdlib>
#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <limits>
#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

using namespace std;

int main(){
	int** mat;
	int r, c;
	int* min;
	int minvalue;
	int maxvalue;
	int* max;
	int mincount;
	int maxcount = 0;
	cin >> r >> c;
	int j , k;
	
	mat = (int**)malloc(sizeof(int*)*r);
	for(int i = 0; i < r; i++){
		mat[i] = (int*)malloc(sizeof(int)*c);
	}
	min = (int*)malloc(sizeof(int)*c);
	max = (int*)malloc(sizeof(int)*c);
	maxvalue = 0;
	for(j = 0; j < r; j++){
		minvalue = std::numeric_limits<int>::max();
		mincount = 0;
		for(k = 0; k < c; k++){
			min[k] = 0;
			cin >> mat[j][k];
			if(mat[j][k] < minvalue){
				minvalue = mat[j][k];
				mincount = 0;
				min[0] = k;
				mincount++;
			}
			else if(mat[j][k] == minvalue){
				min[mincount] = k;
				mincount++;
			}
		}
		if(minvalue > maxvalue){
			maxvalue = minvalue;
			maxcount = mincount;
			for(int p = 0; p < mincount; p++){
				max[p] = min[p];
			} 
		}
	}
	bool flag = false;
	for(int p = 0; p < maxcount; p++){
		for(int q = 0; q < r; q++){
			if(mat[q][max[p]] > maxvalue){
				flag = true;
				break;
			}
		}
		if(!flag){
			cout << maxvalue << endl;
			return 0;
		}
		else{
			flag = false;
			continue;		
		}
	}
	cout << "GUESS\n";
}
