#define lli long long int
#define fri(n) for(long long int i=0;i<n;i++)
#define frj(n) for(long long int j=0;j<n;j++)

lli inline inp()
    {
    lli n=0,a=1;
    char c=getchar_unlocked();
    while(c < '0' || c >'9') {if(c=='-'){a=-1;}c=getchar_unlocked();}
    while(c>='0' && c<='9')
    {
    n=(n<<3)+(n<<1)+c-'0';
    c=getchar_unlocked();
    }
    return a*n;
    }
    inline void print(lli a)
    {
    char num[20];
    lli i=0;
    do
    {
    num[i++]=a%10+48;
    a=a/10;
    }while(a!=0);
    i=i-1;
    while(i>=0)
    putchar_unlocked(num[i--]);
    putchar_unlocked('\n');
    }

int main(){
lli r,c,lx,mx,id[100],flag=0,fl=0,r1=0,fr=0;

frj(c)
a[i][j]=inp();

}
