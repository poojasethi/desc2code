#include<iostream>
using namespace std;
int main()
{
	long long int i,j,r,c,a[102][102],min_r[102],max_c[102],k,flag=0;
	cin>>r>>c;
	for(i=0;i<r;++i)
	for(j=0;j<c;++j)
	cin>>a[i][j];
	
	for(i=0;i<r;++i)
	{
		min_r[i]=a[i][0];
		for(j=0;j<c;++j)
		if(a[i][j]<min_r[i])
		min_r[i]=a[i][j];		
	}
	
	for(j=0;j<c;++j)
	{
		max_c[j]=a[0][j];
		for(i=0;i<r;++i)
		if(a[i][j]>max_c[j])
		max_c[j]=a[i][j];
	}
	
	for(i=0;i<r;++i)
	{
		for(j=0;j<c;++j)
		if(a[i][j]==min_r[i] && a[i][j]==max_c[j])
		{
			k=a[i][j]; flag=1;
		}			
	}
	if(flag)
	cout<<k<<"\n";
	else
	cout<<"GUESS\n";
	return 0;
}
