using namespace std;
#define MAX 104
#define mod 1000000007
typedef long long int ll;
typedef long double lf;

int cost[MAX][MAX];
int row_min[MAX], col_max[MAX];


int main(){
    int r,c;
    scanf("%d%d", &r, &c);
    int i, j;
    for( i=0; i<r; i++)
        row_min[i] = INT_MAX;
    for( i=0; i<c; i++)
        col_max[i] = INT_MIN;
    for( i=0; i<r; i++){
        for( j=0; j<c; j++){
            scanf("%d", &cost[i][j]);
            row_min[i] = min(row_min[i], cost[i][j]);
            col_max[j] = max(col_max[j], cost[i][j]);
        }
    }
    int number;
    bool found = false;
    for( i=0; i<r; i++){
        for( j=0; j<c; j++){
            if( row_min[i] == col_max[j] ){
                found = true;
                number = row_min[i];
            }
        }
    }
    if( found )
        printf("%d\n", number);
    else
        printf("GUESS\n");
    return 0;
}
