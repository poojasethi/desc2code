#include<iostream>
using namespace std;
int a[102][102],row[102],col[102];
int main()
{
	int r,c,i,j,ans;
	
	cin>>r>>c;
	for(i=1;i<=r;i++)
	{
		row[i]=1000000000;
	}
	
	for(j=1;j<=c;j++)
	{
		col[j]=-1;
	}
	for(i=1;i<=r;i++)
	{
		for(j=1;j<=c;j++)
		{
			cin>>a[i][j];
			
			row[i]=min(row[i],a[i][j]);
			col[j]=max(col[j],a[i][j]);
		}
	}
	
	ans=-1;
	
	for(i=1;i<=r;i++)
	{
		for(j=1;j<=c;j++)
		{
			if(a[i][j]==row[i]&&a[i][j]==col[j])
			{
				ans=a[i][j];
				break;
			}
		}
	}
	
	if(ans==-1)
	cout<<"GUESS\n";
	else cout<<ans<<endl;
	return 0;
}
