#include<iostream>
#include <list>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
 

int main()
{
	int r,c;
	scanf("%d %d",&r,&c);
	long long int a[r][c],i,j,k,max,min,row[r],col[c],count,flag;

		for(i=0;i<r;i++)
		{
			for(j=0;j<c;j++)
			{
				scanf("%lld",&a[i][j]);
			}
		}
		for(i=0;i<r;i++)
		{
			for(j=0;j<c;j++)
			{
				if(j==0)
				min=a[i][j];
				else
				{
					if(a[i][j]<min)
					min=a[i][j];		
				}		
			}
			row[i]=min;
		}
		for(j=0;j<c;j++)
		{
			for(i=0;i<r;i++)
			{
				if(i==0)
				max=a[i][j];
				else
				{
					if(a[i][j]>max)
					max=a[i][j];		
				}		
			}
			col[j]=max;
		}
		long long int f=0;
		max=-1;
		for(i=0;i<r;i++)
		{
			for(j=0;j<c;j++)
			{
				if(row[i]==col[j] && f==0)
				{
					max=row[i];
					f++;
				}
				else if(row[i]==col[j])
				{
					if(max<row[i])
					max=row[i];
				}
			}
		}
		if(max==-1)
		printf("GUESS\n");
		else
	printf("%lld\n",max);
	
    return 0;
}
