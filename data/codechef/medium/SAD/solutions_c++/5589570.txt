#include<bits/stdc++.h>
using namespace std;

#ifndef ONLINE_JUDGE
	#define gc getchar
#else
	#define gc getchar_unlocked
#endif

typedef long long ll;

void scanint(ll &x){
	register ll c=gc();
	x=0;
	for(;(c<48||c>57);c=gc());
	for(;(c>47&&c<58);c=gc()){
		x=(x<<1)+(x<<3)+(c-48);
	}
}

int main(){
    
    ll n,R,C,smallest,largest,i,j,flag=0;
    ll A[101][101]={0};
    ll small[101]={0},large[101]={0};
    
    scanint(R);
    scanint(C);
	
	for(i=0;i<R;i++){
		smallest=1000000000;
		for(j=0;j<C;j++){
			scanint(A[i][j]);
			if(A[i][j]<smallest){
				smallest=A[i][j];
			}
		}
		small[i]=smallest;
	}
	
	for(j=0;j<C;j++){
		largest=-1;
		for(i=0;i<R;i++){
			if(A[i][j]>largest){
				largest=A[i][j];
			}
		}
		large[j]=largest;
	}
	
	for(i=0;i<R;i++){
		for(j=0;j<C;j++){
			if(small[i]==A[i][j] && large[j]==A[i][j]){
				printf("%lld\n",A[i][j]);
				flag=1;
				break;
			}
		}
		if(flag==1){
			break;
		}
	}
	
	if(flag==0){
		printf("GUESS");	
	}
	
    return 0;
}
