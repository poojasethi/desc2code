#include<bits/stdc++.h>

using namespace std;

int main()
{
	int r,c;
	int f, i, j, p;
	scanf("%d%d",&r,&c);

	int a[r][c], mi[r], ma[c]; 
	

	for(i=0; i<r; ++i)
	{
			 int min=100000000;
		for(j=0; j<c; ++j)
		{
			scanf("%d",&a[i][j]);
				if(a[i][j]<=min)
				min=a[i][j];
			//	cout<<min;
		}
		mi[i]=min;
			//cout<<mi[i];
	}

	for(j=0; j<c; ++j)
	{
	
		int max=-1;
		for(i=0;i<r; ++i)
		{
			if(a[i][j]>=max)
			max=a[i][j];
		}
		ma[j]=max;
	//	cout<<ma[j];
	}

	for(i=0; i<r; ++i)
	{
		for(j=0; j<c; ++j)
		{
		//	cout<<mi[i]<<" "<<ma[j]<<endl;
			if(ma[j]==mi[i])
			{f=1;p=mi[i];
				cout<<p;
			exit(0);}
			
			else
			{f=0;
			}
			
		}
	}
	
	if(!f)
		cout<<"GUESS";
	

	return 0;
}
