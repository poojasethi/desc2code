#include <iostream>

using namespace std;
bool isPrime(unsigned long n) {
    if (n <= 3) {
        return n > 1;
    }
 
    if (n % 2 == 0 || n % 3 == 0) {
        return false;
    }
 
    for (unsigned short i = 5; i * i <= n; i += 6) {
        if (n % i == 0 || n % (i + 2) == 0) {
            return false;
        }
    }
 
    return true;
}
 
int main(){ 
	unsigned long t, p1,p2,p3,c=0;
	scanf("%ld",&t);	
	while(t!=0){
		c=0;
		for(p3=100;p3>1;p3--)
		{			
			if(isPrime(p3))
			{
				//printf("%ld\n",p3);
			for(p2=2;p2*p2+p3*p3*p3<t;p2=p2+2){
				p1=t-(p2*p2+p3*p3*p3);
				//printf("%ld %ld %ld\n",p1,p2,p3);
				if(isPrime(p1)&&isPrime(p2)){
					c=1;
					break;
				}
				if(p2==2)
					p2--;
			}
			}
			if(c==1)
				break;
		}
		if(p1+p2*p2+p3*p3*p3==t)
		printf("%ld %ld %ld\n",p1,p2,p3);
		else
			printf("%ld %ld %ld\n",0,0,0);

		scanf("%ld",&t);		
	}
	return 0;
}
