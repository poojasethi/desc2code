using namespace std;

int S[1000005],P2[1005],P3[105];
int main()
{
    int d,i,j,k,l,m,n,a,b,c,t,A[100005];
    for(i=0;i<1000005;i++)
    {
        S[i]=0;
        if(i<105)
            P3[i]=0;
        if(i<1005)
            P2[i]=0;
    }
    for(i=2;i<1000005;i++)
    {
        if(S[i]==0)
        {
            for(j=2;i*j<1000005;j++)
                S[i*j]=1;
        }
    }
    j=0;
    k=0;
    l=0;
    for(i=2;i<1005;i++)
    {
        if(S[i]==0)
        {
            if(i<=1000)
                P2[k++]=i*i;
            if(i<=100)
                P3[l++]=i*i*i;
        }
    }
    S[1]=1;
    scanf("%d",&n);
    while(n!=0)
    {
        d=0;
        a=0;
        b=0;
        c=0;
        for(i=0;i<l;i++)
        {
            if(P3[i]>=n)
                break;
            else
            {
                for(j=0;j<k;j++)
                {
                    if(P3[i]+P2[j]>=n)
                        break;
                    else if(S[n-(P3[i]+P2[j])]==0)
                    {
                        d=1;
                        a=n-(P3[i]+P2[j]);
                        b=sqrt(P2[j]);
                        c=cbrt(P3[i]);
                    }
                }
                if(d==1)
                    break;
            }
        }
        printf("%d %d %d\n",a,b,c);
        scanf("%d",&n);
    }
    return 0;
}
