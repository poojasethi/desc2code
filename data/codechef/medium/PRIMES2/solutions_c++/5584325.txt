#include<iostream>
#include<cmath>
using namespace std;
#define ll 1000002
int main()
{
	int a[ll];
	for(int i=0;i<ll;i++)
	a[i]=1;
	a[0]=-1;
	a[1]=-1;
	a[2]=1;
	int temp=2;
	while(temp!=-1)
	{
		int num=temp;
		int count=2;
		while(num*count<ll)
		{   
			a[num*count]=-1;
			count++;
		}
		num=temp;
		temp=-1;
		while(num<ll-1)
		{   num=num+1;
			if(a[num]==1)
			{temp=num;
			break;}
		}
	}
	while(true)
	{
		int num;
		cin>>num;
		int done=-1;
		if(num==0)
		break;
		for(int i=100;i>0;i--)
		{   
			if(a[i]==1)
			{   //cout<<i<<" ";
				int us=num-i*i*i;
				if(us>0)
				{
					for(int j=2;j*j<us;j++)
					{
						if(a[j]==1)
						{
							if(a[us-j*j]==1)
							{
								cout<<us-j*j<<" "<<j<<" "<<i<<endl;
								done=1;
								break;
							}
						}
					}
					if(done==1)
					break;
				}
			}
		}
		if(done==-1)
		cout<<"0 0 0"<<endl;
	}	
	}
