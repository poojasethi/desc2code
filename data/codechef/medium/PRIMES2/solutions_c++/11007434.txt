#include <bits/stdc++.h>

using namespace std;
int a[1000001];
int main()
{
	a[0] = 1;
	a[1] = 1;
	vector<int> p;
	for (int i = 2; i < 1000001; i++)
	{
		if (a[i] == 0) {
			p.push_back(i);
			for (int j = 2; j*i < 1000001; j++)
			{
				a[j*i] = 1;
			}
		}
	}
	int n;
	scanf("%d",&n);
	while (n) {
		int i,j, k;
		int f = 0;
		for (i = 0; i < 25; i++)
		{
			for (j = 0; j < 168; j++)
			{
				int num2 = p[j]*p[j];
				int num3 = p[i]*p[i]*p[i];
				int num1 = n-(num2+num3);
				k = lower_bound(p.begin(),p.end(),num1)-p.begin();
				if ((num2+num3+p[k]) == n)
				{
					f = 1;
					break;
				}
			}
			if (f)
				break;
		}
		if (f)
		{
			cout << p[k] << " " << p[j] << " " << p[i] << endl;
		}
		else
		{
			cout << "0 0 0\n";
		}
		scanf("%d",&n);
	}
}
