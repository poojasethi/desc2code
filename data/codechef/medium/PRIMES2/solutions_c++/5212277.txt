#include<stdio.h>
int prime(int n){
	if (n <= 3) {
		return n > 1;
	}
	else if (n % 2 == 0 || n % 3 == 0) {
		return 0;
	}
	else {
		for (int i = 5; i * i <= n; i += 6) {
			if (n % i == 0 || n % (i + 2) == 0) {
				return 0;
			}
		}
		return 1;
	}
}
int main(){
	int n, p3, p2, p1, temp;
	while (1){
		scanf("%d", &n);
		temp = 0;
		if (n == 0)
			break;
		for (p3 = 2; p3*p3*p3 <= n-6; p3++){
			if (prime(p3)==1){
				for (p2 = 2; (p3*p3*p3) + p2*p2 <= n-2 ; p2++){
					if (prime(p2)==1){
						p1 = n - (p3*p3*p3) - (p2*p2);
						if (prime(p1)==1){
							temp = 1;
							break;
						}
					}
				}
			}
			if (temp == 1)
				break;
		}
		if (temp == 0){
			printf("0 0 0\n");
		}
		else{
			printf("%d %d %d\n", p1, p2, p3);
		}
	}
	return 0;
}
