#include <iostream>
#include <cstdio>
#define gc getchar_unlocked
using namespace std;
int read_int() {
	char c = gc();
	int sign = 1;
  	while((c<'0' || c>'9') && c!='-') c = gc();
  		if(c=='-') {
    		sign = -1;
    		c = gc();
  		}
  	int ret = 0;
  	while(c>='0' && c<='9' ) {
    	ret = 10 * ret + c - 48;
    	c = gc();
  	}
 	 return ret*sign;
}
int main()
{
	int a,f,n,p1,p2,p3;
	int p[1000005];
	for (int i = 0; i <= 1000000; i++) {
		p[i] = 0;
	}
	p[0] = 1;
	p[1] = 1;
	for (int i = 2; i <= 1000000; i++) {
		int j = 2;
		while (i*j <= 1000000) {
			p[i*j] = 1;
			j++;
		}
	}
	f = 0;
	n = read_int();
	while (n != 0) {
		for (int i = 2; i <= 313; i++) {
			for (int j = 2; j <= 43; j++) {
				if (p[i] == 0 && p[j] == 0) {
					a = n - i*i - j*j*j;
					if (a > 0) {
						if (p[a] == 0) {
							printf("%d %d %d\n",a,i,j);
							f = 1;
							break;
						}
					}
				}
			}
			if (f == 1) {
				break;
			}
		}
		if (f == 0)
			printf("0 0 0\n");
		f = 0;
		n = read_int();
	}
}
