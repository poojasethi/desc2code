#include<bits/stdc++.h>
using namespace std;

#define MAX 1000006
typedef long long LL;
bool f_prime[MAX];
LL prime[80000],k=0,flag;
LL p1,p2,p3;

void sieve()
{
    f_prime[0]=f_prime[1]=true;
    for(int i=4;i<=MAX;i+=2) f_prime[i]=true;
    for(int i=3;i*i<=MAX;i+=2)
    {
        if(f_prime[i]==false)
        for(int j=i*i;j<=MAX;j+=i) f_prime[j]=true;
    }
    prime[k++]=2;
    for(int i=3;i<=MAX;i+=2)
    {
        if(f_prime[i]==false) prime[k++]=i;
    }
}

LL Check(LL num)
{
    LL a,b,c,d,f=0;
    if(num < 2) return 0;
   // printf("%lld \n",num);
    for(a=0;a<k;a++)
    {
        p2=prime[a]*prime[a];
        b=num-p2;
       // printf("%lld %lld\n",num,b);
        if(b < 2) break;
        if(f_prime[b] == false and b>1)
        {
            p1=b;
            p2=prime[a];
          //  printf("%lld %lld\n",num,b);
            return 1;
        }
    }
    return 0;
}

int main()
{
    LL num;
    sieve();
    while(scanf("%lld",&num)==1)
    {
        if(num == 0) break;
        flag=0;
        //printf("%d \n",num);
        for(LL i=0;i<k;i++)
        {
            p3=prime[i]*prime[i]*prime[i];
            flag=Check(num-p3);
            p3=prime[i];
            if(flag) break;
        }
        if(flag) printf("%lld %lld %lld\n",p1,p2,p3);
        else printf("0 0 0\n");
    }
    return 0;
}
