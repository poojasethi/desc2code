#include<bits/stdc++.h>
using namespace std;
vector<int> v(1000001);
int main()
{
    int i,j,n,flag,cal1,cal2,cal,val1,val2;
    set<int> s;
    v[0]=v[1]=1;
    for(i=2;i<=1000000;i++)
    {
        if(v[i]==0)
        {
            s.insert(i);
            for(j=2*i;j<=1000000;j=j+i)
            {
                v[j]=1;
            }
        }
    }
    scanf("%d",&n);
    set<int>::iterator it1,it2;
    while(n!=0)
    {
        flag=0;
        for(it1=s.begin();(*it1)<105;it1++)
        {
            val1=(*it1);
            cal1=val1*val1*val1;
            for(it2=s.begin();(*it2)<1005;it2++)
            {
                val2=(*it2);
                cal2=val2*val2;
                cal=n-cal1-cal2;
                if(cal>0)
                {
                    if(v[cal]==0)
                    {
                        flag=1;
                        break;
                    }
                }
            }
            if(flag)
                break;
        }
        if(flag)
        {
            printf("%d %d %d\n",cal,val2,val1);
        }
        else
            printf("0 0 0\n");
        scanf("%d",&n);
    }
    return 0;
}
