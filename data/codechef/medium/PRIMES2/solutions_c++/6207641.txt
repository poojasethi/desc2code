#include<iostream>
using namespace std;
 
int prime(int n){
	if(n<=1)
	return 0;
 
	if(n==2)
	return 1;
 
	for(int i=2;i*i<=n;i++)
	if(n%i==0)
	return 0;
 
	return 1;
}
 
int solve(int n){
	int p1,p2,p3,flag=-1;
	for(p3=100;p3>0;p3--){
		if(p3*p3*p3<n && prime(p3)){
			for(p2=2;p2*p2+p3*p3*p3<n;p2++){
				p1=n-p2*p2-p3*p3*p3;
				if(prime(p1) && prime(p2)){
					flag=1;
					break;
				}
			}
		}
		if(flag==1)
		break;
	}
	if(flag==1)
	cout<<p1<<" "<<p2<<" "<<p3<<endl;
 
	else
	cout<<"0"<<" "<<"0"<<" "<<"0"<<endl;
 
	return 0;
}
 
int main(){
	int n;
	while(1){
		cin>>n;
		if(n==0)
		break;
 
		solve(n);
	}
	return 0;
}
