#include<iostream>
#include<stdio.h>
using namespace std;
int isp(int n)	{
	if ( n < 2)
		return 0;
	if ( n == 2 || n == 3 )
		return 1;	
	int i = 3;
	if ( n % 2 == 0)
		return 0;
	while ( i * i <= n)	{
		if ( n % i == 0)
			return 0;
		i += 2;
	}
	return 1;
}

int main(){
	int N,p1,p2,p3,pcube,i,j,found;
	cin>>N;
	while(N){
	{
		found=0;
		for(p3=97;p3>1;p3--)
		{
			if(((pcube=p3*p3*p3)<N) && isp(p3))
			{
				for(p2=2;(p2*p2+pcube)<N;p2++)
				{
					p1=N-pcube-p2*p2;
					if(isp(p2) && isp(p1))
					{
						found=1;
						break;
					}
				}
				if(found)
					break;
			}
		}
		if(!found)
			p1=p2=p3=0;
		printf("%d %d %d\n",p1,p2,p3);
		scanf("%d",&N);
}
	}
		
	
}
