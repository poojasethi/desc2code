int isprime(int N){
	if(N<2 || (!(N&1) && N!=2))
		return 0;
	for(int i=3; i*i<=N; i+=2){
		if(!(N%i))
			return 0;
	}
	return 1;
}

void yes(int n,int &i,int &j,int &k)
{
			for(i=100;i>1;--i)
			{
				if(i*i*i<n && isprime(i))
				{
					for(j=2;i*i*i+j*j<n;++j)
					{
						k=n-i*i*i-j*j;
						if(isprime(k)&&isprime(j))
						return;					
					}					
				}				
			}
			i=j=k=0;
}
