#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

bool isPrime(int a)
{
    if(a==2||a==3)return true;
    if(a%2==0 || a%3==0 || a==1)return false;

    int b = sqrt(a)+1;
    for(int i=2; i<b; i++)
    if(a%i==0)
    return false;

    return true;
}

void solve(int n, int & a, int & b, int & c)
{
    for(c=100; c>0; c--)
    if(c*c*c<n && isPrime(c))
    for(b=2; b*b+c*c*c<n; b++)
    if(isPrime(b))
    {
        a = n-c*c*c-b*b;
        if(isPrime(a))return;
    }
    a=b=c=0;
}

int main()
{
    int n,a,b,c;
    scanf("%d",&n);
    while(n!=0)
    {
        solve(n,a,b,c);
        printf("%d %d %d\n",a,b,c);
        scanf("%d",&n);
    }
}
