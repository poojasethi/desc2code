def gcd(a,b):
    if b==0:
        return a
    else:
        return gcd(b,a%b)

def prime(n):
    for i in range(2,int(n**0.5)+1):
        if n%i==0:
            return i
    return n


for t in range(int(raw_input())):
    n=int(raw_input())
    N=map(int,raw_input().split())
    g=0
    for i in range(n):
        g=gcd(g,N[i])
    if g==1:
        print -1
    else:
        print prime(g)
