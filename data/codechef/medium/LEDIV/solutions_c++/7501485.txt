#include<bits/stdc++.h>
using namespace std;
#define maxi 100001

bool comp[maxi];
vector<int>v;

void pre()
{
	for(int i=2;i*i<maxi;i++)
	{
		if(comp[i])
		{
			continue;
		}
		for(int j=i;i*j<maxi;j++)
		{
			comp[i*j]=true;
		}
	}
	for(int i=2;i<maxi;i++)
	{
		v.push_back(i);
	}
}

int main()
{
	ios::sync_with_stdio(false);
	pre();
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		int g=a[0];
		for(int i=1;i<n;i++)
		{
			g=__gcd(g,a[i]);
		}
		if(g!=1)
		{
			if(!comp[g])
			{
				cout<<g<<"\n";
			}
			else
			{
				int i=0;
				for(i=0;i<v.size();i++)
				{
					if(g%v[i]==0)
					{
						break;
					}
				}
				cout<<v[i]<<"\n";
			}
		}
		else
		{
			cout<<"-1\n";
		}
	}
	return 0;
}
