#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <math.h>

int GCD(int a, int b)
{
  int t;
  while(b != 0)
  {
    t = a % b;
    a = b;
    b = t;
  }

  return a;
}


int cache[100005];

void CalcCache()
{
  memset(cache, 0, sizeof(cache)); // 0 means unprocessed

  for(int i = 2; i <= 100000; ++i)
  {
    if(cache[i] != 0)
      continue;

    for(int j = i; j <= 100000; j += i)
    {
      if(cache[j] == 0)
        cache[j] = i;
    }
  }
}


int main()
{
  CalcCache();

  int T;
  scanf("%d\n", &T);

  int A[100005];

  while(T--)
  {
    int N;
    scanf("%d\n", &N);

    for(int i = 0; i < N; ++i)
    {
      scanf("%d\n", A + i);
    }

    int G = A[0];
    int i = 1;
    for(; i < N; ++i)
    {
      G = GCD(A[i], G);
      if(G == 1)
        break;
    }

    if(G == 1)
    {
      printf("-1\n");
      continue;
    }

    printf("%d\n", cache[G]);
  }

  return 0;
}
