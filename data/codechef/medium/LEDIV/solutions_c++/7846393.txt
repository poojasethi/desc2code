#define bitcount __builtin_popcount
using namespace std;
typedef long long ll;
int main()
{
    int t; get(t);
    while(t--)
    {
        int n; get(n);
        int x,i,ans=0;
        for(i=0;i<n;i++)
        {
            get(x); ans = gcd(ans,x);
        }
        if(ans==1)
            print(-1);
        else
        {
            int i;
            for( i=2;i*i<=ans;i++)
                if(ans%i==0)
                    break;
            if(i*i>ans)
                print(ans);
            else
                print(i);
        }
    }
    return 0;
}
