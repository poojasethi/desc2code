#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll gcd(ll a,ll b) {return b==0?a:gcd(b,a%b);}
ll anschk(ll hcf)
{
	for(ll i=2;i*i<=hcf;i++)
    {
		if(hcf%i==0)
        return i;
	}
	return hcf;
}
int main()
{
	ll t;
	cin>>t;
	while(t--)
    {
		ll n;cin>>n;
		ll i,a[n];
		for(ll i=0;i<n;i++)
        {
			cin>>a[i];
		}
		ll hcf=0;
		for(ll i=0;i<n;i++)
		{
			hcf=gcd(hcf,a[i]);
		}
		if(hcf==1)
		{
			cout<<"-1\n";
		}
		else
		{
			cout<<anschk(hcf)<<endl;
		}
	}
	return 0;
}
