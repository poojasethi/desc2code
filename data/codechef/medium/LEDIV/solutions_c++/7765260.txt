#include<iostream>
using namespace std;
#include<cstdio>
#include<cmath>
int gcd(int a, int b) { return (b == 0 ? a : gcd(b, a % b)); }
int main(){
	int t, n, g, a;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		g = 0;
		for(int i=0;i<n;i++){
			scanf("%d",&a);			
			g = gcd(a,g);
		}

		if(g==1){
			printf("-1\n");
		}
		else{
			for(int i=2;i<=sqrt(g);i++ ){
				if((g%i)==0){
					g = i;
					break;
				}
			}
			printf("%d\n",g);
		}
	}
}
