#include<stdio.h>
#include<stdlib.h>
#include<math.h>
int a[100010];
int inline gcd(int a,int b)
{
	if(a%b==0)
	return b;
	else
	gcd(b,a%b);
	
}

int main()
{
	int i,sqr,k,t,n,x,flag;
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		scanf("%d",&n);
		scanf("%d",&a[0]);
		x=a[0];
		if(a[0]==1)
		flag=1;
		for(i=1;i<n;i++)
		{
			scanf("%d",&a[i]);
			if(a[i]==1 &&flag==0)
			{
				flag=1;
			}
			else if(flag==0)
			{
				x= (x>a[i])?gcd(x,a[i]):gcd(a[i],x);
				if(x==1)
				flag=1;
			}
		}
		if(flag==1)
		printf("-1\n");
		else
		{
			sqr= sqrt(x);
			for(i=2;i<=sqr;i++)
			{
				if(x%i==0)
				{
					x=i;
					break;
				}
			}
			printf("%d\n",x);
		}
		
	}
	return 0;
	
}
