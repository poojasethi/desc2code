#include<iostream>
using namespace std;
typedef long long int ll;
const int mx=100001;
ll gcd(ll a,ll b)
{
	if(b==0)
	return a;
	else return gcd(b,a%b);
}

int main()
{	
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		ll st;
		int a[n];
		for(int i=0;i<n;i++)
		cin>>a[i];
		ll cf=0;
		for(int i=0;i<n;i++)
		cf=gcd(cf,a[i]);
		
		if(cf==1)
		{cout<<"-1\n";
		}
		else
		{
			st=cf;
			for(int i=2;i*i<=cf;i++)
			if(cf%i==0)
			{
				st=i;
				break;
			}
			cout<<st<<"\n";	
		}
		
		//cout<<st<<"\n";
		
		
	}
}
