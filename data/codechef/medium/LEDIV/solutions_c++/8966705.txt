#include<bits/stdc++.h>
using namespace std;
int smallestPrime[1000001];
int minim(int a, int b)
{
	return (a<b)?a:b;
}
void precalc()
{
	for(int i=0; i<1000001; i++)
		smallestPrime[i]=i;
	for(int i=2; i<1000001; i++)
		if(smallestPrime[i]==i)
			for(int j=2*i; j<1000001; j+=i)
				smallestPrime[j] = minim(smallestPrime[j], i);
}
int gcdr(int a, int b)
{
	if(b==0)
		return a;
	return gcdr(b, a%b);
}
int main()
{
	int t;
	precalc();
	for(scanf("%d", &t); t--; )
	{
		int n;
		scanf("%d", &n);
		int arr[n];
		for(int i=0; i<n; i++)
			scanf("%d", &arr[i]);
		int res;
		if(n==1)
			res=arr[0];
		else{
			res = gcdr(arr[0], arr[1]);
			for(int i=2; i<n; i++)
				res = gcdr(res, arr[i]);
		}
		if(res==1)
			res=-1;
		else{
			res = smallestPrime[res];
		}
		printf("%d\n", res);
	}
	return 0;
}
