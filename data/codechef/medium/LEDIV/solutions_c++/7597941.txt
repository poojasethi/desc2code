#include<bits/stdc++.h>
using namespace std;

int gcd(int a,int b)
{
    if(a%b==0)
    return b;
    else
    return gcd(b,a%b);
}

int main()
{
    int n,t,i,a,g;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        scanf("%d",&a);
        g=a;
        for(i=1;i<n;i++)
        {
            scanf("%d",&a);
            g=gcd(g,a);
        }
        if(g==1)
        g=-1;
        else
        {
            for(i=2;i*i<=g;i++)
            {
                if(g%i==0)
                {
                    g=i;
                    break;
                }
            }
        }
        printf("%d\n",g);
    }
return 0;
}
