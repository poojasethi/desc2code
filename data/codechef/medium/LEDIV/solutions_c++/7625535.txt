#include<stdio.h>
using namespace std;
long int gcd(long int a,long int b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main()
{
    bool f;
    long int t,n,a,i,g;
    scanf("%ld",&t);
    while(t--)
    {
        scanf("%ld",&n);
        scanf("%ld",&g);
        for(i=1;i<n;++i)
        {
            scanf("%ld",&a);
            if(g!=1)
                g=gcd(g,a);
        }
        if(g==1)
            g=-1;
        if(g%2==0)
            g=2;
        for(i=3;i*i<=g;i+=2)
        {
            if(g%i==0)
            {
                g=i;
                break;
            }
        }
        printf("%ld\n",g);
    }
    return 0;
}
