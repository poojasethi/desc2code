#define PUTCHAR putchar_unlocked

inline int inp( )
{
int n=0;
int ch=getchar_unlocked();
while( ch >47 )
n = (n<<3)+(n<<1) + ch-'0', ch=getchar_unlocked();
return n;
}
void sieve()
{
	long int i,j;
    arr[0] = 0;
    for(i=1;i<100010;i++)
    arr[i] = i;
    int r=(int)pow(100010,0.5);
    for(i=2;i<=r+1;i++)
    {
        if(arr[i]==i)
        for(j = 2*i;j<=100010;j=j+i)
        {
            arr[j]=MIN(i,arr[j]);
        }
    }
}
int gcd(int a,int b)
{
int c;
if(a==b||b==0) return a;
c=a>b?a:b;
b=a<b?a:b;
a=c;
c=gcd(b,a%b);
return c;
}
