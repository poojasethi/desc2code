#include<iostream>
using namespace std;
#include<stdio.h>
#include<cmath>
int hcf( int a , int b)
{
	if(b==0) return a;
	else return hcf(b,a%b);
}


int main()
{

	int t;
	scanf("%d",&t);
	while(t--)
	{
	 int n;
	 scanf("%d",&n);
	 int arr , i,ans=0;

	 for(i=0;i<n;i++)
	 {
	   scanf("%d",&arr);
	    ans = hcf(arr,ans);

	 }
   //  cout<< " ans is : " << ans << endl;
	 if(ans==1) cout<<"-1\n";
	 else
     { int j;

       for(j=2;j<=sqrt(ans);j++)
         if(ans%j==0) ans = j;

       cout<< ans << endl;
       }


     }
	}
