#include<iostream>
using namespace std;
typedef long long int lli;  
  
#include<algorithm>
#include<math.h>
 int main()
  {
  	 int t;
  	  cin>>t;
  	  lli maxi=1;
  	   while(t--)
  	    {
  	    	maxi=1;
  	    	  int n;
  	    	   
  	    	   cin>>n;
  	    	    cin>>maxi;
  	    	     for(int i=0;i<n-1;i++)
  	    	      {
  	    	      	lli val; cin>>val;
  	    	      	//if(maxi==1) maxi=val
  	    	      	   maxi=__gcd(maxi,val);
  	    	    
  	    	      }
  	    	        //cout<<maxi<<endl;
  	    	      
  	    	      for(int i=2;i<=sqrt(maxi);i++)
  	           {
  	       	  if(maxi%i==0)
  	       	   {
  	       	   	  maxi=i;
  	       	   	 break;
  	       	   }
  	       }
  	         if(maxi==1)
  	          {
  	          	 cout<<"-1"<<endl;
  	          	
  	          }
  	           else cout<<maxi<<endl;
  	    }
  	      
  	          return 0;
  }
