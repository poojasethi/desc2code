#include <iostream>
#include <math.h>
 
using namespace std;
 
int gcd(int n,int m)
{
  if (n%m ==0) 
  return m;  
  n = n%m;
  return gcd(m,n);
 
}
 
int main(){
  int t,n,i,j,ginv,linv,a[100010],gcd1;
  cin>>t;
  while(t--){
  	cin>>n;
  	for(i=0;i<n;++i)
  		cin>>a[i];
 
  	gcd1=a[0];
    for(i=1;i<n;++i)
    {
      gcd1=gcd(gcd1,a[i]);
 
    
    }
   if(gcd1==1){
   	cout<<"-1"<<endl;
   	continue;
   } 
   
   int x=(int)sqrt(gcd1);
   for(i=2;i<=x;++i){
   	if(gcd1%i==0){
   	    break;
   	}
 
   }
   if(i==x+1){
   	cout<<gcd1<<endl;
   	continue;
   }
   cout<<i<<endl;
 
  }
  return 0;
}
