#include <bits/stdc++.h>

int a[100001];
using namespace std;

int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int main()
{
    int t,n,i,ans;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>a[i];
        ans=a[0];
        for(i=1;i<n;i++)
        {
            if(ans==1)
                break;
            else
                ans=gcd(ans,a[i]);
        }

        if(ans==1)
            cout<<-1<<endl;
        else{
            for(i=2;i<=sqrt(ans);i++)
            {
                if(ans%i==0)
                    ans=i;
            }
            cout<<ans<<endl;
        }
    }
    return 0;
}
