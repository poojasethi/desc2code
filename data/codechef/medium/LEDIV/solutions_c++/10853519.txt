#include<bits/stdc++.h>
using namespace std;
long long findgcd(long long x, long long y)
{

    if(y==0)
        return x;
    else
        return findgcd(y,x%y);
}
long long prime(long long a)
{

    int k, flag=0;
    for(k=2;k<=sqrt(a);k++)

	{
		if(a%k==0){
		flag=1;return k;}
	}
	if(flag==0)
	return a;
}
int main()
{
long long n, t,i, final, result,j, gcd,A[100001];

cin>>t;
for(i=0;i<t;i++)
{
cin>>n;
for(j=0;j<n;j++)
cin>>A[j];
gcd=A[0];
for(j=1;j<n;j++)
gcd=findgcd(gcd,A[j]);
final=prime(gcd);
if(final<gcd)
result=final;
else
    result=gcd;
if(result>1)
    cout<<result<<endl;
else
    cout<<"-1"<<endl;
}
return 0;
}
