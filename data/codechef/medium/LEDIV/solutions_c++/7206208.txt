#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(a==0)
		return b;
	else
		return gcd(b%a,a);
}
 
int fun(int a)
{
	int i,p=pow(a,0.5);
	for(i=2;i<=p;i++)
	{
		if(a%i==0)
			return i;
	}
	return a;
}
 
int main()
{
	int i,j,k,m,n,t,a,c;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int arr[n];
		for(i=0;i<n;i++)
			cin>>arr[i];
		for(i=0;i<n-1;i++)
		{
			a=gcd(arr[i],arr[i+1]);
			arr[i+1]=a;
		}
		if(arr[n-1]==1)
			cout<<"-1"<<endl;
		else
			cout<<fun(arr[n-1])<<endl;
	}
}
