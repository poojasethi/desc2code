#include<cstdio>
#include<cmath>
using namespace std;
typedef long long int ll;
ll gcd(ll a,ll b){
    if(a<b){
        ll tmp=a;
        a=b;
        b=tmp;
    }
    while(b){
        ll tmp=a%b;
        a=b;
        b=tmp;
    }
    return a;
}
int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
    ll n;
    scanf("%lld",&n);
    ll arr[n];
    scanf("%lld",&arr[0]);
    ll gc=arr[0];
    for(ll i=1;i<n;i++){
        scanf("%lld",&arr[i]);
        gc=gcd(gc,arr[i]);
    }
    if(gc==1){
        printf("-1\n");
    }
    else
    {
        ll ans=gc;
        for(ll i=2;i*i<=gc;i++){
            if(gc%i==0){
                ans=i;
                break;
            }
        }
        printf("%lld\n",ans);
    }
    }
}
