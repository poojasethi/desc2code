#include<bits/stdc++.h>
using namespace std;

long long int gcd(long long int a,long long int b)
{
    long long int t;
    while(a)
    {
        t=a;
        a=b%a;
        b=t;
    }
    return b;
}

long long prime(long long a)
{

    int k, flag=0;
    for(k=2;k<=sqrt(a);k++)

	{
		if(a%k==0){
		flag=1;return k;}
	}
	if(flag==0)
	return a;
}

int main()
{
    int x;
    cin>>x;
    long long int final,result;
    while(x--)
    {
        int n;
        cin>>n;
        long long int arr[n],ans;
        for(int i=0;i<n;i++)
            cin>>arr[i];
        ans=arr[0];
        for(int i=1;i<n;i++)
        {
            ans=gcd(ans,arr[i]);
        }
        final=prime(ans);
        if(final<ans)
            result=final;
        else
            result=ans;
        if(result>1)
            cout<<result<<endl;
        else if(ans==1)
        {
            cout<<"-1\n";
        }
    }
}
