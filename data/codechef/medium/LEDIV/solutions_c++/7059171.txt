#include<bits/stdc++.h>
#define ll long long
using namespace std;

ll gcd(ll a,ll b)
{
	if(a>=b)
	{
		if(a%b==0)
		return b;
		return gcd(b,a%b);
	}
	else
	{
		if(b%a==0)
		return a;
		return gcd(a,b%a);
	}
}

ll prm(ll a)
{
	ll i;
	if(a%2==0)
	return 2;
	
	for(i=3;i<=(ll)sqrt(a);i+=2)
	if(a%i==0)
	return i;
	
	return a;
}

int main()
{	
	ll t,n,i,x,g;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>x;
			if(i==0)
			g=x;
			else
			g=gcd(g,x);
		}
		if(g==1)
		cout<<-1<<endl;
		else
		cout<<prm(g)<<endl;
	}
	return 0;
}
