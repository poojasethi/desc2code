#include<bits/stdc++.h>
using namespace std;
# define N 100000
long long int arr[N];
long long int gcd(long long int a,long long int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}
long long int f(long long int a)
{
    for(int i=2; i<=floor(sqrt(a)); i++)
    {
        if(a%i==0)
        {
            return i;
        }
    }
    return a;
}
int main()
{
    long long int t,n,g;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=1; i<=n; i++)
        {
            cin>>arr[i];
        }
        g=0;
        for(int i=1; i<=n; i++)
        {
            g=gcd(arr[i],g);
        }
        if(g==1)
        {
            cout<<"-1"<<endl;
        }
        else
        {
            long long int val=f(g);
            cout<<val<<endl;
        }
    }
    return 0;
}
