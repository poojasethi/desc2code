#include<iostream>
#include<math.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return (gcd(b,a%b));
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,c,d;
		cin>>n;
		int arr[n];
		cin>>c;
		arr[0]=c;
		for(i=1;i<n;i++)
		{
			cin>>arr[i];
			c=gcd(c,arr[i]);
		}
		if(c==1)
		{
			cout<<"-1"<<endl;
			continue;
		}
		d=sqrt(c);
		for(i=2;i<=d;i++)
		{
			if(c%i==0)
			{
				break;
			}
		}
		if(i==d+1)
		{
			cout<<c<<endl;
		}
		else
		{
			cout<<i<<endl;
		}
	}
	return 0;
}
