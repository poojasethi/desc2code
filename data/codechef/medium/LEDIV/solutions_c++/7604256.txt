#include <iostream>
#define read(x) scanf("%d",&x)
#include <cstdio>
using namespace std;

int gcd(int a, int b)
{
    if(a>=b)
    {
        if(a%b == 0)
            return b;
        else
            return gcd(a%b,b);
    }
    else
        return gcd(b,a);
}

int small_div(int n)
{
    for(int i = 2; i*i<=n; ++i)
    {
        if(n%i==0)
            return i;
    }
    return n;
}

int main()
{
    int t;
    read(t);
    while(t--)
    {
        int n;
        read(n);
        int x,y;
        read(x);
        for(int i = 1; i<n; ++i)
        {
            read(y);
            x = gcd(x,y);
        }
        if(x ==1)
            printf("-1\n");
        else
        {
            printf("%d\n",small_div(x));
        }
    }
    return 0;
}
