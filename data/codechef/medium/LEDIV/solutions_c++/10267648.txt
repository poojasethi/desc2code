#include <bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
    return (b==0)?a:gcd(b,a%b);
}
int main() {
int t,i,j,num,arr[100001];arr[0]=0;
    for(i=1;i<100001;i++)
    arr[i] = i;
    int r=(int)sqrt(100001);
    for(i=2;i<=r+1;i++)
    {
        if(arr[i]==i)
        for(j = 2*i;j<=100001;j+=i)
        {
            arr[j]=min(i,arr[j]);
        }
    }
cin>>t;
while(t--)
{
cin>>num;
int a,cal=0;
for(i=0;i<num;++i)
{
    cin>>a;
    cal=gcd(a,cal);
}
if(cal==1)
    cout<<"-1\n";
else
    cout<<arr[cal]<<endl;
}
return 0;
}
