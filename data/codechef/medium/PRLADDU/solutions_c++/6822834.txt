#include <stdio.h>
#include<string.h>
 
int main()
{
	int t,i,n;
	int a[1000000];
	long long int sum;
	scanf("%d",&t);
		
		while(t--)
		{
				sum=0,n=0;
				scanf("%d",&n);
			
				scanf("%d",&a[0]);
				for(i=1;i<n;i++)
				{
					
					scanf("%d",&a[i]);
					
					if(a[i-1]<0)
					{
						sum+=-a[i-1];
						a[i]=a[i]+a[i-1];
					}
					else 
					{
						sum+=a[i-1];
						a[i]=a[i]+a[i-1];
					}
				}
			
		printf("%lld\n",sum);
 
		}
		return 0;
}
