#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		ll a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		ll x=0,ans=0;
		for(int i=0;i<n;i++)
		{
			x+=a[i];
			ans+=abs(x);
		}
		cout<<ans<<"\n";
	}
	return 0;
}
