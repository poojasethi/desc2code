#include <bits/stdc++.h>

using namespace std;
long long int arr[100005];
int main()
{
	long long int t,i,cur,ans,n;
	scanf("%lld",&t);

	while(t--)
	{
		scanf("%lld",&n);
		for(i=0;i<n;i++)
			scanf("%lld",arr+i);

		ans = cur =0;
		for(i=0;i<n;i++)
		{
			cur+=arr[i];
			ans +=abs(cur);
		}
		printf("%lld\n",ans);
	}
	return 0;


}
