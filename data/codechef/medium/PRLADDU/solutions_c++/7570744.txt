#include<bits/stdc++.h>
using namespace std;
int main()
{  long long t,n,ans;
scanf("%lld",&t);
    while(t--)
    {     ans=0;
        scanf("%lld",&n);
        long long A[n];
        for(int i=0;i<n;i++)
            scanf("%lld",&A[i]);
        for(int i=0;i<n;i++)
        {
            if(A[i]>0) //villagers inside
            {
                for(int j=i+1;j<n;j++)
                    if(A[j]<0)
                    {   
                        if(-1*A[j]>=A[i])
                        {
                            ans=ans+(j-i)*A[i];
                            A[j]=A[j]+A[i];
                            A[i]=0;
                            break;
                        }
                        else
                        {   ans=ans+(j-i)*(-1*A[j]);
                            A[i]=A[i]+A[j];
                            A[j]=0;
                        }
                    }
            }
            else if(A[i]<0) //Dino inside
            {
                for(int j=i+1;j<n;j++)
                    {
                        if(A[j]>0)
                        {
                            if(-1*A[i]>A[j])
                            {  ans=ans+(j-i)*A[j];
                               A[i]=A[i]+A[j];
                               A[j]=0;
                            }
                            else
                            {
                                ans=ans+(j-i)*(-1*A[i]);
                                A[j]=A[j]+A[i];
                                A[i]=0;
                                break;
                            }
                            }
                    }
            }
           
        }
 
    	printf("%lld\n",ans);
    }
    return 0;
}
