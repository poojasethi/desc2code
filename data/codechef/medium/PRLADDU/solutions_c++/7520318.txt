# include<stdio.h>
# include<math.h>
int abs(int k)
{	
	if(k>0)
	return k;
	return -k;
}
main()
{
	int t,n,i,j,a[100000];
	long long int c=0;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		for(i=0;i<n;i++)
		{
			if(a[i]>0)
			{
				for(j=i+1;j<n && a[i]>0;j++)
				{
					if(a[j]<0 && abs(a[j])>=a[i])
					{	
						c+=(j-i)*a[i];
						a[j]+=a[i];
						a[i]=0;	
					}
					else if(a[j]<0 && abs(a[j])<a[i])
					{
						c+=(j-i)*(abs(a[j]));		
						a[i]=a[i]-abs(a[j]);
						a[j]=0;
					}
				}
			}
			else if(a[i]<0)
			{
				for(j=i+1;j<n && a[i]<0;j++)
				{
					if(a[j]>0 && (a[j]>=abs(a[i])))
					{	
						c+=(j-i)*abs(a[i]);
						a[j]+=a[i];
						a[i]=0;	
					}
					else if(a[j]>0 && a[j]<abs(a[i]))
					{
						c+=(j-i)*(a[j]);		
						a[i]=a[i]+a[j];
						a[j]=0;
					}
				}
			}
		}
		printf("%lld\n",c);
		c=0;
	}
return 0;
}
