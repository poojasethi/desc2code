#include <iostream>
#include<bits/stdc++.h>
#define ll long long int
// long long mistake :P
using namespace std;
ll arr[1000100];
ll abss(ll a)
{
    if(a<0)
        return (-a);
    else
        return a;
}
int main()
{int t,n;
cin>>t;
while(t--)
{ll finans=0ll;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%lld",arr+i);
    }
    for(int i=0;i<n;i++)
    {
        finans+=abss(arr[i]);
        arr[i+1]+=arr[i];
    }
    printf("%lld\n",finans);
}
    return 0;
}
