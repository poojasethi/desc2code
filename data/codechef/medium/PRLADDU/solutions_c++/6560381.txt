#include <bits/stdc++.h>
using namespace std;

int A[100000+3];
int n;
int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	
	long long ans;
	long long cur_sum = 0;
	while(t--) {
	    ans = 0;
	    cur_sum = 0;
	    scanf("%d",&n);
	    for(int i=0; i<n; i++) {
	        scanf("%d",&A[i]);
	    }
	    
	    for(int i=0;i<n;i++) {
	        cur_sum += A[i];
	        ans += abs(cur_sum);
	    }
	    printf("%lld\n",ans);
	}
	return 0;
}
