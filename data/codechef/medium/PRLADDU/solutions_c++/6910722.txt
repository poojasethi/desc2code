#include<iostream>
using namespace std;
int main()
{
   int T,n,i,j,k,totalLadu;
   long long int ans;
   cin>>T;
   while(T--)
   {
      cin>>n;
      for(ans=totalLadu=i=0;i<n;i++)
      {
         cin>>j;
         totalLadu+=j;
         if(totalLadu>0)
            ans+=totalLadu;
         else
            ans-=totalLadu;
      }
      cout<<ans<<endl;
   }
return 0;
}
