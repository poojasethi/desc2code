#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll d[1000000];
int main()
{
    ll t,n,i,j,prev,sum;
    cin>>t;
    while(t--)
    {
        sum=0;
        cin>>n;
        for(i=1;i<=n;i++)
        {
            cin>>d[i];
        }
        j=1;
        for(i=1;i<=n;i++)
        {
            if(d[i]>0)
            {
                for(;j<=n;j++)
                {
                    ll laddu=d[i];
                    if(d[j]<0)
                    {
                        if(laddu<abs(d[j]))
                        {
                            sum+=(laddu*abs(j-i));
                            d[i]=0;
                            d[j]+=laddu;
                            break;
                        }

                            sum+=abs(d[j])*abs(j-i);
                            d[i]-=abs(d[j]);

                    }
                }
            }
        }
        cout<<sum<<"\n";
    }
    return 0;
}
