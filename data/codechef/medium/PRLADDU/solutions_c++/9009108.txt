#include<iostream>
#include<vector>
#include<stdlib.h>
using namespace std;
int d[100050];
int main(){
 int t,n;
 long long ans,cur;
 cin>>t;
 while(t--){
  cin>>n;
  ans=0,cur=0;
  for(int i=0;i<n;i++){
  cin>>d[i];
  }
  for(int i=0;i<n;i++){    
  cur+=d[i];
  ans+=abs(cur);
 }
 cout<<ans<<endl;
 }
 return 0;
}
