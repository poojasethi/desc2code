#include<iostream>
#include<algorithm>

using namespace std;
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
    int n;
    cin>>n;
    int i;
    long long int a[n],ans=0;
    for(i=0;i<n;i++)
        cin>>a[i];
        ans=abs(a[0]);
   for(i=1;i<n;i++)
   {
       a[i]+=a[i-1];
       ans+=abs(a[i]);
   }
   cout<<ans<<endl;
   }
return 0;

}
