#include<bits/stdc++.h>
using namespace std;
#define pp pair<int,int>
#define make_pair(i,j) mp(i,j)
#define all(c) c.begin(),c.end()
#define ll long long
int main()
{
	ll i,t,n,pos,neg,ans,cal;
	vector<ll> v(100000);
	scanf("%lld",&t);
	while(t--)
	{
		ans=pos=neg=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
			scanf("%lld",&(v[i]));
		while(neg<n)
		{
			while((neg<n)&&(v[neg]>=0))
				neg++;
			while((pos<n)&&(v[pos])<=0)
				pos++;
			if((neg<n)&&(pos<n))
			{
				if((-v[neg])>=v[pos])
				{
					v[neg]=-(-(v[neg])-v[pos]);
					cal=neg-pos;
					if(cal<0)
						cal=-cal;
					ans+=(cal)*v[pos];
					v[pos]=0;
				}
				else
				{
					v[pos]=v[pos]+(v[neg]);
					cal=neg-pos;
					if(cal<0)
						cal=-cal;
					ans+=(cal)*(-v[neg]);
					v[neg]=0;
				}
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
