#include <iostream>
#include <stdio.h>
#include <string.h>
#include <cstdlib>

#define issign(c) 	((c == '+') || (c == '-'))
#define isneg(c)	(c == '-')

int getInt()
{
	char c;
	char sign = '+';
	int result=0;

	while (!isdigit(c=getchar_unlocked()) && !isneg(c));
	if (isneg(c)) {
		sign = c;
		while (!isdigit(c=getchar_unlocked()));
	}
	do
		result = result*10 + (c-'0');
	while (isdigit(c=getchar_unlocked()));
	return (sign == '+') ? result : -result;
}


int main()
{
	int t = getInt();
	while (t--) {
		int n = getInt();
		long long d = getInt();
		long long value = d;
		long long cost = 0;
		int i;
		for (i=1; i<n; i++) {
			d = getInt();
			cost += abs(value);
			value += d;
		}
		std::cout << cost << std::endl;
	}

	return 0;
}
