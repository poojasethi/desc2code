#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
    //freopen("in.txt","r",stdin);
    int TC;
    scanf("%d",&TC);
    while(TC--){
        int N;
        scanf("%d",&N);
        int a[N];
        for(int i=0;i<N;i++){
            scanf("%d",&a[i]);
        }
        int val=0;long long int res=0;
        for(int i=0;i<N;i++){
            val=val+a[i];
            res=res+abs(val);
        }
        printf("%lld\n",res);
    }
}
