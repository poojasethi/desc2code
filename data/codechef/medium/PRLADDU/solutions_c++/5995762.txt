#include <iostream>
#include <stdio.h>
#include <cstdlib>
 
using namespace std;
 
int main()
{
    long int t , n , p = 0 ,num[100001] , j = 0 , k = 0 , num1[100001][2] , num2[100001][2] , temp = 0;
    long long int sum = 0;
    scanf("%ld",&t);
    long int p1 , p2;
    while(t)
    {
        --t;
        temp = 0;
        p = 0;
        k = 0;
        j = 0;
        p1 = 0;
        p2 = 0;
        scanf("%ld",&n);
        for(long int i = 0 ; i < n ; ++i)
        {
            scanf("%ld",&num[i]);
        }
        for(long int i = 0 ;i <= n-1 ;++i)
        {
            if(num[i] >= 0)
            {
                num1[k][0] = i;
                num1[k++][1] = num[i];
            }
            else if(num[i] < 0)
            {
 
                num2[j][0] = i;
                num2[j++][1] = -num[i];
            }
        }
/*
        for(int p = 0 ; p < k ; ++p)
        {
            cout<<num1[p][0]<<" ";
        }
        cout<<endl;
        for(int p = 0 ; p < j ; ++p)
        {
            cout<<num2[p][0]<<" ";
        }
        cout<<endl;*/
        while(p < k && p2 < j)
        {
            if(num1[p][1] > (num2[p2][1]))
            {
                num1[p][1] = num1[p][1] -num2[p2][1];
                temp = num1[p][0] - num2[p2][0];
                if(temp <0){temp = -temp;}
                sum += (num2[p2][1]*temp);
                num2[p2][1] = 0;
                ++p2;
            }
            else if(num1[p][1] == (num2[p2][1]))
            {
                num1[p][1] = num1[p][1] -num2[p2][1];
                temp = num1[p][0] - num2[p2][0];
                if(temp <0){temp = -temp;}
                sum += (num2[p2][1]*temp);
                num2[p2][1] = 0;
                ++p2;
                ++p;
            }
            else
            {
                num2[p2][1] = num2[p2][1] -num1[p][1];
                temp = num1[p][0] - num2[p2][0];
                if(temp <0){temp = -temp;}
                sum += (num1[p][1]*temp);
                num1[p][1] = 0;
                ++p;
            }
        }
        printf("%lld\n",sum);
        sum = 0;
    }
    return 0;
}
