#include<cstdio>
using namespace std;
int main(){
int t;
scanf("%d",&t);
while(t--){
    int n;
    scanf("%d",&n);
    long long int ans=0,curr=0;
    for(int i =0;i<n;i++){
        if(curr>=0)ans = ans + curr;
        else ans-=curr;
        long long int temp;
        scanf("%lld",&temp);
        curr = curr +temp;
    }
 
    printf("%lld\n",ans);
}
return 0;
}
