#include<bits/stdc++.h>
using namespace std;
#define ll long long 
const ll MAX=1000000;
ll d[MAX];

int main()
{
ll t,n,i,j,ans,laddu;
cin>>t;

while(t--)
{
    ans=0;
    cin>>n;
    
    for(i=0;i<n;i++)
        cin>>d[i];
        
        j=0;
    
    for(i=0;i<n;i++)
    {
        if(d[i]>0)//Villager with laddus
        {
            
            for(;j<n;j++)
            {
                laddu=d[i];
                if(d[j]<0)//dinosaur
                {
                    if(laddu<abs(d[j]))
                    {
                        ans+=abs(i-j)*laddu;
                        d[j]+=laddu;
                        d[i]=0;
                        break;
                    }
                    ans+=abs(i-j)*abs(d[j]);
                    d[i]-=abs(d[j]);
                }
            }
        }
    }
    cout<<ans<<"\n";
}
return 0;
}
