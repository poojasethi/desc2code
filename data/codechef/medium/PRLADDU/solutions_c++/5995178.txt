#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
    int n;
	cin>>t;
	while(t--) {
	    long long  cur_sum = 0;
	    long long ans = 0;
	    cin>>n;
	    for(int i=0;i<n;i++) {
	        int a;
	        cin>>a;
	        cur_sum = cur_sum+a;
	        ans = ans + abs(cur_sum);
	    }
	    cout<<ans<<"\n";
	}
	return 0;
}
