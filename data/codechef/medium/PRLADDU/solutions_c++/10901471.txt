#include <bits/stdc++.h>
using namespace std;

typedef long long int ll;

int main()
{
    ios_base::sync_with_stdio(false);
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n;
    	cin>>n;
		vector<ll>d(n+1);
		ll grass=0;
    	for(ll i=0;i<n;i++)
    	{
    		cin>>d[i];
		}
		for(ll i=0;i<n;i++)
		{
			if(d[i]!=0)
			{
				grass += abs(d[i]);
				d[i+1] += d[i];
				d[i] = 0;
			}
		}
		cout<<grass<<endl;
	}
	return 0;
}
