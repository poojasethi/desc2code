#include<bits/stdc++.h>
using namespace std;
int main()
{
long long int i,j,k,t,v1,d1,n,sum;
vector< pair<long long int,long long int> > d,v;
scanf("%lld",&t);
while(t--)
    {
        scanf("%lld",&n);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&j);
            if(j<0)
            {
                d.push_back(make_pair(j,i));
            }
            else
            {
                v.push_back(make_pair(j,i));
            }
        }
        d1=d.size();
        v1=v.size();
        for(i=0,j=0,sum=0;i<d1;)
        {
            if(d[i].first+v[j].first<0)
            {
                sum+=v[j].first*abs(d[i].second-v[j].second);
                d[i].first+=v[j].first;
                j++;
            }
            else if(d[i].first+v[j].first>0)
            {
                sum+=abs(d[i].first)*abs(d[i].second-v[j].second);
                v[j].first+=d[i].first;
                i++;
            }
            else
            {
                sum+=v[j].first*abs(d[i].second-v[j].second);
                i++;
                j++;
            }
        }
        v.clear();
        d.clear();
        printf("%lld\n",sum);
    }
    return 0;
}
