#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t;
    cin>>t;
    while(t--)
    {
        long long n,i,cans=0,cppl=0,x;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>x;
            cans+=abs(cppl);
            cppl+=x;
        }
        cout<<cans<<endl;
    }
    return 0;
}
