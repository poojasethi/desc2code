#include <stdio.h>
#include <stdlib.h>
using namespace std;

int main() {
	// your code goes here
	int t;
    int n;
    int a;
    long long cur_sum,ans;
	scanf("%d",&t);
	while(t--) {
	    ans = 0;
	    cur_sum = 0;
	    scanf("%d",&n);
	    for(int i=0;i<n;i++) {
	        scanf("%d",&a);
	        cur_sum = cur_sum+a;
	        ans = ans + abs(cur_sum);
	    }
	    printf("%lld \n",ans);
	}
	return 0;
}
