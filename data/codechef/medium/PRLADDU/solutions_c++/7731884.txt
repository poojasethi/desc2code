#include<bits/stdc++.h>
typedef long long int ll;
using namespace std;
int main()
{
    ll t,n,i,j,nt,pt,x,mini;
    scanf("%lld",&t);
    while(t--)
    {
        cin>>n;
        ll d[n];
        for(i=0;i<n;i++)
        {
            scanf("%lld",&d[i]);
        }
        ll nt=-1,pt=-1;
        ll grass=0;
        for(i=0;i<n;i++)
        {
            if(d[i]==0)
                continue;
            else if(d[i]<0)
            {
                nt=i;
                pt=-1;
            }
            else if(d[i]>0)
            {
                nt=-1;
                pt=i;
            }
            if(nt==i)
            {
                for(j=i+1;j<n;j++)
                {
                    if(d[j]>0)
                    {
                        x=abs(d[i]);
                        mini=(d[j]>x)?x:d[j];
                        if(d[j]>=x)
                        {
                            d[i]=0;
                            d[j]=d[j]-x;
                        }
                        else
                        {
                            d[i]=d[i]+d[j];
                            d[j]=0;
                        }
                        grass+=mini*(j-i);
                    }
                    if(d[i]==0)
                        break;
                }
            }
            else if(pt==i)
            {
                for(j=i+1;j<n;j++)
                {
                    if(d[j]<0)
                    {
                        x=abs(d[j]);
                        mini=(d[i]>x)?x:d[i];
                        if(d[i]>x)
                        {
                            d[i]=d[i]+d[j];
                            d[j]=0;
                        }
                        else
                        {
                            d[j]=d[i]+d[j];
                            d[i]=0;
                        }
                        grass+=mini*(j-i);
                    }
                    if(d[i]==0)
                        break;
                }
            }
            pt=-1;nt=-1;
        }
        printf("%lld\n",grass);
    }
    return 0;
}
