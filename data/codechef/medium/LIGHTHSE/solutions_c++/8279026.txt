#include <iostream>
#include <map>
#include <stdio.h>
#include <algorithm>

using namespace std;

typedef long long int ll;

typedef struct
{
    ll x;
    ll y;
} Points;

bool operator<(const Points p1, const Points p2)
{
    if (p1.x < p2.x)
        return true;
    else if (p1.x == p2.x)
        return p1.y<p2.y;
    else
        return false;
}

int main()
{
    ll t, n, min_x, min_y, max_x, max_y;
#ifndef ONLINE_JUDGE
    freopen("i_p","r",stdin);
#endif
    cin >> t;
    map<Points, ll> locate;
    map<Points, ll>::iterator mp;
    while (t--)
    {
        cin >> n;
        Points p;
        locate.clear();
        cin >> p.x;
        cin >> p.y;
        locate.insert(make_pair<Points, ll>(p, 1));

        min_x = max_x = p.x;
        min_y = max_y = p.y;
        for (ll i=2; i<=n; i++)
        {
            cin >> p.x;
            cin >> p.y;
            min_x = min(min_x, p.x);
            min_y = min(min_y, p.y);
            max_x = max(max_x, p.x);
            max_y = max(max_y, p.y);
            locate.insert(make_pair<Points, ll>(p, i));
        }

        if (n==1)
        {
            printf("1\n");
            printf("1 NE\n");
            continue;
        }
        mp=locate.begin();
        /*while(pLoc!=loc.end())
        {
            cout << pLoc->second << " ," << pLoc->first.x << ", "<< pLoc->first.y << "\n";
            ++pLoc;
        }*/
        // search for a point at bottom right
        p.x = max_x;
        p.y = min_y;
        mp = locate.find(p);
        if (mp != locate.end())
        {
            // Found point at bottom right!
            printf("1\n");
            printf("%lld NW\n", mp->second);
            continue;
        }

        // search for a point at bottom left
        p.x = min_x;
        p.y = min_y;
        mp = locate.find(p);
        if (mp != locate.end())
        {
            // Found point at bottom left!
            printf("1\n");
            printf("%lld NE\n", mp->second);
            continue;
        }

        // search for a point at top right
        p.x = max_x;
        p.y = max_y;
        mp = locate.find(p);
        if (mp != locate.end())
        {
            // Found point at top right!
            printf("1\n");
            printf("%lld SW\n", mp->second);

            continue;
        }

        // search for a point at top left
        p.x = min_x;
        p.y = max_y;
        mp = locate.find(p);
        if (mp!= locate.end())
        {
            // Found point at top left
            printf("1\n");
            printf("%lld SE\n", mp->second);
            continue;
        }

        // Choose two leftmost points
        map<Points, ll>::iterator left1, left2;

        mp = locate.begin();
        left1 = mp;
        ++mp;
        left2 = mp;
        printf("2\n");
        if (left1->first.y > left2->first.y)
        {
            printf("%lld SE\n", left1->second);
            printf("%lld NE\n", left2->second);
        }
        else
        {
            printf("%lld NE\n", left1->second);
            printf("%lld SE\n", left2->second);
        }
    }
    return 0;
}
