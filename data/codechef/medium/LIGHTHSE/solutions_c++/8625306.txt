#include <iostream>
using namespace std;
 
int main() 
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,cnt=1;;
		cin>>n;
		int xmax,xmin,ymax,ymin;
		xmax=0;
		ymax=0;
		xmin=n+1;
		ymin=n+1;
		long long int a[100005][2];
		a[0][0]=-1000000001;a[0][1]=-1000000001;
		a[n+1][0]=1000000001;a[n+1][1]=1000000001;
		for(i=1;i<=n;++i)
		{
			cin>>a[i][0];
			cin>>a[i][1];
			
			if(a[xmax][0]<a[i][0])
			xmax=i;
			if(a[xmin][0]>a[i][0])
			xmin=i;
			if(a[ymax][1]<a[i][1])
			ymax=i;
			if(a[ymin][1]>a[i][1])
			ymin=i;
			//cout<<a[xmin][0]<<" ";
		}
		//cout<<xmax<<xmin<<ymax<<ymin;
		for(i=1;i<=n;++i)
		{
			if(a[i][0]==a[xmax][0]&&a[i][1]==a[ymax][1])
			{
				xmax=i;
				ymax=i;
			}
			else if(a[i][0]==a[xmax][0]&&a[i][1]==a[ymin][1])
			{
				xmax=i;
				ymin=i;
			}
			else if(a[i][0]==a[xmin][0]&&a[i][1]==a[ymax][1])
			{
				xmin=i;
				ymax=i;
			}
			else if(a[i][0]==a[xmin][0]&&a[i][1]==a[ymin][1])
			{
				xmin=i;
				ymin=i;
			}
		}
		
		if(xmax==ymax)
		printf("%d\n%d\tSW\n",cnt,xmax);
		else if(xmax==ymin)
		printf("%d\n%d\tNW\n",cnt,xmax);
		else if(xmin==ymax)
		printf("%d\n%d\tSE\n",cnt,xmin);
		else if(xmin==ymin)
		printf("%d\n%d\tNE\n",cnt,xmin);
		else
		{
			cnt=2;
			if(a[xmax][1]<=a[xmin][1])
			{
				printf("%d\n%d\tNW\n",cnt,xmax);
				printf("%d\tSE\n",xmin);
			}
			else
			{
				printf("%d\n%d\tNE\n",cnt,xmin);
				printf("%d\tSW\n",xmax);
			}
		}
	}
	// your code goes here
	return 0;
}
