#include <bits/stdc++.h>
using namespace std;
struct lighthouse
{
    int x,y;
};
int main()
{
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        struct lighthouse a[100001];
        int xmax=INT_MIN,pos1=0,pos2=0,p1=0,p2=0,xmin=INT_MAX,ymin=INT_MAX,ymax=INT_MIN;
        for(i=0;i<n;i++)
        {
            scanf("%d %d",&a[i].x,&a[i].y);
            if(a[i].x<=xmin)
            {
                xmin=a[i].x;
                p1=a[i].y;
                pos1=i+1;
            }
            if(a[i].x>=xmax)
            {
                xmax=a[i].x;
                p2=a[i].y;
                pos2=i+1;
            }
            if(a[i].y>=ymax)
                ymax=a[i].y;
            if(a[i].y<=ymin)
                ymin=a[i].y;
        }
        int f=0;
        for(i=0;i<n;i++)
        {
            if(a[i].x==xmax&&a[i].y==ymin)
            {
                f=1;
                printf("1\n%d NW\n",i+1);
                break;
            }
            if(a[i].x==xmax&&a[i].y==ymax)
            {
                f=1;
                printf("1\n%d SW\n",i+1);
                break;
            }
            if(a[i].x==xmin&&a[i].y==ymin)
            {
                f=1;
                printf("1\n%d NE\n",i+1);
                break;
            }
            if(a[i].x==xmin&&a[i].y==ymax)
            {
                f=1;
                printf("1\n%d SE\n",i+1);
                break;
            }
        }
        if(f!=1)
        {
            if(p1<=p2)
            {
                printf("2\n%d NE\n%d SW\n",pos1,pos2);
                //break;
            }
            else
            {
                printf("2\n%d SE\n%d NW\n",pos1,pos2);
                //break;
            }
        }
    }
    return 0;
}
