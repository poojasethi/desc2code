#include <bits/stdc++.h>

using namespace std;

struct coord
{
    int x,y;
};

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        coord arr[n],maxpoint,minpoint;
        maxpoint.x=INT_MIN;
        minpoint.x=INT_MAX;
        int pos=0,maxpos=0,minpos=0,xmax=INT_MIN,xmin=INT_MAX,ymax=INT_MIN,ymin=INT_MAX;
        for(int i=0;i<n;i++)
        {
            scanf("%d %d",&arr[i].x,&arr[i].y);
            if(arr[i].x>=xmax)
            {
                xmax=arr[i].x;
                maxpoint.x=arr[i].x;
                maxpoint.y=arr[i].y;
                maxpos=i+1;
            }
            if(arr[i].x<=xmin)
            {
                xmin=arr[i].x;
                minpoint.x=arr[i].x;
                minpoint.y=arr[i].y;
                minpos=i+1;
            }
            if(arr[i].y<=ymin)
            {
                ymin=arr[i].y;
            }
            if(arr[i].y>=ymax)
            {
                ymax=arr[i].y;
            }
        }
        //printf("%d %d",maxpos,minpos);
        int maxnorth=0,maxsouth=0,minnorth=0,minsouth=0,flag=0;
        for(int i=0;i<n;i++)
        {
            if(arr[i].x==xmax&&arr[i].y==ymax)
            {
                flag=1;
                pos=i+1;
                printf("1\n%d SW\n",pos);
                break;
            }
            if(arr[i].x==xmin&&arr[i].y==ymax)
            {
                flag=1;
                pos=i+1;
                printf("1\n%d SE\n",pos);
                break;
            }
            if(arr[i].x==xmin&&arr[i].y==ymin)
            {
                flag=1;
                pos=i+1;
                printf("1\n%d NE\n",pos);
                break;
            }
            if(arr[i].x==xmax&&arr[i].y==ymin)
            {
                flag=1;
                pos=i+1;
                printf("1\n%d NW\n",pos);
                break;
            }
        }
        if(flag==0)
        {
            if(maxpoint.y>=minpoint.y)
                printf("2\n%d SW\n%d NE\n",maxpos,minpos);
            else
                printf("2\n%d NW\n%d SE\n",maxpos,minpos);
        }
    }
    return 0;
}
