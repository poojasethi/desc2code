#include <iostream>
using namespace std;

int main() {
int n;
cin>>n;
int a[n];
int b[n][n];
for(int i=1;i<=n;i++)
{cin>>a[i];}
int j=1;
int k;
for(int i=1;i<=n;i++)
{k=1;
if(a[i]!=0)
{b[j][1]=i;
while(b[j][k]!=0)
{k++;
b[j][k]=a[b[j][k-1]];
a[b[j][k-1]]=0;
}
j++;
}
}
cout<<j-1<<endl;
for(int i=1;i<j;i++)
{int m=1;
while(b[i][m]!=0)
{
cout<<b[i][m]<<"\t";
m++;
}
cout<<endl;
}
return 0;
}
