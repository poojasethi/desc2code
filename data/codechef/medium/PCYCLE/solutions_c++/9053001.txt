#include<bits/stdc++.h>
#define L long long
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    bool v[n];
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        v[i]=false;
    }
    int p[n][n];int k=0;int pos[n];
    for(int i=0,j=0,l=0;;i++)
    {
        if(!v[j])
        {
            v[j]=true;
            p[k][l]=j+1;
            j=a[j]-1;
            l+=1;
        }
        else
        {
            p[k][l]=p[k][0];
            pos[k]=++l;
            k++;l=0;
            int z;
            for(z=0;z<n;z++)
            {
                if(!v[z])
                {
                    j=z;
                    break;
                }
            }
            if(z==n)
                break;
        }
    }
        cout<<k<<endl;
        for(int i=0;i<k;i++)
        {
            for(int j=0;j<pos[i];j++)
                cout<<p[i][j]<<" ";
            cout<<endl;
        }
    return 0;
}
