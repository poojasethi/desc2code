#include<stdio.h>

int getCtr(int a[],int n){
	int i;
	for(i=1;i<=n;i++)
		if(a[i]==0)
			return i;
	return -1;
}
int main(){
	int n,series[1001],present[1001]={0},i,count=0,j,counter=1;
	scanf("%d",&n);
	int output[n+1][n+1];
	for(i=1;i<=n;i++)
		scanf("%d",&series[i]);
	for(i=1;counter>0;i++){
		for(j=1;present[counter]==0;j++){
			output[i][j]=counter;
			present[counter]++;
			counter=series[counter];
		}
		output[i][j]=-1;
		count++;
		counter=getCtr(present,n);
	}
	printf("%d\n",count);
	for(i=1;i<=count;i++){
		for(j=1;output[i][j]>0;j++)
			printf("%d ",output[i][j]);
		printf("%d\n",output[i][1]);
	}
	return 0;
}
