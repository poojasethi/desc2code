#include<bits/stdc++.h>
using namespace std;
int n;
int indi(int a[1000]);
int main()
{
    int t,i,j,a,x=0,l;
    cin>>n;
    int ar[n];
    for(i=0;i<n;i++)
    cin>>ar[i];
    a=indi(ar);
    int f[n+1][n];
    for(i=0;i<n;i++)
    f[i][0]=0;
    while(a!=-1)
    {
        i=a,l=0,j=1;
        while(l==0)
        {
            f[x][j]=i+1;
            j++;
            f[x][0]++;
            t=ar[i]-1;
            ar[i]=0;
            i=t;
            if(i==a)
            l++;
        }
        f[x][j]=f[x][1];
        f[x][0]++;
        x++;
        a=indi(ar);
    }
    cout<<x<<endl;
    for(i=0;i<x;i++)
    {
        for(j=1;j<=f[i][0];j++)
        {
            cout<<f[i][j]<<" ";
        }
        cout<<endl;
    }
}
int indi(int a[])
{
    int i,l=0;
    for(i=0;i<n;i++)
    {
        if(a[i]!=0)
        {
            l++;
            break;
        }
    }
    if(l>0)
    return i;
    else return -1;
}
