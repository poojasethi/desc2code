#include <iostream>
#include <cmath>
using namespace std;
int main(){
	int n;
	cin>>n;
	int a[n];
	for(int j=0;j<n;j++){
		cin>>a[j];
	}
	int b[3*n];
	for(int j=0;j<3*n;j++){
		b[j]=0;
	}
	int i=0;
	int j=0;
	int temp;
	int c=0;
	while(i<n){
		if(a[i]==-1){
			i++;
			continue;
		}
		else{
			temp = a[i];
			a[i]=-1;
			b[j]=i+1;
			j++;
			while(a[temp-1]!=-1){
				int k=temp-1;
				temp = a[k];
				a[k]=-1;
				b[j]=k+1;
				j++;
			}
			b[j]=temp;
			j++;
			b[j]=-1;
			j++;
			c++;
			i=0;
		}
	}
	cout<<c<<endl;
	for(int m=0;m<j;m++){
		if(b[m]==-1){
			cout<<"\n";
		}
		else{
			cout<<b[m]<<" ";
		}
	}
	return 0;
}
