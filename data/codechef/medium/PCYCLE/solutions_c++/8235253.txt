#include<iostream>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	int a[n+1],vis[n+1],k,j,m,arr[3000],count=0;
	for(int i=1;i<=n;i++)
	scanf("%d",&a[i]);
	for(int i=1;i<=n;i++)
	vis[i]=0;
	m=0;
	for(int i=1;i<=n;i++)
	{
		if(vis[i]==0)
		{
			k=i;
			j=k;
m++;
arr[m]=k;
			vis[k]=1;
			do{
m++;
arr[m]=a[k];
				vis[a[k]]=1;
			k=a[k];
			}while(k!=j);
			m++;
   arr[m]=-1;
count++;
		}
	}
	printf("%d\n",count);
	for(int i=1;i<=m;i++)
	{
		if(arr[i]==-1)
		printf("\n");
		else
		printf("%d ",arr[i]);
	}
	return 0;
}
