#include<cstdio>
#include<vector>
using namespace std;
int main(){
    int N;
    scanf("%d",&N);
    int arr[N+1];
    for(int i=1;i<=N;i++){
        scanf("%d",&arr[i]);
    }
    vector<int> v[N];
    int size=0;
    for(int i=1;i<=N;i++){
        if(arr[i]>0){
            v[size].push_back(i);
            int cur=arr[i];
            arr[i]=-1;
            while(cur!=-1){
                v[size].push_back(cur);
                int next=arr[cur];
                arr[cur]=-1;
                cur=next;
            }
            size++;
        }
    }
    printf("%d\n",size);
    for(int i=0;i<size;i++){
        for(int j=0;j<v[i].size();j++){
            printf("%d ",v[i][j]);
        }
        printf("\n");
    }
}
