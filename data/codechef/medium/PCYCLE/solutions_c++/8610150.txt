#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
 
using namespace std;
 
int a[1005],n,visited[1005];
 
int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	int i=1,z=0;
	visited[i]=true;
	//printf("1 ");
	vector<int> ans;
	ans.push_back(1);	
	while (1) {
		i=a[i];
//		printf("%d ",i);
		ans.push_back(i);
		if (visited[i]) {
//			printf("\n");
			ans.push_back(-1);
			int temp=-1;
			for (int j=1;j<=n;j++) if (!visited[j]){ temp=j;break;}
			if (temp==-1)break;
			else {
				i=temp;
//				printf("%d ",i);
				ans.push_back(i);	
			}
		}
		visited[i]=true;
	}
	int sz=ans.size();
	int r=0;
	for (int i=0;i<sz;i++) if (ans[i]==-1)r++;
	printf("%d\n",r);
	for (int i=0;i<sz;i++) {
		if (ans[i]==-1)printf("\n");
		else printf("%d ",ans[i]);
	}
	return 0;
}
