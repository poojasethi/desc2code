#define ll         long long
#define S(x)       scanf("%d", &x)
#define Sl(x)      scanf("%lld", &x)
#define Sd(x)      scanf("%lf", &x)
#define P(x)       printf("%d\n", x)
#define Pl(x)      printf("%lld\n", x)
#define Pd(x)      printf("%lf\n", x)
#define Pblank()   printf(" ")
#define mem(x,y)   memset(x,y,sizeof(x))
#define F(x,y,z,i) for (x = y; x < z; x = x + i)
#define mod 1000000007

int main()
{
    ll int n;
    Sl(n);
    ll int b[n + 1];
    ll int a[n][n];
    ll int siz[n];
    ll int i, j, k, m = 0, c, temp, p = 0;
    for (i = 1; i <= n; i++){
        Sl(b[i]);
    }

    for (i = 1; i <=n; i++){
        c = 0;
        c=0;
		if (b[i] != -1) {
			a[m][c++] = i;
			k = b[i];
			while (k != i){
				a[m][c++] = k;
				temp = k;
				k = b[k];
				b[temp] = -1;
			}
			a[m++][c++] = i;
			b[i] = -1;
			siz[p++] = c;
		}
    }
    Pl(m);

    for (i = 0; i < m; i++){
        for (j = 0; j < siz[i]; j++){
            cout << a[i][j] << " ";
        }
        cout << endl;
    }

    return 0;
}
