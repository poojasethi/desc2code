#include<bits/stdc++.h>
#define ll long long int
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

int hash[1005],hashe[1005];

int main(){

	int n,c,i,j;
	int a[1005];

	scanf("%d",&n);

	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);

    c=0;

	for(i=1;i<=n;i++){

		if(hashe[i]!=-1){
			c++;
            j=a[i];
			while(i!=j){
				hashe[j]=-1;
				j=a[j];
			}
           hashe[i]=-1;
           
		}
	}

	cout<<c<<endl;

	for(i=1;i<=n;i++){

		if(hash[i]!=-1){
            cout<<i<<" ";

            j=a[i];
			while(i!=j){
				cout<<j<<" ";
				hash[j]=-1;
				j=a[j];
			}
           hash[i]=-1;
           cout<<i<<" ";
           cout<<endl;
		}
	}
	return 0;
}
