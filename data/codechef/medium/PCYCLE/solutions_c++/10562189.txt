#include<stdio.h>
#include<vector>
#define s(n) scanf("%d",&n);
#define p(n) printf("%d ",n);
#define P(n) printf("%d\n",n);
using namespace std;
int main()
{
    int n;
    s(n);
    int a[n];
    bool b[n];
    vector<int> v;
    for(int i=1;i<=n;i++)
        {
         b[i]=0;
        s(a[i])
        }
        int c=0;
        for(int i=1;i<=n;i++)
        {
            if(b[i]!=1)
            {
                c++;
               v.push_back(i);
                int x;
                b[i]=1;
                x=a[i];
               v.push_back(x);
                while(b[x]!=1)
                {
                   b[x]=1;
                x=a[x];
                v.push_back(x);
                }
                v.push_back(0);
            }
        }
        P(c)
        for(int i=0;i<v.size();i++)
        {
            if(v[i]!=0)
                p(v[i])
                else
                printf("\n");
        }
        return 0;
}
