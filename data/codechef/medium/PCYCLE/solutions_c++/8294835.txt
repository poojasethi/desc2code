	#include<iostream>
	using namespace std;

	 
	int main()
	{
		int n,arr[1010]={0};
		int flag[1010]={0};
		int count=0;	
		
		cin>>n;

		for(int i=1;i<=n;i++)
			cin>>arr[i];
		
		for(int i=1;i<=n;i++)
		{
			if(flag[i]==0)
			{
				flag[i]=1;
				count++;
			

				for(int j=arr[i];j!=i;j=arr[j])
					flag[j]=1;
			}
		}

		cout<<count<<endl;
		int flag2[1010]={0};
		
		for(int i=1;i<=n;i++)
		{
			if(flag2[i]==0)
			{
				flag2[i]=1;
				cout<<i<<" ";

				for(int j=arr[i];j!=i;j=arr[j])
				{
					flag2[j]=1;
					cout<<j<<" ";
				}
				cout<<i<<endl;
			}
		}
	
	}
