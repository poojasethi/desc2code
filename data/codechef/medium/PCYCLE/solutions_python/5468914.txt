n = input()
numbers = []
visited = {}
text = raw_input()
numbers.append(0)
for num in text.split():
	numbers.append(int(num))
index = 1
count = 0
res = ""
while index <= n:
	startingIndex = index
	while index <= n:
		visited[index] = 1
		if(numbers[index] == startingIndex):
			res = res + str(index)+" "+str(startingIndex)+"\n"
			count+=1
			break
		else:
			res = res + str(index)+" "
			index = numbers[index]
	index = startingIndex+1
	while index <= n:
		if(index in visited):
			index+=1
		else:
			break
print count
