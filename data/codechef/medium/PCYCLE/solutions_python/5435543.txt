raw_input()
perm = map(int, raw_input().split())
cyc = []
cycl = []
seen = set()
point = 1
from itertools import dropwhile
while True:
    cyc.append(str(point))
    if point in seen:
        cycl.append(cyc)
        cyc = []
        try:
            point = next(dropwhile(seen.__contains__, xrange(1, len(perm) + 1)))
        except StopIteration:
            break
        continue
    seen.add(point)
    point = perm[point - 1]
print len(cycl)
for cyc in cycl:
    print " ".join(cyc)
