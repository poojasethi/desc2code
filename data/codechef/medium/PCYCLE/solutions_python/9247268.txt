n = int(raw_input().strip())
a = map(int, raw_input().strip().split())

b, c = 0, 0
res = ''

while True:
    e = a[b] - 1
    a[b] = 0
    res += str(b+1)
    while e != b:
        res += ' ' + str(e+1)
        t = a[e]
        a[e] = 0
        e = t - 1

    res += ' ' + str(e+1) + '\n'
    c += 1

    for i, ai in enumerate(a):
        if ai != 0:
            b = i
            break
    else:
        break

print c
print res
