n = input()
array = map(int, raw_input().split())
visited = [False]*n
ans = []
for i in range(n):
    current = i
    cycle = []
    while not visited[current]:
        cycle.append(current+1)
        visited[current] = True
        current = array[current]-1
    if len(cycle):
        cycle.append(current+1)
        ans.append(cycle)
print len(ans)
for i in range(len(ans)):
    print ' '.join(str(j) for j in ans[i])
