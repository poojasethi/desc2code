n = int(raw_input())
ar = [0] + [int(c) for c in raw_input().split()]
cycles = []
for i in xrange(1,n+1):
	if ar[i] < 0:
		continue
	tmp = [i]
	while len(tmp) == 1 or tmp[-1] != i:
		tmp.append(ar[tmp[-1]])
		ar[tmp[-2]] = -1
	cycles.append(tmp)
print len(cycles)
for t in cycles:
	print ' '.join(map(str,t))
