import sys

length = int(raw_input())
numbers = map(int, raw_input().strip().split())

nbVisited = 0
results = []
visited = dict()

def walkThrough(array, index):

    global visited

    result = [index + 1]
    visited.setdefault(index, True)
    firstIndex = index
    nextIndex = array[firstIndex] - 1
    if firstIndex == nextIndex or \
        visited.get(nextIndex, False) is True:
        return [firstIndex + 1, nextIndex + 1]
    while firstIndex != nextIndex and \
        visited.get(nextIndex, False) is False:
        result.append(nextIndex + 1)
        visited.setdefault(nextIndex, True)
        nextIndex = array[nextIndex] - 1
    result.append(nextIndex + 1)
    return result


while nbVisited < length:
    for index, num in enumerate(numbers):
        if not visited.get(index, False):
            result = walkThrough(numbers, index)
            nbVisited += (len(result) - 1)
            results.append(result)

print(len(results))
for result in results:
    for index, num in enumerate(result):
        sys.stdout.write(str(num))
        if index != (len(result) - 1):
            sys.stdout.write(' ')
    sys.stdout.write('\n')
