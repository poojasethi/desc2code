#include<bits/stdc++.h>
using namespace std;
int P[1000000];
int main(){
	int t;
	cin >> t;
	while(t--){
		int N, M;
		cin >> N >> M;
		for(int i = 0;i < M;++i){
			cin >> P[i];
		}
		vector<vector<int> > bottles(N);
		for(int i = 0;i < N;++i){
			int cnt;
			cin >> cnt;
			while(cnt--){
				int x;
				cin >> x;
				bottles[i].push_back(x);
			}
			sort(bottles[i].begin(), bottles[i].end());
		}
		int ans = 0;
		for(int i = 0;i < M;++i){
			if(!bottles[P[i]].empty()){
				ans += bottles[P[i]].back();
				bottles[P[i]].pop_back();
			}
		}
		cout << ans << "\n";
	}
	return 0;
}
