#include<bits/stdc++.h>
#include<queue>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		long n,m;
		scanf("%d%d",&n,&m);
		long w[100004]={0};
		for(int i=0;i<m;i++){
			long temp;
			scanf("%ld",&temp);
			w[temp]++;
		}
		priority_queue<long> q[m];
		for(int i=0;i<n;i++){
			long c;
			scanf("%ld",&c);
			for(int j=0;j<c;j++){
				long temp;
				scanf("%ld",&temp);
				q[i].push(temp);
			}
		}
		long ans=0;
		for(int i=0;i<n;i++){
			for(int j=0;j<w[i];j++){
				if(q[i].empty())
				break;
				long temp=q[i].top();q[i].pop();
				ans+=temp;
			}
		}
		printf("%ld\n",ans);
	}
}
