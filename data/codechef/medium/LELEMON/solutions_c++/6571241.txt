void scanint(int &x)
{
    int flag=0;
    register int c = gc();
    if(c == '-') flag=1;
    x = 0;
    for(; (c<48 || c>57); c = gc());
    for(; c>47 && c<58; c = gc())
    {
        x = (x<<1) + (x<<3) + c - 48;
    }
    if(flag == 1)x=-x;
}
