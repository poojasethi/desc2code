#include<iostream>
#include<malloc.h>
using namespace std;
 
int room_order[10000];
int memory[100000]={0};
int index[10000]; 
 
int main() 
{
  int test;
  cin>>test;
  while(test--)
  { int n,m,i=0,j=0,k=0,max,count=0,start,pos,sum=0,num_bottles=0;
    cin>>n>>m;
    for(i=0;i<m;i++)
    cin>>room_order[i];
    for(i=0;i<n;i++)
    {
     cin>>memory[j];
     num_bottles=memory[j];
     index[i]=j;
     for(k=1;k<=num_bottles;k++)
     {
        cin>>memory[++j];
     }
     j++;
    }
    for(i=0;i<m;i++)
   {
    start=index[room_order[i]];
    max=memory[start+1];
    pos=start+1;
    for(k=2;k<=memory[start];k++)
    {
       if(max<memory[start+k])
       { max=memory[start+k];
         pos=start+k;
       }
    }
    memory[pos]=0;
    sum+=max;
   }
   cout<<sum<<"\n";
  }
   return(0);
}
