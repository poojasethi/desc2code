#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int t,n,m;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        int p[m+1],ci[n+1],i,j;
        long long civ[n+1][102],sum=0;
        for(i=0;i<m;i++)
            scanf("%d",&p[i]);
        for(i=0;i<n;i++)
        {
            scanf("%d",&ci[i]);
            for(j=0;j<ci[i];j++)
                scanf("%lld",&civ[i][j]);

            sort(civ[i],civ[i]+ci[i]);
        }
        for(i=0;i<m;i++)
        {
            if(ci[p[i]]!=0)
            {
                sum+=civ[p[i]][ci[p[i]]-1];
                ci[p[i]]--;
            }
        }
        printf("%lld\n",sum);
    }
}
