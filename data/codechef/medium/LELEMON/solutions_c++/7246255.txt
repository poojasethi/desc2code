#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int T,n,m,A[101],B[101];
int main()
{
  	scanf("%d",&T);
	while(T--)
	{
		scanf("%d %d",&n,&m);
		memset(A,0,sizeof(A));
		for(int x,i=0;i<m;i++)
		{
			scanf("%d",&x);
			A[x]++;
		}
		
		long long res=0;
		for(int c,i=0;i<n;i++)
		{
			scanf("%d",&c);
			for(int j=0;j<c;j++)scanf("%d",&B[j]);
			sort(B,B+c);
			if(A[i])
			{
				if(A[i]>=c)for(int j=0;j<c;j++)res+=B[j];
				else{
					for(int j=c-1;j>=(c-A[i]);j--)res+=B[j];
				}
			}
		}
		cout<<res<<"\n";
	}
	return 0;
}
