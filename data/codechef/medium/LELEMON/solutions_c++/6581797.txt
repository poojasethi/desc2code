#include<bits/stdc++.h>
using namespace std;

int cmp(int a, int b)
{
	return (a<b) ? 0:1;
}

int main()
{
	int t;
	scanf("%d",&t);

	while(t--)
	{
		int n, m, sum=0;
		scanf("%d%d",&n,&m);

		int p[m], d[m];

		for(int i=0; i<m; ++i)
		{
			d[i]=0;
		}

		for(int i=0; i<m; ++i)
		{
			scanf("%d",&p[i]);
			d[p[i]]++;
		}

		for(int i=0; i<n; ++i)
		{
			int c;
			scanf("%d",&c);
			 int v[c];

			for(int j=0; j<c; ++j)
				scanf("%d",&v[j]);

			sort(v, v+c, cmp);

			int m = min(d[i], c);

			for(int j=0; j<m; ++j)
			{
				sum+=v[j];
			}

		}

		cout<<sum<<endl;
	}
	return 0;
}
