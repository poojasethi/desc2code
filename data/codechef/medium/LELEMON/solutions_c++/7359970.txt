#include<iostream>
#include<algorithm>
#define mod 1000000007
typedef long long int ll;
using namespace std;
int main()
{
    ll n,t,m,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        ll a[m],lemon=0;
        for(i=0;i<m;i++)
        {
            cin>>a[i];
        }
        ll b[n][105];
        for(i=0;i<n;i++)
        {
            cin>>b[i][0];
            for(j=1;j<=b[i][0];j++)
            {
                cin>>b[i][j];
            }
            sort(b[i]+1,b[i]+b[i][0]+1);
        }
        for(i=0;i<m;i++)
        {
            for(j=b[a[i]][0];j>=1;j--)
            {
                if(b[a[i]][j]!=0)
                {
                    lemon+=b[a[i]][j];
                    b[a[i]][j]=0;
                    break;
                }
            }
        }
        cout<<lemon<<'\n';
    }
    return 0;
}
