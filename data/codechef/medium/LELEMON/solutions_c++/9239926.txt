#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m;
		scanf("%d",&n);
		scanf("%d",&m);
		int a[m];
		int f[n];
		for(int i=0;i<n;i++)
		{
		    f[i]=0;	
		}
		for(int i=0;i<m;i++)
		{
			scanf("%d",&a[i]);
			f[a[i]]++;
		}
		
		long long int sum=0;
		for(int i=0;i<n;i++)
		{
	        int c;	
			scanf("%d",&c);
			int v[c];
			int sum1=0;
			for(int j=0;j<c;j++)
			{
				scanf("%d",&v[j]);
				sum1=sum1+v[j];
			}
			sort(v,v+c);
			sum=sum+sum1;
			if(f[i]<c)
			 {
			 	for(int j=0;j<(c-f[i]);j++)
			 	      sum=sum-v[j];
			 }
		}   
		printf("%lld\n",sum);
	}
}
