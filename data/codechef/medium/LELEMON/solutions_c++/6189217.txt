#include<stdio.h>
#include<iostream>
#include<stdlib.h>
int compare(const void *i,const void *j)
{ return *(int *)i-*(int *)j;
}
int main()
{ int t;
  scanf("%d",&t);
  while(t--)
       { int n,m,c,p[100],s[100]={0},a,sum=0;
         scanf("%d%d",&n,&m);
         for(int i=0;i<m;i++)
            { scanf("%d",&a);
              s[a]++;
            }
         for(int i=0;i<n;i++)
		    { scanf("%d",&c);
		      for(int j=0;j<c;j++)
		         scanf("%d",&p[j]);
		      if(s[i]!=0)
			    { qsort(p,c,sizeof(int),compare);
			      for(int j=0;j<s[i];j++)
			         if(c-1-j>=0) 
					   sum+=p[c-1-j];
			    }   
		    }   
		 printf("%d\n",sum);   
       }
}
