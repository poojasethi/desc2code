
#include<bits/stdc++.h>

using namespace std;

int main()

{

    vector< vector<int> > myvec;

    int i,j,n,m,no,room,p,visit[10003],t;unsigned int vol;

 

    scanf("%d",&t);

    while(t--)

    {

        vol=0;

        scanf("%d%d",&n,&m);

 

        for(i=0;i<m;i++)

            scanf("%d",&visit[i]);

 

            for( i = 0; i < n; i++)

    {

        vector<int> row;

        myvec.push_back(row);

      }

 

        for(i=0;i<n;i++)

        {

            scanf("%d",&no);

            for(j=0;j<no;j++)

            {

                scanf("%d",&p);

                myvec[i].push_back(p);

            }

            sort(myvec[i].begin(),myvec[i].end());

 

        }

 

        for(i=0;i<m;i++)

        {

            room=visit[i];

            if(myvec[room].size()>0)

            {

 

                vol=vol+myvec[room].back();

                myvec[room].pop_back();

            }

        }

        printf("%u\n",vol);

        myvec.clear();

 

    }

    return 0;

}
