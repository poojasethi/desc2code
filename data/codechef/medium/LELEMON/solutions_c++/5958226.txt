#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

struct elephant
{
    int arr[100001];
    int top;
};
typedef struct elephant ele;
int choice[100001];
int bottle[1000][1000];
int c[1000];
int main()
{
    int t,n,m,i,val,j;
    scanf("%d",&t);
    while(t--)
    {
        int ans=0;
        scanf("%d%d",&n,&m);
        memset(choice,0,sizeof(choice));
        for(i=0;i<m;i++)
        {
            scanf("%d",&val);
            choice[val]++;
        }
        for(i=0;i<n;i++)
        {
            scanf("%d",&c[i]);
            for(j=0;j<c[i];j++)
            {
                scanf("%d",&bottle[i][j]);
            }
            sort(bottle[i],bottle[i]+c[i]);
            reverse(bottle[i],bottle[i]+c[i]);
            for(j=0;j<min(c[i],choice[i]);j++)
            {
                //cout<<bottle[i][j]<<endl;
                ans+=bottle[i][j];
            }
        }
        printf("%d\n",ans);

    }
    return 0;
}
