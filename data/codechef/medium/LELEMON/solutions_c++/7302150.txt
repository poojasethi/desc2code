#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
using namespace std;
vector<int> r(10000);
vector<int> l(100);
vector<vector<int> > v(100,vector<int>(100));
int main()
{
    int i,t,n,m,j,elem,vol,sum;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        for(i=0;i<m;i++)
        {
            scanf("%d",&(r[i]));
        }
        for(i=0;i<n;i++)
        {
            scanf("%d",&elem);
            for(j=0;j<elem;j++)
            {
                scanf("%d",&(vol));
                v[i][j]=vol;
            }
            sort(v[i].begin(),v[i].begin()+elem);
            l[i]=elem;
        }
        sum=0;
       for(i=0;i<m;i++)
        {
            if(l[r[i]]>0)
            {
                sum+=v[r[i]][l[r[i]]-1];
                l[r[i]]-=1;
            }
        }
        printf("%d\n",sum);
    }
    return 0;
}
