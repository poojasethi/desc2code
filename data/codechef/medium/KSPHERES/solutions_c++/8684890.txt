#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define sl(l) scanf("%lld",&l)
#define LOOP(i,n) for(ll i=0;i<n;i++)
#define mod 1000000007
int main()
{
	ll n,m,c,R=0,k=1;
	ll h[1003];
	ll l[1003];
	ll e[1003];
	ll f[1003];
	memset(h,0,sizeof(h));
	memset(l,0,sizeof(l));
	sl(n);sl(m);sl(c);
	LOOP(i,n)
	{
		ll j;
		sl(j);
		R = max(j,R);
		h[j]++;
	}

	LOOP(i,m)
	{
		ll j;
		sl(j);
		R = max(j,R);
		l[j]++;
	}

	for(ll i=1;i<=c;i++)
	{
		if(h[i]!=0 && l[i]!=0)
		{
			e[k] = (h[i]*l[i])%mod;
			k++;
		}
	}

	// k is the no of distinct radii spheres
	ll r = k;
	ll j;
	f[1] = e[1]; // no of ways to get the 
	for(ll i=2;i<r;i++)
	{
			j = i;
			while(j>1)
			{
				f[j] += (f[j-1]*e[i]);
				f[j]%=mod;
				j--;
			}

		f[1]+=e[i];
	}

	for(ll i=2;i<=c+1;i++)
	{
		printf("%lld ",f[i]);
	}
	printf("\n");
	return 0;
}
