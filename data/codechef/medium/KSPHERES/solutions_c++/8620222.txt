#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mod 1000000007
#define ll long long int
ll a[1000006],b[1000006];
ll arr[100006][2];
vector<ll>v;
int main()
{
    ll n,m,i,j,c;
    cin>>n>>m>>c;
    for(i=0;i<n;i++)
        cin>>a[i];
    for(i=0;i<m;i++)
        cin>>b[i];
    sort(a,a+n);
    sort(b,b+m);
    i=0,j=0;
    ll ac,bc;
    for(;i<n&&j<m;)
    {
        if(a[i]<b[j])
            i++;
        else if(a[i]>b[j])
            j++;
        else
        {
            ac=1,bc=1;
            while(i<n-1&&a[i]==a[i+1])
                ac++,i++;
            while(j<m-1&&b[j]==b[j+1])
                bc++,j++;
            i++;j++;
            ll x=(ac*bc)%mod;
            v.pb(x);
        }
    }
    /*cout<<"v arr is\n";
    for(i=0;i<v.size();i++)
        cout<<v[i]<<" ";cout<<"\n";*/
    ll len=v.size(),lim=len-1;
    ll cur=1,prev=0;
    arr[len][0]=v[0];
    arr[len-1][0]=1;
    for(i=1;i<len;i++)
    {
        for(j=lim-1;j<len;j++)
            arr[j][cur]=arr[j+1][prev];
        /*if(i==len-1)
            arr[lim-1][cur]=0;*/
        arr[j][cur]=0;
        for(j=lim;j<=len;j++)
            arr[j][cur]=(arr[j][cur]+(v[i]*arr[j][prev])%mod)%mod;
        /*cout<<"for i="<<i<<"\t\t\t";
        for(j=lim;j<=len;j++)
            cout<<arr[j][cur]<<" ";cout<<"\n";*/
        lim--;
        swap(cur,prev);
    }
    j=0;
    for(i=lim+2;i<=len&&j<c;i++,j++)
        cout<<arr[i][prev]<<" ";
    while(j<c)
    {cout<<arr[i][prev]<<" ";i++;j++;}
    return 0;
}
