#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
int res,opt,t,n,m,mat[100][100],visited[100],a,b,c;
int  go(int k,int l)
{
    int i;
	if(l>opt)
        return 0;

	if(k==n)
	{
		if(l<opt)
		{
			opt =l;
			res=1;
		}
		else if(l==opt)
            res++;

	}
	visited[k]=1;
	for(i=1;i<=n;i++)
	{
        if(visited[i]==0 && mat[k][i]>0)
            go(i,l+mat[k][i]);
    }

	visited[k]=0;

	//return res;

}
int main()
{
    int i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&m);
        memset(mat,0,sizeof(mat));
        memset(visited,0,sizeof(visited));
        for(i=1;i<=m;i++)
        {
            scanf("%d %d %d",&a,&b,&c);


            mat[a][b]=c;
            mat[b][a]=c;

        }
        //printf("$$\n");
        res=0;
        opt=999999;
        visited[0]=1;
        go(1,0);
        if(res>0)
            printf("%d\n",res);
    }
    return 0;
}
