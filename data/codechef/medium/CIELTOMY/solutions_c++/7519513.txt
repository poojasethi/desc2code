#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
using namespace std;
int vis[100];
int wt[1000];
vector<pair<int,int> > v[15];

void dfs(int cur,int curwt,int n)
{
	int i,j;
//	printf("%d  %d\n",cur,curwt);
	if(cur==n)
	{
		wt[curwt]++;
		return;
	}
	vis[cur]=1;
	for(i=0;i<v[cur].size();i++)
	{
		if(vis[v[cur][i].first]==0)
		{
			dfs(v[cur][i].first,curwt+v[cur][i].second,n);
		}
	}
	vis[cur]=0;
	return;
}

			wt[i]=0;
		while(m--)

			v[a].push_back(make_pair(b,c));

	}

}
