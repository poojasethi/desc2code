#include <iostream>
using namespace std;
int am[10][20];
bool visited[10];
int lmin=1000;
int ans = 0;
int n;
void go(int k, int len)
{
    if(len>lmin) return;
    if(k==(n-1))
    {
        if(lmin>len)
        {
            lmin = len;
            ans = 1;
        }
        else if ( lmin==len ) ans++;
        // cout<<ans<<endl;
        return;
    }
    visited[k] = true;
    for(int i=0;i<n;i++) if( !visited[i] && am[k][i]>0 ) go(i,len+am[k][i]);
    visited[k] = false;
}
int main()
{
    int t;
    cin >> t;
    while ( t-- )
    {
        int m;
        cin >> n >> m;
        for(int i=0; i<n; i++)
        {
            visited[n] = false;
            for(int j=0;j<n;j++) am[i][j]=0;
        }
        int a, b, c;
        while(m--)
        {
            cin >> a >> b >> c;
            am[a-1][b-1] = am[b-1][a-1] = c;
        }
        lmin = 1000;
        go(0,0);
        cout << ans << endl;
    }
    return 0;
}
