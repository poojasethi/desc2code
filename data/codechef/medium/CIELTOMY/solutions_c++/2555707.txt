using namespace std;

int AdjMat[100][100];
bool visited[100];
int n, m, l, num;

void init()
{
    l = inf;
    num = 0;
    FOR(i, 0, 100)
        visited[i] = false;
    FOR(i, 0, 100)
    {
        AdjMat[i][i] = 0;
        FOR(j, 0, 100)
            AdjMat[i][j] = 0;
    }
}

void go(int k, int len)
{
    if(len > l)
        return ;
    if(k == n)
    {
        if(len < l)
        {
            l = len;
            num = 1;
        }
        else if(len == l)
            num++;
        return ;
    }
    visited[k] = true;
    for(int i = 1; i <= n; i++)
    {
        if(!visited[i] && AdjMat[k][i] > 0)
            go(i, len + AdjMat[k][i]);
    }
    visited[k] = false;
}

int main()
{
    int a, b, c, t;
    scanf("%d", &t);
    while(t--)
    {
        scanf("%d %d", &n, &m);
        init();
        FOR(i, 0, m)
        {
            scanf("%d %d %d", &a, &b, &c);
            AdjMat[a][b] = c;
            AdjMat[b][a] = c;
        }
        go(1, 0);
        printf("%d\n", num);
    }
    return 0;
}
