using namespace std;

inline void si(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(; ((c<48 || c>57) && c != '-'); c = gc());
    if(c=='-')
    {
        neg = 1;
        c = gc();
    }
    for(; c>47 && c<58 ; c = gc())
    {
        x = (x<<1) + (x<<3) + c - 48;
    }
    if(neg)
        x = -x;
}

int main()
{
    int n, m, a, b, c, t;
    int dist[11][11], ways[11][11];
    scanf("%d", &t);
    while(t--)
    {
//        scanf("%d %d", &n, &m);
        si(n);
        si(m);
        FOR(i, 1, n+1)
        {
            dist[i][i] = 0;
            ways[i][i] = 1;
            FOR(j, 1, n+1)
            {
                dist[i][j] = 10000;
                ways[i][j] = 0;
            }
        }
        FOR(i, 0, m)
        {
//            scanf("%d %d %d", &a, &b, &c);
            si(a);
            si(b);
            si(c);
            dist[a][b] = c;
            dist[b][a] = c;
            ways[a][b] = 1;
            ways[b][a] = 1;
        }
        //Floyd-Warshall's algorithm
        FOR(k, 1, n+1)
            FOR(i, 1, n+1)
                FOR(j, 1, n+1)
                {
                    if(i == k || j == k)
                        continue;

                    if(dist[i][k] + dist[k][j] < dist[i][j])
                    {
                        dist[i][j] = dist[i][k] + dist[k][j];
                        ways[i][j] = ways[i][k] * ways[k][j];
                    }
                    else if(dist[i][k] + dist[k][j] == dist[i][j])
                       ways[i][j] += ways[i][k] * ways[k][j];
                }

        printf("%d\n", ways[1][n]);
    }
    return 0;
}
