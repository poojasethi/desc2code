int N, M, x, y, w;
struct Edge {
        int from, to, weight;
        Edge(int from, int to, int weight = 1) :
                from(from), to(to), weight(weight) {
        }
};

class minHeap {
    public:
        bool operator()(const pair<int, int> &a, const pair<int, int> &b) const {
            if (a.first != b.first) {
                return a.first > b.first;
            }

            return a.second > b.second;
        }
};
class Graph {
        int N;
        vector<list<Edge> > adj;
        // 0 based indexing
    public:
        Graph(int n) {
            N = n;
            adj.resize(N);
        }

        void addEdge(int from, int to, int weight = 1) {
            adj[from].push_back(Edge(from, to, weight));
        }

        ~Graph() {
            for (int i = 0; i < adj.size(); i++) {
                adj[i].clear();
            }
            adj.clear();
        }

        void dijkstra(int src, vector<int>&dis) {
            dis.clear();
            dis.resize(N, MOD);
            vector<bool> visited(N, false);
            //for the purpose of priority queue, using sets
            set<pair<int, int> > pq;
            dis[src] = 0;
            pq.insert(make_pair(0, src));
            list<Edge>::iterator it;
            while (!pq.empty()) {
                pair<int, int> p = *(pq.begin());
                pq.erase(pq.begin());
                int d = p.first;
                int v = p.second;
                if (visited[v]) {
                    continue;
                }
                visited[v] = true;
                for (it = adj[v].begin(); it != adj[v].end(); it++) {
                    int newDis = d + it->weight;
                    if (newDis < dis[it->to]) {
                        dis[it->to] = newDis;
                        pq.insert(make_pair(newDis, it->to));
                    }
                }
            }

        }

        void dijkstra2(int src, vector<int>&dis, vector<int>&numPaths) {
            dis.clear();
            dis.resize(N, MOD);
            numPaths.clear();
            numPaths.resize(N, 0);
            vector<bool> visited(N, false);
            priority_queue<pair<int, int>, vector<pair<int, int> >, minHeap> pq;
            dis[src] = 0;
            numPaths[src] = 1;
            pq.push(make_pair(0, src));
            list<Edge>::iterator it;
            while (!pq.empty()) {
                pair<int, int> p = pq.top();
                pq.pop();
                int d = p.first;
                int u = p.second;
                if (visited[u]) {
                    continue;
                }
                visited[u] = true;
                for (it = adj[u].begin(); it != adj[u].end(); it++) {
                    Edge e = *it;
                    int newDis = d + e.weight;
                    if (newDis < dis[e.to]) {
                        dis[it->to] = newDis;
                        pq.push(make_pair(newDis, e.to));
                        numPaths[e.to] = numPaths[u];
                    } else if (newDis == dis[e.to]) {
                        numPaths[e.to] += numPaths[u];
                    }
                }
            }

        }
};
void solve(int cx) {
    scanf("%d %d", &N, &M);
    db(N, M);
    Graph g(N + 1);
    REP(i, M)
    {

        db(x, y,w);
        g.addEdge(x, y, w);
        g.addEdge(y, x, w);
    }
    VI dis;
    VI numPaths;
    g.dijkstra2(1, dis, numPaths);
    db(dis);
    printf("%d\n", numPaths[N]);
}
