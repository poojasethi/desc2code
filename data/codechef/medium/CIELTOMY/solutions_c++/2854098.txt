#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int n,m,a,b,c,kas;
    int ar[15][15],br[15][15];
    cin>>kas;
    for(int t=1;t<=kas;t++)
    {
        cin>>n>>m;
        for(int i=1;i<=n;i++)
            for(int j=1;j<=n;j++)
            {
                if(i!=j)
                    ar[i][j]=9999;
                br[i][j]=0;
            }
        for(int i=1;i<=m;i++)
        {
            cin>>a>>b>>c;
            ar[a][b]=ar[b][a]=c;
            br[a][b]=br[b][a]=1;
        }
        for(int k=1;k<=n;k++)
            for(int i=1;i<=n;i++)
                for(int j=1;j<=n;j++)
                {
                    if(k!=j&&k!=i)
                    {
                        if(ar[i][j]==ar[i][k]+ar[k][j])
                        {
                            br[i][j]+=br[i][k]*br[k][j];
                        }
                        else if(ar[i][j]>ar[i][k]+ar[k][j])
                        {
                            ar[i][j]=ar[i][k]+ar[k][j];
                            br[i][j]=br[i][k]*br[k][j];
                        }
                    }
 
                }
        cout<<br[1][n]<<endl;
 
    }
 
    return 0;
}
