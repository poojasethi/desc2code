#include <iostream>
#include <stdio.h>
#include <bitset>
#include <limits.h>
using namespace std;
int main(){

    int t;scanf("%d",&t);
    int n,m;
    while(t--){
            scanf("%d %d",&n,&m);
            int ct[n+1][n+1];
            int fin[n+1][n+1];
            for(int i=1;i<=n;i++){
            for(int j=1;j<=n;j++){
                    ct[i][j]=0;
                    if(i==j){fin[i][j]=0;continue;}
                    fin[i][j]=1000000;
            }
            }
            int a,b,c;
            
            for(int i=1;i<=m;i++) {
                    scanf("%d %d %d",&a,&b,&c);
                    fin[a][b]=c;
                    fin[b][a]=c;
                    ct[a][b]=1;
                    ct[b][a]=1;
            }
            for(int k=1;k<=n;k++){
                for(int i=1;i<=n;i++){
                        for(int j=1;j<=n;j++){   
                            if(fin[i][k]>=1000000 || fin[k][j]>=1000000) continue;  
                            if(i==k || k==j) continue;
                                if(fin[i][j]>fin[i][k]+fin[k][j]) {
                                      fin[i][j]=fin[i][k]+fin[k][j];
                                      ct[i][j]=ct[i][k]*ct[k][j];      
                                }
                                else if(fin[i][j]==fin[i][k]+fin[k][j]){
                                     ct[i][j]=ct[i][j]+ct[i][k]*ct[k][j];     
                                }    
                        }  
                }
            }
        printf("%d\n",ct[1][n]);                  
    }
}
