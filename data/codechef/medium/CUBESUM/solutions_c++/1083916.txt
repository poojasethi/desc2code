#include<iostream>

using namespace std;

int a[102][102][102];
int b[102][102][102];
int X,Y,Z;

void input()
{
    int i,j,k;

    for(i=0;i<=X;i++)
        for(j=0;j<=Y;j++)
            for(k=0;k<=Z;k++)
            {
                if(i==0||j==0||k==0)
                    b[i][j][k] = 0;
                else
                    cin >> b[i][j][k];
            }
}

void calcA()
{
    int i,j,k;

    for(i=1;i<=X;i++)
        for(j=1;j<=Y;j++)
        {
            for(k=1;k<=Z;k++)
            {
                a[i][j][k] = ( b[i][j][k] - b[i][j-1][k] ) - ( b[i-1][j][k] - b[i-1][j-1][k]) - ( b[i][j][k-1] - b[i][j-1][k-1] - ( b[i-1][j][k-1] - b[i-1][j-1][k-1]));
                cout << a[i][j][k] << " ";
            }

            cout << endl;
        }
}

int main()
{
    int test;

    cin >> test;
    while(test--)
    {
        cin >> X >> Y >> Z;
        input();
        calcA();
    }

    return 0;
}
