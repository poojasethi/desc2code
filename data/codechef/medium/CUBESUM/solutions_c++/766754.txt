#define DEBUG 1

#include <stdio.h>
#include <string.h>

int main(void) {
  int b[101][101][101];
  int x, y, z, X, Y, Z, tc;

  scanf("%d",&tc);
  memset(b, 0, sizeof(b));
  while(tc--) {
    scanf("%d%d%d", &X, &Y, &Z);

    for(x=1;x<=X;x++)
      for(y=1;y<=Y;y++)
	for(z=1;z<=Z;z++)
	  scanf("%d", &b[x][y][z]);

    for(x=1;x<=X;x++)
      for(y=1;y<=Y;y++) {
	printf("%d", b[x][y][1]-b[x][y-1][1]-b[x-1][y][1]+b[x-1][y-1][1]);
	for(z=2;z<=Z;z++)
	  printf(" %d", b[x][y][z]-b[x][y-1][z]-b[x][y][z-1]-b[x-1][y][z]+b[x][y-1][z-1]+b[x-1][y][z-1]+b[x-1][y-1][z]-b[x-1][y-1][z-1]);
	printf("\n");
      }
  }
  return 0;
}
