#include <iostream>
#include <cstdio>
#include <cstdlib>

#define LL long long
#define FOR(i,m,n) for((i)=(m); (i)<(m); (i++))

int main()
{
     int t;
     scanf("%d",&t);
     int X,Y,Z;
     int B[100][100][100];
     int A[100][100][100];
     int i,j,k;
     while(t--){
	  scanf("%d %d %d",&X,&Y,&Z);
	  for(i=0;i<X;i++)
	       for(j=0;j<Y;j++)
		    for(k=0;k<Z;k++){
			 scanf("%d",&(B[i][j][k]));
			 A[i][j][k]=B[i][j][k];
			 if(i-1>=0) A[i][j][k]-=B[i-1][j][k];
			 if(j-1>=0) A[i][j][k]-=B[i][j-1][k];
			 if(k-1>=0) A[i][j][k]-=B[i][j][k-1];
			 if(i-1>=0 && j-1>=0) A[i][j][k]+=B[i-1][j-1][k];
			 if(j-1>=0 && k-1>=0) A[i][j][k]+=B[i][j-1][k-1];
			 if(i-1>=0 && k-1>=0) A[i][j][k]+=B[i-1][j][k-1];
			 if(i-1>=0 && j-1>=0 && k-1>=0) A[i][j][k]-=B[i-1][j-1][k-1];
		    }
	  for(i=0;i<X;i++)
	       for(j=0;j<Y;j++){
		    for(k=0;k<Z;k++){
			 printf("%d ",A[i][j][k]);
		    }
		    printf("\n");
	       }
     }
}
