#include <stdio.h>
using namespace std;

int main(){
int T,x,y,z;
int B[101][101][101];
B[0][0][0] = 0;
for(int i=1;i<101;i++){
        B[0][0][i] = 0;
        B[0][i][0] = 0;
        B[i][0][0] = 0;
    for(int j=1;j<101;j++){
       B[0][i][j] = 0;
       B[i][j][0] = 0;
       B[i][0][j] = 0;
    }
}
scanf("%d",&T);
for(int i=0;i<T;i++){
     scanf("%d",&x);
     scanf("%d",&y);
     scanf("%d",&z);
   for(int p=1;p<=x;p++){
    for(int q=1;q<=y;q++){
        for(int r=1;r<=z;r++){
            scanf("%d",&B[p][q][r]);
            printf("%d ",B[p][q][r]-B[p-1][q-1][r-1]+B[p][q-1][r-1]+B[p-1][q][r-1]+B[p-1][q-1][r]-B[p][q][r-1]-B[p-1][q][r]-B[p][q-1][r]);
        }printf("\n");
    }
   }

}
}
