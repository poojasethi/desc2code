#include <iostream>
#include <stdio.h>
#include <vector>
#include <map>
#include <string>
#include <algorithm>
using namespace std;
int main(){
int t;scanf("%d",&t);
while(t--){
		int x,y,z;
		scanf("%d %d %d",&x,&y,&z);
		int B[x][y][z],A[x][y][z];
		//cout<<x<<" "<<y<<" "<<z<<endl;
		for(int i=0;i<x;i++){
			for(int j=0;j<y;j++){
				for(int k=0;k<z;k++) {
					int tmp;
					scanf("%d",&tmp);
					B[i][j][k]=tmp;
					//cout<<i<<" "<<j<<" "<<k<<" :g "<<B[0][0][0]<<endl;
					}
			}
		}
		
		//cout<<B[0][0][0]<<" "<<B[1][0][0]<<" "<<B[2][0][0]<<endl;
		
		A[0][0][0]=B[0][0][0];
		for(int i=1;i<x;i++) A[i][0][0]=B[i][0][0]-B[i-1][0][0];
		for(int j=1;j<y;j++) A[0][j][0]=B[0][j][0]-B[0][j-1][0];
		for(int k=1;k<z;k++) A[0][0][k]=B[0][0][k]-B[0][0][k-1];
		
		for(int i=1;i<x;i++)
			for(int j=1;j<y;j++)
				A[i][j][0]=B[i][j][0]-B[i-1][j][0]-B[i][j-1][0]+B[i-1][j-1][0];
		
		for(int j=1;j<y;j++)
			for(int k=1;k<z;k++)
				A[0][j][k]=B[0][j][k]-B[0][j][k-1]-B[0][j-1][k]+B[0][j-1][k-1];
				
		for(int i=1;i<x;i++)
			for(int k=1;k<z;k++)
				A[i][0][k]=B[i][0][k]-B[i-1][0][k]-B[i][0][k-1]+B[i-1][0][k-1];
		
		for(int i=1;i<x;i++){
			for(int j=1;j<y;j++){
				for(int k=1;k<z;k++) 
					A[i][j][k]=B[i][j][k]-(B[i-1][j][k]+B[i][j-1][k]+B[i][j][k-1])+(B[i-1][j-1][k]+B[i-1][j][k-1]+B[i][j-1][k-1])-B[i-1][j-1][k-1];
			}
		}
		
		for(int i=0;i<x;i++){
			for(int j=0;j<y;j++){
				for(int k=0;k<z;k++) printf("%d ",A[i][j][k]);
				printf("\n");
			}
			
		}	
}
}
