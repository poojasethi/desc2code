#include<iostream>
#include<cstdio>
#include<vector>

using namespace std;

int mTop[100][100][100];
int mB[100][100][100];
int mPlane[100][100][100];

int main()
{
   // freopen("input.txt","r",stdin);
   // freopen("output.txt","w",stdout);
    int t;
    int x,y,z;
    scanf("%d",&t);
    while(t)
    {
        scanf("%d %d %d",&x,&y,&z);
        for(int i=0;i<x;i++)
        {
            for(int j=0;j<y;j++)
            {
                for(int k=0;k<z;k++)
                {
                    scanf("%d",&mB[i][j][k]);
                }
            }
        }
        for(int i=0;i<x;i++)
        {
            for(int j=0;j<y;j++)
            {
                for(int k=0;k<z;k++)
                {
                    int temp1=0;
                    int temp2=0;
                    int temp3=0;
                    if(k!=0)
                    {
                        temp1=mB[i][j][k-1];
                    }
                    mPlane[i][j][k]=mB[i][j][k]-temp1;

                    if(j!=0)
                    {
                        temp2=mPlane[i][j-1][k];
                    }
                    mTop[i][j][k]=mPlane[i][j][k]-temp2;


                    if(i!=0)
                    {
                        temp3=mTop[i-1][j][k];
                    }
                    //mA[i][j][0]=mB[i][j]-temp;
                    printf("%d ",mB[i][j][k]-temp1-temp2-temp3);
                }
                printf("\n");
            }

            }
        t--;
    }

    return 0;
}
