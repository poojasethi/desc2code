    #include<stdio.h>
     
    int a[102][102][102],x,y,z;
    int main()
    {
    int t;
    int i,j,k;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%d%d%d",&x,&y,&z);
    for(i=1;i<=x;i++,printf("\n"))
    for(j=1;j<=y;j++,printf("\n"))
    for(k=1;k<=z;k++)
    {
    scanf("%d",&a[i][j][k]);
    printf("%d",a[i][j][k]-a[i-1][j][k]-a[i][j-1][k]-a[i][j][k-1]+a[i-1][j-1][k]+a[i-1][j][k-1]+a[i][j-1][k-1]-a[i-1][j-1][k-1]);
    if(k<z)
    printf(" ");
    }
    }
    return 0;
    }
