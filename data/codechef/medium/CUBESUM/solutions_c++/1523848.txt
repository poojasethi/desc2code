#include<stdio.h>
#include<iostream>
#include<math.h>
#include<string.h>

#include<vector>
#include<algorithm>

#define in(i) scanf("%llu",&i);

using namespace std;


int main(){
  unsigned long long tc,a,b,c,B[101][101][101]={0},x=0;
  in(tc);
  
  while(tc--){
    in(a);
    in(b);
    in(c);
    memset(B,0,sizeof(B));
    
    for(int i=1;i<=a;i++)
      for(int j=1;j<=b;j++)
	for(int k=1;k<=c;k++)
	  in(B[i][j][k]);
	
    for(int i=1;i<=a;i++)
      for(int j=1;j<=b;j++){
	for(int k=1;k<=c;k++){
	  if(k!=1)
	    printf(" ");
	  printf("%llu",B[i][j][k] - B[i-1][j][k] - B[i][j-1][k] - B[i][j][k-1] + B[i-1][j-1][k] + B[i-1][j][k-1] + B[i][j-1][k-1] - B[i-1][j-1][k-1]);
	}
	printf("\n");
      }
  }
  return 0;
}
