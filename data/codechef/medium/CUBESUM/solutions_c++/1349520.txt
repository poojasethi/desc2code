#include <iostream>
#include<cstdio>
using namespace std;
int a[101][101][101];
int main()
{
    int t,x,y,z,i,j,k,sum;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&x,&y,&z);
        for(i=1;i<=x;i++)
        {
            for(j=1;j<=y;j++)
            {
                for(k=1;k<=z;k++)
                {
                    scanf("%d",&a[i][j][k]);
                    sum=a[i][j][k]+a[i][j-1][k-1]+a[i-1][j][k-1]+a[i-1][j-1][k]-a[i-1][j-1][k-1]-a[i-1][j][k]-a[i][j-1][k]-a[i][j][k-1];
                    printf("%d ",sum);
                }
                printf("\n");
            }
        }
    }
    return 0;
}
