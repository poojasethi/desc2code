#include <iostream>
#include <cstring>
#include <string>
#include <cstdio>
using namespace std;

int t, x, y, z;
int b[105][105][105];
int a[105][105][105];

int main()
{
    scanf("%d", &t);
    while (t--) {
        scanf("%d %d %d", &x, &y, &z);
        memset(a, 0, sizeof a);
        memset(b, 0, sizeof b);

        for (int i=1; i<=x; i++)
            for (int j=1; j<=y; j++)
                for (int k=1; k<=z; k++)
                    cin >> b[i][j][k];

        for (int i=1; i<=x; i++)
            for (int j=1; j<=y; j++)
                for (int k=1; k<=z; k++)
                    a[i][j][k] = b[i][j][k] - b[i][j][k-1] - b[i][j-1][k] - b[i-1][j][k]
                                + b[i-1][j-1][k] + b[i-1][j][k-1] + b[i][j-1][k-1] - b[i-1][j-1][k-1];

        for (int i=1; i<=x; i++) {
            for (int j=1; j<=y; j++) {
                char *sep = "";
                for (int k=1; k<=z; k++) {
                    printf("%s%d", sep, a[i][j][k]);
                    sep = " ";
                }
                printf("\n");
            }
            //cout << endl;
        }

    }
    return 0;
}
