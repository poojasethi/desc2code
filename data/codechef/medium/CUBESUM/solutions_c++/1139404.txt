#include<stdio.h>
#define FOR(a,i,x) for(i=a;i<x+1;i++)
int main()
{
long long int tc,x,y,z,i,j,k;
scanf("%lld",&tc);
while(tc--)
{
scanf("%lld%lld%lld",&x,&y,&z);
long long int b[x+1][y+1][z+1],ans; //make cube A and B;
FOR(0,i,x)
FOR(0,j,y)
b[i][j][0]=0;
FOR(0,i,x)
FOR(0,j,z)
b[i][0][j]=0;
FOR(0,i,y)
FOR(0,j,z)
b[0][i][j]=0;
 
FOR(1,i,x)
FOR(1,j,y)
FOR(1,k,z)
scanf("%lld",&b[i][j][k]);
 
FOR(1,i,x)
FOR(1,j,y)
{
FOR(1,k,z)
{ans=b[i][j][k]+b[i][j-1][k-1]+b[i-1][j][k-1]+b[i-1][j-1][k]-b[i][j][k-1]-b[i][j-1][k]-b[i-1][j][k]-b[i-1][j-1][k-1];
printf("%lld ",ans);
}
printf("\n");
}
}
return 0;
}
