#include <stdio.h>

int main() {

	int t, x, y, z;
	
	scanf("%d", &t);
	
	while(t--) {
		scanf("%d", &x);
		scanf("%d", &y);
		scanf("%d", &z);
		
		int b[x][y][z];
		
		for(int i=0; i<x; i++)
			for(int j=0; j<y; j++) {
				for(int k=0; k<z; k++) {
					scanf("%d", &b[i][j][k]);
					int minus = 0;
					if(i>0 && j>0 && k>0) {
						minus += b[i][j-1][k] - b[i-1][j-1][k];
						minus += b[i][j][k-1] - b[i-1][j][k-1];
						minus -= b[i][j-1][k-1] - b[i-1][j-1][k-1];
						minus += b[i-1][j][k];
					} else if(i>0 && j>0) {
						minus += b[i-1][j][0];
						minus -= b[i-1][j-1][0];
						minus += b[i][j-1][0];
					} else if(i>0 && k>0) {
						minus += b[i-1][0][k];
						minus -= b[i-1][0][k-1];
						minus += b[i][0][k-1];
					} else if(j>0 && k>0) {
						minus += b[0][j-1][k];
						minus += b[0][j][k-1];
						minus -= b[0][j-1][k-1];
					} else if(i>0) {
						minus += b[i-1][0][0];
					} else if(j>0) {
						minus += b[0][j-1][0];
					} else if(k>0) {
						minus += b[0][0][k-1];
					}
					printf("%d ", b[i][j][k]-minus);
				}
				printf("\n");
			}
				
	}
	
	return 0;
}
