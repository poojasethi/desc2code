#include<iostream>
#include<stdio.h>
using namespace std;
int A[100][100][100],B[100][100][100];
int main()
{
    int T,x,y,z;
    scanf("%d",&T);
    int i,j,k;
    while(T>0)
    {
    scanf("%d %d %d",&x,&y,&z);

        for(i=0;i<x;i++)
        {
            for(j=0;j<y;j++)
            {
                for(k=0;k<z;k++)
                {
                    scanf("%d",&B[i][j][k]);
                }
            }
        }
        for(i=0;i<x;i++)
        {
            for(j=0;j<y;j++)
            {
                for(k=0;k<z;k++)
                {
                    if(i>0&&j>0&&k>0)
                    {
                       A[i][j][k]=B[i][j][k]-B[i][j][k-1]-B[i][j-1][k]-B[i-1][j][k]
                                 +B[i-1][j-1][k]+B[i-1][j][k-1]+B[i][j-1][k-1]
                                 -B[i-1][j-1][k-1];
                    }
                    else if(i>0&&j>0&&k==0)
                    {
                        A[i][j][k]=B[i][j][k]-B[i][j-1][k]-B[i-1][j][k]+B[i-1][j-1][k];
                    }
                    else if(i>0&&k>0&&j==0)
                    {
                        A[i][j][k]=B[i][j][k]-B[i][j][k-1]-B[i-1][j][k]+B[i-1][j][k-1];
                    }
                    else if(j>0&&k>0&&i==0)
                    {
                        A[i][j][k]=B[i][j][k]-B[i][j][k-1]-B[i][j-1][k]+B[i][j-1][k-1];
                    }
                    else if(i==0&&j==0&&k>0)
                    {
                        A[i][j][k]=B[i][j][k]-B[i][j][k-1];
                    }
                    else if(i==0&&k==0&&j>0)
                    {
                        A[i][j][k]=B[i][j][k]-B[i][j-1][k];
                    }
                    else if(j==0&&k==0&&i>0)
                    {
                        A[i][j][k]=B[i][j][k]-B[i-1][j][k];
                    }
                    else
                    {
                        A[i][j][k]=B[i][j][k];
                    }
                    printf("%d ",A[i][j][k]);
                }
                printf("\n");
            }
        }
        T--;
    }
    return 0;
}
