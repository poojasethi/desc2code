{

             {
               for(k=1;k<=z;k++)

           {
              for(j=1;j<=y;j++)

                   a[i][j][k]= b[i][j][k] -b[i-1][j][k] -b[i][j-1][k] -b[i][j][k-1]
                       + b[i-1][j-1][k] + b[i-1][j][k-1] + b[i][j-1][k-1]
                       - b[i-1][j-1][k-1];
                }
              }

           {
             for(j=1;j<=y;j++)
             {

               }printf("\n");
             }
