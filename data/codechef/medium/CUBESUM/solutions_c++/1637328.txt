#define LOOK(x,y,z) ((x >= 0 && y >= 0 && z >= 0) ? B[x][y][z] : 0)

	int T;
	scanf("%d", &T);
	for (int ii = 0; ii < T; ii++)
	{
		int X, Y, Z;
		scanf("%d %d %d", &X, &Y, &Z);
		int B[X][Y][Z];
		//int*** B = new int**[X];

		//	}

		for (int i = 0; i < X; i++)
		{
			for (int j = 0; j < Y; j++)
			{
				for (int k = 0; k < Z; k++)
					scanf("%d", &B[i][j][k]);
			}
		}
		int A[X][Y][Z];
		for (int i = 0; i < X; i++)
		{
			for (int j = 0; j < Y; j++)
			{
				for (int k = 0; k < Z; k++)
				{
					A[i][j][k] = B[i][j][k] - LOOK(i-1,j,k) - LOOK(i,j-1,k) - LOOK(i,j,k-1) + LOOK(i-1,j-1,k) + LOOK(i-1,j,k-1) + LOOK(i,j-1,k-1) - LOOK(i-1,j-1,k-1);
					printf("%d ", A[i][j][k]);
				}
				printf("\n");
			}
		}
	}

}
