int scan() {
	int t = 0;
	char c = getcha();
	while(c<'0' || c>'9') c = getcha();
	while(c>='0' && c<='9') {
		t = 10 * t + c - 48;
		c = getcha();
	}
	return t;
}

int a[101][101][101];
int b[101][101][101];
 
void go() {
	int t = scan();
	while(t--) {
		int x=scan(),y=scan(),z=scan();
		for(int i=1; i<=x; i++) {
			for(int j=1; j<=y; j++) {
				for(int k=1; k<=z; k++) {
					b[i][j][k] = scan();
				}
			}
		}
		for(int i=1; i<=x; i++) {
			for(int j=1; j<=y; j++) {
				for(int k=1; k<=z; k++) {
					a[i][j][k] = b[i][j][k]-b[i-1][j][k]-b[i][j-1][k]-b[i][j][k-1]+b[i-1][j-1][k]+b[i][j-1][k-1]+b[i-1][j][k-1]-b[i-1][j-1][k-1];
				}
			}
		}
		for(int i=1;i<=x; i++) {
			for(int j=1; j<=y; j++) {
				for(int k=1; k<=z; k++) {
					printf("%d ",a[i][j][k]);
				}
				printf("\n");
			}
		}
	}
}
