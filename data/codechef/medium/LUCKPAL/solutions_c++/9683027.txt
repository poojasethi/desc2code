int solve(int x,int w)
{
	int y=n-x-1;
	if(x>=y)
	{
		if(w==4 && x==y && s[x]==L[4])return 0;
		if(w==4 && x==y)return 1;
		if(w==5)return 0;
		return 1000000000;
	}
	if(v[x][w]==id)return mem[x][w];
	v[x][w]=id;
	int ans=1000000000;
	for(char c='a';c<='z';c++)
	{
		int cost=0;
		int ww=w;
		if(c!=s[x])cost++;
		if(c!=s[y])cost++;
		if(w==5);
		else if(c==L[w])ww++;
		else if(c==L[0])ww=1;
		else ww=0;
		ans=min(ans,cost+solve(x+1,ww));
	}
	mem[x][w]=ans;
	return ans;
}
