//#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

typedef long long ll;
int T;
char str[1001];

void solve() {
    int N=strlen(str);
    if(N<9) {
        printf("unlucky\n");
        return ;
    }
    char temp[1001], ans[1001];
    int min=1007, count;
    for(int i=0; i<N-4; i++) {
        count=0;
        strcpy(temp, str);
        if(temp[i]!='l') {
            temp[i]='l';
            count++;
        }
        if(temp[i+1]!='u') {
            temp[i+1]='u';
            count++;
        }
        if(temp[i+2]!='c') {
            temp[i+2]='c';
            count++;
        }
        if(temp[i+3]!='k') {
            temp[i+3]='k';
            count++;
        }
        if(temp[i+4]!='y') {
            temp[i+4]='y';
            count++;
        }
        for(int j=0, k=N-1; j<=k; j++, k--) {
            if(j>=i && k<=i+4 && j!=k) {
                count=1006;
                break;
            }
            if(temp[j]!=temp[k]) {
                if(j>=i && j<=i+4) temp[k]=temp[j];
                else if(k>=i && k<=i+4) temp[j]=temp[k];
                else if(temp[k]<temp[j]) temp[j]=temp[k];
                else temp[k]=temp[j];
                count++;
            }
        }
        if(count<min) {
            strcpy(ans, temp);
            min=count;
        }
        else if(count==min) if(strcmp(ans, temp)>0) strcpy(ans, temp);
    }
    printf("%s %d\n", ans, min);
}

int main() {
    scanf("%d", &T);
    while(T--) {
        scanf("%s", str);
        solve();
    }
}
