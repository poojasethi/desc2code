int checkpalindrome(string &s2, int s, int l, int &count){
	int len= s2.length();
	string s3=s2;

	for(int i = 0,j=len-1; i < len, j>=0; ++i,--j){
		if((i < s && j > l) || (i > l && j < s)){
			if(s2[i] != s2[j]){
				s2[i] = min(s2[i],s2[j]);
				s2[j] = min(s2[i],s2[j]);
				++count;
			}
		}else if(i >= s && i <= l && (j > l || j < s)){
			if(s2[i]!=s2[j]){
				s2[j] = s2[i];
				++count;
			}
		}
		else if(j >= s && j <= l && (i > l || i < s)){
			if(s2[i]!=s2[j]){
				s2[i] = s2[j];
				++count;
			}
		}
		else if(i >= s && i <= l && !(j >= l || j <= s)){
			return INT_MAX;
		}
		else if(j >= s && j <= l && !(i >= l || i <= s)){
			return INT_MAX;
		}
		else{
			if(s2[i]!=s2[j]){
				s2[i] = min(s2[i],s2[j]);
				s2[j] = min(s2[i],s2[j]);
				++count;
			}
		}
	}

	return count;
}
