{

        string st;

        if(st.size()<9)
        {
            printf("unlucky\n");
        }
        else
        {
            int mi = 100000;
			string ans = "";
            FOR(i,(int)st.size()-4)
            {
                int c=0;
                string p=st;
                if (p[i] != 'l')
				{
					p[i] = 'l';
					c++;
				}
				if (p[i+1] != 'u')
				{
					p[i+1] = 'u';
					c++;
				}
				if (p[i+2] != 'c')
				{
					p[i+2] = 'c';
					c++;
				}
				if (p[i+3] != 'k')
				{
					p[i+3] = 'k';
					c++;
				}
				if (p[i+4] != 'y')
				{
					p[i+4] = 'y';
					c++;
				}
				for(j=0,k=st.size()-1;j<=k;j++,k--)
				{
                    if(p[j]!=p[k])
                    {
                        if(j>=i && j<=i+4)
                        {
                            if(k>=i && k<=i+4)
                            {
                                c=10000;
                                break;
                            }
                            else
                            {
                                p[k]=p[j];
                                c++;
                            }
                        }
                        else if (k >= i && k <= i+4)
						{
							if (j >= i && j <= i+4)
							{
								c = 100000;
								break;
							} else
							{
								p[j] = p[k];
								c++;
							}
						}
                        else
                        {
                            char ch = min(p[j],p[k]);
							p[j] = ch;
							p[k] = ch;
							c++;
                        }
                    }
				}
				if(mi>c)
				{
                    mi=c;
                    ans=p;
				}
				else if(mi==c)
				{
                    ans=min(ans,p);
				}
            }
            if (ans == "") cout << "unlucky\n";
			else cout << ans << " " << mi << "\n";
        }
    }
    return 0;
}
