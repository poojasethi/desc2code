#include <cstdio>
#include <cassert>
#include <cstring>
#include <algorithm>

using namespace std;

#define dbg(...) fprintf(stderr, __VA_ARGS__)

int len;
char s[1005], buf[1005];

int main ()
{
// 	freopen ("input.txt", "r", stdin);
 //	freopen ("output.txt", "w", stdout);

	int nt; scanf ("%d", &nt);

	while (nt--)
	{
	 	scanf ("%s", s);
	 	len = strlen (s);

	 	char ans[1005];
	 	int res = (1<<29);

	 	if (len < 9) 
	 	{
	 	 	puts("unlucky");
	 	 	continue;
	 	}

	 	for (int i = 0; i+4 < len; ++i) 
	 	{
			strcpy(buf, s);
			int cnt = 0;
			if (buf[i] != 'l') buf[i] = 'l', ++cnt;	
			if (buf[i+1] != 'u') buf[i+1] = 'u', ++cnt;
			if (buf[i+2] != 'c') buf[i+2] = 'c', ++cnt;
			if (buf[i+3] != 'k') buf[i+3] = 'k', ++cnt;					
			if (buf[i+4] != 'y') buf[i+4] = 'y', ++cnt;

			for (int l = 0, r = len-1; l < r; ++l, --r)
			{
			 	if (buf[l] != buf[r])
			 	{
			 		if (i <= l && l <= i+4)  
			 		 	if (i <= r && r <= i+4)
			 		 	{
			 		 	 	cnt = (1<<29);
			 		 	 	break;
			 		 	}
			 		 	else 
			 				buf[r] = buf[l], ++cnt;
					else 
						if (i <= r && r <= i+4)
							buf[l] = buf[r], ++cnt;
						else 
						{
							char ch = min (buf[l], buf[r]);
							buf[l] = buf[r] = ch;
							++cnt; 	
						}
			 	}
			}

			if (res > cnt || (res == cnt && strcmp(ans, buf) > 0)) 
			{
				res = cnt;
				strcpy(ans, buf);
			}	
	 	}

	 	if (res == (1<<29)) 
			puts("unlucky");
		else 
			printf("%s %d\n", ans, res);
	}
	
	return 0;            
}
