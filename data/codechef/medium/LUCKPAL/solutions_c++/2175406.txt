#include<iostream>
#include<cstdio>
#include<string>
#include<algorithm>
#define INF 10000000
using namespace std;
int main(){
  string word;
  int i,len,up,c,j,k,mi,t;
  scanf("%d",&t);
  while(t--) {
   cin>>word;
   len=word.length();
   if(len < 9) {
    printf("unlucky\n");
    continue;
   }
   string ans="";
   mi=INF;
   for(i=0;i<len-4;i++){
   string p=word;
   c=0;
   if(p[i]!='l')  { p[i]='l';c++;} 
   if(p[i+1]!='u') { p[i+1]='u';c++;} 
   if(p[i+2]!='c') { p[i+2]='c';c++;} 
   if(p[i+3]!='k') { p[i+3]='k';c++;} 
   if(p[i+4]!='y') { p[i+4]='y';c++;} 

   for(j=0,k=len-1;j<=k;j++,k--) {
    if(p[j] != p[k]) {
     if(j>=i && j<=i+4){
     if(k>=i && k<=i+4) { c=INF;break;}
      else {
      p[k]=p[j];
      c++;
      }
    }
    else if(k>=i && k<=i+4){
     if(j>i && j<=i+4) { c=INF;break;}
      else { p[j]=p[k]; c++; }
    }
    else {
      char ch=min(p[j],p[k]);
      p[j]=ch;
      p[k]=ch;
     c++;
    }
   }
   }
  if(mi>c) { mi=c; ans=p;}
  else if(mi==c) {
   if(ans=="")ans=p;
   else ans=min(ans,p);
  }
  }
  if(ans=="")printf("unlucky\n");
 else cout<<ans<<" "<<mi<<endl;
 }
 return 0;
}
