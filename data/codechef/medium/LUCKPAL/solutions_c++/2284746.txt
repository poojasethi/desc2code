#define FOR(i,a,b) for(ull i=a;i<b;i++)
#define FORE(i,a,b) for(ull i=a;i<=b;i++)
#define FORD(i,a,b) for(ull i=a;i>=b;i--)

int main()
{
std::ios_base::sync_with_stdio (false);
  int t;
	cin >> t;

	while(t--)
	{
		string s;
		cin >> s;

		if (s.size() < 9)
		{
			cout << "unlucky\n";
		} else
		{
			int mi = 100000;
			string ans = "";
			for (int i = 0; i < s.size()-4; i++)
			{
				string p = s;
				int c = 0;
				if (p[i] != 'l')
				{
					p[i] = 'l';
					c++;
				}
				if (p[i+1] != 'u')
				{
					p[i+1] = 'u';
					c++;
				}
				if (p[i+2] != 'c')
				{
					p[i+2] = 'c';
					c++;
				}
				if (p[i+3] != 'k')
				{
					p[i+3] = 'k';
					c++;
				}
				if (p[i+4] != 'y')
				{
					p[i+4] = 'y';
					c++;
				}

				for (int j = 0, k = s.size()-1; j <= k; j++,k--)
				{
					if (p[j] != p[k])
					{
						if (j >= i && j <= i+4)
						{
							if (k >= i && k <= i+4)
							{
								c = 100000;
								break;
							} else
							{
								p[k] = p[j];
								c++;
							}
						} else if (k >= i && k <= i+4)
						{
							if (j >= i && j <= i+4)
							{
								c = 100000;
								break;
							} else
							{
								p[j] = p[k];
								c++;
							}
						} else
						{
							char ch = min(p[j],p[k]);
							p[j] = ch;
							p[k] = ch;
							c++;
						}
					}
				}
				if (mi > c)
				{
					mi = c;
					ans = p;
				} else if (mi == c)
				{
					if (ans == "") ans = p;
					else ans = min(ans,p);
				}
			}
			if (ans == "") cout << "unlucky\n";
			else cout << ans << " " << mi << "\n";
		}
	}
}
