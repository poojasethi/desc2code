#include<cstdio>
#include<vector>
#include<set>
#include<map>
#include<queue>
#include<algorithm>
#include<string>
#include<utility>
#include<cassert>
using namespace std;
int f[100000];
pair<int,int> chef[100000];
int main(){
	int i, j,N,K,count=0,cur,nxt;
	assert(scanf("%d",&N)==1);
	assert(scanf("%d",&K)==1);
	for(i=0;i<N;i++){
		assert( scanf("%d",f+i)==1 );
		assert( 0<f[i] && f[i] <= 1000000000 );
	}
	for(i=0;i<N;i++){
		chef[i] = make_pair(f[i], i);
	}
	sort(chef,chef+N);
	cur = 0;
	while(cur< N){
		nxt = cur+1;
		while((chef[cur].first == chef[nxt].first) && (chef[nxt].second - chef[cur].second < K) && (nxt<N)){
			nxt++;                                                                                     }
		cur=nxt;		
		count++;
	}

	printf("%d\n",count);

	return 0;
}
