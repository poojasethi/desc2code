#include <stdio.h>
#include <algorithm>
using namespace std;
typedef struct node
{
	int val;
	int idx;
}node;
bool compare(node a,node b)
{
	if(a.val<b.val)
	return true;
	else if(a.val==b.val)
	{
		if(a.idx<b.idx)
		return true;
		else
		return false;
	}
	else
	return false;
}
node A[100005];
int main()
{
	int i,N,K;
	scanf("%d %d",&N,&K);
	for(i=1;i<=N;i++)
	{
	scanf("%d",&A[i].val);
	A[i].idx=i;
    }
    sort(A+1,A+N+1,compare);
    int range=A[1].idx+(K-1);
    int ans=1;
    int val=A[1].val;
    for(i=2;i<=N;i++)
    {
    	if(val==A[i].val)
    	{
    		if(A[i].idx>range)//out of range
    		{
    			ans+=1;
    			range=A[i].idx+(K-1);
    		}
    	}
    	else
    	{
    		val=A[i].val;
    		range=A[i].idx+(K-1);
    		ans+=1;
    	}
    }
    printf("%d\n",ans);
    return 0;
}
