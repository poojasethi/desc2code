#include <algorithm>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <cstdio>
#include <climits>
#include <iostream>
#include <map>
#include <queue>
#include <string>
#include <fstream>
#include <sstream>
#include <vector>
#include <list>
#include <utility>
#include <set>
#define lend endl

using namespace std;
typedef long long ll;

map<int, bool> checked;
int interval[300000] = {0};
int n, k, j, cnt = 0;

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> k;
	
	for (int i = 0; i < n; ++i) {
		cin >> j;
		if (!checked[j]) {
			checked[j] = true;
			++cnt;
			interval[i + k - 1] = j;
		}
		checked[interval[i]] = false;
	}
	cout << cnt << lend;
}
