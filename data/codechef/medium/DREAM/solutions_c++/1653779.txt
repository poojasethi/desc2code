#include <cstdio>
#include <map>

int main()
{
    int n, k;
    scanf("%d %d", &n, &k);
    int ans = 0;
    std::map<int, int> cookedTill;
    for(int i=1; i<=n; i++) {
        int f;
        scanf("%d", &f);
        if(cookedTill[f] >= i) continue;
        else {
            cookedTill[f] = i + k - 1;
            ans++;
        }
    }
    printf("%d\n", ans);
}
