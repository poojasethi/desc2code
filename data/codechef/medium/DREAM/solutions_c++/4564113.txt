#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main() {
vector<pair<ll,int> > v;
int n,k,i,cnt=0,ctr=0,j;
ll a;
scanf("%d%d",&n,&k);
for(i=1;i<=n;i++) {
scanf("%lld",&a);
v.push_back(make_pair(a,i));
}
sort(v.begin(),v.end());
for(i=0;i<n;i++) {
ctr=1; j=i;
while(ctr<=k) {
    if(v[i].first==v[i+1].first&&v[i+1].second-v[j].second<k) {
        ctr++;
        i++;
    }
    else break;
}
cnt++;
}
printf("%d\n",cnt);
return 0;
}
