#include <cstdio>
#include <map>
using namespace std;

int time[100005];
map <int, int> pos;

int main()  {
    int N, K, i, j, cnt;
    map <int, int>::iterator it;

    scanf("%d %d", &N, &K);
    for (i = 0; i < N; i++)
        scanf("%d", &time[i]);

    for (i = 0, cnt = 0; i < N; i++)    {
        it = pos.find(time[i]);
        if (it != pos.end())    {
            if (i - it->second >= K)    {
                cnt++;
                it->second = i;
            }
        } else {
            pos[time[i]] = i;
            cnt++;
        }
        /*
        if (time[i])    {
            cnt++;
            for (j = 1; j < K; j++)
                if (i + j < N && time[i + j] == time[i])
                    time[i + j] = 0;
        }
        */
    }

    printf("%d\n", cnt);

    return 0;
}
