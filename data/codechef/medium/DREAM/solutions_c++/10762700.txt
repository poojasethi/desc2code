#include<iostream>
#include<algorithm>
using namespace std;
struct node{
	int time;
	int index;
};
struct node food[100005];
bool cmp(const struct node &a,const struct node &b)
{
	if(b.time != a.time)
		return b.time > a.time;
	else
		return b.index > a.index;
}
int main()
{
	int n,k;
	cin >> n >> k;
	for(int i=1;i<=n;i++)
	{
		int x;
		cin >> x;
		food[i].time=x;
		food[i].index=i;
	}
	sort(food+1,food+1+n,cmp);
	int i,j,count=0;
	i=1;
	while(i<=n)
	{
		count++;
		int target=food[i].time;
		int tin=food[i].index;
		while(food[i].time==target && i<=n && food[i].index-tin < k)
		{
			i++;
		}
	}
	cout << count << '\n';
	return 0;
}
