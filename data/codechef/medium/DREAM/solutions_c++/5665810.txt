#include<iostream>
#include<algorithm>
using namespace std;
struct node
{
	int v,idx;
};
node a[100005];
class func
{
	public:
	int operator ()(node a,node b)
	{
		if(a.v<b.v)
		return 1;
		else if(a.v>b.v)
		return 0;
		
		else
		{
			if(a.idx<b.idx)
			return 1;
			return 0;
		}
	}
};
int main()
{
	int n,k,i,j;
	
	scanf("%d%d",&n,&k);
	
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i].v);
		a[i].idx=i;
	}
		sort(a+1,a+n+1,func());
	
	int bit;
	int count=0;
	for(i=1;i<=n;)
	{
		bit=0;
		for(j=i;a[j].idx<=a[i].idx+k-1&&j<=n;j++)
		{
			if(a[j].v!=a[i].v)
			{
				bit=1;
				break;
			}
		}
		if(bit==0)
		{
			i=j;
		}
		else
		{
			i=j;
		}
		count++;
		
	}
	
	printf("%d\n",count);
}
