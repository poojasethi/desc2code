#include<iostream>
#include<algorithm>
using namespace std;
pair<int,int> p[100001];
int main()
{
    int n,k,i,j,f;
    cin>>n>>k;
    for(i=1;i<=n;i++)
    {
        cin>>f;
        p[i]=make_pair(f,i);
    }
    sort(p+1,p+n+1);
  //  pair<int,int>::iterator it;
  //  for(i=1;i<=n;i++)
   // {
    //    cout<<p[i].first<<" "<<p[i].second<<endl;
   // }
    i=j=1;
    int ans=0;
    while(i<=n)
    {
        j=i+1;
        while(j<=n && p[i].first==p[j].first && p[j].second-p[i].second<k)
            j++;
        i=j;
        ans++;
    }
    cout<<ans;
}
