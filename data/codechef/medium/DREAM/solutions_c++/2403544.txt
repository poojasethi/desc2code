#include <cstdio>
#include <algorithm>
using namespace std;
const int N=100000+128;
 
struct node { int n,i; node(){} bool operator < (const node &o) const { return n<o.n || (n==o.n && i<o.i); } };
node in[N];
 
int main()
{
	int n,k,i,j,ans=0,from=0,flag;
	scanf("%d%d",&n,&k);
	for(i=0;i<n;i++) { scanf("%d",&in[i].n); in[i].i=i; }
	sort(in,in+n);
	for(i=0;i<n;)
	{
		ans++;
		from=in[i].i;
		j=i+1;
		while(j<n && in[j].n==in[i].n && in[j].i-from<k) j++;
		i=j;
	}
	printf("%d",ans);
	return 0;
}
