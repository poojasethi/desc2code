vector<pair<int,int> > v;
int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0; i<n; i++)
    {
        int val;
        scanf("%d",&val);
        v.push_back(make_pair(val,i));
    }
    sort(v.begin(),v.end());
    int i=0,j=1;
    long long ans =0 ;
    while(1)
    {

        while(v[j].first == v[i].first && ((v[j].second - v[i].second + 1)<=k)) j++;
        ans++;
        i=j;
        j++;
        if(i>=n) break;
    }
    printf("%lld\n",ans);
}
