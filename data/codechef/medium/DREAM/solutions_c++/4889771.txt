int read() {int nm=0;char ch=gc(); while(ch<'0'||ch>'9') ch=gc(); while(ch>='0'&&ch<='9') {nm=(nm<<1)+(nm<<3)+ch-'0';ch=gc();} return nm;}

int main()
{
  int n,k;
  n=read();
  k=read();
  pair<int ,int > P[n+1];
  for(int i=0;i<n;i++)
  {
     int temp=read();
     P[i].first=temp;
     P[i].second=i;
    // or we can do P[i]=make_pair(temp,i);
  }
  sort(P,P+n);
  int ans=0;
  for(int i=0;i<n;)
  {
     ans++;
     int j=i+1;
     while(j<n)
     {
        if(P[i].first==P[j].first && P[j].second-P[i].second<k) j++;
        else break;
     }
     i=j;
   }
   printf("%d\n",ans);
}
