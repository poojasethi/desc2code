#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <bits/stdc++.h>
using namespace std;

int a[200005];

map<int, bool> p;

int main()
{
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int N, K, i, temp, count;
    scanf("%d %d", &N, &K);
    count = 0;
    for(i=0;i<N;i++){
        scanf("%d", &temp);
        if(!p[temp]){
            p[temp]=1;
            count++;
            a[i+K-1]=temp;
        }
        p[a[i]] = 0;
    }
    printf("%d", count);
    return 0;
}
