using namespace std;
int main(){
	int n, k, f[100000];
	cin>>n>>k;
	int ans=0;
	map<int, int> mp;
	for(int i=0; i<n; i++){
		cin>>f[i];
	}
	for(int i=0; i<n; i++){
		int time = f[i];
		if(mp.count(time)==0 || i>mp[time]){
			ans++;
			mp[time] = i+k-1;
		}
	}
	cout<<ans<<endl;
	return 0;
}
