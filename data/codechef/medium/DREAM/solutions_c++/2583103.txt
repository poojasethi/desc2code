#include <iostream>
#include <algorithm>
#include <utility>
using namespace std;

int f[100000];
pair<int,int> p[100000];

int main() {
    int i, j, n, k, res=0;
    cin >> n >> k;
    for(i=0;i<n;++i) {
        cin >> f[i];
        p[i] = make_pair(f[i],i);
    }
    sort(p,p+n);
    j=1;
    for(i=0;i<n;i=j) {
        for(j=i+1; j<n && p[i].first==p[j].first && (p[j].second-p[i].second)<k; ++j);
        ++res;
    }
    cout << res << endl;
    return 0;
}
