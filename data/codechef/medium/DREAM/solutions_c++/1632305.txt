#include <iostream>
#include <algorithm>
using namespace std;
struct item
{
  int val,ind;
};
bool cmp(item a,item b)
{
  if(a.val==b.val) {return a.ind<b.ind;}
  else return a.val<b.val;
}
int main()
{
  int n,k,i,temp,prev,cnt;
  struct item *a;
  cin >> n >> k;
  a = new item [n];
  for(i=0;i<n;i++)
  {
    cin >> a[i].val;
    a[i].ind = i+1;
  }
  sort(a,a+n,cmp);
  temp = a[0].ind;
  prev = a[0].val;
  cnt = 0;
  for(i=1;i<n;i++)
  {
//     cout << a[i].val << "\t" << a[i].ind << endl;
    if(prev!=a[i].val || a[i].ind-temp>k-1) {cnt++; temp=a[i].ind; prev=a[i].val;}
  }
  cout << cnt+1 << endl;
  return 0;
}
