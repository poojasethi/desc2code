#include <bits/stdc++.h>
using namespace std;
vector<int>adj[1000];
int par[100001];
int find(int x)
{
	if(par[x]==x)
	return x;
	return par[x]=find(par[x]);
}
void unio(int u,int v)
{
	int p_u=find(u);
	int p_v=find(v);
	if(p_u!=p_v)
	{
		par[p_u]=p_v;
	}
}
void init(int n)
{
	for(int i=0;i<n;i++)
	par[i]=i;
}
int main() {
	int t,n,m,q,x,y,u,v;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		init(n);
		while(m--)
		{
			scanf("%d%d",&u,&v);
			unio(u,v);
		}
		scanf("%d",&q);
		while(q--)
		{
			scanf("%d%d",&x,&y);
			if(find(x)==find(y))
			printf("YO\n");
			else
			printf("NO\n");
		}
	}
	return 0;
}
