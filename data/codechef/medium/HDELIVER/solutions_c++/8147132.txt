/* https://www.codechef.com/problems/HDELIVER */
LL ancestor[120];
LL root(LL a)
{
    if(a==ancestor[a])
        return a;
    else
        return root(ancestor[a]);
}
void merge(LL a , LL b)
{
    LL x= root(a);
    LL y= root(b);
    if(x==y)
        return ;
    else
        ancestor[x]=y;
}
int main()
{
    LL t,n,m,a,q,b;
    sll(t);
    while(t--)
    {
        s2ll(n,m);
        rep(i,n)
        {
            ancestor[i]=i;
        }
        rep(i,m)
        {
            s2ll(a,b);
            merge(a,b);
        }
        sll(q);
        while(q--)
        {
            s2ll(a,b);
            LL x= root(a);
            LL y= root(b);
            if(x==y)
                cout<<"YO\n";
            else
                cout<<"NO\n";
        }

    }
    return 0;
}
