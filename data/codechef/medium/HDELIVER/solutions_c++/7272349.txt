#include<stdio.h>
#include<queue>
#include<string.h>
using namespace std;
vector<int> adj[101];
queue<int> q;
bool vis[101];
int bfs(int s)
{
    vis[s]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();
        q.pop();
        int i,len=adj[x].size();
        for(i=0;i<len;i++)
        {
            if(vis[adj[x][i]]==0)
            {
                vis[adj[x][i]]=1;
                q.push(adj[x][i]);
            }
        }
    }
}
void init()
{   int i;
    for(i=0;i<101;i++)
    {
     //   adj[i].clear();
        vis[i]=0;
    }
    while(!q.empty()) q.pop();
}
int main()
{
    int t;
    //cin>>t;
    scanf("%d",&t);
    while(t--)
    {   int q1[3001],q2[3001];
        int n,m,a,b;
      //  cin>>n>>m;
        scanf("%d%d",&n,&m);
        int i;
        for(i=0;i<m;i++)
        {
            //cin>>a>>b;
            scanf("%d%d",&a,&b);
            adj[a].push_back(b);
            adj[b].push_back(a);
        }
        int qr;
      //  cin>>qr;
        scanf("%d",&qr);
        for(i=0;i<qr;i++)
        {
           // cin>>a>>b;
           scanf("%d%d",&q1[i],&q2[i]);
         //   init();
        }
        char an[3001][3];
        for(i=0;i<n;i++)
        {
            if(vis[i]==0)
            {
                bfs(i);
                for(int j=0;j<qr;j++)
                {   if(!i) strcpy(an[j],"YO");
                    if(vis[q1[j]]!=vis[q2[j]]) strcpy(an[j],"NO");
                }
            }
        }
        for(i=0;i<101;i++)
        {
            adj[i].clear();
            vis[i]=0;
        }
        while(!q.empty()) q.pop();
        for(i=0;i<qr;i++)
        {
            printf("%s\n",an[i]);

        }
    }
    return 0;
}
