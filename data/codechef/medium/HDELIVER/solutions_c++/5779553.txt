#include <iostream>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cmath>
#include <stack>
#include <vector>
#define ll 1000000007;
using namespace std;
int main()
{
	int t,n,m,a,b,p;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		vector<int> g[n];
		while(m--){scanf("%d %d",&a,&b); g[a].push_back(b); g[b].push_back(a);}
		int comp[n];
		for(int i=0;i<n;i++)comp[i]=0;
		int num=1;
		stack<int> q;
		for(int i=0;i<n;i++)
		{
			if(comp[i]!=0)continue;
			q.push(i);
			comp[i]=num;
			while(!q.empty())
			{
				int k = q.top();
				q.pop();
				for(int j=0;j<g[k].size();j++)
				{
					int l = g[k][j];
					if(comp[l]!=0)continue;
					comp[l]=num;
					q.push(l);
				}
			}
			num++;
		}
		scanf("%d",&p);
		while(p--)
		{
			scanf("%d %d",&a,&b);
			if(comp[a]==comp[b])printf("YO\n");
			else printf("NO\n");
		}
	}
}
