vector<int> adj[101];
queue<int> q;
bool vis[101][101];
int bfs(int s,int p)
{
    vis[s][p]=1;
    q.push(s);
    while(!q.empty())
    {
        int x=q.front();
        q.pop();
        int i,len=adj[x].size();
        for(i=0;i<len;i++)
        {
            if(vis[adj[x][i]][p]==0)
            {
                vis[adj[x][i]][p]=1;
                q.push(adj[x][i]);
            }
        }
    }
}
