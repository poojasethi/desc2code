#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
int p[100];
int rank[100];

int findset (int x)
{
if(x!=p[x]) p[x] = findset(p[x]);
return p[x];
}


int merge (int x, int y)
{
int px = findset (x);
int py = findset (y);
if(rank[px] >= rank[py]) rank[px]+=rank[py],p[py] = px;
else rank[py]+=rank[px],p[px] = py;
}

int main()
{
int t;
scanf("%d",&t);
while(t--)
{
int n,m,i,x,y;
scanf("%d%d",&n,&m);
for(i=0;i<n;i++) {p[i] = i; rank[i] = 1;}

for(i=0;i<m;i++)
{
scanf("%d%d",&x,&y);
if(findset (x) != findset (y)) merge(x,y);
}

int q;
scanf("%d",&q);
for(i=0;i<q;i++)
{
scanf("%d%d",&x,&y);
if(findset (x) == findset (y)) printf("YO\n");
else printf("NO\n");
}


}

return 0;
}
