#include<iostream>
#include<cstdio>
using namespace std;
inline void inp(int &n ) {
     int ch=getchar();n=0;
     while(  ch >= '0' && ch <= '9' ) n = (n<<3)+(n<<1) + ch-'0', ch=getchar();	
}
int m[200];
int findroot(int n)
{
	while(true)
	{
	if (m[n]==n)break;
	else
	n=m[n];
	}
	return n;
}
int main()
{
ios_base::sync_with_stdio(false);
int t;
inp(t);
while(t--)
{
	
	int n,s;
	inp(n);inp(s);
	for(int i=0;i<n;i++)
        m[i]=i;
	for(int i=0;i<s;i++)
	{
		int a,b,roota,rootb;
		inp(a);inp(b);
		roota=findroot(a);
		rootb=findroot(b);
		if(roota<rootb)
			m[rootb]=roota;
		else
			m[roota]=rootb;			
	}
	int q;
	inp(q);
	for(int i=0;i<n;i++)
		m[i]=findroot(i);
	for(int i=0;i<q;i++)
	{
	int a,b;
	inp(a);inp(b);
	if(m[a]==m[b])
	                printf("%s\n","YO");
            else
               printf("%s\n","NO");	
        }
}
return 0;
		
}
