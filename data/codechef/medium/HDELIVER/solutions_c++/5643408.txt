#include <stdio.h>
#include <math.h>
#include <iostream>
#include <algorithm>
using namespace std;
int color[105];
int thiscolor=0;
bool fast[105][105];
bool dfs(int i,int n){
	if(color[i]!=-1) return false;
	color[i]=thiscolor;
	for(int j=0;j<n;j++){
		if(fast[i][j] && i!=j){
			dfs(j,n);
		}
	}
	return true;
}

int main(){
	int t,n,m;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d",&n,&m);
		int a,b;
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			fast[i][j]=false;
			
		for(int i=0;i<m;i++){
			scanf("%d %d",&a,&b);
			if(a==b) continue;
			fast[a][b]=1;
			fast[b][a]=1;
		}
		for(int i=0;i<n;i++) {color[i]=-1;}
		for(int i=0;i<n;i++) {if(dfs(i,n)){thiscolor++;}}
		int q;scanf("%d",&q);
		while(q--){
			int a,b;
			scanf("%d %d",&a,&b);
			if(color[a]==color[b]) printf("YO\n");
			else printf("NO\n");
		}
	}
}
