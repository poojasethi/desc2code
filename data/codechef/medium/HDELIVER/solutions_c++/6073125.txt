#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int main(){
  int t,n,m,a,b,q,x,y;
  bool mat[102][102];
  scanf("%d",&t);
  while(t--){
    memset(mat,false,sizeof mat);
    scanf("%d %d",&n,&m);
    for(int i=0;i<m;i++){
      scanf("%d %d",&a,&b);
      mat[a][b]=mat[b][a]=true;
    }
    for(int k=0;k<n;k++){
      for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
           mat[i][j]|=(mat[i][k]&mat[k][j]);
        }
      }
    }
    scanf("%d",&q);
    for(int i=0;i<q;i++){
      scanf("%d %d",&x,&y);
      if(mat[x][y] || x==y){
        printf("YO\n");
      }
      else printf("NO\n");
    }
  }
  return 0;
}
