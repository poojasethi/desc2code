#include <stdio.h>
int T, N, M, c, x, y, i, Q;
int nodes[120];
inline int scan()
{
    int n = 0;
    int ch = getchar_unlocked();
    while(ch>='0' && ch<='9')
    {
        n=(n*10)+ch-48;
        ch = getchar_unlocked();
    }
    return n;
}
inline void YES(){putchar_unlocked('Y'),putchar_unlocked('O'),putchar_unlocked('\n');}
 
inline void NO(){putchar_unlocked('N'),putchar_unlocked('O'),putchar_unlocked('\n');}
 
int main() {
	T = scan();
	while(T--) {	
		N = scan();
		M = scan();
		for(i = 0; i < N; i++) {
			nodes[i] = i;
		}
 
		while (M--) {
			x = nodes[scan()];
			y = nodes[scan()];
			if (x != y) {
				for (i = 0 ; i < N; i++) {
					if (nodes[i] == x) {
						nodes[i] = y;
					}
				}
			}
		}
 
		Q = scan();
		for(i = 0; i < Q; i++) {
			x = nodes[scan()];
			y = nodes[scan()];
			if (x == y) YES();
			else NO();
		}
	}
	return 0;
}
