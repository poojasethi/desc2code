#include <bits/stdc++.h>

using namespace std;

class UnionFindDisjointSet {
private:
    vector<int> par, ran;
public:
    UnionFindDisjointSet(int n) {
        par.assign(n, 0);
        ran.assign(n, 0);
        for(int i = 0; i < n; i++)
            par[i] = i;
    }

    int findSet(int x) {
        return (par[x] == x) ? x : par[x] = findSet(par[x]);
    }

    bool isSameSet(int x, int y) {
        return findSet(x) == findSet(y);
    }

    void unionSet(int x, int y) {
        if(!isSameSet(x, y)) {
            int parX = par[x], parY = par[y];
            if(ran[parX] > ran[parY]) par[parY] = parX;
            else {
                par[parX] = parY;
                if(ran[parX] == ran[parY])
                    ran[parY]++;
            }
        }
    }
};

int main() {
    int t;
    scanf("%d", &t);
    while(t--) {
        int n, m;
        scanf("%d %d", &n, &m);
        UnionFindDisjointSet ds(n);
        for(int i = 0; i < m; i++) {
            int a, b;
            scanf("%d %d", &a, &b);
            ds.unionSet(a, b);
        }
        int q;
        scanf("%d", &q);
        for(int i = 0; i < q; i++) {
            int x, y;
            scanf("%d %d", &x, &y);
            if(ds.isSameSet(x, y)) printf("YO\n");
            else printf("NO\n");
        }
    }
    return 0;
}
