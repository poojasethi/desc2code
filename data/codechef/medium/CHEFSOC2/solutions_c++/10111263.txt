#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007

long int pss[1111],dp[1111][1111];




int main()
{
	long int t,n,m,s,i,j,k,l;
	scanf("%ld",&t);
	while(t--)
	{
		scanf("%ld %ld %ld",&n,&m,&s);
		for(i=1;i<=m;i++)
		{
		     scanf("%ld",&pss[i]);
		}
		
		for(i=1;i<=1005;i++)
		{
			for(j=1;j<=1005;j++) dp[i][j]=0;
		}
		
		
		
		for(j=1;j<=m+1;j++)
		{
		    for(i=1;i<=n;i++)
		    {
		    	if(j==1) 
		    	{
		    		dp[i][j]=(i==s);
		    		continue;
				}
		    	long int result=0;
			    if(i-pss[j-1]>=1) result=(result%mod+dp[i-pss[j-1]][j-1]%mod)%mod;
			    if(i+pss[j-1]<=n) result=(result%mod+dp[i+pss[j-1]][j-1]%mod)%mod;
			    dp[i][j]=result;
		    }
	    
	        
	    }
//		
//		for(i=1;i<=n;i++)
//		{
//			for(j=1;j<=m+1;j++) 
//			{
//				printf("%d ",dp[i][j]);
//				
//			}
//			printf("\n");
//		}
//		
//		printf("\n\n");
		for(i=1;i<=n;i++) 
		{
			if(i>1) printf(" "); 
		    printf("%ld",dp[i][m+1]);
		}
		printf("\n");
	}
	
}
