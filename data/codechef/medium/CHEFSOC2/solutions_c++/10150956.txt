#include <stdio.h>
#include <iostream>
#include <malloc.h>

using namespace std;

int MOD = 1000000007;

int main(){
	int no_of_cases,number_of_dogs,number_of_passes,start_from,pass_strength,result;
	cin >> no_of_cases;
	while(no_of_cases--){
		cin >> number_of_dogs >> number_of_passes >> start_from;
		int *prev = (int *)malloc (sizeof(int)*number_of_dogs);
		int *next = (int *)malloc (sizeof(int)*number_of_dogs);
		start_from--;
		for(int i=0;i<number_of_dogs;i++){
			next[i]=(i==start_from);
		}

		while(number_of_passes--){
			cin >> pass_strength;
			for(int i=0;i<number_of_dogs;i++)
				prev[i]=next[i];
			
			for(int i=0;i<number_of_dogs;i++){
				result=0;
				if (i-pass_strength>=0)
					result+=prev[i-pass_strength];
				if(i+pass_strength<number_of_dogs)
					result+=prev[i+pass_strength];
				if (result>=MOD)
					result = result-MOD;
				next[i]=result;
			}
		
		}

		for (int i=0;i<number_of_dogs;i++)
			printf("%d ",next[i]);
		printf("\n");
	}


	return 0;
}
