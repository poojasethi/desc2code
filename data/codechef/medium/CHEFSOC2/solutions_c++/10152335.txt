#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define INF 0x3f3f3f3f
using namespace std;

int n,m,s,a[1005];
ll dp[1005][1005];
ll solve(int i,int m)
{
    if(i<=0||i>n)
        return 0;
    if(m==0)
    {
        if(i==s)
            return 1;
        else
            return 0;
    }
    if(dp[i][m]!=-1)
        return dp[i][m];
    ll ways=0;
    ways+=solve(i-a[m],m-1);
    ways%=MOD;
    ways+=solve(i+a[m],m-1);
    ways%=MOD;
    return dp[i][m]=ways;

}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        memset(dp,-1,sizeof(dp));
        scanf("%d%d%d",&n,&m,&s);
        for(int i=1;i<=m;i++)
        {
            scanf("%d",&a[i]);
        }
        for(int i=1;i<=n;i++)
        {
            printf("%lld ",solve(i,m));
        }
        printf("\n");
    }
    return 0;
}
