#include<bits/stdc++.h>

using namespace std;

#define mod 1000000007

int check(long long int x,long long int n) {
    if((x>=1)&&(x<=n))
        return 1;
    return 0;
}
int main() {
    long long int t,n,m,s,f,i,j,k;
    scanf("%lld",&t);
    while(t--) {
        scanf("%lld%lld%lld",&n,&m,&s);
        vector<long long int> arr(m);
        for(i=0;i<m;i++) {
            scanf("%lld",&arr[i]);
        }

        vector<long long int> t1(n+1,0);
        vector<long long int> t2(n+1,0);
        t1[s]=1;

        for(i=1;i<=m;i++) {
            for(j=1;j<=n;j++) {
//                cout<<"t1s="<<t1[s]<<"\n";
//                cout<<"j="<<j<<" "<<t1[j]<<"\n";
                if(t1[j]) {
//                    cout<<j<<"\n";
                    if(check(j+arr[i-1],n)) {
//                            cout<<"xfgdfg\n";
                        t2[j+arr[i-1]]=(t2[j+arr[i-1]]+t1[j])%mod;
                    }
                    if(check(j-arr[i-1],n)) {
//                        cout<<"etyyuy\n";
                        t2[j-arr[i-1]]=(t2[j-arr[i-1]]+t1[j])%mod;
                    }
                }
//                t1=t2;

            }
            t1=t2;
            for(k=1;k<=n;k++) {
//                    t1[k]=t2[k];
                    t2[k]=0;
                }
        }

        for(i=1;i<=n;i++) {
            printf("%lld ",t1[i]);
        }
        printf("\n");
    }
    return 0;
}
