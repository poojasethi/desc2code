#include<bits/stdc++.h>
using namespace std;

int main(){
	int t,n,m,s,*a;
	long long int mod=1000000007;
	long long int f[1001][1001];
	scanf("%d",&t);
	while(t--){
		scanf("%d %d %d",&n,&m,&s);
		a=new int[m];
		s--;
		for(int i=0;i<m;i++){
			scanf("%d",&a[i]);
		}
		for(int i=0;i<1001;i++){
			for(int j=0;j<=1000;j++){
				f[i][j]=0;
			}
		}
		f[0][s]=1;
		//now execute m passes
		for(int i=0;i<m;i++){
			for(int j=0;j<n;j++){
				if(j+a[i]<n) f[i+1][j+a[i]]=(f[i][j]+f[i+1][j+a[i]])%mod;
				if(j-a[i]>=0) f[i+1][j-a[i]]=(f[i][j]+f[i+1][j-a[i]])%mod;
			}
		}
		for(int i=0;i<n;i++){
			cout<<f[m][i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
