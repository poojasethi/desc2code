#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int i,j,k,l,n,m,t,s,arr[100005],dp[1005][1005];
int main() 
{
	cin>>t;
	while(t--)
	{
		cin>>n>>m>>s;
		for(i=0;i<(m+1);i++)
		{
			for(j=0;j<(n+1);j++)
				dp[i][j]=0;
		}
		for(i=0;i<m;i++)
			cin>>arr[i];
		dp[0][s]=1;
		for(i=0;i<m;i++)
		{
			for(j=1;j<=n;j++)
			{
				if(dp[i][j]>0)
				{
					int l1=j-arr[i];
					int l2=j+arr[i];
					if(l1>0 && l1<=n)
					{
						dp[i+1][l1]+=dp[i][j];
						dp[i+1][l1]%=MOD;
					}
					if(l2>0 && l2<=n)
					{
						dp[i+1][l2]+=dp[i][j];
						dp[l+1][l2]%=MOD;
					}
				}
			}
		}
		for(i=1;i<=n;i++)
			cout<<dp[m][i]<<" ";
		cout<<endl;
	}
	return 0;
}
