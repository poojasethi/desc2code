#include<iostream>
using namespace std;

#define MOD 1000000007

int a[1008];

int main()
{
	int tc,N,M,i,j,s;
	
	cin>>tc;
	
	while(tc--)
	{
	   cin>>N>>M>>s;
	   
	   for(i=1; i<=M; i++)
	    cin>>a[i]; 
	
       int ans[M+1][N+1];
	   
	   for(i=1;i<=N; i++)
	   {
	   	  if(i!=s)
	   	    ans[0][i]=0;
	   	  else
			 ans[0][i]=1;  
	   } 
	
	   for(i=1; i<=M; i++)
	   {
	   	   for(j=1; j<=N; j++)
	   	   {
	   	   	   if(j-a[i]>=1 && j+a[i]<=N)
	   	   	     ans[i][j]=(ans[i-1][j-a[i]]+ans[i-1][j+a[i]])%MOD;
	   	   	   else  if(j-a[i]<1 && j+a[i]<=N)
	   	   	     ans[i][j]=ans[i-1][j+a[i]];
	   	   	    else  if(j-a[i]>=1 && j+a[i]>N)  
	   	   	      ans[i][j]=ans[i-1][j-a[i]];
	   	   	    else
	   	   	      ans[i][j]=0;
	   	   }
	   } 
	
	   for(i=1; i<=N; i++)
	    cout<<ans[M][i]<<" ";
	    cout<<endl;
	}
return 0;	
}
