#include<stdio.h>
long long int arr[1005][1005]={0};
int main()
{
    int n,m,t,i,j,k,s;

    int ps[1005];

    scanf("%d",&t);

    while(t--)
    {
        scanf("%d %d %d",&n,&m,&s);

        for(int i=0;i<m;i++)
        {
            scanf("%d",&ps[i]);
        }

        for(int i=0;i<n;i++)
        {
            if(i==s-1)
            arr[i][0]=1;

            else
            arr[i][0]=0;

            //printf("%lld %d 0\n",arr[i][0],i);
        }

        for(int j=1;j<=m;j++)
        {
            for(int i=0;i<n;i++)
            {
                if(i-ps[j-1]>=0)
                arr[i][j]+=arr[i-ps[j-1]][j-1];

                if(i+ps[j-1]<n)
                arr[i][j]+=arr[i+ps[j-1]][j-1];

                if(arr[i][j]>=1000000007)
                arr[i][j]=arr[i][j]%1000000007;

                //printf("%lld %d %d\n",arr[i][j],i,j);
            }
        }

        for(int i=0;i<n;i++)
        {
            printf("%lld ",arr[i][m]);
        }
        printf("\n");

        for(int i=0;i<n;i++)
        {
            for(int j=0;j<=m;j++)
            {
                arr[i][j]=0;
            }
        }
    }
    return 0;
}
