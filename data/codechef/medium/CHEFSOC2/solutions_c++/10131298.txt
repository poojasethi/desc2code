#include<stdio.h>
#include<iostream>
#define MOD 1000000007
using namespace std;

bool isValid(int N,int pos){
    if(pos>=0&&pos<N)
        return true;
    else
        return false;
}

int main(){

    int t;
    cin>>t;
    while(t--){
        int numDogs;
        int numPass;
        int ballOnDog;
        cin>>numDogs>>numPass>>ballOnDog;
        ballOnDog--;
        int arr[numPass+1];


        for(int i=0;i<numPass;i++){
            cin>>arr[i];

        }
        int dp[numDogs+1][numPass+1];
        for(int i=0;i<=numDogs;i++){
            for(int j=0;j<=numPass;j++){
                dp[i][j]=0;
            }
        }

        if(isValid(numDogs,ballOnDog-arr[0])){
            dp[ballOnDog-arr[0]][0]=1;
        }

        if(isValid(numDogs,ballOnDog+arr[0])){
            dp[ballOnDog+arr[0]][0]=1;
        }
        for(int i=1;i<numPass;i++){
            for(int j=0;j<numDogs;j++){

               if(isValid(numDogs,j-arr[i])){
                    dp[j][i]=(dp[j][i]+dp[j-arr[i]][i-1])%MOD;
               }
                if(isValid(numDogs,j+arr[i])){
                    dp[j][i]=(dp[j][i]+dp[j+arr[i]][i-1])%MOD;
               }
            }
        }

        for(int i=0;i<numDogs;i++){
            cout<<dp[i][numPass-1]<<" ";
        }
        cout<<endl;

    }
    return 0;
}
