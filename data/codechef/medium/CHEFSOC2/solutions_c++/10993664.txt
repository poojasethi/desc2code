#include <bits/stdc++.h>

using namespace std;

#define ll				long long int
#define pb				push_back
#define mp				make_pair
#define MOD				1000000007
#define F				first
#define S				second

ll pow_mod(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a);
		a = (a * a);
		b >>= 1;
	}
	return res;
}


ll arr[1005][1005];		
ll pass[1005];

int main(){

	ios_base::sync_with_stdio(0);

	int t;
	cin >> t;

	while(t--){

		ll n,m,s;
		cin >> n >> m >> s;

		for(int i=1;i<=m;i++)
			cin >> pass[i];

		memset(arr, 0, sizeof(arr[0][0])*1005*1005);
		arr[s-1][0] = 1;

		for(int i = 1;i<=m;i++){
			for(int j=0;j<n;j++){
				if(j+pass[i] < n)
					arr[j][i] = (arr[j][i] + arr[j+pass[i]][i-1])%MOD;
				if(j-pass[i] >= 0)
					arr[j][i] = (arr[j][i] + arr[j-pass[i]][i-1])%MOD;				
			}
		}

		for(int i=0;i<n;i++)
			cout << arr[i][m] << " ";
		cout << "\n";
		
	}

	return 0;
}
