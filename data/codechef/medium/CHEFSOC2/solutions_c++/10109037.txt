#include<iostream>
using namespace std;
#define k 1000000007
int main()
{
	unsigned int * ao,*an,*t;
	int T,N,M,s,i,j,x;
	cin>>T;
	while(T--)
	{
		cin>>N>>M>>s;
		ao=new unsigned int[N];
		an=new unsigned int[N];
		for(i=0;i<N;i++)
		{
			ao[i]=0;
			an[i]=0;
		}
		ao[s-1]=1;
		for(i=0;i<M;i++)
		{
			cin>>x;
			for(j=0;j<N;j++)
			{
				if(ao[j]>0)
				{
					if((j-x)>=0)an[j-x]=(an[j-x]%k+ao[j]%k)%k;
					if((j+x)<(N))an[j+x]=(an[j+x]%k+ao[j]%k)%k;
					ao[j]=0;
				}
			}
			t=an;
			an=ao;
			ao=t;
		}
		for(i=0;i<N;i++)cout<<ao[i]<<" ";
		cout<<endl;
	}
	return 0;
}
