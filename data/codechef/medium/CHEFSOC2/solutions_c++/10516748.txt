#include <bits/stdc++.h>
#define MOD 1000000007LL
#define pb push_back
#define mp make_pair
#define X first
#define Y second

typedef long long LLI;
using namespace std;

int a[1004],s,n;
LLI dp[1004][1004];


LLI mod(LLI a) {

    if(a>=MOD)
        return a%MOD;
    return a;
}

LLI fun(int i, int p) {

    if(i>n || i<1)
        return 0LL;
    if(!p) {
        if(i==s)
            return 1LL;
        else
            return 0LL;
    }

    if(dp[i][p] + 1)
        return dp[i][p];

    return dp[i][p] = mod(fun(i+a[p], p-1) + fun(i-a[p], p-1));
}


int main() {

    int t,m;
    scanf("%d",&t);
    while(t--) {

        memset(dp, -1, sizeof(dp));
        scanf("%d%d%d",&n,&m,&s);
        for(int i=1; i<=m; i++)
            scanf("%d",&a[i]);

        for(int i=1; i<=n; i++)
            printf("%lld ",fun(i, m));

        printf("\n");

    }


    return 0;

}
