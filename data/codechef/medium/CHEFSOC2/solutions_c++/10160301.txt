#include <bits/stdc++.h>

using namespace std;

#define mod 1000000007

int t,n,m,l,ans[1005][1005],s,d,i;
vector<int>v;

int main()
{
 cin>>t;
 while(t--)
 {
     memset(ans,0,sizeof(ans));
     v.clear();
     cin>>n>>m>>s;
     for(i=0;i<m;i++)
     {
         cin>>l;
         v.push_back(l);
     }

for(i=0;i<=m;i++)
{
    for(int j=1;j<=n;j++)
    {
        if(i==0)
        {
            if(s==j)
            ans[i][j]=1;
       }
        else{
               int result=0;
         if(j-v[i-1]>0)
            result+=ans[i-1][j-v[i-1]];
       if(j+v[i-1]<=n)
{
      result+=ans[i-1][j+v[i-1]];
}
ans[i][j]=result%mod;
}
    }
}
       for(int j=1;j<=n;j++)
       {
           cout<<ans[m][j]<<" ";
       }
       cout<<"\n";

 }
    return 0;
}
