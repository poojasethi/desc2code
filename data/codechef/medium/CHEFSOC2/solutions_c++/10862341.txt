//Dead dog walking//

#include<bits/stdc++.h>

using namespace std;

#define MOD 1000000007

void solve(int N,int M,int s)
{
	int dp[1001][1001];
	int A[1001];

	for(int i = 0;i < M;i++)
	{
		cin >> A[i];
	}


	for(int i = 0;i < N;i++)
	{
		dp[0][i] = 0;
	}

	dp[0][s-1] = 1;

	for(int i = 0;i < M;i++)
	{
		for(int j = 0;j < N;j++)
		{
			long long int ans = 0;
			if((j-A[i]) >= 0)
			{
				ans += dp[i][j-A[i]];
			}

			if((j+A[i]) <= N-1)
			{
				ans += dp[i][j+A[i]];
			}
			dp[i+1][j] = ans%MOD;
		}
	}

	for(int i = 0;i < N;i++)
	{
		cout << dp[M][i] << " ";
	}
	cout << endl;
}

int main()
{
	int t;

	cin >> t;

	while(t--)
	{
		int N;
		int M;
		int s;


		cin >> N;
		cin >> M;
		cin >> s;

		solve(N,M,s);
	}
	return 0;
}
