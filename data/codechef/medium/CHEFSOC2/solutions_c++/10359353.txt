#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n,m,s;
        int *arr;
        cin>>n>>m>>s;
        arr = new int[m+1];
        for(int i=1;i<=m;i++)
            cin>>arr[i];
        int **dp = new int*[m+1];
        for(int i=0;i<=m;i++)
            dp[i] = new int[n+1];
        for(int i =1;i<=m;i++){
            for(int j=1;j<=n;j++){
                dp[i][j] = 0;

            }
        }
        if(s+arr[1] <= n)
            dp[1][s+arr[1]] = 1;
        if(s-arr[1]>=1)
            dp[1][s-arr[1]] = 1;
        for(int i=2;i<=m;i++){
            for(int j=1;j<=n;j++){
                if(j+arr[i] <= n)
                    dp[i][j] = (dp[i][j] + dp[i-1][j+arr[i]])%1000000007;
                if(j-arr[i]>=1)
                    dp[i][j] = (dp[i][j] + dp[i-1][j-arr[i]])%1000000007;
            }
        }
        for(int j=1;j<=n;j++)
            cout<<dp[m][j]<<" ";
        cout<<"\n";
    }

}
