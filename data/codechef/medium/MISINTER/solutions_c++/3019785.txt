#include <stdio.h>
#include <cstring>
typedef unsigned long long ull;
const int MOD=1000000007;
int a[100001];

ull solve(int N)
{
	ull res=1;
	memset(a,0,sizeof(int)*(N+1));
	if(N&1) N--,res=26;
	for (int i = 1; i <= N; i++)
	{
		if(a[i]==0)
		{
			res=(res*26)%MOD;
			int j=i;
			while (a[j]==0)
			{
				a[N-j+1]=2;
				a[j]=1;
				if(j&1)
				{
					j=(N>>1)+((j+1)>>1);
				}
				else
				{
					j>>=1;
				}
			}
			if(a[j]==1) res=(res*26)%MOD;
		}
	}	
	return res;
}
int main()
{
	int T,N;
	scanf("%d",&T);
	while (T--)
	{
		scanf("%d",&N);
		printf("%llu\n",solve(N));
	}
}
