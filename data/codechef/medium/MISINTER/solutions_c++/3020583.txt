#include<stdio.h>
int main()
{
	
	int i,x,a[100500],b[100500],t,n;
	scanf("%d",&t);
	while(t--)
	{
		long long int ans=1;
		int count=0,p=1;
		scanf("%d",&n);
	for(i=0;i<=n;i++)
	b[i]=0;
	a[0]=2;
	for(i=0;i<=n/2;i++)
	a[i]=2*i;
	for(i=(n/2)+1;i<=n;i++)
      	{
	     a[i]=p;
	     p=p+2;
        }
        for(i=1;i<=n;i++)
            {
			  if(b[a[i]]==0)
			  {
				b[a[i]]++;
				x=a[i]; 
				while(b[x]<=1)
				  {
					x=a[x];
					b[x]++;  
				  }
				  count++;
			  }
			  
			}
			for(i=1;i<=count;i++)
			{
				ans=ans*26;
			ans=ans%1000000007;
		    }
		    printf("%lld\n",ans);
    }
    
	return 0;
}
