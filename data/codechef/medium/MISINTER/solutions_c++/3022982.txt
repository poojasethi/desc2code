#include<iostream>
//#include<conio.h>
#define c 1000000007
using namespace std;
int a[100002];
int main()
{
    int t,x=1,i,j,k,n;
    long long int prod;
    cin>>t;
    while(x<=t)
    {
               cin>>n;
               if(n%2!=0)
               {
                         n--;
                         prod=26;
               }
               else
               {
                   prod=1;
               }
               for(i=1;i<=n/2;i++)
               {
                                  a[i]=2*i;
               }
               for(j=1;i<=n;i++,j+=2)
               {
                                     a[i]=j;
               }
               int count0=0;
               k=2;
               while(count0!=n)
               {
                               if(a[i]!=0)
                               {
                                          prod=(prod*26)%c;
                               }
                               while(a[i])
                               {
                                          j=a[i];
                                          a[i]=0;
                                          count0++;
                                          i=j;
                               }
                               i=k;
                               k++;
               }
               cout<<prod<<endl;
               x++;
    }
    //getch();
    return(0);
}
