#include<iostream>
#include <stdio.h>
#define N 100000
#define mod 1000000007
using namespace std;

int main()
{
    int t,n;
    cin>>t;
    int arr[N+1],in_cycle[N+1];
    while(t--)
    {
        cin>>n;
        int x=1,i,j,no_of_cycle=0;
        for(i=2;i<=n;i+=2)
           arr[x++]=i;
        for(i=1;i<=n;i+=2)
           arr[x++]=i;

        for(i=1;i<=n;i++)
           in_cycle[i]=0;
        for(i=1;i<=n;i++)
        if(!in_cycle[i])
        {
            no_of_cycle++;
            j=i;
            while(!in_cycle[j])
            {
                in_cycle[j]=1;
                j=arr[j];
            }
        }
    //    printf("%d",no_of_cycle);
        long long ans=1;
        while(no_of_cycle--)
            ans=(ans*26)%mod;
        printf("%lld\n",ans);
    }
}
