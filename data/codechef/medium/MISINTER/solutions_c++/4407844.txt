#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define mod 1000000007
using namespace std;
long long int epow(int a,int n)
{
	if(n==0) return 1;
	long long res=epow(a,n/2);
	res=(res*res)%mod;
	if(n&1)
	res=(res*a)%mod;
	return res;
}
main()
{   std:ios_base::sync_with_stdio(false);
    int t,n;
    long long ans=0;
    cin>>t;
    while(t--)
    {
      	cin>>n;
      	int i,j=0,cnt=0;
        int p[n+1],mark[n+1];
		for(i=2;i<=n;i+=2)
		  p[++j]=i;
	    for(i=1;i<=n;i+=2)
		  p[++j]=i;
		    
	    for(i=1;i<=n;i++) mark[i]=0;
	    
	    for(i=1;i<=n;i++)
	    {
	    	if(!mark[i])
	    	{
	    		cnt++;
	    		for(j=i;!mark[j];j=p[j])
	    		mark[j]=1;
	    	}
	    }
    	ans=epow(26,cnt)%mod;
    	printf("%lld\n",ans);
    }
	return 0;
}
