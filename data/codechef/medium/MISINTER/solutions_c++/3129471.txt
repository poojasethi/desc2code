#include <iostream>
#include <vector>
using namespace std;

inline void getCycle(int start,int n,vector<bool>& used){
	int sub = ((n&1)==0)?-1:0;
	int next=start;
	used[next]=true;
	//cout<<next<<" ";
	next = (2*next+1);
	if(next>=n){
		next %=n;
		next+=sub;
	}
	while(next!=start){
		used[next]=true;
		//cout<<next<<" ";
		next = (2*next+1);
		if(next>=n){
			next %=n;
			next+=sub;
		}
	}
}


int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		vector<bool> used(n,false);
		int sub = ((n&1)==0)?-1:0;
		long long res=1;
		for(int i=0;i<n;++i){
			if(!used[i]){
				getCycle(i,n,used);
				res *=26;
				res %= 1000000007;
			}
		}
		cout<<res<<"\n";
	}
	return 0;
}
