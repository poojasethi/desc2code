#include<iostream>

#define MOD 1000000007
#define mul(a, b) (((a) * (b)) % MOD)
using namespace std;

typedef unsigned long long bigint;

int d[100005];

int root(int i) {
	while(d[i] != i) {
		i = d[i];
	}
	return i;
}

int main() {
	int testCases;
	cin >> testCases;

	while(testCases--) {
		int n;
		bigint ans = 1;
		cin >> n;
		
		for(int i = 0; i <= n; i++) d[i] = i;

		for(int i = 1; i <= n; i++) {
			int j = (i * 2 > n)? (2*(i - 1 - (n/2)) + 1): 2*i;
			int ri = root(i);
			int rj = root(j);
			d[j] = ri;
			d[rj] = ri;
			d[i] = ri;
		}

		for(int i = 1; i <= n; i++) {
			if(d[i] == i) ans = mul(ans, 26);
		}

		cout << ans << endl;
	}
	return 0;
}
