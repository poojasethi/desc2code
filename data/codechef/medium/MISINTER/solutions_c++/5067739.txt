#include<stdio.h>
#include<string.h>
int mod = 1000000007;
int arr[100001];
bool indexx[100001];
int main()
{
    #ifndef ONLINE_JUDGE
    freopen("file.txt","r",stdin);
    #endif
    int T,N,cycle_count,curr_ind,c;
    scanf("%d",&T);
    while(T--)
    {
        cycle_count = 0,c = 1;
        scanf("%d",&N);
        memset(indexx,0,(N+1)*sizeof(bool));
        for(int i=2;i<=N;i+=2)
            arr[c++] = i;
        for(int i=1;i<=N;i+=2)
            arr[c++] = i;
        for(int i=1;i<=N;i++)
        {
            if(indexx[i] == 0)
            {
                    curr_ind = i;
                    indexx[curr_ind] = 1;
                    cycle_count++;
                    while(arr[curr_ind]!=i)
                    {
                        curr_ind = arr[curr_ind];
                        indexx[curr_ind] = 1;
                    }
            }
        }
        long long ans = 26;
        for(int i = 0;i<cycle_count-1;i++){
            ans*=26;
            if(ans>=mod)
                ans %= mod;
        }
        printf("%lld\n",ans);

    }
    return 0;
}
