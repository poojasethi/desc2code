#include<stdio.h>
#define max 1000000007
int main()
{
	int t,n,cnt,pos;
	int *a,*b,*mark;
	scanf("%d", &t);
	while(t--)
	{
		cnt=0;
		pos=1;
		scanf("%d", &n);
		if(n<=2 || n==4)
		{
			printf("%d\n",1);
			continue;
		}
		mark=new int[n+1];
		b=new int[n+1];
		for(int i=1;i<=n;i++)
			mark[i]=0;
		for(int i=2;i<=n;i+=2)
			b[pos++]=i;
		for(int i=1;i<=n;i+=2)
			b[pos++]=i;
		for(int i=1;i<=n;i++)
		{
			if(!mark[i])
			{
				mark[i]=1;
				cnt++;
				int j=b[i];
				while(!mark[j])
				{
					mark[j]=1;
					j=b[j];
				}
			}
		}
		unsigned long long res=1;
		for(int i=1;i<=cnt;i++)
			res=((unsigned long long)res*26)%max;
		printf("%llu\n", res);	
	}
	
	return 0;
}
