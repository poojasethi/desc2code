#include<stdio.h>
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int n;
		scanf("%d", &n);
		bool odd = (n%2!=0);
		if(odd)
			n = n-1;
		int i;
		bool taken[n+1];
		int count = 0;
		for(i=0;i<n+1;i++)
		{
			taken[i] = false;
		}	
		for(i=1;i<=n/2;i++)
		{
			if(!taken[i])
			{
				count++;
				int j = i;
				while(!taken[j])
				{
					taken[j] = true;
					if(j<=n/2)
						j = 2*j;
					else
						j = 2*(j-n/2) - 1;	
				}
			}
		}
		if(odd)
			count++;
		unsigned long long int val = 1;	
		while(count--)	
		{
			val = (val*26)%1000000007;
		}
		printf("%lld\n", val);
	}
	return 0;
}
