#include<stdio.h>
#include<string.h>
int main(){
	long long n, k, p,m,i,j,nc;
	long long a[100005], b[100005];
	int t;
	scanf("%d", &t);
	while(t--){
		scanf("%lld", &n);
		k=n;
		if(n%2 !=0){
		
		--k;
		b[n]=n;
		}
		m = (k/2) + 1;
		j = 1;
		p = 1;
		b[0] = 0;
		b[1]=1;
		memset(a,0,sizeof(a));
		for(i=2;i<=k;i=i+2){
			b[j]=i;
			j++;
			b[m] = p;
			++m;
			p = p+2;
		}
		j=0;
		nc = 0;
		for(i=1;i<=n;i++){
			j = b[i];
			if(a[i]==0){
				while(j!=i){
				a[j]=-1;
				j = b[j];
				
				}
				nc++;	
			}
			
		}
		p = 1;
		while(nc--){
			p = (p * 26)% 1000000007;
		}
		printf("%lld\n", p);
	}
	return 0;
}
