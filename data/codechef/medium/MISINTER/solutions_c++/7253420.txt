#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define m 1000000007
int main()
{
	int t,i,j,s,n,flag;
	ll s1;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&n);
		s=0,s1=1;
		int a[100001]={0};
		if(n==1)
			s=1;
		else
		{
		    for(j=2;j<=n;j++)
			{
				flag=0;
				while(a[j]==0)
				{
					flag=1;
					a[j]=1;
					if(j%2==0)
						j=j/2;
					else
						j=n/2 + (j+1)/2;
				}
				if(flag==1)
					s+=1;
			}
		}
		for(j=1;j<=s;j++)
		{
			s1*=26;
			s1=s1%m;
		}
		printf("%lld\n",s1%1000000007);
	}
return 0;
}
