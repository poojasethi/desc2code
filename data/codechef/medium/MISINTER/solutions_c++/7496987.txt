#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		int a;
	    scanf("%d",&a);
		int arr[100001];
		for(int i=1;i<=a/2;i++)
		{
			arr[i] = i*2;
		}
		int r=1;
		for(int i=(a/2)+1;i<=a;i++)
		{
			arr[i] = r;
			r+=2;	
		}
		long long ans=1;
		for(int i=1;i<=a;i++)
		{
			//cout<<it->first<<" "<<it->second<<endl;
			if(arr[i]!=0)
			{
			
				int ok=1;
				while(ok)
				{
					if(arr[i]==0)
					{
						ans=(ans*26)%MOD;
						ok=0;
					}
					else
					{
						int temp=arr[i];
						arr[i] = 0;
						i=temp;
						
					}
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
