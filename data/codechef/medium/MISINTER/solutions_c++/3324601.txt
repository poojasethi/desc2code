#include<iostream>
#define MOD 1000000007
using namespace std;

long long power(long long n,int m)
{
		
    if(m==0) 
		return 1;
    
    long long x=power(n,m/2);
    if(m%2==0) 
        return (x*x)%MOD;
    else
        return (((x*x)%MOD)*n)%MOD;
}

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int visit[n+1];
		for(int i=0;i<=n;++i)
			visit[i]=0;
		int temp=1,count=0;	
		for(int i=1;i<=n;i+=2)
		{
			if(i==n && n%2)
				++count;
			else if(visit[i]==0)
			{
				int j=(n/2)+temp;
				visit[i]=1;
				visit[j]=1;
				while(j!=i && j<=n)
				{
					if(j%2)
					{
						visit[(n+j+1)/2]=1;
						j=(n+j+1)/2;
					}
					else
					{
						visit[j/2]=1;
						j/=2;
					}
				}
				++count;
			}
			++temp;
		}
		cout<<power(26,count)<<endl;
		//cout<<count<<endl;
	}
	return 0;
}
