#include <iostream>
using namespace std;
int indicate[100001];
int ar[100001];
int main()
{
    int t;
    cin>>t;
    while(t-->0)
    {
        int n;
        cin>>n;

        for(int i=1;i<=n/2;i++)
            ar[i]=2*i;
        for(int j=1;j<=n/2;j++)
            ar[n/2+j]=2*j-1;
        for(int k=1;k<=n;k++)
            indicate[k]=0;

        int ctr=0;
        for(int i=1;i<=n;i++)
        {

            if(indicate[i]==0)
            {
                ctr++;
                for(int j=i;indicate[j]==0;j=ar[j])
                    indicate[j]=1;
            }
            }
            long long int ans=1;
            while(ctr-->0)
            {
                ans=(ans*26)%1000000007;
            }

                cout<<ans<<endl;









    }










}
