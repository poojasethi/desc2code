#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
#define mod 1000000007

void cycle(int a[], int x[], int i)
{
    if(x[i]==1)
	return;
    if(x[i]==0)
    {
	x[i]=1;
	cycle(a,x,a[i]);
    }
}

int main()
{
    int t;
    si(t);
    while(t--)
    {
	int n;
	si(n);
	int a[n+1],i,k=2;
	for(i=1;i<=n;++i)
	{
	    a[i]=k;
	    k += 2;
	    if(k>n)
		k=1;
	}
	int x[n+1];
	ll ans=1;
	memset(x,0,sizeof(x));
	for(i=1;i<=n;++i)
	{
	    if(x[i]==0)
	    {
		ans = (ans*26)%mod;
		cycle(a,x,i);
	    }
	}
	pll(ans); nl;
    }
    return 0;
}
