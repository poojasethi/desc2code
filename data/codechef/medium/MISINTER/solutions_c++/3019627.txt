#include <stdio.h>
#include <cstring>
typedef unsigned long long ull;
const int MOD=1000000007;
bool a[100001];

void brute_force(int N)
{
	ull res=1;
	memset(a,0,sizeof(bool)*(N+1));
	if(N&1) N--,res=26;
	for (int i = 1; i <= N; i++)
	{
		if(a[i]==false)
		{
			res=(res*26)%MOD;
			int j=i;
			while (a[j]==false)
			{
				a[j]=true;
				if(j&1)
				{
					j=(N>>1)+((j+1)>>1);
				}
				else
				{
					j>>=1;
				}
			}
		}
	}	
	printf("%llu\n",res);
}
int main()
{
	int T,N;
	scanf("%d",&T);
	while (T--)
	{
		scanf("%d",&N);
		brute_force(N);
	}
}
