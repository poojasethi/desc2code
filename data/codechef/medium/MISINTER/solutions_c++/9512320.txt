#include <stdio.h>
#define MOD 1000000007
int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		int N;
		scanf("%d", &N);
		int word[100001];
		int itr = 1;
		for(int i = 2; i <= N; i+=2)
			word[itr++] = i;
		for(int i = 1; i <= N; i+=2)
			word[itr++] = i;
		int count = 0;
		int visited[100001] = {0};
		for(int i = 1; i <= N; ++i)
		{
			if(!visited[i])
			{
				int curr = i;
				while(!visited[curr])
				{
					visited[curr] = 1;
					curr = word[curr];
				}
				++count;
			}
		}
		int ans = 1;
		int base = 26;
		while(count)
		{
			if(count&1)
				ans = (long long)ans*base%MOD;
			base = (long long)base*base%MOD;
			count=count/2;
		}

		printf("%d\n", ans);
	}

	return 0;
}
