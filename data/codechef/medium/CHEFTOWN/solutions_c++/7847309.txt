#include<bits/stdc++.h>
using namespace std;
int a[400005];
int maxi[400005],mini[400005];
void take1(int arr[],int n,int k)
{
	deque<int>q(k);
	for(int i=0;i<k;i++)
	{
		while(!q.empty()&&arr[i]>=arr[q.back()])
		q.pop_back();
		q.push_back(i);		
	}
	int c=0;
	for(int i=k;i<n;i++)
	{
		maxi[c++]=arr[q.front()];
		while(!q.empty()&&q.front()<=i-k)
		q.pop_front();
		while(!q.empty()&&arr[i]>=arr[q.back()])
		q.pop_back();
		q.push_back(i);
	}
	maxi[c++]=arr[q.front()];
}
void take2(int arr[],int n,int k)
{
	deque<int>q(k);
	for(int i=0;i<k;i++)
	{
		while(!q.empty()&&arr[i]<=arr[q.back()])
		q.pop_back();
		q.push_back(i);		
	}
	int c=0;
	for(int i=k;i<n;i++)
	{
		mini[c++]=arr[q.front()];
		while(!q.empty()&&q.front()<=i-k)
		q.pop_front();
		while(!q.empty()&&arr[i]<=arr[q.back()])
		q.pop_back();
		q.push_back(i);
	}
	mini[c++]=arr[q.front()];
}
int main()
{
	int n,w;
	scanf("%d%d",&n,&w);
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	take1(a,n,w);
	take2(a,n,w);
	int ans=0;
	for(int i=0;i<=n-w;i++)
	{
		if(maxi[i]-mini[i]==w-1)
		ans++;
	}
	printf("%d\n",ans);
}
