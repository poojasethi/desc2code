using namespace std;

char arr[1001][1001];

int min(int a, int b)
{
	if (a < b)
		return a;
	else
		return b;
}

void precalc()
{
	memset(arr, 0, sizeof(arr));
	arr[0][0] = 'L';
	for (int i = 0; i < 1001; i++)
	{
		for (int j = 0; j < 1001; j++)
		{
			if (arr[i][j] != 'L' && arr[i][j] != 'W')
				arr[i][j] = 'L';
			if (arr[i][j] == 'W')
				continue;
			else
			{
				for (int k = i+1; k < 1001; k++)
					arr[k][j] = 'W';
				for (int k = j+1; k < 1001; k++)
					arr[i][k] = 'W';
				int m = min(1001-i, 1001-j);
				for (int k = 1; k < m; k++)
					arr[i+k][j+k] = 'W';
			}
		}
	}
}

int main()
{
	precalc();
	int T;
	scanf("%d", &T);
	for (int t = 0; t < T; t++)
	{
		int m, n, p, q;
		scanf("%d %d %d %d", &m, &n, &p, &q);
		if (arr[m-p][n-q] == 'L')
			printf("Bob\n");
		else
			printf("Alice\n");
	}	
	return 0;
}
