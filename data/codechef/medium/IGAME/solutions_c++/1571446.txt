#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1000;
int pre[N+5][N+5];

int main()
{
    for (int i=0; i<=N; i++) for (int j=0; j<=N; j++) pre[i][j] = 1;
    for (int i=0; i<=N; i++) for (int j=0; j<=N; j++) if (pre[i][j]) {
        for (int k=i+1; k<=N; k++) {
            pre[k][j]=0;
        }
        for (int k=j+1; k<=N; k++) {
            pre[i][k]=0;
        }
        for (int k=1; i+k<=N && j+k<=N; k++) {
            pre[i+k][j+k]=0;
        }
    }
/*
    for (int i=0; i<10; i++) {
        for (int j=0; j<10; j++)
            printf("%d ", pre[i][j]);
        printf("\n");
    }
*/
    int t, m, n, p, q;
    scanf("%d", &t);
    while (t--) {
        scanf("%d%d%d%d", &m, &n, &p, &q);
        m-=p; n-=q;
        if (pre[m][n]==0)
            printf("Alice\n");
        else
            printf("Bob\n");
    }
    return 0;
}
