#define DEBUG 0

#include <stdio.h>
#include <string.h>

void mark(char a[1001][1001], int x, int y) {
  int i;
  for(i=x;i<=1000;i++)
    a[i][y]=1;
  for(i=y+1;i<=1000;i++)
    a[x][i]=1;
  for(i=1;x+i<=1000&&y+i<=1000;i++)
    a[x+i][y+i]=1;
}

int main(void) {
  char a[1001][1001];
  int r[1001];
  int tc, m, n, p, q, x, y;
 
  memset(a, 0, sizeof(a));
  for(x=0;x<=1000;x++) {
    for(y=0;y<=1000&&a[x][y];y++);
    r[x]=y;
    mark(a,x,y);
    if(DEBUG)
      printf("%d:%d\n", x, y);
  }
 
  scanf("%d", &tc);
  while(tc--) {
    scanf("%d%d%d%d", &m, &n, &p, &q);
    if(r[m-p]==n-q)
      printf("Bob\n");
    else
      printf("Alice\n");
  }
  return 0;
}
