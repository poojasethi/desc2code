#include<iostream>

using namespace std;

static int arr[1001][1001];

void preProcess()
{
    int row[1001]={0},col[1001]={0},diag1[1020]={0},diag2[1020]={0};

    row[0] = 1;
    col[0] = 1;
    for(int i=0;i<1001;i++)
        for(int j=0;j<1001;j++)
        {
            arr[i][j] = 0;

            if(row[i] == 1 || col[j] == 1)
            {
                arr[i][j] = 1;
            }

            if(i==j)
            {
                arr[i][j] = 1;
            }

            if(i<j && diag1[j-i] == 1)
            {
                arr[i][j] = 1;
            }

            if(i>j && diag2[i-j] == 1)
            {
                arr[i][j] = 1;
            }

            if(!arr[i][j])
            {
                arr[i][j] = 0;  //for making code readable;

                row[i] = 1;
                col[j] = 1;

                if(i<j)
                {
                    diag1[j-i] = 1;
                }
                else if(i>j)
                {
                    diag2[i-j] = 1;
                }
            }
        }
}

int main()
{
    preProcess();

    int test,m,n,p,q;
    cin >> test;
    while(test--)
    {
        cin >> m >> n >> p >> q;

        m = m - p;
        n = n - q;

        if(arr[m][n] == 1)
            cout << "Alice"  << endl;
        else
            cout << "Bob" << endl;
    }
    return 0;
}
