#include<stdio.h>
using namespace std;
int main()
{
    int i,j,t,m,n,p,q,k;
    char a[1001][1001];
    for(i=0;i<1000;i++)
    {
              for(j=0;j<1000;j++)
              {
                                 if((a[i][j]!='w')&&(a[i][j]!='l'))
                                 {
                                        a[i][j]='l';
                                 }
                                 if(a[i][j]=='w')
                                 {
                                        continue;
                                 }
                                 else 
                                 {
                                      for(k=j+1;k<1000;k++)
                                      {
                                         a[i][k]='w';
                                         }
                                          for(k=i+1;k<1000;k++)
                                          {
                                             a[k][j]='w';
                                             }
                                          for(k=1;;k++)
                                          {
                                                       if((i+k>999)||(j+k>999))
                                                       {
                                                          break;
                                                          }
                                                          else{a[i+k][j+k]='w';}
                                          }
                                 }
              }
    }
    
    scanf("%d",&t);
    while(t--)
    {
              scanf("%d %d %d %d",&m,&n,&p,&q);
              if(a[m-p][n-q]=='w')
              {
                                  printf("Alice\n");
              }
              else
              {
                  printf("Bob\n");
              }
    }          
    return 0;
}
