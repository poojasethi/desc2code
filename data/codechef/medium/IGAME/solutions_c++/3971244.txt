#include <cstdio>

int grid[1003][1003];

void work() {
  int m, n, p, q;
  scanf("%d %d %d %d", &m, &n, &p, &q);
  int X = m - p;
  int Y = n - q;
  if(grid[X][Y] == 0)
    printf("Bob\n");
  else
    printf("Alice\n");
}

void boom(int x, int y) {
  int i;
  for(i = x + 1; i <= 1000; i++)
    grid[i][y] = 1;
  for(i = y + 1; i <= 1000; i++)
    grid[x][i] = 1;
  for(i = 1; x + i <= 1000 && y + i <= 1000; i++)
    grid[x + i][y + i] = 1;
}


int main() {

  int i, j;
  for(i = 0; i <= 1000; i++)
    for(j = 0; j <= 1000; j++)
      grid[i][j] = 0;

  for(i = 0; i <= 1000; i++) 
    for(j = 0; j <= 1000; j++) 
      if(grid[j][i] == 0)
	boom(j, i);
      
  
  int T;
  scanf("%d", &T);
  while(T--) work();
  return 0;
}
