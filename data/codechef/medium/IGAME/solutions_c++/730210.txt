#include <iostream>
#include <cstdio>

enum state_t {WIN, LOSE, UNCERTAIN };
int igrid[1001][1001];
void preprocess();
int main()
{
    preprocess();
    int t, m, n, p, q;
    scanf ("%d", &t);
    while (t--){
        scanf ("%d %d %d %d", &m, &n, &p, &q);

        if (igrid[m-p][n-q] == WIN) printf ("Alice\n");
        else printf ("Bob\n");
    }
    return 0;
}
void preprocess()
{
    int i, j;
    for (int i = 0; i <= 1000; i++)
        for (int j = 0; j <= 1000; j++)
            igrid[i][j] = UNCERTAIN;

    for (i = 0; i <= 1000; i++){

        for (j = 0; j <= 1000; j++)
            if (igrid[i][j] == UNCERTAIN) break;

        if (j == 1001) continue;
        igrid[i][j] == LOSE;
        for (int k = j+1; k <= 1000; k++)
            igrid[i][k] =  WIN;

        for (int k = i+1; k <= 1000; k++)
            igrid[k][j] = WIN;

        for (int k = 1; (k+i <= 1000)&&(k+j <= 1000); k++)
            igrid[k+i][k+j] = WIN;

    }
    return;
}
