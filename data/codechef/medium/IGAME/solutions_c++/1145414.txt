#include<iostream>

using namespace std;

static int arr[1010][1010];

void preProcess()
{
    int row[1010]={0},col[1010]={0},diag1[1020]={0},diag2[1020]={0};

    row[0] = 1;
    col[0] = 1;
    for(int i=0;i<1000;i++)
        for(int j=0;j<1000;j++)
        {
            arr[i][j] = 0;

            if(row[i] == 1 || col[j] == 1)
            {
                arr[i][j] = 1;
            }

            if(i==j)
            {
                arr[i][j] = 1;
            }

            if(i<j && diag1[j-i] == 1)
            {
                arr[i][j] = 1;
            }

            if(i>j && diag2[i-j] == 1)
            {
                arr[i][j] = 1;
            }

            if(!arr[i][j])
            {
                arr[i][j] = 0;  //for making code readable;

                row[i] = 1;
                col[j] = 1;

                if(i<j)
                {
                    diag1[j-i] = 1;
                }
                else if(i>j)
                {
                    diag2[i-j] = 1;
                }
     //           i>j ? diag1[i-j] = 1 : diag2[j-i] = 1;
            }
        }

//        for(int i = 0;i<10;i++)
//            cout << row[i] << " ";
//        cout << "\n\n\n";
//
//        for(int i = 0;i<10;i++)
//            cout << col[i] << " ";
//        cout << "\n\n\n";
}

void print()
{
     for(int i=0;i<10;i++)
     {
        for(int j=0;j<10;j++)
        {
            cout << arr[i][j] << " ";
        }

        cout << "\n";
     }

}

int main()
{
    preProcess();

//    print();
    int test,m,n,p,q;
    cin >> test;
    while(test--)
    {
        cin >> m >> n >> p >> q;

        m = m - p;
        n = n - q;

        if(arr[m][n] == 1)
            cout << "Alice"  << endl;
        else
            cout << "Bob" << endl;
    }
    return 0;
}
