int a[1001][1001];

void grid()
{
 int i,j,k,r;
 for(i=1;i<=1000;i++)
  a[i][0]=1,a[0][i]=1;
 for(i=0;i<1001;i++)
 for(j=0;j<1001;j++)
  if(!a[i][j])
  {
   for(k=i+1;k<1001;k++)
    a[k][j]=1;
   for(k=j+1;k<1001;k++)
    a[i][k]=1;
   for(r=i+1,k=j+1;(r<1001&&k<1001);r++,k++)
    a[r][k]=1;
  }
}

int main()
{
 //freopen("input.txt","r",stdin);
 //freopen("output.txt","w",stdout);
 grid();
 int test,m,n,p,q;
 cin>>test;
 while(test--)
 {
  cin>>m>>n>>p>>q;
  if(a[m-p][n-q])
   cout<<"Alice"<<endl;
  else
   cout<<"Bob"<<endl;
 }
 return 0;
}
