#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int i,n;
		scanf("%lld",&n);
		long long int a[n],max=0;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a[i]);
			if(a[i]+i>=max)
			{
				max=a[i]+i;
			}
		}
		
		
		
		
			printf("%lld\n",max);
		
		
	}
	return 0;
}
