#include<iostream>
using namespace std;
int main()
{
 long int t,n,w[100000],v,max,j,i;
 cin>>t;
 while(t--)
 {
   max=0;j=0;
   cin>>n;
   for(i=0;i<n;i++)
   {
    cin>>w[i];
    if(w[i]>=max)
    {
     max=w[i];
     j=i;
     }
    }
    v=max+j;
    for(i=j+1;i<n;i++)
    {
      if(v-i<w[i])
      v=w[i]+i;
    }
    cout<<v<<"\n";
    }
    return 0;
    }
