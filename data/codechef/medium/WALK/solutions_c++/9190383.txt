#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        long int n, max = 0;
        scanf("%ld", &n);
        long int arr[n];
        for(int i = 0; i < n; i++)
        {
            scanf("%ld", &arr[i]);
            if(arr[i] + i > max)
            {
                max = arr[i] + i;
            }
        }
        printf("%ld\n", max);
    }
    return 0;
}
