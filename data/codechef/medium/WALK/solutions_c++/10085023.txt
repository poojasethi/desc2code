#include<iostream>
#include<stdio.h>
#include<stdlib.h>
int main()
{
    long int t;
    scanf("%ld",&t);
    while(t--)
    {
        long int n,*a,i,s;
        scanf("%ld",&n);
        a=(long int*)malloc(n*sizeof(long int));
        for(i=0;i<n;i++)
            scanf("%ld",&a[i]);
        s=0;
        for(i=0;i<n;i++)
            if(s<a[i]+i)
            s=a[i]+i;
        printf("%ld\n",s);
    }
    return 0;
}
