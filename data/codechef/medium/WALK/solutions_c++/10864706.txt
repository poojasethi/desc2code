inline int getchar_unlocked()
{
	return getchar();
}

inline long long scn()
{
	long long n = 0, c = getchar_unlocked(),t=0;
	while(c < '0' || c > '9')
	{
		if(c==45)t=1;
		c = getchar_unlocked();
	}
	while(c >= '0' && c <= '9')n = (n<<3) + (n<<1) + c - '0', c = getchar_unlocked();
 
	if(t==0)return n;
	else return -n;
}
 
 
int main()
{
    int tc;
    tc=scn();
    while(tc--)
    {
        int n;
        int a;
        n=scn();
        int p=0;
        for(int i=0;i<n;i++)
        {
            a=scn();

        }

    }

}
