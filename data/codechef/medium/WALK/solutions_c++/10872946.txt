#include<cstdio>
using namespace std;
int MAX(int a,int b){
    if(a<b){
        return b;
    }
    return a;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int speed=0;
    for(int i=0;i<n;i++){
        if(speed<arr[i]+i){
            speed=arr[i]+i;
        }
    }

    }
