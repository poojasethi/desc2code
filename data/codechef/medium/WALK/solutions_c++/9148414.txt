#include<bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("in","r",stdin);
	long t;
	cin >> t;
	while (t--)
	{
    long n;
    cin >> n;
    long a[n] ;
    for( long i = 0 ; i < n ; i++ )
    {
    	cin >> a[i] ;
    }
    long  mx = 0;
    for( long i = 0 ; i < n ; i++ )
    {
    	mx = max ( mx , a[i] + i );
    } 
    cout << mx << endl ;
    }
}
