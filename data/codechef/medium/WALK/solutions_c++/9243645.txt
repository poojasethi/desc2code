#include<iostream>
using namespace std;
int main()
{
int t,k,n,i,v;
cin>>t;
while(t--)
{
cin>>n;
int arr[n];

for(i=0;i<n;i++)
   {
   cin>>arr[i];
   }
k=n;
v=k;
for(i=0;i<n;i++)
    {
       if(arr[i]>v)
           {
            k=k+arr[i]-v;
            v=arr[i];
           }
           v--;
    }
    cout<<k<<endl;

}
return 0;
}
