#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long long int a[n],i;
        for(i=0;i<n;++i)
        cin>>a[i];
        long long int speed=a[n-1];
        for(i=n-2;i>=0;--i)
        {
            ++speed;
            if(speed>=a[i])
            continue;
            else
            {
                speed=a[i];
            }
        }
        cout<<speed<<"\n";
    }
    return 0;
}
