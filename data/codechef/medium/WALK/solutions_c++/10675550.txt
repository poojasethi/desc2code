#include <iostream>
#include <algorithm>
#include <string>
#define M 10000009
using namespace std;

int main(){
        long long int t;
        cin>>t;
        while(t--){
            long long int n;
            cin>>n;
            long long maxvel=0;
            long long int x;
            for(long long int i=0;i<n;i++){
                cin>>x;
                x=x+i;
                if(x>maxvel){
                    maxvel=x;
                }
            }

            cout<<maxvel<<endl;
        }
    return 0;
}
