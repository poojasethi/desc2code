import sys

def minimal_velocity(attract_shop):
  Vmin = 0
  diff = 0
  for shop in attract_shop:
    if Vmin < shop:
      diff += shop - Vmin
      Vmin = shop
    Vmin -= 1
  Vmin = diff
  return Vmin

t = input()
for case in xrange(t):
  nb_shops = input()
  data = map(int,sys.stdin.readline().strip().split())
  print(minimal_velocity(data))
