import sys

inp = sys.stdin.readlines()
ctr = 0

t = int(inp[ctr])
ctr = ctr+1
while t>0:
   n = int(inp[ctr])
   ctr = ctr+1
   a = inp[ctr].split()
   ctr = ctr+1
   ans = int(a[0])
   for j in range(1,n):
      if ans < int(a[j])+j:
         ans = int(a[j])+j
   sys.stdout.write(str(ans)+'\n')
   t = t-1
sys.stdout.flush()
