t=int(raw_input())
inp=[]
maxinp=2
for i in xrange(t):
	inp.append(int(raw_input()))
	if inp[-1]>maxinp:
		maxinp=inp[-1]
 
 
N=maxinp
E={1:(1,1),2:(2,1)}
for n in xrange(3,N+1):
	a,b=E[n-1]
	c,d=E[n-2]
	num=(n**2-3*n+3)*b*d+(2*(n-1)*a*d)-c*b
	deno=(2*n-3)*b*d
	a,b=max(num,deno),min(num,deno)
	while b>0:
		a,b=b,a%b
	E[n]=(num/a,deno/a)
 
for i in xrange(t):
	if E[inp[i]][1]>1:
		print '%d/%d'%(E[inp[i]][0],E[inp[i]][1])
	else:
		print E[inp[i]][0]
