#include<iostream>
#include<set>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cassert>
using namespace std;

vector<int>a,p,l,r,d,ans;

int main()
{

  //  freopen( "in.txt","r",stdin );


    int n,i,j,k,q,mv=0;

    cin>>n>>q;


    a=p=vector<int>(n+1,0);
    ans=l=r=d=vector<int>(q+1,0);


    for( i=1;i<=n;i++ )
    {
        scanf("%d",&a[i]);
        mv=max( mv,a[i] );
    }


    vector<  pair<int,int> > vp;// = vector< pair<int,int> > (n+1) ;

    #define pb push_back

    for( i=1;i<=q;i++ )
    {
        scanf("%d%d%d",&l[i],&r[i],&d[i]);

        l[i]=max( 1,l[i] );
        r[i]=min( n,r[i] );

        if( l[i]-1 )vp.pb( make_pair( l[i]-1,i ) );
        vp.pb( make_pair( r[i],i ) );


        mv=max( mv,d[i] );

    }

    sort( vp.begin(),vp.end() );

    vector<int> sum=vector<int>( mv+1,0 );


    k=0;
    for( i=1;i<=n;i++ )
    {
        for( j=1;j*j<=a[i];j++ )
        {
            if( a[i]%j==0 )
            {
                sum[ j ]++;
                if( j*j!=a[i] )sum[ a[i]/j ]++;
            }
        }

//        cout<<k<<" "<<vp.size()<<endl;

        while( k<(int)vp.size() )
        {
            pair<int,int> u=vp[k];
            if( u.first>i )break;

          //  cout<<u.first<<" "<<u.second<<endl;

            if( u.first==r[ u.second ] )ans[ u.second ]+=sum[ d[u.second] ];
            else ans[ u.second ]-=sum[ d[u.second] ];
            k++;
        }
        /*

        for( j=0;j<(int)vp[ i ].size();j++ )
        {
            pair<int,int> u=vp[i][j];
            if( u.first==0 )ans[ u.second ]-=sum[ d[u.second] ];
            else ans[ u.second ]+=sum[ d[u.second] ];
        }*/

    }

    for( i=1;i<=q;i++ )printf( "%d\n",max(ans[i],0) );

    return 0;
}
