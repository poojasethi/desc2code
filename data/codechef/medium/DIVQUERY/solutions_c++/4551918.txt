int n,q;

	int i,j;
	si(n);
	si(q);
	memset(c,0,sizeof(c));
	for(i=1;i<=n;i++){
		si(a[i]);
	}
	map< pair<int,int> ,int>mp;
	int l,r,k;
	int lr[q],rl[q],kk[q];

	for(i=0;i<q;i++){
		si(l);
		si(r);
		si(k);
		lr[i]=l;
		rl[i]=r;
		kk[i]=k;
		pair <int ,int> t;
		t.first=l-1;
		t.second=k;
		mp[t]=0;
		v[l-1].pb(k);

					c[j]--;
				}
			}
		}

			pair<int,int>t;
			t.first=i;
			t.second=v[i][j];
			mp[t]=c[v[i][j]];
		}
	}
