#include<iostream>

using namespace std;
typedef long long lli;
lli BIT[1000002];

void update(lli index, lli N, lli value)
{
    while (index<=N)
    {
        BIT[index] += value;
        index = index + (index & (-index));
    }
}

lli getsum(lli index, lli N)
{
    lli ans  = 0;
    while(index>0)
    {
        ans += BIT[index];
        index = index - (index & (-index));
    }
    return ans;
}
