#include <bits/stdc++.h>
using namespace std;
 
typedef pair< int , int > pii;
typedef long long LL;
#define fr first
#define se second
#define EPS 1e-8
#define INF 10000*10000*10000LL
stringstream ss;
#define two(x) ( 1LL<<x )
LL mod = 1000000007LL;

/**************************Code****************************/

LL tree[1<<20];

void add( int id, int val )
{
	while( id < two(20) )
	{
		tree[id] += val;
		id += ( id & -id );
	}
}
LL sum( int id )
{
	LL ret = 0;
	while( id )
	{
		ret += tree[id];
		id -= ( id & -id );
	}
	return ret;
}

int main()
{
	int n, q, a, b, x;
	char c;
	cin >> n >> q;
	for( int i = 1 ; i <= n ; i ++ )
		cin >> x, add( i, x );
	while( q -- )
	{
		cin >> c >> a >> b;
		if( c == 'S' )
			cout << sum( ++b ) - ( ++a ? sum(a-1) : 0 ) << endl;
		else if( c == 'G' )
			add( ++a, b );
		else if( c == 'T' )
			add( ++a, -b );
	}
	return 0;
}
