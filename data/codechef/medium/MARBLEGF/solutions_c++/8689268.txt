char type;
ll n,q,l,r,f[1000000+10],x;

void update(int x, ll v){
    while (x<=n){
        f[x]+=v;
        x+=x&(-x);
    }
}

ll get(int x){
    ll sum=0;
    while (x){
        sum+=f[x];
        x-=x&(-x);
    }
    return sum;
}
