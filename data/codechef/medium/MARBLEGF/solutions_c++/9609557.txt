#include<bits/stdc++.h>
#define gc getchar
using namespace std;
typedef long long ll;
int read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
int ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
ll bit[1000005],a[1000005],n;
void update(ll x,ll val)
{
    //cout<<x<<" "<<n<<'\n';
    for(;x<=n;x+=x&-x)
    {
        bit[x]+=val;
        //cout<<bit[x]<<'\n';
    }
}
ll prefixSum(ll x)
{
    ll sum=a[x];
    for(;x>0;x-=x&-x)
    {
        sum+=bit[x];
    }
    return sum;
}
int main()
{
    ll q,i,j,x;
    cin>>n>>q;
    memset(bit,0,sizeof(bit));
    a[0]=0;
    cin>>a[1];
    for(i=2;i<=n;i++)
    {
        cin>>x;
        a[i]=a[i-1]+x;
    }
    char ch;
    while(q--)
    {
       cin>>ch>>i>>j;
       i++;
       if(ch=='S')
       {
           j++;
           if(i==j)
              cout<<a[i]-a[j-1]<<'\n';
           else
           {
               if(i!=1)
                  cout<<prefixSum(j)-prefixSum(i-1)<<'\n';
               else
                  cout<<prefixSum(j)<<'\n';
           }
       }
       else if(ch=='G')
       {
           update(i,j);
           //for(i=1;i<=n;i++)
             //cout<<bit[i]<<' ';
       }
       else
       {
           update(i,-j);
       }
    }
    return 0;
}
