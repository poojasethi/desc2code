#include<iostream>
using namespace std;
void update(long long int arr[],long long int n,long long int i,long long int c)
{
	while(i<=n)
	{
		arr[i]+=c;
		i+=(i&-i);
	}
}
long long int getResult(long long int arr[],long long int p)
{
	long long int sum=0;
	while(p>=1)
	{
		sum+=arr[p];
		p-=(p&-p);
	}
	return sum;
}
int main()
{
	long long int i,n,m,ans,j,u,c;
	char ch;
	cin>>n>>m;
	long long int arr[n+1];
	for(i=0;i<=n;i++)
	arr[i]=0;
	for(i=1;i<=n;i++)
	{   cin>>c;
		update(arr,n,i,c);
	}
	cin.ignore();
	while(m--)
	{
		cin>>ch;
		if(ch=='G')
		{
			cin>>i>>u;
			update(arr,n,i+1,u);
		}
		else if(ch=='T')
		{
			cin>>i>>u;
			update(arr,n,i+1,-u);
		}
		else
		{
			cin>>i>>j;
			ans=getResult(arr,j+1);
			ans-=getResult(arr,i);
			cout<<ans<<endl;
		}
	}
	
	return 0;
}
