#include <cstdlib>
#include <iostream>

using namespace std;
long long int bit[1000000];
long long int sum(long long int index)
{
     long long int ans = 0;
     while(index > 0){
           ans+=bit[index-1];
           index = index - (index&(-1*index));
     }
     return ans;      
}
void increment(long long int index,long long int value,long long int size){
     while(index<=size){
         bit[index-1]+=value;
         index = index+ (index&(-1*index));
       //  cout << index << '\n';
     }
}  

void decrement(long long int index,long long int value,long long int size){
     while(index<=size){
         bit[index-1]-=value;
         index = index+ (index&(-1*index));
     }
}             
     
int main()
{
    long long int N,Q;
    cin >> N >> Q;
    long long int A[N];

    for(long long int i = 0;i<N;++i){
        cin >> A[i];
        //printf("Entered1\n");
        increment(i+1,A[i],N);
    }
   // printf("Entered1\n");
    char C;
    long long int a,b;
    while(Q--) {
          getchar();     
          cin >> C >> a >> b;
          //cout << C << ' ' << a << ' ' << b << '\n';
          if(C=='S'){
             cout << sum(b+1)-sum(a) << '\n';
          }
          else if(C=='G')  increment(a+1,b,N);
          else if(C=='T')  decrement(a+1,b,N);
    }      
                  
    
    return 0;
}
