#include <iostream>
#include <cstdio>
using namespace std;
#define M 2100
double C[M][M];
//C[N][1] * V[1] + ... + C[N][N] * V[N],
// calculating c[n][1].......c[n][n]
 
void calculate(){
for(int n=1;n<=M-1;n++){
    for(int k=1;k<=n;k++){
            // formula derived C[L][k] = 1 − (C[L − 1][k − 1] + C[L − 1][k]) / 2, for k from 1 to L,
        C[n][k] = 1 - (C[n - 1][k - 1] + C[n - 1][k]) / 2;
    }
}
}
int main()
{
    calculate();
    int t;
    scanf("%d",&t);
    while(t--){
 
        int N;
        scanf("%d",&N);
        double sump=0;
        for(int i=1;i<=N;i++){
            int vi;
    scanf("%d",&vi);
    sump += C[N][i] * vi;////C[N][1] * V[1] + ... + C[N][N] * V[N],
        }
        printf("%.3lf\n",sump );
    }
 
 
    return 0;
}
