#define present(c,x) ((c).find(x) != (c).end()) 
#define cpresent(c,x) (find(all(c),x) != (c).end()) 
#define MAX 2002
int main(){
	int t,n,i,j,counter=1,temp;
	double ans=0.0;
	scanf("%d",&t);
	vector< vector<double> >data(MAX,vector<double>(MAX));
	for(i=1;i<MAX;++i){
		for(j=1;j<=counter;++j){
			data[i][j] = 1.0 - 0.5*(data[i-1][j-1]+data[i-1][j]);
 		}
 		++counter;
	}
	while(t--){
		ans=0.0;
		scanf("%d",&n);
		for(i=1;i<=n;++i){
			scanf("%d",&temp);
			ans+=(data[n][i]*(double)temp);
		}
		printf("%lf\n",ans);
	}	
	return 0;
}
