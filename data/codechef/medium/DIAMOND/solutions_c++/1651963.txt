#include <cstdio>

double prob[2002][2002];

void precompute()
{
    prob[1][1] = 1;
    for(int i=2; i<=2000; i++)
        for(int j=1; j<=i; j++)
            prob[i][j] = 1 - (prob[i-1][j] + prob[i-1][j-1]) / 2;
}

int main()
{
    precompute();
    int t;
    scanf("%d", &t);
    while(t--) {
        int n;
        scanf("%d", &n);
        double ans=0;
        int v;
        for(int i=0; i<n; i++) {
            scanf("%d", &v);
            ans += prob[n][i+1] * v;
        }
        printf("%.3f\n", ans);
    }
}
