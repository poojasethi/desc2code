#include<iostream>
#include<cstdio>
#include<vector>
#define MAX 2005
using namespace std;

double dp[MAX][MAX]={0.0};

int main()
{
    #ifndef ONLINE_JUDGE
        freopen("in.txt","r",stdin);
    #endif

    //precompute all the probabilities dp[N][j] in game of 1,2,--N 1st player choose jth
    dp[1][1] = 1.0;
    for(int n=2; n<=2000; n++) {
        dp[n][1] = 0.5*(1 + 1-dp[n-1][1]);
        dp[n][n] = 0.5*(1 + 1-dp[n-1][n-1]);

        for(int j=2; j<=n-1; j++)
            dp[n][j] = 0.5*(1-dp[n-1][j-1] + 1-dp[n-1][j]);
    }
    int t;
    scanf("%d",&t);
    while(t--) {
        int n,v;
        scanf("%d",&n);
        double ans = 0.0;
        for(int i=1; i<=n; i++) {
            scanf("%d",&v);
            ans += dp[n][i]*v;
        }
        printf("%lf\n",ans);
    }
    return 0;
}
