double pro[2004][2004];

#define getcx getchar_unlocked

void inp(int &n) 
{
	n = 0;
	char c = getcx();
	while( c > '9' || c < '0' ) c = getcx();
	while( c >= '0' && c <= '9' ) n = (n << 3 ) + (n<<1) + c -'0',c = getcx();
}

void init()
{
	FOR(i,1,2004) 
		FOR(j,1,i+1) {
			pro[i][j] = 1.0 - 0.5*(pro[i-1][j] + pro[i-1][j-1]);
		}
}

int main()
{
	int t;
	//scanf("%d",&t);
	inp(t);

	init();

	while( t-- ) {
		int n;
		//scanf("%d",&n);
		inp(n);

		int a[n];
		double ans = 0.0;

		REP(i,n) inp(a[i]);//scanf("%d",a+i);

		FOR(i,1,n+1) ans += pro[n][i]*a[i-1];

		printf("%.6lf\n",ans);

	}

}
