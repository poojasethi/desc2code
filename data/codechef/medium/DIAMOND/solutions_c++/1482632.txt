#include<cstdio>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<cctype>
#include<cstring>
using namespace std;
const int maxN=2000;
double C[maxN+1][maxN+1];
int main(){
	for(int n=1;n<=maxN;n++)
		for(int k=1;k<=n;k++)
			C[n][k]=1-(C[n-1][k-1]+C[n-1][k])/2;
	int test;
	for(scanf("%d",&test);test>0;test--){
		int n;
		scanf("%d",&n);
		double sum=0;
		for(int i=1;i<=n;i++){
			int Vi;
			scanf("%d",&Vi);
			sum=sum+Vi*C[n][i];
		}
		printf("%.3lf\n",sum);
	}
	return 0;
}
