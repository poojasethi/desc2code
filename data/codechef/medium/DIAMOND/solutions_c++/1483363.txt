#include<iostream>
#include<stdio.h>
using namespace std;
inline int inp()
{
int noRead=0;
char p=getchar_unlocked();
for(;p<33;){p=getchar_unlocked();};
while(p>32)
{
noRead = (noRead << 3) + (noRead << 1) + (p - '0');
p=getchar_unlocked();
}
return noRead;
};
double c[2001][2001];
void coefficient()
{
    int i,j;
    c[1][1]=1.00;
    for(i=2;i<2001;i++)
    for(j=1;j<=i;j++)
    c[i][j]=1.00-(c[i-1][j-1]+c[i-1][j])/2;
}
int main()
{
    int t,i,j;
    //t=inp();
    scanf("%d",&t);
    coefficient();
    while(t--)
    {
        int n,v;double ans=0;
        //scanf("%d",&n);
        n=inp();
        for(i=1;i<=n;i++)
        {
            v=inp();
            //scanf("%d",&v);
            ans+=(c[n][i]*v);
        }
        printf("%.3lf\n",ans);
    }
    return 0;
}
