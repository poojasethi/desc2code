#include<cstdio>
#include<iostream>
#include<cstring>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<cmath>
using namespace std;
#define  pb push_back

int main()
{
 int T;
 scanf("%d", &T);
 static double dp[2010][2010];
 dp[1][0] = 1.0;
 for (int i = 2; i <= 2000; i++)
   for (int j = 0; j <= i / 2; j++)
       dp[i][j] = dp[i][i - j - 1] = 1.0 - (dp[i - 1][j] + dp[i - 1][j > 0 ? j - 1 : i - 1]) * 0.5; 
  while (T--) 
  {
     int n, x;
     scanf("%d", &n);
     double ans = 0.0;
     for (int i = 0; i < n; i++) scanf("%d", &x), ans += dp[n][i] * x;
     printf("%f\n", ans);
   }
return 0;
}
