#include<iostream>
#include<stdio.h>
#include<string>
#include<set>

#include<stdlib.h>
using namespace std;

#define si(n) scanf("%d",&n)
#define sli(n) scanf("%ld",&n)
#define slli(n) scanf("%lld",&n)
#define sf(n) scanf("%lf",&n)
#define ss(n) scanf("%s",n)
#define pi(n) printf("%d",n)
#define pli(n) printf("%ld",n)
#define plli(n) printf("%lld",n)
#define pf(n) printf("%lf",n)
#define ps(n) printf("%s",n)
#define newLine printf("\n")
#define space printf(" ")

#define max 2000
double arr[max+1][max+1];
int main()
{
	
	for(int i=1;i<=max;i++){
	for(int j=1;j<=i;j++)
	{
		arr[i][j]=1-0.5*(arr[i-1][j]+arr[i-1][j-1]);
	}	
	}
	
	
	int t;
	si(t);
	
	int n;
	
	while(t--)
	{
		si(n);
		int a[n+1];
		
		for( int i=1;i<=n;i++)
		si(a[i]);
		
		double ans=0;
		for(int i=1;i<=n;i++)
		{
			ans+=arr[n][i]*a[i];
		}
		
		
		printf("%0.3lf",ans);
		newLine;
		
		
		
		
	}
	
	
	
	
	
	
	return 0;
}
