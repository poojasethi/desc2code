#include <iostream>
#include <cstdio>
using namespace std;

const int N = 2001;
double a[N][N], b[N][N];
double ans;
int v[N];

void pre() {
    a[1][1] = 1; b[1][1] = 0;
    for (int i=2; i<N; i++) {
        for (int j=1; j<i; j++) {
            a[i][j] = 0.5*b[i-1][j];
            b[i][j] = 0.5*a[i-1][j];
        }
        a[i][i] = 0;
        b[i][i] = 0;
        for (int j=2; j<=i; j++) {
            a[i][j] += 0.5*b[i-1][j-1];
            b[i][j] += 0.5*a[i-1][j-1];
        }
        a[i][1] += 0.5;
        a[i][i] += 0.5;
    }
}

int main()
{
    int t, n;
    pre();
    scanf("%d", &t);
    while (t--) {
        scanf("%d", &n);
        for (int i=1; i<=n; i++) scanf("%d", &v[i]);
        ans = 0;
        for (int i=1; i<=n; i++) {
            ans += v[i]*a[n][i];
        }
        printf("%lf\n", ans);
    }
    return 0;
}
