#include<stdio.h>
#include<malloc.h>
void program();
double **a= (double**)malloc(sizeof(double*)*2000);

void generate()
{
    int i,j;
    for(i=0;i<2000;i++)
    a[i]= (double*)malloc(sizeof(double)*2000);
    a[0][0]=1;
    a[1][0]=0.5;
    a[1][1]=0.5;

    for(i=2;i<2000;i++)
    {
        for(j=0;j<=i;j++)
        {
            a[i][j]= 1- 0.5*(a[i-1][j-1]+ a[i-1][j]);
        }
    }
}
int main()
{
    int t;
    generate();
    scanf("%d",&t);
    while(t--)
    program();
    return 0;
}

void program()
{
    int n,i,l,j;
    int v[2000];
    double ans=0.0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    scanf("%d",&v[i]);
    for(i=0;i<n;i++)
    {
     ans= ans + a[n-1][i]*v[i];
    }
    printf("%.3f\n",ans);
}
