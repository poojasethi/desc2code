#include<stdio.h>
#include<iostream>
using namespace std;

double coef[2001][2001];
void coefficient()
{
    int i,j;
    coef[1][1]=1.00;
    for(i=2;i<2001;i++)
    for(j=1;j<=i;j++)
    coef[i][j]=1.00-(coef[i-1][j-1]+coef[i-1][j])/2;
}
int main()
{
    int i,input,t,n;
    double ans;
    coefficient();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        ans=0;
        for(i=1;i<=n;i++)
        {
            scanf("%d",&input);
            ans+=(coef[n][i]*input);
        }
        printf("%.3lf\n",ans);
    }
    return 0;
}
