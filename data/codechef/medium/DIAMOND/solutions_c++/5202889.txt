#include<cstdio>
#include<iostream>
double c[2005][2005];
using namespace std;
int main()
{
	
	int t;
	c[0][0]=0;
	for(int i=1;i<2005;i++)
	{
		c[i][0]=0;
		for(int j=1;j<=i;j++)
		{	
			c[i][j]=1-(c[i-1][j-1]+c[i-1][j])/2.0;
		}
	}
	scanf("%d",&t);
	while(t--)
	{
		int n,a;
		double ans=0;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a);
			ans+=c[n][i+1]*a;
		}
		printf("%.3f\n",ans);
	}
	return 0;
}
