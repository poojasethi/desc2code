#include <cstring>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstdio>
#include <vector>
using namespace std;
#define maxn 2010
double dp[maxn][maxn];
void init(){
    memset(dp,0,sizeof(dp));
    for(int i=1;i<maxn;i++){
        for(int j=1;j<=i;j++){
            dp[i][j]=1-0.5*(dp[i-1][j-1]+dp[i-1][j]);
        }
    }
}
int main(){
    init();
    int T;
    scanf("%d",&T);
    while(T--){
        int n,x;
        double ans=0;
        scanf("%d",&n);
        for(int i=0;i<n;i++){
            scanf("%d",&x);
            ans+=dp[n][i+1]*x;
        }
        printf("%.12f\n",ans);
    }
    return 0;
}
