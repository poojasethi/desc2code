inline void fs(int &x)
{
	register int c = gc();
	x = 0;
	int neg = 0;
	for(;((c<48 || c>57) && c != '-');c = gc());//c<'0'||c>'9'
	if(c=='-') {neg=1;c=gc();}
	for(;c>=48 && c<=57;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
	if(neg) x=-x;
}
