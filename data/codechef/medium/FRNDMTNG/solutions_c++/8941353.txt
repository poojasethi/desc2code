using namespace std;
 
int readint(){
    int n = 0;
    char c = gc();
    while ( !( '0' <= c && c <= '9' ) ) {
        c = gc();
    }
    while ( '0' <= c && c <= '9' ) {
        n = (n<<3) + (n<<1) + c - '0';
        c = gc();
    }
    return n;
}
