# include <cstdio>
# include<algorithm>
using namespace std;
int main()
{
	long int T;
	scanf("%ld",&T);
	while(T--)
	{
		long double temp[2][2], T1,T2,t1,t2,x;
		scanf("%Lf %Lf %Lf %Lf",&temp[0][0],&temp[1][0],&temp[0][1],&temp[1][1]);
		if(temp[0][0] < temp[1][0])
		{
			T1 = temp[0][0];
			t1 = temp[0][1];
			T2 = temp[1][0];
			t2 = temp[1][1];
		}
		else
		{
			T1 = temp[1][0];
			t1 = temp[1][1];
			T2 = temp[0][0];
			t2 = temp[0][1];
		}
		if(t2 > T1)
		t2 = T1;
		long double a2, a1 = 0.5*(T1-t2)*(T1-t2);
		x = T2-t1;
		if(x > T1)
		x = T1;
		if(T2 >= t1)
		a2 = ((T2-t1)*x)-0.5*(x*x);
		else
		a2 = 0;
		long double c = (a1+a2)/(T1*T2);
		c = 1-c;
		printf("%.6Lf\n",c);
	}
	return 0;
}
