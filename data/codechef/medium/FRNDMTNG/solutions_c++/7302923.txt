#include<iostream>
#include<stdlib.h>

int main()
 {
		
		int t;
		cin>>t;
		
		while(t!=0)
		{	
		t--;
		double T1,t1,T2,t2;
		cin>>T1>>T2>>t1>>t2;
		 double SQ1, SQ2;
	        if (T1 > T2) {
	            swap(T1, T2);
	            swap(t1, t2);
	        }
	        
	        if (t1 > T2) t1 = T2;
	        if (t2 > T1) t2 = T1;
	        
	        SQ1 = T1 * 0.5 * T1;
	        if (T1 > t2)
	            SQ1 -= (T1 - t2) * 0.5 * (T1 - t2);
	        
	        
	        if (t1+T1 <= T2) {
	            SQ2 = T1 * 1.0 * t1;
	        }
	        else {
	            // t1+T1 >= T2
	            SQ2 = T1 * 0.5 * T1;
	            SQ2 += (T1) * 1.0 * (T2-T1);
	            SQ2 -= (T2-t1) * 0.5 * (T2-t1);

	         }
	            double value=(SQ1+SQ2)/(T1*T2);
	            
				cout.precision(6);
	           cout<<value<<endl;
	
	        }       
	

}

	 void swap(double t1, double t2) {
		double temp;
		temp=t1;
		t1=t2;
		t2=temp;
	}
