int main(){
 
    LL t , T1 , T2 , t1 , t2 ;
    cin >> t ;
    while( t -- ){
        cin >> T1 >> T2 ;
        cin >> t1 >> t2 ;
        if( T1 < T2 ){
            swap(T1,T2) ;
            swap(t1,t2) ;
        }
        double ans1 , ans2 ;
        ans1 = 0 ;
        if( t1 < T2 ){
            ans1 = (T2-t1) * ( 0.5 * (T2-t1) + t1 )  ;
            ans1 = ans1 + (T1 - T2 + t1) * T2 ;
            ans1 = T1 * T2 - ans1 ;
        }
        ans2 = 0 ;
        if( t2 < T1 ){
            if( T1 - t2 > T2 ){
                ans2 = 0.5 * T2 * T2 ;
                ans2 = ans2 + t2 * T2 ;
                ans2 = T1 * T2 - ans2 ;
            }else{
                ans2 = (T1-t2) * ( 0.5 * (T1-t2) + T2-T1+t2 )  ;
                ans2 = ans2 + (t2) * T2 ;
                ans2 = T1 * T2 - ans2 ;
            }
        }
        double ans = ans1 + ans2 ;
        ans /= (T1 * T2) ;
        ans = 1 - ans ;
        printf("%.6lf\n",ans) ;
    }
    return 0 ;
}
