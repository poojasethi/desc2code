#include<iostream>
using namespace std;

int main()
{
    int t;
    double area1, area2;
    cin >> t;
    while (t --)
    {
        int T1, T2, t1, t2;
        cin >> T1 >> T2 >> t1 >> t2;
        if (T1 > T2)
        {
            swap(T1, T2);
            swap(t1, t2);
        }

        if (t1 > T2)
            t1 = T2;
        if (t2 > T1)
            t2 = T1;

        area1 = T1 * 0.5 * T1;
        if (T1 > t2)
            area1 -= (T1 - t2) * 0.5 * (T1 - t2);


        if (t1+T1 <= T2)
        {
            area2 = T1 * 1.0 * t1;
        }
        else
        {
            area2 = T1 * 0.5 * T1;
            area2 += (T1) * 1.0 * (T2-T1);
            area2 -= (T2-t1) * 0.5 * (T2-t1);
        }

        cout.precision(6);
        cout << fixed << (area1 + area2) / (T1 * 1ll * T2) << "\n";
    }
    return 0;
}
