#define ll long long int

ll T1,t1,T2,t2;

ll fun(ll t)
 {
     ll left = t-t2;
     ll rt = t + t1;
     left = max(left,(ll)0);
     rt = min(T2,rt);
     if(left < rt) return rt-left;
     return 0;
 }
