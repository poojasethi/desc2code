#include<iostream>
#include<cstdio>
#define lld long long int
using namespace std;
 
int main()
{
	int T;
	double a,b,c;
    lld T1,T2,t1,t2;
    scanf("%d",&T);
	while(T--)
	{
		scanf("%lld%lld%lld%lld",&T1,&T2,&t1,&t2);
		a=b=0.0;
		if(T1-t2>0)
		{
			a=0.5*(T1-t2)*(T1-t2);
			if(T1-t2>T2)
			{
				a=a-0.5*(T1-t2-T2)*(T1-t2-T2);
			}
		}
        
		if(T2-t1>0)
		{
			b=0.5*(T2-t1)*(T2-t1);
			if(T2-t1>T1)
			{
				b=b-0.5*(T2-t1-T1)*(T2-t1-T1);
			}
		}
		c=T1*T2-a-b;
		c=c/(T1*T2);
		printf("%lf\n",c);
	}
	return 0;
}
