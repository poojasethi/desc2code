#include<stdio.h>
#include<math.h>
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
long long int T1,t1,T2,t2,swap;
double prob=0;
scanf("%lld %lld %lld %lld",&T1,&T2,&t1,&t2);
if(T1==0||T2==0)
{
printf("0.000000");
continue;
}
if(T1>T2)
{
swap=T1;
T1=T2;
T2=swap;
swap=t1;
t1=t2;
t2=swap;
}
if(t1>T2)
t1=T2;
if(t2>T1)
t2=T1;
if((T2-t1)<T1)
prob=((T1*T2)-(0.5)*(pow(T1-t2,2)+pow(T2-t1,2)))/(T1*T2);
else
prob=((t1*T1)+(0.5)*(pow(T1,2)-pow(T1-t2,2)))/(T1*T2);
printf("%lf\n",prob);
}
}
