#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
    //freopen("in.txt","r",stdin);
    int TC;
    scanf("%d",&TC);
    while(TC--){
        int T1,T2,t1,t2;
        scanf("%d %d %d %d",&T1,&T2,&t1,&t2);
        double sq1,sq2;
        if(T1 > T2){
            swap(T1,T2);
            swap(t1,t2);
        }
        if(t1>T2) t1=T2;
        if(t2>T1) t2=T1;
        sq1=T1*0.5*T1;
        if(T1 > t2)
            sq1-=(T1-t2)*0.5*(T1-t2);
        if(t1+T1 <= T2){
            sq2=T1*1.0*t1;
        }
        else{
            sq2=T1*0.5*T1;
            sq2+=(T1)*1.0*(T2-T1);
            sq2-=(T2-t1)*0.5*(T2-t1);
        }
        //printf("%1.6lf\n",(sq1+sq2)/(T2*T1*1LL));
        cout.precision(6);
        cout << fixed << (sq1 + sq2) / (T1 * 1ll * T2) << "\n";
    }
}
