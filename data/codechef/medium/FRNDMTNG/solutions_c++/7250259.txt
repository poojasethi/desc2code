#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;cin>>t;
	while(t--)
	{
		long long int T1,T2,t1,t2;
		//cin>>T1>>T2>>t1>>t2;
		scanf("%lld%lld%lld%lld",&T1,&T2,&t1,&t2);
		float s1=0,s2=0,s3=0,s4=0,ans=0,sum=0;
		if(t1==0 && t2==0)
			printf("0.000000\n");
		else
		{
			if(T1>T2)
			{
				swap(T1,T2);
				swap(t1,t2);
			}
			if(T1>=t2)//tri always
			{
				s1=(T1-t2)/(float)T1;
				s1=s1*(T1-t2)/(float)T2;
				s1=s1*0.5;
				sum=sum+s1;

				/*s1=((T1-t2)*(T1-t2))*0.5;*/

			}
			if(T2>T1+t1)//with rect
			{
				
				s2=T1/(float)T2;
				s2=s2*0.5;

				//sum+=s2;

				s3=(T2-(T1+t1))/(float)T2;
				sum=sum+s2+s3;
				/*s2=(T1*T1)*0.5;
				s3=(T2-(T1+t1))*T1;
				sum=s1+s2+s3;
				sum=sum/(double)(T1*T2);*/
				
			}	
			else//without rect
			{
				if(T2>=t1)
				{
					s4=(T2-t1)/(float)T1;
					s4=s4*(T2-t1)/(float)T2;
					s4=s4*0.5;
					
					sum=sum+s4;
					/*s4=(T2-t1)*(T2-t1)*0.5;
					sum=s1+s4;
					sum=sum/(double)(T1*T2);*/
					
				}
			}
			ans=1-sum;
			printf("%0.6f\n",ans);
		}
	}
	return 0;
}
