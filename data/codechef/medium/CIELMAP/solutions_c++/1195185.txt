#include<iostream>
#include<cmath>
#include<iomanip>
using namespace std;
int main()
{
 int t,n,i,j,k,l,m,left=0,right=0;
 double x[2000],y[2000],x1,x2,x3,xx,xx1,y1,y2,y3,yy,yy1,dist=0,slope=0,big=0;
 cin>>t;
 while(t--)
 {
  cin>>n;
  for(i=0;i<n;i++)
  {
   cin>>x[i]>>y[i];
  }
  
  big=0;
  if(n==4)
  {
   
   for(i=0;i<n;i++)
   {
    for(j=0;j<n;j++)
    {
     x1=x[i];y1=y[i];
     x2=x[j];y2=y[j];
     
     if((x2-x1)!=0&&(y2-y1)!=0)
     {
      slope=(y2-y1)/(x2-x1);    
      xx=x1-y1/slope;
     }
     
     else if(y2-y1==0)
     {
      yy=y1;
     }
     
     else if(x2-x1==0)
     {
      xx=x1;
     }
     left=0;right=0;
     for(k=0;k<4;k++)
     {
      
      if(k!=i&&k!=j)
      {
       x3=x[k]; y3=y[k];
       
       if((x2-x1)!=0&&(y2-y1)!=0)
       {
        xx1=x3-y3/slope;
       }
       
       else if(y2-y1==0)
       {
        yy1=y3;
       }
       
       else if(x2-x1==0)
       {
        xx1=x3;
       }
       
       if(xx1>xx&&(y2-y1)!=0)
       right++;
       
       else if(xx1<xx&&(y2-y1)!=0)
       left++;       
       
       else if((y2-y1)==0&&yy1>y1)
       left++;
       
       else if((y2-y1)==0&&yy1<y1)
       right++;       
      }
      
      if(left==2||right==2)
      {
       dist=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
       if(dist>big)
       big=dist;
      }
      
     }
     
     
    }
   }
  }
  else 
  {
   big=0;
   for(i=0;i<n;i++)
   {
    for(j=0;j<n;j++)
    {
     x1=x[i];y1=y[i];
     x2=x[j];y2=y[j];
     
     dist=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
     if(dist>big)
     big=dist;
     
    }
   }   
  }
  
 cout<<setprecision(11)<<big<<"\n"; 
 }

}
