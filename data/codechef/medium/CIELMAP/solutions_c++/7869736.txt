    LL pow_r(LL base,LL n)
    {
    	LL d=base,x=1ll;
    	while(n>0)
    	{
    		if(n&1)
    			x=(x*d);
    		d=(d*d);
    		n>>=1;
    	}
    	return x;
    }

    bool isallowed(PII a,PII b,PII c,PII d)
    {
    	int sign1=(c.S-a.S)*(b.F-a.F)-(b.S-a.S)*(c.F-a.F);
    	int sign2=(d.S-a.S)*(b.F-a.F)-(b.S-a.S)*(d.F-a.F);
    	if((sign1*sign2)<0)return false;
    	return true;
    }
