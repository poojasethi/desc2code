
    #include <stdio.h>
    #include <math.h>
     
    int main ()
    {
    int t, n, i, j, k, l, dist, max, a, b;
    int x[1000], y[1000];
    double ans;
    scanf ("%d", &t);
    while (t--)
    {
    scanf ("%d", &n);
    for (i=0; i<n; i++)
    scanf ("%d %d", &x[i], &y[i]);
    max = 0;
    if (n == 4)
    {
    for (i=0; i<n; i++)
    {
    for (j=i+1; j<n; j++)
    {
    k = 0;
    while (k == i || k == j)
    k++;
    l = k + 1;
    while (l == i || l == j)
    l++;
    a = (y[i] - y[j]) * x[k] - (x[i] - x[j]) * y[k] + (x[i] * y[j] - x[j] * y[i]);
    b = (y[i] - y[j]) * x[l] - (x[i] - x[j]) * y[l] + (x[i] * y[j] - x[j] * y[i]);
    dist = (x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j]);
    if (((a > 0 && b > 0) || (a < 0 && b < 0)) && dist > max)
    max = dist;
    }
    }
    }
    else
    {
    for (i=0; i<n; i++)
    {
    for (j=i+1; j<n; j++)
    {
    dist = (x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j]);
    if (dist > max)
    max = dist;
    }
    }
    }
    ans = sqrt((double)max);
    printf ("%lf\n", ans);
    }
    return 0;
    }
