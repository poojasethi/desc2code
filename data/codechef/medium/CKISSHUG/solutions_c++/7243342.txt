#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#define mod 1000000007
using namespace std;
typedef long long ll;
ll pow(ll base, ll exponent, int modulus)
{
    ll result = 1;
    while (exponent > 0)
    {
        if (exponent % 2 == 1)
            result = (result * base) % modulus;
        exponent = exponent >> 1;
        base = (base * base) % modulus;
    }
    return result;
}
int main()
{
    ll t,n,ans;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        if(n%2==0)
        {
            ans=(2*(pow(2,n/2,mod)-1))%mod;
            ans=(ans+pow(2,n/2,mod))%mod;
        }
        else
        {
            ans=(2*(pow(2,(n+1)/2,mod)-1))%mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
