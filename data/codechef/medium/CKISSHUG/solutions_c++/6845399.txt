#include <bits/stdc++.h>
using namespace std;
#define gi(x) scanf("%lld",&x) 
#define pb push_back
#define mp make_pair
#define all(p) p.begin(),p.end()
typedef long long ll;
ll mod=1000000007ll;
ll power(ll n){
	if(n==0)
		return 1ll;
	if(n==1)
		return 2ll;
	ll p=power(n/2);
	p=(p*p)%mod;
	if(n%2)
		p=(p*2)%mod;
	return p;
}
int main(){
	ll t,n;
	gi(t);
	while(t--){
		gi(n);
		if(n%2){
			ll k=n/2;
			ll ans=1;
			ans+=(2*power(k+1))%mod;
			ans=(ans+mod-3)%mod;
			printf("%lld\n",ans);

		}
		else {
			ll k=(n-1)/2;
			ll ans=1;
			ans+=(2*power(k+1))%mod;
			ans=(ans+mod-3)%mod;
			ans=(ans+power(n/2))%mod;
			printf("%lld\n",ans);
		}
	}
}
