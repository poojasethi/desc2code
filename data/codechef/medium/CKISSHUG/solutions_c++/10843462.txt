#include<stdio.h>
#include<assert.h>

#define ll long long
#define M 1000000007

ll pw(ll a, ll b)
{
  ll r;
  if(b==0) return 1;
  r = pw(a,b/2);
  r = (r*r)%M;
  if(b%2) r = (r*a)%M;
  return r;
}

int main(){
  int T, N;
  ll n1, n2, res;

  assert( scanf("%d",&T)==1 );
  assert( 0<=T && T<=1000 );
  while(T--){
    assert( scanf("%d",&N)==1 );
    assert( 1<=N && N<=1000000000 );

    N++;
    n1 = N/2; n2 = N - n1;
    res = pw(2,n1) + pw(2,n2) - 2;

    printf("%d\n",(int)(res%M));
  }

  return 0;
}
