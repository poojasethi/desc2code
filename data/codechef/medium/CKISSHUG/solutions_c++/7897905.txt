#include<bits/stdc++.h>
#define mod 1000000007
#define gc getchar
typedef long long int ll;
using namespace std;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
ll fastPower(ll base,ll power,ll M)
{
        ll result=1;
        while (power>0)
        {
                if (power%2==1)
                        result = (result*base)%M;
                base = (base*base)%M;
                power/=2;
        }
        return result;
}
int main()
{
    ll t,n,x,y,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n%2==0)
        {
            x=n/2+1;
            y=n/2;
        }
        else
        {
            x=(n+1)/2;
            y=x;
        }
        i=fastPower(2,x,mod);
        j=fastPower(2,y,mod);
        cout<<(i+j-2)%mod<<'\n';
    }
    return 0;
}
