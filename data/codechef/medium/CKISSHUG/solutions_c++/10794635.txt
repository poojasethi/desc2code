
#include<bits/stdc++.h>
#include<cmath>
using namespace std;
const int mod = 1e9 + 7;
long long power(long long a,long long b)
{
    long long ans=1;
    while(b)
    {
        if(b&1)
        {
            ans *= a;
            ans %= mod;
        }
        a *= a;
        a %= mod;
        b = b/2;

    }
    return ans;
}
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        long double n;
        long long ans;
        cin>>n;
        n = (n+1);
        //cout<<ceil(n/2)<< " "<<floor(n/2)<<endl;
        ans = power(2,ceil(n/2)) + power(2,floor(n/2)) -2;
        cout<<ans%mod<<endl;
    }
    return 0;
}
