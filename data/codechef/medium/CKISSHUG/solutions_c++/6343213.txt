#include<bits/stdc++.h>
using namespace std;

#define lli long long
#define pb push_back
#define mod 1000000007
#define gc getchar_unlocked


void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

lli power(int y){
	lli id = 1;
	lli x = 2LL;
	while(y>0){
		if(y&1)
			id= ( (id%mod)*(x%mod) )%mod;
		x=( (x%mod)*(x%mod) )%mod;
		y = y>>1;
	}
	return id;
}

void solve(int n){
	int t = n+1;
	int t1 = ceil(1.0*t/2);
	int t2 = floor(1.0*t/2);
	
	//cout<<t1<<" "<<t2<<endl;
	lli ans = 0;
	ans = ( (ans%mod)+(power(t1)%mod) )%mod;
	ans = ( (ans%mod)+(power(t2)%mod) )%mod;
	
	//cout<<ans;
	ans = ( (ans-2LL+mod)%mod )%mod;
	printf("%lld\n",ans);
}

int main(){
	int t,n;
	scanint(t);
	while(t--){
		scanint(n);
		if(n==1){
			printf("2\n");
			continue;
		}
		if(n==2){
			printf("4\n");
			continue;
		}
		solve(n);
	}
	
}
