#include<bits/stdc++.h>
using namespace std;

#define ll long long
#define MOD 1000000007

ll fast_expo(ll a,ll b,ll c){
	ll x=1,y=a;
	while(b>0){
		if(b&1){
			x=(x*y)%c;
		}
		y=(y*y)%c;
		b=b/2;
	}
	return x%c;
}

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t;
	cin >> t;
	while(t--){

		int n;
		cin >> n;
		ll ans=0;
		if(n&1){
			ans=fast_expo(2,(n+3)/2,MOD)-2;
		}
		else ans=3*fast_expo(2,n/2,MOD)-2;
		ans=ans%MOD;
		cout << ans << "\n";

	}
	return 0;
}
