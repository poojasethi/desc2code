#include<iostream>
#define mod 1000000007
using namespace std;
long long int mypow(int i,long long int j)
{
	long long int k,l;
	k=i;
	if(j==0)
	return 1;
	else if(j==1)
	return i;
	else if (j%2==0)
	{
		l=mypow(i,j/2)%mod;
		l=l*l;
		l%=mod;
		return l;
		
	}
	else
	{
		l=mypow(i,j/2)%mod;
		l=l*l;
		l%=mod;
		l=l*k;
		l%=mod;
		
		return l;
		
	}
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int h,n,w,temp;
		cin>>n;
		w=0;
		if(n % 2==0){
			h=n/2;
			temp =mypow(2,h+1);
			w = w + temp - 1;
			if(temp%2==0)
			w = w+ temp/2 - 1;
			else
			w = w + mypow(2,h) - 1;
			w = w % mod;
		}
 
		else{
	        h=n/2;
			w = 2 * (mypow(2,h+1) - 1);
			w = w % mod;
		}
 
		cout<<w<<endl;
	}
   return 0;

}
