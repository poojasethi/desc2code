#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define F first
#define S second
#define maxn 1005
#define M 1000000007
using namespace std;
 
LL s(LL base,LL exp)
{
	LL r=1;
	while(exp)
	{
		if(exp&1)
		{
			r=(r*base)%M;
		}
		base=base*base;
		base=base%M;
		exp>>=1;
		//cout<<base<<" "<<r<<" "<<exp<<endl;
	}
	//cout<<r<<endl;
	return r;
}
 
int main()
{
	int t,n1,n2,n;
	sc(t);
	LL ans;
	//cout<<s(2,5)<<endl;
	while(t--)
	{
		sc(n);
		ans=0;
		n++;
		n1=n/2;
		n2=n-n1;
		ans+=s(2LL,(LL)n1)+s(2LL,(LL)n2);
		ans-=2;
		if(ans<0)
		{
			ans+=M;
		}
		pl(ans%M);
	}
	return 0;
}
