#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;


long long int fastpower(long long int a, long long int b)
{
if(b==0)
return 1;

else if(b==1)
return a;

else
{
long long int r=fastpower(a,b/2);

if(b%2!=0)
return ((r*r)%mod*a)%mod;

else
return (r*r)%mod;
}
} 

int main()

{
std::ios::sync_with_stdio(false);

int T;
cin>>T;

while(T--)
{
long long int N, ans;
cin>>N;

if(N%2==0)
ans=(((3*fastpower(2, N/2))%mod)-2+mod)%mod;

else
ans=(fastpower(2,((N+3)/2))%mod-2+mod)%mod;

cout<<ans<<"\n";

}
return 0;
}
