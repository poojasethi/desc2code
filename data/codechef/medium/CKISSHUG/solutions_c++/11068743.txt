#include <bits/stdc++.h>
#include <ext/rope>
using namespace __gnu_cxx;
using namespace std;

int be (int e)
{
	if (!e) return 1;
	long long r = be(e/2);
	return (r*r*(e%2 ? 2 : 1))%1000000007;
}

int main()
{
	int T,x;
	cin >> T;
	while (T--)
	{
		cin >> x;
		cout << (be((x+3)/2)-(2-(x%2 ? 0 : be((x+1)/2))))%1000000007 << endl;
	}
}
