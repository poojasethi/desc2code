#include <cstdio>

using namespace std;

#define ll long long
#define mod 1000000007

ll pw (ll a, ll b){
    ll r;
    if(b==0) return 1;
    r = pw(a,b/2);
    r = (r*r) % mod;
    if(b%2) r = (r*a)%mod;
    return r;
}

int main()
{
    int t, N;
    ll n1,n2,res;

    scanf("%d", &t);
    while(t--){
        scanf("%d",&N);

        N++;
        n1 = N/2;
        n2 = N - n1;
        res = pw(2,n1) + pw(2,n2) - 2;
        printf("%d\n", (int) (res%mod));
    }
    return 0;
}
