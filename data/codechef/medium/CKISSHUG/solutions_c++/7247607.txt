#include<bits/stdc++.h>
#define m 1000000007
using namespace std;
long long int exps(long long int n,int x)
{
    if(x==1)
        return n;
    if(x==0)
        return 1;

    if(x&1)
    {
        long long int u =((n%m)*(n%m))%m;
        return((n%m)*(exps(u,(x-1)/2)))%m;
    }
    else
    {
        long long int u=((n%m)*(n%m))%m;
        return (exps(u,x/2))%m;
    }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        cin>>n;
        int u=n>>1;

            long long int r;
        if(n&1)
        {
                    r=(exps(2,u+2)-2+m)%m;

        }
        else
        {
r=(3*exps(2,u)-2+m)%m;

        }
        cout<<r<<endl;

    }



}
