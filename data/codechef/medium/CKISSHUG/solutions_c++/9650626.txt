#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL MOD = 1e9 + 7;
LL power(LL a, LL b){
	LL ans = 1;
	while(b > 0){
		if(b&1) ans = (ans * a) % MOD;
		a = (a * a) % MOD;
		b >>= 1;
	}
	return ans;
}
int main(){
	int t;
	cin >> t;
	while(t--){
		LL N;
		cin >> N;
		LL ans;
		if(N%2 == 0){
			ans = (3 * power(2, N/2)) % MOD;
			ans = (ans + MOD - 2) % MOD;
		} else{
			ans = (power(2, (N/2) + 1) + MOD - 1) % MOD;
			ans = (ans << 1) % MOD;
		}
		cout << ans << "\n";
	}
	return 0;
}
