#include<stdio.h>
#include<math.h>
 
unsigned long long int power(unsigned long long int a,unsigned long long int b)
{ unsigned long long int c;
    if(b==0)
      return 1;
    c=power(a,b/2);
    if(b%2==0)
      return ((c*c)%1000000007);
      else
        return ((c*c*2)%1000000007);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        unsigned long long int n;
        scanf("%llu",&n);
        if(n%2)
          printf("%llu\n",(2*(power(2,(n+1)/2)-1))%1000000007);
          else
          printf("%llu\n",(3*power(2,n/2)-2)%1000000007);
    }
 
    return 0;
}
