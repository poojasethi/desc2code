#include<iostream>
using namespace std;
#define M 1000000007
class kissMe{
	long long int N;
	long long int ans;
	public:
		kissMe()
		{
			cin>>N;
			ans=0;
		}
		long long int power(long long int  i)
		{
			if(i==0)
			return 1;
			long long int res=power(i/2)%M;
			if(i%2==0)
			return res%M*res%M;
			else
			return 2*res%M*res%M;
		}
		void totalKiss()
		{       
				ans+=power((N+3)/2);
				ans-=2;
				if(N%2==0)
				ans+=power(N/2);
			   cout<<(ans+M)%M<<endl;
		}
};
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		kissMe k;
		k.totalKiss();
	}
	return 0;
}
