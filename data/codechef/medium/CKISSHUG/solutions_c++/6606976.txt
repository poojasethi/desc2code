#include<cstdio>
using namespace std;
#define lli long long int
#define MOD 1000000007
lli fun(lli n)
{
	lli ans=1,p=2;
	while(n>0)
	{
		if(n%2==1)
		{
			ans=(ans*p)%MOD;
		}
		p=(p*p)%MOD;
		n=n/2;
	}
	return ans;
}
int main()
{
	lli ans,t,n,val;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		val=fun(n/2);
		if(n%2==0)
		{
			ans=(2*(2*val-1))-val;
		}
		else
		{
			ans=(2*(2*val-1));
		}
		ans=ans%MOD;
		printf("%lld\n",ans);
	}
	return 0;
}
