#include<iostream>
#include<cstdio>
#define MOD  1000000007
using namespace std;
long long int get_expo(long long int n)
{
    long long int ans = 1;
    long long int mul = 2;
    //printf("%lld\n",n);
    while(n != 0)
    {
        if(n % 2 == 1)
        {
            ans *= mul;
            if(ans >= MOD)
                ans %= MOD;
        }
        mul *= mul;
        if(mul >= MOD)
            mul %= MOD;
        n /= 2;
    }
    return ans;
}
int main()
{
   int t,n;
   scanf("%d",&t);
   while(t--)
   {
       scanf("%d",&n);
       int tmp = (n-1)/2;
       tmp++;
       long long int exp = get_expo(1ll*tmp);
       long long ans = 0;
       ans = exp - 1;
       if(n % 2 == 0)
        exp *= 2;
       if(exp >= MOD)
        exp %= MOD;
       ans += exp-1;
       if(ans >= MOD)
        ans %= MOD;
       if(ans < 0)
        ans += MOD;
       printf("%lld\n",ans);
   }
   return 0;
}
