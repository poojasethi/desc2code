#include <bits/stdc++.h>
using namespace std;
const long long  M=1000000007;

typedef long long ll;
ll power(ll a, ll b){
  ll r;
  if(b==0) return 1;
  r = power(a,b/2);
  r = (r*r)%M;
  if(b%2) r = (r*a)%M;
  return r;
}

int main()
{
  int t;
  scanf("%d",&t);
  while(t--){
    ll n;
    scanf("%lld",&n);
    ll ans;
    if(n%2==0){
         ll part1=power(2,(n/2+1));
         ll part2=power(2,n/2);
         ans=part1+part2-2;
         printf("%lld\n",ans%M);

    }
    else{
       ans=2*power(2,((n+1)/2))-2;
    printf("%lld\n",ans%M);

    }


  }
    return 0;
}
