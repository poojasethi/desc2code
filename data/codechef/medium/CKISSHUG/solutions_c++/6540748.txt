#include<iostream>
using namespace std;
#define M 1000000007
long long int findpower(long long int N)
{
   if(N==0)
      return 1;
   if(N==1)
      return 2;
   long long int ans=findpower(N/2);
   ans=ans*ans%M;
   if(N%2)
   return ans*2%M;
   return ans;
}
int main()
{
   long long int T,N,ans;
   cin>>T;
   while(T--)
   {
      cin>>N;
      ans=findpower((N+1)/2);
      if(N%2)
         ans=ans*2-2;
      else
         ans=ans*3-2;
      ans=(ans+M)%M;
      cout<<ans<<endl;
   }
return 0;
}
