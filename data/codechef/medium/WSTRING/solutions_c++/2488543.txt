#include<stdio.h>
#include<string.h>
#include<iostream>

using namespace std;
 int main(){
	 int t,arr[10004][26],run[26],tot;
	 char s[10004];
	 scanf("%d",&t);
	 while(t--){
	 	scanf("%s",s);
		tot=0;
		memset(run,0,26*sizeof(int));
		for(int i=0;s[i]!='\0';i++){
			if(s[i]=='#'){
				for(int i=0;i<26;i++)
					arr[tot][i]=run[i];
				tot++;
			}
			else
				run[s[i]-'a']++;
		}
		for(int i=0;i<26;i++)
			arr[tot][i]=run[i];
		tot++;
		int ans =0;
		for(int i=3;i<tot;i++){
			int len =0;
			for(int j=i-3;j<=i;j++){
				int mx =0;
				for(int k=0;k<26;k++){
					if(j==i-3)
						mx = max(mx,arr[j][k]);
					else if(j==i)
						mx = max(mx,arr[tot-1][k]-arr[j-1][k]);
					else
						mx = max(mx,arr[j][k]-arr[j-1][k]);
				}
				if(mx==0){
					len=0;
					break;
				}
				//printf("%d\n",mx);
				len+=mx;
			}
			///printf("--------------------\n");
			ans=max(ans,len);
		}
		if(ans==0)
			printf("0\n");
		else
		printf("%d\n",ans+3);		
	 }
	 return 0;
 }
