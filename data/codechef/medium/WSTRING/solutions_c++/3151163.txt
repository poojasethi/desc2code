#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
 
 
int main()
{
	int test_case;
	cin>>test_case;
	while(test_case--)
	{
		char str[10005];
		cin>>str;
		int l=strlen(str);
		int lhash[26]={0};
		int rhash[26]={0};
		int hash[26]={0};
		int lmax=0,rmax=0,max=0,j=0,k=0,m=0,hashcount=0;
		int lmaxArr[10004],rmaxArr[10004],maxArr[10004];
		for(int i=0;i<l;i++)
		{
			if(str[i]=='#')
			{
				hashcount++;
				lmaxArr[j++]=lmax;
				maxArr[m++]=max;
				for(int i=0;i<26;i++) hash[i]=0;
				max=0;
			}
			else
			{
				int x=++lhash[int(str[i])-97];
				if(x>lmax) lmax=x;
				int z=++hash[int(str[i])-97];
				if(z>max) max=z;
			}
			if(str[l-i-1]=='#')
			rmaxArr[k++]=rmax;
			else
			{
				int y=++rhash[int(str[l-i-1])-97];
				if(y>rmax) rmax=y;
			}
		}
		if(hashcount<3) cout<<"0\n";
		else
		{
			int maxCount=0;
			int h=0;
			for(int i=0;i<hashcount;i++)
			{
				int p; 
				if(!lmaxArr[i]||!rmaxArr[hashcount-i-3]||!maxArr[i+2]||!maxArr[i+1]) p=0;
				else
				 p=lmaxArr[i]+maxArr[i+1]+maxArr[i+2]+rmaxArr[hashcount-i-3];
			//	cout<<lmaxArr[i]<<" "<<maxArr[i+1]<<" "<<maxArr[i+2]<<" "<<rmaxArr[hashcount-i-3]<<"\n";
				if(p>maxCount) maxCount=p;
			}
			if(maxCount)
			cout<<maxCount+3<<"\n";
			else cout<<"0\n";
		}
	}
	return 0;
}
