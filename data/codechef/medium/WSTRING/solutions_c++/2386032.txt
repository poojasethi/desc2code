#include <iostream>
#include<cstring>
#include<cstdio>
using namespace std;

int main()
{
 int i,j,k,t;
    cin>>t;
    while(t--)
    {
        int left[10000]={0},right[10000]={0},mid[10000]={0},lefti[26]={0},righti[26]={0},midi[26]={0},size=0;
        char s[10100];
        cin>>s;
        for(i=0,k=0;s[i];i++)
        {
            if(s[i]=='#')
            {
                int maxi=0;
                for(j=0;j<26;j++)
                {
                    maxi=max(maxi,lefti[j]);
                }
                left[k++]=maxi;
            }
            else
            {
                lefti[s[i]-'a']++;
            }
        }
        size=k;
       // cout<<" "<<size;
        for(i=0,k=0;s[i];i++)
        {
            if(s[i]=='#')
            {
                int maxi=0;
                for(j=0;j<26;j++)
                {
                    maxi=max(maxi,midi[j]);
                    midi[j]=0;
                }
                mid[k++]=maxi;
            }
            else
            {
                midi[s[i]-'a']++;
            }
        }
        k=size-1;
        for(i=i-1,k;i>=0;i--)
        {
            //cout<<k<<endl;
            if(s[i]=='#')
            {
                int maxi=0;
                for(j=0;j<26;j++)
                {
                    maxi=max(maxi,righti[j]);
                }
                right[k--]=maxi;
            }
            else
            {

                righti[s[i]-'a']++;
            }
        }
       // size=k; cout<<"!"<<size;
       /* for(i=0;i<size;i++)
        {
            cout<<" left "<<left[i]<<" mid "<<mid[i]<<" right "<<right[i]<<endl;
        }
        cout<<endl;*/

        if(size<3)
        printf("0\n");
        else
        {
            int ans=0;
            for(i=0;i<size-2;i++)
            {
                if(left[i]==0||mid[i+1]==0||mid[i+2]==0||right[i+2]==0)
                continue;
                ans=max(ans,left[i]+mid[i+1]+mid[i+2]+right[i+2]);
            }
            if(ans!=0)
            printf("%d\n",ans+3);
            else
            printf("0\n");
        }
    }
    return 0;
}
