// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t, i, n, j, mx, mx1, mx2, mx3, mx4, p, q, r;
    string s;
    cin >> t;
    while (t--) {
        cin >> s;
        int a[26][s.size()+1];
        for (i = 0; i < 26; i++)
            a[i][0] = 0;
        vector <int> v;
        n = s.size();
        for (i = 1; i <= n; i++) {
            for (j = 0; j < 26; j++)
                a[j][i] = a[j][i-1];
            if (s[i-1] == '#')
                v.push_back(i);
            else
                a[s[i-1]-97][i]++;
        }
        mx = 0;
        for (i = 2; i < v.size(); i++) {
            p = v[i-2];
            q = v[i-1];
            r = v[i];
            mx1 = mx2 = mx3 = mx4 = 0;
            for (j = 0; j < 26; j++) {
                mx1 = max(mx1, a[j][p]-a[j][0]);
                mx2 = max(mx2, a[j][q]-a[j][p]);
                mx3 = max(mx3, a[j][r]-a[j][q]);
                mx4 = max(mx4, a[j][n]-a[j][r]);
            }
            if (mx1 && mx2 && mx3 && mx4)
                mx = max(mx, 3+mx1+mx2+mx3+mx4);
        }
        cout << mx << "\n";
    }
    return 0;
}
