typedef long long int lli;

using namespace std;

char str[10010];
int que[10010],size;
int beg[10010],size1;
int en[10010],size2;

int main()
{
    //freopen("inp.txt","r",stdin);
    int t,n;
    scanf("%d",&t);
    int cnt[30];
    while(t--)
    {
    	size=size1=size2=0;
    	scanf("%s",str);
    	mset(cnt,0);
    	n=strlen(str);
    	for(int i=0;i<n;i++)
    	{
    		if(str[i]=='#')
    		{
    			int ma=0;
    			for(int j=0;j<26;j++) if(ma<cnt[j]) ma=cnt[j];
    			que[size++]=ma;
    			mset(cnt,0);
    		}
    		else cnt[str[i]-'a']++;
    	}

        mset(cnt,0);
        for(int i=0;i<n;i++)
        {
            if(str[i]=='#')
            {
                int ma=0;
                for(int j=0;j<26;j++) if(ma<cnt[j]) ma=cnt[j];
                beg[size1++]=ma;
            }
            else cnt[str[i]-'a']++;
        }

        mset(cnt,0);
        size2=size1-1;
        for(int i=n-1;i>=0;i--)
        {
            if(str[i]=='#')
            {
                int ma=0;
                for(int j=0;j<26;j++) if(ma<cnt[j]) ma=cnt[j];
                en[size2--]=ma;
            }
            else cnt[str[i]-'a']++;
        }
        // for(int i=0;i<size;i++) printf("%d ",beg[i]);
        // printf("\n");
        // for(int i=0;i<size;i++) printf("%d ",que[i]);
        // printf("\n");
        // for(int i=0;i<size;i++) printf("%d ",en[i]);
        // printf("\n");
    	int ma=0;
    	for(int i=0;i<=size-3;i++)if(beg[i]&&que[i+1]&&que[i+2]&&en[i+2])
    		if(ma<beg[i]+que[i+1]+que[i+2]+en[i+2]) ma=beg[i]+que[i+1]+que[i+2]+en[i+2];
    	if(ma>0)printf("%d\n",ma+3);
    	else printf("0\n");
    }
    return 0;
    
}
