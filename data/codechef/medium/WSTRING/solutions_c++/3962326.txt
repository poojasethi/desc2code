#include<stdio.h>
#include<string.h>
int a[10000][27]={0};
int res[27]={0};
int max=-1;
int main()
{
   int t,i=0;
   scanf("%d",&t);
   char str[100001];

   while(t--)
   {
       i=0;
       memset(a,0,sizeof(a));
       memset(res,0,sizeof(res));
       max=-1;
       scanf("%s",&str);
       int count=0,vmax=0,rmax=0;

       while(str[i]!='\0')
       {

           if(str[i]=='#')
           {
               a[count][26]=vmax;
               count++;
               vmax=0;
           }

           else
           {
               a[count][str[i]-'a']+=1;
               if(a[count][str[i]-'a']>vmax)
                vmax=a[count][str[i]-'a'];

               if(count>=2)
               {
                 res[str[i]-'a']+=1;

                 if(res[str[i]-'a']>rmax)
                    rmax=res[str[i]-'a'];
               }

          }

          i++;

      }

      res[26]=rmax;
      a[count][26]=vmax;

     // for(i=0;i<count;i++)
       // printf("%d\n",a[i][26]);


      for(i=0;i<=count;i++)
      {

       rmax=0;
       for(int k=0;k<26;k++)
       {
           res[k]=res[k]-a[i+2][k];

           if(res[k]>rmax)
            rmax=res[k];


       }

       res[26]=rmax;

        if(a[i][26]!=0&&a[i+1][26]&&a[i+2][26]!=0&&res[26]!=0)
        {
            // printf("value of i is %d",i);
             if(max<(a[i][26]+a[i+1][26]+a[i+2][26]+res[26]))
                     max=a[i][26]+a[i+1][26]+a[i+2][26]+res[26];
        }

        vmax=0;
      for(int k=0;k<26;k++)
      {
          a[i+1][k]+=a[i][k];

          if(vmax<a[i+1][k])
            vmax=a[i+1][k];
      }
      a[i+1][26]=vmax;


      }

   max=max==-1?0:max+3;
   printf("%d\n",max);

   }




}
