#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int a[100005];
ll sum(int a,int b)
{
    if(a>b)
        return 0;
    return (1LL*b*(b+1)>>1)-(1LL*a*(a-1)>>1);
}
void print(ll x)
{
    if(x%2)
        cout<<"Chef\n";
    else
        cout<<"Mom\n";

}
int main()
{
     int t;
     cin>>t;
     while(t--)
     {
         int n,k,ans;
         cin>>n>>k;
         for(int i=0;i<k;i++)
         {
             cin>>a[i];
         }
         if(k==0)
         {
             ans=sum(1,n)+1;
             print(ans);
         }
         else
         {
             sort(a,a+k);
             ll mx=sum(1,a[0]-1);
             ans=0;
             a[k]=n+1;
             for(int i=0;i<k;i++)
             {
                 if(a[i]>mx)
                 {
                    ans=a[i];
                     break;
                 }
                 mx+=sum(a[i]+1,a[i+1]-1);
             }
             if(ans)
                print(ans);
             else
                print(mx+1);
         }
     }
     return 0;
}
