#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
#define gc getchar_unlocked 
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
ll compute(int a,int b)
{
	int no = b-a-1;
	if(no==1) return (ll) (a+1);
	else return (ll)no*(a+b)/2;
}
int main() 
{
	int t;
	scanint(t);
	while(t--)
	{
		int n,q;
		scanint(n);
		scanint(q);
		int* arr = new int[q];
		for(int i=0;i<q;i++)
			scanint(arr[i]);
		sort(arr,arr+q);
		if(q==0)
		{
			if(compute(0,n+1)%2) printf("Mom\n");
			else printf("Chef\n");
		}
		else if(arr[0]==1) printf("Chef\n");
		else if(arr[0]==2) printf("Mom\n");
		else
		{
			ll sum = 0;
			int i;
			sum = compute(0,arr[0]);
			for(i=1;i<q;i++)
			{
				if(sum>=arr[i-1])
				{
					if(arr[i]!=arr[i-1]+1)	sum+=compute(arr[i-1],arr[i]);
				}
				else break;
			}
			if(sum>=arr[i-1])
			{
				if(arr[i-1]!=n)
					sum+=compute(arr[i-1],n+1);
			}
			if(sum%2) printf("Mom\n");
			else printf("Chef\n");
		}
	}
	return 0;
}
