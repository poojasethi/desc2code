#include<stdio.h>
#include<iostream>
using namespace std;
#include<algorithm>
 
unsigned long long sum(long long a,long long b)
{
	if(a>b)
	return 0;
	
	return ((b*(b+1))-(a*(a-1)))/2;
}
 
int main()
{
	int t,i,flag;
	long long n,a,b,k;
	unsigned long long ans,final;
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		flag=0;
		scanf("%lld%lld",&n,&k);
		long long ar[k+1];
		
		for(i=1;i<=k;i++)
		scanf("%lld",&ar[i]);
		ar[0]=0;
		sort(ar,ar+k+1);
		
		for(i=1;i<=k;i++)
		{
			a=ar[i-1];
			b=ar[i];
			
			if(a<=b)
			ans+=sum(a+1,b-1);
			
			if(ans<ar[i])
			{
			  flag=1;
			  final=ar[i];
		      break;
			}
		}
		
		if(flag==0)
		{
		 ans+=sum(ar[k]+1,n);
	     final=ans+1;
		}
		
		//printf("final=%llu\n",final);
		
		if(final%2==0)
		printf("Mom\n");
		
		else
		printf("Chef\n");
	}
	return 0;
}
