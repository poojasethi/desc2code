#include<stdio.h>
int main()
{
    long long int t,n,k,q,l,r,s,e,i,j,ans;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld%lld",&n,&k,&q);
        char ss[n+10];
        long long int A[n+10],B[n+10],C[2],E[n+10],D[n+10];
        scanf("%s",ss);
        for(i=0;i<n;i++)
        {
            A[i]=ss[i]-48;
        }
        C[0]=0;
        C[1]=0;
        j=0;
        for(i=0;i<n;i++)
        {
            if(i>0)
            {
                C[A[i-1]]--;
            }
            while(j<n)
            {
                if(C[0]<=k&&C[1]<=k)
                {
                    C[A[j]]++;
                    j++;
                }
                else
                {
                    j--;
                    C[A[j]]--;
                    break;
                }
            }
            if(j==n)
            {
                if((C[0]>k)||(C[1]>k))
                {
                    j--;
                    C[A[j]]--;
                }
            }
            B[i]=j-1;
        }

        for(i=0;i<n;i++)
        {
            B[i]=B[i]-i+1;
        }

        D[0]=B[0];
        for(i=1;i<n;i++)
            D[i]=D[i-1]+B[i];


        E[n-1]=n-1;
        j=n-1;
        for(i=n-2;i>=0;i--)
        {
            while(j>=0&&B[j]>i+1-j)
            {
                j--;
            }
            E[i]=j+1;
        }

        while(q--)
        {
            scanf("%lld%lld",&l,&r);
            l=l-1;
            r=r-1;
            ans=0;

            i=E[r];

            if(B[l]>=r+1-l)
            {
                ans=r+1-l;
                ans=(ans*(ans+1))/2;
            }
            else
            {
                if(l==0)
                {
                    ans=r+1-i;
                    ans=(ans*(ans+1))/2;
                    ans=ans+D[i-1];
                }
                else
                {
                    ans=r+1-i;
                    ans=(ans*(ans+1))/2;
                    ans=ans+D[i-1]-D[l-1];
                }
            }
            printf("%lld\n",ans);
        }
    }
    return 0;
}
