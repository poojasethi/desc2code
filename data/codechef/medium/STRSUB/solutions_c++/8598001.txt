#include <bits/stdc++.h> 
#define mod 1000000007
#define gc getchar
using namespace std;
typedef long long int lli;
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int far[100005];
lli sumfar[100005];
char s[100001];
int k,n;
void process()
{
    int c1=0,c0=0,i,j;
    j=1;
    if(s[1]=='1') c1++;
    else c0++;
    for(i=1;i<=n;i++)
    {
        while(j<=n && c1<=k && c0<=k)
        {        
            j++;
            if(j>n) break;   
            if(s[j]=='1') c1++;
            else c0++; 
        }
        far[i]=j;
        if(s[i]=='1') c1--;
        else c0--;
    }
    for(i=1;i<=n;i++)
        sumfar[i]=sumfar[i-1]+far[i];
}
 
int main()
{
    int t,q,i,l,r;    
    scanint(t);
    while(t--)
    {
        scanint(n);scanint(k);scanint(q);
        scanf("%s",s+1);
        sumfar[0]=0;
        process();
        while(q--)
        {
            scanint(l);scanint(r);
            k=upper_bound(far+l,far+r+1,(lli)r)-far;
            k--;
            lli ans=sumfar[k]-sumfar[l-1]+(lli)(r-k)*(r+1)-(lli)r*(r+1)/2+(lli)l*(l-1)/2;
            printf("%lld\n",ans);
        }
    }
}
