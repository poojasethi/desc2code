#include<iostream>
#include<string.h>
#include<stdio.h>
using namespace std;
#define M 1000000007
#define ll long long int
ll t,dp[110][110],a0,a1,a2,a3,s,C[110];
int scan()
{
    int noRead=0;
    char p= getchar_unlocked();
    for(;p<33;)
    {
     p=getchar_unlocked();
    };
    while(p>32)
    {
     noRead = (noRead << 3) + (noRead << 1) + (p - '0');
     p=getchar_unlocked();
    }
    return noRead;
};
ll comb(ll n,ll r)
{
    r=min(r,n-r);
    ll ans=1;
    for(int i=0;i<r;i++)ans=(ans*(n-i))%M;
    ll an2=1;
    for(int i=1;i<=r;i++)an2=(an2*i)%M;
    ll b=M-2,y=an2,an=1;
    while(b!=0)
    {
        if(b%2)an=(an*y)%M;
        b/=2;
        y=(y*y)%M;
    }
    return ((ans*an)%M);
}
int main()
{
    t=scan();
    //cout<<comb(3,2)<<endl;
    while(t--)
    {
        memset(dp,0,sizeof(dp));
        dp[0][0]=1;
        a0=scan();
        a1=scan();
        a2=scan();
        a3=scan();
        s=scan();
        for(int i=1;i<=s;i++)
        {
           int m=a0+a1*i+a2*i*i+a3*i*i*i;
           for(int j=0;j<=s;j++)C[j]=comb(m+j-1,m-1);
            for(int j=0;j<=s;j++)
            {
                int k=j,c=0;
                while(k>=0)
                {
                  //  cout<<i<<" "<<j<<" "<<k<<endl;
                    dp[i][j]=(dp[i][j]+(C[c]*dp[i-1][k])%M)%M;
                    k-=i;
                    c++;
                }
               // cout<<dp[i][j]<<endl;
            }
        }
        printf("%lld\n",dp[s][s]);
    }
}
