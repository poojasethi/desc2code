ll pow(ll a,int b)
{
	ll ans=1;
	while(b>0)
	{
		if(b&1)ans=ans*a%mod;
		b>>=1;a=a*a%mod;
	}
	return ans;
}
ll fac[260001],inv[260001];
ll cal(int a,int b)
{
	if(b>a)return 0;
	if(a<=260000)return fac[a]*inv[b]%mod*inv[a-b]%mod;
	else
	{
		ll ans=1;
		for(int i=1;i<=b;i++)
		ans=ans*(ll)(a-i+1)%mod;
		for(int i=1;i<=b;i++)
		ans=ans*pow((ll)i,mod-2)%mod;
		return ans;
	}
}
