#include<cstdio>
#include<iostream>
using namespace std;
long long a[101];
long long dp[101][101];
long long mod=1000000007;
long long fact[110];
long long exp1(long long a,long long p)
{
	if(a<2)return a;
	long long ret=1;
	while(p)
	{
		if(p&1)
			ret = (ret*a)%mod;
		p>>=1;
		a = (a*a)%mod;
	}
	return ret;
}
long long ncr(long long n,long long r)
{
	long long i;
	long long num=1;
	for(i=n-r+1;i<=n;i++){
		num=num*i;
		num%=mod;
	}
	return (num*exp1(fact[r],mod-2))%mod;
}
int main()
{
	int t;
	scanf("%d",&t);
	int i,j,k;
	long long e;
	fact[0]=1;
	for(e=1;e<110;e++)
		fact[e]=(e*fact[e-1])%mod;
	while(t--)
	{
		long long a1,a2,a3,a4;
		int s;
		scanf("%lld",&a1);
		scanf("%lld",&a2);
		scanf("%lld",&a3);
		scanf("%lld",&a4);
		scanf("%d",&s);
		for(i=0;i<=s;i++)
			a[i]=(a1+a2*i+a3*i*i+a4*i*i*i)%mod;
		for(i=0;i<=s;i++)
			dp[i][0]=0;
		for(i=0;i<=s;i++)
			dp[0][i]=1;
		for(i=1;i<=s;i++)
			for(j=1;j<=s;j++)
			{
				dp[i][j]=dp[i][j-1];
				for(k=1;k*j<=i;k++)
					dp[i][j]=(dp[i][j]+(dp[i-k*j][j-1]*ncr(k+a[j]-1,k))%mod)%mod;
			}
		printf("%lld\n",dp[s][s]);
	}
	return 0;
}
