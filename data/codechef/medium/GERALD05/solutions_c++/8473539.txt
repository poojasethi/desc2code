ll gt[MAXN];
ll igt[MAXN];
 
ll Pow(ll v, int mu){
	if(mu == 0) return 1ll;
	ll re = Pow(v, mu >> 1);
	re = re * re % MOD;
	if(mu & 1)
		re = re * v % MOD;
	return re;
}
 
ll rec(int pos, int rem){
	if(rem == 0) return 1;
	if(pos > rem) return 0;
	ll& re = dp[pos][rem];
	if(re >= 0) return re;
	if(pos == rem)
		return re = a[pos];
	
	re = 0;
	ll tmp;
	for(int i = 0; i * pos <= rem; i++){
		tmp = rec(pos + 1, rem - i * pos);
		if(tmp){
			ll mul = igt[i];
			for(int j = i + a[pos] - 1; j >= a[pos]; j--)
				mul = mul * j % MOD;
			re = (re + tmp * mul) % MOD;
		}
	}
	
	return re;
}
