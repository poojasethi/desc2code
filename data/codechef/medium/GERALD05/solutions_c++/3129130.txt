#include <iostream>
#include <stdio.h>
using namespace std;

int M=1000000007;
int rev[101];

int powr(int &n, int p)
{
  long long a; int b;
  if(p==0)
    return 1;
  else if(p==1)
    return n;
  else if(p%2==0)
  {
    a=powr(n, p/2);
    a*=a;
    if(a>=M)
      a%=M;
    b=a;
    return b;
  }
  else
  {
    a=powr(n, p/2);
    a*=a;
    if(a>=M)
      a%=M;
    a*=n;
    if(a>=M)
      a%=M;
    b=a;
    return b;
  }
}

void revC()
{
  rev[0]=0;
  rev[1]=1;
  for(int i=2; i<=100; i++)
  {
    rev[i]=powr(i, M-2);
  }
}

int ncr(int n, int r)		// r goes to 100 only
{
  if(r>n)
    return 0;
  else if(r==n)
    return 1;
  else if(r==0)
     return 1;
  else
  {
    long long val=1;
    for(int i=0; i<r; i++)
    {
      val*=n-i;
      if(val>=M)
	val%=M;
    }
    
    for(int i=2; i<=r; i++)
    {
      val*=rev[i];
      if(val>=M)
	val%=M;
    }
    
    return val;
  }
}

int main()
{
  int t;
  scanf("%d",&t);//cout<<ncr(10, 3)<<" "<<ncr(100, 2)<<" "<<ncr(6, 4)<<endl;
  revC();
  while(t--)
  {
    int s, a0, a1, a2, a3;
    scanf("%d %d %d %d",&a0, &a1, &a2, &a3);
    scanf("%d",&s);
    long long temp=1;
    long long ob_count[s+1], arr[s+1][s+1];
    for(long long n=0; n<=s; n++)
    {
      temp=a0+a1*n+a2*n*n+a3*n*n*n;
      if(temp>=M)
	temp%=M;
      ob_count[n]=temp;
    }
    ob_count[0]=0;
    
    // filling up the array now
    for(int i=0; i<=s; i++)
    {
      arr[0][i]=0;
      for(int j=i+1; j<=s; j++)
      {
	arr[j][i]=0;
      }
      arr[i][0]=1;
    }
    arr[0][0]=1;
    for(int i=1; i<=s; i++)
    {
      for(int j=1; j<=i; j++)
      {
	arr[j][i]=arr[j-1][i];
	int count=1, k;
	for( k=j; k<=i; k+=j, count++)
	{
	  arr[j][i]+=arr[j-1][i-k]*ncr(ob_count[j]+count-1, count);		// in place of 1 we have the ways to count
	  if(arr[j][i]>=M)
	    arr[j][i]%=M;
	}
      }
      
      for(int j=i+1; j<=s; j++)
	arr[j][i]=arr[i][i];
    }
//     for(int i=0; i<=s; i++)
//     {
//       for(int j=0; j<=s; j++)
// 	cout<<i<<j<<" "<<arr[i][j]<<"\t";
//       cout<<endl;
//     }
    printf("%lld\n",arr[s][s]);
  }
}
