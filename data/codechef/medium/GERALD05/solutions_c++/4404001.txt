#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define M 1000000007
#define lld signed long long int
lld a0,a1,a2,a3;
lld pow(lld a, lld b, lld MOD) {
lld x = 1, y = a;
    while(b > 0) {
        if(b%2 == 1) {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}
lld inv(lld a)
{
    return pow(a,M-2,M);
}
inline lld F(lld i)
{
	return a0+a1*i+a2*i*i+a3*i*i*i;
}
lld dp[101][101];
int main()
{
	int t;
	lld ncj;
	cin>>t;
	//t=500;
	while(t--)
	{
		int s;
		cin>>a0>>a1>>a2>>a3>>s;
		//a0=a1=a2=a3=1000;s=100;
		dp[0][0]=1;
		for(int i=1;i<=s;++i)
			for(int j=0;j<=s;++j)dp[i][j]=0;
		for(int i=1;i<=s;++i)
		{
			int n=F(i);
			ncj=1;
			for(int j=0;i*j<=s;++j)
			{
				if(j>0)ncj=(ncj*(((n+j-1)*inv(j)%M)))%M;
				for(int k=i*j;k<=s;++k)
				{
					dp[i][k]+=(dp[i-1][k-i*j]*ncj);
					dp[i][k]%=M;
				}
			}
		}
		printf("%llu\n",dp[s][s]);
	}
}
