typedef long long LL;






LL qntt[200];

LL inv[200];

LL mod=1000000007;

//return (b^p)%mod
LL modpow(LL b,LL p)
{
	if(!p) return 1;
	if(p%2)
		return (b*modpow(b,p-1))%mod;
	else
	{
		LL c=modpow(b,p/2);
		return (c*c)%mod;
	}
}

LL dp[110][200];

LL com[110][110];



int vis[200][200];
LL mem[200][200];

LL get(int place,int rest)
{
    if(place==0) return 1;
    if(rest==0) return 0;
    if(vis[place][rest])
        return mem[place][rest];

    LL ret=0;

    if(rest)
    ret=(ret+get(place,rest-1))%mod;
    ret=(ret+get(place-1,rest))%mod;


    vis[place][rest]=1;
    mem[place][rest]=ret;
    return ret;
}
