#include <stdio.h>
#include <iostream>
#include <string.h>
#define M 1000000007
#define ll long long

using namespace std;

int dp[105][105],l[105],a0,a1,a2,a3,s;
int c[105][105],combi[105][105],way[105][105];

int mul(ll x, ll y)
{
    return (int)((x*y)%M);
}

int f(int v,int k)
{
   // printf("Enter (%d,%d)\n",v,k);

    if(v==0) return dp[v][k]=1;

    if(k==0) return dp[v][k]=0;

    if(dp[v][k]!=-1) return dp[v][k];

    int i,r=0;

    for(i=0;;i++)
    {
        //printf("AT dp(%d,%d) : now %d\n",v,k,i);
        if(i*k>v) break;

        r=(r+  mul( way[k][i] ,  f(v-i*k,k-1) )  )%M;
    }

   // printf("dp(%d,%d) = %d\n",v,k,r);

    return dp[v][k]=r;
}

ll bigmod(ll a,ll x)
{
    if(x==0) return 1;

    if(x%2) return (a*bigmod(a,x-1))%M;

    return bigmod((a*a)%M,x/2);
}

int inv(int x)
{
    return (int)bigmod(x,M-2);
}

int main()
{
    int i,j,k,t;

    for(i=0;i<=100;i++)
    {
        combi[i][0]=combi[i][i]=1;

        for(j=1;j<i;j++) combi[i][j]=(combi[i-1][j]+combi[i-1][j-1])%M;
    }

    //freopen("in.txt","r",stdin);

    scanf("%d",&t);

    while(t--)
    {
        memset(dp,-1,sizeof(dp));

        scanf("%d %d %d %d %d",&a0,&a1,&a2,&a3,&s);

        for(i=1;i<=s;i++) l[i]=a0+a1*i+a2*i*i+a3*i*i*i;



        for(i=1;i<=s;i++)
        {
            c[i][0]=1;
            c[i][1]=l[i]%M;

            for(j=2;j<=(s/i);j++)
            {
                c[i][j]=mul(c[i][j-1] , mul(inv(j) , l[i]-j+1) );

            }
        }

        for(i=1;i<=s;i++)
        {
            way[i][0]=1;

            for(j=1;j<=(s/i);j++)
            {
                way[i][j]=0;

                for(k=1;k<=j;k++)
                {
                    way[i][j]=(way[i][j]+ mul(c[i][k],combi[j-1][k-1] ) )%M;
                }

            }
        }

        printf("%d\n",f(s,s));

    }

    return 0;
}
