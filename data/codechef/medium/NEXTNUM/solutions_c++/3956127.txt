#include <bits/stdc++.h>
#define ll long long

using namespace std;

int digits(int a[18][10],int b[18],ll n)
{
    ll num=n;
    int r,d=0;
    while(num>0)
    {
        r=num%10;num/=10;
        memset(a[d],0,sizeof(a[d]));
        a[d][r]++;
        b[d]=r;
        d++;
    }
    return(d);
}

ll fact(int n)
{
    ll prod=1;
    for(int i=2;i<=n;i++)   prod*=i;
    return(prod);
}

ll find_comb(int a[18][10],int n1,int n2,int pos)
{
    ll sum=0,count=0;
    if(a[pos][n2]==0)   return(0);
    for(int j=0;j<10;j++)   count+=a[pos][j];
    sum=fact(count);
    for(int j=0;j<10;j++)
    {
        if(j==n1)   sum/=fact(a[pos][j]+1);
        else if(j==n2)  sum/=fact(a[pos][j]-1);
        else sum/=fact(a[pos][j]);
    }
    return(sum);
}

int main()
{
    int T,d=0,a[18][10],b[18];
    cin>>T;
    ll n,res=0;
    while(T--)
    {
        cin>>n;
        d=digits(a,b,n);
        for(int j=0;j<10;j++)
        {
            for(int i=1;i<d;i++)
            {
                a[i][j]+=a[i-1][j];
            }
        }
        res=0;
        for(int l=0;l<d-1;l++)
        {
            for(int i=0;i<b[d-1-l];i++) res+=find_comb(a,b[d-1-l],i,d-l-2);
        }

        cout<<res+1<<endl;
    }
    return(0);
}
