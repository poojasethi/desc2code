#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    unsigned long long int fact[25],t,count[15],l,i,j,ans,temp,k;
    char str[10005];
    fact[0]=1;
    for(i=1;i<=18;i++)
    {
        fact[i]=fact[i-1]*i;
    }
    cin>>t;
    while(t--)
    {
        ans=0;
        cin>>str;
        l=strlen(str);
        for(i=0;i<=9;i++)
        {
            count[i]=0;
        }
        for(i=0;i<l;i++)
        {
            count[str[i]-48]++;
        }
        for(i=0;i<l;i++)
        {
            for(j=0;j<str[i]-48;j++)
            {
                if(count[j]==0)
                continue;
                temp=fact[l-i-1];
                count[j]--;
                for(k=0;k<=9;k++)
                    temp=temp/fact[count[k]];
                ans+=temp;
                count[j]++;
            }
            count[str[i]-48]--;
        }
        ans++;
        cout<<ans<<endl;


    }
}
