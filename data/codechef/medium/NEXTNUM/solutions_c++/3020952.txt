#include<iostream>
#include<algorithm>
#include<string>

using namespace std;




long long int fact[19];


long long int count(int* hash,int i ,int j)
	{
	 long long int res=fact[j];
 	 for(int k=0;k<10;k++)
		{
 		 res/=fact[hash[k]];
		 if(k==i)
		 	res*=hash[i];
		}
	  return res;
	}
		 

int main()
{
 int t;
 cin>>t;
 fact[0]=1;
 for(int i=1;i<=18;i++)
	fact[i]=fact[i-1]*i;
 
	
	 
 while(t--)
	{
	 string num;
 	 cin>>num;
	 int size=num.size();
	
	 long long int sum=0;
	 int hash[10];
	 for(int i=0;i<10;i++)
		hash[i]=0;
	 for(int i=0;i<size;i++)
		hash[num[i]-48]++;
	 
	 
	 

	
	 int j=0;
	 
	 
	 while(j<size)
		{
		 for(int i=0;i<num[j]-48;i++)
		 
			 if(hash[i]>0)	
			 	sum+=count(hash,i,size-j-1);

		 if(hash[num[j]-48])
			{
			 hash[num[j]-48]--;
			 j++;
			}
		}
	 cout<<sum+1<<"\n";
	}
 return 0;
}
