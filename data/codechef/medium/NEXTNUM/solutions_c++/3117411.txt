#include <iostream>
using namespace std;

string n;

long long fact(long long num) {
    if(num<=0) return 1;
    long long res = 1;
    for(long long i=1;i<=num;++i) res*=i;
    return res;
}

long long func(char c, int* dig) {
    long long ans = 0;
    for(int i=0;i<c-'0';++i) {
        if(dig[i]==0) continue;
        dig[i]--;
        long long res = 0;
        for(int j=0;j<10;++j) res+=dig[j];
        res = fact(res);
        for(int j=0;j<10;++j) res/=fact(dig[j]);
        dig[i]++;
        ans+=res;
    }
    return ans;
}

int main() {
    int t;
    cin >> t;
    while(t--) {
        cin >> n;// cout<<n<<endl;
        int dig[10] = {0};
        int len = n.length();
        for(int i=0;i<len;++i) dig[n[i]-'0']++;
        long long ans = 1;
        for(int i=0;i<len;++i) {
            ans = ans+func(n[i],dig);// cout<<ans<<endl;
            dig[n[i]-'0']--;
        }
        cout << ans << endl;
    }
    return 0;
}
