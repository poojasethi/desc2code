#include<iostream>

using namespace std;

#define f(i,a,b) for(i=a;i<b;++i)
#define rep(i,a) for(i=0;i<a;++i)

int t,i,j,a[20],b[10],n,k,temp;
long long q,rank=1;
long long fc[20];

int reverse(int x[],int i)
{
	rep(k,i/2)
	{
		temp=x[k];
		x[k]=x[i-k-1];
		x[i-k-1]=temp;
	}
}

long long fac(int h)
{
	if(fc[h]!=0) return fc[h];
	if(h==0){fc[h]=1;return 1;}
	return h*fac(h-1);
}

long long permute(int x[])
{
	int sum=0;
	rep(k,10)
		sum+=x[k];
	long long ret;
	ret=fac(sum);
	rep(k,10)
		ret/=fac(x[k]);
	return ret;
}

int main()
{
	cin>>t;
	while(t--)
	{
		
		rank=1;
		cin>>q;
		i=0;
		while(q>0)
		{
			a[i++]=q%10;
			b[q%10]++;
			q/=10;
		}
		reverse(a,i);
		n=i;
		rep(i,n)
		{
			rep(j,10)
			{
				if(a[i]!=j && b[j]!=0)
				{
					b[j]--;
					rank+=permute(b);
					b[j]++;
				}
				else if(a[i]==j)
				{
					b[j]--;
					j=10;
				}
			}
	
		}
		cout<<rank<<endl;
	}
	
}
