#include <iostream>
#include <string.h>
#include <stdlib.h>
#include <algorithm>
#include <vector>

using namespace std;

long long int fac(int n){
	if(n==0)return 1;
	else return n*fac(n-1);
}

long long int f(char c,int* nar){
	long long int ans=0;
	for(int i=0;i<c-'0';i++){
		if(nar[i]==0)continue;
		else{
			nar[i]--;
			long long int x=0;
			for(int j=0;j<10;j++)x=x+nar[j];
			x=fac(x);
			for(int j=0;j<10;j++){
				x=x/fac(nar[j]);
			}
			nar[i]++;
			ans=ans+x;
		}
	}
	//cout<<  " ans " << ans<<endl;
	return ans;
}

int main(){
	int n=0;cin>>n;
	for(int i=0;i<n;i++){
		string s;cin>>s;
		int len=s.length();
		int ar[10],nar[10];
		memset(ar,0,sizeof(ar));
		memset(nar,0,sizeof(nar));
		
		for(int j=0;j<len;j++){
			ar[s[j]-'0']++;
			nar[s[j]-'0']++;
		}
		long long int  ans=1;
		for(int j=0;j<len;j++){
				ans=ans+f(s[j],nar);
				nar[s[j]-'0']--;
		}
		cout<<ans<<endl;
	}
}
