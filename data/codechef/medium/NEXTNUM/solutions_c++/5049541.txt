#include<bits/stdc++.h>
#define MAX 1000002
#define mod 1000000007
#define ii pair<int,int>
#define iii pair<int,ii>
#define F first
#define S second
#define mp make_pair
#define pb push_back
#define vi vector<int>
#define vii vector < ii >

using namespace std;

int freq[10];
long long int fact [100];

void preprocess()
{
	fact[0]=1;
	fact[1]=1;
	
	for(int i=2;i<=20;i++)
	fact[i]=fact[i-1]*i;
}

int main()
{
int t;
long long int ans=0;

preprocess();

string str;

scanf("%d",&t);

while(t--)
{
	cin>>str;
 	int n=str.size();
	ans=0;
	
	for(int i=0;i<n;i++)
	{
		memset(freq,0,sizeof(freq));
		int x=str[i]-'0';
		
	//	printf("x = %d\n",x);
		
		for(int j=i+1;j<n;j++)
		{
			int y=str[j]-'0';

			freq[y]++;

		}
		
		for(int j=0;j<x;j++)
		{
			long long int prod=fact[n-i-1];
		//	printf("prod = %lld\n",prod);
			if(freq[j])
			{
		//		printf("x = %d j = %d\n",x,j);
				freq[x]++;
				freq[j]--;
				for(int k=0;k<=9;k++)
				prod=prod/fact[freq[k]];
				freq[j]++;
				freq[x]--;

		//		printf("finally prod = %lld\n",prod);
                ans+=prod;
			}
			
		}
	}
	
	cout<<ans+1<<endl;

}

return 0;
}
