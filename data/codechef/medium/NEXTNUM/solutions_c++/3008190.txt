#include<bits/stdc++.h>
using namespace std;

void precompute(string n,int arr[])
{
	for(int i=0;i<n.size();i++)
	arr[n[i]-'0']++;
}

int main()
{
	long long fact[20];
	fact[0]=1;
	fact[1]=1;
	for(int i=2;i<=18;i++)
	fact[i]=fact[i-1]*i;
	
	int t;
	string str;
	cin>>t;
	while(t--)
	{
		cin>>str;
		int arr[10]={0};
		precompute(str,arr);
		int y;
		long long ans=1;
		for(int i=0;i<str.size()-1;i++)
		{
			int x=str[i]-'0';
			y=0;
			for(int j=0;j<x;j++)
			{
				if(arr[j])
				{
					arr[j]--;
					long long temp=1;
					for(int k=0;k<=9;k++)
					{
						temp*=fact[arr[k]];
					}
					ans+=(fact[str.size()-i-1]/temp);
					arr[j]++;
				}
			}
			arr[x]--;
		}
		cout<<ans<<endl;
	}
}
