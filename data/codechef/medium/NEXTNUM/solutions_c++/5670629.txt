#include<iostream>
#include<cstdio>
#define ll long long int
using namespace std;
 int main()
 {
     ll t,n,i,sum=0;
     int P[10],k,j;
     char ch[20];
     unsigned ll A[19],prod;
     A[0]=1;
     for(i=1;i<=18;i++)
        A[i]=A[i-1]*i;
     scanf("%lld",&t);
     while(t>0)
     {
         t--;
         for(i=0;i<10;i++)
            P[i]=0;
         scanf("%s",&ch);
         int tot=0;
         for(i=0;ch[i]!='\0';i++)
         {
             P[ch[i]-48]++;
             tot++;
         }
         sum=0;
         for(i=0;ch[i]!='\0';i++)
         {
                k=0;
             for(;k<=9&&(ch[i]-48)>k;k++)
             {
                 if(P[k]==0)
                    continue;
                 prod=1;
                 j=0;
                 while(j<=9)
                {
                    if(P[j]==0)
                    {
                        j++;
                        continue;
                    }
                    if(j==k)
                    {
                        prod*=A[P[j++]-1];
                    }
                    else
                    prod*=A[P[j++]];
                }
                sum+=A[tot-(i+1)]/prod;
                //cout<<A[tot-(i+1)]/prod<<endl;
             }
             P[ch[i]-48]--;
         }
         printf("%lld\n",sum+1);
     }
 }
