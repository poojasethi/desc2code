#include<iostream>
#include<cstdio>
#include<vector>
#include<map>
#include<string>
#include<algorithm>
using namespace std;
typedef unsigned long long int ull;
typedef long long int ll;
ull fact[19];
void fct(){
	fact[0] = 1;
	fact[1] = 1;
	fact[2] = 2;
	for(int i = 3;i < 19;i++){
		fact[i] = i*fact[i-1];
	}
}
int gm(int arr[],int l){
	for(int i = l;i < 10;i++){
		if(arr[i]){
			return i;
		}
	}
	return -1;
}
ll ga(int arr[],int in){
	int t = 0;
	for(int i = 0;i < 10;i++){
		t += arr[i];
	}
	t -= 1;
	ull div = 1;
	for(int i = 0;i < 10;i++){
		if(in == i){
			div *= fact[arr[i]-1];
		}else{
			div *= fact[arr[i]];
		}
	}
	ull k = 1;
	if(div) k = fact[t]/div;
	//cout<<"k = "<<k <<endl;
	return k;
}
ll get(string tar,int arr[10]){
	int in = gm(arr,0);
	ll ans = 0;
	while( in != -1 && tar[0]-48 != in){
		if(in < tar[0]-48){
			ans += ga(arr,in);
		}
		in = gm(arr,in+1);
	}
	return ans;
}
int main()
{
	int t;
	fct();
	scanf("%d",&t);
	while(t--){
		string s;
		cin >> s;
		ull ans = 1;
		int arr[10] = {0};
		for(int i = 0;i < s.size();i++){
			arr[s[i]-48]++;
		}
		for(int i = 0;i < s.size();i++){
			ans += get(s.substr(i),arr);
			arr[s[i]-48]--;
		}
		cout<<ans<<endl;
	}
	return 0;
}
