#include <iostream>
#include<string.h>
using namespace std;
#define Int long long
Int fact[20];
Int rank(char *str)
{
  int len=strlen(str);
  int freq[10]={0};
  Int ans=0,dup=1;
  for(int i=len-1;i>=0;i--)
  {
    freq[str[i]-'0']++;
    dup*=freq[str[i]-'0'];
    for(int j=0;j<str[i]-'0';j++)
    {
      if(freq[j])
      ans+=fact[len-i-1]/(dup/freq[j]);
    }
    
  }
  return ans;
}

    }

}
