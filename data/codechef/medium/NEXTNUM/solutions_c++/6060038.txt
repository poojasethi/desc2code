#include<bits/stdc++.h>
using namespace std;
long long fact[20];
void factf()
{

	fact[0]=1;
	for(int i=1;i<=20;i++)
	{
		fact[i]=fact[i-1]*i;
	}
}
int main()
{
	factf();
	int t;
	char s[20];
	scanf("%d",&t);
	while(t--)
	{
		int freq[10]={0};
		unsigned long long rank=1,dp=1;
		scanf("%s",s);
		int l=strlen(s);
		for(int i=l-1;i>=0;i--)
		{
			freq[s[i]-48]++;
			dp*=freq[s[i]-48];
			for(int j=0;j<s[i]-48;j++)
			{
				if(freq[j])
				{
					rank+=fact[l-1-i]*freq[j]/dp;
				}
			}
		}
		printf("%lld\n",rank);
	}
	return 0;
}
