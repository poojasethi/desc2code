#include <stdio.h>

typedef long long LL;

char s[20];
LL fac[20];

int main(int argc, char* argv[])
{
    int h, i, j, k, t;
    fac[0]=1;
    for(i=1;i<20;i++)
    {
        fac[i]=i*fac[i-1];
    }
    scanf("%d", &t);
    for(h=0;h<t;h++)
    {
        scanf("%s", &s);
        int a[10]={0};
        int len=0;
        for(i=0;s[i]!='\0';i++)
        {
            a[s[i]-'0']++;
            len++;
        }
        LL sum=0;
        for(i=0;i<len;i++)
        {
            if(i==len-1)
            {
                sum+=1;
            }
            else
            {
                int max=0;
                for(j=0;j<10;j++)
                {
                    if(a[j])
                    {
                        max=j;
                    }
                }
                int d=s[i]-'0';
                int b[10]={0};
                for(j=0;j<=max;j++)
                {
                    b[j]=a[j];
                }
                for(j=0;j<d;j++)
                {
                    if(a[j])
                    {
                        b[j]--;
                        LL p=fac[len-i-1];
                        for(k=0;k<=max;k++)
                        {
                            if(b[k]>1)
                            {
                                p/=fac[b[k]];
                            }
                        }
                        sum+=p;
                        b[j]++;
                    }
                }
                a[d]--;
            }
        }
        printf("%lld\n", sum);
    }
    return 0;
}
