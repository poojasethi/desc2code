#include <stdio.h>
#include <vector>
#include <string.h>
     
typedef long long int LL;
using namespace std;
     
char s[20];
LL factorial[20];
// Find factorials
void preProcess()
{
  	factorial[0] = 1;
    for(int i = 1; i <= 20; i++)
    factorial[i] = i * factorial[i - 1];
}
void findRank()
{
    int n = strlen(s);
    int frequency[25] = {0};
    LL denominator = 1;
    LL ans = 0;
     
    for(int i = n - 1; i >= 0; i--)
    {
    	frequency[s[i] - '0']++;
    	denominator *= frequency[s[i] - '0'];
    	for(int c = 0; c < s[i] - '0'; c++)
    		if(frequency[c])
    			ans += factorial[n - i - 1] * frequency[c] / (denominator);
    }
    printf("%lld\n", ans + 1) ;
}
     
int main()
{
    int t;
    // Preprocess
    preProcess();
    scanf("%d", &t);
    while(t--)
    {
    	scanf("%s", s);
    	findRank();
    }
    return 0;
}
