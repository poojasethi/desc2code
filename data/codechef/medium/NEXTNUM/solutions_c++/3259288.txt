#include <bits/stdc++.h>
using namespace std;
typedef long long ll;

ll fact[20];
int cont[10];
string s;

ll rank;
int  f(char c,int pos)
{
    if(pos==s.size())
    return 0;
    if(s[pos]==c)
    {
        cont[c-'0']--;
        return f('0',pos+1);
    }
    if(cont[c-'0']==0)
        return f(c+1,pos);
    int g[10];
    memset(g,0,sizeof(g));
    for(int i=pos;i<s.size();i++)
    g[s[i]-'0']++;
    g[c-'0']--;
    ll val=fact[s.size()-pos-1];
    for(int i=0;i<10;i++)
    {
        val=val/fact[g[i]];
    }
    rank=rank+val;
    c=c+1;
    return f(c,pos);
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        rank=0LL;
        memset(cont,0,sizeof(cont));
        cin>>s;
        for(int i=0;i<s.size();i++)
        {
            cont[s[i]-'0']++;
        }
        fact[0]=1LL;
        for(int i=1;i<20;i++)
        {
            fact[i]=i*fact[i-1];
        }
        char c='0';
        f(c,0);
        cout<<rank+1<<endl;
    }
}
