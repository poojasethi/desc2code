typedef long long ll;
typedef long l;

using namespace std;
const double pi = acos(-1.0);

ll fact[19];
int cnt[10];

int main()
{
     fact[0]=1ll;
     for(int i=1;i<20;i++) fact[i]=fact[i-1]*i;
     int t;
     in(t);
     while(t--){
     	char c[20];
     	scanf("%s",c);
     	CLEAR(cnt);
     	ll ans=0,d=1;
     	int y=strlen(c);
     	for(int i=strlen(c)-1;i>=0;i--){
     		int a=c[i]-'0';
     		d*=++cnt[a];
     		for(int j=0;j<a;j++){
     			if(cnt[j]){
     				ans+=(fact[y-i-1]*cnt[j])/d;
     			}
     		}
     	}
        printf("%lld\n",ans+1);
     }
     return 0;
 }
