#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

unsigned long long scanlong()
{
	register int c = getchar();
	unsigned long long x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

int scanint(){
	register int c = getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

void bsort(int *ar, int n)
{
	int i,j;
	for(i=0;i<n-1;i++)
	{
		for(j=n-1;j>i;j--)
		{
			if(ar[j]<ar[j-1])
			{
				ar[j]+=ar[j-1];
				ar[j-1]=ar[j]-ar[j-1];
				ar[j]-=ar[j-1];
			}
		}
	}
}

int main(){
	unsigned long long fact[20]={1,1,2,6,24,120,720,5040,40320};
	int t=0,i,j,c[10]={0},d[10]={0},a[20]={0},b[20]={0},nd=0,ed=0;
	unsigned long long number=0,rank=0,div=1;
	for(i=9;i<20; i++){
		fact[i]=i*fact[i-1];
	}
	t=scanint();
	while(t--){
		number=scanlong();
		for(i=0;i<10;i++){
			c[i]=0;
			d[i]=0;
		}
		i=nd=0;
		while(number!=0){
			ed=number%10;
			c[ed]++;
			d[ed]++;
			a[i]=b[i]=ed;
			i++;
			nd++;
			number=number/10;
		}
		i=nd-1;
		j=rank=0;
		bsort(b,nd);
		while(i>=0){
			if(j<nd && a[i]==b[j]){
				j=0;
				c[a[i]]--;
				i--;
			}
			else if(j<nd  && c[b[j]]!=0){
				c[b[j]]--;
				for(int l=0;l<10;l++){
					div=div*fact[c[l]];
				}
				rank=rank+(fact[i]/div);
				c[b[j]]++;
				div=1;
				j+=d[b[j]];
			}else{
				j++;
			}
		}
		rank++;
		cout<<rank<<"\n";
	}
}
