#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
#define inp(t) scanf("%lld",&n);
#define out(t) printf("%lld\n",t);
using namespace std;
typedef long long int ll;
ll i,j,k,n,t,fact[100],l;
char s[100];
int main()
{
	fact[0]=fact[1]=1;
	for(i=2;i<=18;i++)
	fact[i]=fact[i-1]*i;
    //inp(t);
    scanf("%lld",&t);
    while(t--)
    {
    	ll ans=0;
    	int a[10]={0};
    	scanf("%s",s);
    	l=strlen(s);
    	for(i=0;i<l;i++) a[s[i]-'0']++;
    	//for(i=0;i<10;i++) printf("%d ",a[i]);
    	n=l;
    	for(i=0;i<n;i++)
    	{
    		int p=s[i]-'0';
    		//printf("%d\n",p);
    		for(j=0;j<p;j++)
    		{
    			if(a[j])
    			{
    				ll x=1,y=fact[n-1-i];
    				a[j]--;
    				for(k=0;k<10;k++)
    				x*=fact[a[k]];
    				a[j]++;
    				ans+=y/x;
    			}
    		}
    		a[p]--;
    	}
    	printf("%lld\n",ans+1);
    }
    return 0;
}
