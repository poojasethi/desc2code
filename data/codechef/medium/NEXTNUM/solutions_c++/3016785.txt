#include <iostream>
#include <string.h>
#include <stdlib.h>
#include <algorithm>
#include <vector>

using namespace std;
long long int fa[18];

long long int f(char c,int* nar,int y){
	long long int ans=0;
	for(int i=0;i<c-'0';i++){
		if(nar[i]==0)continue;
		else{
			nar[i]--;
			long long x=fa[y];
			for(int j=0;j<10;j++){
				x=x/fa[nar[j]];
			}
			nar[i]++;
			ans=ans+x;
		}
	}
	//cout<<  " ans " << ans<<endl;
	return ans;
}

int main(){
	fa[0]=1;
	for(int i=1;i<=18;i++)fa[i]=i*fa[i-1];
	
	int n=0;cin>>n;
	for(int i=0;i<n;i++){
		string s;cin>>s;
		int len=s.length(),z=len;
		int nar[10];
		memset(nar,0,sizeof(nar));
		
		for(int j=0;j<len;j++){
			nar[s[j]-'0']++;
		}
		
		long long int  ans=1;
		for(int j=0;j<len;j++){
				ans=ans+f(s[j],nar,z-1);
				nar[s[j]-'0']--;
				z--;
		}
		cout<<ans<<endl;
	}
}
