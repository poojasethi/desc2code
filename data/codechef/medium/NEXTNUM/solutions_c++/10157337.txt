#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL fac[20];
int main(){
	fac[0] = 1;
	for(int i = 1;i < 20;++i)
		fac[i] = i * fac[i - 1];
	int t;
	cin >> t;
	while(t--){
		string s;
		cin >> s;
		int N = s.size();
		int cnt[10] = {0};
		for(int i = 0;i < N;++i){
			++cnt[s[i] - '0'];
		}
		LL ans = 0;
		for(int i = 0;i < N;++i){
			for(int j = 0;j < (s[i] - '0');++j){
				if(cnt[j] == 0) continue;
				LL curAns = fac[N - i - 1];
				--cnt[j];
				for(int k = 0;k < 10;++k){
					curAns /= fac[cnt[k]];
				}
				++cnt[j];
				ans += curAns;
			}
			--cnt[s[i] - '0'];
		}
		printf("%lld\n", ans + 1);
	}
	return 0;
}
