using namespace std;

ll fact[20];

inline void sfast(char *str)
{
    register char c = 0;
    register int i = 0;
    while (c < 33)
        c = gc();

    while(c >= '0' && c <= '9')
    {
        str[i] = c;
        c = gc();
        i = i + 1;
    }
    str[i] = '\0';
}

ll rank(char *str)
{
    ll ans = 0, d = 1;
    int freq[10] = {0}, len = strlen(str);
    reverse(str, str+len);
    for(int i = 0; i < len; i++)
    {
        freq[str[i] - '0']++;
        d *= freq[str[i] - '0'];
        for(int j = 0; j < str[i] - '0'; j++)
            if(freq[j] > 0)
                ans += fact[i]/(d/freq[j]);
    }
    return ans;
}

int main()
{
    fact[0] = 1;
    FOR(i, 1, 20)
        fact[i] = fact[i-1] * i;
    int t;
    scanf("%d", &t);
    while(t--)
    {
        char str[20];
        sfast(str);
//        scanf("%s", str);
        printf("%lld\n", rank(str)+1);
    }
    return 0;
}
