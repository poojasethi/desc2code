#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

long long int n[19];

long long int fact(int m)
{
    if (m==1)
        return 1;
    else
        return m*fact(m-1);
}

void factorial()
{
    int i=1;
    n[0]=1;
    while (i<19)
    {
        n[i]=fact(i);
        i++;
    }
}

int main()
{
    int t,i,j,k,r,l;
    long long int d,b;
    char s[19];
    scanf ("%d",&t);
    factorial();
    while (t--)
    {
        int c[10];
        b=0;
        for (i=0;i<10;++i)
            c[i]=0;
        scanf ("%s",s);
        r=strlen(s);
        for (j=0;j<r;++j)
            c[s[j]-48]++;
        for (i=0;s[i]!='\0';++i)
        {
            for (j=0;j<10;++j)
            {
                if ((s[i]-48)==j)
                {
                    c[j]--;
                    break;
                }
                else if (c[j]!=0)
                {
                    d=n[r-1];
                    {
                        for (k=0;k<10;++k)
                            if (k!=j)
                                d/=(n[c[k]]);
                            else
                                d/=(n[c[k]-1]);
                    }
                    b+=d;
                }
            }
            r--;
        }
        printf ("%lld\n",b+1);
    }
    return 0;
}
