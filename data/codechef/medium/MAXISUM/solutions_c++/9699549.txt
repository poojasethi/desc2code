#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    for(scanf("%d", &t); t--; )
    {
        int n;
        long long a[100001], b[100001];
        long long k;
        scanf("%d %lld", &n, &k);
        for(int i=0; i<n; i++)
            scanf("%lld", &a[i]);
        for(int i=0; i<n; i++)
            scanf("%lld", &b[i]);
        long long sum=0;
        int maxim=0;
        for(int i=0; i<n; i++){
            sum+=a[i]*b[i];
            if(abs(b[i])>abs(b[maxim]))
                maxim=i;
        }
        sum+=(k*abs(b[maxim]));
        printf("%lld\n", sum);
    }
    return 0;
}
