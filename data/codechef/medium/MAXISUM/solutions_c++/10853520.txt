#include<bits/stdc++.h>
#include <cassert>

using namespace std;
typedef long long int ll;
int main() {
    ios::sync_with_stdio(0);
    int t;
    cin >> t;
    while (t --) 
	{
    	ll n,k,b;
		cin>>n>>k;
		ll a[100011] = {0},sum1=0,sum2=0;
		for(ll i=0;i<n;i++)
		{
			cin>>a[i];
		}
		for(ll i=0;i<n;i++)
		{
			cin>>b;
			sum2 = max(sum2,abs(b));
			sum1 += b*a[i];
		}
		sum1 += k * sum2;
		cout<<sum1<<endl;	
    }
    return 0;
}
