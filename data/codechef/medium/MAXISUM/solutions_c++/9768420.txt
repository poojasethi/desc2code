#include<iostream>
#define max 100005
using namespace std;
int main()
{
	int a[max],b[max];
	int n,k;
	
	int t;
	cin>>t;
	while(t--)
	{
	cin>>n>>k;
	int i,j;
	for(i=0;i<n;i++)
	cin>>a[i];
	for(i=0;i<n;i++)
	cin>>b[i];
	
	long long int temp=0;
	int loc=-1;
	
	for(i=0;i<n;i++)
	{
		if(b[i]<0) j=-b[i];
		else j=b[i];
		
		if(temp<j)
		{
			temp=j;
			loc=i;
		}
	}
	
	temp=0;
	if(b[loc]<0) a[loc]-=k;		
	else a[loc]+=k;
	
	temp=0;
	for(i=0;i<n;i++)
	{
		temp=temp+((long long int)a[i])*b[i];
	}
	cout<<temp<<'\n';
	}
	return 0;
}
