#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstdlib>
#define ll long long int
 
using namespace std;
 
int main()
{
    int t;
 
    cin>>t;
 
    while(t--)
    {
        ll n,k;
 
        scanf("%lld %lld",&n,&k);
 
        ll *a=(ll *)calloc(n,sizeof(ll));
 
        ll *b=(ll *)calloc(n,sizeof(ll));
 
        for(ll i=0;i<n;i++)
        {
            scanf("%lld",&a[i]);
        }
 
        for(ll i=0;i<n;i++)
        {
            scanf("%lld",&b[i]);
 
            if(b[i]<0)
            {
                b[i]=b[i]*(-1);
 
                a[i]=a[i]*(-1);
            }
        }
 
        ll total=0;
 
        for(ll i=0;i<n;i++)
        {
            total+=(a[i]*b[i]);
 
        }
 
        ll max=b[0];
 
        for(ll i=1;i<n;i++)
        {
            if(max<b[i])
                max=b[i];
        }
 
        total+=max*k;
 
        printf("%lld\n",total);
    }
 
    return(0);
 }
