#include<bits/stdc++.h>
using namespace std;
int main()
{
	long t;
	cin>>t;
	while(t!=0)
	{
		long long int n,k,x,max,i,dmax,sum=0,j=0;
		cin>>n>>k;
		long long int a[n],b[n],d[n];
		
		for(x=0;x<n;x++)
		{
			cin>>a[x];
		}
		for(x=0;x<n;x++)
		{
			cin>>b[x];
		}
		max=abs(b[0]);
		for(i=0;i<n;i++)
		{
			d[i]=a[i]*b[i];
			if(abs(b[i])>max)
			{
				max=abs(b[i]);
				j=i;
				
			}
		}
		if(b[j]>0)
		{
			d[j]=(a[j]+k)*b[j];
		}
		if(b[j]<0)
		{
			d[j]=(a[j]-k)*b[j];
		}
		for(x=0;x<n;x++)
		{
		sum=sum+d[x];	
		}
		cout<<sum<<"\n";
		t--;
	}
	return 0;
}
