#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#define N 100111
#define LL long long int
using namespace std;

int main()
{
	int test;
	LL A[N], B[N], k,n;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%lld%lld", &n,&k);
		for(int i=0;i<n;i++)
			scanf("%lld", &A[i]);
		
		for(int i=0;i<n;i++)
			scanf("%lld", &B[i]);
		
		LL result = 0;
		LL max = B[0];
		
		for(int i=1;i<n;i++)
			if(abs(B[i]) > max)
				max = abs(B[i]);
		
		for(int i=0;i<n;i++)
			result +=(A[i]*B[i]);
		
		result += max * k;
		cout<<result<<"\n";
	}
	return 0;
}
