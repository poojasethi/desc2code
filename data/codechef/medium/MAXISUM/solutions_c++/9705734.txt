#include <bits/stdc++.h>
#define ll long long int
using namespace std;

int main() {
	ll t;
	cin >> t;

	ll n, k;

	while(t --) {
		cin >> n >> k;
		vector<ll> a(n);
		ll b;

		for(ll i = 0; i < n; i ++) {
			cin >> a[i];
		}
		ll maxm = -1;
		ll sum = 0;
		for(ll i = 0; i < n; i ++) {
			cin >> b;
			maxm = max(maxm, abs(b));
			sum += a[i] * b;
		}

		ll res = sum + k * maxm;
		cout << res << endl;
	}

	return 0;
}
