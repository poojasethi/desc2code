#include<bits/stdc++.h>
#define ll long long int
#define PI 3.14159265359
#define MOD 1000000007
#define SET(x,y) memset(x,y,sizeof(x));


using namespace std;

ll a[1000000],b[1000000];

int main(){
	int t;
	ll n,k,i,max_b,inter;

	scanf("%d",&t);

	while(t--){
		scanf("%lld%lld",&n,&k);

		inter=0;

		for(i=0;i<n;i++)
			scanf("%lld",&a[i]);

		for(i=0;i<n;i++)
			scanf("%lld",&b[i]);
		
        
        max_b=0;

		for(i=0;i<n;i++){
			inter+=(a[i]*b[i]);
			max_b=max(abs(b[i]),max_b);
		}

        inter+=(k*max_b);

        printf("%lld\n",inter);

	}

	return 0;
}
