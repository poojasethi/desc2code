#include<bits/stdc++.h>
using namespace std;
#define LL int long long

LL A[100001];
LL B[100001];
LL mul(int n)
{
    LL res=0;
    int i;
    for(i=0;i<n;i++)
    {
      res+=(A[i]*B[i]);
    }
    return res;
}
LL solve(int n,LL k){
   int i,j,l;
   LL res=INT_MIN;
   LL m=mul(n);
   res=m;
   for(i=0;i<n;i++)
   {
   LL x=m-(B[i]*A[i]);
   A[i]+=k;
   res=max(res,x+(A[i]*B[i]));
   A[i]-=k;
   A[i]-=k;
   res=max(res,x+(A[i]*B[i]));
   A[i]+=k;
   }
   return res;
}
int main()
{
  ios::sync_with_stdio(false);
    int t;
    scanf("%d",&t);
    while(t--)
    {
      int n;
      scanf("%d",&n);
      LL k;
      scanf("%lld",&k);
      int i;
      for(i=0;i<n;i++)
      {
      scanf("%lld",&A[i]);
      }
      for(i=0;i<n;i++)
      {
      scanf("%lld",&B[i]);
      }
      printf("%lld\n",solve(n,k));
    }
    return 0;
}
