#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin >> t;

	int n, k;

	while(t --) {
		cin >> n >> k;
		vector<int> a(n);
		vector<int> b(n);

		for(int i = 0; i < n; i ++) {
			cin >> a[i];
		}
		int maxm = -1;
		for(int i = 0; i < n; i ++) {
			cin >> b[i];
			maxm = max(maxm, abs(b[i]));
		}

		long long int sum = 0;
		for(int i = 0; i < n; i ++) {
			sum += (long long int)a[i] * (long long int)b[i];
		}

		long long int res = sum + (long long int)k * (long long int)maxm;
		cout << res << endl;
	}

	return 0;
}
