#include<stdio.h>
#define N 100111
#define LL long long int


int main()
{
	int test,i;
	LL A[N], B[N], k,n, result, max;
	scanf("%d",&test);
	while(test--)
	{
		result = 0;
		scanf("%lld%lld", &n,&k);
		for( i=0;i<n;i++)
			scanf("%lld", &A[i]);
		
		for( i=0;i<n;i++)
		{
			scanf("%lld", &B[i]);
			result +=(A[i]*B[i]); 
		}
		
		max = B[0];
		
		for( i=1;i<n;i++)
		{
			if(B[i] > max)
				max = B[i];
			if(-B[i] > max)
				max = - B[i];	
		}
			
		result += max * k;
		printf("%lld\n",result);
	}
	return 0;
}
