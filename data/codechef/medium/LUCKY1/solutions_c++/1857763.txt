#include <stdio.h>
#include <map>
using namespace std;

int Q;

// Immer ein bisschen zu viel reservieren, um keine "dummen" Probleme zu bekommen!
long long dp[100009];

map<int, int> DIFF;

int main()
{
	scanf("%d", &Q);
	
	dp[0] = 0;
	DIFF[0] = 1;
	int s4 = 0, s7 = 0;
	for(int i = 1; i <= 100000; i++)
	{
		// ZÃ¤hle Vierer und Siebener zu Zahl hinzu
		int x = i;
		while(x)
		{
			s4 += x % 10 == 4;
			s7 += x % 10 == 7;
			x /= 10;
		}
		
		dp[i] = dp[i - 1] + DIFF[s4 - s7];
		DIFF[s4 - s7]++;
	}
	
	for(int q = 0; q < Q; q++)
	{
		int n;
		scanf("%d", &n);
		printf("%lld\n", dp[n]);
	}
	
	return 0;
}
