#include <iostream>
#include <fstream>
#include <string.h>
#include <stdlib.h>
#include <algorithm> 
#include <cmath>
#include <set>
#include <vector>
using namespace std;

#define C 10 
int t , n , dp[100001] , a[10000001] , b[100001] ;

int f(int x)
{
   int res=0 ; 
   while(x)
   {
      if(x%10==4) res++ ; 
      if(x%10==7) res-- ;
      x/=10 ;
   }
 return res ;
}

int main()
{  
   for(int i=1 ; i<=100000 ; i++)
   {   
       b[i]=b[i-1]+f(i) ;  
       dp[i]=dp[i-1]+a[b[i]+C]+(b[i]==0) ;  
       a[b[i]+C]++ ;
   }

   scanf("%d",&t) ;
   while(t--)
   { 
       scanf("%d",&n) ;
       printf("%d\n",dp[n]) ;
   }   
   
  return 0;
}
