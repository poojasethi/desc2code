using namespace std;

int debug=1;
const int LIM=1e5+5;


/* MAIN PROGRAM STARTS HERE */

LL cnt[LIM];
map<LL,LL> M;


LL calc(LL n,int d)
	{
	LL n1=n,res=0;
	while(n1)
		{
		int p=n1%10;
		if (p==d) res++;
		n1/=10;
		}
	return res;
	}

void solve()
	{
	LL i;
	cnt[0]=0;
	LL  n7=0,n4=0;
	F(i,1,LIM)
		{
		n7+=calc(i,7);
		n4+=calc(i,4);
		cnt[i]=cnt[i-1]+M[n7-n4];
		cnt[i]+=(n7==n4)?1:0;
		M[n7-n4]++;
		}
	}

int main()
	{
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	
	LL T,n,i,j,k,l,m,x,y,z;
	solve();
	cin>>T;
	while(T--)
		{
		cin>>n;
		cout<<cnt[n]<<endl;
		}
	
	return 0;
	}
