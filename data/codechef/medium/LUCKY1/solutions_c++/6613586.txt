#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<map>
using namespace std;
map<int,int> m;
int four[100010],seven[100010],sfour[100010],sseven[100010],res[100010];
int main()
{
    m[0]+=1;
 
    for(int i=1;i<=100000;i++)
    {
            int c=i;
            while(c!=0)
            {
                    if(c%10==4)four[i]+=1;
                    if(c%10==7)seven[i]+=1;
                    c/=10;
            }
 
            sfour[i]+=four[i]+sfour[i-1];
            sseven[i]+=seven[i]+sseven[i-1];
 
            int ans=m[sfour[i]-sseven[i]];
            
            res[i]+=res[i-1]+ans;
            m[sfour[i]-sseven[i]]+=1;
    }
 
    int t;
    scanf("%d",&t);
    while(t--)
    {
            int n;
            scanf("%d",&n);
            printf("%d\n",res[n]);
    }
    return 0;
}
