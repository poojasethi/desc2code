namespace LUCKY1 {
int ans[111000], cnt[111000];

void solve(int test_num) {
  int n;
  scanf("%d", &n);
  printf("%d\n", ans[n]);
 }

int get4(int x) {
  int ans = 0;
  for (; x > 0; x /= 10)
    if (x % 10 == 4)
      ans++;
  return ans;
}

int get7(int x) {
  int ans = 0;
  for (; x > 0; x /= 10)
    if (x % 10 == 7)
      ans++;
  return ans;
}

void init() {
  memset(cnt, 0, sizeof (cnt));
  int fours = 0, sevens = 0;
  cnt[0] = 1;
  for (int i = 1; i <= 100000; i++) {
    fours += get4(i);
    sevens += get7(i);
    ans[i] = ans[i - 1] + cnt[fours - sevens];
    cnt[fours - sevens]++;
  }
}

void solve() {
  #ifdef AZN
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  freopen("azn.txt", "w", stderr);
  #endif
  init();
  int tests;
  scanf("%d", &tests);
  for (int i = 1; i <= tests; i++)
    solve(i);
}
}

int main() {
  LUCKY1::solve();
  return 0;
