using namespace std;
typedef long long ll;
ll ans[1000005];
int f4[1000005];
int f7[1000005];

int main () {
	int i,j,k;
	rep(i,0,100000) {
		f4[i] = f4[i/10]+(i%10 == 4);
		f7[i] = f7[i/10]+(i%10 == 7);
	}	
	map<int,int> m;
	saaf(m);
	m[0]=1;
	rep(i,1,100000) {
		f4[i] += f4[i-1];
		f7[i] += f7[i-1];
		ans[i] = ans[i-1]+m[f4[i]-f7[i]]++;
	}
	int t,n;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		printf("%lld\n",ans[n]);
	}
	return 0;
}
