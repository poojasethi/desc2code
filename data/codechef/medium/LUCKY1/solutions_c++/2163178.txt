#include<stdio.h>

int main()
{
    int t,n,count4[100001],count7[100001];
    long long answer[100001],c[100001];
    for(int i=1;i<=100000;i++)
    {
        count4[i]=count4[i/10];
        if((i%10)==4) count4[i]++;
        count7[i]=count7[i/10];
        if((i%10)==7) count7[i]++;
    }
    int cnt4=0,cnt7=0;
    c[0]=1;
    for(int i=1;i<=100000;i++)
    {
        cnt4+=count4[i];
        cnt7+=count7[i];
        answer[i]=answer[i-1];
        answer[i]+=c[cnt4-cnt7];
        c[cnt4-cnt7]++;
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%d\n",answer[n]);
    }
}
