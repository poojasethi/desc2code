inline int read_int() {
  register char c = getchar_unlocked();
  while(c<'0' || c>'9') c = getchar_unlocked();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = (ret<<3)+(ret<<1) + c - 48;
    c = getchar_unlocked();
  }
  return ret;
}

inline void fastwrite(int a){
char snum[20];
int i=0;
do{
snum[i++]=a%10+48;
a=a/10;
}while(a!=0);
i=i-1;
while(i>=0)
putchar_unlocked(snum[i--]);
putchar_unlocked('\n');
}

using namespace std;

int f4[100001];
int f7[100001];
int dp[100001];    //dp[i]   answer to query i
int norn[100001];  //norn[i] no. of pairs with r=i

int main()
{
    int t,n;
    dp[1]=1;
    norn[1]=1;
    for(int i=1;i<=100000;i++)
    {
        int x=i;
        while(x>0)
        {
            f4[i]+=(x%10==4);
            f7[i]+=(x%10==7);
            x/=10;
        }
    }
    dp[2]=3;dp[3]=6;dp[4]=6;dp[5]=7;dp[6]=9;dp[7]=13;
    norn[2]=2;
    norn[3]=3;
    norn[4]=0;
    norn[5]=1;
    norn[6]=2;
    norn[7]=4;

    for(int i=8;i<=100000;i++)
    {
        if(f4[i]==f7[i])
        {
            dp[i]=1+norn[i-1]+dp[i-1];
            norn[i]=1+norn[i-1];
        }
        else
        {
            if(i==4){dp[i]=6;norn[i]=0;continue;}
            if(i==7){dp[i]=13;norn[i]=4;continue;}
            norn[i]=0;
            int sum=0;
            for(int j=i;j>=2;j--)
            {
                sum+=f4[j]-f7[j];
                if(sum==0)
                {
                    norn[i]=1+norn[j-1];
                    break;
                }
                if(sum>1150)
                    break;
            }
            dp[i]=norn[i]+dp[i-1];
        }
    }
    t=read_int();
    while(t--)
    {
        n=read_int();
        fastwrite(dp[n]);
    }
    return 0;
}
