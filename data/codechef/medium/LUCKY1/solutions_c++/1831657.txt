#include <stdio.h>
#include <map>
using namespace std;

map<int, int> d;

long long dp[100001];
int s_four[100001], s_seven[100001];

int main()
{
	int T;
	scanf("%d", &T);
	
	d[0]++;
	dp[1] = 1;
	for(int i = 2; i <= 100000; i++)
	{
		int four = 0, seven = 0;
		int x = i;
		while(x)
		{
			four  += (x % 10 == 4);
			seven += (x % 10 == 7);
			x /= 10;
		}
		
		s_four[i]  =  s_four [i - 1] + four;
		s_seven[i] =  s_seven[i - 1] + seven;
		
		dp[i] = dp[i - 1] + d[s_four[i] - s_seven[i]] + (four == seven);
		d[s_four[i - 1] - s_seven[i - 1]]++;
	}
	
	for(int t = 0; t < T; t++)
	{
		int N;
		scanf("%d", &N);
		printf("%lld\n", dp[N]);
	}
	
	
	return 0;
}
