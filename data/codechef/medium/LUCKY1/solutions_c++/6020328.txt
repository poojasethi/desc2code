#define Max 100010

typedef long long LL;
typedef unsigned long long ULL;

using namespace std;

LL ara[Max],pos[Max],neg[Max],cumfour[Max],cumseven[Max],fr,svn;

LL fun4(LL x)
{
    LL cnt = 0;
    while(x)
    {
        if(x%10 == 4)cnt++;
        x/=10;
    }
    return cnt;
}

LL fun7(LL x)
{
    LL cnt = 0;
    while(x)
    {
        if(x%10 == 7)cnt++;
        x/=10;
    }
    return cnt;
}

void prework()
{
    for(LL i=1;i<=100000;i++)
    {
        LL f = fun4(i);
        LL s = fun7(i);
        //LL diff = f - s;
        fr+=f;
        svn+=s;
        LL diff = fr - svn;
        if(diff == 0)
        {
            pos[diff]++;
            LL my = pos[diff];
            ara[i] = ara[i-1] + my;
        }
        if(diff < 0)
        {
            LL my = neg[diff];
            ara[i] = ara[i-1] + my;
            neg[diff]++;
        }
        if(diff > 0)
        {
            LL my = pos[diff];
            ara[i] = ara[i-1] + my;
            pos[diff]++;
        }
       // printf("%lld  -->   %lld\n",i,ara[i]);
    }
}

int main()
{
    prework();
    LL t,T,n;
    sl(T);
    rep(t,T)
    {
        sl(n);
        LL ans = ara[n];
        pl(ans);NL;
    }
    return 0;
}
