#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int read()
{
int n=0;
register char ch=getchar_unlocked();
while(ch<'0'||ch>'9')
ch=getchar_unlocked();
while(ch>='0'&&ch<='9')
{
n=(n<<3)+(n<<1)+(ch-'0');
ch=getchar_unlocked();
}
return n;
}
int main()
{
    int n;
    n=read();
    int ar[100001],cc[100001];
    for(int i=0;i<n;i++)
        ar[i]=read();
    sort(ar,ar+n);
    cc[0]=ar[0];
    for(int i=1;i<n;i++)
        cc[i]=cc[i-1]+ar[i];
    int q;
    cin>>q;
    for(int i=0;i<q;i++)
    {
        int s;
        scanf("%d",&s);
        int f=(n)/(s+1);
        if(n%(s+1))
            cout<<cc[f]<<endl;
        else
            cout<<cc[f-1]<<endl;
    }
}
