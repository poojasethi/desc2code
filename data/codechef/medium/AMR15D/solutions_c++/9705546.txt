#include <bits/stdc++.h>
#define ll long long int
#define MAX 100000
#define sc1(x) scanf("%d",&x)
using namespace std;

int arr[MAX+1],sum[MAX+1];
int main()
{
	std::ios::sync_with_stdio(false);
	int n,i;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>arr[i];
	}
	sort(arr+1,arr+1+n);
	for(i=1;i<=n;i++)
	{
		sum[i]=sum[i-1]+arr[i];
	}
	int q;
	cin>>q;
	while(q--)
	{
		int k,num;
		cin>>k;
		if(n%(k+1)==0)
			num=n/(k+1);
		else
			num=n/(k+1)+1;
		cout<<sum[num]<<endl;
	}

	return 0;
}
