#include<stdio.h>
int a[21][20004]={0};
bool visited[1<<21]={false};
int main()
{
    int n,m,i,j,k;
    scanf("%d%d",&n,&m);
    for(i=0;i<m;i++)
    {
        scanf("%d",&k);
        for(j=0;j<k;j++)
        {
            int x;
            scanf("%d",&x);
            a[i][x]=1;
        }
    }
    bool invalid[21][21]={false};
    for(i=0;i<m;i++)
    {
        for(j=i+1;j<m;j++)
        {
            for(k=0;k<=n;k++)
            {
                if(a[i][k] && a[j][k])
                {
                    invalid[i][j]=true;
                    invalid[j][i]=true;
                    visited[(1<<i)|(1<<j)]=true;
                    break;
                }
            }
        }
    }
    int ans=0;
    for(i=0;i<(1<<m);i++)
    {
        //printf("i=%d vis=%d\n",i,visited[i]);
        if(visited[i])
        {
            for(j=0;j<m;j++)
            {
                visited[i|(1<<j)]=true;
            }
        }
        else
        {
            int c=0;
            for(j=0;j<m;j++)
            {
                if(i&(1<<j))
                    c++;
            }
            if(c>ans)
                ans=c;
        }
    }
    printf("%d\n",ans);
    return 0;
}
