inline int in()
{
   int n=0;
   int ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   return n*sign;
}
int a[21][20001],adj[21][21],m,s[21];
int clique(int s[],int l)
{
    int i,j,deg[l],mini;
    while(1)
    {
        int minn=l+5,maxx=0;
        memset(deg,0,sizeof(deg));
        for(i=0;i<l;i++)
        {
            for(j=0;j<l;j++)
            {
                if(adj[s[i]][s[j]] && i!=j)
                {
                    deg[i]++;
       //             cout<<"i="<<s[i]<<" j="<<s[j]<<endl;
                }
            }
        
        }
    /*    cout<<"deg==>"<<endl;
        for(i=0;i<l;i++)
            cout<<deg[i]<<" ";
        cout<<endl;
    */    for(i=0;i<l;i++)
        {
            if(deg[i]<minn)
                minn=deg[i],mini=i;
            if(deg[i]>maxx)
                maxx=deg[i];
        }
    //    cout<<"min="<<minn<<" max="<<maxx<<endl;
     //   system("pause");
        if(minn==maxx)
            return l;
        else
        {
            l--;
            for(i=1;i<mini;i++)
                s[i]=s[i];
            for(;i<=l;i++)
                s[i]=s[i+1];
        }
    }
}
