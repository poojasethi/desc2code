#define mod 360
int mark[22][20020];
int same[22][22];
int m;
vector<int>s;
int ans;
void rec(int t){
  if(t==m){
    ans=max(ans,(int)s.size());
    return;
  }
  rec(t+1);
  bool flag=true;
  for(int i=0;i<s.size();i++){
    if(same[t][s[i]]){
      flag=false;
      break;
    }
  }
  if(flag){
    s.pb(t);
    rec(t+1);
    s.pop_back();
  }
}
int main(){
  int n,k,temp;
  memset(mark,0,sizeof(mark));
  memset(same,0,sizeof(same));
  cin>>n>>m;
  for(int i=0;i<m;i++){
    cin>>k;
    for(int j=0;j<k;j++){
          cin>>temp;
          mark[i][temp]=1;
      }
    }
    for(int i=1;i<=n;i++){
      for(int j=0;j<m;j++){
        for(int k=0;k<m;k++){
          if((mark[j][i]==1)&&(mark[k][i]==1)){
            same[k][j]=1;
            same[j][k]=1;
          }
        }
      }
    }
    rec(0);
    cout<<ans;
    return 0;
  }
