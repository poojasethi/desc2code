#include<iostream>
using namespace std;
int a[22][20010];
int b[22][22];
int c[22];
int n,m;
int dfs(int in){
	if(in>=m)
		return 0;
	int mx=0;
	int tmp;
	for(int i=in;i<m;++i){
		if(c[i]==0){
			for(int j=0;j<m;++j){
				if(b[i][j]==1)
					++c[j];
			}
			tmp=1+dfs(i+1);
			for(int j=0;j<m;++j){
				if(b[i][j]==1)
					--c[j];
			}
			if(tmp>mx)
				mx=tmp;
		}
	}
	return mx;
}
int main(){
	cin>>n>>m;
	for(int i=0;i<m;++i){
		for(int j=0;j<=n;++j){
			a[i][j]=0;
		}
	}
	for(int i=0;i<m;++i){
		for(int j=0;j<m;++j){
			b[i][j]=0;
		}
		c[i]=0;
	}
	for(int i=0;i<m;++i){
		int k;
		cin>>k;
		for(int j=0;j<k;++j){
			int tmp;
			cin>>tmp;
			a[i][tmp]=1;
		}
	}
	for(int i=1;i<=n;++i){
		for(int j=0;j<m;++j){
			for(int k=0;k<m;++k){
				if(a[j][i]==1&&a[k][i]==1){
					b[j][k]=1;
					b[k][j]=1;
				}
			}
		}
	}
	cout<<dfs(0)<<"\n";
	
	return 0;
}
