#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,m;
		cin>>n>>m;
		
		int a[n];
		int sum=0;
		for(int i=0;i<n;i++){
			cin>>a[i];
			sum=sum+a[i];
		}
		sum=sum+m;
		if(sum%n==0)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
		}
		
	return 0;
	}
