#include<bits/stdc++.h>
using namespace std;

int main()
{
    int T,N,M;
    cin>>T;
    int a,b,c;
    while(T--){
        cin>>N>>M;
        std::vector<int> A(N);
        int temp;
        cin>>temp;
        A[0] = temp;
        int max = A[0];
        for(int i = 1; i < N; i++){
            cin>>temp;
            A[i] = temp;
            if(max<temp)
            max = temp;
        }
        int sum = 0;
         for(int i = 0; i < N; i++){
            sum += (max-A[i]);
         }
        if((M-sum)%N == 0 && M>=sum)
        std::cout << "Yes" << std::endl;
        else
        std::cout << "No" << std::endl;
        
    }
   
	return 0;
}
