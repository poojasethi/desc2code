#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m,sum=0,max=-1;
		cin>>n>>m;
		int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
			if(a[i]>max)
			  max=a[i];
		}
		for(int i=0;i<n;i++)
		   sum += max - a[i];
		if(sum == m)
		   cout<<"Yes\n";
		else if(sum<m)
		{
			if((m-sum) % n == 0)
			    cout<<"Yes\n";
			else
			   cout<<"No\n";
		}
		else
		   cout<<"No\n";
	}
}
