#include<stdio.h>

int main()
{
    int i,t,n,m,b[101],d,max,p;
    scanf("%d",&t);
    for(;t>0;t--)
    {
        scanf("%d%d",&n,&m);
        scanf("%d",&b[0]);
        max=b[0];
        for(i=1;i<n;i++)
        {
            scanf("%d",&b[i]);
            if(b[i]>max)
                max=b[i];
        }
        d=0;
        for(i=0;i<n;i++)
            d=d+max-b[i];
        if(m<d)
            printf("No\n");
        else
        {
            p=m-d;
            p=p%n;
            if(p==0)
                printf("Yes\n");
            else
                printf("No\n");
        }
    }
    return 0;
}
