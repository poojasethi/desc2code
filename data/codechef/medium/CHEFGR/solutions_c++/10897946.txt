#include<iostream>
using namespace std;
int main()
{
    int n;

    cin>>n;
    while(n--)
    {
        int m,k,mina=111,sum=0;
        cin>>m>>k;
        int arr[m];
        for(int i=0;i<m;i++)
        {cin>>arr[i];
        if(arr[i]<mina)
            mina=arr[i];
        }
        for(int i=0;i<m;i++)
        {
            sum+=(arr[i]-mina);
        }
        if((sum+k)%m==0)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
   
}
 return 0;
}
