#include<stdio.h>
#include<stdlib.h>
int main()
{
    int T,n,k,i,m;
   scanf("%d",&T);
   while(T--)
   {
       scanf("%d%d",&n,&k);
       if(n==1)
       {
           int a;
           scanf("%d",&a);
           printf("Yes\n");
       }
       else
       {
       int *a=(int*)malloc(n*sizeof(int));
       scanf("%d",a);
       m=a[0];
       for(i=1;i<n;i++)
        {
            scanf("%d",a+i);
            if(a[i]>m)
            m=a[i];
        }
        for(i=0;i<n;i++)
        {
            a[i]=m-a[i];
            k=k-a[i];
        }
        if(k>=0 && k%n==0)
        printf("Yes\n");
        else
        printf("No\n");
        free(a);
       }
   }
}
