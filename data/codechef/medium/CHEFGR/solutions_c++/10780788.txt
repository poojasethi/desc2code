#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	long int t;
	cin>>t;
	while(t--)
	{
		long int n,m,sum=0;
		cin>>n>>m;
		long int a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		for(int i=0;i<n;i++)
		{
			sum +=(a[n-1]-a[i]);
		}
		if(sum==m)
		cout<<"Yes"<<endl;
		else if(sum<m)
             {
                 m=m-sum;
                 if(m%n==0)
            	cout<<"Yes"<<endl;
				else
				cout<<"No"<<endl;
			 }
			else
			cout<<"No"<<endl;
	}
	return 0;
}
