#include<stdio.h>
int main()
{
    int t,i,arr[100],n,m,max,min;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        for(i=0,max=0;i<n;i++)
        {
            scanf("%d",arr+i);
            if(arr[i]>max)
                max=arr[i];
        }
        for(i=0,min=0,max=0;i<n;i++)
            min+=max-arr[i];
        min=m-min;
        if(min<0)
            printf("No\n");
        else
        {
            if(min%n==0)
                printf("Yes\n");
            else
                printf("No\n");
        }
    }
    return 0;
}
