#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int cases;
	int count;
	int n;
	int m;
	int max;
	int arr[100];
	cin>>cases;
	for(register int i = 0 ; i< cases; i++)
	{
		
		count = 0;
		cin>>n;
		cin>>m;
		for(register int k = 0 ; k < n; k++)
		{
			cin>>arr[k];}
		max = *(max_element(arr,arr+n-1));
		for(register int k = 0 ; k < n ;k++)
		{
		
		 count+=(max-arr[k]);}
		if( m == count || (m - count) % n == 0)
		cout<<"Yes"<<endl;
		else
		cout<<"No"<<endl;
	}
}
