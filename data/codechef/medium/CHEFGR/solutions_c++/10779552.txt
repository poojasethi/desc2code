#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main() 
{
	long long int t,n,m,ar[100],max,d;
	cin>>t;
	while(t--)
	{
	    d=0;
	    max=INT_MIN;
	    cin>>n>>m;
	    for(int i=0;i<n;i++)
	    {
	        cin>>ar[i];
	        if(ar[i]>max)
	        {
	            max=ar[i];
	        }
	    }
	    for(int i=0;i<n;i++)
	    {
	        d+=max-ar[i];   
	    }
	    if(m<d)
	    {
	        cout<<"No\n";
	    }
	    else
	    {
	        if((m-d)%n==0)
	        {
	            cout<<"Yes\n";
	        }
	        else
	        {
	            cout<<"No\n";
	        }
	    }
	}
	return 0;
}
