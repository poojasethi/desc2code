#include<iostream>
using namespace std;
int main()
{   int t;
cin>>t;
while(t--)
{ int n,m,a[100],max,sum=0,b[100];
cin>>n>>m;
for(int i=0;i<n;i++)
{  cin>>a[i];}
 max=a[0];
for(int i=0;i<n;i++)
{  if(a[i]>max)
max=a[i];}
for(int i=0;i<n;i++)
{  b[i]=max-a[i];
sum=sum+b[i];
}
if(sum==m||(m-sum)%n==0)
cout<<"Yes\n";
else
cout<<"No\n";}return 0;}
