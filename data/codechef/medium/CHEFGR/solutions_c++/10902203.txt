#include<cstdio>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,m;
        scanf("%d %d",&n,&m);
        int tmp,total=0;
        for(int i=0;i<n;i++){
        scanf("%d",&tmp);
        total+=tmp;
        }
        if((total+m)%n==0){
            printf("Yes\n");
        }
        else
        {
            printf("No\n");
        }
    }
}
