#include<bits/stdc++.h>
using namespace std;
int main()
{
     int t;
     cin>>t;
     while(t--)
     {
            int n,m;
             cin>>n>>m;
             int a[n];
             for(int i=0;i<n;i++)
             {
                  cin>>a[i];
             }
             sort(a,a+n);
             int k=a[n-1];

              int count=0;
             for(int i=0;i<n;i++)
             {
                   if(a[i]<k )
                   {
                         int kk=k-a[i];
                         if(m>=kk)
                         {
                               a[i]+=kk;
                               m-=kk;
                         }
                   }

                   if(a[i]==k)
                   {
                        count++;
                   }


             }

              if(count==n)
              {
                    if(m==0)
                    {
                         cout<<"Yes"<<endl;
                    }
                    else if(m>0)
                    {
                          if(m%n==0)
                          {
                               cout<<"Yes"<<endl;
                          }
                          else
                          {
                               cout<<"No"<<endl;
                          }
                    }
              }
              else
              {
                     cout<<"No"<<endl;
              }
     }


     return 0;

}
