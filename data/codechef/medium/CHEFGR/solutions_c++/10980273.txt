#include<stdio.h>
int main()
{
    int t;
    scanf("%ld",&t);
    while(t!=0)
    {
        int n,m,i,max=0,pos,sum=0;
        scanf("%d%d",&n,&m);
        int arr[n];
        for(i=0;i<n;i++)
        {
            scanf("%d",&arr[i]);
            if(arr[i]>max)
            {
                max=arr[i];
                pos=i;
            }
        }
        for(i=0;i<n;i++)
        {
            sum=sum+(arr[pos]-arr[i]);
        }
        pos=0;
        if(sum==m)
        {
            printf("Yes\n");
        }
        else
        {
            while(sum<m)
            {
                sum=sum+n;
                if(sum==m)
                {
                    pos=1;
                }
            }
            if(pos==1)
            {
                printf("Yes\n");
            }
            else
            {
                printf("No\n");
            }
        }
        t--;
    }
    return 0;
}
