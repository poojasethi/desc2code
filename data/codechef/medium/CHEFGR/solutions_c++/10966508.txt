#include<iostream>
using namespace std;
int a[101];
int main(){
    int t;
    cin>>t;
    while(t--){
        int N,M;
        cin>>N>>M;
        int maxi=0;
        for(int i=0;i<N;i++){
            cin>>a[i];
            if(a[i] > maxi){
                maxi=a[i];
            }
        }
        bool flag=false;
        int sum=0;
        for(int i=0;i<N;i++){
            sum+=(maxi-a[i]);
        }
        if(M >= sum){
            M-=sum;
            if(M % N ==0){
                flag=true;
                cout<<"Yes"<<endl;
            }
        }
        if(flag==false){
            cout<<"No"<<endl;
        }
    }
    return 0;
}
