#include<iostream>
using namespace std;
int main()
{
	int t,n,m;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		int a[n],i,x,sum=0,flag=1;
		for(i=0;i<n;i++)
		{cin>>a[i];sum=sum+a[i];}
		x=(m+sum)/n;
		for(i=0;i<n;i++)
		{
			if(a[i]>x)
			flag=0;
		}
		if((m+sum)%n==0&&flag==1)
		cout<<"Yes"<<"\n";
		else
		cout<<"No"<<"\n";
	}
	return 0;
}
