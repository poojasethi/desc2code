#include<stdio.h>
int main(){
	int t,u;
	scanf("%d",&t);
	for(u=1;u<=t;u++){
		int n,m,e,max=0;
		scanf("%d %d",&n,&m);
		int a[n];
		for(e=0;e<n;e++){
		scanf("%d",&a[e]);
		if(a[e]>max)max=a[e];
	}
		int sum1=0;
		for(e=0;e<n;e++)sum1+=max-a[e];
		if(sum1>m)printf("No\n");
		else{
			if((m-sum1)%n==0)printf("Yes\n");
			else printf("No\n");
		}
	}
	return 0;
}
