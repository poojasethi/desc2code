#include<iostream>
#include<map>
#include<string>
using namespace std;
int main()
{
	long long i,j,a[100000],t,n,m,max,c;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		max=0;
		c=0;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			if(a[i]>=max)
			 max=a[i];
		}
		for(i=0;i<n;i++)
		{
			c+=(max-a[i]);
		}
		if(c==m)
		 cout<<"Yes"<<endl;
		else if((c-m)%n==0)
		 cout<<"Yes"<<endl;
		else
		 cout<<"No"<<endl;
	}
}
