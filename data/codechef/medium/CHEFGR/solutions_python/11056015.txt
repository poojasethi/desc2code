#include <iostream>
using namespace std;
int amax(int a[],int n){
    int ma=a[0];
    for(int i =1;i<n;i++){
        if(a[i]>ma){
            ma=a[i];
        }
    }
    return ma;
}
	

int main() {
	// your code goes herecn>>t;
	
	int t,i,l,z;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;
		int a[n];
		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		z=amax(a,n);
		l=0;
		for(i=0;i<n;i++)
		{
		l=l+z-a[i];	
		}
		if(m>l)
		{
			if((m-l)%n==0)
			cout<<"Yes"<<endl;
			else
			cout<<"No"<<endl;
		}
	else if(l==m)
	cout<<"Yes"<<endl;
	else
	cout<<"No"<<endl;
	}
	
	return 0;
}
