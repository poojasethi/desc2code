#include <bits/stdc++.h>
using namespace std;

int myComp(const void *a, const void *b) {
	return *(int*)b - *(int*)a;
}

int main() {
	int T, a[1000], M, i, N;
	cin>>T;
	while(T--) {
		cin>>N>>M;
		for(i=0; i<N; i++) {
			cin>>a[i];
		}
		qsort(a, N, sizeof(int), myComp);
		for(i=1; i<N; i++) {
			M -= a[0] - a[i];
		}
		if(M < 0) {
			cout<<"No\n";
		}else if(M == 0) {
			cout<<"Yes\n";
		}else if(M > 0) {
			if(M%N == 0) {
				cout<<"Yes\n";
			}else{
				cout<<"No\n";
			}
		}
	}
	return 0;
}
