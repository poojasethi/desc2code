#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t,n,m,i,maxi,used,a,left;
	scanf("%d",&t);
	vector<int>vec;
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(i=0;i<n;i++)
		{
			cin>>a;
			vec.push_back(a);
			
		}
		sort(vec.begin(),vec.end());
		maxi=vec[n-1];
		used=0;
		for(i=0;i<n-1;i++)
		{
			if(vec[i]<maxi) used+=(maxi-vec[i]);
		}
		if(used==m) cout<<"Yes"<<endl;
		else if(used<m)
		{
			left=m-used;
			if(left%n==0)
			 cout<<"Yes"<<endl;
			else  cout<<"No"<<endl;
		}
		else  cout<<"No"<<endl;
		vec.clear();
	}
	return 0;
}
