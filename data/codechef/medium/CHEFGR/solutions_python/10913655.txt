#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,i,e=0,r=0,j,g=0,f;
		scanf("%d%d",&n,&m);
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		for(i=1;i<=n&&n>1;i++)
		{
			for(j=0;j<n-i&&i<n;j++)
			{  if(a[j]>a[j+1]){
			
				int temp=a[j];
				a[j]=a[j+1];
				a[j+1]=temp;
			}
		   }
		    if(i==1)
		    {
		    	g=a[n-i];
			}
			else
			{
			r=g-a[n-i];
			  e=e+r;
		    }
		   
		}
		f=m-e;
		if(e==m)
		{
		  printf("Yes\n");
		}
		else if(f%n==0&&f>0)
		{
			printf("Yes\n");
		}
		else
		{
			printf("No\n");
		}
	}
	return 0;
}
