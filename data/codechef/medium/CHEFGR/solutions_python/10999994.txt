#include<bits/stdc++.h>
using namespace std;

int main(){
int t;
scanf("%d",&t);
while(t--){
    int n,k,x,sum = 0;
    scanf("%d %d",&n,&k);
    for(int i = 1 ; i <= n ; i++){
        scanf("%d",&x);
        sum += x;
    }
    sum += k;
   // cout << sum <<" ";
        if((sum%n) == 0)
        printf("Yes\n");
        else
        printf("No\n");
}
return 0;
}
