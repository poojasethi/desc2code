#include<stdio.h>
int main()
{
  int t;
  scanf("%d",&t);
  while(t)
  {
    int n,m;
    scanf("%d%d",&n,&m);
    int i,a[n],max=0,sum=0;
    for(i=0;i<n;i++){
      scanf("%d",&a[i]);
       if(a[i]>max)
         max=a[i];
      }
       for(i=0;i<n;i++)
       {
         if(a[i]<max)
         sum=sum+(max-a[i]); 
       }
       if((m-sum)%n==0)
       printf("Yes\n");
       else
       printf("No\n");
    t--;
  }
  return 0;
}
