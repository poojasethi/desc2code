#include<iostream>
using namespace std;
int main(){
	int i,m,n,flag,count,a[1000],max,t;
	cin>>t;
	while(t--){
		cin>>n;
		cin>>m;
		max=-1;
		for(i=0;i<n;i++){
			cin>>a[i];
			if(a[i]>max){
				max=a[i];
			}
		}
		count=0;
		for(i=0;i<n;i++){
			count=count+max-a[i];
		}
		
		if(count==m)
		   flag=1;
		else if(count<m){
			if((m-count)%n==0)
			    flag=1;
			else
			   flag=0;
		}
		else
		  flag=0;
		if(flag==1)
		cout<<"Yes\n";
		else
		 cout<<"No\n";
	}
}
