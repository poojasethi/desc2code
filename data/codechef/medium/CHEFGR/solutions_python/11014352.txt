#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m;
        cin>>n>>m;

        int arr[n];
        int max=INT_MIN;
        for(int i=0;i<n;i++)
        {
            cin>>arr[i];
            if(arr[i]>max)
                max=arr[i];
        }

        if(n==1)
        {
            cout<<"Yes\n";
            continue;
        }

        int count=0;
        for(int i=0;i<n;i++)
            count +=(max-arr[i]);

        int diff=m-count;
        if( diff % n == 0 )
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }

}
