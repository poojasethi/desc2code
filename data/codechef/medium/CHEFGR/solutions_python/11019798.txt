#include<iostream>
using namespace std;

int main() {
    int t;
    cin>>t;
    while(t--) {
    	int n,m,x,sum=0;
    	cin>>n>>m;
    	sum+=m;
    	for(int i=0;i<n;i++) {
    	   cin>>x;
    	   sum+=x;
        }   
        if(sum%n==0)
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
}
