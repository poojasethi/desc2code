#include <bits/stdc++.h>
using namespace std;

int main() {
    long long int t,sum,n,m,max,i,a[100005];
	cin>>t;
	while(t--)
	{
	    sum=0;
	    cin>>n>>m;
	    max = INT_MIN;
	    for(i=0;i<n;i++)
	    {
	        cin>>a[i];
	        if(a[i]>max)
	        max = a[i];
	    }
	    //cout<<max<<endl;
	    for(i=0;i<n;i++)
	    sum+=(max-a[i]);
	   // cout<<sum;
	    if(sum==m)
	    cout<<"Yes"<<endl;
	    else if(sum>m)
	    cout<<"No"<<endl;
	    else
	    {
	        if((m-sum)%n==0)
	        cout<<"Yes"<<endl;
	        else
	        cout<<"No"<<endl;
	    }
	}
	return 0;
}
