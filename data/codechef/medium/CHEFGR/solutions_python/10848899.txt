#include<bits/stdc++.h>
using namespace std;

void possible(vector<int> v,long int m)
{
int max=INT_MIN;
long int sum=0;
for(int i=0;i<v.size();i++)
{
if(v[i]>max)
max=v[i];
}

for(int i=0;i<v.size();i++)
{
sum+=(max-v[i]);
}

if(sum==m)
{
cout<<"Yes\n";
}
else if(sum>m)
{
	cout<<"No\n";
}
else
{
	if(((m-sum)%v.size())==0)
	{
		cout<<"Yes\n";
	}
	else cout<<"No\n";
	
}

}

int main()
{
int t=0,n=0,a;
long int m;

cin>>t;
while(t--)
{
vector<int> v;
cin>>n>>m;
while(n--)
{
cin>>a;
v.push_back(a);
}
possible(v,m);
}
return 0;
}
