#include<stdio.h>
int main()
{
    int t,i;
    scanf("%d",&t);
    for(i=0;i<t;i++){
        int n,p,k;
        scanf("%d%d",&n,&k);
        int ar[n];
        for(p=0;p<n;p++){
            scanf("%d",&ar[p]);
        }
        int s=0,max;
        max=ar[0];
        for(p=0;p<n;p++){
            s=s+ar[p];
            if(ar[p]>max)
                max=ar[p];
        }
        if(max*n==s+k||s+k>max*n&&(s+k-max*n)%n==0){
                printf("Yes\n");
        }
        else
            printf("No\n");
}
}
