#include<iostream>
using namespace std;
int main(){
	int t,n,m,max,count,i;
	cin>>t;
	while(t--){
		cin>>n>>m;
		int a[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		max=a[0];
		for(i=0;i<n;i++){
			if(max<a[i])
			max=a[i];
		}
		for(i=0,count=0;i<n;i++){
			count=count+max-a[i];
		}
		m=m-count;
		while(m>0){
			m=m-n;
		}
		if(m==0)
		cout<<"Yes";
		else
		cout<<"No";
		cout<<endl;
	}
	return 0;
}
