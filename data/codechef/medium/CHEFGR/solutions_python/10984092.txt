#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t>0)
    {
        int n,m,sum=0;
            scanf("%d",&n);
            int arr[n];
            scanf("%d",&m);
            for(int i=0;i<n;i++)
                    scanf("%d",&arr[i]);
            sort(arr,arr+n);
            for(int j=0;j<n;j++)
                {
                    sum=sum+arr[n-1]-arr[j];
                    if(sum>m)
                        break;
                }
                if(sum==m || (m-sum)%n == 0)
                    printf("Yes\n");
                else
                    printf("No\n");
t--;
}
return 0;
}
