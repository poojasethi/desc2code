#include<iostream>
using namespace std;
int main(){
ios::sync_with_stdio(false);
int t,n,m;
cin>>t;
while(t--){
    cin>>n>>m;
    int a[n];
    int maxm=0,c=0;
    for(int i=0;i<n;i++){
        cin>>a[i];
        maxm=max(a[i],maxm);
    }
    for(int i=0;i<n;i++){
        if(a[i]<maxm && m>=(maxm-a[i])){
            m-=(maxm-a[i]);
            a[i]+=maxm-a[i];
        }
        if(a[i]==maxm) c++;
    }
    if(c==n && m%n==0) cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
}
}
