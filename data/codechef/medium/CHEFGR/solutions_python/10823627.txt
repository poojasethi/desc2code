#include<bits/stdc++.h>
using namespace std;
int t,i,sum,a[100001],m,ma,n;
int main(){
    scanf("%d",&t);
    while(t--){
        cin>>n>>m;
        sum=0;
        ma=0;
        for(i=0;i<n;i++){
            cin>>a[i];
            ma=max(ma,a[i]);
        }

        for(i=0;i<n;i++)
            sum+=ma-a[i];

        if(sum>m)
            cout<<"No"<<endl;
        else{
            m=m-sum;
            if(m%n==0)
                cout<<"Yes"<<endl;
            else
                cout<<"No"<<endl;

        }
    }
return 0;
}
