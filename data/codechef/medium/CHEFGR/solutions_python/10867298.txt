#include<iostream>
using namespace std;
int main()
{
	long sum,T,N,M,max,i,A[101];
	cin>>T;
	while(T--)
	{
		cin>>N>>M;
		max=0;
		for(i=0;i<N;i++)
		{
			cin>>A[i];
			if(A[i]>max)
			max=A[i];
		}
		sum=0;
		for(i=0;i<N;i++)
		{
			sum+=max-A[i];
		}
		if(sum>M)
		{
			cout<<"No\n";
		}
		else if((M-sum)%N==0)
		cout<<"Yes\n";
		else
		cout<<"No\n";
	}
	return 0;
}
