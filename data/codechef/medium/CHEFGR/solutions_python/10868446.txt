#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		int n,m;
		cin>>n>>m;
		int a[n];
		for(int i=0;i<n;i++)
			cin>>a[i];
		sort(a,a+n);
		for(int i=n-2;i>=0;i--){
			if(a[n-1]>a[i]){
				int temp=a[n-1]-a[i];
				if(m>=temp)
					m=m-temp;
				else{
					cout<<"No"<<endl;
					m=-1;
					break;
				}
				a[i]+=temp;
			}
		
			
		}
		while(m>0){
			if(m>=n)
				m=m-n;
			else{
				cout<<"No"<<endl;
				m=-1;
				break;
			}
		}
		if(m!=-1)
		cout<<"Yes"<<endl;
		
	}
	return 0;
}
