# include<cstdio>
# include<algorithm>
# include<math.h>
# define INF 10000
using namespace std;


int dp[101][101];
void floyd_warshall(int n);

int main()
{

    {

        int frd;
        cin>>frd;
        
        int i,j;
        for(i =0;i<=frd;i++)
        {
            for(j = 0;j<=frd;j++) 
            {
               
                dp[i][j] = INF;
                if(i==j)
                {
 
               
                dp[i][j] = 0;
                }
            }
        }
        int frnd;
        for(i = 1;i<=frd;i++)
        {
            scanf("%d",&frnd);
            
            dp[i][frnd] = 1;
            while(getchar()!='\n')
            {
                scanf("%d",&frnd);
                
                dp[i][frnd] = 1;
            }
        }

        floyd_warshall(frd);
        
        int friend_no = 1;
        
        double sum = 0 ,max_sum = 0;
        for(i=1;i<=frd;i++)

            sum = sum + dp[1][i];
                   
        
        }
        sum = sum/frd;

        for(i=2;i<=frd;i++)

            max_sum = 0;
            for(j=1;j<=frd;j++){
            max_sum = max_sum + dp[i][j];

            max_sum = max_sum/frd;
            if(max_sum<sum)
            {sum = max_sum;friend_no=i;}
        }
        printf("%d %0.6lf\n",friend_no,sum);
    }//end of test cases

}


void floyd_warshall(int n)
{

    for(int k =1;k<=n;k++)
    {
        for(int i =1;i<=n;i++)
        {
            for(int j = 1;j<=n;j++)
            {
                dp[i][j] = min(dp[i][j] , dp[i][k]+dp[k][j]);
            }
        }
   }







}
