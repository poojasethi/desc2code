#define FOR(i,x,y) for(int i=(x);i<(y);i++)
#define RFOR(i,x,y) for(int i=(x);i>=(y);i--)
#define MIN(x,y) ((x)<(y)?(x):(y))
#define MAX(x,y) ((x)>(y)?(x):(y))
#define ABS(x) ((x)>0?(x):-(x))
#define SQ(x) ((x)*(x))
#define mset(x,y) memset(x,y,sizeof(x))

typedef long long int lli;
typedef long double ld;

using namespace std;

int adm[110][110];
int mi[110][110];
int main()
{
   // freopen("inp.txt","r",stdin);
    int t,n,a,sum,ind,mind;
    char c;
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	mset(adm,0);
    	for(int i=0;i<n;i++)
    		for(int j=0;j<n;j++)
    		{
    			if(i==j) mi[i][j]=0;
    			else mi[i][j]=100000000;
    		}
    	for(int i=0;i<n;)
    	{
    		scanf("%d",&a);
    		//printf("%d\n",a);
    		a--;
    		mi[i][a]=mi[a][i]=1;
    		c=getchar();
    		if(c=='\n'||c==EOF) i++;
    	}
    	for(int i=0;i<n;i++)
    		for(int j=0;j<n;j++)
    			for(int k=0;k<n;k++)
    				{
    					mi[j][k]=min(mi[j][k],mi[j][i]+mi[i][k]);
    					}
    	// for(int i=0;i<n;i++)
    	// 	{for(int j=0;j<n;j++)printf("%d ",mi[i][j]);printf("\n");}
    	mind=100000000;
    	for(int i=0;i<n;i++)
    	{
    		sum=0;
    		for(int j=0;j<n;j++) sum+=mi[i][j];
    		if(mind>sum){mind=sum;ind=i;}
    	}
    	printf("%d %Lf\n",ind+1,(ld)mind/(ld)n);
    }
    return 0;
    
}
