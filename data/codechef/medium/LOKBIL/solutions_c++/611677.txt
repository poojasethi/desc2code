#include <cstdio>
#include <cstring>
#include <cstdlib>

const int MAX=100;
const int BUFSIZE = 4*(MAX+1);
char buffer[4*(MAX+1)];
int graph[MAX][MAX];

int main()
{
    int G, N, i, j, k;
    char *tok;
    fgets(buffer, MAX, stdin);
    G = atoi(buffer);
    while(G-- > 0)
    {
        for(i=0; i<100; i++)
            for(j=0; j<100; j++)
                graph[i][j] = (i==j)? 0 : MAX;

        fgets(buffer, BUFSIZE, stdin);
        N = atoi(buffer);
        for(i=0; i<N; i++)
        {
            fgets(buffer, BUFSIZE, stdin);
            tok = strtok(buffer, " ");
            while(tok!=NULL)
            {
                graph[i][atoi(tok)-1] = 1;
                tok = strtok(NULL, " ");
            }
        }

        int d;
        for(k=0; k<N; k++)
            for(i=0; i<N; i++)
                for(j=0; j<N; j++)
                    if ((d = graph[i][k] + graph[k][j]) < graph[i][j])
                        graph[i][j] = d;

        int min_index, min, sum;
        min = MAX*MAX;
        for(i=0; i<N; i++)
        {
            for(sum=j=0; j<N; j++)
                if (graph[i][j] < MAX)
                    sum += graph[i][j];
            if (sum < min)
            {
                min = sum;
                min_index = i;
            }
        }
        printf("%d %.6f\n", min_index+1, double(min)/N);
    }
    return 0;
}
