#include<iostream>
#include<vector>
#include<cstdlib>
#include<cstdio>
using namespace std;
int adj[105][105],dist[105];
int mini(int a,int b)
{
    if(a<b)
        return a;
    else
        return b;
}
int main()
{
    int t,n,i,d;
    char c;
    int j,k;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(j=1;j<=n;j++)
        {
            for(k=1;k<=n;k++)
            {
                adj[j][k]=10000;
            }
        }
        for(i=1;i<=n;i++)
        {
            adj[i][i]=0;
            scanf("%d",&d);
            adj[i][d]=1;
            while(getchar()!='\n')
            {
                scanf("%d",&d);
                adj[i][d]=1;
            }
        }



    for(k=1;k<=n;k++)
    {
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                adj[i][j]=mini(adj[i][j],adj[i][k]+adj[k][j]);
            }
        }
    }
    for(i=1;i<=n;i++)
    {
        dist[i]=0;
        for(j=1;j<=n;j++)
        {
            dist[i]+=adj[i][j];
        }
    }
    int mindist=100000;
    int pos=0;
    for(i=1;i<=n;i++)
    {
       if(mindist>dist[i])
       {
           mindist=dist[i];
           pos=i;
       }
    }

    double mindist1=(1.000000*mindist)/n;
    printf("%d %.6f\n",pos,mindist1);
    }

}
