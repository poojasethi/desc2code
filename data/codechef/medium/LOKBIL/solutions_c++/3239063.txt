#include <iostream>
#include <cstdio>
using namespace std;

int adj[101][101];
int dist[101];
int q[101];
int n;

int bfs(int i) {
    int ans = 0;
    int start=0, end=0;
    q[0] = i;
    for(int j=1;j<=n;++j) dist[j] = -1;
    dist[i] = 0;
    while(start<=end) {
        int cur = q[start++];
        for(int j=1;j<=n;++j) {
            if(adj[cur][j]==1 && dist[j]==-1) {
                dist[j] = dist[cur]+1;
                ans = ans+dist[j];
                q[++end] = j;
            }
        }
    }
    return ans;
}

int main() {
    int t, f;
    scanf("%d",&t);
    while(t--) {
        scanf("%d",&n);
        for(int i=1;i<=n;++i) {
            for(int j=1;j<=n;++j) {
                adj[i][j] = 0;
            }
        }
        for(int i=1;i<=n;++i) {
            scanf("%d",&f);
            adj[i][f] = 1;
            while(getchar()!='\n') {
                scanf("%d",&f);
                adj[i][f] = 1;
            }
        }

        int ind=0;
        int res=1000;
        int ans;
        for(int i=1;i<=n;++i) {
            ans = bfs(i);
            if(ans<res) {
                res = ans;
                ind = i;
            }
        }
        double res1=(double)res/n;
        printf("%d %.6f\n",ind,res1);
    }
    return 0;
}
