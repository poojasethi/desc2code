float bfs(int i)
{
       
       queue<int> r,q;
       int  j,visited[101];
      for(j=0;j<=100;j++)
        visited[j]=0;
       float sum=0;
    
      visited[i]=1;
      r.push(i);
      q.push(0);
      while(!r.empty())
      {
         int c,e;
         e=r.front();
         r.pop();
         c=q.front();
         q.pop();
         sum=sum+c;
         for(j=0;j<b[e];j++)
         {
                            int value=a[e][j];
          if(visited[value]==0)
          {
               r.push(value);
               q.push(c+1);
               visited[value]=1;
          }
          }
       }             
       
       return(sum);
               
         
         
         

}
