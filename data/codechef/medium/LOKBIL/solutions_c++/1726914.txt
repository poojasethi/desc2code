#include<stdio.h>
#include<queue>
using namespace std;
int fl[105];
int gr[105][105];
int n;
void bfs(int m)
{
    // printf("bfs from %d\n",m);
fl[m]=1;
queue<int> q;
q.push(m);
int cur,x,i;
while(!q.empty()){
cur=q.front();
q.pop();
for(i=1;i<=n;i++)
{
if(gr[cur][i])
{
if(fl[i]==0){
fl[i]=fl[cur]+1;
//printf("%d at distance %d\n",i,fl[i]);
q.push(i);
}
}
}
}
}

int main()
{
int t;
scanf("%d",&t);
while(t--){
scanf("%d",&n);
char ch;
int i,j;
for(i=0;i<=n;i++)
for(j=0;j<=n;j++)gr[i][j]=0;
int x;
for(i=1;i<=n;i++)
{
                // printf("added to %d are\n",i);
while(scanf("%d%c",&x,&ch)==2){
//printf("%d ",x);
gr[i][x]=1;
gr[x][i]=1;
if(ch=='\n')break;
}
if(ch!='\n'){
            // printf("%d ",x);
             gr[i][x]=1;
gr[x][i]=1;
}
//printf("\n");
}
int ma=10005;
int ans=0;

for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)fl[j]=0;
bfs(i);
int sum=0;
for(j=1;j<=n;j++)sum+=(fl[j]-1);

if(sum<ma){ma=sum;ans=i;}
}
printf("%d %.6lf\n",ans,(double)ma/n);



}


}
