#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m;
        cin>>n>>m;
        int x[n+1];
        for(int i=0;i<n+1;i++)
        x[i]=0;
        int flag=1;
        int res=0;
        for(int i=0;i<m;i++)
        {
            int p,q;
            cin>>p>>q;
            if(x[p]==0)
            x[p]=1;
            if(x[q]==0)
            x[q]=x[p]+1;
            if(x[p]>x[q])
            flag=0;
            if(res<x[q])
            res=x[q];
        }
        if(flag)
        cout<<res<<endl;
        else
        cout<<"IMPOSSIBLE"<<endl;
    }
    return 0;
}
