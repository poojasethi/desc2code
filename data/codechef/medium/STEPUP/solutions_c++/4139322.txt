#include<iostream>
using namespace std;
int main()
{
    int t,n,e,a,b;
    cin>>t;
    while(t--)
    {
              int ans = 1,maxVal = 0;
              cin>>n>>e;
              int func[n+1];
              for(int i=0;i<=n;i++)
                      func[i]=0;
              for(int i=0;i<e;i++)
              {
                      cin>>a>>b;
                      if(func[a]==0)
                                    func[a]=1;
                      if(func[b]==0)
                                    func[b]=func[a]+1;
                      if(func[b]<func[a])
                                         ans = 0;
                      if(maxVal < func[b])
                                maxVal = func[b];
              }
             
              if(ans == 0)
                     cout<<"IMPOSSIBLE\n";
              else
                  cout<<maxVal<<endl;
    }
    return 0;
}
