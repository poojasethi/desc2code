#include <iostream>
#include <string.h>
using namespace std;
 
int vertex[10010];
int main()
{
	int t;
	cin>>t;
	while(t--){
		int a,b;
		cin>>a>>b;
		memset(vertex, 0, sizeof(vertex));
		int s = 1,max = -1;
		for(int i=0;i<b;i++){
			int x,y;cin>>x>>y;
			if(s){
				if(vertex[x]==0)vertex[x]=1;
				if(vertex[y]==0)vertex[y]=vertex[x]+1;
				if (vertex[y] < vertex[x]) s = 0;
				if (max < vertex[y]) max = vertex[y];
			}
			
		}
		if(!s)cout<<"IMPOSSIBLE\n";
			else cout<<max<<endl;
			
	}
    return 0;
}
