#include <iostream>
#include <iomanip>
#include <math.h>
using namespace std;
 
int main()
{   int x;
    cin>>x;
    while(x--)
    {   long a,b;
        float p,q,s,t,v;
        cin>>a>>b;
        p=a/12;
        if(6*p*p==b)
        {   printf("%.2f\n",p*p*p);
        }
        else
        {   p=((a+sqrt(a*a-24*b))/12);
            q=((a-sqrt(a*a-24*b))/12);
            s=(a-8*p)/4;
            t=(a-8*q)/4;
            v=p*p*s;
            cout<<fixed;
            if(v>q*q*t)
                printf("%.2f\n",v);
            else
                printf("%.2f\n",q*q*t);
        }
    }
    return 0;
}
