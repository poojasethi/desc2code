#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
double volume(float p,float s)
{   float d,h1,v1,v2,h2;
    d=(pow(p,2)/4)-6*s;
    d=sqrt(d);
    h1=((p/2)+d)/6;
    h2=((p/2)-d)/6;
    v1=h1*((s/2)-(h1*((p/4)-h1)));
    v2=h2*((s/2)-(h2*((p/4)-h2)));
    if(v1>=v2)
        return v1;
    else
        return v2;
}
int main()
{   int t,i;
    float p,s,v;
    cin>>t;
    for(i=0;i<t;i++)
    {   cin>>p>>s;
        v=volume(p,s);
        printf("%0.2f\n",v);
    }
    return 0;
}
