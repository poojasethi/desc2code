#include<bits/stdc++.h>
#include<cstdio>
using namespace std;
typedef long long ll;
int main(){
    int t,p,s;
    cin>>t;
    while(t--){
        cin>>p>>s;
        ll D=((ll)p*(ll)p)-((ll)24*(ll)s);
        double d=pow(D,0.5);
        d=((double)p+(double)2*d)*((double)p-d)*((double)p-d)/(double)1728;
        printf("%0.2lf\n",d);
    }
}
