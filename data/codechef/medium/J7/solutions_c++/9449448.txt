#include<iostream>
#include<math.h>
#include<cstdio>

using namespace std;

int main()
{
    int64_t t,p,s,i;
    double b,a,vol,d;
    cin>>t;
    for(i=0;i<t;i++)
    {
        cin>>p>>s;
        d=sqrt((p*p)-(24*s));
        b=(p-d)/12;
        if(b>0)
        {
            a=(p-8*b)/4;
        }
        else
        {
            b=(p+d)/12;
            a=(p-8*b)/4;
        }
        vol=a*b*b;
        printf("%.2f\n",vol);
    }
    return 0;
}
