#include <iostream>
#include<cmath>
#include<iomanip>
using namespace std;

int main() {
	int t;
	double l1,l2,p,a,h1,h2,v1,v2;
	cin>>t;
	while(t--)
	{
		cin>>p>>a;
		l1=(p+sqrt(p*p-24*a))/12;
		h1=(p/4)-2*l1;
		l2=(p-sqrt(p*p-24*a))/12;
		h2=(p/4)-2*l2;
		v1=l1*l1*h1;
		//cout<<v1<<endl;
		v2=l2*l2*h2;
		//cout<<v2<<endl;
		v1>v2?(cout<<fixed<<setprecision(2)<<(v1+0.005)<<endl):(cout<<fixed<<setprecision(2)<<(v2+0.005)<<endl);
	}
	return 0;
}
