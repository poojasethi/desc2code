#include<cstdio>
#include<math.h>
int main()
{
 int t,i;
 scanf("%d",&t);
 for(i=0;i<t;i++)
 {
  int p,s;
  scanf("%d%d",&p,&s);

  float dr,y1,y2,x1,x2,v1,v2;
  dr=p*p-(24*s);

  dr=sqrt(dr);

    //printf("%f ",dr);
  y1=(p+dr)/12;
  // printf("%f ",y1);
  y2=(p-dr)/12;
  // printf("%f ",y2);
  x1=p/4-2*y1;
   //printf("%f ",x1);
  x2=p/4-2*y2;
   //printf("%f ",x2);
  v1=x1*y1*y1;
   //printf("%f ",v1);
  v2=x2*y2*y2;
   //printf("%f ",v2);
  if(v1>v2)
   printf("%0.2f\n",v1);
  else
   printf("%0.2f\n",v2);

 }

 return 0;
}
