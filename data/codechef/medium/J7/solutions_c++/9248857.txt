#include <iostream>
#include<cmath>
#include<cstdio>
using namespace std;

int main() {
	float t,p,s,x,y,h,v;
	cin>>t;
	while(t-->0)
	{
	    cin>>p>>s;
	    h=(p-(sqrt(p*p-24*s)))/12;
	    y=((p/4-h)-(sqrt((p/4-h)*(p/4-h)-4*(s/2-h*(p/4-h)))))/2;
	    x=p/4-h-y;
	    v=x*y*h;
	    printf("%.2f\n",v);
	}
	return 0;
}
