#include<bits/stdc++.h>
#define ll long long int

using namespace std;

int main(){
	ll t,p,s;

	double P,S,V1,V2,x1,x2;
	cin>>t;

	while(t--){
		cin>>p>>s;
		P=(double)(p)/4.00;
		S=(double)(s)/2.00;

		x1=(P+sqrt(P*P-3*S))/3.00;
		V1=(P-2*x1)*x1*x1;

		x2=(P-sqrt(P*P-3*S))/3.00;
		V2=(P-2*x2)*x2*x2;

		if(V1>V2)
			printf("%0.2f\n",V1);
		else
			printf("%0.2f\n",V2);
	}
	return 0;
}
