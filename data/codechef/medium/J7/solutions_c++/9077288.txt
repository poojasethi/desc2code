#include <iostream>
#include <iomanip>
#include <fstream>
#include <cmath>
 
using namespace std;
 
int main(){
    int t;
    scanf("%d",&t);
    
    int P,S;
    double l,V;
    for(int i=0; i<t; i++){
	scanf("%d %d",&P,&S);
	l = P-sqrt(P*P-24*S);
	l /= 12;
	V = S*l/2 - l*l*P/4 + l*l*l;
	cout << fixed << showpoint << setprecision(2) << V<<"\n";
    }
    
    return 0;
}
