#include <iostream>
#include<math.h>
#include<iomanip>

	
using namespace std;

int main() {
	
	int t;
	cin >> t;
	
	while(t--)
	{
		float P, S;
		float l1, l2, l, V;
		
		cin >> P >> S;
		
		l1 = (P + sqrt(P*P - 24*S))/12;
		l2 = (P - sqrt(P*P - 24*S))/12;
		
		if(-P/2 + 6*l1 < 0)
			l = l1;
			
		else
			l = l2;
			
		V = l*(S/2 - l*(P/4 - l));		
		
		 
		cout << fixed;
    	cout << setprecision(2);
    	cout << V << endl;
	}
	return 0;
}
