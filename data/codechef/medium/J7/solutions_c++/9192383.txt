#include<iostream>
#include<iomanip>
#include<math.h>
#include<stdio.h>
using namespace std;
int main()
{
  int t;
  double p,s,a,b,v;
  cin>>t;
  while(t--)
  {
    cin>>p>>s;
    b=p-sqrt((p*p)-(24*s));
    a=(3*p-2*b)/12;
  //  v=(a*((2*p*p)-(24*s)+2*p*sqrt((p*p)-(24*s))))/576;
    v=a*b*b/144;
    //cout<<v<<setprecision(2)<<endl;
    printf("%.2f\n",v);
  }
  return 0;
}
