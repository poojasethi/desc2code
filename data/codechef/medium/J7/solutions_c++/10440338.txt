#include<bits/stdc++.h>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	for(int l=0;l<t;l++)
	{
		float p,s,q1,q2,r1,r2,r,q;
		cin>>p;
		cin>>s;
		q1=(p-sqrt(p*p-24*s))/12;
		q2=(p+sqrt(p*p-24*s))/12;
		if(q1<0)
		q1=0;
		if(q2<0)
		q2=0;
		r1=q1*q1*(p/4-2*q1);
		r2=q2*q2*(p/4-2*q2);
		if(r1>r2)
		r=r1;
		else r=r2;
 
        cout<<setprecision(2)<<fixed<<r<<endl;
 
	}
	return 0;
}
