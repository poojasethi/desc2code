    #include<bits/stdc++.h>
    using namespace std;
    int main(){
    	int n;
    	float p,q,b,l,res;
    	cin>>n;
    	for(int i=0;i<n;i++){
    		cin>>p>>q;
    		p=p/4;
    		q=q/2;
    		b=2*p-pow((4*p*p-12*q),.5);
    		b=b/6;
    		l=p-2*b;
    		res=l*b*b;
    		printf("%.2f\n",res);
    	}
    }
