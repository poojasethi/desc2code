#include<iostream>
#include<cmath>
using namespace std;
 
int main()
{
int T;
cin >> T;
while(T>0)
{
	int P=0,S=0;
	cin >> P >>S;
//	cout<<P<<S<<endl;
	float x1=0,x2=0,y1=0,y2=0,v1=0,v2=0;
	y1=(P+sqrt(P*P-24*S))/12;
	y2=(P-sqrt(P*P-24*S))/12;
	x1=P/4 -2*y1;
	x2=P/4 -2*y2;
//	cout<<x1<<y1<<endl;
//	cout<<x2<<y2<<endl;
	v1=x1*y1*y1;
	v2=x2*y2*y2;
//	cout<<v1<<endl;
//	cout<<v2<<endl;
	if(v1>v2)
	{
		cout<<v1<<endl;
	}
	else
	{
		cout<<v2<<endl;
	}
	

	T--;
}
return 0;
 
 
}
