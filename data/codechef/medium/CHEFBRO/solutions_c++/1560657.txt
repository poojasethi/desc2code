#include <iostream>
#include <cstdio>
using namespace std;

int a[1111][1111];

int main()
{
    int t, c, x[22], y[22], d[22];
    scanf("%d", &t);
    while (t--) {
        scanf("%d", &c);
        for (int i=0; i<c; i++) {
            scanf("%d %d", &x[i], &y[i]);
            --x[i]; --y[i];
            d[i] = (x[i]+y[i])%3;
        }
        int g = d[0];
        for (int i=1; i<c; i++)
            g = g^d[i];
        if (g) printf("MasterChef\n");
        else printf("Football\n");
    }
    return 0;
}
