#define DEBUG 0

#include <stdio.h>

int main(void) {
  int tc, c, n, m, ans;
  scanf("%d", &tc);
  while(tc--) {
    scanf("%d", &c);
    ans=0;
    while(c--) {
      scanf("%d%d", &n, &m);
      ans^=(n+m-2)%3;
    }
    if(ans)
      printf("MasterChef\n");
    else
      printf("Football\n");
  }
  return 0;
}
