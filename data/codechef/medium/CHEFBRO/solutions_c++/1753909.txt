#include<iostream>
#include<stdio.h>
#include<string.h>
#include<vector>
#include<algorithm>
#include<map>
#include<math.h>
#include<climits>
using namespace std;

int main()
{
	int t,c,n,m,ans;
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        scanf("%d",&c);
        while(c--)
        {
            scanf("%d%d",&n,&m);
            ans^=(n+m+1)%3;
        }
        if(ans)
            printf("MasterChef\n");
        else
            printf("Football\n");
    }
	return 0;
}
