#include <iostream>
#include <cstdio>

using namespace std;

#define FOR(i,a,b) for(int i=a;i<b;i++)
#define FORD(i,a,b) for(int i=a;i>=b;i--)
#define REP(i,n) FOR(i,0,n)
#define PB push_back
#define ITER(i,a) for( typeof(a.begin()) i=a.begin();i!=a.end();i++)
#define mod 1000000007
#define MAXN 1000010
int g[1010][1010];

int main(){

	int maxm=0;

	FOR(i,1,1001)
		FOR(j,1,1001)
		{
			if(i==1 && j==1)
			{
				g[i][j]=0;
				continue;
			}
			int mex[10]={0};

			if(i-1>0)
				mex[g[i-1][j]]=2;
			if(i-2>0)
				mex[g[i-2][j]]=2;
			if(i-1>0 && j-1>0)
				mex[g[i-1][j-1]]=2;
			if(i-2>0 && j-2>0)
				mex[g[i-2][j-2]]=2;
			if(j-1>0)
				mex[g[i][j-1]]=2;
			if(j-2>0)
				mex[g[i][j-2]]=2;


			int ctr=0;
			while(mex[ctr]) ctr++;

			g[i][j]=ctr;

			//maxm=max(maxm,g[i][j]);
		}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int C,n,m,ans=0;

		scanf("%d",&C);
		REP(i,C)
		{
			scanf("%d%d",&n,&m);
			ans^=g[n][m];
		}

		if(ans)
			printf("MasterChef\n");
		else
			printf("Football\n");
	}

	return 0;
}
