#include <cstdio>

int main(){
	int T;
	scanf("%d", &T);

	while(T--){
		int C;
		scanf("%d", &C);

		int w[3] = {};
		for (int i = 0; i < C; i++){
			int n, m;
		       	scanf("%d %d", &n, &m);
			w[(n+m-1)%3]++;
		}

		bool result = false;
		if(w[0]&1) result = true;
		else if (w[2]&1) result = true;
		
		if (result) printf("MasterChef\n");
		else printf("Football\n");
	}
	return 0;
}
