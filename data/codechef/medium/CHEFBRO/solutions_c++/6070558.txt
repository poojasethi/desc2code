#include <iostream>
using namespace std;
#include<cstdio>
int main() 
{
     int t,cp,c,n,m;
     cin>>t;
    while(t--)
    {
       cin>>c;
       cp=0;
       while(c--)
       {
           cin>>n>>m;
           cp^=(n+m-2)%3;
       }
       if(cp!=0)
        cout<<"MasterChef\n";
       else
        cout<<"Football\n";
    }
	return 0;
}
