#include <cstdio>
#include <utility>
#include <cstring>

using namespace std;

char table[1024][1024];
int N;
pair < int, int > p[32];

void scan(){
    scanf ( "%d", &N );

    for ( int i = 0; i < N; ++i )
        scanf ( "%d%d", &p[i].first, &p[i].second );
}

void solve(){
    int ans = ( p[0].first + p[0].second + 1 ) % 3;;

    for ( int i = 1; i < N; ++i )
        ans ^= ( p[i].first + p[i].second + 1 ) % 3;

    if ( !ans )
        printf ( "Football\n" );
    else
        printf ( "MasterChef\n" );
}


int main(){
    int tests;

    scanf ( "%d", &tests );

    while ( tests -- ){
        scan();
        solve();
    }
}
