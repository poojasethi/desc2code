#include <cstdio>
#include <utility>
#include <cstring>

using namespace std;

char table[1024][1024];
int N;
pair < int, int > p[32];

void scan(){
    scanf ( "%d", &N );

    for ( int i = 0; i < N; ++i )
        scanf ( "%d%d", &p[i].first, &p[i].second );
}

void solve(){
    int ans = table[ p[0].first ][ p[0].second ];

    for ( int i = 1; i < N; ++i )
        ans ^= table[ p[i].first ][ p[i].second ];

    if ( !ans )
        printf ( "Football\n" );
    else
        printf ( "MasterChef\n" );
}

void fill(){
    for ( int i = 1; i <= 1000; ++i )
        for ( int j = 1; j <= 1000; ++j ){
            char used[7];
            memset ( used, 0, sizeof ( used ) );

            if ( i > 1 )
                used[ table[i - 1][j] ] = 1;
            if ( j > 1 )
                used[ table[i][j - 1] ] = 1;
            if ( i > 2 )
                used[ table[i - 2][j] ] = 1;
            if ( j > 2 )
                used[ table[i][j - 2] ] = 1;
            if ( i > 1 && j > 1 )
                used[ table[i - 1][j - 1] ] = 1;
            if ( i > 2 && j > 2 )
                used[ table[i - 2][j - 2] ] = 1;

            for ( int k = 0; k < 7; ++k )
                if ( !used[k] ){ table[i][j] = k; break; }
        }
}

int main(){
    int tests;

    scanf ( "%d", &tests );

    fill();
    while ( tests -- ){
        scan();
        solve();
    }
}
