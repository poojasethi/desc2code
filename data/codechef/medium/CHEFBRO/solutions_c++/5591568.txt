#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int MAX = 1003;
int grundy[MAX][MAX];
bool mex[5];
void solve(){
    for(int i=1;i<=1000;i++){
        for(int j=1;j<=1000;j++){
            if(i==1 && j==1){
                grundy[i][j] = 0;
                continue;
            }
            memset(mex,false,sizeof mex);
            if(i-1 > 0) mex[grundy[i-1][j]] = true;
            if(i-2 > 0) mex[grundy[i-2][j]] = true;
            if(j-1 > 0) mex[grundy[i][j-1]] = true;
            if(j-2 > 0) mex[grundy[i][j-2]] = true;
            if(i-1 >0  && j-1>0) mex[grundy[i-1][j-1]] = true;
            if(i-2>0 && j-2>0) mex[grundy[i-2][j-2]] = true;
            int res = 0;
            while(mex[res])res++;
            grundy[i][j] = res;
        }
    }
}
int main(){
int t,c,n,m;
solve();
scanf("%d",&t);
while(t--){
  scanf("%d",&c);
  int res = 0;
  for(int i=0;i<c;i++){
    scanf("%d %d",&n,&m);
    res = res xor grundy[n][m];
  }
  if(res) printf("MasterChef\n");
  else printf("Football\n");
}
return 0;
}
