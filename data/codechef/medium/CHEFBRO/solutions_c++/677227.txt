#include <cstdio>
#include <utility>
#include <cstring>

using namespace std;

int N;
pair < int, int > p[32];

void solve(){

    int ans = 0;
    scanf ( "%d", &N );

    for ( int i = 0; i < N; ++i ){
        scanf ( "%d%d", &p[i].first, &p[i].second );
        ans ^= ( p[i].first + p[i].second + 1 ) % 3;
    }

    if ( !ans )
        printf ( "Football\n" );
    else printf ( "MasterChef\n" );

}


int main(){
    int tests;

    scanf ( "%d", &tests );

    while ( tests -- )
        solve();
}
