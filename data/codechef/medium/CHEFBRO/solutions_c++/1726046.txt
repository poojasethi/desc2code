#include <iostream>
 
using namespace std;
 
int e[8];
int a[1024][1024];
 
int main() {
    int cnt;
    for(int i = 1; i <= 1000; i++) for(int j = 1; j <= 1000; j++) {
        cnt = 0;
        if(i > 1) e[cnt++] = a[i-1][j];
        if(i > 2) e[cnt++] = a[i-2][j];
        if(j > 1) e[cnt++] = a[i][j-1];
        if(j > 2) e[cnt++] = a[i][j-2];
        if(i > 1 && j > 1) e[cnt++] = a[i-1][j-1];
        if(i > 2 && j > 2) e[cnt++] = a[i-2][j-2];
        int k = 0, done = 0;
        while(!done) {
            done = 1;
            for(int p = 0; p < cnt && done; p++) if(e[p] == k) done = 0;
            if(!done) k++;
        }
        a[i][j] = k;
    }
    int T;
    cin >> T;
    while(T--) {
        int C;
        cin >> C;
        int nu = 0;
        for(int i = 0; i < C; i++) {
            int p, q;
            cin >> p >> q;
            nu ^= a[p][q];
        }
        if(nu) cout << "MasterChef" << endl;
        else cout << "Football" << endl;
    }
    return 0;
}
