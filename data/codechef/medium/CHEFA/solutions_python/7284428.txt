def chefAndEasyProblem_getMax(n):
    nStr = raw_input().split(" ")
    nInt = [int(x) for x in nStr[:n]]
    nInt.sort(reverse=1)
    return sum(nInt[i] for i in range(len(nInt)) if (i % 2 == 0))

def chefAndEasyProblem():
    for test in range(input()):
        print chefAndEasyProblem_getMax(input())

chefAndEasyProblem()
