#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    //cout << "Hello world!" << endl;
   long long int t,n;
   cin>>t;
   while(t--)
   {

       cin>>n;
       long long int a[n],i,sum=0;
       for(i=0;i<n;++i)
       {

           cin>>a[i];
       }
       sort(a,a+n);
       for(i=n-1;i>=0;i=i-2)
       {

           sum+=a[i];
       }
       cout<<sum<<"\n";
   }

    return 0;
}
