#include<iostream>
#include<cstring>
#include<cstdio>
#include<vector>
#include<cstdlib>
#include<algorithm>
#include<cmath>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    long long int n,num;
    while(t--)
    {
        scanf("%lld",&n);
        vector<long long int> vec;
        for(long long int i=0;i<n;i++)
        {
            scanf("%lld",&num);
            vec.push_back(num);
        }
        sort(vec.begin(),vec.end());
        long long int maxi=0;
        for(long long int i=vec.size()-1;i>=0;i=i-2)
        {
             maxi=maxi+vec[i];
        }
        printf("%lld\n",maxi);
    }
    return 0;
}
