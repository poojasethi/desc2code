#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		long long int a[n],s=0;
		for(int i=0;i<n;i++)
			scanf("%lld",&a[i]);
		sort(a,a+n);
		while(n>0){
			s+=a[n-1];
			n-=2;
		}
		printf("%lld\n",s);
	}
	return 0;
}
