#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	for(scanf("%d", &t); t--; )
	{
		int n;
		scanf("%d", &n);
		long a[n];
		for(int i=0; i<n; i++)
			scanf("%ld", &a[i]);
		sort(a, a+n);
		long long sum=0;
		for(int i=n-1; i>=0; i-=2)
			sum+=a[i];
		printf("%lld\n", sum);
	}
	return 0;
}
