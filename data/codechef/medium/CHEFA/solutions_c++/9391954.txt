#include <iostream>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    long long int sum=0;
    cin>>t;
    while(t--){
            sum=0;
    int n;
    cin>>n;

    long long int *p =new long long int[n];
    for(int i=0;i<n;i++)
        cin>>p[i];
    sort(p,p+n);
    for(int i=n-1;i>=0;i-=2)
      sum=sum+p[i];
    cout<<sum<<"\n";
    }


    return 0;
}
