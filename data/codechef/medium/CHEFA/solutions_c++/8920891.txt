#include <bits/stdc++.h>
using namespace std;

int main() {
    int t;
    cin >> t;

    while(t--) {
        int n;
        cin >> n;

        int a[n],i;
        for(i = 0;i < n;i++) {
            cin >> a[i];
        }
        sort(a,a+n);
        /*for(i = 0;i < n;i++) {
            cout << a[i] << " ";
        }
        cout << "\n";*/

        long long int ans = 0;
        if((n%2) == 1) {
            for(i = 0;i < n;i = i + 2) {
                ans += a[i];
            }
            cout << ans << "\n";
        }

        else {
            for(i = 1;i < n;i = i + 2) {
                ans += a[i];
            }
            cout << ans << "\n";
        }
    }
    return 0;
}
