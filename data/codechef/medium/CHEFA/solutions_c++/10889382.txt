#include <bits/stdc++.h>

using namespace std;

struct node
{
	int x,y;
};
bool cmp(node a,node b)
{
	if (a.x < b.x)
		return true;
	else if (a.x == b.x)
	{
		if (a.y >= b.y)
			return true;
		else
			return false;
	}
	else
		return false;
}
int  main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n;
		scanf("%d",&n);
		vector <int> a(n);
		for (int i = 0; i < n; i++)
			scanf("%d",&a[i]);
		sort(a.rbegin(),a.rend());
		long long int ans = 0;
		for (int i = 0; i < n;)
		{
			ans += a[i];
			i = i+2;
		}
		cout << ans << endl;
	}
	return 0;
}
