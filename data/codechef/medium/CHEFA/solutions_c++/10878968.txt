#include<cstdio>
#include<algorithm>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        ll arr[n];
        for(int i=0;i<n;i++){
            scanf("%lld",&arr[i]);
        }
        sort(arr,arr+n);
        ll s1=0,s2=0;
        for(int i=0;i<n;i++){
            if(i%2){
                s1+=arr[i];
            }
            else
            {
                s2+=arr[i];
            }
        }
        printf("%lld\n",max(s1,s2));
    }
    
}
