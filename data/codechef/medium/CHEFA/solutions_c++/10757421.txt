#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n,sum=0;
        cin>>n;
        long long int a[n];
        for(long long int i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);
        for(long long int i=n-1;i>=0;i-=2)
        {
            sum+=a[i];
        }
        cout<<sum<<endl;
    }
    return 0;
}
