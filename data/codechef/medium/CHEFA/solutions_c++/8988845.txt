#include<cstdio>
#include<stdlib.h>

int cmpfn(const void * a,const void* b)
{
	const int long long *ptr=(const int long long *)a;
	const int long long *ptr2=(const int long long *)b ;
	return (*ptr2)-(*ptr);
}
main()
{
	int long long test;
	int long long a[100010],sum,n,i;
	scanf("%lld",&test);
	while(test--)
	{
		sum=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		   scanf("%lld",&a[i]);
		qsort(a,n,sizeof(int long long),cmpfn);
		i=0;
		while(i<n)
		{
			sum+=a[i];
			i+=2;
		}
		printf("%lld\n",sum);
	}
}
