#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
         long n,a[100001];
         long long sum=0;
        scanf("%ld",&n);
        for(long i =0;i<n;i++){
           scanf("%ld",&a[i]); 
        }
        sort(a,a+n);
        for(long i=n-1;i>=0;i-=2){
            sum+=a[i];
        }
        printf("%lld\n",sum);
    }
}
