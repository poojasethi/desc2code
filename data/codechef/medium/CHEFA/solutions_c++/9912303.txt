#include <iostream>
#include <cstdio>
#include <algorithm>
#include <math.h>
#include <string.h>
#include <stdio.h>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n;
        scanf("%d", &n);
        long long int arr[n], ans = 0;
        for(int i = 0; i < n; i++)
        {
            scanf("%lld", &arr[i]);
        }
        sort(arr, arr + n);

        for(int i = n - 1; i >= 0; i -= 2)
        {
            ans += arr[i];
        }
        printf("%lld\n", ans);
    }
    return 0;
}
