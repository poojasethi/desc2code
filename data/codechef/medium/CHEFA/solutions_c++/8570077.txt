#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int T;
	for(scanf("%d",&T);T--;){
		int n;
		long long int sum=0;
		scanf("%d",&n);
		long int a[n];
		for(int i=0;i<n;i++)
		scanf("%lld",&a[i]);
		sort(a,a+n);
		for(int i=n-1;i>=0;i-=2)
		sum+=a[i];		
		printf("%lld\n",sum);
	}
	return 0;
}
