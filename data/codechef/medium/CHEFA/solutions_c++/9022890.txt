#include<iostream>
#include<stdlib.h>
using namespace std;
int cmpfunc(const void*a, const void*b)
{
	return(*(int*)b - *(int*)a);
}
int main()
{
	long int T, arr[100002], N, i, j;
	long long int res;
	cin >> T;
	while (T > 0)
	{
		res = 0;
		cin >> N;
		for (i = 0;i < N;i++)
			cin >> arr[i];
		qsort(arr, N, sizeof(int), cmpfunc);
		for (i = 0;i < N;i = i + 2)
			res = res + arr[i];
		cout << res << endl;
		T--;
	}
//	cin >> T;
	return 0;
}
