#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long i,n,a[100001],sum=0;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        sort(a,a+n);
        for(i=n-1;i>=0;i=i-2)
        {
            sum=sum+a[i];
        }
        cout<<sum<<"\n";
        
        
    }
    
	return 0;
}
