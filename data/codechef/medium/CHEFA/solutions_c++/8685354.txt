#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,i,s=0,s1=0;
		scanf("%lld",&n);
		long long int a[n],b[n];
		for(i = 0; i < n; i++) scanf("%lld",&a[i]);
		sort(a,a+n);
		for(i = 0; i < n; i++)
		b[i] = a[i];
		for(i = 0; i <n; i++)
		{
			if(i%2 == 0)
			s= s + b[i];
			else
			s1= s1+b[i];
		}
		if(s1 > s)
		printf("%lld\n",s1);
		else
		printf("%lld\n",s);
		
	}
}
