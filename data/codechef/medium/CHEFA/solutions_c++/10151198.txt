#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,i,a[100005];
		long long int sum=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		for(i=n-1;i>=0;i-=2)
			sum+=a[i];
		printf("%lld\n",sum);
	}
	return 0;
}
