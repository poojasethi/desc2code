#include<bits/stdc++.h>
using namespace std;
void pre()
{
  ios_base::sync_with_stdio(false);
  cin.tie(NULL);
  cout.tie(NULL);
}
int main()
{
  pre();
  int t;
  cin>>t;
  while(t--)
  {
      int n;
      long long ans=0;
      cin>>n;
      int a[n],i;
      for(i=0;i<n;i++)
      	cin>>a[i];
      sort(a,a+n,greater<int>());
      for(i=0;i<n;i+=2)
      	ans+=a[i];
      cout<<ans<<endl;
  }
  return 0;
}
