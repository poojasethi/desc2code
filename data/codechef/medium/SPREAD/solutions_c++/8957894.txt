#include<bits/stdc++.h>
using namespace std;
int n,c;
long long int v[1000001];
struct bitree
{
     void update(int i, int z)
     {

         while(i<=n)
         {
             v[i]+=z;
             i+=i&(-i);
         }
     }
     long long int sum(int i)
     {
         long long int r=0;
         while(i)
         {

             r+=v[i];
             i-=i&(-i);
         }
         return r;
     }

};
int main()
{
    int m;
    scanf("%d%d%d",&n,&m,&c);
    bitree bt;

    while(m-->0)
    {
        char ch;
        scanf(" %c",&ch);
        if(ch=='S')
        {

            int u,v,k;
           scanf("%d%d%d",&u,&v,&k);
            bt.update(u,k);
            bt.update(v+1,-k);
        }
        else
        {
            int p;
           scanf("%d",&p);
            long long int s=bt.sum(p)+c;
           printf("%lld\n",s);
        }
    }
}
