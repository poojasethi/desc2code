void update(int idx,int val,int n)
{
	while(idx<=n)
	{
		bit[idx]+=val;
		idx+=(idx&(-idx));
	}
} 
ll query(int idx)
{
	ll rt=0;
	while(idx)
	{
		rt+=bit[idx];
		idx-=(idx&(-idx));
	}
	return rt;
}
