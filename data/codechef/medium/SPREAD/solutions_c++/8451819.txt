#include<bits/stdc++.h>
#define ll long long int
#define gc getchar
using namespace std;
ll BIT[1000005];
int n;
int in() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
 int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
void update(int idx,int v)
{
    while(idx<=n)
    {
        BIT[idx]+=v;
        idx+=(idx&-idx);
    }
}
ll query(int idx)
{
    ll res=0;
    while(idx>0)
    {
        res+=BIT[idx];
        idx-=(idx&-idx);
    }
    return res;
}
int main()
{
    int m,c,u,v,k;
    ll ans;
    char ch;
    //scanf("%d%d%d",&n,&m,&c);
    n=in();m=in();c=in();
    while(m--)
    {
       ch = getchar();
    while(!(ch=='Q' || ch=='S'))
      ch = getchar();
        if(ch=='S')
        {
            //scanf("%d%d%d",&u,&v,&k);
            u=in();v=in();k=in();
            update(u,k);
            update(v+1,-k);
        }
        else
        {
           // scanf("%d",&v);
           v=in();
            ans=c+query(v);
            printf("%lld\n",ans);
        }
    }
    return 0;
}
