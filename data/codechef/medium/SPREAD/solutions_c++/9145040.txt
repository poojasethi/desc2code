#define x getchar_unlocked()
#define y putchar_unlocked
using namespace std;
inline void inp( int &n ) {//fast input function
	n=0;
	int ch=x,sign=1;
	while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=x;}
	while( ch >= '0' && ch <= '9' )
 		n=(n<<3)+(n<<1)+ ch-'0', ch=x;
	n=n*sign;
	}
 
long long int A[1000001]={0};
int n,m,c,a,b,k,p;
 
void update(int idx,int val) {
	while(idx<=n) {
		A[idx]+=val;
		idx+=(idx&-idx);
		}
	}
long long int query(int idx) {
	long long int sum=0;
	while(idx>0) {
		sum+=A[idx];
		idx-=(idx&-idx);
		}
	return sum;
	}
 
int main() {
	char ch;
	inp(n);
	inp(m);
	inp(c);
	update(1,c);
	while(m--) {
		ch=x;
		while(!(ch=='Q' || ch=='S'))
			ch=x;
		if(ch=='Q') {
			inp(p);
			printf("%lld\n",query(p));
			}
		else {
			inp(a);inp(b);inp(k);
			update(a,k);
			update(b+1,-k);
			}
		}
	return 0;
	}
