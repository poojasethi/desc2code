#include<iostream>
#include<stdio.h>
using namespace std;
#define ll long long
#define MAX 1000006

ll fw[MAX];

ll read(int p)
{
    ll sum1=0;
    while(p>0)
    {
        sum1 +=fw[p];
        p -= p&-p;
    }

    return sum1;

}

void update(int n, int idx, int v)
{
    while(idx<=n)
    {
        fw[idx] +=v;
        idx += idx & -idx;
    }
}

int main() {

    int n, m, cc;
    cin >> n >> m >> cc;

    for(int i=0;i<n;i++)
    {
        update(n, i+1, cc);
        update(n, i+1+1, -cc);
    }

    while(m--)
    {
        char c;
        scanf(" %c", &c);
        //cout << "Char is: " << c << endl;
        if(c=='Q')
        {
            //cout << "Q selected\n";
            int p;
            scanf("%d", &p);
            printf("%lld\n", read(p));// read(p) << endl;
        }
        else
        {
            //cout << "Q NOT selected\n";
            int x, y;
            ll v;
            scanf("%d %d %lld", &x, &y, &v);
            x--, y--;
            update(n, x+1, v);
            update(n, y+1+1, -v);
        }
    }


    return 0;
}
