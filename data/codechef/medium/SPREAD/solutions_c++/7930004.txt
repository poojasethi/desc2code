#include <bits/stdc++.h>
using namespace std;
long long int binary[1000001];
void construct(long long int binary[],long long int val,long long int index,long long int l)
{
    while(index<=l)
    {
        binary[index]+=val;
        index=index+(index&-index);
    }
}
long long int getsum(long long int binary[],long long int index)
{
    //index++;
    long long int s=0;
    while(index!=0)
    {
        s=s+binary[index];
        index=index-(index&-index);
    }
    return s;
}
int main()
{
    long long int n,m,c,i,j;
    char a[2];
    scanf("%lld%lld%lld",&n,&m,&c);
    construct(binary,c,1,n);
    for(j=0;j<m;j++)
    {
        scanf("%s",&a);
        if(a[0]=='Q')
        {
           long long int p;
            scanf("%lld",&p);
            printf("%lld\n",getsum(binary,p));
        }
        else if(a[0]=='S')
        {
            long long int u,v,k;
            scanf("%lld%lld%lld",&u,&v,&k);
            construct(binary,k,u,n);
            construct(binary,-k,v+1,n);
        }
    }
    return 0;
}
