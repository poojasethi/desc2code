#include<bits/stdc++.h>
using namespace std;
#define MAX 1000009
long long int BIT[MAX];
long long int n;
inline void update(long long int i,long long int k)
{
    for(;i<=n;i+=(i&-i))
        BIT[i]+=k;
}
inline void update(long long int u,long long int v,long long int k)
{
    update(u,k);
    update(v+1,-k);
}
inline long long int query(long long int i)
{
    long long int sum=0;
    for(;i>0;i-=(i&-i))
        sum+=BIT[i];
    return sum;
}
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    long long int m,c,k,p,u,v;
    char type;
    cin>>n>>m>>c;
    while(m--)
    {
        cin>>type;
        if(type=='S')
        {
            cin>>u>>v>>k;
            update(u,v,k);
        }
        else
        {
            cin>>p;
            cout<<c+query(p)<<"\n";
        }
    }
    return 0;
}
