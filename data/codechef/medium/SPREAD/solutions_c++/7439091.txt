#include <cstdlib>
#include <iostream>
#include <cstdio>

using namespace std;
long long int bit[1000001] = {0};
long long int sum(long long int index)
{
     long long int ans = 0;
     while(index > 0){
           ans += bit[index-1];
           index = index - (index&-index);
     }
     return ans;      
 }
 
void update(long long int index,long long int value,long long int size)
{
     while(index <= size){
           bit[index-1]+=value;
           index = index+(index&-index);
     }      
 } 
 
void update2(long long int l,long long int r,long long int value,long long int size)
{
     update(l,value,size);
     update(r+1,-value,size);
 } 
int main()
{
    long long int n,m,c,l,r,k,p;
    scanf("%lld%lld%lld",&n,&m,&c);
   // cout << n << ' ' << m << ' ' <<  c << '\n'; 
    update2(1,n,c,n);
    int choice;
    while(m--){
        choice = getchar();
        while(choice != (int)'S' && choice != (int)'Q')
            choice = getchar();
          if(choice == (int)'S'){
             scanf("%lld%lld%lld",&l,&r,&k);        
             update2(l,r,k,n);
          }
          else if(choice == (int)'Q'){
             scanf("%lld",&p);
             printf("%lld\n",sum(p));
          }
    }               
    
    
    return 0;
}
