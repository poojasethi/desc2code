ll BIT[1000005];
int n;
void update(int idx, ll val)
{
    while(idx<=n)
    {
        BIT[idx]+=val;
        idx+=(idx&-idx);
    }
}
ll query(int idx)
{
    ll sm=0;
    while(idx>0)
    {
        sm+=BIT[idx];
        idx-=(idx&-idx);
    }
    return sm;
}
