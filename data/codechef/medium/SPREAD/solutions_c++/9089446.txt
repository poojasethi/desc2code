#include <cstdio>
#include <vector>
using namespace std;
int maxvalue;
vector<long long > bit;
void update(int u,long long k){

	int idx = u;
	while(idx<=maxvalue){
		bit[idx] +=k;
		idx+=(idx&-idx);
	}

}

long long read(int q){
	long long answer =0;
	int idx = q;
	while(idx!=0){

		answer+=bit[idx];
		idx-=(idx & -idx);
	}
	return answer;
}


int main(){

	int n,m;
	long long c;
	scanf("%d %d %lld",&n,&m,&c);
	maxvalue=n;
	bit.resize(n+1,0);
	update(1,c);

	char a;
	
	while(m--){
		scanf(" %c",&a);
		if(a=='S'){
			int u,v;
			long long k;
			scanf("%d %d %lld",&u,&v,&k);

			update(u,k);
			update(v+1,-k);
		}
		else{
			int u;
			scanf("%d",&u);
			printf("%lld\n",read(u));
		}
	}

return 0;



}
