#include<string.h>
#include<stdio.h>
#include<iostream>
using namespace std;
main()
{
	long long int sum,n,x,m,l,u,v,c,k,p,i,j;
	char ch;
	scanf("%lld",&n);
	scanf("%lld",&m);
	long long int tree[n+1];
	for(i=0;i<=n;i++)
		tree[i]=0;
	scanf("%lld",&c); 
	while(m--)
	{
		ch=getchar_unlocked();
		while(ch!='Q' && ch!='S')
			ch=getchar_unlocked();
		if(ch=='S')
		{
			scanf("%lld",&u);
			scanf("%lld",&v);
			scanf("%lld",&k);
			l=u;
			x=k;
			while(l<=n)
			{
				tree[l]=tree[l]+x;
				l=l+(l&(-l));
			}
			l=v+1;
			x=-k;
			while(l<=n)
			{
				tree[l]=tree[l]+x;
				l=l+(l&(-l));
			}
		}
		if(ch=='Q')
		{
			scanf("%lld",&p);
			l=p;sum=0;
			while(l>0)
			{
				sum+=tree[l];
				l=l-(l&(-l));
			}
			printf("%lld\n",sum+c);
		}
	}
}
