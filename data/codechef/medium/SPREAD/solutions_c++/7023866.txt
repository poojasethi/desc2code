#include <iostream>
#include<stdio.h>
using namespace std;
typedef long long int ll;
long long int bitree[1000001]={0};
ll getsum(ll index,ll n)
{
 ll sum=0;
	while(index>0)
 {
  sum+=bitree[index];
  index-=index&(-index);
}
return sum;
}
void update(ll index,ll n,ll val)
{
 while(index<=n)
{
 bitree[index]+=val;
 index+=(index)&(-index);
}
}
int main() {
ll n,m,c,p,l,r,num;
char ch;
scanf("%lld %lld %lld",&n,&m,&c);
update(1,n,c);
//cout<<n<<m<<c<<endl;
while(m--)
{
 ch=getchar_unlocked();
while((ch!='Q')&&(ch!='S'))ch=getchar_unlocked(); 
if(ch=='Q')
  {
   scanf("%lld",&p);
  printf("%lld\n",getsum(p,n));
  }
else
 {
  scanf("%lld %lld %lld",&l,&r,&num);
  update(l,n,num);
  update(r+1,n,(-1)*num);
  }
}	
	return 0;
}
