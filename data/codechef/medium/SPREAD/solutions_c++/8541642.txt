bool cmp(int a, int b){ return a>b; }
template<class T> T gcd(T a, T b) { return b ? gcd(b, a % b) : a; }
template<class T> T lcm(T a, T b) { return a * b / gcd(a, b); }

LL n, m, c, BIT[MAXX];

void upd(int idx, LL val){
	while(idx <= n){
		BIT[idx] += val;
		idx += (idx & -idx);
	}
}

LL query(int idx){
	LL sum = 0;
	while(idx > 0){
		sum += BIT[idx];
		idx -= (idx & -idx);
	}
	return sum;
}

			scanf("%d", &a);
			printf("%lld\n", (query(a) + c));
		}else{
			scanf("%d %d %lld", &a, &b, &v);
			upd(a, v);
			upd(b+1, -v);
		}
	}
return 0;
}
