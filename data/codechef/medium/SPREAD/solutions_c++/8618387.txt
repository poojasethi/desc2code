#include<bits/stdc++.h>
using namespace std;
long long BIT[1000005];
long long n,q;
void update(long long ind,long long val)
{
	while(ind<=n)
	{
		BIT[ind]+=val;
		ind+=(ind&(-ind));
	}
}
long long query(long long ind)
{
	long long sum=0;
	while(ind>0)
	{
		sum+=BIT[ind];
		ind-=(ind&(-ind));
	}
	return sum;
}
int main()
{
	long long c;
	scanf("%lld %lld %lld",&n,&q,&c);
	long long i,j;
	update(1,c);
	update(n+1,-c);
	char a[3];long long x,y,z;
	while(q--)
	{
		scanf("%s",a);
		if(a[0]=='S')
		{
			scanf("%lld %lld %lld",&x,&y,&z);
			update(x,z);
			update(y+1,-z);
		}
		else
		{
			scanf("%lld",&x);
			printf("%lld\n",query(x));
		}
	}
	return 0;		
}
