LL tree[1<<20];

void add( int idx, int val )
{
	for( ; idx < (1<<20) ; idx += ( idx&-idx ) )
		tree[idx] += val;
}
LL read( int idx )
{
	LL ret = 0;
	for( ; idx ; idx -= ( idx&-idx ) )
		ret += tree[idx];
	return ret;
}

int main()
{
	int t, n, m, C, a, b, c;
	char ch;
	scanf( "%d%d%d\n", &n, &m, &C );
	while( m -- )
	{
		scanf( " %c", &ch );
		if( ch == 'S' )
		{
			scanf( "%d%d%d", &a, &b, &c );
			add( a, c );
			add( b+1, -c );
		}
		else
		{
			scanf( "%d", &a );
			printf( "%lld\n", C + read( a ) );
		}
	}
	return 0;
}
