#include<bits/stdc++.h>
#define ll long long
#define s(a) scanf("%lld", &a)
#define pr(a) printf("%lld\n", a)
using namespace std;

ll bit[1000005], n, c;

void update(ll ind, ll val)
{
	while(ind<=n)
	{
		bit[ind] += val;
		ind += (ind & -ind);
	}
}
ll query(ll ind)
{
	ll ans=0;
	while(ind>0)
	{
		ans += bit[ind];
		ind -= (ind & -ind);
	}
	return ans+c;
}
int main()
{
	ll m;
	s(n); s(m); s(c);
	while(m--)
	{
		char ch;
		ch=getchar();
		while(ch!='Q' && ch!='S')
			ch=getchar();
		if(ch=='S')
		{
			ll u, v, k;
			s(u); s(v); s(k);
			update(u, k);
			update(v+1, -1*k);
		}
		else
		{
			ll p;
			s(p);
			pr(query(p));
		}
	}
    return 0;
}
