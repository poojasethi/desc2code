 #include<stdio.h>
    #include<string.h>
   
    #define ll int
    
    #define getcx getchar_unlocked
    long long  bit[1000200]={0};
    inline void rl(int *a)
    {
      register char c=0;
      while (c<33) c=getchar_unlocked();
      *a=0;
      while (c>33)
        {
          *a=*a*10+c-'0';
          c=getchar_unlocked();
        }
    }
    void update(ll i,ll x,ll n)
    {
    	while(i<=n)
    	{		
    		bit[i]+=x;
    		i+=(i&-i);
    	}
    }
    long long int query(ll ind)
    {
    	long long sum=0;
    	while(ind>0)
    	{
    		sum+=bit[ind];
    		ind-=(ind&(-ind));
    	}
    	return sum;
    }
    int main()
    {
    	char ch;
    	ll n,m,c,p,u,v,k;
    	rl(&n),rl(&m),rl(&c);
    	update(1,c,n);
    	while(m--)
    	{
    		ch=getcx();
    		while(ch!='Q' && ch!='S')
    		ch=getcx();
    		if(ch=='Q')
    		{
    			rl(&p);
    			printf("%lld\n",query(p));
    		}
    		else
    		{
    			rl(&u),rl(&v),rl(&k);
    			update(u,k,n),update(v+1,-k,n);
    		}
    	}
    }
