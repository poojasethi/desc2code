#include<iostream>
#include<vector>
#include<cstdio>
#define ll long long
using namespace std;
void update(ll idx,ll val,vector<ll>& v,ll n)
{
	while(idx<=n)
	{
		v[idx]+=val;
		idx+=((idx)&(-idx));
	}
}
ll getq(ll idx,vector<ll>& v)
{
	ll sum=0;
	while(idx>0)
	{
		sum+=v[idx];
		idx-=((idx)&(-idx));
	}
	return sum;
}
int main()
{
	ll i,n,m,c,x,y,val;
	char ch;
	scanf("%lld%lld%lld",&n,&m,&c);
	vector<ll> v(n+1);
	update(1,c,v,n);
	while(m--)
	{
		scanf("%c",&ch);
		while(!((ch=='Q')||(ch=='S')))
			scanf("%c",&ch);
			if(ch=='S')
			{
				scanf("%lld%lld%lld",&x,&y,&val);
				update(x,val,v,n);
				update(y+1,-val,v,n);
			}
			else
			{
				scanf("%lld",&x);
				printf("%lld\n",getq(x,v));
			}
	}
	return 0;
}
