#include<bits/stdc++.h>
using namespace std;
#define maxi 1000010
typedef long long ll;

ll bit[maxi];
int n;

void update(int i,ll val)
{
	i+=1;
	while(i<=n)
	{
		bit[i]+=val;
		i+=(i&(-i));
	}
}

ll query(int i)
{
	ll sum=0;
	i+=1;
	while(i>0)
	{
		sum+=bit[i];
		i-=(i&(-i));
	}
	return sum;
}

int main()
{
	int m;
	ll c;
	// cin>>n>>m>>c;
	scanf("%d %d %lld",&n,&m,&c);
	for(int i=0;i<m;i++)
	{
		char a[10];
		scanf("%s",a);
		if(a[0]=='Q')
		{
			int i;
			scanf("%d",&i);
			// cout<<query(i-1)+c<<"\n";
			printf("%lld\n",query(i-1)+c);
		}
		else
		{
			int l,r;
			ll val;
			// cin>>l>>r>>val;
			scanf("%d %d %lld",&l,&r,&val);
			update(l-1,val);
			update(r,-val);
		}
	}
	return 0;
}
