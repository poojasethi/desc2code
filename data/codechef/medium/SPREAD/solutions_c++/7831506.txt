#include<bits/stdc++.h>
using namespace std;
#define lld long long int
#define MAX 1000001
lld BIT[MAX];
#define gc getchar_unlocked
inline int read()
{
	int x;
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
	return x;
}

void update(int x,lld val,int n)
{
   while(x<=n)
   {
   	BIT[x]+=val;
   	x+=(x&(-x));
   }
}

lld query(int x,int n)
{
 lld ans=0;
 while(x>0)
 {
   ans+=BIT[x];
   x=x-(x&(-x));
 }
 return ans;

}

int main()
{
	int ar[MAX],n,i;
	int m,c;
    //scanf("%d %d %d",&n,&m,&c);
    n=read();m=read();c=read();
    for(i=1;i<=n;i++)
    	ar[i]=c;
    update(1,c,n);

    //Scout<<endl;
    while(m--)
    {
    	char s;
    	int a,b;
    	lld val;
    	scanf(" %c",&s);
    	if(s=='S')
    	{
          //scanf("%d %d %lld",&a,&b,&val);
          a=read();b=read();
          scanf("%lld",&val);
          update(a,val,n);
          update(b+1,-val,n);
    	}
    	else
    	{
         
         //scanf("%d",&a);
         a=read();
         printf("%lld\n",query(a,n));
    	}
    }
	return 0;
}
