#include <bits/stdc++.h>
#define ll long long
using namespace std;

const ll maxn = 1000010;
ll n;
ll tree[maxn] = {0};

void update(ll, ll);
ll query(ll);

int main() {
	ios::sync_with_stdio(false);
	cin.tie(0);
	
	char id;
	ll p, u, v, k, m, c;
	cin >> n >> m >> c;
	
	while(m--) {
		cin >> id;
		if (id == 'S') {
			cin >> u >> v >> k;
			update(u, k);
			update(v + 1, -k);
		}
		else {
			cin >> p;
			cout << c + query(p) << '\n';
		}
	}
	return 0;
}

void update(ll idx, ll val) {
	while(idx <= n) {
		tree[idx] += val;
		idx += (idx & -idx);
	}
}

ll query(ll idx) {
	ll sum = 0;
	while(idx > 0) {
		sum += tree[idx];
		idx -= (idx & -idx);
	}
	return sum;
}
