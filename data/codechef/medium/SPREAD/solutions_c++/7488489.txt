#include<string.h>
#include<stdio.h>
#include<iostream>
#define gc getchar_unlocked
void scanint(int &x)
{
	register int c=gc();
	x=0;
	for(;(c<48 || c>57);c=gc());
	for(;c>47 && c<58;c=gc()){x=(x<<1)+(x<<3)+c-48;}
}
using namespace std;
main()
{
	long long int sum,x,l,i,j;
	int n,m,u,v,k,c,p;
	char ch;
	scanint(n);
	scanint(m);
	long long int tree[n+1];
	for(i=0;i<=n;i++)
		tree[i]=0;
	scanint(c); 
	while(m--)
	{
		ch=getchar_unlocked();
		while(ch!='Q' && ch!='S')
			ch=getchar_unlocked();
		if(ch=='S')
		{
			scanint(u);
			scanint(v);
			scanint(k);
			l=u;
			x=k;
			while(l<=n)
			{
				tree[l]=tree[l]+x;
				l=l+(l&(-l));
			}
			l=v+1;
			x=-k;
			while(l<=n)
			{
				tree[l]=tree[l]+x;
				l=l+(l&(-l));
			}
		}
		if(ch=='Q')
		{
			scanint(p);
			l=p;sum=0;
			while(l>0)
			{
				sum+=tree[l];
				l=l-(l&(-l));
			}
			printf("%lld\n",sum+c);
		}
	}
}
