#include <iostream>

using namespace std;

int main()
{
    int t=1,g=0,i,n,q,flipped;
    cin>>t;
    for(int k=1;k<=t;k++)
    {
            cin>>g;
        for(int j=1;j<=g;j++)
        {
            cin>>i>>n>>q;
            if(n%2!=0)
                flipped=((n-1)/2)+1;

            else
                flipped=((n-2)/2)+1;

            if(i==q)
                cout<<n-flipped<<"\n";
            else
                cout<<flipped<<"\n";

        }
    }
    return 0;
}
