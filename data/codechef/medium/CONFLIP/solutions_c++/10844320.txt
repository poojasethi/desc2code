#include<iostream>
#include<vector>
using namespace std;
int main()
{
    int t,g,n;
    short i,q;
    cin>>t;
    while(t--)
    {
        cin>>g;
        while(g--)
        {
           int cou;
           cin>>i>>n>>q;
           if(n%2==0)
            cou=n/2;
           else{
            if(i==q)
                cou=n/2;
            else cou=n/2+1;
           }
            cout<<cou<<endl;
        }
    }
}
