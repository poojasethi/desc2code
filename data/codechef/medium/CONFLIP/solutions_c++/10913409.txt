#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,g,n,i,q;
    cin>>t;
    while(t--)
    {
        cin>>g;
        while(g--)
        {
            cin>>i>>n>>q;
            if(n%2==0)
            cout<<n/2<<endl;
            else if((i+q)%2==0)
            cout<<(n-1)/2<<endl;
            else cout<<(n+1)/2<<endl;
        }
    }
    return 0;
}
