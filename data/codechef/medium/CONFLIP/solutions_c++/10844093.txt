#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%i",&t);
    while(t--)
    {
        int g;
        scanf("%i",&g);
        while(g--)
        {
            int i,n,q,c;
            scanf("%i%i%i",&i,&n,&q);
            c=n/2;
            if(n%2==1)
            {
                if(i+1==q||i+1==q+2)
                    c++;
            }
            printf("%i\n",c);
        }
    }
	return 0;
}
