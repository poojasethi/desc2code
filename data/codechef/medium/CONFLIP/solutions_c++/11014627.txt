#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--)
	{
		int g,j;
		cin>>g;
		//long int i[g+1],n[g+1],q[g+1];
		for(j=1;j<g+1;j++)
		{
			long int i,n,q;
			cin>>i>>n>>q;
			if(n%2==0)
			{
				cout<<n/2<<endl;
			}
			else
			{
				if(i==1)
				{
					if(q==1)
						cout<<n/2<<endl;
					else if(q==2)
						cout<<(n+1)/2<<endl;
				}
				else if(i==2)
				{
					if(q==2)
						cout<<n/2<<endl;
					else if(q==1)
						cout<<(n+1)/2<<endl;
				}
				
			}
			
		}
		
	}
	return 0;
}
