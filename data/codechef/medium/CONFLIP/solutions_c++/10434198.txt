#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,g,q,t;
	long int n;
	cin>>t;
	while(t--)
	{
		cin>>g;
		while(g--)
		{
			cin>>i>>n>>q;
			if(n%2!=0 && i==1 && q==1)
			cout<<n/2<<endl;
			else if(n%2!=0 && i==1 && q==2)
			cout<<(n/2+1)<<endl;
			else if(n%2!=0 && i==2 && q==1)
			cout<<(n/2+1)<<endl;
			else if(n%2!=0 && i==2 && q==2)
			cout<<n/2<<endl;
			else 
			cout<<n/2<<endl;
		}
	}
	return 0;
}
