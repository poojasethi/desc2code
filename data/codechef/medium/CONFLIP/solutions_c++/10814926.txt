#include <iostream>
using namespace std;
int main()
{ long long int t,a,n,s,b,c,res1,res2;
    cin>>t;
    while(t--)
    {
        cin>>n;
        while(n--)
        {
            cin>>a>>b>>c;
            if(b%2==0)
            {res1=b/2;res2=b/2;}
            else if(b%2!=0)
            {
                if(a==1)
                {
                    res2=b-b/2;
                    res1=b-res2;
                }
              else if(a==2)
              {
                  res1=b-b/2;
                  res2=b-res1;
              }
            }
        if(c==1)
            cout<<res1<<endl;
        else if(c==2)
            cout<<res2<<endl;
        }
    }
return 0;
}
