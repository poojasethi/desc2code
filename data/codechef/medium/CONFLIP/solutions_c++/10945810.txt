#include<iostream>
using namespace std;
int main()
{
	long t,g,i,n,q; cin>>t;
	while(t--)
	{
		cin>>g;
		while(g--)
		{
			cin>>i>>n>>q;
			if(i==1)
			{
				if(q==1) cout<<n/2<<endl;
				if(q==2) cout<<n-n/2<<endl;
			}
			if(i==2)
			{
				if(q==1) cout<<n-n/2<<endl;
				if(q==2) cout<<n/2<<endl;
			}
		}
	}
}
