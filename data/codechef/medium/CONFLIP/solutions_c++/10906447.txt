#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    long int t,g,i,n,q;
    scanf("%ld",&t);
    while(t--)
    {
        scanf("%ld",&g);
        while(g--)
        {
            scanf("%ld%ld%ld",&i,&n,&q);
            if(i==q)
            {
                printf("%ld\n",n/2);
            }
            else
                printf("%ld\n",n%2==0?n/2:(n/2+1));
        }
    }
    return 0;
}
