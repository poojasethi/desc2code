#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int t,g,i,q;
    long long int n,h=0,t1=0;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&g);
        while(g--)
        {
            scanf("%d%lld%d",&i,&n,&q);
            if(n%2==0)
            {
                h=t1=n/2;
            }
            else
            {
                if(i==1)
                {
                    h=n/2;
                    t1=h+1;
                }
                else
                {
                    t1=n/2;
                    h=t1+1;
                }
            }
            if(q==1)
            {
                printf("%lld\n",h);
            }
            else
            {
                printf("%lld\n",t1);
            }
        }
    }
    return 0;
}
