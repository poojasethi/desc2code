#include<iostream>

using namespace std;

int main(){
	int testCase ,q,i,gp;
	long long n,x;
	cin>>testCase;
	for(int w=0;w<testCase;w++){
		cin>>gp;
		for(int j=0;j<gp;j++){
			cin>>i>>n>>q;
			x=n%2;
			if(x==0){
				cout<<n/2<<endl;
			}
			else if(q==i){
				cout<<n/2<<endl;
			}
			else if(q!=i){
				cout<<n/2+1<<endl;
			}
			}
	}
}
