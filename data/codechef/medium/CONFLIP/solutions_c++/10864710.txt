#include<cstdio>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int G;
        scanf("%d",&G);
        while(G--){
            int I,N,Q;
            scanf("%d %d %d",&I,&N,&Q);
            int first=N/2;
            int second=N-(N/2);
            if(Q==I){
                printf("%d\n",first);
            }
            else
            {
                printf("%d\n",second);
            }
        }
    }
}
