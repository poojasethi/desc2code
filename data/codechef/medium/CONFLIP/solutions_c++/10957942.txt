#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main(){
	long int t,g,ii,n,q;
	cin>>t;
	for(long int i=0;i<t;i++){
		cin>>g;
		for(long int j=0;j<g;j++){
			cin>>ii>>n>>q;
			if(n%2==1){
				if(ii==1){
					long int a2 = n/2+1;
					long int a1 = n-a2;
					if(q==1){
						cout<<a1<<"\n";
					}
					else{
						cout<<a2<<"\n";
					}
				}
				else{
					long int a1 = n/2+1;
					long int a2 = n-a1;
					if(q==1){
						cout<<a1<<"\n";
					}
					else{
						cout<<a2<<"\n";
					}
				}
			}
			else{
				cout<<n/2<<"\n";
			}
		}
	}
	return 0;
}
