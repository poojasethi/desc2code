#include <iostream>
using namespace std;

int main(){
	long long int t,g,i,n,q,j,k,l,ans;
	cin >> t;
	while(t > 0){
		t--;
		cin >> g;
		while(g > 0){
			g--;
			cin >> i >> n >> q;
			if(i == q)
				ans = n/2;
			else{
				if(n%2 == 0)
					ans = n/2;
				else
					ans = n/2 + 1;
			}
			cout << ans << endl;
		}
	}
	return 0;
}
