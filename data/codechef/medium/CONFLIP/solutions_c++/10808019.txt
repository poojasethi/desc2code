#include <iostream>
using namespace std;

int Constraints(int, int);

int main()
{
	int T, G, I, N, Q;

	T = Constraints(1, 10);




	for (int i = 0; i < T; i++)
	{

		G = Constraints(1, 20000);


		int *II = new int[G];
		int *NN = new int[G];
		int *QQ = new int[G];
		//int *AA = new int[G];

		for (int j = 0; j < G; j++)
		{
			I = Constraints(1, 2);
			N = Constraints(1, 1E9);
			Q = Constraints(1, 2);

			II[j] = I;
			NN[j] = N;
			QQ[j] = Q;
		}



		for (int j = 0; j < G; j++)
		{
			int A, B;

			if (II[j] == 1)
			{
				A = B = NN[j] / 2;
				if (NN[j] & 1 == 1) A++;
			}
			else if (II[j] == 2)
			{
				A = B = NN[j] / 2;
				if (NN[j] & 1 == 1) B++;
			}

			int C = (QQ[j] == 1) ? B : A;

			cout << C << endl;
		}


	}



	cin.get();
	cin.get();
	return 0;
}


int Constraints(int a, int b)
{
	int Inp;
	do
	{
		cin >> Inp;
	} while (Inp < a || Inp > b);

	return Inp;
}
