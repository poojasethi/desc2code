#include<stdio.h>
using namespace std;
int main()
{
	int t,g,i,n,head,tail,q;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&g);
		while(g--)
		{
			scanf("%d%d%d",&i,&n,&q);
			head=n/2;
			tail=n/2;
			if(n%2!=0)
			{
				if(i==2)	head+=1;
				else	tail+=1;
			}
			
			if(q==1)	printf("%d\n",head);
			else	printf("%d\n",tail);
		}
	
	}
	return 0;
}
