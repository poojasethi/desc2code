N = int(raw_input())

for i in xrange(N):
    g = int(raw_input())
    for j in xrange(g):
        l, n, q = map(int, raw_input().split())
    
        if n & 1 :
            if q == l:
                print n / 2 
            else:
                print (n / 2) + 1
        else:
            print n / 2
