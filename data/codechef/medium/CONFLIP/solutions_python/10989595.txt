t = int(raw_input())
while t > 0 :
    t -= 1
    g = int(raw_input())
    while g > 0:
        g-=1
        l,n,q = map(int,raw_input().split())
        if n % 2 == 0 :
            if l==q :
                print n/2
            else :
                print n/2
        else :
            if l==q :
                print n/2
            else :
                print n/2+1
