#include <iostream>
using namespace std;
int main() 
{
	int t, m, w, ww, mm;
	char cm[25000],cw[25000], temp;
	cin>>t;
	getchar_unlocked () ;
	while (t--) 
	{
		m=0;w=0;
		mm=0;ww=0;
		while (1)
		{
			temp=getchar_unlocked() ;
if (temp>='a'&&temp<='z')
			cm[m++]=temp;
			else break;
		}
		while (1)
		{
			temp=getchar_unlocked() ;
if (temp>='a'&&temp<='z')
			cw[w++]=temp;
			else break;
		}
		ww=w-1;
		mm=m-1;
		if (w<=m)
		while (m--){
		if (cm[m]==cw[ww]) ww--;
		if (ww==-1) break;
		} 
		else
		while (w--){
		if (cw[w]==cm[mm]) mm--;
		if (mm==-1) break;} 
		if (mm==-1||ww==-1)
		cout<<"YES\n";
		else 	cout <<"NO\n";
	}
}
