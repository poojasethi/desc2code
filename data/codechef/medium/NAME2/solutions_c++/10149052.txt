#include <bits/stdc++.h>
using namespace std;
int isSS(char str1[], char str2[], int m, int n){
	if(m==0) return 1;
	if(n==0) return 0;
	if(str1[m-1]==str2[n-1]){
		return isSS(str1,str2,m-1,n-1);
	}
	else return isSS(str1,str2,m, n-1);
}
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		int x;
		char str1[25005];
		char str2[25005];
		scanf("%s %s", str1,str2);
		int len1= strlen(str1);
		int len2= strlen(str2);
		if(len1<=len2){
			
			 x = isSS(str1,str2,len1,len2);
			 if(x == 1) printf("YES\n");
			 else if(x==0) printf("NO\n");
		}
		else {
			x = isSS(str2,str1,len2,len1);
			if(x == 1) printf("YES\n");
			else if(x==0) printf("NO\n");
		}
	}
	return 0;
}
