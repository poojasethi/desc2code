#include <iostream>
#include <string>

using namespace std;

bool subseq(string big, string small, int big_len, int small_len)
{
	int i = big_len-1, j = small_len-1;
	
	while(j >= 0 && i >= 0)
	{
		if(big[i] == small[j])
		{
			--j;
		}
		
		--i;
	}
	
	if(j < 0)
		return true;
	
	return false;
}

int main() {
	int t, len1, len2;
	bool can_marry = false;
	
	cin>>t;
	
	while(t--)
	{
		string str1, str2;
		cin>>str1>>str2;
		
		len1 = str1.length();
		len2 = str2.length();
		
		if(len1 >= len2)
			can_marry = subseq(str1, str2, len1, len2);
		else
			can_marry = subseq(str2, str1, len2, len1);
			
		if(can_marry)
			cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	return 0;
}
