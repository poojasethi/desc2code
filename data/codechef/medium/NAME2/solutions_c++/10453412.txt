int main()
{

        int ls=strlen(s);
        int lt=strlen(t);
        int j=0;
        if(ls<=lt)
        {
            for(int i=0;t[i]!='\0';i++)
            {
                if(s[j]==t[i])
                {
                    j++;
                }
            }

            if(s[j]=='\0')
                cout<<"YES"<<endl;
            else
                cout<<"NO"<<endl;
        }
        else
        {
            for(int i=0;s[i]!='\0';i++)
            {
                if(t[j]==s[i])
                    j++;
            }

        }

    }

}
