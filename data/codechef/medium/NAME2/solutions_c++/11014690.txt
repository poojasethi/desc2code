#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s[25001],p[25001];
		cin>>s>>p;
		int i;
		int sl =strlen(s);
		int pl= strlen(p);
		int count=0,k=0,j;
		if(sl<=pl)
		{
			for(i=0;i<sl;i++)
			{
				for(j=k;j<pl;j++)
				{
					if(s[i]==p[j])
					{
						count++;
						k=j+1;
						break;
					}
				}
			}
			if(count==sl)
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
			//cout<<count<<endl;
		}else
		{
			for(i=0;i<pl;i++)
			{
				for(j=k;j<sl;j++)
				{
					if(p[i]==s[j])
					{
						count++;
						k=j+1;
						break;
					}
				}
			}
			if(count==pl)
			cout<<"YES"<<endl;
			else
			cout<<"NO"<<endl;
			//cout<<count<<endl;
		}
	}
}
