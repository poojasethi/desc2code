#include<iostream>
#include<string.h>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        char m[25001],w[25001];
        int i=0,j=0,c=0;
        cin>>m>>w;
        while(m[i]!=0&&w[j]!=0){
            if (m[i]==w[j]){
                i++;
                j++;
                c++;
            }
            else{
                j++;
            }
        }
        int c2;
        i=0,j=0,c2=0;
        while(m[i]!=0&&w[j]!=0){
            if (m[i]==w[j]){
                i++;
                j++;
                c2++;
            }
            else{
                i++;
            }
        }
        int l=0,l2=0;
        for (i=0;m[i]!=0;i++){
            l++;
        }
        for (i=0;w[i]!=0;i++){
            l2++;
        }
        if (c==l||c2==l2)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
