#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string m,w;
        cin>>m>>w;
        int ml = m.length();
        int wl = w.length();
        int i,j;
        if(ml<wl)
        {
        for(i=0,j=0;i<ml && j<wl;)
        {
            if(m[i]==w[j])
                i++;
            j++;
        }
        if(i==ml)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
        }
        else
        {
        for(i=0,j=0;i<wl && j<ml;)
        {
            if(w[i]==m[j])
                i++;
            j++;
        }
        if(i==wl)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
        }
        }
    }
