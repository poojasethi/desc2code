#include<string>

void check(char *a,char *b)
{
	int i=0,j=0,p;
	while(a[i]!='\0')
	{
		if(a[i]==b[j])
		{
			i++;
			j++;
		}
		else
		j++;
		if(b[j]=='\0'&&a[i]!='\0')
		{
			p=0;
			break;
		}
		if(b[j]=='\0'&&a[i]=='\0')
		p=1;
			if(b[j]!='\0'&&a[i]=='\0')
			p=1;
				
	}
	if(p==1)
			cout<<"YES"<<endl;
			if(p==0)
			cout<<"NO"<<endl;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s1,s2;
		cin>>s1>>s2;
		int p,q;
		p=s1.length();
		q=s2.length();
		if(p<q)
		{
			check(&s1[0],&s2[0]);
		}
		else
		{
			check(&s2[0],&s1[0]);
		}	
	}

}
