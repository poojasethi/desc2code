#include <iostream>
#include <string>

using namespace std;
//is A Subsequence of B
int isSubSequence(string A, string B){
	int sizeA=A.size();
	int sizeB=B.size();
	int i=0,j=0;
	while(i<sizeA && j<sizeB){
		if(A[i]==B[j])
			i++;
		j++;
	}
	if(i==sizeA)
		return 1;
	return 0;
}

int main(){
	int T;
	cin >> T;
	string M,W, ans;
	for(int t=0;t<T;t++){
		cin >> M >> W;
		ans = isSubSequence(M,W) || isSubSequence(W,M) ? "YES" : "NO";
		cout << ans << endl;
	}
}
