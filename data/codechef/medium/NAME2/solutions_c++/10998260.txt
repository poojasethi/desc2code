#include<stdio.h>
#include<string.h>
#include<bits/stdc++.h>

bool issubsequence(char b[], char a[], int lb, int la){
	if(lb == 0)	return true;
	if(la == 0)	return false;
	//If last characters of two strings are matching
	if(b[lb - 1] == a[la - 1])	return issubsequence(b, a, lb - 1, la - 1);
	//If last characters are not matching
	return issubsequence(b, a, lb, la - 1);
}

int main(){
	int t, la, lb; scanf("%d", &t); char a[30000], b[30000];
	while(t-- && scanf("%s%s", a, b)){
		la = strlen(a), lb = strlen(b);
		if(issubsequence(a, b, la, lb) == true || issubsequence(b, a, lb, la) == true)	printf("YES\n");
			else	printf("NO\n");
	}
	return 0;
}
