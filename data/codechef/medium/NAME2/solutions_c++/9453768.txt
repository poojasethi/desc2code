#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t;
	string s1,s2;
	cin>>t;
	while(t-->0)
	{
		int i=0,j=0;
		cin>>s1>>s2;
		if(s1.length()>s2.length())
		s1.swap(s2);
		while(s1[i]!='\0')
		{
			while(s2[j]!=s1[i]&&s2[j]!='\0')
			j++;
			if(s2[j]=='\0')
			break;
			i++;
			j++;
		}
		if(s1[i]=='\0')
		cout<<"YES\n";
		else cout<<"NO\n";
	}
}
