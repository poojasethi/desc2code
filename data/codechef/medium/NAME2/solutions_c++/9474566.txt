#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

int main()
{
    int t,i;
    cin>>t;
    char w[25010],m[25010];
    while(t--){
        cin>>w>>m;
        int flag=0;
        int k=strlen(w);
        int l=strlen(m);
        if(k>l){
            int j=0;
            for(i=0;w[i]!='\0'&&j<l;i++){
                if(w[i]==m[j]) j++;
            }
            //cout<<j<<"\n";
            if(j==l) flag=1;
        }else if(l>k){
            int j=0;
            for(i=0;m[i]!='\0'&&j<k;i++){
                if(m[i]==w[j]) j++;
            }
            //cout<<j<<"\n";

            if(j==k) flag=1;
        }else{
            if(!strcmp(w,m)) flag=1;
        }
        if(flag==1) cout<<"YES\n";
        else cout<<"NO\n";
    }

    return 0;
}
