#include<cstdio>
#include<cstring>
using namespace std;
bool is_LCS(char A[],char B[]){
    int n1=strlen(A);
    int n2=strlen(B);
    int i=0,j=0;
    //finding A in B;
    int fail=0;
    while(i<n1){
        char c=A[i];
        bool found=false;
        while(j<n2){
            if(B[j]==c){
                found=true;
            }
            j++;
            if(found){
                break;
            }
        }
        if(found){
            i++;
        }
        else
        {
            fail=1;
            break;
        }
    }
    return !fail;
}
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        char A[25005],B[25005];
        scanf("%s %s",&A,&B);
        if(is_LCS(A,B) || is_LCS(B,A)){
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }
}
