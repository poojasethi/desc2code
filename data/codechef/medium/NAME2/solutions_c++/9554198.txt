#include <iostream>
#include <string.h>

using namespace std;

int main()
{
string s1, s2;
int t, i, j, k, l1, l2, flag;
cin>>t;
int ans[t];
for(i=0;i<t;i++)
{
    cin>>s1>>s2;
    l1=s1.length(); l2=s2.length();
    if(l1>l2) {swap(s1,s2); swap(l1,l2);}
    k=0; flag=0;
    for(j=0;j<l2;j++)
    {

        if(s1[k]==s2[j]) {flag++; k++;}
        if(flag==l1) break;
    }
    if(flag==l1) ans[i]=1; else ans[i]=0;
}
for(i=0;i<t;i++)
{
    if(ans[i]) cout<<"YES\n"; else cout<<"NO\n";
}

return 0;
}
