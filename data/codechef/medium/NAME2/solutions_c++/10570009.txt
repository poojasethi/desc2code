using namespace std;
bool isSubSequence(char str1[], char str2[], int m, int n)
{
   int j = 0; 
   for (int i=0; i<n&&j<m; i++)
       if (str1[j] == str2[i])
         j++;
   return (j==m);
}
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str1[25001];
        char str2[25001];
        cin>>str1>>str2;
        int m = strlen(str1);
        int n = strlen(str2);
        int min,max;
        if(m<=n)
        {
        	min=m;
        	max=n;
        	isSubSequence(str1, str2, min, max)?cout<<"YES\n": 
	    	cout<<"NO\n";
		}
		else
		{
			min=n;
			max=m;
			isSubSequence(str2, str1, min, max)?cout<<"YES\n": 
	    	cout<<"NO\n";
		}
        
                                        
	}
    R;
}
