#include<iostream>
using namespace std;
int main()
{   int flag, t,count,i,j,k;
char str1[25001],str2[25001];
    cin>>t;
while(t--)
{
cin>>str1>>str2;
k=0,flag=1;
for(i=0;str1[i]!='\0';++i)
{   count=0;
    for(j=k;str2[j]!='\0';++j)
    if(str1[i]==str2[j])
    {
        count++;
        k=j+1;
        break;
    }
    if(!count)
    {
        flag=0;
        break;
    }
}
if(!flag)
{
k=0,flag=1;
for(i=0;str2[i]!='\0';++i)
{   count=0;
    for(j=k;str1[j]!='\0';++j)
    if(str2[i]==str1[j])
    {
        count++;
        k=j+1;
        break;
    }
    if(!count)
    {
        flag=0;
        break;
    }
}
}
if(flag)
    cout<<"YES\n";
else
    cout<<"NO\n";
}
return 0;
}
