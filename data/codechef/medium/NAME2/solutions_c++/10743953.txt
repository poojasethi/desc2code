#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int t,i,j;
    char m[25005],w[25005],temp[25005];
    scanf("%d",&t);
    while(t--)
    {
        j=0;
        cin>>m>>w;
        if(strlen(w)>strlen(m))
        {
            strcpy(temp,m);
            strcpy(m,w);
            strcpy(w,temp);
        }
        for(i=0;i<strlen(m);i++)
        {
            if(w[j]==m[i])
            {
                j++;
            }
        }
        if(j==strlen(w))
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
