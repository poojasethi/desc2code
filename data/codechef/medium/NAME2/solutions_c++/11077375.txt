#include <iostream>
using namespace std;

int main() {
	int t,flag=0,len1,len2,i,j,index;
	string m,w;
	cin>>t;
	while(t--)
	{
		cin>>m;
		cin>>w;
		len1=m.length();
		len2=w.length();
		index=0;
		if(len1>=len2)
		{
			for(i=0;i<len2;i++)
			{
				flag=0;
				for(j=index;j<len1;j++)
				{
					if(m[j]==w[i])
					{
						flag=1;
						index=j+1;
						break;
					}	
				}
				if(flag==0)
				break;
			}	
		}
		else
		{
			for(i=0;i<len1;i++)
			{
				flag=0;
				for(j=index;j<len2;j++)
				{
					if(w[j]==m[i])
					{
						flag=1;
						index=j+1;
						break;
					}
				}
				if(flag==0)
				break;
			}		
		}
		if(flag==0)
		cout<<"NO"<<endl;
		else
		cout<<"YES"<<endl;
	}	
	return 0;
	
}
