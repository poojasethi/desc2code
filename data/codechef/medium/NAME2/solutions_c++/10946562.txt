int main()
{
    int i,t,s11,s22;
    string s1,s2;
    cin>>t;
    for(i=1;i<=t;i++)
    {
        cin>>s1;
        cin>>s2;
        s11=s1.length();
        s22=s2.length();
        if(s11<=s22)
            check(&s1[0],&s2[0]);
        else
            check(&s2[0],&s1[0]);
    }

}

void check(char *a,char *b)
{
    int i=0,j=0,p;
    while(a[i]!='\0')
    {
        if(a[i]==b[j])
        {
            i++;
            j++;
        }
        else
            j++;
        if(b[j]=='\0'&&a[i]!='\0')
        {
            p=0;
            break;
        }
        if(b[j]=='\0'&&a[i]=='\0')
            p=1;
        if(b[j]!='\0'&&a[i]=='\0')
            p=1;
    }
    if(p==1)
        cout<<"YES"<<endl;
    else
        cout<<"NO"<<endl;

}
