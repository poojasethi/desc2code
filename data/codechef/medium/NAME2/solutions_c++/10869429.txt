#include<iostream>
#include<string>
using namespace std;
bool check(string a,string b);

bool check(string a,string b){

int i=0;

for(int j=0;j<a.length();j++){
if(a.at(j)==b.at(i)){
i++;
if(i==b.length()){
return true;
}
}
}
if(i==b.length()){
return true;
}else{
return false;
}


}


int main(){
int t;
cin>>t;
while(t-->0){
string i,j;
cin>>i>>j;

if(i.length()>j.length()){

if(check(i,j)){
cout<<"YES"<<endl;
}else{
cout<<"NO"<<endl;
}
}else{
if(check(j,i)){
cout<<"YES"<<endl;
}else{
cout<<"NO"<<endl;
}

}

}
return 0;
}
