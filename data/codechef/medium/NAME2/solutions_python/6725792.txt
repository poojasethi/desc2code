
def stri(s1, s2):
	i = 0
	j = 0
	ret = ""
	l2 = len(s2)
	l1 = len(s1)
	while i < l1 and j < l2:
		if s1[i] == s2[j]:
			ret += s1[i]
			i += 1
			j += 1
		else:
			j += 1		
	return ret == s1

n = int(raw_input())
for i in range(n):
	s = raw_input().split()
	s1 = s[0]
	s2 = s[1]
	if(len(s2) < len(s1)):
		s1,s2 = s2, s1
	if stri(s1, s2):
		print "YES"
	else:
		print "NO"
