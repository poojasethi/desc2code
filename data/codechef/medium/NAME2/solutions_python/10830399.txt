    agla=raw_input().split()
    aage=0
    if len(agla[0])>len(agla[1]):
        pichla=agla[0]
        agla=agla[1]
    else:
        pichla=agla[1]
        agla=agla[0]
    for i in range(len(pichla)):
        if pichla[i]==agla[aage]:
            aage+=1
        if aage==len(agla):
