t= int(raw_input())
while t > 0:
	M,W = raw_input().split()
	k = 0
	l = 0
	for i in range(len(W)):
		if W[i] == M[k]:
			k += 1
			if k == len(M):
				break
	for i in range(len(M)):
		if M[i] == W[l]:
			l += 1
			if l == len(W):
				break
	if k == len(M) or l == len(W):
		print "YES"
	else :
		print "NO"
	t -= 1
