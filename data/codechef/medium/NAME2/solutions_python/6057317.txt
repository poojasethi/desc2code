t = int(raw_input())
from math import sqrt
for _ in xrange(t):
    m, w = raw_input().split()
    def f(m, w):
        n = 0
        for char in m:
            if char == w[n]:
                n += 1
            if n == len(w):
                return True
        return False
    print "YES" if f(m, w) or f(w, m) else "NO"
