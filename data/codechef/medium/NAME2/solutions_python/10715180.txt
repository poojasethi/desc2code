def subsequence(a, b):
	alen = len(a)
	blen = len(b)
	if blen < alen:
		return False
	j = -1
	for i in xrange(alen):
		while True:
			j += 1
			if j >= blen:
				return False
			if a[i] == b[j]:
				break
	return True
		
cases = int(raw_input())
for _dummy in range(cases):
	m, w = raw_input().split()
	if subsequence(m, w) or subsequence(w, m):
		print "YES"
	else:
		print "NO"
