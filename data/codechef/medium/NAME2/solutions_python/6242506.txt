t=int(raw_input())

while t:
    t-=1
    m,w=raw_input().split()
    i,j=0,0
    if len(m)<len(w):
        while i<len(w) and j<len(m):
            if w[i]==m[j]:
                j+=1
            i+=1
        if j==len(m):
            print 'YES'
        else:
            print 'NO'
    else:
        while i<len(w) and j<len(m):
            if w[i]==m[j]:
                i+=1
            j+=1
        if i==len(w):
            print 'YES'
        else:
            print 'NO'
