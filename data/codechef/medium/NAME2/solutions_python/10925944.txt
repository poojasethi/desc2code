for _ in xrange(input()):
    a,b=map(str,raw_input().split())
    la=len(a)
    lb=len(b)
    if la>lb:
        a,b=b,a
        la,lb=lb,la
    j=0
    for i in xrange(lb):
        if b[i]==a[j]:
            j+=1
        if j==len(a):
            break
    if j==la:
        print "YES"
    else:
        print "NO"
