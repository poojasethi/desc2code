def compute(s1, s2):
	i = 0
	j = 0
	len1 = len(s1)
	len2 = len(s2)
	while i < len1 and j < len2:
		if s1[i] == s2[j]:
			i += 1
		j += 1
	return i == len1

t = int(raw_input())

for _ in range(t):
	m, w = raw_input().strip().split(' ')
	if len(m) > len(w):
		m, w = w, m
	if compute(m, w):
		print 'YES'
	else:
		print 'NO'
