#include <cstdio>
#include <cmath>
#include <iostream>
using namespace std;

#define N 11
long long fac[N]={1};
long long dp[N];

long long f(long long x)
{
  if (x > 9876543210ll) return dp[10];
  int a[N];
  int n=0, d;

  for( ;x; x/=10)
    a[n++]=x%10;

  long long ans=dp[n-1];

  int mark[10]={0};

  for(int i=n-1;i>=0;i--)
  {
    for(d=(i== n - 1 ? 1 : 0); d<a[i]; d++)
      if(!mark[d])
        ans+=fac[9-(n-1-i)]/fac[10-n];

    if(mark[d]) break;
    mark[d]=1;
  }
  return ans;
}

int main()
{
  int T;
  long long L,R;

  for(int n = 1; n < N; n++)
    fac[n] = n * fac[n-1];

  dp[0]=0;

  for(int n=1;n<N;n++)
    dp[n] = dp[n-1] + 9 * fac[9]/fac[10-n];

  for(scanf("%d",&T);T--;) {
    scanf("%lld%lld",&L,&R);
    long long resL=f(L);
    long long resR=f(R+1);
    printf("%lld\n",resR-resL);
  }
  return 0;
}
