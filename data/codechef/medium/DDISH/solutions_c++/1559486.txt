#include<cstdio>
#include<cmath>
#include<cstring>
#define ULL unsigned long long
#define LL long long
LL fac[12],cnt_short[12];
char d[20];
LL chef(LL n){
    if(n>=9876543210LL) return cnt_short[11];
    else if(n<10) return n;
    else{
         int sl,len,f_opt,i;
         LL sum;
         sl=sprintf(d,"%lld",n);
        sum=0;
        sum+=cnt_short[sl];
        i=d[0]-'0'-1;
        //printf("l=%d\n",9-sl-1);
        sum+=i*(fac[9]/fac[9-sl+1]);
        for(len=1;len<sl;len++){
          f_opt=d[len]-'0';
          bool f=false;
          for(i=0;i<len;i++){
            if(d[i]<d[len]) f_opt--;
            if(d[i]==d[len]) f=true;
          }
             sum+=f_opt*fac[9-len]/fac[10-sl];
          if(f) break;
        }
        if(len==sl) sum++;                                       
        return sum;
    }                                      
}
int main(){
    int i,t;
    LL x,y,l,r;
    fac[0]=1;
    for(i=1;i<11;i++) fac[i]=fac[i-1]*i;
    cnt_short[0]=cnt_short[1]=0;
    for(i=1;i<11;i++) cnt_short[i+1]=cnt_short[i]+9*fac[9]/fac[10-i];
    scanf("%d",&t);
    while(t--){
       scanf("%lld%lld",&r,&l);
       x=chef(l);
       y=chef(r-1);
       printf("%lld\n",x-y);
    }
    return 0;
}
