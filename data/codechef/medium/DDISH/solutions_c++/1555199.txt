#include<iostream>
#include<stdio.h>
#define ll unsigned long long int
#define N 11
using namespace std;
ll fac[N]={1};
ll dp[N];
ll f(ll x)
{
	if(x>9876543210LL) 
	return dp[10];
	int a[N];
	int n=0;
	for(;x;x/=10) a[n++]=x%10;
	ll ans=dp[n-1];
	int mark[10]={0};
	for(int i=n-1;i>=0;i--)
	{
		int d;
		for(d=(i==n-1?1:0);d<a[i];d++) 
		if(!mark[d])
		ans+=fac[9-(n-1-i)]/fac[10-n];
		if(mark[d]) break;
		mark[d]=1;
	}
	return ans;
}


int main()
{
	ll t,l,r;
	cin>>t;
	for(int n=1;n<N;n++)
		fac[n]=n*fac[n-1];
	dp[0]=0;
	for(int n=1;n<N;n++)
		dp[n]=dp[n-1]+9*fac[9]/fac[10-n];
	
	while(t--)
	{
	cin>>l>>r;	
	ll resL=f(l);
		ll resR=f(r+1);
		printf("%lld\n",resR-resL);
	}
	return 0;
}
