using namespace std;
#define FOR(i,a,b) for(i=a;i<b;i++)
#define REP(i,a) FOR(i,0,a)
#define LL long long
LL D_dish[10000000];
int bi_d[10000000];
int b2i[2000],num;
int b_search(LL N)
{
	int a,b,c;
	a=0;b=num;
	while(a<b)
	{
		
		c=(a+b)/2;
		if(D_dish[c]<=N)
			a=c+1;
		else
			b=c;
	
	}
	return a;
}
int main()
{
	int t,i,rest,bt;
	scanf("%d",&t);
	//First set bit conversion 
	FOR(i,0,10)
		b2i[1<<i]=i;
	
	//Now intial D dish
	num=0;
	FOR(i,1,10)
	{
		D_dish[num]=i;
		bi_d[num]=((1<<10)-1)^(1<<i);
		++num;
	}
	FOR(i,0,num)
	{
		rest=bi_d[i];
		while(rest)
		{
			bt= (rest & -rest);
			D_dish[num]=D_dish[i]*10 +b2i[bt];
			bi_d[num]=(bi_d[i]^bt);
			num++;
			rest^=bt;
		}
	}
	LL R,L;
	while(t--){
	scanf("%lld %lld",&L,&R);
	printf("%d\n",b_search(R)-b_search(L-1));
	}
	return 0;
}
