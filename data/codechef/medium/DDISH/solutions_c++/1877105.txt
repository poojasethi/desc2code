LL v[8877690];
int N = 0;
void solve() {
	LL L,R;
	scanf("%lld%lld",&L,&R);
	int start = lower_bound(v,v + N,L) - v;
	int end = upper_bound(v,v + N,R) - v;
	printf("%d\n",end - start);
}

	queue< pair<LL, int> > q;
	for (int i = 1; i < 10; i++) {
		q.push(mp(i,1 << i));
	}
	while(!q.empty()) {
		LL n = q.front().first;
		int mask = q.front().second;
		q.pop();
		v[N++] = n;
		for (int i = 0; i < 10; i++) {
			if (mask & (1 << i)) continue;
			q.push(mp(n * 10 + i,mask | (1 << i)));

 	while(T--) {
 		solve();
