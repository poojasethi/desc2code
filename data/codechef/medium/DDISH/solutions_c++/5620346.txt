    int match(char *t,int len)
    {

    break;

    {
    p=t[i]-str[i];
    for(j=t[i]-'0'-1;j>=0;j--)
    if(map[j])
    p--;
    map[t[i]-'0']++;
    for(j=i+1,k=9-i;j<len;j++,k--)
    p*=k;
    ans+=p;
    }

    map[t[i]-'0']++;
    else
    break;
    }
    if((i==len)&&(map[t[i-1]-'0']==1))
    ans++;
    return ans;
    }
