#include<stdio.h>
#include<string.h>

int help[11] = {0, 9, 81, 648, 4536, 27216, 136080, 544320, 1632960, 3265920, 3265920};

int handle(char *A, int len_A, int symbol){
	int i, iT, subterm, addterm = 0, j, flag = 1;
	iT = help[len_A]/9;
	for (i = 0; i < len_A; i++){
		A[i] = A[i] - '0';
		subterm = 0;
		for(j = 0; j < i; j++){
			if (A[j] == A[i])
				flag = 0;
			else if(A[j] > A[i])
				subterm = subterm + 1;
		}
		addterm = addterm + (9-subterm-A[i])*iT;
		if (flag == 0){
			break;
		}
		if (i < 9)
			iT = iT/(9 - i);
	}
	if (symbol == 0)
		addterm = addterm + flag;
	return addterm;
}

int main(){
	int T, i, len_L, len_R, ans, j;
	char L[19], R[19];
	scanf("%d", &T);
	for (i = 0; i < T; i++){
		ans = 0;
		scanf("%s", L);
		scanf("%s", R);
		len_L = strlen(L);
		if (len_L > 10)
		{
			printf("0\n");
			continue;
		}
		len_R = strlen(R);
		if (len_R > 10)
			len_R = 11;
		if (len_L != len_R){
			if (len_R != 11)
				ans = handle(L, len_L, 0) + help[len_R] - handle(R, len_R, 1);
			else
				ans = handle(L, len_L, 0);
			for(j = len_L+1; j < len_R; j++)
				ans = ans + help[j];
		}else{
			ans = handle(L, len_L, 0) - handle(R, len_R, 1);
		}
		printf("%d\n", ans);
	}
	return 0;
}
