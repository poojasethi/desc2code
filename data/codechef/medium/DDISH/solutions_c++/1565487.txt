#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int D[11];
int FF[11][11];
int F[] = {1, 1, 2, 6, 24, 120, 720, 5040, 40320, 362880, 3628800};
int dp[] = {0, 9, 90, 738, 5274, 32490, 168570, 712890, 2345850, 5611770, 8877690};

int count(long long n)
{
  int len = 0, i, d, ans = 0;
  if (n > 9876543210ll) return dp[10];

  while (n) {
    D[len++] = n%10;
    n /= 10;
  }
  
  ans = dp[len-1];

  int mark = 0;
  int j = 0;
  for (i = len-1; i >= 0; i--) {
    for (d = (i == len - 1 ? 1 : 0); d < D[i]; d++) {
      if (mark & (1 << d)) continue;

      ans += FF[9-j][10-len];
      //ans += F[9-j]/F[10-len];
    }

    j++;
    if (mark & (1 << D[i])) break;
    mark = mark | (1 << D[i]);
  }

  return ans;
}


int main(int argc, char **argv)
{
  int t, i, j, l, r;
  long long L, R;

  for (i = 0; i < 10; i++)
    for (j = 1; j < 11; j++)
      FF[9-i][10-j] = F[9-i]/F[10-j];

  scanf("%d", &t);

  while (t--) {
    scanf("%lld %lld", &L, &R);

    l = count(L);
    r = count(R + 1);
    printf("%d\n", r - l);
  }

  return 0;
}
