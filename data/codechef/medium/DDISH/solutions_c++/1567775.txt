long long int look[10];

long long int power(long long int a, long long int e){
     long long int res=1;
     while(e>0){
                if(e&1)
                       res*=a;
                e=e>>1;
                a=a*a;
                }
     return res;
     }

int nodig(long long int num){
    int c=0;
    while(num>0){
                 c++;
                 num/=10;
                 }
    return c;
}

int locdig(long long int num, int loc){
    long long int ans=num/power(10ll, loc-1);
    ans%=10;
    return ans;
}

     int nd=nodig(num);

     for(int i=1;i<nd;++i){
             ans+=look[i];
             }
     if(nd>10)
              return ans;
     //cout<<ans<<endl;
     int i=1;
     vector <int> haso(10,0);
     int flag=0;
     int N=nd;
     while(nd>0){
                 int d=locdig(num,nd);
                 long long int c=0;
                 for(int ii=0;ii<d;++ii){
                         if(nd==N && ii==0)
                                  continue;
                         if(haso[ii]==0)
                                        c++;
                         }
                 //cout<<"c="<<c<<endl;
                 int a=10-i;
                 i++;
                 for(int j=0;j<nd-1;++j){
                         c*=a;
                         a--;
                         //cout<<"c="<<c<<endl;
                         }
                 ans+=c;
                 //cout<<"ans="<<ans<<endl;
                 nd--;
                 if(haso[d]==1){
                                flag=1;
                                break;
                                }
                 haso[d]=1;
                 }
     if(flag==0)
                ans++;

     }
