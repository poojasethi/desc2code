#include<cstdio>
#include<iostream>
using namespace std;
long long int maxi=9876543210LL;
int fun(long long int x,int nine[])
{
if(x<10)
return x;
long long int i,j,l,k,d[12],ans,tmp;
bool prev[12];
for(i=0;i<=9;i++)
prev[i]=0;
k=x;i=0;
while(k>0)
     {
     d[i++]=k%10;
      k=k/10;
     }
     i--;
     for(j=i;j>i/2;j--)
     {
     	k=d[j];
     	d[j]=d[i-j];
     	d[i-j]=k;
     }
     l=i;
     ans=i=0;
     while(i<=l)
     {
       if(i>0)
         prev[d[i-1]]=1;
     k=0;
       if(i==0)
         tmp=d[i]-1;
     else
     {
     for(j=0;j<d[i];j++)
      if(prev[j]==0)
        k++;
     tmp=k;
     }
     j=i+1;
     k=9-i;
     while(j<=l)
     	{
     	   tmp=tmp*k;
     	   k--;
     	   j++;
     	}
     	ans+=tmp;
       	if(i>0 && prev[d[i]]==1)
       	break;
     	//cout<<ans<<"****\n";
     	i++;
     }
     if(i==l+1 && prev[d[i-1]]==0)
     {
     //cout<<"hello\n";
     ans+=1;
     }
     ans+=nine[l];
     return ans;
}
int main()
{
int t,nine[12];
long long int l,r,ans[3],i,j,k;
scanf("%d",&t);
nine[0]=0;
for(i=1;i<=10;i++)
{
   r=9,j=9;
   for(k=2;k<=i;k++)
   r*=j--;
   nine[i]=nine[i-1]+r;
}
while(t--)
   {
     i=0;
     scanf("%lld%lld",&l,&r);
     if(l>maxi)
     {
       printf("0\n");
       continue;
     }
     else if(l==maxi)
     {
       printf("1\n");
       continue;
     }
     if(r>maxi)
     r=maxi;
     k=r;
     //cout<<"param\n";
     ans[1]=fun(r,nine);
    ans[2]=fun(l-1,nine);
    //cout<<ans[1]<<" "<<ans[2]<<endl;
     printf("%lld\n",ans[1]-ans[2]);
   }
   return 0;
}
