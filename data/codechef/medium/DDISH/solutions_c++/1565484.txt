#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int D[11];
long long F[11], dp[11];
long long FF[11][11];

int count(long long n)
{
  int len = 0, i, d, ans = 0;
  if (n > 9876543210ll) return dp[10];

  while (n) {
    D[len++] = n%10;
    n /= 10;
  }
  
  ans = dp[len-1];

  int mark = 0;
  int j = 0;
  for (i = len-1; i >= 0; i--) {
    for (d = (i == len - 1 ? 1 : 0); d < D[i]; d++) {
      if (mark & (1 << d)) continue;

      ans += FF[9-j][10-len];
      //ans += F[9-j]/F[10-len];
    }

    j++;
    if (mark & (1 << D[i])) break;
    mark = mark | (1 << D[i]);
  }

  return ans;
}


int main(int argc, char **argv)
{
  int t, i, j, l, r;
  long long L, R;

  F[0] = 1;
  dp[0] = 0;
  for (i = 1; i < 11; i++)
    F[i] = i * F[i-1];

  for (i = 1; i < 11; i++)
    dp[i] = 9 * F[9]/F[10-i] + dp[i-1];

  for (i = 0; i < 10; i++)
    for (j = 1; j < 11; j++)
      FF[9-i][10-j] = F[9-i]/F[10-j];

  scanf("%d", &t);

  while (t--) {
    scanf("%lld %lld", &L, &R);

    l = count(L);
    r = count(R + 1);
    printf("%d\n", r - l);
  }

  return 0;
}
