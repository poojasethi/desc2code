#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <queue>
using namespace std;
typedef long long ll;
ll lucky[8877690];
int n;
inline void init()
{
    queue<ll> q;
    queue<int> used;
    for(int i=1;i<10;++i){ q.push(i); used.push(1<<i); }
    while (!q.empty()) {
        ll p=q.front(); q.pop();
        int mask=used.front(); used.pop();
        lucky[n++]=p;
        for(int i=0;i<10;++i)if(!(mask&(1<<i))){
            q.push(10*p+i);
            used.push(mask|(1<<i));
        }
    }
}
inline int func(ll x)
{
    return upper_bound(lucky,lucky+n,x)-lucky;
}
int main()
{
    init();
    int T; scanf("%d",&T);
    while (T--) {
        ll L,R; scanf("%lld%lld",&L,&R);
        printf("%d\n",func(R)-func(L-1));
    }
    return 0;
}
