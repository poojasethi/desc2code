#include<bits/stdc++.h>
#define lli long long int
lli arr[100004];
using namespace std;
int main()
{
        lli i,sum1,sum,j,k,m,n,t,c,ans;
        scanf("%lld",&n);
        c=0;
        sum=0;
        for(i=0;i<n;i++)
        {
                scanf("%lld",&m);
                if(m<0)
                {
                        arr[c++]=m;
                        sum=sum+m;
                }
        }
        scanf("%lld",&m);
        if(c==0)
                ans=0;
        else
        {
                sort(arr,arr+c);
                if(c<m)
                        ans=-sum;

                else
                {
                        sum1=0;
                        for(j=0;j<m-1;j++)
                                sum1=sum1-arr[j]+arr[m-1];
                        ans=(-arr[m-1])*m+sum1;
                }
        }
        printf("%lld\n",ans);
}
