#include<stdio.h>
#include<iostream>
#include<math.h>  
#include<algorithm>                    
#include<stdlib.h>
#include<cstdio>
#include<utility>
#include<cstring>
using namespace std;
int main()
{  
  long long  n,x,arr[100007],nos,minus,ans,i;
  scanf("%lld",&n);
  for(i=0;i<n;i++)  scanf("%lld",&arr[i]);   scanf("%lld",&x);
  sort(arr,arr+n);
  if(x==0)  { printf("0\n");  return 0;}
  
  minus=0;  ans=0;   nos=n;    
  for(i=n-1;i>=0;i--)
  {   
    arr[i]+=minus;		
     if(arr[i]<0)
	 {   
		if(x<=nos)   { ans=ans+abs(arr[i])*x;  nos--;  minus+=abs(arr[i]); }
		else         { ans=ans+abs(arr[i]); nos--;   }	
	 }
	 else  nos--;
  }
  printf("%lld\n",ans);
  return 0;
}
