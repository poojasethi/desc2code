#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
int main() {
	long long int x,i,p,a[100001],n,amt,k,l;
	amt=0;
	cin>>n;
	for(i=0;i<n;i++)
	cin>>a[i];
	cin>>x;
	sort(a,a+n);
	if(x!=0)
	{
	if(x<=100000)
	{
	if(a[x-1]<0)
	{
	p=abs(a[x-1]);
	amt=amt+(x*p);
	for(k=0;k<n;k++)
	a[k]=a[k]+p;
	}
	}
	for(l=0;l<n;l++)
	{
	if(a[l]<0)
	amt=amt+abs(a[l]);
	}
	cout<<amt;
	}
	else
	cout<<"0";
	// your code goes here
	return 0;
}
