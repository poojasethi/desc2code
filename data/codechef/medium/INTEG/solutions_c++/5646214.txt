/*
	Chef and Integers
	Problem code: INTEG
*/
#include <iostream>
#include <algorithm>
#include <numeric>
using namespace std;

int* array;

long long CostToZero(int* array,int n,int x,long long cost){

	for(int i=0;i<min(n,x);++i)
		cost-=array[i];

	return cost;
}
int main(){
	int n,i,j,x,num;
	cin>>n;

	array = new int[n];
	
	j=0;i=0;
	while(j<n){
		cin>>num;
		if(num<0)
			array[i++]=num;
		++j;
	}

	cin>>x;
	sort(array,array+i);
	cout<<CostToZero(array,i,x,0);
}
