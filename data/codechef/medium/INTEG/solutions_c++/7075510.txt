
#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int main()
{
    long n,x,c;
    long long sum,co;
    c=sum=co=0;
    cin >> n;
    long a[n];
    for(long i=0;i<n;i++)
    {
        long m;
        cin >> m;
        if(m<0)
        {
            a[c]=m;
            c++;
        }
    }
    cin >> x;
    sort(a,a+c);
    while(c>x)
    {
        sum=sum+abs(x*(a[c-1]+co));
        co=-a[c-1];
        c--;
    }
    while(c>0)
    {
        sum=sum+abs(a[c-1]+co);
        c--;
    }
    cout << sum;
}
