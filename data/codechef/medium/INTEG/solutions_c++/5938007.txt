#include <bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);cin.tie(0);
	long long int n;
	scanf("%lld",&n);
	long long int temp;
	vector <int> a;
	for(long long int i=0;i<n;i++)
	{
		scanf("%lld",&temp);
		if(temp<0)
			a.push_back(-temp);

	}
	n=a.size();
	sort(a.begin(),a.end());
	long long int x;
	scanf("%lld",&x);
	// for(long long int i=0;i<n;i++)
	// 	cout<<a[i]<<" ";
	// cout<<endl;
	long long int cost=0,sum=0,k=0,pk=0;

	while(n>0)
	{
		if(n>=x)
		{
			cost=x*a[k];
			sum=a[k];
			for(;k<a.size();k++)
			{
				if(a[k]-sum>0)
				{
					// k--;
					
					break;
				}
			}
			n=n-k+pk;
			pk=k;
		}
		else
			break;

	}
	// cout<<cost<<" "<<sum<<endl;
	for(;k<a.size();k++)
	{
		cost+=a[k]-sum;
		// cout<<a[k]-sum<<" ";
	}
	cout<<cost<<endl;


	return 0;
}
