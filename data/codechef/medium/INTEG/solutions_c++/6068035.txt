#include<iostream>
#include<stdlib.h>
using namespace std;

int comp(const void *a,const void *b)
{
	return (*(long long *)a- *(long long *)b);
}
int main()
{
	int n,j;
	long long a[100005];
	long long x,ans;
	ans=0;
	int i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>a[i];
	}
	cin>>x;
	qsort(a,n,sizeof(long long),comp);
	//for(i=0;i<n;i++)
	//{
	//	cout<<a[i]<<" ";
	//}
	if(x==0)
	{
		cout<<"0";
		return 0;
	}
	j=n-1;
	while(a[j]>=0 && j>=0)
	j--;
	if(j==-1)
	{
		cout<<"0\n";
		return 0;
	}
	if(j>x-1)
	{
		ans=(-a[x])*x;
		for(i=0;i<x;i++)
		ans=ans-a[i]+a[x];
	}
	else
	{
		for(i=0;i<=j;i++)
		ans=ans-a[i];
	}
	cout<<ans;
}
