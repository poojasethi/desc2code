#include<bits/stdc++.h>
using namespace std;
int main()
{
		long long int n,a[100000],i,b,sum=0,d,j=0;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&b);
			if(b<0)
			{
				a[j]=b;
				
				j++;
			}
		}
		
		scanf("\n%lld",&d);
		sort(a,a+j);
		if(d>j)
		{
			for(i=0;i<j;i++)
			{
				sum-=a[i];
			}
			printf("%lld\n",sum);
		}
		else
		{
			for(i=0;i<d;i++)
			{
				sum-=a[i];
			}
			printf("%lld\n",sum);
		}
	return 0;
}
