#include<stdio.h>
#include<iostream>
#include<algorithm>
#include <string.h>
#include <math.h>


using namespace std;

int main()
{
   long long int n,i,j,x,a[100001],min1,ans,sum[100001] = {0};
   cin>>n;
   for(i = 1 ;i <= n ;i++) {
   	cin>>a[i];
   	
   		a[i] = a[i]*-1;
   	
   	
   }
   cin>>x;
   sort(&a[1],&a[n+1]);
  
   if(a[n] <= 0) {
   	cout<<"0";
   	return 0;
   	
   	
   }
   j =1;
   for(i = 1;i <= n ;i++) {
   	
   	
   	if(a[i] > 0) {
   		a[j] = a[i];
   		j++;
   	} 
   	
   }
   //cout<<j<<endl;
   sum[0] = 0;
for(i = 1;i< j;i++) {
	sum[i] = sum[i-1] + a[i];
	
	
	
}
min1 = min(a[j-1]*x,sum[j-1]);
for(i = 1; i< j;i++) {
	ans = a[i]*x + sum[j-1] - sum[i]  -(a[i] *( j - 1 - i));
	if(ans < min1) {
		min1 = ans;
	}	
}
cout<<min1<<endl;

   


return 0;
}
