#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#define mod 1000000007ll
using namespace std;
int T,n,m,k,A[100005];
long long res;
int main()
{
  	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&A[i]);
		if(A[i]<0)res-=A[i],A[m++]=A[i];
	}
	scanf("%d",&k);
	if(k==0||m==0)res=0;
	else
	{
		sort(A,A+m);
		if(k==1)res=-A[0];
		else if(k<m){
			res=0;
			for(int i=0;i<k;i++)res-=A[i];
		}
	}
	cout<<res<<"\n";
	return 0;
}
