int scanint(){
	register int c = getchar();
	int x=0;
	bool neg=false;
	for(;(c<48 || c>57) && c!=45 ;c=getchar());
	if(c==45){
		neg=true;
		c=getchar();
	}
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	if(neg==true)
	x=x*(-1);
	return x;
}

long long scanlong()
{
	register int c = getchar();
	long long x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

int main(){
	int n,a,x;
	vector<int> v ;
	n = scanint();
	for(int i=0 ; i<n ; i++){
		a = scanint();
		if(a<0){
			a = -a ;
			v.push_back(a);
		}
	}
	x = scanint();
	long long cost = 0 ;
	int size = v.size() ;
	if(size == 0 || x == 0)
	{
		cout<<"0";
		return 0 ;
	}
	sort(v.begin() , v.end());
	long long diff = 0 , mini , count;
	
	for(int i=0 ; i<size ; ){
		mini = v[i]-diff ;
		count = size - i;
		if(count >= x){
			cost += (mini * x) ;
		}else{
			cost += (mini * count);
		}
		diff += mini ;
		while(i<size && v[i]-diff == 0)
			i++;
	}
	cout<<cost;
}
