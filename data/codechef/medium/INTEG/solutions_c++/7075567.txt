
#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int main()
{
    long n,x,c;
    long long sum,co;
    c=sum=co=0;
    cin >> n;
    long a[n];
    for(long i=0;i<n;i++)
    {
        long m;
        cin >> m;
        if(m<0)
        {
            a[c]=m;
            c++;
        }
    }
    cin >> x;
    sort(a,a+c);
    c--;
    while(c>=0)
    {
        if(c+1>x)
        {
            sum=sum-x*(a[c]+co);
            co=-a[c];
        }
        else
        {
            sum=sum-(c+1)*(a[c]+co);
            co=-a[c];
        }
        c--;
    }
    cout << sum;
}
