#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const int MAXN = 1e6 + 10;
LL A[MAXN];
int main(){
    int N;
    cin >> N;
    for(int i = 1;i <= N;++i){
        cin >> A[i];
    }
    LL x;
    cin >> x;
    sort(A + 1, A + 1 + N);
    reverse(A + 1, A + 1 + N);
    LL ans = 0;
    LL inc = 0;
    for(int i = 1;i <= N;++i){
        A[i] += inc;
        if(N - i + 1 >= x){
            inc += max(0LL, 0 - A[i]);
            ans += x * max(0LL, 0 - A[i]);
        } else{
            ans += max(0LL, 0 - A[i]);
        }
    }
    cout << ans << "\n";
    return 0;
}
