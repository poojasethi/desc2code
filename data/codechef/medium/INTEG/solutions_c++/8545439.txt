#include <cstdio>
#include <cstdlib>
#include <algorithm>

#define MAX_N 100000

#define READ_INT(_x)					\
  {							\
    int _n_flag = 0;					\
    register int _tmp = 0, _ch = getchar_unlocked();	\
    while (_ch < 48)					\
      {                                                 \
	if (_ch == '-') break;                          \
	_ch = getchar_unlocked();			\
      }                                                 \
    if (_ch == '-')					\
      {							\
	_n_flag = 1;					\
	_ch = getchar_unlocked();			\
      }							\
    while (48 <= _ch)					\
      {							\
	_tmp = (_tmp << 3) + (_tmp << 1) + _ch - 48;	\
	_ch = getchar_unlocked();			\
      }							\
    (_x) = ((_n_flag) ? -1 : 1) * _tmp;			\
  }							\

char _str[20];
#define WRITE_INT(_x)				\
  {						\
    register int _i = 0;			\
    long long _tmp = (_x);			\
    do						\
      {						\
	*(_str + _i++) = _tmp % 10 + 48;	\
	_tmp /= 10;				\
      } while (_tmp != 0);			\
    for (_i--; _i >= 0; _i--)			\
      putchar_unlocked(*(_str + _i));		\
  }						\

int main(void)
{
  long long sum;
  int *a, i, n, x;

  a = (int *) malloc(MAX_N * sizeof *a);
  if (a == NULL)
    {
      perror("malloc");
      return 1;
    }

  READ_INT(n);
  for (sum = i = 0; n-- > 0; )
    {
      READ_INT(x);
      if (x < 0)
	sum += *(a + i++) = -x;
    }
  READ_INT(x);

  if (i <= x)
    {
      WRITE_INT(sum);
      putchar_unlocked('\n');
    }
  else
    {
      std::sort(a, a + i);
      for (sum = 0, n = i - x; n < i; )
	sum += *(a + n++);
      WRITE_INT(sum);
      putchar_unlocked('\n');
    }

  free(a);
  return 0;
}
