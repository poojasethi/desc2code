#include<bits/stdc++.h>
using namespace std;


typedef long long int ll ;


void solve()
{
    int n ;
    scanf("%d",&n) ;
    ll input[n],k=0,sum=0 ;
    for(int i=0  ; i<n ; i++) {
        long long int temp ;
        scanf("%lld",&temp) ;
        if(temp<0) {
            input[k]=temp*(-1) ;
            k++ ;
            sum+=input[k-1] ;
        }
    }
    sort(input,input+k) ;
    ll cost,maxCost=sum,coun=0,tSum=0 ;
    scanf("%lld",&cost) ;
    for(int i=0 ; i<k ; ) {
        ll tCost = input[i]*cost ;
        if(tCost>maxCost) {
            break ;
        }
        else {
            ll curr=input[i] ;
            while(curr==input[i]) {
                tSum+=input[i] ;
                coun++ ;
                i++ ;
            }
            ll ttCost= (sum-tSum) -(k-coun)*curr + tCost ;
            if(ttCost<maxCost) {
                maxCost=ttCost ;
            }
        }
    }
    printf("%lld\n",maxCost) ;
}

int main()
{
    //freopen("input.txt","r",stdin) ;
    //freopen("out.txt","w",stdout) ;
    solve() ;
    return 0 ;
}
