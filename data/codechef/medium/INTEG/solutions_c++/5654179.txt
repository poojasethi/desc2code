#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;

int main()
{
	int t,j=0;	
	long long x,arr[100000],sum=0,tmp;
	scanf("%d",&t);
	
	for(int i=0;i<t;i++)
	{
	scanf("%lld",&tmp);
	
		if(tmp<0)
		{
			arr[j]=tmp;
			sum+=tmp;
			j++;
		}
	}
	arr[j++]=0;
	scanf("%lld",&x);
	sort(arr,arr+j);
	
	
	if(x==0)
	{
		printf("0\n");
		return 0;
	}
	
	else if(x==1)
	{
		printf("%lld\n",arr[0]);
		return 0;
	}
	
	else if(x>abs(sum))
	{
		printf("%lld\n",abs(sum));
		return 0;
	}
	
	 
	else
	{
		long long diff=0;
		
		for(int i=0;i<	j-1;i++)
		{
			long long t1,t2;
			t1=((arr[i+1]-arr[i])*(i+1));	
			t2=(arr[i+1]-arr[i])*x;
			diff+=min(t1,t2);
			}
		
		printf("%lld\n",diff);
	}
return 0;
}
