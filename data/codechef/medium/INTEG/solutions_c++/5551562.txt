#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int n,a,k=0;
    cin>>n;
    int b[n];
    long long int sum=0;
    for(int i=0;i<n;i++)
    {
        cin>>a;
        if(a<0)
            {
            b[k++]=-1*a;
            sum=sum+(-1)*a;
            }
    }
    int x;
    cin>>x;
    if(k<=x)
    {
        cout<<sum<<endl;
    }
    else
    {
        sort(b,b+k);
        sum=0;
        for(int i=k-1,j=0;j<x;i--,j++)
            sum=sum+b[i];
        cout<<sum<<endl;
    }



    return 0;
}
