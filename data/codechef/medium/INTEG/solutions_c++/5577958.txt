#include<fstream>

#define pb push_back
#define mp make_pair
#define fi first
#define se second
#define INF 1000000000
#define LL long long int
#define EPS 1e-11
#define PI 3.14159265359

LL n , a[100003],x , ans,add;

int main(void)
{
    scanf("%lld",&n);
    for (int i = 0 ; i<n; i++) scanf("%lld" , &a[i]);
    scanf("%lld" , &x);

    sort(a, a + n);
    add = ans = 0;
    for (int i = n - 1; i>=0; i--)
    {
        a[i] += add;
        if (a[i]<0)
        {
            if (x <= i+1)
            {
                add -= a[i];
                ans -= (a[i]*x);
            } else
            {
                ans -= a[i];
            }
        }
    }

    printf("%lld\n",ans);
}
