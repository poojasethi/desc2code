using namespace std;

int n;
ill a[20009];
ill b[20009];
int s[20009];
ill ans[10009];
int nn;

void process()
{
    set <pii> ss;
    int i;
    F (i, 0, n) {

   //     pii pp(a[i]+b[i], i);
        ss.insert(pii(a[i]+b[0], i));
    }

    nn = 0;
    while (nn  < min(10009, n*n)) {
          //  cout << nn << endl;
        pii pp = *(ss.begin());
        ss.erase(ss.begin());
        ans[nn++] = pp.first;
        s[pp.second]++;
        int x = pp.second;
        if (s[x] != n) {
            ss.insert(pii(a[x]+b[s[x]], x));
        }
    }
}

int main()
{
  //  freopen ("input.txt", "r", stdin);

    int t;
    S (t);
    while (t--) {
        S (n);
        int q;
        S (q);
        int i;
        F (i, 0, n) S1 (a[i]);
        F (i, 0, n) S1 (b[i]);
        sort (a,a+n);
        sort (b,b+n);
        memset (s, 0, sizeof(s));

        process();
        //cout << "ASDA" << endl;
        while (q--) {
            int x;
            S (x);
            x--;
            printf ("%lld\n", ans[x]);
        }

    }

    return 0;
}
