#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<queue>
using namespace std;
long long int a[20001],b[20001],ans[20001];
int ptr[20001];
pair<int,int> q[20001];
pair<long long int,int> low;
priority_queue<pair<long long int,int> > pq;
int main()
{
	int t,k,Q,i,c;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&k,&Q);
		for(i=0;i<k;i++)
			scanf("%lld",&a[i]);
		for(i=0;i<k;i++)
			scanf("%lld",&b[i]);
		for(i=0;i<Q;i++)
		{
			scanf("%d",&q[i].first);
			q[i].second=i;
		}
		sort(a,a+k);
		sort(b,b+k);
		sort(q,q+Q);
		
		while(pq.size()!=0)
			pq.pop();
		
		for(i=0;i<k;i++)
		{
			ptr[i]=0;
			pq.push(make_pair(-a[i]-b[0],i));
		}
		c=0;
		
		for(i=0;i<Q;i++)
		{
			while(c<q[i].first)
			{
				low=pq.top();
				pq.pop();
				c++;
				ptr[low.second]++;
				if(ptr[low.second]<k)
					pq.push(make_pair(-a[low.second]-b[ptr[low.second]],low.second));
			}
			ans[q[i].second]=-low.first;
		}
		for(i=0;i<Q;i++)
			printf("%lld\n",ans[i]);
	}
	return 0;
}
