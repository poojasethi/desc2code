#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

#define ull long long int


int compr(const void *a, const void *b) {
    return (*(ull *)a - *(ull *)b);
}

int main() {
    int t;
    scanf("%d", &t);
    while(t--) {
        int K, Q;
        scanf("%d%d", &K, &Q);
        ull A[K+1], B[K+1];
        for(int i=1; i<=K; i++)
            scanf("%lld", &A[i]);
        for(int i=1; i<=K; i++)
            scanf("%lld", &B[i]);

        sort(A+1, A+K+1);
        sort(B+1, B+K+1);

        vector<ull> C;
        
        int co = 0;
        for(int i=1; i<=K; i++) {
            int ind = min((10000/i) + 1, K);
            for(int j=1; j <=ind; j++) {
                C.push_back(A[i]+B[j]);
            }
        }
        sort(C.begin(), C.end());
        while(Q--) {
            int qi;
            scanf("%d", &qi);
            printf("%lld\n", C[qi-1]);
        }

    }
    return 0;
}
