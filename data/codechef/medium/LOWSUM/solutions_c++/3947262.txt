#include <iostream>
#include <algorithm>
#include <vector>
#include <cmath>
#include <deque>
#include <string.h>
#include <stdlib.h>
#include<iterator>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define M 1000000
#define MAX 3000

using namespace std;
int main ()
{int p,l,max,ind,i,j,c[20005];
LL inf=3*1e18;

LL ans[10005],min;
vector <LL> a(20005);
vector <LL> b(20005);
a.pb(0);b.pb(0);
int d[505];
scanf("%d",&p);
while (p--)
{
	int k,q;
sc(k);sc(q);
	for ( i=1;i<=k;i++)
	{sl(a[i]);
	c[i]=1;	}
	for ( i=1;i<=k;i++)
	{sl(b[i]);	}
	sort(a.begin()+1,a.begin()+k+1);
	sort(b.begin()+1,b.begin()+k+1);
	a[k+1]=inf;
	b[k+1]=inf;
	max=0;
	for( i=1;i<=q;i++)
	{
		sc(d[i]);
		if(max<d[i])
		max=d[i];
	}
	max=max;
    
    for( i=1;i<=max;i++ )
    {
    	min=inf;ind=-1;
    	for(j=1;j<=k;j++)
    	{
    		if(a[j]+b[c[j]]<min)
    		{
    			min=a[j]+b[c[j]];
    			ind=j;
    		}
    	}
    	ans[i]=min;
    	c[ind]++;
    }
    for(int i=1;i<=q;i++)
    {
    	pl(ans[d[i]]);
    }
    }
return 0;
}
