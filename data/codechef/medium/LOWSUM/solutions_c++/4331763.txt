#include <stdio.h>
#include <algorithm>
#include <set>
#include <vector>
#include <math.h>
using namespace std;

typedef long long INT64;
vector<INT64> A[5];
vector<INT64> B[5];
int Qi[5][500];
int q[5];
int k[5];
vector<INT64> v;

int main(int argc, char* argv[])
{
	int h,i,j,t;
	scanf("%d", &t);
	for(i=0;i<t;i++)
	{
		scanf("%d %d",&k[i],&q[i]);
		A[i].assign(k[i],0);
		B[i].assign(k[i],0);
		for(j=0;j<k[i];j++) scanf("%lld", &A[i][j]);
		for(j=0;j<k[i];j++) scanf("%lld", &B[i][j]);
		for(j=0;j<q[i];j++) scanf("%d", &Qi[i][j]);
	}
	set< pair<INT64,int> >::iterator it;
	for(h=0;h<t;h++)
	{
		sort(A[h].begin(),A[h].end());
		sort(B[h].begin(),B[h].end());
		int maxq=0;
		for(i=0;i<q[h];i++)
		{
			if(Qi[h][i]>maxq)
			{
				maxq=Qi[h][i];
			}
		}
		if(h) v.clear();
		int max=sqrt(maxq);
		if(maxq>max*max) max++;
		int end=(k[h]<max) ? k[h] : max;
		INT64 last=0;
		for(i=0;i<end;i++)
		{
			for(j=0;j<end;j++)
			{
				INT64 x=A[h][i]+B[h][j];
				v.push_back(x);
				if(x>last) last = x;
			}
		}
		int idx1=end;
		int idx2=end;
		for(i=end;i<k[h];i++)
		{
			for(j=0;j<idx1;j++)
			{
				INT64 sum=B[h][i]+A[h][j];
				if(sum>=last)
				{
					idx1=j;
					break;
				}
				else
				{
					v.push_back(sum);
				}
			}
			for(j=0;j<idx2;j++)
			{
				INT64 sum=A[h][i]+B[h][j];
				if(sum>=last)
				{
					idx2=j;
					break;
				}
				else
				{
					v.push_back(sum);
				}
			}
		}
		sort(v.begin(),v.end());
		for(i=0;i<q[h];i++)
		{
			int pos=Qi[h][i];
			printf("%lld\n", v[pos-1]);
		}
	}
	return 0;
}
