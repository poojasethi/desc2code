#include <set>
#include <cstdio>
#include <algorithm>

using namespace std;

int T, K, Q, Qi;
unsigned long long A[20000], B[20000], Sum[10000], MK[20000];
set<pair<unsigned long long, int> > st;
set<pair<unsigned long long, int> >::iterator it;
pair<unsigned long long, int> v;

int main()
{
	scanf("%d", &T);
	while(T--)
	{
		scanf("%d%d", &K, &Q);
		for(int i = 0; i < K; ++i)
		{
			scanf("%llu", A + i);
			MK[i] = 0;
		}
		for(int i = 0; i < K; ++i)
			scanf("%llu", B + i);
		sort(B, B + K);
		st.clear();
		st.insert(make_pair(A[0] + B[0], 0));
		it = st.begin();
		for(int i = 1; i < K; ++i)
			st.insert(++it, make_pair(A[i] + B[0], i));
		int MAX = min(10000, K*K);
		for(int i = 0; i < MAX; ++i)
		{
			it = st.begin();
			v = *it;
			st.erase(it);
			Sum[i] = v.first;
			if(++MK[v.second] < K)
				st.insert(make_pair(A[v.second] + B[MK[v.second]], v.second));
		}
		for(int i = 0; i < Q; ++i)
		{
			scanf("%d", &Qi);
			printf("%llu\n", Sum[Qi - 1]);
		}
	}
}
