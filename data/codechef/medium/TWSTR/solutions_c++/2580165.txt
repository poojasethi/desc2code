#include <bits/stdc++.h>
using namespace std;
#define ll long long int
int main()
{
	int n,i,q;
	cin>>n;
	map<ll,string> m;
	ll v;
	string s;
	map<ll,string>::iterator it;
	for(i=0;i<n;i++)
	{
		cin>>s>>v;
		m[v]=s;
	}
	cin>>q;
	for(i=0;i<q;i++)
	{
		cin>>s;
		int flag=0;
		for(it=m.end();it!=m.begin();)
		{
			it--;
			if(s==(it->second).substr(0,s.length()))
			{
				flag=1;
				cout<<it->second<<endl;
				break;
			}
		}
		if(!flag)
			cout<<"NO\n";
	}
	return 0;
}
