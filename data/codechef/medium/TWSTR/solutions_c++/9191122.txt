#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define INF 1000000007
#define MAX 100005
#define pb(a) push_back(a)
int main()
{
	lli index,max_priority,start,l,q,size,i,j,k,n,t,priority,track[35]={0};
	string st,str,max_str;
	map<string,lli>mp;
	slli(n);
	f(i,0,n)
	{
		cin>>str;
		slli(priority);
		mp[str]=priority;
	}
	size=mp.size();
	map<string,lli>::iterator it;
	it=mp.begin();
	for(i=1;i<=size;i++)
	{
		str=it->first;
		if(str[0]=='-')
			index=30;
		else
			index=str[0]-'a'+1;
		if(track[index]==0)
			track[index]=i;
		it++;
	}
	slli(q);//cout<<"q = "<<q<<endl;
	while(q--)
	{
		cin>>str;//cout<<str<<endl;
		max_priority=-INF;
		it=mp.begin();
		if(str[0]=='-')
			index=30;
		else
			index=str[0]-'a'+1;
		start=track[index];//cout<<"start = "<<start<<endl;
		if(start!=0){
			advance(it,start-1);//cout<<"it->first = "<<it->first<<endl;
			while(it!=mp.end())
			{
				st=it->first;//cout<<st<<endl;
				if(st[0]==str[0])
				{
					if(st.length() >= str.length())
					{
						for(i=0;i<str.length();i++)
						{
							if(str[i]!=st[i])
								break;
						}
						if(i==str.length())
						{
							if(it->second > max_priority)
							{
								max_priority=it->second;
								max_str=it->first;
							}
						}
					}
				}
				else
					break;
				//cout<<max_priority<<endl;
				it++;
			}
		}
		if(max_priority==-INF)
			printf("NO\n");
		else
			cout<<max_str<<endl;
	}
}
