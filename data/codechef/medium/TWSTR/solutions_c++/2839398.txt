#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
struct T{
	char s[1001];
	int v;
};
bool  myf(T a, T b)
{
	return (a.v>b.v);
}
int main()
{
	int t,te,i,j,k,n,q;
	static T inp[1000],x;
	cin>>n;
	char s[1001];
	for(i=0;i<n;i++){
		scanf("%s%d",x.s,&x.v);
		inp[i]=x;
	}
	sort(inp,inp+n,myf);
	cin>>q;
	for(i=0;i<q;i++)
	{
		scanf("%s",s);
		for(j=0;j<n;j++)
		{
			for(k=0;k<strlen(s);k++)
				if(inp[j].s[k]!=s[k])break;
			if(k==strlen(s))break;
		}
		if(j!=n)
			printf("%s\n",inp[j].s);
		else printf("NO\n");
	}
	return 0;	
}
