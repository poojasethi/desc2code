#include <iostream>
#include <algorithm>
using namespace std;
 
int main() {
	int n,q;
	cin>>n;
	string s[n],x,b,c;
	long long int v[n],i,j,sm,pos,temp,flag;
	for(i=0;i<n;i++)
	cin>>s[i]>>v[i];
	cin>>q;
	for(i=0;i<n-1;i++)
	{
		pos=i;
		sm=v[i];
		for(j=i+1;j<n;j++)
		{
			if(sm<v[j])
			{
				sm=v[j];
				pos=j;
			}
		}
		temp=v[pos];
		v[pos]=v[i];
		v[i]=temp;
		x=s[pos];
		s[pos]=s[i];
		s[i]=x;
	}
	while(q--)
	{
		cin>>b;
		int l=b.length();flag=0;
		for(i=0;i<n;i++)
		{
			if(s[i].substr(0,l)==b)
			{
				flag=1;
				cout<<s[i]<<endl;
				break;
			}
		}
		if(flag==0)
		cout<<"NO"<<endl;
	}
	return 0;
}
