#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<algorithm>
using namespace std;
struct product {
long long int p;
char str[1005];
};
bool compare (const struct product  &a,const struct product  &b)
{
return a.p>b.p;
}

int main()
{
struct product s[1003];
long long int n,i,y,j,q,flag,ans;
char str1[1005];
//x=(char*)malloc(1005*sizeof(char));
//str1=(char*)malloc(1005*sizeof(char));
scanf("%lld",&n);
j=0;
for(i=0;i<n;i++)
{
scanf("\n%s",s[j].str);
scanf("%lld",&y);
//s[j].str=x;
s[j].p=y;
j++;
}
sort(s,s+n,compare);
scanf("%lld",&q);
while(q--)
{
ans=-1;
scanf("\n%s",str1);
for(i=0;i<n;i++)
{
flag=0;
for(j=0;str1[j]!='\0';j++)
{
if(str1[j]!=s[i].str[j])
{
flag=1;
break;
}
}
if(flag==0)
{
ans=i;
break;
}
}
if(ans==-1)
printf("NO\n");
else
printf("%s\n",s[ans].str);
}
return 0;
}
