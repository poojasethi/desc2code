#include<iostream>
#include<cstdio>
#include<string.h>
#define scan(a) scanf("%d",&a)
using namespace std;
int main()
{
    int n,i,q,len,j,index,max;
    scan(n);
    string r[n],a;
    int p[n];
    for(i=0;i<n;i++)
    {
        cin>>r[i];
        cin>>p[i];
    }
    cin>>q;
    for(i=0;i<q;i++)
    {  index=-1;
        cin>>a;
        len=a.length();
        index=-1;max=-1000000007;
        for(j=0;j<n;j++)
        {
            if((r[j].substr(0,len)).compare(a)==0)
            {
                if(p[j]>max)
                {
                    max=p[j];
                    index=j;
                }
            }
        }
        if(index==-1)
        {
            cout<<"NO"<<endl;
        }
        else
            {
                cout<<r[index]<<endl;
            }
    }
    return 0;
}
