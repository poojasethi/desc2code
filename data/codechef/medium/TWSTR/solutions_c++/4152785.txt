#include<cstdio>
#include<iostream>
#include<cstring>
#define M 100000000099LL
using namespace std;
#define lld long long int
#define ci(x) x=='-'?26:(int)(x-'a')
struct node
{
    char b;
    bool end;
    lld val,eval;
    node *child[27];
    node()
    {
    	end=false;
        eval=-M;
    	b='\0';
        val=-M;
        for(int i=0;i<=26;++i)child[i]=NULL;
    }
};
node *root,first;
void insert(char s[1003],lld p)
{
    int i,l=strlen(s);
    node* now=root;
    for(i=0;i<l;++i)
    {
        if(!now->child[ci(s[i])])
        {
            now->child[ci(s[i])]=new node;
        }
        if(p>now->val)now->val=p,now->b=s[i];
        now=now->child[ci(s[i])];
    }
	now->eval=p;
	now->end=true;
}
bool search(char s[1005])
{
    node* now=root;
    int l=strlen(s);
    for(int i=0;i<l;++i)
    {
        if(now->child[ci(s[i])])now=now->child[ci(s[i])];
        else return 0;
    }
    printf("%s",s);
    lld hi=max(now->eval,now->val);
    while(now->b!='\0')
    {
        if(hi!=now->val)break;
        printf("%c",now->b);
		now=now->child[ci(now->b)];
    }
    printf("\n");
    return 1;
}
int main()
{
    root=&first;
    lld x;
    char s[1003];
    int t;
    cin>>t;
    while(t--)
    {
        scanf("%s%llu",s,&x);
        insert(s,x);
    }
    cin>>t;
    while(t--)
    {
        scanf("%s",s);
        if(search(s)==0)cout<<"NO\n";
    }
}
