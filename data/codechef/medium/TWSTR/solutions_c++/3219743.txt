inline void fast_int(int *a)
{
	 register char c;
	 while(c<33) c=getchar_unlocked();
	 *a=0;
	 while(c>=33)
	 {
		  *a=*a*10+c-'0';
		  c=getchar_unlocked();	
	 }
}

void solve()
{
	int n,i,j,k,f,t;
	char r[1000][1001],rf[1001];
	int pr[1000];
	fast_int(&n);
	i=0;
	while(i<n)
	{
		scanf("%s %d",r[i],&pr[i]);
		i++;
	}
	fast_int(&t);
	int m=0;
	i=0;
	while(i<t)
	{
		gets(rf);
		j=0;
		m=-1;
		while(j<n)
		{
			k=0;
			f=1;
			while(rf[k]!='\0')
			{
				if(rf[k]!=r[j][k])
				{
					f=0;	
					break;
				}
				k++;
			}
			if(f)
			{
				if(m==-1)
					m=j;
				if(pr[m]<pr[j])
					m=j;
			}
			j++;
		}
		if(m==-1)
			puts("NO");
		else
			puts(r[m]);
		i++;
	}
}

int main()
{
	solve();
    return 0;
}
