using namespace std;
#define gc getchar_unlocked()

int in()
{
  char c;lli t=0,negative=1;
  c=gc;

  while((c<48||c>57)&&c!='-')c=gc;
  if(c=='-')
  {negative=-1;c=gc;}
  while(c>=48&&c<=57)
  {t=(t<<3)+(t<<1)+c-'0';c=gc;}
  return t*negative;
}

int search(char *s1,char *s2,int l)
{
    int ret=1;
    int i=0;
    while(i<l)
    {
            if(s1[i]!=s2[i])
            {
              ret=0;              
              break;
            }
            i++; 
    }
    return ret;
    
}
