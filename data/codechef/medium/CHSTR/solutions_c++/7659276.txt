#include <cstdio>
#include <cstring>
#include <vector>
#include <set>
#include <queue>
#include <stack>

#include <cmath>

#include <iostream>
#define mod 1000000007
#define lli long long int
using namespace std;
vector<lli> f(5001 , 1);
vector<lli> inv(5001 , 1);
long long pow(lli a, lli b, lli MOD)
{
	long long x=1,y=a;
	while(b > 0)
	{
		if(b%2 == 1)
		{
			x=(x*y);
			if(x>MOD) x%=MOD;
		}
		y = (y*y);
		if(y>MOD) y%=MOD;
		b /= 2;
	}
	return x;
}

long long InverseEuler(lli n, lli MOD)
{
	return pow(n,MOD-2,MOD);
}

long long C(int n, int r, lli MOD)
{
	return (f[n]*((inv[r] * inv[n-r])% MOD)) % MOD;
}
int main()
{
	int t,n,q;
	scanf("%d",&t);
	while(t-->0)
	{
		scanf("%d %d",&n,&q);

		string str,temp;
		cin>>str;
		vector<lli> ct(n+1,0);
		priority_queue<string> mypq;
		for(int i=0;i<str.length();i++)
		{
			temp=str.substr(i,n-i);
			mypq.push(temp);
		}
		vector<string> pfarr(n);
		for(int i=n-1;!mypq.empty();i--)
		{
			pfarr[i]=mypq.top();
			mypq.pop();
		}
		string l1,l2;
		l1=pfarr[0];
		vector<lli> lcp(n+1,0);
		vector<lli> v1(n+1,1);
		ct[1]+=l1.length();
		lli maxi=1;
		for(int i=1;i<n;i++)
		{
			l1=pfarr[i-1];l2=pfarr[i];
			lcp[i]=0;
			for(int j=0;j<l1.length() && j<l2.length();j++)
			{
				if(l1.at(j)==l2.at(j))
				{
					lcp[i]++;
				}
				else
				break;
			}
			ct[1]+=l2.length()-lcp[i];
			if(lcp[i]==0)
			v1.assign(n+1,1);
			else if(lcp[i]>=lcp[i-1])
			{
				for(int j=0;j<lcp[i];j++)
				{
					ct[v1[j]]--;
					v1[j]++;
					ct[v1[j]]++;
					if(maxi<v1[j])
					maxi=v1[j];
				}
			}
			else if(lcp[i]<lcp[i-1])
			{
				for(int j=0;j<lcp[i];j++)
				{
					ct[v1[j]]--;
					v1[j]++;
					ct[v1[j]]++;
					if(maxi<v1[j])
					maxi=v1[j];
				}
				for(int j=lcp[i];j<lcp[i-1];j++)
				{
					v1[j]=1;
				}
			}
		}

		for (int i=2; i<=maxi;i++)
		f[i]= (f[i-1]*i) % mod;

		for(int i=2;i<=maxi;i++)
		{
			inv[i]=InverseEuler(f[i],mod);
		}

		vector<lli> result(maxi+1,0);
		for(int i=1;i<=maxi;i++)
		{
			lli po=0;
			for(int j=maxi;j>=i;j--)
			{
				po=po+C(j,i,mod)*ct[j];
				if(po>mod)
				po%=mod;
			}
			result[i]=po;
		}
		lli y;
		for(int i=0;i<q;i++)
		{
			scanf("%lld",&y);
			if(y>maxi)
			{
				printf("0\n");
			}
			else
			{
				printf("%lld\n",result[y]);
			}
		}
	}
}
