#include <bits/stdc++.h>
using namespace std;
#define md int(5000+100)
#define modul int(1e9+7)
#define FOR(i,a,b) for( int i=(a),_b=(b);i<=_b;i++)
#define DOW(i,b,a) for( int i=(b),_a=(a);i>=_a;i--)
int xx[4]={0,0,1,-1};
int yy[4]={1,-1,0,0};
typedef long long ll;
int n,m;
int cnt[md];
int z[md],r[md];
int c[md][md];
char s[md];

void pre()
{
    memset(cnt,0,sizeof(cnt));
    FOR(x,0,n-1)
    {
        int l=0,r=0;
        z[x]=n-x;
        FOR(i,x+1,n-1)
        {
            if (i>r)
            {
                l=r=i;
                while (r<=n-1 && s[r-l+x]==s[r]) r++;
                z[i]=r-l; r--;
            }
            else
            {
                int k=i-l+x;
                if (z[k]<r-i+1) z[i]=z[k];
                else
                {
                    l=i;
                    while (r<=n-1 && s[r-l+x]==s[r]) r++;
                    z[i]=r-l; r--;
                }
            }
        }
        sort(z+x,z+n);

        if (z[x]!=0) cnt[n-x]+=z[x];
        FOR(i,x+1,n-1)
        if (z[i]!=z[i-1])
        {
            cnt[n-i]+=z[i]-z[i-1];
        }
    }
    FOR(i,1,n)
    cnt[i]-=cnt[i+1];
    FOR(k,1,n)
    {
        r[k]=0;
        FOR(i,k,n)
        r[k]=(r[k]+ll(cnt[i])*c[i][k])%modul;
    }
}

int main()
{
    //freopen("inp.txt","r",stdin);
    FOR(i,0,5000)
    c[i][0]=1;
    FOR(i,1,5000)
    FOR(j,1,i)
    c[i][j]=(c[i-1][j-1]+c[i-1][j])%modul;
    int tcase;
    scanf("%d",&tcase);
    FOR(o,1,tcase)
    {
        scanf("%d%d",&n,&m);
        scanf("%s",s);
        pre();
        FOR(i,1,m)
        {
            int x;
            scanf("%d",&x);
            if (x>n) printf("%d\n",0);
            else printf("%d\n",r[x]);
        }
    }
}
