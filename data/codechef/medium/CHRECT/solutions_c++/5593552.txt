#include<iostream>
using namespace std;

int main() {
	int t;
	cin >> t;
	int N,M,K;
	while(t--) {
		cin >> N >> M >> K;
		if((N == 1 && M == 2) || (N == 2 && M == 1) || (N==1 && M==1)) {
			cout << "0" << endl;
		}
		else if ((N == 1) || (M==1)) {
			cout << K << endl;
		}
		else {
			if(K%2 == 0) {
				cout << K/2 << endl;
			}
			else {
				cout << (K+1)/2 << endl;
			}
		}
	}
	return 0;
}
