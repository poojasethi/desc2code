#include<bits/stdc++.h>
#include<iostream>
using namespace std;
int solve(int r,int c,int k){
	if(r==1 && c <=2)return 0;
	if(r==1)return k;
	return ((k+1)/2);
}
int main(){
	int t;
	int n,m,k;
	int c;
	cin>>t;
	for(c=0;c<t;c++){
		cin>>n>>m>>k;
		if(n>m){
			int t;
			t=n;
			n=m;
			m=t;
		}
		cout<<solve(n,m,k)<<endl;
	}
}
