#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int t,k,m,n,ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&n,&m,&k);
        if(n>=2&&m>=2)
            ans=k/2+k%2;
        else if((m>2&&n==1)||(m==1&&n>2))
            ans=k;
        else
            ans=0;
        printf("%d\n",ans);
    }
    return 0;
}
