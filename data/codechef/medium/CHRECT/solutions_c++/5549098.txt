#include<iostream>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        int n,m,k,count=0;
        cin>>n>>m>>k;
        if(n>=2&&m>=2)
            count=(k+1)/2;
        else if((n>2&&m==1)||(m>2&&n==1))
            count=k;
        else
            count=0;
        cout<<count<<endl;
    }
    return 0;
}
