#include<iostream>
using namespace std;
void swap(int &a, int &b)
{
	int t;
	t=a;
	a=b;
	b=t;
}
int main()
{
	int test,n,m,k;
	cin>>test;
	while(test--)
	{
		cin>>n>>m>>k;
		if(n>m)
		swap(n,m);
		if(n==1 && m>2)
		cout<<k<<endl;
		else if(n==1 && m<3)
		cout<<"0\n";
		else
		{
			if(k%2==0)
			cout<<k/2<<"\n";
			else
			cout<<(k/2)+1<<"\n";
		}
	}
	return 0;
}
