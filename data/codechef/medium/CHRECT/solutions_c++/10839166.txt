#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int t,n,m,k;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&n,&m,&k);
        if((m==1&&n==1)||(m==1&&n==2)||(m==2&&n==1))
            printf("0\n");
        else if(m==1||n==1)
            printf("%d\n",k);
        else if(k%2!=0)
            printf("%d\n",(k+1)/2);
        else
            printf("%d\n",k/2);
    }
    return 0;
}
