#include <iostream>
#include <cmath>
using namespace std;

int main() {
	int t,n,m,k,ans;
	float temp;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d %d",&n,&m,&k);
		if((n>1)&&(m>1))
		{
			temp=(ceil)(k/2.0);
			ans=temp;
		}
		else
		{
			if(n==1)
			{
				if(m<=2)
				{
					ans=0;
				}
				else
				{
					ans=k;
				}
			}
			else
			{
				if(n<=2)
				{
					ans=0;
				}
				else
				{
					ans=k;
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
