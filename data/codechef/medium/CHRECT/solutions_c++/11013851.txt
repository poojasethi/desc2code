#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n,m,k;
		cin >> n >> m >> k;
		if (n == 1 && m == 1)
			printf("0\n");
		else
		{
			if (n == 1)
			{
				if (m <= 2)
				{
					printf("0\n");
				}
				else
				{
					printf("%d\n",k);
				}
			}
			else if (m == 1)
			{
				if (n <= 2)
				{
					printf("0\n");
				}
				else
					printf("%d\n",k);
			}
			else
			{
				printf("%d\n",k/2+k%2);
			}
		}
	}
	return 0;
}
