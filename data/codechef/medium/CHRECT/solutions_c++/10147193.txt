#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,n,m,k;
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>k;
        if(n>m)
        {
            n=n+m;
            m=n-m;
            n=n-m;
        }
        if(n==1)
        {
            if(m<=2)
                cout<<"0\n";
            else
                cout<<k<<"\n";
        }
        else
            cout<<ceil(k/2.0)<<endl;
    }
    return 0;
}
