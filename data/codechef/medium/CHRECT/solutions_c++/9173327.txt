#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m,k;
        cin>>n>>m>>k;

        if (n==1&&(m==1||m==2))
            cout<<0<<'\n';
        else if (n==1)
            cout<<k<<'\n';
        else if (m==1&&(n==1||n==2))
            cout<<0<<'\n';
        else if (m==1)
            cout<<k<<'\n';
        else
            cout<<ceil(1.0*k/2)<<'\n';
    }
    return 0;
}
