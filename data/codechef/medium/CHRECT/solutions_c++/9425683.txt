#include<bits/stdc++.h>
using namespace std;
int main()
	{
	int t;
	scanf("%d",&t);
	while(t--)
		{
		int n,m,k;
		scanf("%d",&n);
		scanf("%d",&m);
		scanf("%d",&k);
		if((n==1&&m==1)||(n==2&&m==1)||(n==1&m==2))
			{
			printf("0\n");
			
			}
		else
		if((n==1)||(m==1))
			{
			cout<<k<<endl;
			
			}
		else
			{
			cout<<(k+1)/2<<"\n";
			}
		}
	return 0;
	}
