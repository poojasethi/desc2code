#include<bits/stdc++.h>
# define ll long long 
using namespace std;
int main()
{
	cin.sync_with_stdio(false);
	ll t,n,m,k,i;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n>>m>>k;
		if((n==1 && m <=2)|| (m==1 && n<=2))
		cout<<"0\n";
		else if (n==1 || m==1)
		cout<<k<<endl;
		else 
		cout<<(k+1)/2<<endl;
		
	}
}
