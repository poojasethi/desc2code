#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
	        int r,c,k;
	        r=c=k=1;
	        cin>>r;
	        cin>>c;
	        cin>>k;
	        if(r>c)
	        swap(r,c);
	        if(r==1&&c<=2)
	        {
			cout<<"0"<<endl;
			continue;
			}
	        if(r==1)
		    {
			cout<<k<<endl;
			continue;
			}
		    cout<<(k+1)/2<<endl;
	}
}
