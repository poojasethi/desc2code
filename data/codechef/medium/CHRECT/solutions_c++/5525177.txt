#include <iostream>

using namespace std;

int main()
{
    int t,m,n,k,x,y;
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>k;
        x=max(m,n);
        y=min(m,n);
        if(y==1)
        {
            if(x==1 || x==2)
                cout<<"0\n";
            else
                cout<<k<<"\n";
        }
        else
            cout<<(k+1)/2<<"\n";
    }
    return 0;
}
