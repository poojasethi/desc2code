#include<iostream>
#include<cstdio>
int main()
{
    int ans,n,m,k,t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d",&n,&m,&k);
        ans=0;
        if(n+m==3 || (n==1&&m==1))
            ans=0;
        else if(n==1 || m==1)
            ans=k;
        else
        {
            if(k&1)
                ans=(k+1)/2;
            else
                ans=k/2;
        }
        printf("%d\n",ans);
    }
    return 0;
}
