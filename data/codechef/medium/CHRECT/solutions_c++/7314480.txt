#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<vector>
#include<algorithm>
#include<limits>
#include <stdio.h>
#include<set>
#include<string.h>
#define ll long long
using namespace std;
int main()
{
    int t,n,m,k,res;
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>k;
        if(n==1 and m<=2)
            res=0;
        else if(m==1 and n<=2)
            res=0;
       else if(n==1 or m==1)
            res=k;
       else if(k%2==1)
            res=(k+1)/2;
        else if(k%2==0)
            res=k/2;
        cout<<res<<'\n';
    }
}
