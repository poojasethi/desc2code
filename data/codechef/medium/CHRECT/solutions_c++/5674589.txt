#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,m,temp;
    float k;
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>k;
        if(n>m)
        {
            temp=n;
            n=m;
            m=temp;
        }
            if(n==1)
            {
                if(m==1||m==2)
                {
                    cout<<0<<endl;
                }
                else
                {
                    cout<<k<<endl;
                }
            }
            else
            {
                cout<<ceil(k/2)<<endl;
            }
    }
    return 0;
}
