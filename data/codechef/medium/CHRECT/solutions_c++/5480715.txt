#include<cstdio>
using namespace std;
int main()
{   int t,m,n,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d",&n,&m,&k);
		if((n+m)==2||(n+m)==3)
		{
			printf("0\n");
		}
		else if(m==1||n==1)
		{
			printf("%d\n",k);
		}
		else
		{
			if(k%2==0)
			{
				printf("%d\n",k/2);
			}
			else
			{
				printf("%d\n",k/2+1);
			}
		}
		
	}
	return 0;
}
