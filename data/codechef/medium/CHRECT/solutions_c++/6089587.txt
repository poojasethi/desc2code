#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t; cin>>t;
	while(t-->0)
	{
		int mat[70][70]={};
		int n,m,s=0,k; cin>>n>>m>>k;
		for(int ctr=0;ctr<k;ctr++)
		{
			int i=0,j=0;
			while(1)
			{
				if((i==n-1)&&(j==m-1))
				{
					break;
				}
				if(mat[i+1][j]<=mat[i][j+1])
				{
					if(i==n-1) j++;
					else i++;
					mat[i][j]++;
					continue;
				}
				if(mat[i+1][j]>mat[i][j+1])
				{
					if(j==m-1) i++;
					else j++;
					mat[i][j]++;
				}
			}
		}
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				if((i==n-1)&&(j==m-1)) continue;
				if(s<mat[i][j]) s=mat[i][j];
			}
		}
		cout<<s<<'\n';
	}
}
