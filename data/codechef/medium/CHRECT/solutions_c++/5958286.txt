#include <iostream>

int main(int argc, const char * argv[])
{
    int T,n,m,k;
    
    std::cin>>T;
    for (int i=0; i<T; i++) {
        scanf("%d",&n);
        scanf("%d",&m);
        scanf("%d",&k);
        
        if (n==1 && m==1){
                printf("%d",0);
        }
        else if(n==1 && m==2){
                printf("%d",0);
            }
        
        else if(m==1 && n==2){printf("%d",0);}
        
        else if(m==1 && n>2){
            printf("%d",k);
        }
        
        else if (n==1 && m>2){printf("%d",k);}
        else{
            if (k%2==0) {
                printf("%d",k/2);
            }
            
            else{
                printf("%d",k/2+1);
            }
            
        }
    
        printf("\n");
        
    }
    
    return 0;
}
