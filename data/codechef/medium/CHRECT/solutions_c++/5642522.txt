#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
#define MAX 10001

int main()
{
	int t,n,m,k;
		
	scanf("%d",&t);
	
	while(t--)
	{
		scanf("%d %d %d",&n,&m,&k);
		int tmp=k/2 + k%2;
		
		if(n>1 && m>1)
		printf("%d\n",tmp);
		
		else if( (n==1 && m<=2) || (m==1 && n<=2))
		printf("0\n");
		
		else if( (n==1 && m>2) || (n>2 && m==1) )
		printf("%d\n",k);
	}
	return 0;
}
