#include <cmath>
#include <cstring>
#include <bits/stdc++.h>

using namespace std;

int main(){
	unsigned int i, j, temp, n, m,k,t, rect[70][70];
	scanf("%d\n", &t);
	while(t--){
		scanf("%d%d%d",&n,&m,&k);
		for(i = 0;i<n;i++){
			for(j=0;j<m;j++){	
				rect[i][j]=0;
			}
		}
		
		while(k--){
			i = 0;
			j = 0;
			while(i+j!=n+m-2){
				if(i<n-1 && j<m-1)	{	if(rect[i+1][j]<=rect[i][j+1])	i+=1;
										else j++;
									}
				else if(i == n-1) j++;
				else if(j == m-1) i++;
				rect[i][j]++;
			}
		}
		temp = 0;
		for(i = 0;i<n;i++){
			for(j=0;j<m;j++){
				if(i+j==n+m-2) continue;	
				if(temp<rect[i][j]) 	temp = rect[i][j];
						
			}
		}
		printf("%d\n", temp);
	}
	return 0 ;
}
