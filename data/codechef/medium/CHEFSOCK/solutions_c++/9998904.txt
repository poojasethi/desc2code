#include<stdio.h>

int main()
{
    long long jackcost,sockcost,money;

    scanf("%lld %lld %lld",&jackcost,&sockcost,&money);

    long int curramount;
    curramount=(money-jackcost)/sockcost;

    if(curramount%2==0)
        printf("\nLucky Chef");
    else
        printf("\nUnlucky Chef");
 return 0;


}
