#define ll long long
void check(ll i)
{
  ll sum=0;
  for(int k=1;pow(5,k)<=i;k++)
  {cout<<" op"<<sum<<"  "<<k<<endl;
    sum+=i/pow(5,k);
  }
  cout<<sum;
}
