#include<stdio.h>
 
#define MAXN 5005
#define MAXR 2505
 
#define MOD 1000000007
 
int nCr[MAXN][MAXR];
 
 
int main()
{
	int n,r,m,k,t,M,K,K1;
	long long int val;
 
 
	nCr[0][0] = 0;	
	nCr[1][0] = 1; nCr[1][1] = 1;
	nCr[2][0] = 1; nCr[2][1] = 2;  nCr[2][2] = 1;
	nCr[3][0] = 1; nCr[3][1] = 3;  nCr[3][2] = 3;  nCr[3][3] = 1;
 
	for(n = 4; n<MAXN; n++)
	{
		nCr[n][0] = 1;	
 
		for(r =1; r<=(n/2); r++)
		{
			if(r > n-r-1)
				t = n-r-1;
			else
				t = r;
 
			nCr[n][r] = ((long long)nCr[n-1][t] + (long long)nCr[n-1][r-1]) % MOD;
		}
		
	}
 
 
	while(1)
	{
		scanf("%d %d",&m,&k);
 
		if(!m && !k)
			break;
		
		M = m-2;
		
		if(!(k%2))
		{
				K = (k/2);	  K1 = K-1;
			
			if(K > M-K)
				K = M-K;
 
			if(K1 > M-K1)
				K1 = M-K1;
				
			val = (2 * (long long)nCr[M][K1] * (long long)nCr[M][K]) % MOD ;
 
			printf("%d\n",val);
 
		}
		else
		{
			K =(k-1)/2 ;
			
			if(K > M-K)
				K = M-K;
 
			val = (2 * (long long)nCr[M][K] * (long long)nCr[M][K]) % MOD ;
			
			printf("%d\n", val);
		}
	
	}
 
	return 0;
}
