#include<iostream>
#include<cstdio>
#define ll long long int
#define mod 1000000007
ll fact[5005]={0},infact[5005]={0};
inline ll p(ll a,ll b)
{
    ll temp=1,x=a;
    while(b)
    {
        if(b&1)
            temp=(temp*x)%mod;
        x=(x*x)%mod;
        b>>=1;
    }
    return temp;
}
inline void pre()
{
    ll i;
    fact[0]=1;
    for(i=1;i<5004;i++)
        fact[i]=(fact[i-1]*i)%mod;
    infact[5002]=p(fact[5002],mod-2);
    for(i=5001;i>=0;i--)
        infact[i]=((i+1)*infact[i+1])%mod;
}
inline ll fn(ll n,ll k)
{
    ll temp;
    temp=(((fact[n]*infact[k])%mod)*infact[n-k])%mod;
    return temp;
}
int main()
{
    ll n,k,ans,i;
    pre();
    while(1)
    {
        scanf("%lld %lld",&n,&k);
        if(!n &&!k)
            break;
        ans=(fn(n-2,k/2)*fn(n-2,(k-1)/2))%mod;
        ans=(ans<<1)%mod;
        printf("%lld\n",ans);
    }
    return 0;
}
