#include<cstdio>
using namespace std;
#define lm 1000000007
int C[5010][5010];
int main()
{
     int n,k;
     long long int prod;
    for(int i=0;i<5010;i++)
    {
        for(int j=0;j<=i;j++)
        {
            if(j==0||j==i)
                C[i][j]=1;
            else
                C[i][j]=((C[i-1][j-1])+(C[i-1][j]))%lm;
        }
    }
    while(1)
    {
        scanf("%d %d",&n,&k);
     //   prod=1;
        if(n==0)
            break;
        else
        {
            n=n-2;
            prod=(2*(C[n][k/2]%lm))%lm;
            prod=((prod%lm)*(C[n][((k+1)/2)-1]%lm))%lm;
        }
        printf("%d\n",prod);
    }
}
