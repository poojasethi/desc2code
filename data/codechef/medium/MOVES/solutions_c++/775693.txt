#define DEBUG 1
#define MOD 1000000007
#define MAX 5001

#include <stdio.h>

int f[MAX], finv[MAX];

int p(int x, int y){
  int r;
  if(y==0)
    return 1;
  if(y%2)
    return (((long long unsigned)x*p(x, y-1))%MOD);
  r=p(x, y/2);
  return (((long long unsigned)r*r)%MOD);
}

void precal(void) {
  int i;
  f[0]=1;
  for(i=1; i<MAX; i++)
    f[i]=(((long long unsigned)i*f[i-1])%MOD);
  i--;
  finv[i]=p(f[i],MOD-2);
  for(i--; i>=0; i--)
    finv[i]=(((long long unsigned)(i+1)*finv[i+1])%MOD);
}

int c(int n, int r){
  if(n<r)
    return 0;
  return (((long long unsigned)(((long long unsigned)f[n]*finv[r])%MOD)*finv[n-r])%MOD);
}

int main(){
  int N, K;
  precal();
  while(1) {
    scanf("%d%d", &N, &K);
    if(!N&&!K)
      break;
    printf("%d\n", (int)(((long long unsigned)2*c(N-2, K/2)*c(N-2, (K-1)/2))%MOD));
  }
  return 0;
}
