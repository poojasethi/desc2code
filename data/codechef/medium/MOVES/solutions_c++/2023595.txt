#include<stdio.h>
 
#define MAX 5010
#define MODBY 1000000007
 
typedef long long int lld;
lld fact[MAX],invfact[MAX];
 
int power(int a,int b,int c)
{ 
   lld ans=1,x=a;
   while(b)
   {
      if(b&1) 
      ans=(ans*x)%c;
      x=(x*x)%c;
      b>>=1;
   }                                 
   return (int)ans;
}
 
void precompute()
{   
   int i;   
   fact[0]=1;
   for(i=1;i<MAX;i++)
   fact[i]=(fact[i-1]*i)%MODBY;
   invfact[MAX-1]=power(fact[MAX-1],MODBY-2,MODBY);
   for(i=MAX-2;i>=0;i--)
   invfact[i]=((i+1)*invfact[i+1])%MODBY;
}
 
int C(int n,int r)
{
   return (int)((((fact[n]*invfact[r])%MODBY)*invfact[n-r])%MODBY);
}
 
int main() 
{
   int n,k;
   lld ans;
   precompute();
   while(scanf("%d%d",&n,&k),k)
   {
      ans=C(n-2,k/2);
      ans=ans * C(n-2,(k-1)/2);
      ans=2*ans;
      printf("%d\n",(int)(ans%MODBY));
   }
   return 0;
}
