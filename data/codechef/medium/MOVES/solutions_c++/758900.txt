ll mod = 1000000007;
ll inv(ll a, ll e = mod - 2){
   if(e == 0) return 1;
   ll x = inv(a, e / 2);
   x = x * x % mod;
   if(e % 2 == 1) x = x * a % mod;
   return x;
}

int main(){

   f(i, 0, 5002) I[i] = inv(F[i]);
   int n, k;
   while(scanf("%d %d", &n, &k) && (n || k)){
