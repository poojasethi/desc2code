#include <stdio.h>
#define MOD 1000000007
#define MAXNUM 5000

long long int fact[MAXNUM+2];
long long int ifact[MAXNUM+2];

int getArray(int num, int *array)
{
	int index = -1;
	while(num)
	{
		array[++index] = num&1;
		num>>=1;
	}
	return index;
}

long long int pow(long long int a, int * v, int index)
{
	long long int res = 1;
	while(index>=0)
	{
		res*=res;
		res%=MOD;
		if(v[index--])
		{
			res*=a;
			res%=MOD;
		}
	}
	return res;
}

int main()
{
	int n;
	int k;
	int array[40];
	int index = getArray(MOD-2, array);
	
	fact[0] = 1;
	ifact[0] = 1;
	for(int i = 1; i <= MAXNUM; i++)
	{
		fact[i]=(fact[i-1]*i)%MOD;
		ifact[i]=pow(fact[i], array, index);
	}
	
	scanf("%d %d", &n, &k);
	long long int res;
	while(n || k)
	{
		if(k%2)
		{
			res=(fact[n-2]*ifact[(k-1)>>1])%MOD;
			res*=ifact[(n-2)-((k-1)>>1)];
			res%=MOD;
			res*=res;
			res%=MOD;
			res<<=1;
			res%=MOD;
		}
		else
		{
			res=(fact[n-2]*ifact[k>>1])%MOD;
			res*=ifact[(n-2)-(k>>1)];
			res%=MOD;
			res*=(fact[n-2]*ifact[(k>>1)-1])%MOD;
			res%=MOD;
			res*=ifact[(n-2)-((k>>1)-1)];
			res%=MOD;
			res<<=1;
			res%=MOD;
		}
		printf("%lld\n", res);
		scanf("%d %d", &n, &k);
	}
	return 0;
}
