#include<bits/stdc++.h>
using namespace std;
long long mod=1e9+7;
long long fact[5005];
long long ifact[5005];
long long comb(long long a,long long b)
{
	if(a<b)return 0;
	if(a==b || b==0)return 1;
	else return (((fact[a]*ifact[b])%mod)*ifact[a-b])%mod;
}
main()
{
	long long i,j,k,n;
	fact[0]=ifact[0]=1;
	long long inv[5005];
	inv[1]=1;
	for(i=2;i<5005;i++)
	{
		inv[i]=(mod-(((mod/i)*inv[mod%i])%mod))%mod;
	}
	for(i=1;i<5005;i++)
	{
		fact[i]=(i*fact[i-1])%mod;
		ifact[i]=(inv[i]*ifact[i-1])%mod;
	}
	while(1)
	{
		scanf("%lld %lld",&n,&k);
		if(n==0 && k==0)break;
		printf("%lld\n",(2*((comb(n-2,k/2)*comb(n-2,(k-1)/2))%mod))%mod);
	}
}
