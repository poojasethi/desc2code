#include<stdio.h>
#define M 1000000007
#define lld long long int
using namespace std;
lld f[5001],inf[5001];
lld pow(lld a,lld p)
{
    lld d,ans=1;
    while(p!=0)
    {
               d=p%2;
               p=p/2;
               if(d) ans=(ans*a)%M;
               a=(a*a)%M;
    }
    return ans;
} 
lld comb(int n,int r)
{
    return (f[n]*(pow(((f[n-r]*f[r])%M),M-2)))%M;
}
