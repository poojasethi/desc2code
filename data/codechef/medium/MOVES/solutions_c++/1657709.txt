#include <iostream>
#include <stdio.h>
#define MOD 1000000007

using namespace std;

long long int pow1(int i, int p)
{
    if(p==0) return 1;
    long long int k=pow1(i,p/2);
    if(p & 1==1) return (((i*k)%MOD)*k)%MOD;
    return (k*k)%MOD;
}

int main()
{
    long long int fac[5010],facinv[5010],temp,temp2;
    fac[0]=facinv[0]=fac[1]=facinv[1]=1;
    for(int i=2;i<5003;i++)
    {
            fac[i]=(fac[i-1]*i)%MOD;
            facinv[i]=(facinv[i-1]*pow1(i,MOD-2))%MOD;
            }
    int n,k,l;
    scanf("%d%d",&n,&k);
    while(!(n==0&&k==0))
    {
                        n-=2;
                        l=(k+1)/2-1;
                        k=k/2;
                        temp=(((fac[n]*facinv[k])%MOD)*facinv[n-k])%MOD;
                        temp2=(((fac[n]*facinv[l])%MOD)*facinv[n-l])%MOD;
                        printf("%lld\n",(2*((temp*temp2)%MOD))%MOD);
                        scanf("%d%d",&n,&k);
                        }
    //system("pause");
    return 0;
}
