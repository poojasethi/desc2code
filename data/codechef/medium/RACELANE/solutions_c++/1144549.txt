#include<iostream>
main()
{
      int i,j,arr[10][510],ans[510],t,s,e,l,b,count;
      scanf("%d",&t);
      while(t--)
      {
             for(i=0;i<=9;i++)for(j=0;j<=505;j++)arr[i][j]=1;for(i=0;i<=505;i++)ans[i]=1;   
             scanf("%d",&b);  count=0; 
             while(b--){scanf("%d%d%d",&s,&e,&l);for(i=s;i<=e;i++)arr[l][i]=0;}
             for(i=1;i<=9;i++)for(j=0;j<=505;j++)if(arr[i-1][j]==arr[i][j]&&arr[i][j]==0)ans[j]=0;
             for(i=0;i<=505;i++)if(ans[i]==0)count++;
             printf("%d\n",500-count);
      }
}
