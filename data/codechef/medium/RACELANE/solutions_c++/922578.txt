#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
              int n,ansf=0;
              bool arr[10][501]={0};
             bool ans[501]={0};
              cin>>n;
              int start,end,line;
              while(n--)
              {
                        cin>>start>>end>>line;
                        for(int i=start;i<=end;i++)
                        arr[line][i]=1;
              }
              for(int i=0;i<9;i++)
              {
                      for(int j=1;j<=500;j++)
                      {
                              if(arr[i][j]==1 && arr[i+1][j]==1)
                              ans[j]=1;
                      }
              }
              for(int j=1;j<=500;j++)
              {
                      if(ans[j]==1)
                      ansf++;
              }
              cout<<500-ansf<<endl;
    }
    return 0;
}
