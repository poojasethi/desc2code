#include<iostream>
using namespace std;

int main()
{
	int t,s,e,l,b,m=0,flag;
	int lane[10][500];
	cin>>t;
	while(t--)
	{
		m=0;
		for(int i=0;i<10;i++)
		{
			for(int j=0;j<500;j++)
			{
				lane[i][j]=0;
			}
		}
		cin>>b;
		while(b--)
		{
			cin>>s>>e>>l;
			for(int i =s-1;i<e;i++)
			{
				lane[l][i]=1;
			}
		}
		for(int i=0;i<500;i++)
		{
			flag=0;
			for(int j=0;j<9;j++)
			{
				if(lane[j][i]==1 && lane[j+1][i]==1)
				{
					flag=1;
					break;
				}
			}
			if(flag==0)
			   m++;
		}
		cout<<"\n"<<m;
	}
	return 0;
}
