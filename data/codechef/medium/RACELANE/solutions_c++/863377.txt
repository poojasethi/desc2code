#include<stdio.h>
int line [10][501];
int main()
{
    int t,b,l,s,e,flg,i,j;
    scanf("%d",&t);
    while(t)
    {
            for(i=0;i<=9;i++)
            {
                             for(j=1;j<=500;j++)
                             line[i][j]=0;
            }
            flg=0;
            scanf("%d",&b);
            for(i=1;i<=b;i++)
            {
                             scanf("%d %d %d",&s,&e,&l);
                             for(j=s;j<=e;j++)
                             {
                                           line[l][j]=1;   
                             }
            }
            for(j=1;j<=500;j++)
            for(i=0;i<=8;i++)
            {
                             if(line[i][j]==1&&line[i+1][j]==1)
                             {
                                                                  flg++;
                                                                  break;
                             }
            }
            printf("%d\n",(500-flg));
            t--;
    }   
    return 0;
}
