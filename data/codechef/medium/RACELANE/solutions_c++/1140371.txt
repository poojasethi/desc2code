#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
inline void fastRead(int *a)
{
    register char c=0;
    while (c<33) c=getchar_unlocked();
    *a=0;
    while (c>33)
    {
        *a=*a*10+c-'0';
        c=getchar_unlocked();
    }
}
int tc,s,e,l;
int main()
{
	int i,j;
	fastRead(&tc);
	while(tc--)
	{
		int b;
		int bub[10][510];
		for(i=0;i<=10;i++)
		{
			for(j=0;j<=510;j++)
				bub[i][j]=0;
		}
		scanf("%d",&b);
		for(i=0;i<b;i++)
		{
			fastRead(&s);
			fastRead(&e);
			fastRead(&l);
			for(j=s;j<=e;j++)
			{
				bub[l][j]=1;
			}
		}
		int count=0;
		int track[510]={0};
		for(i=1;i<10;i++)
		{
			for(j=1;j<=500;j++)
			{
				if(bub[i][j]==1 && bub[i-1][j]==1 && track[j]==0) 
				{
					count++;
					track[j]=1;
				}
			}
		}
		printf("%d\n",500-count);
	}
	return(0);
}
