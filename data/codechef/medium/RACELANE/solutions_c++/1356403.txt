#include <iostream>
#include<cstdio>
using namespace std;
int a[15][505],r[505];
int main()
{
    int t,b,s,e,l,i,j,c;
    cin>>t;
    while(t--)
    {
        cin>>b;
        for(i=1;i<=b;i++)
        {
            cin>>s>>e>>l;
            for(j=s;j<=e;j++)
            a[l][j]++;
        }
        c=0;
        for(i=1;i<=9;i++)
        {
            for(j=0;j<=500;j++)
            {
                if(a[i-1][j]>0&&a[i][j]>0)
                r[j]=1;
            }
        }
        for(i=0;i<=500;i++)
        {
            if(r[i]>0)
            c++;
        }
        cout<<500-c<<endl;
        for(i=0;i<=9;i++)
        {
        for(j=0;j<=500;j++)
        {
        a[i][j]=0;
        r[j]=0;
        }
        }

    }
    return 0;
}
