#include<iostream>
using namespace std;
int main()
{
    int a[10][510],i,t,b,s,e,l,cnt=0,ans,j,p;
    cin>>t;
    while(t--)
    {
        cnt=0;
         for(i=0;i<10;i++)
            for(j=0;j<510;j++)
              a[i][j]=0;
        cin>>b;
        while(b--)
        {
            cin>>s>>e>>l;
            for(i=s;i<=e;i++)
            {
                a[l][i]=1;
            }
        }
        for(i=0;i<9;i++)
        {
            for(j=1;j<=500;j++)
            {
                if(a[i][j]==1&&a[i+1][j]==1)
                {
                    cnt++;
                    for(p=i;p<10;p++)
                      a[p][j]=0;
                }
            }

        }
        ans=500-cnt;
        cout<<ans<<endl;

    }
}
