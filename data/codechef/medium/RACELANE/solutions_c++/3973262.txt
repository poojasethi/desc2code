#include<cstdio>
#include<map>
#include<vector>

using namespace std;

int main()
{
vector< map<int,bool> > lanes;
map<int,bool> dummy;
int T,B,S,E,L,count;
scanf("%d",&T);
//for(int i=-1;i<=10;i++)
//	dummy[i]=false;
while(T-->0)
{
	for(int i=0;i<500;i++)
		lanes.push_back(dummy);
	count=0;
	scanf("%d",&B);
	vector<bool> valid(500,true);
	while(B-->0)
	{
		scanf("%d%d%d",&S,&E,&L);
		for(int i=S;i<=E;i++)
		{
			if(valid[i-1])
			{
//				printf("check:%d\n",lanes[i-1][L-1]);	
				if(lanes[i-1][L-1] || lanes[i-1][L+1])
				{
					valid[i-1]=false;
					count++;	
				}
				else
				{
					lanes[i-1][L]=true;
				}
			}
		}
	}
	printf("%d\n",500-count);
	lanes.clear();
}
}
