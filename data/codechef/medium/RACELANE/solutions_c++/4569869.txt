#include<stdio.h>
using namespace std;
int main()
{
    int t,b,arr[10][501],status[501],s,e,i,l,j;
    scanf("%d",&t);
    while(t--)
    {
        for(i=0;i<10;i++)
            for(j=0;j<502;j++)
            arr[i][j]=0;
         for(j=0;j<502;j++)
            status[j]=0;
        scanf("%d",&b);
        for(i=0;i<b;i++)
        {
            scanf("%d%d%d",&s,&e,&l);
            for(j=s;j<e+1;j++)
                arr[l][j]=1;


        }
        int c=0;
        for(i=0;i<9;i++)
        {
            for(j=1;j<501;j++)
            {
                if(!status[j]&&arr[i][j]&&arr[i+1][j])
                {
                    c++;
                    status[j]=1;
                }
            }
        }
        //cout<<c<<endl
        printf("%d\n",500-c);
    }
    return 0;

}
