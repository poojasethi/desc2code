#include <cstdio>
#include <memory.h>
using namespace std;

bool track[10][501];

int main(){
  int t,b,s,e,l;

#ifndef ONLINE_JUDGE
  freopen("in.txt","r",stdin);
#endif
  scanf("%d",&t);
  while(t--){
    memset(track,0,sizeof(track));
    scanf("%d",&b);
    while(b--){
      scanf("%d %d %d",&s,&e,&l);
      for(;s<=e;s++) track[l][s]=true;
    }
    e=500;
    for(s=1;s<501;s++){
      for(l=0;l<9;l++){
        if(track[l][s] && track[l+1][s]){
          e--;
          break;
        }
      }
    }
    printf("%d\n",e);
  }
  return 0;
}
