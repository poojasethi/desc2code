#include<iostream>
using namespace std;

int main()
{
	int t; cin>>t;
	while(t--)
	{	int n[10][501]={0},m=500;
		int b; cin>>b;
		for(int i=0;i<b;i++)
		{	int s,e,l; cin>>s>>e>>l;
			for(int j=s;j<=e;j++)
			n[l][j]=1;	}

		for(int i=1;i<=500;i++)
		{	int fl=1;
			for(int j=0;j<9;j++)
			{	if(fl)
				if(n[j][i] && n[j+1][i])
				{	m--; fl=0;	}	}	}
		cout<<m<<"\n";	}
	return 0;
}
