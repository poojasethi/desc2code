#include<bits/stdc++.h>
using namespace std;
int main() {
	int t,n,i,j,a,b,c;
	cin >> t;
	while(t--) {
		cin >> n;
		int p[505][10];
		memset(p,1,sizeof(p));
		for(i=0;i<n;i++) {
			cin >> a >> b >> c;
			for(j=a;j<=b;j++)
			p[j][c] = 0;
		}
		int c = 0;
		for(i=1;i<=500;i++) {
			for(j=0;j<9;j++) {
				if((p[i][j] == 0 && p[i][j+1] ==0)) {
					c++;
					break;
				}
			}
		}
		cout << 500 - c <<"\n";
	}
}
