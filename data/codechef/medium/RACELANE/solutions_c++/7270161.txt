#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
using namespace std;
 int main()
 {
     int t,c,ans,a,b,flag,flag1,n,i,j;
     int A[501][12];
     scanf("%d",&t);
     while(t>0)
     {
         t--;
         scanf("%d",&n);
         for(i=0;i<=500;i++)
         {
             for(j=0;j<11;j++)
                A[i][j]=0;
         }
         for(i=0;i<n;i++)
         {
             scanf("%d %d %d",&a,&b,&c);
             for(j=a;j<=b;j++)
             {
                 A[j][c]=1;
             }
         }
         c=0;
         for(i=1;i<=500;i++)
         {
             for(j=0;j<9;j++)
             {
                 if(A[i][j]==1&&A[i][j+1]==1)
                 {
                     c++;
                     break;
                 }
             }
         }
         ans=500-c;
         printf("%d\n",ans);
     }
 }
