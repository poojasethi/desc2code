#include<stdio.h>

main()

{

	int t,b;

	int i,s,e,l,j,ctr ;

	scanf("%d",&t);

	bool ispainted[501][10];

	

	while(t--)

	{

		scanf("%d",&b);

		for(i=0;i<501;i++)

		  for(j=0;j<10;j++)

		    ispainted[i][j]=false;

		    ctr=0;

		while(b--)

		{

			scanf("%d%d%d",&s,&e,&l);

			for(i=s;i<=e;i++)

		    {

			   ispainted[i][l]=true;	

		    }

	       }

	    for(i=1;i<501;i++)

	     {for(j=0;j<9;j++)

	       {

				if(ispainted[i][j] && ispainted[i][j+1])

				  {

					ctr+=1;

					break;	

				  }

				

			}

		 }

		 printf("%d\n",500-ctr);

    }

}
