#include <bits/stdc++.h>
int main()
{
	int tc;
	int i,j;
	int s,e,l;
	scanf("%d",&tc);
	while(tc--)
	{
		int b;
		int bub[10][510];
		for(i=0;i<=10;i++)
		{
			for(j=0;j<=510;j++)
			{
				bub[i][j]=0;
			}
		}
		scanf("%d",&b);
		for(i=0;i<b;i++)
		{
			scanf("%d%d%d",&s,&e,&l);
			for(j=s;j<=e;j++)
			{
				bub[l][j]=1;
			}
		}
		int count=0;
		int track[510];
		for(i=1;i<10;i++)
		{
			for(j=1;j<=500;j++)
			{
				if(bub[i][j]==1 && bub[i-1][j]==1 && track[j]==0) 
				{
					count++;
					track[j]=1;
				}
			}
		}
		printf("%d\n",500-count);
	}
	return 0;
}
