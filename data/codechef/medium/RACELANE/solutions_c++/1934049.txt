#include <iostream>
#include <cstring>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while(t--) {
        int n, flag, i, j, x, y, l;

        cin >> n;

        bool ar[501][10];
        int b[501];

        memset(b, 0, sizeof(b));
        memset(ar, 1, sizeof(ar));

        int count = 500;

        for(i = 0; i < n; i++) {
            cin >> x >> y >> l;
            for(j = x; j <= y; j++) {
                ar[j][l] = 0;
                flag = 0;
                if(l - 1 >= 0) {
                    if(!ar[j][l-1]) {
                            flag = 1;
                            b[j]++;
                    }
                }
                if(flag && b[j] == 1) {
                        count--;
                        continue;
                }
                if(l + 1 <= 9) {
                    if(!ar[j][l+1]) {
                            flag = 1;
                            b[j]++;
                    }
                }
                if(flag && b[j] == 1) count--;
            }
        }

        cout << count << endl;

    }
}
