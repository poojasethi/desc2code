#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int c = n/26;
		int m = n%26;
		for(int i=0;i<c;i++)
		{
			for(int i=0;i<26;i++)
			{
				cout<<(char)(i+97);
			}
		}
		for(int i=0;i<m;i++)
		{
			cout<<(char)(i+97);
		}
		cout<<"\n";
	}
}
