#include<cstdio>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        int count =97;
        for(int i=1;i<=n;i++){
            char a=count;
            printf("%c",a);
            count++;
            if(count==123){
                count=97;
            }
        }
        printf("\n");
    }
    return 0;
}
