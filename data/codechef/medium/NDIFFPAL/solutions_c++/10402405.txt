#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#define MAX	 10000

using namespace std;
int T[MAX]; int l;

void get_triangular(){
	T[0] = l = 0;
	for(int i = 1;i <= MAX;i++){
		T[i] = T[i - 1] + i;
		if(T[i] >= MAX){
			l = i + 1;
			break;
		}
	}
}

int main(){
	int t,n;
	get_triangular();
	
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		int pc = 0;
		int r = l - 1;
		while(n != 0){
			while(r > 0 && T[r] > n) r--;
			for(int i = 0;i < r;i++) printf("%c",(char)(pc + 'a'));
			n -= T[r];
			pc = (pc + 1) % 26;
		}
		printf("\n");
	}
	return 0;
}
