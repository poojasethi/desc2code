#include <iostream>
#include <cstdio>

int main() {
  int t, n;
  scanf("%d", &t);

  for (int i= 0; i < t; ++i) {
    scanf("%d", &n);

    for (int j = 0; j < n; ++j) {
      int m = j % 3;
      char ch = (m == 0) ? 'a' : ((m == 1)  ? 'b' : 'c');
      printf("%c", ch);
    }
    printf("\n");
  }

  return 0;
}
