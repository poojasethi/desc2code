#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s="abcdefghijklmnopqrstuvwxyz";
        long long int n;
        cin>>n;
        int l=s.length();
        for(int i=0;i<n/l;i++)
        {
            cout<<s;
        }
        for(int i=0;i<n%l;i++)
            cout<<s[i];
        cout<<"\n";
    }
    return 0;
}
