#include<stdio.h>
int main()
{
    int t,n;
    char alp[27]="abcdefghijklmnopqrstuvwxyz";
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int t1=n/26,z=n%26;
        for(int i=0;i<t1;i++)
            printf("%s",alp);
        for(int i=0;i<z;i++)
            printf("%c",alp[i]);
        printf("\n");
    }
    return 0;
}
