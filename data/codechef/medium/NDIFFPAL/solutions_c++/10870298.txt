#include <iostream>
using namespace std;
#define ll long long 
int main() 
{
  std::ios::sync_with_stdio(false);
	ll t,n;
	
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=1;i<=n;i++)
		{
			if((i+2)%3==0)
				cout<<"a";
			else if((i+1)%3==0)
				cout<<"b";
			else
				cout<<"c";
		}
		cout<<endl;
	}	
	return 0;


}
