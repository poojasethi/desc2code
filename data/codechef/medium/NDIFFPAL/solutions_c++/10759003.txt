#include<iostream>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        char c='a';
        cin>>n;
        while(n>0)
        {
            cout<<c;
            c++;
            n--;
            if(c>'z')
            c-=26;
        }
        cout<<endl;
    }
    return 0;
}
