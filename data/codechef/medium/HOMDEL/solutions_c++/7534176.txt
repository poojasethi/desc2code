#include<stdio.h>
int a[251][251],dis[251][251];

void floyd(int n)
{
    int i,j,k=0;
    for(k=0;k<n;k++)
    {
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                if(dis[i][k]+dis[k][j]<dis[i][j])
                    dis[i][j]=dis[i][k]+dis[k][j];
            }
        }
    }
}
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
            {
                scanf("%d",&a[i][j]);
                dis[i][j]=a[i][j];
            }
    }
    int m;
    scanf("%d",&m);
    floyd(n);
    while(m--)
    {
        int s,g,d;
        long int ans,diff;
        scanf("%d%d%d",&s,&g,&d);
        ans = dis[s][g]+dis[g][d];
        diff = ans - dis[s][d];
        printf("%ld %ld\n",ans,diff);
    }
    return 0;
}
