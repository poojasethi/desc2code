#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long graph[300][300];
	long long dist[300][300];
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			scanf("%lld",&graph[i][j]);
			dist[i][j] = graph[i][j];
		}
	}
	for(int i=1;i<=n;i++)
	{
		for(int l=1;l<=n;l++)
		{
			for(int m=1;m<=n;m++)
			{
				if(dist[l][i] + dist[i][m]<dist[l][m])
				{
					dist[l][m] = dist[l][i] + dist[i][m];
				}
			}
		}
	}
	int m;
	scanf("%d",&m);
	while(m--)
	{
		int s,g,d;
		scanf("%d %d %d",&s,&g,&d);
		s++;
		g++;
		d++;
		long long tim = dist[s][g] + dist[g][d];
		long long tim2 = dist[s][d];
		if(tim2>tim)
		{
			tim2 = tim;
		}
		printf("%lld %lld\n",tim,tim-tim2);

	}
}
