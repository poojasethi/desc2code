    #include <bits/stdc++.h>
     
    using namespace std;
     
    int a[251][251];
     
    int main()
    {
    	int n;
    	scanf("%d",&n);
    	for (int i = 0; i < n; ++i)
    	{
    		for(int j = 0; j < n; ++j)
    			scanf("%d",&a[i][j]);
    	}
    	for(int i = 0; i < n; ++i)
    	{
    		for(int j = 0; j < n; ++j)
    		{
    			for(int k = 0; k < n; ++k)
    			{
    				if(a[i][k] + a[j][i] < a[j][k])
    					a[j][k] = a[i][k] + a[j][i];
    			}
    		}
    	}
     
    	int q;
    	scanf("%d",&q);
    	while(q--)
    	{
    		int s,g,d;
    		scanf("%d %d %d",&s,&g,&d);
    		printf("%d %d\n", a[s][g] + a[g][d], (a[s][g] + a[g][d] - a[s][d]));
    	}
     
    	return 0;
    }
