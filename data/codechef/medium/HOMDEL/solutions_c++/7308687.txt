#include<bits/stdc++.h>
using namespace std;
#define ll long long


int main()
{
    int n,t[251][251],i,j,k,m,s,g,d;
    scanf("%d",&n);
    
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    scanf("%d",&t[i][j]);
    
    for(k=0;k<n;k++)
    for(i=0;i<n;i++)
    for(j=0;j<n;j++)
    {
    	if(t[i][k]+t[k][j]<t[i][j])
    	t[i][j]=t[i][k] + t[k][j];
    }
    
    scanf("%d",&m);
    
    while(m--)
    {
    	scanf("%d %d %d",&s,&g,&d);
    	int t1=t[s][g] + t[g][d];
    	
    	int t2=t1-t[s][d];
    	
    	printf("%d %d\n",t1,t2);
    }

    return 0;
}
