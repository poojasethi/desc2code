#include<bits/stdc++.h>
#define INF 100005
#define inp(x) scanf("%d",&x)
#define print(x) printf("%d",x)

using namespace std;
int main()
{
    int N,M,S,G,D;
    inp(N); 
    int adj[N][N];
    
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            inp(adj[i][j]);
        }
    }
    
    for(int k=0;k<N;k++)
    {
        for(int i=0;i<N;i++)
        {
            for(int j=0;j<N;j++)
            {
                adj[i][j]=min(adj[i][j],adj[i][k]+adj[k][j]);
            }
        }
    }
    
    inp(M);
    while(M--)
    {
        inp(S);
        inp(G);
        inp(D);
        printf("%d %d\n",adj[S][G]+adj[G][D],adj[S][G]+adj[G][D]-adj[S][D]);
    }
   
    
    return 0;
}
