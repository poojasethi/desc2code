int a[251][251];
void compute(int a[][251],int n)
{
    int i,j,k,dist[n][n];
    for (i = 0; i < n; i++)
        for (j = 0; j < n; j++)


            dist[i][j] = a[i][j];

    for(k=0;k<n;k++)
        {
        for(j=0;j<n;j++)
          {
            for(i=0;i<n;i++)
            {


                if(dist[i][j]>dist[i][k]+dist[k][j])
               {
                   dist[i][j]=dist[i][k]+dist[k][j];
               }
        }
    }
}
