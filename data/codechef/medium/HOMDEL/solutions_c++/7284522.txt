#include <iostream>
#include <string>
#include<map>
#include <cstddef>
#include<utility>
#include<string.h>
using namespace std;
int main()
{
    int N,M,S,G,D;
    cin>>N;
    int T[N][N];
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
            cin>>T[i][j];
    }
    for(int k=0;k<N;k++)
      for(int i=0;i<N;i++)
        for(int j=0;j<N;j++)
    {
      if(T[i][k]+T[k][j]<T[i][j])
            T[i][j]=T[i][k]+T[k][j];
    }

    cin>>M;
    while(M--)
    {
        cin>>S>>G>>D;
        long int sum=T[S][G]+T[G][D];
        long int diff=sum-T[S][D];
        cout<<sum<<" "<<diff<<endl;
    }
}
