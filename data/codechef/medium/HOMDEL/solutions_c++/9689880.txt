#include<bits/stdc++.h>
using namespace std;
int A[500][500];
int main(){
	int N;
	cin >> N;
	for(int i = 0;i < N;++i){
		for(int j = 0;j < N;++j){
			cin >> A[i][j];
		}
	}	
	for(int k = 0;k < N;++k){
		for(int i = 0;i < N;++i){
			for(int j = 0;j < N;++j){
				A[i][j] = min(A[i][j], A[i][k] + A[k][j]);
			}
		}
	}
	int M;
	cin >> M;
	while(M--){
		int S, G, D;
		cin >> S >> G >> D;
		cout << A[S][G] + A[G][D] << " " << A[S][G] + A[G][D] - A[S][D] << "\n";
	}
	return 0;
}
