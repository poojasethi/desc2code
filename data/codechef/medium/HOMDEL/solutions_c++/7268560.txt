#include<bits/stdc++.h>
#define pi pair<int,int>
#define MAXN 260
using namespace std;
int n,m,T[MAXN][MAXN],s,g,d,sgd,sd,ts;

void all_pair_shortest_path()
{
	for(int k=0;k<n;k++)
	{
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(T[i][k]+T[k][j]<T[i][j])
				T[i][j]=T[i][k]+T[k][j];
			}
		}
	}
}
int main()
{
	ios_base::sync_with_stdio(0);
	cin>>n;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>T[i][j];
		}
	}
	all_pair_shortest_path();
	cin>>m;
	while(m--)
	{
		cin>>s>>g>>d;
		sgd=T[s][g]+T[g][d];
		sd=T[s][d];
		if(sd<=sgd)
		ts=sgd-sd;
		else
		ts=0;
		cout<<sgd<<" "<<ts<<"\n";
	}
}
