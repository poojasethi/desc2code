#include<stdio.h>
#include<algorithm>
using namespace std;
int mat[250][250];
int inline input()
{
int temp=0;
char c = 0;
while(c<33)
c = getchar_unlocked();
while(c>33)
{
temp = temp * 10 + c -'0';
c = getchar_unlocked();
}
return temp;
}
int main()
{
      int n,m,a,b,c,i,j,k;n = input();
      for(i=0;i<n;i++)for(j=0;j<n;j++)mat[i][j]  = input();
      for(int k=0;k<n;k++)
      for(int j=0;j<n;j++)
      for(int i=0;i<n;i++)
      mat[i][j] = min((mat[i][k]+mat[k][j]),mat[i][j]);
      scanf("%d",&m);
      while(m--)
      {
               a = input(); b = input(); c = input();
               printf("%d %d\n",mat[a][b]+mat[b][c],mat[a][b]+mat[b][c]-mat[a][c]);
                //printf("%d %d\n",mat[a][b]+mat[b][c],mat[a][b]+mat[b][c]-mat[a][c]);
      }
      return 0;
}
