#include<iostream>
#include<stdio.h>
int arr[250][250],dist[250][250];
int main()
{
    int n,i,j,k,m,s,g,d;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            scanf("%d",&arr[i][j]),dist[i][j]=arr[i][j];
    for(k=0;k<n;k++)
        for(i=0;i<n;i++)
            for(j=0;j<n;j++)
                if(dist[i][k]+dist[k][j]<dist[i][j])
                    dist[i][j]=dist[i][k]+dist[k][j];
    scanf("%d",&m);
    while(m--)
    {
        scanf("%d %d %d",&s,&g,&d);
        printf("%d %d\n",dist[s][g]+dist[g][d],dist[s][g]+dist[g][d]-dist[s][d]);
    }
    return 0;
}
