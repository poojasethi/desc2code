#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int n,m,i,j,k,diff,temp,s,d,g;
    scanf("%d",&n);
    int a[n][n];
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        {
            scanf("%d",&a[i][j]);



        }
        for(k=0;k<n;k++)
        {
            for(i=0;i<n;i++)
                for(j=0;j<n;j++)
                a[i][j]=min(a[i][j],a[i][k]+a[k][j]);

        }

        scanf("%d",&m);
        for(i=0;i<m;i++)
        {
              scanf("%d %d %d",&s,&g,&d);
              temp=a[s][g]+a[g][d];
              diff=temp-a[s][d];

              printf("%d %d\n",temp,diff);



        }




}
