#include<iostream>
using namespace std;
int arr[250][250];
int a[250][250];
void floyd(int n)
{int i,j,k;
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
         {
			a[i][j]=arr[i][j];
			if(j==i)
				a[i][j]=0;
	   	}
		for(k=0;k<n;k++)
	       for(i=0;i<n;i++)
	        	for(j=0;j<n;j++)
				     a[i][j]=min(a[i][j],a[i][k]+a[k][j]);

	return;
}

int main()
{ int n,m,i,j,k,s,g,d;
	scanf("%d",&n);
	
	for(i=0;i<n;i++)
	for(j=0;j<n;j++)
    scanf("%d",&arr[i][j]);

     scanf("%d",&m);
     floyd(n);
     
     for(k=0;k<m;k++)
     {
        scanf("%d %d %d",&s,&g,&d);
     	int t1=a[s][g]+a[g][d];
		int t2=a[s][d];

		printf("%d %d\n",t1,t1-t2);
     }
     
     return 0;
}
