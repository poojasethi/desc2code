#include<bits/stdc++.h>
using namespace std;
int main()
{

    int n;
    cin>>n;

    int A[n+1][n+1];
    int i,j,k;

    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        cin>>A[i][j];

    for(k=0;k<n;k++)
    {
        for(i=0;i<n;i++)
            for(j=0;j<n;j++)
        {



            A[i][j]=min(A[i][j],A[i][k]+A[k][j]);
        }
    }

    int m;
    cin>>m;
    while(m--)
    {

        int s,g,d;
        cin>>s>>g>>d;
 int u=A[s][d];
 int y=A[s][g]+A[g][d];
 printf("%d %d\n",y,y-u);





    }
}
