#include<bits/stdc++.h>
using namespace std;
int n;
long int a[252][252];
long int dist[252][252];
void floydWarshall()
{
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
            dist[i][j]=a[i][j];
    }
    for(int k=0;k<n;k++)
    {
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(dist[i][k]+dist[k][j]<dist[i][j])
                    dist[i][j]=dist[i][k]+dist[k][j];
            }
        }
    }
    return ;
}
int main()
{
    int m,s,g,d;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            cin>>a[i][j];
        }
    }
    floydWarshall();
    cin>>m;
    while(m--)
    {
        cin>>s>>g>>d;
        long int x=dist[s][g]+dist[g][d];
        long int y=dist[s][d];
        cout<<x<<" "<<(x-y)<<endl;
    }
    return 0;
}
