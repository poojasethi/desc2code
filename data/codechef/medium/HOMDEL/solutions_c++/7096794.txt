#include<bits/stdc++.h>
using namespace std;
int main() {
	long long int k,t,n,i,j;
	scanf("%lld",&n);
	long long a[n][n];
	for(i=0;i<n;i++) {
		for(j=0;j<n;j++)
		scanf("%lld",&a[i][j]);
	}
	long long dis[n][n];
	for(i=0;i<n;i++) {
		for(j=0;j<n;j++)
		dis[i][j] = a[i][j];
	}
	for(k=0;k<n;k++) {
		for(i=0;i<n;i++) {
			for(j=0;j<n;j++) {
				if(dis[i][k] + dis[k][j] < dis[i][j])
				dis[i][j] = dis[i][k] + dis[k][j];
			}
		}
	}
	long long q,s,g,d;
	scanf("%lld",&q);
	for(i=0;i<q;i++) {
		scanf("%lld%lld%lld",&s,&g,&d);
		printf("%lld %lld\n",dis[s][g] + dis[g][d],(dis[s][g] + dis[g][d] - dis[s][d])); 
	}
	return 0;
}
