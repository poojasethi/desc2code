#include<iostream>
using namespace std;
int main(){
	int i,j,k,t,n;
	int s,g,d;
	cin>>t;
	int a[t][t];
	for(i=0;i<t;i++){
		for(j=0;j<t;j++){
			cin>>a[i][j];
		}
	}
	for(i=0;i<t;i++){
		for(j=0;j<t;j++){
			for(k=0;k<t;k++){
				if(a[j][k] > a[j][i] + a[i][k]){
					a[j][k] = a[j][i] + a[i][k];
				}
			}
		}
	}
	cin>>n;
	for(i=0;i<n;i++){
		cin>>s>>g>>d;
		cout<<a[s][g] + a[g][d]<<" ";
		cout<<a[s][g] + a[g][d] - a[s][d]<<"\n";
	}
	return 0;
}
