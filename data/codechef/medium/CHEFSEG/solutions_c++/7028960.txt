#include <bits/stdc++.h>

using namespace std;

#define pb push_back
#define mp make_pair
#define ll long long int

ll power(ll a, ll b)
{
    if (b == 0)
        return 1;
    ll x = power(a, b/2);
    if (b % 2)
        return x*x*a;
    else
        return x*x;
}

int main()
{
    int t;
    scanf("%d", &t);
    ll x, k, cnt, l, p;
    double n;
    while (t--) {
        scanf("%lld%lld", &x, &k);
        cnt = 0;
        n = (double)x;
        p = k;
        while (k != 0) {
            k /= 2;
            cnt++;
        }
        l = power(2, cnt-1);
        n = (((double)(1+2*(p-l)))*n)/(2.0*((double)l));
        printf("%.6lf\n", n);
    }
    return 0;
}
