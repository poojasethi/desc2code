#include<stdio.h>
#include<math.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        double p;
        long long no;
        scanf("%lf%lld",&p,&no);
        int k=log2(no);
        long long s=pow(2,k);
        double t=p/s;
        no-=s;
        printf("%.15lf\n",t*no+t/2);
    }
}
