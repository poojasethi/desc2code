#include<stdio.h>
#include<math.h>
int main()
{
	long long int t;
	scanf("%lld",&t);
	while(t--)
	{
		long long int i,k,x;
	    double l,c,d=1;
		scanf("%lld%lld",&x,&k);
		for(i=0;i<41;i++)
		{
			if(k<pow(2,i))
			break;
		}	
	 
	     
	    l=k-pow(2,(i-1));

	}

}
