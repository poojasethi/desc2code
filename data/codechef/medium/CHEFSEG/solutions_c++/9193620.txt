#include<stdio.h>
#include<stdlib.h>
 
long long int getLog(long long int n)
{
	long long int count = 0;
	while(n>1)
	{
		n = n/2;
		count++;
	}
	return count;
}
 
long long int power2(long long int n)
{
	
	long long int temp ;
	if(n==0)
		return 1;
	if(n==1)
		return 2;
	else
	{
		temp = power2(n/2);
		if(n%2)
			temp = temp*temp*2;
		else
			temp = temp*temp;
	}
	return temp;
}
 
int main()
{
	int test;
	long long x,k,temp1;
	double ans,temp2;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%lld%lld",&x,&k);
		long long int h;
		if(k==1)
		{
			ans = x/2;
			printf("%lf\n",ans);
		}
		else
		{
			h = getLog(k)+1;
			temp1 = power2(h);
			temp2 =((double)x)/temp1;
			temp1 = temp1/2;
			if(k==temp1)
			{
				ans = temp2;
			}
			else
			{
				ans = temp2 + (k-temp1)*temp2*2;
			}
			printf("%lf\n",ans);	
		}
	}
	return 0;
}
