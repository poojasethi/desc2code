#include<stdio.h>
int main()
{
	double res,pow[45],  k, div, st, x;
	long long int i,t;
	pow[1] = 1;
	for (i = 2; i <= 41; i++)
	{
		pow[i] = (pow[i - 1] + 1) * 2;
		pow[i]--;
	}
	scanf("%lld", &t);
	while (t--)
	{
		scanf("%lf%lf", &x, &k);
		for (i = 1; i <= 41; i++)
		{
			if (k <= pow[i])
				break;
		}
		div = pow[i] + 1;
		if(i>1)
		st = pow[i - 1] + 1;
		if (k > 1)
			res = (((2 * (k - st + 1)) - 1)*x) / div;
		else
			res = x / 2;
		printf("%lf\n", res);
	}
	getchar();
	getchar();
	return 0;
}
