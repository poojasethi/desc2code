ill power(ill x, ill y)
{
    if( y == 0)
        return 1;
    else if (y%2 == 0)
        return power(x, y/2)*power(x, y/2);
    else
        return x*power(x, y/2)*power(x, y/2);

}

int main() {
	input();
	int t;
	ill x, k;
	ill sum[44], value[44];
	sum[0] = 0;
	value[0] = 1;
	for(int i = 1; i < 43; i++) {
		value[i] = power(2, i);
		sum[i] = sum[i-1] + value[i-1];
	}
 	cin >> t;
 	while(t--) {
 		cin >> x >> k;
 		if(k == 1){

 					double d = (double(x)/(double)value[i+1]);
 					ill nth = 1 + ((diff-1)*2);
 					double ans = d*nth;
 					printf("%lf\n", ans);
 					break;
 				}
 			}
 		}
 	}

}
