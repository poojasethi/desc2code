#include<iostream>
#include<math.h>
#include<stdio.h>
using namespace std;
int main()
{
	long long int t,x,k,m,i=0;
     double z,z1;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&x,&k);
		i=0;
		while(1)
		{
			if((k/pow(2,i))>=1)
			i++;
			else
			{
				m=i;
				break;
			}
		}
		//cout<<"m="<<m<<"\n";
		z=x/pow(2,m);
		//cout<<"z="<<z<<"\n";
		z1=(1+(k-pow(2,m-1))*2)*z;
		printf("%lf\n",z1);
	}
	return 0;
}
