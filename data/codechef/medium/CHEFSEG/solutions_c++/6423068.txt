bool double_is_int(long double trouble) {
   long double absolute = abs( trouble );
   return absolute == floor(absolute);
}
long long int fastPower(long long base,long long power)
{
        long long result=1;
        while (power>0)
        {
                if (power%2==1)
                      {
                        result = (result*base);
                      }
                base = (base*base);
                power/=2;
        }
        return result;
}
