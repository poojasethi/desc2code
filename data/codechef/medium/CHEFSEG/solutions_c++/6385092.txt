#include<iostream>
#include<bits/stdc++.h>
using namespace std;

long long int arr[50];

void dp()
{
	for(int i=0;i<45;i++)
		arr[i]=pow(2,i);
}

int main()
{
	int t;
	cin>>t;
	dp();
	
	while(t--)
	{
		long long int l,k;
		cin>>l>>k;
		double ans;
		int i;
		for(i=0;i<45;i++)
			if(arr[i]<=k && arr[i+1]>k)
				break;
		//cout<<i<<"\n\n\n";
		
		ans=k-arr[i]+1;
		ans=ans*2;
		ans--;
		ans/=arr[i+1];
		ans*=l;
		printf("%.6lf\n",ans);
		
	}
	
	return 0;
}
