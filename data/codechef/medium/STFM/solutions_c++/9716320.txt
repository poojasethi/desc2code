#define N 10000010

ll n,m;
ll freq[N];

ll mul(ll a,ll b)
{
    ll c = (ll)(a)*(ll)(b);
    c%=m;
    while(c < 0) c+=m;
    return c;
}

ll add(ll a,ll b) {
    ll c = (ll)(a) + (ll)(b);
    while(c >= m) c-=m;
    return c;
}

void calc() {
    freq[0]=1;
    for(int i=1;i<=m+2;i++) {
        freq[i] = mul(freq[i-1],i);
    }
}

ll get_x(ll val) {
    ll by,c;
    if(val&1) {
        by = (val+1)>>1;
        by%=m;
        if(by < 0) by+=m;

        ll by1 = val%m;
        if(by1 < 0) by1+=m;

        c = (((ll)by1*(ll)by1)%m)*(ll)(by);
        c%=m;
        if(c < 0) c+=m;
    }
    else {
        by = (val)>>1;
        by%=m;
        if(by < 0) by+=m;

        ll by1 = val%m;
        if(by1 < 0) by1+=m;

        c = (((ll)by1*(ll)(by1+1))%m)*(ll)(by);
        c%=m;
        if(c < 0) c+=m;
    }
    

    return c;
}

int main(int argc, char const *argv[])
{
    sc(n); sc(m);

    ll a[n+3];

    for(int i=0;i<n;i++) {
        scanf("%lld",&a[i]);
    }

    calc();

    ll ans = 0;
    for(int i=0;i<n;i++) {
        ll by = (a[i]+1) >= m ? -1 : freq[a[i]+1]-1;
        ans = add(ans , add(by , get_x(a[i])));
    }   

    printf("%lld\n",ans);
    return 0;
}
