##Objective: calculate huge powers of two modulus 1000000007 efficiently

def main():

    cases = int(raw_input())
    base = 2
    modulus = 1000000007
    
    for i in range(cases):
        exponent = int(bin(int(raw_input()))[2:])
        print pow(pow(base,exponent,modulus),2,modulus)
        
main()
