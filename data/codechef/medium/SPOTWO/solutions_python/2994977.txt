
has = {}

def rec(cur_p):
    if cur_p == 0:
            return 1;
    elif cur_p == 1:
            return 2
    val = has.get(cur_p)
    if val:
    	return val
    ret = rec(cur_p/2)
    ret = ret*ret
    ret = ret % 1000000007
    if cur_p %2:
            ret = (ret *2 ) % 1000000007
    has[cur_p] = ret
    return ret
