#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;

long long mod_exp(unsigned long long a, unsigned long long b)
{
	if(b == 0)
                return 1;
	long long x = mod_exp(a, b/2);
	x=(x*x)%mod;
	if(b%2==1)
                x=(x*a)%mod;
	return x;
}

unsigned long long tobinary(long long n)
{
        unsigned long long sum=0,prod=1;
        while(n>0)
        {
                sum=(n%2)*prod+sum;
                prod=prod*10;
                n=n/2;
        }
        return sum;
}

int main()
{
        unsigned long long d;
        long long val,n,t;
        scanf("%lld",&t);

        while(t--)
        {
                scanf("%lld",&n);
                d = tobinary(n);
                val = mod_exp(2,d);
                val = ((val % mod) * (val % mod)) % mod;
                printf("%lld\n",val);
        }
        return 0;
}
