#include<bits/stdc++.h>
#define M 1000000007
using namespace std;
typedef long long ll;
ll s[25],j;
ll powmod(ll a,ll b,ll MOD){
   ll x=1;
   while(b){
      if(b & 1)
         x=(x*a)%MOD;
      a=(a*a)%MOD;
      b>>=1;
   }
   return x%MOD;   
}   
void tobin(ll n){
   ll d=0;
   while(n){
     s[j]=n%2;
     n/=2;
     j++;
   }
}
int main(){
  ll p,t,n,i,d; 
  scanf("%lld",&t);
  while(t--){
     scanf("%lld",&n);
     j=0; 
     tobin(n);
     p=1;
     for(i=0;i<j;i++){
       if(s[i]==1)
         d=powmod(10,i,M-1);
       else
         d=0;  
       p=(p*powmod(4,d,M))%M;
     }
     printf("%lld\n",p);
  }
  return 0;
}
