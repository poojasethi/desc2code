#include<bits/stdc++.h>
using namespace std;
int d[25],w;
unsigned long long convert(int N) 
{
	unsigned long long ans = 0;
	w = 0;
	while (N > 0)
	 {
		w++;
		d[w] = N & 1;
		N /= 2;
	}
 
	for (int i=w;i>=1;i--)
		ans = ans * 10 + d[i];
 
	return ans;
}
#define m 1000000007
void cal(unsigned long long int base,unsigned long long int e)
{
	unsigned long long int result=1;
	while(e>0)
	{
		if(e%2==1)
		result=(result*base)% m;
		base=(base*base)%m;
		e/=2;
	}
	result=(result*result)%m;
	 printf("%llu\n",result);
}
