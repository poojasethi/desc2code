#include <bits/stdc++.h>
#include <vector>
#include <cstdio>
#include <map>
#include <set>
#define MOD 1000000007

using namespace std;

unsigned long long mod_pow(unsigned long long base, unsigned long long exponent) {
	unsigned long long reslt = 1;

	while (exponent > 0) {
		if (exponent % 2 == 1) {
			reslt = (reslt * base) % MOD;
		}
		exponent = exponent >> 1;
		base = (base * base) % MOD;
	}
	return reslt;
}


unsigned long long bin(int n)
{
	unsigned long long r = 0;
	vector<int> v;
    while (n > 0) {
    	v.push_back(n & 1);
    	n >>= 1;
    }
    
    for (int i = v.size() - 1; i >= 0; i--) {
    	r = r * 10 + v[i];
    } 
    return r;
}


int main()
{
	int t, n;
	scanf("%d", &t);
	
	while (t--) {
		scanf("%d", &n);
		
		unsigned long long ans = mod_pow(2, bin(n));
		ans = (ans * ans) % MOD;
		printf("%llu\n", ans);
	}
   	
	return 0;
}
