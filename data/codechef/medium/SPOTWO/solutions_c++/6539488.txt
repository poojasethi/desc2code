#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define F first
#define S second
#define maxn 1005
#define M 1000000007
using namespace std;
 
int main ()
{
	int t,d[25];
	unsigned LL ans,b,n;
	sc(t);
	while(t--)
	{
		//cout<<(1<<20)<<endl;
		sl(n);
		ans=0;
		b=2;
		int i=0;
		memset(d,0,sizeof(d));
		while(n>0)
		{
			i++;
			d[i]=n&1;
			n>>=1;
		}
		n=0;
		for(int j=i;j>=1;j--)
		{
			n=n*10+d[j];
		}
		//cout<<n<<endl;
		ans=1;
		while(n)
		{
			b=b%M;
			if(n&1)
			{
				ans=(ans*b)%M;
			}
			n=n>>1;
			b=(b*b)%M;
		}
		ans=(ans*ans)%M;
		pl(ans%M);
	}
	return 0;
}
