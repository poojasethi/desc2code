unsigned LL power(unsigned LL b,unsigned LL e)
{
	LL r=1;
	while(e)
	{
		if(e&1)
		{
			r*=b;
		}
		b=b*b;
		e>>=1;
	}
	return r;
}

		b=2;

				ans=(ans*b)%M;
