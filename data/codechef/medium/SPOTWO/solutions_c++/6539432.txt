#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define F first
#define S second
#define maxn 1005
#define M 1000000007
using namespace std;
int d[25],nd;
unsigned long long Convert(int N) {
	unsigned long long ans = 0;
	nd = 0;
	while (N > 0) {
		nd++;
		d[nd] = N & 1;
		N >>= 1;
	}
 
	for (int i = nd; i >= 1; i--)
		ans = ans * 10 + d[i];
 
	return ans;
}

int main ()
{
	int t;
	unsigned LL ans,b,n;
	sc(t);
	while(t--)
	{
		//cout<<(1<<20)<<endl;
		sl(n);
		n=Convert(n);
		ans=1;
		b=2;
		while(n)
		{
			b=b%M;
			if(n&1)
			{
				ans=(ans*b)%M;
			}
			n=n>>1;
			b=(b*b)%M;
		}
		ans=(ans*ans)%M;
		pl(ans%M);
	}
	return 0;
}
