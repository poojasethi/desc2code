#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
#define MOD 1000000007
#define pi(x) printf("%d",x)
#define pl(x) printf("%lld",x)
#define pn    printf("\n");
#define si(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
long long bin=0;
void make_bin(unsigned long long n)
{
	if(n==0)
	return;
	else
	{
		make_bin(n/2);
		bin=bin*10+n%2;
	}
}
long long int po(unsigned long long x, unsigned long long y) {
	if (y == 0)
		return 1;
	unsigned long long result = po(x, y >> 1);
	result = (result * result) % 1000000007;
	if (y & 1)
		result = (result * x) % 1000000007;
	return result;
}
int main()
{
	int t;
	si(t);
	while(t--)
	{
		bin=0;
		unsigned long long n;
		sl(n);
		make_bin(n);
		unsigned long long res=po(2,bin)%MOD;
		pl((res*res)%MOD);
		printf("\n");
		
	}
}
