#include<iostream>
#include <cmath>
using namespace std;
typedef unsigned long long ll;
const int modulo=1000000007;
ll powerfunc(ll b,ll c)
{
	if (c == 0)
		return 1;
	ll result = 1, x = b;
	while (c > 0)
	{
		if (c & 1)
		{
			result *= x;
			result = result % modulo;
			c--;
		}
		else
		{
			x *= x;
			x = x % modulo;
			c /= 2;
		}
	}
	result %=modulo;
	return result;
}
ll d2b(ll n) 
{
    ll rem,i,b;
    i=1; b=0;
    while (n!=0)
    {
        rem=n%2;
        n/=2;
        b+=rem*i;
        i*=10;
    }
    return b;
}
