#include<iostream>
#include <cstdio>
using namespace std;

int d[25], nd;

    long long int tobinary(int n) {
	unsigned long long ans = 0;
	nd = 0;
	while (n > 0) {
		nd++;
		d[nd] = n & 1;
		n >>= 1;
	}
	for (int i = nd; i >= 1; i--)
		ans = ans * 10 + d[i];
        return ans;
}
long long int po(unsigned long long x, unsigned long long y) {
	if (y == 0)
		return 1;
	unsigned long long result = po(x, y >> 1);
	result = (result * result) % 1000000007;
	if (y & 1)
		result = (result * x) % 1000000007;
	return result;
}
int main() {
	int t, n;
	scanf("%d", &t);
	for(int i=0;i<t;i++) {
		scanf("%d", &n);
		 long long int ans = po(2, tobinary(n));
		ans = (ans * ans) % 1000000007;
		printf("%lld\n", ans);
	}
	return 0;
}
