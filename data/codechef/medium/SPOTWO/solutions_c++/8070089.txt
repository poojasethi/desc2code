#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
typedef unsigned long long int ll;
ll fastPower(ll base,ll power)
{
    if(power==1)
       return base;
    if(power%2==0)
    {
        ll b=pow(fastPower(base,power/2),2);
        b=b%mod;
        return b;
    }
    else
    {
        ll a=base*pow(fastPower(base,(power-1)/2),2);
        a=a%mod;
        return a;
    }
}
int main()
{
    ll t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        ll s=0,r,i=0;
        while(n>0)
        {
            r=n%2;
            s=s+(r*(unsigned long long int)(pow(10,i++)));
            n=n/2;
        }
        cout<<fastPower(4,s)<<'\n';
    }
    return 0;
}
