#include<bits/stdc++.h>
#define lli long long int
#define MOD 1000000007
using namespace std;
lli power(lli a,lli b)
{
	lli s;
	if(b==0)
		return 1;
	if(b==1)
		return a;
	if(b%2==0)
	{
		s=power(a,b/2);
		return (s*s)%MOD;
	}
	else
	{
		s=power(a,b/2);
		return (a*(s*s)%MOD)%MOD;
	}
}
