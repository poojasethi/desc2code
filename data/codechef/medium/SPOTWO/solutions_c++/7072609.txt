
#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

int n;
unsigned long long sum,prod,bin;

int main()
{
	int test;
	cin >> test;
	while(test--)
    {
        cin >> n;
        sum=1;
        bin=0;
        prod=1;
        while(n>0)
        {
            bin=bin+(n%2)*prod;
            n=n/2;
            prod=prod*10;
        }
        prod=2;
        while(bin>0)
        {
            if(bin%2==1)
                sum=(sum*prod)%1000000007;
            bin=bin/2;
            prod=(prod*prod)%1000000007;
        }
        sum=(sum*sum)%1000000007;
        cout << sum << "\n";
    }
}
