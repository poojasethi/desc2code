unsigned long long power(long long x,unsigned long long y)
{
unsigned long long temp;
if( y == 0)
return 1;
temp = power(x, y/2);
if (y%2 == 0)
return (temp*temp) % 1000000007 ;
else
return (((x*temp)%1000000007) * temp) % 1000000007;
}

unsigned long long binary(long long t)
{
	long long rem,i=1;
	unsigned long long sum=0;
	while(t>0)
     {
        rem=t%2;
        sum=(sum + (i*rem) ) % 1000000006;
        t=t/2;
        i=i*10;
     }
 //cout<<"\n BINARY FOR T IS "<<sum<<endl;
 return sum;
}



int main()
{
