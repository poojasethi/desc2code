#include<iostream>
#include<cstdio>
#include<cmath>
#define mod 1000000007
#define ll long long int
using namespace std;
ll pow(unsigned ll a)
{
    ll result=1,base=2;
    while(a>0)
    {
        if(a%2==1)
                result=((result%mod)*(base%mod))%mod;
            base=((base%mod)*(base%mod))%mod;
            a=a/2;
    }
    return result;
}
unsigned ll bin(ll n)
{
    unsigned ll sum=0,prod=1;
    while(n>0)
    {
            sum=(n%2)*prod+sum;
            prod=prod*10;
            n=n/2;
    }
    return sum;
}
int main()
{
unsigned ll d;ll val,n,t;
scanf("%lld",&t);
while(t>0)
{
    t--;
    scanf("%lld",&n);
    d=bin(n);
    val=pow(d);
    val=((val%mod)*(val%mod))%mod;
    printf("%lld\n",val);
}
}
