#include<iostream>
using namespace std;
#define M 1000000007
long long int getpower(long long int a,long long int N)
{
   if(N==0)
      return 1;
   if(N==1)
      return a;
   long long int res=getpower(a,N/2);
   res=res*res%M;
   if(N%2)
      return res*a%M;
      return res;
}
int main()
{
   long long int T,N,power[21],ans,i,j,k;
   for(power[1]=2,i=2;i<21;i++)
      power[i]=getpower(power[i-1],10);
   cin>>T;
   while(T--)
   {
      cin>>N;
      for(ans=i=1;N;N/=2,i++)
      if(N%2==1)
      ans=ans*power[i]%M;
      cout<<ans*ans%M<<endl;
   }
   return 0;
}
