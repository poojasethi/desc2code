#include<bits/stdc++.h>
using namespace std;
#define ll unsigned long long int
#define mod 1000000007
ll bin(ll n)
{
	vector<ll> v;
	ll temp,r,res=0,i;
	temp=n;
	while(temp)
	{
		r=temp%2;
		v.push_back(r);
		temp/=2;
	}
	reverse(v.begin(),v.end());
	for(i=0;i<v.size();i++)
	{
		//printf("%d",v[i]);
		res=(res*10)+v[i];
	}
	//cout<<endl;
	//cout<< res<<endl;
	return res;
}
ll power(ll a,ll b)
{
	if(b==0)
		return 1;
	if(b==1)
		return a;
	ll temp=power(a,b/2);
	temp=(temp*temp)%mod;
	if(b&1)
		temp=(temp*a)%mod;
	return temp;
}
int main()
{
	ll n,ans,num;
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		num=bin(n);
		ans=power(2,num);
		ans=(ans*ans)%mod;
		printf("%lld\n",ans);
	}
	return 0;
}
