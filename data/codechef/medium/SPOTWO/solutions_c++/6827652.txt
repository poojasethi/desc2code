#include<stdio.h>

int main(){
    unsigned long long ans,pow,p;
    int t,n,l,i,a[21],m=1000000007;
    scanf("%d",&t);
    while(t-->0){
        pow=0;l=0;
        scanf("%d",&n);
        while(n>0){
            a[l++]=n&1;
            n>>=1;
        }
        while(l-->0)
        pow=pow*10+a[l];
        p=2;ans=1;pow%=m-1;
        while(pow>0){
            if(pow&1==1) ans=ans*p%m;
            p=p*p%m;
            pow>>=1;
        }
        ans=ans*ans%m;
        printf("%llu\n",ans);
    }
    return 0;
}
