#include<iostream>
using namespace std;

unsigned long long bin(int n)
{
 int arr[20],i=0;
 unsigned long long num=0;
 while(n!=0)    //stored in rev order
 {
  arr[i++]=n%2;
  //cout<<arr[i-1];
  n=n/2;
 }
 i--;

 for(;i>=0;i--)
 {
     num=num*10+arr[i];
 }
  return num;

}


int power(unsigned long long k ,int m)
{
 if(k==1) return 2;
 unsigned long long f=power(k/2,m);
 //cout<<"#"<<f<<"#";
 unsigned long long p=(f*f)%m;
 //t=f*f;
 //cout<<t;
 //p=(t)%m;
 if(k%2!=0)
  p=(p*2)%m;

 return p;
}

int main()
{
 int t,n,m=1000000007;
 cin>>t;
 unsigned long long  k;
 for(int i=0;i<t;i++)
 {
  cin>>n;
  k=bin(n);
  //cout<<k<<" ";
  //k=k;

  k=power(k,m);
  k=(k*k)%m;

 }

}
