#include<iostream>

using namespace std;

{

while(t--)
{
long long int n;

{

}
long long int ans=1;
//for(long long int i=k-1;i>=0;i--)
//cout<<bin[i];
//cout<<endl;
for(long long int i=k-1;i>=0;i--)
{
if(bin[i]==1)
{
long long int ans2=ans;
for(long long int j=0;j<9;j++)
ans=(ans*ans2)%1000000007;
ans=(ans*2)%1000000007;
//cout<<"a1="<<ans<<endl;
}
else 
{
long long int ans2=ans;
for(int j=0;j<9;j++)
{ans=(ans*ans2)%1000000007;
//cout<<"a0="<<ans<<endl;
}}
}

}

}
