#include<bits/stdc++.h>
using namespace std;
unsigned long long mod=1e9+7ll;
unsigned long long power(unsigned long long a,unsigned long long b)
{
	unsigned long long ans=1,pwr=a%mod;
	while(b)
	{
		if(b%2)ans=((ans*pwr)%mod);
		b/=2;pwr=((pwr*pwr)%mod);//printf("ans=%llu\n",ans);
	}
	return ans;
}
main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		int n;
		unsigned long long n1=0ll,pwr=1ll;
		scanf("%d",&n);
		while(n)
		{
			n1=(n%2)*pwr+n1;
			pwr*=10;n/=2;
		}//printf("%llu\n",n1);
		printf("%llu\n",power(power(2ll,n1),2ll));
	}
}
