#!/usr/bin/env python
# -*- encoding: utf-8 -*-
# pylint: disable=C0111
'''input
2
---+-+-+++
-------
'''


def main():
    def chefch(s):
        return str(min(
            s[::2].count('-') + s[1::2].count('+'),
            s[::2].count('+') + s[1::2].count('-')))

    print '\n'.join([chefch(raw_input()) for _ in xrange(int(raw_input()))])


main()
