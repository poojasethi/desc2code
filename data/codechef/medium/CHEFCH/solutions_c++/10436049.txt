#include<iostream>
using namespace std;
int main()
{
	int test,ans1,ans2;
	string str;
	scanf("%d",&test);
	while(test){
		test--;
		cin>>str;
		ans1=0;
		ans2=0;
		for(int i=0;i<str.length();i++){
			if(i%2==0){
			if(str[i]=='-'){
				ans1++;
			}	
			}
			else{
			  if(str[i]=='+'){
			  	ans1++;
			  }	
			}
		}
		for(int i=0;i<str.length();i++){
			if(i%2==0){
			if(str[i]=='+'){
				ans2++;
			}	
			}
			else{
			  if(str[i]=='-'){
			  	ans2++;
			  }	
			}
		}
		printf("%d\n",min(ans1,ans2));
	}
	return 0;
}
