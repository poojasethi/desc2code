#include<bits/stdc++.h>

using namespace std;

long long int pos(string str)
{
	int n = str.length();
	int flag = 0;
	long long int count=0;
	for(int i=0;i<n;i++){
		if(flag==0 && str[i]!='+'){
			count++;
			flag=1;
		}
		else if(flag==1 && str[i]!='-'){
			count++;
			flag=0;
		}
		else{
			flag=1-flag;
		}
	}
	return count;
}

long long int neg(string str)
{
	int n = str.length();
	int flag = 0;
	long long int count=0;
	for(int i=0;i<n;i++){
		if(flag==0 && str[i]!='-'){
			count++;
			flag=1;
		}
		else if(flag==1 && str[i]!='+'){
			count++;
			flag=0;
		}
		else{
			flag=1-flag;
		}
	}
	return count;
}

int main()
{
	int T;
	cin>>T;
	string str;
	while(T--){
		cin>>str;
		cout<<min(pos(str),neg(str))<<endl;
	}
	return 0;
}
