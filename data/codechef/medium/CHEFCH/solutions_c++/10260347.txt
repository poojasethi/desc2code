#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char str[100001],str1[100001],str2[100001];
		cin>>str;
		int len = strlen(str);
		char ch = str[0];
		char ch1 = str[0] == '+'?'-':'+';
		int i;
		for(i=0;i<len;i++)
		{
			if(i%2==0)
			{
				str1[i] = ch;
				str2[i] = ch1;
			}
			else
			{
				str1[i] = ch1;
				str2[i] = ch;
			}
		}
		str1[len] = '\0';
		str2[len] = '\0';
		int ctr1=0,ctr2=0;
		for(i=0;i<len;i++)
		{
			if(str[i]!=str1[i])
				ctr1++;
			if(str[i]!=str2[i])
				ctr2++;
		}
		cout<<min(ctr1,ctr2)<<endl;
	}
	return 0;
}
