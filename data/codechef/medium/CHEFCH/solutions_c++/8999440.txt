#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int i;
	char a[100000],b[100000];
	for(i=0;i<100000;i++)
	{
		if(i%2==0)
		{
			a[i]='+';
			b[i]='-';
		}
		else
		{
			a[i]='-';
			b[i]='+';
		}
		
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int count=0,count2=0;
		char c[100000];
		scanf("%s",c);
		for(i=0;c[i]!='\0';i++)
		{
			if(a[i]!=c[i])
			{
				count++;
			}
			else if(b[i]!=c[i])
			{
				count2++;
			}
		}
		printf("%lld\n",min(count,count2));
	}
	return 0;
}
