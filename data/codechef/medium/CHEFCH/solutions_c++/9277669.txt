
/* Libraries */

#include<stdio.h>

/* Limits */

#define MAX_SIZE 100001

/* Functions */

char nextChar(char currentChar);
int compare(char *string1, char currentChar);

/* Main Function */

int main(void)
{
    
    int inputs;
    char string1[ MAX_SIZE ];
    
    int result1;
    int result2;
    
    scanf("%d", &inputs);
    
    for(int input = 1; input <= inputs; input++)
    {
        
        scanf("%s", string1);
        
        result1 = compare(string1, '+');
        result2 = compare(string1, '-');
        
        if(result1 < result2)
        {
            printf("%d\n", result1);
        }
        else
        {
            printf("%d\n", result2);
        }
        
    }
    
    return 0;
    
}

/* Function Implementation */

char nextChar(char currentChar)
{
    
    if(currentChar == '+')
    {
        return '-';
    }
    
    return '+';
    
}

int compare(char *string1, char currentChar)
{
    
    int count = 0;
    
    while( (*string1) != '\0' )
    {
        
        if( (*string1) != currentChar )
        {
            count += 1;
        }
        
        currentChar = nextChar(currentChar);
        
        string1 += 1;
        
    }
    
    return count;
    
}
