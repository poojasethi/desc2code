int subset(int total,int val[],int m1)
{
    bool dp[m1+1][total+1];
    for(int i=0;i<=m1;i++)
    {
        dp[i][0]=true;
        
    }
    for(int i=1;i<=m1;i++)
    {
        for(int j=1;j<=total;j++)
        {
            
            
            if(j>=val[i-1])
            {
                dp[i][j]= dp[i-1][j] || dp[i-1][j-val[i-1]];
                
            }
            else
                dp[i][j]=dp[i-1][j];
            
        }
    }
    return dp[m1][total];
    
    
}
