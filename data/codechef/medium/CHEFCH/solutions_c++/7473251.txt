#include <iostream>
#include<stdio.h>
#include<string.h>
#include<math.h>
#include <stdlib.h>
#include<algorithm>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char s[100001];
        scanf("%s",s);
        long long i,s1=0,m1=0,s2=0,m2=0,j=0,k=0,a=0,max1,max2;
        for(i=0;i<strlen(s);i++)
        {
        	if(i%2==0)
        	{
        		if(s[i]=='+')
        		s1++;
        		else
        		m1++;
        		j++;
			}
			else
			{
				if(s[i]=='+')
        		s2++;
        		else
        		m2++;
        		k++;
			}
		}
		max1=s1+m2;
		max2=s2+m1;
		a=max1<max2?max1:max2;
		printf("%lld\n",a);
    }
    return 0;
}
