#include<bits/stdc++.h>
using namespace std;
char a[100001];
int main()
{
   int t,count,l,i;
   cin>>t;
   while(t--)
   {
     count=0;
     cin>>a;
     l=strlen(a);
     if(a[0]=='-')
     {
     for(i=1;i<l;i++)
      {
        if(i%2&&a[i]=='-')
        count++;
        if(i%2==0&&a[i]=='+')
        count++;
      }
    }
    else
    {
      for(i=1;i<l;i++)
      {
        if(i%2&&a[i]=='+')
        count++;
        if(i%2==0&&a[i]=='-')
        count++;
      }
    }
    if(count<l-count)
    cout<<count<<"\n";
    else
        cout<<l-count<<"\n";
}
return 0;
}
