#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	char str[100000];
	long int i,a=0,b=0;
	int t;
	cin>>t;
	while(t--)
	{
	
		cin>>str;
		
		for(i=0;i<strlen(str);i+=2)
		{
		//	cout<<"a : "<<a<<endl;;
		//cout<<"b : "<<b<<endl;
			if(str[i]=='-')
			a++;
			if(str[i+1]=='+')
			a++;
			if(str[i]=='+')
			b++;
			if(str[i+1]=='-')
			b++;
		
		}
	
		if(a>b)
		cout<<b<<endl;
		else
		cout<<a<<endl;
			a=0;
		b=0;
	}
	return 0;
}
