#include<bits/stdc++.h>
using namespace std;
inline int min(int a,int b)
{
	return(a<b?a:b);
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int c=0,d=0,e=0,f=0;
		for(int i=0;i<s.size();i++)
		{
			if(i%2==0 && s[i]=='+')
			{
				c++;
			}
			if(i%2!=0 && s[i]=='-')
			{
				d++;
			}
		}
		for(int i=0;i<s.size();i++)
		{
			if(i%2==0 && s[i]=='-')
			{
				e++;
			}
			if(i%2!=0 && s[i]=='+')
			{
				f++;
			}
		}
		cout<<min(c+d,e+f)<<endl;;
	}
}
