#include<iostream>
using namespace std;

string str;
int main()
{
	int t;
	cin>>t;
	while (t--)
	{
		cin>>str;
		int count1=0,count2=0;
		
		for (int i=0;i<str.size();i++)
		{
			if (i&1 && str[i] == '-') count1++;
			if (!(i&1) && str[i]=='+') count1++;
		}
		
		for (int i=0;i<str.size();i++)
		{
			if (i&1 && str[i] == '+') count2++;
			if (!(i&1) && str[i]=='-') count2++;
		}
		
		if (count1 < count2)
		cout<<count1<<endl;
		else
		cout<<count2<<endl;
		
	}
}
