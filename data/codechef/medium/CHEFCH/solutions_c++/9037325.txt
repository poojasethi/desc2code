#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,t,n,ans1,ans2;
	char s[100001];
	scanf("%d",&t);
	while(t--)
	{
		cin>>s;
		n=strlen(s);
		ans1=ans2=0;
		for(i=0;i<n;i++)
		{
			if(i%2==0)
			{
				if(s[i]=='-')
					ans1++;
				else
					ans2++;
			}
			else
			{
				if(s[i]=='+')
					ans1++;
				else
					ans2++;
			}
		}
		printf("%d\n",min(ans1,ans2));
	}
	return 0;
}
