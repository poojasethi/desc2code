#include <iostream>
#include <bits/stdc++.h>

typedef long long int ll;
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		ll cn1=0,cn2=0;
		string s1,s2;
		for(unsigned int i=0;i<s.length();i++)
		{
			char c, c1;
			if(i%2==0) {c = '+'; c1 = '-';}
			else {c = '-'; c1 = '+';}
			s1 = s1 + c;
			s2 = s2 + c1;
		}
		for(unsigned int i=0;i<s.length();i++)
		{
			if(s[i]!=s1[i]) cn1++;
			if(s[i]!=s2[i]) cn2++;
		}
		ll ans = min(cn1,cn2); 
		cout<<ans<<endl;
	}	
	return 0;
}
