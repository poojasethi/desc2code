#include<bits/stdc++.h>
using namespace std;
char a[1000000],b[1000000],c[1000000];
int main()
{
	int t,i,ans;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",c);
		int n=strlen(c);
		for(int i=0;i<n;i++)
		{
			if(i%2==0)
			{
				a[i]='-';
				b[i]='+';
			}
			else
			{
				a[i]='+';
				b[i]='-';
			}
		}
		int c1=0,c2=0;
		for(int i=0;i<n;i++)
		{
			if(c[i]!=a[i])
			{
				c1++;
			}
			if(c[i]!=b[i])
			{
				c2++;
			}
		}
		ans = min(c1,c2);
		printf("%d\n",ans);
	}
	return 0;
}
