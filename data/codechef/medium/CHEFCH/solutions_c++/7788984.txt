#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <math.h>

using namespace std;

int input()
{
    int n=0;
    char c=getchar_unlocked();
    while(c < '0' || c >'9') {c=getchar_unlocked();}
    while(c>='0' && c<='9')
    {
        n=(n<<3)+(n<<1)+c-'0';
        c=getchar_unlocked();
    }
    return n;
}

char change(char a){
	if(a == '+')
		return '-';
	else
		return '+';
}

int main(){
	int T, count, temp;
	string str;
	char curr, prev;
	T = input();
	while(T--){
		count = 0;
		cin >> str;
		curr = str[0];
		for(int i = 1; i < str.size(); i++){
			prev = curr;
			curr = str[i];
			if(prev == curr){
				curr = change(curr);
				count++;
			}
		}
		temp = str.size()-count;
		cout << min(count, temp) << endl;
	}
}
