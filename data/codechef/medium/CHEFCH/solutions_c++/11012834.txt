#include<iostream>
#include<cstring>
int main()
{ char a[100002];
  long long int t,count1,count2,l,i;
   scanf("%lld",&t);
   while(t--)
   {scanf("%s",a);
    l=strlen(a);
    count1=count2=0;
    for(i=0;i<l;i++)
    {if(i%2==0)
     {
     	if(a[i]=='-')
     	count1+=1;
     	else
     	count2+=1;
     	
     }
     else
     {if(a[i]=='+')
      count1+=1;
      else
      count2+=1;
     }
    }
    if(count1>=count2)
    printf("%lld\n",count2);
    else
    printf("%lld\n",count1);
   }
   return 0;
}
