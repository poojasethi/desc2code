#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;

char a[100005];
int main()
{
	long long int count1, count2, i, t,l;
	scanf("%lld", &t);
	while (t--)
	{
		count1 = count2 = 0;
		scanf("%s", a);
		l = strlen(a);
		a[l + 1] = '\0';
		i = 0;
		while (a[i] != '\0')
		{
			if (a[i] == '+')
				count1++;
			else
				count2++;
			i += 2;
		}
		i = 1;
		while (a[i] != '\0')
		{
			if (a[i] == '+')
				count2++;
			else
				count1++;
			i += 2;
		}
		printf("%lld\n", min(count1, count2));
	}
	getchar();
	getchar();
	return 0;
}
