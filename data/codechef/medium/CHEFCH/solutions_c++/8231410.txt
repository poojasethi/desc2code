#include<bits/stdc++.h>
using namespace std;
int main()
{ int t;
  cin>>t;
  while(t--)
  { char s[100009];
    cin>>s;
    int p,i,val=0,count=0;
	p=0;
    for(i=0;s[i]!='\0';i++)
  	{ if(p==0)
  		{  if(s[i]=='-')
		   ++count;
		   if(s[i]=='+')
		   ++val;
		   p=1;
		  }
  	   else if(p==1)
  	  {  if(s[i]=='+')
  	  	 ++count;
  	  	 if(s[i]=='-')
  	  	 ++val;
  	  	 p=0;
		}	
	}
  	
  	 cout<<min(count,val)<<"\n";
  }
     return 0;	
}
