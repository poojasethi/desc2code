#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,c1,c2,i;
    string str;
    cin>>t;
    while(t--)
    {   c1=0;
         c2=0;
        cin>>str;
        for(i=0;str[i];i++)
        {
            if(i%2==0&&str[i]=='+')
                c1++;
            if(i%2!=0&&str[i]=='-')
                c1++;
            if(i%2!=0&&str[i]=='+')
                c2++;
            if(i%2==0&&str[i]=='-')
                c2++;
        }
        cout<<min(c1,c2)<<endl;
    }
}
