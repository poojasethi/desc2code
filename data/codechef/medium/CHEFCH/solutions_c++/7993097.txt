#include<bits/stdc++.h>
using namespace std;
main()
{
    int T;
    cin>>T;
    while(T-->0)
    {
        char s[100001];
        cin>>s;
        int op;
        if(s[0]=='+')
            op=1;
        else
            op=0;
            int count1=0;
            int count2=1;
            char ch;
            int i;
        for(i=1;s[i]!='\0';i++)
        {
            op=(op+1)%2;
        if(op==1)
            ch='+';
        else
            ch='-';
           if(s[i]!=ch)
            count1++;

        }
        if(s[0]=='+')
            op=1;
        else
            op=0;
            op=(op+1)%2;
        for(i=1;s[i]!='\0';i++)
        {
            op=(op+1)%2;
        if(op==1)
            ch='+';
        else
            ch='-';
           if(s[i]!=ch)
            count2++;

        }
        if(count1<count2)
            cout<<count1<<endl;
        else
            cout<<count2<<endl;
    }
}
