#include<bits/stdc++.h>
using namespace std;
int main()
{
         int j,i,t=0,k=0;
         cin>>j;
         char a[100001];
         fflush(stdin);
         while(j--)
         {
                  scanf("%s",a);
                  for(i=0;a[i]!='\0';i++)
                  {
                           if(i%2==0)
                           {
                                    if(a[i]=='-')
                                    t++;
                                    else if(a[i]=='+')
                                    k++;

                           }
                           else
                           {
                                   if(a[i]=='+')
                                   t++;
                                   else if(a[i]=='-')
                                   k++;
                           }
                  }
                  if(t<k)
                  cout<<t<<"\n";
                  else
                  cout<<k<<"\n";
                  t=0;
                  k=0;
         }
         return 0;
}
