#include<bits/stdc++.h>
using namespace std;
int t,ctr1,ctr2,i,j;
string s;
char x1[100001],x2[100001];
void fastscan(int &x)
{
    register int c;
    x=0;
    c=getchar();
    for(;c>47&&c<58;c=getchar())
        x=(x<<1)+(x<<3)+c-48;

}

int main()
{   //std::ios::sync_with_stdio(false);
    fastscan(t);
  while(t--)
  {
     cin>>s;

       ctr1=0;
       ctr2=0;

      for(i=0;s[i]!='\0';i++)
         {if(i%2==1)
         {x1[i]='-';
          x2[i]='+';
         }
         else
         {x2[i]='-';
          x1[i]='+';
         }
         }

      for(i=0;s[i]!='\0';i++)
      {
          if(s[i]!=x1[i])
            ctr1++;

            if(s[i]!=x2[i])
            ctr2++;
      }

cout<<min(ctr1,ctr2)<<endl;

  }
return 0;
}
