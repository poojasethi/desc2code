#include<iostream>
using namespace std;
#include<string.h>
int main()
{   ios_base::sync_with_stdio(false);  
    cin.tie(NULL);
    int t;
    long len;
    char str[100000];
	cin>>t;
	while(t--)
	{   long count1=0,count2=0;
		cin>>str;
		len=strlen(str);
		char arr1[len+1],arr2[len+1];
		for(long i=0;i<strlen(str);i++)
		 {
		 	
		 	if(i%2==0)
		 	{
		 		arr1[i]='+';
		 		arr2[i]='-';
		 	}
		 	else
		 	{
		 		arr1[i]='-';
		 		arr2[i]='+';
		 	}
		 }
		 for(long i=0;i<strlen(str);i++)
		 {
		 	if(str[i]!=arr1[i])
		 	count1++;
		 	if(str[i]!=arr2[i])
		 	count2++;
		 }
		 if(count1>count2)
		  cout<<count2<<endl;
		  else
		  cout<<count1<<endl;
	}
	return 0;
}
