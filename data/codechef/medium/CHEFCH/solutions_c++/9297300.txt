#include <cstdio>
#include <algorithm>
#define MAX_STR_SIZE 100000
char str[MAX_STR_SIZE + 3];
char pattern1[MAX_STR_SIZE + 3];
char pattern2[MAX_STR_SIZE + 3];

void init_patter() {
    char p1 = '+';
    char p2 = '-';
    for (int i = 0; i <= MAX_STR_SIZE; i++) {
        pattern1[i] = p1;
        pattern2[i] = p2;
        p1 ^= p2 ^= p1 ^= p2;
    }

    pattern1[MAX_STR_SIZE+1] = pattern2[MAX_STR_SIZE+1] = '\0';
}

int main() {
    int t;
    init_patter();
    scanf("%d", &t);
    while (t--) {
        scanf("%s", str);
        int p1_cnt = 0, p2_cnt = 0;
        for(int i = 0; str[i]!=0;i++) {
            if(str[i]!=pattern1[i])
                p1_cnt++;
            if(str[i]!=pattern2[i])
                p2_cnt++;
        }
        printf("%d\n", std::min(p1_cnt, p2_cnt));
    }
    return 0;
}
