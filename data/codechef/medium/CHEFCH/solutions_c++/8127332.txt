#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long t;
    //std::ifstream cin("a.txt");
    cin>>t;
    string s;
    while(t--)
    {
        int count1=0,count2=0;
        cin>>s;
        for(int i=0;i<s.length();i++)
        {
            if(i%2==0)
            {
                if(s[i]=='+')
                    count1++;
                else
                    count2++;
            }
            else
            {
                if(s[i]=='-')
                    count1++;
                else
                    count2++;
            }
        }
        int count=(count1>count2)?count2:count1;
        cout<<count<<endl;
    }
}
