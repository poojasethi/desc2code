#include<iostream>

using namespace std;

int calculate(char a[]) {
	int i=0;
	int c1=0, c2=0;
	for(;a[i]!='\0';i++) {
		if(i%2) {
			if(a[i]=='+')
				c1++;
			else
				c2++;
		}
		else {
			if(a[i]=='-')
				c1++;
			else
				c2++;
		}
	}
	return min(c1,c2);
}

int main () {
	ios::sync_with_stdio(false); 
	int T;
	cin>>T;
	char a[100005];
	while(T--) {
		cin>>a;
		cout<<calculate(a)<<endl;
	}
	return 0;
}
