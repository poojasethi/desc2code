#include<stdio.h>

#define MOD 100000007

long long int power(long long int b, long long int p,long long int n)

{

	long long int x=1,y=b;

	while(p)

	{

		if(p&1)

			x=(x*y)%n;

		y=(y*y)%n;

		p>>=1;

	}

	return x%n;

}

int main()

{

	int T;

	scanf("%d",&T);

	while(T--)

	{

		long long int n;

		long long int a,b,c,d,e,r1,r2;

		scanf("%lld",&n);

		a=2* power(4,n-1,MOD);

		b= power(3,n,MOD);

		c= power(2,n-1,MOD);


		d= power(2,n,MOD);

		e= (power(3,n,2*MOD)+1)/2;

		if(e>d)r1=(e-d)%MOD;

		else r1=(MOD+e-d)%MOD;


		if(a>b)a=a-b;

		else a=MOD+a-b;

		a=a+c;

		a=a+d;


		if(a>e)r2=(a-e)%MOD;

		else r2=(MOD+a-e)%MOD;

		printf("%lld %lld\n",r1,r2);

	}
}
