template<typename T>T absll(T X)
{
    if(X<0)
        return -1*X;
    else
        return X;
}

LL modPow(LL A,LL B,LL C)
{
	LL X=1LL;
	while(B)
	{
		if(B%2LL==1LL)
		{
			X=(X*A)%C;
		}
		A=(A*A)%C;
		B/=2LL;
	}
	return X;
}

	{
		LL N;
		scanf("%lld",&N);
		LL t1=modPow(3,N,MOD);
		t1+=1;
		t1%=MOD;
		t1*=modPow(2LL,MOD-2,MOD);
		t1-=modPow(2,N,MOD);
		t1+=MOD;
		t1%=MOD;
		
		LL t2=2LL*modPow(4,N-1,MOD);
		t2+=MOD-modPow(3,N,MOD);
		t2%=MOD;
		t2+=modPow(2,N-1,MOD);
		t2%=MOD;
		t2-=t1;
		t2+=MOD;
		t2%=MOD;
		
		printf("%lld %lld\n",t1,t2);
	}

}
