#include <bits/stdc++.h>

#define test int t; scanf("%d",&t);while(t--)

typedef long long ll;
using namespace std;

int MOD=1e8+7;

int prod(int a,int b){
	ll aa=a,bb=b;
	return (aa*bb)%MOD;
}
int sum(int a,int b){
	ll aa=a,bb=b;
	return (aa+bb)%MOD;
}
int power(int x,ll n){
	if(n==0)return 1;

	int temp=power(x,n/2);

	if(n%2==0)return prod(temp,temp);
	return prod(temp,prod(temp,x));
}
int inv=power(2,MOD-2);
int main(){
	test{
		ll n; cin >> n;

		int ans1=0,ans2=0;
		ans1=sum(power(3,n),1);
		ans1=(ans1-power(2,n+1)+MOD)%MOD;
		ans1=prod(ans1,inv);

		ans2=sum(power(4,n),sum(power(2,n),power(2,n+1)));
		ans2=(ans2-1+MOD)%MOD;
		ans2=(ans2-power(3,n+1)+MOD)%MOD;
		ans2=prod(ans2,inv);

		cout << ans1 << " " << ans2 << "\n";
	}
}
