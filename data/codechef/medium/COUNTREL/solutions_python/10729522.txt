#include<stdio.h>
long long int m=100000007;

long long int modpowfast(long long int b,long long int e,long long int mod)
{
	long long int ans=1;
	while(e>0)
	{
		if(e&1)
			ans=(ans*b)%mod;
		b=(b*b)%mod;
		e=e>>1;
	}
	return ans;
}

int main()
{
	long long int t,n,ans1,ans2;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		ans1=((modpowfast(3,n,2*m)+1)/2)-modpowfast(2,n,m);
		ans2=((2*modpowfast(4,n-1,m))-modpowfast(3,n,m)+m)%m;
		ans2=(ans2 + modpowfast(2,n-1,m) + modpowfast(2,n,m) - ((modpowfast(3,n,2*m)+1)/2) + m)%m;
		ans1=(ans1+m)%m;
		printf("%lld %lld\n",ans1,ans2);
	}
	return 0;
}
