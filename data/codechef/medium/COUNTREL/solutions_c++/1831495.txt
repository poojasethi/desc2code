#include <stdio.h>
#include <stdlib.h>
#define MOD 100000007
typedef long long int ULL;
 
ULL power(ULL a,ULL n)
{
	ULL temp;
	if(n==1)
		return(a%MOD);
	if(n==0)
		return 1;
	temp = power(a,n/2)%MOD;
	temp = (temp*temp)%MOD;
	if(n%2!=0)
		temp = (temp*(a%MOD))%MOD;
	return temp;
}
 
int main(){
	int tc;
	ULL a,tmp,th,tw,tw1;
	scanf("%d",&tc);
	while(tc--){
		scanf("%lld",&a);
		if(a==1)
		{
			printf("0 0\n");
			continue;
		}
		if(a==2)
		{	
			printf("1 0\n");
			continue;
		}
	
		th = power(3,a);
		tw = power(2,a);
		tw1 = (tw*2)%MOD;
		
		tmp = (th+1-tw1);
		while(tmp<0)
			tmp += MOD;
		if(tmp%2!=0)
			tmp = ((MOD+tmp)/2)%MOD;
		else
			tmp = tmp/2;
		printf("%lld ",tmp);
 
		tmp = (tw*tw)%MOD;
		tmp = (tmp + ((3*tw)%MOD))%MOD;
		tmp = tmp - ((3*th)%MOD);
		tmp = (tmp-1)%MOD;
		while(tmp<0)
			tmp += MOD;
		if(tmp%2!=0)
			tmp = ((MOD+tmp)/2)%MOD;
		else
			tmp = tmp/2;
		printf("%lld\n",tmp);
	}
}
