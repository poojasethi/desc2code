#include<stdio.h>

long long mpow(long long a,long long b)
{
    if(b==0)
        return 1;
    if(b==1)
    return a%100000007;
    long long tp=mpow(a,b/2);
    tp=(tp*tp)%100000007;
    if(b%2)
        tp=(tp*a)%100000007;
    return tp;
}
long long findresult1(long long N)
{
    long long num,den;
    num=(mpow(3,N)-mpow(2,N+1)+1+100000007)%100000007;
    den=mpow(2,100000005);
    return (num*den)%100000007;
}
long long findresult2(long long N)
{
    long long num,den;
    num=(mpow(4,N)-mpow(3,N+1)+3*(mpow(2,N)-1)+2+100000007)%100000007;
    den=mpow(2,100000005);
    return (num*den)%100000007;
}
int main()
{
    int T;
    scanf("%d",&T);
    while(T)
    {
        long long N;
        scanf("%lld",&N);
        long long a1=0,a2=0;
        if(N>1)
            a1=findresult1(N);
        if(N>2)
            a2=findresult2(N);
        printf("%lld %lld\n",a1,a2);
        T--;
    }
}
