#include<iostream>
const int L=100000007,inv2=50000004;
using namespace std;
long long modulo_exponent(int e,long long n)
{
long long x=e;long long temp=1;
	if (n==0) return 1;
if (n%2==0)
{
temp = modulo_exponent(e,n/2);
return (temp*temp)%L;
}
return (modulo_exponent(e,n-1)*e)%L;
}
