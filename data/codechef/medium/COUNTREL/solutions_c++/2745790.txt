#define MOD 100000007
typedef long long int lli;
typedef long double ld;

using namespace std;

lli mpow(lli n, lli p)
{
	if(p==0) return 1;
	if(p==1) return n;
	lli temp=mpow(n,p/2);
	temp=(temp*temp)%MOD;
	if(p%2) return (temp*n)%MOD;
	return temp;
}

int main()
{
    //freopen("inp.txt","r",stdin);
    int t;
    lli n;
    lli ti=mpow(2,MOD-2);
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%lld",&n);
    	lli ans=(mpow(3,n)-mpow(2,n+1)+1)%MOD;
    	ans=(ans*ti)%MOD;
    	if(ans<0) ans+=MOD;
    	printf("%lld ",ans);
    	ans=(mpow(4,n)-mpow(3,n+1)+3*mpow(2,n)-1)%MOD;
    	ans=(ans*ti)%MOD;
    	if(ans<0) ans+=MOD;
    	printf("%lld\n",ans);
    }
    return 0;
    
}
