    #include<stdio.h>
    
    long long int mod=100000007;
    
    long long int pow(long long int n,long long int x)
    {
        if(x==0)
        return 1;
        
        long long int res=pow(n,x/2);
        if(x%2)
        return (((res*res)%mod)*n)%mod;
        return (res*res)%mod;
    }
    
    long long int cons=50000004;
    
    void cal(long long int n)
    {
         long long int t;
         long long int r;
         long long int r1;
         long long int r2;
         long long int p2,p3,p4;
         p2=pow(2,n);
         p3=pow(3,n);
         p4=(p2*p2)%mod;
         t=(p2+p4)%mod;
         t=(t*cons)%mod;
         r=t-p3;
         if(r<0)
         r=r+mod;
         r1=(p3+1-2*p2)%mod;
         if(r1<0)
         r1=r1+mod;
         r1=(r1*cons)%mod;
         r2=r-r1;
         r2=r2%mod;
         if(r2<0)
         r2=r2+mod;
         printf("%lld %lld\n",r1,r2);
    }
