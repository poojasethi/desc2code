#include<iostream>
#include<stdio.h>
#define MOD 100000007LL
using namespace std;
long long pow1(long long n,long long m)
{
  long long res=1;
  while(m)
  {
    if(m%2==1)res=(res*n)%(2*MOD);
    n=(n*n)%(2*MOD);
    m/=2;
  }
  return res;
 }

int main(void)
{
  int t;
  long long n,r1,r2,t1,t2;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%lld",&n);
    t1=(pow1(3,n)+1)%(2*MOD);
    t2=pow1(2,n+1);
    if(t1>=t2)
    r1=(t1-t2)/2;
    else r1=(2*MOD+t1-t2)/2;
    t1=pow1(2,n);
    t2=(pow1(3,n+1)+1)%(2*MOD);
    t1=(t1*((t1+3)%(2*MOD)))%(2*MOD);
    if(t1>=t2)
    r2=(t1-t2)/2;
    else r2=(2*MOD+t1-t2)/2;
    printf("%d ",r1);
    printf("%d\n",r2);
  }
 return 0;
}
