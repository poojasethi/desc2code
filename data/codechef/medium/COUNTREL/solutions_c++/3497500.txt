#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
 
using namespace std;
 
const int Mod=100000007;
 
long long Pow(long long A,long long B)
{
  if(!B)
    return 1;
  long long T=Pow(A*A%Mod,B/2);
  if(B&1)
    T=T*A%Mod;
  return T;
}
 
long long Half(long long X)
{
  X%=Mod;
  if(X<0)
    X+=Mod;
  if(X&1)
    X+=Mod;
  return X/2;
}
 
int main()
{
  int Test;
  scanf("%d",&Test);
  while(Test--)
    {
      long long N;
      cin>>N;
      long long Ans_0=Pow(3,N)-Pow(2,N+1)+1;
      long long Ans_1=Pow(4,N)-Pow(3,N+1)+Pow(2,N)*3-1;
      cout<<Half(Ans_0)<<' '<<Half(Ans_1)<<endl;
    }
  return 0;
}
