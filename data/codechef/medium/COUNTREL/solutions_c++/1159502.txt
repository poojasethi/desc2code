#include<stdio.h>
using namespace std;
#define MOD 100000007
typedef long long	LL;
LL moduloSpecial(LL b, LL p,LL n) 
{
	LL x=1,y=b;
	while(p) 
    {
		if(p&1)x=(x*y)%
        n;
		y=(y*y)%n;
		p>>=1;
	}
	return x%n;
}
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
              LL n;
              LL a,b,c,d,e,r1,r2;
              scanf("%lld",&n);
              a=2*moduloSpecial(4,n-1,MOD);
              b=moduloSpecial(3,n,MOD);
              c=moduloSpecial(2,n-1,MOD);
              d=moduloSpecial(2,n,MOD);
              e=moduloSpecial(3,n,2*MOD)+1;
              e=e/2;
              if(e>d)r1=(e-d)%MOD;
              else r1=(MOD+e-d)%MOD;
              if(a>b)a=a-b;
              else a=MOD+a-b;
              a=(a+c)%MOD;
              a=(a+d)%MOD;
              if(a>e)r2=(a-e);
              else r2=(MOD+a-e)%MOD;
              printf("%lld %lld\n",r1,r2);
              
    }
}
