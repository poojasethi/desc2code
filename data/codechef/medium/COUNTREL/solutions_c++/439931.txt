#include<iostream>
#include<cstdio>

using namespace std;

#define  N 100000007
#define ll long long int

ll mod(ll a,ll b,ll MOD){
	if(b==0) return 1;
	if(b%2) return a*mod(a,b-1,MOD)%MOD;
	ll p=mod(a,b/2,MOD);
	return p*p%MOD;
}
int main(){
	ll t;
	ll a,b,c,d,e,n,r1,r2;
	scanf("%lld",&t);
	while(t--){
		scanf("%lld",&n);
		a=2*mod(4,n-1,N)%N;
		b=mod(3,n,N);
		c=mod(2,n-1,N);
		d=mod(2,n,N);
		e=(mod(3,n,2*N)+1)%(2*N);
		e/=2;
		if(e>d) r1=(e-d)%N;
		else r1=(N+e-d)%N;	
		r2=(a+c)%N;
		if(r2>b) r2=(r2-b)%N;
		else r2=(N+r2-b)%N;
		if(r2>r1) r2=(r2-r1)%N;
		else r2=(N+r2-r1)%N;
		printf("%lld %lld\n",r1,r2);
	}
	return 0;

}
