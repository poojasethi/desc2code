#include<iostream>
using namespace std;

int a[200005]={0};

int main(){
	int n,p;
	cin>>n>>p;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	int cura=1000000000;
	int curd=1000000000;
	for(int i=0;i<11 && i<n-1;i++){
		for(int j=i+1;j<12 && j<n ;j++){
			int num=a[j]-a[i];
			int den=j-i;
			if(num%den==0){
				int d=num/den;
				int a1=a[i]-d*i;
				int count=0;
				for(int k=0;k<n;k++){
					if(a[k]!=(a1+k*d)){
						count++;
					}
				}
				//cout<<count<<endl;
				if(count<=p && a1<cura){
					cura=a1;
					curd=d;
				}
				else if(count<=p && a1==cura && d<curd){
					cura=a1;
					curd=d;
				}
			}
		}
	}
	for(int i=0;i<n;i++){
		cout<<(cura+i*curd)<<" ";
	}
	cout<<endl;
	return 0;
}
