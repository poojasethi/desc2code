#include <algorithm>
#include <stdio.h>
#include <map>
#define  fi        first
#define  se        second
#define  inf       1 << 30
#define  maxn      100003
using    namespace std;
typedef  pair<int,int> pi;

int n,k;
int ar[maxn];
map<int,int>hash;

int main()
{
  scanf("%d%d",&n,&k);
  for(int i=1 ; i<=n ; i++)
    scanf("%d",&ar[i]);
  pi ans=pi(inf,inf);
  for(int i=1 ; i<=k+1 ; i++)
  {
    hash.clear();
    for(int j=1 ; j<=n ; j++)
      if(i!=j && abs(ar[i]-ar[j])%abs(i-j)==0)
        hash[(ar[i]-ar[j])/(i-j)]++;
    map<int,int>::iterator it;
    for(it=hash.begin() ; it!=hash.end() ; it++)
      if((it->se)>=n-k-1)
      {
        int diff=(it->fi);
        pi tmp=pi(ar[i]-diff*(i-1),diff);
        ans=min(ans,tmp);
      }
  }
  for(int i=1 ; i<=n ; i++)
    printf("%d ",ans.fi+(i-1)*ans.se);
  printf("\n");
  return 0;
}
