int a[200001];
 
inline void scanint(int* x)
{
bool m=false;
register char c = getchar_unlocked();
if(c=='-')m=true;
*x = 0;
for(; (c<48)||(c>57);c = getchar_unlocked());
for(; (c>47)&&(c<58);c = getchar_unlocked())
*x = (int)((((*x)<<1) + ((*x)<<3)) + c - 48);
 
if(m==true)*x=(-1)*(*x);
}
 
int scan()
{ int noRead=0;
    char p= getchar();
    for(;p<33;)
    { p=getchar(); }
    while(p>32)
        {
            noRead = (noRead << 3) + (noRead << 1) + (p - '0');
            p=getchar();
        }
    return noRead;
}
 
 
int main()
{
    int n;
    int k;
    
    scanint(&n);
    scanint(&k);
 
    for ( int i = 0; i < n; ++i )
        scanint(&a[i]);
 
    int na = INT_MAX;
    int nd = INT_MAX;
 
    for ( int i = 0; i <= k && i < n; ++i ) {
        for ( int j = i+1; j <= k+1 && j < n; ++j ) {
            int currd = (a[j] - a[i]) / (j - i);
            int curra = a[i] - currd * i;
           
            int countt = 0;
            for ( int x = 0; x < n; ++x) {
                if ( a[x] == ( curra + x * currd ) )
                    continue;
                countt++;
            }
            if ( curra <= na && countt <= k ) {
                if ( (curra == na && currd < nd) || (curra < na) ) {
                    na = curra;
                    nd = currd;
                }
            }
 
        }
    }
 
    for ( int i = 0; i < n; ++i ) {
        printf("%d",na + i * nd);
        if ( i != n -1 )
            printf(" ");
    }
    printf("\n");
 
    return 0;
 
 
 
}
