#include <iostream>
#include <vector>
#include <climits>
using namespace std;

vector<int> A;
int N, K;
int bestA, bestD;

void checkAndUpdate(int I, int J){
	// cout << I <<":"<<J<<endl;
	int currA, currD;
	currD = (A[J]-A[I])/(J-I);
	currA = A[I]-I*currD;
	int changes = 0;
	for(int i=0;i<N && changes<=K;i++){
		if(A[i] != currA+i*currD)
			changes++;
	}
	if(changes <= K){
		// update best
		if(bestA > currA){
			bestA = currA;
			bestD = currD;
		}else if(bestA == currA){
			if(currD < bestD){
				bestD = currD;
			}
		}
	}
}

int main(){
	cin >> N >> K;
	A.assign(N,0);
	for(int i =0;i<N;i++)
		cin >> A[i];
	bestA = INT_MAX;
	for(int i =0; i < min(12,N);i++){
		for(int j = i+1; j < min(12,N); j++){
			checkAndUpdate(i,j);
		}
	}
	for(int i = 0;i < N; i++)
		cout << bestA+bestD*i <<" ";
	return 0;
}
