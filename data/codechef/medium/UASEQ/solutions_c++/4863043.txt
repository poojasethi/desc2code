#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long LL;
pair<LL,LL> ret =make_pair(2000000000,2000000000);

int main()
{
//		ret = min(ret,make_pair(1LL,1LL));
//		printf("%lld %lld\n",ret.first,ret.second);

		int n,k;
		LL* arr;
		scanf("%d %d",&n,&k);
		arr = new LL[n];
		for(int i=0;i<n;i++)scanf("%lld",&arr[i]);

		int left,right;
		LL common_diff,a0;
		for(left=0;left<=k;left++)
		{
				int remain = k-left;
				for(right=max(left+1,n-1-remain);right<n;right++)
				{
						if((arr[right]-arr[left])%(right-left)!=0)continue;
						common_diff=(arr[right]-arr[left])/(right-left);
						a0 = arr[left]-left*common_diff;

						int changed=0;
						for(int i=1;i<n;i++)
						{
								if(arr[i]!=a0+i*common_diff)changed++;
						}
						if(changed<=k)
						{
								ret =min(ret,make_pair(a0,common_diff));
						}
				}
		}

		a0=ret.first;
		common_diff=ret.second;
		printf("%lld",a0);
		for(int i=1;i<n;i++)
		{
				printf(" %lld",a0+i*common_diff);
		}
		printf("\n");
		delete []arr;
		return 0;
}
