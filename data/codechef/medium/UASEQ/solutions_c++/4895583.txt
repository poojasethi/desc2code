int n,k,A,D,arr[100003];
inline void solve()
{
	int lim=min(13,n-1);
	bool flag=true;
	for(int count,a,d,i=1;i<lim;i++)
	{
		a=arr[n-i-1],d=arr[n-i]-arr[n-i-1];
		count=0;
		for(int j=n-i;j<n;j++)
		{
			if(arr[j]!=(a+=d))count++;
		}
		
		a=arr[n-i];
		for(int j=n-i-1;j>=0;j--)
		{
			if(arr[j]!=(a-=d))count++;
		}
		
		if(count<=k)
		{
			flag?(flag=false,A=a,D=d):(A>a?A=a:(a==A&&D>d?D=d:D));
		}
	}
	return ;
}
int main()
{
	cin>>n;
	cin>>k;
	for(int i=0;i<n;i++)cin>>arr[i];
	solve();
	n--;
	for(int i=0;i<n;i++)printf("%d ",(A+D*i));
	printf("%d ",(A+D*n));
	return 0;
}
