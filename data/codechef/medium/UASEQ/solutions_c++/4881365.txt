#include<bits/stdc++.h>

using namespace std;

int a[100005];

int main()
{
	int n,k;
	
	scanf("%d %d",&n,&k);
	
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);

		
	    int ans[100004];
		ans[0]=1000000000;
	    
	    for(int i=0;i<min(n,13);i++)
	    {
			for(int j=i+1;j<min(n,13);j++)
			{

				int temp[100004];
				

				int d=a[j]-a[i];
				int dd;

				if(d%(j-i)==0)
				dd=d/(j-i);

				int start=a[j]-j*dd-dd;

				if(d%(j-i)==0&&start<ans[0])
				{
					d=dd;

					int ctr=0;
					int x=0;
					for(x=0;x<n;x++)
					{
						if(start+d!=a[x])
						{
							temp[x]=start+d;
							ctr++;
						}
						
						else
						temp[x]=start+d;

						if(ctr>k)
						break;
						
						start+=d;
					}


					if(x==n)
					{
						for(int x=0;x<n;x++)
						ans[x]=temp[x];
					}
				}
			}
		}
	
		for(int i=0;i<n;i++)
		printf("%d ",ans[i]);


	return 0;
}
