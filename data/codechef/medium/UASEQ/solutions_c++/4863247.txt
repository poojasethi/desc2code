#include<iostream>
#include<cmath>
#include<algorithm>
#define mod 1000000007
using namespace std;
typedef long long int ll;
int main()
{
    int n,k,i,j,l,s,t;
    ll ap=mod,dp=mod,ax,dx;
    cin>>n>>k;
    ll a[n];
    for(i=0;i<n;i++)
        cin>>a[i];
    l=min(15,n);
    for(i=0;i<l;i++)
    {
        for(j=i+1;j<l;j++)
        {
            s=0;
            if((a[j]-a[i])%(j-i))
                continue;
            dx=(a[j]-a[i])/(j-i);
            ax=a[i]-(i*dx);
            for(t=0;t<n;t++)
                s+=(ax+t*dx!=a[t]);
            if(s<=k)
            {
                if(ap>ax)
                {
                    ap=ax;
                    dp=dx;
                }
                else if(ap==ax && dx<dp)
                    dp=dx;
            }
        }
    }
    for(i=0;i<n;i++)
        cout<<ap+(i*dp)<<" ";
    return 0;
}
