int main(int argc, char const *argv[])
{
	input();
	r.first = 999999999;
	r.second = 999999999;
	int n, k, d, a0;
	cin >> n >> k;
	for (int i = 0; i < n; ++i)
	{
		cin >> a[i];
	}
	for (int i = 0; i < k + 2; ++i)
	{
		for (int j = i + 1; j < k + 2; ++j)
		{
			if ((a[i] - a[j]) % (j - i) == 0) {
				d = (a[j] - a[i]) / (j - i);
				a0 = a[i] - d * i;
				//cout << i << " " << a0 << endl;
				int c = 0;
				if (a0 < r.first || (a0 == r.first && d < r.second)) 
				{
					for (int x = 0; x < n; x++) {
						if (a[x] != a0 + d * x) c++;
					}
					if (c <= k) {
						r.first = a0;
						r.second = d;
					}
				}
			} 
		}
	}
	a0 = r.first;
	d = r.second;
	//cout << a0 << " " << d << endl;
	for (int x = 0; x < n; x++) {
		cout << a0 + d * x << " ";
	}
	cout << endl;

}
