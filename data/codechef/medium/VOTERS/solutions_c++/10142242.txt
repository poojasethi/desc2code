#include<stdio.h>
#include<map>
#include<utility>
using namespace std;
int main()
{
	int a,b,c,d,i,x=0;
	scanf("%d%d%d",&a,&b,&c);
	map<int,int>vote=map<int,int>();
	vote.insert(make_pair(0,0));
	map<int,int>::iterator it;
	for(i=0;i<(a+b+c);i++)
	{
		scanf("%d",&d);
		if(vote.count(d)>0)
		{
			it = vote.find(d);
			it->second++;
			if(it->second>1 && it->second<3)
				x = x+1;
		}
		else
		vote.insert(make_pair(d,1));
		d=0;
	}
	printf("%d\n",x);
	for(map<int,int>::iterator it=vote.begin();it!=vote.end();it++)
	{
		if(it->second>1)
			printf("%d\n",it->first);
	}
	return 0;

}
