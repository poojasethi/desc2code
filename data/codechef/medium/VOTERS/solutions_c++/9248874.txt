#include <iostream>
#include <algorithm>
using namespace std;
int q[150000];
int d[100000];
int main()
{
 long a,b,c;
 cin>>a>>b>>c;
 long n=a+b+c;
 for(long i=0;i<n;i++)
 cin>>q[i];
 sort(q,q+n);
 long m=0;
 //long d[a+b];
 for(long j=0,i=0;i<n;)
 {
  if(q[i]==q[i+1])
  {
      d[j]=q[i];
      j++;
      m++;
      if(q[i]==q[i+2])
        i=i+3;
      else
            i=i+2;
  }
  else
    i++;
 }
 cout<<m<<endl;
 for(long j=0;j<m;j++)
    cout<<d[j]<<endl;
return 0;
}
