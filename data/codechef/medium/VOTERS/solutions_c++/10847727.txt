using namespace std;
 
#define gc getchar_unlocked
#define pc putchar_unlocked
 
inline void scanint(int &x)
{
  char c = gc();
 
  while(c<'0' || c>'9')
   c = gc();
 
  x=0;
 
  while(c>='0' && c<='9')
  {
   x = 10 * x + c - 48;
   c = gc();
  }
 
}
 
inline void fastWrite(int a)
{
 char snum[12];
 int i=0;
 
 do
 {
  snum[i++]=a%10+'0';
  a=a/10;
 }while(a);
 
 i=i-1;
 
 
 while (i>=0)
 {
  pc(snum[i--]);
 }
}
 
 
 
int main()
{
    //freopen("in.txt","r",stdin);
    register int t,n,k,i,j,x;
    int n1,n2,n3;
    scanint(n1);
    scanint(n2);
    scanint(n3);
 
    vector<int>a(n1),b(n2),c(n3),d;
 
    for(i=0;i<n1;i++)
    {
        scanint(a[i]);
    }
 
    for(i=0;i<n2;i++)
    {
        scanint(b[i]);
    }
 
    for(i=0;i<n3;i++)
    {
        scanint(c[i]);
    }
 
    i=0;j=0;k=0;
    while(!(i==n1 && j==n2 && k==n3))
    {
        //select the minimum and check its frequency
        if(i<n1)
            x=a[i];
 
        else
            x=INT_MAX;
 
        if(j<n2)
            if(b[j]<x)
                x=b[j];
 
        if(k<n3)
            if(c[k]<x)
                x=c[k];
 
        //count frequency and move forward
        t=0;
        if(i<n1)
        if(a[i]==x)
        {
            t++;
            i++;
        }
 
        if(j<n2)
        if(b[j]==x)
        {
            t++;
            j++;
        }
 
        if(k<n3)
        if(c[k]==x)
        {
            t++;
            k++;
        }
 
        if(t>=2)
            d.push_back(x);
 
    }
 
    fastWrite(d.size());
    pc('\n');
 
    for(i=0;i<d.size();i++)
    {
        fastWrite(d[i]);
        pc('\n');
    }
 
    return 0;
}
