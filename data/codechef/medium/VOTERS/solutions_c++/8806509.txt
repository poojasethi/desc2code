#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
int a[5000002]={0};

int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */
    int n1,n2,n3,x,count=0,i;

    scanf("%d%d%d",&n1,&n2,&n3);
    n1+=n2+n3;

    while(n1--){

        scanf("%d",&x);
        a[x]++;
        if(a[x]==2){
            count++;
        }
    }
    printf("%d\n",count);
    for(i=1;i<=5000000;++i){
        if(a[i]>=2) printf("%d\n",i);
    }
    return 0;
}
