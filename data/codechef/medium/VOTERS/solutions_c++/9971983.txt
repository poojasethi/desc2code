#include <iostream>
#include<algorithm>
using namespace std;
int main()
{
    long long int a,b,c,i,j,k,count=0;
    cin>>a>>b>>c;
    long long int n=a+b+c;
    long long int arr[n],ans[n];
    j=0;
    for(i=0;i<a;i++)
    {
        cin>>arr[j];
        j++;
    }
    for(i=0;i<b;i++)
    {
        cin>>arr[j];
        j++;
    }
    for(i=0;i<c;i++)
    {
        cin>>arr[j];
        j++;
    }
    sort(arr,arr+n);
    k=0;
    for(i=0;i<n-1;i++)
    {
        if(arr[i+1]==arr[i])
        {
            count++;
            ans[k]=arr[i];
            k++;
            if(arr[i+2]==arr[i])
                i=i+2;
            else
                i=i+1;
        }
    }
    cout<<count<<"\n";
    for(i=0;i<k;i++)
        cout<<ans[i]<<"\n";
    return 0;
}
