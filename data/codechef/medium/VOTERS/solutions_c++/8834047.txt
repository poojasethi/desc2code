void setunion(int a1[50000], int a2[50000], int N1, int N2, int results[50000], int *resnum)
{
    int i, j, k;
    *resnum=0;
    
    for(i=0, j=0;;)
    {
        if(i>=N1) {
            while(j<N2)
                results[(*resnum)++] = a2[j++];
            break;
        }
        if(j>=N2) {
            while(i<N1)
                results[(*resnum)++] = a1[i++];
            break;
        }

        if(a1[i]==a2[j]) {
            results[*resnum] = a1[i];
            *resnum  = *resnum + 1;
            i++, j++;
        }
        else if(a1[i] > a2[j]){
            results[(*resnum)++] = a2[j];
            j++;
        }
        else {
            results[(*resnum)++] = a1[i];
            i++;
        }

    }
}
void intersection(int a1[50000], int a2[50000], int N1, int N2, int results[50000], int *resnum)
{
    int i, j, k;
    *resnum=0;
    
    for(i=0, j=0;;)
    {
        if(i>=N1 || j>=N2)
            break;

        if(a1[i]==a2[j]) {
            results[*resnum] = a1[i];
            *resnum  = *resnum + 1;
            i++, j++;
        }
        else if(a1[i] > a2[j])
            j++;
        else
            i++;

    }
}
