#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define in(t) scanf("%lld", &t)
#define rep(i,n) for(i=0;i<n;i++)
map<ll, int> m;
map<ll, int>::iterator mi;

int main(){
	vector<ll> v;
	ll s=0,n1,n2,n3,i,j;
	in(n1);
	in(n2);
	in(n3);
	rep(i,(n1+n2+n3))
	{
		int n;
	cin>>n;
	mi = m.find(n);
	if( mi == m.end() )
		m.insert(make_pair(n,1));
	else
		mi->second++;
		if( mi->second==2 )
		{
			s++;
			v.push_back(n);
		}
	}
	sort(v.begin(),v.end());
	printf("%lld\n",s);
	rep(i,v.size())
	{
		cout<<v[i]<<endl;
	}
	return 0;
}
