#include <bits/stdc++.h>
using namespace std;
int arr[1000000];
int main() {
    int i,max=0,ans=0,tmp,n1,n2,n3;
        cin>>n1>>n2>>n3;
    for(i=0;i<n1+n2+n3;i++){
        cin>>tmp;
        arr[tmp]++;
        if(max<tmp)
            max=tmp;
    }
    for(i=0;i<=max;i++){
        if(arr[i]>=2)
            ans++;
    }
    cout<<ans<<"\n";
    for(i=0;i<=max;i++)
        {
        if(arr[i]>1)
            cout<<i<<"\n";
    }
                   
    
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    return 0;
}
