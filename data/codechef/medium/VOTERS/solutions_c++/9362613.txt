#include<bits/stdc++.h>

using namespace std;

int main(void)
{
    int n1,n2,n3; scanf("%d %d %d",&n1,&n2,&n3);
    
    map<int,int> mymap;
    
    for(int i=0;i<n1;i++)
    {
       int temp; scanf("%d",&temp);
       if(mymap.find(temp)==mymap.end())
       {
          mymap[temp]=1;
       }
       else
       {
          mymap[temp]++;
       }
    }
    for(int i=0;i<n2;i++)
    {
       int temp; scanf("%d",&temp);
       if(mymap.find(temp)==mymap.end())
       {
          mymap[temp]=1;
       }
       else
       {
          mymap[temp]++;
       }
    }
    for(int i=0;i<n3;i++)
    {
       int temp; scanf("%d",&temp);
       if(mymap.find(temp)==mymap.end())
       {
          mymap[temp]=1;
       }
       else
       {
          mymap[temp]++;
       }
    }
    
    map<int,int>::iterator it=mymap.begin();
    vector<int> voters;
    while(it!=mymap.end())
    {
        if(it->second>=2)
           voters.push_back(it->first);
        it++;
        
    }
    printf("%d\n",voters.size());
    for(int i=0;i<voters.size();i++)
    {
        printf("%d\n",voters[i]);
    }
 }
