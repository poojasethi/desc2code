#include<cstdio>
#define INF 1000000002
using namespace std;
int MAX(int a,int b){
    if(a<b){
        return b;
    }
    return a;
}
int MIN(int a,int b){
    if(a>b){
        return b;
    }
    return a;
}

int main(){
        int N1,N2,N3;
        scanf("%d %d %d",&N1,&N2,&N3);
        int A1[N1+1],A2[N2+1],A3[N3+1];
        for(int i=0;i<N1;i++){
            scanf("%d",&A1[i]);
        }
        for(int i=0;i<N2;i++){
            scanf("%d",&A2[i]);
        }
        for(int i=0;i<N3;i++){
            scanf("%d",&A3[i]);
        }
        A1[N1]=INF;
        A2[N2]=INF;
        A3[N3]=INF;
        int ANS[MAX(N1,MAX(N2,N3))],size=0,i=0,j=0,k=0;
        while(i<N1 || j<N2 || k<N3){
            int min;
            min=A1[i];
            min=MIN(min,A2[j]);
            min=MIN(min,A3[k]);
            int count=0;
            if(min==A1[i]){
                i++;
                count++;
            }
            if(min==A2[j]){
                j++;
                count++;
            }
            if(min==A3[k]){
                k++;
                count++;
            }
            if(count>1){
                ANS[size++]=min;
            }
        }
        printf("%d\n",size);
        for(int i=0;i<size;i++){
            printf("%d\n",ANS[i]);
        }
        }
