#include<bits/stdc++.h>
using namespace std;
int has[9999999];
int main()
{
	//freopen("in","r",stdin);
	long n1,n2,n3;
	cin>>n1>>n2>>n3;
	long a;
	long total=n1+n2+n3;
	for(long i=0;i<total;i++)
	{
		cin>>a;
		has[a]++;
	}
	vector<long>foo;
	int c=0;
	for(long i=0;i<9999999;i++)
	{
     if(has[i]>=2)
     {
     	c++;
     	foo.push_back(i);
     }
	}
	cout<<c<<endl;
	for(int i=0;i<foo.size();i++)
	{
		cout<<foo[i]<<endl;;
	}
}
