#include<bits/stdc++.h>
#define L long long
using namespace std;
int main()
{
    int n1,n2,n3;
    int c=0,t;
    scanf("%d %d %d",&n1,&n2,&n3);
    int a[1000000]={0};
    for(int i=0;i<(n1+n2+n3);i++)
    {
        scanf("%d",&t);
        a[t]++;
        if(a[t]==2)
            c++;

        if(a[i]>1)

    }
    return 0;
}
