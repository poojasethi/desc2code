#include <iostream>
#include <map>
using namespace std;

int main() {
	map<int,int> m;
	map<int,int>::iterator it;
	int term,a,b,c,i,temp,count,c2=0;
	scanf("%d %d %d",&a,&b,&c);
	for(i=0;i<(a+b+c);i++)
	{
		scanf("%d",&temp);
		if(m.find(temp)==m.end())
		{
			m[temp]=1;
		}
		else
		{
			m[temp]++;
		}
	}
	for(it=m.begin();it!=m.end();it++)
	{
		count=it->second;
		if(count>=2)
		{
			c2++;
		}
	}
	printf("%d\n",c2);
	for(it=m.begin();it!=m.end();it++)
	{
		term=it->first;
		count=it->second;
		if(count>=2)
		{
			printf("%d\n",term);
		}
	}
	return 0;
}
