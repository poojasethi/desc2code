#include <bits/stdc++.h>
using namespace std;

int main() {
	int a,b,c;
	cin>>a>>b>>c;
	map<long long,int> mp;
	for(int i=0;i<a+b+c;i++)
	{
		int a;
		cin>>a;
		mp[a]++;
	}
	map<long long,int>::iterator it=mp.begin();
	vector<long long> ans;
	while(it!=mp.end())
	{
		if(it->second>=2)
		ans.push_back(it->first);
	it++;}
	cout<<ans.size()<<endl;
	for(int i=0;i<ans.size();i++)
	cout<<ans[i]<<endl;
	
	return 0;
}
