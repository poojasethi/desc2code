#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int a,b,c;
	scanf("%lld%lld%lld",&a,&b,&c);
	long long int count=0,i,j,d[1000000]={0};
	for(i=0;i<a+b+c;i++)
	{
		scanf("%lld",&j);
		d[j]++;
	}
	for(i=0;i<1000000;i++)
	{
		if(d[i]>1)
		count++;
	}
	printf("%lld\n",count);
	for(i=0;i<1000000;i++)
	{
		if(d[i]>1)
		printf("%lld\n",i);
	}
	return 0;
	
	
}
