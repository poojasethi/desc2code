#include<stdio.h>

int main(){
	int n1,n2,n3,total,count=0,i,max=0,temp;
	char id[1000001]={0};
	scanf("%d%d%d",&n1,&n2,&n3);
	total=n1+n2+n3;
	while(total--){
		scanf("%d",&temp);
		id[temp]++;
		if(id[temp]==2)
			count++;
		if(temp>max)
			max=temp;
	}
	printf("%d\n",count);
	for(i=1;i<=max;i++)
		if(id[i]>1)
			printf("%d\n",i);
	return 0;
}
