#include<iostream>
#include<algorithm>
using namespace std;
int main()
 {
  int count =0,j,num,n1,n2,n3,a[150000],o[50000],k=0;
  cin>>n1>>n2>>n3;
  for(int i=0;i<(n1+n2+n3);i++)
   cin>>a[i];
  sort(a,a+n1+n2+n3);  
  for(int i=0;i<n1+n2+n3;)
   {
    num=a[i];
    j=i;
    while(num==a[j])
     j++;
    if(j-i>=2)
     {
      count++;
      o[k]=a[i];
      i=j;
      k++;
     } 
    else
     i++; 
   }
  cout<<count<<endl;
  for(int i=0;i<count;i++)
   cout<<o[i]<<endl; 
  return 0; 
 }
