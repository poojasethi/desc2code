#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int n1,n2,n3,count=0;
	cin>>n1>>n2>>n3;
	long int a[n1+2],b[n2+2],c[n2+2];
	int hash[1000000];
	for(int i=0;i<1000000;i++)
        hash[i]=0;
	for(int i=0;i<n1;i++)
	{
	cin>>a[i];
	hash[a[i]]++;
	}
	for(int i=0;i<n2;i++)
	{
	cin>>b[i];
	hash[b[i]]++;
	}
	for(int i=0;i<n3;i++)
	{
	cin>>c[i];
	hash[c[i]]++;
	}
	for(int i=0;i<1000000;i++)
	{
	if(hash[i]>=2)
	count++;
	}
	cout<<count<<endl;
	for(int i=0;i<1000000;i++)
	{
	if(hash[i]>=2)
	cout<<i<<endl;
	}
	return 0;
}
