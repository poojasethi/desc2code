n=raw_input()
l=n.split()
l=list(map(int,l))
n1,n2,n3=l[0],l[1],l[2]
l1,l2,l3=[],[],[]
for i in range(n1):
    l1.append(int(input()))
for j in range(n2):
    l2.append(int(input()))
for k in range(n3):
    l3.append(int(input()))
s1,s2,s3=set(l1),set(l2),set(l3)
s4=s1.intersection(s2)
s5=s2.intersection(s3)
s6=s3.intersection(s1)
s=(s4.union(s5)).union(s6)
lis=list(s)
m=len(lis)
print m
lis.sort()
for w in range(m):
    print lis[w]
