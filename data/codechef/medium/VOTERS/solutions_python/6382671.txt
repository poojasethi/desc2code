from sys import stdin

N1, N2, N3 = map(int, stdin.readline().split())
L1 = []
for _ in range(N1):
	L1.append(int(stdin.readline()))
L2 = []
for _ in range(N2):
	L2.append(int(stdin.readline()))
L3 = []
for _ in range(N3):
	L3.append(int(stdin.readline()))

L = sorted(L1+L2+L3)
result = []
i = 0
while i < len(L)-1:
	if L[i] == L[i+1]:
		result.append(L[i])
		i += 2
	else:
		i += 1
print len(result)
for i in result:
	print i
