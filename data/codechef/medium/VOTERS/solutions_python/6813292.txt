# nbNumbers = reduce(lambda x, y: x + y, map(int, raw_input().strip().split()), 0)
nbNumbers = sum(map(int, raw_input().strip().split()))
records = dict()
total = 0

for index in range(nbNumbers):
    num = int(raw_input().strip())
    records.setdefault(num, 0)
    records[num] += 1
    if records[num] == 2:
        total += 1

print(total)
for num in sorted(records.keys()):
    val = records[num]
    if val > 1:
        print(num)
