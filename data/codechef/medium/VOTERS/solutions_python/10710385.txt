def test():
    
    n1,n2,n3 = map(int,raw_input().split())
    dictionary = {}
    total = n1+n2+n3
      
    for e in range(total):
        tmp = int(raw_input())
        if tmp not in dictionary:
            dictionary[tmp] = 1
        else:
            dictionary[tmp] += 1

    result = [e for e in dictionary.keys() if dictionary[e] > 1]
    result = sorted(result)
    print len(result)
    for e in result:
        print e
                

test()
