n1, n2, n3 = map(int,raw_input().split())
total = n1 + n2 + n3
idSet = set()
finalSet = set()
for _dummy in range(total):
	id = int(raw_input())
	if id in idSet:
		finalSet.add(id)
	else:
		idSet.add(id)
finalList = list(finalSet)
finalList.sort()
print len(finalList)

	print id
