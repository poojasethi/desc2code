numbers = []
solution = []
n1, n2, n3 = map(int, raw_input().split())
for i in xrange(n1 + n2 + n3):
	numbers.append(int(raw_input()))
#print numbers
prev = -1
count = 0
for number in sorted(numbers):
	if number == prev: count = count + 1
	else: 
		prev = number
		count = 1
	if count == 2: solution.append(number)
#print solution
print len(solution)
for num in solution: print num
