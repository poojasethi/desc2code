import sys

final , counted , ans = {} ,{} , []
for i in range(sum(map(int,sys.stdin.readline().split()))) :
    k = input()
    if not final.has_key(k) :
        final[k] = 1
    else :
        final[k] += 1
ans = []
for i,j in zip(final.keys(),final.values()) :
    if j >= 2 :
        ans.append(i)
print len(ans)
print "\n".join(str(i) for i in sorted(ans))
