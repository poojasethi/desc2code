using namespace std;

bool marked[1000000];

int main()
{
	int T;
	scanf("%d", &T);
	for (int t = 0; t < T; t++)
	{
		memset(marked, false, sizeof(marked));
		int a, b, N;
		scanf("%d %d %d", &a, &b, &N);
		printf("1 ");
		//marked[1] = true;
		int arr[N];
		arr[0] = 1;
		if (a > b)
			marked[a-b] = true;
		for (int i = 1; i < N; i++)
		{
			int j = arr[i-1]+1;
			while (marked[j])
				j++;
			printf("%d ", j);
			//marked[j] = true;
			arr[i] = j;
			for (int k = 0; k <= i; k++)
			{
				if (a*arr[k]-b*j >= 1)
					marked[a*arr[k]-b*j] = true;
				if (a*j-b*arr[k] >= 1)
					marked[a*j-b*arr[k]] = true;
			}
		}
		printf("\n");
	}
	return 0;
}
