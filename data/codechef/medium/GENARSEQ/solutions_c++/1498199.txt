#include <iostream>
#include <string>
#include <algorithm>
#include <set>
#include <vector>
#include <stdio.h>
#include <string.h>
#include <map>
#include <stdlib.h>
using namespace std;

bool used [1000000];
int main()
{
    int i,j,k,n,x,y,t,mm=0;
    scanf ("%d", &t);
    while (t != 0) {
        scanf ("%d %d %d", &x, &y, &n);

        vector <int> a (n, 0);
         memset(used, 0, sizeof(used));
        for (i = 0; i < n; i++) {
            for (j = 1; used[j]; j++);
            a[i] = j;
            used[j] = true;
            for (j = 0; j <= i; j++) {
                used[max (0, x*a[i] - y*a[j])] = true;
                used[max (0, x*a[j] - y*a[i])] = true;
            }
        }
        for (i = 0; i < a.size(); i++) {
            printf ("%d ", a[i]);
        }
        printf ("\n");

        t--;
    }
    return 0;
}
