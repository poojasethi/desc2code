#include <cstdio>
#include <iostream>
#include <vector>
#include <cstring>
using namespace std;
vector<int> vec;
bool mark[1000001];
int main(){
int t,a,b,n,z;
scanf("%d",&t);
while(t--){
 scanf("%d%d%d",&a,&b,&n);
 int chk=0;
 for(int i=0;i<n;i++){
  for(chk++;mark[chk];chk++);
  vec.push_back(chk);
  for(int j=0;j<=i;j++){
       z=a*vec[i]-b*vec[j];
       if(z<1000001 && chk<z)
         mark[z]=true;
      z=a*vec[j]-b*vec[i];
      if(z<1000001 && chk<z)
        mark[z]=true;
  }
 }
 for(int i=0;i<vec.size();i++){
    printf("%d ",vec[i]);
 }
 printf("\n");
 vec.clear();
 memset(mark,0,sizeof mark);
}
return 0;
}
