#include <cstdio>
#include <cstring>

bool used[1000000];

int main()
{
    int t;
    scanf("%d", &t);

    while(t--) {
        int n, a, b,  x[1002];
        scanf("%d %d %d", &a, &b, &n);
        memset(used, false, sizeof(used));

        x[1] = 1;
        putchar('1');
        if(a > b) used[a-b] = true;

        for(int k=2; k<=n; k++) {
            x[k] = x[k-1] + 1;
            while(used[x[k]]) x[k]++;
            for(int i=1; i<=k; i++) {
                int tmp = a*x[i] - b*x[k];
                if(tmp > 0) used[tmp] = true;
                tmp = a*x[k] - b*x[i];
                if(tmp > 0) used[tmp] = true;
            }
            printf(" %d", x[k]);
        }
        putchar('\n');
    }
}
