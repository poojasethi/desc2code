#include<cstdio>
#include<cstring>
#define MAXN 1001
bool used[2*MAXN*MAXN];
int main()
{
	int t,n,a,b,k,s,x[MAXN];
	scanf("%d",&t);
	while(t--)
	{
	scanf("%d %d %d",&a,&b,&n);
    for(int i=0;i<MAXN;i++)x[i]=0;
    memset(used, 0, sizeof (used));
	 for(k=1;k<=n;k++)
	 {
	    x[k]=x[k-1]+1;
	    while(used[x[k]])x[k]++;
        printf("%d ",x[k]);

	    for(int i=1;i<=k;i++)
	    {
           s=a*x[i]-b*x[k];
           if(s>0)used[s]=1;
           s=a*x[k]-b*x[i];
           if(s>0)used[s]=1;
	    }
	 }
    printf("\n");
	}
}
