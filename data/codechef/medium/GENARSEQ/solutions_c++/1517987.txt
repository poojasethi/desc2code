#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cassert>
using namespace std;

const int N = 2000000;

int x[1005];
bool ok[N];
int a, b, n;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
        scanf("%d %d %d", &a, &b, &n);
        for (int i=0; i<N; i++) ok[i] = true;
        int cnt = 0;
        for (int i=1; i<=N && cnt<n; i++) if (ok[i]) {
            cnt++;
            x[cnt] = i;
            for (int j=1; j<=cnt; j++) {
                int temp = a*x[j]-b*x[cnt];
                if (temp>0 && temp<N)
                    ok[temp] = false;
                temp = a*x[cnt]-b*x[j];
                if (temp>0 && temp<N)
                    ok[temp] = false;
            }
        }
        for (int i=1; i<n; i++) printf("%d ", x[i]);
        printf("%d \n", x[n]);
    }
    return 0;
}
