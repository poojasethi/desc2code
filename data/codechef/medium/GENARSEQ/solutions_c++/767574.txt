#define DEBUG 0
#define MAX 60000
#include <stdio.h>
#include <string.h>
using namespace std;

bool skip[MAX];
int x[1000];

int main(void) {
  int tc, a, b, n, i, k, p;

  scanf("%d", &tc);
  while(tc--) {
    scanf("%d%d%d", &a, &b, &n);
    x[0]=1;
    printf("1");
    memset(skip, false, sizeof(skip));
    if(a-b>1)
      skip[a-b]=true;
    for(k=1;k<n;k++) {
      for(x[k]=x[k-1]+1;skip[x[k]]==true;x[k]++);
      printf(" %d", x[k]);
      for(i=0;i<k;i++) {
	if((p=a*x[i]-b*x[k])>x[k] && p<MAX)
	  skip[p]=true;
	if((p=a*x[k]-b*x[i])>x[k] && p<MAX)
	  skip[p]=true;
      }
      if((p=a*x[k]-b*x[k])>x[k] && p<MAX)
	skip[p]=true;
    }
    printf("\n");
  }
  return 0;
}
