typedef long long ll;
typedef vector<int> vi;
 
const int N = 1000000;
bool used[N];
 
main() {
    ios_base::sync_with_stdio(false);
 
    int t;
    while (cin >> t && t) {
        rep(times, t) {
            int a, b, n;
            cin >> a >> b >> n;
 
            vector<int> ret(n);
            memset(used, 0, sizeof(used));
 
            int cur = 0;
            for (int i = 0; i < n; ++i) {
                for (cur++; used[cur]; cur++);
                ret[i] = cur;
 
                for (int j = 0; j <= i; ++j) {
                    used[max(0, ret[i] * a - ret[j] * b)] = true;
                    used[max(0, ret[j] * a - ret[i] * b)] = true;
                }
            }
            rep(i, sz(ret)) {
                if (i) cout << " ";
                cout << ret[i];
            }
            cout << endl;
        }
    }
}
