const double eps=1e-9;
const ld leps=1e-14;

const int MN=1000+100;
const int MR=MN*MN+MN+100;
int t,a,b,n;
int now[MN];
int p=0;
int mark[MR];
int gl;
int calc(int a)
{
	while (mark[p]==gl)
		p++;
	return p;
}
int main(int argc,char * argv[])
{
	ios_base::sync_with_stdio(false);
	cin>>t;
	while (t-->0)
	{
		gl=2*(t+1)+1;
		p=1;
		cin>>a>>b>>n;
		for (int i=0;i<n;i++) 
		{
			cout<<(now[i]=calc(i))<<" ";
			p++;
			for (int j=0;j<=i;j++)
			{
				int tmp=a*now[i]-b*now[j];
				if (tmp>=p && tmp<MR)
					mark[tmp]=gl;
				tmp=a*now[j]-b*now[i];
				if (tmp>=p && tmp<MR)
					mark[tmp]=gl;
			}
		}
		cout<<endl;
	}
	return 0;
}
