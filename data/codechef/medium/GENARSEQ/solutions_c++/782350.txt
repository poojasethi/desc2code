#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <set>
#include <map>
#include <cmath>
#include <cstdlib>

using namespace std;

typedef long long ll;

int a,b,n,T;
int x[1234],mark[2345678];

int main(){
	
	cin >> T;

	for (int o=1; o<=T; o++){
		cin >> a >> b >> n;
		x[1]=1;
		if (a-b > 0)
			mark[a-b] = o;
		for (int i=2; i<=n; i++){
			int now = x[i-1]+1;
			while (mark[now]==o)
				now++;
			x[i] = now;
			for (int j=1; j<=i; j++){
				int tmp1 = a*x[j] - b*x[i];
				int tmp2 = a*x[i] - b*x[j];
				if (tmp1>x[i] && tmp1<2345678)
					mark[tmp1] = o;
				if (tmp2>x[i] && tmp2<2345678)
					mark[tmp2] = o;
			}
		}
		for (int i=1; i<=n; i++)
			cout << x[i] << ' ';
		cout << endl;
	}

	return 0;
}
