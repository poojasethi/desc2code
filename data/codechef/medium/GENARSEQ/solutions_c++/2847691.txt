#include<iostream>
#include<stdio.h>
#include<cmath>
#include<cstring>
#include<stdlib.h>
#include<algorithm>
#define getcx getchar
using namespace std;
inline int in()
{
   int n=0;
   int ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   return n*sign;
}
int temp[1111000],x[1001];
int main()
{
	int t,i,j,a,b,n,k;
	t=in();
	while(t--)
	{
	    a=in();
	    b=in();
		n=in();
		x[0]=1;
        memset(temp,0,sizeof(temp));
		for(k=1;k<n;k++)
		{
           for(i=0;i<k;i++)
           {
                int ans1=(a*x[k-1])-(b*x[i]),ans2=(a*x[i])-(b*x[k-1]);
                if(ans1>x[k-1])
                    temp[ans1]=1;
                if(ans2>x[k-1])
                    temp[ans2]=1;
       //      if(ans1>6000 || ans2>6000)
       //         cout<<"ans1="<<ans1<<" ans2="<<ans2<<endl;
           }
           for(i=x[k-1]+1;i<100000;i++)
           {
                if(temp[i]==0)
                {
                    x[k]=i;
                    break;
                }
           }
        }
        for(i=0;i<n;i++)
            printf("%d ",x[i]);
        printf("\n");
	}
	return 0;
}
