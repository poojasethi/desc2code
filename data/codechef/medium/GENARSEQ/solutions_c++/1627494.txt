#include <cstdio>
#include <cstring>

using namespace std;

const int MAX_N=1000;
const int BOUND=MAX_N*MAX_N+2*MAX_N;

int seqs[MAX_N+1];
bool marked[BOUND];

void solve(int a, int b, int n)
{
    memset(marked, false, BOUND);
    seqs[0] = 1;
    int cnt = 1;
    for(; cnt<n; cnt++)
    {
	for(int j=0; j<cnt; j++)
	{
	    int z;
	    z = a*seqs[cnt-1]-b*seqs[j];
	    if (z > seqs[cnt-1])
		marked[z] = true;

	    z = a*seqs[j] - b*seqs[cnt-1];
	    if (z > seqs[cnt-1])
		marked[z] = true;
	}

	for(int j=seqs[cnt-1]+1; j<BOUND; j++)
	{
	    if (!marked[j])
	    {
		seqs[cnt] = j;
		marked[j] = true;
		break;
	    }
	}
    }

    for(int i=0; i<n; i++)
	printf("%d ", seqs[i]);
    printf("\n");
}

int main()
{
    int T;
    scanf("%d", &T);
    for(; T>0; T--)
    {
	int a, b, n;
	scanf("%d %d %d", &a, &b, &n);
	solve(a, b, n);
    }
    return 0;
}
