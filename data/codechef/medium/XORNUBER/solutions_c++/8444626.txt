    #include <bits/stdc++.h>
    using namespace std;
    #define ll long long
    int main()
    {
    	ll t; scanf("%lld", &t);
    	while (t--){
    		ll n; scanf("%lld", &n);
    		if (n == 1) printf("2\n");
    		else{
    			if (__builtin_popcountll(n + 1) == 1) printf("%lld\n", n / 2);
    			else printf("-1\n");
    		}
    	}
    	return 0;
    }
