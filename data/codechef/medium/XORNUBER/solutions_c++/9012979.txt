#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,i;
    cin>>t;
    while(t--)
    {
        long long n;
        cin>>n;
        if(n==1)
        {
            cout<<"2\n";
        }
        else
        {
        int temp=int(log2(n))+1;
        int br=1;
        for(i=0;i<temp;i++)
        {
            if(n%2==0)
            {
                br=1;
                break;
            }
            n/=2;
        }
        if(i==temp)
            br=0;
        long long ans=1;
        for(i=1;i<=temp-1;i++)
            ans*=2;
        ans-=1;
        if(br==0)
            cout<<ans<<"\n";
        else
            cout<<"-1\n";
        }
    }
}
