#include<bits/stdc++.h>
using namespace std;
int main()
{
    
    int t,n;
    cin>>t;
    
    while(t--)
    {
        
        cin>>n;
        
        if((n)&(n+1))
        {
        printf("-1\n");
        }
        else
        {
          if((n)==1)
           cout<<2<<endl;
           else
           cout<<n/2<<endl;
            
        }
        
    }
    
    
    return 0;
}
