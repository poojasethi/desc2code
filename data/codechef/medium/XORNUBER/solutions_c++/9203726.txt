
#include <iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cmath>


 typedef unsigned long long int lli;
using namespace std;




int main(){
    lli a[33],power[33],ans[33];

    int i,j;
power[1]=1;
for(i=2;i<=32;i++) power[i]=power[i-1]*2;
a[1]=1; a[2]=3;
for(i=3;i<=32;i++)a[i]=a[i-1]+power[i];
ans[1]=2; ans[2]=1;
for(i=3;i<=32;i++){
        int flag=0;  if(power[i]^power[i-1]==a[i])
for(j=power[i]+1;flag!=1;j+=2){
         if(power[i]^(power[i]-1)==a[i]){ ans[i]=power[i]-1; break;    }
    if(j^(j+1)==a[i]) { ans[i]=j;   flag=1;    }

}
}

int t;
cin>>t;


while(t--){
  lli n;
  cin>>n;

  for(i=1;i<=32;i++){
  if(a[i]==n) {
    cout<<ans[i]<<"\n";
    break;
  }
  if(i==32) cout<<"-1"<<"\n" ;
  }
}

return 0;
}
