using namespace std;

long long t,n,i,ans,temp;

int main()
{
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==1)
        {
            cout<<2<<endl;
            continue;
        }
        ans=-1;
        for(i=0;i<=30;i++)
        {
            temp=(1<<i);
            if((temp^(temp-1))==n)
            {
                ans=temp-1;
                break;
            }
        }
        cout<<ans<<endl;
    }
}
