#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n;
        scanf("%lld",&n);
        bool flag =false;
        if(n==1)
            printf("2\n");
        else
        {
            for(long long int i=2;i<=30;i++ )
            {
                if(n+1==pow(2,i))
                {
                    flag =true;
                    break;
                }
            }
            if(flag)
            {
                printf("%lld\n",(n/2));
            }
            else
                printf("-1\n");

        }
    }
    return 0;
}
