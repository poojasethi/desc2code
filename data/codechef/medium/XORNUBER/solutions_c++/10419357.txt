long long fast_exp(ll a, ll b)
{
    long long x=1,y=a;
    while(b > 0)
    {
        if(b%2 == 1)
        {
            x=(x*y);
            if(x>MOD) x%=MOD;
        }
        y = (y*y);
        if(y>MOD) y%=MOD;
        b /= 2;
    }
    return x;
}

 ll GCD(ll a, ll b)
{
    if (a == 0)

    return GCD(b%a, a);
