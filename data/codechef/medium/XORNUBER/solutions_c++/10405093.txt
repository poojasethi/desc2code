#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<vector>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long int n;
        scanf("%lld",&n);

        if(n==1)
        {
            printf("%d\n",2);
            continue;
        }
        int ans=-1;
        for(int i=1;i<=30 && ans<0;i++)
        {
            int p=(1<<i);

            if( (p ^ (p-1))==n)
                ans=p-1;
        }
        printf("%d\n",ans);
    }


        return 0;
}
