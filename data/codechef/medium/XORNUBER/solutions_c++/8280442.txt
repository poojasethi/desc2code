	#include<iostream>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			int n;
			cin>>n;
			n=n+1;
			int temp=n;
			int flag=0;
			while(n>1)
			{
				if(n%2==1)
					flag=1;
				n/=2;
			}			
			if(flag==1)
				cout<<"-1\n";
			else if(temp!=2)
				cout<<(temp-1)/2<<endl;
			else
				cout<<"2\n";
			t--;
		}	
	}
