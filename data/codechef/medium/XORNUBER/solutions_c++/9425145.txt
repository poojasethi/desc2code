#include <bits/stdc++.h>
using namespace std;

#define maxn (1<<30)

int main()
{
	int t;
	cin >> t;

	assert(t >= 1 && t <= 5000);

	while(t--)
	{
		int n,i;
		cin >> n;	
		assert(n >= 1 && n <= maxn);

		if(n == 1)
		{
			cout << "2" << "\n";
			continue;
		}

		int ans = -1;

		for(i = 1;i <= 30 && ans < 0;i++) {
			int p = (1 << i);

			if((p ^ (p-1)) == n)
				ans = p-1;
		}
		cout << ans << "\n";
	}
	return 0;
}
