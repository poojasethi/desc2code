#include<iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        signed long long int x;
        cin>>x;
        signed long long int m;
        m=x/2;
        if(x==1)
        {
            cout<<2<<"\n";
            continue;

        }
        if((m^(m+1))==x && m>0)
            cout<<m<<"\n";
        else
            cout<<-1<<"\n";


    }

    return 0;
}
