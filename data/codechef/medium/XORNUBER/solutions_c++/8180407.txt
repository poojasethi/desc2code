#include<stdio.h>
 
void getBin(long long int num,int *arr, int *pos) {
	int rem,val;
	if(num==0)
		return ;
	rem = num%2;
	getBin(num/2,arr,pos);
	val=*pos;
	arr[*pos]=rem;
	(*pos)++; 
}

	int min,max,t,i,m,b,arr[32],pos=0,f;
	long long int n,n1;

		pos=0;
		if(n==1) {

		if(n%2==0) {
