bool is_ans_possible(int x){
      int one=0,zero=0;
      while(x){
            if(x%2){
                  one++;
            }
            else
            {
                  zero++;
            }
            x=x/2;
      }
      if(zero==0){
            return true;
      }
      return false;
}
int main() {
      
      int t;
      scanf("%d",&t);
      while(t--){
            int x;
            scanf("%d",&x);
            if(is_ans_possible(x)){
                  if(x!=1)
                  printf("%d\n",(((x+1)/2)-1));
                  else
                  printf("2\n");
            }
            else
