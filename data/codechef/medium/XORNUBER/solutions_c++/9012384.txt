LL power(LL b,LL e,LL MOD)
{
	if(b==0 && e==0) return 1;
	if(b==0 && e>1)  return 0;
	if(b>0 && e==0) return 1;
	LL r=1;
	while(e)
	{
		if(e&1) { r = r*b;  r %= MOD; }
		e >>= 1; b = b*b; b %= MOD;

	}
return r;
}
