#include <iostream>

using namespace std;

int main()
{
   int t,n,i;
   cin>>t;
   while(t--)
   {
       cin>>n;
       if(n==1)
       cout<<2<<"\n";
      else if(n%2==0)
       cout<<-1<<"\n";
       else if(n%2==1)
       {
          i=n>>1;
          if((i^(i+1))==n)
            cout<<i<<"\n";
          else
            cout<<-1<<"\n";
        }
   }
    return 0;
}
