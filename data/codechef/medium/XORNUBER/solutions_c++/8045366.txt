#include <stdio.h>
 
int main() {
	
    int T;
    long long int N,temp,flag;
    
    scanf("%d",&T);
    while(T--){
        scanf("%lld",&N);
        flag=0;
        if(N==1)
          printf("%d\n",2);
        else if(N%2==0)
            printf("-1\n");
        else{
            temp=N;
            while(temp>0){
                if(temp%2==1){
                    temp=temp/2;
                }else{
                    flag=1;
                    break;
                }
            }
            if(flag==0)
                printf("%lld\n",N/2);
            else
                printf("-1\n");
        }
    }
	return 0;
}
