    #include<iostream>
    #include<string.h>
    #include<math.h>
    using namespace std;
     
    int main()
    {
    	ios_base::sync_with_stdio(0);
    	
    	int t,c;
    	long long n,m,r;
    	
    	cin>>t;
    	while(t--)
    	{
    		cin>>n;
    		
    		if(((n+1)&(n)))
    			cout<<"-1\n";
    		else
    		{
    			m=n;
    			c=0;
    			while(m>0)
    			{
    				c++;
    				m/=2;
    			}
    			r=(pow(2,c-1)-1);
    			cout<<(r>0?r:2)<<"\n";	
    			
    		}
    	}
    	
    	return 0;
    }
