    #include<bits/stdc++.h>
    using namespace std;
    int main()
    {
        long long int a[35],n;
        int t,i;
        a[1]=1;
        for(i=2;i<=34;i++)
            a[i]=2*a[i-1]+1;
        cin>>t;
        while(t--)
        {
            cin>>n;
            if(n==1)
                cout<<"2"<<endl;
            else if(n%2==0)
                cout<<-1<<endl;
            else
            {
                int j=1,flag=0;
                while(j<35)
                {
                    if(a[j]==n)
                    {
                        flag=1;
                        break;
                    }
                    if(a[j]>n)
                        break;
                    j++;
                }
                if(flag==0)
                    cout<<-1<<endl;
                else
                {
                    cout<<n/2<<endl;
                }
            }
        }
        return 0;
    }
