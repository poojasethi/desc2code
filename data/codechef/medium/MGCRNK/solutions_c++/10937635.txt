#include <bits/stdc++.h>

using namespace std;

int n;
int g[101][101];
int dp[101][101];
#define inf -987654321
int func(int i,int j)
{
	if (i == n-1 && j == n-1) {
		return 0;
	}
	if (i >= n) {
		//cout << i << endl;
		return inf;
	}
//	cout << "ayush3";
	if (j >= n)
		return inf;
//	cout << "ayush2";
	if (dp[i][j] != inf)
		return dp[i][j];
//	cout << "ayush";
	int k = g[i][j]+func(i+1,j);
	//cout << k << endl;
	int k1 = g[i][j]+func(i,j+1);
	//cout << k1 << endl;
	dp[i][j] = max(k,k1);
	return dp[i][j];
}
int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		scanf("%d",&n);
		for (int i = 0; i < n; i++)
		{
			for (int j = 0; j < n; j++)
			{
				scanf("%d",&g[i][j]);
				dp[i][j] = inf;
			}
		}
		int ans = func(0,0);
		if (ans < 0)
		{
			printf("Bad Judges\n");
			continue;
		}
		double k = double(2*n-3);
		double g = ans/k;
		printf("%lf\n",g);
	}
	return 0;
}
