#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    int t,n,i,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int a[n][n],s[n][n];
        for(i=0;i<n;i++)
            for(j=0;j<n;j++)
                scanf("%d",&a[i][j]);
        for(i=0;i<n;i++)
            for(j=0;j<n;j++)
            {
                if(i>0 && j>0)
                    s[i][j]=a[i][j]+(s[i][j-1]>s[i-1][j]?s[i][j-1]:s[i-1][j]);
                else if(i>0)
                    s[i][j]=a[i][j]+s[i-1][j];
                else if(j>0)
                    s[i][j]=a[i][j]+s[i][j-1];
                else
                    s[i][j]=a[i][j];
            }
        if(s[n-1][n-1]<0)
            printf("Bad Judges\n");
        else
            printf("%f\n",(float)s[n-1][n-1]/(2*n-3));
    }
    return 0;
}
