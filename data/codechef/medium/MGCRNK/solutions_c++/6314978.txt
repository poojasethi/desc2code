#include <iostream>
#include <algorithm>
using namespace std;

int minpath(int cost[][100],int n)
{
int tc[n+1][n+1],i,j;

tc[0][0] = cost[0][0];

for(i=1;i<=n;i++)
    tc[i][0] = tc[i-1][0] + cost[i][0];

for(j=1;j<=n;j++)
    tc[0][j] = tc[0][j-1] + cost[0][j];

for(i=1;i<=n;i++)
    for(j=1;j<=n;j++)
       tc[i][j] = cost[i][j] + max(tc[i-1][j],tc[i][j-1]);

return tc[n][n];
}

int main() 
{
int t;
cin>>t;
while(t--)
 {
 int n,i,j,s[100][100];
 double ans;
 cin>>n;
 for(i=0;i<n;i++)
    for(j=0;j<n;j++)
       cin>>s[i][j];
 ans = double(minpath(s,n-1))/(2*n-3);
 if(ans<0)
   cout<<"Bad Judges"<<endl;
 else
   printf("%.6f\n",ans);
 }
return 0;
}
