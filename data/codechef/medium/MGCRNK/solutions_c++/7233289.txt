#include<iostream>
#include<stdio.h>
using namespace std;
#define small -1000000000
int max(int a,int b)
{
	if(a>b)
	return a;
	else
	return b;
}
int main()
{
	int test,s[103][103],dp[103][103];
	int i,j,n;
	
	cin>>test;
	while(test--)
	{
		cin>>n;
		for(i=1;i<=n;++i)
		dp[0][i]=dp[i][0]=small; //0 subscript ignored cuz it has no judge
		
		for(i=1;i<=n;++i)
		{
			for(j=1;j<=n;++j)
			{
				cin>>s[i][j];
				if(i==1 && j==1)
				dp[i][j]=s[i][j];
				else
				dp[i][j]=max(dp[i-1][j],dp[i][j-1])+s[i][j];
			}
		}
		 double k=((double)(dp[n][n])/(2*n-3));
		if(dp[n][n]>=0)
				printf("%lf\n",k);
			else
				printf("Bad Judges\n");
					
	}
	return 0;
}
