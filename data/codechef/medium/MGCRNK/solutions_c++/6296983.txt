#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,j;
        scanf("%d",&n);
        int a[n][n];

        for(i=0;i<n;i++)
        for(j=0;j<n;j++)
               scanf("%d",&a[i][j]);

        for(i=1;i<n;i++)
        {
            a[0][i]+=a[0][i-1];
            a[i][0]+=a[i-1][0];
        }
        for(i=1;i<n;i++)
        for(j=1;j<n;j++)
        a[i][j]+=max(a[i][j-1],a[i-1][j]);

        float p= a[n-1][n-1]/1.0;
        p=p/(2*n-3);
        if(p<0)
            printf("Bad Judges\n");
        else
        printf("%.6f\n",p);

    }
    return 0;
}
