#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<vector>
#include<string>
#include<cstring>
#include<algorithm>

using namespace std;
#define REP(i,a,b) \
    for(int(i)=int(a);i<=int(b);i++)
#define REPMIN(i,a,b) \
    for(int(i)=int(a);i>=int(b);i--)

int S[105][105],dpTable[105][105];
int main(){
    int T,N;
    double result;
    scanf("%d",&T);
    while(T--){
        scanf("%d",&N);
        REP(i,0,N-1)
            REP(j,0,N-1)
                scanf("%d",&S[i][j]);
        dpTable[N][N] = 0;
        dpTable[N-1][N] = 0;
        REPMIN(i,N-1,0)
            dpTable[N-1][i] = dpTable[N-1][i+1]+S[N-1][i];
        REPMIN(i,N-2,0)
            dpTable[i][N-1] = dpTable[i+1][N-1]+S[i][N-1];
        REPMIN(i,N-2,0){
            REPMIN(j,N-2,0)
                dpTable[i][j] = max(dpTable[i+1][j],dpTable[i][j+1])+S[i][j];
        }
/*        cout<<"dpTable is now "<<endl;
        REP(i,0,N-1){
            REP(j,0,N-1)
                cout<<dpTable[i][j]<<" ";
            cout<<endl;
        } */
        if(dpTable[0][0]<0)
            printf("Bad Judges\n");
        else{
            result = dpTable[0][0]/((double)(2*(N-1)-1));
            printf("%.12lf\n",result);
        }
    }
    return 0;
}
