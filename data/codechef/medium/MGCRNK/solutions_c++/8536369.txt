#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
  int t,n;
  scanf("%d",&t);
  while(t--)
  {
     scanf("%d",&n);
     int arr[101][101];
     for(int i=1; i<=n; i++)
        for(int j=1; j<=n; j++)
            scanf("%d",&arr[i][j]);
     int dp[101][101] = {0};
     for(int i=1; i<=n; i++)
     {
         for(int j=1; j<=n; j++)
           {
               if(i == 1)
                dp[i][j] = dp[i][j-1] + arr[i][j];
               else if(j == 1)
                dp[i][j] = dp[i-1][j] + arr[i][j];
               else
                dp[i][j] = max(dp[i-1][j],dp[i][j-1]) + arr[i][j];
           }
     }
     if(dp[n][n] < 0)
        printf("Bad Judges\n");
     else
     {
         double ans = 0;
         ans = (1.0 * dp[n][n]) / (2*n - 3);
         printf("%.6lf\n",ans);
     }
  }
  return 0;
}
