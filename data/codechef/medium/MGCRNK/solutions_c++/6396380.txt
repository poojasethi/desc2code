#include<iostream>
#include<iomanip>
using namespace std;

int main()
{
       int t, n;
      int s[101][101];
      int i,j;
      cin>>t;
      while(t)
      {
          cin>>n;
          for(i=0;i<n;i++)
          for(j=0;j<n;j++)
          {
                          cin>>s[i][j];
                          if((i==0)&&(j==0))
                          s[i][j]=s[0][0];
                          else if(j==0)
                          s[i][j]+=s[i-1][j];
                          else if(i==0)
                          s[i][j]+=s[i][j-1];
                          else
                          {
                              if(s[i-1][j]>s[i][j-1])
                              s[i][j]+=s[i-1][j];

                              else
                              s[i][j]+=s[i][j-1];

                          }

          }
          double p=(double)s[n-1][n-1]/(2*n-3);
          if(p<0)
          cout<<"Bad Judges\n";
          else
          cout<<fixed<<setprecision(6)<<p<<endl;
           t--;
      }
      return 0;
}
