#include <iostream>
#include <iomanip>
#include <cstdio>
using namespace std;

int main() {
	//ios_base::sync_with_stdio(0);
	int t,n,i,j,mat[100][100];
	double k = 1,mat2[100][100];
	cin>>t;
	while (t--) {
		cin>>n;
		for (i=0;i<n;i++) {
			for (j=0;j<n;j++) cin>>mat[i][j];
		}
		mat2[0][0] = mat[0][0];
		for (i=1;i<n;i++) {
			mat2[i][0] = mat2[i-1][0] + mat[i][0];
		}
		for (j=1;j<n;j++) {
			mat2[0][j] = mat2[0][j-1] + mat[0][j];
		}
		k = 1;
		for (i=1;i<n;i++) {
			for (j=1;j<n;j++) {
					mat2[i][j] = max(mat2[i-1][j],mat2[i][j-1]) + mat[i][j];
			}
		}
		double ans = ((double)(mat2[n-1][n-1]*1.0)/((2*n-3)*1.0));
		if (mat2[n-1][n-1] >= 0)
		printf("%.6f\n",ans);
		else cout<<"Bad Judges\n";
	}
	return 0;
}
