using namespace std;

int maxi(int a ,int b)
{
    if(a>b)
        return a;
    else
        return b;
}
float calculate_dp(int s[100][100],int n)
{
    float dp[n][n];
    dp[0][0]=s[0][0];
    for(int i=1;i<n;i++)
    {
        dp[i][0]=dp[i-1][0]+s[i][0];
    }
    for(int j=1;j<n;j++)
    {
       dp[0][j]=dp[0][j-1]+s[0][j];
    }
        for(int i=1;i<n;i++)
        {
            for(int j=1;j<n;j++)
            {
                dp[i][j]=maxi(dp[i-1][j],dp[i][j-1])+s[i][j];
            }
        }
    return dp[n-1][n-1];
}
int main()
{


    int t;
    cin>>t;
    for(int x=0;x<t;x++)
    {
        int n;
        cin>>n;
        int s[100][100];
        for(int i=0;i<n;i++)
        {
           for(int j=0;j<n;j++)
           {
                cin>>s[i][j];
           }
        }
       float r;
       r=(calculate_dp(s,n)/((2*n)-3));
       if(r<0)
       {
           cout<<"Bad Judges"<<endl;
       }
       else
       {
          printf("%0.6f\n",r);
       }
    }
    return 0;
}
