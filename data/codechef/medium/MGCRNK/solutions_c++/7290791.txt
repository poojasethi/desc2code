#include<bits/stdc++.h>
using namespace std;
#define ll long long


int main()
{
    int t,n,i,j,s[101][101];
    scanf("%d",&t);
    while(t--)
    {
    	scanf("%d",&n);
    	
    	
    	for(i=0;i<n;i++)
    	for(j=0;j<n;j++)
    	scanf("%d",&s[i][j]);
    	
    	for(j=1;j<n;j++)
    	s[0][j]=s[0][j-1]+s[0][j];
    	
    	for(i=1;i<n;i++)
    	s[i][0]=s[i-1][0]+s[i][0];
    	
    	for(i=1;i<n;i++)
    	for(j=1;j<n;j++)
    	s[i][j]=max(s[i-1][j] + s[i][j],s[i][j-1] + s[i][j]);
    	
    	if(s[n-1][n-1]<0)
    	printf("Bad Judges\n");
    	else
    	printf("%.6f\n",(s[n-1][n-1]*1.0)/(2*n-3));
    }
    return 0;
}
