#include<bits/stdc++.h>
# define minvalue -214783648
using namespace std;
inline int numinp(){
   int n=0;
   int ch=getchar_unlocked();int sign=1;
   while(ch < '0' || ch > '9'){
        if(ch=='-')sign=-1;
        ch=getchar_unlocked();
    }
   while(ch >= '0' && ch <= '9')
           n = (n<<3)+(n<<1) + ch-'0', ch=getchar_unlocked();
   return n*sign;
}
int main() {
    int t,n; t = numinp();
    while(t--) {
        n = numinp(); int a[105][105],pmg[105][105];
        for(int i=1;i<=n;i++) for(int j=1;j<=n;j++) {a[i][j] = numinp(); pmg[i][j] = minvalue;}
        for(int i = 1;i<=n;i++) {
            pmg[0][i] = minvalue; pmg[i][0] = minvalue;
        }
        for(int i=1;i<=n;i++) {
            for(int j=1;j<=n;j++) {
                if(i==1 && j==1) pmg[i][j] = a[i][j];
                else if (j==1) pmg[i][j] = a[i][j] + pmg[i-1][j];
                else if (i==1) pmg[i][j] = a[i][j] + pmg[i][j-1];
                else pmg[i][j] = max(pmg[i-1][j],pmg[i][j-1]) + a[i][j];
            }
        }
        if(((float)pmg[n][n]/(n*2-3))< 0) printf("Bad Judges\n");
        else printf("%0.6f\n",((float)pmg[n][n]/(n*2-3)));
    }
    return 0;
}
