#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define F first
#define S second
#define maxn 1005
#define M 1000000007
using namespace std;

int a[105][105],b[105][105];

int main ()
{
	int t,n;
	sc(t);
	float ans;
	while(t--)
	{
		sc(n);
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=n;j++)
			{
				sc(a[i][j]);
			}
		}
		for(int i=0;i<=n;i++)
		{
			b[i][0]=-100000000;
			b[0][i]=-100000000;
		}
		b[0][1]=0;
		b[1][0]=0;
		for(int i=1;i<=n;i++)
		{
			for(int j=1;j<=n;j++)
			{
				b[i][j]=max(b[i-1][j],b[i][j-1])+a[i][j];
				//cout<<b[i][j]<<" ";
			}
			//cout<<endl;
		}	
		//cout<<endl<<b[n][n]<<endl;
		if(b[n][n]<0)
		{
			printf("Bad Judges\n");
		}	
		else
		{
			printf("%0.6f\n",(float)b[n][n]/float(2*n-3));
		}
	}
	return 0;
}
