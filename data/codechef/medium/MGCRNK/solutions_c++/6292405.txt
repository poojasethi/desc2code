    #include<bits/stdc++.h>
    using namespace std;
    //int s[100][100];
     
     
    int main()
    {
    int t,n;
    scanf("%d",&t);
    long ans;
     
    while(t--)
    {
    scanf("%d",&n);
    int s[n+1][n+1];
    for(int i=0;i<n;i++)
    {
    for(int j=0;j<n;j++)
    {
    scanf("%d",&s[i][j]);
    }
    }
    for(int i=1;i<n;i++)
    {
    s[0][i]=s[0][i-1]+s[0][i];
    }
    for(int i=1;i<n;i++)
    {
    s[i][0]=s[i-1][0]+s[i][0];
    }
    for(int i=1;i<n;i++)
    {
    for(int j=1;j<n;j++)
    {
    if(s[i-1][j]>s[i][j-1])
    ans=s[i-1][j];
    else ans=s[i][j-1];
    s[i][j]=ans+s[i][j];
    }
    }
     
    if(ans<0)
    printf("Bad Judges\n");
    else
    {
    double ans1=1.0*ans/(2*n-3);;
    printf("%lf\n",ans1);
    }
     
     
     
     
    }
    return 0;
    }
