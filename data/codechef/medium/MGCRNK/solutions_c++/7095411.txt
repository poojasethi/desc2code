#include<bits/stdc++.h>
using namespace std;
int main() {
	int  t,n,i,j;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		int a[n][n],b[n][n];
		for(i=0;i<n;i++) {
			for(j=0;j<n;j++)
			scanf("%d",&a[i][j]);
		}
		b[0][0] = a[0][0];
		for(i=1;i<n;i++)
		b[0][i] = b[0][i-1] + a[0][i];
		for(i=1;i<n;i++)
		b[i][0] = b[i-1][0] + a[i][0];
		for(i=1;i<n;i++) {
			for(j=1;j<n;j++) {
				b[i][j] = max(b[i-1][j]+a[i][j],b[i][j-1]+a[i][j]);
			}
		}
		double s = b[n-1][n-1];
		double s1 = 2*(n-1)-1;
		if(b[n-1][n-1] >= 0)
		printf("%.6lf\n",s/s1);
		else
		printf("Bad Judges\n");
	}
	return 0;
}
