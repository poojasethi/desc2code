	int i,j;

		for(j=0;j<n;j++)	
			cin>>a[i][j];
	}

		a[0][i]+= a[0][i-1];
		a[i][0]+= a[i-1][0];

			a[i][j]+= max(a[i][j-1], a[i-1][j]);

	}
