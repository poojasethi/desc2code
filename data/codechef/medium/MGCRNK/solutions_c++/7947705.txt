#include<iostream>
#include<stdio.h>
#define ll long long int
using namespace std;
ll dp[1000][1000];
int a[1000][1000];
int main()
{
      int t;
      cin>>t;
      while(t--){
            int n,i,j,k,cnt=0;
            cin>>n;
            for(i=0;i<n;i++)
                  for(j=0;j<n;j++)
                cin>>a[i][j];

             for(i=0;i<=n;i++)
            for(j=0;j<=n;j++)
            {
                  dp[i][j]=0;
            }
            for(i=0;i<n-1;i++)
                  dp[i+1][0]=dp[i][0]+a[i+1][0];
            for(j=0;j<n-1;j++)
                  dp[0][j+1]=dp[0][j]+a[0][j+1];
           for(i=0;i<n-1;i++){
            for(j=0;j<n-1;j++){

                        dp[i+1][j+1]=max(dp[i][j+1],dp[i+1][j])+a[i+1][j+1];


            }
           }
           cnt=2*n-3;
           float ans=float(dp[n-1][n-1])/float(cnt);
           if(ans>=0)
           printf("%.6f\n",ans);
           else
            cout<<"Bad Judges"<<endl;
      }
      return 0;
}
