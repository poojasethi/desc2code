//constructs DP solution
int max_score(int array[][101], int size)
{int i,j;
int dp[size+1][size+1];
for(i=1;i<=size;++i)
    dp[i][0]=dp[0][i]=-1000000;
for(i=1;i<=size;++i)
{
    for(j=1;j<=size;++j)
    {
        if(i==1&&j==1)
            dp[i][j]=array[i][j];
        else
            dp[i][j]=max(dp[i][j-1],dp[i-1][j])+array[i][j];
    }
}
return dp[size][size];
}

int main()
{
    int t,i,j;
    cin >> t;
    int ans;
    while(t--)
    {
		int n;
		cin >> n;
		int array[101][101];
		for(i=1;i<=n;++i)
            for(j=1;j<=n;++j)
            cin>>array[i][j];
    ans=max_score(array,n);
    if(ans>=0)
        printf("%.6lf\n",double(ans/(2*n-3.0)));
    else
        printf("Bad Judges\n");
    }return 0;
    }
