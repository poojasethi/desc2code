#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    ios::sync_with_stdio(0);
    int t,n,i,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int s[n][n],dp[n][n];

        for(i = 0 ; i < n ; ++i)
        {
            for(j = 0 ; j < n ; ++j)
            {
                scanf("%d",&s[i][j]);
            }
        }
        dp[0][0] = s[0][0];

        for(i = 1 ; i < n ; ++i)
            dp[i][0] = dp[i-1][0] + s[i][0];

        for(j = 1 ; j < n ; ++j)
            dp[0][j] = dp[0][j-1] + s[0][j];

        for(i = 1 ; i < n ; ++i)
        {
            for(j = 1 ; j < n ; ++j)
            {
                dp[i][j] = max(dp[i-1][j],dp[i][j-1]) + s[i][j];
            }
        }
        if(dp[n-1][n-1] < 0)
            printf("Bad Judges\n");
        else
        {
            float sum=dp[n-1][n-1];
            sum/=float(2*n-3);
            printf("%.6f\n",sum);
        }
    }
    return 0;
}
