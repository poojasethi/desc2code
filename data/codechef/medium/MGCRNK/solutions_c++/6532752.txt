#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,mp[1000][1000],dp[1000][1000];
		cin>>n;
		for (int i = 0; i <n ; ++i)
		{
			for (int j =0 ; j<n; ++j)
			{
				cin>>mp[i][j];
			}
		}
		double res;
		dp[0][0]=0;
		for (int i = 1; i <n ; ++i)
		{
			dp[0][i] = dp[0][i-1] + mp[0][i];
		}
		for (int i = 1; i <n ; ++i)
		{
			dp[i][0] = dp[i-1][0] + mp[i][0];
		}
		for(int i = 1;i <n ; ++i)
		{
			for (int j =1 ; j<n; ++j)
			{
				if(dp[i][j-1] > dp[i-1][j]) dp[i][j] = dp[i][j-1] + mp[i][j];
				else              dp[i][j] = dp[i-1][j] + mp[i][j];
			}
		}
		if(dp[n-1][n-1] < 0){ puts("Bad Judges"); continue; }
    	res = dp[n-1][n-1] / (double)(2*n-3);
    	printf("%.6f\n",res);


	}
	return 0;
}
