#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
   int t,n;
   scanf("%d",&t);
   while(t--){
        scanf("%d",&n);
    int a[n][n];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
        scanf("%d",&a[i][j]);

    for(int i=1;i<n;i++)
        a[0][i]+=a[0][i-1];

    for(int j=1;j<n;j++)
        a[j][0]+=a[j-1][0];

    for(int i=1;i<n;i++)
        for(int j=1;j<n;j++)
            if(a[i-1][j]>a[i][j-1])
                        a[i][j]+=a[i-1][j];
                        else
                            a[i][j]+=a[i][j-1];

        if(a[n-1][n-1]<0)
            printf("Bad Judges\n");
        else{
            float ans=a[n-1][n-1]/1.0;
            ans/=2*(n-1)-1;
            printf("%.6f\n",ans);
        }

   }


    return 0;
}
