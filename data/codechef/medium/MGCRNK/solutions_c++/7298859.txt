#include<bits/stdc++.h>
using namespace std;
#define max(a,b) a>b?a:b
int main()
{
 int t;
 scanf("%d",&t);
 while(t--)
 {
     int n;
     scanf("%d",&n);
     int s[n][n];
     int i,j;
     for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        scanf("%d",&s[i][j]);
     for(i=1;i<n;i++)
        s[0][i]+=s[0][i-1];
     for(j=1;j<n;j++)
        s[j][0]+=s[j-1][0];
     for(i=1;i<n;i++)
     {
         for(j=1;j<n;j++)
         {
             s[i][j]=max(s[i-1][j]+s[i][j],s[i][j]+s[i][j-1]);
         }
     }
     if(s[n-1][n-1]<0)
        printf("Bad Judges\n");
     else printf("%0.6f\n",s[n-1][n-1]*1.0/(2*n-3));
 }
 return 0;
}
