#include <stdio.h>
#include <iostream>
using namespace std;

int dp[105][105],a[105][105];

int main() {
	// your code goes here
	
	int test;
	cin>>test;
	while(test--)
	 {
	     int n;
	     cin>>n;
	     for(int i=1;i<=n;i++)
	      for(int j=1;j<=n;j++)
	       cin>>a[i][j];
	       
	     dp[1][1] = 0;
	     for(int j=2;j<=n;j++)
	       dp[1][j] = dp[1][j-1] + a[1][j];
	     
	     for(int i=2;i<=n;i++)
	       dp[i][1] = dp[i-1][1] + a[i][1];
	       
	   
	     for(int i=2;i<=n;i++)
	      {
	          for(int j=2;j<=n;j++)
	           dp[i][j] = max(dp[i-1][j],dp[i][j-1]) + a[i][j];
	      }
	      
	      double ans = dp[n][n];
	      ans = ans/(2*n-3);
	      if(dp[n][n] >= 0) printf("%.9lf\n",ans);
	      else printf("Bad Judges\n");    
	       
	       
	 }
	
	return 0;
}
