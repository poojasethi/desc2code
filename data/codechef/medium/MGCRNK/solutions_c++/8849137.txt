// URL : https://www.codechef.com/problems/MGCRNK

#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define M 111

int x[M][M];

int main()
{
    int tc;
    cin >> tc;
    while(tc--)
    {
        int n;
        cin >> n;
        for(int i = 0; i < n; ++i)
         for(int j = 0; j < n; ++j)
           cin >> x[i][j];
        
        // running sum 1st row
        for(int i = 1; i < n; ++i)
                x[0][i] += x[0][i - 1];
        
        // running sum 1st column
        for(int i = 1; i < n; ++i)
                x[i][0] += x[i - 1][0];
        
        for(int i = 1; i < n; ++i)
         for(int j = 1; j < n; ++j)
           x[i][j] += max(x[i - 1][j], x[i][j - 1]);
        
        if(x[n - 1][n - 1] < 0)
        {
            printf("Bad Judges\n");   
        }
        else
        {
            printf("%.6lf\n", (double)x[n - 1][n - 1] / (2 * n - 3));
        }
    }
    return 0;
}
