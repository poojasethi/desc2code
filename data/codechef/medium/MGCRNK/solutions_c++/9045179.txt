#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,t,n;
	double ans;
	vector<vector<int> > v(100,vector<int>(100));
	vector<vector<int> > dp(100,vector<int>(100));
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				scanf("%d",&(v[i][j]));
			}
		}
		dp[0][0]=0;
		for(i=1;i<n;i++)
		{
			dp[0][i]=dp[0][i-1]+v[0][i];
			dp[i][0]=dp[i-1][0]+v[i][0];
		}
		for(i=1;i<n;i++)
		{
			for(j=1;j<n;j++)
			{
				dp[i][j]=v[i][j]+max(dp[i][j-1],dp[i-1][j]);
			}
		}
		if(dp[n-1][n-1]<0)
		{
			printf("Bad Judges\n");
		}
		else
		{
			ans=dp[n-1][n-1];
			printf("%.6lf\n",ans/(2*n-3));
		}
	}
	return 0;
}
