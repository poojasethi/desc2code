#include <bits/stdc++.h>

using namespace std;
int maze[100][100],n,dp[100][100];
long int maxi(long int a,long int b)
{
    if(a>b)
        return a;
    return b;
}
int main()
{
    int t,i,j;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
                scanf("%d",&maze[i][j]);
        }
        dp[0][0]=0;
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                if(i==0 && j==0)
                    continue;
                if(i==0)
                    dp[i][j]=dp[i][j-1]+maze[i][j];
                else if(j==0)
                    dp[i][j]=dp[i-1][j]+maze[i][j];
                else
                    dp[i][j]=maxi(dp[i-1][j],dp[i][j-1])+maze[i][j];
            }
        }
        if(dp[n-1][n-1]<0)
            cout<<"Bad Judges\n";
        else
            printf("%.6lf\n",(double)(dp[n-1][n-1]/(double)(2*n-3)));
    }
    return 0;
}
