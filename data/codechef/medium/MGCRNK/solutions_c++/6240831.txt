ll mul(ll a,ll b,ll m)
{
   ll mod = m;
   a %= m;
   b %= m;
   long double res = a;
   res *= b;
   ll c = (ll)(res / mod);
   a *= b;
   a -= c * mod;
   a %= mod;
   if (a < 0) a += mod;
   return a;
}

ll fpower(ll x,ll p,ll MOD)                         //POWER  MODULO C
{
    ll res = 1;
    while(p>0)
    {
        if(p%2 != 0) res = mul(res,x,MOD);
        p = p/2;
        x = mul(x,x,MOD);
    }
    return res;
}
