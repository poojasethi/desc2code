#include <stdio.h>
#include <iostream>
#include <cstdlib>
#include <math.h>

using namespace std;

int** mat;
int N;

inline int getint()
{
	int num=0;
	char c=getchar_unlocked();
	while(!(c>='0' && c<='9'))
		c=getchar_unlocked();
	while(c>='0' && c<='9')
	{
		num=(num<<3)+(num<<1)+c-'0';
		c=getchar_unlocked();
	}
	return num;
}

int get_max(int x, int y){
	int res[N][N];
	res[0][0] = 0;
	for(int j = 1; j < N; j++){
		res[0][j] = mat[0][j] + res[0][j-1];
		res[j][0] = mat[j][0] + res[j-1][0];
	}
	for(int i = 1; i < N; i++){
		for(int j = 1; j < N; j++){
			res[i][j] = mat[i][j] + max(res[i][j-1], res[i-1][j]);
		}
	}
	return res[N-1][N-1];	
}

int main(){
	int T;	
	T = getint();
	float result;
	for(int i = 0; i < T; i++){
		N = getint();
		mat = (int**)malloc(sizeof(int*)*N);
		for(int j = 0; j < N; j++){
			mat[j] = (int*)malloc(sizeof(int)*N);
		}
		for(int m = 0; m < N; m++){
			for(int n = 0; n < N; n++){
				cin >> mat[m][n];
			}
		}
		result = get_max(0,0);
		if(result < 0)
			cout << "Bad Judges" << endl;
		else
			printf("%.6f\n", (result/(2*N-3)));
		for(int j = 0; j < N; j++){
			delete(mat[j]);
		}
		delete(mat);
	}
}
