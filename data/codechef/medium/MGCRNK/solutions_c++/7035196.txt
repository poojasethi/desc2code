#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
	int i,j,k,m,n,t;
	float a,b;
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
		scanf("%d",&n);
		int s[n][n],dp[n][n];
		for(j=0;j<n;j++)
		{
			for(k=0;k<n;k++)
				scanf("%d",&s[j][k]);
		}
		for(j=0;j<n;j++)
		{
			for(k=0;k<n;k++)
			{
				if(j==0 && k==0)
					dp[j][k]=s[j][k];
				else if(j==0)
					dp[j][k]=dp[j][k-1]+s[j][k];
				else if(k==0)
					dp[j][k]=dp[j-1][k]+s[j][k];
				else
					dp[j][k]=max(dp[j-1][k],dp[j][k-1])+s[j][k];
			}
		}
		a=2*n-3;
		b=dp[n-1][n-1]/a;
		if(b<0)
			printf("Bad Judges\n");
		else
			printf("%f\n",b);
	}
}
