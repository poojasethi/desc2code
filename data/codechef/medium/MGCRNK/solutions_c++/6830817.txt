#include <iostream>
#include <string>
#include <algorithm>
#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <vector>
#include <queue>

using namespace std;
 
int matrix[101][101];
int cost[101][101];
int max(int a , int b)
{
      if(a < b)
	    return b;
      return a;
}
int main()
{
	

	int t, n; 
	cin >> t; 
	while(t--)
	{
		cin >> n;
		for(int i = 1 ; i <= n ; i++)
		{
			for(int j = 1 ; j <= n ; j++)
			{
				cin >> matrix[i][j];
				cost[i][j] = -1000000; 
			}
		}
		cost[1][1] = matrix[1][1]; 
		for(int i = 2 ; i <= n; i++){
		  	cost[1][i] = cost[1][i - 1] + matrix[1][i];
			cost[i][1] = cost[i - 1][1] + matrix[i][1];
		}
		for(int i = 2 ; i <= n ; i++)
		{
		      for(int j = 2 ; j <= n ; j++)
		      {
			    cost[i][j] = max(cost[i - 1][j] , cost[i][j - 1]) + matrix[i][j];
		      }
		}
		if(cost[n][n] < 0)
			printf("Bad Judges\n");
		else		
			printf("%.8f\n" , (float)cost[n][n] / (float(2 * n) - 3.0)); 
	
	}

		
return 0;
}
