#include <iostream>
#include <stdio.h>

using namespace std;

int main()
{
    int kp[200][200];
    int count[200][200];
    int a[200][200];
    int i,j;
    int n;
    int t;


    cin>>t;

    while (t--)
    {


    cin>>n;
    if (n!=1)
    {


    for(i=1;i<=n;i++)
    {
        for (j=1;j<=n;j++)
            cin>>a[i][j];
    }
    count[1][1]=0;
    kp[1][1]=0;
    for (i=2;i<=n;i++)
    {
        kp[i][1]=kp[i-1][1]+a[i][1];
        kp[1][i]=kp[1][i-1]+a[1][i];
        count[i][1]=1+count[i-1][1];
        count[1][i]=1+count[1][i-1];
    }

    for (i=2;i<=n;i++)
    {
        for (j=2;j<=n;j++)
        {
            if ((((double)kp[i-1][j])/count[i-1][j])>(((double)kp[i][j-1])/count[i][j-1]))
            {

                kp[i][j]=kp[i-1][j]+a[i][j];
                count[i][j]=1+count[i-1][j];
            }
            else
            {
                kp[i][j]=kp[i][j-1]+a[i][j];
                count[i][j]=1+count[i][j-1];

            }




        }
    }
   count[n][n]--;



   if (kp[n][n]<0)
    cout<<"Bad Judges"<<endl;
   else
    printf("%.6f\n",((double)kp[n][n])/(count[n][n]));
    }

    else
        cout<<0.000000;

    }




    return 0;
}
