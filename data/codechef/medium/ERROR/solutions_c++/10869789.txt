#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int t,l,v;
    char ar[100005];
    scanf("%d\n",&t);
    while(t--)
    {
        v=0;
        gets(ar);
        l=strlen(ar);
        for(int i=0;i<l-2;i++)
        {
            if(ar[i]=='0' && ar[i+1]=='1' && ar[i+2]=='0')
               {
                   v=1;
                   break;
               }
            if(ar[i]=='1' && ar[i+1]=='0' && ar[i+2]=='1')
            {
                v=1;
                break;
            }
        }
        if(v)
            printf("Good\n");
        else
            printf("Bad\n");
    }
    return 0;
}
