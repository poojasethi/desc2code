#include<bits/stdc++.h>
using namespace std;
int check(const char* str,const char* substring)
{
    char* pos = strstr(str,substring);
    if(pos)
        return 1;
    else
        return 0;
}
int main(){
    int t;
    cin>>t;
    cin.ignore(INT_MAX,'\n');
    while(t--)
    {
        char inp [100000];
        gets (inp);
        int rec,total=-1;
        rec = check(inp,"010");
        total = rec;
        rec = check(inp,"101");
        total = total +rec;
        if(total>0)
            cout<<"Good"<<endl;
        if(total==0)
        {
            cout<<"Bad"<<endl;

        }

    }
    return 0;
}
