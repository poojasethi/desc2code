#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int l,i,f=0;
		l=s.length();
		for(i=0;i<l;i++)
		{
			if(s[i]=='0'&&s[i+1]=='1'&& s[i+2]=='0'||s[i]=='1'&&s[i+1]=='0'&& s[i+2]=='1')
			{
				f=1;
				break;
			}
		}
		if(f)
		cout<<"Good"<<endl;
		else
		cout<<"Bad"<<endl;
		
		
	}
	
return 0;	
}
