#include<iostream>
#include<string>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		string ans="Bad";
		if(s.length()<3)
		{
		cout<<ans<<endl;continue;}
		for(int i=0;i<s.length()-2;i++)
		{
			if( s[i]=='1' && s[i+1]=='0' && s[i+2]=='1'){
			   ans="Good";
			   break;
		}
			if(s[i]=='0' && s[i+1]=='1' && s[i+2]=='0'){
				 ans="Good";
				 break;
			}
			  
		}
	cout<<ans<<"\n";
	}
	return 0;
}
