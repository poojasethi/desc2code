#include<iostream>
#include<string>
#include<bits/stdc++.h>
using namespace std;
int main(){
	unsigned int n, i, lenth;
	scanf("%d\n",&n);
	start:
	while(n--){
		string feedback;
		cin>>feedback;
		// scanf("%s\n",feedback);
		lenth = feedback.length();
		if (lenth<3){
			printf("Bad\n");
			continue;
		}
		for (i=0;i<lenth-2;i++){
			if((feedback[i] == '0' && feedback[i+1] == '1' && feedback[i+2] == '0') ||
				 (feedback[i] == '1' && feedback[i+1] == '0' && feedback[i+2] == '1')){
				printf("Good\n");
				goto start;
				// break;
			}	
		}
		printf("Bad\n");
	}
	return 0;
}
