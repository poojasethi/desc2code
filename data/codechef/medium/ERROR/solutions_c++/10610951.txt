// #include<iostream>
#include<cstdio>
#include<cstring>
// #include<bits/stdc++.h>
using namespace std;
int main(){
	unsigned int n, i, lenth;
	scanf("%d\n",&n);
	char f[100005];
	// cin>>n;
	start:
	while(n--){
		gets(f);
		lenth = strlen(f);
		if (lenth<3){
			printf("Bad\n");
			continue;
		}
		for (i=0;i<lenth-2;i++){
			if((f[i] == '0' && f[i+1] == '1' && f[i+2] == '0') ||
				 (f[i] == '1' && f[i+1] == '0' && f[i+2] == '1')){
				printf("Good\n");
				goto start;
				// break;
			}	
		}
		printf("Bad\n");
	}
	return 0;
}
