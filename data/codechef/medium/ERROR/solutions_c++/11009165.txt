#include <iostream>
using namespace std;

int main() {
	int t,i;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		i=0;
		if(s.length()<3)
		cout<<"Bad"<<endl;
		else
		{
			while(i<s.length()-2)
			{
				if(s[i+1]!=s[i]&&s[i+2]==s[i])   //s[i+2]!=s[i+1]
				{  
					cout<<"Good"<<endl;
					break;
				}
				i++;
			}
			if(i==s.length()-2) //s.length()-1
			cout<<"Bad"<<endl;
		}
	}	
	return 0;
}
