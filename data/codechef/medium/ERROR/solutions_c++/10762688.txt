#include<bits/stdc++.h>
using namespace std;

int main(){
	int t;
	scanf("%d",&t);
	
	while(t--){
		char ss[100005];
		scanf("%s",ss);
		bool flag = 0;
		//cin>>ss;
		for(int i=0;i<sizeof(ss)/sizeof(char)-2;i++){
			if(ss[i]=='1' && ss[i+1]=='0' && ss[i+2]=='1'){
				flag=1;break;
			}
			 if(ss[i]=='0' && ss[i+1]=='1' && ss[i+2]=='0'){
				flag=1;break;
			}
			
		}
		if(flag==1)printf("Good\n");
		else printf("Bad\n");
	}
}
