#include<iostream>
using namespace std;
int main()
{
	int t,i;
	char a[100006];
	cin>>t;
	while(t--)
	{
	int flag=0;		
		scanf("%s",a);
		for(i=0;a[i]!='\0';i++)
		{
			if((a[i]=='0'&&a[i+1]=='1'&&a[i+2]=='0')||(a[i]=='1'&&a[i+1]=='0'&&a[i+2]=='1'))
					{flag=1;	
					break;
					}				
		}
		if(flag==1)
		cout<<"Good"<<endl;
		else
		cout<<"Bad"<<endl;			
		
	}
	return 0;
}
