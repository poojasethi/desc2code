#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int main()
{
    int t;
    char arr[100000];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",arr);
        int flag=0;
        for(int i=0;i<(strlen(arr)-2);i++)
        {
            if(arr[i]==arr[i+2]&&arr[i]!=arr[i+1])
            {
                flag=1;
                break;
            }
        }
        if(strlen(arr)<3)
        {
            printf("Bad\n");
        }
        else if(flag==1)
        {
            printf("Good\n");
        }
        else
            printf("Bad\n");
    }
    return 0;
}
