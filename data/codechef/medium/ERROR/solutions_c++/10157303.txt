#include<stdio.h>
#include<string>
#include<iostream>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int i,p,q,r,x=0;
		string s;
		cin >> s;
		for(string::iterator it=s.begin();it!=s.end();it++)
		{
			p = *it;
			q = *(it+1);
			r = *(it+2);
//			printf("%d %d %d\n",p,q,r);
			if((p==48 && q==49 && r==48) || (p==49 && q==48 && r==49))
			{
				printf("Good\n");
				break;
			}
			x++;
		}
		if(x==s.length())
			printf("Bad\n");
	}
	return 0;
}
