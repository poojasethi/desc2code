#include<bits/stdc++.h>
using namespace std;
 
int main() 
{
	int t,i,j,n,c;
	char a[1000001];
	scanf("%d",&t);
	for(i=0;i<t;i++)
	{
	    c=0;
	    scanf("%s",a);
	    n=strlen(a);
	   
	    for(j=0;j<=n-2;j++)
	    {
	        if(a[j]=='0'&&a[j+1]=='1'&&a[j+2]=='0')
	        {
	            c=1;
	            break;
	        }
	        else if(a[j]=='1'&&a[j+1]=='0'&&a[j+2]=='1')
	        {
	            c=1;
	            break;
	        }
	    }
	   if(c==1)
	   {printf("Good\n");}
	   else
	   {
	       printf("Bad\n");
	   }
	}
	return 0;
}
