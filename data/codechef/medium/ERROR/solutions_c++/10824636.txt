#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{	
		string str;
		cin>>str;
		int i=0,f=0;
		while(str[i])
		{	
			if(str[i+2])
			{			
				if(str[i]=='1' && str[i+1]=='0' && str[i+2]=='1')
				{
					f=1;
					break;
				}
				else if(str[i]=='0' && str[i+1]=='1' && str[i+2]=='0')
				{
					f=1;
					break;
				}
				else
				{
					f=0;
				}
			}
			i++;
		}
	
		if(f==1)
		cout<<"Good"<<endl;	
		else
		cout<<"Bad"<<endl;
	}
	return 0;
}
