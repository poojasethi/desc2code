#include<cstdio>
#include<cstring>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        char str[100003];
        scanf("%s",&str);
        int n=strlen(str);
        int fail=1;
        for(int i=0;i<n-2;i++){
            char a=str[i];
            char b=str[i+1];
            char c=str[i+2];
            if((a=='0' && b=='1' && c=='0') || (a=='1' && b=='0' && c=='1')){
                fail=0;
                break;
            }
        }
        if(fail){
            printf("Bad\n");
        }else
        {
            printf("Good\n");
        }
    }
}
