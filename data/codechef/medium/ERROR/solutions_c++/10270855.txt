#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<vector>
#include<algorithm>
#include<cstdlib>
 using namespace std;

 int main()
 {
     int t;
     scanf("%d",&t);
     while(t--)
     {
         char str[1000000];
         int count=0;
         scanf("%s",&str);
        int len=strlen(str);

         for(int i=0;i<len-2;i++)
         {
              if(str[i]=='0' && str[i+1]=='1' && str[i+2]=='0' || str[i] == '1' && str[i+1] == '0' && str[i+2] == '1')
              {
                 printf("Good\n");
                 count++;
                 break;
             }
         }

         if(count==0)
           printf("Bad\n");
       }
     return 0;
 }
