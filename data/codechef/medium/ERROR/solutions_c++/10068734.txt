#include<bits/stdc++.h>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        int l=s.length();
        int a=1;
        for(int i=0;i<=l-3;i++){
            if((s[i]=='0' && s[i+1]=='1' && s[i+2]=='0')||(s[i]=='1' && s[i+1]=='0' && s[i+2]=='1')){
                cout<<"Good"<<endl;
                a=0;
                break;
            }
        }
        if(a==0){
            continue;
        }
        cout<<"Bad"<<endl;
    }
}
