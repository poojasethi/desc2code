#include <iostream>
#include <bits/stdc++.h>

typedef long long int ll;
using namespace std;
int main()
{
	ll t;
	cin>>t;
	while(t--)
	{
		string s,s1="010",s2="101";
		cin>>s;
		if(s.find(s1) != string::npos || s.find(s2) != string::npos)
			cout<<"Good"<<endl;
		else
			cout<<"Bad"<<endl;	
	}
	return 0;
}
