#include<iostream>
#include<string>
using namespace std;

int main(){
	int t; cin>>t;
	string s;
	long len;
	bool flag;
	while(t--){
		cin>>s;
		flag=false;
		len = s.length();
		for(long i=0;i<len-2;i++){
			if(s[i]!=s[i+1]){
				if(s[i+1]!=s[i+2]){
					cout<<"Good\n";
					flag=true;
					break;}
				else
				   i++;
				}
			}
		if(!flag)
			cout<<"Bad\n";
		}
	return 0;
}
