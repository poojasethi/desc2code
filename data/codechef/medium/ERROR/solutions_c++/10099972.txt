#include <iostream>
#include <string>
#include <stdio.h>

int main()
{
	int numCases;
	const std::string pattern1 = "010";
	const std::string pattern2 = "101";
	std::string input = "";
	std::cin >> numCases;
	for(int i = 0; i < numCases; i++)
	{
		std::cin >> input;
		if((input.find(pattern1) != std::string::npos) || input.find(pattern2) != std::string::npos)
			std::cout << "Good\n";
		else
			std::cout << "Bad\n";
	}
	return 0;
}
