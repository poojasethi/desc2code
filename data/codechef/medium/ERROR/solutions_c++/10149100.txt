#include<iostream>
#include<string.h>
using namespace std;
int main()
{
  long int t,i,j,l,flag=0;
  char s[100000];
  cin>>t;
   while(t--)
  {
    flag=0;
    cin>>s;
    l=strlen(s);
    for(i=0;i<l-2;i++)
    {
      if(s[i]=='1')
      {
	 if(s[i+1]=='0'&&s[i+2]=='1')
	 {
	 cout<<"Good"<<"\n";
	 flag=1;
	 break;
	 }
      }
      else
      {
	if(s[i+1]=='1'&&s[i+2]=='0')
	{
	  cout<<"Good"<<"\n";
	  flag=1;
	  break;
	 }
      }
      }
	 if(flag!=1)
	 cout<<"Bad"<<"\n";
   }
return 0;
}
