#include<iostream>
#include<string.h>
using namespace std;
bool contains(string s,string p)
{
    int len=s.length(),j,k;
    for(int i=0;i<len;i++)
    {
        j=0;
        k=i;
        while(j<3)
        {
            if(s[k]==p[j])
            {
                j++;
                k++;
            }
            else
            {
                j=0;
                break;
            }
        }
        if(j==3)
            return true;
    }
    return false;
}
int main()
{
   int t;
   string s;
   cin>>t;
   while(t--)
   {
     cin>>s;
     if(contains(s,"010")||contains(s,"101"))
        cout<<"Good\n";
     else
        cout<<"Bad\n";
   }

}
