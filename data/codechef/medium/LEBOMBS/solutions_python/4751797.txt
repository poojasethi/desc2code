for i in xrange(int(raw_input())):
    n = int(raw_input())
    S = raw_input()
    cS = list(S)
    for j in xrange(n):
        if S[j] == "1":
            if not j == 0:
                cS[j-1] = "1"
            if not j == n-1:
                cS[j+1] = "1"
    ctr = 0
    for j in xrange(n):
        if cS[j] == "1":
            ctr += 1
    print n-ctr
