t = int(raw_input())
while t:
    n = int(raw_input())
    s = raw_input()
    i=0
    count = 0
    while i<n:
        while i<n and s[i]=='1':
            i+=1
        j=i
        while i<n and s[i]=='0':
            i+=1
        if j == 0 and i==n:
            count+=i-j
        elif (j==0 or i==n) and i-j>1:
            count+=i-j-1
        elif i-j>2:
            count+=i-j-2
    print count
    t-=1
