T=int(raw_input().strip())
while T:
    T-=1
    N=int(raw_input().strip())
    count=0
    arr=[int(x) for x in raw_input().strip()]
    for i in range(0,N):
        if arr[i]==1:
            if i>0 and arr[i-1]==0:
                count-=1
            if i<N-1:
                if arr[i+1]==0:
                    arr[i+1]=2
        elif arr[i]!=2:
            count+=1
    print count
