def numSafePlaces(N, S):
    if N == 1:
        return 1 if S=='0' else 0
    count = N
    if S[0]=='1' or S[1]=='1':
        count -= 1
    for i in range(1, N-1):
        if S[i-1]=='1' or S[i]=='1' or S[i+1]=='1':
            count -= 1
    if S[N-2]=='1' or S[N-1]=='1':
        count -=1
    return count
 
T=int(raw_input())
for _ in range(T):
    N=int(raw_input())
    S=raw_input()
    print numSafePlaces(N, S)
