t = int(raw_input())
for i in range(t):
    n = input()
    a = raw_input()
    b = [False]*n
    for j in xrange(n):
        if a[j] == '1':
            b[j] = True

            if j > 0 and b[j] :
                b[j - 1] = True

            if j< n- 1 and b[j] :
                b[j+1] = True

    print b.count(False)
