import sys

num = int(sys.stdin.readline().rstrip())
while num != 0:
	num_rooms = int(sys.stdin.readline().rstrip())
	room_status = sys.stdin.readline().rstrip()
	
	saved_count = 0  
	for i in xrange(0,num_rooms):
		
		destroyed = False  
		if room_status[i] == '1':
			destroyed = True  
		if i > 0 and room_status[i-1] == '1':	
			destroyed = True  
		if i < num_rooms -1 and room_status[i+1] == '1':
			destroyed = True  
		if not destroyed:
			saved_count += 1  
	print saved_count
	num -= 1
