from sys import stdin
n = lambda: int(stdin.readline())

T = n()
for _ in xrange(T):
    n()
    bs = map(int, stdin.readline().strip())

    bombed = False
    for i, v in enumerate(bs):
        if v:
            if i > 0:
                bs[i - 1] = 1
            bombed = True
        elif bombed:
            bs[i] = 1
            bombed = False

    print len(bs) - sum(bs)
