#include<iostream>
#include<algorithm>
#include<string>


using namespace std;

int main() {
    int t;
    cin>>t;
    while (t--) {
      int n;
        cin>>n;
        string arr;
        cin>>arr;
        int count=0;
        for(int i=0;arr[i];i++)
        {
            if(arr[i]=='0')
            {
                if((i-1)>=0 && arr[i-1]=='0' && (i+1<n) && (arr[i+1]=='0'))
                    count++;
                else if( i-1<0 && i+1==n)
                    count++;
                else if((i-1<0 && arr[i+1]=='0')||(i+1==n && arr[i-1]=='0'))
                    count++;


            }

        }
        cout<<count<<endl;
    }
    return 0;
}
