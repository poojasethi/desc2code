#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i;
		cin>>n;	
		string a,b;
		cin>>a;
		for(i=0;i<n;i++)
		b[i]='0';	
		for(i=0;i<n;i++)
		{
			if(a[i]=='1')
			{
				if(i>0 && i<n-1)			
				b[i]=b[i-1]=b[i+1]='1';		
				else if(i==n-1)
				b[i]=b[i-1]='1';
				else if(i==0)
				b[i]=b[i+1]='1';
				else{}
			}
		}
		int ctr=0;
		for(i=0;i<n;i++)
		{
			if(b[i]=='0')
			ctr++;
		}
		cout<<ctr<<endl;
	}
	return 0;
}
