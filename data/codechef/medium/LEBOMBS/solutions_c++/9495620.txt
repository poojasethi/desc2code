#include <iostream>
 
using namespace std;
#define up(i,p,q) for(i=p;i<q;i++)
#define down(i,p,q) for(i=p;i>q;i--)
typedef long long int BIG;
int main()
{  int t,n,i,cal;
    cin>>t;
    while(t--)
    {
        cin>>n;//3
        char ar[n+1];//4
        cin>>ar;  // 010
        cal=0;
       up(i,1,n-1){if(ar[i]=='1' && ar[i+1]=='0'){ ar[i-1]=ar[i]=ar[i+1]='x';}if(ar[i]=='1'){ar[i]=ar[i-1]='x';}}
       if(ar[0]=='1'){ar[1]='x';}
       if(ar[n-1]=='1'){ar[n-2]='x';}
 
       up(i,0,n){if(ar[i]=='0'){cal++;}}
       cout<<cal<<endl;
 
    }
    return 0;
}
