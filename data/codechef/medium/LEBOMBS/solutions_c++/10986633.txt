#include <iostream>


using namespace std;

int main()
{
    int t;
   
    cin>>t;
    for(int h=0;h<t;h++)
    {



    int n;
   
    cin>>n;
   

    char a[1001];
    int arr[1001];
    cin>>a;
    for(int m=0;m<n;m++)
    {
        arr[m]=1;
    }

    int ans=0;
    if(n==1)
    {
        if(a[0]=='1')
        {
            ans++;
        }

    }
    else{


    
    for(int i=0;i<n;i++)
    {
       if(a[i]=='1')
       {
           if(i==0)
           {
               arr[0]=0;
               arr[1]=0;

           }
           else if(i==n-1)
       {

               arr[n-1]=0;
               arr[n-2]=0;

       }
       else

           {
               arr[i]=0;
               arr[i-1]=0;
               arr[i+1]=0;
           }


       }



    }

    for(int j=0;j<n;j++)
    {
        if(arr[j]==0)
        {
            ans++;
        }
    }
    }

    //cout<<endl<<"Answer is :" <<n-ans<<endl;
    cout<<n-ans<<endl;
    }

    return 0;
}
