#include <bits/stdc++.h>
using namespace std;

int main() {
	int t,n,i,ans[1000],final;
	char x;
	scanf("%d",&t);
	while(t!=0)
	{
		memset(ans,0,sizeof(ans));
		scanf("%d\n",&n);
		for(i = 0; i<n; i++)
		{
			scanf("%c",&x);
			//printf("x is %c\n",x);
			if(x == '1')
			{
				if(i == 0)
				{
					ans[0] = 1;
					ans[1] = 1;
				}
				else
				{
					if(i == n-1)
					{
						ans[n-1] = 1;
						ans[n-2] = 1;
					}
					else
					{
						ans[i]=1;
						ans[i-1]=1;
						ans[i+1]=1;
					}
				}
			}
		}
		final = 0;
		for(i = 0; i<n; i++){
			if(ans[i] == 0)
				final++;
		}
		printf("%d\n",final);
		t--;
	}
	return 0;
}
