#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int a,b,c,d;
       cin>>a;
       char mad[a+1];
       int A[a];
       int B[1000]={0};
       cin>>mad;
       for(b=0;b<a;b++)
       {
       	 A[b]=mad[b]-48;
       }
       for(b=0;b<a;b++)
       {
       	 if(b==0 && A[b]==1)
       	 {
       	 	B[b]++;
       	 	B[b+1]++;
       	 }
       	 else if(b==(a-1) && A[b]==1)
       	 {
       	 	B[b]++;
       	 	B[b-1]++;
       	 }
       	 else if((b>0 && b<(a-1)) && A[b]==1) 
       	 { 
       	 	B[b]++;
            B[b-1]++;
            B[b+1]++;
       	 }
       }
       c=0;
       for(b=0;b<a;b++)
       {
         B[b]==0 ? c++ : c ;
       }
       cout<<c<<endl;
	}
	return 0;
}
