#include<iostream>
using namespace std;
 
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,c=0;
		cin>>n;
		string s;
		cin>>s;
		for(int i=1;i<=n-2;i++)
		if(s[i] == '1')
		{
			if(s[i-1]!='1')
			s[i-1] = 'b';
			if(s[i+1]!='1')
			s[i+1] = 'b';
			s[i] = 'b';
		}
		if(s[0] == '1')
		{
			s[0] = 'b';
			s[1] = 'b';
		}
		if(s[n-1] == '1')
		{
			s[n-1] = 'b';
			s[n-2] = 'b';
		}
		
		for(int i=0;i<n;i++)
		if(s[i]=='0')
		c++;
		cout<<c<<endl;
		}
		return 0;
	}
