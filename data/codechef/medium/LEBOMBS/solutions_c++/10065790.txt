int count_saved(char a[],int n){
	int count =0,i=0 ;
	if(n==0)
		return 0;
	else if(n==1){
		if(a[0]=='0')
			count++;
	}
	else
	{
		if(a[0]=='0' && a[1]=='0')
			count++;
		FOR(i,1,n-1){
			if(a[i]=='0' && a[i-1]=='0' && a[i+1]=='0'){
					count++;
			}
		}
		if(a[n-2]=='0' && a[n-1]=='0')
			count++;
	}
	return count;
}
 
int main(){
	int test;
	scanf("%d",&test);
	while(test--){
		int n,count=0;
		scanf("%d",&n);
		char s[n];
		scanf("%s",s);
		count = count_saved(s,n);
		printf("%d\n",count);
	}
	return 0;
}
