#include<iostream>
using namespace std;
int main(){
	int t,i,count,explode,n;
	string a;
	cin>>t;
	while(t--){
		cin>>n>>a;
		for(i=0,explode=0,count=0;i<a.length();i++){
			if(i==0&&a[i]=='0'&&a[i+1]=='1')
			explode=0;
			else if((a[i]=='1')||(a[i]=='0'&&a[i+1]=='1'))
				explode=1;
			else if(a[i]=='0'&&explode==1)
			explode=0;
			else if(explode==0&&a[i]=='0'){
				++count;
			}
		}
		cout<<count<<endl;
	}
	return 0;
}
