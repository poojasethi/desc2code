#include<stdio.h>
#include<bits/stdc++.h>

#include<string>
#include<math.h>
#define S(n) scanf("%li",&n);
#include<iostream>
#define s(n) scanf("%d",&n);
#define p(n) printf("%d\n",n);
using namespace std;
int main()
{
 int t;
 s(t)
 while(t--)
 {
     int n;
     s(n)
     string s;
     char p[n];
     memset(p,'0',n);
     cin>>s;
     int c=0;
     for(int i=0;i<n;i++)
     {
         if(s[i]=='1'){
                if(i>0&&i<n-1){
          p[i-1]='1';
          p[i]='1';
          p[i+1]='1';
          }
          else if(i<1){
            p[i+1]='1';
            p[i]='1';
          }
          else{
            p[i-1]='1';
            p[i]='1';
          }
          }
     }
     //cout<<s<<endl;
     for(int i=0;i<n;i++)
         if(p[i]=='0')
          c++;
     p(c)
 }
 return 0;
}
