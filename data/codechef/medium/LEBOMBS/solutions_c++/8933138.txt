#include <iostream>
#include<string>
using namespace std;
int main()
{
    int t,n,x;
    cin>>t;
    while(t--)
    {
        cin>>n;
        x=0;
        string s;
        cin>>s;
        if(n == 1)
        {
            if(s[0] == '0')
            {
                x++;
            }
        }
        else
        {
            for (int i = 1; i<n-1; i++)
            {
                if(s[i] == '0')
                    {
                    if(s[i-1] == '0' && s[i+1] == '0')
                       x++;
                    }
            }
            if(s[0] == '0')
                {
                if(s[1] == '0')
                    x++;
                }
            if(s[n-1] == '0')
            {
                if(s[n-2] == '0')
                    x++;
            }
        }
        cout<<x<<endl;
    }
    
    return 0;
}
