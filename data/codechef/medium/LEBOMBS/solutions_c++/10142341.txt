#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t,n,i;
	cin>>t;
	while(t--)
	{
		int c=0;
		cin>>n;
		char s[n];
		char no1[n];
		for(i=0;i<n;i++)
		no1[i]='0';
		scanf("%s",s);
		
		for(i=0;i<n;i++)
		{
			if(s[i]=='1')
			{
				if(i==0)
				{
					no1[0]='1';
					no1[1]='1';
				}
				else if(i==n-1)
				{
					no1[n-1]='1';
					no1[n-2]='1';
				}
				else
				{
					no1[i-1]='1';
					no1[i]='1';
					no1[i+1]='1';
				}
			}
		}
		
		for(i=0;i<n;i++)
		if(no1[i]=='0')
		c++;
		cout<<c<<endl;
	}
	return 0;
}
