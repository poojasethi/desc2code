#include<bits/stdc++.h>
#define ll long long int
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

char a[10005];
char b[10005];

int main(){
	ll t,n,c,i;

	cin>>t;

	while(t--){
		cin>>n;

		SET(b,'0');

		scanf("%s",a);

		if(a[0]=='1'){b[1]='1';b[0]='1';}
		

		if(a[n-1]=='1'){b[n-2]='1';b[n-1]='1';}
		

		for(i=1;i<n-1;i++){
			if(a[i]=='1'){
				b[i]='1';
				b[i-1]='1';
				b[i+1]='1';
			}
		}
        
        c=0;
		for(i=0;i<n;i++)
			if(b[i]=='0')
				c++;

		cout<<c<<endl;
	}
	return 0;
}
