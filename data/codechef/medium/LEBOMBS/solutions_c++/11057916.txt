#include <iostream>
using namespace std;

int main() {
	int t,n,destroyed,i;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cin>>s;
	//	cout<<s<<endl;
		destroyed=0;
		if(s[0]=='1')
		{
			s[0]='2';
			if(s[1]!='1')
			s[1]='2';
			//cout<<"first"<<endl;
		}
		for(i=1;i<n-1;i++)
		{
			//cout<<"for loop"<<endl;
			//cout<<s[i]<<endl;
			if(s[i]=='1')
			{
			//	cout<<"inside for loop"<<endl;
				s[i]='2';
				s[i-1]='2';
				if(s[i+1]!='1')
				s[i+1]='2';	
				//cout<<"second"<<endl;
			}
		}
		if(s[n-1]=='1')
		{
			s[n-1]='2';
			s[n-2]='2';
		//	cout<<"third"<<endl;
		}
	//	cout<<s<<endl;
		for(i=0;i<n;i++)
		{	
			if(s[i]=='2')
			destroyed++;
		}
		cout<<n-destroyed<<endl;
	//	cout<<endl;
	}	
	return 0;
}
