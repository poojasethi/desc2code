#include<cstdio>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int N;
        scanf("%d",&N);
        char str[N+5];
        scanf("%s",&str);
        int arr[N];
        for(int i=0;i<N;i++){
            arr[i]=0;
        }
        for(int i=0;i<N;i++){
            if(str[i]=='1'){
                arr[i]=1;
                if(i-1>=0){
                    arr[i-1]=1;
                }
                if(i+1<N)
                {
                 arr[i+1]=1;   
                }
            }
        }
        int ans=0;
        for(int i=0;i<N;i++){
            ans+=arr[i];
        }
        printf("%d\n",N-ans);
    }
}
