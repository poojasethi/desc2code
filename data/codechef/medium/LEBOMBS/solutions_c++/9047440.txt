#include<iostream>
#include<string>
using namespace std;
int T,N,i,c;
string str;
int main()
{
    cin>>T;
    while(T--)
    {
        c=0;
        cin>>N;
        cin>>str;
        if(str[0]=='1')
        {
            if(str[1]=='0')
                str[1]='b';
        }
        if(str[N-1]=='1')
        {
            if(str[N-2]=='0')
                str[N-2]='b';
        }
        for(i=1;i<N-1;i++)
        {
            if(str[i]=='1')
            {
                str[i-1]='b';
                if(str[i+1]=='0')
                    str[i+1]='b';
            }
        }
        for(i=0;i<N;i++)
        {
            if(str[i]=='0')
            {
                c++;
            }
        }
        cout<<c<<"\n";
    }
    return 0;
}
