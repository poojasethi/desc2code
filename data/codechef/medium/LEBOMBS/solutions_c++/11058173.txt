#include <iostream>
using namespace std;

int main() {
	int t,n,destroyed,i;
	char *s;
	cin>>t;
	while(t--)
	{
		cin>>n;
		s=new char[n+1];
		cin>>s;
		destroyed=0;
		if(s[0]=='1')
		{
			s[0]='2';
			if(s[1]!='1')
			s[1]='2';
		}
		for(i=1;i<n-1;i++)
		{
			if(s[i]=='1')
			{
				s[i]='2';
				s[i-1]='2';
				if(s[i+1]!='1')
				s[i+1]='2';	
			}
		}
		if(s[n-1]=='1')
		{
			s[n-1]='2';
			s[n-2]='2';
		}
		for(i=0;i<n;i++)
		{	
			if(s[i]=='2')
			destroyed++;
		}
		cout<<n-destroyed<<endl;
	}	
	return 0;
}
