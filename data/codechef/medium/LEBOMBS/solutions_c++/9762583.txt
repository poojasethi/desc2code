#include<iostream>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
int n,cnt=0;
cin>>n;
char bom[1001];
int one[1001];
scanf("%s",bom);
for(int i=0;i<=n;i++)
one[i]=0;
for(int i=0;i<n;i++)
{
if(bom[i]=='1')
{
if(i==0)
{
one[0]=1;
one[1]=1;
}
else if(i==n-1)
{
one[n-1]=1;
one[n-2]=1;
}
else
{
one[i-1]=1;
one[i]=1;
one[i+1]=1;
}
}
}
for(int i=0;i<n;i++)
{
if(one[i]==1)
cnt++;
}
cout<<n-cnt<<endl;
}
return 0;
}
