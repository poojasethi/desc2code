#include<iostream>

using namespace std;

int main(){
	int T;
	cin>>T;
	while(T--){
		char S[1000];
		int N,i;
		int ans=0;
		cin>>N;
		scanf("%s",S);
		if(N==1)
			ans=((int)S[0]-48)^1;
		else if(N==2){
			if(S[1]=='1' || S[0]=='1')
				ans=0;
			else
				ans=2;
		}
		else{
			if(S[N-1]==S[N-2] && S[N-2]=='0')
				ans=ans+1;
			if(S[0]==S[1]&& S[1]=='0')
				ans=ans+1;
			for(i=0;i<N-2;i++)
				if(S[i]==S[i+1]&& S[i+1]==S[i+2]&&S[i+2]=='0')
					ans=ans+1;
			}
		cout<<ans<<endl;
	} 
	return 0;
}
