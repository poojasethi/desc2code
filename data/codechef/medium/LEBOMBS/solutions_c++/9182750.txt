#include<bits/stdc++.h>
using namespace std;

int t,n,i,c;
string arr;
int main()
{
    cin>>t;
    while(t--)
    {
        c=0;
        cin>>n;
        cin>>arr;
        if(arr[0]=='1')
        {
            if(arr[1]=='0')
                arr[1]='b';
        }
        if(arr[n-1]=='1')
        {
            if(arr[n-2]=='0')
                arr[n-2]='b';
        }
        for(int i=1;i<n-1;i++)
        {
            if(arr[i]=='1')
            {
                arr[i-1]='b';
                if(arr[i+1]=='0')
                    arr[i+1]='b';
            }
        }
        for(i=0;i<n;i++)
        {
            if(arr[i]=='0')
            {
                c++;
            }
        }
        cout<<c<<"\n";
    }
    return 0;
}
