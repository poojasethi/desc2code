#include<iostream>
#include<vector>
#include<utility>
#include<algorithm>
#include<stdlib.h>
#include<stdio.h>
using namespace std;

int main()
{
	int t,n,i,sum=0;
	char ch;
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>n;
		char a[n+1];
		i=0;
		//char c=getchar();
		while(i<n)
		{cin>>a[i];
		i++;
		}
		//cin>>ch;
		int b[n];
		for(i=0;i<n;i++)
		{
		b[i]=0;
		}
		for(i=0;i<n;i++)
		{
			if(a[i]=='1')
			{
				if(i>=1 && b[i-1]==0)
				{
				b[i-1]=1;
				sum++;
				}
				if(b[i]==0)
				{
				b[i]=1;
				sum++;
				}
				if(i<=n-2){
				b[i+1]=1;
				sum++;
				}
			}

		}
		cout<<n-sum<<endl;

	}
return 0;
}
