#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<math.h>
#include<iomanip>
#include<stdlib.h>
using namespace std;

long long int t,n,ans,i;
char a[1005];
int main()
{
   scanf("%lld",&t);
   while(t--)
   {    ans=0;
       scanf("%lld",&n);
       scanf("%s",&a);
       if(n>1)
       {

       for(i=1;i<n-1;i++)
       {
           if(a[i]=='0')
           {
               if(a[i+1]=='0' && a[i-1]=='0')
                ans++;
           }
       }
       if(a[0]=='0' && a[1]=='0')
        ans++;
       if(a[n-1]=='0' && a[n-2]=='0')
        ans++;
       }
       else if (n==1)
       {
           if(a[0]=='0')
            ans++;

       }
       printf("%lld\n",ans);
   }
   return 0;
}
