#include<string>
#include <iostream>
using namespace std;

int main() 
{
    string s,s1;
    int t,len;
    cin>>t;
    while(t--)
    {
        cin>>len;
        cin>>s;
        s1=s;
        if(len==1)
            {
                if(s=="1")
                    cout<<"0"<<endl;
                else
                    cout<<"1"<<endl;
                continue;
            }
        for(int i=0;i<len;i++)
        {

            if(s.at(i)=='1')
            {
                if(i==0)
                    {
                        s1.at(i)='b';
                        s1.at(i+1)='b';
                    }
                else if(i==len-1)
                    {
                        s1.at(i)='b';
                        s1.at(i-1)='b';
                    }
                else
                    {
                        s1.at(i-1)='b';
                        s1.at(i)='b';
                        s1.at(i+1)='b';   
                    }
                
            }
        }
        int count=0;
        for(int i=0;i<len;i++)
        {
            if(s1.at(i)!='b')
                count++;    
        }
        cout<<count<<endl;
    }
	return 0;
}
