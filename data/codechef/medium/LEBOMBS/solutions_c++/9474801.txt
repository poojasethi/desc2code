#include<iostream>
#include<cstdio>
#include<string>

using namespace std;

int main()
{
    int t;

    scanf("%d",&t);

    while(t--)
    {
        int n,i,sum=0;

        cin>>n;

        string a;

        cin>>a;

        if(n==1)
        {
            if(a[0]=='0')
            sum++;
        }

        else {

                if((a[0]=='0')&&(a[1]=='0'))
                {
                    sum++;
                }

                for(int i=1;i<n-1;i++)
                {
                    if((a[i]=='0')&&(a[i+1]=='0')&&(a[i-1]=='0'))
                    sum++;
                }

                if((a[n-1]=='0')&&(a[n-2]=='0'))
                {
                    sum++;
                }
        }
        printf("%d\n",sum);
    }

    return(0);
}
