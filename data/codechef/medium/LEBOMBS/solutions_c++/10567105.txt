#include<iostream>
#include<bits/stdc++.h>
#define F(i,x,y) for(i=x;i<y;i++)
#define S(n) scanf("%lld",&n)
#define s(n) scanf("%d",&n)
#define P(n) printf("%lld\n",n)
#define p(n) printf("%d\n",n)
#define LLI long long int
#define R return 0
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,i,count=0,b[1001]={0}; s(n);
		char a[n];
		cin>>a;
		if(n==1)
		{
			if(a[0]=='0')
			count=1;
		}
		else if(n==2)
		{
			if(a[0]=='0'&&a[1]=='0')
			count=2;
		}
		else
		{
			F(i,1,n-1)
        	{
	    		if(a[i]=='1')
		     	{
		     		b[i]=1;
		     		b[i-1]=1;
	    			b[i+1]=1;
	    		}
	    	}
	     	if(a[0]=='1')
	    	{				
	    		b[1]=1;
	    		b[0]=1;
	     	}
	    	if(a[n-1]=='1')
	    	{				
	    		b[n-2]=1;
	    		b[n-1]=1;
	    	}
	    	F(i,0,n)
	    	{
	    		if(b[i]==0)
	    		count++;
	     	}
		}		
		p(count);
	}
	R;
}
