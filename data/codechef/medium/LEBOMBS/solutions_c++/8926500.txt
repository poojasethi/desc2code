#include<iostream>
using namespace std ;
#include<string>
 
int main(){
  int t,i,n,ans;
  cin>>t;
while(t--){
     cin>>n;
     ans=0;
     char input[n],temp[n];
     cin>>input;
     for(i=0;i<n;i++){
     	temp[i]='1';
     }
     for(i=0;i<n;i++){
     	if(input[i]=='1' && i==0 ){
           temp[i]='0';
           temp[i+1]='0';
     	}else if(input[i]=='1' && i==n-1 ){     	
     	   temp[i]='0';
     	   temp[i-1]='0';
     	}else if(input[i]=='1'){
           temp[i]='0';
           temp[i+1]='0';
           temp[i-1]='0';
     	}
     }
     for(i=0;i<n;i++){
     	if(temp[i]=='1')
           ans++;
     }
     cout<<ans<<endl;
}
return 0;
}
