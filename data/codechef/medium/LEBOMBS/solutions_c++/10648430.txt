#include <cstdio>
#include <string>
#include <cstdlib>
#include <algorithm>
using namespace std;

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        char s[n];
        int lst[n];
        for(int i=0;i<n;i++){
            lst[i]=0;
        }
        scanf("%s",s);
        //printf("%s\n",s);
        if(s[0]=='1'){lst[0]=1;lst[1]=1;}
        if(s[n-1]=='1'){lst[n-2]=1;lst[n-1]=1;}
        for(int i=1;i<n-1;i++){
            if(s[i]=='1'){
                lst[i-1]=1;
                lst[i]=1;
                lst[i+1]=1;
            }
        }
        int cnt=0;
        for(int i=0;i<n;i++){
          if(lst[i]==0){cnt++;}
        }
        printf("%d\n",cnt);

/*

        }
*/

    }
}
