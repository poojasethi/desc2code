# include <stdio.h>
# include <algorithm>
# define maxl (int)1e5
# define nINF (-(int)1e9-1)
int A[maxl];
long long int maxarr[maxl][2];
int main()
{
	int T,N;
	long long int ovmax;
	scanf("%d",&T);
	while(T--)
	{
		maxarr[0][1]=nINF;
		scanf("%d",&N);
		scanf("%d",A);
		maxarr[0][0]=A[0];
		ovmax=A[0];
		for(int i=1;i<N;i++)
		{
			scanf("%d",A+i);
			maxarr[i][0]=A[i];
			if(maxarr[i-1][0]>0)
				maxarr[i][0]+=maxarr[i-1][0];
			maxarr[i][1]=std::max(maxarr[i-1][1]+A[i],maxarr[i-1][0]);
			ovmax=std::max(ovmax,std::max(maxarr[i][0],maxarr[i][1]));
		}
		printf("%lld\n",ovmax);
	}
	return 0;
}
