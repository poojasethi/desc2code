#include <bits/stdc++.h>
#define LL long long int
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
        LL n;
        cin >> n;
        LL arr[n];
        for(int i = 1; i <= n ; i++) {
            cin >> arr[i];
        }
        LL maxleft[n+2];
        LL maxright[n+2];
        memset(maxleft, 0, sizeof(maxleft));
        memset(maxright, 0, sizeof(maxright));

        for(int i = 1; i <= n ;i++) {
            maxleft[i] = max(arr[i], maxleft[i-1] + arr[i]);
        }

        for(int i = n; i > 0 ; i--) {
            maxright[i] = max(arr[i], maxright[i+1] + arr[i]);
        }

        LL ans = LONG_LONG_MIN;
        for(int i = 1 ; i <= n ; i++){
            ans = max(ans, maxleft[i]);
        }
        for(int i = 1 ; i <= n; i++) {
            ans = max(ans, maxleft[i-1] + maxright[i+1]);
        }
        cout << ans << endl;
    }
    return 0;
}
