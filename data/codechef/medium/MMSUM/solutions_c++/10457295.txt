#include <bits/stdc++.h>
#define forn(i, a, n) for (int i = a; i < n; ++i)
#define forr(i, a, n) for (int i = (n) - 1; i >= a; --i)
#define pb push_back
#define inf 0xdeadbee
 
using namespace std;
const int N = 100005;
typedef vector<int> vi;
typedef pair<int, int> pii;
typedef long long ll;
 
int t, n, a[N];
ll pre[N], suf[N];
 
int main() {
	scanf("%d", &t);
	while (t--) {
		scanf("%d", &n);
		forn (i, 1, n+1) scanf("%d", &a[i]);
		if (*max_element(a + 1, a + n + 1) <= 0) printf("%d\n", *max_element(a + 1, a + n + 1));
		else {
			pre[0] = suf[n+1] = 0;
			forn (i, 1, n+1) pre[i] = max(pre[i-1], 0LL) + a[i];
			forr (i, 1, n+1) suf[i] = max(suf[i+1], 0LL) + a[i];
			ll mx = INT_MIN, s = 0;
			forn (i, 1, n+1) {
				s += a[i];
				if (s > mx) mx = s;
				if (s < 0) s = 0;
			}
			forn (i, 1, n+1) mx = max(mx, pre[i-1] + suf[i+1]);
			printf("%lld\n", mx);
		}
	}
}
