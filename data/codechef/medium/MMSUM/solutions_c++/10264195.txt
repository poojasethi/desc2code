#include<cstdio>
#include<iostream>
using namespace std;
int main()
{

    int t,n,i;
    long long int cs,ms,temp;
    scanf("%d",&t);
    while(t--)
    {
        temp=0;
        scanf("%d",&n);
        long long int a[n],e[n],s[n];
        for(i=0;i<n;i++)
        {
             scanf("%lld",&a[i]);

        }
        ms=a[0];
        cs=a[0];
        s[0]=a[0];
        for(i=1;i<n;i++)
        {
            cs=max(a[i],cs+a[i]);
            ms=max(cs,ms);
            s[i]=cs;

        }

        ms=a[n-1];
        cs=a[n-1];
        e[n-1]=a[n-1];
        for(i=n-2;i>=0;i--)
        {
            cs=max(a[i],cs+a[i]);
            ms=max(cs,ms);
            e[i]=cs;

        }
        for(i=2;i<=n-2;i++)
            ms=max(ms,s[i-1]+e[i+1]);
        printf("%lld\n",ms);

    }
}
