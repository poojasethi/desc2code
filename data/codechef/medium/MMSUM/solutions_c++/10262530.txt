#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int a[n];
        for(int i=0;i<n;i++)
        {
            scanf("%d",&a[i]);

        }
        long long b[n],c[n];
        long long sum1=0,sum2=0;
        b[0]=a[0];
        c[n-1]=a[n-1];
        sum1=a[0];
        sum2=a[n-1];
        for(int i=1;i<n;i++)
        {
            if(sum1<0)
                sum1=a[i];
            else
                sum1+=a[i];
            b[i]=sum1;
        }
        int j=n-2;
        for(int i=n-2;i>=0;i--)
        {
            if(sum2<0)
                sum2=a[i];
            else
                sum2+=a[i];
            c[j]=sum2;
            j--;
        }

        long long maximum,sum3=0;
        for(int i=0;i<n-1;i++)
        {
            sum3=0;
            if(a[i]>=0)
            {
                sum3=max(b[i]+c[i+1],max(b[i],c[i+1]));
            }
            else
            {
                if(i!=0)
                {
                    sum3=max(max(b[i-1],c[i+1]),b[i-1]+c[i+1]);
                }
                else
                {
                    sum3=max(b[i],c[i+1]);
                }
            }
            if(i!=0&&sum3>maximum)
            {
                maximum=sum3;
            }
            else 
            {
                if(i==0)
                    maximum=sum3;
            }

        }
        printf("%lld\n",maximum);
    }
}
