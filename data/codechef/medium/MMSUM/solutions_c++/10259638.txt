#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll a[100005],l[100005],r[100005];
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        memset(l,0,sizeof(l));
        memset(r,0,sizeof(r));
        ll n,ans=LONG_LONG_MIN;
        scanf("%lld",&n);
        for(int i=1;i<=n;i++)
            scanf("%lld",&a[i]);
        for(int i=1;i<=n;i++)
            l[i]=max(l[i-1]+a[i],a[i]);
        for(int i=n;i>=1;i--)
            r[i]=max(r[i+1]+a[i],a[i]);
        for(int i=1;i<=n;i++)
            ans=max(ans,l[i]);
        for(int i=1;i<=n;i++)
            ans=max(ans,l[i-1]+r[i+1]);
        printf("%lld\n",ans);
    }
    return 0;
}
