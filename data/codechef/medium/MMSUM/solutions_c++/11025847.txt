#include<iostream>
#include<stdio.h>
using namespace std;
 
#define GC getchar_unlocked
#define PC putchar_unlocked
 
inline int scan()
{
    int n=0,a=1;
    char c=GC();
 
    while(c < 48 || c >57) 
    {
        if(c=='-')
        {
            a=-1;
        }
        c=GC();
    }
    while(c>47 && c<58)
    {
        n=(n<<3)+(n<<1)+c-'0';
        c=GC();
    }
    return a*n;
}
 
int main()
{
    int t,flag,flagp,i,n;
    
    long int a[100050];
    long long int b[100050],c[100050];
    long long int max,temp;
    
    t=scan();
    while(t--)
    {
        for(i=0;i<100050;i++)
        {
            a[i]=b[i]=c[i]=0;
        }
        
        flagp=0;
        flag=0;
        n=scan();
        //ans=0;
        max=0;
        
        for(i=1;i<=n;i++)
        a[i]=scan();
        
        for(i=1;i<=n;i++)
        {
            if(a[i]>0)
            {
                flag=1;
                break;
            }           
        }
        
        for(i=1;i<=n;i++)
        {
            if(a[i]<0)
            {
                flagp=1;
                break;
            }           
        }
        
        if(flag==0)
        {
            max=a[1];
            for(i=2;i<=n;i++)
            {
                if(a[i]>max)
                {
                    max=a[i];
                    
                }           
            }
            cout<<max<<endl;
        }   
        else if(flagp==0)
        {
            max=0;
            for(i=1;i<=n;i++)
            {
                max=max+a[i];
            }
            cout<<max<<endl;
        }
        else
        {
            b[n]=0;
            for(i=n-1;i>=1;i--)
            {
                b[i]=b[i+1]+a[i+1];
                if(b[i]<0)
                b[i]=0;
            }
            
            c[1]=0;
            for(i=2;i<=n;i++)
            {
                c[i]=c[i-1]+a[i-1];
                if(c[i]<0)
                c[i]=0;
            }
            max=-1000;
            for(i=1;i<=n;i++)
            {
                temp=b[i]+c[i];
                if(temp>max)
                max=temp;
            }
            cout<<max<<endl;
        }
    }
    return 0;   
}
