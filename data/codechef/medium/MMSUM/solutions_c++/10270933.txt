using namespace std;
ll ed[100005],st[100005],ans=1e15;
ll calc_end(ll a[], ll n, ll ed[])
{
    ll i,max_end=0;
    for(i=1;i<=n;i++)
    {
        max_end= max(a[i],max_end+a[i]);
        ans=max(ans,max_end);
        ed[i]=max_end;
    }
}

ll calc_start(ll a[], ll n, ll st[])
{
    ll i,max_end=0;
    for(i=n;i>=1;i--)
    {
        max_end= max(a[i],max_end+a[i]);
        ans=max(ans,max_end);
        st[i]=max_end;
    }
}

int main()
{
    ll t,i;
    cin>>t;
    while(t--)
    {
        ll a[100005],i,n;
        ans= -1e15;
        cin>>n;
        for(i=1;i<=n;i++)
            cin>>a[i];
        calc_end(a,n,ed);
        calc_start(a,n,st);
        for(i=2;i<=n-1;i++)
            ans= max(ans, ed[i-1]+st[i+1]);
        cout<<ans;nl;
    }

}
