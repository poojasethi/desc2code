#include<iostream>
using namespace std;
 
long long int a[100005],b[100005],c[100005];

 int main()
 {
 
      long long int t,n,i,temp;
      scanf("%lld",&t);
      while(t--)
      {
    	scanf("%lld",&n);
    	for(i=0;i<n;i++) scanf("%lld",&a[i]);
    	bool flag=0,flagp=0;
    	for(i=0;i<n;i++)
		{
			if(a[i]>0)
			{
				flag=1;
				break;
			}			
		}
		
		for(i=0;i<n;i++)
		{
			if(a[i]<0)
			{
				flagp=1;
				break;
			}			
		}
		
		if(flag==0)
		{
			long long int maxo=a[0];
			for(i=1;i<n;i++)
			{
				if(a[i]>maxo)
				{
					maxo=a[i];
					
				}			
			}
			cout<<maxo<<endl;
		}	
		else if(flagp==0)
		{
			long long int maxo=0;
			for(i=0;i<n;i++)
			{
				maxo=maxo+a[i];
			}
			cout<<maxo<<endl;
		}
		else
		{
			b[n-1]=0;
			for(i=n-2;i>=0;i--)
			{
				b[i]=b[i+1]+a[i+1];
				if(b[i]<0)
				b[i]=0;
			}
			
			c[0]=0;
			for(i=1;i<n;i++)
			{
				c[i]=c[i-1]+a[i-1];
				if(c[i]<0)
				c[i]=0;
			}
			long long int maxo=-2000000000;
			for(i=0;i<n;i++)
			{
				temp=b[i]+c[i];
				if(temp>maxo)
				maxo=temp;
			}
			printf("%lld\n",maxo);
		
		}
	
		
      }
      
 }
