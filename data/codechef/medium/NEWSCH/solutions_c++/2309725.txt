#include<iostream>
#include<cstdio>
int t, n;
const int MOD = 1000000007;
long long power(int x, int a){
    if(a == 0)
        return 1;
    if(a == 1)
        return x%1000000007;
    else if(a&1){
        return (x*power((x*x)%1000000007,a/2))%1000000007;
    }
    else{
        return power((x*x)%1000000007,a/2)%1000000007;
    }
}
int fastPow(int a, int n) {
	int p = 1;
	while (n > 0) {
		if (n % 2 == 1) {
			p = (long long) p * a % MOD;
		}
		a = (long long) a * a % MOD;
		n /= 2;
	}
	return p;
}
int main(){
    long long res = 0;
    //freopen("input.txt", "r", stdin);
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        if(n&1){
            res = fastPow(3, n)-3;
            if(res < 0)
                res += 1000000007;
            res %= 1000000007;
            printf("%lld\n", res);
        }
        else{
            res = fastPow(3, n)+3;
            res %= 1000000007;
            printf("%lld\n", res);
        }
    }
    return 0;
}
