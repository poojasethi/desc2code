#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
ll pw(ll a,ll b)
{
    if(b==0)
    return 1;
    else if(b==1)
    return a;
    else
    {
        ll val=pw(a,b/2);
        if(b%2==0)
        {
            return (val*val)%mod;
        }
        else
        {
            return ((val*val)%mod)*a;
        }
    }
}
int main()
{
    long long int t,n,ans;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        if(n%2==0)
        {
            ans=((pw(3,n))+3)%mod;
        }
        else
        {
            ans=(pw(3,n)-3)%mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
