#include <cstdio>


const int MOD = 1000000007;
typedef long long LL;
// returns a^n % MOD
// exponentiation by squaring
int powmod(int a,int n)
{
	int p=1;
	for(;n;)
	{
		if(n%2) p=LL(p)*a%MOD;
		if(n/=2) a=LL(a)*a%MOD;
	}
	return p;
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int ans = powmod(3,n)+3*powmod(-1,n);
        printf("%d\n",ans);
    }
}
