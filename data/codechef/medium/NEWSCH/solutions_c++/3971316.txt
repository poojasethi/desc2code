#include <cstdio>


#define MOD 1000000007LL

typedef long long int ll;

ll pow_mod(ll n, ll k) {
  ll ret;
  if(k == 1) {
    return n % MOD;
  }
  if(k == 0) {
    return 1;
  }
  if(k % 2 == 0) {
    ret = n * n;
    ret = ret % MOD;
    ret = pow_mod(ret, (k / 2));
    return ret;
  }
  else {
    ret = pow_mod(n, k - 1);
    ret *= n;
    ret %= MOD;
    return ret;
  }
}

void work() {
  int N;
  scanf("%d", &N);
  int k = N / 2;
  ll pownum = pow_mod(9, k);
  ll ans;
  if(N % 2 == 0) 
    ans = pownum + 3;
  else
    ans = 3 * (pownum - 1);
  ans %= MOD;
  printf("%lld\n", ans);
}
  


int main() {
  int T;
  scanf("%d", &T);
  while(T--) work();
  return 0;
}
