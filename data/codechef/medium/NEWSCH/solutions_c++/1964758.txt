#include <iostream>
#include <stdio.h>

using namespace std;
const int MOD = 1000000007;

int pow(int a, int n)
{
	int p = 1;
	while (n > 0)
    {
		if(n % 2)
			p = (long long) p * a % MOD;
		if(n/=2)
            a = (long long) a * a % MOD;
	}
	return p;
}

int main()
{
	int T,N,ans;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&N);
		 ans = pow(3, N) + 3 * (N % 2 ? -1 : 1);
		if(ans < 0)
        {
			ans += MOD;
		}
		ans=ans%MOD;
        printf("%d \n",ans);
	}
	return 0;
}
