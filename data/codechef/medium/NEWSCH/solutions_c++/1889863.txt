ll mod = (ll)(1e9 + 7);
ll a[2][2] = {{0, 1}, {3, 2}};

ll c[2][2];
void powmatrix (int m){
	//cerr << m << endl;
	if(m == 1) return;
	if(m % 2)
	{
		powmatrix(m - 1);
		b[0][0] = 0;b[1][0] = 3;b[0][1] = 1;b[1][1] = 2;
		forn(i,2){
			forn(j,2){
				c[i][j] = 0;
				forn(k,2){
					c[i][j] += a[i][k] * b[k][j];  
					c[i][j] %= mod;
				}
			}
		}
		a[0][0] = c[0][0];a[1][0] = c[1][0];a[0][1] = c[0][1];a[1][1] = c[1][1];
		
	}
	else {
		powmatrix(m / 2);
		b[0][0] = a[0][0];b[1][0] = a[1][0];b[0][1] = a[0][1];b[1][1] = a[1][1];
		forn(i,2){
			forn(j,2){
				c[i][j] = 0;
				forn(k,2){
					c[i][j] += a[i][k] * b[k][j];   
					c[i][j] %= mod;
				}
			}
		}
		a[0][0] = c[0][0];a[1][0] = c[1][0];a[0][1] = c[0][1];a[1][1] = c[1][1];
		
	}
}
