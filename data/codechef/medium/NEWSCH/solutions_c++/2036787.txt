#include <cstdio>
using namespace std;

long long mod_pow(long long base,long long exponent,long long modulus){
  long long result=1;

  while(exponent){
    if(exponent&1){
      result*=base;
      if(result>=modulus) result%=modulus;
    }
    exponent>>=1;
    base*=base;
    if(base>=modulus) base%=modulus;
  }
  return result;
}

  int t,n;
  long long answer;

#ifndef ONLINE_JUDGE

  while(t--){
    scanf("%d",&n);
    answer=mod_pow(3,n,1000000007);
    if(n&1){

    }
    else{
      answer+=3;
      if(answer>1000000007) answer%=1000000007;
    }
    printf("%lld\n",answer);
  }
  return 0;
}
