#include "stdio.h"
#include "stdlib.h"

#define MOD 1000000007

long long int expo3(long int b)
{
	long long int result=1,a=3;
	while(b)
	{
		if(b&1) result = (result*a)%MOD;
		b >>= 1 ;
		a = (a*a)%MOD;
	}

	return result;
}



int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long int n;
		scanf("%ld",&n);
		long long int result = expo3(n);
		if(n&1) result = (result - 3 + MOD)%MOD;
		else result = (result + 3)%MOD;
		printf("%lld\n",result);
	}

	return 0;
}
