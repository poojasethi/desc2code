int modpow(int a,int b,int mod)
{
	long long product,psequence;
	product=1;
	psequence=a;
	while(b>0)
	{
		if(b&1)
		product=(product*psequence)%mod;
		b>>=1;
		psequence=(psequence*psequence)%mod;
	}
	return (int)product;
}
