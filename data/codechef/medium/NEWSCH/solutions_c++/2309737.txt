#include<iostream>
#include<cstdio>
int t, n;
const int MOD = 1000000007;
long long power(long long x, int a){
    if(a == 0)
        return 1;
    if(a == 1)
        return x;
    else if(a&1){
        return (x*power((x*x)%MOD,a/2))%MOD;
    }
    else{
        return power((x*x)%MOD,a/2);
    }
}
int main(){
    long long res = 0;
    //freopen("input.txt", "r", stdin);
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        if(n&1){
            res = power(3, n)-3;
            if(res < 0)
                res += 1000000007;
            res %= 1000000007;
            printf("%lld\n", res);
        }
        else{
            res = power(3, n)+3;
            res %= 1000000007;
            printf("%lld\n", res);
        }
    }
    return 0;
}
