#include <cstdio>


#define MOD 1000000007LL

typedef long long int ll;

ll pow_mod(ll n, ll k) {
  ll ret;
  if(k == 1) {
    return n % MOD;
  }
  if(k == 0) {
    return 1;
  }
  if(k % 2 == 0) {
    ret = n * n;
    ret = ret % MOD;
    ret = pow_mod(ret, (k / 2));
    return ret;
  }
  else {
    ret = pow_mod(n, k - 1);
    ret *= n;
    ret %= MOD;
    return ret;
  }
}

void work() {
  int N;
  scanf("%d", &N);
  ll pownum = pow_mod(3, N);
  switch(N % 2) {
  case 0:
    pownum += 3;
    break;
  case 1:
    pownum -= 3;
    break;
  }
  pownum %= MOD;
  printf("%lld\n", pownum);
 }
  


int main() {
  int T;
  scanf("%d", &T);
  while(T--) work();
  return 0;
}
