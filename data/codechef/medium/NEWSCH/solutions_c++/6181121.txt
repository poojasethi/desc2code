ll repeat(int a,int b)
{   
    ll x= 1 , y = a;
    while(b > 0)
    {
        if(b%2 == 1)
        { 
           x =( x * y ) % mod;
        } 
        y = (y * y) % mod;
        b /= 2;
    }
    return x;
}
