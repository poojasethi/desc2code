#include<bits/stdc++.h>
using namespace std;
#define ll long long int
//#define M 1000000007
ll pw(ll a, ll b,ll M){
  ll r;
  if(b==0) return 1;
  r = pw(a,b/2,M);
  r = (r*r)%M;
  if(b%2) r = (r*a)%M;
  return r;
}
int main()
{
ll n,t;
scanf("%lld",&t);
while(t--)
{
	scanf("%lld",&n);
	ll power=pw(3,n,1000000007);
	if(n%2==0)
	{
		printf("%lld\n",power+3);
	}
	else printf("%lld\n",power-3);
}

return 0;
}
