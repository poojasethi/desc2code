void copymatrix(ll a[2][2],ll b[2][2],int row,int col)
{
 int i,j;
 for(i=0;i<row;++i)
 for(j=0;j<col;++j)
 a[i][j]=b[i][j];
}
void multiply(ll a[2][2],ll b[2][2],int row,int col)
{
    ll c[2][2];
    int i,j,k;
    for(i=0;i<row;i++)
       for(j=0;j<col;++j)
         c[i][j]=0;

    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            c[i][j]=0;
            for(k=0;k<col;k++)
            {
                c[i][j]=(c[i][j]+a[i][k]*1LL*b[k][j]%mod)%mod;
            }
        }
    }
    copymatrix(a,c,row,col);
}
void matpow(ll a[2][2],int b,int row,int col)
{
     ll c[2][2];
    int i,j;
    for(i=0;i<row;i++)
        for(j=0;j<col;++j)
         c[i][j]=0;
    for(i=0;i<col;i++)
        c[i][i]=1;

    while(b)
    {
        if(b&1)
            multiply(c,a,row,col);
        multiply(a,a,row,col);
        b/=2;
    }
    copymatrix(a,c,row,col);
}
