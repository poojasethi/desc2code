#include <iostream>
#include <cstdio>
#define MOD 1000000007
using namespace std;
long long modPow(long long n)
{
long long x=1,y=3;
while (n > 0)
{
if (n%2 == 1)
x=(x*y)%MOD;
y = (y*y)%MOD;
n/= 2;
}
return x%MOD;
}
int main()
{
long long int ans =0 ;
int T;
scanf ("%d",&T);
while (T--)
{
long long N;
scanf ("%lld",&N);
if (N == 1)
{
printf ("4\n");
continue;
}
if (N%2 == 0)
{
ans = (3 + modPow(N))%MOD;
printf ("%lld\n",ans);
}
else
{
ans = (modPow(N) - 3)%MOD;
printf ("%lld\n",ans);
}
 
 
}
return 0;
}
