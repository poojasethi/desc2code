#define MOD 1000000007

using namespace std;

unsigned long long power(unsigned long long int x,int n)
{
	if(n==1) return x;
	else if(n%2) return (x%MOD)*(power((x*x)%MOD,(n-1)/2)%MOD)%MOD;
	else return power((x*x)%MOD,n/2)%MOD;
}
 
int main()
{
	int nCases;
	scanf("%d",&nCases);
	while(nCases--)
	{
		int n;
		scanf("%d",&n);
		unsigned long long int k=1;
		k=power(3,n);
		if(n%2) k=(k+ MOD-3)%MOD;
		else k=(k+3)%MOD;
		printf("%llu\n",k);
	}
	return 0;
}
