#include <iostream>
using namespace std;
#define ll long long int
#define mod 1000000007

ll power(ll a,ll b)
{
	ll r;
	if(b==0) return 1;
	r=power(a,b/2);
	r=(r*r)%mod;
	if(b%2==1)
	r=(r*a)%mod;
	return r;
}
int main()
{
	int i,t;
	ll tc;
	cin>>t;
	while(t--)
	{
		cin>>tc;
		if(tc%2==0)
	           tc=3+power(3,tc)%mod;				//a(n)=3*(-1)^n+(3)^n
		else
		   tc=-3+power(3,tc)%mod;
		cout<<tc<<endl;
	}
return 0;
}
