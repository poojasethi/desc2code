/*calculated by solving GP of 4*(   3^(n-1) - 3^(n-2) + 3^(n-3) ....+/- 3  )*/


#include<cstdio>
#define ll long long int
#define M 1000000007

ll power3(int n)
{
	ll a=1;
	ll p=3;
	while(n) 
	{
		
		if(n&1)
		{
			a=(a*p)%M;
		}
		
		p=(p*p)%M;
		n=n>>1;
		
	}
	return a;
}


int main()
{
	int t;
	scanf("%d",&t);
	
	while(t--)
	{
		int n;
		scanf("%d",&n);
		ll ans;
	
		ans=power3(n)+3 *(n % 2 ? -1 : 1);
		if(ans<0)
		ans=ans+M;
		
		else if(ans>M)
		ans=ans%M;

		
		printf("%lld\n",ans);
	}
}
