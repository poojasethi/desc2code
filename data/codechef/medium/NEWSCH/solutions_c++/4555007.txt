//Linear homogenous Recurrence relation
#include<iostream>
#include<cstdio>
#define LL long long int
#define mod 1000000007
using namespace std;
//Repeated squaring 
LL power(LL a,LL b)
{	LL temp=a,ans=1;
	while(b>0)
	{	if(b%2==0)
		{	temp=(temp%mod*temp%mod)%mod;
			b/=2;
		}
		else
		{	ans=(ans%mod*temp%mod)%mod;
			b--;
		}
	}
	return ans;
}
int main()
{
	int t;
	LL n,val;
	scanf("%d",&t);
	while(t--)
	{	scanf("%lld",&n);
		val=power(3,n);
		if(n%2==0) val=(val+3)%mod;
		else	   val=(val-3+mod)%mod;
		printf("%lld\n",val);
	}
	return 0;
}
