#include <iostream>
#define MOD 1000000007

using namespace std;

long long int fastpow(int n, int p)
{
    if(p == 0) return 1;

    if(!(p&1)) {
        long long int ans = fastpow(n, p/2);

        return (ans*ans)%MOD;
    }

    return (n*fastpow(n,p-1))%MOD;
}
int main()
{
    int t;
    int n;

    cin >> t;

    while(t--) {
        cin >> n;

        cout << (fastpow(3, n)+((n&1)?-3:3))%MOD << endl;
    }
}
