#include<stdio.h>
#include<math.h>
#include<stdlib.h>

unsigned long long int power(unsigned long long int,unsigned long long int);

int main()
{
    unsigned long long int num,temp=1;
    int test;
    scanf("%d",&test);
    while(test)
    {
        num=0;
        scanf("%d",&num);
        temp=power(3,num);
        temp=temp%1000000007;
        if(num%2==0)
        {
            temp=(temp+3)%1000000007;
        }
        else
        {
            temp=temp-3;
        }
        printf("%lld\n",temp);
        test--;
    }
    return 0;
}

unsigned long long int power(unsigned long long int x,unsigned long long int n)
{
    unsigned long long int res=0;
    if (n==1)
    {
        return x;
    }
    else
    {
        if(n%2==0)
        {
            res=power(x,n/2);
            res=(res*res)%1000000007;
        }
        else
        {
            res=power(x,(n-1)/2);
            res=(x*res*res)%1000000007;
        }
    }
    return res;
}
