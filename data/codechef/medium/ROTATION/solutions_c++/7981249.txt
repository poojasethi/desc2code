#include<stdio.h>
#include<math.h>

int main()
{
	long int n,m,q,count,i,j,k,r,s;
    char ch;
    scanf("%ld %ld",&n,&m);
    long int a[n];
    
    for(i=0;i<n;i++) scanf("%ld",&a[i]);
    
    k=0;
    while(m--)
    {
    	scanf(" %c %ld",&ch,&q);
    	if(ch=='R') //print valus in q index;
    	{
    	    j=(k+q-1)%n;
    	    printf("%ld\n",a[j]);
        }
		else if(ch=='A')
		{
			k=(k-q+n)%n;
		}
		else 
		{
			k=(k+q)%n;
		}
		
	}
	
	 
}
