#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    int n,m,i,j;
    char c;
    sd(n);
    sd(m);
    int a[n+2];
    for(i=0;i<n;i++)
        sd(a[i]);
    j=0;
    while(m--)
    {
        c=getchar();
        c=getchar();
        sd(i);
        if(c=='C')
            j=(j+i)%n;
        else if(c=='A')
            j=(j+n-i)%n;
        else
            printf("%d\n",a[(i-1+j)%n]);
    }
    return 0;
}
