#include<iostream>
using namespace std;
int main()
{
    unsigned long long int n;
    cin>>n;
    unsigned long long int m;
    cin>>m;
    int arr[n+1];
    unsigned long long int i;
    for(i=1;i<n+1;i++)
        cin>>arr[i];
    unsigned long long int clock=0;
    while(m--)
    {
        char a;
        unsigned long long int index;

        cin>>a>>index;

        if (a=='C')
            {clock=clock+index;
            }
        if (a=='A')
            clock =clock+n-index;
        if (a=='R')
        {
            unsigned long long int totalrotation=clock%n;

            index=index+totalrotation;

            if (index>n)
            index=index-n;

            cout<<arr[index]<<'\n';

        }
    }
    return 0;
}
