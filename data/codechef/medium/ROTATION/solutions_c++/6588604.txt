#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[100005];
int main()
{
    int n,m,i,t;
    scanf("%d%d",&n,&m);
    for(i=1;i<=n;i++)
        scanf("%lld",&a[i]);
    //for(i=1;i<=n;i++)
      //  printf("%lld ",a[i]);
   // cout<<"\n";
    int pos=1;
    while(m--)
    {
        char ch;
        int d;
        //cout<<"enter: ";
        cin>>ch>>d;
        if(ch=='R')
        {
            //printf("inisde R: \n");
            t=(pos+d-1)%n;
            if(t==0)
                t=n;
            cout<<a[t]<<"\n";
        }
        else if(ch=='C')
        {
           // printf("inisde C\n:");
            pos=(pos+d)%n;
            if(pos==0)
                pos=n;
        }
        else if(ch=='A')
        {
            //printf("inisde A\n:");
                if(d%n==0)
                    pos=pos;
                else
                {
                    d=d%n;
                    if(d<=pos)
                    {
                        pos=pos-d;
                        if(pos==0)
                            pos=n;
                    }
                        else
                        {
                            t=d-pos;
                            pos=n-t;
                        }
                }
        }
        //printf("pos=%d\n",pos);
    }
    return 0;
}
