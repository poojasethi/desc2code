int arr[100001];

static inline int fastio()
{
    int x=0;
    char c=gc();
    while(c<'0'||c>'9')
        c=gc();
    while(c>='0'&&c<='9')
    {
        x=(x<<1)+(x<<3)+c-'0';
        c=gc();
    }


    return x;
}

void fastWrite(int x)
{
int i=0;
char str[18]={' '};
do
{
    str[i++]=x%10+'0';
    x/=10;
}
while(x!=0);
i--;
for(;i>=0;i--)
{
    pc(str[i]);
}
pc('\n');
return;
}




int main()
{
int n,i,q,k,shift=0;
char str[3];
n=fastio();q=fastio();
for(i=0;i<n;i++)
    arr[i]=fastio();
while(q--)
{
    scanf("%s",str);
    k=fastio();
    if(str[0]=='A')
    {
        shift=(shift+(n-k))%n;
    }
    else if(str[0]=='C')
    {
        shift=(shift+k)%n;
    }
    else
    {

        int pos=(shift+k-1)%n;
        fastWrite(arr[pos]);
        //printf("%d\n",arr[pos]);
    }
}
return 0;
}
