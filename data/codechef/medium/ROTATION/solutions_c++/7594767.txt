int main()
{
	int n,m,x,d;
	char ch;
	vector<int> vec;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)

		vec.push_back(x);
	}
	int start=0;
	while(m--)
	{
		getchar();
		scanf("%c%d",&ch,&d);
		if(ch=='R')
		{
			d=d-1;
			if(start>d)
				printf("%d\n",vec[n-start+d]);
			else
				printf("%d\n",vec[d-start]);
		}
		else if(ch=='C')
			start=(n+start-d)%n;
		else
			start=(start+d)%n;
	}
