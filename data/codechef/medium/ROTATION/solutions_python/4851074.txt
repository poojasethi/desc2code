from sys import stdin
def main():
    a=stdin.readline().split()
    b=map(int,stdin.readline().split())
    n,j=int(a[0]),0
    for i in range(int(a[1])):
        c=stdin.readline().split()
        c1=c[0]
        c2=int(c[1])
        if c1=='R':
            print b[(j+c2-1)%n]
        elif c1=='C':
            j+=c2
        elif c1=='A':
            j-=c2
if __name__ == '__main__':
    main()
