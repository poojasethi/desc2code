#include<iostream>
#include<stdio.h>
using namespace std;
#define MAX 1000
int main()
{
    double ncr[1001*1002/2];
    int T,S,N,M,K,i,j,k;
    double ans;
    cin>>T;
    for(i=0;i<=MAX;i++)
    {
        ncr[i*(i+1)/2+1]=1;
        for(j=1;j<i;j++)
            ncr[i*(i+1)/2+1+j]=ncr[i*(i-1)/2+1+j]+ncr[i*(i-1)/2+1+j-1];
        ncr[i*(i+1)/2+1+i]=1;
    }
    while(T--)
    {
        cin>>S>>N>>M>>K;
        if(K>=N)
        {
            printf("%0.8lf\n",0.0);
            continue;
        }
        if(K==0||N==S)
        {
           printf("%0.8lf\n",1.0);
           continue;
        }
        M--;
        S--;
        N--;
        for(ans=0;K<=M&&K<=N;K++)
        {
            if(S-M>=N-K)
            ans+=ncr[M*(M+1)/2+1+K]*ncr[(S-M)*(S-M+1)/2+1+N-K];
        }
       ans=ans/ncr[S*(S+1)/2+1+N];
        printf("%0.8lf\n",ans);
    }
    return 0;
}
