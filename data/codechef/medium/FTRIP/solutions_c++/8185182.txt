#include <stdio.h>
#include <memory.h>
 
double c[1111][1111];
 
int main() {
  memset(c, 0, sizeof(c));
  for (int i=0;i<=1000;i++) c[i][0] = 1.0;
  for (int i=1;i<=1000;i++)
    for (int j=1;j<=i;j++) c[i][j] = c[i-1][j] + c[i-1][j-1];
  int t;
  scanf("%d", &t);
  while (t--) {
    int s, n, m, k;
    scanf("%d %d %d %d", &s, &n, &m, &k);
    s--; n--; m--;
    double total = 0;
    for (int i=k;i<=m;i++) {
      int j = n-i;
      if (j >= 0 && j <= s-m) total += c[m][i] * c[s-m][j];
    }
    printf("%.6lf\n", total / c[s][n]);
  }
  return 0;
}
