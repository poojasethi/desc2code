#include <stdio.h>
#include <algorithm>
//#include <conio.h>
using namespace std;
#define SMAX 1000

int T, S, N, M, K;

double BioT[SMAX+1][SMAX+1];

void fill() {
    BioT[0][0] = 1.0;
    for(int i = 1; i <= SMAX; i++) {
        BioT[i][0] = BioT[i][i] = 1.0;
        for(int j = 1; j < i; j++)
            BioT[i][j] = BioT[i-1][j-1] + BioT[i-1][j];
    }
}

double F(int a, int b, int n, int k) {
    int imin = max(k, n - b);
    int imax = min(n, a);
    double res = 0.0;
    for(int i = imin; i <= imax; i++)
        res += BioT[a][i] * BioT[b][n-i];
    return res;
}

double solve() {
    double res = F(M-1, S-M, N-1, K) / BioT[S-1][N-1];
    //if(res > 1.0) res = 1.0;
    return res;
}

int main() {
    //freopen("in.txt", "rt", stdin);
    
    fill();
    int T;
    scanf("%d", &T);
    while(T--) {
        scanf("%d%d%d%d", &S, &N, &M, &K);
        printf("%.9lf\n", solve());
    }
    
    //getch();
    return 0;
}
