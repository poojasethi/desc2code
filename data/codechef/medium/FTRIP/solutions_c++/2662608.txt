using namespace std;

double C(int n, int k) {
	double c = 1.0;
	for (int i = 0; i < k; i++) {
		c *= double(n - i) / (1 + i);
	}
	return c;
}

double P(int S, int N, int M, int K) {
	double p = 0;
	for (int f = K; f < min(M, N); f++) {
		p += C(M - 1, f) * C(S - M, N - f - 1);
	}
	return p;
}

double Q(int S, int N) {
	return C(S - 1, N - 1);
}

double calc(int S, int N, int M, int K) {
	return P(S, N, M, K) / Q(S, N);
}

void solve() {
	int T;
	scanf("%d", &T);

	for (int i = 0; i < T; ++i) {
		int S, N, M, K;
		scanf("%d%d%d%d", &S, &N, &M, &K);
		printf("%f\n", calc(S, N, M, K));
	}
}

int main() {
	solve();
	return 0;
}
