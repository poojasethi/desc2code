#include<stdio.h>
double arr[1001][1001];
int max(int a,int b)
{
	if(a<b)return b;
	return a;
}
int min(int a,int b)
{
	if(a>b)return b;
	return a;
}
void init()
{
	int i,j;
	for(i=0;i<1001;i++)
	{
		arr[i][0]=arr[i][i]=1.0;
		for(j=1;j<i;j++)
			arr[i][j]=arr[i-1][j-1]+arr[i-1][j];
	}
}
int main()
{
	init();
	int s,n,m,k,t,i;
	double num,den;
	scanf("%d",&t);
	while(t--)
	{
		num=den=0;
		scanf("%d%d%d%d",&s,&n,&m,&k);
		den = arr[s-1][n-1];
		for(i=max(k, n-1 + m-s);i<=min(n-1,m-1);i++)
			num+=arr[m-1][i]*arr[s-m][n-1-i];
		printf("%.6lf\n",num/den);
	}
	return 0;
}
