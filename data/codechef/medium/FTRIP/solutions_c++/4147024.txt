#define MAXN 1001
double c[MAXN][MAXN];

void initialize()
{
	int i,j;
	c[0][0]=1.0;
	for(i=1;i<MAXN;i++)
	{
		c[i][0] = 1.0;
		for(j=1;j<=i;j++)
			c[i][j] = c[i-1][j]+c[i-1][j-1];
	}
}
int main()
{
	int t;
	int i;
	int s,n,m,k;
	double ans;
	initialize();
	t = scan_d();
	//printf("2choose1 : %lf\n",c[1000][10] );
	while(t--)
	{
		s = scan_d();
		n = scan_d();
		m = scan_d();
		k = scan_d();
		ans = 0;
		for(i=k;i<m;i++)
		{
			if((n-i-1)>=0 && (s-m)>=(n-i-1))
			ans+= c[m-1][i]*c[s-m][n-i-1];
		}
		ans/=c[s-1][n-1];
		printf("%lf\n",ans);
	}
	return 0;
}
