#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

double ncr[1001][1001];

void preprocess() {
  ncr[0][0] = 1;
  for(int i=1;i<1001;++i) {
    ncr[i][0] = ncr[i][i] = 1.0;
    for(int j=1;j<i;++j) ncr[i][j]=ncr[i-1][j]+ncr[i-1][j-1];
  }
}

int main() {
  preprocess();
  int t, s, n, m, k;
  cin >> t;
  while(t--) {
    cin >> s >> n >> m >> k;
    double p, q;
    q = ncr[s-1][n-1];
    p = 0.0;
    int l = min(m-1,n-1);
    for(int f=max(k,n-1+m-s);f<=l;++f) p += ncr[m-1][f]*ncr[s-m][n-f-1];
    printf("%.6f\n",p/q);
  }
  return 0;
}
