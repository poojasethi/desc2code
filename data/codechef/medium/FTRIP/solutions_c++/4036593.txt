#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main(){
    int i,j,k=0,s=0,n=0,m=0,t=0;
    double sum,q;
    double a[1001][1001];
    a[0][0]=1.0;
        for(i=1;i<=1000;i++){
                a[i][i]=1.0;
                a[i][0]=1.0;
            for(j=1;j<i;j++){
                a[i][j]=a[i-1][j]+a[i-1][j-1];
            }
        }
        scanf("%d",&t);
    while(t--){
        scanf("%d %d %d %d",&s,&n,&m,&k);
        sum=0.0;
        for(i=k;i<=min(m-1,n-1);i++){
            sum+=(a[m-1][i]*a[s-m][n-i-1]);
        }

        q=a[s-1][n-1];
        printf("%f\n",sum/q);
    }
return 0;
}
