#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>

using namespace std;

double triangle[1005][1005];  //Pascal's Triangle

double c(short n,short r)
{
	if(triangle[n][r]!=0.0)
		return triangle[n][r];
	else
	{
		double result;
		if(n<2||r<1||r>=n)
			result=1.0;
		else if(r==1||r==n-1)
			result=(double)n;
		else
			result=c(n-1,r)+c(n-1,r-1);
		triangle[n][r]=result;
		return result;
	}
}

void preprocess()
{
	for(int i=0;i<1000;i++)
	{	
		triangle[i][0]=triangle[i][i]=1.0;
		for(int j=1;j<i;j++)
			triangle[i][j]=triangle[i-1][j]+triangle[i-1][j-1];
	}
}

int main()
{
	#ifndef ONLINE_JUDGE
	freopen("input","r",stdin);
	#endif
	
	short t;
	
//	for(int i=0;i<1005;i++)
//		for(int j=0;j<1005;j++)
//			triangle[i][j]=0.0;
	
	preprocess();
	
	cin>>t;
	
	while(t--)
	{
		short s,n,m,k,kmax;
		double result;
		
		cin>>s>>n>>m>>k;
		
		s--;
		n--;
		m--;
		
		if(m<n)
			kmax=m;
		else
			kmax=n;
		
		double den=c(s,n);
		
		result=0.0;
		for(short i=k;i<=kmax;i++)
		{
			double a=c(s-m,n-i),b=c(m,i);
//			printf("%lf %lf\n",a,b);
			result+=a*b;
		}
//		printf("%lf\n%lf\n\n",result,den);
		result/=den;
		
		if(result>1.0)
			result=1.0;
			
		printf("%.6lf\n",result);
	}
	return 0;
}
