double ncr[1010][1010];
void compute_ncr(int n){
  int i,j;
  rep(i,n) rep(j,n) ncr[i][j]=0;
  rep(i,n) ncr[i][i]=1;
  rep(i,n) ncr[i][0]=1;
    
  FOR(i,2,n){
       FOR(j,1,i){
       ncr[i][j]=ncr[i-1][j-1]+ncr[i-1][j]; 
       }
    }
//    rep(i,10 ){ rep(j,10) printf("%lf ",ncr[i][j]); printf("\n");}
    return;
}
int main()
{
   compute_ncr(1010);
   int T,t,N,S,M,K,i;
   double num,den;

   num=0.0;
   scanf("%d %d %d %d",&S,&N,&M,&K);
   den=ncr[S-1][N-1];
   for(i=max(K,N+M-1-S); i<=min(M-1,N-1); i++){
       num+=ncr[M-1][i]*ncr[S-M][N-1-i];
   }
     printf("%.9lf\n",num/den);
   }   
   return 0;
}
