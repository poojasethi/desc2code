#include<iostream>
using namespace std;
double C[1002][1002];
int main()
{
	int S,N,M,K,i,t,j;
	cin>>t;
	
	for(i=0;i<=1000;i++)
	{
		for(j=0;j<=1001;j++)
		C[i][j]=0;
	}
	
	/*for(i=0;i<1000;i++)
	{for(j=0;j<1000;j++)
	cout<<C[i][j]<<" ";
	cout<<endl;}*/
	
    for(i=0;i<=1000;i++)
	C[i][0]=1;
    
	for(i=1;i<=1000;i++)
	{
		for(j=1;j<=i+1;j++)
		C[i][j]=C[i-1][j-1]+C[i-1][j];
	}
	/*for(i=1;i<=5;i++){
     
     for(j=0;j<=i+1;j++){
        printf("%f ",C[i][j]);
        
      }
     printf("\n");
  }*/
  //cout<<C[5][2]<<endl;
	//cin>>t;
	while(t--)
	{
	    cin>>S>>N>>M>>K;
		double P=0.0,Q=0.0,ans;
		Q=C[S-1][N-1];
		//cout<<Q<<endl;
		int kk=min(N-1,M-1);
		//cout<<kk<<endl;
		for(i=K;i<=kk;i++)
		{
			P=P+((C[M-1][i])*(C[S-M][N-i-1]));
		}
		ans=P/Q;
		//cout<<ans<<endl;
		printf("%.6f\n",ans);
	}
	
	
}
