#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

double triangle[1000][1000];  //Pascal's Triangle

double c(short n,short r)
{
	if(triangle[n][r]!=0.0)
		return triangle[n][r];
	else
	{
		double result;
		if(n<=1||r<=0||r>=n)
			result=1.0;
		else
			result=c(n-1,r)+c(n-1,r-1);
		triangle[n][r]=result;
		return result;
	}
}

int main()
{
	#ifndef ONLINE_JUDGE
	freopen("input","r",stdin);
	#endif
	
	short t;
	
	cin>>t;
	
	while(t--)
	{
		short s,n,m,k,kmax;
		double result;
		
		cin>>s>>n>>m>>k;
		
		s--;
		n--;
		m--;
		
		if(k<m+n-s)
			k=m+n-s;
		
		if(m<n)
			kmax=m;
		else
			kmax=n;
		
		result=0.0;
		for(short i=k;i<=kmax;i++)
			result+=c(s-m,n-i)*c(m,i);
		
		result/=c(s,n);
			
		printf("%.6lf\n",result);
	}
	return 0;
}
