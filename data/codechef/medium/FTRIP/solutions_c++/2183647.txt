#include <cstdio>
#include <cstdlib>
#include <iostream>

using namespace std;

double triangle[1000][1000];  //Pascal's Triangle

void preprocess()
{
	for(int i=0;i<1000;i++)
	{	
		triangle[i][0]=triangle[i][i]=1.0;
		for(int j=1;j<i;j++)
			triangle[i][j]=triangle[i-1][j]+triangle[i-1][j-1];
	}
}

int main()
{
	#ifndef ONLINE_JUDGE
	freopen("input","r",stdin);
	#endif
	
	short t;
	
	preprocess();
	
	cin>>t;
	
	while(t--)
	{
		short s,n,m,k,kmax;
		double result;
		
		cin>>s>>n>>m>>k;
		
		s--;
		n--;
		m--;
		
		if(m<n)
			kmax=m;
		else
			kmax=n;
		
		double den=triangle[s][n];
		
		result=0.0;
		for(short i=k;i<=kmax;i++)
			result+=triangle[s-m][n-i]*triangle[m][i];
		
		result/=den;
		
//		if(result>1.0)
//			result=1.0;
			
		printf("%.6lf\n",result);
	}
	return 0;
}
