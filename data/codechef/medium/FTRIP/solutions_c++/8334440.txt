#include <iostream>
#include <iomanip>
#include <string.h>
#include <stdio.h>

using namespace std;

int const maxS = 1000;

double c[maxS+1][maxS+1];
void makec(){

    memset(c,0,sizeof(c));
    for(int i = 0; i<=maxS; i++){
        c[i][0]=c[i][i]=1;
        for(int j=1;j<i;j++){
            c[i][j]=c[i-1][j]+c[i-1][j-1];
        }
    }

}

int main()
{
    makec();
    int T;
    cin>>T;

    for(int i = 0;i<T; i++){

        int S,N,M,K;

        cin>>S>>N>>M>>K;

        double Numerator=0,den = c[S-1][N-1];
        for(int j = max(K,N-1+M-S); j<=min(M-1,N-1); j++){
            Numerator+= c[M-1][j]*c[S-M][N-1-j];
        }
        printf("%.9lf\n", Numerator/den);

    }

    return 0;
}
