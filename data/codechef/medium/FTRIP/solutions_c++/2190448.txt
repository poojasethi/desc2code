const int maxS=1000;
double dp[maxS + 1][maxS + 1];		
int s,n,m,k;
void nCr()
{
	memset(dp, 0,sizeof(dp));
	for(int i = 0; i <= maxS; i++)
	{	
		dp[i][0] = dp[i][i] = 1.0;
		for(int j = 1; j <i; j++)
			dp[i][j] = dp[i-1][j]+dp[i-1][j-1];
	}
}
int main()
{
    nCr();
    int tc;
    cin>>tc;
    while(tc--){
    cin>>s>>n>>m>>k;
    double ans=0.0;
    double d=dp[s-1][n-1]; // select m-1 from s-1;
    for(int i=max(k,n-1+ m-s);i <= min(n-1,m-1);i++)
    {
       ans += dp[m-1][i]*dp[s-m][n-1-i];      
    }
    ans= ans/d;
    printf("%.6lf\n",ans);
   }

   return 0; 
}
