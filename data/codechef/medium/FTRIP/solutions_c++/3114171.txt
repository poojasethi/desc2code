#include<iostream>
#include<stdio.h>
#define MAX 1001
using namespace std;
double temp[MAX][MAX];
int main()
{
	int T,S,N,M,K,i,j;
	double num = 0;
	cin>>T;
	for(i=0;i<MAX;i++)
	{
		temp[i][0]=temp[i][i] = 1.0;
		for(j=1;j<i;j++)
		{
			temp[i][j] = temp[i-1][j]+temp[i-1][j-1];
		}
	}
	while(T--)
	{
		cin>>S>>N>>M>>K;
		num  =0;
		for(i=K;i<=min(N-1,M-1);i++)
		{
			num += temp[M-1][i]*temp[S-M][N-i-1];
		}
		printf("%.9lf\n",num/temp[S-1][N-1]);
	}
	return 0;
}
