#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
long long int getSum(long long int n)
{
    long long int sum;
    /*Single line that calculates sum*/
    for(sum=0; n > 0; sum+=n%10,n/=10);
    if(sum<10)
    	return sum;
    else
    	getSum(sum);
}
long long int function(long long int a,long long int d,long long int l,long long int r)
{
	long long int a_l=a;
		long long int sum;
		long long int SUM=0;
		sum=getSum(a_l);
		long long int nextsum=-1;
		long long int interval=0;
		vector<long long int> v;
		v.push_back(sum);
		while(nextsum!=sum)
		{
			a_l=a_l+d;
			nextsum=getSum(a_l);
			SUM=SUM+nextsum;
			interval=interval+1;
			v.push_back(nextsum);
		//	cout<<nextsum<<" ";
		}
		//cout<<endl;
		//cout<<"SUM= "<<SUM<<" inteval= "<<interval<<endl;
		long long int total_sum,complete_intervals,partial_intervals;
		complete_intervals=(r-l)/interval;
		partial_intervals=(r-l)%interval;
		total_sum=complete_intervals*SUM;
		/*for(int i=0;i<=partial_intervals;i++)
		{
			cout<<v[i]<<" ";
		}
		cout<<endl;*/
		for(int i=0;i<=partial_intervals;i++)
		{
			total_sum=total_sum+v[i];
		}
		return total_sum;
}
int main()
{
	long long int t;
	scanf("%lld",&t);
	while(t--)
	{
	   long long int a,d,l,r;
		scanf("%lld %lld %lld %lld",&a,&d,&l,&r);
		long long int a1,a2;
		a1=function(a,d,1,r);
		a2=function(a,d,1,l-1);
		//cout<<a1<<" "<<a2<<endl;
		a2=a1-a2;
		printf("%lld\n",a2);

	}
}
