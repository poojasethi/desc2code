#include <iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--)
     {
       long long a,d,l,r,x;
       long long s=0;
       cin>>a>>d>>l>>r;

       for(long long i=l;i<l+9;i++)
       {   x=(a%9+(((d%9)*((i-1)%9))%9))%9;
           if(x!=0)
           s+=x;
           else s+=9;
       }
       long long result=((r-l+1)/9)*s;
       for(long long i=l;i<l+(r-l+1)%9;i++)
       {   x=(a%9+(((d%9)*((i-1)%9))%9))%9;
       if(x!=0)
           result+=x;
           else result+=9;
       }
       cout<<result<<endl;
     }
}
