#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll arr[10];
ll solve(ll i)
{
    if(i<=0)
        return 0;
    ll ans=(i/9)*arr[9]+arr[i%9];
    return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        ll a,d,l,r;
        arr[0]=0;
        scanf("%lld%lld%lld%lld",&a,&d,&l,&r);
        int i=1;
        while(i<10){
            ll num=a+(i-1)*d;
            while(num>9){
                ll sum=0;
                while(num!=0){
                    sum+=num%10;
                    num/=10;
                }
                num=sum;
            }
            arr[i++]=num;
        }
        for(i=2;i<10;i++)
            arr[i]+=arr[i-1];
        printf("%lld\n",solve(r)-solve(l-1));
    }
    return 0;
}
