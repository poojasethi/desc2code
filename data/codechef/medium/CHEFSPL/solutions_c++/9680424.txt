#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
 
char s[1000002];
 
bool check(long l, long r, long li, long ri) 
{
	for(long i=l; i<r+1; i++) 
	{
		while( li<=ri && s[li]!=s[i] )
			li++;
		if (li > ri)
			return false;
		li++;
	}
	return true;
}
 
int main()
{
	long d;
	scanf("%ld", &d);
	while(d--)
	{
		
		scanf("%s", s);
		long l = strlen(s);
		bool flag = false;
		if( l==1 )
			printf("NO \n");
		else if( l%2==0 )
		{
			for(long i=0; i<l/2; i++)
				if( s[i]!=s[i+l/2] )
				{
					flag=true;
					break;
				}
			if( flag==true )
				printf("NO \n");
			else
				printf("YES \n");
		}
		else
		{
			flag |= check(0, l/2-1, l/2, l-1);
			flag |= check(l/2+1, l-1, 0, l/2);
			if( !flag )
				printf("NO \n");
			else
				printf("YES \n");
		}
	}
	return 0;
}
