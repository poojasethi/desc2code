#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int test,j,flag;
	char str[1000001];
	scanf("%d",&test);
	while(test){
		test--;
		flag=0;
		scanf("%s",str);
		int len=strlen(str)/2+1;
		if(strlen(str)<2){
			printf("NO\n");
		}
		else if(strlen(str)%2==0){
			for(int i=0,j=len-1;i<len-1&&j<strlen(str);i++,j++){
				if(str[i]!=str[j]){
					flag++;
				}
			}
			if(flag>0){
				printf("NO\n");
			}
			else{
				printf("YES\n");
			}
		}
		else{
		for(int i=0,j=len;i<len&&j<strlen(str);i++,j++){
			if(str[i]!=str[j]){
			flag++;
			j--;	
			}
		}
		if(flag<=1){
			printf("YES\n");
		}
		else{
			flag=0;
			for(int i=0,j=len-1;i<len-1&&j<strlen(str);i++,j++){
			if(str[i]!=str[j]){
			flag++;
			i--;	
			}
		}
		if(flag<=1){
			printf("YES\n");
		}
		else{
			printf("NO\n");
		}
	}
  }
  }
  return 0;
}
