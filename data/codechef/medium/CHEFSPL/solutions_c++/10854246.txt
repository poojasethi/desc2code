ll findans()
{
    ll flag,i,j,count;
    flag=0;
    if(n%2==0)
    {
        for(i=0;i<n/2;i++)
        {
            if(a[i]!=a[i+n/2])
            {
                flag=0;
                return 0;
            }
        }
        return 1;
    }
    else
    {
        flag=0;
        count=0;
        i=0;
        j=(n+1)/2;
        while(count<n/2 && i<=(n/2) && j<n)
        {
            if(a[i]==a[j])
            {
                count++;
                i++;
                j++;
            }
            else
            {
                i++;
            }
        }
        if(count==(n/2))
        {
            return 1;
        }
        else
        {
            count=0;
            i=0;
            j=(n-1)/2;
            while(count<n/2 && i<(n/2) && j<n)
            {
                if(a[i]==a[j])
                {
                    count++;
                    i++;
                    j++;
                }
                else
                {
                    j++;
                }
            }
            if(count==(n/2))
            {
                return 1;
            }
            else
            {
                return 0;
            }
        }
    }
}
