#include<bits/stdc++.h>
using namespace std;
string str;
bool double_string(int a,int a_last,int j,int j_last){
int flag=0,i;
for(i=a;i<=a_last;++i,++j){
        if(j>j_last)return false;
    if(str[i]==str[j])continue;
    else if(str[i]==str[j+1]&&flag==0)
    {
        ++j;flag=1;
    }
    else{return false;}
}
return true;
}
main()
{
int n,k,t,flag,len,mid,i;
scanf("%d",&t);
while(t--){

cin>>str;
len=str.size();
mid=len/2;
if(!(len&1))
{   flag=1;
    for(i=0;i<mid;++i)
    {
    if(str[i]!=str[mid+i])
    {cout<<"NO\n";
    flag=0;
    break;}
    }
    if(flag)printf("YES\n");
}
else if(len==1)printf("NO\n");
else
{  bool check=false;
     check|=double_string(0,mid-1,mid,len-1);
     check|=double_string(mid+1,len-1,0,len/2);
     if(check) cout<<"YES\n";
     else cout<<"NO\n";
}

}
return 0;
}
