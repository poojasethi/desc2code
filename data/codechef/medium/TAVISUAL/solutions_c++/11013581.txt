#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
        long long int n,c,q,l,r;
        cin>>n>>c>>q;
        while(q--)
        {
            cin>>l>>r;
            if(c>=l && c<=r)
            {
                int diff = c-l;
                c=r-diff;
            }
        }
        cout<<c<<"\n";
    }
    return 0;
}
