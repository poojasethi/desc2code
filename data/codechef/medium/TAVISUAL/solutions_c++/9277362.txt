#include <iostream>
using namespace std;

int main()
{
    int t,n,c,q,left,right,diff;
    cin >> t;
    while(t--)
    {
        cin>>n>>c>>q;
        while(q--)
        {
            cin>>left>>right;
            if((c>=left)&(c<=right))
                c=right+left-c;
        }
        cout<<c<<endl;
    }
    return 0;
}
