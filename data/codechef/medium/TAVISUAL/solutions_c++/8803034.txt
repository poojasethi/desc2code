#include<cstdio>
using namespace std;
int main()
{
    int tc;
    scanf("%d",&tc);
    while(tc--)
    {
        long int n,c,q,i,l,r,temp;
        scanf("%ld%ld%ld",&n,&c,&q);
        long int pos=c;
         for(i=0;i<q;i++)
         {
             scanf("%ld%ld",&l,&r);
             if(pos>=l && pos<=r)
             {
                 temp=pos-l;
                 pos=r-temp;
             }
         }
         printf("%ld\n",pos);
    }
    return 0;
}
