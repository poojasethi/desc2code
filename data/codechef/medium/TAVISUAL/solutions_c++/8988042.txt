#include<iostream>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    int t,n,c,q,l,r,i;
    cin>>t;
    while(t--)
    {
        cin>>n>>c>>q;
        for(i=0;i<q;i++)
        {
            cin>>l>>r;
            if(c>=l && c<=r)
                c=l+r-c;
        }
        cout<<c<<endl;
    }
    return 0;
}
