#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,c,q;
		cin >> n >> c >> q;
		int i;
		vector<int> nums(n);
		for( i = 0; i < n; i++ )
		{
			nums[i] = i+1;
		}
		for( i = 0; i < q; i++ )
		{
			int l,r;
			cin >> l >> r;
			if( c >= l && c <= r )
			c = l+r-c;
		}
		cout << c << endl;
	}
	return 0;
}
