#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
      int a,b,c;
      cin>>a>>b>>c;
      int posn=b;
      int l,r;
      while(c--)
      {
      	cin>>l>>r;
      	if(posn<=r && posn>=l)
      	{
      		posn=(l+r)-posn;
      	}
      }
      cout<<posn<<endl;
	}
	return 0;
}
