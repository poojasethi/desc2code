#include<bits/stdc++.h>

using namespace std;

int main()
{
    long tt;
    //ifstream cin ("a.txt");
    cin>>tt;
    int n,c,q,l,r;
    while(tt--){
        cin>>n>>c>>q;
        while(q--){
            cin>>l>>r;
            if(c>=l && c<=r)
                c=r-(c-l);
        }
        cout<<c<<endl;
    }
}
