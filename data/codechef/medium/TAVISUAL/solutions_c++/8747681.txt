#include <iostream>
using namespace std;
long int cal(long int n,long int l,long int r,long int c){
	if(l>c || c>r)
		return c;
	long int pos;
	pos=c-l;
	return r-pos;
}
int main(){
	int t;
	cin>>t;
	while(t--){
		long int n,c,q,l,r;
		cin>>n>>c>>q;
		while(q--){
			cin>>l>>r;
			c=cal(n,l,r,c);
			//cout<<c<<endl;
		}
		cout<<c<<endl;
	}
	return 0;
}
