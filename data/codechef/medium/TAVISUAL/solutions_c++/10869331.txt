#include<cstdio>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        int n,c,q;
        scanf("%d %d %d",&n,&c,&q);
        for(int i=0;i<q;i++){
            int l,r;
            scanf("%d %d",&l,&r);
            if(l<=c && c<=r){
                c=r+l-c;
            }
        }
        printf("%d\n",c );
    }
}
