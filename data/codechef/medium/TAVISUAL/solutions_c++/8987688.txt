#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n,c,q;
        cin>>n>>c>>q;
        int l,r;
        while(q--){
            cin>>l>>r;
            int temp;
            if(c<=r && c>=l){
            temp = l+r-c;
            c=temp;
            }
        }
        cout<<c<<endl;
    }
}
