#include<bits/stdc++.h>

using namespace std;

int main(void)
{
    int t; scanf("%d",&t);
    while(t--)
    {
       int n,c,q; scanf("%d %d %d",&n,&c,&q);
       int pos=c;
       
       while(q--)
       {
          int l,r; scanf("%d %d",&l,&r);
          
          if(pos>=l && pos<=r)
          {
              int mind=min(abs(pos-l),abs(r-pos));
              bool left=abs(pos-l)<=abs(r-pos),
                   right=abs(r-pos)<abs(pos-l);
              if(left)
                pos=r-mind;
              else if(right)
                pos=l+mind;
          }
       }
       printf("%d\n",pos);
   }
   
   return 0;
}
