#include <iostream>
using namespace std;


int main() {
	int t;
	cin>>t;
	while(t--){
	    int n;
	    cin>>n;
	    int h[n],k[n];
	    int m = -1;
	    for(int i=0;i<n;i++){
	        cin>>h[i];
	        if(h[i] > m) m = h[i];
	    }
	    for(int i=0;i<n;i++){
	        cin>>k[i];
	    }
	    m*=2;
	    int dp[m+1],sum=0;
	    for(int i=0;i<=m;i++){
	        dp[i] = 99999;
	    }
	    dp[0] = 0;
	    for(int i=1;i<=m;i++){
	        int minn=99999;
	        for(int j=0;j<n;j++){
	            if(i>=k[j] && (dp[i-k[j]]+1) < dp[i]){
	                dp[i] = dp[i-k[j]]+1;
	            }
	        }
	    }
	    /*for(int i=1;i<=m;i++)
	        cout<<dp[i]<<" ";
	    cout<<"\n";*/
	    for(int i=0;i<n;i++){
	        sum += dp[h[i]*2];
	        //cout<<h[i]*2<<" "<<dp[h[i]*2]<<" ";
	    }
	    
	    cout<<sum<<"\n";
	}
	return 0;
}
