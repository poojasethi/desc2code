#include<iostream>
using namespace std;
#define INF 9999999
int n,x,ans[1001];
void cal(int a[])
{
	int i,j,min,cur;
	ans[0]=0;
	for(j=1;j<=1001;++j)
	{
		min=INF;			//in dp we calculate answer for every subproblem
		for(i=0;i<n;++i)
		{
			cur=ans[j-a[i]]+1;
			if(j>=a[i]  && cur<min)
			min=cur;
		}
		ans[j]=min;
	}
	//cout<<ans[x];	
}
int main()
{
	int a[500],i,test,h[500];
	cin>>test;
	while(test--)
	{
		int z=0;
		cin>>n;
		for(i=0;i<n;++i)
		cin>>h[i];
		for(i=0;i<n;++i)
		cin>>a[i];		
		cal(a);
		for(i=0;i<n;++i)
		z+=ans[2*h[i]];
		//cout<<ans[i];
		cout<<z<<"\n";
	}
	
	return 0;
}
