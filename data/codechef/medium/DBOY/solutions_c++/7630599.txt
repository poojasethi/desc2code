#include <bits/stdc++.h>
using namespace std;
int h[505],k[505];
unsigned long long int sum[1005];
int main() {
	// your code goes here
	int t,n;
	cin>>t;
	while(t--)
	{
	    cin>>n;
	    for(int i=0;i<n;i++)
	    {
	        cin>>h[i];
	        h[i]*=2;
	   }
	        for(int i=0;i<n;i++)
	    {
	        cin>>k[i];
	     }
	     int mx=*max_element(h,h+n);
	     sum[0]=0;
	     for(int i=1;i<=mx;i++)
	    {
	        sum[i]=INT_MAX;
	     }
	     for(int i=1;i<=mx;i++)
	     {
	         for(int j=0;j<n;j++)
	         {
	             if(k[j]<=i)
	             sum[i]=min(sum[i],1+sum[i-k[j]]);
	         }
	     }
	     unsigned long long int s=0;
	   //  for(int i=0;i<=mx;i++)
	   //  cout<<sum[i]<<" ";
	     
	       for(int i=0;i<n;i++)
	        s+=sum[h[i]];
	        cout<<s<<endl;
	}
	
	return 0;
}
