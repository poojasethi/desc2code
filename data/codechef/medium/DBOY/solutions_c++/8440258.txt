#include<bits/stdc++.h>
#define max 100000000
using namespace std;
int main()
{
	int t,n,i,y,j;
	int h[501],k[501];
	long long int ans;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=1;i<=n;i++)
		{
			scanf("%d",&h[i]);
		}
		y=*max_element(h+1,h+n+1);
		for(i=1;i<=n;i++)
		{
			scanf("%d",&k[i]);
		}
		int count[n+1][2*y+1];
		count[0][0]=0;
		for(i=1;i<=2*y;i++)
		{
			count[0][i]=max;
		}
		for(i=1;i<=n;i++)
		{
			for(j=0;j<=2*y;j++)
			{
				count[i][j]=count[i-1][j];
				if(k[i]<=j)
				{
					count[i][j]=min(count[i][j],1+count[i][j-k[i]]);
				}
				
			}
		}
		/*for(i=0;i<=n;i++)
		{
			for(j=0;j<=2*y;j++)
			cout<<count[i][j]<<" ";
			cout<<"\n";
		}*/
		ans=0;
		for(i=1;i<=n;i++)
		{
			ans+=count[n][2*h[i]];
		}
		printf("%lld\n",ans);
	}
	return 0;
}
