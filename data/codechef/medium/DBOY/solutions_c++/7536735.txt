#define sz(c) c.size()

#define inf 1000001
#define gc() getchar_unlocked()
#define pc(x) putchar_unlocked(x)

using namespace std;


typedef vector<int> vi;
typedef vector<vi> vvi;
typedef pair<int,int> pii;
typedef map<int,int> mii;

int N;
int dp[1001];
int pp[501];
int orders[501];


static inline int fastio()
{
    int x=0;
    char c=gc();
    while(c<'0'||c>'9')
    {
        c=gc();
    }
    while(c>='0'&&c<='9')
    {
        x=(x<<1)+(x<<3)+c-'0';
        c=gc();
    }
    return x;
}


static inline void fastWrite(int x)
{
  char s[11];
  int i=0;
  do
  {
      s[i++]=x%10+'0';
      x/=10;
  }while(x);
  --i;
  while(i>=0)
  {
      pc(s[i--]);
  }
  pc('\n');

}










void Initialize()
{
memset(dp,inf,sizeof dp);
dp[0]=0;
for(int i=1;i<1001;i++)
{
    for(int j=0;pp[j]<=i&&j<N;j++)
    {if(dp[i-pp[j]]+1<dp[i])
    dp[i]=dp[i-pp[j]]+1;}
}
}





int main()
{
int T;
//ios_base::sync_with_stdio(false);
//cin>>T;
T=fastio();
while(T--)
{
    //cin>>N;
    N=fastio();
    int i;
    for(i=0;i<N;i++)
        orders[i]=fastio();
        //cin>>orders[i];
    for(i=0;i<N;i++)
        pp[i]=fastio();
        //cin>>pp[i];
    int mincost=0;
    sort(pp,pp+N);
    Initialize();
    for(i=0;i<N;i++)
     mincost+=dp[2*orders[i]];
     fastWrite(mincost);
     //cout<<mincost<<endl;
}
return 0;
}
