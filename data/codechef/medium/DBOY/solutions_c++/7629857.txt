#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <limits.h>

using namespace std;

int input()
{
    int n=0;
    char c=getchar_unlocked();
    while(c < '0' || c >'9') {c=getchar_unlocked();}
    while(c>='0' && c<='9')
    {
        n=(n<<3)+(n<<1)+c-'0';
        c=getchar_unlocked();
    }
    return n;
}

int* H;
int* K;
int R[1000];

void process(){
	int m;
	for(int i = 0; i < 1000; i++){
		m = 2000;
		if(R[i] != 1){
			for(int j = 0; j < (i+1)/2; j++){
				m = min(m, R[j] + R[i-j-1]);
			}
			R[i] = m;
		}
	}
}

int main(){
	int T,N;
	T = input();
	int res = 0;
	while(T--){
		N = input();
		H = new int[N];
		K = new int[N];
		for(int i = 0; i < N; i++){
			H[i] = input();
		}
		for(int i = 0; i < 1000; i++){
			R[i] = 2000;
		}
		for(int i = 0; i < N; i++){
			K[i] = input();
			R[K[i]-1] = 1;
		}
//		for(int i = 0; i < 50; i++){
//			cout << R[i] << "\t";
//		}
//		cout << endl;
		process();	
		res = 0;	
		for(int i = 0; i < N; i++){
			res += R[2*(H[i])-1];
		}
		cout << res << endl;
		delete(H);
		delete(K);
	}
}
