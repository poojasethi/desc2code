#include <limits>
#include <stdio.h>
#include <algorithm>
#include <iostream>
 
using namespace std;

int main()
{
	int t,k,DP[1024];
	int *denominations;
	int *numbers;

	scanf("%d" , &t);

	int min = 100000;
		
	while(t--)
	{
		scanf("%d" , &k);
		numbers = new int[k];
		denominations = new int[k];
		for (int i = 0; i < 1024; ++i)
		{
			DP[i] = min;
		}
		DP[0] = 0;
		
		for(int i=0; i<k;i++)
		{
			scanf("%d" , &numbers[i]);
		} 

		for(int i=0; i<k;i++)
		{
			scanf("%d" , &denominations[i]);
		} 
		
//		sort(denominations , denominations+k);
//		sort(numbers , numbers+k);

		for(int i=0;i<k;i++)
		{
			
			for(int w=denominations[i],f=w;f<1024;f++)
			{
				if(DP[f-w] + 1 < DP[f])
				{
					DP[f] = DP[f-w] + 1;
					// cout<<i<<" "<<f<<" "<<min<<endl;
				}
			}

		}

		int totsum = 0;

		for(int i=0;i<k;i++)
		{
			totsum += DP[2*numbers[i]];
		}

/*		for(int i=0;i<k;i++)
		{
			cout<<numbers[i]<<" ";
		}
		cout<<endl;

		for(int i=0;i<k;i++)
		{
			cout<<denominations[i]<<" ";
		}
		cout<<endl;

		for(int i=0;i<1024;i++)
		{
			cout<<DP[i]<<" ";
		}
		cout<<"\n";
*/
		printf("%d\n" , totsum);

	}

	return 0;
}
