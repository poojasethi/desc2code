#include<bits/stdc++.h>
using namespace std;
int main()
{
   int t,i,num,maxsum,D,min_petrol,p;
scanf("%d",&t);
   while(t--)
   {    maxsum=-1;
      scanf("%d",&num);
      int dist[num+1],array[num+1],petrol[1001];
      memset(petrol,1,sizeof(petrol));
      for(i=0;i<num;++i)
        {scanf("%d",&D);
        dist[i]=2*D;
        maxsum=max(maxsum,dist[i]);
        }
        for(i=0;i<num;++i)
            scanf("%d",&array[i]);
    sort(array,array+num);
    for(p=array[0];p<=maxsum;++p){
        min_petrol=INT_MAX;
        petrol[0]=0,i=0;
    while((p-array[i])>=0 && i<num)
    {
        min_petrol=min(petrol[p-array[i]]+1,min_petrol);
    ++i;
    }
    petrol[p]=min_petrol;
   }
   int count=0;
   for(i=0;i<num;++i)
   {
       count+=petrol[dist[i]];
   }
   printf("%d\n",count);
   }
   return 0;
}
