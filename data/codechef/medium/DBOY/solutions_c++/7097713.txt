#include <iostream>
#define min(a, b) a<b?a:b
using namespace std;
int mat[501][1001];
int main()
{
    int t, n, i, j, h[500], k[501], hmax, sum;
    cin>>t;
    while(t--)
    {
        cin>>n;
        hmax = 0;
        for(i=0; i<n; i++)
        {
            cin>>h[i];
            if(h[i]>hmax)
                hmax = h[i];
        }
        for(i=1; i<=n; i++)
        {
            cin>>k[i];
        }
        mat[0][0] = 0;
        for(i=1; i <= (2*hmax); i++)
            mat[0][i] = 100000;
        for(i=1; i<=n; i++)
        {
            mat[i][0] = 0;
            for(j=1; j <= (2*hmax); j++)
            {
                if(k[i] > j)
                {
                    mat[i][j] = mat[i-1][j];
                }
                else
                {
                    mat[i][j] = min(mat[i-1][j], (1+mat[i][j-k[i]]));
                }
            }
        }
        sum = 0;
        for(i=0; i<n; i++)
        {
            sum += mat[n][2*h[i]];
        }
        cout<<sum<<endl;
    }
    return 0;
}
