#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <math.h>
using namespace std;
#define M 1500
int h[M];
int k[M];
int dp[M][M];
int main()
{
    int big=1000000;
    int t;// test case
    scanf("%d",&t);
    while(t--){
        int n,hmax=0;
        scanf("%d",&n);// no of fuel stations
        for(int i=1;i<=n;i++){
         scanf("%d",&h[i]);
         hmax=max(hmax,h[i]);
         }
         for(int i=1;i<=n;i++){
         scanf("%d",&k[i]);
        }

      dp[0][0]=0;
      for(int i=1;i<=(2*hmax);i++){
        dp[0][i]=big;
      }

      for(int i=1;i<=n;i++)
      {
          for(int j=0;j<=(2*hmax);j++)
          {
          	dp[i][j]=dp[i-1][j];
              if(k[i]<=j){
                dp[i][j]=min(dp[i][j],1+dp[i][j-k[i]]);
              }
          }
      }
       int sum=0;
       for(int i=1;i<=n;i++){
        sum+=dp[n][2*h[i]];
       }
        printf("%d\n",sum);
    }


    return 0;
}
