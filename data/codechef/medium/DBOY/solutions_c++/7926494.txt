#include <iostream>
#include<bits/stdc++.h>
#define pb push_back
using namespace std;
vector< int > v;//vals
vector< int > v2;//orders
int dp[1010];
int main()
{
int t,n,val;
cin>>t;
while(t--)
{
    cin>>n;int sum=0;
    for(int i=0;i<=1010;i++)
        dp[i]=100000000;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&val);
        v2.pb(2*val);//:P
    }
    for(int i=0;i<n;i++)
    {
        scanf("%d",&val);
        v.pb(val);
        dp[val]=1;
    }// input
    sort(v.begin(),v.end());
    for(int i=1;i<=1010;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(i-v[j]>=0)
                dp[i] = min(dp[i],dp[i-v[j]]+1);
            else
                break;
        }
    }
  for(int i=0;i<n;i++)
  {
      sum+=dp[v2[i]];
  }
  printf("%d\n",sum);
  v2.clear();
  v.clear();
}
    return 0;
}
