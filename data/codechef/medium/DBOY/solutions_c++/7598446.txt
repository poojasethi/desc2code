#define INF 1000000007
#define min(a,b) a<b?a:b
int main()
{
int i,j,t,n,max,ans,dis;
scanf("%d",&t);
while(t--)
{

scanf("%d",&n);
max=0;
ans=0;
for(i=0;i<=n;i++)
h[i]=0;
for(i=1;i<=n;i++)
{scanf("%d",&h[i]);
//h[dis]=1;
if(h[i]>max)
max=h[i];
}
for(i=1;i<=n;i++)
scanf("%d",&k[i]);

max=max*2;
for(i=0;i<=n;i++)
for(j=0;j<=max;j++)
{
if(j==0)
dp_pf[i][j]=0;
else if(i==0)
dp_pf[i][j]=INF;
else if(k[i]>j)
dp_pf[i][j]=dp_pf[i-1][j];
else
dp_pf[i][j]=min(dp_pf[i-1][j],1+dp_pf[i][j-k[i]]);
}
//max=max>>1;
for(i=1;i<=n;i++)
{
ans+=dp_pf[n][h[i]<<1];
}
printf("%d\n",ans);

}
return 0;
}
