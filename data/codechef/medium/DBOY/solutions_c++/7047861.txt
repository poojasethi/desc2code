
#include <stdio.h>
#include <algorithm>
 
using namespace std;

int main(int argc , char** argv)
{
	int t,k,numbers[1024],denominations[1024],DP[1024];
	DP[0] = 0;
	scanf("%d" , &t);

	int min = 100000;
		
	while(t--)
	{
		for (int i = 0; i < 1024; ++i)
		{
			DP[i] = min;
		}
		DP[0] = 0;
		scanf("%d" , &k);
		for(int i=0; i<k;i++)
		{
			scanf("%d" , &numbers[i]);
		} 

		for(int i=0; i<k;i++)
		{
			scanf("%d" , &denominations[i]);
		} 
		
		sort(denominations , denominations+k);
		sort(numbers , numbers+k);

		for(int i=1;i<1024;i++)
		{
			
			for(int j=0;j<k;j++)
			{
				if(i - denominations[j] < 0)
					break;
				else if(DP[i - denominations[j]] + 1 < DP[i])
					{
						DP[i] = DP[i - denominations[j]] + 1;
						// cout<<i<<" "<<j<<" "<<min<<endl;
					}
			}

		}

		int totsum = 0;

		for(int i=0;i<k;i++)
		{
			totsum += DP[numbers[i]*2];
		}

/*		for(int i=0;i<k;i++)
		{
			cout<<numbers[i]<<" ";
		}
		cout<<endl;

		for(int i=0;i<k;i++)
		{
			cout<<denominations[i]<<" ";
		}
		cout<<endl;

		for(int i=0;i<1024;i++)
		{
			cout<<DP[i]<<" ";
		}
		cout<<"\n";
*/
		printf("%d\n" , totsum);

	}

	return 0;
}
