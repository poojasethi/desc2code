#include<iostream>
#include<cstdio>
#include<limits.h>
using namespace std;
long long dp[505][2005];
int main()
{
    long long t,n,h[505],k[505],i,j,max,res;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);
        max=-1;
        for(i=1;i<=n;i++)
        {
            scanf("%lld",&h[i]);
            if(h[i]>max)
                max=h[i];
        }
          for(i=1;i<=n;i++)
        {
            scanf("%lld",&k[i]);
        }
        dp[0][0]=0;
        for(i=1;i<=2*max;i++)
        {
            dp[0][i]=1000000000;
        }
        for(i=1;i<=n;i++)
        {  
            for(j=0;j<=2*max;j++)
            {
            if(k[i]<=j)
            dp[i][j]=min(dp[i-1][j],1+dp[i][j-k[i]]);
            else
            dp[i][j]=dp[i-1][j];
           // printf("dp[%d][%d]=%d ",i,j,dp[i][j]);
            }
            // printf("\n");
        }
        res=0;
        for(i=1;i<=n;i++)
        {
            res=res+dp[n][2*h[i]];
        }
        
        printf("%lld\n",res);
    }
    
    return 0;
}
