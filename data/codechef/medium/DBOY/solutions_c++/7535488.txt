#include <bits/stdc++.h>
 using namespace std;

 int h[600],k[600],dp[1200];
int main(){
	int t; cin >> t;
	while(t--){
		int n,maxh=-1,ans=0; cin >> n;
		for(int i=0 ; i<n ; i++) {cin >> h[i]; if(h[i]>maxh) maxh = h[i];}
		for(int i=0 ; i<n ; i++) cin >> k[i];
		for(int i=1 ; i<=2*maxh ; i++) dp[i] = 1000000000; dp[0] = 0;
		for(int i=0 ; i<n ; i++)
			for(int j=0 ; j+k[i]<=2*maxh ; j++)
				dp[j+k[i]] = min(dp[j+k[i]], 1+dp[j]);
		for(int i=0 ; i<n ; i++) ans += dp[2*h[i]];
		cout << ans << endl;
	}
	return 0;
}
