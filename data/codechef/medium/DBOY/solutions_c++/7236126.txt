#include<iostream>
using namespace std;
#include<cstdio>
#include<algorithm>
#define gc getchar_unlocked
void scanint(int &x)
{
	register int c=gc();
	x=0;
	for(;(c<48 || c>57);c=gc());
	for(;c>47 && c<58;c=gc()){x=(x<<1)+(x<<3)+c-48;}
}
main()
{
	int t,n,i,j,maxi;
scanint(t);
	while(t--)
	{	
		scanint(n);
		int h[n],k[n];
		for(i=0;i<n;i++)
			scanint(h[i]);	
		for(i=0;i<n;i++)
			scanint(k[i]);
		maxi=*std::max_element(h,h+n);
	//	cout<<maxi<<"\n";
		int x;
		x=(2*maxi)+1;
		int dp[x];
		dp[0]=0;
		for(i=1;i<=x;i++)
			dp[i]=100000;
		
		for(i=1;i<=x;i++)
		{
			for(j=0;j<n;j++)
			{
				if(k[j]<=i)
					dp[i]=min(dp[i],1+dp[i-k[j]]);
			}			
		}
		
		int sum=0;	
		for(i=0;i<n;i++)
			sum+=dp[2*h[i]];
		printf("%d\n",sum);
	}
}
