
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int

int main()
{
    int t;
    si(t);
    while(t--) {
	int n;
	si(n);
	ll h[n],k[n];
	int cnt[1001]={0};
	for(int i=0;i<n;++i) {
	    sll(h[i]);
	}
	for(int i=0;i<n;++i) {
	    sll(k[i]);
	    cnt[k[i]]++;
	}
	sort(k,k+n);
	int DP[1001];
	DP[0]=0;
	for(int i=1;i<=1000;++i) {
	    DP[i]=1000000;
	}
	for(int i=1;i<=1000;++i) {
	    for(int j=0;j<n;++j) {
		if(i==k[j]) {
		    DP[i]=1;
		}
		else {
		    if(i-k[j]>=0 && DP[i-k[j]]<1000000) {
			DP[i]=min(DP[i],DP[i-k[j]]+1);
		    }
		}
	    }
	}
	int ans = 0;
	for(int i=0;i<n;++i) {
	    ans += DP[2*h[i]];
	}
	pri(ans); nl;
    }
    return 0;
}
