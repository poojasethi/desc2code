#include <bits/stdc++.h>
#define MAX 505
using namespace std;
int h[MAX],k[MAX];
int dp[MAX][2*MAX];
int main(int argc, char const *argv[]) {
    int test,n,i,j,mx=0;
    scanf("%d",&test);
    while(test--)
    {
        mx=0;
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
            scanf("%d",&h[i]);
            mx=max(mx,h[i]);
        }
        for(i=1;i<=n;i++)
            scanf("%d",&k[i]);
        dp[0][0]=0;
        for(i=1;i<=2*mx;i++)
            dp[0][i]=1000000000;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<=2*mx;j++)
            {
                dp[i][j]=dp[i-1][j];
                if(k[i]<=j)
                    dp[i][j]=min(dp[i][j],1+dp[i][j-k[i]]);
            }
        }
        int sum=0;
        for(i=1;i<=n;i++)
        {
            sum+=dp[n][2*h[i]];
        }
        printf("%d\n",sum);
    }
    return 0;
}
