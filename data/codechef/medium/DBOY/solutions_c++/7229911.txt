#include<bits/stdc++.h>
using namespace std;
 #define ll long long
 int main()
 {
 	ll t;
 	scanf("%lld",&t);
 	while(t--)
 	{
 		ll n,j,i;
 		scanf("%lld",&n);
 		ll arr[n+1];
 		ll max=-1;
 		for(i=0;i<n;i++)
 		{
 			scanf("%lld",&arr[i]);
 			if(arr[i]>max)
 			max=arr[i];
 		}
 		max=max*2;
 		ll arrk[n+1];
 		for(i=0;i<n;i++)
 		{
 			scanf("%lld",&arrk[i]);
 		}
 		sort(arrk,arrk+n);
 		ll dp[n+1][max+1];
 			memset(dp,0,sizeof(dp[0][0])*(n+1)*(max+1));
 		for(i=0;i<max+1;i++)
 		{
 			dp[0][i]=100000009;
 		}
 		dp[0][0]=0;
 	
 		for(i=1;i<=n;i++)
 		{
 			for(j=1;j<max+1;j++)
 			{
 				dp[i][j]=dp[i-1][j];
 				if(arrk[i-1]<=j)
 				{
 					
 					dp[i][j]=min(dp[i][j],dp[i][j-arrk[i-1]]+1);
 				}
 				
 			}
 		}
     	/*for(i=1;i<=n;i++)
 		{
 			for(j=1;j<max+1;j++)
 			{
 				
 				{
 					printf("%lld ",dp[i][j]);
 				}
 				
 			}
 			printf("\n");
 		}*/
 		sort(arr,arr+n);
 		j=0;
 		ll sum=0;
 		for(i=1;i<=n;i++)
 		{
 			sum+=dp[n][2*arr[i-1]];
 			
 		}
 		printf("%lld\n",sum);
 	}
 	
	return 0; 
 }
