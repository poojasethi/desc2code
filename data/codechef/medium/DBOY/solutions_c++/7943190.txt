#include<cstdio>
#include<vector>
using namespace std;

vector<int> dp(1001);
void dpsum(int n,vector<int> &arr)
{
for(int i=1;i<=1000;i++) dp[i]=100000;
int temp;
dp[0]=0;
for(int i=1;i<=1000;i++)
 {
    for(int j=0;j<n;j++)
    {
        if(arr[j]<=i)
        {
         temp=i-arr[j];
         if(dp[i]>(dp[temp]+1))
            dp[i]=dp[temp]+1;
        }
    }
 }
}
int main()
{
    int t,n,sum,temp;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        vector<int> h(n);
        vector<int> k(n);
        for(int i=0;i<n;i++) scanf("%d",&h[i]);
        for(int i=0;i<n;i++) scanf("%d",&k[i]);
        dpsum(n,k);
        sum=0;
        for(int i=0;i<n;i++)
        {
            sum+=dp[2*h[i]];
        }
        printf("%d\n",sum);
    }
    return 0;
}
