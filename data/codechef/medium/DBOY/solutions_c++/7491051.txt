/**************************Code****************************/

int a[512], dp[1024];

int main()
{
	int t;
	scanf( "%d", &t );
	while( t -- )
	{
		int n, mx = 0, x;
		scanf( "%d", &n );
		for( int i = 0 ; i < n ; i ++ )
			scanf( "%d", a + i ), mx = max( mx, 2 * a[i] );
		fill( dp, dp+mx+1, 500*500*500+10 );
		dp[0] = 0;
		for( int i = 0 ; i < n ; i ++ )
		{
			scanf( "%d", &x );
			for( int j = 0 ; j +x <= mx ; j ++ )
				dp[j+x] = min( dp[j+x], dp[j] + 1 );
		}
		int ans = 0;
		for( int i = 0 ; i < n ; i ++ )
			ans += dp[ 2*a[i] ];
		printf( "%d\n", ans );
	}
	return 0;
}
