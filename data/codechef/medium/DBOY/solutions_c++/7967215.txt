#include<bits/stdc++.h>
#define INF 10000000
#define ll long long int
using namespace std;
ll dp[100010];
int b[500],a[500];
int n;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,x,mx=0;
        ll ans=0;
        cin>>n;
        for(int i=1; i<=n; i++)
        {
            cin>>x;
            a[i]=2*x;
            mx=max(mx,a[i]);
        }
        for(int i=1; i<=n; i++)
        {
            cin>>b[i];
        }
        for(int i=1; i<=mx; i++)
        {
            dp[i]=INF;
        }
        dp[0]=0;
        for(int i=1; i<=n; i++)
        {
            for(int j=b[i]; j<=mx; j++)
            {
                dp[j]=min(dp[j],1+dp[j-b[i]]);
            }
        }

        for(int i=1; i<=n; i++)
        {
            ans+=dp[a[i]];
        }
        cout<<ans<<endl;
    }
    return 0;
}
