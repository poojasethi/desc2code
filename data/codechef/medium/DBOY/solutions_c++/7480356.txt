#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,m=0,ans=0;
		cin>>n;
		int dis[n],lit[n];
		for(int i=0;i<n;i++)
		{
		cin>>dis[i];
		m=max(m,2*dis[i]);
		}
		for(int i=0;i<n;i++)
		cin>>lit[i];
		int dp[m+2];
		for(int i=0;i<=m;i++)
		dp[i]=99999999;
		dp[0]=0;
		for(int i=0;i<n;i++)
		{
			for(int j=lit[i];j<=m;j++)
			{
				dp[j]=min(dp[j],dp[j-lit[i]]+1);
			}
		}
		for(int i=0;i<n;i++)
		ans+=dp[2*dis[i]];
		cout<<ans<<endl;
	}
}
