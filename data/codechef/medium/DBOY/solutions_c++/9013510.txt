int main(){

    int t;
    scanf("%d",&t);
    while(t--){
        int n,H[502],K[502],dp[1004];
        scanf("%d",&n);
        rep(i,n) scanf("%d",&H[i]);
        rep(i,n) scanf("%d",&K[i]);


        int dist=0;
        rep(i,n){
            dist = max(dist,H[i]);
        }

        dp[0]=0;
        forup(i,1,2*dist)
            dp[i]=MAX;

        for(int i=0;i<n;i++){
            for(int j=K[i];j<=2*dist;j++){
                dp[j] = min(dp[j],1+dp[j-K[i]]);
                /*rep(i,2*dist+1)
                    printf("%d ",dp[i]);
                printf("\n");*/
            }

        }



        int ans=0;
        rep(i,n)
            ans+=dp[2*H[i]];

        printf("%d\n",ans);
    }
    return 0;
}
