#include <iostream>

using namespace std;

#define M 1001

int a[M], x[M], y[M];

int main()
{
	int tc;
	cin >> tc;
	while(tc--)
	{
		int n;
		cin >> n;
		for(int i = 0; i < n; ++i)
		{
			cin >> x[i];
		}
		for(int i = 0; i < n; ++i)
		{
			cin >> y[i];
		}
		// Initialization
		a[0] = 0;
		for(int i = 1; i < M; ++i)
		{
			a[i] = 99999999;
		}
		// Minimizing the fuel filling times
		for(int i = 1; i < M; ++i)
		{
			for(int j = 0; j < n; ++j)
			{
				if(y[j] <= i && a[i] > a[i - y[j]] + 1)
				{
					a[i] = a[i - y[j]] + 1;
				}
			}
		}
		int ans = 0;
		for(int i = 0; i < n; ++i)
		{
			int k = x[i];
			ans += a[2 * k];
		}
		cout << ans << endl;
	}
	return 0;
}
