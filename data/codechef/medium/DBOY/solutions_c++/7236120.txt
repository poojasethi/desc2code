#include<iostream>
using namespace std;
#include<cstdio>
#include<algorithm>
#define repi(x,y) for(i=x;i<y;i++)
#define repj(x,y) for(j=x;j<y;j++)
main()
{
	int t,n,i,j,maxi;
	scanf("%d",&t);
	while(t--)
	{	
		scanf("%d",&n);
		int h[n],k[n];
		for(i=0;i<n;i++)
			scanf("%d",&h[i]);	
		for(i=0;i<n;i++)
			scanf("%d",&k[i]);
		maxi=*std::max_element(h,h+n);
	//	cout<<maxi<<"\n";
		int x;
		x=(2*maxi)+1;
		int dp[x];
		dp[0]=0;
		for(i=1;i<=x;i++)
			dp[i]=100000;
		
		for(i=1;i<=x;i++)
		{
			for(j=0;j<n;j++)
			{
				if(k[j]<=i)
					dp[i]=min(dp[i],1+dp[i-k[j]]);
			}			
		}
		
		int sum=0;	
		for(i=0;i<n;i++)
			sum+=dp[2*h[i]];
		cout<<sum<<"\n";
	}
}
