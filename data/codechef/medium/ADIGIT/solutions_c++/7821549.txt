#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
int n,q,st,e,c;
const int SIZE = 1e5+1;
int dp[10][SIZE];
int arr[SIZE],B1,B2;
void init()
{
    memset(dp,0,sizeof dp);
for(int i=0;i<10;i++){
for(int j=0;j<n;j++){
dp[i][j]=dp[i][j-1]+(arr[j]==i);}}
}





int main()
{
ios_base::sync_with_stdio(false);
cin.tie(0);
string s;
cin>>n>>q;
cin>>s;
for(int i=0;i<n;i++)
   arr[i]=s[i]-'0';
init();
//query
while(q--)
{
    cin>>st;
    B1=B2=0;
    int dig=arr[--st];
    //for calculating B1
    for(int i=0;i<dig;i++)
    {
        B1+=dp[i][st-1]*(dig-i);
    }
    //for calculating B2
    for(int i=dig+1;i<10;++i)
    {
        B2+=dp[i][st-1]*(dig-i);
    }
    //cout<<B1<<" "<<B2<<endl;
    int ans=B1-B2;
    cout<<ans<<endl;
}
return 0;
}
