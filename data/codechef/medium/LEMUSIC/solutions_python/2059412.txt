import operator
t = input()
for i in range(t):
	table = {}
	remaining_table = []
	swtns1 = 0
	swtns2 = 0
	swtns = 0
	count = 1
	n = input()
	for j in range(n):
		b,l = map(int, raw_input().split())
		if b in table:
			if l < table[b]:
				remaining_table.append(table[b])
				table[b] = l
			else:
				remaining_table.append(l)
		else:
			table[b] = l
	
	for j in sorted(table.iteritems(), key = operator.itemgetter(1)):
		swtns1 = swtns1 + j[1]*count
		count = count + 1
	
	count = count - 1
	for j in remaining_table:
		swtns2 = swtns2 + j*count
	
	swtns = swtns1 + swtns2
	print swtns
