#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
#define ll long long int
struct pairs
{
    ll x,y;
};
int compare(pairs a,pairs b)
{
    if( a.x< b.x ) return 1;
    else if( a.x> b.x ) return 0;
    if( a.y< b.y ) return 1;
    return 0;
}
int compare1(pairs a,pairs b)
{
    if( a.y < b.y ) return 1;
    return 0;
}
pairs a[100005],b[100005],c[100005];
int main()
{
    unsigned i,t,n,j,k;
    ll ans,cur;
    scanf("%d",&t);
    while(t--)
    {
        ans=cur=0;
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
            scanf("%lld%lld",&a[i].x,&a[i].y);
        }
        sort(&a[1],&a[1]+n,compare);
        j=2;k=1;
        b[1]=a[1];
        for(i=2;i<=n;i++)
        {
            if(a[i].x==b[j-1].x)
            {
                c[k++]=a[i];
            }
            else
            {
                b[j++]=a[i];
            }
        }
        sort(&b[1],&b[1]+j-1,compare1);
        for(i=1;i<j;i++)
        {
            cur++;
            ans+=(cur*b[i].y);
        }
        for(i=1;i<k;i++)
        {
            ans+=(cur*c[i].y);
        }
        printf("%lld\n",ans);
    }

    return 0;
}
