ll gcd(ll a,ll b) {if(a%b==0) return b;else return gcd(b,a%b);}
#define gc getchar_unlocked
inline int getn(){
	int n = 0, c = gc();
	while(c < '0' || c > '9') c = gc();
	while(c >= '0' && c <= '9')
		n = (n<<3) + (n<<1) + c - '0', c = gc();
	return n;
}
int main()
{
    int t=getn();
    while(t--)
    {
        int n=getn();
        //cin>>n;
        vector<pair<int,int> > A(n);
        for(int i=0;i<n;i++)
        {
            //cin>>A[i].first>>A[i].second;
            A[i].first=getn();
            A[i].second=getn();
        }
        sort(A.begin(),A.end());
        vector<ll> V;
        ll rest=0;
        for(int i=0;i<n;i++)
        {
            if(i==0 or A[i].first>A[i-1].first) V.push_back(A[i].second);
            else rest+=A[i].second;
        }
        //for(int i=0;i<V.size();i++) cout<<V[i]<<endl;
        ll tr=0;
        sort(V.begin(),V.end());
        for(int i=0;i<V.size();i++) tr+=(i+1)*V[i];
        tr+= (rest*V.size());
        cout<<tr<<endl;
    }
}
