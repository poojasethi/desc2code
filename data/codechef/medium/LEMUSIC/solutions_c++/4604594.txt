#include <iostream>
#include <map>
#include <algorithm>
using namespace std;
#define ll long long int
struct pairs
{
    ll x,y;
};
int compare(pairs a,pairs b)
{
    if( a.y< b.y ) return 1;
    return 0;
}
pairs a[100005],b[100005];
int main()
{
    unsigned i,t,n,k;
    ll ans,cur;

    map<ll , int> mp;
    cin>>t;
    while(t--)
    {
        ans=cur=0;
        cin>>n;
        for(i=1;i<=n;i++)
        {
            cin>>a[i].x;
            cin>>a[i].y;
        }
        sort(&a[1],&a[1]+n,compare);
        k=1;
        for(i=1;i<=n;i++)
        {
            if(mp[a[i].x]!=1)
            {
                cur++;
                ans+=(cur*a[i].y);
                mp[a[i].x]=1;
            }
            else
            {
                b[k++]=a[i];
            }
        }
        for(i=1;i<k;i++)
        {
            ans+=(cur*b[i].y);
        }
        cout<<ans<<'\n';
        mp.clear();
    }

    return 0;
}
