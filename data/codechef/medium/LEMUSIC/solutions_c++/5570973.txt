#include<iostream>
#include<vector>
#include<algorithm>
#define ll long long int
using namespace std;
int main()
{
    ll tc,i,n;
    ll sum,t;
    cin>>tc;
while(tc--)
{
    int n;
    cin>>n;
    vector<pair<ll,ll> >a(n);
    vector<ll>b;
    for(i=0;i<n;i++)
    {
        cin>>a[i].first;
        cin>>a[i].second;
    }
    sort(a.begin(),a.end());

    sum=0;
    for(i=0;i<n;i++)
    {
      if(i==0 || a[i-1].first<a[i].first)
            b.push_back(a[i].second);
        else
        sum=sum+a[i].second;
    }
    sort(b.begin(),b.end());
    t=0;
    for(i=0;i<b.size();i++)
        t=t+((i+1)*b[i]);
     sum=sum*b.size();
     cout<<sum+t<<"\n";

}
return 0;
}
