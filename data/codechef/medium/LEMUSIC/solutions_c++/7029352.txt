#include<stdio.h>
#include<utility>
#include<algorithm>
#include<iostream>
using namespace std;
#define max 100005
typedef long long ll;
#define gc getchar_unlocked
void scanint(int &x)
{
	register int c=gc();
	x=0;
	for(;(c<48 || c>57);c=gc());
	for(;c>47 && c<58;c=gc()){x=(x<<1)+(x<<3)+c-48;}
}
main()
{
	pair<int,int> p[max];
	int a[max];
	int n,i,j,b,t,l;
	scanint(t);
	while(t--)
	{
		scanint(n);
		for(i=0;i<n;i++)
		{
			scanint(b);
			scanint(l);
			p[i].first=b;
			p[i].second=l;
		}
		sort(p,p+n);
		int present=-1;
		ll sum=0;
		int k=0;
		for(i=0;i<n;i++)
		{
			if(present!=p[i].first)
				a[k++]=p[i].second;
			else
				sum=sum+p[i].second;
			present=p[i].first;
			
		}
		sort(a,a+k);
		ll ans=0;
		ans=sum*(k);
		for(i=0;i<k;i++)
			ans=ans+((i+1)*(ll)a[i]);
		printf("%lld\n",ans);
	}
}
