#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
using namespace std;
#define ll long long int
struct pairs
{
    ll x,y;
};
int compare(pairs a,pairs b)
{
    if( a.y< b.y ) return 1;
    return 0;
}
int main()
{
    unsigned i,t,n;
    ll ans,cur;
    vector<pairs> a;
    vector<pairs> b;
    map<ll , int> mp;
    pairs p;
    cin>>t;
    while(t--)
    {
        ans=cur=0;
        cin>>n;
        for(i=1;i<=n;i++)
        {
            cin>>p.x;
            cin>>p.y;
            a.push_back(p);
        }
        sort(a.begin(),a.end(),compare);
        for(i=0;i<a.size();i++)
        {
            if(mp[a[i].x]!=1)
            {
                cur++;
                ans+=(cur*a[i].y);
                mp[a[i].x]=1;
            }
            else
            {
                b.push_back(a[i]);
            }
        }
        for(i=0;i<b.size();i++)
        {
            ans+=(cur*b[i].y);
        }
        cout<<ans<<'\n';
        a.clear();b.clear();mp.clear();
    }

    return 0;
}
