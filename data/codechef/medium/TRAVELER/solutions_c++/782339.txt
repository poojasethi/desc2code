const int MN=50+10;
int dis[MN][MN];
bool mark[MN];
map<string,int> mp;
int cnt,n,m,t,k;
string list[MN];
int tono(const string & a)
{
	if (mp.find(a)!=mp.end())
		return mp[a];
	return mp[a]=cnt++;
}

int main(int argc,char * argv[])
{
	ios_base::sync_with_stdio(false);
	cin>>n;
	for (int i=0;i<n;i++)
	{
		string _;
		cin>>_;
		tono(_);
	}
	cin>>m;
	for (int i=0;i<m;i++)
	{
		int a,b;
		int c;
		string _,__;
		cin>>_>>__>>c; a=tono(_); b=tono(__);
		dis[a][b]=c;
	}
	cin>>t;
	while (t-->0)
	{
		int res=0;
		cin>>k;
		int last=-1;
		memset(mark,0,sizeof(mark));
		for (int i=0;i<k;i++)
			cin>>list[i];
		for (int i=0;i<k;i++)
		{
			string _;
			_=list[i];
			int v;
			if (mp.find(_)==mp.end())
			{
				cout<<"ERROR"<<endl;
				goto nex;
			}
			v=tono(_);
			if (mark[v] || (i>0 && dis[last][v]==0))
			{
				cout<<"ERROR"<<endl;
				goto nex;
			}
			mark[v]=1;
			if (i)
				res+=dis[last][v];
			last=v;
		}
		cout<<res<<endl;
nex:;
	}
	return 0;
}
