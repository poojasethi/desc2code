#include<iostream>
#include<map>
#include<cstdio>
#define max -100
using namespace std;
map <string,int> road;
int main()
{
	int n,m;
	scanf("%d",&n);
	char string[30];
	int rmap[52][52];
	int i,j;
	for(i=0;i<52;i++)
		for(j=0;j<52;j++)
			rmap[i][j]=max;
	for(i=0;i<n;i++)
	{
		scanf("%s",string);
		road[string]=i+1;
	}
	scanf("%d",&m);
	char string1[30];
	int val;
	for(i=0;i<m;i++)
	{
		scanf("%s%s%d",string,string1,&val);
		int d1=road.find(string)->second;
		int d2=road.find(string1)->second;
		rmap[d1][d2]=val;
	}
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&val);
		int array[val];
		int flag=1;
		int hash[52];
		for(i=0;i<52;i++)
			hash[i]=0;
		for(i=0;i<val;i++)
		{
			scanf("%s",string);
			array[i]=road.find(string)->second;
			if(array[i]==0)
				flag=0;
			hash[array[i]]++;
			if(hash[array[i]]>1)
				flag=0;
		}
	//	printf("flag=%d\n",flag);
		if(flag==0)
		{
			printf("ERROR\n");
			continue;
		}
		int sum=0;
		for(i=0;i<val-1;i++)
		{
		//	cout<<rmap[array[i]][array[i+1]]<<endl;
		//	printf("%d %d\n",array[i],array[i+1]);
			if(rmap[array[i]][array[i+1]]==max)
				break;
			else
				sum=sum+rmap[array[i]][array[i+1]];
		}
		if(i!=(val-1))
			printf("ERROR\n");
		else
			printf("%d\n",sum);
	}
	return 0;
}
