#include <iostream>
#include <bitset>
using namespace std;
int a[20];
int N,K,M;
bool check(int b[])
{
    for(int i=0;i<N-K+1;i++)
    {
        int max=-1;
        int count=0;
        for(int j=i;j<i+K;j++)
        {
            if(b[j] > max)
            {
                max=b[j];
            }
        }
        for(int j=i;j<i+K;j++)
        {
            if(b[j] == max)
            {
                ++count;
            }
        }
        if(count>=M)
        {
            return false;
        }
    }
    return true;
}
int main() {
	int T;
	cin>>T;
	while(T--)
	{
	    cin>>N>>K>>M;
	    for(int i=0;i<N;i++)
	    {
	        cin>>a[i];
	    }
	    int ans = N+5;
	    for(int i=0;i<(1<<N);i++)
	    {
	        int b[20];
	        bitset<20> set(i);
	        if(set.count() >= ans)
	        {
	            continue;
	        }
	        for(int j=0;j<N;j++)
	        {
	            if(set[j])
	            {
	                b[j]=a[j]+1;
	            }
	            else
	            {
	                b[j]=a[j];
	            }
	        }
	        if(check(b))
	        {
	            ans = set.count();
	        }
	    }
	    if(ans == N+5)
	    {
	        ans = -1;
	    }
	    cout<<ans<<endl;
	}
	return 0;
}
