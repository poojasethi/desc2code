#include <bits/stdc++.h>

using namespace std;

#define pb push_back

string change(string s, int &cnt)
{
	for (int i = s.size()-1; i >= 0; i--) {
		if (s[i] == '1') {
			s[i] = '0';
			cnt--;
		}
		else {
			s[i] = '1';
			cnt++;
			break;
		}
	}
	return s;
}

int cntmax(int b[], int l, int r)
{
	int mx = INT_MIN, cnt, i;
	for (i = l; i <= r; i++) {
		if (b[i] > mx)
			cnt = 0;
		mx = max(mx, b[i]);
		if (mx == b[i])
			cnt++;
	}
	return cnt;
}

int check(int b[], int n, int k, int m)
{
	int mx = INT_MIN, i;
	for (i = 0; i <= n-k; i++) {
		if (cntmax(b, i, i+k-1) >= m)
			return 0;
	}
	return 1;
}

int main()
{
	int t, i, n, m, k, p, a[100], b[100], cnt, j, mn;
	string s;
	scanf("%d", &t);
	while (t--) {
		scanf("%d%d%d", &n, &k, &m);
		s = "";
		p = 1;
		for (i = 0; i < n; i++) {
			scanf("%d", &a[i]);
			s += "0";
			p *= 2;
		}
		cnt = 0;
		mn = INT_MAX;
		for (i = 0; i < p; i++) {
		//	cout << cnt << endl;
			for (j = 0; j < n; j++)
				b[j] = a[j] + s[j] - '0';
			if (check(b, n, k, m) == 1)
				mn = mn > cnt ? cnt : mn;
	//			for (j = 0; mn == 1 && j < n; j++)
	//				cout << b[j] << " ";
	//			cout << endl;
	//		}
			s = change(s, cnt);
		}
		if (mn == INT_MAX)
			mn = -1;
		printf("%d\n", mn);
	}
	return 0;
}
