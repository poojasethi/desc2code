#include<bits/stdc++.h>
using namespace std;

int R[20];
int cnt[20];
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		int N, K, M;
		scanf("%d%d%d", &N, &K, &M);
		int maxe = 0;
		for(int i = 0;i < N;++i){
			scanf("%d", &R[i]);
			maxe = max(maxe, R[i] + 1);
		}
		int ans = 100;
		for(int mask = 0;mask < (1 << N);++mask){
			int curAns = 0;
			for(int i = 0;i < N;++i){
				if(mask & (1 << i)){
					R[i] += 1;
					curAns += 1;
				}
			}
			for(int i = 0;i <= maxe;++i){
				cnt[i] = 0;
			}
			for(int i = 0;i < K;++i){
				cnt[R[i]] += 1;
			}
			bool yes = true;
			int chk = maxe;
			for(int i = K;i <= N;++i){
				while(chk > 0 && cnt[chk] == 0)
					--chk;
				// cout << chk << "\n";
				if(cnt[chk] >= M){
					yes = false;
					break;
				}
				chk = max(chk, R[i]);
				cnt[R[i]] += 1;
				cnt[R[i - K]] -= 1;
			}
			// cout << mask << " " << yes << " " << curAns << "\n";
			if(yes) ans = min(ans, curAns);
			for(int i = 0;i < N;++i){
				if(mask & (1 << i))
					R[i] -= 1;
			}
		}
		if(ans == 100) ans = -1;
		printf("%d\n", ans);
	}
	return 0;
}
