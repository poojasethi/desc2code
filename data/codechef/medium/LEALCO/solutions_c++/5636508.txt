#include<bits/stdc++.h>
using namespace std;
vector<int> dp[20];
int main()
{
    int i,k,d=pow(2,17),j,c,flag,max;
    for(i=0; i<d; i++)
    {
        k=i;
        j=0;
        while(k>0)
        {
            if(k%2==1)
                j++;
            k=k/2;
        }
        dp[j].push_back(i);
    }
    //for(i=0;i<dp[1].size();i++)
    //    printf("%d ",dp[1][i]);
    int t,n,m,a[100],it,temp[101],p;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d",&n,&k,&m);
        for(i=0; i<n; i++)
            scanf("%d",&a[i]);
        d=pow(2,n);
        for(i=0; i<n; i++)
        {
            for(it=0; it<dp[i].size(); it++)
            {
                flag=0;
                if(dp[i][it]>=d)
                    break;
                //printf("%d\n",dp[i][it]);
                for(j=0; j<n; j++)
                    temp[j]=a[j];
                p=dp[i][it];
                j=0;
                while(p>0)
                {
                    if(p%2==1)
                        temp[j]++;
                    j++;
                    p=p/2;
                }
                //for(j=0;j<n;j++)
                //    printf("%d ",temp[j]);
                //printf("\n");
                c=0;
                for(j=0;j<=n-k;j++)
                {
                    c=0;
                    max=-1;
                    for(p=0;p<k;p++)
                        if(temp[j+p]>max)
                    {
                        max=temp[j+p];
                        c=1;
                    }
                    else if(temp[j+p]==max)
                        c++;
                    //printf("%d\n",c);
                    if(c>=m)
                        break;
                }

                if(j==n-k+1)
                {
                    flag=1;
                    break;
                }
                if(flag==1)
                    break;
            }
            if(flag==1)
                    break;
        }
        if(flag==1)
            printf("%d\n",i);
        else
            printf("-1\n");
    }
    return 0;
}
