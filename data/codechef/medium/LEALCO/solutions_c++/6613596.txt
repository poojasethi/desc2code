#include <stdio.h>
#include <math.h>
#include <iostream>
#include <algorithm>
using namespace std;
int fin,k,n,m;
int R[19];
 
void solve(int i,int num){
	if(fin<num) return;
	if(i==n){
		if(fin>num) fin=num;
		return;
	}
	//case1:
	int maxval=-1;int ct=0;
	if(i>=k-1) for(int j=max(0,i-k+1);j<=i;j++) {if(R[j]>maxval) {ct=1;maxval=R[j];} else if(R[j]==maxval) ct++;}
	if(ct>=m) {} 
	else solve(i+1,num);	
	//case2:
	R[i]++;
	maxval=-1;
	ct=0;
	if(i>=k-1) for(int j=max(0,i-k+1);j<=i;j++) {if(R[j]>maxval) {ct=1;maxval=R[j];} else if(R[j]==maxval) ct++;}
	if(ct<m) solve(i+1,num+1);
	R[i]--;
	return;
}
 
int main(){
int t;
scanf("%d", &t);
	while(t--){
		scanf("%d %d %d",&n,&k,&m);
		for(int i=0;i<n;i++) scanf("%d",&R[i]);
		fin=n+1;
		solve(0,0);
		if(fin==n+1) fin=-1;
		printf("%d\n",fin);
	}
}
