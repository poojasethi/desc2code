#include <bits/stdc++.h>
using namespace std;
#define bitcnt __builtin_popcount

int Arr[20];
int Temp[20];
int n, m, k;

bool check(int *Temp)
{
	for(int i = 0; i <= n - k; i++) {
		int cnt = 1;
		int maxx = Temp[i];
		for(int j = i + 1; j < i + k; j++) {
			if(maxx == Temp[j]) {
				cnt++;
			}
			else if(Temp[j] > maxx) {
				maxx = Temp[j];
				cnt = 1;
			}
		}
		if(cnt >= m) {
			return false;
		}
	}
	return true;	
}

void solve()
{
	scanf("%d%d%d", &n, &k, &m);
	for(int i = 0; i < n; i++) {
		scanf("%d", &Arr[i]);
	}
	if(m == 1) {
		puts("-1");
		return;
	}
	int best = -1;
	for(int i = 0; i < (1 << n); i++) {
		for(int j = 0; j < n; j++) {
			if(i & (1 << j)) {
				Temp[j] = Arr[j] + 1; 
			}
			else {
				Temp[j] = Arr[j];
			}
		}
		if(check(Temp)) {
			if(best == -1) best = bitcnt(i);
			else if(bitcnt(i) < best) best = bitcnt(i);
		}
	}
	printf("%d\n", best);
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		solve();
	}
	return 0;
}
