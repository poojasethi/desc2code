#define pb push_back

#define ms1(x) memset(x,-1,sizeof(x));

vector<int> r,nr;

bool sayess(int n,int m,int k){
    for(int i=0;i<=n-k;i++){
        int mx=-1,cntmx=0;
        for(int j=i;j<i+k;j++){
            if(nr[j]==mx) cntmx++;

            if(nr[j]>mx){
                mx= nr[j];
                cntmx=1;
            }
        }
        //printf("%d\n",cntmx);
        if(cntmx>=m) return false;
    }
    return true;
}

int bitcount(int x){
    int res=0;
    while(x>0){
        x-=x&(-x);
        res++;
    }
    return res;
}

int main(){
    int t;
    cin >> t;
    while(t--){
        int n,m,k;
        cin >> n >> k >> m;
        for(int i=0;i<n;i++){
            int x;
            scanf("%d",&x);
            r.pb(x);
        }

        int mn= 1000;

        for(int i=0;i<(1<<n);i++){
            int btcnt= bitcount(i);

            if(mn<=btcnt) continue;

            for(int j=0;j<n;j++){
                if(i & 1<<j) nr.pb(r[j]+1);
                else nr.pb(r[j]);
            }

            if(sayess(n,m,k)){
                mn=btcnt;
            }
            nr.clear();
        }

        if(mn==1000) mn=-1;

        printf("%d\n",mn);

        r.clear();
    }
}
