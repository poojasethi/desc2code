#include <iostream>
using namespace std;

bool isCorrect(int *A, int sz, int K, int M) {
    int i,j,k,maxi, cnt;
    for (i=0;i<=(sz-K);++i) { //no. of ranges
        maxi=A[i]; cnt=0;
        for (j=i; j<(K+i); ++j) {
            maxi=(A[j]>maxi?A[j]:maxi);
            //cout << "> maxi= " << maxi << endl;
        }
        //cout << "maxi for i= " << i << "=" << maxi << endl;
        for (j=i; j<(K+i); ++j) {
            if (A[j]==maxi) ++cnt;
        //    cout << "cnt++" << endl;
        }
        //cout << "cnt=" << cnt << endl;
        if (cnt>=M) return false;
    }
    return true;
}

void print(int *A, int sz) {
    for (int i=0; i<sz; ++i) {
        cout << A[i] << "\t";
    }
    cout << endl;
}
