#include <iostream>
#include "bits/stdc++.h"
using namespace std;
int a[50];
int res;
int n,m,k;
void backtracking(int i, int num)
{
	int add,j;
	
	if (res <= num) {
		return;
	}
 
	
	if (i == n) {
 
	
		if(num<res){
		res = num; }
		return;
	}
 
	
	for ( add = 0; add < 2; add++) {
 
	
		a[i] += add;
 
		
		int cnt = 0;
 
		
		if (i >= k - 1) {
 
			
			int maxR = 0;
			for ( j = i; j > i - k; j--) {
				if(maxR == a[j]) {
					++cnt;
				} else if (maxR < a[j]) {
					maxR = a[j];
					cnt = 1;
				}
			}
		}
 
		
		if (cnt < m) {
			backtracking(i + 1, num + add);
		}
 
		
		a[i] -= add;
	}
	return ;
}
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
		
		scanf("%d",&n);
		 scanf("%d",&k);
		  scanf("%d",&m);
		    
	   for(int i=0;i<n;i++)
	    scanf("%d",&a[i]);
	     res=n+1;
	     backtracking(0,0);
	     if(res==n+1)
	     printf("-1\n");
	     else
	      printf("%d\n",res);
	}
	return 0;
}
