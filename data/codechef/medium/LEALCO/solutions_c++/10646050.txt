template<class T>
void scanint(T &x)
{
    register T c = gc();
    x = 0;
    T neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}


inline bool lt(double a, double b)
{
    return a + EPS < b;
}
inline bool gt(double a, double b)
{
    return a > b + EPS;
}
inline bool eq(double a, double b)
{
    return fabs(a - b) < EPS;
}

ll power(ll base,ll p)
{
    ll temp=1;
    while(p>1)
    {
        if(p%2!=0)
            temp=(1LL*(base*temp)%MOD);
        base=(1LL*(base*base)%MOD);
        p/=2;
    }
    ll ans=(1LL*(base*temp)%MOD);
    return ans;
}

bool check(vint R,int k,int m)
{
	int n=R.size();
	int i,j;
	ff(i,0,n-k)
	{
		int maxi=-1;
		ff(j,i,i+k-1)
		{
			maxi=max(maxi,R[j]);
		}
		int count=0;
		ff(j,i,i+k-1)
		{
			if(R[j]==maxi)
				count++;
		}

		if(count>=m)
			return false;
	}
	return true;
}

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
    	int n,k,m,i,j;
    	cin>>n>>k>>m;
    	vint R(n);
    	ff(i,0,n-1)
    	{
    		cin>>R[i];
    	}
     	int ans=INFI;
     	if(check(R,k,m))
     	{
     		ans=0;
     	}
    	ff(i,1,(1<<n)-1)
    	{
    		int size=__builtin_popcount(i);
    		vint R1(n);
    		ff(j,0,n-1)
    		{
    			R1[j]=R[j];
    			if(i & (1<<j))
    				R1[j]++;
    		}
    		if(check(R1,k,m))
    		{
    			ans=min(size,ans);
    		}
    	}
    	if(ans==INFI)
    		ans=-1;
    	cout<<ans<<endl;

    }
    return 0;
}
