int main()
{   
    ios::sync_with_stdio(false);
    ll t;
    cin>>t;
    while(t--){
        ll n,k,m,i;
        ll ans=INT_MAX;
        cin>>n>>k>>m;
        ll a[n];
        for(i=0;i<n;i++)
            cin>>a[i];
        for(i=0;i<(1<<n);i++){
            ll b[n];
            ll cnt=0;
            for(ll j=0;j<n;j++){
                b[j]=a[j];
              //  cout<<b[j]<<" ";
                if (i & 1<<j){
                    b[j]++;
                    cnt++;
                }
            }
           // cout<<endl;
            if (cnt>=ans) continue;
           //cout<<cnt<<endl;
            int flag=1;
            for(ll j=0;j<n-k+1;j++){
                ll count=0;
                ll maxe=INT_MIN;
                for(int l=j;l<j+k;l++){
                 //  cout<<b[i]<<" ";
                    maxe=max(maxe,b[l]);
                }
                //cout<<endl;
                for(int l=j;l<j+k;l++)
                    if (b[l]==maxe) count++;
            //    cout<<maxe<<" "<<count<<endl;
                if (count>=m) {flag=0;break;}
            }
          //  cout<<cnt<<endl;
            if (flag) ans=min(ans,cnt);

        }
        if (ans>=INT_MAX) cout<<"-1"<<endl;
        else cout<<ans<<endl;
    }
    return 0;
}
