
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int

int check(int A[], int n, int m, int k) {
    for(int i=0;i<=n-k;++i) {
	int freq[40]={0},mx=-1;
	for(int j=i;j<=i+k-1;++j) {
	    freq[A[j]]++;
	    mx=max(mx,A[j]);
	}
	if(freq[mx]>=m) 
	    return 0;
    }
    return 1;
}

int main()
{
    int t;
    si(t);
    while(t--) {
	int n,k,m;
	si(n); si(k); si(m);
	int a[n];
	for(int i=0;i<n;++i) {
	    si(a[i]);
	}
	int ans=100000000;
	for(int i=0;i<(1<<n);++i) {
	    int A[n],mn=0;
	    for(int j=0;j<n;++j) A[j]=a[j];
	    for(int j=0;j<n;++j) {
		if(i&(1<<j)) {
		    mn++;
		    A[j]++;
		}
	    }
		if(check(A,n,m,k)) {
		    ans=min(ans,mn);
		}
	}
	if(ans==100000000) {
	    pri(-1);
	}
	else {
	    pri(ans); 
	}
	nl;
    }
    return 0;
}
