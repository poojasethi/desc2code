int main()
{
int i;
int t;
sint(t);
while(t--)
{
   int n,k,m,num,flag,found=0,c,j,l;
   cin>>n>>k>>m;
   int h[n];
   int main_ans=50;
   FOR(i,0,n-1)
   {
       cin>>h[i];
   }
   FOR(i,0,((1<<n)-1))
   {
       FOR(j,0,n-1)
       {
           if((1<<j)&i)
           h[j]++;
       }
       //check
       flag=1;
       FOR(j,0,n-k)
       {
           //cout<<"hey ";
           num=-1;
           c=0;
           FOR(l,0,k-1)
           {
               num=max(num,h[j+l]);
           }
           FOR(l,0,k-1)
           {
               if(h[j+l]==num)
               c++;
           }
         //  cout<<i<<" "<<j<<" "<<c<<" "<<num<<"\n";
           if(c>=m)
           {
            //   cout<<i<<" "<<j<<" "<<c<<" "<<num<<"\n";
               flag=0;
               break;
           }

       }
       if(flag)
       {
           int ans=0;
           int use=i;
           while(use>0)
           {
               if(use&1)
               {
                   ans++;
               }
               use=use>>1;
           }
           main_ans=min(main_ans,ans);
       }
        FOR(j,0,n-1)
       {
           if((1<<j)&i)
           h[j]--;
       }
   }
   if(main_ans==50)
   main_ans=-1;
    cout<<main_ans<<"\n";
}
return 0;
}
