#include <iostream>
#include "bits/stdc++.h"
using namespace std;
int a[50];
int ans;
int n,m,k;
void solve(int pos,int num)
{
	if(num>=ans) return;
	
	if(pos==n) {
		
		ans=num; return;}
	for(int j=0;j<2;j++){
		a[pos]+=j;
		  int f=0;
	if(pos>=k-1)
	{ 
		  
		int maxa=-1;
		  for(int i=pos;i>=pos-k+1;i--)
		  {
		  	if(maxa<a[i]) maxa=a[i];
		  }
		  for(int i=pos;i>=pos-k+1;i--)
		  {
		  	if(maxa==a[i]) f++;
		  }
		  
	}
	     if(f<m) solve(pos+1,num+j);
	       a[pos]-=j;
	}
	  return;
}
int main() {
	int t;
	scanf("%d",&t);
	while(t--)
	{
		
		scanf("%d",&n);
		 scanf("%d",&k);
		  scanf("%d",&m);
		    
	   for(int i=0;i<n;i++)
	    scanf("%d",&a[i]);
	     ans=n+1;
	     solve(0,0);
	     if(ans==n+1)
	     printf("-1\n");
	     else
	      printf("%d\n",ans);
	}
	return 0;
}
