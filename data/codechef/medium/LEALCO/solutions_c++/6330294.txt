#include<iostream>
#include<limits.h>
#include<math.h>
int checkbad(int newarr[],int n, int k, int m);
using namespace std;
int main()
{
    int t;
    cin>>t;
    int m,k,n,i;
    while(t--)
    {
        cin>>n>>k>>m;

        int arr[n];
        int newarr[n];

        for(i=0;i<n;i++)
            cin>>arr[i];

        long long int mask=0;

        long long int minimum=INT_MAX;

        for(mask=0;mask<pow(2,n);mask++)
         {
             for(i=0;i<n;i++)
                newarr[i]=arr[i];

            int changes=0;

            for(i=0;i<=n-1;i++)
        {
            if (mask&(1<<i))
                {newarr[i]=newarr[i]+1;
                 changes++;}

        }

            int c=checkbad(newarr,n,k,m);

            if (c==1&&(minimum>changes))
            minimum=changes;


         }

        if (minimum!=INT_MAX)
         cout<<minimum<<'\n';
         else
            cout<<-1<<'\n';
    }
    return 0;
}


int checkbad(int newarr[],int n, int k, int m)
{
    int i;


    for(i=0;i<=n-k;i++)
    {
        int countmax=0;
        int j;
        int maximum=-1;
        for(j=i;j<k+i;j++)
            if (newarr[j]>maximum)
            maximum=newarr[j];

        for(j=i;j<k+i;j++)
        {
            if (newarr[j]==maximum)
                countmax++;
        }

        if (countmax>=m)
            return 0;
    }

    return 1;
}
