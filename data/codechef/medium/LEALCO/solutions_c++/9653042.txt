#include<bits/stdc++.h>
 
using namespace std;
 
int T, n, k, m;
int v[32];
 
int main() {
 scanf("%d",&T);
    while(T--)
	 {
       scanf("%d %d %d",&n,&k,&m);
       for (int i=0;i<n;i++)
	    scanf("%d",v+i);
       int best = 1000;
       if (m>1)
       for (int bm=0;bm<(1<<n);bm++)
	    {
           int cand = __builtin_popcount(bm);
           if (cand >= best) continue;
           bool ok = true;
           for (int i=0;i+k<=n;i++) 
		   {
               int maior = -1, cmaior = 0;
               for (int j=i;j<i+k;j++) 
			   {
                   int t = v[j] + ((bm&(1<<j))?1:0);
                   if (t > maior) { maior = t; cmaior = 1;}
                   else if (t == maior) cmaior++;
               }
               if (cmaior >= m)
			    { 
				   ok = false;
				 break;
				 }
           }
           if (ok) best = cand;
       }
 
       printf("%d\n",best==1000?-1:best);
    }
 
    return 0;
}
