const int MOD = 1e9+7;

int n, k, m, a[MAX], b[MAX];

bool isbad()
{
	rep(i, 0, n-k+1)
	{
		int maxi = 0, c = 0;
		rep(j, i, i+k)
			maxi = max(maxi, b[j]);
		rep(j, i, i+k)
			if(b[j] == maxi)
				c++;
		if(c >= m) return 1;
	}
	return 0;
}

					b[i]++, cnt++;
			}
