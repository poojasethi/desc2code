#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    int tc;
    ll n,m;
    scanf("%d",&tc);
    while(tc--)
    {
        scanf("%lld %lld",&n,&m);
        if(n==1 && m==1)
        {
            printf("No\n");
            continue;
        }
        if(n==1 || m==1)
        {
            if(n==2 || m==2)
                printf("Yes\n");
            else
                printf("No\n");
            continue;
        }
        if(n%2==1 && m%2==1)
            printf("No\n");
        else
            printf("Yes\n");
    }
}
