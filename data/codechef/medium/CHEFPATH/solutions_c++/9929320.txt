#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    long long t, n, m;
    cin >> t;
    while (t--)
    {
        cin >> n >> m;
        long long mn, mx;
        mn = min (n, m);
        mx = max (n, m);
        if ((mn == 1 && mx == 2 ) || (mn >= 2 && (!(n % 2) || !(m % 2))))
           cout << "Yes" << endl;
        else
            cout << "No" << endl;
    }
	return 0;
}
