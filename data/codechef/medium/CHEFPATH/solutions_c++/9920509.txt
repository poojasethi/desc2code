#include <iostream>

using namespace std;

int main()
{
    int test;
    cin>>test;
    while(test--)
    {
        long long a,b;
        cin>>a>>b;
        if(a==1 || b==1)
        {
            int s=a-b;
            s=s<0?-s:s;
            if(s==1)
                cout<<"Yes"<<endl;
            else
                cout<<"No"<<endl;
        }
        else{
            if(a%2==0 || b%2==0)
                cout<<"Yes"<<endl;
            else
                cout<<"No"<<endl;
        }
    }
    return 0;
}
