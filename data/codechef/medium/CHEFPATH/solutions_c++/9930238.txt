#include<bits/stdc++.h>
using namespace std;
int main(){
	long long int t;
	cin>>t;
	while(t--){
		long long int n,m;
		cin>>n>>m;
		if(n==1){
			if(m==2)
			printf("Yes\n");
			else printf("No\n");
		}
		else if(m==1){
				if(n==2)
			printf("Yes\n");
			else printf("No\n");
		}
		else if(m%2==0 || n%2==0){
			printf("Yes\n");
		}
		else printf("No\n");
		
	}
	return 0;
}
