#include<iostream>
using namespace std;
main()
{
    long long int n,t,m;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        if(n==1)
         {
             if(m==2)
                cout<<"Yes\n";
             else
                cout<<"No\n";
         }
         else if(m==1)
         {
             if(n==2)
                cout<<"Yes\n";
             else
                cout<<"No\n";
         }
        else if((n%2!=0) && (m%2!=0))
             cout<<"No\n";
        else
             cout<<"Yes\n";
    }
    return 0;
}
