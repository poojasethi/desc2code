#include <iostream>
using namespace std;

int main() {
    
    long long int t;
    long long int n,m;
    cin>>t;
    while(t--){
        
        cin>>n>>m;
        if(n==1&&m==1)
        cout<<"No"<<endl;
        else if((n==1&&m==2)||(m==1&&n==2))
        cout<<"Yes"<<endl;
        else if((n%2==0&&m!=1)||(m%2==0&&n!=1))
        cout<<"Yes"<<endl;
        else
        cout<<"No"<<endl;
    }
    
	return 0;
}
