#include<iostream>
#include<cstdio>
using namespace std;
 
 
int main(){
	
	int t;
	scanf("%d",&t);
 
	long long int n,m;
	while(t--){
		scanf("%lld %lld",&n,&m);

		if((n==1 && m==2) || (n==2 && m==1)){
			printf("Yes\n");
		}
		else if(n==1 || m==1){
			printf("No\n");
		}
		else if( (n%2 == 0) || (m%2 == 0)){
			printf("Yes\n");
		}else{
			printf("No\n");

		}
 
 
	}
 
}
