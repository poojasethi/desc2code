#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    long long int n,m;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld %lld",&n,&m);
        if((n==1 && m>2 )||(m==1 && n>2))
        {
            cout<<"No"<<"\n";
        }
        else
        if(n%2==1 && m%2==1)
        {
            cout<<"No"<<"\n";
        }
        else
            cout<<"Yes"<<"\n";
    }
    return 0;
}
