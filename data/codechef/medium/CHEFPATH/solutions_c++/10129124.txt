#include <iostream>
using namespace std;
#define ll long long

int main()
{   ll t;
    cin>>t;
    while(t--)
    {   ll a,b;
        cin>>a>>b;
        if((a==1 && b!=2) || (b==1 && a!=2))
        {   cout<<"No\n";
            continue;
        }
        if(a%2 && b%2)
        {   cout<<"No\n";
            continue;
        }
        cout<<"Yes\n";
    }
    return 0;
}
