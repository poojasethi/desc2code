#include <iostream>
#define ll long long int
using namespace std;

ll min(ll a,ll b)
{
     return (a<b)?a:b;
}

ll max(ll a,ll b)
{
    return (a>b)?a:b;
}

int main()
{
    int t;
    cin>>t;
    
    while (t--)
     {
          ll n,m;
          cin>>n>>m;
          
          if (min(n,m)==1)
          {
               if (max(n,m)==2)
                cout<<"Yes"<<endl;
               else
                cout<<"No"<<endl;
          }
          
          else if (n%2==1 && m%2==1)
            cout<<"No"<<endl;
            
          else
            cout<<"Yes"<<endl;
     }
     
     return 0;
}
