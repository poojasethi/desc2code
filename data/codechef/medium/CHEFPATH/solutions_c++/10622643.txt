 #include<stdio.h>
unsigned long long int scan()
{
	register int x=0;
	char c=getchar_unlocked();
	for(;c>=48&&c<=57;c=getchar_unlocked())
	x=(x<<1)+(x<<3)+c-48;
	return x;
}
