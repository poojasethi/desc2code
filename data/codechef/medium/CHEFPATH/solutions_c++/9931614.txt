#include <iostream>
#include <stdio.h>
#include <cmath>
using namespace std;

int main ()
{
	int T;
	scanf ("%d\n", &T);
	for (int t=1; t<=T; t++)
	{
		long long r,c;
		const char *res;
		scanf("%lld %lld", &r, &c);
		if (min(r,c)==1)
		{
			if (max(r,c)==2)
				res = "Yes";
			else
				res = "No";
		}
		else
			res=(r%2==1 && c%2==1)?"No":"Yes";
		cout<<res<<endl;
	}
	return 0;
}
