#include <iostream>
#include <stdio.h>
#include <stdlib.h>
using namespace std;
 
int main() {
	long long int n,m,t;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&n,&m);
		if((n==1&&m==2)||(m==1&&n==2))
            printf("Yes\n");
        else if(n==1||m==1)
            printf("No\n");
		else if(n%2==0||m%2==0)
			printf("Yes\n");
		else
            printf("No\n");
	}
	return 0;
}
