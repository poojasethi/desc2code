#include <iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
     int t;
     cin>>t;
     long long int n,m;
     while(t--){
        cin>>n;
        cin>>m;

     if(n==1 ||m==1){
      if(abs(n-m)==1)cout<<"Yes"<<"\n";
      else  cout<<"No"<<"\n";
     }
   else if(n%2!=0 && m%2!=0)
            cout<<"No"<<"\n";
        else
            cout<<"Yes"<<"\n";
}
    return 0;
}
