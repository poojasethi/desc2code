#include <iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        unsigned long long int n, m;
        cin>>n>>m;

        if( (n==1 && m==2) || (n==2 && m==1) ) {
            cout<<"Yes"<<endl;
        }
        else {
            if(n == 1 || m == 1) {
                cout<<"No"<<endl;
            }
            else {
                if(n%2 == 0 || m%2 == 0)
                    cout<<"Yes"<<endl;
                else
                    cout<<"No"<<endl;
            }
        }


    }
    return 0;
}
