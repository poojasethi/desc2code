#include<cstdio>
#include<iostream>
using namespace std;
int t;
long long int n,m;
int main(){
    scanf("%d",&t);
    for(int i=1;i<=t;i++){
        scanf("%lld %lld",&n,&m);
        if (n==1 and m==1) printf("No\n");
        else if ((n==1 or m==1) and n+m>3) printf("No\n");
        else if(n%2==1 and m%2==1) printf("No\n");
        else printf("Yes\n");
        }
return 0;
}
