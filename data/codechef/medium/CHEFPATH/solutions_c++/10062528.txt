#include <cstdlib>
#include <iostream>
#include <stdio.h>
using namespace std;

/*
 * 
 */
int main() {
    
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++)
    {
        long long int n,m;
        scanf("%lld%lld",&n,&m);
        if(n==1||m==1)
        {
            if(n==2||m==2)
                cout<<"Yes\n";
            else
                cout<<"No\n";
        }else
        {
            if(n%2==0||m%2==0)
                cout<<"Yes\n";
            else
                cout<<"No\n";
        }
        
    }
    

    return 0;
}
