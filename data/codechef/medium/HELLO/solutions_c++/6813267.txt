#include <iostream>
using namespace std;
typedef long long LL;
int main() {
    LL t=0,i=0;
    cin>>t;
    while(t--) {
        double d,r,totAmt = 0;
        
        LL u=0,n=0,m=0,c=0;
        
        cin>>d>>u>>n;
        totAmt = (d*u);
        
        LL bestPlan = 0;
        double minAmt = totAmt;
        for(i=1;i<=n;i++) {
            
            cin>>m>>r>>c;
            double newAmt = (((double)(c)/(double)(m)) + r*(double)(u));
            if(newAmt < minAmt) {
                minAmt = newAmt;
                bestPlan = i;
            }
        }
        
        cout<<bestPlan<<endl;
    }
    return 0;
}
