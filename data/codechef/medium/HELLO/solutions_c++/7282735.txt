#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main(int argc, char const *argv[])
{
	int t, u, n, m, c;
	double d, r, cost;
	cin >> t;
	while(t--) {
		cin >> d >> u >> n;
		cost = d * u;
		int ans = -1;
		for(int i = 0; i < n; i++) {
			cin >> m >> r >> c;
			if(cost > (double)c / m + r * u) {
				ans = i;
				cost = (double)c / m + r * u;
			}
		}

		cout << ans + 1 << endl;
	}
	return 0;
}
