#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int T,n,u,m,c,res;
double d,r;
int main()
{
  	scanf("%d",&T);
	while(T--)
	{
		cin>>d>>u>>n;
		double	cur,pre,base=d*u;
		pre=base;
		res=0;
		for(int i=1;i<=n;i++)
		{
			cin>>m>>r>>c;
			cur=u*r+(double)c/m;
			if(cur<pre)
			{
				pre=cur;
				res=i;
			}
		}
		cout<<res<<"\n";
	}
	return 0;
}
