#include <iostream>
using namespace std;

int main() {
	int tc;
	int u, n, m, c;
	int ans;
	double e, _e;
	double d, r;
	
	int i, j;
	
	cin >> tc;
	
	while (tc--) {
		cin >> d >> u >> n;
		e = d * u;
		ans = 0;
		for (i = 1; i <= n; i++) {
			cin >> m >> r >> c;
			_e = ((double)c / (double)m) + (r * u);
			if (_e < e) {
				e = _e;
				ans = i;
			}
		}
//		cout << last_cost << " - " << endl;
		cout << ans << endl;
	}
	
	
	return 0;
}
