#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
     int n;
     double d,u;
     cin>>d>>u>>n;
     double  x1,x2;
     x1=d*u;
     double m,c;
     double r;
     int k=0;
     for(int i=1;i<=n;i++)
     {
         cin>>m>>r>>c;
         x2= r*u+(c/(double)m);
       if( x2<x1)
       {
           x1=x2;
           k=i;
       }
     }
     cout<<k<<"\n";
    }
    return 0;
}
