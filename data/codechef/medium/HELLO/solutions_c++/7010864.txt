#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,i;
    double t,im,ir,init,ans,p;
    cin>>t;
    while(t--)
    {
        cin>>ir>>im>>n;
        p=0;
        double arr[n][3],maxi=1;
        for(i=0;i<n;i++)
        {
            cin>>arr[i][0]>>arr[i][1]>>arr[i][2];
            if(arr[i][0]>maxi)
                maxi=arr[i][0];
        }
        init=ir*im*maxi;
        for(i=0;i<n;i++)
        {
            ans=((maxi/arr[i][0])*arr[i][2])+arr[i][1]*im*maxi;
            if(ans<init)
            {
                init=ans;
                p=i+1;
            }
        }
        cout<<p<<endl;
    }
    return 0;
}
