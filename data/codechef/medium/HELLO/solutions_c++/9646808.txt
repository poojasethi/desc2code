#include<bits/stdc++.h>
using namespace std;
long long M[200];
double D[200];
long long cost[200];
int main(){
	long long t;
	cin >> t;
	while(t--){
		int N;
		long long Minute;
		cin >> D[0] >> Minute >> N;
		M[0] = 1;
		cost[0] = 0;
		long long gc = 1;
		for(int i = 1;i <=  N;++i){
			cin >> M[i] >> D[i] >> cost[i];
			gc = (gc*M[i])/__gcd(M[i], gc);
		}
		long long id = 0;
		double minCost = Minute*D[0]*gc;
		for(long long i = 0;i <= N;++i){
			double newCost = (D[i] * Minute * M[i] + cost[i])*(gc/M[i]);
			// cout << newCost << "\n";
			if(newCost < minCost) {
				minCost = newCost;
				id = i;
			}
		}
		cout << id << "\n";
	}
	return 0;
}
