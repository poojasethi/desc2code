#include <iostream>
using namespace std;
 
int main() {
	int t;
	cin>>t;
	while(t--){
		double d,u,n;
		cin>>d>>u>>n;
		int i;
		double mincost=d*u,mc=0;
		double r,m,c;
		for(i=0;i<n;i++){
			cin>>m>>r>>c;
			double cm=c/m;
			double cost=(r*u)+cm;
			if(cost<mincost){
				mincost=cost;
				mc=i+1;
			}
		}
		cout<<mc<<"\n";
	}
	return 0;
}
