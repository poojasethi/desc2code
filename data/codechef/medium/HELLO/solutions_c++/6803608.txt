#include<stdio.h>
int main()
{
	long long int u;
	int mon[100],t,n,i,n1,j;
	double d,ra[100],val[100],to1,to,to2;
	scanf("%d",&t);
	while(t--)
	{	
		n1=0,j=0,to1=0,to=0,to2=0;
		scanf("%lf%lld%d",&d,&u,&n);
		for(i=0;i<n;i++)
		{
			scanf("%d%lf%lf",&mon[i],&ra[i],&val[i]);
			if(n1<mon[i])
			{
				n1=mon[i];
			}	
		}
		for(i=0;i<n;i++)
		{
			val[i]=(val[i]*n1)/mon[i];
	
		}
		to1=d*u;
		to1=to1*n1;	
		to2=to1;	
		for(i=0;i<n;i++)
		{
			to=(ra[i]*u*n1)+val[i];
			if(to2>to)
			{
				to2=to;
				j=i+1;
			}
		}
		printf("%d\n",j);
	}
return 0;	
}
