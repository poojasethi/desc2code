#include <stdio.h>

int main()
{
 int t, i, n, m, m1, p;
 long int u, c;
 float r, d, tc, min;
 scanf("%d", &t);
 while(t--)
      {
       scanf("%f %ld %d", &d, &u, &n);
       tc = d*u;
       min = tc;
       m1=1;
       p = 0;
       for(i=1; i<=n; i++)
          {
           scanf("%d %f %ld", &m, &r, &c);
           tc = m*u*r + c;
           if(tc*m1 < min*m)
             {
              min = tc;
              m1=m;
              p=i;
             }
          }
       printf("%d\n", p);
      }
 return 0;
}
