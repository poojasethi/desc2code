#include<bits/stdc++.h>
using namespace std;
int main(void)
{
    int t;
    for(scanf("%d", &t); t--; )
    {
        float d;
        int u;
        int n;
        int plan=0;
        scanf("%f %d %d", &d, &u, &n);
        float mini = d*u;
        //cout<<d<<" "<<u<<" "<<mini<<endl;
        for(int i=0; i<n; i++){
            int m;
            float r, c;
            scanf("%d %f %f", &m, &r, &c);
            c/=m;
            float temp = c + (r*u);
            //cout<<temp<<endl;
            if(temp<mini){
                mini = temp;
                plan = i+1;
            }
        }
        printf("%d\n", plan);
    }
    return 0;
}
