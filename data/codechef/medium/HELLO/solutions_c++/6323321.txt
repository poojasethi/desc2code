#include<stdio.h>
int main()
{
    int t,u,n,i,max;
    int m[100],c[100],k;
    double d,r[100],def,cost,min;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lf%d%d",&d,&u,&n);
        max=0;
        for(i=0;i<n;i++)
            scanf("%d%lf%d",&m[i],&r[i],&c[i]);
        for(i=0;i<n;i++)
            if(max<m[i])
                max=m[i];
        def=max*d*u;
        min=def;
        for(i=0;i<n;i++)
        {
            cost=max*r[i]*u;
            cost+=(c[i]*max)/m[i];
            if(min>cost)
            {
                min=cost;
                k=i+1;
            }
        }
        if(min==def)
            printf("0\n");
        else
            printf("%d\n",k);
    }
    return 0;
}
