#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
   int T, U, N, i, ans;
   int M, C;
    double R, D, cost, mincost;
    scanf("%d",&T);
    while(T--)
    {
      scanf("%lf%d%d",&D,&U,&N);
      mincost=D*U;
      for(i=0;i<N;i++)
       {
          cin>>M>>R>>C;
          cost=U*R+C/(1.0*M);
          if(mincost > cost)
           {
           mincost=cost;
             ans=i+1;
            }
        }
 
       printf("%d\n",ans);
     }
 
    return 0;
}
