#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf( "%d", &t );
	while( t -- )
	{
		int n, m, c, u, ind = 0;
		double d, r, mn;
		scanf( "%lf%d%d", &d, &u, &n );
		mn = d * u;
		for( int i = 1 ; i <= n ; i ++ )
		{
			scanf( "%d%lf%d", &m, &r, &c );
			r = c * 1. / m + r * u;
			if( r < mn )
				mn = r, ind = i;
		}
		printf( "%d\n", ind );
	}
	return 0;
}
