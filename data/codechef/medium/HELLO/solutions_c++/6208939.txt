#include<cstdio>
#include<iostream>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#include<cmath>

using namespace std;

int main()
{
    int t=0;
    scanf("%d",&t);



    while(t--)
    {
        double D=0.0,pre_cost=0.0,R=0.0,post_cost=0.0;
        int U=0,N=0,best_plan=0,M=0,C=0;

        scanf("%lf %d %d",&D,&U,&N);

        pre_cost=D*U;


        for(int i=1;i<=N;i++)
        {
            scanf("%d %lf %d",&M,&R,&C);
            post_cost=C/(double)M +U*R;
            if(pre_cost>post_cost)
            {
                best_plan=i;
                pre_cost=post_cost;
            }

        }

        printf("%d\n",best_plan);


    }


return 0;
}
