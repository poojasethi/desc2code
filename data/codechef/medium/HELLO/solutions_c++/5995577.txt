#include <iostream>
#include <cstdio>
#include <cstdlib>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    int U,N,M,C;
    double D,R;
    double res,tmp;
    int index;
    while(t--)
    {
      index = 0;
      scanf("%lf %d %d",&D,&U,&N);
      res = D*U;
      for(int i=0;i<N;i++)
      {
        scanf("%d %lf %d",&M,&R,&C);
        tmp = ((double)C/M + R*U);
        if(tmp < res)
        {
            res = tmp;
            index = i+1;
        }
      }
      printf("%d\n",index);
    }
    return 0;
}
