#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,u,n,m,c;
    float r,d;
    cin>>t;
    while(t--)
    {
        int f=0,i;
        float x,y;
        cin>>d>>u>>n;
        x=d*u;
        for(int i=0;i<n;i++)
        {
            cin>>m>>r>>c;
            y=(c+(m*r*u))/m;
            if(y<x)
            {
                x=y;
                f=i+1;
            }
        }
        cout<<f<<"\n";
    }
    return 0;
}
