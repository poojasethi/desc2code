#include<stdio.h>
#include<math.h>
int main()
{
	long long int u,t,n,i,count=0,final;
	float d,r,temp,old,c,m;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%f%lld%lld",&d,&u,&n);
		old=d*u;
		count=0;
		for(i=1;i<=n;i++)
		{
			scanf("%f%f%f",&m,&r,&c);
			temp=r*u+c/m;
			final=temp-old+1;
			if(final<=0)
			{
				count=i;
				old=temp;
			}
		}
		printf("%lld\n",count);
	}
	return 0;
}
