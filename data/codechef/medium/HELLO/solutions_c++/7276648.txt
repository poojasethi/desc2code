#include<stdio.h>
int main()
{ int t;
   scanf("%d",&t);
   while(t--)
   {  double d,r,cost,minm;
      int u,n,m,c,i,pos;
      scanf("%lf%d%d",&d,&u,&n);
      minm = d*u;
      pos=0;  i=1;
      while(n--)
      { scanf("%d%lf%d",&m,&r,&c);
         cost = (m*r*u+c)/m;
          if(cost<minm)
           { minm = cost;
              pos = i;
           }
           i++;
       }
      printf("%d\n",pos);
   }
   return 0;
}
