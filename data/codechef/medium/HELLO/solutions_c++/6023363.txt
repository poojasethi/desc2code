#include<stdio.h>
#include<algorithm>
int main()
{
    int t,n;
    float d,c,u,m,r;
    scanf("%d",&t);
    do
    {
        scanf("%f%f%d",&d,&u,&n);
        float a[n+1],res;
        a[0]=d*u;
        for(int i=1;i<=n;i++)
        {
            scanf("%f%f%f",&m,&r,&c);
            a[i]=(r*u)+(c/m);
        }
        res=*std::min_element(a,a+n+1);
        for(int i=0;i<=n;i++)
        {
            if(res==a[i]){printf("%d\n",i);break;}
        }
    }
    while(--t);
    return 0;
}
