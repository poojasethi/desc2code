#include<cstdio>
#include<cfloat>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		float D,R,original_rate,d,min=FLT_MAX;
		int U,N,M,C,s,i=1;
		scanf("%f %d %d",&D,&U,&N);
		original_rate=D*U;
		while(N--)
		{
			scanf("%d %f %d",&M,&R,&C);
			d=(float)C/M+R*U;
			if(d<min)
			{
			  min=d;
			  s=i;
		    }
		    i++;
		}
		if(min>original_rate)
		printf("0\n");
		else
		printf("%d\n",s);
	}
	return 0;
}
