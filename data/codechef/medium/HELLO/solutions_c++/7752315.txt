#include<bits/stdc++.h>
using namespace std;
 
int main()
{
	int n,t,count,i;
	float m,u,r,d,c,cm;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%f%f%d",&d,&u,&n);
		count=0;
		cm=d*u;
		for(i=1;i<=n;i++)
		{
			scanf("%f%f%f",&m,&r,&c);
			if((r*u+(c/m))<(cm))
			{
				cm=(r*u+(c/m));
				count=i;
			}
		}
		if(count)
		printf("%d\n",count);
		else
		printf("0\n");
	}
	return 0;
}
