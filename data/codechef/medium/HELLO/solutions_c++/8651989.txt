#include <bits/stdc++.h>
int main()
{
	int t,i,u,n,m,c,index;
	float d,r,net,def,min;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%f%d%d",&d,&u,&n);
		 def=(float)u*d;
		 min=def;
		 index=0;
		for(i=0;i<n;i++)
		{
		   scanf("%d%f%d",&m,&r,&c);
		   net=(float)u*r +(float)c/m;
		   if(min>net)
		   {
		   	min=net;
		   	index=i+1;
		   }
	    }
	    printf("%d\n",index);
	}
	return 0;
}
