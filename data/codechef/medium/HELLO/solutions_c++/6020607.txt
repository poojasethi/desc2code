 #include<iostream>
using namespace std;
#include<limits.h>

int main()
{
 int t;
 cin>> t;

 while(t--)
 { float d; int u , n ;

   cin>> d >> u >> n;

   float charge = d*u;  int flag =0;
   int temp;
   int i ,m,c; float r;
   float ans ;
   for(i=0 ; i<n ; i++)
   { cin>> m >> r >> c;
     ans = m*r*u; //cout<<"ans1 is : " << ans<< endl;;
     ans+= c; //cout<<"ans2 is : " << ans<<endl;
     ans/= m; // cout<<"ans3 is : " << ans <<endl;
     if(ans<charge) {charge = ans; flag=1; temp = i+1;}


   }

   if(flag==0) cout<<"0"<<endl;;
   if(flag==1) cout<<temp<<endl;;



 }}
