#include <iostream>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t-->0)
    {
        float r;
        int u;
        int n;
        cin>>r>>u>>n;
        float min = r*u;
        int minIndex = 0;
        // cout<<"test: "<<t<<endl;
        for(int i=1;i<=n;i++)
        {
            // cout<<"\tPlan:"<<i<<endl;
            int m;
            int c;
            cin>>m>>r>>c;
            float cpm = u*r + (float)c/m;
            // cout<<"\t\tcpm:"<<cpm<<endl;
            if( cpm < min)
            {
                min = cpm;
                minIndex = i;
            }
        }
        cout<<minIndex<<endl;
    }
}
