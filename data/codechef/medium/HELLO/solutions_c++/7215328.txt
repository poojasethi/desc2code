#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{	
	int t,n,i,minind; float d,u,m,r,c,mini,tmp;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%f %f %d",&d,&u,&n);
		mini=u*d;minind=0;
		for(i=0;i<n;i++)
		{
			scanf("%f %f %f",&m,&r,&c);
			tmp=(c/m)+u*r;
//			cout<<tmp<<" "<<mini<<endl;
			if(tmp<mini)
			mini=tmp,minind=i+1;
		}
		printf("%d\n",minind);
	}
	return 0;
}
