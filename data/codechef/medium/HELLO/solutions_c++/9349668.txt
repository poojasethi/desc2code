#include <iostream>
using namespace std;
int main()
{
    int t, u, n, o, i;
    double d;
    int minC, minM, curC, curM;
    double minR, curR;

    cin >> t;

    while (t--)
    {
        cin >> d >> u >> n;
        minC = 0;
        minR = d;
        minM = 1;
        o = 0;
        for (i = 1; i <= n; i++)
        {
            cin >> curM >> curR >> curC;
            if ((curM*curR*u + curC)*minM < (minM*minR*u + minC)*curM)
            {
                o = i;
                minC = curC;
                minR = curR;
                minM = curM;
            }
        }
        cout << o << '\n';
    }
}
