#include<stdio.h>
main()
{

int T;
scanf("%d",&T);
while(T-->0)
{
    float d;
    int u;
    scanf("%f%d",&d,&u);
    float initial=u*d;
    float plan=9999999;
    float cost;
int     i=0;
    int j=0;
    int n;
    scanf("%d",&n);
    while(n-->0)
    {
        j++;
        float m,c;
        float r;
        scanf("%f%f%f",&m,&r,&c);
        cost=r*u + c/m;
        if(cost<plan)
        {
            plan=cost;
            i=j;
        }

    }
    if(plan<initial)
    printf("%d\n",i);
    else
        printf("0\n");
}
}
