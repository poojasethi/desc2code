#include<iostream>
int main()
{ long long int t,n,i,max;
 float d,r,e,e_,u,m,c;
 scanf("%lld",&t);
 while(t--)
 {scanf("%f %f %lld",&d,&u,&n);
  max=0;
  e=d*u;


  for(i=1;i<=n;i++)
  { 
   scanf("%f %f %f",&m,&r,&c);
   e_=r*u+(c/m);
   if(e_<e)
   {max=i;
    e=e_;
   }
  }
  printf("%lld\n",max);
 }
 return 0;
}
