#include<iostream>
#include<stdio.h>
#define gc getchar_unlocked // for linux
//#define gc getchar        // for windows
#define LL long long
using namespace std;

void scanint(int &x)  // Unsigned int data type
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	int t,u,n,ans,i,m,c;
	float dcost, pcost,d,r;
	scanint(t);
	while(t--)
	{
		ans=0;
		scanf("%f %d %d", &d, &u, &n);
		dcost = d*u;
//		printf("dcost: %f\n", dcost);
		for(i=0;i<n;i++)
		{
			scanf("%d %f %d", &m , &r, &c);
			pcost=m*r*u+c;
//			printf("pcost: %f\n", pcost);
			pcost=pcost/m;
//			printf("pcost: %f\n", pcost);
			if(pcost<dcost)
			{
				dcost=pcost;
				ans=i+1;
			}
		}
		printf("%d\n", ans);
	}
	return 0;
}
