#include<iostream>
#include<string.h>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	
	double  i,j,n,t,d,u,m,r,c,ic,fc;
	scanf("%lf",&t);
	while(t--)
	{
		scanf("%lf",&d);
		scanf("%lf",&u);
		scanf("%lf",&n);
		ic=d*u;int ans=0;
		for(i=1;i<=n;i++)
		{
			scanf("%lf",&m);
			scanf("%lf",&r);
			scanf("%lf",&c);
			fc=(c/m)+r*u;
			//printf("%f--%f\n",ic,fc);
			if(fc<ic)
			{
			  ans=i;
			  ic=fc;
			}
			
		}

		printf("%d\n",ans);
		
	}
	return 0;
}
