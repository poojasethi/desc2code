#include <iostream>
using namespace std;

int main() {
	int t,u,n,i,choice,m,c;
	float d,norm,diff,r,temp,prof,t2;
	scanf("%d",&t);
	while(t--)
	{
		choice=0;
		prof=0;
		scanf("%f %d %d",&d,&u,&n);
		for(i=0;i<n;i++)
		{
			scanf("%d %f %d",&m,&r,&c);
			norm=(d*u);
			temp=(u*r)+(c/(float)m);
			t2=norm-temp;
			if(t2>prof)
			{
				choice=i+1;
				prof=t2;
			}
		}
		printf("%d\n",choice);
	}
	return 0;
}
