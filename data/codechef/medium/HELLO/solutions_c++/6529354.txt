#include <iostream>
#include <stdio.h>
using namespace std;

int smaller(float A[], int t)
{
    int x=0, y=0;
    while(y<t)
    {
        if(A[y] > A[x])
            x=y;
        y++;
    }
    return x;
}

int main()
{
    int t, N, i; // test, number of plans, cost of plan
    float D, R,C, U,M,A[100]; // default rate, rates after,//minutes, months

    scanf("%d", &t);
    while(t>0)
    {
        cin>>D;
        cin>>U;
        cin>>N;
        for (int j=0; j<N; j++)
        {
            cin>>M;
            cin>>R;
            cin>>C;
            A[j] = ((U*D)-(C/M+U*R));
        }
        i = smaller(A, N);
    if (A[i]>0)
        printf("%d\n", i+1);
    else
        printf("%d", 0);

        t--;
    }

}
