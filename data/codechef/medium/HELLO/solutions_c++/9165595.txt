#include<iostream>
#include<string.h>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	
	float i,j,n,t,d,u,m,r,c,ic,fc;
	scanf("%f",&t);
	while(t--)
	{
		scanf("%f",&d);
		scanf("%f",&u);
		scanf("%f",&n);
		ic=d*u;int ans=0;
		for(i=1;i<=n;i++)
		{
			scanf("%f",&m);
			scanf("%f",&r);
			scanf("%f",&c);
			fc=(c/m)+r*u;
			//printf("%f--%f\n",ic,fc);
			if(fc<ic){ans=i;ic=fc;
			}
			
		}

		printf("%d\n",ans);
		
	}
	return 0;
}
