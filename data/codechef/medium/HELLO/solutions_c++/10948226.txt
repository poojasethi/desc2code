int main(){
	int i,j,k,t,n,count;
	float month,charge,temp;
	float rate;
	float d,u;
	cin>>t;
	for(i=0;i<t;i++){
		cin>>d>>u>>n;
		temp = d*u;
		//struct plans plan[n];
		count = 0;
		for(j=0;j<n;j++){
			cin>>month>>rate>>charge;
			if(temp > (rate*u + (charge/month))){
				count=j+1;
				temp = rate*u + (charge/month);
			}
		}
		cout<<count<<"\n";
	}
	return 0;
}
