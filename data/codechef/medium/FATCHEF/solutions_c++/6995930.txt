#include<stdio.h>
#define V 1000000009
int main()
{
   int T,N,M,i,j,previousColor;
   long long int ans;
   char ch,str[100005];
   scanf("%d",&T);
   while(T--)
   {
      scanf("%d %d",&N,&M);
      for(i=0;i<N;i++)
         str[i]=0;
      for(i=0;i<M;i++)
      {
         scanf(" %c %d",&ch,&j);
         str[j-1]=ch;
      }
      previousColor=-1;
      for(ans=1,i=0;i<N;i++)
      {
         if(!str[i])
            continue;
         if(previousColor!=-1&&str[i]!=str[previousColor])
            ans=ans*(i-previousColor)%V;
         previousColor=i;
      }
      printf("%lld\n",ans);
   }
return 0;
}
