#include <bits/stdc++.h>
#define MOD 1000000009LL
using namespace std;
typedef long long int LL;
vector<pair<int,char> > v;
int main() {
	int T, pos, n, m;
	char s[5];
	scanf("%d", &T);
	while( T-- ) {
		scanf("%d %d", &n, &m);
		v.clear();
		for(int i=0 ; i<m ; i++) {
			scanf("%s %d", s, &pos);
			v.push_back(make_pair(pos, s[0]));
		}
		sort(v.begin(), v.end());
		LL ans = 1LL;
		for(int i=0 ; i<m-1 ; i++) {
			if( v[i].second != v[i + 1].second ) {
				ans = (ans * (LL)(v[i + 1].first - v[i].first) ) % MOD; 
			}
		}
		printf("%lld\n", ans);
	}
	return 0;
}
