#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;

#define PRIME 1000000009
class node
{
public:
	char c;
	long long int i;
	node(char a,long long int b)
	{
		c=a;
		i=b;
	}
	bool operator<(const node& b) const
	{
		return i<b.i;
	}
};
int main()
{
	int t;
	scanf("%d\n",&t);
	while(t--)
	{
		int n,m;
		
		scanf("%d %d\n",&n,&m);	
		
		vector<node> v;
		char c;
		int x;
		for(int i=0;i<m;i++)
		{
			scanf("%c %d\n",&c,&x);
			
			//cout<<c<<" "<<x<<endl;
			v.push_back(node(c,x));
			// marker.push_back(0);
		}
		sort(v.begin(),v.end());
		long long int res = 1;

		for(int i=1;i<m;i++)
		{
			if(v[i].c !=v[i-1].c)
			{
				res*= (v[i].i - v[i-1].i);
				res%=PRIME;
			}
		}
		printf("%Ld\n",res);
	}
	return 0;
}
