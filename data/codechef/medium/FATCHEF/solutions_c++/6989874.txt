#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define mod 1000000009
#define mp make_pair

int main()
{
    ll t, n, m, i, ans;
    scanf("%lld", &t);
    while (t--) {
        scanf("%lld%lld", &n, &m);
        pair <ll, char> p[m];
        for (i = 0; i < m; i++) {
            getchar();
            scanf("%c%lld", &p[i].second, &p[i].first);
        }
        sort (p, p+m);
        ans = 1;
        for (i = 1; i < m; i++) {
            if (p[i].second != p[i-1].second)
                ans = (ans * (p[i].first - p[i-1].first))%mod;
        }
        printf("%lld\n", ans);
    }
    return 0;
}
