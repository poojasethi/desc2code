#include <stdio.h>
#include<string.h>
 
 
int main()
{
	int N,M,t,d,i,f,mm;
	char cc[100000],ch;
	long long int sum;
	
	scanf("%d",&t);
		
		while(t--)
		{
				scanf("%d",&N);
				scanf("%d",&M);
				memset(cc,'0',100000);
				for(i=0;i<M;i++)
				{
					getchar();
					scanf("%c %d",&ch,&mm);
					cc[mm-1]=ch;
					
				}
					
					
				sum=1;f=-1;
				for(i=0;i<N;i++)
				{
					if(cc[i]>='A' && cc[i]<='Z' && f==-1)
					{
						f=i;
					}
					else if (cc[i]>='A'&&cc[i]<='Z' && cc[i]!=cc[f])
					{
					
						d=i-f;
						sum=(sum*d)%1000000009;
						f=i;
					}
					else if (cc[i]>='A'&&cc[i]<='Z' && cc[i]==cc[f])
					{
						f=i;
					}
				}
				
				printf("%lld\n",sum);
			
			
		}
	
	return 0;
}
