    void scanint(int &x)
{
register int c = gc();
x = 0;
int neg = 0;
for(;((c<48 || c>57) && c != '-');c = gc());
if(c=='-') {neg=1;c=gc();}
for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
if(neg) x=-x;
}
    void printlong(long long n)
{
char op[41];
int size = 40;
while(n)
{
op[size--] = n%10+'0';
n/=10;
}
for(int i=size+1;i<41;i++)
pc(op[i]);
pc('\n');
}
