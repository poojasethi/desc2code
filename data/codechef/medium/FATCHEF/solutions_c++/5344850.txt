#include<iostream>
#include<algorithm>
using namespace std;
#define getcx getchar_unlocked

 inline void inp( int &n ) 
 {
    n=0;
    int ch=getcx();int sign=1;
    while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

    while(  ch >= '0' && ch <= '9' )
            n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
    n=n*sign;
  }


int main()
{
	int tst;
	inp(tst);
	while(tst--)
	{
	
	int n,m;
	inp(n);
	inp(m);
	char ar[100010]={'2'};
	char temp;
	int temp1;
	int br[100010]={0};
	int index=1;
	for(int i=0;i<m;i++)
	{
		//cin>>temp;
		scanf("%c",&temp);
		inp(temp1);
		ar[temp1]=temp;
		br[index]=temp1;
		index++;		
	}
	
	sort(br,br+(m+1));
	unsigned long long ans=1;
	unsigned long long max=(1000000000+9);
	index=1;
	for(;index<m;index++)
	{
		if(ar[br[index]]==ar[br[index+1]])
		{
			continue;
		}
		
		else if(ar[br[index]]!=ar[br[index+1]])
		{
			ans*=(br[index+1]-br[index]);
			ans%=max;
		}
	}
	
	unsigned long long fans=ans%max;
	//cout<<fans<<"\n";
	printf("%llu\n",fans);
	
	}
	return 0;
}
