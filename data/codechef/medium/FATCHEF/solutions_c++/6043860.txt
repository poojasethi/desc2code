#include <algorithm>
#include <vector>
#include <cstring>
#include <string>
#include <cmath>
#include <map>
#define INF 1000000009
int scan()
{ int noRead=0;
char p= getchar();
for(;p<33;)
{ p=getchar(); }
while(p>32)
{
noRead = (noRead << 3) + (noRead << 1) + (p - '0');
p=getchar();
}
return noRead;
}
 
using namespace std;
int a[100001];
int main(void){
 
int t = scan();
while(t--) {
int n = scan();
int m = scan();
 
for(int i = 0; i < n; ++i)
a[i] = INF;
int prev = INF;
for(int i = 0; i < m; ++i) {
//getchar();
char x;
scanf("%c",&x);
//cout<<" c "<<x<<endl;
int num = scan();
//cout<<"num: "<<num;
//cout<<"x is "<<x;
a[num-1] = (int)x - 'A';
prev = min(prev,num-1);
}
long long ans = 1;
//cout<<"prev "<<prev<<endl;
//cout<<"prev + 1 "<<prev<<" n "<<n<<endl;
for(int i = prev + 1; i < n; ++i) {
//cout<<"i "<<i<<" ans : "<<ans<<"a[i] "<<a[i]<<endl;
if(a[i] == INF) {
continue;
}
if(a[i] == a[prev]) {
prev = i;
continue;
}
ans = (long long)(ans%INF * (long long)(i-prev)%INF)%INF;
prev = i;
//cout<<"i "<<i<<" ans : "<<ans<<endl;
}
printf("%lld\n",ans);
}
 
return 0;
 
}
