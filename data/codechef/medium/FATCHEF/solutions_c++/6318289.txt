#include<bits/stdc++.h>
#define mod 1000000009
using namespace std;
int main()
{
	int n,m,index,t;
	long long int ans=1;		
	char color;
	vector<pair<long long int,char> >x;
	scanf("%d",&t);
	while(t--)
	{	ans=1;
		x.clear();
		scanf("%d%d",&n,&m);
		for(int i=0;i<m;i++)
		{
			cin>>color;
			scanf("%d",&index);
			x.push_back(make_pair(index,color));
		}
		sort(x.begin(),x.end());
		for(int j=0;j<x.size()-1;j++)
		{
			if(x[j].second!=x[j+1].second)
			{
				long long int k=(x[j+1].first-x[j].first)%mod;
				ans=(ans*k)%mod;
			}
			else
				ans*=1;
		}
		printf("%lld\n",ans);
	}
	return 0;
}
