#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>
#define con 1000000009
 
using namespace std;

#define gc getchar_unlocked
 
inline long long readI(void)
{
	register long long c = gc();
	long long x = 0;
	for(;(c<48 || c>57);c = gc());
	for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
	return x;
}
 
inline char readC(void)
{
	register char c = gc();
	for(;(c<65 || c>90);c = gc());
	return c;
}

class CharAndInt
{
	public:
		char c;
		long long i;
		CharAndInt(char ch,long long pos)
		{
			c = ch;
			i = pos;
		}
};
 
bool myfunction (CharAndInt a,CharAndInt b)
{
	return (a.i<b.i);
}
 
int main()
{
	long long int t,n,m,i,j;
	t = readI();
	//cin>>t;
	while(t--)
	{
		char c;
		vector<CharAndInt> arr;
		long long res=1,mark;
		scanf("%lld %lld",&n,&m);
		//cin>>n>>m;
		for(i=0;i<m;i++)
		{
			c = readC();
			mark = readI();
			//cin>>c>>mark;
			arr.push_back(CharAndInt(c,mark));
		}
		sort(arr.begin(),arr.end(), myfunction);
		if(m==1)
		{
			res = 1;
		}
		else
		{
			for(i=1;i<m;i++)
			{
				if(arr[i].c!=arr[i-1].c)
				{
					res=(res%con)*(((arr[i].i%con)-(arr[i-1].i%con))%con);
					res = res%con;
				}
			}
		}
		printf("%lld\n",res);
		//cout<<res<<endl;
	}
}
