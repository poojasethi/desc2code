#include <bits/stdc++.h>
using namespace std;

#define REP( i , n ) for( int i = 0 ; i < n ; ++i )

#define pb push_back
#define SZ( v ) ((int)(v).size())
#define all( v ) v.begin() , v.end()

#define mp make_pair
#define fi first
#define se second

#define MOD 1000000009LL

typedef pair< int , int > pii;
typedef long long ll;

int main() {
	ios_base :: sync_with_stdio( 0 );
	int cases;
	cin >> cases;
	REP( tc , cases ){
		int n;
		cin >> n >> n;
		vector< pii > v;
		REP( i , n ){
			char c;
			int pos;
			cin >> c >> pos;
			v.pb( mp( pos , c ) );
		}
		sort( all( v ) );
		ll ans = 1;
		REP( i , SZ( v ) - 1 ){
			if( v[ i ].se != v[ i + 1 ].se ){
				ll len = v[ i + 1 ].fi - v[ i ].fi;
				ans = (ans * len)%MOD;
			}
		}
		cout << ans << '\n';
	}
}
