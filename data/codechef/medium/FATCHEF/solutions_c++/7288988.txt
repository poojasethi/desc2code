#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
#define MOD 1000000009;
int main()
{
    std::ios::sync_with_stdio(false);
    //freopen("in.txt","r",stdin);
    int TC;
    cin >> TC;
    while(TC--){
        int N,M;
        cin >> N >> M;
        vector<pair<int,char> > v;
        for(int i=0;i<M;i++){
            char ch;int p;
            cin >> ch >> p;
            v.push_back(make_pair(p,ch));
        }
        sort(v.begin(),v.end());
        long long int res=1;
        for(int i=1;i<v.size();i++){
            if(v[i].second!=v[i-1].second){
                long long int val=v[i].first - v[i-1].first;
                val=val%MOD;
                res=(res * val)%MOD;
            }
        }
        printf("%lld\n",res);
    }
}
