#define llint long long
#define mod 1000000009

struct P{
	int first1;char second1;
	P (int x, char y){
		first1=x;
		second1=y;
	}
	bool operator<(const P& rhs) const{
		return first1<rhs.first1;
	}
};

int main(){
	int t;scanf("%d\n",&t);
	while(t--){

		int n,m;scanf("%d %d\n",&n,&m);
		vector<P> input;
		for (int i=0;i<m;i++){
			char x;
			int y;
			scanf("%c %i\n",&x,&y);
			//cout<<x<<y<<endl;
			input.push_back(P(y,x));
		}
		sort(input.begin(),input.end());
		llint count =1;
		for (int i=0;i<m-1;i++){
			if (input[i].second1==input[i+1].second1 || input[i+1].first1-input[i].first1==1) continue;
			count = (count*(input[i+1].first1-input[i].first1))%mod;

		}
		printf("%lld\n",count );;
	}
	return 0;
}
