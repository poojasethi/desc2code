#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstdlib>
 
using namespace std;
 
#define FOR(i,a,b) for(int i=a;i<b;i++)
#define K 10000000
#define L 400
#define ll long long
#define s1(a) scanf("%d",&a);
#define s2(a) scanf("%lld",&a);
#define s3(a,b) scanf("%d%d",&a,&b);
#define s4(a) scanf("%c",&a);
#define INF 1000000
 
int n,m,a[110][110],sh[110][110],dp[110][110][2]; 
char str[101];
 
void scan()
{
	char c;
	s3(n,m)
	//getchar();
	FOR(i,0,n) 
	{
        	scanf("%s",str);
        	FOR(j,0,m) 
        		a[i][j] = str[j]-'0';
        }
	FOR(i,0,n)
		FOR(j,0,m)
			sh[i][j]=0;
	//printf("Y\n");
	FOR(i,0,n)
	{
		FOR(j,0,m)
		{
			if(a[i][j]==1)
			{
				if(i!=0)
					++sh[i-1][j];
				if(j!=0)
					++sh[i][j-1];
				if(j!=m-1)
					++sh[i][j+1];
				if(i!=n-1)
					++sh[i+1][j];
			}
		}
	}
}

int minn(int a,int b)
{
	a=a>b?b:a;
	return a;
}
 
void out()
{
	dp[0][0][0]=dp[0][0][1]=sh[0][0] + a[0][0];
	FOR(i,1,n)
	{
		dp[i][0][1]=dp[i-1][0][1] - a[i-1][0] + sh[i][0];
		dp[i][0][0]=INF;
	}
	FOR(i,1,m)
	{
		dp[0][i][0]=dp[0][i-1][0] + sh[0][i] - a[0][i-1];
		dp[0][i][1]=INF;
	}
	FOR(i,1,n)
	{
		FOR(j,1,m)
		{
			dp[i][j][0]=sh[i][j] - a[i][j-1] + min(dp[i][j-1][0],dp[i][j-1][1] - a[i-1][j]);
			dp[i][j][1]=sh[i][j] - a[i-1][j] + min(dp[i-1][j][1],dp[i-1][j][0] - a[i][j-1]);
		}
	}
	/*FOR(i,0,n)
	{
		FOR(j,0,m)
		{
			printf("%d %d\t",dp[i][j][0],dp[i][j][1]);
		}
		printf("\n");
	}*/
	int ans=min(dp[n-1][m-1][0],dp[n-1][m-1][1]);
	printf("%d\n",ans);
}
 
int main()
{
	int t;
	s1(t)
	//getchar();
	while(t--)
	{
		scan();
		out();
	}
	return 0;
}
