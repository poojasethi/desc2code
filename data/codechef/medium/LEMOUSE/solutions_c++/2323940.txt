int n,m;
string s[105];
int u[105][105],l[105][105];

int mouse(int x,int y)
{
	if(x<0||x>=n||y<0||y>=m)
		return 0;
	return s[x][y]=='1';
}

int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	int t;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		rep(i,n)
			cin>>s[i];
		u[0][0]=l[0][0]=mouse(0,0)+mouse(0,1)+mouse(1,0);
		REP(i,1,m)
		{
			u[0][i]=10005;
			l[0][i]=l[0][i-1]+mouse(0,i+1)+mouse(1,i);
		}
		REP(i,1,n)
		{
			u[i][0]=u[i-1][0]+mouse(i,1)+mouse(i+1,0);
			l[i][0]=10005;
			REP(j,1,m)
			{
				u[i][j]=min(u[i-1][j]+mouse(i,j-1),l[i-1][j])+mouse(i,j+1)+mouse(i+1,j);
				l[i][j]=min(u[i][j-1],l[i][j-1]+mouse(i-1,j))+mouse(i,j+1)+mouse(i+1,j);
			}
		}
		cout<<min(u[n-1][m-1],l[n-1][m-1])<<'\n';
	}
	return 0;
}
