#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[105][105],b[105][105][2];
char str[1005];
int main()
{	int t,n,m,i,j,temp1,temp2;
	cin>>t;
	while(t--){
		cin>>n>>m;
		for(i=1;i<=n;i++)
		{	a[i][0]=0;
			scanf("%s",str);
			for(j=1;j<=m;j++)a[i][j]=str[j-1]-'0';
			a[i][j]=0;
		}
		for(j=0;j<=(m+1);j++)a[0][j]=a[n+1][j]=0;
		for(i=0;i<=n+1;i++)
		for(j=0;j<=m+1;j++)
		{
			b[i][j][0]=b[i][j][1]=0;
		}
		for(i=n;i>=1;i--)
		for(j=m;j>=1;j--)
		{	b[i][j][0]=b[i][j][1]=0;
			if(i==n&&j==m){
				b[i][j][0]=b[i][j][1]=(a[i][j]+a[i-1][j]+a[i][j-1]);
				continue;
			}
			if(i==n){
				//FROM DOWN
				b[i][j][1]=1000000;
				//FROM RIGHT
				b[i][j][0]=min((b[i][j+1][0]+a[i-1][j]+a[i][j-1]+a[i+1][j]),(b[i][j+1][1]+a[i-1][j]+a[i][j-1]));
				continue;
			}
			if(j==m){
				//FROM DOWN
				b[i][j][1]=min((b[i+1][j][0]+a[i][j-1]+a[i-1][j]),(b[i+1][j][1]+a[i][j-1]+a[i-1][j]+a[i][j+1]));
				//FROM RIGHT
				b[i][j][0]=1000000;
				continue;
			}
			//FROM DOWN
			b[i][j][1]=min((b[i+1][j][0]+a[i][j-1]+a[i-1][j]),(b[i+1][j][1]+a[i][j-1]+a[i-1][j]+a[i][j+1]));
			//FROM RIGHT
			b[i][j][0]=min((b[i][j+1][0]+a[i-1][j]+a[i][j-1]+a[i+1][j]),(b[i][j+1][1]+a[i-1][j]+a[i][j-1]));
		}
		//cout<<"RIGHT"<<endl;for(i=1;i<=n;i++){for(j=1;j<=m;j++)cout<<b[i][j][0]<<" ";cout<<endl;}
		//cout<<"DOWN"<<endl;for(i=1;i<=n;i++){for(j=1;j<=m;j++)cout<<b[i][j][1]<<" ";cout<<endl;}
		cout<<min(b[1][1][0],b[1][1][1])<<endl;
	}
	return 0;
}
