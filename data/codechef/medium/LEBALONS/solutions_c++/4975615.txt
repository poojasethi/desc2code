#include<stdio.h>
#include<string.h>
int main()
{
	long long t,n,m,i,j,k,x,y;
	long long colors[41],sum[41];
	long long po2[41];		
	po2[0]=1;
	for(i=1;i<=40;i++)
		po2[i]=po2[i-1]*2;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&n,&m);
		memset(colors,0,sizeof(colors));
		memset(sum,0,sizeof(sum));
		for(i=1;i<=n;i++)
		{
			scanf("%lld%lld",&x,&y);
			colors[x]++;
			sum[x]+=y;
		}
		long long ways[41][41],cost[41][41];
		for(i=1;i<=40;i++)
		{
			ways[i][0]=1;
			ways[0][i]=0;
			cost[i][0]=0;
			cost[0][i]=0;
		}
		ways[0][0]=1;
		cost[0][0]=0;
		for(i=1;i<=40;i++)
		{			
			for(j=1;j<=40;j++)
			{
				if(colors[i]>0)
				{
					ways[i][j]=ways[i-1][j]+ways[i-1][j-1]*(po2[colors[i]]-1);
					cost[i][j]=cost[i-1][j]+cost[i-1][j-1]*(po2[colors[i]]-1)+ways[i-1][j-1]*(sum[i])*(po2[colors[i]-1]);
				}
				else
				{
					ways[i][j]=ways[i-1][j];
					cost[i][j]=cost[i-1][j];
				}
			}
		}
		long long num=0,den=0;
		for(i=m;i<=40;i++)
		{
			num+=cost[40][i];
			den+=ways[40][i];
		}
		double d=(num*1.0)/den;
		printf("%lf\n",d);
	}
	return 0;
}
