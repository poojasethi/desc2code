long long int fd(int i)
{
    map<int,int>m;
     map<int,int>::iterator it;

        i=i/2;


    for(j=3;j<=sqrt(i);j=j+2)
    {

        while(i%j==0)
        {
            i=i/j;
            m[j]++;

        }


    }
    if(i>2)
      m[i]++;

    for(it=m.begin();it!=m.end();it++)
    {
        s*=(pow((it->first),(it->second+1))-1)/(it->first-1);

    }
    return s;

}
