#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;

ll arr[100005];
void solve()
{
    for(int i=1;i<=100005;i++)arr[i]=1;
    for(int i=2;i<=100005;i++)
    {
        if(i%2!=0)
        {
            int k=1;
            for(int j=i;j<=100005;j=i*k)
            {
                arr[j]+=i;
                k++;
            }
        }
    }
}
int main()
{
    solve();
    ll t,l,r,ans;
    cin>>t;
    while(t--)
    {
        ans=0;
        cin>>l>>r;
        for(int i=l;i<=r;i++)ans+=arr[i];
        cout<<ans<<endl;
    }
    return 0;
    
}
