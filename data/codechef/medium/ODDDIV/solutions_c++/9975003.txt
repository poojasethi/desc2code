#include <stdio.h>

const int MAX = 1e5;
int test, l, r;
long long sum[MAX + 1];

int main(){
    //freopen("in.txt", "r", stdin);
    scanf("%d", &test);
    for (int i = 1; i <= MAX; i += 2)
    for (int j = 1; j <= MAX / i; ++j)
        sum[i * j] += i;

    for (int i = 2; i <= MAX; ++i)
        sum[i] += sum[i - 1];

    while (test--){
        scanf("%d %d", &l, &r);
        printf("%lld\n", sum[r] - sum[l - 1]);
    }
    return 0;
}
