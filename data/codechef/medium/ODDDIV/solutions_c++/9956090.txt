#include<bits/stdc++.h>
using namespace std;
#define Max 100005
int f[Max];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	for(int i=1;i<=100000;i++)
	{
		for(int j=2;j*j<=i;j++)
		{
			if(i%j==0)
			{
				if(j*j==i)
				{
					if(j%2!=0)
						f[i]+=j;
				}
				else
				{
					if(j%2!=0)
					{
						f[i]+=j;
						if(i/j %2 !=0)
							f[i]=f[i]+i/j;
					}
					else
					{
						if(i/j %2 !=0)
							f[i]=f[i]+i/j;
					}
				}
			}	
		}
		if(i%2!=0)
			f[i]=f[i]+i;
		if(i!=1)
			f[i]+=1;
	}
	int t;
	cin>>t;
	while(t--)
	{
		int l,r;
		cin>>l>>r;
		long long ans=0;
		for(int i=l;i<=r;i++)
		{
			ans=ans+f[i];
		}
		cout<<ans<<endl;
	}

	
	return 0;
}
