#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <cstring>
#include <string>

using namespace std;

typedef long long int64;

#define MAXN 100005
int64 ds[MAXN];

int main(){
   memset(ds, 0, sizeof(ds));
   for(int i=1; i < MAXN; ++i){
      int ni = i;
      while( (ni % 2) == 0) ni /= 2;
      if( ni != i ) ds[i] = ds[ni];
      else {
         for(int64 j=1; j*j <= i; j+= 2){
             if( (i % j) == 0) {
                  ds[i] += j;
                  if( j*j != i ) ds[i] += i/j;
             }
         }
      }
   }
   for(int i=1; i < MAXN; ++i) ds[i] += ds[i-1];
   int tcase;
   for( scanf("%d", &tcase); tcase; tcase--){
       int a, b; scanf("%d %d", &a, &b);
       printf("%lld\n", ds[b] - ds[a-1]);
   }
   return 0;
}
