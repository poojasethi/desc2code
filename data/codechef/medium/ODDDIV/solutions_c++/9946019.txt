#include <bits/stdc++.h>
using namespace std;

int main()
{
        long long t,l,r,ans[100001],n=100001;

        for(int i=1;i<=n;++i) ans[i]=1;
        for(int i=3;i<=n;i+=2) {
                for(int j=i;j<=n;j+=i) ans[j]+=i;
        }
        //for(int i=1;i<=100;++i) cout<<ans[i]<<"\n";

        cin>>t;
        while(t--) {
                cin>>l>>r;
                long long val=0;
                for(int i=l;i<=r;++i) val+=ans[i];
                cout<<val<<"\n";
        }

        return 0;
}
