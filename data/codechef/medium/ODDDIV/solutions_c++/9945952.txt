#include<iostream>
using namespace std;

long long calc(long long x)
{
    long long sum = 0;
    for(int i=1;i<=x;i+=2)
    {
        sum+=i * (x/i);
    }
    return sum;
}

int main()
{
    int T;
    cin >> T;
    while(T--)
    {
        int l, r;
        cin >> l >> r;
        cout << calc(r) - calc(l-1) << "\n";
    }
}
