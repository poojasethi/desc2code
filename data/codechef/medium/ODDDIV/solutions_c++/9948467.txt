#include<bits/stdc++.h>
using namespace std;
long long ans[100005];
int main()
{
	for(int i=1;i<=100000;i++)
	{
		int k=i;
		for(int j=1;j<=sqrt(i);j++)
		{
			if(k%j==0)
			{
				if(j%2)
				ans[i]+=j;
				if(j!=k/j&&(k/j)%2)
				ans[i]+=k/j;	
			}
		}
	}
	for(int i=1;i<=100000;i++)
	ans[i]=ans[i]+ans[i-1];
	int t;
	cin>>t;
	while(t--)
	{
		int l,r;
		cin>>l>>r;
		cout<<ans[r]-ans[l-1]<<endl;
	}
}
