#include <iostream>
using namespace std;
#include <vector>
#include <math.h>
#define ll long long
vector<ll> z;

ll calc(ll n)
{   if(n==1)
        return 1;
    ll t=sqrt(n),s=0;
    for(ll i=1;i<=t;i+=2)
        if(n%i==0)
        {   s+=i;
            s+=n/i;
        }
    if(t*t==n)
        s-=t;
    z[n]=s;
    return s;
}

int main()
{   ll t,i;
    for(i=0;i<100001;i++)
        z.push_back(0);
    cin>>t;
    while(t--)
    {   ll a,b,s=0;
        cin>>a>>b;
        for(i=a;i<=b;i++)
        {   ll x=i;
            while(x%2==0)
                x/=2;
            if(z[x]!=0)
                s+=z[x];
            else
                s+=calc(x);
        }
        cout<<s<<"\n";
    }
    return 0;
}
