#include <iostream>
using namespace std;
#define ll long long
#define lim 100000
ll po(ll a, ll b)

{
	ll re=1;
	while(b>0)
	{
		if(b&1)
		re*=a;
		b=b>>1;
		a*=a;
	}
	return re;
}
ll ans[lim+1];
bool prime[lim+1];
ll getans(ll i,ll temp)
{
	ll num=po(i,temp+1);
	num--;
	ll den=i-1;
	return num/den;
}
int main() {
	for(ll i=0;i<=lim;i++)
    { ans[i]=1;
     prime[i]=0;}
     for(ll i=2;i*i<=lim;i++)
     {
     	if(!prime[i])
     	{
     		for(ll j=2*i;j<=lim;j+=i)
     		prime[j]=true;
     	}
     
     }
     ll t;
     
     for(ll i=3;i<=lim;i+=2)
     {    if(!prime[i])
     	for(ll j=i;j<=lim;j+=i)
     	{
     		ll temp=0;
     		ll q=j;
     		while(q%i==0)
     		{
     			q/=i;
     			temp++;
     		}
     		ans[j]*=getans(i,temp);
     	}
     }
     ans[0]=0;
     ll l,r;
     for(ll i=1;i<=lim;i++)
     ans[i]+=ans[i-1];
     cin>>t;
     while(t--)
     {
     	cin>>l>>r;
     	cout<<ans[r]-ans[l-1]<<endl;
     
     }
}
