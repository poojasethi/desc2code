#include<stdio.h>
typedef long long ll;
ll odd(ll n)
{
	 ll i;
	 ll k1,k2;
	 ll sum=0;
	 for(i=1;i*i<=n;i++)
	 {
	      if(n%i==0)
		  {
		  	k1=i;
		  	k2=n/i;
		  	if(k1==k2)
		  	{
		  		if(k1%2!=0)
		  		sum=sum+k1;
			  }
			  else
			  {
			  	if(k1%2!=0)
			  	sum=sum+k1;
			  	if(k2%2!=0)
			  	sum=sum+k2;
			  }
		   } 
	 }
	 return sum;
}
int main()
{
	ll t,n,i,l,r;
	ll ar[100005];
	scanf("%lld",&t);
	ar[0]=0;
	for(i=1;i<=100000;i++)
	{
		ll k=odd(i);
		ar[i]=k;
	}
	for(i=1;i<=100000;i++)
		ar[i]=ar[i]+ar[i-1];
	while(t--)
	{
		scanf("%lld%lld",&l,&r);
		ll ans=ar[r]-ar[l-1];
		printf("%lld\n",ans);
	}
	return 0;
}
