#include<bits/stdc++.h>
using namespace std;

const int maxn = 10e5+2;
int f[maxn];


//sum of divisors(odd)
void sieve()
{
    memset(f,0,sizeof f);

    for(int i=1;i<maxn;i+=2)
    {
        for(int j=i;j<maxn;j+=i)
            f[j]+=i;
    }
}

int main()
{
    int t;
    scanf("%d",&t);

    sieve();

    while(t--)
    {
        int l,r;
        scanf("%d %d",&l,&r);

        long long sum=0;

        for(int i=l;i<=r;++i)
        {
                sum+=f[i];
        }

        printf("%lld\n",sum);
    }
    return 0;
}
