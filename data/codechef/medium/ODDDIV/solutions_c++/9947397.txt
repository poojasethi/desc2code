#include <iostream>
#include<cstdio>
#include<map>
#include<vector>
#include<string.h>
#include<limits.h>
#include<set>
#include<math.h>

using namespace std;

#define LL long long
#define mod 1000000007

LL sum[100001];

int main()
{
    sum[0] = 0;
    sum[1] = 1;
    sum[2] = 2;

    for(int i=3;i<=100000;i++)
    {
        int m = sqrt(i);
        sum[i] = sum[i-1];
        for(int j=1;j<=m;j++)
            if(i%j==0)
            {
                int d = i/j;
                if(j&1)
                    sum[i] += j;
                if(d!=j && d&1)
                    sum[i] += d;
            }
    }

    int t,l,r;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&l,&r);
        printf("%lld\n",sum[r]-sum[l-1]);
    }
    return 0;
}
