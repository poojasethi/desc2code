#include<bits/stdc++.h>

using namespace std;
typedef long long lli;

int main(void)
{
	lli arr[100005];
	for(int i=0;i<100005;i++)
	    arr[i]=(lli)1;

	for(int i=3;i<100005;i+=2)
	{
		int j=1;
		while(i*j<100005)
		{
			arr[i*j]+=(lli)i;
			j++;
		}
	}
	lli sum[100005]={0};
	sum[0]=0;
	for(int i=1;i<100005;i++)
	{
		sum[i]=sum[i-1]+arr[i];
	}
	
	///for(int i=0;i<15;i++)
	   //cout<<arr[i]<<" ";
	int t; scanf("%d",&t);
	while(t--)
	{
		int l,r; scanf("%d %d",&l,&r);
		if(l>1)
		{
				printf("%lld\n",sum[r]-sum[l-1]);
			
		}
		else if(l==1)
		{
			printf("%lld\n",sum[r]);
		}
		
	}
	
	
	return 0;
}
