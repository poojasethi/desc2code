ULL powermod(ULL x, ULL y)
{
    ULL temp;
    if( y == 0)
       return 1;
    temp = powermod(x, y/2);
    if (y%2 == 0)
        return (temp*temp)%MOD;
    else
    {
        if(y > 0)
            return (x*((temp*temp)%MOD))%MOD;
        else
            return (temp*temp)/x;
    }
}
