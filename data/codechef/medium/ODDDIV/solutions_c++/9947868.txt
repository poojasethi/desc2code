#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin>>t;
	
	long long int f[100006]={0};
	for(int i=1;i<=100001;i+=2)
	{
		for(int d=i;d<=100001;d+=i)
		f[d]+=i;
	}
	
	while(t--)
    {
    	long long int sum=0;
    	int l,r;
    	cin>>l>>r;
    	
    	for(int i=l;i<=r;i++)
    	sum+=f[i];
    	cout<<sum<<endl;
    	
    }
 }
