#include<bits/stdc++.h>
using namespace std;
map<long long int,long long int > a;
long long int sum(long long int n)
{
	if(a[n]!=0)
	return a[n];
	else
	{
		long long int i,z=0;
		for(i=1;i<=sqrt(n);i++)
		{
			if(n%i==0)
			{
				if(i%2!=0)
				{
				z+=i;
			//	cout<<"div = "<<i<<endl;
			    }
				if((n/i)%2!=0 && (n/i)!=i)
				{
				z+=(n/i);
			//	cout<<"div = "<<n/i<<endl;
			    }
			}
		}
		a[n]=z;
	//	cout<<"z = "<<z<<endl;
		return z;
	}
}
main()
{
	ios_base::sync_with_stdio(0);cin.tie(0);
	int t;
	long long int l,r,i,count;
	cin>>t;
	while(t--)
	{
		count=0;
		cin>>l>>r;
		for(i=l;i<=r;i++)
		{
			count+=sum(i);
		}
		cout<<count<<endl;
		}	
}
