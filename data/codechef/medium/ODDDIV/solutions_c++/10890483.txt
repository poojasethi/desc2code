using namespace std;

vector <int> vect;

lli arr[100005];
void pr()
{
    for(int i=1;i<=100000;i++)
    {
        if(i%2!=0)
        {
            lli x=0;
            while(x+(lli)i<=(lli)100000)
            {
                arr[x+i]+=i;
                x=x+i;
            }
        }
    }
}


int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    memset(arr,0,sizeof arr);
    pr();
    cin>>t;
    while(t-->0)
    {
        int x,y;
        cin>>x>>y;
        lli ans=0;
        for(int i=x;i<=y;i++)
            {
                ans+=arr[i];
            }
            cout<<ans<<endl;
    }
    return 0;
}
