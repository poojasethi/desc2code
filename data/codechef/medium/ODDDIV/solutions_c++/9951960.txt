void convert(){
    for(int i=1;i<maxn;i++){
        v[i]=i;
        if(v[i]%2==0){
            while(v[i]%2==0)
                v[i] /= 2;
        }
    }
}
void preprocess(){
    memset(sum,0,sizeof(sum));
    for(int i=1;i<maxn;i++){
        for(int j=1;j<=sqrt(v[i]);j+=2){
            if(v[i]==1)
               sum[i] += 1; 
            else if(v[i]!=1 && v[i]%j==0){
                if(v[i]/j!=j)
                    sum[i] += j+(v[i]/j);
                else
                    sum[i] += j;
            }
        }
    }
}
int main()
{
    ios::sync_with_stdio(false);
    convert();
    preprocess();
    int t;
    cin>>t;
    while(t--){
        int l,r;
        cin>>l>>r;
        ll ans=0;

        for(int i=l;i<=r;i++){
            //cout<<v[i]<<" ";
            ans += sum[i];
        }
        cout<<ans<<endl;
    }
}
