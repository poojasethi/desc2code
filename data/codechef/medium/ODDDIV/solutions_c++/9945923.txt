#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	long long int x,y,l,r;
	long long int arr[100001]={0};
	for(x=1;x<=100000;x++)
	{
		for(y=1;y<=sqrt(x);y++)
		{
			if(x%y==0)
			{	
			if(y%2!=0)
			{
			arr[x]=arr[x]+y;
			}
			if((x/y)!=y && (x/y)%2!=0)
			{
				arr[x]=arr[x]+(x/y);
			}
			}
		}
	}
	while(t!=0)
	{
		long long int sum=0;
		cin>>l>>r;
		for(x=l;x<=r;x++)
		{
			sum=sum+arr[x];
			
		}
		cout<<sum<<"\n";
		t--;
	}
	return 0;
}
