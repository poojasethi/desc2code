#include<bits/stdc++.h>
using namespace std;

typedef long long int  ll;

ll Odd[100005],Sum;
int T,i,j,L,R;

int main() {
	for(i=1;i<=100000;i+=2) {
		for(j=i;j<=100000;j+=i)
			Odd[j]+=i;
	}
	scanf("%d",&T);
	while(T--) {
		Sum=0;
		scanf("%d %d",&L,&R);
		for(i=L;i<=R;++i) Sum+=Odd[i];
		printf("%lld\n",Sum);
	}
	return 0;
}
