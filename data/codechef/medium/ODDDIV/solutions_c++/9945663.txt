#include<stdio.h>
#include<iostream>
#include<math.h>
using namespace std;
long long int divsum(long int n)
{
long long int sum = 1;
long int i=2,x;
 while(i <= sqrt(n))
	    {
	        if( n%i==0) {
	        if(i%2!=0)
	            sum += i;
	            x=n/i;
	            if (i != x  && x%2!=0) {
	                sum += x;
	            }
	        } 
	 
	        i++;
	    }
	return (sum +=(n%2!=0 && n>1?n:0));
    
}
int main()
{
long int t,l,r,i;
long long int sum;
scanf("%ld",&t);
while(t--)
{
scanf("%ld%ld",&l,&r);
sum=0;
for(i=l;i<=r;i++)
{
sum += divsum(i);        
}
printf("%lld\n",sum);
}
return 0;
 }
