const int MAX = 100001;

LL f[MAX];
void pre()
{
	for(int i = 1; i < MAX; i += 2){
		for(int j = i; j < MAX; j += i)
			f[j] += i;
	}
	debugarr(f,1,20);
	rep(i,1,MAX-1) f[i] += f[i-1];
}

int main()
{
	pre();
	int t; S(t);
	while(t--){
		int l, r; 
		SS(l,r);
		printf("%lld\n",f[r] - f[l-1]);
	}
	return 0;
}
