#include <bits/stdc++.h>
using namespace std;

int main()
{
        long long int t,l,r,i,ans[100001];

        for(int i=1;i<=100001;++i) ans[i]=1;
        for(int i=2;i<=100001;++i) {
                if(i%2==1) {
                        for(int j=i;j<=100001;j+=i) ans[j]+=i;
                }               
        }
        //for(int i=1;i<=100;++i) cout<<ans[i]<<"\n";
        //cout<<ans[42]<<"\n";
        
        cin>>t;
        while(t--) {
                cin>>l>>r;
                long long val=0;
                for(int i=l;i<=r;++i) val+=ans[i];
                cout<<val<<"\n";
        }

        return 0;
}
