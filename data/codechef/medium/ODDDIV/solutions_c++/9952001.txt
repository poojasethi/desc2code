#include <bits/stdc++.h>
using namespace std;

int main(){
    int t;
    cin >> t;
    while(t--){
        int l,r;
        cin >> l >> r;
        long long ans = 0;
        for(int i = 1; i <= r; i+=2){
            ans += (r/i)*i;
        }
        for(int i = 1; i <= l-1; i+=2){
            ans -= ((l-1)/i)*i;
        }
        cout << ans << endl;
    }
}
