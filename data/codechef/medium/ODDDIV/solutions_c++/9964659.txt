#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int main() {
	int T;
	scanf("%d",&T);
	while(T--)
	{
	    int l,r;
	    long long int sum=0;
	    scanf("%d",&l);
	    scanf("%d",&r);
	    long long int arr[r+1];
	    memset(arr,0,sizeof(arr));
	    
	    for(int i=1;i<=r;i=i+2)
	    {
	     int j=1;
	     while(i*j<=r)
	     {
	         arr[i*j]+=i;
	         j+=1;
	     }
	    }
	    for(int i=l;i<=r;i++)
	    {
	        sum=sum+arr[i];
	    }
	    cout<<sum<<"\n";
	}
	return 0;
}
