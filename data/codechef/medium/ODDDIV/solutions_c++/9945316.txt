#include <iostream>
using namespace std;
long long sum(int a,int b,int c)
{
     long long res;
     res=(b/c-a/c);
     if(a%c==0)
     res++;
     res=res*c;
     return res;
     
}
int main() {
	int t;
	long long l,r,i;
	cin>>t;
	while(t--)
	{
	     long long t=0;
	     cin>>l>>r;
	     for(i=1;i<=r;i=i+2)
	     {
	          t=t+sum(l,r,i);     
	     }
	     cout<<t<<endl;
	}
	return 0;
}
