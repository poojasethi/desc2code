#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	int t,i,j,l,r;
    cin>>t;
	while(t--)
	{
		cin>>l>>r;
		long long sum=r-l+1;
		for(i=l;i<=r;i++)
		{
	//	sum=0;
			for(j=2;j<=sqrt(i);j++)
			{
				if(i%j==0)
				{
					if(j%2==1)
					sum+=j;
					if(j!=(i/j)&&(i/j)%2==1)
					sum+=i/j;
				}	
			}
			if(i%2==1&&i!=1)
			sum+=i;
		//	printf("%llu\n",sum);
		}
		cout<<sum<<endl;
   }
return 0;	
}
