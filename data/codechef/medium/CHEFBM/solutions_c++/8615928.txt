#include<bits/stdc++.h>
using namespace std;

typedef pair<int,int> pp;
vector<pp> vec[100000+5];
int cl[1000000];
vector<int> d;

int main()
{
    int n,m,p,x,y;
    scanf("%d%d%d",&n,&m,&p);
    for(int i=0;i<p;i++)
    {
        scanf("%d%d",&x,&y);
        vec[x].push_back(make_pair(y,1));

        if(y>1)
            vec[x].push_back(make_pair(y-1,0));
        if(y<m)
            vec[x].push_back(make_pair(y+1,0));
    }

    for(int i=1;i<=n;i++)
    {
        if(vec[i].size())
        {
            vec[i].push_back(make_pair(1,0));
            vec[i].push_back(make_pair(m,0));

            for(int j=0;j<vec[i].size();j++)
                cl[vec[i][j].first]=vec[i][j].first;

            for(int j=0;j<vec[i].size();j++)
            {
                cl[vec[i][j].first]+=vec[i][j].second;
            }

            d.clear();

            for(int j=0;j<vec[i].size();j++)
            {
                d.push_back(vec[i][j].first);
            }

            sort(d.begin(),d.end());

            d.resize(unique(d.begin(),d.end())-d.begin());
            int ans=0;
            for(int j=1;j<d.size();j++)
            {
                ans+=cl[d[j]]-cl[d[j-1]];
                if(cl[d[j]]<cl[d[j-1]])
                {
                    ans=-1;
                    break;
                }
            }
            printf("%d\n",ans);
        }
        else
        {
            printf("%d\n",m-1);
        }
    }
    return 0;
}
