#include<iostream>
#include<algorithm>
#include<vector>
#include<stdio.h>
 
using namespace std;
 
int main()
{
int n,m,p,b,c,sum,pos,x=-1;
scanf("%d%d%d",&n,&m,&p);
vector<int> v[100010];
for(int i=0;i<p;i++)
{
scanf("%d%d",&b,&c);
v[b].push_back(c);
}
int a[m+5];
bool flag=true;
for(int i=1;i<=m;i++)
a[i]=i;
for(int i=1;i<=n;i++)
{
flag=true;
sum=m-1;
if(v[i].empty())
printf("%d\n",sum);
else
{
for(int j=0;j<v[i].size();j++)
a[v[i][j]]++;
for(int j=0;j<v[i].size();j++)
{
pos=v[i][j];
if(pos+1<=m && a[pos]>a[pos+1])
{
flag=false;
break;
}
if(pos-1<1) sum--;
if(pos+1>m) sum++;
}
if(flag)
printf("%d\n",sum);
else
printf("%d\n",x);
for(int j=0;j<v[i].size();j++)
a[v[i][j]]--;
 
}

}
return 0;}
