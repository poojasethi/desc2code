#include <stdio.h>
#include <stdlib.h>
#include <cmath>
#include <string.h>
#include <iostream>
#include <vector>
#include <map>


using namespace std;
int main()
{
  int c[100001];
  map< pair<int, int>,int > coordi;
  map< pair<int,int>,int >::iterator iter;
  map< pair<int,int>,int >::iterator niter;
  int n,m,p,a,b;
  scanf("%d%d%d",&n,&m,&p);
  for (int i=0; i<n; i++)
    c[i]=m-1;
        
  for (int i=0; i<p; i++)
  {
    cin >> a >> b;
    coordi[pair<int,int>(a-1,b-1)]++;
    if(b-1==0)
      c[a-1]--;
    if (b-1==m-1)
      c[a-1]++;
  }
  for(iter=coordi.begin();iter!=coordi.end();++iter)
  {
    niter=iter;
    if(++niter!=coordi.end())
    {
      if((iter->first).first==(niter->first).first)
      {
        if((iter->first).second+1==(niter->first).second)
        {
          if(iter->second > (niter->second)+1)
          {
            c[(iter->first).first]=-1;
          }
        }
        else
        {
          if((iter->second)>1)
          {
            c[(iter->first).first]=-1;
          }
        }
      }
      else 
      {
        if((iter->first).second!=(m-1))
          if((iter->second)>1)
          {
            c[(iter->first).first]=-1;
          }
      }
    }
    else
    {
      if((iter->first).second!=(m-1))
      {
        if((iter->second)>1)
        {
          c[(iter->first).first]=-1;
        }
      }
    }
  }
  for (int i=0; i<n; i++)
  {
    printf("%d\n",c[i]);
  }
  return 0;
}
