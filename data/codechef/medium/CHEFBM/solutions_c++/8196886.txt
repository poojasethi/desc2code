#include<bits/stdc++.h>
using namespace std;


int main() 
{
	std::ios::sync_with_stdio(false); 
	int n,m,p;
	cin>>n>>m>>p;
	vector<int>matrix[n+1],val;
	for(int k=0;k<p;k++)
	{
		int i,j;
		cin>>i>>j;
		--i;
		--j;
		matrix[i].push_back(j);
	
	}
    val.push_back(0);
	for(int i=0;i<m;i++)
	{
		val.push_back(i+1);
	}
	int i;
	for(i=0;i<n;i++)
	{
		int t=matrix[i].size();
		for(int j=0;j<t;j++)
		val[matrix[i][j]]++;
		
		int ans=0;
		for(int j=0;j<t;j++)
		{
		if(matrix[i][j]<m-1&&val[matrix[i][j]]>val[matrix[i][j]+1])
		{
		ans=-1;
		break;
		}
		}
		if(ans!=-1)
		ans=val[m-1]-val[0];
		
		cout<<ans<<"\n";
	    
	    for(int j=0;j<t;j++)
		val[matrix[i][j]]=matrix[i][j];
		
		
	}
	return 0;
}
