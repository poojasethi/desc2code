#include<iostream>
#include<queue>
#include<vector>
#include<cmath>
#include<cstdlib>
using namespace std;
struct chef
{
       int age;
       int rating;
};
class comp1
{
      public:
          bool operator()(chef &c1, chef &c2)
          {
               if(c1.age<c2.age)
                    return true;
               return false;
          }
};
class comp2
{
      public:
          bool operator()(chef &c1, chef &c2)
          {
               if(c1.age>c2.age)
                    return true;
               return false;
          }
};
main()
{
      priority_queue<chef, vector<chef>, comp1> small;
      priority_queue<chef, vector<chef>, comp2> big;
      int n, srat=0, brat=0, a, rat;
      chef temp, ins;
      scanf("%d", &n);
      for(int i=0; i<n; i++)
      {
              scanf("%d%d",  &ins.age, &ins.rating);
              if(i%2==0)
              {
                   if(!big.empty())
                        temp=big.top();
                   if(i==0 || ins.age<temp.age)
                   {
                        small.push(ins);
                        srat+=ins.rating;
                   }
                   else
                   {
                       small.push(temp);
                       srat+=temp.rating;
                       brat-=temp.rating;
                       big.pop();
                       big.push(ins);
                       brat+=ins.rating;
                   }
              }
              else
              {
                   temp=small.top();
                   if(ins.age>temp.age)
                   {
                        big.push(ins);
                        brat+=ins.rating;
                   }
                   else
                   {
                       big.push(temp);
                       brat+=temp.rating;
                       srat-=temp.rating;
                       small.pop();
                       small.push(ins);
                       srat+=ins.rating;
                   }
              }
              printf("%d\n", abs(srat-brat));
      }
}
