#include<iostream>
#include<set>
#include<queue>
#include<cmath>
using namespace std;
typedef pair<int,int> ii;
int main()
{
	int n,a,r,small_n=0,rating1=0,rating2=0;
	scanf("%d",&n);
	set<ii> old;
	priority_queue<ii> young;
	while(n--){
		scanf("%d %d",&a,&r);
		++small_n;
		if(small_n==1){
			young.push(ii(a,r));
			rating1+=r;
			}
		else{
			ii top = young.top();
			if(a<top.first){
				young.push(ii(a,r));
				rating1+=r;
				if(young.size() > (small_n+1)/2){
					ii old_new =  young.top();
					young.pop();
					rating1-=(old_new.second);
					rating2+=(old_new.second);
					old.insert(old_new);
					
				}
			}
			else{
				old.insert(ii(a,r));
				rating2+=r;
				if(old.size() > small_n/2){
					ii young_new = *old.begin();
					old.erase(old.begin());
					rating2-=(young_new.second);
					rating1+=(young_new.second);
					young.push(young_new);
				}
			}
		}	
		if(rating1>rating2)
			printf("%d\n",rating1-rating2);
		else
			printf("%d\n",rating2-rating1);
	}
	return 0;
}
