#include <stdio.h>
#include <set>
using namespace std;

typedef pair<int,int> pii;
typedef set<pii> setii;

int main() {
    
    long N, i, a, r;
    scanf("%d",&N);
    
    long diff = 0;
    setii young, old;
    for (i=1; i<=N; i++) {
        scanf("%d %d",&a,&r);
        if (i%2) {
            if (old.size() == 0 || a < old.begin()->first) {
                young.insert(make_pair(a,r));
                diff += r;
            }
            else {
                pii t = *old.begin();
                old.erase(old.begin());
                old.insert(make_pair(a,r));
                young.insert(t);
                diff += 2*t.second - r;
            }
        }
        else {
            set<pii>::iterator it = young.end(); it--;
            if (a > (*it).first) {
                old.insert(make_pair(a,r));
                diff += -r;
            }
            else {
                it = young.end(); it--;
                pii t = *it;
                young.erase(it);
                young.insert(make_pair(a,r));
                old.insert(t);
                diff += r - 2*t.second;
            }
        }
        printf("%d\n",diff>0?diff:(-diff));
    }
    return 0;
}
