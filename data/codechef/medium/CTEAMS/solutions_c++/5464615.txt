int main()
{
	int n,x=0,y=0;
	cin>>n;
	priority_queue<int> yt;
	priority_queue<int,vector<int>,greater<int> > ot; 
	map<int,int> m;
	REP(i,n)
	{
		int a,r;
		cin>>a>>r;
		m[a]=r;
		
		if(!(i&1))
		{
			if(ot.empty()||a<ot.top())
			{
				yt.push(a);
				x+=r;
//				cout<<"#1.1$"<<endl;
			}
			else 
			{
				int temp;
				temp=ot.top();
				ot.pop();
				y-=m[temp];
				x+=m[temp];
				yt.push(temp);
				ot.push(a);
				y+=r;
//				cout<<"#1.2$"<<endl;
			}
		}
		else
		{
			int temp;
			if(a<yt.top())
			{
				temp=yt.top();
				yt.pop();
				x-=m[temp];
				ot.push(temp);
				y+=m[temp];
				x+=r;
				yt.push(a);
//				cout<<"#2.1$\n";

			}
			else
			{
				ot.push(a);
				y+=r;
//				cout<<"#2.2$\n";
			}
		}
		cout<<abs(x-y)<<endl;
		
	}
