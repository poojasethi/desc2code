#include<iostream>
#include<map>
using namespace std;
int main()
{
    map<int,int> ychef,ochef;
    map<int,int>::iterator iter,iter2;
    int n,i,ychefsum=0,ochefsum=0,age,rating;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>age>>rating;
        if(i%2)     /*add to ychef*/
        {
            if(ychef.begin()==ychef.end())   /*ychef is empty*/
            {
                ychef[age]=rating;
                ychefsum+=rating;
            }
            else
            {
               iter=ochef.begin();
               if(age < iter->first)   /*add to ychef*/
               {
                   ychef[age]=rating;
                   ychefsum+=rating;
               }
               else
               {
                   ochef[age]=rating; /*add to ochef*/
                   ochefsum+=rating;
                   age=iter->first; rating=iter->second; /*remove from ochef*/
                   ochefsum-=rating;
                   ochef.erase(iter);
                   ychef[age]=rating; /*add to ychef*/
                   ychefsum+=rating;
               }
            }
        }
          else
           {
                if(ochef.begin()==ochef.end()) /*ochef is empty*/
                {
                    ochef[age]=rating;
                    ochefsum+=rating;
                }
                else
                {
                    iter=ychef.end(); iter--;
                    if(age>iter->first)    /*add to ochef*/
                    {
                        ochef[age]=rating;
                        ochefsum+=rating;
                    }
                    else
                    {
                        ychef[age]=rating;    /*add to ychef*/
                        ychefsum+=rating;
                        age=iter->first; rating=iter->second;   /*remove from ychef*/
                        ychefsum-=rating;
                        ychef.erase(iter);
                        ochef[age]=rating; /*add to ochef*/
                        ochefsum+=rating;
                    }
                }
           }
        if(ychefsum>ochefsum)
         cout<<ychefsum-ochefsum<<endl;
        else
         cout<<ochefsum-ychefsum<<endl;
    }
return 0;
}
