#include <stdio.h>
#include <set>
using namespace std;

typedef pair<int,int> pii;
typedef set<pii> setii;

inline void scan(long &n) {
    n = 0;
    int ch = getchar();
    while( ch < '0' || ch > '9' ) ch = getchar();
    while( ch >= '0' && ch <= '9' ) n = (n<<3)+(n<<1) + ch-'0', ch = getchar();
}

int main() {
    
    long N, i, a, r;
    scan(N);
    
    long diff = 0;
    setii young, old;
    for (i=1; i<=N; ++i) {
        scan(a); scan(r); 
        if (i%2) {
            if (old.size() == 0 || a < old.begin()->first) {
                young.insert(make_pair(a,r));
                diff += r;
            }
            else {
                pii t = *old.begin();
                old.erase(old.begin());
                old.insert(make_pair(a,r));
                young.insert(t);
                diff += (t.second<<1) - r;
            }
        }
        else {
            pii t = *(--young.end());
            if (a > t.first) {    
                old.insert(make_pair(a,r));
                diff += -r;
            }
            else {                
                young.erase(--young.end());
                young.insert(make_pair(a,r));
                old.insert(t);
                diff += r - (t.second<<1);
            }
        }
        printf("%d\n",diff>0?diff:(-diff));
    }
    return 0;
}
