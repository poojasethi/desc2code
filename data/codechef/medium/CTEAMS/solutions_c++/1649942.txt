#include <cstdio>
#include <cstdlib>

#include<stdio.h>
#include<algorithm>
#include <map>
 //#include<conio.h>
using namespace std;
 
int mabs(int n) {
    return (n < 0 ? -n : n);
}
 
int main() {
    int N;
    scanf("%d",&N);
    int tot = 0;
    int ls = 0;
    map<int,int> yc, oc;
    int nu = 0;
    while(N--) {
        int a, r;
        scanf("%d %d",&a,&r);
        if(nu == 0) {
            ls = r;
            yc[a] = r;
        } else if(nu % 2 == 0) {
            if(a < oc.begin()->first) {
                ls += r;
                yc[a] = r;
            } else {
                ls += oc.begin()->second;
                yc[oc.begin()->first] = oc.begin()->second;
                oc.erase(oc.begin()->first);
                oc[a] = r;
            }
        } else if(nu % 2 == 1) {
            if(a > yc.rbegin()->first) {
                oc[a] = r;
            } else {
                oc[yc.rbegin()->first] = yc.rbegin()->second;
                ls -= yc.rbegin()->second;
                yc.erase(yc.rbegin()->first);
                ls += r;
                yc[a] = r;
            }
        }
        tot += r;
        printf("%d\n",mabs(ls + ls - tot));
        nu++;
    }
    //getch();
    return 0;
}
