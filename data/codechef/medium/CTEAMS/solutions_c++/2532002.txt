#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(0);cin.tie(0);
#define mp make_pair
using namespace std;
#define FOR(i,a,b) for (int i = (a); i <= (b); i++)

typedef pair<int , int > pii;

int main ()
{
priority_queue <pii , vector<pii> , less<pii> >young;
priority_queue <pii , vector<pii> , greater<pii> >old;

int t,i,j,k,n,m,r1,r2,age,rating;

scanf("%d",&n);
r1=0,r2=0;

  FOR(i,0,n-1)
  {
  	scanf("%d %d",&age,&rating);

	 if(young.size() == 0 ){
		young.push(mp(age,rating));
		r1+=rating;}
     else if(young.size() == old.size())
	 {
		if(age > old.top().first)
		{
		   pii ok = old.top();
		   old.pop();
		   r2-=ok.second;

		   young.push(ok);
		   r1+=ok.second;

		   old.push(mp(age,rating));
		   r2+=rating;
		}
		else
			{
			  young.push(mp(age,rating));
		      r1+=rating;
			}
	 }
	 else// young.size = old.size +1
	 {
	    if(age < young.top().first)
		{
		   pii ok = young.top();
		   young.pop();
		   r1-=ok.second;

		   old.push(ok);
		   r2+=ok.second;

		   young.push(mp(age,rating));
		   r1+=rating;
		}
		else
		{
		    old.push(mp(age,rating));
		    r2+=rating;
		}
	 }

   cout<<abs(r1-r2)<<"\n";
  }
 return 0;
}
