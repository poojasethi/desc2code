#include<stdio.h>
#include<algorithm>
#include<math.h>
using namespace std;
struct node{
int x;
int y;
}a[100000];
bool comp(node i,node j)
{
    if(i.x>j.x)
        return false;
    if(i.x==j.x)
    {
        if(i.y<j.y)
            return false;
    }
    return true;
}
int main()
{
    int t,x[100000],y[100000],temp;
    long int n,i,j;
    double d;
    scanf("%d",&t);
    while(t--)
    {
        d=0;
        scanf("%ld",&n);
        for(i=0;i<n;++i)
            scanf("%d%d",&a[i].x,&a[i].y);
        sort(a,a+n,comp);
        for(i=0;i<n-1;++i)
            d+=sqrt(pow(a[i].x-a[i+1].x,2)+pow(a[i].y-a[i+1].y,2));
        printf("%.2lf\n",d);
    }
    return 0;
}
