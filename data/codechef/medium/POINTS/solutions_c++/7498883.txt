#include <cstdio>
#include <string>
#include <algorithm>
#include <memory.h>
#include <cmath>
#include <vector>
using namespace std;
struct point
{
	int x,y;
};
struct minmax
{
	int mi,ma;
};
double dist(int x1, int y1, int x2, int y2)
{
	return sqrt(double((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1)));
}
int main(void)
{
	int t,n,i,x,y,pr;
	double l;
	point a;
	minmax b;
	vector<point> v;
	vector<minmax> v1;
	scanf("%d",&t);
	while (t--)
	{
		v1.clear();
		v.clear();
		b.mi=20000;
		b.ma=-1;
		for (i=0;i<=10000;i++)
			v1.push_back(b);
		scanf("%d",&n);
		for (i=0;i<n;i++)
		{
			scanf("%d%d",&x,&y);
			if (v1[x].ma<y)
				v1[x].ma=y;
			if (v1[x].mi>y)
				v1[x].mi=y;
		}
		x=0;
		l=0;
		while (v1[x].ma==-1)
			x++;
		l+=v1[x].ma-v1[x].mi;
		pr=x;
		x++;
		while (x<=10000)
		{
			while ((x<=10000)&&(v1[x].ma==-1))
				x++;
			if (x<=10000)
			{
				l+=v1[x].ma-v1[x].mi;
				l+=dist(pr,v1[pr].mi,x,v1[x].ma);
				pr=x;
				x++;
			}
		}
		printf("%.2lf\n",l);
	}
	return 0;
}
