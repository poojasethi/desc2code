#include<bits/stdc++.h>
using namespace std;
struct pt
{
    int x,y;
    pt(int a,int b){x=a;y=b;};
};
bool compare(pt p1,pt p2 )
{
    if(p1.x!=p2.x)
    return p1.x<p2.x;
    else return p1.y>p2.y;
}
int main()
{
    int t,t1=0;
    scanf("%d",&t);
    while(t1<t)
    {
        int x,y,n,i;
        double sum;
        vector<pt> arr;
        scanf("%d",&n);
        sum=0;
        for(i=0;i<n;i++)
        {
            scanf("%d %d",&x,&y);
            arr.push_back(pt(x,y));
        }
        sort(arr.begin(),arr.end(),compare);
        for(i=0;i<n-1;i++)
        {
            //cout<<arr[i].x<<" "<<arr[i].y<<endl;
            sum+=(double)sqrt((double)pow(arr[i].x-arr[i+1].x,2)+(double)pow(arr[i].y-arr[i+1].y,2));
            //cout<<"see me"<<sum<<endl;
        }
        printf("%0.2lf\n",sum);
        t1++;
    }
    return 0;
}
