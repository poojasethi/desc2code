#include <bits/stdc++.h>

using namespace std;

struct alpha
{
    int a,b;
};
bool alphacompare(alpha lhs,alpha rhs)
{
    if(lhs.a<rhs.a)
        return true;
    else if(lhs.a==rhs.a)
    {
        if(lhs.b>rhs.b)
            return true;
    }
    return false;
}
int main()
{
    int t,n,i;
    scanf("%d",&t);
    while(t--)
    {
        printf("\n");
        scanf("%d",&n);
        alpha arr[n];
        for(i=0;i<n;i++)
        {
            scanf("%d %d",&arr[i].a,&arr[i].b);
        }
        sort(arr,arr+n,alphacompare);
        double dist=0.0;
        for(i=0;i<n-1;i++)
        {
            dist=dist+sqrt(pow((arr[i+1].a-arr[i].a),2)+pow((arr[i+1].b-arr[i].b),2));
        }
        printf("%.2lf\n",dist);
    }
    return 0;
}
