struct lessKey
{
  bool operator()(int a, int b) const
  {
    return a<b;
  }
};


int main()
{
    int noOfTestCase;
    scanint(noOfTestCase);
    char ch;
    for(int i=0;i<noOfTestCase;++i)
    {
        scanf("%c",&ch);//scanf("%c",&ch);
        int arrLen;
        scanint(arrLen);
        map<int,vector<int> ,lessKey> vectorMap;
        int temp1,temp2;
        for(int j=0;j<arrLen;++j)
        {
            scanint(temp1);scanint(temp2);
            //printf("%d %d",temp1,temp2);
            vectorMap[temp1].push_back(temp2);
        }
        map<int,vector<int>,lessKey>::iterator start=vectorMap.begin();
        map<int,vector<int>,lessKey>::iterator end=vectorMap.end();
        int curX = (*start).first;
        vector<int> vec= (*start).second;
        sort(vec.begin(),vec.end(),greater<int>());
        int curY = vec[vec.size()-1];
        double res=0;
        for(int i=0;i<vec.size()-1;++i)
            res+=double(vec[i])-vec[i+1];
        start++;
        for(;start!=vectorMap.end();++start)
        {
            int tempX = (*start).first;
            vector<int> vec= (*start).second;
            sort(vec.begin(),vec.end(),greater<int>());
            int tempY = vec[vec.size()-1];
            int tempZ = vec[0];
            for(int i=0;i<vec.size()-1;++i)
                res+=double(vec[i])-vec[i+1];
            res+=sqrt(double(tempX-curX)*(tempX-curX) + double(tempZ-curY)*(tempZ-curY));
            curX=tempX;curY=tempY;
        }
        printf("%.2lf\n",res);
        //printf("%d",vectorMap.size());
    }
}

void scanint(int &x)
{
    int flag=0;
    register int c = gc();
    if(c == '-') flag=1;
    x = 0;
    for(; (c<48 || c>57); c = gc());
    for(; c>47 && c<58; c = gc())
    {
        x = (x<<1) + (x<<3) + c - 48;
    }
    if(flag == 1)x=-x;
}
