#include <stdio.h>
#include <math.h>
 
using namespace std;
 
inline void fastRead(int *a)
{
 register char c=0;
 while (c<33) c=getchar_unlocked();
 *a=0;
 while (c>33)
 {
     *a=*a*10+c-'0';
     c=getchar_unlocked();
 }
}
 
int main(){		
	int i, t, n, x, y;
	int yMax[10001];
	int yMin[10001];
	double counter;
	scanf("%i", &t);	
	while(t--){		
		for(i=0;i<=10000;++i){
			yMax[i] = -1;
			yMin[i] = 10001;
		}
		// input
		fastRead(&n);
		//printf("---------%i\n", n);
		for(i=0;i<n;++i){
			fastRead(&x);
			fastRead(&y);
			//scanf("%i %i", &x, &y);
			//printf("%i %i\n", x, y);
			if(y>yMax[x]){
				yMax[x]=y;
			}
			if(y<yMin[x]){
				yMin[x]=y;
			}
		}		
		// Logic
		counter = 0;		
		int sx, sy;		
		int xSlisih, ySlisih;
		for(i=0;i<=10000;++i){
			if(yMax[i]!=-1){
				sx = i;
				counter+=yMax[i]-yMin[i];
				//printf("-- %f\n", counter);
				sy = yMin[i];
				++i;
				break;
			}
		}
		while(i<=10000){
			if(yMax[i]!=-1){
				xSlisih = i-sx;
				ySlisih = yMax[i]-sy;
				counter+= sqrt((xSlisih*xSlisih)+(ySlisih*ySlisih));
				counter+= yMax[i]-yMin[i];
				//printf("++ %f\n", counter);
				sx = i;
				sy = yMin[i];
			}
			++i;
		}
		//printf("== %f\n", counter);
		printf("%.2f\n", floor((counter*100)+0.5)/100);
	}	
	return 0;
}
