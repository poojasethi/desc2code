#include<cstdio>
#include<algorithm>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
typedef struct point{
    int x,y;
}point;
bool comp(const point a,const point b){
    if(a.x<b.x){
        return true;
    }
    else if(a.x==b.x){
        if(a.y>b.y){
            return true;
        }
        else
        {
            return false;
        }
    }
    else
    return false;
}
double dis(int x1,int y1,int x2,int y2){
    return sqrt((((1.0*x2)-(1.0*x1))*((1.0*x2)-(1.0*x1)))+(((1.0*y2)-(1.0*y1))*((1.0*y2)-(1.0*y1))));
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    int n;
    scanf("%d",&n);
    point p[n];
    for(int i=0;i<n;i++){
        scanf("%d %d",&p[i].x,&p[i].y);
    }
    sort(p,p+n,comp);
    int cx=p[0].x;
    int cy=p[0].y;
    double total_dis=0.0f;
    for(int i=1;i<n;i++){
        total_dis+=dis(cx,cy,p[i].x,p[i].y);
        cx=p[i].x;
        cy=p[i].y;
    }
    printf("%.2lf\n",total_dis);
    }
}
