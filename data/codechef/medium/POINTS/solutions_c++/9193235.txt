#include <iostream>
#include <stdio.h>
#include <cmath>
#include <algorithm>
using namespace std;
 
struct point{
    int x;
    int y;
};
 
bool comp(point,point);
 
point a[100001];
 
int main() {
	int t,n,i;
	double ans;
	scanf("%d",&t);
	while(t--){
	    scanf("%d",&n);
	    for(i=0;i<n;i++){
	        scanf("%d%d",&(a[i].x),&(a[i].y));
	    }
	    sort(a,a+n,comp);
	    ans=0;
	    for(i=0;i<n-1;i++){
	        ans+=sqrt(pow(a[i].x-a[i+1].x,2)+pow(a[i].y-a[i+1].y,2));
	    }
	    printf("%.2lf\n",ans);
	}
	return 0;
}
 
bool comp(point a,point b){
    if(a.x==b.x) return a.y>b.y;
    else return a.x<b.x;
}
