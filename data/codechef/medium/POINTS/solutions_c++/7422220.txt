#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define x first
#define y second

bool comp(pair<int,int>A,pair<int,int>B)
{
    if(A.x==B.x)
        return A.y>B.y;
    return A.x<B.x;
}
int main()
{
    int t,n,a,b;
    double dis;
    vector<pair<int,int> >V;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0; i<n; i++)
        {
            cin>>a>>b;
            V.push_back(make_pair(a,b));
        }
        sort(V.begin(),V.end(),comp);
        dis=0;
        for(int i=1; i<V.size(); i++)
            dis+=sqrt((V[i].x-V[i-1].x)*(V[i].x-V[i-1].x)+(V[i].y-V[i-1].y)*(V[i].y-V[i-1].y));
        printf("%.2lf\n",dis);
        V.clear();
    }
    return 0;
}
