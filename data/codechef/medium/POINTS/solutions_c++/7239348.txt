#include <iostream>
#include <cstdio>
#include <vector>
#include <math.h>
#include <algorithm>
#define cin(n) scanf("%d",&n)
#define cout(x) printf("%d",x)
using namespace std;


long mycompare(pair<int,int> v1,pair<int,int> v2)
{
    if(v1.first==v2.first)return v1.second>v2.second;
    return v1.first<v2.first;
}
int main()
{
    vector< pair<int,int> > v;
    int t,i,x,y;
    double result,temp1,temp2;
    long n;
    cin(t);
    while(t--)
    {
        printf("\r\n");
        //printf("\n");
        scanf("%ld",&n);
        v.clear();
        for(i=0;i<n;i++)
        {
            cin(x);
            cin(y);
            v.push_back(make_pair(x,y));
        }
        sort(v.begin(),v.end(),mycompare);
        result=0;
        for(i=1;i<n;i++)
        {
            temp1=(v[i].first-v[i-1].first)*(v[i].first-v[i-1].first);
            temp2=(v[i].second-v[i-1].second)*(v[i].second-v[i-1].second);
            result+= sqrt(temp1+temp2);
        }
        printf("%.2f\n",result);
    }
    return 0;
}
