#include<bits/stdc++.h>
using namespace std;
#define pp pair<double,double>
class comp
{
	public:
		bool operator()(const pp&p1,const pp&p2)
		{
			if(p1.first!=p2.first)
			{
				return(p1.first<p2.first);
			}
			else
			{
				return(p1.second>p2.second);
			}
		}
};
int main()
{ 
	int i,n,t,r;
	char ch;
	double ans,cal;
	vector<pp > v(100000);
	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		scanf("%c",&ch);
		scanf("%c",&ch);
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%lf%lf",&(v[i].first),&(v[i].second));
		}
		sort(v.begin(),v.begin()+n,comp());
		for(i=1;i<n;i++)
		{
			cal=v[i].first-v[i-1].first;
			cal=cal*cal;
			cal+=(v[i].second-v[i-1].second)*(v[i].second-v[i-1].second);
			cal=sqrt(cal);
			ans+=cal;
		}
		printf("%.2lf\n",ans);
	}
	return 0;
}
