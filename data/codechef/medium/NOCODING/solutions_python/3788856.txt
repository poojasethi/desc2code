def string_to_ord(s):
	return map(lambda x: ord(x)-ord('a'), list(s))


def diff25(a,b):
	if a<=b:
		return b-a
	return (26-a)+b


def count_instructions(string):
	insts = 1	#instruction 1 is load the first char
	string = string_to_ord(string)
	for i in xrange(len(string)-1):
		insts += diff25(string[i], string[i+1])
	insts += len(string) #one instruction for each char printed
	return insts


T = int(raw_input())

while T>0:
	T-=1
	word = raw_input()
	insts = count_instructions(word)
	#print(insts)
	if insts <= len(word)*11:
		print("YES")
	else:
		print("NO")
