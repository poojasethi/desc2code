cases = int(raw_input())
for i in xrange(cases):
	word  = raw_input()
	length = len(word)
	basic_steps = length
	if length >=1:
		basic_steps+=1
	extra_steps = 0
	total_steps = 0
	for j in xrange(1,length):
		if word[j]>=word[j-1]:
			extra_steps+=(ord(word[j])-ord(word[j-1]))
			
		else:
			extra_steps+=(26-(ord(word[j-1])-ord(word[j])))
			
	total_steps = basic_steps+extra_steps
	
	if total_steps<=11*length:
		print "YES"
	else:
		print "NO"
