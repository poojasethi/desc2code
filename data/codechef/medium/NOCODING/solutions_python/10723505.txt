assert 1<=t and t<=1000
for i in  range(t):
    list1=[]
    count=0
    string=raw_input()
    for i in string:
        list1.append(ord(i))
    for j in range(0,len(list1)-1):
        if (list1[j+1]>=list1[j]):
            count+=(list1[j+1]-list1[j])
        else:
            count+=(26-list1[j]+list1[j+1])
    count+=1+len(list1)
    if count<=11*len(list1):
        print "YES"
    else:
