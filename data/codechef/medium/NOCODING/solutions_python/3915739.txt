import sys

num = int(sys.stdin.readline().rstrip())
while num != 0:
	word = sys.stdin.readline().rstrip()
	count = 2
	buffer_val = ord(word[0]) - 97
	
	for char in word[1:]:
		val = ord(char) - 97
		if val > buffer_val:
			count += val - buffer_val + 1
		elif val < buffer_val:
			count += val + (25 - buffer_val) + 2
		else:
			count += 1
		buffer_val = val
	
	if count <= 11 * len(word):
		print "YES"
	else:
		print "NO"

	num -= 1
