test=int(raw_input())
d={'a':0,'b':1,'c':2,'d':3,'e':4,'f':5,'g':6,'h':7,'i':8,'j':9,'k':10,'l':11,'m':12,'n':13,'o':14,'p':15,'q':16,'r':17,'s':18,'t':19,'u':20,'v':21,'w':22,'x':23,'y':24,'z':25}
for i in xrange(test):
	s=raw_input()
	c=1				#no of instructions
	for j in range(len(s)-1):
		diff=d[s[j+1]]-d[s[j]]
		if diff>0:
			c+=diff
		elif diff<0:
			c+=(26+diff)
		c+=1
	if c+1<=11*len(s):
		print "YES"
	else:
		print "NO"
