import sys
def calculate(nums):
	x = nums[0]
	count = 0
	for num in nums:
		if(num >= x):
			count += num - x
		else:
			count += 26 + num - x
		x = num
	if(count + len(nums) + 1 <= len(nums) * 11):
		return "YES"
	else:
		return "NO"
 
N = int(sys.stdin.readline().rstrip())
for i in range(0,N):
	n = sys.stdin.readline().rstrip()
	chars = list(n)
	nums = map(ord, chars)
	print calculate(nums)
