#!/usr/bin/python

import sys


def main():
    data = sys.stdin
    nb = int(data.readline())
    for icase in xrange(nb):
        w = map(lambda x: ord(x)-ord('a'), data.readline().strip())
        res = len(w) + 1 # one instr per print plus initial load
        for i in xrange(1, len(w)):
            dif = w[i] - w[i-1]
            if dif < 0:
                dif += 26
            res += dif
        if res <= 11 * len(w):
            print "YES"
        else:
            print "NO"


if __name__ == "__main__":
    main()
