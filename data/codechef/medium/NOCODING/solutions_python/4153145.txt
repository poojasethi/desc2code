n=int(raw_input())
for i in range(0,n):
    cnt=0
    ch=raw_input()
    l=len(ch)
    max=11*l
    for j in range(0,l-1):
        a=ord(ch[j])
        b=ord(ch[j+1])
        if (a>b):
            cnt+=b-a+26
        else:
            cnt+=b-a
    cnt+=l+1
    if cnt<=max:
        print "YES"
    else:
        print "NO"
