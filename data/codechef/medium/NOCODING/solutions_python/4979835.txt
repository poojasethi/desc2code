
test_cases = int(raw_input())

for test_case in range(test_cases):
    string = raw_input()
    prev_char = string[0]
    instruction_count = 1
    for char in string[1:]:
        if prev_char <= char:
            instruction_count += (ord(char) - ord(prev_char))
        else:
            instruction_count += 26 - (ord(prev_char) - ord(char))
            
        prev_char = char

    instruction_count += len(string)

    if instruction_count > len(string) * 11:
        print 'NO'
    else:
        print 'YES'
