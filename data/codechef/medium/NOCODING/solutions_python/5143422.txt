t = int(raw_input())
for i in range(t):
	word = raw_input()
	counter = 2
	word_len = len(word)
	for i in range(1, word_len):
		if ord(word[i]) >= ord(word[i - 1]):
			counter += ord(word[i]) - ord(word[i - 1])
		else:
			counter += 26 + ord(word[i]) - ord(word[i - 1])
		counter += 1
	if counter <= word_len * 11:
		print "YES"
	else:
		print "NO"
