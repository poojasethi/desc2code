import sys

alphabet = list('abcdefghijklmnopqrstuvwxyz')
#print alphabet
t = int(sys.stdin.readline())

for i in range(t):
	s = sys.stdin.readline().strip()
	# load instruction
	numInstructions = 1
	startIndex = alphabet.index(s[0])
	for j in range(1, len(s)):
		currIndex = alphabet.index(s[j])
		#print '{} -> {}'.format(alphabet[startIndex], alphabet[currIndex])
		if currIndex < startIndex:
			increase = 26 - startIndex + currIndex
		else:
			increase = - startIndex + currIndex

		numInstructions += increase
		startIndex = currIndex

	numInstructions += len(s)

	if numInstructions > 11*len(s):
		print 'NO'
	else:
		print 'YES'
