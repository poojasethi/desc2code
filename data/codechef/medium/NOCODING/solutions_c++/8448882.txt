#include<stdio.h>
#include<bits/stdc++.h>
#include<math.h>
using namespace std;
int main()
{
	int t,i,s;
	scanf("%d",&t);
	while(t--)
	{
		char a[10001];
		scanf("%s",a);
		s = 2;
		for(i = 1;i < strlen(a); i++)
		s +=(a[i]-a[i-1]+26)%26+1;
		if(s <= 11*strlen(a)) printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
