#include <iostream>
#include <algorithm>
#include <fstream>
#include <vector>
#include <string>
#include <map>
#include <cstdio>

using namespace std;

int main()
{
	//freopen("i.txt","r",stdin);
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;

		int l=s.size();

		int c=1+l;
		
		for(int i=1;i<l;i++)
		{     
			int a = s[i] - 'a';
			int b= s[i-1] - 'a';

			//cout<<i<<" "<<a<<" "<<b<<"\n";
			if(a>=b)
				c+= (a-b);
			else
				c+= (a-b) + 26;

			
		}
		if(c<=11*l)
			cout<<"YES\n";
		else 
			cout<<"NO\n";
	}
	
}
