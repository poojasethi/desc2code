#include<cstdio>
#include<cstring>
using namespace std;
typedef long long ll;
int Increment_ops(int& x,char c){
    int y=c,r;
    if(y-x>=0){
        r= y-x;
    }
    else
    {
        r=26+y-x;
    }
    x=y;
    return r;
}
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        char str[1005];
        scanf("%s",&str);
        int n=strlen(str);
        int x=str[0],count=2,fail=0;    //Load instruction
        for(int i=1;i<n;i++){
            count+=Increment_ops(x,str[i]);
            count++;
            if(count>11*n){
                fail=1;
                break;
            }
        }
        //printf("%d\n",count);
        if(fail)
        printf("NO\n");
        else
        printf("YES\n");
    }
}
