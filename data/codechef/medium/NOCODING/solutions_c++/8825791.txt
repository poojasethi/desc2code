# include <bits/stdc++.h>
# define MOD 1000000007
# define gc getchar
# define LL long long
# define L long
using namespace std;
template<class T>
void scanint(T &x)
{
    register T c = gc();
    x = 0;
    T neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
int main ()
{
    int t;
    scanint(t);
    while(t--)
    {
        char s[1001];
        scanf("%s",s);
        int ins=1,l=strlen(s),n,p;
        for(int i=1;s[i]!='\0';i++)
        {
            n=s[i]-'a';
            p=s[i-1]-'a';
            if(n>p)
                ins+=(n-p);
            else if(n<p)
                ins+=(26-p+n);
        }
        ins+=l;
        if(ins<=(11*l))
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0 ;
}
