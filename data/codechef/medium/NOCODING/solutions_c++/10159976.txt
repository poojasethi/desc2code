#include<iostream>
#include<string.h>
using namespace std;
int main()
{
 int t,count,i;
 char c[10000];
 cin>>t;
 while(t--)
 {
  count=0;
  cin>>c;
  for(i=0;i<strlen(c)-1;i++)
  {
   if(c[i]<=c[i+1])
    count+= c[i+1]-c[i];
    else
    count+=25-(c[i]-97)+(c[i+1]-97)+1;
   }
   count+=strlen(c)+1;
   if(count<=11*strlen(c))
   cout<<"YES\n";
   else
   cout<<"NO\n" ;
 }
 return 0;
 }
