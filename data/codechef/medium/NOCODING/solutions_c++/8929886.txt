#include<bits/stdc++.h>
using namespace std;
int main()
{
int test;string str;
cin>>test;int t;int i;
int count1;
for(t=0;t<test;t++)
{
cin>>str;
count1=str.size()+1;int val;int  l_val=str[0]-'a';
for(i=1;i<str.size();i++)
{
val=str[i]-'a';
if (val>=l_val)
	count1+=val-l_val;
else
	count1+=26-l_val+val;
//count1+=min(abs(l_val-val),(26-l_val)+val);
l_val=val;
}
//cout<<count1<<" "<<11*str.size()<<" "<<str.size()<<"\n";
	if (count1<=11*str.size())
	cout<<"YES\n";
	else
	cout<<"NO\n";
}

	return 0;
}
