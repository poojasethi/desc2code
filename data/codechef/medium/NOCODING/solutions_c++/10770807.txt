#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int t,n,j,sum,temp;
    char a[1005];
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        sum=0;
        scanf("%s",a);
        n=strlen(a);
        sum+=(n+1);
        for(j=1;j<n;j++)
        {
            temp=int(a[j]-a[j-1]);
            sum+=temp;
            if(int(a[j])<int(a[j-1]))
                sum+=26;
        }
        n*=11;
        if(sum<=n)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
