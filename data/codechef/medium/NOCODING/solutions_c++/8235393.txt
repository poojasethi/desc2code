#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	string s;
//	cout<<(int)(26-('m'-'i'))<<endl;
	scanf("%d",&t);
	while(t--){
		cin>>s;
		int x=s[0]-97,plen=2;//loading &printing first letter
//		cout<<"x= "<<x<<endl;
		for(int i=1;i<s.length();i++){
//	cout<<"x= "<<x<<" s["<<i<<"]= "<<s[i]<<endl;
			if(s[i]==(x+97)){
				plen++;
			}
			else{
				if(x>(s[i]-97))
				plen+=27-abs(s[i]-x-97);
				else
				plen+=abs(s[i]-97-x+1);
				x=s[i]-97;
//				cout<<"x= "<<x<<endl;
	
			}
//			cout<<"plen= "<<p.len<<endl;
		}
//		cout<<"plen= "<<plen<<endl;
		if((s.length()*11)<plen)
			printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
