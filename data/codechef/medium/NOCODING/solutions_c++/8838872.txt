#include<iostream>
using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		int len=s.length();
		long long int score=0;
		for(int i=1;i<s.length();i++){
			int temp=s.at(i)-s.at(i-1);
			if(temp>=0){
				score+=temp;
			}
			else{
				score+=(26+temp);
			}
		}
		(score>=10*len)?cout<<"NO"<<endl:cout<<"YES"<<endl;
	}
	return 0;
}
