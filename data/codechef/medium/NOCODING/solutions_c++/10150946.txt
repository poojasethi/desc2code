#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        char a[1005], t, p;
        scanf("%s", a);
        int l = strlen(a), i, c;
        c = l+1;
        p = a[0];
        for(i=1;i<l;i++)
        {
            t = a[i];
            if(t < p)
                c+= 26 - p + t;
            else
                c+= t - p;
            p = a[i];
        }
        if( c <= 11*l)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
