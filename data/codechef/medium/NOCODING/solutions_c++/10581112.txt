#include<bits/stdc++.h>
using namespace std;
char s[1055];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{

		cin>>s;
		int l=strlen(s);
		int count=2;
		for(int i=1;s[i]!='\0';i++)
		{
		 if(s[i]>=s[i-1])
           count=count+(s[i]-s[i-1]+1);
           else
           count=count+('z'-s[i-1]+s[i]-'a'+1+1);
           if(count>11*l)
           break;
		}
	if(count>11*l)
        cout<<"NO"<<endl;
        else
        cout<<"YES"<<endl;	
		
		
	}
}
