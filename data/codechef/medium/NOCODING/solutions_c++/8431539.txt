#include <iostream>
#include <cmath>
#include <string>
#include <algorithm>
#include <vector>
#include <set>
#include <iterator>
using namespace std;

int main(void){
   
    int t;
    cin>>t;
    while(t--){
        string s;
        cin>>s;
        int counter = 1;
        char x = s[0];
        for(int i=1; i<s.size(); i++){
            int temp = s[i]-x;
            if(temp<0)
                counter+=26+temp;
            if(temp>=0)
                counter+=temp;
            x=s[i];
           
        }
        counter+=s.size();
        if(counter<=11*s.size())
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    
        
    
    
    return 0;
}
