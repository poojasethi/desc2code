#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,c,l,a,b;
    string s;
    string s1 = "abcdefghijklmnopqrstuvwxyz";
    scanf("%d",&t);
    while(t--)
    {
        c=1;
        cin>>s;
        l=s.length();
        for(i=0;i<l-1;i++)
        {
            a=s[i];
            b=s[i+1];
            if(b>=a)
                c+=(b-a);
            else
                c=c+(26-(a-b));
        }
        c+=l;
        if(c<=l*11)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
