#include<iostream>
using namespace std;
int main(){
	int t,i,count;
	cin>>t;
	string s;
	while(t--){
		cin>>s;
		for(i=0,count=2;i<s.length()-1;i++){
			if(s[i+1]<s[i])
			count=count+(25-(s[i]-97)+1)+s[i+1]-97+1;
			else if(s[i+1]>s[i])
			count=count+s[i+1]-s[i]+1;
			else
			count=count+1;
		}
		//cout<<count<<endl;
		if(count<=11*(s.length()))
		cout<<"YES";
		else
		cout<<"NO";
		cout<<endl;
	}
	return 0;
}
