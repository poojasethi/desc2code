#include <iostream>
#include <string>
#include <cmath>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		string s;
		cin>>s;
		int ins=1,i=1,j=0;
		while(s[i]!='\0'){
			ins++;
			if(int(s[i])-int(s[j])>0){
				ins+=int(s[i]-'a')-int(s[j]-'a');
			}else if(int(s[i])-int(s[j])<0){
				ins+=int('z'-s[j])+int(s[i]-'a')+1;
			}
			i++,j++;
		}
		ins++;
		if((s.size())*11>=ins){
			cout<<"YES\n";
		}else{
			cout<<"NO\n";
		}
	}
	return 0;
}
