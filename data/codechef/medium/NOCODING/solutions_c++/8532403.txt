#include <iostream>
#include <cstdio>
#include <math.h>
#include <string.h>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        char str[1001];
        scanf("%s", str);
        int n, a, count = 1, ct;
        a = str[0];
        n = strlen(str);
        for(int i = 1; i < n; i++)
        {
            ct = str[i] - a;
            if(ct < 0)
            {
                ct = 26 + ct;
            }
            count += ct;
            a = str[i];
            //cout<< count<< endl;
        }

        if(count <= n*10)
        {
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }
    return 0;
}
