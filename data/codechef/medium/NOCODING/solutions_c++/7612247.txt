#include <stdio.h>
#include <string.h>
int main()
{
    char name[1001];
    int tc,i,len=0,size=0,count,temp;
    scanf("%d",&tc);
    while(tc--)
    {
        scanf("%s",name);
        count=1;
        len=strlen(name);
        size=len*11;
        count+=len;
        for(i=1;name[i]!='\0';i++)
        {
            if(name[i]==name[i-1]);
            else
            {
                if(name[i]<name[i-1])
                {
                    temp=90-(name[i-1]);
                    temp=temp+name[i]-64;
                }
                else
                {
                    temp=name[i]-name[i-1];
                }
                count=count+temp;
            }
 
        }
        if(count<=size)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
