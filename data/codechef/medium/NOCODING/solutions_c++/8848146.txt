#include<vector>
#include<string>
#include<algorithm>
#include <iostream>
using namespace std;
 
int main() {
   string s;
   int n,i,t;
   cin>>t;
   while(t--)
   {
   int cnt=0;
	cin>>s;
	n=s.length();
 
	for(i=1;i<n;i++)
	{ 
	  if(s[i]>s[i-1])
	  {
	    cnt+=s[i]-s[i-1];
	  }
	  if(s[i]<s[i-1])
	  {
	    cnt+=s[i]-s[i-1];
	    cnt+=26;
	  }
 
	}
	cnt=cnt+n;
	cnt=cnt+1;
	if(cnt>(n*11))
	cout<<"NO"<<endl;
   else
   cout<<"YES"<<endl;
 
	}
	return 0;
}
