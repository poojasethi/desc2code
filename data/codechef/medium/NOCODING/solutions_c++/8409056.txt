#include <iostream>
#include<string.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string ch;
        cin>>ch;
        int instructions=ch.size()+1;
        int i=1;
        while(ch[i]!='\0')
        {
         if(ch[i]!=ch[i-1])
         {
             if((int(ch[i])-97) > (int(ch[i-1])-97))
             {
                 instructions=instructions+((int(ch[i])-97) - (int(ch[i-1])-97) );
             }
             else
             {
                 instructions=instructions+(26-((int(ch[i-1])-97) - (int(ch[i])-97)) );
             }
         }
         i++;
        }
    if(instructions <= 11*ch.size())
        cout<<"YES\n";
    else
        cout<<"NO\n";

    }
    return 0;
}
