#include<bits/stdc++.h>
#include<string.h>

using namespace std;

int main() {
    int i, j, t, c;
    char s[1001];
    cin >> t;
    while(t--) {
        c = 2; i = 1;
        cin >> s;
        j = s[0] - 97;
        while(s[i] != '\0') {
            if(j == s[i] - 97)
                c++;
            else{
                while(j != (s[i] - 97)){
                    j = (j + 1) % 26;
                    c++;
                }
                c++;
            }
            i++;
        }   
        if(c > strlen(s) * 11)
            cout << "NO" << endl;
        else
            cout << "YES" << endl;
    }
    return(0);
}
