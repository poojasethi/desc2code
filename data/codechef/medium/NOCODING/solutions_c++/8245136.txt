#include <iostream>
#include<string.h>
using namespace std;
char s[1005];
int main()
{
    int t,l,i,x;
    cin>>t;
    while(t--)
    {
    cin>>s;
    l=strlen(s);
    x=2;
    for(i=1;s[i]!='\0';i++)
    {
       if(s[i]>=s[i-1])
       x=x+(s[i]-s[i-1]+1);
       else
       x=x+('z'-s[i-1]+s[i]-'a'+1+1);
       if(x>11*l)
       break;
    }
    if(x>11*l)
    cout<<"NO"<<endl;
    else
    cout<<"YES"<<endl;
    }
    return 0;
}
