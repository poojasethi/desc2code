bool isPowerTwo(int n)
{
	while (n!=1)
	{
		if (n%2 != 0)
			return false;
		n = n/2;
	}
	return true;
}

int main()
{
	int t;
	string str;
	cin >> t;
	while(t--)
	{
		cin >> str;
		int l = str.length();
		int i;
		int inst = 2;
		int a,b;
		for (i=1;i<l;i++)
		{
			b = str[i] - 'a';					//a->b
			a = str[i-1] - 'a';	
			//Increment Inst.
			if (a == b)
				inst += 0;				
			else if (a<b)
				inst += b-a;
			else
				inst += (26-a) + b;

			inst++;		//Print
		}
		// cout << inst << endl;
		if (inst <= l*11)
			cout << "YES\n";
		else
			cout << "NO\n";

	}
	return 0;
}
