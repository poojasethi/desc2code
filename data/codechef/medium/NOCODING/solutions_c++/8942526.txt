#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string ar;
        cin>>ar;
        int tsum=0;
        int n=ar.length();
        for(int i=0;i<n-1;i++)
        {
            int y=ar[i+1];
            int x=ar[i];
            if(y>=x)
            {
                tsum=tsum+(y-x);
             
            }
            else
            {
                tsum=tsum+26-x+y;
               
            }
        }
        tsum=tsum+n+1;
       if(tsum<=(11*n)){cout<<"YES\n";}
       else{cout<<"NO\n";}

    }
    return 0;
}
