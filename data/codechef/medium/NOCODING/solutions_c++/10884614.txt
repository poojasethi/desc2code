#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		string s;
		cin >> s;
		int n = s.size();
		int ans = n+1;
		int temp = s[0]-97;
		for (int i = 1; i < n; i++)
		{
			int k = s[i]-97;
			if (k >= temp)
			{
				ans += (k-temp);
				temp = k;
			}
			else
			{
				ans += (25-temp);
				ans += k+1;
				temp = k;
			}
		}
	//	cout << ans << endl;
		if (ans <= (11*n))
			cout << "YES\n";
		else
			cout << "NO\n";
	}
	return 0;
}
