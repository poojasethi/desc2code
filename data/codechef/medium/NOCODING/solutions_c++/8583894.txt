#include<iostream>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    int t;
    unsigned long long int ins,val,i;
    cin>>t;
    while(t--)
    {
        ins=2;
        string arr;
        cin>>arr;
        val=arr[0]-97;
        for(i=1;i<arr.size();i++)
        {
            if((arr[i]-97)>=val)
                ins+=arr[i]-96-val;
            else
                ins+=arr[i]-70-val;
            val=arr[i]-97;
        }
        if(ins<=(11*arr.size()))
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
    return 0;
}
