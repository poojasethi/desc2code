#include<iostream>
using namespace std;
#include<stdio.h>
#include<string.h>
int main()
{
    int t;
    cin>>t;
    char str[1001];
    while(t--)
    {
        scanf("%s",str);
        int len=strlen(str);
        int i, count=len+1;
        for(i=0; str[i+1]!='\0'; i++)
        {
            if(str[i+1]<str[i])
            {
                count=count+26-((int)str[i]-(int)'a')+(int)str[i+1]-(int)'a';
            }
            else if(str[i+1]>=str[i])
            {
                count=count+(int)str[i+1]-(int)str[i];
            }
        }
        if(count<=11*len)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;

    }

    return 0;
}
