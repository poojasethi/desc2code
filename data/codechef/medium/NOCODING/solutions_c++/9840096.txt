#include <iostream>
#include <cstring>
using namespace std;

int main() {
	char a[1009],ch,ch2;
	int t,i,count,ans2,c1,c2,temp;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",a);
		count=2;
		ch=a[0];
		for(i=1;a[i]!='\0';i++)
		{	
			ch2=a[i];
			if(ch2>=ch)
			{
				count+=(ch2-ch);
				count++;
			}
			else
			{
				c1=ch-97+1;
				c2=ch2-97+1;
				temp=26-c1+c2;
				count+=temp;
				count++;
			}
			ch=ch2;
		}
		ans2=strlen(a);
		ans2*=11;
		if(count<=ans2)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
