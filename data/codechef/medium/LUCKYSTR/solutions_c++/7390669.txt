#include <iostream>
using namespace std;

int main() {
	int y,q,flag,j,i,k,n,f,count,l;
	cin>>k>>n;
	string s[k+10],s1[n+10];
	for(i=1;i<=k;i++)
	{
		cin>>s[i];
	}
	for(i=1;i<=n;i++)
	{
		cin>>s1[i];
		if(s1[i].size()>=47)
		 cout<<"Good"<<endl;
		else
		{
			flag=0;
			for(j=1;j<=k;j++)
			{
				int m=0;
				for(q=0;q<s1[i].size();q++)
				{
					if(s1[i][q]==s[j][m])
					{
						 count=0;
						 l=q;
						 for(y=0;y<s[j].size();y++)
						 {
						 	if(s1[i][l]==s[j][y])
						 	{
						 		l++;
						 		count++;
						 	}
						 	else
						 	 break;
						 }
				          if(count==s[j].size())
					       flag++;
		            }
				}
			}
			if(flag==0)
			 cout<<"Bad"<<endl;
			else{
			 cout<<"Good"<<endl;}
		}
	}
	return 0;
}
