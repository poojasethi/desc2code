#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
	int n,k,i,j,l,flag=0;
	char str[51][60];
	char temp[60];
	scanf("%d %d",&k,&n);
	for(i=0;i<k;i++)
	{
		scanf("%s",str[i]);
	}
	
	for(i=0;i<n;i++)
	{
		flag=0;
		scanf("%s",temp);
		l = strlen(temp);
		if(l>=47)
		printf("Good\n");
		else
		{
			for(j=0;j<k;j++)
			{
				if(strstr(temp,str[j])!=NULL)
				{
					flag=1;
					break;
				}
			}
			if(flag==1)
			printf("Good\n");
			else
			printf("Bad\n");
		}
	}
	return 0;
}
