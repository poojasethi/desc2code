#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n,k;
  cin>>k>>n;
  string arr;
  vector<string>vec;
  for(int i=0;i<k;i++)
  {
    cin>>arr;
    vec.push_back(arr);
  }
  while(n--)
  {
    cin>>arr;
    if(arr.length()>=47)
    {
      cout<<"Good"<<endl;
      continue;
    }
    bool flag=false;
     for(int i=0;i<k;i++)
     {
       if(arr.find(vec[i])!=string::npos)
       {
         cout<<"Good"<<endl;
         flag=true;
         break;
       }
    }
    if(!flag)
    {
      cout<<"Bad"<<endl;
    }
  }
  return 0;
  }
