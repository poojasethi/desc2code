#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int main()
{
	int k, n;
	scanf("%d %d", &k, &n);
	char a[k][51];
	for(int i=0; i<k; i++)
		scanf("%s", a[i]);
	char b[51];
	while(n--)
	{
		scanf("%s", b);
		int l = strlen(b);
		if( l>=47 )
			printf("Good \n");
		else
		{
			bool flag=false;
			int l = strlen(b);
			for(int i=0; i<k; i++)
				if( string(b).find(a[i])!=string::npos )
				{
					flag=true;
					break;
				}
			if( flag )
				printf("Good \n");
			else
				printf("Bad \n");
		}
	}
	return 0;
}
