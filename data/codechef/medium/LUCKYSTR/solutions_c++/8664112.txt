#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main()
{
    int k,n,i,j;
    scanf("%d %d",&k,&n);
    char a[k][55],b[55];
    for(i=0;i<k;i++)
    {
        scanf(" %[^\n]s",a[i]);
    }
    for(i=0;i<n;i++)
    {
        scanf(" %[^\n]s",b);
        if(strlen(b)>=47)
            printf("Good\n");
        else
        {
            for(j=0;j<k;j++)
            {
                if(!strstr(b,a[j]))
                    continue;
                else
                    break;
            }
            if(j<k)
                printf("Good\n");
            else
                printf("Bad\n");
        }
    }

    return 0;
}
