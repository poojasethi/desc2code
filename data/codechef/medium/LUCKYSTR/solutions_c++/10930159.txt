#include <bits/stdc++.h>
using namespace std;

int sub(string s1, string s2){
    
    size_t found;
    found=s1.find(s2);
    if(found!=string::npos)return 1;
    return 0;
    
}


int main(){
    
    string a[52],s1;
    int t,k,flag,x,i;
    cin>>t>>k;
    for(i=0;i<t;i++)cin>>a[i];
    while(k--)
    {
        cin>>s1;
        x=s1.length();
        if(x>=47)cout<<"Good"<<endl;
        else {
            flag=0;
        for(i=0;i<t;i++)if(sub(s1,a[i])){flag=1;break;}
            if(flag)cout<<"Good"<<endl;
            else cout<<"Bad"<<endl;
            
        }
    }
    
    
    
    
    return 0;
    
}
