#include <bits/stdc++.h>
 
using namespace std;
 
int main()
{
    int k, n, i, j, flag;
 
    cin >> k >> n;
 
    char a[k][60], b[n][60];
 
    for (i = 0; i < k; i++)
        cin >> a[i];
 
    for (i = 0; i  < n; i++)
        cin >> b[i];
 
    for (i = 0; i < n; i++) {
        if (strlen(b[i]) > 46)
            flag = 1;
 
        else {
            flag = 0;
            for (j = 0; j < k; j++)
                if (strstr(b[i], a[j]) != NULL) {
                    flag = 1;
                    break;
                }
        }
 
        if (flag)
            cout << "Good\n";
 
        else
            cout << "Bad\n";
    }
 
    return 0;
}
