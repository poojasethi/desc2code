bool is_substring(char A[],char B[]){ //is string A is substring of string B??

    int n1=strlen(A);
    int n2=strlen(B);
    for(int i=0;i<n2;i++){
        if(B[i]==A[0]){
            if(n2-i>=n1){
            int fail=0;
            for(int j=i;j<n1+i;j++){
                if(A[j-i]!=B[j]){
                    fail=1;
                    break;
                }
            }
            if(!fail){

            }
        }

}
