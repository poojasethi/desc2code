#include <iostream>
#include <cstdio>
#include <string.h>

using namespace std;

int main()
{
    int k, n;
    scanf("%d %d", &k, &n);
    char lucky[k][51], str[51];
    for(int i = 0; i < k; i++)
    {
        scanf("%s", lucky[i]);
    }
    while(n--)
    {
        scanf("%s", str);
        int flag = 0;
        if(strlen(str) >= 47)
        {
            printf("Good\n");
            continue;
        }
        else
        {
            for(int i = 0; i < k; i++)
            {
                if(strstr(str, lucky[i]) != NULL)
                {
                    printf("Good\n");
                    flag = 1;
                    break;
                }
            }
            if(flag == 0)
            {
                printf("Bad\n");
            }
        }
    }
    return 0;
}
