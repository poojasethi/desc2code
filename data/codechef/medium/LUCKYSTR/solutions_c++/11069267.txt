#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int k,n,i;
	cin>>k>>n;
	string k1[k],n1;
	for(i=0;i<k;i++)
	{
	cin>>k1[i];
	}
	while(n--)
	{
		cin>>n1;
		int flag=0;
		if(n1.length()>=47)
		{
			cout<<"Good"<<endl;
		}
		else
		{
		for(i=0;i<k;i++)
		{
			size_t p=n1.find(k1[i]);
			if(p!=string::npos)
			{
				flag=1;
				break;
			}
			
		}
		if(flag==1)
		cout<<"Good"<<endl;
		else
		cout<<"Bad"<<endl;
	}
			
	}
	
	
	return 0;
}
