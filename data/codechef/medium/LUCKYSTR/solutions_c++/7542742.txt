#include<iostream>
#include<string>
#include<cstring>
using namespace std;
int main(){
	int k,n;
	cin>>k>>n;
	char c[k][50];
	for(int i=0;i<k;i++)
	cin>>c[i];
	while(n--){
		string s;
		cin>>s;
		if(s.length()>=47){
			cout<<"Good"<<endl;
			continue;
		}
		else{
			int p;
			for(int i=0;i<k;i++){
				 p=s.find(c[i]);
				if(p>=0){
					cout<<"Good"<<endl;
					break;
				}
			}
			if(p<0){
				cout<<"Bad"<<endl;
			}
		}
	}
}
