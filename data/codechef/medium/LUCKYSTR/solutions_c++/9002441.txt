#include<iostream>
#include<string>

using namespace std;

string sk[50],sn[50]; 

int main()
{
int k,n,i,j;
std::size_t found;
cin>>k>>n;
for(i=0;i<k;i++)
cin>>sk[i];
for(i=0;i<n;i++)
cin>>sn[i];

for(i=0;i<n;i++)
{
for(j=0;j<k;j++)
{found=sn[i].find(sk[j]);
if(found!=std::string::npos)
break;
}
if(j<k || sn[i].length()>=47)
cout<<"Good\n";
else
cout<<"Bad\n";
}

return 0;
}
