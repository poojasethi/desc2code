#include<bits/stdc++.h>
using namespace std;
int main()
{
	int k,n,i,flag,j,l;
	cin>>k>>n;
	string s1[k];
	string s2;
	for(i=0;i<k;i++)
	cin>>s1[i];
	for(i=0;i<n;i++)
	{
		flag=0;
		cin>>s2;
		l=s2.size();
		if(l>=47)
		{
			cout<<"Good"<<endl;
			continue;
		}
		for(j=0;j<k;j++)
		{
			if(s2.find(s1[j])!=-1)
			{
				flag=1;
				cout<<"Good"<<endl;
				break;
			}
		}
		if(flag==0)
		cout<<"Bad"<<endl;
	}
	return 0;
}
