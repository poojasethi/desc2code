#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
#define endll "\n";
int main()
{
	ios_base::sync_with_stdio(false);
  int k,n;
  cin>>k;
  cin>>n;
	vector<string> lucky_strings;
	string tmp;
	for(int i=0;i<k;i++){
		cin>>tmp;
		lucky_strings.push_back(tmp);
	}
	for(int i=0;i<n;i++){
		cin>>tmp;
		bool good=false;
		if(tmp.size()>=47){
			//good
			good=true;
		}
		else{
			for(int j=0;j<k;j++){
				if(tmp.find(lucky_strings[j])!=-1){
					//good
					good=true;
				}
			}
		}
		if(good){
			cout<<"Good"<<endll;
		}
		else{
			cout<<"Bad"<<endll;
		}
	}
	return 0;
}
