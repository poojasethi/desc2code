#include<stdio.h>
#include<string.h>
int main(){
int numLuckyStr, numPillowStr;
char luckyStr[51][51];
char pillowStr[51][51];
char *ptr;
int i,j,flag;
scanf("%d%d",&numLuckyStr, &numPillowStr);
for(i=0;i<numLuckyStr;i++){
scanf("%s", luckyStr[i]);
}
for(i = 0;i<numPillowStr;i++){
scanf("%s",pillowStr[i]);
}
for(i=0;i<numPillowStr;i++){
flag = 0;
if(strlen(pillowStr[i]) >= 47){
printf("Good\n");
continue;
}
else{
for(j=0;j<numLuckyStr;j++){
ptr = strstr(pillowStr[i],luckyStr[j]);
if(ptr!=NULL){
flag = 1;
break;
}
}
}
if(flag == 1){
printf("Good\n");
}
else{
printf("Bad\n");
}
}
return 0;
}
