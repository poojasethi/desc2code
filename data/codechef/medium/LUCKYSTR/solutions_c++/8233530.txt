#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,k;
	scanf("%d %d",&k,&n);
	string a[k],b[n];
	for(int i=0;i<k;i++){
		cin>>a[i];
	}
	for(int i=0;i<n;i++){
		cin>>b[i];
	}
	
	for(int i=0;i<n;i++){
		int flag=0;
		for(int o=0;o<b[i].length();o++){
			if(b[i][o]!='7' && b[i][o]!='4' ){
				printf("Bad\n");
				flag=1;
				break;
			}
		}
		if(flag==1)
			continue;
		if(b[i].length()>=47){
			printf("Good\n");
		}
		else{
			for(int j=0;j<k;j++){
				if(b[i].find(a[j])!=-1){
					printf("Good\n");
					break;
				}
				if(j==k-1){
					printf("Bad\n");
				}
			}
		}
	}
	return 0;
}
