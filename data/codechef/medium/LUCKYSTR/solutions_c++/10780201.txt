#include<iostream>
#include<string>
using namespace std;
int main(){
	int k,n,i,j,count=0;
	cin>>k>>n;
	string a[k],s;
	for(i=0;i<k;i++)
	cin>>a[i];
	for(i=0;i<n;i++){
		cin>>s;
		if(s.length()>=47)
		cout<<"Good";
		else{
			int found;
			for(j=0,count=0;j<k;j++){
				found=s.find(a[j]);
				if(found!=string::npos){
					count=10;
					break;
				}
			}
			if(count==10)
			cout<<"Good";
			else
			cout<<"Bad";
		}
		cout<<endl;
	}
return 0;
}
