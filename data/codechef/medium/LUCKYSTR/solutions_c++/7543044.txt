// URL : https://www.codechef.com/problems/LUCKYSTR

#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;

#define M 55

char lucky_str[M][M];
char tmp[M];

bool check(char ch[])
{
	int sz = strlen(ch);
	for(int i = 0; i < sz; ++i)
	{
		if(ch[i] != '4' && ch[i] != '7')
			return false;
	}
	return true;
}

int main()
{	
	int n, k;
	scanf("%d%d",&n,&k);
	
	for(int i = 0; i < n; ++i)
		scanf("%s", lucky_str[i]);
	
	for(int i = 0; i < k; ++i)
	{
		scanf("%s",tmp);
		bool flag = false;
		if(check(tmp))
		{
			if(strlen(tmp) >= 47)
			{
				printf("Good\n");
				continue;
			}
			for(int j = 0; j < n; ++j)
				if(strstr(tmp, lucky_str[j]))
				{
					printf("Good\n");
					flag = true;
					break;
				}
		}
		if(!flag)
			printf("Bad\n");
	}
	
	return 0;
}
