import sys

line = [int(s) for s in sys.stdin.readline().rstrip().split(' ')]
num_lucky = line[0]
num = line[1]
lucky_str = []
while num_lucky != 0:
	lucky_str.append(sys.stdin.readline().rstrip())
	num_lucky -= 1

while num != 0:
	flag = 0
	test = sys.stdin.readline().rstrip()
	if len(test) < 47:
		for item in lucky_str:
			if item in test:
				flag = 1
				break
		if flag:
			print "Good"
		else:
			print "Bad"
		
	else:
		print "Good"
	num -= 1
