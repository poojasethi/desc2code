import sys

(k,n)=map(int,sys.stdin.readline().split())
test=0

luck_strings=[]
for i in range(k):
    luck_strings.append(sys.stdin.readline().strip())
for i in range(n):
    val=sys.stdin.readline().strip()
    if len(val)>=47:
        print 'Good'
    else:
        for j in luck_strings:
            test=1
            if j in val:
                print 'Good'
                test=0
                break
        if test:
            print 'Bad'
