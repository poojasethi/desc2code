#codechef - little elephant and strings

k,n = map(int,raw_input().split())
lucky_strings = []
all_stings = []

for _ in range(k):
    lucky_strings.append(raw_input())
    

for _ in range(n):
    
    string = raw_input()
    
    if len(string)>=47:
        print 'Good'
    else:
        result = False
        for lucky in lucky_strings:
            if lucky in string:
                result = True
                break
        print 'Good' if result else 'Bad'
