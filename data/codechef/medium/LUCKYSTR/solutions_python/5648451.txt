k,n = map(int, raw_input().split())
l=[]
for i in range(k):
    l.append(raw_input())
for i in range(n):
    s = raw_input()
    if len(s)>=47:
        print "Good"
    else:
        good = False
        for i in range(k):
            if l[i] in s:
                good = True
                break
        print "Good" if good else "Bad"
