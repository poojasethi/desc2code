k,n=map(int,raw_input().split())
tmp =k
st=[]
while tmp:
  st.append(raw_input().rstrip())
  tmp -=1
while n:
    inp = raw_input().rstrip()
    f=0
    if(len(inp)>=47):
        print 'Good'
    else:
        for i in xrange(0,k):
            if(inp.find(st[i])!=-1):
                f=1
                print 'Good'
                break
        if(not f):
            print 'Bad'
    n -=1
