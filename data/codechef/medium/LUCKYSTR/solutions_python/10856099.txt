k ,n = map(int,raw_input().split())
lis = []
for i in xrange(k):
	lis.append(raw_input())
#print lis
for i in xrange(n):
	a = raw_input()
	if len(a)>=47:
		print "Good"
	else:
		cou = 0
		for i in xrange(k):
			if lis[i] in a:
				print "Good"
				break
			else:
				cou+=1
		if cou == k:
			print "Bad"
