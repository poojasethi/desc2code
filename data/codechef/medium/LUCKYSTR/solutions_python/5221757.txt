inp = map(int,raw_input().split())
K,N = inp[0],inp[1]
LUK = []
for i in range(K):
    LUK.append(raw_input())
ILUK = map(int,LUK)
for j in range(N):
    st = raw_input()
    luk = False
    if len(st) >= 47:
        print 'Good'
        continue
    for lks in LUK:
        if st.find(lks) != -1:
            luk = True
            break
    if luk:
        print 'Good'
    else:
        print 'Bad'
