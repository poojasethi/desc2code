m=raw_input().split();
M,N=map(int,m)
Mlist=[""]*M;
Nlist=[""]*N;
for i in range(M):
	Mlist[i]=raw_input();
for i in range(N):
	Nlist[i]=raw_input();
	
for i in range(N):
	if len(Nlist[i])>=47:
		print "Good"
	else:
		flag=0;
		for j in range(M):
			if Mlist[j] in Nlist[i]:
				print "Good"
				flag=1
				break;
		if flag==0:
			print "Bad"
