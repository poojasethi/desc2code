t=raw_input()
lx=t.split()
k=int(lx[0])
n=int(lx[1])
fav=[]
for i in range(0,k):
    s=raw_input()
    fav.append(s)

for i in range(n):
    s=raw_input()
    fl=0
    if(len(s)<47):
        for j in range(k):
            le=len(fav[j])
            for jjj in range(0,len(s)-le+1):
                if(fav[j]==s[jjj:jjj+le]):
                    fl=1
                    break
    if(fl==1 or len(s)>=47):
        print "Good"
    else:
        print "Bad"
