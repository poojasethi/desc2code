k, n = map(int, raw_input().split())
fav = list()
for i in xrange(k):
	fav.append(raw_input())
for i in xrange(n):
	string = raw_input()
	if len(string) > 46:
		print 'Good'
	else:
		found = False
		for item in fav:
			if item in string:
				found = True
				break
		print 'Good' if found else 'Bad'
