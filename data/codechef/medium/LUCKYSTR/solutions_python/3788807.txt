

K, N = map(int, raw_input().split())

favorites = []

while K>0:
	K-=1
	favorites.append(raw_input())

favorites.sort(lambda a,b: cmp(len(a), len(b)))
K = len(favorites)

while N>0:
	N-=1
	S = raw_input()
	if len(S) >= 47:
		print("Good")
	else:
		found = False
		for i in xrange(K):
			if S.find(favorites[i]) > -1:
				print("Good")
				found = True
				break
		if not found:
			print("Bad")
