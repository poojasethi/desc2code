n,k=map(int,raw_input().split())
a=[]
for i in range(n):
    a.append(raw_input())
for i in range(k):
    y=raw_input()
    if len(y)>=47:
        print "Good"
    else:
        f=0
        for i in a:
            if i in y:
                f=1
                break
        if f:
            print "Good"
        else:
            print "Bad"
