k,n = map(int, raw_input().split())

s1 = []

for _ in xrange(k):
    s1.append(raw_input())
for _ in xrange(n):
    s= raw_input()


    if len(s) >= 47:
        print 'Good'
    else:
        r = False
        for j in xrange(len(s1)):
            if s1[j] in s:
                r = True
                
                break
                
        if r:
            print 'Good'
        else:
            print 'Bad'
