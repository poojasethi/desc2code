#!/usr/bin/env python

def get_if_good_or_bad(lucky_strings, string):
  if len(string) >= 47:
    return "Good"
  for i in lucky_strings:
    if i in string:
      return "Good"
  return "Bad"

def main():
  input_nos = raw_input().split()
  no_of_lucky_strings = int(input_nos[0])
  no_of_strings = int(input_nos[1])
  i = 0
  lucky_strings = []
  while i < no_of_lucky_strings:
    lucky_strings.append(raw_input())
    i += 1
  i = 0
  strings = []
  while i < no_of_strings:
    strings.append(raw_input())
    i += 1
  for string in strings:
    print get_if_good_or_bad(lucky_strings, string)

if __name__ == '__main__':
  main()
