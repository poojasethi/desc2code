k,n=map(int,raw_input().split())
fav=[]
while k>0:
    k=k-1
    fav.append(raw_input().rstrip())
while n:
    found="Bad"
    n-=1
    str=raw_input().rstrip();
    if(len(str)>=47):
        found="Good"
    else:
        for favstr in fav:
            if(favstr in str):
                found="Good"
                break
    print(found)
