k,n=map(int,raw_input().split())
arr={}
for i in range(k):
	arr[i]=raw_input()
for j in range (n):
	m=raw_input()
	b=len(m)
	if b>=47:
		print "Good"
	else:
		i=0
		found=False
		while found!=True and i<k:
			if arr[i] in m:
				found=True
			i+=1
		if found==True:
			print "Good"
		elif found==False:
			print "Bad"
