import sys
f = sys.stdin
T = int(f.readline())
while T:
    N,K = [int(x) for x in f.readline().split()]
    A   = [int(x) for x in f.readline().split()]
    B   = [int(x) for x in f.readline().split()]
    for i in range(0,N) :
        A[i] = K/A[i]
    ans = 0 
    for i in range(0,N) :
        ans = max(ans,A[i]*B[i])
    print ans 
    T -= 1
