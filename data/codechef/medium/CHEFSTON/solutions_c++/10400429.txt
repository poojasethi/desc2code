#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int N, K;
		scanf("%d%d",&N,&K);
		int* A = new int[N];
		int* B = new int[N];
		for(int i = 0; i < N; i++)
			scanf("%d",&A[i]);
		for(int i = 0; i < N; i++)
			scanf("%d",&B[i]);
		ll ans = 0;
		for(int i = 0; i < N; i++)
		{
			ll profit = (K / A[i]) * (ll)B[i];
			ans = max(ans, profit);
		}
		printf("%lld\n",ans);	
	}
	return 0;
}
