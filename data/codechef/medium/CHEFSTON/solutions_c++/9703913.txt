#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    for(scanf("%d", &t); t--; ){
        int n;
        long long k;
        scanf("%d %lld", &n, &k);
        long long a[100001];
        for(int i=0; i<n; i++)
            scanf("%lld", &a[i]);
        long long maxi=0, b;
        for(int i=0; i<n; i++){
            scanf("%lld", &b);
            long long temp = (long)(k/a[i])*b;
            if(temp>maxi)
                maxi=temp;
        }
        printf("%lld\n", maxi);
    }
    return 0;
}
