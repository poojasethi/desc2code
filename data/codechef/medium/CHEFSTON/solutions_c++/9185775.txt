#include<bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("in","r",stdin);
	int t;
	cin>>t;
	while(t--)
	{
		long long n,k;
		cin>>n>>k;
		long long a[n],b[n];
		for(long long i=0;i<n;i++)
			cin>>a[i];
		for(long long i=0;i<n;i++)
			cin>>b[i];
		long long mx=0;
		for(long long i=0;i<n;i++)
		{
			mx=max(mx,(k/a[i])*b[i]);
		}
		cout<<mx<<endl;
	}
}
