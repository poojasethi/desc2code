#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<cstring>
using namespace std;
int t[100000];
int v[100000];
int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	int T,K,N;
	long long ans;
	cin>>T;
	while(T--)
	{
		cin>>N>>K;
		for(int i=0;i<N;++i)
			cin>>t[i];
		for(int i=0;i<N;++i)
			cin>>v[i];
		#define count(X) (K/t[X])*(long long)v[X]
		ans = count(0);
		for(int i=1;i<N;++i)
			ans = max( ans , count(i) );
		cout<<ans<<'\n';
	}
	return 0;
}
