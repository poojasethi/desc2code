#include<bits/stdc++.h>
using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
int t,n;
long int A[100001],B[100001];
long int k;
cin>>t;
while(t--)
{
	cin>>n>>k;
	for(int i=0;i<n;i++)cin>>A[i];
	for(int i=0;i<n;i++)cin>>B[i];
long long max=-1;
	for(int i=0;i<n;i++)
	{
		long long j ;
		j = k;
		j/=A[i];
		j*=B[i];
		if(max<j)
		{
			max=j;
		}
	}
	cout<<max<<endl;
}
}
