#include<iostream>
#include<math.h>
using namespace std;
int main()

{
 int T;
 cin>>T;
 for(int i=0;i<T;i++)
   {
   	long long int N,K;
   	cin>>N>>K;
   	long long int A[N+1],B[N+1];
   	for(long long int j=0;j<N;j++)
   	 cin>>A[j];
   	for(long long int k=0;k<N;k++)
   	 cin>>B[k];
   	long long int ans=-1;
   	for(long long int m=0;m<N;m++)
     ans = max(ans, (K/A[m])*B[m]);
	cout<<ans<<endl;
   }
}
