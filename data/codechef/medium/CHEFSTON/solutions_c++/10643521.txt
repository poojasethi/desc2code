 #include<bits/stdc++.h>
    #define gc getchar_unlocked
    #define pc putchar_unlocked
    #define p(n) printf("%llu\n",n);
    #define s(n) scanf("%li",&n);
    #define S(n) scanf("%llu",&n);
    using namespace std;
     inline long long  int fastread()

        {
          unsigned int i=0;
          char c=gc();
          while(c<'0' || c>'9')
            c=gc();
          for(;c>='0' && c<='9';c=gc())
            i=(i<<3)+(i<<1)+c-'0';
          return i;
        }
    int main()
    {
        int t;
        t=fastread();//scanf("%d",&t);
        while(t--)
        {
           long int n;
           long long k;
            n=fastread();
            k=fastread();//s(n)S(k)
            long long  b;
            long long  a[n];
            for(int i=0;i<n;i++)
            {
                b=fastread();//S(b);
                a[i]=k/b;
            }
            long long max=0;
            for(int i=0;i<n;i++)
            {
                b=fastread();//s(b)
                if(b*a[i]>max)
                    max=b*a[i];
            }
            p(max)
        }
        return 0;
    }
