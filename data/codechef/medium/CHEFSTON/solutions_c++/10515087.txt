#include<bits/stdc++.h>
#define lli long long int 
#define ld long double 
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		lli k;
		scanf("%d %lli",&n,&k);
		lli time[n];
		lli profit[n];
		for(int i=0;i<n;i++)
			scanf("%lli",&time[i]);
		lli max_val=0;
		for(int i=0;i<n;i++){
			scanf("%lli",&profit[i]);
			if(max_val<((k/time[i])*profit[i]))
				max_val=((k/time[i])*profit[i]);
		}
		printf("%lli\n",max_val);
		
		
	}
	return 0;
}
