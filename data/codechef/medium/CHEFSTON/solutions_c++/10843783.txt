#include <iostream>
#include <bits/stdc++.h>

typedef long long int ll;
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n,k,ans,mx=0;
		cin>>n>>k;
		vector<ll>v(n);
		for(ll i=0;i<n;i++)
		{
			ll a;
			cin>>a;
			v[i]=a;
		}
		for(ll i=0;i<n;i++)
		{
			ll a;
			cin>>a;
			ans=k/v[i] * a;
			if(ans>mx) mx = ans;
		}
		cout<<mx<<endl;
	}
	return 0;
}
