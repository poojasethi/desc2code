#include<bits/stdc++.h>
using namespace std ;
int main () {
    int t , n ;
    scanf ( "%d" , &t ) ;
    long long a[100003] , b[100003] , k ;
    while ( t-- ) {
        scanf ( "%d %lld" ,&n , &k ) ;
        int i ;
        for ( i = 0 ; i < n ; i++ ) {
            scanf ( "%lld" , &a[i] ) ;
        }
        for ( i = 0 ; i < n ; i++ ) {
            scanf ( "%lld" , &b[i] ) ;
        }
        long long cnt = k/a[0] ;
        long long p = cnt*b[0] ;
        for ( i = 1 ; i < n ; i++ ) {
            cnt = k / a[i] ;
            p = max ( p , cnt * b[i] ) ;
        }
        printf ( "%lld\n" , p ) ;
    }
    return 0 ;
}
