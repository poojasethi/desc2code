int main(){

    T {
        ll n,k;
        cin>>n>>k;

        ll a[n],b[n];

        for(ll i=0;i<n;i++) cin>>a[i];
        for(ll i=0;i<n;i++) cin>>b[i];

        for(ll i=0;i<n;i++){

            }
        }
        cout<<ans<<endl;
    }
