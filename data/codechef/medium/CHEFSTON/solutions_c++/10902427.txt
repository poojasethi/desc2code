#include<cstdio>
using namespace std;
typedef long long int ll;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k;
        scanf("%d %d",&n,&k);
        ll A[n],B[n];
        for(int i=0;i<n;i++){
            scanf("%lld",&A[i]);
        }
        for(int i=0;i<n;i++){
            scanf("%lld",&B[i]);
        }
        ll max=0;
        for(int i=0;i<n;i++){
           ll tmp=((k/A[i])*B[i]);
            if(tmp>max){
                max=tmp;
            }
        }
        printf("%lld\n",max);
    }
}
