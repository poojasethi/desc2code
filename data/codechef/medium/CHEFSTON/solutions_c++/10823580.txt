#include <iostream>
using namespace std;
#define ll long long int

int main() {
	// your code goes here
	ll t,n,k,profit[100007];
	ll a[100007],b[100007];
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(ll i=0;i<n;i++)
		cin>>a[i];
		for(ll j=0;j<n;j++)
		cin>>b[j];
		for(ll f=0;f<n;f++)
		{
		   profit[f]=(k/a[f])*b[f];  	
		}
		ll max=profit[0];
		for(int l=1;l<n;l++)
		{
			if(profit[l]>max)
			max=profit[l];
		}
		cout<<max<<"\n";
	}
	return 0;
}
