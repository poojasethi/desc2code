#include <iostream>
#include <algorithm>
using namespace std;
#define lli long long int
int main() {
	// your code goes here
	lli t;
	cin>>t;
	while(t--){
		lli n,k,ans=-1110,ss;
		cin>>n>>k;
		lli a[n+1],b[n+1];
		for(int i=1;i<=n;i++){
			cin>>a[i];
		}
		for(int i=1;i<=n;i++){
			cin>>b[i];
		}
		for(int i=1;i<=n;i++){
			ss=k/a[i];
			ss=ss*b[i];
			ans=max(ans,ss);
		}
		cout<<ans<<"\n";
	}
	return 0;
}
