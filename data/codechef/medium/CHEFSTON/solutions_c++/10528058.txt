#include<iostream>
#include<math.h>
using namespace std;

int main()
{
	int T,K,N,*a,*b,i;
	cin>>T;
	unsigned long long int temp,max; 
	
	while(T--)
	{
		max=0;
		cin>>N>>K;
		a=new int[N];
		b=new int[N];
		for(i=0;i<N;i++)
		{
			cin>>a[i];
		}
		for(i=0;i<N;i++)
		{
			cin>>b[i];
			temp=(unsigned long long)floor((K*1.00)/a[i])*(unsigned long long)b[i];
			//cout<<temp;
			if(temp>max)max=temp;
		}
		cout<<max<<endl;
		
		
	}
	return 0;
}
