#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	assert(t >=1 && t <= 5);
	
	while(t--)
	{
		long long int n,k;
		cin >> n >> k;

		long long int a[n],i;
		for(i = 0;i < n;i++) 
		{
			cin >> a[i];
		}	

		long long int b[n];

		for(i = 0;i < n;i++)
		{
			cin >> b[i];
		}

		long long int curr;
		long long int p[n];
		for(i = 0;i < n;i++) {
			p[i] = 0;
		}

		for(i = 0;i < n;i++)
		{
			curr = k;
			while(curr >= a[i]) {
					p[i] += b[i];
					curr = curr - a[i];
					//cout << p[i] << " " << curr << "\n";
			}
		}

		sort(p,p+n);
		cout << p[n-1] << "\n";
	}

	return 0;
}
