#include <bits/stdc++.h>
using namespace std;
string s;
bool solve(int l,int r,int c)   {
  if(l >= r)
    return true;
    if(r-l == 1)  {
      if(s[l]  == s[r])
      return true;
      if(c == 0)
      return true;
      }
   if(s[l] == s[r])
   return solve(l+1,r-1,c);
   if(c == 0) 
     return solve(l+1,r,1)|solve(l,r-1,1);
   return false;
 }
int main() {
	int t;
	cin >> t;
	while(t--)   {
	cin >> s;
	bool ans;
	ans=solve(0,s.size()-1,0);
	if(ans)
	  cout << "YES" << endl;
	else
	  cout << "NO" << endl;
	}
}
