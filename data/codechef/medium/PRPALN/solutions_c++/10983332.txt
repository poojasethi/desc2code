#include <bits/stdc++.h>
using namespace std;
string s;
int solve(int l,int r)
{
	while(l<=r)
	{
		if(s[l]==s[r])
		{
			l++;
			r--;
		}
		else
		{ 
		return 0;
	}
	}
	return 1;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>s;
		int li=s.length();
		int l=0,r=li-1;
		int f=1;
		while(l<=r)
		{
			if(s[l]==s[r])
			{
				l++;
				r--;
			}
			else
			{
				if((!solve(l,r-1))&&(!solve(l+1,r)))
					f=0;
				break;
			}
		}
		if(f)
			cout<<"YES\n";
		else
		 cout<<"NO\n";
	}
 
	return 0;
}
