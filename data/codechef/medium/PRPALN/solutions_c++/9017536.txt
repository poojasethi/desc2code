#include<iostream>
#include<string.h>
using namespace std;
bool palin(string s,int n){
 int i=0,j=n-1;
 while(i<n && j>=0){
   if(s[i]!=s[j])
   return false;
   i++;j--;
  }
  return true;
}
string construct(string s,int in,int n){
  string tmp="";
  for(int i=0;i<=in-1;i++)
  tmp.push_back(s[i]);
  for(int i=in+1;i<n;i++)
  tmp.push_back(s[i]);
  return tmp ;
}
int main(){
 int t,n,i,j;
 string s,tmp;
 cin>>t;
 while(t--){
  cin>>s;
  n=s.size();
  if(palin(s,n))
  printf("YES\n");
  else{
    i=0,j=n-1;
  while(s[i]==s[j] && i<=j){
    i++;j--;
   }
   tmp=construct(s,i,n);
   if(palin(tmp,n-1))
   printf("YES\n");
   else{
   tmp=construct(s,j,n);
   if(palin(tmp,n-1))
   printf("YES\n");
   else
   printf("NO\n");
  }  
  }
 }
 return 0;
}
