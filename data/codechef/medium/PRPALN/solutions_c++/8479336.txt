#include <bits/stdc++.h>

using namespace std;

#define ll long long int

bool func(string str1,string str2)
{
	// cout<<str1<<" "<<str2<<endl;
	ll cnt1=0,cnt2=0,i,j,k,n=str1.length();
	for(i=0;i<n;i++)
	{
		if(str1[i]!=str2[n-i])
			break;
		else cnt1++;
	}
	for(i=n-1;i>=0;i--)
	{
		if(str1[i]!=str2[n-1-i])
			break;
		else cnt2++;
	}
	if((cnt1+cnt2)>=n)
		return true;
	else return false;
}

string reverse(string str)
{
	ll i,n=str.length();
	for(i=0;i<(n/2);i++)
		swap(str[i],str[n-1-i]);
	return str;
}


int main()
{
	ll T,j,k,i,n;
	scanf("%lld\n",&T);
	while(T--)
	{
		char s[100010];
		scanf("%s",s);
		string str(s);
		ll n=str.length();
		string rev=reverse(str);
		if(n==1 || n==2)
		{
			printf("YES\n");
			continue;
		}
		if(n%2==0)
		{
			if(func(str.substr(0,n/2-1),str.substr(n/2,n/2)) || func(rev.substr(0,n/2-1),rev.substr(n/2,n/2)))
				printf("YES\n");
			else printf("NO\n");
		}
		else{
			if(func(str.substr(0,n/2),str.substr(n/2,n/2+1)) || func(rev.substr(0,n/2),rev.substr(n/2,n/2+1)))
				printf("YES\n");
			else printf("NO\n");
		}
	}
	return 0;
}
