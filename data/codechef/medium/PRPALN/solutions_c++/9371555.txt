#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int i,j,count=0,flag=0,k,h,count1=0,count2=0,l,m,n;
		char a[100500];
		scanf("%s",a);
		j=strlen(a)-1;
		k=0;
		
		if(j==2)
		{
			
			
			if(a[0]==a[1]||a[1]==a[2]||a[0]==a[2])
			printf("YES\n");
			else
			printf("NO\n");

		}
	
		else
		{
		
			while(k<j)
			{
			
				if(a[k]!=a[j])
				{
					count++;
					l=k;
					h=j-1;
					m=k+1;
					n=j;
					while(l<h)
					{
						if(a[l]!=a[h])
						{
							count1++;
						}
						l++;
						h--;
						
					}
					while(m<n)
					{
						if(a[m]!=a[n])
						{
							count2++;
						}
						m++;
						n--;
					}
					if(count1==0||count2==0)
					{
						flag=0;
						break;
					}
					else
					flag=1;
				}
				k++;
				j--;
				if(flag==1)
				break;
				
			}
			if(flag==0)
			printf("YES\n");
			else
			printf("NO\n");

		}	
	}
	return 0;
}
