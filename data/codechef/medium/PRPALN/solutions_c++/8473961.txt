#include<bits/stdc++.h>
using namespace std;
int isPalin(string s)
{
    int i;
    int flag=0;
    int n=s.size();
    for(i=0;i<n/2;i++)
        if(s.at(i)!=s.at(n-i-1))
        {
        flag=1;
        break;
        }
        if(flag==1)
            return 0;
        else
            return 1;
}
main()
{

    int T;
    cin>>T;
    while(T-->0)
    {
        string s1,s2,s3;
        cin>>s1;
        int l=s1.size();
        int i;
        for(i=0;i<l/2;i++)
        {
            if(s1.at(i)!=s1.at(l-i-1))
            break;

        }
        s2=s1;
        s3=s1;
        s2.erase(i,1);
        s3.erase(l-i-1,1);
        if(isPalin(s2) || isPalin(s3))
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
}
