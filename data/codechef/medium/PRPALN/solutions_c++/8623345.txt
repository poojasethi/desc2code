

#include <iostream>
#include <cstring>
#include <string>

using namespace std;
#define dim 100009

int pal(char* s, int start, int finish)
{
    int i = start;
    int j = finish;

    for(; i<j;++i,--j)
        if(s[i] != s[j])
        return 0;

    return 1;
}

int T;
char s[dim];
char cop1[dim];
char cop2[dim];
int main()
{
    cin >> T;
    for(;T;--T)
    {
        cin >> s;
        if(pal(s,0,strlen(s)-1))
            cout << "YES" << '\n';
        else
        {
            int lg = strlen(s);
            int min_poz_i = -1;
            int min_poz_j = -1;
            for(int i = 0, j = lg - 1; i < j; i++,j--)
                if(s[i] != s[j])
                {
                    min_poz_i = i;
                    min_poz_j = j;
                    break;
                }
            //imi fac un string din ambele;
            int dm1 = 0;
            for( int i = 0 ; s[i]; i++)
                if(i != min_poz_i)
                    cop1[dm1++] = s[i];
            if(pal(cop1,0,dm1-1))
            {
                cout << "YES" << '\n';
                continue;
            }

            int dm2 = 0;
            for( int i = 0 ; s[i]; i++)
                if(i != min_poz_j)
                    cop2[dm2++] = s[i];
            if(pal(cop2,0,dm2-1))
            {
                cout << "YES" << '\n';
                continue;
            }

            cout << "NO" << '\n';
        }
    }
    return 0;
}
