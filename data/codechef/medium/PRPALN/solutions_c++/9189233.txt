#include<iostream>
using namespace std;

int main(){
	int t;
	string s;
	cin>>t;
	while(t--){
		cin>>s;
		int i;	
		bool flag1=true;
		int n=s.length();
		for(i=0;i<n/2;i++){
			if(s[i]!=s[n-i-1]){
				flag1=false;
				break;
			}
		}
		bool flag2=true;
		for(int j=i;j<n/2;j++){
			if(s[j+1]!=s[n-j-1]){
				flag2=false;
				break;
			}
		}

		bool flag3=true;
		for(int j=i;j<n/2;j++){
			if(s[j]!=s[n-j-2]){
				flag3=false;
				break;
			}
		}
		if(flag1||flag2||flag3)
			cout<<"YES"<<endl;

		else
			cout<<"NO"<<endl;
		//cout<<flag1<<flag2<<flag3<<endl;		
	}

	return 0;
}
