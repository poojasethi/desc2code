#include<iostream>
#include<stdio.h>

#include<string.h>
using namespace std;
char str[100002];
long int st,end,lent;
long int palin(char* str, long int st, long int end)
{
 while(st<end)
 {
  if(str[st]!=str[end])
  {
   return st;
  }
  st++;
  end--;
 }
 return -1;
} 
int main()
{
    int n;
    scanf("%d",&n);
    while(n--)
    {
      scanf("%s",str);
      //fflush(0);
      lent=strlen(str);
      long int ret;
      ret=palin(str,0,lent-1);
      if(ret==-1)
      {
       cout<<"YES\n";
       
      }
      else
      {
       if(palin(str,ret+1,lent-ret-1)==-1 || palin(str,ret,lent-ret-2)==-1)
       {
        cout<<"YES\n";
        
       }
       else
        cout<<"NO\n";
      }
    }
    //getch();
    return 0;
}
