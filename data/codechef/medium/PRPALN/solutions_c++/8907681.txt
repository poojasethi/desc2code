
//Implementare char
#include <iostream>
#include <cstring>
#include <string>

using namespace std;
#define dim 100009

int isPalindrom(char* s, int start, int finish)
{
    int i = start;
    int j = finish;

    for(; i<j; ++i,--j)
        if(s[i] != s[j])
        return 0;

    return 1;
}

int T;
char sir[dim];
char copieSir1[dim];
char copieSir2[dim];
int main()
{
    cin >> T;
    for(;T;--T)
    {
        cin >> sir;
        int dimension = strlen(sir)-1;
        if(isPalindrom(sir,0,dimension))
            cout << "YES" << '\n';
        else
        {
            int min_poz_i = -1;
            int min_poz_j = -1;
            for(int i = 0, j = dimension; i < j; i++,j--)
                if(sir[i] != sir[j])
                {
                    min_poz_i = i;
                    min_poz_j = j;
                    break;
                }
            //imi fac un string din ambele;
            int dm1 = 0;
            for( int i = 0 ; sir[i]; i++)
                if(i != min_poz_i)
                    copieSir1[dm1++] = sir[i];
            if(isPalindrom(copieSir1,0,dm1-1))
            {
                cout << "YES" << '\n';
                continue;
            }

            int dm2 = 0;
            for( int i = 0 ; sir[i]; i++)
                if(i != min_poz_j)
                    copieSir2[dm2++] = sir[i];
            if(isPalindrom(copieSir2,0,dm2-1))
            {
                cout << "YES" << '\n';
                continue;
            }

            cout << "NO" << '\n';
        }
    }


    return 0;
}
