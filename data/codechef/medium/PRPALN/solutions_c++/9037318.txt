using namespace std;
bool palin(char s[],int n)
{
	int i,j;
	for(i=0,j=n-1;i<j;i++,j--)
	{
		if(s[i]!=s[j])
			return false;
	}
	return true;
}
int main()
{
     int i,n,t,j,k,in1,in2;
     bool ans;
     char s[100001],t1[100001],t2[100001];
     scanf("%d",&t);
     while(t--)
     {
     	ans=false;
     	cin>>s;
     	n=strlen(s);
     	if(palin(s,n))
     		ans=true;
     	else
     	{
     		for(i=0,j=n-1;i<j;i++,j--)
			 {
			 	if(s[i]!=s[j])
			 	{
			 		break;
			 	}
			 }
			 in1=in2=0;
			 for(k=0;k<n;k++)
			 {
			 	if(i!=k)
			 	{
			 		t1[in1++]=s[k];
			 	}
			 	if(j!=k)
			 	{
			 		t2[in2++]=s[k];
			 	}
			 }
			 if((palin(t1,n-1))||(palin(t2,n-1)))
			 	ans=true;	
     	}
     	if(ans)
     		printf("YES\n");
     	else
     		printf("NO\n");
     }
    return 0;
}
