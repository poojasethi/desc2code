#include <iostream>
#include <vector>
std::string s;
bool f(int x1,int x2){
    for(int i=0;i<(x2-x1+1)/2;i++){
        if(s.at(x1+i)!=s.at(x2-i)){
            return 0;
        }
    }
    return 1;
};

int main(int argc, const char * argv[]) {
    int t;
    std::cin>>t;
    for(int k=0;k<t;k++){
        std::cin>>s;
        int s_length=(int)s.length();
        bool ans=1;
        int x1=0,x2=s_length-1;
        for(int i=0;i<s_length/2;i++){
            if(s.at(x1+i)!=s.at(x2-i)){
                if(s.at(x1+i+1)==s.at(x2-i)&&s.at(x1+i)==s.at(x2-i-1)){
                    ans=(f(x1+i+1,x2-i)||f(x1+i,x2-i-1));
                }else if(s.at(x1+i+1)==s.at(x2-i)){
                    ans=f(x1+i+1,x2-i);
                }else if(s.at(x1+i)==s.at(x2-i-1)){
                    ans=f(x1+i,x2-i-1);
                }else{
                    ans=0;
                }
                break;
            }
        }
        if(ans==1){
            std::cout<<"YES\n";
        }else{
            std::cout<<"NO\n";
        }
    }
    return 0;
}
