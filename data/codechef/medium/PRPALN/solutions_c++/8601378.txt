typedef long long ll;

const int maxn=1e5+11;

bool isPalin(const char (&str)[maxn], int l, int r){
	while(l<r){
		if(str[l]!=str[r])
			return false;
		l++, r--;
	}
	return true;
}

bool solve(const char (&str)[maxn]){
	int len=strlen(str);
	int l=0, r=len-1;
	while(l<r){
		if(str[l]!=str[r])
			break;
		l++, r--;
	}
	return (isPalin(str, l+1, r) || isPalin(str, l, r-1));
}

int main(){
	ios::sync_with_stdio(false);cin.clear();
	//cerr<<maxn;
	int t;
	cin>>t;
	char str[maxn];
	while(t--){
		cin>>str;
		if(solve(str))
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
}
