int isPalin(string s){
	int i = 0;
	int j = s.length()-1;
	while(i<=j){
		if(s[i]!=s[j]){
			return i;
		}
		i++;
		j--;
	}
	return -1;
}
int main(){
	int t;
	scan(t);
	while(t--){
		string s;
		cin>>s;
		bool ans;
		int index = isPalin(s);
		if(index==-1){
			ans = true;
		}
		else if(index==0){
			if(isPalin(s.substr(1,s.length()-1))==-1||isPalin(s.substr(0,s.length()-1))==-1){
				ans = true;
			}
			else{
				ans = false;
			}
		}
		else{
			string k = s.substr(0,index)+s.substr(index+1,s.length()-1-index);
			if(isPalin(k)==-1){
				ans = true;
			}
			else{
				index = s.length()-index-1;
				k = s.substr(0,index)+s.substr(index+1,s.length()-1-index);
				if(isPalin(k)==-1){
				ans = true;
				}
				else{
					ans = false;
				}
			}
		}
		ans?printf("YES\n"):printf("NO\n");
	}
	return 0;
}
