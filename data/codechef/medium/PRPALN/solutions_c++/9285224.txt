#include<bits/stdc++.h>
using namespace std;
 int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		char a[100006];
		cin>>a;
		long long int l=strlen(a);
		long long int i=0;
		long long int j=l-1;
		while(i<j)
		{
			if(a[i]==a[j])
			{
				i++;
				j--;
		    }
		    else
		    break;
			
		}
		if(i>=j)
		cout<<"YES\n";
		else
		{
			long long int start=i;
			long long int  end=j;
			i=i+1;
			while(i<j)
			{
			if(a[i]==a[j])
			{
				i++;
				j--;
		    }
		    else
		    break;
			}
			if(i>=j)
			cout<<"YES\n";
			else
			{
				i=start;
				j=end;
				j--;
				while(i<j)
			{
			if(a[i]==a[j])
			{
				i++;
				j--;
		    }
		    else
		    break;
			}
			if(i>=j)
			cout<<"YES\n";
			else
			cout<<"NO\n";
			}
			
		}
		
		
	}
	
}
