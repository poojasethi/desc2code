#include<stdio.h>
#include<iostream>
#include<math.h>
#define M 1000000007
using namespace std;
int main(void)
{
	long long int h;
	cin >> h;
	while(h!=0)
	{
		long long int s=pow(2,h),x,i;
		long long int a[s];
		long double p[s];
		for(i=1;i<(pow(2,h));i++)
		{
			cin >> a[i];
			p[i]=a[i];
		}
		for(i=(pow(2,h-1)-1);i>=1;i--)
		{
			 x=p[2*i]>p[2*i+1]?2*i:(2*i+1);
            a[i]=a[x]*a[i]%M;
            p[i]=p[x]*p[i];
			
		}
		printf("%lld\n",a[1]);
	cin >> h;
	}
}
