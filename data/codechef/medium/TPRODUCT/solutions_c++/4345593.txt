#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#define REP(i,a,b) for(i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)

#define M 1000000007

#define ll long long
ll in[111111];
double muld[111111];

int main(){
  int i,j,k,l,m,n,x;

  for(;;){
    scanf("%d",&n); if(!n)break;
    x = (1<<n)-1;
    rep(i,x) scanf("%lld",in+i);
    rep(i,x) muld[i]=log((double)in[i]);

    for(i=x/2-1;i>=0;i--){
      if(muld[i*2+1] > muld[i*2+2]) k = i*2+1; else k = i*2+2;
      in[i] = (in[i]*in[k])%M;
      muld[i] = (muld[i]+muld[k]);
    }

    printf("%lld\n",in[0]);
  }


  return 0;
}
