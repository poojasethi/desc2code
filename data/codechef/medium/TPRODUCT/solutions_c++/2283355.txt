 #include <iostream>
#include <cstdio>
#include <algorithm>
#define MOD 1000000007

using namespace std;

double b[32775];
long long int ar[32775];

double solve(int pos, int n)
{
    if(pos >= (1<<n)-1) return 1;
    double p1, p2;

    p1 = solve(pos*2+1, n);
    p2 = solve(pos*2+2, n);

    if(p1 > p2) {
        ar[pos] = (ar[pos]*(((pos*2+1) < ((1<<n)-1))?ar[pos*2+1]:1))%MOD;
        b[pos] = b[pos]*p1;
    }
    else {
        ar[pos] = (ar[pos]*(((pos*2+2) < ((1<<n)-1))?ar[pos*2+2]:1))%MOD;
        b[pos] = b[pos]*p2;
    }
    return b[pos];
}

int main()
{

    while(1) {

        int n;

        cin >> n;
        if(!n) return 0;
        for(int i = 0; i < (1<<n)-1; i++) { cin >> b[i]; ar[i] = b[i]; }
        solve(0,n);
        cout << ar[0] << endl;
    }
}
