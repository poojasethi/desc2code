#include<stdio.h>
inline long long inp()
    {
    int noRead=0;
    char p=getchar_unlocked();
    for(;p<33;){p=getchar_unlocked();};
    while(p>32)
    {
    noRead = (noRead << 3) + (noRead << 1) + (p - '0');
    p=getchar_unlocked();
    }
    return noRead;
    };
 
long long a[32768];
double b[32768];
	 
int main()
{
	int i,n;
	while(n=inp())
	{
	    n=(1<<n)-1;
	    for(i=1;i<=n;i++)
	    b[i]=a[i]=inp();
	    for(i=n/2;i>=1;i--)
	    {
	        if(b[2*i]>b[2*i+1])
	                {
	                    b[i]*=b[2*i];
	                    a[i]*=a[2*i];
	                }
	                else
	                {
	                    b[i]*=b[2*i+1];
	                    a[i]*=a[2*i+1];
	                }
	                a[i]%=1000000007;
	    }
	    printf("%lld\n",a[1]);
	}
	return 0;
}
