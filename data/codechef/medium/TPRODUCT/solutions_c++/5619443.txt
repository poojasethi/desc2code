#include <iostream>
#include <stdio.h>
#include <vector>
#include <math.h>
using namespace std;
long long int MOD=1000000007;
int main(){
	while(true){
			long long int h;scanf("%lld",&h);
			if(h==0) break;
			long long int n=pow(2.0,h)-1;
			long long int A[n+1];
			for(int i=1;i<=n;i++){
				scanf("%lld",&A[i]);
			}
			vector<long long int> ans[n+1];
			
			for(int j=n;j>=1;j--){
				if(2*j>n){
					ans[j].push_back(A[j]);
				
				}
				else{
					bool left=true;
					if(ans[2*j].size()<ans[2*j+1].size()) {left=false;}
					else if(ans[2*j].size()==ans[2*j+1].size()){
						int n=ans[2*j].size();
						int k=n-1;
						while(true){
							if(ans[2*j][k]>ans[2*j+1][k]) {break;}
							else if(ans[2*j][k]<ans[2*j+1][k]) {left=false; break;}
							k--;
							if(k<0) break;
						}
					}
						
						int t=2*j;
						if(!left) t++;
						
						long long int carry=0;
						long long int a=A[j];
						for(int i=0;i<ans[t].size();i++){
							long long int fullans=ans[t][i]*a+carry;
							ans[j].push_back(fullans%MOD);
							carry=fullans/MOD;
						}
						while(carry!=0){
							ans[j].push_back(carry%MOD);
							carry=carry/MOD;
						}
						
					}
				}
				printf("%lld\n",ans[1][0]);
			}
}
