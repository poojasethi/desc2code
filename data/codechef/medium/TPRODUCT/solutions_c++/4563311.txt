double a[65536];
ll val[65536];
int main(){
	int i,j;
	while(1){
		int h;
		si(h);
		if(h==0){
			break;
		}
		int n=pow(2,h)-1;
		for(i=1;i<=n;i++){
			sd(a[i]);
			val[i]=a[i];
		}
		for(i=n;i>=1;i--){
			if(2*i < n ){
				if(a[2*i] > a[2*i+1]){
					a[i]*=a[2*i];
					val[i]*=val[2*i];
					val[i]=val[i]%MAX;
				}
				else{
					a[i]*=a[2*i+1];
					val[i]*=val[2*i+1];
					val[i]=val[i]%MAX;
				}
			}
		}
		pl(val[1]%MAX);
	}

}
