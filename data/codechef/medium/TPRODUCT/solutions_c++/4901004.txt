#include<iostream>
using namespace std;
#include<math.h>
#define max 1000000007
int main()
{
	int h;
	cin>>h;
	long long int total;
	while(h!=0)
	{
		long long int a[32800],n,i;
		long double b[32800];
		for(i=1;i<pow(2,h);i++)
		{
			cin>>a[i];
			b[i]=(double)a[i];
		}
		n=pow(2,h-1)-1;
		while(n>0)
		{
			if(b[2*n]>b[2*n+1])
            {
                b[n]*=b[2*n];
				a[n]*=a[2*n];
            }
            else
            {
                b[n]*=b[2*n+1];
				a[n]*=a[2*n+1];
            }
            a[n]%=max;
            n--;
		}
		cout<<a[1]<<"\n";
		cin>>h;
	}
	return 0;
}
