#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll mod=1e9+7;

int main()
{
	ios::sync_with_stdio(false);
	int h;
	cin>>h;
	while(h!=0)
	{
		int k=(1<<h)-1;
		// cout<<k<<endl;
		ll a[k+1];
		double b[k+1];
		b[0]=a[0]=0;
		for(int i=1;i<=k;i++)
		{
			cin>>a[i];
			b[i]=a[i];
		}
		for(int i=k/2;i>=1;i--)
		{
			if(b[i*2]>b[i*2+1])
			{
				b[i]*=b[i*2];
				a[i]*=a[i*2];
			}
			else
			{
				b[i]*=b[i*2+1];
				a[i]*=a[i*2+1];
			}
			a[i]%=mod;
		}
		cout<<a[1]<<"\n";
		cin>>h;
	}
	return 0;
}
