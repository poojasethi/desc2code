int main()
{
	int t;
	scanf("%d", &t);
	
	while(t!=0)
	{
		long long int n= pow(2,t)-1;
		long long int a[n];
		long double store[n];
		
		for(int i=0; i<n; i++)
		{
		scanf("%lld", &a[i]);
		store[i]= a[i];
		}
		
		
		for(int i= n/2-1; i>=0; i--)
		{
			if(store[2*i+1]>store[2*i+2])
			{
				a[i]*= a[2*i+1];
				store[i]*= store[2*i+1];
			}
			else
			{
				a[i]*= a[2*i+2];
				store[i]*= store[2*i+2];
				
			}
			
			a[i]= a[i]%1000000007;
		}

}
