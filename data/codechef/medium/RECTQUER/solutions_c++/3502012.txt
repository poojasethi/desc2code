#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <sstream>
#include <vector>
#include <iomanip>
#include <cmath>

using namespace std;
typedef long long LL;
typedef pair<int,int> pii;
int matrix[320][320],dp[320][320][11];
int main()
{
    int t,n,q,x1,y1,x2,y2,chk,cnt;
    t = 1;
    while(t--)
    {
        cin>>n;
        for(int i=1;i<=n;i++)
            for(int j=1;j<=n;j++)
                cin>>matrix[i][j];

        for(int d=1;d<=10;d++)
            for(int i=1;i<=n;i++)
                for(int j=1;j<=n;j++)
                    dp[i][j][d] = dp[i-1][j][d] + dp[i][j-1][d] - dp[i-1][j-1][d] + (matrix[i][j] == d);

        cin>>q;
        for(int i=1;i<=q;i++)
        {
            cin>>x1>>y1>>x2>>y2;
            cnt = 0;
            for(int d=1;d<=10;d++)
            {
                chk = dp[x2][y2][d] - dp[x2][y1-1][d] - dp[x1-1][y2][d] + dp[x1-1][y1-1][d];
                if(chk > 0)
                    cnt++;
            }
            cout<<cnt<<endl;
        }

    }
    return 0;
}
