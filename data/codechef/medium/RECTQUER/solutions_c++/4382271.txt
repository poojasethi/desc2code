#include<iostream>
#include<stdio.h>
using namespace std;
 
inline int read()
{
    int n=0;
    char c=0;
    while(c<33)
        c=getchar();
    while(c>='0'&&c<='9')
    {
        n=(n<<3)+(n<<1)+(c-'0');
        c=getchar();
    }
    return n;
}
 
int main()
{
    int n,i,j,k,temp,x1,x2,y1,y2;
    n=read();
    int a[301][301][11]={0};
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            temp=read();
            for(k=1;k<11;k++)
            {
                a[i][j][k]=a[i-1][j][k]+a[i][j-1][k]-a[i-1][j-1][k];
            }
            a[i][j][temp]++;
        }
    }
    n=read();
    while(n--)
    {
        x1=read();
        y1=read();
        x2=read();
        y2=read();
        temp=0;
        for(k=1;k<11;k++)
        {
            if(a[x2][y2][k]-a[x1-1][y2][k]-a[x2][y1-1][k]+a[x1-1][y1-1][k]>0)
               temp++;
        }
        printf("%d\n",temp);
    }
}
