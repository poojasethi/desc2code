void solve()
{
	int n;
	s(n);
	matrix mat = matrix(n, row(n, 0));
	rep(i, n) rep(j, n) s(mat[i][j]);
	vector<matrix>v(11);
	rep(i, 11) {
		matrix temp = matrix(n, row(n, 0));
		rep(j, n) {
			rep(k, n) {
				temp[j][k] = ((j - 1 >= 0) ? temp[j - 1][k] : 0);
				temp[j][k] += ((k - 1 >= 0) ? temp[j][k - 1] : 0);
				temp[j][k] -= ((j - 1 >= 0 && k - 1 >= 0) ? temp[j - 1][k - 1] : 0);
				temp[j][k] += (mat[j][k] == i);
			}
		}
		v[i] = temp;
	}	
	int q;
	s(q);
	while(q--) {
		int x1, y1, x2, y2;
		s(x1); s(y1); s(x2); s(y2);
		x1--; y1--; x2--; y2--;
		int ans = 0;
		for(int i = 1; i <= 10; i++) {
			int tem = v[i][x2][y2]; 
			tem -= ((y1 - 1 >= 0) ? v[i][x2][y1 - 1] : 0); 
			tem -= ((x1 - 1 >= 0) ? v[i][x1 - 1][y2] : 0);
			tem += ((x1 - 1 >= 0 && y1 - 1 >= 0) ? v[i][x1 - 1][y1 - 1] : 0);
			if(tem) ans++;
		}
		pi(ans);
		printf("\n");
	}
}

int main()
{
	int t = 1;
	//scanf("%d", &t);
	while(t--) {
		solve();
	}
	return 0;
}
