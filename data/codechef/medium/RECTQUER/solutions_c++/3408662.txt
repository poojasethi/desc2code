#include<iostream>
#include<cstdio>

using namespace std;

int main()
{	
	int n,q;
	scanf("%d",&n);
	int * * *h=new int * * [n];
	for(int i=0;i<n;i++)
	{
		h[i]=new int * [n];
		for(int j=0;j<n;j++)
			h[i][j]=new int [11];
	
	}
	int * * a=new int *[n];
	for(int i=0;i<n;i++)
		a[i]=new int [n];
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			scanf("%d",&a[i][j]);
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			for(int k=0;k<11;k++)
				h[i][j][k]=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(i>0)
			{
				for(int k=1;k<=10;k++)
					h[i][j][k] += h[i-1][j][k];
			}
			if(j>0)
			{
				for(int k=1;k<=10;k++)
					h[i][j][k] += h[i][j-1][k];
			}
			if(i>0 && j>0)
			{
				for(int k=1;k<=10;k++)
					h[i][j][k] -= h[i-1][j-1][k];
			}
			h[i][j][a[i][j]]++;
		}
	}
	scanf("%d",&q);
	int as[11];
	int x1,y1,x2,y2;
	while(q--)
	{
		scanf("%d %d %d %d",&x1,&y1,&x2,&y2);
		x1--;x2--;y1--;y2--;
		for(int i=1;i<=10;i++)
			as[i]=h[x2][y2][i];
		if(x1>0&&y1>0)
		{
			for(int i=1;i<=10;i++)
				as[i]+=h[x1-1][y1-1][i];
		}
		if(x1>0)
		{
			for(int i=1;i<=10;i++)
				as[i]-=h[x1-1][y2][i];
		}
		if(y1>0)
		{
			for(int i=1;i<=10;i++)
				as[i]-=h[x2][y1-1][i];
		}
		int ans=0;
		for(int i=1;i<=10;i++)
			if(as[i]>=1)
				ans++;
		printf("%d\n",ans);
	}
	return 0;
}
