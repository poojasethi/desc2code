#include<stdio.h>
#include<stdlib.h>


int a[310][310][11], v[11],b[310][310];
int inline take()
{
	char c;
	int n=0;
	
	c=getchar_unlocked();
	while(c < '0' || c >'9') {c=getchar_unlocked();}
    while(c>='0' && c<='9')
	{
		n=(n<<3)+(n<<1)+c-'0';
		c=getchar_unlocked();
	}
	return n;
}

main()
{
	
	int n,q,x1,x2,y1,y2,i,j,xc,sum,k;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			b[i][j]=take();
			v[b[i][j]]++;
			for(k=1;k<11;k++)
			{
			a[i][j][k]=v[k];
			
			}
		}
	}
	
	q=take();
	while(q--)
	{
		x1=(take()-1);
		y1=(take()-1);
		x2=(take()-1);
		y2=(take()-1);
		
	
		xc=x2-x1;
		sum=0;
		for(i=1;i<11;i++)
		v[i]=0;
	
	for(i=0;i<=xc;i++)
	{
		if(v[b[x1][y1]]==0)
		{
		sum+=1;
		v[b[x1][y1]]=1;
		}
		for(k=1;k<11;k++)
		{
			if((a[x1][y2][k]>a[x1][y1][k]) && v[k]==0)
			{
				sum+=1;
				v[k]=1;
			}
		}
		
		x1++;
	}
	if(sum==0)
	printf("%d\n",sum+1);
	else
	printf("%d\n",sum);
	}
	
		
}
