#include<iostream>
#include<cstdio>
 
inline void inp(int *n)
{
    *n = 0;
    int ch = getchar_unlocked();
    while(ch < '0' || ch > '9') 
    {
        ch = getchar_unlocked();
    }
    while(ch >= '0' && ch <= '9')
        (*n) = ((*n)<<3) + ((*n)<<1) + ch - '0', ch = getchar_unlocked();
}
 
typedef int ll;
ll a[305][305][15],k,p,n,i,j,m,w,x1,y1,x2,y2;
 
 
int main()
{
 
		inp(&n);
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				//scanf("%lld",&p);
				inp(&p);
				for(k=1;k<=10;k++)
				{
					a[i][j][k]=a[i][j-1][k];
					//a[i][j][p]++;
				}
				a[i][j][p]++;
			}
		}
		
		//scanf("%lld",&m);
		inp(&m);
		for(w=1;w<=m;w++)
		{
			//scanf("%lld %lld %lld %lld",&x1,&y1,&x2,&y2);
			inp(&x1);
			inp(&y1);
			inp(&x2);inp(&y2);
			ll b[15]={0};
			for(i=x1;i<=x2;i++)
			{
				for(k=1;k<=10;k++)
				{
					b[k]=b[k]+a[i][y2][k]-a[i][y1-1][k];
				}
			}
			ll cnt=0;
			for(i=1;i<=10;i++)
			{
				if(b[i]) cnt++;
			}
			printf("%d\n",cnt);	
		}
	return 0;
}
