#include<iostream>
using namespace std;
int grundy(int x,int y)
{
	if(x<y)
	swap(x,y);
	
	if(x%y==0)
	return (x/y-1);
	int moves=x/y;
	int g=grundy(x%y,y);
	if(g==0)
	return moves;
	if(moves<=g)
	return moves-1;
	return moves;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int xo=0;
		int n;
		cin>>n;
		int x,y;
		for(int i=0;i<n;i++)
		{
			cin>>x>>y;
			xo^=grundy(x,y);
		}
		if(xo)
		cout<<"YES\n";
		else
		cout<<"NO\n";
	}
	return 0;
}
