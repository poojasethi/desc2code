int Solve(int x, int y){
    if(x == y){
        return 0;
    }
    if(!x || !y){
        return 1;
    }
    if(x < y){
        x = (x + y) - (y = x);
    }
    if(x % y == 0){
        return (x / y - 1);
    }
    if(x / y <= Solve(y, x%y)){
        return x / y - 1;
    }
    return x / y;
}

int main() {
    int T;
    int N;
    cin >> T;
    int x, y;
    while(T--){
        scanf("%d", &N);
        int Ans = 0;
        for(int n=1;n<=N;++n){
            scanf("%d%d", &x, &y);
            Ans ^= Solve(x, y);
        }
        if(Ans){
            printf("YES\n");
        }
        else{
            printf("NO\n");
        }
    }
    return 0;
}
