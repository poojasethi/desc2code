#include <stdio.h>
#include <cstring>
#include <algorithm>
using namespace std;
int grundy(int a , int b)
{
	if(a==b)return 0;
	int  x ;
	if(b>a) swap(a,b);
	if(a%b==0)return a/b-1;
	x = grundy(a%b,b);
	int q =a/b;
	if(x>=q)return q-1;
	else
	{
		if(x-1==q)return x;
		else return q ;
	}
}
int main()
{
    //freopen("in.txt","r",stdin);
	int i , j , t, a ,b,ans,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		n--;
		scanf("%d%d",&a,&b);
		int ans= grundy(a,b);
		while(n--)
		{
			scanf("%d%d",&a,&b);
			ans^=grundy(a,b);
		}
		if(ans)printf("YES\n");
		else printf("NO\n");
	}
	return 0;
}
