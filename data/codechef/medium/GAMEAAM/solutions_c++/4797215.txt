inline void readint(int &x)
{
	register int c = getchar_unlocked();
	x = 0;
	for(;(c<48 || c>57);c = getchar_unlocked());
	for(;c>47 && c<58;c = getchar_unlocked()) {x = (x<<1) + (x<<3) + c - 48;}
}

inline void writeint(int x, char endchar)
{
	if(x == 0)
		putchar_unlocked('0');
	else
	{
		int cnt = 0;
		char ch[20];
		while(x != 0)
		{
			ch[cnt++] = x%10 + '0';
			x /= 10;
		}
		for(int i=cnt - 1; i>=0; --i)
			putchar_unlocked(ch[i]);
	}
	putchar_unlocked(endchar);
}

inline void writestring(const char* ch)
{
	while(*ch != '\0')
		putchar_unlocked(*(ch++));
	putchar_unlocked('\n');
}

int effectivenum(int a, int b)
{
	if(a == b)
		return 0;
	if(a < b)
		return effectivenum(b, a);
	if(a % b == 0)
		return a/b - 1;
	int temp = effectivenum(b, a % b);
	int nummoves = a/b;
	if(nummoves <= temp)
		return nummoves - 1;
	else
		return nummoves;
}

int main()
{
	int t;
	readint(t);
	while(t--)
	{
		int n;
		readint(n);
		int result = 0;
		int n1, n2;
		for(int i=0; i<n; ++i)
		{
			readint(n1);
			readint(n2);
			result ^= effectivenum(n1, n2);
		}
		if(result == 0)
			writestring("NO");
		else
			writestring("YES");
	}
	return 0;
}
