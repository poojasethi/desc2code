#include<stdio.h>
#include<cstdlib>
typedef long long int ll;
ll max(ll a, ll b)
{
    return a>b?a:b;
}
ll min(ll a, ll b)
{
    return a<b?a:b;
}
ll grundy(ll a, ll b)
{
    if(b==a)
    {
        return 0;
    }
    else
    {
        if(a%b==0)
        {
            if(grundy(b,b)< a/b)
            {
                return a/b-1;
            }
        }
        else
        {
            if(grundy(b,a%b) <  a/b)
            {
                return a/b;
            }
            else
            {
                return min(a/b-1,grundy(b,a%b)-1);
            }
        }
    }
}
int main()
{
    int test_cases,n;
    scanf("%d",&test_cases);
    ll a,b;
    ll ans=0;
    for(int i=0;i<test_cases;i++)
    {
        scanf("%d",&n);
        ans=0;
        for(int j=0;j<n;j++)
        {
            scanf("%lld%lld",&a,&b);
            ans=ans^grundy(max(a,b),min(a,b));
        }
        //printf("%lld\n",ans);
        if(ans!=0)
        {
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }
}
