#define max(a,b) (a>b?a:b)

#define getcx getchar_unlocked
inline void inp( int &n )
{
        n=0;
        int ch=getcx();int sign=1;
        while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}
 
        while(  ch >= '0' && ch <= '9' )
                n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
		n=n*sign;
}
 
int sub(int x, int y)
{ 
	int s,sol;
	if(x==y) return 0;
	if(x>y)
	swap(x,y);
	if((y%x)==0) return y/x-1;
	sol=y/x;
	s=sub(y%x,x);
	if(s==0)	return sol;
	if(sol<=s)	return sol-1;
	return sol;
}
 
int main() 
{
	int t,n,i,result,a,b;
	inp(t);
	while(t--)
	{
		inp(n);
		result=0;
		for(i=0;i<n;i++)
		{
			inp(a),inp(b);
			result^=sub(a,b);
		}
		if(result==0)	printf("NO\n");
		else	printf("YES\n");
		
	}
	return 0;
}
