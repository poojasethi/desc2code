#include<bits/stdc++.h>

typedef long long int li;

using namespace std;

int main()
{
    //freopen("input.txt", "r", stdin);
    li t, i, n, t1, t2, j;
    cin >> t;
    while(t--){
        li ans = 0;
        cin >> n;
        //cout << "\n";
        for(j = 0;j < n; ++j){
            vector <li> v;
            cin >> t1 >> t2;
            li t11 = max(t1, t2);
            li t12 = min(t1, t2);
            while(t11 != t12){
                if(t11 % t12 == 0){
                    v.push_back((t11 / t12) - 1);
                    t11 = t12;
                }else{
                    v.push_back((t11 / t12));
                    t11 = t11 - ((t11 / t12) * t12);
                }
                t1 = t11; t2 = t12;
                t11 = max(t1, t2);
                t12 = min(t1, t2);
                //cout << "yes";
            }
            if(v.size() != 0)
            reverse(v.begin(), v.end());
            li pre = 0;
            for(i = 0;i < v.size(); ++i){
                if(v[i] <= pre){
                    pre = v[i] - 1;
                }else{
                    pre = v[i];
                }
            }
            //cout << pre << "\n";
            ans = (ans ^ pre);
            //cout << "no";
        }
        if(ans) cout << "YES\n";
        else cout << "NO\n";
    }
}
