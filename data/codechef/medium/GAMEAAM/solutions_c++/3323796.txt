// The Sprague?Grundy function of the game Euclid
#include <iostream>
using namespace std;

int main() {
    int t, n;
    cin >> t;
    while(t--) {
        long long x, y, mex=0;
        cin >> n;
        while(n--) {
            cin >> x >> y;
            long long t1 = (x*x-y*y)/(x*y);
            if(t1<0) t1 = -t1;
            mex ^= t1;
        }
        if(mex) cout << "YES" << endl;
        else cout << "NO" << endl;
    }
    return 0;
}
