using namespace std;

int getGrundy(int a, int b)
{
    if (b == 0)
		return 0;
	int x = getGrundy(b, a % b);
	if (x == 0)
		if (a % b == 0)
			return (a / b) - 1;
		else
			return (a / b);
	else
		if (a / b <= x)
			return a / b - 1;
		else
			return a / b;
}
int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		int n;
		cin >> n;
		int ans2 = 0;
		for (int i = 0; i < n; i++)
		{
			int a, b;
			cin >> a >> b;
			if (a < b)
				swap(a, b);
			ans2 ^= getGrundy(a, b);
		}
		if (ans2)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
}
