int sprague(int a,int b)
{
	if(a == b){
		return 0;
	}
	if(a < b){
		return sprague(b, a);
	}
	if(a % b == 0){
		return (a / b) - 1;
	}
	int w = sprague(b, a % b);
	
	if(a / b > w){
		return a / b;
	}
	else{
		return (a / b) - 1;
	}
}
int main()
{
	int t,n,p,q;
	int ans;
	cin>>t;
	while(t--)
	{
		ans = 0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&p,&q);
			ans = ans^sprague(p,q);
			
		}
		if(ans == 0){
			printf("NO\n");
		}
		else{
			printf("YES\n");
		}
	}
	return 0;
}
