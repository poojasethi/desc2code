#include<iostream>
using namespace std;
int grundy(int a, int b)
{
    if(a==b) return 0;
    if(b%a==0)return (b-a)/a;
    int temp;

    if(b%a!=0)temp=grundy(b%a,a);
    //else temp=(b-a)/a;
    if((b-a)/a >=temp) return (b/a);
    else return (b-a)/a;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n][2];
        bool flag=true;
        int ans=0;
        for(int i=0;i<n;i++)
        {
            cin>>a[i][0]>>a[i][1];
            int x,temp;
            if(a[i][0]>a[i][1])
            {
                x=a[i][1];
                a[i][1]=a[i][0];
                a[i][0]=x;
            }

            ans=ans^grundy(a[i][0],a[i][1]);
        }//>>temp;

            if(ans==0)cout<<"NO"<<endl;
            else cout<<"YES"<<endl;
        // cout<<"Pagfloyd wins."<<endl;
    }
}
