#include <iostream>
#include <algorithm>



using namespace std;

int solve ( int a , int b )
{
    int temp;
    if ( a == b )
        return 0;
   
    else if ( b % a == 0 )
        return b/a - 1;
    
    else
    {
        temp = solve ( b % a , a );
        if ( temp == 0 )
            return b/a;
        else
            if ( b/a == 1 )
                return 0;
            else if ( b/a <= temp )
                return b/a - 1;
            else
                return b/a;
    }
    
}

int main()
{
    int T;
    cin >> T;
    
    while ( T-- )
    {
        int N;
        cin >> N;
        int x = 0;
        while ( N-- )
        {
            int a , b;
            
            cin >> a >> b;
            
            if ( a > b )
                swap ( a , b );
            
            x ^= solve(a,b);
        }
        
        if ( x )
            cout << "YES\n";
        else
            cout << "NO\n";
    }
    return 0;
}
