int nim(int a, int b)
{
	if(a>b) swap(a,b);
	int k = b/a;
	if(b % a)
	{
		int x = nim(b % a, a);
		if(x >= k) return k-1;
		return k;
	}
	return k-1;
}

int main()
{
	ios::sync_with_stdio(false);
	int t, n, a, b, ans;
	cin >> t;
	while(t--)
	{
		ans = 0;
		cin >> n;
		for(int i = 0; i<n; i++)
		{
			cin >> a >> b;
			ans ^= nim(a, b);
		}
		if(ans)	cout << "YES\n";
		else	cout << "NO\n";
	}
	return 0;
}
