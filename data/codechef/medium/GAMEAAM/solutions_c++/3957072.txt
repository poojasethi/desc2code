#include<iostream>
#include<cmath>
#include<map>
using namespace std;
//int rem[1000][1000]={-1};
std::map < pair<int,int> , int > rem;

int func(int a,int b)
{  //cout<<b<< " "<<a;
   // if(rem[a][b])
   if(rem[make_pair(a,b)]>0)
    return rem[make_pair(a,b)]-1;
   if(b%a==0)
   {  // cout<<" return = "<<(b/a)-1<<" | ";
       rem[make_pair(a,b)]=(b/a);
       return (b/a)-1 ;}
   else
    { int x=func(b%a,a);
      x= x>=floor(b/a)? floor(b/a)-1:floor(b/a);
      //cout<<" return = "<<x<<" | ";
     rem[make_pair(a,b)]=x+1;
      return x;
    }

}
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
      int n;
       cin>>n;
       int ans=0;

       while(n--)
        {
          int a,b;
          cin>>a>>b;
           //ans=func(a,b);
           ans=ans^(a>b?func(b,a):func(a,b));
          //cout<<ans<<endl;
       }
      ans>0?cout<<"YES\n":cout<<"NO\n";

  }

}
