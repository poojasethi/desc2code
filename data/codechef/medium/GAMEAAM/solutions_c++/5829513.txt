#include <cstdio>
#include <vector>

#include <iostream>
using namespace std;

int func(int a,int b)
{
if(a>b)
    swap(a,b);
if(a==b)
    return 0;
if(b>=2*a)
    return 1;
    return 1-func(b%a,a);
}
int grundy(int a,int b)
{
    if(a>b)
        swap(a,b);
if(a==b)
    return 0;
 if(b%a==0)
     return (b/a-1);
 if(func(b%a,a)==0)
     return b/a;
 else
 {
 if(b<2*a)
     return 0;
     int tp=grundy(b%a, a);
     if(b/a<=tp)
         return b/a-1;
     return b/a;
 }
}
int main()
{
    int T,N,i,tp;
    scanf("%d",&T);
    while (T) {
        scanf("%d",&N);
        int a,b;
        tp=0;
        for(i=0;i<N;i++)
        {
            scanf("%d %d",&a,&b);
            tp=tp^grundy(a,b);
        }
        if(tp==0)
            printf("NO\n");
        else
            printf("YES\n");
        T--;
    }
}
