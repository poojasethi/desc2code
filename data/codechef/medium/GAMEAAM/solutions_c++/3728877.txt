#include <bits/stdc++.h>
using namespace std;

int dfs(int n,int m)
{
    if(n > m) swap(n,m);
    if(m % n == 0) return m / n - 1;
    int sg = dfs(n,m % n);
    if(!sg) return m / n;
    if(m / n <= sg) return m / n - 1;
    return m / n;
}

void solve()
{
    int t,n,m;
    int ans = 0;
    scanf("%d",&t);
    while(t --) {
        scanf("%d%d",&n,&m);
        ans ^= dfs(n,m);
    }
    if(ans) puts("YES");
    else puts("NO");
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t --) solve();
    return 0;
}
