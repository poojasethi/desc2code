#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std ;


long long pileSize(long long A,long long B) // A >= B
{
	if (A < B)
		swap(A,B);
	if (A == B)return 0;
	long long C = A%B;
	if (C == 0)
		C = B;
	long long cur = (A-C)/B;
	long long nex = pileSize(B,C);
	if (cur > nex)return cur;
	return cur - 1;
}



int main()
{
//	freopen("game.in","r",stdin);
	int c,c2;
	int tests;
	scanf("%d",&tests);
	while (tests--){
		int n;
		scanf("%d",&n);
		int ret = 0;
		for (c=0;c<n;c++){
			int A,B;
			scanf("%d%d",&A,&B);
			ret^=pileSize(A,B);
		}
		if (ret)
			printf("YES\n");
		else printf("NO\n");
	}
	
	return 0;
}
