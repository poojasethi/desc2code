#include <cstdio>
#include <algorithm>
using namespace std;

int t, n, a, b;

int grundy(int a, int b) {
	return a ? b / a + (grundy(b % a, a) < b / a && b % a) - 1 : 0; 
}

int main() {
    scanf("%d", &t);
    while (t--) {
        scanf("%d", &n);
        int sol = 0;
        while (n--) {
            scanf("%d%d", &a, &b);
            sol ^= grundy(min(a, b), max(a, b));
        }
        puts(sol ? "YES" : "NO");
    }
	return 0;
}
