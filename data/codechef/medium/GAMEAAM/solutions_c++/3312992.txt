#include <cstdio>

using namespace std;

int f(int a, int b)
{
    int d = a / b - 1;
    a %= b;
    if(a == 0)
        return d;
    return d + (d >= f(b, a));
}

int main()
{

    int T;
    scanf("%d", &T);
    while(T--)
    {
        int sol = 0, n;
        scanf("%d", &n);
        while(n--)
        {
            int a, b;
            scanf("%d %d", &a, &b);
            sol ^= a > b ? f(a, b) : f(b, a);
        }
        puts(sol ? "YES" : "NO");
    }
    return 0;
}
