#include <iostream>
using namespace std;
long long nim(long long x, long long y)
{
	if (x < y)
		swap(x, y);
	if (x == y)
		return 0;
	long long step;
	if (x % y == 0)
	{
		step = x / y - 1;
		x = y;
	}
	else
	{
		step = x / y;
		x %= y;
	}
	if (nim(x, y) <= step - 1)
		return step;
	return step - 1;
}
int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin >> t;
	while (t--)
	{
		long long n, ans = 0;
		cin >> n;
		for (int i = 0; i < n; i++)
		{
			long long x, y;
			cin >> x >> y;
			ans ^= nim(x, y);
		}
		if (ans)
			cout << "YES" << endl;
		else
			cout << "NO" << endl;
	}
	return 0;
}
