#include<bits/stdc++.h>
using namespace std;

int main()
    {
        int test, N;
        long long int a,b,temp,res;
        cin>>test;
        while(test)
           {
              res = 0;
              cin>>N;
              while(N)
                 {
                     cin>>a>>b;
                     temp = (a*a - b*b)/(a*b);
                     if(temp<0)
                       temp = -temp;
                     res = res^temp;
                     N = N-1;
                 }
               if(res == 0)
                 cout<<"NO\n";
               else
                 cout<<"YES\n";
               test = test - 1;
           }
        return 0;   
    }
