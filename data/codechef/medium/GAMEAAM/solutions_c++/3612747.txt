#include<bits/stdc++.h>
using namespace std;
#define For(i,a,n) for(int i=a;i<n;i++)
#define sc(a) scanf("%d",&a)
#define sc2(a,b) scanf("%d %d",&a,&b)

int solve(int a,int b)
    {
    if(a<b) swap(a,b);
    if(a%b==0) return a/b-1;
    int c=a/b,s=solve(b,a%b);
    if(c<=s) return c-1;
    return c;
    }

int main()
    {
    
    int a,b,n,t,s;
    for(sc(t);t--;)
       {
       s=0;
       sc(n);
       while(n--)
          sc2(a,b),s^=solve(a,b);
       puts(s?"YES":"NO");
       }
    return 0;
    }
