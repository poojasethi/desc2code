#define ll long long

int getGru(int a , int b)
{
    if(a % b == 0)
        return a / b - 1;

    int x = getGru(b , a % b);
    if(a / b <= x)
        return a / b - 1;
    else
        return a / b;
}
int main(){
    int T;
    cin >> T;
    for(int t = 0;t < T;t ++)
    {
        int n;
        cin >> n;
        int ans = 0ll;
        for(int i = 0;i < n;i ++)
        {
            int a, b;
            cin >> a >> b;
            ans ^= getGru(max(a , b) , min(a  , b));
        }
        if(ans == 0)
            cout << "NO" << endl;
        else
            cout << "YES" << endl;
    }
	return 0;
}
