#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int a[n];
		long long s=0;
		int i;
		int c=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		for(i=0;i<n;i++)
		{
			if(a[i]<2)
			{
				c=1;
				break;
			}
		}
		if(c==1)
		printf("-1\n");
		else
		{
			sort(a,a+n);
			for(i=1;i<n;i++)
			{
				s+=a[i];
			}
			s+=2;
			printf("%lld\n",s);
		}
	}
	return 0;
}
