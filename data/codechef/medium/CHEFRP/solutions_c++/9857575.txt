#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	long long int n,a,min=INT_MAX,ans=0;
	int flag=1;
	scanf("%d",&t);
	while(t--){
		flag=1;
		scanf("%lld",&n);
		ans=0;
		min=INT_MAX;
		while(n--){
			scanf("%lld",&a);
			if(a<2){
				flag=0;
			}else{
				if(a < min)
					min=a;
				ans+=a;
			}
			
		}
		if(flag==0)
			printf("-1\n");
		else
			printf("%lld\n",ans-min+2);
	}
	
	return 0;
}
