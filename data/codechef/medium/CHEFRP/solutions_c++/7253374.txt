#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int w,n;
    cin>>w;
    while(w--){
        cin>>n;
        int arr[n],i;
        for(i=0;i<n;i++)
            cin>>arr[i];
        sort(arr,arr+n);
        int sum=0;
        for(i=n-1;i>0;i--)
            sum=sum+arr[i];
        if(arr[0]>=2){
            sum+=2;
            cout<<sum<<endl;
        }
        else cout<<-1<<endl;
    }
    return 0;
}
