#include<iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--)
	{
		long long int sum = 0;
		int n,flag = 0,arr[100005],m= 1000009;
		cin>>n;
		for(int i = 0;i<n;i++)
		{
			cin>>arr[i];
			sum+=arr[i];
			if(arr[i] < m)
				m = arr[i];
			if(arr[i]<2)
				flag = 1;
		}
		if(flag == 1)
			cout<<"-1\n";
		else
			cout<<(sum+2-m)<<endl;
	}
return 0;
}
