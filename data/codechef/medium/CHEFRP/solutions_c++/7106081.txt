#include <iostream>
#include <cstring>
using namespace std;
const int MAX = 100000 + 1;

int a[MAX];

int main() {
	int t;
	cin >> t;
	while (t--) {
		int n;
		bool shouldBreak = false;
		cin >> n;
		for ( int i = 0; i < n; i++) {
			cin >> a[i];
			if (a[i] < 2) {
				shouldBreak = true;
			}
		}
		
		if (shouldBreak) {
			cout << -1 << endl;
			continue;
		}
		
		int min = a[0];
		for (int i = 1; i < n; i++) {
			if (min > a[i]) min = a[i];
		}
		
		int sum = 0;
		for (int i = 0; i < n;i++) {
			sum += a[i];
		}
		
		cout << sum - min + 2 << endl;
		
		
	}
	return 0;
}
