#include<iostream>
using namespace std;
int main()
{
	long long int test,n,min,i,sum,a[100000];
	cin>>test;
	while(test--)
	{
		cin>>n;
		for(i=0;i<n;++i)
		cin>>a[i];
		sum=0;
		min=a[0];
		int flag=1;
		for(i=0;i<n;++i)
		{
			if(a[i]<min)
			min=a[i];
			sum+=a[i];
			if(a[i]<2)
			{
				flag=0; break;
			}
		}
		if(!flag)
		cout<<"-1\n";
		else
		cout<<sum-min+2<<"\n";
	}
	return 0;
}
