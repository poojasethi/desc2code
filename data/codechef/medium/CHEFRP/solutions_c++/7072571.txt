#include<iostream>
#include<cstdio>
#include<string>

using namespace std;
int main()
{
    int t, A;
    int f=0;
    long int n,s=0;
    
    scanf("%d", &t);
    while(t--)
    {
        s=f=0;
        scanf("%ld",&n);
        int min=10010;
        for(int i=0;i<n;i++){
            scanf("%d",&A);
            s+=A;
            if(A<min) min=A;
            if(A<2) f=1;
        }
        if(f)
        printf("-1\n");
        else
        printf("%ld\n", s-min+2);
    }
    return 0;
}
