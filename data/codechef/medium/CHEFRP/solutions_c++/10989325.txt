#include <bits/stdc++.h>
using namespace std;
 
int main() {
	// your code goes here
	int t,i;
	cin>>t;
	while(t--)
	{
	    long int n,s=0,c=0;
	    cin>>n;
	    long int a[n];
	    for(i=0;i<n;i++)
	    cin>>a[i];
	    for(i=0;i<n;i++)
	    {
	        if(a[i]<2)
	        {
	        c=1;
	        break;}
	    }
	    sort(a,a+n);
	    for(i=n-1;i>0;i--)
	    s=s+a[i];
	    if(c==0)
	    cout<<s+2<<endl;
	    else
	    cout<<-1<<endl;
	}
	return 0;
}
