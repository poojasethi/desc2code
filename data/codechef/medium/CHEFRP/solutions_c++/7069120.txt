#include<iostream>
using namespace std;
int main(){
int t;
cin>>t;
while(t--)
{
    int n,sum=0,min=100000;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++){
        cin>>a[i];
        sum+=a[i];
        if(a[i]<min){
            min=a[i];
        }
    }
        if(min<2)
            cout<<"-1\n";
        else
            cout<<sum-min+2<<endl;
  }

return 0;
}
