#include <stdio.h>
#define ll long long

int main() {
    int z;
    scanf("%d", &z);
    while (z--) {
        int n;
        ll s = 0, m = 1LL<<60;
        scanf("%d", &n);
        for (int i = 0; i < n; i++) {
            ll v;
            scanf("%lld", &v);
            s += v;
            if (m > v) m = v;
        }
        printf("%lld\n", m < 2 ? -1: s - m + 2);
    }
}
