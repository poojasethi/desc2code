#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,i,sum,min,a,count;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		sum=0;
		min=10005;
		count=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a);
			if(a<2)
			{
				count=1;
			}
			else if(a<min)
			{
				min=a;
			}
			sum+=a;
		}
		if(count==1)
		printf("-1\n");
		else
		printf("%d\n",sum+2-min);
		
		
	}
}
