#define readint(n) c=gc();while(c < '0' || c > '9') c = gc();while(c >= '0' && c <= '9'){n =n*10 +c-48;c=gc();}
#define gc getchar_unlocked
#include<stdio.h>
char c;
int main()
{
	int t=0;
	readint(t);
	while(t--)
	{
		long long n=0,sum=0,x,min=1000000,i,flag=0;
		readint(n);
		for(i=0;i<n;++i)
		{
			x=0;
			readint(x);
			if(x<2) {flag=1;}
			sum+=x;
			if(x<min) min=x;
		}
		if(flag==1) printf("-1\n");
		else
		printf("%lld\n",sum-min+2);
	}
	return 0;
}
