#include<stdio.h>
int main(){
    int t,n,a,min,i,sum;
    scanf("%d",&t);
    while(t--){
     scanf("%d",&n);
     for(i=sum=0, min =100000;i<n;i++){
      scanf("%d",&a);
      sum+=a;
      if(a<min) 
       min = a;     
     }
     printf("%d\n",(min<2)?-1:sum-min +2);
    }
      
    return 0;
}
