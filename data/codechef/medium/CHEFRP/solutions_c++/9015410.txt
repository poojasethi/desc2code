//Chef And New Recipe

#include<stdio.h>
#include<algorithm>
#define maxi 100010
using namespace std;


int main()
{
    long long int t,arr[maxi],i,flag,n,sum;
    scanf("%lld",&t);
    while(t--)
    {
        flag=0;sum=0;
        scanf("%lld",&n);
        for(i=0;i<n;i++)
        {
            scanf("%lld",&arr[i]);
            if(arr[i]<2)
            {
                flag=1;
            }
        }
        if(flag)
            printf("-1\n");
        else
        {
            sort(arr,arr+n);
            for(i=n-1;i>=0;i--)
            {
                sum+=arr[i];
            }
            sum=sum-arr[0]+2;
            printf("%lld\n",sum);
        }
    }
    return 0;
}
