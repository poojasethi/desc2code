#include <stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,temp,min=0,sum=0,i;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&temp);
			sum+=temp;
			if(temp<min || i==0)
				min=temp;
		}
		if(min>2)
			sum-=(min-2);
		if(min<2)
			sum=-1;
		printf("%d\n",sum);
	}
	return 0;
}
