#include<bits/stdc++.h>
using namespace std;

int main()
{
    int tc; scanf("%d",&tc);
    while(tc--)
    {
        int n; scanf("%d",&n);
        int a[n];
        int sum=0;
        int m=INT_MAX;
        for(int i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
            sum+=a[i];
            m=min(m,a[i]);
        }
        if(m==1)
        {
            cout<<-1<<endl;
        }
        else
            cout<<sum-m+2<<endl;
    }
    return 0;
}
