#include<bits/stdc++.h>
using namespace std;
int arr[100005];
int mini;
int main()
{
	int t,n,sum,index,mark,i;
	scanf("%d",&t);
	while(t--)
	{
		mini=10000000;
		sum=0;
		index=-1;
		mark=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&arr[i]);
		}
		for(i=0;i<n;i++)
		{
			sum=sum+arr[i];
			if(arr[i]<2)
			{
				printf("-1\n");
				mark=1;
				break;
			}
			else
			{
			if(arr[i]<mini)
			{
			 	mini=arr[i];
			 	
			}	
			}
		}
		if(mark==0)
		{
			printf("%d\n",sum-mini+2);
		}
		
	}
}
