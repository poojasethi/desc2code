#include <iostream>
using namespace std;

int main() {
	// your code goes here
	long long int i,t,n,arr[1000000],counter,times,min;
	cin>>t;
	while(t--)
	   {
	   	cin>>n;
	   	for(i=0;i<n;i++)
	   	   {
	   	   	cin>>arr[i];
	   	   }
	   	 min=arr[0];
	   	 counter=0;
	   	for(i=0;i<n;i++)
	   	   {
	   	   	if(arr[i]<min)
	   	   	  {
	   	   	  	min=arr[i];
	   	   	  }
	   	   	counter=counter+arr[i];
	   	   	
	   	   }
	   	 if(min<2)
	   	   {
	   	   	cout<<"-1"<<endl;
	   	   }
	   	 else
	   	   {times=counter-min+2;
	   	   	cout<<times<<endl;
	   	   }
	   }
	return 0;
}
