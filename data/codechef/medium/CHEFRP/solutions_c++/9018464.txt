//Chef And New Recipe

#include<stdio.h>
#include<algorithm>
#define maxi 100010
using namespace std;


int main()
{
    int t,i,x,flag,n,sum,mini;
    scanf("%d",&t);
    while(t--)
    {
        flag=0;sum=0;
        scanf("%d",&n);
        scanf("%d",&x);
        mini=x;
        sum=x;
        for(i=1;i<n;i++)
        {
            scanf("%d",&x);
            sum+=x;
            if(x<mini)
                mini=x;
            if(x<2)
            {
                flag=1;//Should Never Break An Input
            }
        }
        if(flag)
            printf("-1\n");
        else
            printf("%d\n",sum-mini+2);

    }
    return 0;
}
