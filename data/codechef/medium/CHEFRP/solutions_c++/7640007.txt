#include <stdio.h>
#include <stdlib.h>
#include <iostream>

using namespace std;

int input()
{
    int n=0;
    char c=getchar_unlocked();
    while(c < '0' || c >'9') {c=getchar_unlocked();}
    while(c>='0' && c<='9')
    {
        n=(n<<3)+(n<<1)+c-'0';
        c=getchar_unlocked();
    }
    return n;
}

int main(){
	int T, N, temp;
	int m, flag;
	long long result = 0;
	T = input();
	while(T--){
		N = input();
		result = 0;
		m = 10001;
		flag = 0;
		while(N--){
			temp = input();
			result = (long long)(result + temp);
			m = min(m, temp);
			if(temp < 2){
				flag = 1;
			}
		}
		if(flag == 1){
			printf("-1\n");
			continue;
		}
		result = (long long)(result-(m-2));
		printf("%lld\n",result);
	}
}
