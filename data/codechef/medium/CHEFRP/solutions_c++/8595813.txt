#include<stdio.h>
#include<stdlib.h>

int main()
{
	int i,n,a,flag,t,min;
	long long int sum;
	scanf("%d",&t);
	while(t--)
	{
		
		scanf("%d",&n);
		flag=sum=0;
		min=100010;
		for(i=0;i<n;i++)
		{
			scanf("%d",&a);
			if(a>=2 && flag==0)
			{
				sum+=a;
				min= (a<min)?a:min;
			}
			else
			flag=1;
			
		}
		if(flag==0)
		printf("%lld\n",sum-min+2);
		else
		printf("-1\n");
		
		
	}
	return 0;
}
