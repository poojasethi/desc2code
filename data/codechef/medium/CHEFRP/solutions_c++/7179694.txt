#include<stdio.h>
#include<iostream>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long int n,i,min,m;
        scanf("%ld",&n);
        min=10001;
        long long int s=0,res=0;
        for(i=0;i<n;i++)
            {
                scanf("%ld",&m);
                if(m<min)
                    min=m;
                if(m<2)
                    res=-1;
                s+=m;
            }
        s=s-min+2;
        if(res==0)
            printf("%d\n",s);
        else
            printf("-1\n");
    }
    return 0;
}
