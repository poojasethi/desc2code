#include<bits/stdc++.h>
using namespace std;
int main()
{
    int tc,n,flag,a,rup,min1;
    cin>>tc;
    while(tc--)
    { flag=0;
     rup=0;
     min1=INT_MAX;
        cin>>n;
        for(int i=0;i<n;i++)
        {
            cin>>a;
            if(a<2)
                flag=1;
                rup=rup+a;
            if(a<min1)
                min1=a;

        }
        if(flag==1)
            cout<<"-1"<<"\n";
        else
            cout<<rup-min1+2<<"\n";
    }
}
