#include<stdio.h>
#include<algorithm>
int main() 
{
	int t;
	long int n, i;
	scanf("%d", &t);
	while(t--)
	{
		int flag=1;
		scanf("%ld",&n);
		long int A[n], sum=0;
		for(i=0; i<n; i++)
		{
			scanf("%ld", &A[i]);
			 if(A[i]<2)
			 	flag=0;
		}
		if(flag==0)
		 	printf("%d\n", -1);
		else
		{
			std::sort(A, A+n);
			for(i=1; i<n; i++)
			{
				sum=sum+A[i];
			}
		printf("%ld\n", sum+2);
		}
	}
	return 0;
}
