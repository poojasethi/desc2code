#include <iostream>
using namespace std;

int main() {
	// your code goes here'
	int t;
	cin>>t;
	while(t--){
	    int n,min;
	    long long int sum;
	    cin>>n;
	    int a[n];
	    cin>>a[0];
	    min = a[0];
	    sum = min;
	    for(int i=1;i<n; i++){
	      cin>>a[i];
	      sum+=a[i];
	      if(min > a[i]) min=a[i];
	    } 
	    if(min < 2) cout<<"-1\n";
	    else cout<<sum-min+2<<endl;
	    
	    
	}
	return 0;
}
