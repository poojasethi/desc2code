//Chef And New Recipe

#include<stdio.h>
#include<algorithm>
#define maxi 100010
using namespace std;


int main()
{
    long long int t,arr[maxi],i,flag,n,sum,mini;
    scanf("%lld",&t);
    while(t--)
    {
        flag=0;sum=0;
        scanf("%lld",&n);
        scanf("%lld",&arr[0]);
        mini=arr[0];
        sum=arr[0];
        for(i=1;i<n;i++)
        {
            scanf("%lld",&arr[i]);
            sum+=arr[i];
            if(arr[i]<mini)
                mini=arr[i];
            if(arr[i]<2)
            {
                flag=1;//Should Never Break An Input
            }
        }
        if(flag)
            printf("-1\n");
        else
            printf("%lld\n",sum-mini+2);

    }
    return 0;
}
