#include <iostream>
#include <assert.h>
 
using namespace std;
 
int main() {
std::ios::sync_with_stdio(false);
int t,n,x,minfreq;
long long sum;
cin>>t;
assert(1<=t && t<=10);
while(t--)
{
minfreq=99999999;
sum=0;
cin>>n;
assert(1<=n && n<=100000);
for(int i=0;i<n;i++)
{
cin>>x;
assert(1<=x && x<=10000);
sum+=x;
minfreq=min(minfreq,x);
 
}
if(minfreq<2) 
	cout<<"-1"<<"\n";
else
	cout<<sum-minfreq+2<<"\n";
}
 
return 0;
}
