#include<iostream>
#include<string>
#include<algorithm>
#include<ctype.h>
#include<vector>
#include<set>
#include<math.h>
#include<sstream>

using namespace std;

int main(){
  long long t,n,sum,freq;
    cin>>t;

  while(t--){
    cin>>n;
    vector<int> v;
    sum=0;
   for(int i=0;i<n;i++){
    int a; cin>>a;
    v.push_back(a);
    sum+=a;
   }
   sort(v.begin(),v.end());
   freq=v[0];
   if(freq==1){
    cout<<-1<<endl;
   }else{
    cout<<sum-freq+2<<endl;
   }
  }
  return 0;
}
