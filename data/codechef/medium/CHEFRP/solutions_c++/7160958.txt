#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
     long long t;
     scanf("%lld",&t);
     while(t--)
     {
         
         long long arr[100009];
         long long n,i,j;
         scanf("%lld",&n);
         for(i=0;i<n;i++)
         {
            scanf("%lld",&arr[i]);
         }
         
         long long sum=0;
         long long max=arr[0];
         
         for(i=0;i<n;i++)
         {
         
             if(arr[i]==1)
             break;
             else
             {
             
                 sum+=arr[i];
                 if(arr[i]<max)
                 max=arr[i];
             }
         }
         if(i!=n)
         printf("-1\n");
         else
         printf("%lld\n",sum+2-max);
     }
    
     

   return 0;
}
