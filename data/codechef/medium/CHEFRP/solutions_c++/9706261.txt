#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	long long sum;
	int n,t,i,flag;
	scanf("%d",&t);
	while(t--)
	{
		flag=0;
		sum=0;
		scanf("%d",&n);
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
			sum+=a[i];
		}
		sort(a,a+n);
		if(a[0]<2)
		{
			printf("-1\n");
		}
		else
		{
			sum=sum-a[0]+2;
			printf("%lld\n",sum);
		}
	}
	return 0;
}
