#include<iostream>
#include<climits>
using namespace std;

int minfreq(int a,int b)
{
	return a<b?a:b;
}

int main()
{
	int t,n,*a={0};
	cin>>t;
	for(int i=0;i<t;i++)
	{
		cin>>n;
		a=new int [n];
		for(int j=0;j<n;j++)
		cin>>a[j];
		
		int sum=0;
		int minfreq=INT_MAX;
		
		for(int j=0;j<n;j++)
		{
			sum+=a[j];
			minfreq=min(minfreq,a[j]);
			//cout<<"minfreq"<<minfreq<<" ";
		}
		
		if(minfreq==1)
		cout<<-1<<"\n";
		else
		cout<<(sum-minfreq+2)<<"\n";
	}
	
	return 0;
}
