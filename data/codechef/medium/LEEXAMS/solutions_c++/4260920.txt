#include<iostream>
#include<cmath>
#include<cstdlib>
#include<cstdio>
using namespace std;
int main()
{
    int t,n,m,a[51],b[51],i,j,k=1,flag[17],total,f;
    double answer,p[51],ans;
    cin>>t;
    while(t--)
    {
        cin>>n;
        answer=0;
        for(i=1;i<=n;i++)
        {
            cin>>p[i]>>a[i]>>b[i];
            p[i]/=100;
        }
        if(n>16)
            {
                printf("%0.9lf\n",answer);
                continue;
            }
        answer=0;
        total=pow(2,n);
        for(i=0;i<total;i++)
        {
            f=1;
            ans=1.0;
            for(j=1;j<=16;j++)
                flag[j]=0;
            for(j=1,k=1;j<=n;j++)
                {
            if(k&i)
            {
                if(flag[a[j]]==1)
                    { f=0; break;}
                flag[a[j]]=1;
                ans*=p[j];
            }
            else
            {
                if(flag[b[j]]==1)
                    { f=0;break;}
                flag[b[j]]=1;
                ans*=(1-p[j]);
            }
            k=k<<1;
            }
            if(f==1)
                {
                    answer+=ans;
                }
        }
        printf("%0.9lf\n",answer);
    }
    return 0;
}
