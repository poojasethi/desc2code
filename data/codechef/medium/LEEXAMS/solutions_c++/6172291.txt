#include <bits/stdc++.h>
using namespace std;
struct pree
{
	double val;
	int l,r;
};
int n;
pree a[101];
double memo[1<<16][18][100];
double prob(int mk,int pos,double pb,int p)
{
	int new_mk;
	double r=0;
	/*if(memo[mk][p][(int)pb*100]!=-1)
	{
		return memo[mk][p][(int)pb*100];
	}
	else
	{*/
		if(pos==(n+1))
		{
			r=pb;
			memo[mk][p][(int)pb*100]=r;
			return r;
		}
		else
		{
			if((mk&(1<<(a[pos].l-1)))&&(mk&(1<<(a[pos].r-1))))
			{
				r=r+prob(mk,pos+1,0,0);
			}
			if((mk&(1<<(a[pos].l-1)))==0)
			{
				new_mk=mk|(1<<(a[pos].l-1));
				r=r+prob(new_mk,pos+1,pb*a[pos].val,a[pos].l);
			}
			if((mk&(1<<(a[pos].r-1)))==0)
			{
				new_mk=mk|(1<<(a[pos].r-1));
				r=r+prob(new_mk,pos+1,pb*(1.0-a[pos].val),a[pos].r);
			}
			memo[mk][p][(int)pb*100]=r;
			return r;
		}
	//}
}
int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	double x;
	cin>>t;
	while(t--)
	{
		cin>>n;
		memset(a,0,sizeof(a));
		//memset(memo,-1,sizeof(memo));
		for(int i=0;i<33000;i++)
		{
			for(int j=0;j<18;j++)
			{
				for(int k=0;k<100;k++)
				{
					memo[i][j][k]=-1.0;
				}
			}
		}
		for(int i=1;i<=n;i++)
		{
			cin>>x;
			a[i].val=(x/100.0);
			cin>>a[i].l>>a[i].r;
		}
		if(n<=16)
		printf("%0.9lf\n",prob(0,1,1.0,16));
	else
		printf("0.000000000\n");
	}
}
