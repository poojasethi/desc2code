int tc , n , p[53] , x[53][2] , sol[20];
double ans;

void jalan(int idx)
{
     if (idx==n+1)
     {
                  int occ[17];
                  reset(occ,0);
                  
                  rep(i,1,n) occ[x[i][sol[i]]]++;
                  rep(i,1,16) if (occ[i]>1) return;
                  
                  double ret = 1.0;
                  rep(i,1,n) 
                  {
                             if (sol[i]==0) ret*= (double) p[i]/100;
                             if (sol[i]==1) ret*= (double) (100-p[i])/100;
                  }
                  ans+=ret;
                  return;
     }
     sol[idx] = 0;
     jalan(idx+1);
     sol[idx] = 1;
     jalan(idx+1);
}

              if (n>16) cout<<ans<<endl; else
              {
                        jalan(1);

              }
    }

}
