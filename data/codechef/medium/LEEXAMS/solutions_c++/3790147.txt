#include <iostream>
using namespace std;

int main() {
	float t,k,p[55],a[66],b[66];
	int n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		cin>>p[i]>>a[i]>>b[i];
		
		if(n>16){cout<<"0.0000000"<<endl;continue;}
		float tot=0;
		for(int i=0;i<(1<<n);i++)
		{
			float p1=1,vis[17]={0};
			int j;
			for(j=0;j<n;j++)
			{
				int tem;float ans;
				if(i&(1<<j))
				tem=b[j],ans=100-p[j];
				else
				tem=a[j],ans=p[j];
				if(vis[tem]==1)
				break;
				vis[tem]=1;p1*=(ans/100);
			}
			if(j==n)
			tot+=p1;
		}
		printf("%0.9f\n",tot);
	}
	
	return 0;
}
