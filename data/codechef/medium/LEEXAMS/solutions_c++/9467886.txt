#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define INF 0x3f3f3f3f
using namespace std;

int a[55],b[55],p[55];
double dp[55][1<<17];
int n;
double solve(int i,int mask)
{
    if(i==n)
        return 1;
    if(dp[i][mask]!=-1)
        return dp[i][mask];
    double ret=0.0;
    if(!(mask&(1<<a[i])))
        ret+=solve(i+1,mask|(1<<a[i]))*(p[i]/100.0);
    if(!(mask&(1<<b[i])))
        ret+=solve(i+1,mask|(1<<b[i]))*((100-p[i])/100.0);
    return dp[i][mask]=ret;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            scanf("%d%d%d",&p[i],&a[i],&b[i]);
        }
        fill_n(&dp[0][0],55*(1<<17),-1.0);
        printf("%0.6lf\n",solve(0,0));
    }
    return 0;
}
