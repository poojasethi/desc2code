#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
double myfunction(int i);
int n;
int prob[55];
int a[55];
int b[55];
double ans;
bool visited[17]={false};
double myfunction(int i)
{
    double p=0;
	if(i>=n)
	return 1;
	if(visited[a[i]]==false){
        visited[a[i]]=true;
     p=(prob[i]/100.00)*myfunction(i+1);
     visited[a[i]]=false;
     }
     if(visited[b[i]]==false){
          visited[b[i]]=true;
    p+=((100-prob[i])/100.00)*myfunction(i+1);
    visited[b[i]]=false;
    }
    return p;
	}

int main(){
	int t;
	scanf("%d",&t);
	while(t--){
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    scanf("%d %d %d",&prob[i],&a[i],&b[i]);
    if(n<=16){
      ans=myfunction(0);
      printf("%.9lf\n", ans);
      }
      		else {
			printf("%.9lf\n", 0.0);

    }
}
}
