#include<bits/stdc++.h>
using namespace std;
int a[17],b[17];
double p[17],x=100.0;
int used[17];
double getprob(int i,int n)
{
    double ans=0.0;
    if(i>n)
        return 1.0;
    if(!used[a[i]])
    {
    used[a[i]]=1;
    ans=(p[i]/x)*getprob(i+1,n);
    //printf("%lf\t",ans);
    used[a[i]]=0;
    if(!used[b[i]])
    {used[b[i]]=1;
    ans=ans+((x-p[i])/x)*getprob(i+1,n);
    used[b[i]]=0;
    }
    }
    else
    {
        if(!used[b[i]])
        {
            used[b[i]]=1;
            ans=((x-p[i])/x)*getprob(i+1,n);
            used[b[i]]=0;
        }
    }
return ans;
}

int main()
{
    int t,n,i;
    double ans=0.0;
    scanf("%d",&t);
    while(t--)
    {
        memset(used,0,sizeof(used));
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        scanf("%lf %d %d",&p[i],&a[i],&b[i]);
        if(n==1)
            {
            printf("1.000000000000\n");
            continue;
            }
        x=100.0;
        used[a[1]]=1;
        ans=(p[1]/x)*getprob(2,n);
       // printf("%lf\t",ans);
        used[a[1]]=0;
        used[b[1]]=1;
        ans=ans+((x-p[1])/x)*getprob(2,n);
        used[b[1]]=0;
        printf("%.9lf\n",ans);
    }
return 0;
}
