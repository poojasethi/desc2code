#include<bits/stdc++.h>
using namespace std;
int flag[18],a[52],b[52],n;
double p[52];
double solve(int i,double probability)
{
    if(i==n)
        return probability;
    double p1=0,p2=0;
    if(flag[a[i]]!=1)
    {
        flag[a[i]]=1;
        p1=solve(i+1,probability*(p[i]/100));
        flag[a[i]]=0;
    }
     if(flag[b[i]]!=1)
    {
        flag[b[i]]=1;
        p2=solve(i+1,probability*((100-p[i])/100));
        flag[b[i]]=0;
    }
    return p1+p2;
}
int main()
{
    int t,i;
    double ans;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
            scanf("%lf%d%d",&p[i],&a[i],&b[i]);
        if(n>16)
            printf("0\n");
        else
        {
            ans=solve(0,1);
            printf("%.10lf\n",ans);
        }
    }
    return 0;
}
