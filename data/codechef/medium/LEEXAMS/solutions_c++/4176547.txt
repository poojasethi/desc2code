#include<iostream>
#include<cstdio>
using namespace std;
struct ele
{
    int seen[20];
};
double solve(int a[],int b[],int p[],int i,int cur,int n,struct ele e)
{
    double a1,a2;
    if(i==n)
        return 1;
    else
    {
        e.seen[cur]=1;
        if(e.seen[a[i]]==0||i==0)
        {
            a1=((p[i]*1.0)/100)*solve(a,b,p,i+1,a[i],n,e);
        }
        else
            a1=0;
        if(e.seen[b[i]]==0||i==0)
        {
            a2=((100.0-p[i])/100)*solve(a,b,p,i+1,b[i],n,e);
        }
        else
            a2=0;
        return a1+a2;
    }

}

int main()
{
    int t,n,i;
    cin>>t;
    while(t--)
    {
        int a[100],b[100],p[100];
        struct ele e;
        for(i=0;i<20;i++)
            e.seen[i]=0;
        double ans=0;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>p[i];
            cin>>a[i];
            cin>>b[i];

        }

        if(n>16)
            printf("%0.9lf\n",ans);
        else
        {
            ans=solve(a,b,p,0,0,n,e);
            printf("%0.9lf\n",ans);
        }
    }
    return 0;
}
