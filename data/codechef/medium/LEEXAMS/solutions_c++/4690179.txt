#include <stdio.h>
#include <map>

using namespace std;

int m[55][5];
map<int, bool> mp;

double solve(int n, int i) {
	if(i == n)
		return 1;
	
	double p=0;
	
	if(!mp[m[i][1]]) {
		
		mp[m[i][1]] = true;
		p += solve(n, i+1) * (double)m[i][0]/100;
		mp[m[i][1]] = false;
	}
	
	if(!mp[m[i][2]]) {
		
		mp[m[i][2]] = true;
		p += solve(n, i+1) * (double)(100-m[i][0])/100;
		mp[m[i][2]] = false;
	}
	
	return p;
}

int main() {
	int t;
	scanf(" %d", &t);
	
	while(t--) {
		
		int n;
		scanf(" %d", &n);
		for(int i=0; i<n; i++) {
			scanf(" %d %d %d", &m[i][0], &m[i][1], &m[i][2]);
		}
		
		if(n > 16) {
			printf("0.00000000\n");
			continue;
		}
		
		printf("%.8lf\n", solve(n, 0));
	}
}
