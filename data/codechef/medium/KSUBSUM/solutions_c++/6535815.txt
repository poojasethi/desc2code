#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int merge(int *a, int *b, int k, int l, int sex){
	int i=0, j=0, v=0;
	int m[max(k+l, sex+1)];
	while(v< sex+1 && i<k && j<l){
		if(a[i]>b[j]){
			m[v++] = a[i++];
		}
		else m[v++] = b[j++];
	}
	if(i==k){
		while(j<l && v<sex+1) m[v++] = b[j++];
	}
	else if(j==l){
		while(i<k && v<sex+1) m[v++] = a[i++];
	}
	for(i = 0; i< v; i++) a[i] = m[i];
	return i;
}
int main(){
	int t;
//	cin>>t;
	scanf("%d", &t);
	while(t--){
		int f[2020];
		int s[2020];
		int n, p, q, r;
		scanf("%d %d %d %d", &n, &p, &q, &r);
		//cin>>n>>p>>q>>r;
		int sex = max(p, max(q, r));
		int a[n+1];
		int sum[n+1];
		sum[0]=0;
		for(int i=1; i<= n; i++){
		//	cin>>a[i];
			scanf("%d", &a[i]);
			sum[i] = sum[i-1] + a[i];
		}
		s[0] = 0;
		int y=0;
		for(int i = 1; i<= n; i++){
			int temp[2020];
			for(int j = 0; j< min(i, sex); j++){
				temp[j] = sum[i] - s[j];				
			}
			s[min(sex, i)] = sum[i];
			for(int j=min(sex, i); j>0 ; j--){
				if(s[j]<s[j-1]) swap(s[j], s[j-1]);
				else break;
			}
			y = merge(f, temp, min(sex, y), min(sex, i), sex);				
		}
	//	cout<<f[p-1]<<" "<<f[q-1]<<" "<<f[r-1]<<endl;
		printf("%d %d %d \n",f[p-1], f[q-1], f[r-1]); 
	}
	return 0;
}
