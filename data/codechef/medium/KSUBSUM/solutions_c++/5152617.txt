#define FORD(i,a,b) for(int i=(a),_b=(b);i>=_b;--i)
#define FOREACH(it,c) for(VAR(it,(c).begin());it!=(c).end();++it)
#define REP(i,n) FOR(i,0,n)
#define ALL(c) (c).begin(), (c).end()
#define SORT(c) sort(ALL(c))
#define REVERSE(c) reverse(ALL(c))
#define UNIQUE(c) SORT(c),(c).resize(unique(ALL(c))-(c).begin())

#define MOD 1000000007
int n,k1,k2,k3;
int a[1000000],ans[1000000],temp[1000000],cur[1000000],s[1000000];
int merge(int a[],int p,int b[],int q)
{
	int x=1,y=1,z=0;
	while(z<k3&&x<=p&&y<=q)
	{
		if(a[x]>b[y])
			temp[++z]=a[x++];
		else
			temp[++z]=b[y++];
	}
	while(z<k3&&x<=p)
		temp[++z]=a[x++];
	while(z<k3&&y<=q)
		temp[++z]=b[y++];
	FOR(i,1,z+1)
		a[i]=temp[i];
}
void solve()
{

	s[0]=0;
	int p1=0,p2=0,sum=0,f;
	s[++p1]=0;
	REP(i,n)
	{
		sum+=a[i];
		FOR(j,1,p1+1)
			cur[j]=sum-s[j];
		merge(ans,p2,cur,p1);
		p2=min(p1+p2,k3);
		for(f=p1+1;f>1&&sum<s[f-1];f--)
			s[f]=s[f-1];
		s[f]=sum;
		p1=min(p1+1,k3);
	}

}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		cin>>n>>k1>>k2>>k3;
		REP(i,n)
			cin>>a[i];
		solve();
		printf("%d %d %d\n",ans[k1],ans[k2],ans[k3]);
	}
	return 0;
}
