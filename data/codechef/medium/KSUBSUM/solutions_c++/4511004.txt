#include <cstdio>
#include <algorithm>
#include <functional>

using namespace std;

int i, j, k;
int T, N, k1, k2, k3, fn, ofn, w = 0;
int arr[10001], sum[10001], fs[2][2013];

int main()
{
	scanf("%d", &T);
	while(T--)
	{
		scanf("%d %d %d %d", &N, &k1, &k2, &k3);
		for(i = 0; i < N; ++i)
			scanf("%d", arr + i);
		fn = 0;
		for(i = 0; i < N; ++i)
		{
			for(j = i; j; --j)
				if(sum[j - 1] < 0)
					sum[j] = sum[j - 1] + arr[i];
				else
					break;
			sum[j] = arr[i];
			for(--j; j >= 0; --j)
				sum[j] += arr[i];
			ofn = fn;
			w = !w;
			for(j = 0, k = 0, fn = 0; (j < ofn || k <= i) && fn < 2013; ++fn)
				if(j < ofn && (k > i || fs[!w][j] > sum[k]))
					fs[w][fn] = fs[!w][j++];
				else
					fs[w][fn] = sum[k++];
		}
		printf("%d %d %d\n", fs[w][k1 - 1], fs[w][k2 - 1], fs[w][k3 - 1]);
	}
	return 0;
}
