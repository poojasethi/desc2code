#include <iostream>
#include<queue>

int main() {
	int t;

while(t--)
{
	int n,k1,k2,k3;
	cin>>n>>k1>>k2>>k3;

	arr[0]=0;
	for(int i=1;i<=n;i++)
	{
		int temp;
		cin>>temp;
		arr[i]=arr[i-1]+temp;
		
		
	}
	
priority_queue<int, std::vector<int>, std::greater<int> > my_min_heap;
	for(int i=1;i<=n;i++)
	{
		for(int j=i;j<=n;j++)
		{int temp=arr[j]-arr[i-1];
			if(my_min_heap.size()<k3)
			my_min_heap.push(arr[j]-arr[i-1]);
			else
			{
			if(my_min_heap.top()<temp)
			{
				my_min_heap.pop();
				my_min_heap.push(temp);
				
			}
				
				
			}
	}	}
	   int i=1;;

	    {

	   	y[f++]=my_min_heap.top();
	    	my_min_heap.pop();
