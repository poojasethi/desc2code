int a[N],b[N];
priority_queue<int, vector<int>, std::greater<int> >pq;
int main()
{
    ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    int t,n,k1,k2,k3;
    int sum=0;
    i(t);
    while(t--)
    {
        i(n);i(k1);i(k2);i(k3);
        f(i,1,n)    i(a[i]);
        f(i,1,n)
        {
            sum=0;
            f(j,i,n)
            {
                sum+=a[j];
                if(pq.size()>=k3)
                {
                    if(pq.top()<sum)
                    {
                        pq.pop();
                        pq.push(sum);
                    }
                }
                else    pq.push(sum);
            }
        }
        int k=k3;
        while(!pq.empty())
        {
            b[k--]=pq.top();
            pq.pop();
        }
        o(b[k1]);os;o(b[k2]);os;o(b[k3]);on;
    }



    return 0;
}
