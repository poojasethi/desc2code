#include<cstdio>
#include<queue>

using namespace std;

int main(int argc, const char* args[])
{
	int T, N, K1, K2, K3;
	int arrayVal;
	int S[10001];
	int SortedSubArray[2013];
	scanf("%d", &T);
	while (T > 0)
	{
		S[0] = 0;
		scanf("%d %d %d %d", &N, &K1, &K2, &K3);
		for (int i = 1; i <= N; ++i)
		{
			scanf("%d", &arrayVal);
			S[i] = S[i - 1] + arrayVal;
		}

		int temp;
		priority_queue<int, vector<int>, greater<int> > pqueue;
		for(int i = 0; i < N ; ++i)
		{
			for (int j = i + 1; j < N + 1; ++j)
			{
				temp = S[j] - S[i];
				if (pqueue.size() < K3)
				{
					pqueue.push(temp);
				}
				else
				{
					if (pqueue.top() < temp)
					{
						pqueue.pop();
						pqueue.push(temp);
					}
				}
			}
		}
		
		for (int i = 0; i < K3; i++)
		{
			SortedSubArray[i] = pqueue.top();
			pqueue.pop();
		}

		printf("%d %d %d\n", SortedSubArray[K3 - K1], SortedSubArray[K3 - K2], SortedSubArray[0]);
		T--;
	}
	return 0;
}
