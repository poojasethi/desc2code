#include<iostream>
#include<algorithm>
#include<stdio.h>

using namespace std;

int arr1[4096];
int arr2[4096];
int arr3[4096];
int smal[2048];

int merge(int* a1, int s1, int* a2, int s2, int* res)
{
	int it1 = 0;
	int it2 = 0;
	for(int i=0; i<s1 + s2; ++i)
	{
		if(it1 >= s1)
		{
			res[i] = a2[it2++];
		}
		else if(it2 >= s2)
		{
			res[i] = a1[it1++];
		}
		else
		{
			if(a1[it1] >= a2[it2])
				res[i] = a1[it1++];
			else
				res[i] = a2[it2++];
		}
	}
	int siz = s1 + s2 >= 2012?2012:s1 + s2;
	return siz;
}

//Insert while maintaining descending order
int insert(int* arr, int siz, int val)
{
	for(int i=0; i<siz; ++i)
	{
		if(val > arr[i])
		{
			int temp = val;
			val = arr[i];
			arr[i] = temp;
		}
	}
	arr[siz] = val;
	if(siz + 1 >=2012)
		return 2012;
	else
		return siz + 1;
}

		
int main()
{
	int t;
	int* temp;
	int* prev;
	int* merg;
	int tsi;
	int psi;
	int msi;
	int ssi;
	scanf("%d", &t);
	for(int i=0; i<t; ++i)
	{
		int n, k1, k2, k3;
		scanf("%d %d %d %d", &n, &k1, &k2, &k3);

		prev = arr1;
		psi = 0;

		temp = arr2;
		tsi = 0;

		merg = arr3;
		msi = 0;

		smal[0] = 0;
		ssi = 1;

		int pr = 0;
		for(int j=0; j<n; ++j)
		{
			int tt;
			scanf("%d", &tt);

			int te = tt + pr;
			pr = te;

			tsi = 0;
			for(int k=0; k<ssi; ++k)
			{
				temp[tsi++] = te + smal[k];
			}

			msi = merge(temp, tsi, prev, psi, merg);
			ssi = insert(smal, ssi, -te);

			//Swapping
			int* tempptr = merg;
			int tempsi = msi;
			merg = prev;
			msi = psi;
			prev = tempptr;
			psi = tempsi;
			//Swapping over
		}
		printf("%d %d %d\n", prev[k1 - 1], prev[k2 - 1], prev[k3 - 1]);
	}
	return 0;
}
