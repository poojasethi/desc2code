#include<cstdio>
 
#define rep(i,n) for(int i=0;i<(n);i++)
 
using namespace std;
 
const int INF=1<<29;
 
int merge(int m,int *a,int n,int *b){
	int i=0,j=0,t=0;
	static int c[2012];
	while(t<2012 && (i<m || j<n)){
		if     (i==m) c[t++]=b[j++];
		else if(j==n) c[t++]=a[i++];
		else if(a[i]>b[j]) c[t++]=a[i++];
		else               c[t++]=b[j++];
	}
	rep(k,t) a[k]=c[k];
	return t;
}
 
int main(){
	int T; scanf("%d",&T);
	while(T--){
		int n,k1,k2,k3; scanf("%d%d%d%d",&n,&k1,&k2,&k3); k1--; k2--; k3--;
		int a[10000];
		rep(i,n) scanf("%d",a+i);
 
		int sum[10001];
		rep(i,n) sum[i+1]=sum[i]+a[i];
 
		int n_ans=0,n_mn=1;
		int ans[2012],mn[2013]={0};
 
		rep(i,n){
			int cand[2012];
			rep(j,n_mn) cand[j]=sum[i+1]-mn[j];
 
			n_ans=merge(n_ans,ans,n_mn,cand);
 
			rep(j,n_mn+1) if(j==n_mn || mn[j]>sum[i+1]) {
				for(int k=n_mn-1;k>=j;k--) mn[k+1]=mn[k];
				mn[j]=sum[i+1];
				break;
			}
			if(n_mn<2012) n_mn++;
		}
 
		printf("%d %d %d\n",ans[k1],ans[k2],ans[k3]);
	}
 
	return 0;
}
