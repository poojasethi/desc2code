#include<stdio.h>
#include<math.h>
#define Int long long
int isPaln(Int n,Int base)
{
    Int i,j;
    for(i=1;i<=n;i*=base);
    i=i/base;
    for(j=1;i>j;j*=base,i/=base)
    {
        if((n/i)%base!=(n/j)%base)return 0;
    }
    return 1;
}
int main()
{
    int t;Int n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld",&n);
        if(n==1||n==3||n==5)
        {
            printf("2\n");continue;
        }
        if(n==2||n==4)
        {
            printf("3\n");continue;
        }

        Int i,s=sqrt(n);
        for(i=2;i<=s;i++)
        {
            if(isPaln(n,i))break;
        }
        if(i<=s)
        {
            printf("%lld\n",i);
        }
        else
        {
            for(i=s;i>0;i--)
            {
                s=n/i-1;
                if(n%s==n/s)
                {
                    printf("%lld\n",s);break;
                }

            }
        }
    }
    return 0;
}
