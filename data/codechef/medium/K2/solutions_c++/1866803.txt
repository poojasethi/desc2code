#include<stdio.h>
#include<cmath>

using namespace std;
long long temp,county,rem;
int findpal(long long n,long long b,long long pow)
{
while(pow>1)
{
rem=n%b;
county=n/pow;
if(county==rem)
{
n=(n-(county*pow)-rem)/b;
pow/=b*b;
}
else
return 0;
}
return 1;
}
long long solve(long long n)
{
long long j;
j=2;
long long limit=(long long)sqrt(n);
for (j=2;j<= limit;++j) {
temp=1;
while(temp<=n)
{
temp=temp*j;
}
temp/=j;
if(findpal(n,j,temp))
{
return j;
}
}
 
for (int a=j-1; a>=1; a--) {
temp=n/a;
if(temp!=1||temp!=0)
if(n%(temp-1)==a)
return temp-1;
}
return n+1;
}
int main()
{
int t;
long long n;
scanf("%d",&t);
for(int i=0;i<t;i++)
{
scanf("%lld",&n);
if(n==1)
printf("%d\n",2);
else
{county=solve(n);
printf("%lld\n",county);}
}
}
