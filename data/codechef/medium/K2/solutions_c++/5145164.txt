#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;


bool checkPalin( long long n, long long base )
{
    long long nCopy1 = n;
    long long nCopy2 = n;
    long long lastDigit;
    long long firstDigit;

    long long powerBase = 1;
    while( powerBase <= n )
    {
        powerBase *= base;
    }
    powerBase /= base;

    while( true )
    {
        lastDigit = nCopy1 % base;
        nCopy1 /= base;
        firstDigit = nCopy2 / powerBase;
        nCopy2 -= firstDigit * powerBase;
        powerBase /= base;

        if( lastDigit != firstDigit )
            return false;

        if( nCopy2 == nCopy1 )
            return true;
    }

    return true;

}
int main()
{
    int test;
    scanf( "%d", &test );
    long long num;

    while( test-- )
    {
        scanf( "%lld", &num );
        if( num == 1 )
        {
            printf( "%d\n", 2 );
            continue;
        }

        bool found = false;
        long long limit = sqrt( num + 1 );
        long long i;
        for( i = 2; i <= limit ; ++i )
        {
            if( checkPalin( num, i ) )
            {
                printf( "%lld\n", i );
                found = true;
                break;
            }
        }

        if( !found )
        {
            for( i = num / ( i + 1 ); i >= 1; --i )
            {
                if( num % i == 0 )
                {
                        found = true;
                        printf( "%lld\n", ( num / i ) - 1 );
                        break;
                }
            }
        }
    }
}
