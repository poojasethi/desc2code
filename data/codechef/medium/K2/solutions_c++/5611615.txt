int checkpalindrome(long long n,long long i)
{
	long long rem=0,sum=0,num=n;;
	
		while(n>0)
		{
			rem=n%i;
			sum=sum*i+rem;
			n=n/i;
		}
		if(num==sum)
		return 1;
		else
		return 0;
	
}
