#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;


bool checkPalin( long long n, long long base )
{
    long long lastDigit;
    long long firstDigit;

    long long baseSqr = base * base;
    long long powerBase = 1;
    while( powerBase <= n )
    {
        powerBase *= base;
    }
    powerBase /= base;

    while( powerBase > 1 )
    {
        lastDigit = n % base;
        firstDigit = n / powerBase;

        if( lastDigit == firstDigit )
        {
            n = ( n - ( firstDigit * powerBase ) ) / base;
            powerBase /= baseSqr ;
        }
        else
        {
            return false;
        }
    }

    return true;
}

int main()
{
    int test;
    scanf( "%d", &test );
    long long num;
    bool found;
    long long limit;
    long long i;

    while( test-- )
    {
        scanf( "%lld", &num );
        if( num == 1 )
        {
            printf( "%d\n", 2 );
            continue;
        }

        found = false;
        limit = sqrt( num + 1 );

        for( i = 2; i <= limit ; ++i )
        {
            if( checkPalin( num, i ) )
            {
                printf( "%lld\n", i );
                found = true;
                break;
            }
        }

        if( !found )
        {
            for( i = num / ( i + 1 ); i >= 1; --i )
            {
                if( num % i == 0 )
                {
                    printf( "%lld\n", ( num / i ) - 1 );
                    break;
                }
            }
        }
    }
}
