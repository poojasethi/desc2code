unsigned long long int method(unsigned long long int n){
	if(n==1||n==3||n==5)return 2;
	if(n==2||n==4)return 3;
	if(n==6)return 5;
	int result = 0;
	for(int j=2;j<=(sqrt(double(n))+1e-8);j++){
		if(baseconversion(n,j))
			return j;
		if(n%j==0){
			int r = n/j-1;
			if(r!=j)result = r;
		}
	}
	if(result!=0)return result;
	return n-1;
}
