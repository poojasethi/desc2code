#include<stdio.h>



int is_palin(long long int n,long long int b)
{
     long long int x[50];
     long long int c=0;
    
     while(n>0)
     {
               x[c]=n%b;
               c++;
               n=n/b;
     }
     long long int l=0;
     long long int r=c-1;
     while(l<r)
     {
               if(x[l]!=x[r])
               return 0;
               l++;
               r--;
     }
     return b;
}
