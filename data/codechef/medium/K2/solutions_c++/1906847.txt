using namespace std;

int ar[50];

int generate(unsigned long long int n, unsigned long long int b)
{
    int i = 0;
    while(n > 0) {
        i++;
        ar[i-1] = n%b;
        n /= b;
    }
    return i;
}

bool ispalin(int i)
{
    int j;
    if(i % 2 == 0) {
        for(j = 1; j <= i/2; j++) {
            if(ar[j-1] != ar[i-j]) return false;
        }
    }
    else {
        for(j = 0; j < (i-1)/2; j++) {
            if(ar[j] != ar[i-1-j]) return false;
        }
    }
    return true;
}
