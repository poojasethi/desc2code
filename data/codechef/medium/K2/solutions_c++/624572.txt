#include<iostream>
#include<utility>
#include<list>
#include<vector>
#include<cmath>
#include<algorithm>
typedef long long int ll;
using namespace std;
int main()
{
    int t;
    ll base,idiv;
    scanf("%d",&t);
    while(t--)
    {
       ll n;
       scanf("%lld",&n);
       ll temp;
       if(n<=2)
        base=n+1;
       else
       {
       base=0;
       idiv=1;
       ll s=sqrt(n);
       ll rem;
       for(ll i=2;i<=s;i++)
       {
        temp=n;
        ll rev=0;
        while(temp>=i)
        {
         rem=temp%i;
         rev=(rev+rem)*i;
         temp/=i;
         }
        rev+=temp;
        if(rev==n)
        {
         base=i;
         break;
         }
        ll s=(n/i)-1;
        if(n%i==0&&n%s!=0)
         idiv=i;
       }
      if(base==0)
       base=(n/idiv)-1;
      }
      printf("%lld\n",base);
    }
    return 0;
}
