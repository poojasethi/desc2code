#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,l,j,r,sum=0;
    while(1)
    {
        cin>>n;
        if(n==0)
        break;
        
        int a[n];
        for(i=0;i<n;i++)
        cin>>a[i];
        sort(a,a+n);
        for(i=n-1;i>=2;i--)
        {
            l=0;
            j=i-1;
            while(l<j)
            {
                if((a[j]+a[l])<a[i])
                {
                    sum+=j-l;
                    l++;
                }
                else
                j--;
            }
        }
        cout<<sum<<"\n";
        sum=0;
    }
    return 0;
}
