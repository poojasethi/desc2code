#include<iostream>
#include<algorithm>
#include<vector>
#include<stdio.h>
using namespace std;

int main()
{
    int N;
    scanf("%d",&N);
    while(N!=0)
    {
        vector<long int> v;
        int i,j,k;
        long int num,sum=0;
        for(i=0;i<N;i++)
        {
            scanf("%ld",&num);
            v.push_back(num);
        }
        sort(v.begin(),v.end());
        for(i=0;i<N-2;i++)
        {
            k=i+2;
            for(j=i+1;j<N-1;j++)
            {
                for(;k<N;k++)
                {
                    if(v[i]+v[j]<v[k])
                    {
                        sum+=(N-k);
                        break;
                    }
                }
            }
        }
        printf("%ld\n",sum);
        scanf("%d",&N);
    }
    return 0;
}
