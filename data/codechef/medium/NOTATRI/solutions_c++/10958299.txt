int main()
{
	while(true){
        int n,j,k,c=0;
        cin>>n;
        if(n==0)
            break;
        int i=0,a[n];
        while(i<n){
            cin>>a[i];
            i++;
        }
        sort(a,a+n);
        for(i=n-1;i>=2;i--){
            j=i-1;
            k=0;
            while(k<j){
                if(a[i]>a[j]+a[k]){
                    c+=j-k;

                }

                    j--;  
            }
        }
        cout<<c<<"\n";
