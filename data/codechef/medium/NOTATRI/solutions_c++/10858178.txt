#include<bits/stdc++.h>
#include<algorithm>
using namespace std;
 
int main() {
    while(1) {
    	int n,temp,count=0,j,k;
    	scanf("%d",&n);
    	if(!n)
    	    break;
    	    
    	vector<int> arr;
    	
    	for(int i=0;i<n;i++) {
    		scanf("%d",&temp);
    		arr.push_back(temp);
    	}
    	
        sort(arr.begin(),arr.end());
        
        for(int i=n-1;i>1;i--) {
        	k=i-1;
        	for(j=0;j<k;j++) {
        		while((j<k)&&(arr[j]+arr[k]>=arr[i]))
        		    k--;
        		count+=k-j;    
        	}
        }
        printf("%d\n",count);
    }   	
}
