#include <iostream>
#include<bits/stdc++.h>
using namespace std;
long long a[2003];
long long b_search(long long num,int low,int high)
{
    long long mid=(low+high)/2;
    while(low<=high)
    {
        if(a[mid]>=num)
        {
           high=mid-1;
        }
        else
        {
            low=mid+1;
        }
        mid=(low+high)/2;
    }
    if(a[mid]>num)
    return mid;

    return mid+1;
}
int main() {

	long long n;
	while(1)
	{cin>>n;
	    if(n==0)
	    return 0;
	long long i,j,count=0,k;
	 for(i=0;i<n;i++)
	 {
	     cin>>a[i];
	 }
	 sort(a,a+n);
	 for(i=n-1;i>=2;i--)
	 {   
	     for(j=i-1;j>=1;j--)
	     {
	         
	         long long num=a[i]-a[j];
	       if(num>a[0])
	        count=count+b_search(num,0,j-1);
	     }
	 }
	 cout<<count<<"\n";
	}
	return 0;
}
