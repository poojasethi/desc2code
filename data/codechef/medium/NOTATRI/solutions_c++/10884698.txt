#include <bits/stdc++.h>

using namespace std;

int main()
{
	while (1) {
		int n;
		scanf("%d",&n);
		if (n == 0)
			break;
		int a[n];
		for (int i = 0; i < n; i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		int ctr = 0;
		for (int i = n-1; i >= 0; i--)
		{
			int j = 0;
			int k = i-1;
			while (j < k)
			{
				if (a[j]+a[k] < a[i])
					ctr += (k-j);
				if (a[j]+a[k] < a[i])
					j++;
				else if (a[j]+a[k] >= a[i])
					k--;
			}
		}
		printf("%d\n",ctr);
	}
	return 0;
}
