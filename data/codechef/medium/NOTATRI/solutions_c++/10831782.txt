#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int t;ios::sync_with_stdio(false);
    int x;
    cin>>x;
    while(x!=0)
    {
        long long int arr[x];
        for(int i=0;i<x;i++)
        {
            cin>>arr[i];
        }
        int count = 0;
        sort(arr,arr+x);
        for(int i=0;i<x-2;i++)
        {
            int j=i+1;
            int c = j+1;
            //cout<<c<<endl;
            while(j<x-1)
            {
                //cout<<arr[i]<<' '<<arr[j]<<' '<<arr[c]<<endl;
                while(c<x&&(arr[i]+arr[j]>=arr[c]))
                    c++;
              //  cout<<c<<endl;
                count+=x-c;
                j++;
            }
        }
        cout<<count<<endl;
        cin>>x;
    }
}
