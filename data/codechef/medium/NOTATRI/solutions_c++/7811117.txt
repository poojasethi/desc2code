inline int inp()
{
    register int r=0,c;
    for(c=getchar_unlocked(); c<=32; c=getchar_unlocked());
    if(c=='-')
        return -inp();
    for(; c>32; r=(r<<1)+(r<<3)+c-'0',c=getchar_unlocked());
    return r;
}
int main()
{
    int n; n = inp();
    while(n!=0)
    {
        int a[n],ans=0;
        for(int i=0;i<n;i++)
            a[i] = inp();
        sort(a,a+n);
        for(int i=0;i<n;i++)
        {
            for(int j=i+1;j<n;j++)
            {
                if(a[i]+a[j]<a[n-1])
                    ans += n - (upper_bound(a,a+n,a[i]+a[j]) - a);
            }
        }
        print(ans);
        n = inp();
    }
    return 0;
}
