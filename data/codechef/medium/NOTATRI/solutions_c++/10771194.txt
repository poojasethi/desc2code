#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	while(1){
		long long N;
		cin >> N;
		
		if(!N)
			return 0;
		long long arr[N];
		for(int i=0;i<N;i++)
			cin >> arr[i];
		sort(arr,arr+N);
		long long ans = 0;
		for(long long i=0;i<N-2;i++){
			int k = i+2;
			for(long long j=i+1;j<N-1;j++){
				long long sum = arr[i]+arr[j];
				while(arr[k] <= sum && k < N)
					k++;
				ans += k-j-1;
			}
		}
		long long total = N*(N-1)*(N-2)/6;
		cout << total - ans << endl;
	}
	return 0;
}
