    #define gc getchar_unlocked
    #define pc putchar_unlocked
    #define p(n) printf("%li\n",n);
    #define s(n) scanf("%d",&n);
    #define S(n) scanf("%li",&n);
     inline unsigned int fastread()
    {
      unsigned int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
    using namespace std;
    int main()
    {
                int n;
           n=fastread();
       while(n!=0)
       {

           long int a[n];
           for(int i=0;i<n;i++)
               a[i]=fastread();
               sort(a,a+n);
        long int c=0;
        for(long int j=n-1;j>1;j--)
            {
                long int i=0;
                long int k=j-1;
                while(k>i)
                {
                    if(a[k]+a[i]<a[j])
                        {
                            c=c+k-i;
                            i++;
                        }
                        else
                            k--;
                }
            }
               p(c)
               s(n)
       }
        return 0;
    }
