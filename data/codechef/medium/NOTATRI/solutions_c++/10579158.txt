    #include<stdio.h>
    #include <stdio.h>
    #include <vector>
    #include<algorithm>
    #define gc getchar_unlocked
    #define pc putchar_unlocked
    #define p(n) printf("%li\n",n);
    #define s(n) scanf("%d",&n);
    #define S(n) scanf("%li",&n);
    using namespace std;
    int main()
    {
                int n;
           s(n);
       while(n!=0)
       {

           long int a[n];
           for(int i=0;i<n;i++)
               S(a[i])
               sort(a,a+n);
        long int c=0;
        for(long int j=n-1;j>1;j--)
            {
                long int i=0;
                long int k=j-1;
                while(k>i)
                {
                    if(a[k]+a[i]<a[j])
                        {
                            c=c+k-i;
                            i++;
                        }
                        else
                            k--;
                }
            }
               p(c)
               s(n)
       }
        return 0;
    }
