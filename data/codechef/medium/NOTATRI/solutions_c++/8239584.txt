#include<bits/stdc++.h>
using namespace std;
int main(){
	int  n;
	int arr[20000];
	while(1){
		scanf("%d",&n);
		if(n==0){
			break;
		}
		for(int i=0;i<n;i++){
			scanf("%d",&arr[i]);
		}
		int cnt=0;
		sort(arr,arr+n);
		for(int k=n-1;k>1;k--){
			int i=0,j=k-1;
			while(i<j){
				if(arr[k]>(arr[j]+arr[i])){
					cnt+=(j-i);
					i++;
				}
				else{
					j--;
				}
			}
		}
		printf("%d\n",cnt);
	}
	return 0;
}
