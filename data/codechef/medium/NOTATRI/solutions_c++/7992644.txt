#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,i,j,k,count=0;
	scanf("%d",&n);
	if(n!=0)
	{
		while(n!=0)
		{
			long int l[n];
			for(i=0;i<n;i++)
			scanf("%ld",&l[i]);
			sort(l,l+n);
		for(i=n-1;i>1;i--)
		{
			j=i-1;
			k=0;
			while(k<j)
			{
				if(l[k]+l[j]<l[i])
				{
					count+=j-k;
					k++;
				}
				else
				j--;
			}
		}
			printf("%d\n",count);
			count=0;
			scanf("%d",&n);
		}
	}
	return 0;
}
