#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n;
    scanf("%d",&n);
    while(n){
        int arr[n];
        for(int i=0;i<n;i++){
            scanf("%d",&arr[i]);
        }
        sort(arr,arr+n);
        int ans=0;
        for(int i=n-1;i>1;i--){
            int j=0,k=i-1;
            while(j<k){
                if(arr[i]>arr[j]+arr[k]){
                    ans+=(k-j);
                    j++;
                }
                else
                {
                    k--;
                }
            }
        }
        printf("%d\n",ans);
        scanf("%d",&n);
    }
}
