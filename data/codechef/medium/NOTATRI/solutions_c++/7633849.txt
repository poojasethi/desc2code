#include<bits/stdc++.h>
using namespace std;
int main(){
	int N;
	cin>>N;
	while(N){
		int arr[N];
		for(int i=0;i<N;i++){
			cin>>arr[i];
		}
		sort(arr,arr+N);
		int first,end,count=0;
		for(int i=N-1;i>=2;i--){
			first=0;
			end=i-1;
			while(first<end){
				if(arr[first]+arr[end]<arr[i]){
					count+=end-first;
					first++;
				}
				else
				end--;
			}
		}
		cout<<count<<endl;
		cin>>N;
	}
	return 0;
}
