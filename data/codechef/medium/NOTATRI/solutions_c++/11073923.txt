#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n;
	while(1)
	{
		cin>>n;
		if(n==0)
		break;
		int a[n],i,j,k,c=0;
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		for(i=n-1;i>=2;i--)
		{
			j=i-1;
			k=0;
			while(k<j)
			{
				if(a[k]+a[j]<a[i])
				{
					c+=j-k;
					k++;
				}
				else
				j--;
			}
		}
		cout<<c<<endl;
	}
	return 0;
}
