int main()
{
	int i,j,n,num,ans;
	vector<int> v(2001);
	scanf("%d",&n);
	while(n!=0)
	{
		ans=0;
		map<int,int> m;
		map<int,int> f;
		map<int,int>::iterator it;
		for(i=0;i<n;i++)
		{
			scanf("%d",&(v[i]));
			m[v[i]]++;
		}
		it=m.end();
		i=0;
		for(it=m.begin();it!=m.end();it++)
		{
			f[it->first]=n-i;
			i+=(it->second);
		}		
		for(i=0;i<n-1;i++)
		{
			for(j=i+1;j<n;j++)
			{
				it=f.upper_bound(v[i]+v[j]);
				if(it!=f.end())
				{
					ans=ans+(it->second);
				}
			}
		}
		printf("%d\n",ans);
		scanf("%d",&n);
	}
	return 0;
}
