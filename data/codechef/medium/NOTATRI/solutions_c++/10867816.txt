#include<cstdio>
#include<algorithm>
using namespace std;
long long a[20004];
int main()
{
	long long n,i,j,k,temp,p,ans;
	while(1)
	{
		ans=0;
		scanf("%lld",&n);
		if(n==0)
		break;
		for(i=0;i<n;i++)
		scanf("%lld",&a[i]);
		sort(a,a+n);
		for(i=n-1;i>1;i--)
		{
			j=0;
			k=i-1;
			while(j<k)
			{
				if(a[i]>(a[j]+a[k]))
				{
					ans+=(k-j);
					j++;
				}
				else
				{
					k--;
				}
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
