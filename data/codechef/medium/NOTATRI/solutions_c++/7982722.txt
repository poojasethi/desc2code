#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t){
        int i,count=0,arr[20001],x,r;
        for(i=0;i<t;i++)
           cin>>arr[i];
        sort(arr,arr+t);
        for(i=t-1;i>1;i--){
            x=i-1;
            r=0;
            while(r<x){
                if(arr[r]+arr[x]<arr[i]){
                    count+=x-r;
                    r++;
                }
                else
                 x--;
            }
        }
        cout<<count<<endl;
        cin>>t;
    }
    return 0;
}
