#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int n;
    scanf("%d", &n);
    while(n != 0)
    {
        int arr[n], count = 0, low = 0, high = 0;
        for(int i = 0; i < n; i++)
        {
            scanf("%d", &arr[i]);
        }
        sort(arr, arr + n);

        for(int i = n - 1; i > 1; i--)
        {
            low = 0;
            high = i - 1;
            while(low < high)
            {
                if(arr[low] + arr[high] < arr[i])
                {
                    count += high - low;
                    low++;
                }
                else
                {
                    high--;
                }
            }
        }
        printf("%d\n", count);
        scanf("%d", &n);
    }
    return 0;
}
