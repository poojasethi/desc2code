#include<stdio.h>
#include<algorithm>
using namespace std;

int main()
{
	int n;
	int count;
	int i,x,y;
	while(1)
	{
		scanf("%d",&n);
		if(n==0)
			return 0;
		count=0;
		
		int a[n];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		for(i=n-1;i>1;i--)
		{
			x=0;
			y=i-1;
			while(y>x)
			{
				if(a[x] + a[y] < a[i])
				{
					count+= y-x;
					x++;
				}
				else
				 	y--;
			}
		}
		printf("%d \n",count);
	}
	return 0;
}
