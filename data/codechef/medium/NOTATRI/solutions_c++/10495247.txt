#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	while(1)
	{
		int n;
		cin>>n;
		if(n==0)
		break;
		long int a[n];
		long long ans;
		//ans=ans/6;
		//
		if(n%2==0)
		{
			ans=n/2;
			if((n-1)%3==0)
			{
				ans=ans*((n-1)/3);
				ans=ans*(n-2);
			}
			else if((n-2)%3==0)
			{
				ans=ans*((n-2)/3);
				ans=ans*(n-1);
			}
			else
			{
				ans=ans/3;
				ans=ans*(n-1)*(n-2);
			}
			
		}
		else
		{
			ans=(n-1)/2;
			if((n-2)%3==0)
			{
				ans=ans*((n-2)/3);
				ans=ans*(n);
			}
			else if(n%3==0)
			{
				ans=ans*(n/3);
				ans=ans*(n-2);
			}
			else
			{
				
				ans=ans/3;
				ans=ans*(n)*(n-2);
			}
		}
	//	cout<<ans;
		for(int i=0;i<n;i++)
		cin>>a[i];
//		long long ans=(n)*(n-1)*(n-2);
//		ans=ans/6;
//		cout<<ans;
		sort(a,a+n);
		int i,j;
		long long int count=0;
		for(int i=0;i<n-2;i++)
		{
			int k=i+2;
			for(int j=i+1;j<n-1;j++)
			{
				while(k<n&&a[i]+a[j]>=a[k])
				k++;
				count+=k-j-1;
			}
		}
		ans=ans-count;
		cout<<ans<<"\n";
		
		
	}
}
