int scanint(){
	register int c = getchar();
	int x=0;
	bool neg=false;
	for(;(c<48 || c>57) && c!=45 ;c=getchar());
	if(c==45){
		neg=true;
		c=getchar();
	}
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	if(neg==true)
	x=x*(-1);
	return x;
}

long long scanlong()
{
	register int c = getchar();
	long long x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

typedef long long ll;
const int mod = 1e9 + 7 ;
const int inv = 280000002;

ll modulo(int n){
	if(n==0)
		return 1;
	ll ans = modulo(n/2);
	
	if(n&1){
		ans = (ans*ans) % mod;
		ans = (ans*26) % mod;
	}else{
		ans = (ans*ans) % mod ;
	}
	return ans ;
}

int main(){
	int t,n;
	ll ans,temp;
	t = scanint();
	while(t--){
		n = scanint();
		ans = modulo(n/2);
		if(n & 1){
			temp = (ans * 26) % mod ;
			ans = (ans - 1 + mod) % mod ;
			ans = (ans * 52) % mod ;
			ans = (ans * inv) % mod ;
			ans = (ans + temp) % mod ;
		}else{
			ans = (ans - 1 + mod) % mod ;
			ans = (ans * 52) % mod ;
			ans = (ans * inv) % mod ;
		}
		cout<<ans<<endl;
	}
}
