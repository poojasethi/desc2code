#include<stdio.h>

#define MOD 1000000007
unsigned long long int power(unsigned long long int x, unsigned long long int y)
{
 unsigned long long  int temp;
    if( y == 0)
        return 1;
    temp = power(x, y/2);
    if (y%2 == 0)
        return (temp*temp)%MOD;
    else
        return (x*((temp*temp)%MOD)%MOD);
}
