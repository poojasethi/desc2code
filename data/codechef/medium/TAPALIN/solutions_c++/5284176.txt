#define mod 1000000007

ll big_mod(ll a,ll b)
{
	if(b==0)
		return 1;
	if(b%2==0){
    llu k=big_mod(a,b/2)%mod;
    return (k*k)%mod;
	}
	else
		return (big_mod(a,b-1)%mod*(a%mod))%mod;
}

   //freopen("input.txt","r",stdin);

	K=big_mod(25,mod-2);

		scanf("%lld",&n);
		l=n/2;

		p=(K*p)%mod;

		{

			r=big_mod(26,l+1);
			p=(p+r)%mod;
		}

	}
	return 0;
}
