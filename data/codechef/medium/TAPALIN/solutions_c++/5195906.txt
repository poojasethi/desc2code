#include<iostream>
#define mod 1000000007
using namespace std;
long long int pow(int a,long long int i)
{
    long long int k;
    if(i==1)
        return a;
    if(i==0)
        return 1;
    k=pow(a,i/2);
    if(i%2==0)
        return (k*k)%mod;
    else
        return ((k*k)%mod*a)%mod;
}
int main()
{
    int t,i;
    long long int s,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n%2==0)
            s=((52*(pow(26,n/2)-1))%mod*pow(25,1000000005)%mod)%mod;
       else
            s=(((52*(pow(26,n/2)-1))%mod*pow(25,1000000005)%mod)%mod+pow(26,(n+1)/2))%mod;
            //s=s>0?s:-s;
        cout<<s<<endl;
//cout<<pow(26,50);
    }
    return 0;
}
