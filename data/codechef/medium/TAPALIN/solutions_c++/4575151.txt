 unsigned long long solve(unsigned long long k,unsigned long long n)
 {
     unsigned long long x=k,y=1;
     while(n)
     {
         if(n%2)
            y=product(x,y);
         x=product(x,x);
         n/=2;
     }
     return y%MOD;
 }
 unsigned long long product(unsigned long long x, unsigned long long y)
 {
     unsigned long long a=0,b=x%MOD;
     while(y)
     {
         if(y%2)
            a=(a+b)%MOD;
         b=2*(b%MOD);
         y/=2;
     }
     return a%MOD;
 }
