#include <stdio.h>
#define S(X) scanf("%lld",&X)
#define MOD 1000000007

long long int power(long long int n)
{
    long long int b,r;
    b=26;r=1;
    while(n>0)
    {
        if(n%2!=0)
        r=(r*b)%MOD;
        b=(b*b)%MOD;
        n=n/2;
    }
    return r;
}

int main()
{
    long long t, n, i, p, q;
    S(t);
    while(t--){
        S(n);
        // calculate 26^(n+1)/2 - 1
        p = 1;
        p = power((n+1)/2);
        p=(p+MOD-1)%MOD;

        if(n%2==0){
            p*=52;
        }else{
            p*=27;
        }
        p%=MOD;

        // now we have to divide p by 25. (1/25)%MOD = 25^(MOD-2) % MOD
        q = 1;

        long long int mod2=MOD-2, in = 25;
        while(mod2>0)
        {
            if(mod2%2!=0)
                q=(q*in)%MOD;
            in=(in*in)%MOD;
            mod2=mod2/2;
        }

        p = (p*q)%MOD;
        if(n%2==1) p-=1;
        printf("%lld\n",p);
    }
    return 0;
}
