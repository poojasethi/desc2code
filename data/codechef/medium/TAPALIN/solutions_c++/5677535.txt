 #include<iostream>
    #define mod 1000000007
    long long inv=280000002;
    using namespace std;
    unsigned long long pow(int p)
    {
    unsigned long long a;
    if(p==0)
    return 1;
    if(p==1)
    return 26;
    a=pow(p/2);
    return (((a*a)%mod)*pow(p%2))%mod;
    }
    long long s(int n)
    {
    return ((2*((26*(inv))%mod)%mod)%mod*(pow(n)-1))%mod;
    }
     
    int main()
    {
    int t;
    cin>>t;
    while(t--)
    {
    unsigned long long n,sum=0;
    cin>>n;
    if(n%2==0)
    sum=s(n/2);
    else
    sum=(s(n/2)+pow(n/2+n%2))%mod;
    cout<<sum<<endl;
    }
    return 0;
    }
