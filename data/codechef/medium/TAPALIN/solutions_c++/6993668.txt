#include<iostream>
using namespace std;
#define M 1000000007
class Palindrom{
	long long int n;
	public:
		Palindrom()
		{
			    long long int ans,ans2;
			    cin>>n;
				long long int i=n/2;
				ans=power(26,i+1);
				ans=(ans+M-26)%M;
				ans2=power(25,M-2);
				ans=2*ans*ans2%M;
				if(n%2!=0)
				ans+=power(26,n/2+1);
				cout<<ans%M<<endl;
		}
		long long int power(long long int a,int long long b)
		{
			if(b==0)
			return 1;
			long long int res=power(a,b/2)%M;
			if(b%2==0)
			return res*res%M;
			else
			return res*res%M*a%M;
		}
};

	}

}
