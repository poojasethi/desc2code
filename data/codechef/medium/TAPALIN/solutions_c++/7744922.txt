typedef long long LL;
typedef unsigned long long ULL;

using namespace std;

LL mod = 1000000007;

LL bigmod(LL b,LL p)
{
    if(p == 0)return 1;
    LL ans = bigmod(b,p/2);
    ans *= ans;
    ans %= mod;
    if(p & 1)
    {
        ans *= b;
        ans %= mod;
    }
    return ans;
}

LL fun(LL x)
{
    LL lob = bigmod(26,x+1) - 1;
    if(lob < 0)lob += mod;
    LL hor = 25;
    LL ans = (lob * bigmod(hor,mod-2)) % mod;
    ans--;
    if(ans < 0)ans += mod;
    ans *= 2;
    ans %=mod;
    return ans;
}
