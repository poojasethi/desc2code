#include<stdio.h>
#define MOD 1000000007
long long  modulo(long long  a,long long  b)
{
long long ans;
if(b==0)
return 1;
long long c=modulo(a,b/2);
ans=(c*c)%MOD;
if(b%2)
ans=(((a*c)%MOD)*c)%MOD;
return ans;
}
 
int main()
{
long long int t,n,i,p,temp;
scanf("%lld",&t);
while(t--)
{
long long ans,inv;
scanf("%lld",&n);
p=(n+1)/2;
ans=modulo(26,p);
temp=ans;
ans=(ans-1+MOD)%MOD;
ans=(52*ans)%MOD;
inv=modulo(25,MOD-2);
ans=(ans*inv)%MOD;
if(n%2)
ans=(ans-temp+MOD)%MOD;
printf("%lld\n",ans);
}
}
