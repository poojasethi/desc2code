#include<stdio.h>
 
# define MOD 1000000007
 
 
long long int power(long long int base,long long int exponent)
{
    long long int result = 1;
    while (exponent > 0)
       {
 
        if (exponent%2== 1)
           result = (result * base) %MOD;
        exponent=exponent >> 1;
        base = (base * base)%MOD;
       }
    return result;
 
   }
