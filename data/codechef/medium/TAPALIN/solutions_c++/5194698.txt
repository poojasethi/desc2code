llu pow(llu num,llu n)
{
	llu res =1;
	while(n>0)
	{
		if(n&1)
		{
			res=(res*num)%MOD;
		}
		n=n>>1;
		num=(num*num)%MOD;
	}
	return res;
}


long long inv(long long n){
    return pow(n,MOD-2);
    
}

int main()
{
	int t;
	llu n,res,p,ext,inv25;
	inv25=inv(25)%MOD;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%llu",&n);
		p=(n/2);
		res=pow(26,p);
		if(n&1)
		ext=(26*(res))%MOD;
		res=(2*26*(res-1+MOD))%MOD;
		res=(res*inv25)%MOD;
		if(n&1)
		{
			res=(res+ext)%MOD;
		}
		printf("%llu\n",res);
		
	}
}
