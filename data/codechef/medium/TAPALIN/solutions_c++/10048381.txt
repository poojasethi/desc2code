#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL MOD = 1e9 + 7;
LL power(LL A, LL B){
    LL ans = 1;
    while(B > 0){
        if(B & 1LL){
            ans = (ans * A) % MOD;
        }
        A = (A * A) % MOD;
        B >>= 1;
    }
    return ans;
}
int main(){
    int t;
    cin >> t;
    while(t--){
        LL N;
        scanf("%lld", &N);
        LL ans = 0;
        LL p = power(26, ((N + 1) >> 1));
        LL q = power(26, N/2);
        ans = (26 * ((q + MOD - 1) % MOD)) % MOD;
        ans = (ans * 2) % MOD;
        ans = (ans * power(25, MOD - 2)) % MOD;
        if(N & 1LL)
            ans = (ans + p) % MOD;
        printf("%lld\n", ans);
    }
    return 0;
}
