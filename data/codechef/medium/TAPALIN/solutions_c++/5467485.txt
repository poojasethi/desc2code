#include <iostream>
using namespace std;
#define mod 1000000007
long long pw(long long a,long long b)
{
	long long res=1;
	while(b>0){
		if(b&1)res=(res*a)%mod;
		a=(a*a)%mod;
		b/=2;
	}
	return res%mod;
}
int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
		long long n;
		cin>>n;
		long long inv=280000002;
		long long k=(n+1)/2;
		long long p=pw(26,k);
		long long ans=52*(p-1)%mod*inv%mod;
		if(n%2)ans=(ans+mod-p)%mod;
		cout<<ans<<endl;
	}
	return 0;
}
