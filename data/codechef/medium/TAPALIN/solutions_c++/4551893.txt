#include <cstdio>
using namespace std;

const int mod = 1000000007;

int exp(int a)
{
	if(!a)
	{
		return 1;
	}
	long long b = exp(a/2);
	b *= b;
	b %= mod;
	if(a % 2)
	{
		return (b * 26) % mod;
	}
	else
	{
		return b;
	}
}

int main()
{
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		long long out = 0, temp2 = exp(n/2);
		if(n % 2)
		{
			out += (temp2*26)%mod;
		}
		temp2--;
		while(temp2 % 25)
		{
			temp2 += mod;
		}
		temp2 /= 25;
		temp2 *= 52;
		out += temp2;
		out %= mod;
		printf("%d\n", out);
	}
    return 0;
}
