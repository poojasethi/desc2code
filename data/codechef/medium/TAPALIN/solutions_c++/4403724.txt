#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#define mod 1000000007
using namespace std;
struct matrix
{
	long long a[2][2];
	matrix()
	{
		memset(a,0,sizeof(a));
	}
	
	void unit_matrix()
	{
		a[0][0]=26; a[0][1]=0;
		a[1][0]=1; a[1][1]=1;
	}
	
	matrix operator * (matrix b)
	{
		matrix c;
		memset(c.a,0,sizeof(c.a));
		for(int k=0;k<2;k++)
		 for(int i=0;i<2;i++)
		  for(int j=0;j<2;j++)
		    c.a[i][j]=(c.a[i][j]+a[i][k]*b.a[k][j])%mod;
		    return c;
	}
};
long long pow(int x,long long n)
{
	if(n==0) return 1;
	long long res=pow(x,n/2);
	res=(res*res)%mod;
	
	if(n%2) res=(res*x)%mod;
	return res;
}
matrix pow_matrix(matrix a,long long n)
{
	if(n==1) return a;
	
	matrix b=pow_matrix(a,n/2);
	b=b*b;
	if(n%2) b=(b*a);
	return b;
}
main()
{   std:ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    long long n;
    matrix a;
    while(t--)
    {
    	cin>>n;
    	a.unit_matrix();
    	a=pow_matrix(a,(n+1)/2);
    	long long res=(a.a[1][0]*26)%mod;
    	
    	res=(res*2)%mod;
    	
    	if(n%2) res=(res-pow(26,(n+1)/2))%mod;
    	res=(res+mod)%mod;	
       	printf("%lld\n",res);
    }
	return 0;
}
