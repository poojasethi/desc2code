#include<iostream>
#include<cstring>
typedef long long ll;
using namespace std;
ll mod=1000000007;
long long mul_inv(int a, int b)
{
	ll b0 = b, t, q;
	ll x0 = 0, x1 = 1;
	if (b == 1) return 1;
	while (a > 1) {
		q = a / b;
		t = b, b = a % b, a = t;
		t = x0, x0 = x1 - q * x0, x1 = t;
	}
	if (x1 < 0) x1 += b0;
	return x1;
}

ll power(ll a, ll b)
{
    if(b==0) return 1;
    long long temp=power(a,b/2);
    if(b&1) return (((temp*temp)%mod)*a)%mod;
    else return (temp*temp)%mod;
}
int main()
{
    int t;
    long long n;
    cin>>t;
    ll den=mul_inv(25,mod);
    while(t--)
    {
              cin>>n;
              ll pp=power(26,(n+1)/2);
              ll ans=(((((pp-1+mod)%mod)*26)%mod)*den)%mod;
              ans=(ans*2)%mod;
              if(n&1)
              {
                     ans=(ans-pp+mod)%mod;
              }
              cout<<ans<<endl;
    }
    return 0;
}
