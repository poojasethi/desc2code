#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<stdbool.h>
using namespace std;
long long int power(long long int base,int pow)
{
     long long int prod=1;
     while(pow)
     {
         if(pow & 1)
              prod=(prod*base)%1000000007;
         base=(base*base)%1000000007;
         pow>>=1;
     }
     return prod;
}
int main()
{
      int T;
      long long int N,palno,p;
      scanf("%d",&T);
      while(T>0)
      {
           scanf("%lld",&N);
           p=power(26,N/2);
           palno=(((p-1)%1000000007)*power(25,1000000005))%1000000007;
           palno=(52*palno)%1000000007;
           if(N%2==1)
           {
               p=power(26,(N+1)/2);
               palno=(palno+p)%1000000007;
           }
           printf("%lld\n",palno);
           --T;
      }
      return 0;
}
