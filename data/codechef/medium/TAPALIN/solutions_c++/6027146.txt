#include<iostream>
#define MOD 1000000007
using namespace std;
typedef long long int ulli;
ulli modPow(ulli base, ulli exp, ulli mod)
{
    ulli ans = 1;
    while(exp)
    {
        if(exp%2)
        {
            ans = (ans* base)%mod;
        }
        base = (base * base)%mod;
        exp/=2;
    }
    return ans;
}
ulli sumGP(ulli r, ulli n, ulli mod)
{
    if(n == 0) return 1%mod;
	else if(n == 1) return (1%mod + r%mod)%mod;
	else if(n%2 == 1) return (sumGP(r,1,mod)%mod * sumGP(modPow(r,2,mod)%mod,(n-1)/2,mod)%mod)%mod;
	else return (1%mod +  (r%mod * sumGP(r,1,mod)%mod * sumGP(modPow(r,2,mod),(n-2)/2,mod)%mod)%mod)%mod;
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        if(n==1)cout<<"26\n";
        else if(n==2)cout<<"52\n";
        else
        {
            ulli terms = n/2;
            ulli sum = sumGP(26,terms-1,MOD);
            sum = sum * 52;
            sum = sum % MOD;
            if(n%2==0)//even
            {
                cout<<sum<<endl;
            }
            else
            {
                ulli N = (n+1)/2;
                sum = sum + modPow(26,N,MOD);
                sum = sum % MOD;
                cout<<sum<<endl;
            }
        }
    }
}
