#include<iostream>

#include<cmath>

using namespace std;

const int md=1000000007;
typedef long long doublelong;

int square_by_expoenentiation_and_mod(int a,int b) // a^b
{	
	int product=1;
	while(b)
	{
		if(b%2) 
		{
			product=doublelong(product)*a%md;
		}
		if(b/=2)
		{
			a=doublelong(a)*a%md;
		}
	}
	
	return product;
}

int find_sum(int n)
{
	int modinverse25=square_by_expoenentiation_and_mod(25,md-2);
	
	int k;
	if(n%2==0)
	{
		k=n/2;
	}
	else
	{
		k=(n+1)/2 ;
	}
	int p26=square_by_expoenentiation_and_mod(26,k);
	
	int ans=(doublelong(52))*(p26-1)%md*modinverse25%md;
	
	if(n%2)
	{
	ans=(ans+md-p26)%md;
	}
	return ans;
}
