#include <iostream>
using namespace std;
typedef long long ll;
int main()
{
	ll t,dollars,temp,cntmax,cents,mc,total,currmax,cnt;
	scanf("%lld",&t);
	while(t--)
	{
		cnt=0;
		cntmax=0;
		scanf("%lld %lld %lld",&dollars,&cents,&mc);
		total=dollars*100+cents;
		currmax=total;
		while(mc<=total)
		{
			if(cents<mc)
			{
				dollars--;
				cents+=100;
			}
			cents-=mc;
			temp=dollars;
			dollars=cents;
			cents=temp;
			total=dollars*100+cents;
			cnt++;
			if(total>currmax)
			{
				cntmax=cnt;
				currmax=total;
			}
			if(cnt==10000)
				break;
		}
		printf("%lld\n",cntmax);
	}
	return 0;
}
