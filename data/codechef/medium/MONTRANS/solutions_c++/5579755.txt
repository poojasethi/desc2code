#include <iostream>
using namespace std;

int main() 
{
int t;
cin>>t;
while(t--)
 {
 int a,a1,b1,b,c,temp,count=0,max=0;
 cin>>a>>b>>c;
 a1 = a;
 b1 = b;
 while(a*100+b>=c && count<=1000)
   {
   if(b>=c)
     b=b-c;
   else
     {
     a--;
     b=b+100-c;
     }
   temp=a;
   a=b;
   b=temp;
   count++;
   if(a*100+b>a1*100+b1)
     {
     max = count;
     a1 = a;
     b1 = b;
     }
   }
 cout<<max<<endl;
 }
return 0;
}
