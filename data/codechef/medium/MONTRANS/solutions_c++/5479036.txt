#include<bits/stdc++.h>
using namespace std;
int check[10001];
int main()
{
	int t,a,b,c,i,f,j,l,maxa,maxb,maxc,temp;
	cin>>t;
	while(t--)
	{
	for(i=0;i<=10000;i++)
		check[i]=0;
		cin>>a>>b>>c;
		int u=a*100+b;
		check[u]=1;
		maxa=a;
		maxb=b;
		maxc=0;
		i=0;
		while(1)
		{
			i++;
			if(b>=c)
			{
				temp=a;
				a=b-c;
				b=temp;
				if((a*100+b)>(maxa*100+maxb))
				{
					maxa=a;
					maxb=b;
					maxc=i;
				}
				if(check[a*100+b]==1)
					break;
				else
					check[a*100+b]=1;
			}
			else if(b<c && a>0)
			{
				temp=a-1;
				a=b+100-c;
				b=temp;
				if((a*100+b)>(maxa*100+maxb))
				{
					maxa=a;
					maxb=b;
					maxc=i;
				}
				if(check[a*100+b]==1)
					break;
				else
					check[a*100+b]=1;
			}
			else
			{
				break;
			}
		}
		cout<<maxc<<"\n";
	}
	return 0;
}
