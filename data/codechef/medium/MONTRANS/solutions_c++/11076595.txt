#include <stdio.h>

int main()
{
	int TST,tst,a,b,c,m,mink,maxm,k;
	scanf("%d",&TST);
	for(tst=0;tst<TST;tst++)
	{
		scanf("%d%d%d",&a,&b,&c);
		mink=0;
		maxm=100*a+b;
		for(k=1;k<10000;k++)
		{
			m=100*a+b;
			if(m<c) break;
			m-=c;
			a=m%100;
			b=m/100;
			if(maxm<100*a+b)
			{
				maxm=100*a+b;
				mink=k;
			}
		}
		printf("%d\n",mink);
	}
	return 0;
}
