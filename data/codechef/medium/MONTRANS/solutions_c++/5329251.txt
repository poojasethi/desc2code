#include <iostream>
#include <cstdio>
#define db double

using namespace std;

int main()
{
	int ans,count,temp,t,a,b,c;
	db max,ol,n_no;
	
	scanf("%d",&t);
	for (int k = 0; k < t; k++) {
		scanf("%d%d%d",&a,&b,&c);
		ol = (db)a + (db)b/100;
		ans = 0;
		max = 0.0;
		count = 0;
		while ((a > 0 || b >= c) && (count < 10000)) {
			if (b >= c) {
				b = b-c;
				temp = b;
				b = a;
				a = temp;
			}
			else {
				b = b + 100;
				a--;
				b = b-c;
				temp = b;
				b = a;
				a = temp;
			}
			count++;
			n_no = (db)a+(db)b/100;
			if (n_no - ol > max) {
				ans = count;
				max = n_no - ol;
			}
		}
		printf("%d\n",ans);
	}
}
