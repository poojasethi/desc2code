#include <iostream>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while (t--) {
        int d, c, factor, t1, t2, count, pos, max;

        cin >> d >> c >> factor;

        count = 0;
        pos = 0;
        max = (d * 100) + c;
        while (count <= 10000) {
            t1 = (d * 100) + c;
            if (t1 < factor)
                break;

            t1 -= factor;
            d = t1 % 100;
            c = t1 / 100;
            count++;

            t1 = (d * 100) + c;
            if (t1 > max) {
                max = t1;
                pos = count;
            }
            //cout << count << " " << d << " " << c << endl;
        }

        cout << pos << endl;
    }

    return 0;
}
