#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;

int main(){
    int t,number = 0,i = 1,maxi = 0,a,b,c,money[10000];
    scanf("%d",&t);

 while(t--){
    scanf("%d %d %d",&a,&b,&c);
    money[0] = a*100 + b;
    maxi = money[0];
    do{
        if(b < c){
            if(a > 0)
                a--;
            b = b + 100 - c;
        }
        else
            b = b - c;

        swap(a,b);
        money[i] = a*100 + b;
        if(money[i] > maxi){
            maxi = money[i];
            number = i;
        }
        i++;
    }while(money[i - 1] >= c && i<= 10000);

    printf("%d\n",number);
    number = 0,i = 1,maxi = 0;
 }
  return 0;
}
