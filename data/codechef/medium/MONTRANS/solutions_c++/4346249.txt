#include<iostream>

using namespace std;

int main(){
int t,a,b,c,max_a=0,max_b=0,count,count_max,temp;
cin>>t;
while(t--){
        count=count_max=max_a=max_b=0;
    cin>>a>>b>>c;
    while((100*a+b)>=c&&count<10000){
           //cout<<a<<" "<<b<<"\n";
            if((100*max_a+max_b)<(100*a+b)){
                max_a=a;
                max_b=b;
                count_max=count;
            }
        b-=c;
        if(b<0){
            b+=100;
            a--;
        }
    temp=a;
    a=b;
    b=temp;
        count++;
    }
cout<<count_max<<"\n";
}
return 0;
}
