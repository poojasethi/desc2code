#include<stdio.h>
int main(){
   // freopen("input.txt","r",stdin);
    //freopen("output.txt","w",stdout);
	int T,A,B,C,tc;
	scanf("%d",&T);
	for(tc=0;tc<T;tc++){
		int maxcount=0,count=0,max=0,sum;
		scanf("%d %d %d",&A,&B,&C);
		max=A*100+B;
		while((A>0||B>C)&&count<10000){
			count++;
			if(B>=C)
			    B-=C;
			else
			if(C>B && A>0){
				A-=1;
				B+=100;
                B-=C;
			}
			int temp=A;
			A=B;
			B=temp;
			sum=A*100+B;
		//	printf("%d -> (%d)",temp2,count);
				
			if(sum>max){
				maxcount=count;
				max=sum;
			}
		}
		printf("%d\n",maxcount);
	}
	return 0;
}
