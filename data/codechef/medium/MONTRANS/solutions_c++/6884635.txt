#include <bits/stdc++.h>

using namespace std;

#define s(x) scanf("%d", &x)

int main()
{
    int t, d, c, m, cnt, ans, tmp, md, mc;
    s(t);
    while (t--) {
        s(d);
        s(c);
        s(m);
        cnt = 20000;
        md = d;
        mc = c;
        tmp = ans = 0;
        while (cnt--) {
            if (c < m && d == 0)
                break;
            if (c < m) {
                d--;
                c += 100;
                c -= m;
                swap(d, c);
            }
            else {
                c -= m;
                swap(d, c);
            }
            //cout << d << " " << c << endl;
            ans++;
            if (d > md) {
                md = d;
                mc = c;
                tmp = ans;
            }
            else if (d == md && c > mc) {
                mc = c;
                tmp = ans;
            }
        }
        printf("%d\n", tmp);
    }
    return 0;
}
