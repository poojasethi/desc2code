int main()
{
	int t, a, b, c;
	scanf( "%d", &t );
	while( t -- )
	{
		scanf( "%d%d%d", &a, &b, &c );
		int mx = a * 100 + b, i = 0, ans = 0;
		while( ++ i < 10000 && a * 100 + b >= c )
		{
			if( a && b < c )
				a --, b += 100;
			b -= c;
			swap( a, b );
			if( a * 100 + b > mx )
				mx = a * 100 + b, ans = i;
		}
		printf( "%d\n", ans );
	}
	return 0;
}
