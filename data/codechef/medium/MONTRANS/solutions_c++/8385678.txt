#include <cstdio>

int main() {
	#ifdef _LOCAL_SYSTEM_
		freopen("in", "r", stdin);
	#endif
	int t;
	scanf("%d", &t);

	while(t--) {
		int A, B, C, mi=0, total, max;
		scanf("%d %d %d", &A, &B, &C);
		max=total=A*100+B;
		for(int i=1; i<10000; i++) {
			if(total<C)
				break;
			total-=C;
			total=(total%100)*100+total/100;
			if(total>max) {
				max=total;
				mi=i;
			}
		}
		printf("%d\n", mi);
	}

	return 0;
}
