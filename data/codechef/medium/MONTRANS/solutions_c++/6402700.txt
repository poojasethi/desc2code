#include<bits/stdc++.h>
using namespace std;
int main(){
	int t,a,b,c,ntimes=0;
	scanf("%d",&t);
	while(t--){
		int best=0,curr=0;
		scanf("%d %d %d",&a,&b,&c);
		curr=(a*100)+(b);
		best=curr;
		 ntimes=0;
		 //cout<<"initial : best=curr= "<<best<<endl;
		for(int i=1;i<=10000;i++){
			if(a>0){
				//cout<<"a>0\n";
				if(b<c){
				//	cout<<"b<c\n";
					a=a-1;
					b=b+100;
				}
				//interchanging notes
				int temp=a;
				a=b-c;
				b=temp;
				//cout<<"a ="<<a<<"b= "<<b<<endl;
				curr=(a*100)+(b);
				//cout<<"curr : "<<curr<<endl;
				if(curr>best) {
					best=curr;
					ntimes=i;
					//cout<<"best :"<<best<<" ntimes : "<<ntimes<<endl;
				}
			}
			else if (b>c) {
				//cout<<"a=0 & b>c\n";
				a=(b-c);
				b=0;
				curr=(a*100)+b;
				if(curr>best) {
					best=curr;
					ntimes=i;
				}
				//cout<<"a ="<<a<<"b= "<<b<<"curr="<<curr<<"best ="<<best<<endl;

			}
			else {
				//cout<<"a=0 && b<c no transaction breAking\n";
				break;
			}
			
		}
		printf("%d\n",ntimes);
	}
	return 0;
}
