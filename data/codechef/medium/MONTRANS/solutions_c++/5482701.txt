#include<iostream>
using namespace std;
#include<cstdio>
int main()
{
    int rs,p,t,sb;
    cin>>t;
    while(t--)
    {
        scanf("%d %d %d",&rs,&p,&sb);
        rs=rs*100+p;
        int i=0;
        int mx=rs,loc=0;
        while(rs>sb&&i<10001)
        {
            i++;
            rs=rs-sb;
            p=rs%100;
            rs=(p*100)+(rs/100);
            if(mx<rs)
            {
                loc=i;
                mx=rs;
            }
        }
        cout<<loc<<endl;
    }

}
