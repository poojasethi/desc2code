inline void get(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

int main()
{
	int T;
	get(T);
	for (int __rep = 1; __rep <=T; __rep++)
	{
		int doll, cent, cost;

		get(doll);	
		get(cent);	
		get(cost);

		LL m = doll*100 + cent;
		int trans = 0, mtrans = 0;

		while ((doll > 0 || cent >= cost) && trans < 10000)
		{
			if (cent < cost)
			{
				cent += 100;
				doll--;
			}

			cent -= cost;
			swap(doll,cent);
			trans++;

			if (m < (doll*100+cent))
			{
				m =  doll*100+cent;
				mtrans = trans;
			}
		}

		printf("%d\n", mtrans);
	}
	return 0;
}
