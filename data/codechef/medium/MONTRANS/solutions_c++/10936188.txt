#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int a,b,c;
		cin >> a >> b >> c;
		int ma = a;
		int mb = b;
		int ctr = 0;
		int ans = 0;
		while (ctr < 10000)
		{
			ctr++;
			int f = 0;
			if (b >= c)
			{
				f = 1;
				b = b-c;
			}
			else if (b < c && a > 0)
			{
				f = 1;
				a = a-1;
				b = b+100;
				b = b-c;
			}
			if (f) {
				int temp = a;
				a = b;
				b = temp;
			//	cout << a << " " << b << endl;
				if (a > ma)
				{
					//cout << a << " " << b << endl;
					ma = a;
					mb = b;
					ans = ctr;
				}
				else if (a == ma && b > mb)
				{
					//cout << a <<  " " << b << endl;
					ma = a;
					mb = b;
					ans = ctr;
				}
			}
			else
				break;
		}
		printf("%d\n",ans);
	}
	return 0;
}
