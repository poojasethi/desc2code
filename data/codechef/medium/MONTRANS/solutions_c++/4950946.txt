#include<cstdio>
#include<algorithm>
#include<climits>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int A,B,C,i=0;
        scanf("%d%d%d",&A,&B,&C);
        int profit=INT_MIN;
        int rank=0,temp;
        while(i<1000)
        {
        if(profit<100*A+B)
        {
            profit=100*A+B;
            rank=i;
        }
        if(B>=C)
            B=B-C;
        else if(A>0)
        {
            B=100+B-C;
            A--;
        }
        temp=B;
        B=A;
        A=temp;
        if(100*A+B<C)
            break;
        i++;
        }
        printf("%d\n",rank);
    }
    return 0;
}
