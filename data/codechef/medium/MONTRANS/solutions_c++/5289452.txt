// http://www.codechef.com/problems/MONTRANS

int main()
{
	int T;
	scanf("%d",&T);

	while(T--)
	{
		int A,B,C;

		cin >> A >> B >> C ;

		int max = A*100 + B;
		int counter = 0;
		int max_counter = 0;

		while (A*100 + B >= C)
		{
			counter++;
			if ( counter >= 10000)
			{
				break;
			}
			if ( B >= C)
			{
				B-=C;
				int t = A; A = B; B = t;
				if ( A*100 + B > max)
				{
					max = A*100 + B;
					max_counter = counter;
				}
			}
			else
			{
				A-=1;
				B+=100;
				B-=C;
				int t = A; A = B; B = t;
				if ( A*100 + B > max)
				{
					max = A*100 + B;
					max_counter = counter;
				}
			}

		}

		cout << max_counter << "\n";

	}
	return 0;
}
