#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)

ll done[100][100];
vector <pair<ll,ll> > pt;

int main()
{
	ll t,s,a,b,c,time;

	scanf("%lld", &t);

	while(t--) {
		memset(done,0,sizeof(done));

		scanf("%lld%lld%lld", &a,&b,&c);

		time = 0;

		while(1) {
			if(done[a][b]==1) break;
			done[a][b] = 1;
			pt.push_back(mp(a*100+b,time));
			if(b>=c) {
				b=b-c;
				s = a;
				a = b;
				b = s;
				time++;
			} else if(a>0) {
				a = a - 1;
				b = 100 + b;
				b = b - c;
				s = a;
				a = b;
				b = s;
				time++;
			}
		}

		sort(pt.begin(),pt.end());

		s = pt.size();

		printf("%lld\n", (pt[s-1]).second);

		pt.clear();
	}

	return 0;
}
