#include <stdio.h>

int main()
{
    int t, i, A, B, C, max, temp, n;
    scanf("%d",&t);
    while(t--){
        scanf("%d%d%d",&A,&B,&C);
        max = A*100+B;
        i=1;
        n=0;
        while(A*100+B > C && i < 10000){
            B-=C;
            if(B<0){
                B+=100;
                A-=1;
            }

            temp = A;
            A = B;
            B = temp;

            if(A*100+B > max){
                max = A*100+B;
                n = i;
                //printf("%d %d %d %d\n",A,B,C,i);
            }
            i++;
        }
        printf("%d\n",n);
    }
    return 0;
}
