#include <iostream>
#include <bits/stdc++.h>
using namespace std;
 
int main() 
{
	long long int t,a,b,c,max,p1,p2,v,pr,ans,m;
	cin>>t;
	while(t--)
	{
	    cin>>a>>b>>c;
	    max=0;
	    ans=0;
	    m=a*100+b;
	    for(int i=1;i<10000&&((a*100+b)>=c);i++)
	    {
	        v=a*100+b-c;
	        p1=v%100;
	        p2=v/100;
	        pr=p1*100+p2-m;
	        if(pr>max)
	        {
	            max=pr;
	            ans=i;
	        }
	        a=p1;
	        b=p2;
	    }
	    cout<<ans<<"\n";
	}
	return 0;
}
