#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	std::ios_base::sync_with_stdio(false);
	int t,a,b,c,i,count,max,x;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c;
		count=0;
		i=0;
		max=a*100+b;
		while(a>0 || b>=c)
		{
			i++;
			if(b<c)
			{	b=b+100; a--;  }
			
			b=b-c;
			
			x=a;
			a=b;
			b=x;
			if(i>1000)
			break;
			if(a*100+b > max)
			{
				count=i;
				max=a*100+b;
			}
		}
		cout<<count<<'\n';
	}
	
}
