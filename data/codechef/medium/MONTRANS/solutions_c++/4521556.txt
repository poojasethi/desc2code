#include <cstdio>
using namespace std;

int main()
{
    int t, a, b, c;
    scanf("%i", &t);
    while(t--)
    {
    	scanf("%i%i%i", &a, &b, &c);
    	b += 100*a;
    	int maxb = b, transformations = 0, count = 0;
    	while(b >= c && count < 10000)
    	{
    		count++;
    		b -= c;
    		b = (b%100)*100 + b/100;
    		if(b > maxb)
    		{
    			maxb = b;
    			transformations = count;
    		}
    	}
    	printf("%i\n", transformations);
    }
    return 0;
}
