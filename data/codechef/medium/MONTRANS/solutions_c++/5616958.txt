#include <iostream>
#include <iomanip>
using namespace std;
 
int main() {
	
	int test;
	cin>>test;
	
	while(test--)
	{
		int a,b,c;
		int count=0,step=0;
		double temp,max=0.0;
 
		cin>>a>>b>>c;
		
		temp=a+(float)b/100;
		
		while(a>=0 && count <10000)
		{
			if(temp>max)
			{
				max=temp;
				step=count;
			}
			
			if(b>=c)
			{
				b=b-c;
				temp=b;
				b=a;
				a=temp;
			
				temp=a+(float)b/100;
				count++;
			}
			else{
				b=b+100;
				a--;
			}
		}
		cout<<step<<endl;
	}
 
	return 0;
}
