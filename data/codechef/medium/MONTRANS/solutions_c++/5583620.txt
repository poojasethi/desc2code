#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t-->0)
	{
		int a,b,c;
		cin>>a>>b>>c;
		int step=0;
		int max_val=100*a+b;
		int ans=0;
		int count=0;
		while(a!=0 or b>c)
		{   //cout<<a<<b<<endl;
		if(count++==10003)
		break;
			if(100*a+b>max_val)
			{
				max_val=100*a+b;
				ans=step;
			}
			if(b>=c)
			b=b-c;
			else
			{   if(a>0){
				a=a-1;
				b=b+100-c;
			}
			else
			{break;
		}
			}
			int temp=a;
			a=b;
			b=temp;
			step++;
		}
		cout<<ans<<endl;
	}
}
