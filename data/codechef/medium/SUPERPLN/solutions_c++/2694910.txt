int main()
{
    int t;
    cin>>t;
    while(t--)
    {
	int n;
	cin>>n;
	map<int,pair<pair<int,int>,bool> > v[10001];
	map<int,pair<pair<int,int>,bool> > :: iterator it;
	for(int i=0;i<n;i++)
	{
	    int c1,t1,c2,t2;
	    scanf("%d %d %d %d",&c1,&t1,&c2,&t2);
	    v[c1][t1]=mp(mp(c2,t2),0);
	}
	int sc,st,dc,dt;
	scanf("%d %d %d %d",&sc,&st,&dc,&dt);
	int ans=0,flag=0;
	while(1)
	{
	    if(sc==dc && st<=dt)
		break;
	    it=v[sc].lower_bound(st);
	    if(it==v[sc].end() || (*it).s.s==1)
	    {
		flag=1;
		break;
	    }
	    (*it).s.s=1;
	    sc=(*it).s.f.f;
	    st=(*it).s.f.s;
	    ans++;
	}
	if(!flag)
	    printf("Yes %d\n",ans);
	else
	    printf("No\n");
    }
    return 0;
}
