#include<bits/stdc++.h>
using namespace std;
#define ll long long int
#define mod 1000000007
int main(){
    int t;
    cin>>t;
    cin.ignore();
    while(t--){
        string s;
        getline(cin,s);
        int n=s.size();
        bool lmark[26],umark[26];
        memset(lmark,false,sizeof(lmark));
        memset(umark,false,sizeof(umark));
        int index;
        for(int i=0;i<n;i++){
            if(s[i]>='a' && s[i]<='z'){
                index=s[i]-'a';
                lmark[index]=true;
                umark[index]=true;
            }
            else if(s[i]>='A' && s[i]<='Z'){
                index=s[i]-'A';
                lmark[index]=true;
                umark[index]=true;
            }
        }
        bool flag=false;
        for(int i=0;i<26;i++){
            if(lmark[i]==false){
                char c=i+'a';
                cout<<c;
                flag=true;
                break;
            }
        }
        if(flag==false){
            cout<<"~";
        }
        cout<<endl;
    }
    return 0;
}
