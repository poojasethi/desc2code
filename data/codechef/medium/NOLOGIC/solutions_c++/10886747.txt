#include<bits/stdc++.h>
#define si(x) scanf("%d",&x)
#define pi(x) printf("%d\n",x)
 
using namespace std;
 
int main(){
int t;
si(t);
cin.get();
while(t--){
    string str;
    getline(cin,str);
    int n=str.length();int i;
    bool ch_present[26];
    memset(ch_present,false,sizeof(ch_present));
    for (i=0;i<n;i++) {
        if ('a'<=str[i] && str[i]<='z'){
            ch_present[str[i]-'a']=true;
        }
        if ('A'<=str[i] && str[i]<='Z') {
            ch_present[str[i]-'A']=true;
        }
    }
    bool fail=true;
    for (char c='a';c<='z';c++) {
        if(!ch_present[c-'a']){
            printf("%c\n",c);
            fail=false;
            break;
        }
    }
    if(fail){
        cout<<"~"<<endl;
    }
}
return 0;
}
