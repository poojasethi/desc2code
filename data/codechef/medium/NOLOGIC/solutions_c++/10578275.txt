#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    //fflush(stdin);
    string s;
        getline(cin,s);
    while(t--)
    {
        getline(cin,s);
        int i;
        int arr[26];
        memset(arr,0,sizeof(arr));
        for(i=0;s[i]!='\0';i++)
        {
            if(s[i]>='a' && s[i]<='z')
            {
                arr[s[i]-'a']++;
            }
            else if(s[i]>='A' && s[i]<='Z')
            {
                arr[s[i]-'A']++;
            }
        }
        int flag=0;
        char p;
        for(i=0;i<26;i++)
        {
            if(arr[i]==0)
            {
                flag=1;
                p=i+'a';
                cout<<p;
                break;
            }
        }
        if(flag==0)
            cout<<"~";
        cout<<endl;
    }
    return 0;
}
