#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    char ch;
    scanf("%c",&ch);
    while(t--)
    {
        int c[26]={0};
        bool flag=true;
        string s;
        getline(cin,s);
        for(int i=0;i<s.length();i++)
        {

            if((s[i]|32)>=97&&((s[i]|32)<=122))
            c[(s[i]|32)-97]++;
        }
        for(int i=0;i<26;i++)
        {
            if(c[i]==0)
        {
            cout<<(char)(i+97)<<endl;
            flag=false;
            break;
        }
        }
        if(flag)
        {
            cout<<"~"<<endl;
        }

    }
    return 0;
}
