#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
char a[320];
int main()
{
    int k,i,j,p,t,n,t1;
    char temp1,temp2;
    cin>>t;
    t1=t;
    while(t--)
    {
        k=0;
        if(t==t1-1)
            getchar();
        cin.getline(a,315);
        for(i=65;i<=90;i++)
        {
            p=0;
            temp1=(char)(i);
            temp2=(char)(i+32);
            n=strlen(a);
            for(j=0;j<n;j++)
            {
                if(a[j]==temp1||a[j]==temp2)
                {
                    p++;
                    break;
                }
            }
            if(p==0)
            {
                cout<<temp1;
                k++;
            }
        }
        if(k==0)
            cout<<"~\n";
        else
            cout<<"\n";
    }
    return 0;
}
