#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main() 
{
	long long int t,n;
	cin>>t;
	while(t--)
	{
	    char ar[1000];
	    int freq[26]={0};
	    char ch=getchar();
	    scanf("%[^\n]s",ar);
	    n=strlen(ar);
	    for(int i=0;i<n;i++)
	    {
	        if(ar[i]>='a'&&ar[i]<='z')
	        {
	            freq[ar[i]-97]++;
	        }
	        if(ar[i]>='A'&&ar[i]<='Z')
	        {
	            freq[ar[i]-65]++;
	        }
	    }
	    int flag=0;
	    for(int i=0;i<26;i++)
	    {
	        if(freq[i]==0)
	        {
	            flag=1;
	            cout<<(char)(i+65);
	            break;
	        }
	    }
	    if(flag==0)
	    {
	        cout<<"~";
	    }
	    cout<<"\n";
	}
	return 0;
}
