#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int tc;
	scanf("%ld",&tc);
	char x[2];
	cin.getline(x,2);
	while(tc--)
	{
		char s[100001];
		cin.getline(s,100001);
		long int len;
		for(len=0;s[len]!='\0';len++);
		
		long int arr[26];
		for(int i=0;i<26;i++)arr[i]=0;
		for(long int i=0;i<len;i++)
		{
			if(s[i]>=97&&s[i]<=122)
				arr[s[i]-97]++;
			else if(s[i]>=65 && s[i]<=90)
				arr[s[i]-65]++;
		}
		bool check=0;
		for(int i=0;i<26;i++)
		{
			if(arr[i]==0)
			{
				char x=i+97;
				cout<<x<<"\n";
				check=1;
				break;
			}
		}
		if(check==0)
		{
			cout<<"~\n";
		}
	}
	return 0;
}
