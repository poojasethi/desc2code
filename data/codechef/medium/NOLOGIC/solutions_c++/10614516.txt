int main()
{
	//ios_base::sync_with_stdio(false); cin.tie(0);
	int t,i,l,cnt[26];
	char str[100000], str1[100000];
	inp(t);
	getchar();
	while(t--)
	{
		scanf("%[^\n]s",str);
		getchar();
		//printf("%s\n",str);
		l = strlen(str);
		for(i = 0; i < 26; i++)
		{
			cnt[i] = 0;
		}
		for(i = 0; i < l; i++)
		{
			if(str[i] >= 'a' && str[i] <= 'z')
			{
				cnt[str[i] - 'a'] ++;
			}
			else if(str[i] >= 'A' && str[i] <= 'Z')
			{
				cnt[str[i] - 'A'] ++;
			}
		}
		l = 0;
		for(i = 0; i < 26; i++)
		{
			if(cnt[i] == 0)
			{
				str1[l++] = (char)('a' + i);
			}
		}
		str1[l] = '\0';
		if(l > 0)
		{
			printf("%s\n", str1);
		}
		else
		{
			printf("~\n");
		}
	}
	return 0;
}
