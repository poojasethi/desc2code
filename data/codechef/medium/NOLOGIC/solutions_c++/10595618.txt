#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	char s[400];
	gets(s);
	while(t--)
	{
	    gets(s);
	    int f[26]={0},i,m=0,pos;
	    for(i=0;s[i]!='\0';i++)
	    {
	        if(s[i]>='A'&& s[i]<='Z')
	        s[i]+=32;
	    }
	    for(i=0;s[i]!='\0';i++)
	    {
	        if(s[i]>='a' && s[i]<='z')
	        f[s[i]-'a']++;
	    }
	    for(i=0;i<26;i++)
	    {
	        if(f[i]==0)
	        {
	            pos=i;
	            m=1;
	            break;
	        }
	    }
	    if(m==1)
	    cout<<char(pos+97)<<endl;
	    else
	    cout<<"~"<<endl;
	}
	return 0;
}
