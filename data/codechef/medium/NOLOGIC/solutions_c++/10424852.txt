#include<bits/stdc++.h>
using namespace std;
typedef map<char,int> map1;

void answer(string s1)
{
int i;
map1 m;
for(i=0;i<s1.size();i++)
{
if(s1[i]>='a' && s1[i]<='z')
{
if(m.find(s1[i])==m.end())
m[s1[i]]=1;
}
else
{
if(s1[i]>='A' && s1[i]<='Z')
{
if(m.find(s1[i]-'A'+'a')==m.end())
m[s1[i]-'A'+'a']=1;
}
}
}

char c;

for( c='a' ;c<='z';c++)
{
if(m.find(c)==m.end())
{
cout<<c<<"\n";
return;
}
}

cout<<"~"<<"\n";

}



int main()
{
int t=0;
cin>>t;
getchar();
for(int i=0;i<t;i++)
{
string s1;
getline(cin,s1);
answer(s1);
}
return 0;
}
