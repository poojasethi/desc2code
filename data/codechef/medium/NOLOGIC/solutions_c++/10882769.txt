#include <iostream>
#include<cstdlib>
#include<cstdio>
#include<bitset>
#include<iomanip>
#include<cmath>
#include<climits>
#include<cassert>
#include <cstring>
#include <vector>
using namespace std;

#define iway long long int 
int main() 
{
	iway t;
	scanf("%lld",&t);
	cin.get();
	for (iway i=0;i<t;i++)
  {
		string str; 
		getline(cin,str);
		vector<bool>used(26,false);
		iway len=str.length();
		for (iway i =0;i<len;i++) 
     {
			if ('a'<=str[i]&&str[i]<='z')
      {
				used[str[i]-'a']=true;
			}
			if ('A'<=str[i]&&str[i]<='Z') 
      {
				used[str[i]-'A']=true;
			}
		}
		bool never=true;
		for (char c='a';c<='z';c++) {
			if(!used[c-'a']) 
       {
				cout<<c;
				never=false;
				break;
			}
		}
		if(never)
    {
			printf("~");
		}
		printf("\n");
	}
	return 0;
}
