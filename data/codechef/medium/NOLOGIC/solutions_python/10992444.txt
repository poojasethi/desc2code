#include<stdio.h>
int main()
{
    int t,i,arr[130]={0};
    char c;
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        c=getchar();
        while(c!='\n')
        {
            arr[c]++;
            c=getchar();
        }
        for(i=65;i<=90;i++)
        if(arr[i]==0&&arr[i+32]==0)
        {
            printf("%c\n",i);
            break;
        }
        if(i==91)
            printf("~\n");
        for(i=32;i<=126;i++)
            arr[i]=0;
    }
    return 0;
}
