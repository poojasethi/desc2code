#include<iostream>
#include<cstdio>
#define N 321
using namespace std;
int main()
{	int t,i;
	char que[N];
	cin>>t;
	cin.ignore();
	while(t--)
	{	int map[26]={0};
		gets(que);
		for(i=0;que[i];i++)
		{	if(que[i]>='a' && que[i]<='z')
				map[que[i]-'a']=1;
			else if(que[i]>='A' && que[i]<='Z')
				map[que[i]-'A']=1;
		}
		for(i=0;i<26;i++)
		{	if(map[i]!=1)
			{	printf("%c\n",i+'a');
				break;
			}
		}
		if(i>=26)
			printf("~\n");
	}
	return 0;
}
