#include<stdio.h>
int main()
{
    int t,i;
    char s[500];
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        int f[26]={0},pos;
        gets(s);
        for(i=0;s[i]!='\0';i++)
        {
            if(s[i]>='A'&&s[i]<='Z')s[i]+=32;
        }
        for(i=0;s[i]!='\0';i++)
        {
            if(s[i]>='a'&&s[i]<='z')
                f[s[i]-'a']++;
        }
        int flag=0;
        for(i=0;i<26;i++)
            if(f[i]==0)
        {
            flag=1;
            pos=i;
            break;
        }
        if(flag==1)
        {
            printf("%c\n",pos+97);
        }
        else
            printf("~\n");
    }
    return 0;
}
