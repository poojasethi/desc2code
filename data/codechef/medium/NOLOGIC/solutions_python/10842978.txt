#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
	int t,i,x,*alphabet,j,flag=0;
	char str[315],ans[315];
	scanf("%d",&t);
	while(t--)
	{
		if(flag==0)
		{
			gets(str);
			flag=1;
		}
		gets(str);
		alphabet=(int *)calloc(27,sizeof(int));
		for(i=0;i<strlen(str);i++)
		{
			x=(int)str[i];
			if(x>=97&&x<=122)
				alphabet[x-96]++;
			else if(x>=65&&x<=90)
				alphabet[x-64]++;
		}
		ans[0]='\0';
		j=0;
		for(i=1;i<27;i++)
		{
			if(alphabet[i]==0)
			{
				ans[j++]=(char)(i+64);
				ans[j++]=(char)(i+96);
			}
		}
		ans[j]='\0';
		if(j==0)
			printf("~\n");
		else
			printf("%s\n",ans);
	}
	return 0;
}
