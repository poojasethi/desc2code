#include<stdio.h>
#include<string.h>
#include<stdlib.h>
int main()
{
	int t;
	scanf("%d",&t);
	
	while(t--)
	{
		char quest[315];
		getchar();
		scanf("%[^\n]",quest);
		
		int i,ar[26],c,s,count=0;
		
		for(i=0;i<26;i++)
		{
			ar[i]=0;
		}
		
		for(i=0;i<strlen(quest);i++)
		{
			c=quest[i]-65;
			s=quest[i]-97;
			
			if(c>=0 && c<=25)
			{
				ar[c]++;
			}
			
			else if(s>=0 && s<=25)
			{
				ar[s]++;
			}
		}
		
		/*for(i=0;i<26;i++)
		{
			printf("%d ",ar[i]);
		}*/
		
		for(i=0;i<26;i++)
		{
			if(ar[i]==0)
			{
				printf("%c",(i+65));
				break;
			}
			
			else
			{
				count++;
			}
		}
		
		if(count==26)
		{
			printf("~");
		}
		
		printf("\n");
		
	}
}
