#include<iostream>
using namespace std;
int fact[31][10];
int flag[31] = {0};
void p_fact()
{
	int i,j,n;
	for(i = 2;i<=30;i++)
	{
		n = i;
		for(j = 2;j<=n;)
		{
			if(n%j==0)
			{
				fact[i][flag[i]] = j;
				flag[i]++;
				n = n/j;
			}
			else
			{
				j++;
			}
		}
	}
}
int main()
{
	int t;
	cin>>t;
	p_fact();
	int arr[31];
	while(t--)
	{
		int n,i,j,k,var,prod,ex,prev,p,m,b;
		cin>>n;
		cin>>arr[0];
		prev = arr[0];
		ex = 1;
		for(i = 1;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i]==prev&&ex!=0)
			{
				ex = 1;
			}
			else
			{
				ex = 0;
			}
		}
		if(ex== 1)
		{
			cout<<"0\n";
		}
		else
		{
			p = flag[n];
			int sum = 0;
			for(i = 0;i<p;i++)
			{
				sum += n/(fact[n][i]);
			}
			cout<<sum<<"\n";
			for(i = p-1;i>=0;i--)
			{
				var = fact[n][i];
				if(i == p-1)
				{
					j = 0;
				while(j<n/var)
				{
					cout<<var<<" ";
					for(k = 1;k<=var;k++)
					{
						cout<<(k+j*var)<<" ";
					}
					j++;
					cout<<"\n";
				}
				}
				else
				{
					prod = 1;
					for(j = p-1;j>i;j--)
					{
						prod *= fact[n][j];
					}
					j = 0;
					while(j<n/(prod*var))
					{
						for(m = 1+j*prod*var;m<=prod+j*prod*var;m++)
						{
							cout<<var<<" ";
							for(b = 0;b<var;b++)
							{
								cout<<(b*prod+m)<<" ";
							}
							cout<<"\n";
						}
						j++;
					}
				}
			}
		}
	}
}
