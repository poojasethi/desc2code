int prime(int n)
{
    if(n==1)
    return 1;
    
    if(n%2==0)
    {
              return 2;
    }
 int i;
 for(i=3;i<=n;i=i+2)
 {
                    if(n%i==0)
                    {
                              return i;          
                    }
 }
}
int countequalize(double a[],int start,int end,int count)
{
                     int x=prime(count);
                     int y=count/x;            
                     int i,j,k,sum;
                     if(x==count)
                     {
                                 return 1;
                     }
                     sum=0;
                     sum=x*countequalize(a,start+(i-1)*y,start+(i)*y - 1,y);
                     sum+=y;
                     return sum;    
}
 
equalize(double a[],int start,int end,int count)

}
