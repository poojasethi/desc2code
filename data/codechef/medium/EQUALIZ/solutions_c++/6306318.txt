void func(int i,int j)
{
	if(i==j)
	{
		return ;
	}
	int n=j-i+1;
	int p,q;
	p=a[2*n-2];
	q=a[2*n-1];
	int k,init;
	k=q;
	init=i;
	while(k--)
	{
		func(init,init+p-1);
		
		init=init+p;
	}
	k=i+p-1;
	int r,t;
	init=i;
	r=i;
	while(r<=k)
	{
		printf("%d ",q);
		for(t=0;t<q;t++)
		{
			printf("%d ",r+t*p);
		}
		printf("\n");
		r++;
	}
}
