#include <bits/stdc++.h>
#define sl(l) scanf("%lld",&l)
typedef long long ll;

ll a[31];
ll o[1000][32],z=0;
int isprime(ll n)
{
	int i,b[]={2,3,5,7,11,13};
	for(i=0;i<6;i++)
	{
		if((n%b[i]==0)&&(n>b[i])) return b[i];
	}
	return -1;
}

void oper(ll low,ll high)
{
	ll n = high-low+1;
	ll m = isprime(n);
	if(m==-1)
	{
		o[z][0]=n;
		for(ll i=1;i<=n;i++)o[z][i]=low+i-1;
		z++;
	}
	ll p = n/m; // p is the size of new array m is no of array
	for(ll i=0;i<m;i++)oper(low+i*p,low+(i+1)*p-1);
	for(ll i=0;i<p;i++)
	{
		o[z][0]=m;
		for(ll j=1;j<=m;j++)o[z][j]=low+i+(j-1)*p;
		z++;
	}

}
int main()
{
	ll t;
	sl(t);
	while(t--)
	{
		ll n;
		sl(n);
		for(ll i=0;i<n;i++)sl(a[i]);
		oper(0,n-1);
		printf("%lld\n",z);
		for(ll i=0;i<z;i++){printf("%lld ",o[i][0]);for(ll j=1;j<=o[i][0];j++)printf("%lld ",o[i][j]+1);printf("\n");}
		z=0;
	}
	
	return 0;
}
