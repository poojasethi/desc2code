#include<stdio.h>
int main()
{
	int ans[31]={0,1,1,1,4,1,5,1,12,6,7,1,16,1,9,8,32,1,21,1,24,10,13,1,44,10,15,27,32,1,31};

	int T;
	int N,i,j,k,p2,k1,product2;
	int a;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&N);
		for(i=0;i<N;i++)
			scanf("%d",&a);
		
	
			int p=N,product=1;
			k=N;
			printf("%d\n",ans[N]);
			while(1)
			{
				for(p=p;p>=1;p--)
				{
				if(ans[p]==1 && k%p==0)
				{
					break;
				}
				}
				
				product2=product*p;
				for(i=0;i<N/product2;i++)
				{
					for(k1=0;k1<product;k1++)
					{
						printf("%d",p);
						for(j=0;j<p;j++)
						printf(" %d",i*product2+j*product+k1+1);
					printf("\n");
					}
				}
				product=product2;
			k=k/p;
			if(k==1)
				break;
			}
		
	
	}
	return 0;
	}
