void solve(int s,int e){
    if(e-s+1<2)return;
    int i,j,k,l=e-s+1;
    reps(i,2,l+1){
        if(l%i==0)break;
    }
    //now i will always be  a prime
    k=i;
    if(k==l){
        a.pb(vi());
        reps(j,s,e+1)a.back().pb(j);
        return;
    }
   // cout<<"l="<<l<<" k="<<k<<"\n";
    for(j=s;j<e;j+=(l/k)){
        solve(j,j+l/k-1);
    }
    reps(i,s,s+l/k){// [. . . ][. . .][. . .]
        a.pb(vi());
       for(j=i;j<=e;j+=l/k){
            a.back().pb(j);
       }
    }
}
