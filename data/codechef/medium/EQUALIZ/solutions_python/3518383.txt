void go(vector<int> ind)
{
	int n = SIZE(ind);
	if(n==1)			return;
	int rr = -1;
	FOR(i, 2, n-1)	if(n%i == 0)
	{
		rr = i;		break;
	}
	int i = ~rr ? rr : n;
		int l = 0, r = i;		
		rep(j, n/i)
		{
			cout << i;
			REP(k, l, r)
				cout << " " << ind[k];
			cout << "\n";
			l += i, r += i;
		}
		rep(j, i)
		{
			vector<int> tt;
			for(int k=j; k<n; k+=i)		tt.push_back(ind[k]);
			go(tt);
		}		
}
