#include <bits/stdc++.h>
using namespace std;

int a[100001];

int main(){
    int n;
    cin >> n;
    for(int i = 0; i < n; i++) cin >> a[i];
    long tot = 0;
    for(int i = 0; i < n; i++) tot += a[i];
    if(tot%2 == 0){tot = tot/2;}
    else{ tot = (tot/2) + 1;}
    if(tot < n) cout << n << endl;
    else{cout << tot << endl;}
}
