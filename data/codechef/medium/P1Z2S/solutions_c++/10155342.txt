#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main(){
    ll n, sum, ans, temp;
	
    cin>>n;
    
    sum = 0;
    
    for(ll i = 0; i < n; i++){
        cin>>temp;
        sum+=temp;
    }
    
    if(sum - 2*n > 0){
    	ans = (sum+1)/2;
    } else {
    	ans = n; 
    }
    
    cout<<ans;
    
    return 0;
}
