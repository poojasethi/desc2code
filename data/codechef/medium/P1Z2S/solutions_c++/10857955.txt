#include<iostream>
#include<stdio.h>
int main(){
	int i,n,a[100000];
	int flag=1;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	int avai=0,needed=0;
	avai=2*n;
	for(i=0;i<n;i++){
		needed=needed+a[i];
	}
	if(avai>=needed){
		printf("%d\n",n);
	}
	else{
		do{
			avai=2*(++n);
			if(avai>=needed){
				printf("%d\n",n);
				flag=0;
			}
		}while(flag!=0);
	}
	
	return 0;
}
