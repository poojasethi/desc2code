#include<cstdio>
#define lld long long int 
using namespace std;
int main(int argc, char const *argv[])
{
	lld i,n,a[1000000],sum=0,reserved=0,ans=0;
	scanf("%lld",&n);
	for (i = 0; i < n; ++i)
	{
		scanf("%lld",&a[i]);
		sum+=a[i];
	}
	if (sum < n*2)
		sum=0;
	else 
	{
		sum = sum - n*2;
	}
	if(sum%2==0)
		printf("%lld\n", n+ sum/2);
	else 
		printf("%lld\n", n+sum/2+1);
	
	return 0;
}
