#include<bits/stdc++.h>
 
using namespace std;
 
int main(void)
{
	int n; scanf("%d",&n);
 
	int tvisit=0;
	for(int i=0;i<n;i++)
	{
		int temp; scanf("%d",&temp);
		tvisit+=temp;
	}
 
	int v=n, remvisit=tvisit-n;
	if(remvisit<=v)
	  printf("%d\n",n);
 
    else
       {
       	  if((remvisit-v)%2==0)
       	  {
       	  	printf("%d\n",n+(remvisit-v)/2);
       	  }
       	  else
       	  {
       	  	printf("%d\n",n+(remvisit-v+1)/2);
       	  }
       }
}
