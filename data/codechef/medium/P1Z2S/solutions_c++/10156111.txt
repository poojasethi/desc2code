#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,sum=0,i,val,rem,cn;
    cin>>n;
    vector<int>a;
    for(i=0;i<n;i++)
    {
        cin>>rem;
        a.push_back(rem);

    }
    cn=0;
    val=2*n;

    rem=n;
    sort(a.begin(),a.end());
    i=0;
    while(1)
    {
         while(i<n)
         {

             //if(a[i]!=0)
             //{
                 if(a[i]<=val)
                 {

                     val=val-a[i];
                     a[i]=0;
                     //cout<<val<<" "<<a[i]<<" ";
                     cn++;
                 }
                 else
                 {
                     a[i]=a[i]-val;
                     val=0;
                     break;
                 }
             //}
             i++;
         }

        //cout<<cn<<" ";
        if(cn==n)
            break;

        val=2*(n-cn);
        rem=rem+(n-cn);
    }
    cout<<rem;
    return 0;
}
