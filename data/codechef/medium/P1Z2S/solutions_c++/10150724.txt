#include<bits/stdc++.h>
using namespace std;
#define lli long long int
#define f(i,a,b) for(i=a;i<b;i++)
#define si(a) scanf("%d",&a);
#define slli(a) scanf("%lld",&a);
#define MOD 1000000007
#define MAX 100005
#define pb(a) push_back(a)
lli arr[MAX];
int main()
{
	lli i,j,k,m,n,t,sum=0,ans,diff;
	slli(n);
    f(i,0,n)
    {
    	slli(arr[i]);
    	sum+=arr[i];
    }
    if(sum<=2*n)
    	ans=n;

    else
    {
    	diff=sum-2*n;
    	if(diff%2==0)
    		ans=diff/2;
    	else
    		ans=(diff/2)+1;

    	ans+=n;
    }
    printf("%lld\n",ans);
}
