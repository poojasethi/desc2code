#include<stdio.h>
#include<limits.h>
long long int count[100005];
int main()
{
	long long int i, n, max, a,t;
	scanf("%lld", &t);
	while (t--)
	{
		max = INT_MIN;
		for (i = 1; i <= 100000; i++)
			count[i] = 0;
		scanf("%lld", &n);
		for (i = 0; i < n;i++)
		for (i = 0; i < n; i++)
		{
			scanf("%lld", &a);
			count[a]++;
			if (count[a] > max)
				max = count[a];
		}
		printf("%lld\n", n-max);
	}
	getchar();
	getchar();
	return 0;
}
