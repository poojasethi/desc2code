#include <iostream>
#include <cstdio>
#include <cstdlib>

using namespace std;

int max_elem(int* C)
{
    int M = -1,i=0;
    for(i=0;i<100000;i++)
    {
        if(C[i] > M)
            M = C[i];
    }
    return M;
}

int main()
{
    int T,N,cost = 0;
    int* A;
    int* C = new int[100000];
    cin>>T;
    while(T--)
    {
        cost = 0;
        for(int i=0;i<100000;i++)
            C[i] = 0;
        cin>>N;
        A = new int[N];
        for(int i=0;i<N;i++)
        {
            cin>>A[i];
            C[A[i]]++;
        }
        int mode = max_elem(C);

        cout<<N-mode<<endl;
    }
    return 0;
}
