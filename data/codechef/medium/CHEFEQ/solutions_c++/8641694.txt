#include<bits/stdc++.h>
using namespace std;

void fastscan(int &x)
{
    register int c;
    x=0;
    c=getchar();
    for(;c>47&&c<58;c=getchar())
        x=(x<<1)+(x<<3)+c-48;

}

int n,a[100001],t,i,ans,x;
int main()
{
fastscan(t);
while(t--)
{  ans=0;
 fastscan(n);
  for(i=0;i<n;i++)
  {fastscan(x);
    a[x]++;
   ans=max(ans,a[x]);
   }
for(i=0;i<100001;i++)
    a[i]=0;

    cout<<n-ans<<endl;
}
return 0;
}
