#include<bits/stdc++.h>
#include<iostream>
using namespace std;
#define ll long long int
int main()
{

    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int T;
    int N;
    int X;
    int mm;
    cin>>T;
    while(T--)
    {
        cin>>N;
        map<int,int> mymap;
        int i;mm=0;
        for(i=0;i<N;i++)
        {

            cin>>X;
            mymap[X]++;
            mm=max(mm,mymap[X]);

        }
        int ans=N-mm;
        cout<<ans<<endl;
        mymap.clear();
    }
    return 0;
}
