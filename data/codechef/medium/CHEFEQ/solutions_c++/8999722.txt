#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,i,a,z[100009],sum=0,max=0;
		scanf("%lld",&n);
		for(i=0;i<100009;i++)
		z[i]=0;
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a);
			
				z[a]++;
				if(z[a]>max)
				max=z[a];
			
		}

		printf("%lld\n",n-max);
			
	} 
	return 0;
}
