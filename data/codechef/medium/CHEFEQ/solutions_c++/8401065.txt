#include<bits/stdc++.h>
using namespace std;
int main()
{
         int j,t=1,n,ma=1,i;
         scanf("%d",&j);
         while(j--)
         {
                  t=1;
                  ma=1;
                scanf("%d",&n);
                  long int a[n];
                  for(i=0;i<n;i++)
                    scanf("%d",&a[i]);
                    sort(a,a+n);
                  for(i=1;i<n;i++)
                  {
                           if(a[i]==a[i-1])
                           t++;
                           else
                           {
                           if(ma<t)
                           ma=t;
                           t=1;
                           }
                  }
                if(ma<t)
                ma=t;
                printf("%d\n",n-ma);
         }
         return 0;
}
