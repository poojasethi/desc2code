#include <stdio.h>
#include<string.h>
#define min(a,b) a<b?a:b
#define max(a,b) a>b?a:b
#define gc getchar_unlocked
#define pc putchar_unlocked

inline int sscan()
{
   register int n=0;
   register int ch=gc();
   while(ch<'0'||ch>'9')
        ch=gc();
   while(ch>='0'&&ch<='9')
    {
        n=(n<<3)+(n<<1)+ch-'0';
        ch=gc();
    }
   return n;
}

inline void print(int a)
{
 register int i=0;
char S[20];
while(a>0)
{
    S[i++]=a%10+'0';
a=a/10;
}
--i;
while(i>=0)
pc(S[i--]);
pc('\n');
}


int main()
{
    register int t;
    t=sscan();
    while(t--)
    {
        register int n;
        n=sscan();
        int arr[100002];
        register int i=0,num,ans=0;

        for(i=0;i<100002;i++)
        {
            arr[i]=0;
        }

        for(i=0;i<n;i++)
        {
            num=sscan();
            arr[num]++;
            ans=max(ans, arr[num]);
        }
        print(n-ans);
    }
    return 0;
}
