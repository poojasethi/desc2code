	#include<iostream>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			int n;
			cin>>n;
			int a[100005]={0};
			int temp;
			for(int i=0;i<n;i++)
			{
				cin>>temp;
				a[temp]++;
			}	
			int max=0;
			for(int i=0;i<100000;i++)
			{
				if(a[i]>max)
					max=a[i];
			}
			cout<<n-max<<endl;
			t--;
		}	
	}
