#include<bits/stdc++.h>
using namespace std;
int hash[100005],arr[100005];
int main()
{
	int n,t,x;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=1;i<100005;i++)
		hash[i]=0;
	    for(int i=0;i<n;i++)
	    {
	    	scanf("%d",&arr[i]);
	    	hash[arr[i]]++;
	    }
	    int ans=0;
	    for(int i=0;i<n;i++)
		{
			if(hash[arr[i]]>ans)
			ans=hash[arr[i]];
		}
		printf("%d\n",n-ans);
	}
	
}
