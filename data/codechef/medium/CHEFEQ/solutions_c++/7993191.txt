#include<bits/stdc++.h>
using namespace std;
main()
{
    int T;
    cin>>T;
    while(T-->0)
    {
        int n;
        cin>>n;
        int f[100001]={0};
        int i;
        int x;
        for(i=0;i<n;i++)
        {
            cin>>x;
            f[x]++;
        }
        int freq=0;
        for(i=0;i<100001;i++)
        {
            if(f[i]>freq)
                freq=f[i];
        }
        cout<<(n-freq)<<endl;
    }

}
