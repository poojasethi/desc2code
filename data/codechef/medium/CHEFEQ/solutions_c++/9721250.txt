#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int t,a[100009],i,max,n,temp,ans;
	scanf("%d",&t);
	while(t--)
	{	
		for(i=0;i<100009;i++)
			a[i]=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&temp);
			a[temp]++;
		}
		sort(a,a+100009);
		max=a[100008];
		ans=n-max;
		printf("%d\n",ans);
	}
	return 0;
}
