#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,p,i;
	scanf("%d",&t);
	while(t--)
	{
		long long int arr[100009]={0};
		scanf("%d",&n);
		for( i=0;i<n;i++)
		{
			scanf("%d",&p);
			arr[p]++;
		}
		int x=0;
		for(i=0;i<100009;i++)
			if(x<arr[i])
				x=arr[i];
		printf("%d\n",n-x);
	}
	return 0;
}
