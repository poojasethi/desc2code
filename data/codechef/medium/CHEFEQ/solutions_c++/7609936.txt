#include<bits/stdc++.h>
#include<iostream>
#define MAXN 100001
using namespace std;
#define ll long long int

inline int scan()
{
    int x=0;
    char c=getchar_unlocked();
    while(c<'0'||c>'9')
        c=getchar_unlocked();
    while(c>='0'&&c<='9')
    {
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}



void fastWrite(ll x)
{
    char s[21]={' '};
    int l=0;
    do
    {
        s[l++]=x%10+'0';
        x/=10;
    }while(x);
    --l;
    while(l>=0)
    {
        putchar_unlocked(s[l--]);
    }
    putchar_unlocked('\n');
    return;
}

int A[MAXN];

int main()
{

    int T;
    int N;
    int X;
    int mm;
    T=scan();
    while(T--)
    {
        N=scan();
        memset(A,0,sizeof A);
        int i;mm=0;
        for(i=0;i<N;i++)
        {

            cin>>X;
            A[X]++;
            mm=max(mm,A[X]);

        }
        int ans=N-mm;
        fastWrite(ans);
    }
    return 0;
}
