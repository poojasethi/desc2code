#include<bits/stdc++.h>
using namespace std;
#define maxi 100010

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;;
		int b[maxi]={0};
		for(int i=0;i<n;i++)
		{
			int x;
			cin>>x;
			b[x]++;
		}
		int ind=0,m=0;
		for(int i=0;i<maxi;i++)
		{
			if(m<b[i])
			{
				ind=i;
				m=b[i];
			}
		}
		cout<<n-m<<"\n";
	}
	return 0;
}
