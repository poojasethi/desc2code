#include<iostream>
using namespace std;
int main()
{
	int t;
	long long int n,i,arr[100001],arr1[100001],max,max1;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=0;i<=100000;i++)
		arr[i]=0;
		cin>>arr1[0];
		max=arr1[0];
		arr[arr1[0]]++;
		for(i=1;i<n;i++)
		{
			cin>>arr1[i];
			
			if(arr1[i]>max)
			max=arr1[i];
		    arr[arr1[i]]++;
		}
		//cout<<"max : "<<max<<endl;
		max1=arr[0];
		for(i=1;i<=max;i++)
		{
			if(arr[i]>max1)
			max1=arr[i];
		}
		//cout<<"max1 : "<<max1<<endl;
		cout<<n-max1<<endl;
	}
	return 0;
}
