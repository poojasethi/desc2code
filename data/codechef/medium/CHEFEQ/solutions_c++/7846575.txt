#include <iostream>
using namespace std;

int main() {
	int a[100000],n,t,big,i,temp;
	cin>>t;
	while(t--)
	{
		int num[100001]={0};big=0;
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			temp=a[i];
			if(temp>big)
			big=temp;
			num[temp]++;
		}
		temp=0;
		for(i=1;i<=big;i++)
		{
			if(num[i]!=0)
				if(num[i]>temp)
					temp=num[i];
		}
		cout<<"\n"<<n-temp;
	}
	return 0;
}
