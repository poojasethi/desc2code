#include<bits/stdc++.h>
#include<iostream>
using namespace std;
#define ll long long int
int A[100001];
int main()
{

    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int T;
    int N;
    int X;
    int mm;
    cin>>T;
    while(T--)
    {
        cin>>N;
        memset(A,0,sizeof A);
        int i;mm=0;
        for(i=0;i<N;i++)
        {

            cin>>X;
            A[X]++;
            mm=max(mm,A[X]);

        }
        int ans=N-mm;
        cout<<ans<<endl;
    }
    return 0;
}
