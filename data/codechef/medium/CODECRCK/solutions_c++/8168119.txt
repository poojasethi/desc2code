#include<cstdio>
#include<cmath>


int main()
{
    long long i,k,s,a_i,b_i;
    double a,b;
    scanf("%lld%lld%lld%lld%lld",&i,&k,&s,&a_i,&b_i);
    a=a_i;
    b=b_i;
 //   if((i%2==0&&k%2!=0)||(i%2!=0&&k%2==0))
 if ((i&1) != (k&1))
    {
        i++;
        a=sqrt(2.0)*((a_i+b_i)-sqrt(3.0)*(a_i-b_i));
        b=sqrt(2.0)*((a_i-b_i)+sqrt(3.0)*(a_i+b_i));

    }


    double q=(a+b)*pow(2.0,(2*(k-i))-s);
    printf("%.10f\n",q);
    return 0;
}
