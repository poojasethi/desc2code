#include <stdio.h>
#include <math.h>
int main()
{
	long long int i,k,s;
	double ans,a_i,b_i;
	scanf("%lld%lld%lld",&i,&k,&s);
	scanf("%lf%lf",&a_i,&b_i);
	if((k-i)%2==0)
	{
		ans=a_i+b_i;
		printf("%lf\n",ans*pow(2.0,(2*(k-i))-s));
	}
	else
	{
		ans = sqrt(2)*pow(2.0,(2*(k-i)-1-s))*(a_i+(sqrt(3)*b_i));
		printf("%lf\n",ans);
	}
	return 0;
}
