#include<stdio.h>
#include<stdlib.h>
#include<math.h>
using namespace std;
 
int main()
{
    long long int i,k,s,j;
    double ai,bi,x=sqrt(2.0),y=sqrt(3.0),ans;
    scanf("%lld%lld%lld",&i,&k,&s);
    scanf("%lf%lf",&ai,&bi);
 
        if(!((k-i)&1))
        {
            ans=(ai+bi)*(pow(2.0,((2*(k-i))-s)));
        }
        else
        {
             if(i<=k)
           {double an,bn;
           an=x*((ai+bi)-y*(ai-bi));
           bn=x*((ai-bi)+y*(ai+bi));
           ai=an,bi=bn;
           i++;
           }
             else
           {double an,bn;
           an=(ai+bi-(y*(ai-bi)))/((2*x)+(2*x*y*y));
           bn=(ai-bi+(y*(ai+bi)))/((2*x)+(2*x*y*y));
           ai=an,bi=bn;
           i--;
           }
           ans=(ai+bi)*pow(2.0,2*(k-i)-s);
        }
        printf("%.10lf\n",ans);
 
 
    return 0;
 
}
