#include<cstdio>
#include<cmath>

int main() {
    long long i, k, s, a_i, b_i;
    double a, b;
    scanf("%lld %lld %lld %lld %lld", &i, &k, &s, &a_i, &b_i);
    a =a_i;
    b = b_i;
    if ((i & 1) != (k & 1)) {
        i++;
        a = sqrt(2.0) * ((a_i + b_i) - sqrt(3.0) * (a_i - b_i));
        b = sqrt(2.0) * ((a_i - b_i) + sqrt(3.0) * (a_i + b_i));
    }
    double res = (a + b) * pow(2.0, 2 * (k - i) - s);
    printf("%.10f\n", res);
    return 0;
}
