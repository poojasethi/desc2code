#include<stdio.h>
#include<stdlib.h>
#include<math.h>
using namespace std;

int main()
{
    long long int i,k,s;
    scanf("%lld%lld%lld",&i,&k,&s);
       double a, b;
	const double x = sqrt(2);
	const double y = sqrt(3);
	scanf("%lf %lf", &a, &b);
	if((i & 1) != (k & 1)) {
		double na, nb;
		na = x * ((a + b) - y * (a - b));
		nb = x * ((a - b) + y * (a + b));
		a = na;
		b = nb;
		i++;
	}
	printf("%.10lf\n", (a + b) * pow(2.0, ((k - i) << 1) - s));
    return 0;
}
