#include <bits/stdc++.h>

using namespace std;

int main() {
	long long i, k, s, a_i, b_i;
	cin >> i >> k >> s >> a_i >> b_i;
	double a = a_i, b = b_i;
	if ((i & 1) != (k & 1)) {
		i++;
		a = sqrt(2.0) * ((a_i + b_i) - sqrt(3.0) * (a_i - b_i));
		b = sqrt(2.0) * ((a_i - b_i) + sqrt(3.0) * (a_i + b_i));
	}
	double res = (a + b) * pow(2.0, 2.0 * (k - i) - s);
	printf("%.10f\n", res);
	return 0;
}
