#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll d(ll n)
{
	ll k;
	if(n<0)return 0;
	else
		k=((n+1)*(n+2)*(n+3))/6;
	return k;
}
int main()
{

	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		ll n,a,b,c;
		cin>>n>>a>>b>>c;
		ll ans=d(n)-d(n-a-1)-d(n-b-1)-d(n-c-1)+d(n-a-b-2)+d(n-a-c-2)+d(n-b-c-2)-d(n-a-b-c-3);
		printf("%lld\n",ans );
	}
	return 0;
}
