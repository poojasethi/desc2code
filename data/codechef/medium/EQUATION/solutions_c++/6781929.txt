#include <bits/stdc++.h>
using namespace std;

int main(){
    int t , A , B , C , N ;
    cin >> t ;
    while(t--){
        cin >> N >> A >> B >> C ;
        long long tot = 0 ;
        for(int a = 0 ; a <= A ; a++){
            for(int b = 0 ; b <= B ; b++){
                int x = N - a - b ;
                if(x < 0){
                    break;
                }
                x = min(x , C);
                tot += x+1;
            }
        }
        cout << tot << endl;
    }
}
