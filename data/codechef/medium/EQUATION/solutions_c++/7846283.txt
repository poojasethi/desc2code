typedef long long ll;
int main()
{
    int t; get(t);
    while(t--)
    {
        int n,a,b,c; scanf("%d%d%d%d",&n,&a,&b,&c);
        ll ans=0;
        for(int i=0;i<=a;i++)
        {
            for(int j=0;j<=b;j++)
            {
                if(i+j<=n){

                if(k<c)
                    ans += k+1;
                else
                    ans += c+1;
                }
            }
        }
        printl(ans);
    }

}
