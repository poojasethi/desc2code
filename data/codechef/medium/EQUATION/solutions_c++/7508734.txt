#include <bits/stdc++.h>
using namespace std;
 
int main() {
	// your code goes here
	long long test,n,a,b,c;
	scanf("%lld\n",&test);
	while(test--)
	{
	long long sum=0;
	scanf("%lld %lld %lld %lld\n",&n,&a,&b,&c);
	for(int i=0;(i<=n && i<=a);i++)
	{
	   for(int j=0;(j<=n-i && j<=b);j++)
	   {
	   if(c>(n-i-j))
	   sum+=n-i-j+1;
	   else
	   sum+=c+1;
	   }
	}
	printf("%lld\n",sum);
	}
	return 0;
}
