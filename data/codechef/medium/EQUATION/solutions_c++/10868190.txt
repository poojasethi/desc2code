long long int t,n,a,b,c,d,i,j,N,sum,ans,l1,l2,l3;
int main()
{
    scanf("%lld",&t);
    while(t--)
    {   ans=0;
        scanf("%lld %lld %lld %lld",&n,&a,&b,&c);
        if(n>=a)
        {
            l1=a;
        }
        else
            l1=n;
        for(i=0;i<=l1;i++)
        {   if((n-i)>=b)
        {
            l2=b;
        }
        else
            l2=n-i;
            for(j=0;j<=l2;j++)
            {   if((n-i-j)>=c)
                l3=c;
                else
                    l3=n-i-j;
                N=n-i-j;
                if(N>=c)
                    ans+=c+1;
                else
                    ans+=N+1;
            }
        }
printf("%lld\n",ans);
    }


return 0;
}
