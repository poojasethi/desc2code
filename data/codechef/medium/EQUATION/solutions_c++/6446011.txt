#include<cstdio>
#include<iostream>
using namespace std;

    long long int f(long long int n)
{
    long long int r;
    if(n<0)
    return 0;
    else
    r = (n+1)*(n+2)*(n+3);
        return r/6 ;
}

int main()
{
    long long int n,a,b,c,t,answer;
    cin>>t;
    while(t--)
    {
        cin>>n>>a>>b>>c;
        answer=f(n)-f(n-a-1)-f(n-b-1)-f(n-c-1)+f(n-a-b-2)+f(n-a-c-2)+f(n-c-b-2)-f(n-a-b-c-3);
       cout<<answer<<"\n";
    }
    return 0;
}
