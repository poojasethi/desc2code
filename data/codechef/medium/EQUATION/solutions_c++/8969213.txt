#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	for(scanf("%d", &t); t--; )
	{
		int n, a, b, c;
		scanf("%d %d %d %d", &n, &a, &b, &c);
		long long res=0;
		for(int i=0; i<=a; i++)
			for(int j=0; j<=b; j++){
				int temp = n-i-j;
				if(temp<0)
					continue;
				if(c<temp)
					res+=c+1;
				else
					res+=temp+1;
			}
		printf("%lld\n", res);
	}
	return 0;
}
