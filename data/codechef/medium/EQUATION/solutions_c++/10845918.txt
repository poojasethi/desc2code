#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long
long long int arr[3];
int main()
{
    ll t,n,a,b,c;
    ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        cin>>n>>arr[0]>>arr[1]>>arr[2];
        ll sum = 0;
        sort(arr,arr+3);
        a=arr[0];b=arr[1],c=arr[2];
        for(long long int i=0;i<=a;i++)
        {
            for(long long int j=0;j<=b;j++)
            {
                //cout<<i<<' '<<j<<' '<<endl;
                if(n-(i+j)>=c)
                    sum+=(c+1);
                else if((i+j)<=n)
                    sum+=(n-i-j)+1;
                //cout<<sum<<endl;
            }
        }
        cout<<sum<<endl;
    }
}
