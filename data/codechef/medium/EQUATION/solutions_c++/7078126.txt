#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
void sc(long long &test)
{
	scanf("%lld",&test);
}
int main()
{
	long long test;
	sc(test);
	while(test--)
	{
		long long n,a,b,c;
		sc(n);sc(a);sc(b);sc(c);
		long long ans=0;
		if(a>b)
		{
			swap(a,b);
		}
		if(a>c)
		swap(a,c);
		if(b>c)
		swap(b,c);
		for(long long i=0;i<=a;i++)
		{
			for(long long j=0;j<=b;j++)
			{
				if(i+j>n)
				break;
				ans+=(min(c,n-i-j));
				ans++;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
