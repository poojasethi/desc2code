#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin >> t;
    while(t--) {
        int A,B,C,N;
        cin >> N >> A >> B >> C;
        long long int cnt = 0;
        int a,b;
        for(a = 0;a <= A && a <= N;a++) {
            for(b = 0;b <= N-a && b <= B;b++ ){
                cnt += min(C+1,N-(a+b)+1);
               // cout << cnt << "\n";
            }
        }
        cout << cnt << "\n";
    }
    return 0;
}
