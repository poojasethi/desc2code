#include<cstdio>

long long gcd(long long a, long long b) {
  if (b == 0) return a;
  return gcd(b, a % b);
}


long long ncr(long long n, long long r) {
	if(n<r)
	return 0;
	else
	{
	
  if (r > n - r)
    r = n - r;
 
  long long num = 1;
  long long den = 1;
  for (long long i = n-r+1, j = 1; i <= n; ++i, ++j) {
    num *= i;
    den *= j;
    long long g = gcd(num, den) ;
    num = (num / g) ;
    den = (den / g) ;
  }
  return (num / den) ;
}
}


int main()
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		long long n,a,b,c;
		cin>>n;cin>>a;cin>>b;cin>>c;
		long long ans=0;
		ans=ncr(n+3,3)-ncr(n+2-a,3)-ncr(n+2-b,3)-ncr(n+2-c,3)+ncr(n+1-a-b,3)+ncr(n+1-a-c,3)+ncr(n+1-c-b,3)-ncr(n-a-b-c,3);
		printf("%lld\n",ans);	
   }
	return 0;
}
