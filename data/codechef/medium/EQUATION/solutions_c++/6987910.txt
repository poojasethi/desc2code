#include<iostream>
using namespace std;
int main()
{
    long long t,sum=0;
    int n,a,b,c,j,i;
    cin>>t;
    while(t--)
    {
        sum=0;
        cin>>n>>a>>b>>c;
        for(i=0;i<=a;i++)
        {
            for(j=0;j<=b;j++)
            {
                if(n-i-j>=0)
                sum+=min(c+1,n-i-j+1);
            }
        }
        cout<<sum<<endl;
    }



    return 0;
}
