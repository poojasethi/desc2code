#include<cstdio>
#define MOD 1000000007
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n,A,B,C;
        scanf("%d %d %d %d",&n,&A,&B,&C);
        ll ans=0;
        for(int i=0;i<=A;i++){
            for(int j=0;j<=B;j++){
                int tmp=n-i-j;
                if(tmp>=0 && tmp<=C){
                    ans+=tmp+1;
                }
                else if(tmp>C){
                    ans+=C+1;
                }else
                    break;
            }
        }
        printf("%lld\n",ans);
        
    }
}
