#include <iostream>
using namespace std;

int main() {
   int T;
   cin >> T;
   while (T--) {
      int N, A, B, C;
      cin >> N >> A >> B >> C;
      long long count = 0;
      for (int i = 0; i <= N; i++ ) {
         count += (max(0, min(A,i)+min(B,i)-i+1) * (min(C, N-i)+1));
      }
      cout << count << endl;
   }
   return 0;
}
