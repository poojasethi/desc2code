#include<bits/stdc++.h>

using namespace std;


int main(){
	int t;
	int n,a,b,c;
	cin>>t;
	while(t--){
		cin>>n>>a>>b>>c;
		long long ans = 0;
		for(int i=0;i<=a;i++)
			for(int j=0;j<=b;j++){
				if(i+j <= n){
					int to_add = n-i-j;
					if(to_add >= c)ans += c+1;
					else  if(to_add >= 0)ans+=to_add+1;
				}
			}
		cout<<ans<<endl;
	}

}
