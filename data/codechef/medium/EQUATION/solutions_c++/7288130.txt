#include<bits/stdc++.h>
using namespace std;
long long int f(long long int n)
{
  long long int r;
   if(n<0)
    return 0;
   else
   {
      r=(n+1)*(n+2)*(n+3);
   }
   return (r/6);
}
int main()
{
  long long int t,n,a,b,c;
  scanf("%lld",&t);
  while(t--)
  {
     scanf("%lld%lld%lld%lld",&n,&a,&b,&c);
     long long int arr=f(n)-f(n-a-1)-f(n-b-1)-f(n-c-1)+f(n-a-b-2)+f(n-b-c-2)+f(n-a-c-2)-f(n-a-b-c-3);
     printf("%lld\n",arr);
  }
  return 0;
 }
