#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin >> t;
	while(t--){
		int N, A, B, C;
		cin >> N >> A >> B >> C;
		long long ans = 0;
		for(int i = 0;i <= A && i <= N;++i){
			for(int j = 0;j <= B && i + j <= N;++j){
				ans += min(C, N - i - j) + 1;
			}
		}
		cout << ans << "\n";
	}
	return 0;
}
