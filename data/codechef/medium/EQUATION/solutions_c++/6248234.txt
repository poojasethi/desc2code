#include<iostream>
using namespace std;
typedef long long lld;
int combi(int n)
{
	int ans = (n*(n-1)) /2;
	return ans;
}

int fun(int n, int m)
{
	if ( (n-m) < 0) return 0;
	else if( (n-m) == 0) return 1;
	else
	{
		return combi(n-m+2);
	}
}

 int main()
 {
	 int t,N,a,b,c;
	 cin >> t;
	 while(t--)
	 {
		 cin  >> N >> a >> b >> c;
		 lld ans = 0;
		 for(int n = 0; n <= N; n++)
		 {
			 lld total = combi(n+2);
			 lld complement = 0;
			
				 lld x = fun(n, a+1);
				 lld y = fun(n, b+1);
				 lld z = fun(n, c+1);
				 lld p = fun(n, (a+c+2));
				 lld q = fun(n ,(b+c+2));
				 lld r = fun(n, (a+b+2));
				 lld g = fun(n, (a+b+c+3));
				 complement = x+y+z-p-q-r+g;
				 
			 ans += (total - complement);
			
		 }
		 cout << ans << endl;
	 }
	 return 0;
 }
