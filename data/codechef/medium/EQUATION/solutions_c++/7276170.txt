#include <iostream>
#include <cstdlib>
#include <algorithm>
#include <math.h>

using namespace std;

int main()
{
	int t,N,A,B,C,i;
	cin>>t;
	while(t--)
	{
		long long int cnt=0;
		cin>>N>>A>>B>>C;
	for(i=0;i<=N;i++)
		cnt+=max(0,min(A,i)+ min(B,i)-i+1) * (min(C,N-i)+1);
	cout<<cnt<<endl;
	}
	return 0;
}
