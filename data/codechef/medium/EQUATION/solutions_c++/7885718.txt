#include<stdio.h>
using namespace std;
int main()
{
    int t,i,j,a,k,b,c,n;
    long long int ans;
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        scanf("%d%d%d%d",&n,&a,&b,&c);
        for(i=0;i<=a;++i)
        {
            for(j=0;j<=b;++j)
            {
                if(i+j<=n)
                {
                    k=n-i-j;
                    if(k<c)
                        ans+=k+1;
                    else
                        ans+=c+1;
                }
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
