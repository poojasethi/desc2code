#include<iostream>
#include<map>
#include<set>

using namespace std;

int main()
{

int i,j,m,t,n;
cin>>t;
while(t--)
{

map<string,set<int> > m1;
map<string,set<int> >::iterator it;


m1.clear();

cin>>n;
string s[n];

for(i=0;i<n;i++)
{
  cin>>s[i];
}

int i,j;

for(m=0;m<n;m++)
{
  for(i=0;i<s[m].size();i++)
   {
    for(j=0;j<s[m].size();j++)
    {
      string temp=s[m].substr(i,s[m].size()-j+1);
        m1[temp].insert(m);
    }
   }
}
int op1=0;
string ans;

for(it=m1.begin();it!=m1.end();it++)
{
    if(it->second.size()==n&&it->first.size()>op1)
        ans=it->first,op1=ans.size();
}

cout<<ans<<endl;

}

return 0;

}
