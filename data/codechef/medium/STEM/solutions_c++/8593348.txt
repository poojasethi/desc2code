#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

bool isSubstring(string& word, string& s)
{
	if(word.length() < s.length()) return false;
	for(int i = 0; i < word.length() - s.length() + 1; ++i)
	{
		int j;
		for(j = 0; j < s.length(); ++j)
			if(word[i + j] != s[j])
				break;
		if(j == s.length()) return true;	
	}
	return false;
}

int main() 
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int N;
		scanf("%d",&N);
		vector<string> words(N);
		for(int i = 0; i < N; ++i)
			cin >> words[i];
		string ans = "";
		int l = words[0].length();
		int maxlen = 0;
		for(int i = 0; i < l; ++i)
		{
			for(int j = i; j < l; ++j)
			{
				string s = words[0].substr(i, j - i + 1);
				//cout << s << endl;
				if(j - i + 1 < maxlen) continue;				
				bool ok = true;				
				for(int k = 1; k < N; ++k)
				{
					if(!isSubstring(words[k], s))
					{
						//cout << "i = " << i << "j = " << j << "k = " << k << endl;
						ok = false;
						break;
					}
				}
				if(ok)
				{
					//cout << "i = " << i << "j = " << j << endl;
					if(j - i + 1 > maxlen)
					{
						ans = s;
						maxlen = j - i + 1;
					}
					else if(j - i + 1 == maxlen && s.compare(ans) < 0)
						ans = s;
				}
			}
		}
		cout << ans << endl;
	}
	return 0;
}
