#include <bits/stdc++.h>
using namespace std;

#define DBG 0

string A[30], temp, ans;

void solve(){
	int n;
	ans = 'a';
	cin>>n;
	for(int i = 0; i < n; i++)	cin>>A[i];
	for(int i = 0; i < A[0].length(); i++){
		for(int j = 1; j <= A[0].length(); j++){
			temp = A[0].substr(i, j);
			bool ok = true;
			for(int k = 1; k < n; k++){
				if(A[k].find(temp) == string::npos)	ok = false;
			}
			if(ok){
				if(ans.length() < temp.length())	ans = temp;
				else if(ans.length() == temp.length())	ans = min(ans, temp);
			}
		}
	}
	cout<<ans<<endl;
}

int main(){

	if(DBG){
		freopen("input.txt", "r", stdin);
		freopen("output.txt", "w", stdout);
	}

	cin.tie(NULL);
	iostream::sync_with_stdio(false);

	int t;
	cin>>t;
	while(t--)	solve();
}
