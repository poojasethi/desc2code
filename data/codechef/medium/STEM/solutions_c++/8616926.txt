#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,j,k;
    bool done;
    string a[100],st,sol;
    cin>>t;
    while(t--)
    {
        sol="";
        cin>>n;
        for(i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);
        for(i=0;i<a[0].length();i++)
        {
            done=false;
            for(j=i;j<a[0].length();j++)
            {
                string st;
                st=a[0].substr(i,j-i+1);
             //   cout<<st<<" ";
                for(k=0;k<n;k++)
                {
                    if(a[k].find(st)!=string::npos)
                    {
                       done=true;
                       //cout<<st<<" ";
                    }
                    else
                    {
                        done=false;
                        break;
                    }
                }
                if(done==true && st.length()>=sol.length())
                    {
                     if(st.length()==sol.length())
                     {
                         sol=min(st,sol);
                     }
                     else
                        sol=st;
                    }
            }
        }
        cout<<sol<<"\n";
    }
    return 0;
}
