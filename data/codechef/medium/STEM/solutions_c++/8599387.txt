#include<bits/stdc++.h>
using namespace std;
 
string str1,ans,str[12];
int t,n,i,j,k,cnt;
int main()
{
    scanf("%d",&t);
    while(t--)
    {
        ans="";
        scanf("%d",&n);
        for(i=0;i<n;i++)
                cin>>str[i];
 
        for(i=1;i<=str[0].length();i++)
        {
            for(j=0;j<=str[0].length()-i;j++)
            {
                str1=str[0].substr(j,i);
                cnt=1;
                for(k=1;k<n;k++)
                {
                    if(str[k].find(str1)!=-1)
                        cnt++;
                }
                if(cnt==n)
                {
                    if(ans.length()==str1.length())
                    {
                        if(ans>str1)
                            ans=str1;
                    }
                     else if(ans.length()<str1.length())
                        ans=str1;
                }
            }
        }
        cout<<ans<<"\n";
    }
return 0;
}
