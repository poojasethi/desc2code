#include <bits/stdc++.h>

using namespace std;
string inp[25];
string sub[2500];
int main()
{
    int t,n,i,len=0,count=0;
    string ans="";
    cin>>t;
    while(t--)
    {
        count=0;
        len=0;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>inp[i];
        }
        for(i=0;i<inp[0].size();i++)
        {
            for(int j=i+1;j<inp[0].size();j++)
            {
                sub[count]=inp[0].substr(i,j-i+1);
                count++;
            }
        }
        for(i=0;i<count;i++)
        {
            int flag=0;
            for(int j=1;j<n;j++)
            {
                if(inp[j].find(sub[i])==string::npos)
                {
                    flag=1;
                    break;
                }
            }
            if(flag==0)
            {
                if(sub[i].size()==len)
                    {
                        for(int k=0;k<len;k++)
                        {
                            if(sub[i][k]<ans[k])
                            {
                                ans=sub[i];
                                break;
                            }
                        }
                    }
                    else if(sub[i].size()>len)
                    {
                        ans=sub[i];
                        len=sub[i].size();
                    }
            }
        }


        cout<<ans<<endl;
    }
    return 0;
}
