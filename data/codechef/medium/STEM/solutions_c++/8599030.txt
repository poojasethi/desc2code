#include<stdio.h>
#include<string.h>
#include<stdlib.h>
typedef struct node
{
   char arr[21];
   int len;
}node;

int substr(node a,char b[])
{
  int i,j,m=strlen(b);
  int count;
  
  for(i=0;i<a.len;i++)
  {
      count=0;
      for(j=0;j<m;j++)
      {
         if(a.arr[i+j]==b[j]) count++;
      }
      if(count==m) return 1;
  }
  return 0;

}
