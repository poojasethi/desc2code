void run() {
	int n = in(), _min = 0, _max = 0;
	string s;
	vs str;
	vs ans[30];
	F(i, n) {
		cin >> s;
		str.PB(s);
		if(s.S < str[_min].S) _min = i;
	}
	F(i, str[_min].S) {
		FOR(j, i, str[_min].S) {
			bool isSubstr = true;
			int sz = j-i+1;
			F(k, n) {
				if(k == _min) continue;
				bool found = false;
				for(int l = 0; l < str[k].S-sz+1 && !found; l++) {
					for(int m = 0; m < sz; m++) {
						if(str[_min][m+i] != str[k][l+m]) break;
						if(m == sz-1) found = true;
					}
				}
				isSubstr &= found;
			}
			if(isSubstr) {
				ans[sz].PB(str[_min].substr(i, sz));
				_max = max(_max, sz);
			}
		}
	}
	sort(ALL(ans[_max]));
	printf("%s\n", ans[_max][0].c_str());
}
