#include <bits/stdc++.h>
using namespace std;
string a[30];
int main()
{
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        vector<string>v;
        vector<string>v1;
        scanf("%d",&n);
        int index=-1;
        int mini=50;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
            int l=a[i].length();
            if(l<mini)
            {
                mini=l;
                index=i;
            }
        }
        /*cout<<"fuck";
        for(int i=0;i<n;i++)
        cout<<a[i]<<" ";
        cout<<"\n";
        cout<<index<<" "<<a[index]<<" "<<mini<<"\n";*/
        int flag=-1;
        for(int i=0;i<mini;i++)
        {
            for(int j=1;j<=mini-i;j++)
            {
                string ans=a[index].substr(i,j);
                int c=0;
                for(int k=0;k<n;k++)
                {
                    if(a[k].find(ans)!=std::string::npos)
                    c++;
                }
                if(c==n)
                {

                flag=1;
                v.push_back(ans);
                }
            }
        }
        if(flag==-1)
        {
            cout<<""<<"\n";
            continue;
        }
        int max1=v[0].length();
        for(int i=1;i<v.size();i++)
        {
            if(v[i].length()>max1)
            {
                max1=v[i].length();
            }

        }
        for(int i=0;i<v.size();i++)
        {
            if(v[i].length()==max1)
            v1.push_back(v[i]);
        }
        sort(v1.begin(),v1.end());


        cout<<v1[0]<<"\n";
    }

    return 0;
}
