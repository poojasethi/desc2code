#include <cstdio>
#include <string>
#include <algorithm>
#include <iostream>
#include <vector>
#include <iterator>
#include <math.h>
const int MOD = 1000*1000*1000 + 7;

using namespace std;
string arr[10+10];
char temp[20+10];
int n;
int main(){
    int t;
    scanf("%d", &t);
    while(t--){


        scanf("%d", &n);
        for(int i=0;i<n;i++){
            scanf("%s", temp);
            arr[i]=temp;
        }
        string ans="";
        for(int i=0;i<arr[0].size();i++){
            for(int j=0;j<=i;j++){
                string word = arr[0].substr(j, i-j+1);
                bool flag = true;
                for(int k=0;k<n;k++){
                    if((arr[k].find(word))==-1){
                        flag = false;
                        break;
                    }
                }
                if(flag){
                    if(word.size()>ans.size() || word.size()==ans.size() && word<ans)
                        ans=word;
                }
            }
        }
        printf("%s\n", ans.c_str());
    }


    return 0;
}
