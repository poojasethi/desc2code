#include<bits/stdc++.h>
using namespace std;
bool find(string a,string b)
{
	for(int i=0;i+a.size()<=b.size();i++)
	{
		if(a==b.substr(i,a.size()))
		 return true;
	}
	return false;
}
 
  int main()
  {
  	int t;
  	cin>>t;
  	while(t--)
  	{
  		int n;
  		cin>>n;
  		vector<string> s(n);
  		int len=INT_MAX;
  		string ss;
  		for(int i=0;i<n;i++)
  		{
  			 cin>>s[i];
  			 if(s[i].size()<len)
  			 {
  			  len =s[i].size();
  			  ss=s[i];
  		}
		  }
		 string result;
		 
		  for(int l=1;l<=ss.size();l++)
		  {
		  	 for(int i=0;i+l<=len;i++)
		  	 {
		  	 	int cnt=0;
		  	 	string temp = ss.substr(i,l);
		  	 	for(int i=0;i<n;i++)
		  	 	{
		  	 	 if(find(temp,s[i])) ++cnt;
		  	 	 else break;
		  	 }
		  	 	if(cnt==n)
		  	 	{
		  	 		if(result=="") result=temp;
		  	 	    if(result.size()==temp.size()) result = min(result,temp);
		  	 	    if(result.size()<temp.size()) result =temp;
				   }
		     }
		   } 
		cout<<result<<endl;
	  }
  }
