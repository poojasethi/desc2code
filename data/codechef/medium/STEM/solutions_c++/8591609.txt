#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    char a[1000];
    string b[100],s;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            scanf("%s",a);
            b[i]=a;
        }
        string ans="";
        for(int i=0;i<b[0].size();++i)
        {
            for(int j=0;j<=i;++j)
                 {
                     s=b[0].substr(j,i-j+1);
                    // cout<<s<<endl;
            bool flag=true;
            //cout<<n<<"lkj"<<endl;

            for(int k=0;k<n && flag;++k)
                {

                    flag &=  (b[k].find(s)!= -1);
          /* cout<<flag<<endl;
                    if(b[k].find(s))
                        cout<<s<<"d"<<endl; */
        }
        if(flag)
        {
           if (s.size() > ans.size() || s.size() == ans.size() && s < ans)
                ans=s;
               // cout<<"ans is "<<ans<<endl;
        }
        }
    }
    cout<<ans<<endl;
}
}
