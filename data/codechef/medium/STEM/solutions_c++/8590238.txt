#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;
int main()
{
	int l,i,t,n,k,j;
	bool fg;
	cin>>t;
	while(t--)
	{
		cin>>n;
        string ss[n],x(""),y("");
		for(i=0;i<n;i++)
 		cin>>ss[i];
        l=ss[0].size();
		for(i=0;i<l;i++)
		{
			for(j=0;j<=i;j++)
			{
				fg=true;
				x=ss[0].substr(j,i-j+1);
				for(k=1;k<n;k++)
				{
					if(fg==false)
					break;
					if(ss[k].find(x)==-1)
					fg=false;
				}
				if(fg==true)
				{
					if(x.size()>y.size()||(x.size()==y.size()&&x<y))
					{
						y=x;
					}
				}
			}
		}
		cout<<y<<"\n";
	}
	return 0;
}
