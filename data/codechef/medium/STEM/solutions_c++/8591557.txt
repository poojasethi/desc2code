#include<bits/stdc++.h>
using namespace std;
 
#define lli long long
#define pb push_back
#define mp make_pair
#define gc getchar
#define lld "%I64d"
#define DEBUG(x) cout<<">value ("<<#x<<") : "<<x<<endl;
#define mod 1000000007
#define llu unsigned long long
 
vector<string>vec;
 
int main(){
    int t,n;
    string s;
    cin>>t;
    while(t--){
        vec.clear();
        cin>>n;
        for(int i=0;i<n;i++){
            cin>>s;
            vec.pb(s);
        }
        string ans  = "";
        for(int i=0;i<vec[0].size();i++){
            string sub = "";
            for(int j=i;j<vec[0].size();j++){
                sub += vec[0][j];
                bool ok = true;
                for(int k=1;k<n;k++){
                    if(vec[k].find(sub)==string::npos){
                        ok = false;
                        break;
                    }
                }
                if(ok){
                    if(sub.size() > ans.size())
                        ans = sub;
                    else if(sub.size()==ans.size() && sub < ans)
                        ans = sub;
                }
            }
        }
        cout<<ans<<"\n";
    }
 
    return 0;
}
