#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
string s[10],ans,temp;
int n;
int con(string temp)
{  
    int i;
    for(i=1;i<n;i++)
    {
        if(s[i].find(temp)==string::npos)
            return 0;
    }
    
    return 1;
}
int main()
{
    int t,i,j,counter;
   
    cin>>t;
    while(t--)
    {
       cin>>n;
       
      for(i=0;i<n;i++)
        cin>>s[i];
        
      ans.clear();
        counter=0;
      for(i=0;i<s[0].size();i++)
        for(j=i;j<s[0].size();j++){
            temp=s[0].substr(i,j-i+1);
            
          //  cout<<temp<<endl;
            
            if(con(temp))
            {
                if(counter==0)
                {
                    counter=1;
                    ans=temp;
                }
                else
                {
                    if(temp.size()>ans.size())
                    {
                        ans=temp;
                    }
                    else if(temp.size()==ans.size())
                    {
                        if(temp<ans)
                            ans=temp;
                    }
                }
            }
            
            //cout<<ans<<endl;

        }
        
        cout<<ans<<endl;


   }
        
    return 0;
}
