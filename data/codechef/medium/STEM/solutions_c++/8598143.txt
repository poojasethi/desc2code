#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,j,k;
    cin>>t;
    while(t--)
    {
        vector< string > v;
        cin>>n;
        string s[n];
        for(i=0;i<n;i++)
            cin>>s[i];
        int min=50;
        string base;
        for(i=0;i<n;i++)
        {
            if(s[i].length()<min)
            {
                base=s[i];
                min=s[i].length();
            }
        }
        //cout<<base<<endl;
        for(i=0;i<base.length();i++)
        {
            for(j=i;j<base.length();j++)
            {
                string x=base.substr(i,base.length()-j);
                //cout<<x<<endl;
                for(k=0;k<n;k++)
                {
                    if(s[k].find(x)==string::npos)
                        break;
                }
                if(k==n&&x.length()>0)
                {
                    v.push_back(x);
                   // cout<<"push "<<x<<endl;
                }
            }
        }
        string ans;
        int max=0;
        for(i=0;i<v.size();i++)
        {
            //cout<<v[i]<<endl;
            if(v[i].length()>max)
            {
                max=v[i].length();
                ans=v[i];
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
