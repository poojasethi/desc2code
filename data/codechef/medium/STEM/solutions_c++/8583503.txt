#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
		        map<string,int> map23;

        int k=0;
        while(k<n)
        {
            map<string,int> mp;
            k++;
            string s;
            cin>>s;
            int l=s.length();
            for(int i=0;i<l;i++)
            {
                for(int j=1;j<=l-i;j++)
                    mp[s.substr(i,j)]=1;
                map<string,int>::iterator it;
                for(it=mp.begin();it!=mp.end();++it)
                    if(map23[it->first]==(k-1))
                        map23[it->first]=k;
            }

        }
        string ans;
        int mx=0;
        map<string,int>::iterator it;
            for(it=map23.begin();it!=map23.end();++it)
            {
                string pat=it->first;
                if((it->second)==n && (pat.length()>mx) )
                {
                    mx=pat.length();
                    ans=it->first;
                }
            }
        cout<<ans<<"\n";
    }
}
