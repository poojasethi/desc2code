#include<bits/stdc++.h>
using namespace std;
int dfind(int d,vector<int>& root)
{
    while(d!=root[d])
    {
        d=root[d];
    }
    return d;
}
int main()
{
    int i,j,n,t,q,x,y,f1,f2,op;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        vector<int> maxi(n+1),root(n+1);
        for(i=1;i<=n;i++)
        {
            root[i]=i;
            scanf("%d",&(maxi[i]));

        }
        scanf("%d",&q);
        while(q--)
        {
            scanf("%d",&op);
            if(op==0)
            {
                scanf("%d%d",&x,&y);
                f1=dfind(x,root);
                f2=dfind(y,root);
                root[x]=f1;
                root[y]=f2;
                if(f1==f2)
                {
                    printf("Invalid query!\n");
                }
                else
                {
                    if(maxi[f1]>maxi[f2])
                    {
                        root[f2]=f1;
                    }
                    else if(maxi[f2]>maxi[f1])
                    {
                        root[f1]=f2;
                    }
                }
            }
            else
            {
                scanf("%d",&x);
                f1=dfind(x,root);
                root[x]=f1;
                printf("%d\n",f1);
            }
        }
    }
    return 0;
}
