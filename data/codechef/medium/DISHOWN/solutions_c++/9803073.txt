#include <iostream>
#include<string.h>

using namespace std;

int parent[100001];
int a[100001];
int find_set(int x)
{
//cout<<x<<"\n";
    if(x!=parent[x])
    {
      //  cout<<"oo";
        parent[x]=find_set(parent[x]);
    }
    return parent[x];
}

int mergeit(int x,int y)
{
    int u,v;
u=find_set(x);
v=find_set(y);
//cout<<u<<" "<<v;
if(u==v)
{
    return 0;
}
else
    if(a[u]==a[v])
    return 1;
else
    if(a[u]>a[v])
{
    parent[v]=u;

}
else{
    parent[u]=v;
  //cout<<parent[u];
}
return 1;
}

int main()
{
     ios::sync_with_stdio(0);
   int t,n,i,q,x,z,y,w,f;
cin>>t;
   while(t--)
   {
       memset(parent,0,sizeof(parent));
      cin>>n;
       for(i=1;i<=n;i++)
       {
           cin>>a[i];
     parent[i]=i;
       }
       cin>>q;
       while(q--)
       {
          cin>>z;
           if(z==0)
           {
               cin>>x>>y;
               if(mergeit(x,y)==0)
               {
                  cout<<"Invalid query!"<<"\n";
               }
           }
           else if(z==1)
           {
               cin>>f;
             //  cout<<paren;
             w=find_set(f);
        cout<<w<<"\n";
           }
       }
   }
    return 0;
}
