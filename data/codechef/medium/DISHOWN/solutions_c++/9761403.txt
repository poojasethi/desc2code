int findfun(int x){
    if (parent[x]==x){
        return x;
    }
        else{
            int j=findfun(parent[x]);
            parent[x]=j;
            return j;
        }
}

int p2=findfun(y);
if(score[p1]>score[p2])

else if (score[p2]>score[p1])
    parent[p1]=p2;
}
