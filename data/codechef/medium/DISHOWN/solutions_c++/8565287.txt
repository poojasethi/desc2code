#include<iostream>
#include<cstdio>
using namespace std;
int parent[10002];
int _find(int x)
{
    if(parent[x] == x)
        return x;
    int j = _find(parent[x]);
    parent[x] = j;
    return j;
}
void _set_union(int *s, int x, int y)
{
   int p = _find(x);
   int q = _find(y);
   if(p == q)
   {
       printf("Invalid query!\n");
       return;
   }
   if(s[p] > s[q])
    parent[q] = p;
   else if(s[q] > s[p])
    parent[p] = q;
}

int main()
{
   int t,n,q,type,x,y;
   scanf("%d",&t);
   while(t--)
   {
       scanf("%d",&n);
       int s[10002];
       for(int i=1; i<=n; i++)
        scanf("%d",&s[i]);
       for(int i=1; i<=n; i++)
        parent[i] = i;
       scanf("%d",&q);
       while(q--)
       {
           scanf("%d",&type);
           if(!type)
           {
               scanf("%d%d",&x,&y);
               _set_union(s,x,y);
           }
           else
           {
               scanf("%d",&x);
               int par = _find(x);
               printf("%d\n",par);
           }
       }
   }
    return 0;
}
