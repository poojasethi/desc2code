#include<bits/stdc++.h>
#define  FOR(i,n) for(int i=1;i<=n;++i)
int parent[10001];
int find(int x)
{
    while(x!=parent[x])
    {
        x=parent[x];
        //printf("F\n");
    }
    return x;
}
using namespace std;
int main()
{
    int t,num,Q,x,y,test;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&num);
        int scores[num+1];
        FOR(i,num)
        {
            scanf("%d",&scores[i]);
            parent[i]=i;
        }
        scanf("%d",&Q);
        while(Q--)
        {   int par_x,par_y;
            scanf("%d",&test);
            if(!test)
            {   scanf("%d%d",&x,&y);
                par_x=find(x);par_y=find(y);
                if(par_x==par_y)
                printf("Invalid query!\n");
                if(scores[par_x]>scores[par_y])
                {parent[par_y]=par_x;
                parent[y]=par_x;
                }
                if(scores[par_x]<scores[par_y])
                {parent[par_x]=par_y;
                parent[x]=par_y;
                }
            }
            else
            {
                scanf("%d",&x);
                printf("%d\n",find(x));
            }
        }

    }
    return 0;
}
