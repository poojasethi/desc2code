#include <bits/stdc++.h>
using namespace std;
 
typedef pair< int , int > pii;
typedef long long LL;
#define fr first
#define se second
#define EPS 1e-8
#define INF 10000*10000*10000LL
stringstream ss;
#define two(x) ( 1LL<<x )
LL mod = 1000000007LL;

/**************************Code****************************/

int par[10010], s[10010];

int find( int x )
{
	return x == par[x] ? x : par[x] = find( par[x] );
}

int main()
{
	int t, n, q, x, y;
	scanf( "%d", &t );
	while( t -- )
	{
		scanf( "%d", &n );
		for( int i = 1 ; i <= n ; i ++ )
			scanf( "%d", s + i ), par[i] = i;
		scanf( "%d", &q );
		while( q -- )
		{
			scanf( "%d", &x );
			if( x == 0 )
			{
				scanf( "%d%d", &x, &y );
				x = find(x), y = find(y);
				if( x == y )
					printf( "Invalid query!\n" );
				else
				{
					if( s[x] > s[y] )
						par[y] = x;
					else if( s[y] > s[x] )
						par[x] = y;
				}
			}
			else
			{
				scanf( "%d", &x );
				printf( "%d\n", find(x) );
			}
		}
	}
	return 0;
}
