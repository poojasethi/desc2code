#include<iostream>
#include<stdio.h>
using namespace std;
void createSet(int *parent,int n)
{
    for(int i=1;i<=n;i++)
        parent[i]=i;
}
int findSet(int *parent,int x)
{
    if(x!=parent[x])
    return parent[x]=findSet(parent,parent[x]);
    return x;
}
void mergeSet(int *s,int *parent,int x,int y)
{
    x=findSet(parent,x);
    y=findSet(parent,y);
    if(x!=y)
    {
        if(s[x]>s[y])
            parent[y]=x;
        else if(s[x]<s[y])
            parent[x]=y;
    }
    else
    printf("Invalid query!\n");
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        scanf("%d",&n);
        //cin>>n;
        int s[n+1];
        int parent[n+1];
        for(int i=1;i<=n;i++)
            scanf("%d",s+i);
        createSet(parent,n);
        int q;
        scanf("%d",&q);
        for(int i=0;i<q;i++)
        {
            int type,x,y;
            scanf("%d",&type);
            if(type==0)
            {
                scanf("%d%d",&x,&y);
                mergeSet(s,parent,x,y);

            }
            else
            {
                scanf("%d",&x);
                printf("%d\n",findSet(parent,x));
            }
        }
    }
    return 0;
}
