const int maxn = 1000002;

int father[maxn], s[maxn];

int find(int x)
{
	int y = x;
	while (x != father[x]) {
            //printf("First=%d %d\n",x,father[x]);
		x = father[x];
	}
	/*for (;y != x;) {
		int t = father[y];
		father[y] = x;
		y = t;
        printf("Second=%d %d %d\n",t,y,father[y]);
	}*/
	return x;
}

int main()
{
	int T;
	scanf("%d",&T);
	while(T--){
		int n, q;
		scanf("%d", &n);
		for (int i = 1; i <=n; ++ i) {
			scanf("%d", &s[i]);
			father[i] = i;
		}
        scanf("%d", &q);
		for (int i = 1; i <=q; ++ i) {
			int op, x, y;
			scanf("%d", &op);
			if (op == 0) {
				scanf("%d%d", &x, &y);
				int fx = find(x), fy = find(y);
				father[x]=fx,father[y]=fy;
				if (fx == fy) {
					puts("Invalid query!");
				} else {
					if (s[fx] > s[fy]) {
						father[fy] = fx;
						father[y]=fx;

					} else if (s[fx] < s[fy]){
						father[fx] = fy;
						father[x]=fy;

					}
				}
			} else {
				scanf("%d", &x);
				printf("%d\n", find(x));
			}
		}
	}
	return 0;
}
