#include <bits/stdc++.h>

using namespace std;

int par[1000000],a[1000000];

int fpar(int d)
{
if(d!=par[d])
{
    par[d]=fpar(par[d]);
}
return par[d];
}

int mergei(int x,int y)
{
    int xx,yy;
    xx=fpar(x);
    yy=fpar(y);
    if(xx==yy)
    return 0;
else
    if(a[xx]==a[yy])
    return 1;
else
    if(a[xx]>a[yy])
    par[yy]=xx;
else
    par[xx]=yy;
return 1;
}

int main()
{
   int t,x,y,q,z,n,i;
   scanf("%d",&t);
   while(t--)
   {
       memset(par,0,sizeof(par));
       scanf("%d",&n);
       for(i=1;i<=n;i++)
       {
         scanf("%d",&a[i]);
         par[i]=i;
       }
       scanf("%d",&q);
       while(q--)
       {
           scanf("%d",&z);
           if(z==0)
           {
               scanf("%d %d",&x,&y);
               if(mergei(x,y)==0)
                cout<<"Invalid query!"<<"\n";
           }
           else
           {
               scanf("%d",&x);
               y=fpar(x);
               printf("%d\n",y);
           }
       }
   }
    return 0;
}
