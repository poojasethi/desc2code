#include<bits/stdc++.h>
using namespace std;

#define LL long long
#define sll(a) scanf("%d",&a)
#define s2ll(a,b) scanf("%d%d",&a,&b)
#define plln(a) printf("%d\n",a)

#define rep(i,n) for(i=0;i<n;i++)
#define REP(i,a,b) for(i=a;i<=b;i++)
int par[10010], s[10010];
int head(int i)
{
    int y=i;
    if(par[i]==i)
        return i;
    else
        return par[i]=head(par[i]);
}
int merge(int a, int b)
{
    a=head(a);
    b=head(b);
    if(a==b)
        return 0;
    if(s[a]>s[b])
    {
        par[b]=a;
        //s[a]+=s[b];
    }
    else if(s[b]>s[a])
    {
        par[a]=b;
        //s[b]+=s[a];
    }
    return 1;
}
/* https://www.codechef.com/problems/DISHOWN*/
int main()
{
    int t,n,a,b,i;
    sll(t);
    while(t--)
    {
        sll(n);
        int q;
        REP(i,1,n)
        {
            sll(s[i]);
            par[i]=i;
        }
        sll(q);
        while(q--)
        {
            int ch;
            sll(ch);
            if(ch==1)
            {
                sll(a);
                plln(head(a));
            }
            else
            {
                s2ll(a,b);
                if(!merge(a,b))
                    printf("Invalid query!\n");
            }
        }
    }
    return 0;
}
