#include<bits/stdc++.h>
using namespace std;
int find(vector<int>& root,int i)
{
	while(i!=root[i])
	{
		i=root[i];
	}
	return i;
}
int main()
{
	int t,i,n,q,op,x,y,r1,r2;
	vector<int> root(10000),price(10000);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			root[i]=i;
			scanf("%d",&(price[i]));
		}
		scanf("%d",&q);
		while(q--)
		{
			scanf("%d",&op);
			if(!op)
			{
				scanf("%d%d",&x,&y);
				x--;
				y--;
				r1=find(root,x);
				r2=find(root,y);
				root[x]=r1;
				root[y]=r2;
				if(r1==r2)
				{
					printf("Invalid query!\n");
				}
				else
				{
					if(price[r1]!=price[r2])
					{
						if(price[r1]>price[r2])
						{
							root[r2]=r1;
						}
						else
							root[r1]=r2;
					}
				}
			}
			else
			{
				scanf("%d",&x);
				x--;
				r1=find(root,x);
				root[x]=r1;
				printf("%d\n",find(root,x)+1);
			}
		}
	}
	return 0;
}
