#include<bits/stdc++.h>
using namespace std;
int parent[10000],score[10000],query[10000];
int find(int parent[],int index)
{
	if(parent[index]==index)
	return index;
	return parent[index]=find(parent,parent[index]);
}
int main()
{
	int t,n,i,q,j,k,qr,x,y,par_x,par_y;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=1;i<=n;i++)
		{
			scanf("%d",&score[i]);
			parent[i]=i;
		}
		scanf("%d",&q);
		while(q--)
		{
			scanf("%d",&qr);
			if(qr==0)
			{
			scanf("%d%d",&x,&y);
			par_x=find(parent,x);
			par_y=find(parent,y);
			
			if(par_x==par_y)
			printf("Invalid query!\n");
			else if(score[par_x]==score[par_y])
			{
			}
			else if(score[par_x]>score[par_y])
			parent[par_y]=par_x;
			else if(score[par_y]>score[par_x])
			parent[par_x]=par_y;
			}
			else
			{
				scanf("%d",&k);
			int p=find(parent,k);
			printf("%d\n",parent[p]);
			}
			
		}
	}
	return 0;
}
