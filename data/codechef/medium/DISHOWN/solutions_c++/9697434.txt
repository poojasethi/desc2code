#define e printf("\n");
#define ll long long
const ll mx=10001;
const ll mod=1000000007;
const ll inf=0x3f3f3f3f;
int par[mx+1];
int rnk[mx+1];
int fin(int i)
{
  if(par[i]==i)
    return i;
  par[i]=fin(par[i]);
  return par[i];
}
void uni(int x,int y)
{
  int xr=fin(x);
  int yr=fin(y);
  if(xr==yr)
  {
    printf("Invalid query!\n");
    return ;
  }
  if(rnk[xr]>rnk[yr])
  {
    par[yr]=xr;
  }
  else
  if(rnk[xr]<rnk[yr])  
  {
    par[xr]=yr;
  }
}
int main()
{
  int q,n,i,t,j;
  scanf("%d",&t);
  while(t--)
  {
    scanf("%d",&n);
    for(i=1;i<=n;i++)
      {
        scanf("%d",&rnk[i]);
        par[i]=i;
      }
    scanf("%d",&q);
    while(q--)
    {
      scanf("%d%d",&i,&j);
      if(i)
        printf("%d\n",fin(j));
      else
      {
        scanf("%d",&i);
        uni(j,i);
      }
    }
  }
  return 0;
}
