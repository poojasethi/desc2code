#include<stdio.h>
#include<stdlib.h>
int s[10010];
typedef struct
{
    int parent;
}Subset;

Subset sub[10010];
int find(Subset sub[],int x)
{
        if(sub[x].parent!=x)
        sub[x].parent = find(sub,sub[x].parent);
		return(sub[x].parent);
}


int main()
{
	int i,j,q,n,x,y,t,a,b;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(i=1;i<=n;i++)
		{
			sub[i].parent=i;
			scanf("%d",&s[i]);
		}
		scanf("%d",&q);
		while(q--)
		{
			scanf("%d",&j);
			switch(j)
			{
				case 0:
					scanf("%d%d",&x,&y);
					a= find(sub,x);
					b= find(sub,y);
					if(a==b)
					{
						printf("Invalid query!\n");
					}
					else if(s[a]>s[b])
					{
						sub[b].parent= a;
					}
					else if(s[a]<s[b])
					{
						sub[a].parent= b;
					}
					break;
				case 1:
					scanf("%d",&x);
					printf("%d\n",find(sub,x));
					break;
			}
		}
	}
	return 0;
}
