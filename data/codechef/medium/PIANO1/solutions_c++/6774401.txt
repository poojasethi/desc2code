#include<bits/stdc++.h>
using namespace std;

int main()
{
	long int t,ans,ct,div,temp;
	string s;
	int len,i,n;
	cin>>t;
	while(t--)
	{
		cin>>s;
		cin>>n;
		ct=0;ans=0;
		len=s.length();
		for(i=0;i<len;i++)
		{
			if(s[i]=='T')
			ct=ct+2;
			else
			ct++;
		}
		div=n*12;
		temp=ct;
		while(ct<=div)
		{
			ans=ans+(div-ct);
			ct=ct+temp;
			//cout<<ans<<endl;
		}
		cout<<ans<<endl;
	}
	return 0;
}
