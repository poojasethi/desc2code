#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long T;
    cin>>T;
    while(T--)
    {
        string k;
        cin>>k;
        long long sum=0;
        for(int i=0;i<k.length();i++)
        {
            if(k[i]=='T')
                sum+=2;
            else
                sum+=1;
        }
        long long  M;
        cin>>M;
        long long res=12*M;
        long long res2=0;
        long  long step=sum;
        while(step<=res)
        {
            res2+=res-step;
            step+=sum;
        }
       cout<<res2<<endl;

    }
    return 0;
}
