#include <iostream>
#include <stdio.h>
using namespace std;
int main() {
    int t;
    int n;
    char s[10005];
    scanf("%d",&t);
    while(t--){
        scanf("%s",&s);
        scanf("%d",&n);
        int a = 0;
        for(int p, q, sp = 0; sp < 12 * n; sp++){
            q = sp;
            while(true){
                p = q;
                for(int i = 0; s[i]!='\0'; i++){
                    p += s[i] - 'R';
                }
                if(p >= 12 * n){
                    break;
                }
                q = p;
                a++;
            }
        }
        printf("%d\n",a);
    }
}
