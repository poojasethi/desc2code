#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main()
{
    int testcase,n,count=0,check,i,k,ans=0,range;;
    char a[1000001];
    scanf("%d",&testcase);
    while(testcase--)
    {
        count=k=0;
        cin>>a;
        scanf("%d",&n);
        for(i=0; i<strlen(a); i++)
        {
            if(a[i] == 'T')
                count+=2;
            else
                count+=1;
        }
        range=12*n;
        if((count+1)>range)
            printf("0\n");
        else
        {
            check=0;
            k=count;
            while(range>k)
            {
                check+=range-k;
                k+=count;
            }
            printf("%d\n",check);
        }
    }
    return 0;
}
