#include <bits/stdc++.h>
#define ll long long int

using namespace std;

int main()
{
	int t;
	int n;
	scanf("%d", &t);

	while (t--) {
		string s;
		cin >> s;
		cin >> n;
		ll ans = 0;
		int x, y;
		for (int i = 0; i < 12 * n; i++) {
			y = i;
			while (true) {
				x = y;
				for (int j = 0; j < s.length(); j++) {
					if (s[j] == 'S') {
						x += 1;
					} else if (s[j] == 'T') {
						x += 2;
					}
				}

				if (x >= 12 * n) {
					break;
				}
				y = x;
				ans++;
			}
		}

		printf("%lld\n", ans);
	}

	return 0;
}
