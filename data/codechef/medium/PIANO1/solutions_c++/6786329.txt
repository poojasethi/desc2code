#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define sc(n) scanf("%lld",&n)

int main()
{
	
	ll t,n,i,max,ans,length;
	char a[101];
	sc(t);
	ll sum,noofchar;

	while(t--)
	{
		scanf("%s",a);
		sc(n);
		max=n*12;
		sum=0;
		ans=0;
		length=strlen(a);

		
		for(i=0;i<strlen(a);i++)
		{
			if (a[i]=='S')
			sum=sum+1;
			else
			sum=sum+2;
		}
	//	printf("n,sum.length: %lld %lld %lld\n",n,sum,length);

		for(i=1;i<=max-sum;i++)
		{
			noofchar=max-i;
			ans=ans+(noofchar/sum);

		}
	
		printf("%lld\n",ans);
	}


	return 0;
}
