#include<stdio.h>
int main()
{
	long long int t,n,i,sum,res,div;
	char str[105];
	scanf("%lld", &t);
	while (t--)
	{
		sum = 0;
		res = 0;
		scanf("%s", str);
		scanf("%lld", &n);
		i = 0;
		while (str[i] != '\0')
		{
			if (str[i] == 'S')
				sum += 1;
			else
				sum += 2;
			i++;
		}
		n *= 12;
		n--;
		div = n / sum;
		if (div)
		{
			res += (n - (sum*div) + 1)*div;
			div--;
		}
		while (div != 0)
		{
			res += div*sum;
			div--;
		}
		printf("%lld\n", res);
	}
	getchar();
	getchar();
	return 0;
}
