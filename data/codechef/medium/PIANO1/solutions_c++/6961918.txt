#include <bits/stdc++.h>
 
using namespace std;
 
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		string s;
		cin>>s;
		int l=s.length();
		int i,n,len=0;
		scanf("%d",&n);
		for(i=0;i<l;i++){
			if(s[i]=='S')
				len++;
			else
				len+=2;
		}
		i=1;
		int key=12*n;
		int ans=0;
		while((i*len)+1<=key){
			ans=ans+(key-(i*len+1)+1);
			i++;
		}
		printf("%d\n",ans);
	}
	return 0;
}
