#include<iostream>
#include<stdio.h>

using namespace std;

int main()
{
	char a;
	int x,k,n;
	long t,ans;
	scanf("%ld",&t);
	
	while(t--)
	{
		getchar();		
		x=0;
		do
		{
			a=getchar();
			if(a=='T')
				x=x+2;
			if(a=='S')
				x++;
			//printf("value of a and x are _%c_  _%d_ \n",a,x );
		}while(a!='\n');
		scanf("%d",&k);
		k=k*12;
		n=k/x;
		ans=n*k - (x*n*(n+1))/2;
		printf("%ld\n",ans );
	}
	return 0;
}
