    #include<iostream>
    #include<string.h>
    using namespace std;
    typedef long long ull;
     
    int main()
    {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    ull t,len,i,n,mov,count;
    char str[100001];
    cin>>t;
    while(t--)
    { count=0,mov=0;
    cin>>str;
    len=strlen(str);
    cin>>n;
    for(i=0;i<len;i++)
    {
    if(str[i]=='T')
    mov+=2;
    else
    mov+=1;
    }
    for(i=0;i<n*12;i++)
    {
    ull k=0,temp,flag=0;
    temp=(12*n)-i;
    /// cout<<temp<<" ";
    while(temp>=0 && temp<=(12*n)-i)
    {
    if(k==0)
    {
    if(temp-(mov+1)>=0 )
    {
    temp=temp-(mov+1);
    k=1;
    count++;
    //cout<<temp<<" ";
    }
    else
    {
    flag=1;
    }
    }
    else
    {
    if(temp-mov>=0 )
    {
    //cout<<temp<<" ";
    temp=temp-mov;
    count++;
    }
    else
    {
    flag=1;
    }
    }
    if(flag==1)
    break;
    }
    }	
    cout<<count<<"\n";
    }
    return 0;
    }
