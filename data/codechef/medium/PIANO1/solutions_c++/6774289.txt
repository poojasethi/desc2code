#include<stdio.h>
#include<string.h>
int main()
{long int t;
scanf("%ld",&t);
while(t--)
{int n,i,len=0;
long int ctr=0,oct=0,s=0,p=0;
char arr[101];
scanf("%s",arr);
scanf("%d",&n);
len=strlen(arr);
for(i=0;i<len;i++)
{if(arr[i]=='T')
ctr+=2;
else if(arr[i]=='S')
ctr+=1;
}
oct=12*n;
p=ctr;
while(p<oct)
{s+=oct-p;
p+=ctr;
}
printf("%ld\n",s);
}
return 0;
}
