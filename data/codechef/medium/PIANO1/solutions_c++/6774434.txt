#include <cstdio>
#include <iostream>
#include <cassert>

using namespace std;

typedef unsigned long long u64;

#ifdef _DEBUG_
static const bool _debug_ = true;
#else
static const bool _debug_ = false;
#endif

#define derr if (_debug_) cerr

void preprocess() {
}

void doTest() {
    string S;
    cin >> S;
    int ssz = S.size();
    int N;
    cin >> N;
    derr << " N = " << N << " ssz = " << ssz << " S = " << S << endl;

    int nn, ss;
    int Ssum = 0;
    for (ss=0; ss<ssz; ++ss) {
        if (S[ss] == 'T') {
            Ssum += 2;
        } else if (S[ss] == 'S') {
            Ssum += 1;
        } else {
            assert(false);
        }
    }
    derr << "Ssum = " << Ssum << endl;
    u64 res = 0;
    for (nn=1; nn<=N*12; ++nn) {
        if ((nn+Ssum) > N*12) break;
        res += (N*12-nn)/Ssum;
    }

    cout << res << endl;
}

int main() {
    preprocess();
    int T;
    cin >> T;
    derr << " T = " << T << endl;
    for (int tt=0; tt<T; ++tt) {
        doTest();
    }
    return 0;
}
