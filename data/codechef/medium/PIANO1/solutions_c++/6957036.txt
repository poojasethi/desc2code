#include <iostream>
using namespace std;

int main() {
	int t, n, limit, i, j, delta;
	char s[105];
	long long int ans;
	cin >> t;
	while(t--) {
	    cin >> s >> n;
	    limit = 12 * n;
	    delta = 0;
	    for(i = 0; s[i] != '\0'; i++) {
	        if(s[i] == 'T') delta += 2;
	        else delta++;
	    }
	    ans = 0;
	    for (i = 1; i <= limit; i++) {
	        ans = ans + (limit - i) / delta;
	    }
	    cout << ans << "\n";
	}
	return 0;
}
