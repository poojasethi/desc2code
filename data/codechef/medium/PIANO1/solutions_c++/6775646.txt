#include <cstdio>
 
#define MAX 102
 
using namespace std;
 
int main()
{
    long test_case;
    char pattern[MAX];
    long pattern_value;
    long total_keys;
    long octaves;
    long long int scales;
    scanf("%ld", &test_case);
    while(test_case--)
    {
        scanf("%s",pattern);
        scanf("%ld", &octaves);
        int i = 0;
        pattern_value = 0;
        scales = 0;
        while(pattern[i] != '\0' && pattern[i] != '\n')
        {
            if(pattern[i] == 'T')
                pattern_value += 2;
            else
                pattern_value += 1;
            ++i;
        }
        total_keys = octaves * 12;
        int p = pattern_value;
        while(p < total_keys)
        {
            scales = scales + (total_keys - p);
            p += pattern_value;
        }
        printf("%lld\n",scales);
    }
    return 0;
}
