#include<stdio.h>
#include<string.h>
int main()
{
    int tc,n;
    int len,res,stp;;
    char str[200];
    scanf("%d",&tc);
    while(tc--)
    {
        scanf("%s",str);
        len=strlen(str);

        scanf("%d",&n);
        stp=0;
        for(int i=0;i<len;i++)
        {
            if(str[i]=='T')
                stp+=2;
            else
                stp+=1;
        }

       int i=1;
        res=0;
     for(int i=1;i*stp<n*12;++i)
        {
            res+=((n*12) - (i*stp));
         
        }


        printf("%d\n",res);
    }
    return 0;
}
