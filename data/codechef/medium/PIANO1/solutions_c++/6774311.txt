#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
		string s;
		int n,len=0;
		cin>>s;
		cin>>n;
		int ans=0;
		
		for(int i=0;i<s.length();i++){
			len+=s[i]-'R';
		}
		for(int p=len;p<12*n;p+=len){
            ans+=12*n-p;
        }
        cout<<ans<<endl; 
	}
	return 0;
}
