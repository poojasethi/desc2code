def test():
    #Lapindromes:
    matches = input()
    for _ in range(matches):
        string = raw_input()
        length = len(string)
        first = string[:length/2]
        if length%2 == 0:
            half = length/2
        else:
            half = length/2 + 1

        second = string[half:]
        dict1 = {}

        find = True
        
        if len(first) == len(second):
            for char in first:
                if char not in dict1.keys():
                    dict1[char] = 1
                else:
                    dict1[char] += 1

            for char2 in second:
                if char2 in dict1.keys():
                    dict1[char2] -= 1
                else:
                    dict1[char2] =-1
                    break

            for value in dict1.values():
                if value != 0:
                    find = False
                    break  
                   
        else:
            find = False

        
        if find:
            print 'YES'
        else:
            print 'NO'
            
        
        
        
  
test()
