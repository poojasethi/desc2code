t = int(raw_input())
for _ in xrange(t):
    S = raw_input().strip()
    n = len(S)/2
    a = S[:n]
    b = S[-n:]
    for i in a:
        if i in b:
            a = a[:a.index(i)] + a[a.index(i)+1:]
            b = b[:b.index(i)] + b[b.index(i)+1:]
    if a == b == '':
        print 'YES'
    else:
        print 'NO'
