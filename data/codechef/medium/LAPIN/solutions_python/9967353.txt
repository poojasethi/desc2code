t = int(raw_input())
for _ in xrange(t):
    S = raw_input().strip()
    n = len(S)/2
    a = S[:n]
    b = S[-n:]
    count = 0
    for i in a:
        if a.count(i) != b.count(i) and len(a) == len(b):
            print "NO"
            count = 1
            break
    if count == 0:
        print "YES"
