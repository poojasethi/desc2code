t = int(raw_input())
while t > 0 :
    t -= 1
    s = raw_input()
    left = {}
    right = {}
    z = len(s)
    for i in xrange(z/2) :
        try :
            left[s[i]] += 1
        except :
            left[s[i]] = 1
    f = None
    if z % 2 == 0 :
        f = z/2
    else :
        f = z/2 + 1
    for j in xrange(f,z) :
        try :
            right[s[j]] += 1
        except :
            right[s[j]] = 1
    if left == right :
        print 'YES'
    else :
        print 'NO'
