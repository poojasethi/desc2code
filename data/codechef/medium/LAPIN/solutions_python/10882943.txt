t = input();
while(t > 0):
	s = raw_input(); l = len(s); a = ""; b = "";
	if(l & 1 == 0):
		for i in range(0, l / 2):	a += s[i];
		for i in range(l / 2, l):	b += s[i];
		if(sorted(a) == sorted(b)):	print "YES";
		else:	print "NO";
	else:
		for i in range(0, l / 2):	a += s[i];
		for i in range(l / 2 + 1, l):	b += s[i];
		if(sorted(a) == sorted(b)):	print "YES";
		else:	print "NO";
	t -= 1;
