T = input()
while T:
    str = raw_input()
    str1 = str[:len(str)/2]
    str2 = str[(len(str) +1)/2 :]
    letters = []
    flag=0
    for i in str1:
        letters.append(i)
    for i in str2:
        if letters.count(i) > 0:
            letters.remove(i)
        else:
            flag=1
            break
    if flag == 1:
        print ("NO")
    else:
        print ("YES")
    T -= 1
