for _ in xrange(input()):
    string = raw_input()
    left_part_dict = {}
    right_part_dict = {}
    string_length = len(string)
    if string_length % 2 == 0:
        left_part = string[:string_length/2]
        right_part = string[string_length/2:]
    else:
        left_part = string[:string_length / 2]
        right_part = string[(string_length / 2) + 1:]

    # Populating the dictionary
    for i in range(string_length / 2):
        if left_part[i] not in left_part_dict:
            left_part_dict[left_part[i]] = 1
        else:
            left_part_dict[left_part[i]] += 1
        if right_part[i] not in right_part_dict:
            right_part_dict[right_part[i]] = 1
        else:
            right_part_dict[right_part[i]] += 1

    # Comparing dictionaries
    for i in left_part_dict:
        if i not in right_part_dict or left_part_dict[i] != right_part_dict[i]:
            print "NO"
            break
    else:
        print "YES"
