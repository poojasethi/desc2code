ntc = int(raw_input())
while ntc!=0:
	s = raw_input()
	s1,s2 = "",""
	sz = len(s)
	for i in xrange(len(s)):
		if i<(sz/2):
			s1+=s[i]
		elif not (sz%2==1 and i==sz/2):
			s2+=s[i]
	c=0
	s1 = list(s1)
	s2 = list(s2)
	for i in s1:
		if i in s2:
			v=s2.index(i)
			s2.pop(v)
			c+=1
	print "YES" if c==len(s1) else "NO"
	ntc-=1
