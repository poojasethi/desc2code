def lapindrome(a,b):
    l1 = [0]*26
    l2 = [0]*26
    
    for i in a:
        l1[ord(i)-97]+=1
        
    for i in b:
        l2[ord(i)-97]+=1
    if(l1==l2):
        print "YES"
    else:
        print "NO"
        
if __name__ == "__main__":
    t = input()
    while (t>0):
        s = raw_input()
        l = len(s)
        if(l%2!=0):
            s = s[:l/2]+s[(l/2)+1:]
        l = len(s)
        
        a = s[:l/2]
        b = s[(l/2):]
        lapindrome(a, b)
        t-=1
