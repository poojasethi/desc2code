t=input()
 
for i in range(0,t):
    count = 0
    word = raw_input()
    if len(word)%2 == 0:
        string1 = word[:(len(word)/2)]
        string2 = word[(len(word)/2):]
    else:
        string1 = word[:(len(word)/2)]
        string2 = word[(len(word)/2)+1:]
    for i in string1:
        if(string1.count(i) != string2.count(i)):
            print "NO"
            count = 1
            break
    if count == 0:
