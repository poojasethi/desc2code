# include<iostream>
using namespace std;
 
int main()
{
  int i,t,n=0,j,k,p;
  string s;
  cin>>t;
  for(i=1;i<=t;i++)
  {
    cin>>s;
    for(j=0;s[j]!='\0';j++)
      n++;
    if(n%2==0)
    {
      for(j=0;j<n/2;j++)
      {
        for(k=n/2;k<n;k++)
        {
          if(s[j]==s[k])
          {
            s[k]='\0';
            p=1;
            break;
          }
          else
          {
            p=0;
            
          }
        }
        if(p==0)
          break;
      }
    }
    else
    {
      for(j=0;j<n/2;j++)
      {
        for(k=n/2+1;k<n;k++)
        {
          if(s[j]==s[k])
          {
            s[k]='\0';
            p=1;
            break;
          }
          else
          {
            p=0;
            
          }
        }
        if(p==0)
          break;
      }
    }
    if(p==1)
      cout<<"YES"<<endl;
    else
      cout<<"NO"<<endl;
    n=0;
  }
  return 0;
}
