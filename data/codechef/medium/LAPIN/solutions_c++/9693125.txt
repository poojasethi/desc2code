#include<iostream>
#include<string>
#include<algorithm>
#include<cstdio>

using namespace std;

int main(){
int t,len;
string s1,s2,s3;
cin>>t;
while(t--){
	cin>>s1;
	len = s1.length();
	s2="";s3="";
	for(int i=0;i<len/2;i++)
		s2+= s1[i];
	 
	if(len%2==0){
	for(int i=len/2;i<len;i++)
		s3+=s1[i];}
	else{
	for(int i=len/2 + 1; i<len;i++)
		s3+= s1[i]; }
	sort(s2.begin(),s2.end());
	sort(s3.begin(),s3.end());
	
	if(s2.compare(s3) == 0)
		cout<<"YES\n";
	else
		cout<<"NO\n";}
	return 0;
}
