#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char a[1000];
		scanf("%s",a);
		int l=strlen(a),i;
		int c[26]={0},b[26]={0},count=0;
		if(l%2==0)
		{
			for(i=0;i<l/2;i++)
			{
				b[a[i]-'a']++;
			}
			for(i=l/2;i<l;i++)
			c[a[i]-'a']++;
			for(i=0;i<26;i++)
			{
				if(b[i]!=c[i])
				{
					count++;
					break;
				}
			}
			
		}
		else
		{
			for(i=0;i<l/2;i++)
			{
				b[a[i]-'a']++;
			}
			for(i=(l/2)+1;i<l;i++)
			c[a[i]-'a']++;
			for(i=0;i<26;i++)
			{
				if(b[i]!=c[i])
				{
					count++;
					break;
				}
			}
		}
		if(count==0)
			printf("YES\n");
			else
			printf("NO\n");
	}
	return 0;
}
