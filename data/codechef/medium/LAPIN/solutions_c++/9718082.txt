#include <iostream>
#include <string.h>

using namespace std;

int main()
{
    int t,l,i,a1[26],a2[26],mid=0,flag;
    char st[1000];
    cin>>t;
    while(t--)
    {
        flag=0;
        cin>>st;
        l=strlen(st);
            mid=l/2;
            for(i=0;i<26;i++)
            {
                a1[i]=0;
                a2[i]=0;
            }
            for(i=0;i<mid;i++)
                a1[st[i]-'a']++;
        if(l%2==0)
        {
            for(i=mid;i<l;i++)
                a2[st[i]-'a']++;
        }
        else
        {
            for(i=mid+1;i<l;i++)
                a2[st[i]-'a']++;
        }
            for(i=0;i<26;i++)
            {
                if(a1[i]==a2[i])
                   flag++;

            }
            if(flag==26)
                cout<<"YES\n";
            else
                cout<<"NO\n";

    }

    return 0;
}
