#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n;
    char a[1001];
    int flag,i;
    cin>>t;
      while(t--){
  	      cin>>a;;
            n=strlen(a);
            if(n%2==0){
    	       sort(a,a+n/2);
    	       sort(a+(n-n/2),a+n);
    	    }
			else{
	            sort(a,a+n/2);
	            sort(a+(n/2+1),a+n);
	        }
			 for(i=0;i<n/2;i++){
    		   if(a[i]==a[i+(n-n/2)])
    		      flag=1;
    		   else{
			      flag=0;
				  break;
			   }
		    }
	     if(flag)
	        cout<<"YES\n";
	     else
	        cout<<"NO\n";
	   }
	   return 0;
}
