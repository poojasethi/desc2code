#include<bits/stdc++.h>
using namespace std;
int main()
{
string s,a,b;
int c,n, l,i;
cin>>n;
for(i=0;i<n;i++)
{
	cin>>s;
	l=s.size();
     c=l/2;
     a=s.substr(0,c);
     b=s.substr(c+(l%2),l);
	sort(a.begin(),a.end());
	sort(b.begin(),b.end());
	if(a==b)
	cout<<"YES\n";
	else
	cout<<"NO\n";
}
	return 0;

}
