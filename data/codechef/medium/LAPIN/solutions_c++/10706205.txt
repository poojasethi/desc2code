#include<stdio.h>
#include<string.h>
typedef long long ll;
int main()
{
	ll t;
    scanf("%lld",&t);
    while(t--)
    {
    	char s[1001];
    	ll fl[123]={0};
    	ll fr[123]={0};
    	scanf("%s",s);
    ll i,flag,st,	l=strlen(s);
    	for(i=0;i<l/2;i++)
    	fl[s[i]]++;
    	if(l%2==0)
    	st=l/2;
    	else st=(l/2)+1;
    	for(i=st;i<l;i++)
    	fr[s[i]]++;
    	flag=0;
    	for(i=97;i<123;i++)
    	if(fl[i]==fr[i])
    	continue;
    	else
    	{flag=1;break;}
    	if(flag==1)
    	printf("NO\n");
    	else
    	printf("YES\n");
    	
    	
    	
    }
}
