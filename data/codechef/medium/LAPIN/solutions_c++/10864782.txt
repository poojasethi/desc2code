#include<cstdio>
#include<cstring>
using namespace std;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        char str[1005];
        scanf("%s",&str);
        int n=strlen(str);
        int e1,s2;
        if(n%2==0){
        e1=(n/2)-1,s2=e1+1;
        }else
        {
        e1=(n/2)-1,s2=e1+2;
        }   
        int c1[26]={0},c2[26]={0};;
            for(int i=0;i<=e1;i++){
                c1[str[i] - 'a']++;
            }
            for(int i=s2;i<=n-1;i++){
                c2[str[i] - 'a']++;
            }
            int fail=0;
            for(int i=0;i<26;i++){
                if(c1[i]!=c2[i]){
                    fail=1;
                    break;
                }
            }
            if(fail){
                printf("NO\n");
            }
            else
            {
                printf("YES\n");
            }
    }
}
