#include <iostream>
#include <string>
#include <string.h>

using namespace std;

int main(){
	int T;
	cin >> T;
	int count[26];
	for(int t=0;t<T;t++){
		string S;
		cin >> S;
		int size = S.size();
		memset(count, 0, sizeof(count));
		int l = 0;
		int r = size-1;
		while(l<r){
			count[S[l]-'a']++;
			count[S[r]-'a']--;
			l++;
			r--;
		}
		int flag=0;
		for(int i=0;i<26;i++){
			if(count[i]){
				flag=1;
				break;
			}
		}
		if(flag)
			cout << "NO" <<endl;
		else
			cout << "YES" << endl;
	}
	return 0;
}
