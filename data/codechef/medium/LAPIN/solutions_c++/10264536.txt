#include <iostream>

using namespace std;

#define ll long long;

int main(int argc, char const *argv[])
{
	int t;
	cin>>t;
	while(t--){
		int flag=0, l=0, alpha[26]={0};
		string str;
		cin>>str;
		for (int i=0;str[i]!='\0';i++)
			l++;
		if (l%2==0){
			for (int i=0;i<l;i++){
				if (i<l/2)
					alpha[(int)str[i]-'a']++;
				else alpha[(int)str[i]-'a']--;
			}
		}
		else if (l%2==1){
			for (int i=0;i<l;i++){
				if (i<l/2)
					alpha[(int)str[i]-'a']++;
				if (i>=(l+1)/2)
					alpha[(int)str[i]-'a']--;
			}
		}
		for (int i=0;i<26;i++){
			if (alpha[i]!=0){
				cout<<"NO"<<endl;
				flag=1;
				break;
			}
		}
		if (!flag)
			cout<<"YES"<<endl;
		}
	return 0;
}
