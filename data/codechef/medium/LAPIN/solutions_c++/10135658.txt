#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        char a[1001];
        int i;
        scanf("%s",a);
        int b[150]={0};
        int c[150]={0};
        int l=strlen(a);
        for(i=1;i<=l/2;i++)
        {
            b[a[i-1]]++;
            c[a[l-i]]++;
        }
        int fl=0;
        for(i=0;i<150;i++)
        {
            if(b[i]!=c[i])
            {
                fl=1;
                break;
            }
        }
        if(fl==0)
            cout<<"YES\n";
        else
            cout<<"NO\n";
    }
    return 0;
}
