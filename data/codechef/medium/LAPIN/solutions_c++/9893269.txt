#include <bits/stdc++.h>
using namespace std ;
 int  main ()
{
    int t;
    cin>>t;
    while (t--)
    {
        string name ;
        cin>>name;
        int mid ;
        int length = name.length();
        string substr1,substr2;
        if(length%2==0)
        {
            mid = (length /2);
            substr1 = name.substr(0,mid);
            substr2 = name.substr(mid,length-1);

        }
        else
        {
             mid = (length /2);
            substr1 = name.substr(0,mid);
            substr2 = name.substr(mid+1,length-1);

        }
        int c1 = 0 ;
        int c2 = 0 ;
        int flag = 0 ;
        for(int i = 0 ;i<=(length/2)-1;i++)
        {
            for(int j = 0 ;j<=(length/2)-1;j++)
        {
             if(substr1[i]==substr1[j])
            {
                c1++;
            }

            if(substr1[i]==substr2[j])
            {
                c2++;
            }

        }
        if(c1==c2)
        {

            c1 = 0;c2 = 0;
        }
        else
        {
            flag = 1;
        }
        }
        if(flag==0)
        {
            cout<<"YES"<<endl;
        }
        else
        {
            cout<<"NO"<<endl;
        }

    }
    return 0;
}
