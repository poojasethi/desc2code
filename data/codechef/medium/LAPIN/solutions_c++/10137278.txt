#include<stdio.h>
#include<string.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int i,z;
		char s[10000];
		int s1[10000],s2[10000];
		for(i=0;i<10000;i++)
			s[i]=s1[i]=s2[i]=0;
		scanf("%s",s);
		z = strlen(s);
		if(z%2 != 0)
		{
		for(i=0;i<z/2;i++)
			s1[s[i]-'a']++;
		for(i=(z/2+1);i<z;i++)
			s2[s[i]-'a']++;
		}
		else
		{
			for(i=0;i<z/2;i++)
				s1[s[i]-'a']++;
			for(i=z/2;i<z;i++)
				s2[s[i]-'a']++;
		}
		for(i=0;i<26;i++)
		{
			if(s1[i] != s2[i])
				break;
		}
		if(i==26)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
