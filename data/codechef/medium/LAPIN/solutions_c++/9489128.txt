#include<bits/stdc++.h>
using namespace std;
#define ll long long int

int main()
{
	ll t;
	scanf("%lld",&t);
	while(t--)
	{
		string s;
		cin>>s;
		ll m,n,i,j;
		n = s.length();
		(s.length()&1)?m=(n/2)+1:m=n/2;
		int a[10000]={0};
		int b[10000]={0};
		for(i=0,j=m;i<n/2,j<n;i++,j++)
		{
			a[s[i]]++;
			b[s[j]]++;
		}
		int flag=0;
		for(i=0,j=0;i<10000,j<10000;i++,j++)
		{
			if(a[i]!=b[i])
			{
				flag=1;
				break;
			}
		}
		(flag==0)?cout<<"YES\n":cout<<"NO\n";
	}
	return 0;
}
