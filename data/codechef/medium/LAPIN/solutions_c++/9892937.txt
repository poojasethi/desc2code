
#include<bits/stdc++.h>
using namespace std;
#define ll long long;



main()
 { int T,l,i;

   ios::sync_with_stdio(false);
   string S;
   int hash1[26],hash2[26],lol;
   cin>>T;
   while(T--)

      {cin>>S;
       lol=1;
       l=S.length();
       memset(hash1,0,sizeof(hash1));
       memset(hash2,0,sizeof(hash2));


       if(l%2==0)
       {
           for(i=0;i<l/2;++i)
             {hash1[(S[i]-'a')]+=1;

             }

            for(i=l/2;i<l;++i)
              hash2[(S[i]-'a')]+=1;

       }
       else
       {
           for(i=0;i<l/2;++i)
             {hash1[(S[i]-'a')]+=1;

             }

           for(i=(l/2)+1;i<l;++i)
             hash2[(S[i]-'a')]+=1;
       }
       for(i=0;i<26;++i)
       {

            if(  hash1[i]!=hash2[i])
           {
               lol=0;
               break;
           }

       }

       lol?cout<<"YES\n":cout<<"NO\n";
       S.clear();


      }
 }
