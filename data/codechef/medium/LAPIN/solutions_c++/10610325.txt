#include<iostream>
#include<string>
#include<cstring>

using namespace std;
int main(){
	unsigned int h, i, j, k, t, flag, lent, jmax;
	int n[500];
	char l[500];
	cin>>t;
	while (t--){
		string lapindrome;
		cin>>lapindrome;
		lent = lapindrome.length();
		memset(n,0,sizeof(n));
		jmax = 0;
		flag = 0;
		h = 0;
		for(i=0;i<lent/2;i++){
			for (h=0;h<jmax;h++){
				if (l[h]==lapindrome[i]){
					n[h]+=1;
					// cout<<"BR\n";
					break;
				}
			}
			if(h==jmax){
				l[jmax] = lapindrome[i];
				n[jmax] += 1;
				jmax += 1;
			}
			// else
			// 	break;
		}
		
		for(i=(lent+1)/2;i<lent;i++){
			for (k=0;k<=jmax;k++){
				if (l[k] == lapindrome[i]){
					n[k] -= 1;
					break;
				}
			}
		}
		for (k = 0;k<jmax;k++){
			if(n[k]!=0){
				// cout<<n[k];
				flag = 1;
				break;
			}
		}
		
		if (i!=lent || flag==1){
			cout<<"NO\n";
		}
		else
			cout<<"YES\n";
		
	}
	return 0;
}
