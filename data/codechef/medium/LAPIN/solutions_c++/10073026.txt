#include<bits/stdc++.h>
using namespace std;

int main()
{
    int i,T,Len,flag,s1[27],s2[27];
    char S[1001];
    scanf("%d",&T);
    getchar();
    while(T--)
    {
 flag = 1;
 for(i=0; i<27; i++)  s1[i]=s2[i]=0;
 gets(S);
 Len = strlen(S);
 for(i=0; i<Len/2; i++)
 {
     s1[S[i] - 'a']++;
     s2[S[Len-1-i] - 'a']++;
 }
 for(i=0; i<Len/2; i++)
   if(s1[S[i]-'a'] != s2[S[i] - 'a']) { flag=0; break; }
 if(flag) puts("YES\n");
 else puts("NO\n");
    }
}
