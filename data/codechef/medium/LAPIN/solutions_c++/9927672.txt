#include"bits/stdc++.h"
using namespace std;
int main()
{
	string s;
	int t,x,y;
	cin>>t;
	while(t--)
	{
		int flag=0;
		cin>>s;
		int l=s.length();
		int a[26]={0};
		int b[26]={0};
		if(l%2==0)
		{
			x=l/2-1;
			y=l/2;
		}
		else
		{
			x=l/2-1;
			y=l/2+1;
		}
		for(int i=0;i<l;i++)
		{
			if(i<=x)
			{
				a[s[i]-97]++;
			}
			else
			{
				if(i>=y)
				{
					b[s[i]-97]++;
				}
			}
		}
		for(int i=0;i<26;i++)
		{
			if(a[i]!=b[i])
			{
				flag=1;
				break;
			}
		}
		if(flag==0)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;	
	}
	
}
