#include<iostream>
#include<cstring>
#define fre freopen("in.txt","r",stdin);
#define fr(i,st,en) for(int (i)=st;i<en;i++)
using namespace std;
int main()
{	//fre
	int t;
	cin>>t;
	while(t--)
	{
		string a,b="",c="";
		cin>>a;
		int sz=a.size();
		int i=0;
		int lcnt[26]={0},rcnt[26]={0};
		if(sz%2==0)
		{	for(i=0;i<sz/2;i++)
			{
				b+=a[i];lcnt[a[i]-'a']++;	
			}
			for(int j=i;j<sz;j++)
			{c+=a[j];rcnt[a[j]-'a']++;}
			
			//cout<<b<<" "<<c<<"\n";
		}
		else
		{
			for(i=0;i<sz/2;i++)
			{b+=a[i];lcnt[a[i]-'a']++;}
			for(int j=i+1;j<sz;j++)
			{c+=a[j];rcnt[a[j]-'a']++;}
			//cout<<b<<" "<<c<<"\n";
		}
		bool f=true;
		fr(i,0,26)
		{
			if(lcnt[i]!=rcnt[i])
			{	//cout<<i;
				f=false;
			}
		}
		if(f)
		cout<<"YES\n";
		else cout<<"NO\n";
	}
}
