#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,flag,i,j,n;
	char s[1001];
	cin>>t;
	while(t--)
	{
		cin>>s;
		n = strlen(s);
		flag = 0;
		i=0;
		j=(n+1)/2;
		int ctr1[26] = {0};
		int ctr2[26] = {0};
		while(i<n/2 && j<n)
		{
			ctr1[s[i++]-97]++;
			ctr2[s[j++]-97]++;
		}
		for(i=0; i<26; i++)
		{
			if(ctr1[i]!=ctr2[i])
			{
				flag = 1;
				break;
			}
		}
		if(flag==1)	
			cout<<"NO";
		else
			cout<<"YES";
		cout<<endl;
	}
	return 0;
}
