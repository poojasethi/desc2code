#include<iostream>
#include<string.h>
using namespace std;
int main(){

    int t,n,i,k;
    scanf("%d",&t);
    char s[1010];
    while(t--){
        cin>>s;
        k=strlen(s);
        int a[26]={0};
        for(i=0;i<k/2;++i){
            a[s[i]-'a']++;
        }

        for(i=(k+1)/2;i<k;++i){
            a[s[i]-'a']--;
        }
        int flag=0;
        for(i=0;i<26;++i){
            if(a[i]!=0) {flag=1;break;}
        }

        if(flag==1) cout<<"NO\n";
        else cout<<"YES\n";
    }
}
