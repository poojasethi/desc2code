#include <iostream>
#include<string.h>
using namespace std;

int main() {
	int t,i,j,len,k; 
	char s[1001];
	cin>>t;
	while(t--)
	{
		int a[257]={0};
		cin>>s;
		len=strlen(s);
		k=len%2==0?len/2:((len/2)+1);
		for(i=0;i<len/2;i++)
		a[s[i]]++;
		for(j=k;j<strlen(s);j++)//strlen fun for finding length of char array
		a[s[j]]--;
		for(i=0;i<=256;i++)
		{
			if(a[i]!=0)
			{
			cout<<"NO"<<endl;
			break;
			}
		}
		if(i==257)
		cout<<"YES"<<endl;
	}
	return 0;
}
