#include<iostream>
#include<string.h>
using namespace std;

int main()
{
	int t;
	char s[1001];
	cin>>t;
	while(t--)
	{
	  cin>>s;
	  int l=strlen(s);
	  l--;
	  int i=0,f=0;
	  int a[26]={0};
	  while(i<=l)
	  {
	  	a[s[i]-97]+=1;
	  	a[s[l]-97]-=1;
	  	i++;
	  	l--;
	  }	
	  i=0;
	  while(i<26)
	  {
	  	if(a[i]>0)
	  	   {
	  	   	f=1;
	  	   	break;
	  	   }
	  	   i++;
	  }
	  if(f==0)
	    cout<<"YES"<<endl;
	else
	  cout<<"NO"<<endl;
	}
	return 0;
}
