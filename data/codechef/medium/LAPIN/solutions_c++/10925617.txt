#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,x;
    string s;
    cin>>t;
    while(t--)
    {
        int z=0;
        cin>>s;
        x=s.length();
        int a[26],b[26];
        for(i=0;i<26;i++)
        {
            a[i]=0;
            b[i]=0;
        }
        for(i=0;i<x;i++)
        {
            if(i<(x/2))
            a[s[i]-'a']++;
            else if(x%2==0||i>(x/2))
            b[s[i]-'a']++;
        }
        for(i=0;i<26;i++)
        {
            if(a[i]!=b[i])
            z++;
        }
        if(z==0)
        cout<<"YES"<<endl;
        else cout<<"NO"<<endl;
    }
    return 0;
}
