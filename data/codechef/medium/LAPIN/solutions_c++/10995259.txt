#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--) {
	   char s[1000];
	   int c1[26] = {0};
	   int c2[26] = {0};
	   cin>>s;
	   int sl = strlen(s);
	   for(int i =0, j = sl-1; i < sl/2 ; i++,j--){
	   	 c1[s[i]-'a']++;
	   	 c2[s[j]-'a']++;
	   }
	   int flag = 0;
	   for(int i =0; i < (sl+1)/2; i++){
	         if(c1[s[i]-'a'] != c2[s[i]-'a']){
	         	flag = 1;      
	         }
	   }
 	   if(!flag){
 	   	cout<<"YES\n";
 	   }else {
 	   	cout<<"NO\n";
 	   }
	}
	
}
