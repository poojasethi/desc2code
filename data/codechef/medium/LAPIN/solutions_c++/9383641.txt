#include<bits/stdc++.h>
using namespace std;
#define fs(x) scanf("%f", &x);
#define ds(x) scanf("%lf", &x);
#define is(x) scanf("%d", &x);
#define ls(x) scanf("%ld", &x);
#define lls(x) scanf("%lld", &x);
#define fp(x) printf("%f", x);
#define dp(x) printf("%lf", x);
#define ip(x) printf("%d", x);
#define lp(x) printf("%ld", x);
#define llp(x) printf("%lld", x);
int main()
{
    int i,len,test;
    string str;
    is(test);
    while(test--)
    {
        int a[26]={0},b[26]={0};
        cin>>str;
        len=str.size();
        for(i=0;i<len/2;i++)
        {
            a[str[i]-97]++;
            b[str[len-(i+1)]-97]++;
        }
        int flag=0;
        for(i=0;i<26;i++)
        {
            if(a[i]!=b[i])
            {
                flag=1;
                break;
            }
            else
            continue;
        }
        if(flag==1)
        cout<<"NO"<<endl;
        else
        cout<<"YES"<<endl;
    }
    return 0;
}
