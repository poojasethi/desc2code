#include<iostream>
#include <algorithm>
#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
#define size 1001
#define hashsize 256
int check(int* hash)
{
	int i;
	for(i=97;i<=122;i++)
	if(hash[i]%2==1)
	return 0;
	return 1;
}
void GetHash(char* str,int* hash)
{
	 int i;
   for (i = 0; str[i];  i++)
      hash[str[i]]++;
}
int main()
{
	
	char str[size];
	int t,hash[hashsize],len,i;
	cin>>t;
	while(t--)
	{
		for(i=0;i<hashsize;i++)
		hash[i]=0;
		scanf(" %[^\n]s",str);
		len = strlen(str);	
			if(len%2==0)
			{
				GetHash(str,hash);
			}
			if(len%2==1)
			{
				str[len/2]=' ';
				GetHash(str,hash);
			}
			if(check(hash))
			cout<<"YES"<<"\n";
			else
			cout<<"NO"<<"\n";	
	}
	return 0;
}
