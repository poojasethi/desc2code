#include <bits/stdc++.h>
using namespace std;
long long int dpminl[10005]={0},dpminr[10005]={0},dpmaxr[10005]={0},dpmaxl[10005]={0},min_num,max_num,sum;
int arr[10005]={0};
int main()
{
	int tc,n,i;
	scanf("%d",&tc);
	while(tc--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)
			scanf("%d",&arr[i]);

		min_num=1000000000;
		sum=0;
		for(i=0;i<n;i++)
		{
			sum = sum + arr[i];
			//printf("%lld_",min_num);
			if(sum<min_num)
				min_num = sum; 
			//printf("%d_%lld_%lld\n",arr[i],min_num,sum);
			dpminl[i]=min_num;
			if(sum>0)
				sum=0;
		}
		
		max_num=-1000000000;
		sum=0;
		for(i=0;i<n;i++)
		{
			sum+=arr[i];
			max_num = max(max_num,sum); 
			dpmaxl[i]=max_num;
			if(sum<0)
				sum=0;
		}

		min_num=1000000000;
		sum=0;
		for(i=n-1;i>=0;i--)
		{
			sum+=arr[i];
			min_num = min(min_num,sum); 
			dpminr[i]=min_num;
			if(sum>0)
				sum=0;
		}
		
		max_num=-1000000000;
		sum=0;
		for(i=n-1;i>=0;i--)
		{
			sum+=arr[i];
			max_num = max(max_num,sum); 
			dpmaxr[i]=max_num;
			if(sum<0)
				sum=0;
		}
		
		max_num = -1000000000;
		for(i=0;i<n-1;i++)
		{
			min_num = dpmaxr[i+1]-dpminl[i];
			if(min_num<0)
				min_num = -min_num;
			if(min_num>max_num)
				max_num = min_num;
			//printf("%lld_",min_num);
			min_num = dpminr[i+1]-dpmaxl[i];
			if(min_num<0)
				min_num = -min_num;
			if(min_num>max_num)
				max_num = min_num;
			//printf("%lld\n",min_num);
		}
		printf("%lld\n",max_num);

	}
	return 0;
}
