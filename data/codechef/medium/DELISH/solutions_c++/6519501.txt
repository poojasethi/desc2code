const int M=10005;
i64 t,n,a[M],dp[4][M];

void solve1(int idx){
	i64 cur=0;
	dp[idx][0]=-inf;	
	for(int i=1;i<=n;i++){
		cur=max(a[i],cur+a[i]);
		dp[idx][i]=cur;
		dp[idx][i]=max(dp[idx][i],dp[idx][i-1]);
		
		//trace3(idx,dp[idx][i],a[i]);
	}
	//cout<<endl;
}

void solve2(int idx){
	i64 cur=0;
	dp[idx][n+1]=-inf;	
	for(int i=n;i>=1;i--){
		cur=max(a[i],cur+a[i]);
		dp[idx][i]=cur;
		dp[idx][i]=max(dp[idx][i],dp[idx][i+1]);
		//trace3(idx,dp[idx][i],a[i]);
	}
	//cout<<endl;
}
