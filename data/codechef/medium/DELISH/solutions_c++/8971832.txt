#include<bits/stdc++.h>
using namespace std;

long long int maxf[10000],maxb[10000],minf[10000],minb[10000],a[10000];

long long int max(long int a,long long int b){
	if(a >= b)
		return a;
	return b;
}
long long int min(long int a,long long int b){
	if(a <= b)
		return a;
	return b;
}
long long int min(long long int a){
	if(a < 0)
		return (-1)*a;
	return a;
}
int main(){
	ios::sync_with_stdio(0);
	
	int tc,i,n;
	long long int sum,ans;
	
	cin>>tc;
	while(tc--){
		cin>>n;
		for(i=0;i<n;i++)
			cin>>a[i];
		//setting maxf
		sum = max(0,a[0]);
		maxf[0] = a[0];
		for(i=1;i<n;i++){
			maxf[i] = max(sum+a[i],maxf[i-1]);
			sum = max(0,sum+a[i]);				
		}
		/*for(i=0;i<n;i++)
			cout<<maxf[i]<<" ";
		cout<<endl;*/
		//setting maxb
		sum = max(0,a[n-1]);
		maxb[n-1] = a[n-1];
		for(i=n-2;i>=0;i--){
			maxb[i] = max(sum+a[i],maxb[i+1]);
			sum = max(0,sum+a[i]);
		}
		/*for(i=0;i<n;i++)
			cout<<maxb[i]<<" ";
		cout<<endl;*/
		//setting minf
		sum = min(0,a[0]);
		minf[0] = a[0];
		for(i=1;i<n;i++){
			minf[i] = min(sum+a[i],minf[i-1]);
			sum = min(0,sum+a[i]);
		}
		/*for(i=0;i<n;i++)
			cout<<minf[i]<<" ";
		cout<<endl;*/
		//setting minb
		sum = min(0,a[n-1]);
		minb[n-1] = a[n-1];
		for(i=n-2;i>=0;i--){
			minb[i] = min(sum+a[i],minb[i+1]);
			sum = min(0,sum+a[i]);
		}
		/*for(i=0;i<n;i++)
			cout<<minb[i]<<" ";
		cout<<endl;*/
		ans = INT_MIN;
		for(i=0;i<n-1;i++){
			ans=  max(ans,abs(maxf[i]-minb[i+1]));
			ans = max(ans,abs(minf[i]-maxb[i+1]));
		}
		cout<<ans<<"\n";
	}	
	return 0;
}
