#include <bits/stdc++.h>

#define For(i, be, en) for(i = be; i <= en; i++)
#define Forr(i, be, en) for(int i = be; i >= en; i--)

using namespace std;

int main()
{
    int t, n, l, r, lm, rm;

    long long int a[10005];
    long long int lmin[10005];
    long long int lmax[10005];
    long long int rmin[10005];
    long long int rmax[10005];

    cin >> t;

    while (t--) {
        cin >> n;

        for (int i = 0; i < n; i++) {
            cin >> a[i];
        }

        lmin[0] = lmax[0] = a[0];
        for (int i = 1; i < n; i++) {
            lmin[i] = min(a[i], lmin[i-1] + a[i]);
            lmax[i] = max(a[i], lmax[i-1] + a[i]);
        }

        rmin[n-1] = rmax[n-1] = a[n-1];
        for (int i = n-2; i >= 0; i--) {
            rmin[i] = min(a[i], rmin[i+1] + a[i]);
            rmax[i] = max(a[i], rmax[i+1] + a[i]);
        }

        long long ans = 0;
        for (int i = 0; i < n-1; i++) {
            ans = max(ans, abs(lmin[i] - rmax[i+1]));
            ans = max(ans, abs(lmax[i] - rmin[i+1]));
        }

        cout << ans <<endl;
    }


	return 0;
}
