#include<bits/stdc++.h>
#define lli long long int
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define pd(x) printf("%d",x)
#define plld(x) printf("%lld",x)
 
using namespace std;
 
lli dsum(lli n)
{
	lli sum=0;
	lli i=2;
	while(i <= sqrt(n))
    {
        if(n%i==0) {
            sum+=i;
            if (i != (n / i)) 
            {
                sum += n/i;
            }
        } 
        i++;
    }
	return sum+n+1;
}
 
int main()
{
	int t;
	sd(t);
	while(t--)
	{
		lli n;
		slld(n);
		if(n==1)
		puts("1");
		else
		cout<<dsum(n)<<endl;
	}
}
