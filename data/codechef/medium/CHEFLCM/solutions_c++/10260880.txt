#include <iostream>
#include<stdio.h>
#include<math.h>
using namespace std;

int main() {
	int ts;
	long long n,ans,i;
	scanf("%d",&ts);
	while(ts--){
		scanf("%lld",&n);
		ans=0;
		if(n==1)
		printf("1\n");
		else{
			
		for(i=1;i<=sqrt(n);i++){
			if(n%i==0){
			ans=ans+i;
			if(i!=sqrt(n))
            ans+=n/i;
			}
			
		   
		}
		printf("%lld\n",ans);
		}
	}
	return 0;
}
