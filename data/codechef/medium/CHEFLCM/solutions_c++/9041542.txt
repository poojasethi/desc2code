#include <iostream>
#include<cmath>
using namespace std;

int main() {
	long long int t,n;
	cin>>t;
	while(t--)
	{
	 cin>>n;
	 if(n==1)
	 cout<<"1\n";
	 else
	{ long long int sum=n+1;
	 for(int i=2;i<=sqrt(n);i++)
	 {if(n%i==0)
	 {sum = sum+i+(n/i);
	  if(i==sqrt(n))
	  sum=sum-i;
	 }
	 
	 }
	 cout<<sum<<"\n";
	}
	}
}
