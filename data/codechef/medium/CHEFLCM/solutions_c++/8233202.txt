#include<iostream>
#include<math.h>
using namespace std;
int main()
{
 int tc;
 unsigned long long n,sum,i;
 cin>>tc;
 while(tc--)
 {
  cin>>n;
  sum=0;
  if(n==1)
  cout<<1<<endl;
  if(n==2)
  cout<<3<<endl;
  for(i=2;i<=sqrt(n);i++)
  {
   if(n%i==0)
   {
   	sum+=i;
   if(i!=(n/i))
   sum+=(n/i);
  }
}
  if(n!=1 && n!=2)
  cout<<sum+1+n<<endl;
 }
 return 0;
}
