#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,i,sum=0;
		scanf("%lld",&n);
		if(n==1)
		printf("1\n");
		else if(n==2)
		printf("3\n");
		else
		{
			
			for(i=2;i<=sqrt(n);i++)
			{
				if((n%i)==0)
				{
					sum+=i;
					if(n/i!=i||n/i==n)
					sum+=n/i;
				}
			}
			sum+=1+n;
			printf("%lld\n",sum);
		}
	}
	return 0;
}
