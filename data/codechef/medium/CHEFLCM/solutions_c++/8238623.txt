#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	long int t,i;
	long long int n,sum;
	cin>>t;
	while(t--)
	{
		sum=0;
		cin>>n;
		for(i=2;i<=sqrt(n);i++)
		{
			if(n%i==0)
			{
				sum=sum+i;
				if(n/i!=i)
				sum=sum+n/i;
			}
		}
		if(n!=1)
		sum=sum+1+n;
		else
		sum=sum+1;
		cout<<sum<<endl;
	}
	return 0;
}
