#include <bits/stdc++.h>
using namespace std;
int main()
{
    int t,i,j;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        double sqroot=sqrt(n);
        long long ans=0;
        for(i=1;i<=sqroot;i++)
        {
            if(n%i==0)
            {
                ans+=i;
                ans+=n/i;
            }
        }
        if(sqroot-(int)sqroot==0)
        {
            ans-=(int)sqroot;
        }
        printf("%lld\n",ans);
    }
}
