#include <iostream>
#include <cstdio>
#define  ll long long
using namespace std;
int main()
{
// #ifndef AKHAND_CHUTIYAPA
// 	freopen("../input.txt", "r", stdin);
// 	//freopen("../output.txt","w",stdout);
// #endif
	int t;
	ll n;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%lld", &n);
		ll s = 0;
		for (int j = 1; j * j <= n; j++)
		{
			if (j * j != n && n % j == 0)
				s += (n / j) + j;
			else if (n % j == 0 && j * j == n)
				s += (n / j);
		}
		printf("%lld\n", s);
	}
	return 0;
}
