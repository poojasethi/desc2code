#include<stdio.h>
int main()
{
	long long int t, i, n, count;
	scanf("%lld", &t);
	while (t--)
	{
		count = 0;
		scanf("%lld", &n);
		for (i = 1; i*i <= n; i++)
		{
			if (n%i == 0)
			{
				count += i;
				if ((n / i) != i)
					count += n / i;
			}
		}
		printf("%lld\n", count);
	}
	getchar();
	getchar();
	return 0;

}
