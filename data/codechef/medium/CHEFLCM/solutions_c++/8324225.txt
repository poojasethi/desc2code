	#include<iostream>
	#include<math.h>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			long long int n;
			cin>>n;
			long long int sum=0;
			for(int i=1;i*i<=n;i++)
			{
				if(n%i==0)
					sum=sum+i+n/i;

			}	
			int temp=sqrt(n);
			if(temp*temp==n)
				sum=sum-temp;
			cout<<sum<<endl;
			t--;

		}	
	}
