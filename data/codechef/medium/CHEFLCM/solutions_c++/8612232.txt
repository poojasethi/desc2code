#include <bits/stdc++.h>
using namespace std;
#define ll long long 
int main()
{
    ll t;
    cin>>t;
    for(ll i = 1; i<= t; i++)
    {
        ll n, p;
        long long sum = 0;
        cin >> n;
        for(p = 1; p * p <= n; p++)
        {
            if(n % p == 0){
                sum += p;
                if(p != n / p){
                    sum += n / p;
                }
            }
        }
        cout << sum << '\n';
    }
    return 0;
}
