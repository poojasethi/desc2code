#include <iostream>
using namespace std;

int main() {
	int t;
	cin>>t;
	while(t--){
		long int n;
		cin>>n;
		if(n==1){
			cout<<"1\n";
			continue;
		}
		long long int sum=n+1;
		for(int i=2;i*i<=n;i++){
			if(n%i==0){
				sum+=i;
				sum+=(n/i);
			}
			if(i*i==n){
				sum-=i;
			}
		}
		cout<<sum<<endl;
	}
	return 0;
}
