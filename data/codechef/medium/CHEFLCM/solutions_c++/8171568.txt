#include <iostream>

using namespace std;

int main()
{
    long int t,n,i;
    long long int sum;
    cin>>t;
    while(t--)
    {
        sum=0;
        cin>>n;
        if(n==1)
            sum=1;
        else if(n==2)
            sum=3;
        else if(n>2)
        {
            if(n>=2000000)
        {
        for(i=1;i<(n/20000);i++)
        {
            if(n%i==0)
                sum=sum+i;
        }
        sum=sum+n;
        for(i=2;i<=20000;i++)
        if(n%i==0)
        {
        sum=sum+n/i;
        }
       }
        if(n<2000000)
        {
        for(i=1;i<=(n/2);i++)
        {
            if(n%i==0)
                sum=sum+i;
        }
        sum=sum+n;
        }
        }
        cout<<sum<<endl;
    }
    return 0;
}
