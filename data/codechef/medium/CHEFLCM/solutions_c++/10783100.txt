#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int main()
{
    int t,i;
    long long n,temp,ans;
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        scanf("%lld",&n);
        for(i=1;i<=sqrt(n);i++)
        {
            temp=n/i;
            if(temp*i==n)
            {
                if(i*i!=n)
                    ans+=(i+(n/i));
                else
                    ans+=i;
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
