#include <iostream>
using namespace std;

int main(){
	long long int test,n,g[100000],mini,diff,maxdiff,i,j,k;
	cin >> test;
	while(test>0){
		test--;
		cin >> n;
		mini = 100000000;
		maxdiff = 0;
		for(i=0;i<n;i++){
			cin >> g[i];
			diff = g[i]-mini;
			if(diff > maxdiff)
				maxdiff = diff;
			if(g[i] < mini)
				mini = g[i]; 
		}
		if(maxdiff != 0)
			cout << maxdiff << endl;
		else
			cout << "UNFIT" << endl;

	}
}
