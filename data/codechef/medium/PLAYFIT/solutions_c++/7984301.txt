#include<stdio.h>
int main()
{
	long long int t,n,g[100005],i,count,h[100005],max,min;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		for(i=0;i<n;i++)
			scanf("%lld",&g[i]);
		min=h[0]=g[0];
		for(i=1;i<n;i++)
		{
			if(min>g[i])
				min=g[i];
			h[i]=min;
		}
		max=0;
		for(i=1;i<n;i++)
		{
			if(g[i]-h[i]>max)
			{
				max=g[i]-h[i];
			}
		}
		if(!max)
			printf("UNFIT\n");
		else
			printf("%lld\n",max);
	}
	return 0;
}
