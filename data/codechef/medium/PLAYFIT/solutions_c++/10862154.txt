#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll t;
    scanf("%lld",&t);
    while(t--)
    {
        ll n;
        scanf("%lld",&n);
        ll ar[n];

        for(ll i=0;i<n;i++)
            scanf("%lld",&ar[i]);
      ll  m=ar[0];
      ll  ans=0;
        for(ll i=1;i<n;i++)
        {
            if(ar[i]<m)
            m=ar[i];
            else if(ans<(ar[i]-m))
                ans=(ar[i]-m);
        }
        if(ans==0)
        printf("UNFIT\n");
        else
            printf("%lld\n",ans);



    }

}
