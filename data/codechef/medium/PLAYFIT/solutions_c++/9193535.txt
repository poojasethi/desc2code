#include <iostream>
using namespace std;
int main()
{
        int t;
        cin>>t;
        while(t--)
        {
            long long n,k;
            cin>>n;
            int diff=0,min=1000000;
            for(int i=0;i<n;i++)
            {
                cin>>k;
                if(k<min)
                min=k;
                if(diff<(k-min))
                diff=k-min;
            }
            if(diff>0)
            cout<<diff<<endl;
            else
            cout<<"UNFIT"<<endl;

        }
                    return 0;
}
