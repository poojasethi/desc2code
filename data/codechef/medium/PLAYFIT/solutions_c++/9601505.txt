#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int a[n+2],g=0,min;
		for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
		min=a[0];
		
		for(int i=1;i<n;i++)
		{
			if(a[i]<min)
			min=a[i];
			if(a[i]-min>g)
			g=a[i]-min;
		}
		if(g==0)
		printf("UNFIT\n");
		else
		printf("%d\n",g);
	}
	return 0;
}
