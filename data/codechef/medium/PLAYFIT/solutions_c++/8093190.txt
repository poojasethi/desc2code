#include <stdio.h>
#include <stdlib.h>

int main()
{
	int t,n, max1, max2,i;
	int *G,*H, *I;
	scanf("%d", &t);
	
	while(t--)
	{
		max1 =0;
		scanf("%d", &n);
		G = (int*)malloc(n*sizeof(int));
		H = (int*)malloc(n*sizeof(int));
		I = (int*)malloc(n*sizeof(int));
		for(i=0; i<n; i++)
		{
			scanf("%d", &G[i]);
		}
		
		H[0] = G[0];
		
		for(i=1; i<n; i++)
		{
			if(G[i] < H[i-1])
				 H[i] = G[i];
			else 
				H[i] = H[i-1];
		}
		I[n-1] = G[n-1];
		for(i=n-1; i>0 ; i--)
		{
			if(G[i-1] > I[i])
				I[i-1] = G[i-1];
			else
				I[i-1] = I[i];
		}
		for(i=1; i<n; i++)
			G[i] = I[i] - H[i];
		max1 = G[1];
		for(i=2; i<n; i++)
		{
			if(G[i] > max1)
			max1 = G[i];
		} 
		if(max1>0)
			printf("%d\n", max1);
		else
			printf("UNFIT\n");
			
		free(G);
		free(H);
		free(I);
	}
return 0;
}
