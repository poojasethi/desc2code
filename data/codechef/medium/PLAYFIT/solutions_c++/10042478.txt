#include <iostream>
#include <stdio.h>
using namespace std;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int i,n,min,arr[100000],tmp=0;
        scanf("%d",&n);
        for(i=0;i<n;i++){
            scanf("%d",&arr[i]);
        }
        min=arr[0];
        for(i=0;i<n-1;i++){
            if((arr[i+1] > min) && (tmp < arr[i+1]-min))
                tmp=arr[i+1]-min;
            if(arr[i+1] < min)
                min=arr[i+1];
        }
        if(tmp==0)
            printf("UNFIT \n");
        else
            printf("%d \n",tmp);

    }
}
