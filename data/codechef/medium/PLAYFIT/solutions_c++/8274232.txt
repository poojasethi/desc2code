#include <iostream>
#include<limits.h>
using namespace std;

int main() {
	// your code goes here..
	int t;
	long long arr[100001];
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		for(int i=0;i<n;i++) cin>>arr[i];
		int ans=0;
		int maxi=arr[n-1];
		for(int i=n-2;i>=0;i--)
		{
			if(arr[i]>maxi) maxi=arr[i];
		    if(maxi-arr[i]>ans) ans=maxi-arr[i];
		}
		if(ans==0) cout<<"UNFIT"<<endl;
		else cout<<ans<<endl;
	}
	return 0;
}
