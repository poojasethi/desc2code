#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin>>t;
	long int arr[100005];
	long int x[100005];
	long int nn[100005];
	while(t--){
		int n;
		long int max = 0;
		
		scanf("%d",&n);
		for(int i=0;i<n;i++){
			scanf("%ld",&arr[i]);
			if(i>0){
				if((arr[i]-arr[i-1])>max){
					max = arr[i] - arr[i-1];
				}
			}
		}
		long int minval = arr[0];
		long int maxval = arr[n-1];
		for(int i=n-1,j=0;i>=0&j<n;i--,j++){
			if(arr[j]<minval){
				minval = arr[j];
			}
			if(arr[i]>maxval){
				maxval = arr[i];
			}
			x[i] = maxval;
			nn[j] = minval;
		}
		if(max==0){
			printf("UNFIT\n");
		}
		else{
			long int diff=0;
			for(int i=0;i<n;i++){
		//	cout<<x[i]<<" "<<nn[i]<<"    ";
				if(diff<(x[i]-nn[i])){
					diff = x[i] - nn[i];
				}
			}
			printf("%ld\n",diff);
		}
	}
	return 0;
}
