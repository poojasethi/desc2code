#include<iostream>
#include<algorithm>
#define MAX(a, b) (a)>(b)?(a):(b)
using namespace std;


int main()
{
    int T;
    cin >> T;
    while(T-->0)
    {
        int N, a, b;
        scanf("%d", &N);
        int ans=0;
        scanf("%d", &a);
        for(int i=1;i<N;i++)
        {
            scanf("%d", &b);
            if(b>a)
            {
                ans=MAX(b-a, ans);
            }
            else
                a = b;
        }
        ans>0?printf("%d\n", ans):printf("UNFIT\n");
    }
}
