#include <iostream>
#include<limits.h>
#include<cstdio>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
    long long int n,min=1000000007,ans=-1;
    int gi;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&gi);
        if(ans<gi-min)
            ans=gi-min;
        if(gi<min)
            min=gi;
    }
    if(ans>0)
    cout<<ans<<"\n";
    else
    cout<<"UNFIT\n";
    }
    return 0;
}
