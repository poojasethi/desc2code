	#include<iostream>
	#include<stdio.h>
	using namespace std;
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			int n;
			cin>>n;
			int a[100005];
			for(int i=0;i<n;i++)
			{
				scanf("%d",&a[i]);
			}	
			int max=0;
			int min=a[0];
			for(int i=1;i<n;i++)
			{
				if(a[i]-min>max)
					max=a[i]-min;
				if(min>a[i])
					min=a[i];

			}	
			if(max==0)
				cout<<"UNFIT\n";
			else
				cout<<max<<endl;
			t--;

		}		
	}
