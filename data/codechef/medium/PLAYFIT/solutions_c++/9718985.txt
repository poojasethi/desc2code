#include <iostream>
using namespace std;

int main() {
	int n,t,i,a[100009],min,max,temp,ans;
	scanf("%d",&t);
	while(t--)
	{
		min=0;
		max=0;
		temp=0;
		ans=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		min=a[0];
		max=a[0];
		for(i=1;i<n;i++)
		{
			if(a[i]>max)
			{
				max=a[i];
				temp=max-min;
			}
			else if(a[i]<min)
			{
				temp=max-min;
				if(temp>ans)
				{
					ans=temp;
				}
				min=a[i];
				max=a[i];
			}
		}
		if(temp>ans)
		{
			ans=temp;
		}
		if(ans==0)
			printf("UNFIT\n");
		else
			printf("%d\n",ans);
	}
	return 0;
}
