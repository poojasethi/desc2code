#include<iostream>
#include<math.h>
using namespace std;

int main()
{
    long long t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long long a[n],min,max,mi[n],ma[n];
        for(int i=0;i<n;i++)
            cin>>a[i];
        min=a[0];
        max=a[n-1];
        mi[0]=0;
        ma[n-1]=0;
        for(int i=1;i<n;i++)
        {
            if(a[i]>min)
                mi[i]=a[i]-min;
            else
                mi[i]=0;
            if(a[i]<min)
                min=a[i];
        }
        for(int i=n-2;i>=0;i--)
        {
            if(a[i]<max)
                ma[i]=max-a[i];
            else
                ma[i]=0;
            if(a[i]>max)
                max=a[i];
        }
        int count=0;
        for(int i=0;i<n;i++)
        {
            if(mi[i]>ma[i]&&count<mi[i])
                count=mi[i];
            else if(ma[i]>mi[i]&&count<ma[i])
                count=ma[i];
        }
        if(count==0)
            cout<<"UNFIT"<<endl;
        else
            cout<<count<<endl;
    }
}
