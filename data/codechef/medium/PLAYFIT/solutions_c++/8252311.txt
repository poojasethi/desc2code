# include <iostream>
# include <stdio.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i;
        long a,b,ans=0;
        scanf("%d",&n);
        scanf("%ld",&a);
        for(i=1;i<n;i++)
        {
            scanf("%ld",&b);
            if(b-a>ans)
            ans=b-a;
            else if(a>=b)
            a=b;
        }
        if(ans!=0)
        printf("%ld\n",ans);
        else
        printf("UNFIT\n");
    }
    return 0;
}
