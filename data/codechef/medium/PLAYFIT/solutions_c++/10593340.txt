#include <iostream>
#include <cstdio>
#include<limits.h>
#include <algorithm>
#define gc getchar_unlocked
#define pc putchar_unlocked
using namespace std;
inline unsigned long int fastread()
    {
      unsigned long int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
int main()
{
    int t;
    t=fastread();
    //scanf("%d",&t);
    while(t--)
    {
        int n;
       n=fastread();
     int a[n];
     int diff=-1;
     int mi=INT_MAX;
     for(int i=0;i<n;i++)
     {
         a[i]=fastread();
         mi=min(mi,a[i]);
         diff=max(diff,a[i]-mi);
     }
     if(diff>0)
        printf("%d\n",diff);
     else
        printf("UNFIT\n");

    }
     return 0;
}
