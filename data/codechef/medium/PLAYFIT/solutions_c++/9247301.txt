#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n;
        scanf("%d", &n);
        long long int a[n], b[n], c[n], max = 0, min = 1000000, ans = 0;
        for(int i = 0; i < n; i++) //for input and for minimum array
        {
            scanf("%lld", &a[i]);
            if(a[i] < min)
            {
                min = a[i];
            }
            b[i] = min;
        }
        for(int i = n - 1; i >= 0; i--) //for maximum array
        {
            if(a[i] > max)
            {
                max = a[i];
            }
            c[i] = max;

            if(c[i] - b[i] > ans)
            {
                ans = c[i] - b[i];
            }
        }

        if(ans > 0)
        {
            printf("%lld\n", ans);
        }
        else
        {
            printf("UNFIT\n");
        }
    }
    return 0;
}
