#include <iostream>
#include<stdio.h>
#include<stdlib.h>
using namespace std;

int main()
{
    int t;
   scanf("%d",&t);

    while(t--)
    {
        int n;
        scanf("%d",&n);
        int a[n];
        for(int i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        int tr,m=0,r=0;
        for(int i=n-1;i>=0;i--)
        {
            if(a[i]>m)
                m=a[i];
            tr=m-a[i];
            if(tr>r)
              r=tr;
        }
       if(r==0)
            printf("UNFIT\n");
        else
           printf("%d\n",r);
    }
  return 0;
}
