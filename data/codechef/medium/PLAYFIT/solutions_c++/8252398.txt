#include <bits/stdc++.h>
using namespace std;

int main() {
	long long i,x,t,n,minn=INT_MAX,maxn=-1,maxx=-1;
	cin>>t;
	while(t--){
		minn=INT_MAX;maxn=-1;maxx=-1;
		cin>>n;
		for(i=0;i<n;i++){
			cin>>x;
			if(x<minn)
			minn=x;
			maxx=x-minn;
			if(maxx>maxn)
			maxn=maxx;
		}
	if(maxn>0)	
	cout<<maxn<<endl;
	else
	cout<<"UNFIT"<<endl;
	}
	return 0;
}
