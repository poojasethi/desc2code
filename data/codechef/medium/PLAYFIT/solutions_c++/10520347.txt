#include <stdio.h>
#include <algorithm>
int main()
{
	int t, n, i, j;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		long int goal[n], max=0, diff;
		for(i=0; i<n; i++)
		{
			scanf("%ld",&goal[i]);
		}
		if(n<2)
		{
			printf("UNFIT\n");
		}
		else
		{
			for(i=n-2; i>=0; i--)
			{
			    if(i<n-2)
                {
                    if(goal[i+1]>goal[n-1])
                        goal[n-1]=goal[i+1];
                }
				diff=goal[n-1]-goal[i];
				if(diff>max)
                    max=diff;

			}
			if(max<=0)
			printf("UNFIT\n");
			else printf("%ld\n", max);
		}
	}
	return 0;
}
