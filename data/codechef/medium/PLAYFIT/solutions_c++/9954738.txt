#include<bits/stdc++.h>
using namespace std;
int main(){

	int t;
	int n;
	scanf("%d",&t);
	while(scanf("%d",&n)!=EOF)
	{
		int ans=-1,maxm=-1,minm=1000000007,x;

		for(int i=0;i<n;i++)
		{
			scanf("%d",&x);
			ans=max(ans,x-minm);
			minm=min(minm,x);
		}
		if(ans>0)
			printf("%d\n",ans);
		else
			printf("UNFIT\n");


	}
	return 0;
}
