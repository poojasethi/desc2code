#include<bits/stdc++.h>
using namespace std;
#define gc getchar
void scanint(int &x)
{
    register int c=gc();
    x=0;
    for(;(c<48||c>57);c=gc());
    for(;c>47&&c<58;c=gc()){x=(x<<1)+(x<<3)+c-48;}
}
main()
{
    int t;
    scanint(t);
    while(t--)
    {
        int n,i;
        scanint(n);
        int a[100005];
        scanint(a[0]);
        int low=a[0];
        int ans=0;
        for(i=1;i<n;i++)
        {
         scanint(a[i]);
         if(a[i]>low)
         {
         if(a[i]-low>ans)
         ans=a[i]-low;
         }
         else
         low=a[i];
        }
        if(ans>0)
        printf("%d\n",ans);
        else
        printf("UNFIT\n");
    }
}
