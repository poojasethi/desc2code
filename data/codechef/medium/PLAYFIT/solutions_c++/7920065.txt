#include<stdio.h>

int main()
{
	
	long int arr[100000]; int t;
	scanf("%d\n",&t);
	while(t--)
	{  long int n;
	scanf("%li\n",&n);
		
		for(long int i=0;i<n;i++)
		scanf("%li",&arr[i]);
		long int p[100000]; long int max=0,tmax=0;
		
		for(long int i=0;i<n-1;i++)
		p[i]=arr[i+1]-arr[i];
		
		
		for(long int i=0;i<n-1;i++)
		{  max=max+p[i];
			if(max<0)
			max=0;
			else if (tmax<max)
			tmax =max;
		}
		if(tmax<1)
		printf("UNFIT\n");
		else
		printf("%li\n",tmax);
	}
	
}
