#include<cstdio>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    int min[n];
    int max[n];
    int tmp=INF;
    for(int i=0;i<n;i++){
        if(arr[i]<tmp){
            tmp=arr[i];
        }
        min[i]=tmp;
    }
    tmp=0;
    for(int i=n-1;i>=0;i--){
        if(arr[i]>tmp){
            tmp=arr[i];
        }
        max[i]=tmp;
    }
    int diff=0;
    for(int i=0;i<n;i++){
        if(diff<max[i]-min[i]){
            diff=max[i]-min[i];
        }
    }
    if(diff<=0){
        printf("UNFIT\n");
    }
    else
    {
        printf("%d\n",diff);
    }
    }
}
