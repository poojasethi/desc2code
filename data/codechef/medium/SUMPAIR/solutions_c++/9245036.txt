#include <bits/stdc++.h>
using namespace std;
 
int main() {
	long long i,j,k,l,m,n,t,diff,temp; 
	long long hash[100005] ,a[100005]; 
	scanf("%lld",&t) ; 
	while(t--)
	{
		scanf("%lld %lld",&n,&diff) ; 
		for(i = 0 ; i < n ;i++)
		{
			scanf("%lld",&a[i]) ; 
			hash[i] = 0 ; 
		}
		sort(a,a+n) ; 
		long long sum = 0 ;
		for(i=n-1;i>=1;i--)
		{
		//	printf("fuck off\n")  ; 
			if(hash[i])
			continue; 
			temp = a[i] - a[i-1] ; 
			if(temp < diff )
			{
				sum = sum + a[i] + a[i-1] ; 
				hash[i] = 1 ; 
				hash[i-1] = 1 ; 
			}
		}
		printf("%lld\n",sum) ; 
	}
	return 0;
}
