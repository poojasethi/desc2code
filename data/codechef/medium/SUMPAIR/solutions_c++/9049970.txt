#include<bits/stdc++.h>
using namespace std;
int main()
{
    ios::sync_with_stdio(false);
    int t,i,j,k;
    cin>>t;
    while(t--)
    {
        int n,d;
        cin>>n>>d;
        int arr[n];
        for(i=0;i<n;i++)
            cin>>arr[i];
        sort(arr,arr+n);
        long long ans=0;
        for(i=n-1;i>=1;i--)
        {
            if(arr[i]-arr[i-1]<d)
            {
                ans+=arr[i]+arr[i-1];
                i--;
            }
        }
        cout<<ans<<"\n";
    }
}
