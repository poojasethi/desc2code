#include <iostream>
#include<algorithm>
using namespace std;
int main()
{ long long t,n,i,d,a[100002],sum;
 scanf("%lld",&t);
 while(t--)
 {scanf("%lld %lld",&n,&d);
  for(i=0;i<n;i++)
  scanf("%lld",&a[i]);
  sort(a,a+n);
 
  
  sum=0;
  for(i=n-1;i>=1; )
   
  { if((a[i]-a[i-1])<d)
    {
    sum+=(a[i]+a[i-1]);
    i-=2;
    }
    else
    {i-=1;
    }
    
  }
  printf("%lld\n",sum);
 }
 return 0;
}
