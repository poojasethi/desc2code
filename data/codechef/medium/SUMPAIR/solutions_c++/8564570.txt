#include<bits/stdc++.h>
using namespace std;
int main()
{  int t;
   cin>>t;
   while(t--)
   { long long int a[100008],n,d,i,sum=0;
     cin>>n>>d;
     for(i=0;i<n;i++)
   	  cin>>a[i];
   	  
   	  sort(a,a+n);
   	  
   	  for(i=n-1;i>=1;--i)
   	  if(a[i]-a[i-1]<d)
   	   { sum+=(a[i]+a[i-1]);
   	     --i;
   	   }
   	  	
		cout<<sum<<"\n"; 
   }
   return 0;
}
