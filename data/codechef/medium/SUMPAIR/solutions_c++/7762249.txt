using namespace std;

const int N = 5e5 + 5 + 5;

int t, n, a[N], d;

ll dp[N];

int main ()
{
//	freopen (fname".in", "r", stdin);
//	freopen (fname".out", "w", stdout);
    scanf ("%d", &t);
    while (t--)
    {
    	scanf ("%d%d", &n, &d);
    	for (int i = 1;i <= n;i++)
    		scanf ("%d", &a[i]);
    	sort (a + 1, a + n + 1);
    	for (int i = 2;i <= n;i++)
    	{
    		dp[i] = dp[i - 1];
    		if (a[i] - a[i - 1] < d)
    			dp[i] = max (dp[i], a[i] + a[i - 1] + dp[i - 2]);
    	}
    	printf ("%lld\n", dp[n]);
    }
	return 0;
}
