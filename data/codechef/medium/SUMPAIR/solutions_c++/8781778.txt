using namespace std ;
typedef long long  ll ;
const int MAXN = 100005 ;
int arr[MAXN] ;
int N, T, D ;
ll S ;
int main () {
       cin >> T ;
       while ( T-- ) {
              cin >> N >> D ;
              for ( int i = 0 ; i < N ; i++ )
                     scanf("%d" ,&arr[i] ) ;
              sort ( arr , arr + N ) ;
              for ( int i = N - 1  ; i >= 1 ; i-- ) {
                     if ( arr[i] - arr[i - 1] < D){
                            S += arr[i] + arr[i - 1] ;
                            i-- ;
                     }
              }
              printf("%lld\n" , S )  ;
              S = 0 ;
       }
       return 0 ;
}
