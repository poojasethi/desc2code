#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int t,n,i,sum=0,m,d;
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&n,&d);
        int a[n];
        for(i=0;i<n;i++)
            scanf("%lld",&a[i]);
        sort(a,a+n);
        for(i=n-1;i>=1;i--)
        {
            if((a[i]-a[i-1])<d)
            {
                sum=sum+a[i]+a[i-1];
                i--;
            }
        }
        printf("%lld\n",sum);
        sum=0;
    }
    return 0;
}
