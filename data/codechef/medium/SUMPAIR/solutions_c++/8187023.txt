#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int main() {
	long long int t,n,i,j,ans,a[100005],d;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld %lld",&n,&d);
		for(i=0;i<n;i++) {
			scanf("%lld",&a[i]);
		}
		ans=0;
		sort(a,a+n);
		i=n-2;
		j=n-1;
		while(i>=0) {
			if(a[j] - a[i] < d){
				ans+=(a[i]+a[j]);
				i-=2;
				j-=2;
			}
			else
			{
				i--;
				j--;
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
