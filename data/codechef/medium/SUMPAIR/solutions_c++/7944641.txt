#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
	long long int t,n,d,i,total;
	cin>>t;
	while(t--){
	total=0;
	cin>>n>>d;
	vector<int> arr(n);
	for(i=0;i<n;i++){
		cin>>arr[i];
	}	
	 sort(arr.begin(),arr.end());
	 i=n-1;	 
	while(i>0){
	if(arr[i]-arr[i-1]<d){
	total=total+arr[i]+arr[i-1];
	i=i-2;	
	continue;
	}	
	i=i-1;	
	}	
	cout<<total<<endl;	
	}
	
	
	return 0;
}
