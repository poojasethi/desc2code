#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstdlib>
#include<vector>
#include<cstring>
using namespace std;
int main()
{
	int t,n,d,k,i;
	scanf("%d",&t);
	
	while(t--)
	{
	long long int s=0;
	vector<int>arr;
		
	scanf("%d %d",&n,&d);
	for(i=0;i<n;i++)
	{
	scanf("%d",&k);
	arr.push_back(k);
	}
	
	sort(arr.begin(),arr.end());
	
	for(i=n-1;i>0;i--)
	{
		if((arr[i]-arr[i-1])<d)
		{
		s+=(arr[i]+arr[i-1]);
		i-=1;
		}
		}
		
		printf("%lld\n",s);
		}
	
		return 0;
		}
