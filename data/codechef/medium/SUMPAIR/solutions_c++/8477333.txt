#include <stdio.h>
#include<iostream>
#include<algorithm>
int main(void) {
	// your code goes here
	int t,n,i,d;
	long long sum;
	scanf("%d",&t);
	while(t--)
	{
		int n,d;
		scanf("%d %d",&n,&d);
		int a[n+1];
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		std::sort(a,a+n);
		
		sum=0; 
		for(i=n-1;i>0;)
		{   //printf("i:%d",i);
			
				if((a[i]-a[i-1])<d)
				{ 
					sum+=a[i]+a[i-1];
					
					i-=2;
					
					//sum+=a[i]+a[j];
				}
				else
				{
					i--;
				}
			
			
		 
		}
		printf("%lld\n",sum);
		
	}
	
	return 0;
}
