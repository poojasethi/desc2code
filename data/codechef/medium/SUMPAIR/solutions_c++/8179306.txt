#include <iostream>
#include<string.h>
#include<algorithm>
using namespace std;

int main() {

	ios_base::sync_with_stdio(0);
	long long t,m,x,y,i,j,k,r,s,c,n,d;
	cin>>t;
	while(t--)
	{

		cin>>n>>d;

		long long a[n];
		//memset(b,0,16*n);

		for(i=0;i<n;i++)
		{
			cin>>a[i];
		}
		sort(a,a+n);
		c=0;
		i=n-1;
		while(i>0)
		{
			if((a[i]-a[i-1])<d)
			{
				c+=a[i]+a[i-1];
				i-=2;
			}
			else
                i--;
		}


		cout<<c<<endl;
	}
	return 0;
}
