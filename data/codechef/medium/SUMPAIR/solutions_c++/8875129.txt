#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    long long int t,n,d;
    cin>>t;
    while(t--)
    {
        cin>>n>>d;
        long long int a[n],i;
        for(i=0;i<n;++i)
        cin>>a[i];
        long long int sum=0;
        sort(a,a+n);
        for(i=n-1;i>0;)
        {
            if(a[i]-a[i-1]<d)
            {sum+=a[i]+a[i-1];
             i=i-2;}
             else --i;
        }
        cout<<sum<<"\n";
    }
    return 0;
}
