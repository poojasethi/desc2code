#define getcx getchar
inline void in(int &n)
{
    n=0; int ch = getcx(); int sign = 1;
    while(ch < '0' || ch > '9') { if(ch == '-') sign=-1; ch = getcx(); }
    while(ch >= '0' && ch <= '9') { n = (n << 3) + (n << 1) + ch - '0', ch = getcx(); }
    n = n * sign;
}
int main()
{
    int t;
    in(t);
    while(t--)
    {
        int n,d;
        in(n);
        in(d);

        vector<int> v;
        for(int i=1; i<=n; i++)
        {
            int x;
            in(x);
            v.push_back(x);
        }
        sort(v.begin(),v.end());

        long long int ans=0;
        for(int i=v.size()-1; i>=0; i--)
        {
            if(i-1>=0)
            {
                if(v[i]-v[i-1]<d)
                {
                    ans+=v[i]+v[i-1];
                    i--;
                }
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
