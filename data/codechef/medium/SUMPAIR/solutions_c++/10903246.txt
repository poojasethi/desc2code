#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n,d;
		scanf("%d%d",&n,&d);
		int a[n];
		for (int i = 0; i < n; i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		long long int ans = 0;
		for (int i = n-1; i >= 1;)
		{
			if (a[i]-a[i-1] < d)
			{
				ans += (a[i]+a[i-1]);
				i = i-2;
			}
			else
				i--;
		}
		cout << ans << endl;
	}
}
