#include<bits/stdc++.h>
using namespace std;
int main() {
int t,num,diff,i;
long long sum;
scanf("%d",&t);
while(t--){
    cin>>num>>diff;
    int array[num];
    for(i=0;i<num;++i)cin>>array[i];
    sort(array,array+num);
    i=num-1;
    sum=0;
    while(i>0){
        if((array[i]-array[i-1])<diff)
        {sum+=array[i]+array[i-1];
            i-=2;
        }
        else --i;
    }
    printf("%lld\n",sum);
}
	return 0;
}
