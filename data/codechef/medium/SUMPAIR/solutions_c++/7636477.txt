#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
	int test;
	for(scanf("%d",&test);test>0;--test){
		int n,d;
		scanf("%d%d",&n,&d);
		int A[100010];
		for(int i=0;i<n;++i) scanf("%d",&A[i]);
		sort(A,A+n);
		reverse(A,A+n);
		int i=0;
		long long sum=0;
		while(i<n-1){
			if(A[i]-A[i+1]<d){ sum+=A[i]+A[i+1]; i+=2;}
			else ++i;
		}
		printf("%lld\n",sum);
	}
	return 0;
}
