#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int t,n,d,i;
long long sum;
cin>>t;
while(t--)
{
cin>>n>>d;
int arr[n];
sum=0;
for(i=0;i<n;i++)
cin>>arr[i];
sort(arr,arr+n);
for(i=n-1;i>=1;)
{
if(arr[i]-arr[i-1]<d)
{
sum=sum+arr[i]+arr[i-1];
i=i-2;
}
else
i=i-1;
}
cout<<sum<<"\n";
}
}
