#include <iostream>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;

int main() {
    int t;
    cin>>t;
    int n,d;
    while(t--) {
        cin>>n>>d;
        vector<int> vec;
        for(int i=0;i<n;i++) {
            int ele;
            cin>>ele;
            vec.push_back(ele);
        }
        sort(vec.begin(), vec.end());
        long long int max_sum = 0;
        for(int i=n-1;i>0;i--) {
            if(vec[i]-vec[i-1]<d) {
                max_sum+=vec[i]+vec[i-1];
                i--;
            }
        }
        cout<<max_sum<<endl;
    }
}
