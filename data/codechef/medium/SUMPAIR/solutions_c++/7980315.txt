#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    int t,n,d,i,j,k,a[100005];
    long long int s;
    sd(t);
    while(t--)
    {
        sd(n);
        sd(d);
        for(i=0;i<n;i++)
            sd(a[i]);
        sort(a,a+n);
        s=0;
        j=a[n-1];
        for(i=n-2;i>=0;i--)
        {
            if(j-a[i]<d)
            {
                s+=j+a[i];
                if(i>0)
                    j=a[--i];
            }
            else
                j=a[i];
        }
        printf("%lld\n",s);
    }
    return 0;
}
