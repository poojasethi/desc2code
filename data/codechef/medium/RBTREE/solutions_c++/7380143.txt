{
	int n,i;
	scanf("%d",&n);
	int count=0;
	int low[35],high[35];
	low[1]=1;
	high[1]=1;
	for(i=2;i<=30;i++)
	{
		low[i]=high[i-1]+1;
		high[i]=2*low[i]-1;
	}
	while(n--)
	{
		char q[3];
		long long int ans=0;
		long long int a,b,tmp;
		scanf("%s",q);
		if(q[1]=='i')
		{
			//chnge();
			count=!count;
		}
		else 
		{
			int lev1,lev2;
			scanf("%lld %lld",&a,&b);
			if(a>b)
			{
				tmp=a;
				a=b;
				b=tmp;
			}
			for(i=1;i<=30;i++)
			{
				if(low[i]<=a&&high[i]>=a)
				{
					lev1=i;
					break;
				}
			}
			for(i=1;i<=30;i++)
			{
				if(low[i]<=b&&high[i]>=b)
				{
					lev2=i;
					break;
				}
			}
			
			int levf,dif;
			dif=lev2-lev1;
			while(dif)
			{
				b=b/2;
				dif--;
			}
			levf=lev1;
			while(b!=a)
			{
				b=b/2;
				a=a/2;
				levf--;
			}
			int red=0,blk=0;
			while(lev1>=levf)
			{
				if(lev1%2==0)
				red++;
				else blk++;
				lev1--;
			}
			while(lev2>levf)
			{
				if(lev2%2==0)
				red++;
				else blk++;
				lev2--;
			}
			if(count==0)
			{
				if(q[1]=='b')
				printf("%d\n",blk);
				else
				printf("%d\n",red);
			}
			else
			{
				if(q[1]=='b')
				printf("%d\n",red);
				else
				printf("%d\n",blk);
			}
		}
	}

}
