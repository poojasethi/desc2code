#include <cstdio>
#include <cstring>
using namespace std;
char op[10];
int MSBParity(int x)
{
	for(int i=30; i>=0; i--)
		if (x & (1 << i))
			return (i & 1);
	return -1;
}
int main()
{
	int Q, x=0, y, a, b, c;
	scanf("%d", &Q);
	while (Q--)
    {
		scanf("%s", op);
		if (!strcmp(op, "Qi"))
        {
			x ^= 1;
		}
		else
        {
			if (!strcmp(op, "Qb"))
				y = x;
			else
				y = x ^ 1;
			scanf("%d %d", &a, &b);
			c = 0;
			while (a != b)
			{
				if (a > b)
				{
					if(MSBParity(a) == y)
                        c++;
					a >>= 1;
				}
                else
				{
					if(MSBParity(b) == y)
                        c++;
					b >>= 1;
				}
			}
			if (MSBParity(a) == y)
                c++;
			printf("%d\n", c);
		}
	}
	return 0;
}
