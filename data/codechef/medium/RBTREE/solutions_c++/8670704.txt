using namespace std;
vector<long long int> v1,v2;
//vector<int>::iterator it;
long long int lca(long long int x,long long int y)
{
	long long int s1,s2,ret;
	while(x>=1)
	{
		v1.push_back(x);
		x/=2;
	}
	
	while(y>=1)
	{
		v2.push_back(y);
		y/=2;
	}
	
	s1=v1.size()-1;
	s2=v2.size()-1;
	ret=-1;
	while(s1>=0 && s2>=0)
	{
		if(v1[s1]!=v2[s2])
		{
			ret=v1[s1+1];
			break;
		}
		else
		{
			s1--;
			s2--;
		}
	}
	return ret;
}
long long int rd=0,blk=0;

				long long int lev1,lev2,dif;

	            		blk+=((dif/2)+1);
	            		rd+=(dif/2);
	            	}
	            	else
	            	{
	            		rd+=((dif/2)+1);
	            		blk+=(dif/2);
	            	}
	            }
}
