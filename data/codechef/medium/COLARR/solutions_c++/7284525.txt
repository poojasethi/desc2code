int b[1010][1010],c[1010][1010],arr[1010];;
int main()
{
	int test,i,n,m,k,j;
	int x,prev=0,val,counter;
	VI vec;
	PII mypair;
	VPII vec1;
	scanf("%d",&test);
	while(test--)
	{
		counter=0,prev=0;
		scanf("%d%d%d",&n,&m,&k);
		
		for(int i=0;i<n;i++)
		{
			scanf("%d",&x);
			vec.push_back(x);
			arr[i]=-1;
		}
		
		for(int i=0;i<n;i++){
			for(int j=1;j<=m;j++)
				scanf("%d",&b[i][j]);

			prev+=b[i][vec[i]];
		}
		for(int i=0;i<n;i++){
			for(int j=1;j<=m;j++)
				scanf("%d",&c[i][j]);
		}
		int  maxi;
		for(int i=0;i<n;i++)
		{
			maxi=0;
			x=b[i][vec[i]];
			for(int j=1;j<=m;j++)
			{

				
				val=b[i][j]-c[i][j];
				if(val>x)
					x=val;
			}
			if(x!=b[i][vec[i]]){
				mypair.first=x-b[i][vec[i]];
				mypair.second=i;
				vec1.push_back(mypair);
			}
		}

		sort(vec1.rbegin(),vec1.rend());
		int len=vec1.size();

		for(int i=0;i<len;i++)
		{
			if(counter==k)
				break;
			int x=vec1[i].second;
			arr[x]=vec1[i].first+b[x][vec[x]];
			counter++;			
		}
		
		int sum=0;

		for(int i=0;i<n;i++)
		{
			if(arr[i]==-1)
				arr[i]=b[i][vec[i]];
			sum+=arr[i];
		}

		printf("%d\n",max(sum,prev));
		
		vec1.clear();
		vec.clear();


	}
	return 0;
}
