#include <stdio.h>
#include <algorithm>
#include <vector>
#include <string>
#include <math.h>
#include <iostream> 
#include <map>
#include <set>
#include <functional>
#include <numeric>
using namespace std;
typedef unsigned long long int u64;

static int a[1001];
static int b[1001][1001];
static int c[1001][1001];
static int delta[1001];

int main()
{
    ios_base::sync_with_stdio(false);
    int t,n,m,k;
    cin>>t;
    while(t--)
    {
        cin >> n >> m >> k;
        for(int i = 1; i <= n; i++)
            cin >> a[i];
        for(int i = 1; i <= n; i++)
            for(int j = 1; j <= m; j++)
            {
                cin >> b[i][j];
            }
        for(int i = 1; i <= n; i++)
            for(int j = 1; j <= m; j++)
            {
                cin >> c[i][j];
            }
        for(int i = 1; i <= n; i++)
        {
            delta[i] = 0;
            for(int j = 1; j <= m ; j++)
                delta[i] = max(delta[i], b[i][j] - b[i][a[i]] - c[i][j]);
        }
        sort(delta+1,delta+n+1);
        long int ans = 0L;
        for(int i = 1; i <= n; i++)
            ans += b[i][a[i]];
        for(int i = 1; i <= k ; i++)
            ans += delta[n-i+1];
        cout << ans << endl;
    }
    return 0;
}
