#include<bits/stdc++.h>
using namespace std;
#define ll long long
int n,m,c;
int i=0,j=0;
int a[1010],poscol[1010][1010],negcol[1010][1010];
int getn()
{int  i=0;
 char ch=getchar_unlocked();
 while(ch>'9'||ch<'0')
 ch=getchar_unlocked();
 while(ch<='9'&&ch>='0')
 {i=(i<<1)+(i<<3)+ch-'0';
  ch=getchar_unlocked();
 }
return i;
}
int main(){
    int t;
    t=getn();
    for(i=0;i<t;i++){
    n=getn();
    m=getn();
    c=getn();
    ll paint=0,sum1=0;
   ll  netval[n],inval[n];
    for(j=0;j<n;j++){
    a[j]=getn();
    }
    for(j=0;j<n;j++){
    for(int k=0;k<m;k++){
    poscol[j][k]=getn();
    }
    inval[j]=poscol[j][a[j]-1];
    sum1+=inval[j];
    }
    for(j=0;j<n;j++){
    ll p=INT_MIN;
    for(int k=0;k<m;k++){
    negcol[j][k]=getn();
    ll val=poscol[j][k]-negcol[j][k];
    if(val>p)
    p=val;
    }
    netval[j]=p;
    }
        ll diff[n];
    for(j=0;j<n;j++){
    diff[j]=netval[j]-inval[j];
    }
    sort(diff,diff+n);
    for(j=n-1;j>=0;j--){
    if(paint==c)
    break;
    if(diff[j]>0){
    sum1+=diff[j];
    paint++;
    }
    else
    break;
    }
    printf("%lld\n",sum1);
    }
return 0;
}
