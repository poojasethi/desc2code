#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
    int test;

    scanf("%d",&test);
    while(test--)
    {
        int n,m,k;
        scanf("%d%d%d",&n,&m,&k);
        int a[n+1];
        int start[n+1];
        int changed[n+1];
        int sum=0;
        for(int i=1;i<=n;i++)
        {
            scanf("%d",&a[i]);
            start[i]=0;
            changed[i]=0;
        }
        int b[n+1][m+1];
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=m;j++)
            {
                scanf("%d",&b[i][j]);
            }
        }
        int c[n+1][m+1];
        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=m;j++)
            {
                scanf("%d",&c[i][j]);
            }
        }

        for(int i=1;i<=n;i++)
        {
            start[i]=b[i][a[i]] ;
            sum+=start[i];
            int ma=0;
            for(int j=1;j<=m;j++)
            {
                if((b[i][j]-c[i][j])>ma)
                    ma=b[i][j]-c[i][j];
            }
            changed[i]=ma-start[i];
            if(changed[i]<0)
                changed[i]=0;

        }
        sort(changed+1,changed+n+1);
    changed[0]=0;
    int i=n;
    for(i=n;k>0&&i>=0; i--)
		{

		    sum+=changed[i];
                k--;
		}


    printf("%d\n",sum);
    }
}
