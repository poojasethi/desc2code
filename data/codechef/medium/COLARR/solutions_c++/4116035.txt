#include<bits/stdc++.h>
using namespace std;
#define ll long long
int n,m,c;
int i=0,j=0;
int a[1010],poscol[1010][1010],negcol[1010][1010];
int main(){
    int t;
    scanf("%d",&t);
    for(i=0;i<t;i++){
    scanf("%d%d%d",&n,&m,&c);
    ll paint=0,sum1=0;
   ll  netval[n],inval[n];
    for(j=0;j<n;j++){
    scanf("%d",&a[j]);
    }
    for(j=0;j<n;j++){
    for(int k=0;k<m;k++){
    scanf("%d",&poscol[j][k]);
    }
    inval[j]=poscol[j][a[j]-1];
    sum1+=inval[j];
    }
    for(j=0;j<n;j++){
    ll p=INT_MIN;
    for(int k=0;k<m;k++){
    scanf("%d",&negcol[j][k]);
    ll val=poscol[j][k]-negcol[j][k];
    if(val>p)
    p=val;
    }
    netval[j]=p;
    }
        ll diff[n];
    for(j=0;j<n;j++){
    diff[j]=netval[j]-inval[j];
    }
    sort(diff,diff+n);
    for(j=n-1;j>=0;j--){
    if(paint==c)
    break;
    if(diff[j]>0){
    sum1+=diff[j];
    paint++;
    }
    else
    break;
    }
    printf("%lld\n",sum1);
    }
return 0;
}
