#define CLR(a,x) memset(a,x,sizeof(a))
#define PB push_back
#define INF 1000000000
#define MOD 1000000007
#define MP make_pair
#define tr(container , it) for(typeof(container.begin()) it=container.begin() ; it!=container.end() ; it++)
#define FOR(i,a,b) for(i=a;i<b;i++)
#define REP(i,a) FOR(i,0,a)
#define LLD long long
#define VI vector < int >
#define PII pair < int , int >
#define MAX 1000000000
int A[10004];
int B[1004][1004];
int C[1004][1004];
vector<int> V[1004];
vector<int> L;
int main()
{
    int T,N,M,K;
    scanf("%d",&T);
 
    while(T--)
    {
	for(int i=0;i<=1000;i++)
	    V[i].clear();
	L.clear();
 
	scanf("%d%d%d",&N,&M,&K);
	for(int i=1;i<=N;i++)
	    scanf("%d",&A[i]);
 
	for(int i=1;i<=N;i++)
	    for(int j=1;j<=M;j++)
		scanf("%d",&B[i][j]);
 
	for(int i=1;i<=N;i++)
	    for(int j=1;j<=M;j++)
		scanf("%d",&C[i][j]);
 
	for(int i=1;i<=N;i++)
	    for(int j=1;j<=M;j++)
	       V[i].PB(B[i][j]-C[i][j]);
 
	for(int i=1;i<=N;i++)
	    sort(V[i].begin(), V[i].end());
 
	for(int i=1;i<=N;i++)
	{
	    int h = V[i][M-1];
	    L.PB(h-B[i][A[i]]);
	}
 
	sort(L.begin(), L.end());
 
	int ans = 0;
 
	for(int i=1;i<=N;i++)
	    ans += B[i][A[i]];
 
 
	int i = N-1;
	while(i>=0 && K>0)
	{
	    if(L[i]>0)
	    {
	      ans += L[i];
	      K--;
	    }
	    else
		break;
	    i--;
	}
 
	printf("%d\n",ans);
 
 
    }
 
 
    return 0;
}
