#include<bits/stdc++.h>
using namespace std;

#define sd(x) scanf("%d",&x)
#define ss(x) scanf("%s",x)
#define lld long long int
#define slld(x) scanf("%lld",&x)

int main()
{
    int t,r,g,b,m,ans,rm,bm,gm,ar[101],bb[101],cg[101],i;
	sd(t);
	while(t--)
	{
        sd(r);sd(g);sd(b);sd(m);
        rm=0;bm=0;gm=0;
        for(i=0;i<r;i++)
            {
            sd(ar[i]);
            if(ar[i]>rm)
            rm=ar[i];
            }

        for(i=0;i<g;i++)
            {
            sd(cg[i]);
            if(cg[i]>gm)
            gm=cg[i];
            }
        for(i=0;i<b;i++)
            {
            sd(bb[i]);
            if(bb[i]>bm)
            bm=bb[i];
            }
        for(i=0;i<m;i++)
        {
            if(rm>bm)
            {
                if(rm>gm)
                {
                    rm=rm/2;
                }
                else
                {
                    gm=gm/2;
                }
            }
            else
            {
                if(bm>gm)
                {
                    bm=bm/2;
                }
                else
                {
                    gm=gm/2;
                }
            }
            ans=max(rm,max(bm,gm));
        }
        printf("%d\n",ans);
	}

	return 0;
}
