#include<iostream>
#include<vector>
using namespace std;
int main()
{
    int r,g,b,m,mxr=-1,mxg=-1,mxb=-1,t,val;
    cin>>t;
    while(t--)
    {
        mxr=-1;mxg=-1;mxb=-1;
        cin>>r>>g>>b>>m;
        while(r--) {cin>>val;mxr=max(mxr,val);}
        while(g--) {cin>>val;mxg=max(mxg,val);}
        while(b--){cin>>val;mxb=max(mxb,val);}
        while(m--)
        {
            if(mxr>=mxg && mxr>=mxb) mxr/=2;
            else if(mxg>=mxr && mxg>=mxb) mxg/=2;
            else mxb/=2;
        }
        cout<<max(mxr,max(mxg,mxb))<<endl;
    }
    return 0;
}
