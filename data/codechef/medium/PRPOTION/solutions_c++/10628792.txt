#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
#define MAX 100
unsigned long long int a[MAX],b[MAX],c[MAX];
int la[3];
int main()
{
 int test;
 scanf("%d",&test);
 while(test--)
 {
     int red,gr,bl,m;
     scanf("%d%d%d%d",&red,&gr,&bl,&m);
     for(int i=0;i<red;i++)
        scanf("%llu",&a[i]);
     for(int i=0;i<gr;i++)
        scanf("%llu",&b[i]);
     for(int i=0;i<bl;i++)
        scanf("%llu",&c[i]);
     sort(a,a+red);
     sort(b,b+gr);
     sort(c,c+bl);
     la[0]=a[red-1];
     la[1]=b[gr-1];
     la[2]=c[bl-1];
     for(int i=m-1;i>=0;i--)
     {  sort(la,la+3);
        la[2]=la[2]/2;
       }
 
     sort(la,la+3);
     printf("%d\n",la[2]);
 }
 return 0;
}
