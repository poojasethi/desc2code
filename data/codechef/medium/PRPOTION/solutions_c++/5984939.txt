#include<stdio.h>
# define MINIMUM 2000000009
long long int maxi(long long int x,long long int y,long long int z)
{
 if(x>=y&&z<=x)
 return x;
 else if(y>=x&&y>=z)
  return y;
  else
  return z;   
}
long long int min(long long int x,long long int y)
{
 if(x<y)
 return x;
 else
 return y;     
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
      long long int p,q,r,m,x,max1=0,max2=0,max3=0,maxA,i,minop; 
     scanf("%lld%lld%lld%lld",&p,&q,&r,&m);
     minop=2000000009;
     for(i=0;i<p;i++)
     {
     scanf("%lld",&x);
     if(x>max1)
     max1=x;
     }
     for(i=0;i<q;i++)
     {
     scanf("%lld",&x);
     if(x>max2)
     max2=x;
     }
     for(i=0;i<r;i++)
     {
     scanf("%lld",&x);
     if(x>max3)
     max3=x;
     }
     for(i=0;i<m;i++)
     {           
      maxA=maxi(max1,max2,max3);
      minop=min(maxA,minop);
      if(maxA==max1)
      { 
      max1/=2;
      }
      else if(maxA==max2)
      {
      max2/=2;
      }
      else
      {
      max3/=2;
      }              
     }
     maxA=maxi(max1,max2,max3);
     printf("%lld\n",min(minop,maxA));         
    }
 return 0;     
}
