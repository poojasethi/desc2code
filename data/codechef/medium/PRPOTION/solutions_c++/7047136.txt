#include<iostream>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int r,b,g,m,i;
		long long int max1,max2,max3;
		max1=max2=max3=0;
		cin>>r>>g>>b>>m;
		long long int ar1[r],ar2[g],ar3[b];
		for(i=0;i<r;i++)
		{
		  cin>>ar1[i];
		  if(max1<ar1[i])
		  max1=ar1[i];
	    }
		for(i=0;i<g;i++)
		{
		   cin>>ar2[i];
		   if(max2<ar2[i])
		   max2=ar2[i];
	    }
		for(i=0;i<b;i++)
		{
		   cin>>ar3[i];
		   if(ar3[i]>max3)
		   max3=ar3[i];
	    }
		while(m--)
		{
			if(max1>=max2&&max1>=max3)
			{ 
			    max1=0;
				for(i=0;i<r;i++)
				{
					ar1[i]/=2;
					if(max1<ar1[i])
					max1=ar1[i];
				}
			}
			else if(max1<=max2&&max2>=max3)
			{  
			   max2=0;
				for(i=0;i<g;i++)
				{
					ar2[i]/=2;
					if(max2<ar2[i])
					max2=ar2[i];
				}
			}
			else
			{  
			   max3=0;
				for(i=0;i<b;i++)
				{
					ar3[i]/=2;
					if(max3<ar3[i])
					max3=ar3[i];
				}
			}
		}
		if(max1>=max2&&max1>=max3)
		cout<<max1<<endl;
		else if(max1<=max2&&max2>=max3)
		cout<<max2<<endl;
		else
		cout<<max3<<endl;
	}
	return 0;
}
