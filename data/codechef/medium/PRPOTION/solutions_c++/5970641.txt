#include<stdio.h>
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int r,g,b,m;
		long long int s1=0,s2=0,s3=0,i,y,z=0,a[101],d[101],c[101];
		scanf("%d%d%d%d",&r,&g,&b,&m);
	        
			for(i=0;i<r;i++)
			{
				scanf("%lld",&a[i]);
				if(s1<a[i])
				s1=a[i];
			}
			for(i=0;i<g;i++)
			{
				scanf("%lld",&d[i]);
				if(s2<d[i])
				s2=d[i];
			}
			for(i=0;i<b;i++)
			{
				scanf("%lld",&c[i]);
				if(s3<c[i])
				s3=c[i];
			}
		while(m--)
		{
			y=(s1>=s2)?((s1>=s3)?s1:s3):((s2>=s3)?s2:s3);
		//	printf("%lld ",y);
			if(y==s1)
	        s1=s1/2;
	        
			else if(y==s2)
		    s2=s2/2;
		    
			else 
		    s3=s3/2;
		    
		}
	//	printf("%lld%lld%lld ",s1,s2,s3);
	    z=(s1>=s2)?((s1>=s3)?s1:s3):((s2>=s3)?s2:s3);
		printf("%d\n",z);
	}
	return 0;
}
