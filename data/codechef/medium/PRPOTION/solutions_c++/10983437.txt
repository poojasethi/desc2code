#include<iostream>
using namespace std;
int main()
{
    long int red,green,blue,a,array[3],temp;
    int t,x,y,z,i,j,k,m;
    cin>>t;
    while(t--)
    {
        cin>>x>>y>>z>>m;
        cin>>red;
        for(i=1;i<x;i++)
        {
            cin>>a;
            if(a>red)
             red=a;
        }
        array[0]=red;
        cin>>green;
        for(i=1;i<y;i++)
        {
            cin>>a;
            if(a>green)
             green=a;
        }
        array[1]=green;
        cin>>blue;
        for(i=1;i<z;i++)
        {
            cin>>a;
            if(a>blue)
             blue=a;
        }
        array[2]=blue;


        for(k=0;k<m;k++)
        {
             for(j=0;j<2;j++)
        {
            for(i=j+1;i<3;i++)
        {
            if(array[i]>array[j])
             {
                 temp=array[i];
                 array[i]=array[j];
                 array[j]=temp;
             }
        }
        }
            array[0]/=2;
        }
        if(array[1]>array[0])
         cout<<array[1]<<endl;
        else
         cout<<array[0]<<endl;
    }
}
