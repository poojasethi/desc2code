#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int r,g,b,m;
		cin>>r>>g>>b>>m;
		long long int x,a=0,y=0,c=0,ans=0,i;
		for(i=0;i<r;i++)
		{
			cin>>x;
			a=max(a,x);
		}
		for(i=0;i<g;i++)
		{
			cin>>x;
			y=max(y,x);
		}
		for(i=0;i<b;i++)
		{
			cin>>x;
			c=max(c,x);
		}
		ans=max(a,max(y,c));
		while(m-->0)
		{
			if(a>=y && a>=c){
				a/=2;
			}
			else if(y>=a && y>=c){
				y/=2;
			}
			else{
				c/=2;
			}
		}
		ans=max(a,max(y,c));
		cout<<ans<<endl;
	}
	return 0;
}
