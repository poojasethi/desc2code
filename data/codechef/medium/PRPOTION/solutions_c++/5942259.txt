#include <iostream>
using namespace std;

int max_(int a, int b, int c) {
	if(a>=b && a >=c)
		return a;
	else if(b >= c)
		return b;
	else 
		return c;
}

int main() {
	int i, t, r, g, b, m, input;
	int col[3];
	cin>>t;
	while(t--) {
		int max[3] = {-1};
		cin>>col[0]>>col[1]>>col[2]>>m;
		for(i = 0; i < 3; i++) {
			while(col[i]--) {
				cin>>input;
				if(input > max[i])
					max[i] = input;
			}
		}
		while(m--) {
			if(max[0] >= max[1] && max[0] >= max[2])
				max[0] >>= 1;
			else if(max[1] >= max[2])
				max[1] >>= 1;
			else
				max[2] >>= 1;
			if(!max[0] && !max[1] && !max[2])
				break;
		}
		cout<<max_(max[0], max[1], max[2])<<endl;
	}
	return 0;
}
