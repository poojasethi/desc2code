#include<iostream>
#include<math.h>
#include<cstdio>
#include<algorithm>

using namespace std;

int main(){
    int t;
    cin >> t;
    while(t--){
        int i,r,g,b,m,a1[101],a2[101],a3[101];
        int max1=0,max2=0,max3=0;
        int max4,fmax;
        cin >> r;
        cin >> g;
        cin >> b;
        cin >> m;
        for(i=0;i<r;i++){
            cin >> a1[i];
            if(a1[i]>max1) max1=a1[i];
        }

        for(i=0;i<g;i++){
            cin >> a2[i];
            if(a2[i]>max2) max2=a2[i];
        }
        for(i=0;i<b;i++){
            cin >> a3[i];
            if(a3[i]>max3) max3=a3[i];
        }


        while(m--){
            if(max1>=max2 && max1>=max3){
                max1=max1/2;
            }
            else if(max2>=max1 && max2>=max3){
                max2=max2/2;

            }
            else {
                max3=max3/2;
            }

        }
        max4=max(max1,max2);
        fmax=max(max3,max4);
        cout << fmax << "\n";
    }

return 0;
}
