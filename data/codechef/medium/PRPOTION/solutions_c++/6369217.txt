
    #include <iostream>
    #include <math.h>
    #include <stdio.h>
    #include <string>
    #include <algorithm>
    using namespace std;
     
    int maxm(int a,int b,int c)
    {
    if(a>=b&&a>=c)return a;
    else if(b>=c&&b>=a)return b;
    else if(c>=a&&c>=b)return c;
    }
     
    int main()
    {
    int t;
    scanf("%d",&t);
    int count=0;
    while(count<t)
    {
    int R,G,B,M;
    scanf("%d %d %d %d",&R,&G,&B,&M);
    int r[R],g[G],b[B];
    int max1=0,max2=0,max3=0;
    for(int i=0;i<R;i++){scanf("%d",&r[i]); if(max1<r[i])max1=r[i];}
    for(int i=0;i<G;i++){scanf("%d",&g[i]); if(max2<g[i])max2=g[i];}
    for(int i=0;i<B;i++){scanf("%d",&b[i]); if(max3<b[i])max3=b[i];}
    for(int i=0;i<M;i++)
    {
    if(max1>=max2&&max1>=max3)max1=max1/2;
    else if(max2>=max1&&max2>=max3)max2=max2/2;
    else if(max2>=max2&&max3>=max1)max3=max3/2;
    }
    //cout<<max1<<" "<<max2<<" "<<max3<<endl;
    cout<<maxm(max1,max2,max3)<<endl;
    count++;
    }
    }
