#include<iostream>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
int r,g,b,m,i;
long long int x=0,y=0,z=0,a;
cin>>r>>g>>b>>m;
for(i=0;i<r;i++)
{
    cin>>a;
    if(a>x)
        x=a;
}
for(i=0;i<g;i++)
{
    cin>>a;
    if(a>y)
        y=a;;
}
for(i=0;i<b;i++)
{
    cin>>a;
    if(a>z)
        z=a;
}
for(i=0;i<m;i++)
{
    if(x>=y&&x>=z)
        x=x/2;
    else if(y>=x&&y>=z)
        y=y/2;
    else
        z=z/2;
}
if(x>=y&&x>=z)
    cout<<x;
else if(y>=x&&y>=z)
    cout<<y;
else
    cout<<z;
cout<<'\n';
}
return 0;
}
