#include<bits/stdc++.h>
using namespace std;
int t,i,m,n,r,g,b,x,maxr,maxg,maxb;
int main(){
    scanf("%d",&t);
    while(t--){
        maxr=0,maxg=0,maxb=0;
        int v[100001];
        cin>>r>>g>>b>>m;
        for(i=0;i<r;i++){
            cin>>x;
            maxr=max(maxr,x);
            }
        for(i=0;i<g;i++){
            cin>>x;
            maxg=max(maxg,x);
            }

        for(i=0;i<b;i++){
            cin>>x;
            maxb=max(maxb,x);
            }

        v[0]=maxr;
        v[1]=maxg;
        v[2]=maxb;

        while(m--){
                sort(v,v+3);
                v[2]=v[2]/2;
        }
        sort(v,v+3);
        cout<<v[2]<<endl;
    }
return 0;
}
