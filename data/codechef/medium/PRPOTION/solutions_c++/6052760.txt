
    #include <iostream>
    #include <stdio.h>
    using namespace std;
    int main(){
    int t;
    scanf("%d",&t);
    while(t--){
    int r,g,b,m,maxr,maxg,maxb,tmp;
    scanf("%d %d %d %d",&r,&g,&b,&m);
    scanf("%d",&maxr);
    for(int i=1;i<r;i++) {scanf("%d",&tmp);if(tmp>maxr) maxr=tmp;}
    scanf("%d",&maxg);
    for(int i=1;i<g;i++) {scanf("%d",&tmp);if(tmp>maxg) maxg=tmp;}
    scanf("%d",&maxb);
    for(int i=1;i<b;i++) {scanf("%d",&tmp);if(tmp>maxb) maxb=tmp;}
    for(int i=0;i<m;i++){
    int curmax=max(maxr,max(maxg,maxb));
    if(curmax==maxr) maxr=maxr/2;
    else if(curmax==maxg) maxg=maxg/2;
    else if(curmax==maxb) maxb=maxb/2;
    }
    printf("%d\n",max(maxr,max(maxg,maxb)));
    }
    }
