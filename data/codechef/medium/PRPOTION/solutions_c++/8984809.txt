#include<bits/stdc++.h>
using namespace std;
int main()
  {
   int t,z,n1,big,n2,n3,m,i;
   scanf("%d",&t);
   for(z=0;z<t;z++)
   {
    scanf("%d%d%d%d",&n1,&n2,&n3,&m);
    int a1[n1],a2[n2],a3[n3];
    for(i=0;i<n1;i++)
       scanf("%d",&a1[i]);
    for(i=0;i<n2;i++)
       scanf("%d",&a2[i]);
     for(i=0;i<n3;i++)
        scanf("%d",&a3[i]);
     sort(a1,a1+n1,greater<int>());
     sort(a2,a2+n2,greater<int>());
     sort(a3,a3+n3,greater<int>());
     while(1)
     {
      big=((a1[0]>a2[0])&&(a1[0]>a3[0]))?a1[0]:(a2[0]>a3[0])?a2[0]:a3[0];
      //cout<<endl<<big<<endl;
      if(m==0)
        break;
      if(big==a1[0])
      {
      a1[0]/=2;
 
      }
      else if(big==a2[0])
      {
       a2[0]/=2;

 
      }
     else
      {
       a3[0]/=2;

 
      }
 
     m--;
     }
    printf("%d\n",big);
 
   }
 
   return 0;
  }
