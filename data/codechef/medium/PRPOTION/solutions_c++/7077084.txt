#include<stdio.h>
int maxi(int a,int b,int c)
{
    if(a>=b && a>=c)
        return a;
    else if(b>=a && b>=c)
        return b;
    else
        return c;
}
int main()
{
    int t,R,max,B,G,mr,mg,mb,i,m;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d%d",&R,&G,&B,&m);
        mr=mg=mb=0;
        int r[R],b[B],g[G];
        for(i=0;i<R;++i)
            {scanf("%d",&r[i]);if(r[i]>mr) mr=r[i];}
        for(i=0;i<G;++i)
            {scanf("%d",&g[i]);if(g[i]>mg) mg=g[i];}
        for(i=0;i<B;++i)
            {scanf("%d",&b[i]);if(b[i]>mb) mb=b[i];}
        while(m--)
        {
             max=maxi(mr,mb,mg);
             if(max!=0)
             {
             if(max==mr)
                mr/=2;
             else if(max==mg)
                mg/=2;
             else
                mb/=2;
             }
             else
                break;
        }
        printf("%d\n",maxi(mr,mg,mb));
    }
    return 0;
}
