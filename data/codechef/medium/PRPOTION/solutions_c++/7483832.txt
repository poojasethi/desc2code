#include<bits/stdc++.h>
using namespace std;
long red[105],blue[105],green[105];
int main()
{
	int r,g,b,m,t;
	scanf("%d",&t);
	while(t--)
	{
	scanf("%d%d%d%d",&r,&g,&b,&m);
	for(int i=0;i<r;i++)
	scanf("%d",&red[i]);
	for(int i=0;i<g;i++)
	scanf("%d",&green[i]);
	for(int i=0;i<b;i++)
	scanf("%d",&blue[i]);
    while(m--)
    {
    	long max1=0,max2=0,max3=0;
    	for(int i=0;i<r;i++)
    	{
    		if(max1<red[i])
    		max1=red[i];
    	}
    	for(int i=0;i<g;i++)
    	{
    		if(max2<green[i])
    		max2=green[i];
    	}
    	for(int i=0;i<b;i++)
    	{
    		if(max3<blue[i])
    		max3=blue[i];
    	}
    	long ans=max(max1,max(max2,max3));
    	if(ans==max1)
    	{
    		for(int i=0;i<r;i++)
    		red[i]=red[i]/2;
    	}
    	else if(ans==max2)
    	{
    		for(int i=0;i<g;i++)
    		green[i]/=2;
    	}
    	else
    	{
    		for(int i=0;i<b;i++)
    		blue[i]/=2;
    	}
    }
    long max1=0,max2=0,max3=0;
    	for(int i=0;i<r;i++)
    	{
    		if(max1<red[i])
    		max1=red[i];
    	}
    	for(int i=0;i<g;i++)
    	{
    		if(max2<green[i])
    		max2=green[i];
    	}
    	for(int i=0;i<b;i++)
    	{
    		if(max3<blue[i])
    		max3=blue[i];
    	}
    	long ans=max(max1,max(max2,max3));
    	printf("%ld\n",ans);
    }
}
