#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
    int t, r, g, b, m, rr[100], gg[100], bb[100], i, max;
    cin>>t;
    while(t--)
    {
        cin>>r>>g>>b>>m;
        for(i=0; i<r; i++)
        {
            cin>>rr[i];
        }
        for(i=0; i<g; i++)
        {
            cin>>gg[i];
        }
        for(i=0; i<b; i++)
        {
            cin>>bb[i];
        }
        std::sort(rr, rr+r, std::greater<int>());
        std::sort(gg, gg+g, std::greater<int>());
        std::sort(bb, bb+b, std::greater<int>());
        while(m--)
        {
            if((rr[0]>=gg[0]) && (rr[0]>=bb[0]))
            {
                for(i=0; i<r; i++)
                    rr[i] /= 2;
                std::sort(rr, rr+r, std::greater<int>());
            }
            else if((gg[0]>=rr[0]) && (gg[0]>=bb[0]))
            {
                for(i=0; i<g; i++)
                    gg[i] /= 2;
                std::sort(gg, gg+g, std::greater<int>());
            }
            else
            {
                for(i=0; i<b; i++)
                    bb[i] /= 2;
                std::sort(bb, bb+b, std::greater<int>());
            }
        }
        max=rr[0];
        if(bb[0]>max)
            max=bb[0];
        if(gg[0]>max)
            max=gg[0];
        cout<<max<<endl;
    }
    return 0;
}
