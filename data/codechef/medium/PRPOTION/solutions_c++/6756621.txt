#include<iostream>
#include<algorithm>
using namespace std;
int arr1[100],arr2[100],arr3[100];
int main()
{
	int t,l1=-1,l2=-1,l3=-1,i,r,g,b,m;
	cin>>t;
	while(t--)
	{
		cin>>r>>g>>b>>m;
		for(i=0;i<r;i++)
		{
			cin>>arr1[i];
			if(arr1[i]>l1)
			l1=arr1[i];
		}
		for(i=0;i<g;i++)
		{
			cin>>arr2[i];
			if(arr2[i]>l2)
			l2=arr2[i];
		}
		for(i=0;i<b;i++)
		{
			cin>>arr3[i];
			if(arr3[i]>l3)
			l3=arr3[i];
		}
		while(m--)
		{
			if(l1>=l2 && l1>=l3)
			l1=l1/2;
			else
			if(l2>=l3 && l2>=l1)
			l2=l2/2;
			else
			l3=l3/2;
		}
		if(l1>=l2&&l1>=l3)
		cout<<l1<<"\n";
		else
		if(l2>=l3&&l2>=l1)
		cout<<l2<<"\n";
		else
		cout<<l3<<"\n";
		l1=l2=l3=-1;
	}
	return 0;
}
