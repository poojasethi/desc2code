#include<bits/stdc++.h>
using namespace std;
long long MAX (long long a, long long b)
{
	if (a>b)
			return a;
	else
			return b;
}
int main()
{
	int t,R,G,B,M,i;
	scanf("%d",&t);
    while(t--)
    {
    	scanf("%d %d %d %d", &R, &G, &B, &M);
 
    	long long r[R], g[G], b[B];
    	long long RMax,BMax,GMax;
    	RMax=BMax=GMax=0;
 
    	for(i=0; i<R; i++)
    	{
    			scanf("%lld", &r[i]);
    			if(r[i]>RMax)
    						RMax=r[i];
    	}
 
    	for(i=0; i<G; i++)
    	{
    			scanf("%lld", &g[i]);
    			if(g[i]>GMax)
    						GMax=g[i];
    	}
    	for(i=0; i<B; i++)
		{
    			scanf("%lld", &b[i]);
    			if(b[i]>BMax)
    						BMax=b[i];
    	}
    	long long Max=INT_MAX;
 
    	while(M--)
    	{
    		Max = min(Max,MAX(RMax,MAX(BMax,GMax)));
 
    		if (MAX(RMax,MAX(BMax,GMax))==RMax)
    		{
    			RMax/=2;
 
    		}
       		else if (MAX(RMax,MAX(BMax,GMax))==GMax)
    		{
    			GMax/=2;
 
    		}
    		else
    		{
    			BMax/=2;
 
    		}
    	}
 
    	Max = min(Max,MAX(RMax,MAX(BMax,GMax)));
 
    	printf("%lld\n", Max);
 
    }
    return 0;
}
