#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int r,g,b,m;
		cin>>r>>g>>b>>m;
		vector<int>ar;
		vector<int>ag;
		vector<int>ab;
		int mx=0,my=0,mz=0;
		for(int i=0;i<r;i++)
		{
			int a;
			cin>>a;
			mx = mx < a ? a : mx;
			ar.push_back(a);
		}
		for(int i=0;i<g;i++)
		{
			int a;
			cin>>a;
			my = my < a ? a : my;
			ag.push_back(a);
		}
		for(int i=0;i<b;i++)
		{
			int a;
			cin>>a;
			mz = mz < a ? a : mz;
			ab.push_back(a);
		}
		vector<int>v(3);
		v[0]=mx;
		v[1]=my;
		v[2]=mz;
		for(int i=0;i<m;i++)
		{
			sort(v.begin(),v.end());
			int val = v[2]/2;
			v[2]=val;
		}
		sort(v.begin(),v.end());
		cout<<v[2]<<endl;
	}
	return 0;
}
