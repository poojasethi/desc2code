#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main() {
	int t;
	cin >> t;
	int R,G,B,M;
	long mr, mg, mb;
	while(t--) {
		cin >> R >> G >> B >> M;
		vector<long> vr, vg, vb;
		long temp;
		mr = mg = mb = 0;
		for(int i=0;i<R;i++) {
			cin >> temp;
			vr.push_back(temp);
			if(temp >= vr[mr]) {
				mr = i;
			}
		}
		for(int i=0;i<G;i++) {
			cin >> temp;
			vg.push_back(temp);
			if(temp >= vg[mg]) {
				mg = i;
			}
		}
		for(int i=0;i<B;i++) {
			cin >> temp;
			vb.push_back(temp);
			if(temp >= vb[mb]) {
				mb = i;
			}
		}
		for(int i=0;i<M;i++) {
			temp = max(max(vb[mb],vr[mr]), vg[mg]);
			if(temp == vr[mr]) {
				for(int j=0;j<vr.size();j++) {
					vr[j] = vr[j]/2;
				}
			}	
			else if(temp == vg[mg]) {
				for(int j=0;j<vg.size();j++) {
					vg[j] = vg[j]/2;
				}
			}
			else {
				for(int j=0;j<vb.size();j++) {
					vb[j] = vb[j]/2;
				}
			}
		}
		cout << max(max(vb[mb],vr[mr]), vg[mg]) << endl;
	}
	return 0;
}
