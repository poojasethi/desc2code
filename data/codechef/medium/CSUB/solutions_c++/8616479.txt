#include<bits/stdc++.h>
using namespace std;

int t,n;
char c;

int gcd(int a,int b){
	return((b==0)?a:gcd(b,a%b));
}

int main (){
	scanf("%d",&t);
	
	while(t--){
		scanf("%d\n",&n);
		long long int cnt =0;
		
		while(n--){
			scanf("%c",&c);
			if(c=='1')
				++cnt;
		}
		
		printf("%lld\n", cnt*(cnt+1)/2);
	}
	
	return 0;
}
