#include<bits/stdc++.h>
using namespace std;

int main(){
   long int t;
   scanf("%ld",&t);
   while(t--)
   { long long int countt=0;
       long int n,i;
       scanf("%ld",&n);
       char s[n+1];
       scanf("%s",s);
       for(i=0;i<=n;i++)
       {if(s[i]=='1')
        countt++;}
        printf("%lld\n",countt*(countt+1)/2);
   }
return 0;
}
