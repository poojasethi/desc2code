#include <bits/stdc++.h>

using namespace std;

int t, n;
string s;
long long ans = 0;

int main ()
{
    cin >> t;
    while(t--)
    {
        ans = 0;
        cin >> n >> s;
        for (int i = 0; i < n; i++)
        {
            if (s[i] == '1')
                ans++;
        }
        cout << ans * (ans + 1) / 2 << endl;
    }
    return 0;
}
