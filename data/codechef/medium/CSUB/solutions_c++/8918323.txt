#include<stdio.h>
int main()
{
	long long int t,i,n,j,count;
	char a[100005];	
	scanf("%lld",&t);
	while(t--)
	{
		count=0;
		scanf("%lld ",&n);
		scanf("%s",&a);
		for(i=0;i<n;i++)
		{
			if(a[i]=='1')
			{
				count++;
			}
		}
		count=(count*(count+1))/2;
		printf("%lld\n",count);
	}
	return 0;
}
