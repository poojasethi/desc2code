#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        long long int x,c=0;
        scanf("%d",&n);
        char s[n];
        cin>>s;
        for(int i=0;i<n;i++)
        {
            if(s[i]=='1')
                c++;
        }
        x=(c*(c+1))/2;
        printf("%lld\n",x);
    }
    return 0;
}
