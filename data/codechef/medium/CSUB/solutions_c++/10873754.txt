#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long int ll;
ll count_one(char str[],int n){
    ll ans=0;
    for(int i=0;i<n;i++){
        if(str[i]=='1') ans++;
    }
    return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    int n;
    scanf("%d",&n);
    char str[n+5];
    scanf("%s",&str);
    ll one=count_one(str,n);
    ll ans=(one*(one+1))/2;
    printf("%lld\n",ans);
    }
}
