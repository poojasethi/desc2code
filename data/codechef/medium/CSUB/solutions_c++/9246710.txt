#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        long long int n, count = 0;
        scanf("%lld", &n);
        char str[n + 1];
        scanf("%s", str);
        unsigned long long int ans;
        for(int i = 0; i < n; i++)
        {
            if(str[i] == '1')
            {
                count++;
            }
        }
        ans = count*(count + 1) / 2;
        printf("%llu\n", ans);
    }
    return 0;
}
