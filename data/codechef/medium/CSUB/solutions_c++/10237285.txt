#include <iostream>
using namespace std;
int main()
{
    int t,i,num;
    long long count;
    string str;
    cin>>t;
    while(t--)
    {   count=0;
        cin>>num>>str;
        for(i=0;str[i]!='\0';++i)
        {   if(str[i]=='1')
            ++count;
        }
        cout<<(count*(count+1))/2<<endl;
    }
    return 0;
}
