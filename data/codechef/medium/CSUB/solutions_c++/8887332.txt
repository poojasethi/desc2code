#include <bits/stdc++.h>
using namespace std;

int main() {
    int t,n,i;
    cin >> t;

    while(t--) {
        string s;
        int c = 0;
        cin >> n >> s;
        long long int ans=0;

        for(i = 0;i < n;i++) {
            if(s[i]=='1') {
                c++;
                ans += (long long)(c);
            }
        }
        cout << ans <<endl;
    }
    return 0;
}
