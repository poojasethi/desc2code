#include <iostream>
#include <cstdio>
#include <algorithm>
#define gc getchar_unlocked
#define pc putchar_unlocked
using namespace std;
inline unsigned long int fastread()
    {
      unsigned long int i=0;
      char c=gc();
      while(c<'0' || c>'9')
        c=gc();
      for(;c>='0' && c<='9';c=gc())
        i=(i<<3)+(i<<1)+c-'0';
      return i;
    }
 int main()
 {
     long int t;
     t=fastread();
     //scanf("%li",&t);
     while(t--)
     {
         long int n;
         n=fastread();
         //scanf("%li",&n);
         char a[n];
         scanf("%s",&a);

     }
