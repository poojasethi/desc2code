#include <iostream>
#include <bits/stdc++.h>


using namespace std;

typedef long long int ll;
int main()
{
	std::ios::sync_with_stdio(false);
    ll t;
    cin>>t;
    while(t--)
    {
    	ll n;
    	cin>>n;
    	string s;
    	cin>>s;
    	ll c=0,ans=0;
		for(int i=0;i<s.length();i++)
    	{
    		if(s[i]=='1') c++;
		}
		ans = c * (c+1) / 2;
		cout<<ans<<endl;
	}
	return 0;
}
