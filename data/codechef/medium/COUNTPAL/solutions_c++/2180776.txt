#include <cstdio>
#include <iostream>
using namespace std;

bool a[1005][1005];
int cnt[1005];

#define MOD 1000000007;

int main()
{
    string s;
    cin >> s;
    for(int i=0;i<s.length();i++)
    {
        a[i][i] = true;
    }
    for(int i=0;i<s.length()-1;i++)
    {
        a[i][i+1] = (s[i]==s[i+1]);
    }
    for(int i=2;i<s.length();i++)
    {
        for(int j=0;j+i<s.length();j++)
        {
            a[j][j+i] = a[j+1][j+i-1] && (s[j]==s[j+i]);
        }
    }
    /*for(int i=0;i<s.length();i++)
    {
        for(int j=0;j<s.length();j++)
        {
            cout << a[i][j] << " ";
        }
        cout << endl;
    }
    int sum=0;
    for(int i=0;i<1005;i++)
    {
        sum+=cnt[i];
    }
    cout << sum << endl;*/
    cnt[0] = 1;
    for(int i=1;i<s.length();i++)
    {
        for(int j=0;j<i;j++)
        {
            if(a[j+1][i])   cnt[i] = (cnt[i]+cnt[j])%MOD;
        }
        if(a[0][i]) cnt[i] = (cnt[i]+1)%MOD;
    }
    printf("%d\n",cnt[s.length()-1]);
    return 0;
}
