void go(int pos){
	v[pos].pb(pos+1);
	int l = pos - 1, r = pos + 1;
	while(l >= 0 && r < len){
		if(str[l] == str[r]) v[l].pb(r+1);
		else break;
		l--; r++;
	}
}
void go1(int pos){
	int l = pos - 1, r = pos;
	while(l >= 0 && r < len){
		if(str[l] == str[r]) v[l].pb(r + 1);
		else break;
		l--; r++;
	}
}

int memo[1005];
int dp(int pos){
	if(pos == len) return 1;
	int &ret = memo[pos];
	if(ret != -1) return ret;
	ret = 0;
	FOR(i,0,SIZE(v[pos])){
		int next = v[pos][i];
		ret = (ret + dp(next)) % MOD;
	}
	return ret;
}
