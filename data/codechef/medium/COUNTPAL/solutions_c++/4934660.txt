#include<iostream>
#define MOD 1000000007

using namespace std;

int main()
{
    char s[1005];
    int dp[1005];
    bool ispalin[1005][1005] = {false};
    int i, j;

    cin>>s;

    dp[0] = 1;
    ispalin[0][0] = true;

    for(i=1;s[i]!='\0'; i++)
    {
        ispalin[i][i] = true;
        dp[i] = dp[i-1];

        for(j=0; j<i; j++)
        {
            if(s[j]==s[i] && (j==i-1 || ispalin[j+1][i-1]))
            {
                ispalin[j][i] = true;

                if(j==0)
                    dp[i]++;
                else
                    dp[i] += dp[j-1];

                dp[i] = dp[i]%MOD;
            }
        }
    }

    cout<<dp[i-1]<<"\n";

    return 0;
}
