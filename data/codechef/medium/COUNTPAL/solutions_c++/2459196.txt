#include<iostream>
#include<string.h>
using namespace std;
main()
{
char s[1001];
long n=1000000007;
bool ispal[1001][1001];
long long dp[1001];
cin>>s;
int l=strlen(s);
s[l]='1';
for(int i=l-1;i>=0;--i)
{
    ispal[i][i]=true;
    if(s[i]==s[i+1]) ispal[i][i+1]=true;
    for(int j=i+2;j<l;++j)
    {
        if(s[i]==s[j] && ispal[i+1][j-1])
        {
            ispal[i][j]=true;
        }

    }


}

dp[0]=dp[1]=1;

for(int i=1;i<l;++i)
{

    dp[i+1]=dp[i];

    for(int j=i-1;j>=0;--j)
    {
        if(ispal[j][i]) dp[i+1]+=dp[j];
    }

dp[i+1]=dp[i+1]%1000000007;

}

cout<<dp[l]<<endl;

}
