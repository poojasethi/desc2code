#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main() {
    /* Enter your code here. Read input from STDIN. Print output to STDOUT */   
    int i,j,k,l,c;
    string s;
    cin>>s;
    
    
    l = s.length();
    bool dp[l][l];
    long pal[l];
    
    for(i=0; i<l ; i++)
        for(j=0;j<l;j++)
            dp[i][j] = 0;
        
     
    for(i=0 ; i<l ; i++)
        dp[i][i] = 1;
        
    for(i=0 ; i<l ; i++)
        pal[i] = 0;    
    
    for(i=0 ; i<l-1;i++)
        if(s[i] == s[i+1])
            dp[i][i+1] = 1;
        
        
  
   
    for(k = 2 ; k<l ; k++)
    {
         for(i=0 ; i<l-2 ; i++)
        {
            if(i+k < l && dp[i+1][i+k-1] == 1 && s[i] == s[i+k])
            dp[i][i+k] = 1;
        }
    }
        
    pal[0] = 1;
    
    for(i = 1 ; i< l ; i++)
    {
        for(j=0 ; j<=i ; j++)
        {
            if(dp[j][i] == 1)
            {
                if(j==0) pal[i]++;
                else pal[i]= (pal[i]+ pal[j-1])%1000000007;
            }
        }    
        
    }

    
    cout<<pal[l-1]<<endl;
    
    return 0;
    
    
    
    
    
    
    
}
