if(isPalMem[b][e]!=-1) return isPalMem[b][e];
if(s[b]!=s[e-1]) return isPalMem[b][e]=0;

}
 
int F(int n) {
if(n==0) return 1;
if(FMem[n]!=-1) return FMem[n];

for(int i=0; i<n; i++)

return FMem[n]=res%MOD;
}
