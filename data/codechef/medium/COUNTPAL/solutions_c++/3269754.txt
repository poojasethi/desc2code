#include <iostream>
#include <cstring>
#include <string>
using namespace std;
#define MOD 1000000007
string s;
int FAr[1005];
int isPalAr[1005][1005];
bool isPal(int st,int en);
int F(int n);
int main()
{

    cin>>s;
    for(int i=0;i<1005;i++)
        FAr[i]=-1;
    for(int i=0;i<1005;i++)
        for(int j=0;j<1005;j++)
        isPalAr[i][j]=-1;
        cout<<F(s.length())<<endl;

}

int F(int n)
{
    if(n==0)return 1;
    if(FAr[n]!=-1) return FAr[n];
    int fin=0;
    for(int j=0;j<n;j++)
    {
         if(isPal(j,n)) fin=(fin+F(j))%MOD;

    }
             FAr[n]=fin%MOD;
             return F(n);
}

bool isPal(int st,int en)
{
    if(st==en||st+1==en)
        return true;
    if(isPalAr[st][en]!=-1) return isPalAr[st][en];
    if(s[st]!=s[en-1]) return false;
    return isPalAr[st][en]=isPal(st+1,en-1);


}
