#include <stdio.h>
#include <string.h>
#define MOD 1000000007
#define MX_N 1010

char str[MX_N];
int opt[MX_N];
int palin[MX_N][MX_N];
int arr[MX_N];
int l;

int main() {
    int i,j;
    scanf("%s",str);
    l = strlen(str);
    opt[l] = 1;
    for(i=l-1;i>=0;i--) {
        for(j=0;j<l;j++) {
            if(i >= j) {
                palin[i][j] = 1;
            }else {
                palin[i][j] = (str[i] == str[j] && palin[i+1][j-1]);
            }
            if(palin[i][j]) {
                opt[i] = (opt[i] + opt[j+1])%MOD;
            }
        }
    }
    printf("%d\n",opt[0]);
    return 0;
}
