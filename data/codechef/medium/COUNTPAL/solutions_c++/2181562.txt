#include <cstdio>
#include <cstring>
#define m 1000000007
char str[1002];
int len;

typedef int ll;

ll dp[1002];

ll palin[1002][1002];

bool isPalin(int start, int end){
	if (palin[start][end] == 1)
		return true;
	if (palin[start][end] == 2)
		return false;

	if (end-start<1){
		return true;
	}

	if (str[start] == str[end]){
		if (isPalin(start+1, end-1)){
			palin[start][end] = 1;
			return true;
		}
	}
	palin[start][end] = 2;
	return false;
}

ll cp(int start){
	if (start == len-1 || start == len){
		return 1;
	}
	if (dp[start] != -1){
		return dp[start];
	}
	dp[start]=0;
	for (int i = start; i<len; i++){
		if (isPalin(start, i)){
			dp[start] = (dp[start] + cp(i+1)) % m;
		}
	}
	return dp[start];
}

int main(){
	scanf("%s", str);
	len = strlen(str);

	for (int i=0; i< 1002; i++){
		dp[i] = -1;
	}

	printf("%d", cp(0));
	return 0;
}
