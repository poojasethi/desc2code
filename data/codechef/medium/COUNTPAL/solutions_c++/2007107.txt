#include <iostream>
#include <string>
#include <cstdio>
#include <cstring>
#define M 1000000007

using namespace std;

static bool a[1002][1002];

void pre(string s) {
     int i,j;
     for ( i = 0; i < s.size(); i++ ) {
         a[i][i] = 1;
     }
     for ( i = 0; i < s.size()-1; i++ ) {
         if ( s[i] == s[i+1] ) a[i][i+1] = 1;
     }
     
     for( i = 3; i <= s.size(); i++ ) {
          for ( j = 0; j <= s.size()-i; j++ ) {
              if ( a[j+1][i+j-2] == 1 && s[j] == s[i+j-1] ) {
                 a[j][i+j-1] = 1;
              }
          }
     }
     return;
}

int main()
{
    string s;
    
    int i,j;
    long long dp[1002];
    
    memset(dp, 0, sizeof(dp));
    
    dp[0] = 1;
    
    cin >> s;
    
    pre(s);
    
    for ( i = 1; i <= s.size(); i++ ) {
        for ( j = 1; j <= i; j++ ) {
            if ( a[j-1][i-1] ) {
               dp[i] += (dp[j-1])%M;
               dp[i] = dp[i]%M;
               if ( dp[i] < 0 ) dp[i] += M;
            }
        }
    }
    
    
    printf("%lld\n", dp[(int)(s.size())]);
    
    return 0;
}
