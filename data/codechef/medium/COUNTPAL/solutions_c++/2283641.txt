#include<stdio.h>
#include<string.h>
#define MOD 1000000007
using namespace std;
int main()
{
	int i,j,k,l,n;
	bool a[1002][1002];
	char s[1005];
	int f[1005];
	scanf("%s",s+1);
	f[0]=1;
	n=strlen(s+1);
	for(i=1;i<=n;i++)
		a[i][i]=a[i][i-1]=true;
	for(l=1;l<n;l++)
		for(i=1;i+l<=n;i++)
			a[i][i+l]=(s[i]==s[i+l]&&a[i+1][i+l-1]);
	for(i=1;i<=n;i++)
	{
		f[i]=0;
		for(j=1;j<=i;j++)
			if(a[j][i])
			{
				f[i]+=f[j-1];
				if(f[i]>=MOD)
					f[i]-=MOD;
			}
	}
	printf("%d",f[n]);
	return 0;
}
