#include<iostream>
#include<string.h>
#define MOD 1000000007
using namespace std;
int ans[1001];
int arr[1001][1001];
int main()
{
	memset(arr,0,sizeof(arr));
	memset(ans,0,sizeof(ans));
	char str[1001];
	int i,l,j;
	cin>>str;
	l = strlen(str);
	ans[0] =1 ;
	for(i=0;i<l;i++)
	{
		for(j=1;j<= i+1 ;j++)
		{
			if(str[i]==str[i+1-j]&&(j<=2 || arr[i-1][j-2]))
			{
				arr[i][j] = 1;
				ans[i+1] = (ans[i+1]+ans[i+1-j])%MOD;
			}
		}
	}
	cout<<ans[l]<<endl;
	return 0;
}
