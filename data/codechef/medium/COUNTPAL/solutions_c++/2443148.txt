#include <iostream>
#include <string>
#include <vector>
#include <cstdio>
#include <cstdlib>

using namespace std;
int dp[1000][1000];
int ans[1010]={0};

void printPalindromes(string& str)
{
    int n = str.size();
    for (int i=0; i < n; i++) {
        dp[i][i] = 1;
    }

    for (int L=2; L<=n; L++)
    {
        for (int i=0; i<n-L+1; i++)
        {
            int j = i+L-1; // Set ending index

            if (L == 2)
                dp[i][j] = (str[i] == str[j]);
            else
                dp[i][j] = (str[i] == str[j]) && dp[i+1][j-1];
        }
    }
	ans[n]=1;
	for (int i=n-1; i>=0; i--) {
        ans[i]=0;
        for (int j=i; j<n; j++) {
            if (dp[i][j]) ans[i] = (ans[i]+ans[j+1])%1000000007;
        }
	}
	cout << ans[0] << endl;
}

int main()
{
    string s;
    cin >> s;
    printPalindromes(s);

    return 0;
}
