// Really nice DP. Takes quite a time to be observed. Will remember this.
#include<iostream>
#include<cstdio>
#include<cstring>
#define MOD 1000000007LL
using namespace std;
bool Palin[1005][1005]={};
char s[1005];
long long int ans[1005];
int main()
{
    ans[0] = 1LL;
    scanf("%s",s);
    int n = strlen(s);

    for(int i=0;i<n;++i)
    {
        Palin[i][i]=1;
        if(i+1<n && s[i]==s[i+1])
        {
            Palin[i][i+1] = 1;
        }
    }

    for(int i=0;i<n;++i)
    {
        for(int j=0;i+j+2<n;++j)
        {
            Palin[j][i+j+2] = (Palin[j+1][i+j+1] && s[j]==s[i+j+2]);
        }
    }

    for(int i=1;i<n;++i)
    {
        ans[i] = ans[i-1];
        for(int j=0;j<i;++j)
        {
            if(Palin[j][i])
            {
                if(j>=1)
                ans[i] += ans[j-1];
                else ans[i] += 1;
                ans[i] %= MOD;
            }
        }
    }
    printf("%lld\n",ans[n-1]);
    return 0;
}
