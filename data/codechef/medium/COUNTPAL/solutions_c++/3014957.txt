typedef long long LL;

#define reset(a,b) memset(a, b, sizeof a)

LL dp[1003],pal[1003][1003];
string s;


int cekPal(int i , int j)
{
     if (i>=j) return 1;
     if (s[i]!=s[j]) return pal[i][j] = 0;
     if (pal[i][j]!=-1) return pal[i][j];
     
     return pal[i][j] = cekPal(i+1,j-1);
}

LL solve(int idx)
{
             if (idx==s.length()) return 1;
             if (dp[idx]!=-1) return dp[idx];
             
             LL ret = 0;
             
             for (int i = idx ; i<s.length(); i++)
             {
                 if (cekPal(idx,i)==1) 
                 {
                                         ret+=solve(i+1);
                                         ret%=MOD;
                 }
             }
             
             return dp[idx] = ret;
}

int main()
{
    cin>>s;
    reset(dp,-1);
    reset(pal,-1);
    
    cout<<solve(0)<<endl;

    return 0;
}
