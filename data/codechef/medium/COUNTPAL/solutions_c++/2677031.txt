#include <iostream>
#include <string>
using namespace std;
typedef long long LL;
#define MOD 1000000007

int main() {
  string str;
  cin >> str;
  int l = str.length();
  LL **countpal = new LL*[l];
  for(int i=0;i<l;++i) countpal[i] = new LL[l];
  for(int i=0;i<l;++i) countpal[i][i] = 1;
  
  for(int i=2;i<=l;++i) {
    for(int j=0;(i+j)<=l;++j) {
      countpal[j][j+i-1] = (LL)(str[j]==str[j+i-1]) && (i==2 || countpal[j+1][j+i-2]);
    }
  }
  /*
  for(int i=2;i<=l;++i) {
    for(int j=0;(i+j)<=l;++j) {
      for(int k=0;k<i;++k) {
	countpal[j][j+i-1] = (countpal[j][j+i-1]+(countpal[j][j+k]*countpal[j+k+1][i+j-1])%MOD)%MOD;
      }
    }
  }*/
  
  for(int i=1;i<l;++i) {
    for(int j=0;j<i;++j) {
      countpal[0][i] = (countpal[0][i]+(countpal[0][j]*countpal[j+1][i])%MOD)%MOD;
    }
  }
  
  cout << countpal[0][l-1] << endl;
  return 0;
}
