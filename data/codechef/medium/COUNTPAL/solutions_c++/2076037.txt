#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
#define M 1000000007

using namespace std;

static bool dp[1002][1002];
static long long dp1[1002];

void func(char *s) 
{
     memset(dp, 0, sizeof(dp));
     memset(dp1, 0, sizeof(dp1));
     
     int i,j,k=strlen(s);
     for ( i = 0; i < k; i++ ) {
         dp[i][i] = 1;
     }
     for ( i = 0; i < k-1; i++ ) {
         if ( s[i] == s[i+1] ) {
            dp[i][i+1] = 1;
         }
     }
     for ( i = 3; i <= k; i++ ) {
         for ( j = 0; j <= k-i; j++ ) {
             if ( s[j] == s[i+j-1] && dp[j+1][i+j-2] == 1 ) {
                dp[j][i+j-1] = 1;
             }
         }
     }
     
     dp1[0] = 1;
     dp1[1] = 1;
     
     for ( i = 2; i <= k; i++ ) {
         for ( j = 0; j <= i; j++ ) {
             if ( dp[j][i-1] == 1 ) {
                dp1[i] += (dp1[j]%M);
                dp1[i] = dp1[i]%M;
                if ( dp1[i] < 0 ) dp1[i] += M;
             }
         }
     }
         
     printf("%lld\n", dp1[k]);
     return;
}     
int main()
{   
    char s[1002];
    scanf("%s", s);
    func(s);
    return 0;
}
