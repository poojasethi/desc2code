#include<iostream>
#define M 1000000007
using namespace std;
int main()
{
    bool is[1001][1001]={0};
    int dp[1001]={0},i,j;
    dp[0]=1;
    is[0][0]=1;
    string s;
    cin>>s;
    for(i=1;i<s.size();i++)
    {
                for(j=0;j<=i;j++)
                {
                                if(s[j]==s[i] &&(is[j+1][i-1] || j==i || i-j==1))
                                {
                                                      is[j][i]=1;
                                                      if(j==0) dp[i]=(dp[i]+1)%M;  
                                                      else dp[i]=(dp[i]+dp[j-1])%M;
                                                      //cout<<j<<i<<dp[i]<<endl;
                                }
                }
                
    }
    cout<<dp[s.size()-1]<<endl;
    //cin>>s;
}
