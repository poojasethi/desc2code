#include <iostream>
#include<stdio.h>


int main(){
    long long int t, n, ans;
    scanf("%lld",&t);
    while(t--){
        scanf("%lld",&n);
        ans = n*(n+1);
        ans /= 2;
        ans += n;
        printf("%lld\n",ans);
    }
    return 0;
}
