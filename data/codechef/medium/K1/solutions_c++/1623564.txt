#include <iostream>
#include <cstdio>
#include <cmath>
using namespace std;

const int N = 2005;
const double eps = 1e-9;
int t, a, b, c, n;
int x[N], y[N];
double dist;

double sq(double a) { return a*a; }

bool eq(double a, double b) {
    if (abs(a-b)<eps)
        return true;
    return false;
}

double search(double l, double r) {
    if (eq(l,r))
        return l;
    double x1=l+0.382*(r-l);
    double x2=r-0.382*(r-l);
    double y1 = (-c-a*x1);
    double y2 = (-c-a*x2);
    y1/=b;
    y2/=b;
    double d1 = 0;
    double d2 = 0;
    for (int i=0; i<n; i++) {
        d1 += sqrt(sq(x[i]-x1)+sq(y[i]-y1));
        d2 += sqrt(sq(x[i]-x2)+sq(y[i]-y2));
    }
    if (d1<d2)
        return search(l, x2);
    else
        return search(x1, r);
}

int main()
{
    scanf("%d", &t);
    while (t--) {
        scanf("%d", &n);
        scanf("%d%d%d", &a, &b, &c);
        for (int i=0; i<n; i++) scanf("%d%d", &x[i], &y[i]);
        dist = n*20000;
        double xp = search(-5000, 5000);
        double yp = (-c-a*xp);
        yp/=b;
        double ans = 0;
        for (int i=0; i<n; i++) {
            ans += sqrt(sq(x[i]-xp)+sq(y[i]-yp));
        }
        printf("%.6lf\n", ans);
    }
    return 0;
}
