    double dd(double x,double y,double x2,double y2)
    {
    double ans=sqrt((x-x2)*(x-x2)+(y-y2)*(y-y2));
    return ans;
    }
     
    double dist(int a,int b,int c,double x,vector< pair<int,int> > v)
    {
    double y=(-c-a*x)/(double)b;
    //cout<<x<<" "<<y<<endl;
    double rt=0;
    for(int i=0;i<v.size();i++){
    rt+=dd(x,y,v[i].F,v[i].S);
    }
    return rt;
    }

    vector< pair<int,int> > v;
    for(int i=0;i<n;i++){
    int x,y;
    cin>>x>>y;
    v.PB(MP(x,y));
    }
    double lb=-5000,ub=5000,mid,midl,midh;
    while(fabs(ub-lb)>1e-8 || fabs(lb-ub)>1e-8){
    //cout<<lb<<" "<<ub<<endl;
    mid=(lb+ub)/2;
    midl=(lb+mid)/2;
    midh=(mid+ub)/2;
    double t1=dist(a,b,c,midl,v);
    double t2=dist(a,b,c,midh,v);
    if(t1<t2){
    ub=mid;
    }

    lb=mid;
