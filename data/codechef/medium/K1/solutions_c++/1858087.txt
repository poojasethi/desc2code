#include <cstdio>
#include <cmath>
int T, N, A, B, C, X[2000], Y[2000];

double f(double x)
{
    double y = (-C - A*x)/(B);
    double res = 0;
    for (int i = 0; i < N; i++)
        res += hypot(X[i]-x, Y[i]-y);
    return res;
}

int main()
{
    scanf("%d", &T);
    while (T--)
    {
        scanf("%d", &N);
        scanf("%d%d%d", &A, &B, &C);
        for (int i = 0; i < N; i++)
            scanf("%d%d", &X[i], &Y[i]);
        double lo = -50000, hi = 50000;
        while (hi-lo > 1e-7)
        {
            double l = (lo*2+hi)/3, r = (lo+hi*2)/3;
            if (f(l) > f(r))
                lo = l;
            else
                hi = r;
        }
        printf("%.6lf\n", f(lo));
    }
}
