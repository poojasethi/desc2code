#include<stdio.h>
#include<math.h>
int N;
double A,B,C;
double v1[2500],v2[2500];
double max(double a,double b)
{
return a<b?b:a;
}
double min(double a,double b)
{
return a<b?a:b;
}
double distance(double x, double x0, double y, double y0)
{
return sqrt(pow(x - x0, 2) + pow(y - y0, 2));
}

double dist_from_warehouses(double x1)
{
  double y1=-(A*x1+C)/B;
  double sum=0;
  for(int i=0;i<N;i++)
  {
  sum+=distance(x1,v1[i],y1,v2[i]);
  }
  return sum;

}
double val(double a,double b)
{
double aa,bb;
  while((b-a)>0.00001)
  {
   aa=(2*a+b)/3;
   bb=(a+2*b)/3;
   if (dist_from_warehouses(aa) > dist_from_warehouses(bb))
            a =aa;
        else
            b= bb;
   }
   return (a+b)/2;
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
    scanf("%d",&N);
    scanf("%lf %lf %lf",&A,&B,&C);
     for(int i=0;i<N;i++)
      scanf("%lf %lf",&v1[i],&v2[i]);
      double maxx=-1000000,minn=1000000;
    for(int i=0;i<N;i++)
    {
     maxx=max(maxx,v1[i]);
     minn=min(minn,v2[i]);
    }
 
      printf("%.6lf\n", (double)dist_from_warehouses( val(minn, maxx)));
    }
    return 0;
}
