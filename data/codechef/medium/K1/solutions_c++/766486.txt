#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int T,N,A,B,C,i;
double l,r;
double x[2000],y[2000];
double fd,td;


double function(double px)
{
    double py,res;

    int j;
    res=0.0;
    py=(double)(-C-A*px)/(1.0*B);

    //
    for(j=0;j<N;j++)
    {
        res=res+sqrt(((px-x[j])*(px-x[j]))+((py-y[j])*(py-y[j])));
    }
return res;
}
int main()
{
    scanf("%d",&T);
    while(T>0)
    {
    scanf("%d",&N);
    scanf("%d %d %d",&A,&B,&C);
    for(i=0;i<N;i++)
    {
        scanf("%lf %lf",&x[i],&y[i]);
    }

    l=-5000;
    r=5000;

    while(r-l>0.0000001)
    {
    fd=(double)(r+2*l)/3.0;
    td=(double)(2*r+l)/3.0;

    if(function(fd)>function(td))
        l=fd;
    else
        r=td;
    }
    printf("%0.6lf\n",function(l));
    T--;
    }
    return 0;
}
