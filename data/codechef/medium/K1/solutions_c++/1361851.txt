#include<stdio.h>
#include<iostream>
#include<math.h>
#include<algorithm>

using namespace std;


#define MAXN 2009

double alpha,beta;
int N;
#define mp make_pair
#define pii pair<int,int>
#define X first
#define Y second
#define limit_x 5001.0

pii xy[MAXN];

void Read() {
	
	scanf("%d",&N);
	int a,b,c;
	scanf("%d%d%d",&a,&b,&c);
	
	alpha=(double)-a/b;
	beta=(double)-c/b; 
	
	for (int i=0;i<N;i++) {
		int x,y;
		scanf("%d%d",&x,&y);
		xy[i]=mp(x,y);
	}
	
	return ;
}
		

double get_y(double x) {
	return (double)alpha*x+beta;
}

double dist(double x1,double y1,double x2,double y2) {
	double ret=(double) sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
	return ret;
}

double f(double x) {
	double y=get_y(x);
	
	double ret=0.0;
	
	for (int i=0;i<N;i++)
		ret+=(double)dist(x,y,xy[i].X,xy[i].Y);
		
	return ret;
}

void Ternary() {
	
	double low=-limit_x,up=limit_x;
	
	while ( fabs ( low - up ) > 1e-7 ) {
		double mid=(2*low+up)/3.0;
		double mid1=(low+2*up)/3.0;
		
		if (f(mid) > f(mid1) ) {
			low=mid;
		}
		else 
			up=mid1;
		}
		
		printf("%.6lf\n",f(low));
		
	return;
}


int main () {
	//freopen("plant.in","r",stdin);
	//freopen("plant.out","w",stdout);
	int t;
	scanf("%d",&t);
	for (int i=0;i<t;i++) {
		Read();
		Ternary();
	}
	return 0;
}
