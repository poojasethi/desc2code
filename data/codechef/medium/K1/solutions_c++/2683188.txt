double dist(pair<double,double> a,pair<double,double> b)
{
    return sqrt((a.first-b.first)*(a.first-b.first)+(a.second-b.second)*(a.second-b.second));
}

double ternary(double lt,double rt)
{
    int iter=64;
    double ltt,rtt;
    while(iter--)
    {
        ltt=(2.0*lt+rt)/3.0;
        rtt=(lt+2.0*rt)/3.0;
        if(cost(ltt)<cost(rtt))
            rt=rtt;
        else
            lt=ltt;
    }
    return (lt+rt)/2.0;
}
