#include <bits/stdc++.h>
using namespace std;
#define lf long double
#define eps 1e-8
int X[5004],Y[5004];
double dist(double a,double b,double c,double d)
{
    return sqrt((a-c)*(a-c)+(b-d)*(b-d));
}
int main()
{
    double ans1,ans2,l1,l2,y1,y2,right,left;
    int a,b,c,n,t;
    scanf("%d",&t);
    while(t>0)
    {
        t--;
        scanf("%d",&n);
        scanf("%d %d %d",&a,&b,&c);
        for(int i=0;i<n;i++)
        {
            scanf("%d %d",&X[i],&Y[i]);
        }
        left=-1000000;
        right=1000000;
        while((right-left)>eps){
            l1=(left*2+right)/3.0;
            l2=(left+2*right)/3.0;
            y1=(-c-a*l1)/b;
            y2=(-c-a*l2)/b;
            ans1=ans2=0;
            for(int j=0;j<n;j++)
            {
                ans1=ans1+dist(l1,y1,(double)X[j],(double)Y[j]);
                ans2=ans2+dist(l2,y2,(double)X[j],(double)Y[j]);
            }
            if(ans1>ans2)
            {
                left=l1;
            }
            else{
                right=l2;
            }
      }
       printf("%.7lf\n",ans1);
    }
}
