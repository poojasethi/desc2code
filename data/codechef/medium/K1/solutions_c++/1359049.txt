#include <cstdio>
#include <cmath>

#define MAXN 2005
#define INF 10000

int N;
double x[MAXN], y[MAXN], A, B, C;

void Read() {
	int i;

	scanf ("%d %lf %lf %lf", &N, &A, &B, &C);
	for ( i=1; i<=N; ++i ) {
		scanf ("%lf %lf", &x[i], &y[i]);
	}
}

double sq( double a ) {
	return a*a;
}

double Dist ( double xa, double ya, double xb, double yb ) {
	return sqrt ( sq(xa-xb) + sq(ya-yb) );
}

double Cost ( double coord ) {
	double xi, yi, cost=0.0;
	int i;

	if ( B==0 ) {
		xi = -C/A;
		yi = coord;
	}
	else {
		xi = coord;
		yi = -(A*xi+C)/B;
	}

	for ( i=1; i<=N; ++i ) {
		cost += Dist(xi,yi,x[i],y[i]);
	}

	return cost;
}

void Ternary() {
	double lo, hi, lmid, hmid;
	int i;

	lo = -INF;
	hi = INF;
	for ( i=1; i<=60; ++i ) {
		lmid = (2*lo+hi) / 3.0;
		hmid = (lo+2*hi) / 3.0;
		if ( Cost(lmid) > Cost(hmid) ) {
			lo = lmid;
		}
		else {
			hi = hmid;
		}
	}

	printf ("%lf\n", Cost(lo) );
}

int main ( void ) {
	int t, T;

	scanf ("%d", &T);
	for ( t=1; t<=T; ++t ) {
		Read();
		Ternary();
	}
}
