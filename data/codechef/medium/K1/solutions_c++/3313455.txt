#include<iostream>
#include<cmath>
#include<cstdio>

using namespace std;

struct node {
 double x;
 double y;

} plant[2000];


int A,B,C,N,t;

double inline dist(double x){
  double sum =0;
double y = (-C - A*x)/B;
for(int i = 0; i<N; i++)
     sum += sqrt(   (x-plant[i].x)*(x-plant[i].x) +  (y-plant[i].y)*(y-plant[i].y)   );
return sum;
}



 int main()
{   scanf("%d", &t);
    while(t--){
    cin>>N>>A>>B>>C;
    for(int i =0; i<N; i++)
        cin>>plant[i].x>>plant[i].y;
    double min = plant[0].x,max=plant[0].x;

    for(int i =0; i<N;i++)
    {


        if(plant[i].x >= max)
                max =plant[i].x;
        if(plant[i].x <= min)
            min = plant[i].x;

    }

        double h1, h2;
		for (int i=0; i<100; i++){
			h1 = (min*2+max)/3;
			h2 = (min+2*max)/3;
			if (dist(h1) < dist(h2))
				max = h2;
			 else
				min = h1;

		}


        printf("%0.6lf\n", dist((min+max)/2));


}
}
