#include <stdio.h>
#include <algorithm>
 
 
#define MOD 1000000007
 
 
 
long long int A[100005];
long long int a[105];
long long int an[105];
long long int c[105];
long long int b_coeff[105][105];
 
int main(int argc, const char * argv[]) {
 
  
  
  int i=0;
  int j=0;
 
  for(i=0;i<105;i++){
    b_coeff[i][0]=1;
    b_coeff[i][i]=1;
  }
  for(i=1;i<105;i++){
    for(j=1;j<i;j++){
      b_coeff[i][j]=(b_coeff[i-1][j-1]+b_coeff[i-1][j])%MOD;
    }
  }
  int T;
 
  scanf("%d",&T);
  while(T--){
    int n,k;
    scanf("%d",&n);
    scanf("%d",&k);
    long long int res=0;
    for(i=0;i<n;i++){
      scanf("%lld",&A[i]);
    }
    for(i=0;i<=k;i++)
      c[i]=0;
    std::sort(A,A+n);
    for(i=0;i<n;i++){
      a[0]=1;
      an[0]=1;
      for(j=1;j<=k;j++){
	a[j]=(a[j-1]*A[i])%MOD;
	
	an[j]=(((an[j-1]*(A[i])*(-1))%MOD)+MOD)%MOD;
	
      }
      for(j=0;j<=k;j++){
	
	
	res+=((b_coeff[k][j]*c[j])%MOD)*a[j];
	res = res % MOD;
 
	c[j]+=an[k-j];
	c[j]=c[j]%MOD;
      }
    }
    res = (res*2) % MOD;
    printf("%lld\n",res);
  }
  return 0;
}
