#include<stdio.h>
#include<algorithm>
using namespace std;
long long int a[100005];
long long int mod=1000000007;
long long int powa[105],powna[105];
long long int c[105],comb[105][105];
int main()
{
//	freopen("in.txt","r",stdin);
	int t,n,k,i,j;
	long long int ans=0;
	
	for(i=0;i<=100;i++)
	{
		comb[i][0]=1;
		comb[i][i]=1;
		comb[i][1]=i;
	}
	for(i=3;i<=100;i++)
		for(j=2;j<i;j++)
			comb[i][j]=(comb[i-1][j]+comb[i-1][j-1])%mod;

	scanf("%d",&t);
	while(t--)
	{
		ans=0;
		scanf("%d %d",&n,&k);
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);

//		for(i=0;i<n;i++)
//			printf("%d ",a[i]);
//		printf("\n");
		for(j=0;j<=k;j++)
			c[j]=0;

		for(i=0;i<n;i++)
		{
			powa[0]=1;
			powna[0]=1;
			for(j=1;j<=k;j++)
			{
				powa[j]=(powa[j-1]*a[i])%mod;
				powna[j]=(((powna[j-1]*(a[i])*(-1))%mod)+mod)%mod;
			}
			for(j=0;j<=k;j++)
			{
				ans=(ans+((((comb[k][j]*c[j])%mod)*powa[j])%mod))%mod;
//				printf("%lld ",ans);
				c[j]=(c[j]+powna[k-j])%mod;
			}
			
//			for(j=0;j<=k;j++)
//				printf("%lld ",powna[j]);
//			printf("\n");

//			for(j=0;j<=k;j++)
//				printf("%lld ",c[j]);
//			printf("\n");
		}
		printf("%lld\n",(2*ans)%mod);
	}
	
	return 0;
}
