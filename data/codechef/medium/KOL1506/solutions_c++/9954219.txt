const int maxx = 1e5 + 9;
const int kmaxx = 100 + 9;

int T, N, k;
const ll mod = 1e9 + 7;
ll A[maxx];
ll nCr[kmaxx][kmaxx];
ll C[kmaxx];
ll p[maxx];

int main(){
#ifndef ONLINE_JUDGE
    const clock_t begin_time = clock();
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
   	fastIO
    
    nCr[0][0] = 1;
    for (int i = 1; i < kmaxx; i++) {
        for (int j = 0; j <= i; j++) {
            if (i == j || j == 0) nCr[i][j] = 1;
            else {
                nCr[i][j] = nCr[i - 1][j] + nCr[i - 1][j - 1];
                if (nCr[i][j] >= mod) nCr[i][j] -= mod;
            }
            
        }
    }
    
    
    
    cin >> T;
    
    while (T--) {
        memset(C, 0, sizeof(C));
        cin >> N >> k;
        rep(i, N) cin >> A[i];
        
        sort(A, A + N);
        
        ll ans = 0;
        
        for (int i = 0; i < N; i++) {
            
            p[0] = 1;
            for (int j = 1; j <= k; j++) {
                p[j] = (p[j - 1] * A[i]) % mod;
            }
            
            for (int a = 0; a <= k; a++) {
                ll temp = nCr[k][a];
                temp *= C[a];
                temp %= mod;
                temp *= p[a];
                temp %= mod;
                ans += temp;
                ans %= mod;
            }
            
            for (int a = 0; a <= k; a++) {
                
                ll pw = p[k - a];
                pw %= mod;
                if ((k - a) & 1)
                    pw *= -1;
                C[a] += pw;
                if (C[a] < 0) C[a] += mod;
                C[a] %= mod;
                    
            }
        }
        ans *= 2;
        ans %= mod;
        cout << ans << endl;
    }

    
    
#ifndef ONLINE_JUDGE
   	cout << endl;
    cout << "Time : ";
    cout << float( clock () - begin_time ) / CLOCKS_PER_SEC << endl;
#endif
    
    return 0;
}
