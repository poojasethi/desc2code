#include <iostream>
#include <cstdlib>
#include <stack>
using namespace std;

int main() {
	// your code goes here
	long int a, b;
	cin >> a >> b;
	long int ans = a - b;
	int temp1 = ans;
	stack <int> mystack;

	while(ans > 0){
		mystack.push(ans%10);
		ans = ans/10;
	}
	int flag = 0;
	while( !mystack.empty()){
		if( flag == 0){
			while(1){
				int temp = rand()%10;
				if(temp != 0 && temp != temp1%10){
					cout << temp;
					flag = 1;
					break;
				}
			}
		}
		else{
			cout << mystack.top();
		}
		mystack.pop();
	}
	return 0;
}
