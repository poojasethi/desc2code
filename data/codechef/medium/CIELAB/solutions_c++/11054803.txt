#include <iostream>
#include <stdio.h>
 
using namespace std;
 
int main()
{
    long int a, b;
    cin >> a;
    cin >> b;
    long int corans = a-b;
    if(corans==1){
    	cout << 2;
    	return 0;
    }
    long int lastdigit = corans%10;
    long int fin;
    if(lastdigit==0){
        fin = corans - lastdigit + 1;
    }
    else{
        fin = corans - lastdigit + lastdigit/2;
    }
    cout << fin;
 
    return 0;
}
