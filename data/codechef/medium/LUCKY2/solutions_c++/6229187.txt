int k;
ll dp[1010][800];

ll check(int x) {
	if(x==0)return 0;
	int d=0;
	while(x!=0) {
		d=x%10;
		if(d!=4&&d!=7) return 0LL;
		x/=10;
	}
	return 1LL;
}
 
ll solve(int idx,int cnt) {
	if(idx==k) {
		ll r=check(cnt);
		dp[idx][cnt]=r;
		return r;
	}
	else {
		ll ans=dp[idx][cnt];
		if(ans!=-1) return ans;
		else {
			ans=0;
			for(int i=0;i<=9;i++) {
				int newidx=idx+1;
				int newcnt=cnt;
				if(i==4||i==7) newcnt=cnt+1;
				if(newcnt>777) continue;
				ans=ans+solve(newidx,newcnt);
			}
		}
		dp[idx][cnt]=ans%mod;
		return dp[idx][cnt];
	}
}
