const ll mod=1000000007;
ll powr(int s,int p)
{
	if(p==0)
		return 1;	

	if(p%2==1)
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);
		q=q%mod;
		q=(q*s);	
		return ( q%mod);
	}

	else
	{
		ll q=powr(s,p/2);
		ll a=q;
		q=(a*a);	
		return (q%mod);
	}
}

ll C[1010][1010];
vector <int> L;
int l;

string a,b,s;

void init()
{
	C[0][0]=1;
	loope(i,1,1009)	
	{
		loope(j,0,i)
		{
			if(j==0 || j==i)
				C[i][j]=1;
			else
				C[i][j]=C[i-1][j]+C[i-1][j-1];

			if(C[i][j]>=mod)
				C[i][j]=C[i][j]%mod;
		}
	}
	return;
	
}


void Gen(int x)
{
	if(x>0)
	{
		L.pb(x);
	}

	if(x<=1000)
	{
		Gen(x*10 + 4);
		Gen(x*10 + 7);
	}
	
	return;
}


ll count(int c,int k)
{
	int p=l-k;
	if(p>c)
		return 0;
	if(p<0)
		return 0;

	ll q;
	q=C[c][p];
	q=q%mod;

	q*=powr(2,p);
	q=q%mod;

	q*=powr(8,c-p);	
	q=q%mod;

	return q;
}


ll go(int pos,int k)
{
	ll ret=0;
	if(k>l)
		return 0;

	if(pos==s.size())
	{
		if(k==l)
			return 1;
		
		else
			return 0;
	}

	loop(i,0,s[pos]-'0')
	{
		ret+=count(sz(s)-1-pos,k + (i==4||i==7) );
		ret=ret%mod;
	}

	ret+=go(pos+1,k+(s[pos]=='4' || s[pos]=='7'));
	ret=ret%mod;
	return ret;
}
