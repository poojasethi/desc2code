template <class T> inline void getint(T &num)
{
	bool neg=0;
	num=0;
	char c;
	while ((c=getchar_unlocked()) && (!isdigit(c) && c!='-'));
	if (c=='-')
	{
		neg=1;
		c=getchar_unlocked();
	}
	do num=num*10+c-'0';
	while ((c=getchar_unlocked()) && isdigit(c));
	num*=(neg)?-1:1;
}

const int MOD = 1000000007;

bool islucky(int x)
{
	int &ret=luck[x];
	if (ret!=-1) return ret;
	sprintf(tmp,"%d",x);
	x=1;
	FOR (i,0,strlen(tmp)) x&=(tmp[i]=='4' || tmp[i]=='7');
	return (ret=x);
}

int f(int pos,int cnt,int under)
{
	if (pos==-1) return islucky(cnt);
	int &ret=dp[pos][cnt][under];
	if (ret!=-1) return ret;
	ret=0;
	FOR (i,0,10)
	{
		int lucky=0,nun=under;
		if (i>x[n-pos-1]-'0' && !under) continue;
		if (i<x[n-pos-1]-'0') nun=1;
		if (i==4 || i==7) lucky=1;
		ret=(ret+f(pos-1,cnt+lucky,nun))%MOD;
	}
	return ret;
}
