#include<iostream>
using namespace std;
int maximum(int a,int b)
{
   if(a>b)
      return a;
   return b;
}
int main()
{
   int T,N,S,F,C,i,j,k;
   int event[49][49];
   cin>>T;
   while(T--)
   {
      cin>>N;
      for(i=0;i<49;i++)
      for(j=0;j<49;j++)
      event[i][j]=0;
      for(i=1;i<=N;i++)
      {
         cin>>S>>F>>C;
         event[F][S]=maximum(event[F][S],C);
      }
      for(i=0;i<49;i++)
      {
         for(j=1;j<49;j++)
            event[i][j]=maximum(event[i][j]+event[j][48],event[i][j-1]);
      }
      cout<<event[48][48]<<endl;
   }
return 0;
}
