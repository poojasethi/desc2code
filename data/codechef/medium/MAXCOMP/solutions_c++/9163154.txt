#include<bits/stdc++.h>

using namespace std;

struct event{
    int si;
    int ei;
    int ci;
};

event arr[2005];

bool cmp(event a,event b)
{
    if(a.ei<b.ei)return true;
    if(a.ei==b.ei&&a.si<b.si)return true;
    return false;
}

int fun(int idx)
{
    int low=0,high=idx,ans=-1;
    int ss=arr[idx+1].si;
    while(low<=high)
    {
        int mid=(low+high)/2;
        int ee=arr[mid].ei;
        if(ee<=ss)
        {
            ans=max(ans,mid);
            low=mid+1;
        }
        else
        {
            high=mid-1;
        }
    }
    return ans;
}
int main()
{
    int dp[2002];
    int t;
    scanf("%d",&t);
    while(t--)
    {
       int n;
       scanf("%d",&n);
       for(int i=0;i<n;i++)
       {
           scanf("%d %d %d",&arr[i].si,&arr[i].ei,&arr[i].ci);
       }
       sort(arr,arr+n,cmp);
       dp[0]=arr[0].ci;
       for(int i=1;i<n;i++)
       {
            int idx=fun(i-1);
            if(idx==-1)
                dp[i]=max(dp[i-1],arr[i].ci);
            else
                dp[i]=max(dp[i-1],dp[idx]+arr[i].ci);
       }
       /*for(int i=0;i<n;i++)
        cout<<dp[i]<<" ";*/
       printf("%d\n",dp[n-1]);
    }
    return 0;
}
