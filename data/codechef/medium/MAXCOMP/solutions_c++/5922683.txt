#include<iostream>
#include<stdio.h>
using namespace std;
int max(int a,int b)
{
    if(a>b)
        return a;
    else
        return b;
}

int main()
{
    int t,N,M,X;
	cin >> t;
	while(t--)
    {
        cin>>N;
        int startTime[N],endTime[N],comp[N];
        for(int i=0;i<N;i++)
        {
            cin>>startTime[i]>>endTime[i]>>comp[i];
        }
        int dp[49];
        for(int i=0;i<49;i++)
            dp[i]=0;

        for(int i=1;i<=48;i++)
        {
            int maxvalue = -1;
            for(int temp=0;temp<N;temp++)
            {
                if(endTime[temp]==i)
                {
                    maxvalue = max(maxvalue,comp[temp]+dp[startTime[temp]]);
                }
            }
            dp[i] = max(maxvalue,dp[i-1]);
        }
        cout<<dp[48]<<endl;


    }
    getchar();
        return 0;
}
