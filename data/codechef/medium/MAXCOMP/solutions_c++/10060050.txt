#include<bits/stdc++.h>
using namespace std;
int order[50][50];
int main(){
    int t;
    cin >> t;
    while(t--){
        int N;
        scanf("%d", &N);
        memset(order, 0, sizeof order);
        while(N--){
            int s, f, c;
            scanf("%d%d%d", &s, &f, &c);
            order[s][f] = max(order[s][f], c);
        }
        vector<int> ans(100, 0);
        for(int f = 0;f <= 48;++f){
                int cur = 0;
            for(int s = 0;s < f;++s){
                cur = max(cur, ans[s]);
                ans[f] = max(ans[f], cur + order[s][f]);
            }
        }
        int final_ans = 0;
        for(int i = 0;i <= 48;++i){
            final_ans = max(final_ans, ans[i]);
        }
        printf("%d\n", final_ans);
    }
    return 0;
}
