#include <bits/stdc++.h>
using namespace std;
#define MP make_pair
#define PB push_back
#include<vector>
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
        vector<pair<pair<int,int> , int > > a;
		long ans[2005]={0};
		ans[0]=0;
		a.PB(MP(MP(0,0),0));
		for(int i=1;i<=n;i++)
		{
		    int s,e,c;
		    cin>>s>>e>>c;
		    a.PB(MP(MP(e,s),c));
		}
		sort(a.begin(),a.end());
        int ind[2005];
        for(int i=1;i<=n;i++)
        {
            ind[i]=0;
            for(int j=1;j<i;j++)
            {
                if(a[j].first.first<=a[i].first.second)
                    ind[i]=j;
            }
            ans[i]=max(ans[i-1],ans[ind[i]]+a[i].second);
        }
        cout<<ans[n]<<endl;

	}
}
