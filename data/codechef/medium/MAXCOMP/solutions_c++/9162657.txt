#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> pii;
#define maxi 10005
int dp[maxi];
vector<pair<int,pii > >v;

int bin_search(int idx)
{
    int st=v[idx].second.first;
    int low=0,high=idx-1,ans=-1;
    while(low<=high)
    {
        int mid=(low+high)/2;
        int tmp=v[mid].first;

         if(tmp<=st)
         {
            low=mid+1;
            ans=mid;
         }
         else
         high=mid-1;
    }
    return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);

        v.clear();
        for(int i=1;i<=n;i++)
        {
            int s,d,p;
            scanf("%d %d %d",&s,&d,&p);
            v.push_back(make_pair(d,make_pair(s,p)));
        }
        sort(v.begin(),v.end());
        //for(int i=0;i<v.size();i++)
            //cout<<v[i].first<<" "<<v[i].second.second<<endl;
        for(int i=0;i<maxi;i++)dp[i]=0;
        dp[0]=v[0].second.second;
        for(int i=1;i<n;i++)
        {
            int p=v[i].second.second;
            int res=bin_search(i);
            if(res==-1)
            {
                dp[i]=max(dp[i-1],p);
            }
            else
            {
                dp[i]=max(dp[i-1],dp[res]+p);
            }
        }
        int maxm=-1;
       //for(int i=0;i<n;i++)maxm=max(maxm,dp[i]);
        printf("%d\n",dp[n-1]);
    }
    return 0;
}
