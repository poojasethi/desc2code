#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>

using namespace std;

int main()
{
	int t;
	scanf("%d\n",&t);
	while(t--)
	{
		int n,h,k,j,x,max=0;
		scanf("%d\n",&n);
		int p[49][49]={0};
		int a[49]={0};

		for(int i=0;i<n;i++)
		{
			scanf("%d",&k);
			scanf("%d",&x);
			scanf("%d",&h);
			if(p[k][x]<h) p[k][x]=h;
		}

		for(j=0;j<49;j++)
		{
			max=0;
			for(k=0;k<j;k++)
			{
				if(max<p[k][j]+a[k]) max=p[k][j]+a[k];
			}
			
			a[j]=max;
			
		}

		printf("%d\n",max);
	}

	return 0;
}
