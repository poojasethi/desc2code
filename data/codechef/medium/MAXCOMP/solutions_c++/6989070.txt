#include<iostream>
using namespace std;

int main()
{
   int t,n,f,s,c,i,j,k;
   int dp[49][49];
   cin>>t;
   while(t--)
   {
      cin>>n;
      for(i=0;i<49;i++)
      for(j=0;j<49;j++)
      dp[i][j]=0;
      for(i=1;i<=n;i++)
      {
         cin>>s>>f>>c;
         dp[f][s]=max(dp[f][s],c);
      }
      for(i=0;i<49;i++)
      {
         for(j=1;j<49;j++)
            dp[i][j]=max(dp[i][j]+dp[j][48],dp[i][j-1]);
      }
      cout<<dp[48][48]<<endl;
   }
return 0;
}
