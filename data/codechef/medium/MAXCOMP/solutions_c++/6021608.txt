using namespace std;
int main()
{	int t,i,j,n,x,y,cost,start,end;
	LL dp[50],sum,ans;
	pair<pair<int,int>,int> p[20002];
	sf(t);
	while(t--)
	{	sf(n);
		for(i=0;i<n;i++)
		{	scanf("%d%d%d",&x,&y,&cost);
			p[i]=mp(mp(y,x),cost);
		}
		sort(p,p+n);
		//for(i=0;i<n;i++)
		//	printf("%d %d %d \n",p[i].f.s,p[i].f.f,p[i].s);
		for(i=0;i<=48;i++) dp[i]=-1;
		for(i=0;i<n;i++)
		{	start=p[i].f.s;
			end=p[i].f.f;
			cost=p[i].s;
			sum=0;
			for(j=start;j>=0;j--)
			{	if(dp[j]!=-1)
				{	sum=max(sum,dp[j]);
					
				}
			}
			sum+=cost*(1LL);
			if(dp[end]<sum)
				dp[end]=sum;
			//printf("%d %d\n",end,dp[end]);	
		}
		ans=-1;
		for(i=0;i<=48;i++)
		{	if(dp[i]!=-1)
				if(dp[i]>ans)
					ans=dp[i];
		}
		printf("%lld\n",ans);
	}
	return 0;
}
