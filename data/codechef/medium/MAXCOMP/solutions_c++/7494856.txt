#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

class spe{
public:
int s;
int e;
int c;
};
struct compare{
bool operator()(const spe &spee1,const spe &spee2)
{
        return spee1.e<spee2.e;
}
};
int main()
{
    int t,n,ts,te,tc;
    spe temps;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        vector<int> ans(49,0);
        vector<spe> que(n);
        for(int i=0;i<n;i++)
        {
            scanf("%d%d%d",&ts,&te,&tc);
            temps.s=ts;
            temps.e=te;
            temps.c=tc;
            que[i]=temps;
        }
        sort(que.begin(),que.end(),compare());
        temps.s=50;
        temps.e=50;
        temps.c=50;
        que.push_back(temps);
        int j=0;
        int temp=0;
        for(int i=1;i<=48;i++)
        {
                ans[i]=temp;
                while(que[j].e==i)
                {
                    if(ans[i]<(ans[que[j].s]+que[j].c))
                        ans[i]=ans[que[j].s]+que[j].c;
                    j++;
                }
                temp=ans[i];
        }
        printf("%d\n",ans[48]);
    }
return 0;
}
