#include <stdio.h>
#include <algorithm>
#include <memory.h>
 
using namespace std;
 
int main(){
    int C[49][49];
    int M[49];
    int tt;
    scanf("%d",&tt);
    while(tt--){
        memset(C,0,sizeof(C));
        memset(M,0,sizeof(M));
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;i++){
            int s,e,c;
            scanf("%d %d %d",&s,&e,&c);
            C[s][e] = max(c,C[s][e]); 
        }
        for(int i=0;i<=48;i++)
            for(int j=0;j<i;j++)
                M[i] = max(M[i],M[j]+C[j][i]);
        printf("%d\n",M[48]);
    }
    return 0;
}
