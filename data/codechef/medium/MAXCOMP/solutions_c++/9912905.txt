#include<stdio.h>
#include<vector>
#include<algorithm>
#define pb(v,a) v.push_back(a)
#define mp(a,b) make_pair(a,b)
#define tr(c,it) for(typeof(c.begin()) it=c.begin();it!=c.end();it++)
using namespace std;
vector< pair<pair<int,int>,long long > > v;
int ind[10000];
long long dp[10000];
int main()
{
  //  freopen("input.txt","r",stdin);
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,s,e;
        long long c;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {   ind[i]=-1;
            scanf("%d%d%lld",&s,&e,&c);
            v.push_back(make_pair(make_pair(e,s),c));
        }
        sort(v.begin(),v.end());
       /* tr(v,it)
        {
//      printf("%d %d %d\n",(*it).first.second,(*it).first.first,(*it).second);
        }*/
        dp[0]=v[0].second;
        ind[0]=-1;
        for(i=1;i<n;i++)
        {
            int l,m,r;
            l=0;
            r=i;
            int tmp=-1;
            while(l<r)
            {
                m=l+(r-l)/2;
      //          printf("%d %d %d ",l,m,r);
                if(v[m].first.first<v[i].first.second)
                {
                    tmp=m;
                    l=m+1;
                }
                else if(v[m].first.first==v[i].first.second)
                {
                    tmp=m;
                    l=m+1;
                }
                else
                {
                    r=m;
                }
            }
            if(v[l].first.first>v[i].first.second)
            {
         //       printf("%d %d %d %d\n",l,i,v[l].first.first,v[i].first.second);
                l=tmp;
      //          printf("a");
            }
            ind[i]=l;
            dp[i]=dp[i-1];
            if(ind[i]>=0)dp[i]=max(dp[i],dp[ind[i]]+v[i].second);
            else dp[i]=max(dp[i],v[i].second);
         //   printf("%d %d\n",ind[i],dp[i]);
        }
        printf("%lld\n",dp[n-1]);
        v.clear();
    }
    return 0;
}
