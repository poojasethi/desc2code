#include <bits/stdc++.h>
using namespace std;

char flip(char a)
{
	if(a == '0') return '1';
	return '0';
}

int main()
{
	int t,n,k,i,cnt,cnt2;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		cin>>s;
		if(k==1)
		{
			cnt = cnt2 = 0;
			for(i=0;i<s.length();i++)
			{
				if(s[i]!=i%2 + '0') cnt++;
				if(s[i]!=(i+1)%2 + '0') cnt2++;
			}
			if(cnt < cnt2)
			{
				cout<<cnt<<"\n";
				for(i=0;i<s.length();i++) cout<<(char)(i%2+'0');
				cout<<'\n';
			}
			else
			{
				cout<<cnt2<<"\n";
				for(i=0;i<s.length();i++) cout<<(char)((i+1)%2+'0');
				cout<<'\n';
			}
		}
		else
		{
			cnt = 1;
			s+="\n";
			char prev = '#';
			int prev_index = -1;
			cnt2 = 0;
			for(i=0;i<n+1;i++)
			{
				if(s[i]!=prev)
				{
					if(cnt>k)
					{
						for(int j=prev_index+k;j<i;j+=k+1)
						{
							if(j != i-1) s[j] = flip(s[j]);
							else s[j-1] = flip(s[j-1]);	
							cnt2++;
						}
					}
					prev = s[i];
					prev_index = i;
					cnt = 1;
				}
				else cnt++;
				//cout<<prev<<":"<<cnt<<"\n";
			}
			cout<<cnt2<<'\n'<<s;
		}
	}
	return 0;
}
