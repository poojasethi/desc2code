#include<bits/stdc++.h>
using namespace std;
char arr[100010],str[100010],s[100010];
int zeroes[100010],ones[100010];
int main()
{
	int test,n,k;
	scanf("%d",&test);
	while(test--)
	{
		scanf("%d%d",&n,&k);
		scanf("%s",arr);
		strcpy(str,arr);

		int len=strlen(arr);
		str[len]='\0';

		int counter1=0,counter0=0;

		if(k==1)
		{
			for(int i=0;i<len;i++)
				s[i]=str[i]=i%2+'0';
			for(int i=0;i<len;i++)
			{
				if(arr[i]!=str[i])
					counter0++;
			}
			for(int i=0;i<len;i++)
				str[i]=(i+1)%2+'0';
			for(int i=0;i<len;i++)
			{
				if(arr[i]!=str[i])
					counter1++;
			}
			s[len]='\0';
			if(counter0<counter1)
				printf("%d\n%s\n",counter0,s);
			else
				printf("%d\n%s\n",counter1,str);
		}
		else
		{
			int ans=0;
			
			ones[len-1]=0,zeroes[len-1]=0;
			
			if(arr[len-1]=='1')
				counter1=1;
			else
				counter0=1;
			
			for(int i=len-2;i>=0;i--)
			{
				ones[i]=counter1;
				zeroes[i]=counter0;
				if(arr[i]=='1')
				{
					counter1++;
					counter0=0;
				}
				else
				{
					counter0++;
					counter1=0;
				}
			}
			counter1=counter0=0;
			
			if(arr[0]=='1')
				counter1=1;
			else
				counter0=1;
			
			for(int i=1;i<len;i++){
				if(arr[i]=='1'){
					counter1++;
					if(counter1>k)
					{
						ans++;
						if(zeroes[i]==0){
							arr[i]='0';
							counter0=1;
							counter1=0;
						}
						else{
							arr[i-1]='0';
							counter0=0;
							counter1=1;
						}
					}
					else
						counter0=0;			
				}
				else{
					counter0++;
					if(counter0>k)
					{
						ans++;
						if(ones[i]==0){
							arr[i]='1';
							counter1=1;
							counter0=0;
						}
						else{
							arr[i-1]='1';
							counter1=0;
							counter0=1;
						}
					}
					else
						counter1=0;			
				}
			}
			printf("%d\n%s\n",ans,arr);
		}
	}
	return 0;
}
