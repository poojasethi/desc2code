#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long ll;
#define sl(l) scanf("%lld",&l)
#define pl(l) printf("%lld\n",l)
char str[100002];
char flip(char c)
{
	if(c=='0')return '1';
	else return '0';
}
int main()
{
	ll t;
	sl(t);
	while(t--)
	{
		ll n,k;
		sl(n);sl(k);
		scanf("%s",str);
		if(n<=k)
		{
			ll l=0;
			pl(l);
			printf("%s\n",str);
		}
		else if(k==1)
		{
			ll p=0,q=0;
			for(ll i=0;i<n;i++)
			{
				if(i%2==0)
				{
					if(str[i]=='1')p++;
					else q++;
				}
				else
				{
					if(str[i]=='0')p++;
					else q++;
				}
			}

			if(p<q)
			{
				for(ll i=0;i<n;i++)
				{
					if(i%2==0)
					{
						str[i]='0';
					}
					else str[i]='1';
				}
				pl(p);
				printf("%s\n",str);
			}
			else
			{
				for(ll i=0;i<n;i++)
				{
					if(i%2==0)
					{
						str[i]='1';
					}
					else str[i]='0';
				}
				pl(q);
				printf("%s\n",str);	
			}
		}
		else
		{
			char c;
			ll l,ans=0;
			for(ll i=0;i<n;i++)
			{
				if(i==0)
				{
					c = str[i];
					l = 1;
					continue;
				}
				else if(str[i]==c)
				{
					l++;
					if(l>k)
					{
						if(str[i] == str[i+1])
						{
							str[i] = flip(str[i]);
						}
						else str[i-1] = flip(str[i-1]);
						ans++;
						l=1;
					}
					c = str[i];
				}
				else
				{
					c = str[i];
					l=1;
				}
			}
			pl(ans);
			printf("%s\n",str);
		}
	}
	return 0;
}
