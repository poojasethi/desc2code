{
	long long int   t,n,k,i,count,c1,c2,end,start,cstart,cend,l,flip_cnt;

			int oo = 0;

			int ez = 0;
 
			for ( i = 0; i < l; i++)
			{
				if (i & 1)
				{
					if (s[i] == '0')oz++;
					else oo++;
				}
				else
				{
					if (s[i] == '0')ez++;
					else eo++;
				}
			}
			if ((eo + oz) <= (ez + oo))
			{
				printf("%d\n",eo + oz);
				i = 0;
				bool sw = false;
				while (i < l)
				{
					printf("%c",(sw == false ? '0' : '1'));

				}

				int cnt = 1;
				while (s[i] == s[i - 1] && i < l)
