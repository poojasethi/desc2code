using namespace std;
const int N=100;
string a;
int n,k;

void  input(){
    cin>>n>>k>>a;
}

pair <int,string> sol(string s,int ans){
    for (int i=1;i<s.size();i++)
    if (s[i]==s[i-1]) s[i]='1'+'0'-s[i],ans++;
    return mp(ans,s);
}

void sol(){
    if (k==1) {
        pair <int,string> x=sol(a,0);
        a[0]='1'+'0'-a[0];
        x=min(x,sol(a,1));
        cout<<x.X<<endl<<x.Y<<endl;
        return;
    }
    int ans=0;
    int l=0,r=0;
    while (l<a.size()){
        while (r+1<a.size() && a[r+1]==a[l]) r++;
        for (int i=l+k;i<=r;i+=k+1){
            if (i==r) i--;
            a[i]='1'+'0'-a[i];
            ans++;
        }
        l=r=r+1;
    }
    cout<<ans<<endl<<a<<endl;
}
