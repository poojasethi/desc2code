#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,k;
    scanf("%d",&t);
    while(t--)

    {
        scanf("%d%d",&n,&k);

        string s;
      cin>>s;

        if(k>=n)
        {
             cout<<0<<endl<<s<<endl;
              continue;
        }

           int ans=0;
  if(k==1){
        char arr[n+2];
        int count1=0,count2=0;


        for(int i=0;i<n+2;i=i+2)
            arr[i]='0';
        for(int i=1;i<n+2;i=i+2)
            arr[i]='1';


            for(int i=0;i<n;i++)
                if(s[i]!=arr[i])
                  count1++;

            for(int i=1;i<n+1;i++)
                 if(s[i-1]!=arr[i])
                   count2++;

            if(count1<=count2)
            {
                printf("%d\n",count1);
                for(int i=0;i<n;i++)
                    cout<<arr[i];

                cout<<endl;
            }
            else
            {
                printf("%d\n",count2);
                for(int i=1;i<n+1;i++)
                    cout<<arr[i];
                cout<<endl;
            }

continue;


  }

  for(int i=0;i<n;i++)
  {
      int st=i;

       if(s[i]=='0')
       {    int count=0;
           while(s[i]=='0' && i<n)
           {
               count++;
               i++;
           }

            if(count>k)
            {
                int x=floor(count/(k+1));
                int y=count%(k+1);

                if(x==1)
                {
                    s[st+count/2]='1';
                    ans++;
                }

                else  if(y==k)
                {
                    st=st+k;
                    while(st<i)
                    {
                        ans++;
                        s[st]='1';
                        st+=k+1;
                    }
                }

                else
                {
                    st+=k-1;
                    while(st<i)
                    {
                        ans++;
                        s[st]='1';
                        st+=k+1;
                    }
                }

            } i--;
       }

        if(s[i]=='1')
        {
            int count=0;
            while(s[i]=='1' &&  i<n)
            {
                count++;
                i++;
            }


            if(count>k)
            {    int x=floor(count/(k+1));
                int y=count%(k+1);


                if(x==1)
                {
                    s[st+count/2]='0';
                    ans++;

                }

                else if(y==k)
                {
                    st+=k;
                    while(st<i)
                    {
                        ans++;
                        s[st]='0';
                        st+=k+1;

                    }
                }
                else
                {
                    st+=k-1;
                    while(st<i)
                    {
                        ans++;
                        s[st]='0';
                        st+=k+1;
                    }
                }

            }
            i--;
        }
  }

   cout<<ans<<endl;
   cout<<s<<endl;



    }
}
