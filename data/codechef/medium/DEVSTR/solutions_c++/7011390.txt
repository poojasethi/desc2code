#include <cstdio>
#include <cstring>
#include <algorithm>
#define MAX 100000
using namespace std;
char s[MAX+1],ss[MAX+1];
void flip(int start,int end,int k,int *ans)
{
    char c=1-(s[start]-'0')+'0';
    if(end-start+1<=k)
	return;
    int cur = start+k;
    while(cur<end)
    {
	s[cur] = c;
	(*ans)++;
	cur+=(k+1);
    }
    if(cur==end)
    {
	s[cur-1] = c;
	(*ans)++;
    }
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
	int n,k,ans=0;
	scanf("%d %d %s",&n,&k,s);
	if(k==1)
	{
	    int i=0,ans2=0;
	    int v1=0,v2=1;
	    while(i<n)
	    {
		if(s[i]!=v1+'0')
		    ans++;
 
		if(s[i]!=v2+'0')
		    ans2++;
		i++;
		v1=1-v1;v2=1-v2;
	    }
	    
 
	    int start=0;
	    if(ans2<ans)
	    {
		ans=ans2;
		start=1;
	    }
	    for(int i=0;i<n;i++)
	    {
		s[i] = start + '0';
		start = 1 - start;
	    }
	}
	else
	{
	    int start=0,end=0;
	    while(end<n)
	    {	
		char c = s[start];
		while(end<n && s[end]==c)
		{
		    end++;
		}
		flip(start,end-1,k,&ans);
		start=end;
	    }
	 }
	 //printf("%d\n",ans);
	printf("%d\n%s\n",ans,s);
	
    }
}
