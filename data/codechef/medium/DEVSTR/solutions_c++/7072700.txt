using namespace std;
const int N=100005;
int T,n,k,ans;
char s[N];

inline void work(int l,int r){
    int cur=r-l+1;
    if(l>r || cur<=k)return;
    for(int i=l+k;i<=r;i+=k+1)
        s[i]=char('0'+'1'-s[i]),ans++;
  //  cout<<"work "<<cur<<' '<<l<<' '<<r<<endl;
    if(cur%(k+1)==0)
        swap(s[r],s[r-1]);
}


int main()
{scanf("%d",&T);
 while(T--){
    scanf("%d%d",&n,&k);
    scanf("%s",&s);

    if(k==1){
        int a=0,b=0;
        for(int i=0;i<n;i++)
        if(i&1){
           if(s[i]=='0')a++;
           if(s[i]=='1')b++;
        }
        else{
           if(s[i]=='0')b++;
           if(s[i]=='1')a++;
        }

        ans=MIN(a,b);
        if(a<b)s[0]='0';else s[0]='1';
        for(int i=1;i<n;i++)s[i]=char('0'+'1'-s[i-1]);
    }else{
        ans=0;
        int l=0;
        for(int i=1;i<n;i++){
            if(s[i]==s[i-1])continue;
            work(l,i-1);
            l=i;
        }
        work(l,n-1);
    }
    printf("%d\n",ans);
    for(int i=0;i<n;i++)putchar(s[i]);
    printf("\n");
 }
 return 0;
}
