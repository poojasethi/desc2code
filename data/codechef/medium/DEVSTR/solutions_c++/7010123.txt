#include <iostream>

using namespace std;

int main() {
  int t;
  cin >> t;
  while (t--) {
    int n, k;
    cin >> n >> k;
    string str;
    cin >> str;
    if (k == 1) {
      string out1 = str;
      int bit = 1;
      int cnt1 = 0;
      for (int i=0; i<n; ++i) {
        if (str[i] != (bit + '0')) {
          cnt1++;
          out1[i] = (bit + '0');
        }
        bit = (1-bit);
      }
      string out2 = str;
      bit = 0;
      int cnt2 = 0;
      for (int i=0; i<n; ++i) {
        if (str[i] != (bit + '0')) {
          cnt2++;
          out2[i] = (bit + '0');
        }
        bit = (1-bit);
      }
      if (cnt1 < cnt2)
        cout << cnt1 << endl << out1 << endl;
      else
        cout << cnt2 << endl << out2 << endl;
    } else {
      string out = str;
      int cnt = 0;
      int st = 0;
      int en = 0;
      while (en < n) {
        if (str[en] == str[st]) {
          en++;
        } else {
          if (en - st > k) {
            int ii;
            for (ii=st+k; ii<en; ii+=(k+1)) {
              out[ii] = (('1' - out[ii]) + '0');
              cnt++;
            }
            ii -= (k+1);
            if (ii == en-1) {
              out[ii] = (('1' - out[ii]) + '0');
              out[ii-1] = (('1' - out[ii-1]) + '0');
            }
          }
          st = en;
        }
      }
      if (en - st > k) {
        int ii;
        for (ii=st+k; ii<en; ii+=(k+1)) {
          out[ii] = (('1' - out[ii]) + '0');
          cnt++;
        }
        ii -= (k+1);
        if (ii == en-1) {
          out[ii] = (('1' - out[ii]) + '0');
          out[ii-1] = (('1' - out[ii-1]) + '0');
        }
      }
      cout << cnt << endl << out << endl;
    }
  }
  return 0;
}
