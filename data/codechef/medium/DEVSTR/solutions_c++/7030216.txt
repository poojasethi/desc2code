void flip (string & str , int len , int k   ) {
	int i , j ;
	if(len%(k+1) == 0) {
		str[len-2] = (str[len-2] == '0' ) ? '1' : '0' ;
		for (i = 0; i < len-(k+1); i++) {	
			if((i+1)%(k+1) == 0) str[i] = ((str[i] == '0') ? '1' : '0' ) ;
		}
	} else {
		long long int j = 1; 
		
		for(i = 0 ; i < len; i++) {
			if( (i+1)%(k+1) == 0 ) str[i] = (( str[i] == '0') ? '1' : '0' ) ; 
		}
	} 
}

void binaryString() {
	long long i ,  k , n , n1 , len  , flips  ;
	string s , str , finalString  ; 
 	string zeroString , oneString;
	
	cin >> n >> k >> s;
	str = "";
	finalString = "" ;
	oneString = "";
	zeroString = "";
	flips = 0 ;

	if(k == 1) {
		int zeroflip = 0 , oneflip = 0;
		for(i = 0; i < n; i++){
			if(i%2) zeroString += '0';
			else zeroString += '1';
			
			if(i%2) oneString += '1';
			else oneString += '0';
		} 
		for(i = 0; i < n; i++) {
			if(s[i] != zeroString[i] ) zeroflip++;
			if(s[i] != oneString[i] ) oneflip++;
		}

		if(zeroflip < oneflip ) {
			flips = zeroflip;
			finalString = zeroString;
		} else {
			flips = oneflip;
			finalString = oneString; 
		}

	} else { 

	 
		for(i = 0; i < n-1; i++) {  
			str = str + s[i];
			if(s[i] != s[i+1] ) {
		//		cout << str << endl;
				len = str.size();
				// if length > str then flip the n/k+1 elements 
				if(len > k)  {	
					flip(str, len , k) ;
					flips += len/(k+1) ; 
				}
				finalString += str;
				str = "";
			}
	//		i++; 
		} 

		str += s[i] ;
		//cout << str << endl;
		len = str.size() ; 
		if(len > k ) {	
			flip(str , len , k) ;
			flips += len/(k+1);
			finalString  += str;
		} else finalString += str;

	}
//	cout << " output " << endl; 
	cout  << flips << endl;

}
