#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
int n,k;
scanf("%d %d",&n,&k);
char s[100003];
scanf("%s",s);
int i,j=1,ans=0;
if(k==1)
{
for(i=1;i<n;++i)
{
if(s[i]==s[i-1])
{
++ans;
s[i]=(s[i]=='1')?'0':'1';
}
}
if(n-ans>=ans)
{
printf("%d\n",ans);
printf("%s\n",s);
}
else
{
printf("%d\n",n-ans);
for(i=0;i<n;++i)
(s[i]=='1')?printf("0"):printf("1");
cout<<endl;
}
}
else
{
for(i=1;i<n;++i)
{
if(s[i]==s[i-1])
++j;
else
j=1;
if(j>k)
{
++ans;
if(i==n-1 || s[i+1]==s[i])
s[i]=(s[i]=='1')?'0':'1';
else
s[i-1]=(s[i-1]=='1')?'0':'1';
}
}
printf("%d\n",ans);
printf("%s\n",s);
}


}
return 0;
}
