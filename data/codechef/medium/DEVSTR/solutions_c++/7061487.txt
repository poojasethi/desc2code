#include<stdio.h>
int main()
{
	int t,n,k;
	char c;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&k);int flip=0;
		getchar();
		bool a[n];
		for(int i=0;i<n;i++)
		{	
			scanf("%c",&c);
			a[i]=c-'0';
		}
		int count=1;
		bool curr=a[0];
		if(k>1)
		{
			for (int i = 1; i < n; i++)
			{
					if(a[i]==curr)
					{
						if(count==(k+1))
						{
							a[i-1]=!a[i-1];flip++;
							count=1;
						}
						else if(count==k && i==n-1)
						{
						 	a[i-1]=!a[i-1];
						 	flip++;
							count=1;
						}
						else count++;
					}
					else
					{
						if(count==(k+1))
						{
							a[i-2]=!a[i-2];flip++;
						}
						count=1;
						curr=a[i];
					}			
			}
			printf("%d\n",flip );
			for(int i=0;i<n;i++)
				printf("%d",a[i] );
			printf("\n");
		}
		if(k==1)
		{
			int flip1=0,flip2=1;
			curr=a[0];
			for (int i = 1; i < n; ++i)
			{
				curr=!curr;
				if(a[i]!=curr) flip1++;
				else flip2++;
			}
			if(flip1<flip2)
			{
				printf("%d\n",flip1 );
				curr=a[0];
				for (int i = 0; i < n; ++i)
				{
					printf("%d",curr);
					curr=!curr;
				}
			}
			else
			{
				printf("%d\n",flip2 );
				curr=!a[0];
				for (int i = 0; i < n; ++i)
					{
					printf("%d",curr);
					curr=!curr;
				}
			}
			printf("\n");
		}
		

	}
}
