#include<iostream>
#include<ctype.h>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
int main()
{
	int t;
	int n,k,count,sum;
	char current;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		char string[n+1];
		string[n]='\0';
		count=1;
		sum=0;
		cin>>string;
		current=string[0];
		if(k!=1)
		{
			for(int i=1;i<n;i++)
			{
				if(string[i]==current)
				{
					count++;
					if(count>k)
					{
						if(i<n-1&&string[i+1]==current)
						{
							string[i]=((string[i]-'0'+1)%2)+'0';
						}
						else
						{
							string[i-1]=((string[i]-'0'+1)%2)+'0';
						}
						count=1;
						current=string[i];
						sum++;
					}
				}
				else
				{
					current=string[i];
					count=1;
				}	
			}
			cout<<sum<<"\n"<<string<<"\n";
		}
		else
		{
			int count1=0,count2=0;
			for(int i=0;i<n;i++)
			{
				if(i%2==0)
				{
					if(string[i]=='0')
					{
						count1++;
					}
					else
						count2++;
				}
				else
				{
					if(string[i]=='0')
						count2++;
					else
						count1++;
				}
			}
			if(count1<count2)
			{
				cout<<count1<<"\n";
				for(int i=0;i<n;i++)
				{
					if(i%2==0)
						cout<<"1";
					else
						cout<<"0";
				}
				cout<<"\n";
			}
			else
			{
				cout<<count2<<"\n";
				for(int i=0;i<n;i++)
					if(i%2==0)
						cout<<"0";
					else
						cout<<"1";
				cout<<"\n";
			}
		
		}
	}
		return 0;
}
