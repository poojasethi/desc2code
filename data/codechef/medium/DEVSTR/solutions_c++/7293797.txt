#include<iostream>
#include<bits/stdc++.h>
#include<stdio.h>
#include<string>

#define lld long long int

using namespace std;

int main()
{
    lld t,i,j,n,k;
    char a[100002];
    scanf("%lld ", &t);
    while(t--)
    {
        scanf("%lld%lld", &n,&k);
        scanf("%s",a);
        if(k==1)
        {
            lld one[n+1],zero[n+1];
            lld ct_one=0,ct_zero=0;
            for(i=0; i<n; i+=2)
            {
                one[i]=1;
                zero[i]=0;
            }
            for(i=1; i<n; i+=2)
            {
                one[i]=0;
                zero[i]=1;
            }
            /*
                for(i=0; i<n; i+=1)
                {
                    printf("%lld",one[i]);
                }
                printf("\n");
                for(i=0; i<n; i+=1)
                {
                    printf("%lld",zero[i]);
                }
            */
            for(i=0; i<n; i++)
             {
                if((a[i]-'0')!=one[i])
                    ct_one++;
                if((a[i]-'0')!=zero[i])
                    ct_zero++;
             }
            if(ct_one<=ct_zero)
              {
                   printf("%lld\n",ct_one);
                   for(i=0; i<n; i++)
                   {
                        printf("%lld",one[i]);
                   }
              }
              else
               {
                   printf("%lld\n",ct_zero);
                   for(i=0; i<n; i++)
                   {
                        printf("%lld",zero[i]);
                   }
              }
        }
        else
        {
            lld cnt=1,cx=0;
            for(i=1; i<n; i++)
            {
                if(a[i]!=a[i-1])
                  cnt=1;
                  else
                    cnt++;
                if(cnt>k)
                {
                    if(a[i]!=a[i+1] && a[i-1]=='1')
                         a[i-1]='0';
                     else if(a[i]!=a[i+1] && a[i-1]=='0')
                          a[i-1]='1';
                        else if(a[i]==a[i+1] && a[i+1]=='1' && i+1<n)
                           a[i]='0';
                          else if(a[i]==a[i+1] && a[i+1]=='0' && i+1<n)
                             a[i]='1';
                   cx++;
                   cnt=1;
                }
            }
            printf("%lld\n",cx);
            for(i=0; i<n; i++)
               printf("%c",a[i]);
        }
        printf("\n");
    }
    return 0;
}
