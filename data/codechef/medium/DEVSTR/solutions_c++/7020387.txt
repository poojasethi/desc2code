#include <bits/stdc++.h>


using namespace std;

int T, n, k;

string s;

int main() {
	// freopen("input.txt", "r", stdin);
	ios_base::sync_with_stdio(0);

	cin >> T;
	while (T--) {
		cin >> n >> k;
		cin >> s;
		if (k == 1) {
			int t = 0;
			int cnt1 = 0;
			for (int i = 0; i < n; ++i) {
				cnt1 += (s[i] != '0' + t);
				t ^= 1;
			}

			int cnt2 = 0;
			t = 1;
			for (int i = 0; i < n; ++i) {
				cnt2 += (s[i] != '0' + t);
				// cout << cnt2 <<" "<<s[i] <<" "<<char(t) <<" " << endl;
				t ^= 1;
			}
			if (cnt1 < cnt2) {
				cout << cnt1 << endl;
				t = 0;
				for (int i = 0; i < n; ++i)
					cout << t, t ^= 1;
				cout << endl;
			}
			else {
				cout << cnt2 << endl;
				t = 1;
				for (int i = 0; i < n; ++i)
					cout << t, t ^= 1;
				cout << endl;
			}
		}
		else {
			int res = 0;
			int cnt = 1;
			for (int i = 1; i <= n; ++i) 
				if (s[i] == s[i - 1])
					++cnt;
				else {
					if (cnt > k) {
						res += cnt / (k + 1);
						for (int j = i - cnt + k; j < i; j += k + 1) {
							if (j == i - 1) {
								s[j - 1] = 1 - s[j - 1] + 2 * '0';
								continue;
							}
							s[j] = 1 - s[j] + 2 * '0';
						}
					}
					cnt = 1;
				}
			cout << res << endl << s << endl;
		}
	}
}
