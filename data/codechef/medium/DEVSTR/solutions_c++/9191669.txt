#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<iostream>
using namespace std;

int main()
{
     char A[100005];
     int t,i,n,k,count,flips;
     int odd,even;
     cin>>t;
     while(t>0)
     {
          cin>>n>>k;
          cin>>A;
          if(k==1)
          {
               odd=even=0;
               for(i=0;i<n;i+=2)
               {
                    if(A[i]=='1')
                         even++;
                    else
                         odd++;  
                    if(A[i+1]!='\0')
                         if(A[i+1]=='1')
                              odd++;
                         else
                              even++;
               }
               if(odd<=even)
               {
                    cout<<odd<<endl;
                    for(i=1;i<=n;i++)
                    {
                         cout<<i%2;
                    }
                    cout<<endl;
               }
               else
               {
                    cout<<even<<endl;
                    for(i=0;i<n;i++)
                         cout<<i%2;
                    cout<<endl;
               }
          }
          else
          {
               flips=0;
               count=1;
               for(i=1;i<n;i++)
               {
                    if(A[i]==A[i-1])
                         count++;
                    else
                         count=1;
                    if(count>k)
                    {
                         if(A[i+1]==A[i])
                              A[i]=A[i]^1;
                         else
                              A[i-1]=A[i-1]^1;
                         flips++;
                    }
               }
               cout<<flips<<endl;
               cout<<A<<endl;
          }
          t--;
     }
    }
