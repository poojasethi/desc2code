#include<stdio.h>
#include<algorithm>
using namespace std;
char a[100004];
long long int dp[100004] = { 0 };
int main()
{
	
	long long int t, n, k, flag, count, res,res1,res2, i;
	scanf("%lld", &t);
	while (t--)
	{
		scanf("%lld%lld", &n, &k);
		scanf("%s", a);
		if (a[n - 1] == '0')
			flag = 0;
		else
			flag = 1;
		count = 1;
		for (i = n - 2; i >= 0; i--)
		{
			if (flag == 0)
			{
				if (a[i] == '0')
					count++;
				else
				{
					count = 1;
					flag = 1;
				}
			}
			else
			{
				if (a[i] == '1')
					count++;
				else
				{
					count = 1;
					flag = 0;
				}
			}
			if (count >= k)
			{
				dp[i] = 1;
			}
		}
		dp[n] = 0;
		dp[0] = 0;
		count = 1;
		res = 0;
		for (i = 1; k>1&&i <= n-1; i++)
		{
			if (a[i] == a[i - 1])
				count++;
			else
				count = 1;
			if (count > k)
			{
				if (dp[i + 1] == 1&&a[i]!=a[i+1])
				{
					if (a[i] == '1')
						a[i - 1] = '0';
					else
						a[i - 1] = '1';
				}
				else
				{
					if (a[i] == '1')
						a[i] = '0';
					else
						a[i] = '1';
				}
				res++;
				dp[i] = 0;
				count = 1;
			}
		}
		res1 = res2 = 0;
		if (k == 1)
		{
			for (i = 0; i < n; i++)
			{
				if (i % 2==0)
				{
					if (a[i] == '1')
					{
						
						res1++;
					}
				}
				else
				{
					if (a[i] == '0')
					{
						
						res1++;
					}
				}
			}
			for (i = 0; i < n; i++)
			{
				if (i % 2)
				{
					if (a[i] == '1')
					{
						
						res2++;
					}
				}
				else
				{
					if (a[i] == '0')
					{
						
						res2++;
					}
				}
			}
			res = min(res1, res2);
			if (res == res1)
			{
				for (i = 0; i < n; i++)
				{
					if (i % 2)
						a[i] = '1';
					else
						a[i] = '0';
				}
			}
			else
			{
				for (i = 0; i < n; i++)
				{
					if (i % 2)
						a[i] = '0';
					else
						a[i] = '1';
				}
			}
		}
		printf("%lld\n", res);
		printf("%s", a);
		printf("\n");
	}
	getchar();
	getchar();
	return 0;
}
