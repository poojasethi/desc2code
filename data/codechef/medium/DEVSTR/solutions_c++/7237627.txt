#include <cstdio>
#define MAX 100000
 
using namespace std;
 
void flip(char &a)
{
    a = !(a-'0')+'0';
}
 
long long int i,t,n,k,ans,end[MAX+10],size[MAX+10],j,a; char s[MAX+10];
 
int main()
{
    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld%lld",&n,&k);
        scanf("%s",s);
        ans=a=0;
 
        if(k==1)
        {
            for(i=0;i<n;i+=2)
                a+=s[i]-'0';
            for(i=1;i<n;i+=2)
                a+=!(s[i]-'0');
            ans=n-a;
            for(i=0;i<n;i++)
                s[i]='0'+!(i&1);
            if(n>2*a) {
                for(i=0;i<n;i++)
                flip(s[i]);
                ans=a;}
        }
        else
        {
           j=0;a=1;
           for(i=1;i<n;i++)
            if(s[i]==s[i-1]) a++;
            else{
                    if(a>k)
                        {size[j]=a;
                        end[j]=i-1;j++;
                        }
                    a=1;
                }
            end[j]=n-1;size[j]=a;j++;
            for(i=0;i<j;i++)
            {
                ans+=size[i]/(k+1);
                for(a=end[i]-size[i]+k+1;a<end[i];a+=k+1) flip(s[a]);
                if(a==end[i]) flip(s[a-1]);
            }
        }
        printf("%lld\n%s\n",ans,s);
    }
 
    return 0;
}
