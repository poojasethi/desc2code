#include<bits/stdc++.h>
    #define getcx getchar_unlocked
    inline int inp()
    {
    int ch,n;
    while((ch = getcx())<'0');
    n =(ch-'0');
    while((ch = getcx())>='0')
    n=(n<<3)+(n<<1)+(ch-'0');
    return n;
    }
    int main()
    {
    int t;
    t=inp();
    while(t--)
    {
    int n,k,i;
    char s[100002];
    n=inp();
    k=inp();
    scanf("%s",s);
    if(k==1)
    {
    char s1[n+1],s2[n+1];
    int c1=0,c2=0;
    for(i=0;i<n;i=i+2)
    {
    s1[i]='1';
    s2[i]='0';
    }
    for(i=1;i<n;i=i+2)
    {
    s1[i]='0';
    s2[i]='1';
    }
    s1[n]='\0';
    s2[n]='\0';
    for(i=0;i<n;i++)
    {
    if(s1[i]!=s[i])
    c1++;
    if(s2[i]!=s[i])
    c2++;
    }
    if(c1<c2)
    {
    printf("%d\n",c1);
    printf("%s\n",s1);
    }
    else
    {
    printf("%d\n",c2);
    printf("%s\n",s2);
    }
    }
    else
    {
    int cnt=1,final=0;
    for(i=1;i<n;i++)
    {
    if(s[i]!=s[i-1])
    cnt=1;
    else
    cnt++;
    if(cnt>k)
    {
    if(s[i]!=s[i+1] && s[i-1]=='1')
    s[i-1]='0';
    else if(s[i]!=s[i+1] && s[i-1]=='0')
    s[i-1]='1';
    else if(s[i]==s[i+1] && s[i]=='1' && i+1<n)
    s[i]='0';
    else if(s[i]==s[i+1] && s[i]=='0' && i+1<n)
    s[i]='1';
    cnt=1;
    final++;
    }
    }
    printf("%d\n",final);
    printf("%s\n",s);
    }
    }
    return 0;
    }
