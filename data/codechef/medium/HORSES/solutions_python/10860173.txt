import sys
 
if __name__ == '__main__':
    T = int(sys.stdin.readline())
 
    for i in range(T):
        N = int(sys.stdin.readline())
        S = [int(x) for x in sys.stdin.readline().split()]
 
        S.sort()
 
        min_diff = None
 
        for i in range(len(S)-1):
            new_diff = abs(S[i] - S[i+1])
            if min_diff == None or new_diff < min_diff:
                min_diff = new_diff
 
        print min_diff
