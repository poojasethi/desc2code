#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
int main()
{
int t,n,mini;
scanf("%d",&t);
while(t--)
{
int i;
scanf("%d",&n);int a[n];
for(i=0;i<n;i++)
{
scanf("%d",&a[i]);
}
sort(a,a+n);
mini=a[1]-a[0];
for(i=1;i<n-1;i++)
{
if((a[i+1]-a[i])<mini)
    mini=(a[i+1]-a[i]);
}
printf("%d\n",mini);


}
return 0;
}
