#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main(){
	long int t,n,m;
	cin>>t;
	for(long int i=0;i<t;i++){
		cin>>n;
		vector<long int> v;
		for(long int j=0;j<n;j++){
			cin>>m;
			v.push_back(m);
		}
		long int siz = v.size(),u;
		sort(v.begin(),v.end());
		long int min = v[1]-v[0];
		for(long int ii=0;ii<siz-1;ii++){
			u = v[ii+1]-v[ii];
			//cout<<u<<"\t";
			if(u<min)
				min=u;
		}
		cout<<min<<"\n";
	}
	return 0;
}
