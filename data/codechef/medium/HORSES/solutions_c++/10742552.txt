#include <stdio.h>
#include <iostream>
#include <algorithm>

using namespace std;

int main() {
	int T;
	cin >> T;
	for(int t=0; t<T; t++) {
		int N;
		cin >> N;
		long long S[N];
		for(int n=0;n<N;n++)
			cin >> S[n];
		sort(S,S+N);
		long long min = 1000000001;
		for(int i=N-2;i>=0;i--){
			long long diff = S[i+1]-S[i];
			if(diff<min)
				min = diff;
		}
		cout << min << endl;
	}
}
