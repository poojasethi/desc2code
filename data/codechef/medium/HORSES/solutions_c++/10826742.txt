#include<bits/stdc++.h>
#define ll   long long int
using namespace std;


int main()
{ ll n,m,o,t;

  cin>>t;
  while(t--)
  {     vector<ll>a;
        cin>>n;
        while(n--)
        {
              cin>>m;
              a.push_back(m);
        }
        sort(a.begin(),a.end());
        o=1000000009;
        for(int i=1; i<a.size(); i++)
        {
              o=min(o,a[i]-a[i-1]);
        }
        cout<<o<<endl;


  }


}
