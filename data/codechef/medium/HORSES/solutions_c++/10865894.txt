#include<iostream>
#include<algorithm>
#include<climits>
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t)
    {
        int i,n;
        cin>>n;
        int *a=new int[n];
        for(i=0;i<n;i++)
        {
            cin>>a[i];
        }
        sort(a,a+n);
        int min=INT_MAX;
        for(i=1;i<n;i++)
        {
            if(a[i]-a[i-1]<min)
                min=a[i]-a[i-1];
        }
        cout<<min<<endl;
        t--;
    }
    return 0;
}
