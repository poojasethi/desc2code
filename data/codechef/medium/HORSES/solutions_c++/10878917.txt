#include<stdio.h>
#include<stdlib.h>
#define ll long long int
#define min(a, b) (((a) <= (b)) ? (a) : (b))

int cmpfunc(const void *a, const void *b){
	return(*(ll *)a - *(ll *)b);
}

int main(){
	ll t, n; scanf("%lld", &t);
	while(t-- && scanf("%lld", &n)){
		ll a[n], i, d;
		for(i = 0; i < n; scanf("%lld", &a[i++]));
		qsort(a, n, sizeof(ll), cmpfunc),d = a[1] - a[0];
		for(i = 1; i < n - 1; i++)	d = min(d, a[i + 1] - a[i]);
		printf("%lld\n", d);
	}		
	return 0;
}
