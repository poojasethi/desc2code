
#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int t,n;
	scanf("%d\n",&t);
	//long int first,second;
	while(t--){
	scanf("%d",&n);
	long int arr[n];
	for(int i=0;i<n;i++)
		scanf("%ld",&arr[i]);
	sort(arr,arr+n);
	int min1= 1000000007;
	for(int i=0;i<n-1;i++)
	 {
	 	int diff = arr[i+1]-arr[i];
	 	if(diff<min1)
	 	{
	 		min1 =diff;
	 	}
	 }
	 printf("%ld\n",min1);
   }
	 
	 return 0;
}
