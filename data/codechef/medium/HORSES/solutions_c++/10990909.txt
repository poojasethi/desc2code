#include<bits/stdc++.h>
using namespace std ;
int main()
{  
  int tc ,n ,i ,j;
  cin>>tc;
  while(tc--)
  {
  	cin>>n ;
  	long long int arr[n],min ;
  	for(i=0;i<n;i++)
  	{
  		cin>>arr[i] ;
	  }
	  sort(arr,arr+n) ;
	  min =arr[1]-arr[0] ;
	  for(i=1,j=2;i<n-1;i++,j++)
	  {
	  	 if(arr[j]-arr[i]<min)
	  	 {
	  	 	min=arr[j]-arr[i] ;
		   }
	  }
	cout<<min<<endl ;
  }
}
