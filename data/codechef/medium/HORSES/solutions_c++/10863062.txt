#include<bits/stdc++.h>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int n;
        cin>>n;
        int s[n];
        for(int j=0;j<n;j++)
            cin>>s[j];
        int miN=INT_MAX;
        for(int j=0;j<n-1;j++)
        {
            for(int k=j+1;k<n;k++)
            {
                int num=s[j]-s[k];
                num=abs(num);
                miN=(miN>num)?num:miN;
            }
        }
        cout<<miN<<endl;
    }
}
