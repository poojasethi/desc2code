#include<iostream>
#include <vector>
#include<algorithm>
using namespace std;
int main()
{
	int t,i;
	cin>>t;
	for(i=0;i<t;i++)
	{
		int n;
		cin>>n;
		long int s[n];
		for(int j=0;j<n;j++)
		  cin>>s[j];
		sort(s,s+n);
		long int dif=1000000000;
		for(int k=0;k<n-1;k++)
		{
			if((s[k+1]-s[k])<dif)
			 dif=s[k+1]-s[k];
		}  
		
		cout<<dif<<endl;
	}
}
