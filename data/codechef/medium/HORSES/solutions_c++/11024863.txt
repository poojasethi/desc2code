#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        int n;
        vector<long long int> s;
        cin >> n;
        for(int i = 0; i < n; i++)
        {
            long long int a;
            cin >> a;
            s.push_back(a);
        }
        sort(s.begin(), s.end());
        long long int min = s[1] - s[0];
        for(int i = 0; i < n - 1; i++)
        {
            if((s[i + 1] - s[i]) < min)
                min = s[i + 1] - s[i];
        }
        cout << min << endl;
    }
    return 0;
}
