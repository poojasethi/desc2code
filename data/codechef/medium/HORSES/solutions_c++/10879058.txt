#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T,N,S[5000],i,min;
    cin>>T;

    while(T--)
    {
        cin>>N;
        for(i=0;i<N;i++)
        cin>>S[i];

        sort(S,S+N);


        min=S[1]-S[0];

        for(i=1;i<N-1;i++)
         if((S[i+1]-S[i])<min)
         min=S[i+1]-S[i];

    cout<<min<<endl;
    }
return 0;
}
