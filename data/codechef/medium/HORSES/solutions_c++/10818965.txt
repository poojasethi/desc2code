#include<iostream>
using namespace std;
 
void bubbleSort(int a[],int n){
    for(int i=0;i<n;i++){
        for(int j=0;j<n-1;j++){
             if(a[j]>a[j+1]){
                 int t=a[j];
                 a[j]=a[j+1];
                 a[j+1]=t;} } } }
 
int main(){
        int numbers;
        cin>>numbers;
        
        int s=0;
        while(s<numbers){
            int n;
            cin>>n;
            int a[n];
            for(int i=0;i<n;i++){
                  cin>>a[i];}
            bubbleSort(a,n);
            int minDiff=a[1]-a[0];
            int j=1;
            while(j<n-1){
                int diff=a[j+1]-a[j];
                if(minDiff>diff){
                    minDiff=diff;}
                j=j+1;}
           cout<<minDiff<<endl;
           s=s+1;}  }
