#include <iostream>
#include<algorithm>

using namespace std;


int main(void){
    static int arr[10][5000],N[10],T,i,j,temp,diff;
    cin>>T;
    for(i=0;i<T;i++){
        cin>>N[i];
        for(j=0;j<N[i];j++){
            cin>>arr[i][j];
        }
        sort(arr[i],arr[i]+N[i]);
    }

    for(i=0;i<T;i++){
        temp=arr[i][1]-arr[i][0];
        for(j=1;j<(N[i]-1);j++){
            diff=arr[i][j+1]-arr[i][j];
            if(diff>temp){
                temp=temp;
            }
            else{
                temp=diff;
            }
        }
        cout<<temp<<"\n";
    }
    return 0;
}
