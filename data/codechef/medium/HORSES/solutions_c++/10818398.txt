#include <stdio.h>
#include <algorithm>
#include <limits.h>

using namespace std;

int main() {

	int T,N;
	scanf("%d",&T);

	while(T--) {
		scanf("%d",&N);
		int iArrSkills[N];
		for(int i=0;i<N;i++) {
			scanf("%d",&iArrSkills[i]);
		}

		sort(iArrSkills,iArrSkills+N);

		int iMinDifference = INT_MAX;

		for(int i=1;i<N;i++) {
			int iCurDiff = iArrSkills[i] - iArrSkills[i-1];
			if (iCurDiff < iMinDifference) {
				iMinDifference = iCurDiff;
			}
		}

		printf("%d\n",iMinDifference);
	}

	return 0;
}
