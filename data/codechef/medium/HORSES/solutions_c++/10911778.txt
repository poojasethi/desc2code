#include <iostream>
#include <algorithm>
#include <climits>
using namespace std;

int main() {
	long int t;
	long int n;
	long int diff;
	cin >> t;
	while(t--) {
		cin >> n;
		long int s[n+1];
		for(int i=1; i<n+1; i++) 
			cin >> s[i];
		sort(s+1, s+n+1);
		long int min_diff = 1000000000;
		for(int i=2; i<n+1; i++) {
			diff = s[i] - s[i-1];
			if(diff < min_diff) {
				min_diff = diff;
			}
		}
		cout << min_diff << endl;
	}
	return 0;
}
