#include<iostream>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n,t,s[10000],arr[10000],ans;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>s[i];
		}
		sort(s,s+n);
		
		for(int i=0;i<n-1;i++)
		{
			arr[i]=s[i+1]-s[i];
		}
		sort(arr,arr+n-1);
		ans=arr[0];
		cout<<"\n "<<ans;
		
	}
	return 0;
}
