#include<iostream>
using namespace std;

int main()
{
  int t,i,k,j,n;
  long long s,small;
  cin>>t;
  for(i=1;i<=t;i++)
  {
    cin>>n;
    long long a[n];
    for(j=0;j<n;j++)
      cin>>a[j];
    small=a[0]-a[1];
    if(small<0)
        small=small*-1;
    for(j=0;j<n-1;j++)
    {
      for(k=j+1;k<n;k++)
      {
        s=a[j]-a[k];
        if(s<0)
          s=s*-1;
        if(s<small)
            small=s;
      }
    }
    cout<<small<<endl;
  }
  return 0;
}
