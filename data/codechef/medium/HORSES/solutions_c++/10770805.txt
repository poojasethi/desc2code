#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,i,d;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int ar[n];
        for(i=0;i<n;i++)
        cin>>ar[i];
        sort(ar,ar+n);
        d=ar[1]-ar[0];
        for(i=1;i<n-1;i++)
        {
            if(ar[i+1]-ar[i]<d)
            d=ar[i+1]-ar[i];
        }
        cout<<d<<endl;
    }
    return 0;
}
