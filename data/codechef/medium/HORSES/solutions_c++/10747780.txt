#include <stdio.h>
#include <vector>
#include <algorithm>

using namespace std;

void printVector(vector<int> x)
{
	printf("\n");
	for(int i=0; i<x.size(); i++)
	{
		printf("%d ", x[i]);
	}
}
int main()
{
	int t;
	scanf("%d", &t);

	for(int j=0; j<t; j++)
	{
		int n;
		scanf("%d", &n);

		vector<long long> horses; 
		for(int i=0; i<n; i++)
		{
			long long elem;
			scanf("%lld", &elem);

			horses.push_back(elem);
		}

		sort(horses.begin(), horses.end());

		long long diff = horses[1]-horses[0];
		for(int i=1; i<=n-2; i++)
		{
			long long tempdiff = horses[i+1] - horses[i];
			if(tempdiff<diff)
			{
				diff = tempdiff;
			}
		}
		printf("%lld\n", diff);
	}

	return 0;
}
