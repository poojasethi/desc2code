
#include<bits/stdc++.h>


using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t!=0)
    {
        int i,j;
       long long int n;
        cin>>n;
        int s[n];
        for(i=0;i<n;i++)
        {
            cin>>s[i];
        }
        int a;
        a=abs(s[0]-s[1]);
        for(i=0;i<n-1;i++)
        {
            for(j=i+1;j<n;j++)
            {
                if(abs(s[i]-s[j])<a)
                    a=abs(s[i]-s[j]);
            }
        }
        cout<<a<<endl;
        a=0;
        t--;

    }
return 0;}
