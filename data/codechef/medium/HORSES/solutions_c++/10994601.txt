#include <iostream>
#include <algorithm>

using namespace std;

bool myfunction(int i, int j) { return (i<j); }

int getMinDiff(int *horse_skills, int n){

	int diff = 1000000000;
	for (int i = 2; i <= n; i++){
		if (diff >(horse_skills[i - 1] - horse_skills[i - 2])){
			diff = (horse_skills[i - 1] - horse_skills[i - 2]);
		}
	}
	return diff;
}

int main(){
	int T;
	cin >> T;

	for (int i = 1; i <= T; i++){		
		int N;
		cin >> N;
		int horse_skills[100000];

		for (int j = 1; j <= N; j++){
			int skills;
			cin >> horse_skills[j - 1];
		}

		std::sort(horse_skills, horse_skills + N, myfunction);
		int diff = getMinDiff(horse_skills, N);
		cout << diff << "\n";
	}

	return 0;
}
