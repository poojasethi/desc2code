#include<bits/stdc++.h>
using namespace std;

long long int findMinDiff(long long int arr[],long long int n)
{
    sort(arr, arr+n);
   long long int diff = abs(arr[0]-arr[1]);
   for (long long int i=0; i<n-1; i++)
      if (arr[i+1] - arr[i] < diff)
          diff = arr[i+1] - arr[i];
   return diff;
}

 int main()
 {
    long long  int tc,i,n;
     scanf("%lld",&tc);

     while(tc--)
     {
         scanf("%lld",&n);
        long long int h[n];
         for(i=0; i<n; i++)
            scanf("%lld",&h[i]);
        long long int dif = findMinDiff(  h,  n);
        printf("%lld\n",dif);
     }
 return 0;
 }
