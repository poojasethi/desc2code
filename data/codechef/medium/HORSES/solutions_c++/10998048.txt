#include <iostream>
#include <algorithm>
#include <climits>
#include <cmath>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin >> t;
	for(int i = 0; i < t; i++){
		int n;
		cin >> n;
		long int a[10000];
		for(int j = 0; j < n; j++)
			cin >> a[j];
		sort(a, a+n);
		long int result[10000]={0};
		for(int j = 1; j < n; j++){
			result[j] = abs(a[j-1]-a[j]);
		//	cout << result[j]<<" ";
		}
		
		long int max = INT_MAX;
		for(int j = 1; j <n; j++){
			if(max >= result[j]){
				max = result[j];
			}
		}
		cout << max << "\n";
	}
	return 0;
}
