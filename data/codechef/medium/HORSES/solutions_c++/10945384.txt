#include<iostream>
using namespace std;
void merge(long a[],long l,long m,long r)
{
	long i,j,k,n1=m-l+1,n2=r-m;
	long L[n1],R[n2];
	for(i=0;i<n1;i++) L[i]=a[l+i];
	for(j=0;j<n2;j++) R[j]=a[m+1+j];
	i=0; j=0; k=l;
	while(i<n1&&j<n2)
	{
		if(L[i]<=R[j]) a[k++]=L[i++];
		else a[k++]=R[j++];
	}
	while(i<n1) a[k++]=L[i++];
	while(j<n2) a[k++]=R[j++];
}
void sort(long a[],long l,long r)
{
	if(l<r)
	{
		long m=l+(r-l)/2;
		sort(a,l,m);
		sort(a,m+1,r);
		merge(a,l,m,r);
	}
}
int main()
{
	long t,n; cin>>t;
	while(t--)
	{
		cin>>n;
		long s[n];
		for(long i=0;i<n;i++) cin>>s[i];
		sort(s,0,n-1);
		long min=1000000000;
		for(long i=0;i<n-1;i++)
		{
			if((s[i+1]-s[i])<min) min=s[i+1]-s[i];
		}
		cout<<min<<endl;
	}
}
