#include <iostream>
#include<algorithm>
using namespace std;

int main() {
int t;
scanf("%d",&t);
while(t--)
{
long long int n;
long long int max=999999999;
	scanf("%lld",&n);
	long long int a[n];
	for(int i=0;i<n;i++)
	scanf("%lld",&a[i]);
	sort(a,a+n);
	for(int i=n-1;i>0;i--)
	{
	//printf("%lld ",a[i]);
	if(a[i]-a[i-1]<max)
	max=a[i]-a[i-1];
	}
	printf("%lld\n",max);
}
	return 0;
}
