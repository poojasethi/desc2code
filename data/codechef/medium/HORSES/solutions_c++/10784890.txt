#include <iostream>
#include<algorithm>
using namespace std;

int main() {
	int i,t;
	long long int *a,min,diff,n;
	cin>>t;
	while(t--) //:P should have mentioned it before
	{
		cin>>n;
		a=new long long int[n];
		for(i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		min=a[1]-a[0];
		for(i=1;i<n-1;i++)
		{
			diff=a[i+1]-a[i];
			if(diff<min)
			min=diff;
		}	
		cout<<min<<endl;
	}	
		return 0;
}
