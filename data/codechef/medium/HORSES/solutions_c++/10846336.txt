#include<stdio.h>
#include<math.h>
#include<stdlib.h>
long long int min, d[5000][5000],S[5000];
int main()
{
    long long int  N,  i, j, set1, set2;
    int t;
    scanf("%d", &t);
    while(t--)
    {
        scanf("%lld", &N);
        for(i=0; i<N; i++)
        {
            scanf("%lld", &S[i]);
        }
        for(i=0; i<N; i++)
        {
            for(j=i+1; j<N; j++)
            {
                d[i][j]=abs(S[i]-S[j]);
                min=d[i][j];
            }
        }
        for(i=0; i<N; i++)
        {
            for(j=i+1; j<N; j++)
            {
                if(min>d[i][j])
                {
                    min=d[i][j];
                    set1=i;
                    set2=j;
                }
            }
        }
        printf("%lld\n",min);
    }
    return 0;
}
