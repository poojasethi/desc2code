//#include<stdlib.h>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{int t,i,j,n,k,min,l;
long long int a[100000],diff[100000];
scanf("%d",&t);
for(i=0;i<t;i++)
   {int m=0;
    scanf("%d",&n);
    for(j=0;j<n;j++)
       scanf("%lld",&a[j]);
    sort(a,a+n);
    for(k=1;k<=(n-1);k++)
        {diff[m]=a[k]-a[k-1];
        m++;
		}
	min=diff[0];
	for(l=1;l<m;l++)
	    {if(diff[l]<min)
	       min=diff[l];
		}
	printf("%d\n",min);	  
   }
return 0;
}
