#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,k,m;
        scanf("%d %d %d",&n,&k,&m);
        int a[n];
        int b[m+k];
        for(int i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        int temp;
        for(int i=0;i<n;i++)
        {
            scanf("%d",&temp);
            a[i]-=temp;
        }
        for(int i=0;i<m+k;i++)
        {
            scanf("%d",&b[i]);
        }
        sort(a,a+n);
        sort(b,b+m+k);
        int p2=m+k-1;
        int p1=n-1;
        while(p1>=0 && p2>=0 && a[p1]!=0)
        {
            if(b[p2]>a[p1])
                p2--;
            else
            {
                a[p1]-=b[p2];
                p2--;
                p1--;
            }
        }
        long long int sum=0;
        for(int i=0;i<n;i++)
        {
            sum+=(long long int)a[i];
        }
        printf("%lld\n",sum);
    }
    return 0;
}
