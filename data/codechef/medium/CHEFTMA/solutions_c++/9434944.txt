#include <bits/stdc++.h>

using namespace std;
int planned[100005],complete[100005];
int main(){
	int t,n,k,m,i,j,x;
	multiset<int>button;
	cin>>t;
	while(t--){
		button.clear();
		cin>>n>>k>>m;
		long long int ans=0;
		for(i=0;i<n;++i)
			cin>>planned[i];
		for(i=0;i<n;++i){
			cin>>complete[i];planned[i]-=complete[i];ans+=planned[i];}
		for(i=0;i<k;++i){
			cin>>x;
			button.insert(x);
		}
		for(i=0;i<m;++i){
			cin>>x;
			button.insert(x);
		}
	
		//sort(planned,planned+n);
		multiset<int>::iterator it;
		for(i=0;i<n;++i){
			x = planned[i];
			//cout<<x<<endl;
			if(x==0)
				continue;
			it = button.upper_bound(x);
			if(it==button.begin())
				continue;
			it--;
			//cout<<*it<<endl;
			button.erase(it);
			ans-=(*it);
		}	
		cout<<ans<<endl;
	}

	return 0;
}
