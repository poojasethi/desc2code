#include <bits/stdc++.h>
using namespace std;
 
int main() {
    int t;
    cin >> t;
    while (t--) {
        int n, k, m;
        cin >> n >> k >> m;
        vector<int> a(n);
        for (int i = 0; i < n; ++i)
            cin >> a[i];
        for (int i = 0; i < n; ++i) {
            int dx;
            cin >> dx;
            a[i] -= dx;
        }
        long long incomplete_tasks = accumulate(a.begin(), a.end(), 0LL);
        sort(a.begin(), a.end(), greater<int>());
        vector<int> b(k + m);
        for (int i = 0; i < k + m; ++i)
            cin >> b[i];
        sort(b.begin(), b.end(), greater<int>());
        for (int i = 0, j = 0; i < n && j < k + m; ++j) {
            if (b[j] <= a[i]) {
                incomplete_tasks -= b[j];
                ++i;
            }
        }
        cout << incomplete_tasks << '\n';
    }
}
