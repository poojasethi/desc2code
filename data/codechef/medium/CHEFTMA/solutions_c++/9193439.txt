#include<cstdio>
using namespace std;
long long int ans,var,mvar;
int a[100005],b[100005],c[100005],s[200005];
int main()
{
    int n,i,j,t,k,m,v,top;
    scanf("%d",&t);
    while(t--)
    {
        ans=0;top=0;
        scanf("%d %d %d",&n,&k,&m);
        for(i=1;i<=n;i++)
        {
            scanf("%d",&a[i]);
        }
        for(i=1;i<=n;i++)
        {
            scanf("%d",&v);
            b[a[i]-v]++;
        }
        for(i=0;i<k;i++)
        {
            scanf("%d",&v);
            c[v]++;
        }
        for(i=0;i<m;i++)
        {
            scanf("%d",&v);
            c[v]++;
        }
        top=0;
        for(i=1;i<=100000;i++)
        {
            j=c[i];
            while(j--)
            {
                s[top]=i;
                top++;
            }
        }
        i=100000;j=top-1;
        for(j=top-1;j>=0;j--)
        {
            while(i>=0&&b[i]==0)
                i--;
            if(i<0)
                break;

            if(i>=s[j])
            {
                var=i-s[j];
                ans+=var;
                b[i]--;
            }
        }
        for(i=1;i<=100000;i++)
        {
            mvar=b[i];
            var=i;
            var*=mvar;
            ans+=var;
        }
        printf("%lld\n",ans);

        for(i=0;i<=100000;i++)
        {
            a[i]=b[i]=c[i]=s[i]=0;
        }
        for(;i<200002;i++)
            s[i]=0;
    }
    return 0;
}
