#include<bits/stdc++.h>
using namespace std;

bool myfunction (int i,int j) { return (i>j); }

int main()
{
    int t,n,m,k,a[100010],x,i,j,ans,c[200010];
    scanf("%d",&t);
    while(t--)
    {
        ans=0;
        scanf("%d%d%d",&n,&m,&k);
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        for(i=0;i<n;i++)
            {scanf("%d",&x);a[i]-=x;}
        for(i=0;i<m+k;i++)
            scanf("%d",&c[i]);
        sort(a,a+n,myfunction);
        sort(c,c+m+k,myfunction);
        //for(i=0;i<n;i++)
           // printf("%d ",a[i]);
        i=0;j=0;
        while(i<n&&j<m+k)
        {
            if(c[j]<=a[i])
            {
                a[i]=a[i]-c[j];
                j++;i++;
            }
            else
            {
                j++;
            }
        }
        i=0;
        while(i<n)
        {ans+=a[i];i++;}
        printf("%d\n",ans);
    }
    return 0;
}
