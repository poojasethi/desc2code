#include <iostream>
#include <bits/stdc++.h>

using namespace std;

int main()
{
    double H,s,a1,a2;
    int t;
    cin>>t;
    while(t--)
    {
        cin>>H>>s;
        if(H*H>=4*s)
        {
            a1=sqrt(H*H+4*s);
            a2=sqrt(H*H-4*s);
            cout.precision(2);
  cout << fixed << (a1 - a2) / 2.0 << ' ' << (a1 + a2) / 2.0 << ' ' << H << '\n';

        }
        else
            cout<<"-1"<<endl;
    }
}
