#include<bits/stdc++.h>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
typedef long double ll;
int main()
{
    long t;
    cin>>t;
    while(t--)
    {
        double h,s;
        //scanf("%Lf",&h,&s);
        cin>>h>>s;
        if(h*h < 4*s)
            cout<<-1<<endl;
        else
        {
            double s1 = sqrt(s);
            double b = sqrt(((h*h)+ sqrt((h*h)+4.0*s)*sqrt((h-2*s1)) * sqrt((h+2.0*s1)))/2.0);
            double l = sqrt((h-b)*(h+b));
            printf("%.6lf %.6lf %.6lf\n",l,b,h);

        }
    }
    return 0;
}
