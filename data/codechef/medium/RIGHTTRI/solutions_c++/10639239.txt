int main()
{
    long long int t;
    double h,s;
    cin>>t;
    while(t--)
    {
        scanf("%lf %lf",&h,&s);
        double bh= 2*s;
        double bplh = sqrt(h*h + 2*bh);
        if(h*h -2*bh < 0)
            cout<<"-1\n";
        else
        {
            double bmih = sqrt(h*h - 2*bh);
    //        printf("%lf %lf\n",bplh,bmih);
            double b = (bplh + bmih)/2;
            double h1 = bplh - b;
    //        printf("%lf %lf\n",b,h1);
            double a[3];
            a[0]=h;
            a[1]=b;
            a[2]=h1;
            sort(a,a+3);
            printf("%lf %lf %lf\n",a[0],a[1],a[2]);
        }
       
    }
    return 0;
}
