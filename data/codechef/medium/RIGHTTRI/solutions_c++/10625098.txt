#include<stdio.h>
#include<math.h>
int main(){
	int t;
	long long h,s;
	
	double a,b;
	scanf("%d", &t);
	while(t--){
		scanf("%lld %lld", &h, &s);
		long long sqsum=(h*h);
		long long pr=s*4;
		if(sqsum-pr<0){
			printf("-1\n");
		}
		else{
			double x=fabs(sqrt(sqsum+pr));
			double y=fabs(sqrt(sqsum-pr));
			printf("%lf %lf %lld\n", ((x-y)/2), ((x+y)/2), h);
		}
	}
	return 0;
}
