#include<bits/stdc++.h>
using namespace std;
main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
	double h,s;
	scanf("%lf %lf",&h,&s);
	if(h*h<4*s){
		printf("-1\n");continue;
	}
	double a,b;
	a=sqrt((sqrt(h*h-4*s)*sqrt(h*h+4*s)+h*h)/2);
	b=sqrt(h*h-a*a);
	if(a>b)swap(a,b);
	printf("%lf %lf %lf\n",a,b,h);
	}
}
