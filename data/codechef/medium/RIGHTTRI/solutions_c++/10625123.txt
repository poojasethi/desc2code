
#include<bits/stdc++.h>
using namespace std;
int main(){
 
 
int t;
cin>>t;
 
while(t--){
 
 
long long h,s;
cin>>h>>s;
 
if(h*h-4*s<0)
{
	printf("-1\n");
	continue;
}
double h1=sqrt(h*h+4*s);
double h2=sqrt(h*h-4*s);
double sum=h1+h2;
sum=sum/2;
vector<double> hh;
hh.push_back(sum);
if(sum-h2<0) {
printf("-1\n");
continue;
}
hh.push_back(sum-h2);
hh.push_back(h);
sort(hh.begin(),hh.end());
printf("%lf %lf %lf\n",hh[0],hh[1],hh[2]);
 
}
 
return 0;
}
