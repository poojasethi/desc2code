    #include <bits/stdc++.h>
    #define fast ios_base::sync_with_stdio(0)
    using namespace std;
    int main() {
    fast;
    int test;
    scanf("%d",&test);
    while(test--){
        long double a,b,H,s,temp;
        scanf("%llf%llf",&H,&s);
        //scan(&s);
        temp=H*H-4*s;//(h*h+4*s)(h*h-4*s)
        if(temp>=0){
        a=H*H/2-sqrtl(temp)*sqrtl(H*H+4*s)/2;
        a=sqrtl(a);
        b=H*H/2+sqrtl(temp)*sqrtl(H*H+4*s)/2;
        b=sqrtl(b);
        printf("%0.11llf %0.11llf %0.11llf\n",a,b,H);
        }
        else
            printf("-1\n");
     
    }
