#include <bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	cin >> t;
	double a,b[5];
	double h,s;
	while (t--)
	{
		cin >> h;
		cin >> s;
		double p = (h*h*h*h) - (16*s*s);
		if (p >= 0)
		{
			a = ((h*h) + sqrt(p))/2;
			b[0] = sqrt(a);
			
		//	cout << "a1 = " << setprecision(10) << b <<"\n";
			a = ((h*h) - sqrt(p))/2;
			b[1] = sqrt(a);
			b[2] = h;
			sort(b,b+3);
			printf ("%lf %lf %lf\n", b[0],b[1],b[2]);
		}
		else
			cout << "-1\n";
	//	cout << "a2 = " << b <<"\n";
	}
	return 0;
}
