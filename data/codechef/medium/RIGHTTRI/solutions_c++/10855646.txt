#include <iostream>
#include <map>
#include <climits>
#include <cmath>
#include <cstdio>
#include <cstdlib>
using namespace std;

int main() {
	int T;
	cin>>T;
	while(T--){
        
        double H,S ;
        cin>>H>>S;
        
        double sqH = H*H;
        double sqS = S*S;
        double sq_sqH = sqH * sqH ;
        
        double f1_cof = (sqH - sqrt(sq_sqH - (16 * sqS)))/2;
        double f2_cof = (sqH + sqrt(sq_sqH - (16 * sqS)))/2;
        
        if(f1_cof>0){
            f1_cof = sqrt(f1_cof);
        }
        if(f2_cof>0){
            f2_cof = sqrt(f2_cof);
        }
        //cout <<f2_cof <<" "<<f1_cof <<" "<< H <<endl;
        double s_cof, f_cof;
        if(f1_cof > f2_cof){
            s_cof = ((2*S)/f1_cof);
            f_cof = f1_cof;
        }else{
            s_cof = ((2*S)/f2_cof);
            f_cof = f2_cof;
        }
        //cout <<s_cof <<" "<<f_cof <<" "<< H <<endl;
        if(s_cof > 0 && f_cof>0){
            if(s_cof < f_cof){
                if(s_cof < H){
                    if(H<f_cof){
                        printf("%.6F %.6F %.6F\n",s_cof , H ,f_cof );
                    }else{
                        printf("%.6F %.6F %.6F\n",s_cof ,f_cof,H );   
                    }
                }else{
                    printf("%.6F %.6F %.6F\n",H , s_cof,f_cof);
                }
            }else{
                if(f_cof < H){
                    if(s_cof<H){
                        printf("%.6F %.6F %.6F\n",f_cof,s_cof , H );
                    }else{
                        printf("%.6F %.6F %.6F\n",f_cof, H , s_cof);
                    }
                }else{
                    printf("%.6F %.6F %.6F\n",H ,f_cof, s_cof);
                }
            }
        }else{
            printf("-1\n");
        }
        
        
        
	}
	return 0;
}
