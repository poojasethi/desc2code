# include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        double h,a,temp,temp1,temp2,temp3,temp4,temp5;
        scanf("%lf%lf",&h,&a);
        vector <double> ab;
        ab.push_back(h);
        a=4.0*a;
        temp1=h*h;
        temp=temp1-a;
        temp2=temp1+a;
        if(temp<0)
        {
            cout<<"-1"<<endl;
        }
        else
        {
            temp3=pow(temp,(1.0/2.0));
            temp4=pow(temp2,(1.0/2.0));
            temp5=temp3+temp4;
            temp5=temp5/2.0;
            ab.push_back(temp5);
            temp5=temp4-temp3;
            temp5=temp5/2.0;
            ab.push_back(temp5);
            sort(ab.begin(),ab.end());
            printf("%lf %lf %lf\n",ab[0],ab[1],ab[2]);
            //cout<<ab[0]<<" "<<ab[1]<<" "<<temp5<<endl;
        }
    }
}
