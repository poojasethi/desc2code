#include <bits/stdc++.h>
using namespace std;
 
int main() {
	// your code goes here
	
	int t; 
	cin >> t; 
	double h, s;
	while (t--)
	{
		cin >> h >> s; 
		if (4*s > h*h) // negative d. shouldn't matter.
		{
			printf("-1\n");
			continue;
		}
		double a, b, val1, val2; 
		val1 = sqrt(h*h + 4*s); 
		val2 = sqrt(h*h - 4*s);
		a = (val1 - val2)/2.0;
		b = (val1 + val2)/2.0;
		printf("%lf %lf %lf\n", a, b, h);
	}
	
	return 0;
}
