#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <math.h>
using namespace std;
 
int main() {
    long t;
    double h,s,apb,amb,a,b,temp;
    cin>>t;
    while(t>0)
    {
        cin>>h>>s;
        apb=1.0*sqrt(1.0*(h*h+4*s));
        amb=1.0*sqrt(1.0*(h*h-4*s));
        a=1.0*(apb+amb)/2;
        b=1.0*(apb-amb)/2;
        if(a>0 && b>0)
        {
            if(b>a)
            {
                temp=b;
                b=a;
                a=temp;
            }
        printf("%0.6lf %0.6lf %0.6lf\n",b,a,h);
        }
        else
        cout<<-1<<endl;
        --t;
    }
	return 0;
}
