#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        double h,s,a,b;
        cin>>h>>s;
       // cout<<pow(h,2)<<endl;
        a=(pow(h,2)+4*s);
        b=(pow(h,2)-4*s);

        if(a<=b)
        {
            printf("-1\n");
        }
        else if(a<0||b<0)
        {
            printf("-1\n");
        }
        else
        {
            a=sqrt(a);
            b=sqrt(b);
            double side1,side2;
            side1=(a+b)/(2.0);
            side2=(a-b)/(2.0);
            printf("%0.4lf %0.4lf %0.4lf\n",side2,side1,h);
        }
    }
}
