#include <bits/stdc++.h>
 
using namespace std;
 
int n,m;
 
int main()
{
    int t;
    cin>>t;
    while(t--){
        long double h,s;
        cin>>h>>s;
        bool f=1;
        long double a,b,D1,D2;
        //cout<<h*h-4*s<<" ";
        if(h*h-4*s<0)
        f=0;
        else{
            D1=h*h+4*s;
            D2=h*h-4*s;
            D1=sqrt(D1);
            D2=sqrt(D2);
            a=D1+D2;
            a=a/2;
            b=D1-D2;
            b=b/2;
        }
        if(f)
        cout<<fixed<<setprecision(8)<<b<<" "<<a<<" "<<h;
        else
        cout<<"-1";
        cout<<"\n";
    }
    return 0;
}
