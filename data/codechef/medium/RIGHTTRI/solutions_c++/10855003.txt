#include <bits/stdc++.h>
using namespace std;
#define lf long double
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		lf h,s;
		cin>>h>>s;
		lf pro = 2*s;
		lf sumsq = h*h;
		if(sumsq-2*pro<0)
		{
			cout<<"-1\n";
			continue;
		}
		double x1 = sqrt(sumsq+2*pro);
		double x2 = sqrt(sumsq-2*pro);
		double a[]={(x1+x2)/2,(x1-x2)/2,h};
		sort(a,a+3);
		printf("%.6f %.6f %.6f\n",a[0],a[1],a[2]);
	}
}
