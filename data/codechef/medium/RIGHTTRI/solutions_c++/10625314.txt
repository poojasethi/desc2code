#include<iostream>
#include<math.h>
#include<iomanip>
using namespace std;
int main()
{
 long double t;
 int v=-1;
 cin>>t;
 while(t--)
 {
    long double h,s,p,b,x,y,r;
    long double a[3];
    int i,j;
    cin>>h>>s;
    if((h*h-4.0*s)>=0){
    x=pow((h*h+4.0*s),0.5);
    y=pow((h*h-4.0*s),0.5);
    p=(x+y)/2.0;
    b=(x-y)/2.0;
    //cout<<p<<b;
    a[0]=p;
    a[1]=b;
    a[2]=h;
for(i=0;i<2;i++)
 {
  for(j=0;j<2-i;j++)
   {
    if(a[j]>a[j+1])
     {
       r=a[j];
       a[j]=a[j+1];
       a[j+1]=r;
     }
   }
 }
for(i=0;i<3;i++)
    cout<<fixed<<showpoint<<setprecision(9)<<a[i]<<" ";
    cout<<endl;
    }
 else
    cout<<v<<endl;
}
return 0;
}
