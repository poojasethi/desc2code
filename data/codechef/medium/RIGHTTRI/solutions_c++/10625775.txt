
 #include<bits/stdc++.h>
#define bug(x) cout<<#x<<" => "<<x<<"\n";
#define pr(x) cout<<#x<<"\n";
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		double h,s;
		cin>>h>>s;
		double hsq = h*h;
		double area = 4*s;
		if(hsq<area)
		{
			cout<<"-1\n";
			continue;
		}
		else
		{
			double ans = ((sqrt(hsq+area)+sqrt(hsq-area))/2);
			double ans2 = (sqrt(hsq-(ans*ans)));
			if(ans>ans2)
				printf("%.2f %.2f %.2f\n",ans2,ans,h);
			else
				printf("%.2f %.2f %.2f\n",ans,ans2,h);
			

		}

	}

}
