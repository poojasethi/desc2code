#include<bits/stdc++.h>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    long t;
    cin>>t;
    while(t--)
    {
        double h,s;
        //scanf("%Lf",&h,&s);
        cin>>h>>s;
        if(h*h < 4*s){
            cout<<-1<<endl;
        }
        else {
            double a = sqrt(h*h + 4*s);
            double b = sqrt(h*h - 4*s);
            printf("%.6lf %.6lf %.6lf\n",(a-b)/2.0,(a+b)/2.0,h);
        }
    }
    return 0;
}
