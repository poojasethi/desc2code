from math import sqrt
T = int(raw_input())
for i in range(T):
	H, S = raw_input().split(' ')
	H = int(H)
	S = int(S)
	D = pow(H,4) - 16 * S * S
	if D >= 0:
		root1 = (H * H + sqrt(D))/2
		root2 = (H * H - sqrt(D))/2
		B = sqrt(root1)
		P = sqrt(root2)
		if B >= P:
			print P, B, H
		else:
			print B, P, H
	else: 
		print "-1"
