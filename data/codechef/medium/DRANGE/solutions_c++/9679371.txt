#include<bits/stdc++.h>
using namespace std;


int main()
{
    int t,n,m,w,x,y,z;
    ios::sync_with_stdio(false);
    //scanf("%d",&t);
    cin>>t;
    while(t--)
    {
      // scanf("%d%d",&n,&m);
        cin>>n>>m;
        vector<int> v1(n+1,0);
        while(m--)
        {
            //scanf("%d%d%d%d",&w,&x,&y,&z);
            cin>>w>>x>>y>>z;
             if(w==1)
             {
                 v1[x]+=z;
                 if(y+1<=n)
                 v1[y+1]-=z;

             }
             else
             {
                  v1[x]-=z;
                  if(y+1<=n)
                 v1[y+1]+=z;

             }


        }


        for(int i=1;i<=n;i++)
            v1[i]+=v1[i-1];

            for(int i=1;i<=n;i++)
            v1[i]+=i;


            int m1=INT_MAX,m2=INT_MIN;
            for(int i=1;i<=n;i++)
            {

                    m2= max(v1[i],m2);

                    m1=min(v1[i],m1);

            }

          //printf("%d\n",m2-m1);
           cout<<m2-m1<<"\n";
    }


}
