	inline int inp()
    {
    int n=0;
    char p=getchar_unlocked();
    while(p>='0'&&p<='9')
    {
    n = (n<< 3) + (n<< 1) + (p - '0');
    p=getchar_unlocked();
    };
    return n;
    };
	int main()

		t=inp();
		while(t>0)
		{

			for(int i=1;i<=n;i++)
			{

			int num,x,y,z,max,min;

				{
					arr[x]+=z;

						arr[y+1]-=z;

				else
				{
					arr[x]-=z;
					if(y+1<=n)
						arr[y+1]+=z;

				min=arr[1];
