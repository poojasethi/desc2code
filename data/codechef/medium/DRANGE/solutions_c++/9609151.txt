#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;

int arr[1000009],temp[1000009];
int main()
{
	int t,n,m,x,y,z,w,i,mini,maxi;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(i=1;i<=n;i++)
		{
		arr[i]=i;
		temp[i]=0;
	    }
		while(m--)
		{
			scanf("%d%d%d%d",&w,&x,&y,&z);
			if(w==1)
			{
				temp[x]+=z;
				temp[y+1]-=z;
			}
		     else
		     {
		     	temp[x]-=z;
				temp[y+1]+=z;
		     }
		}
		arr[0]=0;
		temp[0]=0;
		for(i=1;i<=n;i++)
		{
			temp[i]=temp[i-1]+temp[i];
			arr[i]=arr[i]+temp[i];
		//	cout<<"arr="<<arr[i]<<"\n";
		}
		mini=arr[1];
		maxi=arr[1];
		for(i=2;i<=n;i++)
		{
			if(arr[i]<mini)
			mini=arr[i];
			if(arr[i]>maxi)
			maxi=arr[i];
		}
		printf("%d\n",maxi-mini);
	}
	return 0;
}
