int arr[1000000];

int main(){
	int t;
	scan(t);
	while(t--){
		int n,m;
		scan(n);
		scan(m);
		int i;
		arr[0] = 1;
		rep(i,1,n){
			arr[i] = i+1;
			d[i] = 1;
		}
		int w,x,y,z;
		rep(i,0,m){
			scan(w);scan(x);scan(y);scan(z);
			if(w==2){
				z=-z;
			}
			d[x-1]+=z;
			d[y]-=z;
		}
		int max = arr[0];
		int min = arr[0];
		rep(i,1,n){
			arr[i] = arr[i-1] + d[i];
			if(arr[i]>max){
				max = arr[i];
			}
			if(arr[i]<min){
				min = arr[i];
			}
		}
		print(max-min);
	}
