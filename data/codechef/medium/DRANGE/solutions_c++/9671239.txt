#define inf 1000000007

int main()
{
    int t,n,m,w,x,y,z,i,max_res,min_res,ans,j;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        vector<int> d(n+2,1);
        while(m--)
        {
            scanf("%d%d%d%d",&w,&x,&y,&z);
            if(w==2)
                z=-z;
            d[x-1]+=z;
            d[y]-=z;
        }
        max_res=-inf;
        min_res=inf;
        int j=0;
        for(i=1;i<=n;i++)
        {
            int temp=d[i-1]+j;
            max_res=max(max_res,temp);
            min_res=min(min_res,temp);
            j=temp;
            //cout<<temp<<" ";
        }
        ans=max_res-min_res;
        printf("%d\n",ans);
    }
}
