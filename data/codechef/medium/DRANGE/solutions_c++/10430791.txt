#include<bits/stdc++.h>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--){
	int n;
	cin>>n;
	int a[n+1];
	for(int i=0;i<n;i++)
	a[i]=1;
	int m;
	cin>>m;
	for(int i=0;i<m;i++)
	{
		int w,x,y,z;
		scanf("%d%d%d%d", &w, &x, &y, &z);
		//cin>>w>>x>>y>>z;
		if(w==2)
		z=-z;
		x--;
		y--;
		a[x]+=z;
		a[y+1]-=z;
		}
		int b=a[0];
		int imin=1e9,imax=-1e9;
		for(int i=0;i<n;i++,b+=a[i]){
			imin=min(imin,b);
			imax=max(imax,b);
		}
		printf("%d\n",imax-imin);
}
return 0;
}
