using namespace std;
int scan()
{
int t=0;
char c;
c=getchar_unlocked();
while(c<'0' || c>'9')
c=getchar_unlocked();
while(c>='0' && c<='9')
{
t=(t<<3)+(t<<1)+c-'0';
c=getchar_unlocked();
}
return t;
}
void out(int n) {
	if(n<0){
		n=-n;putchar_unlocked('-');
	}
	int i=10;
	char output_buffer[11];
	output_buffer[10]='\n';
	do{
		output_buffer[--i]=(n%10)+'0';n/=10;
	}while(n);
	do{
		putchar_unlocked(output_buffer[i]);
	}
	while(++i<11);
}
int main()
{
     int i,n,t,m,w,x,y,z,mini,maxi;
    t=scan();
    while(t--)
    {
        n=scan();
        m=scan();
        int track=1;
        vector<int> arr(n+1,1);
        while(m--)
        {
            w=scan();
            x=scan();
            y=scan();
            z=scan();
            if(w==2)
                z=-z;
                arr[x-1]+=z;
                arr[y]-=z;
                if(x==1)
                   track+=z;
        }
        mini=track;
        maxi=track;
        int temp;
        for(i=2;i<=n;i++)
        {
            temp=arr[i-1]+track;
            track=temp;
            if(mini>temp)
                mini=temp;
            if(temp>maxi)
                maxi=temp;
        }
        out(maxi-mini);
    }
    return 0;
}
