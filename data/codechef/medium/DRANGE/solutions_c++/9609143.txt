using namespace std;
#define ll long long
const ll mx=1000011;
const ll mod=1000000007;
int ar[mx+1],ss[mx+1];
int main()
{
  int i,j,k,t,r,l,n,v;
  scanf("%d",&t);
  while(t--)
  {
    int mn=INT_MAX;
  int m=INT_MIN;
    memset(ss,0,sizeof ss);
    scanf("%d%d",&n,&k);
    for(i=1;i<=n;i++)
      ar[i]=i;
    while(k--)
    {
      scanf("%d%d%d%d",&i,&l,&r,&v);
      if(i==2)
        v*=-1;
      ss[l]+=v;
      ss[r+1]-=v;
    }
    for(i=2;i<=n;i++)
      ss[i]+=ss[i-1];
    for(i=1;i<=n;i++)
      ar[i]+=ss[i];

    for(i=1;i<=n;i++)
    {
      mn=min(mn,ar[i]);
      m=max(m,ar[i]);
    }
    printf("%d\n",m-mn);
  }
  return 0;
}
