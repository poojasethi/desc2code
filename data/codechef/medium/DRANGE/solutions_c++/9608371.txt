#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() {
	int i,t,n,m,u,v,w,z;
	int maxi,mini;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		vector<int> dp(n+2);
		for(i=0;i<m;i++)
		{
			scanf("%d%d%d%d",&w,&u,&v,&z);
			if(w==1)
			{
				dp[u]+=z;
				dp[v+1]-=z;
			}
			else
			{
				z=-z;
				dp[u]+=z;
				dp[v+1]-=z;
			}
		}
		for(i=1;i<=n;i++)
			dp[i]+=dp[i-1];
		mini=maxi=1+dp[1];
		//cout<<(1+dp[1])<<endl;
		for(i=2;i<=n;i++)
		{
			//cout<<(i+dp[i])<<endl;
			mini=min(mini,i+dp[i]);
			maxi=max(maxi,i+dp[i]);
		}
		printf("%d\n",maxi-mini);
	}
	return 0;
}
