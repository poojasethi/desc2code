#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<climits>
#include<cstring>
using namespace std;
int main()
{
    int m,n,x,y,z,t,w;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&m);
        int *d=new int[n+1];
        memset(d,0,sizeof(int)*(n+1));
        while(m--)
        {
            scanf("%d %d %d %d",&w,&x,&y,&z);
            if(w==2)z=-1*z;
            d[x]+=z;d[y+1]-=z;
        }
        int cur=0,minn=INT_MAX,maxx=INT_MIN;
        for(int i=1;i<=n;i++)
        {
            cur+=d[i];
            if(cur+i<minn)minn=cur+i;
            if(maxx<cur+i)maxx=cur+i;
        }
        printf("%d\n",maxx-minn);
    }
    return 0;
}
