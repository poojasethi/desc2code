#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,ma=0,mi=0,w,x,y,z,i,m;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        int a[n],b[n+1];
        for(i=0;i<n;i++)
        {
            a[i]=i+1;
            b[i]=0;
        }
        b[n]=0;
        while(m--)
        {
            scanf("%d%d%d%d",&w,&x,&y,&z);
            if(w==1)
            {
                b[x-1]+=z;
                b[y]-=z;
            }
            else
            {
                b[x-1]-=z;
                b[y]+=z;
            }
        }
        for(i=1;i<=n;i++)
            b[i]=b[i]+b[i-1];
        for(i=0;i<n;i++)
        {
            a[i]=a[i]+b[i];
        }
        ma=a[0];
        mi=a[0];
        for(i=1;i<n;i++)
        {
            if(ma<a[i])
                ma=a[i];
            if(mi>a[i])
                mi=a[i];
        }
        printf("%d\n",(ma-mi));
    }
    return 0;
}
