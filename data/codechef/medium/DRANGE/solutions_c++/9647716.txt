#include<bits/stdc++.h>
using namespace std;
long long int a[1000001];

int main()
{
	ios::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		long int n,m,w,x,y,z,i;
		cin>>n>>m;
		for(i=1;i<=n;i++)
		a[i]=0;
		while(m--)
		{
			cin>>w>>x>>y>>z;
			if(w==1)
			{
				a[x]=a[x]+z;
				if((y+1)<=n)
				a[y+1]=a[y+1]-z;
			}
			if(w==2)
			{
				a[x]=a[x]-z;
				if((y+1)<=n)
				a[y+1]=a[y+1]+z;
			}
//				for(i=1;i<=n;i++)
////		cout<<" "<<a[i];
////		cout<<"\n";
		}
		for(i=2;i<=n;i++)
		a[i]=a[i]+a[i-1];
		for(i=1;i<=n;i++)
		a[i]=a[i]+i;
		
	
	 long long int min=a[1],max=a[1];
		for(i=1;i<=n;i++)
		{
			if(min>a[i])
			min=a[i];
			if(max<a[i])
			max=a[i];
		}
		cout<<max-min<<"\n";
	}
}
