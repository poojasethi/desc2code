#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

typedef long long ll;

#define gc getchar_unlocked
long long read() {
    char c = gc();
    while(c<'0' || c>'9') c = gc();
    long long ret = 0;
    while(c>='0' && c<='9') {
        ret = 10 * ret + c - 48;
        c = gc();
    }
    return ret;
}

ll A[1000005];

int main()
{
    int t;
    int n, m, w, x, y, z;
    ll range, max_num, min_num;
    t = read();
    while(t--)
    {
        n = read();
        m = read();
        memset(A, 0, sizeof(A));
        for(int i=0;i<m;i++)
        {
            w = read();
            x = read();
            y = read();
            z = read();
            if(w==2)
                z *= -1;
            A[x]   += z;
            A[y+1] -= z; 
        }
        min_num = 10000000000LL;
        max_num = -1 * min_num;
        ll diff = 0;
        for(int i=1;i<=n;i++)
        {
            diff += A[i];
            min_num = min(min_num, i+diff);
            max_num = max(max_num, i+diff);
        }
        range = max_num - min_num;
        printf("%lld\n", range);
    }
    return 0;
}
