#include <bits/stdc++.h>
#include<stdio.h>
using namespace std;
#define pii pair<long long,long long>
#define PI 3.14159265
#define ll long long
#define ff first
#define ss second
#define pb push_back
#define mp make_pair
#define INF 1000000009
#define mod 1000000007
ll n,m,day,u,v;
vector <pii> vec[55555];
pii ans[55555];
ll stamp[55555];
void pre()
{
    for(ll i=0;i<55555;++i)
    {
        vec[i].clear();
        ans[i].ff = i;
        ans[i].ss = 0;
        stamp[i] = 0;
    }
}
void dfs(ll p,ll q,ll r)
{
    ans[p].ff = max(ans[p].ff,r);
    ans[p].ss = q;
    for(ll i=0;i<vec[p].size();++i)
    {
        if(vec[p][i].ss<=q && (ans[vec[p][i].ff].ss<vec[p][i].ss))
        {
            dfs(vec[p][i].ff,vec[p][i].ss,r);
        }
    }
}
int main() 
{
	std::ios::sync_with_stdio(false);
    ll t,value;
    cin>>t;
    while(t--)
    {
        pre();
        cin>>n>>m;
        for(ll i=0;i<n;++i)
        {
            cin>>value;
            stamp[value]++;
        }
        for(ll i=0;i<m;++i)
        {
            cin>>day>>u>>v;
            vec[u].pb(mp(v,day));
            vec[v].pb(mp(u,day));
        }
        for(ll i=55554;i>=1;i--)
        {
            dfs(i,55554,i);
        }
        ll sum = 0;
        for(ll i=0;i<55555;++i)
        {
            sum = sum + (ans[i].ff)*stamp[i];
        }
        cout<<sum<<endl;
    }
	return 0;
}
