#include <cstring>
#include <iostream>
using namespace std;
 
char S[2001];
int best[28][28], best2[28][28];
 
main() {
  int T;
  for (cin >> T; T--;) {
    cin >> S;
    memset(best, 0, sizeof(best));
    for (int i = 0; S[i]; i++) {
      memcpy(best2, best, sizeof(best));
      for (int x = 0; x <= 27; x++)
      for (int y = 0; y <= 27; y++) {
        int c = S[i]-'a'+1;
        if (c >= x) best2[c][y] = max(best2[c][y], best[x][y]+1);
        if (c <= y) best2[x][c] = max(best2[x][c], best[x][y]+1);
      }
      memcpy(best, best2, sizeof(best));
    }
    int ret = 0;
    for (int x = 0; x <= 27; x++)
    for (int y = 0; y <= 27; y++)
      ret = max(ret, best[x][y]);
    cout << ret << endl;
  }
}
