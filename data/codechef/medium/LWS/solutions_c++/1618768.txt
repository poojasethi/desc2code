#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
int dp[2001][27][27];
int main(){
	int t,i,j,k,n;
	string s;
	cin>>t;
	while(t--)
	{
		cin>>s;
		n=s.length();
			for(i=0;i<=n;i++)
				for(j=0;j<27;j++)
					for(k=0;k<27;k++)
						dp[i][j][k]=  -1;
			dp[0][s[0]-'a'+1][0]=1;
			dp[0][0][s[0]-'a'+1]=1;
			dp[0][0][0]=0;
		for(i=1;i<n;i++)
		{	
			int val=s[i]-'a'+1;
			for(j=0;j<27;j++)
				for(k=0;k<27;k++)
				{	dp[i][j][k]=max(dp[i][j][k],dp[i-1][j][k]);
					if( (j==0 || j<=val) && dp[i-1][j][k]!=-1)		dp[i][val][k]=max( dp[i][val][k] , dp[i-1][j][k]+1 );
					if( (k==0 || k>=val) && dp[i-1][j][k]!=-1)	   	dp[i][j][val]=max( dp[i][j][val] , dp[i-1][j][k]+1 );
					//if(j==0 && k==0)  cout<< "val "<<val<<"   "<<dp[i][j][val]<<"\n";
				}
		}

	/*	for(i=0;i<n;i++)
			for(j=0;j<27;j++)
				for(k=0;k<27;k++)
					//cout<<"dp["<<i<<"]["<<j<<"]["<<k<<"] "<<dp[i][j][k]<<"\n"
					;
*/
		int res=-1;
		for (i = 0; i < 27; ++i)
			for (j = 0; j < 27; ++j)
				res=max(res,dp[n-1][i][j]);
		cout<<res<<"\n";
	}
	return 0;
}
