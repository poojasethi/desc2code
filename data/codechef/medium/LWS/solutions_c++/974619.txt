using namespace std;
int T, N, dp[2005][30][30];
char s[2005];
int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%s", &s);
        N = strlen(s);
        for(int last1 = 0; last1 < 30; last1++)
        for(int last2 = 0; last2 < 30; last2++)
            dp[N][last1][last2] = 0;
        for(int i = N - 1; i >= 0; i--)
        for(int last1 = 0; last1 < 30; last1++)
        for(int last2 = 0; last2 < 30; last2++)
            dp[i][last1][last2] = max(dp[i + 1][last1][last2],
                    max((s[i] - 'a' + 1 <= last1?dp[i + 1][s[i] - 'a' + 1][last2] + 1:0),
                    (s[i] - 'a' + 1 >= last2?dp[i + 1][last1][s[i] - 'a' + 1] + 1:0)));
        printf("%d\n", dp[0][28][0]);
    }
    return 0;
}
