#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;

int main(void)
{
	int T,len,p,max1;
	cin>>T;
	string s1;
	char c;
	for(int t=0;t<T;t++)
	{
		cin>>s1;
		len = s1.length();
		int A[len+1][27][27];
		max1 = 0;
        for(int i=0;i<=len;i++)
        {
        	for(int j=0;j<27;j++)
        	{
        	  for(int k=0;k<27;k++)	
        	  {A[i][j][k] = 0;}
			}
		}
			
		for(int i=1;i<=s1.length();i++)
		{
			
			
			for(int x=0;x<27;x++){
		    for(int y=0;y<27;y++)
		    {A[i][x][y] = A[i-1][x][y];}}
		    
		    p = s1[i-1]-96;
		    for(int k=0;k<27;k++)
			{
		      for(int j=0;j<=p;j++)
		      { A[i][p][k] = max(A[i][p][k],A[i-1][j][k]+1);}	
			}
			
			for(int k=0;k<27;k++)
			{
		      for(int j=p;j<=26;j++)
		      { A[i][k][p] = max(A[i][k][p],A[i-1][k][j]+1);}	
			}
			
		}
		
		
		for(int i=0;i<27;i++)
		for(int j=0;j<27;j++)
		{
			if(A[len][i][j]>max1) max1= A[len][i][j];
		}
		cout<<max1<<"\n";
		
	}
}
