#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
using namespace std;

const int N = 2100;

char s[N];
int t, n;
int dp[N][30][30];

int main()
{
    scanf("%d\n", &t);
    while (t--) {
        scanf("%s", s);
        n = strlen(s);
        memset(dp, 0, sizeof dp);
        for (int i=1; i<=26; i++) for (int j=1; j<=26; j++)
            dp[0][i][j] = -1000;
        for (int i=1; i<=n; i++) {
            for (int s1=0; s1<=26; s1++) {
                for (int s2=0; s2<=26; s2++) {
                    int x = s[i-1]-'a'+1;
                    dp[i][s1][s2] = max(dp[i][s1][s2], dp[i-1][s1][s2]);
                    if (s1<=x) {
                        dp[i][x][s2] = max(dp[i][x][s2], dp[i-1][s1][s2]+1);
                    }
                    if (s2>=x) {
                        dp[i][s1][x] = max(dp[i][s1][x], dp[i-1][s1][s2]+1);
                    }
                }
            }
        }
        int ans = -1;
        for (int i=0; i<=26; i++) for (int j=0; j<=26; j++)
            ans = max(ans, dp[n][i][j]);
        printf("%d\n", ans);
    }
    return 0;
}
