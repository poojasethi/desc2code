namespace LWS {
char str[2222];
//dp[i][d][u] = max LWS using i chars with last char of down d, of up u;
int dp[2222][27][27];

void chmax(int&a, int b) {
  if (b > a) a = b;
}

void solve(int test_num) {
  memset(dp, 0xF0, sizeof (dp));
  scanf("%s", str);
  int len = strlen(str);
  dp[0][25][0] = 0;
  for (int i = 0; i < len; i++) {
    int cur = str[i] - 'a';
    for (int d = 0; d < 26; d++) {
      for (int u = 0; u < 26; u++) {
        chmax(dp[i + 1][d][u], dp[i][d][u]);
        if (cur <= d)
          chmax(dp[i + 1][cur][u], dp[i][d][u] + 1);
        if (cur >= u)
          chmax(dp[i + 1][d][cur], dp[i][d][u] + 1);
      }
    }
  }
  int ans = 0;
  for (int i = 0; i < 26; i++)
    for (int j = 0; j < 26; j++)
      chmax(ans, dp[len][i][j]);
  printf("%d\n", ans);
}

void solve() {
  #ifdef AZN
  freopen("input.txt", "r", stdin);
  freopen("output.txt", "w", stdout);
  #endif
  int tests;
  scanf("%d", &tests);
  for (int i = 1; i <= tests; i++)
    solve(i);
}
}

int main() {
  LWS::solve();
  return 0;
