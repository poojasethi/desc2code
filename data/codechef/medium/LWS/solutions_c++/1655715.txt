#define SET(p)     memset(p, -1, sizeof(p))
#define CLR(p)     memset(p, 0, sizeof(p))

int dp[2005][30][30];
main()
{
         int t,i,j,k;
         char s[2005];
         scanf("%d\n",&t);
         while(t--)
         {  CLR(dp);
            scanf("%s",s+1);
            for(k=1;k<=strlen(s+1);k++)
            for(i=1;i<27;i++)
            for(j=1;j<27;j++)
            {   int x=s[k]-'a'+1;
                dp[k][i][j]=max(dp[k][i][j],dp[k-1][i][j]);
                if(i<=x) dp[k][x][j]=max(dp[k][x][j],dp[k-1][i][j]+1);
                if(j>=x) dp[k][i][x]=max(dp[k][i][x],dp[k-1][i][j]+1);   
            }
            int ans=0;
            for(i=1;i<27;i++)
            for(j=1;j<27;j++)
            ans=max(ans,dp[strlen(s+1)][i][j]);
            printf("%d\n",ans);         
         }
         
            
}
