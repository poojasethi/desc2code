#include<bits/stdc++.h>
using namespace std;
int dp[2005][27][27];
int main()
{
	int t;
	string s;
	scanf("%d",&t);
	while(t--)
	{
		cin>>s;
		s="#"+s;
	memset(dp,0,sizeof dp);
			for(int j=0;j<26;j++)
			for(int k=0;k<26;k++)
			dp[0][j][k]=INT_MIN;
		for(int i=1;i<s.length();i++)
		{
			for(int j=0;j<=26;j++)
			{
				for(int k=0;k<=26;k++)
				{
					dp[i][j][k]=max(dp[i][j][k],dp[i-1][j][k]);
					if(s[i]-'a'+1>=j)
					dp[i][s[i]-'a'+1][k]=max(dp[i][s[i]-'a'+1][k],dp[i-1][j][k]+1);
				   if(s[i]-'a'+1<=k)
					dp[i][j][s[i]-'a'+1]=max(dp[i][j][s[i]-'a'+1],dp[i-1][j][k]+1);
				}
			}
		}
		int maxi=INT_MIN;
		for(int i=0;i<=26;i++)
		{
			for(int j=0;j<=26;j++)
			{
				if(maxi<dp[s.length()-1][i][j])
				maxi=dp[s.length()-1][i][j];
			}
		}
		printf("%d\n",maxi);
	}	
}
