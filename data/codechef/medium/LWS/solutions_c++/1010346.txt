#include <stdio.h>

inline int Max(int a, int b){ return (a>b?a:b); }

int t;
char cad[2002];
int DP[2002][26][26];
int n;

int main(){
    scanf("%d",&t);
    while(t--){
        scanf(" %s",cad);
        for(n=0;cad[n];n++)
            cad[n]-='a';
        for(int i=0;i<30;i++)
            for(int j=0;j<30;j++)
                DP[n][i][j]=0;
        for(int pos=n-1;pos>=0;pos--){
            for(int x1=25;x1>=0;x1--){
                for(int x2=0;x2<26;x2++){
                    DP[pos][x1][x2]=0;
                    if(cad[pos]>=x1)
                        DP[pos][x1][x2]=Max(DP[pos][x1][x2],DP[pos+1][cad[pos]][x2]+1);
                    if(cad[pos]<=x2)
                        DP[pos][x1][x2]=Max(DP[pos][x1][x2],DP[pos+1][x1][cad[pos]]+1);
                    DP[pos][x1][x2]=Max(DP[pos][x1][x2],DP[pos+1][x1][x2]);
                }
            }
        }
        printf("%d\n",DP[0][0][25]);
    }
    return 0;
}
