int dp[2010][27][27];
char str[2010];

int main()
{
	int t=scan(),i,j,k;
	while(t--)
	{
		scanf("%s",str);
		memset(dp,0,sizeof(dp));
		int len=strlen(str);
		int n=len;
		for(i=1;i<=n;i++)
		{
			for(j=0;j<26;j++)for(k=0;k<26;k++)dp[i][j][k]=dp[i-1][j][k];
			int curr=str[i-1]-'a';
			for(j=0;j<26;j++)
			{
				for(k=0;k<=curr;k++)
				{
					dp[i][curr][j]=max(dp[i][curr][j],1+dp[i-1][k][j]);
				}
			}
			for(j=0;j<26;j++)
			{
				for(k=curr;k<26;k++)
				{
					dp[i][j][curr]=max(dp[i][j][curr],1+dp[i-1][j][k]);
				}
			}
		}
		int maxi=-1;
		for(i=0;i<26;i++)
		{
			for(j=0;j<26;j++)
				maxi=max(maxi,dp[n][i][j]);
		}
		cout<<maxi<<endl;
	}
	return 0;	
}
