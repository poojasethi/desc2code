#include <iostream>
#include <cstdio>
#include <climits>
#include <algorithm>
#include <queue>
#include <cstring>
#include <cmath>
using namespace std;
int dp[2009][28][28];
void process()
{
	char s[2009];
	memset(dp,0,sizeof(dp));
	gets(s+1);
	int maxi=0,i,j,k;
	for(i=1;s[i]!='\0';i++)
	{
		int c=s[i]-'a';
		for(j=0;j<26;j++)
		{
			for(k=0;k<26;k++)
			{
				dp[i][j][k]=dp[i-1][j][k];
			}
		}
		for(j=0;j<26;j++)
		{
			for(k=0;k<26;k++)
			{
				if(j<=c)
				{
					dp[i][c][k]=max(dp[i][c][k],dp[i-1][j][k]+1);
					maxi=max(maxi,dp[i][c][k]);
				}
				if(k>=c)
				{
					dp[i][j][c]=max(dp[i][j][c],dp[i-1][j][k]+1);
					maxi=max(maxi,dp[i][j][c]);
				}
			}
		}
	}
	printf("%d\n",maxi);
}
int main()
{
	int t,i,j,k,n;
	//freopen("example.txt","r",stdin);
	scanf("%d",&t);
	getchar();
	while(t--)
	{
		process();
	}
	return 0;
}
