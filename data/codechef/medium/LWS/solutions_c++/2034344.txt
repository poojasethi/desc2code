string s;int n;
int dp[2003][30][30];
int solve(int pos,int inc,int dec)
{
    if(pos==n)return 0;
    int x=s[pos]-'a'+1;
    if (dp[pos][inc][dec] != -1) return dp[pos][inc][dec];
    int ansi=solve(pos+1,inc,dec);
    if(x<=inc)
    ansi=max(ansi,1+solve(pos+1,x,dec));
    if(x>=dec)
    ansi=max(ansi,1+solve(pos+1,inc,x));
    dp[pos][inc][dec]=ansi;
    return ansi;
}
