#include<iostream>

using namespace std;

char a[26];

int main()
{
int t;
cin>>t;
cin>>a;
while(t--)
{
string s;
cin>>s;
int i=0;
while(s[i])
{
if(s[i]=='_')
cout<<" ";
else if(s[i]>=97 && s[i]<=122)
cout<<a[s[i]-97];
else if(s[i]>=65 && s[i]<=90)
cout<<(char)(a[s[i]-65]-32);
else
cout<<s[i];
i++;
}
cout<<"\n";
}
return 0;
}
