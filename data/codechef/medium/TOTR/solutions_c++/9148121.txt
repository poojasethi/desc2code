#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    char str[27];
    cin>>str;
    while(t--){
        char ip[101];
        cin>>ip;
        int i;
        for (i=0;ip[i]!=0;i++){
            if (ip[i]=='_')
                cout<<" ";
            else if (ip[i]=='?'||ip[i]=='.'||ip[i]==','||ip[i]=='!')
                cout<<ip[i];
            else if (ip[i]-97>=0&&ip[i]-97<26)
                cout<<str[ip[i]-97];
            else{
                char res=str[ip[i]-65]-32;
                cout<<res;
            }
        }
        cout<<endl;
    }
    return 0;
}
