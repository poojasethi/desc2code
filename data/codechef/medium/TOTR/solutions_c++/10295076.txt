#include<stdio.h>
#include<iostream>


using namespace std;
int main()
{
	int t, p=0;
	char s[27];
	
	cin>>t>>s;

	while(t>0)
	{
		int a[26];
		for(int i=0;i<26;i++)
		{
			a[i]=s[i];
		}
		char str[101];
		scanf("%s",&str);
		int j=0;
		char choice;
		while(str[j]!='\0')
		{
			if(str[j]>=65&&str[j]<=90)
			choice ='A';
			else if(str[j]>=97&&str[j]<=122)
			choice ='B';
			else
			choice = (char)str[j];
			switch(choice)
			{
				case 'A': int q;
						q=str[j]-65;
						q=a[q]-32;
						cout<<(char)q;
						break;
				case 'B': int r;
						r=str[j]-97;
						r=a[r];
						cout<<(char)r;
						break;
				case '_': cout<<" ";
				break;
					case '.': cout<<".";
				break;
					case ',': cout<<",";
				break;
					case '!': cout<<"!";
				break;
					case '?': cout<<"?";
				break;
				
						
			}
			j++;
			
		}
		cout<<endl;
		
       t--;
	
		
	}
	return 0;
}
