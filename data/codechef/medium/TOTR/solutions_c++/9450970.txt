#include<bits/stdc++.h>
using namespace std;

main()
{
    int t;
    char m[27],s[200];
    cin>>t>>m;
    while(t--)
    {
        cin>>s; int i;
        for(i=0;s[i];i++)
        {
             if(islower(s[i]))
                s[i] = m[s[i] - 'a'];
            else if (isupper(s[i]))
                s[i] = toupper(m[s[i] - 'A']);
            else if (s[i] == '_')
                s[i] = ' ';
        }
        cout<<s<<endl;
    }
}
