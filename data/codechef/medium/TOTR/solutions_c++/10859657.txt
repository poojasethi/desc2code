#include <iostream>
#include <cmath>
#include <string>
 
using namespace std ;
 
int main(){
    int T ;
    cin >> T ;
    string M ;
    cin >> M ;
    string A[T];
    for (int i=0; i<T; i++) {
        cin >> A[i] ;
    }
    char key[26] ;
    for (int i=0; i<26; i++) {
        key[i] = M[i] ;
    }
    for (int i=0; i<T; i++) {
        for (int j=0; j<A[i].length(); j++) {
            if ((A[i][j]>64)&&(A[i][j]<91)) {
                cout << char(key[A[i][j]-'A']-'a'+'A') ;
            }
            else if ((A[i][j]>96)&&(A[i][j]<123)){
                cout << key[A[i][j]-'a'] ;
            }
            else if (A[i][j]==95){
                cout << " " ;
            }
            else {
                cout << A[i][j] ;
            }
        }
        cout << endl ;
    }
}
