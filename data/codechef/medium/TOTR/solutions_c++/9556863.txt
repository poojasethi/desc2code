#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	string m,s;
	cin>>t>>m;
	while(t--)
	{
		cin>>s;
		for(int i=0;i<s.length();i++)
		{
			if(islower(s[i]))
			s[i]=m[s[i]-'a'];
			else if(isupper(s[i]))
			s[i]=toupper(m[s[i]-'A']);
			else if(s[i]=='_')
			s[i]=' ';
		}
		cout<<s<<endl;
	}
	return 0;
}
