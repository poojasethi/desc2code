#include<bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("in","r",stdin);
	int t;
	string str,s;
	cin>>t>>str;
	while(t--)
	{
     cin>>s;
     for(int i=0;i<s.size();i++)
     {
     	if(s[i]>='A' && s[i]<='Z')
     	{
     		int x=s[i]-'A';
     		s[i]=str[x]-'a'+'A';
     	}
     	else if(s[i]<='z' && s[i]>='a')
     	{
     		s[i]=str[s[i]-'a'];
     	}
     	else if(s[i]=='_')
     	{
     		s[i]=' ';
     	}

     }
     cout<<s<<endl;
	}
}
