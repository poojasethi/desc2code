#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T,j;
    string s,bs;char fs[200];
    cin>>T>>s;
    map<char,char> mcc;
    for(int i=0;i<s.length();i++)
        mcc[(char)(97+i)] = s[i];
    mcc['_'] = ' ';
    while(T-->0)
    {
        cin>>bs;j=0;
        for(int i=0;i<bs.length();i++)
        {
            if(bs[i]>=65 && bs[i]<=90)
            {
                fs[j++] = (char)(mcc[bs[i]+32]-32);
            }
            else if(mcc.find(bs[i])!=mcc.end())
            {
                fs[j++] = mcc[bs[i]];
            }
            else fs[j++] = bs[i];
        }
        fs[j]='\0';
        cout<<fs<<endl;
    }
}
