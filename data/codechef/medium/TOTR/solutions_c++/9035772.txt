int main()
{
	ios_base::sync_with_stdio(false);
  string input;
  int t;
  string trans2;
  cin>>t;
  cin>>trans2;
  // getline(cin, input);
  // int t = input[0]-'0';
  // string trans2 = input.substr(2,26);
  string trans1 = "abcdefghijklmnopqrstuvwxyz";
  string trans0="ABCDEFGHIJKLMNOPQRSTUVWXYZ";
  string tmp_input;
  while(t--){
    cin>>tmp_input;
    for(int i=0;i<tmp_input.size();i++){
      if(tmp_input[i]=='_'){//If character is underscore, replace it by space
        cout<<" ";
      }
      else if(trans1.find(tmp_input[i])!=-1 ){//if character is present in lowercase letters
        cout<<trans2[trans1.find(tmp_input[i])];
      }
      else if(trans0.find(tmp_input[i])!=-1 ){//if character is present in upper case letters
        cout<<(char)toupper(trans2[trans0.find(tmp_input[i])]);
      }

      else{
        cout<<tmp_input[i];
      }
    }
    cout<<"\n";
  }
	return 0;
}
