#include<cstdio>
using namespace std;
int main(){
    char map_str[28];
    int T;
    scanf("%d %s",&T,&map_str);
    while(T--){
        char str[105];
        scanf("%s",&str);
        int i=0;
        while(str[i]!='\0'){
            char c=str[i];
            if(c>='a' && c<='z'){
                printf("%c",map_str[c-'a']);
            }else if(c>='A' && c<='Z'){
                printf("%c",map_str[c-'A']-'a'+'A');
            }else if(c=='_'){
                printf(" ");
            }
            else{
                printf("%c",c);
            }
            i++;
        }
        printf("\n");
    }
}
