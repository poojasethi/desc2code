#include <iostream>
using namespace std;

int main()
{
    int i,len,t,new_alpha[26],num;
    string alpha,s;
    cin>>t;
    cin>>alpha;
    len=alpha.size();    
    for(i=0;i<len;i++)
    {
        new_alpha[i]=int(alpha[i])-97;
    }
       
    while(t--)
    {
        cin>>s;
        len=s.size();
        for(i=0;i<len;i++)
        {
            num=int(s[i]);
            if(s[i]=='_')
                s[i]=' ';
            else if((num>=97)&(num<=122))
                s[i]=char(new_alpha[num-97]+97);
            else if((num>=65)&(num<=90))
                s[i]=char(new_alpha[num-65]+65);    
        }
        cout<<s<<endl;
    }
    return 0;
}
