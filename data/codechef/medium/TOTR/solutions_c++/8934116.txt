#include <bits/stdc++.h>

#define s(x) scanf("%d", &x)

using namespace std;

int main()
{

	int t,k,l,i,n,j,max;
	
	cin >> t;
	
	string s;
	cin >> s;
	
	while(t--)
	{
		string a;
			cin >> a;
		
		char c;
		n = a.length();
		
		for(i=0;i<n;i++)
		{
			if(a[i] <= 90 && a[i] >= 65)
				c = s[a[i] - 'A'] - 32;
			
			else if(a[i] >= 97)
				c = s[a[i] - 'a'];
			
			else if(a[i] == '_')
				c = ' ';
			
			else 
				c = a[i];
				
			cout << c;		
		}
		cout << endl;
		
	}
	return 0;	
}
