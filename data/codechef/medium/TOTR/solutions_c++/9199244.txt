#include<stdio.h>
#include<iostream>
#include<string>
using namespace std;
int main()
{
	string translation,sentence;
	int t;
	cin >> t >> translation;
	while (t--)
	{
		cin >> sentence;
		for (int i = 0; i < sentence.length(); ++i)
		{
			if (sentence[i] >= 'a'&&sentence[i] <= 'z')
				sentence[i] = translation[sentence[i]-'a'];
			else if (sentence[i] >= 'A'&&sentence[i] <= 'Z')
				sentence[i] = translation[sentence[i] - 'A'] + -'a'+'A';
			else if (sentence[i] == '_')
				sentence[i] = ' ';
		}
		cout << sentence << endl;
	}
	return 0;
}
