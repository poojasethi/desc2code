#include<iostream>
#include<cstring>
#include<map>
using namespace std;
int main()
{	//freopen("out.txt","r",stdin);
	int t;
	cin>>t;
	map<char,char>sm,cap;
	string b;
	cin>>b;
	char c='a',d='A';
	for(int i=0;i<b.size();i++)
	{sm[c++]=b[i];
	cap[d++]=b[i]-('a'-'A');}
	while(t--)
	{
		
		string a;
		cin>>a;
		for(int i=0;i<a.size();i++)
		{
			if(a[i]=='_')
			cout<<" ";
			else if(a[i]>='a'&&a[i]<='z')
			cout<<sm[a[i]];
			else if(a[i]>='A'&&a[i]<='Z')
			cout<<cap[a[i]];
			else cout<<a[i];
			
		}
		cout<<"\n";
		
	}	
}
