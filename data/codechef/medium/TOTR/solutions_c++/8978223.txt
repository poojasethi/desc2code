using namespace std;
typedef long long ll;
int main()
{
    int t; get(t);
    string s; cin >> s;
    int a[26];
    for(int i=0;i<26;i++)
        a[i] = s[i]-'a';
    while(t--)
    {
        cin >> s;
        for(int i=0;i<s.length();i++)
        {
            if(isalpha(s[i]))
            {
                if(s[i]>='a')
                    cout << char(a[s[i]-'a']+'a');
                else
                    cout << char(a[s[i]-'A']+'A');
            }
            else if(s[i]=='_')
                cout << " ";
            else
                cout << s[i];
        }
        cout << endl;
    }
    return 0;
}
