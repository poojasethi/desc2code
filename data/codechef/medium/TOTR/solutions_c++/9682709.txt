#include<stdio.h>
#include<string.h>
int main()
{
int t,i;
char a[27],test[101],tt;
scanf("%d",&t);
scanf("%s",&a);
//printf("%d %c",t,a[8]);
while(t>0)
{
i=0;
scanf("%s",&test);
tt=test[i];
while(tt!='\0')
{
if(tt=='_') printf(" ");
else if(tt>=97&&tt<=122) printf("%c",a[tt%97]);
else if(tt>=65&&tt<=90) printf("%c",a[(tt%65)]-32);
else printf("%c",tt);
tt=test[++i];
}
printf("\n");
t--;
}
return 0;
}
