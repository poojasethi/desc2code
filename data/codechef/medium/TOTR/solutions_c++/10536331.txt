#include<bits/stdc++.h>
using namespace std;
int main()
{	int t; string s;
	cin>>t; cin>>s;
	while(t--)
	{	string a;
	        cin>>a;
	        for(int i=0;i<a.size();i++)
	        {	if(a[i]>='a' && a[i]<='z')
	                      cout<<s[a[i]-'a'];
	                 else if(a[i]>='A' && a[i]<='Z')
	                     cout<<char(s[a[i]-'A'] + 'A'-'a');
	                  else if(a[i]=='_')
	                  cout<<" ";
	                  else
	                  cout<<a[i];       
	         }
	         cout<<"\n";
	   }
return 0;
}
