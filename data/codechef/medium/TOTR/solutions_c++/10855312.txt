#include <bits/stdc++.h>

using namespace std;

int main()
{
	int T;
	char cipher[27], str[101];
	scanf("%d %s", &T, cipher);
	
	while (T--)
	{
		scanf("%s", str);
		int len = strlen(str);
		
		for (int i = 0; i < len; i++)
		{
			if (str[i] == '_')
				printf(" ");
			else if (isupper(str[i]))
				printf("%c", cipher[str[i] - 65] - 32);
			else if (islower(str[i]))
				printf("%c", cipher[str[i] - 97]);
			else
				printf("%c", str[i]);
		}

		printf("\n");
	}
	return 0;
}
