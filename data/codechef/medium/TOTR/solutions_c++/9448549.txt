#include <iostream>
#include <string>
#include <vector>

using namespace std;

int main()
{
    int t;
    string s;
    cin>>t>>s;
    vector<char> a(26);
    for(int i=0;i<26;i++)
        a[i] = s[i];
    while(t--)
    {
    string u;
    cin>>u;
    int ul = u.length();
    for(int i=0;i<ul;i++)
    {
        if(u[i]=='_')
            u[i]=' ';
        else if(isalpha(u[i]))
        {
            if(isupper(u[i]))
            {
                char d = tolower(u[i]);
                int p = d-97;
                u[i] = s[p];
                u[i] = toupper(u[i]);
            }
            else
            {
                char d = u[i];
                int p = d-97;
                u[i] = s[p];
            }
        }
        else
            continue;
    }
    cout<<u<<endl;
    }
}
