#include<iostream>
#include<map>
using namespace std;
map<char , char> m;
int main()
{
    int n;
    string s;
    cin>>n>>s;
    while(n--)
    {


    string s1;
    cin>>s1;
    int i,j;
    for(i=0;i<s.length();i++)
    {

        m[i+97]=tolower(s[i]);
        m[i+65]=toupper(s[i]);
    }
    m['.']='.';
    m['!']='!';
    m['?']='?';
    m[',']=',';
    m['_']=' ';
    for(i=0;i<s1.length();i++)
    {



        cout<<m[s1[i]];
    }
    cout<<endl;
    }
    m.clear();
    return 0;

}
