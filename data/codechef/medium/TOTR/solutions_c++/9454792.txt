#include<iostream>
#include<string>
using namespace std;
int main()
{
	int t;
	string eng;
	cin>>t;
	string test;
	cin>>test;
	while(t-->0)
	{
		int i=0;
		string alpha;
		cin>>alpha;
		for(i=0;i<alpha.length();i++)
		{
			if(alpha[i]>=97&&alpha[i]<=122)
			eng[i]=test[alpha[i]-'a'];
			else if(alpha[i]>=65&&alpha[i]<=90)
			eng[i]=toupper(test[alpha[i]-'A']);
			else if(alpha[i]==95)
			eng[i]=32;
			else 
			eng[i]=alpha[i];
			cout<<eng[i];
		}
		cout<<endl;
	}
}
