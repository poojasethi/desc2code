using namespace std;
int T,i;
string s,tr;
int main()
{
	cin>>T>>tr;
	//cout<<"hello"<<tr<<"\n";
	while(T-->0)
	{
		cin>>s;
		for(i=0;s[i];i++)
			if(s[i]=='_')
				printf(" ");
			else if('a'<=s[i] && s[i]<='z')
				printf("%c",tr[s[i]-'a']);
			else if('A'<=s[i] && s[i]<='Z')
				printf("%c",(tr[s[i]-'A']+'A'-'a'));
			else
				printf("%c",s[i]);
		printf("\n");
	}
	return 0;
}
