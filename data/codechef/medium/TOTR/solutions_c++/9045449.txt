#include <iostream>
using namespace std;

int main() {
	ios_base::sync_with_stdio(false);
	char c;
	string sent,letter;
	int t,len;
	cin>>t>>letter;
	//65-90; 97-122
	while(t-->0)
	{
		cin>>sent;
		len=sent.length();
		for(int i=0;i<len;i++)
		{
			if(sent[i]=='_')
			cout<<" ";
			else if(sent[i]>=65 && sent[i]<=90)
			cout<<(char)(letter[(sent[i])%65]-32);
			else if(sent[i]>=97 && sent[i]<=122)
			cout<<letter[sent[i]%97];
			else cout<<sent[i];
			
		}
		cout<<"\n";
	}
	
	return 0;
}
