x,y=map(str,raw_input().split())
x=int(x)
y=list(y)
z='abcdefghijklmnopqrstuvwxyz'
q=z.upper()
z=list(z)
q=list(q)
for i in xrange(x):
    l=raw_input()
    ans=[]
    for j in range(len(l)):
        if l[j]=='_':
            ans.append(' ')
        elif (ord(l[j])>=65 and ord(l[j])<=90):
            ans.append(y[q.index(l[j])].upper())
        elif (ord(l[j])>=97 and ord(l[j])<=122):
            ans.append(y[z.index(l[j])])
        else:
            ans.append(l[j])
    print ''.join(ans)
