t, m = raw_input().split()
t= int(t)
import string
dict1 = {}
for i in range(len(m)):
    dict1[string.lowercase[i]] = m[i]
m = m.upper()
for i in range(len(m)):
    dict1[string.uppercase[i]] = m[i]
for i in range(t):
    text = raw_input()
    converted_text = []
    for i in text:
        if i.isalpha():
            converted_text.append(dict1[i])
        elif i == "_":
            converted_text.append(" ")
        else:
            converted_text.append(i)
    converted_sentence = "".join(converted_text)
    print converted_sentence
