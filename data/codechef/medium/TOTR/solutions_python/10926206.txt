a,b=map(str,raw_input().split())
dic={}
for i in xrange(26):
    dic[chr(i+97)]=b[i]
    dic[chr(i+65)]=chr(ord(b[i])-97+65)
for _ in xrange(int(a)):
    s=raw_input()
    ans=''
    for i in s:
        try:
            if i=='_':
                ans+=' '
            else:
                ans+=dic[i]
        except:
            ans+=i
    print ans
