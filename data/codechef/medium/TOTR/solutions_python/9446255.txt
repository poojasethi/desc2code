from sys import stdin, stdout
t, ms = stdin.readline().strip().split(' ')
mb = ms.upper()
t = int(t)
while t:
	t -= 1
	s = stdin.readline().strip()
	e = ""
	for i in s:
		if 'a' <= i <= 'z':
			e += ms[ord(i) - 97]
		elif 'A' <= i <= 'Z':
			e += mb[ord(i) - 65]
		elif i == '_':
			e += " "
		else:
			e += i
	stdout.write(e+'\n')
