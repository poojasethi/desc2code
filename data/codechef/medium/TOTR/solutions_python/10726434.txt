line1 = raw_input().split()
cases = int(line1[0])
trans = line1[1]
translist = list(trans)
translist += list(trans.upper())
translist += [' ', '.', ',', '?', '!']
englist = list("abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ")
englist += ['_', '.', ',', '?', '!']
for _dummy in range(cases):
	s = raw_input()
	news = list()
	for c in s:
		news.append(translist[englist.index(c)])
	print ''.join(news)
