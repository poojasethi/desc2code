t=raw_input().split()
conv=(int)(t[0])
byte=t[1]
alpha="abcdefghijklmnopqrstuvwxyz"
list_eng=list(alpha)
list_byte=list(byte)

code=dict(zip(list_eng,list_byte))
code['_']=' '
upper_eng=[x.upper() for x in list(alpha)]
upper_byte=[x.upper() for x in list(byte)]

upper_code=dict(zip(upper_eng,upper_byte))
code.update(upper_code)
code['.']='.'
code['!']='!'
code['?']='?'
code[',']=','

for i in range(conv):
	string=raw_input()
	k=''
	for j in range(len(string)):
		k=k+code[string[j]]
	print k
