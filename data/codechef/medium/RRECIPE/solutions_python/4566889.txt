T = input()

for it in range(0,T):
    inp = raw_input()
    ans = 1
    length = len(inp)
    
    if(length%2 == 0):
        end = length/2
    else:
        end = length/2+1

    for i in range(0,end):
        if(inp[i] == inp[length-i-1]):
            if(inp[i] == '?'):
                ans = (ans*26)%10000009
        elif(inp[i] == '?' or inp[length-i-1] == '?'):
            ans *= 1
        else:
            ans = 0
            break

    print ans
