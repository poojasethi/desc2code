from sys import stdin

def numWays(S):
    lenS = len(S)
    count = 0
    for i in xrange((lenS+1)/2):
        left,right = S[i], S[lenS-i-1]
        if left!="?" and right!="?" and left!=right: return 0
        elif left=="?" and right=="?": count+=1
    return pow(26, count, 10000009)

for trial in xrange(int(stdin.readline())):
    S = stdin.readline().strip()
    print numWays(S)
