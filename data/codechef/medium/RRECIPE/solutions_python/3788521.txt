
def calc_recipe(string):
	N = len(string)
	prod = 1
	if N%2==1 and string[N//2]=='?':
		prod *= 26
	i=0
	count26 = 0
	while i<N//2:
		rev = N-i-1
		if string[i]=='?' and string[rev]=='?':
			count26+=1
		elif string[i]=='?' or string[rev]=='?': #one but not both
			pass
		elif string[i]!=string[rev]:
			prod=0
			break
		i+=1
	prod = prod * 26**count26
	return prod % 10000009



T = int(raw_input())

while T>0:
	T-=1
	string = raw_input()
	prod = calc_recipe(string)
	print(prod)
