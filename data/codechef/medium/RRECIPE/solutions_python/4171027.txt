t=input()
while t:
    st = raw_input().rstrip()
    ans=1
    l=len(st)
    for i in xrange(0,l/2):
        if(st[i]!=st[l-i-1]):
            if(st[i]!='?' and st[l-i-1]!='?'):
                ans=0
                break
        else:
            if(st[i]=='?'):
                ans = (ans*26)%10000009
    if(l%2==1):
        if(st[l/2]=='?'):
            ans = (ans*26)%10000009
    print ans
    t -=1
