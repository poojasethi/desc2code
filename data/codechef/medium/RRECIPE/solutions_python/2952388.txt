t = int(raw_input())
MOD=10000009

for T in range(t):
	s = str(raw_input())
	prod = 1	
	n = len(s)

	for i in range(n/2):
		if s[i] == '?' and s[n-i-1]=='?':
			prod = (prod*26)%MOD
		elif s[i]!=s[n-i-1] and s[i]!='?' and s[n-i-1]!='?':
			prod*=0
			break
	
	if n%2==1 and s[n/2]=='?':
		prod = (prod*26)%MOD

	print(prod)
