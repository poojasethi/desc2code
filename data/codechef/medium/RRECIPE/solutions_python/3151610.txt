def processList(ingred):
    
    result = 0 ; mid = (len(ingred)+1)/2
        
    for j in xrange(mid):
        L , R = ingred[j] , ingred[0-(j+1)]
        if( L == '?' and R == '?'):
            result+=1
        elif(L != '?' and R != '?' and L != R):
            return 0

    return pow(26,result,10000009)

main()
