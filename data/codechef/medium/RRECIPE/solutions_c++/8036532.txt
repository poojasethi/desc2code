#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	    long long int i,j,count,n,ans;
		char str[1000001];
		cin.ignore();
		cin>>str;	
		n=strlen(str);	
		for(i=0,j=n-1,count=1;j>i;i++,j--)
		{  
		   if(str[i]=='?'&&str[j]=='?')
			count=(count*26)%10000009;
			else if(str[i]=='?'||str[j]=='?')
			continue;
			else if(str[i]!=str[j])
			{
				count=0;
				count=count%10000009;
				break;
			}
		}	
		if(i==j)
		{
			if(str[i]=='?')
			count=(count*26)%10000009;
		}
		ans=count%10000009;
		cout<<ans<<endl;
	}
	return 0;
}
