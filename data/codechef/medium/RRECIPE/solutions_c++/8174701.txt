#include<iostream>
#include<cstring>

using namespace std;
int main()
{
    long long MOD =10000009;
    int t;
    cin>>t;
    while(t--)
    {
        string str;
        cin>>str;int res=1;
        long long l=str.length(),i;
        for(i=0;i<(l+1)/2;i++)
        {
            if(str[i]==str[l-1-i])
            {
                if(str[i]=='?')
                res=(res*26)%MOD;

            }
            else
            {
                if(str[i]!='?'&&str[l-1-i]!='?')
                {
                    res=0;
                break;
                }

            }
        }
        cout<<res<<endl;


    }

}
