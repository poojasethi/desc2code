#include<stdio.h>
#include<string.h>
#define m 10000009
using namespace std;
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        char s[1000005];
        scanf("%s", s);
        int l, i, c = 1;
        l = strlen(s);
        int lim;
        if(l %2 == 0)
            lim = l/2;
        else
            lim = l/2+1;
        for(i = 0;i <lim;i++)
        {
            if(s[i] == '?' || s[l-1 - i] == '?')
            {
                if(s[i] == '?' && s[l-i - 1] == '?')
                    c = (c*26)%m;
            }
            else if(s[i] != s[l-1 - i])
            {
                c = 0;
                break;
            }
        }
        printf("%d\n", c);
    }
    return 0;
}
