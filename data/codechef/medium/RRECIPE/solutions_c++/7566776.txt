#include<stdio.h>
#include<string.h>

int main()
{
    int tc;
    long int i,len,limit;
    long long int ans;
    char str[1000000];
    const int mod = 10000009;

    scanf("%d\n",&tc);

    while(tc--)
    {
        scanf("%s",str);

        len = strlen(str);
        ans = 1;
        limit = (len/2)+(len%2);
        for(i=0;i<limit;i++)
        {
            if(str[i]=='?' && str[len-1-i]=='?')
                ans = (ans*26)%mod;
            else if(str[i]!='?' && str[len-1-i]!='?' && str[i]!=str[len-1-i])
            {
                printf("0\n");
                goto label;
            }
        }
        printf("%lld\n",ans);
        label:;
    }
    return 0;
}
