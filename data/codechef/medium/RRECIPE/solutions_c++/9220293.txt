#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

#define m 10000009

int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		char s[1000001];
		scanf("%s", s);
		int l = strlen(s), n=1;
		for(int i=0; i<(l+1)/2; i++)
		{
			if( s[i]=='?' && s[l-i-1]=='?' )
				n = (n*26)%m;
			else if( s[i]!=s[l-i-1] && (s[i]!='?' && s[l-i-1]!='?') )
			{
				n=0;
				break;
			}
		}
		printf("%d \n", n%m);
	}
	return 0;
}
