#include<cstdio>
#include<cstring>
#define MOD 10000009
using namespace std;
typedef long long int ll;
ll count_way(char str[]){
    int n=strlen(str);
    int i=0,j=n-1;
    ll ans=1;
    while(i<=j){
        
        if(str[i]=='?' && str[j]=='?'){
            ans=(ans*26)%MOD;
        }
        if(str[i]!='?'&&str[j]!='?'&&str[i]!=str[j]){
            ans=0;
            break;
        }
        i++;
        j--;
    }
    return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--){
        char str[1000002];
        scanf("%s",&str);
        printf("%lld\n",count_way(str));
    }
}
