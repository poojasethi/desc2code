#include<iostream>
using namespace std;
int main()
{
    int t,i,p,j,count;
    string s;
    cin>>t;
    while(t--)
    {
        cin>>s;
        count=1;
        p=0;
        for(i=0,j=s.length()-1;i<=j;i++,j--)
        {
            if(s[i]!=s[j] && (s[i]!='?' && s[j]!='?'))
            {
                count=0;
                i=j+1;
                p=1;
            }

            else if(s[i]==s[j] && s[i]=='?')
            {
                count=count*26;
                count=count%10000009;
            }
        }
        if(p!=1 && count==0)
        {
            count=1;
        }

        cout<<count<<endl;
    }
}
