#include<cstdio>
#include<cstring>
# define MOD 10000009
using namespace std;
int main(){
	unsigned int n,i,m (1),l;
	char f[1000005];
	scanf("%d\n",&n);
	start :
	while(n--){
		m = 1;
		gets(f);
		l = strlen(f);
		for(i=0;i<((l+1)/2);i++){
			if (f[i]!=f[l-1-i]){
				if (f[i]!='?'&&f[l-1-i]!='?'){
					printf("%d\n", 0);
					goto start;
				}
			}
			else if(f[i]==f[l-1-i] && f[i]=='?')
				m = (m*26)%MOD;
		}
		printf("%d\n",m);
	}
	return 0;
}
