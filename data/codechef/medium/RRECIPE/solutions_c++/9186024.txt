#include<stdio.h>
#include<string.h>
int main()
{
	int t;
	int a,c,d;
		int long long b;
	char recipe[1000000];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s",recipe);
		a=strlen(recipe);
		b=1;
		if((a%2)!=0)
		{
			if(recipe[a/2]=='?')
			{
				b=(b*26)%10000009;
			}
		}
		c=a-1;
		for(d=0;d<c;d++)
		{
			if(recipe[d]==recipe[c])
			{
				if(recipe[d]=='?')
				{
					b=(b*26)%10000009;
				}
			}
			else
			{
				if(recipe[d]=='?' || recipe[c]=='?')
				{
					b=b*1;
				}
				else
				{
					b=0;
					break;
				}
			}
			c--;
		}
		printf("%lld\n",b);
	}
	return 0;
}
