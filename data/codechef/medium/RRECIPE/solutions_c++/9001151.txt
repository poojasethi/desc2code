#include<iostream>
#include<string>

using namespace std;

int main()
{
int t;
cin>>t;
while(t--)
{
string s;
cin>>s;
long long int res=1;
int i,l;
l=s.length();
for(i=0;i<l/2;i++)
{if(s[i]==s[l-i-1] && s[i]=='?' && s[l-i-1]=='?')
{res*=26;
res=res%10000009;}
else if(s[i]!=s[l-i-1] && s[i]!='?' && s[l-i-1]!='?')
break;
}

if(i<l/2)
cout<<"0\n";
else
{
if(l%2==0)
cout<<res%10000009<<"\n";
else
if(s[i]=='?')
cout<<(res*26)%10000009<<"\n";
else
cout<<res%10000009<<"\n";
}



}
return 0;
}
