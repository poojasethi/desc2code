int main(){
	int t;
	si(t);
	string s;
	while(t--){
		cin>>s;
		int l=s.length();
		ll ans=1;
		int i,j;
		for(i=0,j=l-1;i<j;i++,j--){
			if(s[i]=='?' && s[j]=='?')ans*=26,ans%=MOD;
			else if(s[i]=='?' || s[j]=='?')ans%=MOD;
			else if(s[i]!=s[j]){
				ans=0;
				break;
			}
		}
		if(i==j && s[i]=='?')ans*=26,ans%=MOD;
		pll(ans);
	}
	return 0;
