#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	 long long int t,count,l,i,j,k;
	char s[1000001];
	cin>>t;
	while(t--)
	{
		cin>>s;
		k=1;
		l=strlen(s);
		j=l-1;
		for(i=0;i<=(l-1)/2;i++)
		{
				if(j>=l/2)
				{
					if(s[i]!='?'&&s[j]!='?'&&s[i]!=s[j])
					{
					k=0;
					break;
					}
					else if(s[i]!='?'&&s[j]=='?')
					{
					s[j]=s[i];
					k*=1;
					}
					else if(s[i]=='?'&&s[j]!='?')
					{
						s[i]=s[j];
						k*=1;
					}
					else if(s[i]=='?'&&s[j]=='?'&&(i+j)==(l-1))
						k=(k*26)%10000009;
					j--;
				}
		}
	
		cout<<k<<endl;	
	}
	return 0;
}
