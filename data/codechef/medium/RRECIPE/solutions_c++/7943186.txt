#include<bits/stdc++.h>
#define mod 10000009
#define gc getchar
typedef long long int ll;
using namespace std;
ll read_int() {
char c = gc();
while(c<'0' || c>'9') c = gc();
ll ret = 0;
while(c>='0' && c<='9') {
ret = 10 * ret + c - 48;
c = gc();
}
return ret;
}
int main()
{
    ll t,i,j,l,k;
    cin>>t;
    string s;
    while(t--)
    {
        cin>>s;
        l=s.length();
        ll noways=1,f=1;
        if(l==1)
        {
            if(s[0]!='?')
                cout<<1<<'\n';
            else
                cout<<26<<'\n';
        }
        else
        {
            if(l%2!=0)
                k=l/2+1;
            else
                k=l/2;
            for(i=0,j=l-1;i<k;i++,j--)
            {
                if(s[i]!='?' && s[j]!='?')
                {
                    if(s[i]!=s[j])
                    {
                        noways=0;
                        f=0;
                        break;
                    }
                    else
                    {
                        noways*=1;
                        noways%=mod;
                    }
                }
                else if(s[i]=='?' && s[j]=='?')
                {
                    noways*=26;
                    noways%=mod;
                }
                else if(s[i]!='?' && s[j]=='?')
                {
                    noways*=1;
                    noways%=mod;
                }
                else if(s[i]=='?' && s[j]!='?')
                {
                    noways*=1;
                    noways%=mod;
                }
            }
            cout<<noways<<'\n';
        }
    }
    return 0;
}
