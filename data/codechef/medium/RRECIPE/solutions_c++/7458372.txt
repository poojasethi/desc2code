#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

long long md = 10000009;
string s;

int main()
{
	int tc;
	cin >> tc;
	
	while(tc--)
	{
		cin >> s;
		int sz = (int)s.size();
		int p = 0, q = sz - 1;
		long long ans = 1;
		while(p <= q)
		{
			char x = s[p];char y = s[q];
			
			if(x == y && x != '?')
			{
				ans = (ans * 1) % md;
			}
			else if(x == y && x == '?')
			{
				ans = (ans * 26) % md;
			}
			else if(x != y && (x == '?' || y == '?'))
			{
				ans = (ans * 1) % md;
			}
			else
			{
				ans = 0;
			}
			p++;
			q--;
		}
		cout << ans << endl;
	}
	return 0;
}
