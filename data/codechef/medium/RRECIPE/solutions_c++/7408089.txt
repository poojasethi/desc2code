#include<bits/stdc++.h>
#define MOD 10000009
using namespace std;
int main()
{
	int t;
    long int i,j;
	scanf("%d",&t);
	while(t--)
	{
		string s;
		cin>>s;
		long int res=1;
		long int l=s.size();
		for(i=0;i<(l+1)/2;i++)
		{
			if(s[i]==s[l-i-1] && s[i]=='?' && s[l-i-1]=='?')
			{
				res=(res*26)%MOD;
			}
			else if(s[i]!=s[l-i-1] && s[i]!='?' && s[l-i-1]!='?')
			{
				res=0;
				break;
			}
		}
		printf("%ld\n",res);
	}
	return 0;
}
