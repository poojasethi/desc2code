#include<bits/stdc++.h>
#define mod 10000009
using namespace std;
int main()
{
	int t;
	for(scanf("%d", &t); t--; )
	{
		char str[1000001];
		scanf("%s", str);
		int i=0;
		int j=strlen(str)-1;
		long long res=1;
		while(i<=j){
			if(str[i]!=str[j] && str[i]!='?' && str[j]!='?'){
				res=0;
				break;
			}
			if(str[i]==str[j] && str[i]=='?')
				res = (res*26)%mod;
			i++;
			j--;
		}
		printf("%lld\n", res);
	}
	return 0;	
}
