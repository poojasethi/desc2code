#include <iostream>
#include<cmath>
using namespace std;

int main() {
    int f,i,t,k;
    string s;
    cin>>t;
    while(t--)
    {
    	int f=0;
    	long long int ans=1;
    	cin>>s;
    	if(s.size()>1)
    	{
    		if(s.size()%2==0)
    	 k=s.size()/2;
    	  else
    	 k=s.size()/2+1;
    	}
    	else
    	 k=1;
    	for(i=0;i<k;i++)
    	{
    		if(s[i]=='?')
    		{ 
    			if(s[s.size()-1-i]=='?'||i==k){
    			 ans=ans*26;
    			 ans=ans%10000009;
    			}
    		}
    	   else if(96<s[i]<123)
    	   {
    	   	if(s[i]!=s[s.size()-1-i]&&s[s.size()-1-i]!='?')
    	   	{
    	   	f=1;
    	   	break;
    	   	}
    	   }
       }
       if(f==1)
       cout<<"0"<<endl;
       else
       {
       cout<<ans<<endl; }
    }
	return 0;
}
