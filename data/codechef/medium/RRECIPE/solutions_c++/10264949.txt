#include<stdio.h>
#include<string.h>
char arr[1000005];
int main()
{
    int t,n,i,k,j;

    scanf("%d",&t);

    while(t--)
    {
        scanf("%s",&arr);

        n=strlen(arr);
        int flag=0;

        for(int i=0;i<(n+1)/2;i++)
        {
            if((arr[i]!='?')&&(arr[n-i-1]!='?'))
            {
                if(arr[i]!=arr[n-i-1])
                {
                    flag=1;
                    break;
                }
            }
        }

        if(flag==1)
        {
            printf("0\n");
            continue;
        }

        int ctr=1;
        for(int i=0;i<(n+1)/2;i++)
        {
            if(arr[i]=='?')
            {
                if(arr[n-i-1]=='?')
                {
                    ctr=(ctr*26)%10000009;
                }
            }
        }

        printf("%d\n",ctr);
    }
    return 0;
}
