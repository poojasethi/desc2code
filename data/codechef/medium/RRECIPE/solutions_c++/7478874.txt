#include<bits/stdc++.h>
#include<algorithm>
#include<string.h>
#include<math.h>
#include<iomanip>
#include<vector>
#include<map>
#include<set>
#define MOD 10000009
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
typedef double db;
char str[10000001];
int main()
{
  //ios_base::sync_with_stdio(false);
  //cin.tie(NULL);
  
  ll t;
  
   cin>>t;
   while(t--)
   {
    ll ans=1,len,i,j;
   	cin>>str;
   	len=strlen(str);
   	for( i=0, j=len-1;i<j;i++,j--)
   	{
   		if(str[i]!=str[j] && str[i]!='?' && str[j]!='?')
   		{
   			ans=0; break;
		}
		else if((str[i]=='?' && str[j]!='?') || (str[j]=='?' && str[i]!='?'))
		{
			continue;
		}
		else if(str[i]=='?' && str[j]=='?')
		{
			ans=(ans*26)%MOD;
		}
		else
		 continue;
	}
	if(i==j && str[i]=='?')
	 ans=(ans*26)%MOD;
	 cout<<ans<<"\n";
   }
   
     
   	return 0;
}
