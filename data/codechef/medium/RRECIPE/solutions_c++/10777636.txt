#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int t,n,i,p;
long ans;
char a[1000005];
int main()
{
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        p=0; ans=1;
        scanf("%s",a);
        n=strlen(a);
        for(i=0;i<n;i++)
        {
            if((a[i]==a[n-i-1])&&(a[i]=='?'))
            {
                p++;
                if((n%2==0&&i<n/2)||(n%2==1&&i<(n+1)/2))
                    ans=ans*26%10000009;
            }
            else if(a[i]==a[n-i-1]&&(a[i]!='?'))
            {
                p++;
            }
            else if(a[i]=='?')
            {
                p+=2;
            }
        }
        if(p!=n)
            printf("0\n");
        else
            printf("%ld\n",ans);
    }
    return 0;
}
