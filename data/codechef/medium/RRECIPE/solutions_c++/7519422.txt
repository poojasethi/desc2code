#include<cstdio>
#include<iostream>
#include<cstring>
char str[1000001];
using namespace std;
int main()
{
    int t,n,i,j,flag;
    long long counter;
    scanf("%d",&t);
    while(t--)
    {
        cin>>str;
        n=strlen(str);
        if(n==1)
        {
            if(str[0]=='?')
                printf("26\n");
            else
                printf("1\n");
        }
        else
        {
            counter=1;
            if(n%2!=0)
            {
                if(str[n/2]=='?')
                    counter=(counter*26)%10000009;
            }
            for(i=0,j=n-1;i<j;i++,j--)
            {
                if((str[i]=='?'&& str[j]!='?')||(str[i]!='?'&&str[j]=='?'));
                else if(str[i]=='?' && str[j]=='?')
                {
                    counter=(counter*26)%10000009;
                }
                else if(str[i]!=str[j])
                {
                    counter=0;
                    break;
                }
            }
            printf("%lld\n",counter);
        }
    }
    return 0;
}
