#include<iostream>
using namespace std;

int main()
{
	int t,len,count;
	string s;
	cin>>t;
	for(int i=0;i<t;i++)
	{
		len=0;
		cin>>s;
		for(int j=0;s[j]!='\0';j++)
			len++;
	
			count=1;
			
			/*
			if(len%2!=0)
			{
				if(s[len/2]=='?')
				count=26;
			}*/
			
			for(int j=0;j<(len+1)/2;j++)
			{
				if(s[j]==s[len-1-j])
				{
					if(s[j]=='?')
					count=(count*26)%10000009;
				}
				
				else if(s[j]!='?' &&s[len-j-1]!='?' )
				{
						count=0;
						break;
				}
				
						
			}
			
			if(len==1)
			{
				if(s[0]=='?')
				count=26;
			}	
		cout<<count<<"\n";
	
	}

	

	return 0;
	
}
