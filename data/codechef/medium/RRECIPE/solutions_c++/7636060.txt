#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%lld",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
long long int mod=10000009;
char a[1000006];
int main()
{
    long long int t,s,n,i,j,k;
    sd(t);
    while(t--)
    {
        s=1;
        ss(a);
        n=strlen(a);
        for(i=0;i<n/2;i++)
        {
            if(a[i]==a[n-i-1])
            {
                if(a[i]=='?')
                    s=(s*26)%mod;
            }
            else
            {
                if(a[i]!='?'&&a[n-i-1]!='?')
                {
                    s=0;
                    break;
                }
            }
        }
        if(n%2==1&&a[n/2]=='?')
            s=(s*26)%mod;
        printf("%lld\n",s);
    }
    return 0;
}
