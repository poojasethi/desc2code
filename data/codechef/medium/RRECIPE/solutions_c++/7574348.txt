#include<stdio.h>
#include<string.h>
#define mod 10000009
using namespace std;
char s[1000001];
int main()
{
    int t;
    long int l,i;
    long long int ans;
    scanf("%d",&t);
    while(t--)
    {
        ans=1;
        scanf("%s",s);
        l=strlen(s);
        for(i=0;i<l/2;++i)
        {
            if(s[i]==s[l-i-1] && s[i]=='?')
                ans=((ans%mod)*26)%mod;
            if(s[i]!=s[l-i-1] && s[i]!='?' && s[l-1-i]!='?')
                ans=0;
        }
        if(l%2)
        {
            if(s[i]=='?')
                ans=((ans%mod)*26)%mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
