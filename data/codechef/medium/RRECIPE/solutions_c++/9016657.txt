#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 10000009
int main()
{
	int i,t,n,j;
	ll ans;
	char s[1000001];
	scanf("%d",&t);
	while(t--)
	{
		cin>>s;
		n=strlen(s);
		ans=1;
		for(i=0,j=n-1;i<=j;i++,j--)
		{
			if((s[i]!='?')&&(s[j]!='?'))
			{
				if(s[i]!=s[j])
				{
					ans=0;
					break;
				}
			}
			else
			{
				if((s[i]=='?')&&(s[j]=='?'))
				{
					ans=(ans*26)%mod;
				}
			}
		}
		printf("%lld\n",ans);
	}
	return 0;
}
