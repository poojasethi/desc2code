#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
char s[1000005];
#define mod 10000009;
int main() {
	ll t;
	cin >> t;
	while(t--){
		cin >> s;
		ll l =strlen(s);
		ll i,flag = 0,ans = 1,temp;
		if(l == 1){
			if(s[0] == '?') ans = 26;
			else ans = 1;
		}else{
			ll p = l/2;
			if(l%2!=0) p = p + 1;
			for(i=0;i<p;i++){
				if(s[i] == s[l-1-i]){
					if(s[i] == '?'){
						ans = (ans * 26) % 10000009;
						ans = ans % 10000009;
					}
				}else if(s[i] == '?' || s[l-1-i] == '?'){
					ans = ans * 1;
				}else{
					ans = 0;
					break;
				}
			}
			ans = ans % 10000009;
		}
		 cout << ans << endl; 
	}
	return 0;
}
