#include<bits/stdc++.h>
#define D 10000009
#define gcd __gcd
#define getcx getchar
#define pc putchar
#define get(x) scanf("%d",&x)
#define getl(x) scanf("%lld",&x)
#define print(x) printf("%d\n",x)
#define printl(x) printf("%lld\n",x)
#define bitcount __builtin_popcount
using namespace std;
typedef long long ll;

ll ways(string s)
{
    int i=0,j=s.length()-1; ll ans=1;
    while(i<j)
    {
        if(s[i]=='?' && s[j]=='?')
            ans = (ans*26)%D;
        if(s[i]!='?' && s[j]!='?' && s[i]!=s[j])
            return 0;
        i++; j--;
    }
    if(i==j && s[i]=='?')
        ans = (ans*26)%D;
    return ans;
}

int main()
{
    int t; get(t);
    while(t--)
    {
        string s; cin >> s;
        ll ans = ways(s);
        print(ans);
    }
    return 0;
}
