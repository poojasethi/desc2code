#include <iostream>
#include <cstring>
using namespace std;
int main()
{
    int t,s,i,j,n,mod = 10000009;
    cin >> t;
    while(t--) {
        char a[1000001];
        cin >> a;
        s = 1;
        n = strlen(a);
        for(i = 0;i < n/2;++i) {
            if(a[i] == a[n-i-1]) {
                if(a[i] == '?') {
                    s = (s*26) % mod;
                }
            }
            else {
                if(a[i] != '?' && a[n-i-1] != '?')
                {
                    s = 0;
                    break;
                }
            }
        }
        if((n%2) == 1 && a[n/2] == '?')
            s = (s*26) % mod;
        cout << s << "\n";
    }
    return 0;
}
