#include <iostream>
#include <cstdio>
using namespace std;

int main() {
    int T;
    scanf("%d",&T);
    while (T--) {
        int  n;
        scanf("%d",&n);
        int pos=0,neg=0;
        for (int i=0;i<n;i++) {
            int x,y;
            scanf("%d%d",&x,&y);
            if (x-y>0) pos++;
            if (x-y<0) neg++;
        }
        if (pos>0&&neg>0) printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
