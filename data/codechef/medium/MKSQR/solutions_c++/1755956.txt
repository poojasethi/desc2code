#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int t,n,x,y;scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        bool flag1=0,flag2=0,flagmax=0;
        for(int i=0;i<n;++i)
        {
            scanf("%d %d",&x,&y);
            if(x==y)flagmax=1;
            else if((x>=0 && y>=0 && x>y) || (x<=0 && y<=0 && x>y) || (x>=0 && y<=0)){flag1=1;}
            else if((x>=0 && y>=0 && x<y) || (x<=0 && y<=0 && x<y) || (x<=0 && y>=0)){flag2=1;}
        }
        if(flagmax || (flag1&&flag2))printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
