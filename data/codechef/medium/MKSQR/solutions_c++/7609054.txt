#define ll long long int

const int SIZE=1000011;
bool composite[SIZE]={0};
void sieve()
{
    composite[0]=composite[1]=1;
    int i,j;
    for(i=4;i<SIZE;i+=2)
        composite[i]=1;
    int k=(int)(sqrt(SIZE))+1;
    for(i=3;i<=k;i+=2)
    {
        if(!composite[i])
        {

            for(j=i*i;j<SIZE;j+=(i<<1))
            {
                composite[j]=1;
            }
        }

    }
}
int main()
{
int T;
int L;
int R;
ios_base::sync_with_stdio(false);
cin.tie(0);
int N;
cin>>T;
while(T--)
{
    bool Xlesser=false;
    bool Xnotlesser=false;
    cin>>N;
    while(N--)
    {
        cin>>L>>R;
        if(!Xlesser)
        Xlesser=L<R;
        if(!Xnotlesser)
        Xnotlesser=!(L<R);
    }
    printf(Xlesser&&Xnotlesser?"YES\n":"NO\n");
}
return 0;
}
