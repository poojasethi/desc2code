#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
    int t, n, x, y;

    cin >> t;

    while(t--) {

        scanf("%d", &n);

        int pos = 0, neg = 0;

        while(n--) {
            scanf("%d%d", &x, &y);
            if(x - y > 0) pos++;
            else neg++;
        }

        if(!pos || !neg) {
            cout << "NO\n";
        }
        else cout << "YES\n";
    }
}
