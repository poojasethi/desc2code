#include <iostream>
#include <cstdio>
using namespace std;

const int N = 50005;

int n, t;
int x, y, p;
int np, nn;

int main()
{
    scanf("%d", &t);
    while (t--) {
        np = 0; nn = 0;
        scanf("%d", &n);
        for(int i=0; i<n; i++) {
            scanf("%d%d", &x, &y);
            p = x-y;
            if (p>0) np++;
            else if (p<0) nn++;
        }
        if (np==n || nn==n)
            printf("NO\n");
        else {
            printf("YES\n");
        }
    }
    return 0;
}
