#include<iostream>
#include<stdio.h>
 
using namespace std;
 
int main()
{
	int t,n;
	int neg,pos,diff,x,y;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		pos=neg=0;
		while(n--)
		{
			scanf("%d%d",&x,&y);
			diff=x-y;
			if(diff>0)
				pos++;
			else
				neg++;
		}
		if(pos==0||neg==0)
			printf("NO\n");
		else
			printf("YES\n");
	}
	return 0;
}
