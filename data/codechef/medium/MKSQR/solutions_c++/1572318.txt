#include<stdio.h>


int main()
{
    int t;
    scanf("%d",&t);
    
    while(t--)
    {
              int x,y;
              int n;
              scanf("%d",&n);
              int np=0,nn=0;
              while(n--)
              {
                        scanf("%d%d",&x,&y);
                        x=x-y;
                        if(!x)
                        {np=1;nn=1;}
                        else
                        {
                        if(x>0)
                        np=1;
                        if(x<0)
                        nn=1;
                        }
              }
              if(np&&nn)
              printf("YES\n");
              else
              printf("NO\n");
    }
    
    return 0;
}
