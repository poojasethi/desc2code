#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
	int n,t,i,j;
	long long int x,y;
	bool pos=false,neg=false,mid=false;
	scanf("%d",&t);
	for(i=1;i<=t;i++)
	{
		scanf("%d",&n);
		for(j=1;j<=n;j++)
		{
			scanf("%lld %lld",&x,&y);
			if(x>y) {
				pos=true;
            }
			if(x<y) {
				neg=true;
            }
			if(x==y) {
				mid=true;
            }
		}
		if((pos && neg) || mid)  {
			printf("YES\n");
		} else { printf("NO\n");
        }
		pos=false;neg=false;mid=false;
	}
	return 0;
}
