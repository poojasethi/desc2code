    int scan_f( int *location)
{
register int sign=1;
register int digit;
register int result = 0;
if((digit=getchar())=='-')
sign=-1;
else
result = (result << 3) + (result << 1) + digit - '0';
while( (digit = getchar()) >= '0')
result = (result << 3) + (result << 1) + digit - '0';
*location = result*sign;
return 0;
}
