#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
 long long int t,n,x,y,f1,f2,i;
 cin>>t;
 while(t--)
 {
  f1=0,f2=0;
  cin>>n;
  for(i=0;i<n;++i)
  {
   scanf("%lld %lld",&x,&y);
   if(x-y>0)f1=1;
   else f2=1;
   }
   if((f1==1)&&(f2==1))
    printf("YES\n");
   else
    printf("NO\n");
 }
 return 0;
 }
