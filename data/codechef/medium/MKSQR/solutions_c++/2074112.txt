#include<stdio.h>
#include<stdlib.h>
 
int main()
{int t;
long a,b,dif,positive,negative,n,i;
scanf("%d",&t);
while(t--)
{         positive=0;
	  negative=0;
          scanf("%ld",&n);
          for(i=1;i<=n;i++)
          {
                           scanf("%ld%ld",&a,&b);
                           dif=a-b;
                           if(dif<0)
                           negative++;
                           else 
                           positive++;
                           }
                           if(!positive || !negative)
                           printf("NO\n");
                           else printf("YES\n");
                           }
}
