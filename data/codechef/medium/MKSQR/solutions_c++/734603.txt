#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
	int t, n, pos, neg, a1, a2;
	//cin >> t;
	scanf("%d",&t);
	while (t--) {
		pos = 0;
		neg = 0;
		n = 0;
		//cin >> n;
		scanf("%d",&n);
		while (n--) {
			//cin >> a1 >> a2;
			scanf("%d%d",&a1,&a2);
			if (a1 > a2) {
				pos++;
			} else if (a1 < a2) {
				neg++;
			}
		}
		if (pos == 0 || neg == 0) {
			//cout << "NO" << endl;
			printf("NO\n");
		} else {
			//cout << "YES" << endl;
			printf("YES\n");
		}
	}
	return 0;
}
