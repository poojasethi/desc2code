#define DEBUG 0
#include <stdio.h>

int main(void) {
  int tc, n, i, a, b, pos, neg;

  scanf("%d", &tc);
  while(tc--) {
    scanf("%d", &n);
    pos=neg=0;
    for(i=0;i<n;i++) {
      scanf("%d%d", &a, &b);
      if(a>=b)
	pos=1;
      if(b>=a)
	neg=1;
    }
    if(pos&&neg)
      printf("YES\n");
    else
      printf("NO\n");
  }
  return 0;
}
