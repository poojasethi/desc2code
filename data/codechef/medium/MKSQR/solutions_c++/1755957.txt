#include<iostream>

using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        scanf("%d",&n);
        bool ans=0;
        int pos=0;
        while(n--)
        {
            int x,y;
            scanf("%d%d",&x,&y);
            if(pos==0)
        	{
        	    if(x>y)
        	    	pos=1;
    	    	else if(x<y)
    	    		pos=2;
        	}    
            if(x==y || ((pos==1&&x<y) || (pos==2&&x>y)))
            	ans=1;
        }    
        if(ans)
        	printf("YES\n");
    	else
    		printf("NO\n");
    }    
    //cout<<endl;
    //system("pause");
    return 0;
}
