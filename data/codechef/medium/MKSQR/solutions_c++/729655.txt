#include <iostream>
using namespace std;
int n;
int v[50000][2];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		int i;
		int r = 0;
		int l = 0;
		for(i=0;i<n;i++)
		{
			scanf("%d%d",&v[i][0],&v[i][1]);
			if(v[i][0] > v[i][1])
				r++;
			if(v[i][0] < v[i][1])
				l++;
		}
		if(r==0 && l == 0)
			printf("YES\n");
		else
		{
			if(r>0 && l>0)
				printf("YES\n");
			else
				printf("NO\n");
		}
	}
	return 0;
}
