#include <iostream>
#include <cstdio>
using namespace std;

int main() {
  int t, n, x, y, temp;
  bool f1, f2;
  scanf("%d",&t);
  while(t--) {
    scanf("%d",&n);
    f1 = f2 = 0;
    while(n--) {
      scanf("%d%d",&x,&y);
      temp = x-y;
      if(temp<=0) f1 = 1;
      if(temp>=0) f2 = 1;
    }
    if(f1 && f2) printf("YES\n");
    else printf("NO\n");
  }
  return 0;
}
