#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int x,y,t,n,i;
    int flag_positive, flag_negative;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        flag_negative=0; flag_positive=0;
        for(i=0;i<n;i++)
        {
            scanf("%d %d",&x,&y);
            if((!flag_negative)||(!flag_positive))
            {
            if(x-y>0)
                flag_positive=1;
            else
                flag_negative=1;
            }
        }
        if(flag_negative&&flag_positive)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
