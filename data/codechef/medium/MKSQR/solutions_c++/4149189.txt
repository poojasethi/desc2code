#include <stdio.h>

int main ()
{
    int t, n, x, y;
    scanf ("%d", &t);
    while(t--)
    {
        scanf ("%d", &n);
        int pos = 0, neg = 0;
        int i;
        for (i = 0; i < n; i++)
        {
            scanf ("%d %d", &x, &y);
            if (x - y > 0)
                pos = 1;
            else if (x - y < 0)
                neg = 1;
        }
        if (pos && neg)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
