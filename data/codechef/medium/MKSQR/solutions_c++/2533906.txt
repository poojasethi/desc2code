#include <iostream>
#include <cstdlib>
#include <cstdio>

using namespace std;

int main(int argc, char** argv)
{
  int testCases;
  scanf("%d", &testCases);
  
  for (int t=0;t<testCases;t++)
  {
  	int n;
  	int a1;
  	int a2;
  	scanf("%d", &n);
  	int temp;
  	bool posFound = false;
  	bool negFound = false;

  	for (int i=0;i<n;i++)
  	{
  		scanf("%d", &a1);
  		scanf("%d", &a2);
  		temp = a2-a1;
  		if (temp > 0) posFound = true;
  		else if (temp < 0) negFound = true;

  		//if (posFound && negFound) break;  		
  	}

  	if (posFound && negFound)
  		cout<<"YES"<<endl;
  	else
  		cout<<"NO"<<endl;
  }
  
  return 0;
}
