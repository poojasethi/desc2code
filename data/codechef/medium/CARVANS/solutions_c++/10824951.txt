#include <iostream>
using namespace std;

int main() {
  int t;
  cin>>t;
  while(t--) 
  {
      int n;
      cin>>n;
      int a[n];
      for (int i=0;i<n;i++) 
      {
          cin>>a[i];
      }
      int min=a[0],count=1;
      for(int i=1;i<n;i++) 
      {
          if(a[i]<=min)
          {
              min=a[i];
              count++;
          }
      } 
      cout<<count<<endl;
  }
}
