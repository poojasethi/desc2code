#include<iostream>
using namespace std;
int b[101];
int main()
{
    int a,n,k;
    cin >> a >> n >> k;
    int x=a;
    int i;
    for(i=0;i<k;i++)
    {
        b[i]=x%(n+1);
        x=x/(n+1);
        if(x==0)
            break;
    }
    for(i=0;i<k;i++)
        cout << b[i] << " " ;
    return 0;
}
