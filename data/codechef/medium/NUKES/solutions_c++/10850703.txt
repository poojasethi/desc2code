#include<iostream>
using namespace std;
int main()
{
    int p;
    short k,n,i;
    cin>>p>>n>>k;

    for( i=0;i<k;i++)
        {
            cout<<p%(n+1)<<" ";

            p=p/(n+1);
            if(p==0)
                break;
        }
     for(int j=i+1;j<k;j++)
            cout<<'0'<<" ";
    cout<<endl;
    return 0;
}
