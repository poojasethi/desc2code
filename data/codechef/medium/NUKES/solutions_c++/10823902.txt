#include <bits/stdc++.h>
using namespace std;

const int N = 101;
int v[N];

int main(){
	int a, k, n, i = 0;
	scanf("%d %d %d", &a, &n, &k);
	if(n > 0){
		while(a > 0){
			v[i++] = a % (n + 1);
			a /= (n + 1);
			if(i == k)
				break;
		}
	}
	for(i = 0; i < k; ++i)
		printf("%d ", v[i]);
	return 0;
}
