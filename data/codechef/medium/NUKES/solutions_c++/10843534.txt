#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int main()
{
    int n,k,a[101];
    for(int i=0;i<=100;i++)
        a[i]=0;
    scanf("%i%i%i",&a[0],&n,&k);
    for(int i=0;i<k;i++)
    {
        a[i+1]=a[i]/(n+1);
        a[i]%=n+1;
        printf("%i ",a[i]);
    }
    printf("\n");
	return 0;
}
