#include<iostream>
using namespace std;

int main(){
	unsigned int n, i (0), k[100]={0}, a, t;
	cin>>a>>n>>t;
	n+=1;
	while(a && i<t){
		if(a>n-1){
			if(a%n!=0){
				k[i] = a%n;
				a -= k[i];
			}
			a = a/n;
			i++;
		}
		else{
			k[i] = a;
			break;
		}
	}
	for (i = 0 ; i < t ; i++){
		cout<<k[i];
		cout<<" ";
	}
	cout<<"\n";
	return 0;
}
