#include<iostream>
using namespace std;

int main(){
	
		
int particles=0,capacity=0,c=0;
	cin>>particles;
	cin>>capacity;
	cin>>c;
	int chambers[c];
	for(int i=0;i<c;i++){
	chambers[i]=0;
		}
		
		
while(particles-->0){	
	for(int i=0;i<c;i++){
				
		if(chambers[i]<capacity){
			chambers[i]++;
			
			break; 
			} 	
			else{
			chambers[i]=0; //collosion
			}
			
			
		}

}
		
		for(int i=0;i<c;i++){
			cout<<chambers[i]<<" ";
		}

		

return 0;
}
