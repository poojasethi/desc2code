#include <iostream>
using namespace std;

int main() {
	int n,k,*chamber,j,x;
	long long int a;
	cin>>a>>n>>k;
	chamber=new int[k];
	for(int i=a;i>=1;)
	{
		if(chamber[0]==0&&i>=n+1){
		chamber[0]+=n+1;
		i-=n+1;
		}
		else
		{
		chamber[0]++;
		i--;
		}
		j=0;
		while(chamber[j]>n&&j<k)
		{
				chamber[j]=0;
				x=++j;//sigsgev error as ++j sometimes may lead array to be out of bound
				if(x<k)
				chamber[x]+=1;
				else
				break;																	
		}
	}
	for(int l=0;l<k;l++)
	cout<<chamber[l]<<" ";
	delete []chamber;
	return 0;
}
