#!/usr/bin/env python
import sys

if __name__ == '__main__':
    A, N, K = map(int, sys.stdin.readline().strip().split())
    reactors = [0] * K
    reactors[0] = A

    N += 1

    for x in xrange(K-1):
        reactors[x+1] = reactors[x] / N
        reactors[x] = reactors[x] % N

    reactors[-1] = reactors[-1] % N

    print " ".join(map(str, reactors))
