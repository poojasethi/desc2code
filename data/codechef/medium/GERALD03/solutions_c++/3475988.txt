int main()
{
	int T,k,i,j,m,n;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&n);
		for(i=0;i<n;i++)scanf("%d%d",&a[i].x,&a[i].y);
		int x=a[0].x,y=a[0].y,ans=0;
		string s="";
		for(i=1;i<n;i++)
		{
			if(x==a[i].x&&y==a[i].y)continue;
			if(y<=a[i].x)
			{
				if(x<y-1)
				{
					j=y-1-x;ans+=j;
					while(j--)s+="L+";
					x=y-1;
				}
				while(x<a[i].x)ans+=2,s+="R+L+",x++,y++;
				j=a[i].y-y;ans+=j;
				if(j>0)while(j--)s+="R+";
			}
			else
			{
				j=x-a[i].x;ans+=abs(j);
				if(j>0)while(j--)s+="L-";
				else while(j<0)s+="L+",j++;
				j=y-a[i].y;ans+=abs(j);
				if(j>0)while(j--)s+="R-";
				else while(j<0)s+="R+",j++;
			}
			x=a[i].x,y=a[i].y;
		}
		printf("%d\n%s\n",ans,s.c_str());
	}
}
