#include <bits/stdc++.h>

using namespace std;

void getString(int sl,int sr,int el,int er,string &ret) {
	if(sl < el && sl + 1 != sr) {
		ret.push_back('L'); ret.push_back('+');
		getString(sl+1,sr,el,er,ret);
	} else if(sl > el && sl - 1 != sr) {
		ret.push_back('L'); ret.push_back('-');
		getString(sl-1,sr,el,er,ret);
	} else if(sr < er) {
		ret.push_back('R'); ret.push_back('+');
		getString(sl,sr+1,el,er,ret);
	} else if(sr > er) {
		ret.push_back('R'); ret.push_back('-');
		getString(sl,sr-1,el,er,ret);
	}
}
