void fun(int a, int b, int c, int d)
{
	if (a == c && b == d) return;
	else {
		if (((a + 1) <= c) && ((a + 1) != b)) {
			s += "L+";
			fun(a + 1, b, c, d);
		} else if (((a - 1) >= c) && ((a - 1) != b)) {
			s += "L-";
			fun(a - 1, b, c, d);
		} else if (((b + 1) <= d) && ((b + 1) != a)) {
			s += "R+";
			fun(a, b + 1, c, d);
		} else if (((b - 1) >= d) && ((b - 1) != a)) {
			s += "R-";
			fun(a, b - 1, c, d);
		}
	}
}
