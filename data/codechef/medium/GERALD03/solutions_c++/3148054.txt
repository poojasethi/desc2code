string s = "";
int cnt = 0;


void transform(int l1, int r1, int l2, int r2)
{
    while(l1>l2)
    {
        s += "L-";
        l1--;
        cnt++;
    }

    while(l1<l2)
    {
        if((l1+1)==r1 && r1<r2)
        {
            s += "R+";
            r1++;
            cnt++;
        }
        s += "L+";
        l1++;
        cnt++;
    }

    while(r1<r2)
    {
        s += "R+";
        r1++;
        cnt++;
    }

    while(r1>r2)
    {
        s += "R-";
        r1--;
        cnt++;
    }
}

int main()
{
    int t;
    scanf("%d",&t);

    while(t--)
    {
        int n;
        si(n);
        int L[n],R[n];
        cnt = 0;
        s = "";
        loop(i,0,n)
            scanf("%d %d",&L[i],&R[i]);
        
        for(int i=0;i<(n-1);i++)
            transform(L[i],R[i],L[i+1],R[i+1]);
        
        pi(cnt);
        cout << s << endl;
    }
return 0;
}
