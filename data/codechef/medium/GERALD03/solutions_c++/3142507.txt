#include <iostream>
#include <vector>
#include <string>

using namespace std;

int main()
{
    int t,n,i,x,y,a,b,c,d,j;
    vector < pair<int,int> > v;
    ios_base::sync_with_stdio(0);
    cin >> t;
    while ( t-- ) {
          cin >> n;
          v.clear();
          for ( i = 0; i < n; i++ ) {
              cin >> x >> y;
              v.push_back(make_pair(x,y));
          }
          vector <string> ans;
          for ( i = 0; i < n-1; i++ ) {
              a = v[i].first;
              b = v[i].second;
              c = v[i+1].first;
              d = v[i+1].second;
              if ( c >= a && c <= b && d >= a && d <= b ) {
                   for ( j = 1; j <= c-a; j++ ) ans.push_back("L+");
                   for ( j = 1; j <= b-d; j++ ) ans.push_back("R-");
              }
              else if ( c <= a && d >= b ) {
                   for ( j = 1; j <= a-c; j++ ) ans.push_back("L-");
                   for ( j = 1; j <= d-b; j++ ) ans.push_back("R+");
              }
              else if ( c <= a && d <= a ) {
                   for ( j = 1; j <= a-c; j++ ) ans.push_back("L-");
                   for ( j = 1; j <= b-d; j++ ) ans.push_back("R-");
              }
              else if ( c >= b && d >= b ) {
                   for ( j = 1; j <= b-a-1; j++ ) ans.push_back("L+");
                   a = b-1;
                   for ( ;; ) {
                       if ( b < d ) {
                            ans.push_back("R+");
                            b++;
                       }
                       if ( a < c ) {
                            ans.push_back("L+");
                            a++;
                       }
                       if ( a == c && b == d ) break;
                   }
              }
              else if ( c <= a && d >= a && d <= b ) {
                   for ( j = 1; j <= a-c; j++ ) ans.push_back("L-");
                   for ( j = 1; j <= b-d; j++ ) ans.push_back("R-");
              }
              else if ( c >= a && c <= b && d >= b ) {
                   for ( j = 1; j <= c-a; j++ ) ans.push_back("L+");
                   for ( j = 1; j <= d-b; j++ ) ans.push_back("R+");
              }
          }
          cout << ans.size() << endl;
          for ( i = 0; i < ans.size(); i++ ) cout << ans[i];
          cout << endl;
    }
    return 0;
}
