#include<iostream>
using namespace std;
inline int abs(int a)
{
	if(a>=0)
	return a;
	else
	return -1*a;
}
void print(int x1,int y1,int x2,int y2)
{
	while(x1!=x2 || y1!=y2)
	{
		if(x1<x2 && x1+1<y1)
		{
			cout<<"L+";
			x1=x1+1;
			continue;
		}
		if(x1>x2)
		{
			cout<<"L-";
			x1=x1-1;
			continue;
		}
		if(y1<y2)
		{
			cout<<"R+";
			y1=y1+1;
			continue;
		}
		if(y1>y2 && y1-1>x1)
		{
			cout<<"R-";
			y1=y1-1;
			continue;
		}
	}
}
int main()
{
	int t,i,c;
	cin>>t;
	while(t--)
	{
		c=0;
		int n;
		cin>>n;
		int a[n][2];
		for(i=0;i<n;i++)
		{
			scanf("%d %d",&a[i][0],&a[i][1]);
			if(i==0)
			continue;
			c+=abs(a[i][0]-a[i-1][0])+abs(a[i][1]-a[i-1][1]);
		}
		cout<<c<<"\n";
		for(i=0;i<n-1;i++)
		{
			print(a[i][0],a[i][1],a[i+1][0],a[i+1][1]);
		}
		cout<<"\n";
	}
}
