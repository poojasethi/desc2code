#include<iostream>
#include<cstdio>
using namespace std;
char a[10000005];
int main()
{
    int t,n,i,l[2005],r[2005],cnt,k,temp;

    cin>>t;
    while(t--)
    {
        cin>>n;
        k=0;
        cnt=0;
        for(i=1;i<=n;i++)
            scanf("%d%d",&l[i],&r[i]);
        for(i=1;i<=n-1;i++)
        {
            //cout<<"hi";
            while(l[i]!=l[i+1]||r[i]!=r[i+1])
            {
                /*if(l[i]==4&&r[i]==6)
                    cout<<l[i]<<" "<<r[i]<<" "<<l[i+1]<<" "<<r[i+1]<<endl;*/
            if(l[i]<l[i+1]&&l[i]+1<r[i])
            {
                l[i]++;
                cnt++;
                a[k++]='L';
                a[k++]='+';
            }
            else if(l[i]>l[i+1]&&l[i]-1<r[i])
            {
                l[i]--;
                cnt++;
                a[k++]='L';
                a[k++]='-';
            }
            else if(r[i]<r[i+1])
            {
                r[i]++;
                cnt++;
                a[k++]='R';
                a[k++]='+';
            }
            else if(r[i]>r[i+1])
            {
                r[i]--;
                cnt++;
                a[k++]='R';
                a[k++]='-';
            }
            }
        }
        a[k]='\0';
        cout<<cnt<<endl;
        //for(i=0;i<k;i++)
            cout<<a;
        cout<<endl;
    }
}
