#include <bits/stdc++.h>

using namespace std;

int main() {
	//freopen("in.txt","r",stdin);
	int tt;
	scanf("%d",&tt);
	while(tt--) {
		int n;
		scanf("%d",&n);
		stringstream ss;
		int sl,sr,el,er;
		scanf("%d%d",&sl,&sr);
		int len = 0;
		string ret = "";
		for(int i=1;i<n;i++) {
			scanf("%d%d",&el,&er);
			while(sl != el || sr != er) {
				if(sl < el && sl + 1 != sr) {
					ret.push_back('L'); ret.push_back('+');
					sl++;
				} else if(sl > el && sl - 1 != sr) {
					ret.push_back('L'); ret.push_back('-');
					sl--;
				} else if(sr < er) {
					ret.push_back('R'); ret.push_back('+');
					sr++;
				} else if(sr > er) {
					ret.push_back('R'); ret.push_back('-');
					sr--;
				}
			}
		}

		cout << ret.length() / 2 << endl;
		cout << ret << endl;
	}
	return 0;
}
