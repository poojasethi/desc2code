#include<stdio.h>
int main(){
	int L[1000], R[1000], notc, N;
	scanf("%d", &notc);
	int i, total, temp;
	while(notc--){
		total = 0;
		scanf("%d", &N);
		if (N)
			scanf("%d%d", L, R);
		for (i = 1;i<N;i++){
			scanf("%d%d", L+i, R+i);
			if (L[i-1]>L[i])
				temp = L[i-1]-L[i];
			else
				temp = L[i] - L[i-1];
			total += temp;
			if (R[i-1]>R[i])
				temp = R[i-1]-R[i];
			else
				temp = R[i] - R[i-1];
			total += temp;
		}
		printf("%d\n", total);
		for(i = 0;i<N-1;i++){
			if (R[i]<=L[i+1]){
				total = R[i] - L[i]-1;
				while(total--)
					printf("L+");
				total = L[i+1]+1-R[i];
				while(total--)
					printf("R+L+");
				total = R[i+1] - R[i] - (L[i+1]+1-R[i]);
				while(total--)
					printf("R+");
			}
			else{
				if (L[i+1]>L[i]){
					total = L[i+1] - L[i];
					while(total--)
						printf("L+");
				}
				else{
					total = L[i] - L[i+1];
					while(total--)
						printf("L-");
				}
				if (R[i+1]>R[i]){
					total = R[i+1] - R[i];
					while(total--)
						printf("R+");
				}
				else{
					total = R[i] - R[i+1];
					while(total--)
						printf("R-");
				}
			}
		}
		printf("\n");
	}
	return 0;
}
