#include<iostream>
#include<cstring>
using namespace std;
#ifndef ONLINE_JUDGE
inline int getchar_unlocked()
{
	return getchar();
}
#endif
inline int input()
{
	int n = 0, c = getchar_unlocked(),t=0;
	while(c < '0' || c > '9')
	{
		if(c==45)t=1;
		c = getchar_unlocked();
	}
	while(c >= '0' && c <= '9')n = (n<<3) + (n<<1) + c - '0', c = getchar_unlocked();
	
	if(t==0)return n;
	else return -n;
}
char s[9000000];
int main()
{
	int t,n,count=0,a,b,x,y;
	t=input();
	while(t--)
	{
	s[0]='\0';
	count=0;
	n=input();
	x=input();
	y=input();
	n--;
	while(n--)
	{
		a=input();b=input();
	while(x!=a|| y!=b)
	{
	if(x>a && x-1!=y)
	{
		s[count++]='L';
		s[count++]='-';
		x=x-1;
	}
	else if(x<a && x+1!=y)
	{
		s[count++]='L';
		s[count++]='+';
		x++;
	}
	else if(y>b&& y-1!=x)
	{
		s[count++]='R';
		s[count++]='-';
		y--;
	}
	else if(y<b&& y+1!=x)
	{
		s[count++]='R';
		s[count++]='+';
		y++;
	}
	}
	x=a;
	y=b;	
	}
	s[count]='\0';
	printf("%d\n",count/2);
	printf("%s\n",s);
	}
	return 0;
}
