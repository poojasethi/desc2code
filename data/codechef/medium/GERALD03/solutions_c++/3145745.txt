#include <stdio.h>
#include <vector>
using namespace std;
#define inchar getchar_unlocked
inline int inIntNeg() {
	int n=0, ch, neg=0;
	while ((ch = inchar()) <= 32);
	if(ch=='-') neg=1;
	else n=(ch-'0');
	while((ch = inchar()) >= '0') n=(n<<3)+(n<<1)+(ch-'0');
	return neg ? -n : n;
}
int main() {
	int t,n,l,r,i,ln,rn,v;
	vector<int> d;

	t=inIntNeg();
	while(t--) {
		n=inIntNeg();
		l=inIntNeg();
		r=inIntNeg();
		if(n==1) {
			printf("0\n\n");
			continue;
		}
		d.clear();
		for(i=1; i<n; ++i) {
			ln=inIntNeg();
			rn=inIntNeg();

			while(l!=ln || r!=rn) {
				if(l+1<=ln && l+1!=r) {
					d.push_back(1);
					++l;
				}
				else if(l-1>=ln) {
					d.push_back(-1);
					--l;
				}
				else if(r+1<=rn) {
					d.push_back(2);
					++r;
				}
				else if(r-1>=rn && r-1 != l) {
					d.push_back(-2);
					--r;
				}
			}
			l=ln;
			r=rn;
		}
		printf("%d\n",d.size());
		for(i=0; i<(int)d.size(); ++i) {
			v=d[i];
			if(v==1) printf("L+");
			else if(v==-1) printf("L-");
			else if(v==2) printf("R+");
			else printf("R-");
		}
		printf("\n");
	}
	return 0;
}
