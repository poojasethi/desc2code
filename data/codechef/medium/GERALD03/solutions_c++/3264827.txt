#include<stdio.h>
#include<algorithm>
#include<cstring>
#define N 1000003
#include<math.h>
#define K 1000000001
#define IREP(i,a,b) for(i=a;i<=b;++i)
#define DREP(i,a,b) for(i=a;i>=b;--i)
#define scan(x) scanf("%d",&x);
int L[N],R[N];
int count,flag=0;
void fun(int l1,int r1,int l2,int r2)
{if((l1==l2)&&(r1==r2)){flag=1;return;}
if(l1>l2)
{if(l1-1!=r1)
{printf("L-");
fun(l1-1,r1,l2,r2);
}
         
}
 else if(l1<l2)
{if(l1+1!=r1)
{printf("L+");
fun(l1+1,r1,l2,r2);
}}
if(!flag){
 if(r1>r2)
{if(r1-1!=l1)
{printf("R-");
fun(l1,r1-1,l2,r2);
}}
else if(r1<r2)
{if(r1+1!=l1)
{printf("R+");
fun(l1,r1+1,l2,r2);
}
}
         
}
   
}

int main()
{int i,t,n;
scan(t);
while(t--)
{scan(n);count=0;flag=0;
IREP(i,0,n-1){scanf("%d%d",&L[i],&R[i]);if(i>=1){count+=abs((L[i]-L[i-1]));count+=abs((R[i]-R[i-1]));}}
if(n==1){printf("0\n\n");continue;}
printf("%d\n",count); 
IREP(i,0,n-2)
{
fun(L[i],R[i],L[i+1],R[i+1]);flag=0;             
}

printf("\n");          
          
}
return 0;
    
    
    
}
