#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define mod 1000000007
using namespace std;
main()
{   std:ios_base::sync_with_stdio(false);
    int t,n,i,l,r;
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	int il,ir;
    	string ans;
    	
    	for(i=0;i<n;i++)
    	{
    		cin>>l>>r;
    		if(i)
    		{
    		    if(l>il)
				{
					while(il!=ir-1)
					{
						ans+="L+";
						il++;
						if(il==l) break;
					}
					while(il!=l)
					{
						ans+="R+L+";
						il++; ir++;
					}
					while(ir!=r)
					{
						if(ir>r)
						  ans+="R-",--ir;
						else
						  ans+="R+",++ir;
					}
				}	
				else if(l<=il)
				{
					while(il!=l)
					{
						ans+="L-";
						il--;
					}
					while(ir!=r)
					{
						if(ir>r)
						  ans+="R-",--ir;
						else
						  ans+="R+",++ir;
					}
				}
    		}
    		il=l; ir=r;
    	}
    	int l=(int)ans.size();
    	printf("%d\n",l/2);
    	printf("%s\n",ans.c_str());
    }
	return 0;
}
