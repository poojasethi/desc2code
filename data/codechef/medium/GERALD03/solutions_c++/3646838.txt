		cin>>n;
		FOR(i,0,n){

		string ans="";
		FOR(i,0,n-1){
			while(l[i]!=l[i+1] or r[i]!=r[i+1]){
				while(l[i]<l[i+1] and l[i]+1!=r[i]){
					l[i]++;
					ans+="L+";
				}
				while(l[i]>l[i+1]){
					l[i]--;
					ans+="L-";
				}
				if(r[i]<r[i+1]){
					r[i]++;
					ans+="R+";
				}
				if(r[i]>r[i+1] and r[i]-1!=l[i]){
					r[i]--;
					ans+="R-";
				}
			}
		}
		cout<<SZ(ans)/2<<endl;
		cout<<ans<<endl;
	}

}
