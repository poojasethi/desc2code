#include<iostream>
using namespace std;
int main()
{
	 long long int T,N,num1,num2,num3,num4,count,k;
	char arr[8000000];
	cin >> T;
	while (T > 0)
	{
		cin >> N;
		count = 0;
		cin >> num1 >> num2;
		N--;
		arr[0] = '\0';
		k = 0;
		count = 0;
		while (N > 0)
		{
			cin >> num3 >> num4;
			while (1)
			{
				if ((num3 > num1) && (num1+1!=num2))
				{
					arr[k++] = 'L';
					arr[k++] = '+';
					num1++;
					count++;
				}
				else if ((num1 > num3) && (num1-1!=num2))
				{
					arr[k++] = 'L';
					arr[k++] = '-';
					num1--;
					count++;
				}
				else if ((num4 > num2) && (num2+1!=num1))
				{
					arr[k++] = 'R';
					arr[k++] = '+';
					num2++;
					count++;
				}
				else if ((num2 > num4) && (num2-1!=num1))
				{
					arr[k++] = 'R';
					arr[k++] = '-';
					num2--;
					count++;
				}
				else if ((num1 == num3) && (num2 == num4))
				{
					break;

				}

			}
			N--;
		}
		arr[k] = '\0';
		cout << count << endl<<arr<<endl;
       	T--;
	}
	//cin >> T;
	return 0;
}
