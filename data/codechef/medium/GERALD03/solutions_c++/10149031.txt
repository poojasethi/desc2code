#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e3 + 10;
int a[MAXN];
int b[MAXN];
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		int N;
		scanf("%d", &N);
		for(int i = 1;i <= N;++i){
			scanf("%d%d", &a[i], &b[i]);
		}
		int x = a[1], y = b[1];
		string ans = "";
		int cnt = 0;
		for(int i = 1;i <= N;++i){
			while(!(a[i] == x && b[i] == y)){
				while(a[i] < x){
					--x;
					ans += "L-";
					++cnt;
				}
				while(a[i] > x && x < y - 1){
					++x;
					ans += "L+";
					++cnt;
				}
				if(b[i] > y){
					++y;
					ans += "R+";
					++cnt;
				}
				while(b[i] < y && x < y - 1){
					--y;
					ans += "R-";
					++cnt;
				}
			}
		}
		cout << cnt << "\n";
		cout << ans << "\n";
	}
	return 0;
}
