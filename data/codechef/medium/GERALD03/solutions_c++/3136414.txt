using namespace std;
typedef long long ll;
struct structure
{
    int a,b;
}arr[1005],temp;
char str[10000000];
int main()
{
	int t,n,len=0;
	scanf("%d",&t);
	while(t--)
	{
	    scanf("%d",&n);
	    for(int i=0;i<n;i++)
            scanf("%d%d",&arr[i].a,&arr[i].b);
        len=0;
        int cnt=1;
        temp.a=arr[0].a;
        temp.b=arr[0].b;
        while(cnt<n)
        {
            if((temp.a<arr[cnt].a)&&((temp.a+1)!=temp.b))
            {
                temp.a+=1;
                str[len*2]='L';
                str[len*2+1]='+';
                len++;
            }
            else if((temp.a>arr[cnt].a)&&((temp.a-1)!=temp.b))
            {
                temp.a-=1;
                str[len*2]='L';
                str[len*2+1]='-';
                len++;
            }
            else if((temp.b<arr[cnt].b)&&((temp.b+1)!=temp.a))
            {
                temp.b+=1;
                str[len*2]='R';
                str[len*2+1]='+';
                len++;
            }
            else if((temp.b>arr[cnt].b)&&((temp.b-1)!=temp.a))
            {
                temp.b-=1;
                str[len*2]='R';
                str[len*2+1]='-';
                len++;
            }
            if((temp.a==arr[cnt].a)&&(temp.b==arr[cnt].b))cnt++;
        }
        str[len*2]='\0';
        printf("%d\n%s\n",len,str);
	}
	return 0;
}
