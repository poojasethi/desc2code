#include <stdio.h>
#include <iostream>
using namespace std;
 
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n,l,r,curl,curr,count = 0;
        string ans = "";
        scanf("%d", &n);
         scanf("%d%d",&curl,&curr);
        for(int i = 1; i < n; i++)
        {
            scanf("%d%d",&l,&r);
            while(curl != l || curr != r)
            {
                if(l > curl && curl+1 != curr)
                {
                    curl++;
                    ans += "L+";
                }
                else if(l < curl && curl-1 != curr)
                {
                    curl--;
                    ans += "L-";
                }
                else if(r > curr && curr+1 != curl)
                {
                    curr++;
                    ans += "R+";
                }
                else if(r < curr && curr-1 != curl)
                {
                    curr--;
                    ans += "R-";
                }
                count++;
            }
        }
       // cout<<"l,r"<<curl<<curr<<endl;
        cout<<count<<endl<<ans<<endl;
 
    }
    return 0;
}
