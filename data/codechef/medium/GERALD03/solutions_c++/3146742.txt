#include<iostream>
#include<string>
#include<cmath>
#include<cstdlib>
using namespace std;
int main()
{
	
	int a,b,c,d,n,t,i,j,tot;
	cin>>t;
	while(t--) {
		string str="";
		tot=0;
		cin>>n;
		int l[n],r[n];
		for(i=0;i<n;i++) {
			cin>>l[i]>>r[i];
		}
		for(i=0;i<(n-1);i++) {
			a=l[i];
			b=r[i];
			c=abs(l[i+1]-l[i]);
			d=abs(r[i+1]-r[i]);
			c+=d;
			for(j=0;j<c;j++) {
				if(a<l[i+1]&&(b-a)!=1) {
					str+="L+";
					a++;
					tot++;
				}
				else if(a>l[i+1]&&(a-b)!=1) {
					str+="L-";
					a--;
					tot++;
				}
				else if(b<r[i+1]&&(a-b)!=1) {
					str+="R+";
					b++;
					tot++;
					
				}
				else {
					str+="R-";
					b--;
					tot++;
				}
			}
		}
		cout<<tot<<"\n"<<str<<endl;
		
	}
	return 0;
	
}
