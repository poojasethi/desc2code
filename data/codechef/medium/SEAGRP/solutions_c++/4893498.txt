int adj[105][105];
int main()
{
  int t;

  {
    memset(adj,0,sizeof(adj));
    
    n=read();m=read();

bool chek_degree(bool vis[105])
{
  for (int i=1;i<=n;i++)
  {
    if(vis[i]==0)
    {
             // i not matched. match it
       for (int j=1;j<=n;j++)
       {
           if((adj[i][j]==1) and (vis[j]==0))
           {
              vis[i]=1; vis[j]=1;
              if (chek_degree(vis)==true) return true;
              vis[i]=0;
              vis[j]=0;
            }
       }
     return false;
   }
  }
return true;
}
