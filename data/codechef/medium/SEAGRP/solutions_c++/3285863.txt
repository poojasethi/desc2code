// printf("Here I Am in second loop");

// printf("Here I Am entered in if");
if(reduce(n,count-2))

else{
arr[i]=0;

}
}
}
if(j==n){
// printf("Here I Am and Edges over so early!!");

}
}
}
// printf("Here I Am Unable to find solution!!");

}
