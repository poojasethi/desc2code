#include <iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#define pb push_back
#define gc getchar_unlocked
using namespace std;
int n;
vector<int>adj [100];
bool dfs(int u,bool visited[],int match[])
{
	visited[u]=true;
	for(int i=0;i<adj[u].size();i++)
	{	int v=adj[u][i];
		if(!visited[v])
		{	visited[v]=true;
			if(match[v]<0 || dfs(match[v],visited,match))
			{
				match[v]=u;
				match[u]=v;
				return true;
			}
		}
	}
	return false;
}
int maxBPM()
{
	int i,count=0,match[n];
	memset(match,-1,sizeof(match));
	for(i=0;i<n;i++)
	{
		if(match[i]<0)
		{	bool visited[n];
			memset(visited,0,sizeof(visited));
			if(dfs(i,visited,match))
				count++;
		}
	}
	return count;
}
int main() {
	int t,i,u,v,m;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&m);
		for(i=0;i<n;i++)adj[i].clear();
		for(i=0;i<m;i++)
		{
			scanf("%d%d",&u,&v);
			adj[u-1].pb(v-1);
			adj[v-1].pb(u-1);
		}
		if(n&1){printf("NO\n");continue;}
		if(2*maxBPM()==n)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
