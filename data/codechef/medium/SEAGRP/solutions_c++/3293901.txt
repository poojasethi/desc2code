#include<iostream>
using namespace std;
int perfect_match(int *arr[],bool *vis,int n,int c)
{
    if(c==n)return 1;
    int i,j;
    for(i=0;i<n;i++)
    {
                    if(!vis[i])
                    {
                             for(j=0;j<n;j++)
                             {
                                             if(arr[i][j] && !vis[j])
                                             {
                                                          vis[i]=1;vis[j]=1;
                                                          if(perfect_match(arr,vis,n,c+2))return 1;
                                                          vis[i]=0;vis[j]=0;
                                             }
                             }
                             if(j==n)return 0;
                    }
    }return 0;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
              int n,m,i,j;
              cin>>n>>m;
              int *arr[n];
              for(i=0;i<n;i++)
              {
                              arr[i]=new int[n];
              }
              for(i=0;i<n;i++)
              for(j=0;j<n;j++)arr[i][j]=0;
              for(i=0;i<m;i++)
              {
                              int src,dest;
                              cin>>src>>dest;
                              arr[src-1][dest-1]=1;
                              arr[dest-1][src-1]=1;
              }
              bool vis[n];
              for(i=0;i<n;i++)vis[i]=0;
              if(n%2)cout<<"NO\n";
              else
              if(perfect_match(arr,vis,n,0))cout<<"YES\n";
              else
              cout<<"NO\n";
    }
}
