#include<stdio.h>
#include<string.h>

int graph[100][100];
int visited[100];

int perfect_matching(int n,int c)
{
    int i,j;
    int f=0;
    for(j=0;j<n;j++)
    if(visited[j]==0)
    {
        f=1;
        break;
    }
    if(f==0||c>=n)
    return 1;

    if(visited[c]==1)
    return perfect_matching(n,c+1);

    for(i=0;i<n;i++)
    if(graph[c][i]==1&&visited[i]==0)
    {
        visited[c]=visited[i]=1;
        int x=perfect_matching(n,c+1);
        if(x==1)
        return 1;
        else
        {
            visited[c]=visited[i]=0;
        }
    }
    if(visited[c]==1)
    return perfect_matching(n,c+1);
    else
    return 0;
}

int main()
{
    int t;
    scanf("%d",&t);

    while(t--)
    {
        int n,m,i,x,y,j;
        scanf("%d%d",&n,&m);

        for(i=0;i<n;i++)
        for(j=0;j<n;j++)
        graph[i][j]=0;

        for(i=0;i<m;i++)
        {
            scanf("%d%d",&x,&y);
            graph[x-1][y-1]=1;
            graph[y-1][x-1]=1;
        }
        memset(visited,0,sizeof(visited[0])*n);

        if(n%2)
        printf("NO\n");
        else if(perfect_matching(n,0))
        printf("YES\n");
        else printf("NO\n");
    }
    return 0;
}
