vector<int> match;
vector<bool> seen;
vector<vector<int> > g;
// Bipartite Matching using DFS
int bpm(int u)
{
	int temp;
	seen[u]=1;
    for (int v=0;v<g[u].size();v++)
		{
			temp=g[u][v];
            if (seen[temp]) continue;
			seen[temp]=true;
			if (match[temp]<0 || bpm(match[temp]))
			{
				match[temp]=u;
				match[u]=temp;
				return true;
			}
		}
	return false;
}

int main()
{
    std::ios_base::sync_with_stdio(false);
    int t,n,m,a,b;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        g.clear();match.clear();seen.clear();
        g.resize(n+1);match.resize(n+1,-1);seen.resize(n+1);
        while(m--)
        {
            cin>>a>>b;
            g[a].pb(b);
            g[b].pb(a);
        }
        bool d,fail=0;
        if (n%2) fail=1;
        if (!fail)
        {
            for (int i=1;i<=n;i++)
        {
            if (match[i]>0) continue;
            fill(seen.begin(),seen.end(),0);
            //seen[i]=1;
            d=bpm(i);
            if (!d) {fail=1;break;}
        }
        }
        if (fail) cout<<"NO\n";
        else cout<<"YES\n";
    }
return 0;
}
