#include<stdio.h>
#include<stdlib.h>

double mat[105][105];
int ed[101][101];
int n,m;
int sign;

void swap_rows(int i,int j)
{
	double temp;
	int k;
	for(k = 0; k < n; k++)
	{
		temp = mat[i][k];
		mat[i][k] = mat[j][k];
		mat[j][k] = temp;
	}
	if(sign)sign = 0;
	else sign = 1;
}

int det_test()
{
	int i,nzr,j,k;
	double q;
	sign = 0;
	for( i = 0; i < n; i++)
	{
		if(mat[i][i] == 0)
		{
			nzr = i;
			for(k = i + 1;k < n;k++)
				if(mat[k][i] != 0){ nzr = k; break; }
			if(i != nzr)swap_rows(i,nzr);
		}
		if(mat[i][i] != 0)
		{
			for(j = i + 1;j < n; j++)
			{
				q = mat[j][i] / mat[i][i];
				for(k = i ;k < n;k++)mat[j][k] = mat[j][k] - ( q * mat[i][k] );
			}
		}
	}
	double det = mat[0][0];
	for(i = 1; i < n; i++)det = det * mat[i][i];
	return det;
}

int main()
{
	int t;
	int i,j;
 	int vi,vj;
	int arand;

	scanf("%d",&t);
	while(t--)
	{	
		arand = 1;
		scanf("%d%d",&n,&m);

		for(i = 0;i < n;i++)
		for(j = 0;j < n;j++)
		mat[i][j] = 0;

		for(i=1;i<=m;i++)
		{
			scanf("%d%d",&vi,&vj);
			vi--;
			vj--;
			
			if(mat[vi][vj] != 0)continue;
			
			if(vi < vj)
			{
		 		mat[vi][vj] = arand;
				mat[vj][vi] = -arand;
			}
			else if(vi > vj)
			{
				mat[vi][vj] = -arand;
		 		mat[vj][vi] = arand;
			}
			arand++;
		}		
		if(n & 1)puts("NO");
		else if(det_test())puts("YES");
		else puts("NO");
	}
	return 0;
}
