#include <stdio.h>
#include <string.h>
int main()
{
	int i,j,f,k,p,min,tot,t,n,u,v,m,mat[105][105],mark[105],deg[105];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		memset(mat,0,sizeof(mat));
		for(i=0;i<m;i++)
		{
			scanf("%d %d",&u,&v);
			mat[u][v]=mat[v][u]=1;
		}
		if(n&1){printf("NO\n");continue;}
		//calculate degree of each vertex
		for(i=1;i<=n;i++)
		{
			p=0;
			for(j=1;j<=n;j++)
				p+=mat[i][j];
			deg[i]=p;
		}
		memset(mark,0,sizeof(mark));tot=0;
		while(tot!=n)
		{
			min=1000000;
			for(i=1;i<=n;i++)
				if(!mark[i]&&(deg[i]<min)){p=i;min=deg[i];} //choose the min degree vertex
			
			mark[p]=1;tot++;
			min=1000000;f=0;
			for(i=1;i<=n;i++)
				if(!mark[i]&&mat[p][i]&&(deg[i]<min)){min=deg[i];k=i;f=1;}
			if(!f)break;
			mark[k]=1;tot++;
			for(i=1;i<=n;i++)
			{
				if(mat[p][i]){deg[i]--;mat[p][i]=0;}
				if(mat[k][i]){deg[i]--;mat[k][i]=0;}
			}
		}
		if(f)printf("YES\n");else printf("NO\n");
	}
	return 0;
}
