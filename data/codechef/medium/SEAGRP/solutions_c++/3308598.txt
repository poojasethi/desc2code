#include<stdio.h>
int scan()
{
    int n=0;
    int ch=getchar_unlocked();
    while(ch<'0'||ch>'9')
        ch=getchar_unlocked();
    while(ch>='0'&&ch<='9')
    {
        n=(n<<3)+(n<<1)+ch-'0';
        ch=getchar_unlocked();
    }
    return n;
}
int main()
{
	int t,n,m,i,x,y;
	t=scan();
	while(t--)
	{
		n=scan();
		m=scan();
		int g[101][101]={{0}};
		int v[101]={0};
		int d[101]={0};
		for(i=1;i<=m;i++)
		{
			x=scan();
			y=scan();
			if(g[x][y]==0)
			{
    			g[x][y]=g[y][x]=1;
    			d[x]++;
    			d[y]++;
    		}
		}
		if(n%2!=0)
		{
    		printf("NO\n");
    		continue;
    	}
    	while(true)
    	{
    		int mx,my;
    		x=y=0;
    		mx=101,my=101;
    		for(i=1;i<=n;i++)
    		{
    			if(v[i]==0)
    			{
    				if(mx>d[i])
    				{
    					mx=d[i];
    					x=i;
    				}
    			}
    		}
    		for(i=1;i<=n;i++)
    		{
    			if(g[x][i]==1&&v[i]==0&&x!=i)
    			{
    				if(my>d[i])
    				{
    					my=d[i];
    					y=i;
    				}
    			}
    		}
    		if(x!=0&&y!=0)
    		{
    			v[x]=v[y]=1;
    			for(i=1;i<=n;i++)
    			{
    				if(g[x][i]==1)
    				{
    					g[x][i]=g[i][x]=0;
    					d[i]--;
    				}
    				if(g[y][i]==1)
    				{
    					g[y][i]=g[i][y]=0;
    					d[i]--;
    				}
    			}
    		}
    		else
    		if((x==0&&y==0)||(x!=0&&y==0))
    		break;
    	}
    	for(i=1;i<=n&&v[i];i++);
    	if(i>n)
    	printf("YES\n");
    	else
    	printf("NO\n");
	}
	return 0;
}
