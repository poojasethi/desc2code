#include<iostream>
#include<cstdio>
#include<vector>
#include<string>
#include<map>
#include<utility>
#define fi first 
#define se second 
using namespace std;


int main()
{
int t;
scanf("%d",&t);
while(t--)
{
int n,i;
vector<pair<pair<string,string> ,int> > arr;
map<string,int> ma;
map<string,int> q1,q2;

scanf("%d",&n);
string c1,c2,c3;
int x;

for(i=0;i<n-1;i++)
{
pair<pair<string,string> ,int > pa;
cin>>c1>>c2>>x>>c3;

pa.fi.fi = c1;
pa.fi.se = c2;
pa.se = x;


q1[c1] = i;
q2[c2] = i;

arr.push_back(pa);
ma[c1] = 0;
ma[c2] = 0;

}


for(i=0;i<n-1;i++)
{

c1 = arr[i].fi.fi;
c2 = arr[i].fi.se;
x = arr[i].se;

ma[c1]++;
ma[c2]++;

}

int key;
for(i=0;i<n-1;i++)
{
c1 = arr[i].fi.fi;
if(ma[c1] == 1) {key = i; break;}
}


int count = 0,sum = 0;
while(count<n-1)
{

c1 = arr[key].fi.fi;
c2 = arr[key].fi.se;
x = arr[key].se;

cout<<c1<<" "<<c2<<" "<<x<<"$\n";	

key = q1[c2];
count++;

sum+=x;
}

cout<<sum<<"$\n";

}
return 0;
}
