#include <stdio.h>
#include <string.h> // strcmp

int main() {

    int T;
    scanf("%d\n",&T);
    
    while (T-->0) {
        
        int N;
        scanf("%d\n",&N);
        
        char A[5001][51], B[5001][51];
        int  C[5001], prev[5001] = {0};
        int  id;
        
        // read input
        for (int i=1; i<N; i++) {
            int j = 0; char c = getchar();
            while (c != ' ' && c != '\t') {
                A[i][j] = c; c = getchar(); j++;
            }
            A[i][j] = '\0';
            j = 0; c = getchar();
            while (c != ' ' && c != '\t') {
                B[i][j] = c; c = getchar(); j++;
            }
            B[i][j] = '\0';
            scanf("%ld$\n",&C[i]);
        }
        
        // for each i find previous city
        for (int i=1; i<N; i++) {
            int f = 1;          
            for (int j=1; j<N; j++) {
                if (prev[j] != 0) continue; // city has been already scanned
                if (!strcmp(A[i],B[j])) {
                    prev[j] = i;
                    f = 0;
                    break; // previous city found, exit loop for j
                }
            }
            if (f) id = i; // this happen only for the first city
        }

        long sum = 0;
        for (int i=1; i<N; i++) {
            printf("%s %s %ld$\n",A[id],B[id],C[id]);
            sum += C[id];
            id = prev[id];
        }
        printf("%ld$\n",sum);        
    }    
    return 0;
}
