#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		int N;
		scanf("%d", &N);
		int ans = 0;
		map<pair<string, string>, string> cost;
		map<string, string> parent;
		map<string, string> child;
		for(int i = 1;i < N;++i){
			string A, B, C;
			cin >> A >> B >> C;
			cost[make_pair(A, B)] = C;
			parent[B] = A;
			child[A] = B;
			if(parent[A] == "") parent[A] = "";
		}
		string start;
		for(map<string, string> :: iterator it = parent.begin();it != parent.end();++it){
			if(it->second == "")
				start = it->first;
		}
		while(child[start] != ""){
			string &C = cost[make_pair(start, child[start])];
			cout << start << " " << child[start] << " " << C << "\n";
			start = child[start];
			int N = C.size();
			int nc = 0;
			for(int i = 0;i < N - 1;++i){
				nc = (nc * 10 + C[i] - '0');
			}
			ans += nc;
		}
		cout << ans << "$\n";
	}
	return 0;
}
