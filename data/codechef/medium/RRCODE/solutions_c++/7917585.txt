#include<stdio.h>
#include<stdio.h>
int main()
{
	int i,t;
	long long int a,b,XOR,AND,OR,ans,n,k,ians;
	char str[5];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld%lld%lld",&n,&k,&ians);
		scanf("%lld%lld",&a,&b);
		XOR = a^b;
		AND = a&b;
		OR = a|b;
		for(i=2;i<n;i++)
		{
			scanf("%lld",&a);
			XOR = XOR^a;
			AND = AND&a;
			OR = OR|a;
		}
		scanf("%s",str);
		if(k>=1)
		{
			if(str[0]=='X')
				{
					if(k%2==1)
				printf("%lld\n",XOR^ians);
				else
				printf("%lld\n",ians);	
				}
				
			else if(str[0]=='A')
				printf("%lld\n",AND&ians);
			else if(str[0]=='O')
				printf("%lld\n",OR|ians);
		
		}
		else
		printf("%lld\n",ians);
				
	}
}
