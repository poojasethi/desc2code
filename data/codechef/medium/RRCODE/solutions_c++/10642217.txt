#include <iostream>
#include<cstring>
int main()
{ long long int t,n,k,a[1000],i,ans,v;
   char op[5];
  scanf("%lld",&t);
  while(t--)
  {scanf("%lld%lld%lld",&n,&k,&ans);
    for(i=0;i<n;i++)
     {
     scanf("%lld",&a[i]);
     }
     scanf("%s",op);
     if(strcmp(op,"AND")==0)
      { v=a[0];
        for(i=0;i<n;i++)
        {v=v&a[i];
        }
        if(k!=0)
        {ans=ans&v;
        }
      }
      else if(strcmp(op,"OR")==0)
      {
        v=a[0];
        for(i=0;i<n;i++)
        {v=v|a[i];
        }
        if(k!=0)
        {ans=ans|v;
        }
       }
       else
       {v=0;
        for(i=0;i<n;i++)
        {v=v^a[i];
        }
        if(k%2!=0)
        {ans=ans^v;
        }
        else
        {ans=ans^0;
        }
       }
       printf("%lld\n",ans);
  }
  
  return 0;
}
