#include<cstdio>
#include<cstring>
using namespace std;
char str[][10]={"XOR","AND","OR"};
int Operate(int a,int b,int op){
    if(op==0){
        return a^b;
    }
    else if(op==1){
        return a&b;
    }
    else
    {
        return a|b;
    }
}
int find_op(char s[]){
    int o=-1;
    for(int i=0;i<3;i++){
        if(strcmp(s,str[i])==0){
            o=i;
        }
    }
    return o;
}
int F(int n,int k,int ans,int op,int arr[]){
    if(k%2){
        k=3;
    }
    else if(k!=0)
    {
        k=4;
    }
    for(int i=0;i<n;i++){
        for(int j=0;j<k;j++){
            ans=Operate(ans,arr[i],op);
        }
    }
    return ans;
}
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
    int n,k,ans;
    scanf("%d %d %d",&n,&k,&ans);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d",&arr[i]);
    }
    char operation[10];
    scanf("%s",&operation);
    int op=find_op(operation);
    ans=F(n,k,ans,op,arr);
    printf("%d\n",ans);
    }
}
