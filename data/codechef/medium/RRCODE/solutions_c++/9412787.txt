#include <bits/stdc++.h>
using namespace std;

int x(int a,int b) { return(a ^= b);}
int an(int a,int b) {return(a &= b);}
int o(int a,int b) {return(a |= b);}

int main() {
	int t;
	cin >> t;

	while(t--) {
		int n,k,ans;
		cin >> n >> k >> ans;

		int a[n],i;

		for(i = 0;i < n;i++) {
			cin >> a[i];
		}
		char opr[10];
		cin >> opr;

		if(opr [0] == 'A' || opr[0] == 'O') {
			k = min(k,1);	
		}
		else k &= 1;

		while(k--) {
			for(i = 0;i < n;i++) 
				if(opr[0] == 'A')
					ans = an(ans,a[i]);
				else if(opr[0] == 'X')
					ans = x(ans,a[i]);
				else
					ans = o(ans,a[i]);
		}
		cout << ans << "\n";
	}
	return 0;
}
