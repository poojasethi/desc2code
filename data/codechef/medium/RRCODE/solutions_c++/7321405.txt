#include <iostream>
#include<cstdio>

using namespace std;

int main()
{
    long long int t,i,j,k,l,ans,n,m;
    cin>>t;
    while(t--)
    {
    cin>>n>>k>>ans;
    long long int an=ans;
    long long int a[n];
    for(i=0;i<n;i++)
    scanf("%lld",&a[i]);
    string q;
    cin>>q;
    if(q[0]=='A')
    {
    for(i=0;i<n;i++)
    ans=(ans&a[i]);
    }
    else if(q[0]=='X')
    {
    for(i=0;i<n;i++)
    ans=(ans^a[i]);
    }
    else
    {
    for(i=0;i<n;i++)
    ans=(ans|a[i]);
    }

    if(k==0)
    ans=an;
    else if(q[0]=='X' && !(k&1))
    ans=an;

    cout<<ans<<endl;
    }
    return 0;
}
