#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main()
{
	ios::sync_with_stdio(false);
	int t;cin>>t;
	while(t--)
	{
		int n;
		ll k,ans;
		cin>>n>>k>>ans;
		ll a[n];
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		string s;
		cin>>s;
		ll fans=ans;
		if(s[0]=='X')
		{
			for(int i=0;i<n;i++)
			{
				fans^=a[i];
			}
			if(k%2==0)
			{
				fans=ans;
			}
		}
		else if(s[0]=='O')
		{
			if(k!=0)
			{
				for(int i=0;i<n;i++)
				{
					fans|=a[i];
				}
			}
		}
		else
		{
			if(k!=0)
			{
				for(int i=0;i<n;i++)
				{
					fans&=a[i];
				}
			}
		}
		cout<<fans<<"\n";
	}
}
