#include<iostream>
#include<cstring>
#include<stdio.h>
using namespace std;

 int yo(int n, int k, int ans,char s[], int a[])
  {
       int j;
      if(k==0)
        return ans;
      else{
      if(s[0]=='O')
      {
          for(j=0;j<n;j++)
          {
              ans=(ans|a[j]);
          }
      return ans;
      }

      else if(s[0]=='A')
      {
          for(j=0;j<n;j++)
          {
              ans=ans&a[j];
          }
            return ans;


      }
      else if(s[0]=='X')
      {
          if(k%2==0)
            return ans;
          else{
          for(j=0;j<n;j++)
          {
              ans=ans^a[j];
          }
            return ans;}

      }
      }
}

int main()
{
