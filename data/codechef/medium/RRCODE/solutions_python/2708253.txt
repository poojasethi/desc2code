t = int(raw_input())
for l in range(t):
    n,k,ans = map(int,raw_input().strip().split())
    arr = map(int,raw_input().strip().split())
    oper = raw_input().strip()
    if (oper=='XOR'):
        k = k%2
    else:
        k = min(k,1)
    for i in range(k):
        for j in range(n):
            if(oper=='OR'):
                ans = ans|arr[j]
            elif(oper=='AND'):
                ans =ans&arr[j]
            else:
                ans =ans^arr[j]
    print ans
