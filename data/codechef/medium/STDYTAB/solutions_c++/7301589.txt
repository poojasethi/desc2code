#include<bits/stdc++.h>
using namespace std;

int t,i,j;
#define MOD 1000000000

 long long int dp[2005][2005], store[4005][4005];

void banalopehle()
{

	for(i=0;i<4005;++i)
	{
		store[i][i]=1;
		store[i][0]=1;
	}
	for(i=2;i<4005;++i)
	for(j=1;j<i;++j)
	store[i][j]=(store[i-1][j]+store[i-1][j-1])%MOD;
	
}
int main()
{	banalopehle();
	scanf("%d",&t);
	while(t--)
	{	memset(dp,0,sizeof(dp));
		int n,m;
		scanf("%d%d",&n,&m);
		for(j=0;j<=m;++j)
		{
			if(j==0)
				dp[1][j]=1;
			else
				dp[1][j]=(dp[1][j-1]+store[j+m-1][j])%MOD;
		}
		
		for(i=2;i<=n;++i)
		{
			for(j=0;j<=m;++j)
			{	
				if(j==0)
					dp[i][j]=(dp[i-1][j]*store[j+m-1][j])%MOD;
				else
					dp[i][j]=(  dp[i][j-1]+ (dp[i-1][j]*store[j+m-1][j]))%MOD;
				
			}
		}
		printf("%lld\n",dp[n][m]);
		
			}
			return 0;
		}
