#include<bits/stdc++.h>
#define lli long long int
#define MM 1000000000
using namespace std;
lli AB[4012][2012];
lli A[2005];
lli n,m;
void FOOO();
int main(){
	FOOO();
	int t;
	cin>>t;
	while(t--){
		cin>>n>>m;
		lli O=0,ct=n;
			for(int i=2002;i>=0;i--)
			A[i]=0;
		for(int i=m;i>=0;i--){
			A[i]=(AB[m+i-1][m-1]+A[i+1])%MM;
		}
		
		while(ct!=1){
			for(int i=m;i>=0;i--){
				A[i]=((AB[m+i-1][m-1]*A[i])+A[i+1])%MM;
			}
			--ct;
		}
		O=A[0];
		printf("%lld\n",O%MM);
	}
	return 0;
}
