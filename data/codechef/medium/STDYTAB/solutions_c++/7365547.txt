#include<iostream>
#define MOD 1000000000
using namespace std;
 
long long int pascal[4002][4002];
long long int a[2002];
long long int b[2002];
long long int c[2002];
 
void init()
 {
  pascal[0][0]=1;
  for(int i=1;i<4002;i++)
 	for(int j=0;j<=i;j++)
	  if(j==0||j==i)
	   pascal[i][j]=1;
     else
       pascal[i][j]=(pascal[i-1][j]+pascal[i-1][j-1])%MOD;
 }
 
int main()
 
{
 int t,m,n,i,j;
 long long int ans,dummy;
 init();
 cin>>t;
 while(t--)
 {
  cin>>n>>m;
  a[0]=b[0]=c[0]=ans=1;
  for(i=1;i<=m;i++)
  {
   a[i]=pascal[m-1+i][m-1];
   c[i]=a[i];
   ans=(ans+a[i])%MOD;
  }
  for(i=1;i<n;i++)
  {
      for(j=0;j<=m;j++) b[j]=a[j];
      a[0]=ans;
      dummy=ans;
      for(j=1;j<=m;j++)
      {
      	   ans=(ans-(b[j-1]-MOD))%MOD;
           a[j]=ans;
           a[j]=(a[j]*c[j])%MOD;;
           dummy=(dummy+a[j])%MOD;
      }
      ans=dummy;
  }
  cout<<ans<<endl;
 }
}
