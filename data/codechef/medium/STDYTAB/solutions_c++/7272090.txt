#include <iostream>
#include<cstdio>
#define ll long long
using namespace std;
#define mod 1000000000
ll int arr[2000];
ll int c[4010][4010];
ll int minii(ll int a, ll int b)
{
    return a<b?a:b;
}
void init()
{
    c[0][1]=0;
    c[0][0]=1;
    ll int i,j;
    for(i=1;i<=4008;i++)
    {
        for(j=1,c[i][0]=1;j<=minii(i,2001);j++)
        {
            c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
        }
    }
}
int main()
{
   init();
   ll int i,j,k,m,n,t,l,x,y;
   cin>>t;
   while(t--)
   {
 
       cin>>n;
       cin>>m;
       j=0;
       for(i=0;i<=m;i++)
       {
           arr[i]=c[i+m-1][m-1];
           j=(j+c[i+m-1][m-1])%mod;
       }
       for(x=1;x<n;x++)
       {
           k=j;
           l=arr[0];
           arr[0]=k;
           k=(k-l)%mod;
           if(k<0)
            k=k+mod;
           j=arr[0];
           for(i=1;i<=m;i++)
           {
               l=arr[i];
               arr[i]=(c[i+m-1][m-1]*k)%mod;
               k=(k-l)%mod;
           if(k<0)
            k=k+mod;
               j=(j+arr[i])%mod;
           }
       }
       cout<<j<<endl;
 
   }
}
