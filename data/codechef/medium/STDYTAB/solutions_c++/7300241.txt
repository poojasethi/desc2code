#include <bits/stdc++.h>
#define M 1000000000
#define lli long long

using namespace std;

int C[4002][4002];
int dp[2002][2002];
int vis[2002][2002];
int tc;
int m;

//Code for FAST INPUT :)
template <typename T>
inline void fi(T *a)
{
    register char c=0;
    while (c<33) c=getchar_unlocked();
    *a=0;
    int tmp = 0;
    while (c>33)
    {
        if ( c == 45 ) tmp = 1;
        else *a=*a*10+c-'0';
        c=getchar_unlocked();
    }
    if ( tmp == 1 ) *a = 0-(*a);
}

void pre()
{
	C[0][0] = 1;
	for ( int i = 1; i <= 4001; i++ ) {
		for ( int j = 0; j <= i; j++ ) {
			if ( j == 0 || i == j ) C[i][j] = 1;
			else {
				C[i][j] = C[i-1][j-1] + C[i-1][j];
				if ( C[i][j] >= M ) C[i][j] -= M;
			}
		}
	}
	return;
}

int f(int n, int k)
{
	if ( n == 0 ) return 1LL;
	if ( vis[n][k] == tc ) return dp[n][k];
	vis[n][k] = tc;
	int ans = 0;
	if ( k-1 >= 0 ) ans = f(n,k-1);
	lli val = (C[k+m-1][m-1]*(lli)f(n-1,k))%M;
	ans = ans + (int)val;
	if ( ans >= M ) ans -= M;
	dp[n][k] = ans;
	return ans;
}

int main()
{
	int t,n;
	pre();
	fi(&t);
	for ( tc = 1; tc <= t; tc++ ) {
		fi(&n), fi(&m);
		int ans = f(n,m);
		printf("%d\n", ans);
	}
	return 0;
}
