#include<iostream>
#include<cstdio>
#define ll long long
#define m 1000000000
using namespace std;
ll ns[2010][2010];
ll a[2010][2010];
int main()
{
	ll T,i,M,N,j,ans,s;
	for(i=0;i<2005;i++)
	{
		for(j=0;j<2005;j++)
		{
			if(j==0)
				ns[i][j]=0;
			else if(i==0)
				ns[i][j]=1;
			else
				ns[i][j]=(ns[i-1][j] + ns[i][j-1])%m;
		}
	}
	//cout<<ns[5][5];
	scanf("%lld",&T);
	while(T--)
	{
		scanf("%lld %lld",&N,&M);
		for(i=1;i<=N;i++)
		{
			for(j=0;j<=M;j++)
				a[i][j]=ns[j][M];
		}
		for(i=2;i<=N;i++)
		{
			s=0;
			for(j=0;j<=M;j++)
			{
				s=(s + a[i-1][j])%m;
				a[i][j]=(a[i][j] * s)%m;
			}
		}
		ans=0;
		for(j=0;j<=M;j++)
			ans=(ans + a[N][j])%m;
		cout<<ans<<endl;
	}
}
