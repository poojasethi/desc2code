
#include<stdio.h>

 

long long a[2001][2001],b[2001][2001],c[2001][2001];

 

int main()

{

    int i,j,t,n,m;

    int x,y,z;

    a[0][0]=1;

    for(i=1;i<2001;i++)

    {

        a[1][i]=1;

        a[i][0]=1;

    }

    for(i=2;i<2001;i++)

        for(j=1;j<2001;j++)

           a[i][j]=(a[i-1][j]+a[i][j-1])%1000000000;

    scanf("%d",&t);

    while(t--)

    {

        scanf("%d %d",&n,&m);

        if(n==1)

        {

            printf("%d\n",a[m+1][m]);

            continue;

        }
        
        for(i=m;i>=0;i--)

        {

            b[m][i]=0;

            for(j=i;j<=m;j++)

                b[m][i]=(b[m][i] + a[m][j])%1000000000;

        }

        
        for(i=1;i<n;i++)

        {

            for(j=m;j>=0;j--){

            	y = a[m][j]*b[m][j]%1000000000;

            	x = y + c[m][j+1];

                c[m][j] = x%1000000000;

            	z++;

                b[m][j]=x%1000000000;
			}

        }

        printf("%d\n",(c[m][0])%1000000000);

    }

    return 0;

}
