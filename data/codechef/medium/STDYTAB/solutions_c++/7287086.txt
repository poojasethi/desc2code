#include<iostream>
using namespace std;
#define LL long long int
#define MOD 1000000000
int n,s,m,comb[4444][4444],dp[2222][2222];

void setcomb()
{
    for(int i=0;i<=4000;i++)
        comb[i][0]=1 ;
    for(int i=1;i<=4000;i++)
        for(int j=1;j<=i;j++)
            comb[i][j]=(comb[i-1][j-1]+comb[i-1][j])%MOD ;
}
void solve()
{
    for(int i=0;i<=n;i++)
        for(int j=0;j<=m;j++)
            dp[i][j] = 0 ;
    dp[0][0]=1 ;
    for(int i=1;i<=n;i++)
    {
        int s=0 ;
        for(int j=0;j<=m;j++)
        {
            s+=dp[i-1][j] ;
            if(s>=MOD)
                s-=MOD ;
            dp[i][j] = 1LL*s*comb[j+m-1][m-1]%MOD ;
        }
    }

    int ans=0 ;
    for(int i=0;i<=m;i++)
    {
        ans+=dp[n][i] ;
        if(ans >=MOD)
            ans-=MOD ;
    }
    cout<<ans<<endl;
}

int main()
{
    int t;
    cin>>t;
    setcomb();
    while(t--)
    {
        cin>>n>>m;
        solve();

    }
    return 0;
}
