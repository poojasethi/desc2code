#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
long long int mod=1000000000,x;
long long int a[2002][2002];
int p(long long int m,long long int b)
{
    if(b==0)
        return 1;
    if(b==1)
        return m%mod;
    x=p(m,b/2)%mod;
    return (((x*x)%mod)*p(m,b%2))%mod;
}
int main()
{
    long long int t,n,m,i,j,k,f[4002],b[2003],two[4002],five[4002],c;
    n=mod;
    c=n;
    for(i=2;i*i<=n;i++)
    {
        if(n%i==0)
        {
            c-=c/i;
            while(n%i==0)
                n/=i;
        }
    }
    if(n>1)
        c-=c/n;
    f[0]=1;two[0]=0;
    five[0]=0;
    for(i=1;i<=4000;i++)
    {
        two[i]=two[i-1];
        j=i;
        while(j%2==0)
        {
            j/=2;
            two[i]++;
        }
        five[i]=five[i-1];
        while(j%5==0)
        {
            j/=5;
            five[i]++;
        }
        f[i]=(f[i-1]*j)%mod;
    }
    slld(t);
    while(t--)
    {
        slld(n);
        slld(m);
        for(i=0;i<=m;i++)
        {
            j=p(2,two[m+i-1]-two[i]-two[m-1]);
            k=p(5,five[m+i-1]-five[i]-five[m-1]);
            b[i]=(((((j*k)%mod)*f[m+i-1])%mod)*(p((f[i]*f[m-1])%mod,c-1)))%mod;
        }
        a[n][m]=b[m];
        for(i=m-1;i>=0;i--)
            a[n][i]=(a[n][i+1]+b[i])%mod;
        for(i=n-1;i>=1;i--)
        {
            a[i][m]=(a[i+1][m]*b[m])%mod;
            for(j=m-1;j>=0;j--)
                a[i][j]=(a[i][j+1]+(a[i+1][j]*b[j])%mod)%mod;
        }
        printf("%lld\n",a[1][0]);
    }
    return 0;
}
