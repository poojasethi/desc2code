#include <cstdio>
#include <cstring>
 
using namespace std;
 
#define MOD 1000000000
 
int sum[2001][2001],dp[2][2001];
 
int main(){
	int T,N,M;
 
	scanf("%d",&T);
 
	sum[0][0] = 1;
 
	for(int i = 1;i <= 2000;++i){
		sum[i][0] = 1;
 
		for(int j = 1,acc = 0;j <= 2000;++j){
			sum[i][j] = sum[i - 1][j] + sum[i][j - 1];
 
			if(sum[i][j] >= MOD)
				sum[i][j] -= MOD;
		}
	}
 
	while(T--){
		scanf("%d %d",&N,&M);
 
		for(int i = 0;i <= M;++i)
			dp[0][i] = 1;
 
		for(int i = 1,r = 1;i <= N;++i,r ^= 1){
			for(int j = 0;j <= M;++j)
				dp[r][j] = (long long)sum[M][j] * dp[r ^ 1][j] % MOD;
 
			for(int j = 1;j <= M;++j){
				dp[r][j] += dp[r][j - 1];
				if(dp[r][j] >= MOD) dp[r][j] -= MOD;
			}
		}
 
		printf("%d\n",dp[N & 1][M]);
	}
 
	return 0;
}
