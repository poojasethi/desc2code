#include <iostream>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
#define max 2010
 
const int mod = (int)1000000000; 
int P[max][max], perm[max][max],cm[max][max], table[max][max];
 
int main() 
{
    int t;
    cin>>t; 
    while(t--) 
	{
        int n, m;
        cin>>n>>m;
        for(int i = 0; i <= m; i++ )
	    perm[m][i] = 1;
 
        for(int i = m - 1; i >= 0; i-- )
	    for(int j = 0; j <= m; j++ ) {	        
	        P[i][j] = perm[i + 1][j];	        
	        perm[i][j] = P[i][j];
	        if( j > 0 ) {
		    perm[i][j] += perm[i][j - 1];
		    perm[i][j] %= mod;
	        }
	    }        
        for(int i = 0; i <= m; i++ )
	    table[n][i] = 1;
 
        for(int i = n - 1; i >= 0; i-- )
	    for(int j = m; j >= 0; j-- ) {	        
	        cm[i][j] = ((long long) table[i + 1][j] * P[0][j]) % mod; 
	        table[i][j] = cm[i][j];
	        if( j < m ) {
		    table[i][j] += table[i][j + 1];
		    table[i][j] %= mod;
	        }
	    }
 
        int a = 0;
        for(int i = 0; i <= m; i++ )
	    a = (a + cm[0][i]) % mod;
 	cout<<a<<"\n";
    }
 
    return 0;
}
