#include<bits/stdc++.h>
int64_t mod=1000000000;
using namespace std;
int64_t ncr[4000][4000];
int64_t dp[2001][2001];
void fun()
{
    ncr[1][0]=ncr[1][1]=1;
    //cout<<ncr[1][0]<<","<<
    for(int64_t i=2;i<=3999;i++)
    {
        ncr[i][0]=ncr[i][i]=1;
        for(int64_t j=1;j<i;j++)
        {
            ncr[i][j]=(ncr[i-1][j-1]+ncr[i-1][j])%mod;
        }
    }
}
int main()
{
    int64_t t;
    cin>>t;
    fun();
    while(t--)
    {

        int64_t n,m;
        cin>>n>>m;
        dp[1][0]=1;
        for(int64_t j=1;j<=m;j++)
        {
            //cout<<ncr[j+m-1][j]<<","<<dp[1][j-1]<<endl;
            dp[1][j]=(ncr[j+m-1][j]+dp[1][j-1])%mod;
        }
        for(int64_t i=2;i<=n;i++)
        {
            dp[i][0]=1;
            for(int64_t j=1;j<=m;j++)
            {
                //cout<<"i="<<i<<" j="<<j<<endl;
                //cout<<ncr[j+m-1][j]<<","<<dp[i-1][j]<<","<<dp[i][j-1]<<endl;
                dp[i][j]=((ncr[j+m-1][j]*dp[i-1][j])%mod+dp[i][j-1])%mod;
            }
        }
        /*
        for(int64_t row=1;row<=n;row++)
        {
            for(int64_t sum=0;sum<=m;sum++)
            {
                cout<<dp[row][sum]<<" ";
            }
            cout<<endl;
        }
        */
        cout<<dp[n][m]<<endl;
    }
    return 0;
}
