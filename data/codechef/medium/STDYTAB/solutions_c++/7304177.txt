#include<bits/stdc++.h>
using namespace std;
long long int mod=1000000000;
long long int c[4500][2500];
long long int ways[2500][2500];
void nCr(long long int n,long long int m)
{
    long long int i,j;
    for(i=0;i<=n;i++)
    {
        for(j=0;j<=min(i,m);j++)
        {
            if(j==0||j==i)
            {
                c[i][j]=1;
            }
            else
            {
                c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
            }
        }
    }
}
int main()
{
    long long int t,n,m,i,j;
    scanf("%lld",&t);
    nCr(3999,1999);
    while(t--)
    {
        scanf("%lld %lld",&n,&m);
        for(i=0;i<=n;i++)
        {
            for(j=0;j<=m;j++)
            {
                if(i==0 && j==0)
                    ways[i][j]=1;
                else if(i==0 && j!=0)
                {
                    ways[i][j]=(ways[i][j-1]+c[j+m-1][m-1])%mod;
                }
                else if(j==0 && i!=0)
                    ways[i][j]=1;
                else
                    ways[i][j]=(ways[i][j-1]+(ways[i-1][j]*c[j+m-1][m-1])%mod)%mod;
            }
        }
        printf("%lld\n",ways[n-1][m]);
    }
    return 0;
}
