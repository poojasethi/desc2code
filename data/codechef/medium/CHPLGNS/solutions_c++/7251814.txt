#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
using namespace std;
#define ll long long int
#define max 100005
ll a[max];
ll b[max];

int sfunc(int front,int end,ll value)
{
	int mid=(front+end)/2;

	while(value!=b[mid])
	{
		if(value>b[mid]) front=mid+1;
		else if(value<b[mid]) end=mid-1;
		mid=(front+end)/2;
	}
	return mid;
}

int main()
{
    ll i,j,x,y,t,n,m,dmax,temp;
    scanf("%lld",&t);
    while(t--)
    {
        j=0;
        scanf("%lld",&n);
        while(n--)
        {
            dmax=0;
            scanf("%lld",&m);
            for(i=0;i<m;i++)
            {
              scanf("%lld %lld",&x,&y);
              temp=((ll)x*(ll)x+(ll)y*(ll)y);
              if(dmax<temp)dmax=temp;
            }
            a[j]=dmax;
            b[j]=a[j];
            j++;

        }
        n=j;
        for(i=0;i<n;i++) b[i]=a[i];
        sort(b,b+n);
        for(i=0;i<n;i++)
        {
             x=sfunc(0,n-1,a[i]);
             printf("%lld ",x);
        }
        printf("\n");
    }
    return 0;
}
