#include<stdio.h>
#include<algorithm>
using namespace std;
#define Max 100005

struct poly{
    int x;
    int p;
}ver[Max];

bool cmp(const poly &a,const poly &b)
{
    if(a.x < b.x)
        return true;
    return false;
}

int ans[Max];

int main()
{
    int cases;
    int n,m;
    scanf("%d",&cases);

    while(cases--)
    {
        scanf("%d",&n);
        for(int i=0; i<n; i++)
        {
            int a,b,r = -1000000009;
            scanf("%d",&m);
            for(int j=0; j<m; j++)
            {
                scanf("%d %d",&a,&b);
                if(b > r)
                    r = b;
            }
            ver[i].x = r;
            ver[i].p = i;
        }
        sort(ver,ver+n,cmp);

        for(int i=0; i<n; i++)
            ans[ver[i].p] = i;

        for(int i=0; i<n; i++)
            printf("%d ",ans[i]);
        printf("\n");
    }
    return 0;
}
