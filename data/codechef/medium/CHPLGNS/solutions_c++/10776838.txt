#include <iostream>
#include <bits/stdc++.h>
typedef long long int ll;
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll n;
		cin>>n;
		vector<pair<int,int> >mx(n);
		vector<int>a(n);
		for(ll i=0;i<n;i++)
		{
			ll m;
			cin>>m;
			int M=INT_MAX;//minimum x value
			for(ll j=0;j<m;j++)
			{
				int x,y;
				cin>>x>>y;
				if(M>x) M= x;
			}
			mx[i] = make_pair(M,i);	
		}
		sort(mx.begin(),mx.end());
		for(int i=0;i<n;++i)
			a[mx[i].second] = n-i-1;
		for(int i=0;i<n;++i)
			cout<<a[i]<<" ";
		cout<<endl;	
	}
	return 0;
}
