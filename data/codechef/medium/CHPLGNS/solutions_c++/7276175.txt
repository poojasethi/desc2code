#include<cstdio>
#include<iostream>
#include<algorithm>
#include<vector>
#include<climits>
using namespace std;
bool Comp(pair<int,int> l,pair<int,int> r){
    if(l.first != r.first){
        return l.first < r.first;
    }
    else return l.second < r.second;
}
int main()
{
    //freopen("in.txt","r",stdin);
    int TC;
    scanf("%d",&TC);
    while(TC--){
        int M;
        scanf("%d",&M);
        vector<int> A(M,-1);
        vector<pair<int,int> > val;
        for(int i=0;i<M;i++){
            int mxval=INT_MIN;
            int mxid;
            int N;
            scanf("%d",&N);
            for(int j=0;j<N;j++){
                int xi,yi;
                scanf("%d %d",&xi,&yi);
                if(xi >mxval){
                    mxval=xi;
                    mxid=i;
                }
            }
            val.push_back(make_pair(mxval,mxid));
        }
        sort(val.begin(),val.end(),Comp);
        for(int i=0;i<val.size();i++){
            //printf("first = %d second = %d\n",val[i].first,val[i].second);
            A[val[i].second]=i;
        }
        for(int i=0;i<A.size();i++){
            if(i==A.size()-1) printf("%d\n",A[i]);
            else printf("%d ",A[i]);
        }
    }
}
