#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        vector<pair<int,int> >cordinates(n+1);
        vector<int> ans(n+1);
        for(int i=1;i<=n;i++)
        {
            int m;
            scanf("%d",&m);
            int x,y,tx,ty;
            tx=INT_MIN;
            while(m--)
            {
                scanf("%d%d",&x,&y);
                tx=max(tx,x);
            }
            cordinates[i].first=tx;
            cordinates[i].second=i;
        }
        sort(cordinates.begin()+1,cordinates.end());
        for(int i=1;i<=n;i++)
        {
            ans[cordinates[i].second]=i;
        }
        for(int i=1;i<=n;i++)
            printf("%d ",ans[i]-1);
        printf("\n");
    }
    return 0;
}
