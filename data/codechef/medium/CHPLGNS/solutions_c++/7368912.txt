#include<bits/stdc++.h>
using namespace std;

struct polygon{
    int x ;
    int index;
};

bool compare(polygon a,polygon b){
    return a.x<b.x;
}

int main(){
    int t;scanf("%d",&t);
        while(t--){
            int n;scanf("%d",&n);
            polygon A[n];
            int i,j;
            for(i=0;i<n;i++){
                int m;scanf("%d",&m);
                int temp=-1000000000;
                for(j=0;j<m;j++){
                    int x,y;scanf("%d %d",&x,&y);
                    temp=max(temp,x);
                }
                A[i].x=temp;
                A[i].index=i;
            }
            sort(A,A+n,compare);
            int ans[n];
            for(i=0;i<n;i++)
                ans[A[i].index]=i;
            for(i=0;i<n;i++)
                cout<<ans[i]<<" ";
            cout<<endl;
    }
}
