#include<bits/stdc++.h>
using namespace std;

int ans[100005];

int main()
{
  ios_base::sync_with_stdio(false);
  cin.tie(false);

  int t;
  cin>>t;

  while(t--)
  {
      int n;
      cin>>n;

      vector< pair<int,int> > v;

      for(int i=0;i<n;i++)
      {
          int m;
          cin>>m;
          int x,y,temp=1000000005;
          for(int j=0;j<m;j++)
          {
            cin>>x>>y;
            temp=min(x,temp);
          }
          v.push_back(make_pair(temp,i));
      }

      sort(v.begin(),v.end());

      for(int i = 0; i < n; i++)
      {
         ans[v[i].second] = n-i-1;
      }

      for(int i=0;i<n-1;i++)
      {
          cout<<ans[i]<<" ";
      }
      cout<<ans[n-1]<<"\n";

  }

return 0;
}
