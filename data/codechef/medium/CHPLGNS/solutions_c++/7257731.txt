#include <bits/stdc++.h>
using namespace std;
 
typedef long long int ll;
const ll mod=1e9+7;
 
#define fileinput freopen("input.txt","r",stdin)
#define fileoutput freopen("output.txt","w",stdout)
 
vector< pair<ll,ll> > v;
vector<ll> mx;
 
int main()
{
    //fileinput;
 
    ll t;
    scanf("%lld",&t);
 
    while(t--){
       ll n,i;
       scanf("%lld",&n);
       ll ans[100002];
       v.clear();
       for(i=0;i<n;i++){
            ll m,j,k,x,y;
            scanf("%lld",&m);
 
            mx.clear();
            for(j=0;j<m;j++){
                scanf("%lld %lld",&x,&y);
                mx.push_back(x);
            }
 
            ll mx_x=*max_element(mx.begin(),mx.end());
            v.push_back(make_pair(mx_x,i));
       }
 
       sort(v.begin(),v.end());
 
       for(i=0;i<v.size();i++){
            ll p=v[i].second;
            ans[p]=i;
       }
 
       for(i=0;i<n;i++){
         printf("%lld ",ans[i]);
       }
 
       printf("\n");
 
    }
 
    return 0;
}
