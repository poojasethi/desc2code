#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;

typedef long long ll;
#define MAXN 100005
ll b[MAXN];

bool compare(pair<ll,ll> a,pair<ll,ll> b)
{
    return (a.first>b.first);
}

int main()
{
    #ifndef ONLINE_JUDGE
        freopen("CHPLGNS_input.txt","r",stdin);
    #endif // ONLINE_JUDGE

    ll t,n,m,x,y,maxy;
    vector< pair<ll,ll> > v;

    scanf("%lld",&t);
    while(t--)
    {
        scanf("%lld",&n);

        for(ll j=0;j<n;j++)
        {
            scanf("%lld",&m);
            maxy=-1000000000;
            for(ll i=0;i<m;i++)
            {
                scanf("%lld %lld",&x,&y);
                if(y>maxy)
                    maxy=y;
            }
            v.push_back(make_pair(maxy,j));
        }

        sort(v.begin(),v.end(),compare);

        /*for(vector<pair<ll,ll> >::iterator it=v.begin();it!=v.end();it++)
            printf("%lld,%lld ",it->first,it->second);
        printf("\n");*/

        ll j=1;
        for(vector<pair<ll,ll> >::iterator it=v.begin();it!=v.end();it++)
        {
            it->first= n - j;
            b[it->second]=it->first;
            j++;
        }
        for(ll i=0;i<n;i++)
            printf("%lld ",b[i]);
        printf("\n");
        v.clear();
    }
}
