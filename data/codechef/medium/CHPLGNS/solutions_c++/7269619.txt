#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstdlib>
using namespace std;
int main()
{
	int t,n,m;
    int x,y;
    int max_array[100005], new_max_array[100005];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);

		for(int i=0;i<n;i++)
		{
			scanf("%d",&m);
			scanf("%d%d",&x,&y);
			int max =x;

			for(int j=1;j<m;j++)
			{
				scanf("%d%d",&x,&y);

				if(x>max)
					max=x;

			}
			max_array[i]=max;
			new_max_array[i]=max;

            //printf("%d",max);
		}
		sort(new_max_array,new_max_array+n);
		for(int l=0;l<n;l++)
		{
			for(int k=0;k<n;k++)
			{
				if(max_array[l]==new_max_array[k])
                {
					printf("%d ",k);
					break;
				}
			}
		}
		printf("\n");
	}
		return 0;

	}
