#include <bits/stdc++.h>
using namespace std;
struct a{
	int idx;
	int maxi;
} arr[100005];

bool myfunction(a lhs,a rhs)
{
	return lhs.maxi<rhs.maxi;
}	
int main() {
	int t,n,v,j;
	long long int maxx,x,y,max[100005],ans[100005];
	cin>>t;
	while(t--)
	{
		for(int i=0;i<100005;i++)
		ans[i]=-1;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>v;
			//j=0;
			maxx=-1000000001;
			for(int j=0;j<v;j++)
			{
				cin>>x>>y;
				if(x>maxx)
				maxx=x;
				
			}
			arr[i].idx=i;
			arr[i].maxi=maxx;
		}
		sort(arr,arr+n,myfunction);
		for(int i=0;i<n;i++)
		{
			ans[arr[i].idx]=i;
		}
		
		for(int i=0;i<n;i++)
		{
			cout<<ans[i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
