inline void fs(int &x)
{
	register int c = gc();
	x = 0;
	int neg = 0;
	for(;((c<48 || c>57) && c != '-');c = gc());//c<'0'||c>'9'
	if(c=='-') {neg=1;c=gc();}
	for(;c>=48 && c<=57;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
	if(neg) x=-x;
}

bool comp(pair<int,int> a, pair<int,int> b)
{
	return a.first < b.first;
}

int main(int argc, char const *argv[])
{	
	int t;

	while(t--)
	{
		int temp,m,i,j,n,x,y,fin[100001];
		cin>>n;
		pair<int,int> ar[100001];
		for (i = 0; i<n ; ++i)
		{
			cin>>m;
			temp=INT_MAX;
			for (j = 1; j<=m ; ++j)
			{
				cin>>x>>y;
				temp=min(x,temp);
			}
			ar[i]=make_pair(temp,i+1);
		}
		sort(ar,ar+n,comp);
		int z=n-1;
		for( i = 0; i < n; ++i)
		{
			fin[ar[i].second]=z;
			z--;
		}
		for (i = 1; i<=n ; ++i)
		{
			cout<<fin[i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
