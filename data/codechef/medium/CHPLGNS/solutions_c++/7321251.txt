#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    cin >> t;
    vector< pair<int,int> > v;
    while (t--)
    {
        int n;
        cin >> n;
        for (int i = 0; i < n; ++i)
        {
            int m, x, y;
            cin>> m >> x >> y;
            int mx = x;
            m--;
            while (m--)
            {
                cin >> x >> y;
                if (x < mx) mx = x;
            }
            v.push_back(make_pair(mx, i));
        }
        vector<int>ans(n);
        sort(v.begin(), v.end());
        for(int i = 0; i < n; ++i) ans[v[i].second] = n - i - 1;
        for(int i = 0; i < n; ++i)
        {
            cout << ans[i];
            if (i < n - 1) cout << " ";
        }
        cout << endl;
        v.clear();
    }
}
