#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int main()
{
    int t,n,m,x,y,i,j,k,max1,a[100005],b[100005],l,u,m1;
    sd(t);
    while(t--)
    {
        sd(n);
        for(i=0;i<n;i++)
        {
            sd(m);
            max1=-1000000000;
            while(m--)
            {
                sd(x);
                sd(y);
                if(x>max1)
                    max1=x;
            }
            a[i]=max1;
            b[i]=max1;
        }
        sort(a,a+n);
        for(i=0;i<n;i++)
        {
            l=0;
            u=n-1;
            while(l<=u)
            {
                m1=(l+u)/2;
                if(a[m1]==b[i])
                {
                    j=m1;
                    break;
                }
                else if(a[m1]>b[i])
                    u=m1-1;
                else
                    l=m1+1;
            }
            printf("%d ",j);
        }
        printf("\n");
    }
    return 0;
}
