inline long int fastread()
{
long int i=0;
char c=getchar();
while(c<'0')
c=getchar();
while(c>='0'&&c<='9')
{
i=(i<<3)+(i<<1)+c-'0';
c=getchar();
}
return i;
}
int main()
{
	long long int t,n,x,a,b,max,j;
	int i;
	
	t=fastread();
	while(t--)
	{
		vector < pair<long long int,long long int> > s;
		n=fastread();
		for(i=0;i<n;i++)
		{
			max=-100000000;
			x=fastread();
			for(j=0;j<x;j++)
			{
				a=fastread();
				b=fastread();
				if(a>max)
				 max=a;
			}
			s.push_back(make_pair(max,i));
		}
		sort(s.begin(),s.end());
		long long int b[n];
		for(i=0;i<n;i++)
		{
			b[s[i].second]=i;
		}
		for(i=0;i<n;i++)
		{
		cout<<b[i]<<" ";
		}
		cout<<" ";
	}
	return 0;
}
