#include<bits/stdc++.h>

#define for(i,N)  for(int i=0;i<N;++i)

using namespace std;

int main()
{
 int t;
 cin>>t;
 while(t--)
 {
	int n;
	string str,best;
	cin>>n;
	cin>>str;
	best = str;
	int len=str.length();
    for(i,len)
	for(j,len)
	{
	  string nextstr = str;
	  char temp = str[i];
	  nextstr.erase(i,1);
	  nextstr.insert(j,1,temp);
	  if(nextstr < best)
		{
			best=nextstr;
		}
	}
	  cout<<best<<"\n";
 } 
 return 0;
}
