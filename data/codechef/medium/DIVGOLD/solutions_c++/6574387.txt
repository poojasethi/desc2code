#include<iostream>
#include<string>
using namespace std;

void inst(string s,string& s1,int x,int y){
	string sx=s;
	if(x<y){
		char c=sx[x];
		int i;
		for(i=x;(i<y) && (i+1<sx.length());i++){
			sx[i]=sx[i+1];
		}
		sx[i]=c;
	}
	else if(y<x){
		char c=sx[y],ii;
		sx[y]=sx[x];
		for(int i=y+1;i<=x;i++){
			ii=sx[i];
			sx[i]=c;
			c=ii;
		}
	}
	if(sx<s1) s1=sx;
}

int main(){
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		string s,s1;
		cin>>s;
		s1=s;
		for(int i=0;i<n;i++){
			for(int j=0;j<=n;j++){
				inst(s,s1,i,j);
			}
		}
		cout<<s1<<endl;
	}
	
}
