#include<cstdio>
#include<iostream>
#include<cstring>

using namespace std;

char str[100];
char temp[100];
char ans[100];

void replace(int ith , int with)
{
    if(ith==with)return;
    if(ith>with)
    {
        char tp = temp[ith];
        for(int i = ith;i>with;i--)
        {
            temp[i] = temp[i-1];
            //cout<<"*** i"<<i<<endl;
        }
        temp[with] = tp;
    }
    else
    {
        char tp = temp[ith];
        for(int i = ith;i<with;i++)
        {
            temp[i] = temp[i+1];
            //cout<<i<<" ** "<<temp[i]<<endl;
        }
        temp[with] = tp;
    }
    return;
}

int main()
{
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
    #endif
    ios_base::sync_with_stdio(false);

    int t;
    cin>>t;

    while(t--)
    {
        int n;
        cin>>n;

        cin>>str;
        ans[n]='\0';
        temp[n]='\0';
        for(int i=0;i<n;i++)
        {
            ans[i]= str[i];
        }
        for(int i = 0;i<n;i++)
        {
            for(int j = 0;j<n; j++)
            {
                for(int k = 0;k<n;k++)
                {
                    temp[k] = str[k];
                }
                replace(i,j);
                //cout<<i<<" "<<j<<" "<<temp<<endl;
                if(strcmp(ans,temp)>0)
                {
                    for(int k = 0;k<n;k++)
                    {
                        ans[k] = temp[k];
                    }
                }
            }

        }
        cout<<ans<<endl;
    }
}
