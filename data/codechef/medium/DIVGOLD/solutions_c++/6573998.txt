#include <bits/stdc++.h>
using namespace std;

int main() {
    int tc, n, i, j;
    string s, ns, sm;
    string ch;

    cin >> tc;

    while (tc--) {
        cin >> n >> s;
        sm = s;

        for (i = 0; i < n; i++) {
            ch = s.substr(i, 1);
            for (j = 0; j < n; j++) {
                ns = s.substr(0, i) + s.substr(i + 1);;
                ns.insert(j, ch);
                if (ns < sm) {
                    sm = ns;
                }
            }
        }
        cout << sm << endl;
    }

    return 0;
}
