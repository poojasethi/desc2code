# include "stdio.h"
# include "string.h"

char s[55], a[55], b[55];
int n;

void find_lex_small()
{
	int i, j, k, m;
	strcpy(a, s);

	for(i = 0; i < n; i++)
	{
		for(j = i+1; j < n; j++)
		{
			if(s[i+1] <= s[i])
			{
				m = 0;

				for(k = 0; k < i; k++)
					b[m++] = s[k];

				for(k = i+1; k <= j; k++)
					b[m++] = s[k];
				b[m++] = s[i];

				for(k = j+1; k < n; k++)
					b[m++] = s[k];
				b[m] = 0;

				if(strcmp(b, a) < 0)
					strcpy(a, b);
			}

			if( s[j] <= s[i])
			{
				m = 0;

				for(k = 0; k < i; k++)
					b[m++] = s[k];
				b[m++] = s[j];
				for(k = i; k < j; k++)
					b[m++] = s[k];
				for(k = j+1; k < n; k++)
					b[m++] = s[k];
				b[m] = 0;

				if(strcmp(b, a) < 0)
					strcpy(a, b);
			}
		}
	}
	printf("%s\n", a);
}

int main()
{
	int t;
	scanf("%d", &t);

	while(t--)
	{
		scanf("%d", &n);
		scanf("%s", s);

		find_lex_small();
	}
	return 0;
}
