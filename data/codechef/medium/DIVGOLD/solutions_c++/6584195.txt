#include<bits/stdc++.h>
#define readInt(a) scanf("%d\n",&a);
using namespace std;
string s,ans,temp;
char ch;
int main()
{
    int t,n,i,j,k;
    readInt(t);
    while(t--)
    {
        readInt(n);
        cin>>s;
        ans=s;
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                temp=s;
                if(i!=j)
                {
                    ch=temp[i];
                    temp.erase(temp.begin()+i);
                    temp.insert(temp.begin()+j,ch);
                    if(temp.compare(ans)<0)
                        ans=temp;
                }
            }
        }
        cout<<ans<<endl;
    }
}
