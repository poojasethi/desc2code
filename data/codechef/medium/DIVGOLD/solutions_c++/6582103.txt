# include <bits/stdc++.h>
using namespace std;
int main () {
	int t;
	scanf ("%d", &t);
	while (t--) {
		int N, i, j;
		scanf ("%d", &N);
		string s, temp, smin;
		char ch;
		cin >> s;
		smin = s;
		for (i = 0; i < N; i++) {
			for (j = 0; j < N; j++) {
				temp = s;
				ch = s[j];
				temp = temp.erase(j, 1);
				temp = temp.insert(i, 1, ch);
				if (temp < smin) {
					smin = temp;
				}
			}
		}
		cout << smin <<endl;
	}
}
