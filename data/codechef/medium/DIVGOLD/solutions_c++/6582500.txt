int main(){
   int t,n;
   string s,ans,tmp;
   cin>>t;
   while(t--){
      cin>>n>>s;
      ans=s;
      for(int i=0;i<n;i++){
         for(int j=0;j<n;j++){
            tmp=s;
            tmp.erase(i,1);
            tmp.insert(j,1,s[i]);
            //cout<<tmp<<endl;
            ans=min(ans,tmp);
         }
      }
      cout<<ans<<endl;
   }
   return 0;
}
