#include<bits/stdc++.h>
using namespace std;

int main(){
    int t;
    cin >> t;
    while(t--){
        int N;
        string s, ans;
        cin >> N;
        cin >> s;
        ans = s;
        for(int i = 0;i < N;++i){
            string news = s;
            char ch = news[i];
            news.erase(news.begin() + i);
            for(int j = 0;j < N;++j){
                if(news[j] > ch){
                    news.insert(news.begin() + j, ch);
                    break;
                }
            }
            if(news.size() < N)
                news.insert(news.end(), ch);
            ans = min(ans, news);
            //cout << i << " " << news << "\n";
        }
        cout << ans << "\n";
    }


    return 0;
}
