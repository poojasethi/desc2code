    # include <stdio.h>
    # include <string.h>
     
    char f[100];
     
    void traverse( char a[] , int k, int n){
    char b[100], c[100]; int r =0;
    for( int i = 0; i < n; i++){
    if( i == k)continue;
    b[r++] = a[i];
    }
    for( int i = 0 ; i < n; i++){
    r = 0;
    for ( int j = 0; j < i ; j++){
    c[j] = b[r++];
    }
    c[i] = a[k];
    for ( int j = i + 1; j < n ; j++){
    c[j] = b[r++];
    }
    c[n] ='\0';
    if(strcmp(f,c) > 0){
    strcpy(f,c);
    }
    }
    }
     
    int main(){
    int t;
    scanf("%d",&t);
    while(t--){
    char s[100]; int n;
    scanf("%d%s",&n,s);
    strcpy(f,s);
    for( int i = 0; i < n; i++){
    traverse(s, i, n);
    }
    printf("%s\n",f);
    }
    return 0;
    }
