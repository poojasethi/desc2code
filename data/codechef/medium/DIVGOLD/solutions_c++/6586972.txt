#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

string str,ans,temp;
int len;

bool comp(){
    for(int i=0;i<len;i++){
        if(temp.at(i)<ans.at(i))
            return true;
        else if(temp.at(i)>ans.at(i))
            return false;
    }
    return false;
}

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&len);
        cin>>str;
        ans=str;
        temp="";
        for(int i=0;i<len;i++){
            for(int j=0;j<i;j++){
                temp+=str.substr(0,j)+str.at(i)+str.substr(j,i-j)+str.substr(i+1,len-i-1);
                if(comp())
                    ans=temp;
                temp="";
            }
            for(int j=i+1;j<len;j++){
                temp+=str.substr(0,i)+str.substr(i+1,j-i)+str.at(i)+str.substr(j+1,len-j-1);
                if(comp())
                    ans=temp;
                temp="";
            }
        }
        cout<<ans<<endl;
    }
}
