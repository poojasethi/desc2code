#include<stdio.h>
#include<iostream>
#include<string.h>
void swp(char *a,int v,int vv);
void bb(char a[],int i,int j)
{
	int v;
	if(i<j)
	{
		for(v=i;v<j;v++)
		{
			swp(a,v,v+1);
		}
	}
	else
	{
		for(v=i;v>j;v--)
		{
			swp(a,v,v-1);
		}
	}
}
void swp(char *a,int v,int vv)
{
	char sss;
	sss=a[v];
	a[v]=a[vv];
	a[vv]=sss;
}
