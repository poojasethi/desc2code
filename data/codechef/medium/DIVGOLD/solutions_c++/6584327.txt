#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<stdlib.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		string a,b,c;
		int n;
		scanf("%d",&n);
		cin>>a;
		c=a;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				char tmp=a[i];
				string b=a;
				b.erase(i,1);
				b.insert(j,1,tmp);
				if(b<c)
				c=b;
			}
		}
		cout<<c<<"\n";
		
	}
}
