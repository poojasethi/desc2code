# include <bits/stdc++.h>
using namespace std;
int main () {
	int t;
	scanf ("%d", &t);
	while (t--) {
		int N, i, j;
		scanf ("%d", &N);
		string s, temp, smin;
		char s1;
		cin >> s;
		smin = s;
		for (i = 0; i < N; i++) {
			for (j = 0; j < N; j++) {
				temp = s;
				s1 = s[j];
				//cout << "Before inserting : "<< temp <<endl; 
				temp = temp.erase(j, 1);
				//cout << "Before erasing : "<< temp <<endl;
				temp = temp.insert(i, 1, s1);
				//cout << "After inserting : "<< temp <<endl;
				if (temp < smin) {
					smin = temp;
				}
			}
		}
		cout << smin <<endl;
	}
}
