#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>

using namespace std ;

int main()
{
  int T ; 
  scanf( "%d", &T ) ;
  int N ;
  char A[51] ;
  char C[51] ;
  char B[51] ;
  
  for( int t = 1 ; t <= T ; t++ )
    {
      scanf( "%d", &N ) ;
      scanf( "%s", A ) ;
      strcpy( C, A ) ;     //C stores the lexicographically min_string
      
      for( int i = 0 ; i < N ; i++ )
	{
	  char temp = A[i] ;
	  for( int j = 0 ; j < N ; j++ )
	    {
	      strcpy( B, A ) ;//B- current string obtained by permutation
	      if( j > i )
		{
		  for( int k = i ; k < j ; k++ )
		    {
		      B[k] = B[k+1] ;
		    }
		  B[j] = temp ;
		}
	      else if( j < i )
		{
		  for( int k = i ; k > j ; k-- )
		    {
		      B[k] = B[k-1] ;
		    }
		  B[j] = temp ;
		}
	      if( strcmp( B, C ) < 0 )
		{
		  strcpy( C, B ) ;
		}
	    }
	}
      printf( "%s\n", C ) ;

    }
}
