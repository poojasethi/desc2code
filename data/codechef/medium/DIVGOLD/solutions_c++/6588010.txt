#include<stdio.h>
#include<string.h>

int main()
{
	int t;
	scanf(" %d",&t);

	for (int l = 0; l < t; l++)
	{
		char ar[100],cr[100],min[100];
		int len;
		scanf(" %d",&len);
		scanf(" %s",&ar);
		//len=strlen(ar);
		strcpy(min,ar);
		for (int j = 0; j < len; j++)
		{
		strcpy(cr,ar);
        char ch=ar[j];
		int l=j;
		while (l>0)
		{
			ar[l]=ar[l-1];
			l--;
		}
		ar[0]=ch;
		if(strcmp(ar,min)<0)
			strcpy(min,ar);
		//printf("%s\n",ar);
		for (int i = 1; i < len; i++)
		{
		char ch=ar[i]; 
		ar[i]=ar[i-1];
		ar[i-1]=ch;
		if(strcmp(ar,min)<0)
			strcpy(min,ar);
		//printf("%s\n",ar);
		}

		strcpy(ar,cr);
		}
		printf("%s\n",min);
	}
	return 0;
}
