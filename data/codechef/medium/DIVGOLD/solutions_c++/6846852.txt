void move_left() {

	int anspos=0;
	int pos=-1;
	FOR(i,0,n) {

		if(s[i]==s1[i]) {
			ans1[anspos++] = s[i];
		}
		else {
			pos=i;
			break;
		}
	}

	char minch = s[n-1];
	int posmin = n-1;	

	FORR(i,n-2,pos)
		if(s[i]<minch) {
			minch = s[i];
			posmin = i;
		}

	ans1[anspos++] = minch;

	FOR(i,pos,n) {
		if(i!=posmin)
			ans1[anspos++] = s[i];
	}

	ans1[anspos]='\0';
}


void move_right() {

	
	int anspos1=0;	
	int pos=-1;

	FOR(i,0,n) {

		if(s[i]==s1[i]) {
			ans2[anspos1++] = s[i];
		}
		else {
			pos=i;
			break;
		}
	}

	int p1=pos;
	FOR(i,pos+1,n)
		if(s[pos]>=s[i])
			p1=i;
		else 
			break;
		

	FOR(i,pos,n)
		if(i==p1) {
			ans2[anspos1++]=s[i];
			ans2[anspos1++]=s[pos];
		}
		else if(i==pos)
			continue;
		else
			ans2[anspos1++]=s[i];
	

	ans2[anspos1]='\0';

}
