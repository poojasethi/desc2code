#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int T,n;
string S;
int main()
{
	scanf("%d",&T);
	while(T--)
	{
		cin>>n>>S;
		string res=S,cal=S;
		for(int i=0;i<n;i++)
		{
			cal=S;
			int j=i;
			while(j>0)
			{
				swap(cal[j],cal[j-1]);
				if(res>cal)res=cal;
				j--;
			}
			cal=S;
			j=i;
			while(j<n-1)
			{
				swap(cal[j],cal[j+1]);
				if(res>cal)res=cal;
				j++;
			}
		}
		cout<<res<<"\n";
	}
	return 0;
}
