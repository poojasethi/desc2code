#include<bits/stdc++.h>
using namespace std ;
int main()
{
  int t ;
  cin >> t ;
  string s ;
  int n ;

  while( t-- )
  {
    cin >> n ;
    cin >> s ;
    string news , temps ;
    temps = s ;
    for ( int i = 0 ; i < n ; i++)
    {

        for ( int j = 0 ; j < n ; j++)
        {
             news = s ;
            news.erase( i , 1 ) ;
            news.insert( j , 1 , s[i]) ;
            temps = min ( temps , news  ) ;
        }
    }
cout << temps ;
cout << endl ;
  }

  return 0 ;
}
