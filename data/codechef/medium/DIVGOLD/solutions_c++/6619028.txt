#include<cstdio>
#include<cstring>
int main()
{
int n,t,i,j,k;
char s[53],s1[53],ans[53],tmp,tmp1;
scanf("%d",&t);
while(t--)
	{
	scanf("%d",&n);
	scanf("%s",s1);
	for(i=0;i<n;i++)
		for(j=0;j<n;j++)
			{
			strcpy(s,s1);
			if(i<j)
				{
				tmp=s[i];
				for(k=i;k<j;k++)
					s[k]=s[k+1];
				s[j]=tmp;
				}
			else if(j<i)
				{
				tmp=s[i];
				for(k=i;k>j;k--)
					s[k]=s[k-1];
				s[j]=tmp;
				}
			if(i==0 && j==0)
				strcpy(ans,s);
			else
				{
				k=0;
				while(k<n && s[k]==ans[k])
					k++;
				if(k!=n && s[k]<ans[k])
					{
					strcpy(ans,s);
					//printf("%d %d*--------**\n",i,j);
					}	
				}	
				//printf("%d %d*****\n",i,j);
			}
	printf("%s\n",ans);
	}
return 0;
}
