#include<bits/stdc++.h>
using namespace std;

string ch;
set< vector< char > > s;

vector<char> temp,temp1;
char ch1;
int t,n;

void copyVecFast(const vector<char>& original)
{
  temp1.reserve(original.size());
  copy(original.begin(),original.end(),back_inserter(temp1));
}

int main()
{
	cin >> t;
	while(t--)
	{
		s.clear();
		temp.clear();
		temp1.clear();
		cin >> n;
		cin >> ch;
        for(int i=0;i<n;i++)
        {
             temp.push_back(ch[i]);
        }

        for(int i=0;i<n;i++)
        {
        	temp1.clear();
            copyVecFast(temp);   	
        	temp1.erase(temp1.begin()+i);
        	ch1=ch[i];
        	for(int j=0;j<n;j++)
        	{
                temp1.insert(temp1.begin()+j,ch1);
                s.insert(temp1);
                temp1.erase(temp1.begin()+j);
        	}
        }

       vector<char> v = *s.begin();
       
       for(int i=0;i<v.size();i++) cout << v[i]; 

       	cout << endl;

	}
	return 0;
}
