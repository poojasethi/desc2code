#include <bits/stdc++.h>
using namespace std;

int main() {
	// your code goes here
	 int T;
	 int i,h,l,N,H;
	long long max;
	scanf("%d",&T);
	long long arr[1000000];
	while(T--){
		scanf("%d%d",&N,&H);
		memset(arr,0,sizeof(long long)*N);
		for(i=0;i<N;i++){
			scanf("%d %d",&l,&h);
			arr[l]++;
			arr[h+1]--;
		}
		for(i=1;i<N;i++){
			arr[i]+=arr[i-1];
		}
		for(i=1;i<N;i++){
			arr[i]+=arr[i-1];
		}
		max=arr[H-1];
		for(i=1;i<=N-H;i++){
			if(max<arr[i+H-1]-arr[i-1])max=arr[i+H-1]-arr[i-1];
		}
		max= ((long long)N)*(long long)H-max;
		printf("%lld\n",max);
	}
	return 0;
}
