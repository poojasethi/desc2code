#include<iostream>
#include<stdio.h>
#include<cstring>

using namespace std;

#define ll long long int

int main()
{
	ll n,t,i,h,j,a,b,minm,sum;

	scanf("%lld",&t);

	while(t--)
	{
		scanf("%lld %lld",&n,&h);
		// ll arr[n][n];
		ll res[n+1];

		// memset(arr,0,sizeof(arr));
		memset(res,0,sizeof(res));

		for(i=0;i<n;i++)
		{
			scanf("%lld %lld",&a,&b);

			res[a]++;
			res[b+1]--;
		}

		sum=res[0];
		for(i=1;i<n+1;i++)
		{
			sum+=res[i];
			res[i]=sum;
		}

		for(i=0;i<n;i++)
		{
			res[i]=n-res[i];
			// for(j=0;j<n;j++)
			// 	cout<<arr[i][j]<<" ";
			// cout<<" -- "<<res[i]<<"\n";
			// cout<<"\n";
		}

		for(i=1;i<n;i++)
			res[i]+=res[i-1];

		minm=res[h-1];

		for(i=1;i<=n-h;i++)
		{
			a=res[i+h-1]-res[i-1];
			if(a<minm)
				minm=a;
		}

		printf("%lld\n",minm);
	}

	return 0;
}
