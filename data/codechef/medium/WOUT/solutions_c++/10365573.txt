#include<iostream>
using namespace std;

int main()
{
	long long int t,n,h,i,a,b,j;
	long long int sum,max;
	cin>>t;
	while(t--)
	{
		sum = 0;
		cin>>n>>h;
		long long int arr[n+2];
		for(i=0;i<n+3;i++)
		{
			arr[i] = 0;
		}		
		for(i=0;i<n;i++)
		{
			cin>>a>>b;
			arr[a]++;
			arr[b+1]--;
		}
		for(i=1;i<n;i++)
		{
			arr[i]+=arr[i-1];
		}
		for(i=0;i<h;i++)
			sum+=arr[i];
		max = sum;
		b = h;
		a = 0;	
		while(b<n)
		{
			sum-=arr[a];
			sum+=arr[b];
			a++;
			b++;
			if(sum>max)
				max = sum;
		}
		cout<<(h*n)-max<<endl;
	}
	return 0;
}
