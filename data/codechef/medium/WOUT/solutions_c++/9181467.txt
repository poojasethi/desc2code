#include <iostream>
#include <string.h>
using namespace std;
int arr[1000001];
#define max(x,y) (x>y) ? (x):(y)
#define ll long long
int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		memset(arr, 0, sizeof arr);
		ll n,h,ii,hi;
		cin>>n>>h;
		for(int i=0;i<n;i++)
		{
			cin>>ii>>hi;
			arr[ii]++;
			arr[hi+1]--;
		}
		for(int i=1;i<n;i++)
			arr[i]+=arr[i-1];
		ll sum=0;
		ll mx=-1;
		for(int i=0;i<h-1;i++)
			sum+=arr[i];
		for(int i=h-1;i<n;i++)
		{
			sum=(ll)sum+arr[i];
			mx=max(mx,sum);
			sum=(ll)sum-arr[i-h+1];	
		}
		ll p=n*h;
		cout<<(p-mx)<<"\n";
	}
	return 0;
}
