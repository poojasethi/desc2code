#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,h,l,r,i,tot,sum1=0,sum2=0;	
		scanf("%lld%lld",&n,&h);
		
		long long int a[n+1];
		for(i=0;i<n+1;i++)
		{
			a[i]=0;
		}
		tot=n*h;
		for(i=0;i<n;i++)
		{
			scanf("%lld%lld",&l,&r);
			a[l]++;
			a[r+1]--;
		}
		for(i=1;i<n;i++)
		{
			a[i]=a[i]+a[i-1];
			
		}
		for(i=0;i<h;i++)
		{
			sum1+=a[i];
		}
		sum2=sum1;
		for(i=h;i<n;i++)
		{
			if(((a[i]+sum1)-a[i-h])>sum2)
			{
				sum2=(a[i]+sum1)-a[i-h];
				sum1=(a[i]+sum1)-a[i-h];
			}
			else
			{
				sum1=(a[i]+sum1)-a[i-h];
			}
		}
		
		printf("%lld\n",tot-sum2);
	}
	return 0;
}
