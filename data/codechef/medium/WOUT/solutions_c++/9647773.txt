#include<bits/stdc++.h>
using namespace std;
unsigned long long int a[1000003];
int main()
{
	int t;
     long long int i;
	scanf("%d",&t);

	while(t--)
	{
		long long int n,h,j;
		cin>>n>>h;
	     long long int l,r;
		unsigned long long int sum=0,max;
	
		for(i=0;i<n;i++)
		{
		a[i]=0;
		}
		for(i=0;i<n;i++)
		{
			cin>>l>>r;

           a[l]++;
           a[r+1]--;
		    

		}

                for(i=1;i<n;i++)
                a[i]=a[i]+a[i-1];
                sum=0;
        for(i=0;i<h;i++)
        sum=sum+a[i];
        max=sum;
      
		for(i=1;i<=(n-h);i++)
		{
			
	         sum=sum-a[i-1]+a[i+h-1];  
			if(sum>max)
				max=sum;
		}
		cout<<h*n-max<<"\n";
		
	}
}
