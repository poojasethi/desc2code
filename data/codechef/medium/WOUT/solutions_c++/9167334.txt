#include <bits/stdc++.h>
using namespace std;

long long int ans[1000001],chl[1000001],chh[1000001],r[1000001];

int main() {
	long long int i,n,l,h,H,t;
	cin>>t;
	while(t--)
	{
		
		cin>>n>>H;
		
		for(i=0;i<n;i++)
		 chh[i]=0,chl[i]=0,r[i]=0;
		
		for(i=0;i<n;i++)
		{
			cin>>l>>h;
			chl[l]++;
			chh[h+1]++;
		}
		
		long long int cur=n;
		
		for(i=0;i<n;i++)
		{
			cur+=chh[i]-chl[i];
			r[i]=cur;
		}
		
		long long int j,k=-1,temp=0,temp2=0;
		
		for(i=0;i<H;i++)
		 temp+=r[i];
		
		i--;
		
		for(j=0;i<n;j++)
		{
			ans[j]=temp-temp2;
			i++,k++;
			temp+=r[i],temp2+=r[k];
			
		}
		
		sort(ans,ans+j);
		cout<<ans[0]<<endl;
		
		
	}
	return 0;
}
