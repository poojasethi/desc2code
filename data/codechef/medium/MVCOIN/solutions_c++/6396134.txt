#include <stdio.h>

int n, i, ans, t, k;
int a[2222];
int main()
{ scanf("%d",&t);
  while (t--)
  {
    scanf("%d %d",&n,&k);
    for (i=1;i<=n;i++) scanf("%d",a+i);
    a[0] = ans = 0;
    for (i=n;i>=1;i--) 
            ans+=   (a[i]-a[i-1]-1)*  // units to move 
                    ((n-i)/k+1);
    printf("%d\n",ans);
  }
  return 0;
}
