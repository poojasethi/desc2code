#include <iostream>
#include <cstdio>
using namespace std;

int t, n, k;
int x[1001];

int main()
{
    scanf("%d", &t);
    for (int i=0; i<t; i++) {
        scanf("%d%d", &n, &k);
        x[0] = 0;
        for (int i=1; i<=n; i++) {
            scanf("%d", &x[i]);
        }
        int ans = 0;
        int sz = 1;
        for (int i=n; i>0; i--) {
            int d = x[i]-x[i-1]-1;
            ans += d*((sz-1)/k+1);
            sz++;
        }
        printf("%d\n", ans);
    }
    return 0;
}
