#include <iostream>
#include <stdio.h>
using namespace std;
inline int ceil(int a, int b)
{
	if(a%b==0)
	return a/b;
	else
	return (a/b) +1;
}
int main()
{
	int t,n,k,tmp,max, o_c,z_c,stps=0;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d %d",&n,&k);
		int arr[1001]={0};
		for(int i=0; i<n; i++)
		{
			scanf("%d", &tmp);
			arr[tmp]=1;
		}
		max=tmp;
		if(max==n)
		{
			printf("0\n");
			continue;
		}
		o_c=z_c=stps=0;
		max++;
		while(max>0)
		{
			z_c=0;
			while(arr[--max]==1)
			o_c++;
			max++;
			while(arr[--max]==0 && max>0)
			z_c++;
			if(max>0)
			max++;
			stps+=(z_c*ceil(o_c, k));
			//printf("%d %d %d\n",stps, z_c, o_c);
		}
		printf("%d\n", stps);
	}
	return 0;
}
