// MVCOIN
#include <cstdio>
using namespace std;

int main(){
  int t,n,k,i,groupsize,gap,start,secs;
  short initcoinposn[1002];

#ifndef ONLINE_JUDGE
  freopen("in.txt","r",stdin);
#endif
  initcoinposn[0]=0;
  scanf("%d",&t);
  while(t--){
    scanf("%d %d",&n,&k);
    for(i=1;i<=n;i++) scanf("%d",&initcoinposn[i]);
    secs=0;
    start=initcoinposn[n--];
    groupsize=1;
    while(start>1 && n>=0){
      if(initcoinposn[n]+1==start){
        start--;
        groupsize++;
      }
      else{
        gap=start-initcoinposn[n]-1;
        secs+=(gap*(((groupsize-1)/k)+1));
        start-=(gap+1);
        groupsize++;
      }
      n--;
    }
    printf("%d\n",secs);
  }
  return 0;
}
