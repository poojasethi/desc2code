typedef long long LL;
typedef unsigned long long ULL;

using namespace std;

LL n,k,ara[1005];

int main()
{
    LL i,j,x,t,T;
    sl(T);
    rep(t,T)
    {
        mem(ara,0);
        sll(n,k);
        ara[0] = 1;
        rep(i,n)
        {
            sl(x);
            ara[x] = 1;
        }
        LL last = 1000,tot = 0;
        while(1)
        {
            LL idx1,idx2,idx3;
            for(i=last; i>=0; i--)
            {
                if(ara[i])
                {
                    idx1 = i;
                    break;
                }
            }
            idx2 = 0;
            for(i=idx1-1; i>=0; i--)
            {
                if(ara[i] == 0)
                {
                    idx2 = i+1;
                    break;
                }
            }
            if(idx2 == 0)break;
            LL len = idx1 - idx2 + 1;
            LL val = len/k;
            if(len % k)val++;
            for(i=idx2-1; i>=0; i--)
            {
                if(ara[i])
                {
                    idx3 = i;
                    break;
                }
            }
            LL baki = idx2 - idx3 - 1;
            //printf("idx1 %d idx2 %d idx3 %d\n",idx1,idx2,idx3);
            tot += (baki * val);
            last = idx3 + len;
            for(i=idx3; i<=last; i++)ara[i] = 1;
            //for(i=0;i<=last;i++)printf(" %d",ara[i]);
            //NL;
        }
        pl(tot);
        NL;
    }
    return 0;
}
