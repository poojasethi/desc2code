#include<stdio.h>
int main()
{
    int t,c,d,n,k,i,inp,a[1002]={0},s,count;
    scanf("%d",&t);
    c=0;
    while(t--)
    {
        s=1;
        d=c+1;
        count=0;
        scanf("%d%d",&n,&k);
        
        for(i=1;i<=n;i++)
        {
            scanf("%d",&inp);
            a[inp]=d;
        }
        for(i=inp-1;i>0;i--)
        {
            if(a[i]==d)
            {
                s++;
            }
            else
            {
                count+=(s+k-1)/k;
            }
        }
        printf("%d\n",count);
    c+=1;}
    return 0;
}
