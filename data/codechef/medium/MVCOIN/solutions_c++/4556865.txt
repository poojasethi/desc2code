    int main()
    {
        int T;
        scanf("%d",&T);
        for (int i = 0; i < T; i++)
        {      
            int N, K;
            int moves = 0; 
            int a[1002]= {0};
            moves = 0;
            scanf("%d %d",&N,&K);
            int index;
            for(int j=0; j<N;j++ )
            {
                scanf("%d",&index);
                a[index-1]=1;
            }
            int end_itr = index-1;
            for (int itr = index-1; itr >= 0; itr--)
            {
                if (a[itr]==0)
                {
                    moves += (end_itr-itr+K-1)/K;
                    end_itr--;
                }
            }
            printf("%d\n",moves);
            
        }
        return 0;
    }
