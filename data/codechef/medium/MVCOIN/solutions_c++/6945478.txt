#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
int tc,n,k,x;
scanf("%d",&tc);
while(tc--)
{
    int a[1001]={0};
scanf("%d%d",&n,&k);
for(int i=0;i<n;i++)
{
scanf("%d",&x);
a[x]=1;
}
int sum=0,l=0;
for(int i=1000;i>0;i--)
{
if(a[i]==0)
{
sum+=(l-1+k)/k;
}
else
l++;
}
printf("%d\n",sum);
}
return 0;
}
