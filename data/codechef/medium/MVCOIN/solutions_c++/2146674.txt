#include<iostream>

using namespace std;

int main()
{
	int T,ans,l,i,n,k,num;
	int arr[1001];
	cin>>T;
	while(T--)
	{
		for(i=0;i<=1000;i++)
			arr[i]=0;
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>num;
			arr[num]=1;
		}
		l=0;
		ans=0;
		for(i=1000;i>0;i--)
		{
			if(!arr[i])
				ans+=(l+k-1)/k;
			else
				l++;
		}
		cout<<ans<<endl;
	}
	return 0;
}
