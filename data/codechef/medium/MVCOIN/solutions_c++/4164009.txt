#include<iostream>
#include<stdio.h>
using namespace std;

int scanint()
{
	register int c = getchar();
	int x=0;
	for(;(c<48 || c>57);c=getchar());
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	return x;
}

int main(){
	int t,n,k,i,gap,time=0,x[1001],together;
	t=scanint();
	while(t--){
		n=scanint();
		k=scanint();
		for(i=1;i<=n;i++){
			x[i]=scanint();
		}
		together=1;
		i=n-1;
		time=0;
		x[0]=0;
		while(i>=0){
			if(x[i+1]-1==x[i]){
				together++;
				i--;
				continue;
			}
			gap=x[i+1]-x[i]-1;
			if(together<=k){
				time+=gap;
				together++;
				i--;
			}else{
				time+=(((together-1)/k)+1)*gap;
				together++;
				i--;
			}
		}
		cout<<time<<"\n";
	}
}
