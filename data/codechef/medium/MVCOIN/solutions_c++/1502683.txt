#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;

int movingcoins(int n, int k, int loc, bool cell[]){
	int to_return = 0;
//	cout << "n " << n  << " k "<< k << " loc " <<  loc << endl;
	int start = loc-1, end = loc;
	int temp;
	while(start >= 1){
//		cout << "end, start, to_return " << end << " " << start << " " << to_return << endl;
		if (cell[start] == false) {
			temp = end - start;
			if(temp <= k){
				to_return++;
			}
			else{
				if(temp%k == 0) to_return = to_return + temp/k;
				else to_return = to_return + (temp/k) + 1;
			}
			end--;	
		}
		start--;
	}
	return to_return;	
}

int main(){
	int T, N, K, i, j, ans;
	int* location,tmp;
	bool cells[1001];
	for(i=0; i<1001; i++) cells[i] = false;
	cin >> T;
	for (i =0; i<T; i++){
//		cout << "test case :" << i << endl;
		cin >> N >> K;
		for(j=0; j<N; j++) {
			cin >> tmp;
//			cout << tmp << endl;
			cells[tmp] = true;
		}
//		for(j=0; j<1001; j++) { if(cells[j] == true) cout << j << " ";}
//		cout << endl;
		ans = movingcoins(N, K, tmp, cells);
		cout << ans << endl;
		for(j=0; j<1001; j++) cells[j] = false;
	}
	return 0;
}
