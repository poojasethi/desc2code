#include<cstdio>
#include<cstring>

int main()
{
	unsigned t;
	unsigned n,k; 
	unsigned a[1001];
	register int i;
	register unsigned cnt,len;

	scanf("%d",&t);

	while(t--)
	{
		scanf("%u%u",&n,&k);

		a[0] = 0;
		for(i = 1; i <= n; i++)scanf("%u",(a + i));

		i -= 2;
		cnt = 0;
		len = 1;

		while(true)
		{
			while( i >= 0 && (a[i] == (a[i + 1] - 1)) ) 
			{
				len++;
				i--;
			}
			if(i < 0)break;
			cnt += ( ( ((len - 1) / k) + 1 ) * (a[i+1] - a[i] - 1));
			len++;
			i--;
		}
		printf("%u\n",cnt);
	}
	return 0;
}
