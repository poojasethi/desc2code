#include<stdio.h>
int main()
{
	int t,n,k,i,a[1001];
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&k);
		a[0]=0;
		for(i=1;i<=n;i++)
		{
			scanf("%d",&a[i]);
		}
		int count=0;
		int sum=0;
		for(i=n;i>0;i--)
		{
			count++;
			if(count%k!=0)
			sum=sum+(count/k+1)*(a[i]-a[i-1]-1);
			else
			sum=sum+(count/k)*(a[i]-a[i-1]-1);
		}
		printf("%d\n",sum);
	}
	return 0;
}
