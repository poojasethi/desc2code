#include <iostream>
#include <stdio.h>
using namespace std;
inline int FAST_IO();
int main()
{
	int t,n,k,tmp,max, o_c,z_c,stps=0;
	scanf("%d", &t);
	while(t--)
	{
		n=FAST_IO();
		k=FAST_IO();
		int arr[1001]={0};
		for(int i=0; i<n; i++)
		{
			tmp=FAST_IO();
			arr[tmp]=1;
		}
		max=tmp;
		if(max==n)
		{
			printf("0\n");
			continue;
		}
		o_c=z_c=stps=0;
		max++;
		while(max>0)
		{
			z_c=0;
			while(arr[--max]==1)
			o_c++;
			max++;
			while(arr[--max]==0 && max>0)
			z_c++;
			if(max>0)
			max++;
			stps+=(z_c*(o_c%k==0 ? o_c/k : ((o_c/k) +1) ));
			//printf("%d %d %d\n",stps, z_c, o_c);
		}
		printf("%d\n", stps);
	}
	return 0;
}
inline int FAST_IO() 
{
	int xxxx;
	char ch;
	int Negativity=0;
	while (((ch=getchar_unlocked()) < 48 || ch > 57) && ch != '-');
	xxxx=0;
	if (ch == '-') Negativity=1;
	else xxxx = ch-48;
	while ((ch=getchar_unlocked()) >= 48 && ch <= 57) xxxx=xxxx*10+ch-48;
	if (Negativity) return -xxxx;
	else return xxxx;
}
