#define FOR(i,a,b) for(int i=a;i<b;i++)
#define FORL(i,a,b) for(int i=a;i<=b;i++)
#define FORI(i,a,b,inc) for(int i=a;i<b;i=i+inc)
#define FORD(i,a,b) for(int i=a;i>=b;i--)
#define LL long long

using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k;
        scanf("%d%d",&n,&k);
        int arr[n];
        FOR(i,0,n)
                  scanf("%d",&arr[i]);
        int l = 1;
        int ind = n-1;
        int ans = 0;
        while(ind > 0)
        {
            ans = ans + (arr[ind] - arr[ind-1] - 1)*ceil((double)l/(double)k);
            l++;    
            ind--;   
        }
        ans = ans + (arr[0] - 1)*ceil((double)l/(double)k);
        printf("%d\n",ans);
    }
    
    //system("pause");
    return 0;
}
