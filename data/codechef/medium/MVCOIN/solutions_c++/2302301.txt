#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
    int n, i, ans, t, k;
    int a[1001];


    scanf("%d",&t);
    while (t--)
    {
        scanf("%d %d",&n,&k);
        for (i=1;i<=n;i++)
        {
            scanf("%d",a+i);
        }
        std::sort(a+1,a+n+1);
        a[0] = ans = 0;
        for (i=n;i>=1;i--)
        {
            ans += (a[i]-a[i-1]-1)*((n-i)/k+1);
        }
        printf("%d\n",ans);
    }

    return 0;
    
}
