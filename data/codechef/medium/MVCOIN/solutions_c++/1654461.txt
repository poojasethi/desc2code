#include <cstdio>

#define N 1000

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
        int coins, jump;
        int coinPos[N+1]={};
        int ans=0;

        scanf("%d %d", &coins, &jump);
        for(int i=1; i<=coins; i++) scanf("%d", &coinPos[i]);

        for(int i=coins; i>=1; i--) ans += (coinPos[i] - coinPos[i-1] - 1) * ((coins-i)/jump + 1);

        printf("%d\n", ans);
    }
}
