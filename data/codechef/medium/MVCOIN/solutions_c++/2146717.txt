#include<iostream>
#include<cstdlib>

using namespace std;

int main()
{
	int T,ans,l,i,n,k,num;
	
	cin>>T;
	while(T--)
	{
		int *arr=(int *)calloc(1001,sizeof(int));
		cin>>n>>k;
		for(i=0;i<n;i++)
		{
			cin>>num;
			arr[num]=1;
		}
		l=0;
		ans=0;
		for(i=1000;i>0;i--)
		{
			if(!arr[i])
				ans+=(l+k-1)/k;
			else
				l++;
		}
		cout<<ans<<endl;
		free(arr);
	}
	return 0;
}
