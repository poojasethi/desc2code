#include <cstdio>
#include <algorithm>
using namespace std;

#define N 1000
bool has[N];
int main() {
	int tc;
	scanf("%d", &tc);
	for (; tc > 0; tc --) {
		fill(has, has + N, false);
		int n, delta;
		scanf("%d%d", &n, &delta);
		for (int i = 0; i < n; i ++) {
			int x; scanf("%d", &x);
			x --;
			has[x] = true;
		}
		int answer = 0;
		for (int r = N - 1; r >= n; r --) if (has[r]) {
			int l = r;
			while (has[l - 1]) l = l - 1;
			// [l, r] -> [l-1, r-1]
			answer += 1 + (r - l) / delta;
			has[l - 1] = true, has[r] = false;
		}
		printf("%d\n", answer);
	}
}
