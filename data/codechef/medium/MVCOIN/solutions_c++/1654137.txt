#include <stdio.h>
#include <algorithm>
int n, i, ans, t, k;
int a[2222];
main()
{ scanf("%d",&t);
  while (t--)
  {
    scanf("%d %d",&n,&k);
    for (i=1;i<=n;i++) scanf("%d",a+i);
    //std::sort(a+1,a+n+1);
    a[0] = ans = 0;
    for (i=n;i>=1;i--) 
            ans+=   (a[i]-a[i-1]-1)*  // units to move 
                    ((n-i)/k+1)       // cost of moving n-i coins by 1 unit
                    ;
    printf("%d\n",ans);
  }
  return 0;
}
