#include <iostream>
#include <cmath>
#include <string>
using namespace std;



int main(){
        long long int n,k;
        cin>>n>>k;
        long long int lst[n],lst2[n];
        long long max=0,min=0;
        for(long long int i=0;i<n;i++){
            cin>>lst[i];
            if(lst[i]>max)max=lst[i];
            if(lst[i]<min)min=lst[i];
        }
        if(k!=0){
            if(min<0){
                max=max-min;
                for(long long int i=0;i<n;i++){
                    lst[i]=lst[i]-min;
                }
            }
            for(long long int i=0;i<n;i++){
                lst2[i]=max-lst[i];
            }

        }
        long long int *p;
        if(k%2==0){
            p=lst;
        }
        else{
            p=lst2;
        }
        for(long long int j=0;j<n;j++){
            cout<<p[j]<<" ";
            }
            cout<<endl;

    return 0;
}
