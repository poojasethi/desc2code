#include<iostream>
#include<climits>
#define ll long long
using namespace std;
int main()
{
    ll n,k;ios::sync_with_stdio(false);
    cin>>n>>k;
    ll arr[n];
    ll max = LLONG_MIN;
    ll min = LLONG_MAX;
    for(int i=0;i<n;i++)
    {
        cin>>arr[i];
        //if(arr[i]<min)
          //  min = arr[i];
        if(arr[i]>max)
            max = arr[i];
    }
    if(k>=1)
    {
        ll newmax = LLONG_MIN;
        //ll newmin = LLONG_MAX;
        for(int i = 0;i<n;i++)
        {
            arr[i]=max-arr[i];
            if(arr[i]>newmax)
                newmax = arr[i];
        }
        int remain = k-1;
        if(remain%2==0)
        {
            for(int i=0;i<n;i++)
            {
                cout<<arr[i]<<' ';
            }
            cout<<endl;
        }
        else
        {
            for(int i=0;i<n;i++)
            {
                cout<<newmax-arr[i]<<' ';
            }
            cout<<endl;
        }
    }
    else
    {
        for(int i=0;i<n;i++)
        {
            cout<<arr[i]<<' ';
        }
        cout<<endl;
    }
}
