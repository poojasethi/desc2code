#include<bits/stdc++.h>
using namespace std;

int main()
{
long long n,k;	
cin>>n>>k;
long long a[100001];
for(int i=0;i<n;i++)
{ 
	cin>>a[i];
}
long long mn=a[0];
long long mx=a[0];
for(int i=0;i<n;i++)
{
	mn=min(mn,a[i]);
	mx=max(mx,a[i]);
}
  if(k==0)
  { 
 for(int i=0;i<n;i++)
  cout<<a[i]<<" ";
  }

  else if(k%2!=0) 
	{
	for(int i=0;i<n;i++)
	 cout<<mx-a[i]<<" ";
}
  else if(k%2==0)
  { 
   for(int i=0;i<n;i++) 
		cout<<a[i]-mn<<" ";
	}
cout<<endl;	
}
