#include<cstdio>
#include<algorithm>
#include<cmath>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main()
{
    int n;
    ll k;
    scanf("%d %lld",&n,&k);
    ll arr[n];
    ll Min=INF,Max=-INF;
    for(int i=0;i<n;i++){
        scanf("%lld",&arr[i]);
        Min=min(Min,arr[i]);
        Max=max(Max,arr[i]);
    }
    if(k%2){
        for(int i=0;i<n;i++){
         printf("%lld ",Max-arr[i]);   
        }
    }else if(k==0){
        for(int i=0;i<n;i++){
         printf("%lld ",arr[i]);   
        }
    }
    else{
        for(int i=0;i<n;i++){
            printf("%lld ",arr[i]-Min);   
        }
    }
    printf("\n");
    return 0;
}
