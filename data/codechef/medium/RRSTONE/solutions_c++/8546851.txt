#include <iostream>
#include <limits>
#include <cstring>
int main(int argc, const char * argv[]) {
    long long int n,k,a[100001];
    memset(a,0,sizeof(a));
    std::cin>>n>>k;
    std::cin>>a[0];
    long long int max=a[0],min=a[0];
    for(int i=1;i<n;i++){
        std::cin>>a[i];
        if(a[i]>max){
            max=a[i];
        }
        if(a[i]<min){
            min=a[i];
        }
    }
    if(k==0){
        for(int i=0;i<n-1;i++){
            std::cout<<a[i]<<" ";
        }
        std::cout<<a[n-1]<<"\n";
    }else if(k%2==1){
        for(int i=0;i<n-1;i++){
            std::cout<<max-a[i]<<" ";
        }
        std::cout<<max-a[n-1]<<"\n";
    }else{
        for(int i=0;i<n-1;i++){
            std::cout<<a[i]-min<<" ";
        }
        std::cout<<a[n-1]-min<<"\n";
    
    }
    
    return 0;
}
