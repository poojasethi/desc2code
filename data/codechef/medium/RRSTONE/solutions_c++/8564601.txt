#include<bits/stdc++.h>
using namespace std;
long long a[100010],a1[100010],a2[100010];
int main()
{
	long long n,k,maxi=INT_MIN,mini=INT_MAX;
	int i;
	scanf("%lld %lld",&n,&k);
	for(i=0;i<n;i++)
	{
		scanf("%lld",&a[i]);
		if(maxi<a[i])maxi=a[i];
		if(mini>a[i])mini=a[i];
	}
	//long long a1[100010],a2[100010];
	if(k==0)
	{
		for(i=0;i<n;i++)
		printf("%lld ",a[i]);
		printf("\n");return 0;
	}
	for(i=0;i<n;i++)
	a1[i]=maxi-a[i];
	for(i=0;i<n;i++)
	a2[i]=maxi-mini-a1[i];
	if(k%2)
	{
		for(i=0;i<n;i++)
		printf("%lld ",a1[i]);printf("\n");
	}
	else
	{
		for(i=0;i<n;i++)
		printf("%lld ",a2[i]);printf("\n");
	}
}
