#include<iostream>
using namespace std;
int main(){
ios::sync_with_stdio(false);
int n;
long long int k,maxm=0,minm=2000000007;
cin>>n>>k;
long long int a[n];
for(int i=0;i<n;i++){
    cin>>a[i];
    maxm=max(maxm,a[i]);
    minm=min(minm,a[i]);
}
if(k%2==0 && k!=0){
    for(int i=0;i<n;i++){
        a[i]=a[i]-minm;
    }
}
else if(k%2==1){
    for(int i=0;i<n;i++){
        a[i]=maxm-a[i];
    }
}
//OUTPUT
for(int i=0;i<n;i++){
    cout<<a[i]<<" ";
}
}
