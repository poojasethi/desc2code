#include<cstdio>
#include<iostream>
using namespace std;
int main()
{
    int n,k;
    long long int ar[100005];
    cin>>n>>k;
    long long int mn=1000000001,mx=-10000001;
    for(int i=0;i<n;i++)
    {
        scanf("%lld",&ar[i]);
        if(ar[i]>mx)
            mx=ar[i];
        else if(ar[i]<mn)
            mn=ar[i];
    }
    if(k==0)
    {
        for(int i=0;i<n;i++)
        printf("%lld ",ar[i]);
    }
    else if(k%2!=0)
    {
        for(int i=0;i<n;i++)
        printf("%lld ",mx-ar[i]);
    }
    else
    {
        for(int i=0;i<n;i++)
        printf("%lld ",ar[i]-mn);
    }
    cout<<endl;

    return 0;
}
