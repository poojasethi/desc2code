#include<iostream>
using namespace std;

int main()
{
	long long int n,k,j,p;
	cin>>n>>k;
	long long int a[n],max=0,min=2000000005;
	for(long long int i=0;i<n;i++)
	{
		cin>>a[i];
		if(a[i]>max)
		{
			max=a[i];
		}
		else if(a[i]<min)
		{
			min=a[i];
		}
	}
	if(k==0)
	{
		for(p=0;p<n;p++)
		{
			cout<<a[p]<<" ";
		}
	}
	else if(k%2==0)
	{
		for(p=0;p<n;p++)
		{
			cout<<a[p]-min<<" ";
		}
	}
	else
	{
		for(p=0;p<n;p++)
		{
			cout<<max-a[p]<<" ";
		}
	}
}
