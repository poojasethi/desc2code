    #include<iostream>
    using namespace std;
    int main()
    {
    ios_base::sync_with_stdio(false);
    int n,i,k;
    cin>>n>>k;
    long long int a[n],max=-2000000001,min=2000000001;
    for(i=0;i<n;i++)
    {
    cin>>a[i];
    if(a[i]>max)
    max=a[i];
    else if(a[i]<min)
    min=a[i];
    }
    if(k==0)
    {
    for(i=0;i<n;i++)
    cout<<a[i]<<" ";
    }
    else
    {
    k%2==0?k=2:k=1;
    if(k==1)
    for(i=0;i<n;i++)
    cout<<max-a[i]<<" ";
    else
    for(i=0;i<n;i++)
    cout<<a[i]-min<<" ";
    }
    cout<<endl;
    }
