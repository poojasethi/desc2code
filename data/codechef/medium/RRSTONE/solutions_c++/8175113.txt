#include<bits/stdc++.h>
using namespace std;
typedef long long lli;
int main()
{
   lli n,k,i,maxm;
   scanf("%lld %lld ",&n,&k);
   lli arr[n];
   for(i=0;i<n;i++)
    scanf("%lld",&arr[i]);
   if(k==0)
   {
      for(i=0;i<n;i++)
      printf("%lld ",arr[i]);
   }
    else
    {
       maxm=arr[0];
       for(i=0;i<n;i++)
        if(arr[i]>maxm)maxm=arr[i];
       for(i=0;i<n;i++)
        arr[i]=maxm-arr[i];
       if(k&1)
       {
           for(i=0;i<n;i++)
          printf("%lld ",arr[i]);
       }
       else
       {
           maxm=arr[0];
          for(i=0;i<n;i++)
          if(arr[i]>maxm)maxm=arr[i];
          for(i=0;i<n;i++)
          arr[i]=maxm-arr[i];
          for(i=0;i<n;i++)
          printf("%lld ",arr[i]);
       }
    }
    return 0;
}
