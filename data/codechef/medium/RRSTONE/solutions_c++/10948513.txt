#include<iostream>
using namespace std;
int main()
{
	long long n,maxi=0,i,temp,mini=0,t,c;
	cin>>n;
	cin>>t;
	long long a[n];
	for(i=0; i<n; i++)
	cin>>a[i]; 
	a[maxi]=a[0],a[mini]=a[0];
	for(i=0; i<n; i++)
	{
		if(a[i]>a[maxi])
			maxi=i;
		if(a[i]<a[mini])
		mini=i;
	}
	if(t==0)
	{
		for (i=0; i<n; i++)
	cout<<a[i]<<" ";
	return 0;
	}
	else if(t%2) t=1;
	else t=2;
	while(t--)
	{	c=a[maxi];
		for (i=0; i<n; i++)
		a[i]=c-a[i];
		maxi=mini;
	}
	for (i=0; i<n; i++)
	cout<<a[i]<<" ";
	return 0;
}
