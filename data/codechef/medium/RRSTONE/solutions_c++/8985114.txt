#include <stdio.h>
#define getcx getchar_unlocked
int scanint()
{
   int n=0;
   int ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}
 
   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   return(n*sign);
}
int main() 
{
	long long int n,k,max=-1,min=1000000001;
	n=scanint();
	k=scanint();
	long long int a[n],i;
	for(i = 0; i < n; i++) {
		a[i]=scanint();
		if(a[i] > max)
		max = a[i];
		if(a[i] < min)
		min = a[i];
	}
	if(k == 0) {
		for(i = 0; i < n; i++) {
			printf("%Ld ",a[i]);
		}
	}
	else if(k%2) {
		for(i = 0; i < n; i++) {
			printf("%Ld ",max-a[i]);
		}
	}
	else {
		for(i = 0;  i < n; i++) {
			printf("%Ld ",max-min-(max-a[i]));
		}
	}
	printf("\n");
	return 0;
}
