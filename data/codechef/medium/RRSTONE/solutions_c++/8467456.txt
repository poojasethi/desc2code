#include <iostream>
#include <stdio.h>
#include <cmath>
#include <string.h>
#include <stack>
#include <set>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	long long n,k;
	long long i,j,temp;
	cin >> n >> k;
	long long arr[n];
	long long max;

	cin >> arr[0];
	max = arr[0];
	long long min = arr[0];

	for (i=1;i<n;i++)
	{
		cin >> arr[i];
		if (arr[i] > max)
			max = arr[i];
		if (arr[i] < min)
			min = arr[i];
	}

	if (k == 0)
	{
		for (i=0;i<n;i++)
			cout << arr[i] <<" ";
	}
	else
	{
		if (k%2 == 1)
		{
			for (i=0;i<n;i++)
				cout << max-arr[i] <<" ";
		}
		else
		{
			temp = max-min;
			for (i=0;i<n;i++)
				cout << temp-(max-arr[i]) <<" ";
		}
	}

	cout << endl;

	return 0;

}
