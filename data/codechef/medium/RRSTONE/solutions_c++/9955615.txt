    #include <stdlib.h>
    #include <math.h>
    #include <string.h>
    #include<stdio.h>
    #include <ctype.h>
 	
 
    #define MOD 1000000007
 
    typedef long long int ll;
 
    int main(void){
        ll N, K, i, j, a[100005], max;
        scanf("%lld %lld", &N, &K);
        for(i=0;i<N;i++){
            scanf("%lld", &a[i]);
        }
        if(K>0){
            for(i=0;i<(((K+1)%2)+1);i++){
                max=-2000000009;
                for(j=0;j<N;j++){
                    if(a[j]>max){
                        max=a[j];
                    }
                }
                for(j=0;j<N;j++){
                    a[j]=max-a[j];
                }
            }
            for(i=0;i<N;i++){
                printf("%lld ", a[i]);
            }
            printf("\n");
        }
        else{
            for(i=0;i<N;i++){
                printf("%lld ", a[i]);
            }
            printf("\n");
        }
        return 0;
    }
