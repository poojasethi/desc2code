#include<stdio.h>
 
#define Get getchar_unlocked()
int getInt() { int a=0,s=1; char c=0; while(c<33) c=Get; if(c=='-') {s=-1; c=Get;} while(c>33) {a=(a<<3)+(a<<1)+c-'0'; c=Get;} return a*s; }
 
int A[100000],N,K;
 
int main()
{
	N=getInt(); K=getInt();
	if(K==0) {
		while(N--) printf("%d ",getInt());
	}
	else if(K&1) {
		int i,max=-2000000001;
		for(i=0;i<N;i++) {
			A[i]=getInt();
			if(A[i]>max) max=A[i];
		}
		for(i=0;i<N;i++) printf("%lld ",(long long)(max)-A[i]);
	}
	else {
		int i,min=2000000001;
		for(i=0;i<N;i++) {
			A[i]=getInt();
			if(A[i]<min) min=A[i];
		}
		for(i=0;i<N;i++) printf("%lld ",(long long)(A[i])-min);
	}
	return 0;
}
