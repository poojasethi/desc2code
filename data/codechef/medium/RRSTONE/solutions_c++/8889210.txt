int main()
{
    int k,i,n,j;
    cin >> n >> k;
    assert(1 <= n && n <= 100000 && 0 <= k && k <= 2000000000);
    long long int a[n],m;
    for(i = 0;i < n;i++) {
        cin >> a[i];
        assert(abs(a[i]) <= 2000000000);
    }

    if(k % 2 == 1) {
        k = 1;
    }

    else {
        k = min(k,2);
    }
    for(i = 1;i <= k;i++){
            m = a[0];
        for(j = 0;j < n;j++) {
            m = max(m,a[j]);
        }
        for(j = 0;j < n;j++) {
            a[j] = m - a[j];
        }
    }
    for(i = 0;i < n;i++) {
        cout << a[i] << " ";
    }
    cout << "\n";
    return 0;
}
