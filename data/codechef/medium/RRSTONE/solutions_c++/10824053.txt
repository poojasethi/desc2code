#include <iostream>
#include <bits/stdc++.h>


using namespace std;

typedef long long int ll;
int main()
{
	std::ios::sync_with_stdio(false);
    ll n,k;
	cin>>n>>k;
	vector<ll>v(n);
	ll maxim = INT_MIN;
	for(ll i=0;i<n;i++)
	{
		int a;
		cin>>a;
		v[i] = a;
		if(a>maxim) maxim = a;
	}
	if (k % 2 == 1) k = 1; else k = k<2?k:2;
	for(ll i=1;i<=k;i++)
	{
		for(ll j = 0; j < n; j++) maxim = max(maxim, v[j]);
		for(ll j=0;j<n;j++)
		{
			v[j]=maxim-v[j];
		}
	}
	for(ll i=0;i<n;i++)
	{
		cout<<v[i]<<" ";
	}
	cout<<endl;
	return 0;
}
