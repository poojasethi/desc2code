#include<iostream>
using namespace std;
int main()
{
	long long int n,k,max,i,j;
	cin>>n>>k;
	long long int a[n];
	for(i=0;i<n;i++)
	cin>>a[i];
	if(k==0)
	k=0;
    else if(k%2==0)
	 k=2;
	 else
	 k=1;
	for(i=0,max=-1;i<k;i++)
		{
		for(j=0;j<n;j++)
		{
		if(max<a[j])
		max=a[j];
	    }
	    for(j=0;j<n;j++)
		 a[j]=max-a[j];
		}
		for(i=0;i<n;i++)
		cout<<a[i]<<" ";
	return 0;	
}
