class timed(object):
    def __init__(self, func):
        self.func = func
    def __call__(self, *args):
        t1 = time()
        ret = self.func(*args)
        dif_time = time() - t1
        print "%s: returned %s in %f seconds" % (self.func.__name__, ret, dif_time)
        return ret

def alg(s):
    count = 0
    opened = 0
    last_0 = 0
    for c in s:
        if c == '<':
            opened += 1
        elif c == '>':
            if opened > 0:
                opened -= 1
                if opened == 0:
                    last_0 = count + 1
            else:
                break
        count += 1
    if opened != 0:
        count = last_0
    return count
