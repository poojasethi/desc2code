#include<stdio.h>
 
int main()
{   char a[1000001];
	int test,i,l,max;
	scanf("%d\n",&test);
	while(test--)
	{
		max=0;
		l=0;
		gets(a);
		for(i=0;a[i]!='\0';i++)
		{
			if(a[i]=='<')
				l++;
			if(a[i]=='>')
			{
				l--;
				if(l==0)
				{
					max=i+1;
				}
				else if(l<0)
					break;
			}
		}
		printf("%d\n",max);
	}
	return 0;
}
