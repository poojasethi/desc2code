#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int MAXC = 1000010;

int t, deg, fil[MAXC], res;
char s[MAXC];

int main (){
    scanf("%d", &t);
    for(int a=0;a<t;a++){
        deg=res=0;
        s[0]='\0';
        scanf(" %s", s);
        int tam = strlen(s);
        int aux = 0;
        for(int b=0;b<tam;b++){
            if(s[b]=='<'){
                fil[deg++]=b;
            }
            else{
                if(deg==0){
                    aux=b+1;
                    break;
                }
                else{
                    deg--;
                    if(deg==0)
                        res=max(res,b-aux+1);
                }
            }
        }
        printf("%d\n", res);
    }
    return 0;
}
