#include <iostream>
#include <algorithm>
#include <fstream>
#include <vector>
#include <string>
#include <map>
#include <cstdio>

using namespace std;

int main()
{
	
	int t;
	cin>>t;
	while(t--)
	{
		string s;
		cin>>s;
		int a[1000000];
		for(int i=0;i<s.length();i++)
		a[i]=0;
		if(s[0]=='<')
			a[0] = 1;
		else
			a[0] = -1;
		for(int i=1;i<s.length();i++)
			if(s[i]=='<')
				a[i] = a[i-1] + 1;
			else
				a[i] = a[i-1] - 1;
		int ans=-1;
		for(int i=0;i<s.length();i++)
		{
			if(a[i]==0)
				ans=i+1;
			else if(a[i]<0)
				break;
		}
		if(ans>=0)
		cout<<ans<<"\n";
		else
		cout<<"0\n";	

	}
	
}
