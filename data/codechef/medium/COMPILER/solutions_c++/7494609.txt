#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)
#define f first
#define sc second

using namespace std;

stack<ll> s;
ll st[1000001];
string stt;

int main()
{
    ll t,i,j,k,l,ans;
    s(t);
    while(t--) {
        ans=0;
        stt.clear();
        cin>>stt;
        for(i=0;i<stt.length();i++) {
            if(stt[i]=='<') st[i]=-1;
            else st[i]=1;
        }
        for(i=0;i<stt.length();i++) {
            if(s.empty()) {s.push(st[i]);}
            else if(s.top() < 0 && st[i] > 0 && s.top()+st[i]==0) {s.pop();}
            else {s.push(st[i]);}
            if(s.empty()) ans=i+1;
        }
        cout<<ans<<endl;
        while(!s.empty()) s.pop();
    }
    return 0;
}
