#include<bits/stdc++.h>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int T;
	cin>>T;
	while(T--)
	{
		string s;
		cin>>s;

		int ans = 0;
		int st = 0;
		for(int i=0;i<s.size();i++)
		{
			if(s[i] == '<')
			{
				st++;
			}
			else
			{
				st--;
			}

			if(st == 0)
			{
				ans = i+1;
			}
			else if(st < 0)
			{
				break;
			}
		}

		cout<<ans<<"\n";
	}
	return 0;
}
