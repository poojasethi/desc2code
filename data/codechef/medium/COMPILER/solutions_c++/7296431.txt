#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<cmath>
#include<map>
#include<queue>
#include<stack>
using namespace std;
int T,n;
char S[1000005];
int main()
{
	scanf("%d",&T);
	while(T--){
		scanf("\n%[^\n]",S);
		n=strlen(S);
		int res=0;
		stack<char> Q;
		for(int i=0;i<n;i++)
		if(!Q.empty())
		{
			if(S[i]=='<')Q.push(S[i]);
			else Q.pop();
			
			if(Q.empty())res=i+1;
		}
		else if(S[i]=='>')
		{
			res=i;
			break;
		}
		else Q.push(S[i]);
		cout<<res<<"\n";
	}
	return 0;
}
