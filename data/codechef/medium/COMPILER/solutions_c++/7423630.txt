#include<bits/stdc++.h>
#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
#define gc() getchar_unlocked()
#define pc(x) putchar_unlocked(x);
inline int fastio()
{
    int x=0;
    char c=gc();
    while(c<'0'||c>'9')
        c=gc();
    while(c>='0'&&c<='9')
    {
        x=(x<<1)+(x<<3)+c-'0';
        c=gc();
    }
    return x;
}


void fastoutput(int n)
{
    int a[21]={0};
    int l=0;
    if(n==0){
    pc('0');pc('\n');
    return;}
    while(n!=0)
    {
        a[l++]=n%10;
        n/=10;
    }
    l--;
    for(;l>=0;l--)
        pc(a[l]+'0');
    pc('\n');
    return;
}

int main()
{
    int t;
    char str[1000001];
    //scanf("%d",&t);
    t=fastio();
    while(t--)
    {
        //cin>>str;
        scanf("%s",str);
        int i,bal=0,ans=0;
        int l=strlen(str);
        for(i=0;i<l;i++)
        {
            if(str[i]=='<')
                bal++;
            else
                bal--;
            if(bal<0)
                break;
            if(bal==0)
                ans=(i+1);
        }
       fastoutput(ans);
       //printf("%d\n",ans);
    }
    return 0;
}
