#include<bits/stdc++.h>
using namespace std;

int main()
{

    int t;
    scanf("%d",&t);

    while(t--)
    {
        string s;
        cin>>s;
        int sum=0,ans=0;
        for(int i=0;i<s.size();i++)
        {
            if(s[i]=='<')sum++;
            else sum--;
            if(sum==0)
            {
                ans=i+1;
            }
            if(sum<0)break;
        }

        printf("%d\n",ans);
    }
    return 0;
}
