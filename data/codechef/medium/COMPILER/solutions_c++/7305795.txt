#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t,c,i,ans;
    string s;
    scanf("%d",&t);
    cin.ignore();
    while(t--)
    {
        c=0;
        ans=0;
        cin>>s;
        for(i=0;i<s.length();i++)
        {
            if(s[i]=='<')
            {
                c++;
            }
            else
            {
                c--;
                if(c==0)
                    ans=max(ans,i+1);
                if(c<0)
                    break;
            }
        }

            printf("%d\n",ans);

    }
}
