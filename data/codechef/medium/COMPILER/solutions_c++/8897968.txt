 #include <iostream> // for cout

# define limit 1000000
//#define gc getchar_unlocked
using namespace std;
int n,t;
char s[limit+2];
int count;

int main()
{
	int i,p;
	cin>>t;
	while(t-->0)
	{
		count=0;
		p=-1;// last i till where prfix is valid
		//cin>>n;
		scanf("%s",s);
		//printf("%s\n",s);
		for(i=0;s[i];i++)
		{
			if(s[i]=='<')
				count++;
			else // symbol is '>'
				count--;
			if(count==0)
				p=i;
			else if(count<0)
				break;
			//printf("The symbol is %c at %d and count is %d\n",s[i],i,count);
		}
		printf("%d\n",(p+1));
	}
	
	return 0;
}
