#include<bits/stdc++.h>
#include<math.h>
#define ll long long int
using namespace std;

int main(){
    ios_base::sync_with_stdio(false);
    int n;
    cin>>n;
    while(n--){
        string s;
        cin>>s;
        stack<char> myStack;
        ll ans_final = 0;
        ll ans = 0;
        for(int i = 0;i<s.size();i++){
            char temp = s[i];
            ans++;
            if(temp == '<'){
                myStack.push(temp);
            }else{
                if(myStack.size()>0){
                    char temp1 = myStack.top();
                    if(temp1 == '<'){
                        myStack.pop();
                        if(myStack.size() == 0){
                            ans_final = ans;
                        }
                    }else{
                        break;
                    }
                }else{
                    break;
                }
            }
        }
        cout<<ans_final<<endl;
    }
    return 0;
}
