#include<bits/stdc++.h>
using namespace std;
int main()
{
int counter = 0;
char a[1000005];
int t, i;
int max;
scanf("%d", &t);
while(t--) {
	max = counter =  0;
	scanf("%s", a);
	for (i = 0 ; i < strlen(a); i++) {
		if(a[i] == '<') {
			counter++;
		} else {
			counter--;	
			if(counter < 0) {
				break;
			}
			if(counter == 0) {
				max = i + 1;
			}
		}
	}
	printf("%d\n", max);
}
 return 0;
}
