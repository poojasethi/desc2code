#include<bits/stdc++.h>
using namespace std;
int maxim(int a, int b){
    return (a>b)?a:b;
}
int main(void)
{
    int t;
    for(scanf("%d", &t); t--; ){
        char str[1000001];
        scanf("%s", str);
        int ans=0, k=0;
        for(int i=0; i<strlen(str); i++){
            if(str[i]=='<')
                k++;
            else
                k--;
            if(k<0)
                break;
            else if(k==0){
                ans = maxim(ans, i+1);
            }
        }
        printf("%d\n", ans);
    }
}
