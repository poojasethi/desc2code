
#include<bits/stdc++.h>

using namespace std;

char s[1000001];

int main()

{

	int t;

     scanf("%d\n",&t);

	while(t--)

	{

		cin>>s;

		int l=strlen(s);

		int a=0,ans=0;

		for(int i=0;i<l;i++)

		{

			if(s[i]=='<')a++;

			else

			{

			   a--;

			   if(a==0)

			   ans=i+1;

			   if(a<0)

			   break;

			}

		}

		printf("%d\n",ans);

	}

}
