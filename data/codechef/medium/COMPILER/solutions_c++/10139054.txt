#include <iostream>
using namespace std;

int main() {
	int n;
	cin>>n;
	while(n--){
	    char a[1000001];
    	cin>>a;
	    int c=0;
	int ans=0;
	for(long int i=0;a[i]!='\0';i++){
	    if(a[i]=='<')
	    c++;
	    else
	    c--;
	    if(c<0){
	    break;
	    }
	    if(c==0)
	    ans=i+1;
	}
	
cout<<ans<<endl;
	}

	return 0;
}
