#include <stdio.h>
char s[1000002];
int main()
{
	int i, j, ans;
	int t;
	scanf("%d",&t);
	getchar();
	while (t --) {
		gets(s);
		i = 0;
		ans = 0;
		j = 0;
		while (s[i]) {
			if (s[i++] == '<') {
				j++;
			} else {
				j--;
			}
			if (j < 0) {
				break;
			} else if (j == 0) {
				ans = i;
			}
		}
		printf("%d\n",ans);
	}
	
	return 0;
}
