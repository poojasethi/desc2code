#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin >> t;

	while(t--) {
		string s;
		cin >> s;

		int i,t = 0,ans = 0;
		for(i = 0;i < s.size();i++)  {
			if(s[i] == '<')
				t++;
			else {
				t--;
				if(t == 0) {
					ans = max(ans,i+1);
				}

				else if(t < 0)
					break;
			}
		}
		cout << ans << "\n";
	}
	return 0;
}
