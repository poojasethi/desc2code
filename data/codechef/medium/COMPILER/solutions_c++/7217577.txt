using namespace std;

int a[1000001];

int main()
{
	int t;
	cin >> t;
	while (t--)
	{
		string s;
		cin >> s;
		a[0] = 0;
		for (int i = 0; i < s.length(); i++)
		{
			a[i+1] = 0;
			a[i + 1] = a[i];
			if (s[i] == '>')
				a[i + 1]--;
			else
				a[i + 1]++;
		}
		int ans = 0;
		for (int i = 1; i <= s.length(); i++)
			if (a[i] < 0)
				break;
			else if (a[i] == 0)
				ans = i;
		cout << ans << '\n';
	}
	return 0;
}
