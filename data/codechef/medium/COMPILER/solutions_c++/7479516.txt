#include<bits/stdc++.h>
using namespace std;
int main()
{
    int T;
    cin >> T;
    string str;

    for(int i=0; i<T; i++)
    {
        cin >> str;
        int c1=0;
        int c2=0;
        int ans =0;

        for(int j=0; j < str.length(); j++)
        {

            if(str[j] == '<')
                c1++;
            else if(str[j] == '>')
                c2++;
            if(c2 >c1)
                break;
            else if (c2==c1)
                ans = 2*c1;
        }
        cout << ans << endl;
    }
    return 0;
}
