#include<stdio.h>
char s[1000005];
int i,c,ans,t,d;
int main()
{
	scanf("%d",&t);
	while(t--)
	{
		c=0;
		ans=0;
		d=0;
		scanf("%s",s);
		for(i=0;s[i]!='\0';i++)
		{
			if(s[i]=='<')
			c++;
			else
			{
				d++;
				c--;
			}
			if(c<0)
			break;
			if(c==0)
			ans=2*d;
		}
		printf("%d\n",ans);
	}
	return 0;
}
