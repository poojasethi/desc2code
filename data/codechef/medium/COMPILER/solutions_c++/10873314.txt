#include<bits/stdc++.h>
#include <cassert>

using namespace std;
typedef long long int ll;
ll MAX = 1000000;
int a[10][100000];
int main() {
    ios::sync_with_stdio(0);
    int t;
    cin >> t;
    while (t --) 
	{
    	string s;
    	cin>>s;
    	int k=0,ans=0;
    	for(int i=0;i<s.length();i++)
    	{
    		if(s[i] == '<') k++;
    		else
    		{
    			k--;
    			if(k==0) ans = max(ans,i+1);
    			if(k<0) break;
			}
		}
		cout<<ans<<endl;
    }
    return 0;
}
