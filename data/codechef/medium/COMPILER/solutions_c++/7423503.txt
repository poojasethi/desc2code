#include<bits/stdc++.h>
#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int t;
    char str[1000001];
    scanf("%d",&t);
    while(t--)
    {
        cin>>str;
        int i,bal=0,ans=0;
        int l=strlen(str);
        for(i=0;i<l;i++)
        {
            if(str[i]=='<')
                bal++;
            else
                bal--;
            if(bal<0)
                break;
            if(bal==0)
                ans=(i+1);
        }
       printf("%d\n",ans);
    }
    return 0;
}
