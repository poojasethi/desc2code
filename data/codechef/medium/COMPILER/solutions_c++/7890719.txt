#include<iostream>
#include<stack>
#include<cstring>
using namespace std;

char str[1000010];
int main()
{
	int i,j,t,count;
	scanf("%d",&t);
	while(t--)
	{
		count=0;
		scanf("%s",str);
		stack<int> sign;
		for(i=0;str[i]!='\0';i++)
		{
			if(str[i]=='<')
			sign.push(i);
			else if(str[i]=='>')
			{
				if(!sign.empty())
				{
					j = sign.top();
					sign.pop();
					if(str[j]=='<')
					count+=2;
					else
					break;
				}
				else
				break;
			}
		}
		if(!sign.empty())
		{
			while(!sign.empty())
			{
				j=sign.top();
				sign.pop();
			}
			
			count= j;
		}
		
		printf("%d\n",count);
	}
	return 0;
}
