#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
#include<string.h>
#include<math.h>
#define MOD 1000000007
using namespace std;


int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,ans=1,q,freq=1,max;
        scanf("%d",&n);
        scanf("%d",&max);
        n--;
        while(n--)
        {
            scanf("%d",&q);
            if (q==max) freq++;
            else
            {
                if (q>max)
                {
                    max = q;
                    freq = 1;
                }
            }
        }
        while(freq--)
        {
            ans = (ans << 1)%MOD;
            ans%=MOD;
        }
        ans--;
        printf("%d\n",ans);
    }
    return 0;
}
