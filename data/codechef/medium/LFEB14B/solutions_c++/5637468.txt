const long long base = 1000000000 + 7;

using namespace std;

void solve(){
	int n;
	cin >> n;
	int ma = -1;
	int cnt = 0;
	while(n--)
	{
		int x;
		cin >> x;
		if (x > ma)
		{
			ma = x;
			cnt = 0;
		}
		if (x == ma)
			cnt++;			
	}
	int ans = 1;
	while (cnt--)
	{
		ans += ans;
		if (ans >= base)
			ans -= base;		
	}
	ans--;
	cout << ans << endl;
}

int main(){
	ios_base :: sync_with_stdio(false);
	int test = 1;
	cin >> test;
	while (test--){
		solve();
	}
	return 0;
}
