#include<iostream>
#include<algorithm>
using namespace std;
#define y 1000000007
long long int expo(long long int base,long long int exp)
{
    if(exp==1)
    return base;
    else
    {
        if(exp%2 == 0)
        {
            long long int base1=expo(base, exp/2);
            base1=base1%y;
            base1=(base1*base1)%y;
            return base1%y;
        }
        else
        {
            long long int ans = expo(base,(exp-1)/2);
            ans=ans%y;
            ans=(ans*ans)%y;
            base=base%y;
            ans=(base*ans)%y;
            return ans%y;
        }
    }
}

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n],i=0,c=0;
        for(i=0;i<n;i++)
            cin>>a[i];
        sort(a,a+n);
        i=n-1;
        while(a[n-1]==a[i])
        {
            i--;
            c++;
            if(i<0)
                break;
        }
        long long int z=expo(2,c)%y;
        cout<<z-1<<endl;
    }
    return 0;
}
