ll pow1(int x,int y)
{
ll ret=1;
while(y>0)
{
    if(y&1)
    {
        ret=(ret*x)%mod;
    }
 
        x*=x;
        x%=mod;
 
    y>>=1;
}
return(ret%mod);
}
 
 
int scan()
{
    int x=0;
    char c=getchar_unlocked();
    while(c<'0'||c>'9')
    {
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9')
    {
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
 
void fastWrite(ll x)
{
    char ctemp[21];
    int l=0;
    do{
        ctemp[l++]=x%10+'0';
        x/=10;
    }while(x);
    --l;
    while(l>=0)
    {
        putchar_unlocked(ctemp[l--]);
    }
    putchar_unlocked('\n');
}
