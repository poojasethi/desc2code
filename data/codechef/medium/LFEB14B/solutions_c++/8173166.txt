#include<stdio.h>
#include<algorithm>
#include<cmath>
typedef long long ll;
bool sorting(ll a,ll b)
{
return a>b;
}
int main()
{
    ll t,n,i,count,ans,max;
    ll ar[100005];
    scanf("%lld",&t);
    while(t--)
    {
              count=0;
              scanf("%lld",&n);
              for(i=0;i<n;i++)
              scanf("%lld",&ar[i]);
              std::sort(ar,ar+n,sorting);
              max=ar[0];
              for(i=0;i<n;i++)
              {
                              if(max==ar[i])
                              count++;
                              else
                              break;
                              }
                              ans=1;
                      for(i=0;i<count;i++)
                      {
                        ans=2*ans;
                        if(ans>=1000000007)
                        ans=ans-1000000007;
                        }
                        ans--;
                        if(ans<0)
                        ans=ans+1000000007;
                      printf("%lld\n",ans);
                      }
                      return 0;
                      }
