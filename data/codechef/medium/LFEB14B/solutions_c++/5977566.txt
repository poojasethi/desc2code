#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
inline long long  xpow(int n)
{
	long long y=2,ans=1;
	while(n)
	{
		if(n&1)
			ans=(ans*y)%mod;
		y=(y*y)%mod;
		n>>=1;
	}
	return ans;
}
int main()
{
	int t;
	int i,j,k,n,x,c;
	long long ans,xy;
	scanf("%d",&t);
	//cout<<xpow(5)<<"\n";
	while(t--)
	{
		scanf("%d",&n);
		x=c=0;
		for(i=0;i<n;++i)
		{
			scanf("%d",&j);
			if(j>x){
				x=j;c=1;
			}
			else if(j==x)++c;
		}
		//cout<<x<<" "<<c<<"\n";
		printf("%lld\n",xpow(c)-1);
	}
	return 0;
}
