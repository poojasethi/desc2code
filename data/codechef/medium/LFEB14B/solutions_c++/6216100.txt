#include<bits/stdc++.h>
#define mod 1000000007ll
#define ll long long int
ll power(ll bas, ll exp){
    if(exp==1) return bas%mod;
    if(exp==0) return 1;
    else if(!exp%2) return (power((bas*bas)%mod,exp/2)%mod);
    else return (bas*(power(bas,exp-1)%mod))%mod;
}
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n; cin>>n;
        ll count=0,ma=0;
        for(int i=0; i<n; i++){
            ll x; cin>>x;
            if(x>ma){
                ma=x;
                count=1;
            }
            else if(x==ma) count++;
        }
        ll ans = power(2,count);
        cout<<ans-1<<endl;
    }
}
