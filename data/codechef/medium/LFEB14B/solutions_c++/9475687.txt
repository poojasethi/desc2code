#include <bits/stdc++.h>
using namespace std;
#define D 1000000007
typedef long long ll;
#define MAX 100005
ll power[MAX];
int main() 
{
    power[0]=1;
    for(int i=1;i<MAX;i++) power[i] = (power[i-1]*2)%D;
	int t; scanf("%d",&t);
	while(t--)
	{
	    int n; scanf("%d",&n); int a[n],cnt=1;
	    for(int i=0;i<n;i++) scanf("%d",&a[i]); sort(a,a+n);
	    for(int i=n-2;i>=0;i--)
	        if(a[i]!=a[i+1])
	            break;
	        else
	            cnt++;
	    ll finalans = (power[cnt]-1+D)%D;
	    printf("%lld\n",finalans);
	}
	return 0;
}
