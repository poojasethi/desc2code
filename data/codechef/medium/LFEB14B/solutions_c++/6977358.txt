ll modMul(ll x, ll y,ll z) {
    ll result = ((long long)x * y) % z;
    return result;
}

ll poww(ll x,ll y, ll z) {
    if (y == 0)
        return 1;

    ll sqrt = poww(x, y / 2, z);
    ll result = modMul(sqrt, sqrt, z);

    if (y % 2 == 1)
        result = modMul(result, x, z);

    return result;
}
