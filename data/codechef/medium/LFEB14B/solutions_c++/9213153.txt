#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	for(int i=0;i<t;i++)
	{
		int n,max,f=0;
		scanf("%d",&n);
		int a[n];
		for(int j=0;j<n;j++)
		{
			scanf("%d",&a[j]);
		}
		sort(a,a+n);
		for(int j=n-1;j>=0;j--)
		{
			if(a[j]==a[n-1])
			 f++;
			else
			 break; 
		}
		long long int ways=1;
		for(int j=1;j<=f;j++)
		{
			ways=(ways*2)%1000000007;
		}
		ways=ways-1;
		printf("%lld\n",ways);
	}
	return 0;
}
