#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define mod 1000000007

int main()
{
    ios::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n];
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
        }
        int m=a[0];
        for(int i=1;i<n;i++)
        {
            m=max(m,a[i]);
        }
        int k=0;
        for(int i=0;i<n;i++)
        {
            if(a[i]==m)
            {
                k++;
            }
        }
        ll ans=1;
        for(int i=0;i<k;i++)
        {
            ans=(ans*2)%mod;
        }
        cout<<ans-1<<"\n";
    }
    return 0;
}
