#define MOD 1000000007
ll pw(ll base,int p){
	ll ans=1;
	while(p){
		if(p&1){
			ans=(ans*base)%MOD;
		}
		base = (base*base)%MOD;
		p/=2;
	}
	return ans;
}
int main(){
	int t;
	scan(t);
	while(t--){
		int n;
		scan(n);
		int arr[100004];
		int i=0,m=0;
		scan(arr[0]);
		rep(i,1,n){
			scan(arr[i]);
			if(arr[i]>arr[m]){
				m = i;
			}
		}
		int ctr=0;
		rep(i,0,n){
			if(arr[i]==arr[m]){
				ctr++;
			}
		}
		ll ans = pw(2,ctr);
		ans = (ans-1+MOD)%MOD;
		printll(ans); 
	}
	return 0;
}
