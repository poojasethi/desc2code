#include<stdio.h>
#include<iostream>
#include<math.h>
#include<algorithm>
#include<string.h>
#include<vector>
 
using namespace std;
typedef long long ll;
 
#define S(x) scanf("%lld",&x)
#define P(x) printf("%lld\n",x)
#define SARR(_A,_N) for(ll _Z=0; _Z<_N; _Z++) \
scanf("%lld",&_A[_Z])
#define PARR(_A,_N) for(ll _Z=0; _Z<_N; _Z++) \
printf("%lld ",_A[_Z]); \
printf("\n")
ll a[100005];
int main()
{
    ll t;
    S(t);
    while(t--)
    {
              ll n;
              S(n);
              SARR(a,n);
              ll max=a[0];
              ll i;
              for(i=1;i<n;i++)
              {
                if(a[i]>max)
                max=a[i];
              }
              ll count=0;
              for(i=0;i<n;i++)
              {
                 if(a[i]==max)
                 count++;
              } 
              ll ans=1;
              for(i=1;i<=count;i++){
              ans=ans*2;
              ans=ans%1000000007;}
              ans=ans-1;
              P(ans);
    }
  
    return 0;
}
