#include<iostream>
#include<algorithm>
#include<math.h>
#define MOD 1000000007
using namespace std;
int main(){
int t;
cin>>t;
while(t--){
    int n,c=1,ans,i,x;
    cin>> n;
    int arr[n];
for(i=0;i<n;i++)
     cin>>arr[i];
sort(arr,arr+n);
x=(*max_element(arr,arr+n));
for(i=n-2;i>=0;i--){
      if(arr[i]==x)
            c++;
}
ans=1;
		for(i=0;i<c;i++)
            ans=(ans*2)%1000000007;
cout<<(ans-1)<<"\n";
}
return 0;
}
