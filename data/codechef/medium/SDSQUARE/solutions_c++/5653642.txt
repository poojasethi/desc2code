#include<bits/stdc++.h>
using namespace std;

int main()
{
        long long t,a,b,i,temp,dig,flag;
        cin >> t;

        vector<long long> v;

        for(i=0;i<=100000;i++)
        {
                temp = i*i;
                flag=0;
                while(temp>0)
                {
                        dig=temp%10;
                        if(! (dig==0 || dig==1 || dig==4 || dig==9))
                        {
                                flag=1;
                                break;
                        }
                        temp/=10;
                }
                if(!flag)
                v.push_back(i*i);
        }

        //for(i=0;i<v.size();i++)
        //cout << v[i] << " ";


        while(t--)
        {
                cin >> a >> b;
                cout << lower_bound(v.begin(), v.end(), b+1) - lower_bound(v.begin(), v.end(), a) << endl;
        }
        return 0;
}
