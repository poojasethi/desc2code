#include<iostream>
#include<stdio.h>
using namespace std;

long long sqa[1000];

int main(){
	int t; 
	scanf("%d", &t);
	long long x,a,b,val,i,num,k=1,beg,end; bool flag,flag1;
	for(i=1;i<=100000;i++){
		long long x=i*i;
		num=x;  flag=0;
		while(x>0){
			val=x%10;
			x=x/10;
			if(val!=0 && val!=1 && val!=4 && val!=9){
				flag=1;
				break;
			}
		}
		if(flag==0){
			sqa[k++]=num;
		}
	}
	while(t--){
		scanf("%lld %lld", &a, &b);
		flag1=0;
		for(i=1;i<k;i++){
			if(sqa[i]>=a && flag1==0){
				beg=i;
				flag1=1;
			}
			if(sqa[i]>b){
				end=i-1;
				break;
			}
			else if(sqa[i]==b){
				end=i;
				break;
			}
		}
		printf("%lld\n", end-beg+1);
	}

	return 0;
}
