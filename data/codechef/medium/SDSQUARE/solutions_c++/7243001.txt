#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
typedef long long lli;
lli arr[100001];
void solve()
{
    lli num,i;
    arr[0]=1;
    arr[1]=2;
    for(i=2;i<100001;i++)
    {
        int flag=0,d;
        num=i*i;
        while(num>0)
        {
            d=num%10;
            if(d==0 || d==1 || d==4 || d==9)
                num/=10;
            else
            {
                flag++;
                break;
            }
        }
        if(flag==0)
            arr[i]=arr[i-1]+1;
        else
            arr[i]=arr[i-1];
    }
}
int main()
{
    solve();
    int t;
    lli a,b;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld %lld",&a,&b);
        a=ceil(sqrt(a));
        b=ceil(sqrt(b));
        printf("%lld\n",arr[b]-arr[a-1]);
    }
    return 0;
}
