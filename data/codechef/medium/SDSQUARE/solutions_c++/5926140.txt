#include<iostream>
#include<math.h>
using namespace std;
int main(){
	long long int s,a[100000],j,c,b,count,d;
	s=0;
	int t;
	for(long long int i=0;i<=100000;i++){
		j=i*i;
		//cout<<j<<endl;
		while(j!=0){
			d=j%10;
			//cout<<d<<endl;
			if(!(d==0||d==1||d==4||d==9)){
				break;
			}
			j=j/10;
			//cout<<j<<endl;
		}
		if(j==0)
		{
			//cout<<i<<endl;
			a[s]=i*i;
			//cout<<a[s]<<endl;
			s++;
		}
	}
	cin>>t;
	for(int i=0;i<t;i++){
		cin>>c>>b;
		long long int o,p;
		for(o=0;o<s;o++){
			if(a[o]>=c)
			break;
		}
		for(p=s-1;p>=0;p--){
			if(a[p]<=b)
			break;
		}
		cout<<p-o+1<<endl;
	}
}
