#include<bits/stdc++.h>
using namespace std;
int suit(long long n)
{
	while(n)
	{
		int dig=n%10;
		if(dig==0 || dig==1 || dig==4 || dig==9)
		{
			n/=10;
		}
		else
		return 0;
	}
	return 1;
}
main()
{
	int tc;
	long long i;
	vector<long long> v;
	for(i=0;i<=100000;i++)
	{
		if(suit(1ll*i*i))
		v.push_back(i*i);
	}
	scanf("%d",&tc);
	while(tc--)
	{
		long long n1,n2;
		scanf("%lld %lld",&n1,&n2);
		int pos2=lower_bound(v.begin(),v.end(),n2)-v.begin();
		int pos1=lower_bound(v.begin(),v.end(),n1)-v.begin();
		if(v[pos2]!=n2)pos2--;
		printf("%d\n",pos2-pos1+1);
		
	}
	
}
