#include<iostream>
using namespace std;
#include<math.h>
bool issquare(int a)
{
   long long int b=1;
   b=b*a*a;
   for(;b&&b%10==0||b%10==1||b%10==4||b%10==9;b=b/10);
   if(b==0)
      return true;
      return false;
}
int main()
{
   int T,square[100001],a,b,i;
   long long int a1,b1;
   square[0]=0;
   for(i=1;i<=100001;i++)
   {
      if(issquare(i))
      square[i]=square[i-1]+1;
      else
      square[i]=square[i-1];
   }
   cin>>T;
   while(T--)
   {
      cin>>a1>>b1;
      a=sqrt(a1);
      b=sqrt(b1);
      if(a*a!=a1)
      a++;
      cout<<square[b]-square[a-1]<<endl;
   }
return 0;
}
