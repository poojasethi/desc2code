#include<iostream>
#include<cstdio>
#include<math.h>
using namespace std;
long long int MAX = 10000000000ULL;
long long int arr[10000000],squares[100000];
int main()
{
   long long int t;
   arr[0] = 1;
   arr[1] = 4;
   arr[2] = 9;
   long long int top = 0;
   long long int cnt = 3;
   while(1)
   {
      long long int tmp = arr[top]*10;
      if(tmp > MAX)
        break;
      arr[cnt++] = tmp + 0;
      arr[cnt++] = tmp + 1;
      arr[cnt++] = tmp + 4;
      arr[cnt++] = tmp + 9;
      top++;
   }
   //printf("%lld\n",cnt);
   int square_cnt = 0;
   for(int i=0; i<cnt; i++)
   {
       long long int sqt = (long long int )sqrt(1.0*arr[i]);
       if((sqt*sqt) == arr[i])
            squares[square_cnt++] = arr[i];
   }
   scanf("%lld",&t);
   long long int a,b;
   //printf("%lld\n",t);
   while(t--)
   {
       scanf("%lld%lld",&a,&b);
       int ans = 0;
       for(int i=0; i<square_cnt; i++)
       {
           if(a <= squares[i] && squares[i] <= b)
            ans++;
       }
       printf("%d\n",ans);
      //printf("%lld\n",t);
   }
   return 0;
}
