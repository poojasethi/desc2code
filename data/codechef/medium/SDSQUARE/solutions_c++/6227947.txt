#include <iostream>
#include <cmath>
#include <vector>
using namespace std;

int main() {
	long long int tc, cpy, a, b, i, t, count;
	vector<long long int> v;
	
	for (i = 1; i <= 100000; i++) {
		cpy = i * i;
			while (cpy > 0) {
				t = cpy % 10;
				if (t == 2 || t == 3 || t == 5 || t == 6 || t == 7 || t == 8) {
					break;
				}
				cpy /= 10;
			}
			
			if (cpy == 0) {
				v.push_back(i * i);
			}
		}
	
//	for (i = 0; i != v.size(); i++) {
//		cout << v[i] << " ";
//	} cout << endl;
	
	cin >> tc;
	
	while (tc--) {
		cin >> a >> b;
		count = 0;
		
		for (i = 0; i != v.size(); i++) {
			if (v[i] >= a && v[i] <= b) {
				count++;
			}
		}
		cout << count << endl;
	}
	
	return 0;
}
