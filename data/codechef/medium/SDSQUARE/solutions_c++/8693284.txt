#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
char arr[100001];
int main()
{
    int tc;
    long long int i,z,temp,flag,a,b,count;
    for(i=1;i<100001;i++)
    {
        flag=1;
        z=i*i;
        while(z)
        {
            temp=z%10;
            if(temp!=0 && temp!=1 && temp!=4 && temp!=9)
            {
                flag=0;
                break;
            }
            z=z/10;
        }
        if(flag)
            arr[i]=1;
    }

    scanf("%d",&tc);
    while(tc--)
    {
        count=0;
        scanf("%lld %lld",&a,&b);
        a=ceil(sqrt(a));
        b=ceil(sqrt(b));
        for(i=a;i<=b;i++)
        {
            if(arr[i])
                count++;
        }
        printf("%lld\n",count);
    }
    return 0;
}
