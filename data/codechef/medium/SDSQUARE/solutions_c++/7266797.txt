#include<bits/stdc++.h>
#define lli long long int
#define MAX 100005
lli arr[MAX];
using namespace std;
int fun(lli n)
{
	lli i,j,k;
	while(n!=0)
	{
		i=n%10;
		if(i!=0 && i!=1 && i!=4 && i!=9)
			return 0;
		n=n/10;
	}
	return 1;
}
int main()
{
	lli number,a,b,i,j,k,m,n,t,flag=0,count;
	for(i=0;i<MAX;i++)
	{
		number=(i+1)*(i+1);
		if(fun(number))
			arr[flag++]=number;
	}

	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&a,&b);
		i=0;
		while(arr[i]<a)
			i++;
		count=0;
		while(arr[i++]<=b)
			count++;
		printf("%lld\n",count);
	}
}
