int main()
{
	ll i,t,a,b,arr[100005],sq,rem,uni,flag=1,count;
	for (i=1;i<=100000;i++)
	{
		flag=1;
		sq=pow(i,2);
		rem=sq;
		while(rem>0)
		{	
			uni=rem%10;
			rem=rem/10;
			if (uni!=0 && uni!=1 && uni!=4 && uni!=9)
			{
				flag=0;
				break;
			}	
		}
		arr[i]=flag;
	}
	sc(t);
	while(t--)
	{
		sc(a);
		sc(b);
		
		count=0;

		ll start=sqrt(a);
		ll end=sqrt(b);
		
		double check=sqrt(a);

		if (check!=(double)start)
			start=start+1;
		
		for (i=start;i<=end;i++)
		{
			if (arr[i]==1)
				count++;
		}

		printf("%lld\n",count);
	}

	return 0;
}
