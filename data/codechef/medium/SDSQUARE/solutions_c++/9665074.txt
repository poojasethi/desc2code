#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
vector<LL> perF;
bool isP(LL x){
	while(x > 0){
		int dig = x % 10;
		if(!(dig == 0|| dig == 1 || dig == 4 || dig == 9)){
			return false;
		}
		x /= 10;
	}
	return true;
}
int main(){
	LL limit = 1e10;
	perF.push_back(0);
	for(long long i = 1;i*i <= limit;++i){
		if(isP(i*i)){
			perF.push_back(i*i);
		}
	}
	perF.push_back(2*limit);
	// for(int i = 0;i < perF.size();++i){
	// 	cout << perF[i] << "\n";
	// }
	int t;
	cin >> t;
	while(t--){
		LL a, b;
		scanf("%lld%lld",&a,&b);
		LL ans = upper_bound(perF.begin(), perF.end(), b) - 
					upper_bound(perF.begin(), perF.end(), a - 1);
		cout << ans << "\n";
	}
	return 0;
}
