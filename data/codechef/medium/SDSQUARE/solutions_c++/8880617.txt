#include <cstring>
#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <sstream>
#define ll long long int
using namespace std;

vector <ll> v;
ll i;
bool f(ll a)
{
stringstream s;
s<<a;
string b=s.str();
for(int i=0;i<b.size();i++)
{if(b[i]=='0'||b[i]=='1'||b[i]=='4'||b[i]=='9')
continue;
    else return 0;
}
return 1;
}
void fun()
{
 for(i=1;i<=100000;i++)
 {
     if(f(i*i))
        v.push_back(i*i);
 }
}

int main()
{ll a,b;int t,c;
    fun();
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld%lld",&a,&b);
        c=lower_bound(v.begin(),v.end(),b+1)-lower_bound(v.begin(),v.end(),a);
        printf("%d\n",c);
    }
    return 0;
}
