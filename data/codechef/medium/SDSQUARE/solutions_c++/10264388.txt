#include<stdio.h>
long long int verifyPerfectDigit( long long int tmp){
  while( tmp ){
    if(tmp%10!=0 && tmp%10!=1 && tmp%10!=4 && tmp%10!=9){
      return 0;
    }
    tmp = tmp / 10;
  }
  return 1;
}
main(){
  long long int t,i,j,count;
  long long int a,b,square, pdps[200];
  scanf("%lld", &t);
  j=0;
  for( i=1; i<=100000; i++ ){
    square = i*i;
    if( verifyPerfectDigit(square) == 1 ){
      pdps[j++] = square;
    }
  }
  while(t--){
    count = 0;
    scanf("%lld%lld", &a, &b);
    for(i=0; i<j; i++){
      if(pdps[i]>=a && pdps[i]<=b)
        count++;
    }
    printf("%lld\n", count);
  }
  return 0;
}
