#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
  ll z,y,i,s,b,t,x,a[100001];
  s=0;
  a[0]=0;
  for(i=ll(floor(sqrt(1)));i<=100000;i++){
           b=i*i;
           int st=0;
           while(b>0){
             x=b%10;
             if(x!=1 && x!=4 && x!=9 && x!=0){
                st=1;
                break;
             }
             b/=10;
           }
           if(st==0)
             s++;
           a[i]=s;
  }
  cin>>t;
  while(t--){
    cin>>z>>y;
    z=ll(ceil(sqrt(z)));
    y=ll(floor(sqrt(y)));      
    cout<<a[y]-a[z-1]<<endl;
  }                                                   
  return 0;
}
