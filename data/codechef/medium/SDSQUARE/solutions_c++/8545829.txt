#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    long long int squares[10000],square_cnt = 0;
    for(int i=1; i<=100000; i++)
    {
        long long int sqr = 1ll*i*i;
        while(sqr != 0)
        {
            int md = sqr % 10;
            if(md != 0 && md != 1 && md != 4 && md != 9)
                break;
            sqr /= 10;
        }
        if(sqr  == 0)
            squares[square_cnt++] = 1ll*i*i;
    }
    //printf("%lld",square_cnt);
    long long int t;
    scanf("%lld",&t);
   long long int a,b;
   //printf("%lld\n",t);
   while(t--)
   {
       scanf("%lld%lld",&a,&b);
       int ans = 0;
       for(int i=0; i<square_cnt; i++)
       {
           if(a <= squares[i] && squares[i] <= b)
            ans++;
       }
       printf("%d\n",ans);
      //printf("%lld\n",t);
   }
    return 0;
}
