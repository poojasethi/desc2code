#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long i,sq[100005],j=1,cnt,a,b,tmp,flag,t;
    sq[0]=0;
    for(i=1;i<=100000;i++)
    { flag=0;
        tmp=i*i;
        while(tmp!=0)
        {
            if(tmp%10==0||tmp%10==1||tmp%10==4||tmp%10==9)
            {
                tmp=tmp/10;
            }
            else
            {
                flag=1;
                break;
            }

        }
        if(flag==0)
        {
            sq[j]=i*i;
            j++;
        }

    }
    cin>>t;
    while(t--)
    {cnt=0;
        cin>>a>>b;
        for(i=1;sq[i]!=0;i++)
        {
            if(sq[i]>=a&&sq[i]<=b)
                cnt++;
        }
        cout<<cnt<<"\n";
    }
    return 0;
}
