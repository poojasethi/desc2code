#include<bits/stdc++.h>
#define LL long long int
#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)
#define pb push_back
#define F first

using namespace std;

int c[100005];
void f()
{
	LL a;
	int k,d;
	c[0]=1;
	for(LL i=1;i<=100004;i++)
	{
		k=1;
		a=i*i;
		while(a)
		{
			d=a%10;
			if(d!=0 && d!=1 && d!=4 && d!=9)
			{
				k=0;
			}
			a/=10;
		}
		if(k)
			{
				//cout<<i<<endl;
				//cout<<i*i<<endl;
				c[i]=c[i-1]+1;
			}
			else
			{
				c[i]=c[i-1];
			}
	}
}

int main ()
{
	f();
	int t,x,y;
	LL a,b;
	sc(t);
/*	for(int i=0;i<100;i++)
	{

	}
	cout<<endl;*/
	while(t--)
	{
		sl(a);
		sl(b);
		x=sqrt(a);
		y=sqrt(b);
		//cout<<x<<" "<<y<<endl;
		if(x*x==a)
		pf(c[y]-c[x-1]);
		else
		pf(c[y]-c[x]);
	}
	return 0;
}
