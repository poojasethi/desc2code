#include <bits/stdc++.h>

using namespace std;

bool check(long long n)
{
    while(n!=0)
    {
        int temp = n%10;
        if(temp!=0 && temp!=1 && temp!=4 && temp!=9)
            return false;
        n=n/10;
    }
    return true;
}

int main()
{
    vector <long long> v;
    for(long long i=1;i<100001;i++)
    {
        if(check(i*i))
            v.push_back(i*i);
    }
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long long a,b;
        scanf("%lld %lld",&a,&b);
        int ans = upper_bound(v.begin(),v.end(),b) - lower_bound(v.begin(),v.end(),a);
        cout<<ans<<endl;
    }
    return 0;
}
