#include<stdio.h>
#include<stdlib.h>
long long int a[200],k;
inline void calc(void)
{
	long long int i,flag=1,temp;
	a[k++]=0;
	k=0;
	for(i=1;i<=100002;i++)
	{
		long long int sq= i*i;
		while(sq)
		{
			temp= sq%10;
			if(temp==0 || temp==1 || temp==4 || temp==9)
			flag=1;
			else
			{
				flag=0;
				break;
			}
			sq=sq/10;
		}
		if(flag==1)
		a[k++]=i*i;
	}
/*	for(i=0;i<k;i++)
	printf("%lld ",a[i]);
	printf("%d\n",k);*/
}
int main()
{
	int i;
	long long int t,ax,b,ai,bi,fa,fb;
	calc();
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&ax,&b);
		fa=0;
		fb=0;
		for(i=0;i<k;i++)
		{
			if(a[i]>=ax && fa==0)
			{
				ai=i;
				fa=1;
			}
			if(a[i]==b)
			bi=i;
			else if(a[i]>b && fb==0)
			{
				bi=i;
				fb=1;
			}
		}
		printf("%d\n",bi-ai);
	}
	return 0;
	
}
