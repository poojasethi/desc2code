    #include<stdio.h>
    #define SIZE 100002
    int scan()
    {
        int n=0;
        char ch=getchar_unlocked();
        while(ch<'0'||ch>'9')
            ch=getchar_unlocked();
        while(ch>='0'&&ch<='9')
        {
            n=(n<<3)+(n<<1)+ch-'0';
            ch=getchar_unlocked();
        }
        return n;
    }
    void printDouble(double n)
    {
        char num[24];
        int i=0,j;
        int integer = n;
        int decimal = (n-integer)*10000000;
        int temp = decimal;
        while(decimal)
        {
            num[i++] = (decimal%10) + '0';
            decimal/=10;
        }
        for(j=1000000;j>temp;j/=10)
            num[i++] = '0';
        num[i++]='.';
        do
        {
            num[i++] = integer%10 + '0';
            integer/=10;
        }while(integer);
        i--;
        while(i>=0)
            putchar_unlocked(num[i--]);
        putchar_unlocked('\n');
    }
    int main()
    {
        int t,i;
        t=scan();
        double expected[SIZE]={0.0};
        for(i=1;i<=100000;i++)
            expected[i] = 0.45 + 0.1*(expected[i-1]);
        for(i=2;i<=100000;i++)
            expected[i]+=expected[i-1];
        while(t--)
            printDouble(expected[scan()]);
        return 0;
    }
