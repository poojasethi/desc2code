#include <iostream>
#include <cstdlib>
#include <cstdio>

using namespace std;

// Here I store the expected values
double E[100010];
double ans[1000010];

// Precomputing the table
void pre()
{
	E[0] = ans[0] = 0;
	E[1] = 45.0/100.0;		// Because there are 45 such cases out of 100 when they have a carry
	ans[1] = E[1];
	for(int i = 2; i <= 100000; i++)
	{
		E[i] = (45.0/100.0) + (10.0/100.0)*E[i-1]; 		// Both cases accomodated
		ans[i] += ans[i-1] + E[i];
//		cout << i << " " << E[i] << " " << ans[i] << endl;
	}
		
	return;
}


void solve()
{
	int N;
	scanf("%d",&N);
	printf("%lf\n",ans[N]);
	return;
}

int main()
{
	pre();
	int t;
	scanf("%d",&t);
	while(t--)
		solve();
	return 0;
}
