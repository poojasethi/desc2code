#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iomanip>
#include <iostream>
using namespace std;

#define get_int(n) scanf("%d",&n)

#define MAXT 100000
#define MAXN 100000

int t;
int n;

int main () {

	get_int(t);
	for (int i = 0; i < t; i++) {
		get_int(n);

		double answer = 0.5 * double(n);
		double subtract = 0.05;
		for (int j = 0; j < min(n, 10); j++) {
			answer -= subtract;
			subtract /= 10.0;
		}
		cout << setprecision(14) << answer << endl;
	}

	return 0;
}
