#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;
//#define MAX_N 15

#define MAX_N 100005
double p[MAX_N];
double e[MAX_N];

//p(n)=(9+2*p(n-1))/20;

void preprocess(){
	p[0]=0;
	e[0]=0;
	for (int i = 1; i < MAX_N; ++i)
	{
		p[i]=(9+2*p[i-1])/20;
		e[i]=e[i-1]+p[i];
		//cout<<p[i]<<" "<<e[i]<<"\n";
	}
}

int main(){
	preprocess();
	int t;
	cin>>t;
	int n;
	while(t--){
		cin>>n;
		printf("%.6lf\n",e[n]);
		//cout<<e[n]<<"\n";
	}

}
