#include<bits/stdc++.h>
using namespace std;
double arr[100005];
double prob[100005];
#define MAX 100002
void solve()
{
	prob[1]=0.45;
	arr[1]=0.45;
	double temp=0.45;
	for(int i=2;i<MAX;i++)
	{
		arr[i] = (1.0-arr[i-1])*temp + (arr[i-1]*(1.0-temp));
		prob[i] = arr[i] +prob[i-1];
	}
}
int main()
{
	int t;
	scanf("%d",&t);
	solve();
	while(t--)
	{
		int n;
		scanf("%d",&n);
		printf("%.9f\n",prob[n]);
		
	}
}
