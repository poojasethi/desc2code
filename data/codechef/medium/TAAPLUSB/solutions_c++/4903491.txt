#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
using namespace std;
int main()
{
   double d[100010];
   d[0]=0.0;
   d[1]=0.45;
   for(int i=2;i<100010;i++)
   d[i]=(((double)i)*0.45)+(0.1*d[i-1]);
   int t,n;
   scanf("%d",&t);
   while(t--)
   {
     scanf("%d",&n);
     printf("%0.6lf\n",d[n]);
   }
   return 0;
}
