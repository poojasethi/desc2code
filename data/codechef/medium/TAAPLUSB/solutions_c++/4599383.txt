#include <cstdio>
using namespace std;

int main()
{
	double arr[100000][2] = {{0.45, 0.45}};
	for(int i = 1; i < 100000; i++)
	{
		arr[i][0] = 0.45 + 0.1*arr[i-1][0];
		arr[i][1] = arr[i-1][1] + arr[i][0];
	}
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		printf("%f\n", arr[n-1][1]);
	}
    return 0;
}
