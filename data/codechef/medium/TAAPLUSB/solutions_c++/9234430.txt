#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define sz(_v) (int)(_v).size()
#define all(_v) (_v).begin(),(_v).end()
#define pii pair<int,int>
typedef long long ll;
const int INF = 1000000000;
const ll MOD = 1e9 + 7;

const int maxn = 100010;
double ans[maxn];

void calc() {
	ans[1] = 0.45;
	for (int i = 2; i <= maxn - 4; i ++) {
		double aa = ans[i - 1] / (10.00);
		aa += double(0.45) * i;
		ans[i] = aa;
	}
}

int main() {
	ios_base::sync_with_stdio(false);

	calc();
	int t; cin >> t;
	while (t --) {
		int n; cin >> n;
		cout << fixed;
		cout << setprecision(6) << ans[n] << "\n";
	}
	return 0;
}
