#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define mod 1000000007
using namespace std;
main()
{   std:ios_base::sync_with_stdio(false);
    int i,t,n;
    double e[100001],ans[100001];
    e[1]=0.45;
    
    for(i=2;i<=100000;i++)
    e[i]=0.45+0.1*e[i-1];
    
    ans[1]=0.45;
    for(i=2;i<=100000;i++)
    ans[i]=ans[i-1]+e[i];
    
    cin>>t;
    while(t--)
    {
    	cin>>n;
    	printf("%.6lf\n",ans[n]);
    }
    
	return 0;
}
