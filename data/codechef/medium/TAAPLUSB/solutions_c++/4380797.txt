#include<bits/stdc++.h>
using namespace std;
//double arr[100005];
double prob[100001];
#define MAX 100001
void solve()
{
	prob[1]=0.45;
	//arr[1]=0.45;
	double temp=0.45;
	double prev=0.45;
	for(int i=2;i<MAX;i++)
	{
		prev = (1.0-prev)*temp + (prev*(1.0-temp));
		prob[i] = prev +prob[i-1];
	}
}
int main()
{
	int t;
	scanf("%d",&t);
	solve();
	while(t--)
	{
		int n;
		scanf("%d",&n);
		printf("%.9f\n",prob[n]);
		
	}
}
