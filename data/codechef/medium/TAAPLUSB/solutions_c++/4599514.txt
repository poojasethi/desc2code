#include <cstdio>
using namespace std;

int readInt()
{
	char c = getchar_unlocked();
	int x = 0;
	while(!(c >= '0' && c <= '9'))
	{
		c = getchar_unlocked();
	}
	while(c >= '0' && c <= '9')
	{
		x = x*10 + c-'0';
		c = getchar_unlocked();
	}
	return x;
}

void printInt(int a)
{
	char c[5];
	int index = 0;
	while(a)
	{
		c[index] = a%10 + 48;
		a /= 10;
		index++;
	}
	while(index--)
	{
		putchar_unlocked(c[index]);
	}
}

int main()
{
	int t = readInt(), n;
	while(t--)
	{
		n = readInt();
		switch(n)
		{
			case 1:
				putchar_unlocked('0');
				putchar_unlocked('.');
				putchar_unlocked('4');
				putchar_unlocked('5');
				break;
			case 2:
				putchar_unlocked('0');
				putchar_unlocked('.');
				putchar_unlocked('9');
				putchar_unlocked('4');
				putchar_unlocked('5');
				break;
			case 3:
				putchar_unlocked('1');
				putchar_unlocked('.');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('5');
				break;
			case 4:
				putchar_unlocked('1');
				putchar_unlocked('.');
				putchar_unlocked('9');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('5');
				break;
			case 5:
				putchar_unlocked('2');
				putchar_unlocked('.');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('5');
				break;
			default:
				printInt((n-1)/2);
				putchar_unlocked('.');
				putchar_unlocked(((n%2)?4:9)+48);
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				putchar_unlocked('4');
				break;
		}
		putchar_unlocked('\n');
	}
    return 0;
}
