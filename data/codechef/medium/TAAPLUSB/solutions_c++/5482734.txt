#include<cstdio>
using namespace std;
int main()
{   long long int t,n,i;
	scanf("%lld",&t);
	double a[100001],s[100001];
	a[0]=0;
	a[1]=0.45;
	s[0]=0;
	s[1]=0.45;
	
	for(i=2;i<100001;i++)
	{
		a[i]=0.45+(10*a[i-1])/100;
		s[i]=s[i-1]+a[i];
	}
	
	while(t--)
	{
          scanf("%lld",&n);
        
		  
		  printf("%.6f\n",s[n]);
	}
	return 0;
}
