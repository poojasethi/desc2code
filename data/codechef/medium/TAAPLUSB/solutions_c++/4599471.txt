#include <cstdio>
using namespace std;

int main()
{
	double arr[5] = {0.45, 0.945, 1.4445, 1.94445, 2.444445};
	int t, n;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		if(n < 6)
		{
			printf("%f\n", arr[n-1]);
		}
		else
		{
			printf("%d.%d44444\n", (n-1)/2, (n%2)?4:9);
		}
	}
    return 0;
}
