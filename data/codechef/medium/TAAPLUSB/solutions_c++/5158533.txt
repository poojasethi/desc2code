#include<bits/stdc++.h>
using namespace std;
int i,j;
double ans[100005];
 
int main(){
    int t;
    cin >> t;
    int n;
    ans[1] = 0.45;
    ans[2] = 0.945;
    ans[3] = 1.4445;
    ans[4] = 1.94445;
    ans[5] = 2.444445;
    ans[6] = 2.944444;
    double k;
    while(t--){
        scanf("%d",&n);
        if(n<=6) printf("%lf\n",ans[n]);
        else{
            k = (n-1)/2;
            if(n%2) k+=0.444444;
            else k+=0.944444;
            printf("%lf\n",k);
        }
    }
    return 0;
}
