def can_cut(time):
    total_cut = 0
    for i in xrange(N):
        temp = H[i]+time*R[i]
        if temp>=L:
            total_cut+=temp
    return total_cut>=W
N,W,L=map(int,raw_input().split(" "))
H=[0]*N
R=[0]*N
for i in xrange(N):
    H[i],R[i]=map(int,raw_input().split(" "))
t_low=-1
t_high=10**18
while t_high-t_low>1:
    t_mid=(t_high+t_low)/2
    if can_cut(t_mid):
        t_high=t_mid
    else:
        t_low=t_mid
print t_high
