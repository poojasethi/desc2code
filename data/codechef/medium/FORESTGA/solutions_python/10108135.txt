n,w,l=map(int,raw_input().strip().split())
list=[]
for i in range(n):
    h,r=map(int,raw_input().strip().split())
    list.append((h,r,))

def can_cut(t):
    total=0
    for h,r in list:
        height=h+r*t;
        if height>=l:
            total+=height
    return total>=w

low=-1
high=1
while can_cut(high)==False:
    high=2*high
while high-low>1:
    mid=(low+high)/2
    if can_cut(mid):
        high=mid
    else:
        low=mid
print high
