#include<bits/stdc++.h>
using namespace std;
long long h[100005],r[100005],w,n,l;
long long bin(long long low,long long high)
{
	long long mid;
	while(low<high)
	{
		mid=(low+high)/2;
		long long sum=0;
		for(int i=0;i<n;i++)
		{
			if((h[i]+mid*r[i])>=l)
			sum+=(h[i]+mid*r[i]);
			if(sum>=w)
			break;
		}
		if(sum==w)
		return mid;
		else if(sum>w)
		high=mid;
		else
		low=mid+1;
	}
	return low;
}
int main()
{
	long long i;
	cin>>n>>w>>l;
	for(i=0;i<n;i++)
	{
		cin>>h[i]>>r[i];
	}
	cout<<bin(0,1e18)<<endl;
	return 0;
}
