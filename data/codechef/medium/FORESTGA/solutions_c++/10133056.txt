#include<bits/stdc++.h>
using namespace std;
long long ans;
long long h[100005],r[100005],w,n,l;
void bin()
{
	long long mid,high=1000000,low=0;
	high=high*high;
	high=high*1000000;
	while(low<=high)
	{
		mid=low+(high-low)/2;
		long long sum=0;
		for(int i=0;i<n;i++)
		{
			if((h[i]+mid*r[i])>=l)
			sum+=(h[i]+mid*r[i]);
			if(sum>=w)
			break;
		}
		//if(sum==w)
		//return mid;
		if(sum>=w)
		{
			high=mid-1;
			ans=min(ans,mid);
		}
		else
		low=mid+1;
	}
}
int main()
{
	long long i;
	cin>>n>>w>>l;
	for(i=0;i<n;i++)
	{
		cin>>h[i]>>r[i];
	}
	ans=1E18;
	bin();
	cout<<ans<<endl;
	return 0;
}
