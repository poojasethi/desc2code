#include<bits/stdc++.h>

using namespace std;

struct node{
	long long int hi;
	long long int ra;
};

struct node a[100000];

int compare(const void *a,const void *b)
{
	int l = ((struct node *)a)->hi;
	int r = ((struct node *)b)->hi;
	return l-r;
}

bool check(long long int mid,long long int l,long long int w,long long int n)
{
	long long int ans = 0;
	for(int i = 0;i < n;i++)
	{	
		if((a[i].hi+(mid*a[i].ra)) >= l)
		{
			ans+= (a[i].hi + mid*a[i].ra);
			if(ans >= w)
				return true;
		}
	}
	return false;
}

int main()
{
	long long int N;
	long long int L;
	long long int W;

	cin >> N;
	cin >> W;
	cin >> L;

	for(int i = 0;i < N;i++)	
	{
		cin >> a[i].hi;
		cin >> a[i].ra; 
	}

	long long int low = 0;
	long long int high = LLONG_MAX;

	qsort(a,N,sizeof(a[0]),compare);
	long long int mid;
	while(low < high)
	{
		mid = (low+high) >> 1;
		if(check(mid,L,W,N))
		{
			high = mid;
		}
		else
		{
			low = mid+1;
		}
		//cout << "High " << high << " " << "Low "<< low << endl;
	}

	cout << low << endl;
	return 0;
}
