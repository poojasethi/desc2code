#include <bits/stdc++.h>
using namespace std;
#define MAX 1000000000
unsigned long long int i,j,k,l,t,n,m,w,h,r,llow,hhigh,mid;
vector<std::pair<unsigned long long int,unsigned long long int> >v;
int main() 
{
	cin>>n>>w>>l;
	for(i=0;i<n;i++)
	{
		cin>>h>>r;
		v.push_back(make_pair(h,r));
	}
	llow=0;
	hhigh=pow(10,18);
	while(llow<hhigh)
	{
		mid = llow + (hhigh-llow)/2;
		unsigned long long sum=0;
		for(i=0;i<n;i++)
		{
			k=(v[i].first) + (mid*v[i].second);
			if(k>=l)
				sum+=k;
			if(sum>=w)
				break;
		}
		if(sum>=w)
			hhigh=mid;
		else
			llow=mid+1;
	}
	cout<<min(llow,hhigh);
	return 0;
}
