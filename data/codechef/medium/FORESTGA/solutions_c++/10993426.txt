#include <bits/stdc++.h>

using namespace std;

#define ll				long long int
#define pb				push_back
#define mp				make_pair
#define MOD				1000000007
#define F				first
#define S				second

ll pow_mod(ll a, ll b) {
	ll res = 1;
	while(b) {
		if(b & 1)
			res = (res * a);
		a = (a * a);
		b >>= 1;
	}
	return res;
}

int main(){

	ios_base::sync_with_stdio(0);

	unsigned ll n,w,l;
	cin >> n >> w >> l;

	unsigned ll a[n][2];

	for(int i = 0;i<n;i++)
		cin >> a[i][0] >> a[i][1] ;

	unsigned ll lo = 0;
	unsigned ll hi = 1e18;
	unsigned ll best = 1e18;

	while(lo < hi){

		ll mid = (lo + hi) >> 1;
		ll v = 0;

		for(int i = 0;i<n;i++){
			if(a[i][0] + mid*a[i][1] >= l){
				v += a[i][0] + mid*a[i][1];
				if(v >= w)
					break;
			}
		}

		if(v >= w){
			hi = mid;
			best = mid;
		}
		else
			lo = mid + 1;

	}
	cout << best << "\n";

	return 0;
}
