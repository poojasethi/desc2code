#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll h[100005],r[100005];
int main()
{
    ll n,w,l,i,j;
    scanf("%lld %lld %lld",&n,&w,&l);
    for(i=0;i<n;i++)
    {
        scanf("%lld %lld",&h[i],&r[i]);
    }
    ll low=0,high=1e18+5;
    double mid,tmp,sum;
    while(low<=high)
    {
        mid=(high+low)/2;
        sum=0;
        for(i=0;i<n;i++)
        {
            tmp=h[i]+r[i]*mid;
            if(tmp>=l)
                sum+=tmp;
            if(sum>=w)
                break;
        }
        if(sum>=w)
            high=mid-1;
        else
            low=mid+1;
    }
    printf("%lld",low);
    return 0;
}
