#include <bits/stdc++.h>

using namespace std;

#define MAXM 1e18;

long long int n,i,ans,max1,g,w,q,h,m;
pair<long long,long long> a[1000005];

int main()
{
cin>>n>>w>>q;
    max1=0;
    ans=0;
    for(i=0;i<n;i++)
    {
        cin>>h>>m;
        a[i].first=h;
        a[i].second=m;
    if(h>=q)
        ans=ans+h;
    }
    if(ans>=w)
    {
        printf("0\n");
        return 0;
    }

    //cout<<max1;
   long long int l=-1;
   long long int r=MAXM;
    while(l<r)
    {
      long long int mid=(l+r)/2;
        ans=0;
        for(i=0;i<n;i++)
        {
            g=a[i].first+mid*a[i].second;
            if(g>=q)
            {
                ans=ans+g;
            }
            if(ans>=w)
                break;
        }
      //  cout<<ans<<"ans"<<"\n";
        if(ans>=w)
        {
        r=mid;
        }
        else
        {
            l=mid+1;
        }
    }
       printf("%lld\n",r);
    return 0;
}
