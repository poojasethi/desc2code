#include <bits/stdc++.h>
using namespace std;

#define MAX 100002
long long L, W , N;
long long  H[MAX], R[MAX];


int main() {
    
    ios_base::sync_with_stdio(false);
    cin>>N>>W>>L;
    
    for (int i = 0 ; i < N; i++) {
        cin>>H[i]>>R[i];
    }
    
    long long l = 0;
    long long r = 1000000000000000002LL ;
    
    
    while ( l <= r ) {
        long long  mid = (l+r)/2;
        long long temp = 0;
        long long wt_cut = 0;
        bool flag = false;
        
        for (int i = 0; i < N; i++) {
            temp =  (H[i] + ( R[i] * mid) );
            if (temp >= L ) {
                wt_cut += temp;
            }
            
            if (wt_cut >= W) {
                flag = true;
                break;
            }
        }
        
        if (flag) {
            r = mid - 1;
        } else {
            l = mid + 1;
        }
        
    }
    
    cout<<l<<endl;
	return 0;
}
