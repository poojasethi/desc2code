#include <cmath>
#include <cstdio>
#include <algorithm>


int check(float n)
        {
            return (n-(int)n > 0.0 || n<0.0)?1:0;
        }


using namespace std;

int main()
{
    int i,t,n,h[51],d[51];
    int res;
    float sum_h,sum_d,s,x;
    scanf("%d",&t);

    while(t--)
    {
        scanf("%d",&n);

        for(i=0;i<n;i++)
            scanf("%d",&h[i]);

        for(i=0;i<n;i++)
            scanf("%d",&d[i]);

        sum_h=0.0;
        sum_d=0.0;

        for(i=0;i<n;i++)
            sum_h+=h[i];

        for(i=0;i<n;i++)
            sum_d+=d[i];

        if(n==1)
        {
            if(h[0]>=d[0])
                res=h[0]-d[0];
            else
                res=-1;
            goto print;
        }
        if(n==2)
        {
            if(sum_h==sum_d)
                res=max(h[0],d[0])-min(h[0],d[0]);
            else
                res=-1;
            goto print;
        }

        else
        {
            s=(sum_d-sum_h)/(n-2);
            if(check(s))
                {res=-1;
                goto print;}
            for(i=0;i<n;i++)
            {
                x=0.5*(s+h[i]-d[i]);
                if(check(x))
                    {res=-1;
                    goto print;}
            }
            res=s;

        }

            print:
            printf("%d\n",res);
    }
}
