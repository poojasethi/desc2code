#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,t,i,a[51],b[51],steps,final_sum,initial_sum;
    scanf("%d",&t);
    while(t--)
    {
        final_sum=0;initial_sum=0;
        scanf("%d",&n);
        for(i=0;i<n;i++)
            {
                scanf("%d",&a[i]);
                initial_sum+=a[i];
            }
        for(i=0;i<n;i++)
           {
               scanf("%d",&b[i]);
               final_sum+=b[i];
           }
        if(n==2)
        {
            if(final_sum==initial_sum)
                printf("%d\n",abs(b[0]-a[0]));
            else
                printf("-1\n");
        }
        else
        {
            if((final_sum-initial_sum)%(n-2))
                steps=-1;
            else
            {
                steps=(final_sum-initial_sum)/(n-2);
                for(i=0;i<n;i++)
                {
                    if((steps+a[i]-b[i])%2!=0 ||steps+a[i]-b[i]<0)
                    {
                        steps=-1;
                        break;
                    }
                }
            }
            printf("%d\n",steps);
        }
    }
    return 0;
}
