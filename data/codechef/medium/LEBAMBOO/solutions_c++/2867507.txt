#include<cstdio>
using namespace std;
int main()
{
	int t,src[51],dest[51],i,j,n,sum_src,sum_dest;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		sum_src=sum_dest=0;
		//printf(" n = %d   ",n);
		for(i=0;i<n;i++)
		{
			scanf("%d",&src[i]);
			sum_src+=src[i];
			//printf(" %d  %d  ",sum_dest,sum_src);
		}
		for(i=0;i<n;i++)
		{
			scanf("%d",&dest[i]);
			sum_dest+=dest[i];
			//printf(" %d  %d  ",sum_dest,sum_src);
		}
		//printf(" %d  %d  ",sum_dest,sum_src);
		if(n==1)
		{
			if(sum_dest > sum_src)
			{
				printf("-1\n");
			}
			else
			{

					printf("%d\n",sum_src-sum_dest);
			}
		}
		else if(n==2)
		{
			if(sum_dest==sum_src)
			{
				int diff=src[0]-dest[0];
		//		printf("diff = %d  ",diff);
				if(diff<0)
					printf("%d\n",-diff);
				else
					printf("%d\n",diff);
			}
			else
				printf("-1\n");
		}
		else
		{
			int s,diff,x=0;
			diff=sum_dest-sum_src;
			if(diff%(n-2)!=0)printf("-1\n");
			else if(diff<0)printf("-1\n");
			else
			{
				s=diff/(n-2);
				for(i=0;i<n;i++)
				{
					if(((src[i]+s-dest[i])%2==0)&&(src[i]+s-dest[i])/2>=0)
						continue;
					else
						break;
				}
				if(i==n)
					printf("%d\n",s);
				else
					printf("-1\n");
			}
		}

	}
return 0;
}
