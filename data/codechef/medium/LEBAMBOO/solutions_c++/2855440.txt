#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;
 
int main() {
	int hi[51],di[51],diff[51],sum=0;
	int t,tmp1,tmp2,n;
	scanf("%d",&t);
	while(t--)
	{
		sum=0;
		scanf("%d",&n);
		for(int i=0;i<n;i++)
		scanf("%d",&hi[i]);
		for(int i=0;i<n;i++)
		scanf("%d",&di[i]);
		for(int i=0;i<n;i++)
		{diff[i]=-hi[i]+di[i];sum+=diff[i];}
		int ans=0;
		if(n==1)
		{
			if(diff[0]>0){ans=-1;}
			else {ans=diff[0]*(-1);}
		}
		if(n==2)
		{
			if(diff[0]+diff[1]==0)
			ans=abs(diff[0]);
			else 
			ans=-1;
		}
		if(n>2)
		{
		 if(sum<0)
		 	ans=-1;
		 else
		 {
		 	if(sum%(n-2))
		 		ans=-1;
			else
			{
				int counter=sum/(n-2);
				while(counter)
				{
					int minvalue=diff[0];
					int minindex=0;
					for(int i=0;i<n;i++)
						if(diff[i]<minvalue)
						{
							minvalue=diff[i];
							minindex=i;
						}
					diff[minindex]+=1;
					for(int i=0;i<n;i++)
						if(i!=minindex)
							diff[i]-=1;
					counter--;
				}
				int flag=0;
				for(int i=0;i<n;i++)
				{
					if(diff[i]!=0){flag=1;break;}
				}
				if(flag)ans=-1;
				else ans=sum/(n-2);
			}
		 }
		}
		printf("%d\n",ans);
	}
	return 0;
}
