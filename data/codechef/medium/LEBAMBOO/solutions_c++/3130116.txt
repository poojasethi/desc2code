#include <iostream>
#include <cmath>
using namespace std;

int hi[50], di[50];

int main() {
    int t;
    cin >> t;
    while(t--) {
        int n;
        cin >> n;
        if(n==2) {
            cin >> hi[0] >> hi[1];
            cin >> di[0] >> di[1];
            int r1, r2;
            r1 = hi[0]-di[0];
            r2 = di[1]-hi[1];
            if(r1==r2) cout << abs(1.0*r1) << endl;
            else cout << -1 << endl;
        } else {
            int h=0, d=0;
            for(int i=0;i<n;++i) {
                cin >> hi[i];
                h+=hi[i];
            }
            for(int i=0;i<n;++i) {
                cin >> di[i];
                d+=di[i];
            }
            int s = (d-h)/(n-2);
            int check = abs(n-2.0);
            if(((d-h)%check)==0 && s>=0) {
                int xi;
                bool flag = true;
                int sum = 0;
                for(int i=0;i<n;++i) {
                    xi = s+hi[i]-di[i];
                    if(xi/2<0 || xi%2 || xi/2>s) {
                        flag = false;
                        break;
                    }
                    sum += xi/2;
                }
                if(flag && sum==s) cout << s << endl;
                else cout << -1 << endl;
            } else cout << -1 << endl;
        }
    }
    return 0;
}
