#include<cstdio>
using namespace std;
int main()
{
    int t,i,n,f,s,ans,h[55],d[55],g[55];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=0;i<n;i++)
        scanf("%d",&h[i]);
        for(i=0;i<n;i++)
        scanf("%d",&d[i]);
        if(n==1)
        {
            if(h[0]>=d[0])
            printf("%d\n",h[0]-d[0]);
            else printf("-1\n");
        }
        else if(n==2)
        {
            ans=s=0;
            for(i=0;i<n;i++)
            s+=h[i]-d[i];
            if(s==0)
            {
                ans=h[0]-d[0];
                printf("%d\n",(ans<0)?ans*-1:ans);
            }
            else printf("-1\n");
        }
        else
        {
            ans=s=0;
            for(i=0;i<n;i++)
            {g[i]=d[i]-h[i];s+=g[i];}
            if(s<=0 && s%(n-2)) printf("-1\n");
            else
            {
                ans=s/(n-2);f=0;
                for(i=0;i<n;i++)
                {
                    if(ans<g[i] || (ans-g[i])%2)
                    {f=1;break;}
                }
                if(f) printf("-1\n");
                else printf("%d\n",ans);
            }
        }
    }
    return 0;
}
