using namespace std;

int main(){
	int t,n,arr[51],num;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(int i=0;i<n;i++) scanf("%d",arr+i);
		for(int i=0;i<n;i++){
			scanf("%d",&num);
			arr[i]=num-arr[i];
		}
		if(n==1){
			if(arr[0]<=0) printf("%d\n",-arr[0]);
			else printf("-1\n");
		}
		else if(n==2){
			if(arr[0]+arr[1]==0) printf("%d\n",abs(arr[0]));
			else printf("-1\n");
		}
		else{

			int total = 0;
			for(int i=0;i<n;i++) total+= arr[i];
			if(total>=0 && total%(n-2)==0){
				total = total/(n-2);
				bool iscorrect = true;
				int recheck=0;
				for(int i=0;i<n && iscorrect;i++){
					int val = total-arr[i];
					//cout<<val<<" ";
					if(val<0 || val%2 || val/2>total)
						iscorrect=false;
					recheck+=val/2;
				}
				//cout<<endl;
				if(iscorrect && recheck==total)printf("%d\n",total);
				else printf("-1\n");
			}
			else printf("-1\n");
		}
	}
	return 0;
}
