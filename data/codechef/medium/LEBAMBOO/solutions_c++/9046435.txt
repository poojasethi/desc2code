#include<bits/stdc++.h>
using namespace std;
int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
              int n,sumh=0,sumd=0;
              scanf("%d",&n);
              int d[n],h[n];
              for(int i=0;i<n;i++){
                   scanf("%d",&h[i]);
                   sumh+=h[i];
              }
                for(int i=0;i<n;i++){
                   scanf("%d",&d[i]);
                   sumd+=d[i];
                }
            if(n==1)
            {
                        if(h[0]<d[0])
                        printf("-1\n");
		else
		printf("%d\n",h[0]-d[0]);
 
		continue;
            }
            if(n==2)
            {
              if(abs(sumh)==abs(sumd))
                                    printf("%d\n",abs(h[0]-d[0]));
              else
                        printf("-1\n");
 
              continue;
            }
 
if(sumd-sumh<0)
{
            printf("-1\n");
            continue;
}
if((sumd-sumh)%(n-2)==0)
{
   int y, x=(sumd-sumh)/(n-2);
   bool flag=true;
   for(int i=0;i<n;i++)
   {
     y=x+h[i]-d[i];
     if(y&1 || y<0)
     {
                 flag=false;
                 break;
     }
   }
   if(flag)
            printf("%d\n",x);
   else
            printf("-1\n");
}
else
            printf("-1\n");
 
 
  }
  return 0;
}
