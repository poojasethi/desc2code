#define ll long long

#define ins(x) scanf("%s",&x)
#define in1(x) scanf("%d",&x)
#define in2(x,y) scanf("%d %d",&x, &y)
#define in3(x,y,z) scanf("%d %d %d",&x, &y, &z)
#define printc(s) printf("%c\n",s);
#define prints(s) printf("%s\n",s);

#define newline() printf("\n")
#define max(a,b) a>b?a:b
#define min(a,b) a<b?a:b

int main()
{
	int N;
	ll sumh=0, sumd=0;
	int H[100]; int D[100];
	int T; in1(T);
	while(T--){
	in1(N);
	sumh=sumd=0;
	for(int i=0;i<N;++i)
	{
		in1(H[i]); sumh+=H[i];
	}
	for(int i=0;i<N;++i)
	{
		in1(D[i]); sumd+=D[i];
	}
	int diff=sumd-sumh;
	if(N==1)
	{
		if(H[0]<D[0]) printf("-1\n");
		else printf("%d\n",H[0]-D[0]);
	}
	else if(N==2)
	{
		if(H[0]-D[0] == D[1]-H[1]) printf("%d\n",abs(H[0]-D[0]));
		else printf("-1\n");
	}
	else
	{
		if(diff<0) printf("-1\n");
		else if(diff%(N-2)==0)
		{
			int ans=diff/(N-2);
			int i=0;
			for(i=0;i<N;++i)
			{
				int y=H[i]-D[i]+ans;
				if(y<0 || y&1 ) {printf("-1\n");break;}
			}
			if(i==N) printf("%d\n",ans);
		}
		else printf("-1\n");
	}

	}
return 0;
}
