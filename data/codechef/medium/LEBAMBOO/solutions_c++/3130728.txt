#include <cstdio>

#define si(a) scanf("%d", &a)
#define for_range(a) for(int i=0;i<a;i++)
#define abso(a) (a)>0?a:-(a)

using namespace std;

int main()
{
	int t, d[50], h[50], dSum, hSum, n;
	int s, x;
	si(t);
	while(t--)
	{
		dSum = hSum = 0;
		si(n);
		for_range(n) {
			si(h[i]);
			hSum += h[i];
		}
		for_range(n) {
			si(d[i]);
			dSum += d[i];
		}
		if(n == 2 && hSum-dSum == 0)
			printf("%d\n", abso(h[0]-d[0]));
		else if(n > 2) {
			s = dSum - hSum;
			if(s < 0 || s%(n-2)!=0) {
				printf("-1\n");
			} else {
				bool shit = true;
				s /= (n-2);
				for_range(n)
				{
					x = s + h[i] - d[i];
					if(x < 0 || x%2 != 0) {
						printf("-1\n");
						shit = false;
						break;
					}
				}
				if(shit)
					printf("%d\n", s);
			}
			
		} else {
			if(h[0] >= d[0])
				printf("%d\n", h[0] -d[0]);
			else
				printf("-1\n");
		}
	}
	return 0;
}
