#include<stdio.h>
#include<iostream>
#include<stdlib.h>
using namespace std;
int main()
{
    int t,n,h[52],d[52],i,sum,s,ch,a[52],temp;
    cin >> t;
    while(t--)
    {
        ch=0;
        sum=0;
        cin >> n;
        for(i=0;i<n;i++)
            cin >> h[i];
        for(i=0;i<n;i++)
            cin >> d[i];
        for(i=0;i<n;i++)
        {
            sum=sum+(d[i]-h[i]);
            a[i]=d[i]-h[i];
        }

        if(n==2)
        {
            if((h[0]-d[0])==-1*(h[1]-d[1]))
                cout << abs(h[1]-d[1]) << "\n";
            else
                cout << "-1\n";
        }
        else
        {
            if(sum%abs(n-2)==0 && sum/(n-2)>=0)
            {
                s=sum/(n-2);
                for(i=0;i<n;i++)
                {

                    temp=s-a[i];
                    if(temp%2!=0 || temp<0)
                    {
                        ch=1;
                        cout << "-1\n";
                        break;
                    }
                }
                if(ch==0)
                {
                    cout << s << "\n";
                }
            }
            else
                cout << "-1\n";
        }
    }
    return 0;
}
