    #include <iostream>
    using namespace std;
    const int LRG = (int)1e9;
     
    int t;
     
    int main() {
    cin >> t;
    while (t--) {
    int n;
    bool disaster = false;
    cin >> n;
    int h[50];
    int d[50];
    int c[50];
    for (int i = 0; i < n; i++)
    cin >> h[i];
    for (int i = 0; i < n; i++)
    cin >> d[i];
    int res = LRG;
    for (c[0] = 0; c[0] <= 50*50 && !disaster; c[0]++) {
    int sum = c[0];
    for (int i = 1; i < n && !disaster; i++) {
    int x = d[0]-h[0];
    int y = d[i]-h[i];
    if ((x-y)&1) {
    disaster = true;
    break;
    }
    c[i] = (x-y)/2+c[0];
    sum += c[i];
    }
    bool ok = true;
    for (int i = 0; i < n; i++)
    if (h[i]+sum-c[i]-c[i] != d[i] || c[i] < 0) {
    ok = false;
    break;
    }
    if (ok)
    res = min(res, sum);
    }
    if (disaster)
    cout << -1;
    else
    cout << (res == LRG ?-1 :res);
    cout << "\n";
    }
    return 0;
    }
