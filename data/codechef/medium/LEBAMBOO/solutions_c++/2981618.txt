#include<iostream>
#include<vector>
#define MAX 1000000
using namespace std;
int s[1000];
int t[1000];
int d[1000];
int soln[1000];
int main()
{
	int n,test;
	cin>>test;
	while(test--)
	{
		int sum=0;
		bool nosoln=false;
		int ans=0;
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>s[i];
		for(int i=0;i<n;i++)
			cin>>t[i];
		for(int i=0;i<n;i++)
		{
			d[i]=t[i]-s[i];
			sum+=d[i];
		}
		if(n==2)
		{
			if(sum==0)
			{
				ans=max(d[0],d[1]);
			}
			else
			{
				nosoln=true;
			}
		}
		else
		{
			if(sum%(n-2)!=0)
			{
				nosoln=true;
			}
			else
			{
				sum/=(n-2);
				for(int i=0;i<n;i++)
				{
					if((sum-d[i])%2==0&&(sum-d[i])>=0)
					{
						soln[i]=(sum-d[i])/2;
					}
					else
					{
						nosoln=true;
						break;
					}
				}
				for(int i=0;i<n;i++)
				{
					ans+=soln[i];
				}
			}
		}/*
		for(int i=0;i<n;i++)
			cout<<s[i]<<" ";
		cout<<"\n";
		for(int i=0;i<n;i++)
			cout<<t[i]<<" ";
		cout<<"\n";
		for(int i=0;i<n;i++)
			cout<<d[i]<<" ";
		cout<<"\n";
		for(int i=0;i<n;i++)
			cout<<soln[i]<<" ";
		cout<<"\n";*/
		if(nosoln)
			cout<<"-1\n";
		else
			cout<<ans<<endl;
		for(int i=0;i<1000;i++)
		{
			soln[i]=s[i]=d[i]=t[i]=0;
		}
	}
	return 0;
}
