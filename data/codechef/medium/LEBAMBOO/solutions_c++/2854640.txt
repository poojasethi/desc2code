#include <iostream>
#include <cmath>
using namespace std;
int abs(int a) {if(a < 0) return -a; else return a;}
int gcd(int a, int b){
    if(b == 0)
        return a;
    else    return gcd(b, a%b);
}

void solve() {
    int n;
    cin >> n;
    int H[n], D[n];
    int d = 0, h = 0;
    for(int i=0; i<n; i++) {
        cin >> H[i];
        h += H[i];
    }
    for(int i=0; i<n; i++) {
        cin >> D[i];
        d += D[i];
    }
    int del = n-2;

    if(n == 2) {
        if(h == d) {
            cout << abs(D[0]-H[0]) << endl;
        }else {
            cout << -1 << endl;
        }
        return ;
    }
    if(n == 1) {
        if(H[0] < D[0])
            cout << -1 << endl;
        else
            cout << H[0]-D[0] << endl;
    }
    else {
        if((d-h)%(abs(del))){
            cout << -1 << endl;
            return ;
        }
        int step = (d-h)/(del);
        if(step < 0) {
            cout << -1 << endl;
            return ;
        }

        int sum = 0;

        for (int i = 0; i < n; ++ i) {

            int xi2 = H[i]-D[i] + step;
            if (xi2 % 2 != 0 || xi2 / 2 < 0 || xi2 / 2 > step) {
                cout << -1 << endl;
                return;
            }
            sum += xi2 / 2;
        }

        if (sum == step) {
            cout << step << endl;
        } else {
            cout << -1 << endl;
        }
    }
}
int main() {
    int t;
    cin >> t;
    while(t--) {
        solve();
    }
    return 0;
}
