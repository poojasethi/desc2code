#include <stdio.h>
#include <stdlib.h>
 
int getpossible(int *h,int ans,int n)
{
int count=ans;
int i=0;
while(i<n)
{
if((((h[i]+ans)%2)==0)&&(((h[i]+ans)/2)>=0))
{
count=count-((h[i]+ans)/2);
if(count<0)
return -1;
}
else
return -1;
i++;
}
if(count>0)
return -1;
else
return ans;
}
int main()
{
int t;
scanf("%d",&t);
int *h;
int *d;
int n;
int i;
int j;
int sum;
int ans;
for(i=0;i<t;i++)
{
ans=0;
sum=0;
scanf("%d",&n);
h=(int*)calloc(n,sizeof(int));
d=(int*)calloc(n,sizeof(int));
for(j=0;j<n;j++)
scanf("%d",&h[j]);
for(j=0;j<n;j++)
scanf("%d",&d[j]);
for(j=0;j<n;j++)
h[j]=h[j]-d[j];
for(j=0;j<n;j++)
sum=sum+h[j];
if(n>2)
{
if(sum>0||((-1*sum)%(n-2))!=0)
ans=-1;
else
{
ans=(sum/(n-2))*(-1);
ans=getpossible(h,ans,n);
}
}
else
{
if(n==1)
{
if (sum<0)
ans=-1;
else
ans=sum;
}
else
{
if(sum!=0)
ans=-1;
else
{
ans=h[0];
if(ans<0)
ans=ans*(-1);
}
}
}
printf("%d",ans);
if(i<t-1)
printf("\n");
}
}
