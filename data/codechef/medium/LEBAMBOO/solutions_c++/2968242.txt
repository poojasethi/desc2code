using namespace std;

int A[50], B[50];

int main(void){
	//freopen("input.txt", "r", stdin);
	int T;
	cin >> T;
	while(T--){
		int N;
		cin >> N;
		for(int i = 0; i < N; i++) cin >> A[i];
		for(int i = 0; i < N; i++) cin >> B[i];

		bool poss = true;
		bool parity = (A[0] - B[0]) % 2;
		for(int i = 1; i < N; i++){
			bool next = (A[i] - B[i]) % 2;
			if(next != parity) poss = false;
		}

		int x;
		if(poss){
			int sum = 0;
			for(int i = 0; i < N; i++) sum += (B[i] - A[i]);
			if(N == 2){
				x = B[0] == A[0] ? 0 : 1;
				if(sum != 0) poss = false;
			}
			else{
				if(sum % (N-2)) poss = false;
				if(poss){
					x = sum / (N-2);
					for(int i = 0; i < N; i++)
						if(x < B[i] - A[i]) poss = false;
				}
			}
		}
		if(poss) cout<<x<<endl;
		else cout<<"-1"<<endl;

	}


	return 0;
}
