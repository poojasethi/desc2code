int main()
{
    std::ios_base::sync_with_stdio (false);
    #ifndef ONLINE_JUDGE
        freopen("input.txt","r",stdin);
    #endif
    int t,n,c,d,s,ans;
    bool flag;
    cin>>t;
    while(t--){
        cin>>n;
        int a[n],b[n];
        c=0;d=0;
        REP(i,n){cin>>a[i];c+=a[i];}
        REP(i,n){cin>>b[i];d+=b[i];}
        if(n==1){
            if(b[0]-a[0]>0)ans=-1;
            else ans=a[0]-b[0];
        }
        else if(n==2){
            if(b[0]-a[0]+b[1]-a[1]==0)ans=abs(b[0]-a[0]);
            else ans=-1;;
        }
        else{
            if((d-c)%(n-2)||d<c)ans=-1;
            else{
                s=(d-c)/(n-2);
                flag=1;
                REP(i,n){
                    if(s+a[i]-b[i]<0||(s+a[i]-b[i])%2!=0)flag=0;
                }
                if(flag)ans=s;
                else ans=-1;
            }
        }
        cout<<ans<<endl;
    }
return 0;
}
