#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int t,n;
	int a[50],b[50];
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		for(int i=0;i<n;i++)
		{
			cin>>b[i];
		}
		bool ans=true;
		bool par=(a[0]-b[0])%2;
		for(int i=1;i<n;i++)
		{
			bool next=(a[i]-b[i])%2;
			if(next!=par) ans=false;
		}
		int q;
		if(ans)
		{
			int sum=0;
			for(int i=0;i<n;i++)
			{
				sum+=b[i]-a[i];
			}
				if(n==2)
				{
					 q= b[0]==a[0]?0:1;
					if(sum!=0)ans=false;
				}
				else
				{
					 if (sum%(n-2)) ans=false;
						if (ans) 
						{
							//cout<<sum<<"........."<<endl;
							q= sum/(n-2);
							for (int i=0; i<n; i++) if (q<b[i]-a[i]) ans=false;
						}
				}
			
		}
		
		 if (ans) printf("%d\n",q);
		else printf("-1\n");

	}
	return 0;
}
