int check(int *a,int n)
{
        int i,j,m,k=0,p=0;
        for(i=0;i<n;i++)
        {
                if(a[i]==0)
                        k++;
                if(a[i]<0)
                        p++;
        }
        if(k==n)
                return 1;
        else
                return 0;
}

void solve(int *a,int n)
{
        int i,j=0,m,k=a[0],p;
        for(i=0;i<n;i++)
        {
                if(a[i]<k)
                        k=a[i];
        }
        //return k;
        for(i=0;i<n;i++)
        {
                if(j==0 && a[i]==k)
                {
                        a[i]=a[i]+1;
                        j=1;
                }
                else
                {
                        a[i]=a[i]-1;
                }
        }
}
