#include<stdio.h>
#include<stdlib.h>

#define max 1000000

unsigned long long int ans[1000001] ;
void seivetotient() ;

void seiveandtotient(){

	int index ;

	for(index=2;index<=max;index++)
		ans[index] = index ;
	
	int factor ;

	for(index=2;index<=max;index++){
	
		if(ans[index]==index){
		
			for(factor=index;factor<=max;factor+=index)
				ans[factor] = (ans[factor]/index)*(index-1) ;
		
		}
	
	}

	ans[0] = 0 ;
	ans[1] = 0 ;

}

int main(){

	int index ;

	int N ;
	int T ;


	seiveandtotient() ;

	//printf("%d\n",ans[1000]) ;

	for(index=2;index<=max;index++)
		ans[index] += ans[index-1] ;

	

	scanf("%d",&T) ;

	while(T--){
	
		scanf("%d",&N) ;
		printf("%llu\n",ans[N]) ;
	}
}
