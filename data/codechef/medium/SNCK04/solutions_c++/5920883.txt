#include<cstdio>
#define ll long long int
using namespace std;
ll phi[1000001];
ll ans[1000001];
 int main()
 {
     ll i,n,j;
     int t;
     for(i=2;i<=1000000;i++)
     {
         phi[i]=i;
     }
     for(i=2;i<=1000000;i++)
     {
         if(phi[i]==i)
         {
         for(j=i;j<=1000000;j=j+i)
         {
             phi[j]=phi[j]-(phi[j]/i);
         }
         }
     }
     ans[1]=0;
     for(i=2;i<=1000000;i++)
        ans[i]=ans[i-1]+phi[i];
        scanf("%d",&t);
        while(t>0)
        {
            t--;
            scanf("%lld",&n);
            printf("%lld\n",ans[n]);
        }
 }
