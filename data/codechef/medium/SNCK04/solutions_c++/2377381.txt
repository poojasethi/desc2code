#include<stdio.h>
#include<math.h>
#include<string.h>
#define llu long long int
using namespace std;
llu res[1000010],ans[1000010];
int main()
{
	llu i,j,n,t;
	memset(res,0,sizeof(res));
	memset(ans,0,sizeof(ans));
	 i=2;
	while(i<=1000001)
	{
		if(res[i]==0)
		{
			res[i]=i-1;
			for(j=2;i*j<=1000000;j++)
			{
				if(res[i*j]==0 && res[j]!=0)//This Euler Totient function Property
				{
					if(j%i)
					{
						res[i*j]=res[i]*res[j];
					}
					else
					{
						res[i*j]=res[j]*i;
					}
				}
			}
		}
		i++;
	}
	for(i=1;i<=1000000;i++)
	{
		ans[i]=ans[i-1]+res[i];
	}
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		printf("%lld\n",ans[n]);
	}
}
