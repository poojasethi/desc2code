    #include<cstdio>
    #include<cstring>
    #define maxn 1000000
    int n,F[1000005];
    long long s[1000005];
     
     
    inline void Prepare()
    {
    for (int i=2;i<=maxn;++i) if (!F[i])
    {
    F[i]=i-1;
    for (int j=i*2;j<=maxn;j+=i)
    {
    if (!F[j]) F[j]=j;
    F[j]=F[j]/i*(i-1);
    }
    }
    s[1]=F[1];
    for (int i=2;i<=maxn;++i) s[i]=s[i-1]+F[i];
    }
    int main()
    {
    Prepare();
    scanf("%d",&n);
    for (int i=1;i<=n;++i)
    {
    int x;
    scanf("%d",&x);
    printf("%lld\n",s[x]);
    }
    return 0;
    }
