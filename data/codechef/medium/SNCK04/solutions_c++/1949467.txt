#include <cstdio>
using namespace std;

const int N = 1000005;
bool isPrime[N];
unsigned long long totient[N];

void sieveAndTotient() {
    for (int i = 0; i < N; ++i)
        totient[i] = i;

    isPrime[0] = isPrime[1] = false;
    isPrime[2] = true;
    totient[0]=totient[1]=0;
    totient[2]=1;
    for(int i=3;i<N;i++)
    {
        if(i%2!=0)  isPrime[i] = true;
        else
        {
            isPrime[i] = false;
            totient[i] -= totient[i] / 2;
        }
    }

    for (int i = 3; i < N; i += 2)
        if (isPrime[i]) {
            for (int j = i; j < N; j += i) {
                isPrime[j] = false;
                totient[j] -= totient[j] / i;
            }
            isPrime[i] = true;
        }
}

int main() {
    sieveAndTotient();
    for(int i=1;i<N;i++)
    {
        totient[i]+=totient[i-1];
    }
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%llu\n",totient[n]);
    }
    return 0;
}
