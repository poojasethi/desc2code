#include <iostream>

using namespace std;
int a[1000005];
long long int d[1000005];
int main()
{
    int t,n,i,j;
    cin>>t;
     for(i=1;i<=1000000;i++)
     {
        a[i]=i;
     }
        d[1]=0;
        for(i=2;i<=1000000;i++)
        {
            if(a[i]==i)
            {
            for(j=1;i*j<=1000000;j++)
            {
                a[i*j]=a[i*j]-(a[i*j]/i);
            }
            }
            d[i]=a[i]+d[i-1];
        }
    while(t--)
    {
        cin>>n;
        if(n<0)
        n=-n;
        else if(n==0)
        {
            cout<<0<<endl;
            continue;
        }
        cout<<d[n]<<endl;
    }
    return 0;
}
