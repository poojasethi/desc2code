#include <stdio.h>
#define N 1000003
#define LL long long
 
int Phi[N+1];
LL S[N+1];
 
int main()
{
	int i, j, t, n;
 
	for(i=2;i<=N;i++) Phi[i]=i;
	for(i=2;i<=N;i++)
	{
		if(Phi[i]==i)
		{
			for(j=i;j<=N;j+=i)
				Phi[j] = (Phi[j]/i)*(i-1);
		}
		S[i] = S[i-1]+Phi[i];
	}
 
	scanf("%d",&t);
	while(t--) scanf("%d",&n), printf("%lld\n",S[n]);
	return 0;
}
