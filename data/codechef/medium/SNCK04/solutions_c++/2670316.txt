#include <iostream>
using namespace std;

bool isprime[1000001];
long long int phi[1000001];

void find() {
  phi[0] = 0;
  phi[1] = 0;
  for(int i=2;i<1000001;++i) {
    isprime[i] = true;
    phi[i] = i;
  }
  for(int i=2;i<1000001;++i)
    if(isprime[i]) {
      phi[i] = i-1; //since all numbers are relatively prime to prime numbers
      for(int j=i+i;j<1000001;j+=i) {
	isprime[j] = false;
	phi[j] = (phi[j]/i)*(i-1);
      }
    }
  for(int i=1;i<1000001;++i) phi[i]=phi[i]+phi[i-1];
}

int main() {
  find();
  int t;
  cin >> t;
  while(t--) {
    int n;
    cin >> n;
    cout << phi[n] << endl;
  }
  return 0;
}
