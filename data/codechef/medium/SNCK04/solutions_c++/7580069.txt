#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef std::vector<int> vi;
typedef pair<int,int> ii;

vi primes;
int N=1000001;
bool isPrime[1000001];
ll phi[1000001];

void sieve_generator(){
    memset(isPrime,1,sizeof(isPrime));
    for(ll i=1;i<1000001;i++)
        phi[i]=i;
    isPrime[1]=0;
    phi[1]=0;
    for(ll i=2;i<1000001;i++){
        if(isPrime[i]){
            phi[i]--;
            for(ll j=2*i;j<1000001;j+=i){
                isPrime[j]=0;
                phi[j]*=i-1;
                phi[j]/=i;
            }
        }
    }
}

int main(){
    sieve_generator();
    for(ll i=2;i<1000001;i++)
        phi[i]+=phi[i-1] ;
    int t;scanf("%d",&t);
    while(t--){
        ll n;
        scanf("%lld",&n);
        ll ans=phi[n];
        printf("%lld\n",ans);
    }
}
