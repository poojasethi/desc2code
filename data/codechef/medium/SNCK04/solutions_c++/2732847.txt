#include<stdio.h>
int p[1000010];
int phi[1000010];
long long ans[1000010];
void findprimes()
{
     p[0]=1;
     p[1]=1;
     phi[1]=1;
     int i,j;
     for(i=2;i<=1000000;i++)
     phi[i]=i;
     for(i=2;;)
     {
           phi[i]=i-1;    
           for(j=2;i*j>=0&&i*j<=1000000;j++)
           {
              p[i*j]=1;
              phi[i*j]=(phi[i*j]/i)*(i-1);
           }
           i++;
           for(;i<=1000000&&p[i]!=0;i++);
           if(i==1000001)
           break;
     }
}
void findans()
{
     ans[1]=0;
     ans[2]=1;
     int i;
     for(i=3;i<=1000000;i++)
     ans[i]=ans[i-1]+phi[i];
}                                     
main()
{
      findprimes();
      findans();
      int t,n;
      scanf("%d",&t);
      for(;t--;)
      {
           scanf("%d",&n);
           printf("%lld\n",ans[n]);
      }
      return 0;
}
