#include <stdio.h>
#define S 1000005
#include <iostream>
using namespace std;

long long int a[S]={0};
int phi[S];
int main() 
{
   int i,j,t,c;
    for(i=2;i<S;i++)
        {
            phi[i]=i;
        }
    for(i=2;i<S;i++)
        {
            if(phi[i]==i) //This is the case when i is a PRIME. Check for yourself
                {
                 for(j=i;j<S;j+=i) //On all multiples of that prime number, do this
                    {
                        phi[j]=(phi[j]/i)*(i-1);  // same as saying  phi[j]*( 1 - (1/i) ). See Euler's Multiplication Formula 
                   
                    }   
                }
            a[i]=a[i-1]+phi[i];
        }

cin>>t;
while(t--) 
    {
        cin>>c;
        cout<<a[c]<<'\n';
    }

return 0;
}
