#include<stdio.h>
long long int phi[1000001];
main()
{	long long i,j,k,t,n;
	for(i=2;i<=1000000;i++)
		phi[i]=i;
	for(i=2;i<=1000000;i++)
	{	if(phi[i]==i)
		{	for(j=i;j<=1000000;j+=i)
				phi[j]=(phi[j]/i)*(i-1);
		}
	}
	for(i=2;i<=1000000;i++)
	{	phi[i]=phi[i-1]+phi[i];
	}
	scanf("%lld",&t);
	while(t--)
	{	scanf("%lld",&n);
		printf("%lld\n",phi[n]);
	}
	return 0;
}
