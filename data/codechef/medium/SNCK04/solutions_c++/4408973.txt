#include<iostream>
#include<stdio.h>
using namespace std;
#include<bitset>
#define LL long long int
#define MAXN 3000000
LL phi[MAXN + 1], prime[MAXN/10], sz;
LL SUM[MAXN];
bitset <MAXN + 1> mark;
int main()
    {
    for (int i = 2; i <= MAXN; i++ )
        {
        if(!mark[i])
        {
            phi[i] = i-1;
            prime[sz++]= i;
        }
        for (int j=0; j<sz && prime[j]*i <= MAXN; j++ )
        {
            mark[prime[j]*i]=1;
            if(i%prime[j]==0)
            {
                phi[i*prime[j]] = phi[i]*prime[j];
                break;
            }
            else phi[i*prime[j]] = phi[i]*(prime[j]-1 );
        }
       }
       SUM[0]=0,SUM[1]=0;
       for(int i=2;i<=MAXN;i++) SUM[i]= phi[i]+SUM[i-1];
       int t;
       scanf("%d",&t);
       while(t--)
       {
           LL n;
           scanf("%lld",&n);
           printf("%lld\n",SUM[n]);
       }
        //for(int i=0;i<=20;i++) cout<<i<<" "<<SUM[i]<<endl;
    }
