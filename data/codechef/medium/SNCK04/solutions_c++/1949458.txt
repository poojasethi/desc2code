#include <limits.h>
#include <cstdlib>
#include <cmath>
#include <cstdio>
using namespace std;

const int N = 1000005;
bool isPrime[N];
int totient[N];
unsigned long long ans[N];

void sieveAndTotient() {
    for (int i = 0; i < N; ++i)
        totient[i] = i;

    isPrime[0] = isPrime[1] = false;
    isPrime[2] = true;
    totient[0]=totient[1]=0;
    totient[2]=1;
    for(int i=3;i<N;i++)
    {
        if(i%2!=0)  isPrime[i] = true;
        else
        {
            isPrime[i] = false;
            totient[i] -= totient[i] / 2;
        }
    }

    for (int i = 3; i < N; i += 2)
        if (isPrime[i]) {
            if (i < INT_MAX)
                for (int j = i; j < N; j += i) {
                    isPrime[j] = false;
                    totient[j] -= totient[j] / i;
                }
            isPrime[i] = true;
        }
}

int main() {
    sieveAndTotient();
    for(int i=1;i<N;i++)
    {
        ans[i]=ans[i-1]+totient[i];
    }
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%llu\n",ans[n]);
    }
    return 0;
}
