def main():
    n = input()
    i=1
    smallest = float('inf')
    largest = 0
    while i<=n:
        c = map(int,raw_input().split())
        area = abs(c[0]*(c[3]-c[5])+c[2]*(c[5]-c[1])+c[4]*(c[1]-c[3]))/2.0
        if area<=smallest:
            mini=i
            smallest=area
        if area>=largest:
            maxi=i
            largest = area
        i+=1
    print "%d %d" %(mini,maxi)
main()
