def area(triangle):
    x1,y1,x2,y2,x3,y3=map(int,triangle)
    return abs((x1*y2-x2*y1)+(x2*y3-x3*y2)+(x3*y1-x1*y3))
	
t=int(raw_input())
large=small=large_area=0;small_area=1000000000;i=1
while(i<=t):
	triangle=raw_input().split()
	s=area(triangle)
	if(s<=small_area):
		small_area=s
		small=i
	if(s>=large_area):
		large_area=s
		large=i
