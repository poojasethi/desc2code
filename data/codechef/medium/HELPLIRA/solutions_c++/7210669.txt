#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{	
	int n,i,maxind,minind,x1,y1,x2,y2,x3,y3,ar,maxi,mini;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d %d %d %d %d %d",&x1,&y1,&x2,&y2,&x3,&y3);
		ar=(x1*y2 - x2*y1) + (x2*y3 - x3*y2) + (x3*y1 - x1*y3);
		ar=abs(ar);
		if(i==0)
		maxi=mini=ar,maxind=minind=1;
		else
		{
			if(ar>=maxi)
			maxi=ar,maxind=i+1;
			if(ar<=mini)
			mini=ar,minind=i+1;
		}
	}
	printf("%d %d\n",minind,maxind);
	return 0;
}
