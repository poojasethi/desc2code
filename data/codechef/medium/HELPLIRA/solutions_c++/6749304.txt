#include <iostream>
#include<cstdio>
using namespace std;
double area(int x1,int y1,int x2,int y2,int x3,int y3)
{
	double t;
	t=(x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2))/2;
	if(t<0)
	t=-t;
	return t;
}
int main() 
{
	int n,x1,y1,x2,y2,x3,y3,imn,imx;
	scanf("%d",&n);
	double ar,mn=10000001,mx=0;
	for(int i=1;i<=n;i++)
	{
		scanf("%d %d %d %d %d %d",&x1,&y1,&x2,&y2,&x3,&y3);
		ar = area(x1,y1,x2,y2,x3,y3);
		if(mn>=ar)
		{
			imn=i;
			mn=ar;
		}
		if(mx<=ar)
		{
			imx=i;
			mx=ar;
		}
	}
	printf("%d %d",imn,imx);
	return 0;
}
