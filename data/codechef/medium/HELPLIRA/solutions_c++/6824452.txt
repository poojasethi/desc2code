#include<bits/stdc++.h>
#define gc getchar_unlocked // for linux
//#define gc getchar        // for windows
#define LL long long
using namespace std;

int inp()   //Signed long long data type
{
	int n=0;
	int ch=gc();int sign=1;
	while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=gc();}
	while( ch >= '0' && ch <= '9' )
	n = (n<<3)+(n<<1) + ch-'0', ch=gc();
	return(n*sign);
}
void scanint(int &x)  // Unsigned int data type
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
void scanlong(long long &x)     //Unsigned long long data type
{
    register long long c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	int t,ax,ay,bx,by,cx,cy, indexMin, indexMax;
	scanint(t);
	double area,min,max;
	min=100000000;
	max=-1;
	for(int i=0;i<t;i++)
	{
		scanf("%d %d %d %d %d %d",&ax,&ay,&bx,&by,&cx,&cy);
		area=abs((ax*(by-cy)+bx*(cy-ay)+cx*(ay-by))/2);
		if(area>=max)
		{
			max=area;
			indexMax=i+1;
		}
		if(area<=min)
		{
			min=area;
			indexMin=i+1;
		}
	}
	printf("%d %d",indexMin, indexMax);
	return 0;
}
