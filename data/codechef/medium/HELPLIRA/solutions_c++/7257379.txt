#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<vector>
#include<map>
using namespace std;
int n,x[3],y[3];
int main()
{
	scanf("%d",&n);
	vector<int> A(n+1);
	for(int i=1;i<=n;i++){
		scanf("%d %d %d %d %d %d",&x[0],&y[0],&x[1],&y[1],&x[2],&y[2]);
		A[i]=x[0]*y[1]+x[1]*y[2]+x[2]*y[0]-x[1]*y[0]-x[2]*y[1]-x[0]*y[2];
		if(A[i]<0)A[i]*=-1;		
	}
	int sm=n,lr=n;
	for(int i=n-1;i>0;i--){
		if(A[sm]>A[i])sm=i;
		if(A[lr]<A[i])lr=i;
	}
	cout<<sm<<" "<<lr<<"\n";
	return 0;
}
