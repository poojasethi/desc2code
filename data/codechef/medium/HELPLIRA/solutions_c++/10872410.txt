#include <iostream>

int main()
{ long long int n,i,j,x1,x2,x3,y1,y2,y3,maxa,mina,mini,maxi;
  int ar[100];
  scanf("%lld",&n);
  for(j=1;j<=n;j++)
  { 
    
    scanf("%lld %lld %lld %lld %lld %lld",&x1,&y1,&x2,&y2,&x3,&y3);
    
    ar[j]=x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2);
    if(ar[j]<0)
    ar[j]=(-1)*ar[j];
    
    
  }
  maxi=1;
  mini=1;
  mina=ar[1];
  maxa=ar[1];
  for(i=2;i<=n;i++)
  {if(ar[i]>=maxa)
   {maxa=ar[i];
   maxi=i;
   }
   if(ar[i]<=mina)
   {mina=ar[i];
   mini=i;
   }
  }
  printf("%lld %lld",mini,maxi);
  return 0;
}
