#include<iostream>
#include<math.h>
using namespace std;
int main()
{
    int n,ax,ay,bx,by,cx,cy,max,min,i;
    float area,maxa,mina;
    cin>>n;
        cin>>ax>>ay>>bx>>by>>cx>>cy;
        area=fabs(float((ax*(by-cy))+(bx*(cy-ay))+(cx*(ay-by)))*0.5);
        max=min=1;
        maxa=mina=area;

     for(i=2;i<=n;i++)
     {
         cin>>ax>>ay>>bx>>by>>cx>>cy;
         area=fabs(float((ax*(by-cy))+(bx*(cy-ay))+(cx*(ay-by)))*0.5);
        if(area>=maxa)
          {
              max=i;
              maxa=area;
          }
        if(area<=mina)
          {
              min=i;
              mina=area;
          }
     }
    cout<<min<<" "<<max;
}
