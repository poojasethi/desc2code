    #define gc getchar_unlocked
    #define hi int
    inline hi scanint()
    {
      hi n=0,neg=0,c=getchar_unlocked();
      while((c<'0'||c>'9')&&c!='-')c=getchar_unlocked();
      if(c=='-') {neg=1;c=getchar_unlocked();}
      while(c>='0'&&c<='9')n=(n<<3)+(n<<1)+c-'0',c=getchar_unlocked();
      if(neg) n=-n;
      return n;
    }
    int main()
    {
    hi n,i,x1,x2,x3,y1,y2,y3,pos_min,pos_max;
    hi a,b,c,d;
    hi ar;
    hi min=-1,max=-1;
    n=scanint();
    for(i=1;i<=n;i++)
    {
    x1=scanint(); y1=scanint();
    x2=scanint(); y2=scanint();
    x3=scanint(); y3=scanint();
    a=x1*(y2-y3); b=x2*(y3-y1); c=x3*(y1-y2);
    d=a+b+c;
    ar=(hi)abs((float)d);
    ar=(hi)ar/2;
    if(min==-1 && max==-1)
    {
    min=max=ar;
    pos_min=pos_max=i;
    }
    else
    {
    if(ar<=min)
    {
    min=ar; pos_min=i;
    }
    if(ar>=max)
    {
    max=ar; pos_max=i;
    }
    }
    }
    printf("%d %d\n",pos_min,pos_max);
    return 0;
    }
