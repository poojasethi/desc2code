int main(){
    int t, i, ax, ay, bx, by, cx, cy, temp, maxPos = 0, minPos = 0;
    scanf("%d", &t);
    scanf("%d %d %d %d %d %d", &ax, &ay, &bx, &by, &cx, &cy);
    int max = abs(ax*(by - cy) + bx*(cy - ay) + cx*(ay - by));
    int min = max;
    for (i=1; i<t; i++) {
        scanf("%d %d %d %d %d %d", &ax, &ay, &bx, &by, &cx, &cy);
        temp = abs(ax*(by - cy) + bx*(cy - ay) + cx*(ay - by));
        if (temp >= max) {
            max = temp;
            maxPos = i;
        }
        if (temp <= min) {
            min = temp;
            minPos = i;
        }
    }
    printf("%d %d", minPos+1, maxPos+1);

    return 0;
}
