#include<bits/stdc++.h>
using namespace std;
#include<math.h>
int main()
{
    int i,n,maxi=0,mini=0,max,min,ar;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
    int a,b,c,d,e,f,s;
    scanf("%d%d%d%d%d%d",&a,&b,&c,&d,&e,&f);
    ar=(a*(d-f)+c*(f-b)+e*(b-d))/2;
    if(ar<0)
    ar=ar*-1;
    //cout<<ar<<endl;
    if(i==0)
    max=min=ar;
    if(ar<=min)
    {
    mini=i+1;
    min=ar;
    }
    if(ar>=max)
    {
    maxi=i+1;
    max=ar;
    }
    }
    printf("%d %d\n",mini,maxi);
}
