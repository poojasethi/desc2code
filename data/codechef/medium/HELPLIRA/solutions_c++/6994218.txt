#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,al,ah,j;
    double arr[6],areal=0.0,areah=0.0,ans;
    cin>>n;
    for(j=0;j<n;j++)
    {
        for(i=0;i<6;i++)
        {
            cin>>arr[i];
        }
        ans=0.5*(abs(arr[0]*(arr[3]-arr[5])+arr[2]*(arr[5]-arr[1])+arr[4]*(arr[1]-arr[3])));
        if(j==0)
            areal=ans;
        if(ans>=areah)
            {
                areah=ans;
                ah=j+1;
            }
        if(ans<=areal)
            {
                areal=ans;
                al=j+1;
            }
    }
    cout<<al<<" "<<ah<<endl;
    return 0;
}
