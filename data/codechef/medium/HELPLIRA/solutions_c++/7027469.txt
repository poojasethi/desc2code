#include<iostream>
#include<cmath>
using namespace std;

int area( int x1,int y1,int x2,int y2,int x3,int y3)
{
  double a;
  a = (x1*(y2-y3) + x2*(y3-y1) + x3*(y1-y2))/2;

  if(a<0)
  {
     a = a*(-1);
  }

  return a;
}

int main()
{
   int n,i,j,pos_min,pos_max;
   double max,min;
   cin >> n;
   int a[n][6];double b[n];

   for(i=0;i<n;i++)
   {
     for(j=0;j<6;j++)
      {
         cin >> a[i][j];
      }
    }

   for(i=0;i<n;i++)
   {
     b[i] = area(a[i][0],a[i][1],a[i][2],a[i][3],a[i][4],a[i][5]);
   }

   max=0.000000;
   min=pow(10,20);
   
   for(i=0;i<n;i++)
   {
      if( b[i] >= max )
       {
           max = b[i];
           pos_max = i+1;
       }
 
      if( b[i] <= min )
       {  
           min = b[i];
           pos_min = i+1;
       }
    }
  
    cout << pos_min << " " << pos_max <<endl;
return 0;
}
