#include <iostream>
#include <bits/stdc++.h>
#include <limits.h>

using namespace std;

int main()
{
	int n;
	cin>>n;
	float maxim = INT_MIN, minim = INT_MAX,maxInd=0,minInd=0,cnt=0;  
	while(n--)
	{
		float x1,y1,x2,y2,x3,y3;
		cin>>x1>>y1>>x2>>y2>>x3>>y3;
		float a = 0.5*fabs((x1*y2-x2*y1)+(x2*y3-x3*y2)+(x3*y1-x1*y3));
		if(a>=maxim) {maxim = a; maxInd =cnt;}			
		if(a<=minim) {minim = a; minInd =cnt;}
		cnt++;
	}
	cout<<minInd+1<<" "<<maxInd+1<<endl;
	return 0;
}
