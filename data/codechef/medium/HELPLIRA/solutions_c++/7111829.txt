#include<iostream>
using namespace std;
#include<stdlib.h>
int main()
{
	int n,i,j,mini,maxi;
	float area,min,max;
	int ax,bx,cx,ay,by,cy;
	cin>>n;
	cin>>ax>>ay>>bx>>by>>cx>>cy;
	area=abs(ax*(by-cy)+bx*(cy-ay)+cx*(ay-by))/2;
	max=area;min=area;
	maxi=1;mini=1;
	for(i=2;i<=n;i++)
	{
		cin>>ax>>ay>>bx>>by>>cx>>cy;
		area=abs(ax*(by-cy)+bx*(cy-ay)+cx*(ay-by))/2;
		if(area>=max)
		{
			max=area;
			maxi=i;
		}
		if(area<=min)
		{
			min=area;
			mini=i;
		}
	}
	cout<<mini<<" "<<maxi;
	return 0;
}
