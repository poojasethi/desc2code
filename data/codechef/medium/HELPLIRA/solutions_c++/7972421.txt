#include<iostream>
#include<math.h>
#include<stdlib.h>
using namespace std;
int main()
{
	int a,n,i,t,m,x1,y1,x2,y2,x3,y3,max,min;
	cin>>t;
	cin>>x1;
		cin>>y1;
		cin>>x2;
		cin>>y2;
		cin>>x3;
		cin>>y3;
		a=abs(x1*(y2-y3)-x2*(y1-y3)+x3*(y1-y2));
		max=a;
		m=1;
		n=1;
		min=a;
	for(i=2;i<=t;i++)
	{
		cin>>x1;
		cin>>y1;
		cin>>x2;
		cin>>y2;
		cin>>x3;
		cin>>y3;
		a=abs(x1*(y2-y3)-x2*(y1-y3)+x3*(y1-y2));
		//cout<<a<<endl;
		if(a>=max)
		{
			max=a;
			m=i;
		}
		if(a<=min)
		{
			min=a;
			n=i;
		}
	}
	cout<<n<<" "<<m<<endl;
	
	return 0;
}
