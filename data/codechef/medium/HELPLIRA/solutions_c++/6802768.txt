#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    int i=0;
    int posbig=0,posmall=0;
    double small=1000000000,big=0;
    for(i=1;i<=t;i++)
    {
        int x1,y1,x2,y2,x3,y3;
        scanf("%d %d %d %d %d %d",&x1,&y1,&x2,&y2,&x3,&y3);
        double m,c,d,e,area,f;
        m=((1.0)*(y3-y2))/(x3-x2);
        c=y3-m*(x3);
        e=sqrt(1+(m*m));
        f=sqrt(((y3-y2)*(y3-y2))+((x3-x2)*(x3-x2)));
        d=(y1-m*(x1)-c)/e;
        if(d<0)
            d=-1*d;

        area=(0.5)*(d)*(f);
        if(area>=big)
        {
            big=area;
            posbig=i;
        }
        if(area<=small)
        {
        small=area;
        posmall=i;
        }

    }

    cout<<posmall<<" "<<posbig<<endl;



}
