#include<cstdio>
#include<cmath>
using namespace std;
 
int main()
{
    int x1,x2,x3,y1,y2,y3,n,area,min=1,max=1,minval,maxval;
    scanf("%d",&n);
    scanf("%d%d%d%d%d%d",&x1,&y1,&x2,&y2,&x3,&y3);
    area=fabs((x2-x1)*(y3-y1)-(y2-y1)*(x3-x1));
    minval=area;
    maxval=area;
    for(int i=2;i<=n;i++)
    {
        scanf("%d%d%d%d%d%d",&x1,&y1,&x2,&y2,&x3,&y3);
        area=fabs((x2-x1)*(y3-y1)-(y2-y1)*(x3-x1));
        if(minval>=area)
        {
            minval=area;
            min=i;
        }
        if(maxval<=area)
        {
            maxval=area;
            max=i;
        }
   }
   printf("%d %d\n",min,max);
return 0;
}
