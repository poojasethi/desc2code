#include <iostream>
#include <stdlib.h>
#include <math.h>
using namespace std;
typedef long long LL;

LL area(LL x1,LL y1,LL x2,LL y2, LL x3,LL y3) {
    return abs(x1 * (y2-y3) + x2 * (y3-y1) + x3 * (y1 - y2));
}

int main() {
    LL n = 0,i=0,x1=0,x2=0,x3=0,y1=0,y2=0,y3=0,min_area_index=0,max_area_index=0;
    LL min_area=0,max_area=0,ar=0;
    cin>>n;
    for(i=1;i<=n;i++) {
        cin>>x1>>y1>>x2>>y2>>x3>>y3;
        
        ar = area(x1,y1,x2,y2,x3,y3);
        if(i == 1) {
            min_area = ar;
            max_area = ar;
            min_area_index = 1;
            max_area_index = 1;
        }
        if(ar<=min_area) {
            min_area = ar;
            min_area_index = i;
        }
        if(ar>=max_area) {
            max_area = ar;
            max_area_index = i;
        }
    }
    cout<<min_area_index<<" "<<max_area_index<<endl;
    return 0;
}
