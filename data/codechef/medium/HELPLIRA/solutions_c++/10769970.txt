#include<stdio.h>
#include<stdlib.h>
int triarea(int X[4],int Y[4],int n)
{
    int area=0,i,j=n-1;
    for(i=0;i<n;i++)
    {
        area+=(X[j]+X[i])*(Y[j]-Y[i]);
        j=i; 
    }
    return abs(area/2);
}
int main()
{
	int j,n,area[100];scanf("%d",&n);
	for(j=0;j<n;j++)
	{
		int x1,x2,x3,y1,y2,y3;
		scanf("%d %d %d %d %d %d",&x1,&y1,&x2,&y2,&x3,&y3);
		int i,x[4],y[4];
		x[0]=x1;x[1]=x2;x[2]=x3;
		y[0]=y1;y[1]=y2;y[2]=y3;
		area[j]=triarea(x,y,3);		
	}
	int min=0,max=0,minarea=area[0],maxarea=area[0];
	for(j=0;j<n;j++)
	{
		//printf("%d ",area[j]);
		if(minarea>=area[j]) {min=j;minarea=area[j];}
		if(maxarea<=area[j]) {max=j;maxarea=area[j];}
	}
	printf("%d %d\n",min+1,max+1);
}
