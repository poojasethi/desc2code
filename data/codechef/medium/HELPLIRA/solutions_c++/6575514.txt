#define s(n)                        scanf("%d",&n)
#define ss(n)                       scanf("%s",n)

#define w(t)                        while (t--)
#define prt(n)                      printf("%d\n", n)
#define INF                         9999999

using namespace std;

int max(int a, int b) {
	if (a > b) {
		return a;
	}
	return b;
}

struct Point {
	int x1, y1, x2, y2, x3, y3;
};

int abs(int m) {
	if (m >= 0) {
		return m;
	}
	return -m;
}

int main()
{
	int n, mx, mn, mxi = 0, mni = 0;
	s(n);
	
	struct Point P[n];
	vector<int> A;
	
	for (int i = 0; i < n; i++) {
		cin >> P[i].x1 >> P[i].y1 >> P[i].x2 >> P[i].y2	>> P[i].x3 >> P[i].y3;
		int area = abs(P[i].x1 * (P[i].y2 - P[i].y3) + P[i].x2 * (P[i].y3 - P[i].y1) + P[i].x3 * (P[i].y1 - P[i].y2));
		A.push_back(area);
	}
	
	mx = A[0];
	mn = A[0];
	
	forall(i, 1, n) {
		if (A[i] >= mx) {
			mx = A[i];
			mxi = i; 
		}
		if (mn >= A[i]) {
			mn = A[i];
			mni = i;
		}
	}
	
	cout << mni + 1 << " " << mxi + 1 << endl;
	return 0;
}
