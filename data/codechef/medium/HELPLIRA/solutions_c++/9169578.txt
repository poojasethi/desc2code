#include <iostream>
#include<math.h>
#include<stdlib.h>
using namespace std;
long long int min_i,max_i;
int main() {
	long long int n,x1,x2,x3,y1,y2,y3;
	int area,min=0,max=0;
	cin>>n;
	
	
	
	    cin>>x1>>y1>>x2>>y2>>x3>>y3;
	    area=0.5*(x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2));
	    area=abs(area);
	    
	        
	            min=area;
	            min_i=1;
	        
	    
	        
	            max=area;
	            max_i=1;
	        
	
	
	for(int i=1;i<n;i++)
	{
	    cin>>x1>>y1>>x2>>y2>>x3>>y3;
	    area=(x1*(y2-y3)+x2*(y3-y1)+x3*(y1-y2));
	    area=abs(area);
	    if(min>area)
	        {
	            min=area;
	            min_i=i;
	        }
	    if(max<area)
	        {
	            max=area;
	            max_i=i;
	        }
	}
	cout<<min_i+1<<" "<<max_i+1<<"\n";
	return 0;
}
