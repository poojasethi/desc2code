#include <cstdlib>
#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <limits>
#include <math.h>
#define gc getchar_unlocked
 
void scanint(int &x)
{
    register int c = gc(),n=1;
    x = 0;
    for(;(c<48 || c>57);c = gc())
    {
    	if(c=='-')
    		n=-1;
    }
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    x=n*x;
}
 
using namespace std;
 
int main(){	
	int T;
	int x1, y1, x2, y2, x3, y3;
	scanint(T);
	long long min = std::numeric_limits<long long>::max();
	long long max = std::numeric_limits<long long>::min();
	int minindex, maxindex;
	long long A;
	for(int i = 0; i < T; i++){
		scanint(x1);
		scanint(y1);
		scanint(x2);
		scanint(y2);
		scanint(x3);
		scanint(y3);
		A = (long long)((x2 - x1)*(y3 - y1) - (x3 - x1)*(y2 - y1));
		if(A < 0)
			A = -1*A;
		//cout << A << endl;
		if(A <= min){
			min = A;
			minindex = i;
		}
		if(A >= max){
			max = A;
			maxindex = i;
		}
	}
	cout << (minindex+1) << " " << (maxindex+1) << endl;
}
