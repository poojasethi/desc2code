#include <bits/stdc++.h>
using namespace std;
 
#define MAX 1005
 
int s[MAX][MAX], b[MAX][MAX];
double dp[MAX][MAX][2];
 
int main(int argc, char const *argv[])
{
	int n, m;
	long long d;
	scanf("%d %d %lld", &n, &m, &d);
	for(int i = 1; i <= n; i++) {
		dp[0][i][0] = d;
		for(int t = 1; t <= m; t++) {
			scanf("%d %d", &s[i][t], &b[i][t]);
		}
	}
 
	double amax, bmax;
	for(int t = 1, k; t <= m; t++) {
		for(int i = 1; i <= n; i++) {
			amax = dp[t - 1][i][0];
			bmax = dp[t - 1][i][1];
			for(int j = max(1, i - 1); j <= min(n, i + 1); j++) {
				if(t < abs(i - j)) continue;
				k = t - abs(i - j);
				amax = max(amax, dp[k][j][0]);
				bmax = max(bmax, dp[k][j][1]);
				if(k >= 1) {
					amax = max(amax, dp[k - 1][j][1] * b[j][k]);
					bmax = max(bmax, dp[k - 1][j][0] / s[j][k]);
				}
			}
			if(amax > 1e18) {
				printf("Quintillionnaire\n");
				return 0;
			}
			dp[t][i][0] = amax;
			dp[t][i][1] = bmax;
		}
	}
 
	double ans = 0;
	for(int i = 1; i <= n; i++) {
		ans = max(ans, dp[m][i][0]);
	}
	printf("%.10lf\n", ans);
 
	// for(int t = 0; t <= m; t++) {
	// 	for(int i = 1; i <= n; i++) {
	// 		cout << dp[t][i][0] << " ";
	// 	}
	// 	cout << endl;
	// 	for(int i = 1; i <= n; i++) {
	// 		cout << dp[t][i][1] << " ";
	// 	}
	// 	cout << endl << endl;
	// }
	return 0;
}
