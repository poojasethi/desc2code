#include<bits/stdc++.h>
using namespace std;

double f[1005][1005][2];
long long selling[1005][1005],buying[1005][1005];

int main() {
	long long n,m,d;
	cin>>n>>m>>d;

	for(int i=1; i<=n; i++){
		for(int j=0; j<m; j++){
			cin>>selling[j][i]>>buying[j][i];
		}
	}

	for(int i=1; i<=n; i++){
		f[0][i][0]=d;
		f[0][i][1]=0;
	}

    int dem=0;
	for(int i=1; i<=m; i++){
		for(int j=1; j<=n; j++){
			f[i][j][0]=f[i-1][j][0];
			f[i][j][1]=f[i-1][j][1];

				if(i+1<n){
					f[i][j][0]=max(f[i][j][0],f[i-1][j+1][0]);
					f[i][j][1]=max(f[i][j][1],f[i-1][j+1][1]);

					f[i][j][0]=max(f[i][j][0],f[i-1][j][1]*buying[i-1][j] );
					f[i][j][1]=max(f[i][j][1],f[i-1][j][0]/selling[i-1][j]);
				}

				if(i-1>=0){
					f[i][j][0]=max(f[i][j][0],f[i-1][j-1][0]);
					f[i][j][1]=max(f[i][j][1],f[i-1][j-1][1]);

					f[i][j][0]=max(f[i][j][0],f[i-1][j][1]*buying[i-1][j]);
					f[i][j][1]=max(f[i][j][1],f[i-1][j][0]/selling[i-1][j]);
				}

				if(f[i][j][0]>=1e18)  dem=1;
        }
	}

    if(dem==1)   printf("Quintillionnaire\n");
	else{
		double best=0;
		for(int i=1; i<=n; i++){
			if(f[m][i][0]>best)  best=f[m][i][0];
		}
	
		printf("%.12lf\n",best);
	}
}
