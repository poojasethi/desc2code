#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<iostream>
using namespace std;
const int N = 1050;
int n,m;
long double buy[N][N],sell[N][N];
long double f[N][N],g[N][N];
long long getll()
{
	long long res=0;
	char ch=getchar();
	while((ch<'0' || ch>'9') && ch!='-')
		ch=getchar();
	bool fan=0;
	if(ch=='-')
	{
		fan=1;
		ch=getchar();
	}
	while('0'<=ch && ch<='9')
	{
		res=res*10+ch-'0';
		ch=getchar();
	}
	if(fan)
		res=-res;
	return res;
}
void GetData()
{
	int i,j;
	n=getll();
	m=getll();
	long double now=getll();
	for(j=1;j<=n;j++)
		f[0][j]=now;
	for(j=1;j<=n;j++)
	{
		for(i=1;i<=m;i++)
		{
			sell[i][j]=getll();
			buy[i][j]=getll();
		}
	}
}
long double DoIt()
{
	int i,j;
	for(i=1;i<=m;i++)
	{
		for(j=1;j<=n;j++)
		{
			f[i][j]=max(max(f[i-1][j],g[i-1][j]*buy[i][j]),max(f[i-1][j-1],f[i-1][j+1]));
			g[i][j]=max(max(g[i-1][j],f[i-1][j]/sell[i][j]),max(g[i-1][j-1],g[i-1][j+1]));
			if(f[i][j]>1e18l || g[i][j]>1e18l)
				return 1e19l;
		}
	}
	long double res=0.0l;
	for(j=1;j<=n;j++)
		res=max(res,f[m][j]);
	return res;
}
int main()
{
	GetData();
	double ans=DoIt();
	if(ans<=1e18l)
		printf("%.8f\n",ans);
	else
		printf("Quintillionnaire\n");
	return 0;
}
