#include<bits/stdc++.h>
using namespace std;
#define ll long long 
#define aut(r,v) typeof(v) r = (v)
#define FOR(i,a,n) for (int i=(a);i<(n);++i)
#define foreach(c,itr) for(__typeof((c).begin()) itr=(c).begin();itr!=(c).end();itr++)
#define pb push_back
#define mp make_pair
#define vi vector<int >
#define pii pair<int ,int >
#define fr first
#define sc second
#define MOD 1000000007
const double PI=2*acos(0.0); 
 
double dp[1111][1111][3];
vector<pii > rate[1111]; //0-Coin 1-Dollar
int main(){
 

 memset(dp,0,sizeof dp);
 ll n,m,d;
 cin>>n>>m>>d;
 FOR(i,0,n){
    FOR(j,1,m+1){
      int s,b;
      cin>>s>>b;
      rate[j].pb(mp(s,b));
      }//Sell Buy
 }
   FOR(i,0,n){
     dp[0][i][0]=d;
   }
   FOR(t,1,m+1){
     FOR(i,0,n){
        dp[t][i][0]=dp[t-1][i][0];
        dp[t][i][1]=dp[t-1][i][1];
        
        dp[t][i][0]=max(dp[t][i][0],dp[t-1][i-1][0]);
        dp[t][i][1]=max(dp[t][i][1],dp[t-1][i-1][1]);
        
        dp[t][i][0]=max(dp[t][i][0],dp[t-1][i+1][0]);
        dp[t][i][1]=max(dp[t][i][1],dp[t-1][i+1][1]);
        
        dp[t][i][0]=max(dp[t][i][0],dp[t-1][i][1]*rate[t][i].sc);
        dp[t][i][1]=max(dp[t][i][1],dp[t-1][i][0]/rate[t][i].fr);
        
     }
   }
   double ans=INT_MIN;
   FOR(i,0,n)
     ans=max(ans,dp[m][i][0]);
     if(ans > 1e18)
		printf("Quintillionnaire\n");
	else
		printf("%.7lf\n", ans);
 return 0;
}
