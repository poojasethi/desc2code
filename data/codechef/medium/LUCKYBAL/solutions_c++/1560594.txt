#include<iostream>
#include<string.h>
using namespace std;
char a[100010];
main()
{
      int i,t,n,k;
      long long sum;
      scanf("%d",&t);
      while(t--)
      {
                k=0;
                scanf("%s",a);
                n=strlen(a);
                sum=0;
                for(i=0;i<n;i++)
                {
                                if(a[i]=='4'){
                                             sum+=n-i;
                                             k=i-1;
                                             while(a[k]=='7'){
                                                       sum+=n-i;
                                                       k--;     
                                             }
                                } 
                }
                printf("%lld\n",sum);
      }
      return 0;
}
