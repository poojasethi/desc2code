#include<iostream>
#include<string>
using namespace std;

int main()
{
	int t;
	string s;
	long long len;
	long long int ans;
	cin>>t;
	long long int n;
	long long int sum,flag,x;
	while(t--)
	{
		cin>>s;
		len = s.length();
		ans = len * (len+1)/2;
		n = 0;
		sum = 0;
		x = 0;
		flag = 0;
		while(n < len)
		{
			while (s[n] == '7' && n < len) {
				sum++;
				flag = 1;
				n++;
			}
			if(flag == 1)
			{
				sum = sum * (sum+1)/2;
				x = x + (sum);
				sum = 0;
				flag = 0;
			}
			n++;
		}
		ans -= x;
		cout<<ans<<endl;
	}
	return 0;
}
