#include<iostream>
#include<stdio.h>
#include<string.h>
#include<string>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
			char s[1000010];
			scanf("%s",s);

			int n=strlen(s),pos=-1;
			long long ans=0;

			for(int i=0;i<n;i++)
			{
				if(s[i]=='4')
					pos=i;
				ans+=pos+1;
			}
			printf("%lld\n",ans);
	}
	return 0;
}
