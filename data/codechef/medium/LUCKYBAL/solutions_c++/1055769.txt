#include<iostream>
#include<string.h>
using namespace std;
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
 
int main()
{
char str[100500];
int t;
cin>>t;
while(t--)
{
cin>>str;
long long int i,j,l,c=0;
l=strlen(str);
long long int k1=(long long int)(1)*l*(l+1)/2;
for(i=0;i<=l;i++)
{
c=0;
for(j=i;j<=l;j++)
{
if(str[i]=='7' && str[i]==str[j])
{
c++;
}
 
else
{
i=j;
break;
}
}
 
long long int k=(long long int)c*(c+1)/2;
k1=k1-k;
}
 
 
cout<<k1<<"\n";
}
}
