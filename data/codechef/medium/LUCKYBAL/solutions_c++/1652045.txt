using namespace std;

int main()
{
	int T;
	scanf("%d", &T);
	string str;
	long long ans, len;
	for (int t = 0; t < T; t++)
	{
		cin >> str;
		len = str.size();
		long long ans = (len*(len+1))/2;
		long long segLen = 0;
		for (int i = 0; i < len; i++)
		{
			if (str[i] == '7')
				segLen++;
			else
			{
				ans -= (segLen*(segLen+1))/2;
				segLen = 0;
			}
		}
		ans -= (segLen*(segLen+1))/2;
		printf("%lld\n", ans);
	}
	return 0;
}
