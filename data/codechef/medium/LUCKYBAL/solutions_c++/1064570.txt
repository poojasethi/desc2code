#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;
int main()
{
string s1;
long long int tc,l,t,ans;
scanf("%lld",&tc);
while(tc--)
{
t=0;
cin>>s1;
l=s1.length();
ans=l*(l+1)/2;
for(int i=0;i<l;i++)
{if(s1[i]=='7')
t++;
else
t=0;
ans=ans-t;
}
printf("%lld\n",ans);
}
return 0;
}
