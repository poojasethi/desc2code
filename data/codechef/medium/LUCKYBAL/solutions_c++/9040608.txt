#include <string.h>
#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
    int cases;
    scanf("%d",&cases);
    while(cases--)
    {
        char str[100001];
        scanf("%s",str);
        long long int len=strlen(str);
        long long int total=0,flag=0,temp=0;
        for(int i=0;i<len;i++)
        {
            if(str[i]=='4')
            {
                flag=1;
                temp=i;
            }
            if(flag==1)
            {
                total=total+temp+1;
            }
        }
             printf("%lld\n",total);
    }
    return 0;
}
