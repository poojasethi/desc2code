	while (T--)
		{
		string s;
		cin>>s;
		n=s.length();
		LL p=0,cnt=0;
		REP(i,n)
			{
			if (s[i]=='7')
				{
				p=0;
				while(s[i]!='4' && i<n)
					{
					p++;
					i++;
					}
				cnt+=(p*(p+1))/2;
				}
			}
			
		LL ans=(n*(n+1))/2-cnt;
		
		cout<<ans<<"\n";
		}
