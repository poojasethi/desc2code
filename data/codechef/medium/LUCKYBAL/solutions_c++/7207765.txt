#include <iostream>
#include <cstdio>
using namespace std;

char str[100005];

int main() {
	int t, i, j;
	cin>>t;
	while(t--) {
		i = 1;
		scanf("%s",str);
		long long ext = 0;
		long long tot = 0;
		if(str[0] == '4') {
			tot = 1;
			ext = 1;
		}
		while(str[i] != '\0') {
			if(str[i] == '4') {
				tot++;
				int count = 0;
				if(str[i - 1] == '7') {
					j = i - 1;
					while(str[j] == '7'){
						count++;
						j--;
					}
					ext += count;
				}
				tot += ext;
				ext++;
			} else {
				tot += ext;
			}i++;
		}
		cout<<tot<<endl;
	}
	return 0;
}
