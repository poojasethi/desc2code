#define MOD 1000000007
long long power(int a,int b)
{
    long long x=1,y=a;
    while(b>0)
    {
        if(b%2==1)
        {
            x=(x*y);
            if(x>MOD) 
                x%=MOD;
        }
        y=(y*y);
        if(y>MOD) 
            y%=MOD;
        b/=2;
    }
    return x;
}
 
long long invert(int n)
{
    return power(n,MOD-2);
}
