int dc[] = {0,1,0,-1};
int dr[] = {-1,0,1,0};

LL C[500020][6],n,tujuh,ans,x;

int main(void){
    for (int j = 1; j<=5; j++) C[1][j] = 0;
    for (int i = 1; i<=500020; i++) C[i][1] = i;
    
    for (int j = 2; j<=5; j++)
    for (int i = 2; i<=500020; i++) 
    {
        C[i][j] = C[i-1][j-1] + C[i-1][j];
        C[i][j] %= MOD;
    }
    
    cin>>n;
    if (n%2==0) tujuh = 2; else tujuh = 1;
    
    ans = 0;    
    while (1)
    {
          x = (n-tujuh)/2;
          x -= 6;
          if (x<0) break;
          if (x==0) {ans += 1; break;}
          
          ans += C[x+5][5];
          ans %= MOD;
          
          tujuh += 2;
    }
    ans %= MOD;
    
    cout<<ans<<endl;
    
    //system("pause");

}
