#include<stdio.h>
#define M 1000000007
typedef unsigned long long int ll;
ll arr[8]={1, 1, 500000004, 166666668, 41666667, 808333339, 301388891};
ll combo(ll count,ll q)
{
	ll i,k=1;
	if(count<q)
		return 0;
	if(q>count/2)
		q=count-q;
	for(i=1;i<=q;i++)
		k=(k*count--)%M;
	k=(k*arr[q])%M;
	return k;				
}
int main(){
	ll n,out=0;
	scanf("%llu",&n);
	if(n>12){
		out=1;
		ll i,k=(n-13)/2;
		for(i=k;i>0;i--){
			out=(out+combo(i+5,5))%1000000007;
		}
	}
	printf("%llu",out);
	return 0;
}
