#include <iostream>
using namespace std;

#define maxn 1000001
#define mod 1000000007

long long c[maxn][6];

int main() {
    c[0][0] = 1;
    for(int i=1;i<maxn;++i) {
        c[i][0] = 1;
        for(int j=1;j<6;++j) {
            c[i][j] = (c[i-1][j] + c[i-1][j-1]) %mod;
        }
    }

    long long res = 0;
    int n, l, k;
    cin >> n;
    l = n-12;
    for(int i=1;i<=l;++i) {
        if ( ((n-i)&1) == 0 ) {
            k = (n-i) >> 1;
            res = (res + c[k-6+5][5]) % mod;
        }
    }
    cout << res << endl;
    return 0;
}
