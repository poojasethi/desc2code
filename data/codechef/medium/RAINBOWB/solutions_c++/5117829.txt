#include <cstdio>
int main()
{
 long long n,fact,C,i,j,mod=1000000007,mod1=mod,sum;
 scanf("%lld",&n);
 if(n%2==0)
  i=1;
 else
  i=0;
 mod1=mod*120; 
 sum=0;
 for(;i<=n-13;i+=2)
 {
  fact=1;
  for(j=0;j<5;++j)
   fact=(fact*(n-3-i-2*j)/2)%mod1;
  C=fact/120;
  sum=(sum+C)%mod;
 }
 printf("%lld\n",sum);
}
