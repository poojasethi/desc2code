#include<stdio.h>
#include<math.h>
#include<stdlib.h>
#include<iostream>
using namespace std; 
int main()
{
	long long int mod=pow(10,9)+7,res=1,i,j,div2=2,div3=3,div4=4,div5=5,div6=6,k,N;
	cin>>N;
	
	if(N>=13)
	{
		k=(N-13)/2;
		for(i=k+1;i<k+7;i++)
		{
			j = i;
			if(j%6==0)
			{
				j = j/div6;
				div6=1;
			}
			if(j%5==0)
			{
				j = j/div5;
				div5=1;
			}
			if(j%4==0)
			{
				j = j/div4;
				div4=1;	
			}
			if(j%3==0)
			{
				j = j/div3;
				div3=1;
			}
			if(j%2==0)
			{
				j = j/div2;
				div2=1;
			}
			
			res = (res*j)%mod;
		}
		cout<<res<<endl;
	}
	else
		cout<<"0\n";
}
