#include<stdio.h>
#include<string.h>
#define M 1000000
int fact[M+200];
int mod = 1000000007;
int ans = 0;
int N;
int pw(int x,int k){
    int res = 1;
    int mul = x;
    while(k>0){
        if(k%2) res = (1ll*res*mul)%mod;
        mul = (1ll*mul*mul)%mod;
        k/=2;
    }
    return res;
}
int c(int k,int n){
    int r1 = fact[n];
    int r2 = (1ll*fact[k]*fact[n-k])%mod;
    return (1ll*r1*pw(r2,mod-2))%mod;
}
void prepare(){
    fact[0] = 1;
    for(int i=1;i<=M;i++) fact[i] = (1ll*fact[i-1]*i)%mod;
}
void solve(){
    if( N < 13) {
        printf("0");
        return ;
    }
    for(int pos = 1 ; pos <= N ; pos++){
        int tot = N - pos;
        if( tot%2 ) continue;
        tot/=2;
        if( tot < 6) break;
        ans = (ans+c(5,tot-1))%mod;
    }
    printf("%d",ans);
}
main(){
//    freopen("test.inp","r",stdin);
    scanf("%d",&N);
    prepare();
    solve();
}
