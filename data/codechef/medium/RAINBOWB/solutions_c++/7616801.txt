#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MOD 1000000007
ll pw(ll a, ll b)
{
	if(b==0)
		return 1;
	ll temp = pw(a,b/2);
	temp = (temp*temp)%MOD;
	if(b%2 == 0)
		return temp;
	else
		return (temp*a)%MOD;
}
long long int C(int n)
{
    long long int a=1;
    int i;
    for(i=0;i<6;i++)
    {
        a*=(n-i);
        a%=MOD;
    }
    a*=pw(720,MOD-2);
    a%=MOD;
    return a;
}
