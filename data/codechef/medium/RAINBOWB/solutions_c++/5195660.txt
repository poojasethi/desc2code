#define base 1000000007

ll mu(ll a,ll k)
{
	ll s=1,b=a;
	while(k>0)
	{
		if (k%2==1) {s=(s*b)%base;b=(b*b)%base;}
		else b=(b*b)%base;
		k/=2;
	}
	return s;
}
ll gt(ll n)
{
	ll s=1,i;
	for(i=1;i<=n;i++) s=(s*i)%base;
	return s;
}
