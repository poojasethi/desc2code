#include<stdio.h>
//const int MAX = 1000001;
const long long   MOD = 1000000007;
int main()
{
    int reg[2][9];
    int N;
    int k;
    scanf("%d",&N);
    k = (N + 1) / 2;
    if( k < 7)
     {
        printf("0\n");
        return 0;
    }
    if( k == 7 )
    {
        printf("1\n");
        return 0;
    }
    //if( k == 8 )
    //{
    //    printf("7\n");
    //    return 0;
   // }
    for(int j = 0 ; j < 9 ; j++)
    {
        reg[0][j] = 0;
    }
    for(int i = 1 ; i <= k - 7 ; i++ )
    {
        reg[1][7] = 1;
        for(int j = 6 ; j > 0 ; j--)
        {
            reg[1][j] = (reg[1][j + 1]  + reg[0][j]) % MOD;
        }
        for(int j = 7 ; j > 0 ; j--)
        {
            reg[0][j] = reg[1][j];
        }
    }
    long long  res = 0 ;
    for(int i = 1; i <= 7 ;i++)
    {
        res = (res + reg[1][i]) % MOD;
    }
    printf("%lld\n",res);
    return 0;
 
 
 
}
