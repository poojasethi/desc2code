#include<bits/stdc++.h>
using namespace std;
long long int c[10][1000010];
int main()
{
    long long int i,n,j,mod=1000000007,ans=0;
    scanf("%lld",&n);
    //printf("%lld\n",n);
    for(i=0;i<=n+5;i++)
        c[0][i]=1;
    for(i=1;i<=n+5;i++)
    for(j=1;j<=min(5LL,i);j++)
    {
        //printf("%lld %lld %lld\n",j,i,c[j][i]);
        c[j][i]=(c[j][i-1]+c[j-1][i-1])%mod;
    }
    for(i=12;i<n;i+=2)
    {
        ans+=c[5][i/2-1];
        ans%=mod;
    }
    printf("%lld",ans);
}
