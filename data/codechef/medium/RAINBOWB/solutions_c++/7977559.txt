#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int mod=1000000007;
int p(int a,int b)
{
    if(b==0)
        return 1;
    if(b==1)
        return a;
    int c=p(a,b/2);
    return (1ll*((1ll*c*c)%mod)*p(a,b%2))%mod;
}
int main()
{
    int n,i,j,x;
    sd(n);
    if(n<13)
    {
        printf("0");
        return 0;
    }
    if(n%2==0)
        x=n/2 -1;
    else
        x=(n-1)/2;
    j=1;
    for(i=1;i<=6;i++)
    {
        j=(1ll*((1ll*j*(x+1-i))%mod)*p(i,mod-2))%mod;
    }
    printf("%d\n",j);
    return 0;
}
