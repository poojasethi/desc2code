#include <bits/stdc++.h>
using namespace std;
const long long mod=1e9+7;
vector<int> tmp;
long long dp(int x)
{
	int i,j,res[10];
	long long ans=1;
	vector<int> v=tmp;
	for(i=1;i<=5;i++)
	{
		res[i]=(x-i);
		for(j=4;j>=0;j--)
		{
			if(res[i]%v[j]==0) 
			{
				res[i]/=v[j];
				v[j]=1;
			}
		}
	}
	for(i=1;i<=5;i++)
	{

		//printf("%d ",res[i]);
		ans=ans*res[i]%mod;

	}
	//for(j=0;j<5;j++) if(v[j]!=1) printf("WROMG");
	//printf("\n%lld\n",ans);
	return ans;
}
int main()
{
	int i,n;
	long long res=0;
	for(i=1;i<=5;i++) tmp.push_back(i);
	cin>>n;
	for(i=6;i<=n/2;i++)
	{
		if(i*2==n) continue;
		res=(res+dp(i))%mod;
	}
	cout<<res<<endl;
}
