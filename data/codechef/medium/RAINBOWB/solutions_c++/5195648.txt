#include<cstdio>
typedef long long ll;
int a[6];
void div(int x,int n)
{
    while(n--)
    {
        for(int i=0;i<6;++i) if(a[i]%x==0)
        {
            a[i]/=x;
            break;
        }
    }
}
int main()
{
    int n,i,j;
    ll ans=1ll;
    scanf("%d",&n);
    n=((n+1)>>1)-1;
    if(n<6)
    {
        printf("0");
        return 0;
    }
    for(i=0;i<6;++i) a[i]=n-i;
    div(2,4);div(3,2);div(5,1);
    for(i=0;i<6;++i) ans=(ans*(1ll*a[i]))%1000000007;
    printf("%lld",ans);
}
