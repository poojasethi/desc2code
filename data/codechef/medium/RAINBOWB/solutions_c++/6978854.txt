#include <cstdio>
#include <algorithm>
using namespace std;

#define mod 1000000007

int ret, n, m, comb[1000000 + 5][6], i, j;

int func (int n) {
	if (n < 6) return 0;
	return comb[n - 6 + 5][5];
}

int main () {
	scanf("%d", &n);
	comb[0][0] = 1;
	for(i = 1; i <= n; i++) for(j = 1, comb[i][0] = 1; j <= 5; j++) comb[i][j] = (comb[i - 1][j - 1] + comb[i - 1][j]) % mod;
	for(i = 1; i <= n; i++) if ((n - i) % 2 == 0) ret = (ret + func((n - i) / 2)) % mod;
	printf("%d\n", ret);
	return 0;
}
