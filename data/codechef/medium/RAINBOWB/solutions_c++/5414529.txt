#include<iostream>
using namespace std;
int main()
{
	long long  n;
	cin>>n;
	n-=13;
	if(n<0)
	{
		cout<<"0";
		return 0;
	}
	
	
	if(n%2==1)
	{
		n--;
	}
	unsigned long long ans=0;
	unsigned long long c=(1000000000+7);
	while(n>=0)
	{
		unsigned long long m=(n/2);
		unsigned long long temp=1;
		unsigned long long ar[5];
		for(int i=0;i<5;i++)
		{
			ar[i]=m+1+i;
		}
		for(int w=5;w>=2;w--)
		{
			
			for(int i=0;i<5;i++)
			{
				if(ar[i]%w==0)
				{
					ar[i]/=w;
					break;
				}
			}
		}
		
		for(int i=0;i<5;i++)
		{
			ar[i]%=c;
		}
		
		temp=ar[0];
		for(int i=1;i<5;i++)
		{
			temp*=ar[i];
			temp%=c;
		}
		
		//temp%=c;
		ans+=temp;
		ans%=c;
		n-=2;
	}
	
	cout<<(ans%c);
	
	
	
	return 0;
}
