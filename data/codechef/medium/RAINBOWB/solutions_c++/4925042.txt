lli nck(int n)
{

    lli i,ans=1;
    int f2=0,f3=0,f4=0,f5=0,f6=0;
    for(i=5; i>=0; i--)
    {
        //cout<<"ans = "<<ans<<endl;
        ans=ans*(n+i);
        if(ans%6==0 && f6==0)
        {
            ans=ans/6;
            f6=1;
        }
        if(ans%5==0 && f5==0)
        {
            ans=ans/5;
            f5=1;
        }

        {

    return ans;
}
