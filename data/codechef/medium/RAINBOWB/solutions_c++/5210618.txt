#include <cstdio>
#include <iostream>

using namespace std;
typedef long long LL;
const LL MOD = 1000000007;
LL sixInverse;

LL modinverse(LL num){
    LL first = MOD, second = num, q, y1=1, y2=0, temp;

    while(second>1){
        temp = first%second;
        q = first/second;
        first = second;
        second = temp;

        temp = y2 - q*y1;
        y2 = y1;
        y1 = temp;
    }
    if(y1<0)
        y1+=MOD;
    return y1;
}

int main(){
	sixInverse = modinverse(720);
	int n,ni;
	long long ans,temp;
    cin>>ni;

    if(ni<13)
        ans = 0;
    else if(ni==13||ni==14)
        ans = 1;
    else {
        n = (ni-13)/2;
        temp = 1;
        for(int t=1; t<7; ++t){
            temp *= n+t;
            temp %= MOD;
        }
        ans = (temp*sixInverse)%MOD;
    }

    cout<<ans<<endl;
}
