#include<cstdio>
#define mod 1000000007
using namespace std;
typedef unsigned long long int ull;
ull combo[1000001][8];
void build(){
	int i,j;
	for(i=1;i<1000001;i++){
		combo[i][0]=1;
		combo[i][1]=i;
	}
	for(i=2;i<1000001;i++){
		for(j=2;j<8;j++){
			if(j>i)
				combo[i][j]=0;
			else	if(i==j)
						combo[i][j]=1;
					else
						combo[i][j]=(combo[i-1][j]+combo[i-1][j-1])%mod;
		}
	}
}
int main(){
	build();
	ull n,out=0;
	scanf("%llu",&n);
	if(n>12){
		out=1;
		ull i,k=(n-13)/2;
		for(i=k;i>0;i--){
			out=(out+combo[i+5][5])%mod;
		}
	}
	printf("%llu",out);
	return 0;
}
