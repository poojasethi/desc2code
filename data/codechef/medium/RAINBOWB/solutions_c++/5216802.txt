#include<iostream>
#include<bits/stdc++.h>
#define MOD 1000000007



using namespace std;

typedef long long ll;
ll N;

ll modPow(ll x,ll y){
	if(y==0){
		return 1;
	}
	if(y==1){
		return x;
	}
	ll res=modPow(x,y/2);
	res=(res*res)%MOD;
	if(y & 1){
		res=(res*x)%MOD;
	}

	return res;
}


ll func1(ll N1){
	ll P=1;
	ll i;

	for(i=0;i<6;i++){
		P =(P*(N1-i))%MOD;
	}

	ll invfact=modPow(720,MOD-2);
	P=(P*invfact)%MOD;
	return P;
}

int main()
{
	cin>>N;
	ll i;
	if(N<13){
		cout<<"0"<<endl;
	}
	else if(N==13 || N==14){
		cout<<"1"<<endl;
	}
	else{
		if(N & 1){
			cout<<func1((N-1)/2)<<endl;
		}
		else{
			cout<<func1((N/2-1))<<endl;
		}
	}
}
