#include<bits/stdc++.h>
using namespace std;
inline bool overLucky(int N){
	while(N > 0){
		if(N % 10 == 4 || N % 10 == 7)
			return true;
		N /= 10;
	}
	return false;
}
int main(){
	int t;
	cin >> t;
	while(t--){
		int N;
		cin >> N;
		vector<int> facs;
		int i;
		for(i = 1;i*i < N;++i){
			if(N % i == 0){
				facs.push_back(i);
				facs.push_back(N/i);
			}
		}
		if(i*i == N){
			facs.push_back(i);
		}
		int sz = facs.size();
		int cnt = 0;
		for(int i = 0;i < sz;++i){
			if(overLucky(facs[i])){
				++cnt;
			}
		}
		cout << cnt << "\n";
	}
	return 0;
}
