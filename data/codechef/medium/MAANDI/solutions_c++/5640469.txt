#include<iostream>
using namespace std;

int check(int n)
{
	int digit=0;
	while(n>0)
	{
		digit=n%10;
		if(digit==4 || digit==7)
		return 1;
		else
		n=n/10;
	}
	return 0;	
}

int main()
{
	int t;
	cin>>t;
	while(t)
	{
		int n;
		cin>>n;
		int count=0;
		for(int i=1;i*i<=n;i++)
		{
			if(n%i==0)
			{
				count=count+check(i)+check(n/i);
			}
		}
		
		cout<<count<<endl;
		t--;
	}
	return 0;
}
