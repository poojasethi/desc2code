#include<iostream>
using namespace std;

bool check(long n) {
	while(n!=0) {
		if(n%10 == 4 || n%10 == 7) {
			return true;
		}
		n = n/10;
	}
	return false;
}

int main() {
	int t;
	cin >> t;
	long n;
	int count;
	while(t--) {
		cin >> n;
		count = 0;
		for(int i=1;i*i<=n;i++) {
			if(n%i == 0) {
				if(check(i)) {
					//cout << i << endl;
					count++;
				}
				if(check(n/i) && i*i!=n) {
					//cout << "hel" << i << endl;
					count++;
				}
 			}
		}
		cout << count << endl;
	}
}
