#include<bits/stdc++.h>
using namespace std;
 
bool isOverlucky(int n)
{
    while(n)
    {
        if(n%10==4 || n%10==7)
            return true;
        n/=10;
    }
    return false;
}
 
int main()
{
    int tc,kk=1, n;
    cin>>tc;
    while(tc--)
    {
        cin>>n;
        int sq=sqrt(n), ans=0;
        for(int i=1;i<=sq;i++)
        {
            if(n%i==0)
            {
                if(isOverlucky(i))
                    ans++;
                if(isOverlucky(n/i))
                    ans++;
            }
        }
        if(sq*sq==n)   if(isOverlucky(sq)) ans--;
        cout<<ans<<endl;
    }
    return 0;
}
