int main()
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n;
		scanf("%d",&n);
		int len = sqrt(n);
		int ans = 0;
		for (int i = 1; i <= len; i++)
		{
			if (n%i == 0)
			{
				int num1 = i;
				int num2 = n/i;
				int tnum1 = num1;
				int tnum2 = num2;
				while (num1)
				{
					if (num1%10 == 4 || num1%10 == 7) {
						ans++;
						break;
					}
					num1 = num1/10;
				}
				if (num2 != i) {
					while (num2)
					{
						if (num2%10 == 4 || num2%10 == 7) {
							ans++;
							break;
						}
						num2 = num2/10;
					}
				}
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
