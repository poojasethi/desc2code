#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
		vector<int> v;
		long long int n;
		cin>>n;
		for(int i=1; i*i<=n; i++){
			if(!(n%i)){
				int val=n/i;
				v.push_back(i);
				if(i-val) v.push_back(val);
			}
		}
		int s=v.size();
       // cout<<s<<endl;
		int count=0;
		for(int i=0; i<s; i++){
			int x=v[i];
            
			while(x){
				int temp=x%10;
				if(temp==7 || temp==4){
					count++;
                    break;
						
				}
				x/=10;
			}
		}
		cout<<count<<endl;
	}
}
