#include<bits/stdc++.h>
using namespace std;
bool checkUnlucky(long x ){
	int rem=0;
	while(x){
		rem=x%10;
		x=x/10;
		if(rem==4 || rem== 7){
			return true;
		}
	}
	return false;
}
int main(){
	long t,n;

	scanf("%ld",&t);
	while(t--){
		scanf("%ld",&n);
		long count=0;
		for(long i=1;i<=sqrt(n);i++){
			if(n%i==0){
				long  x=n/i;
				long  y=i;
	
				count+=checkUnlucky(y)?1:0;
				if(x!=y){
					count+=checkUnlucky(x)?1:0;
				}
			}
		}
		printf("%ld\n",count);
	}
	return 0;
}
