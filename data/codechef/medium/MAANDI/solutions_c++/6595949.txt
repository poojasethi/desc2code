bool ok(int n) {
	while(n) {
		if(n%10==4 || n%10==7) return true;
		n/=10;
	}
	return false;
}

void solve()
{
	int n;
	s(n);
	int ans = 0;
	for(int i = 1; i * i <= n; i++) 
		if(n % i == 0) {
			if(ok(i)) ans++;
			if(i * i != n && ok(n / i)) ans++;
		}
		pi(ans); pn;
}
