#include <bits/stdc++.h>
using namespace std;
int main()
{
    long long int t,i,n,p,counti,q;
    cin>>t;
    while(t--)
    {
        counti=0;
        cin>>n;
        for(i=1;i<=sqrt(n);i++)
        {
            if(n%i==0)
            {
                p=i;
                while(p!=0)
                {
                    q=p%10;
                    if(q==4||q==7)
                    {
                        counti++;
                        break;
                    }
                    p=p/10;
                }
                p=n/i;
                if(p!=i)
                {while(p!=0)
                {
                    q=p%10;
                    if(q==4||q==7)
                    {
                        counti++;
                        break;
                    }
                    p=p/10;
                }}
            }
        }
    cout<<counti<<endl;
    }
    return 0;
}
