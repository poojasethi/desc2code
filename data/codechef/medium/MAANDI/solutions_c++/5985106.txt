#include<stdio.h>


using namespace std;

int islucky(long long x)
{
    long long  q=x;
    int tag=0;
    while(tag==0&&q>0)
    {
        int r=q%10;
        q=q/10;
        if(r==4||r==7)
        {
            tag=1;
        }
    }
    return tag;
}
int main()
{
    int test;
    cin>>test;
    for(int i=0;i<test;i++)
    {
        long long c=0;
        long long num;
        cin>>num;
        for(long long x=1;x*x<num;x++)
        {
            if(num%x==0)
            {
                if(islucky(x)==1)
                {
                 c++;
                }
                int y=num/x;
                if(islucky(y)==1)
                {
                    c++;
                }

            }
        }
        cout<<c<<endl;
    }
    return 0;
}
