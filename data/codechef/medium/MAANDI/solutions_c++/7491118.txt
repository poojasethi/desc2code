int ok( int x )
{
	while( x )
	{
		if( x % 10 == 4 || x % 10 == 7 )
			return 1;
		x /= 10;
	}
	return 0;
}

int main()
{
	int t;
	scanf( "%d", &t );
	while( t -- )
	{
		int n, ans = 0;
		scanf( "%d", &n );
		for( int i = 1 ; i*i <= n ; i ++ )
			if( n % i == 0 )
			{
				int a = i, b = n / i;
				ans += ok( a ) + ( a != b ) * ok( b );
			}
		printf( "%d\n", ans );
	}
	return 0;
}
