#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;

int checkOverLucky(int x)
{
    while(x)
    {
        int rem=x%10;
        if(rem==4 || rem==7)
            return 1;
        x=x/10;
    }
    return 0;
}
main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int cnt=0;
        int n;
//        n=t+1;
        cin>>n;
        int rt=sqrt(n);
        if((rt*rt)==n)
            cnt+=checkOverLucky(rt);
        for(int i=1;i<=rt;i++)
        {

            if(n%i==0)
            {
                if(i==rt && rt*rt==n)
                    continue;
                cnt+=checkOverLucky(i);
                cnt+=checkOverLucky(n/i);
            }
        }
        cout<<cnt<<endl;
    }
}
