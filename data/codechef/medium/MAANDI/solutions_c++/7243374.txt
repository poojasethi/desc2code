#include<stdio.h>
#include<math.h>
#include<stdio.h>
using namespace std;
int check(long long int a,long long int n)
{  int t=0;
    n=n/a;
	while(a>0)
	{
		if(a%10==4||a%10==7)
		{
		t=1;
		break;
		}
		a/=10;
	}
	while(n>0)
	{
		if(n%10==4||n%10==7)
		{
		t+=1;
		return t;
		}
		n/=10;
	}
	return t;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,count=0,i,x;
		scanf("%lld",&n);
		x=sqrt(n);
		for(i=1;i<=x;i++)
			if(n%i==0)
			count+=check(i,n);
		printf("%lld\n",count);
	}
	return 0;
}
