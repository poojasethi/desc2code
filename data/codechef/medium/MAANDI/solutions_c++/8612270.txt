#include<bits/stdc++.h>
using namespace std;
bool overlucky(long x)
{
	while(x)
	{
		if(x%10==4 || x%10==7)
			return true;
		x/=10;
	}
	return false;
}
int main()
{
	int t;
	for(scanf("%d", &t); t--; )
	{
		long n;
		scanf("%ld", &n);
		long cnt=0;
		for(long i=1; i*i<=n; i++){
			if(n%i)
				continue;
			if(overlucky(i))
				cnt++;
			if(overlucky(n/i))
				cnt++;
		}
		printf("%ld\n", cnt);

	}
	return 0;
}
