#include<stdio.h>
#include<math.h>
int check(int n)
{  int k=0;
    while(n)
    {
        if(n%10==4||n%10==7)
         {
             k=1; break;
         }
        n=n/10;
    }
    return k;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        long int n,i,c=0;
        scanf("%ld",&n);
        for(i=1;i<=sqrt(n);i++)
        {
            if(n%i==0)
             {
                c = c+check(i);
                c=c+check(n/i); }
        }
        printf("%ld\n",c);
    }
    return 0;
}
