#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
bool isLucky(long long n)
{
	//cout<<"Coming here "<<n<<endl;
	while(n>0)
	{
		if(n%10==7 or n%10==4){return true;}
		n/=10;
	}
	return false;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		long long c=0;
		for(int x=1;x*x<=n;x++)
		{
			if(n%x==0)
			{
				//cout<<"checking lucky"<<x<"\n";
				if(isLucky(x))
				{
					c++;
				}
				if(x!=n/x)
				{
					if(isLucky(n/x))
					{
					c++;
					}
				}
			}
		}
		cout<<c<<endl;
	}
	
}
