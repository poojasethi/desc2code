#include <iostream>
int islucky(long long int num)
{ long long int r;
  while(num>0)
  {
  r=num%10;
  if((r==4)||(r==7))
  return 1;
  else
  num=num/10;
 }
 return 0;
}
int main()
{ long long int t,n,i,y,count;
 scanf("%lld",&t);
  while(t--)
  { count=0;
    scanf("%lld",&n);
    for(i=1;i*i<n;i++)
    { 
      if((n%i)==0)
      {
       if(islucky(i))
       count+=1;
      
       y=n/i;
       if(i!=y)
       {
       if(islucky(y))
       count+=1;
       }
      }
      
    }
    printf("%lld\n",count);
  }
  
  return 0;
}
