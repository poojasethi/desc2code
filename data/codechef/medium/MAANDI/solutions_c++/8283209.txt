#include<stdio.h>
#include<math.h>

int main()
{
	int i,j,k,count,t,n,a,z,flag;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		k= sqrt(n);
		count=0;
		for(i=1;i<=k;i++)
		{
			if(n%i==0)
			{
				a=n/i;
				for(z=0;z<=1;z++)
				{
					flag=0;
					while(a)
				{
					if(a%10==4 || a%10==7)
					{
						flag=1;
						break;
					}
					a/=10;
				}
					if(flag==1)
					count++;
					a=i;
				}
			}
			
		}
		printf("%d\n",count);
	}
	return 0;
}
