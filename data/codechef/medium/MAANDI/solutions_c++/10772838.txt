#include <cstdio>

inline bool test(int x)
{
    while (x > 0) {
        int dig = x % 10;
        if (dig == 4 || dig == 7) {
            return true;
        }
        x /= 10;
    }
    return false;
}

int main()
{
    int T;
  scanf("%d",&T);
    while(T--)
    {
        int n;
        scanf("%d", &n);
        int answer = 0;
        for (int d = 1; d * d <= n; ++ d) {
            if (n % d == 0) {
                answer += test(d);
                if (d * d < n) {
                    answer += test(n / d);
                }
            }
        }
        printf("%d\n", answer);
    }
    return 0;
}
