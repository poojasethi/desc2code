#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<cmath>
#include<set>
#include<map>
#include<string>
#include<algorithm>
#define all(c) c.begin(),c.end()
using namespace std;
int main()
{
    int i,n,t,ans,temp,r,flag;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        temp=n;
        flag=ans=0;
        while(temp>0)
        {
            r=temp%10;
            if((r==4)||(r==7))
            {
                flag=1;
                break;
            }
            temp=temp/10;
        }
        if(flag)
            ans++;
        for(i=2;i<=sqrt(n);i++)
        {
            if(n%i==0)
            {
                temp=i;
                flag=0;
                while(temp>0)
                {
                    r=temp%10;
                    if((r==4)||(r==7))
                    {
                        flag=1;
                        break;
                    }
                    temp=temp/10;
                }
                if(flag)
                    ans++;
                if(i!=(n/i))
                {
                    temp=n/i;
                    flag=0;
                    while(temp>0)
                    {
                        r=temp%10;
                        if((r==4)||(r==7))
                        {
                            flag=1;
                            break;
                        }
                        temp=temp/10;
                    }
                        if(flag)
                            ans++;
                }
            }
        }
        printf("%d\n",ans);
    }
    return 0;
}
