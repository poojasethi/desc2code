#include<bits/stdc++.h>
#define lli long long int
using namespace std;
int check(lli n)
{
	lli n1;
	while(n!=0)
	{
		n1=n%10;
		if(n1==4 || n1==7)
			return 1;
		n=n/10;
	}
	return 0;
}
int main()
{
	lli f1,f2,i,j,k,m,n,t,count;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		if(n==1)
			printf("0\n");
		else
		{
			count=0;
			for(i=1;i<=pow(n,0.5);i++)
			{
				if(n%i==0)
				{
					f1=i;
					f2=n/i;
					if(f1==f2 && check(f1))
						count++;
					else
					{
						if(check(f1))
							count++;
						if(check(f2))
							count++;
					}
				}
			}
			printf("%lld\n",count);
		}
	}
}
