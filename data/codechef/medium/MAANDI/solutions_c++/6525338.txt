#include<bits/stdc++.h>
using namespace std;
int chk(int n)
{
	while(n)
	{
		if(n%10==7 || n%10==4)
		return 1;
		n/=10;
	}
	return 0;
}
main()
{
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		int cnt=0,n,i;
		scanf("%d",&n);
		for(i=1;i*i<n;i++)
		{
			if(n%i==0)
			{
				if(chk(i))cnt++;
				if(chk(n/i))cnt++;
			}
			
		}
		if(i*i==n && chk(i))cnt++;
		printf("%d\n",cnt);
	}
}
