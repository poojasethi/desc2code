#include<stdio.h>
#include<iostream>
int check(int n)
{ while(n)
      { if(n%10==7 || n%10==4)
          { return 1;
          }
        n/=10;  
      }
  return 0;    
}
int main()
{ int t;
  scanf("%d",&t);
  while(t--)
       { int n,total=0;
         scanf("%d",&n);
         for(int i=1;i*i<=n;i++)
            { if(n%i==0)
                { total+=check(i);
                  total+=check(n/i);
                }
            }
         printf("%d\n",total);
       }
}
