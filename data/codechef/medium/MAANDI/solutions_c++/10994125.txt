#include <iostream>
#include <cmath>
using namespace std;


int main() {
    int t;
cin >>t;


for( int j=1 ; j<=t; j++)
  {
   
   long long int n , k;    
  cin >>n;
  long long int x,y,count=0;
  int m ,f;
  
  for(long long int i=1; i<sqrt(n); i++)
  {
      if(n%i==0)
      {
          x=i;
        while( x!=0   )
        {
         m = x%10;
         
         if(m==4 || m==7)
         {
             count++;
             break;
         }
         
             
         x = x/10 ;
        }
        
        
        
        k = n /i;
        if(k!=i)
        {
            y=k;
        
         while(   y!=0  )
        {
         f =y%10;
         
         if(f==4 || f==7)
         {
             count++;
             break;
         }
         
          y = y/10 ;   
         
        }
        
           
            
        }
          
          
      }
      
      
  }
  cout << count << endl;
   
   
  
  }
    
    return 0;
}
