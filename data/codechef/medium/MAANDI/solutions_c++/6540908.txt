#define sc(x) scanf("%d",&x)
#define sl(x) scanf("%lld",&x)
#define pf(x) printf("%d\n",x)
#define pl(x) printf("%lld\n",x)

using namespace std;
 
bool f(int a)
{
	while(a)
	{
		if(a%10==4 || a%10==7)
		{
			return true;
		}
		a/=10;
	}
	return 0;
}
 
int main()
{
	int t,n,ans;
	sc(t);
	while(t--)
	{
		sc(n);
		ans=0;
		for(int i=1;i<=sqrt(n);i++)
		{
			
			if(n%i==0)
			{
				if(f(i))ans++;
				if(i!=n/i)
				{
					if(f(n/i))
					ans++;
				}
			}
		}
		pf(ans);
	}
	return 0;
}
