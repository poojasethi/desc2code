#include <cstdio>
#include <cstring>
using namespace std;

int main()
{
    int t, n;
    scanf("%i", &t);
    while(t--)
    {
    	scanf("%i", &n);
    	char board[n+1][n+1];
    	int values[n][n], max = 0;
    	memset(values, 0, sizeof(values));
    	for(int i = 0; i < n; i++)
    	{
    		scanf("%s", board[i]);
    	}
    	for(int i = 0; i < n; i++)
    	{
    		for(int j = 0; j < n; j++)
    		{
    			if(board[j][i] == 'K')
    			{
    				values[j][i] = 1;
    			}
    			if(!values[j][i])
    			{
    				continue;
    			}
    			if(j+2 < n && i+1 < n)
    			{
    				int x = (board[j+2][i+1] == 'P') ? 1 : 0;
    				if(values[j][i]+x > values[j+2][i+1])
    				{
    					values[j+2][i+1] = values[j][i]+x;
    				}
    			}
    			if(j-2 >= 0 && i+1 < n)
    			{
    				int x = (board[j-2][i+1] == 'P') ? 1 : 0;
    				if(values[j][i]+x > values[j-2][i+1])
    				{
    					values[j-2][i+1] = values[j][i]+x;
    				}
    			}
    			if(j+1 < n && i+2 < n)
    			{
    				int x = (board[j+1][i+2] == 'P') ? 1 : 0;
    				if(values[j][i]+x > values[j+1][i+2])
    				{
    					values[j+1][i+2] = values[j][i]+x;
    				}
    			}
    			if(j-1 >= 0 && i+2 < n)
    			{
    				int x = (board[j-1][i+2] == 'P') ? 1 : 0;
    				if(values[j][i]+x > values[j-1][i+2])
    				{
    					values[j-1][i+2] = values[j][i]+x;
    				}
    			}
    		}
    	}
    	for(int i = 0; i < n; i++)
    	{
    		if(values[i][n-1] > max)
    		{
    			max = values[i][n-1];
    		}
    	}
    	printf("%i\n", max-1);
    }
    return 0;
}
