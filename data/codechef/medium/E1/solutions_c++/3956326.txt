#include <iostream>
using namespace std;
main()
{
	int n,x,y,t,i,j,k,l,m,a[1000][1000],b[1000][1000];
	cin>>t;
	string s;
	while(t--)
	{
		
		m=0;
		cin>>n;
		for(i=0;i<n;i++)
			for(j=0;j<n;j++)
				{a[i][j]=0;b[i][j]=-1;}
		for(i=0;i<n;i++)
		{
			m=0;
			cin>>s;
			for(j=0;j<s.length();j++)
			{
				if(s[j]=='P')
					a[i][j]=1;
				if(s[j]=='K')
				{
					x=i;
					y=j;
				}
			}
		}
			for(i=n-1;i>=y;i--)
			{
				for(j=0;j<n;j++)
				{
					m=0;
					if(j-2>=0 && i+1<n)
						if(b[j-2][i+1]>m)
						m=b[j-2][i+1];
					if(j-1>=0 && i+2<n)
						if(b[j-1][i+2]>m)
						m=b[j-1][i+2];
					if(j+1<n && i+2<n)
						if(b[j+1][i+2]>m)
						m=b[j+1][i+2];
					if(j+2<n && i+1<n)
						if(b[j+2][i+1]>m)
						m=b[j+2][i+1];
					b[j][i]=m+a[j][i];
				}
			}

	/*	for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
				cout<<b[i][j]<<a[i][j]<<" ";
			cout<<endl;
		}*/
			cout<<b[x][y]<<endl;
	}
	return 0;
}
