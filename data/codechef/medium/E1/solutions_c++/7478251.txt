#include<iostream>
#include<algorithm>

using namespace std;

int dp[1001][1001];

int main()
{
    int t, n, i, j, pos, val, max1, ans;
    char ch[1001][1001];
    cin>>t;

    while(t--)
    {
        cin>>n;
        val = 0;
        max1 = 0;
        for(i = 0; i < n; i++)
        {
            cin>>ch[i];
        }
        for(i = 0; i < n; i++)
        {
            for(j = 0; j < n; j++)
            {
                if(ch[i][j] == 'K')
                    pos = j;
                dp[i][j] = 0;
            }
        }
        if(pos == n-1)
        {
            cout<<0<<"\n";
            continue;
        }

        /*for(i = 0; i < n; i++)
        {
            if(ch[i][n-1] == 'P')
            {
                dp[i][n-1] = 1;
                val = 1;
            }

        }*/

        for(i = n-1; i >= pos; i--)
        {
            val = 0;
            for(j = 0; j < n; j++)
            {
                if(j-2 >= 0 && i+1 < n)
                {
                    val = max(val, dp[j-2][i+1]);
                }

                if(j+2 < n && i+1 < n)
                {
                    val = max(val, dp[j+2][i+1]);
                }

                if(i+2 < n && j-1 >= 0)
                {
                    val = max(val, dp[j-1][i+2]);
                }

                if(i+2 < n && j+1 < n)
                {
                    val = max(val, dp[j+1][i+2]);
                }
                if(ch[j][i] == 'P')
                    val++;
                dp[j][i] = val;
                if(ch[j][i] == 'K')
                    ans = dp[j][i];
                //max1 = max(max1, dp[j][i]);
                val = 0;
            }
        }
        cout<<ans<<"\n";

        /*for(i = 0; i < n; i++)
        {
            for(j = 0; j < n; j++)
            {
                cout<<dp[i][j]<<" ";
            }
            cout<<"\n";
        }*/

    }
    return 0;
}
