#include <iostream>

using namespace std;

int max(int a,int b,int c,int d)
{
	int ans1,ans2;
	ans1=a>b?a:b;
	ans2=c>d?c:d;
	return ans1>ans2?ans1:ans2;
}

int main()
{
	int i,test,j,a,b,n,flag,ans,x1,x2,x3,x4,mx;
	char arr[1001][1001];
	cin >> test;
	while(test--)
	{
		cin >> n;
		int brr[n][n];
		for(i=0;i<n;i++)
			cin >> arr[i];
/*		
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				cout << arr[i][j];
			}
			cout << "\n";
		}
*/
		flag=1;
		for(i=0;i<n && flag;i++)
		{
			for(j=0;j<n && flag;j++)
			{
				if(arr[i][j]=='K')
				{
					a=i;
					b=j;
					flag=0;
					break;
				}
			}
		}
		
		for(i=0;i<n;i++)
		{
			if(arr[i][n-1]=='P')
				brr[i][n-1]=1;
			else
				brr[i][n-1]=0;
		}
		x1=0;x2=0;x3=0;x4=0;mx=0;
		for(j=n-2;j>=0;j--)
		{
			for(i=0;i<n;i++)
			{
				if((i+1)<n && (j+2)<n)
					x1=brr[i+1][j+2];
			//	if(i==0 && j==0)
			//		cout << "x1" << x1 << endl;
				if((i+2)<n && (j+1)<n)
					x2=brr[i+2][j+1];
				if((i-1)>=0 && (j+2)<n)
					x3=brr[i-1][j+2];
				if((i-2)>=0 && (j+1)<n)
					x4=brr[i-2][j+1];
				mx=max(x1,x2,x3,x4);
				if(arr[i][j]=='P')
					mx++;
				brr[i][j]=mx;
			}
		}
		ans=brr[a][b];
		cout << ans << endl;
	}
	
	return 0;
}
