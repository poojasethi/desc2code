#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int dp[1010][1010];
main()
{   std:ios_base::sync_with_stdio(false);
    int i,j,t,n,ik,jk,ans=-1,row,col,tmp;
    char c;
    cin>>t;
    while(t--)
    {   ans=-1;
    	cin>>n;
    	string s[n];
    	for(i=0;i<n;i++)
    	{   s[i]="";
    		for(j=0;j<n;j++)
    		{
    		  cin>>c;
    		  if(c=='K')
    		  {
    		  	ik=i;
    		  	jk=j;
    		  }
			  s[i]+=c;	
    		}
    	}
    	if(jk==n-1) printf("0\n");
    	else
    	{
    		row=n; col=n;
    		for(i=0;i<row;i++)
    			for(j=0;j<col;j++)
    			dp[i][j]=0;
    		dp[ik][jk]=1;
    		for(j=1;j<col;j++)
    		{
    			for(i=0;i<row;i++)
    			{   
    			   tmp=0;
    			   if(i-1>=0 && j-2>=0) tmp=max(tmp,dp[i-1][j-2]);
    			   if(i+1<n && j-2>=0)  tmp=max(tmp,dp[i+1][j-2]);
    			   if(i-2>=0 && j-1>=0) tmp=max(tmp,dp[i-2][j-1]);
    			   if(i+2<n && j-1>=0)  tmp=max(tmp,dp[i+2][j-1]);
    			   if(tmp)
    			   {
    			     dp[i][j]=tmp;
    			     if(s[i][j]=='P')
    			       dp[i][j]+=1;	
    			   }
    			}
    		}
    		for(i=0;i<row;i++)
    			ans=max(ans,dp[i][col-1]);
    			
    		if(ans==0)
    		ans=1;
    		printf("%d\n",ans-1);
    	}
    }
	return 0;
}
