#include<bits/stdc++.h>
using namespace std;
int main()
{
	cin.sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		char chess[1005][1005];
		int board[1005][1005];
		int n,flag=0;
		cin>>n;
		for(int i=0; i<n; i++)
		{
			cin>>chess[i];
		}
		for(int c=n-1; c>=0; c--)
			{
				for(int r=0; r<n; r++)
				{
				int best=0;
				int curVal=(chess[r][c]=='P'?1:0);
				if(c+2<n && r+1<n)
				{
					best=max(board[r+1][c+2],best);
				}
				if(c+1<n && r+2<n)
				{
					best=max(board[r+2][c+1],best);
				}
				if(c+2<n && r-1>=0)
				{
					best=max(board[r-1][c+2],best);
				}
				if(c+1<n && r-2>=0)
				{
					best=max(board[r-2][c+1],best);
				}
				board[r][c]=best+curVal;
				if(chess[r][c]=='K')
				{
					cout<<board[r][c]<<"\n";
					flag=1;
					break;
					
				}
				
				}
			if(flag)
			break;
			}
	
	}return 0;
}
