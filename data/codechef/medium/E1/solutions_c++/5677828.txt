#include<cstdio>
#include<iostream>
#include<cstdlib>
using namespace std;
int n;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int dp[1001][1001]={0};
        int i,j,x,y;
        scanf("%d",&n);
        char a[1001][1001];
        for(i=0;i<n;i++)
        {
            scanf("%s",a[i]);
            for(j=0;j<n;j++)
            {
                if(a[i][j]=='K')
                {
                    x=i;
                    y=j;
                }

            }
        }
        int e,b,c,d;
        for(j=n-1;j>=y;j--)
        {
            for(i=0;i<n;i++)
            {
                dp[i][j]=0;
                b=0;
                c=0;
                d=0;
                e=0;
                if(i+2<n&&j+1<n)
                {
                    b=dp[i+2][j+1];
                    //cout<<"i and j "<<i<<" "<<j<<" "<<dp[i+2][j+1]<<endl;
                }
                if(i+1<n&&j+2<n)
                {
                    c=dp[i+1][j+2];
                    //cout<<"i and j "<<i<<" "<<j<<" "<<dp[i+1][j+2]<<endl;
                }
                if(i-1>=0&&j+2<n)
                {
                    d=dp[i-1][j+2];
                    //cout<<"i and j "<<i<<" "<<j<<" "<<dp[i-2][j+2]<<endl;
                }
                if(i-2>=0&&j+1<n)
                {
                    e=dp[i-2][j+1];
                    //cout<<"i and j "<<i<<" "<<j<<" "<<dp[i-2][j+1]<<endl;
                }
                dp[i][j]=max(max(max(e,b),c),d);
                if(a[i][j]=='P')
                    dp[i][j]++;
            }
            //cout<<endl;
        }
        printf("%d\n",dp[x][y]);
    }
    return 0;
}
