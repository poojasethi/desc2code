#include<stdio.h>
#include<iostream>
#include<sstream>
#include<math.h>
#include<string.h>
using namespace std;
string s[1002];
long long dp[1002][1002],n,m;
int recur(long long i,long long j)
{
    int flag,ans;
    if(i>=0&&i<n&&j>=0&&j<n)
    {
        if(dp[i][j]!=-1)
            return dp[i][j];
        else
        {
            int x=recur(i-2,j+1);
            int y=recur(i+2,j+1);
            int u=recur(i+1,j+2);
            int w=recur(i-1,j+2);
            if(s[i][j]=='P')
                flag=1;
            else
                flag=0;
            ans=max(max(x,y),max(w,u));
            dp[i][j]=ans+flag;
            return dp[i][j];
        }
    }
    else
        return 0;
}

int main()
{
    long long t,i1,j1,i,j;
    scanf("%lld",&t);
    while(t--)
    {
        m=0;
        scanf("%lld",&n);
        cin.ignore();
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
                dp[i][j]=-1;
        }
        for(i=0;i<n;i++)
            cin>>s[i];
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {
                if(s[i][j]=='K')
                {
                    i1=i;
                    j1=j;
                    break;
                }
            }
        }
        recur(i1,j1);
        for(i=0;i<n;i++)
        {
            for(j=0;j<n;j++)
            {//cout<<dp[i][j]<<" ";
                if(m<dp[i][j])
                m=dp[i][j];
            }//cout<<"\n";
        }
        printf("%lld\n",m);
    }
    return 0;
}
