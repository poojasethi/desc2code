#include <bits/stdc++.h>

using namespace std;

int  main()
{
	int t;
	scanf("%d",&t);
	getchar();
	while (t--)
	{
		string s;
		getline(cin,s);
		int len = s.size();
		if (len == 5 || len == 8) {
			int f1 = 0;
			for (int i = 0; i < len; i++)
			{
				if (i%3 == 0)
				{
					if (s[i] >= 'a' && s[i] <= 'h')
						continue;
					else
					{
						f1 = 1;
						break;
					}
				}
				else if (i%3 == 1)
				{
					if (s[i] >= '1' && s[i] <= '8')
						continue;
					else
					{
						f1 = 1;
						break;
					}
				}
				else if (i%2 == 2)
				{
					if (s[i] != '-') {
						f1 = 1;
						break;
					}
						
				}
			}
			if (f1)
			{
				cout << "Error\n";
				continue;
			}
			int f = 0,x1,y1,x2,y2;
			for (int i = 0; i < len-3;)
			{
				if (s[i] >= 'a' && s[i] <= 'h') {
					x1 = s[i];
					if (s[i+1] >= '1' && s[i+1] <= '8')
						y1 = s[i+1]-48;
					else
					{
						cout << "Error\n";
						f = 1;
						break;
					}
					if (s[i+2] == '-')
					{
						if (s[i+3] >= 'a' && s[i+3] <= 'h')
							x2 = s[i+3];
						else
						{
							cout << "Error\n";
							f = 1;
							break;
						}
						if (s[i+4] >= '1' && s[i+4] <= '8')
							y2 = s[i+4]-48;
						else
						{
							cout << "Error\n";
							f = 1;
							break;
						}
						int d1 = abs(x1-x2);
						int d2 = abs(y1-y2);
						if ((d1 == 1 && d2 == 2) || (d2 == 1 && d1 == 2))
						{
							i = i+3;
							continue;
						}
						else
						{
							cout << "No\n";
							f = 1;
							break;
						}
						
					}
					else {
						cout << "Error\n";
						f = 1;
						break;
					}
				}
				else
				{
					cout << "Error\n";
					f = 1;
					break;
				}		
			}
			if (f == 0)
				cout << "Yes\n";
		}
		else
		{
			cout << "Error\n";
		}
	}
	return 0;
}
