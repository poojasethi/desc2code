#include<bits/stdc++.h>
#define ll long long
using namespace std;


int main()
{	
	ll t;string s;
	bool t1,t2,t3;
	scanf("%lld",&t);
	cin.get();
	while(t--)
	{
		getline(cin,s);
		if(s.size()!=5)
		cout<<"Error\n";
		else
		{
			t1=s[0]>='a' && s[0]<='h' && s[3]>='a' && s[3]<='h';
			t2=s[1]>='1' && s[1]<='8' && s[4]>='1' && s[4]<='8';
			t3=(s[2]=='-');
			if(!(t1 && t2 && t3))
			cout<<"Error\n";
			else
			{
				if(abs(s[0]-s[3])==2)
				{
					if(abs(s[1]-s[4])==1)
					cout<<"Yes\n";
					else
					cout<<"No\n";
				}
				else if(abs(s[1]-s[4])==2)
				{
					if(abs(s[0]-s[3])==1)
					cout<<"Yes\n";
					else
					cout<<"No\n";
				}
				else
				cout<<"No\n";
			}
		}
	}
	return 0;
}
