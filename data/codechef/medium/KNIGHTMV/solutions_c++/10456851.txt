#include<bits/stdc++.h>
using namespace std;
int main()
{
int t,num,count,x,len;
char str[11];
cin>>t;
getchar();
while(t--){
bool flag=false,check=true;
gets(str);
len=strlen(str);
if(len<=5)
{
if( (str[0]>='a'&&str[0]<='h') && (str[3]>='a'&& str[3]<='h') && (str[2]=='-') && str[1]>='1'&&str[1]<='8' && str[4]>='1' && str[4]<='8')
    {
        if(abs(str[0]-str[3])==1 && abs(str[1]-str[4])==2)
            flag=true;
        else if(abs(str[0]-str[3])==2 && abs(str[1]-str[4])==1)
            flag=true;
    }
    else
        check=false;
}
else
check=false;

if(!check)
    cout<<"Error\n";
else if(flag)
    cout<<"Yes\n";
else
    cout<<"No\n";
}
return 0;
}
