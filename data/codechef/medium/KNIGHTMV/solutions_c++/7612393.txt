#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

int abs(int a)
{
    if(a<0)
        return -1*a;
    else
        return a;
}

void process()
{
    int t,diff1,diff2;
    cin >> t;
    char a[15];
    getchar();
    while(t--)
    {
        gets(a);
        if((a[0] >= 'a' && a[0] <= 'h') && (a[3] >= 'a' && a[3] <= 'h') && (a[1] >= '1' && a[1] <= '8') && (a[4] >= '1' && a[4] <= '8') && a[5] == '\0' && a[2] == '-')
        {
            diff1 = a[0] - a[3];
            diff2 = a[1] - a[4];
            if((abs(diff1) == 1 && abs(diff2) == 2) || (abs(diff1) == 2 && abs(diff2) == 1))
                puts("Yes");
            else
                puts("No");
        }
        else
            puts("Error");
    }
}

int main()
{
    process();

}
