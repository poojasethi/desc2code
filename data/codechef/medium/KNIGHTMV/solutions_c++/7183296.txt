#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include <algorithm>
#include <math.h>

int main()
{
#ifdef USE_DATA_TXT
  freopen_s(&stream, "c:\\data.txt", "r", stdin);
#endif
#ifdef PRINT_PERF
  auto start = std::chrono::high_resolution_clock::now();
#endif

  int T;
  scanf("%d", &T);
  getchar();

  char s[16];

  while(T--)
  {
    gets(s);

    if((s[5] != '\0') || (s[2] != '-') || (s[0] < 'a') || (s[0] > 'h') || (s[3] < 'a') || (s[3] > 'h') || (s[1] < '1') || (s[1] > '8') || (s[4] < '1') || (s[4] > '8'))
    {
      printf("Error\n");
    }
    else
    {
      const int h1 = s[0] - 'a';
      const int v1 = s[1] - '1';
      const int h2 = s[3] - 'a';
      const int v2 = s[4] - '1';

      int hdiff = h2 - h1;
      if(hdiff < 0)
        hdiff = -hdiff;
      int vdiff = v2 - v1;
      if(vdiff < 0)
        vdiff = -vdiff;
      if(((vdiff == 2) && (hdiff == 1)) || ((hdiff == 2) && (vdiff == 1)))
        printf("Yes\n");
      else
        printf("No\n");
    }
  }

#ifdef PRINT_PERF
  PrintResultPerf(start);
#endif
#ifdef USE_DATA_TXT
  if(stream)
  {
    fclose(stream);
    stream = NULL;
  }
#endif

  return 0;
}
