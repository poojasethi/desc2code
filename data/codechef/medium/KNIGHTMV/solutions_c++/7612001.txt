#include<stdio.h>
#include<string.h>
using namespace std;
int main()
{
    int l,i;
    long int t;
    char s[11],ch;
    scanf("%ld",&t);
    ch=getchar();
    while(t--)
    {
        gets(s);
        l=strlen(s);
        if(l!=5 || s[2]!='-')
        {
            printf("Error\n");
            continue;
        }
        if(s[0]<'a' || s[0]>'h' || s[3]<'a' || s[3]>'h')
        {
            printf("Error\n");
            continue;
        }
        if(s[1]<'1' || s[1]>'8' || s[4]<'1' || s[4]>'8')
        {
            printf("Error\n");
            continue;
        }
        if(s[3]==s[0]+1 || s[3]==s[0]-1)
        {
            if(s[1]+2==s[4] || s[1]-2==s[4])
                printf("Yes\n");
            else
                printf("No\n");
            continue;
        }
        else if(s[3]==s[0]+2 || s[3]==s[0]-2)
        {
            if(s[1]+1==s[4] || s[1]-1==s[4])
                printf("Yes\n");
            else
                printf("No\n");
            continue;
        }
        else
            printf("No\n");
    }
    return 0;
}
