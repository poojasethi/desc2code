#include<bits/stdc++.h>
using namespace std;
typedef long long lli;
int main()
{
    int t,ctr=0;
    scanf("%d",&t);
    t++;
    while(t--)
    {
        ctr++;
        string str;
        fflush(stdin);
        getline(cin,str);
        int l=str.length();
        if(ctr!=1&&l==5&&str[2]=='-'&&(str[0]>='a'&&str[0]<='h')&&(str[3]>='a'&&str[3]<='h')&&(str[1]>='1'&&str[1]<='8')&&(str[4]>='1'&&str[4]<='8'))
        {
            if((abs(str[0]-str[3])==2&&abs(str[1]-str[4])==1)||(abs(str[0]-str[3])==1&&abs(str[1]-str[4])==2))
                printf("Yes\n");
            else
                printf("No\n");
        }
        else if(ctr!=1)
        {
            printf("Error\n");
        }
    }
    return 0;
}
