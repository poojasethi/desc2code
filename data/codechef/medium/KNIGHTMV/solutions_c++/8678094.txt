#include <bits/stdc++.h>
using namespace std;
int main(void) {
	long long int t,x,y;
	char a[10];
	scanf("%lld",&t);
	gets(a);
	while(t--){
	    char str[10];
	    gets(str);
	    if((strlen(str)==5) && (str[0]>='a' && str[0]<='h') && (str[1]>='1' && str[1]<='8') && (str[3]>='a' && str[3]<='h') && (str[4]>='1' && str[4]<='8') && (str[2]=='-'))
	    {
	        x=abs(str[0]-str[3]);
	        y=abs(str[1]-str[4]);
	    if(x==2 && y==1)
	        printf("Yes\n");
	    else if(x==1 && y==2)
	        printf("Yes\n");
	    else
	        printf("No\n");
	
	    }
	    else
	        printf("Error\n");
	    }
	return 0;
}
