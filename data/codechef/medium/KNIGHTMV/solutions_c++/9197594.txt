#include <bits/stdc++.h>
#include <stdio.h>
using namespace std;
bool check(string s)
{
	if(s[0]>='a' && s[0]<='h' && s[1]>='1' && s[1]<='8' && s[2]=='-' && s[3]>='a' && s[3]<='h' && s[4]>='1' && s[4]<='8' && s.length()==5 )
	return true;
	else
	return false;
}

int main()
{
	//ios_base::sync_with_stdio(false);
	//freopen("output1.txt","w",stdout);
	int t; scanf("%d",&t); getchar();
	while(t--)
		{
			 string s;
			 getline(cin,s);
			if(check(s)==false) 
			 {  //printf("%s\n",s.c_str());
			 	printf("Error\n");
			 }
		    else
			 {
				int a,b,c,d;
				a=s[0]-'a'+1;
				b=s[1]-'0';
				c=s[3]-'a'+1;
				d=s[4]-'0';
				int flag=0;
				if(abs(a-c)==1 && abs(b-d)==2) flag=1;
                if(abs(a-c)==2 && abs(b-d)==1) flag=1;
                if(flag) 
                  printf("Yes\n");
                else
                  printf("No\n");
			 }
		}
	return 0;
}
