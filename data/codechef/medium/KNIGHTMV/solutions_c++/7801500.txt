#include <iostream>
#include <cstring>
#include <cstdio>
 
using namespace std;
 
int main()
{
    int t;
    char ch, a[15];
 
    cin >> t;
    cin.getline(a,15);
 
    while (t--) {
        int i, flag, sum, t1, t2;
 
        cin.getline(a, 15);
 
        flag = 1;
        sum = 0;
        if (strlen(a) != 5)
            flag = 0;
 
        else {
            if (a[0] >= 'a' && a[0] <= 'h' && a[3] >= 'a' && a[3] <= 'h' && a[2] == '-' && a[1] >= '1' && a[1] <= '8' && a[4] >= '1' && a[4] <= '8') {
                t1 = a[0] - a[3];
                if (t1 < 0 && t1 >= -2)
                    sum -= t1;
 
                else if (t1 > 0 && t1 <= 2)
                    sum += t1;
 
                else
                    sum = 10;
 
                t1 = a[1] - a[4];
                if (t1 < 0 && t1 >= -2)
                    sum -= t1;
 
                else if (t1 > 0 && t1 <= 2)
                    sum += t1;
 
                else
                    sum = 10;
            }
 
            else
                flag = 0;
        }
 
        if (flag == 1 && sum == 3)
            cout << "Yes\n";
 
        else if (flag == 1 && sum != 3)
            cout << "No\n";
 
        else
            cout << "Error\n";
    }
 
    return 0;
}
