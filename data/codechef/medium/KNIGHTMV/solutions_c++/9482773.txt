#include <iostream>
#include <stdio.h>

#define abs(a) ((a > 0) ? (a) : -(a))

using namespace std;

int main(){
	int no_cases;
	cin >> no_cases;
	getchar();
	for (int test_case = 1; test_case <= no_cases; test_case++){
		char str[11];
		cin.getline(str, 11);
		//cout << str << endl;
		if (str[0] < 'a' || str[0] > 'h' || str[1] < '1' || str[1] > '8' || str[2] != '-' || str[3] < 'a' || str[3] > 'h' || str[4] < '1' || str[4] > '8' || str[5] != '\0'){
			cout << "Error" << endl;
			continue;
		}
		int a = abs(str[0] - str[3]);
		int b = abs(str[1] - str[4]);
		if ((a == 2 && b == 1) || (a == 1 && b == 2))
			cout << "Yes" << endl;
		else
			cout << "No" << endl;
	}
	return 0;
}
