int checkstring(char *array,int n)
{

	if(((array[0]>='a') && (array[0]<='h')) && ((array[1]>='1') && (array[1]<='8')) && (array[2]=='-') && ((array[3]>='a') && (array[3]<='h')) && ((array[4]>='1') && (array[4]<='8')) && strlen(array)==5)
	return 1;
	else
	return 0;
}
int checkmove(char f1,char s1,char f2,char s2 )
{
	if((abs(f1-f2))==1&&(abs(s1-s2)==2)||(abs(f1-f2))==2&&(abs(s1-s2)==1))
	return 1;
	else
	return 0;
}
