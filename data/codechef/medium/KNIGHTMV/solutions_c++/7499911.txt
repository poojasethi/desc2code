    #include <stdio.h>
    #include <string.h>
     
    int main()
    {
    	int c,r;
    	char str[11];
    	long T,i;
     
    	scanf("%ld",&T);
    	getchar();
     
    	for(i=0;i<T;i++)
    	{
    		gets(str);
     
    		if(str[5]=='\0' && str[0]>='a' && str[0]<='h' && str[3]>='a' && str[3]<='h' && str[2]=='-' && str[1]>='1' && str[1]
     
    <='8'&& str[4]>='1' && str[4]<='8')
    		{
    			c=str[0]-str[3];
    			if(c<0)
    				c=-c;
    			r=str[1]-str[4];
    			if(r<0)
    				r=-r;
    			if((c==1 && r==2) || (c==2 && r==1))
    			{
    				printf("Yes\n");
    			}
    			else
    			{
    				printf("No\n");
    			}
     
    		}
    		else
    		{
    			printf("Error\n");
    		}
    	}
     
    	return(0);
    }
