// Correctness of Knight Move.c
//
#include <stdio.h>
#include <stdlib.h>
 
#ifndef ONLINE_JUDGE
#define getcx getchar
#else
#define getcx getchar_unlocked
#endif
 
inline long getint (void) {
	long x = 0; char ch = getcx();
	while (ch<48 || 57<ch)
		ch = getcx();
	while (48<=ch && ch<=57) {
		x = x*10+ch-48;
		ch = getcx();
	}
	return x;
}
 
char ch, a, b, u, v;
long res;
 
void EndLine (void) {
	while (ch>=32) {
		ch = getcx();
	}
}
 
int main (void) {
#ifndef ONLINE_JUDGE
	freopen( "TESTS\\Correctness of Knight Move.TXT", "r", stdin );
#endif
 
	long t = getint();
 
	while (t--) {
		res = 1;
		ch = getcx();
		if ('a'<=ch && ch<='h') {
			a = ch;
			ch = getcx();
			if ('1'<=ch && ch<='8') {
				b = ch;
				ch = getcx();
				if (ch=='-') {
					ch = getcx();
					if ('a'<=ch && ch<='h') {
						u = ch;
						ch = getcx();
						if ('1'<=ch && ch<='8') {
							v = ch;
							ch = getcx();
							if (ch>=32) {
								res = 0;
								EndLine();
							} else if ((abs( a-u )==1 && abs( b-v )==2) || (abs( a-u )==2 && abs( b-v )==1)) {
								res = 2;
							}
						} else {
							EndLine(); res = 0;
						}
					} else {
						EndLine(); res = 0;
					}
				} else {
					EndLine(); res = 0;
				}
			} else {
				EndLine(); res = 0;
			}
		} else {
			EndLine(); res = 0;
		}
 
		if (res==1) {
			puts( "No" );
		} else if (res==2) {
			puts( "Yes" );
		} else {
			puts( "Error" );
		}
	}
 
	return 0;
}
