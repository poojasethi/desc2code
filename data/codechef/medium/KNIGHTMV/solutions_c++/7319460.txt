#include <iostream>
#include <cmath>

using namespace std;

int main() {
    int t;
    string str;
    cin >> t;
    cin.get();
    while(t --) {
        getline(cin, str);
        if(str.size() != 5 || (str[0] < 'a' || str[0] > 'h') || (str[3] < 'a' || str[3] > 'h') || (str[1] < '1' || str[1] > '8') || (str[4] < '1' || str[4] > '8' || str[2] != '-')) {
            cout << "Error" << endl;
        } else {
            if((abs((double)(str[0] - str[3])) == (double)2 && abs((double)(str[1] - str[4])) == (double)1) || (abs((double)(str[0] - str[3])) == (double)1 && abs((double)(str[1] - str[4])) == (double)2)) {
                cout << "Yes" << endl;
            } else {
                cout << "No" << endl;
            }
        }
    }
    return 0;
}
