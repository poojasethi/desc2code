#include<bits/stdc++.h>
using namespace std;
bool charcrt(char c)
{
    return (c>='a'&&c<='h');
}
bool numcrt(char c)
{
    return (c>='1'&&c<='8');
}
int main()
{
    int t;
    scanf("%d",&t);
    getchar();
    while(t--)
    {
        char arr[151];
        gets(arr);
       // cout<<arr.length()<<" ";
        if((strlen(arr)==5)&&(charcrt(arr[0]))&&(charcrt(arr[3]))&&(numcrt(arr[1]))&&(numcrt(arr[4]))&&(arr[2]=='-'))
        {
            //cout<<arr.at(0)-arr.at(3)<<" "<<arr.at(1)-arr.at(4)<<endl;
            if(abs((arr[0]-arr[3])*(arr[1]-arr[4]))==2) puts("Yes");
            else puts("No");
        }
        else puts("Error");
    }
}
