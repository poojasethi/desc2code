#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
using namespace std;
bool check(char move[10])
{
	if(move[0]<'a' || move[0]>'h' || move[3]<'a' || move[3]>'h')
		return false;
	if(move[1]<'1' || move[1]>'8' || move[4]<'1' || move[4]>'8')
		return false;
	if(move[2]!='-')
		return false;
	return true;
}
int main()
{
	int t;
	for(scanf("%d", &t), getchar(); t--; )
	{
		char move[111];
		gets(move);
		if(strlen(move)!=5){
			printf("Error\n");
		}
		else if(!check(move)){
			printf("Error\n");
		}
		else{
			if((abs(move[0]-move[3])==1 && abs(move[1]-move[4])==2) || (abs(move[0]-move[3])==2 && abs(move[1]-move[4])==1))
				printf("Yes\n");
			else
				printf("No\n");
		}
	}
	return 0;
}
