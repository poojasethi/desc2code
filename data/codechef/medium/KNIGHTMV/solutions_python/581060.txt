x=raw_input()
x=int(x)
ans=[]
for i in range(0,x):
    lin=raw_input()
    if(len(lin)!=5 or lin[0]<'a' or lin[0] >'h' or lin[1]<'1' or lin[1]>'8' or lin[3]<'a' or lin[3] >'h' or lin[4]<'1' or lin[4]>'8' or lin[2]!='-' ):
        ans.append('Error')       
    else:
        su=abs(ord(lin[3])-ord(lin[0]))*abs(ord(lin[4])-ord(lin[1]))
        if(su==2):
            ans.append('Yes')
        else:
            ans.append('No')

for i  in range(0,len(ans)):
    print ans[i]
