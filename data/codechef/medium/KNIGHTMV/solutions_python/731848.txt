from re import match

def process(M):
    test1 = match(r'^([a-h])([1-8])\-([a-h])([1-8])$', M)
    if not test1: return 'Error'
    M = test1.groups()
    M = ord(M[0])-96, int(M[1]), ord(M[2])-96, int(M[3])
    D = abs(M[0]-M[2]), abs(M[1]-M[3])
    if D==(1,2) or D==(2,1): return 'Yes'
    return 'No'

def main():
    T = int(raw_input())
    for t in xrange(T):
        M = raw_input()
        print process(M)

main()
