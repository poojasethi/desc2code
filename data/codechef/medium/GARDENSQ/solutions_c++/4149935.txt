#include<iostream>
#include<stdio.h>
#include<string.h>
#include<cstdio>
#include<math.h>
#include<algorithm>
#define gc getchar_unlocked()
using namespace std;

int main()
{int t,n,m,i,j,msq,count;
char c;
scanf("%d",&t);
while(t--)
{
char gdn[150][150]={};
count=0;
scanf("%d%d",&n,&m);
cin.ignore();
if(n>=m)
    msq=m;
else
    msq=n;
for(i=0;i<n;i++)
{
    for(j=0;j<m;j++)
        scanf("%c",&gdn[i][j]);
		cin.ignore();
}

while(msq>=2)
{//cout<<msq<<"=msq\n";
int d=msq-1;
for(i=0;i<(n-d);i++)
{
    for(j=0;j<m;j++)
        {c=gdn[i][j];
        if(gdn[i][j+d]==c && gdn[i+d][j]==c && gdn[i+d][j+d]==c)
            count++;
        //cout<<i<<j<<" ";
        }
//cout<<"\n";
}
msq--;

}

printf("%d\n",count);
//cout<<"\n";




}


    return 0;
}
