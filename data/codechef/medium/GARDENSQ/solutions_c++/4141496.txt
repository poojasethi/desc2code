#include<stdio.h>
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
              char a[55][55];
              int n,m,i,j,k,count=0;
              scanf("%d%d",&n,&m);
              for(i=0;i<n;i++)
              scanf("%s",a[i]);
              for(i=0;i<n;i++)
              {
                              for(j=0;j<m;j++)
                              {
                                              k=1;
                                              while(i+k<n&&j+k<m)
                                              {
                                                                 if(a[i][j]==a[i+k][j]&&a[i][j]==a[i+k][j+k]&&a[i][j]==a[i][j+k])
                                                                 count++;
                                                                 k++;
                                              }
                              }
              }
             printf("%d\n",count ); 
    }
    return 0;
}
