int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    int T;
    int N;
    int M;
    char s[51][51];
    cin>>T;
    while(T--)
    {
        cin>>N;
        cin>>M;
        int i,j;
        for(i=0;i<N;i++)
        {
            cin>>s[i];
        }
        int cnt=0;
        for(i=0;i<N;i++)
        {
            for(j=0;j<M;j++)
            {

                for(int k=2;k<51;k++)
                {

                    if(i+k-1<N&&j+k-1<M)
                    {
                     char a=s[i][j];
                     char b=s[i+k-1][j];
                     char c=s[i][j+k-1];
                     char d=s[i+k-1][j+k-1];
                     if(a==b&&b==c&&c==d)
                        cnt++;
                    }
                    else
                        break;
                }
            }
        }
        cout<<cnt<<endl;
    }
    return 0;
}
