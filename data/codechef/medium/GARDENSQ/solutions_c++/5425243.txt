#include <iostream>
#include <cstdio>

using namespace std;

int main()
{
	char a[51][51];
	char s[51];
	int count,x,y,j,k,l,i,m,n,t;
	
	scanf("%d",&t);
	for (int d = 0; d < t; d++) {
		x = 1;
		y = 1;
		count = 0;
		scanf("%d%d",&n,&m);
		for (int f = 0; f < n ;f++) {
			scanf("%s",s);
			for (int h = 0; h < m; h++) {
				a[f][h] = s[h];
			}
		}
		while (x < n && y < m) {
			i = 0;
			l = 0;
			j = x;
			k = y;
			while (j < n) {
				while (k < m) {
					if (a[i][l] == a[j][k] && a[i][l] == a[j][l] && a[i][l] == a[i][k]) {
					//	cout << i << " " << j << endl; 
						count++;
					}
					l++;
					k++;
				}
				j++;
				i++;
				l = 0;
				k = y;
			}
			x++;
			y++;
		}
		printf("%d\n",count);
	}
}
