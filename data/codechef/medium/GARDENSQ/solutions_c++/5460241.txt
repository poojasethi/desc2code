#include <stdio.h>
int k,t,m,n,count;
char a[50][51];
int main() {
    scanf("%d",&t);
    while(t--)
     {
        scanf("%d",&n);
        scanf("%d",&m);
        for(int i=0;i<n;i++)
            scanf("%s", a[i]);
        count = 0;
        for(int i=0;i<n-1;i++) {
            for(int j=0;j<m-1;j++) {
                k = 1;
                while(i+k<n and j+k<m) {
                    if(a[i][j]==a[i][j+k] and a[i][j]==a[i+k][j] and a[i][j]==a[i+k][j+k]) count++;
                    k++;
                }
            }
        }
        printf("%d\n", count);
 
    }
    return 0;
}
