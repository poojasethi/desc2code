#include<iostream>
using namespace std;
 
int main()
{
int t;
cin>>t;
while(t--)
    {
    int n,m,count=0;
    char a[55][55];
    cin>>n>>m;
    for(int i=0;i<n;i++)
       cin>>a[i];
    for(int i=0;i<n;i++)
       for(int j=0;j<m;j++)
          for(int k=1;i+k<n && j+k<m;k++)
             if(a[i][j]==a[i+k][j] && a[i][j]==a[i][j+k] && a[i][j]==a[i+k][j+k])
                count++;
    cout<<count<<endl;
    }
return 0;
}
