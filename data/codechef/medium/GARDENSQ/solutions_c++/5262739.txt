#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
    int t,n,m,i,j,k,x,ans;
    char a[100][100];
    scanf("%d",&t);
    while(t--){
        for(i=0;i<100;i++) for(j=0;j<100;j++) a[i][j]='.';
        scanf("%d%d",&n,&m);
        for(i=0;i<n;i++) scanf("%s",a[i]);
        for(i=0,ans=0;i<n;i++)
            for(j=0;j<m;j++){
                for(k=1;i+k<n && j+k<m;k++)
					if(a[i][j]==a[i][j+k] && a[i][j]==a[i+k][j] && a[i][j]==a[i+k][j+k])
						ans++;
            }
        printf("%d\n",ans);
    }
    return 0;
}
