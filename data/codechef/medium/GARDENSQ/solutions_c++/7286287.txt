#include<iostream>
using namespace std;
int main()
{
	int i,j,k,n,m,test;
	char a[55][55];
	cin>>test;
	while(test--)
	{
		int count=0;
		cin>>n>>m;
		for(i=0;i<n;++i)
		//for(j=0;j<m;++j)
		//cin>>a[i][j];
		scanf("%s",a[i]);
		
	for(i=0;i<n;i++)
	{
	 for(j=0;j<m;j++)
	 {
	  {
		for(k=1;i+k<n && j+k<m;k++)
		{
		 if((a[i][j]==a[i][j+k]) && (a[i][j+k]==a[i+k][j+k]) && a[i+k][j+k]==a[i+k][j])
		  count++;
		}
	  }
	 }
	}
		cout<<count<<"\n";
	}
	return 0;
}
