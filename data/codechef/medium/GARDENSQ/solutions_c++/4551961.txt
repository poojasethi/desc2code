#include <cstdio>
using namespace std;

int main()
{
	int t, n, m;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d %d", &n, &m);
		char str[n][m+1];
		for(int i = 0; i < n; i++)
		{
			scanf("%s", str[i]);
		}
		int small = (m < n) ? m : n, count = 0;
		for(int k = 1; k < small; k++)
		{
			for(int i = 0; i < n - k; i++)
			{
				for(int j = 0; j < m - k; j++)
				{
					if(str[i][j] == str[i][j+k] && str[i][j] == str[i+k][j] && str[i][j] == str[i+k][j+k])
					{
						count++;
					}
				}
			}
		}
		printf("%d\n", count);
	}
    return 0;
}
