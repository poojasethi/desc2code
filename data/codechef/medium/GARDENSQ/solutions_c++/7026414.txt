#include<stdio.h>
#include<string.h>
int main()
{
char a[64][64];
int p,i,j,n,m,l,l1,l2,c,t,v,k;
scanf("%d",&t);
while(t--)
{
scanf("%d %d",&n,&m);
for(l=0;l<n;l++)
    scanf("%s",a[l]);
if(n==1 || m==1)        
{
    printf("0\n");
    continue;
}
if(n>m)                 
    v=m;
else
    v=n;
c=0;
while(v>1)              
{
v--;
for(i=0;i<n-v;i++)          
{
    for(j=0;j<m-v;j++)      
        {
            
            if((a[i][j]==a[i][j+v])&&(a[i][j]==a[i+v][j])&&(a[i][j]==a[i+v][j+v]))  //checking equality
                c++;
        }
}
}
printf("%d\n",c);
}
return 0;
}
