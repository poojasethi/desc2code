    #include<stdio.h>
    #include<math.h>
    int main()
    {
     int t,n,m,i,j,k;
     char a[60][60];
     scanf("%d",&t);
     while(t--)
     {
      scanf("%d %d",&n,&m);
      int c=0;
      for(i=0;i<n;i++)
    	 scanf("%s",a[i]);
      for(i=0;i<n;i++)
    	{
    	 for(j=0;j<m;j++)
    	 {
    	  {
    		for(k=1;i+k<n && j+k<m;k++)
    		{
    		 if((a[i][j]==a[i][j+k]) && (a[i][j+k]==a[i+k][j+k]) && a[i+k][j+k]==a[i+k][j])
    		  c++;
    		}
    	  }
    	 }
    	}
    	printf("%d\n",c);
     }
     return 0;
    }
