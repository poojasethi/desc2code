#include <iostream>
using namespace std;
int main()
{
    int t;
    cin >> t;
 
    while(t>0)
    {
        int Y, X;
        cin >> Y >> X;
        char grid[50][50];
        for(int y = 0; y < Y; y++)
        {
            for(int x = 0; x < X; x++)
            {
                cin >> grid[x][y];
            }
        }
        int max_size = X < Y ? X : Y;
        int count = 0;
        for( int s = 1; s < max_size; s++ )
        {
            for( int x = 0; x < X - s; x++)
            {
                for( int y = 0; y < Y - s; y++)
                {
                    if(grid[x][y]==grid[x+s][y]&&grid[x][y]==grid[x][y+s] &&grid[x][y]==grid[x+s][y+s])
                        count ++;
                }
            }
        }
        cout<<count<<endl;
        t--;
    }
}
