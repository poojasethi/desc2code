#include<iostream>
#include<iomanip>
#include<cstdio>
using namespace std;

double dp[100100][50]={0},total[100100][50]={0};

int main()
{
    int i,j,k,l,m,n,t;
    for(i=1;i<=100000;i++)
    {
        dp[i][0]=i;
        total[i][0]=total[i-1][0]+dp[i][0];
    }
    for(i=1;i<=100000;i++)
    {
        for(j=1;j<=40;j++)
        {
            dp[i][j]=(total[i-1][j-1])/i;
            total[i][j]=total[i-1][j]+dp[i][j];
        }
    }
    cin>>t;
    while(t--)
    {
        //cin>>n>>k;
        scanf("%d %d",&n,&k);
        //cout<<fixed;
        if(k>40)
            printf("0.000000\n");//cout<<"0.000000\n";
        else
            printf("%.9lf\n",dp[n][k]);//cout<<setprecision(7)<<dp[n][k]<<"\n";
    }
}
