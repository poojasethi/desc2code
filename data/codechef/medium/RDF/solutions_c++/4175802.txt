#include <cstdio>
#define MAX 100000
#define prec 36
#define FOR(i,m,n) for(i=m;i<=n;i++)

using namespace std;

double f[MAX+1][prec+1];

int main()
{
    long int i,j,t,n,k;

    double sum=0;

    FOR(j,0,MAX)
        f[j][0]=j;

    FOR(i,1,prec)
    {
        sum = 0;
        f[0][i]=0;
        FOR(j,1,MAX)
        {
            sum += f[j-1][i-1];
            f[j][i] = sum / j;
        }
    }

    scanf("%ld",&t);

    FOR(i,1,t)
    {
        scanf("%ld%ld",&n,&k);
        printf("%0.7lf\n",(k<=prec)? f[n][k]:0);
    }

    return 0;
}
