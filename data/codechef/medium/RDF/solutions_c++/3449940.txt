#include<stdio.h>
float arr[100000][36];
int main()
{
    int i,j,t,n,k;
    float sum;
    for(i=1;i<=99999;i++)
    arr[i][0]=i;
    for(j=1;j<36;j++)
    {
        sum=0;
        for(i=1;i<100000;i++)
        {
            sum+=arr[i-1][j-1];
            arr[i][j]=sum/i;
        }
    
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        if(k>35)
        printf("0\n");
        else printf("%f\n",arr[n][k]);
    }
}
