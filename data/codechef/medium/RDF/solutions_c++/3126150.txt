#include<iostream>
#include<vector>
#define MAX 100000
using namespace std;
double record[38][MAX];
void preprocess()
{
	for(int i=1;i<MAX;i++)
	{
		if(i==1)
			record[0][i]=i;
		else
			record[0][i]=record[0][i-1]+i;
	}
	for(int i=1;i<38;i++)
	{
		if(i==1)
			record[i][i]=(i-1)/2.0;
		else
			record[i][i]=record[i-1][i-1]/i;
		for(int j=i+1;j<MAX;j++)
		{
			if(i==1)
				record[i][j]=record[i][j-1]+((j-1)/2.0);
			else
				record[i][j]=record[i][j-1]+(record[i-1][j-1]/j);
		}
	}
}

int main()
{
	preprocess();
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		if(n<=k||k>36)
			cout<<"0\n";
		else	
			printf("%.6lf\n", record[k][n]-record[k][n-1]);
	}
	return 0;
}
