double memo[100010][40];

void init() {

    int i, j, k, n;
    double sum = 0;
    for ( i = 0; i <= 100000; i++ ) {
        memo[i][0] = i;
    }
    for ( i = 0; i < 40; i++ ) {
        memo[0][i] = 0;
    }

    for ( k = 1; k < 40; k++ ) {
        sum = 0;
        for ( n = 1; n <= 100000; n++ ) {
            sum += memo[n-1][k-1];
            memo[n][k] = sum / n;
        }
    }
}

int main () {
    init();
    //freopen ( "input.txt", "r", stdin );
    //freopen ( "output.txt", "w", stdout );

    int kase;
    scanf ( "%d", &kase );

    while ( kase-- ) {
        int n, k;
        scanf ( "%d %d", &n, &k );

        if ( k > 35 ) k = 35;

        double res = memo[n][k];
        printf ( "%.10lf\n", res );
    }

    return 0;
}
