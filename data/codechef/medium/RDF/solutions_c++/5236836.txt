#include<stdio.h>
double a[100001][31];
int getn(){
	int n=0,c=getchar();
	while(!(c>='0' && c<='9'))
		c=getchar();
	while(c>='0' && c<='9')
		n=(n<<3)+(n<<1)+c-48,c=getchar();
	return n;
}
void doit(){
	int i,j;
	double sum;
	for(i=0;i<100001;i++){
		a[i][0]=i;
	}
	for(j=1;j<31;j++){
		sum=0;
		for(i=1;i<100001;i++){
			sum+=a[i-1][j-1];
			a[i][j]=sum/i;
		}
	}
}
int main(){
	int t;
	doit();
	t=getn();
	while(t--){
		int n,k;
		n=getn();
		k=getn();
		if(k>30)
			printf("0\n");
		else
			printf("%lf\n",a[n][k]);
	}
	return 0;
}
