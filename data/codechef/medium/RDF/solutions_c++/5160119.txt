#include<cstdio>
using namespace std;
double a[100001][34]={0};
long long int i,j;
double sum;
int main()
{
	for(i=0;i<100001;i++)
	{
		a[i][0]=i;
	}
	for(j=1;j<34;j++)
	{
		sum=0;
		for(i=j+1;i<100001;i++)
		{
			sum=sum+a[i-1][j-1];
			a[i][j]=sum/i;
		}
	}
	long long int t,k,n;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld%lld",&n,&k);
		if(k>=34)
		{
			printf("0.000000\n");
		}
		else
		{
			printf("%.6lf\n",a[n][k]);
		}
	}
	return 0;
}
