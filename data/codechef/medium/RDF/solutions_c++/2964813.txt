#include<iostream>
#include<cstdio>
using namespace std;
 double rdf[40][100005],sum;
int main()
{
    int t,n,k,i,j;
    for(i=0;i<=100000;i++)
            rdf[0][i]=i;
          for(i=1;i<=39;i++)
           {
               sum=0.000000;
               for(j=1;j<=100000;j++)
               {
                   sum+=rdf[i-1][j-1];
                   rdf[i][j]=sum/j;
               }
           }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        
        if(k>36)
            printf("0.000000\n");
        else if(k>n)
               printf("0.000000\n");
        else
        {
           printf("%.6f\n",rdf[k][n]);
        }
    }
}
