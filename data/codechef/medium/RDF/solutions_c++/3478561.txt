#include      <iostream>
#include      <iomanip>
#define endl '\n'
using namespace std;

const int MAXN = 100001;
const int MAXK = 32;
double DP[MAXN][MAXK];

int main(int argc, char **argv) {
	ios_base::sync_with_stdio(false), cin.tie(0);

	for (int i = 0; i < MAXN; i += 1) DP[i][0] = i;
	for (int k = 1; k < MAXK; k += 1) {
		double s = 0;
		for (int n = 1; n < MAXN; n += 1) {
			DP[n][k] = s / n;
			s += DP[n][k - 1];
		}
	}

	int T, N, K;
	cin >> T;
	while (T--) {
		cin >> N >> K;
		cout << fixed << setprecision(20) << DP[N][min(K, MAXK - 1)] << endl;
	}

	return 0;
}
