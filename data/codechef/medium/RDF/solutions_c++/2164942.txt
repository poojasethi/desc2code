typedef long long int LL;

double pre[100010][37];
double RDF(int i, int j)
{
        if(i<j)
		return 0.0;
        if(j>35)
		return 0.0;
	if(pre[i][j]>-1)
		return pre[i][j];
        
		
	if(j==0)
	{	
	        return i*1.0;
	}
	    
	    {
	   pre[i][j]=(double)(RDF(i-1,j)*(i-1)+ RDF(i-1,j-1))/i;
        return pre[i][j];
	    }
}
int main()
{
        int i,j,t,T;
	double ans;
	rep(i,100010)
		rep(j,37)
                pre[i][j]=-1;		
       si(t);
       rep(T,t)
       {
         si(i); si(j);
         ans= RDF(i,j);
	 printf("%.7lf\n",ans);
       }
	
	return 0;
}
