double memory[100000][36];

int main(){
    int testcases;
    scanf("%d",&testcases);
    FOR(i,1,100000){
        memory[i][0] = (double)i;
        memory[i][1] = (i-1)/2.0;
    }
    FOR(j,2,36){
        double sum = 0.0;
        FOR(i,1,100000){
            sum += memory[i-1][j-1];
            memory[i][j] = sum/(double)i;
        }
    }
    while(testcases--){
        int N,K;
        scanf("%d%d",&N, &K);
        if (K >= 36){
            printf("0.000000\n");
       }
        else{
            printf("%f\n",memory[N][K]);
        }
    }
    
    return 0;
}
