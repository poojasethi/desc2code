#include<iostream>
#include<cstdio>
using namespace std;
double dp[100000][37];
int main()
{
	
	double sum;
	for(int i=0;i<100000;i++)
	dp[i][0]=(double)i;
	for(int j=1;j<37;j++)
			{
				sum=0;
				for(int i=1;i<100000;i++)
				{
				sum=sum+dp[i-1][j-1];
				dp[i][j]=(double)sum/i;}
			}
	int t,n,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		if(k>=37)
		printf("0\n");
		else
		{
			printf("%.10lf\n",dp[n][k]);
		}
	}
	return 0;
}
