    #include <cstdio>
    #include <cstring>
    #define MAX 100000
    #define prec 36
    #define FOR(i,m,n) for(i=m;i<=n;i++)
     
    using namespace std;
     
    double f[prec+1][MAX+1];
     
    void precompute()
    {
    int i;long int j;
    long double sum=0;
    memset(f,0,sizeof(f));
     
    FOR(j,0,MAX)
    f[0][j]=j;
     
    FOR(i,1,prec)
    {
    sum = 0;
    FOR(j,i,MAX-1)
    {
    sum = sum + f[i-1][j];
    f[i][j+1] = sum / (long double)(j+1);
    }
    }
    }
     
    int main()
    {
    long int i,t,n,k;
     
    precompute();
     
    scanf("%ld",&t);
     
    FOR(i,1,t)
    {
    scanf("%ld%ld",&n,&k);
    if(k>prec) printf("0\n");
    else
    printf("%0.7lf\n",f[k][n]);
    }
     
    return 0;
    }
