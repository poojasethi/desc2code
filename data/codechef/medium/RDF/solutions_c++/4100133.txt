#include<bits/stdc++.h>
using namespace std;
double arr[100002][38];
int main()
{
	int t;
	scanf("%d",&t);
	for(int i=0;i<100001;i++)
	 arr[i][0]=i;
	for(int i=1;i<=36;i++)
	{
		double sum =0;
		for(int j=1;j<=100000;j++)
		{
			sum += (arr[j-1][i-1]);
			arr[j][i] = (double)(sum/(double)j);
		}
	}
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		if(k<=36)
		printf("%.6f\n",arr[n][k]);
		else
		printf("0\n");
	}
}
