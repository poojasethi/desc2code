#include<iostream>
using namespace std;
typedef long double LD;
long double ans;
LD dp[100000][35];
void precompute()
{
	int i,j;
	dp[1][0]=1;
	for(i=1;i<100000;i++)
        dp[i][0]=(LD)i;
	for(j=1;j<=34;j++)
    {
        for(i=2;i<100000;i++)
        {
            dp[i][j]=(dp[i-1][j]*(LD)(i-1)+dp[i-1][j-1])/(LD)i;
        }
    }
}

int main()
{
	int t,N,K;
	cin>>t;
	precompute();
	while(t--)
    {
		scanf("%d %d",&N,&K);
		if(K>34)
		    printf("0\n");
		else
		    printf("%0.7Lf\n",dp[N][K]);
	}
return 0;
}
