#include<iostream>

double rdf[100001][37]={0};

inline void input( int *a)
{ register char c=0;
   while(c<33) c=getchar();
    *a=0;
     while(c>33)
     { *a = *a*10 + c - '0';
         c=getchar();
      }
}

void func()
{
int i;
double sum;
for(i=0;i<=100000;i++)
 rdf[i][0]=i;

int j;
for(j=1;j<=36;j++)
{ sum=0;
   for(i=1;i<=100000;i++)
   {
      sum+=rdf[i-1][j-1];
      rdf[i][j]=sum/i;
   }
}
}

int main()
{
int test;
input(&test);
func();
