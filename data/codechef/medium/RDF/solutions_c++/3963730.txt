#include<stdio.h>
#include<string.h>

double A[100001][32];

void scanint(int &x) 
{
	register int c = getchar_unlocked();
	x = 0;
	for (; (c < 48 || c > 57); c = getchar_unlocked())
		;
	for (; c > 47 && c < 58; c = getchar_unlocked())
	{
		x = (x << 1) + (x << 3) + c - 48;
	}
}

void Finit()
{
	
	double sum;
	int i, j;
	
	for(i=0; i<=100000; i++)
	A[i][0]=i;
	
	for(j=1; j<=30; j++)
	{
		sum=0;
		for(i=1; i<=100000; i++)
		{
			sum+=A[i-1][j-1];
			A[i][j]=sum/i;
		}
	}
}

int main()
{
	int T, N, K;
	Finit();
	
	scanint(T);
	
	while(T--)
	{
		scanint(N);
		scanint(K);
		if(K>30)
		printf("0\n");
		else
		printf("%lf\n",A[N][K]);
	}
	
	return 0;
}
