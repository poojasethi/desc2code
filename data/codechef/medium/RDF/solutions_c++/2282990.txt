#include<iostream>
#include<cstdio>
using namespace std;
double arr[100001][37];
int t, n, k, i, j;
int main(){
	for(j=1 ;j<37; ++j)
		arr[1][j] = 0;
	for(i=1; i<=100000; ++i){
		arr[i][0] = i;
		arr[i][1] = (i-1)/2.0;
	}
	for(i=1; i<=100000; ++i){
		for(j=2; j<37; ++j)
			arr[i][j] = ((i-1)*arr[i-1][j]+arr[i-1][j-1])/i;
	}
	scanf("%d", &t);
	while(t-- > 0){
		scanf("%d %d", &n, &k);
		if(k > 36)
			printf("0.0\n");
		else
			printf("%f\n", arr[n][k]);
	}
	return 0;
}
