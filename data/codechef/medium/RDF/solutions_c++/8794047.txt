#include <iostream>
#include <cstdio>
#include <algorithm>
#include <math.h>
using namespace std;
#define M 1000001
double f[M][37];
void fun(){
    double sum=0.0;
for(int i=0;i<=M;i++){
    f[i][0]=(double)i;
}
for(int j=0;j<37;j++){
    f[0][j]=0;
}
for(int j=1;j<=36;j++)
{
    sum=0.0;
    for(int i=1;i<99999+1;i++){
        sum+=f[i-1][j-1];
        f[i][j]=(double)sum/i;
    }
}
}
int main()
{
    fun();
   int t;
   scanf("%d",&t);
   while(t--) {
    int n,k;
    scanf("%d%d",&n,&k);
    if(k>36)
        printf("0.000000\n");
    else
        printf("%8f\n",f[n][k]);


   }
    return 0;
}
