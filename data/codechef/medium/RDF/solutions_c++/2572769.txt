#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

double dp[37][100000];

int main() {
  double sum;
  for(int n=0;n<100000;++n) dp[0][n] = n;
  for(int k=1;k<37;++k) {
    sum = 0.0;
    for(int n=1;n<100000;++n) {
      dp[k][n] = sum/n;
      sum += dp[k-1][n];
    }
  }
  
  int t, n, k;
  scanf("%d",&t);
  while(t--) {
    scanf("%d%d",&n,&k);
    if(k<37) printf("%lf\n",dp[k][n]);
    else printf("0.0\n");
  }
  return 0;
}
