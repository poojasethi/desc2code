#include <iostream>
#include <cstdio>
using namespace std;

typedef long long ll;
double dp[39][100002];

int main()
{
    //cin.sync_with_stdio(false);
    //cout.sync_with_stdio(false);
    for(int i=0;i<100001;i++){
        dp[0][i] = i;
    }
    double sum=0.0;
    for(int j=1;j<37;j++){
        sum = 0.0;
        for(int i=1;i<100001;i++){
            sum += dp[j-1][i-1];
            dp[j][i] = sum/i;
        }
    }
    //memset(dp,-1.0,sizeof(dp));
    int t;
    scanf("%d",&t);
    while(t--){

        int n,k;
        scanf("%d%d",&n,&k);
        if(k>=37 || k >= n){
            cout<<"0\n";
            continue;
        }
        //int tm = solve(n,k,1);
        printf("%lf\n",dp[k][n]);
        //cout<<dp[n][k]<<"\n";
    }
    return 0;
}
