double F[99999][37];
int main()
{
    int T,N,K;
    double sum;
    cin>>T;
    for(int i=0;i<100000;i++)
        F[i][0]=i;
    for (int j=1;j<=36;j++)
    {
        sum = 0;
        for(int i=1;i<=99999;i++)
        {
            sum += F[i-1][j-1];
            F[i][j] = sum / i;

        }
    }
    while(T--)
    {

        scanf("%d%d",&N,&K);
        if(K>=37)
            printf("0.00000\n");
        else
        {
            printf("%8f\n",F[N][K]);
        }


    }
}
