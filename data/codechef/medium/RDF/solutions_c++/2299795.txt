#include<stdio.h>

double rd[100003][36];

void rdf()
{
	int i,j;
	double sum;
	
	for(j=1;j<=35;j++)
	{
		sum=0;
		for(int i=1;i<=100000;i++)
		{
			sum=sum+rd[i-1][j-1];
			rd[i][j]=(sum/(double)i);
		}
	}
}

int main()
{
	 int t,n,k,i,j;
    scanf("%d",&t);
 
    for(k=0; k<=99999; k++)
        rd[k][0]=(double)k;
    rdf();
    while(t--)
    {
        scanf("%d",&n);
        scanf("%d",&k);
        if(k<=35)
        {
            if(k==0)
                printf("%f\n",(double)n);
            else
                printf("%f\n",rd[n][k]);
        }
 
        else
            printf("0\n");
    }
 
	return 0;
}
