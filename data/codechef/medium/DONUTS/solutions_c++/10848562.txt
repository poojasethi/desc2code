#include<bits/stdc++.h>

using namespace std;

#define MAXM 20000+5

int t;
int N, M;

int a[MAXM];

int main() {
	ios_base::sync_with_stdio(false); 
	cin>>t;
	while (t--) {
	    cin>>N>>M;
		for (int i = 0; i < M; i++) {
			int b;
			cin>>b;
			a[i]=b;
		}
		sort(a, a+M);
		int ni = 0, sum = 0;
		for (int i = 0; i < M; i++) {
			sum += (a[i] + 1);
			if (sum >= M) {
			   ni = i;
			   break;
            }
        }
	    cout<<M-ni-1<<endl;
	}
	return 0;	
}
