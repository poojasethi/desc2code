#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,n,m,i,a[100001],c=0;
    scanf("%d",&t);
    while(t--)
    {
      scanf("%d%d",&n,&m);
      for(i=0;i<m;i++)
            scanf("%d",&a[i]);
      sort(a,a+m);
      if(m==1)
        printf("0\n");
      else
       if(m==2)
        printf("1\n");
       else
       {
      m=m-1;c=0;i=0;
      while(i<m)
      {
          if(a[i]!=0)
          {a[i]=a[i]-1;
           c++;
           //cout<<c;
           m=m-1;
          }
          else
            i++;
      }
      printf("%d\n",c);
    }
    }
    return 0;
}
