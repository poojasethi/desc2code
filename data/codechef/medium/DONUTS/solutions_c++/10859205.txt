#include<cstdio>
#include<algorithm>
#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int t,i;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,m,start,end,step,f=0;
		long long int a[20001];
		scanf("%lld%lld",&n,&m);
		for(i=0;i<m;i++)
		{
			scanf("%lld",&a[i]);
			if(a[i]==1)
				f++;
		}
		if(f==m)
			printf("%lld\n",(m/2));
		else
		{
			sort(a,a+m);
			start=0;end=m-1;step=0;
			while(1)
			{
				if(end==0)
					break;
				a[start]--;
				a[end-1]+=a[end]+1;
				step++;
				end--;
				if(a[end]==n)
					break;
				if(a[start]==0)
					start++;
			}
			printf("%lld\n",step);
		}
	}
	return 0;
}
