#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int main()
{
long int t,m,n;
cin>>t;
while(t--)
{
cin>>n>>m;
long int a[m];
for(long int i=0;i<m;i++)
scanf("%ld", &a[i]);
sort(a,a+m);
long int c=0, k=0;
while(m>1)
{
	if(a[k]>0)
    {a[k]--;
    	c++;
        m--;
    }
    else{ m--;
    k++;}
}
cout<<c<<"\n";
}
return 0;

}
