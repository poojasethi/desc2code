#include<bits/stdc++.h>
#include<vector>
#include<algorithm>
#define ull unsigned long long 
#define pb push_back
#define sd(n) scanf("%lld",&n)
using namespace std;
ull q[100000];
int main()
{       ull n,m,sum,ck,shi,t;
    //freopen("input.txt","r",stdin);
    sd(t);

    while(t--)
    {sum=0; 
        sd(n);
        sd(m);
        for(ull o=1;o<=m;o++)
        {
            sd(q[o]);
        }
        sort(q+1,q+m+1);
        ck=1;
        while(true)
        {
            sum+=q[ck-1];
            if(sum==m-ck)
            {shi=sum;
            break;}
            if(sum>m-ck)
            {shi=m-ck+1;
            break;
            }
            ck++;
            
             
        }
        printf("%lld\n",shi);
        
        
    }
}
