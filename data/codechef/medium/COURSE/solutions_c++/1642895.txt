#define N 555
 
int x[N],y[N];
double t[N];
int mark[N];
 
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("in.txt","r",stdin);
	#endif
	int T;
	for(scanf("%d",&T);T--;)
	{
		int n,r,R;
		scanf("%d%d%d",&r,&R,&n);
		int i,v;
		for(i=1;i<=n;i++)
			scanf("%d%d",&x[i],&y[i]);
		t[0]=0;
		for(i=1;i<=n+1;i++) t[i]=1e9;
		fill(mark,0);
		for(;;)
		{
			int u=-1;
			for(v=0;v<=n+1;v++)
				if(!mark[v] && (u<0 || t[u]>t[v])) u=v;
			if(u==n+1) break;
			mark[u]=1;
			double T=t[u],Ti;
			if(u==0)
			{
				for(i=1;i<=n;i++)
					if(!mark[i])
					{
						Ti=max(T,R-hypot(1.*x[i],1.*y[i]));
						MIN(t[i],Ti);
					}
			}
			else
			{
				for(i=1;i<=n;i++)
					if(!mark[i])
					{
						Ti=max(T,hypot(1.*x[i]-x[u],1.*y[i]-y[u]));
						MIN(t[i],Ti);
					}
				Ti=max(T,hypot(1.*x[u],1.*y[u])-r);
				MIN(t[n+1],Ti);
			}
		}
		double ans=min(t[n+1],R-r);
		printf("%.3lf\n",ans);
	}
	return 0;
}
