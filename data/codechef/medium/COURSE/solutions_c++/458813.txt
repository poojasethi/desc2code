#include<cstdio>
#include<cmath>
#define MAX(X,Y) ((X)>(Y)?(X):(Y))
#define MIN(X,Y) ((X)<(Y)?(X):(Y))
#define M 600	
using namespace std;


inline double SQR(double x){double result=x*x; return(result);}
inline double SQRT(double x){double result=pow(x,1/2.00); return result;}
int main(){
double x[M],y[M],d[M]; int id,turn,i,n,r,R,OK[M];
scanf("%d",&turn);
while(turn--){
scanf("%d%d",&r,&R); scanf("%d",&n);
for(i=0;i<n;i++) scanf("%lf%lf",&x[i],&y[i]);
for(i=0;i<n;i++){ d[i]=SQRT( SQR(x[i])+SQR(y[i]) )-r; OK[i]=0;}
OK[n]=0;d[n]=R-r;
while(1){id=-1;
for(i=0;i<=n;i++) if(!OK[i] && (id==-1 || d[id]>d[i])) id = i;
if(id==n) break;
OK[id]=1;
for(i=0;i<n;i++)
if(!OK[i]) d[i]=MIN(d[i],MAX(d[id],SQRT( SQR(x[i]-x[id])+SQR(y[i]-y[id]) )));
d[n] = MIN(d[n],MAX(d[id],R-SQRT( SQR(x[id])+SQR(y[id]) )));
}
printf("%.3lf\n",d[n]);
}
return 0;
}
