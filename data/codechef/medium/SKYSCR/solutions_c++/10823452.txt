#include<iostream>
#include<cmath>
using namespace std;
typedef long long ll;
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    ll m,n;
    while(t--)
    {
        cin>>n>>m;
        if(n>m)
            cout<<n-m<<endl;
        else cout<<m-n<<endl;
    }
    return 0;
}
