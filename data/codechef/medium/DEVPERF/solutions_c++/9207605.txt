#include <cstdio>
#include <climits>
#include <algorithm>
#define MAX_N_VALUE 10000



int main() {
    int t;
    scanf("%d", &t);
    while (t--) {
        int n, m;
        bool flag = false;
        int min_row, min_col, max_row, max_col;
        min_row = min_col = INT_MAX;
        max_row = max_col = -1;
        scanf("%d %d", &n, &m);
        for (int i = 0; i < n; i++) {
            char str[MAX_N_VALUE + 5];
            scanf("%s", str);
            for (int j = 0; str[j]!='\0'; j++) {
                if (str[j] == '*') {
                    flag = true;
                    if (i < min_row)
                        min_row = i;
                    if (i > max_row)
                        max_row = i;
                    if (j < min_col)
                        min_col = j;
                    if (j > max_col)
                        max_col = j;
                }
            }
        }
        // We see max of time needed to reach any of the
        int time = 0;
        if (flag) {
            int start_row = (max_row + min_row) / 2;
            int start_col = (max_col + min_col) / 2;

            time = std::max(start_row - min_row, max_row - start_row);
            time = std::max(time, std::max(start_col - min_col, max_col - start_col));
            time = time + 1;
        }
        printf("%d\n", time);

    }
}
