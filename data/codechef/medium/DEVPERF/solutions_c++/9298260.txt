#include <iostream>
#include <cstdio>
#include <algorithm>
#include <math.h>

using namespace std;

int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n, m, count = 0, ans = 1, minr, maxr = 0, minc, maxc = 0, diffc, diffr;
        scanf("%d %d", &n, &m);
        char arr[n + 1][m + 1];
        minr = n;
        minc = m;
        for(int i = 0; i < n; i++)
        {
            scanf("%s", &arr[i]);
            for(int j = 0; j < m; j++)
            {
//                scanf("%c", &arr[i][j]);
                if(arr[i][j] == '*')
                {
                    count++;
                    if(i < minr)
                    {
                        minr = i;
                    }
                    if(i > maxr)
                    {
                        maxr = i;
                    }

                    if(j < minc)
                    {
                        minc = j;
                    }
                    if(j > maxc)
                    {
                        maxc = j;
                    }
                }
            }
        }
        if(count == 0 || count == 1)
        {
            printf("%d\n", count);
            continue;
        }
        diffc = maxc - minc;
        diffr = maxr - minr;

        diffc = ceil(diffc/2.0);
        diffr = ceil(diffr/2.0);
//        cout<<diffr<<"\n"<<diffc;
        if(diffr > diffc)
        {
            ans += diffr;
        }
        else
        {
            ans += diffc;
        }
        printf("%d\n", ans);
    }
    return 0;
}
