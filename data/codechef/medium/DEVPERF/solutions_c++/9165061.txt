#include<bits/stdc++.h>
using namespace std;
# define bang(i,a,b) for(i=a;i<b;i++) 
int main()
{
	int cases;
	scanf(" %d",&cases);
	while(cases--)
	{
		int i,j,n,m,right=0,left=0,top=0,bottom=0,flag=0,count = 0;
		scanf(" %d %d",&n,&m);
		char s[n][m];
		int row[n],column[m];
		bang(i,0,n)
		row[i] = 0;
		bang(i,0,m)
		{
			column[i] = 0;
		}
        //for(i=0;i<n;i++)
        bang(i,0,n)
        {
        	//for(j=0;j<n;j++)
        	bang(j,0,m)
        	{
        		scanf(" %c",&s[i][j]);
        		if(s[i][j] == '*')
        		{
        			flag=1;
        			row[i] = row[i] + 1;
        			column[j] += 1;
        		}
        	}
        }
        bang(i,0,n)
        {
        	if(row[i]!=0)
        	{	left = i+1;
               //cout << left;
        		break;
        	}
        }
        for(j=n-1;j>=0;j--)
        {
        	if(row[j]!=0)
        	{	right = j+1;
        	//cout << right;
        	break;}
        }
        bang(i,0,m)
        {
        	if(column[i]!=0)
        	{
        		top = i+1;
        	//	cout << top;
        		break;
        	}
        }
        for(j=m-1;j>=0;j--)
        {
        	if(column[j]!=0)
        	{
        		bottom = j+1;
        	//	cout << bottom;
        		break;
        	}
        }
        if(flag == 0)
        {
        printf("0\n");
 //           continue;
        }
        else
        {
         	count += 1;
        
        int max = right-left>=(bottom-top)?right-left:bottom-top;
      //  printf(" %d\n",max);
        //if(max == 1)
        {
        //	count = count+1;
        }
        //else
        {  if(max%2 == 0)
        	{
        		max=max/2;
        		//printf("yeah\n");//do nothing
        	}
        	else
        	{
        		max = max/2 + 1;
        //		printf("odd\n");
        	}
        	bang(i,0,max)
        	{
        		count++;
        	}
        }
        printf("%d\n",count);
    }

	}
return 0;
}
