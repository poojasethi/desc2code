#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#define M 1005
using namespace std;
char str[M][M];
int n,m;
int solve(int x,int y){
	int i,j,ans=0;
	for(i=0;i<n;i++)
		for(j=0;j<m;j++){
			if(str[i][j]!='*')continue; 
			ans=max(ans,max(abs(x-i),abs(y-j))); 
		}
	return ans;
}
int main(){
	int cas,i,j,k;
	scanf("%d",&cas);
	while(cas--){
		scanf("%d %d",&n,&m);
		int x1=n+1,x2=0,y1=m+1,y2=0;
		for(i=0;i<n;i++){
			scanf("%s",str[i]);
			for(j=0;j<m;j++)
				if(str[i][j]=='*'){
					x1=min(x1,i);
					x2=max(x2,i);
					y1=min(y1,j);
					y2=max(y2,j);
				}
		}
		if(x1==n+1){
			puts("0");continue;
		}
		int x=(x1+x2)/2;
		int y=(y1+y2)/2; 
		int ans=max(n,m);
		ans=min(ans,solve(x,y));
		ans=min(ans,solve(x+1,y));
		ans=min(ans,solve(x,y+1));
		ans=min(ans,solve(x+1,y+1));
		printf("%d\n",ans+1);
	}
	return 0;
}
