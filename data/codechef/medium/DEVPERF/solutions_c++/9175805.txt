#include<bits/stdc++.h>
using namespace std;
int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
      int n,m,x1=1005,x2=0,y1=1005,y2=0,f=0;
      scanf("%d %d",&n,&m);
      string s[n];
      for(int i=0;i<n;i++)
        cin>>s[i];
      for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                if(s[i][j]=='*')
                {
                    f=1;
                    x2=max(x2,j);
                    x1=min(x1,j);
                    y2=max(y2,i);
                    y1=min(y1,i);
                }

            }
        }
     if(f)
     {
        int m1=ceil(1.0*(x2-x1)/2);
        int m2=ceil(1.0*(y2-y1)/2);
        int ans=1+max(m1,m2);
        //ans=max(1,ans);
        printf("%d\n",ans);
     }
     else
        printf("0\n");

   }
   return 0;
}
