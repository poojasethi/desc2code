#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>
#include<string>
#include<cstdio>
#include<queue>
#include<iostream>
#include<utility>
#include<vector>
#include<functional>

#define MAX 1009

using namespace std;

int main(){
int tst,n,m,i,j;
scanf("%d",&tst);
char str[MAX][MAX];

while(tst--){
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
        scanf("%s",&str[i]);

    int top=99999,bottom=-99999,left=99999,right=-99999;
    //initial boundaries

    for(i=0;i<n;i++){
            for(j=0;j<m;j++){
                if(str[i][j]=='*')
                {   top=min(top,i);
                    bottom=max(bottom,i);
                    left=min(left,j);
                    right=max(right,j);
                }
            }
    }

    int ans=0;
    int vert=bottom-top;int hori=right-left;
    int mx=max(hori,vert);
    if(top==99999)
    ans=0;
    else
    ans=1+(mx+1)/2;

    printf("%d\n",ans);
}
return 0;
}
