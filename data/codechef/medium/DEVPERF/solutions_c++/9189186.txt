#include<stdio.h>
using namespace std;
int l,u,d,r,dif;
char a[1024];
int main()
{
    int t,i,j,n,m,fl;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&n,&m);
        u=1024;d=-1;l=1024;r=-1;fl=0;
        for(i=0;i<n;i++)
        {
            scanf("%s",a);
            for(j=0;j<m;j++)
            {
                if(a[j]=='*')
                {
                    if(i<u)
                        u=i;
                    if(i>d)
                        d=i;
                    if(j<l)
                        l=j;
                    if(j>r)
                        r=j;
                    fl=1;
                }
            }
        }
        if(d-u>r-l)
        {
            dif=d-u;
        }
        else
        {
            dif=r-l;
        }
        if(dif%2==0)
        {
            dif/=2;
            dif++;
        }
        else
        {
            dif/=2;
            dif+=2;
        }
        if(fl==0)
            printf("0\n");
            else
        printf("%d\n",dif);
    }
    return 0;
}
