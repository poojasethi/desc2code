#include<bits/stdc++.h>
using namespace std;

char grid[1005][1005];

int main()
{
   int t;
   scanf("%d",&t);
   while(t--)
   {
       int n,m;
       scanf("%d %d",&n,&m);
       for(int i=0;i<n;i++)
       {
          scanf("%s",&grid[i]);
       }
       int maxc=0,maxr=0,minc=m,minr=n;

       for(int i=0;i<n;i++)
       {
           for(int j=0;j<m;j++)
           {
               if(grid[i][j]=='*')
               {
                   maxc=max(maxc,j);
                   minc=min(minc,j);
                   maxr=max(maxr,i);
                   minr=min(minr,i);
               }
           }
       }
       int px=(maxr+minr)/2;
       int py=(maxc+minc)/2;
       int mint=-1;
       for(int i=0;i<n;i++)
       {
           if(grid[i][minc]=='*')
           {
              mint=max(mint,max(abs(px-i),abs(py-minc)));
           }

           if(grid[i][maxc]=='*')
           {
              mint=max(mint,max(abs(px-i),abs(py-maxc)));
           }
       }

       for(int j=0;j<m;j++)
       {
           if(grid[minr][j]=='*')
           {
              mint=max(mint,max(abs(px-minr),abs(py-j)));
           }

           if(grid[maxr][j]=='*')
           {
              mint=max(mint,max(abs(px-maxr),abs(py-j)));
           }
       }

       if(minr==n)printf("0\n");
       else
       printf("%d\n",1+mint);
   }
   return 0;
}
