#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int main()
{
    long t,k,i;
    char w[100000];
    long long alphaCount[26];
    long long sortCount[26];
    long long maxCount[26];
    long cnt=0;
    long long changes=0;
    for(i=0;i<26;i++)
        alphaCount[i] = 0;
    scanf("%d",&t);
    while(t-->0)
    {
        scanf("%s %d",w,&k);
        for(i=0;i<26;i++){
            alphaCount[i] = 0;
            sortCount[i] = 0;
        }
        for(i=0;i<strlen(w);i++)
        {
            //printf("character %c : \tindex : %d\n",w[i],w[i]-97);
            alphaCount[w[i]-97]++;
        }
        cnt=0;changes=0;
        long long minimum=strlen(w);
        int temp=0;

        for(i=0;i<26;i++)
        {
            if(alphaCount[i]>0)
            {
                sortCount[cnt] = alphaCount[i];
                cnt++;
            }
        }
      /*  for(i=0;i<cnt;i++)
            printf("%lld\n",sortCount[i]);*/
       // sort(sortCount,sortCount+cnt);
       // cout<<"count "<<cnt<<endl;
        for(int m=0;m<cnt;m++)
        {
            changes=0;
            for(i=0;i<cnt;i++)
            {
                if(sortCount[i]<sortCount[m])
                {
                    changes+=sortCount[i];
                }
                else if(sortCount[i]>sortCount[m]+k)
                {
                    changes+=(sortCount[i]-sortCount[m]-k);
                }
            }
            if(changes<minimum)
                minimum=changes;
        }

        printf("%lld\n",minimum);
    }
    return 0;
}
