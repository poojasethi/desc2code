#include <bits/stdc++.h>

using namespace std;

int t,k,l,j,min1,r,i,u,cur,d;
string s;
int a[27];
vector<int>v;

int main()
{
scanf("%d",&t);
while(t--)
{
  v.clear();
  memset(a,0,sizeof(a));
  cin>>s;
  cin>>k;
  for(i=0;i<s.size();i++)
  {
      a[s[i]-'a']+=1;
  }
  min1=s.size();
for(i=0;i<26;i++)
{
    cur=a[i];
    d=0;
    for(j=0;j<26;j++)
    {
        if(a[j]!=0 && a[j]<cur)
            d=d+a[j];
            else
                if(a[j]-cur>k)
                d=d+(a[j]-cur-k);
    }
    if(d<min1)
        min1=d;
}
cout<<min1<<"\n";
}
    return 0;
}
