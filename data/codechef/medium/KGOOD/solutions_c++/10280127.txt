#include<cstdio>
#include<iostream>
#include<cstring>
#include<climits>
using namespace std;
int main()
{
    int t,n,c,k,i,m,ans;
    char s[100008];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s %d",s,&k);
        n=-5;
        c=0;
        int a[26]={0};
        for(i=0;i<strlen(s);i++)
        {
            a[s[i]-97]++;
        }
        for(i=0;i<26;i++)
        {
            if(a[i]>n)
              n=a[i];

        }
        ans=INT_MAX;
        for(m=1;m<=n;m++)
        {
            c=0;
            for(i=0;i<26;i++)
            {
                if(a[i]<m)
                    c+=a[i];
                if(a[i]>(m+k))
                    c+=a[i]-(m+k);
            }
            ans=min(c,ans);

        }
        printf("%d\n",ans);
    }


}
