#include<bits/stdc++.h>
using namespace std;
int a[30];
vector<int> v;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        memset(a,0,sizeof(a));
        v.clear();
        string st;
        int k,ans=INT_MAX;
        cin>>st>>k;
        for(int i=0;i<st.length();i++)
            a[st[i]-'a']++;
        for(int i=0;i<26;i++)
        {
            if(a[i]!=0)
                v.push_back(a[i]);
        }
        for(int i=0;i<v.size();i++)
        {
            int c=0;
            for(int j=0;j<v.size();j++){
                if(v[j]>v[i]+k)
                    c+=(v[j]-(v[i]+k));
                else if(v[j]<v[i])
                    c+=v[j];
            }
            ans=min(c,ans);
        }
        printf("%d\n",ans);
    }
    return 0;
}
