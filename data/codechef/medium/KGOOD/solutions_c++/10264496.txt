int m,n,k;
int a[30],dp[30][30],mini,maxi;
string s;
int ans(int st,int en)
{
	if(st>=en)
		return 0;
	if((a[en]-a[st])<=k)
		return 0;
	if(dp[st][en]!=-1)
		return dp[st][en];
	dp[st][en]=min((ans(st,en-1)+a[en]-(a[st]+k)),(ans(st+1,en)+a[st]));
	return dp[st][en];
}
int main()
{
	int t;
	si(t);
	while(t--)
	{
		int x,ans1=0;
		for(int i=0;i<30;i++)
		{
			a[i]=0;
			for(int j=0;j<30;j++)
				dp[i][j]=-1;
		}
		cin>>s>>k;
		for(int i=0;i<s.size();i++)
		{
			a[s[i]-'a']++;
		}
		sort(a,a+30);
		for(int i=0;i<30;i++)
		{
			if(!a[i])
			{
				x=i;break;
			}
		}
		ans1=ans(x,29);
		
		cout<<ans1<<endl;
	}

	return 0;
}
