#include<bits/stdc++.h>
#define ll long long
using namespace std;
char s[100002];
ll t,k,hash[27];ll val[27],sum=0,tmp;
int main(){
    scanf("%lld",&t);
    while(t--){
       scanf("%s%lld",s,&k);   
        memset(hash,0,sizeof(hash));
        for(int i=0;s[i]!='\0';i++){
            hash[s[i]-'a']++;
        }
        vector<int > v;
        for(int i=0;i<26;i++){
        if(hash[i]){
            v.push_back(hash[i]); 
           }
        }
        int j=0;
       while(j!=v.size()){
           tmp=v[j];sum=0;
           for(int i=0;i<v.size();i++)
            {
              if(v[i]<tmp) 
              sum+=(v[i]);
              else if(v[i]-tmp>k)
              sum+=(v[i]-tmp-k);
            }
            val[j]=sum;
            j++;
       }
       ll mx=(ll)1e7;
       for(int i=0;i<v.size();i++){
           mx=min(mx,val[i]);
       }
       printf("%lld\n",mx);
       v.erase(v.begin(),v.end());
     }
    return 0;
}
