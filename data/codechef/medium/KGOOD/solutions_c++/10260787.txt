#include<bits/stdc++.h>

using namespace std;

int main()
{
	int t,n,k,*arr,i,j,counter1,counter2,l,temp,m;
	char w[100001];
	scanf("%d",&t);
	while(t--){
		scanf("%s",w);
		scanf("%d",&k);
		arr = new int[26];
		i=0;
		while(w[i]!='\0'){
			arr[w[i]-'a']++;
			i++;
		}
		n=i;
		//sorting letter array
        sort(arr,arr+26);
        //skipping all the zeros in the letter array
        for(i=0;i<26;i++){
            if(arr[i])
                break;
        }
        //since always arr[j]>arr[i], no need for absolute difference
        //if difference greater than k then set as shown
        temp = n;
        for(l=i;l<26;l++){
            m = arr[l];
            counter2=0;
            //cout<<"arr[l]: "<<m<<endl;
            for(j=i;j<26;j++){
                if(arr[j]<m)
                    counter2 += arr[j];
                else if((arr[j]- m)>k){
                    counter2 = counter2 + arr[j] - m - k;
                }
                //cout<<counter2<<" ";
            }
            //cout<<endl;
            if(temp>counter2){
                temp = counter2;
            }
        }
		printf("%d\n",temp);
	}
	return 0;
}
