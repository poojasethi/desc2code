#include<bits/stdc++.h>
using namespace std;
long long minimum(long long a,long long b){
	if(a<b)
		return a;
	return b;	
}
int main(){
	long long t,k,i,min,max,changes,count[26],j;
	char s[100009];
	scanf("%lld",&t);
	while(t!=0){
		t--;
		for(i=0;i<26;i++)
			count[i]=0;
		changes=0;
		scanf("%s",s);
		s[strlen(s)]='\0';
		scanf("%lld",&k);
		for(i=0;i<strlen(s);i++)
			count[s[i]-'a']++;
		min=9999;
		max=-9999;
		for(i=0;i<26;i++){
			if(count[i]>0){
				if(count[i]<min)
					min=count[i];
				if(count[i]>max)
					max=count[i];	
			}
		}
		min=999999;
		for(i=0;i<26;i++){
			changes=0;
			for(j=0;j<26;j++){
				if(count[i]>count[j])
					changes+=count[j];
				else if(count[j]>count[i]+k)
					changes+=count[j]-count[i]-k;
						
			}
			min=minimum(min,changes);
		}
		printf("%lld\n",min);
	}
	return 0;
}
