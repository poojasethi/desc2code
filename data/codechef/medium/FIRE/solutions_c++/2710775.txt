#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include <queue>

main()
{
int nodes[10001][5],n,s,t,test_cases,j,k,temp;
int node_color[10001];
scanf("%d",&test_cases);
std::queue<int> visit;

	while(test_cases--)
	{
		memset(node_color,0,sizeof(node_color));
		scanf("%d%d%d",&n,&s,&t);
		while(visit.size())
			visit.pop();
			
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&k);
			for(j=0;j<k;j++)
				scanf("%d",&nodes[i][j]);
			nodes[i][j]=-1;
		}
		for(int i=0;i<t;i++)
		{
			scanf("%d",&temp);
			node_color[temp]=-1;
			visit.push(temp);
		}
		
		if(node_color[s]==-1)
			{printf("no\n");continue;}
		while(visit.size())
		{
			temp = visit.front();
			visit.pop();
			for(j=0;nodes[temp][j]!=-1;j++)
			{
				if(node_color[nodes[temp][j]]==-1)
					continue;
				node_color[nodes[temp][j]]++;
				if(node_color[nodes[temp][j]]>1)
					{node_color[nodes[temp][j]]=-1;if(node_color[s]==-1)break;visit.push(nodes[temp][j]);}
			}
		}
		if(node_color[s]==-1)
			printf("no\n");
		else
			printf("yes\n");
	}
	
return 0;
}
