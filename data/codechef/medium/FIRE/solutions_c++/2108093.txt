const int maxn = (int)1e4 + 10;
vector<int> g[maxn];
bool good[maxn];
bool can(int v, int p) {
	if (good[v]) return false;
	int u1 = -1;
	int u2 = -1;
	for (int i = 0; i < sz(g[v]); i++) {
		int to = g[v][i];
		if (to == p) continue;
		if (u1 == -1) {
			u1 = to;
		} else {
			u2 = to;
		}
	}
	if (u2 == -1) return true;
	return can(u1,v) || can(u2,v);
}

		--s;
		for (int i = 0; i < n; i++) {
			good[i] = false;
			g[i].clear();
		}
		for (int i = 0; i < n; i++) {
			int c; scanf("%d",&c);
			for (int j = 0; j < c; j++) {
				int id; scanf("%d",&id);
				--id;
				g[i].pb(id);
			}
		}
		for (int i = 0; i < t; i++) {
			int c; scanf("%d",&c);
			--c;
			good[c] = true;
		}
		puts(can(s,-1) ? "yes" : "no");
	}

}
