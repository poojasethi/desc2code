vector<int> adl[10010];

bool chk(int t, int last)
{
	if(save[t]==1) return 0;
	if(adl[t].size()<=1) return 1;
	bool ans=0;
	for(int i=0;i<adl[t].size();i++) if(last!=adl[t][i])
		{
			ans=ans||chk(adl[t][i],t);
			//printf("%d %d\n",t,adl[t][i]);
		}
	return ans;
}
