#include<iostream>
#include<stdio.h>
using namespace std;
bool TrySaving(int nodes[][5] , int node , int parent);

int main()
{
    int t,n,s,m;
    scanf("%d",&t);
    while(t--)
    {
        int nodes[10001][5] = {-1};
        scanf("%d %d %d", &n , &s ,&m );
        for(int x = 1 ; x<=n ; x++)
        {
            scanf("%d" , &(nodes[x][0]));
            for(int y= 1 ; y<= nodes[x][0] ; y++)
            {
                scanf("%d" , &(nodes[x][y]));
            }
        }
        for(int x = 0 ;x< m ;x++)
        {
            int save ;
            scanf("%d" , &save);
            nodes[save][4] = 1;
        }

        if(TrySaving(nodes , s , 0) == true)
        {
            printf("yes\n");
        }
        else
        {
            printf("no\n");
        }
    }
    return 0;
}

bool TrySaving(int nodes[][5] , int node , int parent)
{
    if(nodes[node][4] == 1)
        return false;
    if(nodes[node][0] < 3 && parent !=0)
        return true;
    if(nodes[node][0] < 2 && parent ==0)
        return true;

    bool returnVal = false;
    for(int x= 1; x<= nodes[node][0] ; x++ )
    {
        if(nodes[node][x] == parent)
        continue;
        returnVal = returnVal || TrySaving(nodes , nodes[node][x] ,node ) ;
        if(returnVal == true)
        {
            return returnVal;
        }
    }
    return returnVal;
}
