#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int L,D,S,C;
		cin>>L>>D>>S>>C;
		double ans;
		ans=(D-1)*(log10(1+C))+(log10(S))-(log10(L));
		if(ans>=0)
			cout<<"ALIVE AND KICKING\n";
		else
		{
			cout<<"DEAD AND ROTTING\n";
		}
	}
}
