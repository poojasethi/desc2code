#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
int main()
{
    int t;
    long l,d,s,c,i;
    long long p;
    scanf("%d",&t);
    while(t--)
    {
        cin>>l>>d>>s>>c;
        for(i=1;i<=d;i++)
        {
            p=s*pow((1+c),(i-1));
            if(p>=l)
                break;
        }
        if(p>=l)
            cout<<"ALIVE AND KICKING\n";
        else
            cout<<"DEAD AND ROTTING\n";
    }
    return 0;
}
