#include<iostream>
#include<math.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int l,d,s,c;
		cin>>l>>d>>s>>c;
		int flag=0;
		long long int res=s;
		int x=d-1;
		while(x--)
		{
			res=res+res*c;
			if(res>=l)
			{
				flag=1;
				break;
			}
		}
		if(flag||((d==1)&&res>=l))
		cout<<"ALIVE AND KICKING";
		else
		cout<<"DEAD AND ROTTING";
		cout<<endl;
	}
}
