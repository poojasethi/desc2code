#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ull unsigned long long
int main()
{
	ll t,l,d,s,c,cal,i,st,flag;
	ull ans;
	scanf("%lld",&t);
	while(t--)
	{
		flag=0;
		scanf("%lld%lld%lld%lld",&l,&d,&s,&c);
		ans=s;
		cal=c+1;
		i=0;
		d--;
		st=1;
		while((d)&&(st<=l))
		{
			st=st*cal;
			d--;
		}
		if(st>=l)
		{
			flag=1;
		}
		else
		{
			if(ans*st>=l)
			{
				flag=1;
			}
			else
			{
				flag=0;
			}
		}
		if(flag)
			printf("ALIVE AND KICKING\n");
		else
			printf("DEAD AND ROTTING\n");
	}
	return 0;
}
