#include<bits/stdc++.h>

using namespace std;


int main(){
    //ifstream cin("a.txt");
    long tt;
    cin>>tt;
    long long l,d,s,c;
    while(tt--){
        cin>>l>>d>>s>>c;
        c++;
        long long ans = s;
        bool result = false;
        if(ans >= l)
            result=true;
        for(int i = 1 ; i < d ; i++){
            ans = ans*c;
            if(ans>=l){
                result = true;
                break;
            }
        }
        if(result)
            cout<<"ALIVE AND KICKING"<<endl;
        else
            cout<<"DEAD AND ROTTING"<<endl;
    }
}
