#include<cstdio>
using namespace std;
typedef long long ll;
int main(){
    int T;
    scanf("%d",&T);
    while(T--){
        ll L,D,S,C;;
        scanf("%lld %lld %lld %lld",&L,&D,&S,&C);
        ll X=S;
        bool fail=1;
        for(ll i=1;i<=D;i++){
            if(X>=L){
                fail=0;
                break;
            }
            X=X+(X*C);
        }
        if(fail){
            printf("DEAD AND ROTTING\n");
        }
        else
        {
            printf("ALIVE AND KICKING\n");
        }
    }
}
