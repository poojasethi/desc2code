#include<iostream>
using namespace std;
int main()
{
	long long int t,l,d,s,c;
	cin>>t;
	while(t--)
	{
		cin>>l>>d;
		cin>>s>>c;
		long long int i,num=s;
		for(i=2;i<=d;i++)
		{
			num=num*(1+c);
			if(num>=l)
			break;
		}
		if(num>=l)
		cout<<"ALIVE AND KICKING"<<endl;
		else
		cout<<"DEAD AND ROTTING"<<endl;
	}
	return 0;
}
