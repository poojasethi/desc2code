#include<iostream>
#include<math.h>
#include<stdio.h>
using namespace std;

int main()
{
    int T;
    long long int L,D,S,C;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%lld%lld%lld%lld",&L,&D,&S,&C);
        double sum = S*(pow(C+1,D-1));

            if(sum>=L)
            {
                printf("ALIVE AND KICKING\n");
            }
        else
         printf("DEAD AND ROTTING\n");
    }
    return 0;
}
