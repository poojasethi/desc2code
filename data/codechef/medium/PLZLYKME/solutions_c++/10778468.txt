#include <iostream>

using namespace std;
int main() {
	int T;
	cin >> T;
	for(int t=0;t<T;t++){
		long long L,D,S,C;
		cin >> L >> D >> S >> C;
		long long l = S;
		D--;
		while(D){
			if(l>=L)
				break;
			l = l * (C+1);
			
			D--;
		}
		if(l>=L)
			cout << "ALIVE AND KICKING" << endl;
		else
			cout << "DEAD AND ROTTING" << endl;
	}
	return 0;
}
