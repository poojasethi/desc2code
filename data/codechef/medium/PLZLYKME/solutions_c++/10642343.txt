inline ll int rd()
{
    ll x=0;
    int sign=1;
    char c;
    while(((c=g())<'0'||c>'9')&&c!='-');
    if(c=='-')
    sign=-1,c=g();
    while(c>='0'&&c<='9')
    x=(x<<3)+(x<<1)+c-48,c=g();
    return sign*x;
}
 
int main() {
   ll t;
   cin>>t;
   while(t--)
   {
       double l,d,s,c,ans;
       l=rd();
       d=rd();

       c=rd();
       ans=s*(pow((1+c),(d-1)));
      if(ans>=l)
