#include <iostream>
#include <bits/stdc++.h>
#include <limits.h>
typedef long long int ll;
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		ll l,d,s,c;
		cin>>l>>d>>s>>c;
		ll k=d;
		if(d<2)
			d = s;
		for(ll i=2;i<=k;i++)
		{
			d = (1+c)*s;
			if(d>=l) break; 
			s = d;
		}
		if(d>=l) cout<<"ALIVE AND KICKING"<<endl;
		else cout<<"DEAD AND ROTTING"<<endl;
	}
	return 0;
}
