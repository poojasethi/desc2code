#include<bits/stdc++.h>
using namespace std;
int t,i;
int main(){
    scanf("%d",&t);
    while(t--){

           int flag=0;
           long long int l,d,s,sum=0,c;
           scanf("%lld%lld%lld%lld",&l,&d,&s,&c);
           sum=s;
           while(d>1){
                d--;
                sum+=sum*c;
                    if(sum>=l){
                        flag=1;
                        break;
                }
            }
            //cout<<sum<<endl;
            if(flag || sum>=l)
                printf("ALIVE AND KICKING\n");
            else
                printf("DEAD AND ROTTING\n");

    }
return 0;
}
