#include <iostream>

using namespace std;

void testCase()
{
    int L, D, S, C;
    cin >> L >> D >> S >> C;
    long long int likes = S;
    for (int i = 0; i < D; ++i)
    {
        if (likes >= L)
        {
            cout << "ALIVE AND KICKING" << endl;
            return ;
        }
        likes *= C+1;
    }
    
    cout << "DEAD AND ROTTING" << endl;
}

int main()
{
    int T; cin >> T;
    for (int i = 0; i < T; ++i)
        testCase();
    return 0;
}
