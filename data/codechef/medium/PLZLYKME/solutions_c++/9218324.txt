#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		long long l, d, s, c, n;
		scanf("%lld %lld %lld %lld", &l, &d, &s, &c);
		n=s;
		for(long long i=1; i<d; i++)
		{
			n = s+c*s;
			s=n;
			if( n>=l )
				break;
		}
		if( n>=l )
			printf("ALIVE AND KICKING \n");
		else
			printf("DEAD AND ROTTING \n");
	}
	return 0;
}
