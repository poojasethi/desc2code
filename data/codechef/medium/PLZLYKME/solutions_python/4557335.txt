tc=input()
while tc:
    tc=tc-1
    l,d,s,c=raw_input().split()
    l=(int)(l)
    d=(int)(d)
    s=(int)(s)
    c=(int)(c)
    sum=s;
    while d>1:
        d=d-1
        sum=sum+c*sum
        if sum>=l:
            break
    if sum>=l:
        print("ALIVE AND KICKING")
    else:
        print("DEAD AND ROTTING")
