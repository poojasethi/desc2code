import sys, math

lim = 10**9

n = int(sys.stdin.readline().strip())
for _ in range(n):
    L, D, S, C = map(int, sys.stdin.readline().strip().split())

    if D - 1 >= math.log(lim / float(S), C + 1) + 10:
        print "ALIVE AND KICKING"
        continue

    final = S * pow(C + 1, D - 1)
    if final >= L:
        print "ALIVE AND KICKING"
    else:
        print "DEAD AND ROTTING"
