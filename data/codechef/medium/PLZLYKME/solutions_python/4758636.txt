import sys

for cases in xrange(int(sys.stdin.readline())):
    l,d,s,c = map(int,sys.stdin.readline().split())
    ans = s
    for i in xrange(d-1):
        ans += ans*c
        if ans >= l:
            break
    if ans >= l:
        print "ALIVE AND KICKING"
    else:
        print "DEAD AND ROTTING"
