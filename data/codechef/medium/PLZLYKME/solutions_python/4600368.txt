t=input()
while t>0:
    l,d,s,c=map(int,raw_input().split())
    x=l*1.0/s
    if d==1:
        if s>=l:
            x=1
        else:
            x=0
    else:
        x=pow(x,1.0/(d-1.0))
        if (1+c)>=x:
            x=1
        else:
            x=0 
    if x==1:
        print "ALIVE AND KICKING"
    else:
        print "DEAD AND ROTTING"
    t-=1
