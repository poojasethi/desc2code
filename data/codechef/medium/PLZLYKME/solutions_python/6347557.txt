t = int(raw_input())
for case in range(t):
    l, d, s, c = map(int, raw_input().split())
    temp = 1
    ans = s 
    i = 1
    while i < d:
    
        temp *= 1 + c
        ans = s * temp
        if ans >= l:
            break
        i += 1
   
    if ans >= l:
        print "ALIVE AND KICKING"
    else:
        print "DEAD AND ROTTING"
