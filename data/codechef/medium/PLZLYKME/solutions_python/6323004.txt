a = input()
while (a > 0):
    b = raw_input()
    c,d,e,f = [int(i) for i in b.split()]
    g = 1
    h = 0
    if (e >= c):
        print "ALIVE AND KICKING"
    else:
        while (g < d):
            e = e + f*e
            if (e >= c):
                h  = 1
                g = d
            else:
                g = g+1
        if (h == 1):
            print "ALIVE AND KICKING"
        else:
            print "DEAD AND ROTTING"
    a = a-1
