#include <bits/stdc++.h>

using namespace std;

int main()
{
    set<char> s;
    string in;
    int t;
    scanf("%d",&t);
    while(t--)
    {
        cin>>in;
        int l=in.length();
        for(int i=0;i<l;i++)
            s.insert(in[i]);
        cout<<s.size()<<endl;
        s.clear();
    }
    return 0;
}
