#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,f=0,i,m;
    string a;
    cin>>t;
    while(t--)
    {
        int c[26]={0};
        cin>>a;
        for(i=0;a[i]!='\0';i++)
        {
            m=a[i];
            if(c[m-97]==0)
            {
            f++;
            c[m-97]=1;
            }
        }
        cout<<f<<"\n";
        f=0;
    }
    return 0;
}
