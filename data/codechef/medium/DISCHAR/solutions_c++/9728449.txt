#include <iostream>
#include <cstring>
using namespace std;

int main() {
	int t,n,i,c,ans;
	char a[100009],ch;
	scanf("%d",&t);
	while(t--)
	{
		int b[30];
		for(i=0;i<30;i++)
		{
			b[i]=0;
		}
		scanf("%s",&a);
		n=strlen(a);
		ans=0;
		for(i=0;i<n;i++)
		{
			ch=a[i];
			c=ch;
			if(b[c-97]==0)
			{
				b[c-97]++;
				ans++;
			}
			else
			{
				b[c-97]++;
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
