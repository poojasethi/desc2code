#include<iostream>
#include<cstring>
using namespace std;
int main(){
	int test,count;
	char a[100010];
	int ch[26];
	cin>>test;
	while(test--){
		memset(ch,0,sizeof ch);
		count=0;
		cin>>a;
		for(int i=0;a[i]!='\0';i++){
			if(ch[a[i]-97]==0){
				count++;
				ch[a[i]-97]++;
			}
			if(count==26)
				break;
		}
	cout<<count<<"\n";

	}
return 0;
}
