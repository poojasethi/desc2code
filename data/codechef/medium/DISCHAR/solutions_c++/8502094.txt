#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;

int main(void)
{
    char a[100001];
    int maximum;
    int i;
    bool countChar[257];
    int tc;
    cin>>tc;
    while(tc--)
    {
        maximum = 0;
        for(i = 0; i <= 256; i++)
            countChar[i] = false;
        scanf("%s",a);
        for(i = 0; i < strlen(a); i++)
        {
            if(!countChar[a[i]])
                maximum++;
            countChar[a[i]] = true;
        }
        cout<<maximum<<"\n";
    }
}
