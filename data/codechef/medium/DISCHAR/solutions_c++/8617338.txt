#include <bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
      string s;
      cin>>s;
      int len,count=0;
      len=s.size();
      vector<int>a(12345,0);
      for (int i = 0; i < len; ++i)
      {
      	a[s[i]-'a']++;
      }
      for (int i = 0; i <12345; ++i)
      {
      	if (a[i]>=1)
      	{
      		count++;
      	}
      }
      printf("%d\n",count );
	}
	return 0;
}
