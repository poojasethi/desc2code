#include<bits/stdc++.h>
using namespace std;
int main(void)
{
    int t;
    for(scanf("%d", &t); t--; )
    {
        char str[100001];
        scanf("%s", str);
        bool a[26];
        memset(a, false, sizeof(a));
        int cnt=0, l=strlen(str);
        for(int i=0; i<l; i++)
            a[(int)(str[i]-'a')]=true;
        for(int i=0; i<26; i++)
            if(a[i])
            cnt++;
        printf("%d\n", cnt);
    }
    return 0;
}
