#include<stdio.h>
#include<stdlib.h>

char a[100010];
int b[26];
int main()
{
	int i,j,k,l,t,cnt;
	scanf("%d",&t);
	while(t--)
	{
		cnt=0;
		scanf("%s",a);
		for(i=0;i<26;i++)
		b[i]=0;
		for(i=0;a[i]!='\0';i++)
		{
			if(b[a[i]-'a']==0)
			{
				b[a[i]-'a']++;
				cnt++;
			}
			else
			b[a[i]-'a']++;
		}
		
		printf("%d\n",cnt);
	}
}
