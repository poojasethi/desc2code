#include<bits/stdc++.h>
using namespace std;
char a[100001];
int main()
   {
   int n;
   scanf("%d",&n);
   int f[26];
   while(n--)
      {
      scanf("%s",a);
      memset(f,0,sizeof(f));
      for(int i=0;a[i];i++)
         f[a[i]-'a']=1;
      int ans=0;
      for(int i=0;i<26;i++)
         ans+=f[i];
      printf("%d\n",ans);
//      
}
   return 0;
   }
