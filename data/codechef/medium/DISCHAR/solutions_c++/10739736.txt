#include<iostream>
using namespace std;
int main(){
	int t,i,count;
	cin>>t;
	string s;
	while(t--){
		cin>>s;
		int a[26];
		for(i=0;i<26;i++)
			a[i]=0;
		for(i=0;i<s.length();i++)
			++a[(int)s[i]-97];
		for(i=0,count=0;i<26;i++){
			if(a[i]>0)
			++count;
		}
		cout<<count<<endl;
	}
	return 0;
}
