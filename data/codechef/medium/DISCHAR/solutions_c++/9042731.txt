#include<cstdio>
#include<string.h>
#include<stdlib.h>

int cmpfn(const void *a,const void *b)
{
	char *ptr1=(char *)a;
	char *ptr2=(char *)b;
	return (*ptr1)-(*ptr2);
}
main()
{
	int test;
	char a[100100]={'\0'};
	int long long i,count;
	scanf("%d",&test);
	while(test--)
	{
		count=1;
		fflush(stdin);
		scanf("%s",a);
		qsort(a,strlen(a),sizeof(char),cmpfn);
		for(i=1;i<strlen(a);i++)
			if(a[i]!=a[i-1])count++;
		printf("%lld\n",count);
		for(i=0;i<100100;i++)
		  a[i]='\0';
	}
}
