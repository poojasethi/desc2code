#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long int t,n,k,m,i;
    cin>>t;
    while(t--)
    {
    	string str;
    	cin>>str;
    	long long int i,p[26]={0},ctr=0;
    	for(i=0;str[i]!='\0';++i)
    	{
    	    ++p[str[i]-'a'];
    	}
    	for(i=0;i<26;++i)
    	{
    	    if(p[i]>0)
    	    ++ctr;
    	}
    	cout<<ctr<<"\n";
    }
    return 0;
}
