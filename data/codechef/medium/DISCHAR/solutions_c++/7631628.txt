int distinct(string A, int first , int last)
{
	char B[27];
	for(int i=1;i<27;i++)
	{
	 B[i]=('a'+i-1);
	}
	
	
	int C[27];
	for(int i=0;i<27;i++)
	{
		C[i]=0;
	}
	
	
	
	for(int i=first;i<=last;i++ )
	{
		for(int j=1;j<27;j++)
		{
			if(A[i]==B[j])
			{
				C[j]++;
			}
		}
	}
	int sum=0;
	for(int i=1;i<27;i++)
	{
		if(C[i]>=1)
		{
			sum++;
		}
	}
	
	return sum;
}
