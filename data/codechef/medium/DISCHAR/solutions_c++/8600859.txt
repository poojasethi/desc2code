using namespace std;

#define pb push_back
#define clr(a) memset(a, 0, sizeof(a))
#define forl(i,l,u) for(int i=(l);i<(u);i++)
#define forn(i,n) forl(i,0,n)

typedef long long ll;
typedef pair<int, ll> il;

const int maxn=1e5+11;
const int ascii=256;
int main(){
	ios::sync_with_stdio(false);cin.clear();
	//cerr<<maxn;
	int t;
	cin>>t;
	char str[maxn];
	bool flag[ascii];
	while(t--){
		cin>>str;
		clr(flag);
		int ans=0;
		int len=strlen(str);
		forn(i, len){
			if(!flag[str[i]]){
				ans++;
				flag[str[i]]=true;
			}
		}
		cout<<ans<<endl;
	}
}
