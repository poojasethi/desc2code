using namespace std;

#define maxn 300111
#define logN 20

vector <int> adj[maxn];
int f[maxn][logN], depth[maxn], n,vis[maxn];

void dfs(int u) {
    for (int i = 1; i < logN; i++)
    if(f[u][i - 1]!=-1)
        f[u][i] = f[f[u][i - 1]][i - 1];
        vis[u]=1;

    for (int i = 0; i < (int) adj[u].size(); i++) {
        int v = adj[u][i];

        if (!vis[v]) {
            f[v][0] = u;
            depth[v] = depth[u] + 1;
            dfs(v);
        }
    }
}

int lca (int u, int v) {
    if (depth[u] < depth[v]) swap(u, v);

    for (int i = logN - 1; i >= 0; i--)
        if (f[u][i]!=-1&&(depth[u] -(1<<i))>= depth[v]) {
            u = f[u][i];
        }

    if (u == v) return u;

    for (int i = logN - 1; i >= 0; i--)
        if (f[u][i] != f[v][i]) {
            u = f[u][i];
            v = f[v][i];
        }

    return f[u][0];
}

int dist (int u, int v) {
    int x = lca(u, v);
    int res = depth[u] + depth[v] - 2 * depth[x];
    return res;
}
