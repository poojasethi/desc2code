#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

struct Interval
{
  int x,y;
};

bool operator <(Interval a,Interval b)
{
  return a.y<b.y;
}

int get()
{
  char c;
  while(c=getchar(),(c<'0'||c>'9')&&(c!='-'));
  bool flag=(c=='-');
  if(flag)
    c=getchar();
  int x=0;
  while(c>='0'&&c<='9')
    {
      x=x*10+c-48;
      c=getchar();
    }
  return flag?-x:x;
}

void output(int x)
{
  if(x<0)
    {
      putchar('-');
      x=-x;
    }
  int len=0,data[10];
  while(x)
    {
      data[len++]=x%10;
      x/=10;
    }
  if(!len)
    data[len++]=0;
  while(len--)
    putchar(data[len]+48);
  putchar('\n');
}

int main()
{
  int t=get();
  while(t--)
    {
      int n=get(),m=get()-1;
      static Interval a[200];
      static int data[400];
      for(int i=0;i<n;i++)
        {
          data[i*2]=a[i].x=get()-1;
          data[i*2+1]=a[i].y=get();
        }
      sort(data,data+n*2);
      for(int i=0;i<n;i++)
        {
          a[i].x=lower_bound(data,data+n*2,a[i].x)-data;
          a[i].y=lower_bound(data,data+n*2,a[i].y)-data;
        }
      sort(a,a+n);
      static int s[400];
      memset(s,0,sizeof(s));
      int ans=n;
      for(int i=0;i<n;i++)
        {
          bool flag=true;
          for(int j=a[i].x;flag&&j<a[i].y;j++)
            flag=(s[j]<m);
          if(flag)
            {
              for(int j=a[i].x;j<a[i].y;j++)
                s[j]++;
              ans--;
            }
        }
      output(ans);
    }
  return 0;
}
