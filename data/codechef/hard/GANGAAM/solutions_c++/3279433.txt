vector<pii> V;
bool cmp(const pii &a, const pii &b)
{
    if(a.ff == b.ff)
        return a.ss > b.ff;
    return a.ff < b.ff;
}
bool cmp2(const pii &a, const pii &b)
{
    return a.ff > b.ff;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        V.clear();
        int n,x;
        scanf("%d %d",&n,&x);
        FOR(i,n)
        {
            int s,e;
            scanf("%d %d",&s,&e);
            V.pb(pii(s,e));
        }
        sort(V.begin(),V.end(),cmp);
        vector<bool> vis;
        FOR(i,n)
        {
            vis.pb(false);
        }
        int ans = 0;
        FOR(i,n)
        {
            int cur = 1;
            for(int j=0;j<i;j++)
            {
                if(V[j].ss >= V[i].ff && !vis[j])
                    cur++;
            }
            if(cur >= x)
            {
                ans++;
                vector<pii> G;
                for(int j=0;j<i;j++)
                {
                    if(V[j].ss >= V[i].ff && !vis[j])
                        G.pb(pii(V[j].ss,j));
                }
                G.pb(pii(V[i].ss,i));
                sort(G.begin(),G.end(),cmp2);
                int z = G[0].ss;
                vis[z] = true;
              //  cout << "Select " << z << endl;
            }
        }
        printf("%d\n",ans);
    }
return 0;
}
