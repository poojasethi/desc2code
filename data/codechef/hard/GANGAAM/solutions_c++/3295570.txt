#include<iostream>
#include<cstring>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,x;
		cin>>n>>x;
		int s[n];
		int e[n];
		int c[n];
		for(int i=0;i<n;i++)
		{
			cin>>s[i];
			cin>>e[i];
			c[i]=0;
		}
		for(int i=0;i<n;i++)
		{
			
			for(int j=i+1;j<n;j++)
			{
				if(e[j]<e[i])
				{
					swap(e[j],e[i]);
					swap(s[j],s[i]);
				}
				else if(e[j]==e[i])
				{
					if(s[j]<s[i])
					swap(s[j],s[i]);
				}
			}
		}
			int ans=0;
			for(int i=0;i<n;i++)
			{
				bool flag=0;
				for(int j=0;j<=i;j++)
				{
					if(e[j]>=s[i])
					{
						c[j]++;
					}
					if(c[j]==x)
					flag=1;
				}
				if(flag)
				{
					ans++;
					for(int j=0;j<=i;j++)
					c[j]--;
				}
			}
			cout<<ans<<"\n";
		
	}
	return 0;
}
