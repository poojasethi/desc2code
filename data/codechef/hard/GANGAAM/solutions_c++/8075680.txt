#include <bits/stdc++.h>
using namespace std;
 
const int N = 210;
struct point
{
    int x,y;
    void read() {
        scanf("%d%d",&x,&y);
    }
    friend bool operator < (const point &p,const point &q) {
        return p.y < q.y;
    }
}a[N];
 
void solve()
{
    int visit[N];
    memset(visit,0,sizeof(visit));
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i = 1; i <= n; i ++) a[i].read();
    sort(a + 1,a + n + 1);
    int ans = 0;
    for(int i = 1; i <= n; i ++) {
        bool flag = false;
        for(int j = 1; j <= i; j ++)
            if(a[i].x <= a[j].y) {
                visit[j] ++;
                if(visit[j] == m) {
                    flag = true;
                }
            }
        if(flag) {
            ans ++;
            for(int j = 1; j <= i; j ++)
                if(a[i].x <= a[j].y) {
                    visit[j] --;
                }
        }
    }
    cout << ans << endl;
}
 
int main()
{
    int t;
    cin >> t;
    while(t --) solve();
    return 0;
}
