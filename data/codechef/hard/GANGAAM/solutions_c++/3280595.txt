//#include <fstream>
#include <iostream>
using namespace std;

#include <algorithm>
#define x first
#define y second
#define mp make_pair
#define pii pair<int,int>
#define f cin

//ifstream f("wow.in");
pii A[666];
int P[666];
bool fr[666];

int main()
{
	int tt,nrt,n,i,j,xx;
	f>>nrt;
	
	for(tt=1;tt<=nrt;++tt)
	{
	   f>>n>>xx;
	   int k=0,numar=0;
	   
	   for(i=1;i<=n;++i) 
	   {
			f>>A[i].x>>A[i].y;
    	    P[++k]=A[i].x;
			P[++k]=A[i].y;
	   }
	   sort(P+1,P+k+1);
	   sort(A+1,A+n+1);
	   for(i=1;i<=n;++i) fr[i]=false;
	   
	   int tps=1,result=0,st=0;
	   for(i=2;i<=k;++i)
		   if (P[i]!=P[tps]) P[++tps]=P[i];
	   
	   for(i=1;i<=tps;++i)
	   {
		   while (st+1<=n&&A[st+1].x<=P[i])
		   {
			   ++st;
			   ++numar;
			   fr[st]=true;
		   }
		   for(j=1;j<=st;++j) 
				if (A[j].y<P[i]&&fr[j]==true) fr[j]=false,--numar;
		   
		   while (numar>=xx)
		   {
			   int maxy=0,ind;
			   for(j=1;j<=st;++j) 
				   if (fr[j]&&A[j].y>maxy) maxy=A[j].y,ind=j;
			   
			   fr[ind]=false;
			   --numar;
			   ++result;
		   }
	   }
	   cout<<result<<'\n';
	}
	return 0;
}
