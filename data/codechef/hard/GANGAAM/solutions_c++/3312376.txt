#include<iostream>
#include<cstring>
using namespace std;


int S[210];
int E[210];
int C[210];
int N,X;
int T;


int main(){
    int i,j;
    cin>>T;
    while(T--){
        cin>>N>>X;
        for(i=0;i<N;++i){
            cin>>S[i]>>E[i];
        }
        for(i=0;i<N;++i){
            for(j=i+1;j<N;++j){
                if(E[j]<E[i]){
                    int temp = E[i];
                    E[i] = E[j];
                    E[j] = temp;
                    temp = S[i];
                    S[i] = S[j];
                    S[j] = temp;
                }
                else if(E[j]==E[i]&&S[j]>S[i]){
                    int temp = S[i];
                    S[i] = S[j];
                    S[j] = temp;
                }
            }
        }
        memset(C,0,sizeof(C));
        int res = 0;
        for(i=0;i<N;++i){
            bool flag = false;
            for(j=0;j<=i;++j){
                if(E[j]>=S[i]){
                    C[j]++;
                    if(C[j]==X){
                        flag = true;
                    }
                }
            }
            if(flag){
                ++res;
                for(j=0;j<=i;++j){
                    if(E[j]>=S[i]){
                        C[j]--;
                    }
                }
            }
        }
        cout<<res<<endl;
        
    }
    return 0;
}
