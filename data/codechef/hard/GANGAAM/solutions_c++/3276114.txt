pii lol[205];
int use[205];

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,x;
		scanf("%d%d",&n,&x);
		FOR(a,1,n)
		{
			scanf("%d%d",&lol[a].fi,&lol[a].se);
			use[a] = 0;
		}
		sort(lol+1,lol+n+1);
		int ans = 0;
		FOR(a,1,n)
		{
			int sum = 0;
			FOR(b,1,a)
			{
				if (lol[b].se >= lol[a].fi && !use[b]) sum++;
			}
			if (sum >= x)
			{
				int best = a;
				FOR(b,1,a)
				{
					if (lol[b].se >= lol[a].fi && !use[b])
					{
						if (lol[best].se < lol[b].se) best = b;
					}
				}
				use[best] = 1;
				ans++;
			}
		}
		printf("%d\n",ans);
	}
}
