#include <bits/stdc++.h>
using namespace std;
struct _ { ios_base::Init i; _() { cin.sync_with_stdio(0); cin.tie(0); } } _;

int s[201],e[201],c[201];

int main() 
{ 
	int t,n,x,temp,ans,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>x;
		for(i=0;i<n;i++)
		{
			cin>>s[i]>>e[i];
		}
		for(i=0;i<n;i++)
		{
			for(j=i+1;j<n;j++)
			{
				if(e[j]<e[i])
				{
					temp=e[i];
					e[i]=e[j];
					e[j]=temp;
					temp=s[i];
					s[i]=s[j];
					s[j]=temp;
				}
				else if(e[j]==e[i]&&s[j]>s[i])
				{
					temp=s[i];
					s[i]=s[j];
					s[j]=temp;
				}
			}
		}
		ans=0;
		memset(c,0,sizeof(c));
		for(i=0;i<n;i++)
		{
			bool flag=false;
			for(j=0;j<=i;j++)
			{
				if(e[j]>=s[i])
				{
					c[j]++;
					if(c[j]==x)
					{
						flag=true;
					}
				}
			}
			if(flag)
			{
				ans++;
				for(j=0;j<=i;j++)
				{
					if(e[j]>=s[i])
					{
						c[j]--;
					}
				}
			}
		}
		cout<<ans<<"\n";
	}
	
}
