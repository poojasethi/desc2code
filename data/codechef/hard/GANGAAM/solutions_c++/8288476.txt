#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ull unsigned long long
#define pb push_back
#define ft first
#define se second
#define mp make_pair
int main(int argc, char const *argv[]){
    int t;
    scanf("%d", &t);
    while(t--){
    	int n, K;
    	scanf("%d%d", &n, &K);
    	set<pair<int, pair<int, int> > >init, finit;
    	for(int i = 0; i < n; ++i){
    		int x, y;
    		scanf("%d%d", &x, &y);
    		init.insert(mp(x, mp(y, i)));
    	}//for i
    	set<pair<int, pair<int, int> > > :: iterator it;
    	set<pair<int, pair<int, int> > > :: reverse_iterator rit;
    	int ans=0;
    	while(init.size()>0){
    		it=init.begin();
    		pair<int, pair<int, int> > next=(*it);
    		init.erase(it);
    		it=finit.begin();
    		while(it!=finit.end() && (*it).ft<next.ft){
    			finit.erase(it);
    			it=finit.begin();
    		}
    		finit.insert(mp(next.se.ft, mp(next.ft, next.se.se)));
    		if(finit.size()==K){
    			ans++;
    			rit=finit.rbegin();
    			finit.erase(*rit);
    		}
    	}
    	printf("%d\n", ans);
    }
	return 0;
}
