inline void inp(int &n ) {//fast input function
  n=0;
  int ch=x,sign=1;
  while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=x;}
  while( ch >= '0' && ch <= '9' )
  n=(n<<3)+(n<<1)+ ch-'0', ch=x;
  n=n*sign;
}

int main(){
	int t;
	inp(t);
	while(t--){
		int n,k;
		inp(n);
		inp(k);
		vector<pair<int,int> > times;
		for(int i=0;i<n;i++){
			int s,e;
			inp(s);
			inp(e);
			times.push_back(make_pair(s,e));
		}
		sort(times.begin(),times.end());
		if(n==1){
			printf("%d\n",n);
			continue;
		}
		int check[n];
		int ret = 0;
		memset(check,0,sizeof check);
		for(int i=0;i<n;i++){
			int no_frnds = 1;
			for(int j=0;j<i;j++){
				if(!check[j] && (times[i].first <= times[j].second)){
					no_frnds++;
				}
			}
			if(no_frnds >= k){
				int idx = i;
				for(int j=0;j<i;j++){
					if(!check[j] && (times[i].first <= times[j].second)){
						if(times[j].second > times[idx].second){
							idx = j;
						}
					}
				}
				ret++;
				check[idx]=1;		
			}
		}
		printf("%d\n",ret);
	}
	return 0;
}
