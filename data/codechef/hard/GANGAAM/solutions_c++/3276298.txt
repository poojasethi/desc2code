#include<cstdio>
#include<iostream>
#include<algorithm>
#include<set>
#include<cstring>
#include<cassert>
#include<vector>
#include<utility>
using namespace std;


int main()
{

    int T,i,j,k;


   // freopen("in.txt","r",stdin);

    cin>>T;






    int n,x;
    while(T--)
    {
        cin>>n>>x;

        vector< pair<int,int> >vp;
        for(k=0;k<n;k++){scanf("%d%d",&i,&j);vp.push_back(make_pair(j,i)); }


        sort(vp.begin(),vp.end());



        int res=0;

        int end=0;
        int cnt=0;

        vector<bool> vst(n,0);



        for(k=0;k<n;k++)
        {

//            if(vst[k])continue;

          //  cout<<k<<" ";

            vector<int>v;
            for( i=0;i<n;i++ )
            {
                if(vst[i])continue;
                if( vp[i].second<=vp[k].first && vp[i].first>=vp[k].first )
                {
                    v.push_back( i );
                }
            }

            if( v.size()>=x )
            {
                for(i=x-1;i<v.size();i++)
                {
                    vst[v[i]]=1;
                }//cout<<endl;
                res+=max(0,(int)v.size()-x+1);
            }
        }


        cout<<res<<endl;
    }


    return 0;
}
