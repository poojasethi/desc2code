// CodeChef - GANGAAM
// Greedy Technique

#include <stdio.h>
#include <string.h>
#include <iostream>

using namespace std;

int n, x;
int start0[210];
int end0[210];
int count0[210];

int main()
{
  int t, ans;

  scanf("%d", &t);
  while(t--)
  {
    scanf("%d %d", &n, &x);
    for(int i = 0; i < n; ++i)
     scanf("%d %d", &start0[i], &end0[i]);

    for(int i = 0; i < n; ++i)
     for(int j = i + 1; j < n; ++j)
      if(end0[j] < end0[i])
       swap(start0[i], start0[j]), swap(end0[i], end0[j]);
      else if(end0[j] == end0[i] && start0[j] > start0[i])
       swap(start0[i], start0[j]);

    memset(count0, 0, sizeof(count0));
    ans = 0;
    for(int i = 0; i < n; ++i)
    {
      int flag = 0;
      for(int j = 0; j <= i; ++j)
      {
        if(end0[j] >= start0[i])
        {
          ++count0[j];
          if(count0[j] == x) flag = 1;
        }
      }

      if(flag)
      {
        ++ans;
        for(int j = 0; j <= i; ++j)
         if(end0[j] >= start0[i]) count0[j]--;
      }
    }
        
    printf("%d\n", ans);      
  }

  return 0;
}
