#define mp make_pair

#define ss second

#define all(a)  a.begin(), a.end()
#define allr(a) a.rbegin(), a.rend()
#define SL(n) scanf("%lld", &n)
#define PL(n) printf("%lld", n)
#define fill(a, x) memset(a, x, sizeof(a));
#define mod 1000000007
#define eps 0.000000001

using namespace std;
typedef long long LL;
typedef vector <LL> VL;
typedef map <LL, LL> ML;
typedef pair<LL, LL> PL;
typedef vector <PL> VPL;

int main() {
    int T;
    cin >> T;
    int N, K;
    while(T--){
        scanf("%d%d", &N, &K);
        int S[N + 5];
        int E[N + 5];
        for(int n=0;n<N;++n){
            scanf("%d%d", &S[n], &E[n]);
        }
        for(int n=0;n<N;++n){
            for(int m=n+1;m<N;++m){
                if(E[m] < E[n]){
                    S[n] = S[m] + S[n] - (S[m] = S[n]);
                    E[n] = E[m] + E[n] - (E[m] = E[n]);
                }
                else if(E[m] == E[n] && S[m] > S[n]){
                    S[n] = S[m] + S[n] - (S[m] = S[n]);
                }
            }
        }
        int Ans = 0;
        int Count[N + 5];
        fill(Count, (int)0);
        for(int n=0;n<N;++n){
            bool flag = false;
            for(int m=0;m<=n;++m){
                if(E[m] >= S[n]){
                    Count[m]++;
                    if(Count[m] == K){
                        flag = true;
                    }
                }
            }
            if(flag){
                Ans++;
                for(int m=0;m<=n;++m){
                    if(E[m] >= S[n]){
                        Count[m]--;
                    }
                }
            }
        }
        printf("%d\n", Ans);
    }
    return 0;
}
