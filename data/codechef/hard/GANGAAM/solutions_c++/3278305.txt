using namespace std;

int main(){
	int t;
	cin>>t;
	while(t--){
		int n,x;
		cin>>n>>x;
		vector<pair<int,int> > times;
		for(int i=0;i<n;i++){
			int s,e;
			cin>>s>>e;
			times.push_back(make_pair(s,e));
		}
		sort(times.begin(),times.end());
		if(n==1){
			cout<<n<<endl;
			continue;
		}
		int check[n];
		int ret = 0;
		memset(check,0,sizeof check);
		for(int i=0;i<n;i++){
			int no_frnds = 1;
			for(int j=0;j<i;j++){
				if(!check[j] && (times[i].first <= times[j].second)){
					no_frnds++;
				}
			}
			if(no_frnds >= x){
				int idx = i;
				for(int j=0;j<i;j++){
					if(!check[j] && (times[i].first <= times[j].second)){
						if(times[j].second > times[idx].second){
							idx = j;
						}
					}
				}
				ret++;
				check[idx]=1;		
			}
		}
		cout<<ret<<endl;
	}
	return 0;
}
