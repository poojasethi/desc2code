using namespace std;
typedef long long ll ;
typedef unsigned long long ull;
long long INF = 1e9 ;
typedef pair < long long  , long long  > pii ;
pii a[205] ;
int ct[205] ;
int comp ( pii a , pii b )
{
    if ( a.first == b.first )
        return a.second < b.second ;
    return a.first < b.first ;
}
int main()
{
int t , ans , n , x ;
cin >> t ;
while ( t--)
{
    ans = 0 ;
    cin >> n >> x ;
    memset ( ct , 0 , sizeof(ct) );
    for ( int i = 0 ;  i < n ; i++)
    {
        cin >> a[i].first >> a[i].second ;
    }
    sort ( a , a + n , comp );
    for ( int i = 0 ; i < n ; i++)
    {
        int c = 0 ;
        for ( int j = 0 ; j <= i ; j++)
        {
            if ( a[j].second >= a[i].first)
            {
                if ( ct[j] == 0 )
                {
                    c++ ;
                }
            }
        }
        if ( c >= x )
        {
            ans++  ;
            int u = i ;
            for ( int j = 0 ; j <= i ; j++ )
            {
                 if ( a[j].second >= a[i].first)
            {
                if ( ct[j] == 0 )
                {
                    if ( a[j].second > a[u].second) // subset  //
                        u = j ;
                }
            }
            }
            ct[u] = 1 ;
        }
    }
    cout << ans << endl ;

}
 return 0;
}
