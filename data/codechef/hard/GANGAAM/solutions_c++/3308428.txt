#include<iostream>
#include<cstring>
using namespace std;

long int S[201],E[201];
int C[201];

int main()
{
    int t;
    cin>>t;

    while(t--)
    {
        int n,i,j,x;
        cin>>n>>x;

        for(i=0;i<n;++i)
        cin>>S[i]>>E[i];

        for(i=0;i<n;++i)
        for(j=i+1;j<n;++j)
        if(E[j]<E[i])
        {
            long int temp=S[i];
            S[i]=S[j];
            S[j]=temp;

            temp=E[i];
            E[i]=E[j];
            E[j]=temp;
        }
        else if(E[j]==E[i]&&S[j]>S[i])
        {
            long int temp=S[j];
            S[j]=S[i];
            S[i]=temp;
        }

        memset(C,0,sizeof(C));
        int res=0;
        int f=0;
        for(i=0;i<n;i++)
        {
            f=0;
            for(j=0;j<=i;j++)
            if(E[j]>=S[i])
            {
                C[j]++;
                if(C[j]==x)
                f=1;
            }

            if(f==1)
            {
                ++res;

                for(j=0;j<=i;j++)
                C[j]--;
            }
        }
        cout<<res<<endl;
    }
    return 0;
}
