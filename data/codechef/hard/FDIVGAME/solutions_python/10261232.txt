def grundy(n):
	if n == 0:
		return 0
	k = 1
	while n >= k*12:
		k *= 12
	if n < 2*k:
		return 1
	if n < 4*k:
		return 2
	if n < 6*k:
		return 3
	return 0

t = input()
while t:
	t -= 1
	n = input()
	a = map(int,raw_input().split())
	x = 0
	for i in a:
		x ^= grundy(i)
	if x:
		print "Henry"
	else:
		print "Derek"
