def G(n):
    arr = [0,1,2,2,3,3,0,0,0,0,0,0]
    if n<12:
        return arr[n]
    else:
        return G(n/12)
        
t = int(raw_input())
for i in range(t):
    n = long(raw_input())
    l = map(long,raw_input().split(" "))
    ans = 0
    for j in range(len(l)):
        ans = ans^G(l[j])
    if ans == 0:
        print "Derek"
    else:
        print "Henry"
