#include<iostream>
#include<vector>
#include<string>
#include<algorithm>
using namespace std;
 
int calculate(long long num) {
  if (num == 0)
      return 0;
  else if (num == 1) {
      return 1;
  } else if (num == 2) {
      return 2;
  } else if (num == 3) {
      return 2;
  } else if (num == 4) {
      return 3;
  } else if (num == 5) {
      return 3;
  } else {
      unsigned long long p = 6;
      while (true) {
          if (num >= p && num <= 2*p - 1) {
              return 0;
          }
          if (num >= 2 * p && num <= 4*p - 1) {
              return 1;
          }
          if (num >= 4* p && num <= 8*p - 1) {
              return 2;
          }
          if (num >= 8*p && num <= 12*p - 1) {
              return 3;
          }
          p = p * 12;
      }
  }
}
 
int main() {
    int t;
    cin >> t;
    
    while (t--) {
        int n;
        cin >> n;
        int nim = 0;
        unsigned long long num;
        while (n--) {
            cin >> num;
            nim ^= calculate(num);
        }
        cout << (nim ? "Henry" : "Derek") << "\n";
    }
}
