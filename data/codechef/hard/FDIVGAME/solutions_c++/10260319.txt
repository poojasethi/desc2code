#include <bits/stdc++.h>
using namespace std;

int a[12] = { 0,1,2,2,3,3,0,0,0,0,0,0 };

int g( long long x )
{
	if( x < 12 )
		return a[x];
	return g( x / 12 );
}

int main()
{
	int t;
	cin >> t;
	while( t -- )
	{
		int n, sum = 0;
		long long x;
		cin >> n;
		for( int i = 0 ; i < n ; i ++ )
			cin >> x, sum ^= g( x );
		if( sum )
			cout << "Henry" << endl;
		else
			cout << "Derek" << endl;
	}
	return 0;
}
