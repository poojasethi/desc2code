ll calc_grundy(ll n)
{
    if(n==0)
        return 0;
    else if (n==1)
        return 1;
    else if(n==2 or n==3)
        return 2;
    else if(n==4 or n==5)
        return 3;
    else
        return calc_grundy(n/12);
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n,ans=0,i,x;
        vector<ll>v;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>x;
            v.pb(x);
        }
        for(i=0;i<v.size();i++)
            ans= ans^calc_grundy(v[i]);

        if(ans!=0)
            cout<<"Henry\n";
        else
            cout<<"Derek\n";
    }
}
