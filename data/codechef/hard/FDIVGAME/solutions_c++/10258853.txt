#include<bits/stdc++.h>
using namespace std;
#define ll long long
int d[6]= {2,3,4,5,6};

ll rec(ll a)
{
    while(a>=12)
    {
        a/=12;
    }
    if(a==0) return 0; //G(0) = 0
    if(a>=1 && a<2) return 1; //G(n) = 1 where n = {12^k, 2*12^k -1}
    if(a>=2 && a<4) return 2; //G(n) = 2 where n = {2*12^k, 4*12^k -1}
    if(a>=4 && a<6) return 3; //G(n) = 3 where n = {4*12^k, 6*12^k -1}
    if(a>=6 && a<12) return 0; //G(n) = 0 where n = {6*12^k, 12*12^k -1}
}
int main()
{
    int t,n;
    ll a,nimSum;
    scanf("%d",&t);
    for(int ca=1; ca<=t; ca++)
    {

        scanf("%d",&n);
        nimSum=0LL;
        for(int i=0; i<n; i++)
        {
            scanf("%lld",&a);
            nimSum^=rec(a);
        }
//      printf("Case %d: ",ca);
        if(nimSum) puts("Henry");
        else puts("Derek");
    }
    return 0;
}
