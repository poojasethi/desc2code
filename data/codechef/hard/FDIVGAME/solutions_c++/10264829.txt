#include <bits/stdc++.h>
using namespace std;
typedef long long int li;


li G(li n)
{
	li small[] = {0,1,2,2,3,3,0,0,0,0,0,0};
	if(n < 12) return small[n];
	else return G(n/12);
}



void func()
{
	li n,i,p = 0;
	cin >> n;
	vector<li> arr(n);
	for(i=0;i<n;++i) cin >> arr[i];
	for(i=0;i<n;++i) arr[i] = G(arr[i]);
	for(i=0;i<n;++i) p ^= arr[i];
	if(p==0){
		puts("Derek");
	}
	else{
		puts("Henry");
	}

}



int main()
{
    li test;
    cin >> test;
    while(test--) func();
}
