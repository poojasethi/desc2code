#include <bits/stdc++.h>

using namespace std;
typedef long long int64;

int get_nim(int64 n){
	int64 t=n;
	while(t>=12){
		t=t/12;
	}
	if (t<2){
		return 1;
	}
	if (t<4){
		return 2;
	}
	if (t<6){
		return 3;
	}
	else{
		return 0;
	}
}

int main(){
	int64 inp;
	int64 temp=0;
	int i=0,t,n;
	cin>>t;
	while(i<t){
		cin>>n;
		temp=0;
		for (int j=0;j<n;j++){
			cin>>inp;
			temp=temp^get_nim(inp);
		}
		if (temp==0){
			printf("%s\n","Derek");
		}
		else{
			printf("%s\n","Henry");
		}

		i++;
	}



	return 0;
}
