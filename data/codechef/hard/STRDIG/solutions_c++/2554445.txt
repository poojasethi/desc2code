long long StrToLong(string s){
	long long x=0;
	for (int i=0;i<s.size();i++){
		x=x*10+s[i]-'0';
		if (x>INF)return x;
	}
	return x;
}

string LongToStr(long long x){
	string s="";
	while (x>0){
		s+=char(x%10+'0');
		x/=10;
	}
	string p="";
	for (int i=0;i<s.size();i++)
		p+=s[s.size()-i-1];
	return p;
}

int check(string &s){
	if (s.size()<=10){
		p=s;
		if (p[0]=='0')p='1'+p;
		if (StrToLong(p)<=INF)return 1;
	}
	if (s.size()<=18){
		for (int i=0;i<s.size()-1;i++){
			if (s[i+1]=='0')continue;
			a=s.substr(0,i+1);
			b=s.substr(i+1);
			int j=a.size()-1;
			while ((j>=0)&&(a[j]=='9')){
				a[j]='0';
				--j;
			}
			if (j>=0)a[j]++;
			for (int i=0;(i<=a.size())&&(i<=b.size());i++){
				if ((i==0)||(b.substr(b.size()-i)==a.substr(0,i))){
					p=b+a.substr(i);
					long long x=StrToLong(p);
					if (x>INF)continue;
					if (x-1<1)continue;
					string u=LongToStr(x-1)+LongToStr(x);
					if (x<INF)u+=LongToStr(x+1);
					if (u.find(s)!=string::npos)return 1;
				}
			}
		}
	}
	for (int i=0;i<=10;i++){
		if (i>=s.size())break;
		if (s[i]=='0')continue;
		int tmp=1;
		if (i>tmp)tmp=i;
		for (int j=tmp;j<=10;j++){
			if (i+j-1>=s.size())break;
			long long x=StrToLong(s.substr(i,j));
			if (x>INF)continue;
			bool bad=0;
			if (i!=0){
				if (x-1<1)continue;
				string u=LongToStr(x-1);
				int lu=u.size();
				for (int e=0;e<i;e++)
					if (s[e]!=u[lu-i+e]){
						bad=1;
						break;
					}
				if (bad)continue;
			}
			++x;
			for (int e=i+j;e<s.size();x++){
				if (x>INF){bad=1;break;}
				string v=LongToStr(x);
				for (int o=0;((o<v.size())&&(e<s.size()));o++,e++)
					if (s[e]!=v[o]){bad=1;break;}
				if (bad)break;
			}
			if (bad)continue;
			return 1;
		}
	}
	return 0;
}
