#include <iostream>
#include <vector>

using namespace std;

#define	MOD	1000000007

void mul(vector<long long> &a, const vector<long long> &b)
{
	int n = a.size();
	vector<long long> c(n, 0);
	for (int i = 0; i < n; i++)
		for (int j = 0; i + j < n; j++)
			c[i + j] = (c[i + j] + a[i] * b[j]) % MOD;
	a = c;
}

int main(int argc, char *argv[])
{
	int T;
	cin >> T;
	while (T--) {
		int M, N, X, D, C[32];
		cin >> M >> N >> X >> D;
		for (int i = 0; i <= D; i++) cin >> C[i];
		vector<long long> p(N + 1);
		for (int i = 0; i <= N; i++) {
			p[i] = 0;
			long long t = ((long long) X * i) % MOD, x = 1;
			for (int j = 0; j <= D; j++, x = (x * t) % MOD)
				p[i] = (p[i] + x * C[j] % MOD) % MOD;
		}
		
		vector<long long> a(p.size(), 0);
		a[0] = 1;
		for (int k = M; k; k /= 2) {
			if (k % 2) mul(a, p);
			mul(p, p);
		}
		cout << a[N] << endl;
	}

	return 0;
}
