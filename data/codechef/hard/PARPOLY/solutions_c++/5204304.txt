llu c[15],m,n,x,d,P[MAX];

void calcP()//calc P(num*x)
{
    for(int i=0;i<=n;++i)
    {
        llu pp = i*x;
        if(pp>=MOD)pp%=MOD;
        llu temp = 0,mul=1;
        for(int j=0;j<=d;++j)
        {
            temp = temp + (c[j]*mul);
            if(temp>=MOD) temp%=MOD;
            mul*=pp;
            if(mul>=MOD)mul%=MOD;
        }
        P[i] = temp;
    }
}

void convolute(llu* A,llu* B,llu _size)
{
    llu res[MAX];
    for (int i = 0; i <= _size; ++i)
    {
        res[i] = 0;
        for (int j = 0; j <= i; ++j)
        {
            res[i] = (res[i] + (A[j] * B[i - j]) % MOD) % MOD;
        }
    }
    for(int i = 0; i <= _size; ++i)
        A[i] = res[i];
}

       for(int i=0;i<=d;++i)c[i]=input();
       calcP();
       llu ans[MAX] = {1};
       llu exp = m;
       while(exp)
        {
            if(exp&1)
                convolute(ans,P,n);
            convolute(P,P,n);
            exp>>=1;
        }
       cout<<ans[n]<<endl;
   }
   FlushFASTIO();
   return 0;
}
