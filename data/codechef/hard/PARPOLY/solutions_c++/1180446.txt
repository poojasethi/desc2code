int eval(int i)

int solve(int m,int n)
{
    int &res= dp[m][n];
   // cout<<dp[m][n]<<"\n";
    if(res!=-1) return res;
    if(m==1)
    {
        res= P[n];
        return res;
    }
    ll ans=0LL;
    f(i,0,n)
    {
        ans+= ((ll)solve(m/2,i)*solve(m-m/2,n-i))%mod;
        ans%=mod;
    }
    res= ans;
    return res;
}
