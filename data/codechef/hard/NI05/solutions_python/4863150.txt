long long int pre(int avail[], long long cc)
{
    int j;
    int v=0,ind=-1,i;
    long long int ans=0;
    for(i=0;i<10;i++)
    {
        if(avail[i]!=0){ind=i;v=avail[i];break;}
    }
    if(!v)return cc;
    for(i=0;i<10;i++)
    {
        if(i!=ind && avail[i]!=0)
        {
            int new_arr[10];
            for(j=0;j<10;j++)new_arr[j]=avail[j];
            new_arr[i]--;
            new_arr[ind]--;
            ans = max(ans, pre(new_arr, cc + cost[i][ind]));
//            cout<<i<<" "<<ind<<" "<<cc + cost[i][ind]<<"acost \n";
        }
    }
    if(avail[ind]==2)
    {
                    int new_arr[10];
            for(j=0;j<10;j++)new_arr[j]=avail[j];
        new_arr[ind]=0;
        ans=max(ans, pre(new_arr, cc + cost[ind][ind]));
//        cout<<ind<<" "<<cost[ind][ind]<<"\n";
//        cout<<i<<" "<<ind<<" "<<cc + cost[ind][ind]<<"bcost \n";
    }
    if(avail[ind]==2)
    {
        for(i=0;i<10;i++)
        {
            if(i!=ind && avail[i]==2)
            {
                int new_arr[10];
                for(j=0;j<10;j++)new_arr[j]=avail[j];
                new_arr[ind]=0;
                new_arr[i]=0;
                ans=max(ans,pre(new_arr, cc + 2*cost[i][ind]));
//                cout<<i<<" "<<ind<<" "<<cc + 2*cost[i][ind]<<"ccost \n";
            }
        }
    }
    return ans;
}
