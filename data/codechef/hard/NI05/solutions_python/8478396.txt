void swap(char *a, char *b)
{
	char temp = *a;
	*a = *b;
	*b = temp;
}

void permut(string s,string s1,string s2,int a[10][10])
{
	
	if(s1.size()==1)
	{
		//cout<<(s2+s1)<<endl;
		string temp = s2+s1;
		ll sum = 0;
		for(int i = 0;i<s.size();i++)
		{
			sum+= a[s[i]-'0'][temp[i]-'0'];
		}
		ans = max(ans,sum);
		return;
	}
	for(int i = 0;i<s1.size();i++)
	{
		swap(&s1[0],&s1[i]);
		permut(s,s1.substr(1,s1.size()-1),s2+s1.substr(0,1),a);
		swap(&s1[i],&s1[0]);
	}
}
