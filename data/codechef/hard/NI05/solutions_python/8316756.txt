inline long long int scanint(){
    char c = gc();
    long long int x = 0;
    while(c<'0'||c>'9'){
        c=gc();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=gc();
    }
    return x;
}
