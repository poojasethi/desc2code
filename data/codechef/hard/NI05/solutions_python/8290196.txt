map<vector<int>,int > dp;
int solve(vector<int> x)
{
	if(dp.find(x)!=dp.end())
		return dp[x];
	int i,j;
	int ret=0;
//	for(i=0;i<10;i++)
//		cout<<x[i]<<" ";
//	cout<<endl;
	for(i=0;i<10;i++)
	{
		for(j=0;j<10;j++)
		{
			if(a[i]==0 || a[j]==0 || x[i]==0 || x[j]==0)
				continue;
			if(i==j && x[i]<2)
				continue;
			x[i]--;
			x[j]--;
			ret=max(ret,solve(x)+mat[i][j]);
			x[i]++;
			x[j]++;
		}
	}
	//cout<<ret<<endl;
	dp[x]=ret;
	return ret;
}
