#include <iostream>
#include <stdio.h>
#include<string.h>
using namespace std;
int p[7]={0};
int ans=0;
int count,d[7][7];
void recursive(int j,int sum){
	if(j==count){
		if(ans<sum)
			ans=sum;
	}
	for(int i=0;i<count;i++){
		if(!p[i]){
			p[i]=1;
			sum+=d[j][i];
			recursive(j+1,sum);
			sum=sum-d[j][i];
			p[i]=0;
		}
	}
}
int main() {
	int a[10][10]={{ 55,60,4,25,18,10,12,8,11,50 },{ 60,45,75,23,27,20,24,7,33,12 },{ 4,75,78,32,36,30,36,6,12,65 },{ 25,23,32,15,45,40,48,5,14,23 },{ 18,27,36,45,54,50,60,4,15,12 },{ 10,20,30,40,50,60,72,3,32,34 },{ 12,24,36,48,60,72,84,2,23,34 },{ 8,7,6,5,4,3,2,1,34,123 },{ 11,33,12,14,15,32,23,34,65,48 },{ 50,12,65,23,12,34,34,123,48,71 }};
	int c[10],b,total;
	scanf("%d",&total);
	while(total--){
		count=0;
		for(int i=0;i<10;i++){
			scanf("%d",&b);
		}
		for(int i=0;i<10;i++){
			scanf("%d",&b);
			if(b){
				c[count]=i;
				count++;
			}
		}
	//	printf("%d\n",a[3][4]);
		for(int i=0;i<count;i++){
			for(int j=0;j<count;j++){
				d[i][j]=a[c[i]][c[j]];
	//			cout<<d[i][j]<<" ";
			}
	//		cout<<endl;
		}
		ans=0;
		memset(p,0,sizeof(p));
		recursive(0,0);
		cout<<ans<<endl;
	}
	return 0;
}
