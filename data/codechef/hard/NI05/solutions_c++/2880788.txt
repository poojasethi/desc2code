LL arr[20];
int valid[20];
LL w[10][10] = {55,60,4,25,18,10,12,8,11,50,60,45,75,23,27,20,24,7,33,12,4,75,78,32,36,30,36,6,12,65,25,23,32,15,45,40,48,5,14,23,18,27,36,45,54,50,60,4,15,12,10,20,30,40,50,60,72,3,32,34,12,24,36,48,60,72,84,2,23,34,8,7,6,5,4,3,2,1,34,123,11,33,12,14,15,32,23,34,65,48,50,12,65,23,12,34,34,123,48,71};

map< string, LL > dp;
map< string ,LL >::iterator it;

LL rec(string a){
	it = dp.find(a);
	if(it != dp.end()) return dp[a];
	LL ans = 0;
	LL tmp;
	FOR(i,0,9){
		FOR(j,0,9){
			if(!valid[i] || !valid[j]) continue;
			if(a[i]=='0' || a[j]=='0') continue;
			if(i==j && a[i]!='2') continue;
			a[i]--; a[j]--;
			tmp = w[i][j] + rec(a);
			ans = max(ans,tmp);
			a[i]++; a[j]++;									
		}
	}
	dp[a] =  ans;
	return dp[a];	
}


void input(){
	FOR(i,0,9) cin>>arr[i];
	FOR(i,0,9) cin>>valid[i];
}

int main(){
	int t;
	cin>>t;
	while(t--){
		input();
		string a;
		dp.clear();
		FOR(i,0,9){
			if(arr[i]<2) a.pb(arr[i] + '0');
			else a.pb('2');
		}
		cout<<rec(a)<<endl;		
	}
	return 0;
}
