int solve(int i)
{
	if(i == 10)
		return 0;
	if(ele[i] == 0)
		return solve(i+1);
	if(m.find(ele) != m.end())
		return m[ele];
	int ret = -1000000;
	if(ele[i] > 1)
	{
		//m[ele] = 10000000;
		ele[i] = 0;
		ret = solve(i+1) + v[i*10 + i];
		ele[i] = 2;
	}
	ele[i]--;
	for(int j = i+1; j<10; j++)
	{
		if(ele[j] > 0)
		{
			ele[j]--;
			ret = max(ret, solve(i) + v[i*10 + j]);
			ele[j]++;
		}
	}
	ele[i]++;
	m[ele] = ret;
	return ret;
}
