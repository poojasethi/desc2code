void fun(int pos,int prize)
{
     //cout<<pos<<" "<<pair_count<<" "<<prize<<"\n";
     if(pair_count>7)
     {
                      return;
     }
     
      if(prize>ans)
      ans=prize;
     
     for(int i=pos;i<10;i++)
     {
             if(arr[i]&&count1[i]<=1)
             {
                 count1[i]++;
                 for(int j=pos;j<10;j++)
                 {
                     if(arr[j]&&count1[j]<=1)
                     {
                          count1[j]++; 
                          pair_count++;
                          fun(i,prize+rewards[i][j]);
                          pair_count--;
                          count1[j]--;
                     }
                 }
                 count1[i]--;
             }
     }
}
