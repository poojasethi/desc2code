ll mod;

ll mult(ll x,ll y) {
	if(y==0) return 0;
	ll temp=mult(x,y/2);
	ll ans=0;
	if(y%2==0) ans=temp+temp;
	else ans=temp+temp+x;
	while(ans>=mod) ans-=mod;
	return ans; 
}
