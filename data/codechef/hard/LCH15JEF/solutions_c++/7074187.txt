#include<iostream>
#include<cstring>
#define ll long long
using namespace std;
ll mod;
ll mulmod(ll x,ll y)
{
	if(y==0)
	return 0;
	ll u=mulmod(x,(y/2));
	ll res;
	if(y&1)
	{
		res=u+u+x;
	}
	else
	{
		res=u+u;
	}
	while(res>=mod)
	res-=mod;
	return res;
}
ll powmod(ll a,ll b)
{
	ll res=1;
	while(b)
	{
		if(b&1)
		{
			res=mulmod(res,a);
		}
		a=mulmod(a,a);
		b>>=1;
	}
	return res;
}
char s[10100];
void solve()
{
	cin>>mod;
	cin>>s;
	int l=strlen(s);
	ll ans=1%mod;
	ll a,b;
	for(int i=0;i<l;i++)//if s[i]!='\0' then wrong answer
	{
	//	ll F=0;
		int j=i;
		a=0;
		while(s[j]>='0' && s[j]<='9')
		{
			a=(a*10+s[j]-'0')%mod;
			j++;
		}
		j++;
		j++;
		ll b=1%mod;
		while(s[j]>='0' && s[j]<='9')
		{
			b=powmod(b,10);
			b=mulmod(b,powmod(a,(ll)(s[j]-'0')));
			j++;
		}
		ans=mulmod(ans,b);
		i=j;
	}
	cout<<ans<<"\n";
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	solve();
	return 0;
}
