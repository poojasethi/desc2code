#include<iostream>
#include<cstring>
#define ll long long
using namespace std;
ll mod;
ll mulmod(ll x,ll y)
{
	if(y==0)
	return 0;
	ll u=mulmod(x,(y/2));
	ll res;
	if(y&1)
	{
		res=u+u+x;
	}
	else
	{
		res=u+u;
	}
	while(res>=mod)
	res-=mod;
	return res;
}
char s[10100];
void solve()
{
	cin>>mod;
	cin>>s;
	int l=strlen(s);
	ll ans=1%mod;
	for(int i=0;i<l;i++)
	{
		ll F=0;
		int j=i;
		while(s[j]>='0' && s[j]<='9')
		{
			F=mulmod(F,10);
			F+=s[j]-'0';
			F%=mod;
			j++;
		}
		j++;
		j++;
		ll NF=1%mod;
		while(s[j]>='0' && s[j]<='9')
		{
			ll WNF=1;
			for(int k=1;k<=10;k++)
			WNF=mulmod(NF,WNF);
			NF=WNF;
			for(int k=1;k<=s[j]-'0';k++)
			NF=mulmod(NF,F);
			j++;
		}
		ans=mulmod(ans,NF);
		i=j;
	}
	cout<<ans<<"\n";
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	solve();
	return 0;
}
