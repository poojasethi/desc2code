                          
#include<iostream>
#include<cstdio>
#include<string>
#include<vector>
 
using namespace std;
 
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    string a,b;
    cin>>a>>b;
    vector<vector<int> > lcs(a.size()+1),d(a.size()+1);
    for(int i=0;i<=a.size();i++)
    {
      lcs[i].resize(b.size()+1,0);
      d[i].resize(b.size()+1,1);
    }
    for(int i=1;i<=a.size();i++)
    {
      for(int j=1;j<=b.size();j++)
      {
	if(a[i-1]==b[j-1])
	  lcs[i][j]=lcs[i-1][j-1]+1;
	else
	{
	  if(lcs[i][j-1]>lcs[i-1][j])
	    lcs[i][j]=lcs[i][j-1];
	  else
	    lcs[i][j]=lcs[i-1][j];
	}
      }
    }
    for(int i=1;i<=a.size();i++)
    {
      for(int j=1;j<=b.size();j++)
      {
	d[i][j]=0;
	if(a[i-1]==b[j-1])
	  d[i][j]=d[i-1][j-1];
	else
	{
	  if(lcs[i-1][j]==lcs[i][j])
	    d[i][j]+=d[i-1][j];
	  if(lcs[i][j-1]==lcs[i][j])
	    d[i][j]+=d[i][j-1];
      if (lcs[i - 1][j - 1] == lcs[i][j])
         d[i][j] -= d[i - 1][j - 1];
  
	}
	d[i][j]=(d[i][j]+23102009)%23102009;
      }
    }
    /*for(int i=0;i<=a.size();i++)
    {
      for(int j=0;j<=b.size();j++)
      {
	printf("%d ",d[i][j]);
      }
      printf("\n");
    }*/
    printf("%d %d\n",lcs[a.size()][b.size()],d[a.size()][b.size()]);
  }
  return 0;
}
