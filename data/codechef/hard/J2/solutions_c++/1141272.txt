#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef int LL;

char str1[1002],str2[1002];
LL dp[1002][1002],lcs[1002][1002],mod=/*1000007*/23102009;
LL MX(LL a,LL b){return (a>b)?a:b;}
int main()
{
	LL t,cas=0,len1,len2,i,j;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%s%s",str1+1,str2+1);
		len1=strlen(str1+1);
		len2=strlen(str2+1);
		for(i=0;i<=len1;i++){lcs[i][0]=0;dp[i][0]=0;}
		for(i=1;i<=len2;i++){lcs[0][i]=0;dp[0][i]=0;}
		for(i=1;i<=len1;i++)
		{
			for(j=1;j<=len2;j++)
			{
				if(str1[i]==str2[j]){lcs[i][j]=1+lcs[i-1][j-1];continue;}
				lcs[i][j]=MX(lcs[i-1][j],lcs[i][j-1]);
			}
		}
		for(i=1;i<=len1;i++)
		{
			for(j=1;j<=len2;j++)
			{
				if(str1[i]==str2[j])
				{
					if(dp[i-1][j-1]==0){dp[i][j]=1;}
					else{dp[i][j]=dp[i-1][j-1];}
					continue;
				}
				if(lcs[i-1][j]==lcs[i][j-1])
				{
					if(lcs[i-1][j]==lcs[i-1][j-1]){dp[i][j]=(dp[i-1][j]+dp[i][j-1]-dp[i-1][j-1])%mod;}
					else{dp[i][j]=(dp[i-1][j]+dp[i][j-1])%mod;}
				}
				else if(lcs[i-1][j]>lcs[i][j-1]){dp[i][j]=dp[i-1][j];}
				else{dp[i][j]=dp[i][j-1];}
			}
		}
		dp[len1][len2]+=mod;
		dp[len1][len2]%=mod;
		if(!dp[len1][len2]){dp[len1][len2]=1;}
	//	printf("Case %d: %d\n",++cas,dp[len1][len2]);
		printf("%d %d\n",lcs[len1][len2],dp[len1][len2]);
	}
	return 0;
}
