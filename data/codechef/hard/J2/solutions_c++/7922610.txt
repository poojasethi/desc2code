    #include<string>
    #include<iostream>
    using namespace std;
     
    #define mod 23102009
    int dp1[1002][1002],dp2[1002][1002];
    void func(string a,string b)
    {
    for(int i=0;i<=a.size();i++)
    for(int j=0;j<=b.size();j++)
    {
    if(i==0||j==0)
    {
    dp1[i][j]=0;
    dp2[i][j]=1;
    }
    else if(a[i-1]==b[j-1])
    {
    dp2[i][j]=0;
    dp1[i][j]=dp1[i-1][j-1]+1;
    dp2[i][j]=dp2[i-1][j-1];
    }
    else
    {
    dp2[i][j]=0;
    if(dp1[i-1][j]>=dp1[i][j-1])
    {
    dp1[i][j]=dp1[i-1][j];
    dp2[i][j]=dp2[i][j]+dp2[i-1][j];
    dp2[i][j]=dp2[i][j]%mod;
    }
    if(dp1[i-1][j]<=dp1[i][j-1])
    {
    dp1[i][j]=dp1[i][j-1];
    dp2[i][j]=dp2[i][j]+dp2[i][j-1];
    dp2[i][j]=dp2[i][j]%mod;
    }
    if(dp1[i][j]==dp1[i-1][j-1])
    {
    dp2[i][j]=dp2[i][j]-dp2[i-1][j-1];
    if(dp2[i][j]<0)
    dp2[i][j]=dp2[i][j]+mod;
    dp2[i][j]=dp2[i][j]%mod;
    }
    }
    }
    cout<<dp1[a.size()][b.size()]<<" "<<dp2[a.size()][b.size()]<<endl;
    }
     
    int main()
    {
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
    string a,b;
    cin>>a>>b;
    func(a,b);
    }
    return 0;
    }
