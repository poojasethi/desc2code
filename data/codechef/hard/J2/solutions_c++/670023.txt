#include<iostream>
#include<string>

using namespace std;

int lcs[1001][1001];
long int d[1001][1001];
string x,y;

long int distinct(int n,int m)
{
  if(n<=0||m<=0)
    return 1;
  if(d[n][m]!=-1)
    return d[n][m];
  d[n][m]=0;
  if(x[n-1]==y[m-1])
    d[n][m]=distinct(n-1,m-1);
  else
  {
    if(lcs[n][m]==lcs[n-1][m])
      d[n][m]+=distinct(n-1,m);
    if(lcs[n][m]==lcs[n][m-1])
      d[n][m]+=distinct(n,m-1);
    if(lcs[n][m]==lcs[n-1][m-1])
      d[n][m]-=distinct(n-1,m-1);
  }
  return (d[n][m]+23102009)%23102009;
}

int main()
{
  int t,i,j;
  cin>>t;
  while(t)
  {
    cin>>x>>y;
    for(i=0;i<y.size();i++)
    {
      lcs[i][0]=0;
    }
    for(i=0;i<x.size();i++)
    {
      lcs[0][i]=0;
    }
    for(i=1;i<=x.size();i++)
    {
      for(j=1;j<=y.size();j++)
      {
	if(x[i-1]==y[j-1])
	  lcs[i][j]=lcs[i-1][j-1]+1;
	else if(lcs[i-1][j]>lcs[i][j-1])
	  lcs[i][j]=lcs[i-1][j];
	else
	  lcs[i][j]=lcs[i][j-1];
      }
    }
    
    for(i=0;i<=x.size();i++)
      for(j=0;j<=y.size();j++)
	d[i][j]=-1;
    if(lcs[x.size()][y.size()])
      cout<<"\n"<<lcs[x.size()][y.size()]<<" "<<distinct(x.size(),y.size());
    else
      cout<<"\n0 0";
    t--;
  }
  return 0;
}
