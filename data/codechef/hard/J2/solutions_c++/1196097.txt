#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
int t,l1,l2;
int a[1001][1001][2];
string s1,s2;
int rem=23102009;
scanf("%d",&t);
for(int i=0;i<t;i++)
{
cin>>s1>>s2;
l1=s1.length();
l2=s2.length();
for(int i=0;i<l1;i++)
{
a[i][0][0]=0;
a[i][0][1]=1;
}
for(int j=0;j<l2;j++)
{
a[0][j][0]=0;
a[0][j][1]=1;
}
for(int i=1;i<=l1;i++)
{
for(int j=1;j<=l2;j++)
{
if(s1[i-1]==s2[j-1])
{
a[i][j][0]=1+a[i-1][j-1][0];
a[i][j][1]=a[i-1][j-1][1];
}
else
{
if(a[i][j-1][0]>=a[i-1][j][0])
a[i][j][0]=a[i][j-1][0];
else
a[i][j][0]=a[i-1][j][0];

a[i][j][1]=0;
if(a[i][j][0]==a[i-1][j][0])
a[i][j][1]=a[i-1][j][1];
if(a[i][j][0]==a[i][j-1][0])
a[i][j][1]=(a[i][j][1]+a[i][j-1][1])%rem;
if(a[i][j][0]==a[i-1][j-1][0])
a[i][j][1]-=a[i-1][j-1][1];
if(a[i][j][1]<0)
a[i][j][1]+=rem;
}
}
}
printf("%d %d\n",a[l1][l2][0],a[l1][l2][1]);
}
return 0;
}
