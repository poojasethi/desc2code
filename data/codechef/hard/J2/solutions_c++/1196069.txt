#include<iostream>
#include<vector>
using namespace std;

int main()
{
int x[26],y[26];

int b[1001][1001][2];

int t,l1,l2,cnt;
int rem=23102009;
string s1,s2;
cin>>t;
for(int i=0;i<t;i++)
{
cin>>s1>>s2;
l1=s1.length();
l2=s2.length();
for(int i=0;i<=l1;i++)
{
b[i][0][0]=0;
b[i][0][1]=1;
}
for(int i=0;i<=l2;i++)
{
b[0][i][0]=0;
b[0][i][1]=1;
}
for(int i=1;i<=l1;i++)
{
for(int j=1;j<=l2;j++)
{
if(s1[i-1]==s2[j-1])
{
b[i][j][0]=1+b[i-1][j-1][0];
b[i][j][1]=b[i-1][j-1][1];
}
else
{
if(b[i-1][j][0]>=b[i][j-1][0])
b[i][j][0]=b[i-1][j][0];
else
b[i][j][0]=b[i][j-1][0];

b[i][j][1]=0;
if(b[i][j][0]==b[i-1][j][0])
b[i][j][1]=b[i-1][j][1];
if(b[i][j][0]==b[i][j-1][0])
b[i][j][1]=(b[i][j][1]+b[i][j-1][1])%rem;
if(b[i][j][0]==b[i-1][j-1][0])
b[i][j][1]=b[i][j][1]-b[i-1][j-1][1];
if(b[i][j][1]<0)
{
b[i][j][1]+=rem;
}

}
}
}

if(b[l1][l2][0]==1)
{
for(int i=0;i<26;i++)
{
x[i]=0;
y[i]=0;
}
for(int i=0;i<l1;i++)
{
x[s1[i]-'a']++;
}
for(int i=0;i<l2;i++)
{
y[s2[i]-'a']++;
}
cnt=0;
for(int i=0;i<26;i++)
{
if(x[i]>0 && y[i]>0)
cnt++;
}
cout<<"1 "<<cnt<<endl;
}
else if(b[l1][l2][0]==0)
cout<<"0 1"<<endl;
else
cout<<b[l1][l2][0]<<" "<<b[l1][l2][1]<<endl;
}
return 0;
}
