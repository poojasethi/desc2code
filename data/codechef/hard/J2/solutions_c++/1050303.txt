#include<iostream>
#include<unistd.h>
#include<cstdlib>
#include<cstring>
#include<cstdio>
#include<cmath>
#include<string.h>
#include<algorithm>
#include<map>;
#define CONS 23102009
using namespace std;
char S[1001];
char T[1001];
int mat[1001][1001];
int d[1001][1001];

long int distinct(int i,int j)
{
    if(i<=0||j<=0)
    return 1;
    if(d[i][j]!=-1)
    return d[i][j];
    d[i][j]=0;
    if(S[i-1]==T[j-1])
    d[i][j]=distinct(i-1,j-1);
    else//else if(c[i][j]!=0)
    {
    if(mat[i][j]==mat[i-1][j])
        d[i][j]+=distinct(i-1,j);
    if(mat[i][j]==mat[i][j-1])
        d[i][j]+=distinct(i,j-1);
    if(mat[i][j]==mat[i-1][j-1])
        d[i][j]-=distinct(i-1,j-1);
    }
    return (d[i][j]+CONS)%CONS;
}

int main()
{
   // freopen("input.txt","r",stdin);
   // freopen("output.txt","w",stdout);
    int t,temp;
    int lenS,lenT;
    scanf("%d",&t);
    getchar();
    while(t)
    {
        scanf("%s %s",&S,&T);
        lenS=strlen(S);
        lenT=strlen(T);
        for(int i=0;i<=lenS;i++)
        {
            mat[i][0]=0;
            for(int j=0;j<=lenT;j++)
            {
                d[i][j]=-1;
            }
        }
        for(int i=0;i<=lenT;i++)
        {
            mat[0][i]=0;
        }
        for(int i=0;i<lenS;i++)
        {
            for(int j=0;j<lenT;j++)
            {
                if(S[i]==T[j])
                {
                    //const char* temp=S[i].c_str();
                    temp=mat[i][j];
                    mat[i+1][j+1]=temp+1;
                }
                else if(mat[i+1][j]>mat[i][j+1])
                {
                    mat[i+1][j+1]=mat[i+1][j];
                }
                else
                {
                    mat[i+1][j+1]=mat[i][j+1];
                }
            }
        }

        printf("%d %d\n",mat[lenS][lenT],distinct(lenS,lenT));
        //cout<<lenS<<" "<<lenT;

        t--;
    }

    return 0;
}
