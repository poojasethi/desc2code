#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#include<vector>

#define MOD 23102009

using namespace std;

    //string s1,s2;


   // int lcs[1002][1002];
   // int di[1002][1002];
    char s1[1002],s2[1002];


int main()
{   int cases;

    scanf("%d",&cases);
    while(cases--)
    {
        scanf("%s %s",&s1,&s2);

      //  int l1=s1.length(); int l2=s2.length();

       int l1=strlen(s1);  int l2=strlen(s2);

        //memset(lcs,0,sizeof(lcs));
        //memset(di,2,sizeof(di));

        vector< vector < int > >lcs(l1+1,vector< int >(l2+1,0));
        vector< vector < int > >di(l1+1,vector< int >(l2+1,1));


       

        for(int i=1;i<=l1;i++)
        {
            for(int j=1;j<=l2;j++)
            {
                if(s1[i-1]==s2[j-1])
                {
                    lcs[i][j]=lcs[i-1][j-1]+1;
                    di[i][j]=di[i-1][j-1];
                }
                else
                {
                    lcs[i][j]=max(lcs[i-1][j],lcs[i][j-1]);

                    di[i][j]=0; //cout<<di[i][j]<<endl;

                    if(lcs[i][j]==lcs[i-1][j])
                        di[i][j]=(di[i][j]+di[i-1][j])%MOD;

                    if(lcs[i][j]==lcs[i][j-1])
                        di[i][j]=(di[i][j]+di[i][j-1])%MOD;

                    if(lcs[i][j]==lcs[i-1][j-1])
                    {
                        di[i][j]-=di[i-1][j-1];
                        di[i][j]=(di[i][j]+ MOD )% MOD;
                    }

                }
            }
        }


          

        printf("%d %d\n",lcs[l1][l2],di[l1][l2]);
    }

    return 0;
}
