PII dp[1010][1010] ;
bool isV[1010][1010] ;
int preA[1010][30] ;
int preB[1010][30] ;

char A[1010] ;
char B[1010] ;

int lA,lB,M=23102009 ;

PII cal( int i, int j )
{
    //printf("%d %d\n",i,j) ;
    if(i==lA || j==lB)      return MP(0,1) ;


    if(isV[i][j])      return dp[i][j] ;

    PII ret = MP(0,1),tmp ;
    int k ;
    if( A[i]==B[j] )
    {
        ret = cal( i+1,j+1 ) ;
        ret.ff ++ ;
    }
    else
    {
        rep(k,0,26)
        {
            if(preA[i][k]==lA || preB[j][k]==lB)        continue ;

            tmp = cal( preA[i][k],preB[j][k] ) ;
            //tmp.ff ++ ;
            //puts("sdsd") ;

            if(ret.ff<tmp.ff)         ret = tmp ;
            else if(ret.ff==tmp.ff)   ret.ss = ( ret.ss + tmp.ss ) % M ;
        }
    }

    isV[i][j] = 1 ;
    //printf("%d %d\n",ret.ff,ret.ss) ;
    return dp[i][j] = ret ;
}


void init()
{
    int i,j ;

    rep(j,0,26)     preA[lA][j] = lA ;

    rem(i,lA-1,-1)
        rep(j,0,26)
        {
            if( A[i]==j+'a' )     preA[i][j] = i ;
            else                  preA[i][j] = preA[i+1][j] ;
        }

    rep(j,0,26)     preB[lB][j] = lB ;

    rem(i,lB-1,-1)
        rep(j,0,26)
        {
            if( B[i]==j+'a' )     preB[i][j] = i ;
            else                  preB[i][j] = preB[i+1][j] ;
        }
}
