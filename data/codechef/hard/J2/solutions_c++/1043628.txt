#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int mod = 23102009;

int dp[1010][1010],ans[1010][1010];
int main()
{
	int n,m,t;
	scanf("%d",&t);
	char s[1010],str[1010];
	while(t --) {
		scanf("%s%s",s+1,str+1);
		n = strlen(s + 1);
		m = strlen(str + 1);
		for(int i = 0; i <= n; i ++)
			for(int j = 0; j <= m; j ++)
				dp[i][j] = ans[i][j] = 0;
		for(int i = 0; i <= m; i ++) ans[0][i] = 1;
		for(int i = 0; i <= n; i ++) ans[i][0] = 1;
		for(int i = 1; i <= n;i ++)
			for(int j = 1; j <= m; j ++)
				if(s[i] != str[j]) {
					dp[i][j] = max(dp[i - 1][j],dp[i][j - 1]);
					if(dp[i][j] == dp[i-1][j]) ans[i][j] += ans[i - 1][j];
					if(dp[i][j] == dp[i][j - 1]) ans[i][j] += ans[i][j - 1];
					if(dp[i - 1][j - 1] == dp[i][j]) ans[i][j] -= ans[i - 1][j - 1];
					ans[i][j] = (ans[i][j] % mod + mod ) % mod;
				}
				else {
					dp[i][j] = dp[i - 1][j - 1] + 1;
					ans[i][j] += ans[i - 1][j - 1];
					ans[i][j] = (ans[i][j] % mod + mod ) % mod;
				}
		if(dp[n][m] == 0) ans[n][m] = 0;
		printf("%d %d\n",dp[n][m],ans[n][m]);
	}
	return 0;
}
