

#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

char s1[2000], s2[2000];
char s3[2000], s4[2000];
int len1,len2;
int num;
int LCS[2000][2000];
long long D[2000][2000];


int main() {
    cin >> num;
    int k;
    k = 1;
    while (k <= num) {
        scanf("%s",&s1);
        scanf("%s",&s2);
        len1 = strlen(s1);
        len2 = strlen(s2);
        int i;
        for (i = 1; i <= len1; i++) {
            s3[i] = s1[i - 1];

        }
        for (i = 1; i <= len2; i++) {
            s4[i] = s2[i - 1];
        }
        int j;
        
        for (i = 0; i <= len1; i++)
            LCS[i][0] = 0;
        for (j = 0; j <= len2; j++)
            LCS[0][j] = 0;
        for (i = 1; i <= len1; i++) {
            for (j = 1; j <= len2; j++) {
                if (s3[i] == s4[j])
                    LCS[i][j] = LCS[i - 1][j - 1] + 1;
                else {
                    if (LCS[i - 1][j] > LCS[i][j - 1])
                        LCS[i][j] = LCS[i - 1][j];
                    else
                        LCS[i][j] = LCS[i][j - 1];
                }
            }

        }
        
        for (j = 0; j <= len2; j++) {
            for (i = 0; i <= len1; i++) {
                if (i == 0 || j == 0)
                    D[i][j] = 1;
                else {
                    D[i][j] = 0;
                    if (s3[i] == s4[j])
                        D[i][j] = D[i - 1][j - 1];
                    else {
                        if (LCS[i - 1][j] == LCS[i][j])
                            D[i][j] += D[i - 1][j];
                        if (LCS[i][j - 1] == LCS[i][j])
                            D[i][j] += D[i][j - 1];
                        if (LCS[i - 1][j - 1] == LCS[i][j])
                            D[i][j] -= D[i - 1][j - 1];
                    }
                    D[i][j] = (D[i][j]+23102009)%23102009;
                }
            }
        }
        
            
        if(LCS[len1][len2] == 0)
            D[len1][len2] = 0;
        printf("%d %lld\n", LCS[len1][len2], D[len1][len2] % 23102009);
        k++;
    }
    return 0;
}
