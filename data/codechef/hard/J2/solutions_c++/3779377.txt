#include<iostream>
#include<stdio.h>
#include<string.h>
#include<cmath>
#define MOD 23102009
typedef long long ll;
using namespace std;

ll lcs[1005][1005], dp[1005][1005];

int main()
{
    ll t;
    scanf("%lld",&t);

    while(t--)
    {
        char s1[1005],s2[1005];
        scanf("%s %s",s1,s2);
        ll n = strlen(s1), m = strlen(s2);
        
        for(int i=0;i<=n;i++)
        {
            for(int j=0;j<=m;j++)
            {
                lcs[i][j] = 0;
                dp[i][j] = 0;
            }
        }
        
        
        for(ll i=0;i<=n;i++)
        {
            lcs[i][0] = 0;
            dp[i][0] = 1;
        }
        for(ll j=0;j<=m;j++)
        {
            lcs[0][j] = 0;
            dp[0][j] = 1;
        }
        
        for(ll i=1;i<=n;i++)
        {
            for(ll j=1;j<=m;j++)
            {
                if(s1[i-1]==s2[j-1])
                {
                    lcs[i][j] = lcs[i-1][j-1] + 1;
                    dp[i][j] = dp[i-1][j-1];
                }
                else
                {
                    lcs[i][j] = max(lcs[i-1][j],lcs[i][j-1]);

                    if(lcs[i][j]==lcs[i][j-1])
                        dp[i][j] = dp[i][j-1];

                    if(lcs[i][j]==lcs[i-1][j])
                    {
                        dp[i][j] += dp[i-1][j];
                    }
                    if(lcs[i][j]==lcs[i-1][j-1])
                        dp[i][j] -= dp[i-1][j-1];
                    
                    if(dp[i][j]>=MOD)
                        dp[i][j] %= MOD;
                    else if(dp[i][j]<0)
                        dp[i][j] = (dp[i][j]%MOD + MOD)%MOD;
                }
            }
        }
        printf("%lld %lld\n",lcs[n][m],dp[n][m]);
    }

return 0;
}
