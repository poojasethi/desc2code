#include<iostream>
#include <cstdio> 
#include <cstring> 
#include <cstdlib> 
#include <cctype> 
#include <cmath> 
#include <iostream> 
#include <algorithm> 
#include <vector> 
#include <string> 
#include <sstream>
using namespace std;
int num = 23102009;
int c[1000+1][1+1000],d[1000+1][1+1000];
void LCS(string a,string b)
{
    for(int i=0;i<=a.length()+1;i++)
    {c[i][0]=0;d[i][0]=1;}
    for(int i=0;i<=b.length()+1;i++)
    {c[0][i]=0;d[0][i]=1;}
    for(int i=1;i<=a.length();i++)
    for(int j=1;j<=b.length();j++)
    {
            if(a[i-1]==b[j-1])
            {
                              c[i][j]=c[i-1][j-1]+1;
                              d[i][j]=d[i-1][j-1];
            }
            else
            {
                c[i][j]=c[i][j-1];
                d[i][j]=d[i][j-1];
                if(c[i][j]<c[i-1][j])
                {
                                     c[i][j]=c[i-1][j];
                                     d[i][j]=d[i-1][j];
                }
                else if(c[i][j]==c[i-1][j])
                {
                     d[i][j]+=d[i-1][j];
                     d[i][j]%=num;
                }
                if(c[i][j]==c[i-1][j-1])
                d[i][j]=(d[i][j]-d[i-1][j-1]+num)%num;
            }
    }
}
int main()
{
    int test;
    cin>>test;
    while(test--)
    {
                 string a,b;
                 cin>>a>>b;
                 LCS(a,b);
                 cout<<c[a.length()][b.length()]<<" "<<d[a.length()][b.length()]<<"\n";
    }
    return 0;
}
