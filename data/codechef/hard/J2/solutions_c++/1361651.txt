#include <iostream>
#include <string>
#include <algorithm>
using namespace std;
 
const int Maxn = 1001, num = 23102009;
 
int main()
{
    int t, san[2][Maxn], ilu[2][Maxn];
    string a, b;
    
    cin >> t;
    
    while (t--) {
          
          cin >> a >> b;
          
          fill(san[0], san[0]+Maxn, 0);
          fill(ilu[0], ilu[0]+Maxn, 1);
		 int p = a.length();
		 int q = b.length();
           
          for (int i = 1; i <= p; i++) {
              
              san[1][0] = 0; ilu[1][0] = 1;
              
              for (int j = 1; j <= q; j++) {
                  
                  if (a[i-1] == b[j-1]) {
                            san[1][j] = san[0][j-1] + 1;
                            ilu[1][j] = ilu[0][j-1];
                  } else {
                         
                         san[1][j] = san[0][j];
                         ilu[1][j] = ilu[0][j];
                         
                         if (san[1][j-1] > san[1][j]) {
                                          san[1][j] = san[1][j-1];
                                          ilu[1][j] = ilu[1][j-1];
                         } else if (san[1][j-1] == san[1][j])
                                  ilu[1][j] = (ilu[1][j] + ilu[1][j-1]) % num;
                                  
                        if (san[0][j-1] == san[1][j]) 
                           ilu[1][j] = (ilu[1][j] - ilu[0][j-1] + num) % num;
                           
                 }
                 
             }
             
             for (int j = 1; j <= q; j++) {
                 san[0][j] = san[1][j];
                 ilu[0][j] = ilu[1][j];
             }
             
          }
          
          cout << san[1][q] << " " << ilu[1][q] << endl;
          
    }
    
    return 0;
}
