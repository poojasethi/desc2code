using namespace std;

#define SZ 1050
#define mod 23102009



char s1[SZ],s2[SZ];
int dp[SZ][SZ];
LL memo[SZ][SZ];

int LCS(int i,int j)
{
    if(i<0 || j<0)return 0;

    int &ret=dp[i][j];

    if(ret!=-1)return ret;

    if(s1[i]==s2[j])ret=LCS(i-1,j-1)+1;

    else ret=max(LCS(i,j-1),LCS(i-1,j));

    return ret;
}

LL solve(int i,int j)
{
    if(i<0 || j<0)return 1;

    LL &ret=memo[i][j];
    if(ret!=-1)return ret;

    ret=0;
    if(s1[i]==s2[j])
    {
        ret=solve(i-1,j-1)%mod;
    }
    else
    {
        if(LCS(i,j)==LCS(i-1,j))
        {
            ret=(ret+solve(i-1,j))%mod;
        }
        if(LCS(i,j)==LCS(i,j-1))
        {
            ret=(ret+solve(i,j-1))%mod;
        }
        if(LCS(i,j)==LCS(i-1,j-1))
        {
            ret=(ret-solve(i-1,j-1)+mod)%mod;
        }
    }
    return ret%mod;
}

        int ret = LCS(strlen(s1)-1,strlen(s2)-1);
        memset(memo,-1,sizeof memo);
        printf("%d %lld\n",ret , solve(strlen(s1)-1,strlen(s2)-1));
    }
    return 0;
}
