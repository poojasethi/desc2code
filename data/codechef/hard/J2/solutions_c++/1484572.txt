int mymod(int x)
{
if(x<0)return MOD+x;
if(x<MOD)return x;
return x%MOD;
}
void fillT()
{
    int i,j,T[1001][1001],D[1001][1001];
    for(i=0;i<=la;++i)
      T[i][0]=0; 
    for(i=0;i<=lb;++i)
      T[0][i]=0;
    for(i=1;i<=la;++i)
      {
      for(j=1;j<=lb;++j)
        {
        if(a[i-1]==b[j-1])
        T[i][j]=T[i-1][j-1]+1;
        else
        T[i][j]=max(T[i-1][j],T[i][j-1]);                
        }                
      } 
    ans1=T[la][lb];
    ans2=0;
    for(i=0;i<=la;++i)
    {
      for(j=0;j<=lb;++j)
        {
        if(i==0||j==0)
        D[i][j]=1;
        else
        {
        if(a[i-1]==b[j-1])
          {
          D[i][j]=D[i-1][j-1];
          }
        else
          {
          D[i][j]=0;
          if(T[i][j]==T[i-1][j])
          D[i][j]=mymod(D[i][j]+D[i-1][j]);
          if(T[i][j]==T[i][j-1])
          D[i][j]=mymod(D[i][j]+D[i][j-1]);
          if(T[i][j]==T[i-1][j-1])
          D[i][j]=mymod(D[i][j]-D[i-1][j-1]);
          } 
          }            
        }                  
    } 
    ans2=D[la][lb];  
}
void solve()

	fillT();
    printf("%d %d\n",ans1,ans2);	
}
