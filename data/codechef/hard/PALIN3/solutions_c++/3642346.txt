char s[N],t[N<<1];int dp[N<<1],c[N];
int f[N][3];
void add(int x,int d)
{
	while(x<N)f[x][d]++,x+=x&-x;
}
int query(int x,int d)
{
	int ans=0;
	while(x>0)ans+=f[x][d],x-=x&-x;
	return ans;
}
int query(int x,int y,int d)
{
	if(x>y)return 0;
	int ans=query(y,d)-query(x-1,d);
	//DBG(ans)
	return ans;
}
int main()
{
	int i,j,k,m,T,n;
	while(~scanf("%s",s))
	{
		n=strlen(s);
		ll ans=0;
		memset(f,0,sizeof(f));
		for(i=0;i<n;i++)
		{
			c[i+1]=c[i]+(s[i]-'0');
			//if(s[i]=='0')ans++;
		}
		for(i=1;i<=n;i=j)
		{
			j=i;
			while(j<=n&&c[j]==c[i])j++;
			if(c[i])add(i,c[i]%3);
		}
		m=0;t[m++]='$';
		t[m++]='#';
		for(i=0;i<n;i++)
		t[m++]=s[i],t[m++]='#';
		t[m++]='@';
		m--;
		int id=0,r=0;
		for(i=2;i<m-1;i++)
		{
			if(i<r)j=min(dp[id+id-i],r-i);
			else j=1;
			while(t[i-j]==t[i+j])j++;
			dp[i]=j;
			if(i+j>r){id=i,r=i+j;}
			//cerr<<i<<" "<<dp[i]<<endl;
		}
		for(i=2;i<m-1;i++)
		if(dp[i]>=2)
		{
			int x=i-(dp[i]-2),y=i+dp[i]-2;
			x/=2,y/=2;
			int z=(x+y)/2;
			//cerr<<x<<" "<<c[z]%3<<" "<<y<<endl;
			if(t[i]=='#')
			ans+=query(z+1,y,c[z]%3);
			else 
			{
				j=t[i]-'0';k=j;
				j=(3-j)%3;
				if(j%2==0)j/=2;
				else j=(j+3)/2;
				j=(j+c[z])%3;
				ans+=query(z+1,y,j);
				if(k%3==0)ans++;
			}
		}
		cout<<ans<<endl;
	}
	return 0;
}
