#define N 1000005
using namespace std;
template<class T> inline void mini(T &a,T b){if(b<a) a=b;}

int T[3][N];
int s[N];
ll manacher3(string &cad) {
	int n = cad.size();
	s[0] = 0; f(i,0,n) s[i+1] = s[i] + (cad[i]-'0');
	f(t,0,3) {
		T[t][0] = 0;
		f(i,0,n) {
			T[t][i+1] = T[t][i] + (cad[i]!='0' && s[i+1]%3 == t ? 1 : 0);
		}
	}
	ll res = 0;

	vector<int> d1 (n);
	int l=0, r=-1;
	f(i,0,n){
		int k = (i>r ? 0 : min (d1[l+r-i], r-i)) + 1;
		while (i+k < n && i-k >= 0 && cad[i+k] == cad[i-k])  ++k;
		int t = (s[i] + (cad[i] - '0')*2) % 3;
		int kg = T[t][i+k] - T[t][i];
		res += kg;

		d1[i] = --k;
		if (i+k > r) l = i-k,  r = i+k;
	}

	vector<int> d2 (n);
	l=0, r=-1;
	for (int i=0; i<n; ++i) {
		int k = (i>r ? 0 : min (d2[l+r-i+1], r-i+1)) + 1;
		while (i+k-1 < n && i-k >= 0 && cad[i+k-1] == cad[i-k])  ++k;
		int t = s[i] % 3;
		int kg = T[t][i+k-1] - T[t][i];
		res += kg;

		d2[i] = --k;
		if (i+k-1 > r) l = i-k,  r = i+k-1;
	}

	return res;
}


int main(){
	string s; 
	cin >> s; 
	ll res = manacher3(s);
	f(i,0,s.size()) if (s[i] == '0') res++;
	cout << res << endl;
}
