#define MOD 1000000007

using namespace std;

long long dp[1<<10][1<<10];

long long modpow(long long x, long long n)
{
	long long P = 1;
			
	while(n)
	{
		if(n & 1) P = (P * x) % MOD;
		
		n >>= 1;
		x = (x * x) % MOD;
	}
	
	return P;
}

int main ()
{
	dp[0][0] = 1;
	for(int i=1; i<=1000; i++)
		for(int j=1; j<=i; j++)
			dp[i][j] = (j*dp[i-1][j] + dp[i-1][j-1]) % MOD;
	
	int nCasos;
	scanf("%d", &nCasos);
	
	for(int caso=0; caso<nCasos; caso++)
	{
		int N, M, K;
		scanf("%d %d %d", &N, &M, &K);
		
		long long x = 0, tmp = M;
		
		for(int i=1; i<=min(N, min(M, K)); i++)
		{
			x = (x + dp[N][i] * tmp) % MOD;
			tmp = (tmp * (M-i)) % MOD;
		}
		
		printf("%d\n", (int)x);
	}
	
	return 0;
}
