ll S[1001][1001];
ll mod = 1000000007;

int main(){
   int t; cin >> t;
   clr(S, 0);
   f(i, 0, 1001){
      S[i][0] = i == 0 ? 1 : 0;
      f(j, 1, i + 1) S[i][j] = (j * S[i - 1][j] + S[i - 1][j - 1]) % mod;
   }
   
   while(t--){
      int n, m, k; cin >> n >> m >> k;
      ll res = 0;
      k = min(k, n);
      ll c = m % mod;
      f(i, 1, k + 1){
         res = (res + c * S[n][i]) % mod;
         c = c * (m - i) % mod;
      }
      cout << res << endl;
   }
}
