#include<cstdio>
#define M 1000000007
using namespace std;

long long int dp[1003][1003];

int main()
{
	int t,n=1000,m,k=1000;
	dp[0][0] = 1;
	for(int i=1; i<=n; i++)
		for(int j=1; j<=k; j++)
			dp[i][j] = (dp[i-1][j-1] + j*dp[i-1][j])%M;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d",&n,&m,&k);
		if(k > m) { k = m; }
		if(k > n) { k = n; }
		long long int sum = 0,temp = 1;
		for(int i=1; i<=k; i++)
		{
			temp = (temp*(m-i+1))%M;
			sum = (sum + (temp*dp[n][i])%M)%M;
		}
		printf("%lld\n",sum);
	}
	return 0;
}
