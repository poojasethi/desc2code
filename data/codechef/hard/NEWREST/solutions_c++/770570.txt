#include <cstdio>
using namespace std;
typedef long long i64;
int dp[1001][1001];
const int rem=1000000007;
int T,N,M,K;

int main()
{
    dp[0][0]=1;
    for(int i=1;i<=1000;i++)
        for(int j=1;j<=i;j++)
            dp[i][j]=((i64)dp[i-1][j]*j+(i64)dp[i-1][j-1])%rem;

    scanf("%d",&T);
    while(T--){
        scanf("%d%d%d",&N,&M,&K);
        int res=0,f=1;

        for(int j=1;j<=K;j++){
            f=((i64)f*(M-j+1)%rem);
            res=(((i64)res+(i64)f*dp[N][j])%rem);
        }
        printf("%d\n",res);
    }

    return 0;
}
