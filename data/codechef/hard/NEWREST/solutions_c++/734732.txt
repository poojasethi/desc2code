#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

typedef long long ll;

const ll M=1000000007;

int main(){
	// dp[i][j] := i 日間で j 種類作る方法の数
	//             j 個の区別できない箱に i 個のボールを入れる状況と同じ
	static ll dp[1001][1001];
	dp[0][0]=1;
	for(int i=1;i<=1000;i++) for(int j=1;j<=i;j++) {
		dp[i][j]=(j*dp[i-1][j]+dp[i-1][j-1])%M;
	}

	int T; scanf("%d",&T);
	while(T--){
		int n,m,k; scanf("%d%d%d",&n,&m,&k);
		if(k>n) k=n;

		ll ans=0,mPj=1; // m 個の中から j 個を選ぶ方法の数 (順列)
		for(int j=1;j<=k;j++){
			mPj=mPj*(m-j+1)%M;
			ans=(ans+mPj*dp[n][j])%M;
		}
		printf("%lld\n",ans);
	}

	return 0;
}
