    #include<bits/stdc++.h>
    using namespace std;
    typedef long long ll;
    #define pb push_back
    #define mp make_pair
    #define MOD 1000000007
    #define PI 3.1415926535897932
    #define pii pair < int,int > 
    #define fi first
    #define se second
  
    vector <ll> vf;
    vector <ll> vb;
    ll h[109];
    ll poet(ll num)
    {
    	ll i=1,c=0;
    	while(i<=num)
    	{
    		i=i*2;
    		c++;
    	}
    	return c-1;
    }
    int main()
    {
        ll t,k,i,j;
        scanf("%lld",&t);
        while(t--)
        {
        	vf.clear();
        	vb.clear();
        	for(i=0;i<=105;i++)
        	h[i]=0;
        	scanf("%lld",&k);
        	if(k==1)
        	{
        		printf("1\n1\n");
        		continue;
        	}
        	ll ret=poet(k),rem=k-pow(2,ret);
        	

        	while(rem!=0)
        	{
        		ll rr=poet(rem);
        		h[rr]=1;
        		rem=rem-pow(2,rr);
        	}
        	
        
        	ll curr=1;
        	for(i=ret-1;i>=0;i--)
        	{
        		vb.pb(curr+1);
        		vb.pb(curr);
        		curr+=2;
        		if(h[i]==1)
        		{
        			for(j=vf.size()+i+1;j<=ret;j++)
        			vf.pb(curr++);
        		}
        	}
        	ll s=vf.size()+vb.size();
        	printf("%lld\n",s);
        	for(i=0;i<vf.size();i++)
        	printf("%lld ",vf[i]);
        	for(i=0;i<vb.size();i++)
        	printf("%lld ",vb[i]);
        	printf("\n");
        }
        return 0;
    }
