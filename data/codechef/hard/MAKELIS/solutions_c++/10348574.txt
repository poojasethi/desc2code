    int ans[100], binrep[100];
    int ndig(int k)
    {
    	int c = 0;
    	while(k)
    	{
    		k/=2;
    		c++;
    	}
    	return c;
    }
    int main()
    {

    	int t;
    	cin >> t;
    	while(t--)
    	{
    		int k, n = 0;
    		cin >> k;
    		int la = ndig(k), fi = 0, curr = 1;
    		int temp = la - 1;
    		cout << (3*la - 1) << endl;
    		while(k)
    		{
    			binrep[temp--] = k%2;
    			k/=2;
    		}
    		temp = la;
    		for(int i = 0; i < temp; i++)
    		{
    			if(i == temp-1)
    			{
    				if(binrep[i] == 1)	
    				{
    					ans[fi++] = curr++;
    					ans[la++] = curr;
    				}
    				else
    				{
    					ans[la++] = curr++;
    					ans[fi++] = curr;
    				}
    			}
    			else if(binrep[i]%2)
    			{
    				ans[fi++] = curr++;
    				ans[la++] = curr + 1;

    			}
