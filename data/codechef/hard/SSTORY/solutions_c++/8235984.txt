#include <bits/stdc++.h>
using namespace std;
#define md int(250000+100)
#define modul int(1e9+7)
#define FOR(i,a,b) for( int i=(a),_b=(b);i<=_b;i++)
#define DOW(i,b,a) for( int i=(b),_a=(a);i>=_a;i--)
int xx[4]={0,0,1,-1};
int yy[4]={1,-1,0,0};
typedef long long ll;
int n;
int r[3][md][20],l[3];
int t[3];
char s[3][md];
struct suffix
{
    pair<int,int> p;
    int x,i;
};
suffix a[2*md];

int calc(suffix a, suffix b)
{
    int res=0;
    DOW(k,18,0)
    if (a.i+(1<<k)-1<=l[a.x] && b.i+(1<<k)-1<=l[b.x] && r[a.x][a.i][k]==r[b.x][b.i][k])
    {
        a.i+=(1<<k);
        b.i+=(1<<k);
        res+=(1<<k);
    }
    return res;
}

bool cmp(const suffix &a,const suffix &b)
{
    return a.p<b.p;
}

int main()
{
    //freopen("inp.txt","r",stdin);
    scanf("%s",s[1]+1);
    scanf("%s",s[2]+1);
    l[1]=strlen(s[1]+1);
    l[2]=strlen(s[2]+1);
    FOR(i,1,l[1])
    {
        r[1][i][0]=s[1][i];
    }
    FOR(i,1,l[2])
    {
        r[2][i][0]=s[2][i];
    }
    int dem;
    FOR(k,1,18)
    {
        dem=0;
        FOR(x,1,2)
        FOR(i,1,l[x])
        {
            dem++;
            a[dem].x=x;
            a[dem].i=i;
            if (i+(1<<(k-1))<=l[x])
            {
                a[dem].p=make_pair(r[x][i][k-1],r[x][i+(1<<(k-1))][k-1]);
            }
            else
                a[dem].p=make_pair(r[x][i][k-1],-1);
        }
        sort(a+1,a+1+dem,cmp);
        int d=0;
        FOR(i,1,dem)
        {
            if (i==1 || a[i].p!=a[i-1].p) d++;
            r[a[i].x][a[i].i][k]=d;
        }
    }
    int res=0,pos=0;
    FOR(i,1,dem)
    {
        int x;
        int j=t[3-a[i].x];
        if (j!=0 && a[i].x==2)
        {
            x=a[i].i;
            int gt=calc(a[i],a[j]);
            if (gt>res)
            {
                res=gt;
                pos=x;
            }
            else if (gt==res && x<pos)
            {
                pos=x;
            }
        }

        t[a[i].x]=i;
    }
    memset(t,0,sizeof(t));
    DOW(i,dem,1)
    {
        int x;
        int j=t[3-a[i].x];
        if (j!=0 && a[i].x==2)
        {
            x=a[i].i;
            int gt=calc(a[i],a[j]);
            if (gt>res)
            {
                res=gt;
                pos=x;
            }
            else if (gt==res && x<pos)
            {
                pos=x;
            }
        }
        t[a[i].x]=i;
    }
    if (res!=0)
    {
        FOR(i,pos,pos+res-1)
        {
            printf("%c",s[2][i]);
        }
        cout<<endl;
    }
    cout<<res<<endl;
}
