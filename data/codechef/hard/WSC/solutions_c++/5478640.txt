#include <iostream>
#include <cstdio>
#include <list>
#include <limits.h>
#include <cstring>
#include <string>

using namespace std;

int main()
{
	int t,n,m,x,y,p,i,j;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		int arr[21]={0};
		int maxCap = -20;
		p=m;
		while(p--)
		{
			cin>>x>>y;
			arr[x]++;
			arr[y]++;
			maxCap = max(maxCap,arr[x]);
			maxCap = max(maxCap,arr[y]);
		}
		if(n==1||n==2)
		{
			cout<<"YES"<<endl;
		}
		else if(m>2)
		{
			cout<<"NO"<<endl;
		}
		else if(m==2 && maxCap==1)
		{
			cout<<"NO"<<endl;
		}
		else
		{
			cout<<"YES"<<endl;
		}
	}
}
