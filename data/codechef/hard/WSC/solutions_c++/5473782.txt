#include<bits/stdc++.h>
using namespace std;

int main()
{
    long i,j,x,y,rt,M,N,k=0;
     cin>>rt;
    while( rt-- )
        {
        cin>>N>>M;
        long cnt[20] = {0};
        for( i=1;i<=M;i++ ){
            cin>>x>>y;
            cnt[x]++;
            cnt[y]++;
        }
        long c = 0;
        for( i=1;i<=N;i++ ){
            if( cnt[i] >= 2 ) c++;
        }
        if( M>2 or (M==2 and !c) ) printf("NO\n");
        else printf("YES\n");

    }

    return 0;
}
