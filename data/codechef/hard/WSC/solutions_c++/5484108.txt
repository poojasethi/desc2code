#include<bits/stdc++.h>
using namespace std;
int main()
{
      int t;
      cin>>t;
      while(t--)
      {
            int n,m,x,y;
            cin>>n>>m;
            if(m<2)
            {
                  for(int i=0;i<m;i++)
                  cin>>x>>y;
                  cout<<"YES\n";
                  continue;
            }
          else  if(m>2)
            {
                  for(int i=0;i<m;i++)
                        cin>>x>>y;
                  cout<<"NO\n";
                  continue;
            }
           else if(m==2)
           {
                 int x1,y1,x2,y2;
                 cin>>x1>>y1>>x2>>y2;
                 if(x1!=x2&&x1!=y2&&y1!=y2&&x2!=y1)
                  cout<<"NO\n";
                 else
                  cout<<"YES\n";
                 continue;
           }

      }
      return 0;
}
