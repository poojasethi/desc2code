#include<iostream>
#include<string>
using namespace std;

int main()
{
    int t;
    cin>>t;
    for(int i=0;i<t;i++)
    {
        int n,m;
        cin>>n>>m;
        int a[m][2];
        for(int j=0;j<m;j++)
        {
            for(int k=0;k<2;k++)
            {
                cin>>a[j][k];
            }
        }
        int flag;

        for(int j=1;j<=n;j++)
        {
            flag=0;
            for(int k=0;k<m;k++)
            {
                if(a[k][0]==j || a[k][1]==j)
                flag++;
            }
            if(m<=2 && flag==m)
            {
                cout<<"YES"<<endl;
                break;
            }
        }
        if(m>2 || flag!=m)
        cout<<"NO"<<endl;
    }
return 0;
}
