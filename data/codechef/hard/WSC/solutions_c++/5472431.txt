#include<stdio.h>

int main()
{
    int t;int b[20],n,i,m,x,y,flag=0,q=0;
    scanf("%d",&t);
    while(t--)
    {
              flag=0;q=0;
              scanf("%d%d",&n,&m);
              if(m>2)
              flag=1;
              for(i=0;i<=10;i++)
              b[i]=0;
              for(i=0;i<m;i++)
              {
                              scanf("%d%d",&x,&y);
                              b[x]++;b[y]++;

              }
              if(flag==1)
              printf("NO\n");
              else if(m<2)
              printf("YES\n");
              else
              {
                  for(i=0;i<=10;i++)
                  {
                                    if(b[i]==2)
                                    {
                                               q=1;
                                               break;

                                    }

                  }
                  if(q==1)
                  printf("YES\n");
                  else
                  printf("NO\n");

              }

    }

 return 0;
}
