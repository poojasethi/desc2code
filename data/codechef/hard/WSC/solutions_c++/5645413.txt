#include<cstdio>
int T,N,M,X[25],Y[25];
int main(){
    scanf("%d",&T);
    while(T--){
        scanf("%d %d",&N,&M);
        for(int i=1;i<=M;i++)
            scanf("%d %d",&X[i],&Y[i]);
        if((N==1)||(N==2)) printf("YES\n");
        else
            if(M<2) printf("YES\n");
            else
                if(M>2) printf("NO\n");
                else
                    if(((X[1]==X[2]))||((X[1]==Y[2]))||((Y[1]==Y[2]))||((Y[1]==X[2])))
                        printf("YES\n");
                    else printf("NO\n");

    }
}
