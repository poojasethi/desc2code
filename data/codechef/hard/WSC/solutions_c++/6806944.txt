#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
    int n,m,a,b,temp;
    vector<int> que;
    int flag=0;
    scanf("%d %d",&n,&m);
    for(int i=0;i<m;i++)
    {
        scanf("%d %d",&a,&b);
        if(m==2)
        {
            que.push_back(a);
            que.push_back(b);
        }
    }
    if(n==1 || n==2)
    {
        printf("YES\n");
    }
    else if(n==3)
    {
        if(m<=2)
        printf("YES\n");
        else
            printf("NO\n");
    }
    else if(n>=4)
    {
        if(m<=1)
            printf("YES\n");
        else if(m>=3)
            printf("NO\n");
        else if(m==2)
        {
        sort(que.begin(),que.end());
        temp=que[0];
        for(int i=1;i<4;i++)
        {
            if(temp==que[i])
                flag=1;
                temp=que[i];
        }
        if(flag==0)
            printf("NO\n");
        else if(flag==1)
            printf("YES\n");
        }
    }
    }
    return 0;
}
