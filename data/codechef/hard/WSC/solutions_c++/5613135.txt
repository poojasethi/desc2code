#include<stdio.h>
 
 
int main() {
	int n,m,e[21];
	int d[20][2];
	int i,j,k;
	int ret = 0;
	int t;
	
	scanf("%d",&t);
	while(t--) {
		ret = 0;
		scanf("%d %d",&n,&m);
		for(i=0;i<n;i++) {
			e[i] = 0;
		}
		for(i=0;i<m;i++) {
			scanf("%d %d",&d[i][0], &d[i][1]);
		}
		if(m>2)	{
		
			printf("NO\n");
			continue;
		}
		else {
		if(m==1) {printf("YES\n"); continue;}
		for(i=0;i<m;i++) {
 
			if(e[d[i][0]-1] == 0) {
				e[d[i][0]-1] = 1;
				ret++;
			}
			if(e[d[i][1]-1] == 0) {
				e[d[i][1]-1] = 1;
				ret++;
			}
		}
			if(ret <= 3) {
				 printf("YES\n");
			} else 
				printf("NO\n");
		}
	}
	return 0;
}
