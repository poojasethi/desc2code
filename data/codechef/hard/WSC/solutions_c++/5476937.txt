#include <iostream>
#include <string.h>
using namespace std;

int main()
{
    int n,m,t,a,v,i,count,b;
    int e[30];
    int f[30];
    int arr[30];
    cin >> t;
    while(t--){
    memset(e,0,sizeof(e));
    memset(f,0,sizeof(f));
    memset(arr,0,sizeof(arr));
    cin >> n;
    cin >> m;
    count = 0;
    for (i = 0; i < m; i++){
        cin >> a;
        cin >> b;
        e[a]++;
        f[b]++;
    }
    if (m > 2){
        cout << "NO"<<endl;
        continue;
    }
    if (m == 0 || m == 1){
        cout << "YES"<<endl;
        continue;
    }
    if (m == 2){
        for (i = 0; i < 30; i++){
            if(e[i] > 0){
                arr[i] = 1;
            }
            if(f[i] > 0){
                arr[i] = 1;
            }
        }
        for (i = 0; i < 30; i++){
            if (arr[i] == 1){
                count++;
            }
        }
       // cout << count;
        if (count > 3){
            cout << "NO"<<endl;
        }
        else {
            cout << "YES"<<endl;
        }
        continue;
    }
    }
    return 0;
}
