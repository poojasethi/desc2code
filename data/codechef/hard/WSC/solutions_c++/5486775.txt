#include<bits/stdc++.h>

int main()
{
    int i,t,n,m,c,d,x,y,eats,eaten,h,l;

    scanf("%d",&t);

    while(t--)
    {   
        l=1;
        c=d=0;
        scanf("%d %d",&n,&m);
        if(m==0)
        {
            printf("YES\n");
            continue;
        }

        scanf("%d %d",&x,&y);
        eats=x;
        eaten=y;

        if(m==1)
        {
            printf("YES\n");
            continue;
        }

        scanf("%d %d",&x,&y);
        if(eats==x || eaten==x)
            h=x;
        else if(eaten==y || eats==y)
            h=y;
        else
            l=0;
            
        if(eats==h)
            c++;
        if(x==h)
            c++;

        if(eaten==h)
            d++;
        if(y==h)
            d++;

        for(i=2;i<m;i++)
        {
            scanf("%d %d",&x,&y);
            if(x!=h && y!=h)
                l=0;
            if(l==1)
            {
                if(x==h)
                    c++;
                else if(y==h)
                    d++;
            }
        }

        if(l==1 && (c+d)<=2)
            l=1;
        else
            l=0;

        if(l)
            printf("YES\n");
        else
            printf("NO\n");

    }
    return 0;
}
