#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int t,n,m;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		int a[1000],b[1000];
		for(int i=0;i<m;i++)
		{
			cin>>a[i]>>b[i];
		}
		if(n==1 || n==2)
		{
			cout<<"YES"<<endl;
		}
		else if(m>2)
		{
			cout<<"NO"<<endl;
		}
		else  if(m==2)
		{
			if(b[0]==a[1] || a[0]==b[1] || a[0]==a[1] || b[0]==b[1])
			{
				cout<<"YES"<<endl;
			}
			else
			{
				cout<<"NO"<<endl;
			}
		}
		else
		{
			cout<<"YES"<<endl;
		}
	}
}
