#include<cstdio>
 
#include<algorithm>
#include<iostream>
using namespace std;
int main()
{
    int t;
    int a[100001],x,y,n,m;
    scanf("%d",&t);
 
    while(t--)
    {
          cin>>n>>m;
          int mx;
         for(int i=0;i<=n;i++)
                a[i]=0;
         mx=0;
         for(int i=0;i<m;i++)
         {
            cin>>x>>y;
            a[x]++;
            a[y]++;
        }
        mx=0;
        for(int i=0;i<=n;i++)
        {
            mx=max(a[i],mx);
        }
 
        if(m>2 || (m==2 && mx<2))
        {
            cout<<"NO"<<endl;
        }
        else
        {
            cout<<"YES"<<endl;
        }
 
 
    }
 
 
}
