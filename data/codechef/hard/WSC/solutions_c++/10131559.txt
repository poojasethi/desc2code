	cin>>t;
	while(t--)
	{
    cin>>n>>m;
    s.clear();
    for(int i=1;i<=m;i++)
    {
    	cin>>x[i]>>y[i];
    	if(i<=2)
    	{
	    	s.insert(x[i]);
	    	s.insert(y[i]);
	    }
    	
    }
    if(m>2 ||m==2 && s.size()!=3)

	}

}
