#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
    long int t,n,m,x[100],y[100];
    scanf("%ld",&t);
    while(t--)
    {
        scanf("%ld%ld",&m,&n);
        for(int i=0;i<n;i++)
            scanf("%ld%ld",&x[i],&y[i]);
        if(n>2)
        {
            cout<<"NO"<<endl;
        }
        else if(n<2)
        {
            cout<<"YES"<<endl;
        }
        else{
            if(y[0]==x[1] || y[0]==y[1])
                cout<<"YES"<<endl;
            else if(x[0]==x[1] || x[0]==y[1])
                cout<<"YES"<<endl;
            else cout<<"NO"<<endl;
        }
    }
    return 0;
}
