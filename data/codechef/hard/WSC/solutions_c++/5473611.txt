#include <iostream>
using namespace std;
 
int main() 
{
	int t, n, m, x[20], y[20];
	cin>>t;
	while(t>0)
	{
		cin>>n>>m;
		for(int i=0; i<m; i++)
		{
			cin>>x[i]>>y[i];
		}
		
		if(m>2)
		{
			cout<<"NO\n";
		}
		else if(m<2)
		{
			cout<<"YES\n";
		}
		else
		{
			if(y[0]==x[1] || y[0]==y[1])
			{
				cout<<"YES\n";
			}
			else if(x[0]==x[1] || x[0]==y[1])
			{
				cout<<"YES\n";
			}
			else
			{
				cout<<"NO\n";
			}
		}
		//cout<<endl;
		t--;
	}
	return 0;
}
