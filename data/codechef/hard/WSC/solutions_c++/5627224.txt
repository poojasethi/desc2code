#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;
bool bol[15];

int main()
{
    ios_base::sync_with_stdio(false);
    int t,n,i,m,x,y,cnt;
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        for(i=0;i<n;i++)
            bol[i] = false;
        for(i=0;i<m;i++)
        {
            cin>>x>>y;
            bol[x-1] = true;
            bol[y-1] = true;
        }
        cnt = 0;
        for(i=0;i<n;i++)
        {
            if(bol[i]) cnt++;
        }
        if(cnt<3)
            cout<<"YES"<<endl;
        else if(cnt==3)
        {
            if(m==2) cout<<"YES"<<endl;
            else cout<<"NO"<<endl;
        }
        else cout<<"NO"<<endl;
    }
    return 0;
}
