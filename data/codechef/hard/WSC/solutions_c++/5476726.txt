#include <iostream>
#include <algorithm>
using namespace std;
 
int main(){
 
int t,n,m,i,an,cnt,j,cnt1;
cin>>t;
while(t--){
  cin>>n>>m;
  if(m==0){
       cout<<"YES"<<endl; 
  continue;}
  int p[m+1][2],p1[2*m+1];
  j=1;
  for(i=1;i<=m;i++){
     cin>>p[i][0]>>p[i][1];
     p1[j++]=p[i][0];
     p1[j++]=p[i][1];
  }
  sort(p1+1,p1+2*m+1);
  
  
  
  cnt1=1;
  for(i=1;i<2*m;i++){
    if(p1[i]!=p1[i+1]){
        cnt1++;
    }
  }
 
 
  if(cnt1<=2) cout<<"YES"<<endl;
  else if(cnt1==3&&m==2) cout<<"YES"<<endl;
  else cout<<"NO"<<endl;
}
 
}
