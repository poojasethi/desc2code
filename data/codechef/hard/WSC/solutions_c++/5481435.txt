#include<stdio.h>
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    for(int j=0;j<t;j++)
{
int n,m,cnt[11]={0},x,y,fact=0;
scanf("%d %d",&n,&m);
for(int i=0;i<m;i++)
{
    scanf("%d %d",&x,&y);
    cnt[x]++;
    cnt[y]++;
}
if(n==1 || n==2)
    printf("YES\n");
else
{
for(int i=1;i<=n;i++)
{
if(cnt[i]==2)
fact++;
}

if((fact==1 && m==2)||(m<2))
    printf("YES\n");
else
    printf("NO\n");
}
}
return 0;
}
