#include <iostream>
using namespace std;

int main() {
    int t, n, m, i, j, x, y, c;
    cin >> t;
    while (t--) {
        cin >> n >> m;
        int b[n+1];
        for (i = 0; i <= n; i++) {
            b[i] = 0;
            }
        c = 0;
        for (i = 1; i <= m; i++) {
            cin >> x >> y;
            if (b[x] == 0) c++;
            if (b[y] == 0) c++;
            b[x] = 1;
            b[y] = 1;
            }
        if (c >= 4) cout << "NO\n";
        else if (c < 3) cout << "YES\n";
        else if (m == 2) cout << "YES\n";
        else cout << "NO\n";
    }
    return 0;
}
