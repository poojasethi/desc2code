#include<bits/stdc++.h>
using namespace std;

const int N=22;
int in[N], out[N];

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n, m;
        scanf("%d %d", &n, &m);
        for(int i=0;i<m;i++) {
            int u, v;
            scanf("%d %d", &u, &v);
            in[u]++;
            out[v]++;
        }
        if(m>2) {
            puts("NO");
        } else if(m<2) {
            puts("YES");
        } else {
            if(n==2) {
                puts("YES");
            } else {
                int ctr=0;
                for(int i=0;i<N;i++) {
                    if(in[i] || out[i]) {
                        ctr++;
                    }
                }
                if(ctr<4) {
                    puts("YES");
                } else {
                    puts("NO");
                }
            }
        }
        memset(in, 0, sizeof(in));
        memset(out, 0, sizeof(out));
    }
}
