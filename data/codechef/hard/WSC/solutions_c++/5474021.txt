#include <iostream>
#include <cstdio>

using namespace std;

int main()
{

    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);

    bool ans;
    int n,m,i,j;
    int a[21][2];
    long long tc;
    cin >> tc;

    while(tc--)
    {
        scanf("%d %d",&n,&m);

        for(i=0;i<m;i++)
        {
            scanf("%d %d",&a[i][0],&a[i][1]);
        }

        if (m<2)
        {
            cout << "YES" << endl;
            continue;
        }

        if (m>2)
        {
            cout << "NO" << endl;
            continue;
        }

        ans=false;

        for(i=1;i<=n;i++)
        {
            for(j=0;j<m;j++)
            {
                if (i!=a[j][0] && i!=a[j][1])
                    break;
            }
            if (j==m)
            {
                ans=true;
                break;
            }
        }
        if (ans)
            cout << "YES" << endl;
        else
            cout << "NO" << endl;
    }
	return 0;
}
