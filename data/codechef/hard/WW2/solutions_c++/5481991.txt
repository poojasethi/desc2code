int N,M;
ll temp[30][30];
ll X[30][30],Y[30][30];

void mult(ll A[][30] , ll B[][30])
{
  memset(temp,0,sizeof(temp));
  for(int i=0;i<M;i++)
  {
    for(int j=0;j<M;j++)
    {
      for(int k=0;k<M;k++)
      {
        temp[i][j]+=(A[i][k]*B[k][j])%MOD;
        temp[i][j]%=MOD;
      }
    }
  }
  for(int i=0;i<M;i++)
  {
    for(int j=0;j<M;j++)
    {
      A[i][j]=temp[i][j];
    }
  }
}

void pow(ll A[][30],int b)
{ 
  memset(X,0,sizeof(X));
  memset(Y,0,sizeof(Y));
  for(int i=0;i<M;i++)
  { 
    X[i][i]=1;
    for(int j=0;j<M;j++)
    {
      Y[i][j]=A[i][j];
    }
  } 
  while(b)
  {
    if(b%2)
    mult(X,Y);
    mult(Y,Y);
    b/=2;
  }
  for(int i=0;i<M;i++)
  {
    for(int j=0;j<M;j++)
    {
      A[i][j]=X[i][j];
    }
  }
}


int main()
{
  int t;
  si(t);
  ll A[30][30],B[30][30],C[30][30];
  while(t--)
  {
    si(N);si(M);ll ans=0;
    if(N==1)
    {
      pi(M);nl;
      continue;
    }
    memset(A,0,sizeof(A));
    memset(B,0,sizeof(B));
    memset(C,0,sizeof(C));
    for(int i=0;i<M;i++)
    { 
      B[i][i]=1;
      if(i+1<M)
      {
        B[i][i+1]=1;A[i][i+1]=1;
      }
      if(i-1>=0)
      {
        B[i][i-1]=1;A[i][i-1]=1;
      }
    }
    for(int i=0;i<M;i++)
    {
      for(int j=0;j<M;j++)
      {
        C[i][j]=B[i][j];
      }
    }
    if(N%2==0)
    {
     mult(B,A);
     pow(B,(N/2)-1);
     mult(A,B);
    }
    else
    { 
      mult(B,A);
      pow(B,N/2);
    }
    for(int i=0;i<M;i++)
    {
      for(int j=0;j<M;j++)
      { 
        if(N%2)
        ans+=B[i][j];
        else
        ans+=A[i][j];
        ans%=MOD;
      }
    }
    pl(ans);nl;
  }
}
