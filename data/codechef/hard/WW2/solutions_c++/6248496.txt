#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)

using namespace std;

#define matrix vector<vector<ll> >
ll mod=1000000007;

matrix multiply(matrix a,matrix b,ll k)
{
    ll i,j,w;
    matrix t(k);
    for(i=0;i<k;i++) t[i].resize(k,0);
    for(i=0;i<k;i++) {
        for(j=0;j<k;j++) {
            for(w=0;w<k;w++) {
                t[i][j]=(t[i][j]+a[i][w]*b[w][j])%mod;
            }
        }
    }
    return t;
}

matrix power(matrix a,ll p,ll k)
{
    if(p==1) return a;
    matrix t=power(a,p/2,k);
    t=multiply(t,t,k);
    if(p%2==0) return t;
    t=multiply(t,a,k);
    return t;
}

int main()
{
    ll t,n,k,i,j,x,ans;
    matrix odd(30);
    matrix even(30);

    for(i=0;i<30;i++) {
        odd[i].resize(30,0);
        even[i].resize(30,0);
        for(j=0;j<30;j++) {
            odd[i][j]=0;
            even[i][j]=0;
        }
    }
    for(i=0;i<30;i++) {
        for(j=-1;j<=1;j++) {
            x=i+j;
            if(x<0 || x>=30) continue;
            if(i==x) {
                even[i][x]=1;
                continue;
            }
            else {
                even[i][x]=1;
                odd[i][x]=1;
            }
        }
    }

    s(t);
    while(t--) {
        ans=0;
        s(n);s(k);
        if(n==1) {cout<<k<<endl;continue;}
        if(n==2) {
            for(i=0;i<k;i++) {
                for(j=0;j<k;j++) {
                    ans=(ans+odd[i][j])%mod;
                }
            }
            cout<<ans<<endl;
            continue;
        }
        n--;
        matrix tt=multiply(odd,even,k);

        tt=power(tt,n/2,k);
        if(n%2==1) {
            tt=multiply(tt,odd,k);
        }
        for(i=0;i<k;i++) {
            for(j=0;j<k;j++) {
                ans=(ans+tt[i][j])%mod;
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
