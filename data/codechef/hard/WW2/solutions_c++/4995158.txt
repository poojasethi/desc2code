#include<iostream>
#define MOD 1000000007
using namespace std;
struct sethu
{
    long long int matrix[65][65];
}id;
struct sethu multiply(struct sethu s1,struct sethu s2,long long int m)
{
 struct sethu s3;
 long long int i,j,k;
    for(i=1;i<=m;i++)
    {
        for(j=1;j<=m;j++)
        {
            s3.matrix[i][j]=0;
            for(k=1;k<=m;k++)
            {
                s3.matrix[i][j]+=(s1.matrix[i][k]*s2.matrix[k][j])%MOD;
                s3.matrix[i][j]%=MOD;
            }
        }
    }
    return s3;
}
struct sethu power(struct sethu s1,long long int n,long long int m)
{
 if(n==0)
    return id;
 if(n==1)
    return s1;
 struct sethu s2;
 s2=power(s1,n/2,m);
 s2=multiply(s2,s2,m);
 if(n%2!=0)
        s2=multiply(s2,s1,m);
 return s2;
}
int main()
{
    long long int t;
    cin>>t;
    while(t--)
    {
        long long int j,n,m,i;
        struct sethu even,odd,temp;
        cin>>n>>m;
        for(i=1;i<=m;i++)
        {
            for(j=1;j<=m;j++)
            {
                id.matrix[i][j]=0;
            }
            id.matrix[i][i]=1;
        }
        for(i=1;i<=m;i++)
        {
            even.matrix[i][i]=1;
            even.matrix[i][i-1]=1;
            even.matrix[i][i+1]=1;
            odd.matrix[i][i-1]=1;
            odd.matrix[i][i+1]=1;
        }
        temp=multiply(odd,even,m);
        temp=power(temp,(n-1)/2,m);
        if(n%2==0)
            temp=multiply(temp,odd,m);
        long long int ans=0;
        for(i=1;i<=m;i++)
        {
            for(j=1;j<=m;j++)
            {
                ans+=temp.matrix[i][j];
                ans%=MOD;
            }
        }
        cout<<ans<<endl;

    }
}
