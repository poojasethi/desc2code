#include <iostream>
#include <cstring>
#define MOD 1000000007

using namespace std;

long long iden[30][30];
long long odd[30][30],even[30][30],ans[30][30],T[30][30];

int main()
{
    int t;
    cin>>t;
    while(t--){
        int n,m;
        cin>>n>>m;
        for(int i=0;i<m;i++){
            for(int j=0;j<m;j++){
                odd[i][j] = even[i][j] = iden[i][j] = 0;
            }
        }
        for(int i=0;i<m;i++)
            iden[i][i] = 1;
        for(int i=0;i<m;i++){
            odd[i][i] = 1;
            if(i-1>=0){
                odd[i][i-1] = even[i][i-1] = 1;
            }
            if(i+1<m){
                odd[i][i+1] = even[i][i+1] = 1;
            }
        }
        memset(ans,0,sizeof(ans));
        for(int i=0;i<m;i++){
            ans[i][i] = 0;
        }
        for(int i=0;i<m;i++){
            for(int j=0;j<m;j++){
                for(int k=0;k<m;k++){
                    ans[i][j] = ans[i][j] + (even[i][k]*odd[k][j])%MOD;
                }
                ans[i][j] = ans[i][j]%MOD;
                //cout<<ans[i][j]<<" ";
            }
            //cout<<"\n";
        }
        n-=1;
        int p = n/2;
        memset(T,0,sizeof(T));
        for(int i=0;i<m;i++){
            T[i][i]=1;
        }
        //cout<<p<<"\n";
        while(p!=0){
            long long temp[m][m];
            if(p%2 == 1){
                memset(temp,0,sizeof(temp));
                for(int i=0;i<m;i++){
                    for(int j=0;j<m;j++){
                        for(int k=0;k<m;k++){
                            temp[i][j] = temp[i][j] + (T[i][k]*ans[k][j])%MOD;
                        }
                        //cout<<temp[i][j]<<" ";
                        temp[i][j] = temp[i][j]%MOD;
                    }
                    //cout<<"\n";
                }
                //cout<<"\n\n";
                for(int i=0;i<m;i++){
                    for(int j=0;j<m;j++){
                        T[i][j] = temp[i][j];
                    }
                }
            }
            p/=2;
            memset(temp,0,sizeof(temp));
            for(int i=0;i<m;i++){
                for(int j=0;j<m;j++){
                    for(int k=0;k<m;k++){
                        temp[i][j] = temp[i][j] + (ans[i][k]*ans[k][j])%MOD;
                    }
                    temp[i][j] = temp[i][j]%MOD;
                }
            }
            for(int i=0;i<m;i++){
                for(int j=0;j<m;j++){
                    ans[i][j] = temp[i][j];
                }
            }
        }
        if(n%2){
            long long temp[m][m];
            memset(temp,0,sizeof(temp));
            for(int i=0;i<m;i++){
                for(int j=0;j<m;j++){
                    for(int k=0;k<m;k++){
                        temp[i][j] = temp[i][j] + (T[i][k]*even[k][j])%MOD;
                    }
                    temp[i][j] = temp[i][j]%MOD;
                }
            }
            for(int i=0;i<m;i++){
                for(int j=0;j<m;j++){
                    T[i][j] = temp[i][j];
                    //cout<<T[i][j]<<" ";
                }
                //cout<<"\n";
            }
        }
        long long ans1=0;
        for(int i=0;i<m;i++){
            for(int j=0;j<m;j++){
                ans1 = ans1 + T[i][j];
                ans1%=MOD;
            }
        }
        cout<<ans1<<"\n";
    }
    return 0;
}
