#include<iostream>
#include<cstdio>
#include<algorithm>
#define ll long long int
#define mod 1000000007
using namespace std;
ll odd[31][31],even[31][31];
ll res[31][31],T[31][31];
int m;
int mat_prod(ll A[31][31],ll B[31][31],ll D[31][31])
{
    ll C[31][31];
    for (int i=0;i<m;i++)
    {
        for (int j=0;j<m;j++)
        {
            C[i][j]=0;
            for (int k=0;k<m;k++)
            {
                C[i][j]+=(A[i][k]*B[k][j]);
                if((C[i][j])>=mod)
                    C[i][j]=(C[i][j])%mod;
            }
        }
    }
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<m;j++)
            D[i][j]=C[i][j];
    }
}
void pow_mod(ll A[31][31],int b)
{
    ll B[31][31];
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<m;j++)
            if(i==j)
            B[i][j]=1;
            else
                B[i][j]=0;
    }
    while(b>0)
    {
        if(b%2==1)
            mat_prod(B,A,B);
        mat_prod(A,A,A);
        b=b/2;
    }
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<m;j++)
            res[i][j]=B[i][j];
    }
}

ll count(ll A[31][31])
{
    ll count1=0;
    for(int i=0;i<m;i++)
    {
        for(int j=0;j<m;j++)
        {
            count1+=A[i][j];
            if(count1>=mod)
                count1=count1%mod;
        }
    }
    return count1;
}

 {
     ll n,ans;
     int t,i,j;
     for(j=0;j<31;j++)

                if(abs(j-i)<=1)

 //t=10;

        {

        }
        else

         //   print(T);
