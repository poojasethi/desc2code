void multiply(ll c[40][40],ll b[40][40],ll a[40][40])
{
    ll i,j,k,d[40][40]={0};
    for(i=1;i<=m;i++)
    {
        for(j=1;j<=m;j++)
        {
            for(k=1;k<=m;k++)
            {d[i][j]+=(b[i][k]*a[k][j]);d[i][j]%=mod;}
        }
    }
    for(i=1;i<=m;i++)
    {
        for(j=1;j<=m;j++)
            c[i][j]=d[i][j];
    }
}

void power(ll c[40][40],ll x)
{
    ll d[40][40]={0},i,j;
    for(i=0;i<40;i++)
        d[i][i]=1;
    while(x>0)
    {
        if((x%2)==1)
        {
            multiply(d,d,c);
        }
        x=x/2;
        multiply(c,c,c);
    }
    for(i=1;i<=m;i++)
        {
            for(j=1;j<=m;j++)
            {
                c[i][j]=d[i][j];
            }
        }
}
