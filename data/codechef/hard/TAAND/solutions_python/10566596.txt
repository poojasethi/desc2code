
N = input()
S = []
for i in range(N):
  S += [input()]

#N = 4
#S = [5, 3, 4, 1]
#S = [2,4,8,10]

ma = 0
for i in range(N):
  a = S[i].bit_length()
  if ma < a:
    ma = a

for e in range(ma,-1,-1):
  msk = 1 << e
  t = []
  for s in S:
    if s & msk:
      t += [s]
  if len(t) > 1:
    S = t

m = (1 << ma) - 1
for a in S:
  m &= a
print m
