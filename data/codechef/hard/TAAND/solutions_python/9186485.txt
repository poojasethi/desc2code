n = input()
arr = []
for _ in xrange(n):
    x = input()
    arr += [x]
ans = arr[0]&arr[1]
for i in xrange(30,-1,-1):
    mask = 1<<i
    on = []
    for x in arr:
        if((x&mask) !=0):
            on+=[x]
    if(len(on)>=2):
        arr=on[:]
for i in xrange(len(arr)):
    for j in xrange(i+1,len(arr)):
        ans = max(ans,arr[i]&arr[j])
print ans
