n=input()
arr=[]
for i in range(n):
    arr.append(input())
arr.sort(reverse=True)
m=f=0
for i in range(n):
    for j in range(i+1,n):
        #print arr[i],arr[j]
        temp=arr[i]&arr[j]
        if temp>m:
            m=temp
        if m>=arr[i]:
            f=1
            break
    if f==1:
        break
print m
