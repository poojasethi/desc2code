numbers = int(raw_input().split()[0])
inp = []
while numbers > 0:
    inp.append(int(raw_input().split()[0]))
    numbers -= 1
inp.sort(reverse = True)
ans = 0

numbers = 0
while numbers < len(inp) - 1:
    if inp[numbers] & inp[numbers+1] > ans:
        ans = inp[numbers] & inp[numbers+1]
    numbers += 1
print ans
