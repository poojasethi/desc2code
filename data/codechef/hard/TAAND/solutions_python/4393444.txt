import sys
 
l = []
 
for i in xrange(int(sys.stdin.readline())):
    l.append(int(sys.stdin.readline()))
 
l.sort(reverse=True)
 
ma = -1
for i in xrange(len(l)):
    num = l[i]
    if num > ma:
        for j in xrange(i+1,len(l)):
            t = num&l[j]
            ma = max(t,ma)
    else:
        break
 
print ma
