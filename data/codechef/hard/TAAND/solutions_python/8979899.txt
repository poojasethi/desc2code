n=input()
def brutef(l):
	i=len(l)
	ans=0
	for j in range(i-1):
		for k in range(1,i-j):
			num=l[j]&l[j+k]
			ans=max(num,ans)
	return ans	
ar=[2**x for x in xrange(29,-1,-1)]
l=[]
ans=0
for i in xrange(n):
	l.append(input())
for i in range(30):
	d=[x for x in l if (x>=ar[i])]
	ll=len(d)
	if (ll>1):
		if (ll<100):
			ans+=brutef(d)
			break
		else:
			ans+=ar[i]
			l=[x-ar[i] for x in d]
	else:
		if (ll==1):l.remove(d[0])
print ans
