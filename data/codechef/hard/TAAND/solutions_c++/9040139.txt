#include <bits/stdc++.h>
using namespace std;

int main()
{
    int t,n,i,j;
    //cin>>t;
    //while(t--)
    //{
        cin>>n;
        int a[n];
        for(i=0;i<n;i++) cin>>a[i];
        sort(a,a+n);
        vector<int> v1(a,a+n),v2;
        int ans = 0;
        for(i=30;i>=0;i--)
        {
            if(v1.size() < 2) {ans <<= 1;continue;}
            v2.clear();
            for(j=0;j<v1.size();j++) if(v1[j]&(1<<i)) v2.push_back(v1[j]);
            if(v2.size() < 2) {ans <<= 1;continue;}
            ans <<= 1;
            ans += 1;
            v1 = v2;
        }
        cout<<ans<<'\n';
    //}
    return 0;
}
