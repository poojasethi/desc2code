#include<bits/stdc++.h>

using namespace std;

int main (){
    long long int n;
    cin>>n;
    long long int a[n], i, j;
    long long int num[64] = {0};
    long long int max = 0, c=0;
    
    long long int ind1=0, ind2=0;
    for(i=0; i< n;i++) {
        cin>>a[i];
    }
    
    for(i=0;i<n-1;i++)
     {
         if(a[i]>=max)
         for(j=i+1;j<n;j++)
         {

         }
     }

}
