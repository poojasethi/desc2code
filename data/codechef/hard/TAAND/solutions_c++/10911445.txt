#include<cstdio>
using namespace std;
int count_pattern(int bit_pattern,int arr[],int n){
int count=0;
for(int i=0;i<n;i++){
if((bit_pattern&arr[i])==bit_pattern){
count++;
}
}
return count;
}
int main(){
int t;
//scanf("%d",&t);
t=1;
while(t--){
int n;
scanf("%d",&n);
int arr[n];
for(int i=0;i<n;i++){
scanf("%d",&arr[i]);
}
int ans=0;
for(int i=30;i>=0;i--){
if(count_pattern((ans|1<<i),arr,n)>=2){
ans=(ans|1<<i);
}
}
printf("%d\n",ans);
}
}
