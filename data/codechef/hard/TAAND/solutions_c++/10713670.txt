ll res(ll pat,ll arr[],ll n){
	int cnt=0;
	Rep(i,n){
		if((arr[i]&pat)==pat)
			cnt++;
	}
	return cnt;
}
void solve(){
	ll n;
	scanf("%lld",&n);
	ll arr[n+1],a_max=0;
	Rep(i,n){
		scanf("%lld",arr+i);
		a_max = max(a_max,arr[i]);	
	}
	
	ll ans = 0;
	int l = ceil(log(a_max)/log(2));
	for(int i=l;i>=1;i--){
		 if(res(ans|(1<<(i-1)),arr,n)>=2)
		 	ans |= (1<<(i-1));
		//DEBUG(ans);
	}
	op(ans);
}

int main(){
	solve();
	return 0;
