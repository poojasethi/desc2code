#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k;
    
    int n;

    cin>>n;

    vector<int> a(n),b(n);

    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }

    bool found=false;

    i=31;

    int sum=0;

    while(i>=0 && !found)
    {
        //store all elements with bit i set in vector b
        j=0;

        for(k=0;k<n;k++)
        {
            if(a[k] & (1<<i))
            {
                b[j]=a[k];
                j++;
            }
        }

        if(j>=2)
        {
            sum+=(1<<i);

            for(k=0;k<j;k++)
            {
                a[k]=b[k];
            }

            n=j;
        }

        //cout<<"For i="<<i<<" sum="<<sum<<" n="<<n<<endl;

        i--;
    }

    cout<<sum<<endl;

    return 0;
}
