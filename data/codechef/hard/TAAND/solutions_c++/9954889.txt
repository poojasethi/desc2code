#include<cstdio>
#include<iostream>
 using namespace std;
 int n;
 long long int a[1000000],flag[1000000]={0};
 long long int ff(int l)
{

    if(l==-1)
        {
            return 0;
        }
    long long int i,sc=0,ans=0;

    for(i=0;i<n;i++)
      if(flag[i]==0)
       {
            if((a[i]&(1<<l)))
           {
               sc++;

           }
       }

     if(sc>=2)
     {


         for(i=0;i<n;i++)
                {

                                if((a[i]&(1<<l))==0)
                                         {

                                              flag[i]=1;
                                         }
                }
        ans+=(1<<l);
     }
     //cout<<"ans="<<ans;
     return ans+ff(l-1);
}
