#include<iostream>
#include<cstdio>
#include<string.h>

#define li long int
#define MAX 300005

using namespace std;
int n;
bool start=0;

li a[MAX];
int func(li num){
    int cnt=0;
    for(int x=0;x<n;x++){
        if((num & a[x])==num)
            cnt+=1;
    }
    return cnt;
}



int main(){
    int i,bit=30;
    li res=0,tmp;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%ld",&a[i]);
    }
    res=0;
    for(bit=30;bit>=0;bit--){
        tmp=res|(1<<bit);
        //cout<<bit<<"  "<<tmp<<" res: ";
        if(func(res|(1<<bit))>=2){
            res|=(1<<bit);
        }
         //cout<<res<<endl;
    }

    cout<<res<<endl;

    return 0;
}
