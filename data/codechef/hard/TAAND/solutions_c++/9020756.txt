#define gc getchar

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main(){
int n,i,j;
scanint(n);
int a[n];
for(i=0;i<n;i++){
    scanint(a[i]);
}
int ans=0;
int val;
for(i=0;i<n-1;i++){
        if(a[i]>=ans){
    for(j=i+1;j<n;j++){
       val=(a[i]&a[j]);
       if(val>ans) ans=val;
    }
    }
}
printf("%d\n",ans);
return 0;
}
