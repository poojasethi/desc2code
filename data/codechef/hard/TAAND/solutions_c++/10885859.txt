#include <iostream>
#include <vector>
using namespace std;

int p[300009];

int main(){
	int n; scanf("%d",&n);
	for(int i=0;i<n;i++){
	scanf("%d",&p[i]);
	}

	vector<int> a[2];
	vector<int> temp;
	
	int idx = 0;
		for(int j=0;j<n;j++)a[0].push_back(p[j]);


	
	for(int i=31;i>0;i--){
		temp.clear();
		int mask = 1<<i;
		
		for(int j=0;j<a[idx].size();j++){
			if(mask&a[idx][j])
			temp.push_back(a[idx][j]);
		}
		
		if(temp.size()>=2){
		 idx ^= 1;
		 a[idx]=temp;
		 if(temp.size()==2)break;
		}
		
	}
	
	printf("%d",a[idx][0]&a[idx][1]);
	
	return 0;
}
