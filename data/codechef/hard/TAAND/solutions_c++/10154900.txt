#include<bits/stdc++.h>
using namespace std;
int main(){
	
	long n;
	scanf("%d",&n);
	long a[n],lg[1000]={0},max=-1;
	for(int i=0;i<n;i++){
		scanf("%ld",&a[i]);
		long t=0;
		if(a[i]!=0)
		t=log2(a[i]);
		lg[t]++;
		if(max<t)
		max=t;
	}//if max == 0
	if(max==0){
		printf("0\n");
		exit(1);
	}
	vector<long> m;
	for(int i=max;i>0;i--){
		if(lg[i]>1){
			max=i;
			break;
		}else{
			max=0;
		}	
	}
	//printf("max : %ld\n",max);
	if(max==0){
		printf("0\n");
		exit(1);
	}
	long j=1<<max;//cout<<" j : "<<j<<endl;
	for(int i=0;i<n;i++){
		if(a[i]&j){
			m.push_back(a[i]);
		//	cout<<a[i]<<" ";
		}
	}
	long tet=(1<<max);
	vector<long> ans;
	while(max!=0){
		max=max-1;//cout<<"max : "<<max<<endl;
		for(int i=0;i<m.size();i++){
			if(m[i]&(1<<max)){
				ans.push_back(m[i]);
			//	cout<<":m "<<m[i]<<endl;
			}
				
		}
		if(ans.size() >= 2 ){
			m.clear();
			for(int i=0;i<ans.size();i++){
				m.push_back(ans[i]);
			//	cout<<ans[i]<<" ";
			}
			//cout<<endl;
			tet=tet|(1<<max);
		}
		ans.clear();
	}
	printf("%ld\n",tet);
}
