#include <iostream>
#include <vector>
using namespace std;

int main()
{
    int x = 1<<30;
    //cout<<x;
    int n,A;
    vector<int> a,v,t;
    cin>>n;
    for(int i=1;i<=n;i++) {
        cin>>A;
        a.push_back(A);
    }

    while(x) {

        for(unsigned i=0 ; i<a.size() ; i++) {
            if(x&a[i]) v.push_back(a[i]);
        }
        if(v.size()>=2) {
            t = a; t.clear();
            a = v; v = t;
        }
        else v.clear();
        x >>= 1;
    }
    cout<< (a[0]&a[1]);
    return 0;
}
