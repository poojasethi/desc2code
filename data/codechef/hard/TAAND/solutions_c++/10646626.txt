#include <iostream>
#include <cstdlib>
#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <map>

using namespace std;

#define MAXLOG 32
#define MAXN 3*100005
bool A[MAXN][MAXLOG], in[MAXN];
int num[MAXN];
int N;

int main()
{
	scanf("%d", &N);
	for (int i = 0; i < N; i++)
	{
		int a;
		scanf("%d", &a);
		num[i] = a;
		in[i] = 1;
		for (int j = 0; j < MAXLOG; j++)
		{
			A[i][j] = a & 1;
			a >>= 1;
		}
	}
	for (int j = MAXLOG - 1; j >= 0; j--)
	{
		int cnt = 0;
		for (int i = 0; i < N; i++)
			if (in[i] && A[i][j]) cnt++;
		if (cnt >= 2)
		{
			for (int i = 0; i < N; i++)
				if (in[i] && A[i][j]) in[i] = 1;
				else in[i] = 0;
		}
	}
	int a1, a2;
	int b1 = 0;
	for (int i = 0; i < N; i++)
	{
		if (in[i] && b1 == 0)
		{
			a1 = num[i];
			b1 = 1;
		}
		else if (in[i] && b1 == 1)
		{
			a2 = num[i];
			b1 = 2;
		}
		else if (b1 == 2) break;
	}
	printf("%d\n", a1&a2);
	return 0;
}
