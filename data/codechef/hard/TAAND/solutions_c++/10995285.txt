#include<bits/stdc++.h>
using namespace std;
#define pb push_back
int main(){
	long long n, i, j, k, val, sum=0, c;
	cin>>n;
	vector<long long> v, a;
	for(i=0; i<n; i++){
		cin>>k;
		a.pb(k);
	}
//	cout<<log2(1000000000)<<" "<<(1<<30)<<endl;
	for(i=30; i>=0; i--){
		c=0;
		for(j=0; j<n; j++){
			if((1<<i)&a[j]){
				c++;
				v.pb(a[j]);
			}
		}
	//	cout<<c<<endl;
		if(c<=1){
			v.clear();
		}
		else if(c==2){
			sum=v[0]&v[1];
			break;
		}
		else{
			n=v.size();
		//	sum+=(1<<i);
			a.clear();
			for(k=0; k<n; k++)a.pb(v[k]);
			v.clear();
		}
	}
	if(c!=2)sum=a[0]&a[1];
	cout<<sum<<endl;
}
