#include <bits/stdc++.h>
using namespace std;

int main()  {
    int n;
    cin >> n;
    int a[n],mark[n];//mark for marking the not required numbers
    for(int i = 0; i < n; i++) {
       cin >> a[i];
       }
       memset(mark,0,sizeof(mark));// marking all numbers 0
       int ans = 0;
      for(int i = 30; i >= 0; i--) { // iterarting overall bits
         int c=0;
         for(int j = 0; j < n ; j++) {
            if(mark[j] == 0 && (a[j] & (1 << i))) // if number is reqiured and its ith bit is set
              c++;
           }
         if(c >= 2) { // number with ith bit set are more than two as 1 & 1 == 1
            ans = ans + (1 << i);// including set bit value in answer
            for(int j = 0; j < n; j++) {
                if(mark[j] == 0 && (a[j] & (1 << i)) == 0) // marking the numbers whose ith bit is not set as they are not reqired in answer
                    mark[j]=1;
                }
            }
        }
      cout << ans << endl;
}
