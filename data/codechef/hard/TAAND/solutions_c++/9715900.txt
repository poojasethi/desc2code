#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int main(){
    
    int n;
    cin >> n;
    int a[n];
    int b[n];
    int temp;
    vector<int> v[32];
    for(int i = 0;i < n;i++){
        cin >> a[i];   
    }
    for(int i = 0;i < n;i++){
        temp = a[i];
        for(int j = 0;j < 32;j++){
            
            if(temp & 1) v[j].push_back(i);
            temp = temp/2;
        }
    }
   /* for(int i = 0;i < 32;i++){
        for(int j = 0;j < v[i].size();j++){
            cout << v[i][j] << " ";   
        }
        cout << endl;
    } */
    int maxcount = 0;
    int count = 0;
    for(int i = 0;i < n;i++) b[i] = 0;
    for(int i = 31;i >=0;i--){
        count = 0;
        if(v[i].size() < 2) continue;
        else{
            for(int j = 0;j < v[i].size();j++){
                if(b[v[i][j]] == maxcount){
                    b[v[i][j]]++;
                    count++;
                }
            }
            if(count >= 2) maxcount++; 
             for(int j = 0;j < v[i].size();j++){
                if(b[v[i][j]] > maxcount){
                    b[v[i][j]] = maxcount;
                }
            }
        }
    }
    vector<int> ans;
    //for(int i = 0;i < n;i++) cout << b[i] << " ";
    for(int i = 0;i < n;i++){
        if(b[i] == maxcount) ans.push_back(a[i]);
    }
    if(ans.size() < 2 || maxcount == 0) cout << 0 << endl;
    else{
        temp = ans[0]&ans[1];
        cout << temp << endl;
    }
    return 0;
}
