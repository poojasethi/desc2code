#include<iostream>
#include<algorithm>
#include<vector>
#define llu long long int
using namespace std;
struct tipvalue
{
    llu x;
    llu y;
    llu diff;
};
int cmp(tipvalue a,tipvalue b)
{
    return a.diff<b.diff;
}
int main()
{
   llu n,maxi,maxwithy,x,y;
   cin>>n>>x>>y;
   vector<tipvalue> tips(n);

   for(llu i=0;i<n;i++)
   {
       cin>>tips[i].x;
   }
   for(llu i=0;i<n;i++)
   {
       cin>>tips[i].y;
       tips[i].diff=abs(tips[i].y-tips[i].x);

   }
   maxi=0;
   sort(tips.rbegin(),tips.rend(),cmp);
   for(int i=0;i<n;i++)
   {
       if(tips[i].x>=tips[i].y&&x>0)
       {

               maxi+=tips[i].x;
               x--;
       }
       else if(tips[i].y>=tips[i].x&&y>0)
       {

               maxi+=tips[i].y;
               y--;

       }
       else if(x==0)
       {
            maxi+=tips[i].y;
            y--;
       }
       else
       {
            maxi+=tips[i].x;
               x--;
       }
   }
  cout<<maxi;
}
