#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountll
#define pb push_back
struct abc{
	int f,s,diff;
}z[100005];
bool cmp(abc a,abc b)
{
	return a.diff>b.diff;
}
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int n,x,y,i,j,k,ans;
    sd(n);
    sd(x);
    sd(y);
    for(i=0;i<n;i++)
    	sd(z[i].f);
    for(i=0;i<n;i++)
    {
    	sd(z[i].s);
    	z[i].diff=z[i].f-z[i].s;
    }
    sort(z,z+n,cmp);
    i=0;
    ans=0;
    while(i<n&&i<x&&z[i].diff>=0)
    {
    	ans+=z[i].f;
    	i++;
    }
    if(i==x)
    {
    	for(i;i<n;i++)
    		ans+=z[i].s;
    }
    else
    {
    	for(j=n-1;j>=max(i,n-y);j--)
    		ans+=z[j].s;
    	for(j;j>=i;j--)
    		ans+=z[j].f;
    }
    printf("%d\n",ans);
    return 0;
}
