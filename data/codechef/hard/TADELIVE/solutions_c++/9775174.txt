#include<bits/stdc++.h>
using namespace std;
int ans=0;
int n,x,y;
class pt{
	public:
	int a,b,d;
};
bool cmp(pt o,pt t){
	return o.d>t.d;
}
pt a[100010];
int main(){
	scanf("%d",&n);
	scanf("%d",&x);
	scanf("%d",&y);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i].a);
	}
	for(int i=0;i<n;i++){
		scanf("%d",&a[i].b);
		a[i].d = abs(a[i].a-a[i].b);
	}
	sort(a,a+n,cmp);
	int ao=0,bo=0;
	for(int i=0;i<n;i++){
		if(a[i].a>a[i].b){
			if(ao+1<=x){
				ans+=a[i].a;
				ao++;
			}else{
				ans+=a[i].b;
				bo++;
			}
		}else{
			if(bo+1<=y){
				ans+=a[i].b;
				bo++;
			}else{
				ans+=a[i].a;
				ao++;
			}
		}
	}
	printf("%d",ans);
	return 0;
}
