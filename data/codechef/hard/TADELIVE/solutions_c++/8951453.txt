#include <bits/stdc++.h>
#include <iostream>
#include <cstdio>
using namespace std;

class compare{
public:
	bool operator()(pair<int, int> a, pair<int, int> b){
		return a.first >= b.first; 
	}
};

bool handeled[100001];
int main(){
	int n, x, y;
	scanf("%d %d %d", &n, &x, &y);
	vector<int> tips_a;
	vector<int> tips_b;
	vector<pair<int, int> > diff;

	for (int i = 0; i < n; ++i){
		int tip_a;
		scanf("%d", &tip_a);
		tips_a.push_back(tip_a);
		handeled[i] = false;
	}

	for (int i = 0; i < n; ++i){
		int tip_b;
		scanf("%d", &tip_b);
		tips_b.push_back(tip_b);		
	}

	for (int i = 0; i < n; ++i){
		diff.push_back(make_pair(tips_a[i] - tips_b[i], i));
	}

	sort(diff.begin(), diff.end(), compare());
	int tempx = x;
	int tempy = y;
	int total = 0;
	for (int i = 0; i < diff.size() && x > 0 && diff[i].first > 0; ++i, --x){
		pair<int, int> order = diff[i];
		total += tips_a[order.second];
		handeled[order.second] = true;
	}

	for (int i = diff.size()-1; i >= 0 && y > 0 && diff[i].first < 0; --i, y--){
		pair<int, int> order = diff[i];
		total += tips_b[order.second];
		handeled[order.second] = true;
	}

	int a, any;
	if(diff[tempx-1].first > 0){
		a = 0; any = 0;
	}
	else if(diff[diff.size()-tempy].first < 0){
		a = 1; any = 0;
	}
	else{
		any = 1;
	}

	for (int i = 0; i < n; ++i){
		if(handeled[i] == false){
			if(any == 1){
				total+= tips_a[i];
			}
			else{
				if(a == 1){
					total += tips_a[i];
				}
				else{
					total += tips_b[i];
				}
			}
		}
	}

	printf("%d\n", total);
}
