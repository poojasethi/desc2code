#include<bits/stdc++.h>
using namespace std;
struct val
{

    int diff,in;
};
bool myfunction(val s1,val s2)
{
    return s1.diff>s2.diff;
}
int main()
{

    int n;
    cin>>n;
    int a[n];
    int b[n];
    int x,y;
    val d[n];
    cin>>x>>y;
    int i;
    for(i=0;i<n;i++)
    {

        cin>>a[i];

    }
    for(i=0;i<n;i++)
    {

        cin>>b[i];

    }
    for(i=0;i<n;i++)
    {
       d[i].diff=abs(a[i]-b[i]);
       d[i].in=i;
    }
    sort(d,d+n,myfunction);
  //  for(i=0;i<n;i++)
     //   cout<<d[i].in<<endl;
    int prox=0;
    int proy=0;
    int countx=1;
    int county=1;
    for(i=0;i<n;i++)
    {
       if(a[d[i].in]>b[d[i].in])
       {
          if(countx<=x)
          {
              prox=prox+a[d[i].in];
              countx++;
          }
          else
          {
             proy=proy+b[d[i].in];
             county++;
          }
       }
       else
       {
           if(county<=y)
           {

               proy=proy+b[d[i].in];
               county++;
           }
           else
           {
               prox=prox+a[d[i].in];
               countx++;
           }
       }


    }
cout<<prox+proy<<endl;
return 0;
}
