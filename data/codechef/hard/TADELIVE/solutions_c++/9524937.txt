#include<bits/stdc++.h>
using namespace std;
#define pb push_back
#define mp make_pair
#define ll long long
int main()
{
	int n,x,y;
	cin>>n>>x>>y;
	int a[n],b[n];
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
		cin>>b[i];
	vector < pair<int,int> > rec;
	for(int i=0;i<n;i++)
		rec.pb(mp(abs(a[i]-b[i]),i));
	sort(rec.begin(),rec.end());
	reverse(rec.begin(),rec.end());
	ll ans=0;
	for(int i=0;i<n;i++)
	{
		int ind=rec[i].second;
		if(a[ind]>b[ind])
		{
			if(x)
			{
				ans+=a[ind];
				x--;
			}
			else
			{
				ans+=b[ind];
				y--;
			}
		}
		else
		{
			if(y)
			{
				ans+=b[ind];
				y--;
			}
			else
			{
				ans+=a[ind];
				x--;
			}
		}
	}
	cout<<ans;
	return 0;
}
