int max(int p,int q)
{
    return (p>q)?p:q;
}
int min(int p,int q)
{
    return (p<q)?p:q;
}

int main()
{
    int n,count1,count2,x,y,i,j,ans[100005],sum,a[100005],b[100005];
    VII c;
    SET(ans,0);
    si(n);
    si(x);
    si(y);
    sum=0;
    for(i=0;i<n;i++)
    {
        si(a[i]);
    }
    for(i=0;i<n;i++)
    {
        si(b[i]);
    }
    for(i=0;i<n;i++)
    {
        c.PB(MP(a[i]-b[i],i));
    }
    sort(ALL(c));
    count1=0;
    count2=0;
    for(i=0;i<n;i++)
    {
        if(c[i].F<0)
        {
            count1++;
        }
        if(c[i].F==0)
        {
            count2++;
        }
        if(c[i].F>0)
        {
            break;
        }
    }
    for(i=n-1;i>=0;i--)
    {
        if(x>0 && c[i].F>0)
        {
            ans[c[i].S]=a[c[i].S];
            x--;
        }
    }
    for(i=0;i<n;i++)
    {
        if(y>0 && c[i].F<0)
        {
            ans[c[i].S]=b[c[i].S];
            y--;
        }
    }
    for(i=0;i<n;i++)
    {
        if(x>0 && ans[c[i].S]==0)
        {
            ans[c[i].S]=a[c[i].S];
            x--;
        }
    }
    for(i=0;i<n;i++)
    {
        if(y>0 && ans[c[i].S]==0)
        {
            ans[c[i].S]=b[c[i].S];
            y--;
        }
    }
    sum=0;
    for(i=0;i<n;i++)
    {
        sum+=ans[i];
    }
    dout(sum);
    return 0;
}
