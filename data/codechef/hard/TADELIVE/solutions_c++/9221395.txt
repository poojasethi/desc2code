#include <bits/stdc++.h>

using namespace std;

pair< pair<int, int>, int > array[100000];

bool function(const pair< pair<int, int>, int> &a, const pair< pair<int, int>, int> &b) {
	return abs(a.first.first - a.first.second) > abs(b.first.first - b.first.second);
}

int main() {
	int n, x, y;
	cin >> n >> x >> y;
	for(int i = 0; i < n; i ++) {
		cin >> array[i].first.first;
	}
	for(int i = 0; i < n; i ++) {
		cin >> array[i].first.second;
		if(array[i].first.second > array[i].first.first) {
			array[i].second = 2;
		} else {
			array[i].second = 1;
		}
	}
	sort(array, array + n, function);
	long long answer = 0;
	for(int i = 0; i < n; i ++) {
		if(array[i].second == 1) {
			if(x > 0) {
				x --;
				answer = answer + abs(array[i].first.first);
			} else {
				y --;
				answer = answer + abs(array[i].first.second);
			}
		} else {
			if(y > 0) {
				y --;
				answer = answer + abs(array[i].first.second);
			} else {
				x --;
				answer = answer + abs(array[i].first.first);
			}
		}
	}
	cout << answer;
	return 0;
}
