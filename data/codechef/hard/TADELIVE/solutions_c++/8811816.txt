#include<bits/stdc++.h>
using namespace std;
struct node{
	long long int a,b,diff;
};
bool cmp(node x,node y){
	return x.diff>y.diff;
}
int main(){
	int n,x,y;
	cin>>n>>x>>y;
	vector<node> v(n);
	for(int i=0;i<n;i++){
		cin>>v[i].a;
	}
	for(int i=0;i<n;i++){
		cin>>v[i].b;
		v[i].diff=abs(v[i].b-v[i].a);
	}
	sort(v.begin(),v.end(),cmp);
	long long int ans=0;
	int cnt=0,cnt1=0;
	for(int i=0;i<n;i++){
		if(v[i].a>v[i].b){
			if(cnt<x){
				cnt++;
				ans+=v[i].a;
			}
			else{
				cnt1++;
				ans+=v[i].b;
			}
		}
		else{
			if(cnt1<y){
				cnt1++;
				ans+=v[i].b;
			}
			else{
				cnt++;
				ans+=v[i].a;
			}
		}
	}
	cout<<ans<<"\n";
}
