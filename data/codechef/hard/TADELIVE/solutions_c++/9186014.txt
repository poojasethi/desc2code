#include<bits/stdc++.h>
using namespace std;
pair<int, int> P;
vector<pair<int, int> >V;
vector<pair<int, int> >:: iterator it;

int main()
{
    int N, X, Y;
    cin>>N>>X>>Y;
    int A[N], B[N], diff[N];
    for(int i=0; i<N; i++)
        cin>>A[i];
    for(int i=0; i<N; i++)
        cin>>B[i];
    for(int i=0; i<N; i++)
    {
        diff[i]=A[i]-B[i];
        P=make_pair(abs(diff[i]), i);
        V.push_back(P);
    }
    sort(V.begin(), V.end());
    int tip=0;
    for(it=V.end()-1; it>=V.begin(); it--)
    {
        if(A[it->second]>=B[it->second] && X>0)
        {
            tip+=(A[it->second]);
            X--;
        }
        else if(A[it->second]<=B[it->second] && Y>0)
        {
            tip+=(B[it->second]);
            Y--;
        }
        else if (X>0)
        {
            tip+=A[it->second];
            X--;
        }
        else if (Y>0)
        {
            tip+=B[it->second];
            Y--;
        }
    }
    cout<<tip<<endl;
    return 0;
}
