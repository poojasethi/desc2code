#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define ld long double
#define sl(x) scanf("%lld",&x)
#define pl(x) printf("%lld",x)
#define sld(x) scanf("%Lf",&x)
#define pld(x) printf("%0.0Lf",x)
#define pline() printf("\n")
#define mod 1000000007

struct node{
    ll diff;
    ll pos;
};

bool mycomp(struct node rhs,struct node lhs){
    return rhs.diff>lhs.diff;
}

int main(){
    ll n,x,y,i,sa=0,sb=0,ans=0,p,lost,a[100004],b[100004];
    struct node d[100004];
    sl(n);sl(x);sl(y);
    for(i=0;i<n;i++){
        sl(a[i]);
    }
    for(i=0;i<n;i++){
        sl(b[i]);
    }
    for(i=0;i<n;i++){
        d[i].diff=abs(a[i]-b[i]);
        d[i].pos=i;
    }
    sort(d,d+n,mycomp);
    
    for(i=0;i<n;i++){
        p=d[i].pos;
        if(a[p]>=b[p]){
            if(sa+1<=x){
                sa++;
                ans+=a[p];
            }
            else
            {
                sb++;
                ans+=b[p];
            }
        }
        else{
            if(sb+1<=y){
                sb++;
                ans+=b[p];
            }
            else
            {
                sa++;
                ans+=a[p];
            }
        }
    }
    pl(ans);
    pline();
	return 0;
}
