#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
 
#define max 100009
 
int a[max],b[max];
int c[max];
	
 
int main() {
	int n,x,y;
	scanf("%d %d %d",&n,&x,&y);
	
	for(int i=0;i<n;i++)
	scanf("%d",&a[i]);
	
	for(int i=0;i<n;i++)
	scanf("%d",&b[i]);
	
	long long sum = 0;
	
	int p,q,r;
	p=q=r=0;
	
	for(int i=0;i<n;i++)
	
	if(a[i]>b[i]){
	sum+=a[i];p++;
	}
	else if(b[i]>a[i]){
	sum+=b[i];q++;
	}
	else{
	sum+=a[i];r++;
	}
	
	int d = 0;
	if(p>x){
	d = p-x;
	}
 
	if(q>y){
	d = q-y;
	}
	
	
	if(d){
	int j=0;
	if(p>x){
		for(int i=0;i<n;i++){
			if(a[i]>b[i]){
			c[j++] = a[i]-b[i];
			}
		}
	}
 
	if(q>y){
		for(int i=0;i<n;i++){
			if(a[i]<b[i]){
			c[j++] = b[i]-a[i];
			}
		}
	}
	
	sort(c,c+j);
	
	while(d--){
	sum -= c[d];
	}
	}
		printf("%lld",sum);
 
	return 0;
}
