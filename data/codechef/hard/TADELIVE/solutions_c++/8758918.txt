#include <bits/stdc++.h>
using namespace std;
 
#define gc getchar_unlocked
#define for1(i,m,n) for(i=m;i<n;i++)
#define for2(i,m,n) for(i=m;i>n;i--)
 
inline void get(int *p){register int n=0,a=gc();while(a<'0'||a>'9')a=gc();while(a>='0' && a<='9'){n=(n<<1)+(n<<3)+a-'0';a=gc();}*p=n;}
 
int main()
{
	int i=0,j=0,n,x,y,sum=0,_sum=0,null=0,am=0,bm=0,cm=0;
	get(&n);get(&x);get(&y);
	int a[n],b[n],c[n];
	
	for1(i,0,n)		get(a+i);
	
	for1(i,0,n)
	{
		get(b+i);
		if(a[i]==b[i])
		{
			sum+=a[i];
			++null;
		}
		if(a[i]>b[i])
		{
			sum+=a[i];
			++am;
		}
		if(a[i]<b[i])
		{
			sum+=b[i];
			++bm;
		}
	}
	
	if(am>x)	cm=am-x-null;
	if(bm>y)	cm=bm-y-null;
	if(cm)
	{
		if(am>x)
		{
			for1(i,0,n)
				if(a[i]>b[i])
					c[j++]=a[i]-b[i];
		}
		if(bm>y)
		{
			for1(i,0,n)
				if(b[i]>a[i])
					c[j++]=b[i]-a[i];
		}
		sort(c,c+j);
		while(cm--)
			_sum+=c[cm];
	}
	
	printf("%d",sum-_sum);
	
	return 0;
}
