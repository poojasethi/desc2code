#include<cstdio>
#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
typedef struct
{
    long long int a;
    long long int b;
}p;
bool cc(p c,p d)
{
  return abs(c.a-c.b)>abs(d.a-d.b);

}
int main()
{
    int n,i;
  long long int res=0,ac=0,ab=0,x,y;
      scanf("%d %lld %lld",&n,&x,&y);

      p s[n];
    for(i=0;i<n;i++)
        scanf("%lld",&s[i].a);
     for(i=0;i<n;i++)
        scanf("%lld",&s[i].b);
    sort(s,s+n,cc);
    for(i=0;i<n;i++)
    {
        if(s[i].a>s[i].b)
        {
            if(ac<x)
            {
                ac++;
                res+=s[i].a;
            }
            else
            {
                ab++;
                res+=s[i].b;

            }
        }
        else
        {
             if(ab<y)
            {
                ab++;
                res+=s[i].b;
            }
            else
            {
                ac++;
                res+=s[i].a;

            }

        }
    }
    printf("%lld\n",res);



}
