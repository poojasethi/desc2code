#include <bits/stdc++.h>
#define FOR(i,a,b) for(int i=a; i<=b; i++)
#define FRO(i,a,b) for(int i=a; i>=b; i--)
#define filein(t) freopen(t, "r", stdin)
#define fileout(t) freopen(t, "w", stdout)
#define ll long long
#define mp make_pair
#define f first
#define s second
#define N 1000007
#define BIT(i, n) ( (n >> i) & 1 )
#define ln 1000000009
#define oo 300003

using namespace std;

int n, x, y, ans, C[N], Ma = -ln, u, v, A[N];

int main()
{
    ios_base::sync_with_stdio(0);
    //filein("inp.txt");  fileout("out.txt");
    cin >> n >> x >> y;
    y = n - y;
    for(int i=1; i<=n; i++)
        cin >> A[i];
    for(int i=1; i<=n; i++)
    {
        cin >> v;
        ans += v;
        C[i] = A[i] - v;
    }
    sort(C+1, C+n+1);
    reverse(C+1,C+n+1);
    for(int i=1; i<=n; i++)
        C[i] += C[i-1];
    for(int i=y; i<=x; i++)
        Ma = max(Ma, C[i]);
    cout << ans + Ma;
}
