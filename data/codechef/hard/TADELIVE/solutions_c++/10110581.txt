#include<bits/stdc++.h>
#include <iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

const int MAXN = 100000;
int n, x, y;
int a[MAXN], b[MAXN];

void Init() {
  assert(scanf("%d%d%d", &n, &x, &y) == 3);
  for (int i = 0; i < n; ++i)
   (scanf("%d", a + i) == 1);
  for (int i = 0; i < n; ++i)
    (scanf("%d", b + i) == 1);
}
int dp[100005];
vector<pair <int,int> > v;
void Work() {
  int sum = 0;
  v.clear();
  memset(dp,0,sizeof(dp));
  for (int i = 0; i < n; ++i)
  v.push_back(make_pair(abs(a[i]-b[i]),i));
 
  sort(v.begin(),v.end(),greater< pair <int,int> >());
  
  for (int i = 0; i <n; ++i) {

        int t=v[i].second;
    	if(a[t]>b[t])
    	{
	    	if(x)
	    	{
	    		sum+=a[t];
	    		x--;
	    	}
	    	else
	    	{
	    		sum+=b[t];
	    		y--;
	    	}
	    }
	    else
	    {
	    	if(y)
	    	{
	    		sum+=b[t];
	    		y--;
	    	}
	    	else
	    	{
	    		sum+=a[t];
	    		x--;
	    	}
	    }
  }
 
  printf("%d\n", sum);
}

int main() {
  int cases = 1;
  while (cases--) {
    Init();
    Work();
  }
  return 0;
}
