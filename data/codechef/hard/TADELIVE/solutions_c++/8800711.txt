#include <bits/stdc++.h>
#define maxn 100005
#define ft first
#define sc second

using namespace std;
typedef pair<int,int> II;

int n, a[maxn], b[maxn], X, Y, ans;
II c[maxn];

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    ios::sync_with_stdio(false);
    cin >> n >> X >> Y;
    for(int i=1;i<=n;i++) cin >> a[i];
    for(int i=1;i<=n;i++) cin >> b[i];
    for(int i=1;i<=n;i++) c[i]=II(abs(a[i]-b[i]),i);
    sort(c+1,c+n+1); reverse(c+1,c+n+1);
    int ans=0;
    for(int i=1;i<=n;i++) {
        int u=a[c[i].sc], v=b[c[i].sc];
        if (u>=v && X>0) ans+=u, X--; else ans+=v, Y--;
    }
    cout << ans;
}
