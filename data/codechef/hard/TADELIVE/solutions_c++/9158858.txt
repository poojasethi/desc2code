#include<iostream>
#include<vector>
#include<algorithm>
#include<cstdio>
using namespace std;
vector<int> G[2];
bool myfuction(pair<int,int> p1,pair<int,int> p2)
{
	int diff1=abs(p1.first -p1.second);
	int diff2=abs(p2.first-p2.second);
	return diff1>diff2;
}
int main()
{
	int n,X,Y;
	cin>>n>>X>>Y;
	int tip;
	for(int i=0;i<n;i++)
	{
		cin>>tip;
		G[0].push_back(tip);
	}
	for(int i=0;i<n;i++)
	{
		cin>>tip;
		G[1].push_back(tip);
	}
	vector< pair<int,int> > v;
	for(int i=0;i<n;i++)
	{
		v.push_back(make_pair(G[0][i],G[1][i]));
	}
	sort(v.begin(),v.end(),myfuction);
	int andy=0;
	int bob=0;
	int total_tip=0;
	for(int i=0;i<n;i++)
	{
		if(v[i].first>v[i].second)
		{
			if(andy<X)
			{
				andy++;
				total_tip=total_tip+v[i].first;
			}
			else
			{
				bob++;
				total_tip=total_tip+v[i].second;
			}
		}
		else
		{
			if(bob<Y)
			{
				bob++;
				total_tip=total_tip+v[i].second;
			}
			else
			{
				andy++;
				total_tip=total_tip+v[i].first;
			}
		}
	}
	cout<<total_tip<<endl;
}
