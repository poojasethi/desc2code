#include <iostream>
#include <vector>
#include <bitset>
#include <algorithm>
using namespace std;
#define min(x,y) (x<y) ? (x):(y)
int a[100000],b[100000];
typedef pair<int, int> P;
std::vector<P> diff;
bitset<100000> vis;
int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	int n,x,y;
	cin>>n>>x>>y;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
		cin>>b[i];

	for(int i=0;i<n;i++)
		if(a[i]>b[i])
			diff.push_back(P((a[i]-b[i]),i));

	int sz=diff.size();
	sort(diff.begin(), diff.end());
	reverse(diff.begin(), diff.end());

	int mn=min(sz,x);
	int ans=0;
	for(int i=0;i<mn;i++)
		ans+=a[diff[i].second],vis[diff[i].second]=1;
	
	for(int i=0;i<n;i++)
		if(!vis[i])
			ans+=b[i];
	cout<<ans;
	return 0;
}
