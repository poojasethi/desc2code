#include<iostream>
#include<cmath>
#include<algorithm>
#include<stdio.h>
using namespace std;
int main()
{
	int T,d,a1,a2,p1,p2,q1,q2,a,p,q;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d.%d%d.%d%d.%d%d",&a1,&a2,&p1,&p2,&q1,&q2,&d);
		a=a1*100+a2;
		p=p1*100+p2;
		q=q1*100+q2;
		int num=a*(p*(100-a)+q*100),den=(100-a)*(100-a)*100;
		printf("%d.",num/den);
		num%=den;
		while(d--)
		{
			num*=10;
			printf("%d",num/den);
			num%=den;			
		}
		printf("\n");
	}
	return 0;
}
