#include<iostream>
#include<cmath>
#include<algorithm>
#include<stdio.h>
using namespace std;
int main()
{
	int T,d;
	//std::ios::sync_with_stdio(false);
	double a,p,q;
	scanf("%d",&T);
	while(T--)
	{
		cin>>a>>p>>q>>d;
		float nm=a*(p+q-p*a)*1000000,dn=(1-a)*(1-a)*1000000;
		int num=(int)nm,den=(int)dn;
		printf("%d.",num/den);
		num%=den;
		while(d--)
		{
			num*=10;
			printf("%d",num/den);
			num%=den;
		}
		printf("\n");
	}
	return 0;
}
