{

    ll test;s(test);
    while(test--)
    {
        double a,p,q;
        ll D;
        cin>>a>>p>>q;s(D);

        ll ia=(100.00*a+0.1);
        ll ip=(100.00*p+0.1);
        ll iq=(100.00*q+0.1);
        ll up=ip*ia*(100-ia)+iq*ia*100;
        ll down=(100-ia)*(100-ia)*100;

        ll g=__gcd(up,down);
        up/=g;down/=g;
        ll first=up/down;
        ll len=0;
        ll r;
        ll rem=up%down;
        printf("%lld.",first);
        for(ll i=0;i<D;i++)
        {
            rem*=10;
            ll q=rem/down;
            rem=rem%down;
            printf("%lld",q);
            len++;
        }
        printf("\n");
    }
    return 0;

}
