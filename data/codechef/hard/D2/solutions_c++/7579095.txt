#include<bits/stdc++.h>

using namespace std;

#define inf (1<<29)

int n;
int Sequence[20000+5];

int I[10000+5];


int LisNlogK(int lo,int n)
{
    I[0] = -inf;

    for(int i=1; i<=n; i++)
        I[i] = inf;

    int LisLength=0;

    for(int i=lo; i<lo+n; i++)
    {
        int low = 0,high = LisLength,mid;

        while(low<=high)
        {
            mid = (low+high)/2;

            if(I[mid]<Sequence[i])
                low = mid+1;
            else
                high = mid-1;
        }

        I[low] = Sequence[i];


        if(LisLength<low)
            LisLength = low;
    }



    return LisLength;;

}

int main()
{
    int test;

    scanf("%d",&test);

    while(test--)
    {
        scanf("%d",&n);

        for(int i=0; i<n; i++)
        {
            scanf("%d",&Sequence[i]);
            Sequence[i+n]=Sequence[i];
        }

        int best=1;
        for(int i=0; i<n; i++)
        {
            best=max(best,LisNlogK(i,n));
        }

        printf("%d\n",best);

    }
    return 0;
}
