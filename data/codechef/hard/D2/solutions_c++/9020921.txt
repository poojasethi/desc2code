int lis(int l,int r)
{
	vector<int> st;
	for(int i=l;i<r;i++)
	{
		int num=b[i];
		auto it=lower_bound(st.begin(),st.end(),num);
		if(it==st.end())
		 	st.push_back(num);
		else
			*it=num;
	}
    return st.size();
}
			void solve()
			{
				int maxn=0;
				for(int i=0;i<n;i++)
				maxn=max(maxn,lis(i,i+n));

				printf("%d\n",maxn);
			}
