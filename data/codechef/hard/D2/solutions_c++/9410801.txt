typedef pair<int,int> PII;
typedef vector<int> VI;
typedef vector< PII > VPII;

#define MAXN 20010
int main()
{
    ios::sync_with_stdio(false);
    test()
    {
        int n,i,j;
        cin>>n;
        vector<int> a(2*n);
        vector<pair<int,int > > p(n);
        rep(i,n)
        {
            cin>>a[i];
            a[i+n]=a[i];
            p[i].fi=a[i];
            p[i].se=i;
        }    
        sort(all(p));
        int ans=0;
        for(i=0;i<n;i++)
        {
            VI v;
            int ind=p[i].se;
            int cur=1;
            v.pb(a[ind]);
            for(j=ind+1;j<ind+n;j++)
            {
                if(a[j]<v[0])
                {
                    v[0]=a[ind];
                }
                else if(a[j]>v[sz(v)-1])
                {
                    ++cur;
                    v.pb(a[j]);
                }
                else
                {
                    auto it=upper_bound(all(v),a[j]);
                    *it=a[j];
                }
            }
            ans=max(ans,sz(v));
            v.clear();
        }
        cout<<ans<<el;
    }
    return 0;
}
