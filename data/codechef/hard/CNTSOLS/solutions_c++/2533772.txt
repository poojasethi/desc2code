# define mod 1000000007
using namespace std;
typedef long long ll;
typedef unsigned long long ull;
ll u,d;
int m,n;
int func(int a)
{
    ll temp=d,temp1=a;
    int ans=1;
    if(!d)return 1;
    while(temp)
    {
        if(temp&1)ans=(ans*temp1)%n;
        temp1=(temp1*temp1)%n;
        temp>>=1;
    }
    return ans;
}
int main()
{
	int t;
	int t1,t2,t3;
	scanf("%d",&t);
	ll fans=0;
	bool is1[50];
	while(t--)
	{
	    fans=0;
	    scanf("%lld%lld%d%d",&u,&d,&m,&n);
	    for(int i=0;i<=40;i++)is1[i]=0;
	    for(int i=0;i<n&&i<=u;i++)
        {
            t1=func(i);
            for(int j=0;j<n&&j<=u;j++)
            {
                t2=func(j);
                for(int k=0;k<n&&k<=u;k++)
                {
                    t3=func(k);
                    if((t1+t2+t3)%n==m)
                    {
//                        cout<<((u-i)/n)+1<<" "<<(((u-j)/n)+1)<<" "<<(((u-k)/n)+1);
                        fans=(fans+((((u-i)/n)+1)%mod*(((u-j)/n)+1)%mod*(((u-k)/n)+1)%mod)%mod)%mod;
                    }
                }
            }
        }
        printf("%lld\n",fans);
	}
	return 0;
}
