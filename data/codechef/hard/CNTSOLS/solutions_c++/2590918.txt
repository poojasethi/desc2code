#include<iostream>
#include<cmath>
using namespace std;
#define mod2 1000000007
typedef long long int lli;
int modexpo(lli a, int b,int mod)
{
    lli result=1;
    while(b)
    {
        if(b&1)
        {
            result*= a;
            while(result>=mod)
            result-=mod;
        }
        b=b>>1;
        a*= a;
        while(a>=mod)
        a-=mod;
    }
    return (int)result;
}
lli countsol(int x, int y, int z, int upper, int n)
{
    lli ans=0, save=upper/n;
    int save2=upper%n;

    ans+=save+(x<=save2?1:0);
    ans%=mod2;

    ans*=save+(y<=save2?1:0);
    ans%=mod2;

    ans*=save+(z<=save2?1:0);
    ans%=mod2;
    return ans;
}
int main()
{
    int arr[41],t,i,x,y,z;
    lli upper,d,m,n,ans;

    cin>>t;
    while(t--)
    {
        cin>>upper>>d>>m>>n;

        for(i=0;i<n;i++)
            arr[i]=modexpo(i,d,n);

        ans=0;
        for(x=0;x<n;x++)
            for(y=0;y<n;y++)
                for(z=0;z<n;z++)
                {
                    if((arr[x]+arr[y]+arr[z])%n==m)
                    {
                        ans+=countsol(x,y,z,upper,n);
                        ans%=mod2;
                    }
                }
        cout<<ans<<endl;
    }
}
