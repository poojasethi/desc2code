{

}

ll process(int i,int j,int k)
{
	if(u<=L)
		return 1;
	ll a=soln(i);
	ll c=soln(j);
	ll b=soln(k);
	ll ans=a%K;
	ans=(ans*b)%K;
	ans=(ans*c)%K;
	//printf("%d %d %d\n",i,j,k);
	if(u>L)
	{
		//printf("%lld\n",((((a%K)*(b%K))%K)*(c%K))%K);
		return ans;
	}
}
 
ll f()
{
	ll ans=0LL;
	ll z,l;
	if(u>L)
		l=L;
	else
		l=u+1;
	FOR(i,0,l)
	{
		FOR(j,0,l)
		{
			FOR(k,0,l)
			{
				z=(power(i,d)) + (power(j,d))+ (power(k,d));
				//printf("%lld\n",z);
				if(z%L==m)
				{
					//printf("%d %d %d\n",i,j,k);
					//printf("%lld\n",ans);
					ans=(ans%K + process(i,j,k)%K)%K;
				}
			}
		}
	}
	return ans;
}
 
ll power(ll k,ll p)
{
	ll x=1;
	ll y=k;
	while(p!=0)
	{
		if(p%2==1)
			x=(x*y)%L;
		y=(y*y)%L;
		p=p/2;
	}
	x%=L;
	return x;
}

void out()

	printf("%lld\n",f());
}
