#include<iostream>
using namespace std;
long long int BIGMOD(long long int a,long long int b,long long int m)
{
	if(b == 0) return 1%m;
	
	long long int x = BIGMOD(a, b/2, m);
	x = (x*x)%m;
	
	if(b&1) x = (x*a)%m;
	
	return x;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long int u,d,m,n,mod=1000000007,i,j,k,x;
		cin>>u>>d>>m>>n;
		long long int cnt[50]={0};
		for(i=0;i<n;i++)
		{
			x = BIGMOD(i, d, n);
			if(i <= u)
			cnt[x]+=(u - i)/n + 1;
		}
		long long int ans=0;
		for(i=0;i<n;i++)
		{
			for(j=0;j<n;j++)
			{
				for(k=0;k<n;k++)
				{
					if((i+j+k)%n==m%n)
					ans+=(cnt[i]*((cnt[j]*cnt[k])%mod))%mod;
					ans=ans%mod;
				}
			}
		}
		cout<<ans<<"\n";
	}
	return 0;
}
