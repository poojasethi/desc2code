#include<iostream>
#include<string>
using namespace std;

#define FOR(i,a,b) for(int i=a;i<=b;++i)

typedef long long Int;
const int MODULO = 1000000007;

int binpw(int x,int times,int md)
{
    int ans=1;
    while(times)
    {
        if(times&1)ans=(ans*(Int)x)%md;
        x=(x*(Int)x)%md;
        times/=2;
    }
    return ans;
}

int has_mod[111];
void solve()
{
    int tests;
    cin>>tests;
    while(tests--)
    {
        int up,N,d,m;
        cin>>up>>d>>m>>N;
        FOR(i,0,N-1)has_mod[i] = (up/N)+(up%N>=i);
        Int ans=0;
        FOR(md1,0,N-1)FOR(md2,0,N-1)FOR(md3,0,N-1)
        {
            Int cur=0;
            cur+=binpw(md1,d,N);
            cur+=binpw(md2,d,N);
            cur+=binpw(md3,d,N);

            if(cur%N==m)
            {
                int tmp=has_mod[md1];
                tmp=(tmp*(Int)has_mod[md2])%MODULO;
                tmp=(tmp*(Int)has_mod[md3])%MODULO;
                ans=(ans+tmp)%MODULO;
            }
        }
        cout<<ans<<endl;
    }
}

int main()
{
    solve();
}
