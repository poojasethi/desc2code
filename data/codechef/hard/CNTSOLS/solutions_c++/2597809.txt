typedef long double LD;
typedef long long LL;

#define sz(A) (int)(A).size()

#define pb push_back

const int MOD = int(1e9) + 7;

int upper, N;

LL cnt(int a) {
	return (upper - a) / N + 1;
}

LL power(LL a, int deg) {
	if (!deg)
		return 1 % N;
	if (deg % 2)
		return (a * power(a, deg - 1)) % N;			
	return power((a * a) % N, deg / 2);
}

int main() {
	int t;
	cin >> t;
	for (int i = 0; i < t; i++) {
		int d, m;
		LL res = 0;
		cin >> upper >> d >> m >> N;
		for (int i = 0; i < min(upper + 1, N); i++)
			for (int j = 0; j < min(upper + 1, N); j++)
				for (int k = 0; k < min(upper + 1, N); k++)
					if ((power(i, d) + power(j, d) + power(k, d)) % N == m) {
						res += (((cnt(i) * cnt(j)) % MOD) * cnt(k)) % MOD;
						res %= MOD;
					}					
		cout << res << endl;
	}
	return 0;
}
