#include<iostream>
#include<string.h>
#include<map>
#include<assert.h>
#define MOD (30041975)
#define MAXN (2002)
	using namespace std;
	int E[MAXN][MAXN];
	int A[MAXN][MAXN];
		void calcE()
	{
				int i,j;
				for(i=0;i<MAXN;i++)
			{	
				E[0][i]=0;
				E[1][i]=0;
				E[i][0]=0;
			}		
				E[2][1]=2;
				for(i=3;i<MAXN;i++)
				for(j=1;j<i;j++)
					{
						E[i][j]=((long long)j*E[i-1][j])%MOD+(2*E[i-1][j-1])%MOD;
						E[i][j]%=MOD;
							if(j>=2)
						E[i][j]+=((long long)(i-j)*E[i-1][j-2])%MOD;
						E[i][j]%=MOD;
					}		
	}	
		long long fact(int N)
			{
				int res=1;
				int i;
				for(i=1;i<=N;i++)
				res=((long long)res*i)%MOD;
return res;
	}
		void calcA()
				{
					int i,j;
					for(i=3;i<MAXN;i++)
					A[i][0]=(fact(i))%MOD;	
					for(i=3;i<MAXN;i++)
					A[i][1]=(fact(i)-2+MOD)%MOD;
					for(i=3;i<MAXN;i++)
					for(j=2;i-j>=2;j++)
					A[i][j]=(A[i][j-1]-E[i][j]+MOD)%MOD;
				}	
int main()
	{
		memset(E,0,sizeof(E));
		memset(A,0,sizeof(A));
			calcE();
			calcA();
int i,j;
	for(i=0;i<MAXN;i++)
	for(j=0;j<MAXN;j++)
		if(A[i][j]<0)
		cout<<"FLAG"<<endl;
			for(i=0;i<MAXN;i++)
			for(j=0;j<MAXN;j++)
				if(E[i][j]<0)
				cout<<"FLAG"<<endl;
int T;
	cin>>T;
	while(T--)
{
	int N,K;
	cin>>N>>K;
	cout<<A[N][K]<<endl;
}
return 0;
}
