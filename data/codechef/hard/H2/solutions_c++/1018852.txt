#define FOR(a, b) for (int i = a; i < b; i++)
#define ALL(a) (a.begin(), a.end())
#define SORT(a) sort(ALL(a))
#define SORT_ARR(a, b) sort (a, b)
using namespace std;
int wt[100000], st[100000];
double d[100000];

void distToPoint(double s, int n) {
    FOR(0, n) {
        d[i] = (double)st[i] - s * (double)wt[i];
    }
}

bool possibleSpeed (double s, int n) {
    distToPoint(s, n);
    SORT_ARR(d, d + n);
    FOR (0, n/2 + 1) {
        if (d[i] + d[n - i - 1] < 0)
            return false;
    }
    return true;
}

int main() {
    int n, i, j;
    scanf("%d", &n);
    for (i = 0; i < n; i++) {
        scanf("%d %d", wt + i, st + i);
        wt[i] += 10;
    }

    double high = (100.0 + 100.0) / (60.0 + 60.0);
    double low = (50.0 + 50.0) / (110.0 + 110.0);

    double mid = 0;
    while (1e-8 <= high - low) {
        mid = (low + high) / 2.0;
        if (possibleSpeed(mid, n) == true) {
            low = mid;
        }
        else {
            high = mid;
        }
    }
    printf("%.6lf", mid);
//    getch();
    return 0;
}
