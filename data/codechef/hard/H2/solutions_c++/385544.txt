typedef pair<int,int> pii;
vector <pii> V;
double tmp[ 100001 ];
int n;
bool pos( double val )
{
	int i;
	for(i = 0; i < n ; i ++ ) tmp[ i ] =( ( double ) V[ i ].second - val * V[ i ].first );
	sort( tmp , tmp + n );
	for(i = 0; i < n / 2  ; i ++ ) if( tmp[ i ] + tmp[ n - i - 1 ] < 0 ) return false;
	return true;
}
int main()
{
	while( cin >> n )
	{
		V.clear();
		int a, b, i;
		for(i = 0; i < n ; i ++ ){
			scanf("%d %d",&a,&b);
			V.push_back( pii( a + 10, b) );
		}
		double lo = .4, hi = 1.7, mid;
		while( lo + 1e-8 < hi )
		{
			mid = ( lo + hi ) / 2;
			if( pos ( mid ) ) lo = mid;
			else hi = mid;
		}
		mid = ( lo + hi ) / 2;
		printf("%f\n",mid );		
	}
	return 0;
}
