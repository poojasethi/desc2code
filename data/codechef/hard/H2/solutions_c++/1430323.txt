#include <iostream>
#include<cmath>
#include<algorithm>
#include<cstdio>
 
using namespace std;
 
#define MAX 100000
 
int person[MAX][2];
double dist[MAX];
int num;
 
bool possible(double slope )
{
     double denom = sqrt( 1 + slope*slope);
     int i;
     for( i=0; i<num; i++ )
          dist[i] = (((double)person[i][1]) - slope*( (double)person[i][0]) )/denom;
 
     sort( dist, dist+num );
 
     for( i=0; i<num/2; i++ ) {
          if( dist[i] + dist[num-1-i] < 0 )
              return false;
     }
 
     return true;
}
 
int main()
{
    int i;
    scanf( "%d", &num );
 
    for( i=0; i<num; i++ ){
         scanf( "%d %d", &person[i][0], &person[i][1] );
         person[i][0] += 10;
    }
 
    double high = 1.7, low = 0.4, mid;
    while( high - low > 1e-8 ){
           mid = (high + low)/2;
           if(possible(mid))
               low = mid;
           else
               high = mid;
    }
 
    printf( "%f\n", (low+mid)/2 );

    return 0;
}
