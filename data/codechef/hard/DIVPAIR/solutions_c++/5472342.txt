#include<iostream>
#include<stdio.h>

using namespace std;

int main()
{
    long long t;
    scanf("%lld",&t);
    while(t--)
    {
        long long x,y,n,m,ans;
        scanf("%lld",&n);
        scanf("%lld",&m);

        x=n/m;
        y=n%m;
		ans=0;

		ans += x*x*(m-1);
		ans += x*(x-1);
		if(m%2==0) ans-=x;

		ans/=2;
		ans+=x*y;
		if(y>m/2) ans+= y-m/2;

		printf("%lld\n",ans);

    }
    return 0;
}
