#include <iostream>
#include <cstdio>
using namespace std;

int main()
{
    int tests,n,m;
    scanf("%d",&tests);
    while(tests--)
    {
        scanf("%d %d",&n,&m);
        long long x=n/m;
        long long y=n%m;
        long long ans=0;

        ans+=x*x*(m-1);
        ans+=x*(x-1);
        if(!(m&1)){ans-=x;}

        ans=ans/2;
        ans+=x*y;
        if(y>m/2)
        {
            ans+=y-(m/2);
        }

        printf("%lld\n",ans);
    }
    return 0;
}
