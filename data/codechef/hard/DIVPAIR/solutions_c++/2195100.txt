#include<cstdio>
#define ll long long
int main()
{
    int t;
    ll n,m;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lld %lld",&n ,&m);
        ll x=n/m;//full cycle
        ll y=n%m;//partial cycle
        ll ans=0;
        ans=x*(x-1)*(m-1);//when a b form full cycle
        ans/=2;//eliminating repeated values
        ans+=x*((m-1)/2);
        ans+=x*(x-1)/2;//a=0 and b=0
        ans+=x*y;//when b is from the partial cycle and a from full cycle
        if(y>m/2)//when a & b are from the partial cycle
            ans+=y-m/2;
        printf("%lld\n",ans);
    }
    return 0;
}
