#include<stdio.h>
#define int64 long long

int main()
{
    int T;
    int64 N,M,kc,max,ans;
    scanf("%d",&T);
    
    while(T--)
    {
        scanf("%lld%lld",&N,&M);
        if(M > 2*N-1 || N == 1)
        {
            printf("0\n");
        }
        else
        {
            kc = (N+1)/M;
            if((N+1)%M != 0)
            {
                kc++;
            }
            max = (2*N-1)/M;
            ans = (N+1)*(max+1-kc) - ((max+1-kc)*(max+kc)*M)/2;
            if(M%2 == 0)
            {
                ans += ((M*max*(max+1))/4) - max;
            }
            else
            {
                ans += ( ((M*(max+1)*max)/2) - max - (max/2) )/2;
            }
            printf("%lld\n",ans);
        }
    }
    
    return 0;
}
