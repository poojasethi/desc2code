void solve()
{
	LL n, m;
	sl(n); sl(m);
	LL x = n / m;	//full segments
	LL y = n % m;	//partial segments
	LL ans = x * x * (m - 1);	//a != 0 && b != 0
	ans += (x * (x - 1));	//a == 0 && b == 0
	if(m % 2 == 0) ans -= x;	//for even number same numbers are count with itself
	ans /= 2;
	ans += (x * y); 	//a belongs to full segments and b belongs to partial segment
	LL par = y - m / 2;
	if(par >= 0) ans += par;
	pl(ans); pn;
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		solve();
	}
	return 0;
}
