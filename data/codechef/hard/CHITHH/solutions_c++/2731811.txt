#include <iostream>
#define maxn 100010
#define maxm 316

int a[maxn];
int d[maxn];
int sum[maxn][maxm+10];
int n,m;

using namespace std;

int tinh(int i, int j)
{
    int x,dem;
    if (a[i]==0) return(0);
    x=a[i]; dem=0;
    while (x>0) {
        ++dem;
        if (dem==j) return(x%10);
        x=x/10;
    }
}
int main ()
{
    int i,j,l,v,k;
    int dau,cuoi,mid,t,tt,vt;
    cin>>n;
    for (i=1; i<=n; i++){
        cin>>a[i];
        j=a[i];
        if (a[i]==0) d[i]=1;
        while (j>0) {
            ++d[i]; j=j/10;
        }
    }
    for (i=n; i>=1; i--) {
        if (n-i>maxm) v=maxm; else v=n-i;
        for (j=0; j<=maxm; j++) sum[i][j]=d[i];
        for (j=1; j<=v; j++) sum[i][j]=d[i]+sum[i+j][j];
    }
    cin>>m;
    for (i=1; i<=m; i++) {
        cin>>l>>v>>k;
        if (v<=maxm) {
            vt=0;
            dau=0; cuoi=(n-l)/v;
            while (dau<=cuoi) {
                mid=(dau+cuoi)/2;
                j=l+(mid)*v;
                t=sum[l][v]-sum[j][v]+d[j];
                if (t>=k) {
                    vt=j; tt=t;
                    cuoi=mid-1;
                } else dau=mid+1;
            }
            if (vt==0) cout<<"So sad"<<endl; else cout<<tinh(vt,tt-k+1)<<endl;
        }
        else {
            t=0; tt=1;
            while (l<=n) {
                t+=d[l]; l+=v;
                if (t>=k) {
                    cout<<tinh(l-v,t-k+1)<<endl;
                    tt=0; break;
                }
            }
            if (tt==1) cout<<"So sad"<<endl;
        }
    }
}
