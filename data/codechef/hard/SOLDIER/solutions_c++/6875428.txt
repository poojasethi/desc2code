#include<bits/stdc++.h>
using namespace std;

struct val
{
	int p,q;
};

val arr[7];

int main()
{
	int n,w,i,t,p,q;
	
	scanf("%d%d",&n,&w);
	
	memset(arr,0,sizeof(arr)*7);
	
	int con=0;
	
	for(i=0;i<n;i++)
	{
		scanf("%d%d%d",&t,&p,&q);
		
		if(q>arr[t].q && con-arr[t].p+p<=w)
		{
			arr[t].q=q;
			con-=arr[t].p;
			con+=p;
			arr[t].p=p;
		}
	}
	
	for(i=1;i<=6;i++)
		if(arr[i].p==0)
		{
			cout<<"0\n";
			return 0;
		}
		
	int mini=arr[1].q;
	
	for(i=2;i<7;i++)
		mini=min(arr[i].q,mini);
	
	cout<<mini<<endl;
	
	return 0;
	
}
