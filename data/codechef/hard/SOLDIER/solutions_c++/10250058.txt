#include <stdio.h>
#include <vector>
#include <algorithm>
#include <utility>
using namespace std;
int n, t;
vector<pair<int, int> > G[6];
int sz[6];
int quality;
void travel(int i, int cost, int q){
	if(cost>t||q<=quality)return;
	if(i==6){
		quality = q;
		return;
	}
	for(int j = 0;j<sz[i];j++){
		travel(i+1, cost+G[i][j].first, min(q, G[i][j].second));
	}
}
int main(){
	int d, p, q;
	scanf("%d%d", &n, &t);
	for(int i =0 ;i< n; i++){
		scanf("%d%d%d", &d, &p, &q);
		G[d-1].push_back(make_pair(p, q));
	}
	for(int i =0 ; i<6; i++)sz[i] = (int)G[i].size();
	quality = 0;
	travel(0, 0, 100000000);
	printf("%d\n", quality);
	return 0;
}
