int solve(int k,int pr,int qu)
{
 	if(k==7)return qu;
 	int x=0;
 	for(int i=st[k];i<=ed[k];i++)
 	{
 	 if(e[i].p+pr<=t)
 	 {
 	 	if(e[i].q<qu)
 	  	x=solve(k+1,e[i].p+pr,e[i].q);
 	  	else x=solve(k+1,e[i].p+pr,qu);
 	  	if(x>0)return x;
	 }
	}
	return 0;
}
