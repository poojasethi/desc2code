inline bool func(const pair<int,pair<int,int> >&a ,const pair<int,pair<int,int> >& b){
if(a.first<b.first)return true;
if(a.first==b.first){
if(a.second.first+a.second.second==b.second.first+b.second.second)return a.second.first>b.second.first;
return (a.second.first+a.second.second<b.second.first+b.second.second);
}
return false;
}
vector<vector<int> > dp(10,vector<int>(10100,-1));
inline int solve(vector<vector<pair<int,int> > >&st,int in,int to,int q){
if(to<0 && in<6)return 0;
if(in==6)return q;
if(to<=0 && in<6)return 0;
if(in>6)return 0;
if(dp[in][to]!=-1 && dp[in][to]>=q)return dp[in][to];
//cout<<in<<" "<<to<<"\n";
int maxi=0;
for(int i=0;i<sz(st[in]);i++){
dp[in][to]=max(dp[in][to],solve(st,in+1,to-st[in][i].first,min(q,st[in][i].second)));
maxi=max(dp[in][to],maxi);
}
return maxi;
}
