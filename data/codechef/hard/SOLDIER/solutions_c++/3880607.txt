#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
    int N, T;
    int cnt[7] = {0};

    scanf("%d%d", &N, &T);
    int dp[7][1007] = {0};
    int p[7][107] = {0};
    int q[7][107] = {0};
    int t;
    for (int i = 0; i < N; ++i) {
        scanf("%d", &t);
        if (t < 1 || t > 6) {
            scanf("%d", &t);
            scanf("%d", &t);
        } else {
            scanf("%d%d", &p[t][cnt[t]], &q[t][cnt[t]]);
            cnt[t]++;
        }
    }
    int ans = 0;
    for (int i = 1; i <= 6; ++i) {
        for (int j = 0; j < cnt[i]; ++j) {
            for (int k = T; k >= p[i][j]; --k) {
                if (i == 1 && k - p[i][j] == 0) {
                    dp[i][k] = q[i][j];
                } else if (dp[i - 1][k - p[i][j]] != 0) {
                    int tmp = dp[i - 1][k - p[i][j]] > q[i][j] ? q[i][j] :
                        dp[i - 1][k - p[i][j]];
                    if (tmp > dp[i][k]) dp[i][k] = tmp;
                }
            }
        }
    }
    for (int i = 0; i <= T; ++i) {
        if (dp[6][i] > ans) {
            ans = dp[6][i];
        }
    }
    printf("%d\n", ans);
    return 0;
}
