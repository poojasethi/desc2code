#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> > vqp[6];
int vp[6];
int gtwmq()
{
    int mi=9999999,ty;
    for(int il=0;il<6;il++)
    {
        if(vqp[il].size()==0)
            return -1;
        if(mi>vqp[il][0].first)
        {
            mi=vqp[il][0].first;
            ty=il;
        }
    }
    return ty;
}
int main()
{
    int n,tot;
    cin>>n>>tot;
    if(n<6)
    {
        cout<<"0\n";
        return 0;
    }
    for(int il=0;il<6;il++)
        vp[il]=999999;
    while(n--)
    {
        int t,p,q;
        cin>>t>>p>>q;
        t--;
        vp[t]=min(vp[t],p);
        vqp[t].push_back(make_pair(q,p));
    }
    for(int il=0;il<6;il++)
    {
        sort(vqp[il].begin(),vqp[il].end());
        reverse(vqp[il].begin(),vqp[il].end());
    }
    while(1)
    {
        int curr=tot;
        int ty=gtwmq();
        if(ty==-1)
        {
            cout<<"0\n";
            break;
        }
        curr-=vqp[ty][0].second;
        for(int il=0;il<6;il++)
        {
            if(il==ty)
                continue;
            curr-=vp[il];
            if(curr<0)
                break;
        }
        if(curr<0)
            vqp[ty].erase(vqp[ty].begin());
        else
        {
            cout<<vqp[ty][0].first<<"\n";
            break;
        }
    }
    return 0;
}
