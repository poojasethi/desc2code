#include<iostream>
#include<vector>
#include<stack>
#include<algorithm>

using namespace std;

int main()
{
  int n,t,i,x,y,z,sum=0,min,max,pos,meaning_of_life,flag=0;
  vector<vector<pair<int,int> > > items(7);
  pair<int,int> temp;
  cin>>n>>t;
  for(i=0;i<n;i++)
  {
    cin>>x>>y>>z;
    temp.first=z;
    temp.second=y;
    items[x].push_back(temp);
    //cout<<"im here";
  }
  for(i=1;i<7;i++)
  {
    if(items[i].empty())
    {
      cout<<"\n0";
      return 0;
    }
    sort(items[i].begin(),items[i].end());
    //cout<<"im here 2";
  }
  while(1)
  {
    //cout<<"im here";
    min=32000;
    sum=0;
    for(i=1;i<7;i++)
    {
      if(items[i].empty())
      {
	cout<<"\n0";
	return 0;
      }
      sum+=items[i][items[i].size()-1].second;
      if(min>items[i][items[i].size()-1].first)
	min=items[i][items[i].size()-1].first;
    }
    if(sum<=t)
    {
      cout<<"\n"<<min;
      return 0;
    }
    else
    {
      max=0;
      flag=1;
      for(i=0;i<7;i++)
      {
	if(items[i].size()>=2 && max<items[i][items[i].size()-2].first)
	{
	  max=items[i][items[i].size()-2].first;
	  pos=i;
	}
	if(items[i].size()>1)
	  flag=0;
      }
      if(flag)
      {
	cout<<"\n0";
	return 0;
      }
      items[pos].pop_back();
      //cout<<items[pos][items[i].size()-1].first;
      //cin>>meaning_of_life;
    }
  }
}
