#include <bits/stdc++.h>
#define INF 1e9
#define SPEEDY ios::sync_with_stdio(false); cin.tie(0);
using namespace std;

long long memo[7][1010];
int n, t;
vector<pair<long long, long long> > G[7];

long long dp(int pos, int money_left)
{
	if (money_left < 0) return 0;
	if (pos > 6) return INF;

	if (memo[pos][money_left] != -1)
		return memo[pos][money_left];

	long long ans = 0;
	for (int i = 0; i < G[pos].size(); i++)
		ans = max(ans, min(G[pos][i].second, dp(pos + 1, money_left - G[pos][i].first)));

	return memo[pos][money_left] = ans;
}

int main()
{
	scanf("%d %d", &n, &t);

	for (int i = 0; i < n; i++) {
		long long a, b, c;
		scanf("%lld %lld %lld", &a, &b, &c);
		G[a].push_back(make_pair(b, c));
	}

	memset(memo, -1, sizeof memo);
	printf("%lld\n", dp(1, t));
}
