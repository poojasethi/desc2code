typedef long long vlong;
typedef vector<int> vi;
/***********Template Ends Here***********/

int n, t;

vi price[6];
vi quality[6];

int memo[7][1010];
int done[7][1010], cc = 1;

int dp ( int pos, int left ) {
    //printf ( "%d %d\n", pos, left );
    if ( pos == 6 ) return 1000000;

    if ( done[pos][left] == cc ) return memo[pos][left];

    int i;
    int res = 0, temp;
    for ( i = 0; i < price[pos].size(); i++ ) {
        if ( left - price[pos][i] >= 0 ) {
            temp = min ( quality[pos][i], dp ( pos + 1, left - price[pos][i] ) );
            if ( temp > res ) res = temp;
        }
    }

    done[pos][left] = cc;
    memo[pos][left] = res;
    return res;
}

int main () {
    //freopen ( "input.txt", "r", stdin );
    while ( scanf ( "%d %d", &n, &t ) != EOF ) {
        int i, a, b, c;

        for ( i = 0; i < 6; i++ ) {
            price[i].clear();
            quality[i].clear();
        }

        for ( i = 0; i < n; i++ ) {
            scanf ( "%d %d %d", &a, &b, &c );
            a--;
            price[a].pb ( b );
            quality[a].pb ( c );
        }

        int res = dp ( 0, t  );
        cc++;

        printf ( "%d\n", res );
    }

    return 0;
}
