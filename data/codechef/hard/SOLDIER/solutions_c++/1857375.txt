#include<stdio.h>
#include<assert.h>

inline int MinOf(int a, int b)
{
	return a<=b?a:b;
}

int main()
{
	int N,T,t,p,q,total,best,curr,MinSoFar0,MinSoFar1,MinSoFar2,MinSoFar3,MinSoFar4,MinSoFar5;
	int prices[105][6];
	int quality[105][6];
	int size[6];
	scanf("%d%d",&N,&T);
	for(int i=0; i<6; ++i) size[i] = 0;
	for(int i=0; i<N; ++i)
	{
		scanf("%d%d%d",&t,&p,&q);
		prices[size[t-1]][t-1] = p;
		quality[size[t-1]][t-1] = q;
		size[t-1]++;
	}
	best = 0;
	int i0=0,i1,i2,i3,i4,i5;
	if(size[0] == 0 || size[1] == 0 || size[2] == 0 || size[3] == 0 || size[4] == 0 || size[5] == 0)
	{
		printf("0\n");
		return 0;
	}
	do
	{
		curr = quality[i0][0];
		i1 = 0;
		MinSoFar0 = curr;
		do
		{
			curr = MinOf(MinSoFar0,quality[i1][1]);
			i2 = 0;
			MinSoFar1 = curr;
			do
			{
				curr = MinOf(MinSoFar1,quality[i2][2]);
				i3 = 0;
				MinSoFar2 = curr;
				do
				{
					curr = MinOf(MinSoFar2,quality[i3][3]);
					i4 = 0;
					MinSoFar3 = curr;
					do
					{
						curr = MinOf(MinSoFar3,quality[i4][4]);
						i5 = 0;
						MinSoFar4 = curr;
						do
						{
							curr = MinOf(MinSoFar4,quality[i5][5]);
							if(curr > best)
							{
								total = prices[i0][0] + prices[i1][1] + prices[i2][2] + prices[i3][3] + prices[i4][4] + prices[i5][5];
								if(total <= T) best = curr;
							}
							i5++;
						}
						while(i5 < size[5]);
						i4++;
					}
					while(i4 < size[4]);
					i3++;
				}
				while(i3 < size[3]);
				i2++;
			}
			while(i2 < size[2]);
			i1++;
		}
		while(i1 < size[1]);
		i0++;
	}
	while(i0 < size[0]);
	printf("%d\n",best);
	return 0;
}
