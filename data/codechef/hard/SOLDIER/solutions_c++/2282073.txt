#include<iostream>
#include<vector>
#include<cstdio>
using namespace std;
int main(){
	int n,t;
	scanf("%d %d",&n,&t);
	vector<int> v[7];
	int i;
	int x,y,z;
	int hi=-1;
	int lo=1000000;
	for(i=0;i<n;i++){
		scanf("%d %d %d",&x,&y,&z);
		v[x].push_back(y);
		v[x].push_back(z);
		if(z > hi){
			hi=z;
		}
		if( z < lo){
			lo=z;
		}
	}
	int mid;
	int j;
	int cost;
	int maxx=0;
	int min=10000000;
	while(lo <= hi){
		mid=lo+(hi-lo)/2;
		cost=0;
		for(i=1;i<=6;i++){
			min=10000000;
			for(j=0;j<v[i].size();j+=2){
				if(v[i][j+1]>=mid && v[i][j] <= min){
					min=v[i][j];
				}
			}
			cost=cost+min;
		}
		if(cost <= t){
			lo=mid+1;
			if(mid > maxx){
				maxx=mid;
			}
		}
		else{
			hi=mid-1;
		}
	}
	printf("%d\n",maxx);
	return 0;
}
