#include<iostream>
#include<cstdio>

using namespace std;

//Functions and Global variables
//main
int p[109],q[109],item[109],used[7],price[7];

int main()
{
    int n,t,i,maxiq,bal,j,flag=0,k;
    scanf("%d%d",&n,&t);
    for(i=1;i<=n;i++)
    {
        scanf("%d%d%d",&item[i],&p[i],&q[i]);
    }
    maxiq=-1;
    for(i=1;i<=n;i++)
        if(maxiq<q[i])
            maxiq=q[i];
    for(i=maxiq;i>0;i--)
    {
        bal = t;
        for(j=1;j<7;j++)
        {
            used[j]=0;
            price[j]=0;
        }
        for(k=1;k<7;k++)
        {
            for(j=1;j<=n;j++)
            {
                if(item[j]==k)
                {
                    if(used[item[j]]==0 && q[j]>=i && p[j]<=bal)
                    {
                        used[k]=1;
                        bal = bal-p[j];
                        price[k]=p[i];
                    }
                    else if(used[k]==1 && q[j]>=i && p[j]<price[k])
                    {
                        bal = bal+price[k]-p[j];
                    }
                }
            }
        }
        flag = 0;
        for(j=1;j<7;j++)
        {
            if(used[j]==0)
            {
                flag=1;
            }
        }
        if(flag==0)
        {
            printf("%d\n",i);
            return 0;
        }
    }
    printf("0\n");
    return 0;
}
