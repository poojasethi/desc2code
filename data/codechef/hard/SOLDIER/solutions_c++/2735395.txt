#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#define FOR(i,a,b) for(int i=a; i<b; i++)
#define si1(n) scanf("%d",&n)
#define pb push_back
#define F first
#define S second
using namespace std;

typedef pair<int,int> pi;
vector<pi> V[7];
int N,T;
int P[7][1003];
int solve(int no, int C){
    if(C<0) return 0;
    else if(no>6) return 1<<30;
    else if(P[no][C]!=-1) return P[no][C];
    int &res = P[no][C];
    res=0;
    FOR(i,0,V[no].size()){
        if(C>=V[no][i].F){
            res = max(res , min(V[no][i].S , solve(no+1,C - V[no][i].F)));
        }
    }
    return res;
}

int main(){
    si1(N),si1(T);
    int x=4;
    int t,p,q;
    memset(P,-1,sizeof(P));
    FOR(i,0,N) {
        si1(t), si1(p), si1(q);
        V[t].pb(pi(p,q));
    }
    int res=solve(1,T);
    printf("%d\n",res);

}
