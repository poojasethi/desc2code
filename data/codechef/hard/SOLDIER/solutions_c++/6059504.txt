#include<bits/stdc++.h>
#define MAX 1005

using namespace std;

int dp[7][MAX],ans[MAX];
typedef pair<int,int> pii;
vector<pii> adjList[MAX];

int solve(int type,int remMoney){
	
	if(remMoney<0) return 0;
	if(type>6) return 100000000;	
	if(adjList[type].size()==0) return 0;
	
	if(dp[type][remMoney]!=-1)
      return dp[type][remMoney];
	
	int ans=0,i;
    for(i=0;i<adjList[type].size();i++)  
	  if(remMoney-adjList[type][i].first>=0)
	     ans=max(ans,min(adjList[type][i].second,solve(type+1,remMoney-adjList[type][i].first)));
            	
	return dp[type][remMoney]=ans;
     
}   

int main(void){
	
	int totalItems,i,money;
	
    memset(dp,-1,sizeof dp);
    
	scanf("%d%d",&totalItems,&money);
	
	int cost,type,value;
	for(i=0;i<totalItems;i++){
		
		scanf("%d%d%d",&type,&cost,&value);
		adjList[type].push_back(pii(cost,value));		
		
	}
	
	int ans=solve(1,money);	
	if(ans>1000000)
	  printf("0\n");
	  
	else printf("%d\n",ans);  
	
	return 0;
	
}
