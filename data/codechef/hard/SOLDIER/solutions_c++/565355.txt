#include<iostream>
#include<map>
using namespace std;

int main() {
int I,T;
map<int,int> minQualityTable;
cin>>I>>T;
for(int i = 0 ; i < I ; ++i) {
  int item, itemp , itemq;
  cin>>item>>itemp>>itemq;
  if(minQualityTable.find(item) == minQualityTable.end()) {
    minQualityTable[item] = itemq;
  }
  else if( itemq > minQualityTable[item]) {
    minQualityTable[item] = itemq;
  }

}

  int globalMin = 1000000;
 for(map<int,int>::iterator it = minQualityTable.begin() ; it != minQualityTable.end() ; ++it) {
    if((*it).second < globalMin)
        globalMin = (*it).second;
 }
 if(minQualityTable.size() == 6 )
  cout<<globalMin<<endl;
 else
  cout<<0<<endl;

  return 0;
}
