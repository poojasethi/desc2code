    #include <bits/stdc++.h>
    using namespace std;

    int main(void)
    {
        int totalitems, totalmoney;
        cin >> totalitems >> totalmoney;

        int itemtype[totalitems], price[totalitems], quality[totalitems], temp[totalitems];
        for(int i = 0; i < totalitems; i++) {
            cin >> itemtype[i] >> price[i] >> quality[i];
            temp[i] = quality[i];
        }

        sort(temp, temp + totalitems);
        bool flag = false;
        for(int u = totalitems - 1; u >= 0; u--) {
            int maxquality = temp[u];
            int wallet = totalmoney;
            int flagofweapons[7];
            memset(flagofweapons, 0, sizeof(flagofweapons));

            for(int j = 1; j < 7; j++) {
                int moneyforthatitem = INT_MAX;
                for(int i = 0; i < totalitems; i++) {
                    if(itemtype[i] == j) {
                        if(quality[i] >= maxquality) {
                            if(price[i] <= moneyforthatitem) {
                                moneyforthatitem = price[i];
                            }
                        }
                    }
                }

                wallet -= moneyforthatitem;
                if(wallet < 0) break;
                flagofweapons[j] = 1;
            }
            int k;
            for(k = 1; k < 7; k++) {
                if(flagofweapons[k] == 0) {
                    break;
                }
            }
            if(k == 7) {
                cout << maxquality << endl;
                flag = true;
                break;
            }
        }
        if(!flag) {
            cout << "0" << endl;
        }
        return 0;
    }
