#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

typedef pair<int,int> pii;

int main()
{
    int N,T,a,b,c;
    scanf("%d%d",&N,&T);
    vector< pair<int,int> > inp[6];
    int best[6],mini;
    for(int i=0;i<6;i++) 
        best[i]=0;
    for(int i=0;i<N;i++)
    {
        scanf("%d%d%d",&a,&b,&c);
        inp[a-1].push_back(pair<int,int>(c,b));
        best[a-1]=max(best[a-1],c);
    }
    mini=best[0];
    for(int i=0;i<6;i++)
    {
        sort(inp[i].begin(),inp[i].end());
        mini=min(best[i],mini);
    }
    int ans=0;
    for(int i=1;i<=mini;i++)
    {
        int sum=0;
        bool con=true;
        for(int j=0;j<6;j++)
        {
            a=-1;
            for(vector< pair<int,int> >::iterator it=lower_bound(inp[j].begin(),inp[j].end(),pair<int,int>(i,0));it!=inp[j].end();it++)
            {
                if(a==-1 || a>it->second) 
                    a=it->second;
            }
            if(a==-1) 
                con=false;
            else 
                sum+=a;
        }
        if(con && sum<=T) 
            ans=i;
    }
    printf("%d\n",ans);
    return 0;
}
