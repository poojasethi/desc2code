#include<bits/stdc++.h>

using namespace std;
int main()
{
    int n,t;
    scanf("%d%d",&n,&t);
    int a[n+1][3],low=1,hi=0,istaken[6],ava,minP,flag;
    for(int i=0;i<n;i++)
    {
        scanf("%d%d%d",&a[i][0],&a[i][1],&a[i][2]);
        hi=max(hi,a[i][2]);
    }
    while(low<=hi)
    {
        int mid=(low+hi)/2;
        ava=t;
        for(int i=0;i<6;i++)    istaken[i]=0;
        for(int i=0;i<6;i++)
        {
            minP=1001;
            for(int j=0;j<n;j++)
            {
                if(a[j][0]==i+1)
                {
                    if(a[j][2]>=mid)
                    {
                        if(a[j][1]<=ava)
                        {
                            minP=min(minP,a[j][1]);
                        }
                    }
                }
            }
            if(minP<1001)
            {
                istaken[i]=1;
                ava-=minP;
            }
            else    break;
        }
        flag=0;
        for(int i=0;i<6;i++)
        {
            if(istaken[i]!=1)
            {
                flag=1;break;
            }
        }
        if(flag==1)
        {
            hi=mid-1;
        }
        else
        {
            low=mid+1;
        }
    }
    printf("%d\n",low-1);
    return 0;
}
