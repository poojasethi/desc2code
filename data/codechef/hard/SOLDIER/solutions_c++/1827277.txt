#include <cstdio>
#include <algorithm>
#include <vector>
#include <utility>
using namespace std;

typedef pair<int,int> ii;
const int INF = 2000000000;

int n, r;
int t,p,q;
vector<ii> a[7];
int dp[7][1001];

int main() {
#ifndef ONLINE_JUDGE
	freopen("in.txt", "r", stdin);
#endif
	scanf("%d%d",&n, &r);
	for (int i=0; i<=6; i++) a[i].clear();
	for (int i=0; i<n; i++) {
		scanf("%d%d%d",&t,&p,&q);
		a[t].push_back(ii(p,q));
	}
	for (int i=1; i<=6; i++) for (int j=0; j<=r; j++) dp[i][j] = 0;
	for (int j=0; j<=r; j++) dp[0][j] = INF;
	for (int i=1; i<=6; i++) {
		if (a[i].size()==0) {
			printf("0\n");
			return 0;
		}
		for (int j=0; j<a[i].size(); j++) {
			int p = a[i][j].first;
			int q = a[i][j].second;
			for (int k=0;k+p<=r; k++) {
				dp[i][k+p] = max(dp[i][k+p], min(dp[i-1][k], q));
			}
		}
	}
	printf("%d\n", dp[6][r]);
	return 0;
}
