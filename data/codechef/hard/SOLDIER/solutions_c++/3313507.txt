using namespace std;;

int main()
{
	#ifndef ONLINE_JUDGE
	READ("read.txt");
	#endif
	int n,k,cart[7],t[105],p[105],q[105],minq,ans=0,cnt=0,sum;
	S(n),S(k);
	//assert(n<=101 and n>=0);
	FOR(i,1,n+1){
		S(t[i]),S(p[i]),S(q[i]);
		//assert(t[i]>=1 and t[i]<=6);
	}
	FOR(i,1,n+1){
		if(t[i]<1 or t[i]>6)
			continue;
		FILL(cart,0),cnt=0;
		cart[t[i]]=i;
		cnt++;
		minq=q[i],sum=p[i];
		FOR(j,1,n+1){
			if(j==i) continue;
			
			if(t[j]<1 or t[j]>6)
				continue;
			
			if(cart[t[j]]==0){
				if(q[j]>=minq){
					cart[t[j]]=j;
					cnt++;
					sum+=p[j];
				 }
			}
			else{
				if(q[j]>=minq and p[j]<p[cart[t[j]]]){
					cart[t[j]]=j;
					sum=sum-p[cart[t[j]]]+p[j];
				}
			}
		}
		if(cnt==6 and sum<=k){
			ans=max(ans,minq);
		}
	}
	printf("%d\n",ans);
	return 0;
}
