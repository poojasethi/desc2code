int t, m, n, R, C, used[66777], now, e, maxe;
bool f[66777];
char s[33];
ll pw[33], a[555666], c;
 
bool ok(ll e) {
	int x=0, y=m-1, mid;
	while(x<=y) {
		mid=(x+y)>>1;
		if(a[mid]==e) return 1;
		if(a[mid]<e) x=mid+1; else y=mid-1;
	}
	return 0;
}
void sinh(int i) {
	if(i==n) {
		if(ok(c)) used[e]=now;
		return;
	} 
	sinh(i+1);	
	e+=(1<<i);
	c+=pw[s[i]-'A'];
	sinh(i+1);
	e-=(1<<i);
	c-=pw[s[i]-'A'];	
}
 
int main() {		
	if(debug) openfile; srand(time(NULL));
	
	pw[0]=1;
	foruu(i,1,25) pw[i]=pw[i-1]*base;
	
	scanf("%d\n",&m);
	foru(i,m) {
		gets(s);
		foru(j,strlen(s)) a[i]+=pw[s[j]-'A'];
	}
	sort(a,a+m);
	
	scanf("%d%d%d\n",&R,&C,&t);
	
	n=R*C;
	maxe=1<<n;
	while(t--) {
		foru(i,R) gets(s+i*C);
		
		c=e=0;
		now++;
		sinh(0);
		
		for(e=1; e<maxe; e++) {
			f[e]=0;
			for(int x=e; x>0; x=e&(x-1)) 
				if(used[x]==now && (f[e]|=!f[e^x])) break;
		}
		
		if(f[maxe-1]) puts("Alice"); else puts("Bob");
	}
	
	return 0;	
};
