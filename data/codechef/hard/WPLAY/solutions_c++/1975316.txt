#define ESP (1e-9)
#define N 67000

#define sz(a) int(a.size())
#define pii pair<int,int>
ll pw[35],a[500000],c;
int m,n,use[N],now,e;
bool f[N];char s[35];
bool check(ll e)
{
 int lo=0,hi=m-1,mid;
 while(lo<=hi)
 {
  mid=(lo+hi)>>1;
  if(a[mid]==e)return 1;
  else if(a[mid]>e)hi=mid-1;
  else lo=mid+1;
 }
 return 0;
}
void solve(int i)
{
 if(i==n)
 {
  if(check(c))use[e]=now;
  return;
 }
 solve(i+1);
 c+=pw[s[i]-'A'];e+=(1<<i);
 solve(i+1);
 c-=pw[s[i]-'A'];e-=(1<<i);
}
int main()
{
	//freopen("in.txt","r",stdin);
	int T,i,j,k,p,ro,co;
	pw[0]=1;
	for(i=1;i<=25;i++)pw[i]=pw[i-1]*M;
	scanf("%d\n",&m);
	memset(a,0,sizeof(int)*m);
	for(i=0;i<m;i++)
	{
	 gets(s);
	 for(j=0;j<strlen(s);j++)
	 a[i]+=pw[s[j]-'A'];
	}
	sort(a,a+m);
	scanf("%d%d%d\n",&ro,&co,&T);
	n=co*ro;now=0;
	int maxe=1<<n;
	while(T--)
	{
	 for(i=0;i<ro;i++)
	 gets(s+i*co);
	 c=e=0;now++;
	 solve(0);
	 memset(f,0,sizeof(f));
	 for(e=1;e<maxe;e++)
	 {
	  f[e]=0;
	  for(int x=e;x>=1;x=e&(x-1))
	  if(use[x]==now&&(f[e]|=!f[e^x]))break;
	 }
	 if(f[maxe-1])puts("Alice");
	 else puts("Bob");
	}
	return 0;
}
