using namespace std;
set<long long>st;
string tmp;
char s[20],a[20][20];
int dp[70000],present[70000];
long long hash(string str)
{
    long long h=0;
    for(int i=0;i<str.size();i++)
    h=h*100+str[i];
    return h;
}
int main()
{
    int i,n,j,k,d,t;
    scanf("%d",&d);
    for(i=0;i<d;i++)
    {
      scanf("%s",s);
      tmp=s;
      sort(tmp.begin(),tmp.end());
      st.insert(hash(tmp));
    }
    int r,c;
    scanf("%d%d%d",&r,&c,&t);
    n=r*c;
    while(t--)
    {
        for(i=0;i<r;i++)
        scanf("%s",a[i]);
        for(i=0;i<(1<<n);i++)
        {
            tmp="";
            for(j=0;j<n;j++)
            {
                if(i&(1<<j))
                tmp+=a[j/c][j%c];
            }
            sort(tmp.begin(),tmp.end());
            present[i]=(st.find(hash(tmp))!=st.end());
        }
    for(i=0;i<(1<<n);i++)
    {
        dp[i]=0;
        for(j=i;j>0;j=((j-1)&i))
        if(present[j]&&dp[i^j]==0)
        {
            dp[i]=1;break;
        }
    }
    if(dp[(1<<n)-1]==1)
    printf("Alice\n");
    else printf("Bob\n");
    }
    return 0;
}
