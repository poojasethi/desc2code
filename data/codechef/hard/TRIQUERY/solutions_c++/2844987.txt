#include <stdio.h>
#include <algorithm>
#include <math.h>

using namespace std;
const int maxn = 5e5;

struct point
{
	int x;
	int y;
	int d;
	int pos;

	point(int x, int y)
	{
		x = x;
		y = y;
	}
	point()
	{}

}puntos[maxn],queries[maxn];

int n,q;

int abix[maxn];
int abiy[maxn];

int below[maxn];
int ans[maxn];

int compare_x(const point& a, const point& b)
{
	if(a.x != b.x)return a.x < b.x;
	return a.y < b.y;
}

int compare_m(const point& a, const point& b)
{
	int r1 = a.x + a.y + a.d;
	int r2 = b.x + b.y + b.d;
	return r1 < r2;
}

void update(int pos, int v, int* bit, int l)
{
	while(pos <= l)
	{
		bit[pos] += v;
		pos += (pos&-pos);
	}
}

int query(int pos, int* bit, int l)
{
	int res = 0;
	while(pos)
	{
		res += bit[pos];
		pos -= (pos&-pos);
	}
	return res;
}

void solveless()
{
	sort(puntos,puntos + n ,compare_m);
	sort(queries,queries + q,compare_m);

	for(int i = 0, j = 0; i < q; i++)
	{
		for(; j < n; j++)
		{
			if(puntos[j].x + puntos[j].y > queries[i].x + queries[i].y + queries[i].d)break;
			update(puntos[j].x,1,abix,maxn);
			update(puntos[j].y,1,abiy,maxn);
		}
		ans[queries[i].pos] = j - query(queries[i].x - 1,abix, maxn) - query(queries[i].y - 1,abiy, maxn);
	}
}

void solve()
{
	fill(abix,abix + maxn,0);
	fill(abiy,abiy + maxn,0);

	sort(puntos, puntos + n ,compare_x);
	sort(queries, queries + q,compare_x);

	for(int i = 0, j = 0; i < q ;i++)	
	{
		for(; j < n; j++)
		{
			if(puntos[j].x >= queries[i].x)break;
			update(puntos[j].y,1,abiy,maxn);
		}
		ans[queries[i].pos] += query(queries[i].y - 1,abiy, maxn);
	}
}

int main()
{
	scanf("%d %d", &n, &q);

	for(int i = 0; i < n; i++)
		scanf("%d %d", &puntos[i].x,&puntos[i].y);

	for(int i = 0; i < q; i++)
	{
		scanf("%d %d %d", &queries[i].x,&queries[i].y, &queries[i].d);
		queries[i].pos = i;
	}

	solveless();
	solve();
	for(int i = 0; i < q; i++)
		printf("%d\n",ans[i]);

	return 0;
}
