#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll c[1000111];
int mu[1000111];
int a[100011],p[1000111];
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i=0; i<n; i++)
        scanf("%d",&a[i]);
    mu[1]=1;
    for(i=2; i<=1000000; i++)
        mu[i]=1;
    for(i=2; i<=1000000; i++)
    {
        if(p[i]==0)
        {
            for(j=i; j<=1000000; j+=i)
            {
                if((j/i)%i==0)
                    mu[j]=0;
                else
                    mu[j]=mu[j]*(-1);
                if(j!=i)
                    p[j]=1;
            }
        }
    }
    for(i=0;i<n;i++)
    {
        for(j=1;j<=sqrt(a[i]);j++)
            if(a[i]%j==0)
        {
            c[j]++;
            if(j!=(a[i]/j))
                c[a[i]/j]++;
        }
    }
    ll ans=0LL;
    for(i=1;i<=1000000;i++)
        ans=ans+(mu[i]*((c[i]*(c[i]-1)*(c[i]-2))/6LL));
    printf("%lld\n",ans);
    return 0;
}
