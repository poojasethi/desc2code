

#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int mod = 1e9 + 7 ;
int n, maxi, i, ai, a[1100000], p[1100000], dp[1100000], j;

int ele;
int main () 
{	
	ll sum,ans;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&ele);
		a[ele]++;
		maxi=max(maxi,ele);
	}
	for(i=2;i<=maxi;i++)
	{
		if(!p[i])
		{
			for(j=i;j<=maxi;j+=i)
			{
				if(!p[j])
					p[j]=i;
			}
		}
	}
	ans=0;
	for(i=1;i<=maxi;i++)
	{
		if(i==1) dp[i]=1;
		else if(p[i]==p[i/p[i]])	dp[i]=0;
		else dp[i]=-1*dp[i/p[i]];
		if(dp[i]==0)	continue;
		sum=0;
		for(j=i;j<=maxi;j+=i)
		{
			sum+=a[j];
		}
		ans+=(sum)*(sum-1)*(sum-2)/6*dp[i];
	}
	
	cout<<ans;
}
