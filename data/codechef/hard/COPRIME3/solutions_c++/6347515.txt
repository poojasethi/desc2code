#include <bits/stdc++.h>

#define	st first
#define	nd second
#define	mp make_pair
#define	pb push_back
#define	lli long long int
#define	all( gg )	gg.begin(),gg.end()
#define	foreach( gg,itit )	for( typeof(gg.begin()) itit=gg.begin();itit!=gg.end();itit++ )
#define	FP( ii,aa,bb ) for( lli ii=aa;ii<=bb;ii++ )
#define	FM( ii,aa,bb ) for( lli ii=aa;ii>=bb;ii-- )
#define	debug(ccc)	cout << #ccc << " = " << ccc << endl;

#define	mod	1000000007LL
using namespace std;

int	n,arr[200000],lp[2000000],s[2000000];
vector<int>	pr;
vector< pair<int,int> >	v;

int	f(int x){
	int	r=0;
	while( x!=1 ){
		if( lp[x/lp[x]]==lp[x] )	return	-1;
		r^=1;
		x /= lp[x];
	}
	return	r;
}

lli	g( lli x ){
	return	x*(x-1)*(x-2)/6;
}

int main(){
	lp[1] = 1;
	FP( i,2,1000000 ){
		if( lp[i]==0 ){
			lp[i] = i;
			pr.pb(i);
		}
		for( int j=0;j<pr.size() and pr[j]<=lp[i] and pr[j]*i<=1000000;j++ )
			lp[pr[j]*i] = pr[j];
	}
	int	t;
	FP( i,1,1000000 ){
		t = f(i);
		if( f(i)==-1 )	continue;
		v.pb( mp( i,t ) );
	}
	scanf("%d",&n);
	FP( i,1,n ){
		scanf("%d",&t);
		for( int j=1;j*j<=t;j++ ){
			if( t%j )	continue;
			s[j]++;
			if( j*j!=t )	s[t/j]++;
		}
	}
	lli	res=0;
	foreach( v,it ){
		lli	t = g( s[it->st] );
		if( it->nd==0 )	res += t;
		else	res -= t;
	}

	cout << res << endl;
}
