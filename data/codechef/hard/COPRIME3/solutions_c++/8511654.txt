#include <bits/stdc++.h>
#define filein(t) freopen(t, "r", stdin)
#define fileout(t) freopen(t, "w", stdout)
#define mp make_pair
#define f first
#define s second
#define N 1000005
#define NN 10000000
#define ln 1e10+1
#define BIT(i, j) ((j >> i) & 1)
#define ll long long
#define oo 1000000007
#define ln 1e13

using namespace std;

int P[N], num[N], n;
ll ans = 0;

void Solve(int x)
{
    int y = x, cnt = 0, C = 0;
    while(y > 1)
    {
        cnt++;
        int p = P[y], w = 0;
        while(y % p == 0)   y /= p, w++;
        if(w > 1)   return;
    }
    for(int i=x; i<N; i+=x)
        C += num[i];
    ans += (ll)C * (C-1) * (C-2) / 6 * (cnt % 2 ? -1 : 1);
}

int main()
{
    //filein("inp.txt");  fileout("out.txt");
    int u;
    for(int i=2; i<N; i++)
    {
        if(P[i])    continue;
        for(int j=i; j<N; j+=i)
            P[j] = i;
    }
    cin >> n;
    for(int i=1; i<=n; i++)
    {
        cin >> u;
        num[u]++;
    }
    for(int i=1; i<N; i++)
        Solve(i);
    cout << ans << endl;
}
