#include <bits/stdc++.h>
#define N 1000000
typedef long long ll;
ll ways[N + 1];
int cnt[N + 1],tot[N + 1];
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i = 1 ; i<=n ; i++)
    {
        int u;
        scanf("%d",&u);
        tot[u]++;
    }
    for(i = 1 ; i<=N ; i++)
    {
        for(int j = i ; j<=N; j+=i)
        {
            cnt[i] = cnt[i] + tot[j];
        }
    }
    for(i = N ; i>0 ; i--)
    {
        ll temp = cnt[i];
        ll wa = (temp*(temp-1)*(temp-2))/6;
        if(wa == 0)
        {
            ways[i] = 0;
            continue;
        }
        for(j = 2*i ; j<=N ; j+=i)
        {
            wa = wa - ways[j];
        }
        ways[i] = wa;
    }
    ll ans = ways[1];
    printf("%lld\n",ans);
    return 0;
}
