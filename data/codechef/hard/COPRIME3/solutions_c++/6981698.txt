#include <bits/stdc++.h>
#define maxn 1000000+3
using namespace std;

#define s(n) scanf("%d",&n)
int a[maxn];
int mu[maxn];

int i, j;
int maxa = 0;
long long  C(long long  k) {
	return 1LL* k * (k-1) * (k-2) / 6;
}
int main() {
	// your code goes here
	long long ans = 0;
	int n, aa;
	s(n);
	mu[1] = 1;
	for( i=1; i<=maxn; i++) {
		if(mu[i]!=0) {
			for(int j = i+i; j <=maxn; j+=i) {
				mu[j] -= mu[i];
			}
		}
	}
	for( i=0;i <n;i++) {
		s(aa);
		maxa = max(maxa, aa);
		a[aa]++;
	}
	
	for( i=1; i<=maxa; i++) {
		j = i;
		long long k = 0;
		while(j <= maxa) {
			k += a[j];
			j += i;
		}
		
		ans += C(k) * mu[i];
	}	
	
	printf("%lld\n",ans);
	return 0;
}
