#include<cstdio>
#include<vector>

using namespace std;

int N, maxi, mobius[1000009], init[1000009];
bool ciur[1000009];
long long sol;

int main ()
{
//freopen ("input", "r", stdin);
//freopen ("output", "w", stdout);

scanf ("%d", &N);
for (int i=1; i<=N; i++)
{
    int ai;
    scanf ("%d", &ai), init[ai] ++;
    if (ai > maxi) maxi = ai;
}

for (int i=1; i<=maxi; i++)
    mobius[i] = 1;

for (int i=2; i <= maxi; i++)
    if (ciur[i] == 0)
    {
        for (int j=i; j<=maxi; j+=i)
            ciur[j] = 1, mobius[j] = -mobius[j];
        if (i <= maxi / i)
        {
            for (int j=i * i; j<=maxi; j += i * i)
                mobius[j] = 0;
        }
    }

for (int i=1; i<=maxi; i++)
if (mobius[i] != 0)
{
    int cnt2 = 0;
    for (int j=i; j<=maxi; j+=i)
        cnt2 += init[j];
    sol = (long long) sol + 1LL * mobius[i] * cnt2 * (cnt2 - 1) * (cnt2 - 2) / 6;
}
printf ("%lld\n", sol);

return 0;
}
