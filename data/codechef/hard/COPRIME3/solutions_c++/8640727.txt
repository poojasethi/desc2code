#include <bits/stdc++.h>
using namespace std;
const int lim=1e6;
long long cnt[lim+1];
long long c3(long long x)
{
    return x*(x-1)*(x-2)/6;
}
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    {
        int x;
        scanf("%d",&x);
        cnt[x]+=1;
    }
    for(int i=1;i<=lim;++i)
    {
        for(int j=i+i;j<=lim;j+=i)
            cnt[i]+=cnt[j];
        cnt[i]=c3(cnt[i]);
    }
    for(int i=lim;i>=1;--i)
        for(int j=i+i;j<=lim;j+=i)
            cnt[i]-=cnt[j];
    cout<<cnt[1];
}
