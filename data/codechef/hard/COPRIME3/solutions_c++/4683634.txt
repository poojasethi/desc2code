#include<cstdio>

const int mx = 1000006;
int notprime[mx], prime[mx], pcnt, num[mx], n;
int cnt[mx];
int mul[mx];


void seive() {
    notprime[0] = notprime[1] = 1;
    for(int i = 2; i * i < mx; ++i) if(!notprime[i]) {
        for(int j = i * i; j < mx; j += i) notprime[j] = 1;
    }

    for(int i = 2; i < mx; ++i) if(!notprime[i]) prime[pcnt++] = i;
}

long long ans;

int main() {
    seive();
    scanf("%d", &n);
    for(int i = 0; i < n; ++i) scanf("%d", num + i);

    for(int i = 0; i < n; ++i) {
        for(int j = 1; j * j <= num[i]; ++j) {
            if(num[i] % j == 0) {
                ++cnt[j];
                if(num[i] != j * j) ++cnt[num[i] / j];
            }
        }
    }

    ans = 0;
    for(int i = 1; i < mx; ++i) {
        int mul = 1, m = i;
        bool ok = true;
        for(int j = 0; m > 1 && notprime[m]; ++j) if(m % prime[j] == 0) {
            mul *= -1;
            m /= prime[j];
            if(m % prime[j] == 0) {
                ok = false;
                break;
            }
        }
        if(m > 1 && !notprime[m]) mul *= -1;

        if(ok) ans += mul * (cnt[i] * (cnt[i] - 1ll) * (cnt[i] - 2ll) / 6ll);
    }



    printf("%lld\n", ans);
    return 0;
}
