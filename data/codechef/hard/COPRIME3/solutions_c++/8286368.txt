ll cnt[1000100],
   cnt2[1000100];

int main() {
/*

   \sum{i=1}^n \sum{j=i+1}^n \sum{k=j+1}^n [gcd(ai,aj,ak) = 1]

*/

    memset(cnt, 0, sizeof(cnt));
    int n;
    scanf("%d", &n);
    rep(i,0,n) {
        int x;
        scanf("%d", &x);
        for (int y = 1; y*y <= x; y++) {
            if (x % y == 0) {
                cnt[y]++;
                if (y != x/y) {
                    cnt[x/y]++;
                }
            }
        }
    }

    for (int i = 1000000; i >= 1; i--) {
        ll m = cnt[i];
        cnt2[i] = m * (m-1) * (m-2) / 6;
        for (int j = i+i; j <= 1000000; j += i) {
            cnt2[i] -= cnt2[j];
        }
    }

    cout << cnt2[1] << endl;

    return 0;
}
