#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int p[1000000+5];
int dp[1000000+5];
int A[1000000+5];
int main()
{
    int n;
    cin>>n;

    int maxi=-1;
    for(int i=1;i<=n;i++)
    {
        int a;
        cin>>a;
        A[a]++;
        maxi=max(maxi,a);
    }
    for(int i=2;i<=maxi;i++)
    {
        int j=i;
        if(!p[i])
        {
            while(j<=maxi)
            {
                if(!p[j]) p[j]=i;
                j+=i;
            }
        }
    }
    ll ans=0;
    for(int i=1;i<=maxi;i++)
    {
        if(i==1) dp[i]=1;
        else if(p[i/p[i]]==p[i]) dp[i]=0;
        else dp[i]=-1*dp[i/p[i]];
        if(!dp[i]) continue;
        ll total=0;
        int j=i;
        while(j<=maxi)
        {
            total+=A[j];
            j+=i;
        }
        ans+= (total*(total-1)*(total-2)/6)*dp[i];
    }
    cout<<ans<<endl;
}
