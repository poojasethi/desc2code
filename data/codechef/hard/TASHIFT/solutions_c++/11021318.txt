#include<bits/stdc++.h>
using namespace std;

int main(){
	int i, n, j, k, val;
	cin>>n;
	string a, b;
	cin>>a>>b;
	b=b+b;
//	cout<<a<<endl<<b<<endl;
	int c[n];	
	c[0]=0;
	i=0; j=1;
	while(j<n){
		if(a[i]==a[j]){
			c[j]=i+1;
			i++, j++;
		}
		else{
			if(i!=0){
				i=c[i-1];
			}
			else
			c[j++]=0;
		}
	//cout<<j-1<<" "<<c[j-1]<<"\n";
	}//cout<<endl;
	int d[n];
	memset(d, -1, sizeof d);
	i=0;j=0;
	//search a in b
	while(i<2*n){
		if(b[i]==a[j]){
			if(d[j]==-1)d[j]=i-j;
			i++;
			j++;
		}
		else{
			if(j!=0)
			j=c[j-1];
			else i++;
		}
		if(j==n){
			break;
		}

	}
	i=n-1;
	while(d[i]==-1&&i>=0)i--;
	if(i==-1)cout<<"0\n";
	else cout<<d[i]<<endl;
}
