
#include <bits/stdc++.h>
using namespace std;
#define md int(2e6+100)
#define inf int(1e9+100)
#define FOR(i,a,b) for(int i=(a),_b=(b); i<=_b ;i++)
#define DOW(i,b,a) for(int i=(b),_a=(a); i>=_a ;i--)
#define pii pair<int,int>
#define ll long long

int n;
char a[md],b[md];
int f[md];

int main()
{
    //freopen("inp.txt","r",stdin);
    scanf("%d",&n);
    scanf("%s",a+1);
    scanf("%s",b+1);
    FOR(i,n+1,2*n) b[i]=b[i-n];
    int j=0;
    FOR(i,2,n)
    {
        while (j!=0 && a[i]!=a[j+1]) j=f[j];
        if (a[i]==a[j+1]) j++;
        f[i]=j;
    }
    j=0;
    int res=0,x=0;
    FOR(i,1,2*n)
    {

        while (j!=0 && b[i]!=a[j+1]) j=f[j];
        if (b[i]==a[j+1]) j++;
        if (j>res) { res=j; x=i-j; }
        if (j==n) j=f[j];
    }
    cout<<x<<endl;
}
