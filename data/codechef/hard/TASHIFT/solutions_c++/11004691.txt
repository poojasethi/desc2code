#include<bits/stdc++.h>

using namespace std;

char a[1000000], b[1000000];
long long int   l[1000000];

int main()
{
long long int i,j,k,m,n,len,ans,p,maxi;
len=0;
l[0]=0;
cin>>n;
for(i=0;i<n;i++)
cin>>a[i];
for(i=0;i<n;i++)
cin>>b[i];
for(i=0;i<n &&i+n<1000000;i++)
b[i+n]=b[i];
i=1;
while(i<n)
{
    if(a[i]==a[len])
    {
        len++;
        l[i]=len;
        i++;
    }
    else
    {
        if(len!=0)
        {
            len=l[len-1];
        }
        else
        {
            l[i]=0;
            i++;
        }
    }
}
/*
for(i=0;i<n;i++)
cout<<a[i];
cout<<endl;
for(i=0;i<n;i++)
cout<<l[i];
*/
j=0;
maxi=-1;
i=0;
p=0;
ans=0;
while(i<n)
{
    if(j==n)
    break;

    if(a[j]==b[i])
    {
     //   cout<<endl<<a[j]<<b[i]<<ans<<" "<<p<<" "<<i ;
        j++;
        i++;
        p++;
        if(p>maxi)
        {
         maxi=p;
         ans=i-j;
        }
    }
    else
    {
       // cout<<endl<<a[j]<<b[i]<<ans<<" "<<p<<" "<<i<<"not";
        if(p>0)
        {
         //i=i+p-l[p-1];
            j=l[j-1];
        }
        else
        {
         i++;
        }
        p=0;
    }


}
cout<<ans;
}
