#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define MAX 2000005
using namespace std;

int P[MAX];

void phi(string &s){
	int i,j;
	i = 0;
	j = P[i] = -1;
	while(i < s.size()){
		while(j > -1 && s[i] != s[j]) j = P[j];
		i++;
		j++;
		P[i] = j;	
	}
}

int search(string &a, string &b){
	int i,j;
	i = j = 0;
	phi(a);
	int maxi = 0; int pos = -1;
	while(i < b.size()){
		while(j > -1 && b[i] != a[j]) j = P[j];
		i++;
		j++;
		
		if(j > maxi){
			maxi = j;
			pos = i - j;
		}
		
		if(i == a.size()) j = P[j];
	}
	return pos;
}

int main(){
	int n;
	scanf("%d",&n);
	string a,b;
	cin >> a >> b;
	b = b + b;
	printf("%d\n",search(a,b));
	return 0;
}
