#include <bits/stdc++.h>
using namespace std;
int pre[1000007];
int main() {
	int n;
	cin >> n;
	string a,b;
	cin >> a >> b;
	b = b+b;
	int i=1,j=0;
	pre[0]=0;
	while(i < n) {
		if(a[i] == a[j]) {
			j++;
			pre[i] = j;
			i++;
		}
		else  {
			if(j != 0) {
			   j=pre[j-1];	
			}
			else {
				pre[i]=0;
				i++;
			}
		}
	}
	i=0;
	j=0;
	int res=0,best=0;
	int val=0;
   while(i < 2*n) {
   	if(b[i] == a[j]) {
        i++;
        j++;
        best++;
   	}
   	if(best > res) {
            res = best;
            val = i-j;
        }
    if(j == n/2)
      break;
    if(a[j] != b[i]) {
    	best = 0;
    	if(j != 0) 
    		j = pre[j-1];
    	else
    	    i++;
    }
   }
   cout << val << endl;
	return 0;
}
