using namespace std;
const int M = 1e9 + 7, B = 37;
const int MAXN = 1e6 + 2;

int N;
char a[MAXN], b[MAXN];
int Ahash[MAXN], Bpow[MAXN];

int trymatch(int len)
{
	int bh = 0;
	for (int i = 0; i < len; i++)
		bh = ((1LL * bh * B) % M + (b[i]-'a')) % M;
	if (bh == Ahash[len]) return 0;
	//cout << bh << " ";
	for (int i = 1; i < N; i++)
	{
		bh = (bh - (1LL * (b[i - 1]-'a') * Bpow[len - 1]) % M + M) % M;
		bh = (1LL * bh * B) % M;
		bh = (bh + (b[(i + len - 1) % N]-'a')) % M;
		if (bh == Ahash[len]) return i;
		//cout << bh << " ";
	}
	return -1;
}

int sol = 0;

void bs(int l, int r)
{
	if (l == r)
	{
		int a = trymatch(l);
		if (a != -1)
			sol = a;
	}
	else if (l + 1 == r)
	{
		int a = trymatch(l);
		if (a != -1)
			sol = a;
		a = trymatch(r);
		if (a != -1)
			sol = a;
	}
	else
	{
		int mid = (l + r) / 2;
		int a = trymatch(mid);
		if (a == -1)
			bs(l, mid - 1);
		else
			bs(mid, r);
	}
}

int main()
{
	scanf("%d", &N);
	scanf("%s%s", a, b);
	Bpow[0] = 1;
	for (int i = 1; i <= N; i++)
		Bpow[i] = (1LL * Bpow[i - 1] * B) % M;
	Ahash[0] = 0;
	for (int i = 1; i <= N; i++)
		Ahash[i] = ((1LL * Ahash[i - 1] * B) % M + (a[i - 1]-'a'))%M;
	//trymatch(2);
	bs(1, N);
	printf("%d\n", sol);
	return 0;
}
