
#include<bits/stdc++.h>
using namespace std;
int check(string s,int l1,int l2)
{
    if(l1!=l2)
        return 0;
    vector<int> z;
    z.resize(s.length());
    int n=s.length(),l=0,r=0,k;
    for(int i=1;i<n;i++)
    {
        if(i>r)
        {
            l=r=i;
            while(r<n && s[r-l]==s[r])
                r++;
            z[i]=r-l;
            r--;
        }
        else
        {
            k=i-l;
            if(z[k]<r-i+1)
                z[i]=z[k];
            else
            {
                l=i;
                while(r<n && s[r-l]==s[r])
                    r++;
                z[i]=r-l;
               r--;
            }
        }
    }
    int ans=0;
    int l11=0;
    //0123456789 10 11 12 13 14 15
    //ccadd$bddc c  b  d  d  c  c
    for(int i=l2+1;i<n;i++)
    {
        if(z[i]>ans)
        {
            ans=z[i];
            l11=i-l2-1;
        }
        else
        if(z[i]==ans)
        {
            l11=min(l11,(i-l2-1)%l1);
        }
    }
    return l11;
}
