#include<bits/stdc++.h>
using namespace std ;
int n , m , shift , z[2000004] ;
void zalgo ( string s ) {
    int i , length = s.size() , left = 0 , right = 0 ;
    for ( i = n+1 ; i < length ; i++ ) {
        if ( i > right ) {
            left = right = i ;
            while ( right < length && s[right] == s[right - left] )
                right++ ;
            if ( right - left > m ) {
                m = right - left ;
                z[i] = right - left ;
                shift = i - n - 1 ;
            }
            right-- ;
        }
        else {
            int i1 = i - left ;
            if ( z[i1] < right - i + 1 ) {
                z[i] = z[i1] ;
            }
            else {
                left = i ;
                while ( right < length && s[right] == s[right-left] )
                    right++ ;
                if ( right - left > m ) {
                    m = right - left ;
                    z[i] = right - left;
                    shift = i - n - 1 ;
                }
                right-- ;
            }
        }
    }
}
int main () {
    string a , b ,s ;
    scanf ( "%d" , &n ) ;
    cin >> a >> b ;
    s = a + "$" + b ;
    zalgo ( s ) ;
    printf ( "%d\n" , shift ) ;
    return 0;
}
