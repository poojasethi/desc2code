#include<cstdio>
#include<cstring>
#include<algorithm>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main(){
    int n;
    scanf("%d",&n);
    char s1[n+6],s2[(2*n)+6];
    scanf("%s %s",&s1,&s2);
    s2[2*n]=s2[n];
    for(int i=0;i<n;i++){
        s2[i+n]=s2[i];
    }
    int pre_length[n+1];
    pre_length[0]=0;
    int i=1,j=0;
    while(i<n){
        if(s1[i]==s1[j]){
            pre_length[i]=j+1;
            j++;
            i++;
        }else{
            if(j>0){
            j=pre_length[j-1];
            }
            else{
                pre_length[i]=0;
                i++;
            }
        }
    }

    i=0,j=0;
    int len=0,shift=INF;
    for(int i=0;i<2*n;i++){
        while(j>0 && s1[j]!=s2[i]){
            j=pre_length[j-1];
        }
        if(s1[j]==s2[i]){
            if(j>len){
                shift=i-j;
                len=j;
            }
            else if(j==len){
                shift=min(shift,i-j);
            }
            j++;
            if(j==n){
                break;
            }
        }
    }
    if(shift==INF){
        shift=0;
    }
    printf("%d\n",shift);
}
