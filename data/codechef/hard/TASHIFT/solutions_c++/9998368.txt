#include<bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k;
    int n;
    cin>>n;
    cin.ignore(256,'\n');

    string a,b;

    getline(cin,a);
    getline(cin,b);

    b+=b;

    vector<int> lps(n,0);

    lps[0]=0;

    i=1;j=0;
    while(i<n)
    {
        if(a[i]==a[j])
        {
            j++;

            lps[i]=j;
            i++;
        }

        else
        {
            if(j!=0)
            {
                j=lps[j-1];
            }

            else
            {
                lps[i]=0;
                i++;
            }
        }
    }

    //now, search a in b
    i=0;
    j=0;

    int shifts=0;
    int mx=0;

    while(i<2*n)
    {
        if(b[i]==a[j])
        {
            i++;
            j++;

            if(j>mx)
            {
                mx=j;

                shifts=(i-j)%n;
            }

            else if(j==mx)
            {
                shifts=min(shifts,(i-j)%n);
            }
        }

        if(j==n)
        {
            j=lps[j-1];
        }

        else if(i<2*n && b[i]!=a[j])
        {
            if(j!=0)
                j=lps[j-1];

            else
                i++;
        }
    }

    cout<<shifts<<endl;

    return 0;
}
