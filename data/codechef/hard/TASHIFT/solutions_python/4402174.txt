def border(s):
    dic={}
    dic[0]=-1
    dic[1]=0
    n=len(s)
    j=0
    for i in range(2,n+1):
        while s[i-1]!=s[j] and j!=-1:
            j=dic[j]
        j+=1
        dic[i]=j
    return dic
n=input()
a=raw_input()
b=raw_input()*2
pre=border(a)
m=0
ans=0
cur=0
for i in range(n):
    while a[cur]!=b[i] and cur!=-1:
        cur=pre[cur]
    cur+=1
    if cur>m:
        m=cur
        ans=i-cur+1
