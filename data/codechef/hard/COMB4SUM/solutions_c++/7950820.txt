long long function(vector<long long>v,long long n)
{

      
       long long sum=0;
       for(int i=0;i<n;i++)
       {
            sum+=2*(i*v[i]-(n-i-1)*v[i]);
       
       }

   return sum;
}


int main()
{
    long long t;
    cin>>t;
    long long n;
    while(t--)
    {
        cin>>n;
        long long x;
        vector<long long>v;
        for(int i=0;i<n;i++)
        {
             cin>>x;
             v.push_back(x);
        }
        sort(v.begin(),v.end());
        long long s=function(v,n);
        vector<long long>vv;
        for(int i=0;i<n;i++)
        {
        
              for(int j=1+i;j<n;j++)
              {
              
                    vv.push_back(v[i]+v[j]);
              }
        }
        
     sort(vv.begin(),vv.end());
     long long ss=function(vv,vv.size());
     cout<<ss-(n-2)*s<<"\n";
    }

}
