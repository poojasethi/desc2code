#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

ll func(vector<ll> arr)
{
	ll ret = 0;

	sort(arr.begin(), arr.end());
	int N = arr.size();

	for(int i = 0; i < N; i++)
	{
		ll prd = 2 * arr[i] * ((i) - (N-i-1));
		ret += prd;
	}

	return ret;
}

void solve()
{
	int N;
	scanf("%d", &N);

	vector<ll> arr, brr;
	arr.resize(N);

	for(int i = 0; i < N; i++)
		scanf("%lld", &arr[i]);

	for(int i = 0; i < N; i++)
	{
		for(int j = i+1; j < N; j++)
		{
			ll tsum = arr[i] + arr[j];
			brr.push_back(tsum);
		}
	}

	printf("%lld\n", func(brr) - (N-2) * func(arr));

	return;
}

int main()
{
	int t;
	cin >> t;

	while(t--)
		solve();

	return 0;
}
