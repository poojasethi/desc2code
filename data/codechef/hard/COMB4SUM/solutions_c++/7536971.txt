#include<bits/stdc++.h>
using namespace std;
int arr[1005];
int par[1000005];
long long getAns(int a[],int n)
{
	long long ans=0;
	for(int i=1;i<=n;i++)
	{
		ans+=2*((long long int)(i-1)*a[i]-(long long int)(n-i)*a[i]);
	}
	return ans;
}
int main()
{
	int t,n;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		for(int i=1;i<=n;i++)
		scanf("%d",&arr[i]);
		sort(arr+1,arr+n+1);
		int k=1;
		for(int i=1;i<n;i++)
		{
			for(int j=i+1;j<=n;j++)
			par[k++]=arr[i]+arr[j];
		}
		sort(par+1,par+k);
		long long final=getAns(par,k-1)-(long long)((n-2)*getAns(arr,n));
		printf("%lld\n",final);
	}
}
