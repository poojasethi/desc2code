
#include<bits/stdc++.h>
using namespace std;
#define MAX     1000
#define D(x)    cout << "-> " << x << endl
typedef long long int LL;

const int lim = 200*MAX;
int sq[MAX+11];
int frq[lim+11];

int main()
{
    int i, j, k, t, n, cur;
    LL res;

    scanf("%d", &t);
    while(t--)
    {
        memset(frq, 0, sizeof(frq));
        res = 0;

        scanf("%d", &n);
        for(i = 1; i <= n; i++)
            scanf("%d", &sq[i]);

        for(i = 1; i <= n; i++)
            for(j = 1; j <= n; j++)
                if(i!=j)
                    frq[sq[i]+sq[j]]++;

        for(i = 1; i <= lim; i++)
            frq[i] += frq[i-1];

        for(i = 1; i <= n; i++)
            for(j = i+1; j <= n; j++)
            {
                cur = sq[i] + sq[j];
                res += (LL)frq[cur-1]*cur;
                res -= (LL)(frq[lim] - frq[cur])*cur;

                res -= (LL)abs(sq[i]-sq[j])*2*(n-2);
            }

        printf("%lld\n", res);
    }
    return 0;
}
