#include <bits/stdc++.h>

using namespace std;

int T,N;
int ar[2100];
int F1[450000],F2[450000];
long long G1[450000],G2[450000];
long long res,S1,S2,Z1,Z2;

void upd( int a , int b ){
    for( int i=ar[a]+ar[b]+200000 ; i<=440000 ; i+=(i&-i) ) F1[i]++;
    for( int i=ar[a]+ar[b]+200000 ; i<=440000 ; i+=(i&-i) ) G1[i]+=ar[a]+ar[b];
    S1+=1;
    Z1+=ar[a]+ar[b];

    for( int i=ar[a]-ar[b]+200000 ; i<=440000 ; i+=(i&-i) ) F2[i]++;
    for( int i=ar[a]-ar[b]+200000 ; i<=440000 ; i+=(i&-i) ) G2[i]+=ar[a]-ar[b];
    S2+=1;
    Z2+=ar[a]-ar[b];
}

int main(){

    cin >> T;

    while( T-- ){
        cin >> N;
        res=0;
        S1=S2=Z1=Z2=0;
        memset( F1 , 0 , sizeof F1 );
        memset( F2 , 0 , sizeof F2 );
        memset( G1 , 0 , sizeof G1 );
        memset( G2 , 0 , sizeof G2 );
        for( int i=1 ; i<=N ; i++ ) scanf(" %d",&ar[i]);

        upd( 1 , 2 );

        for( int i=3 ; i<N ; i++ ){

            for( int j=i+1 ; j<=N ; j++ ){
                int sze=0;
                long long sm=0;
                for( int k=ar[i]+ar[j]+200000 ; k ; k-=( k&-k ) ) sze+=F1[k];
                for( int k=ar[i]+ar[j]+200000 ; k ; k-=( k&-k ) ) sm+=G1[k];
                res+=1LL*sze*(ar[i]+ar[j])-sm;
                res+=(Z1-sm)-1LL*(S1-sze)*(ar[i]+ar[j]);
            }

            for( int j=i+1 ; j<=N ; j++ ){
                int sze=0;
                long long sm=0;
                for( int k=ar[j]-ar[i]+200000 ; k ; k-=( k&-k ) ) sze+=F2[k];
                for( int k=ar[j]-ar[i]+200000 ; k ; k-=( k&-k ) ) sm+=G2[k];
                res+=1LL*sze*(ar[j]-ar[i])-sm;
                res+=(Z2-sm)-1LL*(S2-sze)*(ar[j]-ar[i]);
            }

            for( int j=i+1 ; j<=N ; j++ ){
                int sze=0;
                long long sm=0;
                for( int k=ar[i]-ar[j]+200000 ; k ; k-=( k&-k ) ) sze+=F2[k];
                for( int k=ar[i]-ar[j]+200000 ; k ; k-=( k&-k ) ) sm+=G2[k];
                res+=1LL*sze*(ar[i]-ar[j])-sm;
                res+=(Z2-sm)-1LL*(S2-sze)*(ar[i]-ar[j]);
            }
            for( int j=i-1 ; j ; j-- ) upd( j , i );
        }

        cout << res*2LL << endl;

    }

    return 0;
}
