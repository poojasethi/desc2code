#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<vector>

#define pb push_back
using namespace std;

typedef long long int ll;

ll get(vector<int> &changed)
{
	int n = changed.size();

	ll sum = 0;
	
	for(int i=1;i<=n;i++){
		ll V = changed[i-1];
		sum = sum + (ll)(i-1)*V - (ll)(n-i)*V;
	}
	
	return sum*(ll)2;
}

void solve()
{
	vector<int> orig,changed;
	int N;
	scanf("%d",&N);
	for(int i=1;i<=N;i++){
		int x;
		scanf("%d",&x);
		orig.pb(x);
	}
	sort(orig.begin(),orig.end());
	for(int i=0;i<N;i++){
		for(int j=i+1;j<N;j++){
			changed.pb(orig[i]+orig[j]);
		}
	}
	sort(changed.begin(),changed.end());

	ll sum1 = get(changed);
	ll sum2 = get(orig);

	printf("%lld\n",sum1 - sum2*(ll)(N-2));
}

int main()
{
	int T;
	scanf("%d",&T);
	for(int i=1;i<=T;i++){
		solve();
	}
	return 0;
}
