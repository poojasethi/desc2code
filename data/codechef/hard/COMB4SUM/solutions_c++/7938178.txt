#include<cstdio>
#include<iostream>
#include<vector>
#include<set>
#include<algorithm>
#define pii pair< int , int >
#define piii pair< int , pii >
using namespace std;
int a[1003],cntF[1003],cntG[1003];
long long int f[1003],g[1003];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        for(int i=1; i<=n; i++)
        {
            cin>>a[i];
            f[i]=0;
            g[i]=0;
            cntF[i]=0;
            cntG[i]=0;
        }
        vector<piii> v;
        for(int i=1; i<=n; i++)
        {
            for(int j=i+1; j<=n; j++)
            {
                int val=a[i]+a[j];
                v.push_back(piii(val,pii(i,j)));
            }
        }
        sort(v.begin(),v.end());

        long long int ans=0;
        long long int sum=0;
        for(int i=0; i<v.size(); i++)
        {
            int val,x,y;

            val=v[i].first;
            x=v[i].second.first;
            y=v[i].second.second;

            //printf("x and y are %d %d %d\n",x,y,val);

            long long int cnt=i-cntF[x]-cntF[y]-cntG[x]-cntG[y];
            long long int p=cnt;
            p*=val;


            long long int q=sum-f[x]-f[y]-g[x]-g[y];

            //printf("p and q are %lld %lld\n",p,q);

            ans+=(p-q);

            f[x]+=val;
            cntF[x]++;

            g[y]+=val;
            cntG[y]++;

            sum+=val;
        }
        printf("%lld\n",2*ans);
    }
    return 0;
}
