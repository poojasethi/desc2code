#include<bits/stdc++.h>

#define s(a) scanf("%lld",&a)
#define ss(a) scanf("%s",a)

#define MP           make_pair
#define PB           push_back
#define REP(i, n)    for(int i = 0; i < n; i++)
#define INC(i, a, b) for(int i = a; i <= b; i++)
#define DEC(i, a, b) for(int i = a; i >= b; i--)
#define CLEAR(a)     memset(a, 0, sizeof a)

using namespace std;

typedef long long int          LL;
typedef unsigned long long ULL;
typedef vector<long long int>        VI;
typedef pair<int, int>     II;
typedef vector<II>         VII;

LL inp[2005];
int main()
{
      LL t;
      s(t);
      while(t--){
            LL n;
            s(n);
            assert(n>=4);
            LL ans = 0,minus=0;
            VI v;
            REP(i,n)
                  s(inp[i]);
            sort(inp,inp+n);
            REP(i,n)
                  minus += (n-2)*(2*i-n+1)*inp[i];
            REP(i,n)
            INC(j,i+1,n-1)
                  v.PB(inp[i]+inp[j]);
            sort(v.begin(),v.end());
            int len = v.size();
            assert(len==n*(n-1)/2);
            REP(i,len)
                  ans += (2*i-len+1)*v[i];
            ans = ans-minus;
            printf("%lld\n",2*ans);
            v.clear();
      }
      return 0;
}
