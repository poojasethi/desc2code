#include <bits/stdc++.h>
#include <iostream>
#include <cstdio>
using namespace std;

int occ_count[1001];
int a_values[1001];

class compare{
public:
	bool operator()(pair<int, pair<int, int> > a, pair<int, pair<int, int> > b){
		return a.first >= b.first;
	}
};

int main(){
	int t;
	cin >> t;
	while(t--){
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; ++i){
			int a;
			scanf("%d", &a);
			a_values[i] = a;
		}

		vector< pair<int, pair<int, int> > > pair_sums;
		int iter = 0;
		for (int i = 0; i < n; ++i){
			for(int j = i+1; j < n; j++){
				int sum = a_values[i] + a_values[j];
				pair_sums.push_back(make_pair(sum, make_pair(i, j)));
				iter++;
			}
		}

		sort(pair_sums.begin(), pair_sums.end(), compare());

		for(int i = 0; i < n; ++i){
			occ_count[i] = 0;
		}

		int other_pairs = ((n-2)*(n-3))/2;

		long long int ans = 0;
		for (int i = 0; i < iter; ++i){
			int index = i;
			pair<int, pair<int, int> > curr = pair_sums[index];
			int pairs_ahead = i - occ_count[(curr.second).first] - occ_count[(curr.second).second];
			long long int multiplier = other_pairs - 2 * pairs_ahead;
			occ_count[(curr.second).second]++;
			occ_count[(curr.second).first]++;
			// cout << multiplier << " " << curr.first << endl;
			ans += multiplier*curr.first;
 		}
 		// cout << "ans: ";
 		printf("%lld\n", 2*ans);
	}
}
