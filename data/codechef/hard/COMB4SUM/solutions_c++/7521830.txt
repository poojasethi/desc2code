// a is sorted
int64_t get_diff(vector<int> &a)
{
	int64_t ans = 0;
	int64_t sum = 0;

	for (int i = 0; i < a.size(); i++)
	{
		ans += (int64_t) i * (int64_t) a[i] - sum;
		sum += (int64_t) a[i];
	}

	return ans;
}

int main()
{
	int T;
	scanf("%d", &T);

	for (int t = 0; t < T; t++)
	{
		int N;
		scanf("%d", &N);

		vector<int> a(N, 0);
		for (int i = 0; i < N; i++)
		{
			int x;
			scanf("%d", &x);

			a[i] = x;
		}

		std:sort(a.begin(), a.end());

		int n = (N * (N - 1)) / 2;

		vector<int> b;
		b.reserve(n);

		for (int i = 0; i < N; i++)
			for (int j = i + 1; j < N; j++)
				b.push_back(a[i] + a[j]);

		std::sort(b.begin(), b.end());

		int64_t ans = get_diff(b);
		int64_t u = get_diff(a);

		ans -= (int64_t) (N - 2) * u;
		ans *= 2;

		printf("%lld\n", ans);
	}	

	return 0;
}
