#include <bits/stdc++.h>

typedef long long ll;
using namespace std;

ll a[1010],b[1000010];
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    int i,j,t,n,m;
    ll sum1,sum2;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++) cin>>a[i];
        sum1=sum2=0;
        sort(a,a+n);
        for(i=0;i<n;i++)
            sum1+=2LL*(i*a[i]-(n-1-i)*a[i]);
        m=0;
        for(i=0;i<n;i++)
            for(j=i+1;j<n;j++)
                b[m++]=(a[i]+a[j]);
        sort(b,b+m);
        for(i=0;i<m;i++)
            sum2+=2LL*(i*b[i]-(m-i-1)*b[i]);
        cout<<sum2-(n-2)*sum1<<'\n';
    }
    return 0;
}
