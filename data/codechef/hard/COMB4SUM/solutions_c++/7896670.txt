#include<cstdio>
#include<algorithm>
using namespace std;
long long func (int a[],int n)
{
	int i;
	long long sum=0;
	for (i=0;i<n;i++)
		sum+=(long long)(2*i-n+1)*(long long)a[i];
	return sum*(long long)2;
}
int main()
{
	int t,n,N,i,j,k;
	scanf("%d",&t);
	while (t--)
	{
		scanf("%d",&n);
		int a[n];
		for (i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		N=n*(n-1)/2;
		int A[N];
		k=0;
		for (i=0;i<n;i++)
			for (j=i+1;j<n;j++)
					A[k++]=a[i]+a[j];
		sort(A,A+N);
		printf("%lld\n",func(A,N)-(long long)(n-2)*func(a,n));
	}
	return 0;
}
