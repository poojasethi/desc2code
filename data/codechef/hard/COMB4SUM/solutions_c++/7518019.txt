const int N = 100010;
vi a;
vi na;
void solve()
{
	
	int n;
	cin >> n;
	a.resize(n);
	for(int i = 0; i < n; ++i)
		cin >> a[i];
	
	for(int i = 0; i < n; ++i)
	{
		for(int j = i + 1; j < n; ++j)
		{
			int t = a[i] + a[j];
			na.push_back(t);
			if (na.size() > 5 * 1e5)
				break;
		}
	}
	
	sort(a.begin(), a.end());
	sort(na.begin(), na.end());
	int64 ans = 0;
	for(int i = 0; i < n; ++i)
		ans -= a[i] *(int64) (2 * i - n + 1); 
	//cerr << ans << endl;
	ans = ans * (n - 2);
	//cerr << ans << endl;
	for(int i = 0; i < na.size(); ++i)
		ans += na[i] * (int64) (2 * i - (int)na.size() + 1);
 
	cout << 2 * ans << endl;
	a.clear();
	na.clear();
}
