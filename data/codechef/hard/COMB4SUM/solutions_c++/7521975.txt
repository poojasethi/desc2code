ll a[2001];
ll pairs[1000001];
void solve()
{
    
    ll ans=0;
    int n;
    s(n);
    for(int i=0;i<n;i++)
    {
        sl(a[i]);
    }
    sort(a,a+n);
    ll ans1=0;
    for(int i=0;i<n;i++)
    {
        ans1  = ans1+(2*i+1-n)*a[i];
    }
    int len=0;
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            pairs[len++] = a[i]+a[j];

        }
    }
    sort(pairs,pairs+len);
    for(int i=0; i<len; i++) 
    {
        ans = ans + (2*i+1-len)*pairs[i];
    }
    printf("%lld\n",2*(ans-(n-2)*ans1));
    //cout<<ans<<" "<<ans2<<" "<<ans3<<endl;
}
int main()
{
    int t;
    s(t);
    for(int i=0;i<t;i++)
        solve();
}
