#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        long long n,in=0,ans=0,temp=0,len;
        scanf("%lld",&n);
        len=(n*(n-1))/2;
        long long arr[n],arr1[n],arr2[len],arr3[len];
        for(long long i=0;i<n;i++)
            scanf("%lld",&arr[i]);
        sort(arr,arr+n);
        arr1[0]=arr[0];
        for(long long i=1;i<n;i++)
            arr1[i]=arr1[i-1]+arr[i];
        for(long long i=0;i<n;i++){
            for(long long j=i+1;j<n;j++)
                arr2[in++]=arr[i]+arr[j];
        }
        sort(arr2,arr2+len);
        arr3[0]=arr2[0];
        for(long long i=1;i<len;i++)
            arr3[i]=arr3[i-1]+arr2[i];
        for(long long i=len-1;i>0;i--)
            ans+=i*arr2[i]-arr3[i-1];
        for(long long i=1;i<n;i++){
            temp+=(n-i-1)*(i*arr[i]-arr1[i-1]);
            temp+=(i-1)*(i*arr[i]-arr1[i-1]);
        }
        ans-=temp;
        ans*=2;
        printf("%lld\n",ans);
    }
}
