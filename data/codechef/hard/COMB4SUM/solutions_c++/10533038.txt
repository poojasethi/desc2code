#include <iostream>
#include <cstdio>
#include <algorithm>
#define ll long long
using namespace std;


ll t,n,u,ans;
ll arr[1005];
ll pairsum[1000005];

ll method(ll ar[],ll k){
	ll sum = 0;
	
	for(ll i=0;i<k;i++){
		ll temp = i*ar[i] - (k-1-i)*ar[i];
		sum += temp;
	}
	return sum;
	
}

int main() {
	
	scanf("%lld",&t);
	
	while(t--){
	
		scanf("%lld",&n);
		
		for(ll i=0;i<n;i++)
			scanf("%lld",arr+i);
		
		sort(arr,arr+n);
		u = 0;
		for(ll i=0;i<n;i++){
			for(ll j=i+1;j<n;j++){
				pairsum[u] = arr[i]+arr[j];
				u++;
			}
		}
		
		sort(pairsum,pairsum+u);
		
		ans = method(pairsum,u)-(n-2)*method(arr, n);
		printf("%lld\n",ans*2);
		
	}
	
	return 0;
}
