#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<cstring>
#include<vector>
using namespace std;
#define ll long long
int cs,n,i,j;
ll f(vector<ll> A)
{
  ll n,i,s=0;
  sort(A.begin(),A.end());
  n=A.size();
  for(i=0;i<n;i++)
    s+=(i-(n-i-1))*A[i];
  return 2*s;
}
int main()
{
  scanf("%lld",&cs);
  while(cs--)
  {
    scanf("%lld",&n);
    vector<ll> A(n),V;
    for(i=0;i<n;i++)
      scanf("%lld",&A[i]);
    for(i=0;i<n;i++)
      for(j=i+1;j<n;j++)
      V.push_back(A[i]+A[j]);
    printf("%lld\n",f(V)-(n-2)*f(A));
  }
  return 0;
}
