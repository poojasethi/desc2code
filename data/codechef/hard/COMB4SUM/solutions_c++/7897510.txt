#include<iostream>
#include<algorithm>
using namespace std;
long long int temp[1000007],y;
long long int gen(long long int *a,long long int n)
{
long long int sum=0;
for(long long int i=1;i<=n;i++)
sum=sum+2*(2*i-n-1)*a[i-1];
return sum;
}

long long int func(long long int *a,long long int n)
{
sort(a,a+n);
for(int i=0;i<n;i++)
for(int j=i+1;j<n;j++)
temp[y++]=a[i]+a[j];
sort(temp,temp+y);
long long int ans=gen(temp,y)-(n-2)*gen(a,n);
return ans;
}

int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
y=0;
long long int n;
cin>>n;
long long int a[n];
for(int i=0;i<n;i++)
cin>>a[i];
cout<<func(a,n)<<endl;
}
return 0;
}
