#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
long long sum[1000001];
long long a[10001];
int main()
{
	int t; cin>>t;
	while(t--)
	{
		int n; cin>>n;
		for(int i=0;i<n;i++)
		cin>>a[i];
		sort(a,a+n);
		long long ans1=0;
		for(int i=0;i<n;i++)
		{
			ans1+=(2*i+1-n)*a[i];
		}
		ans1*=2;
		int p=0;
		for(int i=0;i<n;i++)
		{
			for(int j=i+1;j<n;j++)
			{
				sum[p++]=a[i]+a[j];
			}
		}
		sort(sum,sum+p);
		long long ans2=0;
		for(int i=0;i<p;i++)
		{
			ans2+=(2*i+1-p)*sum[i];
		}
		ans2*=2;
		printf("%lld\n",ans2-(n-2)*ans1);
	
	}
	return 0;
}
