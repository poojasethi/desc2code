#include <cstdio>
#include <vector>
#include <algorithm>
using std::vector;
using std::sort;
#define LL long long int

int a[2005000];
    int i,j;
int main() {
  int n,t;
  scanf("%d", &t);
  while (t--) {
    scanf("%d", &n);
    for (i = 0; i < n; i++) {
      scanf("%d", &a[i]);
    }
    sort(a, a + n);
    vector<LL> sum;
    LL diffsum = 0;
    LL tsum = 0;
    for (i = 0; i < n; i++) {
      for (j = 0; j < n; j++) {
        if (i == j)
          continue;
        sum.push_back(a[i] + a[j]);
        tsum += a[i] + a[j];
        diffsum += abs(a[i] - a[j]);
      }
    }
    diffsum *= (n - 2) * 4;
    diffsum *= -1;
    sort(sum.begin(), sum.end());
    LL psum = 0;
    n = sum.size();
    for (i = 0; i < n; i++) {
      psum += sum[i];
      diffsum += (sum[i] * (i + 1) - psum);
      diffsum += (tsum - psum - (n - i - 1) * sum[i]);
    }
    printf("%lld\n", diffsum/4);
  }
  return 0;
}
