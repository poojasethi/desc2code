#include <iostream>
#include <algorithm>
using namespace std;
long long int func(long long int A[],long long int n)
{
 long long int i;
 long long int sum=0;
 for (i=1;i<=n;i++)
 {
  long long int temp=2*(((i-1)*A[i])-((n-i)*A[i]));
  sum+=temp;
  


 }


 return sum;
}



int main()
{
 long long int n;
 long long int i,j;
 int t;
  
 cin>>t;
 while (t--)
 {

 long long int A[1003]={0},B[1000006]={0};
 
 cin>>n;
 A[0]=0;
 
 for (i=1;i<=n;i++)
 {
   cin>>A[i];
 }
 sort(A+1,A+n+1);
 

 long int count=1;
 B[0]=0;
 for (i=1;i<=n;i++)
 {
   for (j=i+1;j<=n;j++)
   {
      B[count++]=A[i]+A[j];

   }

 }
 
 count--;
 sort(B,B+count+1);



 //cout<<func(B,count)<<endl;
 cout<<func(B,count)-(n-2)*func(A,n)<<endl;
 }
 return 0;
}
