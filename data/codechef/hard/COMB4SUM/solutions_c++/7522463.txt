#include<bits/stdc++.h>
using namespace std;
#define ll unsigned long long int
#define br printf("\n")
#define loop(i,k,n) for(int i=k;i<n;i++)
#define loopr(i,k,n) for(int i=k;i>=n;i--)
#define arrin(arrname,size) loop(i,0,size) cin>>arrname[i]
#define fastinput() ios_base::sync_with_stdio(false)
#define mod 1000000007
long long sum[10000000];
int main()
{
	fastinput();
    int t;
    cin>>t;
    while(t--)
    {
      int n; cin>>n;
      long long a[n];
	  arrin(a,n);
	  int p=0;
	  sort(a,a+n);
	  loop(i,0,n)	
	  {
	  	loop(j,i+1,n)
	  	{
	  			sum[p++]=a[i]+a[j];
				  }
	  }
	  long long ans=0;
	  sort(sum,sum+p);
	  
	  loop(i,0,p)
	  {
    	ans+=i*sum[i] -(p-1-i)*sum[i] ;
	  }
	  ans*=2;
	  long long ans2=0;
	  loop(i,0,n)
	  {
	  	ans2+=i*a[i] -(n-1-i)*a[i] ;
	  }
	  ans2*=2;
	printf("%lld\n",ans-(n-2)*ans2);
}
	return 0;
}
