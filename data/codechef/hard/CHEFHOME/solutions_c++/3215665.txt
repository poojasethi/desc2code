#include<iostream>
#include<cstdio>
#include<ctime>
#include<algorithm>
#include<cassert>
#define MXN 100005
#define MXV 100000000
using namespace std;
 
typedef long long LL;
 
int X[MXN];
int Y[MXN];
int t,n;
 
int main()
{
    scanf("%d",&t);
 
    while(t-- >0)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            scanf("%d %d",&X[i], &Y[i]);
            assert(X[i]>=-MXV && X[i]<=MXV);
            assert(Y[i]>=-MXV && Y[i]<=MXV);
        }
 
        sort(X,X+n);
        sort(Y,Y+n);
        
        int x = X[n/2] - X[(n-1)/2] + 1;
        int y = Y[n/2] - Y[(n-1)/2] + 1;
        LL ans = 1ll*x*y;
 
        printf("%lld\n",ans);
 
    }
    return 0;  
}
