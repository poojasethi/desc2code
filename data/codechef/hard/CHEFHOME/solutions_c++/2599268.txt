using namespace std;

int main(int argc, char const *argv[])
{
	int t;
	scanf("%d",&t);
	while (t--)
	{
		int n;
		scanf("%d",&n);

		vector<int> x;
		vector<int> y;
		for (int i = 0; i < n; ++i)
		{
			int a,b;
			scanf("%d %d", &a, &b);
			x.push_back(a);
			y.push_back(b);
		}

		sort(x.begin(),x.end());
		sort(y.begin(),y.end());

		if(n%2==1)
			printf("1\n");
		else
		{
			int num_x = x[n/2] - x[n/2 - 1] +1;
			int num_y = y[n/2] - y[n/2 - 1] +1;

			printf("%llu\n", (unsigned long long)num_x*(unsigned long long)num_y);
		}
	}
	return 0;
}
