#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long ll;

int main(void)
{
	int t, m, n, x, y;
	vector<int> X, Y;
	cin >> t;

	while (t--)
	{
		X.clear();
		Y.clear();
		cin >> n;
		m = n;
		while(n--){
			cin >> x >> y;
			X.push_back(x);
			Y.push_back(y);
		}
		sort(X.begin(), X.end());
		sort(Y.begin(), Y.end());
		
		ll a, b;
		a = (X[m/2] - X[(m-1)/2] + 1);
		b = (Y[m/2] - Y[(m-1)/2] + 1);

		cout <<  a * b <<endl;
	}

	return 0;
}
