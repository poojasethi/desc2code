#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
const int MAX=1010;
long long y[MAX],x[MAX];
int main(){
    int n,i,t;
    scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		for(i=0;i<n;i++)scanf("%lld%lld",y+i,x+i);
		if(n&1){
			printf("1\n");
			continue;
		}
		sort(y,y+n);
		sort(x,x+n);
		printf("%lld\n",(1+y[n/2]-y[n/2-1])*(1+x[n/2]-x[n/2-1]));
	}
    return 0;
}
