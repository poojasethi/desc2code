#include<iostream>
#include<cstdio>
#include<ctime>
#include<algorithm>
#include<cassert>
#define MXN 100005
#define MXV 100000000
using namespace std;

typedef long long ll;

int X[MXN];
int Y[MXN];
int t,n;

int main()
{
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            scanf("%d %d",&X[i],&Y[i]);
        }

        sort(X,X+n);
        sort(Y,Y+n);

        int x=X[n>>1]-X[(n-1)>>1]+1;
        int y=Y[n>>1]-Y[(n-1)>>1]+1;
        ll ans=1ll*x*y;

        printf("%lld\n",ans);
    }
    return 0;
}
