#include <algorithm>
#include <cstdio>

using namespace std;

#define wile(n) while(n--) 
#define FOR(i,a,b) for ( int i=a; i<b; i++)
#define ll long long
#define si(n) scanf("%d",&n)
#define prli(n) printf("%lld\n",n)
 
using namespace std;
 
int x[1000], y[1000];
 
int main()
{	int t, n;
	si(t);
	wile(t)
	{	si(n);
		FOR(i,0,n)
			si(x[i]), si(y[i]);
		sort(x, x+n);
		sort(y, y+n);
		ll ans = 0;
		if ( n & 1)		// odd count means only 1 median!!
			ans = 1;
		else
		{	int dx = x[n>>1] - x[(n>>1)-1] + 1;
			int dy = y[n>>1] - y[(n>>1)-1] + 1;
			ans = (long long) dx * dy;
		}
		prli(ans);
	}
	return 0;
}
