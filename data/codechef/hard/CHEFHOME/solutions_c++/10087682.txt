#include<bits/stdc++.h>
using namespace std;
#define ll long long
int x[1003];
int y[1003];
int main()
{
	int t,n,a,b;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(int i=0;i<n;i++)
           cin>>x[i]>>y[i];
        sort(x,x+n);
        sort(y,y+n);
        if(n%2==0)
        {
        a = x[n/2] - x[(n-1)/2] + 1;
        b =  y[n/2] - y[(n-1)/2] + 1;
        }
        else
        {
        a=1;
        b=1;
        }
        ll ans = 1ll*a*b;
        cout<<ans<<endl;
    }
}
