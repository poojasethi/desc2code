#include<iostream>
#include<cstdio>
#include<ctime>
#include<algorithm>
#include<cassert>

using namespace std;
 
typedef long long LL;
 
int X[100005];
int Y[100005];
int t,n;
 
int main()
{
    scanf("%d",&t);
 
    while(t-- >0)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
        {
            scanf("%d %d",&X[i], &Y[i]);
//            assert(X[i]>=-MXV && X[i]<=MXV);
//            assert(Y[i]>=-MXV && Y[i]<=MXV);
        }
 
        sort(X,X+n);
        sort(Y,Y+n);
        
        LL x = X[n/2] - X[(n-1)/2] + 1;
        LL y = Y[n/2] - Y[(n-1)/2] + 1;
        
        printf("%lld\n",x*y);
 
    }
    return 0;  
}
