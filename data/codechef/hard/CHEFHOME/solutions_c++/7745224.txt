LL n,x[1010],y[1010];

LL fun2()
{
    LL mid = n/2;
    LL a = x[mid] - x[mid-1] + 1;
    LL b = y[mid] - y[mid-1] + 1;
    return a*b;
}

int main()
{
    LL t,T,i;
    sl(T);
    rep(t,T)
    {
        sl(n);
        Rep(i,n)

            sl(y[i]);

        sort(x,x+n);
        sort(y,y+n);
        LL ans;
        if(n&1)ans = 1;
        else ans = fun2();
