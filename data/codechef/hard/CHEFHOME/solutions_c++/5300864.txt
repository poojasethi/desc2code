#include<stdio.h>
#include<stdlib.h>

int compare (const void * a, const void * b)
{
  return ( *(int*)a - *(int*)b );
}
int main()
{
	int T,N,X,Y,f;
	int arrX[1001],arrY[1001];
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&N);
		f=N%2;
		for(int i=0;i<N;i++)
		{
			scanf("%d %d",&X,&Y);
			if(f==0)
			{
				arrX[i]=X;
				arrY[i]=Y;
			}
		}

		if(f>0)
		{
			printf("1\n");
		}
		else
		{
			qsort (arrX, N, sizeof(int), compare);
			qsort (arrY, N, sizeof(int), compare);
			long long int mx1=arrX[N/2-1];
			long long int mx2=arrX[N/2];
			long long int my1=arrY[N/2-1];
			long long int my2=arrY[N/2];

			long long int  dx=mx2-mx1;
			if(dx<0) dx=-1*dx;

			long long int dy=my2-my1;
			if(dy<0) dy=-1*dy;

			long long int r=(dy+1)*(dx+1);
			printf("%lld\n",r);
		}
	}
	return 0;
}
