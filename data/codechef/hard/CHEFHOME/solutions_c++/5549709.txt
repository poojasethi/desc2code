#include<iostream>
#include<algorithm>
#define ll long long
using namespace std;

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        ll x[n],y[n];
        for(int i=0;i<n;i++)
        {
            cin>>x[i]>>y[i];
        }

        sort(x,x+n);
        sort(y,y+n);
        ll l = x[n/2]-x[(n-1)/2]+1;
        ll m = y[n/2]-y[(n-1)/2]+1;
        cout<<l*m<<endl;


    }
}
