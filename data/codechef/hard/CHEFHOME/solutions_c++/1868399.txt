#include<stdio.h>
#include<assert.h>
#include<algorithm>
#define int64 long long

int64 X[1000];
int64 Y[1000];

int main()
{
	int T,N;
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&N);
		for(int i=0; i<N; ++i) scanf("%lld%lld",&X[i],&Y[i]);
		if(N&1) printf("1\n");
		else
		{
			std::sort(X,X+N);
			std::sort(Y,Y+N);
			printf("%lld\n",(X[N/2]-X[N/2-1]+1)*(Y[N/2]-Y[N/2-1]+1));
		}
	}
	return 0;
}
