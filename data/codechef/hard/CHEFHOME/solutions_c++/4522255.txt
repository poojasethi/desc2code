#include<bits/stdc++.h>



using namespace std;
bool compare(int a,int b)
{
    return a < b;
}
int main()
{
    int t,n,i,x,y;
    cin >> t;
    while(t--)
    {
        cin >> n;
        vector<int> X(n);
        vector<int> Y(n);
        for(i=0;i<n;i++)
        {
            cin >> x >> y;
            X[i]=x;
            Y[i]=y;
        }
        sort(X.begin(),X.end(),compare);
        sort(Y.begin(),Y.end(),compare);
/*        for(vector<int>::iterator it=X.begin();it!=X.end();it++)
            cout << *it << ' ';
        cout << '\n';*/
        long long int cnt_x,cnt_y;
        if(n%2!=0)
        {
            cnt_x=1;
            cnt_y=1;
        }
        else
        {
            cnt_x=X[n/2]-X[(n/2)-1]+1;
            cnt_y=Y[n/2]-Y[(n/2)-1]+1;
        }
        cout << cnt_x*cnt_y << '\n';
    }
    return 0;
}
