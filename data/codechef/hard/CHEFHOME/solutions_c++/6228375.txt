#include<iostream>
#include<algorithm>
#include<cstdio>

using namespace std;

#define getcx getchar_unlocked
int inp( ) {
int n=0;
int ch=getcx(),sign=1;
while( ch < '0' || ch > '9' ){
if(ch=='-')sign=-1;ch=getcx();
}
while( ch >= '0' && ch <= '9' )
n = (n<<3)+(n<<1) + ch-'0', ch=getcx(); return n*sign; }
int t,n,sx[1001],sy[1001],i,b;
long long int a;
int main()
{t = inp();
while(t--)
{
n = inp();
for(i=0;i<n;i++)
{
sx[i] = inp();
sy[i] = inp();
}
sort(sx,sx+n);
sort(sy,sy+n);
if(n&1)
{
a = 1;
b = 1;
}
else
{
a = sx[n>>1] - sx[(n-1)>>1] + 1;
b = sy[n>>1] - sy[(n-1)>>1] + 1;
}
printf("%lld\n",a*b);
}
return 0;
}
