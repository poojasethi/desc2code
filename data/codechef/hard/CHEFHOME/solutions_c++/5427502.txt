void a_home_for_chef()
{
    int T,i,k,n;
    long long x1,y1,x2,y2;
    vector<long long> x,y;

    cin >> T;
    for(k = 0; k < T; ++k)
    {
        cin >> n;
        x.resize(n);
        y.resize(n);
        for(i = 0; i < n; ++i)
            scanf("%lld %lld", &x[i], &y[i]);
        if(n % 2 == 1)
            cout << 1 << endl;
        else
        {
            sort(x.begin(), x.end());
            sort(y.begin(), y.end());
            
            i  = n / 2;
            x1 = x[i-1];
            x2 = x[i];
            y1 = y[i-1];
            y2 = y[i];
            cout << (x2-x1+1)*(y2-y1+1) << endl;
        }
    }
}

int main()
{
    a_home_for_chef();
    return 0;
}
