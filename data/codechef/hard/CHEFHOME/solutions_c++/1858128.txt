# include <iostream>
# include <stdio.h>
# include <algorithm>
# include <cstring>
using namespace std;
long long x[1001], y[1001];
int main()
{
    long long t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(long long i=0; i<n; i++)
            cin>>x[i]>>y[i];
        if(n&1)
        {
            cout<<1<<endl;
        }
        else
        {
             sort(x,x+n);sort(y,y+n);
             long long first = x[n/2], last = x[n/2-1];
             long long sum = first-last+1;
             first = y[n/2]; last = y[n/2-1];
             long long sum2 = first-last+1;
             sum = sum*(sum2);
             cout<<sum<<endl;
        }
    }
return 0;
}
