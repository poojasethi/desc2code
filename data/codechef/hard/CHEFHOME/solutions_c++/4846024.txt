#include <cstdio>
#include <algorithm>

using namespace std;

int main()
{
    int t;
    int x[1001],y[1001],n,u,v,i;

    scanf("%d",&t);

    while(t--){
        long long ans =1;
        scanf("%d",&n);
        u=0;
        v=0;
        for(i =0 ;i<n;i++){
            scanf("%d %d",x+i,y+i);
        }

        sort(x,x+n);
        sort(y,y+n);

        u= x[n/2]-x[(n-1)/2]+1;
        v= y[n/2]-y[(n-1)/2]+1;
        ans *= (long long)u*(long long)v;
        printf("%lld\n",ans);
    }


    return 0;
}
