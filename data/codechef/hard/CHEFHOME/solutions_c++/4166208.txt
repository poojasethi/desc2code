#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
long long int a[1000006],b[1000006];
int main()
{
	long long int f,n,i,x,y,ans;
	scanf("%lld",&f);
	while(f--)
	{
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		scanf("%lld %lld",&a[i],&b[i]);
		if(n%2==1)
		printf("1\n");
		else
		{
			sort(a,a+n);
			sort(b,b+n);
			x=a[n/2]-a[n/2-1]+1;
			y=b[n/2]-b[n/2-1]+1;
			ans=x*y;
			printf("%lld\n",ans);
		}
	}
	return 0;
}
