#include<iostream>
#include<stdio.h>
#include<algorithm>
#define ll long long
using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        ll x[n],y[n];
        for(int i=0;i<n;i++)
        {
            scanf("%lld%lld",&x[i],&y[i]);
        }

        sort(x,x+n);
        sort(y,y+n);
        ll l = x[n/2]-x[(n-1)/2]+1;
        ll m = y[n/2]-y[(n-1)/2]+1;
        l = l*m;
        printf("%lld\n",l);


    }
}
