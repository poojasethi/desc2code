#include<iostream>
#include<stdio.h>
#include<algorithm>

using namespace std;

typedef long long LL;

int X[100005];
int Y[100005];

int main()
{
    int t, n, x, y;
    LL ans;
    scanf("%d",&t);

    while(t--)
    {
        scanf("%d",&n);
        for(int i=0;i<n;i++)
            scanf("%d %d",X+i, Y+i);

        sort(X,X+n);
        sort(Y,Y+n);

        x = X[n/2] - X[(n-1)/2] + 1;
        y = Y[n/2] - Y[(n-1)/2] + 1;
        ans = 1ll*x*y;

        printf("%lld\n",ans);

    }
    return 0;
}
