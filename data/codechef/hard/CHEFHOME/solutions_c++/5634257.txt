#include <stdio.h>
#include <algorithm>
using namespace std;
int main(){
	int t;scanf("%d",&t);
	while(t--){
			int n;scanf("%d",&n);
			long long int X[n],Y[n];
			for(int i=0;i<n;i++) scanf("%lld %lld",&X[i],&Y[i]);
		    sort(X,X+n);
		    sort(Y,Y+n);
		    long long int ans;
		    if(n%2!=0) ans=1;
			else ans=((X[(n/2)]-X[(n/2)-1])+1)*(1+(Y[(n/2)]-Y[(n/2)-1]));
			printf("%lld\n",ans);
	}
}
