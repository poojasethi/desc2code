#include<bits/stdc++.h>
using namespace std;
#define F(i,a,b) for(int i=(int)(a);i<(int)(b);i++)
#define RF(i,a,b) for(int i=(int)(a);i>=(int)(b);i--)
#define ll long long int
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        ll n,a,b;
        cin>>n;
        ll x[n+2],y[n+2];
        F(i,0,n)
        {
            cin>>x[i];
            cin>>y[i];
        }
        if(n&1)
        {
            cout<<1<<endl;
            continue;
        }
        else
        {
            sort(x,x+n);
            sort(y,y+n);
            a=x[n/2]-x[n/2-1];
            a++;
           b=y[n/2]-y[n/2-1];
            b++;
        }
            cout<<a*b<<endl;
    }
    return 0;
}
