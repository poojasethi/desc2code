#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t, n, i;
    scanf("%d", &t);
    while(t--){
        scanf("%d", &n);
        int x[n], y[n];
        long long sumx, sumy;
        sumx=sumy=0;
        for(i=0; i<n; i++)
            scanf("%d%d", &x[i], &y[i]);
        sort(x, x+n);
        sort(y, y+n);
        if(n%2)
            printf("1\n");
        else{
            sumx = x[(n/2)]-x[(n/2)-1]+1;
            sumy = y[(n/2)]-y[(n/2)-1]+1;
            long long ans = sumx*sumy;
            printf("%lld\n", ans);
        }
    }
    return 0;
}
