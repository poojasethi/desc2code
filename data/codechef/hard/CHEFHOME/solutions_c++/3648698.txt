#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
int x[1001], y[1001];
main()
{
    //freopen("input.txt", "r", stdin);
    //freopen("output.txt", "w", stdout);
    int T, N;
    scanf("%d", &T);
    while (T--)
    {
        scanf("%d", &N);
        for (int i = 1; i <= N; i++)
        {
            scanf("%d%d", &x[i], &y[i]);
        }
        sort(x + 1, x + 1 + N);
        sort(y + 1, y + 1 + N);
        if (N % 2 == 0)
        printf("%lld\n", (x[(N / 2) + 1] - x[N / 2] + 1) * 1ll * (y[(N / 2) + 1] - y[N / 2] + 1));
        else
        printf("%d\n", 1);
    }
    return 0;
}
