#include<iostream>
#include<cstdio>
#include<algorithm>

using namespace std;
int main()
{   int T,i;
    long long int ym,xm;
     scanf("%d",&T);
     while(T--)
     {      int N;
         scanf("%d",&N);

         int x[N],y[N];
         for(i=0;i<N;i++)
           scanf("%d %d",&x[i],&y[i]);

         if(N%2!=0)
         { printf("1\n");}
         else
         {
             sort(x,x+N);
             sort(y,y+N);

            xm=x[N/2]-x[N/2-1]+1;
            ym=y[N/2]-y[N/2-1]+1;

            printf("%lld\n",xm*ym);
         }
     }
    return 0;
}
