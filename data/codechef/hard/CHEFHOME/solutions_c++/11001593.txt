#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

long long get_points(vector<long long> &x, vector<long long> &y, int len){

	if(len % 2 == 0){
		sort(x.begin(), x.end());
		sort(y.begin(), y.end());
		long long x1 = x[len/2]-x[(len/2)-1] + 1;
		long long y1 = y[len/2]-y[(len/2)-1] + 1;
		return (long long)x1 * y1;
	}

	return 1;
}

int main(){
	int t;
	cin>>t;
	while(t-- != 0){
		int n;
		cin>>n;
		vector<long long> x;
		vector<long long> y;
		long long tx, ty;
		for(int i=0;i<n;i++){
			cin>>tx>>ty;
			x.push_back(tx);
			y.push_back(ty);
		}

		cout<<get_points(x, y, n)<<endl;
	}

	return 0;
}
