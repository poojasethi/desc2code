#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
long long int x[1000+10],y[1000+10];
int main()
{
int t,k,n,i;
long long int cx,cy,res;
long long int sum,min;
scanf("%d",&t);
for(k=1;k<=t;k++)
{
scanf("%d",&n);
for(i=0;i<n;i++)
{
scanf("%lld %lld",&x[i],&y[i]);
}
if(n%2!=0)
{
printf("1\n");
}
else
{
sort(x,x+n);
sort(y,y+n);
cx=x[n/2]-x[n/2-1]+1;
cy=y[n/2]-y[n/2-1]+1;
res=cx*cy;
printf("%lld\n",res);
}
}
return 0;
}
