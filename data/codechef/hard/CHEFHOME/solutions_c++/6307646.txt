#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
int n;
cin>>n;
long long int a[n],b[n];
for(int i=0;i<n;i++)
cin>>a[i]>>b[i];
if(n%2!=0)
{
cout<<"1"<<endl;
continue;
}
sort(a,a+n);
sort(b,b+n);
long long int ans=(a[n/2]-a[n/2-1]+1)*(b[n/2]-b[n/2-1]+1);
cout<<ans<<endl;
}
return 0;
}
