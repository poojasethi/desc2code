#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

typedef long long ll;
typedef pair<ll,ll> pii;
vector<ll> x;
vector<ll> y;




int main(){
	int t;
	cin>>t;
	while(t--){
	
		int n;
		cin>>n;
		for(int i=0;i<n;i++){
			ll a,b;
			cin>>a>>b;
			x.push_back(a);
			y.push_back(b);
		}

		if(n%2){
			cout<<1<<endl;
			x.clear();
			y.clear();
			continue;
		}
		sort(x.begin(),x.end());
		sort(y.begin(),y.end());


		ll ans = x[n/2] - x[(n-1)/2] + 1;
		ans = ans * (y[n/2] - y[(n-1)/2] + 1);

		cout<<ans<<endl;
		x.clear();
		y.clear();
	}
}
