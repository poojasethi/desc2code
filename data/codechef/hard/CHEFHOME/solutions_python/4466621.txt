t = int(raw_input().strip())
for c in range(t):
  n = int(raw_input().strip())
  xs = []
  ys = []
  for p in range(n):
    x, y = [int(q) for q in raw_input().strip().split()]
    xs.append(x)
    ys.append(y)
  xs.sort()
  ys.sort()
  ans = 1
  if n % 2 == 0:
    ans *= xs[n/2] - xs[-1 + n/2] + 1
    ans *= ys[n/2] - ys[-1 + n/2] + 1
  print ans
