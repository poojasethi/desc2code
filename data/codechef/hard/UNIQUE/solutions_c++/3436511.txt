const int N=200010;
const int inf=1000000000;
int n,i,j;
char ss[N+1];
int a[N],sa[N],sa2[N],rank[N],rank2[N],sum[N],h[N];
void suffix()
{
    int i,j,p;
    CL(sum,0);
    rep(i,1,n)sum[a[i]]++;
    rep(i,1,100)sum[i]+=sum[i-1];
    _rep(i,n,1)sa[sum[a[i]]--]=i;
    rank[sa[1]]=j=1;rep(i,2,n){if(a[sa[i]]!=a[sa[i-1]])j++;rank[sa[i]]=j;}
     
    for(p=1;j<n;p*=2)
    {
        CL(sum,0);CP(rank2,rank);
        j=0;rep(i,n-p+1,n)sa2[++j]=i;
        rep(i,1,n)if(sa[i]>p)sa2[++j]=sa[i]-p;
        rep(i,1,n)sum[rank[sa2[i]]]++;
        rep(i,1,n)sum[i]+=sum[i-1];
        _rep(i,n,1)sa[sum[rank[sa2[i]]]--]=sa2[i];
        rank[sa[1]]=j=1;rep(i,2,n){if(rank2[sa[i]]!=rank2[sa[i-1]]||rank2[sa[i]+p]!=rank2[sa[i-1]+p])j++;rank[sa[i]]=j;}
    }
    
    h[1]=0;p=0;
    rep(i,1,n)if(rank[i]>1)
    {
		j=sa[rank[i]-1];
		while(a[i+p]==a[j+p])p++;
		h[rank[i]]=p;
		if(p)p--;
	}
}

bool kind;pii res;bool can[N];pii ans[N];int u[N];
struct seg
{
	pii key[N*4],en[N*4];
	void pen(int i,pii s){upmin(key[i],s);upmin(en[i],s);}
	void pdown(int i){if(en[i]!=mp(inf,inf)){pen(i*2,en[i]);pen(i*2+1,en[i]);en[i]=mp(inf,inf);}}
	void upd(int i){key[i]=min(key[i*2],key[i*2+1]);}
	void find(int i,int x,int y,int l,int r)
	{
		if(y<l||x>r)return;
		if(x>=l&&y<=r){if(kind==0)upmin(res,key[i]);else pen(i,res);return;}
		pdown(i);
		find(i*2,x,(x+y)/2,l,r);
		find(i*2+1,(x+y)/2+1,y,l,r);
		upd(i);
	}
	void bui(int i,int x,int y)
	{
		key[i]=en[i]=mp(inf,inf);
		if(x!=y)bui(i*2,x,(x+y)/2),bui(i*2+1,(x+y)/2+1,y);
	}
}T,T2;
 
int main()
{
    //freopen("1.in","r",stdin);
	//freopen("1.out","w",stdout);
	
	scanf("%s",ss+1);n=strlen(ss+1);
	rep(i,1,n)a[i]=ss[i]-'a'+1;
	
	suffix();
	
	T.bui(1,1,n);T2.bui(1,1,n);
	rep(i,1,n)ans[i]=mp(inf,inf);
	rep(i,1,n)u[i]=max(h[i],h[i+1])+1;
	
	#define pd(mid) (i-(mid)+1>=u[rank[mid]])
	rep(i,1,n)
	{
		int l,r,mid;
		for(l=1,r=i;l<r;)
		{
			mid=(l+r+1)/2;
			if(pd(mid))l=mid;else r=mid-1;
		}
		if(pd(l))
		{
			kind=1;res=mp(i-l+1,rank[l]);T.find(1,1,n,l,i);
			kind=1;res=mp(i,0);T2.find(1,1,n,1,l-1);
		}
	}
	
	
	rep(i,1,n)
	{
		kind=0;res=mp(inf,inf);T.find(1,1,n,i,i);upmin(ans[i],res);
		kind=0;res=mp(inf,inf);T2.find(1,1,n,i,i);upmin(ans[i],mp(res.fi-i+1,rank[i]));
	}
	rep(i,1,n)printf("%d %d\n",sa[ans[i].se],ans[i].fi);
    scanf("\n");
    return 0;
}
