#include<cstdio>
#include<algorithm>

using namespace std;

double dp[1005][1005];

int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
    int n,k;
    scanf("%d %d",&n,&k);
    int price[1005],discount[1005];
    for(int i=1;i<=n;i++)
      scanf("%d",&price[i]);
    for(int i=1;i<=k;i++)
      scanf("%d",&discount[i]);
    sort(discount+1,discount+k+1);
    sort(price+1,price+n+1);
    for(int i=1;i<=n;i++)
    {
      for(int j=1;j<=k && j<=i;j++)
      {
	double p=(double)j/(double)i;
	dp[i][j]=(price[i]*discount[j]+dp[i-1][j-1])*p+dp[i-1][j]*(1-p);
      }
    }
    printf("%.3lf\n",dp[n][k]/100);
  }
  return 0;
}
