#include <cstdio>
#include <algorithm>
using namespace std;
 
#define maxN 1000
 
double dp[maxN+1][maxN+1];
 
int main()
{
    int TST;
    scanf("%d",&TST);
    for(int tst=0;tst<TST;tst++)
    {
        int n,k,i,j;
        scanf("%d%d",&n,&k);
 
        int a[maxN+1];
        for(i=1;i<=n;i++) scanf("%d",a+i);
        sort(a+1,a+1+n);
 
        int b[maxN+1];
        for(j=1;j<=k;j++) scanf("%d",b+j);
        sort(b+1,b+1+k);
 
        for(i=1;i<=n;i++)
            for(j=1;j<=k && j<=i;j++)
            {
                double p=1.*j/i;
                dp[i][j]=(a[i]*b[j]+dp[i-1][j-1])*p+dp[i-1][j]*(1-p);
            }
 
        double ans=dp[n][k]/100.;
        printf("%.3lf\n",ans);
    }
    return 0;
}
