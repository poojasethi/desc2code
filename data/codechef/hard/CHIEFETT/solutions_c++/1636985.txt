#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<algorithm>
using namespace std;

int main()
{
    int test;
    scanf("%d",&test);
    while(test--)
    {
        int n,k;
        scanf("%d%d",&n,&k);
        int cost[n+1];
        int discount[k+1];
        for(int i=1;i<=n;i++)
        {
            scanf("%d",&cost[i]);
        }
        for(int j=1;j<=k;j++)
        {
            scanf("%d",&discount[j]);
        }
        sort(cost+1,cost+n+1);
        sort(discount+1,discount+k+1);
        double dynamic[1001][1001];

        for(int i=1;i<=n;i++)
        {
            for(int j=1;j<=k && j<=i;j++)
            {
                double p = 1.*j/i;
                if(i==0)
                {
                    dynamic[i][j]=0;
                }
                else if(j==0)
                {
                    dynamic[i][j]=0;
                }
                else
                dynamic[i][j]=(cost[i]*discount[j]+dynamic[i-1][j-1])*p+(dynamic[i-1][j])*(1-p);
            }
        }

double answer=dynamic[n][k]/100.;
printf("%.3lf\n",answer);

    }

    return 0;
}
