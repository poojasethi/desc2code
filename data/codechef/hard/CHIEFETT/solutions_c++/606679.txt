#include<algorithm>
#include<vector>
#include<cstdio>

using namespace std;

double dp[1001][1001];

int main()
{
	int t,n,k;
	scanf("%d",&t);
	while(t--)
	{
		int arrn[1001],arrk[1001];
		scanf("%d%d",&n,&k);
		for(int i=1; i<=n; i++)
			scanf("%d",&arrn[i]);
		for(int i=1; i<=k; i++)
			scanf("%d",&arrk[i]);

		sort(arrn+1,arrn+n+1);
		sort(arrk+1,arrk+k+1);

		dp[0][0] = 0;
		for(int i=1; i<=n; i++)
		{
			for(int j=1; j<=k && j<=i; j++)
			{
				double p = j/(double)i;
				dp[i][j] = p*(arrn[i]*(arrk[j]/(double)100) + dp[i-1][j-1]) + (1-p)*(dp[i-1][j]);
			}
		}
		printf("%.3lf\n",dp[n][k]);
	}
	return 0;
}
