                                                                     
                                                                     
                                                                     
                                             
#include <iostream>
#include <string>
#include <cstdio>
#include <algorithm>
using namespace std;
int cost[1005];
double discounts[1005];
double dp[1005][2];
void init(int N,int K)
{
	for(int i=0;i<=N+2;i++)
		for(int j=0;j<2;j++)
			dp[i][j]=0.0;
}
int main() 
{
	int t;
	scanf("%d",&t);
	int tmp=0;
	while(t--)
	{
		int N,K;		
		scanf("%d",&N);
		scanf("%d",&K);
		init(N,K);
		for(int i=0;i<N;i++)
		{
			scanf("%d",&cost[i]);
			
		}
		for(int i=0;i<K;i++)
		{

			scanf("%d",&tmp);
			discounts[i]=(double)tmp/100.0;
		}
        sort(cost,cost+N);
        sort(discounts,discounts+K);
		for(int k=K-1;k>=0;k--)
		{
			for(int n=N+k-K;n>=0;n--)
			{
				if(N-n>=K-k)
				{
					double prob=((K-k)*1.0)/((N-n)*1.0);
					dp[n][0]=(dp[n+1][1]+(cost[n]*discounts[k]))*prob;
					dp[n][0]+=dp[n+1][0]*(1-prob);
				}
			}
			for(int i=0;i<=N+2;i++)
				dp[i][1]=dp[i][0];

		}
        double ans=dp[0][0];
        printf("%.3f\n",ans);
	}
}
