#include<cstdio>
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

double ans[1001][1001];
int main()
{
	int t,i,j;
	cin>>t;

	while(t--){
	    
		int n,m;
		cin>>n>>m;
        double a[1001];
		double p;
		double b[1001];
		for(i =1 ;i <= n;i++) {
			cin>>a[i];
		}
		sort(a+1,a+n+1);
		for(i = 1; i <= m ;i++){
			cin>>b[i];
		}
		sort(b+1,b+1+m);
		for(i = 1; i<= n; i++) {
			for(j = 1; j <=m && j <= i; j++){
				p = 1.0*j/i;
				ans[i][j] = (a[i]*b[j] + ans[i-1][j-1])*p  +  ans[i-1][j]*(1-p);
			}
		}
		printf("%0.3lf\n",ans[n][m]/100.0);
	}
	return 0;
}
