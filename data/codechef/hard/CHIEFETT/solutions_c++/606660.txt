#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
const int MAXN = 1000;
double c[MAXN + 1][MAXN + 1],d[MAXN + 1][MAXN + 1];
int a[MAXN],b[MAXN];
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	c[0][0] = 1.0;
	for(int i = 1; i <= MAXN; i++)
	{
		c[i][0] = c[i - 1][0];
		for(int j = 1; j <= i; j++)
			c[i][j] = c[i - 1][j] + c[i - 1][j - 1];
	}
	int t,n,k;
	scanf("%d",&t);
	for(int tt = 0; tt < t; tt++)
	{
		scanf("%d%d",&n,&k);
		for(int i = 1; i <= n; i++)
			scanf("%d",&a[i]);
		for(int i = 1; i <= k; i++)
			scanf("%d",&b[i]);
		sort(a + 1,a + n + 1);
		sort(b + 1,b + k + 1);
		d[0][0] = 0.0;
		for(int i = 1; i <= n; i++)
		{
			d[i][0] = d[i - 1][0];
			for(int j = 1; j <= i; j++)
				d[i][j] = d[i - 1][j] + d[i - 1][j - 1] + c[i - 1][j - 1] * a[i] * b[j] / 100.0;
		}
		printf("%.3lf\n",d[n][k] / c[n][k]);
	}
	return 0;
}
