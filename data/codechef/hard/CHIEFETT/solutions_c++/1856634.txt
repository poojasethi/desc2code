#include<stdio.h>
#include<algorithm>
#include<assert.h>

int main()
{
	//freopen("chefett.in","r",stdin);

	int T,N,K;
	double temp;
	double prices[1000];
	double discount[1000];
	double** dp = new double*[1000];
	for(int i=0; i<1000; ++i) dp[i] = new double[1000];

	scanf("%d",&T);

	while(T--)
	{
		scanf("%d%d",&N,&K);
		for(int i=0; i<N; ++i) scanf("%lf",&prices[i]);
		for(int i=0; i<K; ++i) scanf("%lf",&discount[i]);

		std::sort(prices,prices+N);
		std::sort(discount,discount+K);

		temp = 0;
		for(int i=0; i<N; ++i)
		{
			temp += prices[i];
			dp[i][0] = temp*discount[0]/(i+1);
		}

		for(int j=1; j<K; ++j)
		{
			for(int i=j; i<N; ++i)
			{
				dp[i][j] = (j+1)*(dp[i-1][j-1]+prices[i]*discount[j]) + (i-j)*dp[i-1][j];
				dp[i][j] /= i+1;
			}
		}

		printf("%0.3lf\n",dp[N-1][K-1]/100.0);
	}

	for(int i=0; i<1000; ++i) delete[] dp[i];
	delete[] dp;

	return 0;
}
