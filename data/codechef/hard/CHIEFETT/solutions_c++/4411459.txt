#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;

double A[1002];
double B[1002];
double dp[1002][1002];
     
int main()
{
    int t,n,k;
    double val;
    scanf("%d", &t);
    while ( t-- ) {
          scanf("%d%d", &n, &k);
          for ( int i = 1; i <= n; i++ ) scanf("%lf", &A[i]);
          for ( int i = 1; i <= k; i++ ) {
              scanf("%lf", &B[i]);
              B[i] = B[i]/100.0;
          }
          sort(A+1, A+n+1);
          sort(B+1, B+k+1);
          dp[1][0] = 0.0;
          dp[1][1] = (double)(A[1]*B[1]);
          for ( int i = 2; i <= n; i++ ) {
              for ( int cnt = 0; cnt <= min(i,k); cnt++ ) {
                  if ( cnt == 0 ) {
                       dp[i][cnt] = 0.0;
                       continue;
                  }
                  dp[i][cnt] = dp[i-1][cnt]*(double)(i-cnt) + (double)cnt*(dp[i-1][cnt-1] + (double)A[i]*B[cnt]);
                  dp[i][cnt] = (double)dp[i][cnt]/(double)i;
              }
          }
          printf("%0.3lf\n", dp[n][k]);
    }          
    return 0;
}
