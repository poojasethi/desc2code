#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long ll;
typedef double ld;

ld bin[1024][1024];

int prices[1024];
int disc[1024];
ld discs[1024];

int main()
{
	bin[0][0] = 1;
	
	for (int n = 1; n <= 1000; ++n)
	{
		bin[n][0] = 1;
		for (int k = 1; k <= n; ++k)
			bin[n][k] = bin[n-1][k-1] + bin[n-1][k];
	}
	
	int T;
	scanf("%d\n", &T);
	
	while(T-- > 0)
	{
		int N, K;
		scanf("%d %d\n", &N, &K);
		for (int i = 0; i < N; ++i)
		scanf("%d", &prices[i]);
		for (int i = 0; i < K; ++i)
		scanf("%d", &disc[i]);
		sort(prices, prices + N, greater<int>());
		sort(disc, disc + K, greater<int>());
		ld res = 0.0;
		
		
		for (int p = 0; p < N; ++p)
		{
			for (int i = 0; i <= K; ++i)
			{
				if (p >= i && N - p - 1 >= K - i - 1)
				{
                      res += disc[i] * prices[p] * bin[p][i] * bin[N-p-1][K-i-1];
				}
			}
		}
		printf("%.3lf\n", (res / bin[N][K]) / 100.0);
	}
	return 0;
}
