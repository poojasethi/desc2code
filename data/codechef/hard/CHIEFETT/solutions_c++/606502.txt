#include<iostream>
#include<stdio.h>
#include<math.h>
#include<string.h>
#include<algorithm>
#include<string>
#include<vector>

using namespace std;
#define FOR(i,n) for(i=0;i<n;i++)
#define FOR1(i,n) for(i=1;i<=n;i++)
#define FORab(i,a,b) for(i=a;i<=b;i++)
double ncr[1010][1010]={0};

int main()
{
    int i,j,n,K,t;
    int dis[1130],arr[1010];
    FOR(i,1010)
        FOR(j,1010)if(i<j)ncr[i][j]=0;
    ncr[0][0]=1.0;
    FOR1(i,1000)
        for(j=0;j<=i;j++)
        {
            ncr[i][j]=ncr[i-1][j];
            if(j)ncr[i][j]+=ncr[i-1][j-1];
            //cout<<i<<" "<<j<<endl;
            //cout<<ncr[i][j]<<" "<<ncr[i-1][j]<<" "<<ncr[i][j-1]<<endl;
            //cin>>t;
        }
        //cout<<ncr[1000][500]<<endl;
    cin>>t;

    while(t--)
    {
        cin>>n>>K;
        double ans=0;
        FOR(i,n)scanf("%d",&arr[i]);
        FOR(i,K)scanf("%d",&dis[i]);
        sort(arr,arr+n);
        sort(dis,dis+K);
        FOR(i,n)
        {
            FOR(j,K)
            {
                int l=i;
                int r=j;
                if(l<r)continue;
                if(n-l-1<K-1-r)continue;
                ans+=((ncr[l][r]*ncr[n-l-1][K-1-r]))*(((dis[j]*arr[i])/100.0));;
                //cout<<i<<"s "<<j<<endl;
                //cout<<ncr[l][r]<<" "<<ncr[n-l-1][K-1-r]<<" "<<((dis[j]*arr[i])/100.0)<<"ans "<<ans<<endl;

            }
        }
        //cout<<ncr[n][K]<<endl;
        ans/=ncr[n][K];
        printf("%.3lf\n",ans);
    }
	return 0;
}
