typedef long long llong;
typedef pair<int, int> Item;

const int Maxn = 1000+10;
const int INF = 0x7f7f7f7f;
const double eps = 1e-10;
const double pi = acos(-1.0);
inline int realcmp(double a, double b){ return (a>b+eps) ? 1 : ((a+eps<b)?-1:0); }

double f[Maxn][Maxn];
double item[Maxn], disc[Maxn];

int main()
{
    int cas, n, k;

    scanf("%d", &cas);
    while( cas-- )
    {
        scanf("%d %d", &n, &k);
        for(int i=0; i<n; i++)
            scanf("%lf", &item[i]);
        for(int i=0; i<k; i++)
        {
            scanf("%lf", &disc[i]);
            disc[i] /= 100.0;
        }
        sort(item, item+n);
        sort(disc, disc+k);

        memset(f, 0, sizeof(f));
        for(int i=1; i<=n; i++)
            for(int j=1; j<=i&&j<=k; j++)
            {
                double p = 1.0*j/i;
                f[i][j] = (item[i-1]*disc[j-1]+f[i-1][j-1])*p+f[i-1][j]*(1-p);
            }

        printf("%.3lf\n", f[n][k]);
    }

    return 0;
}
