#include<iostream>
#include<algorithm>
using namespace std;
char buffer[10000];
double C[1024][1024];
void fillC()
{
   C[0][0]=1.0;
   int i,j;
   for(i=1;i<=1000;i++)
   {
       C[i][0]=1.0;
       for(j=1;j<=i;j++)
          C[i][j]=C[i-1][j-1]+C[i-1][j];
   }
}
double a[1024],b[1024];
int main()
{
    int x,t=0;
    gets(buffer);
    for(x=0;buffer[x];x++)
      t=t*10+(buffer[x]-'0');
    //scanf("%d",&t);
   // printf("%d\n",t);
    fillC();
    while(t--)
    {
       int n=0,k=0,i,j,y;
       gets(buffer);
       for(i=0;buffer[i]!=32;i++)
         n=n*10+(buffer[i]-'0');
       for(++i;buffer[i];i++)
         k=k*10+(buffer[i]-'0');
       //scanf("%d%d",&n,&k);
      // printf("%d %d\n",n,k);
       gets(buffer);
       for(i=1,j=-1;i<=n;i++)
       {
          //scanf("%lf",&a[i]);
          y=0;
          for(++j;buffer[j]!=32 && buffer[j];j++)
             y=y*10+(buffer[j]-'0');
          a[i]=(double)y;
       //   printf("%lf ",a[i]);
       }
     //  printf("\n");
       gets(buffer);
       for(j=1,i=-1;j<=k;j++)
       {
          //scanf("%lf",&b[j]);
          y=0;
          for(++i;buffer[i]!=32 && buffer[i];i++)
            y=y*10+(buffer[i]-'0');
          b[j]=(double)y;
          b[j]=b[j]/100.0;
     //     printf("%lf ",b[j]);
       }
    //   printf("\n");
       sort(a+1,a+n+1);
       sort(b+1,b+k+1);
       double res=0.0;
       for(i=1;i<=n;i++)
       {
          for(j=max(1,k-n+i);j<=min(k,i);j++)
          {
       //      printf("%lf/%lf * %lf * %lf * %lf\n",C[n-i][k-j],C[n][k],C[i-1][j-1],a[i],b[j]);
             res+=(C[n-i][k-j]/C[n][k])*C[i-1][j-1]*a[i]*b[j];
          }
       }
       printf("%.3lf\n",res);
    }
    return 0;
}
