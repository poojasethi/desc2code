#include <cstdio>
#include <cstdlib>
 
#include <cstring>
#include <algorithm>
 
using namespace std;
 
double C[1024][1024];
double A[1024], B[1024];
 
int main() {
    memset(C,0,sizeof(double)*1024*1024);
    C[0][0] = 1;
    for(int i = 1; i <= 1000; i++) {
        C[i][0] = 1.0;
        for(int j = 1; j <= i; j++) C[i][j] = C[i-1][j-1] + C[i-1][j];
    }
    int T;
    scanf("%d",&T);
    while(T--) {
        int N, K;
        scanf("%d %d",&N,&K);
        for(int i = 1; i <= N; i++) scanf("%lf",&A[i]);
        sort(A+1, A+1+N);
        for(int j = 1; j <= K; j++) {
            scanf("%lf",&B[j]);
            B[j] /= 100;
        }
        sort(B+1, B+1+K);
        double res = 0;
        for(int i = 1; i <= N; i++) {            
            for(int j = max(1,K-N+i); j <= min(K,i); j++) {
                res += (C[i-1][j-1]/C[N][K])*C[N-i][K-j]*A[i]*B[j];
            }
        }
        printf("%.3lf\n",res);
    }
    return 0;
}
