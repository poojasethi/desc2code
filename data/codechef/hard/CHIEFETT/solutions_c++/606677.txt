#define pii pair < int , int >
#define i64 long long
#define CLR(x) memset(x,0,sizeof x)
#define SET(x,y) memset(x,y,sizeof x)

bool vis[1009][1009];
double dp[1009][1009];
int pr[1009],ds[1009],n,k;

double dfs(int u,int l) {
	//cout<<u<<' '<<l<<'\n';
	//getchar();
	if(l==k) return 0;
	if(vis[u][l]) return dp[u][l];
	vis[u][l]=1;
	if((n-u)==(k-l)) return dp[u][l]=pr[u]*(ds[l]/100.0)+dfs(u+1,l+1);
	double r=0,p;
	p=(double(k-l))/(n-u);
	r=p*(pr[u]*(ds[l]/100.0)+dfs(u+1,l+1));
	r+=(1-p)*dfs(u+1,l);
	return dp[u][l]=r;
}
	

int main(int argc, char **argv)
{
	int i,T;
	cin>>T;
	while(T--) {
		cin>>n>>k;
		for(i=0;i<n;i++) scanf("%d",&pr[i]);
		for(i=0;i<k;i++) scanf("%d",&ds[i]);
		sort(pr,pr+n);
		sort(ds,ds+k);
		CLR(vis);
		printf("%.3lf\n",dfs(0,0)+1e-11);
	}
	return 0;
}
