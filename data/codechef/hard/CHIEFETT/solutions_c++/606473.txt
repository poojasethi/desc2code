#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <set>
#include <map>
#include <vector>
#include <string>
#include <cmath>
#include <cstring>
#include <queue>
#include <stack>
#include <algorithm>
using namespace std;

int a[1005], d[1005], n, m;
double dp[1005][1005];
int color[1005][1005], cc;

int main() {
	int i, j, k, tests;
	for (scanf("%d", &tests); tests > 0; tests--) {
		scanf("%d%d", &n, &m);
		for (i = 0; i < n; ++i)
			scanf("%d", &a[i]);		
		sort(a, a + n);
		for (i = 0; i < m; ++i)
			scanf("%d", &d[i]);			
		sort(d, d + m);
		
		for (j = 0; j <= m; ++j)
			dp[n][j] = 0.0;
		for (i = n - 1; i >= 0; --i) {
			dp[i][m] = 0.0;			
			for (j = m - 1; j >= m - n + i && j >= 0; --j) {
				double p = 1.0 * (m - j) / (n - i);
				dp[i][j] = (dp[i + 1][j + 1] + a[i] * d[j] / 100.0) * p + dp[i + 1][j] * (1.0 - p); 	
			}				
		}
		printf("%.3lf\n", dp[0][0]);			
	}
	return 0;		
}
