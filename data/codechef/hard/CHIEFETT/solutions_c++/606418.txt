#define REP(i,n)    for(int i = 0; i < n; i++)
#define FOR(i,a,b)  for(int i = a; i < b; i++)
#define FORR(i,n)   for(int i = n-1; i >=0; i--)

double comb[1001][1001] = {0.0};
int main(){
//	freopen("input.txt", "r", stdin);
//	freopen("output.txt", "w", stdout);

    int n,k;
    double p[1000];
    double d[1000];

    REP(i,1001){
        comb[i][0] = comb[i][i] = 1.0;
        FOR(j,1,i){
            comb[i][j] = comb[i-1][j]+comb[i-1][j-1];
        }
    }
    int t;
    SI(t);

    while(t--){
        SI(n);SI(k);
        REP(i,n){
            SD(p[i]);
        }
        REP(i,k){
            SD(d[i]);
        }
        double ans = 0;
        sort(p,p+n);
        sort(d,d+k);
        for(int i = 1; i <= n; i++){
            double temp = 0;
            for(int j = 1; j <= k; j++){
                temp += d[j-1]*comb[i-1][j-1]*comb[n-i][k-j];
            }
            ans += temp*p[i-1];
        }
        ans = ans/comb[n][k]/100;
        PD(ans);NL;
    }
}
