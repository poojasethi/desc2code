double dp[1005][1005];
char vi[1005][1005],cs;
int a[1005],b[1005];
double solve(int n,int m){
	if(n <0 || m<0 || n<m) return 0;
	double &ret = dp[n][m];
	if(vi[n][m]==cs) return ret;
	vi[n][m] = cs;
	double p = 1.0 * (m+1) / (n+1);
	ret = p* (solve(n - 1,m - 1) + b[m]/100.0 * a[n]) + (1 - p) * solve(n - 1,m) ;
	return ret;
}
int main(){
	int i,t,n,m;
	scanf("%d",&t);
	cs = 1;
	while(t--){
		scanf("%d %d",&n,&m);
		for(i = 0;i<n;i++){
			scanf("%d",&a[i]);
		}
		for(i = 0;i<m;i++){
			scanf("%d",&b[i]);
		}
		sort(a,a+n);
		sort(b,b+m);
		printf("%.3lf\n",solve(n-1,m-1));
		cs++;
	}
	return 0;
}
