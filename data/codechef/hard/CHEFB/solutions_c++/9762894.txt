#include <map>

using namespace std;
#define MAX 1000006
#define INF 1<<16
#define ff first
#define ss second
#define pii pair<int,int>
#define pi2 pair<pii,int>
std::vector<long long>primes;
bitset<1000006>bs;
int arr[1000005];
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n;
		cin>>n;
		memset(arr,0,sizeof arr);
		while(n--)
		{
			int a;
			scanf("%d",&a);
			for(int i=2;i*i<=a;i++)
			{
				if(a%i==0)
				{
					int c=0;
					while(a%i==0)
					{
						a/=i;
						c++;
					}
					arr[i]=max(arr[i],c);
				}
			}
			if(a>1)
				arr[a]=max(arr[a],1);
		}
		long long ans=0;
		for(int i=0;i<MAX;i++)
			ans+=arr[i];
		cout<<ans<<endl;
	}
	return 0;
}
