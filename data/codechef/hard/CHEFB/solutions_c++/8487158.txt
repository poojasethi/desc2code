#include<bits/stdc++.h>
using namespace std;
vector<int> v(1000001);
vector<int> divn(1000001);
int main()
{
    int i,j,t,n,num,c,val,ans,temp;
    for(i=2;i<=1000000;i++)
    {
        if(v[i]==0)
        {
            divn[i]=i;
            for(j=2*i;j<=1000000;j=j+i)
            {
                v[j]=1;
                divn[j]=i;
            }
        }
    }
    scanf("%d",&t);
    map<int,int>::iterator it;
    while(t--)
    {
        ans=0;
        map<int,int> m;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&num);
            temp=num;
            while(temp!=1)
            {
                val=divn[temp];
                c=0;
                while(temp%val==0)
                {
                    c++;
                    temp/=val;
                }
                if(m[val]==0)
                    m[val]=c;
                else
                {
                    m[val]=max(m[val],c);
                }
            }
        }
        for(it=m.begin();it!=m.end();it++)
        {
            ans+=(it->second);
        }
        printf("%d\n",ans);
    }
    return 0;
}
