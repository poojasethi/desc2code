    #include <bits/stdc++.h>
    using namespace std;
     
    int main() {
    	int t;
    	cin>>t;
    	while(t--)
    	{
    		long int m,n;
    	scanf("%ld%ld",&n,&m);
    		for(long int i=n;i<=m;i++)
    		{int flag=0;
    			if(i==1)
    			{
    			 flag=1;
    			}
    			else if(i==2)
    			{
    				flag=0;    			}
    			else if(i%2==0)
    			{
    				flag=1;
    			}
    			else{
    			long int k=sqrt(i);
    			for(long int j=3;j<=k;j+=2)
    			{
    				if(i%j)
    				{
    					continue;
    				}
    				else{
    					flag=1;
    					break;
    				}
    			}
    			}
    			if(flag==0)
    			{
    				printf("%ld\n",i);
    			}
    		}
    		printf("\n");
    	}
    	
    	
    	return 0;
    }
