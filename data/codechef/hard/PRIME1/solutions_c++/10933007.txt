bool isprime(int a){
//	if(checked[a]==true)return p[a];
//	checked[a]=true;
	bool p=true;
	if(a==1)return false;
	if(a==2)return true;
	if(a%2==0)return false;
	int sq=sqrt(a);
	for(int i=3; i<=sq; i+=2){
		if(a%i==0){
			p=false;break;
		}
	}
	return p;
}

	while(t--){
		ll n, i, j, k, m;
		scanf("%lld %lld", &n, &m);
		for(i=n; i<=m; i++){
			if(isprime(i))printf("%lld\n", i);
		}
		printf("\n");
	}
}
