#include<bits/stdc++.h>
using namespace std;
bool isprime(long n)
{
	if(n==1 || (n%2==0 && n!=2) )
		return false;
	long s=(long)sqrt(n)+1;
	for(long i=3;i<=s;i+=2)
		if(n%i==0)
			return false;
	return true;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long a,b;
		cin>>a>>b;
		for(long i=a;i<=b;i++)
		{
			if(isprime(i))
				cout<<i<<endl;
		}
		cout<<endl;
	}
	return 0;
}
