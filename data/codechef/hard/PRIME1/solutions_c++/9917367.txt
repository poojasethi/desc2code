#include<bits/stdc++.h>
using namespace std;

#define gc getchar_unlocked

typedef long long int ll;
vector<int> P;

int T,N,M;

inline int in() {
	char c=gc();
	int v=0;
	for(;c>='0' && c<='9';c=gc())
		v=v*10+c-'0';
	return v;
}

bool CheckPrime(ll N) {
	int S=sqrt(N);
	for(int i=0;P[i]<=S;++i) if(!(N%P[i])) return 0;
	return 1;
}



int main() {
	P.push_back(2);
	for(int i=3;i<=31700;i+=2) if(CheckPrime(i))
		P.push_back(i);
	T=in();
	while(T--) {
		M=in();N=in();
		if((M==1 && 2<=N) || M==2) { ++M; printf("%d\n",2); }
		if(!(M&1)) ++M;
		for(int i=M;i<=N;i+=2) if(CheckPrime(i))
			printf("%d\n",i);
	}
}
