#include<iostream>
#include<cstdio>
#include<vector>
#include<cmath>
using namespace std;
bool A[1000001];
void genPrime(vector<long long> & V)
{
	for(long long i=0;i<=35000;i++)	A[i] = true;
	A[0]=false;A[1]=false;
	for(long long i=2;i<35000;i++)
	{
		if(A[i])
		{
			V.push_back(i);
			for(long long j=i*i;j<=35000; j+= i)
				A[j]=false;
		}
	}
}
int main()
{
	vector<long long> prime;
	genPrime(prime);
	int t;
	scanf("%d", &t);
	long long temp, flag;
	while(t--){
		long long n, m;
		scanf("%lld %lld",&m,&n);
		//long long temp = sqrt()
		if(m==1) m++;
		for(;m<=n;m++){
			 flag=0;
			 temp=sqrt((double)m);
			 for(int i=0;prime[i]<=temp;i++){
			 	 if(m%prime[i]==0){
					flag=1;
					break;
				 }
			 }
			 if(flag==0)
			 printf("%d\n",m);
		 }
		printf("\n");
	}
	return 0;
}
