    #include<bits/stdc++.h>
    #define S(x) scanf("%lld",&x)
    #define P(x) printf("%lld\n",x)
    using namespace std;
    typedef long long ll;
    bool prime[1000001];
    void pre(){
    	prime[1] = true;
    	
    	for(ll i=2; i<=1000000; i++){
    		if(!prime[i]){
    			for(ll j = i; j*i <= 1000000; j++){
    				prime[j*i] = true;
    			}
    		}
    	}
    }
    int main(){
    	ll t,m,n;
    	pre();
    	for(S(t); t--; ){
    		S(m);S(n);
    		bool ans[1000001];
    		fill(ans,ans+1000001,false);
    		if(m>1000000 || n> 1000000)
    		for(ll i=2; i<=47000; i++){
    			if(!prime[i]){
    				for(ll j = m/i; j*i <= n; j++){
    					ll k = j*i - m;
    					if(k<0)continue;
    					ans[k] = true;
    				}
    			}
    		}
    		
    		for(ll i=m ; i<=n ; i++){
    			if(i<=1000000 && !prime[i]) P(i);
    			else if(i>1000000&& !ans[i-m])P(i);
    		}
    	}
    }
