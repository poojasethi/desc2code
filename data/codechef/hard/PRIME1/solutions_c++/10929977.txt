#include <iostream>
#include <cmath>
#define FOR(i,m,k) for (i = m ; i <= k; i+= 2)
using namespace std;
int main ()
{
	std::ios::sync_with_stdio(false);
	int t;
	cin >> t;
	long long int m;
	long long int n;
	int i, j;
	bool prime = true;
	while (t--)
	{
		cin >> m >> n;
		if (m <= 2)
		{
			m = 3;
			if (n >= 2)
				cout << 2 << endl;
		}
		if (!(m & 1 == 1)) // even number
		{
			// cout << " i worked " << endl;
			m++;
		}
		FOR(i,m,n)
		{
			prime = true;
			FOR(j,3,sqrt(i))
			{
				if (i % j == 0)
				{
					prime = false;
					break;
				}
			}
			if (prime)
				cout << i << endl;
		}
		cout << endl;
	}
}
