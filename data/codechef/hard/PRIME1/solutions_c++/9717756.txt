#include <stdio.h>
bool isitPrime(long n) {
    //check if n is a multiple of 2
    if (n%2==0) return false;
    //if not, then just check the odds
    for(int i=3;i*i<=n;i+=2) {
        if(n%i==0)
            return false;
    }
    return true;
}

    long t,x,y;

    for (;x<=y;x++){

   }
