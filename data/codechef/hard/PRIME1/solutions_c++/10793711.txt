#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<vector>
using namespace std;
int main(){
	vector<char> prime (35001, true);
	vector<int>pl;
	prime[0] = prime[1] = false;
	for(int i=2;i<=2;i++){
		pl.push_back(i);
		for(int j=i*i; j<=35000; j += i){
			prime[j]=false;
		}
	}
	for(int i=3;i<=35000;i += 2){
		if(prime[i]){
			pl.push_back(i);
			for(int j=i*i; j<=35000; j+=i){
				prime[j]=false;
			}
		}
	}
	int t;
	scanf("%d", &t);
	long long temp, flag;
	while(t--){
		long long n, m;
		scanf("%lld %lld",&m,&n);
		//long long temp = sqrt()
		if(m==1) m++;
		for(;m<=n;m++){
			 flag=0;
			 temp=sqrt((double)m);
			 for(int i=0;pl[i]<=temp;i++){
			 	 if(m%pl[i]==0){
					flag=1;
					break;
				 }
			 }
			 if(flag==0)
			 printf("%d\n",m);
		 }
		printf("\n");
	}
	return 0;
}
