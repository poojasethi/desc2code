    #include<bits/stdc++.h>

    bool isprime(long n){
         if(n==1||n%2==0 && n!=2) 
             return false;  
         long s = (long)sqrt(n)+1;
         for(long i=3; i<=s; i = i+2){
             if(n%i==0)
                  return false;
         }
         return true;
    }
    int main(){
      int t;

         cin>>m>>n;
               for(int i=m; i<=n;i++){
                if(isprime(i)) 
                  cout<<i<<"\n";
               }
            }
