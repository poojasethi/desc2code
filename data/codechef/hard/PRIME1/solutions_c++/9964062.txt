#include<cstdio>
#include<iostream>
#include <bits/stdc++.h>
#include<cmath>
using namespace std;
vector<long long int> a;
void func()
{

    int flag,i,j;

    for(i=2;i<=33623;i++)
    {
         //cout<<"bieee";
         flag=0;
        for(j=2;j<=(sqrt(i));j++)
        {

            if((i%j)==0)
               {
                   flag=1;
                   break;
               }
        }

       if(flag==0)
           a.push_back(i);

    }
}

int main()
{

    int t,flag;
    long long int m,n,i,j,p;
    func();


    scanf("%d",&t);
    while(t--)
    {

            scanf("%lld %lld",&m,&n);
            if(m==1)
                m=2;
            for(i=m;i<=n;i++)
            {
                p=sqrt(i);
                flag=0;
                for(j=0;j<a.size() && a[j]<=p;j++)
                {

                        if(i%a[j]==0)
                    {
                        flag=1;
                        break;
                    }

                }
                if(flag==0)
                    printf("%lld\n",i);

            }
            cout<<endl;
    }




}
