#include<iostream>
#include<stdio.h>
bool w[100001];
void calc(int begin ,int end){
    for(int i=2; i*i<=end; i++){
        int s =((begin-1)/i+1)*i;
        if(s==i) s=2*i;
        while(s<=end){
            w[s-begin]=true;
            s+=i;
        }
    }
}
int main(){
    int n,begin,end;
    scanf("%d",&n);
    while(n--){
        for(int i=0;i<100001;i++) w[i]=false;
        scanf("%d%d",&begin,&end);
        calc(begin,end);
        for(int i=0;i<=end-begin;i++)
            if(!w[i] && i+begin!=1)
                printf("%d\n",i+begin);
    }
    return 0;
}
