#include<bits/stdc++.h>
#define ll long long int
#define VMAX 100001
using namespace std;

char is_not_prime[VMAX];
ll p[VMAX];
 void Sieve()
 {
     for(int i=2;i<VMAX;i++)
     {

          if(is_not_prime[i])  continue;

     for(int j=2*i;j<VMAX;j+=i)
       is_not_prime[j]=1;
     }


 }


int main()
{
   int t;

    Sieve();

   int count=0;
  for(int i=0;i<VMAX;i++)
  {
        if(!is_not_prime[i] && i!=1 && i!=0)
          p[count++]=i;

  }


   cin >> t;



    while(t--)
    {

     ll a,b;

     cin >> a >>b;

     if(a==1)
         a++;

      for(ll i=a;i<=b;i++)
      {
           int f=0;

       for(int j=0; p[j]<=sqrt(i);j++)
       {
              if(i%p[j]==0)
              {
                   f=1;
                   break;
              }
       }



                if(f==0)
              cout<<i<<endl;
      }

    }


    }
