#include<iostream>
#include<math.h>
using namespace std;
bool isPrime(int num)
{
  int i=0;
  if(num==1)
    return false;
  if(num==2)
     return true;
  if(num%2==0)
     return false;
  int sqr = sqrt(num);
  for(i=3;i<=sqr;i+=2)
     if(num%i==0)
      return false;

return true;

}

main()
{
int i,t,m,n;
bool flag=false;
cin>>t;

while(t--)
{
  cin>>m>>n;
  flag=false;
  for(i=m;i<=n;i++)
    {
     flag=isPrime(i);
     if(flag==true)
          cout<<i<<endl;
    }
 cout<<endl;
}

}
