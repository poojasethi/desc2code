#define FOR(i,a,b) for(int i=(a);i<=(b);++i)
#define NFOR(i,a,b) for(int i=(a);i>=(b);--i)
#define pb push_back
#define ll long long
using namespace std;
ll stprime[10001]={0};
ll c=0;
void calprime(ll n)
{
    ll sz=floor(sqrt((double)n));
    ll isprime[sz+1];
    c=0;
    FOR(i,0,sz)
    isprime[i]=1;
    isprime[0]=0;
    isprime[1]=0;
    for(ll i=2;i*i*i*i<=n;++i)
    {
        if(isprime[i]){
        for(ll j=i*i;j*j<=n;j+=i)
        {
            isprime[j]=0;
        }}
    }

    for(ll i=0;i*i<=n;++i)
    {
        if(isprime[i])
            stprime[c++]=i;

    }

}
int main()
{
    ll t,n,m;

    cin>>t;
    while(t--)
    {
        cin>>m>>n;
        ll ttprime[n-m];
        FOR(i,0,n-m)
        {ttprime[i]=1;}
        ll co=0;
        calprime(n);
        FOR(i,0,c-1)
        {
            ll x;

            x=m/stprime[i]*stprime[i];
        for(ll j=x;j<=n;j+=stprime[i])
        {
               if(j<m)continue;
               ttprime[j-m]=0;
        }
        }
        FOR(i,0,c-1)
        {
            if(stprime[i]>=m&&stprime[i]<=n)
                cout<<stprime[i]<<endl;
        }
        FOR(i,0,n-m)
        {
            if(ttprime[i]!=0&&(i+m)!=1)
                {
                    cout<<(i+m)<<endl;
                }
        }
    }
    return 0;
}
