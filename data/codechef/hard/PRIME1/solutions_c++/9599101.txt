#include<stdio.h>
#include<cstring>
#include<iostream>
#include<vector>
#define MAX 33000
using namespace std;
vector<int> numbers(MAX,0);
vector<int> prime;
vector<int> ans(MAX,0);
int main()
{
    int i=4,j,k;
    while(i<MAX)
    {
        numbers[i]=1;
        i=i+2;
    }
    i=9;
    while(i<MAX)
    {
        if(numbers[i]!=1)
            numbers[i]=1;
        i+=3;
    }
    i=5;
    while(i*i<MAX)
    {
        k=4;
        if(numbers[i]!=1)
        {
            j=i*i;
            while(j<MAX)
            {
                numbers[j]=1;
                j+=i;
            }
        }
        k=6-k;
        i=i+k;
    }
    for(i=2;i<MAX;i++)
        if(numbers[i]!=1)prime.push_back(i);
    int t,m,n,d;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d",&m,&n);
        if(m==1)
            m=2;
        d=n-m+1;
        int ans[100001];
        memset(ans,0,sizeof ans);
        i=0;
        while(prime[i]*prime[i]<=n)
        {
            if(m%prime[i]==0)
                k=0;
            else
                k=((m/prime[i])+1)*prime[i]-m;
            if(k+m==prime[i])
                ans[k]=0;
            else
                ans[k]=1;
            k+=prime[i];
            while(k<d)
            {
                if(ans[k]!=1)
                    ans[k]=1;
                k+=prime[i];
            }
            i++;
        }
        for(i=0;i<d;i++)
            if(ans[i]==0)
                printf("%d\n",i+m);
    }
    return 0;
}
