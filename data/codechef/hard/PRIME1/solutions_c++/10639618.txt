#include<bits/stdc++.h>
using namespace std;
bool prime[32001]={1};
void seive()
{
	prime[0]=false;
	prime[1]=false;
	for(int i=2;i<=32000;i++)
	prime[i]=true;
//	for(int i=0;i<=320000;i++)
//	cout<<prime[i]<<" ";
	for(int i=2;i*i<=32000;i++)
	{
		if(prime[i])
		{
			for(int j=2*i;j<=32000;j+=i)
			prime[j]=false;
		}
	}
//	cout<<prime[2];
}
int main()
{
	int t;
	cin>>t;

	seive();
	while(t--)
	{
		long long int n,m;
		cin>>m>>n;
		bool p[n-m+1];
		for(int i=0;i<n-m+1;i++)
		p[i]=true;
		if(m<2)
		m=2;
	//	cout<<prime[2];
		for(int i=2;i*i<=n;i++)
		{
			if(prime[i])
			{
				long long int v=(m/i)*i;
				if(v<m)
                v+=i;
                if(v==i)
                v=i*2;
               // cout<<v;
                for(long long int j=v;j<=n;j+=i)
                {
                	p[j-m]=false;
				}
               				
			}
		}
		for(int i=0;i<=n-m;i++)
		{
			if(p[i])
			{
				cout<<i+m<<"\n";
			}
			
		}
		cout<<"\n";
	}

}
