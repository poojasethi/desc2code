#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    long l,r,sq;
    int t,pr=1;
    cin>>t;
    while(t-- > 0)
    {
        cin>>l>>r;
        sq = sqrt(r);
        if(l<=2 && r>=2)
            cout<<"2\n";
        if(l<=3 && r>=3)
            cout<<"3\n";
        if(l<=5 && r>=5)
            cout<<"5\n";
        if(l<=7 && r>=7)
            cout<<"7\n";
pr=0;
        for(long i=l>3?l:3 ;i<=r;i++)
        {

            sq = sqrt(i);
            long j;
            if(i%2==0)
                pr=0;
            else
            for(j=3;j<=sq;j+=2)
            {
                pr=1;
                if(i%j==0){
                        pr=0;
                    break;
                }
            }
            if(pr==1)
            {
                cout<<i<<endl;
                pr=0;
            }
        }
        cout<<endl;
    }
    return 0;
}
