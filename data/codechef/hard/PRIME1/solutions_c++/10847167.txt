#include<iostream>
#include<cstdlib>
#include<algorithm>
#include<cmath>
using namespace std;
int prime(int x,int y);
void checkprime(int x);
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int x,y;
		cin>>x>>y;
		prime(x,y);
	}
	return 0;
}
int prime(int x,int y)
{
	for(int i=x;i<=y;i++)
	{
		checkprime(i);
	}
	cout<<endl;
	return 0;
}
void checkprime(int x)
{
	if(x==2)
	{
		cout<<x<<endl;
		return;
	}
	if(x%2==0)
	return;
	int a=sqrt(x);
	
	for(int i=3;i<=a;i+=2)
	{
		if(x%i==0)
		return;
	}
	
	if(x>2)
	cout<<x<<endl;
	
	return;
}
