#include<bits/stdc++.h>
using namespace std;
int is_prime(int n)
{
int i;
if(n==2)
return 1;
if(n<2 || n%2==0)
    return 0;
for(i=3;i<=sqrt(n);i+=2)
{
if(n%i==0)
return 0;
}
return 1;
}
int main()
{
int t;
int n,m,i;
scanf("%d",&t);
while(t--)
{
cin>>n>>m;
for(i=n;i<=m;i++)
{
if(is_prime(i))
printf("%d\n",i);
}
if(t!=0)
printf("\n");
}
return 0;
}
