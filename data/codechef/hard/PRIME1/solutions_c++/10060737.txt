#include<iostream>
#include<vector>
using namespace std;
int main()
{
    vector <long long int> ar,pr;
    long long int t,n,m,i,j,c=10000000;
    for(i=0;i<=c;i++)
        ar.push_back(i);
    ar[1]=ar[0]=0;
    for(i=2;i<=c;i++)
    {
        if(ar[i]!=0)
        {
            for(j=i+i;j<=c;j+=i)
                ar[j]=0;
            pr.push_back(ar[i]);
        }
    }
    cin>>t;
    while(t--)
    {
        cin>>n>>m;
        for(i=n;i<=m;i++)
        {
            if(i<=c)
            {
                if(ar[i]!=0)
                {
                    cout<<i<<endl;
                }
            }
            else
            {
                for(j=0;j<pr.size();j++)
                {
                    //cout<<pr[j]<<" ";
                    if(i%pr[j]==0)
                        {break;}
                    if(pr[j]>32000)
                        {cout<<i<<endl;break;}
                }
                //if(j>=pr.size())
                    //cout<<i<<endl;
            }
        }
        cout<<endl;
    }
    return 0;
}
