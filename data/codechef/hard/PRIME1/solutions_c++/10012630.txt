#include <iostream>
#include<math.h>
using namespace std;
void primegenrator(int a,int* ar)
{
	int m=0;
	for(int i=2;i<=sqrt(a);i++)
	{
		bool x=0;
		for(int j=2;j<=i/2;j++)
		{
			if(i%j==0)
			{
			x=1;
			break;}
		}
		if(x==0) ar[m++]=i;
	}
}
