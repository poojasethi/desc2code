#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <climits>
#include <numeric>
#include <vector>
#include <cmath>
#include <set>
#include <map>
using namespace std;
#define ll long long
#define mp make_pair
#define pb push_back
#define fi first
#define se second
#define rep(i,n) for(int i=0;i<n;i++)
#define all(a)  a.begin(),a.end()
#define ESP (1e-9)
#define N 100010
#define M 1000000007
#define sz(a) int(a.size())
#define pii pair<int,int>
#define vi vector<int>
set<int>data,val[N];map<int,int>c;
int main()
{
	//freopen("in.txt","r",stdin);
	int T,i,j,k,x,K,n;
	scanf("%d",&T);
	while(T--)
	{
	 c.clear();data.clear();
	 scanf("%d%d",&n,&K);
	 for(i=0;i<n;i++)
	 {
	  scanf("%d",&x);
	  data.insert(x);c[x]++;
	 }
	 bool ok=1;
	 for(map<int,int>::iterator it=c.begin();it!=c.end();it++)
	 {
	  if(it->se>n/K)
	  {ok=0;break;}
	 }
	 if(!ok){puts("-1");continue;}
	 for(set<int>::iterator it=data.begin();it!=data.end();it++)
	 {
	  val[c[*it]].insert(*it);
	 }
	 for(i=n;i>0;i-=K)
	 {
	  k=K;
	  vi res,use;
	  while(!val[i/K].empty())
	  {
	   int x=*val[i/K].begin();
	   val[i/K].erase(x);
	   res.pb(x);
	   data.erase(x);
	   if(i/K>1)
	   {
	    c[x]--;
	    use.pb(x);
	    val[i/K-1].insert(x);
	   }
	   k--;
	  }
	  while(k)
	  {
	   int x=*data.begin();
	   data.erase(x);
	   res.pb(x);
	   val[c[x]].erase(x);
	   if(c[x]>1)
	   {
	    c[x]--;
	    use.pb(x);
	    val[c[x]].insert(x);
	   }
	   k--;
	  }
	  sort(all(res));
	  for(j=0;j<sz(use);j++)
	  data.insert(use[j]);
	  for(j=0;j<K;j++)
	  printf("%d ",res[j]);
	 }
	 puts("");
	}
	return 0;
}
