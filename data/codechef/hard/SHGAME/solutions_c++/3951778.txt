using namespace std;

int T, N, M, xn[1000005], xm[1000005];

int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%d %d", &N, &M);
        
        memset(xn, 0, sizeof xn);
        memset(xm, 0, sizeof xm);
        
        for(int i = 0; i <= (N - 1); i++)
            xn[(N - i - 1) ^ i]++;
        for(int i = 0; i <= (M - 1); i++)
            xm[(M - i - 1) ^ i]++;     

        long long res = (long long)N * M;
        for(int i = 0; i < min(N, M); i++)
            res -= (long long)xn[i] * xm[i];
        printf("%lld\n", res);
    }
    return 0;
}
