#include <bits/stdc++.h>
using namespace std;

namespace SOL{
    const int MAX = 1123456;
    int Fre[MAX];
    void solve(){
        memset(Fre,0,sizeof(Fre));
        long long int n,m;
        cin >> n >> m;
        long long ans = n*m;
        for(int i = 1; i <= n; ++i) ++Fre[(i-1)^(n-i)];

        for(int i = 1; i <= m; ++i)
            ans -= Fre[(i-1)^(m-i)];
        cout << ans << endl;
    }
};

int main()
{
    ios_base::sync_with_stdio(0);
    int t; cin >> t;
    while(t--) SOL::solve();
    return 0;
}
