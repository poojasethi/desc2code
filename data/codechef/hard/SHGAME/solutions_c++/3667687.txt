#include <bits/stdc++.h>
using namespace std;

const int N = 1000010;
int sg[N];

int main()
{
    int t,n,m;
    cin >> t;
    while(t --) {
        cin >> n >> m;
        memset(sg,0,sizeof(sg));
        for(int i = 1; i <= n; i ++)
            sg[(i - 1) ^ (n - i)] ++;
        long long ans = 0;
        for(int i = 1; i <= m; i ++) {
            ans += n - sg[(i - 1) ^ (m - i)];
        }
        cout << ans << endl;
    }
    return 0;
}
