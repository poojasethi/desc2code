inline int getint(){
  char c = getchar();
  for(;c<'0'||c>'9';) c = getchar();
  int x = 0;
  for(;c>='0' && c<='9'; c = getchar()) x*=10, x+=c-'0';
  return x;
}
const int maxn = 1000010;
int n,m,k,x,y,q;
int a[maxn], b[maxn];
long long ans;
int main(int argc, char const *argv[])
{
	int T = getint();
	while(T--){
		n = getint(), m = getint();
		memset(a, 0, sizeof(a));
		memset(b, 0, sizeof(b));
		rep(i,n) a[i^(n-1-i)]++;
		rep(i,m) b[i^(m-1-i)]++;
		ans = 1LL*n*m;
		rep(i,n) ans -= 1LL*a[i]*b[i];
		cout<<ans<<endl;
	}
	return 0;
}
