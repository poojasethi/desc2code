struct triplet
{
    l rem0;
    l rem1;
    l rem2;
};

vector<triplet>tree(400005);
vector<l> lazy(400005);

void build(l index,l low,l high)  //O(N)
{
    if(low==high)
    {
        tree[index].rem0=1;
        tree[index].rem1=0;
        tree[index].rem2=0;
        return;
    }
    if(low>high)
        return;

    l mid=low+(high-low)/2;
    build((2*index)+1,low,mid);
    build((2*index)+2,mid+1,high);
    tree[index].rem0=tree[(2*index)+1].rem0+tree[(2*index)+2].rem0;
    tree[index].rem1=tree[(2*index)+1].rem1+tree[(2*index)+2].rem1;
    tree[index].rem2=tree[(2*index)+1].rem2+tree[(2*index)+2].rem2;
}


void update(l index,l low,l high,l L,l H) // O(log N)
{
    if(lazy[index])
        {
            if(lazy[index]==1)
            {
                l temp=tree[index].rem1;
                tree[index].rem1=tree[index].rem0;
                tree[index].rem0=tree[index].rem2;
                tree[index].rem2=temp;
            }
            else
            {
                l temp=tree[index].rem0;
                tree[index].rem0=tree[index].rem1;
                tree[index].rem1=tree[index].rem2;
                tree[index].rem2=temp;

            }
            if(low!=high)
            {
                lazy[(2*index)+1]=(lazy[(2*index)+1]+lazy[index])%3;
                lazy[(2*index)+2]=(lazy[(2*index)+2]+lazy[index])%3;
            }
            lazy[index]=0;
        }
    if(low>high or L>high or H<low)//Out of range
        return;

    if(low>=L and high<=H)
    {
        l temp=tree[index].rem1;
        tree[index].rem1=tree[index].rem0;
        tree[index].rem0=tree[index].rem2;
        tree[index].rem2=temp;

        if(low!=high)
        {
            lazy[(2*index)+1]=(lazy[(2*index)+1]+1)%3;
            lazy[(2*index)+2]=(lazy[(2*index)+2]+1)%3;
        }
        return;
    }

    l mid=low+(high-low)/2;
    update((2*index)+1,low,mid,L,H);
    update((2*index)+2,mid+1,high,L,H);
    tree[index].rem0=tree[(2*index)+1].rem0+tree[(2*index)+2].rem0;
    tree[index].rem1=tree[(2*index)+1].rem1+tree[(2*index)+2].rem1;
    tree[index].rem2=tree[(2*index)+1].rem2+tree[(2*index)+2].rem2;

}


l get(l index,l low,l high,l L,l H) //O(log N)
{
    if(lazy[index])
        {
            if(lazy[index]==1)
            {
                l temp=tree[index].rem1;
                tree[index].rem1=tree[index].rem0;
                tree[index].rem0=tree[index].rem2;
                tree[index].rem2=temp;
            }
            else
            {
                l temp=tree[index].rem0;
                tree[index].rem0=tree[index].rem1;
                tree[index].rem1=tree[index].rem2;
                tree[index].rem2=temp;
            }
            if(low!=high)
            {
                lazy[(2*index)+1]=(lazy[(2*index)+1]+lazy[index])%3;
                lazy[(2*index)+2]=(lazy[(2*index)+2]+lazy[index])%3;
            }
            lazy[index]=0;
        }

    if(low>high or L>high or H<low)//Out of range
        return 0;

    if(low>=L and high<=H)
    {
       return tree[index].rem0;
    }

    l mid=low+(high-low)/2;
    l p1=get((2*index)+1,low,mid,L,H);
    l p2=get((2*index)+2,mid+1,high,L,H);

    return (p1+p2);

}


main()
 { l N,Q,x,A,B,ans;
   //coolreshab;
 /* freopen("test.txt","r",stdin);
  freopen("out.txt","w",stdout);*/
   sl(N);
   build(0,0,N-1);
   sl(Q);

   while(Q--)

      {
          sl(x);
          switch(x)
          {
              case 0:sl(A);
                     sl(B);
                     update(0,0,N-1,A,B);
                     break;

              case 1:sl(A);
                     sl(B);
                     ans=get(0,0,N-1,A,B);
                     pl(ans);
                     nl;
                     break;
          }


      }
 }
