ll canhai(ll n)
{
    ll fi,la,mid;
    fi=0;
    la=n+1;
    while (fi+1<la)
    {
        mid=fi/2+la/2;
        if (fi%2 && la%2) mid++;
        if (mid>n/mid) la=mid;
        else fi=mid;
    }
    return fi;
}

ll power(ll a, ll n, ll m)
{
    if (n==0) return 1;
    ll tmp,c;
    tmp=c=a;
    n--;
    while (n)
    {
        if (n%2) tmp=tmp*c%m;
        c=c*c%m;
        n/=2;
    }
    return tmp;
}

struct node
{
    int d1,d2;
};

node t[800000+10];
int n,q,a,b,d,f[800000+10];

void doi(int k, int l, int r)
{
    int a2=t[k].d1;
    int b2=t[k].d2;

    t[k].d1=r-l+1-a2-b2;
    t[k].d2=a2;
}

void build_tree(int k, int l, int r, int i, int j)
{
    if (i>r || j<l) return ;
    if (i<=l && r<=j)
    {
        f[k]=(f[k]+1)%3;
        doi(k,l,r);
        return;
    }
    int m=(l+r)/2;
    while (f[k])
    {
        f[k*2]=(f[k*2]+1)%3;
        f[k*2+1]=(f[k*2+1]+1)%3;
        doi(k*2,l,m);
        doi(k*2+1,m+1,r);
        f[k]--;
    }
    build_tree(k*2,l,m,i,j);
    build_tree(k*2+1,m+1,r,i,j);
    t[k].d1=t[k*2].d1+t[k*2+1].d1;
    t[k].d2=t[k*2].d2+t[k*2+1].d2;
}

int get(int k, int l, int r, int i, int j)
{
    if (i>r || j<l) return 0;
    if (i<=l && r<=j) return (r-l+1-t[k].d1-t[k].d2);
    int m=(l+r)/2;
    while (f[k])
    {
        f[k*2]=(f[k*2]+1)%3;
        f[k*2+1]=(f[k*2+1]+1)%3;
        doi(k*2,l,m);
        doi(k*2+1,m+1,r);
        f[k]--;
    }
    int t1=get(k*2,l,m,i,j);
    int t2=get(k*2+1,m+1,r,i,j);
    return t1+t2;
}

}
