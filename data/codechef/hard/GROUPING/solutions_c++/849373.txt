typedef long long ll;

typedef pair<int,int> pie;

const double eps=1e-9;
const ld leps=1e-14;
const int MN=10000+100,INF=(1u<<31)-1;
int cnt[MN];
int now[MN];
int value[MN];
int temp[MN];
bool inlist[MN];
vector <int> list[MN];
pie srt[MN];
int t=-1,n,m,k,p;
int last=-1;
void fn()
{
	if (p!=0)
	{
		int t=now[p-1];
		p--;
		last=value[t];
		inlist[t]=0;
		for (int i=0;i<list[t].size();i++) cnt[list[t][i]]--;
	}
	for (int i=0;i<n;i++)
	{
		int u=srt[i].second;
		if (srt[i].first<last && cnt[u]==p)
		{
			now[p++]=u;
			inlist[u]=1;
			for (int j=0;j<list[u].size();j++) cnt[list[u][j]]++;
			last=srt[i].first;
		}
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin>>t;
	while (t-->0)
	{
	cin>>n>>m>>k;
	p=0;
	last=INF;
	memset(cnt,0,sizeof(cnt));
	memset(inlist,0,sizeof(inlist));
	for (int i=0;i<n;i++) list[i].clear();
	for (int i=0;i<n;i++) cin>>value[i];
	for (int i=0;i<n;i++) srt[i]=pie(value[i],i);
	sort(srt,srt+n,greater<pie>());
	for (int i=0;i<m;i++)
	{
		int a,b; cin>>a>>b;
		list[a].push_back(b);
		list[b].push_back(a);
	}
	for (int i=0;i<k;i++)
	{
		fn();
		for (int i=0;i<n;i++) if (inlist[i])
			cout<<i<<" ";
		cout<<endl;
	}
	if (t)
		cout<<endl;
	}
	return 0;
}
