#include <iostream>
#include <algorithm>
#include <vector>
#include <utility>
#include <set>
using namespace std;

typedef pair<int,int> pie;

const int MN=10000+100,INF=(1u<<31)-1;
int cnt[MN];
int now[MN];
int value[MN];
set <int> inlist;
set <int>::iterator it;
vector <int> list[MN];
pie srt[MN];
int t=-1,n,m,k,p;
int last=-1;
void fn()
{
	if (p!=0)
	{
		int t=now[p-1];
		p--;
		last=value[t];
		inlist.erase(t);
		for (int i=0;i<list[t].size();i++) cnt[list[t][i]]--;
	}
	for (int i=0;i<n;i++)
	{
		int u=srt[i].second;
		if (srt[i].first<last && cnt[u]==p)
		{
			now[p++]=u;
			inlist.insert(u);
			for (int j=0;j<list[u].size();j++) cnt[list[u][j]]++;
			last=srt[i].first;
		}
	}
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin>>t;
	while (t-->0)
	{
		cin>>n>>m>>k;
		p=0;
		last=INF;
		inlist.clear();
		for (int i=0;i<n;i++) list[i].clear(),cnt[i]=0;
		for (int i=0;i<n;i++) cin>>value[i];
		for (int i=0;i<n;i++) srt[i]=pie(value[i],i);
		sort(srt,srt+n,greater<pie>());
		for (int i=0;i<m;i++)
		{
			int a,b; cin>>a>>b;
			list[a].push_back(b);
			list[b].push_back(a);
		}
		for (int i=0;i<k;i++)
		{
			fn();
			for (it=inlist.begin();it!=inlist.end();it++)
				cout<<*it<<" ";
			cout<<"\n";
		}
		if (t)
			cout<<"\n";
	}
	return 0;
}
