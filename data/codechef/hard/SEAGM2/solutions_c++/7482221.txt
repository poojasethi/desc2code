#include <cstdio>
#include <cmath>
#include <algorithm>
#define MAX_N 13
#define MAX 10000

using namespace std;

struct player
{
    long double coeff,exponent;
}players[MAX_N];

bool cmp(player a,player b)
{
    if(a.exponent==b.exponent)
        return a.coeff<b.coeff;
    return a.exponent<b.exponent;
}

int main()
{
    int i,j,n,m,t;double p;long double ans=0,num,denom,cur;
    scanf("%i",&t);
    while(t--)
    {
    scanf("%i%i",&n,&m);
    denom=0;
    for(i=0;i<n;i++)
        {
            players[i].exponent=0;players[i].coeff=1;
            for(j=0;j<m;j++) {
            scanf("%lf",&p);players[i].coeff*=p;
            if(p*MAX<0.1) players[i].exponent=0;
            else if(players[i].coeff<1) {players[i].coeff*=MAX;players[i].exponent-=4;}
            }
            while(players[i].coeff>=10) {players[i].coeff/=10;players[i].exponent++;}
        }
    num=players[0].coeff;j=players[0].exponent;
    sort(players,players+n,cmp);
    denom=0;
    for(i=0;i<n;i++)
    {
        cur=players[i].coeff*powl(10,players[i].exponent-j);
        denom+=cur;
    }
    if(num>0.1) ans=num/denom;
    printf("%0.14lf\n",(double)ans);
    }
    return 0;
}
