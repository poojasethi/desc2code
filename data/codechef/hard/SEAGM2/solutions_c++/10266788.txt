#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n,m,t;
	double a[20][10005];
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=m;j++)
			cin>>a[i][j];
		}
		double ans=1;
		int flag=1;
		for(j=1;j<=m;j++)
		{
			if(a[1][j]<=0.00+1e-9)
			flag=0;
		}
		if(!flag)
		{
			cout<<"0.000000\n";
			continue;
		}
	
		double temp=0;
		
		for(i=1;i<=n;i++)
		{   
		    double tt=1;
			for(j=1;j<=m;j++)
			{
				tt*=a[i][j]/a[1][j];
			}
			temp+=tt;
		}
		if(temp<=0.0+1e-9)
		cout<<"0.000000\n";
		else
		printf("%.6lf\n",1/temp);
		
	}
	
}
