#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
int main()
{
	int i,j,c,indi,t,n,m;
	double array[13][10000],cnt,out;
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		scanf("%d",&m);
		for(i=0;i<n;i++)
            for(j=0;j<m;j++)
                scanf("%lf",&array[i][j]);
		for(j=0,indi=0;j<n;j++){
			if(array[0][j]==0){
				indi=1;
				break;
			}
		}
		if(indi==1)printf("0.000000\n");
		else{
			for(i=1,indi=0,out=1;i<n;i++){
				for(j=0,cnt=1,c=0;j<m;j++){
					cnt=cnt*array[i][j]/array[0][j];
					if(cnt>100000000){
						cnt=cnt/100000000;
						c++;
					}
					if(cnt<0.00000001){
						cnt=cnt*100000000;
						c--;
					}
				}
				if(c>0){
					printf("0.000000\n");
					indi=1;
					break;
				}
				else if(c==0)out=out+cnt;
			}
			if(indi==0)printf("%.6lf\n",1/out);
		}
	}
}
