void solve(){
	
	ld ans=win/r;
	
	if(win==0) ans=0;

}
 
bool input(){
	
	sl(n); sl(m);
	
	win=1.0;
	
	den=0;
	
	F(i,1,n){
		ld temp=1.0;
		F(j,1,m)
		{
			sf(p[i][j]);
			
			p[i][j]*=25;
			
			if(i==1)
			win=win*p[i][j];	
			
			temp=temp*p[i][j];
		}	
		
		den=den+temp;
	}
	
	r=den;
 
	return true;
}
