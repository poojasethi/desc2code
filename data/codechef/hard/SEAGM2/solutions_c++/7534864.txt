#include <stdio.h>
using namespace std;
typedef long long ll;
#define MAXN 14
#define MAXM 10004
long double P[MAXN][MAXM];
void solve() {
    int N, M;
    long double temp, ans;
    scanf("%d %d", &N, &M);
    long double sum = 0;
    long double num = 1l;
    for(int i = 0; i < N; i++) {
        num = 1l;
        for(int j = 0; j < M; j++) {
            scanf("%Lf", &temp);
            temp = temp * 10l;
            P[i][j] = temp;
            num = num * temp;
        }
        sum = sum + num;
    }
    num = 1l;
    for(int i = 0; i < M; i++) num = num * P[0][i];
    if(sum == 0) ans = 0l;
    else ans = num / sum;
    printf("%0.6Lf\n", ans);
}
int main() {
    int t;
    scanf("%d", &t);
    while(t--) solve();
}
