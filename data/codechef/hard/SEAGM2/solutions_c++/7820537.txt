#include <iostream>
#include <cstdio>
using namespace std;

int main(){
    int t;
    cin>>t;
    while(t--){
        int n,m;
        cin>>n>>m;
        double arr[m];
        bool g=true;
        for(int i=0;i<m;i++){
            cin>>arr[i];
            if(arr[i]==0.00)
                g=false;
        }
        if(g){
            double ans=1.00;
            for(int i=1;i<n;i++){
                double temp=1.00,num;
                for(int j=0;j<m;j++){
                    cin>>num;
                    temp*=(num/arr[j]);
                }
                ans+=temp;
            }
            printf("%.6f\n",1/ans);
        }else{
            double num;
            for(int i=1;i<n;i++){
                for(int j=0;j<m;j++)
                    cin>>num;
            }
            printf("0.000000\n");
        }
    }
}
