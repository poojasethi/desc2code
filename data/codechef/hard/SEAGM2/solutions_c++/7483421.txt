#include<cstdio>
using namespace std;
double a[25];
int main()
{
    int t, n, m, i, j;
    scanf("%d", &t);
    double x;
    while(t--)
    {
        scanf("%d %d", &n, &m);
        double sum=0.0;
        for (i=0;i<n;i++)
        {
            a[i]=1.0;
            for(j=0;j<m;j++)
            {
                scanf("%lf",&x);
                a[i]*=x*25;
            }
            sum+=a[i];
        }
	        printf("%.6lf\n",a[0]/((sum>0)?sum:1));       
    }
    return 0;
}
