# include <bits/stdc++.h>
using namespace std;
# define ld long double
# define f(i,c,n) for(i=c;i<n;i++)
# define sfi(p) scanf("%d",&p)
# define sfld(p) scanf("%Lf",&p)


int main()
{
	int t,n,m,i,j;
	sfi(t);
	while(t--)
	{
		sfi(n);sfi(m);
		ld val[13];
		int power[13];
		f(i,0,n)
		{
			ld tval=1;
			int tpower=0;
			ld inp;
			f(j,0,m)
			{
				sfld(inp);
				inp*=1e4;
				tpower+=4;
				tval*=inp;
				while(tval>1e4)
				{
					tval/=1e4;
					tpower-=4;
				}
			}
			val[i]=tval;
			power[i]=tpower;
		}
		ld num,den=0;
		num=val[0];
		if(num<1e-9)
			printf("%.9Lf\n",(long double)0);
		else
		{
			f(i,0,n)
			{
				den+=val[i]*pow(10,(power[0]-power[i]));
			}
			printf("%.9Lf\n",(num/den));
		}
	}
	return 0;
}
