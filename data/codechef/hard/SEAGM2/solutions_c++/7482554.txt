#include<bits/stdc++.h>
 
using namespace std;
 
int main()
{
	ios_base::sync_with_stdio(0);
	int t,i,j,n,m,lulz;
	double lol,temp,ans,meme;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		lol=0;
		double p[n][m];
		for(i=0;i<n;i++)
		{
			for(j=0;j<m;j++)
			{
				cin>>p[i][j];
				p[i][j]*=25;
			}
		}
		meme=1;
		for(i=0;i<m;i++)
		{	
			meme*=p[0][i];
		}
		for(i=0;i<n;i++)
		{
			temp=1;
			for(j=0;j<m;j++)
			{
				temp*=p[i][j];
			}
			lol+=temp;
		}
		ans=meme/lol;
		lulz=ans*1000000000;
		
		if(meme==0) ans=0;
		cout<<ans<<"\n";
	}
	return 0;	
}
