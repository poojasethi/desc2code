#include <stdio.h>
#include <math.h>
using namespace std;
#define MAXN 14
#define MAXM 10001
typedef long long ll;
#define double long double
double P[MAXN][MAXM];
void solve() {
    ll N, M;
    double temp, sum = 0l;
    scanf("%lld %lld", &N, &M);
    for(int i = 0; i < N; i++) {
        double prod = 1l;
        for(int j = 0; j < M; j++) {
            scanf("%Lf", &temp);
            P[i][j] = temp * 10l;
            prod = prod * (P[i][j]);
        }
        sum = sum + prod;
    }
    double num = 1l;
    for(int i = 0; i < M; i++) num = num * P[0][i];
    double ans;
    if(sum == 0) ans = 0l;
    else ans = num / sum;
    printf("%0.6Lf\n", ans);
}
int main() {
    int t;
    scanf("%d", &t);
    while(t--) solve();
}
