#include<cstdio>
#include<iostream>
#define f 10
int main()
{
	int t,n,m;
	long double p,sum,a,q,ans;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&m);
		p=1;
		for(int i=0;i<m;i++){
			scanf("%Lf",&a);a=a*f;
			p*=a;
		}
		sum=p;
		for(int j=1;j<n;j++)
		{ 
			q=1;
			for(int i=0;i<m;i++)
			{
				scanf("%Lf",&a);a*=f;
				q*=a;
			}
			sum+=q;
		}
		if(p==0)
			ans=0;
		else
			ans=p/sum;
		printf("%Lf\n",ans);
	}
	return 0;
}
