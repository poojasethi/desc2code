#include<stdio.h>
int main()
{
    double arr[15][10005];
    int t,n,m,i,j,k;
    double ans;

    scanf("%d",&t);

    while(t--)
    {
        scanf("%d %d",&n,&m);

        for(int i=0;i<n;i++)
        {
            for(int j=0;j<m;j++)
            {
                scanf("%lf",&arr[i][j]);
            }
        }

        double denom=0;

        for(k=0;k<m;k++)
        {
            if(arr[0][k]==0)
            break;
        }

        if(k!=m)
        printf("0.000000\n");

        else
        {
            for(int i=0;i<n;i++)
            {
                double temp=1;
                for(int j=0;j<m;j++)
                {
                    temp=temp*arr[i][j]/arr[0][j];
                }

                denom=denom+temp;
            }

            double ans;

            ans=1/denom;

            printf("%lf\n",ans);
        }
    }
    return 0;
}
