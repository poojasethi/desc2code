import sys, decimal
raw_input = sys.stdin.readline

for tc in range(int(raw_input())):
    N, M = map(int, raw_input().split())
    deno = 0
    for n in range(N):
        product = decimal.Decimal('1.0')
        for p in map(decimal.Decimal, raw_input().split()): product *= p
        
        if n == 0: num = product
        
        deno += product
        
    print "%.6f" % (num/deno if num else 0)
