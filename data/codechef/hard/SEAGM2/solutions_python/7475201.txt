from operator import mul

parse = lambda: [25*i for i in map(float, raw_input().split())]

for _ in xrange(int(raw_input())):
	n, m = map(int, raw_input().split())
	num = reduce(mul, parse())
	den = num
	for _ in xrange(n - 1):
		den += reduce(mul, parse())
	if num == 0:
		den = 1
	print '%.6f' % (num/den)
