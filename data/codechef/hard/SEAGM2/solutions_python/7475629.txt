from operator import mul
from decimal import Decimal
for _ in xrange(int(raw_input())):
	n, m = map(int, raw_input().split())
	num = reduce(mul, map(Decimal, raw_input().split()), Decimal(1))
	den = num
	for i in xrange(n - 1):
		den += reduce(mul, map(Decimal, raw_input().split()), Decimal(1))
	if den == 0:
		print '0.000000'
	else:
		print '%.6f' % (num/den)
