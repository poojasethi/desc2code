LL dp[200010];
int K,a[10],b[10];
#define MOD 1000000007
int mx=0;
vector<LL>prime;
void sol()
{
    for(int j=mx;j>=1;j--)
    {
        LL ret=1;
        FOR(i,K)
        {
            ret*=b[i]/j-(a[i]-1)/j;
            ret%=MOD;
        }
        for(int i=j+j;i<=mx;i+=j)
        {
            ret-=dp[i];
            if(ret<0)ret+=MOD;
            if(ret>MOD)ret-=MOD;
        }
        dp[j]=ret;
    }


}
LL bm(LL a,LL p,LL M)
{
    if(p==0)return 1;
    if(p%2==1)
    {
        LL x=bm(a,p-1,M);
        x*=a;
        x%=M;
        return x;
    }
    else
    {
        LL x=bm(a,p/2,M);
        x*=x;
        x%=M;
        return x;
    }
}
