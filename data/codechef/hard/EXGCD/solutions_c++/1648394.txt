#include<stdio.h>
#include<iostream>
using namespace std;
typedef long long ll;
int t,n,a[9],b[9],m,v[200020],p=1000000007;
ll x,y;
ll pw(ll x,ll y)
{
	ll _=1;
	for(;y;y>>=1,x=x*x%p)
		if(y&1)
			_=_*x%p;
	return _;
}
int main()
{
	for(int i=1;i<=200000;i++)
		v[i]=i;
	for(int i=2;i<=200000;i++)
		if(v[i]==i)
			for(int j=i;j<=200000;j+=i)
				v[j]=v[j]/i*(i-1);
	for(scanf("%d",&t);t--;)
	{
		scanf("%d",&n);
		x=0,y=1;
		for(int i=m=0;i<n;i++)
		{
			scanf("%d%d",a+i,b+i);
			a[i]--;
			m=max(m,b[i]);
			y=y*(b[i]-a[i])%p;
		}
		for(int i=1;i<=m;i++)
		{
			ll u=1;
			for(int j=0;j<n;j++)
				u=u*(b[j]/i-a[j]/i)%p;
			x=(x+u*v[i])%p;
		}
		if(x<0)
			x+=p;
		printf("%lld\n",(p-x)*pw(y,p-2)%p);
	}	
	return 0;
}
