#define N 1000007

using namespace std;

ll k, ans, a[N], S[N];
int n, C[N], cnt;

int Get(int x)
{
    int r = 0;
    while(x)
    {
        r += C[x];
        x -= x & (-x);
    }
    return r;
}

void Up(int x)
{
    while(x < N)
    {
        C[x]++;
        x += x & (-x);
    }
}

int main()
{
    //filein("inp.txt");  fileout("out.txt");
    cin >> n >> k;
    for(int i=1; i<=n; i++)
    {
        scanf("%lld", &a[i]);
        a[i] += a[i-1];
        S[i] = -k*i + a[i];
    }
    S[n+1] = 0;
    sort(S+1, S+n+2);
    for(int i=0; i<=n; i++)
    {
        int pos = lower_bound(S+1, S+n+2, -k*i+a[i]) - S;
        ans += Get(pos);
        Up(pos);
    }
    cout << ans;
}
