using namespace std;

ll int n,k,a[1000004];
ll int b[1000004];
ll int tree[1000004];

int comp(int i,int j)
{
	return (make_pair(b[i],i)<make_pair(b[j],j));
}

int calc(ll int id)
{
	ll int ans=0;
	while(id>0)
	{
	//	pr("*\n");
		ans+=tree[id];
		id-=(id & -id);
	}
	return ans;
}

void up(ll int id)
{
	while(id<1000002)
	{
	//	pr("#\n");
		tree[id]+=1;
		id+=(id & -id);	
	}
}

int main()
{
	sll(n);sll(k);
	ll int i,j,l,m,p,count=0;
	f(i,1,n+1)
	{
		sll(m);
		m-=k;
		if(!i)
			b[i]=m;
		else
			b[i]=m+b[i-1];
		a[i]=i;
	}
	sort(a,a+n+1,comp);

	ff(i,n+1)
	{
		count+=calc(a[i]+1);
		up(a[i]+1);
	}
	pll(count);
