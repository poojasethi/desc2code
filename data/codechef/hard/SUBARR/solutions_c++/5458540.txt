#include<bits/stdc++.h>
using namespace std;
#define ll long long int
ll tree[10000000], A[1000006];
ll maxval = 1000006;
pair< ll, ll > B[1000006];
void update(ll idx, ll val) {
    while(idx <= maxval) {
        tree[idx] += val;
        idx += (idx & -idx);
    }
    //cout << "hi" ;
}
ll query(ll idx) {
    ll ans = 0;
    while(idx > 0) {
        ans += tree[idx];
        idx -= (idx & -idx);
        //cout << "hi" << endl;
    }
    return ans;
}
int main() {
    ll n, k, ans = 0, x;
    cin >> n >> k;
    for(int i = 1; i <= n; i++) {
        cin >> x;
        x -= k;
        if(i == 1) A[i] = x;
        else A[i] = A[i - 1] + x;
    }
    for(int i = 1; i <= n; i++) {
        B[i] = make_pair(A[i], i);
    }
    sort(B + 1, B + n + 1);
    for(int i = 1; i <= n; i++) {
        //cout << B[i].second << endl;
        ans += query(B[i].second);
        if(A[i] >= 0) ans ++;
        update(B[i].second, 1);
    }
    cout << ans;

}
