#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ull unsigned long long
#define pb push_back
#define ft first
#define se second
#define mp make_pair
ll *prefsum;
int *fenwick, *a;
int *ind;
int n, k;
bool comp(int i, int j){
	return mp(prefsum[i], i)<mp(prefsum[j], j);
}
void update(int pos){
	while(pos<n){
		fenwick[pos]+=1;
		pos|=(pos+1);
	}
}
int query(int ind){
	int ans=0;
	while(ind>=0){
		ans+=fenwick[ind];
		ind=(ind & (ind+1))-1;
	}
	return ans;
}
int main(int argc, char const *argv[]){
	fenwick=new int[1000100];
	for(int i = 0; i < n; ++i){
		fenwick[i]=0;
	}//for i
	prefsum=new ll [1000100];
	ind = new int[1000100];
 	scanf("%d%d", &n, &k);
 	for(int j = 0; j < n; ++j){ 
 		int x; 
 		scanf("%d", &x);	
 		x-=k;
 		prefsum[j]=x;
 		ind[j]=j;
 		if(j) prefsum[j]+=prefsum[j-1];
 	}//for j 	
 	sort(ind, ind+n, comp);
 	ll ans=0;
 	for(int i = 0; i < n; ++i){
 		ans+=(query(ind[i]));
 		if(prefsum[ind[i]]>=0) ans+=1;
 		update(ind[i]);
 	}//for i
 	printf("%lld\n", ans);
	return 0;
}
