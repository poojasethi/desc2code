#include <bits/stdc++.h>
using namespace std;

#define ll long long
const int N = 1e6 + 5;

map<ll, int> M;
ll A[N], pos[N], B[N], BIT[N];

void update(int index){
	while(index < N){
		BIT[index]++;
		index += (index&(~(index - 1)));
	}
}

ll query(int index){
	ll tot = 0;
	while(index){
		tot += BIT[index];
		index -= (index&(~(index - 1)));
	}
	return tot;
}

int main(){
	iostream::sync_with_stdio(false);
	cin.tie(NULL);

	int n, k, cntr = 1;
	ll ans = 0;
	cin>>n>>k;
	M[0] = 0;
	for(int i = 1; i <= n; i++){
		cin>>A[i];
		B[i] = B[i - 1] + A[i] - k;
		M[B[i]] = 0;
	}

	for(map<ll, int>::iterator it = M.begin(); it != M.end(); it++)
		M[it->first] = cntr++;

	for(int i = 1; i <= n; i++)	pos[i] = M[B[i]];

	update(M[0]);

	for(int i = 1; i <= n; i++){
		ans += query(pos[i]);
		update(pos[i]);
	}
	cout<<ans<<endl;
}
