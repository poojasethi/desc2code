#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL A[1000001];
LL tree[1100001];
void update(int idx , int val)
{
    for(idx+=1;idx<=1100000;idx+=idx&-idx)
        tree[idx]+=val;
}
int get(int idx)
{
    int ret = 0;
    for(idx+=1;idx;idx-=idx&-idx)
        ret+=tree[idx];
    return ret;
}

int main()
{
    ios::sync_with_stdio(0);cin.tie(0);
    int N , K , i ,temp;
    cin>>N>>K;
    vector < LL > V;
    vector < LL > Pre(N+1,0);
    for(i=1;i<=N;++i)
    {
        cin>>temp;
        Pre[i]=Pre[i-1] + temp - K;
        V.push_back(Pre[i]);
    }
    V.push_back(0);
    sort(V.begin(),V.end());
    LL ans = 0;
    update(lower_bound(V.begin(),V.end(),0)-V.begin(),1);
    for(i=1;i<=N;++i)
    {
        int x= lower_bound(V.begin(),V.end(),Pre[i]) - V.begin();
        ans += get(x);
        update(x,1);
    }
    cout<<ans<<"\n";

    return 0;
}
