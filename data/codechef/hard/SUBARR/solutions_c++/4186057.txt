#define FOR(i,a,b) for (int i=(a),_b=(b);i<_b;i++)
#define FORE(i,a,b) for (int i=(a),_b=(b);i<=_b;i++)
#define FORD(i,a,b) for (int i=(a),_b=(b);i>=_b;i--)
#define ri(n)({\
    int neg=0;\
    n=0;\
    char ch;\
    for(ch=getchar(); ch<'0' || ch>'9'; ch=getchar()) if (ch=='-') neg=1-neg;\
    n=ch-48;\
    for(ch=getchar(); ch>='0' && ch<='9'; ch=getchar()) n=(n<<3)+(n<<1)+ch-48;\
    if (neg) n=-n;\
})

typedef long long ll;
typedef pair<int,int> II;
const int INF=1000000000+7;
const double esp=1e-13;
const double pi=3.141592653589;

struct fa
{
    ll gt;
    int vt;
};

bool cmp(fa a, fa b)
{
    return (a.gt<b.gt || (a.gt==b.gt && a.vt<b.vt));
}

fa sum[1000000+10];
ll a[1000000+10],ans;
int n,k,f[1000000+10];

ll get(int x)
{
    ll gtx=0;
    while (x)
    {
        gtx+=f[x];
        x-=x&(-x);
    }
    return gtx;
}

void update(int x)
{
    while (x<=n)
    {
        f[x]++;
        x+=x&(-x);
    }
}

int main()
{
 //   freopen("SUBARR.inp", "r", stdin);
 //   freopen("SUBARR.out", "w", stdout);
    ri(n); ri(k);
    FORE(i,1,n) ri(a[i]);
    FORE(i,1,n) a[i]-=k;
    sum[0].gt=0;
    FORE(i,1,n)
    {
        sum[i].gt=sum[i-1].gt+a[i];
        sum[i].vt=i;
    }
    sort(sum+1,sum+n+1,cmp);
    ans=0;
    memset(f,0,sizeof(f));
    FORE(i,1,n)
    {
        ans+=(sum[i].gt>=0);
        ans+=get(sum[i].vt);
        update(sum[i].vt);
    }
    cout << ans;
}
