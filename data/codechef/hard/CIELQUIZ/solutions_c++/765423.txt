#include <iostream>
#include <algorithm>
#include <utility>
using namespace std;

int n,k,l,r;
pair <double,int> a[40];

double calc(int l,int r)
{
	double b[40],p=1,s=0;
	for (int i=1;i<=l;i++) b[i]=a[i].first;
	for (int i=1;i<=r;i++) b[l+i]=a[n+1-i].first;
	for (int i=1;i<=k;i++) p*=b[i];
	for (int i=1;i<=k;i++)
	{
		double P=1;
		for (int j=1;j<=k;j++)
			if (i!=j) P*=b[j];
		s+=P;
	}
	return s-p*k;
}

int main()
{
	int test;
	cin >> test;
	while (test--)
	{
		double ans=-1;
		cin >> n >> k;
		for (int i=1;i<=n;i++) 
		{
			cin >> a[i].first; 
			a[i].first*=0.01;
			a[i].second=i;
		}
		sort(a+1,a+n+1);
		for (int i=0;i<=k;i++) 
		{
			double u=calc(i,k-i);
			if (u>ans) ans=u, l=i, r=k-i;
		}
		for (int j=1;j<=l;j++) cout << a[j].second << ' ';
		for (int j=1;j<=r;j++) cout << a[n+1-j].second << ' ';
		cout << endl;
	}
}
