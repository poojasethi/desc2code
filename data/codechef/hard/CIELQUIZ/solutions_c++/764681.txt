#include <stdio.h>
#include <algorithm>
using namespace std;
struct Int{
    int num;
    int id;
};
int n,k;
Int num[40];
bool operator < (const Int a,const Int b){
    return a.num < b.num;
}
Int res[40];
Int bres[40];
double mejor;
void checa(int p){
    int pos = 0;
    int i;
    for(i=0;i<p;i++){
        res[pos] = num[i];
        pos++;
    }   
    for(i=n-1;pos < k;i--){
        res[pos] = num[i];
        pos++;
    }
    double total = 0.0;
    for(int i=0;i<k;i++){
        double prod=1.0;
        for(int j=0;j<k;j++){
            if( j != i){
                prod *= ((double)res[j].num)/(double)100.0;
            }else{
                prod *= 1.0 -((double)res[j].num)/(double)100.0;
            }
        }
        total += prod;
    }
    if( total >mejor){
        mejor = total;
        for(int i=0;i<k;i++){
            bres[i] = res[i];
        }
    }
}
void imprime(){
    for(int i=0;i<k;i++){
        printf("%d ",bres[i].id);
    }
    printf("\n");
}
void caso(){
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++){
        scanf("%d",&num[i].num);
        num[i].id = i+1;
    }
    sort(num,num+n);
    mejor = -1000.0;
    for(int i=0;i<=k;i++){
        checa(i);
    }
    imprime();
}
int main(){
    int tcase;
    scanf("%d",&tcase);
    for(int i=0;i<tcase;i++){
        caso();
    }
    return 0;
}
