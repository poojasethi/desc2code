pair <int,int> p[50];

double calc(vector < pair <int,int> > now)
{
	double sum=0;

	for(int i=0;i<now.S;i++)
	{
		double tmp=(100-now[i].first)/100.0;

		for(int j=0;j<now.S;j++)
			if(i!=j) tmp=tmp*now[j].first/100.0;
		sum+=tmp;
	}
	return sum;
}

int main()
{
	int test;

	for(cin>>test;test;test--)
	{
		int n,k;
		vector < pair <int,int> > ans;
		double res=-1,tmp;

		cin>>n>>k;
		for(int i=0;i<n;i++) 
		{
			cin>>p[i].first;
			p[i].second=i+1;
		}

		sort(p,p+n);
		for(int i=0;i<=k;i++)
		{
			vector < pair <int,int> > pt;

			for(int j=0;j<i;j++) pt.P(p[j]);
			for(int j=n-(k-i);j<n;j++) pt.P(p[j]);
			if(res<(tmp=calc(pt)))
			{
				res=tmp;
				ans=pt;
			}
		}
		for(int i=0;i<k;i++) cout<<ans[i].second<<' ';
		cout<<endl;
	}
	return 0;
}
