#include <cstdio>
#include <cstring>
#include <iostream>

const int Mod = 1000000007;

using namespace std;

int p[44], num[44];

int main (void)
{
  int test, tests;
  int i, j, l, m, t, n, k, z, cnt0;
  int ans[6];
  long double res, tres, ttres;
  scanf ("%d", &tests);
  for (test = 0; test < tests; test++)
  {
    scanf ("%d %d", &n, &k);
    for (i = 0; i < n; i++)
      scanf ("%d", &p[i]);
    for (i = 0; i < n; i++)
      num[i] = i+1;
    for (i = 1; i < n; i++)
      for (j = 1; j < n; j++)
        if (p[j] < p[j-1])
        { 
          t = p[j],	p[j]   = p[j-1], 	p[j-1]   = t;
          t = num[j], 	num[j] = num[j-1], 	num[j-1] = t;
        }
    res = -1;
    memset (ans, 0, sizeof(ans));
    ans[0] = 0; ans[1] = k;
    for (i = 0; i < n; i++)
      for (j = i; j <= n; j++)
        for (l = j; l < n; l++)
          for (m = l; m <= n; m++)
            if (j - i + m - l == k)
            {
              tres = 1;
              cnt0 = 0;
              ttres = 0;
              
              for (z = 0; z < n; z++)
                if ((i <= z && z < j) || (l <= z && z < m))
                {
                  if (p[z] == 0)
                    cnt0++;
                  else tres *= p[z];
                }
              
              if (cnt0 >= 2)
              {
                if (res < 0)
                {
                    res = 0;
                    ans[0] = i, ans[1] = j, ans[2] = l, ans[3] = m;
                }
              }

              else if (cnt0 == 1)
              {
                if (res < tres * 100)    
                {
                    res = tres * 100;
                    ans[0] = i, ans[1] = j, ans[2] = l, ans[3] = m;
                }
              }

              else
              { 
                for (z = 0; z < n; z++)
                  if ((i <= z && z < j) || (l <= z && z < m))
                    ttres += tres / p[z] * (100 - p[z]);

                if (res < ttres)
                {
                  res = ttres;
                  ans[0] = i, ans[1] = j, ans[2] = l, ans[3] = m;
                }
              }
            }
          
    for (z = 0; z < n; z++)
      if ((ans[0] <= z && z < ans[1]) || ((ans[2] <= z && z < ans[3])))
        printf ("%d ", num[z]);
    printf ("\n");    

  }
  return 0;
}
