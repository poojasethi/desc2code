#include <stdio.h>
#include <algorithm>
#include <string.h>
using namespace std;

#define N 45

double aa[N], a[N];
int b[N];
int u[N];
int rr[N];
double res, t, r;

int i,j ,T, k, n,m ,x, y, z, l;
int pr(int x, int y) {
	return aa[x] < aa[y];
}
int main() {
	//freopen("in.txt", "r", stdin);
	scanf("%d", &T);
	while (T --) {
		scanf("%d%d", &n, &m);
		for ( i= 0; i < n; i ++) {
			scanf("%lf", &aa[i]);
			aa[i] /= 100;
			b[i] = i;
		}
		sort(b, b + n, pr);
		
		for (i = 0; i < n; i ++) {
			a[i] = aa[b[i]];
		}

		res = -1;

		for (i = 0; i <= m; i ++) {
			memset(u, 0, sizeof(u));
			for (j = 0; j < i; j ++) {
				u[j] = 1;
			}
			for (j = 0; j < m - i; j ++) {
				u[n-j-1] = 1;
			}
			r = 0;
			for (j = 0; j < n; j ++) {
				if (u[j] != 1) {
					continue;
				}
				t = 1.0 - a[j];
				for (k = 0; k < n; k ++) {
					if (u[k] != 0 && k != j) {
						t *= a[k];
					}
				}
				r += t;
			}
			if (r > res) {
				memset(rr, 0, sizeof(rr));
				res = r;
				for (j = 0; j < i; j ++) {
					rr[b[j]] = 1;
				}
				for (j = 0; j < m - i; j ++) {
					rr[b[n-j-1]] = 1;
				}
			}
		}
		for (i = 0; i < n; i ++) {
			if (rr[i] == 1) {
				printf("%d ", i + 1);
			}
		}
		printf("\n");
	}
	return 0;
}
