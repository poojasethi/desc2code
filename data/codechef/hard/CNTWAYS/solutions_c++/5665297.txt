int fact[800005],facti[800005];
int power(LL a,LL b)
{	int ans=1,temp=a;
	while(b>0)
	{	if(!(b&1))
		{	temp=(temp*1LL*temp)%mod;
			/*if(temp>=mod) temp%=mod;*/
			b=b>>1;
		}
		else
		{	ans=(ans*1LL*temp)%mod;
			/*if(ans>=mod) ans%=mod;*/
			b--;
		}
	}
	return ans;
}
void precompute()
{	int i,j;
	fact[0]=1;facti[0]=1;
	for(i=1;i<=800000;i++)
	{	fact[i]=(fact[i-1]*1LL*i)%mod;
		facti[i]=(power(fact[i],mod-2))%mod;
	}
}
int calc(int a,int b)
{	return (fact[a+b]%mod*(facti[a]%mod*1LL*facti[b]%mod)%mod)%mod;
}
int main()
{	int t,i,j,n,m,a,b,ans;
	precompute();
	//printf("%d %d %d %d \n",facti[5],facti[4],facti[3],facti[2]);
	sf(t);
	while(t--)
	{	sf(n);sf(m);sf(a);sf(b);
		ans=0;
		for(j=0;j<=n-a;j++)
		{	//printf("\n%d %d\n",calc(j,b-1LL),calc(n-j,m-b));
			ans=(ans+(calc(j,b-1LL)*1LL*calc(n-j,m-b))%mod)%mod;
			//printf("%d \n",ans);
		}
		printf("%d\n",ans);
	}
	return 0;
}
