#include <iostream>
#include <stdio.h>

using namespace std;

#define MOD 1000000007
long long fact[800002];
long long ifact[800002];

long long compute(long long x,long long y){
    return (((fact[x]*ifact[y])%MOD) * ifact[x-y])%MOD;
}

long long ways(long long a,long long b){
    return compute(a+b,a);
}

int main()
{
    fact[0]=ifact[0]=1;
    for(int i=1;i<=800000;i++){
        fact[i]=(i*fact[i-1])%MOD;
        long long a,b,result=1;
        b=MOD-2;
        a=fact[i];
        while(b>0){
            if(b%2==1){
                result=(result*a)%MOD;
            }
            a=(a*a)%MOD;
            b/=2;
        }
        ifact[i]=result;
    }
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++){
        int m,n,a,b;
        scanf("%d%d%d%d",&n,&m,&a,&b);
        long long res=0;
        if(a==0||b==0){
            res=ways(n,m);
            printf("%d\n",(int)res);
            continue;
        }

        for(int p=a;p<n+1;p++){
            res+=ways(p,m-b)*ways(n-p,b-1);
            res%=MOD;
        }
        printf("%d\n",(int)res);
    }
    return 0;
}
