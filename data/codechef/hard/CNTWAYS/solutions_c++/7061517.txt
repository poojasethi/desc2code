#include <iostream>
using namespace std;
#define p 1000000007
#define ll long long

ll powmodp(ll a, ll b) {
    ll w=1;  ll x=a;
    while (b) {if (b&1) w=w*x%p; x=x*x%p; b>>=1;    }
    return w;
}

ll odwmodp(ll a) {
  ll u = 1; ll w = a;
  ll x = 0; ll z = p;
  ll q;
  while(w) {
    if(w < z) {
      q = u; u = x; x = q;
      q = w; w = z; z = q;
    }
    q = w / z;
    u -= q * x;
    w -= q * z;
  }
  if(x < 0) x += p;
  return x;    
}

ll divmodp(ll a, ll b) {
    return a*odwmodp(b)%p;
}

ll binom(ll n, ll k) {
    if (k>n/2) k=n-k;
    ll g=1; for(int i=n-k+1;i<=n;++i) g=g*i%p;
    ll d=1; for(int i=2;i<=k;++i) d=d*i%p;
    return divmodp(g,d);
}

int R,N,M,A,B;
ll g1[400001],g2[400001];
int main() {
//for (int i=1; i<10;i++) printf("%d %lld %lld\n",i,powmodp(i,p-2),odwmodp(i));
scanf("%d",&R);
//R=1;
while(R--) {
    scanf("%d%d%d%d",&N,&M,&A,&B);
    //N=M=400000; A=B=1;
    ll n1=M-B+A;    ll k1=M-B;    ll w1=binom(n1,k1);
    ll n2=N-A+B;    ll k2=N-A;    ll w2=binom(n2,k2);
    ll w=w1*w2%p;
    ll wynik=w;
    ll k=min(k1,k2);
    ll d=1; for (int i=1;i<=k;++i) {d=d*(n1-k1+i)%p; d=d*(n2-k2+i)%p;}
    g1[0]=k1*k2%p;
    for (int j=1;j<k;++j) {ll g=(k1-j)*(k2-j)%p; g1[j]=g1[j-1]*g%p;}
    g2[k+1]=1;
    g2[k]=(n1-k1+k)*(n2-k2+k)%p;
    for (int j=k-1;j>=2;--j) {ll g=(n1-k1+j)*(n2-k2+j)%p; g2[j]=g2[j+1]*g%p;}
    ll s=0;
    for (int i=0;i<k;++i) {
        //ll g=1;
        //for (int j=0;j<=i;++j) {g=g*(k1-j)%p; g=g*(k2-j)%p;}
        //for (int j=i+2;j<=k;++j) {g=g*(n1-k1+j)%p; g=g*(n2-k2+j)%p;}
        ll g=g1[i]*g2[i+2]%p;
        s=(s+g)%p;
    }
    w=w*divmodp(s,d)%p;
    wynik=wynik+w;
    printf("%lld\n",wynik%p);
}
return 0;
}
