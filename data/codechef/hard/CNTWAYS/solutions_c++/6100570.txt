#include <bits/stdc++.h>
#define ll long long
using namespace std;
const ll mod=1000000007;
const ll ma=1000000;
ll fact[ma],ifact[ma];
ll exp(ll a,ll b)
{
    ll res=1;
    while(b)
    {
        if(b&1)
        {
            res=(res*a)%mod;
        }
        a=(a*a)%mod;
        b>>=1;
    }
    return res;
}
void cal_fac()
{
    fact[0]=1;ifact[0]=1;
    for(ll i=1;i<=800000;i++)
    {
        fact[i]=(i*fact[i-1])%mod;
        ifact[i]=exp(fact[i],mod-2)%mod;
    }
}
ll c(ll a,ll b)
{
    return ((fact[a]*ifact[b]%mod)*ifact[a-b])%mod;
}
ll ways(ll a,ll b)
{
    return c(a+b,b);
}
int main()
{
    //freopen("i.txt","r",stdin);
    ll r,n,m,a,b;
    cal_fac();
    for(scanf("%lld",&r);r;--r)
    {
      scanf("%lld%lld%lld%lld",&n,&m,&a,&b);
      ll res=0;
      for(ll p=0;p<=n-a;p++)
      {
          res+=(ways(p,b-1)*ways(n-p,m-b))%mod;
          res%=mod;
      }
      printf("%lld\n",res);
    }
    return 0;

}
