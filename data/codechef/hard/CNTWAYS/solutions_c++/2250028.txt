using namespace std;

#define REP(x, n) for (int x = 0; x < (n); ++x)
#define VAR(v, n) __typeof(n) v=(n)
#define FOREACH(i, c) for ( VAR(i, (c).begin()); i != (c).end(); ++i)

#define MOD 1000000007

typedef unsigned long long ULL;
typedef unsigned int UI;

ULL fact[800001];
ULL ifact[800001];


ULL modExp(ULL n, ULL p)
{
	if(p==0)
	{
		return 1;
	}
	else
	{
		ULL res = modExp(n, p/2);
		res = (res*res)%MOD;
		if( p&0x01 )
		{
			res = (res*n)%MOD;
		}
		return res;
	}
}


ULL C(int x, int y)
{
	return (fact[x]*((ifact[x-y]*ifact[y])%MOD))%MOD;
}

ULL ways(int N, int M)
{
	return C(N+M, M);
}


int main()
{
	fact[0] = 1;
	ifact[0] = 1;
	for(int i=1; i<800001; i++) 
	{ 
		fact[i] = (i*fact[i-1])%MOD;
		ifact[i] = modExp(fact[i], MOD-2);
	}
	int T;
	cin>>T;
	while( T-- )
	{
		int N, M, A, B;
		cin>>N>>M>>A>>B;
		ULL res = 0;
		for(int p=0; p <= N-A; p++)
		{
			res += (ways(p, B-1)*ways(N-p, M-B))%MOD;
			if(res >= MOD)
				res = res - MOD;
		}
		cout<<res<<endl;
	}
    return 0;
}
