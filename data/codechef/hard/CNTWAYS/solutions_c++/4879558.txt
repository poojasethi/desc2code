#include <stdio.h>
#define ll long long int
#define mod 1000000007
ll fact[800005];
ll rev[800005];
long long int mypow(ll d, long long t)
{
	long long base = d;
	long long result = 1;	
   	 while (t > 0)
	    {
		if (t & 0x01)
        	{
           		result =(result*base)%mod;
        	}
        	t = t >> 1;
        	if (t > 0)
        	{
            		base = (base*base)%mod;
        	}
	}            
 
	return result;
}
void calc()
{
	fact[0]=1;
	int i;
	for(i=1;i<800005;i++)
	{
		fact[i]=(fact[i-1]*i)%mod;
	}
	rev[0]=1;
	rev[1]=1;
	for(i=2;i<800005;i++)
	{
		rev[i]=mypow(fact[i],mod-2);
	}
	
}

ll nCr(ll n,ll r)
{
	ll temp=(((fact[n]*rev[r])%mod)*rev[n-r])%mod;
	return temp;
}
int main()
{
	ll ans,N,M,A,B,prev,i,temp,temp1,t;
	scanf("%lld",&t);
	calc();
	while(t--)
	{
		scanf("%lld %lld %lld %lld",&N,&M,&A,&B);
		prev=0;
		ans=0;
		for(i=B;i<=M;i++)
		{
		   temp=(nCr((N-A)+i,N-A)-prev+mod)%mod;
		   temp1=(temp*nCr(M-i+A,A))%mod;
		   ans=(ans+temp1)%mod;
		   prev=nCr((N-A)+i,N-A);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
