#include <cstdio>
#include <algorithm>
#include <iostream>

using namespace std;
const long long MOD = (1e9 + 7);
const int MAX = 800005;
long long fact[MAX], ifact[MAX];

long long mpow(long long a, long long b)
{
    long long ret = 1;
    while(b)
    {
        if(b & 1) ret = (ret * a) % MOD;
        b >>= 1;
        a = (a * a) % MOD;
    }
    return ret;
}

int main()
{
    //freopen("input.txt", "r", stdin);
    int t;
    long long n, m, a, b, x, i, ans, y;
    fact[0] = 1;
    ifact[0] = 1;
    for(i = 1;i < MAX;++i)
    {
        fact[i] = (fact[i-1] * i) % MOD;
        ifact[i] = mpow(fact[i], MOD - 2);
    }
    for(scanf("%d", &t);t--;)
    {
        scanf("%lld %lld %lld %lld", &n, &m, &a, &b);
        ans = 0;
        for(i = 0;i <= (n-a);++i)
        {
            x = (fact[i+b-1]*ifact[i]) % MOD;
            x = (x * ifact[b-1]) % MOD;
            y = (fact[n - i + m - b]*ifact[n - i]) % MOD;
            y = (y * ifact[m - b]) % MOD;
            //cout << x << ' ' <<  y << endl;
            ans = (ans + (x * y) % MOD) % MOD;
        }
        if(ans < 0) ans += MOD;
        printf("%lld\n", ans);
    }
    return 0;
}
