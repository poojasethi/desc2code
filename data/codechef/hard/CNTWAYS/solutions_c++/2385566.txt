#include<iostream>
#include<cstdio>
#define mod 1000000007
#define ll long long
int fact[800001];
int ifact[800001];
ll power(int a,int x)
{
    ll b;
    if(x==0)
	return 1;
	else
	{
		b=power(a,x/2);
		if(x%2==0)
		return (b*b)%mod;
		else
		return (((b*b)%mod)*(ll)a)%mod;		
	}
}
int comb(int n,int r)
{
    return (((ll)fact[n]*ifact[r])%mod*(ll)ifact[n-r])%mod;
}
int compute(int a,int b)
{
    return comb(a+b,a);
}
int scan()
{
int t=0;
char c;
c=getchar_unlocked();
while(c<'0' || c>'9')
c=getchar_unlocked();
while(c>='0' && c<='9')
{
t=(t<<3)+(t<<1)+c-'0';
c=getchar_unlocked();
}
return(t);
}
int main(void)
{
	int i,j,k,l,r,n,m,a,b;
    ll p,q,s;
	fact[0]=1;
    ifact[0]=1;
	for(i=1;i<=800000;i++)
	{
		fact[i]=((ll)fact[i-1]*i)%mod;
		ifact[i]=power(fact[i],mod-2)%mod;
	}
    /*for(i=0;i<=100;i++)
    printf("%lld ",ifact[i]);*/
	r=scan();
	while(r--)
	{
        p=0;
		n=scan();
        m=scan();
        a=scan();
        b=scan();
		p=compute(n,m-b);
        for(i=1;i<=n-a;i++)
        {
            q=compute(n-i,m-b);
            s=compute(i,b-1);
            p=(p+(q*s)%mod)%mod;
        }
        printf("%lld\n",p);
	}
	return 0;
}
