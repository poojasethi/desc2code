#include<bits/stdc++.h>

using namespace std;

long long int mod=1000000007;
long long int fact[800010];
    long long int ifact[800010];
long long int power(long long int m,long long int n )
{
long long int temp=m ;
long long int ans=1;
while(n>0)
{
if( n%2 ==0 )
{
    temp= (temp * temp ) % mod ; n/=2 ;
}

else
{
    ans=( ans * temp ) %mod ; n-=1;
}
}
return ans;
}

int main()
{

    long long int n,m,a,b,t,i,ans1,ans2,ans,temp,total,z,dif,tot;
    scanf("%lld",&t);

    fact[0]=1;
    ifact[0]=1;
    for(i=1;i<=800005;i++)
    {
        fact[i]=(fact[i-1]*i)%mod;
        ifact[i]=power(fact[i],mod-2);
    }
    while(t--)
    {
        scanf("%lld%lld%lld%lld",&n,&m,&a,&b);
        ans=0;
        for(i=0;i<=n-a;i++)
        {
            temp=b+i-1;
            ans1=(fact[temp]*ifact[b-1])%mod;
            ans1=(ans1*ifact[temp-b+1])%mod;
            temp=(n-i)+(m-b);
            ans2=(fact[temp]*ifact[n-i])%mod;
            ans2=(ans2*ifact[temp-n+i])%mod;
            ans1=(ans1*ans2)%mod;
            ans=(ans+ans1)%mod;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
