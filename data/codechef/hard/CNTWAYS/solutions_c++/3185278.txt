#include<iostream>
#include<cstdio>
using namespace std;
 
long long fact[800001], finv[800001];
 
long long nCk(long long n, long long k)
{
	long long ans;
	ans = (((fact[n]*finv[k]) % 1000000007)*finv[n-k]) % 1000000007;
	return ans;
}
 
int main()
{
	int i;
	fact[0] = 1;
	for(i = 1; i <= 800000; i++)
		fact[i] = (fact[i-1]*i) % 1000000007;
	finv[800000] = 984388797;
	for(i = 800000-1; i > 0; i--)
		finv[i] = (finv[i+1]*(i+1)) % 1000000007;
	finv[0] = 1;
	int R;
    scanf("%d", &R);
    for (int r = 0; r < R; r++)
    {
        unsigned long long N, M, A, B;
        scanf("%llu %llu %llu %llu", &N, &M, &A, &B);
		unsigned long long answer = 0;
		for(i = 0; i <= N-A; i++)
		{
			answer += nCk(B+i-1, i)*nCk(N-i+M-B, M-B);
			answer = answer % 1000000007;
		}
		printf("%llu\n", answer);
	}
}
