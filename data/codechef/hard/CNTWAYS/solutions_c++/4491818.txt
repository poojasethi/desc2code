#include<stdio.h>
#include<cstring>
#define NUM 1000000007
long long int fact[800000],ifact[800000],inv[800000];
inline void inverse()
{
	inv[1]=1;
	for(int i=2;i<800000;i++)
	{
		inv[i]=NUM-((NUM/i*inv[NUM%i])%NUM);
	}
		
}
inline long long int ways(long long int n,long long int m)
{
	return ((((fact[n+m]*ifact[n])%NUM)*ifact[m])%NUM);
}
main()
{
	int tc,i;
	scanf("%d",&tc);
		fact[0]=ifact[0]=1;
		inverse();
		for(i=1;i<800000;i++)
		{
			fact[i]=(i*fact[i-1])%NUM;
			ifact[i]=(inv[i]*ifact[i-1])%NUM;//if(i<20)printf("%lld  ",ifact[i]);
		}
	while(tc--)
	{
		//memset(fact,1,sizeof(fact));
		//memset(ifact,1,sizeof(ifact));
		//memset(inv,1,sizeof(inv));
		int a,b,m,n;
		scanf("%d%d%d%d",&n,&m,&a,&b);
		long long int ans=0;
		for(i=0;i<=n-a;i++)
		{
			ans=(ans+(ways(i,b-1)*ways(n-i,m-b))%NUM)%NUM;
		}
		printf("%lld\n",ans);
	}
}
