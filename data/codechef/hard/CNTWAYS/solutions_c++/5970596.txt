#include <bits/stdc++.h>
typedef long long int lli;
using namespace std;
#define mod 1000000007ll
lli fact[800001];
lli ifact[8000001];
lli expo(lli a, lli b, lli m)
{
	lli result = 1;
	lli p = a;
	while(b)
	{
		if(b%2)
		{
			result=(result*p)%m;
		}
		p=(p*p)%m;
		b/=2;
	}
	return result;
}
 
lli combine(lli a, lli b)
{
	return (((fact[a] * ifact[b]) % mod) * ifact[a-b]) % mod;
}
 
lli modinv(lli a, lli p)
{
	return expo(a,p-2,p);
}
lli ways(lli a, lli b)
{
	return combine(a+b,b);
	//return (((fact[m+n]%mod * ifact[m]%mod)%mod * (ifact[n]%mod))%mod);
}
int main()
{
	fact[0]=fact[1] = 1;
	ifact[0] = ifact[1]=1;
	//lli mod = 1000000007ll;
	for(lli i =2;i<=800000;i++)
	{
		fact[i] = ((fact[i-1]%mod) * (i%mod))%mod;
		ifact[i] = modinv(fact[i],mod);
	}	
	int r;
	//printf("%lld\n",ways(4,2,1000000007));
	cin>>r;
	while(r--)
	{
		lli n,m,a,b;
		scanf("%lld %lld %lld %lld",&n,&m,&a,&b);
		lli res=0;
			for(lli i=0;i<=n-a;i++)
			{
				//cout<<ways(i,b-1,mod)<<endl;
				//cout<<ways(b,m,mod)<<endl;
				res+=ways(i,b-1)*ways(n-i,m-b);
				res%=mod;
			}
		printf("%lld\n",res);
	}
	return 0;
}
