#include<iostream>
using namespace std;

#define MOD 1000000007

typedef long long ll;

ll fact[800005];
ll invert[800005];
ll fact_invert[800005];

ll pow(ll a, ll b){
	if(b==0){
		return 1;
	}
	ll temp = pow(a,b/2);
	ll ans = temp * temp;
	ans = ans % MOD;
	if(b&1){
		ans = ans * a;
		ans = ans % MOD;
	}
	return ans;
}

void preprocess(){
	fact[0] = 1;
	invert[1] = 1;
	fact_invert[0] = 1;
	for(int i=2;i<=800000;i++){
		invert[i] = (MOD - (MOD/i) * invert[MOD%i] % MOD) % MOD;
	}
	
	for(ll i=1; i<=800000; i++){
		fact[i] = i*fact[i-1];
		fact[i] = fact[i] % MOD;
		fact_invert[i] = invert[i] * fact_invert[i-1];
		fact_invert[i] = fact_invert[i] % MOD;
	}
}

ll ncr(ll n, ll r){

	if(r>n){
		return 0;
	}
	ll temp = fact[n];

	temp = temp * fact_invert[r];
	temp = temp % MOD;

	temp = temp * fact_invert[n-r];
	temp = temp % MOD;

	return temp;	
}


int main(){

	int t;
	preprocess();
	cin>>t;
	while(t--){
		ll n,m,a,b;
		cin>>n>>m>>a>>b;
		ll ans = 0;
		for(ll i=b;i<=m;i++){
			ll temp = ncr(n-a+i , n-a);
			temp = temp * ncr(m-i+a-1 ,a-1);

			temp = temp % MOD;
			ans = ans + temp;
			ans = ans % MOD;
		}	
		cout<<ans<<endl;
	}
}
