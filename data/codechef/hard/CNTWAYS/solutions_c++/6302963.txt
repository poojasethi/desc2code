LL fact[800001],ifact[800001],inv[800001];
LL mod(LL m,LL c)
{
    if(m>=c||m<0)
    return (m%c+c)%c;
    return m;
}

LL power(LL m,LL n)
{
    LL temp=m,ans=1;
    while(n>0)
    {
        if(n%2==0){temp=mod(temp*temp,MAX);n/=2;}
        else
            {ans=mod(ans*temp,MAX);n-=1;}
    }
    return mod(ans,MAX);
}

void calfact()
{
    LL i,j,k;
    fact[0]=1;
    ifact[0]=1;
    inv[1]=1;
    loop1(i,800000)
    {
        fact[i]=i*fact[i-1];
        fact[i]=mod(fact[i],MAX);
        if(i!=1)
        inv[i] =MAX-((MAX/i)*inv[MAX%i]%MAX);
        ifact[i]=mod(ifact[i-1]*inv[i],MAX);
    }
}

LL ways(LL n,LL m)
{
    return mod(fact[n+m]*mod(ifact[n]*ifact[m],MAX),MAX);
}

int main()
{
    std::ios::sync_with_stdio(false);
    LL i,j,k,l,m,n,t,a,b;
    calfact();
    cin>>t;
    while(t--)
    {
        cin>>n>>m>>a>>b;
        LL ans=0;
        loop3(i,0,n-a)
        {
            ans+=(ways(b-1,i))*(ways(n-i,m-b));
            ans=mod(ans,MAX);
        }
        pvn(ans);
    }
    return 0;
}
