using namespace std;

int debug=1;
const int N=1e6+3;

/* MAIN PROGRAM STARTS HERE */

static LL fac[N];
static LL ifac[N];

LL fpow(LL a,LL n)
	{
	if (!n) return 1;
	LL b=fpow(a,n/2);
	b=(b*b)%MOD;
	if (n&1)
		return (a*b)%MOD;
	else return b%MOD;
	}

LL inv_mod(LL a)
	{
	return (fpow(a,MOD-2));
	}

void pre()
	{
	fac[0]=1;
	for (LL i=1; i<N; i++)
		fac[i]=(fac[i-1]*i)%MOD;
	ifac[N-1]=inv_mod(fac[N-1]);
	for (LL i=N-2; i>-1; i--)
		ifac[i]=(ifac[i+1]*(i+1))%MOD;			

	}
	
LL calc(LL n,LL m)
	{
	LL ans=fac[n]%MOD;
	ans=(ans*ifac[n-m])%MOD;
	ans=(ans*ifac[m])%MOD;
	//trace1(ans);
	return ans;
	}

			x=calc(i+b,i);
			y=calc(n-i+m-b,m-b);
			ans=(ans+((x*y)%MOD))%MOD;
			if (i) ans=(ans-(prevx*y)%MOD);

			ans%=MOD;
				prevx=x;
		
			}

		}
