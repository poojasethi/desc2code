using namespace std;
 
const int Mod=1000000007;
 
int Fac[800001],Rev[800001];
 
int Pow(int A,int B)
{
  if(!B)
    return 1;
  int T=Pow((long long)A*A%Mod,B/2);
  if(B&1)
    T=(long long)T*A%Mod;
  return T;
}
 
int FUNCCC(int M,int N)
{
  int Ans=Fac[M+N];
  Ans=(long long)Ans*Rev[M]%Mod;
  Ans=(long long)Ans*Rev[N]%Mod;
  return Ans;
}
