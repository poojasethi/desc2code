#include<iostream>
#include<vector>

using namespace std;

#define MOD 1000000007
#define MAX 800008

typedef vector<long long> vll;
vll fact(MAX),ifact(MAX),invert(MAX);



long long pow(long long base,int exp)
{
    if(exp==0)
        return 1;

    long long ans=pow(base,exp/2);

    if(exp%2==0)
        return (ans*ans)%MOD;
    else
        return (((ans*ans)%MOD)*base)%MOD;
}



long long ways(long long n,long long r){
    n=n+r;
    return (((fact[n]*ifact[r])%MOD)*ifact[n-r])%MOD;

}

void init()
    {
        fact[0]=ifact[0]=1;
        invert[1] = 1;
        for(int i=2;i<MAX;i++){
            invert[i] = (MOD - ((MOD/i) * invert[MOD%i] % MOD) );
        }
        for(long long i=1; i<MAX; i++){
            fact[i] = (i*fact[i-1])% MOD;
            ifact[i] = (invert[i] * ifact[i-1])% MOD;
        }
    }

int main()
{
    init();
    int r,n,m,a,b;
    cin>>r;
    for(int i=0;i<r;i++)
    {
        long long ans=0;
        cin>>n>>m>>a>>b;

        for(int i=0;i<=n-a;i++) {
            if(i==n-a)
                ans += (ways(i,b)*ways(n-i,m-b)) % MOD ;
            else
                ans += (ways(i,b)*ways(n-i,m-b-1)) % MOD ;
                ans %= MOD ;
            }
        cout<<ans<<endl;
    }

return 0;
}
