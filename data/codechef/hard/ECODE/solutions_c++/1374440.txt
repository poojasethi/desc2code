const int maxn=13;
int n;
ll a,mod,g;
ll pw[maxn];
ll pw10[maxn];

ll mul(ll x, ll y){
    ll res=0;
    while(y){
        res+=x*(y&1023);
        y>>=10;
        x=(x<<10)%mod;
        res%=mod;
    }
    return res;
}


int main(){
    cin>>n>>a>>mod>>g;
    pw[0]=1;
    pw10[0]=1;
    REP(i,n) pw[i+1]=mul(a,pw[i]);
    REP(i,n) pw10[i+1]=10*pw10[i];

    int na=pw10[n/2], nb=pw10[(n+1)/2];
    vector<pair<ll,int> > tb;
    REP(i,nb){
        int u=i;
        ll x=0;
        REP(j,(n+1)/2){ x=(x+u%10*pw[j])%mod; u/=10; }
        tb.pb(MP(x,i));
    }
    sort(ALL(tb));


    ll res=-1;
    REP(i,na){
        int u=i;
        ll x=0;
        REP(j,n/2){ x=(x+u%10*pw[(n+1)/2+j])%mod; u/=10; }
        x=(g+mod-x)%mod;
        vector<pair<ll,int> >::iterator it=lower_bound(ALL(tb),MP(x,0));
        if (it!=tb.end() && it->ST==x){res=1ll*mul(i,pw10[(n+1)/2])+it->ND; break;}
    }
    vi s(n,0);
    REP(i,n) s[n-1-i]=res%10, res/=10;
    REP(i,n) printf("%d",s[i]); puts("");
    return 0;
}
