double x[10][2],v[10][2];
int n;
double b[2],u;

double func(int arr[]) {
	double t=0.0,tt=0.0;
	double curx=b[0],cury=b[1];
	double y[10][2];
	for (int i=0;i<n;i++) {
		y[arr[i]][0]=x[arr[i]][0];
		y[arr[i]][1]=x[arr[i]][1];
	}
	for (int i=0;i<n;i++) {
		double dis=sqrt((curx-y[arr[i]][0])*(curx-y[arr[i]][0]) + (cury-y[arr[i]][1])*(cury-y[arr[i]][1]));
		double cph=(y[arr[i]][0]-curx)/dis;
		double sph=(y[arr[i]][1]-cury)/dis;
		double a=u*u-v[arr[i]][0]*v[arr[i]][0]-v[arr[i]][1]*v[arr[i]][1];
		double bb=-2*dis*(v[arr[i]][0]*cph+v[arr[i]][1]*sph);
		double cc=-dis*dis;
		t=(-bb+sqrt(bb*bb-4*a*cc))/(2*a);
		t+=3600;
		tt+=t;
		curx=y[arr[i]][0]+v[arr[i]][0]*t;
		cury=y[arr[i]][1]+v[arr[i]][1]*t;
		for(int j=i+1;j<n;j++){
			y[arr[j]][0]=y[arr[j]][0]+v[arr[j]][0]*t;
			y[arr[j]][1]=y[arr[j]][1]+v[arr[j]][1]*t;
		}
	}
	double dd=sqrt((curx-b[0])*(curx-b[0])+(cury-b[1])*(cury-b[1]));
	tt+=(dd)/u;
	return tt;
}

int main() {
	int tc=1;
	while(1){
		scanf("%d",&n);
		if(n==0)
			break;
		int a[10];
		double ans=1e15;
		for(int i=0;i<n;i++)
			a[i]=i;
		for(int i=0;i<n;i++){
			scanf("%lf%lf%lf%lf",&x[i][0],&x[i][1],&v[i][0],&v[i][1]);
			x[i][0]*=1000;
			x[i][1]*=1000;
			v[i][0]*=(5/18.0);
			v[i][1]*=(5/18.0);
		}
		scanf("%lf%lf%lf",&b[0],&b[1],&u);
		b[0]*=1000;
		b[1]*=1000;
		u*=(5/18.0);
		do{
			ans=min(ans,func(a));
		}while(next_permutation(a,a+n));
		long long int val=(long long int)ceil(ans);
		cout << tc << " ";
		tc++;
		cout<<val/3600<<" "<<(val%3600)/60<<" "<<val%60<<endl;
	}
	return 0;
}
