#define N 100010
int n,f[N],v[N],pre[N],next[N];
void update(int x)
{
	while(x<=n)f[x]++,x+=x&-x;
}
int query(int x)
{
	int ans=0;
	while(x>0)ans+=f[x],x-=x&-x;
	return ans;
}
int main()
{
	int T,i,j,k;
	scanf("%d",&T);
	while(T--)
	{
		string s;
		scanf("%d\n",&n);getline(cin,s);
		s='@'+s+'@';
		for(i=1;i<=n;i++)pre[i]=i-1,next[i]=i+1,f[i]=0;
		ll ans=0;int top=0;
		for(i=1;i<=n-1;i++)
		if(s[i]=='4'&&s[i+1]=='7')v[top++]=i;
		while(1)
		{
			if(top==0)break;
			for(i=top-1;i>=0;i--)
			ans+=v[i]-2*query(v[i]),update(v[i]);
			int top1=0;
			for(i=0;i<top;i++)
			{
				int a=pre[v[i]],b=next[next[v[i]]];
				next[a]=b,pre[b]=a;
				if(s[a]=='4'&&s[b]=='7')v[top1++]=a;
			}
			top=top1;
		}
		printf("%lld\n",ans);
	}
}
