#define MAXN 100011

int N;
char S[MAXN]; //the lucky string
int stp[MAXN];//the stp number when it is removed
int f[MAXN]; //to indicate whether removed or not

stack<int>A; //To find the stp numbers

int max(int a, int b){return a>b?a:b;}
//Fenwick Tree
LL tree[MAXN];
class FenwickTree
{
    public:
        int MaxVal;
        FenwickTree()
        {
            MaxVal = N+5;
        }
        //Read cumulative frequency at index idx
        LL read(int idx)
        {
            LL sum = 0;
            while(idx>0)
            {
                sum+= tree[idx];
                idx -= (idx & -idx);
            }
            return sum;
        }

        //Update frequency at index idx by value val
        void update(int idx, int val)
        {
            while(idx <= MaxVal)
            {
                tree[idx] += (LL)val;
                idx += (idx & -idx);
            }
        }

        //Read actual frequency at index idx
        LL readSingle(int idx)
        {
            return read(idx) - read(idx-1);
        }

        //Returns sum of indices [low, high]
        LL readRange(int low, int high)
        {
            return read(high) - read(low-1);
        }

};
FenwickTree ft;

void solve()
{
    scanf("%d", &N);
    scanf("%s", S);

    //Initialize
    for(int i=0; i<N+5; i++)
    {
        f[i] = 0;
        stp[i] = 0;
    }

    while(!A.empty())
    {
        A.pop();
    }

    //First pass to determine stp
    int j, k;
    for(int i=1; i<=N; i++)
    {
        if(S[i-1] == '4')
        {
            stp[i] = 1;
            A.push(i);
        }
        else //means its a 7
        {
            if(!A.empty())
            {

                j = A.top();
                A.pop();
                f[j] = 1;
                f[i] = 1;
                stp[i] = stp[j];
                if(!A.empty())
                {
                    k = A.top();
                    stp[k] = max(stp[k], 1+ stp[j]);
                    //cout<<i<<" "<<j<<" "<<k<<endl;
                }
            }
        }
    }
    /*for(int i=1;i<=N; i++)
    {
        cout<<f[i]<<" "<<stp[i]<<" "<<i<<endl;
    }*/

    //Second pass to find the sum
    for(int i=0; i<N+5; i++)
    {
        tree[i] = 0;
    }
    ft.MaxVal = N+5;
    LL ans = 0;
    LL diff;
    for(int i=1; i<=N; i++)
    {
        if(S[i-1] == '4' && f[i] == 1)
        {
            diff = ft.read(stp[i]-1);
            //cout<<diff<<" "<<i<<" "<<stp[i]-1<<endl;
            ans += (LL)i - 2*diff;

        }
        if(S[i-1] == '7' && f[i] == 1)
        {
            ft.update(stp[i], 1);
        }
    }
    //printf("%lld\n", ans);
    cout<<ans<<endl;
}

    scanf("%d", &t);
    while(t--)
    {
        solve();
