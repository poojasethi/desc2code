const int nmax = 100100;

char s[nmax];
int f[nmax];

int sum(int i) {
  int res = 0;
  for(; i >= 0; i = (i & (i + 1)) - 1) res += f[i];
  return res;
}

void upd(int i, int d) {
  for(; i < nmax; i |= i + 1) f[i] += d;
}

void solve() {
  int n;
  scanf("%d%s", &n, s);
  MEM(f, 0);
  stack<pair<int, int> > st;
  LL ans = 0;
  REP(i, n) {
    if (s[i] == '4') {
      st.push(MP(i, st.size() + 1));
    } else {
      if (!st.empty()) {
        int lev = st.top().second - st.size();
        int cur = st.top().second;
        st.pop();
        if (!st.empty()) st.top().second = max(st.top().second, cur);
        upd(lev + 1, 1);
        ans += i - sum(lev)*2;
      }
    }
  }
  cout << ans << endl;
}

  while(T--) solve();
