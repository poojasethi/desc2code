int stp[100005];
char a[100005];
LL tree[100005];
int mx;
int M[100005];

void update(int x)
{
    while (x<=mx)
    {
        tree[x] ++;

        x += x & -x;
    }
}

LL query(int x)
{
    LL sm = 0;

    while (x>0)
    {
        sm += tree[x];

        x -= x&-x;
    }

    return sm;
}
