#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<vector>
#include<algorithm>
#include<map>
#include<limits.h>
using namespace std;

int n,A[40],max1;
map<int,int> f;

int doit(int mask)
{
    int cmax=0,cmin=0;
    for(int i=0;i<n;i++)
            if(mask & (1<<i))cmax++;
            else cmin++;
            
    if(cmin==0)return 0;
    if(cmax>=cmin)return 1;
    
    if(f.find(mask)!=f.end())
                         return f[mask];
                                
    int res=INT_MAX;
    
    for(int i=0;i<n;i++)
    {
            int st=mask,en=-1;
            cmax=0,cmin=0;
            for(int j=i;j<n;j++)
            {
                    if(mask & (1<<j))cmax++;
                    else
                    {
                        st+=(1<<j);
                        cmin++;    
                    }        
                    
                    if(cmax>=cmin and st!=mask)
                             en=st;                    
            }
            if(en!=-1)res=min(res,doit(en)+1);

    }      
    return f[mask]=res;
}
int main()
{
    
          int t;
          scanf("%d",&t);
          
          while(t--)
          {
                    scanf("%d",&n);
                    f.clear();
                    for(int i=0;i<n;i++)scanf("%d",&A[i]);
                    max1=-1;
                    for(int i=0;i<n;i++)max1=max(max1,A[i]);
                    
                    int mask=0;
                    for(int i=0;i<n;i++)if(A[i]==max1)mask+=(1<<i);
                    
                    printf("%d\n",doit(mask));    
                              
          }
          
          return 0;
}
