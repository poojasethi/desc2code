#include<stdio.h>

int x12,t,n,a[40],max,maxi,maxj,tmp,maxv,tmp1,ans,cnto,cntz,l;

inline void inp(int &u)
{
scanf("%d",&u);
}

inline int f(int i,int j)
{
tmp1=1000000;
for(int x=j+1;x<n;x++)
{
if(a[x]==1)
{tmp1=x-j;break;}
}
for(int x=i-1;x>=0;x--)
{
if(a[x]==1)
{if((i-x)<tmp1){tmp1=i-x;}break;}
}
return tmp1;
}

inline void change(int i,int j)
{
for(int x=i;x<=j;x++)
{
a[x]=1;
}
}

inline bool ck()
{
for(int i=0;i<n;i++)
{
if(a[i]==-1)
return true;
}
return false;
}

int main()
{
inp(t);
for(int i=0;i<t;i++)
{
inp(n);
max=-1;
for(int i=0;i<n;i++)
{
inp(a[i]);
if(a[i]>max)
max=a[i];
}
for(int i=0;i<n;i++)
{
if(a[i]==max)
a[i]=1;
else
a[i]=-1;
}
ans=0;
while(ck())
{
maxv=1000000;
l=-1;
for(int i=0;i<n;i++)
{
cnto=0;
cntz=0;
if(a[i]==-1)
cnto++;
else
cntz++;
for(int j=i+1;j<n;j++)
{
if(a[j]==-1)
cnto++;
else
cntz++;
if(cnto<=cntz)
{
//printf("fgf %d %d\n",cnto,cntz);
tmp=f(i,j);
if(j-i>l || (j-i==l && tmp<maxv))
{maxv=tmp;maxi=i;maxj=j;l=j-i;}
}
}
}
//printf("%d %d %d\n",maxi,maxj,maxv);
ans++;
change(maxi,maxj);
}
printf("%d\n",ans);
}
return 0;
}
