#define ll long long int
#define ull unsigned long long int
#define maxN 1000006

#define MOD 1000000007
#define rep(i,N) for(int i=0;i<N;i++)

ll A[205],C[205],dp[205];

int main() {

cin.sync_with_stdio(0);
cin.tie(0);

int T;
cin>>T;

while(T--) {
int N,K;
cin>>N>>K;

memset(dp,0,sizeof dp);
memset(C,0,sizeof C);
for(int i=0;i<N;i++) {
	cin>>A[i];
	C[A[i]]++;
}

ll size=0;
dp[0]=1;
for(int i=200;i>=1;i--)
	if(C[i]>0) {
		ll val=1;
		for(ll j=size+1;j<size+C[i];j++)
			val=(val*j)%MOD;
		ll val1=(val*C[i])%MOD;
		ll val2=(val*size)%MOD;
		for(int curr=K;curr>=0;curr--) {
			dp[curr+1] = (dp[curr+1] + ((dp[curr] * val1) % MOD) ) % MOD; 
			dp[curr+1]%=MOD;
			dp[curr]*=val2;
			dp[curr]%=MOD;
	 	}
	//cout<<size<<" "<<dp[1]<<" "<<dp[2]<<" "<<dp[3]<<"\n";
	size+=C[i];
}

ll ans=0;
for(int i=1;i<=K;i++)
	ans=(ans+dp[i])%MOD;
cout<<ans<<"\n";
}

return 0;
}
