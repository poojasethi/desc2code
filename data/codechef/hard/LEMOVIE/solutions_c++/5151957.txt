#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>

using namespace std;

long long jc[500];
long long c[500][500];
long long f[500][500];
int a[500];
int T,n,k;
const int mod = 1e9 + 7;
int main()
{
	jc[0]=1;
	for (int i=1;i<=200;++i) jc[i]=jc[i-1]*i%mod;
	for (int i=0;i<=200;++i) c[i][0]=1;
	for (int i=1;i<=200;++i)
		for (int j=1;j<=i;++j)
			c[i][j]=(c[i-1][j-1]+c[i-1][j])%mod;
	scanf("%d",&T);
	while (T--)
	{
		scanf("%d%d",&n,&k);
		for (int i=1;i<=n;++i) scanf("%d",&a[i]);
		sort(a+1,a+n+1);
		a[0]=0;
		memset(f,0,sizeof(f));
		f[0][0]=1;
		for (int i=1;i<=n;++i)
			for (int j=0;j<i;++j)
				for (int k=0;k<=j;++k){
					if (a[i]>a[j]) 
					f[i][k+1]=(f[i][k+1]+c[n-j-1][i-j-1]*jc[i-j-1]%mod*f[j][k])%mod;
					else
					f[i][k]=(f[i][k]+c[n-j-1][i-j-1]*jc[i-j-1]%mod*f[j][k])%mod;
				}
		long long ans=0;
		for (int i=1;i<=k;++i) ans=(ans+f[n][i])%mod;
		printf("%lld\n",ans);
	}
	return 0;
}
