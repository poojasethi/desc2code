#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int hash[210]={0};
		long long dp[210]={0},size,inp,temp;
		dp[0]=1;
		int n,k;
		cin>>n>>k;
		for(int i=0;i<n;i++)
			{cin>>inp;
			hash[inp]++;
			}
			
		size=0;
		for(int i=200;i>=0;i--)
		{
		
		if(hash[i])
		{
			temp=1;
		for(int j=size+1;j<size+hash[i];j++)
			 temp=(temp*j)%MOD;
		for(int j=k;j>=0;j--)
			{
				dp[j+1]=(dp[j+1] +(((dp[j]*hash[i])%MOD)*temp)%MOD)%MOD;
				dp[j]=(((dp[j]*temp)%MOD)*size)%MOD;
			}
		}
		size=size+hash[i];
	}
	long long sum=0;
	for(int i=0;i<=k;i++)
	sum=(sum+dp[i])%MOD;
	cout<<sum<<endl;
	}
}
