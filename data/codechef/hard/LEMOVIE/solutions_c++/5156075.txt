#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
const int N = 205;
int n, m, k, a[N];
LL f[N];
const int mo = 1000000007;
int main() {
    int TT;
	scanf("%d",&TT);
	while (TT--) {
        int i, j, s = 0;
		LL x, y;
    	scanf("%d%d", &m, &k);
    	for (i = 1; i <= m; i++) {
	        scanf("%d", &a[i]);
	    }
	    sort(a + 1, a + m + 1, greater<int>());
	    n = 0;
	    for (i = 1; i <= m; i++) {
	        j = i;
    	    while (j <= m && a[i] == a[j]) {
        	    j++;
        	}
        	a[++n] = j - i, i = j - 1;
    	}
    	memset(f, 0, sizeof(f));
    	f[0] = 1;
    	for (i = 1; i <= n; i++) {
	        x = y = 1;
	        for (j = s; j <= s + a[i] - 1; j++) {
	            x *= j + 1, y *= j, x %= mo, y %= mo;
	        }
	        x = (x - y + mo) % mo;
	        for (j = k; j >= 0; j--) {
	            f[j] = ((LL)f[j - 1] * x + (LL)f[j] * y) % mo;
	        }
	        s += a[i];
	    }
	    int ans = 0;
	    for (i = 0; i <= k; i++) {
	        ans += f[i], ans %= mo;
	    }
	    printf("%d\n", ans);
	}
	return 0;
}
