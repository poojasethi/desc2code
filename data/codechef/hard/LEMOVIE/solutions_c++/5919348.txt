#include<bits/stdc++.h>
using namespace std;
#define getcx getchar
#define maxn 205
#define F(i,p,n) for(i=p;i<n;i++)
#define I(i,p,q) for(i=p;i>=q;i--)
#define S(x) scanf("%lld",&x)
#define P(x) printf("%lld\n",x)
#define Ps(x) printf("%d ",x)
#define LL long long int
#define pii pair<int,int>
const LL mod=1e9+7;
LL dp[maxn],freq[maxn];

int main()
{
    LL i,j,k,l,m,n,t;
    S(t);
    while(t--)
    {
        vector<LL> v;
        memset(dp,0,sizeof(dp));
        memset(freq,0,sizeof(freq));
        dp[0]=1;
        m=0;
        S(n);
        S(k);
        LL num;
        F(i,0,n)
        {
            S(num);
            freq[num]++;
            if(freq[num]==1)
                v.push_back(num);
        }
        sort(v.begin(),v.end());
        I(i,(LL)v.size()-1,0)
        {
            LL pro=1;
            F(j,1,freq[v[i]])
                pro=(pro*(m+j))%mod;
            I(j,k,0)
            {
                dp[j+1]=(dp[j+1]+((pro*dp[j])%mod)*freq[v[i]])%mod;
                dp[j]=(((pro*dp[j])%mod)*m)%mod;
            }
            m+=freq[v[i]];

        }
        LL ans=0;
        F(i,1,k+1)
        ans=(ans+dp[i])%mod;
        P(ans);
    }
    return 0;
}
