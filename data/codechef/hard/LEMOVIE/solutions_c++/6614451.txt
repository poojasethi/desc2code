#include<bits/stdc++.h>

using namespace std;


#define mod 1000000007


typedef long long int lol;


lol n,k;
lol a[210];

lol dp[210][210][2];


lol foo(lol curr,lol val,int prev_pos);


bool cmp(const lol &a,const lol &b)
{
	return b<a;
}


int main()
{
    int test_case;
    cin>>test_case;
    while(test_case--)
    {
        memset(dp,-1,sizeof(dp));
        cin>>n>>k;
        for(lol i=0;i<n;i++) cin>>a[i];
        sort(a,a+n,cmp);
        cout<<foo(0,0,0)<<"\n";
    }
    return 0;
}


lol foo(lol curr,lol val,int prev_pos)
{
    if(curr==n && val<=k) return 1;
    if(curr==n) return 0;
    if(dp[curr][val][prev_pos]!=-1) return dp[curr][val][prev_pos];
    if(curr==0)
    {
        return dp[curr][val][prev_pos]=foo(curr+1,val+1,1)%mod;
    }
    if(a[curr]==a[curr-1])
    {
        if(prev_pos==1)
        return dp[curr][val][prev_pos]=  (((curr+1)%mod)*((foo(curr+1,val,1)%mod)))%mod;
        else return dp[curr][val][prev_pos]= (foo(curr+1,val+1,1)%mod+(((curr)%mod)*((foo(curr+1,val,0)%mod)))%mod)%mod;
    }
    return dp[curr][val][prev_pos]= (foo(curr+1,val+1,1)%mod+(((curr)%mod)*((foo(curr+1,val,0)%mod)))%mod)%mod;
}
