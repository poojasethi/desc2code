#include<iostream>

#include<cstdio>

#include<cmath>

#define FOR(i,a,b) for(i = (a);i < (b);i++)
#define MOD 1000000007

using namespace std;
long long fact[210],f[210],infact[210];
int w[205];


long long pow(long long a1,int b)
{
	long long  d;

		if(b == 0)
			return 1;
		if(b == 1)
			return a1;
		
		d = pow(a1,b/2);				
		if(b&1)
			return (((d*d)%MOD)*a1)%MOD;
		else
			return (d*d)%MOD; 
}


int main()
{
	int t,cnt,i,y,x,n,k,c,j,num;
	long long ans,prev_t,v,prev_n,sum;
	int a[202];

	fact[0] = 1;
	infact[0] = 1;
	FOR(i,1,205)
	{
		fact[i] = (fact[i-1]*i)%MOD;
		infact[i] = pow(fact[i],MOD-2)%MOD;
	}
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d %d",&n,&k);
		FOR(i,0,n)
			scanf("%d",&a[i]);

		sort(a,a+n);
		//FOR(i,0,n)
		//	printf("%d ",a[i]);

		memset(f,0,sizeof(f));
		memset(w,0,sizeof(w));
		x = 0;
		v = -1;
		FOR(i,0,n)
		{
			
			if(x != a[i])
			{
				v++;
				w[v]++;
				x = a[i];
			}
			else
			{
				w[v]++;
			}
		}
			
		y = 1;
		f[0] = 0; 
		f[1] = fact[w[v]];
		num = 0;
		//FOR(i,0,v+1)
		//	printf("%d ",w[i]);
	
		for(i = v-1;i >= 0;i--)
		{
		//	printf("nishant\n");
			y++;
			num += w[i+1];
			prev_n = f[0];
		//	printf("%d ",num);
			for(j = 1;(j <= y)&&(j <= k);j++)
			{				
				prev_t = prev_n;
				prev_n = f[j];	
				f[j] = ((((prev_t*fact[w[i]+num-1])%MOD)*((infact[num]*w[i])%MOD))%MOD + (((f[j]*fact[w[i]+num-1])%MOD)*infact[num-1])%MOD )%MOD;
		//		printf("%lld ",f[j]);				
				
			}
		}

		sum = 0;		
		FOR(i,1,k+1)
			sum = (sum + f[i])%MOD;
		printf("%lld\n",sum);
	}
	return 0;
}
