#include <bits/stdc++.h>
#define sz 205
#define mod 1000000007
using namespace std;
typedef long long ll;

int n, k;
vector<int> arr;
int next[sz];
ll dp[sz][sz];
ll adjust[sz][sz];

void init() {
	for(int i=1;i<sz;i++) {
		for(int j=0;j<sz;j++) {
			if(j==0) {
				adjust[i][j] = 1;
			} else {
				adjust[i][j] = adjust[i][j-1]*(i+j-1);
				adjust[i][j] %= mod;
			}
		}
	}
}

ll solve() {
	ll ret = 0;
	memset(dp, 0, sizeof dp);
	next[n] = n+1;
	for(int i=n-1;i>0;i--) {
		if(arr[i] == arr[i+1]) {
			next[i] = next[i+1];
		} else {
			next[i] = i+1;
		}
	}
	dp[n+1][0] = 1;
	dp[n][1] = 1;
	for(int i=n-1; i>=1; i--) {
		for(int j=1;j<=n;j++) {
			int r = n-i;
			dp[i][j] = dp[i+1][j] * r;
			dp[i][j] %= mod;
			if(arr[i] != arr[i+1]) {
				dp[i][j] += dp[i+1][j-1];
				dp[i][j] %= mod;
			} else {
				int ii = next[i];
				int temp = n-ii+2;
				int temp1 = ii-i-1;
				dp[i][j] += dp[ii][j-1]*adjust[temp][temp1];
				dp[i][j] %= mod;
			}
		}
	}
	for(int i=1;i<=k;i++) {
		ret += dp[1][i];
		ret %= mod;
	}
	return ret;
}

int main() {
	int t;
	cin>>t;
	init();
	while(t--) {
		arr.clear();
		scanf("%d%d", &n, &k);
		arr.push_back(0);
		for(int i=0;i<n;i++) {
			int a;
			scanf("%d", &a);
			arr.push_back(a);
		}
		sort(arr.begin(), arr.end());
		printf("%lld\n", solve());
	}
	return 0;
}
