using namespace std;
typedef long long LL;

const int N = 205;
int n , m , K , a[N];
LL f[N];
const int Q = 1e9 + 7;
void work() {
    int i , j , s = 0; LL x , y;
    scanf("%d%d",&m,&K);
    for (i = 1 ; i <= m ; ++ i)
        scanf("%d",&a[i]);
    sort(a + 1 , a + m + 1 , greater<int>());
    n = 0;
    for (i = 1 ; i <= m ; ++ i) {
        j = i;
        while (j <= m && a[i] == a[j])
            ++ j;
        a[++ n] = j - i , i = j - 1;
    }
    memset(f , 0 , sizeof(f));
    f[0] = 1;
    for (i = 1 ; i <= n ; ++ i) {
        x = y = 1;
        for (j = s ; j <= s + a[i] - 1 ; ++ j)
            x *= j + 1 , y *= j , x %= Q , y %= Q;
        x = (x - y + Q) % Q;
        for (j = K ; j >= 0 ; -- j)
            f[j] = ((LL)f[j - 1] * x + (LL)f[j] * y) % Q;
        s += a[i];
    }
    int ans = 0;
    for (i = 0 ; i <= K ; ++ i)
        ans += f[i] , ans %= Q;
    printf("%d\n" , ans);
}

int main() {
    int _; scanf("%d",&_); while (_ --)
        work();
    return 0;
}
