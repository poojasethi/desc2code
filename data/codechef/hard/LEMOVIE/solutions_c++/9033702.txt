#include<bits/stdc++.h>

typedef long long  ll;
#define mod 1000000007

using namespace std;
int N,K;
ll A[202];
ll dp[202][202][202][2];
ll calc(int l , int k , int so_far,int big)
{
    if(l==N)
    {
        if(k<=K)
            return big==0?1:0;
        else
            return 0;
    }
    else if(dp[l][k][so_far][big]!=-1)
        return dp[l][k][so_far][big];
    else
    {
        ll ans=0;
        if(A[l]<=so_far)
        {
            ans=(N-l)*calc(l+1,k,so_far,big);
        }
        else{
            ans=calc(l+1,k+1,A[l],0);
            ans%=mod;
            ans=ans+(N-l-1)*calc(l+1,k,so_far,big);
        }
        return dp[l][k][so_far][big]=ans%mod;
    }
}
 int main()
 {
     int t,n,k,i,j;
    ios_base::sync_with_stdio(false);
     ll ans=0;
     cin>>t;
     while(t>0){
        t--;
        cin>>n>>k;
        N=n;
        K=k;
        for(i=0;i<n;i++)
        {
            cin>>A[i];
        }
        memset(dp,-1,sizeof(dp));
        sort(A,A+n);
        ans=calc(0,0,0,0);
        cout<<ans<<endl;
     }
 }
