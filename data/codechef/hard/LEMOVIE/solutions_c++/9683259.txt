#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#define LL long long
using namespace std;
const LL pp=(1e9)+7;
int N,K,Test,x,ans;
int a[210];
LL f[2][210];
LL c[210][210],h[210];
int main(){
	c[0][0]=1;
	for (int i=1;i<=200;i++)
		for (int j=0;j<=i;j++){
			c[i][j]=c[i-1][j];
			if (j)c[i][j]=(c[i][j]+c[i-1][j-1])%pp;
		}
	h[0]=1;
	for (int i=1;i<=200;i++)h[i]=h[i-1]*i%pp;
	scanf("%d",&Test);
	while (Test--){
		scanf("%d%d",&N,&K);
		memset(a,0,sizeof(a));
		for (int i=1;i<=N;i++)scanf("%d",&x),a[x]++;
		memset(f,0,sizeof(f));
		int now=1,next=0;
		f[0][0]=1;
		int num=0;
		for (int i=200;i;i--)if (a[i]){
			swap(now,next);
			memset(f[next],0,sizeof(f[next]));
			for (int j=0;j<=K;j++)
				if (f[now][j]){
					int t1=c[num+a[i]-1][a[i]-1];
					int t2=c[num+a[i]-1][a[i]];
					if (j<K)f[next][j+1]=(f[next][j+1]+f[now][j]*t1%pp*h[a[i]]%pp)%pp;
					f[next][j]=(f[next][j]+f[now][j]*t2%pp*h[a[i]]%pp)%pp;
				}
			num+=a[i];
		}
		ans=0;
		for (int i=0;i<=K;i++)ans=(ans+f[next][i])%pp;
		printf("%d\n",ans);
	}
}
