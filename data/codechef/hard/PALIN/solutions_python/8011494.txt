def inc_one(num):
	for a in xrange(len(num) - 1, -1, -1):
		if int(num[a]) < 9:
			return num[:a] + str(int(num[a])+1) + "0" * (len(num) - a - 1)
	return "1" + "0"*len(num)

def next_pali(num):
	half = (len(num) + 1) / 2
	for b in xrange(half, len(num)):
		a = len(num) - b - 1
		if num[a] == num[b]:
			pass
		if int(num[a]) < int(num[b]):
			"4351539 -> next_pali(4352000) 7/4/3/3"
			"2133 -> next_pali(2200) 4/2/1/2"
			new_num = inc_one(num[:half]) + "0"*(len(num)-half)
			return next_pali(new_num)
		if int(num[a]) > int(num[b]):
			return num[:b] + num[0:a+1][::-1]
	return num
