def nextPalindrome(astr):
	n = len(astr)
	lefthalf = astr[:n/2]
	center = astr[n/2] if n%2 else ''
	righthalf = lefthalf[::-1]
	palindrome = lefthalf + center + righthalf
	
	if palindrome > astr:
		return palindrome
	if center:
		if center != '9':
			center = str(int(center)+1)
			palindrome = lefthalf + center + righthalf
			return palindrome
		else:
			center = '0'
	if lefthalf == '9'*(n/2):
		palindrome = '1' + '0'*(n-1)+ '1'
		return palindrome
		
	t = list(lefthalf)
	p = n/2 - 1
	while t[p] == '9':
		t[p] = '0'
		p -= 1
		
	t[p] = str(int(t[p])+1)
	lefthalf = ''.join(t)
	righthalf = lefthalf[::-1]
	palindrome = lefthalf + center + righthalf
	return palindrome
	
t = input()
for _ in xrange(t):
	string = raw_input()
	print nextPalindrome(string)
