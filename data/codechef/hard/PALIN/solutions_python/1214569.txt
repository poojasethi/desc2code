#!/usr/bin/env python

import array

def comp(val, inp, lng, h):    
    if len(val) != len(inp):
        return (len(val) < len(inp))    
    for i in range(0, lng):
        if val[i] != inp[i]:
            return (val[i] < inp[i])
    return False
    
def getNextPal(val, lng, h):
   
    a = array.array('b', val)
    z  = ord('0')
    x  = ord('9') + 1
    
    for i in range(lng-h-1, -1, -1):       
       d = a[i]+1
       if d == x:
           a[i] = z               
       else:
           a[i] = d
       
           a[lng-h:] = a[h-1::-1]               
           out = a.tostring()           
           return out         
    if a[0] == z:
        a.insert(0, ord('1'))
        lng = lng+1
        h = lng/2
        a[lng-h:] = a[h-1::-1]               
        return a.tostring()   
        
def palin():
    num = input()
    
    while(num):                
        val = raw_input()       
        lng = len(val)
        h = lng/2
        
        if val in str(range(0,9)):
            print str(int(val)+1)
        elif val == '9':
            print '11'
        else:    
            res = val[:h]+val[(lng-1)/2::-1]
                    
            if comp(val, res, lng, h):
                print res
            else:            
                print getNextPal(val, lng, h)
        
        num = num-1    
palin()
