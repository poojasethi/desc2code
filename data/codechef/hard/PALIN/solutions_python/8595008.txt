import fileinput

def main():
  numlines = None
  for line in fileinput.input():
    if numlines is None:
      numlines = int(line)
    else:
      print process(line)
      numlines -= 1
    if numlines <= 0:
      break

def process(line):
  def increment(numstr, pos):
    if numstr[pos] == '9': numstr[pos] = '0'
    else: numstr[pos] = chr(ord(numstr[pos])+1)

  def str_increment(numstr, pos=None):
    if pos == None:
      pos = len(numstr) - 1
    while pos >= 0:
      increment(numstr, pos)
      if numstr[pos] > '0':
        return pos
      pos -= 1
    return -1

  numstr = list(line[:-1])
  if str_increment(numstr) == -1:
    numstr = ['1'] + numstr
  size = len(numstr)
  end = size/2
  left = 0
  while True:
    right = size-left-1
    if numstr[left] < numstr[right]:
      leftmost = str_increment(numstr, right - 1)
      numstr[right] = numstr[left]
      left = min(left, leftmost)
    else:
      numstr[right] = numstr[left]
      left += 1
    if left > end: break
  return ''.join(numstr)

if __name__ == "__main__": main()
