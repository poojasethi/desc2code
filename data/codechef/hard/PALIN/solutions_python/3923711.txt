#!/usr/bin/env python2
# -*- coding: utf-8 -*-

try:
    import psyco
    psyco.full()
except ImportError:
    pass


def main(K):
    """
    >>> main('808')
    '818'
    >>> main('2133')
    '2222'
    """
    L = len(K)

    # build the palindrome
    left = K[:L/2]
    center = K[L/2] if L % 2 else ''
    right = left[::-1]
    palin = left + center + right

    if palin > K:
        return palin

    if center != '':
        if center != '9':
            center = str(int(center) + 1)
            palin = left + center + right
            return palin
        else:
            center = '0'

    # increase the left part
    if left == '9' * (L/2):
        # easy case
        palin = '1' + '0' * (L - 1) + '1'
        return palin
    tab = list(left)
    p = L/2 - 1
    ## propagate the addition
    while tab[p] == '9':
        tab[p] = '0'
        p -= 1
    tab[p] = str(int(tab[p]) + 1)
    left = ''.join(tab)
    # rebuild the palindrome
    right = left[::-1]
    palin = left + center + right

    return palin


if __name__ == "__main__":
    t = int(raw_input()) ## nb of test cases

    for test in xrange(t):
        K = raw_input()
        print main(K)
