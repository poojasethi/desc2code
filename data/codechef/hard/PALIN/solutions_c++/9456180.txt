#include<iostream>
using namespace std;
int a[1000000];
//palimdrome for odd num
void odd(int n)
{
    int i=1;
    while(i<=n/2)
    {
        if (a[n/2-i]<a[n/2+i])
        {
            a[n/2]++;
            break;
        }
        else if (a[n/2-i]==a[n/2+i]) i++;
        else break;
    }
    i=n/2;
    while(i>=0)
    {
        if(a[i]==10)
        {
            a[i]=0;
            a[i-1]+=1;
        }
        i--;
    }
    i=1;
    while(i<=n/2)
    {
        a[n-i]=a[i-1];
        i++;
    }
}
void even(int n)
{
    int i=1;
    while(i<=n/2)
    {
        if (a[n/2-i]<a[n/2+i-1])
        {
            a[n/2-1]++;
            break;
        }
        else if (a[n/2-i]==a[n/2+i-1]) i++;
        else break;
    }
    i=n/2-1;
    while(i>=0)
    {
        if(a[i]==10)
        {
            a[i]=0;
            a[i-1]+=1;
        }
        i--;
    }
    i=1;
    while(i<=n/2)
    {
        a[n-i]=a[i-1];
        i++;
    }
}
void digit_inc(int n)
{
    while(n>0)
    {
        a[n]=a[n-1];
        n--;
    }
    a[0]=1;
}
int main()
{
    int t;
    char b[1000000];
    cin>>t;
    while(t--)
    {
        cin>>b;
        //size of array
        int n=0,i;
        while(b[n]!=0) n++;
        //conversion from char to int
        for(i=0;i<n;i++) a[i]=b[i]-48;
        //adding one
        a[n-1]=a[n-1]+1;
        //removing 10
        i=n;
        while(a[i-1]==10)
        {
            a[i-1]=0;
            a[i-2]+=1;
            i--;
        }
        if (a[0]==0)
        {
            digit_inc(n);
            n++;
        }
        if (n%2!=0) odd(n);
        else even(n);
        for (i=0;i<n;i++) cout<<a[i];
        cout<<endl;
    }
}
