#include<iostream>
#include<stdio.h>
#include<vector>
#include<string.h>
using namespace std;


void copyLeft(char* number, int i,int length)
{
	while(i >= 0)
	{
		number[length-1-i] = number[i];
		i--;
	}
}

int check9(char* number,int i)
{
	while(number[i] == '9')
	{
		number[i] = '0';
		i--;
	}
	return i;
}

int main()
{
	int testCases;
	scanf("%d",&testCases);
	while(testCases--)
	{
		bool palan = true;
		char number[1000005];
		cin >> number;
		int currIndex,length = strlen(number);
		
		int end;
		if(length % 2)
			end = length/2;
		else
			end = length/2 - 1;
		currIndex = length/2 - 1;
		for(int i=currIndex;i>=0;i--)
		{
			if(number[i] == number[length-1-i])
				continue;
			palan = false;
			if(number[i] > number[length-1-i])
				copyLeft(number,i,length);
			else
			{
				int changingIndex = end;
				changingIndex = check9(number,changingIndex);
				number[changingIndex]++;
				copyLeft(number,end,length);
			}
			break;
		}
		if(palan)
		{
			int i = end;
			i = check9(number,i);
			if(i >= 0)
				number[i]++;
			i = end;
			copyLeft(number,i,length);
			if(number[0] == '0')
			{
				number[length] = number[0] = '1';
				number[length + 1] ='\0';
			}
		}
		cout << number << endl;
	}
}
