#include <bits/stdc++.h>
#include <iostream>
using namespace std;

bool checkPalin(string s){
    for(int i=0, j=s.length() - 1;i<=j;i++, j--)
        if(s[i] != s[j])
            return false;
    return true;
}

bool check9(string s){
    for(int i=0;i<s.length();i++)
        if(s[i] != '9')
            return false;
    return true;
}

string for9(string s){
    string final = "1";
    for(int i=0;i<s.length() - 1;i++)
        final.append(1, '0');
    final.append(1, '1');
    return final;
}

string adjustCentre(string s){
    int n, i,mid, carry;
    mid = s.length() / 2;
    i = mid;
    do{
        n = s[i] - 48;
        n++;
        carry = n / 10;
        s[i] = (n % 10) + 48;
        i++;
    }while(i < s.length() && carry);
    for(int i=0, j=s.length() - 1;i<=j;i++, j--)
        s[i] = s[j];
    return s;
}

string prepareString(string s){
    for(int i=0, j=s.length() - 1;i<=j;i++, j--){
        if(s[j] < s[i]){
            s[j] = s[i];
            continue;
        }
        if(s[j] > s[i]){
            s[j] = s[i];
            int n, carry = 1, k=j-1;
            while(carry && k >= 0){
                n = s[k] - 48;
                n++;
                carry = n / 10;
                s[k] = (n % 10) + 48;
                k--;
            }
        }
    }
    return s;
}

string findNext(string s){
    return prepareString(prepareString(s));
}

string findNextPalindrome(string s){
    string res;
    if(checkPalin(s)){
        if(check9(s))
            return for9(s);
        return adjustCentre(s);
    }
    return findNext(s);
}

int main(){
    int t;
    string s;
    scanf("%d", &t);
    while(t--){
        cin >> s;
        cout << findNextPalindrome(s) << endl;
    }
    return 0;
}
