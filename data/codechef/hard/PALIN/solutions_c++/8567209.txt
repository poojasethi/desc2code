#include<bits/stdc++.h>
using namespace std;
bool areAll9(string arr,int len){
	for(int i=0;i<len;i++){
		if(arr[i]!=9)
		return false;
	}
	return true;
}
string nextPalin(string num,int n){
	int mid=n/2;
	int i=mid-1;
	int j=(n%2)?(mid+1):mid;
	bool leftSmall=false;
	while(i>=0 && num[i]==num[j]){
		i--;j++;
	}
	if(i<0 || num[i]<num[j]){
		leftSmall=true;
	}
	while(i>=0){
		num[j++]=num[i--];
	}
	int carry;
	if(leftSmall){
		carry=1;
		i=mid-1;
		if(n%2){
			num[mid]=num[mid]+carry;
			carry=num[mid]/10;
			num[mid]=num[mid]%10;
			j=mid+1;
		}
		else{
			j=mid;
		}
	}
	while (i >= 0)
        {
            num[i] += carry;
            carry = num[i] / 10;
            num[i] %= 10;
            num[j++] = num[i--]; 
        }
	return num;
}
