#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
long long int n;
char a[1000000];
bool alln()
{
    for(int i=0;i<n;i++)
    {
        if(a[i]!='9')
            return false;
    }
    return true;
}
int main()
{
    long long int nu,i,mid,j,carry,temp,y;
    int t,flag;

     scanf("%d\n",&t);
    while(t--)
    {
        flag=0;
     gets(a);

      n=strlen(a);
        if(alln())
        {
            a[0]='1';
            for(i=1;i<n;i++)
            a[i]='0';
            a[n]='1';
            a[n+1]='\0';


        }
        else
        {
           for(i=0;i<n/2;i++)
            {
                if(a[i]>a[n-1-i])
                    flag=1;
               else if(a[i]<a[n-1-i])
                    flag=-1;
                a[n-1-i]=a[i];


            }
            if(n%2==0)
                mid=n/2-1;
            else
                mid=n/2;
            if(flag==0 || flag==-1)
            {
                i=0;
                while(a[mid-i]=='9')
                {
                    a[mid-i]='0';
                    a[n-1-mid+i]='0';
                    i++;

                }
                a[mid-i]++;
                a[n-1-mid+i]=a[mid-i];

            }

        }
        printf("%s\n",a);
    }
}
