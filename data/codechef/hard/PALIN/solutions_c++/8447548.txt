#include<cstdio>
#include<cstring>
using namespace std;
const int MAX=1000000+10;
char num[MAX];
int main(){
	//freopen("PALIN.in","r",stdin);
	int t,n,i,d;
	scanf("%d",&t);
	while(t--){
		scanf("%s",num);
		n=strlen(num);
		//special case 9999
		for(i=0;i<n;i++)if(num[i]!='9')break;
		if(i==n){
			putchar('1');
			for(i=0;i<n-1;i++)putchar('0');
			putchar('1');
			puts("");
			continue;
		}
		for(i=(n>>1)-1;i>=0;i--){
			if(num[i]>num[n-i-1]){
				break;
			}else if(num[i]<num[n-i-1]){
				i=-1;
				break;
			}
		}
		if(i==-1){
			d=n/2-!(n&1);
			num[d]++;
			for(i=d;i>=0;i--){
				if(num[i]>'9'){
					num[i]-=10;
					num[i-1]++;
				}
			}
		}
		for(i=0;i<(n>>1);i++){
			num[n-i-1]=num[i];
		}
		puts(num);
	}
	return 0;
}
