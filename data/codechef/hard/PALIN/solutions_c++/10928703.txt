#include<iostream>
#include<string>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        string number;
        cin>>number;
        int size=number.size();
        bool allnine=true;
        for(int i=0;i<size;i++){
            if(number[i]!='9'){
                allnine=false;
                break;
            }
        }
        if(allnine){
            cout<<"1";
            for(int i=0;i<size-1;i++){
            cout<<"0";
            }
            cout<<"1"<<endl;
            continue;
        }
        else{
            int last=size-1,first=0;
            bool Isgreater=false;
            while(last>=first){
                if(number[first]<number[last]){
                    Isgreater=false;
                }
                else if(number[first]>number[last]){
                    Isgreater=true;
                }
                number[last]=number[first];
                last--;
                first++;
            }

            if(!Isgreater){
                int mid=(size-1)/2;
                last = size-mid-1;
                int first=0;
                if(number[mid]!='9'){
                    number[mid]++;
                    number[last]=number[mid];
                }
                else{
                    while(number[mid-first]=='9'){
                        number[mid-first]='0';
                        number[last+first]='0';
                        first++;
                    }
                    number[mid-first]++;
                    number[last+first]=number[mid-first];
                }
            }
        }
        cout<<number<<endl;
    }
    return 0;
}
