#include <iostream>
#include <string>

using namespace std;
int main ()
{
	std::ios::sync_with_stdio(false);
	int t;
	string number;
	int size;
	int i;
	bool nines = true;
	bool isGreater = false;
	int mid ;
	cin >> t;
	while (t--)
	{
		cin >> number;
		size = number.length();
		nines = true;
		isGreater = false;
		int last, mid;
		for (i = 0 ; i < size; i++)
		{
			if (number[i] != '9')
			{
				nines = false;
				break;
			}
		}
		if (nines)
		{
			number.insert(0, "1");
			for (i = 1 ; i < size ; i++)
				number [i] = '0';
			number[size] = '1';
		}
		else
		{
			last = size ;
			for (i = 0 ; i < size / 2; i++ )
			{
				last --;
				if (number[i] < number[last])
					isGreater = false;
				else if (number[i] > number[last])
					isGreater = true;
				number[last] = number[i];
			}
			//if same or smaller ,increment at mid
			if (!isGreater)
			{
				mid = (size - 1) / 2;
				last = size - mid - 1;
				i = 0;
				if (number[mid] != '9')
				{
					number[mid] ++;
					number [last] = number[mid];
				}
				else
				{
					while (number[mid - i] == '9')
					{
						number[mid-i] = '0';
						number[last + i] = '0';
						i++;
					}				
					number[mid - i] ++;
					number[last + i] = number[mid - i];
				}
			}


		}
		cout << number << endl;
	}
}
