#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main()
{
	char number[1000000],palin[1000000];
	bool carry;
	int t;
	scanf("%d", &t);
	while (t--)
	{
		carry = false;
		cin >> number;
		strcpy(palin, number);
		int n = strlen(number);
		for (int i = 0; i < n / 2; ++i)
			palin[n - i - 1] = palin[i];
		if (strcmp(palin, number) <= 0)
		{
			int i =( n-1 )/ 2;
			do
			{
				if (palin[i] == '9')
				{
					carry = true;
					palin[i] = palin[n - i - 1] = '0';
				}
				else
				{
					carry = false;
					palin[i] = ++palin[n - i - 1];
				}
				i--;
			} while (i >= 0 && carry);
		}
		if (carry == true)
		{
			++palin[n - 1];
			printf("1%s\n", palin);
		}
		else
			printf("%s\n", palin);
	}
}
