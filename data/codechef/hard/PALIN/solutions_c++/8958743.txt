#include<stdio.h>
#include<string.h>
int main(){
	char a[1000002];
	int t;
	scanf("%d",&t);
	while(t--){
	scanf("%s",a);
	int i,l,mid,j;
	l=strlen(a);
	for(i=0;i<l;i++){
		if(a[i]!='9')
			break;
	}
	if(i==l){
		for(i=1;i<l;i++)
			a[i]='0';
		a[0]='1';
		a[l]='1';
		a[l+1]=0;
		printf("%s\n",a);
	}
	else{
	a[l-1]++;
	j=l-1;
	while(a[j]>'9'){
		a[j-1]++;
		a[j]='0';
		j--;
	}   ///
	if(l%2==0){
		mid=l/2-1; 
	}
	else
		mid=(l-1)/2;
	for(i=0;i<=mid;i++){
		if(a[i]!=a[l-1-i]){
			if(a[i]<a[l-1-i]){
				a[l-i-2]++; 
				j=l-2-i;
				while(a[j]>'9'){
				a[j-1]++;
				a[j]='0';
				j--;
				}				//
				a[l-1-i]=a[i];
			}
			else if(a[i]>a[l-1-i]){
				a[l-1-i]=a[i];
			}
		}
	}
	printf("%s\n",a);}
	}
		return 0;
}
