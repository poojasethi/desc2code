#include<bits/stdc++.h>
using namespace std;

int main()
{
    //freopen("in2.in","r",stdin);
    //freopen("out2.out","w",stdout);

    int i,j,k;
    int x,y,z,n;
    int t;
    string str,str2;
    char ch;
    bool done;

    cin>>t;
    cin.ignore(256,'\n');
    
    while(t--)
    {
        getline(cin,str);

        n=str.length();
        done=false;

        i=0;
        j=n-1;

        while(i<(n/2))
        {
            if(str[j]<str[i])
            {
                done=true;
            }

            else if(str[j]>str[i])
            {
                done=false;
            }

            str[j]=str[i];

            i++;
            j--;
        }

        if(!done)
        {
            i=(n/2)-1;
            j=i+1;

            if(n%2==1)
            {
                //for odd length
                
                if(str[n/2]=='9')
                {
                    str[n/2]='0';

                    j++;
                }

                else
                {
                    str[n/2]++;
                    done=true;
                }
            }

            while(i>=0 && !done)
            {
                if(str[i]=='9')
                {
                    str[i]='0';
                    str[j]='0';
                }

                else
                {
                    str[i]++;
                    str[j]++;

                    done=true;
                }

                i--;
                j++;
            }

            if(!done)
            {
                str[0]='1';
                str+='1';

                done=true;
            }

        }

        cout<<str<<endl;
    }

    return 0;
}
