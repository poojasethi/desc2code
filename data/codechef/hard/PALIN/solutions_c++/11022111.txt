#include <cstdio>
#include <cstring>
#include <cstdlib>
 
int main()
{
	char a[1000005];
	int t,len,i,j,mid,flag;
	scanf("%d",&t);
	while(t--)
	{
		flag = 0;
		scanf("%s",a);
		len = strlen(a);
		for(i = 0;i < len;i++)
		{
			if(a[i]!='9')
				break;
		}
		if(i == len)
		{
			a[len] = a[0] = '1';
			for(j = 1;j<len;j++)
			{
				a[j] = '0';
			}
			a[len+1] = '\0';
		}
		else
		{
			flag = 0;
			if(len&1)
				mid = len/2;
			else
				mid = len/2 - 1;
			j = 0;
			while(j<len/2)
			{
				if(a[j] < a[len-j-1]) 
					flag = -1; 
				else if(a[j] > a[len-j-1]) 
					flag = 1;
				a[len-1-j]=a[j];
				j++;
			}
			
			j = 0;
 
		
			if(flag<1)
			{
				
				i = 0;
				while((a[mid-i]=='9'))
				{
					a[len-1-mid+i] = a[mid-i] = '0';
					i++;
				}
				a[mid-i] = (a[mid-i]+1);
				a[len-1-mid+i] = a[mid-i];
			}
 
		}
		printf("%s\n",a);
	}
	return 0;
}
