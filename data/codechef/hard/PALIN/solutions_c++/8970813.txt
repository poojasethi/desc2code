#include <bits/stdc++.h>
 
using namespace std;
 
char str[1000005];

#define lld long long int
 
int main() {
	lld t;
	cin >> t;
	
	while(t--) {
		lld i, len, x = 0;
		
		scanf("%s", str);
		
		len = strlen(str);
		
		for (i = 0; i < len; i++) {
			if (str[i] != '9')
			break;
		}
		
		if (i == len) {
			str[0] = '1';
			for (i = 1; i < len; i++) {
				str[i] = '0';
			}
			str[len] = '1';
			str[len + 1] = '\0';
			cout << str << endl;
			continue;
		}
		
		lld flag = 0, temp, t = 0;
		for(i=0; i<len/2; i++) {
			if(str[i] < str[len-i-1])
				flag = -1;
			else if(str[i] > str[len-i-1])
				flag = 1;
			str[len-i-1] = str[i];
		}
		
		if(flag == -1 || flag==0) {
			if (len % 2 == 0)
				temp = len/2 - 1;
			else
				temp = len/2;
			while (str[temp - t] == '9') {
				str[temp-t] = '0';
				str[len-1-temp+t] = '0';
				t ++;
			}
			str[temp-t] ++;
			str[len-1-temp+t] = str[temp-t];
		}
		
		cout << str << endl;
	}
		
	
	return 0;
}
