#include<iostream>
using namespace std;
#define MAX 1000002
#define intConst 48

int length(char*);
bool areAll9(char*);
void getNextPallindrome(char* c){
	int size= length(c);
	int left= (size/2)-1;
	int right= (size%2==1)?((size/2) + 1):(size/2);
	int i=left;
	int j= right;
	
	while(i>=0 && c[i]==c[j]){
		i--;
		j++;
	}
	
	if(i<0 || c[i]<c[j]){
		int carry=1;
		if(size%2 ==1){
			int t= c[size/2]- intConst;
			t+=carry;
			carry= t/10;
			c[size/2]= (t%10) + intConst;
		}
		
		i=left;
		j= right;
		while(i>=0){
			int t= c[i]- intConst;
			t+=carry;
			carry= t/10;
			c[i]= (t%10) + intConst;
			c[j++]= c[i--];
		}
	}
	else if(c[i]>=c[i]){
		while(i>=0)
			c[j++]= c[i--];
	}
}

int main(){
	char c[MAX];
	int t, size;
	cin>>t;
	
	while(t--){
		cin>>c;
		size= length(c);
		
		if(areAll9(c)){
			cout<<1;
			for(int i=1; i<size; i++)
				cout<<0;
			cout<<1<<endl;
		}
		else{
			getNextPallindrome(c);
			cout<<c<<endl;
		}
	}
}

bool areAll9(char* c){
	for(int i=0; c[i]!='\0'; i++){
		if(c[i]!='9'){
			return false;
		}
	}
	return true;
}

	int i;
	for(i=0; c[i]!='\0'; i++);
