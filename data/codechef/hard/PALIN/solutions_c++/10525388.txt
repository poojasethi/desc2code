#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		char s[1000002];long l,mid;
		int nine=1,j,flag=0;
		scanf("%s",s);
		l=strlen(s);
		for(int i=0;i<l;i++)
			if(s[i]!='9')
			{
				nine=0;
				break;
			}
		if(nine==1)
		{
			s[0]='1';
			for(int i=1;i<l;i++)
				s[i]='0';
			s[l]='1';
			s[l+1]='\0';
		}
		else
		{
			for(int i=0;i<l/2;i++)
			{
				if(s[i]<s[l-i-1])
					flag=1;
				else if(s[i]>s[l-i-1])
					flag=-1;
				s[l-i-1]=s[i];
			}
			if(l%2==0)
				mid=l/2-1;
			else
				mid=l/2;
			if(flag==0 || flag==1)
			{
				j=0;
				while(s[mid-j]=='9')
				{
					s[mid-j]='0';
					s[l-1-mid+j]='0';
					j++;
				}
				s[mid-j]++;
				s[l-1-mid+j]=s[mid-j];
			}
		}
		cout<<s<<endl;
	}
	return 0;
}
