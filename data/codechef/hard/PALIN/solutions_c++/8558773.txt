#include <cstdio>
#include <cstring>
using namespace std;

char str[1000003];

int charToInt(char c){
	return c-'0';
}

char intToChar(int n){
	return (char)('0' + n);
}

int main(){
	int t;
	scanf("%d", &t);
	while(t--){
		scanf("%s", str);
		bool flag1 = true, flag2 = false;
		int len = strlen(str);
		for(int i=0; i< len; i++){
			if(str[i]!='9'){
				flag1 = false;
				break;
			}
		}

		if(flag1){
			printf("1");
			for(int i=0; i<len-1; i++)
				printf("0");
			printf("1\n");
			continue;
		}
		if(len ==1){
            printf("%d\n", charToInt(str[0])+1);
            continue;
		}
		int half = len/2, count =0;
		for(int i=0; i< half; i++){
            if(str[i] == str[len - i - 1]){
                count++;
            }
			else if(str[i] < str[len - i - 1]){
				flag2 = true;
			}
			else{
				flag2 = false;
			}
			str[len - i - 1] = str[i];
		}

		if(count==half || flag2){
			int rem = 1;
			if(len%2==0){
				int left = half - 1;
				int right = half;
				while(rem != 0 && left >= 0){
					int n = charToInt(str[left]) + rem;
					str[left] = intToChar(n%10);
					rem = n/10;
					str[right] = str[left];
					left--;
					right++;
				}
			}else {
				int n = charToInt(str[half]) + rem;
				str[half] = intToChar(n%10);
				rem = n/10;
				int left = half -1;
				int right = half + 1;
				while(rem != 0 && left >= 0){
					int n = charToInt(str[left]) + rem;
					str[left] = intToChar(n%10);
					rem = n/10;
					str[right] = str[left];
					left--;
					right++;
				}
			}
		}
		printf("%s\n", str);

	}
	return 0;
}
