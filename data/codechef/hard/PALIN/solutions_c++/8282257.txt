void inc(char *c, long long int j)
{
	if(*(c + j) == '9')
	{
		*(c + j) = '0';
		inc(c, j - 1);
	}
	else
	{
		int a;
		a = *(c+j);
		a++;
		*(c + j) = a;
		return;
	}
}
