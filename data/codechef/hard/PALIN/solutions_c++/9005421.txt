#include<iostream>
using namespace std;
#include<string>
#include<stdio.h>

main(){
    int t;
    cin>>t;

    while(t--){
        string a;
        cin>>a;

        int len = a.size();

        int pos = len-1;
        while (a[pos]=='9' && pos>=0)
            a[pos--]='0';
        if( pos==-1) {
            a[len-1] = '1';
            cout << '1' << a;
        }
        else {
            a[pos]++;

            // INITIALISE MIDDLE
            int i, i2, j;
            i2 = i = j = len/2;
            if(len%2==0) i2 = (--i);

            // CHECK PARTS
            while (a[i]==a[j] && i>=0) {
                i--;
                j++;
            }

            //IF NOT A PALINDROME ALREADY
            if (i!=-1) {
                // ADD 1 IF REQUIRED
                if (a[j] > a[i]) {
                    int pos = i2;
                    while (a[pos]=='9')
                        a[pos--]='0';
                    a[pos]++;
                }
            }

            // OUTPUT:
            for (int k=0 ; k<=i2 ; k++)
                printf("%c" , a[k]);
            if (len%2)
                i2--;
            while(i2>=0)
                printf("%c", a[i2--]);

        }
        cout<<endl;
    }
}
