vector<int> nextpalin(vector<int> s){
	int num;
	bool great = false;
	bool small = false;
	for(int i=0;i<s.size()/2;i++){
		int a = s[i];
		int b = s[s.size()-1-i];
		if(a<b){
			s[s.size()-1-i] = s[i];
			small = true;
			great = false;
			}
		else if(a>b){
			s[s.size()-1-i] = s[i];
			great = true;
			small = false;
			}
		}
	if(small){
		if(s.size()%2==0){
			int index = s.size()/2 -1;
			bool changed = false;
			while(index>=0){
				if(s[index]!=9){
					s[index]++;
					s[s.size()-1-index] = s[index];
					changed=true;
					break;
					}
				else{
					s[index] = 0;
					s[s.size()-1-index] = 0;
					}
				index--;
				}
			return s;
			}
		else{
			int index = s.size()/2;
			bool changed = false;
			if(s[index]!=9){
				s[index]++;
				changed = true;
				}
			while(index>=0 && !changed){
				if(s[index]!=9){
					s[index]++;
					s[s.size()-1-index] = s[index];
					changed=true;
					break;
					}
				else{
					s[index] = 0;
					s[s.size()-1-index] = 0;
					}
				index--;
				}
			return s;
			}
		}
	else{
		return s;
		}
	}
