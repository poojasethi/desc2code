#include<iostream>
#include<string>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        string number;
        cin>>number;
        int size=number.size();
        bool allnine=true;
        for(int i=0;i<size;i++){
            if(number[i]!='9'){
                allnine=false;
                break;
            }
        }
        if(allnine){
            cout<<"1";
            for(int i=0;i<size-1;i++){
            cout<<"0";
            }
            cout<<"1"<<endl;
            continue;
        }
        else{
            int last=size-1,first=0;
            bool Isgreater=false;
            while(last>=first){
                if(number[first]<number[last]){
                    Isgreater=false;
                }
                else if(number[first]>number[last]){
                    Isgreater=true;
                }
                number[last]=number[first];
                last--;
                first++;
            }

            if(!Isgreater){
                int mid1=(size-1)/2;
                int mid2 = size-mid1-1;
                int i=0;
                if(number[mid1]!='9'){
                    number[mid1]++;
                    number[mid2]=number[mid1];
                }
                else{
                    while(number[mid1-i]=='9'){
                        number[mid1-i]='0';
                        number[mid2+i]='0';
                        i++;
                    }
                    number[mid1-i]++;
                    number[mid2+i]=number[mid1-i];
                }
            }
        }
        cout<<number<<endl;
    }
    return 0;
}
