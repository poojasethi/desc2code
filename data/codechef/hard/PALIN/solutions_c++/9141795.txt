#include<iostream>

using namespace std;
int tests;
string n;

string print_next_palin(string n) {

  // If the string is already a palindrome inc it by 1.
  {
    int k;
    for(k = n.length()-1; n[k] == '9'; --k) n[k] = '0';
    if(k < 0) n = "1" + n;
    else n[k] = n[k] + 1;
  }

  {
  int j = n.length()-1, i=0, k;
  // Algorithm for n > 1 
  for(; i<=j; ++i, --j) {
    if(n[i] == n[j]) continue;
    if(n[i] > n[j]) n[j] = n[i];
    else {

      int k = j-1;
      // Add 1 until there isn't a 9.
      while(n[k] == '9') {
	n[k] = '0';
	k--;
      }
      n[k] = n[k] +  1;
      n[j] = n[i];
    }
    //cout<<n<<endl;

  }
  }

  return n;

}


int main() {
  cin>>tests;
  while(tests--) {
    cin>>n;
    cout<<print_next_palin(n)<<endl;
    cout<<endl;
  }
  return 0;
}
