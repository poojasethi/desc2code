inline int fr() {
    register int c = gc();
	int x = 0; 
    for(; (c<48 || c>57); c = gc());
    for(; c>47 && c<58 ; c = gc()) {
        x = x*10 + c - 48;
}
return x;}
inline void fp(int n){
    if (n / 10 != 0)
        fp(n / 10);
    putchar((n % 10) + '0');
}

int dp[100010][210];

int main() { 
	//TRY("in.txt","out.txt");
	int t=fr(),m,n,a,b,k;
	char s[100010],w[100010];
	while(t--){
		scanf(" %s %s",&s,&w);
		m=strlen(s);
		n=strlen(w);
		a=fr();b=fr();k=fr();
		if(a==0){
			fp(0);pc('\n');
		}
		else if(m-n>k){
			printf("-1\n");
		}
		else{
			memset(dp,127,sizeof(dp));
			//dp[i][j+103] means edit distance b/w a[i] and b[i+j]
			for(int i=0;i<=k;++i){
				dp[0][i+103]=i*a;
			}
			for(int i=0;i<=k;++i){
				dp[i][-i+103]=i*a;
			}
			for(int i=1;i<=m;++i){
				for(int j=max(-i+1,-k);j<=min(n-i,k);++j){
					if(s[i-1]==w[i+j-1])
						dp[i][j+103]=dp[i-1][j+103];
					else{
						dp[i][j+103]=dp[i-1][j+103]+b;//min(dp[i-1][j+104]+a,min(dp[i][j+102]+a,dp[i-1][j+103]+b));
						if(j!=k) dp[i][j+103]=min(dp[i-1][j+104]+a,dp[i][j+103]);
						if(j!=-k) dp[i][j+103]=min(dp[i][j+102]+a,dp[i][j+103]);
					}
				}
			}
			if(dp[m][n-m+103]>k) printf("-1");
			else fp(dp[m][n-m+103]);
			pc('\n');
		}
	}
return 0;  
 }
