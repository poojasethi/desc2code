using namespace std;

int editdistance100(string a, int a_size, string b, int b_size, int addrem, int replace, int total)
{
	if(addrem == 0)
		return 0;
	int N = total/addrem+1;
	int i,j,j_start,temp,ans = 0,mini,*swap,k;
//	int *first = (int*)malloc((2*N+1)*sizeof(int));
//	int *second = (int*)malloc((2*N+1)*sizeof(int));
//	cout<<replace<<" "<<addrem<<" "<<total<<endl;
	int array_one[2*N+1];
	int array_two[2*N+1];
	for(i=0;i<=2*N;i++)
	{
		array_one[i] = 0;
		array_two[i] = 0;
	}
	int *first = array_one;
	int *second = array_two;

	for(i=N;i<=2*N;i++)
		first[i] = (i-N)*addrem;

	for(i=1;i<=a_size;i++)
	{	
		mini = 100000;
		if(i<=N)
		{
			j_start = 1;
			second[N-i] = i*addrem;
		}
		else
			j_start = i-N;
		
		for(j=j_start;j<=min(i+N,b_size);j++)
		{
			if(a[i-1] == b[j-1])
				temp = first[j-i+N];
			else
				temp = first[j-i+N] + replace;

			if(j-i != -N && j-i != N)
			{
				second[j-i+N] = min(temp, min(first[j-i+N+1]+addrem, second[j-i+N-1]+addrem));
			}
			else if(j-i+N == 0)
				second[j-i+N] = min(temp, first[j-i+N+1]+addrem);
			else if(j-i == N)
				second[j-i+N] = min(temp, second[j-i+N-1]+addrem);

			ans = second[j-i+N];
			if(ans<mini)
				mini = ans;
//			cout<< ans<< " ";
		}
//		cout<< endl;
		if(mini>total)
			return -1;
		swap = first;
		first = second;
		second = swap;
	}
	if(ans>total)
		ans = -1;
	return ans;
}


int main()
{
	int t, addrem, replace, total;
	string a, b;
	cin>> t;
	while(t--)
	{
		cin>> a;
		cin>> b;
		cin>> addrem >> replace >> total;
		cout<< editdistance100(a,a.length(),b,b.length(),addrem,replace,total)<< endl;
		a.clear();
		b.clear();
	}
	return 0;
}
