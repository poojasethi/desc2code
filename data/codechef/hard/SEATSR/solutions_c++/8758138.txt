#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>
using namespace std;
int a[100001];
int b[100001];
int main() {
	int t;
	cin>>t;
	while(t--){
		char n1[100001],n2[100001];
		//cin>>n1>>n2;
		scanf("%s%s",&n1,&n2);
		int l1=strlen(n1);
		int l2=strlen(n2);
		//int k=400000;
		int r,s,k;
		cin>>r>>s>>k;
		if(abs(l1-l2)*r>k){cout<<-1<<endl;continue;}
		if(r==0){cout<<0<<endl;continue;}
		for(int i=0;i<=l2;i++)	a[i]=i*r;
		for(int i=0;i<=l2;i++)	b[i]=1000000000;
		for(int i=1;i<=l1;i++){
			a[0]=(i-1)*r;
			b[0]=i*r;
			for(int j=max(1,i-k/r);j<=min(l2,i+k/r);j++){
				if(n1[i-1]==n2[j-1]){
					b[j]=a[j-1];
				}
				else{
					b[j]=min(r+a[j],min(r+b[j-1],s+a[j-1]));
				}
				
			}
			for(int j=max(1,i-k/r);j<=min(l2,i+k/r);j++)	a[j]=b[j];
		}
		if(b[l2]<=k)
		cout<<b[l2]<<endl;
		else cout<<-1<<endl;
	}
	return 0;
}
