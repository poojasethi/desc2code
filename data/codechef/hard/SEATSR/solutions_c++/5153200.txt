int min(int a, int b) {
   return a < b ? a : b;
}
int min(int a, int b, int c)
{
    return min(min(a, b), c);
}
int dp[2][100002];
int ed(char X[], char Y[],int a,int b,int k)
{
    int l, t, c;
    int m = strlen(X)+1;
    int n = strlen(Y)+1;
    for(int i = 0; i < m; i++)
        dp[0][i] = i*a;
    int z = 1,i;
    for(i = 1; i < n; i++)
    {
        int j = i-k;
        int v = j;
        if(j<=0)
            j = 1;
        if(j>1)
            dp[i%2][j-1] = inf;
        else
            dp[i%2][j-1] = (i)*a;
        for(; j <min(v+2*k+1,m); j++)
        {
            l = dp[i%2][j-1];
            l+= a;
            t= dp[(i+z)%2][j];
            t += a;
            c = dp[(i+z)%2][j-1];
            //cout << X[i-1]<< " " << Y[j-1] << endl;
            if(X[j-1] != Y[i-1])
                c+=b;
            dp[i%2][j] = min(l,t,c);
        }
        //for(int f = 0;f<2;f++){
          //  for(int e = 0;e <m;e++)
            //    cout << dp[f][e] << " ";
      //      cout << endl;
        //}
      //  cout << endl << endl;
    }
    if(dp[(i+z)%2][m-1] <= k)
        return dp[(i+z)%2][m-1] ;
    else
        return -1;
}
int main(){
    int t;
    scan(t);
    while(t--){
        int a,b,k;
        char s1[100002],s2[100002];
        scans(s1);scans(s2);
        scan(a);scan(b);scan(k);
        if((a==0)||(b==0&&(strlen(s1)==strlen(s2)))||(strcmp(s1,s2)==0))
            printf("0\n");
        else if(abs(strlen(s1)-strlen(s2))*a>k || (k==0))
			printf("-1\n");

        int l = ed(s1,s2,a,b,k);

        }
    }
}
