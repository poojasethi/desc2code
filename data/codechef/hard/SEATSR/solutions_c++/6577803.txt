#define F first
#define S second

#define rep(i,in1,n) for(i=in1;i<=n;i++)
#define repd(i,in1,n) for(i=in1;i>=n;i--)
#define pf(n) printf("%d ",n);
#define sf(n) scanf("%d",&n)
#define sl(n) scanf("%I64d",&n)
#define nl printf("\n")
#define mem(arr,init) memset(arr,init,sizeof(arr))
#define vi vector<int>
#define vvi vector<vi>

#define pb push_back
#define all(c) (c).begin(),(c).end()
#define present(c,x) ((c).find(x) != (c).end())
#define cpresent(c,x) (find(all(c),x) != (c).end())
#define mp make_pair
#define ep emplace_back//c++11
#define ii pair<int,int>
#define MAX 100000000
using namespace std;
 
string str1,str2;
int dp[100003];
int last[100003];
 
int main()
{
	int i,j,k,t,n,m,a,b,c,x,y,z;
	sf(t);
	
	while(t--)
	{
		mem(dp,0);
		
		cin>>str1;
		
		cin>>str2;
		m=str1.length();
		n=str2.length();
		sf(a); sf(b); sf(k);
		if(a==0 )
		{
			printf("0\n");
			continue;
		}
		if(b==0)
		{
			z=a*abs(m-n);
			if(z>k)
			printf("-1\n");
			else
			printf("%d\n",z);
			continue;
		}
		if(k==0)
		{
			printf("-1\n");
			continue;
		}
		
		
		if(abs(m-n)>k)
		{
			printf("-1\n");
			continue;
		}
		for(i=0;i<max(m,n);i++)
		{
			dp[i]=MAX;
			last[i]=MAX;
		}
			
		for(i=0;i<=min(n,k);i++)
		{
			last[i]=i*a;
		}		
		for(i=1;i<=m;i++)
		{
			
			dp[0]=a*i;			
			for(j=max(1,i-k-1);j<=min(n,i+k+1);j++)
			{
				
				if(str1[i-1]==str2[j-1])
				{
					
					c=last[j-1];
					//continue;
				}
				else
				c=last[j-1]+b;
				dp[j]=min(c,min(last[j]+a,dp[j-1]+a));//check
				last[j-1]=dp[j-1];
				
			}
			last[j]=dp[j-1];
			
			
		}
		if(dp[n]>k)
		printf("-1\n");
		else
		printf("%d\n",dp[n]);
		
		
		
		
	}
 
 
 
 
 
	return 0;
}
