#include <stdio.h>
#include <string.h>
#define min3(a,b,c) ((a<=b&&a<=c)?(a):((b<=a&&b<=c)?b:c))
#define min(a,b) (a<=b?a:b)
#define max(a,b) (a>=b?a:b)
 
long d[203][100001];
char s[100001], w[100001];
 
int main() {
 
int T, a, b, k;
scanf("%d",&T);
while (T--) {
scanf("%s\n%s\n%d %d %d\n",s,w,&a,&b,&k);
long slen = strlen(s), wlen = strlen(w);
if (a==0) {
printf("0\n");
continue;
}
if (slen-wlen>100 || slen-wlen<-100) {
printf("-1\n");
continue;
}
for (long i=0; i<=100000; i++) d[0][i] = d[202][i] = 101;
for (long i=0; i<=100; i++) d[i+101][0] = i*a;
for (long j=0; j<=100; j++) d[-j+101][j] = j*a;
for (long j=1; j<=wlen; j++) {
for (long i=max(1,j-100); i<=min(j+100,slen); i++) {
if (s[i-1] == w[j-1]) d[i-j+101][j] = d[i-j+101][j-1];
else d[i-j+101][j] = min3(d[i-1-j+101][j]+a,d[i-j+1+101][j-1]+a,d[i-j+101][j-1]+b);
}
}
if (d[slen-wlen+101][wlen]<=k) printf("%ld\n",d[slen-wlen+101][wlen]);
else printf("-1\n");
}
return 0;
}
