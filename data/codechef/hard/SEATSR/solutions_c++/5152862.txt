#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> pii;
typedef long long ll;
void fastStream(){cin.tie(0);std::ios_base::sync_with_stdio(0);}

int T;
char S[100010];
char W[100010];
int A,B,K;
int dp[100010][201];
const int base=100;
const int INF=1<<29;
int len1,len2;

int main(){
  
  fastStream();
  cin>>T;
  while(T--){
    cin>>S>>W;
    cin>>A>>B>>K;
    len1=strlen(S);
    len2=strlen(W);
    if(A==0)cout<<0<<endl;
    else if(B==0)cout<<abs(len1-len2)*A<<endl;
    else{
      for(int a=len1;a>=0;a--){
        for(int b=200;b>=0;b--){
          int res=1<<29;
          const int pos1=a;
          const int pos2=(b-base)+a;
          if(a==len1)res=(len2-pos2)*A;
          else if(b==len2)res=(len1-a)*A;
          else{
            if(pos2<len2&&S[pos1]==W[pos2])res=min(res,dp[pos1+1][b]);
            if(pos2<len2)res=min(res,dp[pos1+1][b]+B);
            if(b-1>=0)res=min(res,dp[pos1+1][b-1]+A);
            if(pos2<len2&&b+1<201)res=min(res,dp[pos1][b+1]+A);
          }
          dp[a][b] = res;
        }
      }
      int res = dp[0][base];
      if(res <= K)
        cout<<res<<endl;
      else
        cout<<-1<<endl;
    }
  }
  return 0;
}
