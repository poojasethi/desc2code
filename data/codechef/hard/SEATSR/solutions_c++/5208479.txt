#include<cstdio>
#include<algorithm>
#include<cstring>
const int N = 100005;
using namespace std;
char s[N],t[N];
int f[N][101], g[N][101];
int main() {
    int cas,a,b,k; scanf("%d", &cas);
    while(cas--) {
        scanf("%s%s%d%d%d",s+1,t+1,&a,&b,&k);
        int ls=strlen(s+1), lt=strlen(t+1);
        if(a==0) puts("0");
        else if(abs(ls-lt)>k) puts("-1");
        else {
            for(int i=1; i<=k; i++) f[0][i]=g[i][i]=i*a;
             for(int j, i1, i=1; i<=ls; i++)   {

                    for(j=-k; j<=k; j++){
                    //printf("%d %d\n", i, j);
                    if(j>=0) {
                        i1=i+j;
                        if(i1>lt) break;
                        if(s[i]==t[i1]) f[i][j]=f[i-1][j];
                        else {
                            f[i][j]=min((j==0?g[i][1]:f[i][j-1])+a,f[i-1][j]+b);//insert, replace
                            if(j!=k && f[i][j]>f[i-1][j+1]+a)  f[i][j]=f[i-1][j+1]+a;//remove

                        }
                        //printf("%d %d %d--\n",i,i1,f[i][j]);
                    }else {
                        i1=i+j;
                        if(i1>lt) break;
                        if(i1<=0) continue;
                        if(s[i]==t[i1]) g[i][-j]=g[i-1][-j];
                        else {
                            g[i][-j]=min((j==-1?f[i-1][0]:g[i-1][-j-1])+a,g[i-1][-j]+b);//remove,replace
                            if(j!=-k && g[i][-j]>g[i][-j+1]+a)  g[i][-j]=g[i][-j+1]+a;
                        }
                        //printf("%d %d %d %d %d--\n",i,j,i1,g[i][-j],g[i-1][-j]);
                    }
                }
            }
            int ans;
            if(ls<=lt) ans=f[ls][lt-ls];
            else ans=g[ls][ls-lt];
            printf("%d\n", ans<=k?ans:-1);
        }
    }
}
