#include<algorithm>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
#define si(t) scanf("%d",&t);
#define MAX 100000
#define INF 1000000

int minimum(int x,int y,int z)
{
	if(x<=y && x<=z)
		return x;
	if(y<=x&&y<=z)
		return y;
	return z;	
}

int main()
{
	int T;
	int dp[2][MAX+5];
	char str1[MAX+5],str2[MAX+5];
	si(T);
	while(T--){
	scanf("%s",str1);
	scanf("%s",str2);
	int a,b,k;
	scanf("%d%d%d",&a,&b,&k);
	int l1,l2;
	l1= strlen(str1);
	l2=strlen(str2);
	if(a==0) {
		printf("0\n");
		continue;
		}
	dp[0][0]=0;
	int i;
	for( i=1; i<=l2 && i*a<=k;i++)
	{
		dp[0][i]=dp[0][i-1]+a;
		dp[1][i]=INF;
	}
	dp[1][0]=a;
	int rlim=i-1;
	for( ;i<=l2;i++)
	{
		dp[0][i]=INF;
		dp[1][i]=INF;	
	}
	for(  i=1;i<=l1;i++)
	{
		int maxi=(0>i-rlim)?0:(i-rlim);
		int mini=(l2<i+rlim)?l2:(i+rlim);
		for(int j=maxi;j<=mini;j++)
		{
			if(j==0){
				dp[1][j]=i*a;
				continue;
			}
			if(str1[i-1]!=str2[j-1])
				dp[1][j]=minimum(dp[0][j]+a,dp[1][j-1]+a,dp[0][j-1]+b);
			else
				dp[1][j]=dp[0][j-1];
		}
		for(int j=maxi;j<=mini;j++)
			dp[0][j]=dp[1][j];
	}
		printf("%d\n",(dp[1][l2]<=k)?dp[1][l2]:-1);
	}
    return 0;
}
