#define frev(i,a,b) for(ll i=a;i>=b;i--)

#define gc getchar_unlocked
ll readint()
{
    char c = gc();
    while(c<'0' || c>'9') c = gc();
    ll ret = 0;
    while(c>='0' && c<='9')
    {
        ret = (ret<<3) + (ret<<1) + c - 48;
        c = gc();
    }
    return ret;
}
char p[100005],q[100005];
int dp[100005][205];
int main()
{	
	int t,a,b,k,u,v;
	si(t);
	while(t--)
	{
		ss(p);ss(q);
		si(a);si(b);si(k);
		u=strlen(p);
		v=strlen(q);
		if(a==0) printf("0\n");
		else if(b==0)
		{
			if(abs(u-v)>k) printf("-1\n");
			else printf("%d\n",abs(u-v)*a);
		}
		else{
			if(abs(u-v)>k) printf("-1\n");
			else{
				f(i,0,u) f(j,0,2*k) dp[i][j]=inf;
				f(i,k,2*k) dp[0][i]=(i-k)*a;
				//f(i,0,k) dp[i][k]=i*a;
				f(i,1,u)
				{
					f(j,0,2*k)
					{
						if((i+j-k)<0 || (i+j-k)>v){dp[i][j]=inf;continue;}
						if(i+j-k==0){dp[i][j]=i*a;continue;}	
						int pos=i+j-k-1;
						dp[i][j]=dp[i-1][j]+(p[i-1]!=q[pos])*b;
						if(j<2*k) dp[i][j]=min(dp[i][j],dp[i-1][j+1]+a);
						if(j>0) dp[i][j]=min(dp[i][j],dp[i-1][j-1]+a);

					}
				}
				if(dp[u][v-u+k]>k) printf("-1\n");
				else printf("%d\n",dp[u][v-u+k]);
			}
		}
	}
	return 0;
}
