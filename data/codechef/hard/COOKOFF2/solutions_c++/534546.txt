#include <math.h>
#include <stdio.h>
#include <memory.h>
int i,j,ee,k,t,n,b[105],s[105];
double f[105][105][105],r,x,y;
int main() {
  scanf("%d",&t);
  while (t--) {
    scanf("%d",&n);
    for (i=0; i<n; i++) {
      scanf("%d",&b[i]);
      s[i+1]=s[i]+b[i];
    }
    memset(f,0,sizeof(f));
    f[0][0][0]=1;
    for (i=0; i<=n; i++) for (j=0; j<=i; j++) for (k=0; k<=s[i]; k++) if (f[i][j][k]>0) {
      if (i==n) continue;
      f[i+1][j][k]+=f[i][j][k];
      f[i+1][j+1][k+b[i]]+=f[i][j][k];
    }
    for (r=0,j=1; j<n; j++) {
      for (x=y=k=0; k<=s[n]; k++) if (f[n][j][k]>0) {
        x+=fabs(s[n]-k-k)*f[n][j][k];
        y+=f[n][j][k];
      }
      if (y>0) r+=x/y;
    }
    printf("%.6lf\n",double(r/(n-1)));
  }
  return 0;
}
