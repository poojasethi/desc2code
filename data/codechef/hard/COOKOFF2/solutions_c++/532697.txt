#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 101

int cs,n,i,z,j;
double s,d;
int A[lim];
double M[lim][lim][lim],F[lim],C[lim][lim];
bool B[lim][lim][lim];

double solve(int i,int j,int k)
{
  double &res=M[i][j][k];
  
  if(!B[i][j][k]) 
  {
    if(i==n)
      res=!j && !k;
    else
    {
      res=solve(i+1,j,k);
      if(j>=A[i] && k)
        res+=solve(i+1,j-A[i],k-1)*k/(n+1-k);
    }
    B[i][j][k]=1;
    //printf("%d %d %d %lf\n",i,j,k,res);
  }
  return res;
}
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d",&n);
    for(i=z=0;i<n;i++)
    {
      scanf("%d",&A[i]);
      z+=A[i];
    }
    memset(B,0,sizeof B);
    for(i=1,s=0;i<n;i++)
      for(j=1;j<z;j++)
      {
        s+=solve(0,j,i)*abs(z-2*j);
      }
    printf("%.6lf\n",s/(n-1));
  }  
  return 0;
}
