#include<cstdio>
#include<cstdlib>
#include<cstring>

typedef long double LD;

#ifndef ONLINE_JUDGE
#   define gc getchar
#else
#   define gc getchar_unlocked
#endif
inline void fps(int &x)
{
    register int c = gc();
    x = 0;
    //int neg = 0;
    for(;((c<48 || c>57) /*&& c != '-'*/);c = gc());
    //if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    //if(neg) x=-x;
}

#define EPS 1e-6
#define MX 105
int tok[MX], C;
LD cnt[MX][MX], Cnk[MX][MX];
LD f[MX], fact[MX];

int main() {
	int t;
	
	f[1] = 1;
	fact[0] = fact[1] = 1;
	for(int i = 2; i < MX; ++i) {
		f[i] = i * (i - 1) / 2 * f[i - 1];
		fact[i] = fact[i - 1] * i;
	}
	
	Cnk[0][0] = 1;
	for(int i = 1; i < MX; ++i) {
		Cnk[i][0] = Cnk[i][i] = 1;
		for(int j = 1; j < i; ++j)
			Cnk[i][j] = Cnk[i - 1][j] + Cnk[i - 1][j - 1];
	}
	
	fps(t);
	while(t--) {
		fps(C);
		int S = 0;
		for(int i = 0; i < C; ++i) {
			fps(tok[i]);
			S += tok[i];
		}
		memset(cnt, 0, sizeof(cnt));
		for(int i = 0; i <= C; ++i) for(int j = 0; j <= S; ++j)	cnt[i][j] = 0;
		cnt[0][0] = 1;
		for(int k = 0; k < C; ++k)
			for(int i = C - 1; i >= 0; --i)
				for(int j = 0; j <= S; ++j)
					if(cnt[i][j] > EPS) {
						cnt[i + 1][j + tok[k]] += cnt[i][j];					
						//printf(">>%d %d %lf\n", i + 1, j + tok[k], (double)cnt[i + 1][j + tok[k]]);
					}

		double tot = 0;
		for(int i = 1; i <= C; ++i) {
			for(int j = 1; j <= S; ++j) {
				if(cnt[i][j] > EPS && j <= (S - j)) {
					//printf("%d %d : %lf\n", i, j, (double)cnt[i][j]);
					tot += (double)f[i] * f[C - i];
				}
			}
		}
		//printf("TOT: %lf\n", tot);
		
		LD ans = 0;
		for(int i = 1; i < C; ++i) {
			for(int j = 1; j < S && j < (S - j); ++j) {
				if(cnt[i][j] < EPS) continue;
				ans += abs(S - 2 * j) * f[i] * f[C - i] / f[C] * cnt[i][j] * Cnk[C - 2][i - 1];
			}
		}
		
		//printf("%lf\n", (double)ans);
		printf("%.06Lf\n", ans);
	}
}
