#include<stdio.h>
#include<string.h>
#include<iostream>
#include<math.h>

using namespace std;

#define SZ 105

long double ncr[SZ][SZ];
long double memo[SZ][SZ];
int t[SZ];

int abs(int x)
{
	if(x < 0)
		return (-x);
	return x;
}

int main()
{
	int i, j;
	for(i = 2; i < SZ; i++)
	{
		for(j = 1; j < i; j++)
		{
			if(j == 1)
			{
				ncr[i][j] = 1.0 / i;
			}
			else
			{
				ncr[i][j] = ncr[i][j - 1] * j / (i - j + 1);
			}
		}
	}
	int n, inp, tot, k;
	scanf("%d", &inp);
	while(inp--)
	{
		scanf("%d", &n);
		tot = 0;
		memset(memo, 0, sizeof(memo));
		for(i = 0; i < n; i++)
		{
			scanf("%d", &t[i]);
			tot += t[i];
			for(j = i + 1; j > 0; j--)
			{
				for(k = tot; k >= t[i]; k--)
				{
					if(j == 1)
					{
						if(k == t[i])
							memo[j][k] += 1.0;
					}
					else
					{
						memo[j][k] += memo[j - 1][k - t[i]];
					}
				}
			}
		}
		long double ans = 0.0;
		for(i = 1; i < n; i++)
		{
			for(j = 1; j < tot; j++)
			{
				ans += (memo[i][j] * abs(tot - 2 * j) * ncr[n][i] / (n - 1));
			}
		}
		printf("%.6Lf\n", ans);
	}
	return 0;
}
