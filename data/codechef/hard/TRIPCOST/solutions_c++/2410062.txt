#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#include <set>
#include <map>

using namespace std;

int c[100010], d[100010];
int n, D;

int f(int C)
{
    int dd;
    int res = 0;
    for(int i = 1 ; i <= n ; ++i)
    {
        ++res;
        dd = D;
        int p = i-1;
        int cp = i-1;
        while(i <= n)
        {
            if(dd >= d[i])
            {
                //cout << c[i] << " " << C << "\n";
                if(c[i] <= C || i == n)
                    p = i;
                dd -= d[i];
                ++i;
            }
            else
                break;
        }
        if(p == n) break;
        if(cp == p) return 99999999;
        i = p;
    }
    return res;
}

int main()
{
    int t;
    cin >> t;
    for(int q = 0 ; q < t ; ++q)
    {
        cin >> n >> D;
        for(int i = 1 ; i <= n ; ++i)
            cin >> d[i] >> c[i];
        c[0] = 0;
        c[n] = 0;

        int l = 1, r = n+1, md;

        int MN = f(n+1);
        while(r-l > 1)
        {
            md = (l+r) >> 1;
            if(f(md) > MN)
                l = md;
            else
                r = md;
        }
        if(f(l) > MN) ++l;

        cout << MN << " " << l << "\n";
    }
    return 0;
}
