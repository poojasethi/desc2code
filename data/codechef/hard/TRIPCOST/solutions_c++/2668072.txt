#include <iostream>
#include <vector>
#include <algorithm>
#include <math.h>
using namespace std;

typedef long long ll;
#define maxn 2002
#define f first
#define s second
typedef long long ll;
typedef pair<ll,ll>par;
ll t, n, d;
par num[100009];
int check(ll v)
{
	ll ini = 0;
	ll l = 0;
	while(ini < n)
	{
		ll cost = 0;
		ll pos = -1;
		ll i = ini + 1;
		for(; i <= n + 1 && cost <= d; i++)
		{
			if(num[i].second <= v && cost + num[i].first <= d)
				pos = i;
			cost += num[i].f;
		}
		if(pos == -1)
			return -1;

		ini = pos;
		l++;
	}
	return l;
}

int main()
{
	cin >> t;
	while(t--)
	{
		cin >> n >> d;
		for(ll i = 1; i <= n ;i++)
			cin >> num[i].first >> num[i].second;

		ll mn = check(n + 1);

		ll ini = 0;
		ll fin = n + 1;
		while(fin - ini > 1)
		{
			ll middle = (fin + ini) >> 1;
			if(mn == check(middle))
				fin = middle;
			else
				ini = middle;
		}

		cout << mn << " " << fin << endl;
		par aux = make_pair(0,0);
		fill(num, num + n + 2, aux);
	}
	return 0;
}
