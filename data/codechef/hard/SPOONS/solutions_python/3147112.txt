#!/usr/bin/env python

import sys
from math import factorial

def bin_coef(n, r):
    return factorial(n)//(factorial(r)*factorial(n-r))

def find_maxes():
    maxes = []
    for x in xrange(1,65):
        maxes.append((x, bin_coef(x,x//2)))
    return maxes
max_cache = find_maxes()

def find_min_cities(N):
    for C, n in max_cache:
        if n >= N:
            return C
    return 0
        
if __name__ == '__main__':
    T = int(sys.stdin.readline())

    for x in xrange(0,T):
        N = int(sys.stdin.readline())
        print "%d" % find_min_cities(N)
