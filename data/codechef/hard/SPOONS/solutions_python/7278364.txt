def ncr(a,b):
    A=1
    B=1
    for i in range (a-b+1,a+1):
        A*=i;
    for j in range (1,b+1):
        B*=j;
    return A/B
t=input()
comb=[]
for i in range (0,70):
    comb.append(ncr(i,int(i/2)))
while(t):
    n=int(input())
    for i in range (2,70):
        if(n<=comb[i]):
            print i
            break
    t=t-1
