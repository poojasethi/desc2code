#include <iostream>
using namespace std;
long long int ncr[70][70];
void ncrc()
{
    int i,j;
    for(i=0;i<=70;i++)
    {
        for(j=0;j<=i;j++)
        {
            if(j==0){ncr[i][j]=1;}
            else if(j==1)
            {
                ncr[i][j]=i;
            }
            else
            {
                ncr[i][j]=ncr[i-1][j]+ncr[i-1][j-1];
            }
        }
    }
    
}
int main(int argc, const char * argv[])
{

    int t;
    long long int n;
    int i;
    ncrc();
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=1;i<=70;i++)
        {
            if(ncr[i][i/2]>=n)
            {
                cout<<i<<endl;
                break;
            }
        }
    }
    
    return 0;
}
