#include <bits/stdc++.h>
#include <iostream>
#include <cstdio>
using namespace std;

int main(){
	map<long long int, long long int> mappings;
	long long int value = 1;
	long long int curr = 2;
	mappings.insert(make_pair(1, 1));
	while(curr <= 64){
		if(curr % 2 == 0){
			value = value * 2;
		}
		else{
			long long int denom = curr/2 + 1;
			value = (value/denom) * curr;
		}
		mappings.insert(make_pair(value, curr));
		// cout << curr << " " << value << endl;
		curr++;
	}

	int t;
	cin >> t;
	while(t--){
		long long int n;
		scanf("%lld", &n);
		map<long long int, long long int> :: iterator it = mappings.lower_bound(n);
		printf("%lld\n", it->second);
	}
}
