#include<iostream>
#include<vector>
using namespace std;
int main()
{
    vector<vector<long long> > ncr(100,vector<long long>(100,0));
    for(int i=1;i<100;i++)
    for(int j=0;j<=i;j++)
    {
            if(j==0 || j==i) ncr[i][j]=1;
            else
            {
                ncr[i][j]=ncr[i-1][j-1]+ncr[i-1][j];
            }
    }
    
    int T;
    cin>>T;
    while(T--)
    {
              long long N;
              cin>>N;
              int i;
              for(i=1;ncr[i][i/2]<N;i++);
              cout<<i<<endl;
    }
    //system("pause");
}
