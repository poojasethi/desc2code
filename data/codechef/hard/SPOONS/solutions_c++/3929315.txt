#include<stdio.h>
#define lli long long int
long long int ncr[100][100]={0};
long long int nc2[100]={0};

int main()
{
	int i=0,j,t;
	lli n;
	ncr[0][0]=1;
	for(i=1;i<100;i++)
	{
		for(j=0;j<=i;j++)
		{
			if(j==i || j==0)ncr[i][j]=1;
			else ncr[i][j]=ncr[i-1][j-1]+ncr[i-1][j];
		}
		nc2[i]=ncr[i][i/2];
	}
	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld",&n);
		for(i=0;nc2[i]<n;i++);
		printf("%d\n",i);
	}
	
	return 0;
}
