#include<bits/stdc++.h>
using namespace std;
main()
{
	long long comb[100][100],req[100];
	int i,j;
	for(i=0;i<100;i++)
	{
		for(j=0;j<=i;j++)
		{
			if(j==0 || j==i)comb[i][j]=1;
			else comb[i][j]=comb[i-1][j]+comb[i-1][j-1];
		}
		req[i]=comb[i][i/2];
	}
	int tc;
	scanf("%d",&tc);
	while(tc--)
	{
		long long n;
		scanf("%lld",&n);
		for(i=0;req[i]<n;i++);
		printf("%d\n",i);
	}
}
