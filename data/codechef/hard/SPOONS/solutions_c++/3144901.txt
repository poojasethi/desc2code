void read_int(LLU &ret) {
    char c = gc();
    int neg = 0;
    while(c<'0' || c>'9') c = gc();
    ret = 0;
    if(c=='-')
    {
        neg = 1;
        c = gc();
    }
    while(c>='0' && c<='9') {
        ret = (ret<<1) + (ret<<3) + c - 48;
        //ret = 10 * ret + c - 48;
        c = gc();
    }
    if(neg)
        ret = -ret;
}
