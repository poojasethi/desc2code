#include<cstdio>
#include<cstring>
#include<cstdlib>

using namespace std;

typedef unsigned long long _int64;

int main()
{
	_int64 C[65][65];
	memset(C,0,sizeof(C));
	for(int i=0;i<65;++i)
		for(int j=0;j<=i;++j)
			if(j==0)	C[i][j]=1;
			else		C[i][j]=C[i-1][j-1]+C[i-1][j];
	
	int T;
	scanf("%d",&T);
	while(T--)
	{
		_int64 N;
		scanf("%llu",&N);
		for( int ans=2;ans<65;++ans)
			if(C[ans][ans/2]>=N)
			{
				printf("%d\n",ans);
				break;
			}
	}
	
	return 0;
}
