long long dp[66][66];
long long lim[66];
void solve()
{
	for(int i=0;i<=65;i++)dp[i][0]=1;
	for(int i=1;i<=65;i++)
		for(int j=1;j<=i;j++)
			dp[i][j]=dp[i-1][j-1]+dp[i-1][j];
}
int main()
{
	int t;
	cin>>t;
	solve();
	lim[0]=0,lim[1]=1,lim[2]=2,lim[3]=3,lim[4]=6;
	for(int i=5;i<=65;i++)
			lim[i]=dp[i][i/2];
	while(t--)
	{
		long long n;
		cin>>n;
		for(int i=0;i<=65;i++)
		{
			if(lim[i]>=n)
			{
				cout<<i<<endl;break;
			}
		}
	}
	return 0;
}
