#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
long long int c[101][101];
long long comb(long long int n,long long int r)
{
    int i,j;
    for(i=0;i<100;i++)
        for(j=0;j<=i;j++)
    {
        if(i==0)
            c[i][j]=0;
        else if(j==0)
            c[i][j]=1;
            else if(i==j)
                c[i][j]=1;
        else
        c[i][j]=c[i-1][j]+c[i-1][j-1];
    }
    return c[n][r];

}
int main()
{
    int t;
    long long int n,i;
    cin>>t;
    comb(100,50);
    while(t--)
    {
        cin>>n;
        for(i=2;i<100;i++)
        {
            if(c[i][i/2]>=n)
            break;
        }
        cout<<i<<endl;
    }
    return 0;
}
