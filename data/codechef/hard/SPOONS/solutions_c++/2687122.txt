using namespace std;
 
typedef long long int64;
 
int64 c[128][128];
 
int main()
{
	memset(c,0,sizeof(c));
	for (int i=0;i<128;i++) for (int j=0;j<=i;j++) c[i][j]=(j==0)?1:c[i-1][j-1]+c[i-1][j];
	int t;
	for (scanf("%d",&t);t>0;t--)
	{
		long long int n;
		cin>>n;
		int m=2;
		for (;c[m][m/2]<n;m++);
		printf("%d\n",m);
	}
	return 0;
}
