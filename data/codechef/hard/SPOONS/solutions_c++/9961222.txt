#include <iostream>
using namespace std;

long long ans[100];

void precompute(){
	ans[0] = 1;
	ans[1] = 1;
	ans[2] = 2;
	int i, prev = 2;
	for(i=3; i<=64; i++){
		ans[i] = ans[i-1];
		if(i%2==1){
			ans[i] /= prev;
			ans[i] *= i;
		}
		else{
			ans[i] *= 2;
			prev = i/2 + 1;
		}
	}
	
	/*for(i=1; i<65; i++){
		cout << ans[i] << endl; 
	}*/
}

int main() {
	// your code goes here
	precompute();
	int t, output;
	long long int n;
	cin >> t;
	while(t--){
		cin >> n;
		for(int i=1; i<65; i++){
			if(ans[i] >= n){
				output = i;
				break;
			}
		}
		cout << output << endl;
	}
	return 0;
}
