#include <iostream>

using namespace std;
typedef long long ll;

ll n;
int t;
ll comb[1000000];

void precompute()
{
	comb[1] = 0;
	comb[2] = 2;
	for(int i = 3;;i++)
	{
		if(i % 2 == 0)
		{
			comb[i] = comb[i - 1];
			comb[i]*=(ll)2;
		}
		else
		{
			comb[i] = comb[i - 1];
			comb[i] /= (ll)((i/2) + 1);
			comb[i] *= (ll)i;
		}
		if(comb[i] < 0)break;
	}
}
int main()
{
	cin >> t;
	precompute();
	while(t--)
	{
		cin >> n;
		if(n == 1 || n == 2)
		{
			cout << n << endl;
			continue;
		}
		for(int i = 2; ; i++)
		{
			if(comb[i] >= n)
			{
				cout << i << endl;
				break;
			}
		}

	}
	return 0;
}
