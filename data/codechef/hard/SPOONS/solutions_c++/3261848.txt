LL C[SZ][SZ] = {0};

void combi () {
	FORALL(i, SZ) {
		FORALL(j, i+1) {
			if (!j) C[i][j] = 1;
			else if (j == 1) C[i][j] = i;
			else C[i][j] = C[i-1][j] + C[i-1][j-1];
		}
	}
}

int main() {
	int t;
	combi();
	cin >> t;
	while (t--) {
		LL n;
		cin >> n;
		int i = 1;
		while (i < SZ && C[i][i/2] < n) i++;
		cout << i << endl;
	}
	return 0;
