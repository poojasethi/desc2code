#include<stdio.h>
#include<string.h>

int main()
{
	long long foo[100], N;
	int i, T;
	
	foo[0]=1;
	foo[1]=0;
	foo[2]=1;
	int prev=1;
	
	for(i=3; i<=64; i++)
	{
		foo[i]=foo[i-1];
		
		if(i%2)
		{
			foo[i]/=prev;
			foo[i]*=i;
		}
		else
		{
			foo[i]*=2;
			prev=i/2+1;
		}
	}
	
	scanf("%d",&T);
	
	while(T--)
	{
		scanf("%lld",&N);
		
		if(N==1)
		printf("1\n");
		else if(N==2)
		printf("2\n");
		
		else
		{
			for(i=1; i<=64; i++)
			{
				if(foo[i]>=N)
				break;
			}
			
			printf("%d\n",i);
		}
	}
	return 0;
}
