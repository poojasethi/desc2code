#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<vector>
#define mod 1000000007ll
using namespace std;
long long T,n,d,res[65],C[65][65];
void pre()
{
	C[0][0]=1;
	for(int i=1;i<=64;i++)
	{
		C[i][0]=1;
		d=1;
		for(int j=1;j<=i;j++)
		{
			C[i][j]=C[i-1][j-1]+C[i-1][j];
			d=max(d,C[i][j]);
		}
		res[i]=d;
	}
	//for(int i=1;i<=30;i++)cout<<res[i]<<" ";cout<<endl;
}

int main()
{
	pre();
  	scanf("%lld",&T);
	while(T--)
	{
		scanf("%lld",&n);
		if(n<=4)printf("%lld\n",n);
		else{
			for(int i=1;i<65;i++)
			if(res[i]>=n)
			{
				printf("%d\n",i);
				break;
			}
		}		
	}
	return 0;
}
