#include <iostream>
#include <cstdio>
using namespace std;
typedef long long llint;

llint C[100][100];

int main() {
	C[0][0] = 1;
	for (int i = 1; i <= 70; i++) {
		C[i][0] = 1;
		C[i][i] = 1;
		for (int j = 1; j < i; j++) {
			C[i][j] = C[i-1][j-1] + C[i-1][j];
		}
	}
	
	int t;
	cin >> t;
	while (t--) {
		llint n;
		int res;
		cin >> n;
		
		for (int i = 0; i <= 70; i++)
			if (C[i][i/2] >= n) {
				res = i;
				break;
			}
		
		cout << res << "\n";
	}
	
	return 0;
}
