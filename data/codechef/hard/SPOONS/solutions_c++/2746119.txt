#include <cstdio>
using namespace std;

typedef unsigned long long ull;

const int MAX = 65;

int main()
{
	ull a[MAX][MAX],n;
	for(int i=0;i<MAX;i++)
	{
		a[0][i] = 0;
		a[i][0] = 1;
	}
	for(int i=1;i<MAX;i++)
	{
		for(int j=1;j<i+1;j++)
		{
			a[i][j] = a[i-1][j-1]+a[i-1][j];
		}
	}
	int l,r,mid,t;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%llu",&n);
		l = 0;
		r = MAX-1;
		while(r-l>1)
		{
			mid = (l+r)/2;
			if(a[mid][mid/2]==n)	break;
			else if(a[mid][mid/2]>n)	r = mid;
			else l = mid;
		}
		mid = (l+r)/2;
		if(a[mid][mid/2]<n)	printf("%d\n",mid+1);
		else printf("%d\n",mid);
	}
	return 0;
}
