#include<cstdio>
#include<iostream>
using namespace std;
long long int dp[66][33],arr[65];
long long int ncr(int n,int r)
{
 if(n==r)
 return dp[n][r]=1;
 else if(r==0)
 return dp[n][r]=1;
 else if(r==1)
 return dp[n][r]=(long long int)n;
 else if(dp[n][r])
 return dp[n][r];
 else
 return dp[n][r]=ncr(n-1,r)+ncr(n-1,r-1);
}
int main()
{
    int t;
    long long int n;
    scanf("%d",&t);
    for(int i=0;i<65;i++)
    arr[i]=ncr(i,i/2);
    while(t--)
    {
     scanf("%lld",&n);
     for(int i=0;i<65;i++)
     {
     if(arr[i]>=n)
     {
     printf("%d\n",i);
     break;
     }
     }
    }
}
