#include<stdio.h>


long long arr[75][75] ;
int main(){

	for(int i = 1 ; i<=70 ; i++)
		arr[i][1] = i ;

	for(int i = 2 ; i<=70 ; i++)
		for(int j = 2 ; j<=i ; j++)
			arr[i][j] = arr[i-1][j] + arr[i-1][j-1] ;

	long long ans [75] ;

	for(int i = 2 ; i<=70 ; i++)
		ans[i] = arr[i][i/2] ;

	int T ;
	scanf("%d",&T) ;
	long long N ;
	for(int i = 0 ; i<T ; i++){
	
		scanf("%lld",&N) ;
		int j ;
		for(j = 2 ; j<=70 && N > ans[j] ; j++) ;
		printf("%d\n",j) ;
	}
	return 0 ;
}
