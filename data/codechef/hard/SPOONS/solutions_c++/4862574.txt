#include<cstdio>
using namespace std;
long long int vd[100];
void calc()
{
  vd[0] = 1;
  vd[1] = 0;
  vd[2] = 1;
  long long int prev = 1;
  for(long long int i=3;i<=64;i++)
  {
    vd[i]=vd[i-1];
    if(i%2==1)
    {
      vd[i]/=prev;
      vd[i]*=i;
    }
    else
    {
      vd[i]*=2;
      prev = i/2 + 1;
    }
  }
}

int main()
{
  calc();
    int t;
   long long int n;
   scanf("%d",&t);
   int ans;
   while(t--)
   {
     scanf("%lld",&n);
     if(n==2)
     {
	 printf("2\n");
	 continue;
     }
     for(int i=1;i<=64;i++)
       if(vd[i]>=n)
      {
         ans = i;
         break;
      }
     printf("%d\n",ans);
   }
  return 0;
}
