#include <iostream>
#include <cstdio>

using namespace std;

//#define MAX 1000000000000000000LL

long long T[200][200];
long long maxi[200];
#define MAX 1000000000000000000LL

int C() {
	int i;
	int j;
	for (i = 0; ; i++) {
		T[i][0] = 1;
		maxi[i]= 1;
		for (j = 1; j <= i; j++) {
			T[i][j] = T[i-1][j-1] + T[i-1][j];
			maxi[i] = (maxi[i] > T[i][j]) ? maxi[i] : T[i][j];
//			cout << T[i][j] << " ";
		}
		if(maxi[i]  >= MAX) break;
	//	cout << " maxi " << maxi[i]  << endl;
		
	}
	return 1;
}

int main() {
	int i;
	int T;
	int k;
	long long N;
	C();
	cin >> T;
	for (i = 0; i < T; i++) {
		cin >> N;
		for (k = 0; ; k++) {
			if(maxi[k] >= N) {
			       	cout << k << endl;
				break;
			}
		}
	}
	
	return 0;
}
