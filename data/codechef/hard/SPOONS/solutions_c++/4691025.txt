#include <stdio.h>

#define MAXN 70

long long p[MAXN];

void calcP() {
	p[0] = 1;
	
	int d = 0;
	
	for(int i=1; i<MAXN; i++) {
		if(i % 2)
			d++;
		
		p[i] = p[i-1];
		
		if(p[i] % d == 0) {
			p[i] /= d;
			p[i] *= i;
		}
		else {
			p[i] *= (i/d);
		}
	}
}

int main() {
	
	calcP();
	
	int t;
	scanf(" %d", &t);
	
	for(int i=0; i<t; i++) {
		long long n;
		scanf(" %lld", &n);
		
		int k=1;
		while(k <= 64 && p[k] < n) k++;
		printf("%d\n", k);
	}
}
