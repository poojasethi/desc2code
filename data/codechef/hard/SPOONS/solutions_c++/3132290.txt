#include <iostream>
using namespace std;

long long c[100];

int numdig(long long n) {
    int res = 0;
    while(n) {
        res++;
        n/=10;
    }
    return res;
}

int init() {
    c[0] = 1;
    c[1] = 1;
    c[2] = 2;
    long long prev = 2;
    int i;
    for(i=3;numdig(c[i-1])<=18;++i) {
        c[i] = c[i-1];
        if(i%2) {
            c[i]/=prev;
            c[i]*=i;
        } else {
            c[i]*=2;
            ++prev;
        }
    }
    return i;
}

int main() {
    int limit = init();
    int t;
    cin >> t;
    while(t--) {
        long long n;
        cin >> n;
        for(int i=1;i<limit;++i) {
            if(c[i]>=n) {
                cout << i << endl;
                break;
            }
        }
    }
    return 0;
}
