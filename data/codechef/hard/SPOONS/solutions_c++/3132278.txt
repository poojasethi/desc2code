#include <iostream>
using namespace std;

long long c[65];

void init() {
    c[0] = 1;
    c[1] = 1;
    c[2] = 2;
    long long prev = 2;
    long long i;
    for(i=3;i<65;++i) {
        c[i] = c[i-1];
        if(i%2) {
            c[i]/=prev;
            c[i]*=i;
        } else {
            c[i]*=2;
            ++prev;
        }
    }
}

int main() {
    init();
    int t;
    cin >> t;
    while(t--) {
        long long n;
        cin >> n;
        for(int i=2;i<65;++i) {
            if(c[i]>=n) {
                cout << i << endl;
                break;
            }
        }
    }
    return 0;
}
