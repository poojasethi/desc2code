dictionary = {}
def byte(n):
	cur_max = n
	if n in dictionary:
		cur_max = dictionary[n]
	else:
		if n != 0:
			x = byte(n/2)
			y = byte(n/3)
			z = byte(n/4)
			cur_max = max(n, (x + y + z))
			dictionary[n] = cur_max
	return cur_max
t = 1
while t <= 10:
	x = raw_input()
	if x == "":
		break
	else:
		print byte(int(x))
	t += 1
