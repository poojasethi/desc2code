solutions = dict()
    
def solve(n):
	if n < 12:
    		return n
     
	if solutions.has_key(n):
    		return solutions[n]
     
    	else:
    		ans = solve(n/2) + solve(n/3) + solve(n/4)
    		solutions[n] = ans
    		return ans

    	for x in xrange(10):
    		n = int(raw_input())
    		print solve(n)
