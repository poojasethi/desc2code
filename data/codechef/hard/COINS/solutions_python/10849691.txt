import fileinput

max_values = {}
def compute(inp):
    sol = inp 

    if inp in max_values:
        sol = max_values[inp]
    else:
        if not inp == 0:
            coin1 = compute(int(inp/2))
            coin2 = compute(int(inp/3))
            coin3 = compute(int(inp/4))

            sol = max(inp, coin1 + coin2 + coin3)
            max_values[inp] = sol
    return sol

for line in fileinput.input():
    n = int(line)
    solve = compute(n)
    print solve
