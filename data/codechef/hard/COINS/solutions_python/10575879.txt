import fileinput

marray = {}
def answer(n):
    s = n
    if n in marray:
        s = marray[n]
    else:
        if(n!=0):
            t = answer(int(n/2))
            th = answer(int(n/3))
            f = answer(int(n/4))
            s = max(n, t+th+f)
            
            marray[n] = s
    
    return s
