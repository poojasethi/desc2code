
l = list()

while True: 
	try:
		l.append(int(raw_input()))
	except:
		break


dp = {0:0}
m = max(l)

def gdp(i):
	if dp.has_key(i):
		return dp[i]

	t = gdp(i/2) + gdp(i/3) + gdp(i/4)
	if t < i:
		dp[i] = i
	else:
		dp[i] = t

	return dp[i]



for i in l:
	try:
		print gdp(i)
	except:
		print "Vidhya"
