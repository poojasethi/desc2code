def rec(n):
    if(n not in coins):
        if(n < 12):
            coins[n] = n
            return coins[n]
        else:
            coins[n] = rec(n/2) + rec(n/3) + rec(n/4)
            return coins[n]
    else:
        return coins[n]

coins = {}

    try:
        n = int(raw_input())
    except(EOFError):
