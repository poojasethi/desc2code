
max_dict = {}


def max_earnable(n):

    if n <= 1:
        return n
    elif max_dict.has_key(n):
        return max_dict[n]
    else:
        best = max(n, max_earnable(n/2) + max_earnable(n/3) + max_earnable(n/4))
        max_dict[n] = best
        return best


while True:

    try:
        n = input('')
    except EOFError:
        break

    print(max_earnable(n))
