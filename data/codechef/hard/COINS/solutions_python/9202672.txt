all_ = {0:0, 1:1}

def coins(num):
    try:
        return all_[num]
    except:
        div_sum = coins(num/2) + coins(num/3) + coins(num/4)
        all_[num] = max(div_sum, num)
        return all_[num]

while True:
    try:
        input_ = int(raw_input())
        print coins(input_)
    except Exception, e:
        break
