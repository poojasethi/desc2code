cache_ = {}

def exchange(input__):
    if input__ in cache_:
        return cache_[input__]
    if input__ < 12:
        return input__
    ans_ = exchange(int(input__ / 2)) + exchange(int(input__ / 3)) + exchange(int(input__ / 4))
    cache_[input__] = ans_
    return ans_

while True:
    try:
        input_ = int(raw_input())
        print exchange(input_)
    except Exception, e:
        break
