maxval = {0:0}
def maxv(n):
    if n in maxval:
        return maxval[n]
    else:
        nb2 = maxv(int(n/2))
        nb3 = maxv(int(n/3))
        nb4 = maxv(int(n/4))
        maxval[n] = max(n,nb2 + nb3 + nb4)
        return maxval[n]
    
import sys
while True:
    n = sys.stdin.readline()
    if n == '':
        break
    print maxv(int(n))
