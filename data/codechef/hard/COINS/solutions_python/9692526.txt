import sys
from math import floor

# cache in hash
cache = {0:0}

def break_down (coin):
    if coin in cache:
        return cache[coin]
    else :

        sol = max(coin, break_down(int(coin/2)) + break_down(int(coin/3)) + break_down(int(coin/4)))
        cache[coin] = sol
        return sol
