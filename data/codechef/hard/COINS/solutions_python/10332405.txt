import fileinput

maxs = {}

def solve(n):
  if n==0:
    return 0
  else:
    if n in maxs:
      return maxs[n]
    else:
      c1 = solve(n/2)
      c2 = solve(n/3)
      c3 = solve(n/4)
      res = max(n,c1+c2+c3)
      maxs[n] = res 
      return(res)
        
for line in fileinput.input():
  cv = int(line)
  print solve(cv)
