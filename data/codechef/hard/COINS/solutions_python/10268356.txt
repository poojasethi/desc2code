seq={}
def check(value):
    maxvalue=value
    if value in seq:
         maxvalue = seq[value]   
    else:    
        if value!= 0:
            coin1=check(value/2)
            coin2=check(value/3)
            coin3=check(value/4)
            
            maxvalue=max(value,(coin1+coin2+coin3))
            seq[value]=maxvalue 
    return maxvalue      
tcase=1
while tcase<=10:
    value=raw_input()
    if str(value)=='':
        break
    else: 
        print(check(int(value)))
    tcase+=1
