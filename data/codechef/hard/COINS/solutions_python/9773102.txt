index = {0:0}
def maximumamount(n):
    if n in index:
       return index[n]
    else:
       if (n > (maximumamount(int(n/2)) + maximumamount(int(n/3)) + maximumamount(int(n/4)))):
          index[n] = n
          return index[n]
       else:
          index[n] = (maximumamount(int(n/2)) + maximumamount(int(n/3)) + maximumamount(int(n/4)))
          return index[n]

while True:
      try:
          n = int(input())
          print maximumamount(n)
      except:
          break
