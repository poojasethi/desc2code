import fileinput

solutions = {}


def solve_max(n):
    if n < 12:
        return n
        
    if n in solutions:
        return solutions[n]
    
    
    total = solve_max(int(n/2))
    total += solve_max(int(n/3))
    total += solve_max(int(n/4))
    solutions[n] = total
    return total


for line in fileinput.input():
    if not line:
        continue
    n = int(line)
    result = solve_max(n)
    print result
