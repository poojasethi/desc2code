a={}
def solve(n):
    ans=n
    if n in a:
        ans=a[n]
    else:
        if n!=0:
            x=solve(n/2)
            y=solve(n/3)
            z=solve(n/4)
            ans=max(n,x+y+z)
            a[n]=ans
    return ans
 
acc = []
while True:
    try:
        acc.append(raw_input()) # Or whatever prompt you prefer to use.
    except EOFError:
        break
for i in range(len(acc)):
    acc[i]=int(acc[i])
    result=solve(acc[i])
    print result
