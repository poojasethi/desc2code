import math
#Cache already computed values
computed={}
def maxProfit(number):
	if(number in computed):
		return computed[number]
	if number == 0:
		return 0
	elif number == 1:
		return 1
	else:
		value = int(max(number,maxProfit(int(math.floor(number/2))) + maxProfit(int(math.floor(number/3))) + maxProfit(int(math.floor(number/4)))));
		computed[number] = value
		return value


while True:
    try:
        print maxProfit(int(input()))
    except:
        break
