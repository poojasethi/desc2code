Max= {}
def fun(a):
    if a in Max:
        max_val = Max[a] 
    elif a!=0:
        max_val = max(a, fun(a/2) + fun(a/3) + fun(a/4))
        Max[a] = max_val
    else:
        max_val = 0
    return max_val
            

test = 1
while test <= 10:
    a = raw_input()
    if a == '':
        break
    test += 1
    
    print fun(int(a))
