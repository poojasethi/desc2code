#!/usr/bin/env python
# -*- coding: utf-8 -*-

cache = {0: 0, 1: 1}
def best(n):
    if n in cache:
        return cache[n]
    res = max(n, best(n//2) + best(n//3) + best(n//4))
    cache[n] = res
    return res

if __name__ == "__main__":
    import sys
    for n in map(int, sys.stdin.readlines()):
        print best(n)
