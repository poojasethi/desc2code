import sys

class BytelandMoneyMaker(object):
	"""CodeChef gg"""
	
	def __init__(self):
		self.memoize = {} 

	def get(self,n):
		try:
			return self.memoize[n]
		except KeyError:
			return -1

	def make_money(self,n):
		ret = self.get(n)
		if ret != -1:
			return ret
		a = n / 2
		b = n / 3
		c = n / 4
		if a+b+c > n:
			ret = self.make_money(a) + self.make_money(b) + self.make_money(c)
		else:
			ret = n
		self.memoize[n] = ret
		return ret

money = BytelandMoneyMaker()

		print money.make_money(int(raw_input()))
