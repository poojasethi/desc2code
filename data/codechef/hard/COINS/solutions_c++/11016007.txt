#include <iostream>
#include <cstdio>
#include <cmath>
#include <map>

using namespace std;

map<int, long long> dp;

long long getbig(long long x)
{
	if (x == 0)
		return 0;
	if (dp[x] != 0)
		return dp[x];
	
	long long y = getbig(x/2) + getbig(x/3) + getbig(x/4);
	
	if (y > x)  // if the value is more than original, add it to dp
		dp[x] = y;
	else		// otherwise just save as it is
		dp[x] = x;	
	return dp[x];
}

int main()
{
	long long x;
	while (scanf("%lld", &x) == 1)
	{
		printf("%lld\n", getbig(x));
	}
}
