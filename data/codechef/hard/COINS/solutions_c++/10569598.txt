#include<iostream>
#include<stdio.h>
#include<map>
#include<algorithm>
using namespace std;
map<long long int ,long long int>m;
long long coins(long long int x)
{
    if(x==0)
        return x;
    else if(m[x]>0)
        return m[x];
    else
    {
        m[x]=max(x,(coins(x/2)+coins(x/3)+coins(x/4)));}
        return m[x];
}
int main()
{
long long int n;
while(scanf("%lld",&n)==1) printf("%lld\n",coins(n));
return 0;
}
