#include<iostream>
#include<map>
#include<cstdio>
#include<algorithm>
using namespace std;
map<long long int, long long int>m;
long long int f(long long int n){
    if ( n == 0)    return 0;
    if ( m[n] > 0 ) return m[n];
    else {  
        long long int c=max(n,(f(n/2)+f(n/3)+f(n/4)));
        m[n]=c;
        return m[n];
    }
}
int main(){
    long long int n;
    while (scanf("%lld",&n)>0)
        printf("%lld\n",f(n));
    return 0;
}
