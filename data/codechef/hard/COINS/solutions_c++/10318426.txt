#include<bits/stdc++.h>
using namespace std;

long long get_change(vector<int> &_vec,long long n){
	if(n==0)return 0;
	
	if(n<1000000 && _vec[n]!=-1){
		return _vec[n];
	}
	else{
		long long p1 = get_change(_vec,n/2);
		long long p2 = get_change(_vec,n/3);
		long long p3 = get_change(_vec,n/4);
		if(n<1000000)
		_vec[n] = p1+p2+p3>n?p1+p2+p3:n;
		if(p1+p2+p3>n)return p1+p2+p3;
		else return n;
	}
	
	
	
}
int main(){
	vector<int> _vec(1000005,-1);
	long long n;
	while(scanf("%lld",&n)!=EOF){
		cout<<get_change(_vec,n)<<endl;
	}
	return 0;
}
