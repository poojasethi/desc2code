#include<iostream>
#include<stdio.h>
#include<map>
using namespace std;

map<long long int,long long int> mp;

long long int getBest(long long int n)
{
    if(n<=10)
        return n;
    if(mp.count(n)!=0)
        return mp[n];
    if(n>=((n/2)+(n/3)+(n/4)))
    {
        mp.insert(pair<long long int,long long int>(n,n));
        return mp[n];
    }
    else
    {
        mp.insert(pair<long long int,long long int>(n,getBest(n/2)+getBest(n/3)+getBest(n/4)));
        return mp[n];
    }
}

int main()
{
    long long int n;
    while(scanf("%lld",&n)==1)
        cout<<getBest(n)<<endl;
}
