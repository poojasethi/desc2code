map<ll,ll> m;

ll rec(ll n)
{
    if(n==0)
    {
        return 0;
    }
    if(m[n])
    {
        return m[n];
    }
    return m[n]=max(n,rec(n/2)+rec(n/3)+rec(n/4));
}
