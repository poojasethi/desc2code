#include<iostream>
#include<cstdio>
using namespace std;
long long memory[10000000]={0};
long long count(long long n)
{
	if(n<12)
		return n;
		
	if(n==12)
		return 13;
		
	if(n<1000000 && memory[n] !=0) 		return memory[n];
    long long k =count(n/2)+count(n/3)+count(n/4);
    if(n<1000000)
     	memory[n]=k;
    return k;
}
int main()
{
	long long n=0;
	while(scanf("%lld", &n)!=EOF)
	{
		printf("%lld\n",count(n));
	}
	return 0;
}
