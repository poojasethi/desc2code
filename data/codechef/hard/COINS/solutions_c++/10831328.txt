#include <bits/stdc++.h>
using namespace std;

const long long N = 10000001;
long long dp[N];

long long bk(long long n){
	if(n < 12)
		return n;
	if(n < N){
		if(dp[n] != -1)
			return dp[n];
		return dp[n] = bk(n / 2) + bk(n / 3) + bk(n / 4);
	}
	return bk(n / 2) + bk(n / 3) + bk(n / 4);
}

int main(){
	long long n;
	fill(dp, dp + N, -1);
	while(scanf("%lld", &n) != EOF){
		printf("%lld\n", bk(n));
	}
	return 0;
}
