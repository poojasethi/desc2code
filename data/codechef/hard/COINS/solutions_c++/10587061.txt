#include<bits/stdc++.h>
#include<iostream>
using namespace std;
map<long long int,long long int>a;
long long int fun(long long int n)
{
	if(n==0||n==1)
		return(n);
	else if(!a[n]) 
		a[n]=(max(n,fun(n/2)+fun(n/3)+fun(n/4)));
	return(a[n]);
}
int main()
{
	long long int n;
	while(cin>>n)
	{
		cout<<fun(n)<<endl;
	}
}
