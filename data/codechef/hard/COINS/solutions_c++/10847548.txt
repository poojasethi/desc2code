#include <bits/stdc++.h>
using namespace std;
long long n;
long long a[10000004];

long long coin(long long x) {
    if (x <= 10000000) return a[int(x)];
    if (x > x/2 + x/3 + x/4 || (x == 0)) return x;
    return coin(x/2) + coin(x/3) + coin(x/4);
}

int main() {
    for (int i = 0; i < 12; i++) a[i] = i;
    for (int i = 12; i <= 10000000; i++) a[i] = max(1LL*i, a[i/2] + a[i/3] + a[i/4]);
    while (cin >> n) {
        cout << coin(n) << endl;
    }
}
