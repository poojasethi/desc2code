#include <bits/stdc++.h>

using namespace std;

map<long long,long long>mp;

long long int coins(long long int n)
{
    if(n==0)
        return 0;
    else
        if(mp[n]!=0)
            return mp[n];
        else
            {
        mp[n]=max(n,coins(n/2)+coins(n/3)+coins(n/4));
    }
    return mp[n];
}

int main()
{
      long long int b,n;
   int t;
   t=10;
   mp.clear();
   while(t--)
   {
      cin>>n;
      b=coins(n);
cout<<b<<"\n";
   }
    return 0;
}
