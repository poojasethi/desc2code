#include <iostream>
#include <cstring>
#include <map>
using namespace std;

long long int maxt(map<long long int,long long int> &arr,long int n)
{
    if(n==1||n==0)
        return n;
    else if(arr.find(n)!=arr.end())
        return arr[n];
    else
    {
        long long int t1=maxt(arr,n/2)+maxt(arr,n/3)+maxt(arr,n/4);
        arr[n]=t1>n?t1:n;
        return arr[n];
    }
}
int main()
{
    map<long long int,long long int> arr;
    long long int n;
    while(cin>>n)
    {
        cout<<maxt(arr,n)<<endl;
    }
    return 0;
}
