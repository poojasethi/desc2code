#include<iostream>
#include<cstdio>
#include<algorithm>
#define MAX(a,b) (a>b)?a:b
#define ull unsigned long long int
using namespace std;
ull rec(ull);
int memory[10000000]={0};
int main(){
	ull n=0;
	while(scanf("%lld",&n)!=EOF)
	     printf("%lld\n",max(n,rec(n)));
	return 0;
}

ull rec(ull n){
	if(n<12)
	    return n;
	if(n==12)
	    return 13;
	if(n<1000000)
	 if(memory[n]!=0)
	   return memory[n];
	ull k =rec(n/2)+rec(n/3)+rec(n/4);
	if(n<1000000)
	 memory[n]=k;
	 return k;	
}
