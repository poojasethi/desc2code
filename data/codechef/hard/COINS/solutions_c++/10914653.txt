            #include <iostream>
            #include <algorithm>
            #include <map>
            #include <stdio.h>
             
            using namespace std; 
            unsigned int getMaxSum(unsigned int n, map<unsigned int,unsigned int>& dp){
            	if(n<=4)
            		return n;
            	if(dp.find(n) != dp.end())
            	    return dp[n];
            	unsigned int maxi=n;
            	maxi = max(maxi, getMaxSum(n/2, dp)+getMaxSum(n/3, dp)+getMaxSum(n/4, dp));
            	dp.insert(make_pair(n,maxi));
            	return maxi;
            }
             
            int main(){
            	unsigned int n;
            	while(scanf("%ld",&n)!=EOF){
    	            	map<unsigned int,unsigned int> dp;
    	            	cout<<getMaxSum(n,dp)<<endl;
            	}
            }
