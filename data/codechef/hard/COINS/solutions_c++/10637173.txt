typedef long long int ll;

map<ll,ll>A;

ll f(ll n){
  if(n<12)return n;
  if(n == 12) return 13;
 
 	map<ll,ll>::iterator it = A.find(n);

 	ll temp;
    if(it!=A.end())
      temp =  it->S;
    else
      	temp = f(n>>1)+f(n/3)+f(n>>2);
  	A[n] = temp;
  	return max(temp,n);
}
