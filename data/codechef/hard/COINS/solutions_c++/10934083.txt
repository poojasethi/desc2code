#include <algorithm>
#include <iostream>
using namespace std;

long long memory[10000000] = {0};

long long getMaxCoin(long long n) {
	if(n < 12)
		return n;
	if(n == 12)
		return 13;
		
	if(n<10000000 && memory[n]!=0)
		return memory[n];
			
	
	long long k = getMaxCoin(n/2)+getMaxCoin(n/3)+getMaxCoin(n/4);
	if(k < 10000000)
		memory[n] = k;
		
	return k;
}

int main() {
	long long n;
	while(cin >> n) {
		cout << getMaxCoin(n) << "\n";
	}
	return 0;
}
