#include<bits/stdc++.h>
#include<map>
#define ll unsigned  long
#define max(a,b)   (((a)>(b))?a:b)
using namespace std;
map<ll,ll>v;

ll maxcoin(ll n)
{
    
    if(n==0)
      return n;
    if(v[n]>0)
     return v[n];
     
     else
     {
      
      v[n]=max(n,maxcoin(n/2)+maxcoin(n/3)+maxcoin(n/4));
     
        return v[n]; 
     }
     
}
   int  main()
   {
       unsigned int n;
       
       while((scanf("%d",&n)>0))
       {
           printf("%lu\n",maxcoin(n));
       }
       
       return 0;
   }
