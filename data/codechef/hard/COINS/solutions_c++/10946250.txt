#include<iostream>
#include<cstdio>
#include<map>
#include<algorithm>
#define ll long long
using namespace std;
 
map<ll , ll>M;
 
ll fn(ll n){
	if(n==0)
		return 0;
	ll res=M[n];
	if(res==0){
		res=max(n,fn(n/2)+fn(n/3)+fn(n/4));
		M[n]=res;
	}
	return res;
}
 
int main(){
	ll n;
 
	while(scanf("%lld",&n)!=EOF){
		printf("%lld\n",fn(n));
	}
return 0;
}
