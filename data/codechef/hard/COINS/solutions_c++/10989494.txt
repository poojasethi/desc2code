#include<bits/stdc++.h>
#define max(a,b) a>b?a:b
int a[100000]={0};
unsigned long int func(unsigned long int n)
{
	unsigned long int s=0;
	if(n<12)
	return n;
	else if(n>=0 && n<=100000)
	{
		if(a[n])
		return a[n];
		else
		{
			s=func(n/2)+func(n/3)+func(n/4);
			a[n]=max(s,n);
			return a[n];
		}
	}
	else
	{
		s=func(n/2)+func(n/3)+func(n/4);
		return max(s,n);
	}
}		
int main()
{
	unsigned long int n;
	while(scanf("%lu",&n)!=EOF)
	{	
		printf("%lu\n",func(n));
	}
	return 0;
}
