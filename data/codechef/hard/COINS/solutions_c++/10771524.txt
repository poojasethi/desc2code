#include <bits/stdc++.h>
using namespace std;

long long int dp[100000];

long long int count(long long int n)
{
	if(n < 100000)
		return dp[n];
		
	return max(n, count(n/4) + count(n/3) + count(n/2));
}

int main()
{
	long long int i,j,k,l,m,n;
	
	dp[0] = 0;
	dp[1] = 1;
	
	for(i=2;i<100000;i++){
		k = dp[i/4]+dp[i/3]+dp[i/2];
		dp[i] = max(i,k);
	}
	
	
	while(scanf("%lld",&n) != EOF){
    	printf("%lld\n", count(n));
	}
}
