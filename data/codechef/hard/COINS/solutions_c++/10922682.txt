#include<bits/stdc++.h>
using namespace std;

map <long long, long long> dp;
 
long long ans(long long n) {
     if(n == 0) {
           return 0;
     } 
     if(dp[n] != 0) {
           return dp[n];
     }
     long long sum = ans(n / 2) + ans(n / 3) + ans(n / 4);
     if(sum > n) {
           dp[n] = sum;
     } else {
           dp[n] = n;
     }
     return dp[n];
}

int main()
{
      long long n; 
      while(scanf("%lld",&n) == 1) {
            long long output = ans(n);
            cout << output << endl;
      }
      return 0;
}
