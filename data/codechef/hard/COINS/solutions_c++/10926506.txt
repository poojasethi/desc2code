#include<stdio.h>
typedef long long int ll;
ll max(ll a,ll b){
    if(a<b) return b;
    else return a;
}
ll dp[10000000]={0};
ll count(ll x){
    if(x<12){
        dp[x]=x;
        return x;
    }
    if(x==12){
        dp[x]=13;
        return 13;
    }
    if(x<10000000 && dp[x]){
        return dp[x];
    }
    ll result=count(x/2)+count(x/3)+count(x/4);
    if(x<10000000 && result>=x){
        dp[x]=result;
    }
    if(result>=x)
    return result;
    else
    return x;
}
int main(){
    ll n;
    while(scanf("%lld",&n)!=EOF){
    ll ans=n;
    printf("%lld\n",count(ans));
    }
}
