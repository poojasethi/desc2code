#include <stdio.h>
#include <map>
#include <string>
#include <stdlib.h>

using namespace std;

typedef long long int lli;

inline int getn(){
    int n = 0, c = getchar_unlocked();
    
    if(c == EOF) return -999;
    
    while (c < '0' || c > '9') c = getchar_unlocked();
    
    while (c >= '0' && c <= '9'){
        n = (n << 3) + (n << 1) + c - '0';
        c = getchar_unlocked();
    }
    
    return n;
}

map<lli, lli> mem;

lli split_val(lli n){
    lli n0, n1, n2;
    
    if(mem.count(n)) return mem[n];
    
    n0 = (lli)(n/2);
    n1 = (lli)(n/3);
    n2 = (lli)(n/4);
    
    if (n0+n1+n2 <= n)
        return n;
    
    return mem[n] = split_val(n0) + split_val(n1) + split_val(n2);
    //return split_val(n0) + split_val(n1) + split_val(n2);
}



int main(){
    lli n = 0;
    while(scanf("%lld", &n) != EOF){
    
        printf("%lld\n", split_val(n));
    }
    
    return 0;
}
