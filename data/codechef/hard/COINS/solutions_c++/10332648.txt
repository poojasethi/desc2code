#include <iostream>
#include <cstring>
#include <map>
using namespace std;
map<long long int,long long int> arr;
long long int maxt(long int n)
{
    if(n==1||n==0)
        return n;
    else if(arr.find(n)!=arr.end())
        return arr[n];
    else
    {
        long long int t1=maxt(n/2)+maxt(n/3)+maxt(n/4);
        arr[n]=t1>n?t1:n;
        return arr[n];
    }
}
int main()
{
    long long int n;
    while(cin>>n)
    {
        cout<<maxt(n)<<endl;
    }
    return 0;
}
