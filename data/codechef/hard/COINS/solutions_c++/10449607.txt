#include<stdio.h>
#include<map>
#define max(a,b) a>b?a:b
using namespace std;
map<unsigned long int  ,unsigned long int >m;
unsigned long int b(int n)
{
    if(n==0)
        return 0;
    if(m[n]>0)
        return m[n];
    else
    {
        m[n]=max(n,b(n/2)+b(n/3)+b(n/4));
        return m[n];
        }
}
int main()
{
unsigned int n;
while(scanf("%d",&n)>0)
{
printf("%lu\n",b(n));
}
return 0;}
