#include <iostream>
#include <algorithm>
#include <vector>
#include <stdio.h>
using namespace std;
long long int memory[10000000] = {0};
long long int func(long long int);

int main(){
	long long int n;	
	while(scanf("%lld",&n) != EOF){
		cout<<(func(n))<<endl;
	}
	return 0;
}

long long int func(long long int n){
	if(n < 12){
		memory[n] = n;
		return n;
	}
	else if ( n == 12){
		memory[n] = 13;
		return 13;
	}
	
	else if ( n < 10000000 ){
		if (memory[n] != 0 ){
			return memory[n];
		}
	}
	
	long long int two,three,four;
	two = n/2;
	three = n/3;
	four = n/4;
	
	long long int result = func(two) + func(three) + func(four);
	if(n < 10000000){
		memory[n] = result;
	}
	return result;
}
