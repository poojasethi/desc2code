#include <bits/stdc++.h>
using namespace std;
long long a[100000]={0};
long long coin(long long n)
 {
    if(n<12)
    return n;
    else if(n<=100000&&a[n-1]!=0)
        return a[n-1];
    else
        {
           long long x=coin(n/2)+coin(n/3)+coin(n/4);
           if(x>=n)
           {
            if(n<=100000)
            a[n-1]=x;
           return x;
           }
        }

 }
int main()
 {
    long long n;
    while(scanf("%lld",&n)!=EOF)
    {
        printf("%lld\n",coin(n));
    }
 }
