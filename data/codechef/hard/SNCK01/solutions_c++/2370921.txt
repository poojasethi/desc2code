#include<iostream>
using namespace std;
int main()
{
int t;
cin>>t;
while(t--)
{
int M[50][50];
int n,m;
cin>>n>>m;
for(int i=0;i<n;i++)
{
for(int j=0;j<m;j++)
{
cin>>M[i][j];
}
}
int ans=0;
for(int i=0;i<n;i++)
{
    for(int j=m-2;j>=0;j--)
    {
        if(M[i][j+1]>=M[i][j])
            M[i][j]--;
    }
}
for(int i=0;i<n;i++)
    ans=ans^M[i][0];
if(ans==0)
cout<<"SECOND\n";
else
cout<<"FIRST\n";
}
}
