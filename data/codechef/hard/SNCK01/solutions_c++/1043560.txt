#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main()
{
	int a[55][55];
	int n,m,t;
	scanf("%d",&t);
	while(t --) {
		scanf("%d%d",&n,&m);
		for(int i = 1; i <= n; i ++)
			for(int j = 1; j <= m; j ++)
				scanf("%d",&a[i][j]);
		int ans = 0;
		for(int i = 1; i <= n; i ++) {
			int sg = a[i][m];
			for(int j = m - 1; j > 0; j --)
				if(sg < a[i][j]) sg = a[i][j];
				else sg = a[i][j] - 1;
			ans ^= sg;
		}
		if(ans) printf("FIRST\n");
		else printf("SECOND\n");
	}
	return 0;
}
