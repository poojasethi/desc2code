#include<iostream>
#include<cstdio>
 
using namespace std;
 
#define gc getchar_unlocked
 
inline void getint(int &x)
{
	register int c = gc();
	x = 0;
	for(;(c<48 || c>57);c = gc());
	for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
 
int arr[50];
 
//m>=1
inline int effectivenum(int m)
{
	int prev = 0;
	for(int i=m-1; i>=0; --i)
	{
		if(arr[i] > 0)
		{
			if(arr[i] <= prev)
				prev = (--arr[i]);
			else
				prev = arr[i];
		}
	}
	return prev;
}
			
 
int main()
{
	int t;
	getint(t);
	while(t--)
	{
		int ans = 0;
		int n, m;
		getint(n);
		getint(m);
		for(int i=0; i<n; ++i)
		{
			for(int j=0; j<m; ++j)
				getint(arr[j]);
			ans ^= effectivenum(m);
		}
		if(ans == 0)
			printf("SECOND\n");
		else
			printf("FIRST\n");
	}
	return 0;
}
