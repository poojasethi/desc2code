#include<iostream>

using namespace std;

int main()
{
  int t,n,m,i,j,mat[51][51],result,nim;
  cin>>t;
  while(t)
  {
    cin>>n>>m;
    for(i=0;i<n;i++)
    {
      for(j=0;j<m;j++)
	cin>>mat[i][j];
    }
    result=0;
    for(i=0;i<n;i++)
    {
      nim=mat[i][m-1];
      for(j=m-2;j>=0;j--)
      {
	if(mat[i][j]<=nim)
	  nim=mat[i][j]-1;
	else
	  nim=mat[i][j];
      }
      result^=nim;
    }
    if(result)
      cout<<"FIRST\n";
    else
      cout<<"SECOND\n";
    t--;
  }
  return 0;
}
