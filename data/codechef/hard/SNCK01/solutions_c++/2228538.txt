#include <cstdio>
using namespace std;
 
int n, m;
int a[50][50];
 
int main() 
{
	int t;
	scanf("%d", &t);// test cases
	while (t--) 
	{
		scanf("%d%d", &n, &m); // input N and M
		for (int i=0;i<n;i++)
		{
			for (int j=0;j<m;j++)
			{
				scanf("%d", &a[i][j]);
			}
		}
		for (int i=0;i<n;i++)
		{
			for (int j=m-1;j>=0;j--)
			{
				int next = (j == m-1) ? 0 : a[i][j+1];
				if (a[i][j] < next+1)
					--a[i][j];
			}
		}
		int x = 0;
		for (int i=0;i<n;i++)
		{
			x ^= a[i][0];
		}
		printf(x ? "FIRST\n" : "SECOND\n");
	}
 	return 0;
}
