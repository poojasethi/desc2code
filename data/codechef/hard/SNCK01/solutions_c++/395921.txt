int main () {
	int test;
	cin >> test;
	int arr [50][50];
	while (test-->0) {
		int n, m;
		cin >> n >> m;
		FOR (i, 0, n) {
			FOR (j, 0, m) cin >> arr[i][j];
		}
		
		FOR (i, 0, n) {
			REV (j, m-2, -1) {
				if (arr [i][j+1] >= arr [i][j])
					arr [i][j] --;
			}
		}
		
		int OR = 0;
		FOR (i, 0, n) OR ^= arr [i][0];
		
		if (OR > 0) cout << "FIRST";
		else cout << "SECOND";
		cout << endl;
	}
}
