using namespace std;

#define FOR(i,m,n) for(int (i)=(m);(i)<(n);(i)++)
#define  ll long long

int main(){
    int t; cin>>t;
    while(t--){
        int n,m,a[50][50],sg[50],res=0;
        cin>>n>>m;
        memset(a,0,sizeof(a));
        FOR(i,0,n)
            FOR(j,0,m)
                cin>>a[i][j];

        FOR(i,0,n){
            sg[i]=a[i][m-1];
            for(int j=m-2;j>=0;j--) sg[i]=a[i][j]-(sg[i]>=a[i][j]);
        }
        FOR(i,0,n) res ^= sg[i];
        cout<<((res)? "FIRST" : "SECOND")<<endl;
    }
    return 0;
}
