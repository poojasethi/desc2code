#include<iostream>
#include<stdio.h>
using namespace std;


int arr[50][50];

int calc(int row,int pos, int col){
	
	if(pos==col){
		return 0;
	}

	int steps = arr[row][pos];
	int to = calc(row,pos+1,col);
	if(to>=steps){
		return steps-1;	
	}
	else if(to < steps){
		return steps;
	}
	
}


int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n,m;
		scanf("%d%d",&n,&m);
		
		for(int i=0;i<n;i++){
			for(int j=0;j<m;j++){
				scanf("%d",&arr[i][j]);
				//printf("%d ",arr[i][j]);
			}
			printf("\n");
		}
		int ans = 0;
		for(int i=0;i<n;i++){
			
			int c = calc(i,0,m);
			ans = ans ^ c;

		}
		if(ans==0){
			printf("SECOND\n");
		}
		else{
			printf("FIRST\n");
		}
	}

}
