#include<iostream>
#include<cstdio>
#include<bitset>
#include<algorithm>
using namespace std;

#define FOR(i, a, b) for (int i(a), _b(b); i <= _b; ++i)
#define FORD(i, a, b) for (int i(a), _b(b); i >= _b; --i)
#define REP(i, n) for (int i(0), _n(n); i < _n; ++i)
#define REPD(i, n) for (int i((n)-1); i >= 0; --i)

int n, m;
int a[50][50];

int main() {
	int t;
	scanf("%d", &t);
	while (t--) {
		scanf("%d%d", &n, &m);
		REP(i, n) REP(j, m)
			scanf("%d", &a[i][j]);
		REP(i, n) REPD(j, m) {
			int next = j == m-1 ? 0 : a[i][j+1];
			if (a[i][j] < next+1)
				--a[i][j];
		}
		int x = 0;
		REP(i, n)
			x ^= a[i][0];
		printf(x ? "FIRST\n" : "SECOND\n");
	}

	exit(0);
}
