#include <bits/stdc++.h>
using namespace std;
int arr[100000];

int grundy(int m) {
	int prev = 0;
	for(int i = m-1; i >= 0; i--) {
		if(arr[i] >= 0) {
			if(arr[i] <= prev) {
				prev = (--arr[i]);
			} else {
				prev = arr[i];
			}
		}
	}
	return prev;
}

int main() {
	int t;
	cin>>t;
	int n, m;
	while(t--) {
		cin>>n>>m;
		int ans = 0;
		for(int i = 0 ; i < n ; i++) {
			for(int j = 0; j < m; j++) {
				cin>>arr[j];
			}
			ans	^= grundy(m);
		}
		if(ans == 0) {
			cout<<"SECOND\n";
		} else {
			cout<<"FIRST\n";
		}
	}
	return 0;
}
