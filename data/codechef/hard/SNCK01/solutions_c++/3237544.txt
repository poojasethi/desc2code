#define s(x) scanf("%d",&x)
using namespace std;
int numbers[50][50],r,c;
int get_grundy_row(int row){
    int ans=numbers[row][c-1];
    for(int i=c-2;i>=0;i--){
//        int mex[100];
//        memset(mex,0,sizeof(mex));
        int grundy[numbers[row][i]+1];
        for(int j=1;j<=numbers[row][i];j++){
//            if(i==0){
//                ans=numbers[row][i];
//                grundy[numbers[row][i]]=ans;
//            }
//            else{
//                //mex[ans]=1;
                if(j==1){
                    if(ans)
                        grundy[j]=0;
                    else
                        grundy[j]=1;
//                    mex[grundy[j]]=1;
                }
                else{
//                    int temp=0;
//                    while(mex[temp]){
//                        temp++;
//                    }
                    if((grundy[j-1]+1)!=ans)
                        grundy[j]=grundy[j-1]+1;
                    else
                        grundy[j]=grundy[j-1]+2;
//                    mex[grundy[j]]=1;
                }
//            }
        }
        ans=grundy[numbers[row][i]];
    }
    return ans;
}
int main(void){
    int t;
    s(t);
    while(t--){
        s(r);s(c);
        for(int i=0;i<r;i++)
            for(int j=0;j<c;j++)
                s(numbers[i][j]);
        int ans=get_grundy_row(0);
        for(int i=1;i<r;i++){
            ans^=get_grundy_row(i);
        }
        if(ans){
            printf("FIRST\n");
        }
        else{
            printf("SECOND\n");
        }
    }
    return 0;
}
