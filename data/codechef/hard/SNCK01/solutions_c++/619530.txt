//RETURNS: nimber if on (row,col), num = 1 iff number is 1, num = 0 iff number > 1
int nimber(int row, int col, int num) {
    if(mem[row][col][num] != -1) return mem[row][col][num];
    set<int> s;
    
    //take all
    if(col+1 < M) s.insert(nimber(row,col+1,matrix[row][col+1] == 1));
    else s.insert(0);
    
    if(num == 0) {
        //take all but 1
        s.insert(nimber(row,col,1));
    }
    
    int res = 0;
    while(s.count(res)) ++res;
    return mem[row][col][num] = res;
}
