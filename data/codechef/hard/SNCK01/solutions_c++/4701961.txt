#include<iostream>

using namespace std;

int arr[50];

//m>=1
int effectivenum(int m)
{
	int prev = 0;
	for(int i=m-1; i>=0; --i)
	{
		if(arr[i] > 0)
		{
			if(arr[i] <= prev)
				prev = (--arr[i]);
			else
				prev = arr[i];
		}
	}
	return prev;
}
			

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int ans = 0;
		int n, m;
		cin>>n>>m;
		for(int i=0; i<n; ++i)
		{
			for(int j=0; j<m; ++j)
				cin>>arr[j];
			ans ^= effectivenum(m);
		}
		if(ans == 0)
			cout<<"SECOND"<<endl;
		else
			cout<<"FIRST"<<endl;
	}
	return 0;
}
