using namespace std;

#define NN 51

int game[NN][NN];
int N, M;

void in()
{
 scanf("%d%d", &N, &M);
 for(int i=0; i<N; i++)
 {
     for(int j=0; j<M; j++)
     {
         scanf("%d", &game[i][j]);
     }
 }
}

void solve()
{
    in();
    int ans = 0, delta;
    for(int i=0; i<N; i++)
    {
        delta = 0;
        for(int j=M-2; j>-1; j--)
        {
            if(game[i][j]>game[i][j+1]+delta)delta = 0;
                else delta = -1;
        }
        ans ^= game[i][0] + delta;
    }
    printf("%s\n", ans==0?"SECOND":"FIRST");

}

int main()
{
    #ifndef ONLINE_JUDGE

   // freopen("output.txt","w",stdout);
    #endif

    int t;
    scanf("%d", &t);
    while(t--)
    {
        solve();
    }


}
