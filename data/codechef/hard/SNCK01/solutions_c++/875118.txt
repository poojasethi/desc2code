int g[64];
int dp[64][64];
bool flag[64];

void sol(){	
	int n, m;
	cin >> n >> m;
	int ans = 0;
	FOR(I,1,n) {
		int cc = 0, h = 0;
		FOR(j,1,m) {
			RE("%d",&h);
			if (h > 0) g[cc++] = h;
		}
		dp[cc-1][g[cc-1]] = g[cc-1];
		DFOR(i,cc-2,0) {
			CLR(flag);
			flag[dp[i+1][g[i+1]]] = true;
			int cu = 0;
			FOR(j,1,g[i]){
				while (flag[cu])cu++;
				dp[i][j] = cu;
				flag[cu] = true;
			}
		}
		ans ^= dp[0][g[0]];
	}
	if (ans) WR("FIRST\n");
	else WR("SECOND\n");
}

		sol();
