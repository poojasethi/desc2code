    }

ll dp[2][2][20][2];

ll rec(ll isStart,ll isLess,ll idx,ll hasDigit,ll curDigit)
{
    if(idx>=(long long)v.size())
        return hasDigit;
    ll &ret=dp[isStart][isLess][idx][hasDigit];
    if(ret!=-1LL)
        return ret;
    ret=0;
    ll last=(isLess==1LL)?9:v[idx];
    ll i;
    if(isStart)
    {
        for(i=1;i<=last;i++)
            ret+=rec(0,isLess|(i<v[idx]),idx+1,hasDigit|(i==curDigit),curDigit);
        ret+=rec(1,1,idx+1,hasDigit,curDigit);
    }
    else
    {
        for(i=0;i<=last;i++)
            ret+=rec(0,isLess|(i<v[idx]),idx+1,hasDigit|(i==curDigit),curDigit);
    }
    return ret;
}
