#include <vector>
#include <cstring>
#include <string>
#include <algorithm>
#include <iostream>
#include <cmath>
#include <map>
#include <cassert>

using namespace std;

#define FOR(i, a, b) for(int i = a; i < b; i++)
#define REP(i, n) FOR(i, 0, n)
#define ll long long
#define mp make_pair

const double pi = acos(-1.0);
const double eps = 1e-8;

int n;
int x[2005], y[2005];

int main()
{
    scanf("%d", &n);
    REP(i, n) scanf("%d %d", x+i, y+i);
    ll ret = 0;
    REP(c, n)
    {
        REP(i, n) if(i != c) x[i] -= x[c], y[i] -= y[c];
        x[c] = y[c] = 0;
        
        vector< double > v;
        REP(i, n) if(i != c) v.push_back(atan2(y[i]+0., x[i]));
        sort(v.begin(), v.end());
        REP(i, n-1) v.push_back(v[i] + 2*pi);
        
        int p1 = 0, p2 = 0;
        REP(i, n-1)
        {
            while(v[p1] - v[i] < pi/2 - eps) p1++;
            while(v[p2] - v[i] < 3*pi/2 + eps) p2++;
            
            ret += p2 - p1;
        }
    }
    cout << (ll)n*(n-1)*(n-2)/6 - ret/2 << endl;
    system("pause");
    return 0;
}
