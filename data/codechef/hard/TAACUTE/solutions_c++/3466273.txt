#include <cstdio>

#include <cmath>
#define MAXN 2005
#define PI 3.141592653589793238463
using namespace std;

struct point
{
    int x,y;
    bool mark;
};

long long n;
long long sum;
point a[MAXN];
point b[MAXN];

bool cmp2(point aa,point bb)
{
  if(aa.mark) return 0;
  if(bb.mark) return 1;
  if (aa.y <= 0 && bb.y > 0) return 1;
  if (aa.y > 0 && bb.y <= 0) return 0;
  if (aa.y == 0 && bb.y == 0) return (aa.x < bb.x);
  if (aa.x * bb.y - bb.x * aa.y > 0) return 1;
  return 0;
}

double angle(int k,int j)
{
   return atan2(a[k].y-a[j].y,a[k].x-a[j].x);
}


double sinus(int j,int k)
{
   return a[j].x*a[k].y - a[j].y*a[k].x;
}

double kosin(int j,int k)
{
   return a[j].x*a[k].x + a[j].y*a[k].y;
}


long long firstin,firstout;
int main()
{
    scanf("%d",&n);
    for(int i=0;i<=n-1;i++)
    {
            scanf("%d %d",&b[i].x,&b[i].y);
    }
    for(int i=0;i<=n-1;i++)
    {
        //fixiramo tacku i
         for(int j=0;j<=n-1;j++)
         {
            a[j].x=b[j].x;
            a[j].y=b[j].y;
         }

        for(int j=0;j<=n-1;j++)
        {
            if(i!=j)//norm
            {
                a[j].y-=a[i].y;
                a[j].x-=a[i].x;
                a[j].mark=false;
            }
            else
                a[j].mark=true;
        }
        sort(a,a+n,cmp2);
        firstin=firstout=1;
        for(int j=0;j<=n-2;j++) // od -PI do PI
        {
            while(sinus(j,firstin)>=0 && kosin(j,firstin)>0 && firstin<=n-2)
            {
                firstin++;
            }
            while(!(sinus(firstout,j)>0 && kosin(firstout,j)>0) &&  firstout<=n-2)
            {
                firstout++;
            }
            sum+=firstout-firstin;
        }
    }
    sum=n*(n-1)*(n-2)/6-sum;
    printf("%lld\n",sum);
    return 0;
}
