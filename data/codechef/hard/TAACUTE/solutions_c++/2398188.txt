typedef long long LL;

#define y1 myy1
const LD pi = acos(-1.0);
const LD eps = 1e-8;

#define Point Vect
class Vect // also Point
{
public:
    LD x, y;

    Vect(LD x = 0.0, LD y = 0.0) : x(x), y(y) { }

    Vect(const Vect &o)
    {
        x = o.x;
        y = o.y;
    }

    Vect(const Point &A, const Point &B)
    {
        x = B.x - A.x;
        y = B.y - A.y;
    }

    inline LD distFrom(const Point &pt) const
    {
        return sqrt((x - pt.x) * (x - pt.x) + (y - pt.y) * (y - pt.y));
    }

    inline LD getFi() const
    {
        LD ret = atan2(y, x);
        if (ret < 0.0)
        {
            ret += 2 * pi;
        }
        return ret;
    }

    inline LD getLen() const
    {
        return sqrt(x * x + y * y);
    }

    Vect operator + (const Vect &o) const
    {
        return Vect(x + o.x, y + o.y);
    }

    Vect& operator += (const Vect &o)
    {
        x += o.x;
        y += o.y;
        return *this;
    }

    Vect operator - (const Vect &o) const
    {
        return Vect(x - o.x, y - o.y);
    }

    Vect& operator -= (const Vect &o)
    {
        x -= o.x;
        y -= o.y;
        return *this;
    }

    LD operator * (const Vect &o) const // dot product ` skalyar
    {
        return x * o.x + y * o.y;
    }

    LD operator % (const Vect &o) const // cross product ` vectorakan
    {
        return x * o.y - y * o.x;
    }

    LD angleWith(const Vect &o) const
    {
        return acos((*this * o) / getLen() / o.getLen());
    }
};

bool acute(const Point &A, const Point &B, const Point &C)
{
    Vect u, v;
    bool ret = true;

    u = Vect(A, B);
    v = Vect(A, C);
    ret &= (u.angleWith(v) < pi / 2.0);

    u = Vect(B, A);
    v = Vect(B, C);
    ret &= (u.angleWith(v) < pi / 2.0);

    u = Vect(C, A);
    v = Vect(C, B);
    ret &= (u.angleWith(v) < pi / 2.0);

    return ret;
}

int naive(vector<Point> pt)
{
    int n = sz(pt);
    int ret = 0;
    for (int i = 0; i < n; ++i)
    {
        for (int j = i + 1; j < n; ++j)
        {
            for (int k = j + 1; k < n; ++k)
            {
                if (acute(pt[i], pt[j], pt[k]))
                {
                    ++ret;
                }
            }
        }
    }
    return ret;
}

int main()
{
#ifdef harhro94

    int n;
    cin >> n;
    vector<Point> pt(n);
    for (int i = 0; i < n; ++i)
    {
        cin >> pt[i].x >> pt[i].y;
    }
    LL ans = 0;
    for (int i = 0; i < n; ++i)
    {
        vector<Vect> V;
        for (int j = 0; j < n; ++j)
        {
            if (i == j)
            {
                continue;
            }
            V.pb(pt[j] - pt[i]);
        }
        int m = n - 1;
        vector<LD> mas;
        for (int j = 0; j < m; ++j)
        {
            mas.pb(V[j].getFi());
        }
        sort(all(mas));
        for (int j = 0; j < m; ++j)
        {
            mas.pb(2 * pi + mas[j]);
        }
        int l = 0;
        int r = 0;
        for (int j = 0; j < m; ++j)
        {
            while (l < 2 * m && mas[l] - mas[j] < pi / 2.0 - eps)
            {
                ++l;
            }
            while (r < 2 * m && mas[r] - mas[j] < 1.5 * pi + eps)
            {
                ++r;
            }
            ans += (r - l);
        }
    }
    //cerr << ans << endl;
    //cerr << naive(pt) << endl;
    assert(ans % 2 == 0);
    cout << (LL)n * (LL)(n - 1) * (LL)(n - 2) / 6 - ans / 2 << endl;

}
