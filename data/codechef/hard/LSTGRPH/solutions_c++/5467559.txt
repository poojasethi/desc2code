    int n, m, k;

    S(n), S(m), S(k);

    queue<int> q;

    int v1, v2, v3;

    fori(i,0,m) {
        S(v1), S(v2), S(v3);
        v1--, v2--;
        v[v1].pb(v2);
        v[v2].pb(v1);
        vv[v1].pb(v3);
        vv[v2].pb(v3);
    }
    int last = -1;
    fori(i,0,n) {
        if(!vis[i]) {
            vis[i] = 1;
            q.push(i);
            last = i;
            while(!q.empty()) {
                int pos = q.front();
                q.pop();
                fori(i,0,v[pos].size()) {
                    if(vis[v[pos][i]] == 0) {
                        vis[v[pos][i]] = 1;
                        lab[v[pos][i]] = vv[pos][i]^lab[pos];
                        q.push(v[pos][i]);
                    }
                    else {
                        if(lab[v[pos][i]] != (vv[pos][i]^lab[pos])) {
                            cout << -1 << endl;
                            return 0;
                        }
                    }
                }
            }
        }
    }
    mem(lab,0);
    mem(vis,0);
    lab[last] = k-1;

        if(!vis[i]) {
            vis[i] = 1;
            q.push(i);

                int pos = q.front();

                    if(vis[v[pos][i]] == 0) {
                        vis[v[pos][i]] = 1;
                        lab[v[pos][i]] = vv[pos][i]^lab[pos];

                    }
                }
            }
        }
    }
