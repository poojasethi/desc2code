#include<iostream>
#include<vector>
using namespace std;
int vis[200010],vis2[200010],w,an[200010],u,v,n,m,k,l;
vector<int> adj[200010],we[200010];
void dfs(int n)
{
	if(vis[n])return;
	vis[n]=1;
	for(int i=0;i<adj[n].size();i++)
	{
		dfs(adj[n][i]);
	}
}
void dfs2(int n,int k)
{
	if(vis2[n])return;
	vis2[n]=1;
	an[n]=k;
	for(int i=0;i<adj[n].size();i++)
	{
		dfs2(adj[n][i],we[n][i]^k);
	}
}
int main()
{
	cin>>n>>m>>k;
	for(int i=0;i<m;i++)
	{
		cin>>u>>v>>w;
		adj[u].push_back(v);
		adj[v].push_back(u);
		we[u].push_back(w);
		we[v].push_back(w);
	}
	for(int i=1;i<=n;i++)
	{
		if(!vis[i])
		{
			dfs(i);
			l=i;
		}
	}
	for(int i=1;i<=n;i++)
	{
		if(i!=l)
		{
			dfs2(i,0);
		}
		else
		{
			dfs2(i,k-1);
		}
	}
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<adj[i].size();i++)
		{
			//cout<<i<<" "<<adj[i][j]<<endl;
			//cout<<an[i]<<" "<<an[adj[i][j]]<<" "<<we[i][j]<<endl;
			if((an[i]^an[adj[i][j]])!=we[i][j])
			{
				cout<<-1<<endl;
				return 0;
			}
		}
	}
	for(int i=1;i<=n;i++)cout<<an[i]<<" ";
	cout<<endl;
}
