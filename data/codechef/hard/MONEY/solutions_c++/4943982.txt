#include"map"
#include"stdio.h"
#define MAX 100000
using namespace std;
int find (map<unsigned int,int>* function, unsigned int N,int* lookup)
{
  if(N<MAX)return lookup[N];
  int k=(*function)[N];
//    printf("answered %d, for %u\n",k,N);
  if(k>0)return k;
//   printf("asking %ud",N);
  int sum=1;
  unsigned int it=2;
  while(it*it<N)
  {
    if(N%it==0)
      sum+=find(function,N/it,lookup)+find(function,it,lookup);
    it++;
  }
  if(it*it==N)
    sum+=find(function,it,lookup);
  (*function)[N]=sum;
  return sum;
}
int main()
{
  map<unsigned int,int> function;
  int T;
  scanf("%d",&T);
  function[1]=1;
  int lookup[MAX];
  for(int x=0;x<MAX;x++)
    lookup[x]=0;
  lookup[1]=1;
  for(int i=1;i<MAX;i++)
  {
    for(int j=i<<1;j<MAX;j+=i)
    {
      lookup[j]+=lookup[i];
    }
  }
  while(T--)
  {
    unsigned int N;
    scanf("%u",&N);
//     printf("got %u\n",N);
    int k=find(&function,N+1,lookup);
    printf("%d\n",k);
  }
}
