#include <bits/stdc++.h>
using namespace std;
#define N 50000
int ans[N + 10];
map<unsigned int,int> mp;
void PreCompute()
{
    int i,j;
    ans[1] = 1;
    for(i = 1 ; i<=N ; i++)
    {
        for(j = 2*i ; j<=N ; j = j + i)
        {
            ans[j] = ans[j] + ans[i];
        }
    }
}
int func(unsigned int n)
{
    if(n <= N) return ans[n];
    int v = mp[n];
    if(v > 0) return v;
    else
    {
        int ways = 1;
        unsigned int i;
        for(i = 2 ; i*i <n ; i++)
        {
            if(n%i == 0)
            {
                ways = ways + func(i) + func(n/i);
            }
        }
        if(i*i == n) ways = ways + func(i);
        mp[n] = ways;
        return ways;
    }
}
int main()
{
    PreCompute();
    int tc,t;
    scanf("%d",&tc);
    for(t = 1 ; t<=tc ; t++)
    {
        unsigned int n;
        scanf("%u",&n);
        int ans = func(n+1);
        printf("%d\n",ans);
    }
    return 0;
}
