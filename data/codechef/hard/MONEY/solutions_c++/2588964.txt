#include <cstdio>
#include <cmath>

using namespace std;
int A[10000000];
int validSets(long long n)
{
	if(n == 0)
		return 0;
	int sum = 1;
	if(n < 10000000)
	{
		if(A[n] != 0)
			return A[n];
	}
	int sq = sqrt((double)(n+1));
	for(int i=2 ; i<=sq ; i++)
	{
		if( (n+1) % i == 0)
		{
			sum += validSets(i-1);
			if(i != (n+1)/i)
				sum += validSets( ((n+1)/i) -1);
		}
		
	}
	if(n < 10000000)
		A[n] = sum;
	return sum;
}

int main()
{
	int T ,i , j ;
	long long N; 
	scanf("%d" , &T);
	A[0] = 0;
	A[1] = 1;
	A[2] = 1;
	/*for( i=1 ; i<100000 ; i++)
	{
		A[i] = 1;
	}
	for( j = 2 ; j< 100000 ; j++)
	{
		for( i=2*j-1 ; i<100000 ; i = i+j)
		{
			A[i] += A[j-1];
		}
	}
	/*for(i=1 ; i<15 ; i++)
	{
		printf("%d " , A[i]);
	}
	printf("\n");*/
	while(T--)
	{
		scanf("%lld" , &N);
		//printf("%d--\n" , N);
		printf("%d\n" , validSets(N));
	}
	
	return 0;
}
