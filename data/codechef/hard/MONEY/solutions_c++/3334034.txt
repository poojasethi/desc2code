#include<iostream>
#include<math.h>
#include<stdio.h>
#include<map>
using namespace std;
typedef long long ll;

map < int, ll > m;

ll count(ll n)
{
	

    if (n == 0) {
	return 1;
    }
    if (m.find(n) != m.end()) {
	return m[n];
    }
    int sq = sqrt((ll)n + 1);
	

    ll ans = 0;
    for (int i = 1; i <= sq; i++) {

	if ((n + 1) % i == 0) {
	    int j = (n + 1) / i;
	 
	    if (i - 1 != n)
		ans = ans + count(i - 1);
	    if (i != j && j - 1 != n) {
		ans = ans + count(j - 1);
	    }
	}
    }
	if(ans==0){
		while(1);
	}

    m[n] = ans;
    return ans;
}


int main()
{

    int t;
    cin >> t;
    while (t--) {
	int n;
	cin>>n;
	if (n == 0) {
	    cout << 0 << endl;
	} else{
	    int ans = count(n);
	    cout << ans << endl;
	}
    }

}
