#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
using namespace std;

int fact[5000];
int ans[5000][2];

int main()
{
    unsigned int num;
    int T;
    int i=2,c1=0;
    scanf("%ud",&T);
    while(T--)
    {
        scanf("%ud",&num);
        num++;
        unsigned int x=num;
        memset(fact,0,sizeof(fact));
        i=2;c1=0;
        while(!(x&1))
        {
            x=x>>1;
            fact[c1++]=2;
        }

        int limit=sqrt((double)num);
        i=3;
        while(x>1 && i<=limit)
        {
            while(x%i==0)
            {
                fact[c1++]=i;
                x=x/i;
            }
            i=i+2;
        }

        if(x!=1)
        {
            fact[c1++]=x;
        }

        ans[0][0]=1;
        ans[0][1]=1;

        int c2=1,pos=0,temp=0;
        for(int p=0;p<c1;p++)
        {
            temp=c2;
            for(int q=pos;q<temp;q++)
            {
                ans[c2][0]=fact[p]*ans[q][0];
                ans[c2][1]=0;

                for(int r=0;r<c2;r++)
                {
                    if((ans[c2][0]%ans[r][0])==0)
                    {
                        ans[c2][1]+=ans[r][1];
                    }
                }
                c2++;
            }

            if(p<c1-1 && fact[p]==fact[p+1])
            {
                pos=temp;
            }
            else
            {
                pos=0;
            }
        }

        printf("%d\n",ans[c2-1][1]);
    }
    return 0;
}
