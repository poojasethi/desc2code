/*-----Fast I/O-----*/
inline void inp( int &n ){n=0;int ch=getchar();int sign=1;while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getchar();}while( ch >= '0' && ch <= '9' )n = (n<<3)+(n<<1) + ch-'0', ch=getchar();n=n*sign;}
/*-----End of Fast I/O-----*/

map<unsigned int, lld> memo;

void pre()
{
	memo[1] = 1;
	for(int i=1;i<5000;i++)
		for(int j=2*i;j<5000;j+=i)
			memo[j] += memo[i];
}


lld solve(unsigned int n)
{
	if(present(memo,n)) return memo[n];
	lld ret = 1;
	int i;
	for(i=2;i*i<n;i++)
		if(n%i==0)
			ret+=solve(i) + solve(n/i);
	if(i*i == n)
		ret+=solve(i);
	memo[n]=ret;
	return ret;
}

int main()
{
	int t;
	si(t);
	pre();
	while(t--)
	{
		unsigned int n;
		scanf("%u",&n);
		printf("%lld\n",solve(n+1));
	}
	return 0;
}
