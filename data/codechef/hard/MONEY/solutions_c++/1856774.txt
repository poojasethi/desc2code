#include <iostream>
#include <map>

using namespace std;

int calculateNumUniqueSets(int leftOfMark, int remaining, int step);
map<int, int> foldSets;

int main(int argc, char **argv)
{
    int numTestCases, curTotal;

    cin >> numTestCases;
    while (cin >> curTotal)
    {
        cout << calculateNumUniqueSets(0, curTotal, 1) << endl;
    }

    return 0;
}

int calculateNumUniqueSets(int leftOfMark, int remaining, int step)
{
    int iter = 1;
    int curMarkPos = remaining / step / ++iter;
    int retVal = 1;
    bool bLogarithmicReduction = true;
    while (curMarkPos > 0)
    {
        int curValueLeftOfMark = leftOfMark + curMarkPos * step;
        int curRemaining = remaining - curValueLeftOfMark;

        if (curRemaining % (curValueLeftOfMark + step) == 0)
        {
            int numRightOfMark = (curRemaining / (curValueLeftOfMark + step));
            if (numRightOfMark == 0) break;
            if (foldSets.find(numRightOfMark) != foldSets.end())
            {
                retVal += foldSets[numRightOfMark];
            }
            else
            {
                int newFold = calculateNumUniqueSets(0, curRemaining, curValueLeftOfMark + step);
                foldSets[numRightOfMark] = newFold;
                retVal += newFold;
            }
        }

        if (bLogarithmicReduction && curMarkPos != remaining / step / ++iter)
        {
            curMarkPos = remaining / step / iter;
        }
        else
        {
            curMarkPos--;
            bLogarithmicReduction = false;
        }
    }

    return retVal;
}
