#include<iostream>
#include<vector>
#include<math.h>
#include<map>

using namespace std;

map <long long,long long> M;

void factor(long long n,vector <long long> &v)
{
    v.push_back(1);
    
    for(long long i=2;i<=pow((long double)n,0.5);i++)
    {
        if(n%i==0)
        {
            if(i!=n/i)
            {
                v.push_back(i);     v.push_back(n/i);
            }
            else
            v.push_back(i);
        }
    }
}

long long func(long long n)
{
    if(n==1)
    return 1;
    
    if(M[n]!=0)
    return M[n];
    
    vector <long long> v;
    
    factor(n,v);
    
    long long result=0;
    
    for(int i=0;i<v.size();i++)
    result+=func(v[i]);
    
    M[n]=result;
    
    return result;
}

int main()
{
    int T;
    
    scanf("%d",&T);
    
    for(long long i=0;i<10000;i++)
    {
        func(i);
    }
    
    while(T--)
    {
        long long n;
        
        scanf("%lld",&n);
        
        printf("%lld\n",(long long)func(n+1));
    }
    
    return 0;
}
