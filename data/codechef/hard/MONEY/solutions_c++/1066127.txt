#include<iostream>
#include<vector>
#include<math.h>
#include<map>

using namespace std;

map <long long,long long> M;
vector <long long> Vec;
vector <unsigned int> all_primes;

bool check(unsigned int n)
{

    double p=pow(n,0.5);

    for(int i=0;i<all_primes.size();i++)
    {
        if(all_primes[i]<=p)
        {
            if(n%all_primes[i]==0)
            return 0;
        }
        else
        break;
    }

    return 1;
}


void factor(long long n)
{
    Vec.push_back(1);
    
    for(long long i=2;i<=pow((long double)n,0.5);i++)
    {
        if(n%i==0)
        {
            if(i!=n/i)
            {
                Vec.push_back(i);     Vec.push_back(n/i);
            }
            else
            Vec.push_back(i);
        }
    }
}

void search(long long n,vector <long long> &v)
{
    for(int i=0;i<Vec.size();i++)
    {
        if(n%Vec[i]==0&&n!=Vec[i])
        v.push_back(Vec[i]);
    }
}

long long func(long long n)
{
    if(n==1)
    return 1;
    
    if(M[n]!=0)
    return M[n];
    
    vector <long long> v;
    
    search(n,v);
    
    long long result=0;
    
    for(int i=0;i<v.size();i++)
    result+=func(v[i]);
    
    M[n]=result;
    
    return result;
}
