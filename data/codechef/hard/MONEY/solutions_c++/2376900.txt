#include <cstdio>
#include <algorithm>
#include <cstring>
#include <vector>
using namespace std;

#define MXN 500005
#define SI(x) scanf("%d",&x)
#define PI(x) printf("%d\n",x)
#define PB(x) push_back(x)
typedef long long LL;
 
// Sorted list of factor of (n+1)
vector<int> F;
 
// This array will be filled like a sieve is filled. D[i] denotes the number of ways if we put F[i] as the last element in the current sequence
LL D[MXN];
 
// Input n, as for case 2^31-1 , n+1 may overflow I have used longs
LL n;
 
void solve()
{
	// finds all the factors of n+1
	n++;
	for(LL i=1ll;i*i<=n;i++)
	{
		if(n%i==0)
		{
			F.PB((int)(i));
			LL j = n/i;
			if(i!=j && j!=n) 
			{
				F.PB((int)(j));
			}
		}
	}
	sort(F.begin(),F.end());
}
 
int main()
{
	int t;
	SI(t);
 
	while(t-- >0)
	{	
		scanf("%lld",&n);
		F.clear();
		solve();
		int f = F.size();
 
 
		for(int i=0;i<f;i++){
			D[i] = 1;
		}
 
		int m = F[f-1];
 
		for(int i=1;i<f;i++)
		{
			int x = F[i];
			for(int j=i+1;j<f;j++)
			{		
				int y = F[j];
				if(y%x==0)
				{
					D[j]+=D[i];
				}
			}
		}
		
		LL ans = 0;
 
		for(int i=0;i<f;i++)ans+=D[i];
		printf("%lld\n",ans);
	}
	return 0;
}
