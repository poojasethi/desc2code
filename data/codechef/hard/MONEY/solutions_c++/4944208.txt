/*
  Data Types ka chutiyapa.
  Change unsinged int into int and get a TLE.
*/

#include <bits/stdc++.h>
using namespace std;
map<unsigned int,int> mp;
int func(unsigned int n)
{
    int v = mp[n];
    if(v > 0) return v;
    else
    {
        int ways = 1;
        unsigned int i;
        for(i = 2 ; i*i <n ; i++)
        {
            if(n%i == 0)
            {
                ways = ways + func(i) + func(n/i);
            }
        }
        if(i*i == n) ways = ways + func(i);
        mp[n] = ways;
        return ways;
    }
}
int main()
{
    mp[1] = 1;
    int tc,t;
    scanf("%d",&tc);
    for(t = 1 ; t<=tc ; t++)
    {
        unsigned int n;
        scanf("%u",&n);
        int ans = func(n+1);
        printf("%d\n",ans);
    }
    return 0;
}
