#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
long long dp[500000];
long long factor[500000];
int main()
{
  int tests;
  scanf ("%d", &tests);
  while (tests--) {
    long long n;
    scanf ("%Ld", &n);

    int factors = 0;
    for (long long i = 2; i*i <= n+1; i++) if ((n+1)%i == 0) {
	factor[factors++] = i;
	if (i*i != n+1) factor[factors++] = (n+1)/i;
      }
    sort(factor, factor + factors);
    int fsz = factors;
    for (int i = 0; i < fsz; i++) dp[i] = 1;
    for (int i = 0; i < fsz; i++)
 	for (int j = i-1; j >= 0; j--) if (factor[i]%factor[j] == 0) dp[i] += dp[j];
    
    long long ans = 1;
    for (int i = 0; i < fsz; i++) ans += dp[i];
    printf ("%Ld\n", ans);
  }
  return 0;
}
