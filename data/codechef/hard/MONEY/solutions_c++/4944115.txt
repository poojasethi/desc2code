#include"map"
#include"stdio.h"
#define MAX 100000
using namespace std;
map<unsigned int,int> function;
int lookup[MAX];
//int iter;
int find (unsigned int N)
{
  if(N<MAX)return lookup[N];
  //int k=function[N];
  //if(k>0)return k;
  if(function.find(N) != function.end()) return function[N];
  int sum=1;
  unsigned int it=2;
  //iter++;
  while(it*it<N)
  {
    if(N%it==0)
      sum+=find(N/it)+find(it);
    it++;
  }
  if(it*it==N)
    sum+=find(it);
  function[N]=sum;
  return sum;
}
int main()
{
  int T;
  scanf("%d",&T);
  function[1]=1;
  for(int x=0;x<MAX;x++) lookup[x]=0;
  lookup[1]=1;
  for(int i=1;i<MAX;i++)
  {
    for(int j=i<<1;j<MAX;j+=i)
    {
      lookup[j]+=lookup[i];
    }
  }
  while(T--)
  {
    unsigned int N;
    scanf("%u",&N);
    int k=find(N+1);
    printf("%d\n",k);
  }
}
