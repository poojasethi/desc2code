n_tests = input()
for tc in xrange(n_tests):
	n_piles = input()
	nums = map(int, raw_input().strip().split(' '))
	
	xorval = 0
	for num in nums:
		xorval ^= num

	if xorval == 0:
		print 'First'
	else:
		if n_piles % 2 == 0:
			print 'First'
		else:
			print 'Second'
