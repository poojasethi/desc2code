for _ in range(int(raw_input())):
    n = int(raw_input())
    arr = map(int, raw_input().split())
    count = [0] * 501
    odd, even, xorr = 0, 0, 0
    for i in arr:
        count[i] += 1
        xorr ^= i
    for i in count[1:]:
        if i == 0:
            continue
        if i % 2 == 0:
            even += 1
        else:
            odd += 1
    # print even, odd
    if xorr == 0:
        print "First"
    elif odd % 2 == 1:
        print "Second"
    else:
        print "First"
