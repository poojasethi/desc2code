#include <iostream>
using namespace std;

int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--)
	{
	int n;
	cin>>n;
	int ar[n];
	for(int i=0;i<n;i++)
	cin>>ar[i];
	int ans=ar[0];
	for(int i=1;i<n;i++)
	ans=ans^ar[i];
	if(ans==0)
	{
	cout<<"First"<<endl;
	}
	else
	{
	if(n%2==0)cout<<"First"<<endl;
	else cout<<"Second"<<endl;
	}
	}
	
	return 0;
}
