#include <iostream>
#include<stdio.h>
#include <cstring>
using namespace std;

int main()
{
    
   int t,i,a[100005];
   long x,n;
   cin>>t;
   while(t--)
   {
       cin>>n;
       for(i=0;i<n;i++)
       cin>>a[i];
       x=a[0];
       for(i=1;i<n;i++)
       {
           x=x^a[i];
       }
       if(x==0)
       printf("First\n");
       else
       {
           if(n&1)
           printf("Second\n");
           else
           printf("First\n");
       }

   }
    return 0;
}
