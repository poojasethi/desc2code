using namespace std;

int main()
{
	ll t, n, i;
	sl(t);
	while(t--)
	{
		sl(n);
		ll arr[n];
		rep(i, n)
		{
			sl(arr[i]);
		}
		ll ans = 0;
		rep(i, n)
		{
			ans ^= arr[i];
		}
		if(ans == 0)
		{
			cout << "First\n";
			continue;
		}
		n % 2 ? cout << "Second\n" : cout << "First\n";
	}
    rt 0;
}
