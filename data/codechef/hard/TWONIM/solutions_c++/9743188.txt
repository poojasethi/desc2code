#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t,n,v,win,a;
	cin>>t;
	while(t--)
	{
		cin>>n;
		int i;
		v = 0;
		for(i=0;i<n;i++)
		{
			cin>>a;
			v = v^a;
		}
		int win = 1;
		if(v!=0)
			if(n%2==1)
				win = 0;
		if(win==1)
			cout<<"First";
		else
			cout<<"Second";
		cout<<endl;
	}
	return 0;
}
