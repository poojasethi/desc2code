#include <bits/stdc++.h>

using namespace std;

#define boost ios_base::sync_with_stdio(false);cin.tie(0);cout.tie(0)

typedef long long ll;

int main(){
	boost;
	ll T,N,i,j,k;
	cin >> T ;
	while(T--){
		cin >> N ;
		ll num,sum=0;
		for(i = 0 ; i < N ; i++){
			cin >> num;
			sum = sum ^ num;
		}
		if(sum == 0)
			cout << "First" << endl;
		else{
			if(N%2)
				cout << "Second" << endl;
			else
				cout << "First" << endl;
		}
	}
}
