#include<iostream>
using namespace std;

int main() {
	
	int t,N;
	cin >> t;
	while(t--) {
		cin >> N;
		int ans = 0;
		for (int i = 0; i < N; i++) {
			int a;cin >> a;
			ans ^=a;
		}
		if (ans == 0) {
			cout << "First" << "\n";
		} else {
			if (N % 2  ==0 ) {
				cout << "First" << "\n";	
			}	
			else {
				cout << "Second" << "\n";	
			}
		}
	}
	return 0;

}
