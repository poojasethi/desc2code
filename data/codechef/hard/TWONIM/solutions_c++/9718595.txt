#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int ans=0;
		for(int i=0;i<n;i++)
		{
			int a;
			cin>>a;
			ans=ans^a;
		}
		if(ans==0)
			cout<<"First"<<endl;
		else
		{
			if(n%2==1)
				cout<<"Second"<<endl;
			else
				cout<<"First"<<endl;
		}

	}

	return 0;
}
