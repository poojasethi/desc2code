#include<iostream>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int b,i;
        int xorarr=0;
        for(i=0;i<n;i++)
            {
                cin>>b;
                xorarr=xorarr^b;
            }
        if (xorarr==0)
            {
                cout<<"First\n";
                continue;
            }


        if (n%2==0)
            {
                cout<<"First\n";
            }
        else
            {
                cout<<"Second\n";
            }


    }
    return 0;
}
