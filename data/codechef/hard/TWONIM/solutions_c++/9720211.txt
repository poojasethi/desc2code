#include<iostream>

#include<bits/stdc++.h>
using namespace std;
int main()
{
	long int t,a[502],n,i,k;
	cin>>t;
	while(t--)
	{
		cin>>n;
		cin>>a[0];
		k=a[0];
		for(i=1;i<n;i++)
		{
			cin>>a[i];	
			  k=a[i]^k;	
			
		}
		if(k==0)
		{
			cout<<"First"<<endl;
		}
		else
		{
			if(n%2==0)
			{
				cout<<"First"<<endl;
			}
			else
			cout<<"Second"<<endl;
		}
	}
	//getch();
	return 0;
}
