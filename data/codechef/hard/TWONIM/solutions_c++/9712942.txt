#include <bits/stdc++.h>
using namespace std;
const int mx = 600;
int main(int argc, char const *argv[]) {
    int t;
    scanf("%d", &t);
    while (t--) {
    	int a[mx] = {};
    	int n;
    	scanf("%d", &n);
    	int xo = 0;
    	for (int i = 1; i <= n; ++i) {
    		int x;
    		scanf("%d", &x);
    		xo ^= x;
    		a[x]++;
    	}
    	if (xo == 0) printf("First\n");
    	else {
    		xo = 0;
    		for (int i = 1; i < mx; ++i) xo ^= (a[i] % 2);
    		if (xo == 0) printf("First\n");
    		else printf("Second\n");
    	}
    }
	return 0;
}
