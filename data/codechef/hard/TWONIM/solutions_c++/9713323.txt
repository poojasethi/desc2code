using namespace std;
#define N (int)(5e2+4)
int a[N];
int cnt[N];
void solve(){
   int n;
   cin>>n;
   int Xor=0;
   mem(cnt,0);
   for(int i=1;i<=n;i++){
       cin>>a[i];
       Xor^=a[i];
       cnt[a[i]]++;
   }
   if(Xor==0){
      printf("First\n");
      return;
   }
   int over=1;
   while(over<=n){
        int fd=0;
        for(int i=1;i<=500;i++){
            if(cnt[i]==0)
                continue;
            int nx=Xor^i;
            if(nx==0){
                continue;
            }
            else{
                cnt[i]--;
                Xor^=i;
                fd=1;
                break;
            }
        }
        if(!fd){
           if(over%2)
           printf("Second\n");
           else
           printf("First\n");
           return;
        }
        over++;
   }
   if(n%2)
      printf("Second\n");
    else
      printf("First\n");

}
int main(){
   //ios_base::sync_with_stdio(false);
   int t=1;
   cin>>t;
   for(int i=1;i<=t;i++){
       //printf("Case #%d: ",i);
       solve();
   }
   return 0;
}
