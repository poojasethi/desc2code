#include <bits/stdc++.h>
#define maxn 501

using namespace std;

int n, a[maxn];

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    ios::sync_with_stdio(false);
    int T; cin >> T;
    while (T--) {
        cin >> n;
        for(int i=1;i<=n;i++) cin >> a[i];
        int ans=a[1];
        for(int i=2;i<=n;i++) ans^=a[i];
        if (ans==0) cout << "First\n"; else {
            if (n%2) cout << "Second\n"; else cout << "First\n";
        }
    }
}
