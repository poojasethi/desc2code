#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <queue>
#include <stack>
#include <set>
#include <cmath>
#pragma comment(linker, "/STACK:16777216")
#define inf 1000000000
using namespace std;
typedef long long L;
typedef unsigned long long UL;
int n1,n2,A[51],B[51],dp[51][51][61];
const int lim=60;

int f(int a,int b,int t){
    if(t>lim)return inf;
    if(a==n1&&b==n2)
        return t;
    if(dp[a][b][t]!=0)return dp[a][b][t];

    int res=inf;
    if(a<n1){
        res=min(res,f(a+1,b,max(0,A[a+1]-t))+t);
        res=min(res,f(a+1,b,t+A[a+1]));
    }
    if(b<n2){
        res=min(res,f(a,b+1,max(0,B[b+1]-t))+t);
        res=min(res,f(a,b+1,t+B[b+1]));
    }
    dp[a][b][t]=res;
    return res;
}

int main()
{
    cin>>n1>>n2;
    for(int i=1;i<=n1;i++)cin>>A[i];
    for(int i=1;i<=n2;i++)cin>>B[i];

    cout<<f(0,0,0)<<endl;
    return 0;
}
