#include <cstdio>
#include <cstring>
#include <algorithm>
#define SI ({int x;scanf("%d",&x);x;})
using namespace std;
 
const int inf = 2140000000;
 
int dp[51][51][61];
int c1[51],c2[51];
int n1,n2;
 
int get(int p1, int p2, int t)
{
	if(p1==0 && p2==0) return t;
 
	int &r = dp[p1][p2][t];
	if(r!=-1) return r;
	
	r = inf;
 
	if(p1>0)
	{
		r = min ( r , min(c1[p1],t) + get(p1-1,p2,abs(c1[p1]-t)) );
	}
	
	if(p2>0)
	{
		r = min ( r , min(c2[p2],t) + get(p1,p2-1,abs(c2[p2]-t)) );
	}
	
	return r;
}
 
int main()
{
	int i;
	n1=SI; n2=SI;
	memset(dp,-1,sizeof dp);
	for(i=1;i<=n1;i++) c1[i]=SI;
	for(i=1;i<=n2;i++) c2[i]=SI;
	printf("%d\n", get(n1,n2,0) );
	return 0;
}
