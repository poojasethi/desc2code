#include<iostream>
#include<vector>
#include<cstdlib>
using namespace std;

const int MAXN = 51;
const int MAXT = 61;
const int INF = (int)1e9;
int dp[MAXN][MAXN][MAXT];
int as[MAXN];
int bs[MAXN];


int n1,n2;

int calc(int a,int b,int c)
{
	//cout<<a<<" "<<b<<" "<<c<<" "<<endl;
	if(c >= MAXT) return INF;
	if(a == n1 && b == n2) return c;
	int res = 1000000000;
	int x;
	if(dp[a][b][c] != 0) return dp[a][b][c];

	else {
		if(a<n1) {
			x = calc(a+1,b,max(as[a]-c,0)) + c;
			res = min(res,x);
			x = calc(a+1,b,as[a] + c);
			res = min(res,x);
		} 
		if(b<n2) {
			x = calc(a,b+1,max(bs[b]-c,0)) + c;
			res = min(res,x);
			x = calc(a,b+1,bs[b]+c);
			res = min(res,x);
		}
	}
	dp[a][b][c] = res;
	return res;
}

int main()
{
	cin>>n1;
	cin>>n2;
	
	int x;
	
	for(int i=0;i<n1;i++) cin>>as[i];
	for(int i=0;i<n2;i++) cin>>bs[i];
	
	cout<<calc(0,0,0)<<endl;
	return 0;
}
