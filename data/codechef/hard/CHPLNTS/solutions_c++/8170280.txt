
#include<iostream>

#include<stack>

#define ll long long

using namespace std;

int main(){

	int t;

	cin>>t;

	while(t--){

		stack<int> stk;

		int n;

		cin>>n;

		int a;

		ll ans = 0;

		cin>>a;

		stk.push(a);

		for(int i=2;i<=n;i++){

			cin>>a;

			if(stk.top()<=a) stk.push(a);

			else{

				ans += (stk.top()-a);

				while(!stk.empty() && stk.top()>=a) stk.pop();

				stk.push(a);

			}

		}

		a = stk.top();
		int b;

		while(!stk.empty()){

			b = stk.top();

			stk.pop();

		}

		ans += (a-b);

		cout<<ans<<endl;

	}

}
