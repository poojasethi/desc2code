#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
 
int main()
{
    //freopen("BT10.txt","r",stdin);
    //freopen("BTO10.txt","w",stdout);
    int T;
    scanf("%d", &T);
    while(T--)
    {
        int N;
        ll ans = 0;
        int min_val = -1;
        int curr_max = 0;
        scanf("%d", &N);
        for(int i = 0; i < N; i++)
        {
            int h;
            scanf("%d", &h);
            if(min_val == -1 || h < min_val)
                min_val = h;
            if(h > curr_max)
                curr_max = h;
            else if (curr_max > h)
            {
                ans += curr_max - h;
                curr_max = h;
            }
        }
        ans += curr_max - min_val;
        printf("%lld\n", ans);
    }
    return 0;
}
