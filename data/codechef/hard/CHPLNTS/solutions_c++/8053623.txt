#include <bits/stdc++.h>
using namespace std;
#define ll long long
int main() {
	// your code goes here
	int t;
	ll n;
	cin>>t;
	while(t--)
	{
		stack<ll> s;
		cin>>n;
		ll sum=0;
		for(int i=0;i<n;i++)
		{
			int a;
			cin>>a;
			//cout<<"size="<<s.size()<<endl;
			if(s.size()==0)
			{
				s.push(a);
			}else
			{
				if(a>s.top())
				s.push(a);
				else if(a<s.top())
				{
					sum+=s.top()-a;
					while(s.top()>=a)
					{
						s.pop();
						if(s.size()==0)
						break;
					}
					s.push(a);
				}
			}
			
		}
		if(s.size()>0)
		{
			ll temp=s.top();
			ll sm;
			while(s.size()>0)
			{
				sm=s.top();
				s.pop();
			}
			sum+=temp-sm;
		}
		cout<<sum<<endl;
	}
	return 0;
}
