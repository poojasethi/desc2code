#include <iostream>
#include <algorithm>

using namespace std;
int main(){
	int t;
	cin>>t;
	while(t--){
	string k;
	cin>>k;
	int l=k.length();
	int a[100010][3]={0};
	int b[100010][3]={0};
	for(int i=0;i<l/2;i++){
		if(k[i]==k[l-1-i])a[i][0]=1;
		if(k[i]==k[l-1-i-1])a[i][1]=1;
	}
	for(int i=1;i<l;i++) {
		a[i][0]+=a[i-1][0];
		a[i][1]+=a[i-1][1];
	}
	int ans=0;
	for(int i=0;i<(l+1)/2;i++){
		if(i!=0 && i!=l)
			{if(a[i-1][0]==(i) && (a[(l)/2-1][1]-a[i-1][1])==((l)/2-i)) {ans++;}}
		else{
			if(i==0 && (a[l/2-1][1])==l/2) {ans++;}
		}
	}
	for(int i=0;i<l;i++) {
	//	cout<<a[i][0]<<" "<<a[i][1]<<endl;
	}
	for(int i=0;i<l/2;i++)swap(k[i],k[l-1-i]);
	for(int i=0;i<l/2;i++){
		if(k[i]==k[l-1-i])b[i][0]=1;
		if(k[i]==k[l-1-i-1])b[i][1]=1;
	}
	for(int i=1;i<l;i++) {
		b[i][0]+=b[i-1][0];
		b[i][1]+=b[i-1][1];
	}
	for(int i=0;i<(l+1)/2;i++){
		if(i!=0 && i!=l)
			{if(b[i-1][0]==(i) && (b[(l)/2-1][1]-b[i-1][1])==((l)/2-i)) ans++;}
		else{
			if(i==0 && (b[l/2-1][1])==l/2) ans++;
		}
	}
	if(a[l/2][0]==l/2 && b[l/2][0]==l/2 && l%2==0) {ans++;}
	
	
	cout<<ans<<endl;
	}
}
