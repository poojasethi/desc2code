int cnt,ans;
int next[NN],f[NN];

void check(string s) {
	for(int i=1;i<s.size();i++) {
		swap(s[i],s[i-1]);
		//cout<<s<<"\n";
		//cout<<"initial "<<cnt<<"\n";
		if(s[i-1]==s[next[i-1]] && f[i-1]) {
			f[i-1]=0;
			f[next[i-1]]=0;
			cnt-=2;
		}
		else if(s[i-1]!=s[next[i-1]] && !f[i-1]) {
			f[i-1]=1;
			f[next[i-1]]=1;
			cnt+=2;
		}
		if(f[i]) {
			f[i]=0;
			f[next[i]]=0;
			cnt-=2;
		}
		//cout<<"final "<<cnt<<"\n";
		if(cnt==0) {
			//cout<<"possible\n";
			ans++;
		}
	}
}
