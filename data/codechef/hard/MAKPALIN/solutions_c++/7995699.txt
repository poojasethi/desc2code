#include<bits/stdc++.h>
using namespace std;

int make_palindrome(string s)
{
    int n=s.length();
    int i=-1;
    int j=n-1;
    int ans=0;
    int flag=0;
    while(i<=j)
    {
        if(i==j||i+1==j)
        {
            if(flag==0)
            {
                ans++;
            }
            break;
        }
        if(s[i+1]==s[j-1])
        {
            if(flag==0)
            {ans++;}

            if(s[i+1]!=s[j])
            {
                flag=1;
            }
            i++;
            j--;
        }
        else
        {
                ans=0;
               if(flag==1||s[i+1]!=s[j])
               {
                   break;
               }
               i++;
               j--;
        }
    }
    i=n;
    j=0;
    int lans=0;
    flag=0;
    while(i>j)
    {
        if(i==j||i-1==j)
        {
            if(flag==0)
            {
                lans++;
            }
            break;
        }
        if(s[i-1]==s[j+1])
        {
            if(flag==0)
            {lans++;}

            if(s[i-1]!=s[j])
            {
                flag=1;
            }
            i--;
            j++;
        }
        else
        {
                lans=0;
               if(flag==1||s[i-1]!=s[j])
               {
                   break;
               }
               i--;
               j++;
        }
    }
    ans=ans+lans;
    return(ans);
}
int main()
{
   int t;
   cin>>t;
   while(t--)
   {
       string s;
       cin>>s;
       int n=make_palindrome(s);
       cout<<n<<endl;
   }
}
