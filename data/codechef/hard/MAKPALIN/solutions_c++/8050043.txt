char a[100010];
int len,test,ans;

bool p1[100010],p2[100010],p3[100010];


void createp1()
 {
     int l = len/2;
     int r = (len+1)/2;
     
     if(a[l] == a[r])
          p1[l] = true;
     else
       p1[l] = false;
       
     l--;r++;
     
     for(;l>=1;l--,r++)
      {
          if(a[l] == a[r] && p1[l+1])
           p1[l] = true;
          
          else
          {
              p1[l] = false;
              break;
          }
      }
      
      while(l>=1)
       {
           p1[l] = false;
           l--;
       }
          
     
 }
 


void createp2()
 {
     int l = len/2+1;
     int r = (len+1)/2 + 1;
     
     if(a[l] == a[r])
          p2[l] = true;
     else
       p2[l] = false;
       
     l--;r++;
     
     for(;l>=2;l--,r++)
      {
          if(a[l] == a[r] && p2[l+1])
           p2[l] = true;
          
          else
          {
              p2[l] = false;
              break;
          }
      }
      
      while(l>=2)
       {
           p2[l] = false;
           l--;
       }
          
     
 } 
 

void createp3()
 {
     int num = (len+1)/2;
     
     p3[0] = true;
     
     int l=1,r=len;
     
     for(;l<=num;l++,r--)
      {
      
         if(a[l] == a[r] && p3[l-1])
           p3[l] = true;
          
          else
          {
              p3[l] = false;
              break;
          }
      }
      
      while(l<=num)
       {
           p3[l] = false;
           l++;
       }
     
 }

bool isp1(int l,int r)
 {
     if(l > r) return true;
     return p1[l];
 }
 
 bool isp2(int l,int r)
 {
     if(l > r) return true;
     return p2[l];
 }
 
 bool isp3(int l)
  {
      return p3[l];
  }


int algo2()
 {
     
     createp1();
     createp2();
     createp3();
     
     int k = (len+1)/2;
     
     int cnt = 0;
     
     if(isp1(1,len-1)) cnt++;
     if(isp2(2,len)) cnt++;
     
     for(int i=1,l=2,r=len-2;i<k;l++,r--,i++)
      if(isp1(l,r) && isp3(i)) cnt++;
      
     for(int i=1,l=3,r=len-1;i<k;l++,r--,i++)
      if(isp2(l,r) && isp3(i)) cnt++;
     
     if(len%2 == 0 && isp3(len/2))
      cnt++;
      
     return cnt; 
      
 }


bool palin(string s)
 {
     int r = s.size()-1;
     int l = 0;
     
     while(l<r)
      {
          if(s[l] != s[r])
           return false;
           
          l++;r--; 
      }
      
      return true;
     
 }

int algo1()
 {
     string s,x;
     for(int i=1;i<=len;i++)
      s.push_back(a[i]);
      
     int cnt = 0;

     for(int pos=0;pos<=len;pos++)

          for(char c='a';c<='z';c++)
           {
               x.clear();
               x = s;
               x.insert(pos,1,c);

               if(palin(x))

           }

 }
