#include <bits/stdc++.h>

using namespace std;

#define ll int

ll func(string s1,string s2)
{
	ll n,i,j,k,l,r;
	n=s1.length();
	i=0;
	while(i<(n-1) && s2[i]==s1[n-1-i]) i++;
	l=i;
	i=0;
	while(i<(n-1) && s2[n-2-i]==s1[i]) i++;
	r=i;
	ll ans=0;
	ans+= max(0,r-(n-1-l)+1);
	return ans;
}

void reverse(string& s)
{
	ll i,j,n=s.length();
	for(i=0;i<n/2;i++)
		swap(s[i],s[n-1-i]);
	return;
}


int main()
{
    int t;
    cin>>t;

    while(t--){
	string str,s1,s2;
	cin>>str;
	ll n=str.length(),ans=0;
	if(str.length()%2)
	{
		s1=str.substr(0,(n+1)/2);
		s2 = str.substr((n+1)/2,(n+1)/2-1);
		ans+=func(s1,s2);
		reverse(str);
		s1=str.substr(0,(n+1)/2);
		s2 = str.substr((n+1)/2,(n+1)/2-1);
		ans+=func(s1,s2);
	}
	else{
		s1=str.substr(0,n/2);
		reverse(s1);
		s2 = str.substr(n/2,n/2);
		if(s1==s2)
			ans++;
		s1=str.substr(0,n/2);
		s2 = str.substr(n/2+1,n/2-1);
		ans+=func(s1,s2);
		reverse(str);
		s1=str.substr(0,n/2);
		s2 = str.substr(n/2+1,n/2-1);
		ans+=func(s1,s2);

	}
	cout<<ans<<endl;
    }
	return 0;
}
