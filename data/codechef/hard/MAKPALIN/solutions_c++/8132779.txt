#include<bits/stdc++.h>
using namespace std;
 
const int N=100010;
 
char a[N];
bool lf[N],rf[N],ed[N];
int n;
 
void cal(bool *arr, int l, int r)
{
    int m=(l+r)/2+1;
    arr[m]=true;
    for(int i=m-1;i>=l;i--)
    {
        if(!arr[i+1])
            arr[i]=false;
        else
            arr[i]=(arr[i+1]&&(a[i]==a[l+r-i]));
    }
}
 
void calEnd()
{
    ed[0]=true;
    for(int i=1;i<=n/2;i++)
    {
        if(!ed[i-1])
            ed[i]=false;
        else
            ed[i]=(ed[i-1]&&(a[i]==a[n+1-i]));
    }
}
 
int comp(int pos)
{
    if(pos<=(n+1)/2)
    {
        return lf[pos]&&ed[pos-1];
    }
    else
    {
        pos=n+2-pos;
        return rf[pos+1]&&ed[pos-1];
    }
}
 
int com()
{
    cal(lf, 1, n-1);
  //  printf("%d\n",lf[3]);
    cal(rf, 2, n);
    calEnd();
    int ans=0;
    for(int i=1;i<=n+1;i++)
    {
        ans+=comp(i);
    }
 
    return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%s",a+1);
        n=strlen(a+1);
        int ans;
        if(n==1)
            ans=2;
        else
            ans=com();
        printf("%d\n",ans);
    }
    return 0;
}
