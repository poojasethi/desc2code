#include<cstdio>
const int n=100000;
long long res;
int i,j,Num,last,cnt,p,k,N,T;
int f[n+5],P[n+5],phi[n+5],step[n+5];
int main()
{
	for(i=2;i<=n;++i)
	{
		if(!f[i]) f[i]=P[++Num]=i,phi[i]=i-1;;
		for(j=1;j<=Num&&P[j]<=f[i]&&P[j]<=n/i;++j)
			f[i*P[j]]=P[j],phi[i*P[j]]=phi[i]*(P[j]-(P[j]!=f[i]));
	}
	for(i=3;i<=n;++i) step[i]=step[phi[i]]+1;
	for(scanf("%d",&T);T--;)
	{
		for(res=1,scanf("%d",&N);N--;)
		{
			scanf("%d%d",&p,&k);
			if(p==2) res+=(long long)k-1;
			else res+=(long long)step[p]*k;
		}
		printf("%lld\n",res);
    }
	return 0;
}
