#define inf 1000000000
#define MP make_pair
#define PI acos(-1.0)
#define eps 1e-9
using namespace std;
typedef long long i64;
typedef unsigned long long u64;
const int sz=100000;
int B[sz+1];
i64 C[sz+1];
vector<pair<int,int> > V[sz+1];

int main()
{

    int num,cnt;
    int T,m,p,k,ev;

    for(int i=2;i<=sz;i++)
        if(!B[i]){for(int j=i;j<=sz;j+=i){V[j].push_back(MP(i,-1));B[j]=1;}}


    for(int i=2;i<=sz;i++){
        num=i;
        for(int j=0;j<V[i].size();j++){
            cnt=0;
            while(num%(V[i][j].first)==0){num/=V[i][j].first;cnt++;}
            V[i][j].second=cnt;
        }
    }

    scanf("%d",&T);

    while(T--){
        scanf("%d",&m);

        ev=0;

        while(m--){scanf("%d%d",&p,&k); C[p]+=k; ev+=(p==2);}

        for(int i=sz;i>=3;i--)
            if(C[i]){
                for(int j=0;j<V[i-1].size();j++)C[V[i-1][j].first]+=(V[i-1][j].second*C[i]);
                C[i]=0;
            }

        printf("%lld\n",C[2]+(1-ev));
        C[2]=0;
    }

    return 0;
}
