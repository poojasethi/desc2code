using namespace std;
typedef long long i64;
typedef unsigned long long u64;
const int sz=2013;
int G[sz][sz],B[sz];

int main()
{
    int cnt,res,T,n,k;

    for(int i=1;i<sz;i++){
        memset(B,0,sizeof(B));
        for(int j=1;j<=i;j++){
            G[i][j]=G[i][j-1];
            B[G[i-j][min(i-j,j)]]=1;
            while(B[G[i][j]])G[i][j]++;
        }
    }

    cin>>T;
    while(T--){
        cin>>n>>k;
        res=0;
        for(int i=1;i<=n;i++){
            cin>>cnt;
            res^=G[cnt][min(cnt,k)];
        }
        if(res)cout<<"Nancy"<<endl;
        else cout<<"Zeta"<<endl;
    }

    return 0;
}
