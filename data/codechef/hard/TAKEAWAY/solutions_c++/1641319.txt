#include <iostream>

#define M 2100

using namespace std;

long d[M][M],u[2*M][M],m[2*M],n,k,a,b,f;

void din(void){
    for (long j=1; j<M; ++j)
    for (long i=0; i<M; ++i){
        if (i<j)
        d[i][j]=d[i][j-1];
        else{
            while (u[i][m[i]])
            ++m[i];
            d[i][j]=m[i];
        }

        u[i+j][d[i][j]]=1;
    }
}

void kill(void){
    cin>>n>>k;
    f=0;
    for (long i=0;i<n; ++i){
        cin>>a;
        b=min(a,k);
        f^=d[a][b];
    }

    cout<< (f!=0 ? "Nancy\n":"Zeta\n");

}

int main()
{
    ios_base::sync_with_stdio(0);
    din();
    long t;
    cin>>t;
    while (t--)
    kill();
    return 0;
}
