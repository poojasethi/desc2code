#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define INF 0x3f3f3f3f

using namespace std;
int dp[2015][2015],visited[2015];

int main()
{
    int t;
    scanf("%d",&t);
    for(int i=1;i<=2012;i++)
    {
        int grundy=0;
        for(int k=1;k<=i;k++)
        {
            int temp=dp[i-k][(k>i-k)?i-k:k];
            visited[temp]=i;
            if(temp==grundy)
            {
                while(grundy<2013&&visited[grundy]==i)
                    grundy++;
            }
            dp[i][k]=grundy;
        }
    }
    while(t--)
    {
        int n,k;
        scanf("%d%d",&n,&k);
        int ans=0;
        for(int i=0;i<n;i++)
        {
            int x;
            scanf("%d",&x);
            ans^=dp[x][k>x?x:k];
        }
        if(ans==0)
            printf("Zeta\n");
        else
            printf("Nancy\n");
    }
    return 0;
}
