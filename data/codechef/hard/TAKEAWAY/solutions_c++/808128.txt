const int MX = 2020;
int g[MX+10][MX+10];
bool used[MX+10];

void precalc()
{
	FOR(i, 1, MX)
	{
		memset(used, 0, sizeof used);		
		int mex = 0;
		FOR(j, 1, i)
		{
			used[g[i-j][j]] = true;
			while(used[mex])
				++mex;
			g[i][j] = mex;
		}
		FOR(j, i+1, MX)
			g[i][j] = g[i][j-1];
	}
}
