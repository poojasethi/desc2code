int main() {
    input();
    F(i,0,2020) dp[0][i] = 0;
    F(i,1,2020) {
        bool mex[2020] = {0};
        int last  = 0;
        F(j,1,i+1) {
            mex[dp[i-j][min(i-j,j)]] = 1;
            while ( mex[last] ) last++;
            dp[i][j] = last;
        }
    }
    int t; S(t);
    while ( t-- ) {
        int n; S(n);
        int k; S(k);
        int ans = 0;
        F(i,0,n) {
            int x; S(x);
            ans ^= dp[x][min(k,x)];
        }
        printf("%s\n",(ans?"Nancy":"Zeta"));
    }
    return 0;
}
