void precalc()
{
	for(int i = 1; i<2013; ++i)
	{
		FILL(mark,0);
		int cur = 0;
		for(int j = 1; j<=i; ++j)

			int x = i-j, y = MIN(x,j);
			mark[prec[x][y]] = 1;
			while(mark[cur]) ++cur;
			prec[i][j] = cur;
		}
	}
	return;
}
