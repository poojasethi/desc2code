using namespace std;

inline void readint(int &x)
{
	register int c = getchar_unlocked();
	x = 0;
	for(;(c<48 || c>57);c = getchar_unlocked());
	for(;c>47 && c<58;c = getchar_unlocked()) {x = (x<<1) + (x<<3) + c - 48;}
}

inline void writeint(int x, char endchar)
{
	if(x == 0)
		putchar_unlocked('0');
	else
	{
		int cnt = 0;
		char ch[20];
		while(x != 0)
		{
			ch[cnt++] = x%10 + '0';
			x /= 10;
		}
		for(int i=cnt - 1; i>=0; --i)
			putchar_unlocked(ch[i]);
	}
	putchar_unlocked(endchar);
}

inline void writestring(const char* ch)
{
	while(*ch != '\0')
		putchar_unlocked(*(ch++));
	putchar_unlocked('\n');
}

int val[2013][2013];
int cal[2012];

inline void calc_vals()
{
	int i,j,ptr;
	for(i=1; i<=2012; ++i)
	{
		for(j=0; j<=i; ++j)
			cal[j] = 0;
		ptr = 0;
		for(j=1; j<=i; ++j)
		{
			cal[val[i-j][j < i-j ? j : i - j]] = 1;
			while(cal[ptr] == 1)ptr++;
			val[i][j] = ptr;
		}
		//for(j=i+1; j<=2012; ++j)
		//	val[i][j] = val[i][i];
	}
	return ;
}

int main()
{
	calc_vals();
	int t;
	readint(t);
	while(t--)
	{
		int n, k;
		readint(n);
		readint(k);
		int ans = 0;
		if(k > 2012)
			k = 2012;
		while(n--)
		{
			int temp;
			readint(temp);
			ans ^= val[temp][k < temp ? k : temp];
		}
		if(ans == 0)
			writestring("Zeta");
		else
			writestring("Nancy");
	}
	return 0;
}
