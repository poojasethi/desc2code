#define PB push_back
#define MAX 2020
using namespace std;
 
int grundy[MAX][MAX];
bool moves[MAX+1];
 
void calculateGrundy()
{
     for (int i=0; i<MAX; i++)
     {
          grundy[0][i] = 0;
          grundy[i][0] = 0;
     }
     
     for (int i=1; i<MAX; i++)
     {
         memset(moves,false,sizeof(moves));
         int minGrundy = 0;
         for (int j=1; j<=i; j++)    
         {
             moves[grundy[i-j][j]] = true;
             while (moves[minGrundy]) ++minGrundy;
             grundy[i][j] = minGrundy;
         }
     
         for (int j=i+1; j<MAX; j++)
         {
             grundy[i][j] = grundy[i][i];    
         }
     }     
     
} 
 
int main()
{
    calculateGrundy();
    int t;
    scanf("%d",&t);
    while (t--)
    {
          int n,k,ans = 0;
          scanf("%d%d",&n,&k);
          for (int i=0; i<n; i++)
          {
              int stones;
              scanf("%d",&stones);
              ans ^= grundy[stones][min(stones,k)];
          }   
    ans?printf("Nancy\n"):printf("Zeta\n");
    
    }
 
    return 0;    
}
