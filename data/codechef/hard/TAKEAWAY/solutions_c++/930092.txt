#include<iostream>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int grundy[2013][2013];

void preprocess()
{
    for(int n=1;n<=2012;n++)
    {
        bool temp[2013]={false}; //single array for each row
        int i=0;
        for(int k=1;k<=n;k++)
        {
            temp[grundy[n-k][min(k,n-k)]]=true;
            while(temp[i])
                i++;
            grundy[n][k]=i;
        }
    }
}

int main()
{
    int test,n,k,a;
    preprocess();
    scanf("%d",&test);
    while(test--)
    {
        int ans=0;
        scanf("%d%d",&n,&k);
        for(int i=0;i<n;i++)
        {
            scanf("%d",&a);
            ans^=grundy[a][min(a,k)];
        }
        (ans)?puts("Nancy"):puts("Zeta");
    }
    return 0;
}
