#include<bits/stdc++.h>
#define ll long long
#define mp make_pair
#define pb push_back
#define pi pair<int,int>
using namespace std;
int dp[2013][2013];
void solve(){
    for(int i=1;i<2013;i++){
        int mark[2013];
        int cnt = 0;
        memset(mark,0,sizeof(mark));
        for(int j=1;j<=i;j++){
            mark[dp[i-j][j]]=1;
            while(mark[cnt]) cnt++;
            dp[i][j] = cnt;
        }
        for(int j=i+1;j<2013;j++) dp[i][j] = dp[i][i];
    }
}
int main() {
    solve();
    int t;
    scanf("%d",&t);
    while(t--){
        int n,k;
        scanf("%d%d",&n,&k);
        int ans = 0;
        for(int i=0;i<n;i++){
            int x;scanf("%d",&x);
            ans = ans ^ dp[x][min(k,x)];
        }
        if(ans==0) printf("Zeta\n");
        else printf("Nancy\n");
    }
    return 0;
}
