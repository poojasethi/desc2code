#include<iostream>
//#include<conio.h>
using namespace std;
int grundy[2013][2013];
void find_grundy()
{
int count;
for(int i=1;i<2013;i++)
grundy[0][i]=0;
for(int i=1;i<2013;i++)
{
bool next_visited[2013]={false};
count=0;
for(int j=1;j<=i;j++)
{
next_visited[grundy[i-j][min(j,i-j)]]=true;
while(next_visited[count]==true)
count++;
grundy[i][j]=count;
}
}
}

int main()
{
find_grundy();
int ans;
int x,t,n,k;
scanf("%d",&t);
while(t-->0)
{
ans=0;
scanf("%d%d",&n,&k);
while(n-->0)
{
scanf("%d",&x);
ans=ans^grundy[x][min(x,k)];
}
if(ans==0)
printf("Zeta\n");
else
printf("Nancy\n");
}
//getch();
return 0;
}
