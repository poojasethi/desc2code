typedef pair<int, int>PII;

#define NN 2015

int N,K;
bool check[NN];
int grundy[NN][NN];

int min(int a, int b){return a<b?a:b;}

int mex()
{
    for(int i=0; i<NN; i++)
    {
        if(!check[i])return i;
    }
}

void precompute()
{
  grundy[0][0]=0, grundy[1][1] = 1;
  for(int n=2; n<2013; n++)
  {
      memset(check, 0, sizeof(check));
      for(int k=1; k<n+1; k++)
      {
        check[grundy[n-k][min(n-k,k)]] = 1;
        grundy[n][k] = mex();
      }
  }
}

void solve()
{
    int ans=0, A;
    scanf("%d%d", &N, &K);
    while(N--)
    {
        scanf("%d", &A);
        ans^= grundy[A][min(A, K)];
    }
    printf("%s\n", ans==0?"Zeta":"Nancy");
}

int main()
{
    #ifndef ONLINE_JUDGE

   // freopen("output.txt","w",stdout);
    #endif
    precompute();
     int t;
    scanf("%d", &t);
    while(t--)
    {
        solve();
    }


}
