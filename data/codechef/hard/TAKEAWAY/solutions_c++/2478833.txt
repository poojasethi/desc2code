#define MAXN 2013
bool used[MAXN];
int grundy[MAXN][MAXN];
void Preprocess()
{
	for (int i = 1; i < MAXN; i++) {
		memset(used,false,sizeof(used));
		int next  =0;
		for (int j = 1; j < MAXN; j++) {
			if(i>=j)
			{
				used[grundy[i-j][j]] = true;
				while(next<MAXN&&used[next])
					next+=1;
			}
			grundy[i][j] = next;
		}
	}
}
int main()
{
    int T;
    si(T);
    Preprocess();
    while(T--)
    {
	    int N,K;
	    si(N),si(K);
	    int x[N];
	    for (int i = 0; i < N; i++) {
		si(x[i]);
	    
	    }
	    int res = 0;
	    for (int i = 0; i < N; i++) {
		    res^=grundy[x[i]][min(K,x[i])];
	    }
	    if(res)
		    cout<<"Nancy"<<endl;
	    else
		    cout<<"Zeta"<<endl;
    }
    return 0;
}
