#include<cstdio>

int dp[2000 + 20][2000 + 20];
int mark[40];

int main()
{
    int i,j,min,u,n,x,t,k,f;
    for(i = 1 ; i<=2018 ; i++)
    {

        for(j = 0 ; j<40; j++)
        {
            mark[j] = 0;
        }

        min = 0;

        for(j = 1 ; j<=i ; j++)
        {
            mark[dp[i-j][j]] = 1;

            if(mark[min] == 1)
            {
                while(mark[min] == 1)
                {
                    min++;
                }
            }
            dp[i][j] = min;
        }

        for(j = i+1 ; j<=2018; j++)
        {
            dp[i][j] = dp[i][i];
        }
    }

    scanf("%d",&t);
    for(u = 1 ; u<= t ; u++)
    {
        scanf("%d %d",&n,&k);
        f = 0;

        for(i = 1 ; i<=n ;i++)
        {
            scanf("%d",&x);
            if(k > x)
            {
                f = f^dp[x][x];
            }
            else
            {
                f = f^dp[x][k];
            }
        }

        if(f == 0)
        {
            printf("Zeta\n");
        }
        else
        {
            printf("Nancy\n");
        }
    }
    return 0;
}
