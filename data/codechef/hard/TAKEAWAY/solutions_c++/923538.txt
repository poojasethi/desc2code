#include <stdio.h>
int i,n,t,j,it,k,a,r,f[2020][2020],w[2020];
int main() {
  for (i=1; i<=2012; i++) {
    ++it;
    for (j=1; j<=2012; j++) {
      if (i>=j) w[f[i-j][j]]=it;
      for (f[i][j]=f[i][j-1]; w[f[i][j]]==it; f[i][j]++);
    }
  }
  scanf("%d",&t);
  while (t--) {
    scanf("%d%d",&n,&k);
    if (k>2012) k=2012;
    for (r=i=0; i<n; i++) {
      scanf("%d",&a);
      r^=f[a][k];
    }
    puts(r?"Nancy":"Zeta");
  }
  return 0;
}
