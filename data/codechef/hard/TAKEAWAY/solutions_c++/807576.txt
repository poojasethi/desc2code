using namespace std;

typedef long long ll;

#define M 2050
int n, K;
int a[M];
int grundy[M][M];
bool used[M];

int main(){
  //cin.tie(0);
  //ios_base::sync_with_stdio(false);

  for(int k = 0; k < M; k++)grundy[0][k] = 0;
  for(int i = 1; i < M; i++){
    for(int j = 0; j < M; j++)used[j] = false;
    int mx = 0;
    for(int k = 1; k <= i; k++){
      used[grundy[i-k][k]] = true;
      while(used[mx])mx++;
      grundy[i][k] = mx;
    }
    for(int k = i+1; k < M; k++)
      grundy[i][k] = mx;
  }
  
  int T;
  scanf("%d", &T);
  for(;T;T--){
    scanf("%d%d", &n, &K);
    for(int i = 0; i < n; i++)
      scanf("%d", a+i);
    int g = 0;
    for(int i = 0; i < n; i++)
      g = g ^ grundy[a[i]][min(K, a[i])];
    printf("%s\n", g?"Nancy":"Zeta");
  }

  return 0;
}
