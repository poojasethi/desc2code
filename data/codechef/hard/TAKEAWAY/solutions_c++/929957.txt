#include <iostream>
#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
int gr[2013][2013];

void pre()
{
    for ( int n = 1; n <= 2012 ; n++){
        bool tmp[2013]={false};
        int idx = 0;
        for ( int k = 1; k <= n ; k++){
            tmp[gr[n-k][min(k,n-k)]] = true ;
            while(tmp[idx])idx++;
            gr[n][k] = idx;   
        }   
    }   
}

int main()
{
    pre();
    int T;
    scanf("%d",&T);
    while ( T-- ){
        int n,k;
        scanf("%d%d",&n,&k);
        int a;
        int x=0;
        for ( int i = 0;i< n; i++)scanf("%d",&a),x^=gr[a][min(a,k)];
        if ( x )printf("Nancy\n");
        else printf("Zeta\n");   
    }  
return 0; 
}
