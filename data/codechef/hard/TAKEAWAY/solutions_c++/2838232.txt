#include <iostream>
#include<cstdio>
#include<string.h>
using namespace std;

int grundy[2020][2020];

void pre()
{
    int vis[2030];
    int i,j,x,y,mex;
    for(i=1;i<=2012;i++)
    {
        mex=0;
        memset(vis,0,sizeof(vis));
        for(j=1;j<=i;j++)
        {
            x=i-j;
            y=min(x,j);
            vis[grundy[x][y]]=1;
            while(vis[mex]==1)
            mex++;
            grundy[i][j]=mex;
        }
    }
    return;
}

int main()
{
    int test,n,k,i,x,ans;
    scanf("%d",&test);
    pre();
    while(test--)
    {
        scanf("%d%d",&n,&k);
        ans=0;
        for(i=1;i<=n;i++)
        {
            scanf("%d",&x);
            ans=ans^grundy[x][min(x,k)];
        }
        if(ans!=0)
        printf("Nancy\n");
        else
        printf("Zeta\n");
    }
    return 0;
}
