#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>

#define MOD 1000000007
typedef long long ll;

using namespace std;


int gcd(int a, int b) {
	if(a%b==0)
	{
		return b;
	}
	else
	{
		return gcd(b,a%b);
	}
}

ll pow(ll base, ll n, ll m)
{
	if(n==0)
	{
		return(1);
	}
	if(n==1)
	{
		return(base%m);
	}
	ll tp = pow(base, n/2, m)%m;
	if(n%2==0)
	{
		return((tp*tp)%m);
	}
	else
	{
		return((((tp*tp)%m)*base)%m);
	}
}

ll mmi(ll n, ll m)
{
	return(pow(n, m-2, m));
}

int main() {
	int t;

	ll temp;

	{
		temp = fac[i-1];
		temp = (temp*(i))%MOD;
		fac[i]= temp;

		// temp = facn[i-1];
		// temp = (temp*(i))%MOD;
		// facn[i]= temp;
		//cout<<fac[i]<<" "<<facn[i]<<endl;
	}
	long long int n,k,ans,tp,num,den;
	while(t--)
	{
		scanf("%lld",&n);
		scanf("%lld",&k);

		//ans=n;
		//tp=1;
		num = fac[n+k+1];
		den = fac[k+2];
		// cout<<den<<endl;
		den = (den*fac[n-1])%MOD;
		// cout<<den<<endl;
		den = mmi(den, MOD);
		// cout<<den<<endl;
		ans = (den*num)%MOD;
		// cout<<ans<<endl;
		ans = (2*ans)%MOD;
		// cout<<ans<<endl;
		ans = (ans-n+MOD)%MOD;
		cout<<ans<<endl;
	}
}
