#include <iostream>
#include <stdio.h>

#define mod 1000000007
#define maxn 2123456

using namespace std;

long long fac[maxn],ifac[maxn];

long long pmod(long long a,long long n)
{
	long long res=1;
	while(n)
	{
		if(n&1) res=(res*a)%mod;
		a=(a*a)%mod;
		n>>=1;
	}
	return res;
}

void init()
{
	fac[0]=1;
	for(int i=1;i<maxn;i++) fac[i]=(fac[i-1]*i)%mod;
	ifac[maxn-1]=pmod(fac[maxn-1],mod-2);
	for(int i=maxn-2;i>=0;i--) ifac[i]=(ifac[i+1]*(i+1))%mod;
}
long long ncr(int n,int r)
{
	return (((1ll*fac[n]*ifac[r])%mod)*ifac[n-r])%mod;
}
int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
		
	init();
	int t,n,k;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		printf("%d\n", ((2*ncr(n+k+1,k+2)-n)%mod+mod)%mod);;
	}
	return 0;
}
