#include<bits/stdc++.h>
using namespace std;

const int N=2000007;
const int mod=(int)1e9+7;
long long fact[N];

void preprocess() {
    fact[0]=1;
    for(long long i=1;i<N;i++) {
        fact[i]=(fact[i-1]*i)%mod;
    }
}

long long power(long long a, long long n) {
    long long ans=1;
    while(n) {
        if(n & 1) {
            ans=(ans*a)%mod;
        }
        a=(a*a)%mod;
        n>>=1;
    }
    return ans;
}

int main() {
    int test;
    preprocess();
    scanf("%d", &test);
    while(test--) {
        long long n, k;
        scanf("%lld %lld", &n, &k);
        long long ans=fact[n+k+1];
        ans=(2*ans)%mod;
        long long deno=(fact[k+2]*fact[n-1])%mod;
        ans=(ans*power(deno, mod-2))%mod;
        ans=(ans-n)%mod;
        if(ans<0) {
            ans+=mod;
        }
        printf("%lld\n", ans);
    }
}
