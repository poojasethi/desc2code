#include <iostream>
#define mod 1000000007
#define ll long long int
using namespace std;
ll fact[2000002];

void pre()
{
	fact[0]=1;
	for(int i=1;i<=2000001;i++)
	{
		fact[i] = (fact[i-1]*i)%mod;
	}
}
ll inverse(ll a,ll k)
{
	ll r,r1=k,r2=a,q,t1=0,t2=1,t;
	while(r2!=0)
	{
		q = r1/r2;
		r = r1%r2;
		t = t1 - q*t2;
		r1 = r2;
		r2 = r;
		t1 = t2;
		t2 = t;
	}
	return (t1+k)%k;
	
} 

int main()
{
	pre();
	int t;
	cin>>t;
	while(t--)
	{
		int n,k;
		cin>>n>>k;
		ll ans = fact[n+k+1];
		
		ans = (ans*inverse(fact[n-1],mod))%mod;
		
		ans = (ans*inverse(fact[k+2],mod))%mod;
		
		ans = ((2*ans - n)%mod+mod)%mod;
		cout<<ans<<endl;
	}
	return 0;
}
