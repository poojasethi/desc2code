#include<bits/stdc++.h>
using namespace std;
#define m 1000000007
int T,N,K,fact[2000009];
long long temp;

int pow(int a, int b)
{
    if(b==1) return a;
    if(b&1)return (a*(long long)pow(a,b-1))%m;
    temp=pow(a,b>>1);
    return (temp*temp)%m;
}
int invm(int a)
{
    return pow(a,m-2)%m;
}

int ncr(int n,int r)
{
    if(n<r)return 0;
//    cout<<fact[n]<<endl;cout<<invm(fact[n-r])<<endl;cout<<invm(fact[r])<<endl;
    return (((fact[n]*(long long)invm(fact[n-r]))%m)*(long long)invm(fact[r]))%m;
}

void init()
{
    fact[0]=1;
    for(int i=1;i<=2000005;i++)
    {
        temp=i*(long long)fact[i-1];
        if(temp>m)temp%=m;
        fact[i]=temp;
    }
}

int main()
{
    init();
    cin>>T;
    while(T--)
    {
        cin>>N>>K;
        cout<<(2*(long long)ncr(N+K+1,K+2)-N+m)%m<<endl;
    }
}
