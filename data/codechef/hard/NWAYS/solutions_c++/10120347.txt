LL fact[2000004],inv_fact[1000004];
void factorial()
{
	inv_fact[0]=fact[0]=inv_fact[1]=1;
	for(int i=1;i<=2000050;i++)
	{
		fact[i]=(fact[i-1]*i)%MOD;
	}
	for(int i=2;i<=1000000;i++)
		inv_fact[i]=(MOD - (MOD/i) * inv_fact[MOD%i] % MOD) % MOD;
}
LL nCr(LL n,LL r)
{
	return (((fact[n]*inv_fact[n-r])%MOD) * inv_fact[r])%MOD;
}
LL Inverse(LL a,LL b)
{
	LL x=1,y=a;
	while(b)
	{
		if(b&1)
		x*=y;
		if(x>MOD)x%=MOD;
		y*=y;
		if(y>MOD)y%=MOD;
		b>>=1;
	}
	return x;
}
