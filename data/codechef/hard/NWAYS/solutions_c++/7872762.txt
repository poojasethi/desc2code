#include<bits/stdc++.h>
using namespace std;
#include <iostream>
#include<cstdio>
#include<cstring>
#define ll long long

#define MOD 1000000007
ll b[2000008];
ll mdpw(ll a,ll b,ll p)
{
    ll r=1;
    while(b>0)
    {
        if(b%2!=0){
            r=(r*a)%p;}
        a=(a*a)%p;
        b=b/2;
    }
    return r;
}
ll modinv(ll a,ll c)
{
    return mdpw(a,c-2,c);

}

int main()
{
    b[0]=1;
    for(ll i=1;i<2000008;i++)
        b[i]=(b[i-1]*i)%MOD;
ll t;
scanf("%lld",&t);
while(t--)
{
    ll v,res,n,k,r;
    scanf("%lld",&n);scanf("%lld",&k);
    r=(b[n-1]*b[k+2])%MOD;
    //printf("%d\n",r);
    res=modinv(r,MOD)%MOD;
    v=(b[n+k+1]*res)%MOD;
    res=(2*v)%MOD;
    printf("%lld\n",(res-n+MOD)%MOD);
    
}
return 0;
}
