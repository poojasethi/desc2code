#define MOD 1000000007

const int MAX = 2000010;

ULL modinv(ULL a) {
    ULL res=1,x=MOD-2;
    while(x) {
        if( x%2!=0) {
            res=(res*a)%MOD;
        }
        a=(a*a)%MOD;
        x/=2;
    }
    return res;
}

ULL fact[MAX];

int main()
{
	ios::sync_with_stdio(false);
	fact[0]=1;
	FOR(i,1,MAX) fact[i]=(fact[i-1]*i)%MOD;
	ULL ans=0,t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		ans=(fact[n+k+1] * modinv((fact[n-1]*fact[k+2])%MOD))%MOD;
		ans=(ans*2)%MOD;
		ans=(ans+MOD-n)%MOD;
		cout<<ans<<endl;
	}
	return 0;
}
