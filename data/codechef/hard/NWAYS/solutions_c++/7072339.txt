#include<bits/stdc++.h>

using namespace std;

#define ll long long int
#define MOD 1000000007

ll fact[2100000];

ll pw(ll a, ll b)
{
	if(b==0)
		return 1;
	ll temp = pw(a,b/2);
	temp = (temp*temp)%MOD;
	if(b%2 == 0)
		return temp;
	else
		return (temp*a)%MOD;
}

ll inv_mod(ll a)
{
	return pw(a,MOD-2);
}

ll NCR(ll n, ll r)
{
	ll ans = fact[n];
	ans = (ans*inv_mod(fact[r]))%MOD;
	ans = (ans*inv_mod(fact[n-r]))%MOD;
	return ans;
}

int main()
{
	fact[0] = 1;
	for(int i = 1 ; i < 2100000; i++)
		fact[i] = (fact[i-1]*i)%MOD;
	int t;
	cin>>t;
	
	while(t--)
	{
	    ll ans = 0;
		ll N,K;
		cin>>N>>K;
		
		if(N==1)
		{
				cout<<1<<endl;
				continue;
		}
		else
		{
				
				ans = (2*NCR(N+K+1,K+2)%MOD-N+MOD)%MOD;
				
				cout<<ans<<endl;
		}
		
	}
	return 0;
}
