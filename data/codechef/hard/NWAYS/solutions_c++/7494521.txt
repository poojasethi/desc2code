#include <stdio.h>

typedef long long LL;
const int N=2000002;
const int mod=1000000007;

LL fac[N];
LL inv[N];
LL facinv[N];

int main(int argc, char* argv[])
{
    int i, t, n, k;
    fac[0]=1;
    for(i=1;i<N;i++)
    {
        fac[i]=fac[i-1]*i%mod;
    }
    inv[1]=1;
    for(i=2;i<N;i++)
    {
        inv[i]=inv[mod%i]*(mod - mod/i)%mod;
    }
    facinv[0]=facinv[1]=1;
    for(i=2;i<N;i++)
    {
        facinv[i]=facinv[i-1]*inv[i]%mod;
    }
    scanf("%d", &t);
    for(i=0;i<t;i++)
    {
        scanf("%d %d", &n, &k);
        LL ans=2*fac[k+n+1]*facinv[k+2]%mod*facinv[n-1]%mod;
        ans-=n;
        if(ans<0) ans+=mod;
        printf("%lld\n", ans);
    }
    return 0;
}
