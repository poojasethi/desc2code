#define ll long long
using namespace std;
int mod = 1000000007;
int mpow(int base,int exp)
{
	int res = 1;
	
	while(exp>0){
		if(exp&1) { res=((ll)res*base)%mod;}
		base = ((ll)base*base)%mod;
		exp>>=1;
	}
	return res;
}
const int MAXN = 2000003;
int fac[MAXN],IE[MAXN];
int ways[MAXN];

int _IE(int x)
{
	return mpow(x,mod-2);
}
void init()
{

	fac[0]=fac[1]=1;
	IE[0]=IE[1]=1;
	int i;
	
	
	for(i=2;i<=MAXN;i++) 
	{
		fac[i]=((ll)i*fac[i-1])%mod;
		//IE[i]= ((ll)_IE(i)*IE[i-1])%mod;	
		
	}
}
void pre(int N,int K){
	ll tem;int i;
	for(i=0;i<N;i++)
	{
		tem = fac[K+i];
		tem *= _IE(fac[i]);
		tem %= mod;
		tem *= _IE(fac[K]);
		tem %= mod;
		ways[i] = tem;
	}
}
