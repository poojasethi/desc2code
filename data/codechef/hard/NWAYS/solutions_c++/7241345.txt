#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define MOD 1000000007
ll f[2000010];
ll power(ll a,ll b)
{
    ll x=a;
    ll res=1LL;
    while(b>0)
    {
        if(b&1)
            res=(res*x)%MOD;
        x=(x*x)%MOD;
        b/=2;
    }
    return res;
}
ll invmod(ll a)
{
    return power(a,MOD-2);
}
int main()
{
    int t,n,k;
    f[0]=1LL;
    ll it;
    for(it=1LL;it<2000010LL;it++)
        f[it]=(f[it-1]*it)%MOD;

    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        ll ans=0;
        ans=(f[n+k+1]*invmod((f[n-1]*f[k+2])%MOD))%MOD;
        ans=(ans*2)%MOD;
        ans=(ans-n+MOD)%MOD;
        cout<<ans<<endl;
    }
    return 0;
}
