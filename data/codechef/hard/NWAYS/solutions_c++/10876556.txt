#include <iostream>
#include<cmath>
#include<vector>

using namespace std;

const int MOD =  1000000007;

int exp(int a, int n) {
	int res = 1;
	while ( n ) {
		if ( n&1 ) res=(1LL*res*a)%MOD;
		a = (1LL*a*a)%MOD;
		n >>= 1;
	}
	return res;
}
 
vector<int> fact(2000000+10);
 
void preCal(){
	fact[0] = fact[1] = 1;
	for(int i= 2;i<2000000+10;i++)
		fact[i] = (1ULL * fact[i-1] * i)%MOD;
}

int main() {
	preCal();
	int t; scanf("%d",&t);
	while(t--){
	int n,k; scanf("%d%d",&n,&k);
	long long ans = ((2LL*fact[k+n+1]*exp((1LL*fact[k+2]*fact[n-1])%MOD, MOD-2))%MOD -n + MOD)%MOD;
	printf("%lld\n",ans);
	}
	return 0;
}
