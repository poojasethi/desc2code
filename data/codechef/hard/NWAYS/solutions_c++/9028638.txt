#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define gc getchar
#define M 1000000007
#define ll long long

using namespace std;

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
ll fact[2000005];
void solve(){
ll i;
	fact[0]=fact[1]=1;
	for(i=2;i<=2000005;i++)
		fact[i]=(i*fact[i-1])%M;
}
ll power(ll a,ll b){
if(a==0) return 0;
else if(b==0) return 1;
else if(b&1){
    return (a*power((a*a)%M,b/2))%M;
}
else return power((a*a)%M,b/2);
}

int main(){
    solve();
int t,n,k;
scanint(t);
while(t--){
    scanint(n);scanint(k);
    ll temp,temp1,ans;
    temp=(fact[n+k+1]*power(fact[n-1],M-2))%M;
    temp1=(temp*power(fact[k+2],M-2))%M;
    ans=(2*temp1-n+M)%M;
    printf("%lld\n",ans);
}
return 0;

}
