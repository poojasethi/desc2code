#include<cstdio>
using namespace std;
#define LL long long int
LL fact[2000005]={0};
LL mod=1000000007;
LL power(LL a,LL b)
{   LL temp=a,ans=1;
    while(b>0)
    {   if(b%2==0)
        {	temp=(temp*temp)%mod;
            b/=2;
        }
        else
        {
            ans=(ans*temp)%mod;
            b--;
        }
    }
    return ans;
}
void compute()
{
    LL i;
    fact[0]=fact[1]=1;
    for (i=2;i<=2000005;i++)
        fact[i]=(i*fact[i-1])%mod;
}
int main()
{
    compute();
    LL t,n,k,n1,n2,n3,n4,ans;
    scanf("%lld",&t);
    while (t--)
    {
        scanf("%lld%lld",&n,&k);
        n1=fact[n+k+1];
        n2=fact[k+2];
        n3=fact[n-1];
        n4=((n2%mod)*(n3%mod))%mod;
        ans=power(n4,mod-2)%mod;
        ans=(ans*n1)%mod;
        ans=(2*ans%mod-n+mod)%mod;
        printf("%lld\n",ans);
    }
    return 0;
}
