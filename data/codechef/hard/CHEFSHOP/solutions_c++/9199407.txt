#include<iostream>
#include<cstdio>
#include<algorithm>
#include<climits>
using namespace std;
#define ll long long
ll dp[1<<16];
ll ar[20];
ll V[1<<16];
bool compare(const ll &p,const ll &q){return p>q;}
int main()
{
	ll i,j,t,k,l,m,n,o,p,q,v,x,b,ans;
	scanf("%lld",&t);
	while(t--)
	{
		scanf("%lld",&n);
		m=(1<<n)-1;
		for(i=0;i<=m;i++)
		{
			dp[i]=INT_MAX;
			V[i]=0;
		}
		for(i=1;i<=n;i++)
		scanf("%lld",&ar[i]);
		scanf("%lld",&q);
		while(q--)
		{
			v=0;
			scanf("%lld",&k);
			for(i=0;i<k;i++)
			{
				scanf("%lld",&x);
				x--;
				v|=(1<<x);
			}
			V[v]=k;
		}
		m=(1<<n)-1;
		sort(ar+1,ar+n+1,compare);
		for(i=1;i<=n;i++)
		ar[i]+=ar[i-1];
		dp[0]=0;
		ans=INT_MAX;
		for(i=0;i<=m;i++)
		{
		// cout<<"i="<<i<<" ";
			v=(i^m);
			b=__builtin_popcount(i);
			for(j=v;j;j=((j-1)&v))  // subset of a subset=3^n
			{
			 //  cout<<j<<" ";
				if(V[j])
				{
					dp[i|j]=min(dp[i|j],dp[i]+ar[b+V[j]-1]-ar[b]);
				}
			}
		//	cout<<"\n";
		//	cout<<i<<" "<<dp[i]<<" "<<ar[n]<<" "<<ar[b]<<"\n";
			ans=min(ans,dp[i]+ar[n]-ar[b]);
		}
		printf("%lld\n",ans);
	}
	return 0;
}
