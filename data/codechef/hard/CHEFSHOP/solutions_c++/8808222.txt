#include<bits/stdc++.h>
#define pb push_back

#define ff first
#define ss second
#define all(a) a.begin(), a.end()
#define mod 1000000007

#define si(a) scanf("%d",&a)
#define slld(a) scanf("%lld",&a)

using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< int, int > PII;
 
LL modpow(LL x, LL p, LL M)
{LL ret = 1; while(p){if(p&1) ret = ret*x%M; x = x*x%M; p/=2;} return ret;}
 
const int N = (int)(5e4) + 10;
int a[20],v[20],b[20];
PII p[N];
int dp[1<<17];
 
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
 
	int T; cin>>T;
	while(T--)
	{
		int n; cin>>n;
		for(int i=0; i<n; i++) cin>>a[i];
		sort(a,a+n);
		b[0] = 0;
		for(int i=1; i<=n; i++) b[i] = b[i-1]+a[i-1];
		reverse(a,a+n);
		int m; cin>>m;
		for(int i=0; i<m; i++)
		{
			int cnt,idx; cin>>cnt;
			p[i].ff = cnt;
			p[i].ss = 0;
			while(cnt--)
			{
				cin>>idx; idx--;
				p[i].ss |= 1<<idx;
			}
		}
		sort(p,p+m);
		for(int i=0; i<(1<<n); i++) dp[i] = 0;
		int msk,cnt,len,tot,nmsk,val,bitcnt;
		for(int i=0; i<m; i++)
		{
			msk = p[i].ss;
			cnt = n-p[i].ff;
			len = 0;
			for(int j=0; j<n; j++)
				if((msk&(1<<j))==0)
					v[len++] = j;
			tot = 1<<cnt;
			for(int j=0; j<=tot-1; j++)
			{
				nmsk = 0; bitcnt = n-cnt;
				for(int k=0; k<cnt; k++)
					if(j&(1<<k))
						nmsk |= (1<<v[k]), bitcnt++;
				val = dp[nmsk] + a[bitcnt-1];
				if(val > dp[nmsk|msk])
					dp[nmsk|msk] = val;
			}
		}
		int ans = (1e9);
		for(int i=0; i<(1<<n); i++)
		{
			val = b[n]-dp[i];
			if(val < ans) ans = val;
		}
		//for(int i=0; i<n; i++) ans += a[i];
		//ans -= dp[(1<<n)-1];
		cout<<ans<<endl;
	}
	return 0;
}
