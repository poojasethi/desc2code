#include <iostream>
using namespace std;

long long gcd(long long a, long long b)
{
    if(b==0)
        return a;
    gcd(b, a%b);
}

int main() {
	int t; scanf("%d",&t);
	long long a,b,c,d;
	
	while(t--){
	
	scanf("%lld%lld%lld%lld",&a,&b,&c,&d);
	long long p = gcd(c,d);
	long long ans = ((a>b)?(a-b):(b-a))%p;
	printf("%lld\n",min(ans,p-ans));
	
	}
	
	return 0;

}
