#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
 
int main(){
	long long t, a, b, c, d, g, ans;
    scanf("%lld", &t);
    while(t--){
        scanf("%lld %lld %lld %lld", &a, &b, &c, &d);
    	g = __gcd(c, d);
    	if(a >= b)   ans = (a - b) % g;
    	else    ans = (b - a) % g;
    	printf("%lld\n", min(g - ans, ans));
    }
    return 0;
}
