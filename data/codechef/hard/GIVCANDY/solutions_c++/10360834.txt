#include <bits/stdc++.h>

using namespace std;

long long int xGCD(long long int a, long long int b) {
    if(b==0)
        return a;
    else
        return xGCD(b,a%b);
}

int main()
{
    long long int t;
    //int a = 6, b = -3;
    scanf("%lld",&t);
    while(t--)
    {
        long long int a,b,c,d;
        scanf("%lld %lld %lld %lld",&a,&b,&c,&d);
        long long int g=xGCD(c,d);
        long long int diff;
        if(a>b)
            diff=a-b;
        else
            diff=b-a;
        long long int ans=diff%g;
        if(ans<(g-ans))
            printf("%lld\n",ans);
        else
            printf("%lld\n",g-ans);
    }
    return 0;
}
