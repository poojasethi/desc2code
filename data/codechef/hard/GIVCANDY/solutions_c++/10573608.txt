int max(int p,int q)
{
    return (p>q)?p:q;
}
ll min(ll p,ll q)
{
    return (p<q)?p:q;
}

ll findgcd(ll a,ll b)
{
    if(b==0)
    {
        return a;
    }
    else
    {
        return findgcd(b,a%b);
    }
}

ll mod(ll a)
{
    if(a>0)
    {
        return a;
    }
    else
    {
        return -a;
    }
}

int main()
{
    ll t,a,b,c,d,g,ans;
    sll(t);
    while(t--)
    {
        sll(a);
        sll(b);
        sll(c);
        sll(d);
        if(c>d)
        {
            g=findgcd(c,d);
        }
        else
        {
            g=findgcd(d,c);
        }
        //lldout(g);
        if(g==1)
        {
            cout << "0\n";
        }
        else
        {
            ans=min(mod(a-b)%g,g-(mod(a-b)%g));
            lldout(ans);
        }
    }
    return 0;
}
