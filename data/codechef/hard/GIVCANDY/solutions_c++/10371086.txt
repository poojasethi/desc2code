#include<bits/stdc++.h>
#define ll long long
 
using namespace std;
 
ll gcd(ll a,ll b){
    ll t;
    while(b!=0){
        t=b;
        b=a%b;
        a=t;
    }
    return a;
}
 
int main()
{
    ll t;
    cin>>t;
    ll a,b,c,d,t1,t2,t3,ans;
    while(t--){
        cin>>a>>b>>c>>d;
        t1=max(a,b)-min(a,b);
        t2=gcd(c,d);
        t3=t1%t2;
        ans=min(t3,t2-t3);
        cout<<ans<<endl;
    }
    return 0;
}
