#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
long long gcd(long long c,long long d)
{
    long long d1,d2,rem;
    d2=d;//divident
    d1=c;//divisor
    rem=d2%d1;
    while(rem!=0)
    {
        d2=d1;
        d1=rem;
        rem=d2%d1;
    }
    return d1;
}
long long  mod(long long num,long long g)
{
    if(num>=0)return num%g;
    else return (num%g)+g;
}
int main()
{
    int t,n,i,j;
    long long a,b,c,d,g;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c>>d;
        g=c<d?gcd(c,d):gcd(d,c);
        cout<<min(mod(a-b,g),mod(b-a,g))<<endl;
    }
    return 0;
}
