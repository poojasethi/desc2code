#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mod 1000000007
ll gcd(ll a,ll b){
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int main(){
	#ifndef ONLINE_JUDGE
		freopen("inp.txt","r",stdin);
	#endif
	int t;
	cin>>t;
	ll a,b,c,d,g,ans;
	while(t--){
		cin>>a>>b>>c>>d;
		g=gcd(c,d);
		if(a>b)
		ans=(a-b)%g;
		else
		ans=(b-a)%g;
		ans=min(ans,g-ans);
		cout<<ans<<endl;
	}
	return 0;
}
