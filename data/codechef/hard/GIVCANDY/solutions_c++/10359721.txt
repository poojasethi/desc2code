#include<iostream>
#include<cmath>
using namespace std;

int main()
{
    int t=1,y;
    cin>>y;
    while(t<=y)
    {
        long long int a,b,c,d,r,k1,k2,temp;
        cin>>a>>b>>c>>d;

        if(c>d)
        {
            k1=c;
            k2=d;
            while(k2!=0)
            {
                temp=k1%k2;
                k1=k2;
                k2=temp;
            }

        }
        else
        {
            k1=d;
            k2=c;
            while(k2!=0)
            {
                temp=k1%k2;;
                k1=k2;
                k2=temp;
            }

        }

        if(a>b)
        {
            r=a-b;
        }
        else
        {
            r=b-a;
        }

        long long int k=r%k1;
        long long int q=k1-k;
        long long int z=min(k,q);
        cout<<z<<endl;
        t++;
    }
}
