    #define ll long long
    ll gcd( ll , ll );
    ll min(ll a, ll b){return a>b?b:a;}
    using namespace std;
    int main()
    {
    ll x,p,a,b,c,d,z,s,k;
    cin>>x;
    for(p=1;p<=x;p++)
    {
    cin>>a>>b>>c>>d;
    if(a==b)
    {
    cout<<0<<endl;
    }
    else
    {
    s=a-b<0?(a-b)*-1:a-b;
    z=gcd(c,d);
    //cout<<z;
    k=s%z;
     
    cout<<min(z-k,k)<<endl;
     
    }
    }
    return 0;
    }
    ll gcd( ll p, ll q)
    {
    while (q != 0)
    { ll temp = q; q = p % q; p = temp; } return p;
