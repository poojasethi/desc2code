#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll gcd(ll a,ll b)
{
	if(a%b == 0) return b;
	else return gcd(b,a%b);
}
int main()
{
	ll t,a,b,c,d,ans;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c>>d;
		if(d>c) swap(c,d);
		ans = gcd(c,d);
		cout<<min(abs(a-b)%ans,ans-abs(a-b)%ans)<<"\n";
	}
	return 0;
}
