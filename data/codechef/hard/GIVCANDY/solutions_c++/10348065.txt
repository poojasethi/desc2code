#include <iostream>
 
using namespace std;
 
long long GCD(long long a,long long b){
    if(b == 0)
        return a;
    else
        return GCD(b,a%b);
}
 
int main(){
    int t;
    long long a, b, c, d, gcd, diff, result;
    cin >> t;
    for(int i = 0; i < t; i++){
        cin >> a >> b >> c >> d;
        diff = (a > b)?(a - b):(b - a);
        gcd = GCD(c, d);
        result = diff % gcd;
        cout << ((result < (gcd - result))?result:(gcd - result)) << "\n";
    }
    return 0;
}
