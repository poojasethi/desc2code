#include <iostream>
using namespace std;
#define ll long long 
#define min(x,y) (x<y) ? (x):(y)

ll GCD(ll a, ll b)
{
	if(b==0)
		return a;
	return GCD(b,a%b);
}

ll abss(ll n)
{
	if(n<0)
		return -n;
	return n;
}

int main(int argc, char const *argv[])
{
	ios_base::sync_with_stdio(false);
	int t;
	cin>>t;
	while(t--)
	{
		ll a,b,c,d;
		cin>>a>>b>>c>>d;
		ll gcd = GCD(c,d);
		ll diff = abss(a-b);
		ll ans = min((diff%gcd), ((-diff)%gcd + gcd));
		cout<<ans<<"\n";
	}
	return 0;
}
