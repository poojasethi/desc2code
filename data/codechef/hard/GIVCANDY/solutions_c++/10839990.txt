using namespace std;
typedef long long ll;
typedef pair<int,int> pii;

const int maxn=1e6+10;
ll gcd(ll a,ll b)
{
    return b>0?gcd(b,a%b):a;
}
ll lcm(ll a,ll b)
{
    return a/gcd(a,b)*b;
}
ll extgcd(ll a,ll b,ll &x,ll &y)
{
    ll d=a;
    if (!b)
    {
        d=extgcd(b,a%b,y,x);
        y-=(a/b)*x;
    }
    else
    {
        x=1;
        y=0;
    }
    return d;
}
