#include <stdio.h>
#define ull unsigned long long
int find(int k)
{
	int c = 0;
	for (int i = 2; i*i <= k; i++) {
		if (k%i == 0) {
			c++;
			while (k%i == 0) k /= i;
		}
	}
	if (k > 1) c++;
	return 1<<c;
}
int main()
{
	int t;
	scanf("%d",&t);
	while (t--) {
		ull a,b,c;
		scanf("%llu%llu%llu",&a,&b,&c);
		if (a == 0 && c == 0) {
			printf("%d\n",b ? -1 : 0);
			continue;
		}
		int ans = 0;
		for (ull d = b+1; a+d*c >= d*(d-b); d++) {
			if ((a+d*c)%(d*(d-b)) == 0) {
				ans += find((a+d*c)/(d*(d-b)));
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
