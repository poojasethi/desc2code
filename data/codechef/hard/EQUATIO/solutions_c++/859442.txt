#include <cstdio>

using namespace std;

int nprimefactors(int n)
{
        int i, k, count = 0;
        for (i = 2; i * i <= n; i++) {
                for (k = 0; n % i == 0; n /= i) k++;
                if (k > 0) count++;
        }
        if (n > 1) count++;
        return count;
}

int main(int argc, char *argv[])
{
	int nc;
	scanf("%d", &nc);
	for (int ci = 0; ci < nc; ci++) {
		int a, b, c;
		scanf("%d %d %d", &a, &b, &c);
		long long ans = 0;
		if (a == 0) {
			if (c == 0) {
				ans = b ? -1 : 0;
			} else {
				for (int i = 1; i * i <= c; i++) {
					if (c % i) continue;
					ans += 1LL << nprimefactors(i);
					if (i * i != c)
						ans += 1LL << nprimefactors(c / i);
				}
			}
		} else {
			for (int i = 1; i * i <= a; i++) {
				if (a % i) continue;
				int j = a / i;
				if (i > b && (j + c) % (i - b) == 0)
					ans += 1LL << nprimefactors((j + c) / (i - b));
				if (i != j && j > b && (i + c) % (j - b) == 0)
					ans += 1LL << nprimefactors((i + c) / (j - b));
			}
		}

		printf("%lld\n", ans);
	}

	return 0;
}
