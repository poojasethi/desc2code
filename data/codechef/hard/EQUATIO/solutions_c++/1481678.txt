#include <iostream>


#define long long long

using namespace std;

long a,b,c;

long div(long x){
    long a=0;
    for (long i=2; i*i<=x; ++i)
    if (x%i==0){
        while (x%i==0)
        x/=i;
        a++;
    }
    if (x>1)
    a++;
    return 1<<a;
}


void solve(void){
    long s=0;
    cin>>a>>c>>b;
    if (a==0 && b==0){
        if (c!=0)
        cout<<"-1\n";
        else
        cout<<"0\n";
        return;
    }

    for (long f=c+1; f<=a+b+c; ++f)
    if (a%f==0 && ((a/f)+b)%(f-c)==0)
    s+=div(((a/f)+b)/(f-c));
    cout<<s<<"\n";
}

int main()
{
    long t;
    cin>>t;
    while (t--)
    solve();
    return 0;
}
