#include<stdio.h>
#include<map>

using namespace std;

map <long long, long long> assoc;

long long prime_factors(long long n)    {
    if(assoc[n] != 0)   {
        return assoc[n];
    }
    long long i, j;   long long ans = 0;
    for (i = 2; i * i <= n; i++)    {
        for (j = 0; n % i == 0; n /= i) {
            j++;
        }
        if (j > 0)  {  
            ans++;
            if(assoc[i] == 0) assoc[i] = 1;
        } 
    }
    if (n > 1)  ans++;
    return ans;
}
            
main()  {
    int T;
    scanf("%d", &T);
    while(T--)  {
        long long a, b, c, i, j;
        long long ans = 0;
        scanf("%lld%lld%lld", &a, &b, &c);
        if(a == 0)  {
            if(c == 0)  {
                if(b != 0)  {
                    ans = -1;   goto end;
                }
                else    {
                    ans = 0;    goto end;
                }
            }
            else    {
                for(i = b + c; i >= b + 1; i--)   {
                    if(c % (i - b) == 0)    {
                        ans += (long long)1 << prime_factors(c / (i - b));
                    }
                }
                goto end;
            }
        }
        else    {
                    
            for(i = a; i >= b + 1; i--) {
                if(a % i == 0)  {
                    long long x = a / i + c;
                    long long y = i - b;
                    if(x % y == 0)  {
                        ans += (long long)1 << prime_factors(x / y);
                    }
                }
            }
            end : {}
            printf("%lld\n", ans);
        }
    }
    return 0;
}
