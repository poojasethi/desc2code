ill power(ill a,ill b)
{
	ill ans=1;
	while(b>0)
	{
		if(b%2)ans*=a;
		a*=a;
		b/=2;
	}
	return ans;
}
ill func(ill p)
{
	ill ans=0,i,f=0;
	//if(p==1) return ans;
	while(p>1)
	{
		f=0;
		for(i=2;i*i<=p;i++)
		{
			if(p%i==0) f++;
			while(p%i==0)
			{
				p/=i;
			}
		}
		if(f==0) f++,p=1;
		ans+=f;
	}
	return power(2,ans);
}
