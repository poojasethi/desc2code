#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int f(int n) {
    int ans = 1, i;
  for (i=2;i*i<=n;i++)
    if (n % i == 0) {
      ans *= 2;
      while (n % i == 0) n /= i;
    }
  if (n > 1) ans *= 2;
  return ans;
}

int main()
{
    int g, a, b, c, t;
    scanf("%d", &t);
    while(t--) {
        scanf("%d %d %d", &a, &b, &c);
        if (a==0 && c==0) {
            printf("%d\n", (b==0)?0:-1);
        } else {
            int ans = 0;
            for (g=b+1; (a/g+c)>=(g-b); g++)
                if (a%g == 0 && (a/g+c) % (g-b) == 0) ans += f((a/g+c)/(g-b));
            printf("%d\n", ans);
        }
    }
    return 0;
}
