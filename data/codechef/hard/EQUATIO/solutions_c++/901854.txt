#include<iostream>
using namespace std;

long long int f(long long int N)
{
	long long int size=0;
	for(long long int c=2;c*c<=N;c++)
	{
		if(N%c==0)
		{
			size++;
			N/=c;
			while(N%c==0)
			{
				N/=c;
			}
		}
	}
	if(N>1)
	{
		size++;
	}
	return 1LL<<size;
}

int main()
{
	long long int t;
	cin>>t;
	while(t--)
	{
		long long int ans=0;
		long long int a,b,c;
		cin>>a>>b>>c;
		if(a==0 && c==0 && b!=0)
			ans=-1;
		if(a+c)
		{
			for(long long int d= b+1;d-b<=a/d+c;d++)
			{
				if(a%d==0 && (a/d+c)%(d-b)==0)
				{
					ans+= f((a/d+c)/(d-b));
				}
			}
		}
		cout<<ans<<"\n";
	}
	return 0;
}
