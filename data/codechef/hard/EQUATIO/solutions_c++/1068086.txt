#include<stdio.h>
#include<map>

using namespace std;

map <long long, long long> assoc;

long long prime_factors(long long n)    {
    if(assoc[n] != 0)   {
        return assoc[n];
    }
    long long i, j;   long long ans = 0;
    for (i = 2; i * i <= n; i++)    {
        for (j = 0; n % i == 0; n /= i) {
            j++;
        }
        if (j > 0)  {  
            ans++;
            assoc[i] = 1;
        } 
    }
    if (n > 1)  ans++;
    return ans;
}
            
main()  {
    int T;
    scanf("%d", &T);
    while(T--)  {
        long long a, b, c, i, j;
        long long ans = 0;
        scanf("%lld%lld%lld", &a, &b, &c);
        if(a == 0)  {
            if(c == 0)  {
                if(b != 0)  {
                    ans = -1;   goto end;
                }
                else    {
                    ans = 0;    goto end;
                }
            }
            else    {
                for (i = 1; i * i <= c; i++) {
				if (c % i == 0) {
                     
    				ans += (long long)1 << prime_factors(i);
    				if (i * i != c)
    					ans += (long long)1 << prime_factors(c / i);
    				}
                }
                goto end;
            }
        }
        else    {
                    
            for(i = a; i * i >= a; i--) {
                if(a % i == 0)  {
                    long long x = a / i + c;
                    long long y = i - b;
                    if(i > b && x % y == 0)  {
                        ans += (long long)1 << prime_factors(x / y);
                    }
                    long long x1 = i + c;
                    long long y1 = a / i - b;
                    if((a / i) > b && i != (a / i) && x1 % y1 == 0) {
                        ans += (long long)1 << prime_factors(x1 / y1);
                    }
                }
            }
            end : {}
            printf("%lld\n", ans);
        }
    }
    return 0;
}
