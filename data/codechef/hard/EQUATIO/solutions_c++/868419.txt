#include <iostream>
using namespace std;

int calc(int t) {
    int ans = 1;
    for (int i = 2; i*i <= t; i++) {
        if (t%i == 0) {
            ans *= 2;
            while (t % i == 0) t /= i;
        }
    }
    if (t > 1) ans *= 2;
    return ans;
}

int main(int argc, char const *argv[]) {
    int T;
    cin >> T;
    for (int cas = 0; cas < T; cas++) {
        int a, b, c;
        cin >> a >> b >> c;

        if (a == 0 && c == 0) {
            cout << (b ? -1 : 0) << endl;
            continue;
        }
        int sol = 0;
        for (int d = b+1; (a/d+c) >= (d-b); d++)
            if (a%d == 0 && (a/d+c)%(d-b) == 0) sol += calc((a/d+c)/(d-b));
        cout << sol << endl;
    }
    return 0;
}
