#include<stdio.h>
#include<math.h>
 
int factorise(int n)
{
    int i,j;
    int ans = 1;
    for(i=2; i*i <= n; i++)
    {
             if(n%i == 0)
             {
                ans <<= 1;
                while(n%i == 0)
                          n /= i;
             }
             if(n == 1)
                  break;
    }
    if(n > 1)
         ans <<= 1;
    return ans;
}
 
int main()
{
    int a,b,c,t,ans,d;
    
    scanf("%d",&t);
    while(t--)
    {
              ans = 0;
              scanf("%d %d %d",&a,&b,&c);
              if(a == 0 && c == 0 && b == 0)
                   ans = 0;
              else if(a == 0 && c == 0 && b != 0)
                   ans = -1;
              else
                  for(d=b+1 ; a/d+c >= d-b ; d++)
                            if(a%d == 0  &&  (a/d+c)%(d-b) == 0)
                                   ans += factorise( (a/d+c)/(d-b) );
              printf("%d\n",ans);
    }
    return 0;
}
