int count(int n)
{
    int ret=1;
    for(int i=2;i*i<=n;i++)
    {
        if(n%i==0)
        {
            ret*=2;
            while(n%i==0) n/=i;
        }
    }
    if(n>1) ret*=2;
    return ret;
}

int main()
{
    int t,a,b,c;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c;
        int ans=0;
        if(a==0&&c==0)
        {
            if(b==0) cout<<0<<"\n";
            else cout<<-1<<"\n";
            continue;
        }
        for(int d=b+1;a/d+c>=(d-b);d++)
        {
            if(a%d||(a/d+c)%(d-b)) continue;
            ans+=count((a/d+c)/(d-b));
        }
        cout<<ans<<"\n";
    }
    return 0;
}
