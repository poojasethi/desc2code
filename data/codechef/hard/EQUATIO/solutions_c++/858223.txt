typedef long long int64;
using namespace std;

const int MaxT=2100000;

int nump[MaxT],F[MaxT];

void pre()
{
	F[1]=nump[1]=1;
	for(int i=2;i<MaxT;i++) if(nump[i]==0)
		for(int j=i;j<MaxT;j+=i) nump[j]=i;
}

int calc(int num)
{
	int res=1;
	
	for(;num>1;)
	{
		int tx=nump[num];

		while(num%tx==0) num/=tx;
		res*=2;
	}
	return res;
}

int main()
{
	int test;

	pre();
	for(cin>>test;test;test--)
	{
		int a,b,c;

		cin>>a>>b>>c;
		if((a+b+c)==0) cout<<0<<endl;
		else if((a+c)==0) cout<<-1<<endl;
		else if(a==0)
		{
			int64 res=0;

			for(int i=1;i<=c;i++) if(c%i==0) 
				res+=calc(i);
			cout<<res<<endl;
		}else
		{
			int64 res=0;

			for(int i=b+1;i<=a;i++) if(a%i==0)
				if((a/i+c)%(i-b)==0) res+=calc((a/i+c)/(i-b));
			cout<<res<<endl;
		}
	}
	return 0;
}
