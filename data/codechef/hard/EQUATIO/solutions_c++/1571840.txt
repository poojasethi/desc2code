#include <stdio.h>
#include <iostream>
using namespace std;
#include<math.h>

bool isprime(int n)
	{
		if(n==1) return 0;
		
		int y=sqrt(n);
		
		for(int i=2;i<=y;i++)
		 if(n%i==0) return 0;
		 
		return 1;	
	}


int div(int x)
	
	{
		int sum=0;
		
		for(int i=1;i<=x;i++)
			if(x%i==0 && isprime(i)) 
				sum+=1;
				
		return pow(2,sum);
	}

  
 
  
int main()

{ 
 int T;
 cin>>T;
 
 for(int i=1;i<=T;i++)
 {
  int a,b,c;

  {
  if(c==0) {
            	if(b==0) cout<<"0"<<endl;
            	else cout<<"-1"<<endl;
  
           }
  else {
          int sum=0;
  		for(int i=1;i<=c;i++)
  		if(c%i==0)
  		sum=sum+div(i);
  		
  		cout<<sum<<endl;
  	  }
  }
 
 else
  {
  int ans,d,a1;
  ans=0;
  
  d=b+1;
   
  while(d<=a)
  {
  	if(a%d==0)
  		{
  			a1=a/d;
  			if((a1+c)%(d-b)==0)
  			 ans+=div((a1+c)/(d-b));
 
  		}
  		
     d++;
  }
  
  cout<<ans<<endl;
  }
  }
 return 0;
  
}
