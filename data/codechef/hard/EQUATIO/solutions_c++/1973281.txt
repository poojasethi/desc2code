#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define N 110
int count(int n)
{
 int ans=1,i;
 for(i=2;i*i<=n;i++)
 if(n%i==0)
 {
  ans*=2;
  while(n%i==0)n/=i;
 }
 if(n>1)ans*=2;
 return ans;
}
int main()
{
	//freopen("in.txt","r",stdin);
	int T,i,j,k,a,b,c;
	scanf("%d",&T);
	while(T--)
	{
	 scanf("%d%d%d",&a,&b,&c);
	 if(a==0&&c==0)
	 {
	  printf("%d\n",b?-1:0);continue;
	 }
	 int ans=0;
	 for(int d=b+1;a/d+c>=d-b;d++)
	 if(a%d==0&&(a/d+c)%(d-b)==0)
	 ans+=count((a/d+c)/(d-b));
	 printf("%d\n",ans);
	}
	return 0;
}
