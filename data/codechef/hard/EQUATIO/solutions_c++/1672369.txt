#include <cstdio>

int ways(int n)
{
    int ret=1;
    for(int i=2; i*i<=n; i++) {
        if(n%i==0) ret*=2;
        while(n%i==0) n/=i;
    }
    if(n>1) ret*=2;
    return ret;
}

int main()
{
    int t;
    scanf("%d", &t);
    while(t--) {
        int a, b, c;
        scanf("%d %d %d", &a, &b, &c);
        if(a==0 && c==0)  { printf("%d\n", b ? -1 : 0); continue;}
        int ans = 0;
        for(int d=b+1;(d<=a || a==0) &&  a/d+c>=d-b; d++) if(a%d==0 && (a/d+c)%(d-b)==0) ans += ways((a/d+c)/(d-b));

        printf("%d\n", ans);
    }
}
