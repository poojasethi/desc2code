int main()

    scanf("%d", &te);
    while (te--)
    {
        scanf("%d%d%d", &a, &b, &c);
        if (!a && !c)
        {
            printf("%d\n", b ? -1 : 0);
            continue;
        }
        int ans = 0;
        for (int d = b + 1; a / d + c >= d - b; d++)
            if (a % d == 0 && (a / d + c) % (d - b) == 0)
            {
                int t = (a / d + c) / (d - b), q = 0;
                for (int j = 2; j * j <= t; j++)
                   if (t % j == 0)
                   {
                       while (t % j == 0) t /= j;
                       q++;
                   }
                if (t > 1)
                    q++;
                ans += 1 << q;
            }
        printf("%d\n", ans);
    }
}
