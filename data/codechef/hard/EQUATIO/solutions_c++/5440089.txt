#include <stdio.h>
#define ll unsigned long long
ll res;
ll mypow(ll a,ll b)
{
	ll ans=1;
	while(b>0)
	{
		if(b&1)
		ans=ans*a;
		b=b/2;
		if(b>0)
		a=a*a;
	}
	return ans;
}
void myfun(ll c ,ll i,ll b)
{
	ll g,lim,cnt,powers=0;
	g=b+i;
	ll l;
	l=(c/i)*g;
	if(l%g==0)
	{
	lim=l/g;
			//printf("%lld\n",lim);
	       for(i=2;i*i<=lim;i+=1)
		   {
		   	cnt=0;
		   	while(lim%i==0)
		   	{
		   		//printf("%lld\n",i);
		   		lim=lim/i;
		   		cnt+=1;
		   	}
		   	if(cnt>=1)
		   	powers+=1;
		   }
		   if(lim>1)
		   powers+=1;
		   if(powers>=1)
		   res+=mypow(2,powers-1)*2;
		   else
		   res+=1;
		
	}
}
void fun(ll a,ll b,ll c,ll *l,ll g)
{
	ll lim,powers=0,i;
	int cnt;
	*l=((a/g)+c)*g;
	if(*l%(g-b)==0)
	{
		
		*l=*l/(g-b);
		if(*l%g==0)
		{
			//printf("soln:%lld %lld\n",g,*l);
			//finding the no of primes upto (*l/g)
			lim=*l/g;
			//printf("%lld\n",lim);
	       for(i=2;i*i<=lim;i+=1)
		   {
		   	cnt=0;
		   	while(lim%i==0)
		   	{
		   		//printf("%lld\n",i);
		   		lim=lim/i;
		   		cnt+=1;
		   	}
		   	if(cnt>=1)
		   	powers+=1;
		   }
		   if(lim>1)
		   powers+=1;
		   if(powers>=1)
		   res+=mypow(2,powers-1)*2;
		   else
		   res+=1;
		   		
		}
	}
	
}
