#include <stdio.h>

int count(int n){
    int ans=1,i;
    for(i=2;i*i<=n;i++){
        if(n%i == 0){
            n /= i;
            while(n % i == 0)n /= i;
            ans *= 2;
        }
    }
    if(n > 1)ans *= 2;
    return ans;
}

int main(){
    int t,a,b,c,d,ans;
    scanf("%d",&t);
    while(t--){
        scanf("%d %d %d",&a, &b, &c);
        if(a == 0 && c == 0){
            printf("%d\n",b ? -1 : 0);
            continue;
        }
        ans = 0;
        for(d=b+1;a/d+c >= d-b;d++)
            if(a % d == 0 && (a/d+c) % (d-b) == 0)ans += count((a/d+c)/(d-b));
        printf("%d\n",ans);
    }

    return 0;
}
