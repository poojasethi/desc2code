#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 2000001

int cs,a,b,c,s,i,p,j;
bool B[lim];
int P[lim];

void check(int d)
{
  int t,i,p;

  t=(a/d+c);
  if(d-b>0 && !(t%(d-b)))
  {
    t/=d-b;
	for(i=0,p=0;P[i]*P[i]<=t;i++)
	  if(!(t%P[i]))
	  {
		while(!(t%P[i]))
	      t/=P[i];
		p++;
	  }
	if(t>1)
	  p++;
	s+=1<<p;
  }
}
int main()
{
  for(i=2;;i++)
    if(!B[i])
	{
	  P[p++]=i;
	  if(i*i>lim)
	    break;
	  for(j=i*i;j<lim;j+=i)
	    B[j]=1;
	}
  scanf("%d",&cs);
  while(cs--)
  {
    scanf("%d%d%d",&a,&b,&c);
	s=0;
	if(a)
	{
	  for(i=1;i*i<=a;i++)
	    if(!(a%i))
		{
		  check(i);
		  if(i*i<a)
		    check(a/i);
		}
	  printf("%d\n",s);
	}
	else
	  if(c)
	  {
	    for(i=1;i*i<=c;i++)
		  if(!(c%i))
		  {
			check(i+b);
			if(i*i<c)
			  check(c/i+b);
		  }
		printf("%d\n",s);
	  }
	  else
	    if(b)
		  printf("-1\n");
		else
		  printf("0\n");
  }  
  return 0;
}
