
#include<iostream>
#include<algorithm>
 
using namespace std;
 
#define FOR(i, a, b) for(int i=a; i<b; ++i)
 
struct stad { int start, end; };
bool cmp(stad a, stad b) { return (a.end<b.end); }
int main()
{
    int n, dur; cin>>n; stad a[n];
    FOR(i, 0, n) { cin>>a[i].start>>dur; a[i].end=a[i].start+dur; }
    sort(a, a+n, cmp);
    
    int count=1, ans=1, curr=a[0].end;
    
    while(count<n)
    {
        if(a[count].start>curr) { curr=a[count].end; ++ans;}
        ++count;
    }
    cout<<ans<<endl;
    return 0;
}
