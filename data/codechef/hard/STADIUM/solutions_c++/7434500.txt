#include <bits/stdc++.h>
using namespace std;

vector<pair<int,int> > V;

int main()
{
  int n,a,b,last,ans = 0;
  pair<int,int> p;
  cin>>n;
  for(int i=0; i<n; i++)
        {
            cin>>a>>b;
           V.push_back(make_pair(a+b,a));
        }
  sort(V.begin(), V.end());

  last = 0;

  for(int i = 0; i < n; i++)
  {
    if(V[i].second > last)
    {
      ans++;
      last = V[i].first;
    }
  }

  cout<<ans<<endl;
  return 0;
}
