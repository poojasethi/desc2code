#include <bits/stdc++.h>
using namespace std;
struct node 
{
int start;
int finish;
};
int compare(node a,node b)
{
if(a.finish==b.finish)
{
return b.start<a.start;
}
else
{
return a.finish<b.finish;
}
}
int main()
{
int n,y;
scanf("%d",&n);
node arr[n];
for(int i=0;i<n;i++)
{
scanf("%d%d",&arr[i].start,&y);
arr[i].finish=arr[i].start+y-1;
}
sort(arr,arr+n,compare);
int c=1;
int pivot=arr[0].finish;
for(int i=1;i<n;i++)
{
if(pivot<arr[i].start-1)
{
c++;
pivot=arr[i].finish;
}
}


printf("%d",c);
return 0;
}
