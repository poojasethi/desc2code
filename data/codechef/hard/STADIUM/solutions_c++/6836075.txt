#include<cstdio>
#include<algorithm>
using namespace std;

struct val
{
	int s,e;
};

val *arr = new val[100001];


bool mysort(val a,val b)
{
	if(a.e<b.e)
		return true;
	else if(a.e==b.e && a.s<b.s )
		return true ;
	else
		return false;
}


int main()
{
	int t,i,n,ans,x; 
	
	
		ans=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)		{
			scanf("%d%d",&arr[i].s,&x);
			arr[i].e=arr[i].s+x;
								}
		
		sort(arr,arr+n,mysort);
		
	//	for(i=0;i<n;i++)
		//	printf("%d  %d\n",arr[i].s,arr[i].e);
		
		
		
		x=arr[0].e;		ans=1;
		
		for(i=1;i<n;i++)
		{
			if(arr[i].s>x)
			{	ans++;
			
			x=arr[i].e;	}	
		}
		
		printf("%d\n",ans);
	
	
	return 0;
}
