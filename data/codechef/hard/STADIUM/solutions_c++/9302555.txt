#include<bits/stdc++.h>
using namespace std;
struct time
{
    long int x,y;
};

bool myfunction(struct time s1,struct time s2)

{
    return (s1.y<s2.y);
}
int call(struct time a[],int n)
{
    int i;
    int count=1;
    int j=0;
    for(i=1;i<n;i++)
    {

        if(a[j].y<a[i].x)
        {
            count++;
            j=i;
        }
    }
    return count;
}
int main()
{

 int n;
    int i;
    cin>>n;
    struct time a[n];
    for(i=0;i<n;i++)
    {
        cin>>a[i].x>>a[i].y;

    }

    for(i=0;i<n;i++)
    {

        a[i].y=a[i].x+a[i].y;
    }
    sort(a,a+n,myfunction);


    int res=call(a,n);
    cout<<res<<endl;
    return 0;

}
