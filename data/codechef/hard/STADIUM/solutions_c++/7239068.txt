#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#define cin(n) scanf("%ld",&n)
using namespace std;


long mycompare(pair<long,long> v1,pair<long,long> v2)
{
    return v1.second<v2.second;
}
int main()
{
    vector< pair<long,long> > v;
    long n,start,duration,result,i,j;
    cin(n);

    {
        for(i=0;i<n;i++)
        {
            cin(start);
            cin(duration);
            v.push_back(make_pair(start,(start+duration-1)));
        }
        sort(v.begin(),v.end(),mycompare);
        //cout<<"#"<<v[0].first<<endl;
        result=1;
        j=0;
        for(i=1;i<n;i++)
        {
            if((v[i].first-v[j].second)>=2){result++;j=i;}
        }
        printf("%ld\n",result);
    }
    return 0;
}
