#include<stdio.h>
#include<iostream>
#include<stdlib.h>
using namespace std;

struct slot
{
	int st;
	int end;
};

int compare(const void *a,const void *b)
{
	int x,y;
	x = (*((struct slot *)a)).end - (*((struct slot *)b)).end;
	y= (*((struct slot *)a)).st - (*((struct slot *)b)).st;
	if(x!=0)
	return x;
	return y;
}

void sort(struct slot arr[],int N)
{
	qsort(arr,N,sizeof(struct slot),compare);
}

struct slot arr[100005];

int main()
{
	int N;
	cin>>N;
	for(int i=0;i<N;i++)
	{
		int x,y;
		cin>>x>>y;
		arr[i].st=x;
		arr[i].end=x+y;
	}
	
	sort(arr,N);
	/*for(int i=0;i<N;i++)
	{
		cout<<arr[i].st<<" "<<arr[i].end<<endl;
	}*/
	int C=1;
	int k=0;
	for(int i=1;i<N;i++)
	{
		if(arr[i].st>arr[k].end)
		{
			C++;
			k=i;
		}
	}
	cout<<C<<endl;
}
