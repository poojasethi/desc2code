#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
struct node
{
       int st;
       int fi;
};

bool comp(struct node x, struct node y)
{
     return (x.fi < y.fi);
}
int main()
{
    int n;
    cin >> n;
    
    struct node v[n];
    for(int i = 0; i < n; i++) {
            cin >> v[i].st >> v[i].fi;
            v[i].fi = v[i].fi + v[i].st;
    }
    
    sort(v,v+n,comp);
    
    /*for(int i = 0; i < n; i++) {
            cout << v[i].fi << endl;
    }*/
    
    int count = 0;
    int index = 0;
    while(index < n) {
                count++;
                int l = index;
                index++;
                while(index < n && v[index].st <= v[l].fi){
                                  index++;
                }
    }
    
    cout << count << endl;
    return 0;
}
