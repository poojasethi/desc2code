#include<iostream>
#include<algorithm>
using namespace std;
#include<stdio.h>
#define pii pair<int,int>
bool comp(pii a,pii b)
{
    return a.second<b.second;
}
int main()
{
    int n;
    scanf("%d",&n);
    pii date[n];
    int i,l;
    for(i=0;i<n;i++)
    {
        scanf("%d %d",&date[i].first,&l);
        date[i].second=date[i].first+l;
    }
    sort(date,date+n,comp);
    int count=1,last=date[0].second;
    for(i=1;i<n;i++)
    {
        if(date[i].first>last)
        {
        count++;
        last=date[i].second;
        }
    }
    printf("%d",count);
    return 0;
}
