#include <iostream>
using namespace std;
#include <cstdio>
#include <algorithm>

struct stadium
{
int begin;
int dur;
int end;
};
bool comp(stadium a,stadium b)
{
if(a.end<b.end)
return 1;
return 0;

}
int main()
{
int n,x;
scanf("%d",&n);
 stadium event[n];
for(int i=0;i<n;i++)
{
scanf("%d",&event[i].begin);
scanf("%d",&event[i].dur);
event[i].end=event[i].begin+event[i].dur;
}
sort(event,event+n,comp);
x=event[0].end;
int c=1;
for(int i=1;i<n;i++)
{
if(x<event[i].begin)
{
x=event[i].end;
c++;
}}
printf("%d",c);
return 0;
}
