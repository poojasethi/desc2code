using namespace std;

struct event_t
{
    int start;
    int end;
};

bool operator<(const event_t &a, const event_t &b)
{
    return a.end > b.end;
}

void the_sports_stadium_()
{
    int N,k,n(0),s(-1);
    event_t p;
    priority_queue<event_t> Q;
    
    cin >> N;
    for(k = 0; k < N; ++k)
    {
        scanf("%d %d", &p.start, &p.end);
        p.end += p.start;
        Q.push(p);
    }

    while(!Q.empty())
    {
        p = Q.top();
        Q.pop();
        if(p.start > s)
        {
            s = p.end;
            ++n;
        }
    }
    printf("%d\n", n);
}

int main()
{
    the_sports_stadium_();
    return 0;
}
