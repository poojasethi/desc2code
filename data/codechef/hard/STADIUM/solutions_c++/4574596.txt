#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <stdio.h>

using namespace std;

#define Si(x) scanf("%d",&x)
#define Sii(x,y) scanf("%d %d",&x,&y)
#define Pi(x) printf("%d\n",x)
#define gc getchar_unlocked

typedef pair<int,int> pii;
typedef set<pair<int,int> > spii;
inline int readInt() 
{
	char c = gc();
	while(c<'0' || c>'9') c = gc();
	int ret = 0;
	while(c>='0' && c<='9') 
	{
		ret = (ret<<3) + (ret<<1) + c - 48;
		c = gc();
	}
	return ret;
}

int main() {
	
	spii a;
	int t,start,end;
	t=readInt();
	while(t--)
	{
		start=readInt();
		end=readInt();
		end=start+end;
		a.insert(make_pair(end,start));
	}
	
	spii::iterator i,j,n;
	n=a.end();
	i=j=a.begin();
	int check=(*i).first;
	int count=1;
	for(++i;i!=n;++i)
	{
		
		if((*i).second>check)
		{
			check=(*i).first;
			count++;
			
		}
	}
	Pi(count);
	
	return 0;
}
