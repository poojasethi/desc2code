#include<iostream>
#include<vector>
#include<utility>
#include<algorithm>
using namespace std;

vector <pair <int, int> > arr;

main() {
    int n;
    cin>>n;
    for (int i=1; i<=n; i++) {
        int t1, t2;
        cin>>t1>>t2;
        arr.push_back(make_pair(t1+t2-1, t1));
    }
    sort(arr.begin(), arr.end());

    int ans=1, tmp=arr[0].first;

    for (int i=1; i<n; i++) {
        if (arr[i].second>tmp+1) {
            ans++;
            tmp=arr[i].first;
        }
    }
    cout<<ans<<endl;
}
