#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;

struct event{
    int s,e;
} ev;
int n,i,s,d;
vector<event> v;
int cmp(struct event e1, struct event e2){return e1.e < e2.e ? 1:0;}


int main(){
    cin>>n;
    for(i=0;i<n;i++){
        cin>>s>>d;
        ev.s = s;
        ev.e = s+d;
        v.push_back(ev);
    }
    sort(v.begin(),v.end(),cmp);
    int ans = 1,t=v[0].e;
    for(i=1;i<n;i++){
        if(v[i].s > t){
            ans+=1;
            t=v[i].e;
        }
    }
    cout<<ans<<endl;
    return 0;
}
