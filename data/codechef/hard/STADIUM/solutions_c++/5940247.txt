#include<stdio.h>
#include<stdlib.h>
#include<vector>
#include<algorithm>

using namespace std;

struct Event 
{
 unsigned long int si;
 unsigned long int ei;
 Event(int s,int e):si(s),ei(e){}
 bool operator<(const Event& a)const
 {
	     return ei < a.ei;
 }
};



int main()
{
  int n;
  scanf("%d",&n);
  vector< Event>vi;
  for(int i =0;i<n;i++)
  {
    unsigned long long int s;
    int d;
  scanf("%lld%d",&s,&d);
   vi.push_back( Event(s,s+d));  
  }
  sort(vi.begin(),vi.end());
  int count =1;
  Event last = vi[0];
  for(int i = 1;i<n;i++) 
  {
    if(vi[i].si > last.ei)
    {
      count++;
      last = vi[i];
    }	    
  }
  printf("%d",count);
return 0;
}
