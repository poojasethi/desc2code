#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{

    unsigned long long int n;
    scanf("%llu",&n);
    unsigned long long int i;
    unsigned long long int A[n],B[n],C[n];
    for(i=0;i<n;i++)
    {
        unsigned long long int a,b;
        scanf("%llu %llu",&a,&b);
        C[i]=a*10000000+a+b-1;
    }
    unsigned long long int count=1;
    sort(C,C+n);
    for(i=0;i<n;i++)
    {
        A[i]=C[i]/10000000;
        B[i]=C[i]%10000000;
    }
for(i=0;i<n-1;i++)
{

if((B[i]+1)<A[i+1])
    count++;
else if(B[i]<B[i+1])
    B[i+1]=B[i];


}
printf("%llu \n",count);



}
