#include <algorithm>
#include <stdio.h>
using namespace std;
struct inter{
	int start;
	int end;
	inter(int st,int en){
		start=st;end=en;
	}
	inter(){}
};

struct sortstruct{
    inline bool operator() (const inter& inter1, const inter& inter2){
        return (inter1.end < inter2.end);
    }
} sortq;

int main(){
int n;
scanf("%d",&n);
inter a[n+1];
int start1,len;
for(int i=1;i<=n;i++){
	scanf("%d",&start1);
	scanf("%d",&len);
	a[i]=inter(start1,start1+len-1);
}
sort(a+1,a+n+1,sortq);


int i=1,num=0;
int lastend=-1;;
while(true){
	if(a[i].start>lastend+1){
		num++;
		lastend=a[i].end;
	}
	i++;
	if(i>n) break;
}
printf("%d\n",num);
}
