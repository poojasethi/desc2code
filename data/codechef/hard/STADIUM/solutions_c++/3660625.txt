#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>

std::vector < std::pair <long long, long long> >events;
int main()
{
  long long N, S, D;
  scanf("%lld", &N);

  for (long long i = 0; i < N; i++)
    {
      scanf ("%lld %lld ", &S, &D);
      events.push_back(std::make_pair(S+D, S));
    }
  sort(events.begin(), events.end());

  long long num_events  = 0, prev_end = 0;

  for (long long i = 0 ; i < N; i++)
    {
      if (events[i].second > prev_end)
        {
          prev_end = events[i].first;
          num_events++;
        }
    }
  printf ("%lld", num_events);

  return 0;
}
