#include<iostream>
#include<algorithm>
#include<utility>
using namespace std;
int main()
{       int n;
        cin>>n;
        int blocked;
        pair<int,int> p[n];
        int i,z,count=1;
        for(i=0;i<n;i++)
        {       cin>>p[i].second;
                cin>>z;
                p[i].first=p[i].second+z;
        }
        sort(p,p+n);
        blocked=p[0].first;
        for(i=1;i<n;i++)
        {       if(p[i].second>blocked)
                {       count++;
                        blocked=p[i].first;
                }
        }
        cout<<count;
        return 0;
}
