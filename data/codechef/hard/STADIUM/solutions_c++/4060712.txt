#include <iostream>
#include <cstdio>
#include <algorithm>
#define MAX 100000
#define FOR(i,m,n) for(i=m;i<n;i++)

using namespace std;

struct event
{
    long long int start,end;
    int days;
    void get_event()
    {
        cin>>start>>days;
        end=start+days;
    }
}events[MAX+10];

bool comp(event a,event b)
{return a.end<b.end;}

int main()
{
long long int counter,cur_end,i,n;

cin>>n;

FOR(i,0,n)
    events[i].get_event();

sort(events,events+n,comp);

counter=1;cur_end=events[0].end;

FOR(i,1,n)
    if(events[i].start>cur_end)
        {cur_end=events[i].end;counter++;}

cout<<counter;

return 0;
}
