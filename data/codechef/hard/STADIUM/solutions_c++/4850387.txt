#include<stdio.h>
#include<map>

using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    multimap<int,int> multimap1;
    for(int i=0;i<n;i++)
    {
        int s,d;
        scanf("%d%d",&s,&d);
        d=s+d;
        multimap1.insert(make_pair(d,s));
    }
    int res=0,prev_f;
    multimap<int,int>::iterator it=multimap1.begin();
    if(it!=multimap1.end())
    {
        prev_f=it->first;
        it++;
        res++;
    }
    for(;it!=multimap1.end();it++)
    {
        int st=it->second;
        if(st>prev_f)
        {
            res++;
            prev_f=it->first;
        }
        //printf("\n%d %d ",it->first,it->second);

    }
    printf("%d\n",res);

    return 0;
}
