#include<iostream>
#include<algorithm>
using namespace std;
int main() {
	pair <int, int> events[100001];

	int N,val;
	cin>>N;
	for(int i=0; i<N; i++) {
		cin>>events[i].second;
		cin>>val;
		events[i].first=val+events[i].second;
	}
	sort(events, events+N);
	int count=1;
	int last_end=events[0].first;
	for(int i=1; i<N; i++) {
		if(events[i].second>last_end) {
			count++;
			last_end=events[i].first;
		}
	}
cout<<count<<endl;
	return 0;
}
