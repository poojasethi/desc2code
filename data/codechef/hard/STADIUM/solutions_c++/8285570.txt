#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int main(){
    int n=scan();
    pair<int,int> a[100001];
    for(int i=0;i<n;i++){
        a[i].second=scan();
        a[i].first=scan();
        a[i].first+=a[i].second;
    }
    sort(a,a+n);
    int ans=0;
    int temp=0;
    for(int i=0;i<n;i++){
        if(a[i].second>=temp){
            ans++;
            temp=a[i].first+1;
        }
    }
    printf("%d",ans);
}
