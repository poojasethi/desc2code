#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <stdio.h>

using namespace std;

#define Si(x) scanf("%d",&x)
#define Sii(x,y) scanf("%d %d",&x,&y)
#define Pi(x) printf("%d\n",x)

typedef pair<int,int> pii;
typedef set<pair<int,int> > spii;


int main() {
	
	spii a;
	int t,start,end;
	Si(t);
	while(t--)
	{
		Sii(start,end);
		end=start+end;
		a.insert(make_pair(end,start));
	}
	
	spii::iterator i,j,n;
	n=a.end();
	i=j=a.begin();
	int check=(*i).first;
	int count=1;
	for(++i;i!=n;++i)
	{
		
		if((*i).second>check)
		{
			check=(*i).first;
			count++;
			
		}
	}
	Pi(count);
	
	return 0;
}
