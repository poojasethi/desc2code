#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
bool sort_pred(pair<int,int> a,pair<int,int> b)
{
	return (a.second < b.second);
}
int main()
{
	int T;
	cin>>T;
	int count=0;
	vector< pair<int,int> > v;
	for(int i=0;i<T;i++)
	{
		int s,l;
		cin>>s>>l;
		v.push_back(make_pair(s,s+l));
		
	}
	
	sort(v.begin(),v.end(),sort_pred);
	
	pair<int,int> exe=make_pair(0,0);
	for(int i=0;i<T;i++)
	{
		if(v[i].first > exe.second) 
		{
			exe=v[i];
			count++;
		}
	}
	cout<<count<<endl;
	
}
