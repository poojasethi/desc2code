int scan4n( )
{
    int n = 0;
    char 	c;

    for( c = getc( stdin ); c==' ' || c=='\n' || c == '\t'; c = getc( stdin ));

    for( ; c > 0x2f  &&  c < 0x3a; c = getc( stdin ))
    	n = (n * 10) + (c & 0x0f);

    return( n );
}
