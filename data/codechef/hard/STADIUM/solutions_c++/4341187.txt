#include <iostream>
#include<math.h>
#include<algorithm>


using namespace std;

int main()
{

pair<long long,long long> a[100001];
long long s,d,p,l,n,i;

cin>> n;
for(i=1;i<=n;i++)
{
    cin>>s>>d;
    a[i].first=s+d;
    a[i].second=s;
}
sort(a,a+(n+1));
l=1;
p=a[1].first;

for(i=2;i<=n;i++)
{
    if(p<a[i].second)
    {
        l++;
        p=a[i].first;

    }
}
cout<<l;
}
