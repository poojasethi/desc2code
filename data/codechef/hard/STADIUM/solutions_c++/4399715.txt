    #include<iostream>
    #include<stdio.h>
    #include<vector>
    #include<algorithm>
    using namespace std;
    typedef struct node
    {
        long long int Start;
        long long int Duration;
        long long int End;
    }node;
    vector<node> V;
    bool myfunc(node A, node B)
    {
        return A.End < B.End;
    }
    int main()
    {
        long long int N;
        cin >> N;
        node temp;
        temp.Start = 0;
        temp.Duration = 0;
        temp.End = -1;
        V.assign(N,temp);
        long long int i;
        for(i=0;i<N;i++)
        {
            long long int n1;
            long long int n2;
            cin >>n1>>n2;
            V[i].Start = n1;
            V[i].Duration = n2;
            V[i].End = n1 + n2;
        }
        sort(V.begin(),V.end(),myfunc);
        long long int last_day = -1;
        long long int co = 0;
        for(i=0;i<N;i++)
        {
            if(last_day < V[i].Start)
            {
                last_day = V[i].End;
                co++;
            }
        }
        cout<<co;
    }
