#include <iostream>
#include <algorithm>
using namespace std;

int main() {
int n,x,y;
pair <int,int> a[100009];
cin>>n;
for(int i=0;i<n;i++)
{
cin>>x>>y;
a[i]=make_pair(x+y,x);
}
if(n<=0)
{cout<<0<<"\n";return 0;}
sort(a,a+n);
int max=a[0].first;
int count=1;
for(int i=1;i<n;i++)
{
x=a[i].second;
y=a[i].first;
if(x>max)
{count++;max=y;}
}
cout<<count<<"\n";
return 0;
}
