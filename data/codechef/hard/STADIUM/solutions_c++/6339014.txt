    #include <bits/stdc++.h>		//ry
    
    using namespace std;
     
    int scan(int &x)
    {
    register int c = getchar_unlocked();
    x=0;
    for(;c<48 || c>57;c=getchar_unlocked());
    for(;c>=48&&c<=57;c=getchar_unlocked()) x = (x<<3)+(x<<1)+(c-48);
    return x;
    }
     
     
    struct interval{
    int L,R;
    };
     
    struct cmp
    {
    bool operator()(const interval& a,const interval& b)
    {
    return ((a.R < b.R) || (a.R == b.R && a.L<b.L));
    }
    };
     
    struct interval a[100001];
     
    int main()
    {
    int n;
    scan(n);
    
    for(int i=0;i<n;i++)
    {
    int l,r;
     scan(l);
     scan(r);
       a[i].L = l;
        a[i].R = l + r;
        }
     
    sort(a,a+n,cmp());
     
   
    int last = -1;
    int cnt=0;
    
    for(int i=0;i<n;i++)
    {
    if(a[i].L > last)
     {cnt++;last=a[i].R;}
    }
     
    printf("%d\n",cnt);
    return 0;
    }
