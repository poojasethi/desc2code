#include <stack>
#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;
pair <int,int> A[100001];
main()
{
    int N;
    scanf("%d", &N);
    for (int i = 1; i <= N; i++)
    {
        scanf("%d%d", &A[i].first, &A[i].second);
        A[i].second = A[i].first + A[i].second - 1;
    }
    sort(A + 1 , A + 1 + N);
    stack <int> s;
    for (int i = 1; i <= N; i++)
    {
        if (s.empty())
        {
            s.push(i);
        }
        else
        {
            if (A[i].first - 1 > A[s.top()].second)
            {
                s.push(i);
            }
            else if (A[i].second <= A[s.top()].second)
            {
                s.pop();
                s.push(i);
            }
        }
    }
    printf("%d\n",s.size());
    return 0;
}
