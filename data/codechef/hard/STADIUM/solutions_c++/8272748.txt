#define lli long long int
#define pb push_back

#define pii pair<int,int>

using namespace std;

bool comp(pii a,pii b){
    if(a.second==b.second)return a.first < b.first;
    else return a.second < b.second;
}
int main()
{
    int n,i,j,a,b;
    scanf("%d",&n);
    vector<pii> vec;
    for(i=0; i<n; i++)
    {
        scanf("%d%d",&a,&b);
        vec.pb(make_pair(a,a+b));
    }
    sort(vec.begin(),vec.end(),comp);
    int ans=0;
    j=-1;
    for(i=0; i<n; i++)
    {
        if(vec[i].first > j)
        {
            ans++;
            j=vec[i].second;
        }
    }

    printf("%d",ans);


    return 0;
}
