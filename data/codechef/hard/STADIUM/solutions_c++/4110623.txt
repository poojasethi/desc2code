#include<iostream>
using namespace std;
int event[100000][2];
 void merge(int u,int m,int v)
 {
	 int k=0,i=u,j=m+1;
	 int c[100000][2];
	 while(i<=m && j<=v)
	{
		if(event[i][1]<=event[j][1])
			{
				c[k][0] = event[i][0];
				c[k++][1] = event[i++][1];
			      
		}
		else
			{
				c[k][0] = event[j][0];
				c[k++][1] = event[j++][1];
		    }
	}
	while(i<=m)
	{
				c[k][0] = event[i][0];
				c[k++][1] = event[i++][1];
		
 }
	while(j<=v)
	{
		c[k][0] = event[j][0];
				c[k++][1] = event[j++][1];
	}
	for(i=u;i<=v;i++)
	{
		event[i][0]=c[i-u][0];
		event[i][1]=c[i-u][1];
 }
 }
void sort(int l,int h)
{
	if(l<h)
	{
		int m=(l+h)/2;
		sort(l,m);
		sort(m+1,h);
		merge(l,m,h);
	}
}
