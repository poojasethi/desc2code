#include<iostream>
#include<string>
#include<vector>

using namespace std;

vector<int> kmp(string a, string b)
{
    int ps[b.length()];
    ps[0]=0;
    int i=1,j=0;
    while(i<b.size())
    {
        if(b[i]==b[j])
        {
            ps[i]=j+1;
            i++;
            j++;
        }
        else if(j==0)
        {
            ps[i]=0;
            i++;
        }
        else
            j=ps[j-1];
    }

    vector<int> v;
    i=j=0;
    while(i<a.size())
    {
        if(j==b.size())
        {
            v.push_back(i-j);
            j=ps[j-1];
        }
        else if(a[i]==b[j])
        {
            i++;
            j++;
        }
        else if(j==0)
            i++;
        else
            j=ps[j-1];
    }
    if(j==b.size())
    {
        v.push_back(i-j);
    }
    return v;
}

int main()
{
    int Tn,i,j,k;
    string S,T;
    cin>>Tn;
    while(Tn--)
    {
        cin>>T>>S;

        string p="";
        vector<string> v;
        for(i=0;i<T.size();i++)
        {
            if(T[i]=='*')
            {
                if(p!="")
                {
                    v.push_back(p);
                    p = "";
                }
            }
            else
                p+=T[i];
        }
        if(p!="")
            v.push_back(p);

        if(v.size()==0)
        {
            for(i=0;i<S.size();i++)
                cout<<i+1<<" ";
        }
        else
        {
            vector<int> w;
            for(i=0;i<S.size();i++)
                w.push_back(i-1);

            for(j=0;j<v.size();j++)
            {
                vector<int> u=kmp(S,v[j]);
                k=0;
                for(i=0;i<S.size() && w[i]!=-2;i++)
                {
                    if (k == u.size())
                        w[i] = -2;
                    else if (w[i] >= u[k]) {
                        k++;
                        i--;
                    }
                    else
                        w[i] = u[k] + v[j].size() - 1;
                }
            }

            for(i=0;i<w.size();i++)
                cout<<w[i]+1<<" ";
        }
        cout<<endl;
    }
    return 0;
}
