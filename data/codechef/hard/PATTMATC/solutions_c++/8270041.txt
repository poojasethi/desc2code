void kmp(int k){
    static int ne[N];
    int i,j,n=a.size();
    //DBG(a)
    ne[0]=-1;j=-1;
    for(i=1;i<n;i++){
        while(j>=0&&a[i]!=a[j+1])j=ne[j];
        if(a[i]==a[j+1])j++;
        ne[i]=j;
    }
    j=-1;now++;
    for(i=0;i<m;i++){
        while(j>=0&&s[i]!=a[j+1])j=ne[j];
        if(s[i]==a[j+1])j++;
        if(j==n-1)vis[i]=now;
    }
    dp[m][k]=-1;
    for(i=m-1;i>=0;i--){
        dp[i][k]=dp[i+1][k];
        if(i+n<=m&&vis[i+n-1]==now)dp[i][k]=i;
    }
    //for(i=0;i<m;i++)printf("%d ",dp[i][k]);puts("");
}
