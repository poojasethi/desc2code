#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

int main()
{
	ll x;
	int n;

	scanf("%d %lld", &n, &x);

	vector<ll> v(n);
	for (int i=0; i<n; i++) 
		scanf("%lld", &v[i]);

	sort(v.begin(), v.end());


	int partially_solved = 0, j;
	for (j=0; j<n; j++) {
		if (x - (ll)ceil((double)v[j]/2.0) >= 0) {
			x -= (ll)ceil((double)v[j]/2.0);
			v[j] = v[j] - (ll)ceil((double)v[j]/2.0);
			partially_solved++;
		}
	}

	int full_solved = 0, i;
	for (i=0; i<n; i++) {
		if (x-v[i] >= 0) {
			x -= v[i];
			full_solved++;
		}
	}

	printf("%d %d\n", n-partially_solved, full_solved);

	return 0;
}
