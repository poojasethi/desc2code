#include <iostream>
#include <cstring>
#include <algorithm>
 #include<math.h>
using namespace std;
 
long long a[1000005];
 
int main()
{
    long long n,x,sum=0,pass,fail,p,i,j;
    pass = fail = 0;
    ios_base::sync_with_stdio(0);
    cin >> n >> x;
    for ( i = 0; i < n; i++ ) {
        cin >> a[i];
        sum += a[i];
    }
    if ( x >= sum ) {
         cout << 0 << " " << n << endl;
         return 0;
    }
    sort(a,a+n);
    sum = 0;
    for ( i = 0; i < n; i++ ) {
        if ( sum + ceil(( float)a[i]/2)<= x ) {
             sum +=ceil(( float)a[i]/2);
        }
        else break;
    }
    fail = n-i;
    for ( j = 0; j <= i-1; j++ )  {
        if ( sum + a[j]-ceil(( float)a[j]/2) <= x ) {
             sum += a[j]-ceil(( float)a[j]/2);
             pass++;
        }
    }   
    cout << fail << " " << pass << endl;
    return 0;
}
