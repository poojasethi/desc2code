#include<bits/stdc++.h>
using namespace std;
long long arr[1000010];
int main()
{
	int n,sc=0,fl=0;
	long long x;
	int i,j,k;
	scanf("%d%lld",&n,&x);
	for(i=0;i<n;++i)
		scanf("%lld",&arr[i]);
	sort(arr,arr+n);
	for(i=0;i<n;++i)
	{
		if(x==0&&arr[i]!=0)
			break;
		if(x>=((arr[i]+1)/2))
			{
				x-=(arr[i]+1)/2;
				if(arr[i]==0||arr[i]==1)++sc;
			}
		else break;
	}
	fl=n-i;
	if(x>0)
	for(i=sc;i<n;++i)
	{
		if(x>=(arr[i]/2))
			x-=arr[i]/2,++sc;
		else break;
	}
	cout<<fl<<" "<<sc<<"\n";
	return 0;
}
