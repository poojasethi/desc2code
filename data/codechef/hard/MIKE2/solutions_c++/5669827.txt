#define FOR(i,m,n) for(int i=m; i<n; i++)
 
#define size_ 65536
 
using namespace std;
 
 
int main()
{
    int n;
    long long x;
    cin >> n >> x;
 
    vector<long long> packages;
    vector<long long> given;
    packages.resize(n);
    given.resize(n);
 
    FOR(i,0,n)
    {
        cin >> packages[i];
    }
 
    vector<long long>::iterator iter = packages.begin();
    sort(iter,iter+n);
 
 
    long long succ_packs=0, fail_packs=0;
 
    int i;
    for(i=0; i<n; i++)
    {
        if((long long)ceil(packages[i]/2.0)<=x)
        {
            x = x - (long long)ceil(packages[i]/2.0);
            given[i] = (long long)ceil(packages[i]/2.0);
        }
        else
            break;
    }
 
    fail_packs = n - i;
 
    FOR(j,0,i)
    {
        if(given[j] ==  packages[j])
            succ_packs += 1;
        else if(given[j] < packages[j] && x>= packages[j] - given[j])
        {
            x -= packages[j] - given[j];
            succ_packs += 1;
        }
        else
            break;
    }
 
    cout << fail_packs << " " << succ_packs << endl;
 
    return 0;
}
