#include<bits/stdc++.h>
using namespace std;
main()
{
	int n,j,i;
	long long int  x;
	cin >> n>>x;
	vector<int> a(1000000);
	for(i=0;i<n;i++)
	{
	cin >> a[i];
	}
	sort(a.begin(),a.begin()+n);
	double half;
	for(i=0;i<n;i++)
	{
	half=ceil((double)a[i]/2);
	if(x>=half)
	{
		x=x-half;
	}
	else
		break;
	}
	for(j=0;j<n;j++)
	{
		if((a[j]/2)<=x)
		{
			x=x-(a[j]/2);
		}
		else
			break;
	}

	cout<< n-i <<" " <<j<<endl;
}
