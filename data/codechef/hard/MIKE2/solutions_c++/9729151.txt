lli arr[1000009],dupl[1000009];

{

    for(i=0;i<n;i++)scanf("%lld",&arr[i]);

    for(i=0;i<n;i++)dupl[i]=arr[i];
    int ans = 0,ans2=0;
    for(i=0 ;i < n;i++){
        lli y = (arr[i]+1)/2;
        if( x - y >= 0){
            x=x-y;
            arr[i]-=y;
        }
        else break;
    }
    if(x > 0){
        for(i=0;i<n;i++){
            if(x - arr[i] >=0){
                x-=arr[i];
                arr[i]=0;
            }
            else if(arr[i]-x >=0){
                x=0;
                arr[i]-=x;
                break;
            }
            else break;
        }
    }
    for(i=0;i<n;i++){
        if(arr[i]==0)ans2++;
        else if(arr[i]*2 > dupl[i])ans++;
    }
    printf("%d %d",ans,ans2);


    return 0;
}
