#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll a[1000001];   
int main(){
    int n;
    ll tot;
    cin>>n>>tot;
    for(int i=0; i<n; i++) cin>>a[i];
    sort(a,a+n);
    int half[n+1],full[n+1],nun=0,ff=0;
    memset(half,0,sizeof half);
    memset(full,0,sizeof full);
    for(int i=0; i<n; i++){
        ll val=0;
        if(a[i]%2){
            val=a[i]/2;
            val++;
        }
        else val=a[i]/2;
       // cout<<val<<endl<<" "<<tot<<endl;
        if(tot>=val){
            tot-=val;
            half[i]=1;
        }
        else break;
    }
    for(int i=0; i<n; i++){
        int val=0;
        if(a[i]%2){
            val=a[i]/2;
            val++;
        }
        else val=a[i]/2;
        
        if(half[i]){
            val = a[i]-val;
        }
     //   cout<<val<<endl;
        if(tot>=val){
            tot-=val;
            half[i]=0;
            full[i]=1;
        }
        else break;
    }
    
    for(int i=0; i<n; i++){
    //    cout<<half[i]<<" "<<full[i]<<endl;
        if(full[i]) ff++;
        else if(!half[i]) nun++;
    }
    cout<<nun<<" "<<ff<<endl;
}
