#include<iostream>
#include <string>
#include <vector>
#include<sstream>
#include<queue>
#include<map>
#include<cmath>
#include<algorithm>
 
using namespace std;
 
 
int main()
{
 
    int n, s = 0, f, solve;
    long long x;
 
    cin >> n >> x;
 
    f = n;
 
    vector<int> a(n, 0);
 
    for(int i = 0; i < n; i++) cin >> a[i];

    sort(a.begin(), a.end());
 
    for(int i = 0; i < n; i++) {
        solve = (a[i] + 1)/2;
        
        if(solve <= x) {
            a[i] -= solve;
            x -= solve;
            f--;
        }
    }
 
    for(int i = 0; i < n; i++) {
        solve = a[i];
        if(solve <= x) {
            x -= solve;
            s++;
        }
    }
  
    cout <<f<<" "<<s<< endl;
    return 0;
}
