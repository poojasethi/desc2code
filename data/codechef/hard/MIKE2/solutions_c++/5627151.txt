inline void scan(int &a){
    char ch=a=0;
    while(ch<33)    ch=getc(stdin);
    while(ch>33)    a=a*10+ch-'0',ch=getc(stdin);
}
int min(int a,int b){
    return a>b?b:a;
}
int list[1000010];
int main(){
    int n;
    long long sum=0,rem,x,temp;
    scan(n),scanf("%lld",&x);
    for(int i=0;i<n;i++){
        scan(list[i]),sum+=list[i];
    }
    if(sum<=x){
        printf("0 %d",n);
    }else if(!x){
        printf("%d 0",n);
    }else{
        sort(list,list+n);
        int i=0,j=0;
        for(;i<n;i++){
            if(list[i]&1){
                if((list[i]+1)/2<=x)    x-=(list[i]+1)/2,list[i]/=2;
                else    break;
            }else{
                if(list[i]/2<=x)    x-=list[i]/2,list[i]/=2;
                else    break;
            }
        }
        for(;j<n;j++)
            if(list[j]<=x){
                x=x-list[j];
                continue;
            }else   break;
        printf("%d %d",n-i,j);
    }
    return 0;
}
