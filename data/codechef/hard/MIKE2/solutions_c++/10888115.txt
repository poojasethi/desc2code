#include <iostream>
#include<cstring>
#include<algorithm>
using namespace std;

#define max 1000009

int a[max];

int main() {
	int n;
	long long x;
	scanf("%d %lld",&n,&x);
	
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	
	sort(a,a+n);
	
	int s=0,f=n;
	long long c=0;
	
	for(int i=0;i<n;i++){
	int task = (a[i]+1)/2;
	if(c+task>x)break;
	c += task;
	f--;
	}

	for(int i=0;i<n;i++){
	int task = a[i]/2;
	if(c+task>x)break;
	c += task;
	s++;
	}
	
	printf("%d %d",f,s);
	return 0;
	
}
