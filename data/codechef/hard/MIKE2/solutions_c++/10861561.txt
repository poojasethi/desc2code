#include <iostream>
#include <cstdio>
#include <algorithm>
 
using namespace std;
 
int n, a[ 1000100 ], b[ 1000100 ];
long long x;
 
int main()
{
	scanf("%d%lld", &n, &x);
	for ( int i = 0; i < n; i++ ) scanf("%d", &a[i]);
	sort( a, a + n );
	for ( int i = 0; i < n; i++ ) b[i] = a[i] / 2 + a[i] % 2, a[i] -= b[i];
	int fi = n, se = 0;
	for ( int i = 0; i < n; i++ ) 
		if ( b[i] <= x )
		{
			fi -= 1;
			x -= b[i];
		}
	for ( int i = 0; i < n; i++ )
		if ( a[i] <= x )
		{
			se += 1;
			x -= a[i];
		}
	printf("%d %d\n", fi, se);
	return 0;
}
