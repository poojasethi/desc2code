#include <iostream>
using namespace std;
#define max 1<<17


int a[max];

int main() {
	int t; scanf("%d",&t);
	while(t--){
	int n; scanf("%d",&n);
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	
	long long total = 0;
	
	for(int i=0;i<31;i++){
	
		int cnt[2] = {0,0};
		int sum = 0;
		long long count = 0;
		
		for(int j=0;j<n;j++){
		cnt[sum]++;
		sum = (sum+((a[j]&(1<<i)) != 0))%2;
		count += cnt[sum^1];
		}
		total += count*(1<<i);
	}
	
	printf("%lld\n",total);
	}

	return 0;
}
