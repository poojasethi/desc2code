#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t,has[32],i,j,n,a,sum;
    long long int res;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        res=0;sum=0;
        for(i=0;i<32;i++)
            has[i]=0;
        for(i=1;i<=n;i++)
        {
            scanf("%d",&a);
            sum=sum^a;
            for(j=0;j<32;j++)
            {
                if(sum&(1<<j))
                {
                    res=res+(1LL*(1<<j))*(1LL*(i-has[j]));
                    has[j]++;
                }
                else
                    res=res+(1LL*(1<<j))*(1LL*has[j]);
            }
        }
        printf("%lld\n",res);
    }
    return 0;
}
