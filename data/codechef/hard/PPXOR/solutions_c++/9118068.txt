#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main() {
	ll t,i,j,k,n,p;
	cin>>t;
	int a[100006];
	while (t--) {
		cin>>n;
	    a[0]=0;
		for (i=1; i<=n; i++) {
			cin>>a[i];
			a[i]^=a[i-1];
		}
		long long ans=0;
		for (i=0, p=1; i<30; i++, p<<=1) {
			int c=0;
			for (j=1;j<=n;j++) {
				if (a[j]&p) c++;
			}
			ans+=c*(n-c+1)*p;
		}
		cout<<ans<<endl;
	}
	return 0;
}
