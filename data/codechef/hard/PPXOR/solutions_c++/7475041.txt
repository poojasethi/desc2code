#include <bits/stdc++.h>
#define maxn 100000+3
using namespace std;

typedef long long ll;
int A[maxn], n;


int main() {
	int t;
	scanf("%d",&t);
	while(t--) {
		scanf("%d",&n);
		
		A[0] = 0;
		for(int i=1; i<=n; i++) {
			scanf("%d",&A[i]);
			A[i] ^= A[i-1];
		}
		
		ll p = 1;
		ll  res = 0;
		ll cnt = 0;
		
		for(int i=0; i<=31 ;i ++) {
			cnt = 0;
			for(int j=1; j<=n; j++) {
				if(A[j] & p) {
					cnt++;
				}
			}
			
			res += cnt*(n-cnt+1) * p;
			p<<=1;
		}
		
		printf("%lld\n", res);
	}
}
