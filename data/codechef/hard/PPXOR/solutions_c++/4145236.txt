ll bit[32]={0};
int arr[100005];

int main()
{
	int t,n,a;
	t=1;
	si(t);
	while(t--)
	{
		si(n);ll ans=0;
		for(int i=0;i<n;i++)
		{
			si(a);
			if(i)
			arr[i]=arr[i-1]^a;
			else
			arr[i]=a;
			ans+=arr[i];
			
		}
		for(int i=0;i<n;i++)
		{

			for(int j=0;j<32;j++)
			{
					int d=arr[i]&(1<<j);
					if(i==0)
					{
						if(d)bit[j]++;
						continue;
					}
					if(d==0)
					{
						ans+=(1<<j)*bit[j];

					}
					else
					{
						ans+=(1<<j)*(i-bit[j]);
						bit[j]++;
					}
			}

		}
		pl(ans);nl;
		memset(bit,0,sizeof(bit));
	}
}
