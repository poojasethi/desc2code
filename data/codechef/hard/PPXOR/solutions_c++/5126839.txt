#include <iostream>
using namespace std;
 
int main() {
	int t;
	cin>>t;
	while(t--){
		int n;
		cin>>n;
		int inp[n+1];
		inp[0]=0;
		for(int i=1;i<=n;i++){
			cin>>inp[i];
			inp[i]=inp[i]^inp[i-1];
		}
		long long int ans=0;
		for(int i=0,p=1;i<32;i++,p<<=1){
			int num=0;
			for(int j=0;j<=n;j++){
				if(inp[j]&p)num++;
			}
			ans+=(long long int)(num)*(n+1-num)*p;
		}
		cout<<ans<<endl;
	}
	return 0;
}
