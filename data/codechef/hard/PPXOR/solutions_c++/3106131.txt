#include <iostream>
using namespace std;

int main() {
    long long sum;
    int t, n, ai, ci[32], xor_sum;
    cin >> t;
    while(t--) {
        cin >> n;
        sum = 0;
        xor_sum = 0;
        for(int i=0;i<32;++i) ci[i]=0;
        for(int i=0;i<n;++i) {
            cin >> ai;
            xor_sum = xor_sum^ai;
            int t1=xor_sum;
            long long p = 1;
            for(int in=0;in<32;++in) {
                if(t1&1) {
                    sum = sum+(i+1-ci[in])*p;
                    ++ci[in];
                } else sum = sum+ci[in]*p;
                t1>>=1;
                p<<=1;
            }
        }
        cout << sum << endl;
    }
    return 0;
}
