#include<bits/stdc++.h>
 
using namespace std;
 
#define 	rep(i, n) 		for(i = 0; i < (n); i++)
#define 	FOR(i, a, n) 	for(i = a; i < n; i++)
 
typedef long long ll;
 
#define N	112345
 
ll a[N],pre[N],col[35];
 
ll pwr[35];
 
int main(){
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
	#endif
	ll t,i,j,n;
	cin>>t;
	pwr[0] = 1;
	FOR(i,1,32)		pwr[i] = pwr[i-1] * 2;
	while(t--){
 
		cin>>n;
		FOR(i,1,n+1)	cin>>a[i];
 
		memset(col,0,sizeof col);
 
		pre[0] = 0;
		FOR(i,1,n+1)	pre[i] = pre[i-1] ^ a[i];
 
		ll ans = 0;
 
		FOR(i,1,n+1){
 
			ll cur = 0;
 
			rep( j,32 ){
				int cnt;
				if( pre[i] & (1 << j) ){
					cnt = i - col[j];
				}
				else{
					cnt = col[j];
				}
				cur += pwr[j] * cnt;
			}
 
			ans += cur;
 
			rep( j,32 ){
				if( pre[i] & (1 << j) ){
					col[j]++;
				}
			}
 
		}
		cout<<ans<<endl;
	}
	return 0;
}
