#include <iostream>
#include<cstring>
using namespace std;

int main()
{
    int testCases,n;
    int count[32][2];

    cin>>testCases;

    while(testCases--)  {
        cin>>n;
        memset(count,0,sizeof(count));

        for(int j=0;j<32;++j)   {
            count[j][0] = 1;
        }

        int sum = 0,k;
        long long ans = 0;
        for(int i=0;i<n;++i)    {
            cin>>k;

            sum ^= k;
            for(int p=0;p<32;++p)   {

                long long val = (1<<p);
                if(sum & (1<<p))
                    ans += val*count[p][0];

                else ans += val*count[p][1];
            }

            for(int p=0;p<32;++p)   {
                if(sum & (1<<p))
                    ++count[p][1];

                else ++count[p][0];
            }
        }

        cout<<ans<<endl;
    }

    return 0;
}
