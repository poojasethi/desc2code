#include<iostream>
#include<cmath>
using namespace std;
int main()
{
	long t,n;
	long a[100005];
	cin>>t;
	while(t--)
	{
		cin>>n;
		a[0]=0;
		for(int i=1;i<=n;i++)
		{
		cin>>a[i];
		a[i]=a[i]^a[i-1];
	}
		long long ans=0;
	for(int i=0;i<30;i++)
	{
		int c=0;
		for(int j=0;j<=n;j++)
		{
			if(a[j]&1<<i)
			c++;
		}
		ans+=(long long)c*(n-c+1)*(pow(2,i));
	}
	cout<<ans<<endl;
		
	}
}
