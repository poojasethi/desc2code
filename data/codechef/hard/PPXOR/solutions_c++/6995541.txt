// LUCIFER <3 SLS <3

#include <bits/stdc++.h>

using namespace std;

#define ll long long int
#define s(x) scanf("%lld", &x)

int main()
{
    ll t, n, i, tmp, j, ans, x;
    s(t);
    while (t--) {
        s(n);
        ll y[50];
        memset(y, 0, sizeof(y));
        ans = 0;
        for (i = 0; i < n; i++) {
            s(tmp);
            ans ^= tmp;
            tmp = ans;
            j = 0;
            while (tmp != 0) {
                if (tmp & 1)
                    y[j]++;
                j++;
                tmp >>= 1;
            }
        }
        ans = 0;
        for (i = 0; i < 50; i++) {
            x = n - y[i] + 1;
            ans += (1 << i)*x*y[i];
        }
        printf("%lld\n", ans);
    }
    return 0;
}
