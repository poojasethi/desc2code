#include<iostream>
#include<stdio.h>
#include<math.h>
#include<string>
using namespace std;

int scanint(){
	register int c = getchar();
	int x=0;
	bool neg=false;
	for(;(c<48 || c>57) && c!=45 ;c=getchar());
	if(c==45){
		neg=true;
		c=getchar();
	}
	for(;c>47 && c<58;c=getchar())
	{
		x=(x<<1)+(x<<3)+c-48;
	}
	if(neg==true)
	x=x*(-1);
	return x;
}

const int N = 100000;
typedef long long ll;

void update(ll c[] , int value){
	for(int i=0 ; i<32 ; i++){
		if((value>>i) & 1)
			c[i]++;
	}
}


int main(){
	int t,n,ar[N+1],x[N+1];
	ll sum,c[32]={0};
	t = scanint();
	while(t--){
		
		n = scanint();
		x[0] = 0 ;
		sum = 0 ;
		
		for(int i=1 ; i<=n ; i++){
			ar[i] = scanint();
			x[i] = x[i-1] ^ ar[i] ;
		}
		
		for(int i=0 ; i<32 ; i++)
			c[i] = 0 ;
			
		for(int i=1 ; i<=n ; i++){
			for(int j=0 ; j<32 ; j++){
				if((x[i]>>j) & 1){
					sum += ((i-c[j]) * (ll)pow(2,j));
				}else{
					sum += (c[j] * (ll)pow(2,j));
				}
			}
			update(c,x[i]);
		}
		
		cout<<sum<<endl;
		
	}
}
