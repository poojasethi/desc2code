#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
int t;
int n;
int sum[31]={0};
int main(){
    t=scan();
    while(t--){
    	n=scan();
    	int cur = 0;
    	for(int i=1;i<=n;i++){
    		cur^=scan();
    		for(int j=0;j<31;++j){
    			if(cur&(1<<j)){
    				sum[j]++;
    			}
    		}
    	}
    	long long ans = 0;
    	for(int i=0;i<31;++i){
    		long long temp = sum[i];
    		temp*=1+n-sum[i];
    		temp*=1<<i;
    		ans+=temp;
    		sum[i]=0;
    	}
    	printf("%lld\n",ans);
    }
}
