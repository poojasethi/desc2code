#include <bits/stdc++.h>
#define maxn 100001

using namespace std;

int n, a[maxn], s[maxn];
int cnt[32];

void read_in() {
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",a+i);
}

void update(int x) {
    int i=0;
    while (x) {
        if (x%2) cnt[i]++;
        i++;
        x/=2;
    }
}

long long tong(int R,int x) {
    long long kq=0;
    for(int i=0;i<32;i++)
    if ((x>>i) & 1) kq+=(long long)(R-cnt[i])*(1LL<<i);
    else kq+=(long long)cnt[i]*(1LL<<i);
    return kq;
}

void solve() {
    s[0]=0;
    for(int i=1;i<=n;i++) s[i]=(s[i-1]^a[i]);
    memset(cnt,0,sizeof(cnt));
    long long ans=0;
    for(int i=1;i<=n;i++) {
        ans+=tong(i,s[i]);
        update(s[i]);
    }
    printf("%lld\n",ans);
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    int T; scanf("%d",&T);
    while (T--) {
        read_in();
        solve();
    }
}
