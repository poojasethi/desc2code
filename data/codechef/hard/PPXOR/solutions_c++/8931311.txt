#include<bits/stdc++.h>
using namespace  std;
#define ll long long int
ll A[100010],S[100010],C[100010][33];
int main()
{
    int n,t;
    ll i,j,val;
    scanf("%d",&t);
    while(t>0){
            t--;
    scanf("%d",&n);
    S[0]=0;
    for(i=0;i<32;i++)
        C[0][i]=0;
    for(i=1;i<=n;i++)
    {
        scanf("%lld",&A[i]);
            S[i]=A[i]^S[i-1];
        for(j=0;j<32;j++)
        {
            if(1<<j&S[i])
                C[i][j]=C[i-1][j]+1;
            else
                C[i][j]=C[i-1][j];
        }
    }
     val=0;
    for(i=n;i>0;i--)
    {
        val=val+S[i];
        for(j=0;j<32;j++)
        {
            if(1<<j&S[i])
            {
                val=val+(i-C[i][j])*(1<<j);
            }
            else
            {
                val=val+C[i][j]*(1<<j);
            }
        }
  //  cout<<val<<endl;
    }
    printf("%lld\n",val);
    }
}
