#include <stdio.h>

//typedef __int64 LL;
typedef long long LL;

LL dp0[30];
LL dp1[30];
int d[30];

int main(int argc, char* argv[])
{
    int h, i, j, t, n, a;
    scanf("%d", &t);
    for(h=0;h<t;h++)
    {
        for(i=0;i<30;i++) dp0[i]=1;
        for(i=0;i<30;i++) dp1[i]=0;
        LL sum=0;
        scanf("%d", &n);
        int p=0;
        for(i=0;i<n;i++)
        {
            scanf("%d", &a);
            p^=a;
            a=p;
            for(j=0;j<30;j++)
            {
                d[j]=a&1;
                if(!d[j])
                {
                    sum+=dp1[j]*(1<<j);
                    dp0[j]+=1;
                }
                else
                {
                    sum+=dp0[j]*(1<<j);
                    dp1[j]+=1;
                }
                a>>=1;
            }
        }
        printf("%lld\n", sum);
    }
    return 0;
}
