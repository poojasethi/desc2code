#include<iostream>
using namespace std;
int main()
{
    int test;
    long  boy_l[503],chef_l[503],temp,j;
    long n,p,k,i;
    cin>>test;
    while(test--)
    {
        cin>>n>>k>>p;
        for(i=0;i<k;i++)
        {
            cin>>chef_l[i]>>temp;
        }
        for(i=0;i<p;i++)
        {
            cin>>boy_l[i]>>temp;
        }

        for(i=0;i<p;i++)
        {
            for(j=0;j<k;j++)
            {
                if(boy_l[i]==chef_l[j])
                {
                    break;
                }
            }
            if(j==k)
                    break;  //in case we do not find a case of boys tangerine in last so it terminates
        }
        if(i<p)
        {
             cout<<"No"<<endl;
        }
        else
            cout<<"Yes"<<endl;
    }
    return 0;
}
