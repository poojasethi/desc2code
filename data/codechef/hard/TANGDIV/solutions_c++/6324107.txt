#include<iostream>
int arr[501];
int find(int num,int n)
{
	for(int i=0;i<n;i++)
	{
		if(arr[i]==num) return 1;
	}
	return 0;
}
int main()
{
	int t,c,n,k,p,a,b,flag;
	scanf("%d",&t);
	while(t--)
	{
		c=0; flag = 0;
		scanf("%d %d %d",&n,&k,&p);
		for(int i=0;i<k;i++)
		{
			scanf("%d %d",&a,&b);
			arr[c++]=a;
		}
		for(int i=0;i<p;i++)
		{
			scanf("%d %d",&a,&b);
			if(find(a,c)==0) flag = 1;
		}
		if(flag) printf("No\n");
		else printf("Yes\n");
	}
}
