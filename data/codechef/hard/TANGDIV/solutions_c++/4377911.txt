#include <bits/stdc++.h>
#define PB(n) push_back(n);
using namespace std;
vector <int> segments ;
void solve()
{
	int n,k,p;
	segments.clear();
	scanf("%d%d%d",&n,&k,&p);
	int left, right ;
	for(int i=0;i<k;i++)
	{
		scanf("%d%d",&left,&right);
		segments.PB(left);
	}
	bool flag = true ;
	sort(segments.begin(),segments.end());
	for(int i=0;i<p;i++)
	{
		scanf("%d%d",&left,&right);
		if(!binary_search(segments.begin(),segments.end(),left))
			flag = false ; 
	}
	if(flag)
		printf("Yes\n");
	else
		printf("No\n");
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		solve();		
	}

}
