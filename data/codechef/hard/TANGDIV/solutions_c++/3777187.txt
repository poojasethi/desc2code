#include<iostream>
#include<vector>
# define ll long long
using namespace std;
int main()
{
  ll n;
  int t,i,j,a,b,l,r,k,p,flag,d1;
  cin>>t;
  while(t--)
  {flag=0;
         vector<ll> v; v.clear();
         cin>>n>>k>>p;  d1=p;
         while(k--)
         {
           cin>>a>>b;
           v.push_back(a);
           }
           while(p--)
           {
             cin>>l>>r;
             for(i=0;i<v.size();i++)
             {
                if(l==v[i]) {flag++;  break;}                   
                
                }
                }
                if(flag==d1)
                cout<<"Yes"<<endl;
                else 
                cout<<"No"<<endl;
                }
                return 0;
                }
