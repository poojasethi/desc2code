#include <bits/stdc++.h>

using namespace std;

int main()
{
    int t, n, k, p, i, j;
    cin >> t;
    while (t--) {
        cin >> n >> k >> p;
        vector < pair <int, int> > a(k), b(p);
        for (i = 0; i < k; i++)
            cin >> a[i].first >> a[i].second;
        for (i = 0; i < p; i++)
            cin >> b[i].first >> b[i].second;
        sort(a.begin(), a.end());
        sort(b.begin(), b.end());
        i = 0;
        j = a.size();
        while (i < j && a[0].first != b[0].first) {
            a.push_back(a[0]);
            a.erase(a.begin());
            i++;
        }
        if (i == j) {
            cout << "No\n";
            continue;
        }
        i = 0;
        j = 0;
        while (j < b.size() && a[i].first == b[j].first) {
            while (i < a.size()) {
                if (b[j].first <= b[j].second) {
                    if (b[j].first <= a[i].first && a[i].first <= a[i].second && a[i].second <= b[j].second)
                        i++;
                    else
                        break;
                }
                else {
                    if (a[i].first <= a[i].second) {
                        if (b[j].first <= a[i].first || a[i].second <= b[j].second)
                            i++;
                        else
                            break;
                    }
                    else {
                        if (b[j].first <= a[i].first && a[i].second <= b[j].second)
                            i++;
                        else
                            break;
                    }
                }
            }
            i--;
            if (a[i].second != b[j].second)
                break;
            i++;
            j++;
        }
        if (j == b.size())
            cout << "Yes\n";
        else
            cout << "No\n";
    }
    return 0;
}
