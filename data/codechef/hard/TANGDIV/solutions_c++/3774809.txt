#include<iostream>
using namespace std;
int main()
{
	int n,k,p,t;
	cin>>t;
	while(t--){
		
	
	cin>>n>>k>>p;
	int A[k][2], B[p][2];
	int i,j;
	for(i=0;i<k;i++){
		for(j=0;j<2;j++)
		cin>>A[i][j];
	}
	for(i=0;i<p;i++){
		for(j=0;j<2;j++)
		cin>>B[i][j];
	}
	int F[p];
	for(i=0;i<p;i++){
		int flag=1;
		for(j=0;j<k;j++){
			if(B[i][0]==A[j][0]){flag=0; break;}
		}
		F[i]=flag;
	}
	int kk=1;
			for(i=0;i<p;i++){
			if(F[i]!=0){
				kk=0;break;
			}
		}
		if(kk==1)
		cout<<"Yes"<<endl;
		else cout<<"No"<<endl;
	}
}
