 
using namespace std ;
#include <iostream>
#include "stdio.h"
 
int main(){
 
	int testcases ;
	scanf("%i" , &testcases);
	for(int q=0 ; q<testcases ; q++){
 
		long long int chef[505][2] ;
		long long int boy[505][2] ;
 
		int n , k , p = 0;
		scanf("%i %i %i", &n , &k, &p );
		for(int i=0 ; i<k ; i++){
			int l , r ;
			scanf("%i %i", &l , &r);
			chef[i][0] = l ;
			chef[i][1] = r ;
		}
 
		for(int i=0 ; i<p ; i++){
			int a , b ;
			scanf("%i %i", &a , &b );
			
			boy[i][0] = a ;
			boy[i][1] = b ;
		}
		bool out = true ;
		for(int i=0 ; i<p ; i++){
			int l = boy[i][0];

			bool temp = false; 
			for(int j=0 ; j<k ; j++){
 
				if( l == chef[j][0] ){
						temp = true ;
					}
					
			}
			if(temp == false){
				out = false ;
				break ;
			}
		}

		if(out){
			printf("%s\n" , "Yes");
		}
		else{
			printf("%s\n" , "No");
		}
 
	}
 
	return 0 ;
}
