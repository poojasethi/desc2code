#include <cstdio>
#include <algorithm>

using namespace std;

int K[500];
int P[500];

int main()
{	
	int T,n,k,p,i,j;
	scanf("%d",&T);
	
	while(T--)
	{
		scanf("%d", &n);
		scanf("%d", &k);
		scanf("%d", &p);

		for(i = 0; i < k; i++)
		{
			scanf("%d", &K[i]);
			scanf("%d", &j);
		}
		for(i = 0; i < p; i++)
		{
			scanf("%d", &P[i]);
			scanf("%d", &j);
		}

		
		sort(K,K+k);
		sort(P,P+p);
		for(i = 0; i < p; i++)
		{
			for(j = 0; j < k; j++)
				if(K[j] == P[i]) break;
			
			if(j == k)
				break;
		}
		
		if(i < p) printf("No\n");
		else printf("Yes\n");
	}

	return 0;
}
