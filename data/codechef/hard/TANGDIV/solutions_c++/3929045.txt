/**
 * Note:
 *  Checking left side is sufficient and necessary condition
 *  because apart from the cut segments, remaining also form segments
 *  which cannot be obtained if there is a left side of boy's segment
 *  that do not correspond to the left side of chef's segment
 */
#include <iostream>
#include <set>
using namespace std;

int main() {
    int t;
    long long n, k, p, li, ri, ai, bi;
    cin >> t;
    while(t--) {
        set<long long> s;
        cin >> n >> k >> p;
        while(k--) {
            cin >> li >> ri;
            s.insert(li);
        }
        bool flag = true;
        while(p--) {
            cin >> ai >> bi;
            if ( s.find(ai) == s.end() ) flag = false;
        }
        if ( flag ) cout << "Yes" << endl;
        else cout << "No" << endl;
    }
    return 0;
}
