#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    int test;
    long  boy_l[503],chef_l[503],temp,j;
    long n,p,k,i;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%ld%ld%ld",&n,&k,&p);
        for(i=0;i<k;i++)
        {
            scanf("%ld%ld",&chef_l[i],&temp);
        }
        for(i=0;i<p;i++)
        {
            scanf("%ld%ld",&boy_l[i],&temp);
        }

        for(i=0;i<p;i++)
        {
            for(j=0;j<k;j++)
            {
                if(boy_l[i]==chef_l[j])
                {
                    break;
                }
            }
            if(j==k)
                    break;  //in case we do not find a case of boys tangerine in last so it terminates
        }
        if(i<p)
        {
             printf("No\n");
        }
        else
            printf("Yes\n");
    }
    return 0;
}
