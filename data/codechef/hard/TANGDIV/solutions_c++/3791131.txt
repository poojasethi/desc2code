struct cmp {
	bool operator() (const pair<int, char> &a, const pair<int, char> &b) {
		if (a.first != b.first) return a.first < b.first;
		else return a.second > b.second;
	}
} cmpO;

int t, n, k, p, i, j;
pair<int, char> spaces[2005];
bool child, chef;

bool solve () {
	
	for (i=0; i<k; i++) {
		scanf("%d %d", &spaces[i*2].first, &spaces[i*2+1].first);
		spaces[i*2].second = 'l';
		spaces[i*2+1].second = 'r';
	}
	
	for (i=k; i<p+k; i++) {
		scanf("%d %d", &spaces[i*2].first, &spaces[i*2+1].first);
		spaces[i*2].second = 'a';
		spaces[i*2+1].second = 'b';
	}
	
	sort(spaces, spaces+((k+p)<<1), cmpO);
	
	if (spaces[0].second != 'l' && spaces[0].second != 'r') return false;
	for (i=1; i<((p+k)<<1); i++) {
		if (spaces[i].second == 'a')
			if (spaces[i-1].second=='r' || spaces[i-1].first!=spaces[i].first) return false;
		else if (spaces[i].second == 'b')
			if (spaces[i-1].second=='l' || i>0 && spaces[i-1].first!=spaces[i].first) return false;
	}
	
	return true;
	
}

int main() {
