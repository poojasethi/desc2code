#include<iostream>
#include<cstdio>
#include<set>
#define in(x) scanf("%d", &x)

using namespace std;

int main() {
    int t;
    in(t);
    while(t--) {
        int n,k,p,x,y;
        int flag = 1;
        in(n); in(k); in(p);
        set<int> s;
        for(int i=0; i<k; i++) {
            in(x); in(y);
            s.insert(x);
        }
        for(int i=0; i<p; i++) {
            in(x); in(y);
            if(s.find(x)==s.end()) {
                flag = 0;
            }
        }
        if(flag) printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}
