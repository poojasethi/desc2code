#include<stdio.h>
#define gc getchar_unlocked
int read_int() {
  char c = gc();
  while(c<'0' || c>'9') c = gc();
  int ret = 0;
  while(c>='0' && c<='9') {
    ret = 10 * ret + c - 48;
    c = gc();
  }
  return ret;
}
int main()
{
    int a[505],b[505];
    int i,j,k,n,p,c,tc;
    tc=read_int();
    while(tc--)
    {
        n=read_int();
        k=read_int();
        p=read_int();


        for(i=0;i<k;i++)
          {
              a[i]=read_int();
              c=read_int();
          }


        for(i=0;i<p;i++)
         {
             b[i]=read_int();
             c=read_int();
         }

        for(i=0;i<p;i++)
        {
            for(j=0;j<k;j++)
            {
              if(b[i]==a[j])
                  break;
            }
          if(j==k)
             break;
        }
        if(i<p)
          printf("No\n");
        else
        printf("Yes\n");

    }
    return 0;
}
