#include<stdio.h>
int a[50000001];
int main()
{
    int t,n,pc,pb,l,r,i,size,size2,flag;
    int cp[500];
    int cb[500];
    /*for(i=1;i<=10000000;i++)
          a[i]=0;*/
    scanf("%d",&t);
    while(t--)
    {
         scanf("%d%d%d",&n,&pc,&pb);
         size=pc;
         size2=pb;
         i=0;
         while(pc--)
         {
             scanf("%d%d",&l,&r);
             cp[i]=l;
             a[cp[i]]=1;
             i++;
         }
         i=0;
         flag=1;
         while(pb--)
         {
            scanf("%d%d",&l,&r);
            if(a[l]!=1)
                   flag=0;
         }   
         i=0;
         if(flag==1)
              printf("Yes\n");
         else
              printf("No\n");
         for(i=0;i<size;i++)

              a[cp[i]]=0;
         }     
    }

}
