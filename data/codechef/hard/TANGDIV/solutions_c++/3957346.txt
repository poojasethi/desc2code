#include<bits/stdc++.h>

using namespace std;

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,k,p;
        scanf("%d%d%d",&n,&k,&p);
        set<int> set_k,set_p;
        for(int i=0;i<k;i++)
        {
            int temp1,temp2;
            scanf("%d%d",&temp1,&temp2);
            set_k.insert(temp1);
            set_k.insert( (temp2%n) +1);


        }
        for(int i=0;i<p;i++)
        {
            int temp1,temp2;
            scanf("%d%d",&temp1,&temp2);
            set_p.insert(temp1);
            set_p.insert( (temp2%n) +1);
        }
        bool flag=1;

        for(set<int>::iterator it=set_p.begin();it!=set_p.end();it++)
        {
            if(set_k.find(*it) == set_k.end())
                {flag=0;break;}
            else
                set_k.erase(*it);
        }

        if(flag)
            printf("Yes\n");
        else
            printf("No\n");


    }

    return 0;
}
