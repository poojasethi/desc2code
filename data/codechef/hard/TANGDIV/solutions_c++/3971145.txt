using namespace std;

int main()
{
	int t,k,p;
	int n,i,j,f;
	cin >> t;
	while(t--) {
		f = 0;
		cin >> n >> k >> p;
		int I[k+1],R[k+1],A[p+1],B[p+1];
		/*for(i = 0; i < n; i++) {
			a[i] = 0;
		}*/
		for(i = 0; i < k; i++) {
			cin >> I[i] >> R[i];
		}
		for(i = 0; i < p; i++) {
			cin >> A[i] >> B[i];
		/*	if (a[A[i]] == 1) {
				f = 1;
			}*/
		}
		for(i=0;i<p&&f == 0;i++) {
			for(j = 0; j <k&&f == 0;j++ ) {
				if(I[j] == A[i]){
					break;
				}
			}
			if(j == k) {
				f = 1;
			}
		}
		if (f == 1) {
			cout << "No" << endl;
		} else {
			cout << "Yes" << endl;
		}
	}
	return 0;
}
