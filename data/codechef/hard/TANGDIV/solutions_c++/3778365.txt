#include <iostream>
#include <cstdio>
#include <set>
#include <map>
#include <vector>
using namespace std;

int main() 
{
    int testcases;
    int length,chefpart,studpart,iter,leftend,rightend;
    scanf("%d",&testcases);
    set<int>myset;
    while(testcases--)
    {
      bool flag=false;
      scanf("%d %d %d",&length,&chefpart,&studpart);
      for(iter=0;iter<chefpart;iter++)
      {
        scanf("%d %d",&leftend,&rightend);
        myset.insert(leftend);
      }
      for(iter=0;iter<studpart;iter++)
      {
        scanf("%d %d",&leftend,&rightend);
        if(myset.find(leftend)==myset.end())
         flag=true;
      }
      if(flag)
       printf("No\n");
      else
       printf("Yes\n");
     myset.clear();  
    }
	return 0;
}
