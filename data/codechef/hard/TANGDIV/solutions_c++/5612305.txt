#include<bits/stdc++.h>
using namespace std;
set<int> S;
int main()
{
    int T, N, K, P;
    int x1, y1, x2, y2;
    cin>>T;
    while(T--)
    {
        cin>>N>>K>>P;
        S.erase(S.begin(),S.end());
        while(K--)
        {
            cin>>x1>>y1;
            S.insert(x1);
        }
        bool check = true;
        while(P--)
        {
            cin>>x2>>y2;
            if(S.find(x2)==S.end())check = false;
        }
        if(check==true)cout<<"Yes\n";
        else cout<<"No\n";
    }
}
