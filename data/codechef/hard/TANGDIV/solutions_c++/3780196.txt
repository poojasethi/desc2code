#include <iostream>
using namespace std;
int main(int argc, const char * argv[])
{
    int t;
    int n,k,p,x,y;
    int arr[505];
    int f;
    cin>>t;
    while(t--)
    {
        f=0;
        cin>>n;
        cin>>k;
        cin>>p;
    
        for(int i=0;i<k;i++)
        {
            cin>>x;
            cin>>y;
            arr[i]=y;
        }
        for(int i=0;i<p;i++)
        {
            cin>>x;
            cin>>y;
            for(int j=0;j<k;j++)
            {
                if(arr[j]==y)
                    f++;
            }
            
            
            
        }
        //cout<<f;
        if(f==p)
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    
        
    
    
    }
    return 0;
}
