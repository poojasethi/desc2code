#include <iostream>
#include <map>
#include <vector>
#include <cstdio>
#define ll long long int
using namespace std;

int main()
{
	ll t,i,j,n,k,p,flag;
	scanf("%lld",&t);
	while(t--) {
		flag=1;
		map<ll,ll> a;
		scanf("%lld %lld %lld",&n,&k,&p);
		for(i=0;i<k;i++) {
			scanf("%lld",&j);
			a[j]=1;
			scanf("%lld",&j);
		}
		for(i=0;i<p;i++) {
			scanf("%lld",&j);
			if(a[j]==0) flag=0;
			scanf("%lld",&j);
		}
		if(flag==0) printf("No\n");
		else printf("Yes\n");
	}
	return 0;
}
