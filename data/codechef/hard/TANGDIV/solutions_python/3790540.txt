def solve():
    N, K, P = map(int, raw_input().split())
    pos = set()
    for k in xrange(K):
        l, r = map(int, raw_input().split())
        pos.add(r)
    okay = True
    for p in xrange(P):
        l, r = map(int, raw_input().split())
        if okay and r not in pos:
            okay = False
    return 'Yes' if okay else 'No'

if __name__ == '__main__':
    T = int(raw_input())
    for _ in xrange(T):
        ans = solve()
        print ans
