#include<stdio.h>
typedef long long int ll;
int main()
{
int i,j,k,t,n,life;
ll sum;
bool ok;
scanf("%d",&t);
while(t--)
{
ok =true;
life=0;
scanf("%d",&n);
sum=0;
for(i=0;i<n;i++)
{
scanf("%d",&k);
if(k>0)
life++;
if(k>100)
ok=false;
sum+=k;
}
if(!ok)
printf("NO\n");
else if(sum<100)
printf("NO\n");
else if(sum-100<life)
printf("YES\n");
else
printf("NO\n");
}

}
