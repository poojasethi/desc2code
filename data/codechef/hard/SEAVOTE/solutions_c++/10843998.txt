#include <iostream>
#include <bits/stdc++.h>

typedef long long int ll;
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n,lo_lim=0,up_lim=0;
		cin>>n;
		int add = 0;
		for(int i=1;i<=n;i++)
		{
			int x;
			cin>>x;
			lo_lim += max(0,x-1);
			if(x!=0) add = 1;
			up_lim += x;
		}
		if(lo_lim+add<=100 && up_lim >= 100)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	 }
	return 0;
}
