#include<iostream>
using namespace std;
int main(){
    int t;
    cin>>t;
    while(t--){
        int n,count=0;
        cin>>n;
        int b[n];
        int sum=0;
        for(int i=0;i<n;i++){
            cin>>b[i];
            if(b[i]>0){
            sum=sum+b[i]-1;
            count++;}
        }
        if(sum<100 && (sum+count)>=100)
            cout<<"YES"<<endl;
        else
            cout<<"NO"<<endl;
    }
}
