#include<cstdio>
using namespace std;
typedef long long int ll;
int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        ll sum=0,zero=0;
        for(int i=0;i<n;i++){
            ll tmp;
            scanf("%lld",&tmp);
            sum+=tmp;
            if(tmp==0){
                zero++;
            }
        }
        int tmp=sum - n + zero;
        if(tmp<100 && sum>=100){
            printf("YES\n");
        }
        else{
            printf("NO\n");
        }
    }
}
