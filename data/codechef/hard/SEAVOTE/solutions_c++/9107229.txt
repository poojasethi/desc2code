#include<stdio.h>
int main()
{
	long long int max, min, t, b, i,n;
	scanf("%lld", &t);
	while (t--)
	{
		max = min = 0;
		scanf("%lld", &n);
		for (i = 0; i < n; i++)
		{
			scanf("%lld", &b);
			max += b;
			if (b)
				min += (b - 1);
		}
		min += 1;
		if (100 >= min && 100 <= max)
			printf("YES\n");
		else
			printf("NO\n");
	}
	getchar();
	getchar();
	return 0;
}
