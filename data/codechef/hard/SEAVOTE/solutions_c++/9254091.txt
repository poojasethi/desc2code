#include <cstdio>

using namespace std;

int main()
{
    int tc,n;
    scanf("%d",&tc);
    while(tc--)
    {
        scanf("%d",&n);
        int t;
        int sum=0;
        int m=n;
        for(int i=0;i<n;i++)
        {
            scanf("%d",&t);
            sum=sum+t;
            if(t==0)
            {
                m--;
            }
        }
        sum=sum-100;
        if(sum>=0 && sum<m)
        {
            printf("YES\n");
        }
        else
        {
            printf("NO\n");
        }
    }
    return 0;
}
