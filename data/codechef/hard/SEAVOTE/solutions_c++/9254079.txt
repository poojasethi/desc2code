#include <iostream>

using namespace std;

int main()
{
    int tc,n;
    cin>>tc;
    while(tc--)
    {
        cin>>n;
        int t;
        int sum=0;
        int m=n;
        for(int i=0;i<n;i++)
        {
            cin>>t;
            sum=sum+t;
            if(t==0)
            {
                m--;
            }
        }
        sum=sum-100;
        n=m;
        if(sum>=0 && sum<n)
        {
            cout<<"YES"<<endl;
        }
        else
        {
            cout<<"NO"<<endl;
        }
    }
    return 0;
}
