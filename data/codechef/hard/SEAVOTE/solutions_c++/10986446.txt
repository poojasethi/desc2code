#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main(){
	int t;
	cin>>t;
	while(t--){
		int n, a[10001]={}, i, j, sum=0, z=0, m;
		cin>>n;
		for(i=0; i<n; i++){
			cin>>a[i];
		}
		m=n;
		for(i=0; i<n; i++){
			if(a[i]==0)m--;
			if(a[i]>100)z=1;
			sum+=a[i];
		}
		if(sum>=100&&sum<100+m&&z==0)cout<<"YES\n";else cout<<"NO\n";
	}
}
