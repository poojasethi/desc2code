#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	scanf("%d",&t);
	while(t--){
		int n;
		scanf("%d",&n);
		int arr[n];
		int zero_counter=0;
		int check=0;
		for(int i=0;i<n;i++){
			scanf("%d",&arr[i]);
			if(!arr[i])
				zero_counter++;
			check+=arr[i];
		}
		if(check>100){
			if((check-100)<(n-zero_counter))
				printf("YES\n");
			else
				printf("NO\n");
		}
		else if(check==100)
			printf("YES\n");
		else
			printf("NO\n");
		
		
	}
	return 0;
}
