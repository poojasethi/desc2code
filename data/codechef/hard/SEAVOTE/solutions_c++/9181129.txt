#include <iostream>
using namespace std;

int main () {
	int t,n;
	cin >> t;

	int tot = 0;
	int count = 0;
	int x;
	for(int i = 0; i < t; i++) {
		cin >> n;
		tot = 0;
		count = 0;
		for(int j = 0; j < n; j++) {
			cin >> x;
			if (x != 0) {
				count++;
			}
			tot = tot + x;
		}
		if(tot < 100) {
			cout << "NO" << endl;
		}else if(tot == 100) {
			cout << "YES" << endl;
		}else {
			tot = tot - 100;
			if(tot < count) {
				cout << "YES" << endl;
			}else {
				cout << "NO" << endl;
			}
		}
	}

	return 0;
}
