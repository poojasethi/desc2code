#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long long int n,sum=0,a,i,c=0;
		float x;
		scanf("%lld",&n);
		for(i=0;i<n;i++)
		{
			scanf("%lld",&a);
			sum+=a;
			if(a!=0)
			c++;
		}
		if(sum==100)
		printf("YES\n");
		else if(sum>100&&sum-100<c)
		{
			printf("YES\n");
		}
		else 
		printf("NO\n");
	}
	return 0;
}
