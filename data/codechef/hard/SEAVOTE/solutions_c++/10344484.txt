#include<iostream>
using namespace std;

int main() {

    int t, n, b, sum, rem, counter;
    cin >> t;

    while (t--) {
        cin >> n;
        counter = n;
        sum = 0;

        while (counter--) {
            cin >> b;
            sum += b;

            if (b == 0) {
                n--;
            }
        }

        if (sum < 100) {
            cout << "NO" << endl;
        } else {
            rem = sum - 100;
            if (rem >= n) {
                cout << "NO" << endl;
            } else {
                cout << "YES" << endl;
            }
        }
    }

    return 0;
}
