#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n;
		scanf("%d",&n);
		int sum=0;
		int b_i;
		int zeroes=0;
		for(int i=0;i<n;i++)
		{
			scanf("%d",&b_i);
			sum+=b_i;
			if(b_i!=0)
				zeroes++;
		}
		if(sum==100)
		{
			cout<<"YES\n";
			continue;
		}
		else if(sum<100)
		{
			cout<<"NO\n";
			continue;
		}
		else if(sum-zeroes<100)
		{
			cout<<"YES\n";
		}
		else
		{
			cout<<"NO\n";
		}
	}
}
