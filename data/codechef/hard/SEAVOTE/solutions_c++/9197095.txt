#include<stdio.h>
int main()
{
	int t, n, b;
	long sum;
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d", &n);
		int k = n;
		sum = 0;
		while (k--)
		{
			scanf("%d", &b);
			if (b > 0)
				sum += b;
			else
				--n;
		}
		if (sum < 100)
			printf("NO\n");
		else
		{
			if ((sum - 100.0) / n < 1.0)
				printf("YES\n");
			else
				printf("NO\n");
		}
	}
	return 0;
}
