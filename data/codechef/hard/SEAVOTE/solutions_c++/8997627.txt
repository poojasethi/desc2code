#include<stdio.h>
#include<stdlib.h>

int main()
{
	int i,t,x,n,sum,count;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		sum=count=0;
		for(i=0;i<n;i++)
		{
			scanf("%d",&x);
			sum+=x;
			if(x==0)
			count++;
		}
		if(sum>=100 && sum<=(100+n-1-count))
		printf("YES\n");
		else
		printf("NO\n");
	}
}
