using namespace std;

//objective is to make n using x values
bool check(int x, int n)
{
    if(n>1)
    {
        if(x%n!=0)
            return false;

        x/=n;
    }

    //make 1 using x values

    //factorize
    while(x%2==0)
    {
        x/=2;
    }

    while(x%5==0)
    {
        x/=5;
    }

    if(x==1)
        return true;

    return false;
}

int main()
{
    int i,j,k,n;
    long long int sum;
    int t;

    int y;

    bool result,result2;

    cin>>t;
 
    while(t--)
    {
        cin>>n;
        sum=0;
         
        vector<int> a(n);

        x=0;
        result=true;

        for(i=0;i<n;i++)
        {
            cin>>a[i];

            if(a[i]==0)
            {
                x++;
            }

            if(a[i]>100)
            {
                result=false;
            }

            sum+=a[i];
        }

        n=(n-x);
         
        if(result && sum>=100 && sum<=(n+99))
        {

        }

    }

}
