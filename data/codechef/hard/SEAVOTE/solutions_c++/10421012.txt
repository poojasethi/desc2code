#include<iostream>
using namespace std;
int main()
{
	int t,m,i;
	cin>>t;
	while(t--)
	{
	    long long n,s=0;
		cin>>n;
		long long a[n];
		m=n;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			if(a[i]==0)
			{
				m--;
			}
			else
			s+=a[i];
		}
		s-=100;
		if(s>=0 && s<m)
			cout<<"YES"<<endl;
		else
			cout<<"NO"<<endl;
	}
}
