#include<iostream>
using namespace std;



int main()
{
    int  T;
    cin >> T;
    while(T--){
    long long n=0,sum=0,a[100001],N;
    cin >> N;
    for(int i=0;i<N;i++){
        cin >> a[i];
        sum+=a[i];
        if(a[i]!=0)
            n++;
    }
    sum = sum-100;
    if(sum<n && sum>=0){
        cout << "YES\n";
    }
    else
        cout << "NO\n";
    }

}
