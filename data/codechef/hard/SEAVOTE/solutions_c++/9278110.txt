#include<bits/stdc++.h>
#include <cstdio>
#include<algorithm>
using namespace std;

int main()
{
    int t,n,x,add,lo,i,up;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&n);
        up=0;lo=0;add=0;
        for(i=0;i<n;i++){
            scanf("%d",&x);
            lo=lo+max(0,x-1);

            if(x!=0)
                add=1;
            up=up+x;
        }
        cout<<(((lo+add<=100 && up>=100)? "YES":"NO"))<<endl;
    }
    return 0;
}
