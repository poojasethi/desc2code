#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int N;
		scanf("%d",&N);
		int b, lsum = 0, rsum = 0;
		for(int i = 0; i < N; i++)
		{
			scanf("%d",&b);
			if(b != 0)	
				lsum += (b - 1);
			rsum += b;
		}
		if(rsum >= 100 && lsum < 100)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
