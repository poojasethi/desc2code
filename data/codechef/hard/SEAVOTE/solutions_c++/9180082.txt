#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll arr[10010];
int main()
{
	ll t,i,n;
	cin>>t;
	while(t--)
	{
		cin>>n;
		ll sum=0;
		ll count=0;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			if(arr[i]!=0)
			sum+=arr[i]-1;
			else
			count++;
			
		}
		n-=count;
		if(sum+n<100)
		cout<<"NO"<<endl;
		else if(sum+n==100)
		cout<<"YES"<<endl;
		else
		{
			if(sum>=100)
			cout<<"NO"<<endl;
			else
			{
				ll aa=100-sum;
				if(n>aa)
				cout<<"YES"<<endl;
				else
				cout<<"NO"<<endl;
			}
		}
	}
}
