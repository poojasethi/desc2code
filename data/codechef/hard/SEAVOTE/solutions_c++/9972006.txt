#include<bits/stdc++.h>
using namespace std;
 
int main()
{
    int i,j,k,n;
    int sum;
    int t;
    cin>>t;
 
    while(t--)
    {
        cin>>n;
        sum=0;
        
        k=0;
        for(i=0;i<n;i++)
        {
            cin>>j;
            sum+=j;

            if(j==0)
                k++;
        }
        
        n=n-k;

        if(sum>=100 && sum<=(n+99))
        {
            cout<<"YES";
        }
         
        else
            cout<<"NO";
         
        cout<<endl;
    }
	return 0;
}
