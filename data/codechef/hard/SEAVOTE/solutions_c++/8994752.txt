
#include <iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cmath>

using namespace std;


int main(){

int t,i;
cin>>t;
while(t--){
    int n,sum=0,count=0;
    cin>>n;
    int a[n];
    for(i=0;i<n;i++){
        cin>>a[i];
        if(a[i]==0) count++;
        sum+=a[i];

    }
    n=n-count;
    if (sum==100)cout<<"YES" <<"\n";
    else if(sum<100)cout<<"NO"<<"\n" ;
    else if(sum-n<100)cout<<"YES"<<"\n";

        else  cout<<"NO"<<"\n";

}

return 0;
}
