#include <bits/stdc++.h>
using namespace std;
 
#define FOR(i,a,b) for(int i=a;i<=b;++i)
 
 
int main() 
{
    ios_base::sync_with_stdio(false);cin.tie(0);
    int T;
    cin>>T;
    while(T--)
    { int flag=1;
     int n,n2;cin>>n;n2=n;int s=0;FOR(i,1,n){int t;cin>>t;if(t==0)--n2;s+=t;if(t>100)flag=0;}
     if(flag&&s>=100&&(s-100)<n2)cout<<"YES\n";
     else cout<<"NO\n";
    }
	return 0;
}
