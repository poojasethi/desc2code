

T = input()
for i in range(T):
    N = input()
    votes = map(int, raw_input().split())
    non_zero_count = 0
    for vote in votes:
        if vote > 0:
            non_zero_count += 1
    
    total = sum(votes)
    if total < 100:
        print "NO"
    elif total > 99 + non_zero_count:
        print "NO"
    else:
        print "YES"
