t=int(raw_input())
for i in xrange(t):
    n=int(raw_input())
    a=raw_input().split()
    s=0
    count=0
    for j in xrange(n):
        a[j]=int(a[j])
        if a[j]==0:
            count+=1
        s+=a[j]
    #print s,s-n
    if s<100:
        print "NO"
    elif s==100:
        print "YES"
    elif s-n+count<100:
        print "YES"
    else:
        print "NO"
