num_of_cases = int(raw_input())

for x in xrange(num_of_cases):
	num_of_numbers = int(raw_input())

	numbers = map(int, raw_input().split())
	numbers = [x for x in numbers if x != 0]
	num_of_numbers = len(numbers)

	sum_numbers = sum(numbers)

	if sum_numbers < 100:
		print "NO"
	elif sum_numbers >= 100 and sum_numbers <= 100 + num_of_numbers - 1:
		print "YES"
	else:
		print "NO"
