def run():
	t = input()
	for _ in xrange(t):
		# For each test, case first line contains integer N - number of opinions.
		N = input()
		# Next line contains N integers B1, B2, ..., BN as defined in the problem.
		B = map(int, raw_input().split())
		sumB = sum(B)
		n0 = len([a for a in B if a == 0])
		lower_bound = sumB - (N-n0)
		if lower_bound < 100 <= sumB:
			print 'YES'
		else:
			print 'NO'

run()
