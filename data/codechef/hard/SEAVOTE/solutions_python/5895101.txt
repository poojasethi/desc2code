t = input()
for _ in range(t):
    n = input()
    b = raw_input().split()
    b = map(int,b)
    num_of_zero = sum(x == 0 for x in b)
    non_zero = n - num_of_zero

    if sum(b) < 100:
        print 'NO'
    elif sum(b) > 100 and (sum(b)-non_zero) >= 100:
        print 'NO'
    else:
        print 'YES'
