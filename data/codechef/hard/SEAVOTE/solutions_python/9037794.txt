from pprint import pprint as pp
def GI(): return int(raw_input())
def GIS(): return map(int, raw_input().split())

def main():
    T = GI()
    for t in xrange(T):
        N = GI()
        B = GIS()
        excess = sum(B) - 100
        if excess < 0:
            print 'NO'
        elif excess < N - B.count(0):
            print 'YES'
        else:
            print 'NO'

main()
