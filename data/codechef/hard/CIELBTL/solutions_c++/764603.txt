double dp[200][200][10]; // our hp, his hp, mana
double sum[200][200][10];

int va, sa, vb, sb, m;

double getSum(int i, int j, int mana, double ans)
{
	double s = sum[i][j-1][mana] + sum[i-1][j][mana] - sum[i-1][j-1][mana];
	int tx = i-sb;
	int ty = j-sa;

	int x = max(tx,0);
	int y = max(ty,0);

	if(x)
		s -= sum[x-1][j][mana];
	if(y)
		s -= sum[i][y-1][mana];
	if(x && y)
		s += sum[x-1][y-1][mana];

	tx = max(0,-tx);
	ty = max(0,-ty);

	int ones = ty*(i-max(x,1)+1);
	s += ones;
	if(y == 0 && x == 0)
		s += ((tx+1)*(ty+1)-1)*ans;

	return s;
}

double f(double ans)

			sum[i][0][mana] = 1+sum[i-1][0][mana];
		}
		for(int j = 1; j <= vb; j++)
		{
			dp[0][j][mana] = 0;
			sum[0][j][mana] = ans;
		}
	}
	fori(mana,m+1)
		for(int i = 1; i <= va; i++)
			for(int j = 1; j <= vb; j++)
			{
				dp[i][j][mana] = 0;
				if(mana)
					dp[i][j][mana] = dp[(i+1)/2][(j+1)/2][mana-1];
				double area = (sa+1)*(sb+1) - 1;
				double b = getSum(i,j,mana,ans);				
				b /= area;
				dp[i][j][mana] = max(dp[i][j][mana],b);

				sum[i][j][mana] = dp[i][j][mana] + 
					sum[i-1][j][mana]+sum[i][j-1][mana]
					-sum[i-1][j-1][mana];
