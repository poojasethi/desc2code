#define FOR(i,a,b) for (int i(a); i < (b); i++)
#define REP(i,n) FOR(i,0,n)
#define SORT(v) sort((v).begin(),(v).end())
#define UN(v) sort((v).begin(),(v).end()),v.erase(unique(v.begin(),v.end()),v.end())
#define CL(a,b) memset(a,b,sizeof(a))
#define pb push_back
 
int va,vb,sa,sb,m;
 
double d[2][101][101];
double s[101][101];
 
double solve(double P){
	for(int M=0;M<=m;M++){
		int curr = M%2;
		int prev = 1-curr;
 
		REP(i,va+1)REP(j,vb+1){
 
			d[curr][i][j] = 0;
 
			if(i==0 && j==0) d[curr][i][j] = P;
			else if(i==0) d[curr][i][j] = 0;
			else if(j==0) d[curr][i][j] = 1;
			else{
				double sum = 0;
 
				//[i-sa] .. [i]  * [j-sb] ... [j]
 
				int beg_i = i - sb;
				int beg_j = j - sa;
 
				if(beg_i <=0 && beg_j <=0) sum += (-beg_i+1)*(-beg_j+1)*P;
				if(beg_i <=0) sum += 0 * 1;
				if(beg_j <=0) sum += 1 * (-beg_j+1) * (i - max(beg_i,1)+1);
				beg_i = max(beg_i,1);
				beg_j = max(beg_j,1);
 
				sum += s[i-1][j] + s[i][j-1] - s[i-1][j-1];
				sum -= s[beg_i-1][j];
				sum -= s[i][beg_j-1];
				sum += s[beg_i-1][beg_j-1];
 
				
				d[curr][i][j] = sum / ((sa + 1) * (sb + 1)-1);
				if(M) d[curr][i][j] = max(d[curr][i][j], d[prev][(i+1)/2][(j+1)/2]);
			}
 
			s[i][j] = d[curr][i][j];
			if(i) s[i][j] += s[i-1][j];
			if(j) s[i][j] += s[i][j-1];
			if(i&&j) s[i][j] -= s[i-1][j-1];
		}
	}
	return d[m%2][va][vb];
}
