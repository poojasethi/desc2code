static inline void scan(int& x){
    x = 0;
    register char c = getchar_unlocked();
    while(c<'0'||c>'9') c = getchar_unlocked();
    while(c>='0'&&c<='9'){
            x=(x<<1)+(x<<3)+c-'0';
    c=getchar_unlocked();
    }
}
