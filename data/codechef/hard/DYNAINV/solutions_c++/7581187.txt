#include<stdio.h>
using namespace std;
int n, q, a[100000 + 5], f[100000 + 5], i, t, j, ret, x, y;
int main(int argc, char * const argv[])
{

    scanf("%d %d",&n,&q);
    for( i=1;i<=n;i++)
        scanf("%d",&a[i]);
    for(i = 1; i <= n; i++) {
		t = 0;
		for(j = a[i]; j; j = j & (j - 1)) t += f[j];//printf("%d ",t);}
		for(j = a[i]; j <= n; j = (j | (j - 1)) + 1) ++f[j];//printf("@ %d ",fwt[j]);

		ret = (ret + i - t - 1) % 2;
	//	printf("ret %d" ,ret);// All we need is just a parity of this number, so we take it modulo 2
	}
	for(i=0;i<q;i++){
        scanf("%d %d",&x,&y);
        ret ^= 1;
		printf("%d\n", ret);
	}
	return 0;
}
