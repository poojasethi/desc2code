#include<iostream>
#include<algorithm>
using namespace std;
long long int n,a[100010],b[100010];
long long int binary_searh(long long int low,long long int high,long long int value){
while(low<high){
long long int mid=low+(high-low)/2;
if(b[mid]==value)
return mid;
else{
if(b[mid]<value)
low=mid+1;
else
high=mid-1;
}
}
return low;
}
void update(long long int p,long long int c[]){
for(p++;p<=n;p+=p&(-p)){
c[p]++;
}
return;
}
long long int read(long long int p,long long int c[]){
long long int sum=0;
for(p;p>0;p-=p&(-p)){
sum+=c[p];
}
return sum;
}
int main(){
long long int i,c[100010]={0},sum=0,d,x,y,q;
int even;
cin>>n>>q;
for(i=0;i<n;i++){
cin>>a[i];
b[i]=a[i];
}
sort(b,b+n);
for(i=0;i<n;i++){
a[i]=binary_searh(0,n-1,a[i]);
}
for(i=n-1;i>=0;i--){
sum+=read(a[i],c);
update(a[i],c);
}
if(sum%2==0)
even=1;
else
even=0;
while(q--){
cin>>x>>y;
if(even){
cout<<"1"<<endl;
even=0;
}
else{
cout<<"0"<<endl;
even=1;
}
}
return 0;
}
