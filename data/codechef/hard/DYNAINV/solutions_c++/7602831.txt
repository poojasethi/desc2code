#include<bits/stdc++.h>
using namespace std;

const int N=123456;
int a[N], temp[N];

int _merge(int from, int mid, int to) {
    int i=from, j=mid+1, k=from, res=0;
    while(i<=mid && j<=to) {
        if(a[i]<=a[j]) {
            temp[k++]=a[i++];
        } else {
            temp[k++]=a[j++];
            res+=(mid-i+1);
        }
    }
    while(i<=mid) {
        temp[k++]=a[i++];
    }
    while(j<=to) {
        temp[k++]=a[j++];
    }
    for(int i=from;i<=to;i++) {
        a[i]=temp[i];
    }
    return res;
}

int invcnt(int from, int to) {
    int res=0;
    if(from<to) {
        int mid=from+(to-from)/2;
        res+=invcnt(from, mid);
        res=res%2;
        res+=invcnt(mid+1, to);
        res=res%2;
        res+=_merge(from, mid, to);
        res=res%2;
    }
    return (res%2+2)%2;
}

int main() {
    int n, q;
    scanf("%d %d", &n, &q);
    for(int i=0;i<n;i++) {
        scanf("%d", &a[i]);
    }
    int ans=invcnt(0, n-1);
    ans=ans%2;
    while(q--) {
        int l, r;
        scanf("%d %d", &l, &r);
        ans=ans^1;
        printf("%d\n", ans);
    }
}
