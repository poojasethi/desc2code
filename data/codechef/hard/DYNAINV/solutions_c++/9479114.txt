#include<bits/stdc++.h>
using namespace std;
int bit[100001];
int a[100001];
int getcount(int index)
{
    int sum=0;

    while(index>0)
    {
        sum+=bit[index];
        index-=(index&(-index));
    }
    return sum;
}
void update(int index,int n,int val)
{
    while(index<=n)
    {
       bit[index]+=val;
       index+=index&(-index);
    }
}
int  solve(int a[],int n)
{
    int max1=n;
    int coun_inv=0;
    int i;
    memset(bit,0,sizeof(bit));
    for(i=n-1;i>=0;i--)
    {
        coun_inv+=getcount(a[i]-1);
        update(a[i],max1,1);

    }
    return coun_inv;
}
