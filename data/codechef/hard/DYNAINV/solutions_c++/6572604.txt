#include<bits/stdc++.h>
using namespace std;

#define MAX 1000010

int a[MAX];
int n,i;
int solve()
{
    int ans=0;
    for(i=1;i<=n;i++)
    {
        if(a[i]!=i) {
            swap(a[i],a[a[i]]);
            ans++;
            i--;
        }
    }
    return ans;
}
int main()
{
    int q; cin >> n >> q;
    for(i=1;i<=n;i++)
        cin >> a[i];
    int ans=solve();
    ans=ans%2;
    while(q--)
    {
        int x,y;
        scanf("%d%d",&x,&y);
        ans=ans^1;
        printf("%d\n",ans);
    }
    return 0;
}
