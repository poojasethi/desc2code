#include <iostream>
using namespace std;

int f[100005];
long long ans;
int n,q,p;


int get(int x){
    int d = 0;
    while (x <= n) {
        d += f[x];
        x = x | (x + 1);
    }
    return d;
}
 
void add(int x) {
    while (x >= 0){
        f[x] += 1;
        x = x & (x + 1);
        x--;
    }
}

int main() {
	scanf("%d%d",&n,&q);
	
	for(int i=0;i<n;i++){
		scanf("%d",&p);
		ans += get(p);
		add(p);
	}
	ans %= 2;
	for(int i=0;i<q;i++){
		scanf("%d %d",&p,&p);
		ans = 1^ans;
		printf("%lld\n",ans);
