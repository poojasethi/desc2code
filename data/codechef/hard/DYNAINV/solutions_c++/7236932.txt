#include<bits/stdc++.h>
using namespace std;

const int N=123456;
int bit[N], a[N];

void update(int idx, int val) {

    while(idx<=N) {
        bit[idx]+=val;
        idx+=(idx & -idx);
    }
}

long long get(int idx) {

    long long sum=0;
    while(idx>0) {
        sum+=(long long)bit[idx];
        idx-=(idx & -idx);
    }
    return sum;
}

int main() {
    int n, q;
    scanf("%d %d", &n, &q);
    for(int i=1;i<=n;i++) {
        scanf("%d", &a[i]);
        update(i, 1);
    }
    long long ans=0;
    for(int i=1;i<=n;i++) {
        ans+=get(a[i]-1);
        update(a[i], -1);
    }
    bool res=ans%2;
    while(q--) {
        int x, y;
        scanf("%d %d", &x, &y);
        res^=1;
        printf("%d\n", res);
    }
}
