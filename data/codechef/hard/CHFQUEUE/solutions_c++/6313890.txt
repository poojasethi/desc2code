#include<stdio.h>
#include<stack>

#define mod 1000000007

using namespace std;

int main()
{
    int n,k,i,j;
    long long mul=1;
    stack<int> s;
    scanf("%d %d",&n,&k);
    int a[n];
    for(i=0;i<n;i++)
    {
    scanf("%d",&a[i]);
    while(!s.empty() && a[s.top()]>a[i])
    {
        j= s.top();
        mul= (mul*(i-j+1))%mod;
        s.pop();
    }
    s.push(i);
    }
    printf("%lld\n",mul);
    return 0;
}
