inline void inp( int &n )
{
n=0;
int ch=inchar();int sign=1;
while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=inchar();}
while( ch >= '0' && ch <= '9' )
n = (n<<3)+(n<<1) + ch-'0', ch=inchar();
n=n*sign;
}

typedef pair<int,int> ii;
stack<ii> s;

int a[mx];


int main() {

    int n,k;
    inp(n);inp(k);
    FOR(i,1,n) inp(a[i]);
    ll ans = 1;
    FOR(i,1,n)
     {
        if(s.empty() || a[i] >= s.top().first ) s.push(ii(a[i],i));
        else
         {
           while(!s.empty() && a[i] < s.top().first)
            {
              ii x = s.top();
              s.pop();
              ans = (ans*((ll)(i-x.second+1)))%mod;
            }            
           s.push(ii(a[i],i));          
         }    
     }
    printf("%lld\n",ans);
    
    
    //getchar();
    //getchar();
	return 0;
}
