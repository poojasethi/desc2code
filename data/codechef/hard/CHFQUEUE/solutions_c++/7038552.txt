#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main()
{
    int n,k,i,x;
    long long int res=1;
    pair<int,int> mypair;
    stack< pair<int,int> > mystack;
    mypair=make_pair(0,0);
    mystack.push(mypair);
    scanf("%d%d",&n,&k);
    for(i=1;i<=n;i++)
    {
        scanf("%d",&x);
        while(mystack.size()!=0 && mystack.top().first>x)
        {
            res=(res*(i-mystack.top().second+1))%mod;
            mystack.pop();
        }
        mypair=make_pair(x,i);
        mystack.push(mypair);
    }
    printf("%lld\n",res);
    return 0;
}
