#include <iostream>
#include <stack>

int main() {
    int N, K;
    long long modNum = 1000000007, ans = 1;
    std::cin >> N >> K;
    std::stack<std::pair<int, int> > chefstack;
    for (int i = 0; i < N; ++i) {
        int n;
        std::cin >> n;
        std::pair<int, int> chef(n, i);
        while (chefstack.size() >= 1 && chef.first < chefstack.top().first){
            ans *= chef.second - chefstack.top().second + 1;
            ans %= modNum;
            chefstack.pop();
        }
        chefstack.push(chef);
    }
    std::cout << ans << std::endl;
    return 0;
}
