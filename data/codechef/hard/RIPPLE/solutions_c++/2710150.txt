void subarrays(int t){

	char temp[n+1];

		}	
	}
	else {//add
		ctr=0;
		for(int i=1;i<=t;i= i << 1){if(t&i)temp[n-ctr]=1;ctr++;}
		ctr=n;
		ctr2=0;//balance
		while(ctr>=0){
			if(veca[ctr]+temp[ctr]+ctr2==3){}
