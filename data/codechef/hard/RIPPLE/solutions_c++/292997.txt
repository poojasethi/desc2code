#include<iostream>
#include<string>
#include<vector>
#include<cstdio>
#include<algorithm>

using namespace std;

int main(){
	int t;
	char a[200000],b[200000];
	int n;
	int as,bs;
	int c,sum;
	scanf("%d",&t);
	while(t!=0){
		t--;
		scanf("%d %s %s",&n,&a,&b);
		bs=n;
		as=0;
		for(int i=0;i<n;i++){
			if(a[i]=='1'){
				as--;
			}
			if(b[i]=='1'){
				as++;
			}
			if(a[i]=='1'){
				if(b[i]=='1'){
					b[i]='0';
				}else{
					b[i]='1';
					c=i-1;
					while(b[c]!='1'){
						b[c--]='1';
					}
					b[c]='0';
				}
			}
		}
		b[n]='0';
		b[n+1]='\0';
		bs=n;
		if(as>0){
			while(as!=0){
				if((as^(as-1))==1){
					if(b[bs]=='1'){
						b[bs]='0';
					}else{
						b[bs]='1';
						c=bs-1;
						while(b[c]!='1'){
							b[c--]='1';
						}
						b[c]='0';
					}
				}
				as=as>>1;
				bs--;
			}
		}else{
			as=-as;
			while(as!=0){
				if((as^(as-1))==1){
					if(b[bs]=='0'){
						b[bs]='1';
					}else{
						b[bs]='0';
						c=bs-1;
						while(b[c]!='0'){
							b[c--]='0';
						}
						b[c]='1';
					}
				}
				as=as>>1;
				bs--;
			}
		}
		c=0;
		for(int i=0;i<n+1;i++){
			if(c==0 && b[i]=='0'){
				//do nothing
			}else{
				printf("%c",b[i]);
				c=1;
			}
		}
		printf("\n");
	}
	return 0;
}
