#include <stdio.h>

#define MAX 100000


void oduzmi(char *niz1, int n, char *niz2, int m)
{
    int temp, i;

    temp = 0;
    for (i = 0; i < m; i++) {
	niz1[i] -= niz2[i] + temp;
	if (niz1[i] < 0) {
	    niz1[i] += 2;
	    temp = 1;
	} else
	    temp = 0;
    }
    while (temp) {
	niz1[i]--;
	if (niz1[i] < 0) {
	    niz1[i] += 2;
	    temp = 1;
	} else
	    temp = 0;
	i++;
    }
}


int pretvori(char *niz, int n)
{
    int i = 0;

    while (n) {
	niz[i++] = n & 1;
	n >>= 1;
    }

    return i;
}

int ispis(char *niz, int n)
{
    int i;
    for (i = n - 1; i >= 0 && niz[i] == 0; i--);
    if (i < 0)
	printf("0");

    for (; i >= 0; i--)
	printf("%d", niz[i]);
    printf("\n");
}
