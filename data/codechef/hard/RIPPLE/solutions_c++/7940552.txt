#include<cstdio>
 
using namespace std;
 
int l,n;
char s1[100010],s2[100010],s3[100010];
 
void make(int v)
{
    l = 0;
    if(v<0) v *= -1;
    while(v) {
        s3[l++] = v%2 + '0';
        v /= 2;
    }
    s3[l] = '\0';
}
 
void comp(char s[])
{
    for(int i=n-1;i>=0;i--) {
        s[i] = (char)(!(s[i]-'0')+'0');
    }
}
 
void ad(char s[])
{
    int i=0;
    while(s[i] == '1') {
        s[i] = '0';
        i++;
    }
    s[i] = '1';
}
 
void add(char s1[],char s2[])
{
    int v = 0;
    for(int i=0;i<n;i++) {
        v += (s1[i]-'0') + (s2[i]-'0');
        s1[i] = (v%2 + '0');
        v /= 2;
    }
}
 
void rev(char str[])
{
    char ch;
    for(int i=0;i<n/2;i++) {
        ch = str[i];
        str[i] = str[n-i-1];
        str[n-i-1] = ch;
    }
}
 
int main()
{
    int t;
    scanf("%d",&t);
    while(t--) {
        scanf("%d",&n);
        scanf("%s",s2);
        scanf("%s",s1);
 
        s1[n] = s2[n] = '0';
        s1[n+1] = s2[n+1] = '\0';
        n++;
 
        int x1,x2;
        x1 = x2 = 0;
 
        for(int i=0;i<n;i++) x1 += (s1[i]=='1');
        for(int i=0;i<n;i++) x2 += (s2[i]=='1');
 
        rev(s1);
        rev(s2);
 
        s1[n] = s2[n] = '0';
        s1[n+1] = s2[n+1] = '\0';
        n++;
 
        make(x1-x2);
        for(int i=l;i<n;i++) s3[i] = '0';
        s3[n] = '\0';
 
        l = n;
        comp(s2);
        ad(s2);
 
        if(x1-x2 > 0) comp(s3);
        if(x1-x2 > 0) ad(s3);
 
        add(s1,s2);
 
        add(s1,s3);
        bool f = 0;
        for(int i=n-1;i>=0;i--) {
            if(s1[i] == '1') f = 1;
            if(f) printf("%c",s1[i]);
        }
        printf("\n");
    }
}
