int cal1(int *b2,int *b1)
{
    int count1=0;
    b2[100004]=0;
    for(int i=0;i<n;i++)
    {
        if(b1[100004-i]==1)
        {
            b2[100004-i-1]=1;
            count1++;
        }
        else
            b2[100004-i-1]=0;
    }
    return count1;
}
void sub(int *a2,int *b2,int *ans)

              ans[100004-i]=0;

          }
        }

        {
          if(a2[100004-i]>b2[100004-i])
          {
              b=1;
              ans[100004-i]=1;
          }
          else
          {
              ans[100004-i]=b2[100004-i]-a2[100004-i];
          }
        }
    }
    return;
}
void to_binary(int *dec,int d)
{
    int i=0;
    while(d!=0)
    {
        dec[100004-i]=d%2;
        d/=2;
        i++;
    }
    return;
