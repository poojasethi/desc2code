#include <cstdio>
#include <iostream>

int bits[2][200000];

void sum (int b[]) {
	int carry = 0;
	int i = 0;
	do {
		carry = b[i] + carry;
		b[i] = carry&1;
		carry = carry>>1;
		i++;
	} while (carry);
}

void compliment (int b[]) {
	int _xor = 0;
	for (int i=0; i<200000; i++) {
		b[i] = b[i] ^ _xor;
		if (b[i])
			_xor = 1;
	}
}

void add (int a[], int b[]) {
	int carry = 0;
	int tmp = 0;
	for (int i=0; i<200000; i++) {
		tmp = a[i] + b[i] + carry;
		carry = 0;
		if (tmp > 1) {
			a[i] = tmp&1;
			carry = 1;
		} else
			a[i] = tmp;
	}
}

int main(int argc, char** argv) {
	
	int tc; 
	scanf("%d", &tc);
	int bitNum;
	
	int i, j;
	while (tc--) {
		scanf("%d", &bitNum);
		getchar();
		
		for (i=0; i<200000; i++)
			bits[0][i] = bits[1][i] = 0;
		
		bits[0][bitNum-1] = getchar()-'0';
		for (i=bitNum-2; i>=0; i--)
			bits[0][i] = getchar()-'0'+bits[0][i+1];
		getchar();
		
		sum(bits[0]);
		compliment(bits[0]);
		
		bits[1][bitNum-1] = getchar()-'0';
		for (i=bitNum-2; i>=0; i--)
			bits[1][i] = getchar()-'0'+bits[1][i+1];
		getchar();
		
		sum(bits[1]);
		add(bits[1], bits[0]);
		
		int j = 0;
		for (i=199999; i>=0; i--) {
			if (bits[1][i]) j=1;
			if (j)
				printf("%d", bits[1][i]);		
		}
		printf("\n");
	}			
	return 0;
}
