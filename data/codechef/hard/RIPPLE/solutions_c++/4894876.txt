#include<bits/stdc++.h>
using namespace std;
#define MAX 1000010
int f_a[MAX],f_b[MAX];
int aa[MAX],bb[MAX];
int ans_b[MAX],ans_a[MAX];int final_ans[MAX];
void trace(int* f_a,int n)
{
	for(int i=1;i<=n;i++)
	   cout<<f_a[i]<<" ";
	  cout<<endl;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
	  int n;
	  string a,b;
	  cin>>n>>a>>b;
	  for(int i=1;i<=n;i++)
	  {
	  	aa[i] = a[n-i]-'0';
	  	bb[i]=  b[n-i]-'0';
	  }
	  for(int i=0;i<n+10;i++)
	  {
	  	f_a[i]=0;f_b[i]=0;ans_a[i]=0;ans_b[i]=0;final_ans[i]=0;
	  }
	  for(int i=1;i<=n;i++)
	  {
	  	if(aa[i]==1)
	  	{
	  		f_a[1]   += 1;
	  		f_a[i+1] -= 1;
	  	}
	  	if(bb[i]==1)
	  	{
	  		f_b[1] +=1;
	  		f_b[i+1] -=1;
	  	}
	  }
	  
	  for(int i=1;i<=n;i++)
	  {
	  	f_a[i] += f_a[i-1];
	  	f_b[i] += f_b[i-1];
	  }
	  f_a[n+1]=0;
	  f_b[n+1]=0;
	  int in_a=0,in_b=0;
	  for(int i=1;i<=n;i++)
	  {
	  	if(f_a[i]&1)
	  	 ans_a[i]=1;
	  	else
	  	 ans_a[i]=0;
	  	 
	  	f_a[i+1] += (f_a[i]/2);
	  	if(f_b[i]&1)
	  	 ans_b[i]=1;
	  	else
	  	 ans_b[i]=0;
	  	 
	  	f_b[i+1] += (f_b[i]/2);
	  	//cout<<f_a[i+1]<<" "<<f_b[i+1]<<endl;
	  	int j,k;
	  	if(i==n)
	  	{
	  	//cout<<f_a[i+1]<<" "<<f_b[i+1]<<endl;
	  		 j=i;
	  		while(f_a[j+1]>0)
	  		{
	  			if(f_a[j+1]&1)
	  	 		ans_a[j+1]=1;
	  			else
	  	 		ans_a[j+1]=0;
	  	 		f_a[j+2] += (f_a[j+1]/2);
	  	 		j++;
	  	 		//cout<<"avneet"<<endl;
	  		}
	  		 k=i;
	  		while(f_b[k+1]>0)
	  		{
	  			//cout<<"avneetb"<<endl;
	  			if(f_b[k+1]&1)
	  	 		ans_b[k+1]=1;
	  			else
	  	 		ans_b[k+1]=0;
	  	 		f_b[k+2] += (f_b[k+1]/2);
	  	 		k++;
	  		}
	  	}
	  	in_a=j;in_b=k;
	  }
	  int tr=0;
      if(tr)
      {
      	trace(ans_b,in_a);
      	trace(ans_a,in_b);
      }
	  for(int i=1;i<=in_b;i++)
	  {
	  	ans_b[i]-=ans_a[i];
	  	while(ans_b[i]<0)
	  	{
	  		ans_b[i]+=2;
	  		ans_b[i+1]--;
	  	}
	  	//cout<<ans_b[i]<<" ";
	  }
	  //cout<<endl;
	  //trace(ans_b,in_b);
	  bool flag=false;
	  for(int i=0;i<in_b;i++)
	  {
	  	//cout<<ans_b[in_b-i];
	  	if(ans_b[in_b-i]==0 && !flag)
	  	 {
	  	    //cout<<i<<endl;
		    continue;
		 }
	  	if(ans_b[in_b-i])
	  	 flag=true;
	  	if(flag)
	  	{
	  	 cout<<ans_b[in_b-i];
	    }
	  }
	  cout<<endl;
	  //printf("%s\n",final_ans);
	  
	}
	
}
