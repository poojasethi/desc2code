const int limit = 200005;
char str[limit];
unsigned X;
 
void srand1(unsigned S){

}
 
unsigned nextInteger1(void){
	X = X * 1103515245 + 12345;
	return (X / 65536) % 32768;
}
 
bool generator1(int N, unsigned S){
	srand1(S);
	for(int i=1;i<=N;i++){
		int c = nextInteger1() % 2;
		if(c != (str[i-1]-'0') ) return false;
	}
	return true;
}
 
bool check1(int n){
	for(unsigned s=0;s<=1000000;s++){
		if(generator1(n,s))
			return true;
	}
	return false;
}
