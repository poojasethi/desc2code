unsigned X; // we assume that unsigned is a 32bit integer type

void srand1(unsigned S){
  X = S;
}

unsigned nextInteger1(void){
  X = X * 1103515245 + 12345;
  return (X / 65536) % 32768;
}

int generator1(int N, unsigned S, char str[]){
  srand1(S);
  for(int i=0;i<N;i++){
    if(nextInteger1() % 2){
    	if(str[i]!='1')
    		return 0;
    }
    else{
    	if(str[i]!='0')
    		return 0;
    }
  }
  return 1;
}


unsigned x, y, z, w; // we assume that unsigned is a 32bit integer type

void srand2(unsigned S){
  x = S;
  y = x * S;
  z = y * S;
  w = z * S;
}

unsigned nextInteger2(void){
  unsigned t = x ^ (x << 11);
  x = y; y = z; z = w;
  return w = (w ^ (w >> 19)) ^ (t ^ (t >> 8)); 
}

int generator2(int N, unsigned S, char str[]){
  srand2(S);
  for(int i=0;i<N;i++){
    if(nextInteger2() % 2){
    	if(str[i]!='1')
    		return 0;
    }
    else{
    	if(str[i]!='0')
    		return 0;
    }
  }
  return 1;
}
