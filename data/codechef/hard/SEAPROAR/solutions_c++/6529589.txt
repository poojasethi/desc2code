#include<iostream>
#include<cstdio>
using namespace std;

int generator(char str[],int x)
{
    int seed;
    int flag;
    for(int i = 0;i < 35;++i)
    {
        x = x*1103515245+12345;
        seed = x>>16;
        if((seed&1) == 1 && str[i] == '0' || !(seed&1) && str[i] == '1')
            flag = 1;
        else
        {
             flag = 0;
             break;
        }

    }
    return flag;
}

int main()
{
    int t;
    scanf("%d",&t);
    char str[200001];
    while(t--)
    {
        scanf("%s",str);
        int res = 0;
        for(int i = 0;i <= 1000000;++i)
        {
            res = generator(str,i);
            if(res == 1)
                break;
        }
        if(res == 1)
            printf("LCG\n");
        else
            printf("Xorshift\n");
    }
    return 0;
}
