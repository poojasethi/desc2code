using namespace std;

struct node
{
    node *son[2];
};

typedef node *Node;
Node root;

Node Make_node()
{
    Node P = new node();
    P -> son[0] = P -> son[1] = NULL;
    return P;
}

int n, a[N];

void Update(int x)
{
    Node P = root;
    for(int i=29; i>=0; i--)
    {
        int u = BIT(i, x);
        if(P -> son[u] == NULL)  P -> son[u] = Make_node();
        P = P -> son[u];
    }
}

int Get(int x)
{
    int res = 0;
    Node P = root;
    for(int i=29; i>=0; i--)
    {
        int u = BIT(i, x);
        if(P -> son[1-u] != NULL)
        {
            res |= (1 << i);
            P = P -> son[1-u];
            continue;
        }
        P = P -> son[u];
    }
    return res;
}

int main()
{
    //filein("CHXORR.inp");   fileout("CHXORR.out");
    int T;
    cin >> T;
    while(T--)
    {
        int ans = 0;
        cin >> n;
        for(int i=1; i<=n; i++)
            cin >> a[i];
        root = Make_node();
        for(int i=n-1; i>=1; i--)
        {
            Update(a[i+1]);
            for(int j=1; j<i; j++)
                ans = max(ans, Get(a[j] ^ a[i]));
        }
        cout << ans << endl;
    }
}
