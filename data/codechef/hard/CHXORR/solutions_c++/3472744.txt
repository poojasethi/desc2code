{
	int p=0;

}
int find(int x)
{
	int p=0,res=0;
	for(int i=30;i>=0;i--)
	{
		bool j=x&(1<<i);
		if(c[p][!j])res+=(1<<i),p=c[p][!j];
		else p=c[p][j];
	}
	return res;
}
