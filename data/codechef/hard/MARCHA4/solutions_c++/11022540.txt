#include <bits/stdc++.h>

using namespace std;

int n, k, mod;

// In k chu so dau cua n^n
void dau() {
    long double nlgn=n*log10l(n);
    long double first=pow(10,nlgn-floor(nlgn)+k-1);
    printf("%d",int(first));
}


// Tinh k chu so cuoi cuar n^n
int luythua(int a,int n,int mod) {
    if (n==0) return 1;
    int t=luythua(a,n/2,mod);
    t=((long long)t*t) % mod;
    if (n%2) t=((long long)t*a) % mod;
    return t;
}
void cuoi() {
    mod=1;
    for(int i=1;i<=k;i++) mod*=10;
    char res[15];
    sprintf(res,"%d",luythua(n,n,mod));
    int len=strlen(res);
    for(int i=len+1;i<=k;i++) printf("0");
    printf("%s\n",res);
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    #endif // ONLINE_JUDGE
    int T; scanf("%d",&T);
    while (T--) {
        scanf("%d%d",&n,&k);
        dau();
        printf(" ");
        cuoi();
    }
}
