#include<cstdio>
#include<cstring>
#include<cmath>
 
int n,k,test;
 
inline int Calc(int x,int y,int MOD)
{
	int tmp=x,ans=1;
	for (;y;y>>=1)
	{
		if (y&1) ans=((long long)ans*tmp)%MOD;
		tmp=((long long)tmp*tmp)%MOD;
	}
	return ans;
}
int main()
{
	for (scanf("%d",&test);test--;)
	{
		scanf("%d%d",&n,&k);
		int MOD=1;
		for (int i=1;i<=k;++i) MOD*=10;
		long double t=n*1.0*log10l(n*1.0);
		long long len=(long long)(t)+1;
		int ans=Calc(n,n,MOD);
		long long kk=len-k;
		long double x=pow(10,t-(long double)kk);
		printf("%d ",(int)(x));
		for (MOD=MOD/10;MOD>1 && ans<MOD;MOD=MOD/10)
			printf("0");
		printf("%d\n",ans);
	}
	return 0;
}
