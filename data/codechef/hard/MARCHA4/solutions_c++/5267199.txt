#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>

using namespace std;

#define For(i,a,b) for(int i = a;i <= b; ++i)
typedef long double lb;
typedef long long ll;

int NTest, NN, KK;
lb NlogN;
ll f, nn, tmp, scs;

const ll d[] = {1, 10, 100, 1000, 10000, 100000, 1000000, 10000000, 100000000, 1000000000};
const char zero[11] = "0000000000";
char *ch[11];

int main(void) {

	For (i, 1, 10) ch[i] = (char*)(&(zero[10-i]));

	scanf("%d", &NTest);
	For (itest, 1, NTest) {
		scanf("%d%d", &NN, &KK);
		
		NlogN = lb(NN)*log10l(lb(NN));
		f = ll(powl( lb(10.0), NlogN - ll(NlogN) + KK - 1 ));

		nn = NN % d[KK];
		tmp = 1LL;
		while (NN > 1) {
			if (NN&1) {
				tmp = (tmp * nn) % d[KK];
			}
			nn = (nn * nn) % d[KK];
			NN >>= 1;
		}
		nn = (nn * tmp) % d[KK];

		if (nn > 0) scs = ll(log10l(nn)) + 1;
		else scs = 1;

		if (KK-scs > 0)
			printf("%lld %s%lld\n", f, ch[KK-scs], nn);
		else
			printf("%lld %lld\n", f, nn);

	}
	return 0;
}
