long long int Pow[10]={1,10,100,1000,10000,100000,1000000,10000000,100000000,1000000000};

long long int last_digit(long long int a,long long int b,int k)
{
    if(b==0)
        return 1;
    if(b==1)
    {
       // cout<<"kk"<<a<<endl;
        return a % Pow[k];

    {
        if(b%2!=0)
        {
            //cout<<"hello"<<endl;
            return ((a % Pow[k]) * (last_digit(a,b-1,k) % Pow[k]) % Pow[k]);
        }
        else
        {
            //cout<<"hello2"<<endl;
            return last_digit(( (a % Pow[k]) * (a % Pow[k]) % Pow[k]),b/2,k);
        }
    }
}
