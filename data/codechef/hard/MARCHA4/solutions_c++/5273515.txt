long long least_nums(long long b, long long p, long long mod)
{
    long long y(b),x(1);
    while(p > 0)
    {
        if(p & 1)
            x = (x * y) % mod;
        y = (y * y) % mod;
        p >>= 1;
    }
    return x;
}

void the_rise_and_fall_of_power()
{
    int T,k,iter;
    long long i,n,p,q,s;
    long double t;

    cin >> T;
    for(iter = 0; iter < T; ++iter)
    {
        cin >> n >> k;
        for(i = 0, p = 1; i < k; ++i)
            p *= 10;
        t = n * log10l(double(n));
        q = (long long)(pow(10, t-(long long)t + k-1));
        s = least_nums(n, n, p);
        
        printf("%d ", q);
        for (p = p / 10; p > 1 && s < p; p = p / 10)
            printf("0");
        printf("%d\n", s);
    }
}

int main()
{
    the_rise_and_fall_of_power();
    return 0;
}
