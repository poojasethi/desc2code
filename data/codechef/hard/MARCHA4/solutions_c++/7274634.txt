ll pow_mod(ll a,ll m)
{
    ll res=1,b;
    b=a;
    while(b>0)
    {
        if(b%2==1)
            res=((res%m)*(a%m))%m;
        a=((a%m)*(a%m))%m;
        b=b/2;
    }
    return res;
}
ll first_d(ll a,int k)
{
    ll res;
   long double x;
    x=a*log10((long double)a);
    res=pow(10,x-floor(x)+k-1);
    return res;
}
