#include<stdio.h>
#include<math.h>

using namespace std;
typedef long long int ll;
typedef long double ld;

ll power(ll a, ll b, ll mod){
	ll x = 1;
	ll y = a, p = b;
	while(p > 0){
		if(p&1)
			x = (x * y) % mod;
		y = (y * y) % mod;
		p >>= 1;
	}
	return x;
}

int main(){
	int t, k, digits;
	ll n, mod, last, first;
	scanf("%d",&t);
	while(t--){
		digits = 1;
		mod = 1;
		scanf("%lld %d",&n,&k);
		for(int i=0;i<k;i++)
			mod *= 10;
		last = power(n,n,mod);
		first = last;
		ld z = log10l(n)*n;
		ld m = k - 1 - floor(z);
		if(m < 0.0) 
			first = powl(10.0L, m + z);
		printf("%lld ",first);
		if(last!=0)
			digits = log10(last)+1;
		for(int i=0;i < k-digits;i++)
			printf("0");
		printf("%lld\n",last);
	}
}
