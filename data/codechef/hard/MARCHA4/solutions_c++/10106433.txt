LL firstK(LL n,LL k)
{
   long double intpart;
   //First k digits of n^n = 10^(x-floor(x)+k-1) .... where x = n*log10(n)
      return (LL) floorl( powl(10.0,modfl( ((long double)n * log10l((long double)n)) , &intpart ) + k - 1 ) );
    
}
LL lastK(LL n,LL k)
{
    LL powk = 1 ;
    for(int i=0;i<k;i++)
     powk*=10 ;
    LL ans = 1;
    LL tmp = n%powk ;
    while(n)
    {
        if(n&1)
         ans = (ans*tmp)%powk ;
        tmp =(tmp*tmp)%powk ; 
        n = n>>1 ;
        
    }
    return ans%powk ;
}
