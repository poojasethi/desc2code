#include<iostream>
#include<stdio.h>
#include<cmath>
using namespace std;
typedef long long ll;

ll pow(ll a, ll b, ll mod){

	if(b==0)return 1;
	ll ans = pow(a,b/2,mod);
	ans = ans * ans;
	ans = ans % mod;
	if(b&1) ans = ans * a;
	ans = ans % mod;
	return ans;

}


int main(){


	int t;
	cin>>t;
	while(t--){
		int n,k;
		scanf("%d%d",&n,&k);
		long double flr = floor(n*log10l(n));
		long double x = log10l(n);
		long double N = n;
		long double p = pow(10, N*x - flr + k -1);
		printf("%lld ",(ll)p);

		ll ten = 1;
		for(int i=0;i<k;i++){
			ten = ten * 10;
		}
		ll second = pow(n,n,ten);
		printf("%0*lld\n",k,second);
	}


}
