long long fastpowlow(long long n , long long k)
{
	long long p=1;
	long long a =n ;
	while(n>0)
	{
		if(n%2 == 1)
		{
			p = (long long)p*a % k;
		}
		a = (long long)a*a % k;
		n /=2;
	}
	return p;
}
int length(int n)
{
	int i=0;
	while(n)
	{
		i++;
		n /= 10;
	}
	return i;
}
