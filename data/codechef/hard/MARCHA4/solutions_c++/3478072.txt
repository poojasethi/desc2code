long long int A[11] = {1, 10, 100, 1000, 10000, 100000, 1000000, 10000000, 100000000, 1000000000};
long long int bin(int n,int base,int M)
{
    if (base == 0)
    {
        return 1;
    }
    else
    {
        if (base % 2 == 0)
        {
            long long int x = bin(n , base / 2, M);
            return (x * x) % M;
        }
        else
        {
            return (n * bin(n , base - 1, M)) % M;
        }
    }
}
long long int last(long long int n,int k)
{
    return bin(n, n, A[k]);
}
