#include <bits/stdc++.h>

using namespace std;

void firstkdig(long long n, int k)
{
    long double ans;
    long double y;
    y = n*log10l(n);
    ans = floorl(powl(10.0, y-floor(y)+k-1));
    printf("%0u ", (unsigned int)ans);
}

void lastkdig(long long n, int k)
{
    int i;
    long long mod = 1;
    for(i=1; i<=k; i++){
        mod = (mod*10);
    }
    long long res=1, y=n;
    while(n>0){
        if(n%2){
            res = (res*y)%mod;
        }
        y = (y*y)%mod;
        n/=2;
    }
    printf("%0*lld\n", k, res);
}

int main()
{
    int t, k;
    long long n;
    scanf("%d", &t);
    while(t--){
        scanf("%lld%d", &n, &k);
        firstkdig(n, k);
        lastkdig(n, k);
    }
    return 0;
}
