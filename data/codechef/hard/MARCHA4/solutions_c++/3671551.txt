long int last(long long a, long long b, long long c){
	long long x = 1, y = a;
	while(b > 0){
		if(b%2 == 1)
			x = (x*y)%c;
		y = (y*y)%c;
		b /= 2;
	}
	return x%c;
}

int length(long long int number){
	int len = 0;
	if(number == 0)return 1;
	while(number > 0){
		number /= 10;
		len++;
	}
	return len;
}
