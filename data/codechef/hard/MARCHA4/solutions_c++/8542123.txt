ll PowMod(ll x, ll e, ll mod) {
	if (e == 0)
		return 1 % mod;
	if (e == 1)
		return x % mod;
	ll res = PowMod(x, e >> 1, mod);
	res = (res * res) % mod;
	if (e & 1)
		res = (res * x) % mod;
	return res % mod;
}

ll first(ll n, ll k) {
	long double left = n * log10l(n);
	long double right = powl(10, left - (ll)left);
	ll ans = right * powl(10, k);
	return ans;
}

ll len(ll n) {
	if (n == 0)
		return 1;
	ll cnt = 0;
	while (n) {
		n = n / 10;
		cnt++;
	}
	return cnt;
}
