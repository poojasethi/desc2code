#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
typedef long long ll;
const int m=41;
const int maxn=100010;
ll lcm=5342931457063200ll;
ll tot[m],sum[m],v[m];
pair<int,int> a[maxn];
int n;
ll gcd(ll x,ll y)
{
   if (!y) return x;
   return gcd(y,x%y);
}

int main()
{
    int i,j,cases,x,y,k;
    ll tmp,ans1,ans2,g,t;
    for (scanf("%d",&cases);cases;cases--)
    {
        for (i=0;i<m;i++) tot[i]=0,sum[i]=0,v[i]=0;
        scanf("%d",&n);
        for (i=0;i<n;i++)
        {
            scanf("%d%d",&x,&y);
            a[i]=make_pair(x,y+20);
        }
        sort(a,a+n);
        ans1=0;
        ans2=0;
        for (i=0;i<n;i++)
        {
            for (j=a[i].second+1;j<m;j++)
                v[j-a[i].second]+=1LL*a[i].first*tot[j]-sum[j];
            tot[a[i].second]++;
            sum[a[i].second]+=a[i].first;
        }
        for (i=1;i<m;i++)
        {
            ans1+=v[i]/i;
            ans2+=v[i]%i*(lcm/i);
            ans1+=ans2/lcm;
            ans2%=lcm;
        }
        g=gcd(lcm,ans2);
        if (ans1==0&&ans2==0)
        {
           puts("0");
        }
        else
        if (ans1>0&&ans2>0)
        printf("%lld %lld/%lld\n",ans1,ans2/g,lcm/g);
        else
        if (ans1>0)
        printf("%lld\n",ans1);
        else printf("%lld/%lld\n",ans2/g,lcm/g);
    }
    return 0;
}
