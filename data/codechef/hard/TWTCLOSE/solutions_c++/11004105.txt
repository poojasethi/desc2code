#include<bits/stdc++.h>
using namespace std ;

int main()
{  
 int n,k ;
 cin>>n>>k ;
 int s[1001]={0} ;
 int ans=0 ;
 while(k--)
 {
 	char str[10] ;int ind;
 	cin>>str ;
 	 
 	if(strcmp(str,"CLICK")==0)
 	{
 		 cin>>ind ;
 		 if(s[ind-1]==0)
 		 {
 		 	ans++ ;
 		 	s[ind-1]=1 ;
		  }
		  else{
		  	ans-- ;
		  	s[ind-1]=0 ;
		  }
	 }
	 else{
	 	ans=0 ;
	 	for(int i=0;i<n;i++)
	 	s[i]=0 ;
	 }
	 cout<<ans<<endl ; 
 	
 }
}
