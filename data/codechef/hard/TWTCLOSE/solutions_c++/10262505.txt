#include <bits/stdc++.h>

using namespace std;

int main()
{
   int a[1010]={0},open=0,n,k,x;
   char str[20];
   cin>>n>>k;
   while(k--)
   {
       cin>>str;
       if(strcmp(str,"CLICK")==0)
       {
            cin>>x;
            if(a[x]==0)
            {

                a[x]=1;
                open++;
            }
            else
            {

                a[x]=0;
                open--;
            }
       }
       else
       {
           for(int i=0;i<1010;i++)
           {
               a[i]=0;
           }
           open=0;
       }
       cout<<open<<endl;
   }
   return 0;

}
