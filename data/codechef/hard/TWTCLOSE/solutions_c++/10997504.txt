#include<iostream>
#include<stdio.h>
#include<string.h>

using namespace std;
int main()
{
    int n,k;
    cin>>n>>k;
    cin.ignore();
    bool a[n];
    char s1[10];

    for(int i=0;i<n;i++)
    {
        a[i] = 0;//all close initially
    }

    int count =0;
    for(int i=0;i<k;i++)
    {
        int m;
        cin>>s1;
        if(s1[2] == 'O'){
            count = 0;
            cout<<count<<endl;
            for(int j=0;j<n;j++)
            {
                a[j] = 0;
            }
        }
        else{
                cin>>m;
                    if(a[m-1])
                    {
                        count--;
                        cout<<count<<endl;
                        a[m-1] = 0;
                    }
                    else{
                        count++;
                        cout<<count<<endl;
                        a[m-1] = 1;
                    }
            }
        }



}
