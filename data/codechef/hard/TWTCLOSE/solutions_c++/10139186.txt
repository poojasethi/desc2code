#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() {
    int n,k;
    cin>>n>>k;
    int a[n+1];
    for(int i=0;i<n+1;i++)
    a[i]=0;
    while(k--){
        int c=0;
        char str[100];
        cin>>str;
        int num;
        if(strcmp(str,"CLOSEALL")==0){
            for(int i=0;i<n+1;i++)
            a[i]=0;
        }
        else{
            cin>>num;
            if(a[num]==0)
            a[num]=1;
            else
            a[num]=0;
        }
        for(int i=0;i<n+1;i++){
            if(a[i]==1)
            c++;
        }
        cout<<c<<endl;
        
    }

	return 0;
}
