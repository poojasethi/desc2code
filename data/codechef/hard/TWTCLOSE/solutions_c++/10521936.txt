#include<stdio.h>
#include<string.h>
int main(){
    char str[][9]={"CLICK","CLOSEALL"};
    int n,k;
    scanf("%d %d",&n,&k);
    int tmp[n+2];
    for(int y=0;y<=n;y++){
        tmp[y]=0;
    }
    int open=0;
    for(int i=0;i<k;i++){
        char s[9];
        scanf("%s",&s);
        if(strcmp(s,str[0])==0){
            int x;
            scanf("%d",&x);
            if(tmp[x]){
                tmp[x]=0;
                open--;
            }
            else
            {
                tmp[x]=1;
                open++;
            }
        }
        else if(strcmp(s,str[1])==0){
            if(open)
            for(int y=0;y<=n;y++){
            tmp[y]=0;
            }
            
            open=0;
        }
        printf("%d\n",open);
    }
}
