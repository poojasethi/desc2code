#include<bits/stdc++.h>
using namespace std;
int main(){

    int n,k,i,mcount=0,t;
    string str;
    cin>>n>>k;

    bool arr[1002]={0};

    while(k--){

        cin>>str;

        if(!str.compare("CLICK")){

            cin>>t;
            if(arr[t]){

                arr[t]=0;
                mcount--;
            }

            else{

                arr[t]=1;
                mcount++;
            }


        }
        else{

        memset(arr,0,n+1);
        mcount = 0;


        }


        cout<<mcount<<endl;


    }

}
