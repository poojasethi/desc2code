#include<stdio.h>
#include<string.h>
typedef long long ll;
int main()
{
	ll n,k,i,x;
	ll count=0;
	char at[10];
	ll g;
	ll ar[1003];
	memset(ar,0,sizeof(ar));
	scanf("%lld%lld",&n,&k);
	while(k--)
	{
	scanf("%s",at);
	g=strlen(at);
	if(g==8)
	{
		memset(ar,0,sizeof(ar));
		count=0;
	printf("0\n");
	continue;
}
	else{
		scanf("%lld",&x);
		if(ar[x]==0)
		{
			ar[x]=1;
			count++;
		}
		else
		{
			ar[x]=0;
			count--;
		}
	}
	printf("%lld\n",count);	
	}
	return 0;
}
