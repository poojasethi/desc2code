#include <bits/stdc++.h>
using namespace std;
int state[1001] = {0};

int main()
{   int n,k;
	#ifndef ONLINE_JUDGE
	    freopen("input.txt","r",stdin);
	    // freopen("output.txt","w",stdout);
	  #endif
	cin>>n>>k;
	
	for(int i=0;i<k;i++)
	{
		string a; int s; int count=0;
		cin>>a;
		if(a == "CLICK")
			cin>>s;


		if(a == "CLICK")
			state[s] = abs(state[s]-1);
				

		else
		{
			for(int j=0;j<=n;j++)
				state[j] = 0;
		}

		for(int j=0;j<=n;j++)
			if(state[j])
				count++;

		cout<<count<<endl;	
	}

 	return 0;
}
