#include<bits/stdc++.h>
#define max 1001
using namespace std;
int main()
{
	int n,k,i,j,l,a[max],count=0;
	cin>>n>>k;
	string str;
	for(i=0;i<n;i++)
	a[i]=0;
	while(k--)
	{
		cin>>str;
		count=0;	
		if(str=="CLICK")
		{
			cin>>l;
			a[l-1]=a[l-1]+1;			
		}
		else if (str=="CLOSEALL" )
		{
			for(i=0;i<n;i++)
			a[i]=0;
		}
		for(i=0;i<n;i++)
		{
			count=count+a[i]%2;
			
		}
		cout<<count<<"\n";
	}
}
