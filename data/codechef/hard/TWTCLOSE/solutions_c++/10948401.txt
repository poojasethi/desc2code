#include<iostream>
#include<set>
#include<cstdio>
using namespace std;
int main()
{
	int n,k;
	cin>>n>>k;
	set<int> s;
	string S;
	int A[1001],B[1001]={0};
	string s1="CLOSEALL";
	for(int i=0;i<k;i++)
	{	
		cin>>S;
		if(S.compare(s1)!=0)
		{
		cin>>A[i];
		if(B[A[i]]!=1)
		{
			s.insert(A[i]);
			B[A[i]]=1;
		}
		else
		{
			s.erase(A[i]);
			B[A[i]]=0;
		}
		cout<<s.size()<<endl;
		}
		else
		{
			s.clear();
			for(int j=0;j<1001;j++)
			{
				B[j]=0;
			}
			cout<<"0"<<endl;
		}
	}
}
