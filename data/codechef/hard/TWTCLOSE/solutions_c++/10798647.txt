#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

#define MX 1010

int main()
{
	int n, q, t;
	int open[MX];
	char command[MX];

	scanf("%d %d",&n,&q);

	for(int i=1;i<=n;i++) open[i] = 0;

	while(q--)
	{
		scanf("%s",command);
		if(strcmp(command,"CLICK")==0)
		{
			scanf("%d",&t);
			open[t]^=1;
		}
		else
		{
			for(int i=1;i<=n;i++) open[i] = 0;
		}

		int openCount = 0;
		for(int i=1;i<=n;i++) openCount += open[i];

		printf("%d\n",openCount);
	}

	return 0;
}
