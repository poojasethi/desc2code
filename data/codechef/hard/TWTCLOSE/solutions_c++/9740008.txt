#include<bits/stdc++.h>
using namespace std;
#define Max 1005
int arr[Max];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	int n,k;
	cin>>n>>k;
	int open=0;
	while(k--)
	{
		string s;
		int ind;
		cin>>s;
		if(!(s=="CLOSEALL"))
		cin>>ind;
		if(s=="CLICK")
		{
			if(arr[ind-1]==0)
				arr[ind-1]=1,open++;
			else if(arr[ind-1]==1)
				arr[ind-1]=0,open--;
		}
		else if(s=="CLOSEALL")
		{
			for(int i=0;i<n;i++)
			{
				if(arr[i]==1)
					arr[i]=0,open--;
			}
		}
		cout<<open<<endl;
	}
	return 0;
}
