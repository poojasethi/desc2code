#include<iostream>
#include<stdio.h>
#include<vector>
using namespace std;
int main()
{
    int n,k,x,i;
    cin>>n>>k;
    vector<int>v(n+1,0);
    char a[20];
    while(k--)
    {
        cin>>a;
        int c=0;
        if(a[2]=='I'){cin>>x;
        if(!v[x])
            v[x]=1;
        else
            v[x]=0;}
            else
            {
                for(i=1;i<=n;++i)
                    v[i]=0;
            }
            for(i=0;i<=n;++i)
            {
                if(v[i])
                    ++c;
            }
            cout<<c<<endl;
    }
    return 0;
}
