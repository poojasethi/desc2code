#include<bits/stdc++.h>
using namespace std;
int main(){
	int n, k, i, j, c=0, l, m;
	map<int, int> mp;
	string s;
	cin>>n>>k;
	while(k--){
		cin>>s;
		l=s.length();
		if(l==5){
			cin>>m;
			if(mp[m]){
				mp[m]=0;
				c--;
			}
			else {
				mp[m]=1;
				c++;
			}
		}
		else{
			for(i=1; i<=n; i++)mp[i]=0;
				c=0;
		}
		printf("%d\n", c);
	}
}
