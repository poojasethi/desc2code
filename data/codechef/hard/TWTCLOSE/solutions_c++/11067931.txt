#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,q,k;
    string s;
    cin >> n >> q;
    int h[n];
    memset(h,0,sizeof(h));
    while(q--){
        cin >> s;
        if(s=="CLOSEALL"){
            for(int i = 0; i<n; i++)
                h[i] = 0;
        }
        else{
            cin >> k;
            h[k-1] = !h[k-1];
        }
        int ans = 0;
        for(int i = 0; i<n; i++)
            ans += h[i];
        cout << ans << "\n";
    }

    return 0;
}
