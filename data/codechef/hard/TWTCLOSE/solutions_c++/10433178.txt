#include<iostream>
#include<string>
using namespace std;
int integerpoint(string );
int main()
{
    int n,k;
    cin>>n>>k;
    //cout<<n<<" "<<k<<endl;
    int a[n];
   for(int i=0;i<n;i++)
    a[i]=0;
    int c=0;
    cin.ignore();
    for(int i=0;i<k;i++)
    {
        string s;
        getline(cin,s);
        //<<s[6]<<endl;
        if((int)s[6]>47&&(int)s[6]<58)
        {
            int z=integerpoint(s);
            if(a[z-1]==0)
            {
            a[z-1]=1;
            c++;
            }
            else
            {
            a[z-1]=0;
            c--;
            }
            cout<<c<<endl;
        }
        else
        {
            for(int j=0;j<n;j++)
            a[j]=0;
            c=0;
            cout<<c<<endl;
        }
    }
    return 0;
}
int integerpoint(string s)
{
    int x=0;
    for(int i=6;i<s.size();i++)
    {
        x=x*10+((int)s[i]-48);
    }
    return x;

}
