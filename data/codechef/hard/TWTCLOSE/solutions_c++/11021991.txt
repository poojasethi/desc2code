#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n,k;
    cin >> n >> k;
    vector<bool> arr(n,false);
    int c=0;
    while(k--)
    {
        string s;
        cin >> s;
        if(s=="CLICK")
        {
            int x;
            cin >> x;
            if(arr[x-1]==false)
            {
                c++;
                arr[x-1]=true;
            }
            else
            {
                c--;
                arr[x-1]=false;
            }
        }
        else
        {
            c=0;
            for(int i=0;i<n;i++)
            {
                arr[i]=false;
            }
        }
        cout << c << endl;
    }
    return 0;
}
