#include <iostream>
#include <set>
#include <string>

using namespace std;

const int MAX_N = 1000, MAX_K = 1000;


int main()
{
    std::ios_base::sync_with_stdio(0);
    int n, k;
    string command;
    int nr;
    set<int> s;
    cin >> n >> k;
    while (k--)
    {
        //getline(cin, command);
        //getline(cin, command, ' ');
        cin >> command;
        //cout << "Comm: " << command << endl;
        if (command == "CLOSEALL")
        {
            s.clear();
            cout << "0\n";
        }
        else
        {
            cin >> nr;
            set<int>::iterator it = s.find(nr);
            if (it == s.end())
                s.insert(nr);
            else
                s.erase(it);
            cout << s.size() << "\n";
        }
    }
    return 0;
}
