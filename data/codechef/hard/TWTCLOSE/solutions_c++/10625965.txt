#include <bits/stdc++.h>
using namespace std;
int main()
{
  int n,k;
  cin>>n>>k;
  vector< int > v(n+1,0);
 int cnt  = 0;
  while(k--)
  {
    string s;
    int p;
    cin>>s;

    if(s[2]=='I')
    {
      cin>>p;
      if(v[p]==0)
       { v[p] = 1;
        cnt++;
      }
      else
       {
        v[p] = 0;
        cnt--;
      }
    }
    else
    {
      for (int i = 1; i <=n ; ++i)
      {
        v[i] = 0;
      }
      cnt=0;
    }
    cout<<cnt<<endl;
    

  }
  return 0;
}
