#include<bits/stdc++.h>
using namespace std;

int main() {
	int n,k;
	cin>>n>>k;
	int open[1000];
	for(int i = 0; i < n; i++) open[i] = 0;
	char click[10];
	while(k--){
		cin>>click;
		if(strcmp(click,"CLICK")==0) {
			int x;
			cin>>x;
		        open[x-1] = (open[x-1] == 0 ? 1 : 0);
		}else {
		     for(int i = 0; i < n; i++) open[i] = 0; 
		}
		int count = 0;
	        for(int i = 0; i < n; i++) count = count + open[i];
	        cout<<count<<endl;
	}
	
}
