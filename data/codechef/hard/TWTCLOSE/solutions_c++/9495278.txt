#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int n,k,tweet[1001]={},inp,co=0,compare;
	char input[10];
	cin>>n>>k;
	while(k--)
	{
		cin>>input;
		compare=strcmp(input,"CLOSEALL");
		if(compare==0)
		{
			co=0;
			for(int i=1;i<=n;i++)
			tweet[i]=0;
		}
		else
		{
			cin>>inp;
			if(tweet[inp]==0)
			{
				tweet[inp]=1;
				co++;
			}
			else
			{
				tweet[inp]=0;
				co--;
			}
		}
		cout<<co<<endl;
		
	}
}
