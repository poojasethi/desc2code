#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int open[1007]={0};
	int m,p;
	scanf("%d %d",&m,&p);
	int i;
	int total=0;
	for(i=0;i<p;i++)
	{
		string a;
		int tem;
		cin>>a;
		if(!strcmp("CLICK",&a[0]))
			{
				cin>>tem;
				if(open[tem])
					{
						open[tem]=0;
						total--;
					}
				else
					{
						open[tem]=1;
						total++;
					}
			}
			else
			{
				total=0;
				int j;
				for(j=1;j<=m;j++)
					open[j]=0;
			}
			cout<<total<<endl;
	}
	return 0;
}
