#include <bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("in","r",stdin);
	int n,k;
	cin>>n>>k;
	int hash[n+1];
	string s;
	int x;
	memset(hash,0,sizeof hash);
	while(k--)
	{
		cin>>s;
     if(s=="CLOSEALL")
     {
     	memset(hash,0,sizeof hash);
     	cout<<0<<endl;
     }
     else if(s=="CLICK")
     {
     	int c=0;
     	cin>>x;
     	hash[x] ^= 1; 
     	for(int i=1;i<=n;i++)
     	{
     		if(hash[i]==1)
     			c++;
     	}
     	cout<<c<<endl;
     }
	}
}
