#include <iostream>
#include<stdlib.h>
#include<string.h>
using namespace std;

int main()
{
		int n,k,i;
		cin>>n>>k;
		string str;
		n++;
		int arr[n];
		memset(arr,0,n*sizeof(int));
		int cnt=0;
		for(;k--;)
		{
			cin>>str;
			if(str=="CLOSEALL")
			{	
				cnt=0;
				memset(arr,0,n*sizeof(int));
			}
			else
			{int  val;

				cin>>val;
				if(arr[val]==0)
					cnt++;
				else
					cnt--;
				arr[val]=~arr[val];
			}
			cout<<cnt<<"\n";
		}
    return 0;
}
