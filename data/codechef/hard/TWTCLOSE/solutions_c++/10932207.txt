#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#include<string>
using namespace std;
int main()
{
    int n,k,*a,i,count=0,j;
    scanf("%d %d",&n,&k);
    a=(int*)malloc(sizeof(int)*(n+1));
    for(i=1;i<=n;i++)
        a[i]=0;
    string s;
    int x;
    for(i=1;i<=k;i++)
    {
        cin>>s;
        if(s=="CLICK")
        {
            cin>>x;
            if(a[x]==0)
            {
                a[x]=1;
                count++;
            }
            else
            {
                a[x]=0;
                count--;
            }
        }
        else if(s=="CLOSEALL")
        {
            for(j=1;j<=n;j++)
                a[j]=0;
            count=0;
        }
        cout<<count<<"\n";
    }
    return 0;
}
