#include <bits/stdc++.h>
using namespace std;

int main() {
	int n, k;
	cin >> n >> k;
	vector<int> ar(n+1);
	string s;
	int open = 0;
	for(int i = 0; i < k; i++) {
		cin >> s;
		if(s == "CLICK")
		{
			int x; cin >> x;
			ar[x] = !ar[x];
			if(ar[x]) open++;
			else open--;
		}
		else {
			ar = vector<int>(n+1);
			open = 0;
		}
		cout << open << "\n";
	}
	return 0;
}
