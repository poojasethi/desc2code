#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int n, k;
    cin >> n >> k;
    int t[n+1];
    memset(t, 0, sizeof t);
    int open = 0, m;
    string s;
    while (k--)
    {
        cin >> s; 
        if (s == "CLICK")
        {
            cin >> m;
            t[m] ^= 1;
            if (t[m])
                open++;
            else
                open--;
        }
        else
        {
            memset(t, 0, sizeof t);
            open = 0;
        }
        cout << open << endl;
    }
	return 0;
}
