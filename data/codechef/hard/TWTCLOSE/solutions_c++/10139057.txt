#include<stdio.h>
int main()
{
	int n,k,i,j,ctr=0,num;
	int arr[1005]={0};
	char str[15];

	scanf("%d %d",&n,&k);

	for(int i=0;i<k;i++)
	{
		scanf("%s",str);

		if(str[2]=='I')
		{
			scanf("%d",&num);

			if(arr[num]==0)
			{
				arr[num]=1;
				ctr++;
			}

			else if(arr[num]==1)
			{
				arr[num]=0;
				ctr--;
			}
		}

		else if(str[2]=='O')
		{
			for(int j=1;j<=n;j++)
			{
				arr[j]=0;
			}

			ctr=0;
		}

		printf("%d\n",ctr);
	}
	return 0;
}
