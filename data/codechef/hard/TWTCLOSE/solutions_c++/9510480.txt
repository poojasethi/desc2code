#include<stdio.h>
#include<string.h>
#define S(a,x) scanf("%"#a,&x)  //#a expands to "a" we use this bcoz, arguments are not replaced within quoted strings
#define PS(a,x) printf("%"#a" ",x)  //print with space
#define PN(a,x) printf("%"#a"\n",x)  //print with newline
#define FOR(i,a,b) for( i=a;i<b;i++)
#define FORD(i,a,b) for( i=a;i>=b;i--)
#define REP(i,n) FOR(i,0,n)
#define Max(a, b) ((a>b)?a:b)
#define Min(a, b) ((a>b)?b:a)
#define MAXSIZE 1010
char isOpen[MAXSIZE]; //initially all close 0
int openCount=0;
int main()
{
    int n,q,num;
    char s[15];
    S(d,n);
    S(d,q);
    while(q--) //O(qn) = 10^6
    {
        S(s,s);
        if(!strcmp(s,"CLICK"))
        {
            S(d,num);
            num--; //0 based index
            isOpen[num]=!isOpen[num]; //toggle
            if(isOpen[num])
                openCount++;
            else
                openCount--;
        }
        else //closeall
        {
            memset(isOpen,0,n+1); //O(n)
            openCount=0;
            
        }
        PN(d,openCount);
    }
	return 0;
}
