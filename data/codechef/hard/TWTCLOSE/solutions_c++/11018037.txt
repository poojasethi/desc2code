#include <iostream>
using namespace std;

int main(){
    int n, c;
    cin >> n >> c;
    int a[10000] = {0};
    int count = 0;
    for(int j = 1; j <= c; j++){
        string click;
        int no;
        cin >> click;
        if( click == "CLOSEALL"){
            count = 0;
            for(int i = 1; i<= n; i++)
                a[i] = 0;
        }
        else{
        	cin >> no;
            if( a[no] == 0){
                a[no] = 1;
                count++;
            }
            else{
                a[no] = 0;
                count--;
            }
        }
        cout << count << "\n";
    }

return 0;
}
