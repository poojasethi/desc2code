#include<stdio.h>
#include<stdlib.h>
#include<math.h>
#include<string.h>
#include<assert.h>
#define REP(i,a,b) for(i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)

int is_same(char a[], char b[], int len){
  int i;
  rep(i,len) if(a[i]!=b[i]) return 0;
  return 1;
}

int main(){
  int N, K, X;
  char com[20];
  int i, res;
  int opened[1200], st[1200], st_size;

  scanf("%d%d",&N,&K);
  assert(1<=N && N<=1000 && 1<=K && K<=1000);

  rep(i,N) opened[i] = 0;
  st_size = 0;
  res = 0; /* number of opened tweets */

  while(K--){
    scanf("%s",com);
    if(is_same(com,"CLICK",6)){
      scanf("%d",&X);
      assert(1<=X && X<=N);
      X--;
      st[st_size++] = X; /* store using tweets */
      if(opened[X]==0) opened[X] = 1, res++;
      else             opened[X] = 0, res--;
    } else if(is_same(com,"CLOSEALL",9)){
      rep(i,st_size) opened[st[i]] = 0; /* CLOSE all using tweets */
      st_size = 0;
      res = 0;
    } else {
      assert(0);
    }

    printf("%d\n",res);
  }

  return 0;
}
