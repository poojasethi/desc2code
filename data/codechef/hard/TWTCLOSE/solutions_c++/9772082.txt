#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int getNum(char *buf){
    int i=0,j=0,len=strlen(buf);
    char num[5];
    while(i<len&&buf[i]!=' ')
        i++;

    if(buf[i]!=' ')
        return -1;
    i++;
    while(i<len)
        num[j++]=buf[i++];
    num[j]='\0';
    return atoi(num);

}
int main()
{
    int n,k;
    scanf("%d %d",&n,&k);
    int tweets[n],count=0,opened=1,closed=0;
    for(count=0;count<n;count++)
        tweets[count]=closed;
    count=0;
    char buf[100];
    getchar();
    for(;k>0;k--){
        fgets (buf, 100, stdin);
        int tweetNo=getNum(buf);
        if(tweetNo==-1)
        {
            count=0;
            opened+=2;
            closed=opened-1;
        }
        else{
            tweetNo--;
            if(tweets[tweetNo]==opened){
                count--;
                tweets[tweetNo]=closed;
            }
            else
            {
                count++;
                tweets[tweetNo]=opened;

            }
        }
        printf("%d\n",count);
    }
    return 0;
}
