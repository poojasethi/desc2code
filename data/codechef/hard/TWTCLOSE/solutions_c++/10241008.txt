#include<iostream>
using namespace std;
int main()
{
int i,n,k,x,c=0;
string st;
cin>>n>>k;
int a[n];
 for(i=0;i<n;i++)
a[i]=0;
while(k--)
{
   
cin>>st;
if(st[2]=='I')
    {
cin>>x;
if(a[x-1]==0)
a[x-1]=1;
else
a[x-1]=0;
} 
    else
  {
for(i=0;i<n;i++)
a[i]=0;
    }
    
for(i=0;i<n;i++)
if(a[i]==1)
c++;
cout<<c<<endl;
c=0;
}
return 0;
}
