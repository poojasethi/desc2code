#include <iostream>
#include <string>
#include <set>
#include <sstream>
using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
	string command, inp;
	set <int> tweets;
	set<int>::iterator it;
	int counter = 0;
	int number;
	int n, k;
	cin >> n >> k;//n tweets k clicks
	while (k--)
		{
			cin >> command ;
			// number = str2int(inp.substr(inp.find(" ")));
			// command = inp.substr(0, inp.find(" "));
			if (command == "CLOSEALL")
			{
				tweets.clear();
				counter = 0;
			}
			else 
			{
				cin >> number;
				if ((it = tweets.find(number)) == tweets.end())
				{
					tweets.insert(number);
					counter ++;
				}
				else
				{
					tweets.erase(it);
					counter --;
				}
			}
			// cout << command << number;
			cout << counter <<endl;
		}
}
