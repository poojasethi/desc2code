#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int main()
{
	int N, K, A[1000], i, count=0;
	scanf("%d %d", &N, &K);
	for(i=0;i<N;i++)
		A[i]=0;
	while(K--)
	{
		char s[10];
		scanf ("%s", s);
		if(strcmp(s,"CLOSEALL")==0)
		{
			for(i=0;i<N;i++)
				A[i]=0;
			count=0;
		}
		else
		{
			int x;
			scanf("%d", &x);
			A[x-1]=1-A[x-1];
			if(A[x-1])
				count++;
			else
				count--;
		}
		printf("%d\n", count);
	}
	return 0;
}
