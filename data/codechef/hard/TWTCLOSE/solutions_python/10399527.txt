(n,k) = map(int,raw_input().split())
l = [0 for i in range(n)]
count = 0
for i in range(k):
    s = raw_input()
    if s[5:]=='ALL':
        l = [0 for i in range(n)]
        count = 0
    else:
        num = int(s[6:])
        if l[num-1]==0:
            count=count+1
            l[num-1]=1
        else:
            count = count - 1
            l[num-1]=0
    print count
