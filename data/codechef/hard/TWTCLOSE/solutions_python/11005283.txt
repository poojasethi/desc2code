input = raw_input("").split(" ")
n = int(input[0])
n += 1
k = int(input[1])
tweets = [0]*n
count = 0
j = 1
while j <= k:
    input = raw_input()
    if input != 'CLOSEALL':
        index = int(input.split(" ")[1])
	if tweets[index] == 0:
	    count += 1
	    tweets[index] = 1
	else:
	    count -= 1
            tweets[index] = 0
    else:
        tweets = [0]*n
        count = 0
    j += 1
    print count
