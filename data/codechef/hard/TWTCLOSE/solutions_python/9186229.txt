def main():
	s = raw_input().strip().split()
	M = int(s[0])
	N = int(s[1])
	arr = []
	mx = 0
	c = 0
	for i in xrange(M):
		arr.append(False)
	for j in xrange(N):
		s = raw_input().strip()
		if(s == 'CLOSEALL'):
			for i in xrange(mx):
				arr[i] = False
				c = 0
		else:
			temp = int(s[6:])
			if(temp>mx):
				mx = temp
			if (arr[temp-1] == False):
				arr[temp-1] = True;
				c+=1
			else:
				arr[temp-1] = False;
				c-=1
		print c

	return

main()
