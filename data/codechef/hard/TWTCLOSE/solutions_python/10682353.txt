n,k=map(int,raw_input().split())
counter=0
dic={}
for i in range(1,n+1):
    dic[i]=0

for i in range(k):
    ans=0
    x=list(map(str,raw_input().split()))
    if len(x)==1:
        for j in dic:
            dic[j]=0
    else:
        x[1]=int(x[1])
        if dic[x[1]]==1:
            dic[x[1]]=0
        else:
            dic[x[1]]=1
    for j in dic:
        ans+=dic[j]
    print ans
