n, k = map(int, raw_input().split())

status = [False] * n
sum = 0
for i in xrange(k):
	message = raw_input()
	if message == "CLOSEALL":
		for j in xrange(n):
			status[j] = False
		sum = 0
	else:
		tweetNo = int(message.split()[1]) - 1
		status[tweetNo] = not status[tweetNo]
		if status[tweetNo]:
			sum += 1
		else:
			sum -= 1
	print sum
