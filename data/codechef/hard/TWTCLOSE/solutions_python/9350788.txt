#!/bin/env python2.7
n, k = map(int, raw_input().split(" "))
tweets = [0]*n
counter = 0

for i in range(k):
	s = raw_input()
	
	if s=="CLOSEALL":
		counter = 0
		tweets = [0]*n
	else:
		pos = int(s.split(" ")[1]) - 1
		if tweets[pos]:
			tweets[pos] = 0
			counter -= 1
		else:
			tweets[pos] = 1
			counter += 1
	print counter
