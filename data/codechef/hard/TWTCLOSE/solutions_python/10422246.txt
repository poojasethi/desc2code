import sys

n , t = map(int,sys.stdin.readline().split())
tweets ,closed = [0]*(n+1) ,n
for __ in range(t) :
	s = raw_input().split()
	if len(s) == 1 :
		closed = n
		tweets = [0]*(n+1)
		print 0
	else :
		tweet = int(s[1])
		tweets[tweet] ^= 1
		if tweets[tweet] == 0 :
			closed += 1
		else :
			closed -= 1
		print n-closed
