n,k=raw_input().split(" ")
n=int(n)
k=int(k)
a=[0 for x in range(n)]
t=k
count=0
while t:
    t-=1
    try:
        m=raw_input().split(" ")[1]
        m=int(m)
        if a[m-1]==1:
            a[m-1]=0
        else:
            a[m-1]=1
    except:
        a=[0 for x in range(n)]
        count = 0
    for i in a:
        if i == 1:
           count+=1
    print(count)
    count=0
