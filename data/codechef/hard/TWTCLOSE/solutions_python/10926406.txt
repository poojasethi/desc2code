n,k=map(int,raw_input().split())
N=[0]*n
count=0
for i in range(k):
    y=raw_input()
    if y=='CLOSEALL':
        count=0
        N=[0]*n
    else:
        y=y.split()
        b=int(y[1])
        if N[int(b)-1]==0:
            count+=1
            N[int(b)-1]=1
        else:
            count-=1
            N[int(b)-1]=0
    print count
