tweets_n, clicks_n = map(int, raw_input().split(" "))

tweets = [0]*tweets_n

for i in xrange(clicks_n):
	click = raw_input()
	if click == "CLOSEALL":
		tweets = [0]*tweets_n
	elif click[0:5] == "CLICK":
		tweet_no = int(click[6:]) - 1
		tweets[tweet_no] = (tweets[tweet_no] + 1) % 2
	# printing number of open tweets
	open_tweets = 0
	for j in xrange(tweets_n):
		if tweets[j] == 1:
			open_tweets = open_tweets + 1
	print open_tweets
