import sys
n_tweets, n_clicks = map(int,sys.stdin.readline().strip().split())
tweets = [0 for x in xrange(0,n_tweets)]
input_lines = sys.stdin.readlines()
for line in input_lines:
    if "CLICK" in line:
        number = int(line.strip().split()[1])-1
        if tweets[number] == 0:
            tweets[number] = 1
        else:
            tweets[number] = 0
    else:
        tweets = [0 for x in xrange(0,n_tweets)]
    sum_tweets = sum(tweets)  
    print sum_tweets
