tmp=list(raw_input().split())
n=int(tmp[0])
m=int(tmp[1])

tweet=[]
for i in range(0,n):
	tweet.append(0)
for i in range(0,m):
	tmp=raw_input()
	if tmp[2]=='O':
		for j in range(0,n):
			tweet[j]=0
	else:	
	    
		change=int(tmp[6:len(tmp)])-1
		
		if tweet[change]==1:
			tweet[change]=0
		else:
			tweet[change]=1
	
	count=0
	for j in range(0,n):
		if tweet[j]==1:
			count+=1
	print count
