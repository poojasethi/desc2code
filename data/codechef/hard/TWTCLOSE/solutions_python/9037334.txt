import fileinput


open_tweets = set()
open_count = 0
first = True
for line in fileinput.input():
    if not line:
        continue
        
    if first:
        first = False
        N, K = map(int, line.split())
        continue
    
    line_split = line.split()
    if line_split[0] == "CLOSEALL":
        open_tweets = set()
        open_count = 0
    elif line_split[0] == "CLICK":
        tweet_num = int(line_split[1])
        if tweet_num in open_tweets:
            open_count -= 1
            open_tweets.remove(tweet_num)
        else:
            open_tweets.add(tweet_num)
            open_count += 1
    else:
        raise Exception
        
    print open_count
