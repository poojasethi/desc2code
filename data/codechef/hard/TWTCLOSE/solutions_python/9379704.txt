import sys

class T(object):
    def __init__(self, N):
        self._N = N
        self.close_all()

    def close_all(self):
        self._status = [False] * self._N
        self._num_open = 0
        return 0

    def touch(self, i):
        door_i = self._status[i - 1]
        self._status[i - 1] = not door_i
        self._num_open += -1 if door_i else 1
        return self._num_open


close_all = object()


def get_action(text):
    if text == "CLOSEALL\n":
        return close_all
    else:
        return int(text[5:])


def main():
    doors, clicks = map(int, raw_input().split())
    house = T(doors)
    for line in sys.stdin:
        action = get_action(line)
        if action == close_all:
            print house.close_all()
        else:
            print house.touch(action)

main()
