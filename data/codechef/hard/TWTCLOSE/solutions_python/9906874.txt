[N, K] = [int(x) for x in raw_input().rstrip().split(" ")]

m = [0]*N
s = 0
for _ in range(K):
    op = raw_input().rstrip()
    if op == 'CLOSEALL':
        m = [0]*N
        s = 0
    else:
        [t, i] = op.split(" ")
        i = int(i)
        if m[i-1] == 1:
            s -= 1
            m[i-1] = 0
        else:
            s += 1
            m[i-1] = 1
    print(s)
