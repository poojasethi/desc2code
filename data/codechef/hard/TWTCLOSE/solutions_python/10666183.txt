N, K = map(int, raw_input().split())
tweetState = {i:False for i in range(1, N+1)}
openTweetCnt = 0
for k in range(K):
    action = raw_input()
    if action == "CLOSEALL":
        tweetState = {i: False for i in range(1, N+1)}
        openTweetCnt = 0
    else:
        tweet = int(action.split("CLICK ")[1])
        tweetState[tweet] = not tweetState[tweet]
        openTweetCnt += (1 if tweetState[tweet] else -1)
    print openTweetCnt
