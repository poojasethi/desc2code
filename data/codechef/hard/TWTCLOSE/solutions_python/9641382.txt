
if __name__ == "__main__":
    n, k = map(int, raw_input().split())
    states = [0] * n
    
    for _ in range(k):
        ln = raw_input().split()
        if ln[0] == "CLICK":
            i = int(ln[1])-1
            states[i] = not states[i]
        elif ln[0] == "CLOSEALL":
            states = [0] * n
        print (sum(states))
