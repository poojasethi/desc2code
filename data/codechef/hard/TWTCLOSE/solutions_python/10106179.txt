n,k = [int(p) for p in raw_input().split()]

listq=[]
for i in range(1,int(n)+1):
    listq.append('CLICK '+str(i))
dic={key: 0 for key in listq}

def clk(n,k,x):
    if 'CLICK' in x:
        dic[x] += 1
    elif x =='CLOSEALL':
        for i in range(1,int(n)+1):dic['CLICK '+str(i)] = 0
    for t in dic:
        dic[t]%=2
    return sum(dic.values())

cnt = 0
while cnt<k:
    cnt+=1
    try:
        x=raw_input()
        print clk(n,k,x)
    except:
        break
