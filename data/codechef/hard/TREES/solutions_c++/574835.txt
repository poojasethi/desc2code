void dfs(int u)
	{
	vis[u]=true;
	dp[u][0]=1;
	
	FOR(i,0,adj[u].s()-1)
		{
		int v=adj[u][i];
		if(vis[v]) continue;
		dfs(v);
		
		FOR(i,0,K) temp[i]=dp[u][i];
		
		FOR(j,0,K)
		for(int k=0;j+k+1<=K;k++)
		dp[u][max(j,k+1)]+=temp[j]*dp[v][k];
		}
	}
