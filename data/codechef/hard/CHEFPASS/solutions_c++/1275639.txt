#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 257
#define lim2 1001
#define lim3 10

int i,s,cs,n,m;
bool M[lim];
char A[lim2][lim3],B[lim2][lim3];
int D[lim];

void f(char A[lim2][lim3],int &n)
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%s",A[i]);
}
void dfs(int x)
{
  int i,y=x;

  M[x]=1;
  if(D[x]&1)
    s++;
  for(i=0;i<m;i++)
  {
    if(B[i][0]==x)
      y=B[i][1];
    else if(B[i][1]==x)
      y=B[i][0];
    if(!M[y])
      dfs(y);
  }
}
int main()
{
  scanf("%d",&cs);
  while(cs--)
  {
    f(A,n);
    f(B,m);
    memset(D,0,sizeof D);
    for(i=0;i<n;i++)
    {
      D[A[i][0]]++;
      D[A[i][1]]++;
    }
    memset(M,0,sizeof M);
    for(i=0;i<lim;i++)
      if(!M[i])
      {
        s=0;
        dfs(i);
        if(s&1)
          break;
      }
    printf("%s\n",(i==lim)?"YES":"NO");
  }
  return 0;
}
