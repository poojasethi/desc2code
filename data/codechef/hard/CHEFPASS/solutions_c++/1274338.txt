int toint(char a)
{
    if(a>='a'&&a<='z')
    return 25+(a-'a');
    else
    return (a-'A');
}

void dfs(int i)
{
     v[i]=1;
     
     for(int j=0;j<52;j++)
     {
                      if(g[i][j]==1&&v[j]==0)
                      dfs(j);
     }
}

{

    {
              for(i=0;i<52;i++)

              g[i][j]=0;
