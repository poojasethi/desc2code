#include<cstdio>
#define mod 1000000007
#define mat_mul(c,a,b,n){long long temp[65][65],k;for(long long i=0;i<n;i++)for(long long j=0;j<n;j++)for(k=temp[i][j]=0;k<n;k++)temp[i][j]=(temp[i][j]+a[i][k]*b[k][j])%mod;for(long long i=0;i<n;i++) for(long long j=0;j<n;j++)c[i][j]=temp[i][j];}
long long a[70][70],res[70][70],n,d;
long long state(long long i,long long j)
{
if(i<0||i>d) return 65;
if(j<0||j>d) return 65;
return((d+1)*i+j);
}

{

for(i=0;i<=d;i++)
for(j=0;j<=d;j++)
a[state(i,j)][state(i+1,j)]=a[state(i,j)][state(i-1,j+1)]=a[state(i,j)][state(i,j-1)]=1;
long long d1=(d+1)*(d+1);
for(i=0;i<=d1;i++) res[i][i]=1;


while(n)
        {
        if(n%2) mat_mul(res,res,a,d1)
        mat_mul(a,a,a,d1)
        n/=2;
        }

long long sum=0;
