#define MOD 1000000007
using namespace std;
int n,dp[1<<10][102],shirt[11][102];
long long int rec(int mask,int tid)
{
    if(mask == (1<<n)-1)return dp[mask][tid] = 1;
    if(tid == 101)return 0;
    if(dp[mask][tid] !=-1)return dp[mask][tid];
    long long int ans = 0;
    ans+=rec(mask,tid+1);
    for(int i = 0;i<n;i++)
    {
        if(shirt[i][tid] && !(mask&(1<<i)))
            ans+=rec(mask|1<<i,tid+1);
        if(ans>MOD)ans-=MOD;
    }
    return dp[mask][tid] = ans;
}
