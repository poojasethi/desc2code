using namespace std;
typedef long long int ll;
#define MOD 1000000007
ll arr[15][105]={0},n,dp[105][(1<<10)+5]={0};
ll recc(ll id,ll bit)
{
    if(bit==(1<<n)-1)
    {
        dp[id][bit]=1;
        return dp[id][bit];
    }
    if(id==101)
    {
        dp[id][bit]=0;
        return dp[id][bit];
    }
    if(dp[id][bit]!=-1)
        return dp[id][bit];
    ll i,val=0,a;
    a=recc(id+1,bit);
    val=(val%MOD+a%MOD)%MOD;
    for(i=0;i<n;i++)
    {
        if(arr[i][id]==0)
        continue;
        if((bit&(1<<i))==0)
        {
            a=recc(id+1,(bit|(1<<i)));
            val=(val%MOD+a%MOD)%MOD;
        }
    }
    dp[id][bit]=val;
    return dp[id][bit];
}
int main()
{
    ll tc,i,val,j;
    string str="",temp="";
    cin>>tc;
    while(tc--)
    {
        memset(dp,-1,sizeof(dp));
        memset(arr,0,sizeof(arr));
        cin>>n;
        getline(cin,str);
        for(i=0;i<n;i++)
        {
            getline(cin,str);
            //cout<<str<<endl;
            stringstream ss(str);
            //vector<ll> token;
            while(ss >> temp)
            {
                //cout<<"**** "<<temp<<endl;
                val=0;
                for(j=0;j<temp.size();j++)
                {
                    val=(val*10)+(temp[j]-'0');
                }
               // cout<<val<<endl;
                //token.push_back(val);
                arr[i][val]=1;
            }
        }
        cout<<recc(0,0)<<endl;
    }
    return 0;
}
