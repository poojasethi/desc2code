ll dp[2000][105] = {0};

int masking(int bitmask , int cur,int n)
{
	int x,i;
	ll ans=0;
	if(cur == 101)
	{
		x = (1<<n) - 1;
		if(x == bitmask)
			return 1;
		return 0;
	}
	if(dp[bitmask][cur] != -1)
		return dp[bitmask][cur];

	ans = masking(bitmask,cur+1,n);
	//trace3(bitmask,cur,ans);
	for(i=0;i<v[cur].size();i++)
	{
		x = v[cur][i];
		if( (bitmask & (1<<x)) == 0)
			ans += masking((bitmask|(1<<x)) , cur+1 , n);
		if(ans > MOD)
			ans -= MOD;
	}
	return dp[bitmask][cur] = ans;
}
