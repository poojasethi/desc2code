#include<bits/stdc++.h>
#include<sstream>
using namespace std;
#define mod 1000000007
std::vector<int>arr[105];
int n,t,i,j,k;
string x;
char c;
long long int dp[105][1200];

int convert(string x)
{
    int ans=0;
    for(int i=0; i<x.length(); i++)
        ans=ans*10+(x[i]-'0');
    return ans;
}

int main()
{
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);

        for(i=0; i<=100; i++)
            arr[i].clear();
        
        scanf("%c",&c);
        std::string line;
        
        for(i=0; i<n; i++)
        {
            getline(std::cin, line);

            std::istringstream iss(line);
            while (iss >> x)
            {
                arr[convert(x)].push_back(i);
            }
        }

        memset(dp,0,sizeof(dp));

        for(i=0; i<=100; i++)
            dp[i][0]=1;

        for(i=1; i<=100; i++)
        {
            for(k=1; k<(1<<n); k++)
            {
                dp[i][k]=dp[i-1][k];
                for(j=0; j<arr[i].size(); j++)
                {
                    if(k&(1<<arr[i][j]))
                    {
                        dp[i][k]=(dp[i][k]+ dp[i-1][k^(1<<arr[i][j])])%mod;
                    }
                }
                // printf("\n%d %d %d\n",i,k,dp[i][k]);
            }
        }
        printf("%lld\n",dp[100][(1<<n)-1]);
    }
    return 0;
}
