lli solve(int mask,int cur)
{
	// cout<<" mask" <<mask<<" cur "<<cur<<endl;
	 if(mask==maxi) return 1;
	if(cur>100)
	{
		return 0;
	}
	if(dp[mask][cur]!=-1) return dp[mask][cur];
	else
	{
		 lli ret=0;
		 ret=solve(mask,cur+1);
		 
		 for(int i=0;i<n;i++)
		 {
		 	 if(id[i][cur]!=0 && (!(mask & (1<<i))))
		 	  {
		 	  	ret=(ret+solve(mask |(1<<i),cur+1))%mod;
			   }
		 }
		 dp[mask][cur]=ret;
		 return ret;
	}
}
