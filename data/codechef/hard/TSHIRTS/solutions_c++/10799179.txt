int a[11][105];int dp[1050][105],n;
int rec(int mask,int id) {
  //  TR2(mask,(1<<n));
    if(mask==(1<<n)-1) {
      //  TR1(mask);
        return 1;
    }
    if(id>100)
        return 0;
    if(dp[mask][id]!=-1)
        return dp[mask][id];
    int res=(rec(mask,id+1))%MOD;
    REP(i,0,n) {
        if((!(mask&(1<<i)))&&a[i][id]==1) {
            res=(res+rec(mask|(1<<i),id+1))%MOD;
        }
    }
    return dp[mask][id]=res;
}
