using namespace std;

int debug=1;
const int N=2e5+3;

/* MAIN PROGRAM STARTS HERE */

static LL dp[4000][110];
int n;
VL A[108];

int toint(const string &s) 
	{ 
	stringstream ss; ss << s; int x; ss >> x; 
	return x; 
	}

LL calc(LL x,int k)
	{
	if (x==(1<<n)-1) 
		return 1;
	if (k==101) return 0;
	if (dp[x][k]!=-1) return dp[x][k];
	LL ret=0;
	for (int i=0; i<A[k].size(); i++)
		{
		int p=A[k][i];
		if ((x|(1<<p))!=x)
			{
			ret+=calc(x|(1<<p),k+1);
			ret=ret%MOD;
			}
		}
	ret+=calc(x,k+1);
	ret=ret%MOD;
	return dp[x][k]=ret;
	}

int main()
	{
	int T;
	LL i,j,k,x,y,z,p,q,r;
	cin>>T;
	while (T--)
		{
		memset(dp,-1,sizeof(dp));
        string ss,temp;
        cin >> n;
        getchar();
        for(i=0; i<=100; i++)
            A[i].clear();
        for(i=0; i<n; i++)
        	{
            getline(cin,ss);
            stringstream s (ss);
            while(s>>temp)
                A[(toint(temp))].push_back(i);
        	}
  	
        cout << calc(0,1) << endl;
		}
	
	return 0;
	}
