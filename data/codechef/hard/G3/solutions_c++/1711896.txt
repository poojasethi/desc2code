int main()
{
    int t,n;
    int arr[10002];
    int grundy[10002]; int start;
    scanf("%d",&t); arr[0]=0; int g,pos;
    while(t--)
    {
        scanf("%d",&n);arr[0]=0;
        for(int i=1;i<=n;i++)scanf("%d",&arr[i]);
        n++;
        sort(arr,arr+n);
        for(int i=0;i<n-1;i++){
            grundy[i]=arr[i+1]-arr[i]-1;
        }
        if(n&1)start=1;
        else start=0;
        g=0; pos=-1;
        for(int i=start;i<n-1;i+=2){
            g^=grundy[i];
            //cout<<" .. "<<i<<" "<<grundy[i]<<endl;
        }

        if(g==0){
            printf("Johnny wins\n\n");
            continue;
        }
        printf("Mary wins\nMove ");

        int tmp,diff;
        for(int i=start;i<n-1;i+=2)
        {
            pos=i;
            if(grundy[i]==0)continue;
            tmp=g^grundy[i];
            if(tmp==grundy[i])continue;
            if(tmp>grundy[i])
            {
                //cout<<" i1 "<<i<<endl;
                if(i==0)continue;
                diff=tmp-grundy[i];
                if( (arr[pos]-1-diff)<arr[pos-1] )continue;
                //printf("");
                cout<<arr[pos]<<" to "<<arr[pos]-diff<<endl;
                break;
            }
            else
            {
                //cout<<" i2 "<<i<<endl;
                diff=grundy[i]-tmp;
                cout<<arr[pos+1]<<" to "<<arr[pos+1]-diff<<endl;
                break;
            }
        }


    }

return 0;
}
