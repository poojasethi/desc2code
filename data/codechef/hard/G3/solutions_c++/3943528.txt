#include<stdio.h>
#include<algorithm>
using namespace std;
int t;
	long n,i,move,a[100006],p[1000006],ans;
int main()
{
	
	scanf("%d",&t);
	while(t--)
	{
		scanf("%ld",&n);
		for(i=0;i<n;i++)
		{
			scanf("%ld",&a[i]);
		
		}
		
		sort(a,a+n);
		
		for(i=n-1;i>0;i--)
		{
			p[i]=a[i]-a[i-1]-1;
		}
		
		p[0]=a[0]-1;
		ans=0;
		for(i=n-1;i>-1;i-=2)
		{
			ans^=p[i];
		}
		
		if(ans>0)
		printf("Mary wins\n");
		else
		printf("Johnny wins\n");
		if(ans!=0)
		{
			for(i=1-n%2;i<n;i+=2)
			{
				if((ans^p[i])<p[i])
				{
					move=p[i]-(ans^p[i]);
					printf("Move %ld to %ld\n",a[i],a[i]-move);
					break;
				}
				else if(i>0)
				{
					
					if(p[i-1]>=((ans^p[i])-p[i]))
					{
						printf("Move %ld to %ld\n",a[i-1],a[i-1]-((ans^p[i])-p[i]));break;
					}
				}
			}
			
		}
	}
}
