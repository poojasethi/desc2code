#include	<stdlib.h>
#include	<stdio.h>
#include	<string.h>
#include	<math.h>
#define FF(i,b) for(i=0;i<(b);i++)
#define FOR(i,a,b) for(i=(a);i<=(b);i++)
#define ZERO(x) do{memset(x,0,sizeof(x));}while(0)
#define SWAP(a,b) do{int _tmp=a;a=b;b=_tmp;}while(0)
#define INFI 2000000000
#define INFF 1e100
typedef long long  LL;
int a[10001]={0},n;
int main ( int argc, char *argv[] )
{
	int T,cas;
	scanf("%d",&T);
	FOR(cas,1,T){
		int p=0;
		scanf("%d",&n);
		for(int i=1;i<=n;i++){
			scanf("%d",&a[i]);
			if(!((n^i)&1))p^=a[i]-a[i-1]-1;
		}
		if(!p)printf("Johnny wins\n");
		else{
			printf("Mary wins\n");
			for(int i=(n&1)?1:2;i<=n;i+=2){
				int d=(a[i]-a[i-1]-1)^p;

				if(i!=n&&a[i]-d-1<a[i-1]&&a[i]-d-1>a[i-2]){
					printf("Move %d to %d\n",a[i-1],a[i]-d-1);break;
				}
				if(d<a[i]-a[i-1]-1){printf("Move %d to %d\n",a[i],a[i-1]+d+1);break;}

			}
		printf("\n");
		}
	}
	return EXIT_SUCCESS;
}				/* ----------  end of function main  ---------- */
