#include <stdio.h>
#define S(a) for(i=1;i<=n;i+=a)
#define T(x,y,z) if(x){printf("Move %d to %d\n",y,z);break;}

int fall, n, p, i, d, a[10010], b[10010];

main()
{
for(scanf("%d",&fall); fall--; puts(""))
	{
	p=!scanf("%d",&n);
	S(1)scanf("%d",a+i);
	S(1)b[i]=a[n-i+1]-a[n-i]-1;
	S(2)p^=b[i];
	puts((!p)?"Johnny wins":"Mary wins");
	if(p)for(i=n;i>0;i--)if(i&1)
		{
		T(((d=p^b[i])<b[i]),a[n-i+1],a[n-i]+d+1);
		T((i!=n&&b[i]<d&&d<=b[i]+b[i+1]),a[n-i],a[n-i]-d+b[i]);
		}
	}
return 0;
}
