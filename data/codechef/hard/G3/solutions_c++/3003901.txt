using namespace std;
ll gcd(ll a,ll b) {if(b==0) return a; return gcd(b,a%b);}

ll power(ll b,ll exp,ll m)
 {ll ans=1;
  b%=m;
  while(exp)
   {if(exp&1)
     ans=(ans*b)%m;
    exp>>=1;
	b=(b*b)%m;
   }
  return ans;
 }



#define MXN 10007

int T,n,SG,a[MXN],b[MXN];

int main()
{
	for (scanf("%d",&T);T--;){
		scanf("%d",&n);SG=0;
		for (int i=1;i<=n;++i) scanf("%d",a+i);
		for (int i=1;i<=n;++i) b[i]=a[n-i+1]-a[n-i]-1;
		for (int i=1;i<=n;i+=2) SG^=b[i];
		if (!SG) puts("Johnny wins");
		else{
			puts("Mary wins");
			for (int i=n;i>0;--i) if (i&1){
				int d=SG^b[i];
				if (d<b[i])
                {printf("Move %d to %d\n",a[n-i+1],a[n-i]+d+1);break;}
				if (i!=n && b[i]<d && d<=b[i]+b[i+1])
                 {printf("Move %d to %d\n",a[n-i],a[n-i]-d+b[i]);break;}
			}
		}
		putchar('\n');
	}
	return 0;
}
