#define FOR(i,j,k) for(i = j;i < k;i++)
#define RFOR(i,j,k) for(i = k-1;i >= j;i--)
#define LL long long
#define GET(x) scanf("%d",&x)
#define OUT(x) printf("%d\n",x)
#define SET(x) memset(x,0,sizeof(x))
using namespace std;
int a[10005];
int main()
 { //Dec
   int t,n,gr;
   int p,q,tt,df,i,j;
   GET(t);
   while(t--)
    { GET(n);
      gr = 0;
      FOR(i,1,n+1)
        GET(a[i]);
      a[0] = 0;
      for(i = n;i > 0;i -= 2)
        { //cout<<a[i]-a[i-1]-1<<"  ";
          gr ^= (a[i]-a[i-1]-1);
        }
      j = i+1;
      bool dn = false;
      //cout<<gr<<"\n";
      if(gr)
       { for(;!dn && j < n;j+=2)
          { tt = gr ^ (a[j+1]-a[j]-1);
            p = a[j];
            q = a[j+1];
            df = q-p-1;
            //cout<<tt<<"  "<<df<<"\n";
            if(tt > df)
             { p = p - (tt-df);
               if(p > 0 && p > a[j-1])
                 { printf("Mary wins\nMove %d to %d\n",a[j],p);
                   dn = true;
                   break;
                 }
             }
            else if(tt < df)
             { q = q - (df-tt);
               if(q > a[j]) 
                { printf("Mary wins\nMove %d to %d\n",a[j+1],q);
                  dn = true;
                  break;
                }
             }
          }
       }
      if(!dn)
        printf("Johnny wins\n");
      printf("\n");
    }
 return 0;
 }
