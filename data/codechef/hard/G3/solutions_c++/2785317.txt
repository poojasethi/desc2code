#include <bits/stdc++.h>
using namespace std;

int main() {
  ios_base::sync_with_stdio(0);
  int t; cin>>t;
  while(t-->0) {
    int n; cin>>n;
    vector<int> a(n+1);
    for(int i=1; i<=n; i++) cin>>a[i];
    int c = 0;
    for(int i=(n&1)?1:2; i<=n; i+=2) c^=(a[i]-a[i-1]-1);
    if(!c) cout<<"Johnny wins"<<endl<<endl;
    else {
      for(int i=(n&1)?1:2; i<=n; i+=2) {
	int d = a[i]-a[i-1]-1, k = d - (c^d);
	if(k<0 && i!=1 && a[i-1]+k>a[i-2]) { cout<<"Mary wins\nMove "<<a[i-1]<<" to "<<a[i-1]+k<<endl<<endl; break; } 
	else if(k>0 && a[i]-k>a[i-1]) { cout<<"Mary wins\nMove "<<a[i]<<" to "<<a[i]-k<<endl<<endl; break; }
      }
    }
  }
  return 0;
}
