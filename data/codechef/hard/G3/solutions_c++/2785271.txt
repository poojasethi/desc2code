#include <bits/stdc++.h>
using namespace std;

vector<int> a;

int main() {
  ios_base::sync_with_stdio(0);
  int t;
  cin>>t;
  while(t-->0) {
    int n;
    cin>>n;
    a.resize(n+1);
    a[0] = 0;
    for(int i=1; i<=n; i++) {
      cin>>a[i];
    }
    int c = 0;
    for(int i=(n&1)?1:2; i<=n; i+=2) {
      c^=(a[i]-a[i-1]-1);
      assert(a[i]>a[i-1]);
    }
    bool f= false;
    if(c>0) {
      for(int i=(n&1)?1:2; i<=n && (!f); i+=2) {
	int d = a[i]-a[i-1]-1;
	int k = d-(c^d);
	if(k==0) continue;
	else if(k<0 && i!=1 && a[i-1]+k>a[i-2]){
	  cout<<"Mary wins\nMove "<<a[i-1]<<" to "<<a[i-1]+k<<endl<<endl;
	  f = true;
	}
	else if(k>0 && a[i]-k>a[i-1]){
	  cout<<"Mary wins\nMove "<<a[i]<<" to "<<a[i]-k<<endl<<endl;
	  f = true;
	}
      }
    }
    if(!f) {
      cout<<"Johnny wins"<<endl<<endl;
    }
  }
  return 0;
}
