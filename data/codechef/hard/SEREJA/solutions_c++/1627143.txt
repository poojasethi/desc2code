inline int inp()
{
    int n=0,s=1;
    char p=get();
    if(p=='-')
    s=-1;
    while((p<'0'||p>'9')&&p!=EOF)
    p=get();
    while(p>='0'&&p<='9')
    {
    n = (n<< 3) + (n<< 1) + (p - '0');
    p=get();
    };
    return n*s;
}
int a[100005],m[100005],M[100005];
long long c[100005];
int main()
{
    int i,n,j,k,l,t;
     n=inp();
     for(i=0;i<n;i++)
     a[i]=inp();
     long long ans=1;
     c[0]=1;
     for(i=1;i<n;i++)
     {
         M[i]=max(a[i],a[i-1]);
         m[i]=min(a[i],a[i-1]);
         j=i-1;
         while(j>0&&a[i]>M[j])
         {
             M[j]=a[i];
             j--;
         }
         while(j>0&&a[i]<m[j])
         {
             m[j]=a[i];
             j--;
         }
         for(k=j+1;k<=i;k++)
         c[k]=(c[k-1]*(M[k]-m[k]))%mod;
         ans=(ans*c[i])%mod;
     }
     printf("%lld\n",ans);
    return 0;
}
