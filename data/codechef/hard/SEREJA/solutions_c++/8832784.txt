#define MOD 1000000007

int a[100005];
int mx[100005];
int mn[100005];
int pr[100005]={0};

int main()
{
    int n = fi();
    ll ans=1;

    forall(i,0,n){
        a[i]=fi();
    }
    fill(mx,mx+n,INT_MIN);
    fill(mn,mn+n,INT_MAX);

    pr[0] = 1;

    forall(j,0,n){
        int i,k1=j,k2=j,k;
        while(mx[k1]<a[j] && k1>=0){
            mx[k1]=a[j];
            k1--;
        }

        while(mn[k2]>a[j] && k2>=0){
            mn[k2]=a[j];
            k2--;
        }

        k=min(k1,k2);

        if(k<0){
            k=0;
        }

        for(i=k;i<j;i++){
            pr[i+1]= ((ll)pr[i]*(mx[i]-mn[i])) % MOD;
        }
        ans = (ans*pr[j]) % MOD;
    }

    printf("%lld\n", ans);

    return 0;
}
