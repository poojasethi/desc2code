#define ALL(i,n)    for(i = 0; i < (n); i++)
#define FOR(i,a,b)  for(i = (a); i < (b); i++)
#define SET(p)      memset(p,-1,sizeof(p))
#define CLR(p)      memset(p,0,sizeof(p))
#define S(n)	    scanf("%d",&n)
#define P(n)	    printf("%d\n",n)
#define Sl(n)	    scanf("%lld",&n)
#define Pl(n)	    printf("%lld\n",n)
#define Sf(n)       scanf("%lf",&n)
#define Ss(n)       scanf("%s",n)
#define LL long long
#define ULL unsigned long long
#define pb push_back
#define mod 1000000007
#define inf 2000000000
using namespace std;
LL a[100005],mn[100005],mx[100005],prod[100005],ans=1;

int main()
{
int n,m,i,j,k,k1,k2;
S(n);
for(i=1;i<=n;i++)
    Sl(a[i]);
mn[1]=a[1];
mx[1]=a[1];
prod[0]=1;

for(i=2;i<=n;i++)
{
    k1=i-1;
    while(k1>0 && a[i]>mx[k1])
    {
        mx[k1]=a[i];
        k1--;
    }

    k2=i-1;
    while(k2>0 && a[i]<mn[k2])
    {
        mn[k2]=a[i];
        k2--;
    }

    k=min(k1,k2)+1;

    for(j=k;j<i;j++)
    {
        prod[j]=(prod[j-1]*(mx[j]-mn[j]))%mod;
    }
    mx[i]=a[i];
    mn[i]=a[i];
    ans=(ans*prod[j-1])%mod;
}
Pl(ans);
return 0;
}
