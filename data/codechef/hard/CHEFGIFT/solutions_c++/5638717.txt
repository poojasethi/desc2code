#include <iostream>
#include<cstdio>
#define inf 10000009
using namespace std;
int main()
{
    int t,n,m,d,answer,i,j,l,k;
    int toll[101][101],cost[101][101];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d%d",&d,&n,&m);
        for(i=0;i<m;i++)
        {
            for(j=0;j<n;j++)
                scanf("%d",&toll[i][j]);
        }
        for(i=0;i<m;i++)
        {
            for(j=0;j<m;j++)
                scanf("%d",&cost[i][j]);
        }
        for(k=0;k<m;k++)
        {
        for(i=0;i<m;i++)
        {
            for(j=0;j<m;j++)
                cost[i][j]=min(cost[i][j],cost[i][k]+cost[k][j]);
        }
        }
        answer=inf;
        for(i=1;i<n;i++)
        {
            for(j=0;j<m;j++)
            {   l=toll[j][i],toll[j][i]=inf;
            	for(k=0;k<m;k++)
                    toll[j][i]=min(toll[j][i],toll[k][i-1]+cost[k][j]);
                toll[j][i]+=l;
            }
        }
        for(i=0;i<m;i++)
            answer=min(answer,toll[i][n-1]);
     answer=d-answer;
     if(answer>=0)
     printf("%d\n",answer);
    else
        printf("%d\n",-1);
    }
    return 0;
}
