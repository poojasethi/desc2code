#include<cstdio>
#include<iostream>
#include<vector>
#include<cstring>
#include<algorithm>
#include<set>
#include<cmath>
#include<bitset>
#include<map>
#include<queue>
#define test(t) while(t--)
#define cin(n) scanf("%d",&n)
#define cout(n) printf("%d\n",n)
#define rep(i,a,n) for(i=a;i<n;i++)
#define vi vector<int>
#define vii vector< pair<int,int> >
#define mii map<int,int>
#define pb push_back
#define mp make_pair
#define imax 1000000007
#define ill long long
#define vvi vector<vector<ill> >
using namespace std;
int t,d,n,m,k,toll[101][101],jump[101][101],i,j,dis[101][101];

int main()
{
    cin(t);int ans;
    test(t)
    {
        cin(d);cin(n);cin(m);
        for(i=1;i<=m;i++)
        {
            for(j=1;j<=n;j++)
            {
                cin(toll[i][j]);
            }
        }
        for(i=1;i<=m;i++)
        {
            for(j=1;j<=m;j++)
            {
                cin(jump[i][j]);
            }
        }
        for(k=1;k<=m;k++)
        {
            for(i=1;i<=m;i++)
            {
                for(j=1;j<=m;j++)
                {
                    jump[i][j] = min(jump[i][j],jump[i][k]+jump[k][j]);
                }
            }
        }
        ans=imax;
        memset(dis,0,sizeof(dis));
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=m;j++)
            {
                dis[i][j]=imax;
                for(k=1;k<=m;k++)
                {
                    dis[i][j] = min(dis[i][j],dis[i-1][k]+jump[k][j]+toll[j][i]);
                }
                //cout<<dis[i][j]<<" ";
                if(i==n) ans=min(ans,dis[i][j]);
            }
           // cout<<"\n";
        }
        cout(max(-1,d-ans));
    }

    return 0;
}
