#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

const int INF=1<<29;

int main(){
	int T; scanf("%d",&T);
	while(T--){
		int d,n,m; scanf("%d%d%d",&d,&n,&m);
		int toll[100][100],cost[100][100];
		rep(i,m) rep(j,n) scanf("%d",toll[i]+j);
		rep(i,m) rep(j,m) scanf("%d",cost[i]+j);

		// Warshall-Floyd
		rep(k,m) rep(i,m) rep(j,m) cost[i][j]=min(cost[i][j],cost[i][k]+cost[k][j]);

		int dp[100][100]; // dp[i][j] := 道路 i の料金所 j を過ぎた地点にいるときの最小合計料金
		rep(i,m) rep(j,n) dp[i][j]=(j==0?toll[i][0]:INF);
		rep(j,n-1) rep(i,m) {
			rep(k,m) dp[i][j+1]=min(dp[i][j+1],dp[k][j]+cost[k][i]+toll[i][j+1]);
		}

		int ans=INF;
		rep(i,m) ans=min(ans,dp[i][n-1]);
		ans=d-ans;
		if(ans<0) ans=-1;
		printf("%d\n",ans);
	}

	return 0;
}
