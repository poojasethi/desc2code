#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

#define MX 100010
#define SI ({int x;scanf("%d",&x);x;})

pair<int,int> A[MX];
int cnt[MX], xlim[MX];

int main()
{
	for(int t = SI;t--;)
	{
		int n = SI, k = 0;
		for(int i=0;i<n;i++) A[i].second = SI, A[i].first = SI;
		sort(A,A+n);
		xlim[0] = (int)(1e9+5);
		for(int i=n-1;i>=0;i--)
		{
			int x = A[i].second;
			if(k==0 || x <= xlim[k]){ xlim[++k] = x; cnt[k] = 1; }
			else
			{
				int lo = 0, hi = k;
				while(hi-lo>1)
				{
					int md = (lo+hi)/2;
					if(x>xlim[md]) hi = md;
					else lo = md;
				}
				xlim[hi] = x; cnt[hi]++;
			}
		}
		printf("%d\n",k);
		for(int i=1;i<=k;i++) printf("%d ",cnt[i]);
		puts("");
	}
	return 0;
}
