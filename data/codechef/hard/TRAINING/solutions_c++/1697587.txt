#include<iostream>
#include<vector>
#include<map>
#include<algorithm>
#include<utility>

using namespace std;

vector< pair<int,int> > a(100000);
map <int,vector<int> > mp;
map<int,vector<int> >::iterator it;


bool comp( pair<int,int> r, pair<int,int> s)
{
	if(r.first==s.first)
		return r.second>s.second;
	return r.first< s.first;
}

int main(){
	
	int t,i,j,n,temp,k,p,q;
	cin>>t;
	while(t--)
	{
		cin>>n;
		k=0;
		mp.clear();
		a.clear();
		for(i=0;i<n;i++)
		{
			cin>> p;
			cin>> q;
			a.push_back(make_pair(q,-p));
		}
		sort(a.begin(),a.end(),comp);

		//for(i=0;i<n;i++)
		//{
		//	cout<< a[i].first<<" ";
		//	cout<<a[i].second<<"\n";
		//}
		

		for(i=n-1;i>=0;i--)
		{	//cout<<"kkk  ";
			if(mp.upper_bound(a[i].second)!=mp.end())
			{	//cout<<"hhh  ";
				it=mp.upper_bound(a[i].second);
				int temp=it->second[0];
				//.pop_front();

				 it->second.erase (it->second.begin());
				 if(it->second.size()==0)
				 	mp.erase(it);
				mp[a[i].second].push_back(temp+1);
			}
			else
				mp[a[i].second].push_back(1);
		}
		
		for(it=mp.begin();it!=mp.end();it++)
		{
			k+= it->second.size();
		}
		cout<<k<<"\n";
		for(it=mp.begin();it!=mp.end();it++)
		{
			for(j=0;j<it->second.size();j++)
			cout<< it->second[j]<<" ";
		}
		cout<<"\n";
		
	}
	return 0;
}
