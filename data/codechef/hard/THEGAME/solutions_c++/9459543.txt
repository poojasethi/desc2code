#include <bits/stdc++.h>
#define ll long long
#define MOD 1000000007
#define INF 0x3f3f3f3f
using namespace std;
#define MAX 50005
int di[]={1,0,-1,0};
int dj[]={0,1,0,-1};
string grid[MAX];
int visited[MAX];
int r,c;

int dfs(int i,int j)
{
    visited[i*c+j]=1;
    int ret=1;
    for(int k=0;k<4;k++)
    {
        int ni=i+di[k];
        int nj=j+dj[k];
        if(ni>=0&&ni<r&&nj>=0&&nj<c&&grid[ni][nj]=='o'&&!visited[ni*c+nj])
        {
            ret+=dfs(ni,nj);
        }
    }
    return ret;
}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        scanf("%d%d",&r,&c);
        for(int i=0;i<r;i++)
        {
            cin>>grid[i];
        }
        memset(visited,0,sizeof(visited));
        vector<int> comp;
        for(int i=0;i<r;i++)
        {
            for(int j=0;j<c;j++)
            {
                if(grid[i][j]=='o'&&!visited[i*c+j])
                {
                    comp.push_back(dfs(i,j));
                }
            }
        }
        double ans=1.0;
        for(int i=1;i<comp.size();i++)
        {
            ans+=(1.0*comp[i])/(comp[i]+comp[0]);
        }
        printf("%0.9lf\n",ans);
    }
    return 0;
}
