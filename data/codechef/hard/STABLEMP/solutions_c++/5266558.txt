#include <stdio.h>
#include <iostream>
using namespace std;
int women[501][501], men[501][501], couple[501], single[501], arr[501];
int main()
{
    int t,i,j,temp,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        for(i=1;i<=n;i++)
        {
            couple[i]=-1;
            single[i]=1;
            arr[i]=1;
        }
        for(i=1;i<=n;i++)
        {
            scanf("%d",&temp);
            for(j=1;j<=n;j++)
            {
                scanf("%d",&women[temp][j]);
            }
        }
        int temp2;
        for(i=1;i<=n;i++)
        {            
            scanf("%d",&temp2);
            for(j=1;j<=n;j++)
            {
                scanf("%d",&temp);
                men[temp2][temp]=j;
            }
        }
        int all_married=0;
        while(all_married<n)
        {
            for(i=1;i<=n;i++)
            {
                if(single[i]==0 || arr[i]>n)
                    continue;
                temp=women[i][arr[i]];
                arr[i]++;
                if(men[temp][i]<men[temp][couple[temp]] || couple[temp]<0)
                {
                    if(couple[temp]>0)
                        single[couple[temp]]=1;
                    else
                        all_married++;
                    single[i]=0;
                    couple[temp]=i;
                }
            }
        }
        for(i=1;i<=n;i++)
        {
            printf("%d %d\n",i,couple[i]);
        }
    }
    return 0;
}
