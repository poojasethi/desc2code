#include<iostream>
#include<queue>
#include"stdio.h"

using namespace std;

#define MAX 501

int main()
{
	int a,b;
	cin >> a;											//number of test cases
	while(a--)
	{
		cin >> b;										//number of matches
		int menpref[MAX][MAX],wmenpref[MAX][MAX],wife[MAX],husband[MAX],nextpref[MAX];
		husband[0] = 0; wife[0] = 0; nextpref[0] = 0;						//finishing off the first elements
		int index,pref;
		queue<int> unmarried;
		for (int i = 1; i <= b; ++i)								//adding the preferences of women
		{
			scanf("%d",&index);
			for (int j = 1; j <= b; ++j)
			{
				scanf("%d",&pref);
				wmenpref[index][pref] = j;
			}
			
		}
		for (int i = 1; i <= b; ++i)								//adding the preferences of men
		{
			unmarried.push(i);
			husband[i] = 0;
			wife[i] = 0;
			nextpref[i] = 1;
			scanf("%d",&index);
			for (int j = 1; j <= b; ++j)
			{	
				scanf("%d",&menpref[index][j]);
			}
		
		}
		
	
	
		int presentwoman;
		int presentman;
		
		while(!unmarried.empty())
		{
			presentman = unmarried.front();
			unmarried.pop();
			
			while(1)
			{
				presentwoman = menpref[presentman][nextpref[presentman]];
				if(husband[presentwoman] == 0)									//in case the woman doesn't hav a husband
				{
					husband[presentwoman] = presentman;
					wife[presentman] = presentwoman;
					nextpref[presentman]++;
					break;
				}
				else if(wmenpref[presentwoman][presentman] < wmenpref[presentwoman][husband[presentwoman]])	//if the woman like the present man more than her husband
				{
					wife[husband[presentwoman]] = 0;
					unmarried.push(husband[presentwoman]);
					husband[presentwoman] = presentman;
					wife[presentman] = presentwoman;
					nextpref[presentman]++;
					break;
				}
				else												//if none of the above
				{
					nextpref[presentman]++;
				}
			}
		}
	
	for (int i = 1; i <=b; ++i)
	{
		printf("%d %d\n",i,wife[i]);
	}
	
	}
	
	return 0;
}
