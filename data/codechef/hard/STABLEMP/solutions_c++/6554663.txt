#include<stdio.h>
#include<math.h>
#include<stdlib.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;

int main (){
    int t;
    scanf("%d",&t);
    while (t--){
        int n;
        scanf ("%d",&n);
        int men[n+1][n+1], women[n+1][n+1];
        int propose[n+1],engaged_men[n+1],engaged_women[n+1];
        for (int i=1;i<=n;i++){
            propose[i]=1; engaged_men[i]=0; engaged_women[i]=0;
        }
        int temp;
        for (int i=1;i<=n;i++){
            scanf ("%d",&temp);
            for (int j=1;j<=n;j++){
                scanf ("%d",&temp);
                women[i][temp] = j;
            }
        }
        for (int i=1;i<=n;i++){
            scanf ("%d",&temp);
            for (int j=1;j<=n;j++){
                scanf ("%d",&temp);
                men[i][j] = temp;
            }
        }

        for (int z=0;z<n;z++){
            for (int i=1;i<=n;i++){
                if (engaged_men[i]==0){
                    int j = propose[i];
                    while ((engaged_women[men[i][j]])!=0 && women[men[i][j]][i] > women[men[i][j]][engaged_women[men[i][j]]]) {j++;}
                    if (engaged_women[men[i][j]]==0){
                        propose[i]=j; engaged_men[i]=men[i][j]; engaged_women[men[i][j]] = i;
                    }
                    else {
                        engaged_men[engaged_women[men[i][j]]] = 0;
                        propose[i]=j; engaged_men[i]=men[i][j]; engaged_women[men[i][j]] = i;
                    }
   //                 printf ("%d %d+++\n",i,engaged_men[i]);
                }

            }
        }
        for (int i=1;i<=n;i++){
            printf("%d %d\n",i,engaged_men[i]);
        }

    }

}
