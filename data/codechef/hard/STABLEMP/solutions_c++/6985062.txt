#include <iostream>
#include <stdio.h>
#include <stack>
#include <queue>
using namespace std;
int women[505][505];
int currently_engaged[505];
int main()
{
    int t,i,j,m,n,w,x;
    scanf("%d",&t);
    while(t--)
    {
        queue<int> men[505];
        stack<int> men_not_engaged;
        scanf("%d",&n);
        for(i=0;i<n;i++)
        {
            scanf("%d",&w);
            currently_engaged[w]=0;
            for(j=0;j<n;j++)
            {
                scanf("%d",&x);
                women[w][x]=n-j;
            }
        }
        for(i=0;i<n;i++)
        {
            scanf("%d",&m);
            men_not_engaged.push(m);
            for(j=0;j<n;j++)
            {
                scanf("%d",&x);
                men[m].push(x);
            }
        }
        while(!men_not_engaged.empty())
        {
            m=men_not_engaged.top();
            men_not_engaged.pop();
            for(i=1;i<=n;i++)
            {
                w=men[m].front();
                men[m].pop();
                if(currently_engaged[w]==0)
                {
                    currently_engaged[w]=m;
                    break;
                }
                else if(women[w][currently_engaged[w]]<women[w][m])
                {
                    men_not_engaged.push(currently_engaged[w]);
                    currently_engaged[w]=m;
                    break;
                }
            }
        }
        for(int i=1;i<=n;i++){
			printf("%d %d\n",currently_engaged[i],i);
		}
    }
    return 0;
}
