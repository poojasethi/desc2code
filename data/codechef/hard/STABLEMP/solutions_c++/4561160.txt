#include<iostream>
#include<queue>
using namespace std;
int men[501][501],women[501][501],couples[500];
queue<int> freeMen;
bool find(int A[],int m1,int m2,int len){
	int index1=0,index2=0;
	bool found1=false,found2=false;
	for(int i=1;i<=len;i++){
		if(m1==A[i]){found1=true;index1=i;}
		if(m2==A[i]){index2=i;found2=true;}
		if(found1&&found2)break;
	}
	if(index1<index2)return false;
	else return true;
}
int main(){
//	freopen("input.txt","r",stdin);
//	freopen("output.txt","w",stdout);
	int T,n,temp;
	cin>>T;
	for(int tc=0;tc<T;tc++){
		cin>>n;
		for(int i=1;i<=n;i++){
			cin>>temp;
			women[i][0]=0;
			for(int j=1;j<=n;j++){
				cin>>temp;
				women[i][j]=temp;
			}
		}
		for(int i=1;i<=n;i++){
			cin>>temp;
			freeMen.push(i);
			men[i][0]=1;
			for(int j=1;j<=n;j++){
				cin>>temp;
				men[i][j]=temp;
			}
		}
		while(!freeMen.empty()){
			int m=freeMen.front();
			freeMen.pop();
			int choice=men[m][0];
			int w=men[m][choice];
			if(women[w][0]==0){ //women is free
				couples[m]=w;
				women[w][0]=m;
			}
			else{
				bool isBetter=find(women[w],women[w][0],m,n);
				if(isBetter){
					couples[m]=w;
					freeMen.push(women[w][0]);
					women[w][0]=m;
				}
				else{
					freeMen.push(m);
					men[m][0]++;
				}
			}
		}
		for(int i=1;i<=n;i++){
			cout<<i<<" "<<couples[i]<<endl;
		}
	}
}
