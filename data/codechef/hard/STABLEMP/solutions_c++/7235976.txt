int man[501][501],woman[501][501],freeman[501],freewoman[501];
int main()
{
	int a,t,n,cnt,i,j,b;
	sf(t)
	while(t--)
	{
		mem(man,0);
		mem(woman,0);
		mem(freeman,0);
		mem(freewoman,0);
		sf(n)
		for(i=1;i<=n;i++)
		{
			sf(b)
			for(j=1;j<=n;j++)
			{
				sf(a)
				woman[b][a]=j;				
			}
		}
		for(i=1;i<=n;i++)
		{
			sf(b)
			for(j=1;j<=n;j++)
			{
				sf(man[b][j])
			}
		}
		cnt=0;
		while(cnt<n)
		{
			for(i=1;i<=n;i++)
			{
				if(freeman[i]!=0)
				{
					continue;
				}
				for(j=1;j<=n;j++)
				{
					if(freewoman[man[i][j]]==0)
					{
						freewoman[man[i][j]]=i;
						freeman[i]=man[i][j];
						cnt++;
						break;
					}
					else if(woman[man[i][j]][freewoman[man[i][j]]]>woman[man[i][j]][i])
					{
						freeman[freewoman[man[i][j]]]=0;
						freewoman[man[i][j]]=i;
						freeman[i]=man[i][j];
						break;						
					}
				}
			}
		}
		for(i=1;i<=n;i++)
		{
			printf("%d %d\n",i,freeman[i]);
		}
	}
	return 0;
}
