#include <iostream>
#include <stdio.h>
#include <bitset>
#include <queue>
using namespace std;
int women[505][505];
int balcony[505];
int main()
{
    int t,i,j,n,x,w,m;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        bitset<505> g;
        queue<int> men[505];
        for(i=0;i<n;i++){
            scanf("%d",&w);
            balcony[w]=0;
            for(j=0;j<n;j++)
            {
                scanf("%d",&x);
                women[w][x]=n-j;
            }
        }
        for(i=0;i<n;i++){
            scanf("%d",&m);
            for(j=0;j<n;j++)
            {
                scanf("%d",&x);
                men[m].push(x);
            }
        }
        while(g.count()!=n)
        {
            for(i=1;i<=n;i++)
            {
                w=men[i].front();
                //cout<<"women: "<<w<<endl;
                if(balcony[w]==0){
                    balcony[w]=i;
                    g.set(w);
                    //cout<<"count: "<<g.count()<<endl;
                }
                else if(women[w][balcony[w]]<women[w][i])
                {
                    men[balcony[w]].pop();
                    balcony[w]=i;
                    //cout<<"else\n";
                }
                else if(women[w][balcony[w]]>women[w][i])
                {
                    men[i].pop();
                }
                //cout<<"1\n";
            }
            //cout<<"2\n";
        }
        for(int i=1;i<=n;i++){
			printf("%d %d\n",balcony[i],i);
		}

    }
    return 0;
}
