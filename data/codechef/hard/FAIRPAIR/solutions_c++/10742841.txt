#include<iostream>
using namespace std;
int main()
{
	int n,t,S[1000],T[1000],esc,task[1000];
	cin>>t;
	while(t--)
	{	esc=0;
		cin>>n;
		for(int i=0;i<n;i++)
			cin>>S[i];
		for(int i=0;i<n;i++){
			task[i]=i+1;
			cin>>T[i];
		}
		for(int i=0;i<n;i++)
		{
			if(S[i]==T[i])
			{
				for(int j=0;j<n;j++)
				{
					if(S[i]!=T[j] && T[i]!=S[j])
					{
						swap(T[i],T[j]);
						swap(task[i],task[j]);
						break;
					}
				}
			}
		}
		for(int i=0;i<n;i++)
		{
			if(S[i]==T[i])
				esc++;
		}
		cout<<esc<<endl;
		for(int i=0;i<n;i++)
			cout<<task[i]<<" ";
		cout<<"\n";
	}
	return 0;
}
