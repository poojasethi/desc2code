#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
#define ll long long
#define mod 1000000007
#define bitcount    __builtin_popcountll
#define pb push_back
int a[1000006],b[1000006];
vector<int> q[1000006];
int main()
{
    //freopen("in.txt","r",stdin);
    //freopen("out.txt","w",stdout);
    int t,n,i,j,k,max1,ans;
    for(i=2;i<=1000000;i++)
    {
    	if(a[i]==0)
    	{
    		for(j=i;j<=1000000;j+=i)
    		{
    			a[j]=1;
    			q[j].pb(i);
    		}
    	}
    }
    sd(t);
    while(t--)
    {
    	sd(n);
    	for(i=1;i<=n;i++)
    		sd(a[i]);
    	for(i=0;i<=1000000;i++)
    		b[i]=0;
    	ans=1;
    	for(i=0;i<q[a[1]].size();i++)
    		b[q[a[1]][i]]=1;
    	max1=0;
    	for(i=2;i<=n;i++)
    	{
    		for(j=0;j<q[a[i]].size();j++)
    		{
    			max1=max(max1,b[q[a[i]][j]]);
    			b[q[a[i]][j]]=i;
    		}
    		ans=max(ans,i-max1);
    	}
    	if(ans==1)
    		ans=-1;
    	printf("%d\n",ans);
    }
    return 0;
}
