#include <iostream>
#include <vector>
#include <cstring>

using namespace std;

int v[1000010][9];
int acnt[1000010];
int primes[1000010];
int lat[1000010];
int bl[100000];
int cnt=0;

void factor(){
    primes[2] = 0;
    //v[2].push_back(2);
    for(int i=2;i<1000001;i++){
        if(primes[i] == 0){
            v[i][acnt[i]++] = i;
            bl[cnt++] = i;
            for(int j=i+i;j<1000001;j+=i){
                v[j][acnt[j]++] = i;
                primes[j] = 1;
            }
        }
    }
}

int main()
{
    cin.sync_with_stdio(false);
    cout.sync_with_stdio(false);
    factor();
    int t;
    cin>>t;
    while(t--){
        int n;
        cin>>n;
        int a[n];
        memset(lat,-1,sizeof(lat));
        for(int i=0;i<n;i++){
            cin>>a[i];
        }
        int ans=1,ind1=0,sz=1;
        for(int i=0;i<acnt[a[0]];i++){
            lat[v[a[0]][i]] = 0;
            //cout<<v[a[0]][i]<<" "<<lat[v[a[0]][i]]<<"\n";
        }
        for(int i=1;i<n;i++){
            for(int j=0;j<acnt[a[i]];j++){
                //cout<<ind1<<" "<<v[a[i]][j]<<" "<<lat[v[a[i]][j]]<<"\n";
                if(lat[v[a[i]][j]] >= ind1){
                    //cout<<ind1<<" "<<v[a[i]][j]<<" "<<lat[v[a[i]][j]]<<"\n";
                    ind1 = lat[v[a[i]][j]]+1;
                }
                lat[v[a[i]][j]] = i;
            }
            //cout<<ind1<<"\n";
            sz = i - ind1 + 1;
            if(sz > ans){
                ans = sz;
            }
        }
        if(ans > 1){
            cout<<ans<<"\n";
        }else{
            cout<<"-1"<<"\n";
        }
    }
    return 0;
}
