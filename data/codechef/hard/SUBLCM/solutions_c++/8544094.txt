#include <stdio.h>
#include <string.h>
#include <algorithm>
#define MAX 1000000
#define LL long long
#define getchar getchar_unlocked
using namespace std;

int inline scan()
{
    int N = 0;
    char C;
    C=getchar();
    while (C < '0' || C>'9') C=getchar();
    while (C >= '0' && C <= '9')
    {
        N = (N<<3) + (N<<1) + C - '0';
        C=getchar();
    }

    return N;
}

int dp[MAX];
int primefactor[MAX+1];
int primepos[MAX];
int func(int num,int i)
{
    int maxpos=-1;
    while(num>1)
    {
        int p=primefactor[num];
        maxpos=max(maxpos,primepos[p]);
        primepos[p]=i;
        while(num%p==0) num/=p;
    }
    return maxpos;
}
int main()
{
    for(LL i=2;i<=MAX;++i)
    {
        if(primefactor[i]==0)
        {
            primefactor[i]=i;
            for(LL j=i*i;j<=MAX;j+=i)
            {
                if(primefactor[j]==0) primefactor[j]=i;
            }
        }
    }
    int t,n,num,ans;
    t=scan();
    while(t--)
    {
        memset(primepos,-1,sizeof(primepos));
        ans=0;
        n=scan();
        int p=-1;
        for(int i=0;i<n;++i)
        {
            num=scan();
            p=max(p,func(num,i));
            ans=max(ans,i-p);
        }
        if(ans<2) ans=-1;
        printf("%d\n",ans);
    }
    return 0;
}
