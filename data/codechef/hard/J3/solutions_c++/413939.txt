#include<stdio.h>

void assign(int i,int j,int n,int num);
int solve(int n,int curr);

int board[501][501];
int current[501];

int main()
{
    int testCases,t;
    int n,i,j,k;
    int t1,t2;
    scanf("%d",&testCases);
    for(t=1;t<=testCases;t++)
    {
        scanf("%d%d",&n,&k);
  
        for(i=0;i<k;i++)
        {
            scanf("%d%d",&t1,&t2);
            board[t1][t2] = 2;
        }
        
        solve(n,1); 
        for(i=1;i<=n;i++)
        {
            printf("%d ",current[i]);
        }       
        printf("\n"); 
        
                
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=n;j++)
            {
                board[i][j] = 0;
            }    
        } 
    }   
    return 0;   
}

int solve(int n,int curr)
{
    int i,j,k;
    for(i = 1;i<=n;i++)
    {
        if(board[i][curr]==0)
        {
            assign(i,curr,n,1);
            if(curr==n)
                return 1;
            if(solve(n,curr+1))
                return 1;
            for(j=1;j<=n;j++)
            {
                for(k=1;k<=n;k++)
                {
                    if(board[j][k]==1)
                      board[j][k] = 0;
                }    
            }
            for(j=1;j<curr;j++)
            {
                assign(current[j],j,n,1);
            }        
        }    
    }
    return 0;    
}
  

void assign(int i,int j,int n,int num)
{
    int x,y;
    current[j] = i;
    for(x=i;x<=n;x++)
    {
        if(board[x][j]==2)
            break;
        board[x][j] = num;
    }
    for(x=i;x>=1;x--)
    {
        if(board[x][j]==2)
           break;
        board[x][j] = num;
    } 
    for(x=j;x<=n;x++)
    {
        if(board[i][x]==2)
          break;
        board[i][x] = num;
    }
    for(x=j;x>=1;x--)
    {
        if(board[i][x]==2)
          break;
        board[i][x] = num;
    }
    for(x=i,y=j;x<=n && y <=n;x++,y++)
    {
        if(board[x][y]==2)
           break;
        board[x][y] = num;
    }  
    for(x=i,y=j;x<=n && y>=1;x++,y--)
    {
        if(board[x][y]==2)
           break;
        board[x][y] = num;
    }
    for(x=i,y=j;x>=1 && y <=n;x--,y++)
    {
        if(board[x][y]==2)
           break;
        board[x][y] = num;
    }
    for(x=i,y=j;x>=1 && y>=1;x--,y--)
    {
        if(board[x][y]==2)
           break;
        board[x][y] = num;
    }                 
}
