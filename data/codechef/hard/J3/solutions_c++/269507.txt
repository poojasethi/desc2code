void place_queen( int i , int j  )
{
	
	int ii = i , jj = j,x,y;
	//For that row and column
	if(graph[i][j]==0)
		graph[i][j]=j;
	for(x=i-1;x>=1;x--)
	{
		if(graph[x][j]==-1)
			break;
		if(graph[x][j]==0)
			graph[x][j]=j;
	}
	for(x=i+1;x<=N;x++)
	{
		if(graph[x][j]==-1)
			break;

			graph[x][j]=j;
	}
	for(y=j-1;y>=1;y--)

			graph[i][y]=j;
	}
	for(y=j+1;y<=N;y++)
	{

			break;
		if(graph[i][y]==0)
			graph[i][y]=j;
	}
	x=i+1;

		y++;
	}
	x=i-1;

	{
		if(graph[x][y]==-1)
			break;
		if(graph[x][y]==0)
			graph[x][y]=j;
		x--;

	x=i-1;

}
void remove_queen(int jj)
{

}
int recurse(int j )
{
	if ( j > N )
		return 1;
	/*cout << endl ;

	cout << endl;*/
	for ( int i =1 ; i <=N ; i ++)
	{
		
		if ( graph[i][j] == 0 )
		{
			place_queen(i,j);
			arr[i][j] = 1 ;
			result[j]=i;

			{

			remove_queen(j);

}
