int b[500][500],d[500],n;
bool attack(int r,int c)
{
 int ro,co;
 for(int i=c-1;i>=0;i--)
 if(b[r][i]==-1)break;
 else if(b[r][i]==1)return 1;
 ro=r-1,co=c-1;
 while(ro>=0&&co>=0)
 {
  if(b[ro][co]==-1)break;
  else if(b[ro][co]==1)return 1;
  ro--,co--;
 }
 ro=r+1,co=c-1;
 while(ro<n&&co>=0)
 {
  if(b[ro][co]==-1)break;
  else if(b[ro][co]==1)return 1;
  ro++,co--;
 }
 return 0;
}
int f(int c)
{
 for(int i=0;i<n;i++)
 {
  if(b[i][c]==-1)continue;
  if(!attack(i,c))
  {
   b[i][c]=1;d[c]=i;
   if(c==n-1)return 1;
   if(f(c+1))return 1;
   b[i][c]=0;d[c]=0;
  }
 }
  return 0;
}
