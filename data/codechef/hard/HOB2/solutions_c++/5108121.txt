#include<iostream>
#include<stdio.h>
#define inf 31415926
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int r,n,inc;
        scanf("%d%d",&r,&n);
        int stay[n];
        fill_n(stay,n,inf);
        int room[n],vt[n],curr;
        if( r>=n ){
            for(int i = 0; i<n; i++){
            scanf("%d%d", &vt[i], &inc);
            room[i] = i - inc;
            if( i == r-1 )
                curr = i;
            }
        }
        else{
            for(int i = 0; i<r; i++){
            scanf("%d%d", &vt[i], &inc);
            room[i] = i - inc;
            if( i == r-1 )
                curr = i;
            }
            for(int i = r; i<n; i++){
            scanf("%d%d", &vt[i], &inc);
            if( inc == r ){
                room[i] = 0;
            }
            else{
                room[i] = r-1-inc;
                stay[curr] = vt[i] - vt[curr];
                curr = i;
            }
        }
        }

        for( int i = 0; i<n; i++ ){
            printf("%d %d\n", room[i],stay[i]);
            //cout<<room[i]<<" "<<stay[i]<<"\n";
        }
    }
}
