#include <iostream>
#include <stdio.h>
#include <string>
#include <string.h>
#include <vector>
#include <queue>
#include <map>

#include <cmath>
#include <sstream>
#include <iomanip>
using namespace std;
#define limit 31415926

int stay_time[314159];
int room[314159];
int arr_time[314159];

int main(int argc, char const *argv[])
{
	int tc;
	int n,r;
	scanf("%d",&tc);
	
	while (tc>0){
		scanf("%d%d",&r,&n);
		int times,inconv,last=r-1;
		for (int i = 0; i < n; ++i){
			scanf("%d%d",&times,&inconv);
			arr_time[i]=times;
			if (i<r){
				room[i]=i-inconv;
				stay_time[i]=limit;
			}
			else{
				if (inconv==r){
					room[i]=0;
				}
				else{
					room[i]= r - 1 - inconv;
					stay_time[last]=arr_time[i]-arr_time[last];
					last=i;
				}
				stay_time[i]=limit;
			}
		}

		for (int i = 0; i < n; ++i){
			printf("%d %d\n",room[i],stay_time[i]);
		}
		
		tc--;	
	}
	
	return 0;
}
