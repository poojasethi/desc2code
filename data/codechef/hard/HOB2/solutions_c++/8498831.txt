ll VR[314165],VT[314165],time_[314165];
void pre()
{
	ll i;
	for(i=0;i<314165;i++)
	{
		VT[i]=31415926;
	}
}



int main()
{
	ll t,r,n,a,b,i,prev;
	pre();
	sl(t)
	while(t--)
	{
		sl(r)
		sl(n)
		for(i=1;i<=n;i++)
		{
			sl(a)
			sl(b)
			time_[i]=a;
			VR[i]=i-b-1;
			if(i==r)
				break;
		}
		prev=i;
		for(i=i+1;i<=n;i++)
		{
			sl(a)
			sl(b)
			time_[i]=a;
			if(b==r)
			{
				VR[i]=0;
				continue;
			}
			VR[i]=r-b-1;
			VT[prev]=a-time_[prev];
			prev=i;
		}
		for(i=1;i<=n;i++)
		{
			printf("%lld %lld\n",VR[i],VT[i]);
			VT[i]=31415926;
		}
	}
	return 0;
}
