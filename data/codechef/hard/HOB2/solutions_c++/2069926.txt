#include <stdio.h>
#include <string.h>
#include <math.h>
#include <stdlib.h>
#include <time.h>
#include <vector>
#include <set>
#include <map>
#include <algorithm>
using namespace std;
const int MAXN = 320000;
const int MAXT = 31415926;
int t[MAXN],c[MAXN],q[MAXN],need[MAXN];
int main()
{
    int T;
    scanf("%d",&T);
    while (T--)
    {
        int n,m;
        scanf("%d%d",&n,&m);
        for (int i=0;i<m;i++)
            scanf("%d%d",&t[i],&c[i]);
        for (int i=0;i<n&&i<m;i++)
        {
            q[i]=i-c[i];
            need[i]=MAXT-t[i];
        }
        int pre=n-1;
        for (int i=n;i<m;i++)
        {
            if (c[i]==n)
            {
                q[i]=0;
                need[i]=MAXT-t[i];
            }
            else
            {
                q[i]=n-1-c[i];
                need[pre]=t[i]-t[pre];
                need[i]=MAXT-t[i];
                pre=i;
            }
        }
        for (int i=0;i<m;i++)
            printf("%d %d\n",q[i],need[i]);
    }
    return 0;
}
