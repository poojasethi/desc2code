const int maxN = 314159;
int tim[maxN], room[maxN], stay_time[maxN];

int main() {
    int T;
    scanf("%d", &T);
    for (int t = 0; t < T; ++t) {
        int R, N;
        scanf("%d%d", &R, &N);
       
        int last = R - 1;
        for (int k = 0; k < N; ++k) {
            int inconv;
            scanf("%d%d", &tim[k], &inconv);
           
            room[k] = 0;
            stay_time[k] = 31415926;
            if(inconv < R) {
               
                room[k] = min(k, R - 1) - inconv;
                if (k >= R) {
                 
                    stay_time[last] = tim[k] - tim[last];
                    last = k;
                }
            }
        }
        for (int k = 0; k < N; ++k) {
            printf("%d %d\n", room[k], stay_time[k]);
        }
    }
    return 0;
}
