const int maxN=314159;
int t[maxN],room[maxN],stay_time[maxN];

int main(){
	int test;
	for(scanf("%d",&test);test>0;test--){
		int R,N;
		scanf("%d%d",&R,&N);
		int last=R-1;
		for(int i=0;i<N;i++){
			int inconv;
			scanf("%d%d",&t[i],&inconv);
			room[i]=0;
			stay_time[i]=31415926;
			if(inconv<R){
				room[i]=min(i,R-1)-inconv;
				if(i>=R){ stay_time[last]=t[i]-t[last]; last=i;}
			}
		}
		for(int i=0;i<N;i++) printf("%d %d\n",room[i],stay_time[i]);
	}
	return 0;
}
