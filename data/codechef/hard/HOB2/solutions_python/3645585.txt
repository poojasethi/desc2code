int main()
{
	#ifndef ONLINE_JUDGE
	READ("read.txt");

	int t,r,n,ind,lTime,iConv;
	ios::sync_with_stdio(false);
	cin>>t;
	while(t--){
		cin>>r>>n;
		ind=0;
		for(int i=0;i<n;i++){
			cin>>lTime>>iConv;
			gArrive[i]=lTime;
			if(ind<r){
				gRoom[i]=ind-iConv;
				gLeave[i]=31415926;
				room[ind]=i;
				ind++;
			}
			else{
				if(iConv==r){
					gRoom[i]=0;
					gLeave[i]=31415926;
				}
				else{
					gLeave[room[r-1]]=lTime-gArrive[room[r-1]];
					gRoom[i]=r-1-iConv;
					room[r-1]=i;
					gLeave[i]=31415926;
				}
			}
		}
		for(int i=0;i<n;i++){
			cout<<gRoom[i]<<" "<<gLeave[i]<<endl;
		}
	}
	return 0;
}
