#include<iostream>
using namespace std;
int arr[320000][4];
int rooms[320000][2];
#define MAX 31415926
int main()
{
    int t,r,n,i,k,v;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&r,&n);
        for(i=0;i<n;i++)
        {
           scanf("%d%d",&arr[i][0],&arr[i][3]);
        }
        k=0;
        for(i=0;i<n;i++)
        {
            arr[i][1]=0;
            arr[i][2]=MAX;
            if(arr[i][3]<r)
            {
                v=min(r-1,i)-arr[i][3];
                arr[i][1]=v;
                if(i>=r-1)
                {
                   arr[i][2]=1;
                   k=i;
                }
            }
            else
               arr[k][2]=arr[i][0]-arr[k][0]+1;
        }
        for(i=0;i<n;i++)
          printf("%d %d\n",arr[i][1],arr[i][2]);
    }
    return 0;
}
