     #include<iostream>
    #include<stdio.h>
    #define intin(a) scanf("%d",&a)
    #define LL long long
    using namespace std;
    LL a[100002],p[100002],b[100002],t[100002],psum[100002],expect;
    void sort(int lo,int hi)
    {
    if(lo<hi)
    {
    int m=(lo+hi)/2;
    sort(lo,m);
    sort(m+1,hi);
    //==========merging
    int i,j,k,size;
    i=0;j=lo;psum[m-j+1]=0;
    while(j<=m)
    { b[i]=a[j]; psum[m-j]=p[m-i]+psum[m-j+1];
    t[i++]=p[j++];
    }
    size=i;i=0;k=lo;
    while(k<j&&j<=hi)
    { if(b[i]<=a[j])
    { a[k]=b[i];
    p[k++]=t[i++];
    }
    else
    { a[k]=a[j]; expect+=(psum[i]*p[j]);
    p[k++]=p[j++];
    }
    }
    while(k<j)
    { a[k]=b[i];
    p[k++]=t[i++];
    }
    }
    }
    int main()
    {
    int i,cases,n,d;
    intin(cases);
    while(cases--)
    {
    scanf("%d%d",&n,&d);
    for(i=0;i<2*n;i+=2)
    { scanf("%lld",&a[i]);
    a[i+1]=a[i]+d;
    }
    for(i=0;i<2*n;i+=2)
    { scanf("%lld",&p[i+1]);
    p[i]=100-p[i+1];
    }
    expect=0;
    //for(i=0;i<2*n;i++)cout<<a[i]<<" ";cout<<"\n";
    sort(0,2*n-1);
    //for(i=0;i<2*n;i++)cout<<a[i]<<" ";cout<<"\n";
    printf("%.4f\n",(double)expect/10000);
    }
    }
