#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <climits>
#include <numeric>
#include <vector>
#include <cmath>
using namespace std;
const int N = 100001;
#define ll long long
#define mp make_pair
int s[N],p[N/2],n;pair<int,int>c[N];
void update(int x,int v)
{
 while(x<=n)
 s[x]+=v,x+=x&-x;
}
int read(int x)
{
 int res=0;
 while(x>0)
 {
 	res+=s[x];x-=x&-x;
 }
 return res;
}
int main()
{
	//freopen("in.txt","r",stdin);
	int T,i,j,m,d,a;
	scanf("%d",&T);
	while(T--)
	{
	 scanf("%d%d",&n,&d);
	 memset(s,0,sizeof(s));
	 for(i=0;i<n;i++)
	 {
	  scanf("%d",&a);
	  c[2*i]=mp(a,2*i);c[2*i+1]=mp(a+d,2*i+1);
	 }
	 for(i=0;i<n;i++)scanf("%d",&p[i]);
	 sort(c,c+2*n);
	 double ans=0;
	 for(j=2*n-1;j>=0;j--)
	 {
	  i=c[j].second;
	  if(i&1)
	  {
	   i>>=1;
	   ans+=read(i)*p[i];
	   update(i+1,p[i]);
	  }
	  else
	  {
	   i>>=1;
	   ans+=read(i)*(100-p[i]);
	   update(i+1,100-p[i]);
	  }
	 }
	 printf("%.4f\n",ans/10000.0);
	}
	return 0;
}
