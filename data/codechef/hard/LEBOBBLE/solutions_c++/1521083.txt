#include<cstdio>
int T,N,D,i,X,B[100005],A[100005],a[100005];
long long Ans,P[100005];
void sort(int l,int r)
{
	int mid=(l+r)/2,k,k1,k2;
	long long sum=0;
	if(l==r) return;
	sort(l,mid),sort(mid+1,r);
	k1=l,k2=mid+1;
	for(k=l;k<=r;++k)
	{
		if(k1<=mid&&B[A[k1]]<=B[A[k2]]||k2>r) a[k]=A[k1],Ans+=sum*P[A[k1++]];
		else  a[k]=A[k2],sum+=P[A[k2++]];
	}
	for(k=l;k<=r;++k) A[k]=a[k];
}
int main()
{
	for(scanf("%d",&T);T--;)
	{
		scanf("%d%d",&N,&D);
		for(i=1;i<=N;++i) scanf("%d",&B[2*i-1]),B[2*i]=B[2*i-1]+D;
		for(i=1;i<=N;++i) scanf("%d",&X),P[2*i]=X,P[2*i-1]=100-X;
		Ans=0;
		for(i=1;i<=2*N;++i) A[i]=i;
		sort(1,2*N);
		printf("%.4lf\n",Ans/10000.0);
	}
	return 0;
}
