#include <cstdlib>
#include <iostream>

using namespace std;

int main(int argc, char *argv[])
{
    int tests;
    float result;
    int count;
    
    scanf("%d", &tests);
    
    while (tests--){
    
          scanf("%d", &count);
          
          float x = count;
          result=0;
          for ( int i = 1; i<=count; i++){
          
              result =result + x/i;
          }
          printf("%.1f\n", result);
    }
    
    //system("PAUSE");
    return EXIT_SUCCESS;
}
