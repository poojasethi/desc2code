double dp[SIZE];
void precal(){
    //for(int i = 1;i<SIZE;i++) dp[i][i] = 0.0;
    for(int i = 1;i<SIZE;i++){
          double value = 0.0;
            for(int j = i-1;j>=0;j--){
                    value+=1.0*i/(i-j);
            }
            dp[i]=value*1.0;
    }
}
