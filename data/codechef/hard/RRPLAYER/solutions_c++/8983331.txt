#include<stdio.h>
double d[3001];
int main()
{
    int t,n,i;
    d[1]=1.0;
    double a,b;
    for(i=2;i<=3000;++i)
    {
        a=1.0*(i-1),b=1.0*i;
        d[i]=(b/a)*(d[i-1])+1.0;
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%0.1lf\n",d[n]);
    }
    return 0;
}
