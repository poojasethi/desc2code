#include <stdio.h>
float sum[3333][3333];
int main(void){
	
	int i, j;
	int T, N;

	for(i = 1; i <= 3000; i++){
		sum[i][i] = 0;
		for(j = i - 1; j >= 0; j--)
			sum[i][j] = sum[i][j + 1] + 1.0*i/(i-j);
	}
	scanf("%d", &T);
	for(i = 0; i < T; i++){
		scanf("%d", &N);
		printf("%0.1f\n", sum[N][0]);
	}
	return 0;
}
