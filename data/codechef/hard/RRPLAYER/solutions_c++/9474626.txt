// expectation val for collecting n coupons
double solveCouponCollectorProblem(int n){
	double ans = 0, num = n;
	for(int i=1; i<=n; i++){
		ans += (num/i);
	}
	return ans;
}

double ar[3001];

void precompute(){
	for(int i=1; i<=3000; i++)
		ar[i] = solveCouponCollectorProblem(i);
}

int main(){
	precompute();
	int t,n;
	cin >> t;
	while(t--){
		cin >> n;
		printf("%.6lf\n", ar[n]);
	}
	return 0;
}
