#include <fstream>
#include <iostream>

using namespace std;
void skaityti();
void rasyti();
double t,N[3001];
double skaiciuoti(int j);
int main()
{
    skaityti();
    rasyti();
}
void skaityti()
{

    cin>>t;
    for(int i=0;i<t;i++)
    {
        cin>>N[i];
    }

}
void rasyti()
{
    ofstream fr("rez.txt");
    for(int i=0;i<t;i++)
    {
        cout<<skaiciuoti(i)<<endl;
    }
    fr.close();
}
double skaiciuoti(int j)
{
    double ats,k=2;
    if(N[j]==1)
    {
        ats=1;
    }

    else
    {
        ats=1;
        while(k<=N[j])
    {
            ats+=(1/k);
            k++;
    }
    ats*=N[j];
    }

    return ats;
}
