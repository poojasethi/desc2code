#include<iostream>
#include<cstring>

using namespace std;
#define REP(i,a,b) \
    for(int(i)=int(a);i<=int(b);i++)
#define REPMIN(i,a,b) \
    for(int(i)=int(a);i>=int(b);i--)

double dpTable[3005][30005];

void fillTable(){
    REP(i,1,3004)
        REP(j,1,3004)
            dpTable[i][j] = 0;

    REP(i,1,3004)
        REPMIN(j,i-1,0)
            dpTable[i][j] = dpTable[i][j+1] + 1.0 * i/(i-j);
}

int main(){
    int T,N;
    scanf("%d",&T);
    fillTable();
    while(T--){
        scanf("%d",&N);
        printf("%lf\n",dpTable[N][0]);
    }
    return 0;
}
