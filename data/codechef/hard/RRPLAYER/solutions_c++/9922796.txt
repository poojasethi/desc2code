#include<bits/stdc++.h>
#define ll long long int

using namespace std;
ll gcd(ll m,ll n)
{
	return n == 0 ? m : gcd(n,m % n);
}

int main(){
int t;
cin>>t;
while(t--){
   int  n,i;
   cin>>n;double sum=0.0;
   for(i=n;i>=1;i--){
    sum+=(double)n/(double)i;
   }
   cout<<setprecision(5)<<sum<<endl;
}
return 0;
}
