#include<iostream>
#include<cstdio>
double ar[3005];
int main()
{
    int t,i,n;
    double ans;
    ar[0]=0.0;
    ar[1]=1.0;
    for(i=2;i<=3000;i++)
    {
        ar[i]=ar[i-1]+(1.0/i);
    }
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        ans=ar[n]*n;
        printf("%.1lf\n",ans);
    }
    return 0;
}
