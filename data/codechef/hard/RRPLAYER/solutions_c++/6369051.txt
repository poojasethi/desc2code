#include <iostream>

using namespace std;

int main() {
  int T;
  cin >> T;
  double* res = new double[T];
  
  for (int i = 0 ; i < T ; ++i) {
    int N;
    cin >> N;
    
    res[i] = 0; 
    for (int j = 0 ; j < N ; j++) {
      res[i] += (double)N * (1.0/(double)(j+1));
    }
  }
  
  for (int i = 0 ; i < T ; ++i) {
    cout << res[i] << endl;
  }
}
