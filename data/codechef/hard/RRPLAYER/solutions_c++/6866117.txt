#include <iostream>
#include <cstdio>
#define FOR(i,n) for (int (i) = 0; (i) < (n); (i)++ )
using namespace std;

int main(){ //freopen("in.txt","r",stdin);
	int t; cin >> t; while(t--){
		double n; cin >> n;
		double ans = 0;
		for (int i=1; i <= n; i++){
			ans += (double)1/i;
		}
		printf("%.2f\n",ans*n);
	}
	return 0;
}
