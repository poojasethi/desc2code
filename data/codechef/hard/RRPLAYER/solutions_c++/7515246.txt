#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        int i=1;
        float ans=0;
        while(i<=n)
        {
            ans+=(float)1/i;
            i++;
        }
        ans*=(float)n;
        printf("%.1f\n",ans);
    }
}
