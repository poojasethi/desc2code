#include <bits/stdc++.h>
using namespace std;
double dp[3001][3001];
void pre()
{
	for(int i = 0; i <= 3000; i++ ){
		dp[i][0] = 0;
		for(int j = 1; j <= i; j++){
			dp[i][j] = dp[i][j-1]+ (double)i/(double)j;
		}
	}
}
int main()
{
	pre();
	int test;
	scanf("%d",&test);
	while(test--){
		int n;
		scanf("%d",&n);
		printf("%lf\n",dp[n][n]);
	}
	return 0;
}
