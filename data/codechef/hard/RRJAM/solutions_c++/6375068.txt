#include <bits/stdc++.h>

#define fi first
#define se second
#define o ((b+e)/2)

using namespace std;

typedef long long int Lint;

typedef pair <int,int> ii;
Lint F1[110000],F2[110000];
Lint ar[110000],X[110000],Y[110000];
int N,M,srt[110000];
ii as[110000];
priority_queue <ii , vector<ii> , greater <ii> > q;

int comp( const int &a , const int &b ){ 	return as[a].fi < as[b].fi; }

Lint find1( int i , Lint sum ){
	for( ; i ; i-=(i&-i) ) sum+=F1[i];
	return sum;
}

Lint find2( int i , Lint sum ){
	for( ; i ; i-=(i&-i) ) sum+=F2[i];
	return sum;
}

int main(){
	
	cin >> N;
	
	for( int i=1 ; i<=N ; i++ ) scanf(" %lld",&ar[i]);
	for( int i=1 ; i<=N ; i++ ){
		Lint a; scanf(" %lld",&a);
		ar[i]=a-ar[i];
	}
	
	cin >> M;
	for( int i=1 ; i<=M ; i++ ){
		int a,b; scanf(" %d %d",&a,&b);
		Lint c,d; scanf(" %lld %lld",&c,&d);
		as[i]=ii(a,b); X[i]=c; Y[i]=d; srt[i]=i;
	}
	
	sort( srt+1 , srt+1+M , comp );
	
	for( int i=1, j=1 ; i<=N ; i++ ){
		while( as[srt[j]].fi == i && j<=M ){
			q.push( ii( as[srt[j]].se , srt[j] ) );
			for( int k=srt[j] ; k<=M ; k+=(k&-k) ) F1[k]+=X[srt[j]]-Y[srt[j]]*i;
			for( int k=srt[j] ; k<=M ; k+=(k&-k) ) F2[k]+=Y[srt[j]];
			j++;
		}
		while( !q.empty() && q.top().fi == i-1 ){
			int x=q.top().se;
			for( int k=x ; k<=M ; k+=(k&-k) ) F1[k]-=X[x]-Y[x]*as[x].fi;
			for( int k=x ; k<=M ; k+=(k&-k) ) F2[k]-=Y[x];
			q.pop();
		}
		if( ar[i] <=0 ){ printf("0 "); continue; }
		int b=1,e=M,res=-1;
		while( b <= e ){
			if( find1( o , 0  )+find2( o , 0 )*i >= ar[i] ){ res=o; e=o-1; }
			else b=o+1;
		}
		printf("%d ",res);
	}
	
	return 0;
}
