struct pp
{
	LL X;
	LL Y;
	LL Z;
}AA[100005];

struct qq
{
	LL l,r,x,y;
}qu[100005];
int answ[100005];
bool done[100005];
LL a[100005],b[100005];
int main()
{
	ios_base::sync_with_stdio(false);
	LL n,q,t,m,l,k,ans,i,j,res=0,fl;
	t=1;
	LL r,x,y;
	//si(t);

	memset(answ,-1,sizeof answ);
	cin>>(n);
	Loop(i,n)
		cin>>(a[i]);
	Loop(i,n)
		cin>>(b[i]);
	Loop(i,n)
		a[i]=a[i]-b[i];

	Loop(i,n)
	{
		if(a[i]>=0)
		{
			done[i]=1;
			answ[i]=0;
		}
	}


	cin>>(q);
	int Q=q,ct=0,offset=0;
	loop(q,Q)
	{
		cin>>(l);
		cin>>(r);
		cin>>(x);
		cin>>(y);
		
		qu[ct].l=l;
		qu[ct].r=r;
		qu[ct].x=x;
		qu[ct].y=y;
		ct++;


		AA[l].X+=x;
		AA[r+1].X-=x;
		AA[l].Y+=y;
		AA[r+1].Y-=y;
		AA[l].Z+=y*l;
		AA[r+1].Z-=l*y;

		if(ct*ct>=Q or q == Q-1)
		{
			
			LL summ[3]={};
			Loop(i,n)
			{
				summ[0]+=AA[i].X;
				summ[1]+=AA[i].Y;
				summ[2]+=AA[i].Z;

				LL temp=(a[i]+summ[0]+summ[1]*i-summ[2]);
			//	DB2(i,temp)
				if(temp>=0)
				{
					if(done[i])
						continue;
					done[i]=1;
					loop(j,ct)
					{
						if(i>=qu[j].l and i<= qu[j].r)
						{
							a[i]+=qu[j].x+qu[j].y*(i-qu[j].l);
							if(a[i]>=0)
							{
								answ[i]=offset+1+j;
								break;
							}
						}
					}


				}
				a[i]=temp;
			}
			offset+=ct;
			ct=0;
		       

			Loop(i,n)
				AA[i].X=AA[i].Y=AA[i].Z=0;

		}
	}

	Loop(i,n)
		cout<<(answ[i])<<' ';
	


		return 0;
	}
