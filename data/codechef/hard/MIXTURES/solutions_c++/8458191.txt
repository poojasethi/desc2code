#include<iostream>
#include<climits>
using namespace std;
int main()
{
    int n,i,j,k,l,temp;
    while(cin>>n)
    {
        n++;
        int arr[n],dp[n][n];
        arr[0]=0;
        for(i=1;i<n;i++)
            cin>>arr[i],arr[i]+=arr[i-1];
        for(i=1;i<n;i++)
            dp[i][i]=0;
        for(l=2;l<n;l++)
        {
            for(i=1;i<=n-l+1;i++)
            {
               j=i+l-1;
               dp[i][j]=INT_MAX;
               for(k=i;k<j;k++)
               {
                   temp=dp[i][k]+dp[k+1][j]+(((arr[k]-arr[i-1])%100)*((arr[j]-arr[k])%100));
                   if(dp[i][j]>temp)
                    dp[i][j]=temp;
                    //cout<<dp[i][j]<<" ";
               }
               //cout<<endl;
            }
        }
        cout<<dp[1][n-1]<<endl;
    }
    return 0;
}
