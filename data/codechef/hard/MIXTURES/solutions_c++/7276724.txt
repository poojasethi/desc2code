#include<bits/stdc++.h>
using namespace std;

int T,N,A[102],smoke[102][102],help[102];
bool done;

int color(int a,int b)
{
    if(!done)
    for(int i=0;i<N;i++)
    {
        help[i+1]=A[i]+help[i];
    }
    done=true;

    return (help[b+1]-help[a])%100;
}
int solve(int i, int j)
{
    if(smoke[i][j]!=-1)return smoke[i][j];
    if(i==j)return smoke[i][j]=0;

    if(j==i+1)return smoke[i][j]=A[i]*A[j];

    int imin=INT_MAX,ans=0;;

    for(int k=i;k<j;k++)
    {
        ans=solve(i,k);
        ans+=solve(k+1,j);
        ans+=color(i,k)*color(k+1,j);
        imin=min(ans,imin);
    }

    return smoke[i][j]=imin;
}
int main()
{
        while(cin>>N)
        {
            done=false;
            memset(smoke,-1,sizeof(smoke));
            for(int i=0;i<N;i++)cin>>A[i];
            cout<<solve(0,N-1)<<endl;
        }
}
