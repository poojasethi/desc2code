#include<iostream>
#include<climits>
#include<algorithm>
#include<cstdio>
using namespace std;
int ar[105];
int store[105][105];
int dp[105][105];
void pre(int n)
{
	int i,j;
	for(i=1;i<=n;i++)
	store[i][i]=ar[i];
	for(i=1;i<=n;i++)
	{
		for(j=i+1;j<=n;j++)
		{
			store[i][j]=store[i][j-1]+ar[j];
			store[i][j]%=100;
		}
	}
}
int solve(int n)
{
	int i,j,l,k;
	for(i=1;i<=n;i++)
	{
		dp[i][i]=0;
	}
	for(l=2;l<=n;l++)
	{
		for(i=1;i<=n-l+1;i++)
		{
			j=i+l-1;
			dp[i][j]=INT_MAX;
			for(k=i;k<=j-1;k++)
			{
				dp[i][j]=min(dp[i][j],dp[i][k]+dp[k+1][j]+store[i][k]*store[k+1][j]);
			}
		}
	}
	return dp[1][n];
}
int main()
{
	int i,n;
	while(scanf("%d",&n)==1)
	{
		for(i=1;i<=n;i++)
		scanf("%d",&ar[i]);
		pre(n);
		int ans=solve(n);
		printf("%d\n",ans);
	}
	return 0;
}
