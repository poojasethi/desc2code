#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;
int arr[105],dp[105][105];
int sum(int i,int j){
	int s=0;
	for(;i<=j;i++){
		s+=arr[i];
		s%=100;
	}
	return s;
}
int main() {
	int n,temp;
	while(scanf("%d",&n)!=EOF){
		
		for(int i=0;i<n;i++) scanf("%d",&arr[i]);
		memset(dp,0,sizeof(dp));
		for(int i=0;i<=n;i++) dp[i][i]=0;
		for(int l=2;l<=n;l++){
			for(int i=0;i<n-l+1;i++){
				int j=i+l-1;
				temp=1000000000;
				for(int k=i;k<=j;k++){
					temp=sum(i,k)*sum(k+1,j)+dp[i][k]+dp[k+1][j];
					if(dp[i][j]!=0) 
					dp[i][j]=min(dp[i][j],temp);
					else dp[i][j]=temp;
				}
			}
		}
		printf("%d\n",dp[0][n-1]);
	}
	return 0;
}
