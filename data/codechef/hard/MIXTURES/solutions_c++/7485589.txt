#include<iostream>
#include<cstdio>
using namespace std;
#define inf 1000000



int main()
{
	int n;
	int smoke[100][100],m[100][100],colour[100];
	while(scanf("%d",&n)!=EOF)
	{
		for(int i=0;i<n;i++)
		{
			scanf("%d",&m[i][i]);
			colour[i]=m[i][i];
		}
		int i,j,k,l,q,sum,min;
		for(i=0;i<n;i++)
			smoke[i][i]=0;
		for(i=0;i<n;i++)
		{
			sum=m[i][i];
			for(j=i+1;j<n;j++)
			{
				sum+=colour[j];
				m[i][j]=sum%100;
			}
		}
		for(l=1;l<=n;l++)
		{
			for(i=0;i<n-l+1;i++)
			{
				j=i+l;
				min=inf;
				for(k=i+1;k<=j;k++)
				{
					if((smoke[i][k-1]+smoke[k][j]+m[i][k-1]*m[k][j])<min)
					min=smoke[i][k-1]+smoke[k][j]+m[i][k-1]*m[k][j];
				}
				smoke[i][j]=min;
			}
		}
		printf("%d\n",smoke[0][n-1]);
	}
	return 0;
}
