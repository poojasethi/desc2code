#include<iostream>
#include<cstdio>
#define MAX_VAL 1000000000
using namespace std;

int p[101][101];
int m[101][101];
int Matrix(int n)
{
	int L,i,k,j,q;
    for (L = 2; L <= n; L++)   
    {
        for (i = 0; i <= n-L; i++)
        {
            j = i+L-1;
            m[i][j] = MAX_VAL;
            for (k = i; k < j; k++)
            {
                q = m[i][k] + m[k+1][j] + p[i][k]*p[k+1][j];
                if (q < m[i][j])
                    {
                    	m[i][j] = q;
                    	p[i][j] = (p[i][k]+p[k+1][j])%100;
                    }
            }
        }
    }
 
    return m[0][n-1];
}

int main()
{
	int l,r;
	while(scanf("%d",&l) != EOF)
	{
		for(r = 0; r < l;r++)
		{
			scanf("%d",&p[r][r]);
		}
		cout << Matrix(l)<<endl;
	}
	return 0;
}
