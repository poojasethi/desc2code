#include <bits/stdc++.h>
#define ll long long int
#define si(a) scanf("%d", &a)
#define sl(a) scanf("%lld", &a)
#define sc(a) scanf("%c", &a)
#define pb push_back
#define mp make_pair
#define SET(x, a) memset(x, a, sizeof(x))

using namespace std;
int a[105];
int dp[105][105];

int sum(int x, int y)
{
	int s = 0;
	for (int i = x; i <= y; i++) {
		s += a[i];
		s %= 100;
	}
	return s;
}


int solve(int i, int j)
{
	if (i == j) {
		return 0;
	}
	if (dp[i][j] != -1) {
		return dp[i][j];
	}
	int ans = INT_MAX;
	for (int k = i; k < j; k++) {
		int tmp = (solve(i, k) + solve(k + 1, j) + sum(i, k) * sum(k + 1, j));
		if (tmp < ans) {
			ans = tmp;
		}
	}

	dp[i][j] = ans;

	return dp[i][j];
}

int main()
{
	int n;
	while (cin >> n) {
		SET(dp, -1);
		for (int i = 1; i <= n; i++) {
			si(a[i]);
		}
		printf("%d\n", solve(1, n));
	}
	return 0;
}
