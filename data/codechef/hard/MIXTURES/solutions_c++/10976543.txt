#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll a[101], m[101][101]={}, p[101][101]={};
int main(){
	int n;
//	cin>>t;
	while(cin>>n){
		ll i, j, k, d;
	//	cin>>n;
		memset(m, 0, sizeof m);
		for(i=0; i<n; i++){
			cin>>a[i];
		}
		for(i=0; i<n; i++){
			m[i][i]=0;
			p[i][i]=a[i];
		}
		for(d=1; d<n; d++){
			for(i=0; i<n-d; i++){
				j=i+d;
				m[i][j]=m[i][i]+m[i+1][j]+p[i][i]*p[i+1][j];
				p[i][j]=(p[i][i]+p[i+1][j])%100;
				for(k=i+1; k<j; k++){
						if(m[i][j]>m[i][k]+m[k+1][j]+p[i][k]*p[k+1][j]){
							m[i][j]=m[i][k]+m[k+1][j]+p[i][k]*p[k+1][j];
							p[i][j]=(p[i][k]+p[k+1][j])%100;
						}
				}
			}
		}
		cout<<m[0][n-1]<<endl;
	}
}
