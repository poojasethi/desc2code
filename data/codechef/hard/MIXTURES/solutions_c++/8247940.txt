#include <bits/stdc++.h>
using namespace std;
int a[100];
long long int memo[100][100];
int sum(int i,int j)
{
	int add=0;
	for(;i<=j;i++)
	{
		add+=a[i];
		add%=100;
	}
	return add;
}
long long int rec(int i,int j)
{
	if(i==j)
	return (memo[i][j]=0);
	if(memo[i][j]!=-1)
	return memo[i][j];
	long long int count;
	int mini=memo[i][j];
	for(int k=i;k<j;k++)
	{
		count=rec(i,k)+rec(k+1,j)+sum(i,k)*sum(k+1,j);
		if((count<mini)||(mini==-1))
		mini=count;
	}
	return (memo[i][j]=mini);
}
int main() 
{
	int n,i,j;
	long long int ans;
	while((scanf("%d",&n))==1)
	{
		for(i=0;i<n;i++)
		scanf("%d",&a[i]);
		memset(memo,-1,sizeof(memo));
		ans=rec(0,n-1);
		cout<<ans<<endl;
	}
	return 0;
}
