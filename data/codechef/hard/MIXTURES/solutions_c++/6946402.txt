int dp[105][105],sum[105][105],a[105];

int solve(int x,int y)
{
    if(x==y) return 0;
    if(y==x+1) return a[x]*a[y];
    int &ret = dp[x][y],temp;
    if(ret==-1)
    {
        for(int i = x ; i+1 <= y ;i++)
        {
            temp = sum[x][i]*sum[i+1][y] + solve(x,i) + solve(i+1,y);
            if(ret==-1 || temp < ret) ret = temp;
        }
    }
    return ret;
}

    {
        for(int i=0;i<n;i++) gi(a[i]);
        for(int i=0;i<n;i++) sum[i][i] = a[i];
        for(int i=0;i<n;i++)
        {
            for(int j=i+1;j<n;j++)
            {
                sum[i][j] = sum[i][j-1]+a[j];
                if(sum[i][j]>=100) sum[i][j]-=100;
            }
        }
        SET(dp,-1);
        pi(solve(0,n-1));

    }

     return 0;
 }
