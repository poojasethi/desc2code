#include<bits/stdc++.h>
using namespace std;
int main()
{

    while(1)
    {

        int n;
        cin>>n;
        if(!cin.fail())
        {
        int i;
        int A[n+1];
        for(i=1;i<=n;i++)
            cin>>A[i];

       long long int s[n+1][n+1];
       long long int B[n+1][n+1];
       int j,k;
       for(i=1;i<=n;i++)
       {
        s[i][i]=0;
       B[i][i]=A[i];
       }
       int h;
       long long int mini=1000000000;
       for(i=2;i<=n;i++)
       {
           for(j=1;j<=n-i+1;j++)
           {

             h=j+i-1;
             long long int mini=1000000000;
             for(k=j;k<h;k++)
             {
                 long long int r=s[j][k]+s[k+1][h]+B[j][k]*B[k+1][h];
                 if(r<mini)
                    mini=r;
             }
             s[j][h]=mini;
             int y;
             B[j][h]=0;
             for(y=j;y<=h;y++)
                B[j][h]=(A[y]%100+B[j][h])%100;

           }
       }
    cout<<s[1][n]<<endl;
    }
    else if(cin.eof())
        return 0;
    else
    {
        string u;
        cin>>u;
    }
}
}
