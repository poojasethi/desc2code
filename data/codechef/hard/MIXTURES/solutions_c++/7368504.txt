#include <cstdio>

int sum_mod_100(int * array, int a, int b)
{
	int suma = 0;
	for (int i = a; i <= b; i++)
		suma += array[i];
	return suma % 100;
}


int main()
{
	int N, i, j, k;
	long long min, smoke;
	int mixtures[100];
	long long B[100][100];
	for (i = 0; i < 100; i++)
		B[i][i] = 0;
	while (scanf("%d", &N) > 0)
	{
		for (i = 0; i < N; i++)
			scanf("%d", &mixtures[i]);
		for (i = 1; i < N; i++)
		{
			for (j = 0; j < N - i; j++)
			{
				min = 10000000;
				for (k = 0; k < i; k++)
				{
					smoke = B[j][j + k] + B[j + k + 1][j + i] + sum_mod_100(mixtures, j, j + k) * sum_mod_100(mixtures, j + k + 1, j + i);
					if (smoke < min)
						min = smoke;
				}
				B[j][j + i] = min;
			}
		}
		printf("%d\n", B[0][N - 1]);
	}
	return 0;
}
