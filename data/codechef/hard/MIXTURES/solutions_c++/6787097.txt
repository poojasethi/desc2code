#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
#include <cstring>
#define MAX 100
#define MOD 100

using namespace std;

int dp[MAX][MAX];
int dpa[MAX][MAX];

int A[MAX];

int solve(int i,int j){
	if(i == j){
		dp[i][j] = 0;
		dpa[i][j] = A[i]; 
		return 0;
	}
	
	if(dp[i][j] != -1) return dp[i][j];
	
	dp[i][j] = (1 << 20);
	for(int k = i;k < j;k++){
		int a = solve(i,k) + solve(k + 1,j);
		if(dpa[i][k]*dpa[k + 1][j] + a < dp[i][j]){
			dp[i][j] = dpa[i][k]*dpa[k + 1][j] + a;
			dpa[i][j] = (dpa[i][k] + dpa[k + 1][j]) % MOD;
		}
	}
	return dp[i][j];
}


int main(){
	int n;
	while(scanf("%d",&n) != EOF){
		for(int i = 0;i < n;i++) scanf("%d",&A[i]);
		memset(dp,-1,sizeof dp);
		printf("%d\n",solve(0,n - 1));
	}
	return 0;
}
