#define N 200010
using namespace std;
int main()
{
    int n;
    while(scanf("%d",&n)!=EOF)
    {
        ll dp[101][101],a[101][101],b[101];
        for(int i=1;i<=n;i++)
            cin>>b[i];
        for(int i=1;i<=n;i++)
        {
            dp[i][i]=0;
            a[i][i]=b[i];
        }
        for(int l=2;l<=n;l++)
        {
            for(int i=1;i<=n-l+1;i++)
            {
                int j=i+l-1;
                dp[i][j]=INT_MAX;
                for(int k=i;k<j;k++)
                {
                    ll q=dp[i][k]+dp[k+1][j]+a[i][k]*a[k+1][j];
                    if(dp[i][j]>q)
                    {
                        dp[i][j]=q;
                        a[i][j]=(a[i][k]+a[k+1][j])%100;
                    }
                }
            }
        }
        cout<<dp[1][n]<<endl;
    }
    return 0;
}
