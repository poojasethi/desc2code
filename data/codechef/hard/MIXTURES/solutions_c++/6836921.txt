#include <bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define sz size()
#define all(a)  a.begin(),a.end() 
#define SL(n) scanf("%lld",&n)
#define PL(n) printf("%lld",n)
#define fill(a, x) memset(a,x,sizeof(a));
#define mod 1000000007

using namespace std;
typedef long long LL;
typedef vector <LL> VL;
typedef map <LL, LL> ML;
typedef pair<LL, LL> PL;
typedef vector <pair <LL, LL> > VPL;

VL In;
LL dp[105][105];

LL Sum(LL a, LL b){
    LL sum=0;
    for(LL i=a;i<=b;++i){
        sum+=In[i];
    }
    sum%=100;
    return sum;
}

LL solve(LL a, LL b){
    if(dp[a][b]!=-1){
        return dp[a][b];
    }
    if(a==b){
        return 0;
    }
    for(LL i=a; i<b;++i){
        LL tans=solve(a, i) + solve(i+1, b) + Sum(a, i)*Sum(i+1, b);
        if(dp[a][b] == -1 || tans < dp[a][b]){
            dp[a][b]=tans;
        }
    }
    return dp[a][b];
}

int main(){
    LL N;
    while(scanf("%lld", &N)!=EOF){
        In.clear();
        In.resize(0);
        for(LL n=0;n<N;++n){
            LL x;
            SL(x);
            In.pb(x);
        }
        fill(dp, (LL)-1);
        cout << solve(0, N-1) << endl;
    }
    return 0;
}
