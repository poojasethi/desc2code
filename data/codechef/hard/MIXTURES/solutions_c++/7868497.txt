#include <iostream>
#include <stdio.h>
#include <string.h>

using namespace std;

int main()
{
    int t, n, i, j, k, len, a[102], sum[102][102];
    int prod[102][102];
    while(scanf("%d",&n)>0){
        memset(sum, 0, sizeof(sum));
        memset(prod, 1000000, sizeof(prod));
        for(i=0; i<n; i++){
            scanf("%d",&a[i]);
            sum[i][i]=a[i];
            prod[i][i]=0;
        }
        for(len=2; len<=n; len++){
            for(i=0; i<=n-len; i++){
                j=i+len-1;
                {
                    for(k=i; k<j; k++){
                        prod[i][j]=min(prod[i][j], sum[i][k]*sum[k+1][j]+prod[i][k]+prod[k+1][j]);
                    }
                    sum[i][j]=(sum[i][i]+sum[i+1][j])%100;
                }
            }
        }
        printf("%d\n",prod[0][n-1]);
    }
    return 0;
}
