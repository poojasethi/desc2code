#include <iostream>
#include <string>
#include <climits>
using namespace std;

void debug(int arr[102][102], int n)
{
	for(int i = 1; i <= n; i++) {
		for(int j = 1; j <= n; j++)
			cout << arr[i][j] << " ";
		cout << endl;
	}
	cout << endl;
}

int main(int argc, char const *argv[])
{
	int t, n;
	int smoke[102][102], color[102][102];
	
	while(cin >> n) {
		for(int i = 1; i <= n; i++) {
			cin >> color[i][i];
			smoke[i][i] = 0;
		}

		for(int l = 2; l <= n; l++) {
			for(int i = 1; i <= n - l + 1; i++) {
				int j = i + l - 1;
				smoke[i][j] = INT_MAX;
				for(int k = i; k <= j - 1; k++) {
					int q = smoke[i][k] + smoke[k + 1][j] + color[i][k] * color[k + 1][j];
					if(q < smoke[i][j]) {
						smoke[i][j] = q;
						color[i][j] = (color[i][k] + color[k + 1][j]) % 100;
					}
				}
			}
		}
		// debug(smoke, n);
		// debug(color, n);
		cout << smoke[1][n] << endl;
	}
	return 0;
}
