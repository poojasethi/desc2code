#include<iostream>
#include<cstdio>
using namespace std;
int max1=1000000000;
int main()
{
	int n;
	while(scanf("%d",&n)!=EOF)
	{
		int color[n][n];
		for(int i=0;i<n;i++)
			cin>>color[i][i];
    int smoke[n][n];
    for(int i=0;i<n;i++)
    {
      for(int j=0;j<n;j++)
      {
        smoke[i][j]=0;
      }
    }
		for(int len=2;len<=n;len++)
    {
      for(int i=0;i<n-len+1;i++)
      {
        int j=i+len-1;
        int min1=max1;
        for(int k=i;k+1<=j;k++)
        {
          int temp=smoke[i][k]+smoke[k+1][j]+(color[i][k]*color[k+1][j]);
          if(temp<min1)
          {
            min1=temp;
            smoke[i][j]=temp;
            color[i][j]=(color[i][k]+color[k+1][j])%100;
          }
        }
      }
    }
    printf("%d\n",smoke[0][n-1]);
	}
}
