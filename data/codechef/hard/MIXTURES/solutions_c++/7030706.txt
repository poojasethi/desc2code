#include<iostream>
#include<stdio.h>
#include<cstring>
using namespace std;

int main()
{
    int T,N;
    while(scanf("%d",&N)>0)
    {
        int Ar[105];
        for(int i=0;i<N;i++)
            cin>>Ar[i];
        long long result[105][105][2];
        memset(result,0,sizeof(result));
        for(int i=0;i<N;i++)
        {
            result[i][i][0]=0;
            result[i][i][1]=Ar[i];
        }


     //As matrix chain multiplication
        for(int l=1;l<N;l++)
        {
            for(int i=0;i<N-l;i++)
            {
                long long temp=1000000000;
                for(int k=0;k<l;k++)
                {
                    temp=min(temp,result[i][i+k][0]+result[i+k+1][i+l][0]+result[i][i+k][1]*result[i+k+1][i+l][1]);
                }
                result[i][i+l][0]=temp;
                result[i][i+l][1]=0;
                for(int j=i;j<=i+l;j++)
                    result[i][i+l][1]=(result[i][i+l][1]+Ar[j])%100;
            }
        }

        /*for(int i=0;i<N;i++,cout<<"\n")
            for(int j=0;j<N;j++)
            cout<<result[i][j][0]<<"::"<<result[i][j][1]<<"    ";
        */
        cout<<result[0][N-1][0]<<"\n";
    }
    return 0;
}
