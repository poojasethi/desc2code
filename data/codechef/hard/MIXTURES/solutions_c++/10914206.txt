#include<cstdio>
#define INF 1000000000
using namespace std;
int main(){
    int n;
    while(scanf("%d",&n)!=EOF){
        int color[n][n],smoke[n+1][n+1];
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                color[i][j]=0;
            }
        }
        for(int i=0;i<n;i++){
            scanf("%d",&color[i][i]);
        }
        for(int i=0;i<n;i++){
            for(int j=i+1;j<n;j++){
                for(int k=i;k<=j;k++){
                    color[i][j]+=color[k][k];
                }
                color[i][j]%=100;
            }
        }
        for(int i=0;i<=n;i++){
                for(int j=0;j<=n;j++)
                smoke[i][j]=0;
        }
        for(int size=1;size<=n;size++){
            for(int start=0;start<n-size;start++){
                int end=start+size;
                if(end>start){
                        int ans=INF;
                    for(int mid=start;mid<end;mid++){
                        int tmp=smoke[start][mid]+smoke[mid+1][end]+(color[start][mid]*color[mid+1][end]);
                if(tmp<ans){
                    ans=tmp;
                }
                    }
                    smoke[start][end]=ans;
                }
            }
        }
        printf("%d\n",smoke[0][n-1]);
    }
}
