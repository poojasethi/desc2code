#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll arr[1000];
ll dp[500][500];
ll sum[1000];
int main()
{
	ll n,i,j,k;
	while(cin>>n)
	{
		
		sum[0]=0;
		for(i=0;i<n;i++)
		{
			cin>>arr[i];
			sum[i+1]=(sum[i]+arr[i])%100;
			dp[i][i]=0;	
		}
		
		if(n==1)
		cout<<"0"<<endl;
		else
		{
			for(i=2;i<=n;i++)
			{
				for(j=0;j<=n-i;j++)
				{
					k=j+i-1;
					dp[j][k]=999999999999LL;
					for(ll l=j;l<k;l++)
					{
						dp[j][k]=min(dp[j][k],dp[j][l]+dp[l+1][k]+((sum[l+1]-sum[j]+100)%100)*((sum[k+1]-sum[l+1]+100)%100));
						
					}
				//	cout<<j<<" "<<k<<" "<<dp[j][k]<<endl;
				}
			}
			cout<<dp[0][n-1]<<endl;
		}
	}
}
