#include <stdio.h>

#define rep(i,a,b) for(i=(a);i<(b);++i)

#define oo 9999999

int smoke[101][101], color[101][101], a[101];
int n;

int main(){
	int i, j, k, len, tmp;
	while(scanf("%d",&n) != EOF){
		rep(i,0,n) scanf("%d",&a[i]);

		rep(i,0,n) rep(j,0,n) smoke[i][j] = oo;
		
		int sum = 0;
		rep(i,0,n){
			sum = color[i][i] = a[i];
			smoke[i][i] = 0;
			rep(j,i+1,n){
				sum += a[j];
				color[i][j] = sum % 100;
			}
		}

		rep(len, 2, n+1) {
			rep(i, 0, n-len+1) {
				j = i+len-1;
				rep(k, i, j) {
					tmp = smoke[i][k] + smoke[k+1][j] + color[i][k] * color[k+1][j];
					if(tmp < smoke[i][j]) 
						smoke[i][j] = tmp;
				}
			}
		}
		printf("%d\n", smoke[0][n-1]);
	}
}
