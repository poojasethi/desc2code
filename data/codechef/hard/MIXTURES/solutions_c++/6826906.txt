#include <iostream>
#include<stdlib.h>
#include<stdio.h>
using namespace std;

int main() {
	// your code goes here
	int energy[105][105], node[105][105],i,j,len,en,max_energy,n;
	while(scanf("%d",&n)==1)
	{
		for(i=1;i<=n;i++)
		{
			cin>>node[i][i];
			energy[i][i]=0;
		}
		for(len=1;len<n;len++)
		{
			for(j=1;j<=n-len;j++)
			{
				max_energy=100000000;
				for(i=j;i<j+len;i++)
				{
					en=energy[j][i]+energy[i+1][j+len]+(node[j][i]*node[i+1][j+len]);
					if(en<max_energy)
					{
						max_energy=en;
						node[j][j+len]=(node[j][i]+node[i+1][j+len])%100;
					}
				}
				energy[j][j+len]=max_energy;
			}
		}
		cout<<energy[1][n]<<"\n";
	}
	return 0;
}
