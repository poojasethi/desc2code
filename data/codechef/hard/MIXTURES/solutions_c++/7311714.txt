#include<bits/stdc++.h>
using namespace std;

const int N=123;
const int mod=100;
int fsum[N][N], f[N][N], a[N];

int main() {
    int n;
    while(scanf("%d", &n)!=-1) {
        for(int i=1;i<=n;i++) {
            scanf("%d", &a[i]);
        }
        for(int i=1;i<=n;i++) {
            fsum[i][i]=a[i];
            f[i][i]=0;
        }
        for(int l=2;l<=n;l++) {
            for(int i=1;i<=n-l+1;i++) {
                int j=i+l-1;
                int temp=(int)1e9;
                for(int k=i;k<j;k++) {
                    temp=min(temp, f[i][k]+f[k+1][j]+fsum[i][k]*fsum[k+1][j]);
                    fsum[i][j]=(fsum[i][k]+fsum[k+1][j])%mod;
                }
                f[i][j]=temp;
            }
        }
        printf("%d\n", f[1][n]);
    }
}
