inline int fi (){//fast input function

    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = getcx());
    if(c=='-') {neg=1;c=getcx();}
    for(;c>47 && c<58;c = getcx()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;

}

#define INF 2000000009

int main()
{
    init();
    int n;
    int v[102];
    int dp[102][102];
    int mix[102][102];

    while(scanf("%d",&n)>0){

        forall(i,1,n+1) v[i]=fi();

        forall(i,1,n+1){
            forall(j,1,n+1){
                dp[i][j] = INF;
            }
        }

        forall(i,1,n+1){
            dp[i][i] = 0;
            mix[i][i]= v[i];
        }

        forall(L,2,n+1){ // chain length
            forall(i,1,n-L+2){ // i is starting of chain
                int j = i+L-1; // j is ending of chain
                dp[i][j] = INF;
                forall(k,i,j+1){
                    int r = dp[i][k]+dp[k+1][j] + mix[i][k]*mix[k+1][j];
                    if(r<dp[i][j]){
                        dp[i][j] = r;
                        mix[i][j] = (mix[i][k]+mix[k+1][j])%100;
                    }
                }
            }
        }

        printf("%d\n",dp[1][n]);

    }

    return 0;
}
