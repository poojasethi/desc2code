#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
    long arr[200],sum[200][200],smk[200][200];
    long long minimum;
    int n;
    while(scanf("%d",&n)!=EOF)
    {
        minimum=0;
        for(int i=1;i<=n;i++)
            scanf("%d",&arr[i]);
        for(int i=1;i<=n;i++)
        {
                sum[i][i]=arr[i];
                smk[i][i]=0;
        }
        for(int i=1;i<n;i++)
        {
            for(int j=1;i+j<=n;j++)
            {
                minimum= 1000000000;
                for(int k=j;k<j+i;k++)
                {
                    if(smk[j][k]+smk[k+1][j+i]+sum[j][k]*sum[k+1][j+i]<=minimum)
                    {
                        smk[j][j+i]=smk[j][k]+smk[k+1][j+i]+sum[j][k]*sum[k+1][j+i];
                        sum[j][j+i]=(sum[j][k]+sum[k+1][j+i])%100;
                        minimum=smk[j][j+i];
                    }
                }
            }
        }
        printf("%ld\n",smk[1][n]);
    }
    return 0;
}
