import sys
lines=sys.stdin.readlines()
x=len(lines)
o=0
while o < x:
    n=int(lines[o].strip())
    a=map(int,lines[o+1].strip().split())
    m=[]
    for i in range(n):
        m.append(list([0]*n))
    for j in range(1,n):
        for i in range(j-1,-1,-1):
            m[i][j]=10000000
            for k in range(i,j):
                q=m[i][k]+m[k+1][j]+(sum(a[i:k+1])%100)*(sum(a[k+1:j+1])%100)
                if q<m[i][j]:
                    m[i][j]=q
    o+=2
    print m[0][n-1]
