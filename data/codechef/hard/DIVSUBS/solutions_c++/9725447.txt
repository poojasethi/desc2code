#include<bits/stdc++.h>
using namespace std;

int main()
{
    int T;
    cin>>T;
    for(;T;--T)
    {
        int  N , i ;
        cin>>N;
        vector < int > A(N);
        vector < int > Pre(N,0);
        vector < int > M(N,-1);
        for(i=0;i<N;++i)
        {
            cin>>A[i];
            Pre[i] = ((i!=0?Pre[i-1]:0) + A[i])%N;
        }

        for(i=0;i<N;++i)
        {
            if(Pre[i]==0)
            {
                cout<<i+1<<"\n";
                for(int j=0;j<=i;++j)
                    cout<<j+1<<" ";
                break;
            }

            if(M[Pre[i]]!=-1)
            {
                cout<<i-M[Pre[i]]+1<<"\n";
                for(int j=M[Pre[i]];j<=i;++j)
                    cout<<j+1<<" ";
                break;
            }
            M[Pre[i]] = i+1 ;
        }

        cout<<'\n';
    }
    return 0;
}
