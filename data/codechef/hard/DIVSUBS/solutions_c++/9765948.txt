#include<bits/stdc++.h>
using namespace std;
int Test,N,a[100010],Start,End;
map<int,int>pos;
main()
{
    scanf("%d",&Test);
    while(Test--)
    {
        pos.clear();pos[0]=1;
        Start=0;End=-2;
        long long sum=0;
        scanf("%d",&N);
        for(int i=1;i<=N;i++)
        {
            scanf("%d",&a[i]);
            sum=(sum+a[i])%N;
            if(pos[sum])Start=pos[sum],End=i;
            pos[sum]=i+1;
        }
        printf("%d\n",End-Start+1);
        for(int i=Start;i<=End;i++)
            printf("%d ",i);
        printf("\n");
    }
}
