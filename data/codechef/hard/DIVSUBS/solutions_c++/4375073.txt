#include <cstdio>
#include <algorithm>
using namespace std;

struct x
{
	long long val;
	int pos;
}b[100000];

bool comp(const x a,const x b)
{
	if(a.val == b.val)
		return (a.pos<b.pos);
	return (a.val<b.val);
}

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,a[100000];
		b[0].val = 0;
		b[0].pos=0;
		scanf("%d",&n);
		for(int i=0;i<n;i++) scanf("%d",&a[i]);

		for(int i=1;i<=n;i++)
		{
			b[i].val = (b[i-1].val+a[i-1])%n;
			b[i].pos = i;
		}
		sort(b,b+n+1,comp);
		int l,r;
		for(int i=0;i<=n;i++)
		{
			if(i == 0)
			{
				if(b[0].val == b[1].val)
				{
					l = b[0].pos;
					r = b[1].pos-1;
					break;
				}
			}
			else if(i == n)
			{
				if(b[n-1].val == b[n].val)
				{
					l = b[n-1].pos;
					r = b[n].pos-1;
					break;
				}
			}
			else
			{
				if(b[i].val == b[i-1].val)
				{
					l = b[i-1].pos;
					r = b[i].pos-1;
					break;
				}
				else if(b[i].val == b[i+1].val)
				{
					l = b[i].pos;
					r = b[i+1].pos-1;
					break;
				}
			}
		}
		printf("%d\n",r-l+1);
		for(int i=l;i<=r;i++)
		{
			printf("%d ",i+1);
		}
		printf("\n");
	}
	return 0;
}
