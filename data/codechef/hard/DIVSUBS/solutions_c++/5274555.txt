#include <iostream>
#define f first
#define s second
using namespace std;

int main()
{
    int t,n,i,j,x[100001],sum[100001],is,ie;

    cin>>t;

    for(j=1;j<=t;j++)
    {
        cin>>n;
        sum[0]=0;
        for(i=1;i<=n;i++)
            {cin>>x[i];
            sum[i]=(sum[i-1]+x[i])%n;
            }

      pair<int,bool> v[n];
      for(i=0;i<n;i++)
        v[i].s=false;



    v[0].f=0;
    v[0].s=true;

    for(i=1;i<=n;i++)
    {
       if(v[sum[i]].s==false)
       {
         v[sum[i]].s=true;
         v[sum[i]].f=i;

       }
       else
       {
           is=v[sum[i]].f;
           ie=i;
           break;
       }
    }

       cout<<ie-is<<endl;

       for(i=is+1;i<=ie;i++)
        cout<<i<<" ";

       cout<<endl;






    }
}
