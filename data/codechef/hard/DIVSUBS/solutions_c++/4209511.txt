#include <bits/stdc++.h>

#define MAXN 100005
#define st first
#define nd second

using namespace std;

typedef pair< int,int > pii;
typedef long long ll;

int N,TC;
int A[MAXN];
int H[MAXN];

int main()
{
	int i,j;

	cin >> TC;
	
	for( int t=1; t<=TC; t++ )
	{
		memset( H,-1,sizeof H );
	
		scanf("%d",&N);
		
		for( i=1; i<=N; i++ )
		{
			scanf("%d",&A[i]);
			A[i] %= N;
		}	
		
		H[0] = 0;
		for( i=1; i<=N; i++ )
		{
			A[i] = ( A[i] + A[i-1] ) % N;
			
			if( H[ A[i] ] != -1 ) break;
			H[ A[i] ] = i;
		}
		
		printf("%d\n",i-H[ A[i] ]);
		for( j=H[ A[i] ]+1; j<=i; j++ )
			printf("%d ",j);
		printf("\n");	
	}
}
