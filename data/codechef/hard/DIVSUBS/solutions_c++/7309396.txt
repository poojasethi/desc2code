#include<bits/stdc++.h>
using namespace std;

const int N=123456;
int a[N], f[N];

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n;
        scanf("%d", &n);
        a[0]=0;
        for(int i=1;i<=n;i++) {
            int num;
            scanf("%d", &num);
            a[i]=(a[i-1]+num)%n;
        }
        int l, r;
        memset(f, -1, sizeof(f));
        for(int i=0;i<=n;i++) {
            if(f[a[i]]>=0) {
                l=f[a[i]];
                r=i;
                break;
            } else {
                f[a[i]]=i;
            }
        }
        printf("%d\n", r-l);
        for(int i=l+1;i<=r;i++) {
            printf("%d ", i);
        }
        printf("\n");
    }
}
