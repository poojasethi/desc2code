#include<iostream>
using namespace std;
#define max 100000
int main()
{
	int t,n,i,j,flag,k,m[max+1];
	long long int a[max],sum;
	cin>>t;
	while(t--)
	{
		m[0]=0;
		cin>>n;
		sum=0;
		for(i=0;i<n;i++)
		{
			cin>>a[i];
			sum+=a[i];
			m[i+1]=sum%n;
		}
		flag=1;
		for(i=n;flag&&i>=1;i--)
			for(j=i-1;flag&&j>=0;j--)
			{
				if(m[i]==m[j])
				{
					flag=0;
					cout<<i-j<<endl;
					for(k=j+1;k<i;k++)
						cout<<k<<" ";
					cout<<k<<endl;
				}
			}
	}
	return 0;
}
