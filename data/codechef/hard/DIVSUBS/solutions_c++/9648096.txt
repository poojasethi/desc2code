#include <bits/stdc++.h>
using namespace std;

int main() {
	int i,t,n;
	vector<int> v(100002);
	vector<pair<int,int> > b(100002);
	scanf("%d",&t);
	while(t--)
	{
	    scanf("%d",&n);
	    for(i=1;i<=n;i++)
	        scanf("%d",&(v[i]));
	   b[0].first=0;
	   b[0].second=0;
	   for(i=1;i<=n;i++)
	   {
	       b[i].first=(b[i-1].first+v[i])%n;
	       b[i].second=i;
	   }
	   sort(b.begin(),b.begin()+n+1);
	   int l,r;
	   for(i=1;i<=n;i++)
	   {
	       if(b[i].first==b[i-1].first)
	       {
	           l=b[i-1].second;
	           r=b[i].second;
	       }
	   }
	   printf("%d\n",r-l);
	   for(i=l+1;i<=r;i++)
	   {
	       printf("%d ",i);
	   }
	}
	return 0;
}
