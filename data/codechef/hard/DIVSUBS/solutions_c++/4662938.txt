#include <bits/stdc++.h>
using namespace std;

typedef long long ll;


const ll mod=1000000007ll;
double eps=1e-8;

	
#define sc(x) scanf("%d",&(x))
#define scl(x) scanf("%lld",&x)
#define pb push_back
#define mp make_pair
#define fori(n) for(int i=0;i<n;i++)
#define forj(n) for(int j=0;j<n;j++)



////************
int main()
{
	
	int tt;
	sc(tt);
	while(tt--)
	{
		int n;sc(n);
		ll a[n];
		for(int i=0;i<n;i++)
		{
			scl(a[i]);
			a[i]=a[i]%n;
		}
		ll b[n+1];
		b[0]=0;
		for(int i=1;i<=n;i++)
		{
			b[i]=b[i-1]+a[i-1];
			b[i]=(b[i]%n);
		}
		//fori(n+1)cout<<b[i]<<" ";cout<<endl;
		
		map<ll,int> M;
		for(int i=1;i<=n;i++)
		{
			if(b[i]==0)
			{
				printf("%d\n",i);
				for(int j=1;j<=i;j++)
				{
					printf("%d ",j);
				}
				cout<<endl;
				break;
			}
			else if(M[b[i]]!=0)
			{
				printf("%d\n",i-M[b[i]]);
				for(int j=M[b[i]]+1;j<=i;j++)
				{
					//cout<<"j "<<j<<endl;
					printf("%d ",j);
				}
				cout<<endl;
				break;
			}
			else
			{
				//cout<<"i "<<i<<" m[b[i]] "<<M[b[i]]<<endl;
				M[b[i]]=i;
				//cout<<"i "<<i<<" m[b[i]] "<<M[b[i]]<<endl;
			}
		}
	}


}
