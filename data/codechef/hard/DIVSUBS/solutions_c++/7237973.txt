#include <cstdio>
#include <cstring>
#define MAX 100000
 
using namespace std;
 
long long int a,sum;long int marked[MAX+10];
 
int main()
{
    long int i,t,n,left,right,size;
    scanf("%ld",&t);
    while(t--)
    {
    memset(marked,-1,sizeof(marked));
    sum=marked[0]=0;
    scanf("%ld",&n);
    for(i=1;i<=n;i++)
    {scanf("%lld",&a);sum=(sum+a)%n;
    if(marked[sum]>-1) {right=i;left=marked[sum]+1;}    
    marked[sum]=i; 
    }
    size=right+1-left;
    printf("%ld\n",size);
    for(i=left;i<=right;i++)
    printf("%ld ",i);
    }
    return 0;
}
