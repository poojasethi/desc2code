#include <iostream>
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <queue>
#include <algorithm>
#define lli long long int
#define pb push_back
#define pii pair<int,int>

using namespace std;

int arr[100009];
int main()
{
    int t,n,i,j,k,a,b;
    scanf("%d",&t);
    while(t--){
        lli s=0;
        scanf("%d",&n);
        vector<pii> vec;
        vec.pb(make_pair(0,0));
        for(i=1; i<=n; i++)
        {
            scanf("%d",&arr[i]);
            s=(s+arr[i])%n;
            vec.pb(make_pair(s,i));
        }

        sort(vec.begin(),vec.end());
       
        for(i=1;i<=n;i++){
            if(vec[i].first == vec[i-1].first){
                a=vec[i-1].second;
                b= vec[i].second;
                break;
            }
        }

        printf("%d\n",b-a);
        for(i=a+1;i<=b;i++)printf("%d ",i);
        printf("\n");
    }

    return 0;
}
