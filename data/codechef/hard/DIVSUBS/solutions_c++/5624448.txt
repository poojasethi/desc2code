#include<iostream>
using namespace std;

int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n],b[n+1],k=0,mark[100010]={0},p,q;
        b[0]=0;
        for(int i=0;i<n;i++)
        {
            cin>>a[i];
            b[i+1]=(a[i]+b[i])%n;
        }
        for(int i=1;i<=n;i++)
        {
            if(b[i]==0)
            {
                p=0;
                q=i;
                break;
            }
            if(mark[b[i]]!=0)
            {
                p=mark[b[i]];
                q=i;
                break;
            }
            mark[b[i]]=i;
        }
        cout<<q-p<<endl;
        for(int i=1;p+i<=q;i++)
            cout<<p+i<<" ";
            cout<<endl;
    }
    return 0;
}
