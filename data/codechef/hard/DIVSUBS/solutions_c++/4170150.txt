#include <iostream>
#include <cstdlib>
#include <cstdio>
using namespace std;
 
typedef long long ll;
 
void solve()
{
	long N;
	scanf("%ld",&N);
	ll * ptr = (ll *) malloc((N+1) * sizeof(ll));
	long * B = (long *) malloc((N+1) * sizeof(long));
	for(int i = 0; i <= N; i++) B[i] = -1;
	B[0] = 0;
	ll l,r,a, v;
	ptr[0] = 0;
	for(int i = 0; i < N; i++)
	{
		scanf("%lld",&a);
		ptr[i+1] = ptr[i] + a;
		v = ptr[i+1] % N;
//		cout << ptr[i+1] <<  " " << a << " " << B[v] << endl;
		if(B[v] != -1)
		{
			l = B[v] + 1;
			r = i+1;
		}
		else
		{
			B[v] = i+1;
		}
	}
 
	free(ptr) , free(B);
	cout << (r - l + 1) << endl;
 
	for(int i = l; i <= r; i++)
	{
		cout << i << " " ;
	}
	cout << endl;
	return;
}
 
int main() {
	// your code goes here
	int t;
	cin >> t;
	while(t--)
		solve();
	return 0;
}
