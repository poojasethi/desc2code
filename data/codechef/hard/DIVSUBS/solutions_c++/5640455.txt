#include<stdio.h>
#include<iostream>
#include<algorithm>
#define LL long long
using namespace std;
 
LL a[100001],b[100001],m[100001];
 
 
int main()
{
	LL sum,t,i,j,n,g,x1,x2,temp;
	cin>>t;
	while(t--)
	{
		cin>>n;
		for(i=1;i<=n;i++)
		{
			m[i] = 0;
		}
 
		b[0] = 0;
		for(i=1;i<=n;i++)
		{
			cin>>a[i];
			b[i] = b[i-1] + a[i];
		}
		g = 0;
		for(i=1;i<=n;i++)
		{
			temp = b[i] % n;
			if(temp == 0)
			{
				g = 1;
				break;
			}
			m[temp]++;
 
		}
		if(g == 1)
		{
			cout<<i<<endl;
			for(j=1;j<=i;j++)
			{
				cout<<j<<" ";
			}
			cout<<endl;
			continue;
		}
 
		for(i=1;i<=n;i++)
		{
			if(m[i] >= 2)
			{
				temp = i;
				break;
			}
		}
 
		sum = temp;
 
		g = 0;
		for(i=1;i<=n;i++)
		{
			if(g == 0)
			{
				if(b[i] % n == sum)
				{
					g = 1;
					x1 = i;
				}
			}
			else
			{
				if(b[i] % n == sum)
				{
					x2 = i;
				}
			}
		}
		cout<<(x2 - x1)<<endl;
		for(i=x1+1;i<=x2;i++)
		{
			cout<<i<<" ";
		}
		cout<<endl;
 
	}
 
	return 0;
}
