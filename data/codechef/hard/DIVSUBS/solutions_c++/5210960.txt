

#include <iostream>
#include <cassert>
#include <cstdio>
#include <algorithm>
using namespace std;

#define maxn 1000000 + 5

int n, fl, tn, was[maxn], i, a[maxn], f[maxn], j, ti, sum;

int main () {
	scanf("%d", &tn);
	for(ti = 1; ti <= tn; ti++) {
		scanf("%d", &n);
		sum += n;
		fl = 0;
		for(i = 1; i <= n; i++) {
			scanf("%d", &a[i]);
			if (a[i] % n == 0 && !fl) {
				printf("1\n%d\n", i);
				fl = 1;
			}
		}
		if (!fl) { 
			for(i = 0; i < n; i++) was[i] = 0;
			was[0] = 1;
			for(i = 1; i <= n; i++) {
				f[i] = (f[i - 1] + a[i]) % n; 
				if (was[f[i]]) break; 
				was[f[i]] = 1;
			}
			for(j = 0; j <= i; j++) if (f[j] == f[i]) break; 
			printf("%d\n", i - j); 
			for(++j; j <= i; j++) {
				printf("%d", j);
				if (j != i) printf(" ");
			}
			puts("");
		}
	}
	assert(1 <= sum && sum <= 100000);
    return 0;
}
