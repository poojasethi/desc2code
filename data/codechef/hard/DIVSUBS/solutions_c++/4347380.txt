#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
int T,n,L,R;
int main()
{
	scanf("%d",&T);
	while(T--)
	{
		scanf("%d",&n);
		vector<int> check(n+1);
		check[0]=1;
		bool flag=true;
		for(int a,b=0,i=1;i<=n;i++)
		{
			scanf("%d",&a);
			b=(b+a)%n;
			if(check[b])
			{
				flag=false;
				L=check[b],R=i;
			}
			check[b]=i+1;
		}
		if(flag)printf("-1\n");
		else
		{
			printf("%d\n",R-L+1);
			while(L<R)printf("%d ",L++);
			printf("%d\n",R);
		}
	}
	return 0;
}
