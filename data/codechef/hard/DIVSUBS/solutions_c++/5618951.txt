#include<bits/stdc++.h>
using namespace std;
int arr[100001];
int cum[100001];
int mulc[100001];
static bool began;
int multiple;
int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&multiple);
        int i,j;
        bool single = false;
        for(i = 1;i<=multiple;i++)
        {
            scanf("%d",&arr[i]);
            if(arr[i]%multiple==0 && single==false)
            {
                single = true;
                printf("1\n%d\n",i);
            }
            cum[i] = 0;
            mulc[i] = 0;
        }
        if(single==false)
        {
            cum[0] = 0;
            for(i=1;i<=multiple;i++)
            {
                cum[i] = cum[i-1]+arr[i];
                cum[i]%=multiple;
                if(mulc[cum[i]]==1)break;
                mulc[cum[i]]=1;
            }
            for(j = 0;j<=i;j++)
            {
                if(cum[j]==cum[i])break;
            }
            //found the duplicate already
            printf("%d\n",i-j);
            for(++j;j<=i;j++)
            {
                printf("%d ",j);
            }
            printf("\n");
        }
    }
    return 0;
}
