#include <stdio.h>

void solution()
{
    int n,i;
    scanf("%d",&n);
    int A[n],sum[n];

    for(i=0;i<n;i++) scanf("%d",A+i);
    sum[0]=A[0]%n;

    for (i=1;i<n;i++)
    {
        sum[i] = sum[i-1] + A[i];
        sum[i] = sum[i]%n;
    }
    /*
    for (i=0;i<n;i++) printf("%d ",sum[i]);
    printf("\n");
    */
    int freq[n],j,inv[n];
    for (i=0;i<n;i++) freq[i]=0;

    for (i=0;i<n;i++)
    {
        if (sum[i]==0)
        {
            printf("%d\n",i+1);
            for (j=0;j<=i;j++) printf("%d ",j+1);
            printf("\n");
            break;
        }
        else
        {
            freq[sum[i]]++;
            if (freq[sum[i]]==1) inv[sum[i]]=i;
            else if (freq[sum[i]]==2)
            {
                printf("%d\n",i-inv[sum[i]]);
                for (j=inv[sum[i]]+1;j<=i;j++) printf("%d ",j+1);
                printf("\n");
                break;
            }
        }
    }
}

int main()
{
    int t;
    scanf("%d",&t);
    while(t--) solution();

    return 0;
}
