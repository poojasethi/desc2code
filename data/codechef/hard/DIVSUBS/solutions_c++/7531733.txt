#include <iostream>
#include <stdio.h>
#include <cstring>
#define maxn 100000+50
#define gc getchar_unlocked

typedef long long ll;
typedef float fl;
typedef double db;
typedef unsigned long long ull;
using namespace std;

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    int neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}
void scanint(ll &x)
{
    register ll c = gc();
    x = 0;
    ll neg = 0;
    for(;((c<48 || c>57) && c != '-');c = gc());
    if(c=='-') {neg=1;c=gc();}
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    if(neg) x=-x;
}

ll arr[maxn];
ll b[maxn];
ll freq[maxn];

int main(){
	int testCases;
	scanint(testCases);
	
	ll n,i,j;
	ll pos;
	int flag = 1;
	
	while(testCases){
		scanint(n);
		flag = 1;
		
		for(i = 0;i<n;++i){
			scanint(arr[i]);
			if((arr[i]%n)==0 && flag==1){pos = i+1;flag = 0;}
		}
		if(flag==0) printf("1\n%lld\n",pos);
		if(flag==1){ memset(freq,0,sizeof(freq));
			b[0] = 0;
			freq[0]=1;
			for(i=1;i<=n;++i){
				b[i] = (b[i-1]+arr[i-1])%n;
				if(freq[b[i]]!=0) break;
				
				++freq[b[i]];
			}
			for(j=0;j<=i;++j) if(b[j]==b[i]) break;
			printf("%lld\n",i-j);
			for(++j;j<=i;++j) printf("%lld ",j);
			printf("\n");
		}
	--testCases;
	}
return 0;
}
