#include <bits/stdc++.h>

#define MAX 100005

using namespace std;

int a[MAX];

int main()
{
	int t, n;
	scanf("%d", &t);

	while (t--) {
		scanf("%d", &n);
		for (int i=0; i<n; i++) {
			scanf("%d", &a[i]);
		}
		
		vector<pair<int, int> > p(n+1);

		p[0] = make_pair(0, 0);
		for (int i=1; i<=n; i++) {
			p[i].first = (a[i-1] + p[i-1].first) % n;
			p[i].second = i;
		}
		sort(p.begin(), p.end());
		for (int i=0; i<n; i++) {
			if (p[i].first == p[i+1].first) {
				printf("%d\n", p[i+1].second - p[i].second);
				
				if (p[i].second < p[i+1].second) {
					for (int j=p[i].second+1; j<=p[i+1].second; j++)
						printf("%d ", j);
					printf("\n");
				} else {
					for (int j=p[i+1].second+1; j<=p[i].second; j++)
						printf("%d ", j);
					printf("\n");
				}
				break;
			}
		}
	}

	return 0;
}
