#include<cstdio>
#include<cstring>

const int mx = 100005;
int a[mx], last[mx], n;
int main() {
    int t;
    scanf("%d", &t);
    while(t--) {
        scanf("%d", &n);
        for(int i = 1; i <= n; ++i) scanf("%d", a + i);

        memset(last, -1, sizeof(last));

        int cur = 0;
        int ans1 , ans2;
        for(int i = 0; i <= n; ++i) {
            cur = (cur + a[i]) % n;
            if(last[cur] != -1) {
                ans1 = last[cur] + 1;
                ans2 = i;
                break;
            }
            last[cur] = i;
        }

        printf("%d\n", ans2 - ans1 + 1);
        for(int i = ans1; i <= ans2; ++i) printf("%d ", i);
        printf("\n");
    }
    return 0;
}
