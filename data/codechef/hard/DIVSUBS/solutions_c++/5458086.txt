#include <cstdio>
#include <map>
using namespace std;
int n,t;
int ds[100005];
map <int, int> m;

void ispisi(int l, int r) {
  printf("%d\n",r-l);
  for (int i=l+1; i<=r; i++) printf("%d ", i);
  printf("\n");
  for (int i=r+1; i<=n; i++) scanf("%d", &ds[i]);
}

int main (){
  scanf("%d",&t);
  while (t--) {
    scanf("%d",&n);
    ds[0]=0;
    m.clear();
    for (int i=1;i<=n;i++) {
        scanf("%d",&ds[i]);
        ds[i]=(ds[i]+ds[i-1])%n;
        if (ds[i]==0) {ispisi(0,i); break;}
        if (m[ds[i]]) {ispisi(m[ds[i]],i); break;}
        else m[ds[i]]=i;
    }
  }
  return 0;
}
