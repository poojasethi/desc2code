#include <iostream>
#include <cstring>
#include <cstdio>
#include <string>
using namespace std;
#define MOD 1000000007
typedef long long int64;
int64 comb[5001][5001];
void preprocess()
{
	for(int i=0;i<=5000;++i)comb[i][0]=1;
	for(int i=1;i<=5000;++i)
	{
		for(int j=1;j<=i;++j)
			comb[i][j]=(comb[i-1][j]+comb[i-1][j-1])%MOD;
	}
}

int main()
{
	preprocess();
	int t;string str;
	for(cin>>t;t>0;t--)
	{
		cin>>str;
		int n=str.size(),n4=0;
		for(int i=0;i<n;++i)if(str[i]=='4')n4++;
		if(n4==0 || n4==n)cout<<1<<endl;
		else cout<<((comb[n][n4]-comb[n-2][n4-1])%MOD+MOD)%MOD<<endl;
	}

	return 0;
}
