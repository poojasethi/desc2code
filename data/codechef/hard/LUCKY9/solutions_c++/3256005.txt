#include<cstdio>
#include<iostream>
#include<cstring>
using namespace std;
char s[5000 + 10];
typedef long long int ll;
ll fac[5000 + 10];
ll invfac[5000 + 10];
ll invmod[5000 + 10];
int main()
{
    ll mod = 1000000007;
    invmod[1] = 1;
    invfac[1] = 1;
    invfac[0] = 1;
    fac[1] = 1;
    fac[0] = 1;

    for(int i = 2 ; i<= 5000 ; i++)
    {
        ll temp;
        temp = i;
        fac[i] = (fac[i-1]*temp)%mod;
        invmod[i] = ((-mod/i)*invmod[mod%i])%mod + mod;
        invfac[i] = (invfac[i-1]*invmod[i])%mod;
    }
    int tc,t;
    scanf("%d",&tc);
    for(t = 1 ; t<= tc ; t++)
    {
        scanf("%s",s + 1);
        int len = strlen(s + 1);
        int c = 0;
        for(int j = 1 ; j<= len ; j++)
        {
            if(s[j] == '4')
            c++;
        }
        ll ans ;
        if(c > 0 && c < len)
        {
            ll t1,t2;
            t1 = fac[len];
            t1 = (t1*invfac[c])%mod;
            t1 = (t1*invfac[len - c])%mod;
            t2 = fac[len - 2];
            t2 = (t2*invfac[c - 1])%mod;
            t2 = (t2*invfac[len - c - 1])%mod;
            ans = t1 - t2;
            if(ans < 0) ans = ans + mod;
        }
        else
        {
            ans = 1;
        }
        printf("%lld\n",ans);
    }
    return 0;
}
