#include<iostream>
#include<string>
#include<cstdio>
using namespace std;

#define MAX 5001
#define MOD 1000000007

int dp[MAX][MAX];

void compute()
{
    dp[1][1]=1;
    for(int i=1;i<MAX;i++) dp[0][i]=dp[i][0]=1;

    for(int i=1;i<MAX;i++)
    {
        for(int j=1;j<MAX;j++)
        {
            if(i==1 && j==1) continue;

            dp[i][j]=(dp[i-1][j]+dp[i][j-1])%MOD;
        }
    }
}

int main()
{
    compute();
    int t;
    scanf("%d",&t);
    while(t--)
    {
        string str;
        cin>>str;
        int fCount=0,sCount=0;
        for(int i=0;i<str.size();i++)
        {
            if(str[i]=='4')
                fCount++;
            else
                sCount++;
        }
        printf("%d\n",dp[fCount][sCount]);
    }
    return 0;
}
