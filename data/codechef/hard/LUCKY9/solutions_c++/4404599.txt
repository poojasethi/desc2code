#include <iostream>
#include <stdio.h>
#include <vector>
#include <algorithm>
#include <string.h>
#include <math.h>
using namespace std;
int t,n;
long long base=1000000007;
long long C[5001][5001];
char s[5001];
void khoitao(){
        C[0][0]=1;
        for (int i=1;i<=5000;i++){
            C[i][0]=1;
            for (int j=1;j<=i;j++) C[i][j]=(C[i-1][j]+C[i-1][j-1])%base;
        }
}
long long xuli(){
    int d4=0;
    n=strlen(s);
    for (int i=0;i<n;i++)
        if (s[i]=='4') d4++;
    if ((d4==n) || (d4==0)) return(1);
    return(((C[n][d4]-C[n-2][d4-1])%base+base)%base);
}
int main(){
    //freopen("LUCKY9.inp","r",stdin);
    //freopen("LUCKY9.out","w",stdout);
    khoitao();
    scanf("%d\n",&t);
    while (t){
        gets(s);
        //cout<<s<<endl;
        printf("%d\n",xuli());
        t--;
    }
}
