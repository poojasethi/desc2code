LLI powx(LLI a, LLI n)
{
	LLI ans = 1;
	while(n > 0)
	{
		if(n&1)
			ans = (ans*a) % M;
		a = (a*a) % M;
		n >>= 1;
	}
	return ans;
}

LLI comb(int n,int r)
{
	if(r < 0)
		return 0;
	LLI num = 1, den = 1;
	for(int i=1; i<=r; i++)
	{
		den = (den*i) % M;
		num = (num*n--) % M;
	}

	den = powx(den,M-2);
	num = (num*den) % M;
	return num;
}
