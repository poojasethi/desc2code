    int t,n,m,i,j,k,l,d;
    LL v;
    dp[0][27][18][9][9]=1;
    for(n=1;n<=36;n++)
    {
        if(n&1)

            for(i=0;i<82;i++)

                            for(d=1;d<10;d++)
                            {
                                dp[n][i+g2[d]][j+g3[d]][k+g5[d]][l+g7[d]]+=dp[n-1][i][j][k][l];
                            }
                        }
                    }

        }
