#include<cstdio>
int i,T,L,N2,N3,N5,N7,p;
long long V;
unsigned int Zero[36],f[36][109][73][37][37];
int main()
{
	f[0][54][36][18][18]=1;
	f[0][55][36][18][18]=1;
	f[0][54][37][18][18]=1;
	f[0][56][36][18][18]=1;
	f[0][54][36][19][18]=1;
	f[0][55][37][18][18]=1;
	f[0][54][36][18][19]=1;
	f[0][57][36][18][18]=1;
	f[0][54][38][18][18]=1;
	for(i=0;i<35;++i)
	{
		for(N2=0;N2<109;++N2)
			for(N3=0;N3<73;++N3)
				for(N5=0;N5<37;++N5)
					for(N7=0;N7<37;++N7)
						if(f[i][N2][N3][N5][N7])
						{
							if(i&1) p=1;
							else p=-1;
							f[i+1][N2][N3][N5][N7]+=f[i][N2][N3][N5][N7];
							f[i+1][N2+p][N3][N5][N7]+=f[i][N2][N3][N5][N7];
							f[i+1][N2][N3+p][N5][N7]+=f[i][N2][N3][N5][N7];
							f[i+1][N2+p+p][N3][N5][N7]+=f[i][N2][N3][N5][N7];
							f[i+1][N2][N3][N5+p][N7]+=f[i][N2][N3][N5][N7];
							f[i+1][N2+p][N3+p][N5][N7]+=f[i][N2][N3][N5][N7];
							f[i+1][N2][N3][N5][N7+p]+=f[i][N2][N3][N5][N7];
							f[i+1][N2+p+p+p][N3][N5][N7]+=f[i][N2][N3][N5][N7];
							f[i+1][N2][N3+p+p][N5][N7]+=f[i][N2][N3][N5][N7];
							if(i+2<=35) Zero[i+2]+=f[i][N2][N3][N5][N7]*9u;
						}
		if(i+2<=35) Zero[i+2]+=Zero[i]*90u;
	}
	for(scanf("%d",&T);T--;)
	{
		scanf("%d%lld",&L,&V);
		--L;
		if(V==0)
		{
			printf("%u\n",Zero[L]);
			continue;
		}
		for(N2=54;V%2==0;V/=2,++N2);
		for(N3=36;V%3==0;V/=3,++N3);
		for(N5=18;V%5==0;V/=5,++N5);
		for(N7=18;V%7==0;V/=7,++N7);
		if(V!=1||N2>108||N3>72||N5>36||N7>36) printf("0\n");
		else printf("%u\n",f[L][N2][N3][N5][N7]);
	}
	return 0;
}
