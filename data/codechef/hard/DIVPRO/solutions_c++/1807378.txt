#include<cstdio>
#include<map>
#include<cstring>
#include<vector>

int T;
int a,b,c,d,e,f,g,h;
short i,j,now;
unsigned int v;
unsigned int dp[2][109][73][37][37];
long long t;
std::vector<long long>pow[4];
std::vector<unsigned int>p9,p10;
std::map<long long,unsigned int>ans[37];
std::map<long long,unsigned int>::const_iterator it;
short fac[10][4];

int main()
{
    pow[0].reserve(55);
    pow[0].push_back(1ll);
    for(i=1;i<=54;++i)
        pow[0].push_back(pow[0].back()<<1ll);
    pow[1].reserve(37);
    pow[1].push_back(1ll);
    for(i=1;i<=36;++i)
        pow[1].push_back(pow[1].back()*3ll);
    pow[2].reserve(19);
    pow[2].push_back(1ll);
    pow[3].reserve(19);
    pow[3].push_back(1ll);
    for(i=1;i<=18;++i)
    {
        pow[2].push_back(pow[2].back()*5ll);
        pow[3].push_back(pow[3].back()*7ll);
    }
    for(i=2;i<=9;++i)
    {
        j=i;
        while(j%2==0)
        {
            ++fac[i][0];
            j/=2;
        }
        while(j%3==0)
        {
            ++fac[i][1];
            j/=3;
        }
        if(j%5==0)
            ++fac[i][2];
        if(j%7==0)
            ++fac[i][3];
    }
    dp[0][54][36][18][18]=1;
    now=1;
    p9.reserve(37);
    p10.reserve(37);
    p9.push_back(1u);
    p10.push_back(1u);
    for(j=1;j<=36;++j)
    {
        p9.push_back(p9.back()*9u);
        p10.push_back(p10.back()*10u);
        i=(j-1)>>1;
        ans[j][0]=p9[j-i]*(p10[i]-p9[i]);
        memset(dp[now],0,sizeof(dp[now]));
        if(now)
            for(a=0;a<109;++a)
                for(b=0;b<73;++b)
                    for(c=0;c<37;++c)
                        for(d=0;d<37;++d)
                        {
                            v=dp[now^1][a][b][c][d];
                            if(v)
                                for(i=1;i<10;++i)
                                    if((e=a+fac[i][0])<109 && (f=b+fac[i][1])<73 &&  (g=c+fac[i][2])<37 && (h=d+fac[i][3])<37)
                                        dp[now][e][f][g][h]+=v;
                        }
        else
            for(a=0;a<109;++a)
                for(b=0;b<73;++b)
                    for(c=0;c<37;++c)
                        for(d=0;d<37;++d)
                        {
                            v=dp[now^1][a][b][c][d];
                            if(v)
                                for(i=1;i<10;++i)
                                    if((e=a-fac[i][0])>=0 && (f=b-fac[i][1])>=0 &&  (g=c-fac[i][2])>=0 && (h=d-fac[i][3])>=0)
                                        dp[now][e][f][g][h]+=v;
                        }
        for(a=54;a<109;++a)
            for(b=36;b<73;++b)
                for(c=18;c<37;++c)
                    for(d=18;d<37;++d)
                        if(dp[now][a][b][c][d])
                            ans[j][pow[0][a-54]*pow[1][b-36]*pow[2][c-18]*pow[3][d-18]]=dp[now][a][b][c][d];
        now^=1;
    }
    scanf("%d",&a);
    while(a--)
    {
        scanf("%hd %lld",&i,&t);
        if((it=ans[i].find(t))!=ans[i].end())
            printf("%u\n",it->second);
        else
            puts("0");
    }
    return 0;
}
