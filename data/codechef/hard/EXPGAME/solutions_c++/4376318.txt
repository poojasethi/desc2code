#include <iostream>
#include <algorithm>
#include <cmath>
#include <set>
#include <cstdio>
#define forn(i,s,n)for ( i = s;i < n; i++)

using namespace std;

int main() {
       
    int sg[100333] = {0};
    int deg[7]={0},i,j;
    forn(i,1,7)deg[i] = pow(i,i);// forn(i,1,7)cout << deg[i] << " ";
    
    forn(i,1,100000){
        set<int> s;
        for(int j = 1; i >= deg[j]; j++)
            s.insert(sg[i-deg[j]]);
        int g = 0;
        while(s.find(g) != s.end())g++;
        sg[i] = g;
    }
    // forn(i,1,29)cout << sg[i] << " ";
    int tc;
    scanf("%d", &tc);
    while (tc--){
        int n;  
        scanf("%d", &n);
        int ans = 0;
        for (i = 1; i <= n; i++){
            int x;
            scanf("%d", &x);
            ans ^= sg[x];
        }          
        if (ans) printf("Little Chef\n");
        else printf("Head Chef\n");
    }
}
