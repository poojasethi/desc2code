#include <iostream>
#define SIZ 100002

using namespace std;

void populateGrundi(int grundi[]) {
	int p[] = {0, 1, 4, 27, 256, 3125, 46656, 823543};
	         //0  1  2  3   4    5     6      7
	int i, j;
	int hasChild[8];

	grundi[0] = 0;
	for(i=0 ; i<SIZ ; i++) {
		hasChild[0]=hasChild[1]=hasChild[2]=hasChild[3]=hasChild[4]=hasChild[5]=hasChild[6]=hasChild[7]=0;
		for(j=1 ; j<=6 && p[j]<=i ; j++) {
			hasChild[grundi[i-p[j]]] = 1;
		}
		for(j=0 ; j<8 ; j++) {
			if(!hasChild[j]) {
				grundi[i] = j;
				break;
			}
		}
	}
}

int main() {
	int t, n, a, i, result;
	int grundi[SIZ];

	populateGrundi(grundi);
	
	cin>>t;
	while(t--) {
		result = 0;
		cin>>n;
		
		while(n--) {
			scanf("%d", &a);
			result ^= grundi[a];
		}
		if(result == 0)
			printf("Head Chef\n");
		else
			printf("Little Chef\n");
	}

	return 0;
}
