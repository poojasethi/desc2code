int grundy[100005];
int vis[100005];

int64 powe(int a,int b){
	int64 ret=1;
	while(b){
		if(b&1){
			ret=ret*a;			
		}
		a=a*a;
		b/=2;
	}
	return ret;
}


void pre(){
	grundy[0]=0;
	int timer=1;
	for(int i=1;i<=100000;i++){
		
		for(int j=1;powe(j,j)<=i;j++){
			vis[grundy[i-powe(j,j)]]=timer;
		}
		int val=0;
		while(vis[val]==timer)
		val++;
		grundy[i]=val;
		timer++;
	}
	
}
int main(){
	pre();
	int t,n,i,val;
	cin>>t;
	while(t--){
		cin>>n;
		int xr=0;
		for(i=0;i<n;i++){
			scanf("%d",&val);
			xr^=grundy[val];
		}
		if(xr)
		printf("Little Chef\n");
		else
		printf("Head Chef\n");
	}
	return 0;
}
