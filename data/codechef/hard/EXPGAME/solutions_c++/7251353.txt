#include <bits/stdc++.h>

using namespace std;

int vis[100005];
int dp[100005];
vector <int> v;

int f(int x)
{
    if ( vis[x] ) return dp[x];
    vis[x] = true;
    set <int> ss;
    for ( int i = 0; i < v.size(); i++ ) {
        if ( x - v[i] >= 0 ) ss.insert(f(x-v[i]));
    }
    int ans = 0;
    while ( ss.find(ans) != ss.end() ) ans++;
    dp[x] = ans;
    return ans;
}

int main()
{
    v.clear();
    v.push_back(1), v.push_back(4), v.push_back(27), v.push_back(256), v.push_back(3125), v.push_back(46656);
    int t,n,x;
    cin >> t;
    while ( t-- ) {
        cin >> n;
        int ans = 0;
        for ( int i = 0; i < n; i++ ) {
            cin >> x;
            ans ^= f(x);
        }
        if ( ans ) puts("Little Chef");
        else puts("Head Chef");
    }
    return 0;
}
