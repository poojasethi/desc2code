#include <iostream>
#include<cstdio>
#include<string.h>
using namespace std;

int main()
{
   int t,i,j;
    int a[]={1,4,27,256,3125,46656};
   int sg[100005],f[8];
   sg[0]=0;
   for(i=1;i<=100000;i++)
   {memset(f,0,sizeof(f));
 //  cout<<i<<endl;
       for(j=0;;j++)
       {
           if(i>=a[j])
           f[sg[i-a[j]]]=1;
           else break;
       }
       for(j=0;;j++)
         if(!f[j]) break;
        sg[i]=j;
    }
    //cout<<sg[4];
   scanf("%d",&t);
   while(t--)
   {
       int n;
       scanf("%d",&n);
       int ans=0,x;
       for(i=0;i<n;i++)
       {
           cin>>x;
           ans^=sg[x];
       }
       if(ans) printf("Little Chef\n");
       else printf("Head Chef\n");
   }
    return 0;
}
