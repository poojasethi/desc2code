#include<bits/stdc++.h>
using namespace std;
int a[100001];
int b[6]={1,4,27,256,3125,46656};
int arr[6];
void init()
{
    a[0]=0;
   int i,j,k;
   for(i=1;i<=100000;i++)
   {
      k=5;
     arr[0]=0;arr[1]=1;arr[2]=2;arr[3]=3;arr[4]=4;arr[5]=5;

      while(i<b[k])
      k--;
      // cout<<k<<"\n";
      while(k>=0)
      {
         arr[a[i-b[k]]]=-1;
         k--;
      }

      k=0;
      while(1)
      {
         if(arr[k]!=-1)
         {
         a[i]=arr[k];
         break;
         }
         else
         k++;
       }
    }
}
int main()
{
    init();

        n=n-1;
        while(n--)
        {
            cin>>num;

        }
         if(value==0)
