
#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int

ll x[11];
int g[100005];

void comp() {
    for(ll i=1;i<11;++i) {
	x[i]=pow(i,i);
    }
}

void grundy() {
    g[0]=0;
    g[1]=1;
    int a[100005];
    for(int i=2;i<=100000;++i) {
	for(int j=0;j<11;++j) {
	    if(i-x[j]>=0) {
		a[g[i-x[j]]]=i;
	    }
	    else {
		break;
	    }
	}
	for(int j=0;j<=100000;++j) {
	    if(a[j]!=i) {
		g[i]=j;
		break;
	    }
	}
    }
}

int main()
{
    memset(g,-1,sizeof(g));
    comp();
    grundy();
    int t;
    si(t);
    while(t--) {
	int n;
	si(n);
	int x=0,a;
	for(int i=0;i<n;++i) {
	    si(a);
	    x = x ^ g[a];
	}
	if(x) {
	    cout<<"Little Chef\n";
	}
	else {
	    cout<<"Head Chef\n";
	}
    }
    return 0;
}
