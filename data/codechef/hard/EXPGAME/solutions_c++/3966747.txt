using namespace std;

int T, N, a[105], dp[100005];
int main(){
    for(int i = 0; i < 100005; i++){
        vector <int> v;
        for(int j = 1; j <= 6; j++){
            int k = 1;
            for(int p = 0; p < j; p++)k *= j;
            
            if(i >= k)v.push_back(dp[i - k]);
        }
        sort(v.begin(), v.end());
        
        dp[i] = 0;
        for(int j = 0; j < v.size(); j++){
            if(v[j] < dp[i])continue;
            if(v[j] > dp[i])break;
            dp[i]++;
        }
    }
    
    scanf("%d", &T);
    while(T--){
        scanf("%d", &N);

        int res = 0;
        for(int i = 0; i < N; i++){
            scanf("%d", &a[i]);
            res ^= dp[a[i]];
        }
        
        if(res == 0)printf("Head Chef\n");
        else printf("Little Chef\n");
    }
    
    return 0;
}
