#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int MAX = 100000;
int req[] = {1,4,27,256,3125,46656};
int grundy[MAX+5];
void pre(){
  grundy[0] = 0;
  for(int i=1;i<=MAX;i++){
     set<int> st;
     set<int>::iterator it;
     for(int j=0;j<6;j++){
        if(req[j] <= i)
           st.insert(grundy[i-req[j]]);
        else
            break;
     }
     int res = 0;
     for(it=st.begin();it!=st.end();it++){
        if((*it) != res) break;
        res++;
     }
     grundy[i] = res;
  }
}
int main(){
int t,n,x;
pre();
scanf("%d",&t);
while(t--){
    scanf("%d",&n);
    int res = 0;
    for(int i=0;i<n;i++){
        scanf("%d",&x);
        res = res xor grundy[x];
    }
    if(res) puts("Little Chef");
    else puts("Head Chef");
}
return 0;
}
