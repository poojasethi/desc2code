#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define ld long double
#define sl(x) scanf("%lld",&x)
#define pl(x) printf("%lld",x)
#define sld(x) scanf("%Lf",&x)
#define pld(x) printf("%0.0Lf",x)
#define pline() printf("\n")
#define mod 1000000007

ll dp[100004];
ll s[]={1, 4, 27, 256, 3125, 46656};
void SG(){
    ll i,j;
    set<ll> mset;set<ll>::iterator it;
    dp[0]=0;
    for(i=1;i<=100000;i++){
        mset.clear();
        for(j=0;j<6 && s[j]<=i;j++){
            //cout<<dp[i-s[j]]<<" ";
            mset.insert(dp[i-s[j]]);
        }
        ll c=0,f=0;
        for(it=mset.begin();it!=mset.end();it++){
            //cout<<*it<<" ";
            if((*it)!=c){
                f=1;
                dp[i]=c;
                break;
            }
            c++;
        }
        if(f==0)
        dp[i]=c;
        //cout<<dp[i]<<" ";
    }
}

int main()
{
    ll t,n,x,i,ans;
    SG();
    sl(t);
    while(t--){
        sl(n);
        ans=0;
        for(i=0;i<n;i++){
            sl(x);
            ans^=dp[x];
        }
        if(ans==0)
			printf("Head Chef\n");
		else
			printf("Little Chef\n");

    }
    
    return 0;
}
