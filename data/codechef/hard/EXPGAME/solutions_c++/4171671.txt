#include <iostream>
#include <cstdio>
#include <cstring>
#define FOR(i,m,n) for(i=m;i<=n;i++)
#define MAX 100000
#define max_grundy_number 6
#define min_grundy_number 0
#define win "Little Chef"
#define loss "Head Chef"

using namespace std;

int removable[]={0,1,4,27,256,3125,46656},grundy[MAX+1];
void update_grundy()
{
    long int i;
    bool grundy_reachable[max_grundy_number+1];
    int j;
    FOR(i,1,MAX)
    {
        memset(grundy_reachable,false,sizeof(grundy_reachable));

        for(j=1;removable[j]<=i && j<=max_grundy_number;j++)
           grundy_reachable[grundy[i-removable[j]]]=true;

        FOR(j,min_grundy_number,max_grundy_number)
            if(!grundy_reachable[j])  {grundy[i]=j;break;}
    }
}

int main()
{
    int i,j,t,n,grundy_eq;long int a;
    scanf("%i",&t);
    grundy[0]=0;
    update_grundy();
    FOR(i,1,t)
    {
        scanf("%i",&n);
        scanf("%ld",&a); grundy_eq = grundy[a];
        FOR(j,2,n)
        { scanf("%ld",&a); grundy_eq = grundy_eq ^ grundy[a];}
        if(grundy_eq) printf(win);
        else printf(loss);
        printf("\n");
    }
    return 0;
}
