#include<bits/stdc++.h>
using namespace std;
vector<int> pick(8,0);
vector<int> grundyvalue(100005, -1);
int grundy(int x)
{
	if(grundyvalue[x]!=-1)
	return grundyvalue[x];
	vector<int> mov;
	set<int> s;
	for(int i=0;i<7 && (x-pick[i]>=0);i++)
	{
		mov.push_back(x-pick[i]);
	}
	for(int i=0;i<mov.size();i++)
	{
		s.insert(grundyvalue[mov[i]]);
	}
	int ret=0;
	set<int>::iterator it=s.begin();
	while(it!=s.end())
	{
		if(*it==ret)
		{
			it++;
			ret++;
		}
		else
		break;
	}
	grundyvalue[x]=ret;
	return ret;
}
int main()
{
	for(int i=1;i<8;i++)
	{
		pick[i-1]=pow(i,i);
	}
	for(int j=0;j<=100000;j++)
	{
		grundyvalue[j]=grundy(j);
	}
	int t, n, x, ans;
	cin>>t;
	while(t--)
	{
		ans=0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>x;
			ans= ans^grundy(x);
		}
		if(ans==0)
		cout<<"Head Chef\n";
		else
		cout<<"Little Chef\n";
	}
	return 0;
}
