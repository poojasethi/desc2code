#include <iostream>
#include <cstdio>
using namespace std;

int grundy[100005];
int arr[6];

void init(){
	arr[0] = 1;
	arr[1] = 4;
	arr[2] = 27;
	arr[3] = 256;
	arr[4] = 5*5*5*5*5;
	arr[5] = 6*6*6*6*6*6;
}

void grundycal(){
	
	grundy[0] = 0;
	int gg;
	for(int i=1;i<=100000;i++){
		
		for(int j=0;j<=6;j++){
			gg = -1;
			for(int k=0;k<6;k++){
				if(arr[k]>i)
					break;
				if(grundy[i-arr[k]]==j){
					gg = j;
					break;
				}
			}
			if(gg==-1){
				gg = j;
				break;
			}
		}
		grundy[i] = gg;
	}
	
}

int t,n,a,xr;

int main() {
	
	init();
	grundycal();
	
	scanf("%d",&t);
	
	while(t--){
		scanf("%d",&n);
		xr = 0;
		for(int i=0;i<n;i++){
			scanf("%d",&a);
			xr ^= grundy[a];
		}
		
		if(!xr)
			printf("Head Chef\n");
		else
			printf("Little Chef\n");
	}
	
	return 0;
}
