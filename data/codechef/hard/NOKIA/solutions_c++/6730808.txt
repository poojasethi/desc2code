#include<stdio.h>
#include<iostream>
#include<algorithm>
using namespace std;
int dp[31];
int solve(int idx)
{
	if(idx==1)
		return 2;
	else if(idx==2)
		return 5;
	else if(dp[idx]!=-1)
		return dp[idx];
	else
	{
		int mid=(idx+1)/2;
		dp[idx]= solve(mid-1)+solve(idx-mid)+idx+1;
        return dp[idx];
	}
}
int main()
{
	int tc;
	scanf("%d",&tc);
	for(int i=0;i<=30;i++)
			dp[i]=-1;
	while(tc--)
	{
		int n,d;
		scanf("%d%d",&n,&d);
		int maxi=n*(n+3)/2;
		int mini=solve(n);
		if(d<=maxi && d>=mini)
			printf("0\n");
		else if(d<mini)
			printf("-1\n");
		else
			printf("%d\n",d-maxi);
	}
	return 0;
}
