#include<cstdio>
#include<cstring>
#include<algorithm>
#define INF 1000000000
using namespace std;
typedef long long int ll;
int main(){
    int min_[1001],max_[1001];
    min_[0]=0;
    max_[0]=0;
    for(int i=1;i<=1000;i++){
        min_[i]=(i+1) + min_[i/2] + min_[i-1-(i/2)];
        max_[i]=(i+1)+max_[i-1];
    }
    int t;
    scanf("%d",&t);
    while(t--){
        int n,m;
        scanf("%d %d",&n,&m);
        if(max_[n]<=m){
            printf("%d\n",m-max_[n]);
        }
        else if(min_[n]<=m){
            printf("0\n");
        }
        else
        {
            printf("-1\n");
        }
        }
}
