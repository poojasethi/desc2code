#include <bits/stdc++.h>
#define inf 2000000000

using namespace std;

int n, m;
int d[31], f[31], g[31];

void calc(int n) {
    if (n==0) return;
    if (d[n]) return;
    d[n]=1;
    f[n]=inf;
    g[n]=-inf;
    for(int i=1;i<=n;i++) {
        calc(i-1);
        calc(n-i);
        f[n]=min(f[n],f[i-1]+f[n-i]);
        g[n]=max(g[n],g[i-1]+g[n-i]);
    }
    f[n]+=(n+1);
    g[n]+=(n+1);
}

int main() {
    //freopen("inp.txt","r",stdin);
    int T; scanf("%d",&T);
    for(int t=1;t<=T;t++) {
        int n, m; scanf("%d%d",&n,&m);
        memset(d,0,sizeof(d));
        calc(n);
        if (m<f[n]) printf("-1\n"); else {
            if (m>=g[n]) printf("%d\n",m-g[n]);
            else printf("0\n");
        }
    }
}
