#include<iostream>
 
using namespace std;
 
int minimum_val(int n){
if(n<=0) return 0;
else if(n==1) return 2;
else return (n+1+minimum_val(n/2)+minimum_val(n-1-n/2));
}
 
int main(){
int t;
cin>>t;
while(t--){
    int n,m;
    cin>>n>>m;
    int max,min;
    max=n*(n+3)/2;
    min=minimum_val(n);
    if(min>m){
        cout<<"-1\n";
    }
    else{
        if(max>=m) cout<<"0\n";
        else cout<<(m-max)<<"\n";
    }
}
return 0;
}
