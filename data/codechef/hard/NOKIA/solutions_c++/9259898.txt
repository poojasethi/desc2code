#include <iostream>

using namespace std;
int minl(int n)
{
    int f,l;
    if(n==0)
    {
        return 0;
    }
    else
    {
        if(n%2==0)
        {
            f=(n/2)-1;
            l=n/2;
        }
        else
        {
            f=l=n/2;
        }
    }
    return (n+1+minl(f)+minl(l));
}
int main()
{
    int tc,n,m,m1,m2;
    cin>>tc;
    while(tc--)
    {
        cin>>n>>m;
        m1=minl(n);
        m2=(n*(n+3))/2;
        if(m<m1)
        {
            cout<<"-1"<<endl;
        }
        else
        {
            if(m1<=m && m2>=m)
            {
                cout<<"0"<<endl;
            }
            else
            {
                cout<<m-m2<<endl;
            }
        }
    }
    return 0;
}
