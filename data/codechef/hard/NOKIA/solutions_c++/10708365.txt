#include <iostream>

using namespace std;

const int MAX_N = 30;

class Knowledge
{
public:
    void init()
    {
        min[0] = max[0] = 0;
        for (int i = 1; i <= MAX_N; ++i)
        {
            max[i] = i+1 + max[i-1];
            min[i] = i+1 + min[i/2] + min[(i-1)/2];
        }
    }
    
    int getMax(int n) const {
        return max[n];
    }
    
    int getMin(int n) const {
        return min[n];
    }
    
private:
    int min[MAX_N+1];
    int max[MAX_N+1];
};

int main()
{
    Knowledge k; k.init();
    int T; cin >> T;
    while (T--)
    {
        int N, M;
        cin >> N >> M;
        if (k.getMin(N) > M)
            cout << -1 << endl;
        else if (k.getMax(N) >= M)
            cout << 0 << endl;
        else
            cout << M - k.getMax(N) << endl;            
    }
    return 0;
}
