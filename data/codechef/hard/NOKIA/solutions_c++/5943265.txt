#include<iostream>
using namespace std;
#include<cstdio>
int solve(int n)
{
    int ans=n+1;
    if(n==0) return 0;
    if(n==1) return 2;
    if(n%2==0)
    {
        ans+=solve(n/2);
        ans+=solve(n/2-1);
    }
    else
    {
        ans+=2*solve(n/2);
    }
    return ans;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,m;
        scanf("%d%d",&n,&m);
        int mn=solve(n);
        int mx=n*(n+3)/2;
        if(m<mn) printf("-1\n");
        else if(m>mx) printf("%d\n",m-mx);
        else printf("0\n");
    }
    return 0;
}
