#include<bits/stdc++.h>
using namespace std;
typedef long long lint;
lint min_wire(lint n)
{
    lint res=n+1;
    if(n==0)
        return 0;
    if(n==1)
        return 2;
    if(n&1)
        res+=2*min_wire(n/2);
    else
        res+=min_wire(n/2)+min_wire(n/2-1);
    return res;

}
void solve(lint n, lint m)
{
    lint minm, maxm, ans;
    maxm=((n+1)*(n+2))/2 -1;
    minm=min_wire(n);
    //cout<<minm<<"pr"<<endl;
    if(m>=minm && m<=maxm)
        ans=0;
    else if(m>maxm)
        ans=m-maxm;
    else
        ans=-1;
    printf("%lld\n", ans);
}
int main()
{
    int test;
    scanf("%d", &test);
    while(test--)
    {
        lint n, m;
        scanf("%lld %lld", &n, &m);
        solve(n, m);
    }
    return 0;
}
