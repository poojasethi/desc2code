#include<iostream>
using namespace std;
int mini[32], maxi[32];
main () {
    mini[1]=2; mini[2]=5;
    for (int i=3; i<=31; i++) mini[i]=mini[i/2]+mini[(i-1)/2]+i+1;
    for (int i=1; i<=31; i++) maxi[i]=((i+2)*(i+1))/2-1;
    int t;
    cin>>t;
    while (t--) {
        int n, m;
        cin>>n>>m;
        if (m>=mini[n] && m<=maxi[n]) cout<<0<<endl;
        else if (m<mini[n]) cout<<-1<<endl;
        else cout<<m-maxi[n]<<endl;
    }
}
