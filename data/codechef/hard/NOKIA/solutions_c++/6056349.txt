using namespace std;
int findans(int n)
{
    if(n==1)
        return 2;
        else if(n==0)
            return 0;
        else
        {
            int ans;
            if(n%2==0)
                ans=n+1;
            else
                ans=((n/2)+1)*2;
                return ans+findans(n/2)+findans(n-n/2-1);
        }
}
 int main()
 {
long long int t,d,k,max1,min1,n;
scanf("%lld",&t);
while(t>0)
{
    t--;
    scanf("%lld %lld",&n,&k);
    max1=(((n)*(n+1))/2)+(n);
    if(k>=max1)
    {
        printf("%lld\n",k-max1);
    }
    else
    {
        min1=findans(n);
       // cout<<min1<<endl;

            printf("-1\n");
        else

    }
}
}
