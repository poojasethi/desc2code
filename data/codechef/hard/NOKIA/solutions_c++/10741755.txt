//<---- Solution ---->
ll minN(int N){
	if(N==0)
		return 0;
	if(N==1)
		return 2;
	if(N==2)
		return 5;
	else {
		return N+1+minN(N/2)+minN(N-1-N/2);
	}
}
void solve(){
	int n,m;
	inpp(n,m);
	int maxi = n*(n+3)/2;
	int mini = minN(n);
	//DEBUG(mini);
	if(mini<=m && m<=maxi)
		op(0);
	else if(m<mini)
		op(-1);
	else if(m>maxi)
		op(m-maxi);
}

int main(){
	int tc;
	inp(tc);
	while(tc--)
		solve();
	return 0;
}
