#include <iostream>
using namespace std;

int minFind(int n)
{
	if(n==2)
		return 5;
	if(n==1)
		return 2;
	return(n+1+minFind(n/2)+minFind(n-n/2-1));
}

int main()
{
	int i,t,m,n,max,min,output;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>n>>m;
		max=(n+1)*(n+2)/2-1;
		min=minFind(n);
		//cout<<max<<"**"<<min<<endl;
		if(m>max)
			output=m-max;
		else if(m<min)
			output=-1;
		else
			output=0;
		cout<<output<<endl;
	}
	return 0;
}
