#include<bits/stdc++.h>

using namespace std;

int minq[51],maxq[51];

int mina(int n)
{
	if(n==0)
	return 0;
	
	if(minq[n]!=0)
	return minq[n];
	
	return mina(n/2) + mina(n - n/2 - 1) + n+1;	
}

int maxa(int n)
{
	if(n==0)
	return 0;
	
	if(maxq[n]!=0)
	return maxq[n];
	
	return maxa(n-1) + n+1;	
}


int main()
{
    int i,j,k,l,n,m,t;
   
    memset(minq,0,sizeof(minq));
 	memset(maxq,0,sizeof(maxq)); 	
	 cin>>t;
	 
 	while(t--)
 	{
 		cin>>n;
 		cin>>m;
 	    minq[n]=mina(n);
		maxq[n]=maxa(n);	
 		//cout<<minq[n]<<"  "<<maxq[n]<<endl;
 	    if(m<minq[n])
 	    cout<<"-1"<<endl;
 	    else if(m<=maxq[n])
 	    cout<<"0"<<endl;
 	    else
 	    cout<<m-maxq[n]<<endl;
	 }
 
 
 	
}
