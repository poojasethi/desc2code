#include <iostream>
#include <cstdio>
#include <climits>
#include <algorithm>
#include <queue>
#include <cstring>
#include <cmath>
using namespace std;
int mini[1009];
void preprocess()
{
	int i,j,k,l;
	mini[0]=0;
	mini[1]=2;
	for(i=2;i<=1000;i++)
	{
		mini[i]=mini[(i/2+1)-1]+mini[i-(i/2+1)]+i+1;
	}
}
int main()
{
	preprocess();
	int t,i,j,k,n;
	//freopen("example.txt","r",stdin);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		int x=(n*(n+1))/2+n;
		if(k<mini[n])
			printf("-1\n");
		else if(k>x)
			printf("%d\n",k-x);
		else
			printf("0\n");
	}
	return 0;
}
