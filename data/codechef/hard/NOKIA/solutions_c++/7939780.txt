#include<iostream>
#include<cstdio>

int main()
{

    int t,a[50],b,c,i,j,k;
    scanf("%d",&t);
    a[1]=2;
    a[0]=0;
    for(i=2;i<33;i++)
    {
        if(i%2==0)
        {
            b=i/2;
            c=b-1;
        }
        else
        {

            b=i/2;
            c=b;
        }
        a[i]=i+1+a[b]+a[c];
    }

    while(t--)
    {
        scanf("%d%d",&c,&b);
        int d=((c+1)*(c+2))/2;
        d=d-1;
        if(b<=d && b>=a[c])
            printf("0\n");
        else if(b>=d)
            printf("%d\n",b-d);
        else
            printf("-1\n");
    }
}
