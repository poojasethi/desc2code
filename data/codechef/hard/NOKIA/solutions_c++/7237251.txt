#include<stdio.h>


int memo[1001][31];
int rec(int n,int m)
{
    int a,b;
    if(m<0)
        return 0;
    if(m==0&&n!=0)
        return 0;
    if(n==0&&m!=0)
        return 0;
    if(memo[m][n]!=-1)
        return memo[m][n];
    if(n==0&&m==0)
        return 1;
    for(int i=0;i<n;i++)
    {
        for(int k=0;k<=m-n-1;k++)
        {
            a=rec(i,k);
            b=rec(n-i-1,m-n-1-k);
            if(a&&b)
            {
                memo[m][n]=1;
                return 1;
            }
        }
    }
    return 0;
}
