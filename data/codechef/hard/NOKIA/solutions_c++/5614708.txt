#include <bits/stdc++.h>
using namespace std;
int main() 
{
	int minn[35],maxx[35];
	minn[0]=0;
	maxx[0]=0;
	for(int yu=1;yu<=35;yu++)
	{
		minn[yu]=minn[yu/2]+minn[yu-1-yu/2]+yu+1;
		maxx[yu]=(yu*(yu+3))/2;
	}
	int t,i,j,k,l,n,m;
	cin>>t;
	while(t--)
	{
		cin>>n>>m;
		if(m<minn[n])
			cout<<"-1"<<endl;
		else if(m>maxx[n])
			cout<<(m-maxx[n])<<endl;
		else
			cout<<"0"<<endl;
	}
	return 0;
}
