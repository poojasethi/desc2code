int ans(int a,int b)
{
	if(b-a==2)
		return 2;
	if(b-a<2)
		return 0;
	return b-a + ans(a,(a+b)/2)+ans((a+b)/2,b);
}

void solve()
{
	int n,m,mx,mn,temp;
	scanf("%d%d",&n,&m);
	mx=n*(n+1)/2+n;
	mn=ans(0,n+1);
	//printf("%d %d\n",mn,mx);
	if(m<mn)
		temp=-1;
	else if(m>mx)
		temp=m-mx;
	else
		temp=0;
	printf("%d\n",temp);
}
