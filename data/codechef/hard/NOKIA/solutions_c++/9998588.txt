#include<bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k;
    int n,m,t;
    int mx,mn;

    vector<set<int> > a(31);

    set<int> temp;
    temp.insert(0);
    a[0]=temp;

    temp.clear();
    temp.insert(2);
    a[1]=temp;

    set<int> :: iterator it,it2;

    for(i=2;i<31;i++)
    {
        j=((i-1)/2);
        
        temp.clear();
        while(j>=0)
        {
            k=(i-1-j);

            //add all the values
            for(it=a[j].begin();it!=a[j].end();it++)
            {
                for(it2=a[k].begin();it2!=a[k].end();it2++)
                {
                    temp.insert(i+1+(*it)+(*it2));
                }
            }

            j--;
        }
        
        a[i]=temp;

        /*
        cout<<i<<" : ";

        for(it=a[i].begin();it!=a[i].end();it++)
        {
            cout<<(*it)<<" ";
        }
    
        cout<<endl;

        getchar();
        */
    }

    cin>>t;

    while(t--)
    {
        cin>>n>>m;
    
        //find mx and min 
        if(m<*(a[n].begin()))
        {
            cout<<-1;
        }
        
        else
        {
            //now, find
            it=a[n].begin();
            for(;it!=a[n].end() && (*it)<=m;it++);

            it--;

            mn=(m-(*it));
            cout<<mn;
        }


        cout<<endl;
    }

    return 0;
}
