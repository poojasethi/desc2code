#include<iostream>
using namespace std;
int min(int n)
{
	if (n==0)
		return 0;
	if(n==1)
	return 2;
	else 
	return n+1+min(n/2)+min(n-n/2-1); 
}
int main()
{
	int t;
	cin>>t;
	for(int i=0;i<t;i++)
	{	int n,k;
		cin>>n>>k;
		int maxlen=(n+1)*(n+2)/2-1;
		//cout<<sum<<endl;
		//int x=k-sum+1;
		int minlen=min(n);
		//cout<<minlen<<" "<<maxlen<<endl;

		if(minlen<=k&&k<maxlen)
		{
			cout<<0<<endl;
			continue;
		}
		else
		{
			if(k>=maxlen)
				cout<<k-maxlen<<endl;
			else
				cout<<-1<<endl;
			continue;
		}
	}
}
