#include <iostream>
#include <stdio.h>
#include <limits.h>
#include <math.h>
#include <stdlib.h>
using namespace std;
int max_array[35][35],min_array[35][35];
int main()
{
    int t,i,j,k,l,n,m,maxi,qmax,qmin,mini;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&m);
        for(i=0;i<=n;i++)
        {
            max_array[i][i+1]=0;
            min_array[i][i+1]=0;
        }
        for(l=2;l<=n+1;l++)
        {
            for(i=0;i<=n-l+1;i++)
            {
                maxi=INT_MIN;
                mini=INT_MAX;
                j=i+l;
                for(k=i+1;k<j;k++)
                {
                    qmax=max_array[i][k]+max_array[k][j]+j-i;
                    qmin=min_array[i][k]+min_array[k][j]+j-i;
                    if(qmax>maxi)
                        maxi=qmax;
                    if(qmin<mini)
                        mini=qmin;
                }
                max_array[i][j]=maxi;
                min_array[i][j]=mini;

            }
        }
        //printf("%d %d\n",min_array[0][n+1],max_array[0][n+1]);
        if(m>max_array[0][n+1])
            printf("%d\n",m-max_array[0][n+1]);
        else if(m>=min_array[0][n+1]&&m<=max_array[0][n+1])
            printf("0\n");
        else
            printf("-1\n");

    }
    return 0;
}
