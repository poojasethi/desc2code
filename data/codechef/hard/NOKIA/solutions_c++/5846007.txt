	#include<iostream>
	using namespace std;
	long long int func(int lower,int upper)
	{
		long long int min=0;
		int mid=(lower+upper)/2;	
		//cout<<"lower="<<lower<<endl;
		//cout<<"mid="<<mid<<endl;
		if(lower+1<upper)
		{
			
			min=func(lower,mid)+func(mid,upper)+upper-lower;
		}
		return min;
	}
	
	
	
	int main()
	{
		int t;
		cin>>t;
		while(t>0)
		{
			int n,m;
			cin>>n;
			cin>>m;
			long long int max=(n*(n+3))/2;
			long long int min;
			min=func(0,n+1);
		//	cout<<"min="<<min<<endl;
			if(m<min)
				cout<<"-1\n";
			else if(max<m)
				cout<<m-max<<endl;
			else
				cout<<"0\n";
			t--;
		
		}
	}
