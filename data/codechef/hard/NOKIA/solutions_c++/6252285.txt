#include<iostream>
using namespace std;

int func(int beg,int last)
{
if(last-beg<=1)
return 0;
if(last-beg==2)
return 2;
int mid=(beg+last)/2;
return last-beg+func(beg,mid)+func(mid,last);
}

int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
int n,m;
cin>>n>>m;
int wire1=func(0,n+1),wire2=(n+1)*(n+2)/2-1;
//cout<<wire<<endl;
if(m<wire1)
cout<<"-1"<<endl;
else if(m>=wire1&&m<=wire2)
cout<<"0"<<endl;
else
cout<<m-wire2<<endl;

}
return 0;
}
