#include <iostream>
#include <stdio.h>
#include <math.h>
#include <algorithm>
using namespace std;
int main(){
	int minz[31],maxz[31];
	minz[0]=0;maxz[0]=0;
	for(int i=1;i<=30;i++){
		minz[i]=minz[i/2] + minz[i-1-i/2]+i+1;
		maxz[i]=maxz[i-1]+i+1;
	}
int t;
scanf("%d",&t);
while(t--){
		int n,m;
		scanf("%d %d",&n,&m);
		if(minz[n]>m) printf("-1\n");
		else if(maxz[n]>=m) printf("0\n");
		else if(maxz[n]<m) printf("%d\n",m-maxz[n]); 
}
}
