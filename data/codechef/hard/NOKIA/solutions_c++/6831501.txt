#include<cstdio>
#include<algorithm>
#include<iostream>
#define in(a)  scanf("%d",&a)
using namespace std;

int calmin(int n)
{
	if(n==0)	
		return 0;
	if(n==1)
		return 2;
	
	int ans=(n+1) + calmin(n/2) + calmin(n-1-(n/2));
	
	return ans;
}

int calmax(int n)
{
	if(n==0)
		return 0;
	if(n==1)
		return 2;
	
	int ans=(n+1) + calmax(n-1);
	
	return ans;
}

int main()
{
	int t,n,l,minv,maxv;
	in(t);
	while(t--)
	{
		in(n);	in(l);
		
		minv=calmin(n);
		maxv=calmax(n);
		
		if(minv>l)
			{	cout<<"-1\n";   continue;	}
		
		if(maxv>=l)
			cout<<"0\n";
		else 
			cout<<l-maxv<<endl;
		
	}
	
	return 0;
}
