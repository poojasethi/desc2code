#include <cstdio>

inline int max(int n)
{
	return (n * (n + 3)) / 2;
}

int main()
{
	int T, N, M, i;
	int W[31];
	W[0] = 0;
	W[1] = 2;
	W[2] = 5;
	for (i = 3; i <= 30; i++)
		W[i] = i + 1 + W[i / 2] + W[i / 2 - (i + 1) % 2];
	scanf("%d", &T);
	for (i = 0; i < T; i++)
	{
		scanf("%d %d", &N, &M);
		if (M < W[N])
			printf("%d\n", -1);
		else if (M < max(N))
			printf("%d\n", 0);
		else
			printf("%d\n", M - max(N));
	}
	return 0;
}
