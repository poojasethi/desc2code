#include <cstdio>
#include <cstring>
#include <iostream>
#include <vector>
#include <algorithm>
#define MAX 32
using namespace std;

int F[MAX];

int get_min(int a, int b){
	if(b - a == 1) return 0;
	int med = (a + b) >> 1;
	return get_min(a,med) + get_min(med,b) + (b - a);
}


int main(){
	int t,n,m;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d",&n,&m);
		int maxi = ((n + 1)*(n + 2))/2 -1;
		int mini = get_min(0,n + 1);
		if(m < mini) printf("-1\n");
		else
		{
			if(m > maxi) printf("%d\n",m - maxi);
			else printf("0\n");
		}
	}
	return 0;
}
