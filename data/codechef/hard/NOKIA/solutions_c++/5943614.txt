#include<iostream>
#include<cstdio>
using namespace std;
int minLen(int val)
{
    if(val==1)
    return 0;
    else{
         int half=val/2;
         return half+(val-half)+minLen(half)+minLen(val-half);
    }
}
int maxLen(int len)
{
    if(len==1)
    return 0;
    else{
         return 1+(len-1)+maxLen(len-1);
    }
}
int main()
{
    int N,M,T;
    cin>>T;
    while(T--) {
        cin>>N>>M;
        int min1 = minLen(N+1);
        int max1 = maxLen(N+1);
        if(M < min1) {
            cout<<"-1"<<endl;
        }
        else if(M >= min1 && M <= max1) {
            cout<<"0"<<endl;

        }
        else {
            cout<<M - max1<<endl;
        }
    }
    getchar();
    return 0;
}
