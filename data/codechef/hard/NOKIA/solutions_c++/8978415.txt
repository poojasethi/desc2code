#include<stdio.h>
int min[31]={0},max[31]={0};
int rec(int n){
	if(min[n]){
		return min[n];
	}
	else{
		if(n%2==0){
			min[n]=n+1+rec(n/2)+rec((n/2) -1);
		}
		else{
			min[n]=n+1+(2*rec((n-1)/2));
		}
		return min[n];
	}
}
int main(){
	min[0]=0;max[0]=0;
	min[1]=2;max[1]=2;
	min[2]=5;max[2]=5;
	min[3]=8;max[3]=9;
	int i,n,t,m;
	for(i=4;i<=30;i++){
		max[i]=i+1+max[i-1];
		//printf("%d ",max[i]);
	}
	scanf("%d",&t);
	while(t--){
		scanf("%d%d",&n,&m);
		if(m>=max[n]){
			printf("%d\n",m-max[n]);
		}
		else if(m<rec(n)){
			printf("-1\n");
		}
		else {
			printf("0\n");
		}
		
	}
	
	return 0;
}
