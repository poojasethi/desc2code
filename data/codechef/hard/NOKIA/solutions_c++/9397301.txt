#include <iostream>
#include <cstdio>
using namespace std;

int t,m,n,l,helper[30],max;

int min(int n){
    if(n==0){
        return 0;
    }
    else if(helper[n]!=0){
        return helper[n];
    }
    else{
        helper[n]=min(n/2)+min(n-1-(n/2))+n+1;
        return helper[n];
    }
}

int main()
{

    scanf("%d",&t);
    while(t--){
        scanf("%d %d",&n,&m);
        l=min(n);
        int max=(n*(n+3))/2;
        if(max<m){
            printf("%d\n",(m-max));
        }
        else if(l>m){
            printf("-1\n");
        }
        else
            printf("0\n");
    }
    return 0;
}
