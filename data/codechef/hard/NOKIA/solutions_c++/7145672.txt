#include<stdio.h>

using namespace std;

int main()
{
	int T,N,M;
	scanf("%d",&T);
	int minmax[30][2];
	minmax[0][0]=2;minmax[0][1]=2;
	minmax[1][0]=5;minmax[1][1]=5;
	for(int i=3;i<=30;i++)
	{
		minmax[i-1][0]=i+1+minmax[(i/2)-1][0]+minmax[((i-1)/2)-1][0];
		minmax[i-1][1]=i+1+minmax[i-2][1];
	}
	
	while(T>0)
	{
		scanf("%d %d",&N,&M);
		int m1=minmax[N-1][0],m2=minmax[N-1][1];
		if(M>=m2)
		{
			printf("%d\n",M-m2);
		}
		else if (M<m1)
		{
			printf("-1\n");
		}
		else
		{
			printf("0\n");
		}
		T--;
	}
}
