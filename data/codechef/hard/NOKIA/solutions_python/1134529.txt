MAXN = 30

hash = [[2,2] for i in range(MAXN+1)]
hash[0] = [0,0]

for i in range(2, MAXN + 1):
	hash[i][0] = (i+1) + hash[i/2][0] + hash[i-i/2 - 1][0]
	hash[i][1] = (i+1) + hash[i-1][1]

t = int(raw_input())

for i in range(t):
	(n,m) = tuple(map(int, str(raw_input()).split(' ')))

	if m>=hash[n][0] and m<=hash[n][1]:
		print(0)
	elif m>hash[n][1]:
		print(m - hash[n][1])
	else:
		print(-1)
