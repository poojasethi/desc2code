def compute_max(N):
	return ((N+1)*(N+2)/2 - 1)

def compute_min(N):
	if (N==0):
		return 0
	if (N==1):
		return 2
	else:
		if N%2==0:
			return (N+1) + compute_min((N-1)/2) + compute_min(N/2)
		else:
			return (N+1) + 2*compute_min(N/2)

T = int(raw_input())
for t in range(T):
	[N, M] = [int (i) for i in raw_input().split()]
	maxi = compute_max(N)
	mini = compute_min(N)
	if M > (maxi):
		print M-maxi
	elif M >= mini:
		print 0
	else:
		print -1
