#include<bits/stdc++.h>
using namespace std;
bool ispal(char *s ,int start, int end){
	for(int i=start,j = end ; i<=(start+end)/2; i++,j--){
		if(s[i]!= s[j]) return false;
	}
	return true;
}
int main()
{
    int t,i,j,n,a[5002];
    char str[5002];
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        scanf("%s",&str);
        a[0]=1;
       for(i=1;i<n;i++)
       {
           a[i]=a[i-1]+1;
           for(j=1;j<i;j++)
           {
               if(ispal(str,j,i))
                {
                    a[i]=min(a[i],a[j-1]+1);
                    
                }
           }
           if(ispal(str,0,i))
            a[i]=1;
       }

        printf("%d\n",a[n-1]);
    }
    return 0;
}
