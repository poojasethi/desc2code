#include <string.h>
#include <stdio.h>
#include <iostream>
#include <map>
#include <vector>
#include <algorithm>
 
#define NMAX 5001
 
#define min(a,b) ((a) <= (b) ? (a) : (b))
 
using namespace std;
 
int main () {
 
	vector <int> r[NMAX];
	
	int m, n, p[NMAX], i, j, t, tst, lim, vol, sum, k;
	char v[NMAX];
	int pmin, pmax = NMAX * NMAX;
	
	scanf("%d", &tst);
	
	for (t = 0; t < tst; t++) {
	
		scanf ("%d ", &n);
		
		for (i = 0; i<n; i++) {
			scanf("%c", &v[i]);
			//printf("lit = %c\n", v[i]);
			r[i].clear();
			p[i] = 1;
			r[i].push_back(i);
	       
	       	if (i) {
	       		
				pmin = p[i-1] + 1;
				
				if (v[i - 1] == v[i]) {
						r[i].push_back(i - 1);
						if (i > 1)
							pmin = min(pmin, p[i-2]+1);
						else
							pmin = 1;	
				}
				m = r[i-1].size();
				//printf("m = %d i = %d\n", m, i-1);
				for (j = 0; j < m; j++) {
					k = r[i-1][j];
					if (k && v[k - 1] == v[i]) {
						r[i].push_back(k - 1);
						if (k > 1)
							pmin = min(pmin, p[k-2]+1);
						else
							pmin = 1;
					}
				}
				p[i] = pmin;
	       	}	
		}
		
		printf("%d\n", p[n-1]);
		//for (i = 0; i<n; i++) 
			//printf("%d ", p[i]);
		
	
	} 
	
	return 0;
}
