#include<bits/stdc++.h>
using namespace std;

const int N=5005;
int dp[N],check[N][N];
char s[N];
int main() {
    int t;scanf("%d",&t);
    while(t--) {
        int n;
        scanf("%d",&n);
        scanf("%s",s);
        memset(check,0,sizeof(check));
        memset(dp,0,sizeof(dp));
        for(int i=0;i<n;i++) {
            dp[i]=n-i;
        }
        for(int i=n-1;i>=0;i--) {
            for(int j=i;j<n;j++) {
                if(s[i]==s[j] && (j-i<2 ||check[i+1][j-1]==1)){
                    check[i][j]=1;
                    dp[i]=min(dp[i],1+dp[j+1]);
                }
            }
        }
        printf("%d\n",dp[0]);
    }
}
