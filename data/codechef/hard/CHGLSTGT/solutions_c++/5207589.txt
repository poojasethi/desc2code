#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int min(int a, int b){
	return (a<b)?a:b;
}

bool checkPalindrome(char *s, int start, int end){
	for(int i=start,j = end ; i<=(start+end)/2; i++,j--){
		if(s[i]!= s[j]) return false;
	}
	return true;
}

int main(){
	int te;
	scanf("%d",&te);
	while(te--){
		int n;
		scanf("%d",&n);
		char s[n];
		scanf("%s",s);
		int a[n];
		a[0] = 1;
		for(int i=1; i<n; i++)
		{
			a[i] = a[i-1]+1; //not the part of any palindrome;
			for(int j = 1; j<i; j++){
				if(checkPalindrome(s, j, i)){
					a[i] = min(a[i], 1+a[j-1]);
					//break;
				}
			}
			if(checkPalindrome(s, 0, i)) a[i] = 1;
		}
		printf("%d\n", a[n-1]);
	}
	return 0;
}
