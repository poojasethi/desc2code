#include <bits/stdc++.h>
using namespace std;
string s;
int dp[5001];
int checkPalin[5001][5001];
int rec(int i){
    if(i==s.size())
        return 0;
    if(dp[i]!=-1)
        return dp[i];
    dp[i]=1000000000;
    for(int j=i;j<s.size();j++){
        if(checkPalin[i][j]==1)
            dp[i]=min(dp[i],1+rec(j+1));
    }
    return dp[i];
}
int main(){
    int t;
    cin>>t;
    int n;
    while(t--){
        cin>>n;
        cin>>s;
        memset(dp,-1,sizeof dp);
        memset(checkPalin,0,sizeof checkPalin);
        for(int i=0;i<s.size();i++)
            checkPalin[i][i]=1;
        for(int i=2;i<=s.size();i++){
            for(int j=0;j<s.size()-i+1;j++){
                if(i==2){
                    if(s[j]==s[i+j-1])
                        checkPalin[j][i+j-1]=1;
                }
                else{
                    if(s[j]==s[i+j-1] and checkPalin[j+1][i+j-2]==1)
                        checkPalin[j][i+j-1]=1;
                }
            }
        }
        cout<<rec(0)<<endl;
    }
    return 0;
}
