		for(int j = 0; j < n; ++j) tot[i][j]=0;

	int l, r, i;
	for (i = 0; i < n; i++){
        l = r = i;
        while (l >= 0 && r < n && s[l] == s[r]){
            tot[l][r] = true;
            l--;

        }
        l = i - 1;

void dp(){	 
	int mini[n+1];
	for(int i = 0; i <= n; mini[i]=5001, ++i);
	mini[0]=0;
	mini[1]=1;
	for(int i = 1; i < n; ++i){
		for(int j = 0; j <= i; ++j)
			if(tot[i-j][i]) mini[i+1]=min(mini[i+1], mini[i-j]+1);
	}
	printf("%d\n", mini[n]);
}
int main(int argc, char const *argv[]){
