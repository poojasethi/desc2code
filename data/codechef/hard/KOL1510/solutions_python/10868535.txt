#define mt 18

vec_<bool>arr[mt];
int n;
string st,ans;

void go(int i)
{
    repa(ch,st)
    if(ch == 'B')
        arr[i].pb(0);
    else
        arr[i].pb(1);
}

void get_ans(int mask,int len)
{
    for(int i=len-1;i>=0;i--)
    if(mask &(1<<i))
        ans+='G';
    else
        ans+='B';
}

bool check2(int mask,int len,int i)
{
//    deb("yo");
    int len2=sz(arr[i]),j=0,k=0;
//    deb(len2);
//    deb("hola");
    k=len-1;
    while(j<len2 && k>=0)
    {
        if((mask & (1<<k)) && arr[i][j])
        {
            k--;
            j++;
        }
        else if(!(mask & (1<<k)) && !arr[i][j])
        {
            k--;
            j++;
        }
        else
            k--;
    }
    return j==len2;
}

bool check(int mask,int len)
{
    rep(i,n)
    if(!check2(mask,len,i))
        return 0;
    return 1;
}
