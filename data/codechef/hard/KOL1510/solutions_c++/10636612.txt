#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
#define sci(fd) scanf("%d",&fd)
#define scll(fd) scanf("%lld",&fd)
#define pb push_back
#define mp make_pair
#define MOD 1000000007
#define PI 3.1415926535897932
#define pii pair < int,int > 
#define fi first
#define se second
const ll infi=1000000000000000009;
char str[20][12];
int main()
{
    int t;
    sci(t);
    while(t--)
    {
        int i,k,j,n,len=0;
        sci(n);
        for(i=0;i<n;i++)
        {
            scanf("%s",str[i]);
            int le=strlen(str[i]);
            len=max(le,len);
        }
        int tot=pow(2,2*len),ans=30,ind=-1;
        int st[40];
        for(i=0;i<tot;i++)
        {
            for(j=0;j<2*len;j++)
            {
                if(((1<<j)&i)==0)
                st[j]='B';
                else
                st[j]='G';
            }
            int maxlen=0;
            for(j=0;j<n;j++)
            {
                int itt=0;
                for(k=0;k<2*len;k++)
                {
                    if(str[j][itt]==st[k])
                    itt++;
                    if(str[j][itt]=='\0')
                    break;
                }
                if(str[j][itt]!='\0')
                break;
                maxlen=max(k+1,maxlen);
            }
            if(j!=n)
            continue;
            if(maxlen<ans)
            {
                ans=maxlen;
                ind=i;
            }
        }
        for(j=0;j<ans;j++)
        {
            if(((1<<j)&ind)==0)
            printf("B");
            else
            printf("G");
        }
        printf("\n");
    }
    return 0;
}
