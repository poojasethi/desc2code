void sieve()
{
    long long i,j,a[41100]={0};
    for(i=2;i<=41000;i++)
        if(!a[i])
            for(j=i*i;j<=41000;j+=i)
                a[j]=1;
    for(i=2;i<=41000;i++)
        if(!a[i])
            prime.push_back(i);
}

void initp(int p)
{
    num=0;
    for(LL i=2;i*i<=p;i++)
    {
        if(p%i==0)
        {
            pf[num++]=i;
            while(p%i==0) p/=i;
        }
    }
    if(p>1)
        pf[num++]=p;
}
LL totient(LL m)
{
    initp(m);
    LL x = 1,y = m;
    for(int i=0;i<num;i++)
    {
        x *= pf[i]-1;
        m /= pf[i];
    }
    return x*m;
}
LL quitpow(LL n,LL m, LL mod)
{
    LL ans=1,k=n;
    if(n==1) return 1;
    while(m)
    {
        if(m&1) ans=(ans*k)%mod;
        m>>=1;
        k=(k*k)%mod;
    }
    return ans;
}
