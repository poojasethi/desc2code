#include <stdio.h>
#include <string.h>
#include <stack>
#include <iostream>
using namespace std;
stack <int> box;
int connections[26][26];
int arr[26];
int check()
{
    int i,j;
    while(box.empty()==false)
    {
        i=box.top();
        box.pop();
        if(arr[i]==0)
            continue;
        arr[i]=0;
        for(j=0;j<26;j++)
        {
            if(connections[i][j]==1)
            {
                if(arr[j]==1)
                    box.push(j);
            }
        }
    }
    for(i=0;i<26;i++)
    {
        if(arr[i]==1)
            return 0;
    }
    return 1;
}
int connected()
{
    int i,j;
    for(i=0;i<26;i++)
    {
        for(j=0;j<26;j++)
        {
            if(connections[i][j]==1)
                break;
        }
        if(connections[i][j]==1)
                break;
    }
    if(i<26)
    {
        box.push(i);
        return check();
    }
    else
        return 1;
}
int main()
{
    int t,n,i,j,a,b,starts,ends,temp;
    int in[26]={0}, out[26]={0};
    char str[1001];
    scanf("%d",&t);
    while(t--)
    {
        for(i=0;i<26;i++)
        {
            for(j=0;j<26;j++)
            {
                connections[i][j]=0;
            }
            arr[i]=0;
            in[i]=0;
            out[i]=0;
        }
        scanf("%d",&n);
        i=0;
        while(i<n)
        {
            scanf("%s",str);
            temp=strlen(str)-1;
            a=str[0]-97;
            b=str[temp]-97;
            in[a]++;
            out[b]++;
            connections[a][b]=1;
            connections[b][a]=1;
            arr[a]=1;
            arr[b]=1;
            i++;
        }
        starts=0;
        ends=0;
        for(i=0;i<26;i++)
        {
            if(in[i]==out[i]);
            else if(in[i]==out[i]+1)
            {
                starts++;
            }
            else if(out[i]==in[i]+1)
            {
                ends++;
            }
            else
            {
                starts=100;
                ends=100;
                break;
            }
        }
        if(starts<=1 && ends<=1 && connected())
            printf("Ordering is possible.\n");
        else
            printf("The door cannot be opened.\n");
    }
	return 0;
}
