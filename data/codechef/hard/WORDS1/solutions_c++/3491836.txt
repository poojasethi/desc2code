int v[26],g[26][26];
void dfs(int ch)
{
	v[ch]=0;
	for(int i=0;i<26;i++)
	if(g[ch][i]||g[i][ch])
	g[ch][i]=0,g[i][ch]=0,dfs(i);
}

	int T,n,m,i,j,k;

	{
		scanf("%d",&n);

		memset(v,0,sizeof(v));
		memset(g,0,sizeof(g));

		{

			d[x]++,d[y]--;
			v[x]=1,v[y]=1;
			g[x][y]=1;
		}
