#include<stdio.h>
#include<string.h>

int mrk[26],g[26][26];

void traverse(int j)
{
	int i;
	mrk[j]=1;
	for(i=0;i<26;i++)
	{
		if(g[j][i] && mrk[i]==0)
		{
			traverse(i);
		}
	}
}


int main()
{
	int t,n,i,j,k,a[26][2],st,ed,flag,cnt,e,f;
	char s[1010];
	scanf("%d",&t);
	while(t--)
	{
		st=0,ed=0;
		memset(a,0,sizeof(a));
		memset(g,0,sizeof(g));
		memset(mrk,0,sizeof(mrk));
		k=0;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%s",s);
			e=s[0]-'a';
			f=s[strlen(s)-1]-'a';
			a[e][1]++;
			a[f][0]++;
//			if(e!=f)
//			{
				g[e][f]=1;
//				g[f][e]=1;
//			}
		}
		for(i=0;i<26;i++)
		{
			if(a[i][1]==a[i][0]+1)
			{
				++st;
				k=i;
			}
			else if(a[i][0]==a[i][1]+1)
			{
				ed++;
			}
			else if(a[i][0]!=a[i][1])
			{
				break;
			}
		}
		//printf("%d %d %d\n",i,st,ed);
		flag=0;
		cnt=0;
		traverse(k);
		if(i==26 && st==ed && ed<=1)
		{
			for(i=0;i<26;i++)
			{
				if((a[i][0]>0 || a[i][1]>0) && mrk[i]==0)
				{
//					cnt++;
//					if(cnt==2)
//					{
						flag=1;
						break;
//					}
//					traverse(i);
				}
			}
			if(flag==0)
			{
				printf("Ordering is possible.\n");
			}
			else
			{
				printf("The door cannot be opened.\n");
			}
		}
		else
		{
			printf("The door cannot be opened.\n");
		}
	}
	return 0;
}
