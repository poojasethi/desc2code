void dfs(int co,int u,int fa)
{
	sz[u]=0;dp[co][u][0]=1;
	for(int i=0;i<g[u].size();i++)
	{
		int v=g[u][i];
		if(v==fa)continue;
		dfs(co,v,u);
		for(int j=0;j<=sz[u]+sz[v];j++)h[j]=0;
		for(int j=0;j<=sz[u];j++)
		if(dp[co][u][j])
		{
			for(int k=0;k<=sz[v];k++)
			add(h[j+k],(ll)dp[co][u][j]*dp[co][v][k]%mod);
			for(int k=0;k<=min(lim[co]-1,sz[v]);k++)
			add(h[j+k+1],(ll)dp[co][u][j]*dp[co][v][k]%mod);
		}
		sz[u]+=sz[v];
		for(int j=0;j<=sz[u];j++)dp[co][u][j]=h[j];
	}
	sz[u]++;
}
void dfs1(int co,int u,int fa)
{
	sz[u]=0;f[co][u][0]=1;
	for(int i=0;i<g[u].size();i++)
	{
		int v=g[u][i];
		if(v==fa)continue;
		dfs1(co,v,u);
		for(int j=0;j<=sz[u]+sz[v];j++)h[j]=0;
		for(int j=0;j<=sz[u];j++)
		if(f[co][u][j])
		{
			for(int k=0;k<=sz[v];k++)
			add(h[j+k+1],(ll)f[co][u][j]*f[co][v][k]%mod);
			for(int a=0;a<3;a++)
			if(a!=co)
			{
				int b=3-co-a;
				for(int k=0;k<=sz[v];k++)
				if(dp[b][v][k])
				{
					int res=(ll)f[co][u][j]*dp[b][v][k]%mod;
					for(int w=0;w<=sz[v]-k-1;w++)
					{
						if(w+1>lim[a])break;
						int x=sz[v]-k-1-w;
						add(h[j+x],(ll)res*c[sz[v]-k-1][w]%mod);
					}
				}
			}
		}
		sz[u]+=sz[v];
		for(int j=0;j<=sz[u];j++)f[co][u][j]=h[j];
	}
	sz[u]++;
}
