void dfs(int co,int u,int fa)
{
	sz[u]=0;dp[co][u][0]=1;
	for(int i=0;i<g[u].size();i++)
	{
		int v=g[u][i];
		if(v==fa)continue;
		dfs(co,v,u);
		for(int j=0;j<=sz[u]+sz[v];j++)h[j]=0;
		for(int j=0;j<=sz[u];j++)
		if(dp[co][u][j])
		{
			for(int k=0;k<=sz[v];k++)
			add(h[j+k],(ll)dp[co][u][j]*dp[co][v][k]%mod);
		}
		sz[u]+=sz[v];
		for(int j=0;j<=sz[u];j++)dp[co][u][j]=h[j];
	}
	for(int j=0;j<=sz[u];j++)pre[co][u][j]=dp[co][u][j];
	for(int j=lim[co]-1;j>=0;j--)add(dp[co][u][j+1],dp[co][u][j]);
	sz[u]++;
}

		for(int j=0;j<=sz[u];j++)f[co][u][j]=h[j];
	}
	if(u==1)
	{
		for(int i=0;i<lim[co];i++)
		add(ans,f[co][1][i]);
		return ;
	}
	h[0]=0;
	for(int i=0;i<=sz[u];i++)h[i+1]=f[co][u][i];
	for(int a=0;a<3;a++)
	if(a!=co)
	{
		int b=3-co-a;

				add(h[x],(ll)pre[b][u][j]*c[sz[u]-1-j][k]%mod);

}
