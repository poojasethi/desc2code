int search(int a[],int low,int high,int value)
{
	int mid;
	while(low<=high)
	{
		int mid=(low+high)/2;
		if(a[mid]>=value)
		high=mid-1;
		else low=mid+1;
	}
	return low;
}

int main()
{
	int n;
	cin>>n;
	int a[n];
	for(int i=0;i<n;i++)
	cin>>a[i];
	int m;
	cin>>m;
	while(m--){
		int t,d;
		cin>>t>>d;
		int pos=search(a,0,n-1,t);
		int temppos;
		
		if(pos>=n || a[pos]>t)
		temppos=pos-1;
		else 
		temppos=pos;
		
		while(temppos>0 && a[temppos-1]+d>=a[temppos])
		{
			int k=search(a,0,temppos-1,a[temppos]-d);
			temppos=k;
		}
		cout<<temppos+1<<"\n";
	}
}
