using namespace std;

int n, a[N], Ma[21][N];

int Get_max(int d, int c)
{
    int lo = log2(c - d + 1);
    return max( Ma[lo][d], Ma[lo][c-(1<<lo)+1] );
}

int main()
{
   // filein("inp.txt");  fileout("out.txt");
    int q, t, d;
    cin >> n;
    for(int i=1; i<=n; i++)
        scanf("%d", &a[i]);    a[n+1] = ln;
    for(int i=1; i<=n; i++)
        Ma[0][i] = a[i+1] - a[i];
    for(int j=1; j<=20; j++)
        for(int i=1; i<=n; i++)
            Ma[j][i] = max(Ma[j-1][i], Ma[j-1][i+(1<<(j-1))]);
    cin >> q;
    while(q--)
    {
        scanf("%d%d", &t, &d);
        int pos = upper_bound(a+1, a+n+2, t) - a - 1;
        int l = 0;
        int r = pos;
        while(r - l > 1)
        {
            int x = (l + r) / 2;
            if(Get_max(x, pos-1) <= d)  r = x;
            else l = x;
        }
        printf("%d\n", r);
    }
}
