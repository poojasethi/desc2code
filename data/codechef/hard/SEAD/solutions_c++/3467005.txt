#include <iostream>
#include <vector>

using namespace std;

int main()
{
        int n;
        ios_base::sync_with_stdio(false);
        cin >> n;
        vector<int> v(n);

        for(int i = 0; i < n; i++) {
            cin >> v[i];
        }

        v.push_back(10000007);

        int q, t, d;

        cin >> q;

        for(int i = 0; i < q; i++) {
            cin >> t >> d;

            int low, high;
            low = 0, high = n-1;
            int ans = -1;
            while(low <= high) {
                int mid = (low+high)/2;
                if(v[mid] > t) {
                    high = mid-1;
                }
                else {
                    low = mid+1;
                    ans = max(ans, mid);
                }
            }

            if(ans == -1) {
                cout << n << endl;
                continue;
            }
            else {
                while(ans > 0 && v[ans-1]+d >= v[ans]) {
                    int low, high, mid;
                    low = 0;
                    high = ans-1;
                    int val = ans;
                    while(low <= high) {
                        mid = (low+high)/2;
                        if(v[mid] >= v[ans]-d) {
                            val = min(mid, val);
                            high = mid-1;
                        }
                        else {
                            low = mid+1;
                        }
                    }
                    ans = val;
                }
                cout << ans+1 << endl;
            }
        }
}
