#include <cstdio>
#include <algorithm>
#define mp make_pair
#define pb push_back
#define fir first
#define sec second
using namespace std;
typedef long long ll;

inline void R(int &x) {
	char ch = getchar(); x = 0;
	for (; ch<'0'; ch=getchar());
	for (; ch>='0'; ch=getchar()) x = x*10+ch-'0';
}
const int N = 100005;
int n, q, a[N], b[N], f[N][20];
int solve(int pos, int d) {
	for (int j=19; j>=0; --j) {
		if (f[pos][j]<=d)
			pos -= (1<<j);
	}
	return pos;
}
void run() {
	R(n);
	for (int i=1; i<=n; ++i) R(a[i]);
	for (int i=2; i<=n; ++i)
		b[i] = a[i] - a[i-1];
	b[1] = 1<<30;
	for (int i=1; i<=n; ++i) {
		f[i][0] = b[i];
		for (int j=1; j<20; ++j)
			if (i-(1<<(j-1))>=1)
				f[i][j] = max(f[i][j-1], f[i-(1<<(j-1))][j-1]); else
				f[i][j] = f[i][j-1];
	}
	R(q);
	int t, d, k;
	while (q--) {
		R(t); R(d);
		k = upper_bound(a+1, a+n+1, t) - a - 1;
		printf("%d\n", solve(k, d));
	}
}
int main() {
	run();
	return 0;
}
