#include <iostream>
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
int n,m,t,d,x,y,val,seg[4*100010];
vector<int> a;
void Update(int st,int en ,int k)
{
    if(x>en || y<st)
        return;
    if(st==en)
    {
        seg[k]=val;
        return;
    }
    int mid=(st+en)/2;
    Update(st,mid,k*2) , Update(mid+1,en,k*2+1);
    seg[k]=max(seg[k*2],seg[k*2+1]);
}
int Query(int st,int en ,int k)
{
    if(x>en || y<st)
        return -1;
    if(st>=x && en<=y )
        return seg[k];
    int mid=(st+en)/2;
    return max(Query(st,mid,k*2) , Query(mid+1,en,k*2+1));
}
int main()
{
    //freopen("r.txt","r",stdin);
    scanf("%d",&n);
    a=vector<int>(n);
    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);
    for(int i=1;i<n;i++)
        x=y=i , val=a[i]-a[i-1] , Update(0,n-1,1);
    scanf("%d",&m);
    for(int i=0;i<m;i++)
    {
        scanf("%d %d",&t,&d);
        int st=0 , en;
        en=upper_bound(a.begin(),a.end(),t) - a.begin() - 1;
        while(st<en)
        {
            int mid=(st+en)/2,val1=0,val2=0;
             x=st , y=mid , val1=Query(0,n-1,1);
             x=mid+1 , y=en , val2=Query(0,n-1,1);
             if(val2 > d)
                st=mid+1;
             else if(val1>d)
                en = mid;
             else
                en=st;
        }
        printf("%d\n",st+1);
    }

    return 0;
}
