VL prime;
bool is_prime[1000005];
void sieve(ll N)  // O(N * log(log(N)))
{
    MEM(is_prime,N,1);
    is_prime[1]=is_prime[0]=0;
    l i,j;
    for(i=2;i*i<=N;++i)
    {
        if(is_prime[i])
        {
            for(j=2*i;j<=N;j+=i)
                {
                    is_prime[j]=0;
                }
        }
    }
   FORAB(i,2,N)
    {
        if(is_prime[i])
            prime.pb(i);

    }

}
