#include<stdio.h>
#include<map>
#include<math.h>
#include<vector>
int main(){
    int t;
    scanf("%d",&t);
    for(int j=0;j<t;j++){
        int n;
        scanf("%d",&n);
        std::map<int,int> m;
        std::vector<int> v;
        for(int i=0;i<n;i++){
            int tmp;
            scanf("%d",&tmp);
            int f=0;
            while(tmp%2==0){
                if(m[2]==0)
                v.push_back(2);
                m[2]++;
                tmp=tmp/2;
            }
            for(int k=3;k<=sqrt(tmp);){
                if(tmp%k==0){
                    if(m[k]==0)
                    v.push_back(k);
                    
                    m[k]++;
                    tmp=tmp/k;
                }
                else
                    k++;
            }
            if(tmp!=1){
                if(m[tmp]==0)
                v.push_back(tmp);
                m[tmp]++;
            }
            
        }
        long long ans=1;
        std::vector<int>::iterator it;
      for(it=v.begin() ; it < v.end(); it++) {
      int g=*it;
      ans*= m[g]+1;
      }
        printf("%lld\n",ans);
        }
return 0;
}
