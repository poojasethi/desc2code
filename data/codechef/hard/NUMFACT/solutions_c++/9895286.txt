#include <iostream>
#include<map>
#define ll long long
using namespace std;
int k=0;
bool a[1000000]={0};
int p[100000];
void sieve()
{
int i,j;
for(i=2;i<1000000;i++)
{
if(!a[i])
{
p[k++]=i;
for(j=i*2;j<1000000;j+=i)
{a[j]=1;
}
}
}
}
int main() {
sieve();
	int t,n,a,i,j;
	ll ans;
	map<int,int>m;
	cin>>t;
	while(t--)
	{m.clear();ans=1;
	cin>>n;
	for(i=0;i<n;i++)
	{cin>>a;
	for(j=0;j<k;j++)
	{
	while(a%p[j]==0&&a!=1)
	{a/=p[j];
	if(m.find(p[j])!=m.end())
	m[p[j]]++;
	else m.insert(make_pair(p[j],1));
        }
        }
        if(a!=1)
        if(m.find(a)!=m.end())
        m[a]++;
        else m.insert(make_pair(a,1));
	}
	map<int,int>::iterator l;
	for(l=m.begin();l!=m.end();l++)
    ans*=((*l).second)+1ll;
    cout<<ans<<endl;
	}
	return 0;
}
