#include <iostream>
#include <limits.h>
#include <algorithm>
#include <math.h>
using namespace std;
int main(){
    
    int t;
    int n;
    int a[10];
    int pf[1000005];
    int factors;
    cin >> t;
    int k;
    long long ans;
    int root;
    int maxi,max2;
    while(t--){
        cin >> n;
        ans = 1;
        maxi = INT_MIN;
        for(int i = 0;i < n;i++) {
           cin >> a[i];
            if(a[i] > maxi) maxi = a[i];
        }
        for(int i = 0;i <= maxi;i++) pf[i]= 0;
        max2 = INT_MIN;
        for(int i = 0;i < n;i++){
            k = 2;
            root = sqrt(a[i]);
            while(a[i] > 1){
                if(k > root) {
                    //cout << "came here with k " << k << " and a[" << i << "] " << a[i] << endl;
                    pf[a[i]]++;
                    if(max2 < a[i]) max2 = a[i];
                    break;
                }
                if(a[i]%k == 0){
                  if(max2 < k) max2 = k;
                  pf[k]++;
                    a[i] /= k;
                    continue;
                } 
                k++; 
            }
        }    
        for(int i = 0; i<= max2;i++){
            if(pf[i] != 0){
                ans *= (pf[i]+1);
            }
        }
        cout << ans << endl;
          
   }
    return 0;
}
