#include<bits/stdc++.h>
#define ll long long
#define mp make_pair
#define pb push_back
using namespace std;
int prime[1000001];
map<int,int> dp;
int main()
{
   for(int i=2;i<=1000000;i++) prime[i]=1;
   for(int i=2;i<=1000;i++){
    if(prime[i]){
        for(int j=i*i;j<=1000000;j+=i) prime[j]=0;
    }
   }
   int t;
   scanf("%d",&t);
   while(t--){
        dp.clear();
    int n;
    scanf("%d",&n);
    int arr[n];
    for(int i=0;i<n;i++) scanf("%d",&arr[i]);
    for(int i=2;i<=1000;i++){
        if(prime[i]){
            int cnt = 0;
            for(int j=0;j<n;j++){
                while(arr[j]%i==0 && arr[j]!=0){
                    arr[j]/=i;
                    cnt++;
                }
            }
            dp[i] += cnt;
        }
    }
    for(int i=0;i<n;i++) if(arr[i] > 1) dp[arr[i]]++;
    map<int,int>:: iterator it;
    ll ans = 1;
    for(it=dp.begin();it!=dp.end();it++){
        ans *= (it->second +1);
    }
    cout<<ans<<endl;
   }
return 0;
}
