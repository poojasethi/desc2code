#include <bits/stdc++.h>
# define s(x) scanf("%lld",&x)
using namespace std;
#define pb push_back
#define mp make_pair
#define ll long long
#define ms(a,b) memset((a),(b),sizeof(a))
#define inf 1000000007
#define f(i,j) for(int i=0;i<j;i++)
int main()
{
    ll t,n,p=1000001,sum,ans,i,j;
    s(t);
    vector<ll>q(p,0);
    vector<ll>a;
    // prime numbers less than 1000001
    for(i=2;i<sqrt(p);i++)
    {
        if(q[i]==0)
        {
            a.push_back(i);
            for(j=i*i;j<p;j+=i)
                q[j]=1;
        }
    }
//    for(j=i;j<p;j++)
//    {
//        if(q[j]==0)
//            a.push_back(j);
//    }
    while(t--)
    {
        s(n);
        ll val=0;
        ll b[n];
        for(i=0;i<n;i++)
        {
            s(b[i]);
            if(b[i]>val)
                val=b[i];
        }
        ans=1;
        for(i=0;a[i]<=sqrt(val)&&i<a.size();i++)
        {
            sum=0;
            for(j=0;j<n;j++)
            {
                //no. of times a particular prime number divides given numbers is sum
                while(b[j]%a[i]==0)
                {
                    sum++;
                    b[j]/=a[i];
                }
            }
            ans*=(sum+1);
        }
        for(i=0;i<n;i++)
        {
            if(b[i]!=1)
            {
                sum=1;
                for(j=i+1;j<n;j++)
                {
                    if(b[i]==b[j])
                    {
                        b[j]=1;
                        sum++;
                    }
                }
                ans*=(sum+1);
            }
        }
        printf("%lld\n",ans);
    }
    return 0;
}
