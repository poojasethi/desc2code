#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define MAX 1000005

int P[MAX];
int F[MAX];

void get_primes(){
	memset(P,-1,sizeof P);
	for(int i = 2;i*i <= MAX;i++){
		if(P[i] != -1) continue;
		for(int j = 2*i;j < MAX; j += i){
			if(P[j] == -1) P[j] = i;
		}
	}
}

void factorize(int a){
	while(P[a] != -1){
		F[P[a]]++;
		a = a/P[a];
	}
	if(a > 1) F[a]++;
}

int main(){
	int t,n,a;
	get_primes();
	scanf("%d",&t);
	while(t--){
		scanf("%d",&n);
		memset(F,0,sizeof F);
		for(int i = 0;i < n;i++){
			scanf("%d",&a);
			factorize(a);
		}
		int r = (F[2] + 1);
		for(int  i = 3;i < MAX;i += 2) r *= (F[i] + 1);
		
		printf("%d\n",r);
		
	}
	return 0;
}
