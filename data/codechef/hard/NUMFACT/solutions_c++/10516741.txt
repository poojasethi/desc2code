#include<stdio.h>
#include<algorithm>
#include<math.h>
#include<vector>
using namespace std;
bool cmp(long long int a,long long int b)
{
	return(a>b);

}
int main()
{
	long long int t,n,total,i,j,s,a,count;
	vector<long long int > factors;
	scanf("%lld",&t);
	while(t--)
	{
		factors.clear();
		count=1;
		total=1;
		scanf("%lld",&n);
		while(n--)
		{
			scanf("%lld",&a);
			while(a%2==0)
			{
				factors.push_back(2);
				a=a/2;
			}
			for(i=3;i<=sqrt(a);i=i+2)
			{
				while(a%i==0)
				{
					factors.push_back(i);
					a=a/i;
				}
			}
			if(a>2)
				factors.push_back(a);
		}
		sort(factors.begin(),factors.end(),cmp);
		s=factors.size();
		count=1;
		i=1;
		while(i<s)
		{
			if(factors[i]!=factors[i-1])
			{
				total=total*(count+1);
				count=1;

			}
			else
				count++;
			i++;	
		}
		total=total*(count+1);
		printf("%lld\n",total);

	}
	return 0;
}
