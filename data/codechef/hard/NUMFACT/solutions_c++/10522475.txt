static int prime[1000000];

int main(){
	int t,n,num;
	long long ans;
	scanf("%d",&t);
	while(t--){
		ans=1;
		for (int i = 0; i < 1000000; ++i)
		{
			prime[i]=0;
		}
		scanf("%d",&n);
		while(n--){
			scanf("%d",&num);
			for (int i = 2; i*i <= num; ++i)
			{
				while(num%i==0){
					prime[i]++;
					num/= i;
				}
			}
			prime[num]++;
		}
		for (int i = 2; i < 1000000; ++i)
			ans*= (prime[i]+1);
		printf("%lld\n",ans);
	}
	return 0;
}
