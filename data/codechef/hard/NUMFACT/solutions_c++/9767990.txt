#include <iostream>
#include<cmath>
using namespace std;
#define num 1000002

bool a[num+2];
int prime[num+2];
void sieve(){
	int i,j;
	a[1]=1;
	for(i=2;i<=num;++i){
		if(a[i]==0){
			for(j=2*i;j<=num;j+=i){
				a[j]=1;
			}
		}
	}
	
	int l=0;
	for(i=1;i<=num;++i) if(a[i]==0) prime[l++]=i;

}
int main() {
	// your code goes here
	sieve();
	int t,n,i,x,j;
	cin>>t;
	while(t--){
		cin>>n;
		int g[num+2]={0};
		for(i=0;i<n;++i){

			cin>>x;
			j=0;
			while(prime[j]<=sqrt(x)){
				
				while(x%prime[j]==0){
					x/=prime[j];
					g[prime[j]]++;
				}
				j++;
			}
			if(x>1) g[x]++;
		}
		long long count=1;
		for(i=0;i<=num;++i){
		    //if(g[i]!=0){cout<<i<<" ";}
			count*=(g[i]+1);
		}
		if(count!=1)cout<<count<<"\n";
		
	}
	
	return 0;
}
