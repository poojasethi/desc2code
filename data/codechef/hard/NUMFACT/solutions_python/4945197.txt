def factorize(n):
    factors = []
    factor = 2
    number = n
    while number>1:
        factor = nextp(number, factor)
        factors.append(factor)
        number/= factor
    return factors

def nextp(n, f):
    if n%2 == 0:
        return 2
    else:
        for i in xrange(max(f,3), int(n**.5)+1,2):
            if n%i == 0:
                return i
    return n



n = int(raw_input())
for i in xrange(n):
    raw_input()
    nos = map(int, raw_input().split())
    factdict = {}
    for num in nos:
        temp = factorize(num)
        for j in temp:
            if j not in factdict:
                factdict[j] = 1
            else:
                factdict[j] += 1
    print reduce(lambda x, y: x*y, [factdict[x]+1 for x in factdict] )
