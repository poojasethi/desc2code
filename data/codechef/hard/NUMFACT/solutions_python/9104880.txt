import fileinput
import math







def gen_prime_numbers(N):
    prime_number_list = []
    prime_number_set = set()
    
    num2check = 2
    while num2check <= N:
        max_divisor = int(math.sqrt(num2check))
        
        found_divisor = False
        for i in xrange(len(prime_number_list)):
            prime = prime_number_list[i]
            remainder = num2check % prime
            
            #not prime
            if remainder == 0:
                found_divisor = True
                break
            
            if prime >= max_divisor:
                break
        
        if not found_divisor:
            prime_number_list.append(num2check)
            prime_number_set.add(num2check)
            
        num2check += 1
    return prime_number_list, prime_number_set





def update_prime_factor_map(factor_map, N):
    
    global prime_number_list
    global prime_number_set
    
    num2check = N
    
    
    for i in xrange(len(prime_number_list)):
        
        
        #see if we are prime
        if num2check in prime_number_set:
            if num2check not in factor_map:
                factor_map[num2check] = 0
            factor_map[num2check] += 1
            break
        
#        #otherwise see if we should break
#        prime = prime_number_list[i]
#        if prime >= int(math.sqrt(num2check)):
#            break
        
        prime = prime_number_list[i]
        #otherwise check divisor
        remainder = num2check % prime
        
        #divide until we cannot
        while remainder == 0:
            if prime not in factor_map:
                factor_map[prime] = 0
            factor_map[prime] += 1
            
            num2check = num2check / prime
            remainder = num2check % prime
