for t in xrange(int(raw_input())):
	length = raw_input()
	array = map(int, raw_input().split())
	diction = dict()
	for number in array:
		i = 2
		while i*i <= number:
			while number % i == 0:
				diction[i] = diction.get(i, 0) + 1
				number /= i
			i += 1
		if number > 1:
			diction[number] = diction.get(number, 0) + 1
	ans = 1
	for key, value in diction.items():
		ans *= (value + 1)
	print ans
