import sys
from collections import Counter

def primes(n):
    primfac = []
    d = 2
    while d*d <= n:
        while (n % d) == 0:
            primfac.append(d)
            n /= d
        d += 1
    if n > 1:
       primfac.append(n)
    return primfac

def readtestcase():
    n = int(sys.stdin.readline())
    num = map(int, sys.stdin.readline().split())
    res = []
    for i in range(0,n):
        res = res + primes(num[i])
    c = Counter(res)
    result = 1
    for item in c:
        result = result * (c[item] + 1)
    print result


t = int(sys.stdin.readline())
for i in range(0,t):
    readtestcase()
