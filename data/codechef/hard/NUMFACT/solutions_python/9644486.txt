#!/usr/bin/python
 
from collections import defaultdict
def find_first_factor(n):
    if n%2 == 0:
        return 2
    for i in range(3,n/2,2):
        if n%i == 0:
            return i
    return n
 
def find_prime_factors(n):
    result = []
    while n > 1:
        result.append( find_first_factor(n) )
        n=n/result[-1]
    return result
 
if __name__ == "__main__":
    n = int(raw_input())
    for i in range(n):
        m = int(raw_input())
        a = map(int, raw_input().split())
        r = []
        for j in a:
            r += find_prime_factors(j)
        d = defaultdict(int)
        for p in r:
            d[p] += 1
        s = 1
        for k in d:
            s *= d[k] + 1
        print s
