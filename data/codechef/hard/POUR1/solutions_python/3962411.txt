def solve(A,B,C):
	ans=1
	a=A
	b=0
	while(a!=C and b!=C):
#		print "A=",A,"B=",B,"C=",C,"a=",a
		x=min(a,B-b)
		b=b+x
		a=a-x
		ans=ans+1
		if(a==C or b==C):
			break
		if(a==0):
			a=A
			ans=ans+1
		if(b==B):
			b=0
			ans=ans+1
	return ans
