#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
    return gcd(b,a%b);
}

int cnt(int A, int B, int C) {
	int move = 1, a = A, b = 0, pour;
	while(a != C && b != C) {
		pour = min(a, B-b);
		b += pour;
		a -= pour;
		move++;
		if(a == C || b == C)
            break;
		if(b==B)
		 b = 0,
		  move++;
		else if(a==0)
		 a = A, move++;
	}
	return move;
}
int main()
{
    int t;
   cin>>t;
     while(t--) {
        int a,b,c;
        cin>>a>>b>>c;
        if (c > a && c > b)
            cout<<"-1\n";
        else if (c == 0)
            cout<<"0\n";
        else if (c == a || c == b)
             cout<<"1\n";
        else if (c%gcd(a, b) != 0)
             cout<<"-1\n";
        else {
            cout<<min(cnt(a,b,c),cnt(b,a,c))<<endl;;
        }
    }
    return 0;
}
