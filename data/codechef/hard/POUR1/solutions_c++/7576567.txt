#include<bits/stdc++.h>
using namespace std;
#define min(a,b) a<b?a:b
int pour(int A,int B,int C)
{
int count=1,a=A,b=0,d;
while(a!=C && b!=C){
           d=min(a,B-b);
           b+=d;
           a-=d;
           count++;
           if(a==C || b==C)
                   break;
           if(a==0){
                    a=A;
                    count++;
           }
           if(b==B){
                b=0;
                count++;
           }
     }
     return count;
}
int gcd(int a,int b){
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int main(){
int t,a,b,c;
scanf("%d",&t);
while(t--){
           scanf("%d%d%d",&a,&b,&c);
           if(c>a && c>b)
                  printf("-1\n");
           else if(c%gcd(a,b) != 0)
                printf("-1\n");
           else if(c==a || c==b)
                printf("1\n");
           else
               printf("%d\n",min(pour(a,b,c),pour(b,a,c)));
}
return 0;
}
