#include<iostream>
#define SWAP(x, y) {x^=y;y^=x;x^=y;}
using namespace std;

int gcd(int a, int b);

int main()
{
	int setCnt;
	cin>>setCnt;
	while (setCnt-- > 0)
	{
		int a, b, c;
		cin>>a>>b>>c;
		if (a < b)
			SWAP(a,b);
        if (c > a || c % gcd(a, b) != 0)
		{	
			cout<<"-1"<<endl;
			continue;
		}

		if (a == c || b == c)
		{
			cout<<"1"<<endl;
			continue;
		}

        int steps = 0;
		int vesselOne = 0, vesselTwo = 0;
        int vesselOneAlt = 0, vesselTwoAlt = 0;
        

		while(vesselOne!=c && vesselTwo!=c && vesselOneAlt!=c && vesselTwoAlt!=c)
		{
            if(vesselOneAlt == 0)
            {
                vesselOneAlt = b;
            }
            else if (vesselTwoAlt == a)
            {
                vesselTwoAlt = 0;
            }
            else
            {
                int remain = a - vesselTwoAlt;
                vesselOneAlt -= remain;
                vesselTwoAlt = a;
                
                if(vesselOneAlt < 0)
                {
                    vesselTwoAlt += vesselOneAlt;
                    vesselOneAlt = 0;
                }
            }
            
			if(vesselOne == 0)
			{
				vesselOne = a;
			}
			else if (vesselTwo == b)
			{
				vesselTwo = 0;
			}
			else
			{
				int remain = b - vesselTwo;
				vesselOne -= remain;
				vesselTwo = b;

				if(vesselOne < 0)
				{
					vesselTwo += vesselOne;
					vesselOne = 0;
				}
                
			}
            steps++;

		}
		cout<<steps<<endl;

	}
}
int gcd(int a, int b)
{
    while(b != 0)
    {
        int c = a % b;
        a = b;
        b = c;
    }
    return a;
}
