#include<bits/stdc++.h>
using namespace std;
#define ll long long
int gcd(int a, int b){
	if(b==0)return a;
	return gcd(b, a%b);
}
int func(int a, int b, int d){
	int from=a, to=0, steps=1, temp;
	while(from!=d&&to!=d){
		temp=min(b-to, from);
		from-=temp;
		to+=temp;
		steps++;
		if(from==d||to==d)break;
		if(from==0){
			from=a;
			steps++;
		}
		if(to==b){
			to=0;
			steps++;
		}
	}
	return steps;
}
int main(){
	int t, i, j;
	cin>>t;
	while(t--){
		int a, b, c, i, j;
		cin>>a>>b>>c;
		if(c>max(a, b)||c%gcd(a, b))
			cout<<-1<<endl;
		else cout<<min(func(a, b, c), func(b, a, c))<<endl;
	}
}
