#include<bits/stdc++.h>
using namespace std;
int max(int a,int b)
{
    return a>b?a:b;
}
int min(int a,int b)
{
    return a>b?b:a;
}
int solve1(int a,int b,int c)
{
    int x=0, y=0,ans=0;
    while(x!=c&&y!=c){
    if(x==0)
    {
        x=a;
        ans++;
    }
    else if(y==b)
    {
        y=0;
        ans++;
    }
    else
    {
        int val1=min(x,b-y);
        ans++;
        x=x-val1;
        y=y+val1;
    }
    }
    return ans;
}
int solve(int a,int b,int c)
{
    if(c==0) return 0;
    if(c>max(a,b)) return -1;
    if(c%__gcd(a,b)!=0)
        return -1;
    else
    {
        //cout<<c%__gcd(a,b)<<" ";
        int val1=solve1(a,b,c);
        int val2=solve1(b,a,c);
        return min(val1,val2);
    }
}
int main()
{
    int t,a,b,c;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c;
        int ans=solve(a,b,c);
        cout<<ans<<endl;
    }

    return 0;
}
