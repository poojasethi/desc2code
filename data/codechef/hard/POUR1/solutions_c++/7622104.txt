ll pour(ll A,ll B, ll C){
    ll a=A,b=0,c=1,x;
    while(a!=C && b!=C){
        x=min(a,B-b);
        a=a-x;
        b+=x;
        c++;
        if(a==C || b==C){break;}
        if(a==0){a=A; c++;}
        if(b==B){b=0; c++;}
    }
    return c;
}
