#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
    if(b==0)
        return a;
    return gcd(b,a%b);
}
int pour(int a,int b,int c)
{
    int mov=1,x=a,y=0,tfr;
    while(x!=c && y!=c)
    {
        tfr=min(x,b-y);
        y+=tfr;
        x-=tfr;
        mov++;
        if(x==c|| y==c)
            break;
        if(x==0)
        {
            x=a;
            mov++;
        }
        if(y==b)
        {
            y=0;
            mov++;
        }
    }
    return mov;
}
int main()
{
    int t,a,b,c;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&a);
        scanf("%d",&b);
        scanf("%d",&c);
        int d=gcd(a,b);
        if(c%d!=0 || ((c>a)&& (c>b)))
        {
            printf("-1\n");
            continue;
        }
        int ans=min(pour(a,b,c),pour(b,a,c));
        printf("%d\n",ans);
    }
    return 0;
}
