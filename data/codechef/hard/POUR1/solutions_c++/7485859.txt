#include<stdio.h>


int gcd(int a,int b)
{
	if(b==0)
	{
		return a;
	}
	else return gcd(b,a%b);
}

int main()
{
	long i,j,k1,k2,k,a,b,c,t;
	scanf("%ld",&k);
	while(k--)
	{
		scanf("%ld %ld %ld",&a,&b,&c);
		k1=0,k2=0;
		if(b>a)
		{
			t=a;
			a=b;
			b=t;
		}
		if(c>a)
		{
			printf("-1\n");
			continue;
		}
		if(c%gcd(a,b)!=0)
		{
			printf("-1\n");
			continue;
		}
		i=0,j=0;
		for(;;)
		{
			//printf("%ld %ld\n",i,j);
			if(i==0)
			{
				i=a;
				k1++;
			}
			if(i==c)
			{
				break;
			}
			t=i;
			i-=(b-j);
			if(i<0)
			{
				i=0;
			}
			j+=t;
			if(j>b)
			{
				j=b;
			}
			k1++;
			if(i==c || j==c)
			{
				break;
			}
			if(j==b)
			{
				j=0;
				k1++;
			}
			//printf("1\n");
			/*if(j==0)
			{
				j=b;
				i=0;
				k1++;
			}*/
		}
		i=0,j=0;
		for(;;)
		{
			if(j==0)
			{
				j=b;
				k2++;
			}
			if(j==c)
			{
				break;
			}
			t=j;
			j-=(a-i);
			if(j<0)
			{
				j=0;
			}
			i+=t;
			if(i>a)
			{
				i=a;
			}
			k2++;
			if(i==c || j==c)
			{
				break;
			}
			if(i==a)
			{
				i=0;
				k2++;
			}
		}
		t=(k2>k1?k1:k2);
		printf("%ld\n",t);
	}
	return 0;
}
