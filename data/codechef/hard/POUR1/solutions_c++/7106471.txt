#include <stdio.h>
#define DEBUG 0
#define SIMPLE 1

int gcd(int m, int n)
{
		//printf("m %d n %d\n", m, n);
		if(n==0) return m;
		return gcd(n, m%n);
}

int main()
{
		int T;
		scanf("%d", &T);
		while(T--)
		{
				int a, b, c;
				scanf("%d %d %d", &a, &b, &c);
				
				if(a<b)
				{
						int t=a;
						a=b;
						b=t;
				}
				
				if(c==0)
				{
					printf("0\n");
					continue;
				}
				else if(c%gcd(a,b)!= 0 || c>a)
				{
					printf("-1\n");
					continue;
				}
				
				int a1=0, b1=0, a2=0, b2=0, s1=0, s2=0;
				
				while(1)
				{
					//if(a1<0 || b1<0 || a2<0 || b2<0) break;

					//empty the recipient vessels
					b1==b ? (s1++, b1=0) : 1;
					a2==a ? (s2++, a2=0) : 1;
					if (DEBUG) printf("a1 %d b1 %d s1 %d || a2 %d b2 %d s2 %d\n", a1, b1, s1, a2, b2, s2);
					
					
					//fill the primary vessels
					a1==0? (s1++, a1=a) : 1;
					b2==0? (s2++, b2=b) : 1;
					if (DEBUG) printf("a1 %d b1 %d s1 %d || a2 %d b2 %d s2 %d\n", a1, b1, s1, a2, b2, s2);
					if(a1==c || b1 == c || a2==c || b2==c) break;
					
					//transfer water from primary vessels to secondary vessels
					if(a1<=(b-b1))
					{
						(b1+=a1, a1=0, s1++);
					}
					else
					{
						if(SIMPLE){
							a1-=(b-b1), b1=b, s1++;
						}
						else
						{
							if(c > (a%b) && (c - (a1%b))%b == 0)
							{
								int k = (c - (a1%b))/b;
								int l = a1/b;
								a1=c;
								b1=0;
								s1 += (l-k)*2 - 1;
							}
							else
							{
								a1%=b, b1=0, s1+=(a1/b)*2;
							}
						}
					}
					
					if(b2>=(a-a2))
					{
						b2-=(a-a2), a2=a, s2++;
					}
					else
					{
						if(SIMPLE)
						{
							a2+=b2; b2=0; s2++;
						}
						else
						{
							if( c>a2 && (c-a2)%b == 0)
							{
								a2=c;
								b2=0;
								s2 += 2*((c-a2)/b) - 1;
							}
							else
							{
								a2+=b2, b2=0, s2++;
							}
						}
					}
					
					if (DEBUG) printf("a1 %d b1 %d s1 %d || a2 %d b2 %d s2 %d\n", a1, b1, s1, a2, b2, s2);
					if(a1==c || b1 == c  || a2==c || b2==c) break;
				}
				
				printf("%d\n", s1<s2?s1:s2);
		}
}
