#include<bits/stdc++.h>
using namespace std;

int calculate(int a, int b, int c)
{
    int i,j;
    int x,y;
    i=0;
    j=0;

    x=0;

    while(i!=c && j!=c)
    {
        if(i==0)
        {
            //refill a
            i=a;
            x++;
        }

        else if(j==b)
        {
            //vacate b
            j=0;
            x++;
        }

        else
        {
            //transfer from a to b
            int scope=(b-j);

            if(i>=scope)
            {
                i=i-scope;
                j=b;
            }

            else
            {
                j=j+i;
                i=0;
            }

            x++;
        }
    }


    y=0;
    i=0;
    j=0;

    while(i!=c && j!=c)
    {
        if(j==0)
        {
            //refill b
            j=b;
            y++;
        }

        else if(i==a)
        {
            //vacate a
            i=0;
            y++;
        }

        else
        {
            //transfer from b to a
            int scope=(a-i);

            if(j>=scope)
            {
                j=j-scope;
                i=a;
            }

            else
            {
                i=i+j;
                j=0;
            }

            y++;
        }
    }

    return min(x,y);
}

int main()
{
    int a,b,c,c1;
    int t;

    cin>>t;

    while(t--)
    {
        cin>>a>>b>>c;
        c1=-1;
       
        if(c<=max(a,b) && c%__gcd(a,b)==0)
        {
            c1=calculate(a,b,c);
        }

        cout<<c1<<endl;
    }

    return 0;
}
