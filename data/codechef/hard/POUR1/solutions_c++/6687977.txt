#include<stdio.h>

using namespace std;

int gcd(int x1,int x2)
{
	return (x1==0||x2==0)?x1+x2:gcd(x2,x1%x2);
}

main()
{
	int testcases;
	scanf("%d",&testcases);
	while(testcases>0)
	{
		int n1,n2,n3;
		scanf("%d\n%d\n%d",&n1,&n2,&n3);
		if(n1<n2)
		{n1=n1+n2;n2=n1-n2;n1=n1-n2;
		}
		if(n3==0){printf("0\n");
		}
		else if (n3>n1){printf("-1\n");
		}
		else if(n3==n1 || n3==n2){printf("1\n");
		}
		else
		{
			int g=gcd(n1,n2);
			bool spcase=(g==n2);
			if(n3%g!=0)
			{
				printf("-1\n");
			}
			else if (spcase)
			{
				int s1=(n1-n3)/n2,s2=(n3/n2);
				s1=(s1>s2)?2*s2:2*s1;
				printf("%d\n",s1);
			}
			else
			{
				int num_op1=0,num_op2=0,x=n1;
				while(1){
				if((x-n3)%n2==0)
				{
					num_op1+=((x-n3)*2)/n2;
					break;
				}
				else
				{
					num_op1+=((int)(x/n2))*2+4;
					x=n1-n2+x%n2;
				}
				}
				x=0;
				while(1)
				{
					if((n3-x)%n2==0)
					{if(n3==x){num_op2-=2;
					}
						num_op2+=((n3-x)*2)/n2;
						break;
					}
					else
					{
						num_op2+=((int)(n1-x)/n2)*2+4;
						x=n2-((n1-x)%n2);
					}
				}
				num_op1=(num_op1<num_op2)?num_op1:num_op2;
				printf("%d\n",num_op1);
			}
		}
		testcases--;
	}
}
