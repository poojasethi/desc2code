#include<bits/stdc++.h>
using namespace std;
int main(){

    int t,a,b,c;
    cin.sync_with_stdio(false);
    cin.tie(0);
    cin >> t;
    while(t--){
        cin >> a >> b >> c;
        if(c > max(a,b) || (a == b && a != c)){
            cout << "-1\n";
            continue;
        }
        if(a > b)
            swap(a,b);
        if(a == c || b == c){
            cout << "1\n";
            continue;
        }
        int g = __gcd(a, b);
        if(c%g){
            cout << "-1\n";
            continue;
        }
        int s = 0, a1 = 0, b1 = 0;
        while(1){
            if(a1 == 0){
                a1 = a;
                s++;
            }
            else if(b1 == b){
                b1 = 0;
                s++;
            }
            else if((b - b1) >= a1){
                b1 += a1;
                a1 = 0;
                s++;
                if(b1 == c || (b - b1) == c)
                    break;
            }
            else if((b - b1) < a1){
                a1 -= (b - b1);
                b1 = b;
                s++;
                if(a1 == c)
                    break;
            }
        }
        cout << s << "\n";
    }
    return 0;

}
