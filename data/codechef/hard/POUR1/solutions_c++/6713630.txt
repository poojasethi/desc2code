#include <bits/stdc++.h>

using namespace std;

int rot(int a,int b,int c) {
    if (a==c || b==c) return 1;
    int ans=0, ta=0, tb=0;
    while (ta!=c && tb!=c) {
        if (ta==0) {ans++; ta=a;} else {
            int t=min(ta,b-tb);
            ans++; ta-=t; tb+=t;
            if (ta!=c && tb==b) {ans++; tb=0;}
        }
    }
    return ans;
}

int gcd(int a,int b) {
    while (b) {
        int r=a%b;
        a=b;
        b=r;
    }
    return a;
}

int main() {
    //freopen("inp.txt","r",stdin);
    int T; scanf("%d",&T);
    for(int t=1;t<=T;t++) {
        int a, b, c; scanf("%d%d%d",&a,&b,&c);
        if ((c>a && c>b) || c%gcd(a,b)!=0) printf("-1\n"); else {
            int ds=rot(a,b,c);
            ds=min(ds,rot(b,a,c));
            printf("%d\n",ds);
        }
    }
}
