int ans(int a,int b,int c)
{
	//printf("Here\n");
	int steps=0,x=0,y=0,mn;
	while(x!=c&&y!=c)
	{
		if(x==0)
			x=a;
		else if(y==b)
			y=0;
		else
		{
			mn=min(x,b-y);
			x-=mn;
			y+=mn;
		}
		steps++;
		//printf("Steps =%d\n",steps);
	}
	return steps;
}

void solve()
{
	int a,b,c;
	scanf("%d%d%d",&a,&b,&c);
	//printf("Got it\n");
	printf("%d\n",((c<=a||c<=b)&&((c%__gcd(a,b))==0))?(min(ans(a,b,c),ans(b,a,c))):-1);
}
