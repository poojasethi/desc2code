#include<stdio.h>
#include<stdlib.h>
 
int transferWater( int cap1, int cap2, int cap ){
    int count = 0, q1 = 0, q2 = 0;
 
    while( q1 != cap && q2!=cap ){
        if( q1 == cap1 )
            q1 = 0;
        else if( q2 == 0 )
            q2 = cap2;
        else{
            if( q1 + q2 > cap1 ){
                q2 = q1 + q2 - cap1;
                q1 = cap1;
            } else{
                q1 = q1 + q2;
                q2 = 0;
            }
        }
        count++;
        if( q1 == 0 && q2 == 0 )
              return -1;
    }
 
    return count;
 
}
 
int performTest(){
    int i, cap1, cap2, cap, gcd, count[2];
    scanf( "%d %d %d", &cap1, &cap2, &cap );
 
    count[0] = transferWater( cap1, cap2, cap);
    count[1] = transferWater( cap2, cap1, cap);
 
    if( count[0] < count[1] )
        return count[0];
    return count[1];
}
 
int main(){
 
    int numTest;
    scanf( "%d", &numTest );
 
    while( numTest-- )
        printf( "\n%d", performTest() );
 
 
    return 0;
}
