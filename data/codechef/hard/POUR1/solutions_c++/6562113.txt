#include <iostream>
using namespace std;

int operations(int a,int b,int c,int method)
{
	int r,k,flag,count;

	if(c==0)
		return 0;

	if(a==b)
		return(a==c?1:-1);

	if(c==a || c==b)
		return 1;

	if(method)
	{
		r=0;
		flag=0;
		count=0;
		while(r!=0 || flag==0)
		{
			if(!flag)
				flag=1;
			k=r+b-c;
			//cout<<k<<"***"<<endl;
			if(k%a==0)
				return(count+2*k/a);
			count+=2+2*((r+b)/a);
			r=(r+b)%a;
			//cout<<r<<endl;
			//cout<<count<<"/**"<<endl;
		}
		return -1;
	}

	else
	{
		r=0;
		flag=0;
		count=0;
		while(r!=0 || flag==0)
		{
			if(!flag)
				flag=1;
			if((c-r)%a==0)
				return(count+2*(c-r)/a);
			if((b-r)%a==0)
				break;
			k=(b-r)/a;
			r=r+a*(k+1)-b;
			if(r==c)
				return(count+2*(k+1));
			count+=2*(k+2);
		}
		return -1;
	}
}

int main()
{
	int a,b,c,t,n;
	int i,tmp,x,y;
	cin>>t;
	for(i=0;i<t;i++)
	{
		cin>>a>>b>>c;
		if(a>b)
		{
			tmp=b;
			b=a;
			a=tmp;
		}
		if(c>b)
			n=-1;
		else
		{
			x=operations(a,b,c,1);
			//cout<<x<<endl;
			y=operations(a,b,c,0);
			//cout<<y<<endl;
			if(x==-1)
				n=y;
			else if(y==-1)
				n=x;
			else
				n=(x<y)?x:y;
		}
		cout<<n<<endl;
	}
	return 0;
}
