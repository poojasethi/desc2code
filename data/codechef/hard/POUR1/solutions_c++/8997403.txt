#include<iostream>
#include<cstdio>
using namespace std;
int pour(int A,int B,int c)
{
	int a=A;
	int  b=0;
	int tbfr;
	int move=1;
	while(a!=c && b!=c)
	{
		int tbfr=min(a,B-b);
		a-=tbfr;
		b+=tbfr;
		move++;
		if(a==c||b==c)
		{
			return move;
		}
		if(a==0)
		{
			a=A;
			move++;
		}
		if(b==B)
		{
			b=0;
			move++;
		}
	}
	return move;
}
int gcd(int a,int b)
{
	if(b==0)
		return a;
	else
		gcd(b,a%b);
}
int main()
{
	int t;
	scanf("%d",&t);
	int a,b,c;
	while(t--)
	{
		//int n;
		//cin>>n;
		int a,b,c;
		cin>>a>>b>>c;
		if(c>a && c>b)
		{
			cout<<"-1"<<endl;
		}
		else if(c%gcd(a,b)!=0)
		{
			cout<<"-1"<<endl;
		}
		else if(a==c || b==c)
		{
			cout<<"1"<<endl;
		}
		else
		{
			cout<<min(pour(a,b,c),pour(b,a,c))<<endl;
		}
	}
}
