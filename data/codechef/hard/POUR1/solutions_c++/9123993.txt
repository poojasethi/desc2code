#include<bits/stdc++.h>
using namespace std;
#define ll long long int
int gcd(int a,int b)
{
    if(a%b==0)
        return b;
    return gcd(b,a%b);
}
int calc_steps(int A,int B,int C)
{
    int steps=1,a=A,b=0,minn;
    while(a!=C && b!=C)
    {
           minn=min(a,B-b);
           b+=minn;
           a-=minn;
           steps++;
           if(a==C || b==C)
                break;
           if(a==0)
           {
                a=A;
                steps++;
           }
           if(b==B)
           {
                b=0;
                steps++;
           }
    }
    return steps;
}
int find_ans(int a,int b,int c)
{
    if((c==a)||(c==b))
        return 1;
    else if(c==abs(a-b))
        return 2;
    else
        return min(calc_steps(a,b,c),calc_steps(b,a,c));
}
int main()
{
    int test,num,i,j,a,b,c,ans;
    scanf("%d",&test);
    while(test--)
    {
        scanf("%d%d%d",&a,&b,&c);
        if(max(a,b)<c)
            printf("-1\n");
        else if(c%gcd(a,b) != 0)
            printf("-1\n");
        else
        {
            if(b>a)
                swap(a,b);
            ans=find_ans(a,b,c);
            printf("%d\n",ans);
        }
    }
}
