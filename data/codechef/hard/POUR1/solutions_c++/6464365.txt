int gcd(int a, int b) {
    while (b != 0) {
        int c = a % b;
        a = b;
        b = c;
    }
    return a;
}

int pourWater(int a, int b, int c){
    
    int steps = 0;
    int vesselOne = 0, vesselTwo = 0;
    
    while (vesselOne!=c && vesselTwo!=c) {
        
        if (vesselOne == 0) {
            vesselOne = a;
        }
        else if (vesselTwo == b) {
            vesselTwo = 0;
        }
        else {
            int remain = b - vesselTwo;
            vesselOne -= remain;
            vesselTwo = b;
            
            if(vesselOne < 0){
                vesselTwo += vesselOne;
                vesselOne = 0;
            }
            
        }
        steps++;
    }
    
    return steps;

}
