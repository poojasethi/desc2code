#include <bits/stdc++.h>
using namespace std;
 
typedef pair<int,int> ii;
typedef pair<ii,int> iii;
 
bool vis1[40001][2];
bool vis2[2][40001];
 
int gcd(int a, int b)
{
	if(b==0)
		return a;
	return gcd(b,a%b);
}
 
int main()
{
	int t;
	scanf("%d",&t);
 
	for(int casenum=1; casenum<=t; casenum++)
	{
		int a,b,c;
		scanf("%d %d %d",&a,&b,&c);
 
		if(b<a)
			swap(a,b);
 
		if(c>b || c%gcd(a,b)!=0)
		{
			puts("-1");
			continue;
		}
 
		for(int i=0; i<=a; i++)
			vis1[i][0] = vis1[i][1] = false;
		for(int i=0; i<=b; i++)
			vis2[0][i] = vis2[1][i] = false;
 
		vis1[0][0] = true;
		vis2[0][0] = true;
		deque<iii> dq;
		dq.push_back(iii(ii(0,0),0));
 
		int ans = -1;
		while(dq.size())
		{
			iii tmp = dq.front();
			dq.pop_front();
 
			int c1 = tmp.first.first;
			int c2 = tmp.first.second;
			int dist = tmp.second;
 
			if(c1==c || c2==c)
			{
				ans = dist;
				break;
			}
 
			deque<ii> temp;
			temp.push_back(ii(a,c2));
			temp.push_back(ii(0,c2));
			temp.push_back(ii(c1,b));
			temp.push_back(ii(c1,0));
			int diff = min(c1,b-c2);
			temp.push_back(ii(c1-diff,c2+diff));
			diff = min(a-c1,c2);
			temp.push_back(ii(c1+diff,c2-diff));
 
			for(int i=0; i<temp.size(); i++)
			{
				if(temp[i].first==0 || temp[i].first==a)
				{
					int p1 = temp[i].first?1:0;
					if(vis2[p1][temp[i].second])
						continue;
					vis2[p1][temp[i].second] = true;
				}
				else
				{
					int p2 = temp[i].second?1:0;
					if(vis1[temp[i].first][p2])
						continue;
					vis1[temp[i].first][p2] = true;
				}
 
				dq.push_back(iii(ii(temp[i].first,temp[i].second),dist+1));
			}
		}
 
		printf("%d\n",ans);
	}
 
	return 0;
}
