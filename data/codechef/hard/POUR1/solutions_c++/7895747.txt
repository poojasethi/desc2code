#include<bits/stdc++.h>
using namespace std;
int gcd(int a,int b)
{
	if(b==0)
	return a;
	else
	return gcd(b,a%b);
}
int solve(int a,int b,int c)
{
int ca=0,cb=0,p,r=0;
while(ca!=c && cb!=c)
{
if(ca==0)
ca=a;
else if(cb==b)
cb=0;
else
{
int z=b-cb;
int y=ca;
p=min(y,z);
ca=ca-p;
cb=cb+p;
}
r++;
}
return r;
}
int main()
{
	int t,a,b,c,g;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d%d",&a,&b,&c);
		if(a<c && b<c)
		printf("-1\n");
		else if(c%gcd(a,b)!=0)
		printf("-1\n");
		else
		{
			printf("%d\n",min(solve(a,b,c),solve(b,a,c)));
		}
	}
	return 0;
}
