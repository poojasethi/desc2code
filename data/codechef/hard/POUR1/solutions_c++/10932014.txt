#include<cstdio>
#include<algorithm>
using namespace std;
int gcd(int a,int b){
    if(a<b){
        int tmp=a;
        a=b;
        b=tmp;
    }
    while(b){
        int tmp=a%b;
        a=b;
        b=tmp;
    }
    return a;
}
int main(){
int t;
scanf("%d",&t);
while(t--){
    int capa,capb,c;
    scanf("%d %d %d",&capa,&capb,&c);
    if(capa<capb){
        int tmp=capa;
        capa=capb;
        capb=tmp;
    }
    if(c>capa || c%gcd(capa,capb)){
        printf("-1\n");
        continue;
    }
    if(c==0){
        printf("0\n");
        continue;
    }
    //trying to fill a
    int ans=1,a1,a2,a=0,b=capb;
    while(a!=c && b!=c){
    int fill_in_a=min(capa-a,b);
    a+=fill_in_a;
    b-=fill_in_a;
    ans++;
    if(a==c || b==c){
        break;
    }
    if(a==capa){
        a=0;
        ans++;
    }
    if(b==0){
        b=capb;
        ans++;
    }
    }
    a1=ans;
    //trying to fill b
    a=capa;
    b=0;
    ans=1;
    while(a!=c && b!=c){
    int fill_in_b=min(a,capb-b);
    a-=fill_in_b;
    b+=fill_in_b;
    ans++;
    if(a==c || b==c){
        break;
    }
    if(a==0){
        a=capa;
        ans++;
    }
    if(b==capb){
        b=0;
        ans++;
    }
    }
    a2=ans;
    printf("%d\n",min(a1,a2));
}
}
