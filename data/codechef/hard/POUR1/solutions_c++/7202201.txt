#include<cstdio>
#include<algorithm>

using namespace std;

int agua(int A,int B,int C){
int mov=1,a=A,b=0,dif;
    while(a!=C && b!=C){
           dif=min(a,B-b);
           b+=dif;
           a-=dif;
           mov++;
           if(a==C || b==C)
                   break;
           if(a==0){
                    a=A;
                    mov++;
           }
           if(b==B){
                b=0;
                mov++;
           }
     }
     return mov;
}

int gcd(int a,int b){
    if(b==0){
        return a;
    }
    return gcd(b,a%b);
}

int main(){
    int ntest,a,b,c;
    scanf("%d",&ntest);
        while(ntest--){
            scanf("%d%d%d",&a,&b,&c);
            if(c>a && c>b)
                  printf("-1\n");
            else if(c%gcd(a,b) != 0)
                printf("-1\n");
            else if(c==a || c==b)
                printf("1\n");
            else
               printf("%d\n",min(agua(a,b,c),agua(b,a,c)));
    }
    return 0;
}
