#include<iostream>
#include<algorithm>

using namespace std;

int gcd(int a,int b)
{
    if(b==0)
        return a;
    else
        return gcd(b,a%b);
}

int moves(int A,int B,int c)
{
    int a,b,diff,count=0;;

    // fill A;
    a=A;
    b=0;
    count++;

    // pour a to b
    while(a!=c||b!=c)
    {
        diff = min(a,B-b);
        count++;
        a = a - diff;
        b = b + diff;

        if(a==c||b==c)
            break;

        if(a==0)
        {
            a=A;
            count++;
        }
        else if(b==B)
        {
            b = 0;
            count++;
        }
    }
    return count;
}

void solve()
{
    int A,B,c;

    cin>>A>>B>>c;

    if(c>A&&c>B)
        cout<<"-1";
    else if(c%gcd(A,B)!=0)
        cout<<"-1";
    else if(c==A||c==B)
        cout<<"1";
    else{
        cout<<min(moves(A,B,c),moves(B,A,c));
    }

    cout<<"\n";
}

int main()
{
    int T;
    cin>>T;
    for(int i=0;i<T;i++)
    {
        solve();
    }
    return 0;
}
