#include<stdio.h>
#include<algorithm>
#include<math.h>
#include<queue>
using namespace std;
int mini(int bda,int chota,int c)
{
    int a=bda,b=0,steps=1;
    int val;
    while(a!=c || b!=c)
    {
        val=min(a,chota-b);
        a-=val;
        b+=val;
        steps++;
        if(a==c || b==c)
            break;
            if(a==0)
               {
                    a=bda;
                    steps++;
               }
               if(b==chota)
               {
                   b=0;
                   steps++;
               }

    }
    return steps;
}
int main()
{
    //freopen("spoj.txt","r",stdin);
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int bda,chota,c;
        scanf("%d %d %d",&bda,&chota,&c);

        if(bda ==c || chota==c)
            printf("1\n");
        else if(c>bda && c>chota)
            printf("-1\n");
        else if(c%__gcd(bda,chota)!=0)
            printf("-1\n");
        else
        {
            int ans=min(mini(bda,chota,c),mini(chota,bda,c));
            printf("%d\n",ans);
        }
    }
    return 0;
}
