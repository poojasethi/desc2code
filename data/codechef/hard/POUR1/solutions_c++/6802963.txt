#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>

using namespace std;

int main(){
	int t,a,b,c;
	scanf("%d",&t);
	while(t--){
		scanf("%d %d %d",&a,&b,&c);
		int aux = __gcd(a,b);
		if(c % aux != 0 || c > max(a,b)) printf("-1\n");
		else{
			int fp = a; int fs = 0;
			int sp = 0; int ss = b;
			int cont1 = 1; int cont2 = 1;
			while(fp != c && fs != c && sp != c && ss != c){
				
				int tfr1 = min(fp,b - fs);
				fp -= tfr1;
				fs += tfr1;
				
				int tfr2 = min(ss,a - sp);
				sp += tfr2;
				ss -= tfr2;
				
				cont1++,cont2++;
				
				
				//printf("(%d %d) (%d %d)\n",fp,fs,sp,ss);
				
				if(fp == c || fs == c || sp == c || ss == c) break;
				
				if(fp == 0) fp = a,cont1++;
				if(fs == b) fs = 0,cont1++;
				
				if(ss == 0) ss = b,cont2++;
				if(sp == a) sp = 0,cont2++;
				
				//printf("(%d %d) (%d %d)\n",fp,fs,sp,ss);
			}
			if(fp == c || fs == c) printf("%d\n",cont1);
			else printf("%d\n",cont2);
		}
	}
	return 0;
}
