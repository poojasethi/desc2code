import sys

n = int(sys.stdin.readline())
ad_list = []
ad_mat = []

for i in xrange(n):
    frnds = sys.stdin.readline().strip()
    ad_mat.append(frnds)
    ad_list.append(int(frnds,2))

result = 0

for a in xrange(n):
    for b in xrange(a+1,n):
        if (ad_mat[a][b] == '0' and (ad_list[a] & ad_list[b] != 0)):
            result += 2
            

print result
