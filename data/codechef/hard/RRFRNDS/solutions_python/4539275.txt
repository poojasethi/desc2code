friendrelations=int(input(""))
edges=[0]*friendrelations
tobin=[0]*friendrelations
for g in range (0,friendrelations):
    x=raw_input("")
    edges[g]=x
    tobin[g]=int(x,2)
ans=0
for g in range (friendrelations):
    for y in range (g+1,friendrelations):
        if (edges[g][y]=='0'):
            if ((tobin[g]&tobin[y])!=0):
                ans+=2
print(ans)
