#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#include<list>
#include<vector>
#define ll long long

using namespace std;


ll adj[2001][2001];

vector<int> v[20001];


int main()
{
	ll n,i,j,k,count=0;
	string str;
	cin>>n;
	
	for(i=0;i<n;i++)
	{
		cin>>str;
		for(j=0;j<n;j++)
		{
			if(str[j]=='1')
			{
	         adj[i][j]=1;
			v[i].push_back(j);
		    }
		    else
		    adj[i][j]=0;
		}
	}
	
	
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n && j!=i;j++)
		{
			if(adj[i][j]==0)
			{
				for(k=0;k<v[i].size();k++)
				{
				if(adj[v[i][k]][j])
				{
					count++;
					break;
				}
			   }
			}
		}
	}
	
	cout<<2*count<<"\n";
	return 0;
}
