#include <bits/stdc++.h>
using namespace std;
 
char s[2002];
bitset<2000> f[2000];
 
int main()
{
	int n;
 
	scanf("%d\n",&n);
 
	for(int i=0;i<n;i++)
	{
		gets(s);
		for(int j=0;j<n;j++)if(s[j]=='1')f[i].set(j);
	}
 
	int ans=0;
 
	for(int i=0;i<n;i++)
	for(int j=i+1;j<n;j++)
	if(!f[i].test(j) && (f[i]&f[j]).any())ans+=2;
 
	printf("%d\n",ans);
}
