#include<iostream>
using namespace std;
#include<bits/stdc++.h>
char arr[2020][2020];
int main()
 {
 	int n;
 	 cin>>n;
 	
 	vector<int>li[2020];
     for(int i=0;i<n;i++)
      cin>>arr[i];
      int con[2020];
      for(int i=0;i<n;i++)
       {
       	int c=0;
       	 for(int j=0;j<n;j++)
       	  {
       	  	if(arr[i][j]=='1')
       	  	 {
       	  	 	   c++;
       	  	       li[i].push_back(j);
       	  	   
						
       	  	 }
       	  	  con[i]=c;
       	  }
       }
       int res=0;
         for(int i=0;i<n;i++)
          {
          	 for(int j=i+1;j<n;j++)
          	  {
          	  	 if(arr[i][j]=='0')
          	  	  {
          	  	  	 for(int k=0;k<con[i];k++)
          	  	  	  {
          	  	  	  	if(arr[li[i][k]][j]=='1')/* i is connected with k and k is connected with j*/
          	  	  	  	 {
          	  	  	  	 	res++;
          	  	  	  	 	break;
          	  	  	  	 }
          	  	  	  }
          	  	  }
          	  }
          }
           cout<<2*res<<endl;
       return 0;
 }
