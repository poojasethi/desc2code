#include<bits/stdc++.h>
using namespace std;
int mat[2010][2010];
main()
{
	bitset<2010> frnd[2010];
	int i,j,n;
	char str[2010];
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%s",str);
		for(j=0;j<n;j++)
		{
			mat[i][j]=str[j]=='1'?1:0;
			if(str[j]=='1')
			frnd[i].set(j);
		}
	}
	int ans=0;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(i!=j)
			{
				//bitset<2010> temp;
				//temp.set(j);
				if(!frnd[i].test(j) && (frnd[i]&frnd[j]).any())
				ans++;
			}
		}
	}
	printf("%d\n",ans);
}
