#include<stdio.h>
#include<iostream>
using namespace std;
unsigned long long  a[2005][50];
char inp[2005][2005];
int main()
{
int n;
scanf("%d",&n);
for(int i=0;i<n;i++)
{
scanf("%s",inp[i]);
for(int j=0;j<n;j++)
{
if(inp[i][j]=='1')
{

a[i][j/64]|=1ll<<(j%64);
}
}
}
int grp=(n+63)/64,cnt=0;
for(int i=0;i<n;i++)
{
for(int j=0;j<n;j++)
{
    if(i==j)
    continue;
if(a[i][j/64]&(1ll<<(j%64)))
continue;
for(int k=0;k<=grp;k++)
{
if(a[i][k]&a[j][k])
{
cnt++;
break;
}
}
}
}
printf("%d\n",cnt);
return 0;
}
