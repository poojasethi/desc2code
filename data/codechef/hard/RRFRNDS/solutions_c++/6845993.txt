#include<iostream>
#include<cstdio>
#include<bitset>
#include<algorithm>

//	gc

#define in(a) scanf("%d",&a)
#define out(a) printf("%d\n",a)
#define ins(a) scanf("%s",&a);
#define fo(i,n) for(i=0;i<n;i++)
#define lim 2010
#define bset bitset<lim>


using namespace std;

bset val[lim];
char str[lim][lim];

int main()
{
	int cnt,i,n,j;
	
	in(n);
	
	fo(i,n)
	{
		ins(str[i]);
		
		val[i].reset();
		
		fo(j,n)
			if(str[i][j]=='1')	val[i].set(j);
	}
	
	cnt=0;
	
	fo(i,n)
	{
		bset tem;
		
		tem.reset();
		
		fo(j,n)
			if(str[i][j]=='1')
				tem.set(j);
			
		fo(j,n)
			if(i!=j && str[i][j]=='0')
				if( (tem & val[j]).any() )
					cnt++;
				
	}
	
	out(cnt);
	
	return 0;
}
