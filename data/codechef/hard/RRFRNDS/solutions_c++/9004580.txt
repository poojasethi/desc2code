#include<stdio.h>
#include<stdlib.h>

int a[2010][2010],b[2010][2010],l[2010];

int main()
{
	int i,j,k,n;
	long long int count;
	char x;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%c",&x);
		k=0;
		for(j=0;j<n;j++)
		{
			scanf("%c",&x);
			a[i][j]=x-'0';
			if(x=='1')
			b[i][k++]=j;
		}
		l[i]=k;
		
	}
	count=0;
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(a[i][j]!=1)
			{
				for(k=0;k<l[i];k++)
				{
					if(a[b[i][k]][j])
					{
						count+=2;
						break;
					}
				}
			}
		}
	}
	printf("%lld\n",count);
	return 0;
}
