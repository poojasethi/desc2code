char a[2001][2001] ;

int adj[2001][2001] ;

int len[2001] ;

int countx = 0 ;

int main()
{
    ios_base::sync_with_stdio(false) ;

    cin.tie(0) ;

    int N ;

    cin >> N ;

    for(int i = 0 ; i < N ; i++)
        {
            cin >> a[i] ;

            int k = 0 ;

            for(int j = 0 ; j < N ; j++)
            {
                if(a[i][j] == '1')
                    adj[i][k++] = j ;
            }

          len[i] = k ;
        }

    for(int i = 0 ; i < N - 1 ; i++)
    {
        for(int j = i + 1 ; j < N ; j++)
        {
            if(a[i][j] == '0')
            {
                for(int k = 0 ; k < len[i] ; k++)
                {
                    if(a[adj[i][k]][j] == '1')
                    {
                        countx++ ;
                        break ;
                    }

                }
            }
        }
    }


    cout << countx * 2 << endl ;

    return 0 ;
}
