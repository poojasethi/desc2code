#include<iostream>
#include<queue>
#include<algorithm>
#include<cstring>
#include<cstdio>
#include<bitset>
#include<stdlib.h>
using namespace std;
 
char temp[2001];
bitset<2001> t[2001];            
int list[2001][2001], len[2001]; 
#define sf(x) scanf("%d",&x);
#define pf(x) printf("%d",x);
#define pn() printf("\n");
 
int main(){
 
	int n, count = 0;
	sf(n)
	for (int i = 0; i < n; ++i)
	{
		scanf("%s", temp);
		int k = 0;
		for(int j = 0; j < n; ++j)
		{
			if(temp[j] == '1')
			{
				t[i].set(j);
				list[i][k++] = j;
			}
		}
		len[i] = k;
	}
 
	for(int i = 0; i < n; ++i)
	{
		for(int j = i+1; j < n; ++j)
		{
			if(t[i].test(j)) continue;
			for(int k = 0; k < len[i]; ++k)
			{
				if(t[list[i][k]].test(j))
				{
					count++;
					break;
				}
			}
		}
	}
 
	pf(count * 2)
}
