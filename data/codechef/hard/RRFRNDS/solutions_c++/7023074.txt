#include<bits/stdc++.h>
using namespace std;

int n;

char s[2010][2010];
int b[2010][2010]={0};
int qw[2010]={0};

int main ( ){

   long long int ce = 0;

     int  n , t , i, j ,k ;

    scanf("%d",&n);

    for (  i = 0 ; i < n ; i++ ){
        scanf("%s",s[i]);
        t = 0;
        for (  j = 0 ; j < n ; j++ ){
            if(s[i][j] == '1' )
            {
                b[i][t++] = j;

            }

        }
        qw[i] = t ;
        //cout<<s[i]<<endl;
    }

    for (  i = 0 ; i < n ;i++ ){
        for ( j = i+1 ; j < n ; j++ ){
                if(s[i][j] == '0'){
                    for ( k = 0 ; k < qw[i] ; k++ ){
                        if( s[b[i][k]][j] == '1' ){
                            ce++;
                            break;
                    }
                }
            }
        }
    }

   printf("%lld\n",2*ce) ;

    return 0;
}
