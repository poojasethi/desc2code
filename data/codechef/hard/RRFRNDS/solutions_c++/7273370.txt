#include<bits/stdc++.h>
#define MAXN 2005
#define ll long long
using namespace std;
char G[MAXN][MAXN];
int fl[MAXN][MAXN];	//fl->friend list
int cnt[MAXN];                //c[i]->number of friends of i
ll req;
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	scanf("%s",G[i]);
	
	for(int i=0;i<n;i++)
	{
		int k=0;
		for(int j=0;j<n;j++)
		{
			if(G[i][j]=='1')
			{
				fl[i][k]=j;
				k++;
			}
		}
		cnt[i]=k;
	}
	for(int i=0;i<n;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			if(G[i][j]=='0')
			{
			for(int k=0;k<cnt[i];k++)
			{
				if(G[fl[i][k]][j]=='1')
				{
					req++;
					break;
				}
			}
			}
		}
	}
	printf("%lld\n",2*req);
}
