#include<iostream>
#include<queue>
#include<algorithm>
#include<cstring>
#include<cstdio>
#include<stdlib.h>
// #include <bits/stdc++.h>
using namespace std;

char mat[2001][2001];              // Adjacency matrix
int list[2001][2001], len[2001];   // Adjacency list
#define sf(x) scanf("%d",&x);
#define pf(x) printf("%d",x);
#define pn() printf("\n");

int main(){

	int n, count = 0;
	sf(n)
	for (int i = 0; i < n; ++i)
	{
		scanf("%s", mat[i]);
		int k = 0;
		for(int j = 0; j < n; ++j)
		{
			if(mat[i][j] == '1') list[i][k++] = j;
		}
		len[i] = k;
	}

	for(int i = 0; i < n; ++i)
	{
		for(int j = i+1; j < n; ++j)
		{
			if(mat[i][j] != '0') continue;
			for(int k = 0; k < len[i]; ++k)
			{
				if(mat[list[i][k]][j]=='1')
				{
					count++;
					break;
				}
			}
		}	
	}

	pf(count * 2)
}
