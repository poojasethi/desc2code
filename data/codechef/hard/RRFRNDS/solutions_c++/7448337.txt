#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define sch(x) scanf("%c",&x)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int
int a[2001][2001]={0};

int main()
{
    int n,i,j;
    char x;
    si(n);
    for(i=0;i<n;++i)
    {
	sch(x);
	for(j=0;j<n;++j)
	{
	    sch(x);
	    a[i][j]=x-'0';
	}
    }
    ll ans=0;
    ll temp,k;
    vector< ll > v[n];
    for(i=0;i<n;++i)
    {
	temp=0;
	k=31;
	for(j=0;j<n;++j)
	{
	    if(k<0)
	    {
		v[i].push_back(temp);
		temp=0;
		k=31;
	    }
	    if(a[i][j]==1)
		temp = temp | (1ll<<k);
	    k--;
	}
	v[i].push_back(temp);
    }
    for(i=0;i<n;++i)
    {
	for(j=i+1;j<n;++j)
	{
	    if(a[i][j]==1) continue;
	    int flag=0;
	    for(k=0;k<v[i].size();++k)
	    {

//		cout<<v[i][k]<<" ";
		if( (v[i][k]&v[j][k])>0 )
		{
		    flag=1;
		    break;
		}
	    }
//		nl;
	    if(flag)
		ans++;
	}
    }
    ans *= 2;
    pll(ans); nl;
    return 0;
}
