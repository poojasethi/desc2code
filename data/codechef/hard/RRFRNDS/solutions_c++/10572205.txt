int main(){
	int n,temp;
	string s;
	cin >>n;
	vector < vector <int> > frnd(n, vector <int> (n));
	vector < list<int> > graph(n);
	for (int i = 0; i < n; ++i)
	{
		cin >> s;
		for (int j = 0; j < n; ++j)
		{
			// scanf("%d",&temp);
			if (s[j]=='1')
			{
				frnd[i][j]=1;
				graph[i].push_back(j);
			}
			else frnd[i][j]=0;
		}
	}
	int cnt =0;
	for (int i = 0; i < n; ++i)
	{
		for (int j = 0; j < n; ++j)
		{
			if (frnd[i][j]==0 && i!=j)
			{
				list<int>::iterator it;
				for (it=graph[j].begin(); it!=graph[j].end(); ++it)
				{
					if (frnd[i][(*it)]==1)
					{
						cnt++;
						break;
					}
				}
			}
		}
	}
	cout << cnt << endl;
	return 0;

}
