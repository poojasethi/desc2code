#include<bits/stdc++.h>
#define AIN(a,b,c) assert(a<=b && b<=c)
using namespace std;
typedef long long ll;
char str[2001][2001];
ll arr[2001][50];
int main()
{
	std::ios_base::sync_with_stdio(false);
	ll n,i,j,k,y,z,a,b;
	cin>>n;
	//AIN(1,n,2000);
	for(i=0;i<n;i++)
	{
		cin>>str[i];
		for(j=0;j<n;j++)
		{
			if(str[i][j]=='1')
			{
				a=j/64;
				b=j%64;
				arr[i][a]|=1LL<<b;
			}
		
		}
		
	}
	ll count=0;
	ll x=(n+63)/64;
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(i!=j)
			{
				y=j/64;
				z=j%64;
				if(!(arr[i][y]&(1LL<<z)))
				{
					for(k=0;k<=x;k++)
					{
						if(arr[i][k]&arr[j][k])
						break;
					}
					if(k<=x)
					{
						count++;
					}
				}
			}
		}
	}
	cout<<count<<endl;
}
