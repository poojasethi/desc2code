#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<iostream>
using namespace std;

int b[2005][2005],len[2005];
char str[2005][2005];
int main()
{
	int n,i,j,k;
	long long int cont=0;
	//scanf("%d",&n);
	cin>>n;
	for(i=0;i<n;i++)
	{
		k=0;
		scanf("%s",str[i]);
		for(j=0;j<n;j++)
		{
			if(str[i][j]=='1')
				b[i][k++]=j;
		}
		len[i]=k;
	}
	for(i=0;i<n;i++)
	{
		for(j=i+1;j<n;j++)
		{
			if(str[i][j]=='0')
			{
				for(k=0;k<len[i];k++)
				{
					if(str[b[i][k]][j]=='1')
					{
						cont++;
						break;
					}
				}
			}
		}
	}
	//printf("%lld\n",2*cont);
	cout<<2*cont<<endl;
	return 0;
}
