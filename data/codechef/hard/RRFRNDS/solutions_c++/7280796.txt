#include<bits/stdc++.h>
using namespace std;

int N;
bitset<2003> ar[2003];
int main()
{
    cin>>N;
    char ch=getchar();
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            ch=getchar();
            if(ch=='1')
            {
                ar[i].set(j);
            }
        }
        ch=getchar();
    }

    int ans=0;
    for(int i=0;i<N;i++)
    {
        for(int j=0;j<N;j++)
        {
            if(i==j)continue;
            if(ar[i][j]!=1)
            {
                if((ar[i]&ar[j]).any())ans++;
            }
        }
    }
    cout<<ans<<endl;
}
