#include <iostream>
#include <cstdio>
#include <cstring>
#define maxn 2005
#include <vector>
#define ll long long 
using namespace std;
int n;
vector<int>V[maxn];
int val[maxn][maxn];
ll ans=0;
void input(){
	cin>>n;
    for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++){
			scanf("%1d",&val[i][j]);
		}
	}
	for(int i=0;i<maxn;i++){
	    V[i].clear();
	}
  	ans=0;
}
void doit()
{
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			if(val[i][j]){
			     V[i].push_back(j);
			}
		}
	}
	for(int i=0;i<n;i++){
		for(int j=i+1;j<n;j++)
		{
			if(val[i][j]==0){
	              for(int k=0;k<V[i].size();k++)
				  {
					  if(val[V[i][k]][j]){
					          ans++;
							  break;
					  }
				  }
			}
		}
	}
	cout<<ans*2<<endl;
}
int main()
{
	//freopen("in.txt","r",stdin);
	input();
	doit();
	return 0;
}
