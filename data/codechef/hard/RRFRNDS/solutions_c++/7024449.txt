#include <bits/stdc++.h>

using namespace std;

int main()
{
	int n;
	scanf("%d", &n);

	vector<bitset<2005> > v(n);

	for(int i=0; i<n; i++) {
		cin >> v[i];
	}

	int cnt=0;
	for (int i=0; i<n; i++) {
		for (int j=0; j<n; j++) {
			if (j!=n-i-1 && !v[i][j]) {
				bitset<2005> c = v[i]&v[n-j-1];
				if (c.any())
					cnt++;
			}
		}
	}

	printf("%d\n", cnt);

	return 0;
}
