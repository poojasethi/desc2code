#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
int id[2001][2001]={0};

int main()
{
	int N;
	long long count=0;
	cin>>N;
	char input[N][N];
	
	int total[N];
	
	
	for(int i=0;i<N;i++)
		cin>>input[i];
 
 
	for(int i=0;i<N;i++)
	{
		int t=0;
		for(int j=0;j<N;j++)
		{
			if(input[i][j]=='1')
    				id[i][t++]=j;
    		}
    		total[i]=t;
	}
 
	for(int i=0;i<N;i++)
	{
		for(int j=i+1;j<N;j++)
		{
			if(input[i][j]=='0')
			{
				for(int k=0;k<total[i];k++)
				{
					if(input[id[i][k]][j]=='1')
					{
						count+=2;
						break;
					}
				}
			}
		}
	}
 
	cout<<count<<endl;
	return 0;
}
