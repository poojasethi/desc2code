#include <iostream>
#include <stdio.h>
#include <bitset>
using namespace std;

int main()
{
    int n,c=0,i,j;
    string s;
    scanf("%d",&n);
    bitset<2005> b[2005];
    for(i=0;i<n;i++)
    {
        cin>>s;
        for(j=0;j<n;j++)
        {
            if(s[j]=='1')
                b[i].set(j);
        }
    }
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(!b[i].test(j))
            {
                if((b[i]&b[j]).any())
                {
                    c+=2;
                }
            }
        }
    }
    printf("%d",c);
    return 0;
}
