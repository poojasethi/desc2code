//<---- Solution ---->
vector<string> S;
void solve(){
	int n;
	inp(n);
	string s;
	vector< vector<int> > V(n);
	vector< vector<int> > M(n,vector<int>(n));
	Rep(i,n){
		cin>>s;
		S.pb(s);
		Rep(j,s.size()){
			if(s[j]=='1' && !M[i][j] && !M[j][i]){
				V[i].pb(j);
				V[j].pb(i);
			}
		}
	}
	int cnt=0;
	Rep(i,n){
		//op(V[i].size());
		Rep(j,n){
			if(S[i][j]=='0'&& M[i][j]!=1 && i!=j)
			Rep(k,V[j].size()){
				if(S[i][V[j][k]]=='1'){
					cnt++;
					M[i][j]=1;
					break;	
				}
			}
		}
	}
	op(cnt);
}

int main(){
	solve();
	return 0;
}
