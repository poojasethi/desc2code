#include <iostream>
using namespace std;

#define max 2000

int arr[max][max];
char str[max][max];
int len[max] = {0};
int main(){

	int n,k;
	scanf("%d\n",&n);
	for(int i=0;i<n;i++){
	gets(str[i]);
	k=0;
	for(int j=0;j<n;j++){
	if(str[i][j] == '1'){
	arr[i][k]= j; k++;
	}
	len[i]=k;
	}}
	
	int count = 0;

	for(int i=0;i<n-1;i++){
	for(int j=i+1;j<n;j++){
		if(str[i][j]=='0'){
			for(int k=0;k<len[i];k++){
				if(str[j][arr[i][k]]=='1'){
					count++;break;
				}
			}
		}
	}
	
	}
	
	printf("%d\n",count*2);
	return 0;
}
