#include <stdio.h>
#define DEBUG 0

char frnds[2002][2002];
int adj[2002][2002] = {0};
int len[2002] = {0};

int scanInt()
{
  int n=0;
  char ch=getchar();

  while(ch<'0' || ch>'9') ch=getchar();

  while(ch >= '0' && ch<='9')
    {
      //      printf("%c %d\n", ch, n);
      n += (n<<3) + (n<<1) + (ch-'0');
      ch=getchar();
    }

  return n;
} 
  

int main()
{
  int N;
  scanf("%d", &N);

  for(int i=0; i<N; i++)
    {
      scanf("%s", frnds[i]);

      len[i] = 0;
      
      for(int j=0; j<N; j++)
	{
	  if(frnds[i][j] == '1')
	    {
	      adj[i][len[i]++] = j;
	    }
	}
    }

  if(DEBUG)
    {
      printf("Printing graph matrix\n");
      
      for(int i=0; i<N; i++)
	{
	  for(int j=0; j<N; j++)
	    {
	      printf("%c ", frnds[i][j]);
	    }
	  printf("\n");
	}

      printf("Printing adjacency lists\n");
  
      for(int i=0; i<N; i++)
	{
	  for(int j=0; j<len[i]; j++)
	    {
	      printf("%d ", adj[i][j]);
	    }
	  printf("\n");
	}	  
    }  

  long long int requests = 0;
  
  for(int i=0; i<N; i++)
    {
      for(int j=i+1; j<N; j++)
	{
	  if(frnds[i][j] == '0')
	    {
	      for(int k=0; k<len[i]; k++)
		{
		  if(frnds[adj[i][k]][j] == '1')
		    {
		  
		      if(DEBUG)
			{
			  printf("%d -> %d -> %d\n", i, adj[i][k], j);
			}
		      
		      requests++;
		      break;
		    }
		}
	    }
	}
    }

  if(DEBUG)
    {
      printf("Printing answer\n");
    }
  
  printf("%lld\n", requests*2);
  
  return 0;
}
