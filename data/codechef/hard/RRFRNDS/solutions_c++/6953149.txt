#include <bits/stdc++.h>
#define mod 1000000007
using namespace std;

string s[2005];
bitset<2005>bset[2005];

int main()
{
    int n,i,j,u,w,k;
    long long int ans;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        cin>>s[i];
    for(i=0;i<n;i++)
    {
        bset[i].reset();
        for(j=0;j<n;j++)
        {
            if(s[i][j]=='1')
                bset[i].set(j);
        }
    }
    ans=0;
    for(i=0;i<n;i++)
    {
        for(j=i+1;j<n;j++)
        {
            if(s[i][j]=='0'&&(bset[i]&bset[j]).any())
                ans=ans+2;
        }
    }
    printf("%lld\n",ans);
    return 0;
}
