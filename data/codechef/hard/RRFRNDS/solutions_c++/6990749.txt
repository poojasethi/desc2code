#include<iostream>
#include<cstdio>
#include<bitset>
using namespace std;

int main()
{
    bitset<2005> b[2005];
    string str;
    int n,count=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        cin >> str;
        for(int j=0;j<n;j++)
        {
            if(str[j]=='1')
                b[i].set(j);
        }
    }
    for(int i=0;i<n;i++)
    {
        for(int j=i+1;j<n;j++)
        {
            if(b[i].test(j)==0)
            {
                if((b[i]&b[j]).any())
                    count+=2;
            }
        }
    }
    printf("%d\n",count);

    return 0;
}
