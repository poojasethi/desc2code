# include <bits/stdc++.h>
# define mod 1000000007
# define rep(i,a,b) for(i=a;i<=b;i++)
# define repd(i,a,b) for(i=a;i>=b;i--)
# define saaf(h) (h).clear()
# define pb push_back
# define mp make_pair

using namespace std;
typedef long long ll;

int main () {
	int t,n,a[8],i,j,k;
	int g[256];
	bool h[256];
	g[0]=g[1]=0,g[2]=1;
	for ( i=3 ; i < 256 ; i++ ) {
		memset(h,false,sizeof h);
		
		for (j=1 ; j < i-1 ; j++ ) {
			h[g[j]^g[i-j]]=true;
			h[g[j]^g[i-j-1]]=true;
		}
		for(j=0;h[j];j++);
		g[i]=j;
	}

	scanf("%d",&t);
	while (t--) {
		
		scanf("%d",&n);
		for ( i=0 ; i < n; i++) 
			scanf("%d",a+i);
		int ans=0;
		for (i=0 ; i < n ; i++)
			ans ^= g[a[i]];
		printf(ans?"Alice\n":"Bob\n");
	}
	return 0;
}
