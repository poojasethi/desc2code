#include<iostream>
#include<cstdio>
#include<set>
using namespace std;

int main()
{
    int *grundy=new int[256];
    grundy[0]=0;
    grundy[1]=0;
    for(int i=2;i<=256;i++)
    {
        set <int> s;
        for(int j=1;j<=(i-1);j++)
        {
               s.insert(grundy[j]^grundy[i-j]);
               if(i-j!=1)
               s.insert(grundy[j]^grundy[i-j-1]);
        }
        int ret=0;
        while(s.find(ret)!=s.end()) ret++;
        grundy[i]=ret;
        s.clear();
    }
    int l,t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        int exor=0;
        for(int i=0;i<n;i++)
        {
            scanf("%d",&l);
            exor^=grundy[l];
        }
        if(exor) printf("Alice\n");
        else printf("Bob\n");
    }
    return 0;
}
