void pre(){
	FOR(i,0,MN-1) {
		CLR(flag);
		FOR(j,2,i-2) {
			flag[grand[j] ^ grand[i-j]] = true;
			if ((i % 2 == 1) && (j % 2 == 1)) {
				flag[grand[j] ^ grand[i-j+1]] = true;
				if (i-j-1 >= 2) flag[grand[j] ^ grand[i-j-1]] = true;
			}
		}
		FOR(j,0,MN-1) if (!flag[j]) {
			grand[i] = j;
			break;
		}
	}

}
