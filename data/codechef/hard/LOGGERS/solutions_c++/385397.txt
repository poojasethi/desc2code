#include <cstdio>
using namespace std;

int sg[251], flag[200];

int main()
{
    int cases, n, cookie;
    sg[1] = 0;
    for (int i = 2; i <= 250; i++) {
        ++cookie;
        for (int j = 1; j <= i-j; j++)
            flag[sg[j]^sg[i-j]] = cookie;
        for (int j = 1; j <= i-j-1; j++)
            flag[sg[j]^sg[i-j-1]] = cookie;
        for (; cookie == flag[sg[i]]; sg[i]++);
    }
    
    for (scanf("%d", &cases); cases--; ) {
        int a, ans = 0;
        scanf("%d", &n);
        for (int i = 0; i < n; i++)
            scanf("%d", &a), ans ^= sg[a];
        puts(ans ? "Alice" : "Bob");
    }
}
