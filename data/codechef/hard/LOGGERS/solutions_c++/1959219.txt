typedef pair<int, int>PII;

#define NN 255

int N;
bool check[NN];
int grundy[NN];

int mex()
{
    for(int i=0; i<NN; i++)
    {
        if(!check[i])return i;
    }
}

void precompute()
{
    grundy[0] = 0, grundy[1] = 0;

    for(int i=2; i<251; i++)
    {
        memset(check, 0, sizeof(check));
        for(int j=0; j<i; j++)
        {
           check[ (grundy[j]^grundy[i-j-1]) ] = 1;
        }
        for(int j=1; j<i; j++)
        {
            check[(grundy[j]^grundy[i-j])] = 1;
        }
        grundy[i] = mex();
    }

}

void solve()
{
 int ans=0, len;
 scanf("%d", &N);
 while(N--)
 {
     scanf("%d", &len);
     ans^=grundy[len];
 }
 printf("%s\n", ans==0?"Bob":"Alice");

}

int main()
{
    #ifndef ONLINE_JUDGE

    #endif
    precompute();

    int t;
    scanf("%d", &t);
    while(t--)
    {
       solve();
    }

}
