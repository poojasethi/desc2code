#include<cstdio>
#include<cstring>
int i,j,T,N,x,Ans;
int f[505],g[255];
int main()
{
	for(i=0;i<=250;++i)
	{
		memset(f,0,sizeof(f));
		for(j=1;j<=(i-2);++j) f[g[j]^g[i-j]]=f[g[j]^g[i-j-1]]=1;
		if(i>1) f[g[i-1]^g[1]]=1;
		for(g[i]=0;f[g[i]];++g[i]);
	}
	for(scanf("%d",&T);T--;)
	{
		scanf("%d",&N),Ans=0;
		for(i=0;i<N;++i) scanf("%d",&x),Ans^=g[x];
		if(Ans) printf("Alice\n");
		else printf("Bob\n");
    }
	return 0;
}
