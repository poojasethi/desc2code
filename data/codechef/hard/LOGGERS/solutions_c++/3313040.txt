#include<iostream>
#include<algorithm>
#include<cstring>
#include<cstdio>
#include<set>
#include<algorithm>
using namespace std;
int g[260];
int main()
{
		bool x[260];
		g[1]=0;
		g[2]=1;
		for(int i=3;i<260;i++)
		{
			g[i]=0;
			memset(x,0,260);
			for(int j=1;j<i-1;j++)
			{
				x[g[j]^g[i-j]]=1;
				x[g[j]^g[i-j-1]]=1;
				
			}
			for(int j=0;j<i;j++)
			{
				if(x[j]==0)
				{
					g[i]=j;break;
				}
			}
		}
		
	int t;
	cin>>t;
	while(t--)
	{
		int n;int ans=0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			int num;
			cin>>num;
			ans=ans^g[num];
			
		}
		if(ans==0)cout<<"Bob\n";
		else cout<<"Alice\n";
	}
}
