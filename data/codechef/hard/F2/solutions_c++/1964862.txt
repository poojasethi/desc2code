int a,b,n,l,dp[1001][501],next[501];
char s[501];
int solve(int a,int k)
{
 if(abs(b-a)>n-k||a<0||a>l)return 0;
 int &ret=dp[a][k];
 if(ret!=-1)return ret;
 ret=0;
 if(a==b)ret+=1;
 if(k!=n)
 {
  ret=(ret+solve(s[k]=='+'?a+1:a-1,k+1))%mod;
  if(next[k]!=n)
  ret=(ret+solve(s[k]=='+'?a-1:a+1,next[k]+1))%mod;
 }
 return ret;
}
int main()
{
  int T,i,j,k;
  //freopen("in.txt","r",stdin);
  scanf("%d",&T);
  while(T--)
  {
   scanf("%d%d%d%d%s",&n,&l,&a,&b,s);
   if(b>n)
   {
    a-=(b-n);l-=(b-n);b-=(b-n);
   }
   if(l>b+n)l=b+n;
   int nextplus=n,nextminus=n;
   for(int i=n-1;i>=0;i--)
   if(s[i]=='+')next[i]=nextminus,nextplus=i;
   else next[i]=nextplus,nextminus=i;
   memset(dp,-1,sizeof(dp));
   printf("%d\n",solve(a,0));
  }
  return 0;
}
