#include<iostream>
#include<cstdio>
#include<string>
#include<queue>

using namespace std;

#define max(x, y) ((x) > (y) ? (x) : (y))
#define min(x, y) ((x) < (y) ? (x) : (y))

#define mod 7051954

long long int arr[501];
long long int vals[501][501];
char mas[501];

int main()
{
	int t;
	scanf("%d", &t);
	while(t--)
	{
		long long int n, l, a, b;
		scanf("%lld %lld %lld %lld", &n, &l, &a, &b);
		long long int maxdis = 0, mindis = 0, currdis = 0;
		string str;
		cin>>str;
		string::iterator it;
		int i=0;
		for(it=str.begin(); it != str.end(); ++it)
		{
			mas[i] = *it;
			//scanf("%c", &mas[i]);
			//intf("%c", mas[i]);
			if(mas[i] == '+')
				maxdis++;
			else
				mindis--;
			i++;
		}
		//cout<<endl;
		long long int rightlim = min(l - a, maxdis);
		long long int leftlim = max(-a, mindis);
		//cout<<leftlim<<"    "<<rightlim<<endl;
		if(b - a > rightlim || b - a < leftlim)
		{
			printf("0\n");
			continue;
		}
		int size = rightlim - leftlim + 1;
		for(int i=0; i<size; ++i)
			arr[i] = 0;
		for(int i=0; i<size; ++i)
		{
			for(int j=0; j<=n; ++j)
			{
				vals[i][j] = 0;
			}
		}
		vals[-leftlim][0] = 1;
		arr[-leftlim] = 1;
		queue<int> plus;
		queue<int> minus;
		plus.push(0);
		minus.push(0);
		int strpos = 0;
		long long int ans = 0;
		for(it = str.begin(); it != str.end(); ++it)
		{
			strpos++;
			if(*it == '+')
			{
				while(plus.size() != 0)
				{
					int index = plus.front();
					plus.pop();
					for(int k=0; k<size - 1; ++k)
					{
						vals[k + 1][strpos] = (vals[k + 1][strpos] + vals[k][index]) % mod;
					}
				}
			}
			else
			{
				while(minus.size() != 0)
				{
					int index = minus.front();
					minus.pop();
					for(int k=1; k<size; ++k)
					{
						vals[k - 1][strpos] = (vals[k - 1][strpos] + vals[k][index]) % mod;
					}
				}
			}
			plus.push(strpos);
			minus.push(strpos);
		}
		for(int i=0; i<=str.size(); ++i)
			ans = (ans + vals[b - a - leftlim][i]) % mod;

		cout<<ans<<endl;
	}
	return 0;
}
