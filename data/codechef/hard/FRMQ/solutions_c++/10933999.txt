#include <cstdio>
#include <cmath>
#define MAX(x, y) (x > y ? x : y)
#define MIN(x, y) (x < y ? x : y)
using namespace std;

const int N = 1e5;
int table[20][N], lg[N];
int main(){
	int n, i, j, m, x, y;
	double lg2 = log(2);
	long long ans = 0;
	scanf("%d", &n);
	for(i = 0; i < n; ++i){
		scanf("%d", &table[0][i]);
		lg[i + 1] = log(i + 1) / lg2;
	}
	scanf("%d %d %d", &m, &x, &y);
	for(j = 1; j <= lg[n]; ++j)
		for(i = 0; i + (1 << j - 1) < n; ++i)
			table[j][i] = MAX(table[j - 1][i], table[j - 1][i + (1 << j - 1)]);
	while(m--){
		j = MAX(x, y);
		i = MIN(x, y);
		ans += MAX(table[lg[j - i + 1]][i], table[lg[j - i + 1]][j + 1 - (1 << lg[j - i + 1])]);
		x += 7;
		if(x >= n - 1){
			if(n - 1 < 7)
				x %= n - 1;
			else
				x -= n - 1;
		}
		y += 11;
		if(y >= n){
			if(n < 11)
				y %= n;
			else
				y -= n;
		}
	}
	printf("%lld\n", ans);
	return 0;
}
