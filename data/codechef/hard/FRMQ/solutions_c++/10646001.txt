#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int a[100005],b[100][100005],p[100],l1[100005];
int main()
{
    int n,m,x,y,i,j,k,l,r,max1;
    long long s;
    sd(n);
    p[0]=1;
    for(i=1;p[i-1]<=n;i++)
        p[i]=p[i-1]*2;
    l1[1]=0;
    for(i=2;i<=100000;i++)
        l1[i]=l1[i/2]+1;
    for(i=0;i<n;i++)
        sd(a[i]);
    for(i=0;i<n;i++)
        b[0][i]=i;
    for(j=1;p[j]<=n;j++)
    {
        for(i=0;i+p[j]-1<n;i++)
        {
            if(a[b[j-1][i]]>a[b[j-1][i+p[j-1]]])
                b[j][i]=b[j-1][i];
            else
                b[j][i]=b[j-1][i+p[j-1]];
        }
    }
    sd(m);
    sd(x);
    sd(y);
    s=0;
    for(i=0;i<m;i++)
    {
        l=min(x,y);
        r=max(x,y);
        /*max1=a[l];
        for(j=l;j<=r;j++)
        {
            if(a[j]>max1)
                max1=a[j];
        }*/
        k=l1[r-l+1];
        max1=max(a[b[k][l]],a[b[k][r-p[k]+1]]);
        s+=max1;
        //x=(x+7)%(n-1);
        //y=(y+11)%n;
        x+=7;
        y+=11;
        while(x>=n-1)
            x-=n-1;
        while(y>=n)
            y-=n;
    }
    printf("%lld\n",s);
    return 0;
}
