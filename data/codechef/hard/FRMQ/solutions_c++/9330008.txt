#include<iostream>
#include<cstdio>
#include<algorithm>
#include<string>
#include<cstring>
using namespace std;
#define ll int
#define e cout<<endl;
ll store[21],logs[100001],n,ar[100001],rmq[17][100001];
void RMQ()
{
   ll i,j,k,x,y;
  for(i=0;i<n;i++)
     rmq[0][i]=ar[i];

  for(k=1;k<=logs[n];k++)
  {
    for(i=n-1;i>=0;i--)
    {
         rmq[k][i]=max(rmq[k-1][i],rmq[k-1][i+store[k-1]]);
    }
  }
}
ll query(ll l,ll r)
{
  ll k=logs[r-l+1];
  return max(rmq[k][l],rmq[k][r-store[k]+1]);
}
int main()
{
  ll i,j,k,m,x,y;
  long long sum=0LL;
  store[0]=1;logs[0]=-1;
  for(i=1;i<=20;i++)
    store[i]=store[i-1]<<1;
  for(i=1;i<=100000;i++)
    logs[i]=logs[i>>1]+1;

  scanf("%d",&n);
  for(i=0;i<n;i++)
   scanf("%d",&ar[i]);
   scanf("%d%d%d",&m,&x,&y);

  RMQ();
  for(i=1;i<=m;i++)
  {
    sum+=query(min(x,y),max(x,y));
     x+=7;y+=11;
    while(x>=(n-1))
    x-=(n-1);
    while(y>=n)
    y-=n;
  }
  printf("%lld",sum);
  return 0;
}
