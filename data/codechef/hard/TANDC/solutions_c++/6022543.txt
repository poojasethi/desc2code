#include<bits/stdc++.h>
using namespace std;
typedef long long lint;
#define mod 1000000007
lint m, r;
lint power(lint a, lint n)  //a^n
{
    lint ans=1;
    while(n)
    {
        if(n & 1)
            ans=(ans*a)%mod;
        a=(a*a)%mod;
        n>>=1;
    }
    return ans;
}
void solve()
{
    lint ans, temp;
    temp=((power(m, r)%mod-(m%mod))+mod)%mod;
    ans=(temp%mod *(power(m-1, mod-2)%mod));
    printf("%lld\n", (ans-(r-1)+mod)%mod);
}
int main()
{
    int test;
    scanf("%d", &test);
    while(test--)
    {
        scanf("%lld %lld", &m, &r);
        solve();
    }
    return 0;
}
