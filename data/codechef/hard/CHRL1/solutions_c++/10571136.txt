int max(int p,int q)
{
    return (p>q)?p:q;
}
int min(int p,int q)
{
    return (p<q)?p:q;
}

int main()
{
    VII a;
    int t,n,k,i,temp1,temp2,max,cost,weight,num,j;
    si(t);
    while(t--)
    {
        a.clear();
        si(n);
        si(k);
        for(i=0;i<n;i++)
        {
            si(temp1);
            si(temp2);
            a.PB(MP(temp1,temp2));
        }
        num=pow(2,n);
        max=INT_MIN;
        for(i=0;i<num;i++)
        {
            temp1=i;
            cost=0;
            weight=0;
            j=0;
            while(temp1>0)
            {
                if(temp1 & 1==1)
                {
                    cost+=a[j].F;
                    weight+=a[j].S;
                }
                temp1=temp1>>1;
                j++;
            }
            if(cost<=k)
            {
                if(weight>max)
                {
                    max=weight;
                }
            }
        }
        dout(max);
    }
    return 0;
}
