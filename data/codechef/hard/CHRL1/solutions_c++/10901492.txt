long cost[10], weight[10];

long Knapsack(long k, long n) {
	if(k == 0 || n == 0) return 0;

	if(cost[n-1] > k) return Knapsack(k, n-1);

	else return max(weight[n-1] + Knapsack(k-cost[n-1], n-1) , Knapsack(k, n-1));
}

int main() {
	BOOST;

		long n, k;
		cin>>n>>k;

		rep(i,0,n) cin>>cost[i]>>weight[i];

		cout << Knapsack(k, n) << "\n";
	}

	return 0;
}
