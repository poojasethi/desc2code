#include <iostream>
#include <cmath>
#include <climits>
#include <cstdlib>
using namespace std;
int cost[1000],weight[1000];
int knapsack(int n,int k)
{
    // cout << k << endl;
    if(k == 0)
        return 0;
    if(k < 0)
    {
        cout << "error" << endl;
        exit(1);
    }
    if(n == 0)
        if(cost[n] <= k)
            return weight[n];
        else
            return 0;
    if(k-cost[n] < 0)
        return knapsack(n-1,k);
    else
        return max(knapsack(n-1,k) , weight[n] + knapsack(n-1,k-cost[n]));
}
int main()
{
    int t;
    cin >> t;
    int n,k;
    while(t--)
    {
        cin >> n >> k;
        for(int i = 0 ; i < n; i++)
        {
            cin >> cost[i] >> weight[i];
        }
        cout << knapsack(n-1,k) << endl;
    }
	return 0;
}
