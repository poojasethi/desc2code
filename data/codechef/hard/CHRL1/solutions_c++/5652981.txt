#include<stdio.h>
struct hello
{
    long int c,w;
}a[12];
int main()
{
    int n,k,i,t,p,j;
    long int w1,c1,mx,x;
    scanf("%d",&t);
    while(t--)
    {
        mx=0;
        scanf("%d%d",&n,&k);
        for(i=0;i<n;i++)
            scanf("%ld%ld",&a[i].c,&a[i].w);
        p=1;
        for(i=1;i<=n;i++)
            p=p*2;
        p--;
        for(i=0;i<=p;i++)
        {
            x=i;
            j=0;
            w1=c1=0;
            while(x>0)
            {
                if(x&(1))
                {
                    w1+=a[j].w;
                    c1+=a[j].c;

                }
                if(c1>k)
                    break;
                j++;
                x=x>>1;
            }
            if(c1<=k && w1 >mx)
                mx=w1;
        }
        printf("%ld\n",mx);
    }


    return 0;
}
