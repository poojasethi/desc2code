// CHRL1 - Codechef
#include <bits/stdc++.h>
#define maxn 11

using namespace std;

int n, k, cost[maxn], weight[maxn];
long long ans,GT,KL;

void thu(int i) {
    if (i>n) {ans=max(ans,KL); return;}
    thu(i+1);
    if (GT+cost[i]<=k) {
        KL+=weight[i];
        GT+=cost[i];
        thu(i+1);
        KL-=weight[i];
        GT-=cost[i];
    }
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    int T; scanf("%d",&T);
    while (T--) {
        scanf("%d%d",&n,&k);
        for(int i=1;i<=n;i++) scanf("%d%d",cost+i,weight+i);
        ans=KL=GT=0;
        thu(1);
        printf("%I64d\n",ans);
    }
}
