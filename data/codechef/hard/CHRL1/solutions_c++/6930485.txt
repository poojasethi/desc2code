#define M(a,x) memset(a,x,sizeof(a))
#define sci(x) scanf("%d",&x);
#define scl(x) scanf("%lld",&x);
#define scs(x) scanf("%s",x);
#define print(x) printf("%d",x);
#define nl printf("\n")

#define F(i,a,n) for(int i=a;i<n;i++)
#define INF 100000000000000000LL
#define MAX 1000005

int main()
{
    int t;sci(t);
    while(t--) {
        int n,k;sci(n);sci(k);
        pair<int,int> v[n];
        F(i,0,n) {
            sci(v[i].first);sci(v[i].second);
        }
        long long mx=-1;
        F(i,1,1<<n) {
            long long cost=0,wt=0;
            F(j,0,n)
            {
                if(i&(1<<j)) {
                    wt+=v[j].second;
                    cost+=v[j].first;
                }
            }
            if(cost<=k) {
                mx=max(wt,mx);
            }
        }
        cout<<mx<<endl;
    }
    return 0;
}
