#define PI acos(-1.0)
#define inf INT_MAX
using namespace std;
typedef long long int LL;

LL cost[20] , weight[20] , curr = 0 , cst;

int main()
{
LL T , N , K , j , ans = 0;
scanf("%lld",&T);

while(T--)
{
	ans = 0;
	scanf("%lld %lld",&N,&K);
	for(int i=1;i<=N;i++)
		scanf("%lld %lld",&cost[i] , &weight[i]);

	for(int i=0;i<(1<<N);i++)
	{
		cst = 0;
		curr = 0;
		for(j=0;j<N;j++)
		{
			if(i&(1<<j))
			{	
				cst = cst + cost[j+1];			
				curr = curr + weight[j+1];							
			}
		}
		if(cst<=K)
			ans = max(ans , curr);
	}
	printf("%lld\n", ans);
}
return 0;
}
