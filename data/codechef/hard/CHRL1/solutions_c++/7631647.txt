int Knapsack(int sum,int idx,int cost[],int w[],int n)
{	if(idx==n) return 0;
	if(sum<cost[idx]) return Knapsack(sum,idx+1,cost,w,n);
	return max(Knapsack(sum,idx+1,cost,w,n),w[idx]+Knapsack(sum-cost[idx],idx+1,cost,w,n));
}

		cout<<Knapsack(k,0,cost,w,n)<<endl;
	}
