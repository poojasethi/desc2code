#include <iostream>
using namespace std;

int main() {
	int t;
	int n;
	long long int tot;

	cin >> t;

	for(int i = 0; i < t; i++) {
		cin >> n;
		cin >> tot;
		long long int a[n][2];
		for(int j = 0; j < n; j++) {
			cin >> a[j][0] >> a[j][1];
		}

		int s = (1 << n);
		long long int max = 0;
		for( int k = 1; k <= s; k++) {
			long long int cur = 0;
			long long int cost = 0;
			for(int l = 0; l < n; l++) {
				int z = k & (1 << l);
				if(z > 0) {
					cur = cur + a[l][1];
					cost = cost + a[l][0];
				}
			}
			if(cost <= tot) {
				if(cur > max) {
					max = cur;
				}
			}
		}
		cout << max << endl;
	}

	return 0;
}
