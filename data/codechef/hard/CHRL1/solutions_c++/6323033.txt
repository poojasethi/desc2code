using namespace std;

int main()
{
    int t,k,i,j,n,m,a,b,c,ara[11],ara1[11];
    long long res=0,weight,cost=0,Max;

    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        for(i=0; i<n; i++)
            scanf("%d%d",&ara1[i],&ara[i]);

        Max=0;
        for(i=0; i<(1<<n); i++)
        {
            cost=0;
            weight=0;
            for(j=0; j<n; j++)
            {
                if(i&(1<<j))
                {
                    weight+=ara[j];
                    cost+=ara1[j];
                }
            }

            if(cost<=k&&weight>Max)
                Max=weight;
        }
        printf("%lld\n",Max);
    }
    return 0;
}
