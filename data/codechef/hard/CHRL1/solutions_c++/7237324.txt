#include<bits/stdc++.h>
using namespace std;

const int N=12;

int cost[N], weight[N];

int knapsack(int n, int k) {
    if(n==0 || k==0) {
        return 0;
    } else if(k<cost[n]) {
        return knapsack(n-1, k);
    } else {
        return max(knapsack(n-1, k), knapsack(n-1, k-cost[n])+weight[n]);
    }
}

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n, k;
        scanf("%d %d", &n, &k);
        for(int i=1;i<=n;i++) {
            scanf("%d %d", &cost[i], &weight[i]);
        }
        printf("%d\n", knapsack(n, k));
    }
}
