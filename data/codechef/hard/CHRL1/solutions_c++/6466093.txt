#include<iostream>
#include<time.h>

using namespace std;

struct orange
{
    int weight;
    int cost;
};
int main()
{
   ios_base::sync_with_stdio(false);
   //clock_t start = clock();

   int t,N,K;
   cin>>t;
   while(t--)
   {
       cin>>N>>K;
       orange s[10];
       int i;
       int ans=0;
       for (i=0;i<N;i++) cin>>s[i].cost>>s[i].weight;
       for (i=1;i<(1<<N);i++)
       {
           int c=0,w=0,j=i,k=0;
           while(j)
           {
               if (j%2)
               {
                   c+=s[k].cost;
                   w+=s[k].weight;
                   //cout<<'c'<<c<<' ';
                   if (c>K) {w=0;break;}
               }
               j/=2;
               k++;
           }
           //cout<<'w'<<w<<' ';
           if (w>ans) ans = w;
           //cout<<ans<<' ';
       }
       cout<<ans<<'\n';
   }
   //start = clock() - start;
   //cout<<start/CLOCKS_PER_SEC*1000<<"ms";
   return 0;
}
