pair<int,int> p[12];
pair<ll,ll> p1[1050];

void func(int n){

    int k = pow(2,n);

    for(int i=0;i<k;i++){

        for(int j=0;j<n;j++){

            if(i & (1<<j)) {p1[i].first+=p[j].first; p1[i].second+=p[j].second;}
        }
    }
}
int main(){

    int t; cin>>t;

    while(t--){

        for(int i=0;i<12;i++) {p[i].first=0;p[i].second=0;}

        for(int i=0;i<1050;i++) {p1[i].first=0;p1[i].second=0;}

        int n,k; cin>>n>>k;

        for(int i=0;i<n;i++) cin>>p[i].first>>p[i].second;

        func(n);

        ll size = pow(2,n),big=p1[0].second;

        for(int i=0;i<=size;i++){

            if(p1[i].first<=k){big=max(big,p1[i].second);}
        }

        cout<<big<<"\n";
