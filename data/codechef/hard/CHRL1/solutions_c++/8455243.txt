#include <bits/stdc++.h>
#define filein(t) freopen(t, "r", stdin)
#define fileout(t) freopen(t, "w", stdout)
#define ll long long
#define mp make_pair
#define f first
#define s second
#define N 107
#define BIT(i, n) ( (n >> i) & 1 )
#define ln 1000000009
#define oo 300003

using namespace std;

int n, k, C[N], W[N], T;
ll ans;

int main()
{
   // filein("inp.txt");  fileout("out.txt");
    cin >> T;
    while(T--)
    {
        ans = 0;
        cin >> n >> k;
        for(int i=1; i<=n; i++)
            cin >> C[i] >> W[i];
        for(int i=0; i<(1<<n); i++)
        {
            ll We = 0;
            ll Co = 0;
            for(int j=0; j<n; j++)
                if(BIT(j, i))
                {
                    We += W[j+1];
                    Co += C[j+1];
                }
            if(Co <= k) ans = max(ans, We);
        }
        cout << ans << endl;
    }
}
