#include<bits/stdc++.h>
using namespace std;
main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,i,j;
		long long int k;
		cin >> n >>k;
		int c[n],w[n];
		for(i=0;i<n;i++)
		{
			cin >> c[i] >> w[i];
		}
		int weight=0,cost=0,ans=0;
		for(i=0;i<(1<<n);i++)
		{
			weight=0;
			cost=0;
			for(j=0;j<n;j++)
			{
			if(i&(1<<j))
			{
				weight+=w[j];
				cost+=c[j];
			}
			}
			if(cost<=k)
			{
			ans=max(ans,weight);
			}
		}
		cout << ans <<endl;
	}
}
