#include<stdio.h>
#include<string.h>
long long int t,n,w[12],c[12],k,i,j,x[12],y[12],fw;
void bknapsack(long long int i,long long int cp,long long int cw)
{
	//implementing backtracking algorithm to calculate the maximum profit value
	long long int j;
	if(i==(n+1))
	{
		if(cw>fw)
		fw=cw;
		return;
	}
	if((cp+c[i])<=k)
	{
		bknapsack(i+1,cp+c[i],cw+w[i]);
	}
	bknapsack(i+1,cp,cw);
}
int main()
{
	//freopen("cook.txt","r",stdin);
	scanf("%lld",&t);
	while(t--)
	{
		fw=-1;
		scanf("%lld%lld",&n,&k);
		for(i=1;i<=n;i++)
		scanf("%lld%lld",c+i,w+i);
		bknapsack(1,0,0);
		printf("%lld\n",fw);
	}
	return 0;
}
