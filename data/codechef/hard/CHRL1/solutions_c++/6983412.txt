using namespace std;

typedef long long ll;

int wt[20], cost[20];

int ks(int i, int wt[], int cost[], int n, int curr_wt, int W)
{
	if (i == n) {
		if (curr_wt <= W)
			return 0;
		else 
			return -inf;
	}
	
	int ans1 = cost[i] + ks(i+1, wt, cost, n, curr_wt+wt[i], W);
	int ans2 = ks(i+1, wt, cost, n, curr_wt, W);
	
	return max(ans1, ans2);
}

int main()
{
	int t, n, k;
	si(t);

	while (t--) {
		si(n);
		si(k);

		F(i, 0, n) {
			si(wt[i]);
			si(cost[i]);
		}

		printf("%d\n", ks(0, wt, cost, n, 0, k));
	}

	return 0;
}
