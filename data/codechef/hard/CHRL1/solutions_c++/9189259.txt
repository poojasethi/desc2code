#include <iostream>
using namespace std;
int max(int a,int b)
{
    if(a>b)
    {
        return a;
    }
    else
    {
        return b;
    }
}
int compute(int k,int n,int cost[],int weight[],int l)
    {
        if(n==l || k==0)
        {
            return 0;
        }
     
        else if(cost[n]>k)
        {
            return compute(k,n+1,cost,weight,l);
        }
        else
        {
            return max(compute(k,n+1,cost,weight,l),weight[n]+compute(k-cost[n],n+1,cost,weight,l));    
        }
        
    }
int main()
{
    int t;
     int i;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int cost[10000],weight[10000];
       
        for(i=0;i<n;i++)
        {
            cin>>cost[i]>>weight[i];
        }
        cout<<compute(k,0,cost,weight,n)<<endl;
    }
    return 0;
}
