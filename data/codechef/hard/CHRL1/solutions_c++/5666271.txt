#include <bits/stdc++.h>
using namespace std;

#define sd(x) scanf("%d", &x)
#define N 11

int c[N], w[N];

inline void Solve(){
    int n, i, x, j, cost, ans = 0, weight;
    sd(n); sd(x);
    for(i = 0; i < n; i++){
        sd(c[i]); sd(w[i]);
    }
    for(i = 0; i < (1<<n); i++){
        cost = 0;
        weight = 0;
        for(j = 0; j < n; j++){
            if(((1<<j) & i) == 0) continue;
            cost += c[j];
            weight += w[j];
        }
        if(cost > x) continue;
        if(weight > ans){
            ans = weight;
        }
    }
    printf("%d\n", ans);
}

int main(){
    //freopen("input.txt", "r", stdin);
    int t;
    sd(t);
    while(t--){
        Solve();
    }
    return 0;
}
