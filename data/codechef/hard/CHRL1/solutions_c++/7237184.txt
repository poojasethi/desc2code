#include<bits/stdc++.h>
using namespace std;

const int N=12;

struct Orange {
    int price;
    int weight;
}orange[N];

int main() {
    int test;
    scanf("%d", &test);
    while(test--) {
        int n, k;
        scanf("%d %d", &n, &k);
        for(int i=0;i<n;i++) {
            scanf("%d %d", &orange[i].price, &orange[i].weight);
        }
        int ans=0;
        for(int i=0;i<(1<<n);i++) {
            int total_weight=0;
            int total_price=0;
            for(int j=0;j<n;j++) {
                if(i & (1<<j)) {
                    total_weight+=orange[j].weight;
                    total_price+=orange[j].price;
                }
            }
            if(total_price<=k && total_weight>ans) {
                ans=total_weight;
            }
        }
        printf("%d\n", ans);
    }
}
