#include <iostream>
using namespace std;

int a[10],p[10];
int l,e,y,o,v;
int maximum(int a,int b)
{
    if(a>b)
        return a;
    return b;
}
int ai(int pos,int sum,int val)
{
    if(sum>v)
        return 0;
    if(pos>=e)
        return val;
    return maximum(ai(pos+1,sum,val),ai(pos+1,sum+a[pos],val+p[pos]));
}
int main()
{
    cin >> l;
    while(l--)
    {
        cin >> e >> v;
        for(int i=0;i<e;i++)
            cin >> a[i] >> p[i];
        cout << ai(0,0,0) << endl;
    }
    return 0;
}
