#include<iostream>
using namespace std;
int main()
{
    ios_base::sync_with_stdio(false);
    int t;
    cin>>t;
    while(t--)
    {
        int n,k;
        cin>>n>>k;
        int c[n],w[n];
        for(int i=0;i<n;i++)
            cin>>c[i]>>w[i];
        long long int ans=0,temp1,temp2;
        for(int i=0;i<=(1<<n);i++)
        {
            temp1=k,temp2=0;
            for(int j=0;j<n;j++)
            {
                if(i&1<<j)
                {
                   temp1=temp1-c[j];
                   temp2=temp2+w[j];
                }
            }
            if(temp1>=0)
            {
                ans=max(ans,temp2);
            }
        }
        cout<<ans<<endl;
    }
    return 0;
}
