#include<bits/stdc++.h>
using namespace std;
int main()
{
    int sum1,sum2,m,t,n,k,i,j,p,x,y;
vector< pair<int,int> > a;
cin>>t;
while(t--)
{
    cin>>n>>k;
    for(i=0;i<n;i++)
    {
        cin>>x;
        cin>>y;
        a.push_back(make_pair(x,y));
    }
    x=pow(2,n);
    for(i=1,m=0;i<x;i++)
    {
        for(j=0,sum1=0,sum2=0;j<n;j++)
        {
            if(i&(1<<j))
            {
                sum1+=a[j].first;
                sum2+=a[j].second;
            }
        }
            if(sum1<=k&&sum2>m)
            {
                m=sum2;
            }
        a.clear();
    }
    cout<<m<<"\n";
}
}
