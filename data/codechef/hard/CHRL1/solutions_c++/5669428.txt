#include<iostream>
#include<stdio.h>
using namespace std;
long long int cost[20],weigh[20];
long long int eval(long long int ct[],long long int wt[],long long int rp,long long int size)
{   if(size==0||rp==0)
    {  return 0; }
    else if(ct[size-1]>rp)
    {  return eval(ct,wt,rp,size-1); }
    else
    {  return max((wt[size-1]+eval(ct,wt,rp-ct[size-1],size-1)),eval(ct,wt,rp,size-1)); }
}    
int main()
{    long long int t,i,n,k,ans;
     cin>>t;
     while(t--)
     {  cin>>n>>k;
        for(i=0;i<n;i++)
        {  cin>>cost[i]>>weigh[i];  }
        ans=eval(cost,weigh,k,n);
        cout<<ans<<endl;
     }
     return 0;
}
