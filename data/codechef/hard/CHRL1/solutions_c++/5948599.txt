#include<bits/stdc++.h>
using namespace std;
long long a[2][15];
int main()
{
	int t,n,i,j,p,q;
	long long k,ans,tmp,cost;
	scanf("%d",&t);
	while(t--)
	{
			ans=0;
			scanf("%d%lld",&n,&k);
			for(i=0;i<n;++i)
			{
				scanf("%lld%lld",&a[0][i],&a[1][i]);
			}

			p=(1<<n)-1;
			//cout<<"p"<<p<<"\n";
			while(p)
			{
				q=p;
				tmp=0;i=0;cost=0;
				while(q)
				{
					if(q&1) {tmp+=a[1][i];cost+=a[0][i];}
					q>>=1;
					++i;
				}
				//cout<<tmp<<" "<<cost<<"\n";
				p-=1;
				if(cost<=k&&tmp>ans)ans=tmp;
			}
			printf("%lld\n",ans);

	}
	return 0;
}
