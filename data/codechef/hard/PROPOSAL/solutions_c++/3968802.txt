#include <iostream>
#include <algorithm>

using namespace std;

 int arr[10] = {0};

bool check(int a)
{
	while(a)
	{
		if(arr[a%10] != 1)
			return false;
		a /= 10;
	}
	return true;
}

int main()
{
	int n, t, total = 0;
	cin >> n;
	while(n--)
	{
		cin >> t;
		arr[t] = 1;
	}

	for (int i = 111; i < 999; ++i)
	{
		if(check(i))
		{
			for (int j = 11; j < 99; ++j)
			{
				if(check(j))
				{
					int x = i * (j%10);
					int y = i * (j/10);
					int z = x + 10*y;

					if(x <1000 && y<1000 && z<10000 && check(x) && check(y) && check(z))
						total++;

				}
			}


		}
	}

	cout << total;
	return 0;
}
