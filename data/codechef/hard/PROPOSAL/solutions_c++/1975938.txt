// PROPOSAL
#include <cstdio>
using namespace std;

/* 6 4 2
     8 1
--------
 D B 9
   7 5 3
--------
 E C A 3 */

int main(){
  int n,digits[9],sols=0,carry3,carry5,carry9,carrya,carryb,carryc;
  int i1,i2,i4,i6,i8;
  int d1,d2,d3,d4,d5,d6,d7,d8,d9,da,db,dc,dd,de;
  bool validdigit[10]={false};

#ifndef ONLINE_JUDGE
  freopen("in.txt","r",stdin);
#endif
  scanf("%d",&n);
  for(i1=0;i1<n;i1++){
    scanf("%d",&digits[i1]);
    validdigit[digits[i1]]=true;
  }
  for(i1=0;i1<n;i1++){
    d1=digits[i1];
    for(i2=0;i2<n;i2++){
      d2=digits[i2];
      d3=d1*d2;
      if(d3<10) carry3=0;
      else{
        carry3=d3/10;
        d3-=(carry3*10);
      }
      if(!validdigit[d3]) continue;
      for(i4=0;i4<n;i4++){
        d4=digits[i4];
        d5=d1*d4+carry3;
        if(d5<10) carry5=0;
        else{
          carry5=d5/10;
          d5-=(carry5*10);
        }
        if(!validdigit[d5]) continue;
        for(i6=0;i6<n;i6++){
          d6=digits[i6];
          d7=d1*d6+carry5;
          if(d7>9 || !validdigit[d7]) continue;
          for(i8=0;i8<n;i8++){
            d8=digits[i8];
            d9=d2*d8;
            if(d9<10) carry9=0;
            else{
              carry9=d9/10;
              d9-=(carry9*10);
            }
            if(!validdigit[d9]) continue;
            da=d5+d9;
            if(da<10) carrya=0;
            else{
              carrya=da/10;
              da-=(carrya*10);
            }
            if(!validdigit[da]) continue;
            db=d4*d8+carry9;
            if(db<10) carryb=0;
            else{
              carryb=db/10;
              db-=(carryb*10);
            }
            if(!validdigit[db]) continue;
            dc=d7+db+carrya;
            if(dc<10) carryc=0;
            else{
              carryc=dc/10;
              dc-=(carryc*10);
            }
            if(!validdigit[dc]) continue;
            dd=d6*d8+carryb;
            if(dd>9 || !validdigit[dd]) continue;
            de=dd+carryc;
            if(de<10 && validdigit[de]) sols++;
          }
        }
      }
    }
  }
  printf("%d\n",sols);
  return 0;
}
