bool f[10];
int n;

bool isvalid(int x)
{
	while (x > 0)
	{
		if (!f[x % 10]) return false;
		x /= 10;
	}
	return true;
}

void solve() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		int t;
		scanf("%d", &t);
		f[t] = true;
	}
	int res = 0;
	for (int i = 111; i <= 999; i++) if (isvalid(i))
		for (int j = 11; j <= 99; j++) if (isvalid(j))
		{
			int a = i * (j % 10);
			int b = i * (j / 10);
			int c = i * j;
			if (a < 1000 && b < 1000 && c < 10000 && isvalid(a) && isvalid(b) && isvalid(c))
			{
				res++;
			}
		}
	printf("%d", res);
}

int main() {
	//freopen(infile, "rt", stdin);
	//freopen(outfile, "wt", stdout);
	solve();
	return 0;
