#include<iostream>
using namespace std;
main()
{
    int n,a[10]={0},x[10]={0},cnt=0;
    cin>>n;
    for(int i=0;i<n;i++) { cin>>a[i]; x[a[i]]=1; }
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<n;j++)
        {
            for(int k=0;k<n;k++)
            {
                for(int l=0;l<n;l++)
                {
                    for(int m=0;m<n;m++)
                    {
                        int flag=0,cntdigit=0;
                        int num1=a[i]*100+a[j]*10+a[k];
                        int num2=a[l];
                        int num3=a[m];
                        int res=num1*num2;
                        while(res>0)
                        {
                            int val=res%10;
                            cntdigit++;
                            res/=10;
                            if(x[val]==0) { flag=1; break; }
                        }
                        if(cntdigit>3) flag=1;
                        cntdigit=0;
                        res=num1*num3;
                        while(res>0)
                        {
                            int val=res%10;
                            res/=10;
                            cntdigit++;
                            if(x[val]==0) { flag=1; break; }
                        }
                        res=num1*(num2*10+num3);
                        if(cntdigit>3) flag=1;
                        cntdigit=0;
                        while(res>0)
                        {
                            int val=res%10;
                            res/=10;
                            cntdigit++;
                            if(x[val]==0) { flag=1; break; }
                        }
                        if(cntdigit>4) flag=1;
                        if(flag==0) {  cnt++; }
                    }
                }
            }
        }
    }
    cout<<cnt<<endl;
}
