#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int arr[209][209],a[200005];
int f[200005][205];
int g[200005][205];
int main()
{
    int n,m,k,t,mini,x,y;
    scanf("%d",&t);
    while(t--)
    {
    scanf("%d %d %d",&m,&k,&n);
    memset(arr,0,sizeof(arr));
    for(int i=1;i<=k;i++)
    {
        scanf("%d %d",&x,&y);
        arr[x][y]=1;
        arr[y][x]=1;
    }
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    for(int l=1;l<=m;l++)
    {
        for(int i=1;i<=m;i++)
        {
            for(int j=1;j<=m;j++)
            {
                if(arr[i][l] && arr[l][j])
                    arr[i][j]=1;
            }
        }
    }
    for(int j=1;j<=m;j++)
        g[0][j]=0;
    for(int i=1;i<=n;i++)
    {
        mini=999999;
        for(int j=1;j<=m;j++)
        {
            f[i][j]=999999;
            if(j==a[i] || arr[a[i]][j])
            {
                if(j==a[i])
                    f[i][j]=g[i-1][j];
                else
                    f[i][j]=g[i-1][j]+1;
            }
            if(f[i][j]<mini)
                mini=f[i][j];
            g[i][j]=mini;
        }
    }
    mini=999999;
    for(int i=1;i<=m;i++)
    {
        if(f[n][i]<mini)
            mini=f[n][i];
    }
    if(mini==999999)
        printf("-1\n");
    else
        printf("%d\n",mini);
    }
    return 0;
}
