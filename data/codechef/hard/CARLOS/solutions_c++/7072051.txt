#include <bits/stdc++.h>
using namespace std;
#define inf 10000000
int dp[200001][201];
bool path[201][201];
int arr[200001];

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int m,k,n,a,b;
        scanf("%d%d%d",&m,&k,&n);
       /* int arr[n+1],dp[n+1][m+1];
        bool path[m+1][m+1];*/
        for(int i=1;i<=m;i++){
            for(int j=1;j<=m;j++)
                path[i][j]=false;
        }
        for(int i=1;i<=k;i++){
            scanf("%d%d",&a,&b);
            path[a][b]=path[b][a]=true;
        }
        for(int k=1;k<=m;k++){
            for(int i=1;i<=m;i++){
                for(int j=1;j<=m;j++)
                    path[i][j]=path[i][j]||(path[i][k]&&path[k][j]);
            }
        }
        for(int i=1;i<=n;i++)
            scanf("%d",&arr[i]);
        for(int i=1;i<=m;i++){
            if(arr[1]==i)
                dp[1][i]=0;
            else if(path[arr[1]][i])
                dp[1][i]=1;
            else
                dp[1][i]=inf;
        }
        for(int j=2;j<=n;j++){
            int cost=dp[j-1][1];
            for(int i=1;i<=m;i++){
                cost=min(cost,dp[j-1][i]);
                if(arr[j]==i)
                    dp[j][i]=cost;
                else if(path[arr[j]][i])
                    dp[j][i]=cost+1;
                else
                    dp[j][i]=inf;
            }
        }
        int ans=dp[n][1];
        for(int i=2;i<=m;i++)
            ans=min(ans,dp[n][i]);
        if(ans>=inf)
            printf("-1\n");
        else
            printf("%d\n",ans);
    }
}
