#include<iostream>
#include<vector>
#include<string>
#include<stdio.h>
#include<stdlib.h>
#include<cstring>
using namespace std;

int vis[17000];

void dfs(int u,int n)
{
    vis[u]=1;
    for(int i=0;i<n;i++)
    {
        if(u&(1<<i))
            cout<<"1 ";
        else cout<<"-1 ";

    }
    cout<<endl;
    for(int i=0;i<n;i++)
    {
        int nv=(u^(1<<i));
        if(vis[nv]==0)
        {
            dfs(nv,n);
        }
    }
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
      int n,d,v;
      cin>>n>>d;
      int sb=0;
    memset(vis,0,sizeof(vis));
      for(int i=0;i<n;i++)
      {
          cin>>v;
          if(v==1)(sb=sb|(1<<i));
      }
      if(d>=4)
        dfs(sb,n);
    else cout<<"-2\n";
    }
    return 0;

}
