#include<bits/stdc++.h>
using namespace std;
int ans[1<<16];
void make()
{
	ans[0]=0;
	ans[1]=1;
	ans[2]=3;
	ans[3]=2;
	for(int i=3;i<16;i++)
	{
		for(int j=0;j<(1<<i-1);j++)
		ans[j+(1<<i-1)]=((1<<i-1)+ans[(1<<i-1)-(1+j)]);
	}
}
int main()
{
	make();

	int t;
	cin>>t;
	while(t--)
	{
		int n,d;
		cin>>n>>d;
		int idx=0;
		for(int i=0;i<n;i++)
		{
			int x;
			cin>>x;
			if(x==1)  idx+=(1<<i);
		}
		if(d<4)
		{
			cout<<"-2"<<endl;
			continue;
		}
		int j=0;
		for(int i=0;i<=(1<<15);i++)
		{
			if(idx==ans[i])
			{
				j=i;
				break;
			}
		}
		for(int i=j;i<(1<<n);i++)
		{
			for(int k=0;k<n;k++)
			{
				if(ans[i]&(1<<k)) printf("1 ");
				else printf("-1 ");
			}
			printf("\n");
		}
		for(int i=0;i<j;i++)
		{
			for(int k=0;k<n;k++)
			{
				if(ans[i]&(1<<k)) printf("1 ");
				else printf("-1 ");
			}
			printf("\n");
		}
	}
	return 0;
}
