#include <bits/stdc++.h>
using namespace std;
 
int n, D, a;
int bit;
 
void print() {
	if( (1 & bit)  == 0)  {
		putchar('-');
	}
	
	putchar('1');
	for(int i=1 ;i<n;i++) {
		putchar(' ');
		if( (bit &(1<<i) )  == 0) {
			putchar('-');
		}
		
		putchar('1');
	}
	putchar('\n');
}
 
void fun(int pos) {
	if(pos == n) {
		return ;
	}
	
	fun(pos+1);
	bit ^= (1 << pos);
	print();
	fun(pos+1);
}
int main() {
	// your code goes here
	int t;
	scanf("%d",&t);
	while(t--) {
		bit = 0;
		scanf("%d %d",&n, &D);
		for(int i=0; i<n; i++) {
			scanf("%d",&a);
			if(a > 0 ) {
				bit|= (1<<i);
			}
		}
		
		if(D < 4) {
			printf("-2\n");
			continue;
		} 
		print();
		fun(0);
	}
	return 0;
}
