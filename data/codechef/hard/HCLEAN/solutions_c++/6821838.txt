#include<iostream>
using namespace std;
 
void func(int arr[20],int k)
{
     if(arr[k]==1)
     arr[k]=-1;
     else
     arr[k]=1;
}
void out(int arr[20],int n)
{
     
         for(int j=0;j<n;j++)
         {
                 cout<<arr[j]<<" ";
         }
         cout<<endl;
 }
void out1(int arr[20],int i, int n)
{
     if(i==2)
     {
         out(arr,n);
         for(int j=0;j<3;j++)
         {
                 func(arr,j%2);
                 out(arr,n);
         }
     
     
     }
     else
     {
         out1(arr,i-1,n);
         func(arr,i-1);
         out1(arr,i-1,n);
     }
     
}
 
 
int main()
