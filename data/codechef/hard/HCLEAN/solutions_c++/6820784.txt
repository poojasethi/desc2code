#include <bits/stdc++.h>

using namespace std;

int A[15];
vector <int> v;
int vis[1<<15];
int n;

void f(int mask)
{
    v.push_back(mask);
    if ( vis[mask] == (1<<n) ) return;
    for ( int i = 0; i < n; i++ ) {
        int mask1 = mask^(1<<i);
        if ( vis[mask1] == 0 ) {
            vis[mask1] = vis[mask] + 1;
            f(mask1);
            break;
        }
    }
}
int main()
{
    int t,d;
    cin >> t;
    while ( t-- ) {
        cin >> n >> d;
        for ( int i = 0; i < n; i++ ) cin >> A[i], assert(A[i] == -1 || A[i] == 1);
        if ( d < 4 ) {
            puts("-2");
            continue;
        }
        v.clear();
        memset(vis, false, sizeof(vis));
        vis[0] = 1;
        f(0);
        for ( int i = 0; i < (1<<n); i++ ) {
            for ( int j = 0; j < n; j++ ) {
                if ( v[i]&(1<<j) ) cout << -A[j];
                else cout << A[j];
                if ( j == n-1 ) cout << endl;
                else cout << " ";
            }
        }
    }
    return 0;
}
