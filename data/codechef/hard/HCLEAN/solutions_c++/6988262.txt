
#include <iostream>

using namespace std;

int recur(int a[],int n,int c)
{
    if(c<n-1)
    {
        recur(a,n,c+1);
        if(a[c]==1)
            a[c]=-1;
        else
            a[c]=1;
        recur(a,n,c+1);
    }
    else
    {
        for(int i=0;i<n;i++)
            cout << a[i] << " ";
        cout << "\n";
        if(a[c]==1)
            a[c]=-1;
        else
            a[c]=1;
        for(int i=0;i<n;i++)
            cout << a[i] << " ";
        cout << "\n";
    }
}

int main()
{
	int test;
	cin >> test;
	for(int q=1;q<=test;q++)
	{
        int n,d;
        cin >> n;
        cin >> d;
        int a[n];
        for(int i=0;i<n;i++)
            cin >> a[i];
        if(d<4)
            cout << -2 << "\n";
        else
            recur(a,n,0);
	}
}
