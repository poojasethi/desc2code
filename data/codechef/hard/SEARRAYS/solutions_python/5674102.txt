test= (int)(raw_input())

mod=1000000007

while(test):
    str=raw_input()
    str=str.split()
    n=int(str[0])
    k=int(str[1])
   
    dp=list()
    dp.append(1)
    
    if(k==1):
      dp.append(2)
    else:
      dp.append(1)
      
    for i in range(2,n+2):
       if(i-k)<0:
         dp.append(dp[i-1])
       else:
         dp.append(dp[i-1]+dp[i-k])
       dp[i]%=mod
       
    print int(dp[n])
    test-=1
