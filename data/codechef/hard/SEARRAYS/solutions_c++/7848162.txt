ll expo(int x,int y)
{
  ll tmp=x,ans=1ll;
  for(;y;y>>=1){
    if(y&1){
      ans=(ans*tmp)%mod;
    }
    tmp=(tmp*tmp)%mod;
  }
  return ans;
}
           
ll dp[100010];

ll solve(int n,int k)
{
   if(dp[n]!=-1) return dp[n];
   if(n<k)
   {
     dp[n]=1;
     return dp[n];
   }
   if(n==k)
   {
     dp[n]=2;
     return dp[n];
   }
   dp[n]=solve(n-1,k);
   dp[n]+=solve(n-k,k);
   dp[n]%= mod;
   return dp[n] ;
 }  

int main()
{
  int t;
  cin>>t;
  while(t--)
  {
     memset(dp,-1,sizeof(dp));
     int n,k;
     scanf("%d%d",&n,&k);
     if(k==1)
     {
       cout<<expo(2,n)<<"\n";
       continue;
     }  
     cout<<solve(n,k)<<endl;
  }
}
