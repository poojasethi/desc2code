#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int n,k;

int main()
{
  long int dp[100005];
  int t;
  cin>>t;
  int n,k;
  while(t--)
  {
    cin>>n>>k;
    dp[1] = 1;
    if(k==1)
    dp[1]++;
    for(int i=2;i<=n;i++)
    {
      if(i<k)
      {
         dp[i] = dp[i-1];
         continue;
      }
      if(i>k)
      dp[i] = (dp[i-1] + dp[i-k])%mod;
      else
      dp[i] = 2;
    }
    cout<<dp[n]<<endl;
  }
  return 0;
}
