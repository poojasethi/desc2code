#include <iostream>
#include <string.h>
#include <cstdio>
using namespace std;
long long int t,n,k,dp[100010],mod=1000000007;
long long int DFS(int pos)
{
    if(pos>n)
        return 0;
    if(pos==n)
        return 1;
    if(dp[pos]!=-1)
        return dp[pos];
    long long int ret=DFS(pos+1);///don`t start from here
    ret+=DFS(pos+k);/// start from here
    return (dp[pos]=ret%mod ) ;
}
int main()
{
    //freopen("r.txt","r",stdin);
    scanf("%lld",&t);
    for(int u=0;u<t;u++)
    {
        memset(dp,-1,sizeof dp);
        scanf("%lld %lld",&n,&k);
        printf("%lld\n",DFS(0)%mod);
    }

    return 0;
}
