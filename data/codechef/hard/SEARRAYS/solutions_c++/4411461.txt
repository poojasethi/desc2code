#include<iostream>
using namespace std;
#define M 1000000007
#define lld signed long long int
lld dp[100000+5];
lld psum[100000+5];
int main()
{
	int t,n,k;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		for(int i=0;i<=n;++i)dp[i]=psum[i]=0;
		dp[0]=1;
		psum[0]=1;
		for(int i=1;i<=k;++i)
		{
			dp[i]=dp[i-1];
			psum[i]=dp[i];
		}
		for(int i=k;i<=n;++i)
		{
			dp[i]=dp[i-1]+psum[i-1-k];
			if(i%k==0)dp[i]++;
			psum[i]=dp[i]+psum[i-k];
			dp[i]%=M;
			psum[i]%=M;
		}
		cout<<dp[n]<<endl;
	}
}
