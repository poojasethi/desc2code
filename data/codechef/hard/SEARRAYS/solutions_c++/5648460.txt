#include<iostream>
#include<fstream>
#include<string.h>
#include<stdio.h>
#include<math.h>
#include<algorithm>
#include<string>
#include<vector>

using namespace std;

#define MOD 1000000007

int max(int a,int b)
{
	return (a>b)?a:b;
}

int min(int a,int b)
{
	return (a<b)?a:b;
}

int mod(int x)
{
	return (x<0)?(-x):x;
}

int DP[100005];
int process(int n,int k)
{
	DP[0]=1;	
	
	for(int i=1;i<=n;i++)
	{
		if(i>k)
			DP[i]=(DP[i-k]+DP[i-1])%MOD;
		else if(i==k)
			DP[i]=2;
		else
			DP[i]=1;
	}
	
	return DP[n];
}

int main()
{
	ios_base::sync_with_stdio(false);
	int t,k,n;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		cout<<process(n,k)<<endl;
	}
	
	return 0;
}
