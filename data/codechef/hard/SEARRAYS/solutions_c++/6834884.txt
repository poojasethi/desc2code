const int MOD = 1000000007LL;
const int sz = 100002;
 
int main(){
    int test; si1(test);
    while(test--){
        int n, k; si2(n,k);
        int F[sz];
        FOR(i,0,n+1){
            if(i<k) F[i] = 1;
            else if(i==k) F[i]=2;
            else {
                F[i] = F[i-k] + F[i-1];
                F[i]%=MOD;
            }
        }
        cout<<F[n]%MOD<<endl;
    }
    return 0;
}
