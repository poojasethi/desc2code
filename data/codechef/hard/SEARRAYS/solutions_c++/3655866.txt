#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define N 100100
#define P 1000000007

int n, k, c[N];
int main()
{
  int t;
  scanf("%d", &t);
  while(t--) {
    scanf("%d%d", &n, &k);
    for(int i = 0; i < k; i++)
      c[i] = 1;
    for(int i = k; i <= n; i++) {
      c[i] = c[i-1] + c[i-k];
      if(c[i] >= P) c[i] -= P;
    }
    printf("%d\n", c[n]);
  }
  return 0;
}
