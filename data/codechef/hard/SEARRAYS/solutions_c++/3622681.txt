#include <bits/stdc++.h>
using namespace std;

const int N = 100010;
const long long mod = 1000000007;

long long pow(long long a,long long b)
{
    long long ans = 1;
    while(b) {
        if(b & 1) ans = ans * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return ans;
}

long long fac[N],rev[N];

void init()
{
    fac[0] = rev[0] = 1;
    for(long long i = 1; i < N; i ++) {
        fac[i] = fac[i - 1] * i % mod;
        rev[i] = rev[i - 1] * pow(i,mod - 2) % mod;
    }
}

long long com(long long n,long long m)
{
    if(n < m) return 0;
    return fac[n] * rev[m] % mod * rev[n - m] % mod;
}

void solve()
{
    int n,m;
    cin >> n >> m;
    long long ans = 0;
    for(int i = 0; i * m <= n; i ++)
        ans += com(i + n - i * m,i);
    cout << ans % mod << endl;
}

int main()
{
    int t;
    scanf("%d",&t);
    init();
    while(t --) solve();
    return 0;
}
