#include<iostream>
using namespace std;

typedef long long int ll;

#define mod 1000000007

int main()
{
	ll n,k,i;
	ll test;
	cin>>test;
	while(test--)
	{
	cin>>n>>k;
	ll dp[n+2];
	dp[0]=1;
	
	if(n<k)
	cout<<"1"<<endl;
	else if(n==k)
	cout<<"2"<<endl;
	else
	{
		if(k==1)
		 dp[1]=2;
		 else dp[1]=1;
		 for(i=2;i<=n;i++)
		 {
		 	dp[i]=0;
		 	if(i-k>=0)
		 	   dp[i]= dp[i-1]+dp[i-k];
		 	else
		 	   dp[i]=dp[i-1];
		 	dp[i]%=mod;
		 }
		 cout<<dp[n]%mod<<endl;
	}
	}
	return 0;
}
