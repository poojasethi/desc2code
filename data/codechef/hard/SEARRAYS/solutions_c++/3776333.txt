#include <iostream>
#include <cstdio>
#include <algorithm>
#include <memory.h>
#define ll long long
using namespace std;

ll dp[100005];

int main()
{
    int t;
    cin >> t;
    while(t--)
    {
        memset(dp,0,sizeof(dp));
        dp[0] = 1;
        int n, k;
        scanf("%d%d",&n,&k);
        for(int i = 1; i <= n; i++)
        {
            dp[i] = dp[i-1];
            if(i-k >= 0) dp[i] += dp[i-k];
            dp[i] %= 1000000007;
        }
        printf("%lld\n",dp[n]);

    }
    return 0;
}
