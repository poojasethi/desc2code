#include <iostream>

using namespace std;

int main()
{
    int t,n,k,mod=1000000007,dp[100001],i,j;
    dp[0]=1;
    cin>>t;

    for(j=1;j<=t;j++)
    {
        cin>>n>>k;
        for(i=1;i<k;i++)
            dp[i]=1;

        for(i=k;i<=n;i++)
            dp[i]=(dp[i-1]+dp[i-k])%mod;

        cout<<dp[n]<<endl;
    }
}
