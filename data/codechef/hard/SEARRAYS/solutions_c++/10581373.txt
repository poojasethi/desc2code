#include <stdio.h>
#include <math.h>
#define MOTA 1000000007
int main(){
	int t;scanf("%d",&t);
	while(t--){
		int k,n;
		scanf("%d",&n);
		scanf("%d",&k);
		long f[n+1];
		f[0] = 1;
		if(k==1) f[1]=2;
		else f[1]=1;
		for(int i=2;i<=n;i++){
			f[i] = f[i-1];
			if(i-k>=0) f[i] = (f[i] + f[i-k]) % MOTA;	
		}
		printf("%ld\n",f[n]);
	}

}
