#include <iostream>
#define MOD 1000000007
#define lli long long int

using namespace std;
lli min(lli a, lli b)
{
    return (a<b)?a:b;
}
int main()
{
    lli i,t,n,k,a,ans,f[100001]={0};
    cin >> t;
    while(t--)
    {
        cin >> n >> k;
        a = 0;
        ans = 0;
        if(n<2*k)
        {
            cout << n-k+2 << "\n";
            continue;
        }
        for(i=0; i<k; i++)
        {
            f[i] = 0;
            ans += f[i];
            ans %= MOD;
        }
        for(i=k; i<2*k; i++)
        {
            f[i] = 1;
            ans += f[i];
            ans %= MOD;
        }
        a = 0;
        for(i=2*k; i<=n; i++)
        {
            a = a+f[i-k];
            a %= MOD;
            f[i] = a+1;
            ans += f[i];
            ans %= MOD;
        }
        cout << (1+ans)%MOD << "\n";
    }
    return 0;
}
