#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ull unsigned long long
#define pb push_back
#define ft first
#define se second
#define mp make_pair
int main(int argc, char const *argv[]){
    int t;
    scanf("%d", &t);
    while(t--){
    	int n, k;
    	scanf("%d%d", &n, &k);
    	int a[n+1];
    	a[0]=1;
    	int mod=1000000007;
    	for(int z = 1; z <= n; a[z]=0, ++z);
    	for(int i = 1; i <= n; ++i){
    		a[i]+=a[i-1];
    		if(i-k>=0) a[i]+=a[i-k];
    		a[i]%=mod;
    	}
    	printf("%d\n", a[n]);
    }
	return 0;
}
