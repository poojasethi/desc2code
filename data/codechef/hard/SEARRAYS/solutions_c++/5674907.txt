#include <bits/stdc++.h>
using namespace std;

#define LL long long
#define MOD 1000000007
#define N 112345

LL dp[N];

inline void Solve(){
    LL n, k, i;
    cin>>n>>k;
    for(i = 1; i <= k - 1; i++){
        dp[i] = 1;
    }
    dp[k] = 2;
    for(i = k + 1; i <= n; i++){
        dp[i] = dp[i - 1] + dp[i - k];
        dp[i] %= MOD;
    }
    cout<<dp[n]<<endl;
    return;
}

int main(){
    //freopen("input.txt", "r", stdin);
    LL t;
    cin>>t;
    while(t--){
        Solve();
    }
    return 0;
}
