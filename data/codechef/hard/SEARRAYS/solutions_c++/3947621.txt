#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
int a[100001];
int b[100001];
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,k;
		scanf("%d%d",&n,&k);
		memset(a,0,sizeof(a));
		memset(b,0,sizeof(b));
		a[0]=1;
		b[0]=1;
		for(int i=1;i<=n;i++)
		{
			if(i%k==0)
			 a[i] =  1;
			// int p=i-1;
			//while(p>=0)
			{
				a[i] += b[(i-1)%k]%mod;
				a[i] %= mod;
				b[i%k] = (b[i%k] + a[i])%mod; 
				//p = p-k;
			}
		}
		cout<<a[n]%mod<<endl;
	}
}
