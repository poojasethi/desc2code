using namespace std;

const int MOD = 1000000007;

int T, n, k, dp[100005];
int main(){
    scanf("%d", &T);
    while(T--){
        scanf("%d %d", &n, &k);
        
        dp[0] = 1;
        for(int i = 1; i <= n; i++)
            dp[i] = (dp[i - 1] + (i - k >= 0 ? dp[i - k] : 0)) % MOD;
        printf("%d\n", dp[n]);
    }
    return 0;
}
