const long long base = 1000000000 + 7;

using namespace std;


void solve(){
	int n, k;
	cin >> n >> k;
	int *f = new int[n + 1];
	f[0] = 1;
	for (int i = 1; i <= n; i++)
	{
		f[i] = f[i - 1];
		if (i - k >= 0)
		{
			f[i] += f[i - k];
		}
		if (f[i] >= base)
		{
			f[i] -= base;
		}
	}
	cout << f[n] << endl;
	delete[] f;
}

int main(){
	ios_base :: sync_with_stdio(false);
	int test = 1;
	cin >> test;
	while (test--){
		solve();
	}
	return 0;
}
