#include <stdio.h>

#define mod 1000000007

using namespace std;

long long dp0[100001],dp1[100001];

int main(int argc, char **argv)
{
	int T,i,N,K;
	
	for(scanf("%d",&T);T>0;T--)
	{
		scanf("%d%d",&N,&K);
		for(i=0;i<=K;i++)
		{
			dp0[i]=1;
			dp1[i]=0;
		}
		dp1[K]=1;
		for(i=K+1;i<=N;i++)
		{
			dp0[i]=(dp0[i-1]+dp1[i-1])%mod;
			dp1[i]=(dp0[i-K]+dp1[i-K])%mod;
		}
		printf("%lld\n",(dp0[N]+dp1[N])%mod);
	}
	
	return 0;
}
