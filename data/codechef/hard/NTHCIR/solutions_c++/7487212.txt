#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,p,m,b;
	double r[4];
	long t;
	scanf("%ld",&t);
	scanf("%lld%lld%lld%lld",&n,&p,&m,&b);
	scanf("%lf%lf%lf%lf",&r[0],&r[1],&r[2],&r[3]);
    double a1=-1/r[0];
	double a2=1/r[1];
	double a3=1/r[2];
   //cout<<(a3*(a1+a2))+a1*a2<<endl;
	double cons;
	double cons1=sqrt((a3*(a1+a2))+a1*a2)-3*(a1+a2);
	double cons2=-sqrt((a3*(a1+a2))+a1*a2)-3*(a1+a2);
	double test1=(((4*(a1+a2))+cons1)*((4*(a1+a2))+cons1))-(a1*a2);
	test1=(a1+a2)/test1;
	double test2=(((4*(a1+a2))+cons2)*((4*(a1+a2))+cons2))-(a1*a2);
	test2=(a1+a2)/test2;
	//cout<<test1<<endl;
	if(test1==r[3])
	cons=cons1;
	else
	cons=cons2;
	double ans=0;
	while(t--)
	{
	long long find=(p*n)%m+b;
	double temp;
	if(find<=4)
	temp=r[find-1];
	else
	{
		temp=(((find*(a1+a2))+cons)*((find*(a1+a2))+cons))-(a1*a2);
		temp=temp/(a1+a2);
	}	
	ans+=1/temp;
	n=find;
    }
	printf("%0.6lf\n",ans);
}
