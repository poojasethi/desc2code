#include <cstdio>
int main()
{
	int t,n,p,m,b;
	double r1,r2,r3,r4,ans=0.0,cans,a1,a2,a3,a4,a5;
	scanf("%d %d %d %d %d",&t,&n,&p,&m,&b);
	scanf("%lf %lf %lf %lf", &r1,&r2,&r3,&r4);
	for (;t--;)
	{
		n=(((long long)n*(long long)p)%m)+b;
		if (n==1) 
		cans=r1;
		else if(n==2)
		cans=r2;
		else if(n==3)cans=r3;
		else if (n == 4)
		cans=r4;
		else
		{
			a1=-1.0/r1;
			a2=1.0/r2;
			a3=1.0/r3;
			a4=1.0/r4;
			double c1=a4-a3-7*(a1+a2);
			double c2=a3-9*(a1+a2)-3*c1;
			cans=1.0/((a1+a2)*(long long)n*(long long)n+c1*n+c2);
		}
		ans+=cans;
	}
	printf("%.6f\n",ans);
	return 0;
}
