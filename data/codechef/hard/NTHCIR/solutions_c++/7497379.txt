LL t, n, m, p, b, nt;
LD a1, a2, a3, a4, r1, r2, r3, r4, ans=0;
int main() {
	cin >> t >> n >> p >> m >> b;
	cin >> r1 >> r2 >> r3 >> r4;
	a1 = -(1 / r1);
	a2 = 1 / r2;
	a3 = 1 / r3;
	a4 = 1 / r4;
	while(t--) {
		n = (p * n) % m + b;
		if(n > 4) ans += 1.0 / (a4 + (n-4) * ((a4 - a3) + (n-3) * (a1 + a2)));
		else {
			switch (n) {
				case 1: ans += r1;break;
				case 2: ans += r2;break;
				case 3: ans += r3;break;
				case 4: ans += r4;break;
			}
		}
	}	
	printf("%0.6Lf\n", ans);
}
