#include <bits/stdc++.h>
using namespace std;
 
#define ll 			long long int
#define ld 			long double
 
int main()
{
    int t,p,m,b;
    ll n;
    ld r1,r2,r3,r4;
    cin>>t>>n>>p>>m>>b>>r1>>r2>>r3>>r4;
    ld ans=0;
    ld k,c;
    k=(1/r4-1/r3)-7*(1/r2-1/r1);
    c=k/2;
    ld k1=(1/r2-1/r1);
    ld k2=1/(r1*r2);
    for(int i=0;i<t;i++)
    {
        n=(p*n)%m+b;
        if(n==1) ans+=r1;
        else if(n==2) ans+=r2;
        else if(n==3) ans+=r3;
        else if(n==4) ans+=r4;
        else
        {
            ans+=1/((n*n*k1)+(2*c*n)+(c*c/k1)+(k2/k1));       
        }
    }
    cout.precision(6);
    cout<<fixed<<ans;    
    return 0;
}
