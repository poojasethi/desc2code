#include<stdio.h>
int main()
{
    long long int t,n,p,m,b,i;
    double r1,r2,r3,r4,sum=0;

    scanf("%lld",&t);
    scanf("%lld %lld %lld %lld",&n,&p,&m,&b);
    scanf("%lf %lf %lf %lf",&r1,&r2,&r3,&r4);

    while(t--)
    {
        n=(p*n)%m+b;

        sum=sum+1/(((n-3)/r4)-((n-4)/r3)+((n-4)*(n-3)*((1/r2)-(1/r1))));
    }

    printf("%lf\n",sum);

    return 0;
}
