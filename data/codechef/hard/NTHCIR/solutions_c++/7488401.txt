inline double err(double a, double b){
    return abs(a - b);
}

int main(){
    //freopen("0.txt", "r", stdin);

    int t;
    double dn, r1, r2, r3, r4, pr3, qr3, rr3;
    long long  P, M, B, N;
    double ans = 0;
    cin>>t;
    cin>>N>>P>>M>>B;
    cin>>r1>>r2>>r3>>r4;
    double x3 = (1 / r4 + 1 / r1 - 1 / r3 - 1 / r2) / 4;
    double y3 = (1 / r2 + 1 / r1) / 4;
    double r = (1 / r2 - 1 / r1) / 4;
    double xn, yn;
    bool flag = false;
    if(x3 < 0){
        flag = true;
        x3 = -x3;
    }
    //cout<<x3<<" "<<y3<<endl;
    while(t--){
        N = (N * P) % M + B;
        dn = N - 3;
        xn = x3 + dn * (flag == true ? -r : r) * 2;
        yn = y3;
        ans += 1 / (xn * xn / r + yn * yn / r - r);
        //cout<<t<<" "<<ans<<" "<<dn<<endl;
        //system("pause");
        //printf("%lld %0.6f\n", N, x);
    }
    printf("%0.6f\n", ans);
    return 0;
}
