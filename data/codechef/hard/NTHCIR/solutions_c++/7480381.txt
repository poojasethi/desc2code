typedef long long ll;
typedef pair<int,int> PII;
typedef vector<int> VI;
typedef vector< PII > VPII;
typedef vector< pair< int , PII > > VPIII;

int main(int argc, char const *argv[])
{
	int t;
	ll n,p,m,b;
	double a1,a2,a3,a4,an,sum=0,r1,r2,r3,r4,rn,c1,c2;// an is the curvature of nth circle, rn is the radius of nth circle 
	cin>>t;
	scanf("%lld %lld %lld %lld",&n,&p,&m,&b);
	scanf("%lf %lf %lf %lf",&r1,&r2,&r3,&r4);
	a1=-1/r1;
	a2=1/r2;
	a3=1/r3;a4=1/r4;
	while(t--)
	{
		n=(p*n)%m+b;
		if(n==1)
			rn=r1;
		else if(n==2)
			rn=r2;
		else if(n==3)
			rn=r3;
		else if(n==4)
			rn=r4;
		else //n>4
		{
			c1=a4-a3-7*(a1+a2);
			c2=4*a3-3*a4+12*(a1+a2);
			an=((a1+a2)*n*n) + (n*c1) + c2;//curvature
			rn=1/an; //nth radius 
		}
		sum=sum+rn;
	}
	printf("%0.6lf\n",sum);
	return 0;
}
