#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
string x;
bool check()
{
for(int i=0;i<(x.size()+1)/2;i++)
if(x[i]!=x[x.size()-i-1])
return false;
return true;
}

long long int func(string &s)
{
long long int ans=0;
for(int i=1;i<=s.size();i++)
for(int p=2;(i*p)<=s.size();p++)
{
x.clear();
x.push_back(s[i-1]);
for(long long int j=i*p;j<=s.size();j=j*p)
{
x.push_back(s[j-1]);
if(check())
ans++;
}
}
return ans+s.size();
}

int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
string s;
cin>>s;
cout<<func(s)<<endl;
}
return 0;
}
