#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll t,n,x,ans;
string str,temp;

void rwwww()
{
    for (ll i = 0; i < (ll)temp.size(); ++i)
        if(temp[i]!=temp[temp.size()-i-1])
            return;
    ans++;
}

int main()
{
    cin>>t;
    while(t--)
    {
        cin>>str;
        n=str.size();
        ans=n;

        for (ll i = 1; i <= n; ++i)
        {
            for (ll j = 2; i*j <= n ; j++)
            {
                temp="";
                temp+=str[i-1];
                x=i*j;

                while(x<=n)
                {
                    temp+=str[x-1];
                    x*=j;
                    rwwww();
                }
            }
        }
        cout<<ans<<endl;
    }        
    return 0;
}
