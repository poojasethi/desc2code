#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    string s;
    ll t,l,p,i,st,en,ans;
    cin>>t;
    while(t--)
    {
        cin>>s;
        s='$'+s;        //increasing length by 1
        ll ans=0;
        l=s.length();
        //check for odd length palindrome
        for(p=2;p<=l;p++)
        {
            for(i=1;i*p<=l;i++)
            {
                if(i%p==0)
                {
                    st=i/p;
                    en=i*p;
                    while(s[st]==s[en])             //we start with palindrome of length 3 and move towards increasing length in both directions
                    {
                        ans++;
                        if(st%p==0)
                        {
                            st/=p;
                        }
                        else
                        {
                            break;
                        }
                        en*=p;
                        if(en>l)
                        {
                            break;
                        }
                    }
                }
            }
        }
        //check for even length palindrome
        for(p=2;p<=l;p++)
        {
            for(i=1;i*p<=l;i++)
            {
                st=i;
                en=i*p;
                while(s[st]==s[en])         //we start with palindrome of length 2 and move towards increasing length in both directions
                {
                    ans++;
                    if(st%p==0)
                    {
                        st/=p;
                    }
                    else
                    {
                        break;
                    }
                    en*=p;
                    if(en>l)
                    {
                        break;
                    }
                }
            }
        }
        cout<<ans+l-1<<"\n";
    }
    return 0;
}
