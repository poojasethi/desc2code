#include <bits/stdc++.h>

using namespace std;

const int MAXL = 500000;

string s;

void solve() {
    cin>>s;
    long long ans = s.length();
    int l = s.length();
    assert(l <= MAXL);
    s = "#" + s;

    for (long long p = 2; p <= l; p++) {
        //1
        for (long long i = 1; i * p <= l; i++) {
            if (i % p == 0) {
                long long st = i / p;
                long long fin = i * p;
                while (s[st] == s[fin]) {
                    ans++;
                    if (st % p == 0) {
                        st /= p;
                    }
                    else {
                        break;
                    }
                    fin *= p;
                    if (fin > l ) {
                        break;
                    }
                }
            }
        }
        //2
        for (long long i = 1; i * p <= l; i++) {
            long long st = i;
            long long fin = i * p;
            while (s[st] == s[fin]) {
                ans++;
                if (st % p == 0) {
                    st /= p;
                }
                else {
                    break;
                }
                fin *= p;
                if (fin > l ) {
                    break;
                }
            }
        }
    }
    cout<<ans<<"\n";
}

int main() {
    int test;
    cin>>test;
    assert(test <= 10);
    while (test--) {
        solve();
    }
    return 0;
}
