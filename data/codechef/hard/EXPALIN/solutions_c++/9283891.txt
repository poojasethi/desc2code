#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		char a[1000500],b[1000500];
		long long int i,j,sum=0,si,d,k,l,e,flag=0;
		cin>>(a+1);
		si=strlen(a+1);
		for(i=2;i<=si;i++)
		{
			for(j=1;j*i<=si;j++)
			{
				b[0]=a[j];
				d=j;
				d*=i;
				k=1;
				while(d<=si)
				{
					b[k]=a[d];
					for(l=0,e=k;l<=k/2;l++,e--)
					{
						if(b[l]!=b[e])
						flag=1;
					}
					if(flag==0)
					{
						sum++;
					}
					else
					flag=0;
					k++;
					d*=i;
					
				}
			}
		}
		printf("%lld\n",si+sum);
	}
	return 0;
}
