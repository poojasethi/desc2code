#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
const int mx=500000;
char str[mx+1];
int a1[mx+1],a2[mx+1];
#define ll long long
long long ct;
int main()
{
        ll i,j,k,l,m,n,o,p,fd,w,gc,d,gh,t;
        scanf("%lld",&t);
        while(t--)
        {
                scanf("%s",str);
                for(i=0;str[i];i++)
                        a1[i+1]=str[i]-'0';
                n=i;
               ct=n;
                for(i=1;i<=n;i++)
                {
                        for(p=2;p*i<=n;p++)
                        {
                                a2[0]=a1[i];k=1;
                           for(j=i*p;j<=n;j*=p)
                           {
                              a2[k]=a1[j];
                              k++;
                              bool flag=true;
                              for(l=0;l<k/2;l++)
                              {
                                if(a2[l]!=a2[k-l-1])
                                  {
                                        flag=false;
                                        break;
                                  }
                              }
                              if(flag)
                                ct++;
                           }
                        }
                }
                printf("%lld\n",ct);
        }
        return 0;
}
