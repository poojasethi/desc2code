#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;

char temp[1000];
char str[500001];
char str1[500001];

int check_palindrome(long long int low,long long int high)
{
	int i=low;
	int j=high;
	while(1)
	{
		if(i>=j)
		break;
		if(temp[i]==temp[j])
		{
			i++;
			j--;
		}
		else
		return 0;
	}
	return 1;
}


int main()
{
	long long int t,i,j,l,count=0,sum=0,x,k=0;
	scanf("%lld",&t);
	while(t--)
	{
		
		scanf("%s",str);
		l=strlen(str);
		for(i=0;i<l;i++)
		str1[i+1]=str[i];
		sum=l;
		for(i=1;i<=l;i++)
		{
			count=2;
			for(count=2;count*i<=l;count++)
			{
			j=i;
			k=0;
			temp[k++]=str1[i];
			j=j*count;
			for(;j<=l;j=j*count)
			{
			    temp[k++]=str1[j];
				x=check_palindrome(0,k-1);
				if(x==1)
				{
				sum+=1;	
			    }
			}
		   }
		}
		printf("%lld\n",sum);
		sum=0;
	}
	return 0;
}
