    #include<bits/stdc++.h>
    using namespace std;
    char ar[500001];
    string str;
    bool pal()
    {
    	int l=str.length();
    	for(int i=0;i<(l+1)/2;i++)
    	{
    		if(str.at(i)==str.at(l-1-i))
    		continue;
    		else
    		return 0;
    	}
    	return 1;
    }
    int main()
    {
    	int t;
    	cin>>t;
    	while(t--)
    	{
    		string s;
    		cin>>s;
    		int l=s.length();
    		long long ans=l;
    		for(int i=0;i<l;i++)
    		ar[i+1]=s.at(i);
    		for(int i=1;i<=l;i++)
    		{
    			long long int idx;
    			for(int p=2;(i*p)<=l;p++)
    			{
    				idx=i;
    				str.clear();
    				str.push_back(ar[idx]);
    				while(idx*p<=l)
    				{
    					idx*=p;
    					str.push_back(ar[idx]);
    					if(pal())
    					ans++;
    				}
    			}
    		}
    		cout<<ans<<endl;
    	}
    }
