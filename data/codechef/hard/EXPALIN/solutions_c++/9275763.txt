char ar[N];
int idx;

bool ispal()
{
    int l=0;
    int h=idx-1;
    while(h>l)
    {
        if(ar[l++]!=ar[h--])
        {
            return false;
        }
    }
    return true;
}

int main()
{
    int t,n,j;
    string s;
    ll ans,cnt,k;
    i(t);
    while(t--)
    {
        cnt=0;
        i(s);
        n=s.length();
        int m=n/2;
        f(i,1,m)
        {
            for(j=2;i*j<=n;++j)
            {
                idx=0;
                ar[idx++]=s[i-1];
                k=i*j;
                while(k<=n)
                {
                    ar[idx++]=s[k-1];
                    if(ispal())     ++cnt;
                    k*=j;
                }
            }
        }
        ans=cnt+n;
        o(ans);on;
    }
    return 0;
}
