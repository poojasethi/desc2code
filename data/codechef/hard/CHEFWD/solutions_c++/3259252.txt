#include<cstdio>
#include<iostream>
using namespace std;
typedef long long int ll;
typedef long long int L;
#define MAX 2048
ll mod;
int a[10][10],b[10][10],c[5][5];

void mulaa(int);
void mulab(int);
int func(ll);

char ob[MAX], *at = ob, *end = ob + MAX - 1;
void flush_buffer()
{
fwrite(ob, 1, at - ob, stdout);
at = ob;
}
void buffer(int k)
{
if(at + 10 > end) flush_buffer();
int d[10], dc = 0;
if(k < 0) *at++ = '-', k = -k;
if(k==0) d[dc++]=0;
while(k) d[dc++] = k % 10, k /= 10;
for(int k = dc; k--; *at++ = d[k] + '0');
*at++=10;
}
inline L inp()
{
    L noRead=0;
    char p=getchar_unlocked();
    for(;p<33;)
    {
     p=getchar_unlocked();
    };
    while(p>32)
    {
     noRead = (noRead << 3) + (noRead << 1) + (p - '0');
     p=getchar_unlocked();
    }
    return noRead;
};

int main()
{
    int tc,t;
    scanf("%d",&tc);
    mod = 1000000007;
    for(t = 1 ; t<= tc ; t++)
    {
        ll n;
        //scanf("%lld",&n);
        n = inp();
        int ans = 0;
        switch(n)
        {
            case 1: ans = 0;break;
            case 2: ans = 5;break;
            case 3: ans = 18;break;
            case 4: ans = 44;break;
            default : ans = func(n - 4);
        }
        //printf("%d\n",ans);
        buffer(ans);
        flush_buffer();
    }
    return 0;
}
void mulaa(int x)
{
    //for(int i = 1 ; i<= x ; i++)
    //for(int j = 1 ; j<= x ; j++)
    //c[i][j] = 0;

    for(int i = 1 ; i<=x ; i++)
    for(int j = 1 ; j<=x ; j++)
    {
    ll temp = 0;
    for(int k = 1 ; k<=x ; k++)
    {
        temp = temp + (ll)a[i][k]*a[k][j];
    }
    temp = temp%mod;
    c[i][j] = (int)temp;
    }

    for(int i = 1 ; i<= x ; i++)
    for(int j = 1 ; j<= x ; j++)
    a[i][j] = c[i][j];
}
void mulab(int x)
{
    //for(int i = 1 ; i<= x ; i++)
    //for(int j = 1 ; j<= x ; j++)
    //c[i][j] = 0;

    for(int i = 1 ; i<=x ; i++)
    for(int j = 1 ; j<=x ; j++)
    {
    ll temp = 0;
    for(int k = 1 ; k<=x ; k++)
    {
        temp = temp + (ll)b[i][k]*a[k][j];
    }
    temp = temp%mod;
    c[i][j] = (int)temp;
    }

    for(int i = 1 ; i<= x ; i++)
    for(int j = 1 ; j<= x ; j++)
    b[i][j] = c[i][j];
}
int func(ll n)
{
    int g[5];
    g[1] = 0;
    g[2] = 5;
    g[3] = 18;
    g[4] = 44;
    int ans = 0;
    int i,j;
    for(i = 1 ; i<=4 ; i++)
    {
        for(j = 1 ; j<= 4 ; j++)
        {
            a[i][j] = 0;
            b[i][j] = 0;
        }
        b[i][i] = 1;
    }
    a[1][1] = 2;
    a[1][2] = 1;
    a[1][3] = 1000000005;
    a[1][4] = 1000000006;
    a[2][1] = 1;
    a[3][2] = 1;
    a[4][3] = 1;
    while(n > 0)
    {
        if(n&1) mulab(4);
        n = n>>1;
        mulaa(4);
    }
    ll temp = 0;
    for(i = 1 ; i<=4 ; i++)
    {
        temp = temp + (ll)g[i]*b[1][5-i];
    }
    temp = temp%mod;
    ans = (int)temp;
    return ans;
}
