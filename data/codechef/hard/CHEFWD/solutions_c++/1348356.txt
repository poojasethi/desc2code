inline long long int divmod(long long int x)
{
if(x==1)
return 5;
long long int div=divmod(x/2);
if(x%2==0)
return (div*div)%rem;
return (((5*div)%rem)*div)%rem;
}

inline void mat_mul()
{
for(int i=0;i<2;i++)
{
for(int j=0;j<2;j++)
{
c[i][j]=0;
for(int k=0;k<2;k++)
{
c[i][j]=(c[i][j]+(b[i][k]*b[k][j])%rem)%rem;
}
}
}
for(int i=0;i<2;i++)
{
for(int j=0;j<2;j++)
{
b[i][j]=c[i][j];
}
}
}

inline void mat_mul2()
{

{

{

}
}
}
for(int i=0;i<2;i++)
{
for(int j=0;j<2;j++)
{
b[i][j]=c[i][j];
}
}
}
inline void mat_pow(unsigned long long int x)
{
if(x==1)
{initialize(b);return;}
if(x%2==0)
{
mat_pow(x/2);
mat_mul();
}
else
{
mat_pow(x/2);
mat_mul();
mat_mul2();
}
}
