LL f(int mask, int len){
	vector<int> v;
	int zero = 0;
	FOR(i,0,n-1) {
		int h = m[i];
		bool ok = true;
		int mas = 0;
		FOR(j,0,len-1) {
			int d = h % 10;
			int need = (mask & (1 << j)) ? 4 : 7;
			if (d > need) {
				ok = false;
				break;
			} else if (d == need){
				mas |= (1 << j);
			}
			h /= 10;
		}
		if (h > 0) ok = false;
		if (ok) {
			if (mas > 0) v.PB(mas);
			else zero++;
		}
	}
	int all = 0;
	FA(i, v) all |= v[i];
	if (all != (1 << len) - 1) return 0;
	
	FOR(i,0,(1<<len)-1) dp[i] = 0;
	dp[0] = 1 << zero;
	int ma = 0;
	int r = SZ(v);
	FA(i, v) {
		int mas = v[i];
		DFOR(j, ma, 0) {
			int next = j | mas;
			dp[next] += dp[j];
			if (next > ma) ma = next;
		}
	}
	ass(dp[(1<<len)-1] > 0);
	return dp[(1<<len) - 1];
}

void sol(){

	LL ans = 0;
	FOR(i,1,9) {
		FOR(j,0,(1<<i)-1)
			ans += f(j, i);
	}
	cout << ans << endl;
}
