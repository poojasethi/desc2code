#include <cstdio>
#include <algorithm>
using namespace std;
int Mat[2005][2005],Count[2005],l;
long long ans;
int main()
{
	int m,n,i,j;
	scanf("%d%d\n",&n,&m);
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			Mat[i][j]=getchar()-'0';
		}
		getchar();
	}
	for(i=n;i;i--)
	{
	  for(j=1;j<=m;j++)
		     if(Mat[i][j])++Count[j]; else Count[j]=0;
		
		for(j=1,l=0;j<=m;j++)
		ans+=l=min(l+1,Count[j]);
	}
	printf("%lld\n",ans);
	return 0;
}
