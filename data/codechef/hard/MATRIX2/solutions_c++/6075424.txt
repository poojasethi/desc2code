#include<bits/stdc++.h>
using namespace std;
int arr[2001][2001];
int main()
{
	int n,m;
	int i,j;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;++i)
		for(j=0;j<m;++j)
			scanf("%1d",&arr[i][j]);
			//arr[i][j]=1;

	long long  count=0;

	for(i=0;i<m;++i)
			count+=(long long )arr[n-1][i];

	for(i=0;i<(n-1);++i)
		count+=arr[i][0];
	for(i=n-2;i>=0;--i)
	{
		for(j=1;j<m;++j)
		{
			if(arr[i][j]==1)
			{
				arr[i][j]=min(arr[i+1][j],arr[i][j-1])+1;
				count+=(long long )arr[i][j];
			}
		}
	}
	cout<<count<<"\n";
	/*for(i=0;i<n;++i,cout<<"\n")
		for(j=0;j<m;++j)
			cout<<dp[i][j]<<" ";*/
	return 0;
}
