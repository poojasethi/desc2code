#define ll long long
const ll mx=100001;
const ll mod=1000000007;
const ll inf=0x3f3f3f3f;
int dp[2001][2001];
string s[2001];
void solve()
{
    int n,m,i,j;
    ll ans=0;;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
        cin>>s[i];
    for(i=0;i<n;i++)
        dp[i][0]=s[i][0]-48;
    for(i=0;i<m;i++)
        dp[0][i]=s[0][i]-48;
    for(i=1;i<n;i++)
    {
        for(j=1;j<m;j++)
        {
            if(s[i][j]=='1')
                dp[i][j]=min(dp[i-1][j-1],dp[i-1][j])+1;
            else
                dp[i][j]=0;
        }
    }
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
        ans+=dp[i][j];
    cout<<ans;
    e
}
int main()
{
    solve();
    return 0;
}
