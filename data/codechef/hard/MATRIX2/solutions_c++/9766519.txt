#include<iostream>
#include<cstdio>
#include<cmath>
#define ll long long
using namespace std;


ll arr[2002][2002],g[2002][2002],dp[2002][2002];

string str[20000];
int main()
{
	ll n,m,i,count=0,sum=0,j;
	cin>>n>>m;
	for(i=0;i<n;i++)
	cin>>str[i];
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(str[i][j]=='0')
			arr[i][j]=0;
			else
			arr[i][j]=1;
		}
	}
	
	for(i=0;i<n;i++)
	{
		if(arr[i][0]==1)
		dp[i][0]=1;
		else
		dp[i][0]=0;
	}
	
	count=0;
	for(j=1;j<m;j++)
	{
		count=0;
		for(i=n-1;i>=0;i--)
		{
			if(arr[i][j]==1)
			{
			count++;
			g[i][j]=count;
		    }
		    else
		    {
		    	count=0;
		    	g[i][j]=0;
		    }
		}
	}
	
	
	for(i=0;i<n;i++)
	{
		for(j=1;j<m;j++)
		{
			dp[i][j]=min(1+dp[i][j-1],g[i][j]);
		}
	}
	
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			//cout<<"dp="<<dp[i][j]<<" ";
			sum+=dp[i][j];
		}
		//cout<<"\n";
	}
	cout<<sum<<"\n";
	return 0;
}
