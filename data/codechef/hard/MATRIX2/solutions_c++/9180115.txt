#include<iostream>
using namespace std;

int a[2000][2000];
int table[2000][2000];
int sum[2000][2000];
int main(){
	int n,m;
	char c;
	cin>>n>>m;

	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			cin>>c;
			a[i][j]=c-'0';
		}
	}


	for(int i=0;i<m;i++)
		sum[0][i]=a[0][i];

	for(int i=1;i<n;i++){
		for(int j=0;j<m;j++){
			if(a[i][j]==0)
				sum[i][j]=0;
			else
				sum[i][j]=sum[i-1][j]+1;
			
		}
	}

	long long ans=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			if(i==0 || j==0)
				table[i][j]=a[i][j];
			else{
				table[i][j]=min(table[i-1][j-1]+1,sum[i][j]);
			}
				
			ans+=(long long)table[i][j];		
		}
	}
	cout<<ans<<endl;
	return 0;
}
