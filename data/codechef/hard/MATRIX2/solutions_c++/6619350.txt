#include<bits/stdc++.h>
using namespace std;
int minimum(int a,int b)
{
    if(a<b)
        return a;
    else
        return b;
}
int dp[2002][2002];
char a[2002][2002];
int main()
{
    int n,m,i,j;long long int ans=0;;
    scanf("%d%d",&n,&m);
    gets(a[0]);
    for(i=0;i<n;i++)
        gets(a[i]);
    for(i=0;i<n;i++)
        dp[i][0]=a[i][0]-48;
    for(i=0;i<m;i++)
        dp[0][i]=a[0][i]-48;
    for(i=1;i<n;i++)
    {
        for(j=1;j<m;j++)
        {
            if(a[i][j]=='1')
                dp[i][j]=minimum(dp[i-1][j-1],dp[i-1][j])+1;
            else
                dp[i][j]=0;
        }
    }
    for(i=0;i<n;i++)
        for(j=0;j<m;j++)
        ans+=dp[i][j];
    printf("%lld\n",ans);
}
