/*Dynamic Programming - O(N^2 Solution)*/
#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int mat[2005][2005]={{0}},dp[2005][2005]={{0}},up[2005][2005]={{0}};
char ch;
int n,m,i,j;
long long int ans;
int main()
{
	ans=0;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{	scanf(" %c",&ch);
			mat[i][j]=ch-'0';
			if(mat[i][j]==1)
				up[i][j]=up[i-1][j]+1;
			else
				up[i][j]=0;	
		}
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{	if(mat[i][j]==1)
			{
				dp[i][j]=min(dp[i-1][j-1]+1,up[i][j]);
				ans+=dp[i][j];
			}
			
		}	
	printf("%lld",ans);
	return 0;	
}
