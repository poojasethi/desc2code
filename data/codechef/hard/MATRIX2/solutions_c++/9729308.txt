#include<bits/stdc++.h>
using namespace std;
long long int dp[2002][2002]={};
char a[2002][2002];
int main()
{
long long int i,j,n,m;
ios_base::sync_with_stdio(false);
cin>>n>>m;
long long int count=0;
for(i=0;i<n;i++)
{
    cin>>a[i];
}
for(i=1;i<=n;i++)
{
    for(j=1;j<=m;j++)
    {
      if(a[i-1][j-1]-'0'==1)
      {
          dp[i][j]=1;
      }
    }
}
for(i=1;i<=n;i++)
{
    for(j=1;j<=m;j++)
    {
      if(a[i-1][j-1]-'0'==1)
       {
          dp[i][j]=1+min(dp[i-1][j],dp[i-1][j-1]);
             count+=dp[i][j];
       }
    }
}
cout<<count<<endl;
}
