    #include<stdio.h>
    int a[2001][2001];
    int min(int x,int y)
    {
        return ( x< y)?x:y;
    }

    int main()
    {
        int n,m,i,j;
        char b[2001];
        long long san=0;
        scanf("%d%d",&n,&m);
        for(i=0;i<n;i++)
        {
            getchar();
            scanf("%s",b);
            for(j=0;j<m;j++)
                if(b[j]=='0')
                    a[i][j]=0;
                else
                    a[i][j]=1;
        }
        for(i=n-1;i>=0;i--)
        {
            for(j=m-1;j>=0;j--)
            {
                if(a[i][j])
                    a[i][j]=1+min(a[i][j+1],a[i+1][j+1]);
    //printf("%d %d %d\n", i , j , a[i][j]);
                    san+=a[i][j];
            }
        }
    printf("%lld\n",san);
    return 0;
    }
