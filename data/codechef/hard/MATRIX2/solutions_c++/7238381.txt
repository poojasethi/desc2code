#include<bits/stdc++.h>

using namespace std;

#define si(x) scanf("%d",&x)
#define pri(x) printf("%d",x)
#define sll(x) scanf("%lld",&x)
#define sch(x) scanf("%c",&x)
#define pll(x) printf("%lld",x)
#define sstr(s) scanf("%s",s)
#define pstr(s) printf("%s",s)
#define nl printf("\n")
#define ll long long int

ll dp[2000][2000]={0};
int cd[2000][2000]={0},a[2000][2000]={0};

int main()
{
    int n,m,i,j;
    si(n); si(m);
    char ch,x;
    for(i=0;i<n;++i)
    {
	sch(ch);
	for(j=0;j<m;++j)
	{
	    sch(x);
	    a[i][j]=x-'0';
	}
    }
    int count[m];
    ll ans=0;
    memset(count,0,sizeof(count));
    memset(cd,0,sizeof(cd));
    memset(dp,0,sizeof(dp));
    for(i=0;i<n;++i)
    {
	for(j=0;j<m;++j)
	{
	    if(a[i][j]==1)
	    {
		if(i==0)
		    cd[i][j]=1;
		else
		    cd[i][j]=cd[i-1][j]+1;
	    }
	}
    }
    for(i=0;i<n;++i)
    {
	for(j=0;j<m;++j)
	{
	    if(a[i][j]==1)
	    {
		if(i==0 || j==0)
		{
		    count[j]=dp[i][j]=1;
		    ans++;
		}
		else
		{
		    count[j]=count[j-1]+1;
		    int w=cd[i-1][j];
//			cout<<"CD "<<w<<endl;
		    if(w >= dp[i-1][j-1])
		    {
			dp[i][j]=dp[i-1][j-1]+1;
			ans += dp[i][j];
		    }
		    else
		    {
			dp[i][j]=w+1;
			ans += dp[i][j];
		    }
		}
	    }
	}
    }
    cout<<ans<<endl;
/*
    for(i=0;i<n;++i)
    {
	for(j=0;j<m;++j)
	    cout<<dp[i][j]<<" ";
	nl;
    }
*/
    return 0;
}
