#include <iostream>
 
using namespace std;
 
int main()
{
    int n, m;
 
    cin >> n >> m;
    char a[n+1][m+1];
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= m; j++) {
            cin >> a[i][j];
        }
    }
 
    int f[2][m+2];
    int ans[2][m+2];
    for(int i = 0; i <= m+1; i++) {
        f[(n-1)&1][i] = 0;
        ans[(n-1)&1][i] = 0;
    }
    f[n&1][0] = ans[n&1][0] = 0;
    long long res = 0;
    for(int i = n; i > 0; i--) {
        for(int j = 1; j <= m; j++) {
            if(a[i][j] == '1') {
                ans[i&1][j] = ans[(i-1)&1][j]+1;
            }
            else {
                ans[i&1][j] = 0;
            }
            f[i&1][j] = min(f[i&1][j-1]+(a[i][j]=='1'?1:0), ans[i&1][j]);
            res += f[i&1][j];
           
        }
        
    }
    cout << res << endl;
}
