#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

typedef long long ll;

const int MAX=2005;
int dp[MAX][MAX];

int main() {
    int n, m;
    cin >> n >> m;
    vector<string> s(n+1);

    for (int i = 1; i <= n; i++) {
        cin >> s[i];
        s[i] = "#"+s[i];
    }
    for (int i = n; i >= 1; i--) {
        for (int j = m; j >= 1; j--) {
            if (s[i][j]=='1') {
                dp[i][j]=min(dp[i][j+1], dp[i+1][j+1])+1;
            }
        }
    }

    /*for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            cout << dp[i][j] << " ";
        }
        cout << endl;
    }*/

    ll answer=0;
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            answer += dp[i][j];
        }
    }

    cout << answer << endl;

}
