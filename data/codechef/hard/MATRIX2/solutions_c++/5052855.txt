#include<stdio.h>
 
int main(int agrc,char*agrv[]){
  int N,M;scanf("%d %d",&N,&M);
  int *hle=new int[M];
  int *diagle=new int[M];
  char *s=new char[M];
  for(int i=0;i<M;i++){
    hle[i]=0;
    diagle[i]=0;
  }
  long long int count=0;
  for(int r=0;r<N;r++){
    scanf("%s",s);
    for(int c=0;c<M;c++)
      if(s[c]=='1'){
        hle[c]++;
      }else{
        hle[c]=0;
      }
    for(int c=M-1;c>0;c--){
      if(hle[c]>diagle[c-1]){
        diagle[c]=diagle[c-1]+1;
      }else{
        diagle[c]=hle[c];
      }
      count+=diagle[c];
    }
    diagle[0]=s[0]-48;
    count+=diagle[0];
  }
  printf("%lld",count);
  delete [] hle;
  delete [] diagle;
  delete [] s;
  return 0;
}
