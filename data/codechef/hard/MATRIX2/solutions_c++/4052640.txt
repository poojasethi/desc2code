#define get getchar_unlocked

//fast function to scan various integer data types
template <typename T>
inline T input() {
	T n = 0;
	int s = 1;
	char p = get();

	while((p < '0' || p > '9') && p != EOF && p != '-')
		p = get();

	if(p == '-'){ s = -1, p = get(); }

	while(p >= '0' && p <= '9') {
		n = (n << 3) + (n << 1) + (p - '0');
		p = get();
	}
	return n*s;
}

#define MAX 2009

int main(int argc, char** argv) {
#ifdef LOCAL
    freopen("in", "r", stdin);
#endif

    ios_base::sync_with_stdio(false);

    int n, m;
    cin >> n >> m;
    lli ans = 0;

    short DP[MAX][MAX];

    string s[MAX];

    rep(i, n) { 
    	cin >> s[i];
    }

    for(int i = n-1; i >= 0; i--) {
    	for(int j = 0; j < m; j++) {
    		if(s[i][j] == '0')
    			DP[i][j] = 0;

    		else if(j == 0 || i == n-1)
    			DP[i][j] = 1;

    		else {
    			DP[i][j] = min(DP[i][j-1], DP[i+1][j]) + 1;
    		}
    		ans += DP[i][j];
    	}
    }

    cout << ans << endl;

	return 0;
}
