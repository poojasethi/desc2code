int n,m,d[2002][2002],con[2002][2002],dp[2002][2002],flag[2002][2002];
int rec(int i,int j)
{
    if(i>n or j>m)return 0;
    if(d[i][j]==0)return 0;
    if(flag[i][j])return dp[i][j];
    if(con[i][j]>=1+rec(i+1,j+1))
        dp[i][j]=1+rec(i+1,j+1);
        else
            dp[i][j]=con[i][j];
    flag[i][j]=1;
    if(dp[i][j]<=1)dp[i][j]=1;
    return dp[i][j];
}
int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;++i)
    {
        for(int j=1;j<=m;++j)
        {
            char c;
            cin>>c;
            if(c=='1')d[i][j]=1;
            else d[i][j]=0;
        }
    }
    for(int i=1;i<=n;++i)
    {
        for(int j=m;j>=1;j--)
        {
            if(d[i][j]==0)con[i][j]=0;
            else con[i][j]=1+con[i][j+1];
        }
    }
    lld ans=0;
    for(int i=1;i<=n;++i)
    {
        for(int j=1;j<=m;++j)
        {
  //          cout<<rec(i,j)<<' ';
            ans+=rec(i,j);
        }
//        cout<<endl;
    }
    cout<<ans<<endl;
}
