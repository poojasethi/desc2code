#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int dp[5005][5005];
char a[5005][5005];
int n,m;
ll minm(ll a,ll b)
{
    return (a<b)?a:b;
}
ll ok(int i,int j)
{
    if(i>=0&&i<n&&j>=0&&j<m)
        return dp[i][j];
    return 0;
}
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)
    {
       scanf("%s",a[i]);
    }
  ll res=0;
    for(int j=0;j<m;j++)
    {
        dp[0][j]=a[0][j]=='1';
        res+=dp[0][j];
    }
    for(int i=1;i<n;i++)
    {
        for(int j=0;j<m;j++)
        {
            if(a[i][j]=='1')
            {
                dp[i][j]=minm(ok(i-1,j),ok(i-1,j-1))+1;
                res+=dp[i][j];
            }
        }
    }
    printf("%lld\n",res);
    return 0;
}
/*
 dp[i][j]= max size of square submatrix such that all cells on and above the main diagonal is 1.
 here (i,j) is the bottom right corner of half submatrix.
*/
