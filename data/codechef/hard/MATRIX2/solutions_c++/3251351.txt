#include<iostream>
#include<stdio.h>
#include<math.h>
#define l long long int
using namespace std;
int main()
{
	l count=0;
	l n,m,**mat;
	cin>>n>>m;

	//allocating the memory for matrix
	mat=new  l*[n];
	for(l i=0;i<n;i++)
 		mat[i]=new l[m];


	char c;


	//taking the input matrix
	for(l i=0;i<n;i++)
		{
			c=getchar();
			for(l j=0;j<m;j++)
			{
				c=getchar();
				mat[i][j]=(int)(c-48);
			}

		}

	for(l i=1;i<n;i++)
		for(l j=1;j<m;j++)
		{
			if(mat[i][j])
				mat[i][j]=min(mat[i-1][j],mat[i-1][j-1])+1;
		}
	for(l i=0;i<n;i++)
		for(l j=0;j<m;j++)
			count+=mat[i][j];

 		cout<<count;
		delete mat;
return 0;
}
