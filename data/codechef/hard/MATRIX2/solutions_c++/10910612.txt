#include<bits/stdc++.h>

using namespace std;

long long n, ar[2001][2001], m;

int main()
{
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
	{
		string s;
		cin >> s;
		for (int j = 0; j < m; j++)
			ar[i][j + 1] = s[j] - 48;	
	}
	
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			if (ar[i][j] == 1)
			{
				ar[i][j] = min(ar[i - 1][j], ar[i - 1][j - 1]) + 1;
			}
		}
	}
	long long ans = 0;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= m; j++)
		{
			ans += ar[i][j];
		}
	}
	cout << ans;
	return 0;
}
