#include<bits/stdc++.h>
using namespace std;
int arr[2001][2001],dp[2001][2001];
int main()
{
	int n,m;
	int i,j;
	cin>>n>>m;
	for(i=0;i<n;++i)
		for(j=0;j<m;++j)
			scanf("%1d",&arr[i][j]);
			//arr[i][j]=1;

	long long  count=0;

	for(i=0;i<m;++i)
		{
			dp[n-1][i]=arr[n-1][i];
			count+=dp[n-1][i];
		}

	for(i=0;i<(n-1);++i)
	{
		dp[i][0]=arr[i][0];
		count+=dp[i][0];
	}
	for(i=n-2;i>=0;--i)
	{
		for(j=1;j<m;++j)
		{
			if(arr[i][j]==1)
			{
				dp[i][j]=min(dp[i+1][j],dp[i][j-1])+1;
				count+=(long long )dp[i][j];
			}
		}
	}
	cout<<count<<"\n";
	/*for(i=0;i<n;++i,cout<<"\n")
		for(j=0;j<m;++j)
			cout<<dp[i][j]<<" ";*/
	return 0;
}
