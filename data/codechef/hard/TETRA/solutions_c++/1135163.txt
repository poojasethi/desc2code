#include<stdio.h>
#include<math.h>
double area(double a,double b,double c)
{
	double s=0,result;
	s=(a+b+c)/2;
	result=sqrt(s*(s-a)*(s-b)*(s-c));
	return result;
}
int main()
{
	double L,M,N,O,P,Q,A,B,C,D,E,F,vol;
	int t;
	double v,s,r;
	scanf("%d",&t);
	while(t--)
	{
		s=0;
		scanf("%lf%lf%lf%lf%lf%lf",&Q,&M,&O,&N,&P,&L);
		F=L*L;
		B=M*M;
		D=N*N;
		C=O*O;
		E=P*P;
		A=Q*Q;
		v=A*C*D + B*C*D + A*B*E + B*C*E+ B*D*E + C*D*E + A*B*F + A*C*F + A*D*F + C*D*F + A*E*F + B*E*F- C*C*D - C*D*D - B*B*E - B*E*E - A*A*F - A*F*F- A*B*D - A*C*E - B*C*F - D*E*F; 
		vol=sqrt(v)/12;
		s=area(M,N,Q)+area(O,P,Q)+area(M,L,O)+area(L,P,N);
		r=(vol*3)/s;
		printf("%.4lf\n",r);
	}
	return 0;
}
