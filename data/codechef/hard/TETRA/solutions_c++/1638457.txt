double areatri(int a, int b, int c)
{
	double s = ((double)(a+b+c))/2.0;
	double ar = s*(s-a)*(s-b)*(s-c);
	return sqrt(ar);
}

double volume(int U, int V, int W, int u, int v, int w)
{
	double a, b, c, d, X, x, Y, y, Z, z;
	X = (double)((w-U+v)*(U+v+w));
	x = (double)((U-v+w)*(v-w+U));
	Y = (double)((u-V+w)*(V+w+u));
	y = (double)((V-w+u)*(w-u+V));
	Z = (double)((v-W+u)*(W+u+v));
	z = (double)((W-u+v)*(u-v+W));
	a = sqrt(x*Y*Z);
	b = sqrt(y*Z*X);
	c = sqrt(z*X*Y);
	d = sqrt(x*y*z);
	//cout << a << " " << b << " " << c << " " << d << endl;
	double vol = (-a+b+c+d)*(a-b+c+d)*(a+b-c+d)*(a+b+c-d);
	//cout << vol << endl;
	vol = sqrt(vol);
	vol /= 192.0*(double)(u*v*w);
	return vol;
}

int main()
{
	cout << setprecision(4) << fixed;
	int T;
	scanf("%d", &T);
	int u, v, w, U, V, W;
	for (int t = 0; t < T; t++)
	{
		scanf("%d %d %d %d %d %d", &w, &v, &u, &U, &V, &W);
		double S = 0.0, Vol;
		S += areatri(V, w, u);
		S += areatri(U, w, v);
		S += areatri(W, v, u);
		S += areatri(U, V, W);
		//cout << S << endl;
		Vol = volume(U, V, W, u, v, w);
		cout << 3.0*Vol/S << "\n";
	}
	return 0;
}
