#include <cstdio>
#include <cmath>
using namespace std;
double heron(double a,double b,double c)
{
double s=(a+b+c)/2.0;
    return sqrt(s*(s-a)*(s-b)*(s-c));
}
int main()
{
int t;
scanf("%d",&t);
while(t--)
{
    double U,V,W,u,v,w;
    scanf("%lf %lf %lf %lf %lf %lf",&U,&V,&w,&W,&v,&u);
    double X=(w-U+v)*(U+v+w);
    double x=(U-v+w)*(v-w+U);
    double Y=(u-V+w)*(V+w+u);
    double y=(V-w+u)*(w-u+V);
    double Z=(v-W+u)*(W+u+v);
    double z=(W-u+v)*(u-v+W);
    double a=sqrt(x*Y*Z);
    double b=sqrt(y*X*Z);
    double c=sqrt(z*X*Y);
    double d=sqrt(x*y*z);
    double volume=sqrt((-a+b+c+d)*(a-b+c+d)*(a+b-c+d)*(a+b+c-d))/(192*u*v*w);
    double a1=heron(U,V,W),a2=heron(U,w,v),a3=heron(V,w,u),a4=heron(W,v,u);
    printf("%.4lf\n",(3*volume)/(a1+a2+a3+a4));
}

    return 0;
}
