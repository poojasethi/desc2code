#include<bits/stdc++.h>
double area(double, double, double);

int main()
{
double a, A, b, B, c, C, d, D, e, E, f, F, r, S, V;
int N;
scanf("%d", &N);
while (N--)
{
scanf("%lf %lf %lf %lf %lf %lf", &d, &e, &f, &a, &b, &c);
A = a * a;
B = b * b;
C = c * c;
D = d * d;
E = e * e;
F = f * f;
V = sqrt(A * F * (-A + B + C + D + E - F) + B * E * (A - B + C + D - E + F) + C * D * (A + B - C - D + E + F)
- (A + F) * (B + E) * (C + D) / 2.0 - (A - F) * (B - E) * (C - D) / 2.0) / 12.0;
S = area(a, b, c) + area(a, d, e) + area(b, d, f) + area(c, e, f);
r = V * 3.0 / S;
printf("%.4lf\n", r);
}
return 0;
}

double area(double a, double b, double c)
{
double s;
s = (a + b + c) / 2.0;
return sqrt(s * (s - a) * (s - b) * (s - c));
}
