using namespace std;

double area(double a, double b, double c)
{
    double s = (a + b + c)/2.0;
    return sqrt( s*(s - a)*(s - b)*(s - c) );
}

int main()
{
    int t = SI;
    while(t--)
    {
        double wx, wy, wz, xy, xz, yz, S = 0, vol;
        scanf("%lf %lf %lf %lf %lf %lf", &wx, &wy, &wz, &xy, &xz, &yz);
        S += area(wx, wz, xz);
        S += area(wy, wz, yz);
        S += area(wx, wy, xy);
        S += area(xy, yz, xz);
        vol = sqrt( 4 * wx*wx * wy*wy * wz*wz
                    - wx*wx * (wy*wy + wz*wz - yz*yz)*(wy*wy + wz*wz - yz*yz)
                    - wy*wy * (wz*wz + wx*wx - xz*xz)*(wz*wz + wx*wx - xz*xz)
                    - wz*wz * (wx*wx + wy*wy - xy*xy)*(wx*wx + wy*wy - xy*xy)
                    + (wy*wy + wz*wz - yz*yz)*(wz*wz + wx*wx - xz*xz)*(wx*wx + wy*wy - xy*xy) )/12;
        printf("%0.4lf\n", vol*3.0/S);
    }
    return 0;
}
