#include <iostream>
#include<cmath>
using namespace std;
double areatriangle(double a,double b,double c)
{
	double s;
	s=(a+b+c)/2.0;
	return sqrt(s*(s-a)*(s-b)*(s-c));
}
int main(void) 
{
	int t;
	double area,vol,a,b,c,d,e,f,A,B,C,D,E,F;
	cin>>t;
	while(t--)
	{
		cin>>a>>b>>c>>d>>e>>f;
		A=a*a;
		B=b*b;
		C=c*c;
		D=d*d;
		E=e*e;
		F=f*f;
		area=0;
		area+=areatriangle(f,b,c);
		area+=areatriangle(a,b,d);
		area+=areatriangle(d,e,f);
		area+=areatriangle(a,e,c);
		vol=(sqrt((A*C*D + B*C*D + A*B*E + B*C*E+ B*D*E + C*D*E + A*B*F + A*C*F + A*D*F + C*D*F + A*E*F + B*E*F) - (A*B*D + A*C*E + B*C*F + D*E*F + C*C*D + C*D*D + B*B*E + B*E*E + A*A*F + A*F*F)))/12.0;
		printf("%.4lf\n",(vol*3.0)/area);	
	}
	return 0;
}
