#include<iostream>
#include<cmath>
#include<cstdio>
#define s(a) a*a
#define sq(a) a*a
using namespace std;
double area(double x, double y, double z)
{
    double s = (x+y+z)/2;
    return sqrt(s*(s-x)*(s-y)*(s-z));
}
int main()
{
    double a,b,c,d,e,f,ans,v,r;
    double m,n,o,p;
    int t;
    cin>>t;
    while(t--)
    {
        cin>>a>>b>>c>>d>>e>>f;
        ans = ( s(a)+s(b)+s(c)+s(d)+s(e)+s(f) ) * ( s(a*f) + s(b*e)+s(c*d));
        ans = ans - 2*sq(a*f)*(a*a+f*f);
        ans = ans - 2*sq(b*e)*(b*b+e*e);
        ans = ans - 2*sq(c*d)*(c*c+d*d);
        ans = ans - (((sq(a)+sq(f))*(sq(b)+sq(e))*(sq(c)+sq(d)))/2);
        ans = ans + (((sq(a)-sq(f))*(sq(b)-sq(e))*(sq(c)-sq(d)))/2);
        v = sqrt(ans) / 12.0;
        m = area(a,d,b);
        n = area(a,e,c);
        o = area(d,f,e);
        p = area(b,c,f);
        double r = 3.0*v/(m+n+o+p);
        printf("%.4lf\n",r);
    }
    return 0;
}
