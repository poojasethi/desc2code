#include <cstdio>
#include <cmath>
double sqr(double a);
 
double area(double s1,double s2,double s3);
int main()
{
double a[6],S1,S2,S3,S4,S5,S6,vol,S;
int n,i;
scanf("%d", &n);
while(n--) {
S = 0;
for(i=0;i<6;++i)
{
scanf("%lf",&a[i]);
}
S1=a[2];S2=a[3];
S3=a[4];S4=a[1];
S5=a[5];S6=a[0];
S+=area(a[2],a[1],a[5]);
S+=area(a[0],a[2],a[4]);
S+=area(a[0],a[1],a[3]);
S+=area(a[3],a[4],a[5]);
vol=sqrt((4*sqr(S1)*sqr(S3)*sqr(S5)-sqr(S1)*sqr(sqr(S3)+sqr(S5)-sqr(S2))- sqr(S3)*sqr(sqr(S5)+sqr(S1)-sqr(S4))- sqr(S5)*sqr(sqr(S1)+sqr(S3)-sqr(S6))+ (sqr(S3)+sqr(S5)-sqr(S2))*(sqr(S5)+sqr(S1)-sqr(S4))*(sqr(S1)+sqr(S3)-sqr(S6))))/12.0;
printf("%.4lf\n", vol*3.0/S);
}
return 0;
}
 
double sqr(double a)
{
return (a*a);
}
 
double area(double s1, double s2, double s3)
{
double s = (s1 + s2 + s3)/2.0;
return sqrt(s*(s-s1)*(s-s2)*(s-s3));
}
