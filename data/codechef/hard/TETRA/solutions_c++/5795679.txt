#include<stdio.h>
#include<math.h>
double area(double a,double b,double c)
{
double s;
s=(a+b+c)/2.0;
return sqrt(s*(s-a)*(s-b)*(s-c));
}
int main()
{
int t;
double sarea,vol,a,b,c,d,e,f,A,B,C,D,E,F;
scanf("%d",&t);
while(t--)
{
scanf("%lf%lf%lf%lf%lf%lf",&a,&b,&c,&d,&e,&f);
A=a*a;
B=b*b;
C=c*c;
D=d*d;
E=e*e;
F=f*f;
sarea=0;
sarea+=area(f,b,c);
sarea+=area(a,b,d);
sarea+=area(d,e,f);
sarea+=area(a,e,c);
vol=(sqrt((A*C*D + B*C*D + A*B*E + B*C*E+ B*D*E + C*D*E + A*B*F + A*C*F + A*D*F + C*D*F + A*E*F + B*E*F) - (A*B*D + A*C*E + B*C*F + D*E*F + C*C*D + C*D*D + B*B*E + B*E*E + A*A*F + A*F*F)))/12.0;
printf("%.4lf\n",(vol*3.0)/sarea);
}
return 0;
}
