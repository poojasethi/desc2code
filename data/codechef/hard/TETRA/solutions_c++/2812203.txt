#include<stdio.h>
#include<math.h>
const double CONS = 1e-9;
double surface(double a,double b,double c)
{
	return sqrt((a + b + c) * (-a + b + c) * (a - b + c) * (a + b - c)) / 4.0;
}
double vol(double a,double b,double c,double d,double e,double f)
{
	double u,v,w;
	u=b*b+c*c-d*d;
	v=c*c+a*a-e*e;
	w=a*a+b*b-f*f;
	return sqrt(4.0*a*a*b*b*c*c-a*a*u*u-b*b*v*v-c*c*w*w+u*v*w)/12.0;
}
int main()
{
	int t;
	double WX,WY,WZ,XY,XZ,YZ,surf,rad;
	scanf("%d",&t);
	while(t--)
	{
	scanf("%lf%lf%lf%lf%lf%lf",&WX,&WY,&WZ,&XY,&XZ,&YZ);
	surf = surface(WX, WY, XY) + surface(WX, XZ, WZ) + surface(WY, YZ, WZ) + surface(XY, XZ, YZ);
	rad = vol(WX, WY, WZ, YZ, XZ, XY)* 3.0 / surf;
	printf("%.4lf\n", rad + CONS);
	}
	return 0;
}
