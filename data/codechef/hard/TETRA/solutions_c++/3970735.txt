#include<iostream>
#include<math.h>
using namespace std;
double area(double a1,double a2,double a3)
{
	double s=(a1+a2+a3)/2.0;
	return sqrt(s*(s-a1)*(s-a2)*(s-a3));
}
double SQ(double a)
{
  return (a)*(a);
}
int main()
{
	int t;
	double wx,wy,wz,xy,xz,yz,vol,S;
	scanf("%d",&t);
	while(t)
	{
		
		 
		scanf("%lf%lf%lf%lf%lf%lf",&wx,&wy,&wz,&xy,&xz,&yz);

		S=area(wz,wy,yz);
		S+=area(wx,wz,xz);
		S+=area(wx,wy,xy);
		S+=area(xy,xz,yz);
		vol=sqrt(
			  (4*SQ(wz)*SQ(xz)*SQ(yz)
			  -SQ(wz)*SQ(SQ(xz)+SQ(yz)-SQ(xy))
			 - SQ(xz)*SQ(SQ(yz)+SQ(wz)-SQ(wy))
            - SQ(yz)*SQ(SQ(wz)+SQ(xz)-SQ(wx)) 
            + (SQ(xz)+SQ(yz)-SQ(xy))
              *(SQ(yz)+SQ(wz)-SQ(wy))
              *(SQ(wz)+SQ(xz)-SQ(wx))
			  ))/12.0;

		 printf("%.4lf\n",vol*3.0/S);
		t--;
	}

	return 0;
}
