//Piero della Francesca's Tetrahedron Formula
#include<stdio.h>
#include<math.h>
int main()
{int t;double a,b,c,d,e,f,A,B,C,D,E,F,s1,s2,s3,s4,a1,a2,a3,a4,AREA,V,R;
 scanf("%d",&t);
 while(t--)
 {scanf("%lf %lf %lf %lf %lf %lf",&a,&b,&c,&d,&e,&f);
  s1=(a+b+d)/2;  s2=(d+e+f)/2;  s3=(a+c+e)/2;  s4=(b+c+f)/2;
  a1=sqrt(s1*(s1-a)*(s1-b)*(s1-d));  a2=sqrt(s2*(s2-d)*(s2-e)*(s2-f));
  a3=sqrt(s3*(s3-a)*(s3-c)*(s3-e));  a4=sqrt(s4*(s4-b)*(s4-c)*(s4-f));
  AREA=a1+a2+a3+a4;
  A=a*a; B=b*b; C=c*c; D=d*d; E=e*e; F=f*f; 
  V=  A*C*D + B*C*D + A*B*E + B*C*E + B*D*E + C*D*E + A*B*F + 
      A*C*F + A*D*F + C*D*F + A*E*F + B*E*F - A*B*D - A*C*E - 
      B*C*F - D*E*F - C*C*D - C*D*D - B*B*E - B*E*E - A*A*F - A*F*F;
  V=sqrt(V)/12;
  R=(V*3)/AREA;
  printf("%.4lf\n",R);
  }
  return 0;
}
