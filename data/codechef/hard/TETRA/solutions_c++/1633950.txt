#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstdlib>
using namespace std;

double area(double a,double b,double c)
{      double s;
       s=(a+b+c)/2;
       double area1;
       area1=sqrt(s*(s-a)*(s-b)*(s-c));
       return area1;
}

int main()
{
    int t;
    
    scanf("%d",&t);
    if(t<1)
    exit(1);
    
    double u[t],v[t],w[t],U[t],V[t],W[t];
    double volume[t],sarea[t];
    

    
    for(int i=0;i<t;i++)
    {
            scanf("%lf %lf %lf %lf %lf %lf",&u[i],&v[i],&w[i],&W[i],&V[i],&U[i]);
            volume[i]=sqrt(4*u[i]*u[i]*v[i]*v[i]*w[i]*w[i]- u[i]*u[i]*(v[i]*v[i]+w[i]*w[i] - U[i]*U[i])*(v[i]*v[i]+w[i]*w[i]-U[i]*U[i]) - v[i]*v[i]*(w[i]*w[i]+u[i]*u[i] - V[i]*V[i])*(w[i]*w[i]+u[i]*u[i]-V[i]*V[i]) - w[i]*w[i]*(u[i]*u[i] + v[i]*v[i] - W[i]*W[i])*(u[i]*u[i]+v[i]*v[i]-W[i]*W[i]) + (v[i]*v[i] +w[i]*w[i] - U[i]*U[i])*(w[i]*w[i]+u[i]*u[i]- V[i]*V[i])*(u[i]*u[i]+v[i]*v[i] - W[i]*W[i]))/12; 
            
            sarea[i]=area(u[i],V[i],w[i])+area(u[i],W[i],v[i])+area(v[i],U[i],w[i])+area(W[i],V[i],U[i]);

    }        
    
    
    
for(int i=0;i<t;i++)
{
        double ans;
        ans=(3*volume[i])/sarea[i];
        printf("%.4lf\n",ans);
            
}
return 0;
}
