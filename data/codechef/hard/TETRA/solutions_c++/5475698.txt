#include<iostream>
#include<math.h>
#include<iomanip>
using namespace std;
int main()
{
	double a,b,c,d,e,f;
	double x,y,p,q,r;
	double V,A1,A2,A3,A4;
	double R;
	int T;
	cin>>T;
	while(T)
	{
		cin>>a>>c>>e>>b>>f>>d;
		y = (a*a+c*c-b*b)/(2*c);
		x = sqrt(a*a-y*y);
		q = (c*c + e*e - d*d)/(2*c);
		p = (a*a + e*e - f*f -2*q*y)/(2*x);
		r = sqrt(e*e - q*q - p*p);
		V = fabs(r*c*x/6);
		A1 = fabs(c*x)/2;
		A2 = 0.5*sqrt(r*r*y*y + r*r*x*x + (p*y-q*x)*(p*y-q*x));
		A3 = fabs(c)/2*sqrt(p*p+r*r);
		A4 = 0.5*sqrt(r*r*(y-c)*(y-c) + r*r*x*x + (p*y-p*c-q*x+c*x)*(p*y-p*c-q*x+c*x));
		R = 3*V/(A1+A2+A3+A4);
		T--;
		cout<<fixed<<setprecision(4)<<R<<endl;
	}
	return 0;
}
