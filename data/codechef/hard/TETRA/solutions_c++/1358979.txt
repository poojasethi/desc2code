#include<stdio.h>
#include<math.h>
double area(double p,double q,double r)
{
 double e,d=0;
 d=(p+q+r)/2;
 e=sqrt(d*(d-p)*(d-q)*(d-r));
 return e;
} 
int main()
{
 int t;
 double u,v,w,x,y,z;
 double a,b,c,d,e,f,vol1;
 double s,r,vol;
 scanf("%d",&t);
while(t--)
{
 s=0;
 scanf("%lf%lf%lf%lf%lf%lf",&u,&v,&w,&x,&y,&z);
 a=u*u;b=v*v;c=w*w;d=x*x;e=y*y;f=z*z;
 vol1=((d+f)*(a*c+b*e))+((d+e)*(b*c+a*f))+((e+f)*(c*d+a*b));
 vol1-=(c*d*(c+d))+(b*e*(b+e))+(a*f*(a+f))+(d*(a*b+e*f))+(c*(a*e+b*f));
 vol=sqrt(vol1)/12;
 s=area(u,v,x)+area(w,y,u)+area(v,w,z)+area(x,y,z);
 r=((3.0)*vol)/s;
  printf("%.4lf\n",r);
}
return 0;
}
