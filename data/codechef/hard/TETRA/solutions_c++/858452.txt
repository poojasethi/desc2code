using namespace std;

double edges[6];

double sa(double a,double b,double c)
{
    double s=(a+b+c)/2;
    //cout<<"s = "<<s<<endl;
    double prod=sqrt(s*(s-a));
    double prod2=sqrt((s-b)*(s-c));
    //cout<<"sa^2 = "<<prod*prod2<<endl;
    return prod*prod2;
}

int main()
{
   // freopen("input.txt","r",stdin);
   // freopen("output.txt","w",stdout);
    int m,n,t,temp;

    scanf("%d",&t);

    while(t)
    {
        for(int i=0;i<6;i++)
        {
            cin>>edges[i];
            //cout<<edges[i]<<endl;
        }
        double A=edges[0]*edges[0];
        double B=edges[1]*edges[1];
        double C=edges[2]*edges[2];
        double D=edges[3]*edges[3];
        double E=edges[4]*edges[4];
        double F=edges[5]*edges[5];
        double form = - (A*B*D) - (A*C*E) - (B*C*F) - (D*E*F) + (A*C*D) + (B*C*D) + (A*B*E) + (B*C*E) + (B*D*E) + (C*D*E) + (A*B*F) + (A*C*F) + (A*D*F) + (C*D*F) + (A*E*F) + (B*E*F) - (C*C*D) - (C*D*D) - (B*B*E) - (B*E*E) - (A*A*F) - (A*F*F);
        double V  = (sqrt(form))/12;
        double S=0;
        S=sa(edges[0],edges[1],edges[3])+sa(edges[0],edges[2],edges[4])+sa(edges[3],edges[4],edges[5])+sa(edges[1],edges[2],edges[5]);
        double ans = (3*V)/S;
        //cout<<form<<endl;
        //cout<<V<<endl;
        printf("%.4f\n",ans);

        t--;
    }
    return 0;
}
