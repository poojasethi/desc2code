#include<iostream>
#include<vector>
#include<cmath>
#include<cstdio>

using namespace std;

float ff(float a,float b,float c);
int main()
{
	int i,j,k,n;
	float u,U,v,V,W,w,X,x,Y,y,Z,z,a,b,c,d,vol,a1,a2,a3,a4;
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%f%f%f%f%f%f",&U,&V,&w,&W,&v,&u);
		X=(w-U+v)*(U+v+w);
		x=(U-v+w)*(v-w+U);
		Y=(u-V+w)*(V+w+u);
		y=(V-w+u)*(w-u+V);
		Z=(v-W+u)*(v+W+u);
		z=(W-u+v)*(u-v+W);
		a=sqrt(x*Y*Z);
		b=sqrt(y*X*Z);
		c=sqrt(z*X*Y);
		d=sqrt(x*y*z);
		vol=(sqrt((-a+b+c+d)*(a-b+c+d)*(a+b-c+d)*(a+b+c-d))/(192*u*v*w));
		a1=ff(U,v,w);
		a2=ff(u,V,w);
		a3=ff(u,v,W);
		a4=ff(U,V,W);
		printf("%.4lf\n",3*vol/(a1+a2+a3+a4));
	}
	return 0;
}
float ff(float a,float b,float c)
{
	float aa;
	aa=(sqrt((-a+b+c)*(c+a-b)*(a+b-c)*(a+c+b))/4);
	return aa;
}
