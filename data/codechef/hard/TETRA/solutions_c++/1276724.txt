#include <iostream>
#include <iomanip>
#include <math.h>

using namespace std;

double volFromSides(double w, double v, double u, double U, double V, double W) {
	double X=(-U+v+w)*(U+v+w);
	double x=(U-v+w)*(U+v-w);
	double Y=(u-V+w)*(u+V+w);
	double y=(u+V-w)*(-u+V+w);
	double Z=(u+v-W)*(u+v+W);
	double z=(-u+v+W)*(u-v+W);
	double a=sqrt(x*Y*Z);
	double b=sqrt(X*y*Z);
	double c=sqrt(X*Y*z);
	double d=sqrt(x*y*z);
	return sqrt((-a+b+c+d)*(a-b+c+d)*(a+b-c+d)*(a+b+c-d))/(192.0*u*v*w);
}

double area(double a, double b, double c) {
	return sqrt((a+b+c)*(-a+b+c)*(a-b+c)*(a+b-c))/4.0;
}

int main() {
	int T;
	cin>>T;
	for(int t=0;t<T;t++) {
		double wx,wy,wz,xy,xz,yz;
		cin>>wx>>wy>>wz>>xy>>xz>>yz;
		double volume = volFromSides(wx,wy,wz,xy,xz,yz);
		double area1 = area(wy,wz,yz);
		double area2 = area(wx,wz,xz);
		double area3 = area(wx,wy,xy);
		double area4 = area(xy,xz,yz);
		cout<<fixed<<setprecision(4)<<3*volume/(area1+area2+area3+area4)<<endl;
	}
	
}
