/*144 V^2  = - ABD - ACE - BCF - DEF + ACD + BCD + ABE + BCE
           + BDE + CDE + ABF + ACF + ADF + CDF + AEF + BEF
           - CCD - CDD - BBE - BEE - AAF - AFF

144 V^2  = - ABC - ADE - BDF - CEF + ACD + BCD + ABE + BCE
           + BDE + CDE + ABF + ACF + ADF + CDF + AEF + BEF
           - CCD - CDD - BBE - BEE - AAF - AFF*/
#include<iostream>
#include<cmath>
using namespace std;
 long long int n,a,b,c,d,e,f,A,B,C,D,E,F;
  long double V,S,r,p1,p2,p3,p4;
int main()
{

scanf("%lld",&n);
while(n--)
{
  scanf("%lld%lld%lld%lld%lld%lld",&a,&b,&c,&d,&e,&f);
  A=a*a;C=c*c;D=d*d;
  B=b*b;E=e*e;F=f*f;
  V=sqrt(- A*B*D - A*C*E - B*C*F - D*E*F + A*C*D + B*C*D + A*B*E + B*C*E+ B*D*E + C*D*E + A*B*F + A*C*F + A*D*F + C*D*F + A*E*F + B*E*F- C*C*D - C*D*D - B*B*E - B*E*E - A*A*F - A*F*F)/12;
p1=(long double)(e+d+f)/2;p2=(long double)(a+c+e)/2;p3=(long double)(a+b+d)/2;p4=(long double)(c+b+f)/2;
  S=sqrt((p1*(p1-e)*(p1-d)*(p1-f)))+sqrt((p2*(p2-a)*(p2-c)*(p2-e)))+sqrt((p3*(p3-a)*(p3-d)*(p3-b)))+sqrt((p4*(p4-c)*(p4-b)*(p4-f)));
  r=3*V/S;
printf("%.4Lf\n",r);
}
return 0;
}
