#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
double area(int a,int b,int c)
{
    double s=(a+b+c)/2.0;
    double t=0;
    t=sqrt((s*(s-a)*(s-b)*(s-c)));
    return t;
}

int main()
{
    int t;
    double u,v,w,U,V,W;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%lf%lf%lf%lf%lf%lf",&u,&v,&w,&W,&V,&U);
        double vol=0;
        vol=sqrt(((4*u*u*v*v*w*w)-(u*u*(v*v+w*w-U*U)*(v*v+w*w-U*U))-(v*v*(w*w+u*u-V*V)*(w*w+u*u-V*V))-(w*w*(u*u+v*v-W*W)*(u*u+v*v-W*W))+((v*v+w*w-U*U)*(w*w+u*u-V*V)*(u*u+v*v-W*W))))/12.0;
        double a1 = area(u,v,W);
		double a2 = area(U,v,w);
		double a3 = area(V,u,w);
		double a4 = area(W,U,V);
        double r;
        r=(vol*3.0)/(a1+a2+a3+a4);
        printf("%.4lf\n",r);
    }
}
