#include<iostream>
#include<math.h>
#include<iomanip>

using namespace std;
int main()
{
  int n;
  cin>>n;
  while(n--)
  {

    double a,b,c,d,e,f;
    cin>>a>>b>>c>>d>>e>>f;
    double A,B,C,D,E,F;
      A = a*a;
      B = b*b;
      C = c*c;
      D = d*d;
      E = e*e;
      F = f*f;
     double v = (sqrt(- A*B*D - A*C*E - B*C*F - D*E*F + A*C*D + B*C*D + A*B*E + B*C*E + B*D*E + C*D*E + A*B*F + A*C*F + A*D*F + C*D*F + A*E*F + B*E*F - C*C*D - C*D*D - B*B*E - B*E*E - A*A*F - A*F*F))/12.0;
      double s1 = (a + b + d)/2.0;
      double s2 = (a + c + e)/2.0;
      double s3 = (d + e + f)/2.0;
      double s4 = (b + c + f)/2.0;
      double a1 = sqrt(s1*(s1-a)*(s1-b)*(s1-d));
      double a2 = sqrt(s2*(s2-a)*(s2-c)*(s2-e));
      double a3 = sqrt(s3*(s3-d)*(s3-e)*(s3-f));
      double a4 = sqrt(s4*(s4-b)*(s4-c)*(s4-f));
      double s = a1 + a2 + a3 + a4;
      double r = (3.0*v)/s;
      cout<<setprecision(4)<<fixed<<r;
      cout<<endl;
  }
  return 0;
}
