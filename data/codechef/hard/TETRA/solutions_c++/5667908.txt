#include<iostream>
#include<cmath>
#include <iomanip>
using namespace std;

double area(double a,double b, double c)
{
	double ar=0;
	double s=(a+b+c)/2;
	ar=sqrt(s*(s-a)*(s-b)*(s-c));
	return ar;
}

int main()
{
	int t;
	cin>>t;
	while(t)
	{
		double u,v,w,U,V,W,x,y,z,X,Y,Z,a,b,c,d;
		cin>>w>>u>>v>>V>>U>>W;
		X=(w-U+v)*(w+U+v);
		x=(U-v+w)*(v-w+U);
		Y=(u-V+w)*(u+V+w);
		y=(V-w+u)*(w-u+V);
		Z=(v-W+u)*(W+u+v);
		z=(W-u+v)*(u-v+W);
		a=sqrt(x*Y*Z);
		b=sqrt(y*X*Z);
		c=sqrt(z*Y*X);
		d=sqrt(x*y*z);
		double volume=sqrt((-a+b+c+d)*(a-b+c+d)*(a+b-c+d)*(a+b+c-d))/(192*u*v*w);
		a=area(U,V,W);
		b=area(U,v,w);
		c=area(V,u,w);
		d=area(u,W,v);
		double radius= 3*volume;
		radius=radius/(a+b+c+d);
		std::cout << std::fixed;
  		std::cout << std::setprecision(4) << radius << '\n';
		t--;
	}
	return 0;
}
