//Sphere in a tetrahedron
#include<iostream>
#include<cmath>
#include<iomanip>
using namespace std;
int main()
{
    int t;
    cin>>t;
    float a,b,c,d,e,f;
    float A,B,C,D,E,F;
    for(int i=0;i<t;i++)
    {
            cin>>a>>d>>e>>b>>c>>f;
            A = a*a;
            B = b*b;
            C = c*c;
            D = d*d;
            E = e*e;
            F = f*f;
            float vol = (sqrt(- A*B*D - A*C*E - B*C*F - D*E*F + A*C*D + B*C*D + A*B*E + B*C*E + B*D*E + C*D*E + A*B*F + A*C*F + A*D*F + C*D*F + A*E*F + B*E*F - C*C*D - C*D*D - B*B*E - B*E*E - A*A*F - A*F*F))/12.;
            float s1 = (a + b + d)/2.;
            float s2 = (a + c + e)/2.;
            float s3 = (d + e + f)/2.;
            float s4 = (b + c + f)/2.;
            float a1 = sqrt(s1*(s1-a)*(s1-b)*(s1-d));
            float a2 = sqrt(s2*(s2-a)*(s2-c)*(s2-e));
            float a3 = sqrt(s3*(s3-d)*(s3-e)*(s3-f));
            float a4 = sqrt(s4*(s4-b)*(s4-c)*(s4-f));
            float surf_area = a1 + a2 + a3 + a4;
            float r = (3.*vol)/surf_area;
            //cout<<vol<<" "<<surf_area<<endl;
            cout<<setprecision(4)<<fixed<<r<<endl;
    }
    cin.ignore();
    cin.get();
    return 0;
}
