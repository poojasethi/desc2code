int main()
{
  long long t;
  long double u,v,w,U,V,W,x,y,z,X,Y,Z;
  double a,b,c,d,vol,area,s,ans;
  cin>>t;
  while(t--){
    cin>>u>>v>>w>>W>>V>>U;
    X=(w-U+v)*(U+v+w);
    x=(w+U-v)*(U+v-w);
    Y=(u-V+w)*(V+w+u);
    y=(u+V-w)*(V+w-u);
    Z=(v-W+u)*(W+u+v);
    z=(v+W-u)*(W+u-v);
    z=z>=0?z:-z;
    x=x>=0?x:-x;
    y=y>=0?y:-y;
    X=X>=0?X:-X;
    Y=Y>=0?Y:-Y;
    Z=Z>=0?Z:-Z;
//     cout<<x<<" "<<y<<" "<<z<<" "<<X<<" "<<Y<<" "<<Z<<endl;
    a=sqrt(x)*sqrt(Y)*sqrt(Z);
    b=sqrt(X*y*Z);
    c=sqrt(X*Y*z);
    d=sqrt(x*y*z);
//     cout<<a<<" "<<b<<" "<<c<<" "<<d<<endl;
    vol=sqrt((-a+b+c+d)*(a-b+c+d)*(a+b-c+d)*(a+b+c-d))/192;
//     cout<<vodl<<endl;
    vol/=u;
    vol/=v;
    vol/=w;
    area=0;
    s=U+V+W;s/=2.0;
    area+=sqrt(s*(s-U)*(s-V)*(s-W));
    s=U+v+w;s/=2.0;
    area+=sqrt(s*(s-U)*(s-v)*(s-w));
    s=u+V+w;s/=2.0;
    area+=sqrt(s*(s-u)*(s-V)*(s-w));
    s=u+v+W;s/=2.0;
    area+=sqrt(s*(s-u)*(s-v)*(s-W));
    ans=3*vol/area;
    printf("%0.4f\n",ans);
  }
  return 0;
}
