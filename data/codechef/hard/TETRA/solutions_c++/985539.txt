#include<stdio.h>
 
#include<math.h>
int main ()
{
int t,i;
float u,v,w,U,V,W, vol,s1,s2,s3,s4,a1,a2,a3,a4,area,radius[30];
scanf("%d", &t);
for( i=1; i<=t; i++)
{
scanf("%f ",&u);
scanf("%f", &v);
scanf("%f", &w);
scanf("%f", &W);
scanf("%f", &V);
scanf("%f", &U);
vol=sqrt(4*pow(u,2)*pow(v,2)*pow(w,2) - pow(u,2)*(pow((pow(v,2)+pow(w,2)-pow(U,2)),2)) - pow(v,2)*(pow((pow(w,2)+pow(u,2)-pow(V,2)),2)) - pow(w,2)*(pow((pow(u,2)+pow(v,2)-pow(W,2)),2)) + (pow(v,2)+pow(w,2)-pow(U,2))*(pow(w,2)+pow(u,2)-pow(V,2))*(pow(u,2)+pow(v,2)-pow(W,2)))/12 ;
s1=(u+v+W)/2;
s2=(U+V+W)/2;
s3=(U+v+w)/2;
s4=(u+V+w)/2;
a1=sqrt(s1*(s1-u)*(s1-v)*(s1-W));
a2=sqrt(s2*(s2-U)*(s2-V)*(s2-W));
a3=sqrt(s3*(s3-U)*(s3-v)*(s3-w));
a4=sqrt(s4*(s4-u)*(s4-V)*(s4-w));
area=a1+a2+a3+a4;
radius[i]=(3*vol)/(area);
}
for(i=1;i<=t;i++)
{
printf("%.4f \n",radius[i]);}
return 0;
}
