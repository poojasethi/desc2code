#include<iostream>
#include<stdio.h>
#include<math.h>
using namespace std;
double area(double a,double b,double c)
{
	double s = (a+b+c)/2.0;//semiperimeter
	
	double A = sqrt(s*(s-a)*(s-b)*(s-c));
	
	return A;
}
 
int main()
{
	
	int test;
	scanf("%d",&test);
	while(test--)
	{
		double u,v,w,W,V,U;
		scanf("%lf%lf%lf%lf%lf%lf",&u,&v,&w,&W,&V,&U);
		double a1 = area(u,v,W);
		double a2 = area(U,v,w);
		double a3 = area(V,u,w);
		double a4 = area(W,U,V);
		double temp=0;
		temp=a1+a2+a3+a4;
		double vol=sqrt(4*u*u*v*v*w*w- u*u*(v*v+w*w - U*U)*(v*v+w*w-U*U) - v*v*(w*w+u*u - V*V)*(w*w+u*u-V*V) - w*w*(u*u + v*v - W*W)*(u*u+v*v-W*W) + (v*v +w*w - U*U)*(w*w+u*u- V*V)*(u*u+v*v - W*W))/12;
		
		double ans=(vol*3.0)/temp;
		printf("%.4lf\n",ans);
		
	}
	return 0;
}
