def det(M):

	if(n==0) :
		return 0;
	elif(n==1):
		return M[0][0];

	sign=1
	delta=0
	for i in xrange(len(M)):
		if (M[0][i]!=0):
			delta+=sign*M[0][i]*det(eliminate(M,0,i));
		sign*=-1;

	return delta;
