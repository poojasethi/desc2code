#include<iostream>
#include<algorithm>
#include<stdio.h>
#define ll unsigned long long int
#define mod 747474747
using namespace std;

ll graph[6670][6670]={{0}};

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int i,j,k,l,m,n,d;
        ll in[6670][10],maxDistanceTo[6670]={0},visited[6670]={0},ans=1,include=1,index=0,disTo=0,temp;
        scanf("%d %d",&n,&d);
        for(i=1;i<=n;i++)
        {
            for(j=1;j<=d;j++)
                scanf("%llu",&in[i][j]);
        }
        for(i=1;i<=n;i++)
        {
            for(j=i+1;j<=n;j++)
            {
                temp=0;
                for(k=1;k<=d;k++)
                {
                    temp=(temp+((in[i][k]-in[j][k])*(in[i][k]-in[j][k])));
                }
                graph[i][j]=temp;
                graph[j][i]=temp;
            }
            graph[i][i]=0;
        }
        for(i=1;i<=n;i++)
        {
            maxDistanceTo[i]=graph[1][i];
        }
        visited[1]=1;
        index=1;
        include=1;
        while(include<n)
        {
            disTo=0;
            for(i=1;i<=n;i++)
            {
                if(!visited[i]&&(disTo<=maxDistanceTo[i]))
                {
                    disTo=maxDistanceTo[i];
                    index=i;
                }
            }
            visited[index]=1;
            include++;
            if(disTo!=0)
            ans=(ans*((disTo)%mod))%mod;
            for(i=1;i<=n;i++)
            {
                if(maxDistanceTo[i]<graph[index][i]&&!visited[i])
                {
                    maxDistanceTo[i]=graph[index][i];
                }
            }
        }
        printf("%llu\n",ans);
    }
    return 0;
}
