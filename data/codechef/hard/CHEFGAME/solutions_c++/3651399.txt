#include <cstdio>
#include <cstring>
#define mod 747474747

using namespace std;

long long dis[7000],max;
int t,n,i,j,x,y,d,v[7000][6],vaz[7000];
long long sol;

long long distanta(int a,int b)
{
	long long x=0;
	for(int i=0;i<d;i++) x+=((long long)v[a][i]-v[b][i])*(v[a][i]-v[b][i]);
	return x;
}

int main()
{
    //freopen("chefgame.in", "r", stdin);
    //freopen("chefgame.out", "w", stdout);
    scanf("%d",&t);
    for(;t;t--)
    {
    	sol=1;
    	scanf("%d%d",&n,&d);
    	for(i=1;i<=n;i++)
			for(j=0;j<d;j++) scanf("%d",&v[i][j]);
		memset(dis,-1,sizeof(dis));
		memset(vaz,0,sizeof(vaz));
		dis[1]=0;
		x=1;
		vaz[1]=1;
		for(i=1;i<n;i++)
		{
			max=1;
			y=1;
			for(j=1;j<=n;j++)
			if(!vaz[j])
			{
				if(distanta(x,j)>dis[j]) dis[j]=distanta(x,j);
				if(dis[j]>max) max=dis[j], y=j;
			}
			sol=(((long long)sol)*(max%mod))%mod;
			vaz[y]=1;
			x=y;
		}
		printf("%lld\n",sol);
    }
    return 0;
}
