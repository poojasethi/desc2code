LL dist(int i,int j)
{
    LL res=0;
    REP(k,d)
    {
        LL tmp=(a[i][k]-a[j][k]);
        res+=tmp*tmp;
    }
    return res;
}

    while(t--)
    {
        cin>>n>>d;
        REP(i,n) REP(j,d) cin>>a[i][j];

        vector<LL> df(n,0);
        vector<bool> it(n,0);

        long ans=1;
        it[0]=1;
        FOR(j,1,n) df[j]=dist(0,j);
        while(1)
        {
            int index=-1;mx=0;
            REP(j,n) if(!it[j]&&df[j]>mx){index=j;mx=df[j];}
            if(mx==0) break;
            ans=(ans*(mx%mod))%mod;
            it[index]=1;
            REP(j,n) if(!it[j]) df[j]=max(df[j],dist(index,j));
        }
        cout<<ans<<endl;
    }

}
