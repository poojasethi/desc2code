#include <iostream>
#include <cstdio>
#define MOD 747474747
using namespace std;

long data[7000][5];
int n,d;
long long distances(int i,int j){
    long long res=0;
    for(int k=0;k<d;k++)res=(res+((long long)data[i][k]-data[j][k])*(data[i][k]-data[j][k]));
return res;
}

int main()
{
    ios_base::sync_with_stdio(0);
    //freopen("in.txt","r",stdin);
    int t;cin>>t;
    long long ans,tmp;
    while(t--){
        ans=1;
        cin>>n>>d;
        bool visited[n];
        long long dist[n];
        for(int i=0;i<n;i++){
            for(int j=0;j<d;j++){
                cin>>data[i][j];
            }
            visited[i]=0;
        }
        visited[0]=1;dist[0]=0;
        for(int i=1;i<n;i++)dist[i]=distances(i,0);
        while(1){
            int z=0;tmp=0;
            for(int i=1;i<n;i++){if(!visited[i]&&dist[i]>tmp){z=i;tmp=dist[i];}}
            if(tmp==0)break;
            ans=(ans*(tmp%MOD))%MOD;visited[z]=1;
            for(int i=0;i<n;i++){if(!visited[i]&&distances(z,i)>dist[i]){dist[i]=distances(z,i);}}
        }
        cout<<ans<<endl;
    }
return 0;
}
