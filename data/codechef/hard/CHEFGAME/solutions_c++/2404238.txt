#define mod 747474747
int arr[6667][5];
ll dis[6667];
bool q[6667];
int n,d;
ll fun(int a,int b)
{

    ll ans=0;
    for(int i=0;i<d;i++)
    {
        ans+=(1LL*(arr[a][i]-arr[b][i])*(arr[a][i]-arr[b][i]));
    }
    return ans;
}
int main()
{
	//freopen("ip.in","r",stdin);
	//freopen("op.in","w",stdout);
	int t;s(t);
	while(t--)
	{

        s2(n,d);
        for(int i=1;i<=n;i++)
        {
            for(int j=0;j<d;j++)
            {
                s(arr[i][j]);
            }
        }
        memset(dis,-1,sizeof(dis));
        for(int i=0;i<=6666;i++)q[i]=1;
        int ans=1;
        dis[1]=1;
        for(int i=1;i<=n;i++)
        {
            ll m=-1;int in;
            for(int i=1;i<=n;i++)if(q[i]&&dis[i]>m){m=dis[i];in=i;}
            if(m<=0)break;
            //cout<<m<<" "<<in<<endl;
            ans=(1LL*ans*(m%mod))%mod;
            q[in]=0;
            for(int i=1;i<=n;i++)
            {
                if(q[i])
                {
                    dis[i]=max(dis[i],fun(in,i));
                }
            }
            //for(int i=1;i<=n;i++)cout<<dis[i]<<" ";
            //cout<<endl;

        }
        printf("%d\n",ans);




	}
}
