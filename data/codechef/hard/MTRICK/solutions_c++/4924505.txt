    ll multiply(ll a,ll b,ll mod)
    {
    	if(b==1)
    	return a%mod;
    	else if(b==0)
    	return 0;
	    else if(b%2==0)
	    return (2*multiply(a,b/2,mod))%mod;
	    else
	    return (a+multiply(a,b-1,mod))%mod;
    }
    ll pow(ll base,ll pow,ll mod)
    {
	    base%=mod;
	    ll res=1;
	    while(pow)
	    {
		    if(pow%2==1)
		    res=(res*base)%mod;
		    base=(base*base)%mod;
		    pow/=2;
	    }
	    return res;
    }
