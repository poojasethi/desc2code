#include<iostream>
using namespace std;
long long l[1001],k[1001],d[1001];
long long multiple(long long a, long long b, long long c) // a * b % c
{
    if (b == 0) {
        return 0;
    }
    long long ret = multiple(a, b >> 1, c);
    ret = (ret + ret) % c;
    if (b & 1) {
        ret = (ret + a) % c;
    }
    return ret;
}
int main()
{
    char str[1001];
    int t,n,i,begin,end;
    long long a,b,c;
    cin>>t;
    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>l[i];
            k[i]=1;
            d[i]=0;
        }
        k[n]=1;
        d[n]=0;
        cin>>a>>b>>c;
        cin>>str;
        begin=0;
        end=n-1;
        int direction=1;
        for(i=1;i<=n;i++)
        {
            if(str[i-1]=='R')
            {
                k[i]=k[i-1];
                d[i]=d[i-1];
                swap(begin,end);
                direction=(-1)*direction;
            }
            else if(str[i-1]=='A')
            {
                k[i]=k[i-1];
                d[i]=(d[i-1]+a)%c;
            }
            else
            {
                k[i]=multiple(k[i-1],b,c);
                d[i]=multiple(d[i-1],b,c);
            }
            //cout<<k[i]<<" "<<d[i]<<endl;
            long long ans=multiple(l[begin],k[i],c);
            ans=(ans+d[i])%c;
            cout<<ans<<" ";
            begin=begin+direction;
        }
        cout<<endl;
    }
    return 0;
}
