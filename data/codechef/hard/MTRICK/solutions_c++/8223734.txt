using namespace std;
 
typedef long long LL;
typedef unsigned long long ULL; 
 
ULL mulmod(ULL a, ULL b, ULL c) // a * b % c
{
    if (b == 0) 
	{
        return 0;
    }
    ULL ret = mulmod(a, b >> 1, c);
    ret = (ret + ret) % c;
    if (b & 1) {
        ret = (ret + a) % c;
    }
    return ret;
}
