long long int timesmod( long long int x, long long int y)
{
  x %= c;
  y %= c;
  long long int a = x < y ? x : y; // min
  long long int b = x < y ? y : x; // max
  long long int product = 0;
  for (; a != 0; a >>= 1, b = (b+b)%c )
    if (a&1) product = (product+b)%c;
  return product;
}
