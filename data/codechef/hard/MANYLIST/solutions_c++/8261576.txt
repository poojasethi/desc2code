#include <iostream>
#include <stdio.h>
#include <set>
#include <utility>

using namespace std;

const int N = 25e4 + 7;

set<pair<int,int> > s[N];
set<pair<int,int> >::iterator it;

int fwt[N];

int n,m,id,l,r,x;

void update(int from,int val)
{
	while(from<=N) 
	{
		fwt[from]+=val;
		from += from&-from;
	}
}
int query(int pos)
{
	int r=0;
	while(pos>=1)
	{
		r += fwt[pos];
		pos -= pos&-pos;
	}
	return r;
}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif

	cin>>n>>m;
	for(int i=0;i<m;i++)
	{
		scanf("%d",&id);
		if(id == 0)
		{
			scanf("%d%d%d",&l,&r,&x);
			it = s[x].lower_bound(make_pair(l+1,0));
			if(it != s[x].begin())
			{
				--it;
				if(it->second >= l) l = it->first;
			}
			while(1)
			{
				it = s[x].lower_bound(make_pair(l,0));
				if(it == s[x].end()) break;
				if(it->first <= r)
				{
					r = max(r,it->second);
					update(it->first,-1);
					update(it->second+1,1);
					s[x].erase(it);
				}
				else break;
			}
			s[x].insert(make_pair(l,r));
			update(l,1);
			update(r+1,-1);
		}
		else
		{
			scanf("%d",&x);
			printf("%d\n",query(x));
		}
		// for(int i=1;i<=n;i++) printf("%d ",query(i));puts("");
	}
	return 0;
}
