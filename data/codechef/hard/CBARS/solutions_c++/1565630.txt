#include <iostream>
#define MOD 1000000007LL
#define N (1<<a)
using namespace std;

long long a,b;
long long A[70][70];

long long check(long long s, long long t)
{
	for (long long i=0 ; i<a-1 ; i++) {
		long long b1, b2, b3, b4;
		b1 = (s >> i) & 1;
		b2 = (s >> i+1) & 1;
		b3 = (t >> i) & 1;
		b4 = (t >> i+1) & 1;
		if(b1 == b2 && b3 == b4 && b1 == b3) return 0;
	}
	return 1;
}

void init()
{
	for (long long i=0; i<N; i++)
		for(long long j=0; j<N; j++) {
			A[i][j] = check(i,j);
		}
}


void matmult(long long C[][70], long long A[][70], long long B[][70])
{
	long long tmp[70][70] = {0};
	for (long long i=0; i<N; i++)
		for (long long j=0; j<N; j++)
			for (long long k=0; k<N; k++)
				tmp[i][j] = (tmp[i][j] + A[i][k] * B[k][j] % MOD) % MOD;
	
	for (long long i=0; i<N; i++)
		for (long long j=0; j<N; j++)
			C[i][j] = tmp[i][j];
}


void matpow(long long b[][70], long long res[][70], long long e)
{
	long long tmp[70][70] = {0};
	for (long long i=0 ; i<N ; i++)
		for (long long j=0 ; j<N ; j++)
			res[i][j] = i == j, tmp[i][j] = b[i][j];


	while (e) {
		if (e & 1)
			matmult(res, res, tmp);
		matmult(tmp, tmp, tmp);
		e >>= 1;
	}
}

int main()
{
	cin >> a >> b;
	
	init();
	
	if (b == 1) {
		cout << (N) << endl;
		return 0;
	}
	
	long long ans[70][70];
	matpow(A, ans, b-1);

	long long res=0;
	for (long long i=0; i<N; i++)
		for (long long j=0; j<N; j++)
			res = (res + ans[i][j]) % MOD;
	
	cout << res << endl;
	return 0;
}
