#include <stdio.h>
#include <iostream>
#include <vector>
#define vvi vector <vector <long long int> >
#define ll long long
#define M 1000000007

using namespace std;

ll a;
vvi v;

vvi cr(int x,int y)
{
    vvi r;

    for(int i=0;i<x;i++)
    {
        vector <ll> v;

        for(int j=0;j<y;j++) v.push_back(0);

        r.push_back(v);
    }

    return r;
}

vvi mat_mul(vvi X,vvi Y)
{
    vvi x=X,y=Y;
    vvi z=cr(x.size(),y[0].size());;

    int i,j,k;

    for(k=0;k<x[0].size();k++)
    {
        for(i=0;i<x.size();i++)
        {
            for(j=0;j<y[0].size();j++) z[i][j]=(z[i][j]+x[i][k]*y[k][j])%M;
        }
    }

    return z;
}

vvi power(vvi x,ll n)
{
    vvi z;
    int i,j,k;

    if(n==0)
    {

        z=cr(x.size(),x[0].size());

        for(i=0;i<x.size();i++) z[i][i]=1;

        return z;
    }

    if(n%2)
    {
        return mat_mul(x,power(x,n-1));
    }

    return power(mat_mul(x,x),n/2);
}

bool ok(ll x,ll y)
{
    bool a1,a2,a3,a4;

    for(int i=0;i<a-1;i++)
    {
        a1=( (x & (1<<i) ) >0);
        a2=( (x & (1<<(i+1)) ) >0);
        a3=( (y & (1<<i) ) >0);
        a4=( (y & (1<<(i+1)) ) >0);


        if(a1==a2 && a2==a3 && a3==a4) return 0;

    }

    return 1;

}

int main()
{
    ll i,j,k,b,ans;
    vvi ret;

    while(cin>>a>>b)
    {
        v.clear();
        ret.clear();
        ans=0;

        v=cr(1<<a,1<<a);

        for(i=0;i<(1<<a);i++)
        {
            for(j=0;j<=i;j++) v[i][j]=v[j][i]=ok(i,j);
        }

        ret=power(v,b-1);

        for(i=0;i<(1<<a);i++)
        {
            for(j=0;j<(1<<a);j++) ans=(ans+ret[i][j])%M;
        }

        cout<<ans<<endl;
    }

    return 0;




}
