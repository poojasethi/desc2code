const int mod = (int)1e9 + 7;
const int maxn = 1 << 6;
int a;
LL b;
int m[maxn][maxn],res[maxn][maxn],tmp[maxn][maxn];
void mul(int A[maxn][maxn], int B[maxn][maxn], int C[maxn][maxn]) {
	for (int i = 0; i < (1 << a); i++) {
		for (int j = 0; j < (1 << a); j++) {
			tmp[i][j] = 0;
			for (int k = 0; k < (1 << a); k++) {
				tmp[i][j] += A[i][k] * (LL)B[k][j] % mod;
				tmp[i][j] %= mod;
			}
		}
	}
	for (int i = 0; i < (1 << a); i++) {
		for (int j = 0; j < (1 << a); j++) {
			C[i][j] = tmp[i][j];
		}
	}
}
int main()
{
	#ifndef ONLINE_JUDGE

	cin >> a >> b;
	for (int mask = 0; mask < (1 << a); mask++) {
		for (int sub = 0; sub < (1 << a); sub++) {
			bool ok = true;
			for (int i = 0; i < a - 1; i++) {
				int sum = ((mask >> i) & 1) + ((mask >> (i + 1)) & 1) + ((sub >> i) & 1) + ((sub >> (i + 1)) & 1);
				if (sum == 0 || sum == 4) {
					ok = false;
					break;
				}
			}
			if (ok) {
				m[mask][sub]++;
			}
		}
	}
	b--;
	for (int i = 0; i < (1 << a); i++) {
		res[i][i] = 1;
	}
	while(b > 0) {
		if (b & 1) {
			mul(res,m,res);
		}
		mul(m,m,m);
		b >>= 1;
	}
	int ans = 0;
	for (int i = 0; i < (1 << a); i++) {
		for (int j = 0; j < (1 << a); j++) {
			ans += res[i][j];
			ans %= mod;
		}
	}
	cout << ans << endl;
	return 0;
}
