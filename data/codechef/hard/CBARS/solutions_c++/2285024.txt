ill f(ill i,ill j,ill a)
{
    ill x=i&j,y=i|j,f=-1;ill k;
    k=a;
    while(k--)
    {
        if(f==1 &&(x%2)==1) return 0;
        f=x%2;
        x/=2;
    }
    f=-1;
    k=a;
    while(k--)
    {
        if(f==0 && (y%2)==0) return 0;
        f=y%2;
        y/=2;
    }
    return 1;
}
vvi mul(vvi a,vvi b)
{
    vvi ret;
    ill i,j,k;
    ret.resize(a.size());
    for(i=0;i<a.size();i++)
    {
        ret[i].resize(a.size(),0);
        for(j=0;j<a.size();j++)
        {
            for(k=0;k<a.size();k++)
            {
                ret[i][j]+=(a[i][k]*b[k][j])%imax;
                ret[i][j]%=imax;
            }
        }
    }
    return ret;
}
ill exp(vvi a,ill n)
{
    vvi ret;ill i,j,ans;
    ret.resize(a.size());
    for(i=0;i<a.size();i++) ret[i].resize(a.size(),0),ret[i][i]=1;
    while(n)
    {
        if(n%2) ret=mul(ret,a);
        a=mul(a,a);
        n/=2;
    }
    ans=0;
    for(i=0;i<a.size();i++)
    {
        for(j=0;j<a.size();j++)
        {
           ans+=ret[i][j];
           ans%=imax;
           //cout<<ret[i][j]<<" ";
        }
        //cout<<"\n";
    }
    return ans;
}
