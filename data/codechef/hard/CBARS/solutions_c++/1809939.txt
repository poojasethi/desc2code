#include<cstdio>
#include<cstring>
const long long oo=1000000007;
long long H,Ans;
int i,j,K,W;
long long A[64],B[64][64],a[64],b[64][64];
void Mul1()
{
	int i,j;
	memset(a,0,sizeof(a));
	for(i=0;i<K;++i) if(A[i])
		for(j=0;j<K;++j) if(B[i][j])
			a[j]=(a[j]+A[i]*B[i][j])%oo;
	memcpy(A,a,sizeof(A));
}
void Mul2()
{
	int i,j,k;
	memset(b,0,sizeof(b));
	for(i=0;i<K;++i)
		for(j=0;j<K;++j) if(B[i][j])
			for(k=0;k<K;++k) if(B[j][k])
				b[i][k]=(b[i][k]+B[i][j]*B[j][k])%oo;
	memcpy(B,b,sizeof(B));
}
void Mul(long long N)
{
	for(;N;N/=2)
	{
		if(N&1) Mul1();
		Mul2();
	}
}
bool Check(int x,int y)
{
	int i,t1,t2,t3,t4;
	for(i=0;i<W-1;++i)
	{
		t1=(x&(1<<i))>0;
		t2=(x&(1<<i+1))>0;
		t3=(y&(1<<i))>0;
		t4=(y&(1<<i+1))>0;
		if(t1==t2&&t1==t3&&t1==t4) return 0;
	}
	return 1;
}
int main()
{
	scanf("%d%lld",&W,&H);
	K=1<<W;
	for(i=0;i<K;++i)
		for(j=0;j<K;++j)
			if(Check(i,j))
				B[i][j]=1;
	for(i=0;i<K;++i) A[i]=1;
	Mul(H-1);
	for(i=0;i<K;++i) Ans=(Ans+A[i])%oo;
	printf("%lld\n",Ans);
	return 0;
}
