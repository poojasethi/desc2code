#include<cstdio>
#include<iostream>
#include<cmath>
using namespace std;
typedef long long int ll;
ll a[65][65],b[65][65],c[65][65];
int ar[65][10];
ll mod,d;
int f;
void fill(int f)
{
    int i,j;
    int x = pow(2,f);
    for(i = 0 ; i<x ; i++)
    {
        for(j = 0 ; j<x ; j++)
        {
            int ans = 1;
            for(int k = 0 ; k<f -1 ; k++)
            {
                if(ar[i][k] == ar[i][k + 1] && ar[j][k] == ar[j][k + 1] && ar[i][k] == ar[j][k])
                {
                    ans = 0;
                    break;
                }
            }
            a[i+1][j+1] = ans;
        }
    }
}
void mulaa(int x)
{
    for(int i = 1 ; i<= x ; i++)
    for(int j = 1 ; j<= x ; j++)
    c[i][j] = 0;

    for(int i = 1 ; i<=x ; i++)
    for(int j = 1 ; j<=x ; j++)
    for(int k = 1 ; k<=x ; k++)
    c[i][j] = (c[i][j] + (a[i][k]*a[k][j])%mod)%mod;

    for(int i = 1 ; i<= x ; i++)
    for(int j = 1 ; j<= x ; j++)
    a[i][j] = c[i][j];
}
void mulab(int x)
{
    for(int i = 1 ; i<= x ; i++)
    for(int j = 1 ; j<= x ; j++)
    c[i][j] = 0;

    for(int i = 1 ; i<=x ; i++)
    for(int j = 1 ; j<=x ; j++)
    for(int k = 1 ; k<=x ; k++)
    c[i][j] = (c[i][j] + (b[i][k]*a[k][j])%mod)%mod;

    for(int i = 1 ; i<= x ; i++)
    for(int j = 1 ; j<= x ; j++)
    b[i][j] = c[i][j];
}
ll getfunc()
{
    int x = pow(2,f);
    ll ans = 0;

    for(int i = 1 ; i<=x ; i++)
    for(int j = 1 ; j<=x ; j++)
    b[i][j] = 0;

    for(int i = 1 ; i<=x ; i++)
    b[i][i] = 1;

    for(int i = 0 ; i<x ; i++)
    {
        int temp = i;
        for(int k = 0 ; k<6 ; k++)
        {
            ar[i][k] = temp%2;
            temp = temp/2;
        }
    }

    fill(f);
    /*for(int i = 1 ;i<=x ; i++)
    {
        for(int j = 1 ; j<=x ; j++)
        printf("%d ",a[i][j]);

        printf("\n");
    }*/
    while(d > 0)
    {
        if(d&1) mulab(x);
        d = d/2;
        mulaa(x);
    }
    ans = 0;
    for(int i = 1 ; i<=x ; i++)
    {
        for(int j = 1 ; j<=x ; j++)
        {
            ans = (ans + b[i][j])%mod;
        }
    }
    return ans;
}
int main()
{
    mod = 1000000007;
    scanf("%d %lld",&f,&d);
    //printf("%d %lld\n",f,d);
    d--;
    ll ans = getfunc();
    printf("%lld\n",ans);
    return 0;
}
