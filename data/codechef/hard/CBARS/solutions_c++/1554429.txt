#include <cstdio>
#include <string>
#include <algorithm>
#define md 1000000007
using namespace std;
FILE *f,*g;
long long b;
int a,lm,i,j,lm2,lm3;
typedef int vector[100][100];
vector nr,p,ax;
int sol;

inline int k(int x,int y,int pz)
{
    int c1,c2,c3,c4;
    c1=( (x&(1<<pz))!=0)?1:0;
    c2=( (x&(1<<(pz+1)))!=0)?1:0;
    c3=((y&(1<<pz))!=0)?1:0;
    c4=((y&(1<<(pz+1)))!=0)?1:0;
    if (c1+c2+c3+c4==0 || c1+c2+c3+c4==4) return 0;
    return 1;
}

void inmult(vector a, vector b)
{
    int i,j,k;

    for (i=0;i<lm2;i++)
        for (j=0;j<lm2;j++)
            for (k=0,ax[i][j]=0;k<lm2;k++)
                ax[i][j]=(int)(((long long)ax[i][j]+((long long)a[i][k]*b[k][j])%md)%md);
    for (i=0;i<lm2;i++)
        for (j=0;j<lm2;j++)
            a[i][j]=ax[i][j];

}

void pre()
{
    int i,j,q;
    bool ok,ok2;
    lm=(1<<(a));
    lm3=lm-1;
    lm2=lm/2;


    for (i=0;i<lm;i++)
        nr[i][i]=1;

    for (i=0;i<lm2;i++)
        for (j=0;j<lm2;j++)
        {
            for (q=0,ok=true,ok2=true;q<a-1;q++)
            {
                if (k(i,j,q)==0) ok=false;
                if (k(i,lm3-j,q)==0) ok2=false;
            }
            if (ok)
                p[i][j]=1;
            else
                p[i][j]=0;
            if (ok2)
                p[i][j]++;
        }





}

void ridic(long long b)
{
    long long i;
    int j;

    for (;b;inmult(p,p),b>>=(long long)1)
        if ((long long)b&1)
            inmult(nr,p);
}

int main()
{
	scanf("%d%lld",&a,&b);




    pre();
    if (b==1)
    {
        printf("%d",lm);
        return 0;
    }

    ridic(b-1);

    for (sol=0,i=0;i<lm2;i++)
        for (j=0;j<lm2;j++)
            sol=int(((long long)sol+((long long)2*nr[i][j])%md)%md);
    printf("%d",sol);

	return 0;
}
