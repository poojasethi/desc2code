double A[100], D[100];

int main() {

  /*

  A[0] = 1; D[0] = 1;
  double v = 1;
  for (int k = 1; k < 100; k++) {
    v *= -1. / k;
    D[k] = D[k - 1] + v;
    A[k] = A[k - 1] * k;
  }

  cout.precision(5); cout << fixed;
  int T;  cin >> T;
  while (T--) {
    int n, r; cin >> n >> r;
    double ans = 0;
    for (int c = 2; c <= n; c++)
      if (r % c == 0)
        ans += A[n - 1] * D[n - c];
    ans /= D[n] * A[n];
    cout << n * ans << endl;
  }
  //cerr << ((double)clock() / CLOCKS_PER_SEC) << endl;
  

  return 0;
}
