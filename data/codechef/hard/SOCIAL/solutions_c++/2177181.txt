#include<iostream>
#include<cstdio>
using namespace std;

int main()
{
    int T; // Number of test cases
    int N; // Number of chairs
    int R; // Number of ringings
    double factorialArray[51], derangeArray[51]; // Helper arrays
    double result; // Result :P
    int i, j, k; // Loop variables
    factorialArray[0] = 1;
    factorialArray[1] = 1;
    derangeArray[0] = 1;
    derangeArray[1] = 0;
    for (i = 2; i < 51; i++)
    {
        factorialArray[i] = factorialArray[i - 1] * i;
        derangeArray[i] = (i - 1) * (derangeArray[i - 1] + derangeArray[i - 2]);
    }
    cin >> T;
    for (j = 0; j < T; j++)
    {
        cin >> N >> R;
        result = 0;
        for (k = 2; k <= N; k++)
        {
            if (R % k == 0)
            {
                result += factorialArray[N - 1] / factorialArray[N - k] * derangeArray[N - k];
            }
        }
        result /= derangeArray[N];
        result *= N;
        printf ("%.5f\n", result);
    }
    return 0;
}
