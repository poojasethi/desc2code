#include<stdio.h>
#define s1(x) scanf("%d",&x)
#define s2(x,y) scanf("%d%d",&x,&y)
int main()
{
	long double fact[51],derange[51];
	fact[0]=fact[1]=1;
	derange[0]=1;
	derange[1]=0;
	for(int i=2;i<=50;i++)
	{
		fact[i]=i*fact[i-1];
		derange[i]=(i-1)*(derange[i-1]+derange[i-2]);
	}
	int t,n,r;
	long double ans;
	s1(t);
	while(t--)
	{
		ans=0;
		s2(n,r);
		for(int i=2;i<=n;i++)
		{
			if(r%i==0)
			{
				ans+=fact[n-1]/fact[n-i]*derange[n-i];
			
			}
		
		}
		ans/=derange[n];
		ans*=n;
		printf("%.5Lf\n",ans);
	}
	
}
