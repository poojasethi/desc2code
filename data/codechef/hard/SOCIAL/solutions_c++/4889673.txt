#include <cstring>
#include<iostream>
 #include <cmath>
#include<iomanip>

 #define inf 1000000000

 using namespace std;

 typedef long long L;

 typedef unsigned long long UL;

 double P[51],F[51];

 int t,n,r;

  

 int main()

 {

     F[0]=1;F[1]=1;F[2]=2;

     P[0]=1;P[1]=0;P[2]=1;

     for(int i=3;i<=50;i++){

         F[i]=i*(F[i-1]);

         P[i]=(i-1)*(P[i-1]+P[i-2]);

     }

  

     cin>>t;

     while(t--){

         cin>>n>>r;

         double res=0.0;

         for(int i=2;i<=n;i++)

             if(!(r%i))

                 res+=(F[n-1]/F[n-i]*P[n-i])/P[n];

         res*=(1.0*n);

         cout.setf(ios::fixed);

         cout.precision(5);

         cout.setf(ios::showpoint);

         cout<<res<<endl;

     }

  

     return 0;

 }
