#include <bits/stdc++.h>

using namespace std;
int arr[100009];
int main()
{
   int n;
   scanf("%d",&n);
   int i;
   for(i=0;i<n;i++)
    scanf("%d",&arr[i]);
   vector<int> d;
   vector<int>::iterator it;
   for(i=n-1;i>=0;i--){
    it=lower_bound(d.begin(),d.end(),arr[i]);
    if(it==d.end())
        d.push_back(arr[i]);
    else *it=arr[i];
   }
   printf("%d\n",d.size());
    return 0;
}
