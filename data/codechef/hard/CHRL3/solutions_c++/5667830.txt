#include <bits/stdc++.h>
using namespace std;

#define N 112345
#define sd(x) scanf("%d", &x)

multiset<int> st;

int main(){
    //freopen("input.txt", "r", stdin);
    int n, i, x;
    multiset<int>::iterator it;
    sd(n);
    for(i = 0; i < n; i++){
        sd(x);
        if((int)(st.size()) == 0){
            st.insert(x);
        }
        else if(x > *st.rbegin()){
            st.erase(--st.end());
            st.insert(x);
        }
        else if(x < *st.begin()){
            st.insert(x);
        }
        else{
            it = st.upper_bound(x);
            st.erase(--it);
            st.insert(x);
        }
    }
    printf("%d\n", (int)(st.size()));
    return 0;
}
