#include <cstdio>
#include <set>
using namespace std;
int a[100005];
int main()
{
	int n;
	int x;
	scanf("%d",&n);
	set<int> A;
	for (int i=0;i<n;i++)
	{
		scanf("%d",&x);
		if (A.size() == 0)
			A.insert(x);
		else
		{
			if (x < *A.begin())
			{
				A.insert(x);
			}
			else if (x > *A.rbegin())
			{
				A.erase(--A.end());
				A.insert(x);
			}
			else if (A.find(x) == A.end())
			{
				set<int>::iterator it = A.upper_bound(x);
				it--;
				A.erase(it);
				A.insert(x);
			}
		}
	}
	printf("%d\n",A.size());
	return 0;
}
