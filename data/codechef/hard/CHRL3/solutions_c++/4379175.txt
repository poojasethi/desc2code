#include <cstdio>
#include <algorithm>

using namespace std;

int last_element[100000],top=-1;
 
int bin_search(int x,int l,int r)
{
	int mid = (l+r)/2,temp1,temp2;
	if(l > r)
		return -1;
	if(x >= last_element[mid])
	{
		temp1 = mid;
		temp2 = bin_search(x,l,mid-1);
		if(temp2 == -1)
			return temp1;
		return (max(last_element[temp1],last_element[temp2])==last_element[temp1])?temp1:temp2;
	}	
	if(x < last_element[mid])
	{
		return bin_search(x,mid+1,r);
	}
}

int main()
{
	int n,a[100000],temp;
	scanf("%d",&n);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);

	for(int i=0;i<n;i++)
	{
			temp = bin_search(a[i],0,top);
			if(temp == -1)
			{
				last_element[++top] = a[i];
			}
			else
			{
				last_element[temp] = a[i];
			}
	}	
	printf("%d\n",top+1);
	return 0;
}
