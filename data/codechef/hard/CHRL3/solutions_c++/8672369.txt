int longest(int a[],int n){

  int dp[n],len=1,i;dp[0]=a[0];
  for(i=1;i<n;i++){
    if(a[i]<dp[0])dp[0]=a[i];
    else if(dp[len-1]<a[i])dp[len++]=a[i];
    else{
      int idx=lower_bound(dp,dp+len,a[i])-dp;
      dp[idx]=a[i];
    }}
    return len;
  }
