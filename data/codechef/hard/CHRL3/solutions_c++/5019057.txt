#include<iostream>
#include<algorithm>
using namespace std;
int lis(int *a,int n){
    int count=1;
    int *l=new int [n];
    l[0]=a[0];
    for(int i=1;i<n;i++){
        if(a[i]>l[count-1]){
            l[count]=a[i];
            count++;
            //cout<<a[i]<<endl;
        }
        else{
           // (*(upper_bound(l,l+count,a[i])-1))=a[i];
           int j=lower_bound(l,l+count,a[i])-l;
           //cout<<a[i]<<" "<<l[j]<<endl;
           l[j]=a[i];
        }
    }
    //for(int i=0;i<count;i++)cout<<l[i]<<" ";
   // cout<<endl;
    return count;
}
int main(){
   int n;
   cin>>n;
   int *a=new int [n];
   for(int i=0;i<n;i++){
        cin>>a[i];
   }
   reverse(a,a+n);
   cout<<lis(a,n)<<endl;
}
