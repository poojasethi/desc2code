#include<stdio.h>
#include<iostream>
#include<vector>
using namespace std;
 
int main(){
int n;
cin>>n;
vector <int> vals;
vals.push_back(0);
for (int i = 1 ; i <= n; i ++ )
    {
        int x;
        scanf("%d",&x);
        if (x < vals[vals.size()-1]){vals.push_back(x);continue;}
        //bool ya = 1 ;
        for (int w = 0 ; w < vals.size(); w ++)
        {
            if (x >= vals[w])
            {
                vals[w]=x;
              //  ya =0;
                break;
            }
        }
      //  if (ya)vals.push_back(x);
    }
    int ap = vals.size();
    printf("%d \n",ap);
    return 0;
}
