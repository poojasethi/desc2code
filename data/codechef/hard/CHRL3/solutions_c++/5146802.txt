#include<cstdio>
#include<iostream>
#include<set>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	int a;
	set< int > v;
	for(int i=0;i<n;i++)
	{
		scanf("%d",&a);
		if(v.size()==0) v.insert(a);
		else
		{
			if(a<*v.begin())
				v.insert(a);
			else if(a>*v.rbegin())
			{
				v.erase(--v.end());
				v.insert(a);
			}
			else if(v.find(a)==v.end())
			{
				set<int>::iterator it=v.upper_bound(a);
				it--;
				v.erase(it);
				v.insert(a);
			}
		}		
	}
	printf("%d\n",(int)v.size());
	return 0;
}
