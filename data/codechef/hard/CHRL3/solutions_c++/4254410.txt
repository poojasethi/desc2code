#include <iostream>
#include<stdio.h>
using namespace std;

int tailtable[100005];
int ceilindex(int st,int end,int key)
{
	while (end-st>1)
	{
		int mid=st+(end-st)/2;
		if(tailtable[mid]>=key)
			end=mid;
		else 
			st=mid;
		
	}
	return end;
}
int LongestIncreasingSubsequenceLength(int arr[],int n)
{
	tailtable[0]=arr[0];
	int len=1;
	for(int i=1;i<n;i++)
	{
		if(arr[i]<tailtable[0])
			tailtable[0]=arr[i];
		else if(arr[i]>tailtable[len-1])
			tailtable[len++]=arr[i];
		else 
		{
			int k=ceilindex(0,len-1,arr[i]);
			tailtable[k]=arr[i];
		}
	}
	return len;
}
int main() {
    int n,arr[100005];
	scanf("%d",&n);
	for(int i=n-1;i>=0;i--)
		scanf("%d",&arr[i]);
    printf("%d\n",LongestIncreasingSubsequenceLength(arr, n));
	
    return 0;
}
