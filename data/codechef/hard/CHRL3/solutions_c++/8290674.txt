#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ull unsigned long long
#define pb push_back
#define ft first
#define se second
#define mp make_pair
int main(int argc, char const *argv[]){
    int n;
    scanf("%d", &n);
    set<int> v;
    int x;
    scanf("%d", &x);
    v.insert(-x);
    for(int i = 1; i < n; ++i){
    	scanf("%d", &x);
    	set<int> :: iterator it;
    	it=v.upper_bound(-(x+1));
    	if(it==v.end()) v.insert(-x);
    	else{
    		v.erase(it);
    		v.insert(-x);
    	}
    }
    printf("%d\n", (int)v.size());
	return 0;
}
