#include <iostream>
#include <vector>
#include <algorithm>
 
using namespace std;
vector <int> v;
vector <int> c;
int n,t;
int bin(int y);
int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	cin >> n;
	c.resize(n+1);
	for(int i = 0; i < n; i++){
		cin >> c[i];
	}
	v.push_back(c[0]);
	for(int i = 1; i < n; ++i){
		t = bin(c[i]);
		if(t == v.size())
			v.push_back(c[i]);
		else v[t] = c[i];
	}
	cout << v.size() << "\n";
	return 0;
}
 
int bin(int y){
    int hi,lo,mid;
 
    lo = 0;
    hi = v.size()-1;
 
    while(lo < hi){
        mid = (hi+lo)/2;
        if(v[mid] > y){
            lo = mid+1;
        }else{
            hi = mid;
        }
    }
 
    if(v[lo] > y) return lo + 1;
    return lo;
}
