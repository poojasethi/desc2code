#include<iostream>
#include<algorithm>
#include<set>
using namespace std;

set<int> st;
set<int> ::iterator it;

int main()
{
    int i,j,k,l,m,n,a[100100];
    cin>>n;
    for(i=1;i<=n;i++)
        cin>>a[i];
    int flag=0;
    st.insert(a[1]);
    for(i=2;i<=n;i++)
    {
        if(a[i]<*st.begin())
        {
            st.insert(a[i]);
        }
        else if(a[i]>*st.rbegin())
        {
            st.erase(--st.end());
            st.insert(a[i]);
        }
        else
        {
            it=st.upper_bound(a[i]);
            it--;
            st.erase(it);
            st.insert(a[i]);
        }
    }
    cout<<st.size();
    return 0;
}
