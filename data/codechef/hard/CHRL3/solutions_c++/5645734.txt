vector<int>v;
int arr[100005], dp[100005];
int main()
{
  int t;
  //si(t);
  t=1;
  while(t--)
  {
    int n;
    si(n);
    int ma=1;
    for(int i=0;i<n;i++)
    {
      si(arr[i]);
    }
    reverse(arr,arr+n);
    v.pb(arr[0]);
    dp[0]=1;
    for(int i=1;i<n;i++)
    {
      if(arr[i] > v[v.size()-1] )
      {
        v.pb(arr[i]);
        dp[i]=v.size();
        ma=max(ma,dp[i]);
      }
      else
      {
        int p = lower_bound(v.begin(),v.end(),arr[i])-v.begin();
        v[p]=arr[i];
        dp[i] = p+1;
        ma=max(ma,dp[i]);
      }
    }
    pi(ma);nl;
    v.clear();
  }
}
