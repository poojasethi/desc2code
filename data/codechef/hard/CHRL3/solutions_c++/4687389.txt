#include<cstdio>
#include<cstring>

const int mx = 100005;
int seg[4 * mx], cnt[mx];
int n, a[mx];

void put(int i, int a, int b, int q, int v) {
    if(a == b) {
        cnt[q] += v;
        if(cnt[q] == 0) seg[i] = -1;
        else seg[i] = q;
        return;
    }

    int m = (a + b) / 2, l = 2 * i + 1, r = 2 * i + 2;
    if(q <= m) put(l, a, m, q, v);
    else put(r, m + 1, b, q, v);

    if(seg[r] == -1) seg[i] = seg[l];
    else seg[i] = seg[r];
}

int get(int i, int a, int b, int q) {
    if(a > q) return -1;
    if(q >= b) return seg[i];

    int m = (a + b) / 2, l = 2 * i + 1, r = 2 * i + 2;
    int al, ar;
    al = ar = -1;

    al = get(l, a, m, q);
    ar = get(r, m + 1, b, q);

    if(ar == -1) return al;
    else return ar;
}

int main() {
    scanf("%d", &n);
    for(int i = 0; i < n; ++i) scanf("%d", a + i);

    memset(seg, -1, sizeof(seg));

    int ans = 0;
    for(int i = 0; i < n; ++i) {
        int x = get(0, 0, mx - 1, a[i]);
        if(x == -1) ++ans;
        else put(0, 0, mx - 1, x, -1);
        put(0, 0, mx - 1, a[i], 1);
    }

    printf("%d\n", ans);
}
