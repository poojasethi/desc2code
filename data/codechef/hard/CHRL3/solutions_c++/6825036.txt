#include<bits/stdc++.h>
using namespace std;
inline int scan(){
    char c = getchar_unlocked();
    int x = 0;
    while(c<'0'||c>'9'){
        c=getchar_unlocked();
    }
    while(c>='0'&&c<='9'){
        x=(x<<1)+(x<<3)+c-'0';
        c=getchar_unlocked();
    }
    return x;
}
vector<int> bit(100001,0);
stack<int> arr;
int n;
int query(int idx){
    int res=0;
    while(idx){
        res=max(res,bit[idx]);
        idx&=idx-1;
    }
    return res;
}
void update(int idx,int val){
    while(idx<=100000){
        bit[idx]=max(bit[idx],val);
        idx+=idx&-idx;
    }
}
int main(){
    n=scan();
    for(int i=0;i<n;++i){
        arr.push(scan());
    }
    while(!arr.empty()){
        update(arr.top(),query(arr.top()-1)+1);
        arr.pop();
    }
    printf("%d\n",query(100000));
    bit.clear();
}
