#include<bits/stdc++.h>

using namespace std;

#define sd(x) scanf("%d",&x);
#define slld(x) scanf("%I64d",&x);
#define LL long long
#define LD long double
#define PB push_back
#define MP make_pair
#define F first
#define S second
#define Fill(a, b) memset(a, b, sizeof(a))
#define INF 2000000009

typedef pair<int,int> PII;
typedef vector<int> VI;

#define N 20
#define PN 4000000


int adj[N][N],dp[PN];
LL ans[PN];

void solve()
{
	int n,m,a,b;
	sd(n);sd(m);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			adj[i][j]=0;
		}
	}
	for(int i=0;i<m;i++)
	{
		sd(a);sd(b);
		a--;b--;
		adj[a][b]=1;
		adj[b][a]=1;
	}
	dp[0]=0;
	for(int i=0;i<(1<<n);i++)
	{
		for(int j=0;j<n;j++)
		{
			//cout<<"j "<<i<<" "<<j<<endl;
			if((i&(1<<j))==0)
			{
				dp[i|(1<<j)]=dp[i];
				for(int k=0;k<n;k++)
				{
					//cout<<j<<" "<<k<<" "<<adj[j][k]<<endl;
					if(adj[j][k]==1)
					{
						if(i&(1<<k))dp[i|(1<<j)]--;
						else dp[i|(1<<j)]++;
						//cout<<(i|(1<<j))<<" "<<dp[i|(1<<j)]<<" "<<k<<" "<<j<<" "<<adj[j][k]<<endl;
					}
				}
			}
		}
	}
	for(int i=0;i<(1<<n);i++)
	{
		ans[i]=INF;
	}
	ans[0]=0;
	for(int i=0;i<(1<<n);i++)
	{
		for(int j=0;j<n;j++)
		{
			ans[i|(1<<j)] = min(ans[i|(1<<j)],ans[i]+dp[i]);
		}
	}
	printf("%lld\n",ans[(1<<n)-1]);
}

int main()
{
	int t=1;
	//scanf("%d",&t);
	while(t--)
	{
		solve();
	}
}
