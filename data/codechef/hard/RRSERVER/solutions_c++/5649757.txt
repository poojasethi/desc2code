#define pb push_back
#define f  first

using namespace std;

typedef pair<int,int> pr;
typedef long long ll;
int dp[2000000];
int dp2[2000000];
vector<pr> arr;
int dis[25][25];
int deg[22];

void func(int a,int m,int n)
{
	int i,j,x1,x2,k,z;

	dp2[0] = 0;
	for(i = 1;i < a;i++)
	{
		for(j = 0;j < n;j++)
		{
		 	if(i&(1<<j))
			{
				z = (i^(1<<j));
				break;
			} 
		}
		dp2[i] = dp2[z] + deg[j];
	//	cout<<dp2[i]<<endl;
		for(k = j+1;k < n;k++)
		{
		
			if((i&(1<<k)))
			{
				
				dp2[i] = dp2[i] - (2*dis[j][k]);
				//cout<<dis[j][k]<<"it is dis[j][k] "<<j<<" "<<k<<endl;
			}
		}
	}
	/*for(i = 1;i < a;i++)
	{
		cout<<dp2[i]<<"  ";
	}*/
	//cout<<endl;

}

int main()
{
	int n,m,i,a,j,z,x,y;
	
	
	scanf("%d %d",&n,&m);
		
	for(i = 0;i < m;i++)
	{
		scanf("%d %d",&x,&y);
		x--;
		y--;
		/* sir zero based indexing for ease */
		if(x == y)
			continue;
		deg[x]++;
		deg[y]++;
		dis[x][y] = 1;
		dis[y][x] = 1;
		arr.pb(pr(x,y));	
	}
	
	a = 1;
	for(i = 1;i <= n;i++)
	{
		a = a*2;
	}
	
	func(a,m,n);
	for(i = 0;i <= a;i++)
	{	
		dp[i] = 10000000;
	}
	dp[0] = 0;
	for(i = 1;i < a;i++)
	{
		for(j = 0;j < n;j++)
		{
			if(i&(1<<j))
			{
				z = i^(1<<j);
				dp[i] = min(dp[i],dp[z]+dp2[i]);			
			}
		}
	}
	printf("%d\n",dp[a-1]);
	return 0;
}
