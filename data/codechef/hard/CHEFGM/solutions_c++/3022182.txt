long double val(const vector<int> &pile) {
	if(len(pile) == 0) {
		return 0;
	}

	vector<bool> even;
	for(int i = 0; i < len(pile); i++) {
		even.push_back(pile[i] % 2 == 0);
	}

	bool neg = false;
	if(!even[0]) {
		for(int i = 0; i < len(pile); i++) {
			even[i] = !even[i];
		}

		neg = true;
	}

	long double cur = 0;
	for(int i = 0; i < len(pile); i++) {
		if(i > 0 && pile[i] == pile[i - 1]) {
			continue;
		}

		if(even[i]) {
			cur += 1.0L;
		} else {
			break;
		}
	}

	long double curAdd = 1;
	for(int i = cur; i < len(pile); i++) {
		if(i > 0 && pile[i] == pile[i - 1]) {
			continue;
		}

		if(even[i]) {
			cur = cur + curAdd / 2.0L;
		} else {
			cur = cur - curAdd / 2.0L;
		}
		curAdd /= 2.0L;
	}

	if(neg) {
		cur = -cur;
	}

	return cur;
}

int main(__attribute__((unused)) int argc, __attribute__((unused)) char **argv) {
	int t, k;
	scanf("%d", &t);
	while(t--) {
		scanf("%d", &k);
		vector<vector<int> > piles(k);
		for(int i = 0; i < k; i++) {
			int v;
			scanf("%d", &v);
			piles[i].resize(v);
			for(int j = 0; j < len(piles[i]); j++) {
				scanf("%d", &v);
				piles[i][j] = v;
			}
			sort(piles[i].begin(), piles[i].end());
		}

		long double ans = 0;
		for(int i = 0; i < len(piles); i++) {
			ans += val(piles[i]);
		}

		if(ans > 0) {
			printf("EVEN\n");
		} else if(ans < 0) {
			printf("ODD\n");
		} else {
			printf("DON'T PLAY\n");
		}
	}
	
	return 0;
}
