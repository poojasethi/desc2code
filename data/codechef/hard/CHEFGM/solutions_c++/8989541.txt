#include<bits/stdc++.h>
using namespace std;
#define LL long long

LL pile,n,t;
LL arr[100];
LL calculate() {
    LL res = 0;
    LL value = (((LL)1)<<40);
    res = (arr[0]%2==0)?value:-value;
    bool is_changed = false;
    for(LL i=1; i<n; ++i) {
        if(arr[i]%2 != arr[i-1]%2) {
            is_changed = true;
        }
        if(is_changed) value /= 2;
        res+=(arr[i]%2==0)?value:-value;
    }
    return res;
}
int main() {
    scanf("%d",&t);
    while(t--) {
        scanf("%d",&pile);
        LL res = 0;
        for(int i=0; i<pile; ++i) {
            scanf("%d",&n);
            for(int j=0; j<n; ++j)
                scanf("%d",&arr[j]);
            sort(arr,arr+n);
            res += calculate();
        }
        if(res > 0 ) printf("EVEN\n");
        else if(res < 0 ) printf("ODD\n");
        else printf("DON'T PLAY\n");
    }
    return 0;
}
