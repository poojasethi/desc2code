#include <cstdio>
 
inline int ri()
{
    register char c = 0;
    while(c<33) c = getchar_unlocked();
    int x = 0;
 
    while(c > 33)
    {
        x = x * 10 + c - '0';
        c = getchar_unlocked();
    }
    return x;
}
 
int a[100][45], sz[100];
 
void insort(int i)
{
	int j, tmp, hole;
	for(j=1; j<sz[i]; j++)
	{
		tmp = a[i][j];
		hole = j;
		while(hole > 0 && tmp < a[i][hole-1])
			a[i][hole] = a[i][hole-1], hole--;
		a[i][hole] = tmp;
	}
}
 
int main()
{
	int t, k, i, j, ok, p2, parity;
	long long gt, num;
	t = ri();
	while(t--)
	{
		k = ri();
		for(i=0; i<k; i++)
		{
			sz[i] = ri();
			for(j=0; j<sz[i]; j++)
				a[i][j] = ri();
			insort(i);
		}
		gt = 0;
		for(i=0; i<k; i++)
		{
			ok = 0, p2 = 0, num = 0, parity = a[i][0] & 1;
			for(j=0; j<sz[i]; j++)
			{
				if((a[i][j] & 1) != parity)
					ok = 1;
				if(ok) p2++;
				if(a[i][j] & 1)
					num += 1LL << (-p2 + 45);
				else
					num -= 1LL << (-p2 + 45);
			}
			gt += num;
		}
		if(gt > 0)
			printf("ODD\n");
		else if(gt < 0)
			printf("EVEN\n");
		else
			printf("DON'T PLAY\n");
	}
	return 0;
}
