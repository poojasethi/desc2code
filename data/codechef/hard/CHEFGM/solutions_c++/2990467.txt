#include <cstring>
#include <string>
#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;


typedef long long ll;

int a[45];

ll value(int n)
{
   ll sc = 0; 
   ll x = (ll(1))<<(45);
   ll oe= a[0]&1;
   if(oe) sc=x;
   else   sc=-x;
   
   bool flag = 0;
   
   for(int i=1; i<n; i++)
   {
      ll t1=a[i],t2=a[i-1];
      if(t1==t2) continue;
      
      if( (t1&1) != (t2&1) ) flag = 1;     
      if(flag) x= x>>1;     
      if(t1&1) sc+=x;
      else     sc+=-x;
   }
   return sc;
}

int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
    int k,n;
     scanf("%d",&k); 
     
     ll x = 0;
     for(int i=0; i<k; i++)
     {
       scanf("%d",&n);
       for(int j=0; j<n; j++) 
       scanf("%d",&a[j]);
       
       sort(a,a+n);
       x += value(n);
     }
     if(x < 0 ) printf("EVEN\n");
     else if(x > 0 ) printf("ODD\n");
     else printf("DON'T PLAY\n");
  }
  return 0; 
}
