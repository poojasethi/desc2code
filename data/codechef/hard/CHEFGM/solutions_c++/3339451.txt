#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

int t,k;
int a,b[101];

inline double SN(){
    double x=0;
    int i=0;
    while(i<a && b[i]%2==b[0]%2){
        if(b[0]%2==0){
            x=x+1;
        }
        else x-=1;
        i++;
    }
    double k=2.0;
    while(i<a){
        if(b[i]%2==0){
            x+=1/k;
        }
        else x-=1/k;
        i++,k*=2.0;
    }
    return x;
}
int main(){
    scanf("%d",&t);
    while(t--){
        scanf("%d",&k);
        double snEven=0;
        for(int i=0;i<k;i++){
            scanf("%d",&a);
            for(int j=0;j<a;j++){
                scanf("%d",&b[j]);
            }
            sort(b,b+a);
            snEven+=SN();
        }
        if(snEven>0) puts("EVEN");
        else if(snEven<0) puts("ODD");
        else puts("DON'T PLAY");
    }
    return 0;
}
