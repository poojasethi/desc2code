
#include<iostream>
#include<vector>
#include<algorithm>
#include<fstream>
#include <stdio.h>
using namespace std;
int main()
{
    //freopen("test.txt","r",stdin);
    int t;
    cin>>t;
    while(t--)
    {
        int k;
        scanf("%d",&k);
        long long sum = 0;
        for(int i=0;i<k;i++)
        {
           
            int ni;
            scanf("%d",&ni);
            vector<int> values;
            for(int j=0;j<ni;j++)
            {
                int l;
                scanf("%d",&l);
                values.push_back(l);
            }
            sort(values.begin(),values.end());
            long long total = 1LL<<48;
            int flag=0;
            sum += (values[0]%2==0)?total:-total;
            for(int i1=1;i1<ni;i1++)
            {
                if (values[i1]%2 != values[i1-1]%2){
                    flag = 1;
                }
                if (flag) total /= 2;
                sum += (values[i1]%2==0)?total:-total;
            }
        }
        if(sum>0)
            cout<<"EVEN"<<endl;
        else if(sum<0)
            cout<<"ODD"<<endl;
        else
            cout<<"DON'T PLAY"<<endl;
    }
    return 0;
}
