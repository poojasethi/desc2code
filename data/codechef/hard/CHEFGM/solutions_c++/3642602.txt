void get(int &x){

    while ((c=getchar_unlocked())<'0');

    }
}

int t,k,ni,n[50];

double valueofpile(int ni, int *n) {
	sort(n,n+ni);
	double w=0;
	int i=0,p0=n[0]%2;
	while (i<ni && n[i]%2 == p0) {w+=1; i++;}
	double d=1;
	while (i<ni) {
		d =d/2;
		if (n[i]%2 == p0)
			w+=d;
		else
			w-=d;
		i++;
	}
	return w;
}
