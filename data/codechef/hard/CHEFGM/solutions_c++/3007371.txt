    unsigned long long readint() {
    unsigned long long cc = getchar_unlocked(), ret = 0;
    for (; cc < '0' || cc > '9';)
    cc = getchar_unlocked();
    for (; cc >= '0' && cc <= '9';) {
    ret = ret * 10 + cc - '0';
    cc = getchar_unlocked();
    }
    return ret;
    }
    unsigned long long pow(unsigned long long p)
    {
    unsigned long long x=1,b=2;
    while(p)
    {
    if(p&1)
    x=x*b;
    b=b*b;
    p/=2;
    }
    return x;
    }
