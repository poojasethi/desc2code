#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>

using namespace std;
typedef long long ll;
const ll MAX = 1LL<<46;
int num, a[50];

ll solve() {
    int i;
    ll val, sum;
    val = MAX;
    sort(a, a + num);
    if(a[0] % 2 == 1) val = -val;
    i = 0;
    sum = 0;
    while(i < num && (a[i]%2 == a[0]%2)) {
        sum += val; i++;
    }
    for(int j = i; j < num; ++j) {
        if((a[j] % 2) != (a[j-1] % 2)) {
            sum += (val = -val/2);
        }
        else sum += (val = val/2);
    }
    return sum;
}

int main() {
    int test, n;
    ll ans;
    scanf("%d", &test);
    for(int t = 0; t < test; ++t) {
        scanf("%d", &n);
        ans = 0;
        for(int i = 0; i < n; ++i) {
            scanf("%d", &num);
            for(int j = 0; j < num; ++j) {
                scanf("%d", &a[j]);
            }
            ans += solve();
        }
        if(ans > 0) printf("EVEN\n");
        else if(ans < 0) printf("ODD\n");
        else printf("DON'T PLAY\n");
    }
    return 0;
}
