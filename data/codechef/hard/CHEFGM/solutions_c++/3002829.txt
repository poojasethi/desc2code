#include<stdio.h>
#include<iostream>
#include<algorithm>

using namespace std;
int arr[100][45],s[100];

main()
{
    //freopen("CHEFGM.txt","r",stdin);
    int t;
    cin>>t;
    while(t--)
    {
        long long gameval=0;
        int k;
        cin>>k;
        for(int i=0;i<k;i++)
        {
            cin>>s[i];
            for(int j=0;j<s[i];j++)
            {
                cin>>arr[i][j];
            }
            sort(arr[i],arr[i]+s[i]);
        }
        for(int i=0;i<k;i++)
        {
            for(int j=0;j<s[i];j++)
            {
                if(arr[i][j]%2) arr[i][j]=-1;
                else arr[i][j]=1;
            }
        }
        for(int i=0;i<k;i++)
        {
            long long val=35184372088832LL; /* 2^45*/
            bool iscont=true;
            for(int j=0;j<s[i];j++)
            {
                if(iscont)
                {
                    if(arr[i][0]!=arr[i][j])
                    {
                        val=val/2;
                        iscont=false;
                    }
                }
                else
                {
                    val=val/2;
                }
                gameval+=arr[i][j]*val;
            }

        }
        if(gameval==0LL)
        {
            printf("DON'T PLAY\n");
        }
        else if(gameval<0)
        {
            printf("ODD\n");
        }
        else
        {
            printf("EVEN\n");
        }


    }
}
