#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int arr[50];
int x;
ll power(ll a,ll b){
ll res = 1;
 while(b){
    if(b&1)
        res = res*a;
    a = a*a;
    b >>= 1;
 }
 return res;
}
ll solve(){
ll V = power(2,50);
ll res = 0;
if(arr[0]%2==0)
    res += V;
else
    res -= V;
bool change = false;
for(int i=1;i<x;i++){
  if(arr[i]%2 != arr[i-1]%2)
    change = true;
  if(change)
    V >>= 1;
  if(arr[i]%2==0)
    res += V;
  else
    res -= V;
}
return res;
}
int main(){
int t,n;
cin >> t;
while(t--){
    cin >> n;
    ll res = 0;
    while(n--){
        cin >> x;
        for(int i=0;i<x;i++)
            cin >> arr[i];
        sort(arr,arr+x);
        res += solve();
    }
    if(res > 0)
        cout << "EVEN" << endl;
    else if(res < 0)
        cout << "ODD" << endl;
    else
        cout << "DON'T PLAY" << endl;
}
return 0;
}
