#define max 105
#define max1 46
using namespace std;

int inline scan()
 { 
  int N = 0; 
  char C;
  C=getchar_unlocked();
   while (C < '0' || C>'9') 
   C=getchar_unlocked();
    while (C >= '0' && C <= '9')
	 {
	   N = (N<<3) + (N<<1) + C - '0'; 
	   C=getchar_unlocked();
     }
   return N;
 }
int k,n,t,i,j,u,l;
int game[max][max1];
int arr[max1];
float sumx,sumy;
 
float power(float x, int y)
{
    float temp;
    if( y == 0)
       return 1;
    temp = power(x, y/2);       
    if (y%2 == 0)
        return temp*temp;
    else
    {
        if(y > 0)
            return x*temp*temp;
        else
            return (temp*temp)/x;
    }
}
