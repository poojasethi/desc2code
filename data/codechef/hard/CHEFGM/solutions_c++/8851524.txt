#include<bits/stdc++.h>
using namespace std;

int main()

{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
      long long  int ans=0;
		int flag=0;
		while(n--)
		{
				long long int pow=1024LL*1024*1024*512;
				
		        int flag=0;
			int m;
			cin>>m;
			long long int a[m+10];
			for(int i=0;i<m;i++)
			{
				cin>>a[i];
			}
			sort(a,a+m);
			for(int i=0;i<m;i++)
			{
				if(a[i]%2!=a[0]%2 && flag==0)
				{
					flag=1;
				}
			//	cout<<pow<<"\t";
				if(flag==1)
				{
					pow/=2;
				}
			//		cout<<pow<<"\t";
				if(a[i]%2==0)
				ans+=pow;
				else
				ans-=pow;
			}
	    }
	  //  cout<<ans;
	    	if(ans>0)
			{
				cout<<"EVEN\n";
			}
			else if(ans<0)
			{
				cout<<"ODD\n";
			}
			else if(ans==0)
			{
				cout<<"DON'T PLAY\n";
			}
	}
	
	
	
}
