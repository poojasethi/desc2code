#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int t,k,ni,n[50];

double valueofpile(int ni, int *n) {
	sort(n,n+ni);
	double w=0;
	int i=0,p0=n[0]%2;
	while (i<ni && n[i]%2 == p0) {w+=1; i++;}
	double d=1;
	while (i<ni) {
		d =d/2;
		if (n[i]%2 == p0)
			w+=d;
		else
			w-=d;
		i++;
	}
	return w;
}

int main() {
	cin >> t;
	while (t--) {
		cin >> k;
		double w=0;
		while (k--) {
			cin >> ni;
			for (int j=0;j<ni;j++)
				cin >> n[j];
			double w1 =valueofpile(ni,n);
			if (n[0]%2==0)
				w += w1;
			else
				w -= w1;
		}
		cout << (w>0 ? "EVEN" : (w<0 ? "ODD" : "DON'T PLAY")) << '\n';
	}
	return 0;
}
