using namespace std;
int T,K,N[100];
int A[100][50];
double tot;

double evaluate(int pile[], int N, int odd){
	int index=0, sign=(odd)?-1:1;
	double val=0, base=2;
	while(index<N && pile[index]%2==odd) index++;
	val+=(sign*index);
	while(index<N){
		if(pile[index]%2) val+=(-1.0/base);
		else val+=(1.0/base);
		base*=2.0;
		index++;
	}
	return val;
}

int main()
{
		scanf("%d",&T);
        while(T-->0)
        {
        	tot=0.0;
            scanf("%d",&K);
            for(int i=0;i<K;i++)
            {
            	scanf("%d",&N[i]);
                for(int j=0;j<N[i];j++)
                {
                    scanf("%d",&A[i][j]); 
                }
                sort(A[i],A[i]+N[i]);
                int o=A[i][0]&1;
				tot+=evaluate(A[i],N[i],o);
            }	
            if(tot>0)
                printf("EVEN\n");
            else if(tot<0)
            	printf("ODD\n");
            else
				printf("DON'T PLAY\n");
        }
}
