#include <iostream>
#include <stdio.h>
#include <stdlib.h>

using namespace std;

inline int ri() {
    register char c = 0;
    while(c<33) c = getchar_unlocked();
    int x = 0;

    while(c > 33)
    {
        x = x * 10 + c - '0';
        c = getchar_unlocked();
    }
    return x;
}

int compare(const void * one,const void * two){
    int a = *((int*)one);
    int b = *((int*)two);
    if (a==b)
    	return 0;
    else if (a < b)
        return -1;
    else
    	return 1;
}

int main() {
	int t = ri();

//	cin >> t;

	for (int i = 0; i < t; i++) {
		float balance = 0;
		int pileCount = ri();

		for (int j = 0; j < pileCount; j++){
			int pileSize = ri();
			float divider = 1;
//			cin >> pileSize;
			int data[pileSize];
			for (int u = 0; u < pileSize; u++) {
//				cin >> data[u];
				data[u] = ri();
			}

			qsort(data, pileSize, sizeof(int), compare);


			int changed = 0;
			for (int u = 0; u < pileSize; u++) {
				if (u > 0 && data[u] % 2 != data[u-1] %2) {
					changed = 1;
				}
				if (changed == 1) {
					divider *= 2;
				}

				if (data[u] % 2 == 0){
					balance += 1 / divider;
				} else {
					balance -= 1 / divider;
				}
			}
		}

		if (balance > 0) {
			cout << "EVEN" << endl;
		} else if (balance < 0) {
			cout << "ODD" << endl;
		} else {
			cout << "DON'T PLAY" << endl;
		}
	}


	return 0;
}
