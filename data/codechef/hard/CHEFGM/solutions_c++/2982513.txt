#include<stdio.h>
#include<iostream>
#include<math.h>
#include<algorithm>
using namespace std;
int a[100][45];

#define g getchar_unlocked()
 
int scan()//fast input output
{
int t=0;
char c;
c=g;
while(c<'0' || c>'9')
c=g;
while(c>='0' && c<='9')
{
t=(t<<3)+(t<<1)+c-'0';
c=g;
}//end fast input output
return(t);
}

int main()
{
	int i,j,tc,k,n,size[100],parity;
	long long v,sum;
	scanf("%d",&tc);
	while(tc--)
	{
		k=scan();
		//scanf("%d",&k);
		for(i=0;i<k;i++)
		{
			n=scan();
			//scanf("%d",&n);
			size[i]=n;
			for(j=0;j<n;j++)
			a[i][j]=scan();//scanf("%d",&a[i][j]);
			sort(a[i],a[i]+n);
		}
		sum=0;
		for(i=0;i<k;i++)
		{
			v=1LL<<45;
			j=0;
			parity=a[i][j]&1;
			//printf("%d %d %d\n",j,parity,size[i]);
			while(j<size[i] && ((a[i][j]&1)==parity))
			{
				j++;
			if(parity)
			sum-=v;
			else
			sum+=v;
			}
			v/=2;
				for(;j<size[i];j++)
				{
					if(a[i][j]&1)
					sum-=v;
					else
					sum+=v;
					v/=2;
				}	
		}
		if(sum==0)
		printf("DON'T PLAY\n");
		else if(sum>0)
		printf("EVEN\n");
		else
		printf("ODD\n");
	}
	return 0;
}
