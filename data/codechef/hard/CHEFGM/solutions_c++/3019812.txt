#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cassert>

using namespace std;

unsigned long long fastexp1(int base, int power)
{
	  if(power==0) return 1;
	  unsigned long long result=fastexp1(base,power/2);
	  result=result*result;
	  if(power%2==1) result=result*base;
	  return result;
}

int main()
{
	  int t,k,ni;
	  scanf("%d",&t);
	  while(t--)
	  {
	  	  scanf("%d",&k);
	  	  double value[k];
	  	  double total;
	  	  for(int j=0; j<k; j++)
	  	  {
									scanf("%d",&ni);
									unsigned long v[ni];
									int pile[ni];
	  	  	  for(int i=0; i<ni; ++i)
	  	  	  {
												scanf("%lu",&v[i]);
	  	  	  }
	  	  	  sort(v,v+ni);
	  	  	  int i1=0;
	  	  	  for(int i=0; i<ni; ++i)
	  	  	  {
												
												if(v[i]%2==1) pile[i1]=1;
												else pile[i1]=2;
												i1++;
	  	  	  }
	  	  	  /*cout<<"\n";
	  	  	  for(int i=0; i<ni; ++i)
	  	  	  {
												
												cout<<pile[i]<<" ";
	  	  	  }
	  	  	  cout<<"\n";*/
	  	  	  int flag=0, ctr=1;
	  	  	  for(int i=0; i<i1; ++i)
	  	  	  {
	  	  	  	  if(flag==0 && pile[i]==pile[i+1]) continue;
	  	  	  	  if(flag==0)
	  	  	  	  {
	  	  	  	    if(pile[i]==2) value[j]=i+1;
	  	  	  	    else value[j]=-(i+1);
	  	  	  	    flag=1;
														continue;	
	  	  	  	  }
	  	  	  	  if(pile[i]==2) value[j]+=(double)1/fastexp1(2,ctr);
	  	  	  	  else value[j]-=(double)1/fastexp1(2,ctr);
	  	  	  	  ctr++;
	  	  	  }
	  	  	  ctr=1;
	  	  	  flag=0;
	  	  	  //printf("\n%lf ",value[j]);
	  	  	  total+=value[j];
	  	  	  //printf("%lf\n",total);
	  	  }
	  	  if(total<0) printf("ODD\n");
	  	  else if(total>0) printf("EVEN\n");
	  	  else printf("DON'T PLAY\n");
	  	  total=0;
	  }
	  return 0;
}
