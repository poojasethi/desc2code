typedef long long ll;

int a[45];

ll value(int n)
{
   ll sc = 0; 
   ll x = (ll(1))<<(45);
   ll oe= a[0]&1;
   if(oe) sc=x;
   else   sc=-x;
   
   bool flag = 0;
   
   for(int i=1; i<n; i++)
   {
      ll t1=a[i],t2=a[i-1];
      if(t1==t2) continue;
      
      if( (t1&1) != (t2&1) ) flag = 1;     
      if(flag) x= x>>1;     
      if(t1&1) sc+=x;
      else     sc+=-x;
   }
   return sc;
}

inline long long inp()
    {
    int noRead=0;
    char p=getchar_unlocked();
    for(;p<33;){p=getchar_unlocked();};
    while(p>32)
    {
    noRead = (noRead << 3) + (noRead << 1) + (p - '0');
    p=getchar_unlocked();
    }
    return noRead;
    }
    
    
int main()
{
  int t;
  scanf("%d",&t);
  while(t--)
  {
    int k,n;
     k=inp(); 
     
     ll x = 0;
     for(int i=0; i<k; i++)
     {
       n=inp();
       for(int j=0; j<n; j++) 
       a[j]=inp();
       
       sort(a,a+n);
       x += value(n);
     }
     if(x < 0 ) printf("EVEN\n");
     else if(x > 0 ) printf("ODD\n");
     else printf("DON'T PLAY\n");
  }
  return 0; 
}
