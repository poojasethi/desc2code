set<ll>a;
long long lob , hor;

int main()
{
    FAST;
    int ts , k , n ; cin>>ts;
    ll sign;
    while(ts--){

        sign = 0;
        ll tot = 0 ; //lob
        cin>>k;
        rep(i,1,k){
            cin>>n;
            a.clear();
            rep(i,1,n){
                ll x;
                cin>>x;
                a.insert(x);
            }
            ll cnt = 0;

            lob = 0 ;
            hor = 1 ;
            bool chk = true;
            ll prev = (*a.begin());
            sign= 0;
            for(set<ll>::iterator it = a.begin();it!=a.end();it++){

                ll val = (*it);

                if( val%2 == 0 && prev%2 != 0)  chk = false;
                if( val%2 != 0 && prev%2 == 0)  chk = false;
                prev = val;

                if(chk){
                    if(val%2==0) sign++;
                    else         sign--;
                }
                else{
                    cnt++;
                    ll one;
                    if(val%2==0)  one = +1;
                    else          one = -1;

                    ll nhor = (1LL<<cnt);
                    ll nlob = one;

                    hor  = nhor;
                    lob  = lob*2 + nlob;
                }
            }

//            cout<<"--> "<<lob<<endl<<"--> "<<hor<<endl;
//            cout<<sign<<endl;

            ll d = (45 - cnt);

//            cout<<"d "<<d<<endl;
            tot += lob*(1LL<<d) + sign*(1LL<<45);
        }

        if(tot >  0) cout<<"EVEN"<<endl;
        else if( tot < 0 ) cout<<"ODD"<<endl;
        else cout<<"DON'T PLAY"<<endl;
    }

	return 0;
}
