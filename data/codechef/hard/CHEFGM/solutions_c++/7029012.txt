#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
  ll t,n,k,i,a[46];
  double gamevalue=0;
  scanf("%lld",&t);
  while(t--){
    gamevalue=0; 
    scanf("%lld",&k);
    while(k--){
       scanf("%lld",&n);
       for(i=0;i<n;i++)
           scanf("%lld",&a[i]);
       sort(a,a+n);
       double sign=1;
       i=1;
       if(a[0]%2==0)
           sign=1;
       else
           sign=-1;   
       gamevalue+=sign ;    
       while(i<n && a[i]%2==a[i-1]%2){
          gamevalue+=sign;
          i++;
       }  
       double pr=1;
       for(;i<n;i++){
          if(a[i]%2==0)
              sign=1;
          else
              sign=-1;   
          gamevalue+=(sign*double(pr)/2);
         // cout<<"val="<<sign*double(pr)/2<<endl;
          pr=abs(double(pr)/2);   
       }   
       // cout<<gamevalue<<endl;  
    }
    //cout<<gamevalue<<endl;
    if(gamevalue > 0 ) 
       printf("EVEN\n"); 
    else if(gamevalue < 0 )  
       printf("ODD\n"); 
    else 
       printf("DON'T PLAY\n");
  } 
  return 0;
}
