bool kom(ev a, ev b){
	return a.t < b.t;
}

PII rek(int x){
	if (sov[x]){
		return dp[x];
	}else{
		dp[x] = MP(1, -ar[x].t);
		sov[x] = 1;
		
		FOR(i,x+1,n){
			int dx = abs(ar[x].x - ar[i].x);
			int dy = abs(ar[x].y - ar[i].y);
			int dt = ar[i].t - ar[x].t;
			if (dx + dy <= dt){
				dp[x] = max(dp[x], MP(1 + rek(i).F, rek(i).S) );
			}
		}
		
		return dp[x];
	}
}
