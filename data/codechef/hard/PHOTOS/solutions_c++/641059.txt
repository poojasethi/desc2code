typedef long long LL;
typedef vector<int> VI;
typedef pair<int,int> PI;
 
#define SI ({int x;scanf("%d",&x);x;})
#define SET(v,x) memset(v,x,sizeof v)
 
#define FOR(i,a,b) for(int _n(b),i(a);i<_n;i++) 
#define REP(i,n) FOR(i,0,n)
#define TC REP(kase,SI)
 
#define SZ(v) int(v.size())
#define PB(x) push_back(x)
#define ALL(v) (v).begin(),(v).end()
#define SORT(v) sort(ALL(v))
 
#define MP(x,y) make_pair(x,y)
#define F first
#define S second
 
#define MX 1005
 
int x[MX], y[MX];
PI v[MX];
int bt[MX], bl[MX];
 
int main()
{
	TC
	{
		int n=SI;
 
		FOR(i,1,n+1)
		{
			x[i]=SI; y[i]=SI;
			v[i].F=SI; v[i].S=i;
		}
 
		sort(v,v+n+1);
 
		for(int i=n;i>=0;i--)
		{
			bt[i]=1; bl[i]=v[i].F;
			int cx = x[v[i].S], cy = y[v[i].S];
 
			FOR(j,i+1,n+1)
			{
				int nx = x[v[j].S], ny = y[v[j].S];
				if(abs(cx-nx)+abs(cy-ny)<=v[j].F-v[i].F)
				{
					if(bt[i]<bt[j]+1 || (bt[i]==bt[j]+1 && bl[i]>bl[j]) )
						bt[i]=bt[j]+1, bl[i]=bl[j];
				}
			}
		}
 
		if(bt[0]==1) puts("No Photos");
		else printf("%d %d\n",bt[0]-1,bl[0]);
	}
 
	return 0;
}
