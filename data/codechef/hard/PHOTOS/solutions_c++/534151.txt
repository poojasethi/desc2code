#include <iostream>
#include <math.h>

int main()
{
int fall, n, i, j, max, z[10][1005];

for(std::cin >> fall >> n; fall--; std::cin >> n)
	{
	z[0][0]=z[6][0]=z[2][0]=z[3][0]=0;
	for(i=1; i<=n; std::cin>>z[0][i]>>z[6][i]>>z[3][i], z[2][i]=i++);
	for(i=1; i<=n; i++)
		for(j=i+1; j<=n; j++)
			if(z[3][z[2][i]]>z[3][z[2][j]])
				{
				z[2][j]^=z[2][i];
				z[2][i]^=z[2][j];
				z[2][j]^=z[2][i];
				}
	for(i=n; i>=0; z[4][z[2][i--]]=max+1)
		for(max=0, z[5][z[2][i]]=z[3][z[2][i]], j=i+1; j<=n; j++)
			if(((fabs(z[0][z[2][i]]-z[0][z[2][j]])+fabs(z[6][z[2][i]]-z[6][z[2][j]]))<=(z[3][z[2][j]]-z[3][z[2][i]])))
				if(max<z[4][z[2][j]])
					{
					max=z[4][z[2][j]];
					z[5][z[2][i]]=z[5][z[2][j]];
					}
				else if(max==z[4][z[2][j]]&&z[5][z[2][i]]>z[5][z[2][j]])
					z[5][z[2][i]]=z[5][z[2][j]];
	if(z[4][0]==1)
		std::cout << "No Photos" << std::endl;
	else
		std::cout << z[4][0]-1 << " " << z[5][0] << std::endl;
	}
return 0;
}
