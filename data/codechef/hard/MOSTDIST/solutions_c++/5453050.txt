int log(int n) {int ans=0;while(n>0) {n/=2;ans++;}return ans-1;}
ll mod(ll a){return a>0?a:-a;}

struct node{
	ll x,y,index;
	ll sum;
};
node makenode(ll x,ll y,ll index,ll sum) {
	node xx;
	xx.x=x,xx.y=y,xx.index=index;
	xx.sum=sum;
	return xx;
}
bool operator<(const node& lhs, const node& rhs)
{
    return lhs.sum < rhs.sum;
}
ll calc(node p,ll a,ll b) {
	return mod(p.x-a)+mod(p.y-b);
}
priority_queue <node> h1,h2,h3,h4;
int main() {
	ll ans=0,q,n;
	scanll(q);
	vector< ii > v;
	v.pb(make_pair(-1,-1));
	bool rem[mx6]={0};
	while(q--) {
		char ch=getchar();
		//prllf("ch:%c\n",ch);
		if(ch=='-') {
			scanll(n);
			n^=ans;
			rem[n]=1;
		}
		else if(ch=='+') {
			ll a,b;
			scanll(a),scanll(b);
			a^=ans;
			b^=ans;
			v.pb(make_pair(a,b));
			h1.push(makenode(a,b,v.size()-1,a+b)),h2.push(makenode(a,b,v.size()-1,a-b)),h3.push(makenode(a,b,v.size()-1,-a+b)),h4.push(makenode(a,b,v.size()-1,-a-b));
			//prllf("push size:%d\n",(ll)s.size());
		}
		else {
			ll a,b;
			scanll(a),scanll(b);
			a^=ans;
			b^=ans;
			node p,q,r,s;
			while(rem[h1.top().index]==1) {
				h1.pop();
			}
			p = h1.top();
			while(rem[h2.top().index]==1) {
				h2.pop();
			}
			q = h2.top();
			while(rem[h3.top().index]==1) {
				h3.pop();
			}
			r = h3.top();
			while(rem[h4.top().index]==1) {
				h4.pop();
			}
			s = h4.top();
			ll m = max4(calc(p,a,b),calc(q,a,b),calc(r,a,b),calc(s,a,b));
			printf("%lld\n",m);
			ans=m;
		}
	}	
}
