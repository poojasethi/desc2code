using namespace std;
 
long long l,r;
int nl,nr,al[20],ar[20];
 
int prod (int d4,int d7,int d) {
    int h=(d7+d-d4)/2;
    if ((0<=h)&&(h<=d)) return ((d7+d-h)*(d4+h));
    ++h;
    if ((0<=h)&&(h<=d)) return ((d7+d-h)*(d4+h));
    return max((d7+d)*d4,(d4+d)*d7);
}
 
int f(int l,int r,int d4,int d7,int d) {
    if (l>r) return 0;
    int kq=0;
    kq=max(kq,prod(d4,d7,d));
    if ((l<=4)&&(4<=r)) kq=max(kq,prod(d4+1,d7,d));
    if ((l<=7)&&(7<=r)) kq=max(kq,prod(d4,d7+1,d));
    return kq;
}
 
void xuly () {
    nl=nr=0;
    memset(al,0,sizeof(al));
    memset(ar,0,sizeof(ar));
    while (l>0) {
        ++nl;
        al[nl]=l%10; l/=10;
    }
    while (r>0) {
        ++nr;
        ar[nr]=r%10; r/=10;
    }
    int i=nr,d4=0,d7=0;
    while ((i>0)&&(ar[i]==al[i])) {
        if (ar[i]==4) ++d4;
        if (ar[i]==7) ++d7;
        --i;
    }
    if (i==0) {
        printf("%d\n",d4*d7); return;
    }
    int kq=0;
    kq=max(kq,f(al[i]+1,ar[i]-1,d4,d7,i-1));
    int j=i,e4=d4,e7=d7;
    while (j>0) {
        if (al[j]==4) ++e4;
        if (al[j]==7) ++e7;
        --j;
        if (j>0) kq=max(kq,f(al[j]+1,9,e4,e7,j-1));
    }
    kq=max(kq,e4*e7);
    j=i,e4=d4,e7=d7;
    while (j>0) {
        if (ar[j]==4) ++e4;
        if (ar[j]==7) ++e7;
        --j;
        if (j>0) kq=max(kq,f(0,ar[j]-1,e4,e7,j-1));
    }
    kq=max(kq,e4*e7);
    printf("%d\n",kq);
}
 
int main () {
    //freopen ("LUCKY8.INP","r",stdin);
    int t;
    scanf("%d",&t);
    while (t>0) {
        scanf("%lld%lld",&l,&r);
        xuly ();
        --t;
    }

}
