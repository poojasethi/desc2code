int dp[19][19][19][2][2];


int go(int nd, int n4, int n7, int c1, int c2){
	if(~dp[nd][n4][n7][c1][c2]) return dp[nd][n4][n7][c1][c2];
	int& ref = dp[nd][n4][n7][c1][c2];
	if(nd == 0) return n4*n7;
	int lb = 0, ub = 10;
	if(c1) lb = (a/pot[nd-1])%10;
	if(c2) ub = 1+(b/pot[nd-1])%10;
	Fr(i, lb, ub){
		int nc1, nc2, nn4 = n4, nn7 = n7;
		if(i == lb) nc1 = c1;
		else nc1 = 0;
		if(i == ub-1) nc2 = c2;
		else nc2 = 0;
		if(i == 4) nn4++;
		if(i == 7) nn7++;
		ref = max(ref, go(nd-1, nn4, nn7, nc1, nc2));
	}
	return ref;
}

		int ndig = 0;
		while(auxb) ndig++, auxb /= 10;
		printf("%d\n",go(ndig, 0, 0, 1, 1));
	}

}
