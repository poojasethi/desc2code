int cas;
string L,R;
int dp[20][20][20][2][2],vis[20][20][20][2][2],len;
int rec(int pos,int four,int seven,int low,int high){
   if(pos==len) return four*seven;
   int &ret=dp[pos][four][seven][low][high];
   if(vis[pos][four][seven][low][high]==cas) return ret;
   vis[pos][four][seven][low][high]=cas;
   ret=-1;
   int l=(low==1)?L[pos]-'0':0;
   int h=(high==1)?R[pos]-'0':9;
   for(int i=l;i<=h;i++){
     int _low=low&(L[pos]-'0'==i);
     int _high=high&(R[pos]-'0'==i);
     ret=max(ret,rec(pos+1,four+(i==4),seven+(i==7),_low,_high));
   }
   return ret;
}
int main(){
   int t;
   cas=-1;
   mem(dp,cas);
   mem(vis,cas);
   scanf("%d",&t);
   while(t--){
      cas++;
      cin>>L>>R;
      len=R.size();
      while(L.size()!=len){
         L='0'+L;
      }
      printf("%d\n",rec(0,0,0,1,1));
   }
   return 0;
}
