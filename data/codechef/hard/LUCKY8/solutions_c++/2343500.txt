using namespace std;

long long L, R;
int memo[21][21][21][2][2];
string L_, R_;

string convertLL(long long x){
       stringstream temp;
       temp << x;
       return temp.str();
       }

int f(int p, int c4, int c7, bool greater, bool smaller){
     int &x = memo[p][c4][c7][greater][smaller];
     if (x != -1) return x;
     if (p == 20){
           return x = c4*c7;
        }
     else{
           int s = L_[p] - '0';
           int t = R_[p] - '0';
           
           if (greater) s = 0;
           if (smaller) t = 9;
           
           //cout<<s<<" "<<t<<endl;
           int y = 0;
           for(int i = s; i <= t; i++){
                   y = max(y, f(p+1, c4+(i==4), c7+(i==7), greater||(i>s), smaller||(i<t)));
                   }
           x = y;
          }
     return x;
     }

int main(){
    
    int T;
    scanf("%d", &T);
    
    for(int i = 0; i < T; i++){
            scanf("%lld%lld",&L,&R);
            L_ = convertLL(L);
            while(L_.length() < 20) L_ = '0' + L_;
            //cout<<L_<<endl;

            R_ = convertLL(R);
            while(R_.length() < 20) R_ = '0' + R_;
            //cout<<R_<<endl;
            
            for(int a = 0; a <= 20; a++)
                    for(int b = 0; b <= 20; b++)
                            for(int c = 0; c <= 20; c++)
                                    for(int d = 0; d < 2; d++)
                                            for(int e = 0; e < 2; e++)
                                                    memo[a][b][c][d][e] = -1;
            printf("%d\n",f(0,0,0,0,0));
            }
            
    return 0;
}
