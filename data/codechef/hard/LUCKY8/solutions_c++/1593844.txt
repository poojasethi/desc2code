int zn(unsigned long long x)

{

}

unsigned long long get(int er)

{

         unsigned long long ans=0;

         for (int i=1;i<=er;i++)

         {

             ans=(unsigned long long) ans+ rr[i]*br[i];

         }

         return ans;

}

char solve( char kol)

{
    

    rr[kol]=1;

    for (char i=kol-1;i>0;i--)

        rr[i]=(unsigned long long)rr[i+1]*10;

    int ans=0;
    
   

    unsigned long long h=l;

    for (char i=kol;i>=1;i--)

    {

        mas[i]=h%10;

        h=h/10;    

    }

 

    for (char i=kol;i>=0;i--)

        for (char j=kol;j>=0;j--)

        {

            if (i+j>kol) continue;

            fl=0;
            
       

            if (i*j<=ansss) continue;

            iter=0;

            if ( rec(i, j, kol-i-j,  0, 1) )

               ansss=max(ansss, i*j);    

        }
  

    return ans;

}
