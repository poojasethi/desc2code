using namespace std;

ll C[N][N], Pow[N], L, R;
int S[N];

int Case(int x)
{
    if(x < 4)   return x + 1;
    if(x < 7)   return x;
    return x - 1;
}

ll Get(int c4, int c7, ll X)
{
    ll num = 0, cnt = 0;
    while(X)
    {
        S[++cnt] = X % 10;
        X /= 10;
    }
    reverse(S+1, S+cnt+1);
    for(int i=1; i<=cnt; i++)
    {
        if(c4 + c7 <= cnt-i) num += C[c4][cnt-i] * C[c7][cnt-i-c4] * Pow[cnt-i-c4-c7] * Case(S[i] - 1);
        if(S[i] > 4 && c4 - 1 + c7 <= cnt-i && c4 > 0)   num += C[c4-1][cnt-i] * C[c7][cnt-i-c4+1] * Pow[cnt-i-c4-c7+1];
        if(S[i] > 7 && c4 + c7 - 1 <= cnt-i && c7 > 0)   num += C[c4][cnt-i] * C[c7-1][cnt-i-c4] * Pow[cnt-i-c4-c7+1];
        if(S[i] == 4)   c4--;
        if(S[i] == 7)   c7--;
        if(c4 < 0)  return num;
        if(c7 < 0)  return num;
    }
    return num + (c4 == 0 && c7 == 0);
}

int main()
{
    //filein("inp.txt");  fileout("out.txt");
    int T;
    cin >> T;
    for(int i=0; i<=18; i++)    C[0][i] = 1;
    for(int i=1; i<=18; i++)
        for(int j=1; j<=18; j++)
            C[i][j] = C[i][j-1] + C[i-1][j-1];
    Pow[0] = 1;
    for(int i=1; i<=18; i++)
        Pow[i] = Pow[i-1] * 8;
  //  T = 1000;
    while(T--)
    {
        cin >> L >> R;
        //L = (ll)rand() * rand() * rand() * rand();
       // R = (ll)rand() * rand() * rand() * rand();
       // if(L > R)   swap(L, R);
        int ans = 0;
        for(int i=1; i<=18; i++)
            for(int j=1; j<=18-i; j++)
                if(Get(i, j, L-1) < Get(i, j, R))
                    ans = max(ans, i * j);
        cout << ans << endl;
    }
}
