#include<iostream>
#include<algorithm>
using namespace std;
int dp[1001],n;
int steps;

void calc_prefix(int gap){
int i,j;
if(gap>n) return;
cout<<n/gap<<" ";
for(i=gap;i<=n;i+=gap)
{
	cout<< i- gap/2 <<"+"<<i<<"="<<i<<" ";
}
cout<<"\n";

calc_prefix(gap*2);

cout<<(n-gap/2)/gap<<" ";
for(i=gap+gap/2;i<=n;i+=gap)
{
	cout<< i- gap/2 <<"+"<<i<<"="<<i<<" ";
}
cout<<"\n";
}




int main(){
	int i,j;
	cin>>n;
	int n1=n;steps=0;
	while(n1>1)
	{
		n1/=2;
		steps++;
	}
	cout<<steps + steps-1 +1<<"\n" ;
	calc_prefix(2);
	
}
