#include <stdio.h>
#include <vector>
using namespace std;

int n;

typedef struct {
	int a; int b; int c;
} oper;

vector<vector<oper> > comp;

void solve(int k) {
	vector<oper> par;
	for (int i=k;i+k<=n;i+=2*k) {
		par.push_back((oper){i,i+k,i+k});
	}
	comp.push_back(par);
	if (4*k<=n) solve(2*k);
	
		par.clear();
		for (int i=3*k;i<=n;i+=2*k) {
			par.push_back((oper){i-k,i,i});
		}
		comp.push_back(par);
	
}

int main() {
	scanf("%d",&n);
    solve(1);
    printf("%d\n",comp.size());
    int total=0;
    for (int i=0;i<comp.size();i++) {
    	printf("%d",comp[i].size());
    	for (int j=0;j<comp[i].size();j++) {
    		printf(" %d+%d=%d",comp[i][j].a,comp[i][j].b,comp[i][j].c);
    	}
    	total+=comp[i].size();
    	printf("\n");
    }
    return 0;
}
