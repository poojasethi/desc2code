#include<cstdio>
#include<vector>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

struct command{ int op1,op2,res; };

int main(){
	vector<command> ans[19];
	rep(t,10) rep(i,1024) if((i&(1<<t+1)-1)==(1<<t+1)-1) ans[t].push_back((command){i-(1<<t),i,i});
	rep(t, 9) rep(i,1023) if((i&(1<<t+1)-1)==(1<<t+1)-1) ans[18-t].push_back((command){i,i+(1<<t),i+(1<<t)});

	int n; scanf("%d",&n);

	puts("19");
	rep(t,19){
		int m=0;
		rep(i,ans[t].size()) if(ans[t][i].res<n) m++;
		printf("%d",m);

		rep(i,ans[t].size()) if(ans[t][i].res<n) {
			printf(" %d+%d=%d",ans[t][i].op1+1,ans[t][i].op2+1,ans[t][i].res+1);
		}
		puts("");
	}

	return 0;
}
