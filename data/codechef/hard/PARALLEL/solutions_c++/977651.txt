#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main(int argc, char *argv[])
{
	int n, i, k, steps;
	scanf("%d", &n);
	for (k = 1, steps = 0; k < n; k *= 2) steps++;
	steps = (steps - 1) * 2 + (k == n ? 1 : 0);
	printf("%d\n", steps);

	for (k = 2; k <= n; k *= 2) {
		printf("%d", n / k);
		for (i = k; i <= n; i += k) printf(" %d+%d=%d", i - k / 2, i, i);
		printf("\n");
	}
	
	for (k /= 2; k > 1; k /= 2) {
		if (n == k) continue;
		printf("%d", (n - k / 2) / k);
		for (i = k + k / 2; i <= n; i += k) printf(" %d+%d=%d", i - k / 2, i, i);
		printf("\n");
	}
	
	
	return 0;
}
