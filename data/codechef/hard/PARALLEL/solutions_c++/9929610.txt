#include <iostream>
#include <stdio.h>

using namespace::std;

void num_steps(int n, int p)
{
	if (n == 1)
		return;

	int d = (p << 1);
	if (n == 2)
	{
		printf("%d %d+%d=%d\n", 1, p, d, d);
		return;
	}

	printf("%d ", n/2);
	for (int i = 1; i <= n/2; i++)
	{
		int j = (i << 1);
		printf("%d+%d=%d ", p * (j - 1), p * j, p * j);
	}
	printf("\n");

	num_steps(n/2, d);

	if (n % 2 == 0)
		printf("%d ", n/2 - 1);
	else
		printf("%d ", n/2);

	for (int i = 1; i <= n/2; i++)
	{
		int j = (i << 1);
		if ((j + 1) > n)
			break;

		printf("%d+%d=%d ", p * j, p * (j + 1), p * (j + 1));
	}
	printf("\n");
}

int get_val(int n)
{
	if (n <= 2)
		return n/2;

	int ans = get_val(n/2);
	return (ans + 2);
}

main()
{
	int n;

	scanf("%d", &n);
	printf("%d\n", get_val(n));

	num_steps(n, 1);
}
