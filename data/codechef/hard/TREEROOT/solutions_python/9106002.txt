import operator

t = int(raw_input())

while t>0:
    n = int(raw_input())
    id_l,sid_l = [], []
    while n:
        id,sid = map(int,raw_input().split())
        id_l.append(id)
        sid_l.append(sid)
        n-=1
    zipped = zip(id_l,sid_l)
    suum =0
    for i in range(len(zipped)):
        suum += (operator.sub(zipped[i][0],zipped[i][1]))
    print suum

    t-=1
