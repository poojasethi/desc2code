import sys
import math
def main():
    t = int(sys.stdin.readline())
    for i in xrange(t):
        n = int(sys.stdin.readline())
        root=0
        for j in xrange(n):
            id,sum = map(int, sys.stdin.readline().split(' '))
            root=root+id-sum
        print root
if __name__=='__main__':
    main()
