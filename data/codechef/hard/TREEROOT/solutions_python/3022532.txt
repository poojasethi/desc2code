T = int(raw_input())

for _ in xrange(T):
    N = int(raw_input())
    root = 0
    
    # since child's sum is parent's id
    # after the following operation root's id 
    # will remain. provided there's a root.
    for _ in xrange(N):
        id, s = map(int, raw_input().split())
        root += id - s
        
    print root
