def main():
    t = int(raw_input())
    for i in xrange(t):
        n = int(raw_input())
        root = 0
        for j in xrange(n):
            id, sum_id = raw_input().split()
            id = int(id)
            sum_id = int(sum_id)
            root += id - sum_id
        print root
if __name__ == "__main__":
    main()
