

T = input()
for i in range(T):
    N = input()
    
    sum_of_children = 0
    sum_if_ids = 0
    for j in range(N):
        id_node, sum_child = map(int, raw_input().split())
        sum_of_children += sum_child
        sum_if_ids += id_node
    
    root = sum_if_ids - sum_of_children
    print root
