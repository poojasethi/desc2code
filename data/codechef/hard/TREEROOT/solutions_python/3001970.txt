import sys

def get_int():
    return int(sys.stdin.readline())

def get_ints():
    return [int(i) for i in sys.stdin.readline().split()]

def run():
    t = get_int()
    while t:
        n = get_int()
        root = 0
        while n:
            a, b = get_ints()
            root += a - b
            n -= 1
        print root
        t -= 1

run()
