#include<bits/stdc++.h>
using namespace std;
typedef pair<int,int> pii;
vector<pii> ar;

int main()
{
    int t;
    scanf("%d",&t);

    while(t--)
    {
        ar.clear();
        int n;
        scanf("%d",&n);
        for(int i=0;i<n;++i)
        {
            int a,b;
            scanf("%d",&a);
            scanf("%d",&b);
            ar.push_back(make_pair(a,b));
        }

        if(ar.size()==1)
        {
            printf("%d\n",ar[0].first);
            continue;
        }

        int sum=0;

        for(int i=0;i<ar.size();++i)
        {
            sum += (ar[i].first-ar[i].second);
        }
        printf("%d\n",sum);
    }
}
