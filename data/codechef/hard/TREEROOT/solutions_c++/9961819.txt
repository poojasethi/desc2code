#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,j,k;
    int n,m;
    
    int t;
    cin>>t;

    while(t--)
    {
        cin>>n;
        
        k=0;
        
        for(m=0;m<n;m++)
        {
            cin>>i>>j;
            
            k+=(i-j);
        }

        cout<<k<<endl;

    }

    return 0;
}
