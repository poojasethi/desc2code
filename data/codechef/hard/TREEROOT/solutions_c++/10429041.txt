#include <iostream>
using namespace std;
int main(){
	int testcases;
	cin>>testcases;
	while(testcases-- > 0){
		int numberOfNodes;
		cin>>numberOfNodes;
		int sumOfAll = 0;
		while(numberOfNodes-- > 0){
			int parent,sumOfChildren;
			scanf("%d %d",&parent, &sumOfChildren);
			sumOfAll += parent - sumOfChildren; 
		}
		cout<<sumOfAll<<endl;
	}
	return 0;
}
