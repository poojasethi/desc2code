#include <iostream>
using namespace std;
 
int main() {

	int t;
	cin >> t;
	for(int i=0;i<t;i++){
		int n;
		cin >> n;
		int sumNode=0;
		int sumChild=0;
		int x1,x2;
		for(int i=0;i<n;i++){
			cin >> x1 >> x2;
			sumNode+=x1;
			sumChild+=x2;
		}
		cout << sumNode - sumChild << endl;
	}
	return 0;
}
