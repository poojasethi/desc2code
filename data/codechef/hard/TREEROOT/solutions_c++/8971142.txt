#include <bits/stdc++.h>
 
using namespace std;
 
char str[1000005];

#define lld long long int
 
int main() {
	lld t;
	cin >> t;
	
	while(t--) {
		lld n, root = 0;
		cin >> n;
		
		while(n--) {
			lld id, sum;
			cin >> id >> sum;
			
			root = root + (id - sum);
		}
		
		cout << root << endl;
	}
	
	return 0;
}
