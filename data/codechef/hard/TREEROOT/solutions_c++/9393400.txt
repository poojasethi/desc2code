
#include<stdio.h>

 int main(){
 int t, n, id[50],child[50];
int i,s;

//printf("enter value of t\n");
scanf("%d",&t);
while(t-->0){
s=0;
             //printf("enter the value of n\n");
scanf("%d",&n);

//printf("enter data\n");
for (i=0;i<n;i++){
scanf("%d %d",&id[i],&child[i]);}

for (i=0;i<n;i++){
s+=id[i];}
for (i=0;i<n;i++){
s-=child[i];}

printf("%d\n",s);

} 
  
 return 0;
 
 }
