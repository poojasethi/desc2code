#include <iostream>

using namespace std;

class Solver
{
public:
    void solve()
    {
        int id, children;
        int sumCh = 0;
        int sumAll = 0;
        int n; cin >> n;
        while (n--)
        {
            cin >> id >> children;
            sumCh += children;
            sumAll += id;
        }
        cout << sumAll - sumCh << endl;
    }
    
private:
};

int main()
{
    int T; cin >> T;
    while (T--)
    {
        Solver s;
        s.solve();
    }
    return 0;
}
