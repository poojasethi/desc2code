#include <bits/stdc++.h>

using namespace std;

int main() {
	cin.sync_with_stdio(false);
	int t, x, y, n;
	cin >> t;
	while(t --) {
		cin >> n;
		long long sum_x = 0, sum_y = 0;
		for(int i = 0; i < n; i ++) {
			cin >> x >> y;
			sum_x += x;
			sum_y += y;
		}
		cout << sum_x - sum_y << endl;
	}
	return 0;
}
