#include<stdio.h>
int main(){
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++){
        int n;
        scanf("%d",&n);
        int sum_id=0,sum_child=0;
        for(int j=0;j<n;j++){
            int id,child_s;
            scanf("%d %d",&id,&child_s);
            sum_id+=id;
            sum_child+=child_s;
        }
        printf("%d\n",sum_id-sum_child);
    }
    return 0;
}
