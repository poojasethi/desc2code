#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int a[1000];
		int i,j,sum=0;
		for(i=0;i<1000;i++)a[i]=-1;
		int sum1=0;
		for(i=0;i<n;i++)
		{
			int k;
			cin>>k;sum=sum+k;
			cin>>j;sum1=sum1+j;a[k-1]=j;
		}
		for(i=0;i<1000;i++)
		{
			if(sum-(i+1)==sum1&&a[i]>=0)
			{
				cout<<i+1<<"\n";
			}
		}
	}
	return 0;
}
