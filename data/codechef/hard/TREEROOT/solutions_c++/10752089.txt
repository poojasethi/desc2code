#include <iostream>
using namespace std;

int sol(){
    int n;cin >> n;
    int sum_id=0;
    int sum_sid=0;
    int t1,t2;
    for(int i=0;i<n;i++){
        cin >> t1 >> t2;
        sum_id+=t1;
        sum_sid+=t2;
    }
    cout << sum_id - sum_sid << endl;
}


int main() {
	// your code goes here
	int t;cin >> t;
	for(int i=0;i<t;i++){
	    sol();
	}
	return 0;
}
