#include <stdio.h>

int soma[32];
int bin[32][32];
long long a[32], pos[1<<16];

int main()
{
	for(int i = 0; i < 31; i++)
		for(int j = 0; j <= i; j++)
			if(j == 0 || j == i) bin[i][j] = 1;
			else bin[i][j] = bin[i-1][j] + bin[i-1][j-1];
	
	for(int i = 0; i < 32; i++)
		for(int j = 1; j <= i; j+=2)
			soma[i] += bin[i][j];

	int tc;
	scanf("%d", &tc);
	while(tc--) {
		int n;
		long long sol = 0;
		scanf("%d", &n);
		for(int i = 0; i < n; i++)
			scanf("%lld", &a[i]);
		pos[0] = -1;
		for(int i = 1; i < (1<<n); i++) {
			int pop = __builtin_popcount(i);
			pos[i] = pos[i&(i-1)] & a[__builtin_ctz(i)];
			if(pop%2)
				sol += soma[pop]*(1LL<<__builtin_popcountll(pos[i]));
			else
				sol -= soma[pop]*(1LL<<__builtin_popcountll(pos[i]));
		}
		printf("%lld\n", sol);
	}
	return 0;
}
