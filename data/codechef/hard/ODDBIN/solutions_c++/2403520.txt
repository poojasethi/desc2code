#include <cstdio>
 
long long omask[1<<15]={-1};
 
int main(){
	int T, m;
	long long a;
	for(scanf("%d", &T); T--; ){
		scanf("%d", &m);
		unsigned long long res=0;
		for(int i=0; i<m; i++){
			scanf("%lld", &a);
			for(int j=0; j<1<<i; j++){
				omask[(1<<i)+j]=omask[j]&a;
				int n=__builtin_popcount(j);
				int c=n+__builtin_popcountll(omask[(1<<i)+j]);
				if(c<64)
					res+=(n&1 ? -1ll : 1ll)<<c;
			}
		}
		printf("%llu\n", res);
	}
	return 0;
}
