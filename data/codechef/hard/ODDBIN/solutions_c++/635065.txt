#include <stdio.h>

int soma[32];
int bin[32][32];
long long a[32], pos[1<<16];

int main()
{
	for(int i = 1; i < 32; i++)
		if(!(i%2)) soma[i] = -(1<<(i-1));
		else soma[i] = (1<<(i-1));


	int tc;
	scanf("%d", &tc);
	while(tc--) {
		int n;
		long long sol = 0;
		scanf("%d", &n);
		for(int i = 0; i < n; i++)
			scanf("%lld", &a[i]);
		pos[0] = -1;
		for(int i = 1; i < (1<<n); i++) {
			int pop = __builtin_popcount(i);
			pos[i] = pos[i&(i-1)] & a[__builtin_ctz(i)];
			sol += soma[pop]*(1LL<<__builtin_popcountll(pos[i]));
		}
		printf("%lld\n", sol);
	}
	return 0;
}
