#include<bits/stdc++.h>
using namespace std;
bool cmp(pair<long long int,long long int > p1,pair<long long int,long long int > p2){
    return p1.second==p2.second ? p1.first>p2.first:p1.second>p2.second; 
}
int main(){
    int t,n;
    cin>>t;
    while(t--){
        cin>>n;
        pair< long long int, long long int > v[n];
        for(int i=0;i<n;i++)
        cin>>v[i].first;
        for(int i=0;i<n;i++)
        cin>>v[i].second;
        sort(v,v+n,cmp);
        long long int dp[5002];
        memset(dp, 0, sizeof(dp));
        for(int i=0;i<n;i++){
            for(int j=i+1;j>0;j--){
                dp[j]=max(dp[j],dp[j-1]+v[i].first+(j-1)*v[i].second);
            }
        }
        for(int i=1;i<=n;i++)
        cout<<dp[i]<<" ";
        cout<<endl;
    }
}
