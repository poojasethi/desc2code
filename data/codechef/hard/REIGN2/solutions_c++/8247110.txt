#include <bits/stdc++.h>

using namespace std;

#define ll long long int

typedef pair<ll,ll> ii;

ll dp[5005][5005];
ii sp[5005];

bool comp(ii x,ii y)
 {
     return x.second > y.second;
 }

int main() {
	// your code goes here
	
	int test;
	cin>>test;
	
	while(test--)
	 {
	   int n;
	   cin>>n;
	   
	   for(int i=1;i<=n;i++)
	    cin>>sp[i].first;
	    
	   for(int i=1;i<=n;i++)
	    cin>>sp[i].second;
	    
	   sort(sp+1,sp+n+1,comp); 
	   
	   dp[1][1] = sp[1].first;
	   
	   for(int i=2;i<=n;i++)
	   {
	    dp[i][i] = dp[i-1][i-1] + sp[i].first + (i-1)*sp[i].second;
	    dp[i][1] = max(dp[i-1][1] , sp[i].first);
	   }
	   
	   
	   for(int i=3;i<=n;i++)
	    {
	        for(int j=2;j<i;j++)
	         {
	             dp[i][j] = max(dp[i-1][j],dp[i-1][j-1] + sp[i].first + (j-1) *sp[i].second);
	             
	         }
	    }
	   
	   
	    for(int i=1;i<=n;i++)
	     cout<<dp[n][i]<<" ";
	    
	    cout<<"\n"; 
	 }
	
	
	
	return 0;
}
