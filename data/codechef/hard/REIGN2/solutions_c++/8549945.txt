#include<bits/stdc++.h>

#define ll long long
const ll mod = 1000*1000*1000+7;

using namespace std;

struct pt
{
  ll a;
  ll b;

};

ll mx(ll x,ll y)
{
  if(x<y)
    return y;
  return x;
}

bool cmp(pt const &l,pt const &r)
{

    return l.a - r.b;
}

ll ta[5005], tb[5005];

bool my(int i,int j)
{
  if(tb[i]!=tb[j])
    return (tb[i] > tb[j]);
  else
    return (ta[i] > ta[j]);
}

    for(i=0;i<n;++i)
      cin>>tb[i];
    for(i=0;i<n;++i)
    {
      pt temp;
      temp.a = a[i];
      temp.b = b[i];
      p.push_back(temp);
    }
    ll rank[5005];
    for(i=0;i<n;++i)
      rank[i] = i;
    sort(rank,rank+n,my);
    for(i=0;i<n;++i)
    {
      a[i] = ta[rank[i]];
      b[i] = tb[rank[i]];
    }

  /*  for(i=0;i<n;++i)

    /*for(i=0;i<n;++i)

    ll dp[10][5005],j;

      for(j=0;j<=n;++j)
      {
        if(i==0)
        {
          dp[i%2][j] = 0;
          continue;
        }
        if(j==0)
        {
          dp[i%2][j] = 0;
          continue;
        }
        if(j>i)
        {
          dp[i%2][j] = 0;
          continue;
        }
        dp[i%2][j] = mx(dp[(i-1)%2][j-1] + a[i-1] + (j-1)*b[i-1], dp[(i-1)%2][j]);
      }
