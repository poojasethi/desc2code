ll a[5005][5005];

bool cmp( const pair<ll,ll>a, const pair<ll,ll>b)
{
    if(a.second == b.second)
        return a.first>b.first;

    return a.second>b.second;
}

void solve(ll a[5005][5005],ll n, const vector<pair<ll,ll> >p)
{
    ll i,j;
    a[1][1] = p[1].first;

    for(i=1;i<=n;i++)
    {
        for(j=1;j<=i;j++)
            a[i][j] =  max( a[i-1][j], a[i-1][j-1] + p[i].first +(j-1)* p[i].second );
    }
}

int main()
{
    ll t;
    ip(t);
    while(t--)
    {
        ll n,val,i;
        ip(n);

        memset(a,0,sizeof(a));

        vector<pair<ll,ll> >p(n+1);

        for(i=1;i<=n;i++)
        {
                ip(val);
                p[i].first=val;
        }
        for(i=1;i<=n;i++)
        {
                ip(val);
                p[i].second=val;
        }
        sort(p.begin()+1, p.end(),cmp);
        solve(a,n,p);
        for(i=1;i<=n;i++)
            printf("%lld ",a[n][i]);
        nl;

  }
}
