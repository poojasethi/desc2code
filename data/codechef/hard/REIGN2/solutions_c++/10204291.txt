#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cassert>
#include <vector>
#include <bits/stdc++.h>
using namespace std;
pair<int, int> a[5001];
long long f[5001];

int main()
{
    int t, n;
    cin>>t;
   while(t--) {
      cin>>n;
        for (int i = 0; i < n; ++ i) 
            scanf("%d", &a[i].second);
            
        for (int i = 0; i < n; ++ i) {
           scanf("%d", &a[i].first);
        }

        sort(a, a + n);
        reverse(a, a + n);

        memset(f, 0, sizeof(f));
        for (int i = 0; i < n; ++ i) {
            for (int j = i; j >= 0; -- j) {
                f[j + 1] = max(f[j + 1], f[j] + (long long)j * a[i].first + a[i].second);
            }
        }
        for (int i = 1; i <= n; ++ i) {
            printf("%lld%c", f[i], i == n ? '\n' : ' ');
        }
    }
    return 0;
}
