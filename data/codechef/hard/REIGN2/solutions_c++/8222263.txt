#include<bits/stdc++.h>
#define INF (1LL<<60)
using namespace std;

long long DP[5005][5005];//index,taken so far
pair<long long,long long>castle[5005];
int N;

bool comp(pair<long long,long long>p1, pair<long long,long long>p2)
{
    if(p1.second==p2.second)
        return (p1.first<p2.first);
    return (p1.second>p2.second);
}

int main()
{
    int i,j,T;
   // freopen("reign2.in","r",stdin);
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&N);
        for(i=1; i<=N; i++)
            scanf("%lld",&castle[i].first);
        for(i=1; i<=N; i++)
            scanf("%lld",&castle[i].second);


        sort(castle+1,castle+N+1,comp);
        //for(i=1; i<=N; i++)  printf("%lld %lld\n",castle[i].first,castle[i].second);
        for(i=0; i<=N; i++)
            for(j=0; j<=N; j++)  DP[i][j]=0;

        DP[1][0]=0;DP[1][1]=castle[1].first;

        for(i=2; i<=N; i++)
        {
            DP[i][0]=0;
            for(j=1; j<=i; j++)
            {
                DP[i][j]=max(DP[i-1][j],DP[i-1][j-1]+castle[i].first+castle[i].second*(j-1));
            }
        }


        for(i=1; i<=N; i++)
        {
            if(i>1)  printf(" ");
            printf("%lld",DP[N][i]);
        }
        puts("");
    }
    return 0;
}
