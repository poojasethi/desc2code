#include <bits/stdc++.h>

using namespace std;

#define ll long long

const int maxn = 5010;

ll dp[maxn][maxn];
vector<pair<ll, ll> > v;

int main()
{
	//freopen("i.txt", "r", stdin);
	int t, n, i, j;
	scanf("%d", &t);
	while(t--){
		v.clear();
		scanf("%d", &n);
		v.resize(n+1);
		for(i=1; i<=n; i++)
			scanf("%lld", &v[i].second);
		for(i=1; i<=n; i++)
			scanf("%lld", &v[i].first);
		sort(v.begin()+1, v.end());
		reverse(v.begin()+1, v.end());
		for(i=0; i<=n; i++){
			for(j=0; j<=n; j++)
				dp[i][j] = 0;
		}
		for(i=1; i<=n; i++){
			for(j=1; j<=i; j++){
				dp[i][j] = max(dp[i-1][j-1]+v[i].second+v[i].first*(j-1), dp[i-1][j]);
			}
		}
		for(i=1; i<=n; i++)
			printf("%lld ", dp[n][i]);
		printf("\n");
	}
	return 0;
}
