// compiling with gcc11 or above

// PIE is defined as M_PI

const int mod = 1000000000 + 7 ;
const int maxn = 100000 + 9 ;

bool cmp(pii a,pii b){
	if(a.sd==b.sd){
		return a.ft > b.ft;
	}
	return a.sd > b.sd;
}

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int t;
	cin >> t;
	while(t--){
		int n;
		cin >> n;
		vector< pii > vec( n+1 , mp(0,0) );
		for(int i=1;i<=n;i++){
			cin >> vec[i].ft;
		}
		for(int j=1;j<=n;j++){
			cin >> vec[j].sd;
		}
		sort(vec.begin()+1,vec.end(),cmp);
		vector< vector<ll> > dp( n+1 , vector<ll> (n+1,0) );
		for(int j=1;j<=n;j++){
			for(int i=j;i<=n;i++){
				dp[i][j]=max(dp[i-1][j],dp[i-1][j-1]+vec[i].ft+(j-1)*((ll)vec[i].sd));
			}
		}
		for(int i=1;i<=n;i++){
			cout << dp[n][i] << " ";
		}
		cout << "\n";
	}
	return 0;
}
