const int N = 5010; 
int n, m;
int64 day[N];
int64 a[N], b[N];
pair <int64, int64> ba[N];
void solve()
{
	cin >> n;
	for(int i = 0; i < n; ++i)
		cin >> a[i];
	for(int i = 0; i < n; ++i)
		cin >> b[i];
	for(int i = 0; i < n; ++i)
	{
		ba[i] = mp(b[i], a[i]);
	}
	sort(ba, ba + n);
	memset(day, -1, sizeof day);
	day[0] = 0;
	for(int i = n - 1; i >= 0; --i)
	{
		for(int j = n - 1; j >= 0; --j)
		{
			if (day[j] == -1)
				continue;
			day[j + 1] = max(day[j + 1], day[j] + ba[i].first * j + ba[i].second); 
		}
	}
	for(int i = 1; i <= n; ++i)
		cout << day[i] << ' ';
}
