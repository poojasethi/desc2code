#include<bits/stdc++.h>
#define s(n) scanf("%d",&n)
#define ll long long
#define pb push_back
using namespace std ;


ll dp[5001][5001];

bool compare(pair<int,int> a,pair<int,int> b){
  return a.first>b.first ;
}
int main(){

int t,n1,n2,n,p;
int a[5001],b[5001],i,j;

cin>>t;
while(t--){

  cin>>n;
  pair<ll,ll>p[n+1];
  for(i=0;i<n;++i){
      s(a[i]);
  }
  for(i=0;i<n;++i){
      s(b[i]);
  }
  for(i=0;i<n;++i){
     p[i].first = b[i];
     p[i].second = a[i];
  }
  sort(p,p+n,compare);


  for(i=0;i<n;++i)
    for(j=0;j<=i+1;++j)
        dp[i][j] = 0;

  for(i=0;i<n;++i){
    for(j=1;j<=i+1;++j){
       if(i==0){
         dp[0][1] = p[0].second ;
       }else{
         dp[i][j] = max(dp[i-1][j],dp[i-1][j-1]+(j-1)*p[i].first+p[i].second);
       }
    }
  }
  for(i=1;i<=n;++i)
    printf("%lld ",dp[n-1][i]);
   printf("\n");

}
return 0;
}
