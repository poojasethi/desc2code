#include <bits/stdc++.h>

using namespace std;

struct node
{
    long a,b;
};

int cmp(node,node);

long long dp[5005];
node c[5005];
int main()
{
    /*ios::sync_with_stdio(0);
    cin.tie(NULL);
    cout.tie(NULL);*/

    long long t,n,i,j;
    cin>>t;

    while(t--)
    {
        cin>>n;
        for(i=0;i<n;i++)
            cin>>c[i].a;
        for(i=0;i<n;i++)
            cin>>c[i].b;

        sort(c,c+n,cmp);
        memset(dp,0,sizeof(dp));

        for(i=1;i<=n;i++)
            for(j=i;j>0;j--)
                dp[j]=max(dp[j],dp[j-1]+c[i-1].a+(j-1)*c[i-1].b);

        for(i=1;i<=n;i++)
            cout<<dp[i]<<" ";
        cout<<endl;
    }
    return 0;
}

int cmp(node m,node n)
{
    return(m.b>n.b);
}
