#include <bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define sz size()
#define all(a)  a.begin(), a.end()
#define allr(a) a.rbegin(), a.rend()
#define SL(n) scanf("%lld", &n)
#define fill(a, x) memset(a, x, sizeof(a));
#define mod 1000000007
#define eps 0.000000001

using namespace std;
typedef long long LL;
typedef vector <LL> VL;
typedef map <LL, LL> ML;
typedef pair<LL, LL> PL;
typedef vector <PL> VPL;

bool dbg = 0;

LL dp[5005];
LL dp1[5005];
int N;
int a[5005];
int b[5005];
pair<int, int> Inp[5005];

int main() {
    int T;
    cin >> T;
    while(T--){
        scanf("%d", &N);
        for(int n=1;n<=N;++n){
            scanf("%d", &Inp[n].ss);
        }
        for(int n=1;n<=N;++n){
            scanf("%d", &Inp[n].ff);
        }
        sort(Inp + 1, Inp + N + 1);
        for(int n=1;n<=N;++n){
            a[n] = Inp[n].ss;
            b[n] = Inp[n].ff;
        }
        fill(dp, (LL)0);
        fill(dp1, (LL)0);
        for(int n=N;n>0;--n){
            for(int m=1;(m+n)<=(N+1);++m){
                dp1[m] = max(dp[m], dp[m-1] + a[n] + (m - 1) * 1ll * b[n]);
            }
            for(int m=1;(m+n)<=(N+1);++m){
                dp[m] = dp1[m];
            }
        }
        for(int n=1;n<=N;++n){
            cout << dp[n] << " ";
        }
        cout << endl;
    }
    return 0;
}
