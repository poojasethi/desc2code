#include<bits/stdc++.h>
#define ll long long
#define fi first
#define se second
ll mpow(ll a, ll n,ll mod)
{ll ret=1;ll b=a;while(n) {if(n&1)
    ret=(ret*b)%mod;b=(b*b)%mod;n>>=1;}
return (ll)ret;
}
using namespace std;
#define mem(x,a) memset(x,a,sizeof(x))
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define all(v) v.begin(),v.end()
using namespace std;
#define N (int)(5e3+4)
ll dp[2][N];
vector<pii> v;
void solve(){
    int n;
    cin>>n;
    v.resize(n);
    for(int i=0;i<n;i++){
        int x;
        cin>>x;
        v[i].se=x;
    }
    for(int i=0;i<n;i++){
        int x;
        cin>>x;
        v[i].fi=-x;
    }
    mem(dp,0);
    sort(v.begin(),v.end());
    for(int i=1;i<=n;i++){
        int pre=(i-1)%2;
        int nx=i%2;
        for(int j=1;j<=i;j++){
            ll p=max(dp[pre][j-1]+(-v[i-1].fi)*(1ll)*(j-1)+v[i-1].se,dp[pre][j]);
            dp[nx][j]=max(p,dp[nx][j]);
        }
    }
    for(int i=1;i<=n;i++){
        cout<<dp[n%2][i]<<" ";
    }
    printf("\n");
}
int main(){
   //ios_base::sync_with_stdio(false);
   int t=1;
   scanf("%d",&t);
   for(int i=1;i<=t;i++){
       //printf("Case #%d: ",i);
       solve();
   }
   return 0;
}
