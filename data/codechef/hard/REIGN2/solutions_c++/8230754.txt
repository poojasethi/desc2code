#include <bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define pii pair<int, int>
#define pll pair<long long, long long>
#define vi  vector<int>
#define pb  push_back
#define mp  make_pair
#define M(a,x) memset(a,x,sizeof(a))
#define sci(x) scanf("%d",&x);
#define scl(x) scanf("%lld",&x);
#define scs(x) scanf("%s",x);
#define print(x) printf("%d",x);
#define nl printf("\n")
#define printl(x) printf("%lld",x)
#define F(i,a,n) for(int i=a;i<n;i++)
#define INF 100000000000000000LL
#define fr first
#define se second
#define LL long long
const int N = 5005;
LL dp[N][N];
vector< pair<LL,LL> > v;
int main() {
    int t;sci(t);
    while(t--) {
        v.clear();
        M(dp,0);
        int n;sci(n);
        F(i,0,n) {
            int x;
            cin>>x;
            v.pb(mp(0,x));
        }
        F(i,0,n) {
            int x;sci(x);
            v[i].fr=x;
        }
        v.pb(mp(INF,1));
        sort(v.rbegin(),v.rend());
        F(i,1,n+1) {
            for(int j=1;j<=i;j++) {
                dp[i][j]=max(dp[i-1][j],dp[i-1][j-1]+v[i].fr*(j-1)+v[i].se);
            }
        }
/*      F(i,1,n+1) {
            F(j,1,i+1) {
                cout<<dp[i][j]<<" ";
            }
            cout<<endl;
        }*/
        F(i,1,n+1) {
            printl(dp[n][i]);printf(" ");
        }
        cout<<endl;
    }
}
