#include <iostream>
#include <algorithm>
#include <cassert>
using namespace std;
class code
{
public:
	long long int a,b;
};

bool myf(const code &a,const code &b)
{
	return a.b>b.b;
}
static long long int arr[5005][5005];
int main()
{
	long long int t,n,u,j,i;
	code k[5005];
	cin>>t;
	while (t--)
	{
		cin>>n;
		for (i=0;i<n;i++)
		{
			cin>>k[i].a;
		}
		for (i=0;i<n;i++)
		{
			cin>>k[i].b;
		}
		sort(k,k+n,myf);
		for (i=0;i<n;i++)
		{
			for (j=0;j<min(n,i+1);j++)
			{
				if (i==0)
				{
					arr[i][j]=k[i].a+(j)*k[i].b;					
				}
				else if (j==0)
				{
					arr[i][j]=max(arr[i-1][j],k[i].a+(j)*k[i].b);
				}
				else if (j==i)
				{
					arr[i][j]=arr[i-1][j-1]+k[i].a+(j)*k[i].b;
				}
				else
				{
					arr[i][j]=max(arr[i-1][j],arr[i-1][j-1]+k[i].a+(j)*k[i].b);
				}
			}
		}
		for (i=0;i<n;i++)
		{
			cout<<arr[n-1][i]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
