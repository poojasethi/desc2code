#include <bits/stdc++.h>
using namespace std ;
#define ft first
#define sd second  
#define ll long long 
#define pii pair<int,int>
#define mp make_pair
#define pii pair<int,int>
const int mx=5111;
ll dp[mx][mx];
bool cmp(const pii &a,const pii &b)
{
	return a.sd>b.sd;
}
int main()
{ 
	int i,t,n,j; 
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		vector<pii> cc(n+1,mp(0,0));
		for(i=1;i<=n;i++) 
		scanf("%d",&cc[i].ft);
		for(i=1;i<=n;i++) 
		scanf("%d",&cc[i].sd);
		sort(cc.begin()+1,cc.end(),cmp);
		for(i=1; i<=n; i++)
		{
			for(j=1; j<=i; j++)
			{
				dp[j][i] = 0;
			}
		}
		for(i=1; i<=n; i++)
		{
			for(j=1;j<=i;j++)
			{
				dp[j][i]=max(dp[j][i],1LL*cc[i].ft+1LL*(j-1)*cc[i].sd+dp[j-1][i-1]);
				dp[j][i]=max(dp[j][i],dp[j][i-1]);
			}
		}
		for(i=1;i<=n;i++) 
		printf("%lld ",dp[i][n]);
		printf("\n");
	}
	return 0;
}
