#include <bits/stdc++.h>
using namespace std;
typedef struct link
{
long long int a,b;
}li;
bool comp(const struct link &x,const struct link &y)
{
return x.b>y.b;
}
long long int dp[5005][5005];
li *s;
int main() {
	
	long long int t,j,i,n;
	cin>>t;
	while(t--)
	{
	cin>>n;
	s=(li *)malloc(sizeof(li)*n);
	for(i=0;i<n;i++) cin>>s[i].a;
	for(i=0;i<n;i++) cin>>s[i].b;
	sort(s,s+n,comp);
	memset(dp,sizeof(dp),0);
	for(j=1;j<=n;j++)
	{
	for(i=j;i<=n;i++)
	dp[i][j]=max(dp[i-1][j],dp[i-1][j-1]+s[i-1].a+(j-1)*s[i-1].b);
	}

for(j=1;j<=n;j++)
cout<<dp[n][j]<<" ";
cout<<endl;
}
	return 0;
}
