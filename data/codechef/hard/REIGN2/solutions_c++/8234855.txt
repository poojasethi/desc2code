#define FOR(i,a,b) for(int i=(a);i<(b);i++)
#define SZ(x) ((int)(x).size())
#define FORV(i,x) FOR(i,0,SZ(x))
#define FORD(i,a,b) for(int i=(a);i>=(b);i--)

#define ULL unsigned long long
#define LL long long
#define mset(x,v) memset(x,v,sizeof x)
#define F first
#define S second

LL dp[5010][5010];

int main()
{
	ios::sync_with_stdio(false);
	LL t,a[5010],b[5010],n;
	pair<LL,LL>  v[5010];
	cin>>t;
	while(t--)
	{
		cin>>n;
		FOR(i,0,n) cin>>v[i].S;
		FOR(i,0,n) cin>>v[i].F;
		sort(v,v+n,greater<pair<int,int> >());
		mset(dp,0);
		FOR(i,1,n+1)
		FOR(j,1,i+1) dp[i][j]=max(dp[i-1][j],dp[i-1][j-1]+v[i-1].S+v[i-1].F*(j-1));
		FOR(i,1,n+1) cout<<dp[n][i]<<" ";cout<<endl;
	}
	return 0;
}
