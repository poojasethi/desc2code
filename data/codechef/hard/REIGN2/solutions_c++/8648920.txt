#include<iostream>
#include<algorithm>
#define ll long long
using namespace std;

bool comp(pair<ll, ll>i, pair<ll, ll>j ){
	if(i.second>j.second) return true;
	return false;
}

ll dp[5000][5001];
pair<ll, ll> a[5000];

int main(){
	ll t;
	cin>>t;
	ll n;
	
	while(t--){
		cin>>n;
		for(ll i=0;i<n;i++) cin>>a[i].first;
		for(ll i=0;i<n;i++) cin>>a[i].second;
		
		sort(a, a+n, comp);
		
		
		
		for(ll i=0;i<n;i++) dp[i][0]=0;
		dp[0][1]=a[0].first;
		
		for(ll j=2;j<=n;j++){
			dp[j-1][j]=dp[j-2][j-1]+a[j-1].first+(j-1)*a[j-1].second;
		}
		
		for(ll j=1;j<=n;j++){
			for(ll i=j;i<n;i++){
				dp[i][j]=max(dp[i-1][j], dp[i-1][j-1]+a[i].first+(j-1)*a[i].second);
			}
		}
		
		for(ll i=1;i<=n;i++){
			cout<<dp[n-1][i]<<" ";
		}cout<<endl;
	}
}
