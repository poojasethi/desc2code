#include<bits/stdc++.h>
using namespace std;


typedef long long ll;
const int MAX = 5005;

ll dp[MAX][MAX];

struct input
{
	ll Ai;
	ll Bi;
};

input in[MAX];

void INITIALISE()
{
	
	for(int i = 0 ; i < MAX ; i++)
		for(int j = 0 ; j < MAX ; j++)
			dp[i][j] = 0;
}

bool comp(const input &x, const input &y)
{
	if(x.Bi == y.Bi)
		return x.Ai > y.Ai;
	return x.Bi > y.Bi;
}

void solve()
{
	INITIALISE();
	int N;
	scanf("%d",&N);

	for(int i = 0 ; i < N ; i++)
		cin>>in[i].Ai;
	for(int i = 0 ; i < N ; i++)
		cin>>in[i].Bi;

	sort(in,in+N, comp);

	for(int i = 1 ; i <= N ; i++)
	{
		for(int j = 1 ; j <= N ; j++)
		{
			if(j > i)
				break;
			dp[i][j] = max(dp[i-1][j] , dp[i-1][j-1]+in[i-1].Ai+(in[i-1].Bi*(j-1)) );
		}
	}

	for(int i = 1 ; i <= N ; i++)
		cout<<dp[N][i]<<" ";
	cout<<endl;
}

int main()
{
    int test;
    scanf("%d",&test);

    while(test--)
        solve();

	return 0;
}
