This is the bad habit that I get using Codeforces */
#include <bits/stdc++.h>
#define lli long long int
#define s(x) scanf("%lld", &x)
#define pb push_back
#define mp make_pair
#define fr first
#define sc second
#define MAXX 5005

using namespace std;

lli dp[MAXX][MAXX];
lli arr[MAXX], brr[MAXX];

vector<pair<lli, lli> > vv;

int main()
{
    lli tcase,n,i,j,k,ans,temp;

    s(tcase);

    while (tcase--) {
        vv.clear();
        s(n);

        for (i = 1; i <= n; ++i) {
            s(arr[i]);
        }

        for (i = 1; i <= n; ++i) {
            s(brr[i]);
        }

        for (i = 1; i <= n; ++i) {
            vv.pb(mp(brr[i], arr[i]));
        }


        sort(vv.rbegin(), vv.rend());

        dp[0][0] = 0;

        // dp[i][j] means max cost of picking i from strting j items


        for (i = 1; i <= n; ++i) {
            dp[i][i-1] = 0;
            for (j = i; j <= n; ++j) {
                dp[i][j] = max(dp[i][j-1],  dp[i-1][j-1] + vv[j-1].sc + (i-1) *  vv[j-1].fr);
            }
        }


        for (i = 1; i <= n; ++i) {
            printf("%lld ", dp[i][n]);
        }

        printf("\n");
    }

    return 0;
}
