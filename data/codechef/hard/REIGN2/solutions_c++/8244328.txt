#include <iostream>
#include <bits/stdc++.h>
using namespace std;
long long int dp[5001][5001];
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        int a[n+1],b[n+1];
        vector<pair<int,int> > arr;
        for(int i=1;i<=n;i++)
        {
            cin>>a[i];
        }
        for(int i=1;i<=n;i++)
        {
            cin>>b[i];
        }
        for(int i=1;i<=n;i++)
        {
            arr.push_back(make_pair(b[i],a[i]));
        }
        sort(arr.begin(),arr.end());
        reverse(arr.begin(),arr.end());
        //int dp[n+1][n+1];
        memset(dp,0,sizeof(dp));
        for(int j=1;j<=n;j++)
        {
            for(int i=j;i<=n;i++)
            {

                dp[i][j]=max(arr[i-1].second+(long long int)(j-1)*arr[i-1].first+dp[i-1][j-1],dp[i-1][j]);
            }
        }

        for(int i=1;i<=n;i++)
            cout<<dp[n][i]<<" ";
        cout<<endl;
    }
    return 0;
}
