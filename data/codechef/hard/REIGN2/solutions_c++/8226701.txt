#include<bits/stdc++.h>
using namespace std;

#define ll long long
const int N = 5e3 + 5;

struct z {
    int x, y;
    bool friend operator < (z p, z q) {
        return p.y > q.y;
    }
} a[N];

int n;
ll dp[N][N];

int main() {
    int tc;
    cin >> tc;
    for (int cs = 1; cs <= tc; cs++) {
        cin >> n;
        for (int i = 1; i <= n; i++) cin >> a[i].x;
        for (int i = 1; i <= n; i++) cin >> a[i].y;
        sort(a + 1, a + n + 1);
        memset(dp, 0, sizeof dp);
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= i; j++) {
                dp[i][j] = dp[i - 1][j];
                dp[i][j] = max(dp[i][j], dp[i - 1][j - 1] + a[i].x + (ll)(j - 1) * a[i].y);
            }
        }
        for (int i = 1; i <= n; i++) {
            if (i > 1) cout << " ";
            cout << dp[n][i];
        }
        cout << endl;
    }
    return 0;
}
