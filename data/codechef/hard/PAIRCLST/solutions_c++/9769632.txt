i64 Bigmod(i64 a,i64 b)
{
    i64 res=1;
    i64 pw=a%mod;
    while(b>0)
    {
       if(b&1)res=(res*pw)%mod;
       pw=(pw*pw)%mod;
       b/=2;
    }
    return res;
}

{

    X=x;

}

int n,m,k;
int ok[400005];
vector<pair<int,int> >g[400005];
i64 d[N];
int vis[N];
priority_queue< pair<int,int>, vector<pair<int,int> >, greater< pair<int,int> > > s;
int main()
{
   ///freopen("input.txt","r",stdin);
   s3(n,m,k);
   for(int i=1;i<=k;i++)
   {
       int a;
       scanf("%d",&a);
       ok[a]=1;
   }
   for(int i=1;i<=m;i++)
   {
       int x,y,z;
       scanf("%d %d %d",&x,&y,&z);
       g[x].pb(mp(y,z));
       g[y].pb(mp(x,z));
   }
   rep(i,1,n+10)d[i]=inf;
   i64 ans=inf+10;
    for(int i=1;i<=n;i++)
    {
        if(!ok[i])continue;

        while(!s.empty())
        {
            d[s.top().second]=inf;
            s.pop();
        }
        s.push(mp(0,i));
        d[i]=0;
        int cnt=0;
        while(!s.empty())
        {
            int v=s.top().second;
            s.pop();
            if(vis[v]==i)continue;
            cnt+=ok[v];
            vis[v]=i;
            if(d[v]>ans)
                break;
            if(ok[v] && cnt==2)
            {
                ans=min(ans,d[v]);
                break;
            }
            int len=g[v].size();
            for(int i=0;i<len;i++)
            {
                int to=g[v][i].first;
                int cst=g[v][i].second;
                if(d[v]+cst<d[to])
                {
                    d[to]=d[v]+cst;
                    s.push(mp(d[to],to));
                }
            }
            d[v]=inf;
        }

    }
    cout<<ans<<endl;


   return 0;
}
