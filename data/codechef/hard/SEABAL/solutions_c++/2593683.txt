#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<int>range[100003];
int a[100003],rig[100003];
int main()
{
    int n,m,k,ans=0,i,j,x,b;
    cin>>n>>m;
    for(i=1;i<=n;i++)cin>>a[i];
    for(i=1;i<=n+1;i++)rig[i]=i;
    for(i=0;i<m;i++){cin>>j>>b;range[j].push_back(b);}
    cin>>k;
    for(i=1;i<=n;i++)sort(range[i].begin(),range[i].end(),greater<int>());
    for(i=0;i<k;i++)
    {
                    cin>>x;
                    x+=ans;
                    a[x]--;
                    if(!a[x])
                    {
                             int b=rig[x+1];
                             for(j=x;rig[j]==x;j--)
                             {
                                                   while(range[j].size() && range[j].back()<b){range[j].pop_back();ans++;}
                                                   rig[j]=b;
                             }  
                    }
                    cout<<ans<<endl;
    }
}
