#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<int>range[100003];
int main()
{
    int n,m,k,ans=0,i,j,x,a[100003],b,right[100003];
    cin>>n>>m;
    for(i=1;i<=n;i++)cin>>a[i];
    for(i=1;i<=n+1;i++)right[i]=i;
    for(i=0;i<m;i++){cin>>j>>b;range[j].push_back(b);}
    cin>>k;
    for(i=1;i<=n;i++){sort(range[i].begin(),range[i].end(),greater<int>());}
    for(i=0;i<k;i++)
    {
                    cin>>x;
                    x+=ans;
                    a[x]--;
                    if(!a[x])
                    {
                             int b=right[x+1];
                             for(j=x;!a[j] && j>0;j--)
                             {
                                                   while(range[j].size() && range[j].back()<b){range[j].pop_back();ans++;}
                                                   right[j]=b;
                             }  
                    }
                    cout<<ans<<endl;
    }
}
