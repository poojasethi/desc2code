typedef pair<int,int> PII;
typedef pair<double, double> PDD;
typedef vector<int> VI;

#define IN(A, B, C) assert( B <= A && A <= C)

//typedef int LL;
//typedef __int64 LL;

vector<PII> V;
int a[100005];
int counts[100005], lefts[100005], rights[100005];
int info[400005];
int n, m, ans;

int Left(int at) { if(at == lefts[at]) return at; return lefts[at] = Left(lefts[at]); }
int Right(int at) { if(at == rights[at]) return at; return rights[at] = Right(rights[at]); }

void build(int at, int idl, int idr)
{
	if(idl == idr)
	{
		info[at] = V[idl].second;
		return;
	}

	int mid = (idl + idr)/2;
	build(at*2, idl, mid);
	build(at*2 + 1, mid + 1, idr);

	info[at] = MIN(info[at*2], info[at*2 + 1]);
}

int query(int at, int idl, int idr, int L, int H, int l, int r)
{
	if(H < idl || L > idr) return 0;
	if(L <= idl && idr <= H && info[at] > r) return 0;

	if(idl == idr)
	{
		V[idl].second = 1000000000;
		info[at] = 1000000000;
		return 1;
	}

	int mid = (idl + idr)/2;

	if(query(at*2, idl, mid, L, H, l, r)) 
	{
		info[at] = MIN(info[at*2], info[at*2 + 1]);
		return 1;
	}

	if(query(at*2 + 1, mid + 1, idr, L, H, l, r)) 
	{
		info[at] = MIN(info[at*2], info[at*2 + 1]);
		return 1;
	}

	return 0;
}

void process(int at)
{
	if(at && a[at-1]==0) lefts[at] = at - 1;
	if(at + 1 < n && a[at + 1]==0) lefts[at + 1] = at;

	if(at + 1 < n && a[at + 1]==0) rights[at] = at + 1;
	if(at && a[at - 1]==0) rights[at - 1] = at;

	int l, r;

	l = Left(at);
	r = Right(at);

	int low, hi;

	low = (l == 0 ? 0 : counts[l - 1]);
	hi  = counts[r] - 1;

	while( query(1, 0, m - 1, low, hi, l, r) )
		ans++;
}

int main()
{
	int i, l, r, k, pos;

	scanf("%d %d", &n, &m);
	IN(n, 1, 100000);
	IN(m, 1, 100000);

	for(i = 0; i < n; i++)
	{
		scanf("%d", &a[i]);
		IN(a[i], 1, 100000);
	}

	for(i = 1; i <= m; i++)
	{
		scanf("%d %d", &l, &r);
		IN(l, 1, r);
		IN(r, l, n);

		l--;
		r--;
		counts[l]++;
		
		V.push_back( PII(l, r) );
	}

	sort(V.begin(), V.end());

	for(i = 1; i <= n; i++)
		counts[i] += counts[i - 1];

	for(i = 0; i < n; i++)
		lefts[i] = rights[i] = i;

	build(1, 0, m - 1);
	ans = 0;
	for(i = 0; i < n; i++)
		if(!a[i])
			process(i);

	scanf("%d", &k);
	while(k--)
	{
		scanf("%d", &pos);
		pos--;
		pos += ans;

		IN(pos, 0, n - 1);
		a[pos]--;

		if(a[pos])
		{
			printf("%d\n", ans);
			continue;
		}

		process(pos);
		printf("%d\n", ans);
	}

	return 0;
}
