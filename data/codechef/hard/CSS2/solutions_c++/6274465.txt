#include<cstdio>
#include<map>

using namespace std;

struct slog{int val, priority;};

int val, priority, N, M;
long long p, id, attr;
map<long long, slog> mapa;

int main() {
    scanf("%d %d", &N, &M);
    while(N--) {
        scanf("%lld %d %d %d", &id, &attr, &val, &priority);
        p=(id<<20)+attr;
        if(mapa[p].priority<=priority) {
            mapa[p].priority=priority;
            mapa[p].val=val;
        }
    }
    while(M--) {
        scanf("%lld %d", &id, &attr);
        printf("%d\n", mapa[(id<<20)+attr].val);
    }
    return 0;
}
