#include<cstdio>
#include<vector>
using namespace std;
struct node
{
    int attr;
    int priority;
    int value;
};
int main()
{
    vector<vector<node> >v(1000000);
    struct node temp;
    int n,m,a;
    scanf("%d%d",&n,&m);
    while(n--)
    {
        scanf("%d%d%d%d",&a,&temp.attr,&temp.value,&temp.priority);
        v[a].push_back(temp);
        
    }
    while(m--)
    {
      int a,b,last=0,ans=0,i;
      scanf("%d%d",&a,&b);
      for(i=0;i<v[a].size();i++)
      {
        if(v[a][i].attr==b)
        {
            if(v[a][i].priority>=last)
            {
                last=v[a][i].priority;
                ans=v[a][i].value;
            }
        }
      }
      printf("%d\n",ans);
    }
    return 0;
}
