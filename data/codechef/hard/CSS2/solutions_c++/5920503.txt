#define     TEST            int _t; s(_t); while(_t--)

typedef     pair<ll,ll> pii;
typedef struct node
{
    int attr;
    int val;
    int pr;
}node;

int main()
{
    vector<vector<node> >v(1000001);
    int n,m;
    s(n);
    s(m);
    while(n--)
    {
        int a;
        struct node temp;
        scanf("%d%d%d%d",&a,&temp.attr,&temp.val,&temp.pr);
        v[a].push_back(temp);
    }
    while(m--)
    {
        int a,b,ans,last=0;
        s(a);
        s(b);
        for(int i=0;i<v[a].size();i++)
        {
            if(v[a][i].attr==b)
            {
                if(v[a][i].pr>=last)
                {
                    last=v[a][i].pr;
                    ans=v[a][i].val;
                }
            }
        }
        cout<<ans<<endl;
    }
return 0;
}
