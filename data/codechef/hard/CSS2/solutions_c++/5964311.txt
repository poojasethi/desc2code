#include <iostream>
#include <cstdio>
#include <map>
#include <utility>

using namespace std;

typedef long long ll;

int main()
{
	ll id, attr, val, pr, n, m;

	map<ll, map<ll, pair<ll, ll> > > hash;
	
	scanf("%lld %lld", &n, &m);

	for (int i=0; i<n; i++) {
		scanf("%lld %lld %lld %lld", &id, &attr, &val, &pr);

		if (hash[id][attr].first > pr) continue;
		else hash[id][attr]=make_pair(pr, val);
	}

	for (int i=0; i<m; i++) {
		scanf("%lld %lld", &id, &attr);
		printf("%lld\n", hash[id][attr].second);
	}

	return 0;
}
