bool less1(int a[],int b[])
{
	if(a[0]!=b[0])return a[0]<b[0];
	for(int i=a[0];i>=1;i--)
	if(a[i]!=b[i])return a[i]<b[i];
	return 0;
}
void sum(int a[],int b[],int c[])
{

	memset(c,0,sizeof(c));
	c[0]=max(a[0],b[0]);

	{
		x+=a[i]+b[i];
		c[i]=x%md;x/=md;
	}
	while(x>0)
	{
		c[0]++;
		c[c[0]]=x%md,x/=md;
	}
}
void sub(int a[],int b[])
{

		a[i]-=b[i];
	}
	while(a[0]>1&&a[a[0]]==0)a[0]--;
}
