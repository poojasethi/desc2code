from math import *

def B(n,k):
	if ((k>n) or (k<0)):
		return 0
	return factorial(n)/(factorial(k)*factorial(n-k))

def S(m,l,k):
 	global ans
	if (l==0):
		return
	b=B(l-1,k)
	if (m<=b):
		ans+='0'
		S(m,l-1,k)
	else:
		ans+='1'
		S(m-b,l-1,k-1)

t=input('')
for i in range(0,t):
	ans=""
	n,k,m=raw_input('').split()
	n,k,m=[int(n),int(k),int(m)]
	if (B(n,k)>=m):
		S(m,n,k)
		print ans
	else:
		print -1
