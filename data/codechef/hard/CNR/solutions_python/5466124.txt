import math

def comb(n,k):
	return math.factorial(n)/(math.factorial(k) * math.factorial(n-k))

test=int(raw_input())
while test:
	test=test-1
	s=raw_input()
	n,k,m=(long(x) for x in s.split())
	res=''
	if comb(n,k)<m:
		print -1
		continue
			
	for val in range(0, n):
		if (n-(val+1)-k)>=0:
				c = comb(n-(val+1), k)
	    
		else:
				c = 0

		if c<m:
			res = res + '1'
			k = k-1
			m = m-c
		else:
			res = res + '0'
		
	print res
