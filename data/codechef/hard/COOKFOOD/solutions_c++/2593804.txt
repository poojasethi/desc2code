const Lint mod = 1000000007LL;

const int MAXN = 200010;

Lint Pow(Lint a,Lint b){
	
	Lint R=1LL;
	
	while(b){
		
		if(b&1)	R=(R*a)%mod;
		
		b>>=1;
		a=(a*a)%mod;
		
	}
	
	return R;
	
}
