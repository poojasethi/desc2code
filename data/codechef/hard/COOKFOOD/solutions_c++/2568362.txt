#include<iostream>
#define M 1000000007
#define llu long long int
using namespace std;
llu pow(llu a,llu b)
{
    llu y=a,ans=1;
    while(b!=0)
    {
               llu d=b%2;
               b=b/2;
               if(d)ans=(ans*y)%M;
               y=(y*y)%M;
    }
    return ans;
}
int main()
{
    llu t,n,k,ans,x;
    cin>>t;
    while(t--)
    {
              cin>>n>>k;
              ans=(k*(k-1));
              ans=ans%M;
              x=k*k-3*k+3;
              x=x%M;
              ///cout<<ans<<" "<<x<<endl;
              if(k==0)cout<<1<<endl;
              else
              {
                  cout<<(pow(x,n-1)*ans)%M<<endl;
              }
   }
}
