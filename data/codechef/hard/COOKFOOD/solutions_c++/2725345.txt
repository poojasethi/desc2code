#include<iostream>
#define ll long long
#define mod 1000000007
using namespace std;
ll power(ll a,ll b)
{
    ll c;
    if(b==0)
        return 1;
    else if(b==1)
        return a;
    else
    {
        c=power(a,b/2);
        if(b%2)
            return (((c*c)%mod)*a)%mod;
        else
            return ((c*c)%mod);
    }
}
int main(void)
{
    ll n,k,i,j,l,t;
    cin>>t;
    while(t--)
    {
        cin>>n>>k;
        j=(k*(k-1))%mod;
        l=power(k-1,2)-(k-2);
        if(l<=0)
            l+=mod;
        i=power(l,n-1);
        j=(j*i)%mod;
        cout<<j<<endl;
    }
    return 0;
}
