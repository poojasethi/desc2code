LL powmod(LL a,LL b)
{
    if(b==0)return 1;
    LL x=powmod(a,b/2);
    LL y=(x*x)%MOD;
    if(b%2)
        return (a*y)%MOD;
    return y%MOD;
}
int main()
{
    int t;
    cin>>t;
    while(t--) {
        LL n,k;
        cin>>n>>k;
        LL ans=0;
        ans=(k*(k-1))%MOD;
        LL vl=((k-1)*(k-1))%MOD;
        vl=(vl-(k-2)+MOD)%MOD;
        ans=(ans*powmod(vl,n-1))%MOD;
        cout<<ans<<endl;
    }
return 0;
}
