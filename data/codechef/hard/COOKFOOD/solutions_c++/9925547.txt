# include <bits/stdc++.h>
# define mp make_pair
using namespace std;
typedef long long ll;
ll mod = 1e9+7;
ll pw(ll a, ll b) {
    ll res=1;
    while(b) {
       if(b&1) res=(res*a)%mod;
       a=(a*a)%mod;
       b>>=1;
    }
    return res%mod;
}

main () {
	int t;
	scanf("%d",&t);
	while (t--) {
		ll n,k;
		scanf("%lld%lld",&n,&k);
		ll ans = (k*(k-1))%mod;
		ll ans1 = ((k-1)*(k-1) - (k-2) + mod)%mod;
		ans1 = pw(ans1,n-1);
		printf("%lld\n",(ans*ans1)%mod);
	}
}
