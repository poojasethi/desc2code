#include <iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#define M 1000000007
#define LL long long

using namespace std;

LL f(LL n, LL k){
    LL ret = 1, bs = n;
    while(k > 0){
        if(k&1) ret = ret * bs %M;
        bs = bs*bs %M;
        k >>= 1;
    }
    return ret;
}

int main(){
    LL t,n,k;
    scanf("%lld", &t);
    while(t--){
        scanf("%lld%lld", &n,&k);
        LL p1 = k*(k-1) % M;
        LL p2 = f((k*k - 3*k + 3) %M, n-1);
        
        printf("%lld\n", p1*p2%M);
    }
    return 0;   
}
