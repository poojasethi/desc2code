unsigned long long convertToDecimalInt(vector<unsigned long long>
num,int base){
        unsigned long long ret=0;
        for(int i=0;i<num.size();i++){
                ret+=num[i]*basePower[base][i];
        }
        return ret;
}

vector<unsigned long long> convertToIntVector(string str,int &minbase){
        vector<unsigned long long> ret(str.length(),0);
        minbase=0;
        int dec;
        for(int i=0,j=str.length()-1;i<str.length();i++,j--){

                if( isalpha(str[i])){
                        ret[j]=10+str[i]-'a';
                }
                else {
                        ret[j]=str[i]-'0';
                }
                if (minbase<ret[j])minbase=ret[j];
        }
        return ret;
}
