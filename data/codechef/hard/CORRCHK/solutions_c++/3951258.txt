using namespace std;


int T, value[300];
char s[30], As[10], Bs[10], Cs[10];

int main(){
    scanf("%d", &T);

    for(char c = '0'; c <= '9'; c++)
        value[c] = c - '0';
    for(char c = 'a'; c <= 'f'; c++)
        value[c] = c - 'a' + 10;
    
    int res = 0;
    while(T--){
        scanf("%s", s);
        
        int min_base = 2;
        for(int i = 0; s[i]; i++)
            if((s[i] != '+') && (s[i] != '='))min_base = max(min_base, value[s[i]]);
        
        int i = 0, j = 0;
        while(s[i] != '+'){
            As[j] = s[i];
            i++; j++;
        }
        As[j] = 0;
        
        i++;
        j = 0;
        
        while(s[i] != '='){
            Bs[j] = s[i];
            i++; j++;
        }
        Bs[j] = 0;
        
        i++;
        j = 0;
        while(s[i]){
            Cs[j] = s[i];
            i++; j++;
        }
        Cs[j] = 0;
        
        bool ok = false;
        for(int Q = min_base + 1; Q <= 16; Q++){
            long long A = 0, e = 1;
            for(int i = strlen(As) - 1; i >= 0; i--){
                A += e * value[As[i]];
                e *= Q;
            }
            long long B = 0;
            e = 1;
            for(int i = strlen(Bs) - 1; i >= 0; i--){
                B += e * value[Bs[i]];
                e *= Q;
            }
            long long C = 0;
            e = 1;
            for(int i = strlen(Cs) - 1; i >= 0; i--){
                C += e * value[Cs[i]];
                e *= Q;
            }
            if(A + B == C){
                ok = true;
                break;
            }
        }
        res += ok;
    }
    printf("%d\n", res);
    
    return 0;
}
