#include<cstring>
#include<iostream>
using namespace std;

int val(char a)
{
if(a>='0'&&a<='9')
return a-'0';
return a-'a'+10;
}
long long int convert(char *a,long long int base)
{
long long int ans=0;
for(int i=0;a[i]!='\0';i++)
ans=ans*base+val(a[i]);
return ans;
}
int func(char *a,char *b,char *c)
{
int beg=0;
for(int i=0;a[i]!='\0';i++)
beg=max(beg,val(a[i]));
for(int i=0;b[i]!='\0';i++)
beg=max(beg,val(b[i]));
for(int i=0;c[i]!='\0';i++)
beg=max(beg,val(c[i]));
return beg+1;
}

int main()
{
ios_base::sync_with_stdio(false);
int t,ans=0,i=0,r=0;
cin>>t;
while(t--)
{
i=0,r=0;
char a[15]={'\0'},b[15]={'\0'},c[15]={'\0'},d,s[100]={'\0'};
cin>>s;
//cout<<s<<endl;
for(i=0;;i++)
{
if(s[i]=='+')
break;
a[i]=s[i];
}
i++;
for(;;i++)
{
if(s[i]=='=')
break;
b[r++]=s[i];
}
i++;
r=0;
for(;s[i]!='\0';i++)
c[r++]=s[i];
//cout<<a<<" "<<b<<" "<<c<<endl;
int beg=func(a,b,c);
for(i=beg;i<=16;i++)
{
if(convert(a,i)+convert(b,i)==convert(c,i))
{
ans++;
break;
}
}
}
cout<<ans<<endl;
return 0;
}
