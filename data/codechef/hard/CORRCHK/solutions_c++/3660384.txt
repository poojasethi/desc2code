int base(char *str)
{
	
	int soln=0,val,i;
	//determing highest possible base of number
	for(i=0;str[i]!='\0';i++)
	{
		if(str[i]=='a')
		val=10;
		else if(str[i]=='b')
		val=11;
		else if(str[i]=='c')
		val=12;
		else if(str[i]=='d')
		val=13;
		else if(str[i]=='e')
		val=14;
		else if(str[i]=='f')
		val=15;
		else 
		val=str[i]-'0';
		if(soln<val)
		soln=val;
	}
	return soln+1;
	
}
long long int convert_to_decimal(char *str2,int b,int n)
{
	long long int ans=0;
	int val,i;
	for(i=n-1;i>=0;i--)
	{
	if(str2[i]=='a')
	val=10;
	else if(str2[i]=='b')
	val=11;	
	else if(str2[i]=='c')
		val=12;
		else if(str2[i]=='d')
		val=13;
		else if(str2[i]=='e')
		val=14;
		else if(str2[i]=='f')
		val=15;
		else 
		val=str2[i]-'0';
      ans=ans+(long long int)pow(b,n-1-i)*val;
      
	}
	return ans;
}
