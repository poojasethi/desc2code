#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <string>

using namespace std;

const int MAXN = 300;
typedef long long int llint;

int t;
char state[MAXN];
int rjesenje = 0;

int pretvori(char znak){
     if(znak >= '0' && znak <= '9') return znak - '0';
     return znak - 'a' + 10;
}

llint pretvori2(string broj, int baza){
      llint ret = 0;
      for(int i = 0; i < broj.size(); i++)
         ret = ret * baza + (pretvori(broj[i]));
      return ret;
}


int main(){
    int j;
    scanf("%d", &t);

    for(int i = 0; i < t; i++){
              string A = "", B = "", C = "";

              cin >> state;
              int len = strlen(state);

              for(j = 0; state[j] != '+'; j++)
                  A += state[j];
              for(++j; state[j] != '='; j++)
                  B += state[j];
              for(++j; j < len; j++)
                  C += state[j];

              int najveca = 0;
              for(int x = 0; x < len; x++){
                    if(state[x] == '+' || state[x] == '=') continue;
                    najveca = max(najveca, pretvori(state[x]) + 1);
              }

              int moze = 0;
              for(int b = najveca; b <= 16; b++){
                   if(pretvori2(A, b) + pretvori2(B, b) == pretvori2(C, b)) moze = 1;
              }

              rjesenje += moze;
    }

    printf("%d", rjesenje);

    return 0;
}
