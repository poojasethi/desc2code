#include <iostream>
#include <cstdio>
#include <vector>
#include <string.h>
#include <map>
using namespace std;
long long int t,res;
vector<char>a,b,c;
char in[100];
map<char,int>mapi;
long long int Solve(vector<char>s,int base)
{
    long long int p=1 , ret=0;
    for(int i=s.size()-1;i>=0;i--)
        ret+=(mapi[s[i]]*p) , p*=base;
    return ret;
}
int F(vector<char>s)
{
    int mx=0;
    for(int i=0;i<s.size();i++)
        mx=max(mx,mapi[s[i]]);
    return mx;
}
void Do_Map()
{
    int val=0;
    for(char i='0';i<='9';i++)  mapi[i]=val++;
    for(char i='a';i<='f';i++)  mapi[i]=val++;
}
int main()
{
    //freopen("r.txt","r",stdin);
    Do_Map();
    scanf("%lld",&t);
    for(int u=0;u<t;u++)
    {
        int i=0 ; a.clear() , b.clear(), c.clear() ;
        char in[100];
        scanf("%s",in);
        while(in[i]!='+') a.push_back(in[i++]);i++;
        while(in[i]!='=') b.push_back(in[i++]);i++;
        while(i<strlen(in)) c.push_back(in[i++]);
        int st=max( max(F(a),F(b)) , F(c));
        //printf("%d ",st);
        for(int base=st+1;base<=16;base++)
        {
            long long int A,B,C;
            A=Solve(a,base) ,B=Solve(b,base) , C=Solve(c,base);
            if(A+B==C)
            {
                res++;
                break;
            }
        }
    }
    printf("%lld",res);

    return 0;
}
