#include <bits/stdc++.h>
using namespace std;

char s[110];

int chg(char c)
{
    if(c >= '0' && c <= '9') return c - '0';
    return c - 'a' + 10;
}

bool check(int base)
{
    long long x = 0, y = 0,z = 0;
    int i = 0;
    while(s[i] != '+') {
        int xx = chg(s[i]);
        if(xx >= base) return false;
        x = x * base + xx;
        i ++;
    }
    i ++;
    y = x;
    x = 0;
    while(s[i] != '=') {
        int xx = chg(s[i]);
        if(xx >= base) return false;
        x = x * base + xx;
        i ++;
    }    
    i ++;
    z = x;
    x = 0;
    while(s[i]) {
        int xx = chg(s[i]);
        if(xx >= base) return false;
        x = x * base + xx;
        i ++;
    }
    return x == y + z;
}

int main()
{
    int t;
    scanf("%d",&t);
    int ans = 0;
    while(t --) {
        scanf("%s",s);
        for(int i = 2; i <= 16; i ++)
            if(check(i)) {
                ans ++;
                break;
            }
    }
    cout << ans << endl;
    return 0;
}
