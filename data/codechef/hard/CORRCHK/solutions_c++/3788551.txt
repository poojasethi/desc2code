#include <bits/stdc++.h>

#define st first
#define nd second
#define mp make_pair
#define pb push_back
#define lli long long int
#define all( gg )	gg.begin(),gg.end()
#define foreach( gg,itit )	for( typeof(gg.begin()) itit=gg.begin();itit!=gg.end();itit++ )
#define FP( ii,aa,bb ) for( lli ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb ) for( lli ii=aa;ii>=bb;ii-- )

using namespace std;

char s[100];
string a,b,c;

lli f( char c ){
	if( c<='9' )	return c-'0';
	return c-'a'+10;
}

lli toten( string s,lli q ){
	lli r=0;
	foreach( s,it )
		r = r*q+f(*it);
	return r;
}

lli ctrl( string a,string b,string c ){
	lli mini=2;
	foreach( a,it )	mini = max( mini,f(*it)+1 );
	foreach( b,it )	mini = max( mini,f(*it)+1 );
	foreach( c,it )	mini = max( mini,f(*it)+1 );
	FP( i,mini,16 )
		if( toten( a,i )+toten( b,i )==toten( c,i ) )	return 1;
	return 0;
}

int main(){
	lli t,r=0;
	cin >> t;
	while( t-- ){
		scanf("%s",s+1);
		a.clear();
		b.clear();
		c.clear();
		lli ind=1;
		while( s[ind]!='+' )	a += s[ind++];
		ind++;
		while( s[ind]!='=' )	b += s[ind++];
		ind++;
		while( s[ind] )	c += s[ind++];
		r += ctrl( a,b,c );
	}
	cout << r << endl;
}
