#include <iostream>
#include <cstdio>
#include <map>
using namespace std;
map<char, int> m;
long long get(string s, int base)
{
        long long pow = 1;
        long long val = 0;
        for(int i = s.size()-1; i >= 0; i--) {
            val = val + m[s[i]] * pow;
            pow = pow * base;
        }
        return val;
}

int main()
{
    int t;

    cin >> t;
    char c1[100], c2[100], c3[100];
    for(int i = 0; i <= 9; i++) {
        m[(char)(i+'0')] = i;
    }
    for(int i = 0; i < 6; i++) {
        m[(char)(i+'a')] = i+10;
    }
    int c = 0;
    while(t--) {
        string ss;
        cin >> ss;
        string s1, s2, s3;
        int base = 1;
        s1 = s2 = s3 = "";
        int pos = 0;
        while(ss[pos]!='+') {
                pos++;
                s1 += ss[pos-1];
                base = max(base, m[ss[pos-1]]);
        }
        pos++;
        while(ss[pos]!='=') {
                pos++;
                s2 += ss[pos-1];
                base = max(base, m[ss[pos-1]]);
        }
        pos++;
        while(pos < ss.size()) {
                pos++;
                s3 += ss[pos-1];
                base = max(base, m[ss[pos-1]]);
        }
        //pos++;

        for(int i = base+1; i <= 16; i++) {
            long long v1, v2, v3;
            v1 = get(s1, i);
            v2 = get(s2, i);
            v3 = get(s3, i);
            if(v1+v2 == v3) {
                c++;
                break;
            }
        }
    }
    cout << c << endl;
}
