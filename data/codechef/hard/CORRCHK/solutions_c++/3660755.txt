#define sz(x) (int)(x).size()

typedef long long ll;

int n;
string s, a, b, c;

int get(char x) {
 return (x <= '9') ? x - '0' 
                   : x - 'a' + 10;
}

ll tran(string x,ll base) {
 ll res = 0;
 for (int i = 0; i < sz(x); ++i) {
  res = res * base + get(x[i]);
 } 
 return res;
}

int main() {

 cin >> n;
 int ans = 0;
 while (n--)  {
  cin >> s;                                        
  bool pl = 0, eq = 0; 
  a = b = c = "";
  int mx = 0;
  for (int i = 0; i < sz(s); ++i) {
   if (s[i] != '+' && s[i] != '=')
    mx = max (mx, get(s[i]) + 1);

   if (s[i] == '=') eq = 1;
   else if (s[i] == '+') pl = 1;
   else if (eq) c += s[i];
   else if (pl) b += s[i];
   else a += s[i]; 
  }
  bool good = 0;
  for (int i = mx; i <= 16; ++i) {
   if (tran(a, i) + tran(b, i) == tran(c, i)) good = 1;
  }
  ans += good;
 }
 printf ("%d", ans);

 return 0;
}
