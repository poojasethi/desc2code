#include <bits/stdc++.h>
using namespace std;

long long int func(char a[], int base) {
	int i;
	long long int t=0;
	for(i=0;i<strlen(a);i++) {
		if((a[i]>='0' && a[i]<='9') && (a[i]-'0')<base){
			t= t *base + a[i]-'0';
		}
		else if ((a[i]>='a' && a[i]<='f') && a[i]-'a' + 10 <base) t = t*base + a[i]-'a'+10;
		else return 0;
	}
	return t;
}
int main()
{
	int t,e,f,g,d,ans=0,i;
	scanf("%d",&t);
	while(t--) {
		d=0;
		e=0;
		f=0;
		g=0;
		char a[10],b[10],c[10],x[100];
		scanf("%s",x);
		for(i=0;i<strlen(x);i++) {
			if(x[i]=='+' || x[i]=='=') d++;
			else if(d==0) a[e++]=x[i];
			else if(d==1) b[f++]=x[i];
			else if(d==2) c[g++]=x[i];
		}
		a[e]='\0';b[f]='\0';
		c[g]='\0';
		for(i=2;i<=16;i++) {
			if((func(a,i) + func(b,i)==func(c,i)) && func(a,i)!=0 && func(b,i)!=0 && func(c,i)!=0) {
				ans++;
				break;
			}
		}
	}
	printf("%d\n",ans);
}
