#include<cstdio>
#include<climits>
#define INF LLONG_MAX;
long long int min(long long int a,long long int b)
{
    if(a<b)
        return a;
    return b;
}
long long int max(long long int a,long long int b)
{
    if(a>b)
        return a;
    return b;
}
int main(void)
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,m,i,j,k;
        scanf("%d %d",&n,&m);
 
        long long int cost[n][m];
        long long int discount[n][m];
        long long int mini=INF;
 
        for(i=0;i<n;++i)
            for(j=0;j<m;++j)
            scanf("%lld",&cost[i][j]);
 
        for(i=0;i<n;++i)
            for(j=0;j<m;++j)
            scanf("%lld",&discount[i][j]);
        long long int dp[n][m];
 
        for(i=1;i<n;++i)
            for(j=0;j<m;++j)
            dp[i][j]=INF;
 
        for(i=0;i<m;++i)
        {
            dp[0][i]=cost[0][i];
            mini=min(dp[0][i],mini); //mini stores the minimum cost of the first item here
        }
 
        for(i=1;i<n;++i)
        {
            long long int mini2=INF;
            for(j=0;j<m;++j)
            {
                long long int temp=max(0,cost[i][j]-discount[i-1][j]);
                dp[i][j]=min(dp[i-1][j]+temp,mini+cost[i][j]);
                mini2=min(dp[i][j],mini2);
            }
            mini=mini2;
        }
 
        long long int ans=INF;
        for(i=0;i<m;++i)
            ans=min(ans,dp[n-1][i]);
        printf("%lld\n",ans);
    }
    return 0;
}
