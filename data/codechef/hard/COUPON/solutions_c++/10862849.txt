#include<bits/stdc++.h>
using namespace std;
#define LL int long long
vector <vector<LL> > dp;
vector <vector<LL> > price;
vector <vector<LL> > disc;
LL n,m;

int main()
{
	int t;
    scanf("%d",&t);
	while(t--)
	{
      scanf("%lld %lld",&n,&m);
      int i,j;
      dp.clear();
      disc.clear();
      price.clear();
      dp.resize(n);
      disc.resize(n);
      price.resize(n);
      for(i=0;i<n;i++)
      {
      	dp[i].resize(m);
      	price[i].resize(m);
      	disc[i].resize(m);
      }
      for(i=0;i<n;i++)
      {
      for(j=0;j<m;j++)
      {
      scanf("%lld",&price[i][j]);
      }
      }
       for(i=0;i<n;i++)
      {
      for(j=0;j<m;j++)
      {scanf("%lld",&disc[i][j]);
      }
      }
      LL curmin=(1LL<<44LL);
      for(i=0;i<m;i++)
      {
         curmin=min(curmin,price[0][i]);
         dp[0][i]=price[0][i];
      }
      for(i=1;i<n;i++)
      {
      for(j=0;j<m;j++)
      {
      dp[i][j]=min(curmin+price[i][j],dp[i-1][j]+max(0LL,price[i][j]-disc[i-1][j]));
      }
      curmin=(1LL<<44LL);
      for(j=0;j<m;j++)
      {
      curmin=min(curmin,dp[i][j]);
      }
      }
      printf("%lld\n",curmin);
	}
	return 0;
}
