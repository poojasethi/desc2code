#include <iostream>
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <queue>
#include <algorithm>
#define lli long long int
#define pb push_back
#define mod 1000000007
#define pii pair<int,int>

using namespace std;
vector<lli> cost[100009],dis[100009];
lli n,m;
lli dp[2][100009];

int main()
{
    lli t,i,j,k,a,b;
    scanf("%lld",&t);
    while(t--){
        scanf("%lld%lld",&n,&m);
        for(i=0;i<=n;i++){
            cost[i].clear();
            dis[i].clear();
        }
        for(i=0;i<n;i++){
            for(j=0;j<m;j++){
                scanf("%lld",&a);
                cost[i].pb(a);
            }
        }

        for(i=0;i<n;i++){
            for(j=0;j<m;j++){
                scanf("%lld",&a);
                dis[i].pb(a);
            }
        }

        lli minm=1e15;
        for(i=0;i<m;i++){
            dp[0][i]=cost[0][i];
            minm= min(minm,cost[0][i]);
        }

        for(i=1;i<n;i++){
            a=i%2;
            lli temp=1e15;
            for(j=0;j<m;j++){
                lli newcost= max(0ll,cost[i][j]- dis[i-1][j]);
                dp[a][j]=min(newcost + dp[1-a][j] , minm + cost[i][j]);
                temp=min(temp,dp[a][j]);
            }
            minm=temp;

        }
        printf("%lld\n",minm);
    }

    return 0;
}
