#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll minm(ll a,ll b)
{
    return (a<b)?a:b;
}
ll maxm(ll a,ll b)
{
    return (a>b)?a:b;
}
int main()
{
    int t;
    cin>>t;
    while(t--){
   int n,m;
   ll mn,x;
    cin>>n>>m;
    int p[n][m];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
            cin>>p[i][j];
    }
    int d[n][m];
    ll dp[n][m];
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
            cin>>d[i][j];
    }
    mn=p[0][0];
    for(int j=0;j<m;j++)
    {
        dp[0][j]=p[0][j];  // cost of buying first item at any shop is just price of that item without discount
      mn=minm(mn,dp[0][j]);
    }
    for(int i=1;i<n;i++)
    {
        x=1e18;
        for(int j=0;j<m;j++)
        {
            // case 1:  ith item at same shop
                   // ans of buying till (i-1)th item at same shop +max(price-discount,free)
            // case 2: ith item at any other shop
                  // min ans till (i-1)th item + price of ith item at any shop without discount
            dp[i][j]=minm(dp[i-1][j]+maxm(0,p[i][j]-d[i-1][j]),mn+p[i][j]);
            x=minm(x,dp[i][j]);
        }
       mn=x;
    }
    cout<<mn<<endl;
    }
    return 0;
}
