#include<bits/stdc++.h>
#define bug(x) cout<<#x<<" => "<<x<<"\n"
#define pr(x)  cout<<x<<"\n"
#define sc(x)	scanf("%d",&x)
#define scl(x) scanf("%lld",&x)

using namespace std;


int main()
{
//	freopen("input.txt","r",stdin);
//	freopen("output.txt","w",stdout);
	int cases=0;
	int t;
	cin>>t;
	while(t--)
	{
		int n,m;
		cin>>n>>m;

		long long  price[n][m],discount[n][m];
		long long dp[n][m];
		memset(dp,-1,sizeof dp);
		int x;
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				cin>>price[i][j];
			}
		}

		for(int i=0;i<n;i++)
		{
			for(int j=0;j<m;j++)
			{
				cin>>discount[i][j];
			}
		}

		for(int j=0;j<m;j++)
			dp[0][j]=price[0][j];

		for(int i=1;i<n;i++)
		{
			long long int minn = 1e18;
			for(int j=0;j<m;j++)
			{
				minn=min(dp[i-1][j],minn);
			}
			for(int j=0;j<m;j++)
			{
				int p = price[i][j]-discount[i-1][j];
				p = p>0?p:0;
				dp[i][j]=min(price[i][j]+minn,p+dp[i-1][j]);

			}

		}
		long long ans = 1e18;
		for(int i=0;i<m;i++)
		{
			ans=min(ans,dp[n-1][i]);
		}
		cout<<ans<<"\n";
	}
}
