#include <bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define pii pair <ll, ll>
#define mod 1000000007
#define INF 1e17
using namespace std;
typedef long long ll;

vector<ll> c[100005],d[100005],dp[100005];

main() {
	int t,n,m,i,j,k;
	scanf("%d",&t);
	while (t--) {
		scanf("%d%d",&n,&m);
		for (i=1 ; i <= n ; i++) {
			c[i].clear();
			d[i].clear();
			dp[i].clear();
			dp[i].resize(m+1);
		}
		for (i=1 ; i <= n ; i++)
			for (j=1 ; j <= m ; j++)
			{
				if (j==1) c[i].push_back(1e7);
				int lol ; scanf("%d",&lol);
				c[i].push_back(lol);
			}
		for (i=1 ; i <= n ; i++)
			for (j=1 ; j <= m ; j++)
			{
				if (j==1) d[i].push_back(1e7);
				int lol ; scanf("%d",&lol);
				d[i].push_back(lol);
			}
		ll x = 1e17;
		for (i=1 ; i <= m ; i++) {
			dp[1][i] = c[1][i];
			x = min(x,dp[1][i]);
		}
		for (i=2 ; i <= n ; i++) {
			ll y = 1e17;
			for (j=1 ; j <= m ; j++) {
				dp[i][j] = min(dp[i-1][j] + ((c[i][j]>=d[i-1][j])?c[i][j]-d[i-1][j]:0), c[i][j] + x);
				y = min(y,dp[i][j]);
			}
			x = y;
		}
		printf("%lld\n",x);
	}
	return 0;
}
