const int MD = 1000000007;

string s;
int ans, ls, cur, __, maxp, mins;

int main() {
	sIO();
	cin.sync_with_stdio(0);
	cin >> __;
	for (int _ = 0; _ < __; ++_) {
		cin >> s;
		ls = s.length();
		cur = 0;
		for (int i = 0; i < ls; ++i)
			cur += (s[i] == '7');
		ans = cur;
		maxp = cur;
		mins = 0;
		for (int i = 0; i < ls; ++i) {
			if (s[i] == '4') ++cur; else --cur;
			ans = max(ans, cur - mins);
			mins = min(mins, cur - maxp);
			maxp = max(maxp, cur);
		}
		cout << ans << endl;
	}
	return 0;
}
