using namespace std;
int cnt1[100100];
int cnt2[100100];
char s[100100];
int f[100100];
int main()
{
#ifdef Fcdkbear
	freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
#endif
	int t;
	scanf("%d",&t);
	while (t--)
	{
		MEMS(cnt1,0);
		MEMS(cnt2,0);
		scanf("%s",&s);
		int n=strlen(s);
		FOR(i,0,n)
		{
			if (i)
				cnt1[i]=cnt1[i-1];
			if (s[i]=='4')
				cnt1[i]++;
		}
		for (int i=n-1; i>=0; --i)
		{
			cnt2[i]=cnt2[i+1];
			if (s[i]=='7')
				cnt2[i]++;
		}
		int res=0;
		if ((cnt1[n-1]==n) || (cnt2[0]==n))
			res=n;
		int cur=0;
		FOR(i,0,n)
		{
			int k=0;
			if (i)
				k=f[i-1];
			f[i]=MAX(k,i+1-cnt1[i]-cnt1[i]+cur);
			if (s[i]=='4')
				res=MAX(res,cnt1[i]+cnt2[i]+f[i]);
			cur=MAX(cur,cnt1[i]-(i+1-cnt1[i]));
		}
		cout<<res<<endl;
	}
	return 0;
}
