void solve() {
  string s;
  cin>>s;
  int dp[4]={0,0,0,0};
  int n = s.size();
  for(int i=0; i<n; i++) {
    if(s[i]=='4')
      dp[0]++ , dp[2]++;
    else 
      dp[1]++ , dp[3]++;
    for(int j=0; j<3; j++)
      dp[j+1] = max(dp[j+1],dp[j]);
  }
  cout<<dp[3]<<endl;

}


int main() {
  int t;
  cin>>t;
  for(int i=0; i<t; i++)solve();

  return 0;
}
