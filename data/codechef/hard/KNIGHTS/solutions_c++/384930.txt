// Maximum Bipartite Matching Using Network Flows in O(V.E).
struct Graph
{
	vector < vector <int> > adj, cap, res;
	int source, sink, nodes, L, R;
	vector <int> parent;
	void initialize(int L, int R) {
		nodes = L + R + 2;
		this -> L = L; this -> R = R;
		adj.resize(nodes);
		cap.resize(nodes, vector <int> (nodes, 0));
		source = nodes - 2; sink = nodes - 1;
	}
	void addEdge(int u, int v) {
		v = v + L;
		adj[u].push_back(v);
		adj[v].push_back(u);
		cap[u][v] = 1;
	}
	bool bfs() {
		queue <int> q; q.push(source);
		vector <int> marked(nodes, false);
		marked[source] = true;
		parent.clear(); parent.resize(nodes, -1);
		while (!q.empty() && marked[sink] == false) {
			int u = q.front(); q.pop();
			for (int i = 0; i < size(adj[u]); ++i) {
				int v = adj[u][i];
				if (marked[v] == false && res[u][v] > 0)
					marked[v] = true, parent[v] = u, q.push(v);
			}
		}
		return marked[sink];
	}
	int matching() {
		int totalFlow = 0;
		for (int i = 0; i < L; ++i)
			addEdge(source, i - L);
		for (int i = 0; i < R; ++i)
			addEdge(i + L, sink - L);
		res = cap;
		for (bool hasPath = bfs(); hasPath == true; hasPath = bfs()) {
			++totalFlow;
			for (int v = sink; v != source; v = parent[v])
				res[parent[v]][v]--, res[v][parent[v]]++;
		}
		return totalFlow;
	}
};

struct Solver
{
	vector <string> vs; int rows, cols;
	inline int toEdge(int u, int v) {
		return (cols * u + v) >> 1;
	}
	inline bool valid(int u, int v) {
		return u >= 0 && v >= 0 && u < rows && v < cols && vs[u][v] != '#';
	}
	int solve(int rows, int cols, vector <string>& v_s) {
		vs = v_s; this -> rows = rows; this -> cols = cols;
		int black = (rows * cols) >> 1, white = (rows * cols + 1) >> 1;
		Graph graph; graph.initialize(white, black);
		int dx[] = {2, -2, 2, -2, 1, -1, 1, -1};
		int dy[] = {1, 1, -1, -1, 2, 2, -2, -2};
		for (int i = 0; i < rows; ++i)
			for (int j = 0; j < cols; ++j)
				if (((i + j + 1) & 1) && vs[i][j] != '#')
					for (int k = 0; k < 8; ++k)
						if (valid(i + dx[k], j + dy[k]))
							graph.addEdge(toEdge(i, j), toEdge(i + dx[k], j + dy[k]));
		int nodes = 0;
		for (int i = 0; i < rows; ++i)
			for (int j = 0; j < cols; ++j)
				if (vs[i][j] != '#')
					++nodes;
		int result = nodes - graph.matching();
		return result;
	}
};
