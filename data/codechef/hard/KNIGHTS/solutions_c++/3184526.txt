#define MAX 32
typedef long long int LL;
using namespace std;

int r,c,match[MAX][MAX];
char g[MAX][MAX];
bool seen[MAX][MAX];
int dx[] = {1, 1, 2, 2,-1,-1,-2,-2};
int dy[] = {2,-2, 1,-1, 2,-2, 1,-1};


inline bool valid(int i,int j)
{
	return i>=0 && i<r && j>=0 && j<c && g[i][j] != '#';
}

bool dfs(int i,int j)
{
	if(seen[i][j])		return false;
	seen[i][j] = true;
	for(int k=0;k<8;k++)
	{
		int ni,nj;
		ni=i+dx[k],nj=j+dy[k];
		if(!valid(ni,nj))	continue;
		if(match[ni][nj] == -1 || dfs(match[ni][nj] /c, match[ni][nj] %c))
		{
			match[i][j] = ni*c + nj;
			match[ni][nj] = i*c + j;
			return true;
		}
	}
	return false;	
}

bool augment(void)
{
	int i,j;
	memset(seen,0,sizeof(seen));
	for(i=0;i<r;i++)
	for(j=0;j<c;j++)
	{
		if(g[i][j] == '#')	continue;	
		if(((i+j)&1) && match[i][j] == -1 && dfs(i,j))	return true;
	}
	return false;
}	

int main()
{
	int t,i,j,b,tot;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&r,&c);
		b=tot=0;		
		for(i=0;i<r;i++)
		{
			scanf("%s",g[i]);
			for(j=0;j<c;j++)			
				if(g[i][j] == '#')	b++;		
		}
		memset(match,-1,sizeof(match));
		
		while(augment())
			tot++;	//printf("%d  %d  %d\n",r*c,b,tot);
		printf("%d\n",r*c - b - tot);
	}
	return 0;
}
