#include<cstdio>
#include<algorithm>
using namespace std;
int a[100002] , s[2500002];
    
int main() {
    int n,k,i,j,t=0;
    scanf("%d",&n);
    scanf("%d",&k);
    for(i=0;i<n;i++)
            scanf("%d",&a[i]);
    sort(a,a+n);
    for(i=1;i<n;i++) {
            for(j=0;j<n-i;j++) {
                               if(t>10*k) break;
                               else s[t++] = a[j] ^ a[j+i];
            }
    }
    sort(s,s+t);
    for(i=0;i<k;i++)
            printf("%d ",s[i]);
    return 0;                  
}
