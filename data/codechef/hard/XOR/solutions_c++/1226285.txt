#include <cstdio>
#include <algorithm>
#define maxn 100005
#define maxm 250005

using namespace std;

int a[maxn], ans[maxm << 1], n, m;

int main()
{
	//freopen("xor.in", "r", stdin);
	//freopen("xor.out", "w", stdout);
	
	scanf("%d %d", &n, &m);
	for (int i = 1; i <= n; ++i)
		scanf("%d", &a[i]);
	sort(a + 1, a + n + 1);
	unsigned int s = (1LL << 32) - 1;
	for (int i = 0; i <= 31; ++i)
	{
		long long cnt = 0;
		for (int j = 2, pre = 1; j <= n && cnt < m; ++j)
			if ((a[j] & s) == (a[j - 1] & s))
				cnt += j - pre;
			else
				pre = j;
		if (cnt >= m)
		{
			if (i != 0)
			{
				ans[0] = 0;
				for (int j = 2, pre = 1; j <= n; ++j)
					if ((a[j] & s) == (a[j - 1] & s))
						for (int k = pre; k < j; ++k)
							ans[++ans[0]] = a[k] ^ a[j];
					else
						pre = j;
				sort(ans + 1, ans + ans[0] + 1);
			}
			for (int j = 1; j <= m; ++j)
				printf("%d%c", ans[j], j == m ? '\n' : ' ');
			break;
		}
		s -= 1 << i;
	}
	
	return 0;
}
