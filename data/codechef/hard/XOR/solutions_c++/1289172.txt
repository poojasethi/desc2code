#include <iostream>
#include <cstdio>
#include <string>
#include <string.h>
#include <algorithm>
#include <math.h>
using namespace std;
typedef long long ll;

const int N = 250000;
int a[N];
int ans[N*10];
int n,m;

int main(){
    while (scanf("%d%d",&n,&m)==2){
        for (int i=1;i<=n;++i) scanf("%d",&a[i]);
        sort(a+1,a+n+1);
        int l=-1,r=31;
        ll sum=0,j=0;
        while (l<r-1){
            int mid=(l+r)>>1;
            sum=0;j=1;
            for (int i=1;i<n;++i)
                if ((a[i]^a[i+1])<(1<<mid)) ++j;
                else{
                    sum+=(j*(j-1))/2;
                    j=1;
                }
            sum+=(j*(j-1))/2;
            if (sum>=m) r=mid;
            else l=mid;
        }
        if (!r){
            while (m--) printf("0 ");
            puts("");
            continue;
        }
        sum=0;j=1;
        for (int i=2;i<=n;++i)
            if((a[i]^a[i-1])<(1<<r))
                for (int x=j;x<i;++x)
                    ans[++sum]=a[i]^a[x];
            else j=i;
        sort(ans+1,ans+sum+1);
        for (int i=1;i<=m;++i) printf("%d ",ans[i]);
        puts("");
    }
    return 0;
}
