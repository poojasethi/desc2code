#include <cstdio>
#include <cstdlib>
#include <cctype>
#include <cmath>
#include <iostream>
#include <sstream>
#include <cstring>
#include <vector>
#include<algorithm>
using namespace std;
int a[100005];
int ans[2500004];
 
int main()
{
    int i,j,k,n;
    scanf("%d%d",&n,&k);
    for(i=0;i<n;i++)scanf("%d",&a[i]);
    sort(a,a+n);
    int l=0;
    for(i=1;i<n;i++){
                     for(j=0;j+i<n;j++){if(l>10*k)break;else ans[l++]=a[j]^a[j+i];}
                     }
sort(ans,ans+l);
for(i=0;i<k;i++)printf("%d ",ans[i]);
printf("\n");  
system("pause");  
    return 0;}
