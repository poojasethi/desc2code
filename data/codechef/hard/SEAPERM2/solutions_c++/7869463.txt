#include<bits/stdc++.h>
#define MAX 310
#define mod 10000007
using namespace std;
int q[MAX][MAX],tmp[MAX],ans[MAX];
int main(){
 
    cin.sync_with_stdio(false);
    cin.tie(0);
    int t, n;
    cin >> t;
    while(t--){
        cin >> n;
        int hsh[n],chk[n];
        memset(hsh,0,sizeof hsh);
        memset(chk,0,sizeof chk);
        for(int i = 1;i <= n; ++i){
            for(int j = 1;j < n; ++j){
                cin >> q[i][j];
                hsh[i-1] = (hsh[i-1]*33 + q[i][j])%mod;
            }
        }
        sort(hsh,hsh + n);
        for(int i = 1;i <= n; ++i){
            int r = i%n + 1,po = 0,v = 0;
            for(int j = 1;j < n && po == 0; ++j)
                if(q[r][j] == 1)
                    po = j;
            for(int j = 1;j <= n; ++j){
                if(j == po){
                    tmp[j] = 1;
                    v = 1;
                }
                else
                    tmp[j] = q[i][j - v] + 1;
            }
            for(int j = 1;j <= n; ++j){
                chk[j-1] = 0;
                v = tmp[j];
                for(int k = 1;k <= n; ++k){
                    if(j != k){
                        if(tmp[k] > v)
                            chk[j-1] = (chk[j-1]*33 + tmp[k] - 1)%mod;
                        else
                            chk[j-1] = (chk[j-1]*33 + tmp[k])%mod;
                    }
                }
            }
            sort(chk,chk + n);
            int flag = 0;
            for(int j = 0;j < n && flag == 0; ++j)
                if(chk[j] != hsh[j])
                    flag = 1;
            if(flag == 0){
                    for(int j = 1;j <= n; ++j)
                        ans[j] = tmp[j];
                    break;
            }
            v = 0;
            for(int j = 1;j <= n; ++j){
                if(j == (po+1)){
                    tmp[j] = 1;
                    v = 1;
                }
                else
                    tmp[j] = q[i][j - v] + 1;
            }
            for(int j = 1;j <= n; ++j){
                chk[j-1] = 0;
                v = tmp[j];
                for(int k = 1;k <= n; ++k){
                    if(j != k){
                        if(tmp[k] > v)
                            chk[j-1] = (chk[j-1]*33 + tmp[k] - 1)%mod;
                        else
                            chk[j-1] = (chk[j-1]*33 + tmp[k])%mod;
                    }
                }
            }
            sort(chk,chk + n);
            flag = 0;
            for(int j = 0;j < n && flag == 0; ++j)
                if(chk[j] != hsh[j])
                    flag = 1;
            if(flag == 0){
                    for(int j = 1;j <= n; ++j)
                        ans[j] = tmp[j];
                    break;
            }
 
        }
        for(int i = 1;i <= n; ++i)
            cout << ans[i] << " ";
        cout << "\n";
    }
    return 0;
}
