#include<stdio.h>
#include<math.h>
#include<algorithm>
#define PI 3.1415926535897
//#define max(x,y) x>y?x:y
//#define min(x,y) x<y?x:y
#define get getchar_unlocked
using namespace std;
 
inline int scan()
{
    int n=0,s=1;
    char p=get();
    if(p=='-')  s=-1;
    while((p<'0'||p>'9')&&p!=EOF&&p!='-') p=get();
    if(p=='-') s=-1,p=get();
    while(p>='0'&&p<='9') { n = (n<< 3) + (n<< 1) + (p - '0'); p=get(); }
    return n*s;
}
 
double MaxVolume(double W, double H) 
{
    double r = min(W/PI, 2*H/3),Ans = PI/4*r*r*(H-r);
    double hp = H/(PI+1);
    double D = min(W/2, hp);
    if(2*hp-W > 0) 
    {
        double wp = W/((PI+1)*(PI+1));
        D = max(D, min(W, hp+wp-sqrt(wp*(wp+2*hp-W))));
    }
    Ans = max(Ans, PI/4*D*D*W);
    return Ans;
}
 
int main() 
{
    int T,W,H;
    T = scan();
    while(T--)
    {
        W = scan(); H = scan(); 
        printf("%.12lf\n",max(MaxVolume(W, H), MaxVolume(H, W)));
    }
  return 0;
}
