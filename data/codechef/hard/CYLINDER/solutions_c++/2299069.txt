#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

const double PI = 2 * acos(0.);

double maxV(double w,double h)
{
	double x = min(w / PI, 2 * h / 3);
	double ans = PI/4 * x * x * (h - x);
	double hp = h / (PI + 1),d = min(w / 2, hp),wp = w / ((PI+1) * (PI+1));
	if (2 * hp - w > 0)
		d = max(d, min(w, hp + wp - sqrt(wp * (wp + 2 * hp - w))));
	ans = max(ans, PI/4 * d * d * w);
	return ans;
}
int main() {
	int t,w, h;
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d %d",&w,&h);
		printf("%.12lf\n",max(maxV(w, h),maxV(h, w)));
	}
	return 0;
}
