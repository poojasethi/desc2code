#include <bits/stdc++.h>
#include <iostream>
#include <cstdio>
using namespace std;

int main(){
	int t;
	cin >> t;
	while(t--){
		int n;
		scanf("%d", &n);
		long long int curr_max, curr_min;
		scanf("%lld", &curr_min);
		curr_max = curr_min;
		for (int i = 1; i < n; ++i){
			int next;
			scanf("%d", &next);
			long long int a1 = curr_min + next;
			long long int a2 = curr_min * next;
			long long int a3 = curr_min - next;
			long long int a4 = curr_max + next;
			long long int a5 = curr_max - next;
			long long int a6 = curr_max * next;
			curr_min = min(min(min(min(min(a1, a2), a3), a4), a5), a6);
			curr_max = max(max(max(max(max(a1, a2), a3), a4), a5), a6);
			// cout << i << " " << curr_min << " " << curr_max << endl;
		}
		printf("%lld\n", curr_min);
	}
}
