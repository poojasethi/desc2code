#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n;
        cin>>n;
        long long int a[15];
        
        for(int i=0;i<n;i++)
            cin>>a[i];
        long long int best = a[0];
        long long int worst =a[0];
        long long int new_best = a[0];
        long long int new_worst =a[0];
        for(int i=1;i<n;i++)
        {
            if(a[i]>=0)
            {
                new_best = max(best - a[i],max(best+a[i],best*a[i]));
                new_worst = min(worst - a[i],min(worst+a[i],worst*a[i]));
            }
            else
            {
                new_best = max(best - a[i],max(best+a[i],worst*a[i]));
                new_worst = min(worst - a[i],min(worst+a[i],best*a[i]));
            }
            worst = new_worst;
            best = new_best;
        }
        cout<<worst<<"\n";



    }


}
