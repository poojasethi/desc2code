#include <iostream>
using namespace std;

long long int c,low[11],high[11];
long long int recur(int n, int a[]){
	for(int i=1;i<n;i++)
	{
		low[i]=min(min(low[i-1]+a[i],min(low[i-1]-a[i],low[i-1]*a[i])),min(high[i-1]+a[i],min(high[i-1]-a[i],high[i-1]*a[i])));
		high[i]=max(max(low[i-1]+a[i],max(low[i-1]-a[i],low[i-1]*a[i])),max(high[i-1]+a[i],max(high[i-1]-a[i],high[i-1]*a[i])));
	}
	return low[n-1];
}

int main() {
	int t,test,n,a[11];
	 std::ios::sync_with_stdio(false);
	cin>>test;
	for(t=0;t<test;t++)
	{
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>a[i];
		}
		high[0]=low[0]=a[0];
		printf("%lld\n",recur(n,a));

	}
	// your code goes here
	return 0;
}
