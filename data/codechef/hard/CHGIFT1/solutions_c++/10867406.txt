#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t,i,j,k,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        long long arr[n];
        long long a[n];
        long long b[n];
        for(i=0;i<n;i++)
            cin>>arr[i];
        a[0]=arr[0];
        b[0]=arr[0];
        for(i=1;i<n;i++)
        {
            if(arr[i]>0)
            {
            a[i]=max(arr[i]+a[i-1],max(a[i-1]*arr[i],a[i-1]-arr[i]));
            b[i]=min(arr[i]+b[i-1],min(b[i-1]*arr[i],b[i-1]-arr[i]));
            }
            else
            {
               a[i]=max(arr[i]+a[i-1],max(b[i-1]*arr[i],a[i-1]-arr[i]));
            b[i]=min(arr[i]+b[i-1],min(a[i-1]*arr[i],b[i-1]-arr[i]));
            }
        }
        cout<<b[n-1]<<"\n";
    }
}
