#include <bits/stdc++.h>
#define maxn 11

using namespace std;

int n, a[maxn];
long long f[maxn], g[maxn];

void read_in() {
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",a+i);
}

void solve() {
    f[1]=g[1]=a[1];
    for(int i=2;i<=n;i++)
    if (a[i]>0) {
        f[i]=max(f[i-1]+a[i],max(f[i-1]-a[i],f[i-1]*a[i]));
        g[i]=min(g[i-1]+a[i],min(g[i-1]-a[i],g[i-1]*a[i]));
    } else {
        f[i]=max(max(f[i-1]+a[i],f[i-1]-a[i]),g[i-1]*a[i]);
        g[i]=min(min(g[i-1]+a[i],g[i-1]-a[i]),f[i-1]*a[i]);
    }
}
int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    int T; scanf("%d",&T);
    while (T--) {
        read_in();
        solve();
        printf("%lld\n",g[n]);
    }
}
