#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll int arr[20];
ll int mindp[20];
ll int maxdp[20];
int n;
int main()
{

    int t;
    cin>>t;
    while(t--)
    {
        int i;
        cin>>n;
        for(i=0;i<n;i++)
        {
            cin>>arr[i];
        }
        mindp[0]=arr[0];
        maxdp[0]=arr[0];
        for(i=1;i<n;i++)
        {
            mindp[i]=min(min(min(min(min(maxdp[i-1]*arr[i],maxdp[i-1]-arr[i]),maxdp[i-1]+arr[i]),mindp[i-1]*arr[i]),mindp[i-1]-arr[i]),mindp[i-1]+arr[i]);
            maxdp[i]=max(max(max(max(max(maxdp[i-1]*arr[i],maxdp[i-1]-arr[i]),maxdp[i-1]+arr[i]),mindp[i-1]*arr[i]),mindp[i-1]-arr[i]),mindp[i-1]+arr[i]);
        }
            cout<<mindp[n-1]<<endl;

    }


return 0;

}
