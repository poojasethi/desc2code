def find_min_value(lst):
    mini = lst[0]
    maxi = lst[0]
    for i in range(1, len(lst)):
        tmp = mini
        mini = min(tmp + lst[i], tmp - lst[i], tmp * lst[i], maxi + lst[i], maxi - lst[i], maxi * lst[i])
        maxi = max(tmp + lst[i], tmp - lst[i], tmp * lst[i], maxi + lst[i], maxi - lst[i], maxi * lst[i])
    return mini
