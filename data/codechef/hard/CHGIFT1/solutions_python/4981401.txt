t = int(raw_input())
for i in xrange(t):
    n = int(raw_input())
    a = map(int, raw_input().split())
    mi = a[0]
    ma = a[0]
    for j in xrange(1,n):
        b = mi*a[j]
        c = mi+a[j]
        d = mi-a[j]
        e = ma*a[j]
        f = ma+a[j]
        g = ma-a[j]
        mi = min(b,c,d,e,f,g)
        ma = max(b,c,d,e,f,g)
    print mi
