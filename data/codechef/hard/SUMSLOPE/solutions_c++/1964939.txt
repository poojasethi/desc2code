ll p[17];
int c[1001],a[1001];
void pre()
{
 a[0]=0;c[0]=0;p[0]=1;
 int i;
 for(i=1;i<17;i++)p[i]=p[i-1]*10;
 for(i=1;i<1000;i++)
 {
   int x=i/10,y=i%10,z=x/10;
   x%=10;
   if(x<y&&x<z||(x>y&&x>z))
   a[i]=1,c[i]=1;
   c[i]+=c[i-1];
 }
}
ll solve(ll n)
{
 if(n<100)return 0;
 else if(n<1000)return c[n]-c[100];
 int b[19],ptr=0,v,k;
 ll suf[19];suf[0]=0;
 while(n)
 {
  b[ptr++]=n%10;n/=10;
  suf[ptr]=suf[ptr-1]+(ll)b[ptr-1]*p[ptr-1];
 }
 ll ans=0,tail=0;
 for(int i=ptr-1;i>=2;i--)
 {
  	k=i;
  	v=b[k]*100+b[k-1]*10+b[k-2];
  	if(tail)
  	{
  	 ans+=(tail-1)*570*p[k-2];
  	 ans+=(ll)(570-c[100])*p[k-2];
	}
	if(v)
	{
	 if(tail)ans+=c[v-1]*p[k-2];
	 else ans+=(ll)(c[v-1]>c[100]?c[v-1]-c[100]:0)*p[k-2];
	}
	if(a[v])ans+=suf[k-2]+1;
	tail=tail*10+b[i];
 }
 return ans;
}
int main()
{
  int T,i,j,k;ll n,m;
  pre();
 // freopen("in.txt","r",stdin);
  scanf("%d",&T);
  while(T--)
  {
   scanf("%lld%lld",&n,&m);
   printf("%lld\n",solve(m)-solve(n-1));
  }
  return 0;
}
