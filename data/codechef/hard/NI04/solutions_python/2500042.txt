#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int main()
{
    int t;

    cin >> t;

    while(t--) {
            int p, q, n, m, k;
            cin >> p >> q >> m >> n >> k;
            int gone = 0;
            queue<int> ready;
            int landed = 0;
            for(int i = 1; i <= k+1; i++) {
                if(!ready.empty() && i-ready.front() >= n) {
                    q++;
                    ready.pop();
                }
                if(p) {
                    p--;
                    landed++;
                    ready.push(i);
                }
                else if(q) {
                    q--;
                    gone++;
                }
                if(i%m == 0) p++;
            }

            cout << landed << " " << gone << " " << p << " " << q+ready.size() << endl;
    }
}
