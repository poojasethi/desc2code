#include <stdio.h>
#include <string.h>

#define SZ 100009

int takeOffs[SZ] = {0};

int main()
{
    int t, p, q, m, n, k;
#ifndef ONLINE_JUDGE
    freopen( "ni04.txt","r", stdin );
#endif
    scanf( "%d\n", &t );
    while( t-- )
    {
        int landed = 0, takeOff = 0, time;
        int wtl = 0;
        int wto = 0;
        int isAnyLanding;
        memset( takeOffs, 0, sizeof( int) * SZ );
        scanf( "%d %d %d %d %d", &p, &q, &m, &n, &k );
        wtl = p;
        wto = q;
        takeOffs[0] = q;

        for( time = 0; time <= k; ++time )
        {
            if( time && !(time % m ))
                wtl++;

            if( wtl )
            {
                landed++;
                takeOffs[time+n]++;
                takeOffs[time+1] += takeOffs[time];
                wtl--;
                wto++;
            }
            else
            {
                if( takeOffs[time] )
                {
                    takeOffs[time]--;
                    takeOffs[time+1] += takeOffs[time];
                    wto--;
                    takeOff++;
                }
            }
        }
        if( !( (k+1) % m ) ) 
            wtl++;
        printf("%d %d %d %d\n", landed, takeOff, wtl, wto );
    }
    return 0;
}
