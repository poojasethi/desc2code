using namespace std;

#define rep(i,a,b) for(typeof(a) i=(a);i<(b);i++)
#define ll long long int
#define ii pair<int,int>
#define gi(x) scanf("%d",&x)
#define Clear(x,val) memset(x,val,sizeof(x))
#define SZ(v) (v).size()
#define MAX 100009

int main()
{
	int T; gi(T);
	while( T-- ) {
		int p , q , m , n , k;
		gi(p); gi(q); gi(m); gi(n); gi(k);
		vector<int> PQ , PQ1;PQ.clear(); PQ1.clear();	

		rep( i , 0 , p ) { PQ.push_back(0); }
		for( int i = m; i <= (k+1); i+=m ) { PQ.push_back(i); }
		rep( i, 0 , q ) { PQ1.push_back(0); }

		int tm = 0;		
		int aland = 0 , atako = 0;				
		int ls = 0 , ln = PQ.size() , ts = 0 , tn = PQ1.size();
	

		while( tm <= k ) {
			bool flag = false;
			if( ls < ln ){
				int lt = PQ[ls];
				if( lt <= tm ) { aland += 1; PQ1.push_back(lt+n); tn++; ls++; flag = true; } 			
			}
			if(!flag) {
				if( ts < tn ) {
					int tk = PQ1[ts];
					if( tk <= tm ) { atako += 1; ts++; }
				} 
			}
			tm += 1;
		}
		cout << aland << " " << atako << " " << (ln-ls) <<" " << (tn-ts) << "\n"; 
	}	
	return 0;	
}
