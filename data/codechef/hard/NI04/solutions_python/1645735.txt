#include<stdio.h>
#include<stdlib.h>
int q[100002];
int main() {
    int t,p,r,k,i,m,n,l,f;
    scanf("%d",&t);
    while(t--) {
               scanf("%d%d%d%d%d",&p,&q[0],&m,&n,&k);
               l=f=r=0;
               for(i=0;i<=k;i++) {
                                 if(i&&!(i%m)) p++;
                                 if(p) {
                                       p--;
                                       l++;
                                       if((i+n)<=k) q[i+n]++;
                                       else r++;
                                 }
                                 else if(q[i]) {
                                      q[i]--;
                                      f++;
                                 }
                                 q[i+1]+=q[i];     
               }
               if(!((k+1)%m)) p++;
               printf("%d %d %d %d\n",l,f,p,q[k]+r);
               if(t) memset(q,0,4*(k+2));
    }
    return 0;
}
