
#include<stdio.h>
#include<queue>

using namespace std;
int main()
{
    long long countl,countt,i,t,n,m,p,q,k,l;
    //queue<long long> P,Q;
    scanf("%lld",&t);
    while(t--)
    {
 queue<long long> P,Q;

    scanf("%lld %lld %lld %lld %lld",&p,&q,&m,&n,&k);

    l=p;

    while(l--)
    {
        P.push(0);

    }
    l=q;
    while(l--)
    Q.push(0);
    countl=0;
    countt=0;

    for(i=1;i<=k+1;i++)
    {

        if(!P.empty())
        {

            P.pop();
            Q.push(i);
            countl++;

        }
        else

             if(!Q.empty()&&(Q.front()==0||Q.front()+n<=i))
            {

            Q.pop();
             countt++;
        }
         if(i%m==0)
        {
            P.push(i);

        }

    }
printf("%lld %lld %d %d\n",countl,countt,P.size(),Q.size());

    }
return 0;

}
