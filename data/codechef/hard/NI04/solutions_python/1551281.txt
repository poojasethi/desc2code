#include<stdio.h>
int time[1000000]={0};
int main()
{

	int t;
	scanf("%d",&t);
	while(t--)
	{
		int n,m,q,p,r,k,i,j;
		scanf("%d%d%d%d%d",&p,&q,&m,&n,&k);
		int plane_land = p,plane_take=q,taked_off = 0,total_land=0;
		r = 0;
		int x = 0;
		for(i=0;i<=k;i++)
		{
			if( x!=r  && i==time[x]+n)
			{
				plane_take++;
				x++;
			}
			if(plane_land)
			{
				time[r++] = i;
				total_land++;
				plane_land--;
			}
			else
			{
				if(plane_take)
				{
					plane_take--;
					taked_off++;
				}
			}
			if((i+1)%m==0)
				plane_land++;
		}
		printf("%d %d %d %d\n",total_land,taked_off,plane_land,plane_take+(r-x));
	}
	return 0;
}
