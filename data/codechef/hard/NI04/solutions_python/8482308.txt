#include <bits/stdc++.h>
using namespace std;
int tim[100003];
int main(){
	int t;cin>>t;
	while(t--){
		int p,q,m,n,k;
		cin>>p>>q>>m>>n>>k;
		int land = p;
		int takeoff = q;
		int landed =0;
		int takedoff =0;
		int dekhte =q;
		for(int i =0;i<=k;i++){
			//cout<<landed<<" "<<takedoff<<" "<<land<<" "<<takeoff<<endl;
			if(((i)%(m))==0&&i!=0)land++;
			if(tim[i]!=0)takeoff+=tim[i];
			if(land>0){
				land--;
				landed++;
				dekhte++;
				if((i+n)<=k+1)tim[i+n]++;
			}
			else if(takeoff>0){
				takeoff--;
				takedoff++;
				dekhte--;
			}
			
			//cout<<landed<<" "<<takedoff<<" "<<land<<" "<<takeoff<<endl;
		}
		if((k+1)%m==0)land++;
		cout<<landed<<" "<<takedoff<<" "<<land<<" "<<dekhte<<endl;
		for(int i =0;i<100003;i++){
			tim[i]=0;
		}
	}
	
}
