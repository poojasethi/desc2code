#include<stdio.h>

int main()
{
	int p,q,m,n,k;

	int cases;
	
	scanf("%d",&cases);
	
	while(cases--)
	{
	scanf("%d%d%d%d%d",&p,&q,&m,&n,&k);
	int q1=0,q2=0;
	int t=0;
	int m1;
	int i=0;
	int j=0;
	int pland=0;
	int ptoff=0;
	int time[100000] = {0};
	
	m1=m;
	
	
	while(t<=k)
	{
		
		if(p) //when plane is in the air
		{
			p--;
			pland++;
			time[i]=t;
			i++;
			t++;
			q1++;
		}
		
		else //when there is no plane in air
		{
			if(q) //plane already waiting to take off since starting
			{
				q--;
				ptoff++;
				t++;
			}
			else // plane waiting to take off after landing
			{
				if(t >= (time[j]+n) && q1)
				{
					ptoff++;
					q1--;
					t++;
					j++;
				}
				else
				t++;
			}
		}
		
		if(t==m1)
		{
			p++;
			m1=m1+m;
		}
	}
	
	q2=q+q1;
	printf("%d %d %d %d\n",pland,ptoff,p,q2);
	
	}
	
	return 0;
}
