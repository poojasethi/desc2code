#include <cstdio>

int main(void) {
    int kases;
    scanf("%d", &kases);
    while (kases--) {
        int m, n, p, q, k;
        int queue_landing[1000001] = {0};
        int already_landed = 0, already_takenoff = 0;
        scanf("%d%d%d%d%d", &p, &q, &m, &n, &k);
        for (int i = 0; i <= k; i++) {
            if (p > 0) {
                p--;
                already_landed++;
                if (i+n <= k+n)
                    queue_landing[i+n] = 1;
            }
            else if (q > 0) {
                q--;
                already_takenoff++;
            }
            // Check for any ready flights
            q += queue_landing[i+1];
            if ((i+1)%m == 0)
                p++;
            //printf("At turn %d: %d %d %d %d\n", i, already_landed, already_takenoff, p, q);
        }
        for (int i = k+2; i <= k+n; i++)
            q += queue_landing[i];
        printf("%d %d %d %d\n", already_landed, already_takenoff, p, q);
    }
}
