int p, q, m, n, k;
void solve()
{
    queue<int> grnd;
    for(int i=0; i<q; ++i)grnd.push(0);
    int tof=0, lnd=0;
    for(int tym=0; tym<=k; ++tym)
    {
        if(tym > 0 && tym%m == 0)p++;

        if(p)
        {
            p--;
            lnd++;
            grnd.push(tym+n);
        }
        else
        {
            if(!grnd.empty())
            {
                int tt = grnd.front();
                if(tt <= tym)
                {
                    tof++;
                    grnd.pop();
                }
            }
        }
    }
    if((k+1)%m==0)p++;

    printf("%d %d %d %d\n", lnd, tof, p, grnd.size());
}

void readInput()
{
    scanf("%d%d%d%d%d", &p, &q, &m, &n, &k);

}

int main()
{
    //freopen("input.txt", "r", stdin);
    //freopen("out.txt", "w", stdout);
    int testcases;
    scanf("%d", &testcases);
    for(int caseno=1; caseno<=testcases; caseno++)
    {
        //printf("Case #%d: ", caseno);
        readInput();
        solve();
    }
    return 0;
}
