#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<math.h>
#include<string.h>
#include<map>
#include<set>
#include<vector>
#include<queue>
 
using namespace std;
 
int main()
{
    int t,n,m,p,q,k,i;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d %d %d %d %d",&p,&q,&m,&n,&k);
        queue<int> Q;
 
        int totalLandWaiting = p;
        int totalTakeoffWaiting = q;
        int alreadyTakenoff = 0, alreadyLanded = 0;
 
        for(i=0;i<=k;i++)
        {
 
            if(!Q.empty())
            {
                int firstTime = Q.front();
                if(firstTime ==i)
                {
                    Q.pop();
                    totalTakeoffWaiting++;
                }
            }
 
            if(i%m == 0 && i!=0)
                totalLandWaiting++;
 
            if(totalLandWaiting>0)
            {
                totalLandWaiting--;
                alreadyLanded++;
                Q.push(i+n);
            }
 
            else if(totalTakeoffWaiting>0)
            {
                totalTakeoffWaiting--;
                alreadyTakenoff++;
            }
            
 
        }
        if(!Q.empty())
            {
                int firstTime = Q.front();
                if(firstTime ==i)
                {
                    Q.pop();
                    totalTakeoffWaiting++;
                }
            }
 
        if(i%m == 0 && i!=0)
                totalLandWaiting++;
        printf("%d %d %d %d\n",alreadyLanded,alreadyTakenoff,totalLandWaiting,(alreadyLanded+q-alreadyTakenoff));
    }
}
