#include <bits/stdc++.h>
using namespace std;

const int N = 1e5 + 100;
int wt[N];

int main() {
	ios_base::sync_with_stdio(0);
	int t; cin >> t;

	while(t--) {
		int p, q, m, n, k;
		cin >> p >> q >> m >> n >> k;
		memset(wt, 0, sizeof  wt);
		int init = q;
		wt[0] = q;
		int landed = 0; int takenoff = 0;

		for(int i=0; i<=k; i++) {

			if(i>0 && i%m==0) 
				p++;

			if(p) { 
				landed++; p--;
				wt[i+n]++;
			}
			else if(wt[i]) {
				takenoff++; wt[i]--;
			}

			wt[i+1] += wt[i];
		} 

		if((k+1)%m == 0)
			p++;

		cout << landed << " " << takenoff << " " << p <<  " " << landed + q - takenoff << endl;

	}
}
