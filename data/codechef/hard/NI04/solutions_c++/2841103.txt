#include <bits/stdc++.h>

using namespace std;

int main ()
{
	int T;
	int p, q, n, m, k, land[100005], landed, takeoff;
	scanf ("%d", &T);
	while (T--)
	{
		scanf ("%d %d %d %d %d", &p, &q, &m, &n, &k);
		int M = q;
		landed = takeoff = 0;
		memset (land, 0, sizeof (land));
		if (p)
		{
			land[0] = 1;
			p--;
			landed++;
			M++;
		}
		else if (q)
		{
			land[0] = 0;
			q--;
			takeoff++;
			M--;
		}
		else
			land[0] = 0;
		for (int t = 1; t <= k; t++)
		{
			if (t - n >= 0)
				q += land[t - n];
			if (t % m == 0)
				p++;
			if (p)
			{
				land[t] = 1;
				p--;
				landed++;
				M++;
			}
			else if (q)
			{
				land[t] = 0;
				q--;
				takeoff++;
				M--;
			}
			else
				land[t] = 0;
		}
		if ((k + 1) % m == 0)
			p++;
		printf ("%d %d %d %d\n", landed, takeoff, p, M);
	}
	return 0;
}
