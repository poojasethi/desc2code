#include<cstdio>
#include<queue>
using namespace std;
std::queue<int>r;
int cnt,cnt1,p,q,m,n,k,t;
int main()
{
    int i,j,flag;
    scanf("%d",&t);
    while(t--)
    {
        cnt1=cnt=0;
        while(!r.empty())
            r.pop();
        scanf("%d %d %d %d %d",&p,&q,&m,&n,&k);
        for(i=1; i<=k+1; i++)
        {
          if(!r.empty())
            {
                if(r.front()+n<=i)
                {r.pop();
                q++;
                }
            }
            if(p>0)
            {
                cnt++;
                p--;
                r.push(i);
            }
            else
            {
                if(q>0)
                {
                    q--;
                    cnt1++;
                }
            }
            if(i%m==0)
            p++;
 
        }
 
 
        printf("%d %d %d %d\n",cnt,cnt1,p,q+r.size());
    }
 
 
    return 0;
}
