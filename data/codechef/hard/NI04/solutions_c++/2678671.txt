#include <cstdio>
#include <algorithm>
#include <cstring>
#include <queue>
using namespace std;
typedef pair<int,int> ii;

int p, q, m, n, k;
int ans;

queue<int> qu;
int landed, takedOff, onGround, inAir;
int cdOnLand;

int main()
{
    //freopen("in.txt", "r", stdin);
    int tc;
    scanf("%d", &tc);
    while (tc--)
    {
        scanf("%d %d %d %d %d", &p, &q, &m, &n, &k);
        while (!qu.empty()) qu.pop();
        for (int i=0; i<q; i++) qu.push(0);
        landed = 0;
        takedOff = 0;
        onGround = q;
        inAir = p;
        cdOnLand = m;
        for (int i=1; i<=k+1; i++)
        {
            
            if (inAir)
            {
                landed++;
                onGround++;
                inAir--;
                qu.push(i+n);
            }
            else if (onGround && qu.front() <= i)
            {
                takedOff++;
                onGround--;
                qu.pop();
            }
            if(i%m == 0) inAir++;
        }
        printf("%d %d %d %d\n", landed, takedOff, inAir, onGround);
    }
}
