#include <iostream>
#include<queue>
#include<math.h>
#include<algorithm>
#include<map>
using namespace std;

int main()
{
    int t;cin>>t;
    while(t--)
    {
        int p,q,m,n,k,ina=0,lan=0,qi;
        cin>>p>>q>>m>>n>>k;
        qi=q;
        int i=1;
        queue<int> onland;
        while(i<=k+1)
        {

           if(p)
           {//cout<<'p'<<i<<endl;
               p--;lan++;
               onland.push(i);
           }
           else if(q)
           {//cout<<'q'<<i<<endl;
               q--;
               ina++;
           }
           else
           {
               if(!onland.empty()&&i-onland.front()>=n) {ina++;onland.pop();}
           }
            
            if(i%m==0) p++;i++;
        }
        cout<<lan<<' '<<ina<<' '<<p<<' '<<lan+qi-ina<<endl;
    }
    return 0;
}
