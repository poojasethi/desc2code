#include <iostream>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        int n,m,tt=0;
        cin>>n>>m;
        string s[n];
        int A[n][m],X[30],Y[30];
        for(int i=0;i<n;i++) cin>>s[i];
        for(int i=0;i<n;i++)
            for(int j=0;j<m;j++)
                if(s[i][j]=='L')
                {
                    A[i][j]=tt;
                    X[tt]=i;Y[tt]=j;
                    tt++;
                }
        int f=0;
        for(int i=0;i<(1<<tt);i++)
        {
            int VV[n][m];
            memset(VV,0,sizeof(VV));
            int val=i,p=0;
            while(val!=0)
            {
                if(val&1)
                    for(int j=0;j<X[p];j++) VV[j][Y[p]]=1;
                val=val>>1;
                p++;
            }               
            int pp=0;
            for(int j=0;j<n;j++)
            {
                int ma=-1,mb=m;
                for(int k=0;k<m;k++)
                {
                    if(s[j][k]=='L' && (i&(1<<A[j][k]))==0)
                    {
                        ma=max(ma,k);
                        mb=min(mb,k);
                    }
                }
                if(ma!=-1 && ma!=mb) continue;
                if(ma==-1)
                {
                    int vv=0;
                    for(int k=0;k<m;k++)
                        if(s[j][k]=='E' && VV[j][k]==0)
                            vv=1;
                    if(vv==1)
                    {
                        pp=1;
                        break;
                    }
                    else continue;
                }
                int l=0,r=0;
                for(int k=0;k<ma;k++)
                    if(s[j][k]=='E' && VV[j][k]==0)
                        l++;
                for(int k=ma+1;k<m;k++)
                    if(s[j][k]=='E' && VV[j][k]==0)
                        r++;
                if(l!=0 && r!=0)
                {
                    pp=1;
                    break;
                }
            }
            if(pp==0)
            {
                f=1;
                break;
            }
        }
        if(f==0) cout<<"Impossible\n";
        else cout<<"Possible\n";
    }
    return 0;
}
