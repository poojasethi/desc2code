int pos[17][2],n,m;
int o=0,count1;
int func(char s[][51],int k,int c){
	int i,j,flag,p,q,c1=0,c2=0,c3=0;
	if(!c) return 1;
    if(k==count1) return 0;
    for(j=pos[k][1]-1;j>=0;j--){
           if(s[pos[k][0]][j]=='E')
               c1++;
           s[pos[k][0]][j]++;
                                   
		
	}
    if(c1)
	if(func(s,k+1,c-c1))return 1;
    
	for(j=pos[k][1]-1;j>=0;j--){
	  
			s[pos[k][0]][j]--;
		
	}
	for(i=pos[k][0]-1;i>=0;i--){
		if(s[i][pos[k][1]]=='E'){
            
            c2++;
        }
		s[i][pos[k][1]]++;
	}
	if(c2)
    if(func(s,k+1,c-c2))return 1;
    
	for(i=pos[k][0]-1;i>=0;i--){
			s[i][pos[k][1]]--;
		
	}
	for(j=pos[k][1]+1;j<m;j++){
		if(s[pos[k][0]][j]=='E')
            c3++;
			s[pos[k][0]][j]++;
		
	}
    if(c3)
	if(func(s,k+1,c-c3)) return 1;
    
	for(j=pos[k][1]+1;j<m;j++){
		
			s[pos[k][0]][j]--;
		
	}
	if(!c1&&!c2&&!c3) if(func(s,k+1,c-c2))return 1;
        return 0;
}
