int n,m,len;
vector< pair<int,int> > laser;
char a[60][60];

int rec(int id,int left)
{
    bool gone=false;
    if(left==0)return 1;
    if(id==len)return (left==0);
    int posi,posj,t,tot,i,j;
    posi=laser[id].F;
    posj=laser[id].S;

    vector<pair<int,int> > temp;

    for(t=posj-1;t>=0;t--)if(a[posi][t]=='E'){
        a[posi][t]='.';
        temp.push_back(make_pair(posi,t));
    }
    tot=temp.size();
    if(!temp.empty()){
        if(rec(id+1,left-tot))return 1;gone=true;}
    for(i=0;i<tot;i++)a[temp[i].F][temp[i].S]='E';
    temp.clear();


    for(t=posi-1;t>=0;t--)if(a[t][posj]=='E'){
        a[t][posj]='.';
        temp.push_back(make_pair(t,posj));
    }
    tot=temp.size();
    if(!temp.empty()){
        if(rec(id+1,left-tot))return 1;gone=true;}
    for(i=0;i<tot;i++)a[temp[i].F][temp[i].S]='E';
    temp.clear();


    for(t=posj+1;t<m;t++)if(a[posi][t]=='E'){
        a[posi][t]='.';
        temp.push_back(make_pair(posi,t));
    }
    tot=temp.size();
    if(!temp.empty()){
        if(rec(id+1,left-tot))return 1;gone=true;}
    for(i=0;i<tot;i++)a[temp[i].F][temp[i].S]='E';
    temp.clear();
    if(!gone)if(rec(id+1,left))return 1;
    return 0;
}
