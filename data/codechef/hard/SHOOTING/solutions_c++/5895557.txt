#include<iostream>
#include<cstdio>
#include<stack>
using namespace std;
struct laser
{
    int x,y;
};
int call(laser S[],int counte,char ch[][50],int n,int m,char ch1,int size)
{
    if(counte==0)
        return 1;
    else if(size<0)
        return 0;
    else
    {
        bool a,b,c;
        int x,y,i,j,e=0;
    x=S[size].x;
    y=S[size].y;
    e=0;
    for(j=y;j>=0;j--)
    {
        if(ch[x][j]=='E')
        {
            e++;
            ch[x][j]=ch1;
        }
    }
    if(e!=0)
    a=call(S,counte-e,ch,m,n,ch1+1,size-1);
    else
        a=0;
    for(j=y;j>=0;j--)
    {
        if(ch[x][j]==ch1)
        {
            ch[x][j]='E';
        }
    }
    if(a==1)
        return a;
    e=0;
    for(j=y;j<m;j++)
    {
        if(ch[x][j]=='E')
        {
            e++;
            ch[x][j]=ch1;
        }
    }
    if(e!=0)
    b=call(S,counte-e,ch,m,n,ch1+1,size-1);
    else b=0;
    for(j=y;j<m;j++)
    {
        if(ch[x][j]==ch1)
        {
            ch[x][j]='E';
        }
    }
    e=0;
    if(b==1)
        return b;
    for(i=x;i>=0;i--)
    {
        if(ch[i][y]=='E')
        {
            e++;
            ch[i][y]=ch1;
        }
    }
    c=call(S,counte-e,ch,m,n,ch1+1,size-1);
    for(i=x;i>=0;i--)
    {
        if(ch[i][y]==ch1)
        {
            e++;
            ch[i][y]='E';
        }
    }
    return c;
    }
}
 int main()
 {
     laser S[17];
     int t,n,m,i,j,countl=0,counte=0,size=0;
     char ch[50][50];
    scanf("%d",&t);
    while(t>0)
    {
        t--;
        countl=0;counte=0;
        scanf("%d %d",&n,&m);
        for(i=0;i<n;i++)
            {
                scanf("%s",&ch[i]);
            }
            for(i=0;i<n;i++)
            {
                for(j=0;j<m;j++)
                {
                    if(ch[i][j]=='L')
                    {
                        countl++;
                        S[size].x=i;
                        S[size].y=j;
                        size++;
                    }
                if(ch[i][j]=='E')
                    counte++;
                }
            }
            if(call(S,counte,ch,n,m,'a',size-1))
                printf("Possible\n");
            else
                printf("Impossible\n");
                size=0;
    }
 }
