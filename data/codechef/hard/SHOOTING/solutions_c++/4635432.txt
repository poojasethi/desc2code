#include<iostream>
#include<stdio.h>
using namespace std;

char a[55][55];
int ax[20],ay[20],L[55][55],E[55][55];

int main()
{
    int t;
    cin >> t;
    int curr = 0;

    while(t--)
    {
        int n,m,cnt=0;
        scanf("%d %d",&n,&m);
        
        for(int i=0;i<n;i++)
        {
                for(int j=0;j<m;j++)
                {
                    a[i][j] = getchar();
                    while(a[i][j]!='.' && a[i][j]!='E' && a[i][j]!='L')
                        a[i][j] = getchar();

                    if(a[i][j]=='L')
                    {
                        cnt++;
                        ax[cnt] = i;
                        ay[cnt] = j;
                    }
                }
        }
        
        int ans = -1;

        for(int mask=0;mask<(1<<cnt);mask++)
        {
            curr++;
           /* for(int i=0;i<n;i++)
                for(int j=0;j<m;j++)
                    E[i][j] = ((a[i][j]=='E') ? 1 : 0);
            
            for(int i=0;i<n;i++)
                for(int j=0;j<m;j++)
                    L[i][j] = ((a[i][j]=='L') ? 1 : 0);
            */

            for(int i=1;i<=cnt;i++)
            {
                if(mask & (1<<(i-1)))
                {
                    int x1 = ax[i], y1 = ay[i];
                    L[x1][y1] = curr;
                    while(x1>=0)
                    {
                        E[x1][y1] = curr;
                        x1--;
                    }
                }
            }
            
            int r[55] = {0};
            
/*            for(int i=0;i<n;i++)
            {
                for(int j=0;j<m;j++)
                {
                    r[i] += L[i][j];
                }
            }
*/
            for(int j=0;j<cnt;j++)
                if(!(mask & (1<<j)))
                    r[ax[j+1]]++;
            
            int flag = 0;
                        
            for(int i=0;i<n;i++)
            {
                int fl = 0, fr = 0, ll = 0;
            
                if(r[i]<2)
                {
                    for(int j=0;j<m;j++)
                    {
                        if(a[i][j]=='E' && E[i][j]!=curr)
                        {
                            if(!fl)
                            {
                                if(!ll)
                                    fl = 1;
                                else
                                    fr = 1;
                            }
                            else if(ll)
                            {
                                fr = 1;
                                break;
                            }
                        }
                        else if(a[i][j]=='L' && L[i][j]!=curr)
                            ll = 1;
                    }
                    if((!ll && fl) || (fl && ll && fr))
                    {
                        flag = 1;
                        break;
                    }
                }
            }
            //cout << flag << endl;
            if(!flag)
            {
                ans = 1;
                break;
            }    
        }

        if(ans==1)
            printf("Possible\n");
        else
            printf("Impossible\n");
    }
return 0;
}
