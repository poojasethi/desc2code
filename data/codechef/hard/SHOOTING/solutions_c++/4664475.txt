#include <bits/stdc++.h>

using namespace std;
#define pb push_back
#define mp make_pair
#define ff first
#define ss second

typedef long long i64;

int main(){
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	
	ios_base::sync_with_stdio(false);
	i64 n,m,t,i,j;
	string s[55];
	cin>>t;
	while(t--){
		
		cin>>n>>m;
		for(i=0;i<n;i++)
		cin>>s[i];
		
		vector<long long> e(n,0);
		for(i=0;i<n;i++){
			for(j=0;j<m;j++){
				
				if(s[i][j]=='E')
				e[i]|=(1LL<<j);
			}
		}
		
		vector<pair<int,int> >l;
		for(i=0;i<n;i++){
			for(j=0;j<m;j++){
				
				if(s[i][j]=='L')
				l.pb(mp(i,j));
				
			}
		}	
		bool ok;	
		for(i=0;i<(1<<l.size());i++){
			vector<pair<int,int > >la(n,mp(-1,m+1));
			ok =true;
			int x=l.size()-1;
			long long b=0;
			
			for(j=n-1;j>=0;j--){
				while(x>=0 && l[x].ff>=j){
					if(i&(1<<x)) b|=(1LL<<(l[x].ss));
					else la[l[x].ff].ff=max(la[l[x].ff].ff,l[x].ss),la[l[x].ff].ss=min(la[l[x].ff].ss,l[x].ss);
					x--;			
				}
				long long bb=(( 1LL<<m)-1-b)&e[j];
				
				if(bb==0)
				continue;
				if(la[j].ff==-1){ok=false;break;}
				if(la[j].ff==la[j].ss)
					if( (bb& ((1LL<<la[j].ff)-1))!=0)
						if((bb&((1LL<<la[j].ff)-1))!=bb)
						{ok=false;break;}
			}
			if(ok==true)
			break;
		}
		if(ok==true)
		cout<<"Possible\n";
		else
		cout<<"Impossible\n";
	}
	return 0;
}
