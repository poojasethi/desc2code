#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define mod 1000000007

using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< int, int > PII;

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	int T;
	cin>>T;
	while(T--)
	{
		int n,m,cnt=0;
		cin>>n>>m;
		vector<string> s(n);
		vector<PII> l;
		VVI b(n);
		for(int i=0; i<n; i++)
		{
			b[i].resize(m);
			cin>>s[i];
			for(int j=0; j<m; j++)
			{
				b[i][j]=-1;
				if(s[i][j]=='L')
				{
					l.pb(mp(i,j));
					b[i][j]=cnt++;
				}
				else if(s[i][j]=='E')
					b[i][j]=-2;
			}
		}

		int f1=0;
		for(int bit=0; bit<(1<<cnt); bit++)
		{
			int col=0;
			f1=1;
			for(int i=n-1; i>=0; i--)
			{
				int fl=0,fr=0,f=0,count=0;
				for(int j=0; j<m; j++)
				{
					if(b[i][j]==-1)continue;
					if(b[i][j]>=0)
					{
						if(bit&(1<<b[i][j]))
						{
							col |= (1<<j);
						}
						else
						{
							count++;
							f=1;
						}
					}
					if(b[i][j]==-2 && (col&(1<<j))==0)
					{
						if(f)fr++;
						else fl++;
					}
				}
				if(fl+fr==0)continue;
				if(count>=2)continue;
				if(count==1 && (fr==0 || fl==0))continue;
				f1=0;
				break;
			}
			if(f1)break;
		}

		if(!f1)
			cout<<"Impossible\n";
		else
			cout<<"Possible\n";
	}
	return 0;
}
