#include <iostream>
#include <algorithm>
#include <cstring>
#include <stdio.h>
using namespace std;
 
int n,m;
char mat[55][55];
 
int col[55];
bool ok;
 
void backtr(int lin){
    if(lin==(n+1)){
        ok=true;
        return;
    }
 
    bool exista=false;
    for(int j=1;j<=m;j++)
        if(!col[j] && mat[lin][j]=='E'){
            exista=true;
            break;
        }
 
    if(!exista){
        for(int j=1;j<=m;j++)
            if(mat[lin][j]=='L')
                col[j]++;
 
        backtr(lin+1);
 
        for(int j=1;j<=m;j++)
            if(mat[lin][j]=='L')
                col[j]--;
 
        return;
    }
 
    int minim=m+1;
    int maxim=0;
    for(int j=1;j<=m;j++)
        if(!col[j] && mat[lin][j]=='E'){
            maxim=max(maxim,j);
            minim=min(minim,j);
        }
 
    int lasere=0;
    for(int j=1;j<=m;j++)
        if(mat[lin][j]=='L')
            col[j]++,lasere++;
 
    if(!lasere){
        for(int j=1;j<=m;j++)
            if(mat[lin][j]=='L')
                col[j]--;
        return;
    }
 
    for(int j=1;j<minim;j++)
        if(mat[lin][j]=='L'){
            col[j]--;
            backtr(lin+1);
            if(ok)
                break;
 
            col[j]++;
        }
 
    for(int j=maxim+1;j<=m;j++)
        if(mat[lin][j]=='L'){
            col[j]--;
            backtr(lin+1);
            if(ok)
                break;
 
            col[j]++;
        }
 
    if(lasere>=2)
    for(int i=1;i<=m;i++)
        for(int j=i+1;j<=m;j++)
            if(mat[lin][i]=='L' && mat[lin][j]=='L'){
                col[i]--;col[j]--;
                backtr(lin+1);
                col[i]++;col[j]++;
 
                if(ok)
                    return;
            }
 
    for(int j=1;j<=m;j++)
        if(mat[lin][j]=='L')
            col[j]--;
}
 
int main()
{
    int t=0;
    cin>>t;
 
    while(t--){
        cin>>n>>m;
 
        for(int i=n;i>=1;i--){
            cin.get();
            cin.get(mat[i]+1,55);
        }
 
        ok=false;
        for(int i=1;i<=m;i++)
            col[i]=0;
 
        backtr(1);
 
        if(ok)
            printf("Possible\n");
        else
            printf("Impossible\n");
    }
 
    return 0;
}
