#include <stdio.h>
#include <stdlib.h>
#define MODS 1000000007
long long changes[200001];
long long sums;
long long mfact;
long long tmp;
long long ans;
long long pluss;
int n,m;
int t;
inline void init()
{
 sums=0;
 ans=0;
 mfact=1;
 for(int i=1;i<=m;i++)
 {
  mfact=(long long)mfact*i;
  mfact=mfact%MODS;
 }
 for(int i=1;i<=n;i++)
 {
  changes[i]=0;
 }
}
main()
{
 scanf("%d",&t);
 for(int tests=1;tests<=t;tests++)
 {
  scanf("%d%d",&n,&m);
  init();
  changes[m+1]=(long long)mfact;
  changes[m+1]=changes[m+1]%MODS;
  for(int i=1;i<=m;i++)
  {
  tmp=m-i;
  changes[m+i+1]=(long long)mfact*tmp;
  changes[m+i+1]=changes[m+i+1]%MODS;
  }
  sums=changes[m+1];
  for(int i=m+2;i<=n;i++)
  {
   changes[i]=changes[i]+sums;
   changes[i]=changes[i]%MODS;
   sums=sums-changes[i-m];
   sums=sums+changes[i];
   sums=sums%MODS;
   sums=sums+MODS;
   sums=sums%MODS;
  }
  for(int i=0;i<m;i++)
  {
   tmp=m-i;
   pluss=(long long)tmp*changes[n-i];
   pluss=pluss%MODS;
   ans=ans+pluss;
   ans=ans%MODS;
  }
  if(2*m>n)
  {
   for(int i=n+1;i<=2*m;i++)
   {
    ans=ans+changes[i];
    ans=ans%MODS;
   }
  }
  printf("%lld\n",ans);
 }
 //system("pause");
 return 0;
}
