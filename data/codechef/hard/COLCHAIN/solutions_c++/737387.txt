#include <cstdio>

const int mod = 1000000007;

int n, m;
long long sol[100020], ss[100020];
long long fac[100020];

int main()
{
	int i;
	long long ans, k;
	fac[0] = 1;
	for ( i = 1; i <= 100000; i++ )
	{
		fac[i] = fac[i-1] * i;
		if ( fac[i] >= mod ) fac[i] = fac[i] % mod;
	}
	int t, T;
	scanf( "%d", &T );
	for ( t = 0; t < T; ++t )
	{
		scanf( "%d %d", &n, &m );
		ans = 0;
		ss[n+1] = 0;
		for ( i = n; i > 0; i-- )
		{
//			if ( i+m > n )
				//sol[i] = 0;
			if ( i+m >= n )
				sol[i] = 1;
			else
			{
				sol[i] = ss[i+1]-ss[i+m+1]+mod;
				if ( sol[i] >= mod ) sol[i] = sol[i] - mod;
			}
			ss[i] = ss[i+1] + sol[i];
			if ( ss[i] >= mod ) ss[i] = ss[i] - mod;
		}
		k = 0;
		for ( i = 1; i <= m; i++ )
		{
			k = k + sol[i];
//			printf( "%d %lld _ %lld %lld\n", i, sol[i], k, fac[m] );
			if ( k >= mod ) k = k - mod;
			ans = ans + k*fac[m];
			if ( ans >= mod ) ans = ans % mod;
		}
		printf( "%lld\n", ans );
	}
	return 0;
}
