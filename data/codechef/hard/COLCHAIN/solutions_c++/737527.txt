#define INF 1e9
#define pb push_back
#define mp make_pair
#define forn(i,n) for (int i = 0; i < n; i++)
#define MAXN 222222
#define ADD  111111

using namespace std;

typedef long long ll;

const ll MOD = INF + 7;

ll dp[MAXN], p[MAXN];
int tests;
ll n, m;

void init();
ll ways(int k);

ll mod(ll a)
{
    return (a % MOD + MOD) % MOD;
}

void init()
{

    ll sum = m;
    for (int i = ADD; i < MAXN; i++)
    {
        dp[i] = sum;
        sum -= dp[i - m];
        sum += dp[i];
        sum = mod(sum);
    }
}

ll ways(int k)
{
    return dp[k + ADD - 1];
}

int main()
{
    //freopen("input.txt", "r", stdin);
    cin >> tests;
    p[0] = 1;
    for (ll i = 1; i < MAXN; i++)
    {
        p[i] = mod(p[i - 1] * i);
    }
    for (int i = 0; i < ADD; i++)
    {
        dp[i] = 1;
    }
    while(tests--)
    {
        cin >> n >> m;
        init();
        ll ans = 0;
        for (ll i = 0; i < m; i++)
        {
            ans = mod(ans + ((m - i)) * ways(n - m - i - 1));
        }
        cout << mod(ans * p[m]) << endl;
    }
    return 0;
}
