#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#define MODD 1000000007

long long numways[100010];

long long fact[100010];
long long N,M;
int main()
{
    int T;
    scanf("%d",&T);
    fact[0]=1;
    for(int i=1;i<=100002;i++)
        fact[i]=(fact[i-1]*i)%MODD;
    for(int t=1;t<=T;t++)
    {
        scanf("%lld%lld",&N,&M);
        long long answer=0;
        long long cumulsum=0;
        for(int i=1;i<=N;i++)
        {
            //sum from numways[i-M] to numways[i-1]
            if(i-M<=0)
            {
                numways[i]=(cumulsum+M-i+1)%MODD;
                cumulsum=(cumulsum+numways[i])%MODD;
            }
            else
            {
                numways[i]=cumulsum;
                cumulsum=(cumulsum-numways[i-M]+numways[i])%MODD;
                cumulsum=(cumulsum+MODD)%MODD;
            }
                
        }
        for(long long reppos=1;reppos<=M;reppos++)
        {
            answer=(answer+(M+1-reppos)*fact[M]%MODD*(N-M-reppos>0?numways[N-M-reppos]:1)%MODD)%MODD;
        }
        while(answer<0)
            answer%=MODD;
        printf("%lld\n",answer%MODD);
    }
}
