#include <cmath>
#include <cstdio>
#include <algorithm>
using namespace std;

#define P	99991
#define n	200005
#define DB	double
#define I64	long long
#define For(i,a,b)	for(int i=a;i<b;i++)

const	DB	Pi=acos(-1);
int		N,M,K,Ans,G[n],Num[n],rev[n];

struct	Cpx{
		DB	x,y;
		Cpx	operator +	(const Cpx&b)	{return	(Cpx){x+b.x,y+b.y};}
		Cpx	operator -	(const Cpx&b)	{return	(Cpx){x-b.x,y-b.y};}
		Cpx	operator *	(const Cpx&b)	{return	(Cpx){x*b.x-y*b.y,x*b.y+y*b.x};}
}A[n],B[n],w[2][n];

int		Pow(int a,int b){
		int	ret=1;
		for (;b;b>>=1,a=1ll*a*a%P)
			if	(b&1)	ret=1ll*ret*a%P;
		return	ret;
}
void	Pre(){
		For(i,0,M){
			w[0][i]=(Cpx){cos(2*Pi/M*i),sin(2*Pi/M*i)};
			w[1][i]=(Cpx){cos(2*Pi/M*i),-sin(2*Pi/M*i)};	rev[i]=0;
			for (int x=1,t=i;x<M;x<<=1,t>>=1)	(rev[i]<<=1)|=t&1;
		}
}
void	FFT(Cpx*A,int f){
		For(i,0,M)	if	(rev[i]>i)	swap(A[i],A[rev[i]]);
		for (int i=1;i<M;i<<=1)
		for (int j=0,t=M/(i<<1);j<M;j+=i<<1)
		for (int k=0,l=0;k<i;k++,l+=t){
			Cpx	x=A[j+k],y=A[j+k+i]*w[f][l];
			A[j+k]=x+y;	A[j+k+i]=x-y;
		}
}

void	Work(int l,int r,int f){
		if	(l==r)	{
			G[l]=Pow(!f?55048:44944,Num[l]);	return;
		}	int Mid=l+r>>1;
		Work(l,Mid,f);	Work(Mid+1,r,f);

		int L=Mid-l+2,R=r-Mid+1;M=1;
		for (;M<L||M<R;M<<=1);	M<<=1;	Pre();
		For(i,1,L)	A[i]=(Cpx){G[l+i-1],0};	A[0]=(Cpx){1,0};
		For(i,L,M)	A[i]=(Cpx){0,0};	FFT(A,0);

		For(i,1,R)	B[i]=(Cpx){G[Mid+i],0};	B[0]=(Cpx){1,0};
		For(i,R,M)	B[i]=(Cpx){0,0};	FFT(B,0);
		For(i,0,M)	A[i]=A[i]*B[i];		FFT(A,1);
		for (int i=l;i<=r;i++)	G[i]=(I64)(A[i-l+1].x/M+.5)%P;
}

int		main(){
		scanf("%d%d",&N,&K);
		for (int i=1;i<=N;i++)	scanf("%d",&Num[i]);
		Work(1,N,0);Ans+=G[K];
		Work(1,N,1);Ans-=G[K];
		printf("%d\n",(1ll*Ans*22019%P+P)%P);
}
