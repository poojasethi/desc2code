#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int main()	
{
	int t,n,f,i,j,a,count,sum;
	
	scanf("%d",&t);
	for(i=1;i<=t;i++)
	{
		
	
		scanf("%d",&n);
		scanf("%d",&f);
		int arr[n];
		
		for(j=0;j<n;j++)
		{
			scanf("%d",&a);
			while(a%f==0)
			a=a/f;
			arr[j]=a;
		}
			sort(arr,arr+n);
			sum=0;
			count=1;
			for(j=0;j<n-1;j++)
			{
				if(arr[j]!=arr[j+1])
				{
					count=0;
				}
				sum+=count;
				count++;
			}
			printf("%d\n",sum);
	}
		
		
	
}
