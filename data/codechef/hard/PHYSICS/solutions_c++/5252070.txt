#include<iostream>
#include<algorithm>
using namespace std;

typedef long long int ll;

int main()
{
	ll test;
	cin>>test;
	while(test--)
	{
    ll n,f,i,a;
    cin>>n>>f;
    ll height[n+1];
    for(i=0;i<n;i++)
    {
        cin>>a;
        while(a%f==0)
         a/=f;
         height[i]=a;
         //cout<<height[i]<<" ";
    }
    sort(height,height+n);
    ll res=height[0],ans=0,cnt=1;
    for(i=1;i<n;i++)
    {
        if(height[i]==res)
         cnt++;
         else
         {
             ans+= (cnt*(cnt-1))/2;
             
             res=height[i];
             cnt=1;
         }
    }
    ans+= (cnt*(cnt-1))/2;
    cout<<ans<<endl;
	}
    return 0;
}
