#include <iostream>
#include <cstdio>
#include <map>
#include <algorithm>
using namespace std;

int h[100005];
int n,f;
std::map<int,int> mp;

int main()
{
    int T;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d %d",&n,&f);
        for(int i=0;i<n;i++)
        {
            scanf("%d",&h[i]);
        }

        mp.clear();
        for(int i=0;i<n;i++)
        {
            int base=h[i];
            while(base%f == 0)
            {
                base=base/f;
            }
            mp[base]++;
        }

        int ans=0;
        for(map<int,int>::iterator itr=mp.begin();itr!=mp.end();itr++)
        {
            int e = itr->second;
            ans+=(e*(e-1))/2;
        }

        printf("%d\n",ans);
    }
    return 0;
}
