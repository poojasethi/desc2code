#include<cstdio>
#include<map>
#include<algorithm>

using namespace std;

int T, N, F, H[10005], brojac, j;
map<int, int> mapa;
map<int, int> ::iterator it;


int main() {
    scanf("%d", &T);
    while(T--) {
        scanf("%d %d", &N, &F);
        for(int i=1; i<=N; i++)
            scanf("%d", &H[i]);
        sort(H+1, H+N+1);
        for(int i=1; i<=N; i++) {
            for(j=H[i]; j%F==0; j/=F);
            brojac+=mapa[j]++;
        }
        //for(it=mapa.begin(); it!=mapa.end(); it++)
            //brojac+=it->second;
        printf("%d\n", brojac);
        brojac=0;
        mapa.clear();
    }
    return 0;
}
