#include<iostream>
using namespace std;
#include<map>
int main()
{
	int t;
	cin>>t;
	int n;
	int h[10000]= {0};
	map<int,int> m;
	long long int f;
	while(t--)
	{
		cin>>n>>f;
		for	(int i =0;i<n;i++)
		{
			cin>>h[i];
		}
		m.clear();
		for(int i = 0;i<n;i++)
		{
			int tem = h[i];
			while(tem%f==0){
				tem/=f;
			}
			m[tem]++;
		}
		int ans =0;
		for(map<int,int>::iterator it = m.begin();it!=m.end();it++)
		{
			int k = it->second;
			ans+=(k*(k-1))/2;
			
		}
		
		cout<<ans<<endl;
	}
}
