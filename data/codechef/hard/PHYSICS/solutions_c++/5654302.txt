#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
#define gc getchar

void scanInt(long long& x)
{
register int c = gc();
x = 0;
for(; (c < 48 || c > 57); c = gc());
for(; (c > 47 && c < 58); c = gc())
x = (x << 1) + (x << 3) + c - 48;
}
int main()
{
	long long t,n,f,i,j,a,count,sum;

	scanInt(t);
	while(t--)
	{
		scanInt(n);
		scanInt(f);
		long long arr[n];
		for(j=0;j<n;j++)
		{
			scanInt(a);
			while(a%f==0)
			a=a/f;
			arr[j]=a;
		}
			sort(arr,arr+n);
			sum=0;
			count=1;
			for(j=0;j<n-1;j++)
			{
				if(arr[j]!=arr[j+1])
				{
					count=0;
				}
				sum+=count;
				count++;
			}
			printf("%lld\n",sum);
	}
    return 0;
}
