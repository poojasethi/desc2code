#include<stdio.h>
#include<algorithm>
using namespace std;
 
#define getcx getchar_unlocked
inline void inp( int &n )//fast input function
{
n=0;
 int ch=getcx();
 int sign=1;
while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}
 
while( ch >= '0' && ch <= '9' )
n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
n=n*sign;
}
 
int main()
{
    int t, n, f, i, a, count;
    inp(t);
    while(t--)
    {
        inp(n);
        inp(f);
        int arr[n];
        for(i = 0 ; i < n; i++)
        {
            inp(a);
            while(a % f == 0)
                a /= f;
            arr[i] = a;
        }
        sort(arr, arr + n);
        count = 1;
        int ans = 0;
        for(i = 0; i < n - 1; i++)
        {
            if(arr[i] != arr[i + 1])
            {
                count = 0;
            }
            ans += count;
            count++;
        }
       printf("%d\n", ans);
    }
    return 0;
}
