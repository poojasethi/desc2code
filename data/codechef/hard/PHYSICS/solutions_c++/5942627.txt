#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
main()
{
	int t;
	cin >> t;
	while(t--)
	{
		int n,f,x,i,count,ans;
		cin >> n>>f;
		long long int a[10001];
		for(i=0;i<n;i++)
		{
			cin >> x;
			while(x%f==0)
			x=x/f;
			a[i]=x;
		}
		sort(a,a+n);
		count=1;
		ans=0;
		for(i=0;i<n-1;i++)
		{
			if(a[i]!=a[i+1])
				count=0;
			ans=ans+count;
			count++;
		}
		cout << ans<<endl;
	}
}
