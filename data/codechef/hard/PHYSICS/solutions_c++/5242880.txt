typedef unsigned long long int uint64;
typedef long long int int64;
map<int,int>m;
map<int,int>::iterator it;
int main(){
	int t,n,i;
	cin>>t;
	while(t--){
		m.clear();
		uint64 f;
		scanf("%d%llu",&n,&f);
		int ma=0;
		while(n--){
			scanf("%d",&i);
			m[i]++;
			ma=max(ma,i);
		}
		int ans=0;
		for(it=m.begin();it!=m.end();it++){
			uint64 val=it->first;
			int freq=it->second;
			ans+=(freq*(freq-1)/2);
			while(1){
				val=val*f;
				if(val>ma)
				break;
				ans+=freq*m[val];
			}
		}
		printf("%d\n",ans);
	}
	return 0;
}
