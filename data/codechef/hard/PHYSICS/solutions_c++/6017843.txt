#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    long long int t, n, f, i, a, count;
    cin >> t;
    while(t--)
    {
        cin >> n >> f;
        long long int arr[n];
        for(i = 0 ; i < n; i++)
        {
            cin >> a;
            while(a % f == 0)
                a /= f;
            arr[i] = a;
        }
        sort(arr, arr + n);
        count = 1;
        long long int ans = 0;
        for(i = 0; i < n - 1; i++)
        {
            if(arr[i] != arr[i + 1])
            {
                count = 0;
            }
            ans += count;
            count++;
        }
        cout << ans << endl;
    }
    return 0;
}
