using namespace std;

const int N = 1e4 + 5;

int t, a[N], n, f, last, res; 

map < int, int > cnt;
 
int main ()
{
//	freopen (fname".in", "r", stdin);
//	freopen (fname".out", "w", stdout);
	scanf ("%d", &t);
	while (t --)
	{
		scanf ("%d%d", &n, &f);
		for (int i = 1;i <= n;i ++)
			scanf ("%d", &a[i]);
		cnt.clear ();
		sort (a + 1, a + n + 1);
		res = 0;	
		for (int i = 1;i <= n;i ++)
		{
			ll tmp = a[i];
			last = -1;
			while (tmp)
			{
				res += cnt[tmp];
				if (tmp % f != 0)
					break;
				tmp /= f;
			}
			cnt[a[i]] ++;
		}
		printf ("%d\n", res);             				
	}
	return 0;
}
