ll power(ll b, ll e)
{
    ll p = 1;
    while (e > 0)
    {
       if(e&1)
        {
          p=(p*b)%mod;
        }
        e=e>>1;
        b=(b*b)%mod;
    }
    return p;
}

int main()
{
    ll t,n,f,i,x,j,ans,c;
    s1(t);
    while(t--)
    {
        s1(n);
        s1(f);
        ll a[n];
        for(i=0;i<n;i++)
        {
            s1(a[i]);
        }
        ans=0;
        sort(a,a+n);
        for(i=0;i<n;i++)
        {
            for(j=a[i];j<mod;j=j*f)
            {
                if(binary_search(a+i+1,a+n,j))
                    ans+=upper_bound(a+i+1,a+n,j)-lower_bound(a+i+1,a+n,j);
            }
        }
        p1(ans);
    }
return 0;
}
