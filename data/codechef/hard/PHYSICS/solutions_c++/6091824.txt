#include<iostream>
#include<algorithm>
using namespace std;

int se(int *a,int beg,int n,int x)
{
int mid,last=n-1;
while(beg<=last)
{
mid=(beg+last)/2;
if(a[mid]==x)
{
int c=0;
int p=a[mid],i=mid;
while(p==a[i]&&i<n)
{
i++;
c++;
}
for(i=mid-1;i>=beg;i--)
{
if(a[i]==p)
c++;
else
break;
}
return c;
}
else if(a[mid]>x)
last=mid-1;
else
beg=mid+1;
}
return 0;
}

long long int func(int *a,int n,long long int f)
{
long long int temp,maxi=a[n-1],c=0;
for(int i=0;i<n-1;i++)
{
temp=a[i];
while(temp<=a[n-1])
{
c=c+se(a,i+1,n,temp);
temp=temp*f;
}
}
return c;
}

int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
int n;
long long int f;
cin>>n>>f;
int a[n];
for(int i=0;i<n;i++)
cin>>a[i];
sort(a,a+n);
cout<<func(a,n,f)<<endl;
}
return 0;
}
