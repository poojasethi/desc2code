inline void readint(int &x)
{
	register int c = getchar_unlocked();
	x = 0;
	for(;(c<48 || c>57);c = getchar_unlocked());
	for(;c>47 && c<58;c = getchar_unlocked()) {x = (x<<1) + (x<<3) + c - 48;}
}

inline void writeint(int x, char endchar)
{
	if(x == 0)
		putchar_unlocked('0');
	else
	{
		int cnt = 0;
		char ch[20];
		while(x != 0)
		{
			ch[cnt++] = x%10 + '0';
			x /= 10;
		}
		for(int i=cnt - 1; i>=0; --i)
			putchar_unlocked(ch[i]);
	}
	putchar_unlocked(endchar);
}

inline void writestring(const char* ch)
{
	while(*ch != '\0')
		putchar_unlocked(*(ch++));
	putchar_unlocked('\n');
}

int a[15] = {1<<0, 1<<1, 1<<2, 1<<3, 1<<4, 1<<5, 1<<6, 1<<7, 1<<8, 1<<9, 1<<10, 1<<11, 1<<12, 1<<13, 1<<14};
int b[1<<15];
void generate_basic()
{
	b[0] = 0;
	for(int i=1; i<15; ++i)
	{
		int num = (1<<i) - 1;
		b[num] = i;
		for(int j=0; j<num; ++j)
			b[num + j + 1] = b[j];
	}
	//for(int i=0; i<(1<<14); ++i)
	// cout<<b[i]<<" ";
	//cout<<endl;
	return ;
}
void print(int a1, int n, int x, int y)
{
	int tot = 1<<n;
	int temp = a[x];
	a[x] = a[y];
	a[y] = temp;
	for(int i=0; i<tot; ++i)
	{
		writeint(a1, ' ');
		a1 = a1 ^ a[b[i]];
	}
	putchar_unlocked('\n');
	temp = a[x];
	a[x] = a[y];
	a[y] = temp;
	return ;
}
int main()
{
	generate_basic();
	int t;
	readint(t);
	while(t--)
	{
		int n, a1, b1;
		readint(n);
		readint(a1);
		readint(b1);
		int tot = 1 << n;
		int flag = -1;
		for(int i=0; i<15; ++i)
		{
			if((a1 ^ b1) == a[i])
			{
				//cout<<((a1 ^ b1) ^ a[0])<<endl;
				//cout<<i<<"Here"<<endl;
				flag = i;
				break;
			}
		}
		if(flag == -1)
		{
			print(a1, n, 0, 0);
		}
		else
		{
			if(n<=2)
			{
				putchar_unlocked('-');
				putchar_unlocked('1');
				putchar_unlocked('\n');
				continue;
			}
			if(flag == 0)
				print(a1, n, 0, 1);
			else if(flag == n-1)
			{
				print(a1, n, n-1, 1);

			else

				//cout<<"Here"<<endl;

	}
