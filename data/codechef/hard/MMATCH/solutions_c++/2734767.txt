#include <stdio.h>
#include <string.h>
 
int childrenCount, bedCount, caseCount;
 
struct Grid
{
   char c[100];
   char b[100];
   char bc[100][100];
};
 
void ClearCell(Grid * const pG, int const b, int const c, int &bCount, int &cCount)
{
   int ii;
 
   pG->bc[b][c] = 0;
   pG->b[b] = 0;
   pG->c[c] = 0;
   --bCount;
   --cCount;
 
   for (ii = 0; ii < bedCount; ++ii)
   {
      if (pG->bc[ii][c])
      {
         pG->bc[ii][c] = 0;
         --(pG->b[ii]);
         if (pG->b[ii] == 0)
         {
            --bCount;
         }
      }
   }
 
   for (ii = 0; ii < childrenCount; ++ii)
   {
      if (pG->bc[b][ii])
      {
         pG->bc[b][ii] = 0;
         --(pG->c[ii]);
         if (pG->c[ii] == 0)
         {
            --cCount;
         }
      }
   }
}

int GetTheOnes (Grid * const pG, int &bCount, int &cCount)
{
   int ii, jj, kk;
   int happy = 0;
 
   for (ii = 0; ii < bedCount; ++ii)
   {
      if (pG->b[ii] == 1)
      {
         ++happy;

         for (jj = 0; !(pG->bc[ii][jj]); ++jj);
         pG->c[jj] = 0;
         --cCount;

         for (kk = 0; kk < bedCount; ++kk)
         {
            if (pG->bc[kk][jj])
            {
               pG->bc[kk][jj] = 0;
               --(pG->b[kk]);
               if (pG->b[kk] == 1 && kk < ii)
               {
                  ii = kk - 1;
               }
               else if (pG->b[kk] == 0)
               {
                  --bCount;
               }
            }
         }
      }
   }

   for (ii = 0; ii < childrenCount; ++ii)
   {
      if (pG->c[ii] == 1)
      {
         ++happy;

         for (jj = 0; !(pG->bc[jj][ii]); ++jj);
         pG->b[jj] = 0;
         --bCount;

         for (kk = 0; kk < childrenCount; ++kk)
         {
            if (pG->bc[jj][kk])
            {
               pG->bc[jj][kk] = 0;
               --(pG->c[kk]);
               if (pG->c[kk] == 1 && kk < ii)
               {
                  ii = kk - 1;
               }
               else if (pG->c[kk] == 0)
               {
                  --cCount;
               }
            }
         }
      }
   }

   return happy;
}

int Tree (Grid g, int const b, int const c, int bCount, int cCount);
 
int FindHappy(int const minHappy, int const maxHappy, int const bCount, int const cCount, Grid const &g)
{
   int happy = minHappy;
   
   for (int ii = 0; ii < bedCount; ++ii)
   {
      for (int jj = 0; jj < childrenCount; ++jj)
      {
         if (g.bc[ii][jj])
         {
            int t = Tree(g, ii, jj, bCount, cCount) + minHappy;
            if (t > happy)
            {
               happy = t;
 
               if (maxHappy == happy)
               {
                  return happy;
               }
            }
         }
      }
   }
 
   return happy;
}
 
int Tree (Grid g, int const b, int const c, int bCount, int cCount)
{
   ClearCell(&g, b, c, bCount, cCount);
   int minHappy = GetTheOnes(&g, bCount, cCount) + 1;

   int maxHappy = (bCount <= cCount) ? bCount : cCount;
   if (maxHappy == 0)
   {
      return minHappy;
   }
   maxHappy += minHappy;
   
   return FindHappy(minHappy, maxHappy, bCount, cCount, g);
}

int TreeRoot (Grid &g)
{ 
   int bCount = 0;
   int cCount = 0;
 
   for (int ii = 0; ii < bedCount; ++ii)
   {
      if (g.b[ii])
      {
         ++bCount;
      }
   }
 
   for (int ii = 0; ii < childrenCount; ++ii)
   {
      if (g.c[ii])
      {
         ++cCount;
      }
   }
   
   int minHappy = GetTheOnes(&g, bCount, cCount);
   
   int maxHappy = (bCount <= cCount) ? bCount : cCount;
   if (maxHappy == 0)
   {
      return minHappy;
   }
   maxHappy += minHappy;
   
   return FindHappy(minHappy, maxHappy, bCount, cCount, g);
} 
 
int main()
{
   int tcCount, ii;
   Grid g;
   scanf("%d", &tcCount);
 
   for (ii = 0; ii < tcCount; ++ii)
   {
      scanf("%d %d", &childrenCount, &bedCount);
      if (childrenCount > 100 || bedCount > 100) return -1; 
 
      memset(&g, 0, sizeof(struct Grid));
 
      while (1)
      {
         int c, b;
 
         scanf("%d %d", &c, &b);
 
         if (c == 0 && b == 0) break;
         --b;
         --c;
 
         g.bc[b][c] = 1;
         ++(g.c[c]);
         ++(g.b[b]);
      }
 
      printf("%d\n", TreeRoot(g));
   }
 
   return 0;
}
