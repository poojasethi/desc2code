using namespace std;
const int n_max=100005;
int n,Q,i,o,m,x,y;
struct Ele{
       int ANS,l,r,L,R;
};
Ele a[4*n_max];

void add(int k,int x){
     a[k].ANS=a[k].L=a[k].R=1;
     a[k].l=a[k].r=x;
     int  e=1;
     while(k>1){
        k/=2;

        a[k].l=a[2*k].l;
        a[k].r=a[2*k+1].r;

        a[k].ANS=MAX(a[2*k].ANS,a[2*k+1].ANS);
        if(a[2*k].r==a[2*k+1].l-1)a[k].ANS=MAX(a[k].ANS,a[2*k].R+a[2*k+1].L);


        a[k].L=a[2*k].L;
        if(a[2*k].L==e && a[2*k].r==a[2*k+1].l-1)a[k].L+=a[2*k+1].L;

        a[k].R=a[2*k+1].R;
        if(a[2*k+1].R==e && a[2*k].r==a[2*k+1].l-1)a[k].R+=a[2*k].R;
      //  cout<<k<<' '<<a[k].ANS<<' '<<a[k].L<<' '<<a[k].R<<endl;
        e*=2;
     }
     //system("pause");
}


main()
{scanf("%d%d",&n,&Q);

 m=1;
 while(m<n)m*=2;
 for(i=1;i<=n;i++)scanf("%d",&o),add(m+i-1,o);


 printf("%d\n",a[1].ANS);
 while(Q--){
    scanf("%d%d",&x,&y);
    add(m+x-1,y);
    printf("%d\n",a[1].ANS);
 }

}
