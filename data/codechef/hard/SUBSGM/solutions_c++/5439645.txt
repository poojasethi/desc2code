#include <iostream>
#define si 100005
using namespace std;

struct data {int pref,suff,ans;} ;
data t[si*4]={0};
int x[si];

data combine ( data l, data r,int ss,int se, int m ) {
	data res ;
	res.pref=l.pref;
	res.suff=r.suff;
	res.ans=max ( l. ans , r. ans );
	if(l.pref==m-ss+1&&x[m+1]-x[m]==1)
	res. pref = max ( l. pref , m-ss+1 + r. pref ) ;
	if(r. suff == se-m&&x[m+1]-x[m]==1)
	res. suff = max ( r. suff , se-m + l. suff ) ;
	if(x[m+1]-x[m]==1)
	res. ans = max ( max ( l. ans , r. ans ) , l. suff + r. pref ) ;
	return res ;
}

data make_data (  ) {
	data res ;

	res. pref = res. suff = res. ans = 1  ;
	return res ;
}

void ct(int ss,int se,int in )
{ int m= ss + (se -ss)/2,i;
   if(ss==se)


{t[in]=make_data();
//cout<<t[in].ans<<endl;
return ;}

ct(ss,m,in*2);
ct(m+1,se,in*2+1);

    t[in]=combine(t[in*2],t[in*2+1],ss,se,m);
return;
}

int upd (int q,int ss,int se,int in)
{
    if (ss==q&&se==q)

          return 0;


        if (se < q || ss > q)
        return 0;

        int m= ss + (se -ss)/2;


        upd(q,ss,m,in*2);
        upd(q,m+1,se,in*2+1);
        t[in]=combine(t[in*2],t[in*2+1],ss,se,m);
}

int main()
{
    int n,i,j,m,p,y;
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    data ans;
    cin>>n>>m;
    for(i=1;i<=n;i++)
        cin>>x[i];
    ct(1,n,1);
    ans=t[1];
    cout<<ans.ans<<endl;
    for(i=1;i<=m;i++)
    {
        cin>>p>>y;
        x[p]=y;
        upd(p,1,n,1);
        ans=t[1];
        cout<<ans.ans<<endl;
    }

}
