#define mod 1000000007
const ll mod2 = mod-1;
const int p2 = (mod-1)>>1;
const int maxn = 1e5+100;
ll fact1[maxn],fact2[maxn];
ll powr(ll a, ll b,ll c)
{
    a%=c;
    ll x = 1;
    while(b)
    {
        if(b&1)
            x=(x*a)%c;
        a=(a*a)%c;
        b>>=1;
    }
    return x;
}
ll exp2(ll n)
{
    ll ret = 0;
    n>>=1;
    while(n)
    {
        ret+=n;
        n>>=1;
    }
    return ret;
}

ll exp2C(ll n,ll k)
{
    return exp2(n) - exp2(k)-exp2(n-k);
}
void pre()
{
    fact1[0] = fact1[1]=1;
    fact2[0]=fact2[1]=1;
    for(int i = 2;i<maxn;i++)
    {
        fact1[i] = (fact1[i-1]*i)%mod;
        ll _prod = i;
        while(_prod%2 == 0)
            _prod/=2;
        fact2[i] = (fact2[i-1]*_prod)%mod2;
    }
}
ll inv(ll a,ll b)
{
    return powr(a,b-2,b);
}
ll md1(ll n, ll k)
{
    if(n<k||k<0)
        return 0;
    if(n==k||k==0)
        return 1;
    ll a = fact1[n],b = fact1[k],c = fact1[n-k];
    return (a*inv((b*c)%mod,mod))%mod;
}
ll md2(ll n, ll k)
{
    if(n<k||k<0)
        return 0;
    if(n==k||k==0)
        return 1;
    ll a = fact2[n],b = fact2[k],c = fact2[n-k];
    ll _exp2 = exp2C(n,k);
    return (powr(2,_exp2,mod2)*((a*powr((b*c)%mod2,p2 -2, mod2))%mod2))%mod2;
}
ll ans1(ll prod, ll n)
{
    ll x = n/2;
    ll ret = powr(prod,md2(n-1,x-1),mod);
    if(n&1)
        ret+=powr(prod,md2(n-1,x),mod);
    return ret%mod;
}

{   pre();
    int t;

    while(t--)
    {

        ll len = md1(n,n/2);
        ll prod = 1;
        for(int i = 0;i<n;i++)
        {
            scanf("%lld%lld",&a,&b);
            prod=(prod*b)%mod;
        }
        if(n==1)
        {
            cout<<1<<" "<<(1+prod)%mod<<endl;
            continue;
        }
        cout<<len<<" "<<ans1(prod,n)<<endl;
    }

}
