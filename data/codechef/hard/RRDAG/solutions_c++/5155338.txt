#include <cstdio>
#include <cstring>
#include <queue>
#include <algorithm>
using namespace std;

const int N = 1500+5;
priority_queue<int> qu;
int n, deg[N], g[N][N];
char s[N][N];

int main() {
    scanf("%d", &n);
    for(int i = 0;i < n; i++)
        scanf("%s", s[i]);
    int sum = 0;
    for(int i = 0;i < n; i++) {
        for(int j = 0;j < n; j++) if(s[i][j] == '1')
            deg[i]++, sum++;
        if(deg[i] == 0) qu.push(i);
    }
    while(!qu.empty()) {
        int u = qu.top(); qu.pop();
        for(int i = 0;i < n; i++) if(s[i][u] == '0' && s[u][i]=='0' && !g[u][i]) {
            if(i != u)
                g[i][u] = 1;
        }
        for(int i = 0;i < n; i++) if(s[i][u] == '1') {
            if(!(--deg[i])) qu.push(i);
        }
    }
    printf("%d\n", n*(n-1)/2 - sum);
    for(int i = 0;i < n; i++) {
        for(int j = 0;j < n; j++) if(g[i][j])
            printf("%d %d\n", i+1, j+1);
    }
    return 0;
}
