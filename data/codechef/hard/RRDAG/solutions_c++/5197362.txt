#include <bits/stdc++.h>

#define REP(i, n) for (int i = 0; i < (int) (n); ++i)
#define REPP(i, a, b) for (int i = (int) (a); i <= (int) (b); ++i)

using namespace std;

const int N = 1505;
int n, m;
int out[N];
char mp[N][N];
bool vis[N];

int main() {
	//freopen("1.in", "r", stdin);
	
	scanf("%d", &n);

	REPP(i, 1, n) scanf("%s", mp[i] + 1);
	REPP(i, 1, n) REPP(j, 1, n) {
		mp[i][j] -= '0';
		if (mp[i][j]) out[i]++, m++;
	}
	printf("%d\n", n * (n - 1) / 2 - m);

	int cnt = n;
	vector<pair<int, int> > ans;
	while (cnt) {
		for (int i = n; i >= 1; --i) if (out[i] == 0 && !vis[i]) {
			vis[i] = 1;
			cnt--;
			REPP(j, 1, n) if (!vis[j]) {
				if (mp[j][i]) {
					out[j]--;
				}
				else if (i != j) {
					ans.push_back(make_pair(j, i));
				}
			}
			break;	
		}
	}
	sort(ans.begin(), ans.end());
	REP(i, ans.size()) {
		printf("%d %d\n", ans[i].first, ans[i].second);
	}

	return 0;
}
