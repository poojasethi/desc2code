int used[maxN],outdeg[maxN];
int A[maxN][maxN];

int main() {

int N;
cin>>N;

for(int i=0;i<N;i++) {
	string s;
	cin>>s;
	for(int j=0;j<N;j++) {
		if(s[j]=='1') {
			outdeg[i]++;
			A[i][j]=1;
		}
	}
}

vector<pii> edges;
edges.clear();
int n=N;
while(n--) {
	int v=-1;
	for(int i=0;i<N;i++) 
		if(outdeg[i]==0&&used[i]==0)
			v=i;
	used[v]=1;
	for(int i=0;i<N;i++)
		if(used[i]==0&&A[i][v]==0) {
			edges.pb(mp(i+1,v+1));
			}
	for(int i=0;i<N;i++)
		if(A[i][v])
			--outdeg[i];
}

sort(edges.begin(),edges.end());
cout<<edges.size()<<"\n";
for(int i=0;i<edges.size();i++)
 cout<<edges[i].ff<<" "<<edges[i].ss<<"\n";

}
