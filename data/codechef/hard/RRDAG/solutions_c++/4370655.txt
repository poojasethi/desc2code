#include <iostream>
#include <stdio.h>
#include <algorithm>
#include <utility>
#include <vector>
#include <string.h>

#define sz 1505

using namespace std;

int arr[sz][sz];
int top[sz];
int c[sz];
int cnt;
int n;

void solve(vector<int> a) {
	if(a.size() == 0) return;
	memset(c, 0, sizeof(c));
	for(int i=0;i<a.size();i++) c[a[i]] = 1;
	vector<int> que;
	c[a[0]] = 2;
	que.push_back(a[0]);
	for(int ck=0;ck<que.size();ck++) for(int k=que[ck],i=0;i<n;i++) if(arr[i][k] && (c[i]==1)) {c[i]=2; que.push_back(i);}
	vector<int> bb,cc;
	for(int i=1;i<a.size();i++) {
		if(c[a[i]] == 2) bb.push_back(a[i]);
		else cc.push_back(a[i]);
	}
	solve(bb);
	top[a[0]] = (cnt++);
	solve(cc);
}

int main() {
	scanf("%d",&n);
	for(int i=0;i<n;i++) {
		char tch[sz];
		scanf("%s",tch);
		for(int j=0;j<n;j++) arr[i][j] = (tch[j] == '1');
	}
	cnt = 0;
	vector<int> a;
	for(int i=0;i<n;i++) a.push_back(i);
	solve(a);
	int m = 0;
	for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(top[i] < top[j] && !arr[i][j]) m++;
	printf("%d\n",m);
	for(int i=0;i<n;i++) for(int j=0;j<n;j++) if(top[i] < top[j] && !arr[i][j]) printf("%d %d\n",i+1, j+1);
	return 0;
}
