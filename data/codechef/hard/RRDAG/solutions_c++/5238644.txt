#include <bits/stdc++.h>
using namespace std;

typedef pair<int, int> pii;

int a[1511][1511];
int outd[1511];
bool Remove[1511];
char str[1511];

int main() {

	int n = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf(" %s", str);
		for (int j = 0; j < n; ++j) {
			a[i][j] = str[j] - '0';
			if (a[i][j]) outd[i]++;
		}
	}

	vector<pii> ans;

	for (int x = 0; x < n; ++x) {
	
		int v = 0;
		for (int j = n - 1; j >= 0; --j) if (!Remove[j] && outd[j] == 0) {
			v = j;
			break;
		}
		Remove[v] = true;
		for (int j = 0; j < n; ++j) if (!Remove[j] && !a[j][v]) {
			ans.push_back(pii(j, v));
		}
		for (int j = 0; j < n; ++j) if (!Remove[j] && a[j][v]) --outd[j];
	}
	printf("%d\n", ans.size());
	sort(ans.begin(), ans.end());
	for (int i = 0; i < ans.size(); ++i) printf("%d %d\n", ans[i].first+1, ans[i].second+1);
//	for (auto x : ans) printf("%d %d\n", x.first+1, x.second+1);

	return 0;
}
