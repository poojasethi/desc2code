void fscani(int *x)
{
	int n=0;int sign=1;char c=getcx();
	while(c<'0' || c>'9'){if(c=='-') sign=-1;c=getcx();}
	while(c>='0' && c<='9'){n=(n<<1)+(n<<3)+(c-'0');c=getcx();}
	n=n*sign;*x=n;
}

bool a[2510][2510];
int n;
int out[2510];
bool edges[2510][2510];
bool marked[2510];
int main()
{
	S(n);
	int ans = 0;
	char s[n+1];
	FOR(i,n)
	{
		SS(s);
		FOR(j,n)
		{
			a[i][j] = s[j]-'0';
			if(a[i][j])
				out[i]++;
		}
	}
	FOR(i,n)
	{
		int vertex = -1;
		for(int j=n-1;j>=0;j--)
			if(!marked[j] && out[j] == 0) 
			{
				vertex = j;
				marked[j] = true;
				break;
			}
		assert (out[vertex] == 0);
		FOR(j,n)
			if((j != vertex) && (a[j][vertex] == 0) && (!marked[j])) 
			{
				edges[j][vertex] = 1;ans++;
			}
		FOR(j,n)
			if(a[j][vertex] == 1)
			{
				out[j]--;
			}
	}
	printf("%d\n",ans);
	FOR(i,n)
		FOR(j,n)
			if(edges[i][j])
				printf("%d %d\n",i+1,j+1);
