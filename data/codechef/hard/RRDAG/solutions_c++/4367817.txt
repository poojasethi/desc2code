#define maxn 401

ifstream fin ("C.in");
ofstream fout ("C.out");

char s[2002][2002];
bool viz[2002];
int n,pred[2002],t;
vector<pair<int,int> > ans;

int main()
{
    //freopen ("C.in","r",stdin);

    scanf ("%d",&n);

    for (int i=1; i<=n; ++i)
    {
        scanf ("%s",s[i]+1);

        for (int j=1; j<=n; ++j)
        {
            if (s[i][j] == '1')
                pred[i]++;
        }
    }

    for (int i=1; i<=n; ++i)
    {
        for (int j=n; j>=1; --j)
        {
            if (!viz[j] && !pred[j])
            {
                viz[j] = 1;
                for (int k=1; k<=n; ++k)
                {
                    if (s[k][j] == '1')
                      --pred[k];
                    else if (!viz[k])
                        ans.push_back (make_pair(k,j));
                }
                break;
            }
        }
    }

    sort (ans.begin(),ans.end());

    cout<<ans.size()<<"\n";

    for (int i=0; i<ans.size(); ++i)
    {
        cout<<ans[i].first<<" "<<ans[i].second<<"\n";
    }

}
