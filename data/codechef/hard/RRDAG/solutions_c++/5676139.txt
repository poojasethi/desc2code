#include <bits/stdc++.h>
using namespace std;
typedef pair <int,int> ii;
#define N 1500
char s[N + 1][N + 2];
vector <ii> ans;
int cnt[N + 1];
int mark[N + 1];
int main()
{
    int n,i,j;
    scanf("%d",&n);
    for(i = 1 ; i<=n ; i++)
    {
        scanf("%s",s[i]+1);
        for(j = 1 ; j<=n ; j++)
        {
            if(j == i)
            {
                s[i][i] = '0';
                continue;
            }
            if(s[i][j] == '1') cnt[i]++;
        }
    }
    while(1)
    {
        int save = -1;
        for(i = 1 ; i<=n ; i++)
        {
            if(cnt[i] == 0 && mark[i] == 0) save = i;
        }
        if(save == -1) break;
        for(i = 1 ; i<=n ; i++)
        {
            if(mark[i] == 1 && s[save][i] == '0')
            {
                ans.push_back(ii(save,i));
            }
            if(s[i][save] == '1')
            {
                cnt[i]--;
            }
        }
        mark[save] = 1;
    }
    int sz = ans.size();
    sort(ans.begin(),ans.end());
    printf("%d\n",sz);
    for(i = 0 ; i<sz ; i++)
    {
        printf("%d %d\n",ans[i].first,ans[i].second);
    }
    return 0;
}
