#include <cstdio>
#include <cmath>
#include <iostream>
#include <set>
#include <algorithm>
#include <vector>
#include <map>
#include <cassert>
#include <string>
#include <cstring>

using namespace std;

#define rep(i,a,b) for(int i = a; i < b; i++)
#define S(x) scanf("%d",&x)
#define P(x) printf("%d\n",x)

typedef long long int LL;
typedef pair<int, int > pii;
const int MAXN = 1505;

bool used[MAXN];
vector<pii > ans;
int in[MAXN];
char s[MAXN][MAXN];
int out[MAXN];

int main() {
	int n;
	S(n);
	rep(i,0,n) {
		scanf("%s",s[i]);
		rep(j,0,n) {
			if(s[i][j] == '1') {
				out[i]++;
			}
		}
	}


	rep(i,0,n) {
		int id = -1;
		for(int j = n-1; j >= 0; j--) if(!used[j] && (id == -1 || out[j] < out[id])) {
			id = j;
		}

		used[id] = 1;
		rep(j,0,n) if(!used[j] && s[id][j] == '0' && s[j][id] == '0' && id != j) {
			ans.push_back(make_pair(j+1, id+1));
			s[j][id] = '1';
			out[j]++;
		}

	}
	sort(ans.begin(), ans.end());
	P((int)ans.size());
	rep(i,0,ans.size()) {
		printf("%d %d\n",ans[i].first, ans[i].second);
	}
	return 0;

}
