#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
#define N 2222
typedef pair<int,int> pii;
char a[N][N],p=0;
bool vis[N],b[N][N];
int outd[N];
int main(){
	int i,j,k,n,e = 0;
	p=0;
	memset(b,0,sizeof(b));
	memset(outd,0,sizeof(outd));
	scanf("%d",&n);int ans=n*(n-1)/2;
	for(i = 0;i<n;i++){
		scanf("%s",a[i]);
		for(j = 0;j<n;j++){
			e+= a[i][j] -  '0';
			if(a[i][j]=='1')outd[i]++,--ans;
		}
	}
	for(i = 0;i<n;++i){//order is important 
		int p=-1;for(j=0;j<n;++j)if(!vis[j] && !outd[j])p=j;
		for(j = 0;j<n;j++){
			if(j==p || vis[j])continue;
			if(a[j][p]=='1') {
				--outd[j];
			}else b[j][p] = 1;
		}
		vis[p]=1;
	}
	printf("%d\n",ans);//must be n-1 + n-2 + .. + 1 
	for(i = 0;i<n;i++)for(j = 0;j<n;j++)if(b[i][j])printf("%d %d\n",i+1,j+1);
	return 0;
}
