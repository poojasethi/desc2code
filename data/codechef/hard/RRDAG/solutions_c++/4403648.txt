#include <iostream>
#include <bits/stdc++.h>
using namespace std;

char a[1505][1505];
bool vis[1505],b[1505][1505];
int outDegree[1505];

int main()
{
    int i,j,n,e=0;
    vector<pair<int,int> > eg;

    scanf("%d",&n);

    for(i=0;i<n;i++)
    {
        scanf("%s",a[i]);
        for(j=0;j<n;j++)
        {
            e+=a[i][j]-'0';
            if(a[i][j]=='1')
            {
                outDegree[i]++;
            }
        }
    }

    int cnt=0,p=0;
    while(1)
    {
        for(i=n-1;i>=0;i--)
        {
            if(vis[i]||outDegree[i])continue;

            for(j=0;j<n;j++)
            {
                if(a[j][i]=='1')
                {
                    outDegree[j]--;
                    continue;
                }

                if(j==i||vis[j])continue;
                b[j][i]=1;
                p++;
            }

            vis[i]=1;
            break;
        }

        if(i<0)break;
        cnt++;
    }

    printf("%d\n",p);

    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(b[i][j])
                printf("%d %d\n",i+1,j+1);
        }
    }

    return 0;
}
