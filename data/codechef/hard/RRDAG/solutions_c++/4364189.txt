#include <iostream>
#include <memory.h>
#include <cstdio>
using namespace std;
int main()
{
    int n,i,j;
    cin >> n;
    char ch[1505][1505];
    bool NEW[n+1][n+1];
    int outdegree[n+1];
    int vis[n+1];
    memset(NEW, 0 , sizeof(NEW));
    memset(vis,0,sizeof(vis));
    memset(outdegree,0,sizeof(outdegree));
    for ( i = 0 ; i < n ; i++)
    {
        scanf("%s",ch[i]);
        for ( j = 0 ; j < n ; j++)
        {
            if(ch[i][j] == '1')
            {
                outdegree[i] ++;
            }
        }
    }
    int cnt = 0;
    while(1)
    {
        for(i = n - 1 ; i >= 0 ; i--)
        {
            if(vis[i] || outdegree[i])
              continue;
            vis[i] = 1;
                for(j = 0; j < n ; j++)
                {
                    if(ch[j][i] == '1')
                    {
                        outdegree[j] --;
                        continue;
                    }
                    if( j == i || vis[j])
                      continue;
                    NEW[j][i] = 1;
                    cnt++;
                }
            
              break;
        }
        if(i < 0)
           break;
    }
    
    cout << cnt << endl;
    for(i = 0 ; i < n ; i++)
    {
        for(j= 0; j < n ; j++)
        {
            if(NEW[i][j] == 1)
            printf("%d %d\n",i+1, j+1);
        }
    }
    return 0;
}
