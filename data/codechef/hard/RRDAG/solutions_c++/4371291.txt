#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;
char A[1501][1501];
bool visited[1501],B[1501][1501];
int n,u,outdegree[1501];
static int res=0;
inline void toposort()
{
	while(true)
	{
		u=-1;
		for(int i=n-1;i>=0;i--)
		if(!visited[i]&&outdegree[i]==0)
		{
			u=i;
			break;
		}
		
		if(u==-1)break;
		
		visited[u]=true;
		for(int j=0;j<n;j++)
		if(j!=u&&!visited[j]&&A[j][u]!='1')
		{
			res++;
			B[j][u]=true;
		}
		
		for(int i=0;i<n;i++)if(A[i][u]=='1')outdegree[i]--;
	}
	return;
}
int main()
{
	scanf("%d",&n);
	for(int count,i=0;i<n;i++)
	{
		scanf("\n%[^\n]",A[i]);
		count=0;
		for(int j=0;j<n;j++)if(A[i][j]=='1')count++;
		outdegree[i]=count;
	}
	
	toposort();
	printf("%d\n",res);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)if(B[i][j])printf("%d %d\n",i+1,j+1);
	}
	return 0;
}
