#include <stdio.h>
#include <string.h>
#include <vector>
#include <algorithm>
#include <queue>

#define MOD 1000000007

using namespace std;

typedef long long int lld;
typedef pair<int, int> pii;
typedef pair<pii, pii> line;

#define F first
#define S second

char mat[2020][2020];
int mark[2020];
int in[2020];
pii ls[2020];
priority_queue<int> pq;
vector<pii> ans;

int main()
{
  memset(mark, 0, sizeof mark);
  memset(in, 0, sizeof in);
  int n, i, j, k, ct = 0;
  scanf("%d", &n);

  for (i = 0; i < n; i++)
  {
    scanf(" %s", mat[i]);
    for (j = 0; j < n; j++)
      if (mat[i][j] == '1')
        in[i]++;
  }

  for (i = 0; i < n; i++)
    if (in[i] == 0)
      pq.push(i);

  while (!pq.empty())
  {
    int tp = pq.top();
    pq.pop();
    mark[tp] = 1;

    for (i = 0; i < n; i++)
      if (!mark[i] && mat[i][tp] == '0')
        ans.push_back(pii(i, tp));
      else if (!mark[i])
      {
        in[i]--;
        if (in[i] == 0)
          pq.push(i);
      }
  }

  sort(ans.begin(), ans.end());

  printf("%d\n", (int)ans.size());
  for (i = 0; i < (int)ans.size(); i++)
    printf("%d %d\n", ans[i].F + 1, ans[i].S + 1);
  
    
  return 0;
}
