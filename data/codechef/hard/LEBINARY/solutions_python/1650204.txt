A = [2, 2]
for i in range(3, 3400):
	if (i&1 == 0):
		A.append(A[-1]*2)
	else:
		A.append((A[-1]*(i-2))/(i/2))
	
#print len(str(A[-1]))

def bsearch(x):
	s = 0
	e = len(A)-1
	while (s <= e):
		mid = (s+e)/2
		if (A[mid] == x):
			return mid
		elif (A[mid] < x):
			s = mid+1
		else:
			e = mid-1
	return -1

T = int(input())
for t in range(T):
	N = (int)(input())
	if (N&1 != 0):
		print "NO"
	else:
		if (bsearch(N) == -1):
			print "NO"
		else:
			print "YES"
