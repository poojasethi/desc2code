import math
N=3400
f=[0]*(N+100)
def prep():
	f[1]=1
	for i in range(2,N+1):
		if(i&1):f[i]=f[i-1]*i/(i-(i/2))
		else :f[i]=f[i-1]*i/(i/2)
	for i in range(1,N+1):f[i]=f[i]*2
prep()
T=int(input())
while T:
	n=int(input())
	flag=0
	for i in range(1,N+1):
		if(f[i]==n):
			flag=1
			break
		if(f[i]>n):break
	if(flag):print("YES")
	else : print("NO")
	T-=1
