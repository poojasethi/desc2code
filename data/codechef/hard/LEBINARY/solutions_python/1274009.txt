a=set([2,4])
lastres = 2
i=2
while len(str(lastres))<=1000:
    lastres = lastres*2*(2*i-1)/(i)
    a.add(lastres)
    a.add(2*lastres)
    i+=1
    
t = int(raw_input())
for i in range(t):
    n = int(raw_input())
    if n in a:
        print "YES"
    else:
        print "NO"
