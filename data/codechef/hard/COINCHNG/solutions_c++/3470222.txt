int c,s;ull k,f[60][505];
int init()
{
	int i,j;
	for(i=0;i<=s;i++)f[0][i]=i<c;
	for(i=1;i<60;i++)
	{
	for(j=0;j<=s;j++)
	{
		f[i][j]=0;
		for(int n=0;n<=min(j,c-1);n++)
		f[i][j]+=f[i-1][j-n];
	}
	if(f[i][s]>=k)return i;
	}
	return -1;
}
ull pow(ull a,int b)
{
	ull res=1;
	while(b)
	{
		if(b&1)res=res*a;
		a*=a,b>>=1;
	}
	return res;
}
ull query(int i,int s,ll k)
{
	if(i==0)return s;
	ull res=0;
	for(int n=0;n<c;n++)
	{
		if(f[i-1][s-n]>=k)
		{
			res+=query(i-1,s-n,k);break;
		}
		k-=f[i-1][s-n];
		res+=pow((ull)c,i);
	}
	return res;
}

{
    int t,i,j,T,n;

	while(T--)
	{
		scanf("%d%d%lld",&c,&s,&k);
		i=init();
		printf("%llu\n",query(i,s,k));
	}
    return 0;
}
