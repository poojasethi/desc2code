ll dp[100][600];
int main()
{
	int T;
	cin>>T;
	while(T--) {
		int C, S;
		ll K;
		cin>>C>>S>>K;
		if(S==0) {
			cout<<0<<endl;
			continue;
		}
		CLR(dp,0);
		int i, j;
		dp[0][0] = 1;
		int p = 1;
		bool ok = false;
		int d;
		while(true) {
			FOR(i,0,C) if(i<=S) {
				FOR(j,0,S+1) dp[p][j+i] += dp[p-1][j];
				if(i==0) continue;
				if(K<=dp[p-1][S-i]) {
					ok = true;
					d = i;
					break;
				} else {
					K -= dp[p-1][S-i];
				}
			}
			if(ok) break;
			p++;
		}
		ll ans = 0;
		ans = ans * C + d;
		p--;
		S -= d;
		while(p>0) {
			FOR(i,0,C) {
				if(dp[p-1][S-i] >= K) break;
				K -= dp[p-1][S-i];
			}
			ans = ans * C + i;
			S -= i;
			p--;
		}
		cout<<ans<<endl;
	}
	return 0;
}
