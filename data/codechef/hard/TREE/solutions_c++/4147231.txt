//fast function to scan various integer data types
template <typename T>
inline T input() {
	T n = 0;
	int s = 1;
	char p = get();

	while((p < '0' || p > '9') && p != EOF && p != '-')
		p = get();

	if(p == '-'){ s = -1, p = get(); }

	while(p >= '0' && p <= '9') {
		n = (n << 3) + (n << 1) + (p - '0');
		p = get();
	}
	return n*s;
}

lli mod = 1000000007;

inline long long multiply(long long a,long long b) // O(1) for (a*b)%m
{
	a %= mod;
	b %= mod;
	long double res = a;
	res *= b;
	long long c = (long long)(res / mod);
	a *= b;
	a -= c * mod;
	a %= mod;
	if (a < 0) a += mod;
		return a;
}
 
inline long long power(long long a,long long b)
{
	long long ans=1;
	while(b)
	{
		if(b&1)
		{
			ans=multiply(ans,a);
		}
		a=multiply(a,a);
		b>>=1;
	}
	return ans;
}
