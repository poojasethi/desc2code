#include<iostream>
#include<cstdio>
#define M 1000000007
using namespace std;
long long int expo(long long int a,long long int b)
{
long long int res=1;
while(b>0)
{
if(b&1)
res=(res*a)%M;
a=(a*a)%M;
b/=2;
}
return res;
}
 
int main()
{
long long int n,k;
cin>>n>>k;
if(k==1)
{
if(n==1)
{
cout<<"1\n";
}
else if(n<k)
cout<<"0\n";
else
cout<<((expo(n*k,n*k-2))%M)%M<<endl;
}
else
{
//(n^(n-2)) * ((n-1)^(n*(k-1))) * (k^(n*k-2))
//cout<<(((expo(n,n-2)*(expo(n-1,(n*(k-1)))))%M)*(expo(k,(n*k-2)))%M<<endl;
cout<<(((expo(n,n-2)*expo(n-1,n*(k-1)))%M)*expo(k,n*k-2))%M<<endl;
}
//system("pause");
}
