#include<bits/stdc++.h>
using namespace std;
#define MOD 1000000007
#define ll long long
ll exp(ll a,ll b)
{
	ll r=1;
	ll m=a;
	while(b>0)
	{
		if(b&1)	r=(r*m)%MOD;
		m=(m*m)%MOD;
		b>>=1;
	}
	return r;
}
int main()
{
	int n,k;scanf("%d%d",&n,&k);
	ll t=exp((1LL*k*n),1LL*n-2LL);
	t=(t*exp(1LL*k*(n-1),1LL*(k-1)*n))%MOD;
	cout<<t<<endl;
	return 0;
}
