#include <cstdio>
#include <cassert>

#define MOD 1000000007

inline int mul(int a, int b){
	return (long long)a*b%MOD;

}

int power(int base, long long exponent){
	int sq=base, res=1;
	for(; exponent; exponent>>=1){
		if(exponent&1)
			res=mul(res, sq);
		sq=mul(sq, sq);
	}
	return res;
}

// N^(N-2) * (N-1)^((K-1)*N) * K^(K*N-2)
int solve(int n, int k){
	return n == 1 ? k==1 : mul(mul(power(n, n-2), power(n-1, (k-1)*n)), power(k, k*n-2));
}

int main(){
	int N, K;
	scanf("%d %d", &N, &K);
	printf("%d\n", solve(N, K));
	return 0;
}
