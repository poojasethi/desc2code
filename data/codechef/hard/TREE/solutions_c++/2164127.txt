#include<stdio.h>

unsigned long long int pow(unsigned long long int a,unsigned long long int b,unsigned long long int m)
{
   unsigned  long long int res=1;
   while(b>0)
   {
        if(b&1)  res=(res*a)%m;
        b=b>>1;   a=(a*a)%m;
   }
return res;
}


int main()
{
    unsigned long long int n,k;
    unsigned long long int m=1000000007;
    scanf("%llu%llu",&n,&k);
    
    unsigned long long int temp1=pow(k*n,n-2,m);
     unsigned long long int temp2=pow(k*n-k,k-1,m);
     temp2=pow(temp2,n,m);
     unsigned long long int ans=(temp1 * temp2) %m;
    printf("%llu",ans);
}
