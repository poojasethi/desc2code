#include<stdio.h>
#define mod 1000000007
long long int cal(long long int n,long long int k)
{
    long long odd=1;
    if(k==0)
    return 1;
    while(k>0)
    {
        if(k%2)
        odd=(odd*n)%mod;
        k/=2;
        n=(n*n)%mod;
        
    }
    return odd%mod;
}
