#include <iostream>

using namespace std;

const int mod =  1000000007;
/*
idea 
(k*n)^n-2)*(k*(n-1))^(k-1)*n)
*/

typedef long long LL;

//(a^b)%mod
LL mod_pow(LL a, LL b, LL mod){
    if(!b)return 1;
    LL temp = mod_pow(a,b/2,mod);
    temp = (temp*temp)%mod;
    if(b&1)temp = (temp*a)%mod;
    return temp;
}
int main() {
	int n,k;
    cin>>n>>k;
    LL p1 = mod_pow(k*n, n-2,mod);
    LL p2 = mod_pow(k*(n-1),(k-1)*n,mod);
    LL ret = (p1*p2)%mod;
    cout<<ret<<endl;
	return 0;
}
