
#include<stdio.h>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long LL;
#define MOD 1000000007

int power(LL a,LL n)
{
     LL r =1;
    for(;n>0;n>>=1)
   {
         if(n&1) r = (r*a)%MOD;
          a = (a*a)%MOD;
   }
      return r;
 }
int main()
{ 
   long long n,k,ans,res;
   scanf("%lld %lld",&n,&k);
  
   ans =power(k*n,n-2);
   res = power(k*(n-1),(k-1)*n);
   res = (ans*res)%MOD;
   printf("%lld",res);
  return 0;
}
