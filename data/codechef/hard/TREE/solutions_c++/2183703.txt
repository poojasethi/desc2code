#include <cstdio>
#include <cstdlib>
#define ull unsigned long long
#define mod 1000000007
using namespace std;

ull power(ull base,ull exp)
{
	ull prod=1,b=base;
	while(exp!=0)
	{
		if(exp&1)
		{
			prod=(prod*b)%mod;
		}
		b=(b*b)%mod;
		exp>>=1;
	}
	return prod;
}

int main()
{
	int n,k;
	scanf("%d%d",&n,&k);
	ull num;
	if(k==1)
	{
		if(n==1)
		{
			printf("1\n");
		}
		else
		{
			printf("%d\n",power(n,n-2));
		}
	}
	else if(k==2)
	{
		if(n==1)
		{
			printf("0\n");
		}
		else
		{
			printf("%d\n",(((power(n,n-2) * power(n-1,n)) % mod) * power(4,n-1))%mod);
		}
	}
	else
	{
		if(n==1)
		{
			printf("0\n");
		}
		else
		{
			printf("%d\n",(((((power(n,n-2) * power(n-1,n)) % mod) * power(9,n-1))%mod) * power(3*n-3,n))%mod);
		}
	}
	return 0;
}
