ll pow(ll a , ll b)
{
    if(b == 0) return 1;
    ll ans = 1ll;
    ans = pow(a , b / 2ll);
    ans = ans * ans % mod;
    if(b & 1ll) ans = ans * a % mod;
    return ans;
}
