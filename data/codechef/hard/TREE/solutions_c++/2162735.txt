#include<iostream>
#include<algorithm>
using namespace std;
#define Maxn 1000000007
inline long long  power(long a ,long b )
{
    if(!b)
        return 1%Maxn;
    long long ans = 0;
    long long temp = power(a,b/2);
    ans=temp*temp%Maxn;
    if(b&1)
        ans=ans*a%Maxn;
    return ans;
}
main()
{
    int n,k;
    cin>>n>>k;

    cout<<power(k*(n-1),n*(k-1))*power(k*n,n-2)%Maxn;



}
