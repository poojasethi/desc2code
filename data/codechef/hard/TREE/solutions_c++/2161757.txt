long long int powr(long long int a,long long int p,long long int mm)
{
     if(p==0) return 1;
     else if(p%2==0)
     {
         long long int x= powr(a,p/2,mm);
         return (x*x)%mm;
     }
     else return (a%mm * powr(a,p-1,mm)%mm)%mm;
}

    }
