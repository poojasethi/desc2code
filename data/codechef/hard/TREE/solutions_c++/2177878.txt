#include<stdio.h>
#define o 1000000007
long long int mod_pow(long long int num, long long int pw,long long int mod)
{
    long long int test;
    for(test = 1; pw; pw >>= 1)
    {
        if (pw & 1)
        test = (test * num) % mod;
        num = (num * num) % mod;
    }

    return test;

}
int main()
{
    long long int n,k,s;
    scanf("%lld %lld",&n,&k);
    if(n==1)
    {
        if(k==1)
            s=1;
        else if(k==2 || k==3)
            s=0;
    }
    else
        s=(((mod_pow(n,(n-2),o) * mod_pow((n-1),(n*(k-1)),o))%o) * mod_pow(k,(n*k-2),o))%o;
    printf("%lld\n",s);
    return 0;
}
