#include<stdio.h>
#define ull unsigned long long
#define MOD 1000000007
long long power(long long a, unsigned long long int b)
{
ull r=1,x=a;
if(a==0)
return 0;
while(b>0)
{
if(b&1)r=(r*x)%MOD;
x=(x*x)%MOD;
b>>=1;
}
return r%MOD;
}
int main()
{
    long long n,k,answer;
        scanf("%lld%lld",&n,&k);
        if(n==1&&k>1)
        printf("0\n");
        else if(n==1&&k==1)
        printf("1\n");
        else
        {
            answer=(((power(n,n-2)*power(n-1,n*(k-1)))%1000000007)*power(k,n*k-2))%1000000007;
            printf("%lld\n",answer);
        }
    return 0;
}
