struct mat
{
	ll x[2][2];
	void init()
	{
		memset(x,0,sizeof(x));
	}
	mat operator *(const mat&a)const
	{
		mat b;b.init();
		for(int k=0;k<2;k++)
		for(int i=0;i<2;i++)
		for(int j=0;j<2;j++)
		b.x[i][j]=(b.x[i][j]+x[i][k]*a.x[k][j]%mod)%mod;
		return b;
	}
};
mat pow(mat a,int n)
{
	mat b;b.init();b.x[0][0]=b.x[1][1]=1;
	while(n)
	{
		if(n&1)b=b*a;
		a=a*a;n>>=1;
	}
	return b;
}
