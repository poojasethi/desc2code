#include <iostream>
using namespace std;

#define ULL unsigned long long

void fib(ULL n, ULL&x, ULL&y,ULL mod){
    if(n==1){
        x = y = 1;
        return ;
    }
     
    if(n&1){
        fib(n-1, y, x,mod);
        y=(y+x)%mod;
    }else{
        ULL a, b;
        fib(n>>1, a, b,mod);
        y = (a*a+b*b)%mod;
        x = (a*b + a*(b-a+mod))%mod;
    }
}
 
ULL ansfib(ULL n,ULL mod)
{if(n==0||n==1)
{return n;}
ULL x, y;
fib(n,x,y,mod);
return x;}


int main()
{int t;
cin>>t;
while(t--)
{ULL n;
scanf("%lld",&n);
printf("%lld\n",ansfib(n,1000000007));
}
return 0;}
