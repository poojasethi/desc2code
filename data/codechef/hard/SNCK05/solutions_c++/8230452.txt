#include<stdio.h>
#include<vector>
#include<map>
#define mod 1000000007
#define ll long long int 
using namespace std;
map < ll , ll > store;
map < ll , ll > :: iterator it;

long long int fib(long long int n){
	it =  store.find(n);
	if( it != store.end() )
		return it->second;
	ll k = fib(n/2);
	if( n % 2 == 0 )
	{
		ll l = fib( n/2 -1);
		k = (k*k + l*l)%mod;
	}
	else{
		k = ( k*fib( n/2 +1) + k*fib(n/2 -1))%mod;
	}
	store.insert( make_pair ( n, k));
	return k;
}	

int main(){

	ll n, t, k;

	store.insert( make_pair( 0, 1));
	store.insert( make_pair( 1, 1));
	store.insert( make_pair( 2, 2));
	store.insert( make_pair( 3, 3));



	while( t--){
		scanf("%lld",&n);
		printf("%lld\n", fib(n-1));
	}

}
