#include<iostream>
#include<stdio.h>
using namespace std;
#define MOD 1000000007
class matr
{
	public:
	long long int a;
	long long int b;
	long long int c;
	long long int d;
	matr(long long int a_, long long int b_, long long int c_, long long int d_)
	{
		a=a_;
		b=b_;
		c=c_;
		d=d_;
	}
	matr()
	{
	}
	matr operator * (matr x)
	{
		long long int p = (a*x.a+b*x.c)%MOD;
		long long int q = (a*x.b + b*x.d)%MOD;
		long long int r = (c*x.a + d*x.c)%MOD;
		long long int s = (c*x.b + d*x.d)%MOD;
		return matr(p,q,r,s);
	}
	
};

matr m[32];

void calc_matr()
{
	m[0].a=1; 
	m[0].b=1; 
	m[0].c=1; 
	m[0].d=0; 
	for(int i=1;i<31;++i)
	{
		m[i] = m[i-1]*m[i-1];
	}
}

long long int an[1000001];

int main()
{
	calc_matr();
	int t;
	scanf("%d",&t);
	int a = t;
	/*for(int i=0;i<31;++i)
	{
		printf("%lld %lld\n%lld %lld\n\n",m[i].a,m[i].b,m[i].c,m[i].d);
	}*/
	while(t-- >0)
	{
		long long int n;
		cin>>n;
		matr ans(1,0,0,1);
		int i=0;
		if(n==0)
		{
			cout<<0<<endl;
			continue;
		}
		while(n>0)
		{
			if(n%2==1)
			{
				ans = ans*m[i];
			}
			i++;
			n/=2;
		}
		an[a-t-1]=ans.b;
	}
	for(int i=0;i<a;++i)
		printf("%lld\n",an[i]);
	return 0;
}
