#include <bits/stdc++.h>
#define mp make_pair
#define pb push_back
#define ll long long
using namespace std;
string s;
int calc(char a,char b,char c)
{
    int ca=0,cb=0,cc=0;
    int ans=0,len=s.length();
    for(int i=0;i<len;i++)
    {
        if(s[i]==a)
        {
            ans+=(i-ca);
            ca++;
            cb++;
            cc++;
        }
        else if(s[i]==b)
        {
            ans+=(i-cb);
            cb++;
            cc++;
        }
        else
        {
            ans+=(i-cc);
            cc++;
        }
    }
    return ans;
}
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
    int t,rgb,rbg,gbr,grb,brg,bgr,r,g,b;
    cin>>t;
    while(t--)
    {
        cin>>s;
        rgb=calc('r','g','b');
        rbg=calc('r','b','g');
        gbr=calc('g','b','r');
        grb=calc('g','r','b');
        brg=calc('b','r','g');
        bgr=calc('b','g','r');
        cout<<min(rgb,min(rbg,min(gbr,min(grb,min(brg,bgr)))))<<endl;
    }
	return 0;
}
