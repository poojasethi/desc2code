#define nrep(i,j,k) for(i = j; i >= k; i--)
#define scs(str) scanf("%s", str)
#define sci(a) scanf("%d",&a)
#define scl(a) scanf("%lld",&a)
#define scd(a) scanf("%lf", &a)
#define fr first
#define se second
#define mp make_pair

using namespace std;

int main()
{
    int t;
    sci(t);
    while(t--) {
        string s;
        cin >> s;
        int i, j, n = s.length();
        char permut[6][3] = {{'r','g','b'}, {'r','b','g'}, {'g','r','b'}, {'g','b','r'}, {'b','g','r'}, {'b','r','g'}};
        int ans = INT_MAX;
        rep(i, 0, 6) {
            int red = 0;
            int green = 0;
            int blue = 0;
            int temp = 0;
            rep (j, 0, n) {
                if(s[j] == permut[i][0]) {
                    temp += green + blue;
                    red++;
                } else if(s[j] == permut[i][1]) {
                    temp += blue;
                    green++;
                } else {
                    blue++;
                }
            }
            ans = min(ans, temp);
        }
        pri(ans); nl;
    }
}
