#include <bits/stdc++.h>
using namespace std;

char s[100005];

int cal(string s1){
	int l=strlen(s) , ans=0 , c1=0 , c2=0 , c3=0;
	for(int i=0 ; i<l ; i++){
		if(s[i]==s1[0]) { c1++; ans+=(c2+c3); }
		else if(s[i]==s1[1]) { c2++; ans+=(c3); }
		else  c3++; 
	}
	return ans;
}

int main() {
	int t;
	
	scanf("%d" , &t);
	while(t--){
		scanf("%s" , &s);
		int min1=min(cal("rgb") , cal("rbg"));
		int min2=min(cal("grb") , cal("gbr"));
		int min3=min(cal("bgr") , cal("brg"));
		min1=min(min1 , min2);
		min1=min(min1 , min3);
		cout<<min1<<endl;
	}
	return 0;
}
