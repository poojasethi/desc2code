#define maxn 100000+3
using namespace std;
char s[maxn];
int len;
long long ans ;

long long min(long long a, long long b) {
    return a<b?a:b;
}
void solve(char ch1, char ch2, char ch3) {
    long long cur_min = 0;
    
    long long cnt1 = 0;
    long long cnt2 = 0;
    long long cnt3 = 0;
    
    for(int i=0; i<len; i++) {
        
        if(s[i] == ch1) {
            cur_min += (cnt2 + cnt3);
            cnt1++;
        } else if(s[i] == ch2) {
            cur_min += (cnt3);
            cnt2++;
        } else {
            cnt3++;
        }
    }
    ans = min(ans, cur_min);
    
}
