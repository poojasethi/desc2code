#include<iostream>
#include<cstdio>
#include<cstring>
#include<climits>
using namespace std;
char c[]={'r','g','b'};

int getagain(char s[], char a, char b)
{
    int i=-1,j,k=-1,coun=0;
    for(j=0;j<strlen(s);j++)
    {
        if(s[j]!=b)
            k++;
        if(s[j]==a)
        {
            coun+=(k-i-1);
            i++;
        }
    }
    return coun;

}


int get(char s[], char a)
{
    int i=-1,coun=0,j,mini=INT_MAX;
    for(j=0;j<strlen(s);j++)
    {
        if(s[j]==a)
        {
            coun+=(j-i-1);
            i++;
        }
    }
    for(i=0;i<3;i++)
    {
        if(c[i]!=a)
            mini=min(mini,coun+getagain(s,c[i],a));
    }
return mini;

}

int order(char s[])
{
    int r =min(min(get(s,'r'),get(s,'b')),get(s,'g'));
    return r;

}

int main()
{

    int t,n, result;
     char s[100005];

     scanf("%d",&t);
    while(t--)
    {
        scanf("%s",s);
        result=order(s);
        printf("%d\n",result);

    }

    return 0;
}
