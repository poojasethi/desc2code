#include<bits/stdc++.h>
using namespace std;
int main(){
	int t;
	cin >> t;
	while(t--){
		string s;
		cin>>s;
		int ans=INT_MAX;
		int currentans=0;
		int counter=0,i;
		//START R END G
		for(int i=0;i<s.length();i++){
			if(s[i]=='r')
			currentans+=counter;
			else
			counter++;
		}
		counter=0;
		for(i=s.length()-1;i>=0;i--){
			if(s[i]=='g')
			currentans+=counter;
			else if(s[i]=='b')
			counter++;
		}
		ans=min(ans,currentans);
		
		//START G END R
		currentans=0;
		counter=0;
		for(int i=0;i<s.length();i++){
			if(s[i]=='g')
			currentans+=counter;
			else
			counter++;
		}
		counter=0;
		for(int i=s.length()-1;i>=0;i--){
			if(s[i]=='r')
			currentans+=counter;
			else if(s[i]=='b')
			counter++;
		}
		ans=min(ans,currentans);
		
		//START R END B
		currentans=0;
		counter=0;
		for(int i=0;i<s.length();i++){
			if(s[i]=='r')
			currentans+=counter;
			else
			counter++;
		}
		counter=0;
		for(int i=s.length()-1;i>=0;i--){
			if(s[i]=='b')
			currentans+=counter;
			else if(s[i]=='g')
			counter++;
		}
		ans=min(ans,currentans);
		
		//START B END R
		currentans=0;
		counter=0;
		for(int i=0;i<s.length();i++){
			if(s[i]=='b')
			currentans+=counter;
			else
			counter++;
		}
		counter=0;
		for(int i=s.length()-1;i>=0;i--){
			if(s[i]=='r')
			currentans+=counter;
			else if(s[i]=='g')
			counter++;
		}
		ans=min(ans,currentans);
		
		//START G END B
		currentans=0;
		counter=0;
		for(int i=0;i<s.length();i++){
			if(s[i]=='g')
			currentans+=counter;
			else
			counter++;
		}
		counter=0;
		for(int i=s.length()-1;i>=0;i--){
			if(s[i]=='b')
			currentans+=counter;
			else if(s[i]=='r')
			counter++;
		}
		ans=min(ans,currentans);
		
		//START B END 
		currentans=0;
		counter=0;
		for(int i=0;i<s.length();i++){
			if(s[i]=='b')
			currentans+=counter;
			else
			counter++;
		}
		counter=0;
		for(int i=s.length()-1;i>=0;i--){
			if(s[i]=='g')
			currentans+=counter;
			else if(s[i]=='r')
			counter++;
		}
		ans=min(ans,currentans);
		cout<<ans<<"\n";
		
	}
}
