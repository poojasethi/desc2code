typedef long long ll;

using namespace std;

int main()
{
    ios_base::sync_with_stdio(0);cin.tie(0);
    int t,l,c1,c2,c3,i,j,s,mini;
    char a[100005],b[7][4]={{"rgb"},{"rbg"},{"bgr"},{"brg"},{"gbr"},{"grb"}};
    cin>>t;
    while(t--)
    {
        cin>>a;
        mini=INT_MAX;
        for(j=0;j<6;j++)
        {
        i=0;
        c1=0,c2=0,c3=0;s=0;
        while(a[i]!='\0')
        {
            if(a[i]==b[j][0])
                {c1+=1;s+=c3+c2;}

            else if(a[i]==b[j][1])
                {c2+=1;s+=c3;}
            else
                c3+=1;

            i++;
        }
        mini=min(mini,s);
        }
        cout<<mini<<endl;

    }
    return 0;
}
