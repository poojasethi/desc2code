#include<iostream>
#include<cstring>
#include<vector>
#include<cstdio>
#define MOD 1000000007
#define N 5000000
#define gc getchar_unlocked

using namespace std;

inline int scan()
{
    register int n=0,c=gc();
    while(c<'0'||c>'9')
    c=gc();
    while(c<='9'&&c>='0'){
        n=n*10+c-'0';
        c=gc();
    }
    return n;
}

int pperm[N+1];
int prime[N+1];
bool isprime[N+1];
int main(){

	int t,n;
	memset(isprime,1,sizeof(isprime));
	isprime[0]=isprime[1]=0;
	for(int i=2;i*i<=N;i++){
		if(isprime[i]){
			for(int j=i*i;j<=N;j+=i)
				isprime[j]=0;
		}
	}

	for(int i=1;i<=N;i++){
		prime[i]=((long long)isprime[i]+prime[i-1])%MOD;
	}

	pperm[1]=1;
	for(int i=2;i<=N;i++){
		pperm[i]=((long long)(1+prime[i])*(long long)pperm[i-1])%MOD;
	}
	t=scan();
	while(t--){
		n=scan();

		printf("%d\n",pperm[n]);
	}

	return 0;
}
