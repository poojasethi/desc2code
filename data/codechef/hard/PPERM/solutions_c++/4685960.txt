#include<iostream>
#include<cstdio>

using namespace std;

int pri[5000001];
long long int ans[5000001];

#define gc getchar_unlocked
#define mod 1000000007LL

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

void writeint(long long int x)
{
	if(x == 0)
		putchar_unlocked('0');
	else
	{
		int cnt = 0;
		char ch[20];
		while(x != 0)
		{
			ch[cnt++] = x%10 + '0';
			x /= 10;
		}
		for(int i=cnt - 1; i>=0; --i)
			putchar_unlocked(ch[i]);
	}
	putchar_unlocked('\n');
}

int ques[500001];

int main()
{
	for(int i=2; i<=5000000; ++i)
	{
		if(pri[i] == 0)
		{
			for(int j= 2*i; j<=5000000; j+=i)
			{
				pri[j] = 1;
			}
		}
	}
	int pricount = 1;
	ans[1] = 1;
	ans[2] = 2;
	ans[3] = 6;
	pricount = 3;
	for(int i=4; i<=5000000; ++i)
	{
		if(pri[i] == 0)
			pricount++;
		ans[i] = (ans[i - 1] * pricount) % mod;
	}
	int t;
	int n;
	scanint(t);
	for(int i=0; i<t; ++i)
		scanint(ques[i]);
	for(int i=0; i<t; ++i)
		writeint(ans[ques[i]]);
	return 0;
}
