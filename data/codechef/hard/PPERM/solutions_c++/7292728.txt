#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
 
#define mod 1000000007
 
using namespace std;
 
#define MAX 5000001 
#define ll long long
int seive[MAX] ;
ll cp[MAX] ;
ll get_num()
{
    ll num = 0;
    char c = getchar_unlocked();
    int flag = 0;
    while(!((c>='0' & c<='9') || c == '-'))
        c=getchar_unlocked();
    if(c == '-')
    {
        flag = 1;
        c=getchar_unlocked();
    }
    while(c>='0' && c<='9')
    {
        num = (num<<1)+(num<<3)+c-'0';
        c=getchar_unlocked();
    }
    if(flag==0)
        return num;
    else
        return -1*num;
}
void populate_seive()
{
    ll i, j ;
    seive[1] = 0 ;
    for(i=2;i*i<MAX;i++)
    {
        if(seive[i]==0)
        {
            for(j=i*i;j<MAX;j+=i)
              seive[j] = 1 ;
        }
    }
}
int main() {
	ll t,N,i,cnt ;
	t = get_num() ;
	
	populate_seive() ;
	cp[1] = 1 ;
    cnt = 1 ;  
	for(i=2;i<MAX;i++)
	{
	    if(seive[i]==0)
	    {
	        cnt++;
	    }
	    cp[i] = ((cp[i-1]%mod)*(cnt%mod))%mod ;
	}
	while(t--)
	{
	   N = get_num() ;
	   printf("%lld\n",cp[N]) ; 
	}
	return 0;
}
