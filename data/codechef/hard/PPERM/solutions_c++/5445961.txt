#include <iostream>
#include <cstdio>
using namespace std;

const int maxn=5000005;
bool iscomp[maxn];
int prod[maxn];
const int MOD=1000000007;

void precompute()
{
    prod[1]=1;
    long long cur=1;
    int last = 1;
    for(int i=2;i<maxn;i++)
    {
        if(!iscomp[i])
        {
            cur++;
            if(i<2500)
            {
                for(int j=i*i;j<maxn;j+=i)
                {
                    iscomp[j]=true;
                }
            }
        }
        last=prod[i]=(last*cur)%MOD;
    }
}

int main()
{
    precompute();
    int T=1,N;
    scanf("%d",&T);
    while(T--)
    {
        scanf("%d",&N);
        printf("%d\n",prod[N]);
    }
    return 0;
}
