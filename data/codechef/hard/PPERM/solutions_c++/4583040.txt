#include<cstdio>
#include<cstring>
using namespace std;
#define MAXN 5000100
#define MOD 1000000007
bool isPrime[MAXN];
long long int primes[MAXN];
long long int pperm[MAXN];
int main()
{
    isPrime[1]=false;
    for(long long int i=2;i<=MAXN;i++) isPrime[i]=true;
    for(long long int i=2;i*i<=MAXN;i++){
        if(isPrime[i]){
            for(long long int j=i*i;j<=MAXN;j+=i)
                isPrime[j]=false;
        }
    }
    primes[0]=0;
    for(long long int i=1;i<=MAXN;i++)
        primes[i]=primes[i-1]+(isPrime[i]?1:0);
    int t;
    scanf("%d",&t);
    pperm[1]=1;
    for(int i=2;i<MAXN;i++){
        pperm[i]=(((1+primes[i])%MOD)*(pperm[i-1]%MOD))%MOD;
    }
    while(t--){
        long long int n;
        scanf("%lld",&n);
        printf("%lld\n",pperm[n]);
    }
}
