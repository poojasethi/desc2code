using namespace std;
vector<long long int> ans;
int main()
{
    ans.push_back(1);
    ans.push_back(1);
    int i;
    int j;
    int prime=1;
    vector<bool> P;
    P.assign(5000001,true);
    for(i=2; i*i<=5000000; i++)
    {
        if(P[i]==true)
        {
            for(j=i*i; j<=5000000; j=j+i)
            {
                P[j]=false;
            }
        }
    }
    for(i=2;i<=5000000;i++)
    {
        if(P[i])
        {
            prime++;
        }
        ans.push_back(((ans[i-1] % 1000000007)*(prime % 1000000007))% 1000000007);
    }
    int T;
    scanf("%d",&T);
    while(T--)
    {
        int no;
        scanf("%d",&no);
        printf("%d\n",ans[no]);
    }
}
