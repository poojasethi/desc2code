{
	int i,j;
	for(i=2;i<=5000001;i++)

		if(a[i]==0)

	 		primes[k++]=i;
	 		j=2;
	 		while((i*j)<=5000001)
	 		{
	 			a[i*j]=1;
	 			j++;
	 		}
	 	}
	}
	int c=1;
	arr[1]=1;
	for(int i=2;i<=5000001;i++)
	{
		if(a[i]==0)
			c++;
		arr[i]=(arr[i-1]*c)%1000000007;
	}
}
