#include <bits/stdc++.h>
#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;


#define getcx getchar_unlocked
#define putcx putchar_unlocked
#define MAX 5000000
#define MOD 1000000007
bool isprime[MAX+10];
int arr[MAX+10];
long long ans[MAX+10];

inline void readInt( int &n ) {
    n=0;
    int ch=getcx();
    while( ch < '0' || ch > '9' ){ch=getcx();}
    while(  ch >= '0' && ch <= '9' )
        n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
}

inline void writeInt(long long n) {
	int i = 19;
	char buf[20];
	buf[19] = '\n';

	while (n) {
		buf[--i] = n % 10 + '0';
		n /= 10;
	}

	do {
		putcx(buf[i]);
	} while (buf[i++] != '\n');
}

int main()
{
    int t, i, j, n;
    memset(isprime, true, sizeof(isprime));
    memset(arr, 0, sizeof(arr));
    for(i=4; i<=MAX; i+=2)
        isprime[i] = false;
    for(i=3; i<sqrt(MAX); i+=2){
        if(isprime[i]){
            for(j=i*i; j<=MAX; j+=i){
                isprime[j] = false;
            }
        }
    }
    arr[1] = 1;
    for(i=2; i<=MAX; i++){
        if(isprime[i])
            arr[i] = arr[i-1]+1;
        else
            arr[i] = arr[i-1];
    }
    ans[1] = 1;
    for(i=2; i<=MAX; i++){
        ans[i] = (ans[i-1]*arr[i])%MOD;
    }
    readInt(t);
    while(t--){
        readInt(n);
        writeInt(ans[n]);
    }
    return 0;
}
