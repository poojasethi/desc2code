#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#define FM 5000001
using namespace std;
bool isPrime[FM];
int nPrimes[FM];
long long permutations[FM];
void sieve()
{
    int i,j;
    long long temp;
    isPrime[1]=false;
    isPrime[0]=false;
    nPrimes[0]=0;
    nPrimes[1]=0;
    permutations[0]=0;
    permutations[1]=1;
    for(i=2;i<FM;i++)
    {
        isPrime[i]=true;
        nPrimes[i]=0;
        permutations[i]=0;
    }
    for(i=2;i<FM;i++)
    {
        nPrimes[i]=nPrimes[i-1];
        if(isPrime[i]==true)
        {
            nPrimes[i]++;
            for(j=2*i;j<FM;j+=i)
            {
                isPrime[j]=false;
            }
        }
        temp = (1+nPrimes[i])*permutations[i-1];
        permutations[i] = temp%1000000007;
    }
}
int main()
{
    sieve();
    int t,n;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d",&n);
        printf("%lld\n",permutations[n]);
    }
    return 0;
}
