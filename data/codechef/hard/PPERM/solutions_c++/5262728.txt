#include <bits/stdc++.h>
#define mod 1000000007
#define maxval 5000000
#define x getchar_unlocked()
#define lli long long int
using namespace std;

int prime[5000002];
lli ans[5000002];
lli primes[5000002];


void sieve(){
	memset(prime,1,sizeof prime);
	prime[0] = 0;
	prime[1] = 0;
	for(int i=2;i<=maxval;i++){
		if(prime[i]){
			for(int j=2*i;j<=maxval;j=j+i){
				prime[j] = 0;
			}
		}
	}
	int cnt =0;
	for(int i=0;i<=maxval;i++){
		if(prime[i])	cnt++;
		primes[i] = cnt; 
	}
}

void calc(){
	sieve();
	ans[1] = 1;
	for(int i=2;i<=maxval;i++){
		ans[i] = ans[i-1]*(1+primes[i]);
		ans[i] = ans[i]%mod;
	}
}

inline void inp(int &n ) {//fast input function
  n=0;
  int ch=x,sign=1;
  while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=x;}
  while( ch >= '0' && ch <= '9' )
  n=(n<<3)+(n<<1)+ ch-'0', ch=x;
  n=n*sign;
}

int main(){
	int t;
	inp(t);
	calc();
	while(t--){
		int n;
		inp(n);
		printf("%lld\n",ans[n]);
	}
	return 0;
}
