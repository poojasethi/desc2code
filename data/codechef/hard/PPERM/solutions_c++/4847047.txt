#include<stdio.h>
#include<string.h>

#define M 1000000007

using namespace std;

bool prime[5000006];
//int cum_prime[5000001],
long long int res[5000001];

int main()
{
    memset(prime,true,sizeof prime);
    int count1=1;
    //cum_prime[1]=1;
    res[1]=(long long)1;
    for(int i=2;i<=5000000;i++)
    {
        if(prime[i])
        {
            count1++;
            for(int j=2*i;j<=5000000;j=j+i)
                prime[j]=false;
        }
        //cum_prime[i]=count1;
        res[i]=(res[i-1] * count1) % M;

    }
    //printf("%d",count1);

    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        printf("%d\n",res[n]);
    }

    return 0;
}
