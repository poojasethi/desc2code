int main()
{
    int t;
    scanf("%d",&t);
    long long *dp;
    bool *prime;
    dp=(long long *)malloc(sizeof(long long)*5000001);
    prime=(bool *)malloc(sizeof(bool)*5000001);
    dp[1]=1;
    int top=2;
    int prime1=0,n,i,j;
    for(i=2;i<=5000000;i++)
    {
        prime[i]=true;
    }
    for(i=2;i<=5000000;i++)
    {
        if(prime[i])
        {
            for(j=i+i;j<=5000000;j=j+i)
            {
                prime[j]=false;
            }
        }
    }
    while(t--)
    {
        scanf("%d",&n);
        while(n>=top)
        {
            if(prime[top])
            prime1=prime1+1;
            dp[top]=((prime1+1)*dp[top-1])%p;
            top++;
        }
        printf("%lld\n",dp[n]);
    }
    return 0;
}
