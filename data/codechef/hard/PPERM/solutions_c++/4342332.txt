#include <cstdio>
#include <cstring>
#define max_n 5000000
#define mod 1000000007

using namespace std;

bool isprime[max_n+10];long int choice[max_n+10];
long long int prime_perm[max_n+10];

void seive()
{
    long int i,j;
    memset(isprime,true,sizeof(isprime));
    isprime[1]=false;choice[1]=1;
    for(i=2;i<=max_n;i++)
    {
        choice[i]=choice[i-1];
        if(isprime[i])
           {
                choice[i]++;
                for(j=i<<1;j<=max_n;j+=i)
                    isprime[j]=false;
           }
    }
}

void precompute()
{
    seive();
    prime_perm[1]=1;
    for(long int i=2;i<=max_n;i++)
        prime_perm[i] = (choice[i] * prime_perm[i-1]) % mod;
}

int main()
{
    precompute();
    long int t,n;

    scanf("%ld",&t);

    while(t--)
    {
        scanf("%ld",&n);
        printf("%lld\n",prime_perm[n]);
    }

    return 0;
}
