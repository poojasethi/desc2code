#define sn(i) scanf("%s",i)
#define maxn 5000001
#define mod 1000000007
using namespace std;

bool isPrime[maxn]; int primesum[maxn];
int dp[maxn];

int main() {
	// your code goes here
	int n,t;
	memset(isPrime,true,sizeof(isPrime));
	isPrime[1] = false;
	for(int i = 2; i*i < maxn; i++)
		if(isPrime[i])
		for(int j = i*i; j < maxn; j+= i)
				isPrime[j] = false;

	primesum[0] = 0;

	for(int i = 1; i < maxn; i++)
		primesum[i] = primesum[i-1] + ((isPrime[i]) ? 1 : 0);


	dp[1] = 1;

	for(int i = 2; i < maxn; i++)
		dp[i] = (((1 + primesum[i])%mod)*(long long)dp[i-1])%mod;

	in(t);
	while(t--)
	{
		in(n);
		printf("%d\n",dp[n]);
	}
	return 0;
}
