// PPERM - Codechef
#include <bits/stdc++.h>
#define maxP 5000000
#define mod 1000000007

using namespace std;
typedef long long lint;

int E[5000001], P[5000001], nP;
int dp[5000001];

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    for(int i=2;i<=maxP;i++) {
        if (E[i]==0) E[i]=P[++nP]=i;
        for(int j=1;j<=nP && P[j]<=E[i] && P[j]<=maxP/i;j++) E[P[j]*i]=P[j];
    }
    E[1]=0;
    for(int i=2;i<=maxP;i++) if (E[i]==i) E[i]=E[i-1]+1; else E[i]=E[i-1];
    dp[1]=1;
    for(int i=2;i<=maxP;i++) dp[i]=(lint(1+E[i])*dp[i-1]) % mod;
    int T; scanf("%d",&T);
    while (T--) {
        int n; scanf("%d",&n);
        printf("%d\n",dp[n]);
    }
}
