using namespace std;

void prime_permutations()
{
    int k,T,N,i;
    long long j;
    static long long nums[5000001];
    static long long prime_nums[5000001];
    static int prime_flag[5000001];

    nums[1] = 1;
    prime_nums[1] = 1;
    for(i = 2; i < 5000001; ++i)
    {
        if(prime_flag[i] == 0)
        {
            for(j = (long long)i*i; j < 5000001; j += i)
                prime_flag[j] = 1;
            prime_nums[i] += 1;
        }
        prime_nums[i] += prime_nums[i-1];
        nums[i] = (nums[i-1] * prime_nums[i]) % 1000000007;
    }
    cin >> T;
    for(k = 0; k < T; ++k)
    {
        scanf("%d", &N);
        printf("%d\n", nums[N]);
    }
}

int main()
{
    prime_permutations();
    return 0;
}
