#include<bits/stdc++.h>
#define N 5000010
#define ll long long
#define mod 1000000007
using namespace std;

bool sieve[N];
ll primes[N];
ll dp[N];

main() {
    dp[1]=1; primes[1]=1;
    for (ll i=2; i<N; i++) {
        if (!sieve[i]) {
            for (ll j=2*i; j<N; j+=i) sieve[j]=1;
            primes[i]=primes[i-1]+1;
        }
        else primes[i]=primes[i-1];
        dp[i]=dp[i-1]*primes[i];
        dp[i]%=mod;
    }
    ll t, n;
    scanf("%lld", &t);
    while (t--) {
        scanf("%lld", &n);
        printf("%lld\n", dp[n]);
    }
}
