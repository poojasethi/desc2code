#include <bits/stdc++.h>
# define s(x) scanf("%lld",&x)
using namespace std;
#define pb push_back
#define mp make_pair
#define ll long long
#define ms(a,b) memset((a),(b),sizeof(a))
#define inf 1000000007
#define f(i,j) for(int i=0;i<j;i++)
ll num=5000001;
vector<ll> a(num,0);
vector<ll> ans(num,0);
int main()
{
    ll t,n,i,j;
    ans[1]=1;
    ll val=1;
    for(i=2;i<(floor)(sqrt(num));i++)
    {

        if(a[i]==0)
        {
            for(j=i*i;j<num;j+=i)
            {
                a[j]=1;
            }
            val++;
            ans[i]=((ans[i-1]*val)%inf);
        }
        else
        {
            ans[i]=((ans[i-1]*val)%inf);
        }
    }
    for(j=i;j<num;j++)
    {
        if(a[j]==0)
        {
            val++;
            ans[j]=((ans[j-1]*val)%inf);
        }
        else
        {
            ans[j]=((ans[j-1]*val)%inf);
        }
    }
    s(t);
    while(t--)
    {
        s(n);
        printf("%lld\n",ans[n]);
    }
    return 0;
}
