#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
#define mod 1000000007
#define ll long long int
ll A[5000002];
bool P[5000002];
 int main()
 {
     int i,j,t,n,c;
     for(i=0;i<=5000000;i++)
        P[i]=1;
     P[0]=P[1]=0;
    for(i=2;i<3000;i++)
    {
        if(P[i]==1)
        for(j=i+i;j<=5000000;j=j+i)
            P[j]=0;
    }
    A[1]=1;
    c=1;
    for(i=2;i<=5000000;i++)
    {
        if(P[i]==1)
           {
               //cout<<i<<endl;
            c++;
           }
        A[i]=((A[i-1]%mod)*(c%mod))%mod;
    }
    scanf("%d",&t);
    while(t>0)
    {
        t--;
        scanf("%d",&n);
        printf("%lld\n",A[n]);
    }
 }
