#include<bits/stdc++.h>
#define ll long long
#define mp make_pair
#define pb push_back
#define pi pair<int,int>
#define MX 5000001
#define mod 1000000007
using namespace std;
int prime[MX];
ll dp[MX];
void pre(){
    for(int i=2;i<MX;i++) prime[i] = 1;
    prime[1]=1;
    for(int i=2;i<=2236;i++){
        if(prime[i]){
            for(int j= i*i;j<MX;j+=i){
                prime[j]=0;
            }
        }
    }
    for(int i=0;i<MX;i++){
        prime[i]+=prime[i-1];
    }
    dp[1]=1;
    for(int i=2;i<MX;i++){
        dp[i] = (1ll*prime[i]*dp[i-1])%mod;
    }
}
int main() {
    pre();
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        printf("%lld\n",dp[n]);
    }
    return 0;
}
