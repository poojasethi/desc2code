bool prima[5000005]={};
int dp[5000005]={};

void DP() {
     int i,j;
     long long x=0;
     for (i=2;i<=5000000;i++) {
         if (prima[i]==false) {
            j=i*2;
            while (j<=5000000) {
                  prima[j]=true;
                  j+=i;
            }
         }
     }
     j=1;
     dp[1]=1;
     for (i=2;i<=5000000;i++) {
         if (prima[i]==false) j++;
         x=dp[i-1];
         x*=j;
         x%=1000000007;
         dp[i]=x;
     }
}
