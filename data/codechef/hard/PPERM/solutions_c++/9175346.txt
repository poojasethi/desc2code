#include<bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> pi;

#define MP make_pair
#define PB push_back
#define F first
#define S second
#define MOD 1000000007
#define INF (1 << 30)
#define N 5000005
#define MAX 5000000

ll dp[N];
bool isPrime[N];
ll primes[N]; // primes[i] = No.of.primes till 'i'

void sieve()
{
	int i, j;
	memset(isPrime, true, sizeof isPrime);
	for(i = 2; i <= sqrt(MAX); i++)
	{
		if(isPrime[i])
		{
			for(j = i * i; j <= MAX; j += i)
				isPrime[j] = false; 
		}
	}
	primes[1] = 1;
	for(i = 2; i < N; i++)
		primes[i] = primes[i - 1] + (isPrime[i] ? 1 : 0);
}

void precompute()
{
	sieve();
	dp[1] = 1;
	for(int i = 2; i < N; i++)
		dp[i] = (primes[i] * dp[i - 1]) % MOD;
}

int main()
{
	// ios::sync_with_stdio(false);
	int t, n;
	precompute();
	scanf("%d", &t);
	while(t--)
	{
		scanf("%d", &n);
		printf("%lld\n", dp[n]);
	}
	return 0;
}
