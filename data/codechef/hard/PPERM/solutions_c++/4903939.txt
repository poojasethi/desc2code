    #include<cstdio>
    #include<iostream>
    #include<algorithm>
    using namespace std;
    const int K =5000010;
    #define mod 1000000007
    	 bool isPrime[K];
    	  int prime[K],pprime[K];
    int main()
    {
    long long int g,g1;
    int t,n;
    isPrime[0] = isPrime[1] = false;
    for (int i = 2; i < K; ++i) {
    isPrime[i] = true;
    }
    for (int i = 2; i * i < K; ++i) {
    if (isPrime[i]) {
    for (int j = i * i; j < K; j += i) {
    isPrime[j] = false;
    }
    }
    }
    prime[0]=0;
    for(int i=1;i<K;i++)
    prime[i]=prime[i-1]+(isPrime[i]?1:0);
    pprime[0]=1;
    for(int i=1;i<K;i++)
    {
    	g1=((1+prime[i])%mod);
    g=(g1*pprime[i-1])%mod;
    g=g%mod;
    pprime[i]=g;
    }
    scanf("%d",&t);
    while(t--)
    {
    scanf("%d",&n);
    printf("%d\n",pprime[n]);
    }
    return 0;
    }
