#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
#define ll long long
const int mx=5000001;
const ll mod=1000000007;
ll dp[mx];
vector<int> store(mx);
vector<bool> ar(mx,true);
void pre()
{
	int i,j;
	store.push_back(2);
	for(i=2;i<=mx;i+=2)
	ar[i]=false;
	for(i=3;i<=mx;i+=2)
	{
		if(ar[i]==true)
		{
			for(j=2*i;j<=mx;j+=i)
			{
				ar[j]=false;
			}
		}
	}
	store[0]=0;
	for(i=1;i<=5000000;i++)
	{
	   store[i]=store[i-1]+(ar[i]==true);
	}
}
int main()
{
	int i,j,n,t;
  	pre();
  	dp[1]=1;
  	dp[2]=2;
  	for(i=3;i<=5000000;i++)
  	{
		dp[i]=dp[i-1];
		dp[i]*=(store[i]+1);
		dp[i]%=mod;
		
  	}
	scanf("%d",&t);
  	while(t--)
  	{
  		scanf("%d",&n);
  		printf("%lld\n",dp[n]);
  	}
  	return 0;
}
