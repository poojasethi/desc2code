void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}



bool a[5000001];
ll primes[5000001],perm[5000001];
void sieve(){
 primes[0]=0,perm[0]=0,perm[1]=1;
int i,j;
for(i=0;i<5000001;i++){ a[i]=true;}
a[0]=false;a[1]=false;
for(i=2;i<2237;i++){
    for(j=i*i;j<5000001;j+=i){
        a[j]=false;
    }
}
for(i=1;i<5000001;i++){
    primes[i]=primes[i-1]+(a[i]?1:0);
}
for(i=2;i<5000001;i++){
    perm[i]=(1+primes[i])*(perm[i-1]%M);
    perm[i]%=M;
}
}
int main(){
sieve();
int t,n;
scanint(t);
while(t--){
    scanint(n);
    printf("%lld\n",perm[n]);
}
return 0;
}
