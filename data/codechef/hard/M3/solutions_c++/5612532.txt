#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
using namespace std;
const int maxN=200;
#define fi "B.INP"
#define fo "B.OUT"
#define foru(i,l,r) for(int i=l;i<=r;++i)
#define ll unsigned long long
int a[maxN][maxN],b[maxN][maxN],num[maxN],tmp_num[maxN],res[maxN],test,n;
int main(){
    //freopen(fi,"r",stdin);
    //freopen(fo,"w",stdout);
    scanf("%d",&test);
    while (test--){
        scanf("%d",&n);
        foru(i,1,n) num[i]=0;
        foru(i,1,n)
            foru(j,1,n) scanf("%d",&a[i][j]),num[i]+=(a[i][j]==1);
        foru(t,1,n){
            foru(i,1,n) tmp_num[i]=num[i];
            foru(i,1,n)
                foru(j,1,n) b[i][j]=a[i][j];
            foru(i,1,n)
                if (b[t][i]==2) b[t][i]=1,b[i][t]=0,++tmp_num[t];
            int ok=1;
            foru(i,1,n){
                if (!ok) break;
                if (i==t) continue;
                if (tmp_num[i]>tmp_num[t]) ok=0;
                if (tmp_num[i]<tmp_num[t]) continue;
                foru(j,1,n)
                    if ((tmp_num[i]==tmp_num[j])&&(b[i][j]==2)) ok=0;
            }
            res[t]=ok;
        }
        foru(i,1,n) printf("%d",res[i]);
        printf("\n");
    }
    fclose(stdin);
    fclose(stdout);
}
