#include <cstdio>
#include <iostream>
#include <cstring>
#include <string>
#include <algorithm>
using namespace std;
#define MX 142

int score[MX], s[MX], n;
int toplay[MX][MX], tp[MX][MX];

int canWin(int ct)
{
	int i,j,k,a,b;
	for(i=0;i<n;i++)
		if(tp[ct][i])
		{
			s[ct]++;
			tp[ct][i]=tp[i][ct]=0;
		}

	for(i=0;i<n;i++)
	{
		if(i==ct) continue;
		if(s[i]>s[ct]) return 0;
		for(j=0;j<n;j++)
		{
			if(tp[i][j]==1 && s[i]==s[ct] && s[j]==s[ct]) return 0;
		}
	}
	return 1;
}


int main()
{
	int t,i,j,k; char in[502],*tok;
	for(scanf("%d",&t);t--;)
	{
		scanf("%d ",&n);
		
		for(i=0;i<n;i++) 
		{
			score[i]=0;
			for(j=0;j<n;j++)
				toplay[i][j]=0;
		}
		
		for(i=0;i<n;i++)
		{
			gets(in);
			tok = strtok(in," ");
			for(j=0;j<n;j++)
			{
				if(*tok=='2') toplay[i][j]=1;
				else if(*tok=='1') score[i]++;
				tok = strtok(NULL," ");				
			}				
		}
		


		for(k=0;k<n;k++)
		{
			for(i=0;i<n;i++) for(j=0,s[i]=score[i];j<n;j++) 
				tp[i][j]=toplay[i][j];
			in[k] = '0'+canWin(k);
		}
		in[n]='\0';
		puts(in);
	}
	return 0;
}
