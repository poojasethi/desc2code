#include <bits/stdc++.h>
#define filein(t) freopen(t, "r", stdin)
#define fileout(t) freopen(t, "w", stdout)
#define f first
#define s second
#define cs 137
#define ll long long
#define mp make_pair
#define BIT(i, j) ((j >> i) & 1)
#define N 1005
#define NN 100005
#define ln 1e9+1
#define oo 1000000007
#define Pa pair < int, int >
#define PD pair < double, double >
#define Line pair < Pa, ll >
#define PP pair < ll, Pa >
#define Pii pair < Pa, Pa >

using namespace std;

int T, n, m, A[N][N], DP[N][N];

int main()
{
    ios_base::sync_with_stdio(0);
    //filein("TABLECOV.inp"); fileout("TABLECOV.out");
    cin >> T;
    while(T--)
    {
        cin >> n >> m;
        memset(DP, 0, sizeof(DP));
        memset(A, 0, sizeof(A));
        for(int i=1; i<=n; i++)
            for(int j=1; j<=m; j++)
                cin >> A[i][j];
        for(int i=1; i<=n; i++)
            for(int j=m; j>=1; j--)
                DP[i][j] = max(DP[i-1][j+1] + A[i][j], max(DP[i][j+1], DP[i-1][j]));
        cout << DP[n][1] << endl;
    }
}
