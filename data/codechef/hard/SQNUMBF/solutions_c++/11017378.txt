#include<bits/stdc++.h>
#define rep(i,n) for(i=0;i<n;++i)
#define ll long long
using namespace std;

ll gcd(ll a,ll b)
{
    return (!b?a:gcd(b,a%b));
}
ll i,j,arr[101],ans,x,temp,num;
#define ll long long
void solve()
{
    cin>>num;
    rep(i,num)scanf("%lld",&arr[i]);

    rep(i,num-1)
    {
        for(j=i+1;j<num;++j)
        if(gcd(arr[i],arr[j])>1)
            {
                printf("%lld\n",gcd(arr[i],arr[j]));return;
            }
    }
    rep(j,num)
    {   x=arr[j];
        for(i=1;i*i*i<=x;++i){
        if(i>1 && !( x % (i*i) ))
        {
        printf("%lld\n",i);
        return;
        }
        if(!( x%i ))
        {      temp=x/i;
            ans=(ll)sqrt(temp);
            if(ans>1&&ans*ans==temp)
            {
            printf("%lld\n",ans);
            return;
            }
        }

    }
}
}
int main() {
int t;
cin>>t;
while (t--) {
solve();
}
    return 0;
}
