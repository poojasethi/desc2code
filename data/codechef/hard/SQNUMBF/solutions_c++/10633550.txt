#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n,i,f=0;
        long long s,s1,s2,j;
        scanf("%d",&n);
        long long a[n+69];
        for(i=0;i<n;i++)
        scanf("%lld",&a[i]);
        for(i=0;i<n;i++)
        {
            for(j=i+1;j<n;j++)
            {
                s=__gcd(a[i],a[j]);
                if(s>1)
                {
                  printf("%lld\n",s);
                  f=1;
                  break;
                }
            }
            if(f==1)
                break;
        }
        if(f==0)
        {
            for(i=0;i<n;i++)
            {
                long long x=a[i];
                for(j=1;j*j*j<=x;j++)
                {
                    if(j>1&&x%(j*j)==0)
                    {
                        printf("%lld\n",j);
                        f=1;
                        break;
                    }
                    if(a[i]%j==0)
                    {
                        s1=x/j;
                        s2=(long long)sqrt(s1);
                        if((s2*s2)==s1 && s2>1)
                        {
                        printf("%lld\n",s2);
                        f=1;
                        break;
                        }
                    }
                }
                if(f==1)
                    break;
            }
        }
    }
    return 0;
}
