#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long t,i;
    scanf("%lld",&t);
    while(t--)
    {
        long long n,s,j,f=0,s1,s2;
        scanf("%lld",&n);
        long long ar[n+69];
        for(i=0;i<n;i++)
        scanf("%lld",&ar[i]);
        for(i=0;i<n;i++)
        {
            for(j=i+1;j<n;j++)
            {
                if(i!=j){
                s=__gcd(ar[i],ar[j]);
                if(s>1){
                    printf("%lld\n",s);f=1;break;}}
            }
            if(f==1)
                break;
        }
           if(f==0)
        {
            for(i=0;i<n;i++)
            {
                for(j=2;j*j*j<=ar[i];j++)
                {
                    s=j*j;
                    if(ar[i]%s==0)
                    {
                        printf("%lld\n",j);
                        f=1;
                        break;
                    }
                }
                if(f==1)
                    break;
            }
        }
        if(f==0)
        {
           for(i=0;i<n;i++)
            {
                for(j=1;j*j*j<=ar[i];j++)
                {
                    if(ar[i]%j==0)
                    {
                        s1=ar[i]/j;
                        s2=(long long)sqrt(s1)+1E-6;
                        if(s2*s2==s1&&s2>1){
                        printf("%lld\n",s2);
                        f=1;
                        break;}
                    }
                }
                if(f==1)
                    break;
            }
        }
    }
    return 0;
}
