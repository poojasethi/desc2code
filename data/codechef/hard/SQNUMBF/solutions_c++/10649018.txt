#include <bits/stdc++.h>
using namespace std;
long long gcd(long long a,long long b)
  {
      if(b==0)
        return a;
      return gcd(b,a%b);
  }

int main()
  {
      int t;
      long long x,result,y;
      scanf("%d",&t);
      while(t--)
      {
          result=1;
          int n;
          scanf("%d",&n);
          long long a[n];
          for(int i=0;i<n;i++)
            scanf("%lld",&a[i]);
            for(int i=0;i<n-1;i++)
            {
                for(int j=i+1;j<n;j++)
                {
                    x=gcd(a[i],a[j]);
                    if(x>1)
                    {
                      result=x;
                      break;
                    }

                }
                if(result>1)
                    break;
            }

            if(result==1)
            {
               for(int i=0;i<n;i++)
               {  for(long long j=1;j*j*j<=a[i];j++)

                   {
                          if(j>1&&a[i]%(j*j)==0)
                          {
                              result=j;
                              break;
                          }
                          if(a[i]%j==0)
                          {
                              y=sqrt(a[i]/j);
                              if(y*y==a[i]/j)
                              {
                                  result=y;
                                  break;
                              }
                          }
                   }
                   if(result>1)
                    break;
               }
            }
            printf("%lld\n",result);
      }
  }
