#include<bits/stdc++.h>
#define ll long long int
#define mp make_pair
#define pb push_back
#define si(x) scanf("%d",&x)
#define pi(x) printf("%d\n",x)
#define s(x) scanf("%lld",&x)
#define p(x) printf("%lld\n",x)

using namespace std;

vector<ll> p;
bool b[1000000];
void sieve(){
    int i,j;
    p.pb(2);
    for(i=3;i<=1000000;i+=2){
        if(!b[i]){
            p.pb(i);
            for(j=3*i;j<=1000000;j+=2*i){
                b[j]=true;
            }
        }
    }
}

int main(){
memset(b,false,sizeof(b));
sieve();
int t;
si(t);
while(t--){
    int n,i,j;
    si(n);
    ll x=1;ll a[n+1];int flag=0;
    for(i=1;i<=n;i++){
        s(a[i]);
    }

    for (i=1;i<n;i++){
        for (j=i+1; j<=n;j++) {
            ll ans =__gcd(a[i],a[j]);
            if (ans>1) {
                p(ans);flag=1;break;
            }
        }
        if(flag) break;
    }
    if(!flag){
        for(i=1;i<=n;i++){
            for(j=0;(j<p.size() && p[j]*p[j]<=a[i]);j++){
                ll x=p[j];
                if(a[i]%(x*x)==0){
                    p(x);flag=1; break;
                }
               if(a[i]%x==0){
                a[i]/=x;
               }
            }
            if(flag) break;
            ll y=(ll)(sqrt((double)(a[i])));
            if(y>1 && a[i]==y*y) {
                p(y);break;
            }
        }
    }

}
return 0;
}
