#define pb push_back
#define ppb pop_back
#define mp make_pair
#define all(x) (x).begin(),(x).end()
#define sz(x) (int)(x).size()
#define LL long long
#define bit __builtin_popcountll
#define sqr(x) (x) * (x)
using namespace std;
typedef pair<int, int> pii;
const double eps = 1e-9;
const double pi = acos(-1.0);
const int maxn = 66666;
vector<pii> g[maxn];
bool used[maxn];
LL s[maxn];
void dfs(int v) {
	used[v] = true;
	for (int i = 0; i < sz(g[v]); i++) {
		int to = g[v][i].first;
		int add = g[v][i].second;
		if (used[to] && s[v] + add != s[to]) {
			cout << -1 << endl;
			exit(0);
		}
		s[to] = s[v] + add;
		if (!used[to]) dfs(to);
	}
}
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif
	int n,m; cin >> n >> m;
	for (int i = 0; i < m; i++) {
		int x,y,z; scanf("%d%d%d",&x,&y,&z);
		g[x - 1].pb(mp(y,z));
		g[y].pb(mp(x - 1,-z));
	}	
	for (int i = 0; i <= n; i++) {
		if (used[i]) continue;
		s[i] = 0;
		dfs(i);
	}
	for (int i = 1; i <= n; i++) {
		cout << s[i] - s[i - 1] << " ";
	}
	return 0;
}
