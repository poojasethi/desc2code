#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
#include<stdlib.h>
#include<string.h>
using namespace std;
int vis[1000010];
struct two
{
        int y;
        long long c;
        two(int a=0,long long b=0)
        {
            y=a;
            c=b;
        }
};
vector<two> v[1000010];
vector<long long> res(1000010,0);

void dfs(int x)
{
                vis[x]=1;
                for(int i=0;i<(int)v[x].size();i++)
                {
                    int t=v[x][i].y;
                    if(vis[t])
                    {
                        if(res[t]!=res[x]+v[x][i].c)
                        {
                                printf("-1\n");
                                exit(0);
                        }
                    }
                    else
                    {
                            res[t]=res[x]+v[x][i].c;
                            dfs(t);
                    }
                }
            
}
int main()
{
    int n,m;
    scanf("%d %d",&n,&m);
    
    for(int i=0;i<m;i++)
    {
            int x,y;
            long long z;
            scanf("%d %d %lld",&x,&y,&z);
            v[--x].push_back(two(y,z));
            v[y].push_back(two(x,-z));
    }

    for(int i=0;i<=n;i++)
        if(!vis[i])dfs(i);

    for(int i=1;i<=n;i++)
            printf("%lld ",res[i]-res[i-1]);

    printf("\n");
    return 0;
}
