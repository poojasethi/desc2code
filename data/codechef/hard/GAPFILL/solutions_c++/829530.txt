#define PB push_back
#define LL long long
#define MOD 1000000007
using namespace std;

LL power(LL a, LL b)
{
   LL x = 1LL, y = a;
   while (b>0)
   {
        if (b&1)
        {
           x = (x*y) % MOD;        
        }      
        y = (y*y) % MOD;
        b /= 2;
   }        
return x;
}  

int main()
{
    int t;
    scanf("%d",&t);
    while (t--)
    {
          LL m,n;
          LL ans = 1LL;
          bool o = false;
          
          scanf("%lld",&m);
          if (m&1LL)
          {
             --m;
             o = true;        
          } 
          
          scanf("%lld",&n);
          if (n&1LL)
          {
              --n;
              o = true;     
          }
    
    LL p = ((m%(MOD-1)) * (n%(MOD-1))) % (MOD-1);
    ans = power(2LL,p);
    if (o)
    {
       ans += ans;
       if (ans>=MOD) ans -= MOD;      
    }
    printf("%lld\n",ans);
    }
    return 0;    
}
