#include<stdio.h>
#include<math.h>
#include<vector>

using namespace std;

int main()
{
    int mod=1000000007;
    vector<long long> pwr;
    pwr.push_back(2);
    for(long long i=1;i<64;i++)
	pwr.push_back((pwr[i-1]*pwr[i-1])%mod);

    int t;
    scanf("%d",&t);
    while(t--)
    {
	long long n,m,flag=0;
	scanf("%lld%lld",&n,&m);
	flag+=(m%2 | n%2);
	n=n-(n%2);
	m=m-(m%2);
	long long ans=1;
	n=n%(mod-1);
	m=m%(mod-1);
	n=(n*m+flag)%(mod-1);
	for(long long i=0;i<log2(n)+1;i++)
	    if(n&1<<i)
	    {
		ans=ans*pwr[i];
		ans=ans%mod;
	    }

	printf("%lld\n",ans);
    }
    return 0;
}
