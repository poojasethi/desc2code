#include <bits/stdc++.h>
using namespace std;
 
const int MOD = 1000000007;
const int MODP = MOD - 1;
typedef unsigned long long ull;
 
int bigMod(ull a, ull b)
{
	ull res=1;
	while(b)
	{
		if(b&1)
			res=(res*a)%MOD;
		a=(a*a)%MOD;
		b>>=1;
	}
	return (int)res;
}
 
int main() {
	ios_base::sync_with_stdio(0);cin.tie(0);
	int t;
	ull n, m, res;
	//cin>>t;
	scanf("%d", &t);
	//cout<<t<<endl;
	//cout<<bigMod(2, 9)<<endl;
	while(t--)
	{
		//cin>>n>>m;
		scanf("%llu%llu", &n, &m);
		bool odd =false;
		if(n&1)
		{	n--; odd=true;}
		if(m&1)
		{	m--;odd=true;}
		int p=((n%MODP)*(m%MODP))%MODP;
		res=bigMod(2, p);
		if(odd)
		{
			res=(2*res)%MOD;
		}
		//cout<<res<<endl;
		printf("%llu\n", res);
	}
	//cout<<"End"<<endl;
	return 0;
}
