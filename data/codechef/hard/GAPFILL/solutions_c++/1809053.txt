const LL mod = (LL)1e9 + 7;
LL binpow(LL a, LL b) {
	LL res = 1;
	while(b > 0) {
		if (b & 1) {
			res = res * a % mod;
		}
		a = a * a % mod;
		b >>= 1;
	}
	return res;
}
