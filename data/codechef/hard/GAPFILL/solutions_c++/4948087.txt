#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
ll BASE = 1000000007LL;
ll solve(ll n){
    //cout<<n<<"\n";
    ll z=1,p=2;
    while(n){
        if(n%2==0){
            n>>=1;
            p*=p;
            p%=BASE;
        }
        else {
            z = (z*p)%BASE;
            n--;
        }
    }
    return z;
}

int test(){
    ll n,m;
    bool odd = false;
    scanf("%lld %lld",&n,&m);
    if (n%2){
        n--;
        odd=true;
    }
    if (m%2){
        m--;
        odd=true;
    }
    ll b = BASE-1;
    ll ans = solve(((m%b)*(n%b))%b);
    //cout<<ans<<"\n";
    if (odd){
        ans = ans*2;
        ans%=BASE;
    }
    cout<<ans<<"\n";
    return 0;
}

int main(){
    int T; scanf("%d",&T);
    while(T--){test();}
}
