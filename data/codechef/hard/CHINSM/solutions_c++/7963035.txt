#include<bits/stdc++.h>
using namespace std;

#define REP(i,a,b) for(int i=a;i<b;i++)
#define rep(i,n) REP(i,0,n)

#define ll long long
#define ull unsigned ll

int arr[100099];
int ind[100099];
int main()
{
    int n,k;
    scanf("%d %d",&n,&k);
    int maxA=-1;
    for(int i=0;i<n;i++)
    {
        scanf("%d",arr+i);
        maxA = max(arr[i],maxA);
    }
    memset(ind,-1,sizeof ind);
    ll ans =0;
    int ci =-1;
    for(int i=0;i<n;i++)
    {
        int num = arr[i];

        if(k>=num)
        {
            ans += (i-(ci+1)+1);
            ind[num] = i;
          //  cout<<"check: "<<i<<endl;
            continue;
        }
        if(k==0 && num ==1)
        {
            ans += 1;
            ind[num] = i;
            ci = max(ci,i-1);
            continue;
        }
        for(ll f=0;f<=maxA;f+=num)
        {
            ll chk = f+k;
            if(chk==0)
                continue;
            if(chk<=100000 && ind[chk]>ci && ind[chk]!=-1)
            {
                ci = ind[chk];
            }
        }
        ans += (i-(ci+1)+1);
        ind[num] = i;
        //cout<<"check: "<<ans<<endl;
    }
    printf("%lld\n",ans);
    return 0;
}
