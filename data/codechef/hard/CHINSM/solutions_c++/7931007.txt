#include<stdio.h>
#include<math.h>
//#define gc getchar_unlocked
#define gc getchar
long int C[100001];
long int min(long int a,long int b)
{
    if(a>b)
        return b;
    return a;
}
long int scanint()
{
    register int c = gc();
    long int x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    return x;
}

int main()
{
    long int n,k,i,I,j,l,ii,jj,maxi;
    long long int ans;
    n=scanint();
    k=scanint();
    long int A[n],B[n][2],D[n],CC[n];
    j=-1;
    for(i=0;i<n;i++)
    {
        A[i]=scanint();
        CC[i]=n;
    }
    D[0]=0;
    I=0;
    for(i=1;i<n;i++)
    {
        if(A[D[I]]<A[i])
        {
            while(I>=0&&A[D[I]]<A[i])
            {
                CC[D[I]]=i;
                I--;
            }
            I++;
            D[I]=i;
        }
        else
        {
            I++;
            D[I]=i;
        }
    }
    for(i=0;i<=100000;i++)
        C[i]=n;
    I=0;
    for(i=n-1;i>=0;i--)
    {
        l=A[i]-k;
        if(l>0)
        {
            j=n;
            for(ii=1;ii<=sqrt(l);ii++)
            {
                if(l%ii==0)
                {
                    if(A[i]%ii==k)
                        j=min(j,C[ii]);
                    jj=l/ii;
                    if(A[i]%jj==k)
                        j=min(j,C[jj]);
                }
            }
            if(j!=n)
            {
                if(I>0&&B[I-1][1]>j||I==0)
                {
                    B[I][0]=i;
                    B[I][1]=j;
                    I++;
                }
            }
        }
        else if(l==0)
        {
            if(CC[i]!=n)
            {
                j=CC[i];
                if(I>0&&B[I-1][1]>j||I==0)
                {
                    B[I][0]=i;
                    B[I][1]=j;
                    I++;
                }
            }
        }
        C[A[i]]=i;
    }
    ans=0;
    maxi=-1;
    for(i=0;i<n;i++)
    {
        if(I>0&&B[I-1][1]==i)
        {
            maxi=B[I-1][0];
            I--;
        }
        ans=ans+i-maxi;
    }
    printf("%lld",ans);
    return 0;
}
