#include <bits/stdc++.h>
using namespace std;
 
const int N = 200000;
 
int a[N];
int n, k;
int main() {
	scanf("%d %d", &n, &k);
	int maxi = 0;
	for(int i = 0;i < n;i++){
		scanf("%d", &a[i]);
		maxi = max(maxi, a[i]);
	}
	int idx = 0;
	vector<int > last(N, -1);
	long long int ans = 0;
	for(int i = 0;i < n;i++){
		if(a[i] <= k) ans += (i - idx + 1);
		else{
			int cnt = idx;
			for(int j = k;j <= maxi; j += a[i]){
				if(last[j] + 1 > cnt) cnt = last[j] + 1;
			}
			idx = cnt;
			ans += i - idx + 1;
		}
		last[a[i]] = i;
	}
	printf("%lld", ans);
	return 0;
}
