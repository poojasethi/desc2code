#include<bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> pi;

#define MP make_pair
#define PB push_back
#define F first
#define S second
#define MOD 1000000007
#define MAX 100005

int flag[MAX];

int main()
{
	ios::sync_with_stdio(false);
	int n, k, i, j, maxi;
	ll ans;
	cin >> n >> k;
	int a[n];
	maxi = 0;
	for(i = 0; i < n; i++)
	{
		cin >> a[i];
		maxi = max(maxi, a[i]);
	}
	for(i = 0; i <= maxi; i++)
		flag[i] = -1;
	if(a[n - 1] > k)
	{
		for(i = k; i <= maxi; i += a[n - 1])
			flag[i] = n - 1;
	}
	int last = n;
	ans = 1;
	for(i = n - 2; i >= 0; i--)
	{
		if(a[i] > k)
		{
			if(flag[a[i]] != -1)
				last = min(last, flag[a[i]]);
			ans += (last - i);
			for(j = k; j <= maxi; j += a[i]) // This element can act both as 'a' and 'b' and so the updation.
				flag[j] = i;
		}
		else if(a[i] == k)
		{
			if(flag[a[i]] != -1)
				last = min(last, flag[a[i]]);
			ans += (last - i);
		}
		else
			ans += (last - i);
	}
	cout << ans << endl;
	return 0;
}
