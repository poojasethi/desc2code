LL A[100005]={};
LL B[100005]={};
LL ma=0;
LL f(LL a,LL b)
{
    return b-a;
}


int main()
{
    ios::sync_with_stdio(false);
    LL n,k,i,j,m=0,ans=1;
    cin>>n>>k;
    for(i=1;i<=n;i++)
        cin>>A[i],ma=max(ma,A[i]);
    B[A[1]]=1;
    for(i=2;i<=n;i++)
    {
        if(A[i]<=k)
        {
            ans+=f(m,i);
        }
        else
        {
            for(j=k;j<=ma;j+=A[i])
            {
                assert(j%A[i]==k);
                m=max(B[j],m);
            }
            ans+=f(m,i);
        }
        B[A[i]]=i;
    }
    cout<<ans<<endl;
    return 0;
}
