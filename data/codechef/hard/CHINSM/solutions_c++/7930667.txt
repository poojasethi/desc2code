#include<stdio.h>
#define gc getchar_unlocked
//#define gc getchar
long int C[100001];
long int scanint()
{
    register int c = gc();
    long int x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
    return x;
}
long int max(long int a,long int b)
{
    if(a>b)
        return a;
    return b;
}
int main()
{
    long int n,k,i,I,j;
    long long int ans;
    n=scanint();
    k=scanint();
    long int A[n],B[100005],maxA;
    maxA=0;
    for(i=0;i<n;i++)
    {
        A[i]=scanint();
        if(A[i]>maxA)
            maxA=A[i];
    }
    for(i=0;i<=100000;i++)
        C[i]=-1;

    ans=0;
    long int maxi=-1;
	for(i=0;i<n;i++)
	{
		if(k<A[i])
		{
			for(j=k;j<=maxA;j+=A[i])
			{
				if(C[j]!=-1)
				{
					maxi=max(maxi,C[j]);
				}
			}
		}
		ans=ans+i-maxi;
		C[A[i]]=i;
	}
    printf("%lld",ans);
    return 0;
}
