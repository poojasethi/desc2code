#include <bits/stdc++.h>
#define md 1000000000
#define ll long long
#define gc getchar_unlocked
using namespace std;
int h[100010];
int a[100010];
int main (){
	#ifndef ONLINE_JUDGE
			freopen("input.txt","r",stdin);	
	#endif
   int n,k;
   scanf("%d%d",&n,&k);
   int last=0,lastk=0;
   ll ans=0;
   for(int i=1;i<=n;i++){
	   scanf("%d",&a[i]);
   }
   
   for(int i=1;i<=n;i++){
	   
	   last=max(last,h[a[i]]);
	   int to=last;
	   if(a[i]>k)
			to=max(to,lastk);
		last=to;
		if(a[i]==k)
			lastk=i;
		else if(a[i]>k){
			int l=a[i]-k;
			
			for(ll j=1;j*j<=l;j++)
			{
				if(l%j==0){
					if(j>k)
					h[j]=i;
					if(l/j>k)
					h[l/j]=i;
				}
			}
			
		}
		ans+=i-to;
   }
   cout<<ans<<endl;
    return 0;
}
