#define MAX 100007
#define EPS 1e-9
int divv[MAX];
int main()
{
    //READ("in.in");
    //WRITE("out.out");
    int n,k,a;
    scanf("%d%d",&n,&k);
    ll res=0,limit=0,lastk=0,now;
    FOR(i,1,n)
    {
        scanf("%d",&a);
        now=divv[a];
        if(a>k)
        now=max(now,lastk);
        now=max(now,limit);
        limit=now;
       // pr2(now,res);
        res+=(i-now);
        if(a>k)
        {
            a-=k;
            for(int j=1;j*j<=a;++j)
            {
                if(a%j==0)
                {
                    if(j>k)divv[j]=i;
                    if(a/j>k)divv[a/j]=i;
                }
            }
        }
        else if(a==k)lastk=i;
    }
    printf("%lld\n",res);
    return 0;
}
