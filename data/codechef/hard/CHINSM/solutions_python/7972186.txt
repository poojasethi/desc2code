c=[]
for i in range(100001):
    c.append(-1)
n,k=map(int,raw_input().split())
l=list(map(int,raw_input().split()))
maxl=max(l)
ans=0
maxi=-1
for i in range(n):
    if k<l[i]:
        for j in range(k,maxl+1,l[i]):
            if c[j]!=-1:
                maxi=max(maxi,c[j])
    ans+=i-maxi
    c[l[i]]=i
print ans
