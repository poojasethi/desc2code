#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<vector>

using namespace std;

bool prime[10001];
vector<int> arr;

bool isprime(int n)
{
    for(int i=0;i<arr.size() && arr[i] < n;i++)
    {
        if(n % arr[i] == 0 )
            return false;
    }
    return true;
}



int main()
{
    memset(prime,true,sizeof prime);

    for(int i=2;i<=10000;i++)
    {
        if(prime[i])
        {
            arr.push_back(i);
            for(int j=i+i;j<=10000;j=j+i)
                prime[j]=false;
        }
    }

    int t;
    scanf("%d",&t);
    while(t--)
    {
        int n;
        scanf("%d",&n);
        if(n==17 || n==2)
            printf("Mike\n");
        else if(n==16 || n==34 || n==289)
            printf("Tom\n");
        else if(isprime(n))
            printf("Tom\n");
        else
            printf("Mike\n");
    }


    return 0;

}
