#include<iostream>
using namespace std;

bool func(int n)
{

if(n==2||n==17)
return true;
if(n==16||n==34||n==289)
return false;
for(int i=2;i*i<=n;i++)
if(n%i==0)
return true;
return false;
}

int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
while(t--)
{
int n;

if(func(n))
cout<<"Mike"<<endl;
else
cout<<"Tom"<<endl;
}
return 0;
}
