int solve(int n)
{
    if(n==17 || n==2)
        return 1;
    if(n==34 || n==16 || n==289)
        return 0;
    for(int i=2;i*i<=n;i++)
    {
        if(n%i==0)
            return 1;
    }
    return 0;

}
