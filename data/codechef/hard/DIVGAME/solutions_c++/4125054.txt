#include <cstdio>
#define s(x) scanf("%d",&x)
#define l int
bool Check(l n)
{
	if(n==2||n==17) return false;
	if(n==16||n==34||n==289) return true;
	for(l i=2;i*i<=n;i++)
	{
		if(n%i==0)
		return false;
	}
	return true;
}
l main()
{
	l t;
	s(t);
	while(t--)
	{
		l n;
		bool ans;
		s(n);
		ans=Check(n);
		if(ans==false) printf("Mike\n");
		else printf("Tom\n");
	}
	return 0;
}
