#include<iostream>
#include<stdio.h>

using namespace std;

int n;

void solve()
{
    if ( n == 2 ) 
    {
        cout<<"Mike"<<endl;
        return;
    }
    if ( n == 16 ) 
    {
        cout<<"Tom"<<endl;
        return;
    }
    if ( n == 17 ) 
    {
        cout<<"Mike"<<endl;
        return;
    }
    if ( n == 34 )
     {
         cout<<"Tom"<<endl;
         return ;
     }
    if ( n == 289 ) 
    {
        cout<<"Tom"<<endl;
        return;
    }
    for ( int i = 2; i * i <= n; i++ )
        if ( n % i == 0 )
        {
            cout<<"Mike"<<endl;
            return ;
        }
    cout<<"Tom"<<endl;    
    return;
}

int main()
{
    int T;
    cin>>T;
    while(T--)
    {
        cin>>n;
        solve();
    }
    return 0;
}
