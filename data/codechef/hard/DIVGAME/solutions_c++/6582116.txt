#include<iostream>

using namespace std;

bool check(int n)
{
	if(n==2||n==17){
		return true;
	}
	else if(n==16||n==34||n==289){
		return false;
	}
	for(int i = 2 ; i * i<=n ; i++){
		if(n%i == 0){
			return true;
		}
	}
	return false;
}

int main()
{
	int t;
	int n;
	cin>>t;
	while(t--){
		cin>>n;
		if(check(n)){
			cout<<"Mike"<<endl;
		}
		else cout<<"Tom"<<endl;
	}
	return 0;
}
