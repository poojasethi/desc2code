bool game(long int n)
{long int i;
	if(n==2)
	return(true);
	if(n==17)
	return(true);
	if(n==34)
	return(false);
	if(n==289)
	return(false);
	if(n==16)
	return(false);
	
	for(i=2;i<=sqrt(n);i++)
	{
		if(n%i==0)
		{
		return(true);	
		}
		
	}
	return(false);
}
