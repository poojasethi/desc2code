#include<iostream>
#include<cmath>
using namespace std;

bool isprime(int n)
{
	int s = sqrt(n);
	for(int i=2;i<=s;i++)
	{
		if(n%i==0) return false;
	}
	return true;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int n;
		cin>>n;
		int res=0;
		switch(n)
		{
			case 1:
			case 16:
			case 34:
			case 289:
				res = 2;
				break;
			case 2:
			case 17:
				res = 1;
				break;
		}
		
		
		if(res==0)
		{
			if(isprime(n))
				res = 2;
			else
				res = 1;
		}
		//cout<<n<<" ";
		if(res == 2)
			cout<<"Tom\n";
		else
			cout<<"Mike\n";
		
	}
	return 0;
}
