#include <bits/stdc++.h>
# define s(x) scanf("%lld",&x)
using namespace std;
#define pb push_back
#define mp make_pair
#define ll long long
#define ms(a,b) memset((a),(b),sizeof(a))
#define inf 1000000007
#define f(i,j) for(int i=0;i<j;i++)
ll check[10001]={0};
vector<ll>prime;
int main()
{
    ll t,n,i,j,m,p;
    s(t);
    for(i=2;i<101;i++)
    {
        if(check[i]==0)
        {
            prime.push_back(i);
            for(j=i*i;j<10001;j+=i)
            {
                check[j]=1;
            }
        }
    }
    for(j=i;j<10001;j++)
    {
        if(check[j]==0)
            prime.push_back(j);
    }
    while(t--)
    {
        s(n);
        if(n==289||n==16||n==34)
        {
            printf("Tom\n");
            continue;
        }
        m=n;
        p=prime.size();
        //printf("%lld nkmlcv\n",prime[p-1]);
        for(i=0;i<prime.size();i++)
        {
            p=prime[i];
            if(p==n)
            {
                if(p!=17&&p!=2)
                    printf("Tom\n");
                else
                    printf("Mike\n");
                break;
            }
            else if(n%p==0)
            {
                printf("Mike\n");
                break;
            }
        }
        if(i==prime.size())
        {
            printf("Tom\n");
            continue;
        }
    }
    return 0;
}
