#include <bits/stdc++.h>
using namespace std;

bool isprime(int n)
{
    for(int i = 2; i * i <= n; i ++)
        if(n % i == 0) return false;
    return true;
}

int main()
{
    int t,n;
    scanf("%d",&t);
    while(t --) {
        scanf("%d",&n);
        if(isprime(n)) {
            if(n == 2 || n == 17) puts("Mike");
            else puts("Tom");
        }
        else {
            if(n == 16 || n == 34 || n == 289) puts("Tom");
            else puts("Mike");
        }
    }
    return 0;
}
