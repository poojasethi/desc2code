#include <cstdio>
#include <cmath>
#define FOR(i,m,n) for(i=m;i<n;i++)

using namespace std;

bool isprime(long int n)
{
if(n==2) return true;
if(!(n & 1)) return false;

int i,temp=sqrt(n);

for(i=3;i<=temp;i+=2)
    if(n % i ==0) return false;
return true;
}

bool win(long int n)
{
    if(n==2 || n==17) return true;
    if(n==1 || n==16 || n==34 || n==289) return false;
    return !isprime(n);
}

int main()
{
    int t;long int n;
    scanf("%i",&t);
    while(t--)
    {
        scanf("%ld",&n);
        if(win(n)) printf("Mike");
        else printf("Tom");
        printf("\n");
    }
    return 0;
}
