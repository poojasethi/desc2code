#include <cstdio>

bool winner (int n){
    if (n == 2 || n == 17) return true;
    if (n == 16 || n == 34 || n == 289) return false;

    for (int i = 2; i * i <= n; i++)
        if (!(n % i)) return true;
    return false;
}

int main(){

    int t;

    scanf ("%d", &t);

    while (t--){
        int n;
        scanf ("%d", &n);
        if (winner(n))
            printf ("Mike\n");
        else
            printf ("Tom\n");
    }

    return 0;
}
