#include <bits/stdc++.h>
#define maxa 1000000

using namespace std;

int E[maxa+1], P[maxa+1], nP;

void Ankin() {
    for(int i=1;i<=maxa;i++) E[i]=i;
    nP=0;
    for(int i=2;i<=maxa;i++) {
        if (E[i]==i) P[++nP]=i;
        for(int j=1;j<=nP && P[j]<=E[i] && P[j]<=maxa/i;j++) E[P[j]*i]=P[j];
    }
}

bool isPrime(int x) {
    if (x<=maxa) return (E[x]==x);
    for(int j=1; j<=nP && P[j]<=x/P[j];j++)
        if (x%P[j]==0) return false;
    return true;
}

int main() {
    #ifndef ONLINE_JUDGE
    freopen("inp.txt","r",stdin);
    freopen("out.txt","w",stdout);
    #endif // ONLINE_JUDGE
    Ankin();
    int T; scanf("%d",&T);
    while (T--) {
        int x; scanf("%d",&x);
        if (isPrime(x)) {
            if (x==2 || x==17) printf("Mike\n");
            else printf("Tom\n");
        } else {
            if (x==16 || x==34 || x==289) printf("Tom\n");
            else printf("Mike\n");
        }
    }
}
