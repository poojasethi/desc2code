#include<bits/stdc++.h>
using namespace std;
int prime[10005];
int a[10009]={0};
int main()
{
	prime[0]=2;
	int  p=1;
	for(int i=3;i<10000;i+=2)
	{
		if(!a[i])
		{
			prime[p++]=i;
			for(int j=i*i;j<10000;j+=2*i)
			{
				a[j]=1;
			}
		}
	}
	//for(int i=0;i<100;i++)
	//cout<<prime[i]<<"\t";
	int t;
	cin>>t;
	while(t--)
	{
		long long int n,i;
		cin>>n;
		if(n==2|| n==17)
		cout<<"Mike\n";
		else if(n==289||n==34|| n==16)
		{
			cout<<"Tom\n";
		}
		else
		{
		
		for( i=0;i<p && prime[i]<n;i++)
		{
			if(n%prime[i]==0)
			{
				break;
			}
		}
		if(i<p && prime[i]<n)
		{
			cout<<"Mike\n";
		}
		else
		cout<<"Tom\n";
	   }
	}
}
