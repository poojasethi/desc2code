#include <bits/stdc++.h>
using namespace std;
int Solve(int n)
{
	if(n==2)
		return true;
	if(n==17)
		return true;
	if(n==16)
		return false;
	if(n==34)
		return false;
	if(n==289)
		return false;
	for(int i = 2;i*i<=n;i++)
	{
		if(n%i==0)
			return true;
	}
	return false;
}
int main()
{
	int T;
	cin>>T;
	while(T--)
	{
		int N;
		cin>>N;
		if(Solve(N))
			cout<<"Mike"<<endl;
		else
			cout<<"Tom"<<endl;
	}
	return 0;
}
