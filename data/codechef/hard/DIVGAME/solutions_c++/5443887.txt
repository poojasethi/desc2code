#include <cstdio>
#include<cmath>
#define TOM printf("Tom\n")
#define MIKE printf("Mike\n")
using namespace std;
int main()
{
    int i,flag;
    int t,N;
    scanf("%d",&t);
    while(t--){
        scanf("%d",&N);
        if(N==2 || N==17)
            MIKE;
        else if(N==16 || N==34 || N==289)
            TOM;
        else{
           int m=sqrt(N);
           flag=0;
           for(i=2;i<=m;i++){
                if(N%i==0){
                    flag=1;
                    MIKE;
                    break;
                }
           }
           if(flag!=1)
                TOM;
        }
    }
    return 0;
}
