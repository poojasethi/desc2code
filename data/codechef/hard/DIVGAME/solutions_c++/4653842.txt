#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
int isPrime(int n){
    int i;
    int j=sqrt(n);
    for (i=3;i<=j;i+=2)
        if (n%i==0)
            return 0;
    return 1;
}

main()
{
    int t;
    scanf("%d",&t);
    for(;t;t--)
    {
        int n;
        scanf("%d",&n);
        if(n==16||n==34||n==289) { printf("Tom\n"); continue; }
        if(n%2==0||n==17) { printf("Mike\n"); continue; }
        if(isPrime(n)) printf("Tom\n"); else printf("Mike\n");
    }
 return 0;
}
