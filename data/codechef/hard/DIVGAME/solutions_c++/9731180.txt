using namespace std;
#define N (int)(1e5+4)
bool mark[N];
vector<int> v;
void sieve(){
    for(int i=2;i<=10005;i++){
        if(mark[i])
            continue;
        v.pb(i);
        for(int j=2*i;j<=10005;j+=i){
            mark[j]=1;
        }
    }
}
void solve(){
    int n;
    scanf("%d",&n);
    int ans=0;
    if(n<=10005){
       ans=mark[n];
    }
    else{
        for(int i=0;i<v.size();i++){
            if(n%v[i]==0){
               ans=1;
               break;
            }
        }
    }
    if(n==2||n==17||n==16||n==34||n==289){
       ans=1-ans;
    }
    if(ans)
        printf("Mike\n");
    else
        printf("Tom\n");
}
int main(){
   //ios_base::sync_with_stdio(false);
   sieve();
   int t=1;
   scanf("%d",&t);
   for(int i=1;i<=t;i++){
       //printf("Case #%d: ",i);
       solve();
   }
   return 0;
}
