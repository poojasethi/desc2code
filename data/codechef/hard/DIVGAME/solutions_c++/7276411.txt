#include<iostream>
#include<cstdio>
#include<cmath>
#include<algorithm>
#include<vector>
#define ll long long int
 bool DIV[1000002];
 bool P[1000002];
using namespace std;
 int main()
 {
     vector<ll>Pr;
     ll n;
     int i,j,a,k;
     for(i=0;i<=1000000;i++)
        P[i]=1;
     P[0]=P[1]=0;
     for(i=2;i<1000;i++)
     {
         if(P[i]==1)
         for(j=i+i;j<=1000000;j=j+i)
         {
             P[j]=0;
         }
     }
     for(i=0;i<=1000000;i++)
     {
         if(P[i]==1)
            Pr.push_back(i);
     }
     a=Pr.size();
     for(i=2;i<=1000000;i++)
     {
         if(P[i]==1)
         {
             DIV[i]=1;
         }
         else
            DIV[i]=0;
     }
     DIV[2]=DIV[17]=0;
     DIV[16]=DIV[34]=DIV[289]=1;
     int t,ans,b;
    scanf("%d",&t);
     while(t>0)
     {
         t--;
         scanf("%lld",&n);
      //   n=1000003;
         if(n<=1000000)
         {
             if(DIV[n]==1)
                printf("Tom\n");
             else
                printf("Mike\n");
                continue;
         }
         else
         {
             ans=2;
             b=sqrt(n);
            for(i=0;i<a&&Pr[i]<=b;i++)
            {
                if(n%Pr[i]==0)
                    {
                        ans=1;
                        break;
                    }
            }
            if(ans==2)
                printf("Tom\n");
            else
                printf("Mike\n");
         }
     }
 }
