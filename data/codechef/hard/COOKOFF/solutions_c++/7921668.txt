#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
 
#define SI ({int x;scanf("%d",&x);x;})
 
#define MX 1000000
#define MXN 1010
 
int a[MXN];
int N,n,c1,c2;
 
int hmm(int d)
{
	int i1 = lower_bound(a,a+N,c1)-a;
	int i2 = upper_bound(a,a+N,c2)-a;
	int cnt = i2-i1, i, p, t = 0;
	
	p=c1;
	for(i=i1;i<i2;i++)
	{
		if(a[i]-p>d) return 0;
		if(i+1<i2 && a[i+1]-p>d) p=a[i], t++;
		if(i+1==i2 && t<n) p=a[i],t++;
	}
	
	if(c2-p>d || t>n) return 0;
	if(cnt>=n) return 1;
 
	p=c1;
	for(i=i1-1;i>=0;i--)
	{
		if(p-a[i]>d) break;
		cnt++; p=a[i];
	}
	if(cnt>=n) return 1;
 
	p=c2;
	for(i=i2;i<N;i++)
	{
		if(a[i]-p>d) break;
		cnt++; p=a[i];
	}
	if(cnt>=n) return 1;
 
	return 0;
}
 
 
int main()
{
	for(int kases=SI;kases--;)
	{
		N=SI; n=SI; c1=SI; c2=SI;
		if(c1>c2) swap(c1,c2);
		for(int i=0;i<N;i++) a[i]=SI;
		sort(a,a+N);
 
		if(hmm(0)){ puts("0"); continue; }
 
		int lo=0, hi=MX;
		while(lo+1<hi)
		{
			int mid=(lo+hi)>>1;
			if(hmm(mid)) hi=mid;
			else lo=mid;
		}
 
		printf("%d\n",hi);
	}
 
	return 0;
}
