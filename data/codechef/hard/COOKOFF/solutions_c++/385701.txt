#include <cstdio>
#include <algorithm>
using namespace std;
int t,m,n,c,cc,d,dd,l,r,h,a[1111],i,j;
bool u[1111];
bool check() {
  int z=0,i,j,l,r;
  for (i=cc; i<=dd; ) {
    for (j=i; j<dd && a[j+1]-a[i]<=h; j++);
    if (j==i) break; else z++;
    i=j;
  }
  if (dd==j) z--;
  if (d-a[j]>h || z>m) return false;
  z=dd-cc+1; l=cc-1; r=dd+1;
  while (z<m) {
    if (l<0 && r>=n) break;
    if (l>=0 && a[l+1]-a[l]<=h) l--,z++; else
      if (r<n && a[r]-a[r-1]<=h) r++,z++; else break;
  }
  return (z>=m);
}
int main() {
  scanf("%d",&t);
  while (t--) {
    scanf("%d%d%d%d",&n,&m,&c,&d);
    if (c>d) swap(c,d);
    for (i=0; i<n; i++) scanf("%d",&a[i]);
    sort(a,a+n);
    for (i=0; i<=n+2; i++) u[i]=false;
    for (i=n-1; i>=0; i--) if (a[i]<=c) {
      i++;
      for (j=n; j>i; j--) a[j]=a[j-1];
      a[i]=c; u[i]=true; cc=i;
      break;
    }
    if (i<0) {
      i=0;
      for (j=n; j>i; j--) a[j]=a[j-1];
      a[i]=c; u[i]=true; cc=i;
    }
    n++;
    for (i=n-1; i>=0; i--) if (a[i]<=d) {
      i++;
      for (j=n; j>i; j--) a[j]=a[j-1];
      a[i]=d; u[i]=true; dd=i;
      break;
    }
    if (i<0) {
      i=0;
      for (j=n; j>i; j--) a[j]=a[j-1];
      a[i]=d; u[i]=true; dd=i;
    }
    n++; l=0; r=1000000;
    while (l<r) {
      h=(l+r)/2;
      if (check()) r=h; else l=h+1;
    }
    printf("%d\n",r);
  }
  return 0;
}
