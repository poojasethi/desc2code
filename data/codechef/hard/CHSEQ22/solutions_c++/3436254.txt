LL expmod(LL a,LL b)
{
	LL x=1,y=a;
	while(b>0)
	{
		if(b&1) x=(x*y)%mod;
		y=(y*y)%mod;
		b>>=1;
	}
	return (x)%mod;
}

void dfs(int x,int visited[],vector<int> adj[])
{
	nodes++;
    visited[x]=1;
    int siz=adj[x].size();
    for(int i=0;i<siz;i++)
    {
        if(visited[adj[x][i]]==0)
            dfs(adj[x][i],visited,adj);
    }
}
