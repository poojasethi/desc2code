vector<ll int> v[100002];
ll int visit[100002]={0};

ll int dfs(int i)
{
	visit[i]=1;
	ll	int ans=1;
	for(ll int j=0;j<v[i].size();j++)
	{
		if(!visit[v[i][j]])
		ans+=dfs(v[i][j]);
	}
	return ans;
}

ll int f(ll int b,ll int e, ll int m)
{
ll int remainder;
ll int x = 1;

while (e != 0)
{
remainder = e % 2;
e= e/2;

if (remainder == 1)
x = (x * b) % m;  //(if x*b overflows use multiply mod(x,b,m))
b= (b * b) % m; // New base equal b^2 % m
}
return x;
}
