#include<bits/stdc++.h>

#define LL long long
#define inf 1000000
#define pb push_back
#define vsort(v) sort(v.begin(),v.end())
#define pi acos(-1)
#define clr(a,b) memset(a,b,sizeof a)
#define bclr(a) memset(a,false,sizeof a)
#define MOD 1000000007ll
#define MP make_pair
#define MX 100000

using namespace std;
int par[MX+10];

int find_par(int x){
    if(par[x]==0) return x;
    return par[x]=find_par(par[x]);
}

int main(){
    ios_base::sync_with_stdio(0);
    cin.tie();
    int n,m;
    cin>>n>>m;
    LL res=1ll;
    for(int i=0;i<m;i++){
        int a,b;
        cin>>a>>b;
        b++;
        a=find_par(a);
        b=find_par(b);
        if(a!=b){
            par[a]=b;
            res=(res*2ll)%MOD;
        }
    }
    cout<<res<<endl;
    return 0;
}
