vector<int>G[100001];

LL pow(int a,int b)
{
    if(b==0)return 1;
    LL tmp=pow(a,b/2);
    tmp=(tmp*tmp)%MOD;
    if(b%2==0)return tmp;
    return (tmp*2)%MOD;
}

int vis[100001];
int cnt=0;
void dfs(int i)
{
    cnt++;
    vis[i]=1;

    for(int j=0;j<G[i].size();++j)
    {
        if(!vis[G[i][j]])dfs(G[i][j]);
    }
}

    int n,m;
    s(n); s(m);

    int a,b;
    for(int i=0;i<m;++i)
    {
        s(a); s(b);
        G[a].pb(b+1);
        G[b+1].pb(a);
    }

    LL ans=1;
    for(int i=1;i<=n;++i)
    {
        if(!vis[i])
        {
            cnt=0;
            dfs(i);
            ans=(ans*pow(2,cnt-1))%MOD;
        }
    }

    cout<<ans<<endl;

     return 0;
}
