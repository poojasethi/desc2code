#include<cstdio>
#define For(i,n) for(int i=1;i<=n;i++)
#define N 100010
#define mo 1000000007
int n,m,x,y,ans,fa[N];
int getfa(int x)
{
    if (!fa[x]) return x;
    return fa[x]=getfa(fa[x]);
}
int main()
{
    scanf("%d%d",&n,&m);
    ans=1;
    For(i,m)
    {
        scanf("%d%d",&x,&y);
        y++;
        x=getfa(x);y=getfa(y);
        if (x!=y)
        {
            fa[x]=y;
            ans<<=1;
            if (ans>=mo) ans-=mo;
        }
    }
    printf("%d",ans);
    return 0;
}
