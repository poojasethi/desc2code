int f[N],b[N];
int find(int a)
{
	if(a==f[a])return a;
	return f[a]=find(f[a]);
}
ll pow(ll a,int b)
{
	ll ans=1;
	while(b>0)
	{
		if(b&1)ans=ans*a%mod;
		a=a*a%mod;b>>=1;
	}
	return ans;
}
int main()
{
	int n,m,i,j,k;
	while(~scanf("%d%d",&n,&m))
	{
		for(i=1;i<=n+1;i++)f[i]=i,b[i]=1;
		while(m--)
		{
			scanf("%d%d",&i,&j);
			j++;
			i=find(i);j=find(j);
			if(i!=j)
			{
				if(b[i]>b[j])
				{
					f[j]=i;b[i]+=b[j];
				}
				else f[i]=j,b[j]+=b[i];
			}
		}
		int ans=0;
		for(i=1;i<=n+1;i++)
		if(f[i]==i)ans+=b[i]-1;
		printf("%d\n",(int)pow(2LL,ans));
	}
}
