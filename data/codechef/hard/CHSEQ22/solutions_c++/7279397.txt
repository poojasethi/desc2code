#include<iostream>
#include<cstdio>
#include<vector>
#include<cstring>
#include<algorithm>
#include<set>
#define mod 1000000007ll
using namespace std;
typedef vector<int> vi;
typedef vector<vi> vvi;

long long expo(long long a,long long p)
{
	long long res=1;
	while(p)
	{
		if(p&1)res=res*a%mod;
		p>>=1;
		a=a*a%mod;
	}
	return res;
}
int n,m;
bool visited[100005];
int dfs(vvi& adj,int u)
{
	int count=0;
	visited[u]=true;
	for(int i=0;i<adj[u].size();i++)
	if(!visited[adj[u][i]])
	{
		count++;
		count+=dfs(adj,adj[u][i]);
	}
	return count;
}
int main()
{
	scanf("%d %d",&n,&m);
	vvi adj(n+2,vi(0));
	for(int x,y,i=0;i<m;i++)
	{
		scanf("%d %d",&x,&y);
		adj[x].push_back(y+1);
		adj[y+1].push_back(x);
	}
	
	for(int i=0;i<=n;i++)visited[i]=0;
	long long res=0,count=0;
	for(int i=1;i<=n;i++)
	if(!visited[i])
	{
		count+=dfs(adj,i);
	}
	res=expo(2,count);
	cout<<(res+mod)%mod<<"\n";
	return 0;
}
