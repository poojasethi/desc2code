template<class T>
inline void readInt( T &n )//fast input function
{
   n=0;
   T ch=getcx();int sign=1;
   while( ch < '0' || ch > '9' ){if(ch=='-')sign=-1; ch=getcx();}

   while(  ch >= '0' && ch <= '9' )
           n = (n<<3)+(n<<1) + ch-'0', ch=getcx();
   n=n*sign;
}

pair<char, int> v[100010];
int *P, K;
LL V[200020];

void solve() {
    int C = 0, R = 0;
    rbe(i, 1, K) {
        int j = 2 * C - i;

        P[i] = R > i ? min(R - i, P[j]) : 0;

        while(v[i - P[i] - 1] == v[i + P[i] + 1]) P[i]++;

        if(i + P[i] > R) {
            C = i;
            R = i + P[i];
        }
    }
}
