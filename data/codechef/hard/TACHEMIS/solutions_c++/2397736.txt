using namespace std;
 
#define LL long long
 
template <typename T> T sqr(T x) { return x * x; }
template <typename T> T abs(T x) { return x < 0 ? -x : x; }
 
#define N 1000111
 
int t, n;
char a[N];
int b[N], da[N], db[N], sum[N];
 
int get_sum(int l, int r)
{
	if (l > r)
		return 0;
 
	if (l == 0)
		return sum[r];
 
	return sum[r] - sum[l - 1];
}
 
int main()
{
	// freopen("input.txt", "r", stdin);
	// freopen("output.txt", "w", stdout);
	scanf("%d", &t);
	while (t--)
	{
		scanf("%d\n", &n);
		for (int i = 0; i < n; i++)
		{
			sum[i] = da[i] = db[i] = 0;
			scanf("%c %d\n", &a[i], &b[i]);
		}
 
		sum[0] = b[0];
		for (int i = 1; i < n; i++)
			sum[i] = sum[i - 1] + b[i];
 
		int l = 0, r = -1;
		for (int i = 0; i < n; i++)
		{
			int k = (i > r ? 1 : min(da[l + r - i], r - i));
			while (i + k < n && i - k >= 0 && a[i + k] == a[i - k]) ++k;
			da[i] = k--;
			if (i + k > r)
			{
				l = i - k;
				r = i + k;
			}
		}
		l = 0, r = -1;
		for (int i = 0; i < n; i++)
		{
			int k = (i > r ? 1 : min(db[l + r - i], r - i));
			while (i + k < n && i - k >= 0 && b[i + k] == b[i - k]) ++k;
			db[i] = k--;
			if (i + k > r)
			{
				l = i - k;
				r = i + k;
			}
		}
 
		long long ans = 0;
		for (int i = 0; i < n; i++)
		{
			ans += (long long) (b[i] * 1ll * (b[i] - 1ll) / 2ll) + b[i];
			if (da[i] == db[i])
				ans += (long long) get_sum(i + 1, i + da[i] - 1);
			else if (da[i] > db[i])
			{
				int k = min(da[i], db[i]);
				l = min(b[i - k], b[i + k]);
				ans += (long long) get_sum(i + 1, i + k - 1) + l;
			}
			else
			{
				int k = min(da[i], db[i]);
				--k;
				if (k == 0)
					continue;
				l = min(b[i - k], b[i + k]);
				ans += (long long) get_sum(i + 1, i + k - 1) + l;
			}
		}
		cout << ans << endl;
	}
    return 0;
}
