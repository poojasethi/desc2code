#include <cstdio>
#include <vector>
#include <ctime>
#include <cstring>
#define pii pair<int,int>
#define ull unsigned long long
using namespace std;
vector<pair<int,int> > v;
ull find (int n)
{
	int *p = new int [n+2];
	ull *sum = new ull [n+2];
	int i;
	int j;
	int c = 0;
	int r = 0;
	ull ans = 0;
	sum[0] = 0;
	p[0] = 0;
	for (i  = 1; i < n+1; i++) {
		p[i] = i < r ? min(p[2*c-i],r-i) : 0;

		while ( v[i+p[i]+1] == v[i-p[i]-1]) p[i]++;
		if(p[i] + i > r ) {
			c = i;
			r = p[i] + i;
		}
		ull x = (ull)v[i].second;
		ans += ((x+1)*x)/2;
		sum[i] = sum[i-1] + (ull)v[i].second;
	}
	for (i = 1; i < n+1; i++ ) {
		ans += sum[i+p[i]] - sum[i];
		if (i+p[i]+1 < n+1 && i-p[i]-1 >= 0 ) {
			if ( v[i+p[i]+1].first == v[i-p[i]-1].first ) {
				ans += (ull)min(v[i+p[i]+1].second,v[i-p[i]-1].second);
			}
		}
	}
	return ans;
}
int main()
{
	int t;
	int n;
	int i;
	int tym = clock();
	scanf("%d",&t);
	while (t--) {
		char str[10];
		int k;
		scanf("%d",&n);
		v.push_back(pii(-1,0));
		for (i = 0; i < n; i++) {
			scanf("%s",str);
			scanf("%d",&k);
			v.push_back(pii(str[0]-'A',k));
		}
		v.push_back(pii(-1,-1));
		ull ans = find(n);
		printf("%llu\n",ans);
		v.clear();
	}
	while ((float)(clock()-tym)/CLOCKS_PER_SEC < .8);
	return 0;
}
