#include <bits/stdc++.h>
using namespace std;
#define ms(a) memset(a, 0, sizeof a)

static int b[1001][1001];
static int c[1001], r[1001], d1[2002], d2[1001], d3[1001];

int n, m, t;

int main()
{
    cin >> t;
    while(t--)
    {
        ms(b); ms(c); ms(r); ms(d1); ms(d2); ms(d3);
        cin >> n >> m;

        for(int i=1; i<=m; ++i)
        {
            int x, y;
            cin >> x >> y;
            b[x][y]=r[x]=c[y]=d1[x+y]=1;
            if(y>=x)d2[y-x]=1;
            else d3[x-y]=1;
        }

        int cnt=0;
        for(int i=1; i<=n; ++i)
        {
            for(int j=1; j<=n; ++j)
            {
                int tmp=0;
                int chk=(j>=i)? d2[j-i]: d3[i-j];
                if(!r[i] && !c[j] && !d1[i+j] && !chk)
                {
                    if(i+1<=n && j+2<=n && b[i+1][j+2])tmp++;
                    if(i+1<=n && j-2>0 && b[i+1][j-2])tmp++;
                    if(i-1>0 && j+2<=n && b[i-1][j+2])tmp++;
                    if(i-1>0 && j-2>0 && b[i-1][j-2])tmp++;
                    if(i+2<=n && j+1<=n && b[i+2][j+1])tmp++;
                    if(i+2<=n && j-1>0 && b[i+2][j-1])tmp++;
                    if(i-2>0 && j+1<=n && b[i-2][j+1])tmp++;
                    if(i-2>0 && j-1>0 && b[i-2][j-1])tmp++;
                }
                if(tmp>=2)cnt++;
            }
        }
        cout << cnt << endl;
    }
    return 0;
}
