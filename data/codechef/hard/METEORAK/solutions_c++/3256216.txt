#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#define max1 1007
#define gc getchar_unlocked
using namespace std;
int a[max1][max1];
int dp[max1][max1];
int l[max1][max1];
int r[max1][max1];
int up[max1][max1];
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main()
{
    int n,m,k,aa,bb;
    scanint(n);
    scanint(m);
    scanint(k);
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
                a[i][j]=1;
    for(int i=0;i<k;i++)
    {
        scanint(aa);
        scanint(bb);
        a[aa-1][bb-1]=0;
    }
    /*cout<<"a=";
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++)
                cout<<a[i][j]<<" ";
        cout<<endl;}*/
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
        {
                if(!i)
                {
                                if(!a[i][j])
                                up[i][j]=0;
                                else
                                up[i][j]=1;
                }
                else if(!a[i][j])
                up[i][j]=0;
                else
                up[i][j]=up[i-1][j]+1;
        }
    for(int i=0;i<n;i++)
    {
        int q=-1;
        for(int j=0;j<m;j++)
        {
                if(!j)
                {
                                l[i][j]=a[i][j];
                                if(!a[i][j])
                                q=j;
                }
                else if(!i)
                {
                                if(!a[i][j])
                                {
                                         l[i][j]=0;
                                         q=j;                       
                                }
                                else
                                l[i][j]=j-q;
                }
                else
                {
                                if(!a[i][j])
                                {
                                         l[i][j]=0;
                                         q=j;
                                }
                                else if(!a[i-1][j])
                                          l[i][j]=j-q;
                                else
                                {
                                          l[i][j]=min(j-q,l[i-1][j]);
                                }
                }
        }
    }
    for(int i=0;i<n;i++)
    {
        int q=m;
        for(int j=m-1;j>=0;j--)
        {
                        if(j==m-1)
                        {
                                       r[i][j]=a[i][j];
                                       if(!a[i][j])
                                       q=j;
                        }
                        else if(!i)
                        {
                                                if(!a[i][j])
                                                {
                                                             r[i][j]=0;
                                                             q=j;
                                                }
                                                else
                                                r[i][j]=q-j;
                        }
                        else
                        {
                                                if(!a[i][j])
                                                {
                                                              r[i][j]=0;
                                                              q=j;
                                                }
                                                else if(!a[i-1][j])
                                                               r[i][j]=q-j;
                                                else
                                                r[i][j]=min(q-j,r[i-1][j]);
                        }
        }
    }
    /*cout<<"up="<<endl;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
                cout<<up[i][j]<<" ";
        cout<<endl;
    }
    cout<<"left="<<endl;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
                cout<<l[i][j]<<" ";
        cout<<endl;
    }
    cout<<"right="<<endl;
    for(int i=0;i<n;i++)
    {
        for(int j=0;j<m;j++)
                cout<<r[i][j]<<" ";
        cout<<endl;
    }*/
    int w[max1];
    for(int i=0;i<n;i++)
    {
        memset(w,0,sizeof(w));
        for(int j=0;j<m;j++)
                w[up[i][j]]=max(w[up[i][j]],l[i][j]+r[i][j]-1);
        int temp=0;
        for(int j=0;j<=i;j++)
        {
                temp=max(temp,w[i-j+1]);
                dp[j][i]=temp*(i-j+1);
        }
        for(int j=i;j>=0;j--)
                dp[j][i]=max(dp[j][i],max(dp[j][i-1],dp[j+1][i]));
    }
    int q,x,y;
    scanint(q);
    for(int i=0;i<q;i++)
    {
        scanint(x);
        scanint(y);
        printf("%d\n",dp[x-1][y-1]);
    }
    return 0;
}
