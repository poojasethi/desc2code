#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int dp[1005][1005],l[1005],r[1005],area[1005][1005];
int main()
{
	int n,m,k,q,x,y;
	scanf("%d%d%d",&m,&n,&k);
	while(k--)
	{
		scanf("%d%d",&x,&y);
		dp[x][y]=1;
	}
	for(int i=1;i<=m;i++)for(int j=1;j<=n;j++)
	if(dp[i][j]==0)dp[i][j]=dp[i-1][j]+1;
	else dp[i][j]=0;
	
	for(int i=1;i<=m;i++)
	{
		for(int j=0;j<=n;j++)l[j]=r[j]=j;
		
		for(int j=n;j>0;j--)
		{
			while(r[j]+1<=n&&dp[i][r[j]+1]>=dp[i][j])r[j]=r[r[j]+1];
		}
		/*cout<<"\nnext...........................\n";
		for(int j=1;j<=m;j++)cout<<next[j]<<" ";*/
		
		for(int j=1;j<=n;j++)
		{
			while(l[j]-1>0&&dp[i][l[j]-1]>=dp[i][j])l[j]=l[l[j]-1];
		}
		/*cout<<"\nprev...........................\n";
		for(int j=1;j<=m;j++)cout<<prev[j]<<" ";
		cout<<"\narea...........................\n";*/
		for(int j=1;j<=n;j++)
        {
        	if(dp[i][j]!=0)area[i-dp[i][j]+1][i]=max(area[i-dp[i][j]+1][i],r[j]-l[j]+1);
        	//cout<<A[i][j]*area[i-A[i][j]+1][i]<<" ";
        }
		int width=0;
		for(int j=1;j<=i;j++)
		{
            width=max(width,area[j][i]);
			area[j][i]=width*(i-j+1);
		}
	}
	for(int j=1;j<m;j++)
	{
		for(int i=1;i+j<=m;i++)
		{
			area[i][i+j]=max(area[i][i+j],max(area[i][i+j-1],area[i+1][i+j]));
		}
	}
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d%d",&x,&y);
		printf("%d\n", area[x][y]);
	}
}
