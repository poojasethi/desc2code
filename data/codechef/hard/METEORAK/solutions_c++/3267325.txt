void width()
{
	pair<int ,int> stk[1003];
	int indx=0,top=0,curnt=0,last=0,i=0,j=0,rr=0;
	for( i=1; i<=n; i++)
	{
		indx=0;
		stk[indx++]=make_pair( hight[i][1],1);
		for( j=2; j<=m; j++)
		{
			curnt=hight[i][j];
			last=j;
			while(indx and stk[indx-1].first > curnt)
			{
				row_width[i][ (i-1)+(stk[indx-1].first) ]=max(row_width[i][(i-1) + stk[ indx-1 ].first ], j-stk[indx-1].second );
				last=stk[indx-1].second;
				indx--;
			}
			if(!indx or stk[indx-1].first < curnt)	stk[indx++]=make_pair( hight[i][j],last);
		}
		while(indx)
		{
			row_width[i][i+stk[indx-1].first-1]=max(row_width[i][i+stk[indx-1].first-1],j-stk[indx-1].second);
			indx--;
		}

		for(rr = n; rr >= i; rr--)
			row_width[i][rr]=max(row_width[i][rr],row_width[i][rr+1]);

		for(rr = i; rr <= n; rr++)
			row_width[i][rr]*=(rr-i+1);
	}

	for( i = 2; i <= n ; i++ )
	{
		for( j = 1; i+j-1<=n ; j++ )
		{
			row_width[j][i+j-1]=max(row_width[j][i+j-1] , row_width[j][i+j-2]	);
			row_width[j][i+j-1]=max(row_width[j][i+j-1] , row_width[j+1][i+j-1]	);
		}
	}
}
