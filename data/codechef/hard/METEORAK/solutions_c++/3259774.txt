#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<queue>
#include<vector>
#include<string>
using namespace std;

#define ll long long
#define llu unsigned long long int
#define N 1010
#define all(x) x.begin(),x.end()
#define pii pair<int,int>
#define mp make_pair
#define pb push_back
#define x first
#define y second
#define inf 0x3f3f3f3f
int a[N][N],c[N][N],d[N][N],up[N][N],l[N][N],r[N][N],s[N];
int main()
{
	int n,m,i,j,k;
	while(~scanf("%d%d%d",&n,&m,&k))
	{
		memset(a,0,sizeof(a));
		while(k--)
		{
			scanf("%d%d",&i,&j);
			a[i][j]=1;
		}
		for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		if(a[i][j])up[i][j]=0;
		else up[i][j]=up[i-1][j]+1;
		for(i=1;i<=n;i++)
		{
			int top=0;
			for(j=1;j<=m;j++)
			{
				while(top&&up[i][j]<=up[i][s[top]])top--;
				if(top==0)l[i][j]=0;
				else l[i][j]=s[top];
				s[++top]=j;
			}
			top=0;
			for(j=m;j>=1;j--)
			{
				while(top&&up[i][j]<=up[i][s[top]])top--;
				if(top==0)r[i][j]=m+1;
				else r[i][j]=s[top];
				s[++top]=j;
			}
		}
		memset(c,0,sizeof(c));
		memset(d,0,sizeof(d));
		for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
		{
			c[i-up[i][j]+1][i]=max(c[i-up[i][j]+1][i],r[i][j]-l[i][j]-1);
		}
		for(int len=n;len>=1;len--)
		for(i=1;i+len-1<=n;i++)
		{
			j=i+len-1;
			c[i][j]=max(c[i][j],c[i-1][j]);
			c[i][j]=max(c[i][j],c[i][j+1]);
		}
		for(int len=1;len<=n;len++)
		for(i=1;i+len-1<=n;i++)
		{
			j=i+len-1;
			d[i][j]=max(c[i][j]*(j-i+1),max(d[i+1][j],d[i][j-1]));
		}
		scanf("%d",&k);
		while(k--)
		{
			scanf("%d%d",&i,&j);
			printf("%d\n",d[i][j]);
		}
	}
}
