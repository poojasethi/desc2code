#include<cstdio>
#include<iostream>
using namespace std;
int up[1003][1003];
int left1[1003][1003];
int right1[1003][1003];
int stack[1003];
int c[1003][1003];
int result[1003][1003];
int main()
{
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	int i,j;
	for(i=0;i<n+1;i++)
		for(j=0;j<m+1;j++)
			up[i][j]=1;
	int val1,val2;
	for(i=0;i<k;i++){
		scanf("%d%d",&val1,&val2);
		up[val1][val2]=0;
	}
	for(j=1;j<=m;j++)
	{
		up[0][j]=0;
		for(i=1;i<=n;i++)
			up[i][j]=up[i][j]*(up[i-1][j]+1);
	}
	for(i=1;i<=n;i++)
	{
		int top=0;
		for(j=1;j<=m;j++)
		{
			if(up[i][j]==0)
			{
				left1[i][j]=0;
				stack[top++]=j;
				continue;
			}
			left1[i][j]=1;
			while(top>0 && up[i][stack[top-1]]>=up[i][j])
			{
				left1[i][j]+=left1[i][stack[top-1]];
				top--;
			}
			stack[top++]=j;
		}
		top=0;
		for(j=m;j>=1;j--)
		{
			if(up[i][j]==0)
			{
				right1[i][j]=0;
				stack[top++]=j;
				continue;
			}
			right1[i][j]=1;
			while(top>0 && up[i][stack[top-1]]>=up[i][j])
			{
				right1[i][j]+=right1[i][stack[top-1]];
				top--;
			}
			stack[top++]=j;
		}
	}
	for(i=0;i<n+2;i++)
		for(j=0;j<n+2;j++)
			c[i][j]=0;
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			c[i-up[i][j]+1][i]=max(c[i-up[i][j]+1][i],right1[i][j]+left1[i][j]-1);
	int len;
	for(len=n-1;len>=1;len--)
		for(i=1;i+len-1<=n;i++)
			c[i][i+len-1]=max(c[i][i+len-1],max(c[i-1][i+len-1],c[i][i+len]));
	/*	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++)
		cout<<c[i][j]<<" ";
		cout<<endl;
		}*/
	for(i=0;i<n+2;i++)
		for(j=0;j<n+2;j++)
			result[i][j]=0;
	for(len=1;len<=n;len++)
		for(i=1;i+len-1<=n;i++)
			result[i][i+len-1]=max(c[i][i+len-1]*len,max(result[i+1][i+len-1],result[i][i+len-2]));
	int q;
	scanf("%d",&q);
	while(q--)
	{
		scanf("%d%d",&val1,&val2);
		printf("%d\n",result[val1][val2]);
	}
	return 0;
}
