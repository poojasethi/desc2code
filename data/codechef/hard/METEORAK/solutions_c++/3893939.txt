#include <iostream>
#include <cstring>
#include <cstdio>
#include <vector>
#include <map>
#include <algorithm>
using namespace std;

int d[1005][1005];
int h[1005], l[1005], r[1005];
int ans[1005][1005];
int main()
{
//	freopen("in.txt", "r", stdin);
	int n, m, k, q, x, y;
	scanf("%d%d%d", &n,&m,&k);
	for(int i=0;i<k;++i)
	{
		scanf("%d%d", &x, &y);
		d[x-1][y-1]=1;
	}
	for(int i=0;i<n;++i)
	{
		for(int j=0; j<m;++j)
		{
			if(d[i][j])h[j]=0;
			else h[j]=h[j]+1;
		}
		for(int j=0; j<m; ++j)
		{
			if(j==0)l[j]=-1;
			else
			{
				int tmp=j-1;
				while(tmp!=-1 && h[j]<=h[tmp])tmp=l[tmp];
				l[j]=tmp;
			}
		}
		for(int j=m-1; j>=0; --j)
		{
			if(j==m-1)r[j]=m;
			else
			{
				int tmp=j+1;
				while(tmp!=m && h[j]<=h[tmp])tmp=r[tmp];
				r[j]=tmp;
			}
		}
		for(int j=0; j<m; ++j)if(h[j])ans[i][h[j]]=max(ans[i][h[j]], r[j]-l[j]-1);
		for(int j=i; j>=1; --j)
		{
			ans[i][j]=max(ans[i][j], ans[i][j+1]);
		}	
		for(int j=1; j<=i+1; ++j)
		{
			ans[i][j]=max(ans[i][j]*j, max(ans[i][j-1], (i==0)?0:ans[i-1][j-1]));
		}
	}
	scanf("%d", &q);
	for(int i=1; i<=q;++i)
	{
		scanf("%d%d", &x, &y);
		printf("%d\n", ans[y-1][y-x+1]);
	}	
	return 0;
}
