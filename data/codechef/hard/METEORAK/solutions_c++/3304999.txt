#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
#define MAX 1001
#ifndef ONLINE_JUDGE
inline int getchar_unlocked(){return getchar();}
#endif
inline int scnf(){int n = 0, c = getchar_unlocked(),t=0;while(c < '0' || c > '9'){if(c==45)t=1;c = getchar_unlocked();}while(c >= '0' && c <= '9')n = (n<<3) + (n<<1) + c - '0', c = getchar_unlocked();if(t==0)return n;else return -n;}

inline int mx(int a,int b){return a>b?a:b;}
int m,n,k,q,x,y,l[MAX],r[MAX],dp[MAX][MAX],C[MAX][MAX];
int main()
{
	m=scnf();n=scnf();k=scnf();
	while(k--)
	{
		x=scnf();y=scnf();
		dp[x][y]=1;
	}
	for(int i=1;i<=m;i++)
	{
		for(int j=1;j<=n;j++)
		if(dp[i][j])dp[i][j]=0;
		else dp[i][j]=dp[i-1][j]+1;
	}
	
	for(int i=1;i<=m;i++)
	{
		for(int j=n;j>0;j--)
		{
			r[j]=j;
			while(r[j]+1<=n&&dp[i][r[j]+1]>=dp[i][j])r[j]=r[r[j]+1];
		}
		for(int j=1;j<=n;j++)
		{
			l[j]=j;
			while(l[j]-1>0&&dp[i][l[j]-1]>=dp[i][j])l[j]=l[l[j]-1];
		}
		
		//width updation 
		for(int j=1;j<=n;j++)if(dp[i][j])C[i-dp[i][j]+1][i]=mx(C[i-dp[i][j]+1][i],r[j]-l[j]+1);
		
		//partial Area updation by mutiplying width with hight
		int height=0;
		for(int j=1;j<=i;j++)
		{
			height=mx(height,C[j][i]);
			C[j][i]=height*(i-j+1);
		}
	}
	//now complete area updation
	for(int i=1;i<m;i++)
	{
		for(int j=1;i+j<=m;j++)
		C[j][i+j]=mx(C[j][i+j],mx(C[j][i+j-1],C[j+1][i+j]));
	}
	q=scnf();
	while(q--)
	{
		x=scnf();y=scnf();
		printf("%d\n",C[x][y]);
	}
	return 0;
}
