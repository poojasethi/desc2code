#include <iostream>
#include <cstring>
using namespace std;
typedef long long int ll;
int main() {
	// your code goes here
	ll t;
	cin >> t;
	while(t--) {
	    string s1, s2, a, b;
	    cin >> s1 >> s2;
	    ll l1, l2;
	    l1 = s1.size();
	    l2 = s2.size();
	    ll i;
	    a = "";
	    b = "";
	    char ch = s1[0];
	    a = a + ch;
	    for(i = 1; i < l1; i++) {
	        if(s1[i] != ch) {
	            ch = s1[i];
	            a = a + ch;
	        }
	    }
	    ch = s2[0];
	    b = b + ch;
	    for(i = 1; i < l2; i++) {
	        if(s2[i] != ch) {
	            ch = s2[i];
	            b = b + ch;
	        }
	    }
	    if(a == b) {
	        cout << "Yes\n";
	    }
	    else {
	        cout << "No\n";
	    }
	}
	return 0;
}
