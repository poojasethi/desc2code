#include <bits/stdc++.h>
using namespace std;

int main()
{
    int i,t;
    cin >> t;
    while(t--)
    {
        string s,t,s1,t1;
        cin >> s >> t;
        s1+=s.at(0);
        
        for(i=0;i<s.size()-1;i++)
        if(s[i]!=s[i+1])
        s1+=s.at(i+1);
        
        t1+=t.at(0);
        
        for(i=0;i<t.size()-1;i++)
        if(t[i]!=t[i+1])
        t1+=t.at(i+1);
        //cout << s1 << endl << t1;
        if(s1.compare(t1) == 0)
        cout << "Yes" << endl;
        else
        cout << "No" << endl;
    }
    return 0;
}
