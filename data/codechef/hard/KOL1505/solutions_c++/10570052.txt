#include <stdio.h>
#include <stdlib.h>
#include <string.h>

int main()
{
    int t,i,j;
    scanf("%d",&t);
    while(t--)
    {
        char s[100],t[100];
        scanf("%s",s);
        scanf("%s",t);
        i=0,j=0;
        while(i<strlen(s)&&j<strlen(t))
        {
            while(i<strlen(s)-1)
            {
                if(s[i]==s[i+1])
                i++;
                else break;
            }
            while(j<strlen(t)-1)
            {
                if(t[j]==t[j+1])
                j++;
                else break;
            }
            if(s[i]!=t[j])
            break;
            i++;j++;

        }
        if(i==strlen(s)&&j==strlen(t))
        printf("Yes\n");
        else printf("No\n");
    }
    return 0;
}
