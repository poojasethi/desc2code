#include<iostream>
#include<string>
using namespace std;
int main()
{
	int T,i,j,temp;
	cin>>T;
	while(T)
	{
		string s,t;
		i=0,j=0,temp=1;
		cin>>s>>t;
		while(i<s.length()||j<t.length())
		{
			if(s[i] != t[j])
			{
				temp=0;
				break;
			}
			else
			{
				while(s[i]==s[i+1])
				i++;
				while(t[j]==t[j+1])
				j++;
			}
			i++;j++;
		}
		if(temp)
		{
			cout<<"Yes\n";
		}
		else
		{
			cout<<"No\n";
		}
		T--;
	}
}
