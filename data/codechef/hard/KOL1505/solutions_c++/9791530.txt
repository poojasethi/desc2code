#include <bits/stdc++.h>
using namespace std;
#define aa long long
string ans(string s)
{
    return string(s.begin(),unique(s.begin(),s.end()));
}
int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        string s1,s2;
        cin>>s1>>s2;
        if(ans(s1) == ans(s2))
        {
            cout<<"Yes"<<endl;
        }
        else cout<<"No"<<endl;
    }
}
