#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    std::ios::sync_with_stdio(false);
    cin>>t;
    while(t--)
    {
        string s,t;
        cin>>s;
        cin>>t;
        char s1[55],t1[55];
        long i,j,k,n;
        i=j=k=0;
        n=s.size();
        while(i<n)
        {
            s1[k++]=s[i];
            while(s1[k-1]==s[i])
            {
                i++;
            }
        }
        s1[k]='\0';
        i=j=k=0;
        n=t.size();
        while(i<n)
        {
            t1[k++]=t[i];
            while(t1[k-1]==t[i])
            {
                i++;
            }
        }
        t1[k]='\0';
       // cout<<s1<<" "<<t1<<"\n";
        if(strcmp(s1,t1)==0)
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
