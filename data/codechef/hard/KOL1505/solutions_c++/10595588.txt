#include<string>
#include<iostream>
#include<utility>
#include<vector>
using namespace std;

int main()
{
	int n,i,temp;
	cin>>n;
	getchar();
		
	while(n--){
		string a,b;
		getline(cin,a);
		getline(cin,b);
		bool flag=true;
		int j=0,count=1;
		vector<char> storeA;
		vector<char> storeB;
		char old=a.at(0);
		for(i=1;i<a.length();i++)// && j<b.length();)
		{
			if(a.at(i)==old)
				count++;
			else{
				storeA.push_back(old);
				old=a.at(i);
				count=1;
			}
		}
		storeA.push_back(old);
		old=b.at(0);
		count=1;
		for(i=1;i<b.length();i++)// && j<b.length();)
		{
			if(b.at(i)==old)
				count++;
			else{
				storeB.push_back(old);
				old=b.at(i);
				count=1;
			}
		}
		storeB.push_back(old);
		if(storeA.size()==storeB.size())
		{
			for(i=0;i<storeA.size();i++)
			{
				if(storeA[i]!=storeB[i])
				{
					//cout<<i<<endl;
					flag=false;
					break;
				}
			}
			if(flag)
				cout<<"Yes\n";
			else
				cout<<"No\n";
		}
		else{
			//cout<<"dasda\n"<<storeA.size()<<" "<<storeB.size()<<endl;
			cout<<"No\n"<<endl;
		}

	}
	return 0;
}
