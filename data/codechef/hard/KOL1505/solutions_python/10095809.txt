#include <bits/stdc++.h>
#include <string.h>
using namespace std;
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    char a[52];
    char b[52];
    char ap[52];
    char bp[52];
    scanf("%s",a);
    scanf("%s",b);
    // cout<<strlen(a);
    int m;
    m=0;
    for(int i=0;i<strlen(a);i++)
    {
      if(a[i]==a[i+1])
      {
        // cout<<"passed"<<endl;
      }
      else
      {
        // cout<<m<<" "<<a[i]<<endl;
        ap[m]=a[i];
        m++;
      }
    }
    ap[m]='\0';
    // for(int i=0;i<m;i++)
    // {
    //   cout<<ap[i]<<endl;
    // }
    // cout<<strlen(ap)<<endl;
    m=0;
    for(int i=0;i<strlen(b);i++)
    {
      if(b[i]==b[i+1])
      {
        // cout<<"passed"<<endl;
      }
      else
      {
        // cout<<m<<" "<<a[i]<<endl;
        bp[m]=b[i];
        m++;
      }
    }
    bp[m]='\0';
    if(strcmp(ap,bp)==0)
    {
      cout<<"Yes"<<endl;
    }
    else
    {
      cout<<"No"<<endl;
    }
  }
  return 0;
}
