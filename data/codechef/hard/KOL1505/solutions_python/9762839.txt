#include <iostream>
#include <vector>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 51;
char s[N], t[N];

void jalebiBai() {


    int T=0;
    scanf("%d", &T);

    while(T--) {

        scanf("%s%s", s, t);
        int ls = strlen(s);
        int lt = strlen(t);
        ls = unique(s, s+ls) - s;
        lt = unique(t, t+lt) - t;
        s[ls] = 0;
        t[lt] = 0;
        //cout << s << endl;
        //cout << t << endl;

        if(strcmp(s,t)==0) cout << "Yes" << endl;
        else cout << "No" << endl;

    }

}


int main()
{
    //cout << "Hello World!" << endl << endl;

    jalebiBai();

    return 0;
}
