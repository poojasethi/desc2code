#include <bits/stdc++.h>
using namespace std;

int main()
{
	int T;
	cin>>T;
	while(T--){
		char str[100],str2[100];
		scanf("%s%s",str,str2);
		int n = strlen(str);
		int n2 = strlen(str2);
		char s1[100],s2[100];
		int j=0;
		for(int i=0;i<n;i++,j++){
			s1[j]=str[i];
			char ch = str[i];
			while(str[i]==ch) i++;i--;
		}
		s1[j]='\0';
		j=0;
		for(int i=0;i<n2;i++,j++){
			s2[j]=str2[i];
			char ch = str2[i];
			while(str2[i]==ch) i++;i--;
		}
		s2[j]='\0';
		//cout<<s1<<" "<<s2<<endl;
		if(strcmp(s1,s2)==0)
			printf("Yes\n");
		else printf("No\n");
	}
	return 0;
}
