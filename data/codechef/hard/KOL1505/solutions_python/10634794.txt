#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
    int t,i,j;
    string a,b;
    vector<char> v1,v2;
    cin>>t;
    while(t--)
    {
        cin>>a>>b;
        v1.clear();
        v2.clear();
        for(i=0;a[i];i++)
        {
            for(j=i;a[j];j++)
                if(a[j]!=a[i])
                break;
            v1.push_back(a[i]);
            i=j-1;
        }
        for(i=0;b[i];i++)
        {
            for(j=i;b[j];j++)
                if(b[j]!=b[i])
                break;
            v2.push_back(b[i]);
            i=j-1;
        }
        if(v1.size()==v2.size())
        {
            for(i=0;i<v1.size();i++)
                if(v1[i]!=v2[i])
                break;
            if(i==v1.size())
                cout<<"Yes"<<endl;
            else
                cout<<"No"<<endl;
        }
        else
            cout<<"No"<<endl;
    }
    return 0;
}
