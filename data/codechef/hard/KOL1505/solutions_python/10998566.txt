#include<bits/stdc++.h>
using namespace std;

#define MAX 100

char str1[MAX],str2[MAX];
vector<char>S1; vector<char>S2;

int main(){
	char ref;
	int t,n1,n2,i;
	cin>>t;
	while(t--){
		cin>>str1;
		cin>>str2;
		n1=strlen(str1);
		n2=strlen(str2);
		S1.push_back(str1[0]); ref=str1[0];
		for(i=1;i<n1;++i){
			if(str1[i]==ref){
				continue;
			}
			else{
				S1.push_back(str1[i]);
				ref=str1[i];
			}
		}
		S2.push_back(str2[0]); ref=str2[0];
		for(i=1;i<n2;++i){
			if(str2[i]==ref){
				continue;
			}
			else{
				S2.push_back(str2[i]);
				ref=str2[i];
			}
		}
		if(S1.size()==S2.size()){
			int flag=0;
			for(i=0;i<S1.size();++i){
				if(S1[i]!=S2[i]) flag=1;
			}
			if(!flag) cout<<"Yes"<<endl;
			else cout<<"No"<<endl;
		}
		else{
			cout<<"No"<<endl;
		}
		S1.clear(); S2.clear();
	}
	return 0;
}
