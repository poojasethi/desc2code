#include <iostream>
#include <vector>
#include <string>

using namespace std;

string remove_duplicates(string c) {
    if(c.length()==1) return c;
    string d = c.substr(0,1);
    for (int j = 1; c[j]; j++) {
        if (c[j] != c[j-1]) d += c[j];
    }
    //cout << d << endl;
    return d;
}

void jalebiBai() {


    int test=0, T=0;
    cin >> T;
    std::vector<string> out(T);

    for(test=0; test<T; test++) {

        string s, t;
        cin >> s;
        cin >> t;

        if(s.length()==0 || t.length()==0) {
            out[test] = "No";
            continue;
        }

        s = remove_duplicates(s);
        t = remove_duplicates(t);

        if(s.compare(t)==0) out[test] = "Yes";
        else out[test] = "No";

    }

    for(test=0; test<T; test++) cout << out[test] << endl;

}


int main()
{
    //cout << "Hello World!" << endl << endl;

    jalebiBai();

    return 0;
}
