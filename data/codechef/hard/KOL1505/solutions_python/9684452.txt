#include <bits/stdc++.h>

using namespace std;

const int N = 110;

char s[N], t[N];

bool check(char *s, char *t) {
  int ls = strlen(s), lt = strlen(t);
  if (ls != lt) return 0;
  for (int i = 0; i < ls; ++i) {
    if (s[i] != t[i])
      return 0;
  }
  return 1;
}

int main() {
  int T;
  scanf("%d", &T);
  while (T--) {
    scanf("%s%s", s, t);
    int ls = strlen(s);
    int lt = strlen(t);
    ls = unique(s, s + ls) - s;
    lt = unique(t, t + lt) - t;
    s[ls] = 0;
    t[lt] = 0;
    puts(check(s, t) ? "Yes" : "No");
  } 
  return 0;
}
