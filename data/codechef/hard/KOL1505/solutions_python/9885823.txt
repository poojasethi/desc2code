#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	
	cin>>t;
	while(t--)
	{
		string a,b,aa,bb;
		cin>>a>>b;
		aa.push_back(a[0]);
		for(int i=1;a[i]!='\0';i++)
		{
			if(a[i]!=a[i-1]) aa.push_back(a[i]);
		}
		//aa.push_back('\0');
		bb.push_back(b[0]);
		for(int i=1;b[i]!='\0';i++)
		{
			if(b[i]!=b[i-1]) bb.push_back(b[i]);
		}
		//bb.push_back('\0');
		//cout<<aa<<" "<<bb<<"\n";
		if(aa==bb) cout<<"Yes\n";
		else cout<<"No\n";
	}
}
