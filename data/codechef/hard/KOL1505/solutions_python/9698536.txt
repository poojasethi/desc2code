#include<iostream>
#include<cstdio>
#include<string.h>
#include<algorithm>

using namespace std;

const int MAX=99;
char a[MAX],b[MAX],a_new[MAX],b_new[MAX];

int main(){
    int t,i,j,lena,lenb;
    scanf("%d",&t);
    while(t--){
        scanf("%s",a);
        scanf("%s",b);
        lena=strlen(a);
        lenb=strlen(b);

        a_new[0]=a[0];
        j=1;
        for(i=1;i<lena;i++){
            if(a[i]!=a[i-1]){
                a_new[j]=a[i];
                j+=1;
            }
        }
        lena=j;

        b_new[0]=b[0];
        j=1;
        for(i=1;i<lenb;i++){
            if(b[i]!=b[i-1]){
                b_new[j]=b[i];
                j+=1;
            }
        }
        lenb=j;

        if(lena==lenb){
            bool flag=1;
            for(i=0; i<lena && flag; i++){
                if(a_new[i]!=b_new[i]){
                    flag=0;
                }
            }
            if(flag==0){
                printf("No\n");
            }
            else{
                printf("Yes\n");
            }
        }
        else{
            printf("No\n");
        }

    }

    return 0;
}
