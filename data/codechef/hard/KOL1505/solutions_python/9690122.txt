#include <iostream>
#include <cstring>
using namespace std;
 
string Convert(string s) {
    long long int size = s.size(), i;
    string result (1, s[0]);
    i=0;
    long long int j=0;
 
    while(i<s.size()){
        if(s[i] != result[j]){
            result = result + s[i];
            j++;
        }
        i++;
    }
    return result;
}
 
int main (){
    long long int t;
 
    cin>>t;
 
    while (t--) {
        string a,b;
        cin>>a;
        cin>>b;
        if(Convert(a) == Convert(b)){
            cout<<"Yes\n";
        } else{
            cout<<"No\n";
        }
    }
}
