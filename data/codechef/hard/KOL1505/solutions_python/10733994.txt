#include<bits/stdc++.h>
using namespace std;
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        string s,p;
        cin>>s>>p;
        for(int i=1;i<s.length();i++)
        {
            if(s[i]==s[i-1])
            {
                s.erase(s.begin()+i);
                i--;
            }

        }

        for(int i=1;i<p.length();i++)
        {
            if(p[i]==p[i-1])
            {
                p.erase(p.begin()+i);
                i--;
            }
        }

        if(s.compare(p)==0)
            cout<<"Yes\n";
        else
            cout<<"No\n";
    }
    return 0;
}
