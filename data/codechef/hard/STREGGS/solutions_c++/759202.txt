#include<cstdio>
#include<cstdlib>
#include<climits>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define LIM 502
int a[LIM];
unsigned int dp[LIM][LIM];
int main()
{
	int t, n, k, i, j, K;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&K);
		for(i = 0; i < n; i++)
		scanf("%d", &a[i]);
		
		sort(a, a + n);
		memset(dp, 127, sizeof(dp));
		dp[0][n] = 0;
		
		for(k = 1; k <= K; k++)
		for(i = 0; i < n; i++)
		{
			dp[k][i] = INT_MAX;
			for(j = n; j > i; j--)
			dp[k][i] = min(dp[k][i] , dp[k - 1][j] + (j - i)*a[i]);	
		}

		unsigned int ans = INT_MAX;
		for(i=0; i<n; i++)
		ans = min(ans, dp[K][i]);
		printf("%u\n",ans);
	}
	return 0;	
}
