#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
 
int dp[501][501];
 
int main(){
	int T, K, N, A[501];
	scanf("%d", &T);
	while(T--){
		scanf("%d %d", &N, &K);
		for(int i=0; i<N; i++)
			scanf("%d", A+i);
		sort(A, A+N);
		memset(dp, 0x77, sizeof(dp));
		dp[0][0]=0;
		for(int i=0; i<N; i++)
		for(int j=0; j<=i; j++)
		for(int k=j; k<=i && (k==0 || j!=0); k++){
			dp[j+1][i+1]=min(dp[j+1][i+1], dp[j][k]+A[i]);
			dp[j][k]+=(k==0 ? 0 : A[k-1]);
		}
		int res=0x7fffffff;
		for(int i=0; i<=N; i++)
			res=min(res, dp[K][i]);
		printf("%d\n", res);
	}
	return 0;
}
