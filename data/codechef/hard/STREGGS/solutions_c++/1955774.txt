#include<iostream>
#include<stdio.h>
#include<string.h>
#include<vector>
#include<algorithm>
#include<map>
#include<math.h>
#include<climits>
using namespace std;

long long arr[510], dp[510][510];

int main()
{
    long long t, n, k, sum;
    scanf("%lld",&t);
    while(t--)
    {
        sum=0;
        scanf("%lld%lld",&n,&k);
        for(int i=1; i<=n; i++)
            scanf("%lld",&arr[i]);
        sort(arr+1, arr+n+1);
        for(int i=0; i<=n; i++)
        {
            //dp[0][i]=INT_MAX;
            dp[i][0]=0;
        }
        for(int i=1; i<=n; i++)
        {
            sum+=arr[i];
            dp[i][i]=sum;
        }
        for(int j=1; j<=k; j++)
        {
            for(int i=j+1; i<=n; i++)
            {
                long long temp=INT_MAX;
                for(int l=j; l<=i; l++)
                {
                    long long temp2=arr[l]*(i-l+1);
                    temp2+=dp[l-1][j-1];
                    if(temp2<temp)
                        temp=temp2;
                }
                dp[i][j]=temp;
            }
        }
//        for(int i=1; i<=n; i++)
//        {
//            for(int j=1; j<=n; j++)
//                printf("%d  ",dp[i][j]);
//            printf("\n");
//        }
        printf("%lld\n",dp[n][k]);
    }
    return 0;
}
