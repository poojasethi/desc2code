#include<stdio.h>
#include<algorithm>

using namespace std;

int arr[505];
int dpp[505][505];
int n,k;

int solve(int ,int );

int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
              scanf("%d%d",&n,&k);
              int i,j;
              
              for(i=1;i<=n;i++)
              scanf("%d",&arr[i]);
              
              
              
              
              sort(arr+1,arr+n+1);
              
             
     
              
              for(i=0;i<=n;i++)
               for(j=0;j<=k;j++)
               dpp[i][j]=0;
               
               
              int ans;
              
             
              ans=solve(n,k);
              
              printf("%d\n",ans);
              
              
    }
    
    return 0;
}


int solve(int a, int b)
{
    
    
    
    int i;
    int temp;
    
    
    
    
    if(dpp[a][b]!=0)
    return dpp[a][b];
    
    if(b==0)
    return 0;
    
    int ans=0;
    
    
    if(a==b)
    {
            //printf(" in ");
            
            for(i=a;i>0;i--)
            ans += arr[i];
            
            dpp[a][b] = ans;
            return dpp[a][b];
    }
    ans=   arr[a] + solve(a-1,b-1);
    
    //int g=1;
    int g=1;
    for(i=a-1;i>=b;i--)
    {
                     temp =   arr[a-g]*g +     solve(a-g,b);
                     if(temp<ans)
                     ans = temp;
                     
                     g++; 
    }
    
    dpp[a][b] =  ans;
    
    return dpp[a][b];
    
}
