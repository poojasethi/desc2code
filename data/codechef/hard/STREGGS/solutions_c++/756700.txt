#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <assert.h>
#include <stdint.h>
#include <algorithm>

int Min(int a, int b) {
  return (a < b ? a : b);
}

int compare_inc (const void * a, const void * b) {
  if ( *(int*)a - *(int*)b > 0 ) return 1;
  else if ( *(int*)a - *(int*)b < 0 ) return -1;
  else return 0;
}

int dp[501][501];
int A[501];
 
int main() {
  int T, K, N;
  scanf("%d", &T);
  for(int t=1; t<=T; t++) {
    scanf("%d%d", &N, &K);
    for(int i=0; i<N; i++) {
      scanf("%d", &A[i]);
    }
    qsort(A, N, sizeof(int), compare_inc);

    for(int k=0; k<=K; k++) {
      for(int j=0; j<=N; j++) {
	dp[k][j] = 0x3FFFFFFF;
      }
    }	
    dp[0][N]=0;
    
    for(int k=1; k<=K; k++) {
      for(int j=0; j<=N; j++) {
	for(int i=0; i<j; i++) {
	  dp[k][i] = Min(dp[k][i], dp[k-1][j] + (j-i)*A[i]);
	}
      }
    }
    int minTap = dp[K][N];
    for(int i=0; i<N; i++) {
      minTap=Min(minTap, dp[K][i]);
    }
    printf("%d\n", minTap);
  }
  return 0;
}
