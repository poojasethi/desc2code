#define ALL(i,n)    for(i = 0; i < (n); i++)
#define FOR(i,a,b)  for(i = (a); i < (b); i++)
#define SET(p)      memset(p,-1,sizeof(p))
#define CLR(p)      memset(p,0,sizeof(p))
#define S(n)	    scanf("%d",&n)
#define P(n)	    printf("%d\n",n)
#define Sl(n)	    scanf("%lld",&n)
#define Pl(n)	    printf("%lld\n",n)
#define Sf(n)       scanf("%lf",&n)

using namespace std;

int a[505],dp[505][505];

int main()
{
    int t,n,m,i,j,k,l;
    S(t);
    while(t--)
    {
        S(n);  S(k);
        for(i=1;i<=n;i++)
        {
            S(a[i]);
        }

        sort(a+1,a+n+1);

        for(i=1;i<=n;i++)
        {
            dp[i][1]=(a[i]*(n-i+1));

            for(j=2;j<=min(i,k);j++)
            {
                dp[i][j]=dp[j-1][j-1]+ (a[i]-a[j-1])*(n-i+1);
                for(l=j;l<i;l++)
                {
                    dp[i][j]=min(dp[i][j], dp[l][j-1]+ (a[i]-a[l])*(n-i+1));
                }
            }
        }

        int ans=1000000000;
        for(i=k;i<=n;i++)
            ans=min(ans,dp[i][k]);

        P(ans);
    }
    return 0;
}
