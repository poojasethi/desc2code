#include<iostream>
#include<algorithm>
#include<limits.h>
#include<stdio.h>
using namespace std;
int main() {
int dp[505][505],t,n,k,a[505],i,j,l;
	scanf("%d",&t);
	while(t--) {
		//cin>>n>>k;
		scanf("%d%d",&n,&k);
		for(i=0;i<n;i++) {
			scanf("%d",&a[i]);
		}
		sort(a,a+n);
		for(i=0;i<=n;i++) {
			for(j=0;j<=n;j++) {
				dp[j][i]=INT_MAX;
			}
		}
		for(i=0;i<n;i++) {
			dp[i][0]=0;
			dp[i][1]=(n-i)*a[i];
		}
		 int min = INT_MAX;
		if(k==1) {
			for(i=0;i<n;i++) {
				int temp = a[i]*(n-i);
				if(temp<min)
				min = temp;
			}
		}
		else {
		for(i=2;i<=k;i++) { //broken
			for(j=0;j<n;j++) { //last
			if(j>=i-1) {
				for(l=i-2;l<j;l++) { 
					 int temp = dp[l][i-1]+(n-j)*(a[j]-a[l]);
					if(temp<dp[j][i])
					   dp[j][i] =temp;
					  // cout<<j<<" "<<i<<" "<<dp[j][i]<<" "<<endl;
					
				}
			}
				
			}
		}
		
		for(i=k-1;i<n;i++) {
			if(min>dp[i][k])
			min = dp[i][k];
		}
	}
		printf("%d\n",min);
	}
}
