#include<iostream>
#include<stdio.h>
#include<climits>
#include<cstring>
#include<math.h>
#include<algorithm>
#include<map>

using namespace std;

int main()
{
    int t,n,k;
    scanf("%d",&t);
    while(t--)
    {
        int a[505];
        scanf("%d%d",&n,&k);
        for(int i=1;i<=n;i++)
            scanf("%d",&a[i]);
        sort(a+1,a+n+1);
        long long dp[505][505];
        for(int i=0;i<=n;i++)
            dp[0][i]=0;
        for(int i=1;i<=n;i++)
        {
            dp[i][i]=0;
            for(int j=1;j<=i;j++)
                dp[i][i]+=a[j];
        }
        for(int i=1;i<=k;i++)
        {
            for(int j=(i+1);j<=n;j++)
            {
                long long minim=INT_MAX;
                for(int l=j;l>=i;l--)
                {
                    long long temp;
                    temp=dp[i-1][l-1]+a[l]*(j-l+1);
                    if(minim>temp)
                        minim=temp;
                }
                dp[i][j]=minim;
            }
        }
        printf("%lld\n",dp[k][n]);
    }
    return 0;
}
