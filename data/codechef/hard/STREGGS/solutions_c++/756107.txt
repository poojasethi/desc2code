
#include <iostream>
#include <algorithm>
#include <cstdio>

#define REP(i,n)        for(int i = 0; i < n; i++)
#define FOR(i,a,b,k)    for(int i = a; i < b; i+=k)

using namespace std;

int a[500], N, K;
int ans[500][501];

int get(int i, int j){
    if(i >= j) return ans[i][j];
    return 1e9;
}

int solve(){
    REP(i,N){
        ans[i][1] = a[i]*(N-i);
    }

    FOR(k,2,K+1,1){
        for(int i = N-k; i >=0; i--){
            ans[i][k] = 1e9;
            for(int j = i+1; j <= N-k+1; j++){
                ans[i][k] = min(ans[i][k], a[i]*(j-i) + ans[j][k-1]);
            }
        }
    }

    int ret = 1e9;
    REP(i,N-K+1){
        ret = min(ret, ans[i][K]);
    }
    return ret;
}

int main(){
    //freopen("input.txt", "r", stdin);
    int t;
    scanf("%d", &t);

    while(t--){
        scanf("%d %d", &N, &K);
        REP(i,N){
            scanf("%d", &a[i]);
        }
        sort(a,a+N);
        printf("%d\n", solve());
    }
}
