#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cstring>
#include<map>
#include<cmath>
#include<string>

using namespace std;

int a[500];
int dp[500][500];
int main()
{
    int n,k,t,i,j;

    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);          
        for(i=0;i<n;i++)
            scanf("%d",&a[i]);
        sort(a,a+n);
        for(i=0;i<n;i++)
            dp[i][1]=(n-i)*a[i];
        
        for(i=n-2;i>=0;i--)
            for(int K=2;K<=min(k,n-i);K++)
            {
                dp[i][K]=999999999;
                for(j=i+1;j<n;j++)
                    if(n-j>=K-1)
                        dp[i][K]=min(dp[i][K], dp[j][K-1] + (j-i)*a[i]);
            }
        
        int out=999999999;
        for(i=0;i<=n-k;i++)
            out=min(out,dp[i][k]);
        printf("%d\n",out);
    }    
//    system("pause");
    return 0;
}
