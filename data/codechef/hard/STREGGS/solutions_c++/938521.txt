#include <iostream>
#include <cstdio>
#include <climits>
#include <algorithm>
#include <queue>
#include <cstring>
#include <cmath>
using namespace std;
int main()
{
	int t,i,j,k,n,l;
	int a[600];
	//freopen("example.txt","r",stdin);
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d%d",&n,&k);
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		int dp[k+5][n+5],mini,x;
		for(i=0;i<n;i++)
			dp[0][i]=(n-i)*a[i];
		for(i=1;i<k;i++)
		{
			for(j=i;j<n;j++)
			{
				mini=INT_MAX;
				for(l=i-1;l<j;l++)
				{
					x=dp[i-1][l]-a[l]*(n-j);
					mini=min(mini,x);
				}
	 			dp[i][j]=mini+(n-j)*a[j];
			}
		}
		mini=INT_MAX;
		/*for(i=0;i<k;i++)
		{
			for(j=0;j<n;j++)
				cout<<dp[i][j]<<" ";
			cout<<endl;
		}*/
		for(i=k-1;i<n;i++)
		{
			mini=min(mini,dp[k-1][i]);
		}
		cout<<mini<<endl;
	}
	return 0;
}
