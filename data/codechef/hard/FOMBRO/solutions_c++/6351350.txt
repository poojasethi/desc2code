#include <bits/stdc++.h>
using namespace std;
typedef long long int lli;
#define MAX 1000005

lli a1[MAX],a2[MAX],a3[MAX];

lli modpow(lli a, lli b, int m)
{
    lli ans=1;
    while(b)
    {
        if(b&1)
            ans=(ans*a)%m;
        a=(a*a)%m;
        b>>=1;
    }
    return ans;
}

int main()
{
    int t,n,m,q,i,j,r;
    cin>>t;
    while(t--)
    {
        scanf("%d%d%d",&n,&m,&q);
        a1[0]=1;
        a3[0]=1;
        for(i=1;i<=n/2;i++)
            a1[i]=(a1[i-1]*modpow(i,i-1,m))%m;
        if(n&1) a2[n/2]=n/2+1;
        else a2[n/2]=1;
        for(i=n/2-1;i>=1;i--)
            a2[i]=((a2[i+1]*(i+1)%m)*(n-i))%m;
        for(i=1;i<=n/2;i++)
            a3[i]=(a3[i-1]*modpow(n-i+1,i,m))%m;
        while(q--)
        {
            scanf("%d",&r);
            r=min(r,n-r);
            lli ans=(((a1[r]*modpow(a2[r],r,m))%m)*(a3[r]))%m;
            printf("%lld\n",ans);
        }        
    }
}
