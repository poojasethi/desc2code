char P[500001];
vector<int> T;
void table(int n)
{	
  for(int i=0;i<n;i++)T[i]=-1;
  int cnd=0;
  int i=2;
  T[0]=-1;T[1]=0;
  for(;i<=n;)
  {
    if(P[i-1]==P[cnd])T[i++]=++cnd;
    else if (cnd>0) cnd=T[cnd];
    else T[i++]=0;
  }
}
int recur(int n)
{
	int x,y;
	if(n == 1 )return n;
	x = T[n];
	while (2*x != n) {
		if(n == 1 )return n;
		y = n-x; 
		if (y>x)return n;
		if (n%y)return n;
		n = y;
		x = T[n];
	}
	return x;
}
int main()
{
  int t,n,a,b;cin(t);
  while(t--){
    scanf("%s",P);
    n=strlen(P);
    T.clear();T.resize(n+1);
    table(n);
    int r=recur(n);
    cin(a);cin(b);
    int i;int ans=0;
    for(i=1;i<=2*n;i++)
    {
      ans+=(i&1)?a:b;
      ans%=n;
      if(ans%r==0){printf("%d\n",i);break;}
    }
  }

}
