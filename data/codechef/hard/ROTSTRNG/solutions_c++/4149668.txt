#include <iostream>     // std::cout
#include <algorithm>   // std::next_permutation, std::sort
#include <cstring>
#include <cstdio>
#include <cassert>

using namespace std;

#define max 500005
#define ass(n,a,b) assert(n>=a&&n<=b)

char str[2*max];

int main ()
{
    long int t,x,i,N,n,m,j,p,c=0;
    scanf("%d",&t);
    ass(t,1,10);
    while(t--)
    {
        c=0;
        scanf("%s",str);
        n = strlen(str);
        N = 2*n;
        //ass(n,1,500000);
        scanf("%ld%ld",&m,&p);
        int f[N+5],flag[N+5];
        for(i=0;i<n;i++)
            str[i+n]=str[i];
        for(i=1;i<=N;i++)
            flag[i]=0;
        flag[0]=1;
        i=1;
        j=0;
        f[0]=0;
        while(i<N)
        {
            if(str[i]==str[j]){
                f[i]=j+1;
                i++;
                j++;
            }
            else if(j>0){
                j=f[j-1];
            }
            else
            {
                f[i]=0;
                i++;
            }
            if(j==n-1)
            {
                flag[i-n+1]=1;
                j = f[n-1]-1;
            }
        }
        //for(i=1;i<=N;i++)
          //  cout<<flag[i]<<" ";
        int ind=0,ans=0,k;
        for(k=0;k<n;k++)
        {
            ind-=m; ans++; if(ind<0) ind+=n; if(flag[ind]) break;
            ind-=p; ans++; if(ind<0) ind+=n; if(flag[ind]) break;
        }
        cout<<ans<<endl;
    }
    return 0;
}
