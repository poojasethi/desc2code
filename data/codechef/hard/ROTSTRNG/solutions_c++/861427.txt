void sol(){

	int a, b;
	RE("%s %d %d",s,&a,&b);
	int n = strlen(s);
	ass(n > 0);
	p[1] = 0;
	FOR(i, 2, n) {
		int j = p[i-1];
		while (j > 0 && s[j] != s[i-1]) j = p[j];
		if (s[j] == s[i-1]) p[i] = j + 1;
		else p[i] = 0;
	}
	int k = n;
	if (n % (n - p[n]) == 0) k = n - p[n];
	
	int cur = 0;
	int ans = -1;
	a %= k;
	b %= k;
	FOR(I,1,k+k+1){
		if (I & 1) cur += a;
		else cur += b;
		if (cur >= k) cur -= k;
		if (cur == 0) {
			ans = I;
			break;
		}
	}
	cout << ans << endl;
	
}
