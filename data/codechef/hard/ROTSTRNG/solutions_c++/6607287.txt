char str[limit];
int F[limit];
bool S[limit];
void failure(string str)
{
	int size = str.size();
	F[0] = F[1] = 0;
	for(int i = 2; i <= size; i++) {
		int j = F[i - 1];
		while(true) {
			if(str[j] == str[i - 1]) {
				F[i] = j + 1; break; 
			}
			if(j == 0) {
				F[i] = 0; break;
			}
			else {
				j = F[j];
			}
		}
	}	
}

void print(int *arr, LL n)

	rep(i, n) cout << arr[i] << " ";
	cout << endl;
}

void solve()
{
	scanf("%s", str);
	string s = string(str);
	int l = s.size();
	failure(s);
	int mlen = l - F[l];
	if(mlen != 0 && (l % mlen == 0)) {
		memset(S, false, sizeof(S));
		for(int i = mlen; i < l; i += mlen) {
			S[i] = true;
		}
	}
	int n, m;
	s(n); s(m);
	int prein = 0;
	for(int i = 1; true; i++) {
		if(i & 1) {
			prein = (prein + n) % l;
			if(prein == 0 || S[prein] == 1) {
				pi(i); pn; return;
			}
		}
		else {
			prein = (prein + m) % l;
			if(prein == 0 || S[prein] == 1) {
				pi(i); pn; return;
			}
		}
	}
}

int main()
{
	int t;
	scanf("%d", &t);
	while(t--) {
		solve();
	}
	return 0;
}
