#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <climits>
#include <numeric>
#include <vector>
#include <cmath>
using namespace std;
const int MAX_N = 50000 + 10;
int n;
double L, R;
double x[MAX_N], y[MAX_N], z[MAX_N];
#define X first
#define Y second
pair<double,double>a[100020];
const double EPS = 1e-9;
bool check(double u) {
	int cnt = 0;
	for (int i = 0; i < n; ++i) {
		if (y[i] <= u)
			continue;
		double p = (y[i] - u) * z[i];
		double l = x[i] - p, r = x[i] + p;
		l = max(l, L);
		r = min(r, R);
		if (l + EPS <= r) {
			a[cnt].X = l;
			a[cnt].Y = r;
			++cnt;
		}
	}

	sort(a, a + cnt);
	double r = L;
	for (int i = 0; i < cnt; ++i) {
		if (a[i].X <= r + EPS) {
			r = max(r, a[i].Y);
		} else
			return false;
	}
	return r >= R - EPS;
}

int main() {
	cin >> n >> L >> R;
	for (int i = 0; i < n; ++i) {
		scanf("%lf%lf%lf", x + i, y + i, z + i);
		z[i] = tan(z[i] / 180 * M_PI);
	}
	double L = 0, R = 1000;
	while (L + 1e-7 < R) {
		double M = (L + R) / 2;
		if (check(M))
			L = M;
		else
			R = M;
	}
	printf("%0.10lf\n", L);
	return 0;
}
