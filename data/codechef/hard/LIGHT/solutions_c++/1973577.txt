#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#include <climits>
#include <numeric>
#include <vector>
#include <cmath>
using namespace std;
const int N = 50000 + 10;
int n;
double l,r;
double x[N], y[N], z[N];
#define X first
#define Y second
pair<double,double>a[100020];
const double eps = 1e-9;
bool check(double m)
{
 int cnt=0;
 for(int i=0;i<n;i++)
 {
  if(y[i]<=m)continue;
  double p=(y[i]-m)*z[i];
  double l1=x[i]-p,r1=x[i]+p;
  l1=max(l1,l);r1=min(r1,r);
  if(l1+eps<=r1)
  {
   a[cnt].X=l1,a[cnt++].Y=r1;
  }
 }
 sort(a,a+cnt);
 double t=l;
 for(int i=0;i<cnt;i++)
 if(a[i].X<=t)
 t=max(t,a[i].Y);
 else return 0;
 return t+eps>=r;
}
int main()
{
	//freopen("in.txt","r",stdin);
	int T,i,j;
	while(~scanf("%d%lf%lf",&n,&l,&r))
	{
	 for(i=0;i<n;i++)
	 {
	  scanf("%lf%lf%lf",&x[i],&y[i],&z[i]);
	  z[i]=tan(z[i]/180*M_PI);
	 }
	 double l=0,r=1000,m;
	 while(l+1e-7<r)
	 {
	  	m=(l+r)/2;
	  	if(check(m))l=m;
	  	else r=m;
	 }
	 printf("%.10lf\n",l);
	}
	return 0;
}
