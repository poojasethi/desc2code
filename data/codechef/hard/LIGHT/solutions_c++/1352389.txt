#include<cstdio>
#include<cmath>
#include<vector>
#include<utility>
#include<algorithm>
using namespace std;

#define SZ 50000
#define EPS 1e-9
#define PI acos(-1.0)
int n;
double l, r;
double x[SZ], y[SZ], deg[SZ];

bool f(double h) {
     vector<pair<double, double> > vp;
     for(int i=0; i<n; i++) {
         if(y[i]-h>=EPS) {
               double r=(y[i]-h)*tan(deg[i]/180*PI);
               vp.push_back(make_pair(x[i]-r, x[i]+r));
         }
     }
     if(vp.empty()) return false;
     sort(vp.begin(), vp.end());
     double x=vp[0].first, y=vp[0].second;
     for(int i=1; i<vp.size(); i++) {
          if(y-vp[i].first>-EPS) y=max(y, vp[i].second);
          else {
               if(l-x>-EPS && y-r>-EPS) return true;
               x=vp[i].first;
               y=vp[i].second;
          }
     }
     return (l-x>-EPS && y-r>-EPS);
}

int main() {
    scanf("%d %lf %lf", &n, &l, &r);
    for(int i=0; i<n; i++)
    scanf("%lf %lf %lf", x+i, y+i, deg+i);
    
    double low=-1, high=1000, h=-1;
    while(high-low>=EPS) {
         double mid=(low+high)/2;
         if(f(mid)) {
              low=mid;
              if(mid-h>-EPS) h=mid;
         }
         else {
              high=mid;
         }
    }
    
    printf("%.10lf\n", h);
}
