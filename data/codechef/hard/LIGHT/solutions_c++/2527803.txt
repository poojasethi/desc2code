#include <cstdio>
#include <algorithm>
#include <cmath>
#include <iostream>

using namespace std;

struct linea {
    char op;
    double pos;
    bool operator <(const linea &otro) const {
        if(pos<otro.pos)
            return true;
        return false;
    }
} L[100005];

bool sip;
int ctos, hay, va, antes, lleva;
double A[50005], B[50005], C[50005], x=0, y=0, izq, der;
double mid, dist;
const double p=3.14159265;

int main()
{
    scanf("%d%lf%lf",&ctos,&izq,&der);
    for(int i=1; i<=ctos; i++){
        scanf("%lf%lf%lf",&A[i],&B[i],&C[i]);
        if(B[i]>y)
            y=B[i];
    }
    while(y-x>=.0000001){
        mid=(x+y)/2.0;
        hay=0;
        for(int i=1; i<=ctos; i++){
            if(mid<=B[i]){
                dist=tan(C[i]*p/180.0)*(B[i]-mid);
                ++hay;
                L[hay].pos=A[i]-dist;
                L[hay].op='+';
                ++hay;
                L[hay].pos=A[i]+dist;
                L[hay].op='-';
            }
        }
        sort(L+1, L+hay+1);
        sip=false;
        va=1;
        while(va<=hay){
            antes=va;
            lleva=1;
            ++va;
            while(lleva>0){
                if(L[va].op=='+')
                    ++lleva;
                else
                    --lleva;
                ++va;
            }
            --va;
            if(L[antes].pos<=izq and L[va].pos>der)
                sip=true;
            ++va;
        }
        if(sip)
            x=mid;
        else
            y=mid;
    }
    printf("%.10lf\n",x);
    return 0;
}
