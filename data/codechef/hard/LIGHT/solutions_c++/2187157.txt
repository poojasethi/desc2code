#define FOR(i,x,y) for(int i=(x);i<(y);i++)
#define MIN(x,y) ((x)<(y)?(x):(y))
#define MAX(x,y) ((x)>(y)?(x):(y))
#define ABS(x) ((x)>0?(x):-(x))
#define SQ(x) ((x)*(x))
#define PI 3.14159265359

using namespace std;

int n;
double l,r;
double x[50100],y[50100],z[50100];

struct data
{
    int s;
    double pos;
    bool operator < (const data &d) const
    {
        return pos>d.pos;
    }
};

priority_queue<data> que;
bool check(double h)
{
    while(que.size())que.pop();
    data temp;
    FOR(i,0,n){
        if(y[i]<h) continue;
        temp.pos=x[i]-(y[i]-h)*tan(PI*z[i]/180);
        temp.s=1;
        que.push(temp);
        temp.pos=x[i]+(y[i]-h)*tan(PI*z[i]/180);
        temp.s=-1;
        que.push(temp);
    }
    int count=0;
    temp.pos=l,temp.s=0;que.push(temp);
    temp.pos=r,temp.s=0;que.push(temp);
    while(que.size())
    {
        count+=(que.top()).s;
        //printf("%lf %d\n",que.top().pos,que.top().s);
        if(count==0)if(que.top().pos<=r && que.top().pos>=l)return 0;
        que.pop();
    }
    return 1;
}

int main()
{
    scanf("%d %lf %lf",&n,&l,&r);
    FOR(i,0,n) scanf("%lf%lf%lf",&x[i],&y[i],&z[i]);
    double low=0,high=1000;
    while(high-low>0.000001)
    {
        if(!check((low+high)/2)) high=(low+high)/2;
        else low=(low+high)/2;
    }
    //printf("%d\n",check(3.01));
    printf("%.7lf",(low+high)/2);
}
