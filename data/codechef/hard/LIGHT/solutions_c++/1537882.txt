const double PI=acos(-1.0);
const double EPS = 1e-7;
#define foreach(it, v) for (typeof((v).begin()) it = (v).begin(),stop=(v).end(); it != stop; ++it)
#define foreach_r(it, v) for (typeof((v).rbegin()) it = (v).rbegin(),stop=(v).rend(); it != stop; ++it)
template<class T> inline void Swap(T& a,T& b){a^=b,b^=a,a^=b;}

int n;
double L,R,x[50000+10],y[50000+10],z[50000+10];

inline bool check(double height){
    map<double,int> m;
    for(int i=0;i<n;++i){
        double dist = tan(z[i]) * (y[i]-height);
        if(dist>0){
            m[x[i] - dist] = 1;
            m[x[i] + dist] = -1;
        }
    }
    int nr=0;
    foreach(it,m){
        if(nr == 0 && it->first - L > EPS  && R - it->first > EPS)
            return false;
        if(it->first - R >= EPS)
            break;
        nr += it->second;
    }
    return nr>0;
}

        cin>>x[i]>>y[i]>>z[i];
        z[i]=z[i]/180.0*PI; //rad
    }
    
    //search
    
    double left=0,right=1000+EPS,m;
    for(int step=0;step<50;++step){
        m=left+(right-left)/2.0;
        if(check(m)) left = m;
        else right = m;
    }
    
    cout<<setprecision(9)<<fixed<<left<<'\n';

}
