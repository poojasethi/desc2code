#include <cstdio>
#include <algorithm>
#include <iostream>
#include <vector>
#include <cmath>

#define MAX 50010
#define INF 1100
#define EPS 1e-9

using namespace std;

double L, R;
int N;

struct light {
	double x,y,z;
} pontos[MAX];

const double PI=acos(-1.0);

pair<double,double> seg[MAX];

main()
{
scanf("%d %lf %lf",&N,&L,&R);
for(int i=0; i<N;i++)
	{
        scanf("%lf %lf %lf",&pontos[i].x,&pontos[i].y,&pontos[i].z);
	}

double ini=0;
double fim=INF;

while(fim>ini+EPS)
	{
        double meio=(ini+fim)/2;
        for(int i=0; i<N; i++)
		{
		if(meio>pontos[i].y+EPS)
			{
			seg[i].first=INF;
                	seg[i].second=INF;
	                continue;
			}
		seg[i].first=pontos[i].x-(pontos[i].y-meio)*tan(PI*pontos[i].z/180);
		seg[i].second=pontos[i].x+(pontos[i].y-meio)*tan(PI*pontos[i].z/180);
		}

	sort(seg,seg+N);
	double maior=-INF;
	bool passo=false;

	for(int i=0; i<N; i++)
		{
		if(seg[i].first<=L&&seg[i].second>=L&&!passo)
			{
			maior=seg[i].second;
	                passo=true;
        	        continue;
			}

		if(!passo) continue;
		if(maior+EPS>=R) break;
		if(seg[i].first<=maior+EPS)
			{
			maior=max(maior,seg[i].second);
			}
		else
			{
			passo=false;
			break;
			}
		}

        if(maior+EPS<R)
		{
		passo=false;
		}

	if(!passo)
		fim=meio;
        else
		ini=meio;

	}

printf("%.9lf\n",(ini+fim)/2+EPS);

return 0;

}
