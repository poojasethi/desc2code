#include <bits/stdc++.h>
#define _ ios_base::sync_with_stdio(false);cin.tie(0);
using namespace std;
#define pb push_back
#define pob pop_back
#define pf push_front
#define pof pop_front
#define mp make_pair
#define all(a) a.begin(),a.end()
#define bitcnt(x) __builtin_popcountll(x)
#define MOD 1000000007
#define MAXN 131075
#define BASE 31
typedef unsigned long long int uint64;
typedef long long int int64;
 

double x[50005],y[50005],z[50005];
double l,r;
int n;

bool chck(double val){
	double eps=1e-9;
	vector<pair<double,double> >v;
	for(int i=0;i<n;i++){
		double base=y[i]-val;
		if(base<=0)
		continue;
		base=base*tan(z[i]);
		v.pb(mp(x[i]-base,x[i]+base));
	}	
	sort(all(v));
	if(v.empty())
	return false;
	vector<pair<double,double> >interv;
	double curl=v[0].first,curr=v[0].second;
	for(int i=1;i<v.size();i++){
		if(v[i].first>curr){
			interv.pb(mp(curl,curr));
			curl=v[i].first;
			curr=v[i].second;
		}
		else{
			curr=max(curr,v[i].second);
		}
	}
	interv.pb(mp(curl,curr));
	for(int i=0;i<interv.size();i++){
		if((interv[i].first<=l)&&(interv[i].second>=r))
		return true;
	}
	return false;
}

int main(){
	int i;
	double pi=acos(-1);
	cin>>n>>l>>r;
	
	for(i=0;i<n;i++){
		cin>>x[i]>>y[i]>>z[i];
		z[i]*=pi;
		z[i]/=180.0;
	}
	double lo=0.0,hgh=1000,ans=-1;
	
	for(i=0;i<=100;i++){
		double mid=(lo+hgh);
		mid/=2;
		if(chck(mid)){
			ans=mid;
			lo=mid;
		}
		else{
			hgh=mid;
		}
	}
	printf("%.6lf\n",ans);
	return 0;
}
