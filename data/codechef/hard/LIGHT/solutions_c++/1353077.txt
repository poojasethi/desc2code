#include <cstdio>
#include <cstring>
#include <cmath>
#include <iostream>
#include <algorithm>
using namespace std;
const int N=50010;
const double pi=acos(-1.0);
const double EPS=1e-10;
int n;
double L,R,x[N],y[N],z[N];
struct node{
    double a,b;
}q[N];
bool cmp(node aa,node bb){
    return aa.a<bb.a;
}
int main(){
    scanf("%d%lf%lf",&n,&L,&R);
    for(int i=0;i<n;i++){
    scanf("%lf%lf%lf",x+i,y+i,z+i);
    z[i]=tan(z[i]/180.0*pi);
    }
    double l=0,r=1000.0,mid;
    for(int k=0;k<220;k++){
        mid=(l+r)*0.5;
        int j=0;
        for(int i=0;i<n;i++){
        if(y[i]<mid) continue;
        double t=(y[i]-mid)*z[i];
        q[j].a=max(L,x[i]-t);
        q[j].b=min(R,x[i]+t);
        if(q[j].a<q[j].b) j++;
        }
        sort(q,q+j,cmp);
        double right=L;
        int i;
        for( i=0;i<j;i++){
            if(q[i].a-right>EPS) break;
            else right=max(right,q[i].b);
        }
        if(i<j||R-right>EPS) r=mid;else l=mid;
        if(r-l<EPS) break;
    }
    printf("%.9lf\n",mid);
}
