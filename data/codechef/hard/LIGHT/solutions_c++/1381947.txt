
#include<stdio.h>
#include<iostream>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<math.h>
#define X first
#define Y second
using namespace std;
#define fr(i,n) for(int i=0;i<n;i++)
#define fo(i,n) for(int i=1;i<=n;i++)
#define fe(i,n) for(__typeof(n.begin()) i=n.begin();i!=n.end();i++)
pair<double,double>a[100020];
double x[100020];
double y[100020];
double z[100020];
double l,r,pi=acos(-1.);
int n,c;
int ok(double u)
{
	c=0;
	for(int i=0;i<n;i++)
		if(y[i]>u)
		{
			a[c].X=x[i]-(y[i]-u)*tan(z[i]);
			a[c].Y=x[i]+(y[i]-u)*tan(z[i]);
			c++;
		}
	sort(a,a+c);
	double p=l;
	for(int i=0;i<c;i++)
	{
		if(a[i].X>p+1e-9)
			return 0;
		p=max(p,a[i].Y);
		if(p>=r)
			return 1;
	}
	return 0;
}
int main()
{
	scanf("%d %lf %lf",&n,&l,&r);
	for(int i=0;i<n;i++)
	{
		scanf("%lf%lf%lf",x+i,y+i,z+i);
		z[i]*=pi/180;
	}
	double l=0,r=1000;
	for(int _=0;_<200;_++)
	{
		if(ok((l+r)/2))
			l=(l+r)/2;
		else
			r=(l+r)/2;
	}
	printf("%.10lf",(l+r)/2);
	return 0;
}
