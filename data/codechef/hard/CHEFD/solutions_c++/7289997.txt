void scan(int &x)
{
    register int c = gx();
    x = 0;
    for(;(c<48 || c>57);c = gx());
    for(;c>47 && c<58;c = gx())
    {
        x = (x<<1) + (x<<3) + c - 48;
    }
}
lld modu(lld a,lld b,lld c)
{
	lld res=1;
	while(b>0)
	{
		if(b&1)
		{
			res = (res*a)%c;
		}
		b=b>>1;
		a = (a*a)%c;
	}
	return res%c;
}
int n,m,q,l,r,p,d;
