using namespace std;
set<int> ss[6];
set<int>::iterator it,it2;
int arr[100009];
int main()
{
    int m,i,j,l,r,p,n,typ,a;
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%d",&arr[i]);
        if(arr[i]%2==0)ss[2].insert(i);
        if(arr[i]%3==0)ss[3].insert(i);
        if(arr[i]%5==0)ss[5].insert(i);
    }
    scanf("%d",&m);
    while(m--){
        scanf("%d",&typ);

        if(typ==1){
            vector<int> del;
            scanf("%d%d%d",&l,&r,&p);
            it=lower_bound(ss[p].begin(),ss[p].end(),l);
            for(it2=it;it2!=ss[p].end();it2++){
                a=*it2;
                if(a > r)break;
                arr[a]/=p;
                if(arr[a]%p!=0)del.pb(a);
            }
            for(i=0;i<del.size();i++){
                ss[p].erase(del[i]);
            }
        }
        else {
            scanf("%d%d",&l,&a);
            if(arr[l]%2==0)ss[2].erase(l);
            if(arr[l]%3==0)ss[3].erase(l);
            if(arr[l]%5==0)ss[5].erase(l);
            arr[l]=a;
            if(arr[l]%2==0)ss[2].insert(l);
            if(arr[l]%3==0)ss[3].insert(l);
            if(arr[l]%5==0)ss[5].insert(l);
        }
    }
