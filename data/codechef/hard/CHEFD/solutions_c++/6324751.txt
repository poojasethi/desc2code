#include <bits/stdc++.h>

using namespace std;

int A[100005];
set <int> s[10];
set <int> :: iterator it,it1;
int main()
{
    int n,q,type,x,y,p;
    cin >> n;
    for ( int i = 0; i < n; i++ ) {
        cin >> A[i];
        if ( A[i]%2 == 0 ) s[2].insert(i);
        if ( A[i]%3 == 0 ) s[3].insert(i);
        if ( A[i]%5 == 0 ) s[5].insert(i);
    }
    cin >> q;
    while ( q-- ) {
        cin >> type;
        if ( type == 1 ) {
            cin >> x >> y >> p;
            x--, y--;
            it = s[p].lower_bound(x);
            vector <int> v;
            for ( it1 = it; it1 != s[p].end(); it1++ ) {
                if ( (*it1) > y ) break;
                A[*it1] /= p;
                v.push_back(*it1);
            }
            for ( int i = 0; i < v.size(); i++ ) {
                s[p].erase(s[p].find(v[i]));
                if ( A[v[i]]%2 == 0 ) s[2].insert(v[i]);
                if ( A[v[i]]%3 == 0 ) s[3].insert(v[i]);
                if ( A[v[i]]%5 == 0 ) s[5].insert(v[i]);
            }
        }
        else {
            cin >> x >> y ;
            x--;
            if ( A[x]%2 == 0 ) s[2].erase(s[2].find(x));
            if ( A[x]%3 == 0 ) s[3].erase(s[3].find(x));
            if ( A[x]%5 == 0 ) s[5].erase(s[5].find(x));
            A[x] = y;
            if ( A[x]%2 == 0 ) s[2].insert(x);
            if ( A[x]%3 == 0 ) s[3].insert(x);
            if ( A[x]%5 == 0 ) s[5].insert(x);
        }
    }
    for ( int i = 0; i < n; i++ ) cout << A[i] << " ";
    return 0;
}
