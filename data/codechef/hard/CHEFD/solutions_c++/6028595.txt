#include <bits/stdc++.h>
using namespace std;

int main() {
    long long n,arr[100005],i,m,f,l,r,d,p;
    set<long long> myset[6];
    set<long long> :: iterator it;
    vector<long long> del;
    vector<long long> :: iterator it1;
    scanf("%lld",&n);
    for(i=0;i<n;i++)
    {
        scanf("%lld",&arr[i]);
        if(arr[i]%2==0)
            myset[2].insert(i);
        if(arr[i]%3==0)
            myset[3].insert(i);
        if(arr[i]%5==0)
            myset[5].insert(i);
    }

    scanf("%lld",&m);
    while(m--)
    {
        scanf("%lld",&f);
        if(f==1)
        {
            scanf("%lld %lld %lld",&l,&r,&p);
            l=l-1;
            r=r-1;
            for(it=myset[p].begin();it!=myset[p].end();it++)
            {
                if(*it>=l)
                    break;
            }
            while(it!=myset[p].end() && *it<=r)
            {
                arr[*it]/=p;
                if(arr[*it]%p)
                {
                    del.push_back(*it);

                }
                it++;
            }
            for(it1=del.begin();it1!=del.end();it1++)
                myset[p].erase(*it1);

            del.clear();
        }
        else
        {
            scanf("%lld %lld",&i,&d);
            i=i-1;
			if(arr[i]%2==0)
            	myset[2].erase(i);
			if(arr[i]%3==0)
           	 	myset[3].erase(i);
			if(arr[i]%5==0)
            	myset[5].erase(i);
            arr[i]=d;
            if(d%2==0)
                myset[2].insert(i);
            if(d%3==0)
                myset[3].insert(i);
            if(d%5==0)
                myset[5].insert(i);
        }

    }
     for(i=0;i<n;i++)
            cout<<arr[i]<<" ";

        cout<<endl;
    return 0;
}
