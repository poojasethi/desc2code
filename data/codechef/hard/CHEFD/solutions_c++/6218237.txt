#include<bits/stdc++.h>
using namespace std;
set<int> s2,s3,s5;
set<int> :: iterator it,it1;
int arr[1000000];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		scanf("%d",&arr[i]);
		if(arr[i]%2==0)s2.insert(i);
		if(arr[i]%3==0)s3.insert(i);
		if(arr[i]%5==0)s5.insert(i);
	}
	int m,l,r,p,d,x,ind;
	scanf("%d",&m);
	while(m--)
	{
		scanf("%d",&x);
		if(x==1)
		{
			scanf("%d%d%d",&l,&r,&p);
			if(p==2)
			{
				it=s2.lower_bound(l);
				while(it!=s2.end()&&(ind=*it)<=r)
				{
					arr[ind]/=2;
					if(arr[ind]%2!=0)
					{
						it1=it;
						++it;
						s2.erase(it1);
					}
					else ++it;
				}
			}
			else if(p==3)
			{
				it=s3.lower_bound(l);
				while(it!=s3.end()&&(ind=*it)<=r)
				{
					arr[ind]/=3;
					if(arr[ind]%3!=0)
					{
						it1=it;
						++it;
						s3.erase(it1);
					}
					else ++it;
				}
			}
			else
			{
				it=s5.lower_bound(l);
				while(it!=s5.end()&&(ind=*it)<=r)
				{
					arr[ind]/=5;
					if(arr[ind]%5!=0)
					{
						it1=it;
						++it;
						s5.erase(it1);
					}
					else ++it;
				}
			}
		}
		else
		{
			scanf("%d%d",&l,&d);
			if(arr[l]%2==0)
			{
				s2.erase(l);
			}
			if(arr[l]%3==0)
			{
				s3.erase(l);
			}
			if(arr[l]%5==0)
			{
				s5.erase(l);
			}
			if(d%2==0)
				s2.insert(l);
			if(d%3==0)
				s3.insert(l);
			if(d%5==0)
				s5.insert(l);
			arr[l]=d;
		}
	}
	for(int i=1;i<=n;++i)
			printf("%d ",arr[i]);
	return 0;
}
