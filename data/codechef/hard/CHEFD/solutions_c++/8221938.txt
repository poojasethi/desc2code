#include<bits/stdc++.h>
using namespace std;
int data[100005];
set<int>st[6];
int main()
{

   int n;
   scanf("%d",&n);
   for(int i=1;i<=n;i++)
   {
    scanf("%d",&data[i]);
    if(data[i]%2==0)st[2].insert(i);
    if(data[i]%3==0)st[3].insert(i);
    if(data[i]%5==0)st[5].insert(i);
   }
  int q;
  scanf("%d",&q);
  while(q--)
  {
       int l,r,p,d;
        int type;
        scanf("%d %d",&type,&l);
        int ans=0;
        if(type==1)
        {
           scanf("%d %d",&r,&p);
            set<int> ::iterator pos=lower_bound(st[p].begin(),st[p].end(),l);
            vector<int>del;
            for(set<int>::iterator it=pos;it!=st[p].end();it++)
            {
               if(*it>r)break;
               ans++;
                data[*it]/=p;
                if(data[*it]%p)del.push_back(*it);
            }
            int len=del.size();
            for(int i=0;i<len;i++)
                 st[p].erase(del[i]);
            //printf("%d\n",ans);
        }
        else
        {
          scanf("%d",&d);
           if(data[l]%2==0)st[2].erase(l);
           if(data[l]%3==0)st[3].erase(l);
           if(data[l]%5==0)st[5].erase(l);
           data[l]=d;
           if(data[l]%2==0)st[2].insert(l);
           if(data[l]%3==0)st[3].insert(l);
           if(data[l]%5==0)st[5].insert(l);

        }

  }
  for(int i=1;i<=n;i++)
    printf("%d ",data[i]);
   return 0;
}
