#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n, i, c, w, m, pt1, pt2;
    long long x, y;
    scanf("%d", &n);
    vector<int> wt1;
    vector<int> wt2;
    m=0;
    for(i=0; i<n; i++){
        scanf("%d%d", &w, &c);
        m+=w;
        if(w==1){
            wt1.push_back(c);
        }
        else{
            wt2.push_back(c);
        }
    }
    pt1=pt2=0;
    sort(wt1.begin(), wt1.end());
    sort(wt2.begin(), wt2.end());
    reverse(wt1.begin(), wt1.end());
    reverse(wt2.begin(), wt2.end());
    long long ans[m];
    long long cost=0;
    for(i=2; i<=m; i+=2){
        if(pt1==wt1.size())
            x = 0;
        else if(pt1==wt1.size()-1)
            x = wt1[wt1.size()-1];
        else
            x = wt1[pt1]+wt1[pt1+1];
        if(pt2==wt2.size())
            y = 0;
        else
            y = wt2[pt2];
        if(x>y){
            cost+=x;
            pt1+=2;
        }
        else{
            cost+=y;
            pt2++;
        }
        ans[i] = cost;
    }
    pt1=pt2=0;
    if(wt1.size()>=1){
        ans[1] = cost = wt1[0];
        pt1++;
    }
    else{
        ans[1]=cost=0;
    }
    for(i=3; i<=m; i+=2){
        if(pt1==wt1.size())
            x = 0;
        else if(pt1==wt1.size()-1)
            x = wt1[wt1.size()-1];
        else
            x = wt1[pt1]+wt1[pt1+1];
        if(pt2==wt2.size())
            y = 0;
        else
            y = wt2[pt2];
        if(x>y){
            cost+=x;
            pt1+=2;
        }
        else{
            cost+=y;
            pt2++;
        }
        ans[i] = cost;
    }
    for(i=1; i<=m; i++){
        printf("%lld ", ans[i]);
    }
    cout << endl;
    return 0;
}
