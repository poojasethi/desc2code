#include <iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

long long int wt1[100005],wt2[100005];
long long int dp[200005];
int ind1[200005],ind2[200005];
int main()
{
    int n,i,j,k,m,w;
    long long int c,x,y;
    scanf("%d",&n);
    j=0;k=0;m=0;
    for(i=0;i<n;i++)
    {
        scanf("%d%lld",&w,&c);
        m=m+w;
        if(w==1)
        {
            wt1[j]=c;
            j++;
        }
        else if(w==2)
        {
            wt2[k]=c;
            k++;
        }
    }
    sort(wt1,wt1+j);
    sort(wt2,wt2+k);
    dp[0]=0;
    ind1[0]=j;
    ind2[0]=k;
    for(i=1;i<=m;i++)
    {
        x=0;y=0;
        dp[i]=dp[i-1];
        ind1[i]=ind1[i-1];
        ind2[i]=ind2[i-1];
        if(i-1>=0&&ind1[i-1]>=1)
            x=dp[i-1]+wt1[ind1[i-1]-1];
        if(i-2>=0&&ind2[i-2]>=1)
            y=dp[i-2]+wt2[ind2[i-2]-1];
        if(x>y&&x>dp[i])
        {
            dp[i]=x;
            ind1[i]=ind1[i-1]-1;
            ind2[i]=ind2[i-1];
        }
        else if(y>=x&&y>dp[i])
        {
            dp[i]=y;
            ind1[i]=ind1[i-2];
            ind2[i]=ind2[i-2]-1;
        }
        if(i!=m)
        printf("%lld ",dp[i]);
        else if(i==m)
            printf("%lld",dp[i]);
    }
    return 0;
}
