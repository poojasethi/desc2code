#include<iostream>
#include<vector>
#include<algorithm>
#define ll long long
using namespace std;

struct element {
    ll ans;
    ll cnt1;
    ll cnt2;
};

vector <ll> twos;
vector <ll> ones;
vector <element> dp(200010);

main() {
    ll n, t1, t2, m=0;
    scanf("%lld", &n);
    for (int i=1; i<=1000; i++) {
        ones.push_back(0);
        twos.push_back(0);
    }
    for (ll i=0; i<n; i++) {
        scanf("%lld%lld", &t1, &t2);
        if (t1==1) ones.push_back(t2);
        if (t1==2) twos.push_back(t2);
        m+=t1;
    }
    sort(ones.rbegin(), ones.rend());
    sort(twos.rbegin(), twos.rend());

    dp[0].ans=0;
    dp[0].cnt1=0;
    dp[0].cnt2=0;
    dp[1].ans=ones[0];
    dp[1].cnt1=1;
    dp[1].cnt2=0;
    printf("%lld ", dp[1].ans);

    for (ll i=2; i<=m; i++) {
        if (dp[i-1].ans+ones[dp[i-1].cnt1]>dp[i-2].ans+twos[dp[i-2].cnt2]) {
            dp[i].ans=dp[i-1].ans+ones[dp[i-1].cnt1];
            dp[i].cnt1=dp[i-1].cnt1+1;
            dp[i].cnt2=dp[i-1].cnt2;
        }
        else {
            dp[i].ans=dp[i-2].ans+twos[dp[i-2].cnt2];
            dp[i].cnt1=dp[i-2].cnt1;
            dp[i].cnt2=dp[i-2].cnt2+1;
        }
        printf("%lld ", dp[i].ans);
    }
    printf("\n");
}
