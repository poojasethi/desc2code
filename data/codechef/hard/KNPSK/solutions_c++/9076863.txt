# include <vector>
# include <algorithm>
#include <stdio.h>
# define limit (int)1e5
using namespace std;
int N,W,C,i,M;
long long int cost[2*limit+1];
vector <int> one, two;
int main()
{
	int counter1=0,counter2=0;
	scanf("%d",&N);
	for(i=0;i<N;i++)
	{
		scanf("%d%d",&W,&C);
		if(W==1)
			one.push_back(-C);
		else
			two.push_back(-C);
		M+=W;
	}
	sort(one.begin(),one.end());
	sort(two.begin(),two.end());
	
	for(i=2;i<=M;i+=2)
	{
		if(counter1<one.size())
		{
			cost[i]=cost[i-2]-one[counter1];
			if(counter1+1<one.size())
				cost[i]=cost[i]-one[counter1+1];
			if(counter2<two.size()&&cost[i]<cost[i-2]-two[counter2])
				cost[i]=cost[i-2]-two[counter2++];
			else
				counter1+=2;
		}
		else // counter2 < two.size()
			cost[i]=cost[i-2]-two[counter2++];
	}
	counter1=0;
	if(counter1<one.size())
		cost[1]=-one[counter1++];
	counter2=0;
	for(i=3;i<=M;i+=2)
	{
		if(counter1<one.size())
		{
			cost[i]=cost[i-2]-one[counter1];
			if(counter1+1<one.size())
				cost[i]=cost[i]-one[counter1+1];
			if(counter2<two.size()&&cost[i]<cost[i-2]-two[counter2])
				cost[i]=cost[i-2]-two[counter2++];
			else
				counter1+=2;
		}
		else // counter2 < two.size()
			cost[i]=cost[i-2]-two[counter2++];
	}
	printf("%lld",cost[1]);
	for(i=2;i<=M;i++)
		printf(" %lld",cost[i]);
	printf("\n");
	return 0;
}
