#include <iostream>
#include <stdio.h>
#include <string.h>
#include <math.h>
#include <vector>
#include <string>
#include <map>
#include <set>
#include <queue>
#include <algorithm>
#define lli long long int
#define pb push_back
#define mod 1000000007
#define pii pair<int,int>

using namespace std;
multiset<lli> s1,s2,s11,s22;
multiset<lli>::reverse_iterator it1,it2;
multiset<lli>::iterator it3,it4;

lli ans[300009];
int main()
{
    int n,i,j,k,a,b;
    cin>>n;
    lli s=0;
    for(i=1;i<=n;i++){
        scanf("%d%d",&a,&b);
        if(a%2==1){
            s1.insert(b);
            s11.insert(b);
        }
        else {
            s2.insert(b);
            s22.insert(b);
        }
        s+=a;
    }
    memset(ans,0,sizeof(ans));
    ans[0]=0;

    lli temp1,temp2;
    //for even weights

    for(i=2;i<=s;i+=2){

        temp1=0;
        temp2=0;
        if(s1.size()>0){
            it1 = s1.rbegin();
            temp1=*it1;
            it1++;
        }
        if(s1.size()>1)temp1+=(*it1);


        if(s2.size()>0){
            it2=s2.rbegin();
            temp2=*it2;
        }

        if(temp1 > temp2){
            ans[i]=ans[i-2]+ temp1;
            if(s1.size()>0){
                it3=s1.end();
                it3--;
                s1.erase(it3);
            }

            if(s1.size()>0){
                it3=s1.end();
                it3--;
                s1.erase(it3);
            }
        }
        else {
            ans[i]=ans[i-2]+temp2;
            if(s2.size()>0){
                it4=s2.end();
                it4--;
                s2.erase(it4);
            }
        }
    }
    //for odd weights

     if(s11.size() > 0 ){
         ans[1]=*(s11.rbegin());
         it3=s11.end();
         it3--;
         s11.erase(it3);
     }

    for(i=3;i<=s;i+=2){
        temp1=0;
        temp2=0;
        if(s11.size()>0){
            it1 = s11.rbegin();
            temp1=*it1;
            it1++;
        }
        if(s11.size()>1)temp1+=(*it1);


        if(s22.size()>0){
            it2=s22.rbegin();
            temp2=*it2;
        }

        if(temp1 > temp2){
            ans[i]=ans[i-2]+ temp1;
            if(s11.size()>0){
                it3=s11.end();
                it3--;
                s11.erase(it3);
            }
            if(s11.size()>0){
                it3=s11.end();
                it3--;
                s11.erase(it3);
            }
        }
        else {
            ans[i]=ans[i-2]+temp2;
            if(s22.size()>0){
                it4=s22.end();
                it4--;
                s22.erase(it4);
            }
        }
     }

    for(i=1;i<=s;i++){
        cout<<ans[i]<<" ";
    }


    return 0;
}
