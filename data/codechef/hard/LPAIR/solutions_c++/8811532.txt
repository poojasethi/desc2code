#include<bits/stdc++.h>
using namespace std;
const int MAXN=2e5+7;
long long int bit[MAXN];
void update(int a){
	for(;a<2e5+3;a+=(a & -a)){
		bit[a]+=1;
	}
}
long long int query(int a){
	int ans=0;
	for(;a>0;a-=(a & -a)){
		ans+=bit[a];
	}
	return ans;
}
int main(){
	std::ios::sync_with_stdio(false);
	int n;
	cin>>n;
	map<int,int>m;
	vector<pair<int,int> > v;
	for(int i=0;i<n;i++){
		int x,y;
		cin>>x>>y;
		v.push_back(make_pair(x,y));	
		m[x]=1;
		m[y]=1;
	}
	int val=1;
	for(map<int,int>::iterator vi=m.begin();vi!=m.end();vi++){
		vi->second=val;
		val++;
	}
	for(int i=0;i<n;i++){
		v[i].first=m[v[i].first];
		v[i].second=m[v[i].second];
	}
	sort(v.begin(),v.end());
	long long int ans=0;
	for(int i=0;i<n;i++){
		long long int temp=query(2e5+2)-query(v[i].second);
		ans+=temp;
		update(v[i].second+1);
	}
	cout<<ans<<"\n";
}
