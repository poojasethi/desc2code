#include <bits/stdc++.h>
using namespace std;

#define ll long long int
#define mp(i,j) make_pair(i,j)

vector <pair<ll,ll> > m;
vector <pair<ll,ll> > f;
vector <pair<ll,ll> > p;

ll BIT[100010];

ll update(ll i,ll n)
{
	while(i<=n) {
		(BIT[i])++;
		i = i + (i&(-i));
	}
}

ll query(ll i,ll n)
{
	ll sum = 0;

	while(i>=1) {
		sum += BIT[i];
		i = i - (i&(-i));
	}
	return sum;
}
