#include <bits/stdc++.h>
#define MAX 200005
#define lli long long

using namespace std;

int tree[MAX];

void update(int idx)
{
    while ( idx <= MAX-4 ) {
        tree[idx] += 1;
        idx += (idx & (-idx));
    }
    return;
}

int query(int idx)
{
    int ans = 0;
    while ( idx > 0 ) {
        ans += tree[idx];
        idx -= (idx & (-idx));
    }
    return ans;
}

int main()
{
    int q,x,y,n,cnt=1;
    map <int,int> mp;
    map <int,int> :: iterator it;
    vector < pair<int,int> >  v;

    lli ans = 0;

    cin >> q;
    while ( q-- ) {
        cin >> x >> y;
        mp[x] = 1, mp[y] = 1;
        v.push_back(make_pair(x,y));
    }

    for ( it = mp.begin(); it != mp.end(); it++ ) it->second = cnt++;

    sort(v.begin(),v.end());

    n = (int)v.size();

    for ( int i = n-1; i >= 0; i-- ) {
        ans += (long long)query(mp[v[i].second]-1);
        update(mp[v[i].second]);
    }

    cout << ans << endl;

    return 0;
}
