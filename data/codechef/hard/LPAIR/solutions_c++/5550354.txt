#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
typedef long long LU;
typedef pair<LU,LU> pi;
LU cnt = 0;
void mergesort(pi  a[],long  p,long  q,long  r,long n)
{
     long int x[n],k=0,i=p,j=r+1;
     while(i<=r && j<=q)
     {
      if(a[i].second>a[j].second)
      {
      x[k++]=a[i++].second;
      cnt+=((q-j)+1);
      }
      else
      x[k++]=a[j++].second;
     };
     if(i<=r)
     {
      for(;i<=r;i++)
      x[k++]=a[i].second;
     }
     else
     {
      for(;j<=q;j++)
      x[k++]=a[j].second;
     }
     for(i=0;i<k;i++)
     a[i+p].second=x[i];
}
void ms(pi  a[],long p,long q,long  n)
{
     if(p<q)
     {
            int r=(p+q)/2;
            ms(a,p,r,n);
            ms(a,r+1,q,n);
            mergesort(a,p,q,r,n);
     }
}
int main()
{
 	int n;
	scanf("%d",&n);
	pi p[n];
	for(int i=0;i<n;i++)
	{
		scanf("%lld%lld",&p[i].first,&p[i].second);
	}	
	sort(p,p+n);
    cnt=0;
    ms(p,0,n-1,n);
    printf("%lld\n",cnt);
    
	return 0;
}
