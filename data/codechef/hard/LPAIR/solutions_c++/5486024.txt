void mergesort(LL *a, LL low, LL high,LL &cou)
{
    LL mid;
    LL lc=0,rc=0,sc=0;
    if (low < high)
    {
        mid=(low+high)/2;
        mergesort(a,low,mid,lc);
        mergesort(a,mid+1,high,rc);
        merge(a,low,high,mid,sc);
        cou=lc+rc+sc;
    }
    return;
}
void merge(LL *a, LL low, LL high, LL mid,LL &sc)
{
    LL i, j, k, c[mid+high+1];
    i = low;
    k = low;
    j = mid + 1;
    while (i <= mid && j <= high)
    {
        if (a[i] < a[j])
        {
            c[k] = a[i];
            k++;
            i++;
        }
        else
        {
            c[k] = a[j];
            sc+=(mid-i+1);
            k++;
            j++;
        }
    }
    while (i <= mid)
    {
        c[k] = a[i];
        k++;
        i++;
    }
    while (j <= high)
    {
        c[k] = a[j];
        k++;
        j++;
    }
    for (i = low; i < k; i++)
    {
        a[i] = c[i];
    }
}
