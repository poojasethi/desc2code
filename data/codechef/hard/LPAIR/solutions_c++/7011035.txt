#include <bits/stdc++.h>
#define lli long long int
#define s(x) scanf("%lld", &x)
#define MAXX 100005
#define trm(x) (x&(-x))
#define pb push_back
#define mp make_pair
#define fr first
#define sc second

using namespace std;

lli arr[MAXX], brr[MAXX], bit[MAXX], n;

vector<pair<lli, lli> > vv;

// Here j < i & arr[j] > arr[i]

// For Tle convert lli to int
lli query(lli x)
{
    lli ans = 0;

    while (x > 0) {
        ans += bit[x];
        x = x - trm(x);
    }

    return ans;
}

lli updt(lli x, lli val)
{
    while (x <= n) {
        bit[x] += val;
        x = x + trm(x);
    }
}

int main()
{
    //freopen("inp.txt", "r", stdin);

    lli tcase,cntr,i,j,k,m,ans;

    s(n);

    for (i = 0; i < n; ++i) {
        s(j);
        s(k);

        vv.pb(mp(j,k));

    //    bit[i] = 0;
    }

    //bit[i] = 0;

    sort(vv.begin(), vv.end());

    memset(brr, 0, sizeof(brr));

    for (i = 1; i <= n; ++i) {
        arr[i] = vv[i-1].sc;
        brr[i] = arr[i];
    }

    sort(brr+1, brr+n+1);

    for (i = 1; i <= n; ++i) {
        arr[i] = (lli)(lower_bound(brr+1, brr+n+1, arr[i]) - (brr));
        //  cout << arr[i] << " ";
    }
        //cout << endl;
    cntr = 0;

    for (i = n; i >= 1; --i) {
        cntr = cntr + query(arr[i]-1);

        updt(arr[i], 1);
    }

    printf("%lld\n", cntr);

    return 0;
}
