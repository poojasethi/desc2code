#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair<int,int> ii;
typedef long long ll;
#define pb push_back
const int maxn=100000+11;

int bit[maxn]={};

void insert(int key,int n){
	while(key<=n){
		bit[key]+=1;
		key+= (key & -key);
	}
}

int query(int key){
	int ans=0;
	while(key>0){
		ans+=bit[key];
		key-= (key & -key);
	}
	return ans;
}

bool fCmp(ii a,ii b){
	return a.first<b.first;
}


bool sCmp(ii a,ii b){
	return a.second<b.second;
}

int main(){
	int n;
	cin>>n;
	int f[maxn];
	vector<ii> l;
	for (int i = 0; i < n; ++i)
	{
		int a,b;
		cin>>a>>b;
		l.pb(ii(a,b));
	}
	sort(l.begin(),l.end(),sCmp);

	for (int i = 0; i < l.size(); ++i)
	{
		l[i].second=i+1;
	}

	sort(l.begin(),l.end(),fCmp);
	ll iter=0;
	for (int i = 0; i < l.size(); ++i)
	{
		int ls=query(l[i].second);
		iter+=i-ls;
		iter+=l[i].second-ls-1;
		insert(l[i].second,n);
	}
	cout<<iter/2<<"\n";
}
