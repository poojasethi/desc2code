#define pb push_back
#define mp make_pair
#define fi first
#define se second
#define INF 1000000000
#define LL long long int
#define EPS 1e-11
#define PI 3.14159265359

LL BIT[100003] , n , ans , bil[100003];
pair<LL , LL> mar[100003] , temp[100003];

void update(LL idx)
{
     while (idx<=100000)
     {
           BIT[idx] += 1;
           idx += (idx&(-idx));
     }
}

LL query(LL idx)
{
     LL ret = 0;
     while (idx > 0)
     {
           ret += BIT[idx];
           idx -= (idx&(-idx));
     }
     return ret;
}

int main(void)
{
    scanf("%lld",&n);
    for (int i =0 ; i<n; i++) scanf("%lld%lld",&mar[i].fi , &mar[i].se);
    
    sort(mar,mar+n);
    for (int i = 0 ; i<n; i++)
    {
        temp[i].fi = mar[i].se;
        temp[i].se = i + 1;
    }
    
    sort(temp , temp + n);
    for (int i = 0; i<n; i++) bil[temp[i].se] = i + 1;
    
    
    ans = 0 ;
    for (int i = 1; i<=n; i++)
    {
        ans += (i-1-query(bil[i]));
        update(bil[i]);
    }
    
    printf("%lld\n",ans);
    
    //system("pause");

}
