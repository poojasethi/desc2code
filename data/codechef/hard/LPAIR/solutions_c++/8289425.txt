#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define ull unsigned long long
#define pb push_back
#define ft first
#define se second
#define mp make_pair
int *fenwick;
int n;					 // Dont declare n again an error
void init(){
    fenwick=new int [n]; // Change to LL if needed
    for(int z = 0; z < n; fenwick[z]=0, ++z);
}
void update(int pos, int extra){
    while(pos<n){
        fenwick[pos]+=extra;
        pos|=(pos+1);
    }
}
int query(int pos){
    int ans=0;
    while(pos>=0){
        ans+=fenwick[pos];
        pos=(pos& (pos+1))-1;
    }
    return ans;
}
int main(int argc, char const *argv[]){
    scanf("%d", &n);
    map<pair<int, int>, int> pr;
    pair<int, int> mat[n];
    for(int i = 0; i < n; ++i){
    	scanf("%d%d", &mat[i].ft, &mat[i].se);
    }
    sort(mat, mat+n);
    for(int i = 0; i < n; ++i){
    	swap(mat[i].ft, mat[i].se);
    	pr.insert(mp(mat[i], i));
    }
    sort(mat, mat+n);
    ll ans=0;
    init();
    for(int j = 0; j < n; ++j){
    	ans+=(query(n-1)-query(pr[mat[j]])); 
    	update(pr[mat[j]], 1);
    }
    printf("%lld\n", ans);

	return 0;
}
