#include <bits/stdc++.h>
using namespace std;
long long int  findval(long long int  female,long long int  bit[],long long int  n){
	long long int  sum = 0;
	long long int  index = female+1;
	while(index<=n){
		sum += bit[index];
		index = index + (index & (-index));
		}
	return sum;
	}

void update(long long int  female,long long int  bit[],long long int  n){
	long long int  index = female+1;
	while(index>0){
		bit[index]++;
		index = index - (index & (-index));
		}
	}

int main(){
	long long int  n;
	cin>>n;
	long long int  a[n];
	long long int  b[n];
	map<long long int ,long long int > m;
	long long int  bit[n+1];
	for(long long int  i=0;i<n;i++){
		long long int  male,female;
		scanf("%lld %lld",&male,&female);
		a[i] = male;
		b[i] = female;
		m[male] = female;
		bit[i] = 0;
		}
	bit[n] =0;
	sort(a,a+n);
	sort(b,b+n);
	map<long long int ,long long int > findindex;
	for(long long int  i=0;i<n;i++){
		findindex[b[i]] = i;
		}
	long long int  ans=0;	
	for(long long int  i=0;i<n;i++){
		long long int  male = a[i];
		long long int  female = m[male];
		long long int  index = findindex[female];
		//cout<<index<<endl;
		ans += findval(index,bit,n);
		update(index,bit,n);
		/*
		for(long long int  i=0;i<=n;i++){
			cout<<bit[i]<<" ";
			}
		cout<<endl;*/
		}
	cout<<ans<<endl;
	}
