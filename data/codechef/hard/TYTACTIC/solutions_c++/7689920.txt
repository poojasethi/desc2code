void dfsorder(int src)
{
	sz_of_subtree[src] = 1;
	mapped[src] = ++cnt;
	
	fr(i,0,(int)G[src].size()-1)
	{
        int des = G[src][i];
        if (sz_of_subtree[des] == 0)
         {
            dfsorder(des);
            sz_of_subtree[src] += sz_of_subtree[des];
        }
    }
    
    //bug2(src,sz_of_subtree[src]);
}


void update(int ind, int val) 
{
    while (ind <= n) 
    {
        node[ind] += val;
        ind += ind & -ind;
    }
}


int query (int ind) 
{
    int sum = 0;
    while (ind)
     {
        sum += node[ind];
        ind -= ind & -ind;
    }
    return sum;
}
