#include<iostream>
#include<algorithm>
#include<vector>
typedef std::pair<int,int> p;
using namespace std;


int main()
{
    int T;
    cin >> T;
    while(T--)
    {
        int r,a,b,count=0,l=1000000000,N;
        vector<p> v;
        cin >> N;

        for(int i=0;i<N;i++){
            cin >> a >> b;
            v.push_back(p(b,a));
            if(a<l)l=a;
        }
        l--;
        sort(v.begin(),v.end());
        for(int i=0;i<N;i++){
            if(v[i].second>l){
                count++;
                l=v[i].first;
            }

        }
        cout << count << "\n";
    }
}
