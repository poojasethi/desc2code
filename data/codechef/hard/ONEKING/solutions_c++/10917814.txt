#include<cstdio>
#include<algorithm>
#define INF 1000000000
using namespace std;
typedef long long int ll;
typedef struct KINGDOM{
    int start,end;
}KINGDOM;
bool comp(const KINGDOM a,const KINGDOM b){
    if(a.end<b.end){
        return true;
    }
    return false;
}

int main(){
    int t;
    scanf("%d",&t);
    while(t--){
        int n;
        scanf("%d",&n);
        KINGDOM arr[n];
        bool cover[n];
        for(int i=0;i<n;i++){
            cover[i]=0;
            scanf("%d %d",&arr[i].start,&arr[i].end);
        }
        sort(arr,arr+n,comp);
        int j=0,ans=0;
        int start=-INF;
        for(int i=0;i<n;i++){
            if(cover[i]) continue;
            ans++;
            for(int j=i;j<n;j++){
                if(arr[j].start<=arr[i].end){
                    cover[j]=1;
                }
            }

        }
        printf("%d\n",ans);
    }
}
