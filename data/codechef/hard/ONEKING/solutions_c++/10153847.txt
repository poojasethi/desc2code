#include<bits/stdc++.h>
using namespace std;
bool func(const pair<long,long> &a,const pair<long,long> &b){
	if(a.second < b.second)
		return true;
	if(a.second == b.second)
		return a.first<b.first;
	return false;		
}
int main(){
	long t;
	long long n;
	scanf("%ld",&t);
	while(t--){
		scanf("%lld",&n);
		pair<long,long> p[n];
		for(int i=0;i<n;i++){
			scanf("%ld%ld",&p[i].first,&p[i].second);
		}
		long ans=0,j=0;
		sort(p,p+n,func);
		for(int i=0;i<n;i++)
		//printf("%ld %ld\n",p[i].first,p[i].second);
		for(int i=0;i<n;i++){
			if(j==n)
				break;
			long cur=p[i].second;
			ans++;
			for(j=i+1;j<n;j++){
				if(p[j].first>cur){
					//printf("b : %ld %ld %ld\n",p[j].first,cur,j);
					i=j-1;
					break;
				}
			}
		}
		printf("%ld\n",ans);
	}	
}
