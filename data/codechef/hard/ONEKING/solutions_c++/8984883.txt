#include<iostream>
#include<math.h>
#include<cstdio>
#include<algorithm>
#define gc getchar

using namespace std;
void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}
int main(){
    int t;
    scanint(t);
    while(t--){
        int n,i,j;
        scanint(n);
        int a,b,ans[2001];
        for(i=0;i<2001;i++){
            ans[i]=-1;
        }
        for(i=1;i<=n;i++){
            scanint(a);
            scanint(b);
            if(ans[b]<a){
                ans[b]=a;
            }
        }
        int max1=-1;
        int count1=0;
        for(i=0;i<2001;i++){
            if(max1<ans[i]){
                count1++;
                max1=i;
            }
        }

        printf("%d\n",count1);
    }
    return 0;
}
