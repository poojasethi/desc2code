#include <iostream>
#include <utility>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n,t1,t2;
        cin>>n;
        vector< pair<int,int> > v;
        for(int i=0;i<n;i++)
        {
            cin>>t1>>t2;
            v.push_back(make_pair(t1,t2));
        }
        sort(v.begin(),v.end());
        int count=1;
        int l,r;
        l=v[0].first;
        r=v[0].second;
        for(int i=1;i<n;i++)
        {
            if(v[i].first>r)
            {
                count++;
                l=v[i].first;
                r=v[i].second;
            }
            else
            {
                l=v[i].first;
                if(v[i].second<r)
                {
                    r=v[i].second;
                }
            }
        }
        cout<<count<<endl;
    }
    return 0;
}
