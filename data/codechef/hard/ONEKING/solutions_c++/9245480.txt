#define pii pair <int ,int >
typedef struct points {
	int l, r;
	int i;
}points;

bool cmp(const struct points &x, const struct points &y) {
	if (x.l == y.l) {
		return x.r < y.r;
	}
	return x.l < y.l;
}

int main() {
	int t;
	sc(t);
	while (t--) {
		int n;
		sc(n);

		struct points p[n + 3];
		for (int i = 0; i < n; i++)
		{
			int x, y;
			sc(x); sc(y);

			p[i].l = x;
			p[i].r = y;
			p[i].i = i;
		}

		sort(p, p + n, cmp);

		/*for (int i = 0; i < n; i++)
		{
			cout << p[i].l << " " << p[i].r << endl;
		}*/
		
		int ans=1;
		vector <pair <int ,int > > vi;
		vi.pb(pii(p[0].l,p[0].r));
		
		for(int i=1;i < n;i++) {
			//cout << vi[0].ff << " "<< vi[0].ss << endl;
			int k = vi[0].ss;
			if(p[i].l <= k) {
				vi.clear();
				vi.pb(pii(p[i].l,min(k,p[i].r)));
			}
			else {
				vi.clear();
				ans++;
				vi.pb(pii(p[i].l,p[i].r));
			}
		}
		
		pf(ans); pn;
	}
	return 0;
}
