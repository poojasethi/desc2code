#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		long int n,a,b,c[10000],count=-2,i,k,flag=0,e,f,q=1;
		for(i=0;i<10000;i++)
		c[i]=-2;
		scanf("%ld",&n);
		while(n--)
		{
			scanf("%ld%ld",&a,&b);
			if(c[b]<a)
			{
				c[b]=a;
			}
			
		}
		count=-1;
		for(i=0;i<10000;i++)
		{
			if(c[i]>count)
			{
				flag++;
				count=i;
			}
		}
		
		printf("%ld\n",flag);
	}
	return 0;
}
