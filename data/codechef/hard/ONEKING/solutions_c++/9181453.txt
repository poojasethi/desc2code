#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
vector<pair<ll,ll> > v;
bool cmp(pair<ll,ll> a,pair<ll,ll> b)
{
	if(a.first<b.first)
	return 1;
	else if(a.first==b.first)
	{
		if(a.second>=b.second)
		return 1;
		else
		return 0;
	}
	else
	return 0;
}
int main()
{
	ll t,n,i,a,b;
	cin>>t;
	while(t--)
	{
		v.clear();
		cin>>n;
		for(i=0;i<n;i++)
		{
			cin>>a>>b;
			v.push_back(make_pair(a,b));
		}
		ll count=0;
		sort(v.begin(),v.end());
		if(n==1)
		{
			cout<<"1"<<endl;
			continue;
		}
		ll last=v[0].second;
		count+=1;
		for(i=1;i<n;i++)
		{
			if(v[i].first>last)
			{
				count+=1;
				last=v[i].second;
			}
			else
			{
				last=min(last,v[i].second);
			}
		}
		cout<<count<<endl;
	}
}
