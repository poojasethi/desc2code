#include <iostream>
#include<set>
using namespace std;

int main() {
	set < pair < int, int > > iset;
	int t,test,a,b,n;
	long long int cnt;
	std::ios::sync_with_stdio(false);
	cin>>test;
	for(int t=0;t<test;t++)
	{
		cnt=0;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			cin>>a>>b;
			iset.insert(make_pair(b,a));
		}
		a=iset.begin()->second;
		b=iset.begin()->first;
//		cout<<a<<" "<<b<<" ";
		cnt=1;
		int q=iset.size();
//		cout<<q;
		iset.erase(make_pair(iset.begin()->first,iset.begin()->second));
		for(int i=1;i<q;i++)
		{
			if(iset.begin()->second>b)
			{
				cnt++;
				a=iset.begin()->second;
				b=iset.begin()->first;	
			}
				iset.erase(make_pair(iset.begin()->first,iset.begin()->second));
		}
		cout<<cnt<<endl;
		iset.clear();
	}
	// your code goes here
	return 0;
}
