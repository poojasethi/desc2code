#include<bits/stdc++.h>
using namespace std;

int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		vector < pair<int,int> > v;
		pair <int,int> tmp;
		int a,b,i,count=1,n,start,end;
		scanf("%d",&n);
		for(i=0;i<n;i++)
		{
			scanf("%d %d",&a,&b);
			tmp= make_pair(a,b);
			v.push_back(tmp);
		}
		sort(v.begin(),v.end());
		start= v[0].first;
		end= v[0].second;
		for(i=1;i<n;i++)
		{
			if(v[i].first>end)   //for completely outside 
			{
				start= v[i].first;
				end= v[i].second;
				count++;
			}
			else if(v[i].first>=start && v[i].second<=end)   // for completely inside
			{
				start= v[i].first;
				end= v[i].second;
			}
			else    //for partially inside and partially outside
			{
				start= v[i].first;
			}
		}
		printf("%d\n",count);
	}
	return 0;
}
