#include <iostream>
#include <climits>
#include <bits/stdc++.h>

typedef long long int ll;
using namespace std;  

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		vector<pair<int,int> >p;
		int m = INT_MAX;
		int n;
		cin>>n;
		for(int i=0;i<n;i++)
		{
			int a,b;
			cin>>a>>b;
			p.push_back(make_pair(b,a));
			if(a<m) m = a;
		}
		m--;
		sort(p.begin(),p.end());
		int cnt=0;
		for(int i=0;i<n;i++)
		{
			if(p[i].second>m)
			{
				cnt++;
				m = p[i].first;
			}
		}
		cout<<cnt<<endl;
	}
	return 0;
}
