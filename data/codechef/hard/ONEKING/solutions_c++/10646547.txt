#include <iostream>
#include <cstdlib>
#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <map>

using namespace std;
const int MAXN = 2003;
int T, N;
bool b[MAXN];
int maxa[MAXN];

int main()
{
	scanf("%d", &T);
	while (T--)
	{
		for (int i = 0; i < MAXN; i++)
		{
			b[i] = maxa[i] = 0;
		}
		int sol = 0;
		scanf("%d", &N);
		for (int i = 0; i < N; i++)
		{
			int aa, bb;
			scanf("%d%d", &aa, &bb);
			b[bb] = 1;
			if (maxa[bb] < aa) maxa[bb] = aa;
		}
		int lastb = -1;
		for (int i=0; i<=2000; i++)
			if (b[i])
			{
				if (maxa[i] > lastb)
				{
					lastb = i;
					sol++;
					//cout << lastb;
				}
			}
		printf("%d\n", sol);
	}
	return 0;
}
