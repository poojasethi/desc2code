int main()

	while(t--)
	{
		int n;

		vector < pair <int,int > > v;
		int a,b;
		for(int i=0;i<n;i++)
		{
			scanf("%d%d",&a,&b);
			v.push_back( make_pair (b,a));
		}

		int c=1;

		}
