#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int test,n,i,a,b,ans;
	vector< pair<int,int> >v;
	scanf("%d",&test);
	while(test){
		test--;
		ans=0;
		scanf("%d",&n);
		for(i=0;i<n;i++){
			scanf("%d%d",&a,&b);
			v.push_back(make_pair(b,a));
		}
		sort(v.begin(),v.end());
		a=v[0].first;
		ans++;
		for(i=1;i<n;i++){
			if(v[i].second>a){
				ans++;
				a=v[i].first;
			}
		}
		printf("%d\n",ans);
		v.clear();
	}
	return 0;
}
