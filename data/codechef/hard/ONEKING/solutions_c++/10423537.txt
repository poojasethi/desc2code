#include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

int solve(vector<pair<int, int> >& v)
{
	sort(v.begin(), v.end());
	vector<pair<int, int> > res;
	res.push_back(v[0]);
	for(int i = 1; i < v.size(); i++)
	{
		pair<int, int> p = res[res.size() - 1];
		res.pop_back();
		//cout << p.second << " " << p.first << endl;
		if(v[i].second <= p.first)
		{
			p.second = max(p.second, v[i].second);
			res.push_back(p);
		}
		else
		{
			res.push_back(p);
			res.push_back(v[i]);
		}
	}
	return res.size();
}

int main()
{
	int T;
	scanf("%d",&T);
	while(T--)
	{
		int N;
		scanf("%d",&N);
		vector<pair<int, int> > v(N);
		for(int i = 0; i < N; i++)
			scanf("%d%d",&v[i].second, &v[i].first);
		printf("%d\n",solve(v));
	}
	return 0;
}
