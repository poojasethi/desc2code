#include<bits/stdc++.h>
using namespace std;
bool func(pair<int, int> a, pair<int, int> b){
	if(a.first==b.first)return (a.second<b.second);
	return (a.first<b.first);
}
#define pb push_back
int main(){
	int t;
	cin>>t;
	vector<int> v, u;
	pair<int, int> p[100001];
	while(t--){
		int n, k, i, j, a, b, c=0;
		cin>>n;
		v.clear(), u.clear();
		for(i=0; i<n; i++){
			cin>>a>>b;
			p[i].first=a;
			p[i].second=b;
		}
		sort(p, p+n, func);
		for(i=0; i<n; i++){
			v.pb(p[i].second);
			u.pb(p[i].first);
		//	cout<<u[i]<<" "<<v[i]<<endl;
		}
		i=n-1;
		int val=u[i];
	//	cout<<upper_bound(u.begin(), u.end(), 6)-u.begin();
	//	cout<<upper_bound(u.begin(), u.end(), 5)-u.begin();
	//	cout<<upper_bound(u.begin(), u.end(), 1)-u.begin();
		while(i>=0){
			while(i>=0&&v[i]>=val){
				i--;
			}
			if(i!=-1)
			val=u[i];
			c++;
		}
		cout<<c<<endl;
	}
}
