#include <iostream>
#include <utility>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
    int tc;
    cin>>tc;
    while(tc--)
    {
        int n,t1,t2,l=100000000;
        cin>>n;
        vector< pair<int,int> > v;
        for(int i=0;i<n;i++)
        {
            cin>>t1>>t2;
            v.push_back(make_pair(t2,t1));
            if(l>t1)
            {
                l=t1;
            }
        }
        l--;
        sort(v.begin(),v.end());
        int count=0;
        for(int i=0;i<n;i++)
        {
            if(v[i].second>l)
            {
                count++;
                l=v[i].first;
            }
        }
        cout<<count<<endl;
    }
    return 0;
}
