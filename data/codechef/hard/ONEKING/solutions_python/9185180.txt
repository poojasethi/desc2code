for i in range(int(raw_input())):
    n = int(raw_input())
    llist = []
    m = 2000
    for i in range(n):
        a,b = map(int,raw_input().split())
        llist.append((b,a))
        m = min(m,a)
    llist.sort()
    m = m-1
    c = 0
    for i in llist:
        if i[1]>m:
            m = i[0]
            c += 1
    print c
