T = input()
for _ in xrange(0, T):

    kingdoms = []

    N = input()
    for _ in xrange(0, N):

        L, R = map(int, raw_input().split())
        kingdoms.append((R, L))

    kingdoms = sorted(kingdoms)

    number_of_bombs_required = 0
    bomb_position = -1

    for r, l in kingdoms:
        if l > bomb_position:
            number_of_bombs_required += 1
            bomb_position = r

    print(number_of_bombs_required)
