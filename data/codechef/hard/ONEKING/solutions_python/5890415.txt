t = int(raw_input())
for case in range(t):
    n = int(raw_input())
    arr = [map(int, raw_input().split()) for x in range(n)]
    arr.sort(key=lambda x:x[1])
    r = arr[0][1]
    ans = 1
    for i in range(1, n):
        if arr[i][0] > r:
            ans += 1
            r = arr[i][1]
        else:
            r = min(r, arr[i][1])
    print ans
