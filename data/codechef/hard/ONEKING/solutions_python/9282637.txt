import fileinput

foundT = False
for line in fileinput.input():
    if not foundT:
        foundT = True
        T = int(line)
        first_test_line = True
        continue
    
    if first_test_line:
        first_test_line = False

        test_lines_left = N
        start_map = [-1] * 2001
        continue

    start_map[bi] = max(ai, start_map[bi])

    if test_lines_left == 0:
        first_test_line = True
        
        last_bomb = -1
        bomb_count = 0
        for i in range(2001):
