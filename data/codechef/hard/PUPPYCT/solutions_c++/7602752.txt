//#include<stdio.h>

//#include<string.h>
#include<bits/stdc++.h>
using namespace std;
#define sd(x) scanf("%d",&x)
#define slld(x) scanf("%lld",&x)
#define ss(x) scanf("%s",x)
int a[2000005],b[2000005],c[2000005];
int main()
{
    int t,n,i,j,k,x,y;
    long long int s;
    sd(t);
    while(t--)
    {
        for(i=0;i<=2000003;i++)
        {
            a[i]=0;
            b[i]=0;
        }
        sd(n);
        sd(k);
        s=0;
        while(k--)
        {
            sd(x);
            sd(y);
            a[x-y+n]++;
            b[x+y-1]++;
        }
        for(i=1;i<=2*n-1;i++)
        {
            if(a[i]>0)
               s+=n-abs(i-n);
            if(b[i]>0)
                s+=n-abs(i-n);
        }
        //printf("%lld\n",s);
        c[1]=min(1,a[n]);
        c[2]=min(1,a[n-1])+min(1,a[n+1]);
        for(i=3;i<=n;i++)
            c[i]=c[i-2]+min(1,a[n+i-1])+min(1,a[n-i+1]);
        for(i=1;i<=2*n-1;i++)
        {
            if(b[i]>0)
            {
                if(i<=n)
                    s-=c[i];
                else
                    s-=c[2*n-i];
            }
        }
        printf("%lld\n",1ll*n*n-s);
    }
    return 0;
}
