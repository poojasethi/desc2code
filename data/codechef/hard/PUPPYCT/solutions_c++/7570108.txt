//vector< pair<ll,ll> > v[2];

ll sum[2][2*MX];
bool vis[2*MX];

pair<ll,ll>p[MX];

int main()
{
    ll n , k , ts , x , y;
    scanf("%lld",&ts);
    while(ts--){ //rep(i,0,1)v[i].clear();
        memset(vis , 0 , sizeof vis);
        memset(sum, 0 , sizeof sum);

        scanf("%lld%lld",&n, &k);
        ll cat = 0 ;
        rep(i,1,k){
            scanf("%lld%lld",&x,&y);
            p[i].xx  = x , p[i].yy =  y;

            if(!vis[x-y+n]) cat += n - abs(x-y);
            else continue;
                vis[x-y+n] = 1;

                int drow , uprow;
                int dcol , upcol;

                if(x>=y) {
                    dcol = 1;
                    drow = 1+abs(x-y);

                    uprow = n;
                    upcol = n-abs(x-y);
                }
                else {
//                    dcol = (y-x)+1
                    drow = 1;
                    dcol = 1 + abs(y-x);

                    upcol = n;
                    uprow = n - abs(y-x);
                }


            if( (x-y)%2==0 ){
                sum[0][drow+dcol]++;
                sum[0][uprow+upcol+1]--;
            }
            else{
//                cout<<"s "<<drow+dcol<<" ";
                sum[1][drow+dcol]++;
//               cout<<"s "<<uprow+upcol<<" ";
                sum[1][uprow+upcol+1]--;
            }
        }

        rep(i,1,2*n){
            sum[1][i] += sum[1][i-1];
            sum[0][i] += sum[0][i-1];
        }

//        cout<<"c "<<cat<<endl;

        memset(vis , 0 , sizeof vis);
        rep(i,1,k){   ll x  = p[i].xx , y = p[i].yy;

            if(vis[x+y]) continue;
               vis[x+y] = 1;

            int tot = (x + y-1);
            if(tot>n){
                tot = 2*n - tot;
            }

            if((x-y)%2==0) cat += tot - sum[0][x+y];
            else           cat += tot - sum[1][x+y];
        }

        printf("%lld\n",(n*n-cat));
    }

    return 0;
}
