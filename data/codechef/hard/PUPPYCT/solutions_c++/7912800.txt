#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define pii pair<int,int>
#define F first
#define S second
#define mp make_pair
#define pb push_back
#define read(x) scanf("%d",&(x))
#define readll(x) scanf("%lld",&(x))
 
int x[100000],y[100000];
int lower[1000006],upper[1000006],diag[2000006];
int main()
{
    int t;
    int n,k,i,j,odd,even,cur;
    ll ans;
    read(t);
    while(t--)
    {
        ans=0;
        scanf("%d%d",&n,&k);
        for(i=0;i<k;i++)
            scanf("%d%d",x+i,y+i);
        for(i=j=n+1;i>1;i--,j++)
            diag[i]=diag[j]=i-1;
        for(i=0;i<n;i++)
            lower[i]=upper[i]=n-i;
 
        for(i=0;i<k;i++)
        {
            if(diag[x[i]+y[i]])
            {
                ans+=diag[x[i]+y[i]];
                diag[x[i]+y[i]]=0;
            }
        }
        odd=even=0;
        cur=1;
        for(i=n-1;i>=0;i--)
        {
            if(cur)
            {
                odd+=(diag[i+2]?0:1)+((i==n-1)?0:(diag[n+n-i]?0:1));
                lower[i]-=odd;
            }
            else
            {
                even+=(diag[i+2]?0:1)+(diag[n+n-i]?0:1);
                lower[i]-=even;
            }
            upper[i]=lower[i];
            cur^=1;
        }
 
        for(i=0;i<k;i++)
        {
            if(x[i]>=y[i])
            {
                if(lower[x[i]-y[i]])
                {
                    ans+=lower[x[i]-y[i]];
                    lower[x[i]-y[i]]=0;
                }
            }
            else if(upper[y[i]-x[i]])
                {
                    ans+=upper[y[i]-x[i]];
                    upper[y[i]-x[i]]=0;
                }
        }
        printf("%lld\n",n*1ll*n-ans);
    }
    return 0;
}
