#include <stdio.h>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
int d[2000000]={0};
int u[2000000]={0};
int checke[2000000]={0};
int checko[2000000]={0};
int temp[2000000]={0};
int main() {
	// your code goes here
	int t;
	cin>>t;
	while(t--){
	long long n,k;
	cin>>n>>k;
	if(k<=1){
		long long ans=n*n;
		//cout<<n<<k<<ans;
		int t1,t2;
		while(k--){
		scanf("%d%d",&t1,&t2);
		int temp1=t1+1,temp2=t2+1;
		while(temp1<=n && temp2<=n) {ans--; temp1++;temp2++;}
		temp1=t1+1;temp2=t2-1;
		while(temp1<=n && temp2>0) {ans--; temp1++;temp2--;}
		temp1=t1-1;temp2=t2+1;
		while(temp1>0 && temp2<=n) {ans--; temp1--;temp2++;}
		temp1=t1-1;temp2=t2-1;
		while(temp1>0 && temp2>0) {ans--; temp1--;temp2--;}
		ans--;
		}
		cout<<ans<<endl;
		
	}else{
	memset(d,0,8*n);
	memset(u,0,8*n);
	memset(checke,0,8*n);
	memset(checko,0,8*n);
	long long ans=0;
	for(int i=0;i<k;i++){
		int t1,t2;
		scanf("%d%d",&t1,&t2);
		if(d[t1-t2+n]==0)
		ans+=n-(max(t1,t2)-min(t1,t2));
		d[t1-t2+n]=1;
		u[t1+t2]=1;
	}
	//cout<<ans;
	//ans=0;
	memset(temp,0,8*n);
	//for(int i=0;i<=2*n;i++) {cout<<i<<" "<<d[i]<<endl;}
	for(int i=0;i<=2*n;i++) temp[abs(n-i)]+=d[i];
	for(int i=0;i<=n;i++) {d[i]=temp[i];}
	//for(int i=0;i<=2*n;i++) {cout<<i<<" "<<d[i]<<endl;}
	checke[0]=d[0];
	checko[1]=d[1];
	for(long long i=2;i<=n;i+=2) {checke[i]=checke[i-2]+d[i];}
	for(long long i=3;i<=n;i+=2) {checko[i]=checko[i-2]+d[i];}
	//for(int i=0;i<n;i++) cout<<checko[i]<<endl;
	//cout<<ans<<endl;
	for(long long i=2;i<=2*n;i+=2){ 
		ans+=(n-max(n+1,i)+min(n+1,i)-checke[(n-max(n+1,i)+min(n+1,i))-1])*u[i];
	//	cout<<ans<<endl;
	}
	//cout<<ans<<endl;
	
	for(long long i=3;i<=2*n;i+=2){
		ans+=(n-max(n+1,i)+min(n+1,i)-checko[(n-max(n+1,i)+min(n+1,i))-1])*u[i];
	//	cout<<ans<<endl;
	}
	
	ans=n*n-ans;
	cout<<ans<<endl;
	//for(int i=0;i<=n;i++) cout<<d[i]<<endl;
	}}
	//system("pause");
	return 0;
}
