#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

long long cnt[2000001];
bool tell1[1000000],tell2[1000000];

struct node{
    long long x,y;
};

int main(){
    long long t;
    scanf("%lld",&t);
    while(t--){
        long long n,k,ans;
        scanf("%lld%lld",&n,&k);
        node arr[k];
        for(long long i=0;i<=2*n;i++)
            cnt[i]=0;
        for(long long i=0;i<n;i++)
            tell1[i]=tell2[i]=false;
        ans=n*n;
        for(long long i=0;i<k;i++){
            scanf("%lld%lld",&arr[i].x,&arr[i].y);
            if(cnt[arr[i].x+arr[i].y]==0){
                cnt[arr[i].x+arr[i].y]=1;
                if(arr[i].x+arr[i].y<=n+1)
                    ans-=arr[i].x+arr[i].y-1;
                else
                    ans-=(2*n-arr[i].x-arr[i].y+1);
            }
        }
        for(long long i=2;i<=2*n;i++)
            cnt[i]+=cnt[i-2];
        for(long long i=0;i<k;i++){
            if(arr[i].x==arr[i].y){
                if(!tell1[0]){
                    tell1[0]=tell2[0]=true;
                    ans-=(n-cnt[2*n]);
                }
            }else if(arr[i].x<arr[i].y){
                if(!tell2[arr[i].y-arr[i].x]){
                    tell2[arr[i].y-arr[i].x]=true;
                    ans-=(n+arr[i].x-arr[i].y-cnt[2*n+arr[i].x-arr[i].y]+cnt[arr[i].y-arr[i].x]);
                }
            }else{
                if(!tell1[arr[i].x-arr[i].y]){
                    tell1[arr[i].x-arr[i].y]=true;
                    ans-=(n-arr[i].x+arr[i].y-cnt[2*n-arr[i].x+arr[i].y]+cnt[arr[i].x-arr[i].y]);
                }
            }
        }
        printf("%lld\n",ans);
    }
}
