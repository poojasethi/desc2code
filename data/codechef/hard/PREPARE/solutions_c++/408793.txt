#define INF 1000000000
using namespace std;

typedef pair<int,int> ii;
vector<ii> v;
bool vis[1005];
int n;
bool possible(int T)
{
        memset(vis,false,sizeof(vis));
        vis[0]=true;
        int t = 0 ;
        for(int i=0;i<n;i++)
        {
                if( v[i].first <= T )
                        continue;
                else
                        t+=v[i].second;
                for(int j=T-v[i].second;j>=0;j--)
                        vis[j+v[i].second] = vis[j+v[i].second] | vis[j];
        }
        for(int i=T;t-i<=T;i--)
        {
                if( vis[i] )
                        return true;
        }
        return false;
}
int main()
{
        int tc,low,mid,high;
        scanf("%d",&tc);
        while(tc--)
        {
                scanf("%d",&n);
                ii P;
                for(int i=0;i<n;i++)
                {
                        scanf("%d%d",&P.first,&P.second);
                        v.push_back(P);
                }
                low=0,high=1000;
                while(high > low+1 )
                {
                        mid = (low+high)/2;
                        if(possible(mid) == true )
                                high = mid;
                        else
                                low = mid ;
                }
                printf("%d\n",high);
                v.clear();
        }
        return 0;
}
