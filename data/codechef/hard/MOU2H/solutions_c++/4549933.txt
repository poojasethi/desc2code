#include<bits/stdc++.h>
using namespace std;
typedef long long int LL;
#define MAX 1000000
#define ext 4000000
#define MOD 1000000009

int hi[MAX+10], dif[MAX+10];
LL dp[MAX+10];
int last[8000010];

int main()
{
    int i, j, k, t, n;

    scanf("%d", &t);
    while(t--)
    {
        scanf("%d", &n);
        for(i = 1; i <= n; i++)
            scanf("%d", &hi[i]), dif[i-1] = hi[i]-hi[i-1];

        n--;
        dp[0] = 1;
        for(i = 1; i <= n; i++)
            dp[i] = ((2*dp[i-1])- last[dif[i]+ext]+MOD)%MOD, last[dif[i]+ext] = dp[i-1];

        for(i=1; i <= n; i++) last[dif[i]+ext] = 0;
        printf("%lld\n", dp[n]-1);
    }
    return 0;
}
