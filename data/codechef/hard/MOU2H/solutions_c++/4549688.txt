#include <iostream>
#include <string.h>
#include <stdio.h>
using namespace std;
#define MOD 1000000009
#define MAX 1000005
long long int A[MAX], S[MAX];
const int maxY = 4000000;
long long int pr[2 * maxY + 10];

int main()
{
	int t;
	scanf("%d", &t);
	memset(pr, -1, sizeof(pr));
	while(t--) {
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; ++i) {
			scanf("%lld", &A[i]);
			if (i)
				A[i - 1] = A[i] - A[i - 1] + maxY;
		}
		if (n == 1) {
			printf("1\n");
			continue;
		}
		for (int i = 0; i + 1 < n; ++i) {
			S[i] = (i > 0 ? S[i - 1] : 0) - (pr[A[i]] > 0 ? S[pr[A[i]] - 1] : 0) + (pr[A[i]] == -1);
			S[i] = ((S[i] + S[i - 1]) % MOD + MOD) % MOD;
			pr[A[i]] = i;
		}
		for (int i = 0; i + 1 < n; ++i)
			pr[A[i]] = -1;
		printf("%lld\n", S[n - 2]);
	}
	return 0;
}
