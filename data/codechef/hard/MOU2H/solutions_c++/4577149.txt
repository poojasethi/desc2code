#define BASE 4000000
#define MOD 1000000009

using namespace std;

int last[8000005];
long long int dp[1000007];
int a[1000007];
long long int sum[1000007];

void input() {

int main()
{
	input();
	int t;
	scanf("%d", &t);
	while (t--) {
		int n;
		scanf("%d", &n);
		for (int i = 0; i < n; ++i) scanf("%d", &a[i]);
		for (int i = 1; i < n; ++i)	last[BASE + a[i] - a[i - 1]] = 0;
		dp[0] = 1;
		sum[0] = 1;
		int b;
		for (int i = 1; i < n; ++i)
		{
			int b = a[i] - a[i - 1];
			dp[i] = sum[i - 1];
			if (last[b + BASE] != 0) {
				sum[i] = dp[i] + sum[i - 1] - sum[last[b + BASE] - 1];
				last[b + BASE] = i;
			} else {
				sum [i] = dp[i] + sum[i - 1];
				last[b + BASE] = i;
			}
			sum[i] = (sum[i] + MOD) % MOD;
		}
		sum[n - 1]--;
		if (sum[n-1] < 0) sum[n-1] += MOD;
		printf("%lld\n", sum[n - 1]);
	}
	return 0;
}
