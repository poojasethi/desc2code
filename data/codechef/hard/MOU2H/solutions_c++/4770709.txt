#include <algorithm>
#include <string.h>
#include <stdio.h>
#define  maxn      1000001
#define  maxd      4000000
#define  mod       1000000009
using    namespace std;

int T,n,ans;
int ar[maxn];
int sum[2*maxd+1];

int main()
{
  scanf("%d",&T);
  while(T--)
  {
    for(int i=1 ; i<n ; i++) sum[ar[i]]=0;
    scanf("%d",&n);
    for(int i=1 ; i<=n ; i++)
      scanf("%d",&ar[i]);
    for(int i=1 ; i<n ; i++)
      ar[i]=ar[i+1]-ar[i]+maxd;
    int tot=1;
    for(int i=1 ; i<n ; i++)
    {
      int tmp=(tot-sum[ar[i]]+mod)%mod;
      tot=(tot+tmp)%mod;
      sum[ar[i]]=(sum[ar[i]]+tmp)%mod;
    }
    printf("%d\n",(tot+mod-1)%mod);
  }
  return 0;
}
