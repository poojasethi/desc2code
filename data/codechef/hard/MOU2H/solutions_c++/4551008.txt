const int INF=(1<<29);
const int MAX=1000099;

ll dp[MAX],A[MAX],last[8*MAX];
int add=4000000;
ll mod = 1000000009;

int main()
{
    onlyCPP;
    int t,n,x,pre;
    cin>>t;
    while(t--)
    {
        cin>>n;
        cin>>pre;
        for(int i=1;i<n;i++)
        {
            cin>>x;
            A[i]=(x-pre)+add;
            pre=x;
            last[A[i]]=-1;
        }

        dp[0]=1;
        for(int i=1;i<n;i++)
        {
            dp[i]=(dp[i-1]*2)%mod;
            int k=last[A[i]];
            last[A[i]]=i;

            if(k==-1) continue;
            dp[i]=(dp[i]-dp[k-1]+mod)%mod;
        }
        cout<<(dp[n-1]-1+mod)%mod<<endl;
    }
    return 0;
}
