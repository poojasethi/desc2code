long long dp[1000002]={0};
int last[10000000]={0};
int diff[1000002]={0};

int main()
{
#ifdef DEBUG
    cout<<"Debugging\n";
#endif
ios::sync_with_stdio(false);
FILE *fp,*fo;
fp=fopen("input.txt","r");
fo=fopen("output.txt","w");

int t,i,n,x,prev;
long long ans;
cin>>t;
while(t--)
{
    ans=0;
    //memset(last,-1,sizeof(last));
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>x;
        if(i!=0)
            diff[i]=x-prev;
        prev=x;
    }
    n--;
    for(i=1;i<=n;i++)
    {
    	last[diff[i]+4000000]=-1;
    }
    dp[0]=1LL;
    for(i=1;i<=n;i++)
    {
        x=diff[i];
        dp[i]=(2LL*dp[i-1]);
        dp[i]%=mod;
        if(last[x+4000000]!=-1)
        {
            dp[i]-=dp[last[x+4000000]-1];
            if(dp[i]<0)
                dp[i]+=mod;
        }
        last[x+4000000]=i;
    }
    ans=dp[n]-1;
    if(ans<0)
        ans+=mod;
    cout<<ans<<"\n";
}



return 0;
}
