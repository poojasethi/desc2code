#define pii pair<int,int>
#define pb push_back
#define mp make_pair

#define y second
#define all(x) x.begin(),x.end()
#define ll long long
#define inf 2100000007
#define eps 1e-7
#define pi acos(-1)
#define N 2020
int a[N],sl[N],sr[N],dp[2][N],pre,now,r[N],l[N],m;
void cal()
{
    now^=1,pre^=1;
    int i;sr[m+1]=0;
    for(i=1;i<=m;i++)sl[i]=max(0,sl[i-1]+a[i]);
    for(i=m;i>=1;i--)sr[i]=max(0,sr[i+1]+a[i]);
    l[0]=-inf;r[m+1]=-inf;
    for(i=1;i<=m;i++)l[i]=a[i]+max(l[i-1],sl[i-1]+dp[pre][i]);
    for(i=m;i>=1;i--)r[i]=a[i]+max(r[i+1],sr[i+1]+dp[pre][i]);
    for(i=1;i<=m;i++)dp[now][i]=max(l[i]+sr[i+1],r[i]+sl[i-1]);
}
int main()
{
    //freopen("in.txt","r",stdin);
    int i,j,k,n,T;
    scanf("%d",&T);
    while(T--)
    {
        memset(dp[1],0,sizeof(dp[1]));
        scanf("%d%d%d",&n,&m,&k);
        int x=0,p,q,M;
        if(k==2)scanf("%d%d%d%d",&x,&p,&q,&M);
        int cur=x;
        now=1,pre=0;
        for(i=1;i<=n;i++)
        {
            if(k==1)
            for(j=1;j<=m;j++)scanf("%d",&a[j]);
            else
            {
                for(j=1;j<=m;j++)
                {
                    cur=(cur*p+q)%M;a[j]=x-cur;
                }
            }
            cal();
        }
        int ans=-inf;
        for(i=1;i<=m;i++)ans=max(ans,dp[now][i]);
        printf("%d\n",ans);
    }
}
