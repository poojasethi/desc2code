def ncr(x,y):
    if(x==y):
        return 1
    elif(y==1):
        return x
    res=1
    if(y>x/2):
        y=x-y
    for i in range(1,y+1):
        res=res*(x-y+i)
        res=res/i
    return res
t=input()
while(t!=0):
    t=t-1
    a,b=map(int,raw_input().split())
    a=a-1;
    b=b-1;
    ans=ncr(a,b)
    print ans
