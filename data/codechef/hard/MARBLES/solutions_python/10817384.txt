from math import factorial

def solution():
    T = int(raw_input().strip())
    for i in xrange(T):
        n, k = map(int, raw_input().strip().split(' '))
        if n == 0:
            print "%d"%(1)
            continue
        else:
            print "%d"%n_1cn_k(n-1, n-k)

def n_1cn_k(n, k):
    val = 1
    if k > n//2:
        k = n - k
    for i in xrange(1, k+1):
        val *= n
        val /= i
        n -= 1
    return val

solution()
