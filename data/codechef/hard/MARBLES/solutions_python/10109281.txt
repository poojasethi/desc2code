k = int(raw_input())

def fct(n,k):
    ct=Fraction(1,1)
    if n-k>k:
        while k>0:
           ct*=Fraction(n,k)
           k-=1
           n-=1
        return ct
    else:
        while n-k>0:
           ct*=Fraction(n,n-k)
           #k+=1
           n-=1
        return ct
