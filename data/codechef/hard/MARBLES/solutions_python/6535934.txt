def comb(m,n):
    if m<n:
        return 0
    if m==n:
        return 1
    m-=1
    n-=1
    n=min(n,m-n)
    n1=n
    num=1
    den=1
    for i in range(n1):
        num=num*m
        den=den*n
        m-=1
        n-=1
    total=num/den
    return total

l=[]
t=int(raw_input())
for i in range(t):
    n, k = map(int, raw_input().split())
    l.append(comb(n,k))


for j in l:
    print j
