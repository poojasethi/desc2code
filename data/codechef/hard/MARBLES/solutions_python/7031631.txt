def nCr(a,b):
    mem1=1
    mem3=1
    if(a-b<=b):
        for i in range(b,a):
            mem1=mem1*i;
        
        for i in range(2, a-b+1):
            mem3=mem3*i
        return mem1/mem3
    else:
        for i in range(a-b+1,a):
            mem1=mem1*i;
        
        for i in range(2,b):
            mem3=mem3*i
        return mem1/mem3
        

t=input()
while(t):
    n,k=map(int,raw_input().split())
    a=nCr(n,k)
    print a
    t=t-1
