import operator

def combi(n, k):
    big, small = k, n-k
    if small>big: big, small = small, big

    numerator = reduce(operator.mul, xrange(big+1, n+1), 1)
    denominator = reduce(operator.mul, xrange(2, small+1), 1)
    return numerator/denominator
