def com(n,r):
    if r>n:
        return 0
    else:
        if r>(n/2):
            return com(n,n-r)
        else:
            prod=1
            for i in range(0,r,1):
                 prod=prod*(n-i)
                 prod=prod/(i+1)
            return (prod)
