from math import factorial

def combination(n,r):
    upperterm = 1
    lowerterm = factorial(min(r,n-r))
    upperiter = max(r, n-r) + 1
    while (upperiter <= n):
           upperterm = upperterm * upperiter
           upperiter = upperiter + 1  
    return (upperterm/lowerterm)


def main():
    T = int(raw_input())
    for i in range(T):
        n, k = map(int, raw_input().split())
        print combination(n-1, k-1)

main()
