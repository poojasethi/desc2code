def ncr(n,r):
    if n == r:
        return 1
    elif r == 1:
        return n
    res = 1
    if r > n/2:
        r = n-r
    for i in range(1,r+1):
        res = res* (n - r + i)
        res = res / i
    return res

t = input()
    
for i in range(t):
    n,k = map(int,raw_input().split())
    print ncr(n-1,k-1)
