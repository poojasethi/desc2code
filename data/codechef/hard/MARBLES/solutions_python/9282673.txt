import sys

def read(inFile):
    t = int(inFile.readline())
    for _ in xrange(t):
        yield [int(x) for x in inFile.readline().split()]

def write(outFile, answer):
    outFile.write(str(answer)+'\n')

def solve(n, k):
    toPlay = n - k
    result = 1
    for x in xrange(1, k):
        result = result*(toPlay + x)/x
    return result

for case in read(sys.stdin):
    write(sys.stdout, solve(*case))
