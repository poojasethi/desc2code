import sys
t=input()

def fact(n,k):
	if (n//2)<k:
		k=n-k
	up=1
	t=n
	while t!=(n-k):
		up *=t
		t -=1
	down=1
	for each in range(1,k+1):
		down=down*each
	#print up
	#print down
	return up//down
