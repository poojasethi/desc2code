

def ncr(n,k):
    ans=1
    if k>n-k:
        k=n-k
    j=1
    while j<=k:
        if n%j==0:
            ans=ans*(n/j)
        elif ans%j==0:
            ans=ans/j*n
        else:
            ans=(ans*n)/j

        j=j+1
        n=n-1
    return ans

T=int(raw_input())
for t in range(T):
    a=raw_input().split()
    n=int(a[0])
    k=int(a[1])
    ans=ncr(n-1,k-1)
    print ans
