import sys

def solve(a,b) :
    if a//2 < b :
        b = a - b
    answer = 1
    back , front = a , 1
    for i in range(b) :
        answer = (answer * back)
        back -= 1
        answer = answer/(front)
        front += 1
    return answer

for __ in range(input()) :
    n , m = map(int,sys.stdin.readline().split())
    print solve(n-1,m-1)
