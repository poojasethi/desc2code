t = input()
for i in range(0, t):
	n,m = raw_input().split()
	n = int(n)
	m = int(m)
	mu = 1
	mu1 = 1
	if 2*m < n:
		for j in range(n-m+1,n):
			mu *= j
		for j in range(1,m):
			mu1 *= j
		print(mu/mu1)
	else:
		for j in range(m,n):
			mu *= j
		for j in range(1,n-m+1):
			mu1 *= j
		print(mu/mu1)
