def fact(n) :
    if n == 0 or n == 1 :
        return 1
    else :
        return n*fact(n-1)
def a(n,l) :
    prod = 1
    for i in xrange(l+1,n+1) :
        prod *= i
    return prod
t = int(raw_input())
while t > 0 :
    t -= 1
    n,k = map(int,raw_input().split())
    l = None
    b = None
    if n-k > k-1 :
        l = n-k
        b = k-1
    else :
        l = k-1
        b = n-k
    print a(n-1,l)/fact(b)
