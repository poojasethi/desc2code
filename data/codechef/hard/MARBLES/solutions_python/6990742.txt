def fact(n):
	ans=1
	for i in xrange(2,n+1):
		ans *= i
	return ans

def intellians(n,k):
	if (k>n/2):
		return intellians(n,n-k)
	else:
		ans=1
		for i in range(1,k+1):
			ans = ans*(n-k+i)
			ans = ans/i
		return ans
