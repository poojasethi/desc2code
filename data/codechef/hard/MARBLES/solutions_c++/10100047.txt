ll nCr(int n,int r)
{
    int i=0;
    ll result=1;
    if(r > n/2)
    {
        r = n-r;
    } 
    for (i = 0; i < r; i++)
    {
        result *= (n-i);
        result /= (i+1);
    }
    return result;
}
