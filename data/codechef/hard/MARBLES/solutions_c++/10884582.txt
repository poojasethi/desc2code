 long long int comb (int n, int r)
{
	if (r==0)
		return 1;
	if (n == r)
		return 1;
	if (r == n-1 || r == 1)
		return n;
	if (r > n/2)
		r = n-r;
	int x = n, y = 1;
	long long int product = 1;
	int i = 1;
	while (i <= r)
	{
		product = product * x;
		product = product / y;
		y++;
		x--;
		i++;
	}
	return product;
}
