long long c(long long n,long long k)
{
    if(n == 0) return 1;
    if(k == 0) return 1;
    if(k == 1 || k == n-1) return n;
    if(k > n/2)
        k = n-k;
    long long num = 1;

    for(long long i = 0;i<k;i++)
    {
        num = num * (n-i)/(i+1);
    }

    return num;
}
