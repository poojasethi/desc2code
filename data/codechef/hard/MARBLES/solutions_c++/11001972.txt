#include<iostream>
#include<stdio.h>
using namespace std;
long long int calc(long long int n, long long int r){

    if(r>n/2){
        r = n-r;
    }

    long long int ans = 1;
    for(int i=1;i<=r;i++)
    {
        ans*=n-r+i;
        ans/=i;
        //cout<<ans<<" ";

    }
    return ans;

}
int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long int n, k;
        cin>>n>>k;
        n--;
        k--;
        cout<<calc(n,k)<<endl;

    }

}
