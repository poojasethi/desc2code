#include<stdio.h>

int main()

{     

      unsigned long int n,k,t;

      scanf("%ld",&t);

      while(t--)

      {

               scanf("%ld %ld",&n,&k);

               --n,--k;

               if(k>n-k)

                        k=n-k;

               unsigned int j=k;

               double num=1;

               for(;j;num*=n,j--,n--);

               for(;k;num/=k,k--);

               printf("%.f\n",num); 

      }   

      return 0;   

}
