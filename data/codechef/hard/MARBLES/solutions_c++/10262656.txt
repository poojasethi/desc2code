#include <iostream>

using namespace std;

long long ncr(long long n,long long r)
{
    long long res=1,i;
    if(r>n/2)
    {
        r=n-r;
    }
    for(i=0;i<r;i++)
    {
        res = res*n;
        res = res/(i+1);
        n--;
    }
    return res;
}

int main()
{
    int t;
    cin>>t;
    while(t--)
    {
        long long n,k;
        cin>>n>>k;
        cout<<ncr(n-1,n-k)<<endl;
    }
    return 0;
}
