#include <stdio.h>
using namespace std;

unsigned long long funct(int n, int k)
{
	unsigned long long result = 1;
	if(n==k)
			return 1;
		if(k==0)
			return 1;
		if(k==1 || k==n-1)
			return n;
		if(k>n/2) k=n-k;
		
		for(int i=1;i<=k;++i)
		{
			result*=n;
			n--;
			result/=i;
		}
		return result;
}

int main()
{
	long int t,n,k;
	scanf("%ld",&t);
	while(t--)
	{
		
		scanf("%ld%ld",&n,&k);
	 	
		printf("%llu\n",funct(n-1,k-1));
	}
	return 0;
}
