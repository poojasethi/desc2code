#include <bits/stdc++.h>
using namespace std;
long long nCr(int n,int r)
{
  long long result=1;
  if (r > n/2)
  {
    r = n - r;
  }
  for (int i = 0; i < r; i++)
  {
    result *= (n-i);
    result /= (i+1);
  }
  return result;
}
 
int main(void)
{
  int t;
  cin >> t;
  while(t--)
  {
    int n,r;
    cin >> n >> r;
    cout << nCr(n-1,r-1) << endl;
  }
  return 0;
}
