#include<stdio.h>
#include<bits/stdc++.h>
using namespace std;
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		int m,p;
		cin>>m>>p;
		if(p>m)
			{
				cout<<0<<endl;
				continue;
			}
		double ans=1;
		m=m-1;
		p=p-1;
		while(p!=0)
		{
			ans=ans*((double)m/p);
			m=m-1;
			p=p-1;
		}
		printf("%.lf\n",ans);
	}
	return 0;
}
