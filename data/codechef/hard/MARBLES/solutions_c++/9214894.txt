#include <iostream>
#include <fstream>

using namespace std;

int T, n, k;

long long int solve (int n, int k)
{
	long long res; 
	int i;

	if (k > n / 2) 
		k = n - k;

	res = 1;
	i = 0;
	while (i < k)
	{
		res *= n;
		res /= i + 1;

		n--;
		i++;
	}

	return res;
}

int main()
{
	//ifstream cin ("main.in");

	unsigned long long i;
	int64_t ans;

	cin >> T;
	while (T--)
	{
		cin >> n >> k;

		ans = solve (n - 1, n - k);

		cout << ans << endl;
	}

	return 0;
}
