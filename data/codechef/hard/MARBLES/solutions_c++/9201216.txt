#include<iostream>
#include<stdint.h>
using namespace std;

long long int C(unsigned long int n, long long int r)
{
    long long int res=1;
    if(r>n/2)
        r=n-r;
    for(int i=0; i<r; i++){
        res*=n;
        res/=i+1;
        n--;
    }
    return res;
}

int main()
{
    int t;
    cin>>t;
    int64_t x;
    unsigned long int n, k, N, R;
    while(t--){
        cin>>n>>k;
        N=k; R=n-k;
        x=C(N+R-1, R);
        cout<<x<<endl;
    }
}
