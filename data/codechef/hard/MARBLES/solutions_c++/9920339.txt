#include<bits/stdc++.h>
using namespace std;

typedef unsigned long int ll;

ll T,N,K,i,j;
double Ans;

int main() {
	scanf("%ld",&T);
	while(T--) {
		scanf("%ld %ld",&N,&K);
		--N;
		--K;
		if(K>(N-K)) K=N-K;
		j=K;
		Ans=1;
		for(;j;Ans*=N,j--,N--);
		for(;K;Ans/=K,K--);
		printf("%.f\n",Ans);
	}
	return 0;
}
