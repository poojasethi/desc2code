#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;
 
ll comb(int n,int r){
    int i=0;
    ll result=1;
    if(r>n/2){
        r=n-r;
    }
    for(int i=0;i<r;i++){
        result*=(n-i);
        result/=(i+1);
    }
    return result;
} 

int main()
{
    ios_base::sync_with_stdio(false);
    
    int t;
    cin >> t;
    while(t--){
        int n,k;
        cin >> n >> k;
        ll ans=comb(n-1,k-1);
        cout << ans <<endl;
    }

}
