#include<stdio.h>
typedef long long int ll;
ll ans_sol(ll n , ll k){
    n--;
    k--;
    if (k > n) return 0;
    if (k * 2 > n) k = n-k;
    if (k == 0) return 1;

    ll result = n;
    for(int i = 2; i <= k; ++i ) {
        result *= (n-i+1);
        result /= i;
    }
    return result;
}
int main(){
    int t;
    scanf("%d",&t);
    for(int i=0;i<t;i++){
         ll n,r;
        scanf("%lld %lld",&n,&r);
        printf("%lld\n",ans_sol(n,r));
    }
}
