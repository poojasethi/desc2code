#include<stdio.h>
unsigned long long  nChoosek( unsigned long n, unsigned long  k )
{
    if (k > n) return 0;
    if (k * 2 > n) k = n-k;
    if (k == 0||k==n) return 1;
    unsigned long long result = n;
    for( int i = 2; i <= k; i++ ) {
        result *= (n-i+1);
        result /= i;
    }
    return result;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
   long  int n,k;
    scanf("%li %li",&n,&k);
    printf("%llu\n",nChoosek(n-1,k-1));
    }
    return 0;
}
