#include<bits/stdc++.h>
using namespace std;

long long int combo(long long int n,long long int k){

    if(k*2>n)
        k=n-k;
    long long int i,result = 1;
    for(i=1;i<=k;i++){

        result*= (n-i+1);
        result/= i;
    }

    return result;


}

int main(){

    int t;
    long long int n,k;

    cin>>t;

    while(t--){

        cin>>n>>k;
        cout<<combo(n-1,k-1)<<endl;

    }


}
