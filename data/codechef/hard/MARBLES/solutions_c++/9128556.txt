#include<iostream>

using namespace std;


typedef long long int li;
int tests;
li n, k;

li solve(li n, li r) {
  li limit = r+1;
  li ret = 1;
  if(r > n/2) r = n-r;
  for(li i=1; i<=r; ++i, n--) {
    ret *= n;
    ret /= i;
  }
  return ret;
}

int main() {
  cin>>tests;

  while(tests--) {
    cin>>k>>n;
    if(n == k) {
      cout<<1<<endl;
      continue;
    }
    k = k > n ? k-n : k;
    n = n+k-1;
    cout<<solve(n, k)<<endl;
  }
  return 0;
}
