 typedef unsigned long long int lli;
using namespace std;
lli gcd(lli a,lli b)
{
if(b==0)
return a;
else
return gcd(b,a%b);
}
lli fac(lli n){

if(n==0) return 1;
else return n*fac(n-1);

}
lli foo(lli n,lli r)
{
     if(n==r) return 1;
     if(r==1) return n;
     return ((n-r+1)*foo(n,r-1))/r;
}
int main(){
int t;
cin>>t;
while(t--){
    lli n,k;
    cin>>n>>k;
    lli y=1,r,r1;
    //n-1cn-k//
    if(n-k>=k-1){
        r=n-k;
        r1=k-1;

    }
else{
    r=k-1;
    r1=n-k;
}
lli count=1;
n--;
while(count<=r1){
    y*=n--;
    y/=count++;

}
