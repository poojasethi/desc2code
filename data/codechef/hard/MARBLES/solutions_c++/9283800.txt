#include<bits/stdc++.h>
using namespace std;
long long ans=1;
long long combination(long n,long m)
{
      m=min(m,n-m);
      for(int i=0;i<m;i++)
      {
          ans=ans*(n-i);
          ans=ans/(i+1);
      }
      return ans;
}
int main()
{
    //cout<<combination(29,6)<<endl;
     long cs;
     cin>>cs;
     while(cs--)
     {
         long n,m;
         cin>>n>>m;
         if(m>n) cout<<0<<endl;
         else if (m==n) cout<<1<<endl;
         else
         {
             ans=1;
             cout<<combination(n-1,m-1)<<endl;
         }
     }
}
