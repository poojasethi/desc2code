#include<bits/stdc++.h>
using namespace std;

int main()
{
    //freopen("in2.in","r",stdin);
    //freopen("out2.out","w",stdout);

    int i,j,k;
    long long int x,y,p,q,num,den,z;
    int n;

    int t;
    cin>>t;

    while(t--)
    {
        cin>>n>>k;
        n=n-k;

        if(n<0)
            return 0;

        x=n;
        y=k-1;

        if(x<y)
            swap(x,y);

        //find factorial
        //(x+y-1)*(x+y-2)*...*(x+)
        //y!
       
        //limits
        p=x+y;
        q=y;

        num=1;
        den=1;

        x=x+1;
        y=1;

        while(x<=p)
        {
            num=num*x;
            den=den*y;

            //divide to avoid overflow
            z=__gcd(num,den);
            num/=z;
            den/=z;

            x++;
            y++;
        }

        cout<<num<<endl;

    }
   
    return 0;
}
