#include<stdio.h>
using namespace std;

unsigned long long fun(int n,int r)
{
  unsigned long long ans=1;
  if(r>n-r)
    r=n-r;
    for(int i=1;i<=r;i++)
        ans=(ans*(n-i+1))/(i);
    return ans;
}

int main()
{
    int k,n,t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        printf("%llu\n",fun(n-1,n-k));
    }
}
