#include<stdio.h>
using namespace std;
 
unsigned long long calc(unsigned long long a,int b,int c)
{
   unsigned long long res,a1=(a/c),a2=(a%c),b1=(b/c),b2=(b%c),z;
    res=(c*a1)*b1;
    res+=a1*b2;
    res+=a2*b1;
    z=(a2*b2)/c;
    res+=z;
    return res;
}
 
unsigned long long fun(int n,int r)
{
  unsigned long long ans=1;
  if(r>n-r)//---------------------------IMP
    r=n-r;
    for(int i=1;i<=r;i++)
        ans=(ans*(n-i+1))/(i);
    return ans;
}
 
int main()
{
    int k,n,t;
    scanf("%d",&t);
    while(t--)
    {
        scanf("%d%d",&n,&k);
        printf("%llu\n",fun(n-1,n-k));
    }
}
