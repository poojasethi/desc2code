#include<stdio.h>
long long C(int n, int r)
{
    if(r > n / 2) r = n - r;

    long long ans = 1;
    int i;

    for(i = 1; i <= r; i++)
    {
        ans *= n - r + i;
        ans /= i;
    }

    return ans;
}
int main()
{
    int t,n,r;

    scanf("%d",&t);

    while(t--)
    {
        scanf("%d %d",&n,&r);

        printf("%lld\n",C(n-1,r-1));
    }
    return 0;
}
