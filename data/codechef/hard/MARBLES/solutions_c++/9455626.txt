#include<iostream>
using namespace std;
long long int f(long long int n,long long int r)
{
	if(n==r) return 1;
	if(n-r<r) r=n-r;
	if(r==0) return 1;
	long long int ans=1,i;
	for(i=1;i<=r;i++)
	{
		ans=(ans*(n-r+i))/(i);
	}
	return ans;
}
main()
{
	long long int n,k,t;
	cin>>t;
	while(t--)
	{
		cin>>n>>k;
		cout<<f(n-1,n-k)<<"\n";
	}
}
