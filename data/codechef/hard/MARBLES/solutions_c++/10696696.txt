    int gcd ( int a, int b )
    {
    	//printf ( "%llu %llu\n",a,b );
    	if (a<b)
    	{
    		b = a+b -(a=b);
    	}
    	//printf ( "%llu %llu\n",a,b );
    	int temp;
    	while ( b!=0 )
    	{
    		temp = a;
    		a = b;
    		b = temp%a;
    	}
    	return a;
    }
