#include<bits/stdc++.h>
using namespace std;

long long int per(long long int n,long long int k)
{
   if(k>n || k<0)
        return 0;

        if(k>n-k)
            k=n-k;
        long long int ans=1;
        for(int i=0;i<k;i++)
            ans=ans*(n-i),ans/=i+1;
        return ans;
}




int main()
{
    int t;
    cin>>t;

    while(t--)
    {
         long long int n,k;
        cin>>n>>k;
        cout<<per(n-1,k-1)<<endl;
    }

return 0;

}
