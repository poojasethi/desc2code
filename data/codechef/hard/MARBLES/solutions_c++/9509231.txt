long long nCr(long long n, long long r) {
	if(n < r)
		return 0;
	if(r > n/2)
		r = n - r;
	long long ans = 1;
	for(int i = 0; i < r; ++i) {
		ans *= (n - i);
		ans /= (i + 1);
	}

	return ans;
}

int main()
{
	int T;
	scanf("%d", &T);

	while(T--) {
		long long N;
		long long K;
		scanf("%lld%lld", &N, &K);
		long long n = K + (N - K) - 1;
		long long r = N - K;
		long long ans = nCr(n, r);
		printf("%lld\n", ans);
	}
    return 0;
}
