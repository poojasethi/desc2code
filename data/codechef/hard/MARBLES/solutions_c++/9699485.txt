#include<bits/stdc++.h>
using namespace std;
#define lli long long int

lli C(lli n,lli r){
    if(r>n/2)
        r=n-r;
    lli ans=1;
    for(lli i=0;i<r;i++){
        ans*=(n-i);
        ans/=(i+1);
    }
    return ans;
}

int main(){
    lli t,n,k;
    cin>>t;
    while(t--){
        cin>>n>>k;
        cout<<C(n-1,k-1)<<endl;
    }
    return 0;
}
