long long int combination(int n , int r)
{
    if(n==r)
        return 1;

    if(r==0)
        return 1;

    if(r==1 || r==n-1)
        return n;
    
    long long int d=1;
    
    int i=1;

    if(r > n/2)
        r = n-r;
    
    int x=n,y=1;
    
    while(i<=r)
    {
        d *= x;
        d /= y;
        y++;
        x--;
        i++;
    }

    return d;
}
