#include <bits/stdc++.h>
using namespace std;

int main() {
	int t;
	cin >> t;
	int n, k;
	while(t--) {
		cin >> n >> k;
		n -= k;
		/*
			ans = C( n+k-1, k-1 ) = (n+k-1)!/(k-1)!/(n)!
								= (n+1)*(n+2)*...(n+k-1)
								--------------------------
								 		(1)(2)...(k-1)
				C(29,6) = 29!/6!/23!
				18*19/1/2 * 20*
		*/
		long long ans = 1;
		long long lastDenom = 0;
		for(int i = 1; i < k; i++) {
			ans *= (n+i);
			while(lastDenom < i && ans % (1+lastDenom) == 0)
				ans /= ++lastDenom;
		}
		cout << "\n";
		
		cout << ans << "\n";
	}
	return 0;
}
