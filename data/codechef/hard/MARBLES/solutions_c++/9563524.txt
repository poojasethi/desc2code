#include<bits/stdc++.h>
using namespace std;

long long int ans(long long int n,long long int r)
{
	
	if(n==r)
		return 1;
	if(r==0)
		return 1;
	if(r==1 || r==n-1)
		return n;
	
	long long int d=1;
	
	int i=1;
	if(r > n/2)
		r = n-r;
	
	
	
	int x=n,y=1;
	
	while(i<=r){
		d *= x;
		d /= y;
		y++;
		x--;
		i++;
	}
	
	return d;
}



int main()
{
	int t;
	long long int n,r;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld%lld",&n,&r);
		n-=r;
		printf("%lld\n",ans(n+r-1,r-1));
		
	}
	return 0;
}
