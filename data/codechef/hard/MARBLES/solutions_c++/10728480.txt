#include <bits/stdc++.h>
using namespace std;
long long int comb(int n,int k)
{
  long long int r = 1;
  if(k-1>(n-1)/2)
    k = n-k+1;
  for (int i = 1; i <= k-1 ; ++i)
   {
     r*=(n-i);
     r/=i;
  }
   return r;
}
int main()
{
  int t;
  cin>>t;
  while(t--)
  {
    int n,k;
    cin>>n>>k;
    cout<<comb(n,k)<<endl;
  }
}
