#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)
#define pb push_back
#define mp make_pair
#define f first
#define sc second
#define inf 10e16

using namespace std;

double dp[101][101];
double w[101];

int main()
{
	ll t,n,i,j,k,l,x,y,z,ww,r,col,prev,c;
	s(t);
	while(t--) {
		s(n);s(c);s(k);
		for(i=0;i<=n;i++) {
			for(j=0;j<=c;j++) dp[i][j]=0;
		}
		for(i=0;i<=n;i++) dp[i][1]=1;
		for(i=1;i<=k;i++) {
			s(l);s(r);
			for(j=l;j<=r;j++) {
				for(col=0;col<=c;col++) w[col]=0;
				for(prev=0;prev<c;prev++) {
					for(col=0;col<c;col++) {
						w[(col*prev)%c]+=(dp[j][prev]/(2*c));
					}
					w[prev]+=(dp[j][prev]/2);
				}
				for(col=0;col<c;col++) {
					dp[j][col]=w[col];
				}
			}
		}
		double sum=0;
		for(i=1;i<=n;i++) {
			double kk=0;
			for(j=0;j<c;j++) {
				kk=kk+j*dp[i][j];
			}
			sum=sum+kk;
		}
		
		printf("%.9lf\n",sum);
	}
			
	return 0;
}
