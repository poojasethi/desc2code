#include<bits/stdc++.h>
using namespace std;
#define ll long long
double dp[70][200][70];
int main()
{
    ll t,n,c,k,l,r,i,j,q,m;
    cin>>t;
    while(t--)
    {
        cin>>n>>c>>k;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<c;j++)
            {
                for(m=1;m<=k;m++)
                {
                    dp[i][j][m]=0;
                }
            }
        }
        for(i=1;i<=n;i++)
        {
            for(j=0;j<c;j++)
            {
                if(j==1)
                {
                    dp[i][j][0]=1;
                }
                else
                {
                    dp[i][j][0]=0;
                }
            }
        }
        q=k;
        for(k=1;k<=q;k++)
        {
            cin>>l>>r;
            for(i=1;i<=n;i++)
            {
                for(j=0;j<c;j++)
                {
                    if(i>=l&&i<=r)
                    {
                        for(m=0;m<c;m++)
                        {
                            dp[i][(j*m)%c][k]+=(dp[i][j][k-1]/(2*c));
                        }
                        dp[i][j][k]+=dp[i][j][k-1]*(0.5);
                    }
                    else
                    {
                        dp[i][j][k]+=dp[i][j][k-1];
                    }
                }
            }
        }
        double sum=0;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<c;j++)
            {
                sum+=j*dp[i][j][q];
            }
        }
        cout<<setprecision(18)<<sum<<"\n";
    }
    return 0;
}
