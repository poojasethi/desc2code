#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define mp make_pair
#define pb push_back
double p = 0.5;
double dp[52][102][52];
vector <pair<ll,ll> > v;
int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
ll t,n,c,k,i,j,s,e,turn,l,r,oldc,newc,latestc;
double ans;
cin>>t;
while(t--)
{
    cin>>n>>c>>k;
    memset(dp,0,sizeof(dp));
    v.clear();

    for(turn=1;turn<=k;turn++)
    {
        cin>>l>>r;
        v.pb(mp(l,r));
    }

    for(i=1;i<=n;i++)
        dp[i][1][0]=1;
    for(turn=1;turn<=k;turn++)
    {
        l=v[turn-1].first;
        r=v[turn-1].second;

        for(i=l;i<=r;i++)
        {
            for(oldc=0;oldc<c;oldc++)
            {
                dp[i][oldc][turn]+=dp[i][oldc][turn-1]*p;
                for(newc=0;newc<c;newc++)
                {
                    latestc=(oldc*newc)%c;
                    dp[i][latestc][turn]+=(dp[i][oldc][turn-1]*p*(1.0/c));
                }
            }
        }

        for(i=1;i<l;i++)
        {
            for(oldc=0;oldc<c;oldc++)
                dp[i][oldc][turn]+=dp[i][oldc][turn-1];
        }
        for(i=r+1;i<=n;i++)
        {
            for(oldc=0;oldc<c;oldc++)
                dp[i][oldc][turn]+=dp[i][oldc][turn-1];

        }
    }

    ans=0;
    for(i=1;i<=n;i++)
    {
        for(j=0;j<c;j++)
            ans=ans+dp[i][j][k]*j;
    }
    cout<<setprecision(18)<<ans<<"\n";
}
return 0;
}
