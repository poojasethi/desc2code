int objects, turns, colors;
std::vector<pair<int,int> > range;

double dp[51][51][102];

double rec(int turn, int ind, int color)
{
	if(turn == turns)
	{
		return color;
	}

	double &ret = dp[turn][ind][color];
	if(ret >= 0)
		return ret;

	ret = 0.0;

	int i, j;

	int l = range[turn].first, r = range[turn].second;

	if(ind >= l && ind <= r)
	{
		fl(i,0,colors)
		{
			ret += rec(turn + 1, ind, (color * i) % colors) * 0.5 * ( 1 / (colors * 1.0) );
		}

		ret += rec(turn + 1, ind, color) * 0.5;
	}
	else
	{
		ret += rec(turn + 1, ind, color);
	}

	return ret;
}

int main()
{
	int i, j, k;

	int cases;

	si(cases);

	wl(cases)
	{

		fl(i,0,51)
			fl(j,0,51)
				fl(k,0,102)
					dp[i][j][k] = -1.0;

		range.clear();

		si(objects);	si(colors);	si(turns);

		fl(i,0,turns)
		{
			int l, r;
			si(l);	si(r);
			l--;	r--;
			range.PB(MP(l,r));
		}

		double ans = 0.0;

		fl(i,0,objects)
		{
			ans += rec(0,i,1);
		}

		printf("%.10lf\n", ans);

	}


	return 0;
}
