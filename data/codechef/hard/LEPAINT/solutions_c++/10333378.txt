#include <bits/stdc++.h>

using namespace std;

double dp[55][55][105],ans;

int main()
{
    int t,i,j,k,m,n,c,p,l,r;
    scanf("%d",&t);
    while(t--)
    {
        memset(dp,0,sizeof(dp));
        scanf("%d %d %d",&n,&c,&p);
        for(i=1;i<=n;i++)
        {
            dp[0][i][1]=1;
        }
        for(j=1;j<=p;j++)
        {
            scanf("%d %d",&l,&r);
            for(i=1;i<=n;i++)
            {
                 for(k=0;k<c;k++)
               {
                if(i>=l && i<=r)
                {
                    for(m=0;m<c;m++){
                   dp[j][i][(k*m)%c]+=dp[j-1][i][k]/(2*c);
                }
                  dp[j][i][k]+=dp[j-1][i][k]*0.5;
                }
                else
                    dp[j][i][k]+=dp[j-1][i][k];
            }
            }
        }
        ans=0;
        for(i=1;i<=n;i++)
        {
            for(j=0;j<c;j++)
            {
              ans+=j*dp[p][i][j];
            }
        }
        cout<<fixed<<setprecision(10)<<ans<<"\n";
    }
    return 0;
}
