#include<iostream>
#include<cstring>
using namespace std;
 
double func(int n,int c,int k)
{
double dp[53][53][200];
memset(dp,0.0,sizeof(dp));
for(int i=1;i<=n;i++)
dp[0][i][1]=1;
for(int i=1;i<=k;i++)
{
int l,r;
cin>>l>>r;
for(int j=1;j<=n;j++)
for(int p=0;p<c;p++)
if(j>=l&&j<=r)
{
for(int m=0;m<c;m++)
dp[i][j][(m*p)%c]=dp[i][j][(m*p)%c]+dp[i-1][j][p]/(2.00*(double)c);
dp[i][j][p]=dp[i][j][p]+.5*dp[i-1][j][p];
}
else
dp[i][j][p]=dp[i][j][p]+dp[i-1][j][p];
}
double ans=0;
for(int i=1;i<=n;i++)
for(int j=0;j<c;j++)
ans=ans+((double)(j))*dp[k][i][j];
return ans;
}
 
int main()
{
ios_base::sync_with_stdio(false);
int t;
cin>>t;
cout<<fixed;
cout.precision(9);
while(t--)
{
int n,c,k;
cin>>n>>c>>k;
cout<<func(n,c,k)<<endl;
}
return 0;
}
