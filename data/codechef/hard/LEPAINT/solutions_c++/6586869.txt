#include <bits/stdc++.h>
#define gc getchar_unlocked
#define fo(i,n) for(i=0;i<n;i++)
#define Fo(i,k,n) for(i=k;i<n;i++)
#define lli long long int
#define pii pair<int,int>
#define vi vector<int>
#define pb push_back
#define mp make_pair
#define mod 1000000007

using namespace std;
void sd(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
	int t;
	sd(t);
	int i,n,c,k;
	
	int l , r , color , x, turn;
	while(t--)
	{
		sd(n);
		sd(c);
		sd(k);
		double dp[k+1][n][c];
		
		fo(i,n)
		fo(turn,k+1)
		fo(color,c)
			dp[turn][i][color] = 0.0;
		fo(i,n)
			dp[0][i][1] = 1.0;
		Fo(turn,1,k+1)
		{
			sd(l);
			sd(r);
			l--;
			r--;
			fo(i,n)
			{
				fo(color,c)
				{
					if( i < l | i > r)
					{
						dp[turn][i][color] = dp[turn-1][i][color];
						continue;
					}
					dp[turn][i][color] += dp[turn-1][i][color]*0.5;
					fo(x,c)
						dp[turn][i][(color*x)%c] +=	dp[turn-1][i][x]*0.5/c;
				
				}
			}
		}
		double ans = 0;
		fo(i,n)
		fo(color,c)
			ans += dp[k][i][color]*color;
			
		
		printf("%.20f\n",ans);
	
	}
	
	return 0;
}
