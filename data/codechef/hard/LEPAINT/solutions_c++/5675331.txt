#include <iostream>
using namespace std;

int main()
{
	int t;
	cin>>t;
	while(t--)
	{
	double dp[51][51][101];
	for(int i=0;i<51;i++)
	{
		for(int j=0;j<51;j++)
		{
			for(int k=0;k<101;k++)
			{
				dp[i][j][k]=0;
			}
		}
	}
	int n,K,C;
	cin>>n>>C>>K;
	int l[K],r[K];
	int a[n];
	for(int i=1;i<=K;i++)
	{
		cin>>l[i]>>r[i];l[i]--;r[i]--;
	}
	for(int i=0;i<n;i++)
	{
		dp[0][i][1]=1;
		a[i]=1;
	}
	for(int k=1;k<=K;k++)
	{
		for(int j=0;j<n;j++)
		{
			if(j>=l[k]&&j<=r[k])
			{
				for(int c=0;c<C;c++)
				{
					dp[k][j][c]+=0.5*dp[k-1][j][c];
					for(int cc=0;cc<C;cc++)
					{
						dp[k][j][(c*cc)%C]+=(0.5*dp[k-1][j][c])/C;
					}
				}
			}
			else
			{
				for(int c=0;c<C;c++)
				{
					dp[k][j][c]=dp[k-1][j][c];
				}
			}
		}	
	}
	double sum=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<C;j++)
		{
			sum+=dp[K][i][j]*j;	
		}
	}
	cout.precision(7);
	cout<<fixed<<sum<<"\n";
	}
	return 0;
}
