#include<bits/stdc++.h>
#define inf (1e+9)

using namespace std;

pair<int,int> c[20],t[20];

int n,dp[1<<8][20][20][20];

int rec(int mask,int pos,int t1,int t2)
{
    int x,y;

    if(dp[mask][pos][t1][t2]!=-1) return dp[mask][pos][t1][t2];

    int &ret=dp[mask][pos][t1][t2];
    ret=inf;

    if(pos>=11&&pos<=18) ///on tool
    {
        x=t[pos-10].first;
        y=t[pos-10].second;
    }
    else
    {
        x=c[pos].first;
        y=c[pos].second;
    }

if(mask == (1<<n)-1)	return ret=abs(x)+abs(y);

    if(t1>0&&t2>0)
    {
        ret=min(abs(x-c[t1].first)+abs(y-c[t1].second)+rec(mask|1<<(t1-1),t1,0,t2),abs(x-c[t2].first)+abs(y-c[t2].second)+rec(mask|1<<(t2-1),t2,t1,0));
    }
    else if(t1>0)
    {
        ret=min(ret,abs(x-c[t1].first)+abs(y-c[t1].second)+rec(mask|1<<(t1-1),t1,0,0));

        for(int i=0;i<n;i++)
        {
            if((!((1<<i)&mask))&&t1!=i+1)
            {
                ret=min(ret,abs(x-t[i+1].first)+abs(y-t[i+1].second)+rec(mask,i+11,t1,i+1));
            }
        }
    }
    else if(t2>0)
    {
        ret=min(ret,abs(x-c[t2].first)+abs(y-c[t2].second)+rec(mask|1<<(t2-1),t2,0,0));

        for(int i=0;i<n;i++)
        {
            if((!((1<<i)&mask))&&t2!=i+1)
            {
                ret=min(ret,abs(x-t[i+1].first)+abs(y-t[i+1].second)+rec(mask,i+11,i+1,t2));
            }
        }
    }
    else
    {
        for(int i=0;i<n;i++)
        {
            if(!((1<<i)&mask))
                ret=min(ret,abs(x-t[i+1].first)+abs(y-t[i+1].second)+rec(mask,i+11,i+1,0));
        }
    }

    return ret;
}

int main()
{

int cases;
cin>>cases;

while(cases--)
{
  cin>>n;

  for(int j=1;j<=n;j++)
   cin>>c[j].first>>c[j].second>>t[j].first>>t[j].second;

  memset(dp,-1,sizeof(dp));

  cout<<rec(0,0,0,0)<<endl;

}

return 0;
}
