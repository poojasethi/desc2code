int cc[100];

int px[2][10], py[2][10];
int dp[1<<10][10][10][10][2];
int n;

int solve(int mask, int p1, int p2, int last, bool flag)
{
   // trace5(p1, p2, mask, last, flag);
  //  getchar();
    if(mask == 0) return abs(px[flag][last]) + abs(py[flag][last]);
    if(dp[mask][p1][p2][last][flag] != -1) return dp[mask][p1][p2][last][flag];
    int ans = 1e9;

    if(p1 == n+1) {
        fori(i,0,n) {
            if(i == p2) continue;
            if(mask&(1<<i)) {
                ans = min(ans, abs(px[flag][last]-px[1][i]) + abs(py[flag][last]-py[1][i])+solve(mask, i, p2, i, 1));
            }
        }
    }
    if(p2 == n+1) {
        fori(i,0,n) {
            if(i == p1) continue;
            if(mask&(1<<i)) {
                ans = min(ans, abs(px[flag][last]-px[1][i]) + abs(py[flag][last]-py[1][i])+solve(mask, p1, i, i, 1));
            }
        }
    }
    if(p1 != n+1) {
        ans = min(ans, abs(px[flag][last]-px[0][p1])+abs(py[flag][last]-py[0][p1]) + solve(mask^(1<<p1), n+1, p2, p1, 0));
    }
    if(p2 != n+1) {
        ans = min(ans, abs(px[flag][last]-px[0][p2])+abs(py[flag][last]-py[0][p2]) + solve(mask^(1<<p2), p1, n+1, p2, 0));
    }
    return dp[mask][p1][p2][last][flag] = ans;
}
