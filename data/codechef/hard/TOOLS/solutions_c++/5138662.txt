int dp[1<<8][17][10][10];

// person i, item = 8+i
int f(int pos, int mask, int h1, int h2){
    if(dp[mask][pos][h1][h2] != -1)
        return dp[mask][pos][h1][h2];
    //cout << mask << " " << pos << " " << h1 << " " << h2 << endl;
    //pau;
    if(mask == ((1 << n)-1)){
//        cout << mask << " " << h1 << " " << h2 << endl;
        return a[pos][0] + a[pos][1];
    }
    int ans = 1e9;
    dec;
    if(!pos && !mask && h1==9 && h2==9){
//        cout << "haha11\n";
        frl(i, 0, n){
            ans = min(ans, abs( b[8+i][0]) + abs(b[8+i][1]) +  f(8+i, 0, i, 9));
        }
        return ans;
    }
    if(h1 < 9){
        //give h1
//        cout << "haha22\n";
        if(pos < 8)
            ans = min(ans, abs(a[pos][0] - a[h1][0]) + abs(a[pos][1] - a[h1][1]) +  f(h1, mask | (1 << h1), 9, h2));
        else {
            ans = min(ans, abs(b[pos][0] - a[h1][0]) + abs(b[pos][1] - a[h1][1]) +  f(h1, mask | (1 << h1), 9, h2));
        }
    }
    if(h2 < 9){
        //give h2
//        cout << "haha33\n";
        if(pos < 8)
            ans = min(ans, abs(a[pos][0] - a[h2][0]) + abs(a[pos][1] - a[h2][1]) +  f(h2, mask | (1 << h2), h1, 9));
        else {
            ans = min(ans, abs(b[pos][0] - a[h2][0]) + abs(b[pos][1] - a[h2][1]) +  f(h2, mask | (1 << h2), h1, 9));
        }
    }
    if(h1 == 9){
        //take h1
//        cout << "haha44\n";
        frl(i, 0, n){
            if(!(mask & (1 << i))){
                if(pos < 8)
                    ans = min(ans, abs(a[pos][0] - b[i+8][0]) + abs(a[pos][1] - b[i+8][1]) +  f(8+i, mask, i, h2));
                else {
                    ans = min(ans, abs(b[pos][0] - b[i+8][0]) + abs(b[pos][1] - b[i+8][1]) +  f(8+i, mask, i, h2));
                }
            }
        }
    }
    if(h2 == 9){
        //take h2
//        cout << "haha55\n";
        frl(i, 0, n){
            if(!(mask & (1 << i))){
                if(pos < 8)
                    ans = min(ans, abs(a[pos][0] - b[i+8][0]) + abs(a[pos][1] - b[i+8][1]) +  f(8+i, mask, h1, i));
                else {
                    ans = min(ans, abs(b[pos][0] - b[i+8][0]) + abs(b[pos][1] - b[i+8][1]) +  f(8+i, mask, h1, i));
                }
            }
        }
    }
    return dp[mask][pos][h1][h2] = ans;
}
