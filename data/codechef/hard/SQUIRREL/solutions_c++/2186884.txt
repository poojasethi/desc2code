#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#define fr(i,n) for(i=0;i<n;i++)
#define Fr(i,m,n) for(i=m;i<n;i++)
#define max(x,y) (x)>(y) ? (x) : (y)
#define mod 1000000007
using namespace std;
typedef long long int ll;
int t[10005],p[10005],f[10005];
int main()
{
    int n,m,k,tt,i,j;
    ll l,h,tm,c;
    scanf("%d",&tt);
    while(tt--)
    {   l=0; h=mod; c=0;
        scanf("%d%d%d",&m,&n,&k);
        Fr(i,1,m+1) scanf("%d",&t[i]);
        Fr(i,1,m+1) scanf("%d",&p[i]);
        
        while(l<h)
        {
        tm = (l+h)/2;
        Fr(i,1,m+1)
            f[i]=(tm>=t[i] ? (tm-t[i])/p[i] +1 : 0);
        sort(f+1,f+m+1);
        c=0;
        for(i=m,j=0;i>0&&j<n ;i--,j++)
            c+= f[i];
        if(c<k)   l=tm+1;
        else      h=tm;
        }
        printf("%lld\n",l);
    }
    return 0;
}
