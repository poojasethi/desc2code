typedef vector<int> VI;

#define N 11111
int a[N];
PII pt[N];

int main()
{
	#ifndef ONLINE_JUDGE
	freopen("in.txt","r",stdin);
	#endif
	int T;
	for(scanf("%d",&T);T--;)
	{
		int n,m,k,i;
		scanf("%d%d%d",&n,&m,&k);
		for(i=0;i<n;i++) scanf("%d",&pt[i].second);
		if(m>n) m=n;
		int q=(k-1)/m;
		for(i=0;i<n;i++)
		{
			scanf("%d",&pt[i].first);
			a[i]=pt[i].second+q*pt[i].first;
		}
		sort(pt,pt+n);
		sort(a,a+n);
		int L=a[0]-1;
		int R=a[m-1];
		while(L+1<R)
		{
			int M=(L+R)/2;
			for(i=0;i<n;i++)
				a[i]=M<pt[i].second?0:(M-pt[i].second)/pt[i].first+1;
			sort(a,a+n);
			reverse(a,a+n);
			int s=0;
			for(i=0;i<m && s<k;i++)
				s+=a[i];
			if(s>=k) R=M; else L=M;
		}
		printf("%d\n",R);
	}
	return 0;
}
