#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{	int test,i,m,n,k,t[10001],p[10001];
	scanf("%d",&test);
	while(test--)
	{
		scanf("%d%d%d",&m,&n,&k);
		for(i=0;i<m;i++)
		scanf("%d",&t[i]);
		for(i=0;i<m;i++)
		scanf("%d",&p[i]);
		int min=t[0]+(k-1)*p[0];
		//printf("%d\n",min);
		for(i=1;i<m;++i)
		{	
			t[i]+(k-1)*p[i];
		//	printf("%d\n",q[i]);
			if(t[i]+(k-1)*p[i]<min)
			min=t[i]+(k-1)*p[i];
		}
		int st=0,end=min,mid,s[m],res=min;
		
		while(st<=end)
		{long long sum=0ll;
			mid=st+(end-st)/2;
			for(i=0;i<m;++i)
			{
				if(mid>=t[i])
				s[i]=(mid-t[i])/p[i]+1;
				else
				s[i]=0;
			}
			
			sort(s,s+m);
			for(i=0;i<n&&i<m;++i)
			sum+=s[m-1-i];
			if(sum<k)
			{
				st=mid+1;
			}
			else
			{
				end=mid-1;
				res=mid;
			}
		}
		printf("%d\n",res);
	}
	return 0;
}
