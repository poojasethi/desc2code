#include<iostream>
#include<algorithm>
#define BUF 4096
char ibuf[BUF];
int ipt = BUF;
int readInt() 
{
    while (ipt < BUF && ibuf[ipt] < '0') ipt++;     
    if (ipt == BUF)
    {
         fread(ibuf, 1, BUF, stdin);
         ipt = 0;
         while (ipt < BUF && ibuf[ipt] < '0') ipt++;
    }
    int n = 0;
    while (ipt < BUF && ibuf[ipt] >= '0') n = (n*10)+(ibuf[ipt++]-'0');
    if (ipt == BUF)
    {
         fread(ibuf, 1, BUF, stdin);
         ipt = 0;
         while (ipt < BUF && ibuf[ipt] >= '0') n = (n*10)+(ibuf[ipt++]-'0');
    }
    return n;
}
using namespace std;
int m,n,k,T[10001],P[10001],nuts[10001],i;
long long total;
bool f(int t)
{
    for(i=0;i<m;i++)
    {
       if(t>=T[i])
         nuts[i]=(t-T[i])/P[i]+1;
       else
         nuts[i]=0;
    }
    nth_element(nuts,nuts+m-n,nuts+m);
    total=0;
    for(i=m-n;i<m;i++)
       total+=nuts[i];
    return total>=k;
}
int main()
{
    int tc;
    tc=readInt();
//    scanf("%d",&tc);
    while(tc--)
    {
        int i,j,t=1,u,l=0,mid,x;
        m=readInt();
        n=readInt();
        k=readInt();
        //scanf("%d%d%d",&m,&n,&k);
        n=n<m?n:m;
        for(i=0;i<m;i++)
          T[i]=readInt();
//           scanf("%d",&T[i]);
        for(i=0;i<m;i++)
          P[i]=readInt();
  //         scanf("%d",&P[i]);
        u=T[0]+P[0]*k;
        for(i=1;i<m;i++)
           u=u>T[i]+P[i]*k?T[i]+P[i]*k:u;
        while(l<u)
        {
           mid=(u+l)>>1;
           if(f(mid))
           u=mid;
           else
           l=mid+1;
        }
        printf("%d\n",u);
    }
    return 0;
}
