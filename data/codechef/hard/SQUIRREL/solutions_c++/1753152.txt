#include<vector>

{
    int t;

    {

       for(i=0;i<m;i++) cin>>t[i];

       }

          for(i=0;i<m;i++)
          {
              if(t[i]<=mid) coll[i]=(mid-t[i])/p[i]+1;
              else coll[i]=0;
          }
          sort(coll,coll+m);
         // long long tot=0;
          for(i=0;i<n&&i<m;i++)
          tot+=coll[m-i-1];
//cout<<"tot collectef"<<tot<<"  time"<<mid<<endl;
           if(tot<k) {
           s=mid+1;}
           else{
           e=mid-1;
          if(mint>mid) mint=mid;}
       }
           cout<<mint<<endl;
           }

}
