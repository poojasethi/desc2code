#include <iostream>
#include<algorithm>
using namespace std;
int ti[10005],p[10005],a[10005];
int main()
{
    int t,m,n,k,i;
    long long int l,h,mid,s;
    cin>>t;
    while(t--)
    {
        cin>>m>>n>>k;
        for(i=0;i<m;i++)
        cin>>ti[i];
        for(i=0;i<m;i++)
        cin>>p[i];
        l=0;h=1000000000;
        while(l<h)
        {
            mid=(l+h)/2;
            for(i=0;i<m;i++)
            {
                if(mid>=ti[i])
                a[i]=((mid-ti[i])/p[i])+1;
                else
                a[i]=0;
            }
            sort(a,a+m);
            s=0;
            for(i=m-1;i>=0&&i>=m-n;i--)
            s=s+a[i];
            if(s<k)
            l=mid+1;
            else
            h=mid;
        }
        cout<<l<<endl;
    }
    return 0;
}
