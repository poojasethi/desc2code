char dum;
#define MX 100005
int n,m,k,t[MX],p[MX];

int solve(int mid){
	
	lli sum=0;
	VI v;
	FOR(i,0,m){
		if(mid>=t[i]){
			v.pb(1+(mid-t[i])/p[i]);
		}
		else v.pb(0);
	}
	SORT(v);REV(v);
	for(int i=0;i<n and i<m;i++){
		sum+=v[i];
	}
	return sum>=k;
}
