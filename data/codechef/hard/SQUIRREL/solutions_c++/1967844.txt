{

 nth_element(num,num+m-n,num+m);

 for(int i=m-n;i<m;i++)

}
int bin(int l,int r)
{
 int m;
 while(l<r)
 {
  m=(l+r)>>1;
  if(comp(m))r=m;
  else l=m+1;
 }
 return l;
}
