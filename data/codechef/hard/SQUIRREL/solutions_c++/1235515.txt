LL pow_r(LL base,LL n)
{
	LL d=base,x=1ll;
	while(n>0)
	{
		if(n&1)
			x=(x*d);
		d=(d*d);
		n>>=1;
	}
	return x;
}

    char c;int num=0;
    c=getchar_unlocked();
    while(!(c>='0' && c<='9')) c=getchar_unlocked();
    while(c>='0' && c<='9'){
        num=(num<<3)+(num<<1)+c-'0';

    }

}
