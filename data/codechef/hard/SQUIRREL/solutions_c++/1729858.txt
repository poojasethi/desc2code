#define MAXN 10002
#define __ printf("\n")
using namespace std;

int nkasus;
int n,m;
int t[MAXN],p[MAXN];
LL tot,mau;
int ar[MAXN],ans;

bool bisa(int x){
	REP(i,m){
		ar[i] = 0;
		if (t[i] <= x){
			ar[i] = 1 + (x - t[i])/p[i];
		}
		
		ar[i] = -ar[i];	
	}
	
	sort(ar, ar + m);
	
	tot = 0;
	REP(i,n){
		tot -= (LL)ar[i];
	}
	
	return (tot >= mau);
}

int main(){
	scanf("%d", &nkasus);
	REP(jt,nkasus){
		scanf("%d%d%lld", &m, &n, &mau);
		
		n = min(n, m);	
			
		REP(i,m){
			scanf("%d", &t[i]);
		}
		REP(i,m){
			scanf("%d", &p[i]);
		}
		
		int ki,ka;
		ki = 0;
		ka = 2000000000;
		while (ki <= ka){
			int tgh = (ki + ka) >> 1;
		
			if (bisa(tgh)){
				ans = tgh;
				ka = tgh - 1;
			}else{
				ki = tgh + 1;
			}
		}
		
		printf("%d\n", ans);
	}
		
	return 0;
}
