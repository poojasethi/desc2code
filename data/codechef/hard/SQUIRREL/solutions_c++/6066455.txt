#include <bits/stdc++.h>

using namespace std;

#define gc getchar_unlocked

void scanint(int &x)
{
    register int c = gc();
    x = 0;
    for(;(c<48 || c>57);c = gc());
    for(;c>47 && c<58;c = gc()) {x = (x<<1) + (x<<3) + c - 48;}
}

int main()
{
    int t, n, m ,k, i, j, lo, hi, mid, x;
    scanint(t);
    while(t--){
        scanint(m);
        scanint(n);
        scanint(k);
        int arr[m], t[m], p[m];
        for(i=0; i<m; i++)
            scanint(t[i]);
        for(i=0; i<m; i++)
            scanint(p[i]);
        lo=0, hi=1000000300;
        while(lo<hi){
            mid=(lo+hi)>>1;
            for(i=0; i<m; i++){
                if(mid>=t[i])
                    arr[i]=((mid-t[i])/p[i])+1;
                else
                    arr[i]=0;
            }
            sort(arr, arr+m);
            bool dir=true;
            x=0;
            for(i=m-1, j=0; i>=0 && j<n; j++, i--){
                x+=arr[i];
                if(x>=k){
                    dir=false;
                    break;
                }
            }
            if(dir)
                lo=mid+1;
            else
                hi=mid;
        }
        printf("%d\n", lo);
    }
    return 0;
}
