#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;
typedef long long tint;

const int MAXN = 10001;
int t[MAXN],p[MAXN]; 
tint nuts[MAXN];

int main()
{
	int tests; scanf("%d",&tests);
	while(tests-- > 0){
		int n,m,k;
		scanf("%d%d%d",&m,&n,&k);
		for(int i = 0; i < m; i++)
			scanf("%d",&t[i]);
		for(int i = 0; i < m; i++)
			scanf("%d",&p[i]);

		tint l = 0, r = t[0]+((k-1)*1LL)*p[0]+1;
		int up = min(m,n);
		while(r - l > 1){
			tint mid = (r+l)/2;
			for(int i = 0; i < m; i++)
				nuts[i] = (mid >= t[i]) ? (1 + (mid-t[i])/p[i]) : 0;

			sort(nuts,nuts+m);
			tint s = 0;
			for(int j = 0; j < up; j++){
				s += nuts[m-1-j];
			}
			if(s >= k) r = mid; else l = mid;
		}
		printf("%lld\n",r);
	}
}
