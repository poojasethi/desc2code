#include <bits/stdc++.h>

using namespace std;

#define type(x) __typeof((x).begin())
#define foreach(it, x) for(type(x) it = (x).begin(); it != (x).end(); it++)

typedef long long ll;
typedef pair < int, int > ii;

const int inf = 1e9 + 333;
const ll linf = 1e18 + inf;

const int N = 1e5 + 5;

int n, m;
char s[N];
ll dp[N][2][2][2];

ll f(int x, int l1, int l2, int l3) {
	if(x > n) {
		return l2 == n - 1;
	}
	if(l1 < x - 3 and l2 < x - 2 and l3 < x - 2)
		return 0;
	ll &r = dp[x][l1 & 1][l2 & 1][l3 & 1];
	if(r != -1) {
		return r;
	}
	r = 0;
	for(int i = 1; i <= m; i++) {
		int newL1 = (i == s[x - 1] ? l1 + 1 : l2);
		int newL2 = (i == s[x] ? l2 + 1 : max(newL1, l3));
		int newL3 = (i == s[x + 1] ? l3 + 1 : newL2);
		r += f(x + 1, newL1, newL2, newL3);
	}
	//printf("x = %d l1 = %d l2 = %d l3 = %d r = %d\n", x, l1, l2, l3, r);
	return r;
}

int main() {

	scanf("%d %d", &n, &m);

	scanf("%s", s + 1);

	for(int i = 1; i <= n; i++) {
		s[i] = s[i] - 'a' + 1;
	}

	memset(dp, -1, sizeof(dp));

	printf("%lld\n", f(1, 0, 0, 0));

	return 0;

}
