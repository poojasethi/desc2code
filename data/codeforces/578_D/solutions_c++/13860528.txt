#include <bits/stdc++.h>
using namespace std;
int const N = 101001;
char s[N];
int main() {
    int n, m; scanf("%d%d", &n, &m);
    scanf(" %s", s);
    long long c = 1;
    for (int i = 0; i < n; ++i)
        if (i > 0 && s[i] != s[i - 1]) ++c;
    long long ans = c * n * (m - 1);
    for (int i = 1; i < n; ++i) {
        if (s[i] != s[i - 1]) {
            int t = 2;
            for (int j = i + 1; j < n; ++j) {
                if (s[j] != s[j - 2]) {
                    break;
                }
                i = j;
                ++t;
            }
            ans -= 1LL * t * (t - 1) / 2;
        }
    }
    printf("%I64d\n", ans); 
    return 0;
}
