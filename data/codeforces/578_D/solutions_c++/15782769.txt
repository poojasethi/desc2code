#include <cstdio>
#include <cstring>

#define L 100005

typedef __int64 LL;

char s[L];

main() {
	int n, m;
	scanf("%d %d %s", &n, &m, s);
	int cnt = 1;
	for (int i = 1; i < n; i ++) if (s[i] != s[i - 1]) cnt ++;
	LL rlt = 1LL * n * cnt * (m - 1);
	int len = 0;
	for (int i = 1; i < n; i ++) if (s[i] != s[i - 1]) {
		if (i > 1 && s[i] == s[i - 2]) {
			len ++;
			rlt -= len + 1;
		} else rlt --, len = 0;
	}
	printf("%I64d\n", rlt);
}
