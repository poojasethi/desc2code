#include <bits/stdc++.h>
using namespace std;
typedef unsigned long long int ulli;
typedef long long int lli;
#define pb push_back
#define ft first
#define se second
#define mp make_pair

int main(){
	int n, m;
	cin >> n >> m;
	string s;
	cin >> s;
	lli res = n * (m - 1);
	for(int i = 1; i < n; i++){
		if(s[i] != s[i - 1])
			res += n * (m - 1);
	}
	lli pos = 1;
	for(int i = 1; i < n; i++){
		if(pos == 1){
			if(s[i] == s[i - 1])
				continue;
			else
				pos++;
		}
		else{
			if(s[i] == s[i - 2])
				pos++;
			else{
				res -= (pos * (pos - 1)) / 2;
				if(s[i] == s[i - 1])
					pos = 1;
				else
					pos = 2;
			}
		}
	}
	res -= (pos * (pos - 1)) / 2;
	cout << res << endl;
}
