#include <bits/stdc++.h>
using namespace std ;
#define LL long long 
#define MAX 200004

string S ;

int main()
{
  int N , M ;
  cin >> N >> M ;
  
  cin >> S ;

  LL answer = N*M-N ;

  for(int i = 1 ; i < N ; i++)
    if(S[i]!=S[i-1])
      answer += (N*M-N) ;


  LL curr = 1 ;
  
  for(int i = 1 ; i  < N ; i++)
  {
    if(curr == 1)
    {
      if(S[i]==S[i-1])
        continue ;
      else 
        curr++ ;
    }
    else
    {
      if(S[i] == S[i-2]) curr++ ;
      else
      {
        answer -= (curr*(curr-1))/2 ;
        if(S[i] == S[i-1]) curr = 1 ;
        else curr = 2 ;
      }
    }
  }

  answer -= (curr*(curr-1))/2 ;

  cout << answer << endl ;   
}
