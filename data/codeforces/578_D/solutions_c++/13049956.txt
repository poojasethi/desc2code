#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
int n, m, id[100005];
long long ans = 0;
char s[100005];
int hashi(int x, int y)
{
	if (x == y) return -1;
	if (x > y) swap(x, y);
	return x * 1000 + y;
}
int main()
{
	scanf("%d%d", &n, &m);
	scanf("%s", s + 1);
	ans = 1LL * (m - 1) * n;
	for (int i = 2; i <= n; i++)
		if (s[i - 1] != s[i])
			ans += 1LL * (i - 1) * (m - 1);
	for (int i = 1; i <= n - 1; i++)
		if (s[i] != s[i + 1])
			ans += 1LL * (n - i) * (m - 1);
	for (int i = 1; i <= n - 1; i++) id[i] = hashi(s[i], s[i + 1]);
	for (int i = 1; i <= n - 1; i++)
		if (id[i] != -1 && (i == 1 || id[i] != id[i - 1]))
			for (int j = i; j <= n - 1 && id[j] == id[i]; j++)
				ans -= j - i + 1;
	cout << ans << endl;
	return 0;
}
