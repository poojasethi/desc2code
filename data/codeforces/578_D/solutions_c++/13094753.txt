#include <bits/stdc++.h>
#define rep(i, a, b) for(int i = (a); i <= (b); ++i)
#define per(i, a, b) for(int i = (a); i >= (b); --i)
typedef long long LL;
using namespace std;

const int MAXN = 100009;

int n, m;
char s[MAXN];
LL ans;

int main()
{
#ifndef ONLINE_JUDGE

    freopen("D.out", "w", stdout);
#endif

    scanf("%d%d", &n, &m);
    scanf("%s", s), ans = n*m-n;
    rep(i, 1, n-1)
        if(s[i] != s[i-1])
            ans += n*m-n;
    LL len = 1;
    rep(i, 1, n-1)
        if(len == 1)
        {
            if(s[i] == s[i-1])
                continue;
            else len++;
        }
        else
        {
            if(s[i] == s[i-2])
                len++;
            else
            {
                ans -= (len*(len-1))>>1;
                if(s[i] != s[i-1])
                    len = 2;
                else len = 1;
            }
        }
    ans -= (len*(len-1))>>1;
    cout << ans << endl;

    return 0;
}
