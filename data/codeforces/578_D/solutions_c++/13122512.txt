#include <cstdio>

int main() {
  int n, m;
  char a[100001];
  long long ret;
  scanf("%d%d", &n, &m);
  ret = 1LL * n * n * (m - 1);
  scanf("%s", a);
  int c = 0;
  for (int i = 1; i < n; i++) {
    if (a[i] == a[i - 1]) {
      ret -= n * (m - 1);
      c = 0;
    } else if (i > 1 && a[i] == a[i - 2]) {
      c++;
      ret -= c + 1;
    } else {
      ret--;
      c = 0;
    }
  }
  printf("%I64d\n", ret);
  return 0;
}
