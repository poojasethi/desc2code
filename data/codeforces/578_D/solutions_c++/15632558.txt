#include <iostream>

using namespace std;

typedef long long int ll;

int n,m;
int v[1000000];
int numle[1000000];
int numri[1000000];
int len[1000000];

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>m;
  ll sol=(m-1)*n;
  for (int i=1;i<=n;i++) {
    char c;
    cin>>c;
    v[i]=c-'a';
  }
  for (int i=2;i<=n;i++) {
    numle[i]=numle[i-1]+(v[i-1]!=v[i]);
    sol+=(m-1)*numle[i];
  }
  for (int i=n-1;i>=1;i--) {
    numri[i]=numri[i+1]+(v[i]!=v[i+1]);
    sol+=(m-1)*numri[i];
  }
  for (int i=2;i<=n;i++) {
    if (v[i-1]==v[i]) len[i]=0;
    else if (i>2 and v[i-2]==v[i]) len[i]=1+len[i-1];
    else len[i]=1;
    sol-=len[i];
  }
  cout<<sol<<endl;
}
