#include <cstring>
#include <cstdio>
#include <algorithm>

using namespace std;
typedef long long LL;

const int MAXN = 1e5 + 5;

char S[MAXN];
int N, M;

int main() {
	scanf("%d%d", &N, &M);
	scanf("%s", S + 1);
	LL Ans = 1ll * N * M - N;
	for (int i = 2; i <= N; i ++) 
		if (S[i] != S[i - 1]) Ans += 1ll * N * M - N;
	
	LL Cnt = 1;
	for (int i = 2; i <= N; i ++) {
		if (Cnt == 1) {
			if (S[i] == S[i - 1]) continue; else Cnt ++;
		} else {
			if (S[i] == S[i - 2]) Cnt ++; else {
				Ans -= Cnt * (Cnt - 1) / 2;
				if (S[i] == S[i - 1]) Cnt = 1; else Cnt = 2;
			}
		}
	}
	Ans -= Cnt * (Cnt - 1) / 2;
	printf("%I64d\n", Ans);
}
