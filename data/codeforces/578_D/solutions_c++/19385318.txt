#include <bits/stdc++.h>

using namespace std;

#define NN 110000
#define MOD 1000000007
#define inf 0x3f3f3f3f
#define eps 1e-8
#define eprintf(...) fprintf(stderr, __VA_ARGS__)

typedef long long INT;
typedef pair<int, int> pii;

int N, M;
int com[NN];
char str[NN];

int main() {
	scanf("%d%d", &N, &M);
	scanf("%s", str);
	
	INT ans=(M-1)*N;
	for (int i=0; i<N-1; i++) {
		if (str[i]!=str[i+1]) {
			ans+=(M-1)*N-1;
		}
	}
	
	com[0]=com[1]=0;
	for (int i=2; i<N; i++) {
		if (str[i]!=str[i-1] && str[i]==str[i-2]) com[i]=com[i-1]+1;
		else com[i]=0;
		ans-=com[i];
	}
	
	printf("%I64d\n", ans);
	
	return 0;
}
