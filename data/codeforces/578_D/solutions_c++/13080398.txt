#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long int64;

char s[100000 + 10];
int64 ans;
int two[100000 + 10];
int n, m;

int doit(char x, char y) {
    if(x == y)
        return -1;
    x -= 'a', y -= 'a';
    if(x > y)
        swap(x, y);
    return x * 100 + y;
}

int main() {
    scanf("%d%d", &n, &m);
    scanf("%s", s + 1);
    ans += (int64) n * (m - 1);
    for(int i = 1; i <= n - 1; i++)
        if(s[i] != s[i + 1])
            ans += (int64) (n - i) * (m - 1);
    for(int i = 2; i <= n; i++)
        if(s[i - 1] != s[i])
            ans += (int64) (i - 1) * (m - 1);
    for(int i = 1; i <= n - 1; i++)
        two[i] = doit(s[i], s[i + 1]);
    for(int i = 1; i <= n - 1; i++)
        if(two[i] != -1 && (i == 1 || two[i] != two[i - 1])) {
            for(int j = i; j <= n - 1 && two[i] == two[j]; j++)
                ans -= (j - i + 1);
        }
    printf("%I64d\n", ans);
    return 0;
}
