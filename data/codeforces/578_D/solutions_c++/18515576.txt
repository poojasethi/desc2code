#include <bits/stdc++.h>

const int MAXN = 100005;

int n, m;
char s[MAXN];
long long ans;

void input()
{
	scanf("%d%d%s", &n, &m, s + 1);
}

void solve()
{
	ans = n * (m - 1);
	for(int i = 1; i < n; ++i)
		if(s[i] != s[i + 1])
			ans += n * (m - 1);
	for(int i = 1; i < n; i++)
	{
		if(s[i] != s[i + 1])
		{
			int cur = 0;
			while(i + cur <= n && s[i + cur] == s[i + (cur & 1)])
				cur++;
			cur--;
			ans -= (long long)cur * (cur + 1) / 2;
			i += cur - 1;
		}
	}
}

void output()
{
	std::cout << ans << std::endl;
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("lcs.in", "r", stdin);
	freopen("lcs.out", "w", stdout);
#endif

	input();
	solve();
	output();

#ifndef ONLINE_JUDGE
	fclose(stdin);
	fclose(stdout);
#endif
	return 0;
}
