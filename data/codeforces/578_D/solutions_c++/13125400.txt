#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll n,m;
char s[100010];
int main(){
    scanf("%I64d%I64d\n",&n,&m);
    scanf("%s",s+1);
    s[0]='-';
    ll ans=n*n*(m-1)-n+1,t=2,k=0;
    for (int i=2; i<=n; i++){
        if (s[i]!=s[i-1] && s[i]==s[i-2]) t++;
        else {ans-=((t-1)*(t-2)/2); t=2;} 
        if (s[i]==s[i-1]) k++;
    }
    ans-=((t-1)*(t-2)/2);
    ans-=k*(n*(m-1)-1);
    printf("%I64d",ans);
}
