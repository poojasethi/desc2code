#include<bits/stdc++.h>
using namespace std;
const int N = 200000 + 5;
typedef long long LL;
char a[N];
int f[N];
int main(){
	int n, m;
	LL ans = 0;
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= n; i++) scanf(" %c", &a[i]);
	for (int i = 1; i <= n; i++)
		if (a[i] != a[i - 1]) ans += n * LL(m - 1);
	f[1] = 0;	
	for (int i = 2; i <= n; i++)
		if (a[i] != a[i - 1]){
			f[i] = 1;
			if (a[i] == a[i - 2]) f[i] = f[i - 1] + 1;
			ans -= f[i];
		}
	printf("%I64d\n", ans);
}
