#include <bits/stdc++.h>
#define maxn 100009
using namespace std;
char s[maxn];
long long sum[maxn];
int n,m;
int main()
{
	scanf("%d%d",&n,&m);
	int cnt=0;
	scanf("%s",s);
	for(int i=1;i<n;i++)
		if(s[i]!=s[i-1])
			cnt++;
	long long ans=1LL*(cnt+1)*(1LL*n*m-n);
	for(int i=1;i<n;i++)
	{
		if(s[i]!=s[i-1])
		{
			sum[i]=1;
			if(i-2>=0&&s[i]==s[i-2])
			{
				sum[i]+=sum[i-1];
			}
		}
		ans-=sum[i];
	}
	printf("%I64d\n",ans);
	//system("pause");
	return 0;
}
