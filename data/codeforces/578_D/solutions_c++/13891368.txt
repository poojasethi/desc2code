const int N=1e5+5;
const int mod=1e9+7;
const int inf=1e9+7;

int n,m,i;
int dp[N][2][2][2];

string str;

int f(int x,int l1,int l2,int l3){

	if(x==n+1)
		return l2==n-1;

	if(l1<x-3 or l2<x-2 or l3<x-2)
		return 0;

	int &r=dp[x][l1&1][l2&1][l3&1];

	if(r!=-1)
		return r;

	int i,newl1,newl2,newl3;

	r=0;

	for(i=1 ; i<=m ; i++){
		newl1=str[x-1]==i?l1+1:l2;
		newl2=str[x]==i?l2+1:max(newl1,l3);
		newl3=str[x+1]==i?l3+1:newl2;
		r+=f(x+1,newl1,newl2,newl3);
	}

	return r;
}

main(){

	ios_base::sync_with_stdio(0);

	cin>>n>>m;

	cin>>str;

	str=' '+str;

	for(i=1 ; i<=n ; i++)
		str[i]=str[i]-'a'+1;

	memset(dp,-1,sizeof dp);

	cout<<f(1,0,0,0);
}
