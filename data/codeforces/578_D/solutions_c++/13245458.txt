#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

long long f[100005][8];
int n,m;
char s[100005];
inline int get(int i,int j) {
	return (i>>j)&1;
}
int main() {
	scanf("%d%d",&n,&m);
	scanf("%s",s+1);
	for (int i=1;i<=n;i++) s[i]-='a';
	s[n+1]=-1;
	for (int i=0;i<m;i++) {
		int t=1;
		if (i==s[1]) t|=2;
		if (i==s[2]||i==s[1]) t|=4;
		f[1][t]++;
	}
	for (int i=2;i<=n;i++) {
		for (int j=0;j<8;j++) {
			int d[4]={i-3+get(j,0),i-2+get(j,1),i-2+get(j,2)};
			for (int k=0;k<m;k++) {
				int d2[4]={0,0,0};
				for (int l=1;l<=3;l++) {
					if (s[i-2+l]==k) d2[l]=d[l-1]+1; else d2[l]=max(d2[l-1],d[l]);
				}
				if (d2[1]>=i-2&&min(d2[2],d2[3])>=i-1) f[i][(d2[1]-(i-2))|((d2[2]-(i-1))<<1)|((d2[3]-(i-1))<<2)]+=f[i-1][j];
			}
		}
	}
	printf("%I64d\n",f[n][0]+f[n][1]+f[n][4]+f[n][5]);
	return 0;
}
