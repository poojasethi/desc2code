#include <stdio.h>
#include <iostream>
#define ll long long

using namespace std;

ll n,m;

ll cntG;

struct Group{
    char symbol;
    int length;
    bool operator==( Group &rhs )const{
        if ( symbol == rhs.symbol && length == rhs.length )
            return true;
        return false;
    }
}g[ 100009 ];

int main(){

    int i;
    char lltmp,ltmp,tmp;

    cntG = 0;

    scanf("%I64d %I64d\n",&n,&m);

    ll deltaansw = 0;
    ll dcnt = 0;

    lltmp = -2;
    ltmp = -1;
    for ( i=1; i<=n; i++){
        scanf("%c",&tmp);
        if ( tmp == ltmp ){
            g[ cntG - 1 ].length++;
        }else{
            g[ cntG ].symbol = tmp;
            g[ cntG ].length = 1;

            cntG++;
        }

        if ( i>1 && tmp != ltmp ){
            if ( tmp == lltmp ){
                dcnt ++;
            }else{
                dcnt = 1;
            }
        }else{
            dcnt = 0;
        }

        deltaansw+=dcnt;

        lltmp = ltmp;
        ltmp = tmp;
    }

    printf("%I64d", cntG * ( m * n - n ) - deltaansw);


}
