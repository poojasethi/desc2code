#!/usr/bin/env python
import sys

MOD = 10**9+7

(n, m) = map(int, raw_input().split())

segments = []
points = []
point_hash = {}

for i in range(m):
	(a, b) = map(int, raw_input().split())
	segments.append((a, b))
	for point in [a, b]:
		if point in point_hash:
			continue
		point_hash[point] = None
		points.append(point)

if n not in point_hash or 0 not in point_hash:
	print(0)
	sys.exit()

points.sort()
for (i, point) in enumerate(points):
	point_hash[point] = i

b_as = {}

for (a, b) in segments:
	a = point_hash[a]
	b = point_hash[b]
	if b not in b_as:
		b_as[b] = [a]
	else:
		b_as[b].append(a)

P = len(points)
sums = {-1: 0, 0: 1}

for point in range(1, P+1):
	sums[point] = sums[point-1]
	if point not in b_as:
		continue
	count = 0
	for a in b_as[point]:
		count += sums[point-1]-sums[a-1]
	sums[point] = (sums[point]+count)%MOD

print((sums[P-1]-sums[P-2])%MOD)
