#include <iostream>
#include <algorithm>
#define pii pair<long long, long long>
#define f first
#define s second

using namespace std;

const int maxm = 100*1000+5;

int n;
int m;
pii b[maxm];
long long dp[maxm];
long long sum[maxm];

int bs(int h, int t, long long tar) {
	if (t == h)
		return h;
	long long mid = (t+h)/2;
	if (tar <= b[mid].f)
		return bs(h, mid, tar);
	else
		return bs(mid+1, t, tar);
}

int main() {
	cin >> n >> m;
	for (int i = 0; i < m; ++i)
		cin >> b[i].s >> b[i].f;
	sort(b, b+m);
	for (int i = 0; i < m; ++i)
		if (!b[i].s)
			dp[i] = 1;
	for (int i = 0; i < m; ++i) {
		int x = bs(0 , m-1, b[i].f);
		if (x)
			dp[i] += sum[x-1];
		x = bs(0 , m-1, b[i].s);
		if (x)
			dp[i] -= sum[x-1];
		dp[i]%=1000000007;
		sum[i] = sum[i-1]+dp[i];
	}
	long long ans = 0;
	for (int i = 0; i < m; ++i)
		if (b[i].f == n) {
			ans += dp[i];
			ans %= 1000000007;
		}
	cout << ans%1000000007 << endl;
	return 0;
}
