#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <queue>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define f first
#define s second
#define mod 1000000007 
#define lim 100001

int n,m,i,s,j;
pair <int,int> A[lim];
priority_queue <pair<int,int> > Q;
int M[lim];

int main()
{
  scanf("%d%d",&n,&m);
  for(i=0;i<m;i++)
    scanf("%d%d",&A[i].s,&A[i].f);
  sort(A,A+m);
  reverse(A,A+m);
  for(i=0;i<m && A[i].f==n;i++)
  {
    M[i]=1;
	Q.push(make_pair(A[i].s,1));
	s=(s+1)%mod;
  }
  for(;i<m;)
  {
    while(!Q.empty() && Q.top().f>A[i].f)
	{
	  s=(s-Q.top().s+mod)%mod;
	  Q.pop();
    }
	for(j=i;i<m && A[i].f==A[j].f;i++)
	  M[i]=s;
    for(i=j;i<m && A[i].f==A[j].f;i++)
	{
	  Q.push(make_pair(A[i].s,M[i]));
	  s=(s+M[i])%mod;
	}
  }
  s=0;
  for(i=0;i<m;i++)
    if(!A[i].s)
	  s=(s+M[i])%mod;
  printf("%d",s);
  return 0;
}
