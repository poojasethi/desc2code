#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
typedef pair<ll,ll> ii;
int main()
{
 ll n,m;
 ll mod=1e9+7;
 cin>>n>>m;
 vector <ii> bus (m);
 vector <ll> end(m); 
 vector <ll> a(m),s(m+1);
 
 for(ll i=0;i<m;i++)
 {
     ll s,t;
     
     cin>>s>>t;
     bus[i]=make_pair(t,s);
     end[i]=t;
 }
 s[0]=0;
 sort(end.begin(),end.end());
 sort(bus.begin(),bus.end());
 for(ll i=0;i<m;i++)
 {
    ll t=lower_bound(end.begin(),end.end(),bus[i].second)-end.begin();
    ll e=lower_bound(end.begin(),end.end(),bus[i].first)-end.begin();
    a[i]=(s[e]-s[t]+mod+(bus[i].second==0?1:0))%mod;
   
    s[i+1]=(mod+a[i]+s[i])%mod;
   
 }
 ll r=0;
 for(ll i=m-1;i>=0;i--)
     if(bus[i].first==n)
         r=(r+a[i]+mod)%mod;
 cout<<r<<endl;
 return 0;
     
 }
