
#include <cstdio>
#include <algorithm>

#define MOD 1000000007

using namespace std ;

struct bus{
    int s, t ;
} ;

bus bs[111111] ;
int dp[111111] ;
int ss[111111] ;

bool cmp(bus b1, bus b2){
    if(b1.t != b2.t)
        return b1.t < b2.t ;
    return false ;
}

int main()
{
    int n, m ;
    scanf("%d%d", &n, &m) ;
    for(int i = 1 ; i <= m ; i++)
        scanf("%d%d", &bs[i].s, &bs[i].t) ;
    sort(bs+1, bs+m+1, cmp) ;
    int ans = 0 ;
    for(int i = 1 ; i <= m ; i++){
        bus tmp ;
        tmp.t = bs[i].s ;
        dp[i] = ss[lower_bound(bs+1, bs+i, bs[i], cmp)-bs-1]-ss[lower_bound(bs+1, bs+i, tmp, cmp)-bs-1]+(!bs[i].s) ;
        //printf("dp[%d] = ss[%d] - ss[%d] + %d\n", i, lower_bound(bs+1, bs+i, bs[i], cmp)-bs-1, lower_bound(bs+1, bs+i, tmp, cmp)-bs-1, (!bs[i].s)) ;
        while(dp[i] < 0)   dp[i] += MOD ;
        dp[i] %= MOD ;
        ss[i] = (ss[i-1]+dp[i])%MOD ;
        if(bs[i].t == n)    ans = (ans+dp[i])%MOD ;
    }
    printf("%d\n", ans) ;
    return 0 ;
}
