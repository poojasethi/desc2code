#include <cstdio>
#include <cstring>
#include <algorithm>
#define N 200500
using namespace std;

const long long MOD = 1000000007LL;

int len, n, m, t, dis[N], ex[N], ey[N], qd[N], qn[N], l[N];
long long S[N];

int main(){

	scanf("%d%d", &len, &m);
	
	t = 0;
	for (int i=0; i<m; i++){
		scanf("%d%d", &ex[i], &ey[i]);
		dis[t++] = ex[i];
		dis[t++] = ey[i];
	}
	dis[t++] = 0;
	dis[t++] = len;
	
	sort(dis, dis + t);
	n = 0;
	for (int i=0; i<t; i++){
		if (i==0 || dis[i] != dis[i-1]) dis[n++] = dis[i];
	}
	
	
	t = 0;
	for (int i=0; i<m; i++){
		ex[i] = lower_bound(dis, dis+n, ex[i]) - dis;
		ey[i] = lower_bound(dis, dis+n, ey[i]) - dis;
		qd[++t] = ex[i];
		qn[t] = l[ey[i]];
		l[ey[i]] = t;
	}
	
	S[0] = 1;
	for (int i=1; i<n; i++){
		S[i] = 0;
		for (int j=l[i]; j; j=qn[j]){
			S[i] += S[i-1];
			if (qd[j]) S[i] += MOD - S[qd[j]-1];
		}
		S[i] = (S[i] + S[i-1]) % MOD;
	}
	
	long long ret = (S[n-1] - S[n-2] + MOD) % MOD;
	printf("%d\n", (int)ret);
	
	return 0;
}
