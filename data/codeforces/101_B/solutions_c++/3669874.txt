//In the Name of God
#include <iostream>
#include <cstring>
#include <string>
#include <vector>
#include <algorithm>
#include <set>
#include <map>

using namespace std;
typedef long long LL;

const LL maxn = 100000 + 10;
LL n , m , x[maxn] , y[maxn], ted , MOD = 1000000007;
map <LL , LL> ans;
map <LL , vector <LL> > st , en;
set <LL> s;

int main(){
  cin >> n >> m;
  for(LL i = 0 ; i < m ; i++){
    cin >> x[i] >> y[i];
    en[y[i]].push_back(i);
    st[x[i]].push_back(i);
    s.insert(x[i]);
    s.insert(y[i]);
  }
  if(en[n].empty()){
    cout << 0 << endl;
    return 0;
  }
  ans[n] = 1;
  ted = LL(en[n].size());
  set <LL> ::iterator it = s.end();
  it--;
  it--;
  for( ; it != s.begin() ; it--){
    //    cout << (*it) << endl;
    ans[*it] = ted;
    for(LL j = 0 ; j < st[*it].size() ; j++){
      ted -= ans[y[st[*it][j]]];
      if(ted < 0)
	ted += MOD;
    }
    ted = (ted + (LL(en[*it].size()) * ans[*it]) % MOD) % MOD;
  }
  ans[*(s.begin())] = ted;
  cout << ans[0] << endl;
  return 0;
}
