#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<map>
#include<list>
using namespace std;
int n,m;
struct bus{
  int l,r;
}s[100005];
map<int,int> M,M2;
int p=0,ans[200005];
list<int> L[200005];
int main(){
  scanf("%d %d",&n,&m);
  M[n]=1;
  M[0]=1;
  for(int i=1;i<=m;i++){
    scanf("%d %d",&s[i].l,&s[i].r);
    if(M.count(s[i].l)==0) M[s[i].l]=1;
    if(M.count(s[i].r)==0) M[s[i].r]=1;
  }
  for(map<int,int>::iterator it=M.begin();it!=M.end();it++){
    M2[(*it).first]=++p;
  }
  for(int i=1;i<=m;i++){
    s[i].l=M2[s[i].l];
    s[i].r=M2[s[i].r];
    L[s[i].r].push_back(s[i].l);
  }
  ans[0]=0;
  ans[1]=1;
  for(int i=2;i<=p;i++){
    ans[i]=0;
    for(list<int>::iterator it=L[i].begin();it!=L[i].end();it++){
      ans[i]+=ans[i-1]-ans[(*it)-1];
      if(ans[i]<0) ans[i]+=1000000007;
      ans[i]%=1000000007;
    }
    ans[i]+=ans[i-1];
    ans[i]%=1000000007;
  }
  printf("%d\n",(ans[p]-ans[p-1]+1000000007)%1000000007);
  return 0;
}
