#include<bits/stdc++.h>
using namespace std;
#define mod 1000000007
#define lo long long int
#define pb push_back
#define mk make_pair
vector<lo>all;
map<lo,vector<lo> >bus_end;
lo ans[500001],fin[500001],s[500001],t[500001];

int main() 
{
    lo i,j,n,m,x,y,cur,tot,beg,ind;
    cin>>n>>m;
    all.pb(0);
    all.pb(n);
    for(i=0;i<m;i++)
    {
        cin>>s[i]>>t[i];
        all.pb(s[i]);
        all.pb(t[i]);
        bus_end[t[i]].pb(s[i]);
    }
    sort(all.begin(),all.end());
    all.resize(unique(all.begin(), all.end()) - all.begin());
    ans[0]=1;
    for(i=1;i<all.size();i++)
    {
        cur=all[i];
        tot=0;
        for(j=0;j<bus_end[cur].size();j++)
        {
            beg=bus_end[cur][j];
            ind=lower_bound(all.begin(),all.end(),beg)-all.begin();
            tot=(tot+ans[i-1]-ans[ind-1])%mod;
        }
        fin[i]=tot;
        ans[i]=(ans[i-1]+tot)%mod;
        
    }
    cout<<(fin[all.size()-1]+mod)%mod<<endl;
    return 0;
}
