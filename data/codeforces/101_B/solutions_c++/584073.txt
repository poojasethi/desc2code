#include <iostream>
#include <queue>
#include <cmath>
#include <vector>
#include <map>
#include <set>
#include <sstream>
#include <cstring>
#include <iomanip>
#include <algorithm>

using namespace std;

typedef long long LL;
typedef pair<LL,LL> P;

#define MAXN 100100
#define MOD 1000000007

LL tt[MAXN];
LL ans[MAXN];
LL sum[MAXN];

int main(){

  string ln;
  LL n,m,s,t,x;
  vector<P> v;

  getline(cin,ln);
  istringstream(ln)>>n>>m;

  for(LL i=0;i<m;i++){
    getline(cin,ln);
    istringstream(ln)>>s>>t;
    v.push_back(P(t,s));
  }
  sort(v.begin(),v.end());

  LL res=0;

  for(LL j=0;j<m;j++){
    s=v[j].second;
    t=v[j].first;
    tt[j]=t;

    int i1=lower_bound(tt,tt+j,s)-tt;
    int i2=lower_bound(tt,tt+j,t)-tt;

    ans[j]=(s==0?1:0);
    ans[j]+=(i2>0?sum[i2-1]:0);
    ans[j]-=(i1>0?sum[i1-1]:0);
    ans[j]%=MOD;

    sum[j]=(j>0?sum[j-1]:0)+ans[j];
    sum[j]%=MOD;

    if(t==n){
      res+=ans[j];
      res%=MOD;
    }
  }

  cout<<(res+MOD)%MOD<<endl;

  return 0;
}
