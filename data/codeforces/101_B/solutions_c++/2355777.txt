#include <iostream>
#include <set>
#include <vector>
#include <utility>
#include <algorithm>
using namespace std;
#define MOD 1000000007
#define PLL pair<long long, long long>
int main(){
   long long n,m,s,t;
   cin>>n>>m;
   vector< PLL > v;
   for(int i=0; i<m; i++){
      cin>>s>>t;
      v.push_back(make_pair(t,s));
   }
   v.push_back(make_pair(0,-1));
   sort(v.begin(),v.end());
   long long sum=0,val=0;
   multiset< PLL > ms;
   for(int i=v.size()-1; i>=0;){
      while(!ms.empty() && v[i].first < -(ms.begin()->first)){
         sum=(MOD+sum-(ms.begin()->second))%MOD;
         ms.erase(ms.begin());
      }
      long long cur=sum;
      for(int j=i; i>=0 && v[i].first==v[j].first;i--){
         val=(v[i].first==n)?1:cur;
         sum=(sum+val)%MOD;
         ms.insert(make_pair(-v[i].second,val));
      }
   }
   cout<<val<<endl;
   return 0;
}
