#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int x,y;
} b[N];
int i,j,k,n,m,t,nn;
int a[N],f[N],h[N];
inline bool cc1(const ww &a,const ww &b) {
	return a.y<b.y;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,m) {
		int x,y;
		scanf("%d%d",&x,&y);
		a[++t]=x,a[++t]=y;
		b[i]=(ww){x,y};
	}
	a[++t]=0,a[++t]=n;
	sort(a+1,a+t+1);
	For(i,1,t) if (i==1||a[i]!=a[nn]) a[++nn]=a[i];
	For(i,1,m) {
		b[i].x=lower_bound(a+1,a+nn+1,b[i].x)-a;
		b[i].y=lower_bound(a+1,a+nn+1,b[i].y)-a;
	}
	sort(b+1,b+m+1,cc1);
	f[1]=1; j=1;
	For(i,1,nn) {
		for (;j<=m&&b[j].y==i;j++) {
			int A=(h[i-1]-h[b[j].x-1])%mo;
			f[i]=(f[i]+A)%mo;
		}
		h[i]=(h[i-1]+f[i])%mo;
	}
	if (f[nn]<0) f[nn]+=mo;
	printf("%d\n",f[nn]);
	return 0;
}
