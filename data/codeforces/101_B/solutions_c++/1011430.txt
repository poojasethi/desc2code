#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
#define MD 1000000007

map<int,int> bt;
pair<int,int> seg[100000];
void addmd(int &x,int y) {
        x=(x+y)%MD;
}
void add(int i,int a) {
        for(i+=5;i<1000100000;i+=i&-i) {
                addmd(bt[i],a);
        }
}
int query(int i) {
        int s=0;
        for(i+=5;i;i-=i&-i) {
                addmd(s,bt.count(i)?bt[i]:0);
        }
        return s;
}
int main() {
        int N,M;
        cin>>N>>M;
        for(int i=0;i<M;i++) {
                cin>>seg[i].second>>seg[i].first;
        }
        sort(seg,seg+M);
        add(0,1);
        for(int i=0;i<M;i++) {
                int s=seg[i].second,t=seg[i].first;
                int c=(query(t-1)-query(s-1)+MD)%MD;
                add(t,c);
        }
        cout<<(query(N)-query(N-1)+MD)%MD;
}
