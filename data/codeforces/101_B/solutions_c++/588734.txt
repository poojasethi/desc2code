#include<stdio.h>
#include<algorithm>
using namespace std;

const int MAXM = 100010; 
const int MOD = 1000000007;
int n,m;

struct seg
{
	int a,b;
	bool operator <(const seg &a)const
	{
		return b < a.b;
	}
}p[MAXM];

int f[MAXM],sum[MAXM];

int find(int x)
{
	int l = 0,r = m;
	while (l <= r)
	{
		int mid = (l+r)/2;
		if (x > p[mid].b)
		{
			l = mid+1;
		}else
			r = mid-1;
	}
	return l-1;
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++)
	{
		scanf("%d%d",&p[i].a,&p[i].b);	
	}
	
	sort(p+1,p+m+1);
	p[0].b = 0;
	sum[0] = 1;	
	
	for (int i=1;i<=m;i++)
	{
		int l = find(p[i].a);
		int r = find(p[i].b);
		
		if (p[i].a == 0) 
			f[i] = sum[r];
		else 				
			f[i] = (sum[r] - sum[l]) % MOD;
		
		sum[i] = (sum[i-1] + f[i]) % MOD;
	}
	int ans = 0;
	for (int i=1;i<=m;i++)
	if (p[i].b == n)
		ans = (ans + f[i]) % MOD;
		
	if (ans < 0) ans += MOD; 
	printf("%d\n",ans);
}
