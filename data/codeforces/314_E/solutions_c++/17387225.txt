#include<cstdio>
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;

int n,m;
unsigned int ans;
unsigned int f[200010];
char a[100010];

int main()
{int i,j,S;
 
 scanf("%d",&n);
 scanf("%s",a+1);
 if((n&1) || n-m-n<0)
 {	printf("%d\n",0);
 	return 0;
 }
 S=n;
 f[S]=1;
 for(i=1;i<=n;i++)
 {	if(a[i]!='?')
 	{	m++;
 		if(!(i&1))
 		{	S--;
 			f[S-1]=0;
 		}
	 	continue;
	}
	for(j=S+min((i+1)/2,(n/2+1)/2);j>S;j--) f[j]+=f[j-1];
	if(i&1)
	{	f[S]=0;
		S++;
	}
 }
 ans=f[S];
 for(i=1;i<=(n-m-m)/2;i++) ans=ans*25;
 printf("%u\n",ans);
 return 0;
}
