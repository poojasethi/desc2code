#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define MAXN 100005
using namespace std ;
typedef unsigned uint ;

int N ;
int dp[MAXN], num ;
char ss[MAXN] ;

int main()
{
	int i, j, half, to ;
	scanf("%d", &N) ;
	scanf("%s", ss) ;
	half = (N>>1) ;
	dp[1] = 1 ;
	if(N&1)
	{
		printf("0\n") ;
		return 0 ;
	}
	for(i = 1; i <= N; i ++)
	{
		if(ss[i-1] == '?')
		{
			to = max(1, i-half+1) ;
			for(j = i/2+1; j >= to; j --)
				dp[j] += dp[j-1] ;
		}
		else num ++ ;
	}
	uint ans = dp[N/2] ;
	for(i = 1; i <= half-num; i ++)
		ans *= 25 ;
	printf("%u\n", ans) ;
	return 0 ;
}
