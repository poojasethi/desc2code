#include <stdio.h>

#define u32 unsigned int

#define MAXN 100005

int n;
char s[MAXN];
u32 f[2][MAXN];

int main()
{
	scanf("%d",&n);
	if (n&1) return puts("0"),0;
	scanf("%s",s+1);
	int i,j;
	int cnt=n/2;
	for (i=1;i<=n;i++) cnt-=s[i]!='?';
	if (cnt<0) return puts("0"),0;
	u32 *now=f[0],*pre=f[1];
	now[0]=1;
	for (i=1;i<=n;i++) {
		u32 *tmp=now;now=pre;pre=tmp;
		int lim=n-i<i ? n-i : i;
		if (s[i]=='?') {
			for (j=lim;j>=0;j-=2) now[j]=pre[j-1]+pre[j+1];
		} else {
			for (j=lim;j>=0;j-=2) now[j]=pre[j-1];
		}
		int lim1=n-(i-1)<i-1 ? n-(i-1) : i-1;
		for (j=lim1;j>0;j-=2) pre[j]=0;
	}
	u32 ans=now[0];
	while (cnt--) ans*=25;
	printf("%u\n",ans);
}
