#include<cstdio>
using namespace std;
int main()
{
	static unsigned int g[100005],*f=g+3;
	int n;
	scanf("%d",&n);
	if(n&1)
		return puts("0"),0;
	f[0]=1;getchar();
	for(int i=1;i<=n;++i)
		if(getchar()=='?')
			for(int j=i&1;j<=i;j+=2)
				f[j]=f[j-1]*25+f[j+1];
		else
			for(int j=i&1;j<=i;j+=2)
				f[j]=f[j-1];
	printf("%u\n",f[0]);
	return 0;
}
