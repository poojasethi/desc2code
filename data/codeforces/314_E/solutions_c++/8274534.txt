#include <cstdio>
const int N=1e5+5;
char s[N];
int n,m,q;
unsigned f[N>>1];
int main(){
	scanf("%d%s",&n,s+1);
	if(n&1) return puts("0"),0;
	m=n>>1;
	f[0]=1;
	for(int i=1;i<=n;i++) if(s[i]=='?')
		for(int j=i>>1;j&&j>=i-m;j--) f[j]+=f[j-1];
	else q++;
	unsigned ans=f[m];
	for(int i=0;i<m-q;i++) ans*=25;
	printf("%u\n",ans);
}
