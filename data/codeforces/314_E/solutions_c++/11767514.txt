#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
const int SIZEN=100010;
unsigned int f[2][SIZEN];
char s[SIZEN];
int n;
void work(void){
	int o,w,x,y;
	f[0][0]=1;
	for(int i=1;i<=n;i++){
		o=i&1;
		w=o^1;
		x=min(i,n-i);
		if(o==0) y=2;
		else y=1;
		if(s[i]=='?'){
			f[o][0]=f[w][1];
			for(int j=y;j<=x;j+=2){
				f[o][j]=f[w][j-1]*25+f[w][j+1];
			}
		}
		else{
			f[o][0]=0;
			for(int j=y;j<=x;j+=2){
				f[o][j]=f[w][j-1];
			}
		}
	}
	o=n&1;
	printf("%u\n",f[o][0]);
}
void read(void){
	scanf("%d",&n);
	scanf("%s",s+1);
}
int main(){
	//freopen("t1.in","r",stdin);
	read();
	work();
	return 0;
}
