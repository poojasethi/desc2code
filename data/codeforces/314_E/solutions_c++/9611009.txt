#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
#define N 110000
int n,m;
unsigned int dp[N];
char str[N];
int main(){
    scanf("%d",&n);
    scanf("%s",str+1);
    for(int i=1;i<=n;i++)
      if(str[i]!='?')m++;
    if(n&1 || m>n/2){
      printf("0\n");
      return 0;
    }
    dp[0]=1;
    for(int i=1;i<=n;i++){
      int l=max(0,i-n/2),r=i/2;
      if(str[i]=='?')
        for(int j=r;j>=l;j--)dp[j]+=dp[j-1];
      dp[r+1]=0;
    }
    for(int i=1;i<=n/2-m;i++)dp[n/2]*=25;
    cout<<dp[n/2]<<endl;
    return 0;
}
