#include <cstdio>
using namespace std;
const int N=100004;
int n;
unsigned f[N];
char s[N];
int main()
{
	int i,j; scanf("%d%s",&n,s+1);
	if (n%2)
	{
		printf("0\n");
		return 0;
	}
	f[1]=1; for (i=2;i<=n+1;i++) f[i]=0;
	for (i=1;i<=n;i++)
	    if (s[i]=='?') for (j=i%2;j<=i;j+=2) f[j+1]=f[j]*25+f[j+2];
		else for (j=i%2;j<=i;j+=2) f[j+1]=f[j];
	printf("%u\n",f[1]);
	return 0;
}
