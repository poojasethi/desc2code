#include<cstdio>
unsigned int sum[50010];
int n;
char k[100100];
int main()
{
    int i,j,cnt=0;
    scanf("%d\n",&n);
    gets(k+1);
    for(i=1;i<=n;++i)
        if(k[i]!='?')
            ++cnt;
    if(n&1||cnt>n/2)
    {
        puts("0");
        return 0;
    }
    sum[0]=1;
    for(i=1;i<=n;++i)
        if(k[i]=='?')
            if(i<=n/2)
                for(j=i>>1;j;--j)
                    sum[j]+=sum[j-1];
            else
                for(j=i>>1;i-(n>>1)<=j;--j)
                    sum[j]+=sum[j-1];
    for(i=n/2-cnt;i;--i)
        sum[n>>1]*=25;
    printf("%u\n",sum[n>>1]);
    return 0;
}
