#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
int n,m,z,x,y,o,w;
unsigned int f[2][100005];
char s[100005];
int main(){
	//freopen("cf314E.in","r",stdin);freopen("cf314E.out","w",stdout);
	scanf("%d\n",&n);
	scanf("%s",s+1);
	f[0][0]=1;
	for (int i=1;i<=n;++i) {
		o=i&1;w=(i+1)&1;x=min(i,n-i);
		if (o==0) y=2;
		else y=1;
		if (s[i]=='?'){
			f[o][0]=f[w][1];
			for (int j=y;j<=x;j+=2) f[o][j]=f[w][j-1]*25+f[w][j+1];
		}
		else {
			f[o][0]=0;
			for (int j=y;j<=x;j+=2) f[o][j]=f[w][j-1];
		}
	}
	o=n&1;
	printf("%u",f[o][0]);
	return 0;
}
