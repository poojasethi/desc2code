#include <cstdio>
#include <algorithm>

using namespace std;

int n;
unsigned f[100010];

int main()
{
	scanf("%d\n", &n);
	if (n & 1)
	{
		puts("0");
		return 0;
	}
	int m = n / 2, chn = m;
	f[0] = 1;
	for (int i = 1; i <= n; ++i)
		if (getchar() == '?')
		{
			for (int j = i / 2; j >= m - (n - i) && j; --j)
				f[j] += f[j - 1];
		}
		else --chn;
	while (chn-- > 0) f[m] *= 25;
	printf("%u\n", f[m]);
}
