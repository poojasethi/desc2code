#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;
typedef unsigned U;
U f[100005];
int sum[100005];
int i,j,i1,i2,n;
char S[100005];

int main(){
  scanf("%d\n",&n);
  scanf("%s",S+1);
  if (n&1) {puts("0");return 0;}
  for (i=n;i>0;i--)
    sum[i] = sum[i+1]+(S[i]=='?');
  if (sum[1]<n/2) {puts("0");return 0;}
  f[0] = 1;
  for (i=1;i<=n;i++)
  if (S[i]=='?')
    for (j=i>>1;sum[i+1]>=(n>>1)-j&&j>0;j--)
      f[j] += f[j-1];
  U ans = f[n>>1];
  for (i=1;i<=sum[1]-n/2;i++)
    ans = ans*25U;
  printf("%u\n",ans);
  return 0;
}
