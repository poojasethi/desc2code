#include <cstdio>
#include <cstring>
#include <algorithm>
#define MXN 100010
using namespace std;

char S[MXN];

int N, L, R;
unsigned int dp[MXN * 3];

int main (){
	scanf("%d%s", &N, S);

	dp[L = N+2] = 1; R = L+1;
	for(int i = 0; i < N; i++){
		if(S[i] != '?') --L;
		else{
			dp[++R] = 0;
			dp[++R] = 0;
			for(int j = R-1; j >= L+1; j--)
				dp[j] += dp[j-2] * 25;
			dp[L++] = 0;
		}
		while(R - L >= (N - i)) dp[R--] = 0;
	}
	printf("%u\n", dp[L]);
	//printf("%lf\n", 1. * clock() / CLOCKS_PER_SEC);
}
