#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;

#define maxn 100007

char a[maxn];
unsigned int f[maxn+maxn];
int n, l, r;

int main (){
    long long zz;
    int i, j;

	scanf("%d", &n);
	scanf("%s", a);

	zz=0; l = n+2; r = n+3; f[l] = 1;
	for(i = 0; i < n; i++){
        if (a[i] == '?') {
			r += 2;
			for(j = r-1; j >= l+1; j--) {
			    ++zz;
				f[j] += f[j-2] * 25;
			}
			f[l++] = 0;
        }
        else --l;
		while(r-l >= n-i) f[r--] = 0;
	}
	printf("%u\n", f[l]);
}
