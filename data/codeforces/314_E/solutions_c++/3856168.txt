#include <cstdio>
#define MXN 100010

char S[MXN];
unsigned int N, L, R, dp[MXN * 3];

int main (){
	scanf("%d%s", &N, S);
	dp[L = N+2] = 1; R = L+1;
	for(int i = 0; i < N; i++){
		if(S[i] != '?') --L;
		else{
			R += 2;
			for(int j = R-1; j >= L+1; j--)
				dp[j] += dp[j-2] * 25;
			dp[L++] = 0;
		}
		while(R - L >= (N - i)) dp[R--] = 0;
	}
	printf("%u\n", dp[L]);
}
