#include <cstdio>

const int MAXN = 100005;

int N, M, cnt;
unsigned ans, f[MAXN>>1];
char s[MAXN];

int main()
{
	scanf("%d%s", &N, s + 1); f[0] = 1, M = N >> 1;
	if (N & 1) return puts("0"), 0;
	for (int i = 1; i <= N; i ++)
		if (s[i] == '?')
			for (int j = i >> 1; j && j >= i - M; j --)
				f[j] += f[j - 1];
		else ++ cnt;
	ans = f[M];
	for (int i = 0; i < M - cnt; i ++) ans *= 25;
	printf("%u\n", ans);
	return 0;
}
