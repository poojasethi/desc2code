#include<bits/stdc++.h>
#define rep(i,A,B) for(int i=A;i<=B;i++)
#define dep(i,B,A) for(int i=B;i>=A;i--)
#define mn 100011
using namespace std;
typedef unsigned int u32;
inline u32 powmod(u32 a,u32 b){u32 tmp=1;for(;b;b>>=1){if(b&1)tmp=tmp*a;a=a*a;}return tmp;}
u32 ans,f[mn];int n;char s[mn];
int main(){
    scanf("%d%s",&n,s+1);
    if(n&1)return puts("0"),0;
    f[0]=1;int cnt=n,m=n>>1;
    rep(i,1,n)if(s[i]=='?'){int lim=max(0,i-m);dep(j,i>>1,lim)f[j]+=f[j-1];--cnt;}
    if(cnt>m)puts("0");else printf("%u\n",f[m]*powmod(25,m-cnt));
    return 0;
}
