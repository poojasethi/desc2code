#include <bits/stdc++.h>
#define MAX_N 100010 

using namespace std; 
typedef unsigned ll; 

ll dp[MAX_N]; 
int n,tmp; 
string s; 

int main () 
	{
		cin >> n; 
		cin >> s; 
		int m = n / 2; 

		int tmp = 0; 
		if (n % 2 == 1) 
			{
				cout << 0; 
				return 0; 
			}
			
		dp[0] = 1; 
		for (int i = 1; i <= n; i++) 
			if (s[i-1] == '?') 
				for (int j = i / 2; j && j >= (i - m); j--) dp[j] += dp[j-1]; 
		else tmp++; 
		ll res = dp[m];
		for (int i = 0; i < m - tmp; i++) res *= 25;  
		cout << res; 
		
		return 0; 
	}
