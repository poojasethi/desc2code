#include <cstdio>
const int N = 100000 + 9;
unsigned int f[N],ans = 1;
int n,cnt;
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("314E.in","r",stdin);
	freopen("314E.out","w",stdout);
	#endif
	scanf("%d\n",&n);
	if (n & 1) return puts("0");
	f[0] = 1;
	for (int i = 1; i <= n; ++i) {
		char c = getchar();
		if (c == '?')
			for (int t = i & 1,j = i >> 1; j && t <= n - i; t += 2,--j)
				f[j] += f[j - 1];
		else ++cnt;
	}
	cnt = (n - cnt * 2) / 2;
	for (int i = 1; i <= cnt; ++i) ans *= 25;
	ans *= f[n / 2];
	printf("%u\n",ans);
}
