# include <cstdio>

# define REP(i, n) for (int i = 1; i <= n; i ++)

char s[101000];
int n, c;
unsigned f[101000];
int main () {
	scanf ("%d%s", &n, s + 1);
	int h = n >> 1;
	if (n & 1) return puts ("0"), 0;
	REP (i, n) if (s[i] == '?') ++ c;
	if (c < n / 2) return puts ("0"), 0;
	f[0] = 1;
	REP (i, n) if (s[i] == '?') for (int j = i >> 1; j >= i - h && j; -- j) f[j] += f[j - 1];
	int ans = f[h];
	REP (i, c - n / 2) ans *= 25;
	printf ("%u\n", ans);
	return 0;
}
