#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 2e5 + 5;

unsigned n, a[N], f[N];

int gc(){char c; while(c=getchar(), c<=' '); return c=='?';}
int min(int x, int y){return x<y?x:y;}

int main()
{
	scanf("%d", &n);
	if(n&1) {puts("0"); return 0;}
	for(int i=1; i<=n; i++) a[i] = gc();
	
	int ze = n;
	f[n] = 1;
	
	for(int i=1; i<=n; i++)
	{
		if(a[i])
		{
			int ep = ze + min(i-1, n-i+1);
			f[ep+1] = f[ep+2] = 0;
			for(int j=ep; j>=ze; j--) f[j+2] += f[j] * 25;
			++ze;
		}
		else f[--ze] = 0;
	}
	printf("%u\n", f[ze]);
	
	return 0;
}
