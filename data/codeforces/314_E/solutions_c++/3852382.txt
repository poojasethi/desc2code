#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
char c[120000];
int i,j,k,l,n,m,tot,q,r;
unsigned int ans,f[120000];
int main(){
	scanf("%d",&n);
	scanf("%s",c+1);
	for(i=1;i<=n;++i)q+=1-(c[i]=='?');
	if(n&1 || q>n/2){
		printf("0\n");
		return 0;
	}
	f[0]=1;
	for(i=1;i<=n;++i){
		l=max(0,n/2-(n-i+1));
		r=i/2;
		if(c[i]=='?')for(j=r;j>=l;--j)f[j+1]+=f[j];
		f[r+1]=0;
	}
	ans=f[n/2];
	for(i=1;i<=n/2-q;++i)ans*=25;
	printf("%u\n",ans);
}
