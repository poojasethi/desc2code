#include <iostream>
#include <map>
#include <algorithm>
#define LL long long
using namespace std;
int n, b, k;
map<int, int> M;
int main(){
  ios_base::sync_with_stdio(0);
  cin>>k>>b>>n;
  M[0]++;
  bool back=false;
  int want=b;
  if(want==0) back=true;
  if(want==k-1) want=0;
  int mod=k-1;
  LL ans=0;
  LL sum=0;
  int zeros=0;
  for(int i=0; i<n; i++){
    LL num;
    cin>>num;
    if(num==0){
      LL diff=zeros+1;
      zeros++;
      if(back) ans+=diff;
      if(!back && want==0) ans-=diff;
    }else{
      zeros=0;
    }
    sum+=num;
    sum%=mod;
    LL left=sum+mod-want;
    left%=mod;
    if(!back) ans+=M[(int)left];
    M[sum]++;
  }
  cout<<ans;
  return 0;
}
