#include<stdio.h>
#include<map>
using namespace std;

typedef __int64 lld;
const int maxn=100005;
map<lld,lld>p;
lld k,b,n;
int a[maxn];

int main(){
    int i,j;
    lld ans;
    while(scanf("%d%d%d",&k,&b,&n)!=EOF){
        p.clear();
        for(i=0;i<n;i++){
            scanf("%d",&a[i]);
        }
        lld sum=0;
        lld s=0,z=0;
        ans=0;
        p[0]=1;
        for(i=0;i<n;i++){
            sum=(sum+a[i])%(k-1);
            if(a[i]==0){
                s++;
                z+=s;
            }else
                s=0;
            lld t=(sum-b+k-1)%(k-1);
            ans+=p[t];
            p[sum]++;
        }
        if(b==0)ans=z;
        if(b==k-1) ans-=z;
        printf("%I64d\n",ans);
    }
    return 0;
}
