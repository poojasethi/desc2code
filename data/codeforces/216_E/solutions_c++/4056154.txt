#include <algorithm>
#include <stdio.h>
#include <map>
using    namespace std;
typedef  long long llint;

int k,b,n;
int sum,cnt;
map<int,int>hash;

int main()
{
  llint ans=0;
  scanf("%d%d%d",&k,&b,&n);
  hash[0]=1;
  for(int i=0,x ; i<n ; i++)
  {
    scanf("%d",&x);
    if(x==0) cnt++;
    else     cnt=0;
    sum+=x;
    sum%=(k-1);
    int tmp=(sum-b+k-1)%(k-1);
    if(b==k-1)
      ans+=(llint)hash[tmp]-cnt;
    else if(b==0)
      ans+=(llint)cnt;
    else
      ans+=(llint)hash[tmp];
    hash[sum]++;
  }
  printf("%I64d\n",ans);
  return 0;
}
