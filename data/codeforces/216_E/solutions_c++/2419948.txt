#include <iostream>
#include <cstdio>
#include <map>

using namespace std;

int k, b, n;
long long sz, s;

int main()
{
	scanf("%d%d%d", &k, &b, &n);
	int zeros = 0;
	int t = 0;
	map<int, int> cnt;
	cnt[0] = 1;
	for (int i = 0; i < n; ++i) {
		int x;
		scanf("%d", &x);
		if (x != 0) zeros = 0;
		else {
			++zeros;
			sz += zeros;
		}
		t = (t + x) % (k - 1);
		s += cnt[((t - b) + (k - 1)) % (k - 1)];
		++cnt[t];
	}
	if (b == 0) cout << sz;
	else if (b == k - 1) cout << s - sz;
	else cout << s;
	cout << endl;
}
