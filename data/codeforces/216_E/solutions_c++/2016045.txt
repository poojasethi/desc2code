#include <iostream>
#include <map>
using namespace std;

map<int,int> m;
long long ans,sum[100001];
int k,b,n,x,cnt[100001];

int main(){
    cin>>k>>b>>n;
    if(!b){
        for(int i=1;i<=n;i++){
            cin>>x;
            cnt[0]=(!x)?(cnt[0]+1):0;
            ans+=cnt[0];
        }
    }else{
        k--,m[0]=1;
        for(int i=1;i<=n;i++){
            cin>>x;
            cnt[i]=(!x)?(cnt[i-1]+1):0;
            sum[i]=(sum[i-1]+x)%k;
            int tmp=(sum[i]-b+k)%k;
            ans+=m[tmp];
            m[sum[i]]++;
            if(tmp==sum[i]) ans-=cnt[i];
        }
    }
    cout<<ans<<endl;
}
