#include<algorithm>
#include<iostream>
#include<cstdio>
using namespace std;
int b[234567];
bool t[234567];
int main()
{
	int n,k,a,m,beg,end,ans=0;
	cin>>n>>k>>a>>m;
	for(int i=1;i<=m;i++)scanf("%i",b+i);
	beg=1,end=m;
	while(beg<=end)
	{
		int mid=(beg+end)/2,sum=0,now=0;
		for(int i=1;i<=n;i++)t[i]=false;
		for(int i=1;i<=mid;i++)t[b[i]]=true;
		for(int i=1;i<=n;i++)
		{
			if(now==a)sum++,now=0;
			else if(t[i])now=0;
			else now++;
		}
		sum+=now/a;
		if(sum<k)end=mid-1;
		else ans=max(ans,mid),beg=mid+1;
	}
	if(ans!=m)cout<<ans+1<<endl;
	else cout<<-1<<endl;
	return 0;
}
