#include <bits/stdc++.h>
#define fi first
#define se second

using namespace std;

typedef pair <int,int> ii;

int N,K,A,elde,dongu;
set <ii> st;

int main(){
	scanf("%d %d %d",&N,&K,&A);
	A++;
	elde=(N+1)/A;
	
	st.insert(ii(N+1,0));
	
	scanf("%d",&dongu);
	
	for(int i=1;i<=dongu;i++){
		int a;
		scanf("%d",&a);
		
		set <ii> :: iterator it;
		it=st.lower_bound(ii(a,0));
		ii x=*it;
		st.erase(it);
		elde-=(x.fi-x.se)/A;
		elde+=(x.fi-a)/A;
		elde+=(a-x.se)/A;
		st.insert(ii(x.fi,a));
		st.insert(ii(a,x.se));
		
		if(elde<K){
			printf("%d\n",i);
			return 0;
		}
	}
	
	printf("-1\n");

	return 0;
}
