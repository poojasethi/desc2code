#include <iostream>
#include <set>

using namespace std;

set<int> s;
set<int>::iterator it;

inline int func(int l, int r, int sz) {
	return (r - l) / (sz + 1);
}

int main() {
	int cnt, l, r, x, n, k, a, m, ans = -1;

	ios::sync_with_stdio(0);

	cin >> n >> k >> a;
	cin >> m;

	s.insert(0);
	s.insert(n+1);

	cnt = func(0, n+1, a);

	for (int i = 1; i <= m; i++) {
		cin >> x;
		s.insert(x);

		it = s.find(x);
		++it;
		r = *it;
		--it; --it;
		l = *it;

		cnt -= func(l, r, a);
		cnt += func(l, x, a);
		cnt += func(x, r, a);

		if (cnt < k) {
			ans = i;
			break;
		}
	}

	cout << ans << endl;

	return 0;
}
