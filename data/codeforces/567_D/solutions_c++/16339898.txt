#include<iostream>
#include<cstdio>
#include<cstring>
#include<set>
using namespace std;
int cu,n,k,a;
set<int>b;
void df(int x){
    set<int>::iterator it=b.upper_bound(x);
    int fa=*it,fb=*(--it);
    cu=cu-(fa-fb)/(a+1)+(fa-x)/(a+1)+(x-fb)/(a+1);
    return;
}
int main(){
    int m,x,j=-1;
    cin>>n>>k>>a>>m;
    b.insert(0);
    b.insert(n+1);
    cu=(n+1)/(a+1);
    for(int i=1;i<=m;i++){
        cin>>x;
        df(x);
        if(cu<k){j=i;break;}
        b.insert(x);
    }
    printf("%d\n",j);
    return 0;
}
