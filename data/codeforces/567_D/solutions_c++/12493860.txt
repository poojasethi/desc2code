#include<bits/stdc++.h>
using namespace std;
set <int> s;
int main()
{
int n,m,k,a,x,tot=0,stn=-1;cin>>n>>k>>a>>m;
    s.insert(0);s.insert(n+1);
    std::set<int>::iterator l,r;
    tot=(n+1)/(a+1);
    for (int i=0;i<m;i++)
    {
        int x,u=0,g=0;
        cin>>x;
        r=s.upper_bound(x);
        l=s.lower_bound(x);
        l--;
        int q=*r,w=*l;
        u=q-w-1;u=(u+1)/(a+1);
        s.insert(x);
        int p=(x-w)/(a+1)+(q-x)/(a+1);
        tot-=(u-p);
        if (tot<k && stn==-1)
            stn=i+1;
    }
    cout<<stn;
    return 0;
}
