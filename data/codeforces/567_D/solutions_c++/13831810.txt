#include <stdio.h>
#include <set>

using namespace std;

int a;

int f(int x) {
	return (x+1)/(a+1);
}

int main (void) {
	int n, k;
	scanf ("%d%d%d", &n, &k, &a);
	int m;
	scanf ("%d", &m);
	set<int> s;
	s.insert(0);
	s.insert(n+1);
	int r = f(n);
	int ans = -1;
	for (int i = 0; i < m; i++) {
		int x;
		scanf ("%d", &x);
		set<int>::iterator it;
		it = s.lower_bound(x);
		int end = *it;
		it--;
		int beg = *it;
		r = r - f(end-1-beg) + f(x - 1 - beg) + f(end - 1 - x);
		if (r < k) {
			ans = i+1;
			break;
		}
		s.insert(x);
	}
	printf ("%d\n", ans);
}
