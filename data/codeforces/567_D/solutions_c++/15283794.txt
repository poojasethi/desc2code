#include <bits/stdc++.h>
using namespace std;
int len(int x,int a){
	return (int) ((x+1)/(a+1));
}
int main(){
	int tot=0,m,x,n,k,a,i,j,org,diff,d1,d2;cin>>n>>k>>a;
	cin>>m;
	set<int> s;s.insert(0),s.insert(n+1);
	tot = len(n,a);
	set<int>::iterator it;
	for(i=1;i<=m;++i){
		cin>>x;
		s.insert(x);
		it = s.find(x);--it;
		int prev= *it;++it,++it;
		int next= *it;
		diff= len(next-prev-1,a)- len(next-x-1,a)-len(x-prev-1,a);
		assert(diff>=0);
		tot-=diff;
		if(k>tot){
			cout<<i<<endl;return 0;
		}
	}
	cout<<-1<<endl;
}
