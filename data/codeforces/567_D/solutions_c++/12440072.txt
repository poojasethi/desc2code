#include <bits/stdc++.h>
using namespace std;
int main(){
	int n, k, a;
	cin>>n>>k>>a;
	int m;
	cin>>m;
	set<int> cov, negcov; 
	cov.insert(0), cov.insert(n + 1);
	negcov.insert(0), negcov.insert(-n - 1);
	int x;
	int ans = (n + 1) / (a + 1);
	for (int i = 0; i < m; i++) {
		cin>>x;
		int next = (*cov.upper_bound(x));
		int prev = (*negcov.upper_bound(-x));
		ans -= (next + prev) / (a + 1);
		ans += (x + prev) / (a + 1);
		ans += (next - x) / (a + 1);
		if(ans < k) {
			cout<<i + 1<<endl;
			return 0;
		}
		cov.insert(x);
		negcov.insert(-x);
	}
	cout<<-1<<endl;
}
