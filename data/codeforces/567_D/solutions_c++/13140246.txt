#include <bits/stdc++.h>

using namespace std;

int n, k, a, x, m, ans = -1;
set<int> s;

int get(int m)
{
	return m/a;
}

int main()
{
	scanf("%d %d %d", &n, &k, &a);
	a++;
	s.insert(0);
	s.insert(n+1);
	scanf("%d", &m);
	
	int sum = get(n+1);
	
	for(int u = 1; u <= m; u++)
	{
		scanf("%d", &x);
		s.insert(x);
		set<int>::iterator it = s.find(x);
		it--;
		int L = *it;
		it++; it++;
		int R = *it;
		int La = get(x - L), Ra = get(R - x);
		sum += La + Ra - get(R-L);
		if(sum < k && ans == -1) ans = u;
	}
	printf("%d\n", ans);

	return 0;
}
