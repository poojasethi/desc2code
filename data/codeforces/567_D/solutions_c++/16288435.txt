#include<stdio.h>
#include<set>
#include<iostream>
using namespace std; 
set<int>p;
set<int>::iterator it;
int ship[200005];
int main()
{
	int n,k,a,m,sum,x=-1;
	scanf("%d%d%d",&n,&k,&a);
	sum=(n+1)/(a+1);
	p.insert(0);
	p.insert(n+1);
	scanf("%d",&m);
	for(int i=1;i<=m;i++)	scanf("%d",&ship[i]);
	for(int i=1;i<=m;i++)	
	{
		it=p.lower_bound(ship[i]);
		int t1=*it;
		it--;
		int t2=*it;
		int t=ship[i];
		sum=sum-(t1-t2)/(a+1)+(t-t2)/(a+1)+(t1-t)/(a+1);
		p.insert(ship[i]);
		if(sum<k)
		{
			x=i;
			break;
		}
	}
	printf("%d\n",x);
}
