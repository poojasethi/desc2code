#include<bits/stdc++.h>

using namespace std;


set<int> num;

int main()
{
	int n,k,a,m;
	cin >> n>> k >>a>>m;
	int ships= (n+1)/(a+1);
	num.insert(0);
	num.insert(n+1);
	for(int i=0;i<m;i++)
	{
			int id;
			cin >> id;
			int s= *(--num.lower_bound(id));
			int e= *(num.lower_bound(id));
			if(e==id)
				continue;
			ships= ships+ (id-s)/(a+1)+ (e-id)/(a+1)- (e-s)/(a+1);
			num.insert(id);
			if(ships<k)
				{
					cout<< i+1;
					return 0;
				}
	}	
	cout<< -1;
}
