#include <bits/stdc++.h>
using namespace std;
set <int> s;
set <int> :: iterator it;

int n,k,a,x,m,ma,t,f;

int main()
{
    cin>>n>>k>>a>>m;

    s.insert(0);s.insert(n+1);
    ma=(n+1)/(a+1);
    for(int i=0;i<m;i++) {cin>>x;
        it=s.lower_bound(x);
        t=*it-1; it--; f=*it+1;
        ma+=(x-f+1)/(a+1)+(t-x+1)/(a+1)-(t-f+2)/(a+1);
        if (ma<k) {cout<<i+1; return 0;}
        s.insert(x);
    }
    cout<<-1;
}
