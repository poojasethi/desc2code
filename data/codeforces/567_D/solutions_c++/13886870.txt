#include <iostream>
#include <set>

using namespace std;

int main()
{
    int n, k, a, m, x;
    cin>>n>>k>>a>>m;
    set <int> s;
    s.insert (0);
    s.insert (n+1);
    int ans = (n+1)/(a+1);
    int cnt = 0;
    while (m--) {
        cnt++;
        cin>>x;
        set <int> :: iterator it = s.upper_bound(x);
        int up = *(it--);
        int lo = *(it);
        ans -= (up-lo)/(a+1) - (up-x)/(a+1) - (x-lo)/(a+1);
        if (ans < k) {
            cout<<cnt;
            return 0;
        }
        s.insert(x);
    }

    cout<<"-1";
    return 0;
}
