#include<iostream>
#include<stdio.h>
#include<set>
using namespace std;
set<int>sp;
int main(){
	int n,k,a,m,i,sum,x,flag=0;
	scanf("%d%d%d%d",&n,&k,&a,&m);
	set<int>::iterator it;
	sp.clear();
	sp.insert(0);
	sp.insert(n+1);
	sum=(n+1)/(a+1);
	for(i=1;i<=m;i++){
		scanf("%d",&x);
		sp.insert(x);
		it=sp.find(x);
		int lef=*(--it);
		it++;
		int rig=*(++it);
		sum-=(rig-lef)/(a+1);
		sum+=(rig-x)/(a+1);
		sum+=(x-lef)/(a+1);
		if(sum<k){
			flag=1;
			break;
		}
	}
	if(flag)cout<<i<<endl;
	else cout<<-1<<endl;
	return 0;
}
