#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <set>
using namespace std;
int main()
{
	int n, k, a;
	scanf("%d%d%d", &n, &k, &a);
	a ++;
	int m;
	scanf("%d", &m);
	set<int> ss;
	ss.insert(0);
	ss.insert(n + 1);
	int cnt = (n + 1) / a, i;
	for (i = 1; i <= m; ++ i)
	{
		int x;
		scanf("%d", &x);
		int L = *(-- ss.lower_bound(x));
		int R = *ss.upper_bound(x);
		//printf("*** %d %d\n", L, R);
		ss.insert(x);
		cnt -= (R - L) / a;
		cnt += (x - L) / a;
		cnt += (R - x) / a;
		if (cnt < k) break;
	}
	if (i > m) i = -1;
	printf("%d\n", i);
}
