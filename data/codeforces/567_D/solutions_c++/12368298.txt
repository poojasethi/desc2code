#include <cstdio>
#include <cstring>
#include <set>
using namespace std;

set <int> T;
int n,k,a,cnt;

inline int Calc(int x) {return (x+1)/(a+1);}

int main()
{
	//freopen("test.in","r",stdin);
	scanf("%d%d%d",&n,&k,&a);
	int Time,flag=-1;
	scanf("%d",&Time);
	T.insert(0);T.insert(n+1);
	cnt=Calc(n);
	for (int i=1;i<=Time && flag==-1;i++)
	{
		int x,l,r;
		scanf("%d",&x);
		l=*(--T.lower_bound(x));
		r=*T.upper_bound(x);
		cnt-=Calc(r-l-1);
		cnt+=Calc(x-l-1);
		cnt+=Calc(r-x-1);
		T.insert(x);
		if (cnt<k) flag=i;
	}
	printf("%d\n",flag);
	return 0;
}
