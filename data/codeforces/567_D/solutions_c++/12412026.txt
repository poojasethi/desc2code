#include <iostream>
#include <set>
using namespace std;
int n,k,a;

int maxx(int lo, int hi){
	return (hi-lo )/(a+1);
}
int main(){
	cin>>n>>k>>a;
	
	int mx = (n+1)/(a+1);
	
	set<int> s;
	s.insert(0);
	s.insert(n+1);
	
	int m;
	cin>>m;
	
	int t,lo,hi;
	set<int>::iterator it;
	for(int i = 1; i<=m;i++){
		cin >> t;
		it = s.lower_bound(t);
		hi = *(it--);
		lo = *it;
		s.insert(t);
		mx = mx + maxx(lo, t) + maxx(t, hi) - maxx(lo,hi);
//		cout<<lo<<" "<<t<<" "<<hi<<endl;
//		cout<<i << " " << mx << endl;
		if(mx < k){
			cout<<i<<endl;
			return 0;
		}
	}
	cout<<-1<<endl;
	return 0;
}
