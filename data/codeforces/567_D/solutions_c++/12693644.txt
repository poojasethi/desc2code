#include <bits/stdc++.h>
using namespace std;
set <int> myset;
set <int> :: iterator it;
set <int> :: iterator it1;
int main() {
	int m,l,sh,si,s,to,i=0,le,ri;
	cin>>l>>sh>>si;
	cin>>s;
	to = (++l)/(++si);
	myset.insert(0);
	myset.insert(l);
	while(s-- && to>=sh) {
		cin>>m;
		it = myset.upper_bound(m);
		ri = *it;
		le = *--it;
		to -= (ri-le)/si;
		to += (ri-m)/si+(m-le)/si;
		myset.insert(m);
		i++;
	}
	printf("%d\n",to<sh?i:-1);
	return 0;
}
