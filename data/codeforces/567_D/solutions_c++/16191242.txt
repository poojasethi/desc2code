#include <iostream>
#include <set>
using namespace std;
int n, k, a, m;
set <int> tr;
set <int> ::iterator it;
int main()
{
	cin >> n >> k >> a; a++;
	tr.insert(0); tr.insert(n + 1);
	cin >> m;
	int p = n / a; if (n % a == a - 1) p ++;
	for (int i = 1; i <= m; ++ i)
	{
		int x; cin >> x;
		it = tr.upper_bound(x);
		int r = *it; it --; 
		int l = *it;
		p -= (r - l) / a;
		p += (x - l) / a + (r - x) / a;
		if (p < k) {cout << i; return 0;}
		tr.insert(x);
	}
	cout << -1;
	return 0;
}
