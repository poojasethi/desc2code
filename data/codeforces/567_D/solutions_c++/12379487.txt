#include<bits/stdc++.h>
using namespace std;
set<int> s;
int n,k,a,m,x,v;
int main(){
    scanf("%d%d%d",&n,&k,&a);
    scanf("%d",&m);
    v=(n+1)/(a+1);
    s.insert(0);
    s.insert(n+1);
    for (int i=1; i<=m; i++){
        scanf("%d",&x);
        s.insert(x);
        int l=*(--s.lower_bound(x)),r=*s.upper_bound(x); 
        v-=(r-l)/(a+1);
        v+=(x-l)/(a+1);
        v+=(r-x)/(a+1);
        if (v<k){
            printf("%d",i);
            return 0;
        }
    }
    printf("-1");
}
