#include <iostream>
#include <set>
#include <stdio.h>
using namespace std;

int main(void)
{
	int n,k,a,m,x;
	set<int> S;
	scanf("%d%d%d%d",&n,&k,&a,&m);
	int ans = (n+1)/(a+1);
	S.insert(0);
	S.insert(n+1);
	for(int i = 0;i < m;i++){
		scanf("%d",&x);
		int c=*S.lower_bound(x);
        int d=*--S.lower_bound(x);
        S.insert(x);
        ans -= (c-d)/(a+1);
        ans += (c-x)/(a+1)+(x-d)/(a+1);
        if(ans < k){
        	printf("%d\n",i+1);
        	return 0;
        } 
	}
	cout << "-1" << endl;
}
