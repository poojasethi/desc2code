#include<bits/stdc++.h>
using namespace std;
int n, k, a, m, x, mid, ans = -1, space, last, b[200005];
bool used[200005];
int main(int argc, char **argv) {
	scanf("%d%d%d%d", &n, &k, &a, &m);
	for(int i=0 ; i<m ; ++i) scanf("%d", b+i);
	int l = 0, r = m-1;
	while(l<=r){
		mid = (l+r) >> 1 , space = 0, last=0;
		memset(used, 0, n);
		for(int i=0 ; i<=mid; ++i) used[b[i]-1] = 1;
		for(int i=0; i<n ; ++i)
			if(used[i]) space += (i-last+1)/(a+1) , last = i+1;
		space += (n-last+1)/(a+1);
		if(k > space) r = mid-1, ans = mid+1;
		else l = mid+1;
	}
	cout<<ans<<endl;
	return 0;
}
