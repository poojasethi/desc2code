#include <cstdio>
#include <cstring>
#include <set>

using namespace std;

int main() {
  set<int> s;
  int n, k, a, max;
  scanf("%d%d%d", &n, &k, &a);
  s.insert(0);
  s.insert(n + 1);
  max = (n + 1) / (a + 1);
  int m;
  scanf("%d", &m);
  for (int i = 0; i < m; i++) {
    int t;
    scanf("%d", &t);
    set<int>::iterator it1 = s.lower_bound(t);
    set<int>::iterator it2 = it1;
    it2--;
    max -= (*it1 - *it2) / (a + 1);
    max += (t - *it2) / (a + 1) + (*it1 - t) / (a + 1);
    if (max < k) {
      printf("%d\n", i + 1);
      return 0;
    }
    s.insert(t);
  }
  puts("-1");
  return 0;
}
