#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<set> 
using namespace std;
set<int> sep;
int main(){
//	freopen("training1_2.txt","r",stdin);
	int j,x,n,l,r,s,k,a,i,m;
	set<int>::iterator iter;
	scanf("%d%d%d",&n,&k,&a);
	scanf("%d",&m);
	s=(n+1)/(a+1);
	sep.insert(0);
	sep.insert(n+1);
	for(i=0;i<m&&s>=k;i++){
		scanf("%d",&x);
		iter=sep.lower_bound(x);
		r=*iter;
		iter--;
		l=*iter;
		sep.insert(x);
		s=s-(r-l)/(a+1)+(r-x)/(a+1)+(x-l)/(a+1);
	}
	if(i==m&&s>=k){
		i=-1;
	}
	printf("%d\n",i);
	return 0;
}
