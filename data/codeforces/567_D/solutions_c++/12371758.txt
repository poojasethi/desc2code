#include <iostream>
#include <cstdio>
#include <set>
using namespace std;

const int maxn =(1e5)*2+10;

int n,k,m,a,x[maxn],ans,res;
set<int> tr;

int main(){
	scanf("%d%d%d",&n,&k,&a);
	scanf("%d",&m);
	set<int>::iterator itr,l,r;
	res=(n+1)/(a+1);
	ans=-1;
	tr.insert(0);tr.insert(n+1);
	for(int i=1;i<=m;i++){
	  scanf("%d",&x[i]);
	  if(tr.find(x[i])!=tr.end())continue;
	  tr.insert(x[i]);
	  itr=tr.lower_bound(x[i]);
	  l=itr;r=itr;l--;r++;
	  res=res-(*r-*l)/(a+1)+(*itr-*l)/(a+1)+(*r-*itr)/(a+1);
	  if(res<k){ans=i;break;}
    }
    printf("%d",ans);
    return 0;
}
