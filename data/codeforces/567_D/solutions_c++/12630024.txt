#include<stdio.h>
#include<set>
using namespace std;
set<pair<int,int> > st;
set<pair<int,int> >::iterator it;
int n,k,m,a,i,s,t,l,r;
int f(int l,int r){
	return (r-l+2)/(a+1);
}
int main(){
	scanf("%d%d%d",&n,&k,&a);
	scanf("%d",&m);
	s=f(1,n);
	st.insert(make_pair(n,1));
	for(i=0;i<m;i++){
		scanf("%d",&t);
		it=st.lower_bound(make_pair(t,0));
		l=it->second;
		r=it->first;
		st.erase(it);
		if(t>l){s+=f(l,t-1);st.insert(make_pair(t-1,l));}
		if(r>t){s+=f(t+1,r);st.insert(make_pair(r,t+1));}
		s-=f(l,r);
		if(s<k){printf("%d",i+1);return 0;}
	}
	printf("-1");
}
