#include<cstdio>
using namespace std;
int x[2000001]={0};
int n,k,a,m,xx;
bool ok(int id){
	int cnt=0,now=0;
	for (int i=1;i<=n;i++){
		if (x[i]>0 && x[i]<=id){
			now = 0;
			continue;
		}
		now++;
		if (now == a){
			now = 0;
			i++;
			cnt++;
		}
	}
	return (cnt>=k);
}
int main(){
	scanf("%d%d%d",&n,&k,&a);
	scanf("%d",&m);
	for (int i=0;i<m;i++){
		scanf("%d",&xx);
		x[xx]=i+1;
	}
	int l=1,r=m;
	while (l<r){
		if (l+1==r){
			l=ok(l)?r:l;
			break;
		}
		int mid=(l+r)>>1;
		if (ok(mid))l=mid;else r=mid;
	}
	if (ok(l)) puts("-1");else
	printf("%d\n",l);
	return 0;
}
