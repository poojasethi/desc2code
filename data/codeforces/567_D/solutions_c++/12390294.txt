#define N (1<<18)
#include <bits/stdc++.h>
using namespace std;

bool p[N];
int n,k,a,m,f[N],Left,Right;

bool judge(int x)
{
	memset(p,0,sizeof(p));
	for(int i=1;i<=x;i++)
		p[f[i]]=1;

	int cnt=0,ck=0;
	for(int i=1;i<=n;i++)
	{
		if(p[i])
			ck=0;
		else
		{
			ck++;
			if(ck==a)
				cnt++,ck=-1;
		}
	}
	return cnt<k;
}

int main()
{
	cin>>n>>k>>a>>m;
	for(int i=1;i<=m;i++)
		scanf("%d",f+i);

	for(Left=1,Right=m+1;Left<Right;)
	{
		int mid=(Left+Right)>>1;
		if(judge(mid))
			Right=mid;
		else
			Left=mid+1;
	}
	cout<<(Left==m+1?-1:Left)<<endl;
	return 0;
}
