#include <bits/stdc++.h>
#define type(x) __typeof(x.begin())
#define maxn 200050
using namespace std;
int n,m,k,q,ar[maxn];
set <int> st ;
int main()
{
	scanf("%d %d %d %d",&n,&m,&k,&q);
	for(int i=1;i<=q;i++)
	{	
		scanf("%d",&ar[i]);
	}
	int cur=(n+1)/(k+1);
	st.insert(0) ;
	st.insert(n+1) ;
	for(int i=1;i<=q;i++)
	{
		type(st)it=st.upper_bound(ar[i]) ;
		type(st)itt =--it;
		it++;
		int uz=(*it)-(*itt)-1;
		cur-=(uz+1)/(k+1);
		cur+=(ar[i]-(*itt))/(k+1);
		cur+=((*it)-ar[i])/(k+1);
		st.insert(ar[i]);
		if(cur<m)
		{
			printf("%d\n",i);
			return 0; 
		}
	}
	printf("-1\n");
	return 0;
}
