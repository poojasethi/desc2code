#include<bits/stdc++.h>
using namespace std;

int main() {
	int N, K, A;
	scanf("%d%d%d", &N, &K, &A);
	set<int> bord;
	int ships = (N+1)/(A+1);
	bord.insert(0);
	bord.insert(N+1);
	int M; scanf("%d", &M);
	for(int i = 1; i <= M; ++i) {
		int x; scanf("%d", &x);
		set<int>::iterator it = bord.upper_bound(x);
		int r = *(it--), l = *it;
		int t1 = x-l-1, t2 = r-x-1, len = t1+t2+1;
		//printf("%d %d\n", t1, t2);
		ships -= ((len+1)/(A+1) - (t1+1)/(A+1) - (t2+1)/(A+1));
		if(ships < K) {
			printf("%d\n", i);
			return 0;
		}
		bord.insert(x);
	}
	printf("-1\n");
}
