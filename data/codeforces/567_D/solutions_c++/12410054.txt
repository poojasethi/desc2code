#include<bits/stdc++.h>
using namespace std;

int n, k, a, m, x, before, after, available;
set<int> sa;

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d%d%d", &n, &k, &a, &m);
	sa.insert(-1), sa.insert(n), ++a;
	available = (n+1)/a;
	for(int i=1 ; i<=m ; ++i){
		scanf("%d", &x);
		after = *sa.lower_bound(x);
		before = *--sa.lower_bound(x)+1;
		sa.insert(x-1);
		available -= (after-before+1)/a ;
		available += (x-before)/a + (after-x+1)/a;
		if(k > available){
			printf("%d\n", i);
			return 0;
		}
	}
	puts("-1");
	return 0;
}
