#include <bits/stdc++.h>

#define NN 200010

using namespace std;

set <int> pos;
set <int> :: iterator it;

int p[NN];

int main (){
	
	int N, k, a, M, i;
	scanf("%d %d %d %d", &N, &k, &a, &M);
	
	pos.insert(0);
	pos.insert(N+1);
	
	for(i=1; i<=M; i++) scanf("%d", p+i);
	
	int tot=(N+1)/(a+1), st, ed;
	
	for(i=1; i<=M; i++){
		it=pos.lower_bound(p[i]);
		ed=*it; it--; st=*it;
		pos.insert(p[i]);
		tot-=(ed-st)/(a+1);
		tot+=(ed-p[i])/(a+1);
		tot+=(p[i]-st)/(a+1);
		if(tot<k) break;
	}
	
	if(i>M) i=-1;
	printf("%d\n", i);
	
	return 0;
}
