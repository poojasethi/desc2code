#include<set>
#include<iostream>

using namespace std;

#define F 300000

int main()
{
	int val;
	int n , k , a;
	cin >> n >> k >> a;
	val = (n+1)/(a+1);
	set<int>S;
	S.insert( 0 );
	S.insert( n+1 );
	int m;
	cin >> m;
	int b;
	int ans=F;
	set<int>::iterator it;
	int x ,y;
	for( int i = 0 ; i < m ; i++ )
	{
		cin >> b;
		it = S.upper_bound(b);
		x = *it;
		it--;
		y = *it;
		val -= (x-y)/(a+1);
		val += (x-b)/(a+1);
		val += (b-y)/(a+1);
		if( val < k ) ans = min( ans , i + 1 );
		S.insert( b );
	}
	if( ans != F ) cout << ans << endl;
	else cout << -1 << endl;
}
