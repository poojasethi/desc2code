#include<bits/stdc++.h>
using namespace std;
set<int> s;
typedef set<int>::iterator cou;

int main(){
int n,k,a,m,re=-1,to,te,l,r;
scanf("%d%d%d%d",&n,&k,&a,&m);
s.clear();
to=(n+1)/(a+1);
s.insert(0);s.insert(n+1);
for(int i=0;i<m;i++){
    scanf("%d",&te);
    cou i1=s.upper_bound(te);
    r=(*i1);
    l=(*(--i1));
    //cout<<to<<" "<<l<<"    "<<r<<endl;
    to-=(r-l)/(a+1);
    s.insert(te);
    to+=(te-l)/(a+1);
    to+=(r-te)/(a+1);
    if(to<k){re=(i+1);break;}
}
printf("%d\n",re);
return 0;
}
