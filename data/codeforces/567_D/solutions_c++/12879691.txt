#include<cstdio>
#include<cstring>
#include<algorithm>
#include<set>
using namespace std;
int n,k,a,f[1000055],m;
set<int> se;
int main(){
    scanf("%d%d%d",&n,&k,&a);
    se.insert(0);se.insert(n+1);
    f[a]=1;for (int i=a+1;i<=n;i++) f[i]=f[i-a-1]+1;
    scanf("%d",&m);
    int now=f[n];
    for (int i=1;i<=m;i++){
        int x;scanf("%d",&x);
        int l=*(--se.lower_bound(x)),r=*se.lower_bound(x);
        now-=f[r-l-1];
        now+=f[x-l-1]+f[r-x-1];
        se.insert(x);
        if (now<k){
            printf("%d",i);return 0;
        }
    }
    puts("-1");
}
