#include<stdio.h>
#include<set>
using namespace std;
set<int>st;
int main()
{
	int n,k,a;
	scanf("%d%d%d",&n,&k,&a);
	st.clear();
	st.insert(0);st.insert(n+1);
	int maxship=(n+1)/(a+1);
	int m;
	scanf("%d",&m);
	for (int i=1;i<=m;i++)
	{
		int x;
		scanf("%d",&x);
		if (st.count(x)>0) continue;
		set<int>::iterator p=st.upper_bound(x);
		int R=*p,L;
		p--;L=*p;
		maxship-=(R-L)/(a+1);
		maxship+=(R-x)/(a+1);
		maxship+=(x-L)/(a+1);
		if (maxship<k) {printf("%d\n",i);return 0;}
		st.insert(x);
	}
	printf("-1\n");
	return 0;
}
