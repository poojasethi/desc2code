#include<cstring>
#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cmath>
#include<set>
#define rep(i,l,r) for (int i=l;i<=r;i++)
#define maxn 400500
#define ll long long
using namespace std;
set<int> b;
int n,x,tot,k,a,m;
int main(){
	scanf("%d%d%d",&n,&k,&a);
	a++;
	tot=(n+1)/a;
	b.insert(0); b.insert(n+1);
	scanf("%d",&m);
	for (int i=1;i<=m;i++){
		scanf("%d",&x);
		set<int>:: iterator l,r=b.lower_bound(x);
		l=r;l--;
		tot-=(*r - *l)/a;
		tot+=(*r-x)/a+(x-*l)/a;
		b.insert(x);
		if (tot<k) {
			printf("%d\n",i);
			return 0;
		}
	}
	puts("-1");
	return 0;
}
