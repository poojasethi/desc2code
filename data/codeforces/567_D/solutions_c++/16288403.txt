#include<iostream>
#include<cstdio>
#include<set>
using namespace std;
set<int>sp;
int main(){
	int n,k,a,x,i,t,sum,flag=1,u=-1;
	cin>>n>>k>>a>>x;
	sp.insert(0);
	sp.insert(n+1);
	sum=(n+1)/(a+1);
	for(i=1;i<=x;i++){
		scanf("%d",&t);
		sp.insert(t);
		set<int>::iterator it;
		it=sp.find(t);
		int d1=*(--it);
		it++;
		int d2=*(++it);
		it--;
		sum=sum-(d2-d1)/(a+1)+(d2-t)/(a+1)+(t-d1)/(a+1);
		if(sum<k&&flag){
			flag=0;
			u=i;
		}
	}
	cout<<u<<endl;
	return 0;
}
