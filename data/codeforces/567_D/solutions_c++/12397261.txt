#include<iostream>
#include<algorithm>
#include<set>
using namespace std;

int n,k,a,m,x;
set<int> s;
set<int>::iterator ite;

int main()
{
    ios::sync_with_stdio(false);
    cin>>n>>k>>a>>m;
    s.insert(0);
    s.insert(n+1);
    int v=(n+1)/(a+1);
    for(int i=0;i<m;i++){
        cin>>x;
        ite=s.upper_bound(x);
        int r=*ite,l=*(--ite);
        v-=(r-l)/(a+1);
        v+=(x-l)/(a+1);
        v+=(r-x)/(a+1);
        if(v<k){
            cout<<i+1<<endl;
            return 0;
        }
        s.insert(x);
    }
    cout<<-1<<endl;
    return 0;
}
