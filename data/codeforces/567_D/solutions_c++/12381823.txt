#include<stdio.h>
#include<set>
#define V 200001
using namespace std;

set<int> s;
set<int>::iterator it;
int lf;
int rt;

int main(){
	int i,n,k,a;
	int m,p,last,cnt;
	int len;
	bool great=false;
	scanf("%d %d %d",&n,&k,&a);
	scanf("%d",&m);
	s.insert(0);
	s.insert(n+1);
	cnt = (n+1)/(a+1);
	for(i=0;i<m;i++){
		scanf("%d",&p);
		it = s.upper_bound(p);
		rt = *it;
		lf = *(--it);
		cnt -= (rt-lf)/(a+1);
		cnt += (p-lf)/(a+1);
		cnt += (rt-p)/(a+1);
		s.insert(p);
		if(cnt<k){
			great = true;
			printf("%d\n",i+1);
			break;
		}
	}
	if(!great)
		printf("%d\n",-1);	
	return 0;
}
