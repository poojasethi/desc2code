#include<iostream>
#include<cstdio>
#define maxn 1000005
using namespace std;
int n,Q;
int pos1,pos2;
int res[maxn];
int main()
{
	 scanf("%d",&n);
	 scanf("%d",&Q);
	 while(Q--)
	 {
	 	int type;
	 	scanf("%d",&type);
		if(type==1)
		{
			int x;
			scanf("%d",&x);
			x%=n;
			pos1+=x;
			pos2+=x;
		}
		else
		{
			if(pos1%2==1)pos1--;else pos1++;
			if(pos2%2==1)pos2++;else pos2--;
		}
		pos1=(pos1+n)%n;
		pos2=(pos2+n)%n;
	}
	for(int i=1;i<=n;i++)
	{
		if(i%2==1)res[(pos1+n+i-1)%n+1]=i;
		else res[(pos2+i+n-1)%n+1]=i;
	}
	for(int i=1;i<=n;i++)
	{
		printf("%d ",res[i]);
	}
	puts("");
	return 0;
}
