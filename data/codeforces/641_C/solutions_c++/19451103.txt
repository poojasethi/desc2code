#include<bits/stdc++.h>
using namespace std;
#define rep(i, a, b) for(i=(a); i<(b); i++)
#define NN 1111111
int a[NN];

int main() {
	
	int n, Q, tp, x;
	scanf("%d%d", &n, &Q);
	
	int s1, s2, i;
	s1=s2=0;
	for(;Q--;) {
		scanf("%d", &tp);
		if(tp==1) {
			scanf("%d", &x);
			s1+=x; s2+=x;
			s1=(s1+n)%n;
			s2=(s2+n)%n;
			
		} else {
			if(s1%2==0) s1++, s2--;
			else s2++, s1--;
		}
	}
	rep(i, 1, n+1) {
		if(i%2) a[(i+s1-1)%n+1]=i;
		else a[(i+s2-1)%n+1]=i;
	}
	rep(i, 1, n+1) {
		if(i>1) printf(" ");
		printf("%d", a[i]);
	} puts("");
	
	return 0;
}
