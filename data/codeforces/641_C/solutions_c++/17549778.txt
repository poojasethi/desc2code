#include <cstdio>

using namespace std;

int a[2];
int ans[1000010];

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n,q;scanf("%d%d",&n,&q);
	while (q--)
	{
		int op;scanf("%d",&op);
		if (op==1)
		{
			int x;scanf("%d",&x);
			for (int i=0;i<2;i++) a[i]=(a[i]+x)%n;
		}
		else
		{
			for (int i=0;i<2;i++)
				if ((a[i]+i)%2==0) a[i]--; else a[i]++;
		}
	}
	for (int i=1;i<=n;i++) ans[((i+a[i&1])%n+n)%n]=i;
	for (int i=1;i<=n;i++) printf("%d ",ans[i%n]);
	return 0;
}
