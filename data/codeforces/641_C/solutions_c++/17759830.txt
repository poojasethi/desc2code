#include<cstdio>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;
int n,q;
int mv[2],ans[1000006];
bool f;
inline void add(int &x,int y){
	x+=y;
	x=((x%n)+n)%n;
	return;
}
int main(){
	scanf("%d%d",&n,&q);
	while(q--){
		int op;
		scanf("%d",&op);
		if(op==2){
			add(mv[f],-1);
			add(mv[f^1],1);
			f^=1;
		}
		else{
			int x;
			scanf("%d",&x);
			add(mv[0],x);
			add(mv[1],x);
			if(x&1)
				f^=1;
		}
	}
	rep(i,n)
		if(i & 1)
			ans[(mv[0]+i)%n]=i+1;
		else
			ans[(mv[1]+i)%n]=i+1;
	rep(i,n)
		printf("%d%c",ans[i],i==n-1?'\n':' ');
	return 0;
}
