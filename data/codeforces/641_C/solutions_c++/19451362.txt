#include <bits/stdc++.h>

#define NN 1000010
using namespace std;

int p[NN];

void move(int &a){
	if(a%2) a++;
	else a--;
}

int main (){
	
	int N, Q;
	scanf("%d %d", &N, &Q);
	
	int id, x;
	int one=1, two=2;
	
	while(Q--){
		scanf("%d", &id);
		if(id==1){
			scanf("%d", &x);
			one+=x; one%=N;
			if(one<=0) one+=N;
			two+=x; two%=N;
			if(two<=0) two+=N;
		}
		else {
			move(one);
			move(two);
		}
	}
	
	for(int i=0; i<N; i+=2){
		p[(one+i)%N]=i+1;
		p[(two+i)%N]=i+2;
	}
	p[N]=p[0];
	
	for(int i=1; i<=N; i++) printf("%d ", p[i]);
	
	return 0;
}
