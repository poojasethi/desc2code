#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,q,ans[1000010],delta[2];
void add(int &x,int a){x=(x+a+2*n)%n;}
int main(){
	scanf("%d%d",&n,&q);
	for(int i=1,op,x;i<=q;++i){
		scanf("%d",&op);
		if(op==1)scanf("%d",&x),add(delta[0],x),add(delta[1],x);
		else{
			if((delta[1]+n)&1)add(delta[1],-1),add(delta[0],1);
			else add(delta[1],1),add(delta[0],-1);
		}
	}
	for(int i=1;i<=n;++i)ans[(i+delta[i&1]+2*n)%n]=i;
	ans[n]=ans[0];for(int i=1;i<=n;++i)printf("%d%c",ans[i],i==n?'\n':' ');
}
