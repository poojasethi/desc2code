#include <iostream>
#include <string>
#include <vector>
#include <cmath>

#define rep(i,n) for(int i=0;i<n;i++)
#define INF      100000000

using namespace std;

int n,m,N,cap[210][210],cost[210][210],used[210],a,sz,mf,res;
string str,t;

int dfs(int v,int flow){
  if(v==N)return flow;
  used[v] = 1;
  rep(i,N+1){
    if(cap[v][i] && !used[i]){
      int tmp = dfs(i,min(flow,cap[v][i]));
      if(tmp){
    cap[v][i] -= tmp;
    cap[i][v] += tmp;
    return tmp;
      }
    }
  }
  return 0;
}

int main(){

  cin>>str>>n;
  sz = str.size();
  N = n+27;
  rep(i,sz){
    cap[str[i]-'a'+n+1][N]++;
  }

  rep(i,n){
    cin>>t>>a;
    cap[0][i+1] = a;cost[0][i+1] = i+1;
    rep(j,t.size())cap[i+1][t[j]-'a'+n+1]++;
  }
  while(1){
    fill(used,used+N,0);
    int f = dfs(0,INF);
    if(!f)break;
    else mf += f;
  }

  
  if(mf != str.size()){
    cout<<-1<<endl;
  }
  else {
    for(int i=1;i<=n;i++){
      res += cap[i][0]*i;
    }
    cout<<res<<endl;
  }

  return 0;


}
