#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>

using namespace std;

int S,T;
int n,a[30],ans,tot;
char str[10100];
int head[1010],next[100100],key[100100],len[100100],flow[100100],cnt;
int dis[1010],minr[1010],lk[1010],vis[1010];
int q[100100];

void add(int x,int y,int z,int c)
{
	next[cnt]=head[x],key[cnt]=y,flow[cnt]=z,len[cnt]=c,head[x]=cnt++;
	next[cnt]=head[y],key[cnt]=x,flow[cnt]=0,len[cnt]=-c,head[y]=cnt++;
}

bool spfa()
{
	int h=1,t=2;
	memset(dis,0x3f,sizeof dis);
	memset(vis,0,sizeof vis);
	dis[S]=0;
	vis[S]=1;
	q[1]=S;
	minr[S]=0x3f3f3f3f;
	while(h<t)
	{
		int v=q[h++];
		vis[v]=0;
		for(int i=head[v];~i;i=next[i])
		{
			int u=key[i];
			if(flow[i]&&dis[u]>dis[v]+len[i])
			{
				dis[u]=dis[v]+len[i];
				minr[u]=min(minr[v],flow[i]);
				lk[u]=i;
				if(!vis[u])
				{
					vis[u]=1;
					q[t++]=u;
				}
			}
		}
	}
	if(dis[T]==0x3f3f3f3f) return false;
	return true;
}

int update()
{
	int i=T;
	while(i!=S)
	{
		ans+=minr[T]*len[lk[i]];
		flow[lk[i]]-=minr[T];
		flow[lk[i]^1]+=minr[T];
		i=key[lk[i]^1];
	}
	return minr[T];
}

int main()
{
	memset(head,-1,sizeof head);
	scanf("%s",str+1);
	int len=strlen(str+1);
	for(int i=1;str[i];i++)
		a[str[i]-'a'+1]++;
	scanf("%d",&n);
	S=0,T=n+26+1;
	for(int i=1;i<=26;i++)
		add(n+i,T,a[i],0);
	for(int i=1,x;i<=n;i++)
	{
		memset(a,0,sizeof a);
		scanf("%s%d",str+1,&x);
		for(int j=1;str[j];j++)
			a[str[j]-'a'+1]++;
		add(S,i,x,i);
		for(int j=1;j<=26;j++)
			add(i,n+j,a[j],0);
	}
	while(spfa()) tot+=update();
	if(tot!=len) printf("-1\n");
	else printf("%d",ans);
	return 0;
}
