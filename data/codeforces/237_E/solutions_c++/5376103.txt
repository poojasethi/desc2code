#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
#define INF 999999999
using namespace std;

int n, N;
int c[150][150], vst[150];
int flow(int p, int fl){
	if(p == N-1) return fl;
	vst[p] = 1;
	for(int i = 0; i < N; i++) if(!vst[i] && c[p][i]){
		int f = flow(i, min(fl, c[p][i]));
		if(f){
			c[p][i] -= f;
			c[i][p] += f;
			return f;
		}
	}
	return 0;
}

int main (){
	char s[110];
	scanf("%s%d", s, &n); N = n+28;
	for(int i = 0; s[i]; i++)
		c[s[i]-'a'+1+n][N-1]++;
	for(int i = 0; i < n; i++){
		scanf("%s%d", s, &c[0][i+1]);
		for(int j = 0; s[j]; j++)
			c[i+1][s[j]-'a'+1+n]++;
	}
	while(1){
		fill(vst, vst+N, 0);
		if(!flow(0, INF)) break;
	}
	int ans = 0;
	for(int i = 1; i <= n; i++) for(int j = 1; j <= 26; j++)
		ans += i * c[n+j][i];
	for(int i = n+1; i <= n+26; i++) if(c[i][N-1]) return puts("-1"), 0;
	printf("%d\n", ans);
}
