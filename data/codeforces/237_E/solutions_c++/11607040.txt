#include <algorithm>
#include <string.h>
#include <limits.h>
#include <stdio.h>
#define  maxl      105
#define  maxN      128
using    namespace std;

int N,n,m,k;
int a[maxl];
char t[maxl];
int mark[maxN];
char s[maxl][maxl];
int ar[maxN][maxN];

int dfs(int node , int way)
{
  if(node==N)
    return way;
  mark[node]=k;
  for(int i=0 ; i<=N ; i++)
    if(ar[node][i] && mark[i]!=k)
    {
      int tmp=dfs(i,min(way,ar[node][i]));
      if(!tmp)
        continue;
      ar[node][i]-=tmp;
      ar[i][node]+=tmp;
      return tmp;
    }
  return 0;
}

int main()
{
  scanf("%s%d",t,&n);
  N=n+27;
  m=strlen(t);
  for(int i=0 ; i<m ; i++)
    ar[t[i]-'a'+n+1][N]++;
  for(int i=1 ; i<=n ; i++)
  {
    scanf("%s%d",s[i],&a[i]);
    ar[0][i]=a[i];
    int len=strlen(s[i]);
    for(int j=0 ; j<len ; j++)
      ar[i][s[i][j]-'a'+n+1]++;
  }
  while(1)
  {
    k++;
    if(!dfs(0,INT_MAX))
      break;
  }
  int res=0,cnt=0;
  for(int i=1 ; i<=n ; i++)
  {
    int tmp=ar[i][0];
    cnt+=tmp;
    res+=tmp*i;
  }
  if(cnt!=m)
    res=-1;
  printf("%d\n",res);
  return 0;
}
