#define inf 10000000    
int n,vis[1000],adj[200][200];
char str[200];
int augpath(int v,int flow)
{
    int temp;
    if(v==n+27) return flow;
    vis[v]=1;
    for(int i=0;i<n+28;i++)
    {
        if(vis[i]==0 and adj[v][i])
        {
            temp=min(adj[v][i],flow);
            temp=augpath(i,temp);
            if(temp)
            {
                adj[i][v]+=temp;
                adj[v][i]-=temp;
                return temp;
            }
        }
    }
    return 0;
}
int main()
{
    scanf("%s",str);
    si(n);
    int len=strlen(str);
    int i,temp,j;
    for(i=0;i<len;i++) adj[n+str[i]-96][n+27]++;
    for(i=0;i<n;i++)
    {
        scanf("%s",str);
        len=strlen(str);
        si(temp);
        adj[0][i+1]=temp;
        for(j=0;j<len;j++) adj[i+1][n+str[j]-96]++; 
    }
    while(augpath(0,inf))
    {
        for(i=0;i<n+30;i++) vis[i]=0;
    }
    int ans=0;
    for(i=1;i<=n;i++)
        for(j=n+1;j<n+27;j++) ans+=adj[j][i]*i;
    for(i=n+1;i<n+27;i++)
        if(adj[i][n+27]) ans=-1;
    printf("%d\n", ans);    
}
