#include<stdio.h>

#define QF Q.front()
#define ST S.top()
#define PF printf
#define SF scanf
#define EL PF("\n");
using namespace std;

int i,j,n,FLOW=0,check=1,sum=0;
char t[101],por[128];
char s[101][101];
int cap[128][128];
int flow[128][128];
int visit[128];

void dfs(int p){
    visit[p]=1;
    if(p==127){
        check=1;
        while(p!=0){
            flow[por[p]][p]++;
            flow[p][por[p]]--;
            p=por[p];
        }
        for(i=0;i<=127;i++){
            visit[i]=0;
        }
        FLOW++;
        return;
    }
    for(int k=0;k<=127;k++){
        if(cap[p][k]-flow[p][k]>0 && visit[k]==0){
            por[k]=p;
            dfs(k);
        }
        if(check){
            break;
        }
    }
    return;
}

int main(){
    gets(t);
    for(i=0;t[i]!='\0';i++){
        cap[101+t[i]-'a'][127]++;
    }
    SF("%d",&n);
    for(i=1;i<=n;i++){
        SF("%s",s[i]);
        for(j=0;s[i][j]!='\0';j++){
            cap[i][101+s[i][j]-'a']++;
        }
        SF("%d",&cap[0][i]);
    }

    while(check){
        check=0;
        dfs(0);
    }

    if(FLOW<strlen(t)){
        PF("-1");
    }
    else{
        for(i=1;i<=n;i++){
            sum+=flow[0][i]*i;
        }
        PF("%d",sum);
    }
    return 0;
}
