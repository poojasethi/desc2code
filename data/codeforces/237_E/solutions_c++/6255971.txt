#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <vector>
#define mm 1005
#define clr(A) memset(A,0,sizeof(A))
#define INF 1000000
using namespace std;
typedef long long LL;
typedef pair<int,int > P;

char st[mm];
int n,N;
int vis[mm];
int map[150][150];
void input()
{
    scanf("%s%d",st,&n);
	N = n + 27;
	clr(map);
	for(int i = strlen(st)-1;i>=0;i--)
	map[st[i]-'a'+1+n][N]++;
	for(int i = 1;i<=n;i++)
	{
	    int v;
	    scanf("%s%d",st,&v);
	    map[0][i] = v;
	    for(int j = strlen(st)-1;j>=0;j--)
	    map[i][st[j]-'a'+1+n]++;
	}
}
int extend(int k,int flow)
{
    if(k==N) return flow;
    vis[k] = 1;
    for(int i = 0;i<=N;i++)
    if(!vis[i] && map[k][i])
    {
        int res = extend(i,min(flow,map[k][i]));
        if(res){
            map[k][i] -= res;
            map[i][k] += res;
            return res;
        }
    }
    return 0;
}
int main()
{
	//freopen("chef.in","r",stdin);
	input();
	while(1)
	{
	    clr(vis);
	    if(!extend(0,INF)) break;
	}
	int ans = 0;
	for(int i = n+1;i<=n+26;i++)
	if(map[i][N]) return printf("-1\n"),0;
	for(int i = 1;i<=n;i++)
	for(int j = 1;j<=26;j++)
	ans += i * map[n+j][i];
	printf("%d\n",ans);
    return 0;
}
