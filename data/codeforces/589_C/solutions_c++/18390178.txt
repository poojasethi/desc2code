	template <class Type> Type read()
	{
		Type aa;
		int bb = 1, ch;
		while (ch = getc(), (ch < '0' || ch > '9') && ch != '-');
		ch == '-' ? aa = 0, bb = -1 : aa = ch - '0';
		while (ch = getc(), ch >= '0' && ch <= '9') aa = aa * 10 + ch - '0';
		return bb == 1 ? aa : -aa;
	}

namespace lightningwork
{
	// my space

	long long (*read)() = defs::read<long long>;

	char str[111];
	
	int n, m, len, k[100010];
	
	int sum[111][26], tsum[111][111][26];

	long long siz[111], top;

	int dfs(register int i, register long long p)
	{
		if (i == 0) return str[p] - 'a';
		if (siz[i] >= p) return dfs(i - 1, p);
		register long long t = p - siz[i] - k[i];
		if (t <= 0) t += siz[i];
		return dfs(i - 1, t);
	}

	void init()
	{
		register int ch = getc();
		for (register int i = 1; ch >= 'a' && ch <= 'z'; ++i, ch = getc())
		{
			str[i] = ch;
			memcpy(sum[i], sum[i - 1], sizeof sum[i]);
			++sum[i][ch - 'a'];
			++len;
		}
		n = read();
		m = read();
		register long long L = len;
		for (int i = 1; i <= n; ++i)
		{
			k[i] = read();
			if (L <= (long long) 1e18)
			{
				siz[++top] = L;
				k[i] %= L;
				L <<= 1;
			}
		}
		for (register int i = 1; i <= top; ++i)
		{
			for (register int j = 1; j <= k[i]; ++j)
			{
				memcpy(tsum[i][j], tsum[i][j - 1], sizeof tsum[i][j]);
				++tsum[i][j][dfs(i, siz[i] + j)];
			}
		}
	}
	
	long long dfs(register int i, register long long p, register int ch)
	{
		if (p == 0) return 0;
		if (p <= siz[i]) return dfs(i - 1, p, ch);
		if (i == 0) return sum[p][ch];
		register long long ret = siz[i] / len * sum[len][ch];
		p -= siz[i];
		if (p > k[i])
		{
			ret += dfs(i - 1, p - k[i], ch) + tsum[i][k[i]][ch];
		} else
		{
			ret += tsum[i][p][ch];
		}
		return ret;
	}
	
	void solve()
	{
		register long long l = read();
		register long long r = read();
		register int ch = getc();
		while (ch < 'a' || ch > 'z') ch = getc();
		printf("%lld\n", dfs(top, r, ch - 'a') - dfs(top, l - 1, ch - 'a'));
	}

	void work()
	{
		init();
		while (m--)
		{
			solve();
		}
	}
}

int main()
{
	file_able setfile("work");
	lightningwork::work();
	file_able closefile();
	return 0;
}
