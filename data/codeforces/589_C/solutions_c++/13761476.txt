#include <bits/stdc++.h>
using namespace std;

#define ms(s, n) memset(s, n, sizeof(s))
#define FOR(i, a, b) for(int i = (a); i < (b); i++)
#define FORd(i, a, b) for(int i = (a) - 1; i >= (b); i--)
#define sz(a) int((a).size())
typedef long long ll;
typedef unsigned long long ull;
typedef long double ld;
typedef pair<int, int> pi;
typedef vector<int> vi;
typedef vector<pi> vii;
ll gcd(ll a, ll b) {ll r; while (b != 0) {r = a % b; a = b; b = r;} return a;}
ll lcm(ll a, ll b) {return a / gcd(a, b) * b;}
#define EPS 1e-9
const int mod = 1000000007;
const int INF = (int) 1e9;
const ll LINF = (ll) 1e18;
const ld PI = 2 * acos(0);

const int maxn = 100010;
const int maxc = 26;
char s[maxn];
int n, m;
int len;
int k[maxn];
int sum[maxn][maxc];
ll f[110][65][maxc];

ll calc(ll pos, int d, int c) {
    if (pos < 0) return 0;
    if (pos < len) return sum[pos][c];
    ll m = len * (1LL << (d - 1));
    if (2 * m - pos < 110 && f[2 * m - pos][d][c] != -1) return f[2 * m - pos][d][c];
    if (pos >= m) {
        ll res = (1LL << (d - 1)) * sum[len - 1][c];
        pos -= m;
        int r = k[d] % m;
        if (r <= pos + 1) {
            res += (1LL << (d - 1)) * sum[len - 1][c] - calc(m - r - 1, d - 1, c);
            res += calc(pos - r, d - 1, c);
        }
        else res += calc(m - r + pos, d - 1, c) - calc(m - r - 1, d - 1, c);
        if (m - pos < 110) return f[m - pos][d][c] = res; else return res;
    }
    if (2 * m - pos < 110) return f[2 * m - pos][d][c] = calc(pos, d - 1, c); else return calc(pos, d - 1, c);
}

void solve() {
    ms(sum, 0); ms(f, -1);
    scanf("%s%d%d", s, &n, &m);
    FOR(i, 1, n + 1) scanf("%d", k + i);
    len = strlen(s);
    FOR(i, 0, len) {
        if (i) FOR(j, 0, maxc) sum[i][j] = sum[i - 1][j];
        sum[i][s[i] - 'a']++;
    }
    ll tmp = len; int up = 0;
    while (tmp < (ll) 1e18 && up < n) {tmp <<= 1; up++;}
    while (m--) {
        ll l, r; char c[10]; scanf("%I64d%I64d%s", &l, &r, c); l--; r--;
        ll res = calc(r, up, c[0] - 'a') - calc(l - 1, up, c[0] - 'a');
        printf("%I64d\n", res);
    }
}

int main() {
	//ios_base::sync_with_stdio(0); cin.tie(NULL);
#ifdef _LOCAL_
	freopen("in.txt", "r", stdin); //freopen("out.txt", "w", stdout);
#endif
	solve();
#ifdef _LOCAL_
	//printf("\nTime elapsed: %dms", 1000 * clock() / CLOCKS_PER_SEC);
#endif
	return 0;
}
