#include<cstdio>
using namespace std;

int main() {
	int cnt1, cnt2, x, y;
	scanf("%d %d %d %d", &cnt1, &cnt2, &x, &y);
	int l = 1, r = 2e9;
	while (l < r) {
		int m = l + (r - l) / 2;
		if (cnt1 <= (m - m / x) && cnt2 <= (m - m / y) && cnt1 + cnt2 <= (m - m / (x * y)))
			r = m;
		else l = m + 1;
	}
	printf("%d\n", l);
	return 0;
}
