#include<bits/stdc++.h>
using namespace std;

long long n, m, x, y, l, r, mid;

main() {
    ios_base::sync_with_stdio(0);
    cin >> n >>  m >> x >> y;
    l=0; r=2e9;
    //cout << 2e9 << '\n';
    while (l<r) {
        mid = (l+r)/2;
        (mid-mid/x>=n&&mid-mid/y>=m&&mid-mid/(x*y)>=m+n) ? r=mid : l=mid+1;
    }
    cout  << r << '\n';
}
