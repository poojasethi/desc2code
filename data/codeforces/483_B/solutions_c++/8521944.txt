#include <stdio.h>
#include <math.h>
#include <string.h>
#include <iostream>
using namespace std;
typedef long long ll;
int n1,n2,x,y;
ll l=1,r=2000000000,m;
int main()
{
	cin>>n1>>n2>>x>>y;
	int z=x*y;
	while(l<r){
		m=(l+r)/2;
		if((m-m/x)>=n1&&(m-m/y)>=n2&&(m-m/z)>=(n1+n2))
			r=m;
		else l=m+1;
	}cout<<r<<endl;
}//8521873
