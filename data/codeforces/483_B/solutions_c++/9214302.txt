#include <iostream>
using namespace std;

int main() {
    long long c1, c2, x, y, beg=0,end=100000000000,mid;
    cin>>c1>>c2>>x>>y;
    while(beg<end)
    {
    	mid=(beg+end)/2;
    	if(c1<=mid-mid/x&&c2<=mid-mid/y&&(c1+c2)<=mid-mid/(x*y))
    	end=mid;
    	else 
    	beg=mid+1;
    	
    }
    cout<<end;
    return 0;
}
