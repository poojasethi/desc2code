#include <iostream>

using namespace std;

int main(){

	int cnt1, cnt2, x, y;
	cin >> cnt1 >> cnt2 >> x >> y;


	int left = 1, right = 0x7FFFFFFF, m;

	while (left < right){
		m = left + (right-left)/2;
		if (cnt1 <= m-m/x && cnt2 <= m-m/y && cnt1+cnt2 <= m-m/(x*y)){
			right = m;
		}
		else{
			left = m+1;
		}
	}

	cout << right;
	return 0;
}
