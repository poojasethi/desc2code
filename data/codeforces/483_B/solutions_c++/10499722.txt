#include <cstdio>
int gcd(int a,int b){
	return a?gcd(b%a,a):b;
}
int main(){
	int a,b,x,y;
	scanf("%d%d%d%d",&a,&b,&x,&y);
	long long l=1,r=4000000001;
	while(r>l){
		long long m=(l+r)>>1;
		if(m-m/x>=a&&m-m/y>=b&&m-m/(x/gcd(x,y)*y)>=a+b){
			r=m;
		}else{
			l=m+1;
		}
	}
	printf("%lld\n",r);
	return 0;
}
