#include<bits/stdc++.h>
using namespace std;
#define int long long
int cnt1,cnt2,x,y;
main()
{

    cin>>cnt1>>cnt2>>x>>y;
    int l=1,r=1e+18;
    while(r>l)
    {
        int mid=(l+r)/2;
        if(cnt1<=mid-mid/x && cnt2<=(mid-mid/y) && cnt1+cnt2<=mid-mid/(x*y))
            r=mid;
        else
            l=mid+1;
    }
    cout<<r;
}
