#include <iostream>
using namespace std;

int main() {
    long long l=0,r=~0ull>>1;
    long long cnt1,cnt2,x,y;
    cin>>cnt1>>cnt2>>x>>y;
    long long m;
    while(l<r) {
        m=(l+r)>>1;
        if(m-m/x>=cnt1&&m-m/y>=cnt2&&m-m/(x*y)>=cnt1+cnt2) r=m;
        else l=m+1;
    }
    cout<<r<<endl;
    return 0;
}
