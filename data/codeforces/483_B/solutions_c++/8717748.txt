#include <cstdio>
#include <iostream>
using namespace std;
long long l=2 , r=2000000000, c[2], x[2];
int main(){
   scanf("%lld %lld %lld %lld", &c[0], &c[1], &x[0], &x[1]);
   while(l!=r){
      long long w = (l+r)/2;
      if((w-w/x[0]>=c[0]) && (w-w/x[1]>=c[1]) && (w-(w/(x[0]*x[1]))>=c[0]+c[1])) r=w;
      else l=w+1;
   }
   printf("%lld", l);
}
