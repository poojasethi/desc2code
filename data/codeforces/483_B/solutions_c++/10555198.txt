#include <iostream>
using namespace std;
int main(){
    long long n, k, x, y, f = 1, l = 2e9; cin >> n >> k >> x >> y;
    while (f < l){
        long long m =( f + l - 1) / 2;
        if (n <= (m - m / x) && k <= (m - m / y) && k + n <= (m - m / (x*y)))l = m;
        else f = m + 1;}
    cout << l << endl;
    return 0;
}
