#include<bits/stdc++.h>
using namespace std;
typedef long long LL;

int main()
{
    LL cnt1,cnt2,x,y;
    cin>>cnt1>>cnt2>>x>>y;
   LL m,l=0,r=1LL<<32;
    while(l<r)
    {
         m=(l+r)>>1;
         if(m-(m/x)>=cnt1&&(m-(m/y))>=cnt2&&m-(m/(x*y))>=cnt1+cnt2)
            r=m;
         else l=m+1;
    }
    cout<<r<<endl;
}
