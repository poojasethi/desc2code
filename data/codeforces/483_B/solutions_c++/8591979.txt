#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,m,a,b,min;
	while(scanf("%d %d %d %d",&n,&m,&a,&b)!=EOF)
	{
		int i=2,s=2e9;
		while(i<s)
		{        
			min=i+(s-i)/2;
			if(n<=(min-min/a)&&m<=(min-min/b)&&n+m<=min-min/(a*b))
				s=min;
			else i=min+1;
		}
		printf("%d\n",s);	
	}
	return 0;
}
