#include <stdio.h>
int main()
{
	int n,m,x,y;
	while(~scanf("%d%d%d%d",&n,&m,&x,&y))
	{
		int l=1,r=2e9;        
		while(l<r)
		{              
			int mid=l+(r-l)/2;
			if(n<=(mid-mid/x)&&m<=(mid-mid/y)&&n+m<=mid-mid/(x*y))   
				r=mid;
			else
			  	l=mid+1;
		}
		printf("%d\n",r);	
	}
	return 0;
}
