#include<stdio.h>
int main (void){
	int cnt1,cnt2,x,y;
	scanf("%d%d%d%d",&cnt1,&cnt2,&x,&y);
	int l=1,r=2e9;//端点 
	while(l<r){
		int mid=l+(r-l)/2;
		if(mid-mid/x>=cnt1&&mid-mid/y>=cnt2&&mid-mid/(x*y)>=cnt1+cnt2){
			r=mid;
		}
		else l=mid+1;
	}
	printf("%d\n",r);
	return 0;
}
