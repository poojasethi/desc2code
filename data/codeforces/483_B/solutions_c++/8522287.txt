#include<cstdio>
using namespace std;
int main()
{
    int a,b,c,d;
    int ba=0,bb=2147483647,bc;
    scanf("%d%d%d%d",&a,&b,&c,&d);
    while(ba<bb)
    {
        bc=(long long)ba+bb>>1;
        if(bc-bc/c/d>=a+b&&bc-bc/c>=a&&bc-bc/d>=b)
            bb=bc;
        else ba=bc+1;
    }
    printf("%d\n",ba);
}
