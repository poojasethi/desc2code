#include <cstdio>
int main(){
	long long a, b, x, y, maxx, l = 1, r = 100000000000007;
	scanf("%I64d %I64d %I64d %I64d", &a, &b, &x, &y);
	while(l != r) {
		long long mid = (l + r) / 2;
		if(mid - (mid / x) >= a && mid - (mid / y) >= b &&  mid - (mid / (x * y)) >= a + b)
			r = mid;
		else l = mid + 1;
	}
	printf("%I64d", l);
}
