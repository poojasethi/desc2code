#include <iostream>
#include <cmath>
int main(void){ 
using namespace std;
int a,b,c,x,y,nx,ny,nxy,min1,min2,Rreal,Rideal,d;
cin>>a>>b>>x>>y;
c=a+b;
do {
c=c+d;
ny=c/y;
nx=c/x;
nxy=c/(x*y);
Rreal=c-ny-nx+nxy;
min1=a;if ((ny-nxy)<a) min1=ny-nxy;
min2=b;if ((nx-nxy)<b) min2=nx-nxy;
Rideal=a-min1+b-min2;
d=Rideal-Rreal;
}while (d>0);
cout<<c;
}
