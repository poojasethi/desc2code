#include<stdio.h>

int main(void){
	int cnt1,cnt2,x,y;
	scanf("%d %d %d %d",&cnt1,&cnt2,&x,&y);
	int l,r;
	l=1;
	r=2000000000;
	while(l<r){
		int m=l+((r-l)>>1);
		if((cnt1<=m-m/x)&&(cnt2<=m-m/y)&&((cnt1+cnt2)<=m-m/(x*y))){
			r=m;
		}
		else{
			l=m+1;
		}
	}
	printf("%d\n",r);
	return 0;
}
