def satisfies(cnt1, cnt2, x, y, v):
    '''
    tells whether you can form presents using the numbers
    '''
    bothXY = v / (x * y)
    onlyX = v/ x - bothXY
    onlyY = v/ y - bothXY
    none = v - bothXY - onlyX - onlyY
    cnt1 = max(cnt1 - onlyY, 0)
    cnt2 = max(cnt2 - onlyX, 0)
    return (cnt1 + cnt2 <= none)

def function(v):
    '''
    returns a tuple of bothXY, onlyX, onlyY, none
    '''
    bothXY = v / (x * y)
    onlyX = v/ x - bothXY
    onlyY = v/ y - bothXY
    none = v - bothXY - onlyX - onlyY
    return (bothXY, onlyX, onlyY, none)

import sys
cnt1, cnt2, x, y = map(int, sys.stdin.readline().split())
leftBound = cnt1 + cnt2 #inclusive bounds
rightBound = (cnt1 + cnt2) * 2
while(rightBound - leftBound > 0):
    v = (leftBound + rightBound) / 2
    if(satisfies(cnt1, cnt2, x, y, v)):
        rightBound = v
    else:
        leftBound = v + 1
