def analyze(v, x, y):
    divx = v/x
    divy = v/y
    divxandy = v/(x*y)
    divxonly = divx - divxandy
    divyonly = divy - divxandy
    suits_x_only = divyonly
    suits_y_only = divxonly
    suits_none = divxandy
    suits_both = v - suits_none - suits_x_only - suits_y_only
    return suits_x_only, suits_y_only, suits_none, suits_both

def split_fits(need_x, need_y, available_x, available_y, available_both):
    need_more_x = max(need_x - available_x, 0)
    need_more_y = max(need_y - available_y, 0)
    return need_more_x + need_more_y <= available_both

def main(data):
    c1, c2, x, y = map(int, data.split())
    def test(v):
        sx, sy, sn, sb = analyze(v, x, y)
        return split_fits(c1, c2, sx, sy, sb)
    left = 1
    right = 100000000000
    lval = test(left)
    rval = test(right)
    while (right - left) > 1:
        mid = (left + right)/2
        t = test(mid)
        if t:
            right = mid
        else:
            left = mid
    print right
