cnt1,cnt2,x,y=map(int,raw_input().split())


def is_ok(val):
    nx=val/x
    ny=val/y
    ncom=val/(x*y)

    tx=val-nx
    ty=val-ny
    tcom=val-nx-ny+ncom

    return tx>=cnt1 and ty>=cnt2 and tx+ty-tcom>=cnt1+cnt2



begs=1
ends=100000000**2

while(begs<ends):
    mid=(begs+ends)/2

    if(is_ok(mid)):
        ends=mid
    else:
        begs=mid+1


print ends
