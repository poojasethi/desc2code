def test(n,c1,c2,x,y):        
    xy=x*y
    c1-=(n/y-n/xy)
    c2-=(n/x-n/xy)
    if c1<=0:c1=0
    if c2<=0:c2=0
    n=n-n/x-n/y+n/xy
    return n>=c1+c2

def main():
    c1,c2,x,y=map(long,raw_input().strip().split())

    lo=1;hi=10**20

    while lo<hi:
        m=(lo+hi)/2
        if test(m,c1,c2,x,y):hi=m
        else:lo=m+1
    print lo
    
    







main()
