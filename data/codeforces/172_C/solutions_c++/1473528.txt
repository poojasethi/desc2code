#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

const int MAXN=100010;

int n,m;
int t[MAXN],x[MAXN],ans[MAXN];

void init()
{
	cin>>n>>m;
	for(int i=0;i<n;++i)
		cin>>t[i]>>x[i];
}
void solve()
{
	int head=0,backTime=0;
	while(head<n)
	{
		vector<pair<int,int> > car;
		for(int j=0;head<n && j<m; ++head,++j)
			car.push_back(make_pair(x[head],head));
		int leaveTime=max(backTime,t[head-1]);
		sort(car.begin(),car.end());
		int way=0;
		for(int j=0;j<car.size();++j)
		{
			int s=j;
			while(j+1<car.size() && car[j+1].first==car[s].first)j++;
			for(int i=s;i<=j;++i)
				ans[car[i].second]=leaveTime+car[s].first+way;
			way+=(j-s+1)/2+1;
		}
		backTime=leaveTime+way+2*car[car.size()-1].first;
	}
	for(int i=0;i<n;++i)
		cout<<ans[i]<<' ';cout<<endl;
}
int main()
{
	init();
	solve();
	return 0;
}
