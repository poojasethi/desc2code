#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct note {int t,x,id;}a[100005],b[100005];
int n,m,i,j,k,Time=0,ans[100005];
bool cmpt(const note i,const note j){return i.t<j.t;}
bool cmpx(const note i,const note j){return i.x<j.x;}
int main()
{
	scanf("%d%d",&n,&m);
	for (i=1;i<=n;i++)
		scanf("%d%d",&a[i].t,&a[i].x),a[i].id=i;
	sort(a+1,a+n+1,cmpt);
	for (i=0;i<n;){
		j=0; while (j<m && i<n) b[++j]=a[++i];
		if (b[j].t>Time) Time=b[j].t;
		sort(b+1,b+j+1,cmpx); b[0].x=0; b[j+1].x=b[j].x+100;
		for (int k=1;k<=j;k++){
			int run=b[k].x-b[k-1].x,num=1; Time+=run;
			while (b[k+1].x==b[k].x) ans[b[k].id]=Time,k++,num++;
			ans[b[k].id]=Time; Time+=1+num/2;
			}
		Time+=b[j].x;
		}
	for (int i=1;i<n;i++) printf("%d ",ans[i]);
	printf("%d\n",ans[n]); return 0;
}
