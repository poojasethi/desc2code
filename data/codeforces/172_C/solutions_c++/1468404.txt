#include <cstdio>
#include <map>
using namespace std;

enum {
    MAXM = 100 * 1010
};

map<int, int> stop;
int want[MAXM];

int main() {
    int n, m, time = 0, i, j, cnt = 0;
    scanf("%d %d", &n, &m);
    for(i = 0; i < n; i++) {
        int t, x;
        scanf("%d %d", &t, &x);
        stop[x]++;
        want[cnt] = x;
        time = max(time, t);
        cnt++;
        if(cnt == m || i == n - 1) {
            int prevx = 0;
            for(map<int, int>::iterator it = stop.begin(); it != stop.end(); it++) {
                int k = it->second;
                x = it->first;
                time += x - prevx;
                stop[x] = time;
                time += 1 + k / 2;
                prevx = x;
            }
            for(j = 0; j < cnt; j++) {
                printf("%d\n", stop[want[j]]);
            }
            time += prevx;
            cnt = 0;
            stop.clear();
        }
    }
    return 0;
}
