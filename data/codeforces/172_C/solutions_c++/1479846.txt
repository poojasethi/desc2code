#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 100000 + 10;
typedef pair<int, int> ii;
ii a[N];
int ans[N];

int main() {
  int n, m;
  scanf("%d %d", &n, &m);
  int T = 0, X = 0;
  for (int i = 0; i < n; i += m) {
    int sz = min(m, n - i);
    for (int j = 0; j < sz; ++j) {
      int x, y;
      scanf("%d %d", &x, &y);
      a[j] = ii(y, i + j);
      T = max(T, x);
    }
    sort(a, a + sz);
    for (int j = 0, k; j < sz; j = k) {
      T += a[j].first - X;
      X = a[j].first;
      for (k = j; k < sz && a[j].first == a[k].first; ++k) {
        ans[a[k].second] = T;
      }
      T += (k - j) / 2 + 1;
    }
    T += X;
    X = 0;
  }
  bool first = true;
  for (int i = 0; i < n; ++i) {
    if (first) first = false;
    else putchar(' ');
    printf("%d", ans[i]);
  }
  puts("");
  return 0;
}
