#include <iostream>
#include <queue>

#define MAX 100005
using namespace std;

class point
{
	public:
		int x,w;
};

int N,M,T;
int D[MAX];
int A[MAX];
priority_queue<point> q;
point t,tt;

bool operator<(point a,point b)
{
	return a.w > b.w;
}

int Go()
{
	int where=0,s=0,i;
	while(!q.empty())
	{
		t=q.top(); q.pop();
		if(t.w == where) 
		{
			A[s]=t.x;
			s++;
		}
		else
		{
			if(s)
			{
				for(i=0;i<s;i++) D[A[i]]=T;
				T+=1+(s/2);
			}
			T+=t.w-where;
			A[0]=t.x;
			s=1;
			where=t.w;
		}
	}
	if(s)
	{
		for(i=0;i<s;i++) D[A[i]]=T;
		T+=1+(s/2);
	}
	return where;
}

int main()
{
	int x,i,b;
	cin >> N >> M;
	for(i=1;i<=N;i++)
	{
		cin >> b >> t.w;
		t.x=i;
		q.push(t);
		if(b > T) T=b;
		if(i == N) x=Go();
		else if(!(i%M))
		{
			x=Go();
			T+=x;
		}
	}
	for(i=1;i<=N;i++) cout << D[i] << " "; cout << endl;
	return 0;
}
