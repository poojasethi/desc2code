#include <iostream>
#include <algorithm>
#include <cstdio>
#include <vector>
#include <queue>
#define f first
#define s second
#define mp make_pair
#define pb push_back
using namespace std;
priority_queue < pair < int ,int > > p1,p2,p3;
pair <int ,int > P,P2;
int nm,c;
void solve()
{
	int t=0,pl=0,s;
	while(!p2.empty())
	{
		while(p1.size()!=c && !p2.empty())
		{
			P=p2.top(); p2.pop();
			t=max(-P.f,t);
			swap(P.f,P.s);
			p1.push(P);
		}
		pl=0;
		while(!p1.empty())
		{
			P=p1.top(); p1.pop();
			t+=-P.f - pl;
			
			P2.f=P.s;
			P2.s=t;
			p3.push(P2);

			pl=-P.f;
			s=1;
			while(!p1.empty() && p1.top().f == P.f)
			{
				s++;
				P=p1.top();	p1.pop();

				P2.f=P.s;
				p3.push(P2);
			}
			t+=1+s/2;
		}
		t+=pl;
	}

	while(!p3.empty())
	{
		P=p3.top(); p3.pop();
		printf("%d ",P.s);
	}
}
int main()
{
	scanf("%d %d",&nm,&c);
	for(int i=1;i<=nm;i++)
	{
		scanf("%d %d",&P.f,&P.s);
		P.f*=-1;
		P.s*=-1;
		p2.push(P);
	}
	solve();
	return 0;
}
