#include <iostream>
#include <vector>
#include <map>
#define R(i,n) for(int i=0;i<(n);++i)
#define G(i,l,r) for(int i=(l);i<(r);++i)
#define MP make_pair
#define X first
#define Y second
#define T multimap<int,int>::iterator
using namespace std;
int main()
{
	int n,m,i=0,st=0,ct=0;
	cin >> n >> m;
	vector<int> t(n),x(n),r(n);
	R(i,n) cin >> t[i] >> x[i];
	while (i<n)
	{
		multimap<int,int> mmp;
		for (; i<n && i-st<m; ++i)
			mmp.insert(MP(x[i],i));
		ct=max(t[i-1],ct);
		for (T it=mmp.begin(); it!=mmp.end();)
		{
			int c=mmp.count(it->X);
			pair<T,T> l=mmp.equal_range(it->X);
			if (it==mmp.begin())
				ct+=it->X;
			else
			{
				T h=it;
				--h;
				ct+=it->X-h->X;
			}
			for (T jt=l.X; jt!=l.Y; ++jt)
				r[jt->Y]=ct;
			ct+=1+c/2;
			it=l.Y;
		}
		ct+=mmp.rbegin()->X;
		st=i;
	}
	cout << r[0];
	G(i,1,n) cout << ' ' << r[i];
	cout << endl;
	return 0;
}
