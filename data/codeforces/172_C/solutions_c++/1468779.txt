#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

bool solve()
{
	int n,m,tm=0;
	if (scanf("%d%d",&n,&m)==EOF)
		return false;
	vector<int> ans(n);
	vector<pair<int,int> > st;
	for (int i=0;i<n;++i)
	{
		int t,x;
		scanf("%d%d",&t,&x);
		st.push_back(make_pair(x,i));
		if (t>tm)
			tm = t;
		if (st.size()==m || i==n-1)
		{
			sort(st.begin(),st.end());
			x = 0;
			for (int i=0;i<st.size();)
			{
				tm += st[i].first-x;
				x = st[i].first;
				int k=0;
				for (k=0;i+k<st.size()&&st[i+k].first==x;++k)
					ans[st[i+k].second] = tm;
				tm += 1+k/2;
				i += k;
			}
			tm += x;
			st.clear();
		}
	}
	for (int i=0;i<n;++i)
		printf("%d ",ans[i]);
	printf("\n");
	return true;
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	while (solve());
#endif
	solve();
	return 0;
}
