#include <iostream>
#include <map>
#include <vector>

int main(){
	int n, m;
	int i;
	std::map<int, std::vector<int> > x;
	std::map<int, int> w;
	std::cin>>n>>m;
	int now = 0;
	for(;n ; n-=m){
		x.clear();
		int pos, time;
		if(n < m) m = n;
		for(i=0; i<m; ++i){
			int x_;
			std::cin>>time>>x_;
			x[x_].push_back(time);
		}
		pos = 0;
		if(now < time) now = time;
		for(std::map<int,std::vector<int> >::iterator i = x.begin();
			i != x.end(); ++i){
			now += i->first - pos;
			pos = i->first;
			for(std::vector<int>::iterator j = i->second.begin();
				j != i->second.end(); ++j){
				w[*j] = now;
			}
			now += 1+i->second.size()/2;
		}
		now += pos;
	}
	for(std::map<int,int>::iterator i = w.begin();
		i != w.end(); ++i)
		std::cout << i->second << ' ';
	std::cout << std::endl;
	return 0;
}
