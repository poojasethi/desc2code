#include<cstdio>
#include<cstring>
#include<algorithm>
#define fi first
#define se second
#define pii pair<int,int>
using namespace std;
int n,m,t[100010],nowt,pos,ans[100010];
pii a[100010];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;++i){
        scanf("%d%d",&t[i],&a[i].fi);
        a[i].se=i;
    }
    for(int i=1,j;i<=n;i=j+1){
        j=min(n,i+m-1);
        sort(a+i,a+j+1);
        nowt=max(nowt,t[j]);
        pos=0;
        for(int k=i,l;k<=j;k=l){l=k;
            while(l<=j&&a[l].fi==a[k].fi)++l;
            nowt+=a[k].fi-pos;
            for(int h=k;h<l;++h)ans[a[h].se]=nowt;
            nowt+=1+(l-k)/2;
            pos=a[k].fi;
        }
        nowt+=pos;
    }
    for(int i=1;i<=n;++i)
        printf("%d ",ans[i]);
    return 0;
}
