#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
int n,m;
int t[100001],x[100001],z[100001],tt[100001];
int main()
{
	int bt,bx,na,nb,qn;
	while(scanf("%d%d",&n,&m)==2)
	{
		bt=0;
		bx=0;
		na=0;
		nb=0;
		for(int i=0;i<n;i++) scanf("%d%d",&t[i],&x[i]);
		for(int i=0;i<n;i++) z[i]=x[i];
		while(1)
		{
			if(t[na]>bt) bt=t[na];
			nb=na;
			while((nb-na+1)<=m) 
			{
				if(t[nb]>bt) bt=t[nb];
				nb++;
				if(nb==n) break;
			}
			nb--;
			std::sort(z+na,z+nb+1);
			for(int i=na;i<=nb;i++)
			{
				int j=i;
				while(z[j]==z[i]) 
				{
					j++;
					if(j>nb) break;
				}
				bt=bt+z[i]-bx;
				bx=z[i];
				for(int s=na;s<=nb;s++) 
				{
					if(x[s]==bx) 
					{
						tt[s]=bt;
					}
				}
				bt=bt+(j-i)/2+1;
				i=j-1;
			}
			bt=bt+bx;
			bx=0;
			na=nb+1;
			if(nb==n-1) break;
		}
		for(int i=0;i<n;i++) printf("%d ",tt[i]);
		printf("\n");
	}
	return 0;
}
