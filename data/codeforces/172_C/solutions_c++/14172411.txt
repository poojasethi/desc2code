#include<cstdio>
#include<algorithm>

int n,m,i,j,k,p[110000],f[110000],Now,x[110000],T[110000],t;

bool cmp(int a,int b){return x[a]<x[b];}

int main()
{	
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)scanf("%d%d",T+i,x+i);
	for(i=0;i<n;i+=m)
	{	for(t=0,j=i;j<n&&j<i+m;j++)p[t++]=j;
		if(Now<T[j-1])Now=T[j-1];
		std::sort(p,p+t,cmp);
		for(j=0;j<t;j=k)
		{	Now+=x[p[j]];
            for(k=j;k<t&&x[p[j]]==x[p[k]];k++)f[p[k]]=Now;
			Now+=1+(k-j)/2;
			if(k<t)Now-=x[p[j]];else Now+=x[p[j]];
		}
	}
	for(i=0;i<n;i++)printf("%d ",f[i]);
	return 0;
}
