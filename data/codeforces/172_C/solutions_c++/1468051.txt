
#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;
typedef long long i8;

int n,m, t,x;
int arv[100111];
vector< pair<int,int> > bus;

main() {
	scanf("%d%d",&n,&m);
	
	int out=0, time=0;
	while (out<n) {
		// seat
		bus.clear();
		for (int i=0; i<m && out+i<n; i++) {
			scanf("%d%d",&t,&x);
			time=max(time,t);
			bus.push_back(make_pair(x,out+i));
		}
		
		// forward
		out += bus.size();
		sort(bus.begin(),bus.end());
		int pa=0, fr=0, to, cnt;
		while (pa<bus.size()) {
			to=bus[pa].first;
			time += (to-fr);
			cnt=1;
			arv[bus[pa].second]=time;
			while (1) {
				pa++;
				if (pa>=bus.size() || bus[pa].first>to) 
					break;
				arv[bus[pa].second]=time;
				cnt++;
			}
			time += 1+cnt/2;
			fr=to;
		}
		
		// back
		time += fr;
	}
	
	for (int i=0; i<n; i++)
		printf("%d ", arv[i]);
	printf("\n");
}
