#include<set>
#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n,m; scanf("%d%d",&n,&m);
	int t[100000],x[100000];
	rep(i,n) scanf("%d%d",t+i,x+i);

	int k=0; // 処理した学生の番号
	multiset<int> bus;
	for(int t_now=0;k<n;){
		// バスに学生が乗る
		int k0=k;
		while(k<n && bus.size()<m){
			bus.insert(x[k++]);
		}

		static int ans[10001];
		// バスが走る
		int t0=max(t_now,t[k-1]); // バスが出発した時刻
		int t1=0; // 降車にかかった時間
		int t2=*bus.rbegin(); // バスが終点まで進むのにかかる時間
		while(!bus.empty()){
			int x=*bus.begin();
			int c=bus.count(x);
			ans[x]=t0+t1+x;
			t1+=1+c/2;
			bus.erase(x);
		}
		t_now=t0+t1+2*t2;

		for(;k0<k;k0++) printf("%d ",ans[x[k0]]);
	}

	return 0;
}
