#include<iostream>
#include<algorithm>
#include<set>

using namespace std;

#define pip pair<pair<int, int>, int>
#define F first
#define S second
#define MP make_pair

set<pip > s;
pip a[100005];
int ans[100005];

int main()
{
	int n, m;
	cin >> n >> m;
	for(int i = 0; i < n; i++)
	{
		cin >> a[i].F.F >> a[i].F.S;
		a[i].S = i;
	}
	sort(a, a + n);
	int t = 0;
	for(int i = 0; i < n; i++)
	{
		int st;
		for(int j = i; j < min(n, i + m); j++)
		{
			s.insert(MP(MP(a[j].F.S, a[j].F.F), a[j].S));
			st = a[j].F.F;
		}
		t = max(t, st);
		while(s.size())
		{
			pip p = *s.begin();
			s.erase(p);
			t += p.F.F;
			ans[p.S] = t;
			int x = 1;
			while(p.F.F == (*s.begin()).F.F)
			{
				ans[s.begin()->S] = t;
				s.erase(s.begin());
				x++;
			}
			t += 1 + x/2;
			t -= p.F.F;
			if(s.size() == 0)
				t += 2 * p.F.F;
		}
		i += m - 1;
	}
	for(int i = 0; i < n; i++)
		cout << ans[i] << ' ';
	cout << endl;
	return 0;
}
