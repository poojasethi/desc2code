#include <map>
#include <iostream>
#include <cstdio>
#include <queue>
#define REP(i,n) for(int i=0; i<(int)(n); i++)

#define FOR(i,c) for(__typeof((c).begin()) i=(c).begin(); i!=(c).end(); i++)

inline int getInt(){ int s; scanf("%d", &s); return s; }

using namespace std;

int main(){
  int n = getInt();
  int m = getInt();
  int last = 0;

  map<int, vector<int> > bus;

  for(int i = 0; i < n; i++){
    int t = max(last, getInt());
    int x = getInt();

    bus[x].push_back(i % m);

    if(i % m == m - 1 || i == n - 1){
      vector<int> ret(i % m + 1);
      int extra = 0;

      FOR(it, bus){
    int k = it->second.size();

    FOR(v, it->second){
      ret[*v] = it->first + t + extra;
    }

    extra += 1 + k / 2;
      }

      last = bus.rbegin()->first * 2 + t + extra;

      bus.clear();

      REP(j, ret.size()) printf("%d%c", ret[j], i == n - 1 && j == ret.size() - 1 ? '\n' : ' ');
    }
  }

  return 0;
}
