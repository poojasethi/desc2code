#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <map>
using namespace std;

typedef long long LL;
const int N = 100005;
LL t[N], x[N], r[N];
int main(void) {
	int n, m;
	scanf("%d%d", &n, &m);
	LL last=0;
	for(int i=0;i<n;i++) scanf("%I64d%I64d", &t[i], &x[i]);
	for(int i=0;i<n;i+=m) {
		map<LL, LL> a, ot;
		LL st=last;
		for(int j=0;j<m && i+j<n;j++) {
			st = max(st, t[i+j]);
			a[x[i+j]]++;
		}
		LL sv=0;
		for(map<LL, LL>::iterator it=a.begin(); it!=a.end();it++) {
			ot[it->first] = sv;
			sv += 1+(it->second)/2;
		}
		for(int j=0;j<m && i+j<n;j++) {
			r[i+j] = st + x[i+j] + ot[x[i+j]];
			last = max(last, st + x[i+j] + sv+x[i+j]);
		}
	}
	for(int i=0;i<n;i++)
		printf("%I64d ", r[i]);
	puts("");
	return 0;
}
