#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
int n,m,i,j,k,t[100000],x[100000],l,r,T,f;
vector<int> y,z;
int main(){
	for(scanf("%d%d",&n,&m);i<n;++i)scanf("%d%d",t+i,x+i);
	for(i=0;l<n;l=r){
		r=min(l+m,n);
		for(y.clear();i<r;++i){
			y.push_back(x[i]);
			T=max(T,t[i]);
		}
		sort(y.begin(),y.end());
		z.clear();
		for(j=f=0;j<y.size();j=k){
			for(k=j+1;k<y.size()&&y[j]==y[k];++k);
			z.push_back(T+=(y[j]-f));
			T+=k-j+2>>1;f=y[j];
		}
		y.resize(unique(y.begin(),y.end())-y.begin());
		for(j=l;j<r;++j)
			t[j]=z[lower_bound(y.begin(),y.end(),x[j])-y.begin()];
		T+=f;
	}
	for(i=0;i<n;++i)printf("%d%c",t[i],((i+1==n)?'\n':' '));
	return 0;
}
