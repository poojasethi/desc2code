#include <iostream>
#include <map>
#include <vector>
using namespace std;

#define FOREACH(i, c) for(__typeof((c).begin()) i = (c).begin(); i != (c).end(); ++i)
#define FOR(i, a, b) for (int i = a; i < int(b); i++)
#define X first
#define Y second
#define ll long long

int n, m;
vector<ll> res;

int main() {
	cin >> n >> m;
	res.resize(n);
	map<int, vector<int> > mp;
	ll p = 0;
	FOR(cur, 0, n) {
		pair<ll, ll> V;
		cin >> V.X >> V.Y;
		mp[V.Y].push_back(cur);
		if ((cur+1)%m == 0 || cur+1 == n) {
			p = max(p, V.X);
			FOREACH(it, mp) {
				FOR(i, 0, it->Y.size()) res[it->Y[i]] = p+it->X;
				p += it->Y.size()/2+1;
			}
			p += 2*mp.rbegin()->X;
			mp.clear();
		}
	}
	FOR(i, 0, n) cout << res[i] << " "; cout << endl;
	return 0;
}
