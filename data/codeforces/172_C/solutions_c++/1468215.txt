#include <stdio.h>
#include <math.h>
#include <iostream>
#include <sstream>
#include <string.h>
#include <ctype.h>
#include <stdlib.h>
#include <vector>
#include <set>
#include <algorithm>
#include <map>
#include <queue>
#include <string>

using namespace std;

struct T
{
	int t,x,i;
}a[200000];

bool operator <(T t1,T t2)
{
	return t1.x<t2.x;
}

int n,m,i,l,to,j,k,o;
__int64 ans[200000],t;

int main()
{
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&a[i].t,&a[i].x);
		a[i].i=i;
	}
	for(i=0;i<n;i+=m)
	{
		l=i+m<n?i+m:n;
		t=max(t,(__int64)a[l-1].t);
		to=0;
		sort(a+i,a+l);
		for(j=i,k=i;j<l;j++)
			if(a[j].x!=a[k].x)
			{
				for(o=k;o<j;o++)
					ans[a[o].i]=t+to+a[k].x;
				to+=1+(j-k)/2;
				k=j;
			}
		for(o=k;o<l;o++)
			ans[a[o].i]=t+to+a[k].x;
		to+=1+(l-k)/2;
		t+=to+2*a[l-1].x;
	}
	for(i=0;i<n;i++)
		printf("%I64d ",ans[i]);
	printf("\n");
	return 0;
}
