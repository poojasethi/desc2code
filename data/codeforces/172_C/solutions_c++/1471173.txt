#include <cstdio>
#include <vector>
#include <map>
#include <algorithm>

using namespace std;

#define mp make_pair
#define pb push_back
#define F first
#define S second

vector<pair<int, pair<int, int > > > s;
vector<pair<int, pair<int, int > > > b;
int res[100000];

int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	
	for (int i = 0; i < n; ++i)
	{
		int a, b;
		scanf("%d%d", &a, &b);
		s.pb(mp(b,mp(a, i)));
	}
	
	int now = 0;
	for (int i = 0; i < n; )
	{
		b.clear();
		int pos = 0;
		while (i < n && b.size() < m)
			b.pb(s[i++]);
		if (now < b[b.size()-1].S.F) now = b[b.size()-1].S.F;
		sort(b.begin(), b.end());
		for (int j = 0; j < b.size(); )
		{
			int z = b[j].F;
			int c = 0, k = j;
			while (b[j].F == z && j < b.size()) ++j, ++c;
			now += z-pos;
			pos = z;
			while (k < j) res[b[k++].S.S] = now;
			now += 1+c/2;
		}
		now += pos;
	}
	
	for (int i = 0; i < n; ++i)
		printf("%d ", res[i]);
	printf("\n");
	
	return 0;
}
