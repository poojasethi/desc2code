#include<iostream>
#include<algorithm>
using namespace std;

#define N 100005
int n,m;
int t[N],x[N],ans[N];
int curtime,curpos;
pair<int,int>p[N];
int main()
{
    int i,j;
    cin>>n>>m;
    curpos=curtime=0;

    for(i=0;i<n;i++) cin>>t[i]>>x[i];
    
    for(i=0;i<n;i+=j)
    {

        for(j=0;i+j<n&&j<m;j++)p[j]=make_pair(x[i+j],i+j);
        
        if(curtime<t[i+j-1]) curtime=t[i+j-1];
        sort(p,p+j);
        for(int k=0;k<j;)
        {
            int a,b;

            for( a=0;k+a<j&&p[k+a].first==p[k].first;a++) ans[p[k+a].second]=p[k].first-curpos+curtime;
            curtime+=p[k].first-curpos+1+a/2;
            curpos=p[k].first;
            k+=a;
        }
        
        curtime+=curpos;
        curpos=0;
    }
    for(i=0;i<n;i++) cout<<ans[i]<<endl;
}
