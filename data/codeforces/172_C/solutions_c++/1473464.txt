#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,m;
long long tim,ans[200000];
struct Point {
	int a,b,c;
} p[200000];
bool cmp(Point const &a,Point const &b)
{
	return a.b<b.b;
}
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i) 
	{
		scanf("%d%d",&p[i].a,&p[i].b);
		p[i].c=i;
	}
	for (int i=1;i<=n;i+=m)
	{
		if (n-i<m) m=n-i+1;
		if (p[i+m-1].a>tim) tim=p[i+m-1].a;
		sort(p+i,p+m+i,cmp);
		int k=p[i].b,kk=1;
		for (int j=i+1;j<=i+m-1;++j)
			if (k==p[j].b)
			{
				kk++;
			}
			else
			{
				for (int l=j-1;l>=j-kk;--l)
					ans[p[l].c]=tim+p[l].b;
				tim+=1+kk/2;
				kk=1;k=p[j].b;
			}
			for (int l=i+m-1;l>=i+m-kk;--l)
				ans[p[l].c]=tim+p[l].b;
			tim+=1+kk/2;
		tim+=p[i+m-1].b+p[i+m-1].b;
	}
	for (int i=1;i<=n;++i) printf("%I64d ",ans[i]);
	cout<<endl;
	return 0;
}
