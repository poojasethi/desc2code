#include <cstdio>
#include <cstring>
#include <algorithm>
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
#define FOE(i,s,e) for (int i=(s); i<=(e); i++)
#define CLR(a, x) memset(a, x, sizeof(a))
#define LLD long long
using namespace std;

struct dat{
	int p, id;
	bool operator < (dat const &T) const{return p < T.p;}
};

int n, m, t, ts, st, sum, rt, res[100005];
dat a[100005];

int main(){
	scanf("%d%d", &n, &m);
	
	t = rt = 0;
	FOE(i,1,n){
		scanf("%d%d", &ts, &a[t].p), a[t].id = i-1;
		t++;
		if (i == n || i % m == 0){
			st = max(ts, rt);
			sort(a, a + t);
			sum = 0;
			for (int j=0, k=0; j<t; j=k){
				while (k < t && a[j].p == a[k].p){
					res[a[k].id] = st + a[k].p + sum;
					k++;
				}
				sum += 1 + (k - j) / 2;
			}
			rt = st + a[t-1].p * 2 + sum;
			t = 0;
		}
	}
	
	FOR(i,0,n) printf("%d%c", res[i], i < n-1 ? ' ' : '\n');
	return 0;
}
