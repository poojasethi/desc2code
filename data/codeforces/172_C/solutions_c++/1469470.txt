#include <iostream>
#include <queue>

#define MAX 100005
using namespace std;

class point
{
	public:
		int x,w;
};

int T,N,M,Dis[MAX],Ar[MAX];
point t;
priority_queue<point> q;

bool operator<(point a,point b)
{
	if(a.w == b.w) return a.x > b.x;
	return a.w > b.w;
}

int Bus()
{
	int wh=0,s=0,i;
	while(!q.empty())
	{
		t=q.top(); q.pop();
		if(t.w == wh) 
		{
			Ar[s]=t.x;
			s++;
		}
		else
		{
			if(s)
			{
				for(i=0;i<s;i++) 
					Dis[Ar[i]]=T;
				T+=s/2+1;
			}
			T+=t.w-wh;
			Ar[0]=t.x;
			s=1;
			wh=t.w;
		}
	}
	if(s)
	{
		for(i=0;i<s;i++) 
			Dis[Ar[i]]=T;
		T+=s/2+1;
	}
	return wh;
}

int main()
{
	cin >> N >> M;
	int x,i,bb;
	for(i=1;i<=N;i++)
	{
		t.x=i;
		cin >> bb >> t.w;
		q.push(t);
		if(bb > T) T=bb;
		if(i == N) x=Bus();
		else if((i%M) == 0)
		{
			x=Bus();
			T+=x;
		}
	}
	for(i=1;i<=N;i++) cout << Dis[i] << " "; cout << endl;
	return 0;
}
