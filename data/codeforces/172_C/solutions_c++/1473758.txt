#include <cstdio>
#include <algorithm>
#define N 100005
#define FI(a, b, c) for(int a = b; a <= c; a++)
#define FD(a, b, c) for(int a = b; a >= c; a--)
using namespace std;

int n, m, a[N], T, c;
struct pas{
	int t, x, o;
	bool operator < (const pas &T) const{ return x < T.x;}
} p[N];

int main(){
	scanf("%d %d", &n, &m);
	FI(i, 1, n) scanf("%d %d", &p[i].t, &p[i].x), p[i].o = i;
	
	FI(i, 1, n){
		int j = min(n, i + m - 1);
		T = max(T, p[j].t);
		c = 0;
		
		sort(p + i, p + j + 1);
		FI(k, i, j){
			int l = k;
			while(l < j && p[l + 1].x == p[k].x) l++;
			
			T += p[k].x - c;
			FI(x, k, l) a[p[x].o] = T;
			T += 1 + (l - k + 1) / 2;
			c = p[k].x;
			
			k = l;
		}
		
		T += c;
		i = j;
	}
	
	FI(i, 1, n) printf("%d%c", a[i], i == n ? '\n' : ' ');
	scanf("\n");
}
