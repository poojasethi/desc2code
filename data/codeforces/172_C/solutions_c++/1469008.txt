#include <iostream>
#include <algorithm>
#include <queue>
#define f first
#define s second
#define mp make_pair
#define put(x,y) {pq.push(mp(-x.s,y));Bus_size++;}
using namespace std;
typedef pair <int,int> ii;
int N,M,Time,Bus_size;
int In[100010];
ii A[100010];
priority_queue <ii> pq;
void read()
{
	int x,y;
	cin >> N >> M;
	for(int i=1;i<=N;i++)
		cin >> A[i].f >> A[i].s;
}
void Go()
{
	int x,y,yer=0,ind,say;
	while(!pq.empty()){
		x=-pq.top().f;	y=pq.top().s;
		Time+=x-yer;
		say=0;
		while(!pq.empty() && -pq.top().f == x){
			In[y]=Time;
			yer=x;
			pq.pop();
			y=pq.top().s;
			say++;
		}
		Time+=(say/2)+1;
	}
	Time+=yer;
	Bus_size=0;
}
void solve()
{
	for(int i=1;i<=N;i++){
		put(A[i],i);
		if(Time < A[i].f)
			Time=A[i].f;
		if(Bus_size == M || i == N)
			Go();
	}
}
void print()
{
	int i;
	for(i=1;i<=N;i++)
		cout << In[i] << " ";
	cout << endl;
}
int main()
{
	read();
	solve();
	print();
	return 0;
}
