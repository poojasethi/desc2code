#include <stdio.h>
#include <string.h>
#include <iostream>
#include <set>
#include <string>
#include <map>
using namespace std;

const int nmax=100005;
const int mmax=10005;
map<int,int>pew;
int xx[nmax];
int stop[mmax];

int solve(int);

int main()
{
	//freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
	int beg=0,n,x,m,i,j,t,k=0,tbegin=0;
	scanf("%d%d",&n,&m);
	for (i=0;i<n;i++)
	{
		scanf("%d%d",&t,&x);
		xx[i]=x;
		pew[x]++;k++;
		if (k==m || i==n-1)
		{
			tbegin=solve(max(t,tbegin));
			for (j=beg;j<=i;j++)
			{
				printf("%d ",stop[xx[j]]);
			}
			k=0;
			beg=i+1;
			pew.clear();
		}
	}
	return 0;
}

int solve(int tbegin)
{
	map<int,int>::iterator it=pew.begin();
	int x=0;int xxx=it->first;
	
	while (it!=pew.end())
	{	
		xxx=it->first;
		stop[xxx]=tbegin+(xxx)-x;
		tbegin+=1+xxx-x+it->second/2;
		it++;
		x=xxx;
	}
	return tbegin+xxx;
}
