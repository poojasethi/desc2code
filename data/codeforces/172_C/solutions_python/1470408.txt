#!/usr/bin/env python

from itertools import groupby

def chunks(l, n):
    for i in xrange(0, len(l), n):
        yield l[i:i+n]

class Passenger:
    def __init__(self, time, dest, no):
        self.time = time
        self.dest = dest
        self.no = no
        self.arrival = None

n, m = (int(x) for x in raw_input().split())
current_time = 0
passengers = []
for i in xrange(n):
    time, dest = (int(x) for x in raw_input().split()) 
    passengers.append(Passenger(time, dest, i))

for chunk in chunks(passengers, m):
    departure_time = max(current_time, chunk[-1].time)
    chunk.sort(key=lambda passenger: passenger.dest)
    total_unload_time = 0
    for dest, group in groupby(chunk, key=lambda passenger: passenger.dest):
        group = list(group)
        arrival_time = departure_time + dest + total_unload_time
        total_unload_time += 1 + len(group) / 2
        for passenger in group:
            passenger.arrival = arrival_time
    current_time = departure_time + 2 * chunk[-1].dest + total_unload_time
