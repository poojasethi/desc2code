n, m = [int(x) for x in raw_input().split(' ')]
students = [None] * n
times = [None] * n

    students[i] = tuple([int(x) for x in raw_input().split(' ')])
t = 0
x = 0
i = 0
while i != n:
    j = min((i + m, n))
    t = max((students[j - 1][0], t))
    locations = sorted([ k for k in range(i, j) ], lambda a, b: cmp(students[a][1], students[b][1]))
    left = 0
    while left != j - i:
        t += students[locations[left]][1] - x
        x = students[locations[left]][1]
        count = 0
        while left + count < j - i and students[locations[left + count]][1] == x:
