#include <iostream>

using namespace std;

int main(){
	int n, m, x, y, a[100100], b[100100], u[100100], v[100100], k=0;
	cin >> n >> m >> x >> y;
	for (int i = 0 ; i < n; i++){
		cin >> a[i];
	}
		
	for (int i = 0 ; i < m; i++){
		cin >> b[i];
	}
	int i=0, j=0;
	while (i < n && j < m){
		if (a[i]-x<=b[j] && b[j] <= a[i]+y){
			u[k]= ++i; v[k++]= ++j;
		} else if (b[j] < a[i]-x) {j++;
		} else i++;
	}

	cout << k << endl;
	for (int i = 0 ; i < k; i++){
		cout << u[i] << " " << v[i] << endl;
	}
	return 0;
}
