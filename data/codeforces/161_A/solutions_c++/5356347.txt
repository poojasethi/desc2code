#include <iostream>

using namespace std;
int n,a[100001],b[100001],c[100001],d[100001],x,y,kil=0,m,j,i;
int main()
{
    cin>>n>>m>>x>>y;
    for(i=1;i<=n;i++)cin>>a[i];
    for(i=1;i<=m;i++)cin>>b[i];j=1;
    for(i=1;i<=n;i++){
        if ((a[i]-x<=b[j]) && (a[i]+y>=b[j])){c[++kil]=i;d[kil]=j++;}
        else if (a[i]-x>=b[j]){ i--;j++;}
    if (j>m) break;}
    cout<<kil<<endl;
    for(i=1;i<=kil;i++)cout<<c[i]<<" "<<d[i]<<endl;
    return 0;
}
