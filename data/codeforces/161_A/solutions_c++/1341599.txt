#include <cstdio>

int n,m,x,y;
int a[100010];
int b[100010];
int na;
int ans[100010][2];

int main(){
	scanf("%d%d%d%d",&n,&m,&x,&y);
	for (int i=0; i<n; ++i)
		scanf("%d", a+i);
	for (int i=0; i<m; ++i)
		scanf("%d", b+i);
	int l=0, r=0;
	while (l<n && r<m){
		if (b[r]<a[l]-x)
			++r;
		else
			if (b[r]>a[l]+y)
				++l;
			else {
				ans[na][0]=l;
				ans[na][1]=r;
				++l;++r;
				++na;
			}
	}
	printf("%d\n", na);
	for (int i=0; i<na; ++i)
		printf("%d %d\n", ans[i][0]+1, ans[i][1]+1);
}
