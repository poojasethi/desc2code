#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int p[100001],c[100001],l1[100001],r1[100001];
int n,m,x,y,l,r,a;
int main()
{
	cin>>n>>m>>x>>y;
	int i;
	for(i=1;i<=n;i++)
		cin>>p[i];
	for(i=1;i<=m;i++)
		cin>>c[i];
	l=1,r=1;
	while(l<=n&&r<=m)
	{
		if(p[l]+y<c[r])
			l++;
		else if(p[l]-x>c[r])
			r++;
		else
			l1[++a]=l++,r1[a]=r++;
	}
	cout<<a<<endl;
	for(i=1;i<=a;i++)
		cout<<l1[i]<<" "<<r1[i]<<endl;
	return 0;
}
