#include <cstdio>
#define FOR(x,y,z) for(int x=y;x<=z;++x)
int n,m,x,y,a[100005],b[100005],id,ans,px[100005],py[100005];
int main(){
	scanf("%d%d%d%d",&n,&m,&x,&y);
	FOR(i,1,n) scanf("%d",&a[i]);
	FOR(i,1,m) scanf("%d",&b[i]);
	id=1;
	ans=0;
	FOR(i,1,n){
		while(id<=m&&(b[id]<a[i]-x)) ++id;
		if (id<=m&&b[id]<=a[i]+y){
			px[++ans]=i;
			py[ans]=id;
			++id;
		}
	}
	printf("%d\n",ans);
	FOR(i,1,ans) printf("%d %d\n",px[i],py[i]);
	return 0;
}
