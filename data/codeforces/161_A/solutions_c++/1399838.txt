#include<stdio.h>
int main()
{
	int y,x,a,b,c=0,d,e,f,i,j,A[1001000],B[101000],S[100100][2],z=1;
	scanf("%d %d %d %d",&a,&b,&x,&y);
	for(i=1;i<=a;i++)
		scanf("%d",&A[i]);
	for(i=1;i<=b;i++)
	{
		scanf("%d",&B[i]);
	}
	i=1;
	j=1;
	while(1)
	{
		if(j>b || i>a)
			break;
		if((A[i]-x)<=B[j] && (A[i]+y)>=B[j])
		{
			S[z][0]=i;
			S[z][1]=j;
			i++;
			j++;
			z++;
			continue;
		}
		if((A[i]+y)<B[j])
			i++;
		if((A[i]-x)>B[j])
			j++;
		
	}
	printf("%d\n",--z);
	for(i=1;i<=z;i++)
		printf("%d %d\n",S[i][0],S[i][1]);
	return 0;
}
