#include <iostream>
using namespace std;
#define M 100010
int A[M],B[M],RA[M],RB[M];

int main(){
	int i,j,n,m,x,y,k=0;
    cin>>n>>m>>x>>y;
    for ( i=1;i<=n;++i)
    	cin>>A[i];
    for ( i=1;i<=m;++i)
        cin>>B[i];
    for ( i=j=1;i<=n && j<=m;++i){
       	for (;j<=m && B[j]<A[i]-x;++j);
    	if (j<=m && B[j]<=A[i]+y)
	    	RA[k]=i,RB[k]=j,k++,j++;
	}
	cout<<k<<endl;
	for ( i=0;i<k;++i)
		cout<<RA[i]<<" "<<RB[i]<<endl;
	return 0;
}
