#include<iostream>
#include<algorithm>

using namespace std;
int ar[100010],br[100010],A[100010][2];

int main()
{
	int n,m,x,y,i,j,ans;
	cin>>n>>m>>x>>y;

	for(i=0;i<n;i++)
		cin>>ar[i];

	for(i=0;i<m;i++)
		cin>>br[i];

	i=j=ans=0;
	while(i<n && j<m)
	{
		if(ar[i]-x <= br[j] && ar[i]+y >= br[j])
		{
			A[ans][0]=i+1;
			A[ans][1]=j+1;
			i++;
			j++;
			ans++;
		}
		else if(ar[i]-x > br[j])
		{
			j++;
		}
		else
		{
			i++;
		}
	}
	cout<<ans<<endl;
	for(i=0;i<ans;i++)
		cout<<A[i][0]<<' '<<A[i][1]<<endl;

	return 0;

}
