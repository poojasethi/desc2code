#include<iostream>
using namespace std;
int main()
{
long n,m,x,y,i,j,c=0;
cin>>n>>m>>x>>y;
long *a= new long [n];
long *b= new long [m];
long *u= new long [n];
long *v= new long [m];
for(i=0; i<n; i++)
cin>>a[i];
for(j=0; j<m; j++)
cin>>b[j];
for(i=0,j=0; i<n && j<m; )
if(b[j]>=a[i]-x && b[j]<=a[i]+y)
{
u[c]=i+1;
v[c]=j+1;
c++;
j++;
i++;
}
else if(b[j]<a[i]-x)
j++;
else
i++;
cout<<c<<'\n';
for(i=0; i<c; i++)
cout<<u[i]<<' '<<v[i]<<'\n';
delete a;
delete b;
delete u;
delete v;
return 0;
}
