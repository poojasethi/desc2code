#include <iostream>
using namespace std;
int main()
{
int a[100005],b[100005],a1[100005],b1[100005];
int n,m,x,y,n1=0;
cin>>n>>m>>x>>y;
for (int i=1;i<=n;i++) cin>>a[i];
for (int i=1;i<=m;i++) cin>>b[i];
for (int i=1,j=1;i<=n&&j<=m;)
{
 if (a[i]-x<=b[j]&&b[j]<=a[i]+y)
 {n1++; a1[n1]=i;b1[n1]=j;j++;i++;continue;}
 if (a[i]-x>b[j]) {j++;continue;}
 if (b[j]>a[i]+y)  {i++;continue;}
 }
  cout<<n1<<endl;
  for (int i=1;i<=n1;i++) cout<<a1[i]<<" "<<b1[i]<<endl;
return 0;
}
