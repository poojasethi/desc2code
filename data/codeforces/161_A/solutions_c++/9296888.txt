#include <iostream>
#include <vector>

using namespace std;

int main()
{
  int n,m,x,y;
  cin>>n>>m>>x>>y;
  vector<int> a(n+1);
  for (int i=1;i<=n;i++)
    cin>>a[i];
  int ia=1;
  vector<pair<int,int> > sol;
  for (int j=1;j<=m and ia<=n;j++) {
    int b;
    cin>>b;
    while (ia<=n and b>a[ia]+y) ia++;
    if (ia<=n and a[ia]-x<=b) {
      sol.push_back(pair<int,int> (ia,j));
      ia++;
    }
  }
  cout<<int(sol.size())<<endl;
  for (int i=0;i<int(sol.size());i++)
    cout<<sol[i].first<<" "<<sol[i].second<<endl;
}
