#include <cstdio>

int main() {
	int n, m, x, y, i, j, count, A[100000], B[100000];
	int S[100000], V[100000];
	scanf("%d %d %d %d", &n, &m, &x, &y);
	for (i = 0; i < n; i++)
		scanf("%d", &A[i]);
	for (i = 0; i < m; i++)
		scanf("%d", &B[i]);
	count = 0;
	j = 0;
	for (i = 0; i < n; i++) {
		while (A[i] - x > B[j] && j != m)
			j++;
		if (j == m)
			break;
		if (A[i] + y >= B[j]) {
			S[count] = i + 1;
			V[count] = j + 1;
			count++;
			j++;
		}
	}
	printf("%d\n", count);
	for (i = 0; i < count; i++)
		printf("%d %d\n", S[i], V[i]);
	return 0;
}
