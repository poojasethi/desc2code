#include <cstdio>
#include <cstring>

#define N 100010

int main()
{
	int n,m,x,y;
	int a[N],b[N],link[N]={0};
	scanf("%d%d%d%d",&n,&m,&x,&y);
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	for(int i=0;i<m;i++) scanf("%d",&b[i]);
	int p=0,ans=0;
	for(int i=0;i<n;i++)
	{
	//	printf("%d %d\n",i,p);
		while(p<m && a[i]-x>b[p]) p++;
		if(p<m && b[p]<=a[i]+y)
		{
			link[i]=p+1;
			ans++; p++;
		}
	}
	printf("%d\n",ans);
	for(int i=0;i<n;i++) if(link[i])
		printf("%d %d\n",i+1,link[i]);
	return 0;
}
