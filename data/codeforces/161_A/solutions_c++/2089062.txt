#include<iostream>
using namespace std;
int a[100050],b[100050];
int main(){
	long long n,m,x,y;
	while(cin>>n>>m>>x>>y){
		for(int i=0;i<n;++i)cin>>a[i];
		for(int i=0;i<m;++i)cin>>b[i];
		int ans=0;
		for(int i=0,k=0;i<n&&k<m;){
			if(a[i]+y>=b[k]&&a[i]-x<=b[k]){
				ans++;i++;k++;
			}
			else if(a[i]+y<b[k]){
				i++;
			}
			else k++;
		}
		cout<<ans<<endl;
		for(int i=0,k=0;i<n&&k<m;){
			if(a[i]+y>=b[k]&&a[i]-x<=b[k]){
				i++;k++;
				cout<<i<<" "<<k<<endl;
			}
			else if(a[i]+y<b[k]){
				i++;
			}
			else k++;
		}
	}
}
