#include<stdio.h>
int a[100100],b[100100],aa[100100],ab[100100],t;
int main(){
	int n,m,x,y,i,j;
	scanf("%d %d %d %d",&n,&m,&x,&y);
	for(i=0;i<n;i++)
		scanf("%d",a+i);
	for(i=0;i<m;i++)
		scanf("%d",b+i);
	j=t=0;
	for(i=0;i<n;i++){
		for(;j<m&&b[j]<=a[i]+y;j++){
			if(a[i]-x<=b[j]&&b[j]<=a[i]+y){
				aa[t]=i+1;
				ab[t]=j+1;
				j++;
				t++;
				break;
				}
			}
		}
	printf("%d\n",t);
	for(i=0;i<t;i++)
		printf("%d %d\n",aa[i],ab[i]);
	//while(getchar());
	return 0;
	}
