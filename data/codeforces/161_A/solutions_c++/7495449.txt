#include<cmath>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,m,i,it,x,y,ans,a[400000],b[400000],v[400000][2];
int main(){
 scanf("%d%d%d%d",&n,&m,&x,&y);
 for(i=1;i<=n;++i)scanf("%d",&a[i]);
 for(i=1;i<=m;++i)scanf("%d",&b[i]);
 ans=0,it=1;
 for(i=1;i<=n;++i){
  while((b[it]<a[i]-x)&&(it<=m))++it;
  if(it>m)break;
  if(b[it]>a[i]+y)continue;
  v[++ans][1]=i,v[ans][2]=it,++it;
 }
 printf("%d\n",ans);
 for(i=1;i<=ans;++i)printf("%d %d\n",v[i][1],v[i][2]);
 return 0;
}
