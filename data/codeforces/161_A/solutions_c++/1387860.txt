#include<stdio.h>
#include <iostream>
#include<string.h>
using namespace std;
int a[1000000],b[1000000],c[1000000][3];
int n,k,x,y,i=0,j=0,s=0,z=0;
int main(){
	cin>>n>>k>>x>>y;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<k;i++)
		scanf("%d",&b[i]);
	i=0;
	j=0;
	while(1){
		if(b[j]>a[i]+y) i++;
		if(a[i]-x>b[j]) j++;
		if(a[i]-x<=b[j] && b[j]<=a[i]+y){
			s++;
			c[s][0]=i+1;
			c[s][1]=j+1;
			i++;
			j++;
		}
		if(i>=n || j>=k) break;
	}	
	cout<<s<<endl;
	for(i=1;i<=s;i++)
		printf("%d %d\n",c[i][0],c[i][1]);
	return 0;
}
