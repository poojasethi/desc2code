#include<cstdio>
#include<cstring>
const int MAXN=100010;
int a[MAXN],b[MAXN],n,m,x,y;
int l[MAXN],r[MAXN],ans=0;
int main() {
	scanf("%d %d %d %d",&n,&m,&x,&y);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(int i=0;i<m;i++)
		scanf("%d",&b[i]);
	int p=0;
	for(int i=0;i<n;i++) {
		while(b[p] < a[i]-x) p++;
		if(p>=m) break;
		if(a[i]-x <= b[p] && a[i]+y >= b[p]) {
			l[ans]=i+1,r[ans]=p+1;
			ans++,p++;
		}
	}
	printf("%d\n",ans);
	for(int i=0;i<ans;i++)
		printf("%d %d\n",l[i],r[i]);
	return 0;
}
