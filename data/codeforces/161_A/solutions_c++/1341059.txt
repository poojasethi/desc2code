#include<stdio.h>
#define ran 100010
int n,m,x,y,l;
int a[ran],b[ran];
int c[ran],d[ran];
int main(){
	scanf("%d%d%d%d",&n,&m,&x,&y);
	for(int i=0; i<n; i++)
		scanf("%d",&a[i]);
	for(int i=0; i<m; i++)
		scanf("%d",&b[i]);
	l=0;
	for(int i=0,j=0; i<n; i++){
		for(; j<m && b[j]<a[i]-x; j++);
		if(j==m || b[j]>a[i]+y)continue;
		c[l]=i;
		d[l++]=j;
		j++;
	}
	printf("%d\n",l);
	for(int i=0; i<l; i++)
		printf("%d %d\n",c[i]+1,d[i]+1);
	return 0;
}
