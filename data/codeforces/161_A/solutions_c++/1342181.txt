#include<stdio.h>
int n,m;
int x,y;
int ans;
int aa[200000];
int ab[200000];
int a[200000];
int b[200000];
int main(){
	scanf("%d %d %d %d",&n,&m,&x,&y);
	for(int i = 0 ; i < n; i++ ) scanf("%d",&a[i]);
	for(int i = 0 ; i < m; i++ ) scanf("%d",&b[i]);
	int pos = 0 ;
	for(int i = 0 ; i < m ; i++ ){
		for( ; pos < n && a[pos]-x <= b[i] ; ){
			if( b[i] <= a[pos]+y ){
				aa[ans] = pos;
				ab[ans] = i;
				ans++;
				pos++;
				break;
			}
			pos++;
		}
	}
	printf("%d\n",ans);
	for(int i = 0 ; i < ans ; i++ ) printf("%d %d\n",aa[i]+1,ab[i]+1);
}
