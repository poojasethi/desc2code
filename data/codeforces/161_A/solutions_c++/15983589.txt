#include<iostream>
using namespace std;
int main()
{
	int n,m,x,y,i=0,j=0,count=-1;
	cin>>n>>m>>x>>y;
	int a[n],b[m];
	for(i=0;i<n;i++)cin>>a[i];
	for(i=0;i<m;i++)cin>>b[i];
	int l[100001][2];i=0;
	while(i<n&&j<m)
	{
		if(b[j]<=(a[i]+y)&&b[j]>=a[i]-x)l[++count][0]=i+1,l[count][1]=j+1,i++,j++;
		else if(b[j]<a[i]-x)j++;
		else i++;
	}
	cout<<count+1<<endl;
	for(int i=0;i<=count;i++)cout<<l[i][0]<<" "<<l[i][1]<<endl;
}
