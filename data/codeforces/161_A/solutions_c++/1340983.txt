#include <iostream>
#include <algorithm>
using namespace std;
const int MN=1<<20;
int as[MN], bs[MN];
int rs[MN][2];
int main() {
	int n,m,x,y;
	cin>>n>>m>>x>>y;
	for(int i=0; i<n; ++i) cin>>as[i];
	for(int i=0; i<m; ++i) cin>>bs[i];
	int ai=0,bi=0;
	int r=0;
	while(ai<n && bi<m) {
		int a = as[ai];
		int b = bs[bi];
		if (b<a-x) {
			++bi;
		} else if (b>a+y) {
			++ai;
		} else {
			rs[r][0]=++ai;
			rs[r++][1]=++bi;
		}
	}
	cout<<r<<'\n';
	for(int i=0; i<r; ++i) {
		cout<<rs[i][0]<<' '<<rs[i][1]<<'\n';
	}
}
