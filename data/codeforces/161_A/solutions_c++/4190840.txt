#include <iostream>

using namespace std;

int main(){
    long long n,m,x,y,j=0; int *a,b,answ[100001][2]={0}; cin >> n >> m >> x >> y; a=new int[n];
    for(int i=0 ; i<n ; i++) cin >> a[i];
    for(int i=0 , k=0 ; i<n && k<m ; k++){
        cin >> b;
        while( b>a[i]+y && i<n) i++; 
        if(a[i]-x<=b && b<=a[i]+y && i<n ){ answ[j][0]=i+1; answ[j][1]=k+1; j++; i++; }
    }
    cout << j << endl;
    for(int i=0 ; i<j ;i++){
        cout << answ[i][0] << " " << answ[i][1] << endl;
    }
}
