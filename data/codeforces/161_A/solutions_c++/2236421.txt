#include<stdio.h>
int a[100010];
int b[100010];
int c[100010][2];
int main(){
	int i,j,n,m,x,y,top;
	scanf("%d%d%d%d",&n,&m,&x,&y);
	for(i=1;i<=n;i++)
	    scanf("%d",&a[i]);
	for(j=1;j<=m;j++)
	    scanf("%d",&b[j]);
	for(i=1,j=1,top=0;i<=n;i++)
	    for(;j<=m&&b[j]<=a[i]+y;j++)
		    if(b[j]>=a[i]-x){
			c[top][0]=i;
			c[top][1]=j;
			j++;
			top++;
			break;
			}
	printf("%d\n",top);
	for(i=0;i<top;i++)
	    printf("%d %d\n",c[i][0],c[i][1]);
	return 0;
	}
