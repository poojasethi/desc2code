#include <iostream>
#include <math.h>

using namespace std;

int main()
{
    int i=1,j=1,k=0,l,m,n,x,y;
    cin>>n>>m>>x>>y;
    int a[n],b[m];
    for (i=0;i<n;i++) cin>>a[i];
    for (j=0;j<m;j++) cin>>b[j];
    i=0;j=0;
    while (i<n&&j<m) if (a[i]-x>b[j]) j++; else if (a[i]+y<b[j]) i++; else {i++;j++;k++;}
    cout<<k<<endl;
    i=0;j=0;
    while (i<n&&j<m) if (a[i]-x>b[j]) j++; else if (a[i]+y<b[j]) i++; else cout<<++i<<" "<<++j<<endl;
    return 0;
}
