#include<cstdio>
#include<algorithm>
using namespace std;
int a1[100000],a2[100000],a3[100000];
int main()
{
	int n,m,a,b,c=0,o=0;
	scanf("%d%d%d%d",&n,&m,&a,&b);
	for(int i=0;i<n;++i)scanf("%d",a1+i);
	for(int i=0;i<m;++i)scanf("%d",a2+i);
	sort(a1,a1+n);sort(a2,a2+m);
	for(int i=0;i<m;++i)
	{
		while(c<n&&a2[i]>a1[c]+b)++c;
		if(c==n)break;
		if(a2[i]<a1[c]-a)continue;
		a3[i]=++c;++o;
	}
	printf("%d\n",o);
	for(int i=0;i<m;++i)
	if(a3[i])printf("%d %d\n",a3[i],i+1);
}
