#include<bits/stdc++.h>
#define pb push_back
using namespace std;
int main()
{
	int n,m,x,y,i,j,k=0,a[100010],b[100010];
	vector<int> p;
	cin>>n>>m>>x>>y;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<m;i++)
		scanf("%d",&b[i]);
	for(i=0,j=0;i<n && j<m;i++)
	{
		if(a[i]-x<=b[j] && b[j]<=a[i]+y)
		{
			k++;
			p.pb(i+1);
			p.pb(j+1);
			j++;
		}
		else if(b[j]<a[i]-x)
		{
			j++;
			i--;
		}
	}
	cout<<k<<endl;
	for(i=0;i<p.size();i+=2)
		printf("%d %d\n",p[i],p[i+1]);
	return 0;
}
