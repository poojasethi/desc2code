#include<cstdio>
#define N 100005
int b[N],c[N];
struct node{
	int l,r;
}a[N];
int main()
{
	//freopen("in.txt","r",stdin);
	int i,j,s,n,m,x,y,num;
	while(~scanf("%d%d%d%d",&n,&m,&x,&y)){
	for(i=1; i<=n; i++)
	{
		scanf("%d",&num);
		a[i].l = num-x;
		a[i].r = num+y;
	}
	for(s=j=i=1;i<=m;i++)
	{
		scanf("%d",&num);
		for(;a[j].l<=num && j<=n;j++)
			if(a[j].r>=num)
			{
				b[s] = j;
				c[s] = i;
				s++;
				j++;
				break;
			}
	}
	printf("%d\n",s-1);
	for(i=1;i<s;i++)
		printf("%d %d\n",b[i],c[i]);}
	return 0;
}
