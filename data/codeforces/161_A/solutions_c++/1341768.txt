#include <stdio.h>
main()
{
int n,m,x,y,i,j,k=0,a[100100],b[100100],u[100100],v[100100];
scanf("%d %d %d %d",&n,&m,&x,&y);
for (i=0; i<n; i++)
scanf("%d",&a[i]);
for (i=0; i<m; i++)
scanf("%d",&b[i]);
i=j=0;
while (i<n && j<m)
{
if (b[j]>=a[i]-x && b[j]<=a[i]+y) { u[k]=i+1; v[k]=j+1; k++; i++; j++; continue; }
if (b[j]<a[i]-x) j++;
if (b[j]>a[i]+y) i++;
}
printf("%d\n",k);
for (i=0; i<k; i++)
printf("%d %d\n",u[i],v[i]);
return 0;
}
