#include <iostream>
#include <vector>

using namespace std;

int main () {
	int i, n, m, x, y, f = 0, s = 0;
	cin >> n >> m >> x >> y;
	vector<int> a(n), b(m);
	vector< pair<int,int> > v;
	for (i = 0; i < n; ++i)
		cin >> a[i];
	for (i = 0; i < m; ++i)
		cin >> b[i];
	while (f < n && s < m)
		if (b[s] < a[f] - x)
			++s;
		else if (b[s] > a[f] + y)
			++f;
		else
			v.push_back (make_pair (++f, ++s));
	cout << v.size() << endl;
	for (i = 0; i < v.size(); ++i)
		cout << v[i].first << " " << v[i].second << endl;
	return 0;
}
