#include <cstdio>
#include <algorithm>
using namespace std;

int main(){
	int n,m,x,y; scanf("%d %d %d %d",&n,&m,&x,&y);
	int a[100005]; int b[100005]; pair<int,int> pii[100005]; int cc=0;
	for(int i=0;i<n;i++) scanf("%d",&a[i]);
	for(int i=0;i<m;i++) scanf("%d",&b[i]);
	for(int i=0,j=0;i<n;i++){
		while(j<m&&b[j]<a[i]-x)
			j++;
		if(j<m&&b[j]<=a[i]+y){
			pii[cc++]=make_pair(i+1,j+1);
			j++;
		}
	}
	printf("%d\n",cc);
	for(int i=0;i<cc;i++)
		printf("%d %d\n",pii[i].first,pii[i].second);
	return 0;
}
