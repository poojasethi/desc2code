#include<iostream>
using namespace std;
int main(){
    int n,m,x,y,a[100000],b[100000],u[100000],v[100000],d=0;
    cin>>n>>m>>x>>y;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<m;i++)
        cin>>b[i];
    int p=0,q=0;
    while(p<n&&q<m){
        if(a[p]-x>b[q])
            q++;
        else if(a[p]+y<b[q])
            p++;
        else
            u[d]=++p,v[d++]=++q;
    }
    cout<<d<<" ";
    for(int i=0;i<d;i++)
        cout<<u[i]<<" "<<v[i]<<" ";    
}
