#!/usr/bin/env python2

n, m, x, y = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
i = j = 0

result = []

try:
    for i in range(len(a)):
        while a[i] - x > b[j]:
            j += 1
        if a[i] + y >= b[j]:
            result.append((i + 1, j + 1))
            j += 1
except IndexError:
    pass
print(len(result))

for (i, j) in result:
    print i, j
