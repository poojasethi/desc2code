r=lambda:map(int,raw_input().split())
n,m,x,y=r()
a=r()
b=r()
i=0;j=0
s=[]
while(True):
    if a[i]-x<=b[j]<=a[i]+y:
        i+=1
        j+=1
        s.append(str(i)+" "+str(j))
    elif a[i]<b[j]:
        i+=1
    else:
        j+=1
    if i>=n or j>=m:
        break
l=len(s)
print l
for x in s:
    print x
