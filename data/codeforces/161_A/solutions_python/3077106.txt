n,m,x,y = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
i = 0
j = 0
ans = []
while i<n and j<m:
  if a[i]-x > b[j]:
    j+=1
  elif a[i]+y < b[j]:
    i+=1
  else:
    ans.append((i+1, j+1))
    i+=1
    j+=1
print len(ans)
for an in ans:
  print an[0], an[1]
