import string

def ii():
    ii = []
    ss = string.split(raw_input())
    for s in ss:
        ii.append(int(s))
    return ii

a = ii()
n = a[0]
m = a[1]
x = a[2]
y = a[3]

i = 0
j = 0

k = 0
ps = []
nn = ii()
mm = ii()
while (i < n) and (j < m):
    d = mm[j] - nn[i]
    if (d < -x):
        j += 1
    elif (d > y):
        i += 1
    else:
        ps.append((i + 1, j + 1))
        i += 1
        j += 1
        k += 1
print k
for p in ps:
    print str(p[0]) + ' ' + str(p[1])
