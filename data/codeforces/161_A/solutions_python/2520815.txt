n, m, x, y = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
c = []
j = 0
for i, v in enumerate(a):
	while j < m:
		if v + y < b[j]:
			break
		elif b[j] < v - x:
			j = j + 1
		else:
			c.append((i + 1, j + 1))
			j = j + 1
			break
print len(c)
print '\n'.join(' '.join(map(str, x)) for x in c)
