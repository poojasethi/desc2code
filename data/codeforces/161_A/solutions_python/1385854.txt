#!/usr/bin/env python

import sys
import pdb

fun = lambda:map(int, raw_input().split())

n, m, x, y = fun()

a = fun() + [9e9]
b = fun()
c = []

i = 0

for j, bj in enumerate(b):

	if i == n: break 

	while (a[i] + y < bj):
		i += 1

	if a[i] - x <= bj:
		c += [(i+1,j+1)]
		i += 1

print len(c)
for a, b in c:
	print a, b
