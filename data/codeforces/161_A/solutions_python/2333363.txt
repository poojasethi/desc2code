army,vests,x,y = map(int,raw_input().split(' '))
armySizes = map(int,raw_input().split(' '))
vestSizes = map(int,raw_input().split(' '))

presA,presV=0,0
ret=[]
while(presA<army and presV<vests):
    A = armySizes[presA]
    V = vestSizes[presV]
    if( V <= (A+y) and V>= (A-x)):
        ret.append((presA+1,presV+1))
        presA+=1
        presV+=1
    elif (V > (A+y)):
        presA+=1
    elif (V < (A-x)):
    	presV+=1

print len(ret)
for t in ret:
	print t[0],t[1]
