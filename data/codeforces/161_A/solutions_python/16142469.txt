n, m, x, y = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
cnt = 0
ans = []
i = j = 0
while i < n and j < m:
    lower, upper = a[i] - x, a[i] + y
    if b[j] < lower:
        j += 1
    elif b[j] > upper:
        i += 1
    else:
        ans.append([i+1, j+1])
        i += 1
        j += 1
        cnt += 1
print cnt
for i in range(cnt):
    print ans[i][0], ans[i][1]
