n,m,x,y = map(int,raw_input().split())
a = map(int,raw_input().split())
a.sort()
b = map(int,raw_input().split())
b.sort()
i = j = 0
c = []
while i<n and j<m:
    if a[i]<b[j]-y:
        i = i+1
    elif a[i]>b[j]+x:
        j = j+1
    else:
        c.append([i+1,j+1])
        i = i + 1
        j = j + 1
print len(c)
for i in c: 
    print ' '.join(str(h) for h in i)
