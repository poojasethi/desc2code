from sys import stdin as I

n, m, x, y = map(int, I.readline().split())
a = map(int, I.readline().split())
b = map(int, I.readline().split())

a.sort()
b.sort()
i = 0
j = 0

rv = []
while i < n and j < m:
    if a[i]-x <= b[j] <= a[i] + y:
        rv.append((i+1, j+1))
        i += 1
        j += 1
    elif b[j] < a[i]-x:
        j += 1
    elif a[i]+y < b[j]:
        i += 1
        
print len(rv)
for x, y in rv:
    print x, y
