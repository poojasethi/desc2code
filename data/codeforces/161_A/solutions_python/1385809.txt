#!/usr/bin/python -O
import sys
import collections

n, m, x, y = tuple(map(int, sys.stdin.readline().split()))
a = map(int, sys.stdin.readline().split())
b = map(int, sys.stdin.readline().split())
sol = []
for i in xrange(len(b) - 1, -1, -1):
  if not a:
    break
  if b[i] > a[-1] + y:
    continue
  while a and b[i] < a[-1] - x:
    a.pop()
  if a and a[-1] - x <= b[i] <= a[-1] + y:
    a.pop()
    sol.append((len(a) +1, i + 1))
print len(sol)
for (u, v) in sol:
  print u, v
