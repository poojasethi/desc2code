n,m,x,y = map(int, raw_input().split())
a = map(int, raw_input().split()) # soldiers
b = map(int, raw_input().split()) # armors
ai = 0
bi = 0
c = 0

s = ''

while ai < n and bi < m:
  if a[ai]-x <= b[bi] <= a[ai]+y:
    s += str(ai+1) + ' ' + str(bi+1) + '\n'
    ai += 1
    bi += 1
    c += 1
  else: 
    if a[ai]-x > b[bi]:
      bi += 1
    else:
      ai += 1

print str(c) + '\n' + s
