n,m,xx,yy = map(int, raw_input().split())
a = map(int, raw_input().split())
aa = [(x,i+1) for i,x in enumerate(a)]
b = map(int, raw_input().split())
bb = [(x,i+1) for i,x in enumerate(b)]
aa.sort()
bb.sort()
r = []
i = 0
#print aa,bb,xx,yy
for x,xn in bb:
    while i<n and aa[i][0]+yy < x: i+=1
    #print i, aa[i], x
    if i>=n: break
    y,yn = aa[i]
    if y-xx > x: continue
    i+=1
    r.append((yn,xn))
print len(r)
for a,b in r:
    print a,b
