import sys

n,m,x,y,=map(int,raw_input().split())
a=map(int,raw_input().split())
b=map(int,raw_input().split())

pa,pb=0,0
r={}
while pa<n and pb<m:
    if b[pb]>=a[pa]-x and b[pb]<=a[pa]+y:
        r[pa+1]=pb+1
        pa+=1
        pb+=1
    else:
        if a[pa]-x > b[pb]:
            pb+=1
        else:
            pa+=1
        
print len(r)
for x in r.keys():
    print x,r[x]
