n, m, x, y = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())

res ={}

k = 0
s = 0
i = 0
while (k < n):	
	while (i < m) and (b[i] < a[k] - x):
		i += 1
	if (i < m):
		if (b[i] <= a[k] + y):
			s += 1
			res[k] = i
			i += 1		
	k += 1

print s
for u, v in res.items():
	print u + 1, v + 1
