I = lambda:map(int, raw_input().split())
n,m,x,y = I()
a = I()
b = I()
i, j = 0, 0
r = []
while i < n and j < m:
    if a[i] - x <= b[j] and a[i] + y >= b[j]:
        r.append((i+1, j+1))
        i, j = i+1, j+1
    elif b[j] > a[i] + y:
        i += 1
    else:
        j += 1
print len(r)
for xa, xb in r:
    print xa, xb
