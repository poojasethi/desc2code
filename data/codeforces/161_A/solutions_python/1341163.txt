
n,m,x,y = map(int,raw_input().split())
S = map(int,raw_input().split())
V = map(int,raw_input().split())
j = 0

res = []

for i,s in enumerate(S):
    while j<m and s-x>V[j]:
        j+=1
    if j<m and s-x<=V[j]<=s+y:
        res.append((i+1,j+1))
        j+=1

print len(res)
for x,y in res:
    print x,y
