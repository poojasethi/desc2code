r=lambda:map(int,raw_input().split())

n,m,x,y=r()
a=r()
b=r()
i=0
j=0
r=[]
while i < n and j < m:
    if a[i] + y < b[j]:
        i += 1
    elif a[i] - x > b[j]:
        j += 1
    else:
        i += 1
        j += 1
        r += [[i,j]]

print len(r)
print '\n'.join(map(lambda x: ' '.join(map(str, x)), r))
