n, m, x, y = map(int, raw_input().split())
ai = map(int, raw_input().split())
bi = map(int, raw_input().split())
i = 0
ans = []
for j in range(m):
    while i < n and bi[j] > ai[i] + y: i += 1
    if i < n and bi[j] >= ai[i] - x:
        ans.append((i, j))
        i += 1
print len(ans)
for en in ans:
    print en[0]+1, en[1]+1
