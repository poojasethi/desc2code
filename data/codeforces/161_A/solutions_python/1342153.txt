n, m, x, y = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
a.sort()
b.sort()
i = 0
j = 0
k = 0
c = []
while (i < n) & (j < m):
        if (a[i] - x <= b[j]) & (b[j] <= a[i]+y):
                k += 1
                c.append([i+1, j+1])
                i += 1
                j += 1
        elif a[i] < b[j]:
                i += 1
        else:
                j += 1
print k
#print c
for i in xrange(k):
       print c[i][0], c[i][1]
