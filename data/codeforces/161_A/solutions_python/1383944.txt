n, m, x, y = map(int, raw_input().split(' '))

w = map(int, raw_input().split(' '))
b = map(int, raw_input().split(' '))
res = []
cnt = 0
k2 = 0
for k1, j in enumerate(b):
    while (k2 < n) and (w[k2] + y < j):
        k2 += 1
    if (k2 >= n):
        break
    if (w[k2] - x <= j):
        res.append([k2 + 1, k1 + 1])
        cnt +=1
        k2 += 1

print cnt
for i in res:
    print i[0], i[1]
