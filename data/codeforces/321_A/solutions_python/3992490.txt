I=lambda:map(int, raw_input().split())
a,b=I()
s=raw_input()
n=len(s)
d=[(0,0)]*(n+1)
x,y=0,0
for i in xrange(n):
    c=s[i]
    if c=='U':y+=1
    elif c=='R':x+=1
    elif c=='D':y-=1
    else:x-=1
    d[i+1]=(x,y)
dx,dy=x,y
for x,y in d:
    px,py=a-x,b-y
    if (x==a and y==b) or (dx and px%dx==0 and px/dx>0 and px/dx*dy==py) or (dy and py%dy==0 and py/dy>0 and py/dy*dx==px):
        print 'Yes';exit()
print 'No'
