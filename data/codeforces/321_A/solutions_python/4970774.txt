a,b=map(int,raw_input().split())
s=raw_input()
x=s.count('R')-s.count('L')
y=s.count('U')-s.count('D')
k=max(b/y if y else 1,a/x if x else 1) - 30
if k>0:
    a-=k*x
    b-=k*y
#print a,b,k,x,y
i=0
while (a != 0 or b!=0) and i<70001:
    j=i%len(s)
    if s[j] == 'R': a-=1
    elif s[j] == 'L': a+=1
    elif s[j] == 'U': b-=1
    elif s[j] == 'D': b+=1
#    if i > 290: print s[j],a,b
    i+=1
    
if a==b==0: print "Yes"
else: print "No"
