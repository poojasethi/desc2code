R=raw_input
a,b=map(int,R().split())
s=R()
def f(s):
 c=s.count
 return c('R')-c('L'),c('U')-c('D')
X,Y=f(s)
x,y=0,0
q=0
for i in range(len(s)):
 c,d=f(s[:i])
 A=a-c;B=b-d
 if (X and A%X) or (Y and B%Y):continue
 if X==Y==0:
  if A==B==0:q=1
 elif X==0:
  if A==0 and B//Y>=0:q=1
 elif Y==0:
  if B==0 and A//X>=0:q=1
 elif A//X==B//Y and A//X>=0:q=1
print["No","Yes"][q]
