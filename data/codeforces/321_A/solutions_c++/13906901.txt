#include<bits/stdc++.h>

using namespace std;

typedef long long ll;

int main()
{
    string s;
    int a,b,xx,yy;
    int n,m;
    cin>>n>>m;
    cin>>s;
    a=b=0;

    for (int i=0;i<s.size();i++)
    {
        if (s[i]=='L') a--;
        else if (s[i]=='R') a++;
        else if (s[i]=='U') b++;
        else b--;
    }
    int p;
    if (a) p=n/a;
    else if (b) p=m/b;
    else p=0;

    for (int j=max(0,p-100);j<=p+100;j++)
    {
        xx=yy=0;
        for (int i=0;i<s.size();i++)
        {
            if (a*j+xx==n&&b*j+yy==m)
            {
                cout<<"Yes";
                return 0;
            }

        if (s[i]=='L') xx--;
        else if (s[i]=='R') xx++;
        else if (s[i]=='U') yy++;
        else yy--;
        }
    }
    cout<<"No";
    return 0;
}
