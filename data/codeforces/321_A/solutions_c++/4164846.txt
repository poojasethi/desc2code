#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,x,y,X,Y,xx,yy,p;
char c[12000];
int main(){
	scanf("%d%d",&X,&Y);
	scanf("%s",c+1);
	n=strlen(c+1);
	x=y=0;
	for(i=1;i<=n;++i){
		if(c[i]=='R')x++;
		if(c[i]=='L')x--;
		if(c[i]=='D')y--;
		if(c[i]=='U')y++;
	}
	if(x)p=X/x;
	else if(y)p=Y/y;
	else p=0;
	for(j=max(0,p-100);j<=p+100;++j){
		xx=yy=0;
		for(i=1;i<=n;++i){
			if(x*j+xx==X && y*j+yy==Y){
				printf("Yes\n");
				return 0;
			}
			if(c[i]=='R')xx++;
			if(c[i]=='L')xx--;
			if(c[i]=='D')yy--;
			if(c[i]=='U')yy++;
		}
	}
	printf("No\n");
}
