#include <string>
#include <algorithm>
#include <iostream>
#include <cstdio>
#include <map>

using namespace std;

int xx[110],yy[110],cnt;

void upd(int x,int y)
{
    ++cnt;
    xx[cnt]=x,yy[cnt]=y;
}
int jud(long long a,long long b,long long aa,long long bb)
{
    if(a*bb-b*aa)
	return 0;
    if(a==0&&b==0)
	return 1;
    if(aa==0&&bb==0)
	return 0;
    if(a*aa<0||b*bb<0)
	return 0;
    if(aa&&a%aa)
	return 0;
    if(bb&&b%bb)
	return 0;
    return 1;
}
char s[110];
int main()
{
    int a,b;
    int i,j,k;
    scanf("%d%d",&a,&b);
    scanf("%s",s);

    int x=0,y=0;
    for(i=0;s[i];i++)
    {
	if(s[i]=='U')
	    y++;
	else if(s[i]=='D')
	    y--;
	else if(s[i]=='L')
	    x--;
	else
	    x++;
	upd(x,y);
    }
    for(i=0;i<=cnt;i++)if(jud(a-xx[i],b-yy[i],xx[cnt],yy[cnt]))
    {
	puts("Yes");
	return 0;
    }
    puts("No");
}
