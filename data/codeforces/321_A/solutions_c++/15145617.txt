#include<bits/stdc++.h>
using namespace std;
int main()
{
    //return 0;
    int X,Y,x=0,y=0,dx,dy,g[2][109]={0};
    string s;
    cin>>X>>Y>>s;
    for(int i=0;i<s.length();i++)
    {
        if(s[i]=='L')x--;
        if(s[i]=='R')x++;
        if(s[i]=='D')y--;
        if(s[i]=='U')y++;
        g[0][i+1]=x;
        g[1][i+1]=y;
    }
    
    for(int i=0;i<=s.length();i++)
    {
        int p=X-g[0][i],q=Y-g[1][i];
        int k=1;
        if(x!=0)k=p/x;
        if(y!=0)k=q/y;
        if(k>=0 && p==k*x && q==k*y)
        {
            cout<<"Yes";return 0;
        }
    }
    cout<<"No";
    return 0;
}
