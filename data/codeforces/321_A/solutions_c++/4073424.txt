#include<string.h>
#include<stdio.h>
int main()
{
	int l,i,j,n,m,a,b,x[102],y[102],t,flag;
	char str[102];
	while(scanf("%d%d%s",&n,&m,str+1)!=EOF)
	{
		memset(x,0,sizeof(x));
		memset(y,0,sizeof(y));
		l=strlen(str+1);
		for(i=1;i<=l;i++)
		{
			if(str[i]=='U') {x[i]=x[i-1];y[i]=y[i-1]+1;}
			if(str[i]=='D') {x[i]=x[i-1];y[i]=y[i-1]-1;}
			if(str[i]=='L') {x[i]=x[i-1]-1;y[i]=y[i-1];}
			if(str[i]=='R') {x[i]=x[i-1]+1;y[i]=y[i-1];}
		}
		flag=0;
		for(i=0;i<=l;i++)
		{
			a=n-x[i];
			b=m-y[i];
			t=(x[l]?a/x[l]:(y[l]?b/y[l]:1));
            if(t>=0&&x[l]*t==a&&y[l]*t==b)
            {flag=1;break;}
		}
		if(flag==0) printf("No\n");
		else printf("Yes\n");
	}
}
