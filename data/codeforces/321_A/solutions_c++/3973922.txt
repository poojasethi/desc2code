#include <iostream>

using namespace std;

int x,y;
int xx,yy;
string s;
int f[110][2];

int main() {
	cin >> xx >> yy;

	x=0,y=0;
	cin >> s;
	for (int i=0;i<(int)s.size();++i) {
		if (s[i]=='U') y++;
		if (s[i]=='D') y--;
		if (s[i]=='L') x--;
		if (s[i]=='R') x++;
		if (xx==x && yy==y) {
			cout <<"Yes\n";
			return 0;
		}
		f[i][0]=x,f[i][1]=y;
	}
	f[s.size()][0]=0;
	f[s.size()][1]=0;
	int dx=x,dy=y;
	for (int i=0;i<=(int)s.size();++i) {
		int px=xx-f[i][0],py=yy-f[i][1];
		if (dx==0 && px!=0) continue;
		if (dy==0 && py!=0) continue;
		if (dx==0 || (px%dx==0 && px/dx>=0))
			if (dy==0 || (py%dy==0 && py/dy>=0))
				if (dx==0 || dy==0 || px/dx==py/dy) {
					cout << "Yes\n";
					return 0;
				}
	}
	cout << "No\n";
}
