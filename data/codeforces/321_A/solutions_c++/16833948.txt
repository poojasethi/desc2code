#include<stdio.h>
#include<string.h>
int main()
{
	char zu[110];
	int x[110]={0},y[110]={0};
	int i,j,t,n,a,b,time,sumx,sumy,len,A,B;
	scanf("%d%d",&a,&b);
	scanf("%s",zu+1);
	len=strlen(zu+1);
	for(i=1;i<=len;i++)
	{
		if(zu[i]=='U')
		{
			x[i]=x[i-1];
			y[i]=y[i-1]+1;
		}
		else if(zu[i]=='D')
		{
			x[i]=x[i-1];
			y[i]=y[i-1]-1;
		}
		else if(zu[i]=='L')
		{
			x[i]=x[i-1]-1;
			y[i]=y[i-1];
		}
		else if(zu[i]=='R')
		{
			x[i]=x[i-1]+1;
			y[i]=y[i-1];
		}
	}
	for(i=0;i<=len;i++)
	{  
		A=a-x[i];
		B=b-y[i];
		if(y[len]==0)
			sumy=1;
		else
			sumy=B/y[len];
		if(x[len]==0)
			sumx=sumy;
		else
			sumx=A/x[len];
		time=sumx;
		if(time>=0)
		{  
			if(x[len]*time==A&&y[len]*time==B)
			{
				printf("Yes\n"); 
				return 0;  
			}
		}
	}
	printf("No\n");
	return 0;
}
