#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int main(){
	long long a,b;
	string str;
	while(cin>>a>>b){
		cin>>str;
	 	long long x=0,y=0;int flag=0;
		for(int i=0;str[i];i++){
			if(str[i]=='L')x--;
			if(str[i]=='R')x++;
			if(str[i]=='U')y++;
			if(str[i]=='D')y--; 
		}
		long long x1=0,y1=0;int len=str.length();
		for(int i=0;i<=len;i++){
 		int lena=a-x1;
        int lenb=b-y1;
        int t=(x?lena/x:(y?lenb/y:1));
        if(t>=0&&x*t==lena&&y*t==lenb){
            flag=1;break;
         }
			if(str[i]=='L')x1--;
			if(str[i]=='R')x1++;
			if(str[i]=='U')y1++;
			if(str[i]=='D')y1--;
		}
		if(flag)puts("Yes");
		else puts("No");
	}
}
