#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
int a,b,n,x,y;
char s[1001];
void work(int t){
	int xx=x*t,yy=y*t;
//	printf("t %d x y %d %d xx yy %d %d\n",t,x,y,xx,yy);
	if(xx==a&&yy==b){printf("Yes\n"); exit(0);}
	for(int i=1;i<=n;i++){
		if(s[i]=='U') yy++;
		if(s[i]=='D') yy--;
		if(s[i]=='L') xx--;
		if(s[i]=='R') xx++;
		if(xx==a&&yy==b){printf("Yes\n"); exit(0);}
	}
}
int main(){
	scanf("%d%d",&a,&b);
	scanf("%s",s+1); n=strlen(s+1);
	x=0,y=0;
	for(int i=1;i<=n;i++){
		if(s[i]=='U') y++;
		if(s[i]=='D') y--;
		if(s[i]=='L') x--;
		if(s[i]=='R') x++;
	}
	
	if(x){
		int t=a/x+102;
		for(int i=max(0,t-205);i<=t;i++) work(i);	
	}
	else if(y){
		int t=b/y+102;
		for(int i=max(0,t-205);i<=t;i++) work(i);
	}
	work(0);
	printf("No\n");
}
