#include<cstdio>
#include<cstring>
#define N 100000
int a,b,x,y,xx,yy,i;
char s[N];
int main()
{
    scanf("%d%d",&a,&b);
    scanf("%s",s);
    for (i=0;i<strlen(s);i++)
    {
        if (s[i]=='R') xx++;
        if (s[i]=='L') xx--;
        if (s[i]=='U') yy++;
        if (s[i]=='D') yy--;
    } 
    
    for (i=0;i<strlen(s);i++)
    {
        if (((xx==0)&&(a-x==0))||(xx!=0)&&((a-x)%xx==0))
        if (((yy==0)&&(b-y==0))||(yy!=0)&&((b-y)%yy==0))
        if ((xx*yy==0)||((a-x)/xx==(b-y)/yy))
        if ((xx==0)||((a-x)/xx>=0))
        if ((yy==0)||((b-y)/yy>=0))
        {
            printf("Yes");
            return 0;
        }
        if (s[i]=='R') x++;
        if (s[i]=='L') x--;
        if (s[i]=='U') y++;
        if (s[i]=='D') y--;
    }
    printf("No");
}
