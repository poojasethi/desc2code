#include<cstdio>
#include<algorithm>
using namespace std;
int a,b,dx,dy,kx,ky,x,y;
char s[1001];
int main(){
    scanf("%d%d",&a,&b);
    scanf("%s",s);
    for(int i=0;s[i];i++){
        if(s[i] == 'R') dx++;
        else if(s[i] == 'L') dx--;
        else if(s[i] == 'U') dy++;
        else dy--;
    }
    if(a == 0 && b == 0){
        printf("Yes\n");
        goto ou;
    }
    for(int i=0;s[i];i++){
        if(s[i] == 'R') x++;
        else if(s[i] == 'L') x--;
        else if(s[i] == 'U') y++;
        else y--;
        kx = dx == 0 ? 0 : (a-x)/dx;
        ky = dy == 0 ? 0 : (b-y)/dy;
        if((kx == ky || dx == 0 || dy == 0) && kx >= 0 && ky >= 0 && kx*dx+x == a && ky*dy+y == b){
            printf("Yes\n");
            goto ou;
        }
    }
    printf("No\n");
    ou:;
    return 0;
}
