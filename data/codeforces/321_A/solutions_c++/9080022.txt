#include<stdio.h>
#include<string.h>
int main()
{
	int i,len,a,b,x=0,y=0;

		if( s[i] == 'U' )
			y+=1;

			x+=1;
		else if(s[i] == 'L' )
			x-=1;
		else if( s[i] == 'D' )
			y-=1;
	}
	i=len-1;
	do{
		if(( a==0 ||( x &&  a%x == 0 && a/x >=0 ))&&(b==0 || (y && b%y == 0 && b/y >= 0)))
		{
			if(  x == 0 || y == 0 || ( a/x == b/y ) )
			{
				printf("Yes\n");
				return 0;
			}

		}
		if( s[i] == 'U' )
			b+=1;
		else if( s[i] == 'R' )
			a+=1;
		else if(s[i] == 'L' )
			a-=1;
		else if( s[i] == 'D' )
			b-=1;

}
