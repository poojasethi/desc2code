#include <cstdio>

using namespace std;

int main()
{
    int a, b, i, x=0, y=0;
    static char s[105];
    scanf("%d%d%s", &a, &b, s);
    for (i=0; s[i]; i++) {
        if (s[i]=='U') y++;
        if (s[i]=='D') y--;
        if (s[i]=='L') x--;
        if (s[i]=='R') x++;
    }
    int dx=0, dy=0;
    for (i=0; !i||s[i-1]; i++) {
        //t*x+dx==a &&  t*y+dy==b
        if (!x&&!y) if (a==dx&&b==dy) break;
        if (x&&!y) if (b==dy&&(a-dx)%x==0&&(a-dx)/x>=0) break;
        if (!x&&y) if (a==dx&&(b-dy)%y==0&&(b-dy)/y>=0) break;
        if (x&&y) if ((a-dx)%x==0&&(b-dy)%y==0&&(a-dx)/x==(b-dy)/y&&(a-dx)/x>=0) break;
        if (s[i]=='U') dy++;
        if (s[i]=='D') dy--;
        if (s[i]=='L') dx--;
        if (s[i]=='R') dx++;
    }
    puts(s[i]||s[i-1] ? "Yes" : "No");
    return 0;
}
