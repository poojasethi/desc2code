#include <cstdio>
#include <cstring>
int d[200][2];
char s[200];
int a,b,x[1000],y[1000];
int main()
{
	scanf("%d %d\n",&a,&b);
	scanf("%s",s);
	x[0]=y[0]=0;
	d['U'][1]=1;
	d['D'][1]=-1;
	d['L'][0]=-1;
	d['R'][0]=1;
	for (int i=0;i<strlen(s);++i) {
		x[i+1]=x[i]+d[s[i]][0],y[i+1]=y[i]+d[s[i]][1];
		if (x[i]==a && y[i]==b) {puts("Yes");return 0;}
	}
	if (x[strlen(s)])
		for (int i=0;i<strlen(s);++i)
			if ((a-x[i])%x[strlen(s)]==0 && (a-x[i])/x[strlen(s)]>=0 && (a-x[i])/x[strlen(s)]*y[strlen(s)]+y[i]==b) {puts("Yes");return 0;}
	if (y[strlen(s)])
		for (int i=0;i<strlen(s);++i)
			if ((b-y[i])%y[strlen(s)]==0 && (b-y[i])/y[strlen(s)]>=0 && (b-y[i])/y[strlen(s)]*x[strlen(s)]+x[i]==a) {puts("Yes");return 0;}
	puts("No");
}
