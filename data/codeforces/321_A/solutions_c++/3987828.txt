#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
    int a,b,flag=0,x=0,y=0;
    string s;
    cin>>a>>b>>s;
    for(int i=0;i<s.size();++i)
    {
        if(s[i]=='U')++y;
        else if(s[i]=='D')--y;
        else if(s[i]=='L')--x;
        else ++x;
    }
    int c=0,d=0;
    for(int i=0;i<s.size();++i)
    {
        if(s[i]=='U')++d;
        else if(s[i]=='D')--d;
        else if(s[i]=='L')--c;
        else ++c;
        int k=-1,l=-1;
        if(x!=0&&(a-c)%x==0)k=(a-c)/x;
        if(y!=0&&(b-d)%y==0)l=(b-d)/y;
        if(k>=0&&k==l)flag=1;
        if(x==0&&a==c&&y==0&&b==d)flag=1;
        if(x==0&&a==c&&l>=0)flag=1;
        if(y==0&&b==d&&k>=0)flag=1;
    }
    if(a==0&&b==0)flag=1;
    if(flag)printf("Yes\n");
    else printf("No\n");
}
