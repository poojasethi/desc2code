#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
int main(){
    ll a, b;
    cin>>a>>b;
    char str[1111];
    cin>>str;
    int n=strlen(str);
    ll p=0,q=0,x=0,y=0;
    for(int i=0;i<n;++i){
        if(str[i]=='U')++y;
        if(str[i]=='D')--y;
        if(str[i]=='R')++x;
        if(str[i]=='L')--x;
    }
    p=x;
    q=y;
    x=0;
    y=0;
    bool ok=false;
    if(a==0&&b==0) ok=true;
    for(int i=0;i<n;++i){
        if(str[i]=='U')++y;
        if(str[i]=='D')--y;
        if(str[i]=='R')++x;
        if(str[i]=='L')--x;
        if(a==x&&b==y) {ok=true;break;}
        ll r;
        if(p) r=(a-x)/p;
        if(q) r=(b-y)/q;
        if((p||q)&&r>=0&&r*p+x==a&&r*q+y==b) {ok=true;break;}
    }
    if(ok) cout<<"Yes"<<endl;
    else cout<<"No"<<endl;
}
