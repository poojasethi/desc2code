#include<stdio.h>
#include<string.h>
#include<math.h>
int main()
{
	int a,b,i,j,len,p,q;
	char str[205];
 	while(scanf("%d%d",&a,&b)!=EOF)
 	{
	 	scanf("%s",str);
	    len=strlen(str);
	    int x=0,y=0;
	    for(i=0;i<len;i++)
	    {
 			if(str[i]=='U')y++;
 			if(str[i]=='R')x++;
 			if(str[i]=='L')x--;
 			if(str[i]=='D')y--;
 			//str[i+len]
 		}
		int flag=0;
			int x1=0,y1=0;
			
			for(i=0;i<=len;i++)
			{
			    int a1=a-x1;
				int b1=b-y1;
				if(x!=0)p=a1/x;
				else p=1000000005;
				if(y!=0)q=b1/y;	
				else q=1000000005;
				if(p>q)p=q;
				if(p*x==a1&&p*y==b1&&p>=0)
				{
					flag=1;break;
				}
				if(str[i]=='U')y1++;
	 			if(str[i]=='R')x1++;
	 			if(str[i]=='L')x1--;
	 			if(str[i]=='D')y1--;
			}
		if(flag)printf("Yes\n");
		else printf("No\n");
    }
}
