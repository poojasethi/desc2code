#include<bits/stdc++.h>
using namespace std;
#define sz 100005
#define ll long long
int main(){
  int a,b,dx[102],dy[102],dxx=0,dyy=0,len,k;
  string s;
  cin>>a>>b>>s;
  if(a==0 && b==0) {
    puts("Yes");
    return 0;
  }
  len=s.size();
  for(int i=0;i<len;i++){
     if(s[i]=='U'){
        dyy++;
     }
     else if(s[i]=='D'){
        dyy--;
     }
     else if(s[i]=='L'){
        dxx--;
     }
     else {
        dxx++;
     }
     dx[i]=dxx;
     dy[i]=dyy;
  }
  k=0;
  for(int i=0;i<len;i++){
     if(dxx!=0) k=(a-dx[i])/dxx;
     if(dyy!=0) k=(b-dy[i])/dyy;
     if(k<0) continue;
     if(k*(dxx)+dx[i]==a && k*(dyy)+dy[i]==b){
         puts("Yes");
         return 0;
     }
  }
  puts("No");
  return 0;
}
