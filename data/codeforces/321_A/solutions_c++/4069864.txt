#include<cstdio>
#include<cstring>
int x,y,Dx,Dy,Px,Py,i,l;
char S[105];
bool Work()
{
	if(Dx==0&&Dy==0)
	{
		if(x==Px&&y==Py) return 1;
		return 0;
	}
	if(Dx==0)
	{
		if(x==Px&&(y-Py)%Dy==0&&(y-Py)/Dy>=0) return 1;
		return 0;
	}
	if(Dy==0)
	{
		if(y==Py&&(x-Px)%Dx==0&&(x-Px)/Dx>=0) return 1;
		return 0;
	}
	if((x-Px)%Dx==0&&(x-Px)/Dx>=0&&y-Py==(x-Px)/Dx*Dy) return 1;
	return 0;
}
int main()
{
	scanf("%d%d",&x,&y);
	scanf("%s",S);
	l=strlen(S);
	for(i=0;i<l;++i)
	{
		if(S[i]=='U') ++Dy;
		if(S[i]=='D') --Dy;
		if(S[i]=='L') --Dx;
		if(S[i]=='R') ++Dx;
	}
	for(i=0;i<l;++i)
	{
		if(Work())
		{
			printf("Yes\n");
			return 0;
		}
		if(S[i]=='U') ++Py;
		if(S[i]=='D') --Py;
		if(S[i]=='L') --Px;
		if(S[i]=='R') ++Px;
	}
	printf("No\n");
	return 0;
}
