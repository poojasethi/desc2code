#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;

int main()
{
	char str[110];
	int ex,ey;
	int x,y,i;
	while(scanf("%d %d",&ex,&ey)!=EOF)
	{
		scanf("%s",str);
		int n=strlen(str);
		x=y=0;
		for(i=0;i<n;i++)
		{
			if(str[i]=='L')x--;
			if(str[i]=='R')x++;
			if(str[i]=='U')y++;
			if(str[i]=='D')y--; 
		}
		int x1=0,y1=0;
		int fg=0;
		for(i=0;i<n;i++)
		{
			int tx=ex-x1;
			int ty=ey-y1;
			int tmp;
			if(x!=0)
			{
				tmp=tx/x;
			}
			else
			{
				if(y!=0)
				{
					tmp=ty/y;
				}
				else tmp=1;
			}
			if(tmp>=0&&x*tmp==tx&&y*tmp==ty)
			{
				fg=1;
				break;
			}
			if(str[i]=='L')x1--;
			if(str[i]=='R')x1++;
			if(str[i]=='U')y1++;
			if(str[i]=='D')y1--;	
		}
		if(fg)
			printf("Yes\n");
		else printf("No\n");
	}
	return 0;
}
