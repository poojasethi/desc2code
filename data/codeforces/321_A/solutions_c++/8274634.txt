#include <cstdio>
#include <cstdlib>
char s[105];
int xx,yy,dx[128],dy[128];
int a,b;
void check(){
	int tx,ty;
	if(xx){
		if(a%xx) return;
		tx=a/xx;
		if(tx<0) return;
	}
	else{
		if(a) return;
		tx=-1;
	}
	if(yy){
		if(b%yy) return;
		ty=b/yy;
		if(ty<0) return;
	}
	else{
		if(b) return;
		ty=-1;
	}
	if(tx==-1||ty==-1||tx==ty){
		puts("Yes");
		exit(0);
	}
}
int main(){
	scanf("%d%d",&a,&b);
	scanf("%s",s);
	dy['U']=1;
	dy['D']=-1;
	dx['L']=-1;
	dx['R']=1;
	for(int i=0;s[i];i++) xx+=dx[s[i]],yy+=dy[s[i]];
	for(int i=0;s[i];i++){
		check();
		a-=dx[s[i]];
		b-=dy[s[i]];
	}
	puts("No");
}
