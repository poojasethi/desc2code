#include <bits/stdc++.h>
using namespace std;
int a[1005],b[1005];
int main()
{
	int x,y;
	cin>>x>>y;
	string s;
	cin>>s;
	int p=0,q=0;
	if(x==p&&y==q){
		cout<<"Yes";
		return 0;
	}
	int n = s.length();
	for(int i=0;i<n;i++)
	{
		if(s[i]=='U')q++;
		if(s[i]=='D')q--;
		if(s[i]=='L')p--;
		if(s[i]=='R')p++;
		a[i] = p;
		b[i] = q;
	}
	for(int i=0;i<n;i++)
	{
		int k = 0;
		if(p){
			k = (x-a[i])/p;
		}
		else if(q){
			k = (y-b[i])/q;
		}
		if(k<0) k =0;
		if(x == k*p+a[i] && y == k*q+b[i]){
			cout<<"Yes";
			return 0;
		}
	}
	cout<<"No";
	return 0;
}
