#include<iostream>
#include<string>
using namespace std;

int main(){
	long long int x, y; string s;
	cin>>x>>y>>s;
	long long int X=0,Y=0;
	long long int xx[s.length()], yy[s.length()];
	for (int i=0;i<s.length();i++){
		xx[i] = X, yy[i]=Y;
		if (s[i]=='R') X++;
		else if (s[i]=='U') Y++;
		else if (s[i]=='L') X--;
		else if (s[i]=='D') Y--;
	}
	for (int i=0;i<s.length();i++){
		if (((X!=0 || x==xx[i])&&(X==0||(x-xx[i])%X==0)) && ((Y!=0 || y==yy[i])&&(Y==0||(y-yy[i])%Y==0))){
			if (((x-xx[i])*Y==(y-yy[i])*X)){
				if ((X==0 || (x-xx[i])/X>=0) && (Y==0||(y-yy[i])/Y>=0)){
					cout<<"Yes";
					return 0;
				}
			}
		}
	}
	cout<<"No";
}
