#include <iostream>
#include <cstdio>
#include <algorithm>

using namespace std;

int x,y,N,X,Y,a,b;
int yol[105][2];
string S;

int main()
{
	int i,j;

	scanf("%d %d",&x,&y);

	cin >> S;
	N=S.size();

	for(i=0;i<N;i++)
	{
		yol[i][0]=X; yol[i][1]=Y;

		switch(S[i])
		{
			case 'U' : Y++; break;
			case 'D' : Y--; break;
			case 'L' : X--; break;
			case 'R' : X++; break;
		}
	}
	

	for(i=0;i<N;i++)
	{
		a = x - yol[i][0];
		b = y - yol[i][1];

		if(!X && !Y)
		{
			if(!a && !b) 
			{ puts("Yes"); return 0; } 
			continue;
		}
			

		if(!X) 
		{
			if(!a && !(b % Y) && (b/Y)>=0)
			{ puts("Yes"); return 0; }
		}

		else if(!Y)
		{
			if(!b && !(a % X) && (a/X)>=0)
			{ puts("Yes"); return 0; }
		}

		else 
		{
			if(!(a % X) && !(b % Y) && (a/X == b/Y) && (a/X)>=0)
			{ puts("Yes"); return 0; }
		}
	}

	puts("No"); return 0;
}
