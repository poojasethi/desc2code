#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long ll;

ll a[100],b[100];

ll lowbit(ll x) {return x&(-x);}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n;scanf("%d",&n);
	ll w=0;int cnt=0;
	for (int i=1;i<=n;i++)
	{
		ll x,y;
		scanf("%I64d%I64d",&x,&y);
		w^=x;x^=y;
		for (int j=1;j<=cnt;j++)
			if (x&b[j]) x^=a[j];
		if (x) {a[++cnt]=x;b[cnt]=lowbit(x);}
	}
	for (int i=1;i<=cnt;i++) if (w&b[i]) w^=a[i];
	if (w) {puts("1/1");return 0;}
	ll t=1;for (int i=1;i<=cnt;i++) t<<=1;
	printf("%I64d/%I64d\n",t-1,t);
	return 0;
}
