#include <bits/stdc++.h>

using namespace std;

int D[100005];

int main()
{
	int n,i,k,f=0,t=1;
	cin >> n ;
	memset(D,-1,sizeof D);
	for(i=0 ; i<n ; i++)
	{
		scanf("%d",&k);
		if(k==i)
		{
			f++;
			continue ;
		}
		D[k]=i;
		if(D[i]==k and t)
		{
			f++;
			t=0;
		}
	}
	cout << f+(f!=n) << endl ;
}
