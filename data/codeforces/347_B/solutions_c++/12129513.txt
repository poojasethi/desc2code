#include<cstdio>
#include<cstring>
int a[100008],n;
int main()
{
	int i,ans=0,f=0;
	scanf("%d",&n);
	memset(a,-1,sizeof(a));
	for (i=0;i<n;i++){
		scanf("%d",&a[i]);
		if (a[i]==i) ans++;
		else if (a[a[i]]==i) f=1;
	}
	if (ans<n) ans=ans+f+1;
	printf("%d\n",ans);
	return 0;
}
