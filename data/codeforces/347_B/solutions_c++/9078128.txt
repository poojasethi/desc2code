#include <iostream>
#include <algorithm>
using namespace std;

int main(){
	int n;
	cin>>n;
	int a[n];
	int mx=1;
	int ans=0;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	for(int i=0;i<n;i++){
		if(a[i]==i){
			ans+=1;
		}else if(a[a[i]]==i){
			mx=2;
		}
	}
	ans+=mx;
	cout<<min(n,ans)<<"\n";
	return 0;
}
