#include <iostream>

using namespace std;

int v[1000000];

int main()
{
  int n;
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>v[i];
  int c=0;
  int mas=1;
  for (int i=0;i<n;i++) {
    if (v[i]==i) c++;
    if (v[i]!=i and v[v[i]]==i)
      mas=2;
  }
  if (c==n)
    cout<<n<<endl;
  else
    cout<<c+mas<<endl;
}
