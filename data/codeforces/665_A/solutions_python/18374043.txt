#!/usr/bin/python2
# -*- coding: utf-8 -*-

import sys
import os

def main():
    a, ta = map(int, sys.stdin.readline().split())
    b, tb = map(int, sys.stdin.readline().split())
    hhmm = sys.stdin.readline().strip()
    hh = int(hhmm[:2])
    mm = int(hhmm[3:])
    t1 = hh*60 + mm
    t2 = t1 + ta
    c = 0
    for i in xrange(5*60, 24*60, b):
        if i < t2 and i + tb > t1:
            c += 1
    print c


if __name__ == '__main__':
    main()
