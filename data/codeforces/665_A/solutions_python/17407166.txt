a, ta = [int(x) for x in raw_input().split(' ')]
b, tb = [int(x) for x in raw_input().split(' ')]
time_string = raw_input()
hh = int(time_string[0:2])
mm = int(time_string[3:])

start = hh * 60 + mm
end = start + ta
cur = 300
#print cur, end

count = 0
# find the beginning time interval
while cur < end and cur <= 23 * 60 + 59:
    if cur > start or start < cur + tb:
        count = count + 1
    cur = cur + b

print count
