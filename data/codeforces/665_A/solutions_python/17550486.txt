a, ta = map(int, raw_input().split(" "))
b, tb = map(int, raw_input().split(" "))
h, m = map(int, raw_input().split(":"))

beg_time = h * 60 + m
end_time = beg_time + ta

beg = 5 * 60
end = beg + tb
ans = 0

while beg <= 23 * 60 + 59:
  if max(beg_time, beg) < min(end_time, end):
    ans += 1
  beg += b
  end = beg + tb

print ans
