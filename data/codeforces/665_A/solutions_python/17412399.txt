a, ta = map(int, raw_input().split())
b, tb = map(int, raw_input().split())
hour, minute = map(int, raw_input().split(':'))
simiondepart=60*hour+minute
simionarrive=simiondepart+ta
bdeparttime=300
barrivetime=bdeparttime+tb
count=0
for i in xrange((18*60+59)/b+1):
    if max(bdeparttime,simiondepart)<min(barrivetime,simionarrive):
        count+=1
    bdeparttime+=b
    barrivetime=bdeparttime+tb
print count
