#include<bits/stdc++.h>

using namespace std;

int main()
{
    int a, ta, b, tb, HH, MM, MM1;


    scanf("%d %d %d %d %d:%d",&a,&ta,&b,&tb,&HH,&MM);

    MM += HH*60;

    MM1 = MM+ta;


        int cnt=0;
    for(int i=300; i<(24*60); i+=b){
        if(i<MM1&&i+tb>MM)
           cnt++;
    }


    printf("%d",cnt);

}
