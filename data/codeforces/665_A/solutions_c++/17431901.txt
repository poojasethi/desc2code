#include<cstdio>
using namespace std;
int main(){
    int a,ta,b,tb,h,m,t1,t2,sum;
    while(~scanf("%d%d%d%d%d:%d",&a,&ta,&b,&tb,&h,&m)){
        t1=60*h+m-300;t2=t1+ta;sum=0;
        for(int t=0;t<19*60;t+=b){
            if(t+tb>t1&&t<t2) ++sum;
        }
        printf("%d\n",sum);
    }
    return 0;
}
