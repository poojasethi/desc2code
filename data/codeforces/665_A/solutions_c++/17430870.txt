#include<iostream>

using namespace std;

int main()
{
	ios::sync_with_stdio(0);
	int c, ta, b, tb, count = 0, t, t1;
	char o; 
	cin>>c>>ta>>b>>tb>>t>>o>>t1;
	
	t = t*60 + t1;
	t1 = 300;
	
	while( t1 + tb < t )
	{
		t1 += b;	
	}
	if( t1+tb == t )
		t1 += b;
	for( ; t1 < t+ta && t1 < 1440 ; t1 += b, count++ );
	cout<<endl<<count;
}
