#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a,b,c,d,e,f;
    scanf("%d%d%d%d",&a,&b,&c,&d);
    scanf("%d:%d",&e,&f);
    int st=(e-5)*60+f;
    int ed=st+b;
    int ans=0;
    for(int i=0;i<(24-5)*60;i+=c){
        if(i>=ed)break;
        if(i+d>st)ans++;
    }
    printf("%d\n",ans);
}
