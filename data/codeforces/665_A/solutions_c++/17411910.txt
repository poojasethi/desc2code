#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a,ta,b,tb,h,m,t1,t;
	scanf("%d%d",&a,&ta);
	scanf("%d%d",&b,&tb);
	scanf("%d:%d",&h,&m);
	int x1=h*60+m;
	int x2=x1+ta;
	int count=0;
	for(t1=300;t1<1440;t1=t1+b)
	{
		t=t1+tb;
		if(t1<x2 && x1<t)
			count++;
	}
	printf("%d\n",count);
	return 0;
}
