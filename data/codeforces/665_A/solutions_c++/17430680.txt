#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>

using namespace std;

int main(){
    int a,ta,b,tb;
    scanf("%d%d%d%d",&a,&ta,&b,&tb);
    int h,f;
    scanf("%d:%d",&h,&f);
    f += 60*h;
    int ans = 0;
    for(int i=300;i<24 * 60;i+=b)
        if(i+tb > f && i < f+ta)
            ans++;
    cout<<ans<<endl;
}
