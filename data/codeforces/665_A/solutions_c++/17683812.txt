#include<iostream>
#include<cstdio>
#include<string>
using namespace std;
int main()
{
    int a,ta,b,tb,h,m,ans=0;
    cin>>a>>ta;
    cin>>b>>tb;
    scanf("%d:%d",&h,&m);
    m+=h*60;
    for(int i=300;i<24*60;i+=b)
    {
        if(i+tb>m&&i<m+ta)
            ans++;
    }
    cout<<ans<<'\n';
    return 0;
}
