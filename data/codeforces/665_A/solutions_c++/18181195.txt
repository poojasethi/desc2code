#include <stdio.h>
int main()
{
	int a,ta,b,tb;
    while(~scanf("%d%d%d%d",&a,&ta,&b,&tb))
    {
    	int h,m;
		scanf("%d:%d",&h,&m);
        m+=60*h;
        int ans = 0;
        for(int i=300;i<24*60;i+=b)
        {
        	if(i+tb>m&&i<m+ta)
			   ans++;
        }        
        printf("%d\n",ans);
    }
    
	return 0;
}
