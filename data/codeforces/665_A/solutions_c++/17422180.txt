#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a,b,t1,t2,h,m;
    scanf("%d %d",&a,&t1);
    scanf("%d %d",&b,&t2);
    scanf("%d:%d",&h,&m);
    //cout<<h<<m;
    m+=h*60;
    int ans=0;
    for(int i=300;i<24*60;i+=b){
        if(i+t2>m&&i<m+t1) ans++;
    }
    cout<<ans;
    return 0;
}
