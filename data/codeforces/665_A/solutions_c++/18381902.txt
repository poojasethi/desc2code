#include <bits/stdc++.h>
using namespace std;

int main() {

int a,b,ta,tb,h,m;

	cin>>a>>ta;
	cin>>b>>tb;

	scanf("%d:%d",&h,&m);

	int x1=h*60+m;
	int y1=x1+ta;

	int x2,y2;

	long long int count=0;

	for(x2=5*60;x2<24*60;x2+=b)
	{
		y2=x2+tb;

		if(max(x1,x2)<min(y1,y2))
		 count++;
	}

	cout<<count;

	return 0;
}
