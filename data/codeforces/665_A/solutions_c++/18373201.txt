#include <iostream>
#include<stdio.h>
using namespace std;

int main()
{
    int a,ta,b,tb,h,m,i;
    scanf("%d%d%d%d%d:%d",&a,&ta,&b,&tb,&h,&m);
    int t=60*h+m,count=0;
    for(i=300;i<1440;i+=b)
    {
        if((i<t+ta)&&(i+tb>t))
            count++;
    }
    printf("%d\n",count);
    return 0;
}
