#include <bits/stdc++.h>

using namespace std;

int main() {
	int a, ta;
	cin >> a >> ta;
	int b, tb;
	cin >> b >> tb;
	int hh, mm;
	scanf("%d:%d", &hh, &mm);
	int x1 = hh * 60 + mm, x2 = x1 + ta;
	int cnt = 0;
	for (int t1 = 300; t1 < 1440; t1 += b) {
		int t2 = t1 + tb;
		if (t1 < x2 && x1 < t2) cnt++;
	}
	cout << cnt << endl;
}
