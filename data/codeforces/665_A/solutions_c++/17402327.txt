#include<bits/stdc++.h>

int main(){
	int a,ta,b,tb,x,y,ans;
	ans = 0;
	scanf("%d%d%d%d",&a,&ta,&b,&tb);
	{
		int h,m;
		scanf("%d:%d",&h,&m);
		x = (h-5)*60+m;
	}
	for(int i = 0;;i++){
		if(x >= b*i+tb )  continue;
		else if(x+ta <= b*i || b*i >= 19*60)  break;
		ans ++ ;
	}
	printf("%d\n",ans);
	return 0;
}
