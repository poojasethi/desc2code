#include <bits/stdc++.h>
using namespace std;

int main()
{
    int a,ta,b,tb,h,m,count=0;
    cin>>a>>ta;
    cin>>b>>tb;
    scanf("%d:%d",&h,&m);
    int time=h*60+m;
    for(int i=300;i<1440;i=i+b)
    {
        int s=i+tb;
        if(i<time+ta&&s>time)
            count++;
    }
    cout<<count;
    return 0;
}
