#include <bits/stdc++.h>
using namespace std;

int main()
{
   int a, ta, b, tb, h, m, t, tm, cnt=0;
   cin>>a>>ta>>b>>tb;
   cin>>h; cin.ignore(); cin>>m;
   t=h*60+m; tm=t+ta;
   //cout<<t<<endl;
   int s=t-tb;
   for(int i=300; i<1440; i+=b){
      if(i>s and i<tm) cnt++;
   }
   cout<<cnt<<endl;
}
