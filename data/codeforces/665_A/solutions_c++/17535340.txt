#include <iostream>
using namespace std;
int a,ta,b,tb;

string s;
int main() {
	int count=0;
	cin>>a>>ta>>b>>tb;
    cin>>s;
	int h,m,x;
	h=(s[0]-48)*10+(s[1]-48);
	m=(s[3]%-48)*10+(s[4]%-48);
	x=h*60+m;
	
    	for(int i=5*60;i<=24*60-1;i=i+b)
	{
		if(x<i+tb && x+ta>i)
	{ count=count+1;
}
}
cout<<count;

}
