#include <cstdio>
#include <algorithm>
using namespace std;
int main()  
{  
    int a,ta,b,tb,h,m,be,end,ans=0;  
    scanf("%d%d%d%d%d:%d",&a,&ta,&b,&tb,&h,&m);   
    be=h*60+m;  
    end=min(be+ta,24*60);  
    for(int i=5*60;i<end;i+=b)  
        if(i+tb>be)  ans++;  
    printf("%d\n",ans);  
    return 0;  
}
