#include<iostream>
#include<iomanip>
#include<cmath>
using namespace std;
int main (){
	int n,k;
	cin>>n>>k;
	double x[1001],y[1001],s=0;
	for(int i=0;i<n;i++)
	cin>>x[i]>>y[i];
	for(int i=0;i<n-1;i++)
		s+=sqrt(pow(x[i]-x[i+1],2)+pow(y[i]-y[i+1],2));
	cout<<fixed<<setprecision(9)<<(s/50)*k;
	return 0;}
