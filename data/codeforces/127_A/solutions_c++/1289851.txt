#include <cstdio>
#include <cmath>

using namespace std;

int main() {
	int n,k;
	double x,y;
	scanf("%d%d%lf%lf",&n,&k,&x,&y);
	double res=0;
	for(int i=1;i<n;i++) {
		double p,q;
		scanf("%lf%lf",&p,&q);
		res+=sqrt((p-x)*(p-x)+(q-y)*(q-y));
		x=p; y=q;
	}
	printf("%.9lf\n",res*k/50);
	return 0;
}
