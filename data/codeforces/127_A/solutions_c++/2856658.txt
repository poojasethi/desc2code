#include <iostream>
#include <cmath>
#include <iomanip>
using namespace std;

int main(){
	unsigned int n, i, k;
	cin>>n>>k;
	double x[n], y[n], sum=0.0;
	for(i=0;i<n;i++)
		cin>>x[i]>>y[i];
	for(i=0;i<n-1;i++)
		sum += sqrt(pow(x[i]-x[i+1],2) + pow(y[i]-y[i+1],2));
	cout<<setprecision(20)<<k*(sum/50.0)<<endl;
}
