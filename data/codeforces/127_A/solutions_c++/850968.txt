#include<stdio.h>
#include<math.h>
int main()
{
   int N,n;
   double  sum=0;
   scanf("%d%d",&N,&n);
   int a[N],b[N],i;
   scanf("%d%d",&a[0],&b[0]);
   for(i=1;i<N;i++)
   {
        scanf("%d%d",&a[i],&b[i]);
	sum=sum+sqrt(pow(a[i]-a[i-1],2)+pow(b[i]-b[i-1],2));
   }
   printf("%.9f\n",(sum/50)*n);
   return 0;
}
