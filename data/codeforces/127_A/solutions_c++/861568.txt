#include<stdio.h>
#include <math.h>
int main(){
	int n,k,i;
	int x[110];
	int y[110];
	int X=0,Y=0;
	double res=0;
	scanf("%d %d",&n,&k);
	for(i=0;i<n;i++){
		scanf("%d %d",&x[i],&y[i]);
	}
	for(i=0;i<n-1;i++){
		X=x[i]-x[i+1];
		Y=y[i]-y[i+1];
		res+=sqrt(X*X+Y*Y);
	}
	printf("%.9lf\n",res/50.0*k);
			
	return 0;
}
