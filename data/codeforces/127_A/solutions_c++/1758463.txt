#include <stdio.h>
#include <math.h>
int main(){
int n;
double k;
double x0,y0,x,y,d=0;
scanf("%d%lf",&n,&k);
scanf("%lf%lf",&x0,&y0);
while(n!=1){
        scanf("%lf%lf",&x,&y);
        d=sqrt(pow((x-x0),2.0)+pow((y-y0),2.0))+d;
        x0=x;y0=y;
        n=n-1;
}
printf("%lf",k*d/50.0);
return 0;
}
