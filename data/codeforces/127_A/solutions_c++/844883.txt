#include<iostream>
#include<math.h>
#include<cstdio>
using namespace std;

int main()
{
	float n,k;
	float x =0;
	cin >> n>>k;
	int xi,yi;
	cin >> xi>>yi;
	n--;
	int x2,y2;
	while(n--)
	{
		cin>>x2>>y2;
		x+=sqrt(float((x2-xi)*(x2-xi) + (y2-yi)*(y2-yi)));
		xi=x2;
		yi=y2;
	}
	printf("%.9f\n",x*k/50);
	return 0;
}
