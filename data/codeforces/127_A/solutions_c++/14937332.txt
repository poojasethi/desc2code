#include<bits/stdc++.h>
using namespace std;
int main()
{
	double l=0,t;
	int n,m,x,y,a,b,i;
	scanf("%d%d",&n,&m);
	scanf("%d%d",&x,&y);
	for(i=0;i<n-1;++i)
	{
		scanf("%d%d",&a,&b);
		l+=sqrt((a-x)*(a-x)*1.0+(b-y)*(b-y)*1.0);
		x=a;
		y=b;
	}
	printf("%.15lf\n",l/50.0*m);
}
