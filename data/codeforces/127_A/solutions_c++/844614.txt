#include <stdio.h>
#include <math.h>

int main()
{
	int a,b;
	float x,y,lx,ly,dif=0;
	scanf("%d %d",&a,&b);
	scanf("%f %f",&lx,&ly);
	int i;
	for(i=1;i<a;i++)
	{
		scanf("%f %f",&x,&y);
		dif = dif + sqrt(pow((lx - x ),2) + pow((ly-y),2));
		lx = x;
		ly = y;
	}
	dif = dif*b;
	printf("%f\n",dif/50);
}
