#include<stdio.h>
#include<math.h>
#include<string.h>
#include<stdlib.h>

typedef long long int LL;
#define MOD 1000000007
#define M 103000

char a[M], b[M],c[M];
int n, k, ca, cb;

int main(void) {
	scanf("%d%d%s%s", &n, &k, a, b);
	k = n - k;
	for (int i = 0; i<n; i++) 
		if (a[i] == b[i])
			if(ca < k) ca++, cb++, c[i] = a[i];
			else c[i] = (a[i] - 'a' + 1) % 26 + 'a';
	for (int i = 0;i<n;i++) {
		if(a[i] != b[i]) {
			if (ca == k && cb == k) {
				char j;
				for (j = 'a'; j == a[i] || j == b[i]; j++);
				c[i] = j;
			}
			else if (ca < cb)  ca++, c[i] = a[i];
			else cb++, c[i] = b[i];	
		}
	}
	if (ca == k && cb == k) c[n] = '\0', puts(c);
	else puts("-1");
	return 0;
}
