#include <bits/stdc++.h>
#define B begin()
#define E end()
#define RE resize
#define mP make_pair
#define pB push_back
#define X first
#define Y second
#define S size()
#define ST setprecision
typedef long long ll;
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	int n,t,w;
	string x,y,q;
	cin>>n>>t>>x>>y;
	t=n-t;
	for(int i=0;i<x.S;i++)
		q+='0';
	for(int i=0;i<x.S;i++)
	{
		if(t>0)
		{
			if(x[i]==y[i])
			{
				q[i]=x[i];
				x[i]=y[i]=-1;
				t--;
			}
		}
	}
	int l=2*t;
	for(int i=0;i<x.S;i++)
	{
		if(x[i]==-1)
			continue;
		if(l)
		{
			if(l%2==0)
				q[i]=x[i];
			else
				q[i]=y[i];
			l--;
		}
		else
		{
			for(char j='a';j<='z';j++)
				if(j!=x[i] && j!=y[i])
				{
					q[i]=j;
					break;
				}
		}
	}
	if(l==0)
		cout<<q;
	else
		cout<<-1;
	return 0;
}
