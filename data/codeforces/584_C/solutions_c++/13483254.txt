#include <iostream>
#include <string>
using namespace std;


string a,b;
int n,k;
char sol[1001000];

char get_diff(char a,char b){
	for(char h='a';;h++){
		if(h!=a && h!=b){
			return h;
		}
	}
}
int main(){
	cin>>n>>k;
	cin>>a>>b;
	k=n-k;
	int dif=0,same=0;
	for(int i=0;i<n;i++){
		if(a[i]==b[i]){
			same++;
		} else {
			dif++;
		}
	}
	if(same+dif/2<k){
		cout<<-1<<endl;
		return 0;
	}
	for(int i=0;i<n;i++){
		if(a[i]==b[i]){
			if(k>0){
				sol[i]=a[i];
				k--;
			} else {
				sol[i]=get_diff(a[i],b[i]);
			}
			
		}
	}
	int g=k;
	for(int i=0;i<n;i++){
		if(a[i]!=b[i]){
			if(k>0){
				sol[i]=a[i];
				k--;
			} else if (g>0) {
				sol[i]=b[i];
				g--;
			} else {
				sol[i]=get_diff(a[i],b[i]);
			}
		}
	}
	for(int i=0;i<n;i++){
		cout<<sol[i];
	}
}
