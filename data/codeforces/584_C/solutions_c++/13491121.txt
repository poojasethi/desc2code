#include <string>
#include <iostream>
using namespace std;
int main(){
	string a,b;int n,t,dif;
	cin>>n>>t>>a>>b;
	string ret(n,0);
	dif = n-t;
	for(int i=0;i<n;++i){
		if(dif==0)break;
		if(a[i]==b[i])
			ret[i]=a[i],--dif;
	}
	int x=dif,y=dif;
	for(int i=0;i<n;++i){
		if(x==0&&y==0)break;
		if(ret[i]==0){
			if(x){ ret[i]= a[i];--x;continue;}
			if(y)ret[i]=b[i],--y;
		}
	}
	if(x!=0||y!=0){
		cout<<-1<<endl;return 0;
	}
	for(int i=0;i<n;++i){
		if(ret[i]==0){
			for(int j=0;j<26;++j)
				if(a[i]-'a'!=j && b[i]-'a' !=j){
					ret[i]= char('a'+j);break;
				}
		}
	}
	cout<<ret<<endl;
}
