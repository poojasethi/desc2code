#include <cstdio>
#include <iostream>
#include <string>
using namespace std;
int main(){
	string s1,s2;
	int d1=0,d2=0;
	int n,t;
	cin>>n>>t>>s1>>s2;
	int t2=n-t;
	string s3(n,'a');
	for (int i=0;i<n;i++){
		if (d1<t2 && s1[i]==s2[i]){
			s3[i]=s1[i];
			d1++;
			d2++;
		}
		else{
			if (s1[i]!='a'){
				s3[i]='a';
			}
			else{
				s3[i]='b';
			}
		}
			
	}
	for(int i=0;i<n;i++){
		if(s1[i]!=s2[i]){
			if(d2<t2){
				s3[i]=s2[i];
				d2++;
			}
			else if(d1<t2){
				s3[i]=s1[i];
				d1++;
			}
			else{
				if (s1[i]!='a' && s2[i]!='a'){
					s3[i]='a';
				}
				else if(s1[i]!='b' && s2[i]!='b'){
					s3[i]='b';
				}
				else{
					s3[i]='c';
				}
			}
		}
	}
	if (d1==t2 && d2==t2){
		cout<<s3;
	}
	else{
		cout<< "-1";
	}
	
	return 0;
}
