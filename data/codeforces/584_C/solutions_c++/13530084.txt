#include <cstdio>
#define MAX 200000
int n, t, diff, same = 0, samed = 0;
int s_same, s_samed = 0;
char s1[MAX], s2[MAX];
char other(char a, char b)
{
	for (char c = 'a'; c <= 'z'; c++)
		if (a != c && b != c)
			return c;
	return '!';
}
int main()
{
	scanf("%d%d", &n, &t);
	scanf("%s%s", s1, s2);
	for (int i = 0; i < n; i++)
		if (s1[i] == s2[i])
			same++;
	diff = n - same;
	s_same = n - t - same;
	if (2 * s_same > diff)
		printf("-1");
	else
		for (int i = 0; i < n; i++)
			if (s1[i] == s2[i] && samed < n - t)
				putchar(s1[i]),
				samed++;
			else if (s1[i] != s2[i] && (s_samed < s_same * 2))
			{
				if (s_samed % 2)
					putchar(s1[i]);
				else putchar(s2[i]);
				s_samed++;
			}
			else
				putchar(other(s1[i], s2[i]));
	return 0;
}
