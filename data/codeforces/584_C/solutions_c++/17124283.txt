#include <iostream>
using namespace std;
int main()
{
	int n, t, m, k, i, g=0, t1, t2, t3;
	string s1, s2;
	cin>>n>>t>>s1>>s2;
	for (m=i=0; i<n; i++) if (s1[i]!=s2[i]) m++;
	if (t<(m+1)/2) {
		cout<<-1<<endl;
		return 0;
	}
	t1=m-min(m, t); t2=t1; t3=t+t1-m;
	for (i=0; i<n; i++) {
		if (s1[i]==s2[i]) {
			if (t3>0) {
				if (s1[i]=='a') s1[i]++;
				else s1[i]--;
				t3--;
			} 
		}
		else {
			if (t1>0) {
				t1--;
			}
			else if (t2>0) {
				s1[i]=s2[i];
				t2--;
			}
			else {
				if (s1[i]!='a'&&s2[i]!='a') s1[i]='a';
				else if (s1[i]!='b'&&s2[i]!='b') s1[i]='b';
				else s1[i]='c';
			}
		}
	}
	cout<<s1<<endl;
	return 0;
}
