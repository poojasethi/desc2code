#include <iostream>
#include <string>

using namespace std;
char otherLetter(char a, char b) {
	if (a > b)
		return otherLetter(b, a);
	if (b == a + 1 || b == a) {
		if (b == 'z')
			return 'a';
		return 'z';
	} else {
		return (a + b) >> 1;
	}
}
int main()
{
	int n, t;
 	string s1, s2;
 	cin >> n >> t;
 	cin >> s1 >> s2;
 	string ans = s1;
 	int sim = n - t;
 	int swap = -1;
 	for (int i = 0; i < n; i++) {
 		if (sim == 0) {
 			ans[i] = otherLetter(s1[i], s2[i]);
 		} else {
	 		if (s1[i] == s2[i]) {
	 			sim--;
	 		} else {
	 			if (swap == -1) {
	 				swap = i;
	 				ans[i] = otherLetter(s1[i], s2[i]);
	 			} else {
	 				ans[swap] = s1[swap];
	 				ans[i] = s2[i];
	 				swap = -1;
	 				sim--;
	 			}
	 			
	 		}
	 	}
 	}
 	if (sim == 0)
 		cout << ans << endl;
 	else 
 		cout << "-1\n";
	return 0;
}
