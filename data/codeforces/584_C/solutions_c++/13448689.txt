#include<cstdlib>
#include<cstdio>
#include<cstring>

using namespace std;

const int N=100100;

char a[N],b[N],c[N];
int n,m,i,j,k;

int main(){
	scanf("%d%d",&n,&m);
	scanf("%s",a+1);
	scanf("%s",b+1);
	for(i=1;i<=n;i++)k+=a[i]!=b[i];
	if(k>(m<<1)){
		printf("-1\n");
		return 0;
	}
	m<<=1;
	for(i=1;i<=n;i++){
		if(a[i]!=b[i]){
			if(k<m){
				for(j='a';j<='z';j++)
					if(j!=a[i] && j!=b[i]){
						c[i]=j;
						m-=2;
						k--;
						break;
					}
			}else{
				if(m&1)c[i]=a[i];
				else c[i]=b[i];
				m--;
				k--;
			}
		}
	}
	for(i=1;i<=n;i++){
		if(a[i]==b[i]){
			if(m){
				if(a[i]=='a')c[i]='b';
				else c[i]='a';
				m-=2;
			}else c[i]=a[i];
		}
	}
	printf("%s\n",c+1);
	return 0;
}
