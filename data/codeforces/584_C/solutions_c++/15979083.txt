#include <bits/stdc++.h>
using namespace std; 

int n, t; 
string a, b; 
char ans[100005];
char getfir(int i) {
    for(char j = 'a'; j<='z'; j++) if (j != a[i] && j != b[i]) return j; 
    return 0;
}
int main() {
    ios::sync_with_stdio(0); 
    cin >> n >> t >> a >> b; 
    int need = n-t; 
    for(int i=0; i<n && need; i++) if (a[i] == b[i]) ans[i] = a[i], need--;
    bool turn =0;
    for(int i=0; i<n && need; i++) {
        if(a[i] == b[i]) continue; 
        if (turn) ans[i]= b[i], need--; 
        else ans[i] = a[i]; 
        turn = !turn;
    }
    if (need) { cout << "-1\n"; return 0; } 
    for(int i=0; i<n; i++) if(ans[i] < 'a') ans[i] = getfir(i);
    cout << ans << "\n";
    return 0; 
}
