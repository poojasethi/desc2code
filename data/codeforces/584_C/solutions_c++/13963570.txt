#include<cstdio>
#include<cstdlib>
#include<ctime>
#define N 100010
using namespace std;
int n,t,flag[N];
char s1[N],s2[N],s[N];
int main()
{
	srand(time(0));
	scanf("%d%d",&n,&t);
	scanf("%s%s",s1+1,s2+1);
	int cnt=0;
	for(int i=1;i<=n;i++)
		if(s1[i]==s2[i])
		{
			flag[i]=1;
			cnt++;
		}
	if(n-t>(n+cnt)/2)
	{
		printf("-1\n");
		return 0;
	}
	int curr=n-t;
		for(int i=1;i<=n;i++)
			if(curr&&flag[i])
			{
				s[i]=s1[i];
				curr--;
			}
			else while(1)
			{
				int cur=rand()%26+'a';
				if(s1[i]==cur||s2[i]==cur) continue;
				s[i]=cur;
				break;
			}
	if(!curr)
	{
		printf("%s",s+1);
		return 0;
	}
	for(int i=1;i<=n;i++)
		if(curr&&!flag[i])
		{
			int d=i+1;
			while(flag[d]) d++;
			s[i]=s1[i];
			s[d]=s2[d];
			flag[d]=1;
			curr--;
		}
	printf("%s",s+1);
	return 0;
}
