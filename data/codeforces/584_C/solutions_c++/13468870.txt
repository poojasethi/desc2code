#include<cstdio>

char input[2][100001];
char result[100001];

int main(){
	int n,t;
	scanf("%d%d",&n,&t);
	scanf("%s%s",input[0],input[1]);
	int target=n-t, q=0;
	int p=0, temp;
	for(int i=0;i<n;i++){
		if(q<target){
			if(input[0][i]!=input[1][i]){
				result[i]=input[p][i];
				p = (p+1)%2;
				if(p==1)
					temp=i;
				else
					q++;
			}
			else{
				result[i]=input[0][i];
				q++;
			}
		}
		else{
			
			for(char aa='a';aa<='z';aa++)
				if(aa!=input[0][i] && aa!=input[1][i]){
					result[i]=aa;
					break;
				}
		}
	}
	if(p==1){
				for(char aa='a';aa<='z';aa++)
				if(aa!=input[0][temp] && aa!=input[1][temp]){
					result[temp]=aa;
					break;
				}
				p = (p+1)%2;
	}
	result[n]=0;
	if(q!=target)
		printf("-1");
	else
		printf("%s",result);
	return 0;
}
