#include<bits/stdc++.h>
using namespace std;
char s1[100005],s2[100005],s3[100005]={0};
int main()
{
	int n,t;
	char x;
	cin>>n>>t>>s1>>s2;
	int same=0;
	for(int i=0;i<n;i++)
	{
		same+=s1[i]==s2[i];
		for(x='a';x<='z' && (x==s1[i] || x==s2[i]);x++);
		s3[i]=x;
	}
	int ctr=0;
	for(int i=0;i<n &&ctr!=min(same,(n-t));i++)
	{
		if(s1[i]==s2[i])
		{
			ctr++;
			s3[i]=s1[i];
		}
	}
	bool poss=true;
	if(same<n-t)
	{
		ctr=0;
		int i;
		for(i=0;i<n && ctr!=(n-t-same);i++)
		{
			if(s1[i]!=s2[i])
			{
				ctr++;
				s3[i]=s1[i];
			}
		}
		if(ctr!=n-t-same)
			poss=false;
		ctr=0;
		for(;i<n && ctr!=(n-t-same);i++)
		{
			if(s1[i]!=s2[i])
			{
				ctr++;
				s3[i]=s2[i];
			}
		}
		if(ctr!=n-t-same)
			poss=false;
	}
	s3[n]=0;
	if(poss)
		cout<<s3<<endl;
	else
		cout<<-1<<endl;
	return 0;
}
