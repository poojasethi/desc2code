#include <bits/stdc++.h>
using namespace std;
int get(string &s1,string &s2){
int n=s1.size(),d=0;
for(int i=0;i<n;i++) if(s1[i]!=s2[i]) d++;
return d;
}
char choose(char a,char b) {
for(char i='a';i<='z';i=(char)(i+1)) if(i!=a&&i!=b) return i;
}
int main(){
int n,t;cin>>n>>t;
string s1,s2;cin>>s1>>s2;
int diff=get(s1,s2);
if(t<(diff+1)/2||t>n) {cout<<"-1\n";return 0;}
string ans=s1;
int ct=0;
for(int i=0;i<n&&ct<t-(diff+1)/2;i++) if(s1[i]==s2[i]) {ans[i]=s1[i]=='a'?'z':'a';ct++;}
t-=ct;
int k=diff-t;
ct=0;
for(int i=0;i<n&&ct<k;i++) if(s1[i]!=s2[i]) {ans[i]=s1[i];ct++;s1[i]=s2[i];}
ct=0;
for(int i=0;i<n&&ct<k;i++) if(s1[i]!=s2[i]) {ans[i]=s2[i];ct++;s2[i]=s1[i];}
for(int i=0;i<n;i++) if(s1[i]!=s2[i]) ans[i]=choose(s1[i],s2[i]);
cout<<ans<<endl;
return 0;}
