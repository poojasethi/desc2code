#include<cstdio>
char s1[100000],s2[100000],s[100000];
int main(){
    //freopen("d://d.txt","r",stdin);
    int n,t,i;scanf("%d%d",&n,&t);
    scanf("%s%s",s1,s2);
    int t1=n-t,t2=t1;
    char j;
    for(i=0;i<n;i++)for(j=97;j<123;j++)if(j!=s1[i]&&j!=s2[i]){s[i]=j;break;}
    for(i=0;i<n;i++){
        if(!t2)break;
        if(s1[i]==s2[i]){
            t1--;t2--;
            s[i]=s1[i];
        }
    }
    for(i=0;i<n;i++){
        if(!t2)break;
        if(s1[i]!=s2[i]){
            if(t1>=t2){t1--;s[i]=s1[i];}
            else {t2--;s[i]=s2[i];}
        }
    }
    if(t1||t2)printf("-1");
    else printf("%s",s);
    return 0;
}
