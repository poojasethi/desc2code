#include<bits/stdc++.h>
#define ll long long int
using namespace std;
char c[123456];
int main(){
int n,t,x,y;
cin>>n>>t;
string a,b;
cin>>a>>b;
x=y=n-t;
for(int i=0;i<n;i++){
	if(a[i]==b[i] && x>0 && y>0){
		c[i]=a[i];
		x--; y--;
	}
	else if(a[i]==b[i]){
		for(int j=0;j<26;j++){
			if(a[i]-'a' !=j){
				c[i]=(char)j+'a';
				break;
			}
		}
	}
}
for(int i=0;i<n;i++){
	if(a[i]==b[i]) continue;
	if(x>0){
		c[i]=a[i];
		x--;
	}
	else if(y>0){
		c[i]=b[i];
		y--;
	}
	else{
		for(int j=0;j<26;j++){
			if(a[i]-'a' != j && b[i]-'a' !=j){
				c[i]=(char)j+'a';
				break;
			}
		}
	}
}
c[n]='\n';
x=0,y=0;
for(int i=0;i<n;i++){
	y+=(b[i]==c[i]);
	x+=(a[i]==c[i]);
}
if(x==y && x==n-t){
	cout<<c<<endl;
}
else cout<<"-1\n";
return 0;
}
