#include<cstdio>
#include<algorithm>
using namespace std;
char s1[100001],s2[100001];
int f(int a,int b)
{
	if(a>b)swap(a,b);
	if(a>97)return 97;
	if(b>98)return 98;
	return 99;
}
int main()
{
	int n,k,a=0,b;bool bl=1;
	scanf("%d%d%s%s",&n,&k,s1,s2);
	for(int i=0;i<n;++i)
	if(s1[i]!=s2[i])++a;
	if(2*k<a){puts("-1");return 0;}
	b=min(n-a,k-(a+1)/2);a=a+b-k<<1;
	for(int i=0;i<n;++i)
	if(s1[i]!=s2[i])
	{
		if(a)--a,putchar(bl?s1[i]:s2[i]),bl^=1;
		else putchar(f(s1[i],s2[i]));
	}
	else if(b)--b,putchar(s1[i]==97?98:97);
	else putchar(s1[i]);
	puts("");
}
