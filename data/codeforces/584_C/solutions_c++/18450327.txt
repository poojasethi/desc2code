#include<iostream>
using namespace std;
#define inc(c) (c=(c-'a'+1)%26+'a')
int d,k,f,n;
int main()
{
    string a,b,c;
    cin>>n>>k>>a>>b;
    c=a;
    for(int i=0;i<n;i++) if(a[i]!=b[i])
    {
        d++;
        if(inc(c[i])==b[i]) inc(c[i]);
    }
    d=(d-k)*2;
    for(int i=0;i<n;i++)
        if(d<0 && a[i]==b[i])
            inc(c[i]),d+=2;
        else if(d>0 && a[i]!=b[i] )
        {
            d--;
            c[i]= ((f)?b[i]:a[i]);
            f^=1;
        }
    if(!d)  cout<<c;
    else cout<<-1;
}
