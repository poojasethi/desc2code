#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main() {
	string a, b;
	int n, m;
	cin >> n >> m;
	cin >> a >> b;
	string c(n,'-');
	int similar = 0, different = 0;
	for(int i = 0; i < a.size(); i++) {
		if(a[i] == b[i]) similar++;
		else different++;
	}
	int x = n-m;
	if(x > similar + different/2) {
		cout << -1;
	} else {
		for(int i = 0; i < n && x; i++) {
			if(a[i] == b[i]) {
				c[i] = a[i];
				x--;
			}
		}
		int x1 = x, x2 = x;
		for(int i = 0; i < n; i++) {
			
			if(a[i] != b[i]) {
				if(x1) {
					c[i] = a[i];
					x1--;
				} else if(x2) {
					c[i] = b[i];
					x2--;
				}
			}
			if(c[i] == '-') {
				c[i] = 'a';
				while(c[i] == a[i] || c[i] == b[i]) c[i]++;
			}
		}
		cout << c;
	}
	
	return 0;
}
