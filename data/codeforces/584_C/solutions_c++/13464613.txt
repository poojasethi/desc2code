#include<iostream>
#include<stack>
using namespace std;
int main()
{
	stack<int>x,y;
	int m,n;
	string a,b,ans;
	cin>>m>>n;
	int nowa=m-n,nowb=m-n;
	cin>>a>>b;
	ans=a;
	for(int i=0;i<a.size();i++)
	{
		if(a[i]!='a'&&b[i]!='a')ans[i]='a';
		if(a[i]!='b'&&b[i]!='b')ans[i]='b';
		if(a[i]!='c'&&b[i]!='c')ans[i]='c';
		if(a[i]==b[i])x.push(i);
		else y.push(i);
	}
	while(x.size()&&nowa)ans[x.top()]=a[x.top()],nowa--,nowb--,x.pop();
	while(y.size()&&nowa)ans[y.top()]=a[y.top()],nowa--,y.pop();
	while(y.size()&&nowb)ans[y.top()]=b[y.top()],nowb--,y.pop();
	if(nowb)
	{
		cout<<-1<<endl;
		return 0;
	}
	cout<<ans<<endl;
	return 0;
}
