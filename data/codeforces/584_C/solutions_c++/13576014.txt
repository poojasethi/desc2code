#include <bits/stdc++.h>
using namespace std;

const int maxn = 1e5+123;

int N,T;
char ar[maxn],er[maxn],ans[maxn];

int main(){
	
	scanf("%d%d%s%s",&N,&T,ar+1,er+1);
	
	T = N - T;
	
	int same = 0;
	
	for( int i=1 ; i <= N ; i++ ){
		same += ar[i] == er[i];
		if( ar[i] == er[i] and T - same >= 0 )
			ans[i] = ar[i];
	}
	
	T -= same;
	
	if( 2*T > N-same ){
		printf("-1\n");
		return 0;
	}
	T += T;
	T = max(0,T);
	
	for( int i=1,f=0 ; i <= N and T ; i++ )
		if( !ans[i] ){
			if( !f )
				ans[i] = ar[i];
			else
				ans[i] = er[i];
			f = !f;
			T--;
		}
	
	for( int i=1 ; i <= N ; i++ )
		if( !ans[i] ){
			if( ar[i] != 'a' and er[i] != 'a' )
				ans[i] = 'a';
			else if( ar[i] != 'b' and er[i] != 'b' )
				ans[i] = 'b';
			else
				ans[i] = 'c';
		}
	
	printf("%s",ans+1);
	
	return 0;
}
