#include <bits/stdc++.h>

using namespace std;

char a[100001], b[100001], c[100001];

int main(){
	int n,t,i,j,k,s,d,x,y,z;

	scanf("%d%d", &n, &t);

	scanf("%s", a);
	scanf("%s", b);

	for(i=0,s=0;i<n;i++)
		s+=(a[i]==b[i]);

	t=n-t;

	d=n-s;

	if(t > s + d/2 )
		printf("-1\n");
	else{
		if(t<s+1){
			s=t;
			x=y=0;
		}
		else{
			x=y=t-s;
		}

		for(i=0;i<n;i++){
			if(a[i]==b[i]){
				if(s>0){
					c[i]=a[i];
					s--;
				}
				else{
					for(j=0;j<26 && a[i]==('a'+j);j++);

					c[i]='a'+j;
				}
			}
			else{
				if(x>0){
					c[i]=a[i];
					x--;
				}
				else{
					if(y>0){
						c[i]=b[i];
						y--;
					}
					else{
						for(j=0;j<26 && (a[i]==('a'+j) || b[i]==('a'+j));j++);

						c[i]='a'+j;
					}
				}
			}
		}

		c[n]='\0';

		printf("%s\n", c);
	}

	return 0;
}
