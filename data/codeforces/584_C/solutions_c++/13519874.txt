#include <bits/stdc++.h>
using namespace std;

int n,t;
char s1[100005],s2[100005];
int dif[100005];
int nx[100005];

int main(){
	memset(dif,-1,sizeof(dif));
	scanf("%d%d%s%s",&n,&t,s1,s2);
	int r=n;
	if(r<t){puts("-1");return 0;}
	int k=-1;
	for(int i=n-1;i>=0;--i){
		if(s1[i]!=s2[i]){
			nx[i]=k;
			k=i;
		}
	}
	for(int i=0;i<n;++i){
		if(dif[i]>=0)continue;
		if(s1[i]!=s2[i]){
			if(r==t||nx[i]<0)dif[i]=0;
			else {
				dif[i]=1;
				dif[nx[i]]=2;
				r--;
			}
		}
	}

	for(int i=0;i<n;++i){
		if(dif[i]>=0)continue;
		if(s1[i]==s2[i]){
			if(r==t)dif[i]=0;
			else {
				dif[i]=1;
				r--;
			}
		}
	}

	if(r>t){puts("-1");return 0;}

	for(int i=0;i<n;++i){
		char c;
		if(!dif[i]){
			for(c='a';c==s1[i]||c==s2[i];c++);
		}
		else if(dif[i]==1)c=s1[i];
		else c=s2[i];
		putchar(c);
	}
	puts("");	
	return 0;
}
