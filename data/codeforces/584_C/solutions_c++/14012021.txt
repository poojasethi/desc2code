#include<stdio.h>
#include<iostream>
#include<string.h>
#include<stdlib.h>
using namespace std;
const int N=100008;
char a[N],b[N],c[N];
int main(void)
{
	int i,n,t,s1,s2,s3,s4,sa;
	memset(a,0,sizeof(a));
	memset(b,0,sizeof(b));
	memset(c,0,sizeof(c));
	scanf("%d%d%s%s",&n,&t,a+1,b+1);
	s1=s2=s3=s4=0;
	for(i=1;i<=n;i++)if(a[i]==b[i])s1++;else s2++;
	if(t*2<s2){cout<<-1;return 0;}
	if(t>s2){s3=t-s2;s4=s2;}
	else {s4=t*2-s2;sa=t-s4;}
	for(i=1;i<=n;i++)
	if(a[i]==b[i]){if(s3){s3--;c[i]='a';while(c[i]==a[i]||c[i]==b[i])c[i]++;}else c[i]=a[i];}
	else if(s4){s4--;c[i]='a';while(c[i]==a[i]||c[i]==b[i])c[i]++;}
	else if(sa){sa--;c[i]=b[i];}
	else {c[i]=a[i];}
	printf("%s",c+1);
	return 0;
}
