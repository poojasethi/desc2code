#include <cstdio>
#include <cstring>
using namespace std;
const int max_len = 1e5 + 5;
char sa[max_len],sb[max_len],ans[max_len];
int main()
{
	int n,t,same=0;
	scanf("%d%d%s%s",&n,&t,sa,sb);
	t = n - t;
	for(int i=0;i<n;i++)
		if(sa[i] == sb[i])
			same++;
	int max_same = same + (n - same) /2;
	if(t > max_same ){printf("-1\n");return 0;}
	bool isa=true;
	for(int i=0;i<n && t;i++)
		if(sa[i] == sb[i])
			ans[i] = sa[i],t--;
	for(int i=0;i<n;i++)
	{
		if(ans[i] != 0)continue;
		if(t > 0)
		{
			ans[i] = (isa?sa[i]:sb[i]);
			isa=!isa;
			if(isa)t--;
		}else
		{
			for(int j='a';j<='z';j++)
				if(j != sa[i] && j!= sb[i])
				{
					ans[i] = (j);
					break;
				}
		}
	}
	ans[n] = 0;
	puts(ans);
	return 0;
}
