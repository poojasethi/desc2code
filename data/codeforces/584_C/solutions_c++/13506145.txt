#include<bits/stdc++.h>

using namespace std;

int n, t, same, dif;
char a[100010], b[100010], ans[100010];

char fun(int i) {
	for(char x='a' ; ; ++x)
		if(x != a[i] && x!=b[i])
			return x;
	return 'a';
}

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d%s%s", &n, &t, a, b);
	t = n-t;
	for(int i=0 ; i<n ; ++i)
		a[i] == b[i] ? ++same : ++dif;
	if(same + dif/2 < t) return puts("-1"), 0;
	for(int i=0 ; i<n ; ++i)
		if(a[i] == b[i])
			if(t > 0) ans[i] = a[i], --t;
			else ans[i] = fun(i);
	int k = t;
	for(int i=0 ; i<n ; ++i)
		if(a[i] != b[i])
			if(t > 0) ans[i] = a[i], --t;
			else if(k > 0) ans[i] = b[i], --k;
			else ans[i] = fun(i);
	puts(ans);
	return 0;
}
