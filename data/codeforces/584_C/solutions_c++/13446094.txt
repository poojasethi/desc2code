#include <iostream>
#include <vector>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	int n, t;
	cin >> n >> t;
	string a, b;
	cin >> a >> b;
	string ans;
	vector<int> v1, v2;
	for (int i = 0; i < n; i++)
		if (a[i] == b[i])
		{
			ans += a[i];
			v1.push_back(i);
		}
		else
		{
			char c = 'a';
			while (c == a[i] || c == b[i])
				c++;
			ans += c;
			v2.push_back(i);
		}
	int x = v2.size();
	for (int i = 0; i < (int)v1.size() && x < t; i++)
	{
		char c = 'a';
		while (c == a[v1[i]])
			c++;
		ans[v1[i]] = c;
		x++;
	}
	for (int i = 0; i + 1 < (int)v2.size() && x > t; i += 2)
	{
		ans[v2[i]] = a[v2[i]];
		ans[v2[i + 1]] = b[v2[i + 1]];
		x--;
	}
	if (x != t)
		cout << -1 << endl;
	else
		cout << ans << endl;
	return 0;
}
