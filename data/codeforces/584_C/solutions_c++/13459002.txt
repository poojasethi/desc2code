#include <iostream>
#include <queue>
#include <string>
#define mod 1000000007
#include <vector>

using namespace std ;

int main ()
{

	int n, t;
	cin >> n >> t;

	string s1, s2, s;
	cin >> s1 >> s2;

	s = s1;

	for (int i = 0; i < n; ++i)
	{
		for (char c = 'a'; c <= 'z'; ++c)
			if (c != s1[i] && c != s2[i])
			{
				s[i] = c;
				break;
			}
	}

	int d1 = n, d2 = n;

	for (int i = 0; i < n; ++i)
	{
		if (s1[i] == s2[i])
		{
			if (d1 > t && d2 > t)
				s[i] = s1[i], --d1, --d2;
		}
	}

	for (int i = 0; i < n; ++i)
	{
		if (s1[i] != s2[i])
		{
			if (d1 > t)
				s[i] = s1[i], --d1;
			else
			{
				if (d2 > t)
					s[i] = s2[i], --d2;
			}
		}
	}

	if (d1 == t && d2 == t)
		cout << s;
	else
		cout << -1;

	return 0;


}
