n,k=map(int,raw_input().split())
a=list(raw_input())
b=list(raw_input())
c=a[:]
d=[i for i in range(n) if a[i]!=b[i]]
while k and len(d)>k:
  i=d.pop()
  c[i]=a[i]
  i=d.pop()
  c[i]=b[i]
  k-=1
while d and k:
  i=d.pop()
  c[i]=[x for x in 'abc' if x!=a[i] and x!=b[i]][0]
  k-=1
if k:
  for i in range(n):
    if a[i]==b[i]:
      c[i]=[x for x in 'ab' if x!=a[i]][0]
      k-=1
      if not k:
        break
print -1 if k or d else ''.join(c)
