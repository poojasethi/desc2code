#!/usr/bin/env python
from sys import stdin as cin

P = 10**9+7

def other_than(c1, c2):
    if c1 != 'a' and c2 != 'a':
        return 'a'
    if c1 != 'b' and c2 != 'b':
        return 'b'
    return 'c'

def main():
    n, t = map(int, next(cin).split())
    s1 = next(cin).strip()
    s2 = next(cin).strip()
    difn = sum(c1 != c2 for (c1,c2) in zip(s1, s2))

    change_neqs = 0
    if difn % 2:
        change_neqs += 1
        difn -= 1
        n -= 1
        t -= 1

    if t < difn // 2:
        return '-1'

    change_eqs = 0
    if t > difn:
        change_eqs = t - difn
        t = difn

    change_neq1 = change_neq2 = difn - t
    change_neqs += difn - change_neq1*2

    s = []
    for (c1, c2) in zip(s1, s2):
        if c1 == c2:
            if change_eqs:
                c = other_than(c1, c2)
                change_eqs -= 1
            else:
                c = c1
        else:
            if change_neqs:
                c = other_than(c1, c2)
                change_neqs -= 1
            elif change_neq1:
                c = c2
                change_neq1 -= 1
            elif change_neq2:
                c = c1
                change_neq2 -= 1
            else:
                raise Exception('bad condition')
        s.append(c)

    return ''.join(s)


print main()
