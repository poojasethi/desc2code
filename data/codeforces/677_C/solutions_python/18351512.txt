def c2i(c):
    if '0'<=c<='9':
        return int(c)
    if 'A'<=c<='Z':
        return ord(c)-ord('A')+10
    if 'a'<=c<='z':
        return ord(c)-ord('a')+36
    if c=='-':
        return 62
    return 63

cnt=[0]*64
for i in range(64):
    for j in range(64):
        cnt[i&j]+=1
M=10**9+7
print reduce(lambda a,x:a*x%M, (cnt[c2i(c)] for c in raw_input()), 1)
