def solve():
    l=[0]*(64)
    s=raw_input()
    count=0
    for x in s:
        if x>='a':
            l[ord(x)-61]+=1
        elif x=='_':
            l[63]+=1
        elif x=='-':
            l[62]+=1
        elif x>='0' and x<='9':
            l[ord(x)-48]+=1
        else:
            l[ord(x)-55]+=1
    # print l
    for x in xrange(63):
        if l[x]>0:
            count+=(6-bin(x).count('1'))*l[x]
    print (3**count)%1000000007
solve()
