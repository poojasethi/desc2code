d={}
for i in range(10):
	d[str(i)]=i
for i in range(26):
	d[chr(ord('A')+i)]=10+i
for i in range(26):
	d[chr(ord('a')+i)]=36+i
d['-']=62
d['_']=63
freq=[0]*64
for i in range(64):
	for j in range(64):
		if i&j<64:
			freq[i&j]+=1
			freq[i&j]%=(10**9+7)
s=raw_input()
ans=1
for x in s:
	ans = (ans*freq[d[x]])%(10**9+7)
print ans
