str = raw_input()
zcnt = 0
mod = 10**9 + 7

def bn(s):
    bv = bin(s)[2:]
    b = len([i for i in bv if i is '0']) + (6 - len(bv))
    return b

for s in str:
    if s.isdigit():
        zcnt += bn(int(s))
    elif s.isalpha():
        if s.islower():
            zcnt += bn(ord(s) - ord('a') + 36)
        else:
            zcnt += bn(ord(s) - ord('A') + 10)
    elif s is '-':
        zcnt += bn(62)
    elif s is '_':
        zcnt += bn(63)
print pow(3,zcnt,mod)
