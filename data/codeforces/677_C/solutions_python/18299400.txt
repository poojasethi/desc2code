def convert(num):
	ans = ""
	while num > 0:
		ans += str(num % 2)
		num /= 2

	return "".join(["0" for i in xrange(6 - len(ans))]) + ans[::-1]
	
	
st = raw_input()
d = {}
for i in xrange(48, 58):
	d[chr(i)] = i - 48
	
cont = 10
for i in xrange(65, 91):
	d[chr(i)] = cont
	cont += 1

cont = 36
for i in xrange(97, 123):
	d[chr(i)] = cont
	cont += 1
	
d['-'] = 62
d['_'] = 63

bin = ""
for i in st:
	bin += convert(d[i])
	
print (3 ** bin.count('0')) % (10 ** 9 + 7)
