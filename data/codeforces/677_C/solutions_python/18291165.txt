def conv(char):
    if char in '0123456789':
        return int(char)
    k = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
    if char in k:
        return 10 + k.index(char)
    k = k.lower()
    if char in k:
        return 36 + k.index(char)
    if char == '-':
        return 62
    return 63

x = [0] * 65
for i in range(64):
    for j in range(64):
        x[i&j]+=1

p = 1
for i in raw_input():
    p *= x[conv(i)]
    p %= (10**9+7)
print(p)
