mod=1000000007
s=raw_input()
l=[3**((6-(len(bin(i)[2:])))+bin(i)[2:].count('0')) for i in range(64)]
ans=1
for i in s:
    if i=='-':
        ans*=l[-2]
    elif i=='_':
        ans*=l[-1]
    elif i.isdigit():
        ans*=l[int(i)]
    elif i.isupper():
        ans*=l[ord(i)-55]
    else:
        ans*=l[ord(i)-61]
    ans%=mod
print ans
