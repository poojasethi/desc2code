def f(c):
  if '0' <= c <= '9': return ord(c) - ord('0')
  if 'A' <= c <= 'Z': return ord(c) - ord('A') + 10
  if 'a' <= c <= 'z': return ord(c) - ord('a') + 36
  if c == '-': return 62
  return 63

s, res, mod = raw_input(), 1, 10**9 + 7
for c in s:
  for i in range(6):
    if (f(c) >> i) & 1 == 0:
      res = (res * 3) % mod
print res
