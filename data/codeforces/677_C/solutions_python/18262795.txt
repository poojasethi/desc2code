
MOD = 10**9+7

d = {}
for i in xrange(10):
    d[chr(i+ord('0'))] = i
for i in xrange(26):
    d[chr(i+ord('A'))] = i+10
    d[chr(i+ord('a'))] = i+36
d['-'] = 62
d['_'] = 63

wk = sorted((w, k) for k, w in d.items())
di = [k for _, k in wk]

ma = {}
for i in xrange(64):
    for j in xrange(64):
        z = (i & j)
        if di[z] not in ma:
            ma[di[z]] = 1
        else:
            ma[di[z]] += 1

s = raw_input()
ans = 1
for ch in s:
    ans *= ma[ch]
    ans %= MOD
print ans
