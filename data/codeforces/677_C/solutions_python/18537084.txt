s=raw_input()
t=0
for i in s:
	if(i.isalpha()):
		if(i.islower()):
			t+=bin(ord(i)-ord('a')+36)[2:].zfill(6).count('0')
		else:
			t+=bin(ord(i)-ord('A')+10)[2:].zfill(6).count('0')
	else:
		if(i.isdigit()):
			t+=bin(ord(i)-ord('0'))[2:].zfill(6).count('0')
		elif(i=='-'):
			t+=1
print pow(3,t,1000000007)
