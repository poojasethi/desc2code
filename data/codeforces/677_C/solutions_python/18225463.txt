tab = [0] * 64
for i in range(64):
  for j in range(64):
    tab[i & j] += 1

s = raw_input()
arr = []
for i in range(len(s)):
  if '0' <= s[i] <= '9':
    arr.append(ord(s[i]) - ord('0'))
  elif 'A' <= s[i] <= 'Z':
    arr.append(ord(s[i]) - ord('A') + 10)
  elif 'a' <= s[i] <= 'z':
    arr.append(ord(s[i]) - ord('a') + 36)
  elif s[i] == '-':
    arr.append(62)
  else:
    arr.append(63)

mod = 1e9 + 7
ans = 1
for a in arr:
  ans = (ans * tab[a]) % mod

print int(ans)
