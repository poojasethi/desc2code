lowercase = 'abcdefghijklmnopqrstuvwxyz'
uppercase = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ'
digits = '0123456789'

B64 = digits + uppercase + lowercase + '-_'
DIVISOR = (10 ** 9 + 7)

s = raw_input()
result = 1
for i in s:
    result = (result * 3 ** '{0:#08b}'.format(B64.index(i))[2:].count('0')) % DIVISOR

print result
