def conv(c):
    if c == '-': return 62
    if c == '_': return 63
    if c.isdigit(): return ord(c)-ord('0')
    if c.isupper(): return ord(c)-ord('A')+10
    if c.islower(): return ord(c)-ord('a')+36
def cnt(v):
    c = list(bin(conv(v))[2:])
    return c.count('0')+6-len(c)
print pow(3, sum(map(cnt, list(raw_input()))), 1000000007)
