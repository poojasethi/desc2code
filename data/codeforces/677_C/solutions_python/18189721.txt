import sys

W = sys.stdin.readline().rstrip('\n\r')

def val(c):
  if c>='0' and c<='9':
    return ord(c)-ord('0')
  if c>='A' and c<='Z':
    return ord(c)-ord('A')+10
  if c>='a' and c<='z':
    return ord(c)-ord('a')+36
  if c=='-':
    return 62
  return 63

def P(c):
  x = 0
  for i in range(6):
    if c&2**i==0:
      x+=1
  return 3**x

S = 1
for c in W:
  S = S*P(val(c))%(10**9+7)
print S
