#include<iostream>
#include<string>
using namespace std;

string s="0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz-_";
const int MOD=1000000007;
long long ans=1;

int main(){
	string str;
	cin>>str;
	for(int i=0;i<str.size();i++){
		int n=s.find(str[i]);
		for(int j=0;j<6;j++){
			if(n&(1<<j))
				continue;
			ans=(ans*3)%MOD;
			}
		}
	cout<<ans<<endl;
	return 0;
	}
