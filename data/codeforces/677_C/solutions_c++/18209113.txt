#include <cstring>
#include <cstdio>
const int mod=1e9+7;
char s[100005];
int n,sum[64],ans=1;
inline int NUM(char c)
{
	if(c>='0'&&c<='9') return c-'0';
	if(c>='A'&&c<='Z') return c-'A'+10;
	if(c>='a'&&c<='z') return c-'a'+36;
	if(c=='-') return 62;return 63;
}
int main()
{
	for(int i=0;i<64;i++)
		for(int j=0;j<64;j++)
			sum[i&j]++;
	scanf("%s",s);
	n=strlen(s);
	for(int i=0;i<n;i++)
		ans=(long long)ans*sum[NUM(s[i])]%mod;
	printf("%d",ans);
}
