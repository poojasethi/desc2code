#include <bits/stdc++.h>
using namespace std;
typedef long long lli;
char s[100004];
int main ()
{
  int i, r = 1, j;
  scanf ("%s", s);
  for (i = 0; s[i]; i++) {
    for (j = s[i] == '-' ? 62 : s[i] == '_' ? 63 :
         isdigit (s[i]) ? s[i] - '0' : s[i] <= 'Z' ? s[i] - 'A' + 10 :
         s[i] - 'a' + 36; j < 63; j |= j + 1) {
      //printf ("%d\n", j);
      r = r * 3LL % 1000000007;
    }
  }
  printf ("%d\n", r);
  return 0;
}
