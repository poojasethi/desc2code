#include<bits/stdc++.h>
using namespace std;
int mod=1000000007;
char s[1005000];
long long a[80],b[1005000];
long long res;
int main(){
	for(int i=0;i<64;i++){
		for(int j=0;j<64;j++)
			a[(i&j)]++; 
	}
	cin>>s;
	int len=strlen(s);
	for(int i=0;i<len;i++){
		if(s[i]<='9'&&s[i]>='0')b[i]=s[i]-'0';
		else if(s[i]<='z'&&s[i]>='a')b[i]=s[i]-'a'+36;
		else if(s[i]<='Z'&&s[i]>='A')b[i]=s[i]-'A'+10;
		else if(s[i]=='-')b[i]=62;
		else b[i]=63;		
	}
	res=1;
	for(int i=0;i<len;i++){
		res*=a[b[i]];
		if(res>mod)res%=mod;
	}
	cout<<res<<endl;
}
