#include<cstdio>
#define M 1000000007
main(){
long long n,i=0,j,t,s=1;
char a[100005];
fgets(a,100005,stdin);
for(i=0;a[i]>30;i++){
	if(a[i]==95)t=63;
	else if(a[i]==45)t=62;
	else if(a[i]<60)t=a[i]&15;
	else if(a[i]&32)t=(a[i]&31)+35;
	else t=(a[i]&31)+9;
	for(j=1;j<64;j<<=1)if(!(t&j)){
		s*=3;
		s%=M;
	}
}
printf("%I64d",s);
}
