#include<bits/stdc++.h>
using namespace std;
long long dictionary(char a){
long long b=a;
if(b==95){return 63;}
else if(b==45){return 62;}
else if(b>=65 and b<=90){return b-55;}
else if(b>=48 and b<=57){return b-48;}
else{return b-61;}
}
int main(){
vector<long long>vec(64);
for(long long i=0;i<=63;i++){
for(long long j=0;j<=63;j++){
long long mu=i&j;
vec[mu]++;
}
}
string p;
cin>>p;
long long k=1;
for(long long i=0;i<p.size();i++){
    k=(k%(1000000007)*(vec[dictionary(p[i])]))%1000000007;
}
cout<<k<<endl;
}
