#include <bits/stdc++.h>

#define mod 1000000007

using namespace std;

int main() {

	long long x,ans = 1;
	string s;
	cin >> s;

	for(int i = 0; i < s.size(); i++){

		if(isdigit(s[i])) x = s[i]-'0';
		else if(isupper(s[i])) x = s[i]-'A'+10;
		else if(islower(s[i])) x = s[i]-'a'+36;
		else if(s[i] == '-') x = 62;
		else if(s[i] == '_') x= 63;

		for(int j = 0; j < 6; j++){
			if( !((x >> j)&1) ){
				ans *= 3;
				ans = ans%mod;
			}
		}
	}

	cout << ans << endl;

	return 0;
}
