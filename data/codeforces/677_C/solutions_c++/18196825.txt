#include<iostream>
#include<math.h>
using namespace std;
int main(){
string s;
cin>>s;long long int ans=1,u=1000000007;
for(long i=0;i<s.size();i++){
    int g=s[i]-'0';
    if(g>=17&&g<=42){g=g-7;}
    if(g>=49&&g<=74)g=g-13;
    if(s[i]=='-')g=62;if(s[i]=='_')g=63;
    int l=0;//cout<<g<<" "<<s[i]<<endl;
    while(g!=0){
        if(g%2==1)l++;g=g/2;
    }
    l=6-l;long long int p=pow(3,l);ans=ans*p;ans=ans%u;
}
cout<<ans<<endl;










}
