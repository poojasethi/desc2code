#include <iostream>
#include <string>

using namespace std;

typedef long long int ll;

ll modulo=1e9+7;

int maneras[1000];

string s;
ll sol=1;

int main()
{
  for (int i=0;i<64;i++)
    for (int j=0;j<64;j++)
      maneras[i&j]++;
  cin>>s;
  for (int i=0;i<int(s.size());i++) {
    char c=s[i];
    int x;
    if (c>='0' and c<='9') x=c-'0';
    if (c>='A' and c<='Z') x=c-'A'+10;
    if (c>='a' and c<='z') x=c-'a'+36;
    if (c=='-') x=62;
    if (c=='_') x=63;
    sol=sol*maneras[x]%modulo;
  }
  cout<<sol<<endl;
}
