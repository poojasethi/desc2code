#include <iostream>
#define MOD 1000000007
using namespace std;
 
int get(char c) {
	if(c >= '0' && c <= '9') return c-'0';
	if(c >= 'A' && c <= 'Z') return c-'A'+10;
	if(c >= 'a' && c <= 'z') return c-'a'+36;
	if(c == '-') return 62;
	return 63;
}
int main() {
	int t;
 
	string s;
 
	cin >> s;
	long long ans = 1;
	for(int i = 0; i < s.size(); i++) {
		int digit = get(s[i]);
		int val = 1;
		for(int j = 0; j < 6; j++) {
			if(digit&(1<<j)) continue;
			val = val * 3;
		}
		ans = (ans * val)%MOD;
	}
	cout << ans << endl;
 
	return 0;
}
