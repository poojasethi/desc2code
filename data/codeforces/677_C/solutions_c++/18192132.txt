#include <bits/stdc++.h>
using namespace std;
int cnt[64];
long long mod = 1e9 + 7;
int main() {
	for(int i = 0; i < 64; i++) {
		for(int j = 0; j < 64; j++) {
			cnt[i & j]++;
		}
	}
	long long ans = 1;
	string s;
	cin >> s;
	for (int i = 0; i < int(s.length()); ++i)
	{
		if(s[i] >= '0' && s[i] <= '9') ans *= cnt[s[i] - '0'];
		else if(s[i] >= 'A' && s[i] <= 'Z') ans *= cnt[s[i] - 'A' + 10];
		else if(s[i] >= 'a' && s[i] <= 'z') ans *= cnt[s[i] - 'a' + 36];
		else if(s[i] == '-') ans *= cnt[62];
		else ans *= cnt[63];
		ans %= mod;
	}
	cout << ans << endl;
}
