#include<bits/stdc++.h>
using namespace std;
const int mod = 1e9+7;
string s;
int get(char c)
{
    if(c>='0'&&c<='9')return c-'0';
    if(c>='A'&&c<='Z')return c-'A'+10;
    if(c>='a'&&c<='z')return c-'a'+36;
    if(c=='-')return 62;
    if(c=='_')return 63;
}
int main()
{
    cin>>s;
    long long ans = 1;
    for(int i=0;i<s.size();i++)
    {
        int p = get(s[i]);
        for(int j=0;j<6;j++)
            if(!((p>>j)&1))
                ans=ans*3%mod;
    }
    cout<<ans<<endl;
}
