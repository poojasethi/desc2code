#include<iostream>
#include<string>
using namespace std;
long long ans;
long f[100],i,j,l,k;
string s;
int main()
{
	for(i=0;i<=63;i++)
	for(j=0;j<=63;j++)
	f[i&j]++;
	cin>>s;l=0;ans=1;
	while(s[l]) l++;
	for(i=0;i<l;i++)
	{
		k=64;
		if((s[i]>='0') && (s[i]<='9')) k=s[i]-'0';
		else if((s[i]>='A') && (s[i]<='Z')) k=s[i]-'A'+10;
		else if((s[i]>='a') && (s[i]<='z')) k=s[i]-'a'+36;
		else if(s[i]=='-') k=62;
		else k=63;
		ans=ans*f[k]%1000000007;
	}
	cout<<ans;
}
