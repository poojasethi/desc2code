#include <iostream>
using namespace std;

int main()
{
    string s;long long int x,k=1;
    cin>>s;
    for (int i=0;i<s.size();i++){
        x=s[i];
        if (x>=48 && x<=57) x-=48;
        else if (x>=65 && x<=90) x-=55;
        else if (x>=97 && x<=122) x-=61;
        else if (s[i]=='-') x=62;
        else x=63;

        for (int i=0;i<6;i++){
            if (x%2==0) k=(k*3)%1000000007;
            x=x/2;
        }
    }
    cout<<k;
    return 0;
}
