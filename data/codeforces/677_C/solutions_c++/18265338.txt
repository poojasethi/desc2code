#include <iostream>
#include <map>

using namespace std;

int main(int argc, char** argv) {
	string A;
	cin>>A;
	int i,N=A.size();
	map<char,unsigned char>D;
	for(i=0;i<10;i++)D[char('0'+i)]=i;
	for(i=0;i<26;i++)D[char('A'+i)]=i+10;
	for(i=0;i<26;i++)D[char('a'+i)]=i+36;
	D['-']=62; D['_']=63;
	long long ans=1, M=1000000007;
	for(i=0;i<N;i++)
		{
		 unsigned P=~D[A[i]] & 63;
		 int k=0;
		 while(P){
		 	k++; P&=P-1;
		 }
		 while(k--)ans*=3;
		 ans%=M;
		}
	cout<<ans;
	return 0;
}
