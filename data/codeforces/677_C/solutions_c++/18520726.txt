#include <cstdio>

const int MOD = 1000000007;

char str[100001];

int to_int(char a)
{
	if (a >= '0' && a <= '9') return a - '0';
	if (a >= 'A' && a <= 'Z') return a - 'A' + 10;
	if (a >= 'a' && a <= 'z') return a - 'a' + 36;
	if (a == '-') return 62;
	return 63;
}

int main() {
	scanf("%s", str);
	int ans = 1;
	for (int i = 0; str[i]; ++i) {
		int n = to_int(str[i]);
		for (int j = 0, k = to_int(str[i]); j < 6; ++j, k /= 2) {
			if (k % 2 == 0) {
				ans = (ans * 3ll) % MOD;
			}
		}
	}
	printf("%d\n", ans);

	return 0;
}
