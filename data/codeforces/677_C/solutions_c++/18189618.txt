#include<cstdio>

const long long mod=1000000007;
char word[100010];
long long A;

int main()
{
	scanf("%s\n",word);
	A=1;
	for(int i=0,k;word[i];i++)
	{
		if(word[i]>='0'&&word[i]<='9') k=word[i]-'0';
		else if(word[i]>='A'&&word[i]<='Z') k=word[i]-'A'+10;
		else if(word[i]>='a'&&word[i]<='z') k=word[i]-'a'+36;
		else if(word[i]=='-') k=62;
		else if(word[i]=='_') k=63;

		for(int d=0;d<6;d++)
		{
			if(!(k&1)) A=3*A%mod;
			k>>=1;
		}
	}
	printf("%I64d\n",A);
	return 0;
}
