#include <bits/stdc++.h>
typedef long long ll;
using namespace std;

string s;
ll cnt[100],ret=1;
ll Mod = 1000000000+7LL;
int indxOf(char c){
	if(c>='0'&&c<='9')
		return c-'0';
	else if(c>='A'&&c<='Z')
		return 10+c-'A';
	else if(c>='a'&&c<='z')
		return 36+c-'a';
	else 
		return c=='-'?62:63;
}
int main(){
	for(int i=0;i<64;i++)
		for(int j=0;j<64;j++)
			cnt[i&j] ++;
	
	cin >> s;
	int n = s.length(); 
	for(int i=0;i<n;i++){
		ret = (ret%Mod * cnt[indxOf(s[i])]%Mod)%Mod;
	}
	cout << ret << endl;
	return 0;
}
