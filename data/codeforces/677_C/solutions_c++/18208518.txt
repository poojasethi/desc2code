#include<bits/stdc++.h>
using namespace std;
const int mod=1e9+7;

int hash(char c){
	if(c=='-')
	  return 62;
	if(c=='_')
	  return 63;
	if(c>='0'&&c<='9')
	  return c-'0';
	if(c>='a'&&c<='z')
	  return 36+c-'a';
	return c-'A'+10;
}

int main(){
	string s;
	cin>>s;
	long long ans=1;
	for(int i=0;i<s.length();i++){
		int q=hash(s[i]);
		int tmp=0;
		while(q>0){
			if(q%2==1)
			  tmp++;
			q/=2;
		}
		tmp=6-tmp;
		while(tmp--)
			ans=(ans*3)%mod;
	}
	cout<<ans<<endl;
	return 0;
}
