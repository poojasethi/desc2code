#include<bits/stdc++.h>
using namespace std;
const int maxn=1e5+100;
char s[maxn];
typedef long long LL;
const LL mod=1e9+7;
LL fac[maxn];
int cal(char c)
{
	if('0'<=c&&c<='9')return c-'0';
	else if('A'<=c&&c<='Z')return c-'A'+10;
	else if('a'<=c&&c<='z')return c-'a'+36;
	else if('-'==c)return 62;
	else return 63;
}
int main()
{
	for(int i=0;i<64;++i)
		for(int j=0;j<64;++j)++fac[i&j];
	scanf("%s",s);
	int n=strlen(s);
	LL res=1;
	for(int i=0;i<n;++i)
		res=fac[cal(s[i])]*res%mod;
	printf("%I64d\n",res);
	return 0;
}
