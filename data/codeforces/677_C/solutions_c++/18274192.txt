#include<bits/stdc++.h>
using namespace std;
int v[256];
int ans[256];

int main()
{	v['-']=62;
	v['_']=63;
	for(int i='0';i<='9';i++)v[i]=i-'0';
	for(int i='A';i<='Z';i++)v[i]=i-'A'+10;
	for(int i='a';i<='z';i++)v[i]=i-'a'+36;
	for(int i=0;i<=63;i++)
	{	for(int j=0;j<=63;j++)ans[i&j]++;
	}
	string a;
	cin>>a;//scanf("%s",a);
	long long int resul=1;
	int n=a.length();
	for(int  i=0;i<n;i++)resul=(resul*ans[v[(int)a[i]]])%1000000007;
	cout<<resul<<endl;
	return 0;
}
