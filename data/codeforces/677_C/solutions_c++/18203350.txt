#include<bits/stdc++.h>
using namespace std;
const long long md = 1e9+7;
int main(void) {
	string s;
	cin>>s;
	long long ans = 1;
	for(int i=0;i<s.length();i++) {
		long long int t;
		if(s[i] >= '0' && s[i] <= '9')
			t = s[i] - '0';
		if(s[i] >= 'A' && s[i] <= 'Z')
			t = s[i] - 'A' + 10;
		if(s[i] >= 'a' && s[i] <= 'z')
			t = s[i] - 'a' + 36;
		if(s[i] == '_')
			t = 63;
		if(s[i]== '-')
			t = 62;
		for(int j=0;j<6;j++) {
			if((t%2)==0) {
				ans = (ans * 3)%md;
			}
			t/=2;
		}
		
	}
	cout<<ans<<endl;
	return 0;
}
