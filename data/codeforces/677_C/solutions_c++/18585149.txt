#include <bits/stdc++.h>
using namespace std;
#define m 1000000007;

int main()
{
	string s;
	int i,val,n,count=0,j;
	long long int a=1;
	cin>>s;
	for(i=0;i<s.length();i++)
	{
		if(s[i]=='-')
			val=62;
		else if(s[i]=='_')
			val=63;
		else if(s[i]>='0' && s[i]<='9')
			val=s[i]-'0';
		else if(s[i]>='a' && s[i]<='z')
			val=s[i]-'a'+36;
		else if(s[i]>='A' && s[i]<='Z')
			val=s[i]-'A'+10;

		for(j=0;j<6;j++)
		{
			n=val&1;
			val=val>>1;
			if(n==0)
				count++;
		}
	}
	for(i=0;i<count;i++)
	{
		a*=pow(3,1);
		a=a%m;
	}
	cout<<a<<endl;
	return 0;
}
