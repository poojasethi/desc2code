#include<cstdio>
#include<cstring>
const int mod=1e9+7,N=100005;
int a[64],x;
char s[N];
long long ans;
int main(){
	for (int i=0;i<64;i++){
		a[i]=1;
		for (int j=0;j<6;j++)
		 if (!((i>>j)&1)) a[i]=3ll*a[i]%mod; 
	}
	scanf("%s",s);
	int len=strlen(s);ans=1;
	for (int i=0;i<len;i++){
		if (s[i]>='0' && s[i]<='9') x=s[i]-'0';
		else if (s[i]>='A' && s[i]<='Z') x=10+s[i]-'A';
		else if (s[i]>='a' && s[i]<='z') x=36+s[i]-'a'; 
		else if (s[i]=='-') x=62;
		else x=63;
		ans=ans*a[x]%mod;
	}
	printf("%I64d",ans);
	return 0;
}
