#include<cstdio>
using namespace std;
const int mod=1000000007;

char s[200000];
long long ans;
int pow[10],map[500];

int main(){
	for (int i=0;i<=9;i++) map['0'+i]=i;
	for (int i=0;i<26;i++) map['A'+i]=i+10;
	for (int i=0;i<26;i++) map['a'+i]=i+36;
	map['-']=62;
	map['_']=63;
	ans=1;
	scanf("%s",s);
	pow[0]=1;
	for (int i=1;i<=6;i++) pow[i]=pow[i-1]*3;
	for (int i=0;s[i]!='\0';i++){
		int a=map[s[i]],cnt=0;
		while(a){
			if (a&1) cnt++;
			a>>=1;
		}
		ans=ans*pow[6-cnt]%mod;
	}
	printf("%I64d",ans);
	return 0;
}
