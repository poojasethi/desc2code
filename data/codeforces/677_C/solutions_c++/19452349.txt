#include<iostream>
#include<cstring>
typedef long long ll;
const ll mod=1e9+7;
int cnt(char t){int s=int(t);return s==95?63:s==45?62:s>96?s-61:s>64?s-55:s-48;}
main(){
    ll a[64],rep=1,i=0,j;
    char r[100000];
    for(;i<64;a[i++]=0);
    std::cin>>r;
    for(i=0;i<64;i++)for(j=0;j<64;j++)a[i&j]++;
    for(i=0;i<strlen(r);)rep=rep*a[cnt(r[i++])]%mod;
    std::cout<<rep;
}
