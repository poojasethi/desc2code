#include <cstring>
#include <algorithm>
#include <iostream>
#include <string>
using namespace std;
string s;
long long int n,sum[64],ans=1;
int main()
{
 long long int mod=1e9+7;
	for(int i=0;i<64;i++)
		for(int j=0;j<64;j++)
			sum[i&j]++;
	cin>>s;
	n=s.size();
	for(int i=0;i<n;i++)
{
int t;
char c=s[i];
if(c>='0'&&c<='9') t=c-'0'; else
	if(c>='A'&&c<='Z') t=c-'A'+10; else
	if(c>='a'&&c<='z') t=c-'a'+36; else
	if(c=='-') t=62; else t=63;
		
ans=(long long)ans*sum[t]%mod;
}
	cout<<ans;
}
