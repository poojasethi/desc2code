#include <iostream>

using namespace std;

int main()
{
	long long t, s = 1;
	string a;
	cin >> a;
	for(int i = 0; i < a.length(); i++)
	{
		if(a[i] == '_') t = 63;
		else if(a[i] == '-') t = 62;
		else if(a[i] < 60) t = a[i] & 15;
		else if(a[i] & 32) t = (a[i]&31) + 35;
		else t=(a[i] & 31) + 9;
		for(int j = 1; j < 64; j <<= 1)
			if(!(t & j))
			{
				s *= 3;
				s %= 1000000007;
			}
	}
	cout << s << endl;
}
