#include<cstdio>
#include<cstring>
using namespace std;
int i,j,l;
long long ans[70],res=1,P=1e9+7;
char s[100010];
int main()
{
	for (i=0;i<=63;i++)
	  for (j=0;j<=63;j++)
	    ans[i&j]++;
	scanf("%s",s);
	l=strlen(s);
	for (i=0;i<l;i++)
	{
		if (s[i]>='0'&&s[i]<='9') res=res*ans[s[i]-'0']%P;
		else if  (s[i]>='A'&&s[i]<='Z') res=res*ans[s[i]-'A'+10]%P;
		else if  (s[i]>='a'&&s[i]<='z') res=res*ans[s[i]-'a'+36]%P;
		else if  (s[i]=='-') res=res*ans[62]%P;
		else if  (s[i]=='_') res=res*ans[63]%P;
	}
	printf("%I64d\n",res);
 }
