#include<cstdio>

const long long mod=1000000007;
char input[100010];
long long A;

int main()
{
	scanf("%s\n",input);

	A=1;
	for(int i=0,k;input[i];i++)
	{
		if(input[i]>='0'&&input[i]<='9') k=input[i]-'0';
		else if(input[i]>='A'&&input[i]<='Z') k=input[i]-'A'+10;
		else if(input[i]>='a'&&input[i]<='z') k=input[i]-'a'+36;
		else if(input[i]=='-') k=62;
		else if(input[i]=='_') k=63;

		for(int p=0;p<6;p++)
		{
			if(!(k&1))
			{
				A+=A<<1;
				A%=mod;
			}
			k>>=1;
		}
	}
	printf("%I64d\n",A);
	return 0;
}
