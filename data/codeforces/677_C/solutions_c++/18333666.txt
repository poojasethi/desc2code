#include<cstdlib>
#include<cstdio>
#include<cstring>
using namespace std;
const int MOD=1e9+7;
char s[100005];
int sum[65];
int change(char c)
{
 if(c>='0'&&c<='9') return c-'0';
 else if(c>='A'&&c<='Z') return c-'A'+10;
 else if(c>='a'&&c<='z') return c-'a'+36;
 else if(c=='-') return 62;
 else if(c=='_') return 63;
}
int main()
{
 for(int i=0;i<64;i++)
  for(int j=0;j<64;j++)
   sum[i&j]++;
 long long ans=1;
 scanf("%s",s);
 for(int i=0;i<strlen(s);i++)
  ans=((ans*sum[change(s[i])]))%MOD;
 printf("%d\n",ans);
}
