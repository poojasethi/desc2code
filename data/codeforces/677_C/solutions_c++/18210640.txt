#include <bits/stdc++.h>
using namespace std;


int main() {
	string s;
	cin>>s;
	char m[66];
	int cu=0;
	for(char i='0';i<='9';i++)
		m[cu++]=i;
	for(char i='A';i<='Z';i++)
			m[cu++]=i;
	for(char i='a';i<='z';i++)
			m[cu++]=i;
	m[cu++]='-';
	m[cu++]='_';
	long long  ss=1;
	for(int i=0;i<s.size();i++)
	{	long long t=0;
		for(int j=0;j<cu;j++)
		for(int k=0;k<cu;k++)
			if((j&k)<=65 &&m[j&k]==s[i])
			t++;
		ss=(ss*t)%1000000007;
	}
	cout<<ss<<endl;
		return 0;
}
