#include <bits/stdc++.h>
using namespace std;
const int mod = 1e9+7;
#define ll long long
int a[65];
int main()
{
for(int i=0;i<64;i++)
for(int j=0;j<64;j++)
a[i&j]++;

string s;
cin>>s;
long long int n=s.length();
long long int ans =1;
for(int i=0;i<n;i++)
{
if(s[i]>='0'&&s[i]<='9') ans*=a[s[i]-'0']%mod;
if(s[i]>='A'&&s[i]<='Z') ans*=a[s[i]-'A'+10]%mod;
if(s[i]>='a'&&s[i]<='z') ans*=a[s[i]-'a'+36]%mod;
if(s[i]=='-') ans*=a[62]%mod;
if(s[i]=='_') ans*=a[63]%mod;
ans=ans%mod;
}
cout<<ans<<endl;
}
