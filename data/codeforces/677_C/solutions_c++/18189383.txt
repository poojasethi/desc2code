#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
const long long mo=1000000007;
int a[100],i,j,l;
long long ans=1;
char s[100010];
int change(char c)
{
	if (c>='0' && c<='9') return c-'0';
	if (c>='A' && c<='Z') return c-'A'+10;
	if (c>='a' && c<='z') return c-'a'+36;
	if (c=='-') return 62;
	return 63;
}
int main()
{
		for (i=0;i<64;++i)
			for (j=0;j<64;++j) ++a[i&j];
		scanf("%s",s);l=strlen(s);
		for (i=0;i<l;++i) ans=ans*a[change(s[i])]%mo;
		printf("%I64d\n",ans);
}
