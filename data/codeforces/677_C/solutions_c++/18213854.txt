#include<iostream>
#include<string>

using namespace std;

int ctoi(char a)
{
	if(a >= '0' && a <= '9') return a - '0';
	if(a >= 'A' && a <= 'Z') return a - 'A' + 10;
	if(a >= 'a' && a <= 'z') return a - 'a' + 36;
	if(a == '-') return 62;
	return 63;
}

int main()
{
	long long int ans=1;
	string a;
	cin>>a;
	for(int i=0; i<a.size(); i++)
	{
		int temp = ctoi(a[i]);
		for(int j=0; j<6; j++)
		{
			if(!(temp & 1))
			{
				ans *= 3;
				ans %= 1000000007;
			}
			temp = temp >> 1;
		}
	}
	cout << ans << endl;
}
