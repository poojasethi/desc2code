#include<bits/stdc++.h>

using namespace std;



int main() {
	char s[100010];

	scanf("%s", s);

	long long int num, i, j, mod = 1000000007, ans = 1;
	for(i=0; s[i]!='\0'; ++i) {
		if(s[i]>='0' && s[i]<='9')
			num = s[i]-'0';
		else if(s[i]>='A' && s[i]<='Z') 
			num = s[i] - 'A' + 10;		
		else if(s[i]>='a' && s[i]<='z') 
			num = s[i] - 'a' + 36;
		else if(s[i] == '-') 
			num = 62;
		else 
			num = 63;
		for(j=0; j<6; ++j)
			if((num&(1<<j)) == 0)
				ans = (3*ans)%mod;
	}

	printf("%lld\n", ans);

}
