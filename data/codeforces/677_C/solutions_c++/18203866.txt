#include<bits/stdc++.h>
using namespace std;
const int mod=1e9+7;
int t[200],pre[200];
char s[100005];
int main()
{
   memset(t,0,sizeof(t));
   memset(pre,0,sizeof(pre));
   for(int i=0;i<10;i++)
    t['0'+i]=i;
   for(int i=0;i<26;i++)
   {t['a'+i]=36+i;
    t['A'+i]=10+i;
   }
   t['-']=62;
   t['_']=63;
   for(int i=0;i<64;i++)
    for(int j=0;j<64;j++)
    pre[i&j]++;

   scanf("%s",s);
   long long  ans=1;
   for(int i=0;i<strlen(s);i++)
    ans=ans*pre[t[s[i]]]%mod;
    printf("%lld\n",ans);
}
