#include<iostream>
#include<string>
using namespace std;
long long q[1000]= {0},num=1;
int i,j,p;
string s;


int main() {
	
	for(i=0; i<=63; i++) {
		for(j=0; j<=63; j++) 
			q[i&j]++;
	}

	cin>>s;
	for(i=0; i<s.length(); i++) {
		if(s[i]>='0'&&s[i]<='9') 		p=s[i]-'0';
		else if(s[i]>='A'&&s[i]<='Z') 	p=s[i]-'A'+10;
		else if(s[i]>='a'&&s[i]<='z') 	p=s[i]-'a'+36;
		else if(s[i]=='-') 				p=62;
		else if(s[i]=='_') 				p=63;
		num*=q[p];
		num%=1000000007;
	}
	cout<<num<<endl;
	return 0;
}
