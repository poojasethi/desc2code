#include <iostream>
#define mod int(1e9 + 7)
using namespace std;

string s;
int n;
long long res = 1;

int get( char c ){
	if( c == '-' ) return 62;
	if( c == '_' ) return 63;
	if( c <= '9' ) return c - '0';
	if( 'A' <= c && c <= 'Z' ) return 10 + c - 'A';
    return 36 + c - 'a';
}

int main() {
	cin >> s;
	for(int i = 0 ; i < s.size() ; i++){
	    int now = get(s[i]);
	    for(int i = 0 ; i < 6 ; i++)
		    if( !( (1 << i) & now ) ) res = (res * 3) % mod;
	}
	cout << res << endl;
}
