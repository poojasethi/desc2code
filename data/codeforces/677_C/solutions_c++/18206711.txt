#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll m,n,i,j,k,a[100005],x,mm=1e9+7,ans=1;
map<char,ll>mp;
string s;
int main()
{
   ios_base::sync_with_stdio(false);
   j=0;
   for(i='0';i<='9';i++)
    mp[i]=j++;
    j=10;
   for(i='A';i<='Z';i++)
    mp[i]=j++;
    j=36;
   for(i='a';i<='z';i++)
    mp[i]=j++;
    mp['-']=62;
    mp['_']=63;
   cin>>s;
   j=s.size();
   for(i=0;s[i];i++)
   {
       x=mp[s[i]];
       for(j=0;j<6;j++)
       if(((1<<j)&x)==0)
       ans=(ans*3)%mm;
   }
   cout<<ans;

}
