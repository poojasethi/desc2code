#include <bits/stdc++.h>

using namespace std;


int valor(char c){
	if(c>='0' && c<='9') return c-'0';
	if(c>='A' && c<='Z') return c-'A'+10;
	if(c>='a' && c<='z') return c-'a'+36;
	if(c=='-') return 62;
	return 63;
}
int cant(int n){
	int res=0;
	for(int i=0;i<6;i++){
		if(n%2==0) res++;
		n/=2;
	}
	return res;
}


int main(){
	int modul=1e9+7;
	string s;
	cin>>s;
	long long int r=1;
	for(int i=0;i<(int)s.size();i++){
		r*=pow(3, cant(valor(s[i])));
		r%=modul;
	}
	cout<<r<<endl;
	return 0;
}
