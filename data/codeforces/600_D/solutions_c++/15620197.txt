#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
long double A(long double r,long double w,long double d){
  long double c=acos((r*r+d*d-w*w)/(2*r*d));
  return r*r*(c-sin(2*c)/2);
}
int main(){
  long double x,y,r,u,v,w,d,pi=2*acos(0);
  cin>>x>>y>>r>>u>>v>>w;
  if(r>w)swap(x,u),swap(y,v),swap(r,w);
  d=sqrt((x-u)*(x-u)+(y-v)*(y-v));
  if(d>=r+w)puts("0");else if(d<=w-r)printf("%.32f\n",(double)(pi*r*r));
  else printf("%.32f\n",(double)(A(r,w,d)+A(w,r,d)));
}
