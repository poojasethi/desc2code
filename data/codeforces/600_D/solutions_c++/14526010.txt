#include <bits/stdc++.h>
using namespace std;
double xa,xb,ya,yb,r1,r2;
const double eps=1e-12;
long double Hypot(long double a,long double b)
{
    return sqrt(a*a+b*b);
}
long double solve()
{
    long double pi=acos(-1.0);
    long double r=Hypot(xa-xb,ya-yb);
    if(r>=r1+r2-eps)
        return 0.0;
    if(r+r1<=r2+eps)
        return pi*r1*r1;
    if(r+r2<=r1+eps)
        return pi*r2*r2;
    long double ans=0,thi;
    thi=acos((r1*r1+r*r-r2*r2)/(2*r1*r));
    ans+=r1*r1*(thi-sin(thi)*cos(thi));
    thi=acos((r2*r2+r*r-r1*r1)/(2*r2*r));
    ans+=r2*r2*(thi-sin(thi)*cos(thi));
    return ans;

}
int main()
{
    scanf("%lf%lf%lf",&xa,&ya,&r1);
    scanf("%lf%lf%lf",&xb,&yb,&r2);
    printf("%.15lf\n",(double)solve());
    return 0;
}
