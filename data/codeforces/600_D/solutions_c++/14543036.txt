#include <cmath>
#include <cstdio>
#define pi acos(-1)

double r1,r2,t;
struct p { double x,y; } p1,p2;

long double ans () {
    long double d=sqrtl((p1.x-p2.x)*(p1.x-p2.x)+(p1.y-p2.y)*(p1.y-p2.y));
    if (d >= r1+r2) return 0;
    if (d <= r1-r2) return pi*r2*r2;
    if (d <= r2-r1) return pi*r1*r1;
    long double h1,h2,a=0;
    h1 = acosl((r1*r1+d*d-r2*r2)/(2*r1*d));
    h2 = acosl((r2*r2+d*d-r1*r1)/(2*r2*d));
    a += h1*r1*r1;
    a -= r1*r1*sinl(h1)*cosl(h1);
    a += h2*r2*r2;
    a -= r2*r2*sinl(h2)*cosl(h2);
    return a;
}

int main () {
    scanf("%lf%lf%lf", &p1.x, &p1.y, &r1);
    scanf("%lf%lf%lf", &p2.x, &p2.y, &r2);
    printf("%.15lf\n", (double)ans());
    return 0;
}
