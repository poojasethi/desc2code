#include<bits/stdc++.h>
using namespace std;
int main(){
    long double X,Y,R,x,y,r;
    long double pi=acos(-1);
    cin>>X>>Y>>R;
    cin>>x>>y>>r;

    if(R>r) swap(X,x),swap(Y,y),swap(R,r);

    long double d=sqrt((X-x)*(X-x)+(Y-y)*(Y-y));

    cout<<fixed<<setprecision(20);

    if(d>=R+r) printf("0");
    else if(r-R>=d) cout<<pi*R*R;
    else{
        long double A=acos((d*d-r*r+R*R)/(2*d*R));
        long double a=acos((d*d-R*R+r*r)/(2*d*r));
        long double ans=r*r*a+R*R*A;

        A=A*2,a=a*2;
        if(A<pi) ans=ans-0.5*R*R*sin(A);
        else ans=ans+0.5*R*R*sin(A-pi);
        if(a<pi) ans=ans-0.5*r*r*sin(a);
        else ans=ans+0.5*r*r*sin(a-pi);
        //cout<<A<<endl<<a<<endl;
        cout<<ans;}
}
