#include <bits/stdc++.h>

using namespace std;

const long double pi = 3.1415926535897932384;

long double p1,q1,r1,p2,q2,r2,d;
long double alpha,beta;
long double s1,s2;

void input()
{
	cin >> p1 >> q1 >> r1;
	cin >> p2 >> q2 >> r2;
	d = sqrt(pow(p1-p2,2)+pow(q1-q2,2));
}


void solve()
{
	cout.precision(40);
	if(r1 > r2){swap(r1,r2) , swap(p1,p2) , swap(q1,q2);}
	if(d >= r1+r2) {cout <<  0 << endl; return;}
	if(r1+d <= r2) {cout << r1*r1*pi << endl; return;}
	alpha = acos((r1*r1-r2*r2+d*d) / (2*r1*d))*2;
	beta = acos((r2*r2-r1*r1+d*d) / (2*r2*d))*2;
	s1 = (long double)0.5*r1*r1*alpha - 0.5*sin(alpha)*r1*r1;
	s2 = (long double)0.5*r2*r2*beta - 0.5*sin(beta)*r2*r2;
	
	cout << s1 + s2 << endl;
}

int main()
{
	input();
	solve();
	return 0;
}
