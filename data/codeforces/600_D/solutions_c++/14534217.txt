#include<bits/stdc++.h>
using namespace std;
const long double pi=acos(-1.0);
long double X1,Y1,X2,Y2,a,b;
int main(){
    cin>>X1>>Y1>>a;
    cin>>X2>>Y2>>b;
    long double c=sqrt((X1-X2)*(X1-X2)+(Y1-Y2)*(Y1-Y2));
    cout.precision(12);
    if(a>=b&&c<=a-b)cout<<fixed<<pi*b*b<<endl;
    else if(b>=a&&c<=b-a)cout<<fixed<<pi*a*a<<endl;
    else if(a+b<=c)printf("%.20lf",(double)0);
    else {
        //long double di=sqrt((a+b+c)*(-a+b+c)*(a-b+c)*(a+b-c))/c;
        //long double S=di*c/2;
        long double A=acos((a*a+c*c-b*b)/(2*a*c))*2;
        long double B=acos((b*b+c*c-a*a)/(2*b*c))*2;
        long double yo=0.5*A*a*a-0.5*a*a*sin(A);
        long double yoo=0.5*B*b*b-0.5*b*b*sin(B);
        cout<<fixed<<yo+yoo<<endl;
    }
    return 0;
}
