#include <cstdio>
#include <cmath>
#include <iostream>
using namespace std;
typedef long double ld;
const ld pi=3.14159265358;
ld calc(int r1,int r2,ld d)
{
    ld alpha=acos( (pow((ld)r2,2)+pow(d,2)-pow((ld)r1,2))/(2*d*r2) );
    ld s1=alpha*pow((ld)r2,2),s2=sin(alpha)*cos(alpha)*pow((ld)r2,2);
    return s1-s2;
}
int main()
{
    int x1,y1,x2,y2,r1,r2;
    scanf("%d %d %d",&x1,&y1,&r1);
    scanf("%d %d %d",&x2,&y2,&r2);
    ld d=sqrt ( pow((ld)(x1-x2),2) + pow((ld)(y1-y2),2) );
    cout.precision(20);
    if (d>=r1+r2) //相离
        cout<<0;
    else if (abs(r2-r1)>=d) //内含
        cout<<pi*min(pow((ld)r1,2),pow((ld)r2,2));
    else
        cout<<calc(r1,r2,d)+calc(r2,r1,d);
    return 0;
}
