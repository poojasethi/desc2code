#include<bits/stdc++.h>

using namespace std;



int main(){
long double PI=3.1415926535897932384;
long double x0,x1,y0,y1,r0,r1,d,S0,S1,a1,a2;
cin>>x0>>y0>>r0>>x1>>y1>>r1;

d=sqrt((x1-x0)*(x1-x0)+(y1-y0)*(y1-y0));
cout.precision(9);

if (d>(r0+r1))   cout<<0;
else if (d<=abs(r1-r0) && r0>=r1) cout<<PI*r1*r1;
else if (d<=abs(r1-r0) && r0<=r1) cout<<PI*r0*r0;
 else {   S0=acos((r0*r0+d*d-r1*r1)/(2*r0*d))*2;
    S1=acos((r1*r1+d*d-r0*r0)/(2*r1*d))*2;
    a1=(long double)0.5*S1*r1*r1-0.5*r1*r1*sin(S1);
    a2=(long double)0.5*S0*r0*r0-0.5*r0*r0*sin(S0);
cout<<a2+a1;}
}
