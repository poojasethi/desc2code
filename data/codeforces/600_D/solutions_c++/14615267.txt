#include <bits/stdc++.h>

using namespace std;

int main()
{
    //freopen("a.txt", "r", stdin);
    long double x1,y1,r1,x2,y2,r2,d,a1,a2,s1,s2;
    long double PI = acos(-1);
    cout.precision(9);
    while(cin>>x1>>y1>>r1>>x2>>y2>>r2)
    {
        d=sqrt((x2-x1)*(x2-x1)+(y2-y1)*(y2-y1));
        if(d >= r1 + r2)
        {
            cout << 0 << endl;
            continue;
        }
        if (d + r2 <= r1) { cout<<PI*r2*r2 << endl; continue; }
        if (d + r1 <= r2) { cout<<PI*r1*r1 << endl; continue; }
        a1 = acos((r1*r1+d*d-r2*r2)/(2*r1*d));
        a2 = acos((r2*r2+d*d-r1*r1)/(2*r2*d));
        s1 = a1*r1*r1 - cos(a1)*sin(a1)*r1*r1;
        s2 = a2*r2*r2 - cos(a2)*sin(a2)*r2*r2;
        cout << s1+s2 << endl;
    }
    return 0;
}
