#include <bits/stdc++.h>
using namespace std;
#define sqr(x) ((x) * (x))
typedef long double LD;
typedef long long int LL;
LL x, X, y, Y, r, R, di;
LD d;
LD area(bool other) {
	if(other) swap(R, r);
	LD alpha = acosl((sqr(r) + di - sqr(R)) / (2 * r * d));
	return sqr(r) * (alpha - sinl(alpha) * cosl(alpha));
}
int main() {
	cin >> x >> y >> r >> X >> Y >> R;
	di = sqr(x - X) + sqr(y - Y);
	if(sqr(r-R) >= di) {
		cout << setprecision(15) << M_PI * sqr(min(r, R)) << endl;
	} else if(sqr(r+R) <= di) {
		cout << 0.0 << endl;
	} else {
		d = sqrtl(di);
		LD ar = area(false) + area(true);
		cout << setprecision(15) << ar << endl;
	}
	return 0;
}
