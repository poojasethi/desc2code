#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
#define PI (3.14159265358979323846264338324990142)
using namespace std;

long double xx,yy,r1,x2,y2,r2;
long double o1,o2,d,s1,s2;

int main()
{
    cin>>xx>>yy>>r1;
    cin>>x2>>y2>>r2;
    d=sqrt((xx-x2)*(xx-x2)+(yy-y2)*(yy-y2));
    if(d>=r1+r2) {printf("0\n"); return 0;}
    if(d+r2<=r1) {printf("%.10lf\n",(double)(PI*r2*r2)); return 0;}
    if(d+r1<=r2) {printf("%.10lf\n",(double)(PI*r1*r1)); return 0;}
    o1=acos((r1*r1+d*d-r2*r2)/(2*r1*d))*2;
    s1=o1*r1*r1-r1*r1*sin(o1);
    o2=acos((r2*r2+d*d-r1*r1)/(2*r2*d))*2;
    s2=o2*r2*r2-r2*r2*sin(o2);
    printf("%.10lf\n",double(.5*(s1+s2)));
    return 0;
}
