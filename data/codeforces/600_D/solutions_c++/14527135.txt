#include <bits/stdc++.h>
using namespace std;

#define mp(x, y) make_pair((x), (y))

typedef long long ll;

long double x1, y11, r1;
long double x2, y2, r2;
long double d;

int main()
{
	cout << setprecision(12);
	cin >> x1 >> y11 >> r1;
	cin >> x2 >> y2 >> r2;
	d=sqrt((x1-x2)*(x1-x2)+(y11-y2)*(y11-y2));
	if(d<=max(r1,r2)-min(r1,r2)) {
		cout << (4.0*atanl(1.0)*min(r1, r2)*min(r1, r2)) << endl;
	} else if(d>=r1+r2) {
		cout << 0.0 << endl;
	} else {
		long double a1, a2;
		a1=acos((r1*r1+d*d-r2*r2)/(2.0*r1*d));
		a2=acos((r2*r2+d*d-r1*r1)/(2.0*r2*d));
		long double ans=(a1*r1*r1-r1*r1*cos(a1)*sin(a1))+(a2*r2*r2-r2*r2*cos(a2)*sin(a2));
		cout << ans << endl;
	}

	return 0;
}
