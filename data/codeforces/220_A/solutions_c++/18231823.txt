#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,arr[100005],i,c,arr2[100005];
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d",&arr[i]);
		arr2[i] = arr[i];
	}c=0;
	sort(arr,arr+n);
	for(i=0;i<n;i++){
		if(arr[i]!=arr2[i])c++;
	}
	if(c<=2)printf("YES\n");
	else printf("NO\n");
	return 0;
}
