#include<stdio.h>
#include<algorithm>
using namespace std;

int main(){
	int n,a[100010],b[100010],cnt=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++){scanf("%d",&a[i]);b[i]=a[i];}
	sort(b,b+n);
	for(int i=0;i<n;i++){if(a[i]!=b[i]){cnt++;}}
	if(cnt==0||cnt==2){puts("YES");}
	else{puts("NO");}
	return 0;
}
