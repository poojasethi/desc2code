#include<cstdio>
#include<algorithm>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n; scanf("%d",&n);
	int a[100000],b[100000];
	rep(i,n) scanf("%d",a+i), b[i]=a[i];

	sort(b,b+n);

	int cnt=0;
	rep(i,n) if(a[i]!=b[i]) cnt++;
	puts(cnt<=2?"YES":"NO");

	return 0;
}
