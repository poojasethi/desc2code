
def solve():
    str = raw_input()
    length = len(str)-1
    while length > 0:
          if str[length] != '/':
             break
          length -= 1
    
    ans = str[0]
    pre = str[0]
    for i in range(1,length+1):
        if str[i] == '/' and pre == '/':
           continue
        ans += str[i]
        pre = str[i]
    print ans
          
if __name__ == "__main__":
    solve()
