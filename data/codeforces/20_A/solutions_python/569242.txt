#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

def solve(s):
    P = [e for e in s.split('/') if len(e)>0]
    if len(P) == 0:
        return '/'
    else:
        return '/%s' % ('/'.join(P))

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

s = input.readline().strip()

a = solve(s)
output.write('%s\n' % a)
