pyt = raw_input('')

dl = len(pyt)
normPyt = ''
opr = 0
kol = 0

for x in range(dl):
    dob = ''

    if pyt[x] == '/':
        if kol == 1:
            opr = 1

        kol = 1
    else:
        kol = 0
        opr = 0

    if opr == 0:
        dob = pyt[x]

    normPyt = normPyt + dob

dl = len(normPyt)

if dl > 1:
    if normPyt[dl-1] == '/':
        normPyt = normPyt[0:dl-1]

print normPyt
