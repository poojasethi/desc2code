#!/usr/bin/python
import string
import re

s = raw_input()
if re.match("^/*$", s):
	print "/"
	exit(0)
result = re.match("^(.*[^/])/+$", s)
if result:
	s = result.group(1)
t = []
slashed = False
for i in range(len(s)):
	if s[i] == '/':
		if slashed:
			continue
		else:
			t.append('/')
			slashed = True
	else:
		slashed = False
		t.append(s[i])
print string.join(t, '')
