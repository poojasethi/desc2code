#!usr/bin/env/python
path = raw_input()
new_path = ''
flag = 0
for i in xrange(len(path)):
	if path[i] == '/' and flag:
		pass
	elif path[i] == '/':
		flag = 1
		new_path += path[i]
	else:
		flag = 0
		new_path += path[i]
while len(new_path) > 1 and new_path[-1] == '/':
	new_path = new_path[:-1]
print new_path
