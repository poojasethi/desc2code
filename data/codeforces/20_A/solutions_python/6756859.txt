# -*- coding: utf-8 -*-
import sys
def fun():
    try:
        s=map(str,raw_input().split())
    except ValueError:
        print 'Invalid number'
        return
    ss=""
    b=False;
    for i in s[0]:
        
        if i=='/':
            if b==False:
                ss=ss+i
            b=True
        else:
            b=False
            ss=ss+i
    if len(ss)>1 and ss[-1]=='/':
        ss=ss[0:-1]
    print ss;
if __name__ == "__main__":
    fun()
