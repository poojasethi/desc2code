import sys

def nextLine():
	return sys.stdin.readline().strip()

def removeSlash(l):
	return l.split('/')

def makeStr(k):
	s = ""
	for i in k:
		if i != '':
			s += '/'
			s += i
	return s


line = nextLine();

for i in line:
	if i != '/':
		print "%s" % makeStr(removeSlash(line))
		sys.exit(0)
print "/"
