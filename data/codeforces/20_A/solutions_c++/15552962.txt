#include <iostream>

using namespace std;

int main()
{
	string s;
	cin >> s;
	cout << s[0];
	int pt = s.size() - 1;
	while (s[pt] == '/')
		pt--;
	for (int i = 1; i <= pt; i++)
		if (s[i] == '/' && s[i - 1] != '/')
			cout << s[i];
		else if (s[i] != '/')
			cout << s[i];
	cout << endl;
	return 0;
}
