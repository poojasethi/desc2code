#include<cstdio>
#include<cstring>
#include<cctype>
using namespace std;
int main(){
    int flag=0,k=0;
    char ch,q[105];
    while((ch=getchar())&&ch!=EOF){
        if(k==0||islower(ch)||q[k-1]!='/')q[k++]=ch;
    }
    if(q[k-1]=='/'&&k>1)k--;
    for(int i=0;i<k;i++)printf("%c",q[i]);

}
