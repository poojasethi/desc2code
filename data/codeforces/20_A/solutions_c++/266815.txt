#include <stdio.h>

#define N 128

int main()
{
  char a[N], b[N];
  int i, j;

  scanf("%s", a);

  for (i=j=0; a[i]; )
  {
    b[j] = a[i];
    j++;

    if (a[i]=='/')
      while(a[i]=='/') i++;
    else 
      i++;
  }

  if (j>1 && b[j-1]=='/')
    j--;

  b[j] = 0;
  printf("%s\n" , b);
  return 0;
}
