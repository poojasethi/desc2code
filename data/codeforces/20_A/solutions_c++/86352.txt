#include <iostream>
#include <string>

using namespace std;

int main()
{
  string s;
  cin>>s;
  bool hay=false;
  //cout<<s[0];
  for (int i=1;i<int(s.size());i++) {
    if (s[i]!='/') {
      if (s[i-1]=='/') {
	cout<<'/';
	hay=true;
      }
      cout<<s[i];
    }
  }
  if (not hay) cout<<'/';
  cout<<endl;
}
