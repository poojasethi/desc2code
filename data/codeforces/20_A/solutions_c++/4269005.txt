#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
int main (){
	int i,f;
	string s;
	cin >> s;
	f=s.size()-1;
	while(s[f]=='/' && f>0) f--;
	f++;
	s=s.substr(0,f);
	for(i=0;i<s.size();i++)
	{
		if(i && s[i-1]=='/' && s[i]=='/')
			continue;
		printf("%c",s[i]);
	}
	puts("");
	return 0;
}
