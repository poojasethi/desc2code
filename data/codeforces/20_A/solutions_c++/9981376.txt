#include <iostream>
#include <string>
using namespace std;
int main()
{
	int result;
	bool spd=false,f=false;
	string s;
	cin>>s;
	for(int i=0;i<s.size();i++){
		if(s[i]>='a' && s[i]<='z'){
			if(spd)cout<<"/";
			spd=false;
			cout<<s[i];
			f=true;
		}
		else spd=true;
	}
	if(!f)cout<<"/";
	return 0;
}
