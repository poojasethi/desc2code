#include<bits/stdc++.h>
using namespace std;
int main ()
{
    string s;
    int i,j,cnt=0,d=0;
    cin>>s;
    for( i=0; i<s.size(); i++ ){
        if( s[i]=='/' ){
            d=0;
            continue;
        }
        cnt++;
        if( d==0 )cout<<'/';
        d++;
        cout<<s[i];
    }
    if( !cnt )cout<<'/';
}
