#include <iostream>
#include <string>

using namespace std;

int main () {
	string s;
	cin >> s;
	for(int i = 0;i < s.size();) {
		if(s[i]=='/'&&s.size()>i+1&&s[i+1]=='/') {
			s.erase(s.begin()+i);
		} else
			i++;
	}
	if(s.size()>1 && s[s.size()-1]=='/') s.erase(s.begin()+s.size()-1);
	cout << s << endl;
	return 0;
}
