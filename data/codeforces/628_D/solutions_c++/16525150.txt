#include <bits/stdc++.h>
using namespace std;
#define LL long long
const int pp=(1e9)+7;
int M,D,ans,len;
char a[2010],b[2010];
int f[2010][2010];
int pow10[2010];
int calc(char a[]){
	int res=0;
	int rest=0;
	for (int i=1;i<=len;i++){
		for (int j=0;j<a[i]-'0';j++){
			if ((i&1)&&j==D)continue;
			if ((i&1)==0&&j!=D)continue;
			int rr=((LL)j*pow10[i]+rest)%M;
			res=(res+f[i+1][(M-rr)%M])%pp;
		}
		if (((i&1)&&a[i]-'0'!=D)
			||(!(i&1)&&a[i]-'0'==D))
				rest=((LL)(a[i]-'0')*pow10[i]+rest)%M;
		else break;
	}
	return res;
}
int main(){
	scanf("%d%d",&M,&D);
	scanf("%s%s",a+1,b+1);
	len=strlen(a+1);
	f[len+1][0]=1;
	for (int i=len;i;i--){
		if (i==len)pow10[i]=1;
		else pow10[i]=pow10[i+1]*10%M;
		for (int j=0;j<M;j++){
			if (!f[i+1][j])continue;
			if (i&1){
				for (int k=0;k<10;k++)
					if (k!=D)
						f[i][((LL)k*pow10[i]+j)%M]=(f[i][((LL)k*pow10[i]+j)%M]+f[i+1][j])%pp;
			}else f[i][((LL)D*pow10[i]+j)%M]=(f[i][((LL)D*pow10[i]+j)%M]+f[i+1][j])%pp;
		}
	}
	ans=(calc(b)-calc(a)+pp)%pp;
	int rest=0;
	for (int i=1;i<=len;i++){
		rest=((LL)rest*10+b[i]-'0')%M;
		if ((!(i&1)&&b[i]-'0'!=D)
			||((i&1)&&b[i]-'0'==D)){
			rest=-1;
			break;
		}
	}
	if (!rest)ans=(ans+1)%pp;
	printf("%d",ans);
}
