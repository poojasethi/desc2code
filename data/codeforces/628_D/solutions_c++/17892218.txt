int dp[N][N][2][2],m,d,na,nb;
string a,b;
int f(int x,int md,bool byk,bool kck){
	if(x == nb) return md==0;
	int &r = dp[x][md][byk][kck];
	if(r != -1) return r;
	r=0;
	FOR(i,0,9){
		if(i==d and x&1) continue;
		if(i!=d and !(x&1)) continue;
		if(i < (a[x]-'0') and !byk) continue;
		if(i > (b[x]-'0') and !kck) continue;
		r = (r + f(x+1 , (md*10+i)%m , byk|(i>(a[x]-'0')) , kck|(i<(b[x]-'0'))))%mod;
	}
	return r;
}
