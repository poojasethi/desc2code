#include <bits/stdc++.h>
#define MOD 1000000007

using namespace std;

const int MAXN = 2005;

int len;
int n,d;
int uno;
char A[MAXN];
char B[MAXN];
int pre[MAXN];
int memo[MAXN][2][2005];

int DP(int id,bool f,int mod){
	if(!uno && !A[id]) return mod==0;
	else if(!A[id]) return f && mod==0;
	
	if(memo[id][f][mod]!=-1) return memo[id][f][mod];
	
	int r = 0;
	for(int i= ( !id ? 1 : 0 );i<=9 && ( i<=A[id]-48 || f );i++)
		if(id&1 && i==d)
			r = ( r + DP( id + 1 , max( f , i<A[id]-48 ) , ( mod + pre[len - id - 1] * i ) % n ) ) % MOD;
		else if(!(id&1) && i!=d)
			r = ( r + DP( id + 1 , max( f , i<A[id]-48 ) , ( mod + pre[len - id - 1] * i ) % n ) ) % MOD;
		
	return memo[id][f][mod] = r;
}

int main(){

	scanf("%d %d",&n,&d);
	
	scanf("\n%s",A);
	scanf("\n%s",B);
	
	len = strlen(A);
	
	pre[0] = 1;
	for(int i=1;i<MAXN;i++)
		pre[i] = ( pre[i - 1] * 10 ) % n;
	
	for(int i=0;A[i];i++)
		for(int j=0;j<2;j++)
			for(int h=0;h<n;h++)
				memo[i][j][h] = -1;
		
	swap( A , B );
		
	int r = DP( 0 , 0 , 0 );
	
	swap( A , B );
	
	for(int i=0;A[i];i++)
		for(int j=0;j<2;j++)
			for(int h=0;h<=n;h++)
				memo[i][j][h] = -1;
	uno = 1;
	
	r = ( r - DP( 0 , 0 , 0 ) + MOD ) % MOD;
	
	printf("%d\n",r);
	return 0;
}
