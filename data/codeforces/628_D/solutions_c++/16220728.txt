#include<cstdio>
#include<iostream>
#include<algorithm>
#include<cctype>
#include<string>
#include<cstring>

using namespace std;
typedef long long ll;

int m, d, n;
char a[2010], b[2010];
ll dp[2010][2000][2][2];
const int mod = 1000000007;

ll calc(int i, int rem, bool more, bool less) {
	if(i == n) return rem == 0;
	ll &ret = dp[i][rem][more][less];
	if(ret != -1) return ret;
	int x = more ? 0 : a[i]-'0';
	int y = less ? 9 : b[i]-'0';
	if(i & 1) {
		if(d+'0' > b[i] && !less)
			return ret = 0;
		if(d+'0' < a[i] && !more)
			return ret = 0;
		return ret = calc(i+1, (rem*10+d)%m, d+'0' > a[i] ? 1 : more, d+'0' < b[i] ? 1 : less)%mod;
	}
	ret = 0;
	for(int j=x ; j<=y ; ++j) {
		if(j == d) continue;
		ret += calc(i+1, (rem*10+j)%m, j+'0' > a[i] ? 1 : more, j+'0' < b[i] ? 1 : less);
		ret %= mod;
	}

	return ret%mod;
}

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d", &m, &d);
	scanf("%s %s", a, b);
	n = strlen(a);
	memset(dp, -1, sizeof dp);
	ll ans = calc(0, 0, 0, 0);
	cout<<ans<<endl;
	return 0;
}
