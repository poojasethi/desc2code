#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define ff(i,a,b) for(ll i = a;i<=b;i++)
#define fo(i,a,b) for(ll i = a;i<b;i++)
#define fod(i,a,b) for(ll i = a;i>=b;i--)
#define pb push_back
#define fi first
#define se second
#define mp make_pair
typedef pair <ll, ll> ii;
typedef vector <ll> vi;
const ll oo = 1e7;
const int MODE = 1e9+7;

int n,d;
ll f[2001][2][2][2001];
string a,b;

ll sol(int pos,int x,int y,int mod){
    if(pos == a.size())
    {
        if(mod == 0)return 1ll;else return 0ll;
    }
    if(f[pos][x][y][mod]!=-1)return f[pos][x][y][mod];
    int lo,hi;
    if(x) lo = 0;else lo = a[pos] - '0';
    if(y) hi = 9; else hi = b[pos] - '0';
    ll ans = 0ll;
    for(int i = lo;i<=hi;i++)
    {
        if(pos%2==0 && i==d)continue;
        if(pos%2!=0 && i!=d)continue;
        int nx,ny;
        if(x==0 && i==lo)nx = 0;else nx = 1;
        if(y==0 && i==hi)ny = 0;else ny = 1;
        ans = (ans + sol(pos+1,nx,ny,(mod*10 + i)%n))%MODE;
    }
    return f[pos][x][y][mod] = ans%MODE;
}

int main(){
    cin>>n>>d>>a>>b;
    memset(f,-1,sizeof f);
    cout<<sol(0,0,0,0);
}
