#include <iostream>
#include <cstring>
#include <cstdio>
const int mo=1e9+7;
int f[2005][10][2005],c[2005],m,ans=0,i,j,k,d,len;
char a[2005],b[2005];
int work()
{
	int ans=0;
	for (int i=1,j,k=0;i<=len;++i)
	{
		j=a[i]-'0';
		for (--j;j>=0;--j)
			ans=(ans+f[i][j][(m-k)%m])%mo;
		(k+=c[i]*(a[i]-'0'))%=m;
		if (i&1)
		{
			if (a[i]-'0'==d) return ans;
		}
		else if (a[i]-'0'!=d) return ans;
	}
	return ans;
}
int check()
{
	int j=0;
	for (int i=1;i<=len;++i)
	{
		j=(j*10+a[i]-'0')%m;
		if (i&1)
		{
			if (a[i]-'0'==d) return 0;
		}
		else if (a[i]-'0'!=d) return 0;
	}
	if (j) return 0;
	return 1;
}
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d%d\n%s\n%s\n",&m,&d,a+1,b+1);
	len=strlen(b+1);
	if (len&1^1) f[len][d][d%m]=1;
		else for (i=0;i<=9;++i) if (i!=d) f[len][i][i%m]=1;
	for (c[len]=1,i=len-1;i;--i) c[i]=10*c[i+1]%m;
	for (i=len;i>1;--i)
		for (j=0;j<=9;++j)
			if (j!=d)
			for (k=0;k<m;++k)
			if (i&1) (f[i-1][d][(k+c[i-1]*d)%m]+=f[i][j][k])%=mo;
				else (f[i-1][j][(k+c[i-1]*j)%m]+=f[i][d][k])%=mo;
	ans=-work();
	for (i=1;i<=len;++i) a[i]=b[i];
	ans+=work()+check();
	printf("%d\n",(ans+mo)%mo);
}
