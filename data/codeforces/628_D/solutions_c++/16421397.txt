const int MN=2048+10;
const int ME=30000+10;
const int Mod=1000000007;
typedef long long LL;
typedef unsigned long long ULL;
const int Intmax=(~(1<<31));
const int Intmin=(1<<31);
const LL LLmax=(~(1LL<<63));
const LL LLmin=(1LL<<63);
const ULL ULLmax=(~(0ULL));
int m,d;
LL dp[MN][MN][2][2];
char s1[MN],s2[MN];
int len;
LL dfs(int id,int mod,int a,int b){
	if(id==len) return mod==0;
	if(dp[id][mod][a][b]!=-1) return dp[id][mod][a][b];
	int i;
	LL ans=0;
	for(i=0;i<10;i++){
		if(!(id&1)&&i==d) continue;
		if((id&1)&&i!=d) continue;
		if(i<s1[id]-'0' && !a) continue;
		if(i>s2[id]-'0' && !b) continue;
		ans=(ans+dfs(id+1,(mod*10+i)%m,(i>s1[id]-'0')|a,(i<s2[id]-'0')|b))%Mod;
	}
	return dp[id][mod][a][b]=ans;
}
