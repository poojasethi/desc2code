#include<stdio.h>
#include<string.h>
char st1[2005]={'\0'},st2[2005]={'\0'};
int m,d,dg[2005][2005][2][2]={0},len;
void calc(int k,int mod,int fs,int fb)
{
   //printf("\n%d %d %d %d %d\n",k,mod,fs,fb,dg[k][mod][fs][fb]);
   if(dg[k][mod][fs][fb]==0) return;
   int nmod;
   for(int i=0;i<=9;i++)
   {
     if(k%2==1&&i!=d) continue;
     if(k%2==0&&i==d) continue;
     if(fs==0&&i<st1[k+1]-'0') continue;
     if(fb==0&&i>st2[k+1]-'0') continue;
     nmod=((mod*10)+i)%m;
     //printf("%d %d %d\n",k+1,i,nmod);
     dg[k+1][nmod][fs|i>st1[k+1]-'0'][fb|(i<st2[k+1]-'0')]+=dg[k][mod][fs][fb];
     dg[k+1][nmod][fs|i>st1[k+1]-'0'][fb|(i<st2[k+1]-'0')]%=(1000000007);
   }
}
int main()
{
   scanf("%d %d",&m,&d);
   scanf("%s %s",st1+1,st2+1);
   len=strlen(st1+1);
   //if(len==1&&d!=0) {printf("0"); return 0;}
   dg[0][0][0][0]=1;
   calc(0,0,0,0);
   int i,j,k,l;
   for(i=1;i<len;i++)
      for(j=0;j<m;j++)
      for(k=0;k<=1;k++)
      for(l=0;l<=1;l++)
      calc(i,j,k,l);
   int ans=(dg[len][0][0][0]+dg[len][0][0][1]+dg[len][0][1][0]+dg[len][0][1][1])%(1000000007);
   printf("%d",ans);
return 0;
}
