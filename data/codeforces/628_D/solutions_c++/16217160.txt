#include <iostream>
#include <cstring>

#define FOR(i, a, b) for(__typeof(a) i = a; i < b; ++i)

using namespace std;

int dp[2001][2001][2][2], m, d, MOD = 1e9 + 7;
string a, b;

int cnt(int idx, int mod, int bi, int sm){
	if(idx == b.size())
		return mod == 0;
	if(dp[idx][mod][bi][sm] != -1)
		return dp[idx][mod][bi][sm];
	long long ans = 0;
	FOR(digit, 0, 10) {
		if(!(idx & 1) && digit == d) continue;
		if((idx & 1)  && digit != d) continue;
		if(digit < a[idx] - '0' && !bi) continue;
		if(digit > b[idx] - '0' && !sm) continue;
		ans += cnt(idx+1, ((mod * 10) + digit) % m, bi | (digit > a[idx] - '0'), sm | (digit < b[idx] - '0')) % MOD;
	}
	ans %= MOD;
	return dp[idx][mod][bi][sm] = ans;
}

int main() {
	ios::sync_with_stdio(false); cin.tie(0);
	memset(dp, -1, sizeof(dp));
	cin >> m >> d >> a >> b;
	cout << cnt(0, 0, 0, 0);
	return 0;
}
