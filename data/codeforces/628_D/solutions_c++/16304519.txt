#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int mm = int(1e9) + 7;
int m, d, ans, f[2002][2001][2], len;
char A[2005], B[2005];

inline void Mod(int &x, int y) {
	x += y;
	if (x >= mm) x -= mm;
}

inline int Solve(char *s, int k) {
	memset(f, 0, sizeof(f));
	f[0][0][1] = 1;//1 equal
	for (int i = 1; i <= len; ++i) {
		for (int j = 0, g; j < m; ++j) {
			if (i & 1) {
				for (int t = 0; t <= 9; ++t)
					if (t != d) {
						g = (j * 10 + t) % m;
						Mod(f[i][g][0], f[i - 1][j][0]);
						if (t < s[i] - '0') Mod(f[i][g][0], f[i - 1][j][1]);
						else if (t == s[i] - '0') Mod(f[i][g][1], f[i - 1][j][1]);
					}
			} else {
				g = (j * 10 + d) % m;
				Mod(f[i][g][0], f[i - 1][j][0]);
				if (d < s[i] - '0') Mod(f[i][g][0], f[i - 1][j][1]);
				else if (d == s[i] - '0') Mod(f[i][g][1], f[i - 1][j][1]);
			}
		}
	}
	if (k == 1) return f[len][0][1] + f[len][0][0];
	return f[len][0][0];
}

int main()
{
	scanf("%d%d", &m, &d);
	scanf("%s", A+1);
	len = strlen(A+1);
	scanf("%s", B+1);
	ans = Solve(B, 1) - Solve(A, 0);
	if (ans < 0) ans += mm;
	cout << ans << endl;
}
