using namespace std;

typedef long long ll;

const int inf = 0x3f3f3f3f;
const int mod = 1e9 + 7;

int f[2002][2002][2][2];
bool vis[2002][2002][2][2];
int a[10001];
int m, d, n;
int b[10001];
char str[10001];

void upd(int& a, int b)
{
	a += b;
	if (a >= mod) a -= mod;
}

int Work(int now, int j, int maska, int maskb)
{
	if (now == n + 1) return j % m == 0;
	if (vis[now][j][maska][maskb]) return f[now][j][maska][maskb];
	vis[now][j][maska][maskb] = true;
	int tmp = 0;
	for (int i = 0; i < 10; i ++)
	{
		if ((now & 1) && i == d) continue;
		if (now % 2 == 0 && i != d) continue;
		if (i < a[now] && maska) continue;
		if (i > b[now] && maskb) continue;
		upd(tmp, Work(now + 1, (j * 10 + i) % m, maska & (i == a[now]), maskb & (i == b[now])));
	}
	return (f[now][j][maska][maskb] = tmp);
}

int main( )
{
	scanf("%d %d", &m, &d);
	scanf("%s", str);
	n = (int )strlen(str);
	for (int i = 1; i <= n; i ++)
		a[i] = str[i - 1] - '0';
	scanf("%s", str);	
	for (int i = 1; i <= n; i ++)
		b[i] = str[i - 1] - '0';
	printf("%d\n", Work(1, 0, 1, 1));
	return 0;
}
