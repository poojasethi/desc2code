#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
using namespace std;
#define maxn 2222
#define LL __int64
const LL MOD = 1e9+7;

int m,d,len;
int bit[maxn];
LL dp[maxn][maxn];

LL dfs(int pos,int yushu,bool yes){
    if(pos==len) return yushu==0;
    if(!yes&&dp[pos][yushu]!=-1) return dp[pos][yushu];
    LL ans=0;
    int End=yes?bit[pos]:9;
    for(int i=0;i<=End;i++){
        if(pos%2==0&&i==d) continue;
        if(pos%2==1&&i!=d) continue;
        ans=(ans+dfs(pos+1,(yushu*10+i)%m,yes&&(i==End)))%MOD;
    }
    if(!yes) dp[pos][yushu]=ans;
    return ans;
}

LL Count(char str[]){
    for(int i=0;i<len;i++) bit[i]=str[i]-'0';
    return dfs(0,0,1);
}

LL is_judge(char str[]){
    int yushu=0;
    for(int i=0;i<len;i++){
        int v=str[i]-'0';
        if((i&1)&&v!=d) return 0;
        if((i&1)==0&&v==d) return 0;
        yushu=yushu*10+v;
        yushu%=m;
    }
    return yushu==0?1:0;
}

int main(){
    memset(dp,-1,sizeof(dp));
    while(scanf("%d%d",&m,&d)!=EOF){
        char a[maxn],b[maxn];
        scanf("%s%s",a,b);
        len=strlen(a);
        printf("%I64d\n",(Count(b)-Count(a)+is_judge(a)+MOD)%MOD);
    }

    return 0;
}
