#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int f[2002][10][2000],a[2002],b[2002],i,j,k,n,m,d,ans=0,mo=1000000007;
char s[2002];
int work(int x)
{
	int i,j,k=0,sum=0;
	for (i=1;i<=n;++i)
	{
		for (j=0;j<a[i];++j) sum=(sum+f[i][j][(m-k)%m])%mo;
		if ((i&1 && a[i]==d) || (!(i&1) && a[i]!=d)) return sum*x;
		k=(k+a[i]*b[i])%m;
	}
	if (!k && x==1) ans++;
	return sum*x;
}
int main()
{
		scanf("%d%d\n",&m,&d);
		scanf("%s\n",s);n=strlen(s);
		for (i=0;i<n;++i) a[i+1]=s[i]-'0';
		for (b[n]=1,i=n-1;i;i--) b[i]=b[i+1]*10%m;
		if (!(n&1)) f[n][d][d%m]=1;
			else for (i=0;i<=9;++i) if (i!=d) f[n][i][i%m]=1;
		for (i=n;i>1;i--)
		if (i&1)
		{
			for (j=0;j<=9;++j)
			if (j!=d) for (k=0;k<m;++k) (f[i-1][d][(d*b[i-1]+k)%m]+=f[i][j][k])%=mo;
		}
		else
		{
			for (j=0;j<=9;++j)
			if (j!=d) for (k=0;k<m;++k) (f[i-1][j][(j*b[i-1]+k)%m]+=f[i][d][k])%=mo;
		}
		ans=work(-1);
		scanf("%s\n",s);
		for (i=0;i<n;++i) a[i+1]=s[i]-'0';
		ans=(ans+work(1)+mo)%mo;
		printf("%d\n",ans);
}
