#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define MAXX 1010
using namespace std;
int w[MAXX],h[MAXX],n,i,j,k,tot,ans=(int)2e9,a[MAXX],b[MAXX],c[MAXX],t,y;
int check(int t)
{
	int nl=0,tot=0,lenc=0;
	for(int i=1;i<=n;i++)
	{
		if (h[i]>t)
		{
			if (w[i]>t) return 2147483647;
			else nl++,tot+=h[i];
 		}
 		else
 		{
 			tot+=w[i];
 			if (w[i]<=t&&h[i]<w[i]) c[++lenc]=h[i]-w[i];
		}
	}
	if (nl>n/2) return 2147483647;
	sort(c+1,c+1+lenc);
	if (lenc>n/2-nl) lenc=n/2-nl;
	for(int i=1;i<=lenc;i++)
		tot+=c[i];
	return tot;
}
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)
		scanf("%d%d",&w[i],&h[i]);
	for(t=1;t<=1000;t++)
	{
		memset(c,0,sizeof(c));
		int che=check(t);
		if (che==2147483647) continue;
		ans=min(ans,t*che);
	}
	cout<<ans<<endl;
}
