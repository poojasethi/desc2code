#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

struct elem {
  int w,h;
};

bool operator<(elem e1,elem e2)
{
  return e1.h-e1.w<e2.h-e2.w;
}

int main()
{
  int n;
  cin>>n;
  vector<elem> v(n);
  for (int i=0;i<n;i++)
    cin>>v[i].w>>v[i].h;
  sort(v.begin(),v.end());
  int sol=1000000001;
  for (int h=1000;h>=0;h--) {
    int sumw=0;
    int giros=0;
    vector<elem> nextv;
    bool error=false;
    for (int i=0;i<n and not error and 2*giros<=n;i++) {
      if (v[i].h>h) {
	if (v[i].w>h) error=true;
	else {
	  giros++;
	  sumw+=v[i].h;
	}
      } else nextv.push_back(v[i]);
    }
    if (error or 2*giros>n) break;
    for (int i=0;i<int(nextv.size());i++) {
      if (nextv[i].h<nextv[i].w and nextv[i].w<=h and 2*(giros+1)<=n) {
	giros++;
	sumw+=nextv[i].h;
      } else sumw+=nextv[i].w;
    }
    sol=min(sol,h*sumw);
  }
  cout<<sol<<endl;
}
