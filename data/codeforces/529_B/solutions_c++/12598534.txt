#include<cstdio>
#include<algorithm>
using namespace std;
struct ff{
	int w,h;
}p[1005];
int ans=1e9+1,n,h[1005];
bool cmp(int a,int b){ return a>b; }
int check(int high)
{
	int num=0,tot=0,len=0;
	for (int i=1; i<=n; ++i)
	{
		if (p[i].h<=high){  
			len+=p[i].w;
			if (p[i].h<p[i].w && p[i].w<=high) h[++tot]=p[i].w-p[i].h;
		}
		else {
			if (p[i].w>high) return 0;
			len+=p[i].h;
			++num;
			if (num>n/2) return 0;
		}
	}
	sort(h+1,h+1+tot,cmp);
	for (int i=1; i<=min(tot,n/2-num); ++i) len-=h[i];
	ans=min(ans,len*high);
	//printf("get %d\n",high);
}
int main()
{
	scanf("%d",&n);
	for (int i=1; i<=n; ++i)
	  scanf("%d%d",&p[i].w,&p[i].h);
	for (int i=1; i<=n; ++i) check(p[i].h);
	for (int i=1; i<=n; ++i) check(p[i].w);
	printf("%d",ans);
}
