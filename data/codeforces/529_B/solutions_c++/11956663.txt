#include<stdio.h>
#include<algorithm>
using namespace std;
int w,n,arr[1010],dap=2147483647;
struct data
{
    int x,y;
}a[1010];
void pro()
{
    int i,j,sum,t;
    for(i=1;i<=1000;i++)
    {
        w=0; sum=0; t=n/2;
        for(j=1;j<=n;j++)
        {
            if(a[j].x<=i && a[j].y<=i)
            {
                if(a[j].x>a[j].y) arr[++w]=a[j].y-a[j].x;
                sum+=a[j].x;
                continue;
            }
            if(a[j].x>i && a[j].y>i) break;
            if(a[j].y>i) t--, sum+=a[j].y;
            else sum+=a[j].x;
        }
        if(j!=n+1 || t<0) continue;
        sort(arr+1,arr+w+1);
        for(j=1;j<=min(t,w);j++)
        {
            sum+=arr[j];
        }
        if(dap>sum*i) dap=sum*i;
    }
}
int main()
{
    int i;
    scanf("%d",&n);
    for(i=1;i<=n;i++) scanf("%d %d",&a[i].x,&a[i].y);
    pro();
    printf("%d",dap);
    return 0;
}
