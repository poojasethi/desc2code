#include <bits/stdc++.h>
using namespace std;
const int maxn = 1005;
int w[maxn], h[maxn];
pair <int, int> c[maxn];
int main() {
	int n, up = 0, down = 0;
	cin >> n;
	for(int i = 1; i <= n; i++) {
		cin >> w[i] >> h[i];
		up = max(up, max(w[i], h[i]));
		down = max(down, min(w[i], h[i]));
	}
	int ans = 0x7fffffff;
	for(int i = down; i <= up; i++) {
		int idx = 0, cnt = 0, x = 0;
		for(int j = 1; j <= n; j++) {
			if(h[j] > i) {
				cnt++;
				x += h[j];
			} else if(w[j] <= i && h[j] < w[j]) {
				c[idx++] = make_pair(w[j] - h[j], j);
			} else {
				x += w[j];
			}
		}
		if(cnt > n / 2) continue;
		sort(c, c + idx);
		reverse(c, c + idx);
		for(int j = 0; j < min(idx, n / 2 - cnt); j++) x += h[c[j].second];
		for(int j = min(idx, n / 2 - cnt); j < idx; j++) x += w[c[j].second];
		//printf("i = %d, x = %d\n", i, x);
		ans = min(ans, x * i);
	}
	cout << ans << endl;
	return 0;
}
