#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<math.h>
#include<vector>
#include<set>
#include<map>
#include<queue>
using namespace std;
char ch[1000001];
struct P{
	int w,h;
}p[1000];
bool cmp(P a,P b){
	return a.w-a.h>b.w-b.h;
}
int main(){
	int n;
	while(scanf("%d",&n)!=EOF){
		for(int i=0;i<n;i++){
			scanf("%d%d",&p[i].w,&p[i].h);
		}
		sort(p,p+n,cmp);
		int ans=1000*1000*1000;
		for(int h=1;h<=1000;h++){
			int ct=n/2,sum=0;
			for(int i=0;i<n&&sum!=-1;i++){
				if(p[i].h>h){
					if(p[i].w>h){
						sum=-1;
					}
					else if(ct<=0){
						sum=-1;
					}
					else{
						ct--;
						sum+=p[i].h;
					}
				}
			}
			if(sum==-1)continue;
			for(int i=0;i<n;i++){
				if(p[i].h<=h){
					if(ct>0&&p[i].w<=h&&p[i].h<p[i].w){
						ct--;
						sum+=p[i].h;
					}
					else sum+=p[i].w;
				}
			}
			ans=min(ans,sum*h);
		}
		printf("%d\n",ans);
	}
}
