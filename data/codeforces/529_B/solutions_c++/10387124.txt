#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
int x[1005],y[1005];
int n;
void read(){
	int i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d %d",&x[i],&y[i]);
}
vector < int > v;
int dene(int a){
	int i,t=0,m=n/2;
	v.clear();
	for(i=1;i<=n;i++){
		if(x[i]>a && y[i]>a)
			return 1000000000;
		else if(y[i]>a){
			m--;
			t+=y[i];
		}
		else if(x[i]>a)
			t+=x[i];
		else if(x[i]<=y[i])
			t+=x[i];
		else{
			v.push_back(x[i]-y[i]);
			t+=x[i];
		}
	}
	if(m<0)
		return 1000000000;
	sort(v.begin(),v.end());
	for(i=v.size()-1;i>=0 && m;i--,m--)
		t-=v[i];
	return a*t;
}
void solve(){
	int mn=1000000000,i;
	for(i=1;i<=1000;i++)
		mn=min(mn,dene(i));
	printf("%d\n",mn);
}
int main(){
	read();
	solve();
	return 0;
}
