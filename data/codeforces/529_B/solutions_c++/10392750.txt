#include<cstdio>
#include<algorithm>
#define MAXN 1010
using namespace std;
struct mia
{
    int h,w;
    friend inline bool operator<(const mia &x,const mia &y)
    {
        return x.w-x.h>y.w-y.h;
    }
}a[MAXN];
int n,ans=1e9;
int w[MAXN],h[MAXN];
inline void solve(int x)
{
    int sum=0,cnt=0,k=n/2;
    for(int i=1;i<=n;i++)
    {
        if(h[i]>x&&w[i]>x)
            return;
        if(h[i]>=w[i]&&h[i]<=x||w[i]>x&&h[i]<=x)
            sum+=w[i];
        else if(h[i]>x&&w[i]<=x)
            k--,sum+=h[i];
        else
            a[++cnt]=(mia){h[i],w[i]};
    }
    sort(a+1,a+1+cnt);
    for(int i=1;i<=cnt;i++)
        if(k>0)
            sum+=a[i].h,k--;
        else
            sum+=a[i].w;
    if(k>=0)
        ans=min(ans,sum*x); 
}
int main()
{
    int i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%d%d",w+i,h+i);
    for(i=1;i<=1000;i++)
        solve(i);
    printf("%d\n",ans);
    return 0;
}
