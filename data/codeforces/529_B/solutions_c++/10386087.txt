#include <queue>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define M 1010
using namespace std;
int n,ans=0x3f3f3f3f;
int w[M],h[M];
void Check(int limit)
{
    int i,width=0,left=n>>1;
    static priority_queue<int> heap;
    while(heap.size()) heap.pop();
    for(i=1;i<=n;i++)
    {
        if(w[i]>limit&&h[i]>limit)
            return ;
        if(w[i]>limit)
            width+=w[i];
        else if(h[i]>limit)
            --left,width+=h[i];
        else
            width+=w[i],heap.push(w[i]-h[i]);
    }
    if(left<0) return ;
    for(;left;left--)
    {
        if(heap.empty()) break;
        int x=heap.top();heap.pop();
        if(x<0) break;
        width-=x;
    }
    ans=min(ans,width*limit);
}
int main()
{
    int i;
    cin>>n;
    for(i=1;i<=n;i++)
        scanf("%d%d",&w[i],&h[i]);
    for(i=1;i<=1000;i++)
        Check(i);
    cout<<ans<<endl;
    return 0;
}
