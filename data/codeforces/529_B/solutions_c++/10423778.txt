#include <iostream>
#include <fstream>
#include <algorithm>
using namespace std;
long oo=2000000000;
long n,tong,m,kq;
long a[10000],b[10000],c[10000];
long tinh(long h){
	  long w=tong,sl=0,m=0;
	  for (long i=1;i<=n;i++){
	  	  if ((a[i]>h)&&(b[i]>h)) return oo ; else
			if (b[i]>h){	 
			   sl=sl+1;
			   w=w-a[i]+b[i];} else
			if ((a[i]<=h)&&(a[i]>b[i])){
				m=m+1;
				c[m]=b[i]-a[i];
			};
		}
	  if (sl>n/2) return oo;
	  if (m>0) sort(c+1,c+m+1);
	  for (long i=1;i<=m;i++){
	  	  if (sl>=n/2) break;
		   w=w+c[i];
		   sl=sl+1; 
	  };
	  return w*h;
}
int main()
{
	  cin>>n;
	  tong=0;
	  for (long i=1;i<=n;i++){
	  	  cin>>a[i]>>b[i];
	  	  tong=tong+a[i];
	  };
	  kq=oo;
	  for (long i=1;i<=1000;i++)
	     kq=min(kq,tinh(i));
	  cout<<kq;
	  return 0;
}
