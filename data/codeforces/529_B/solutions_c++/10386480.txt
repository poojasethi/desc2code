#include <bits/stdc++.h>

#define MAX 1010

using namespace std;

int n,w[MAX],h[MAX];
int v[MAX];

int main(){

	int resp=1000000010;
	scanf("%d",&n);

	for(int i=0;i<n;i++){
		scanf("%d %d",&w[i],&h[i]);	
	}

	vector<int> v2;
	v2.reserve(MAX);
	
	for(int H=1;H<MAX;H++){
		bool passo=true;
		int cont=n/2;
		int total=0;
		for(int i=0;i<n;i++) v[i]=0;
		for(int i=0;i<n;i++){
			if(h[i]>H){
				if(w[i]>H){
					passo=false;
					break;				
				}else{
					total+=h[i];
					v[i]=1;
					cont--;			
				}	
			}else{
				total+=w[i];
			}		
		}
		if(cont<0) continue;
		if(!passo) continue;
		v2.clear();
		for(int i=0;i<n;i++){
			if(v[i]) continue;
			if(w[i]>H) continue;
			v2.push_back(h[i]-w[i]);								
		}		
				
		sort(v2.begin(),v2.end());
		for(int i=0;i<cont;i++){
			if(v2[i]>=0) break;		
			total+=v2[i];
		}
		resp=min(resp,total*H);					
	}

	printf("%d\n",resp);
}
