#include <bits/stdc++.h>
typedef long long LL;
#define rep(i,n) for(int i = 0;i < n;i ++)
#define pair(x, y) make_pair(x, y)

using namespace std;
typedef pair<int, int> P;
const int N = 1024;
int w[N], h[N], f[N];
int main() {
	int n;
	scanf("%d", &n);
	rep(i, n)
		scanf("%d%d", w + i, h + i);
	int answer = 1e9;
	for(int H = 1000; H; H --) {
		priority_queue<P> Q;
		memset(f, 0, sizeof f);
		int cnt = 0;
		rep(i, n) {
			if(h[i] > H) {
				if(w[i] > H) {
					cnt = -1;
					break;
				} else 	f[i] = 1;
				cnt ++;
			} else if(w[i] <= H && w[i] > h[i]) {
				Q.push(pair(w[i] - h[i], i));
			}
		}
		if(cnt > n / 2)	continue;
		if(cnt == -1)	break;
		while(!Q.empty() && cnt < n / 2) {
			P t = Q.top();
			f[t.second] = 1;
			Q.pop();
			cnt ++;
		}
		int tw = 0;
		rep(i, n) if(f[i]) {
			tw += h[i];
		} else {
			tw += w[i];
		}
		answer = min(answer, tw*H);
	}
	printf("%d\n", answer);
	return 0;
}
