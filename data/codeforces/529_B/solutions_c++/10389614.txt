#include <algorithm>
#include <iostream>
#include <stdio.h>
#include <vector>

#define fi first
#define se second
#define pb push_back

using namespace std;

long long res,n,a[2001],b[2001],ans,h,i;
long long bb[2001];
main()
 {
   cin>>n;
   for (i=1;i<=n;i++)
     {cin>>a[i]>>b[i]; }
  ans=1000000000;
    for (h=1;h<=1000;h++)
     {
      res=0; int pos=0;
      for (i=1;i<=n;i++)
       if (a[i] > h && b[i] > h) break;
       if (i != n+1) continue;
      for (i=1;i<=n;i++)
       if (b[i] > h) { res+=b[i]; pos++; } else
        if (a[i] > h) { res+=a[i]; continue; }
       if (pos > n/2) continue;
       int m=0;
      for (i=1;i<=n;i++)
      {
      if (b[i] > h) continue;
      if (a[i] > h) continue;
      res+=a[i];
      if (a[i] > b[i])
      bb[++m]=a[i]-b[i];
      }
      sort(bb+1,bb+m+1);
      reverse(bb+1,bb+m+1);
      for (int j=1;j<=m && pos < n/2;j++)
      res-=bb[j],pos++;
      ans=min(ans,res*h);
     }
   cout<<ans<<endl;
 }
