pair<int,int>a[1005];
int n,i,b[1005],W,H,q,ans,cnt,flag;
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
  {
    scanf("%d%d",&a[i].x,&a[i].y);
    //b[i].x=a[i].y;b[i].y=a[i].x;
  }
  if (n==1) {printf("%d\n",a[1].x*a[1].y);return 0;}
  /*sort(a+1,a+n+1);
  for (i=1;i<=n;i++) sum+=a[i].y;

    sum-=,ans=min(ans,sum*a[i-1].x);*/
  ans=2100000000;
  for (H=1;H<=1000;H++)
  {
    flag=1;cnt=0;W=0;q=0;
    for (i=1;i<=n;i++)
    {
      if (a[i].y>H)
      {
        if (a[i].x>H) flag=0;
        else cnt++,W+=a[i].y;
        continue;
      }
      else if (a[i].x>H) W+=a[i].x;
      else W+=a[i].x,b[++q]=a[i].y-a[i].x;
    }
    if (cnt>n/2||!flag) continue;
    cnt=n/2-cnt;
    sort(b+1,b+q+1);
    for (i=1;i<=cnt&&b[i]<0;i++) W+=b[i];
    ans=min(ans,W*H);
  }
  printf("%d\n",ans);
}
