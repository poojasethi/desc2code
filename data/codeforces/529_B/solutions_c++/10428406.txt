#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int n,w[1001],h[1001],ans,q[1001];
bool cmp(int x,int y){
    return w[x]-h[x]>w[y]-h[y];
}
void work(int H,int k){
	int sum=0,tot=0;
	for(int i=1;i<=n;++i){
		if(w[i]<=H && h[i]<=H){
		    if(w[i]<=h[i])sum+=w[i];
			else q[++tot]=i;
		}
		else if(h[i]<=H)sum+=w[i];
		else if(w[i]<=H)sum+=h[i],--k;
		else return;
	}
	if(k<0)return;
	sort(q+1,q+tot+1,cmp);
	for(int i=1;i<=tot;++i)
	    sum+=(i<=k ? h[q[i]]:w[q[i]]);
	ans=min(ans,sum*H);
}
int main(){
	scanf("%d",&n);
	int H=0;
	for(int i=1;i<=n;++i){
		scanf("%d%d",&w[i],&h[i]);
		ans+=w[i],H=max(H,h[i]);
	}
	ans*=H;
	for(int i=1;i<=1000;++i)work(i,n/2);
	printf("%d",ans);
}
