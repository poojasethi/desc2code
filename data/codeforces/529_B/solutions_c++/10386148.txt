struct ee{
	int w,h;
	bool operator<(const ee &b)const{return w-h>b.w-b.h;}
};
ee a[1100];
int m,n,i,j,k,l=0,t,s=0,ans=1999999999,h[2100];

int main() {
	cin>>n;
	for (i=0;i<n;i++) {
		scanf("%d%d",&a[i].w,&a[i].h);
		s+=a[i].w;
		h[i*2]=a[i].w;
		h[i*2+1]=a[i].h;
		l=max(l,min(a[i].w,a[i].h));
	}
	sort(a,a+n);
	t=s;
	for (i=0;i<2*n;i++)
	if (h[i]>=l) {
		s=t;
		k=n/2;
		for (j=0;j<n;j++)
		if (a[j].h>h[i]) {
			k--;
			s-=a[j].w-a[j].h;
		}
		for (j=0;(j<n && k>0);j++)
		if (a[j].w>a[j].h && a[j].w<=h[i]) {
			k--;
			s-=a[j].w-a[j].h;
		}
		if (k>=0)
		ans=min(ans,s*h[i]);
	}
	cout<<ans<<endl;
    return 0;
}
