#include <bits/stdc++.h>
using namespace std;
typedef long long int ll;
int H[1005],W[1005];
vector<ll> vec;
int main(){
 int n;
 cin >> n;
 for(int i=0;i<n;i++) cin >> W[i] >> H[i];
 ll maxi = LONG_LONG_MAX;
 for(int h=1;h<=1000;h++){
	ll sum = 0;
        vec.clear();
  	for(int i=0;i<n;i++){
		ll a = H[i]<=h ? W[i] : INT_MAX;
		ll b = W[i]<=h ? H[i] : INT_MAX;
		sum += a;
		vec.push_back(b-a);
	}
	sort(vec.begin(),vec.end());
	for(int i=0;i<n/2;i++){
		if(vec[i]>=0) break;
		sum += vec[i];
	}
	maxi = min(maxi,h*sum);
 }
 cout << maxi << endl;
 return 0;
}
