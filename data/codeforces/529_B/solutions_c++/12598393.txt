#include<cmath>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define N 10000
#define INF 0x3f3f3f3f
#define mem(a,b) memset(a,b,sizeof(a))
using namespace std;
int w[N];
int h[N];
int sta[N];
int main()
{
	int n,maxh=-INF;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
	{
		scanf("%d%d",&w[i],&h[i]);
		maxh=max(maxh,w[i]);
		maxh=max(maxh,h[i]);
	}
	int ans=INF;
	for(int high=1;high<=maxh;++high)
	{
		int lie=0;
		int top=0;
		int flag=1;
		int width=0;
		for(int i=1; i<=n && flag ; ++i)
		{
			if(h[i]<=high)
			{
				if(w[i]<=high)sta[++top]=h[i]-w[i];
				width+=w[i];
			}
			else
			{
				if(w[i]<=high)
				{
				    width+=h[i];
				    ++lie;
				    flag=lie<=n/2;
				}
				else flag=0;
			}
		}
		if(lie<n/2)
		{
			sort(sta+1,sta+top+1);
			for(int i=1; lie<n/2 && sta[i]<0 && i<=top ; width+=sta[i] , ++i , ++lie);
		}
		if(flag) ans=min(ans,high*width);
	}
	printf("%d",ans);
}
