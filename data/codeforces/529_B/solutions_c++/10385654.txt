#include<bits/stdc++.h>
using namespace std;
long long int n,x[1243],y[1234],ans=1999999999;
void solve(long long int cur)
{
    long long int i,cnt=0,sum=0;
    priority_queue <long long int> qu;
    for(i=1;i<=n;i++)
    {
        if(x[i]>cur && y[i]>cur) return;
        if(x[i]>cur) {cnt++; sum+=x[i]; if(cnt>n/2) return; continue;}
        if(y[i]>x[i] && y[i]<=cur) qu.push(y[i]-x[i]);
        sum+=y[i];
    }
     //printf("%lld %lld %lld\n",cnt,cur,sum);
    while(!qu.empty() && cnt+1<=n/2)
    {
        sum-=qu.top();
        qu.pop(); cnt++;
    }
   // printf("%lld %lld %lld\n",cnt,cur,sum);
    ans=min(ans,cur*sum);
}
int main()
{
    long long int i;
    //freopen("test.inp","r",stdin);
    scanf("%lld",&n);
    for(i=1;i<=n;i++)
    {
        scanf("%lld %lld",&y[i],&x[i]);
    }
    for(i=1;i<=1010;i++)
    {
        solve(i);
        //printf("%lld %lld\n",i,ans);
    }
    cout<<ans;
}
