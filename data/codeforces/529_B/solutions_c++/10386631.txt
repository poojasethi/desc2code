#include<cstdio>
#include<algorithm>
using namespace std;
const int maxn=1050;
int n,w[maxn],h[maxn],ans=2e9;
int h0[maxn*2],hs=0,w0[maxn],ws=0,ww;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&w[i],&h[i]);
		h0[hs++]=w[i],h0[hs++]=h[i];
	}
	sort(h0,h0+hs);
	for(i=0;i<hs;i++)
	{
		int t=n/2;
		ww=ws=0;
		for(j=1;j<=n&&t>=0;j++)
			if(w[j]>h0[i]&&h[j]>h0[i])
				t=-1;
			else if(w[j]>h0[i])
				ww+=w[j];
			else if(h[j]>h0[i])
				ww+=h[j],t--;
			else ww+=w[j],w0[ws++]=h[j]-w[j];
		if(t<0) continue;
		sort(w0,w0+ws);
		for(j=0;j<t&&w0[j]<0;j++)
			ww+=w0[j];
		//printf("%d %d  %d\n",h0[i],ww,t);
		ans=min(ans,h0[i]*ww);
	}
	printf("%d\n",ans);
	return 0;
}
