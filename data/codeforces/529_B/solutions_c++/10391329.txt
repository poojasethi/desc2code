#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <algorithm>
#include <vector>
#define inf (1<<30)
using namespace std;

struct data{int w,h,d,f;}a[1111];
vector<int> b;

int cmp1(data a,data b){return a.h>b.h;}

int cmp2(data a,data b){return a.d<b.d;}

int main(){
	int n;
	cin>>n;
	for (int i=0;i<n;i++){
		scanf("%d%d",&a[i].w,&a[i].h);
		a[i].d=a[i].h-a[i].w;
		b.push_back(a[i].h);
		b.push_back(a[i].w);
		a[i].f=-1;
	}
	sort(b.begin(),b.end());
	int ans=inf;
	sort(a,a+n,cmp2);
	for (int i=0;i<b.size();i++){
		int sum=0,cnt=0,ok=1;
		for (int j=0;j<n;j++) if (a[j].h>b[i]){
			sum+=a[j].h;
			if (a[j].w>b[i]) ok=0;
			a[j].f=i;
			cnt++;
		}
		if (!ok || cnt>n/2) continue;
		for (int j=0;j<n;j++) if (a[j].f!=i){
			if (a[j].w>b[i] || cnt+1>n/2 || a[j].w<a[j].h) sum+=a[j].w;
			else{
				sum+=a[j].h;
				cnt++;
			}
		}
		ans=min(ans,b[i]*sum);
	}
	cout<<ans<<endl;
}
