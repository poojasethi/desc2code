#include <cstdio>
#include <algorithm>
#include <queue>
#define N 1005
#define fi(a, b, c) for(int a = (b); a < (c); a++)
#define fd(a, b, c) for(int a = (b); a > (c); a--)
#define FI(a, b, c) for(int a = (b); a <= (c); a++)
#define FD(a, b, c) for(int a = (b); a >= (c); a--)
#define fe(a, b, c) for(int a = (b); a; a = c[a])
using namespace std;

int n, w[N], h[N], ans;
priority_queue<int> q;

int main(){
	scanf("%d", &n);
	fi(i, 0, n) scanf("%d %d", &w[i], &h[i]);
	
	ans = 1 << 30;
	FI(m, 1, 1000){
		int t = n / 2, s = 0;
		bool ok = 1;
		while(q.size()) q.pop();
		fi(i, 0, n){
			if(h[i] > m){
				if(w[i] > m){
					ok = 0;
					break;
				}
				t--;
				s += h[i];
			}else{
				if(w[i] <= m) q.push(w[i] - h[i]);
				s += w[i];
			}
		}
		
		if(!ok || t < 0) continue;
		
		while(t && q.size() && q.top() > 0){
			t--;
			s -= q.top();
			q.pop();
		}
		
		ans = min(ans, s * m);
	}
	
	printf("%d\n", ans);
	scanf("\n");
}
