#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
using namespace std;
const int N=1005,INF=1500000005;
struct sl{
	int h,w;
}peo[N];
int n,m,z,x,y,ans;
bool c[N];
inline bool cmp(sl x,sl y){
	return x.w-x.h>y.w-y.h;
}
inline int calc(int l){
	memset(c,0,sizeof(c));
	int x=0,ans=0;
	for (int i=1;i<=n;++i){
		if (peo[i].h>l && peo[i].w>l) return INF;
		if (peo[i].h>l && peo[i].w<=l){
			++x;ans+=peo[i].h;c[i]=true;
		}
		if (peo[i].h<=l && peo[i].w>l){
			ans+=peo[i].w;c[i]=true;
		}
	}
	if (x>n/2) return INF;
	for (int i=1;i<=n;++i)
	if (c[i]==false){
		if (peo[i].h<peo[i].w && x<n/2) ans+=peo[i].h,++x;
		else ans+=peo[i].w;
	}
	return ans*l;
}
int main(){
	//freopen("cf529B.in","r",stdin);freopen("cf529B.out","w",stdout);
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%d%d",&peo[i].w,&peo[i].h);
	sort(peo+1,peo+n+1,cmp);
	ans=INF;
	for (int i=1;i<=1000;++i) 
	ans=min(ans,calc(i));
	printf("%d",ans);
}
