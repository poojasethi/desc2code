#include<cstdio>
#include<algorithm>
using namespace std;
#define inf 99999999999999999ll
typedef long long LL;
const LL NN=11011;
LL n,b[NN];
struct ppt
{
	LL x,y;
} q[NN];
bool cmp(int x,int y) {return x>y;}
int main()
{
	//freopen("1.in","r",stdin);freopen("1.out","w",stdout);
	scanf("%lld",&n);
	LL i,j;
	for (i=1;i<=n;i++)
	{
		scanf("%lld%lld",&q[i].x,&q[i].y);
		//if (q[i].x<q[i].y) swap(q[i].x,q[i].y);
	}
	LL ans=inf;
	for (i=1500;i>=0;i--)
	{
		bool flag=true;
		LL t=0,num=0,m=0;
		for (j=1;j<=n;j++)
		{
			if (q[j].x>i&&q[j].y>i) flag=false;
			else if (q[j].x<=i&&q[j].y>i) num++,t+=q[j].y;
			else if (q[j].x>i&&q[j].y<=i) t+=q[j].x;
			else if (q[j].x<=q[j].y) t+=q[j].x;
			else t+=q[j].x,b[++m]=q[j].x-q[j].y;
		}
		if (num+num>n) continue;
		if (!flag) continue;
		sort(b+1,b+m+1,cmp);
		for (LL k=1;k+num<=n/2&&k<=m;k++) t-=b[k];
		ans=min(ans,i*t);
	}
	printf("%lld\n",ans);
	return 0;
}
