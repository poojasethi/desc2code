#include <iostream>
#include <queue>
using namespace std;
int X=1000000005,H[1005],W[1005],N;
int g(int h, int k, int w) {
    priority_queue<int> q;
    for(int i=1;i<=N;i++){
        if (H[i]<=h && W[i]<=h){
             q.push(W[i]-H[i]);
             w+=W[i];
        }
        else if (W[i]<=h && k){
             w+=H[i];
             k--;
        }
        else if (H[i]<=h)
             w+=W[i];
        else return 1000000005;
    }
    while (!q.empty() && k && q.top()>0) {
        w-=q.top(); 
        q.pop();
        k--;
    }
    return w*h;
}

int main()
{
    cin >> N;
    for(int i=1;i<=N;i++)cin>>W[i]>>H[i];
    for(int h=1;h<=1000;h++) X=min(X,g(h,N/2,0));
    cout<<X<<endl;
    return 0;
}
