#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<set>
using namespace std; 

int n,h[1003],w[1003]; 
multiset<int> se; 
int main(){ 
	scanf("%d",&n); 
	for(int i=1;i<=n;i++) 
		scanf("%d%d",&w[i],&h[i]); 
	int ans=1<<30; 
	int sum=0; 
	for (int i=1;i<=n;i++) sum+=w[i]; 
	for ( int i=1;i<=1000;i++){ 
		int k=0;
		int len=0; 
		int kt=0; 
		se.clear();
		for( int j=1;j<=n;j++){ 
			if (min(h[j],w[j])>i){ 
				k=-1;
				break; 
			}
			if ( h[j]>i) { 
				k++; 
				len+=h[j]; 
			}
			else { 
				if ( w[j]>i) len+=w[j];
				else if ( w[j]>h[j]) { 
					len+=w[j]; 
					kt++; 
					se.insert(h[j]-w[j]); 
				}else len+=w[j]; 
			}
		}
		if ( k==-1) continue; 
		if ( k>n/2 ) continue; 
		int le=min(kt,n/2-k); 
		for ( set<int> ::iterator p=se.begin();le;le--,p++)   len+=(*p);
		ans=min(ans,i*len); 
	}
	printf("%d\n",ans); 
	return 0; 
}
