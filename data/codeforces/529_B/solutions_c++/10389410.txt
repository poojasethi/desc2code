#include <iostream>
#include <queue>
#include <climits>

using namespace std;

#define NMAX 1007

int hfix,rotate,cw,flag,i,j,d,N,ans;
int h[NMAX],w[NMAX];
priority_queue < int > pq;

int main()
{

cin >> N;

for (i=1;i<=N;++i)
cin >> w[i] >> h[i];

ans = INT_MAX;

for (hfix=1;hfix<=1000;++hfix)
{
    rotate = N/2 ;
    flag = true;
    cw = 0;
    while (pq.size()) pq.pop();

    for (j=1;j<=N;++j)
    {
        if (i==j) continue;

        if (h[j]>hfix)
        {
            if (w[j]>hfix || !rotate) flag = false;
            cw += h[j];

            --rotate;

            continue;
        }

        if (w[j]>hfix || h[j]>=w[j])
        {
            cw += w[j];
            continue;
        }

        pq.push(w[j]-h[j]);
        cw += w[j];
    }

    while (rotate && pq.size())
    {
        d = pq.top();
        pq.pop();

        cw -= d;
        --rotate;
    }

    if (flag) ans = min(ans,hfix*cw);
}

cout<<ans<<'\n';

return 0;
}
