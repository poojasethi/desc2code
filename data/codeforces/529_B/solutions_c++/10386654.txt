#include<iostream>
#include<map>
#include<set>
#include<vector>
#include<queue>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<cmath>
#define ll long long
#define inf 2100000000
using namespace std;
int n;
vector<int>a;
int w[1010],h[1010];
int main(){
    scanf("%d",&n);
    int re=0,ans=inf;
    for(int i=0;i<n;i++) {
        scanf("%d%d",&w[i],&h[i]);
    }
    for(int hh=1;hh<=1000;hh++){
        int flag=1;
        int cnt=0,ww=0;
        a.clear();
        for(int i=0;i<n;i++) {
            if(w[i]>hh&&h[i]>hh) {flag=0;break;}
            if(h[i]>hh) {cnt++;ww+=h[i];continue;}
            if(w[i]<=hh&&h[i]<=hh) a.push_back(h[i]-w[i]);
            ww+=w[i];
        }
        sort(a.begin(),a.end());
        if(flag==0) continue;
        if(cnt>n/2) continue;
        int i=0;
        while(cnt<n/2&&i<a.size()&&a[i]<0){
            ww+=a[i];i++;cnt++;      
        }
        ans=min(ans,hh*ww);
    }
    printf("%d\n",ans);
}
