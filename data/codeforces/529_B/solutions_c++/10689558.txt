#include <algorithm>
#include <cstring>

#define fi first
#define se second
#define PA pair<int,int>
#define VI vector<int>
#define VP vector<PA >
#define mk(x,y) make_pair(x,y)
#define N 2010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,an;
int x[N],y[N],xx[N],yy[N],a[N];
inline void work(int h) {
	int i,s=0,t=0,len=0;
	For(i,1,n) xx[i]=x[i],yy[i]=y[i];
	For(i,1,n) {
		if (xx[i]>h&&yy[i]>h) return;
		if (yy[i]>h) swap(xx[i],yy[i]),s++;
	}
	if (s*2>n) return;
	For(i,1,n) if (xx[i]<=h&&yy[i]<=h&&xx[i]>yy[i]) a[++t]=xx[i]-yy[i];
	For(i,1,n) len+=xx[i];
	sort(a+1,a+t+1);
	reverse(a+1,a+t+1);
	For(i,1,min(t,n/2-s)) len-=a[i];
	an=min(an,len*h);
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d%d",&x[i],&y[i]);
	an=(int)2e9;
	For(i,1,1000) work(i);
	printf("%d\n",an);
	return 0;
}
