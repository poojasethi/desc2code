#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e6+10;
int n,a[maxn],id[maxn],bit[maxn];
long long ans;
bool _less(int x,int y){
    return a[x]>a[y];
}
int get(int x){
    int ans=0;
    while (x){
        ans+=bit[x];
        x-=x&-x;
    }
    return ans;
}
void update(int x){
    while (x<=n){
        bit[x]++;
        x+=x&-x;
    }
}
int main(){
    scanf("%d",&n);
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    for (int i=1; i<=n; i++) id[i]=i;
    sort(id+1,id+n+1,_less);
    for (int t,j,i=1; i<=n; i++){
        j=get(id[i]-1);
        t=id[i]-j-1;
        t=(n-i)-t;
        ans+=1LL*j*t;
        update(id[i]);
    }
    cout<<ans;
}
