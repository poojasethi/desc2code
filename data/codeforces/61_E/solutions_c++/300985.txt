#include <cstdio>
#include <algorithm>
#define MAXN 1000000

using namespace std;

int f[MAXN], n, a[MAXN], b[MAXN];

int sum (int i) {
    int res = 0;
    for (;i >= 0; i = (i & (i + 1)) - 1)
        res += f[i];
    return res;
}

int update (int i, int x) {
    for (; i < n; i |= i + 1)
        f[i] += x;
}

int main () {
    scanf ("%d", &n);
    for (int i = 0; i < n; ++i) {
        scanf ("%d", a + i);
        b[i] = a[i];
    }
    sort (b, b + n);
    for (int i = 0; i < n; ++i) 
        a[i] = lower_bound (b, b + n, a[i]) - b;
    long long res = 0;
    for (int i = 0; i < n; ++i) {
        int gr = sum (n - 1) - sum (a[i]);
        int le = a[i] - (i - gr);
        res += 1ll * gr * le;   
        update (a[i], 1);
    }
    printf ("%I64d", res);
    return 0;
}
