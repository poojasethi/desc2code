#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1000006;

int n, a[MAXN], x[MAXN], t[MAXN];
long long res;

void up(int i)
{
    while(i <= n)
    {
        t[i]++;
        i += i & (-i);
    }
}

int get(int i)
{
    int res = 0;
    while( i > 0)
    {
        res += t[i];
        i -= i & (-i);
    }
    return res;
}

int main()
{
    ios_base::sync_with_stdio(0);

    cin >> n;
    for(int i=1; i<=n; i++) cin >> a[i], x[i] = a[i];

    sort( x + 1, x + n + 1);

    res = 0;
    int f1, f2;
    for(int i=n; i>=1; i--)
    {
        int pos = lower_bound( x + 1, x + n + 1, a[i]) - x;
        f1 = get(pos);
        up(pos);
        f2 = (n - pos) - (n - i - f1);
        res += 1ll*f1 * f2;
    }
    cout << res;

    return 0;
}
