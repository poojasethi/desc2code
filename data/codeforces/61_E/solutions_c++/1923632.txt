#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

const int N=1000000;
int tem[N+10],num[N+10],arry[N+10];

bool cmp(int a,int b)
{
	return arry[a]>arry[b];
}
int n;

void add(int k)
{
    while(k<=n)
    {

        num[k]++;
        k+=(k&-k);
    }
}

long long sum(int k)
{
    long long ret=0;
    while(k)
    {
        ret+=num[k];
        k-=(k&-k);
    }
    return ret;
}

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		tem[i]=i;
		scanf("%d",arry+i);
	}
	sort(tem+1,tem+n+1,cmp);
	long long ans=0,k;
	for(int i=1;i<=n;i++)
	{
	    k=sum(tem[i]);
	    //printf("%I64d\n",k);
	    ans+=k*(n-i-tem[i]+k+1);
	    add(tem[i]);
	}
	printf("%I64d\n",ans);
	return 0;
}
