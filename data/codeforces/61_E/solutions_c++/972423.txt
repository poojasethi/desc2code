#include<iostream>
#include<algorithm>
using namespace std;
#define z 1000100
#define g for(i=0;i<n;i++)
#define u(s) for(j=s;j<z;j+=j&-j)
#define q(s) c=0;for(j=s;j;j-=j&-j)
int x[z],y[z],a[z],s[z];
int main(){
    int n,i,j,c;
    long long r,f=0;
    cin>>n;
    g {cin>>a[i];s[i]=a[i];}
    sort(s,s+n);
    g a[i]=lower_bound(s,s+n,a[i])-s+1;
    g u(a[i])y[j]++;
    g {
        u(a[i])y[j]--;
        q(a[i])c+=y[j];
        r=c;
        q(z-a[i])c+=x[j];
        r*=c;
        f+=r;
        u(z-a[i])x[j]++;
    }
    cout<<f;
}
