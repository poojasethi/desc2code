#include <iostream>
#include <algorithm>
using namespace std;
const int maxN = 1e6 + 9;
long long fen1[maxN];
long long fen2[maxN];
pair<int,int> arr[maxN];
long long ans;
long long get(int id,long long *fen){
	id++;
	long long ret = 0;
	while(id){
		ret += fen[id];
		id -= id & -id;
	}
	return ret;
}
void add(int id,long long val,long long *fen){
	id++;
	while(id < maxN){
		fen[id] += val;
		id += id & -id;
	}
	return;
}
int main(){
	int n;
	cin >> n;
	for(int i = 0; i < n;i++){
		cin >> arr[i].first;
		arr[i].second = i;
	}
	sort(arr,arr+n);
	reverse(arr,arr+n);
	for(int i = 0; i < n;i++){
		int id = arr[i].second;
		long long one = get(id,fen1);
		ans += get(id,fen2);
		add(id,1,fen1);
		add(id,one,fen2);
	}
	cout << ans << endl;
	return 0;
}
