#include <iostream>
#include <stdio.h>
#include <algorithm>
using namespace std;
#define N 1000500
//courtesy -- vlad07
typedef long long ll;
#define P pair<ll, ll>
int main(){
   ll n, res=0, cl;
   ll o[N]={0}, t[N]={0};
   P a[N];
   scanf("%I64d",&n);
   for(int i=0; i<n; i++){
      scanf("%I64d",&a[i].first);
      a[i].second=i+1;
   }
   sort(a,a+n);
   for(int i=n-1; i>=0; i--){
      cl=0;
      for(int j=a[i].second-1; j>0; j&=(j-1)){
         cl+=o[j];
         res+=t[j];
      }
      for(int j=a[i].second; j<N; j=(j|(j-1))+1){
         o[j]++;
         t[j]+=cl;
      }
   }
   cout<<res<<endl;
   return 0;
}
