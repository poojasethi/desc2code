#include<cstdio>
#include<algorithm>
using namespace std;
int n,tree[1000010],a[1000010],b[1000010];
bool cmp(int a1,int a2)
{
    return a[a1]<a[a2];
}
void update(int pos,int add)
{
    while(pos<=n)
    {
        tree[pos]+=add;
        pos+=pos&(-pos);
    }
}
int query(int pos)
{
    int ans=0;
    while(pos)
    {
        ans+=tree[pos];
        pos-=pos&(-pos);
    }
    return ans;
}
long long ans;
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    scanf("%d",&a[i]);
    for(int i=1;i<=n;i++)b[i]=i;
    sort(b+1,b+1+n,cmp);
    for(int i=1;i<=n;i++)a[b[i]]=i;
    for(int pos=2,temp;pos<n;pos++)
    {
        update(a[pos-1],1);
        temp=query(a[pos]);
        ans+=(long long)(pos-1-temp)*(a[pos]-1-temp);
    }
    printf("%I64d",ans);
}
