#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 1000010

using namespace std;

typedef long long LL;

int x[maxn],xn;
int a[maxn];
LL lt[maxn],rt[maxn];
int t[maxn];
int n;
void ins(int p){
    for (;p<=xn;p+=(p&(p^(p-1)))) t[p]++;
}
int ask(int p){
    int ret=0;
    for (;p>0;p-=(p&(p^(p-1)))) ret+=t[p];
    return ret;
}
int main(){
    scanf("%d",&n);
    for (int i=0;i<n;i++){
        scanf("%d",a+i);
        x[i]=a[i];
    }
    sort(x,x+n);
    xn=unique(x,x+n)-x;
    for (int i=0;i<n;i++){
        a[i]=upper_bound(x,x+xn,a[i])-x;
        lt[i]=i-ask(a[i]-1);
        ins(a[i]);
    }
    memset(t,0,sizeof(t));
    for (int i=n-1;i>=0;i--){
        rt[i]=ask(a[i]);
        ins(a[i]);
    }
    LL ans=0;
    for (int i=0;i<n;i++) ans+=lt[i]*rt[i];
    printf("%I64d\n",ans);
    return 0;
}
