#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int a[1001000];
int r[1001000];
long long ans;
int n, sumi;
int c[1001000];

bool cmp(int i, int j){
    return a[i]<a[j];
}

int lowbit(int x){
    return x & (-x);
}

int sum(int pos){
    int s = 0;
    while (pos){
        s += c[pos];
        pos-=lowbit(pos);
    }
    return s;
}

void update(int pos){
    while (pos<=n){
        c[pos]++;
        pos += lowbit(pos);
    }
}

int main(){
    scanf("%d", &n);
    for (int i = 1; i <= n; i++){
        scanf("%d", &a[i]);
        r[i] = i;
    }
    sort(r+1, r+n+1, cmp);
    memset(c, 0, sizeof c);
    ans = 0;
    for (int i = 1; i <= n; i++){
        update(r[i]);
        sumi = sum(r[i]);
        ans += (long long)(r[i] - sumi) * (long long)(sum(n) - sumi);
    }
    cout<<ans<<endl;
    return 0;
}
