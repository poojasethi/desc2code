#include <algorithm>
#include <cstdio>
using namespace std;
int a[1111111],b[1111111],c[1111111];

int main(){
	int n,p,q,i,x;
	scanf("%d",&n);
	for(i=0;i<n;i++)	scanf("%d",a+i);
	copy(a,a+n,b);
	sort(b,b+n);
	for(i=0;i<n;i++)	a[i]=upper_bound(b,b+n,a[i])-b;
	fill(b,b+n,0);
	for(i=1;i<=n;i++)	for(x=i;x<=n;x+=x&-x)	b[x]++;
	long long ans=0;
	for(i=0;i<n;i++){
		for(x=a[i];x<=n;x+=x&-x)	b[x]--;
		p=q=0;
		for(x=a[i];x>0;x-=x&-x)	p+=b[x];
		for(x=a[i];x>0;x-=x&-x)	q+=c[x];
		ans+=p*(long long)(i-q);
		for(x=a[i];x<=n;x+=x&-x)	c[x]++;
	}
	printf("%I64d\n",ans);
	return 0;
}
