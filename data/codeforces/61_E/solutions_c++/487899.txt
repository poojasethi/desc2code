#include<cstdio>
#include<algorithm>
using namespace std;

const int maxn=1000000+10;
pair<int,int> tmp[maxn];
int A[maxn];
long long P[maxn],Q[maxn];
long long t[maxn],res;
int n;

int main()
{
    scanf("%d",&n);
    for (int i=1;i<=n;i++)
        scanf("%d",&tmp[i].first),tmp[i].second=i;
    sort(tmp+1,tmp+1+n);
    for (int i=1;i<=n;i++) A[tmp[i].second]=i;
    for (int i=n;i;i--)
    {
        for (int j=A[i]-1;j;j-=j&-j) P[i]+=t[j];
        for (int j=A[i];j<=n;j+=j&-j) t[j]++;
    }
    for (int i=0;i<=n;i++) t[i]=0;
    for (int i=n;i;i--)
    {
        for (int j=A[i]-1;j;j-=j&-j) Q[i]+=t[j];
        for (int j=A[i];j<=n;j+=j&-j) t[j]+=P[i];
        res+=Q[i];
    }
    printf("%I64d\n",res);
}
