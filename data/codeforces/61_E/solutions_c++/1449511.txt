#include <iostream>
#include <algorithm>

#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl

using namespace std;

const int inf=1e9;
long long a[1000010],b[1000010];
int n,m,k[1000010],t[1000010];
long long res;

bool cmp(int a,int b){return t[a]<t[b];}

int main(){
  
  cin>>n;
  rep(i,n)cin>>t[i],k[i]=i;
  sort(k,k+n,cmp);rep(i,n)k[i]++;
  reverse(k,k+n);

  rep(i,n){
    int c,s=0;
    c=k[i]-1;while(c>0){s+=a[c];c-=c&(-c);}
    c=k[i];while(c<=n){a[c]++;c+=c&(-c);}
    c=k[i]-1;while(c>0){res+=b[c];c-=c&(-c);}
    c=k[i];while(c<=n){b[c]+=s;c+=c&(-c);}
  }

  pr(res);

  return 0;

}
