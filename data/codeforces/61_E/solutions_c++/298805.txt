#include <cstdio>
#include <iostream>
#include <map>
#include <cstring>
#include <algorithm>
using namespace std;

map<int, int> tab;
int A[1010000], B[1010000], num[1010000], sum[1010000];
typedef long long ll;
int N;

int ask(int k) {
	int ret = 0;
	while (k > 0) {
		ret += sum[k];
		k -= (k^(k-1))&k;
	}
	return ret;
}

int update(int k) {
	while (k <= N) {
		++sum[k];
		k += (k^(k-1))&k;
	}
}

int main() {
	scanf("%d", &N);
	
	for (int i=0; i<N; ++i) scanf("%d", &num[i]), sum[i] = num[i];
	sort(sum, sum+N);
	for (int i=0; i<N; ++i) tab[sum[i]] = i + 1;

	memset(sum, 0, sizeof(sum));
	for (int i=0; i<N; ++i) {
		int k = tab[num[i]];
		A[i] = i - ask(k);
		update(k);
	}
	
	memset(sum, 0, sizeof(sum));
	for (int i=N-1; i>=0; --i) {
		int k = tab[num[i]];
		B[i] = ask(k);
		update(k);
	}
	
	ll ret = 0;
	for (int i=0; i<N; ++i) ret += ll(A[i]) * B[i];
	cout << ret << endl;
	return 0;
}
