const int N = 1000000 + 5;
int a[N],b[N],t[N];
int n;
ll ans;
void add(int x, int v) {
	for(;x<N;x+=x&-x) t[x]+=v;
}
int query(int x) {
	int ret = 0;
	for(;x>0;x-=x&-x) ret+=t[x];
	return ret;
}
int main(){

		int num_large_prev = (n-a[i]) - (n-i-1-num_small_succ);
		ans += (ll) num_small_succ * num_large_prev;
