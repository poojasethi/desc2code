#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;

int n, N, arr[1<<20|1], compressed[1<<20|1];
pair<int,int> sor[1<<20|1];
long long bit[2][1<<20|1];

void addValue(int bitInd, int i, long long val){
	for(; i<=N; i+=(i&-i)) bit[bitInd][i]+=val;
}

long long getSum(int bitInd, int i){
	long long sum=0;
	for(; i; i-=(i&-i)) sum+=bit[bitInd][i];
	return sum;
}

int main(){
	scanf("%d", &n);
	for(int i=1; i<=n; i++){
		scanf("%d", &arr[i]);
		sor[i] = make_pair(arr[i], i);
	}
	sort(sor+1, sor+1+n);
	for(int i=1; i<=n; i++) compressed[sor[i].second]=i;
	for(N=1; N<n; N<<=1);
	long long ans=0;
	for(int i=n; i>=1; i--){
		ans += getSum(1, compressed[i]);
		addValue(1, compressed[i], getSum(0, compressed[i]));
		addValue(0, compressed[i], 1);
	}
	printf("%lld\n", ans);
}
