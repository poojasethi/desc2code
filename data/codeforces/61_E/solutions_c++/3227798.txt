#include <iostream>
#include <algorithm>
using namespace std;
typedef long long LL;
LL n,a[1<<20],b[1<<20],c[1<<20],w;
int main(void){
  cin>>n;
  for(LL i=0;i<n&&cin>>a[i];++i)b[i]=a[i];
  sort(b,b+n);
  for(LL i=0;i<n;++i){
    LL u=lower_bound(b,b+n,a[i])-b+1,t=0;
    for(LL j=u;j>0;j-=j&-j)t+=c[j];
    w+=(i-t)*(u-1-t);
    while(u<=n)++c[u],u+=u&-u;
  }
  cout<<w<<endl;
}
