#define N 1001001
typedef long long int LL;

using namespace std;
int n,t[N];
int s[N];

struct BIT{
	int b[N];
	BIT(){ memset(b,0,sizeof(b)); }
	void add(int a){ for(int i=a; i<N; i+=i&-i) b[i]++; }
	int  cnt(int a){ 
		int ans=0; 
		for(int i=a; i; i-=i&-i) ans+= b[i]; 
		return ans; 
	}
} b;

int main(){

	scanf("%d",&n);
	for(int i=0; i<n; i++)scanf("%d",&t[i]);
	
	memcpy(s,t,n*sizeof(int));	
	sort(s,s+n);
	for(int i=0; i<n; i++)t[i] = lower_bound(s,s+n,t[i]) - s + 1;

	LL ans=0;
	for(int i=0; i<n; i++){
		LL k = b.cnt(t[i]);
		LL ft = i-k, bk = t[i]-1-k;
		ans += ft*bk;
		b.add(t[i]);
	}
	cout << ans << endl;
	return 0;
}
