#include <iostream>
#include <algorithm>
#include <cstdio>
#include <map>
using namespace std ;
#define lim 1000500

int N ;
pair<int,int> val[lim] ;

long long  mayores[lim], mayores2[lim] ;

int main() {
    int i, j, b ;
    map <int,int> :: iterator it ;
    long long res = 0LL, tmp ;

    scanf("%d",&N ) ;
    for( i=0; i<N; i++ ) {
        scanf("%d",&val[i].first ) ;
        val[i].second = i+1 ;
    }
    sort( val, val+N ) ;

    for( i=0; i<N; i++ ) {
        tmp = 0LL ;
        for( b=val[i].second+1;  b<=N;   b+=( b & -b)   )  {
            res += mayores2[b] ;
            tmp += mayores[b] ;
        }
        for( b=val[i].second; b>0; b -= ( b & -b)   )  {
            mayores[b] ++ ;
            mayores2[b] += tmp ;
        }
    }

    cout << res << endl ;

    return 0 ;
}
