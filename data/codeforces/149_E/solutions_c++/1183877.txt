#include<stdio.h>
#include<string.h>
#define M 100005
using namespace std;

char s[M],str[M],ss[M],tmp[1005];

int next[1005];

void get_next(int n){
	memset(next,-1,sizeof(next));
	int i=0,j=-1;
	while(i<n){
		if(j==-1 || ss[i]==ss[j]){
			i++,j++;
			next[i]=j;
		}else j=next[j];
	}
}

int kmp(int n,int m){
	int i,j;
	get_next(m);
	i=0,j=0;
	int dp[1005];
	memset(dp,0,sizeof(dp));
	while(i<n && j<m){
		if(j==-1 || s[i]==ss[j]){
			i++,j++;
		}else j=next[j];
		if(j>0 && !dp[j]) dp[j]=i;
	}
	if(dp[m]) return 1;
	
	for(i=0;i<m;i++){
		tmp[i]=ss[m-i-1];
	}
	tmp[m]='\0';
	strcpy(ss,tmp);
	get_next(m);
	i=0,j=0;
	while(i<n && j<m){
		if(j==-1 || str[i]==ss[j]){
			i++,j++;
		}else j=next[j];
		if(j>0 && dp[m-j]){
			if(dp[m-j]-1+j<n && dp[m-j]<=n-i)return 1;
		}
	}
	return 0;
}

int main(){
	int i,j,n,len,m,cnt;
	while(scanf("%s",s)!=EOF){
		n=strlen(s);
		for(i=0;i<n;i++){
			str[i]=s[n-i-1];
		}
		str[n]='\0';
		scanf("%d",&m);
		cnt=0;
		while(m--){
			scanf("%s",ss);
			len=strlen(ss);
			if(len<2) continue;
			if(kmp(n,len)) cnt++;
		}
		printf("%d\n",cnt);
	}
	return 0;
}
