#include<cstdio>
#include<cstring>
const int N=100005;
int ans,next[1005],d1[N],d2[N];
char s1[N],s2[N];
char a1[1005],a2[1005];
void get_next(char a[])
{
	int i,k;
	i=0,k=-1;
	next[0]=-1;
	while(a[i])
	{
		if(k==-1||a[i]==a[k])
		{
			i++,k++;
			if(a[i]!=a[k])
				next[i]=k;
			else
				next[i]=next[k];
		}
		else
			k=next[k];
	}
}
int kmp(char s[],char a[],int d[])
{
	int i,k;
	i=0,k=0;
	while(s[i])
	{
		if(k==-1||s[i]==a[k])
		{
			d[i]=k+1;
			i++,k++;
			if(!a[k])
				return 1;
		}
		else
			k=next[k];
	}
	return 0;
}
int max(int a,int b){if(a>b)return a;return b;}
int main()
{
	int i,j,d,n,m;
	scanf("%s",s1);
	scanf("%d",&n);
	d=strlen(s1);
	for(i=d-1;i>=0;i--)
		s2[d-i-1]=s1[i];
	s2[d]=0;
	for(i=0;i<n;i++)
	{
		scanf("%s",a1);
		d=strlen(a1);
		if(d==1)
			continue;
		for(j=d-1;j>=0;j--)
			a2[d-j-1]=a1[j];
		a2[d]=0;
		get_next(a1);
		if(kmp(s1,a1,d1))
			ans++;
		else
		{
			get_next(a2);
			kmp(s2,a2,d2);
			m=0;
			d=strlen(s1);
			for(j=0;j<d-1;j++)
			{  
				m=max(m,d1[j]);
				if(m+d2[d-j-2]>=strlen(a1))
				{
					ans++;
					break;
				}
			}
		}
	}
	printf("%d\n",ans);
	return 0;
}
