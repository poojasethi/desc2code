#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n, x, m, l, ok, ret = 0;
int f[1005], pre[1005], suf[1005];
char s[100005], w[1005];

void precompute(){
	x = f[0] = -1;
	for (int j=1; j<=l; j++){
		while (x >= 0 && w[x] != w[j-1]) x = f[x];
		f[j] = ++x;
	}
}

int main(){
	scanf("%s", s);
	n = strlen(s);

	scanf("%d", &m);
	while (m--){
		scanf("%s", w);
		l = strlen(w);
		if (l == 1) continue;
		pre[0] = -1;
		suf[0] = (1<<29);
		for (int i=1; i<=l; i++) pre[i] = (1<<29), suf[i] = -1;
		
		precompute();
		x = 0;
		for (int j=0; j<n; j++){
			while (x >= 0 && w[x] != s[j]) x = f[x];
			++x;
			pre[x] = min(pre[x], j);
			if (x == l) x = f[x];
		}
		
		reverse(w, w + l);
		precompute();
		x = 0;
		for (int j=n-1; j>=0; j--){
			while (x >= 0 && w[x] != s[j]) x = f[x];
			++x;
			suf[x] = max(suf[x], j);
			if (x == l) x = f[x];
		}
		
		ok = 0;
		for (int i=0; i<=l; i++)
		if (pre[i] < suf[l-i]) ok = 1;
		ret += ok;
	}

	printf("%d\n", ret);
	return 0;
}
