#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> pie;

#define L first
#define R second
#define MP make_pair
#define PB push_back

string s, k;

const int maxn = 1e5 + 100;

int dp[2][maxn], next[maxn];

void kmp(int id){
	next[0] = 0;
	int match = 0;
	for(int i=1;i<k.size();i++){
		while(match != 0 and k[match] != k[i])
			match = next[match - 1];
		if(k[i] == k[match])
			match ++;
		next[i] = match;
	}
	match = 0;
	for(int i=0;i<s.size();i++){
		while(match != 0 and k[match] != s[i])
			match = next[match - 1];
		if(k[match] == s[i])
			match ++;
		dp[id][i] = match;
		if(match == k.size())
			match = next[match - 1];
	}
}

int main(){
	ios_base::sync_with_stdio(false);
	cin >> s;
	int m, ans = 0;
	cin >> m;
	for(int i=1;i<=m;i++){
		cin >> k;
		if(k.size() == 1)
			continue;
		kmp(0);
		reverse(k.begin(), k.end()), reverse(s.begin(), s.end());
		kmp(1);
		reverse(k.begin(), k.end()), reverse(s.begin(), s.end());
		int untilNow = 0;
		for(int i=1;i<s.size();i++){
			untilNow = max(untilNow, dp[0][i - 1]);
			if(untilNow + dp[1][s.size() - i - 1] >= k.size()){
				ans++;
				break;
			}
		}
	}
	cout << ans << endl;
}
