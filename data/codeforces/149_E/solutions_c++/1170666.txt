#include <cstring>
#include <cstdio>
#include <algorithm>
#define N 100005
using namespace std;

char s[N],w[1005];
int n,m,AC=0,l,y,f[1005],pre[1005],sub[1005],ok;


void FAIL(){
	int x;
	f[0] = x = -1;
	for (int i=1;i<=l;i++){
		while (x>=0 && w[x]!=w[i-1]) x=f[x];
		f[i]=++x;
	}
}

int main(){
	
	scanf("%s",s);
	n = strlen(s);
	scanf("%d",&m);
	
	for (int C=1;C<=m;C++){
		
		scanf("%s",w);
		l = strlen(w);
		if (l==1) continue;
		for (int i=0;i<=l;i++)
			pre[i] = (1<<29),
			sub[i] = -1;
		
		FAIL();
		y = 0;
		for (int i=0;i<n;i++){
			while (y>=0 && s[i]!=w[y]) y=f[y];
			++y;
			pre[y] = min(pre[y],i);
			if (y==l) y=f[y];
		}
		
		for (int i=0;i<l/2;i++) swap(w[i],w[l-i-1]);
		FAIL();
		
		y = 0;
		for (int i=n-1;i>=0;i--){
			while (y>=0 && s[i]!=w[y]) y=f[y];
			++y;
			sub[y] = max(sub[y],i);
			if (y==l) y=f[y];
		}
		
		ok = 0;
		for (int i=0;i<=l;i++)
			if (pre[i]<sub[l-i]) ok=1;
		AC+=ok;
		//printf(" OK = %d\n",ok);	
	}
	
	printf("%d\n",AC);
	
	//scanf("\n");
	return 0;
}
