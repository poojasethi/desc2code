#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<string>
#include<iostream>
#include<cmath>
#include<algorithm>
#include<string>
#include<vector>
#include<map>
#include<set>
#include<queue>
using namespace std;

typedef unsigned long long big;

const int N=102000;
char str[N],s[1200];
int n,m,f[N],g[N],next[1200];
void getnext()
{
	int i,j=0,len=strlen(s+1);
	next[1]=0;
	for(i=2;i<=len;i++)
	{
		while(j&&s[j+1]!=s[i])j=next[j];
		if(s[j+1]==s[i])j++;
		next[i]=j;
	}
}
int main()
{
	int ans=0,i,j,len;
	scanf("%s",str+1);
	n=strlen(str+1);
	scanf("%d",&m);
	while(m--)
	{
		scanf("%s",s+1);
		len=strlen(s+1);
		if(len==1)continue;
		getnext();
		for(j=0,i=1;i<=n;i++)
		{
			while(j&&s[j+1]!=str[i])j=next[j];
			if(s[j+1]==str[i])j++;
			f[i]=max(j,f[i-1]);
			if(j==len)j=next[j];
		}
		reverse(s+1,s+1+len);
		getnext();
		for(j=0,i=n;i;i--)
		{
			while(j&&s[j+1]!=str[i])j=next[j];
			if(s[j+1]==str[i])j++;
			g[i]=max(j,g[i+1]);
			if(j==len)j=next[j];
		}
		for(i=2;i<=n;i++)
			if(f[i-1]+g[i]>=len)
			{
				ans++;
				break;
			}
	}
	printf("%d\n",ans);
	return 0;
}
