#include <string>
#include <cstring>
#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

char str[1005];
int z[1005];
char st[100005];
int z2[100005][2];

void match(int idx){
  z[0] = -1; z[1] = 0; int j = 0;
  for(int i=2; str[i]; ++i) {
     while(j>=0 && str[j+1]!=str[i]) j = z[j]; j += 1;
     z[i] = j;
  }
//  for(int i=1; str[i]; ++i) printf("%d ",z[i]); printf("\n");
  z2[0][idx] = -1; j = 0;
  for(int i=1; st[i]!='\0'; ++i){
     while(j>=0 && str[j+1]!=st[i]) j = z[j];  j += 1;
     z2[i][idx] = j; //printf("%d ",z2[i][idx]);
     z2[i][idx] = max(z2[i][idx],z2[i-1][idx]);
  }
 // printf("\n");
}

int main(){
   scanf("%s",&st[1]);
   int k; scanf("%d",&k); int r = 0;
   for(int i=0; i<k; ++i){
      scanf("%s",&str[1]); match(0); 
      int l1 = 1; while(str[l1]!='\0') ++l1;
      if(l1==2) continue;
      int l2 = 1; while(st[l2]!='\0') ++l2; 
      reverse(st+1,st+l2); reverse(str+1,str+l1); match(1);
      reverse(st+1,st+l2);
      for(int i=1; i<l2; ++i){
          int a = z2[i][0]; if(l2-i-1<=0) continue;
          int b = z2[l2-i-1][1];
          if(a+b>=l1-1) {
               ++r; break;
          }
      }
   }
   cout<<r<<endl;
   return 0;
}
