#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>

using std::max;

const int fin=0,maxn=100010;
char a[maxn],b[maxn];
int p[maxn],L[maxn],R[maxn];

int main(){
  if(fin){
    freopen("t.in","r",stdin);
    freopen("t.out","w",stdout);
  }
  int n,M,m,j,i,ans=0;
  bool found;
  gets(a),n=strlen(a);
  scanf("%d\n",&M);
  while(M--){
    gets(b),m=strlen(b);
    if(m==1)continue;
    found=false;
    for(p[0]=j=-1,i=1;i<m;++i){
      while(j!=-1&&b[j+1]!=b[i])j=p[j];
      if(b[j+1]==b[i])++j;
      p[i]=j;
    }
    for(j=-1,i=0;i<n;++i){
      while(j!=-1&&b[j+1]!=a[i])j=p[j];
      if(b[j+1]==a[i])++j;
      if(j==m-1){found=true;break;}
      L[i]=j;
    }
    if(found){++ans;continue;}
    for(i=1;i<n;++i)L[i]=max(L[i],L[i-1]);
    for(p[m-1]=j=m,i=m-2;i>=0;--i){
      while(j!=m&&b[j-1]!=b[i])j=p[j];
      if(b[j-1]==b[i])--j;
      p[i]=j;
    }
    for(j=m,i=n-1;i;--i){
      while(j!=m&&b[j-1]!=a[i])j=p[j];
      if(b[j-1]==a[i])--j;
      R[i]=j;
      if(L[i-1]+1>=j){found=true;break;}
    }
    if(found)++ans;
  }
  printf("%d",ans);
  return 0;
}
