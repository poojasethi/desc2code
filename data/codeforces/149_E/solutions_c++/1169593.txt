#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[100010],str[100010],ss[100010],tmp[1010];
int nxt[1010];
void get_next(int n)
{
	memset(nxt,-1,sizeof(nxt));
	int i=0,j=-1;
	while(i<n){
		if(j==-1||ss[i]==ss[j]){
			i++,j++;
			nxt[i]=j;
		}
		else j=nxt[j];
	}
}
int kmp(int n,int m)
{
	get_next(m);
	int dp[1010];
	memset(dp,0,sizeof(dp));
	int i=0,j=0;
	while(i<n&&j<m){
		if(j==-1||s[i]==ss[j]) i++,j++;
		else j=nxt[j];
		if(j>=1&&!dp[j]) dp[j]=i;
	}
	if(dp[m]) return 1;
	for(i=0;i<m;i++)
		tmp[i]=ss[m-i-1];
	tmp[m]=0;
	strcpy(ss,tmp);
	get_next(m);
	i=j=0;
	while(i<n&&j<m){
		if(j==-1||str[i]==ss[j]) i++,j++;
		else j=nxt[j];
		if(j>=1&&dp[m-j]) {
			if(dp[m-j]-1+j<n&&dp[m-j]<=n-i) return 1;
		}
	}
	return 0;
}
int main()
{
	scanf("%s",s);
	int n=strlen(s);
	for(int i=0;i<n;i++)
		str[i]=s[n-i-1];
	str[n]=0;
	int m;
	scanf("%d",&m);
	int ans=0;
	while(m--){
		scanf("%s",ss);
		if(strlen(ss)<=1) continue;
		if(kmp(n,strlen(ss))) ans++;
	}
	cout<<ans<<endl;
	//cin>>ans;
	return 0;
}
