#include<cstdio>
#include<iostream>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
string s,rs,t;
int n,nxt[1009],m1[100009],m2[100009];
void pref(string &a,int *p){    
    p[0]=-1;
    for(int j,i=0;a[i];i++){
        for(j=p[i];j>=0&&a[i]!=a[j];j=p[j]);
        p[i+1]=j+1;
    }
}
void kmp(string &a,string &b,int *p,int *f){
    int m=0;
    for(int j=0,i=0;a[i];i++){
        for(;j>=0&&a[i]!=b[j];j=p[j]);
        m=max(m,++j);
        f[i+1]=m;
    }
}
int main(){
    while(cin>>s){
        int ans=0;
        rs=s;
        reverse(rs.begin(),rs.end());
        for(cin>>n;n;n--){
            cin>>t;                        
            pref(t,nxt);
            kmp(s,t,nxt,m1);
            reverse(t.begin(),t.end());
            pref(t,nxt);
            kmp(rs,t,nxt,m2);
            int l1=s.size(),l2=t.size();
            if(l2>1)
                for(int i=1;i<=l1;i++)
                    if(m1[i]+m2[l1-i]>=l2){
                        ans++;break;
                    }
        }
        cout<<ans<<endl;
    }
}
