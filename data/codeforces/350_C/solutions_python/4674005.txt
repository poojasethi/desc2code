#!/bin/env python
# coding=utf-8
import sys

def main():
    n = int(raw_input())
    points = [map(int, line.split()) for line in sys.stdin]
    points_for_sort = []
    step = 0
    for x in points:
        step += (x[0] and 2) + (x[1] and 2) + 2
        if x[0] > 0:
            if x[1] > 0:
                points_for_sort.append((x[0], x[1], 'RL', 'UD'))
            else:
                points_for_sort.append((x[0], -x[1], 'RL', 'DU'))
        else:
            if x[1] > 0:
                points_for_sort.append((-x[0], x[1], 'LR', 'UD'))
            else:
                points_for_sort.append((-x[0], -x[1], 'LR', 'DU'))
    points_for_sort.sort()
    print step
    for x in points_for_sort:
        if x[0]:
            sys.stdout.write('1 %s %s\n' % (x[0], x[2][0]))
        if x[1]:
            sys.stdout.write('1 %s %s\n' % (x[1], x[3][0]))
        sys.stdout.write('2\n')
        if x[0]:
            sys.stdout.write('1 %s %s\n' % (x[0], x[2][1]))
        if x[1]:
            sys.stdout.write('1 %s %s\n' % (x[1], x[3][1]))
        sys.stdout.write('3\n')
    

if __name__ == '__main__':
    main()
