#include<algorithm>
#include<iostream>
#include<cmath> 
using namespace std;
struct point{int x,y;};
bool tmp(point a,point b){return abs(a.x)+abs(a.y)<abs(b.x)+abs(b.y);}
int main()
{
	int n,sum=0;
	cin>>n;
	point a[n];
	for(int i=0;i<n;i++)cin>>a[i].x>>a[i].y;
	for(int i=0;i<n;i++)
	{
		if(a[i].x==0)sum-=2;
		if(a[i].y==0)sum-=2; 
	}
	sort(a,a+n,tmp);
	cout<<n*6+sum<<endl;
	for(int i=0;i<n;i++)
	{
		if(a[i].x>0)cout<<"1 "<<a[i].x<<" R\n";
		else if(a[i].x<0)cout<<"1 "<<-a[i].x<<" L\n";
		if(a[i].y>0)cout<<"1 "<<a[i].y<<" U\n";
		else if(a[i].y<0)cout<<"1 "<<-a[i].y<<" D\n";
		cout<<"2\n";
		if(a[i].x>0)cout<<"1 "<<a[i].x<<" L\n";
		else if(a[i].x<0)cout<<"1 "<<-a[i].x<<" R\n";
		if(a[i].y>0)cout<<"1 "<<a[i].y<<" D\n";
		else if(a[i].y<0)cout<<"1 "<<-a[i].y<<" U\n";
		cout<<"3\n";
	}
	return 0;
}
