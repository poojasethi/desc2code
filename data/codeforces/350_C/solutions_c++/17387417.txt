#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
#define LL __int64
using namespace std;
const int maxn=100010;

struct node
{
	int x,y,s;
}a[maxn];
bool cmp(node a,node b)
{
	return a.s<b.s;
}

int main()
{
	int n,f=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&a[i].x,&a[i].y);
		a[i].s = abs(a[i].x) + abs(a[i].y);
		if(a[i].x==0 || a[i].y==0) f++;
		
	}	
	sort(a,a+n,cmp);
	
	printf("%d\n",n*6-f*2);
	for(int i=0;i<n;i++)
	{
		int xx=a[i].x, yy = a[i].y;
		
		if(xx>0) printf("1 %d R\n",xx);
		else if(xx<0) printf("1 %d L\n",-xx);	
			
		if(yy>0) printf("1 %d U\n",yy)	;
		else if(yy<0) printf("1 %d D\n",-yy);	
			
		printf("2\n");
		if(yy>0) printf("1 %d D\n",yy)	;
		else if(yy<0) printf("1 %d U\n",-yy);
		if(xx>0) printf("1 %d L\n",xx);
		else if(xx<0) printf("1 %d R\n",-xx);
		
					
		printf("3\n");
	}
	
	return 0;
}
