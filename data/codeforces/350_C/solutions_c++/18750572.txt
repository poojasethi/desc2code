#include<bits/stdc++.h>
#define newl '\n'
using namespace std;
const int N=1e5+5;
long long x,y,n,i,ans;
typedef struct
{
	long long a,b;
}pp;
pp p[N]; 
bool compa(pp ta,pp tb)
{
	return  abs(ta.a)+abs(ta.b)<abs(tb.a)+abs(tb.b);
}
int main()
{
	ios::sync_with_stdio(false);cin.tie(0);
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>p[i].a>>p[i].b;
	    if(p[i].a)ans+=2;
	    if(p[i].b)ans+=2;
	    ans+=2;
	}
	cout<<ans<<newl;
	sort(p,p+n,compa);
	for(i=0;i<n;i++)
	{
		x=p[i].a;y=p[i].b;
		if(x>0)
			cout<<"1 "<<x<<" R"<<newl;
		if(x<0)
		    cout<<"1 "<<-x<<" L"<<newl;
        if(y>0)
			cout<<"1 "<<y<<" U"<<newl;
		if(y<0)
		    cout<<"1 "<<-y<<" D"<<newl;
		cout<<2<<newl;
		if(x>0)
			cout<<"1 "<<x<<" L"<<newl;
		if(x<0)
		    cout<<"1 "<<-x<<" R"<<newl;
        if(y>0)
			cout<<"1 "<<y<<" D"<<newl;
		if(y<0)
		    cout<<"1 "<<-y<<" U"<<newl;
		cout<<3<<newl;
	}
	return 0;
}
