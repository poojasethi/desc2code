#include <cstdio>
#include <algorithm>
#define abs(x) ((x)>0?(x):-(x))

using namespace std;

int n,i,j,k;

struct _n{
	int x,y;
} g[100001];

bool cmp(const _n &a, const _n &b){
	return a.y==b.y?abs(a.x)<abs(b.x):abs(a.y)<abs(b.y);
}

int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%d%d",&g[i].x,&g[i].y);
	}
	sort(g,g+n,cmp);
	for(i=0,j=0;i<n;i++){
		j+=2*(1+(g[i].x!=0)+(g[i].y!=0));
	}
	printf("%d\n",j);
	for(i=0;i<n;i++){
		if (g[i].x) printf("1 %d %c\n",abs(g[i].x),g[i].x>0?'R':'L');
		if (g[i].y) printf("1 %d %c\n",abs(g[i].y),g[i].y>0?'U':'D');
		printf("2\n");
		if (g[i].y) printf("1 %d %c\n",abs(g[i].y),g[i].y>0?'D':'U');
		if (g[i].x) printf("1 %d %c\n",abs(g[i].x),g[i].x>0?'L':'R');
		printf("3\n");
	}
	return 0;
}
