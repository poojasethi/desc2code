#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
#define abs(x) (x>=0?x:-x)
using namespace std;
const int maxn=100050;
struct P{
	int x,y;
}p[maxn];
int n;
bool cmp(const P&a,const P &b)
{
	return (abs(a.x)+abs(a.y))<(abs(b.x)+abs(b.y));	
}

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d%d",&p[i].x,&p[i].y);
	}	
	sort(p,p+n,cmp);
	int step=0;
	for(int i=0;i<n;i++)
	{
		if(abs(p[i].x)==0 || abs(p[i].y)==0) step+=4;
		else step+=6;
	}
	printf("%d\n",step);
	for(int i=0;i<n;i++){
		if(abs(p[i].x)){
			printf("1 %d %c\n",abs(p[i].x),p[i].x<0?'L':'R');
		}
		if(abs(p[i].y)){
			printf("1 %d %c\n",abs(p[i].y),p[i].y<0?'D':'U');	
		}
		printf("2\n");
		if(abs(p[i].y)){
			printf("1 %d %c\n",abs(p[i].y),p[i].y<0?'U':'D');	
		}
		if(abs(p[i].x)){
			printf("1 %d %c\n",abs(p[i].x),p[i].x<0?'R':'L');
		}
		printf("3\n");
	}
	
	return 0;
}
