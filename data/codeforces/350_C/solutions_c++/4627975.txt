#include <cstdio>
#include <algorithm>

using namespace std;

struct bomb {
	int x, y;
	bool operator <(const bomb &T) const {
		return abs(x) + abs(y) < abs(T.x) + abs(T.y);
	}
};

bomb b[111111];

int main() {
	int i, n, cnt;
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf("%d%d", &b[i].x, &b[i].y);
	}
	sort(b, b + n);
	cnt = 0;
	for (i = 0; i < n; i++) {
		cnt += 2;
		if (b[i].x) cnt += 2;
		if (b[i].y) cnt += 2;
	}
	printf("%d\n", cnt);
	for (i = 0; i < n; i++) {
		if (b[i].x) printf("1 %d %c\n", abs(b[i].x), b[i].x > 0 ? 'R' : 'L');
		if (b[i].y) printf("1 %d %c\n", abs(b[i].y), b[i].y > 0 ? 'U' : 'D');
		puts("2");
		if (b[i].x) printf("1 %d %c\n", abs(b[i].x), b[i].x < 0 ? 'R' : 'L');
		if (b[i].y) printf("1 %d %c\n", abs(b[i].y), b[i].y < 0 ? 'U' : 'D');
		puts("3");
	}
	return 0;
}
