#include <bits/stdc++.h>

using namespace std;


struct point
{
    int x,y;
} g[100001];

bool cmp(const point &a, const point &b){
	return abs(a.x)+abs(a.y)<abs(b.x)+abs(b.y);
}

int main()
{
    int n,i,j,k;
    scanf("%d",&n);
    for(i=0; i<n; i++)
    {
        scanf("%d %d",&g[i].x,&g[i].y);
    }
    sort(g,g+n,cmp);
    for(i=0,j=0; i<n; i++)
    {
        j+=2*(1+(g[i].x!=0)+(g[i].y!=0));
    }
    printf("%d\n",j);
    for(i=0; i<n; i++)
    {
        if (g[i].x) printf("1 %d %c\n",abs(g[i].x),g[i].x>0?'R':'L');
        if (g[i].y) printf("1 %d %c\n",abs(g[i].y),g[i].y>0?'U':'D');
        printf("2\n");
        if (g[i].y) printf("1 %d %c\n",abs(g[i].y),g[i].y>0?'D':'U');
        if (g[i].x) printf("1 %d %c\n",abs(g[i].x),g[i].x>0?'L':'R');
        printf("3\n");
    }
    return 0;

}
