#include <iostream>
#include <cstdlib>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;
typedef pair <int, int> pii;
vector <pair<int, pii> > bombs;
int main() {
	int n;
	cin >> n;
	int x,y;
	ll ops = 0;
	for(int i = 0; i < n; i++) {
 		cin >> x >> y;
 		int metric = abs(x) + abs(y);
 		bombs.push_back(make_pair(metric, make_pair(x,y)));
 		if(x == 0 || y == 0) ops += 4;
 		else ops += 6;
 	}
 	cout << ops << endl;
 	sort(bombs.begin(), bombs.end());
 	for(int i = 0; i < bombs.size(); i++) {
 		pair<int, pii> b = bombs[i];
 		x = b.second.first, y = b.second.second;
 		if(x) cout << "1 " << abs(x) << ((x>0)?" R":" L") << endl;
 		if(y) cout << "1 " << abs(y) << ((y>0)?" U":" D") << endl;
 		cout << "2" << endl;
  		if(x) cout << "1 " << abs(x) << ((x>0)?" L":" R") << endl;
 		if(y) cout << "1 " << abs(y) << ((y>0)?" D":" U") << endl;
 		cout << "3" << endl;
  	}
	return 0;
}
