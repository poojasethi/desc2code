#include <stdio.h>
#include <algorithm>
using namespace std;
const int N = 100005;
struct node {
	int x, y;
}b[N];
int cmp(const node idx, const node idy) {
	return abs(idx.x)+abs(idx.y) < abs(idy.x)+abs(idy.y);
}
int main() {
	int n;
	scanf("%d", &n);
	int ans = 0;
	for(int i=0; i<n; i++) {
		scanf("%d%d", &b[i].x, &b[i].y);
		if(b[i].x == 0 || b[i].y == 0) ans += 4;
		else ans += 6;
	}
	sort(b, b+n, cmp);
	printf("%d\n", ans);
	for(int i=0; i<n ;i++) {
		if (b[i].x != 0) printf("1 %d %c\n", abs(b[i].x), b[i].x > 0 ? 'R' : 'L');
		if (b[i].y != 0) printf("1 %d %c\n", abs(b[i].y), b[i].y > 0 ? 'U' : 'D');
		puts("2");
		if (b[i].x != 0) printf("1 %d %c\n", abs(b[i].x), b[i].x > 0 ? 'L' : 'R');
		if (b[i].y != 0) printf("1 %d %c\n", abs(b[i].y), b[i].y > 0 ? 'D' : 'U');
		puts("3");
	}
	return 0;
}
