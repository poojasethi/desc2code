#include<algorithm>
#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
int n,ans,k;
struct bom{ int x,y;} v[100001];
bool cmparator(const bom &a, const bom &b){
	return a.y==b.y?abs(a.x)<abs(b.x):abs(a.y)<abs(b.y);
}
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>v[i].x>>v[i].y;
	}
	sort(v,v+n,cmparator);
	for(int i=0;i<n;i++){
		ans+=2*(1+(v[i].x!=0)+(v[i].y!=0));
	}
      cout<<ans<<"\n";
	for(int i=0;i<n;i++){
		if (v[i].x) cout<<"1"<<" "<<abs(v[i].x)<<" "<<(v[i].x>0?'R':'L')<<"\n";
		if (v[i].y) cout<<"1"<<" "<<abs(v[i].y)<<" "<<(v[i].y>0?'U':'D')<<"\n";
		cout<<"2"<<"\n";
		if (v[i].y) cout<<"1"<<" "<<abs(v[i].y)<<" "<<(v[i].y>0?'D':'U')<<"\n";
		if (v[i].x) cout<<"1"<<" "<<abs(v[i].x)<<" "<<(v[i].x>0?'L':'R')<<"\n";
		cout<<"3"<<"\n";
	}
	return 0;
}
