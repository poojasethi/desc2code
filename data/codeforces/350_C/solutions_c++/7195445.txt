#include<bits/stdc++.h>
using namespace std;
#define ll long long
struct p1{
int x,y,val;
}p[100004];

bool sort1(p1 a,p1 b){
return a.val<b.val;
}

int main(){
int i,x,y,n;
cin>>n;
int ans=0;
for(int i=0;i<n;i++){
scanf("%d",&p[i].x);
scanf("%d",&p[i].y);
p[i].val=abs(p[i].x)+abs(p[i].y);
if(p[i].x!=0)
ans+=2;
if(p[i].y!=0)
ans+=2;
ans+=2;
}
sort(p,p+n,sort1);
printf("%d\n",ans);
for(i=0;i<n;i++){
if(p[i].x<0)
printf("1 %d L\n",abs(p[i].x));
else
if(p[i].x>0)
printf("1 %d R\n",abs(p[i].x));
if(p[i].y>0)
printf("1 %d U\n",abs(p[i].y));
else
if(p[i].y<0)
printf("1 %d D\n",abs(p[i].y));
printf("2\n");
if(p[i].x<0)
printf("1 %d R\n",abs(p[i].x));
else
if(p[i].x>0)
printf("1 %d L\n",abs(p[i].x));
if(p[i].y<0)
printf("1 %d U\n",abs(p[i].y));
else
if(p[i].y>0)
printf("1 %d D\n",abs(p[i].y));
printf("3\n");
}
return 0;
}
