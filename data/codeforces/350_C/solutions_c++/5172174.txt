#include <cstdio>
#include <algorithm>
using namespace std;

struct oo { int x, y; };

oo a[100030];


int cmp(oo a, oo b) 
{ 
	return abs(a.x) + abs(a.y) < abs(b.x) + abs(b.y);
}

int main()
{
	int n;
	int ans = 0;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		scanf("%d%d", &a[i].x, &a[i].y);
		ans += 2;
		if (a[i].x != 0) ans += 2;
		if (a[i].y != 0) ans += 2;
	}
	printf("%d\n", ans);
	sort(a, a + n, cmp);
	for (int i = 0; i < n; i++)
	{
		int x = a[i].x;
		int y = a[i].y;
		if (x != 0)
			printf("1 %d %c\n", abs(x), (x > 0 ? 'R' : 'L'));
		if (y != 0)
			printf("1 %d %c\n", abs(y), (y > 0 ? 'U' : 'D'));
		printf("2\n");
		if (x != 0)
			printf("1 %d %c\n", abs(x), (x > 0 ? 'L' : 'R'));
		if (y != 0)
			printf("1 %d %c\n", abs(y), (y > 0 ? 'D' : 'U'));
		printf("3\n");
	}
	return 0;
}
