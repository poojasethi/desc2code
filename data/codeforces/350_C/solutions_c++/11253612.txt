#include<bits/stdc++.h>
using namespace std;
#define ll long long
#define N 1000000007
#define MAX 100001
ll i,j,x,y,k,l,m,n;
struct node 
{
  ll x;
  ll y;
};
bool comp(node a,node b)
{
  return (abs(a.x)+abs(a.y))<(abs(b.x)+abs(b.y));
}
int main()
{
//cin>>n;
//ios_base::sync_with_stdio(false);
cin>>k;
node a[k];
ll cnt=0;
for(i=0;i<k;i++)
{
  scanf("%lld",&a[i].x);
  scanf("%lld",&a[i].y);
  if(a[i].x==0 || a[i].y==0)
  cnt++;
}
sort(a,a+k,comp);
cout<<k*6-cnt*2<<endl;
for(i=0;i<k;i++)
{
    if (a[i].x) printf("1 %lld %c\n",abs(a[i].x),a[i].x>0?'R':'L');
    if (a[i].y) printf("1 %lld %c\n",abs(a[i].y),a[i].y>0?'U':'D');
    printf("2\n");
    if (a[i].y) printf("1 %lld %c\n",abs(a[i].y),a[i].y>0?'D':'U');
    if (a[i].x) printf("1 %lld %c\n",abs(a[i].x),a[i].x>0?'L':'R');
    printf("3\n");
}
return 0;
}
