#include<stdio.h>
#include<iostream>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<vector>
#include<string>
using namespace std;
struct node
{
	int x;
	int y;
	int f;
}maze[100010];
int n,len;
bool cmp(node a,node b)
{
	return a.f<b.f;
}
void change(int a,int b)
{
	if(a>0)  printf("1 %d R\n",a);
	if(a<0)  printf("1 %d L\n",-a);
	if(b>0)  printf("1 %d U\n",b);
	if(b<0)  printf("1 %d D\n",-b);
	printf("2\n");
	if(b>0)  printf("1 %d D\n",b);
	if(b<0)  printf("1 %d U\n",-b);
	if(a>0)  printf("1 %d L\n",a);
	if(a<0)  printf("1 %d R\n",-a);
	printf("3\n"); 
}
int main()
{
	scanf("%d",&n);
	len=0;
	for(int i=0;i<n;i++)
	{
		scanf("%d %d",&maze[i].x,&maze[i].y);
		maze[i].f=abs(maze[i].x)+abs(maze[i].y);
		if(maze[i].x!=0&&maze[i].y!=0)  len+=6;
		else  len+=4;
	}
	sort(maze,maze+n,cmp);
	printf("%d\n",len);
	for(int i=0;i<n;i++)  change(maze[i].x,maze[i].y);

	return 0;
}
