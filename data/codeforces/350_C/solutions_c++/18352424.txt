#include<bits/stdc++.h>
using namespace std;
#define mp make_pair
#define pb push_back
typedef long long int ll;
vector<pair<ll,pair<ll,ll> > >v;
int main()
{
    ll i,n,x,y,p=0;
    scanf("%I64d",&n);
    for(i=0;i<n;i++){scanf("%I64d %I64d",&x,&y);if(x==0 || y==0){p++;}v.pb(mp(abs(x)+abs(y),mp(x,y)));}
    sort(v.begin(),v.end());
    cout<<(6*v.size()-2*p)<<endl;
    for(i=0;i<n;i++)
    {
        x=v[i].second.first;y=v[i].second.second;
        if(x<0){printf("1 %I64d L\n",abs(x));}
        else if(x>0){printf("1 %I64d R\n",x);}
        if(y<0){printf("1 %I64d D\n",abs(y));}
        else if(y>0){printf("1 %I64d U\n",y);}
        printf("2\n");
        if(x<0){printf("1 %I64d R\n",abs(x));}
        else if(x>0){printf("1 %I64d L\n",x);}
        if(y<0){printf("1 %I64d U\n",abs(y));}
        else if(y>0){printf("1 %I64d D\n",y);}
        printf("3\n");
    }
    return 0;
}
