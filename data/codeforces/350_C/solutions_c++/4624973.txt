#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<climits>
#include<algorithm>
using namespace std;
struct cs{
	int f;
	int x;
	int y;
};
inline bool operator<(cs a,cs b){return a.f<b.f;}
cs s[120000];
int n;
int main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("%d%d",&s[i].x,&s[i].y);
		s[i].f=abs(s[i].x)+abs(s[i].y);
	}
	sort(s,s+n);
	int ans=0;
	for (int i=0;i<n;i++){
		ans+=2;
		if (s[i].x) ans+=2;
		if (s[i].y) ans+=2;
	}
	printf("%d\n",ans);
	for (int i=0;i<n;i++){
		if (s[i].x)
			printf(s[i].x>0?"1 %d R\n":"1 %d L\n",abs(s[i].x) );
		if (s[i].y)
			printf(s[i].y>0?"1 %d U\n":"1 %d D\n",abs(s[i].y) );
		printf("2\n");
		if (s[i].x)
			printf(s[i].x>0?"1 %d L\n":"1 %d R\n",abs(s[i].x) );
		if (s[i].y)
			printf(s[i].y>0?"1 %d D\n":"1 %d U\n",abs(s[i].y) );
		printf("3\n");
	}
}
