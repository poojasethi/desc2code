#include<bits/stdc++.h>
using namespace std;
#define Int long long
struct node{
	Int x,y,d;
};
node a[100005];
bool cmp(node a,node b){
	return (a.d < b.d);
}
int main(){
	Int i,ans,n;
	cin>>n;
	ans=2*n;
	for(i=0;i<n;i++){
		cin>>a[i].x>>a[i].y;
		a[i].d = (a[i].x * a[i].x + a[i].y * a[i].y); 
		if(a[i].x!=0)ans+=2;
		if(a[i].y!=0)ans+=2;
	}
	sort(a,a+n,cmp);
	cout<<ans<<endl;
	for(i=0;i<n;i++){
		if(a[i].x > 0 ){
			cout<<"1 "<<a[i].x <<" R\n";
		}else if(a[i].x < 0 ){
			cout<<"1 "<<-1*a[i].x <<" L\n";
		}
		if(a[i].y > 0){
			cout<<"1 "<<a[i].y <<" U\n";
		}else if(a[i].y < 0){
			cout<<"1 "<<-1*a[i].y <<" D\n";
		}
		cout<<"2\n";
		if(a[i].y < 0){
			cout<<"1 "<<-1*a[i].y <<" U\n";
		}else if(a[i].y > 0){
			cout<<"1 "<<a[i].y <<" D\n";
		}
		if(a[i].x < 0 ){
			cout<<"1 "<<-1*a[i].x <<" R\n";				
		}else if(a[i].x > 0 ){
			cout<<"1 "<<a[i].x <<" L\n";
		}
		cout<<"3\n";
	}
	return 0;
}
