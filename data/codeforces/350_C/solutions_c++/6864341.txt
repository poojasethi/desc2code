#include<iostream>
#include<string>
#include<vector>
#include<math.h>
#include<algorithm>
using namespace std;

int n,i,ans,k;
struct _n{
	int x,y;
} v[100001];
bool cmp(const _n &a, const _n &b){
	return a.y==b.y?abs(a.x)<abs(b.x):abs(a.y)<abs(b.y);
}
int main(){
	cin>>n;
	for(i=0;i<n;i++){
		cin>>v[i].x>>v[i].y;
	}
	sort(v,v+n,cmp);
	for(i=0;i<n;i++){
		ans+=2*(1+(v[i].x!=0)+(v[i].y!=0));
	}
	cout<<ans<<"\n";
	for(i=0;i<n;i++){
		if (v[i].x) cout<<"1"<<" "<<abs(v[i].x)<<" "<<(v[i].x>0?'R':'L')<<"\n";
		if (v[i].y) cout<<"1"<<" "<<abs(v[i].y)<<" "<<(v[i].y>0?'U':'D')<<"\n";
		cout<<"2"<<"\n";
		if (v[i].y) cout<<"1"<<" "<<abs(v[i].y)<<" "<<(v[i].y>0?'D':'U')<<"\n";
		if (v[i].x) cout<<"1"<<" "<<abs(v[i].x)<<" "<<(v[i].x>0?'L':'R')<<"\n";
		cout<<"3"<<"\n";
	}
	return 0;
}
