#include <cstdio>
#include <algorithm>
#define abs(x) ((x)>0?(x):-(x))

using namespace std;

int n,i,j,k;

struct _n{
    int x,y;
} p[100001];

bool cmp(const _n &a, const _n &b){
    return a.y==b.y?abs(a.x)<abs(b.x):abs(a.y)<abs(b.y);
}

int main(){
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d%d",&p[i].x,&p[i].y);
    }
    sort(p,p+n,cmp);
    for(i=0,j=0;i<n;i++){
        j+=p[i].x==0||p[i].y==0?4:6;
    }
    printf("%d\n",j);
    for(i=0;i<n;i++){
        if (p[i].x) printf("1 %d %c\n",abs(p[i].x),p[i].x>0?'R':'L');
        if (p[i].y) printf("1 %d %c\n",abs(p[i].y),p[i].y>0?'U':'D');
        printf("2\n");
        if (p[i].y) printf("1 %d %c\n",abs(p[i].y),p[i].y>0?'D':'U');
        if (p[i].x) printf("1 %d %c\n",abs(p[i].x),p[i].x>0?'L':'R');
        printf("3\n");
    }
    return 0;
}
