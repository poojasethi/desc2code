#include<bits/stdc++.h>
#define ii pair<int,int> 
#define pb push_back
#define f first
#define s second
using namespace std;
ii v[100100];
int n,sum=0;
bool compare(ii p1,ii p2){
	return abs(p1.f)+abs(p1.s)<abs(p2.f)+abs(p2.s);
//	return (p1.f==p2.f)?abs(p1.s)<abs(p2.s):abs(p1.f)<abs(p2.f);
}
void do_work(){
	for(int i=0;i<n;i++){
		int x=v[i].f,y=v[i].s;
		if(x)printf("1 %d %c\n",abs(x),(x>0)?'R':'L');
		if(y)printf("1 %d %c\n",abs(y),(y>0)?'U':'D');
		printf("2\n");
		if(y)printf("1 %d %c\n",abs(y),(y>0)?'D':'U');
		if(x)printf("1 %d %c\n",abs(x),(x>0)?'L':'R');
		printf("3\n");
	}
}
int main(){
	int x,y;
	cin>>n;
	for(int i=0;i<n;i++){
		scanf("%d%d",&x,&y);
		v[i].f=x;
		v[i].s=y;
		sum+=2*(1+(x!=0)+(y!=0));
	}
	sort(v,v+n,compare);
	cout<<sum<<endl;
	do_work();
	
}
