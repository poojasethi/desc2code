#include <cstdio>
#include <cstdlib>
#include <algorithm>

using namespace std;

struct bomb {
	int x, y;
	bool operator < (const bomb& rhs) const {
		return abs(x) + abs(y) < abs(rhs.x) + abs(rhs.y);	
	}
};

int n;
bomb b[100000];

void process(int bi) {
	bomb t = b[bi];
	if (t.x) {
		printf("1 %d %c\n", abs(t.x), t.x < 0 ? 'L' : 'R');
	}
	if (t.y) {
		printf("1 %d %c\n", abs(t.y), t.y < 0 ? 'D' : 'U');
	}
	printf("2\n");
	if (t.x) {
		printf("1 %d %c\n", abs(t.x), t.x < 0 ? 'R' : 'L');
	}
	if (t.y) {
		printf("1 %d %c\n", abs(t.y), t.y < 0 ? 'U' : 'D');
	}
	printf("3\n");
}

int main() {
	scanf("%d", &n);
	int t = 0;
	for (int i = 0; i < n; ++i) {
		scanf("%d %d", &b[i].x, &b[i].y);
		if (b[i].x == 0 || b[i].y == 0) {
			t++;
		}
	}
	printf("%d\n", 2 * n + 4 * n - 2 * t);
	sort(b, b + n);
	for (int i = 0; i < n; ++i) {
		process(i);
	}
	return 0;
}
