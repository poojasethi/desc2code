

#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
#define N 120000
using namespace std;
struct ppp{
	int x,y;
}a[N];
int i,n,ans;
bool cmp(const ppp&a,const ppp&b){
	return abs(a.x)+abs(a.y)<abs(b.x)+abs(b.y);
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d%d",&a[i].x,&a[i].y);
	sort(a+1,a+n+1,cmp);
	ans=0;
	for(i=1;i<=n;++i){
		if(a[i].x)ans+=2;
		if(a[i].y)ans+=2;
		ans+=2;
	}
	printf("%d\n",ans);
	for(i=1;i<=n;++i){
		if(a[i].x>0)printf("1 %d R\n",a[i].x);
		if(a[i].x<0)printf("1 %d L\n",-a[i].x);
		if(a[i].y>0)printf("1 %d U\n",a[i].y);
		if(a[i].y<0)printf("1 %d D\n",-a[i].y);
		printf("2\n");
		if(a[i].x>0)printf("1 %d L\n",a[i].x);
		if(a[i].x<0)printf("1 %d R\n",-a[i].x);
		if(a[i].y>0)printf("1 %d D\n",a[i].y);
		if(a[i].y<0)printf("1 %d U\n",-a[i].y);
		printf("3\n");
	}
}
