#include<bits/stdc++.h>
using namespace std;

int n;
struct node{
	int x, y, d;
	bool operator<(const node & n) const {
		return d < n.d;
	}
}a[100010];

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	int k = 0;
	for(int i=0 ; i<n ; ++i)  {
		scanf("%d%d", &a[i].x, &a[i].y);
		a[i].d = abs(a[i].x) + abs(a[i].y);
		if(a[i].x == 0 || a[i].y == 0)
			k += 4;
		else
			k += 6;
	}
	sort(a, a+n);
	printf("%d\n", k);
	for(int i=0 ; i<n ; ++i) {
		if(a[i].x)
			printf("1 %d %c\n", abs(a[i].x), a[i].x > 0 ? 'R' : 'L');
		if(a[i].y)
			printf("1 %d %c\n", abs(a[i].y), a[i].y > 0 ? 'U' : 'D');
		puts("2");
		if(a[i].x)
			printf("1 %d %c\n", abs(a[i].x), a[i].x > 0 ? 'L' : 'R');
		if(a[i].y)
			printf("1 %d %c\n", abs(a[i].y), a[i].y > 0 ? 'D' : 'U');
		puts("3");
	}
	return 0;
}
