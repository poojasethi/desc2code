#include<bits/stdc++.h>
using namespace std;
int x,y,k,n,i;
pair <int, pair<int,int> > a[100003];
main()
{
    scanf("%d",&n);
    for (i=0;i<n;i++)
        {scanf("%d%d",&x,&y);
        if (x==0||y==0) k+=4;
        else k+=6;
        a[i]=make_pair(abs(x)+abs(y),make_pair(x,y));}
    sort(a,a+n);
    printf("%d\n",k);
    for (i=0;i<n;i++)
    {   x=a[i].second.first;
        y=a[i].second.second;
        if (x>0)
            printf("1 %d R\n",x);
        if (x<0)
            printf("1 %d L\n",-x);
         if (y>0)
            printf("1 %d U\n",y);
        if (y<0)
            printf("1 %d D\n",-y);
        printf("2\n");
        if (x>0)
            printf("1 %d L\n",x);
        if (x<0)
            printf("1 %d R\n",-x);
         if (y>0)
            printf("1 %d D\n",y);
        if (y<0)
            printf("1 %d U\n",-y);
        printf("3\n");
    }
}
