#include<iostream>
using namespace std;
int main(){
	int n;
	cin>>n;
	int ar[n][2],count=0;
	int tempx , tempy;
	static	int ar1[2001][2001]={0};
	for(int i =0 ; i < n;i++){
		cin>>ar[i][0]>>ar[i][1];
			ar1[ar[i][0]+1000][ar[i][1]+1000]=1;		
}
		for(int i=0 ; i < n;i++){
			for(int j =i+1; j < n ;j++){
			tempx = ar[i][0] + ar[j][0];
			tempy = ar[i][1] + ar[j][1];
			if( (tempx%2) == 0 && (tempy%2)==0){
				if( ar1[(tempx/2) +1000][(tempy/2)+1000] != 0){
				count++;
			}
	}
		}
}
		cout<<count<<endl;
	}
