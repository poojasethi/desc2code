#include <cstdio>
#include <iostream>
#define base 1000
using namespace std;
struct point
{
	int x,y;
}p[3008];
bool ok[2009][2009];

int main()
{
	int n;
	cin>>n;
	for (int i=0;i<n;i++)
	{
		cin>>p[i].x>>p[i].y;
		ok[p[i].x+base][p[i].y+base]=1;
	}
	int ans=0;
	for (int i=0;i<n;i++)
		for (int j=i+1;j<n;j++)
			if ( ((p[i].x+p[j].x)&1)==0 && ((p[i].y+p[j].y)&1)==0)
			{
				int x=(p[i].x+p[j].x)/2;
				int y=(p[i].y+p[j].y)/2;
				if (ok[x+base][y+base]) ans++;
			}
	cout<<ans<<endl;
	return 0;
}
