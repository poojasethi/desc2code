#include <iostream>

using namespace std;
int a[3001][3001];
int main()
{
    int n,x[3000],y[3000],i,count=0,j;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>x[i]>>y[i];
        x[i]+=1000;
        y[i]+=1000;
        a[x[i]][y[i]]=1;
    }
    for(i=0;i<n;i++)
    {
        for(j=0;j<n;j++)
        {
            if(i!=j&&(x[i]+x[j])%2==0&&(y[i]+y[j])%2==0&&a[(x[i]+x[j])/2][(y[i]+y[j])/2]>0)
            count++;
        }
    }
    cout<<count/2;
    return 0;
}
