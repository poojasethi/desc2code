#include <iostream>
#include <stdio.h>
using namespace std;
const int S=1000;
int a[2010][2010];
int n;
int x[3333],y[3333];
int main()
{
    cin>>n;
    for(int i=0;i<n;i++) cin>>x[i]>>y[i],a[x[i]+S][y[i]+S]++;
    int ans=0;
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
        {
            int xx=x[i]+x[j];
            int yy=y[i]+y[j];
            if((xx&1 )|| (yy&1) ) continue;
            ans+=a[xx/2+S][yy/2+S];
        }
    cout<<ans<<endl;
}
