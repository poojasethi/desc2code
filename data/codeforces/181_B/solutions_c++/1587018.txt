#include <cstdio>

using namespace std;

bool exi[2010][2010];
int x[3001],y[3001];

int main(void)
{
  int n;
  scanf("%d",&n);
  for(int i=1;i<=n;++i)
  {
    scanf("%d %d",&x[i],&y[i]);
    exi[x[i]+1000][y[i]+1000]=true;
  }
  int ans=0;
  for(int i=1;i<n;++i)
  {
    for(int j=i+1;j<=n;++j)
    {
      if(((x[i]+x[j])%2!=0)||((y[i]+y[j])%2!=0))
        continue;
      int p=(x[i]+x[j])/2,q=(y[i]+y[j])/2;
      if(exi[p+1000][q+1000])
        ++ans;
    }
  }
  printf("%d\n",ans);
  return 0;
}
