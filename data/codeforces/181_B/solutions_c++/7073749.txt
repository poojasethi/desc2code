#include <iostream>
#include <stdio.h>
using namespace std;
int a[3005][3005],b[3005],c[3005];
int main(){
	int n,res=0;
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("%d%d",&b[i],&c[i]);
		c[i]+=1000;
		b[i]+=1000;
		a[b[i]][c[i]]=1;
		
	}
	int xx,yy;
	for (int i=0;i<n;i++){
		for (int j=i+1;j<n;j++){
			if ((b[i]+b[j])%2==0 && (c[i]+c[j])%2==0){
				yy=(c[i]+c[j])/2;
				xx=(b[i]+b[j])/2;
				if (a[xx][yy]==1){
				    res++;
			    }
			}
			
		}
	}
	printf("%d",res);
	return 0;
}
