#include<cstdio>
#include<iostream>
#include<cstring>
#include<map>
#include<algorithm>
using namespace std;
int x[3009],y[3009];
int cnt[2009][2009];
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>x[i]>>y[i];
        cnt[x[i]+1000][y[i]+1000]++;
    }
    int ans=0;
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
            if(!((x[i]+x[j])%2)&&!((y[i]+y[j])%2))
                ans+=cnt[((x[i]+x[j])>>1)+1000][((y[i]+y[j])>>1)+1000];
    cout<<ans<<endl;
}
