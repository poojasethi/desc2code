#include<cstdio>
#include<cstring>
#include<cstdlib>
using namespace std;
int a[2100][2100],x[3100],y[3100];
int n,ans;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i)
		scanf("%d%d",&x[i],&y[i]),a[x[i]+1000][y[i]+1000]=1;
	for (int i=1;i<=n;++i)
		for (int j=i+1;j<=n;++j)
		{
			int X=x[i]+x[j],Y=y[i]+y[j];
			if ((X&1)||(Y&1)) continue;
			X/=2,Y/=2;
			ans+=a[X+1000][Y+1000];
		}
	printf("%d\n",ans);
	//system("pause");
}
