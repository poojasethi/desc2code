#include <cstdio>
#include <algorithm>
#include <utility>
#include <set>

using namespace std;

int n, ans;
pair<int,int> v[20010];
set< pair<int,int> > st;

int main()
{
	scanf("%d", &n);
	for (int i=0; i<n; i++) scanf("%d%d", &v[i].first, &v[i].second), st.insert(v[i]);
	for (int i=0; i<n; i++)
		for (int j=i+1; j<n; j++) {
			int x = v[i].first + v[j].first;
			int y = v[i].second + v[j].second;
			if (abs(x) % 2 == 0 && abs(y) % 2 == 0) ans += st.count(make_pair(x/2,y/2));
		}
	printf("%d\n", ans);
	return 0;
}
