#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;
const int maxn=2013;
bool map[maxn][maxn];
int x[maxn*2],y[maxn*2],n;
int main(){
	while(cin>>n){
		memset(map,0,sizeof(map));
		for(int i=1;i<=n;i++){
			cin>>x[i]>>y[i];
			x[i]+=1000;
			y[i]+=1000;
			map[x[i]][y[i]]=1;
		}
		int sum=0,xx,yy;
		for(int i=1;i<n;i++)
			for(int j=i+1;j<=n;j++){
				xx=x[i]+x[j];
				yy=y[i]+y[j];
				if((xx%2==0)&&(yy%2==0)&&map[xx/2][yy/2]==1)
					sum++;
			}
			cout<<sum<<endl;
	}
}
