#include<iostream>
#define rep(i,n) for(int i=1;i<=n;i++)
using namespace std;
int ans,n,i,j,x[3001],y[3001];
bool map[2005][2005];
int main(){
	cin>>n;
	rep(i,n){
		cin>>x[i]>>y[i];
		x[i]+=1000;
		y[i]+=1000;
		map[x[i]][y[i]]=true;
	}
	rep(i,n-1)
	for(j=i+1;j<=n;j++)
	if((x[i]+x[j])%2==0 && (y[i]+y[j])%2==0 && map[(x[i]+x[j])/2][(y[i]+y[j])/2])
	ans++;
	cout<<ans;
	return 0;
}
