#include<iostream>
using namespace std;

int n,s;
int a[30000],b[30000];
bool f[10000][10000];

int main()
{
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i]>>b[i];
        a[i]+=1000;
        b[i]+=1000;
        f[a[i]][b[i]]=true;
    }
    
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
        {
            if((a[i]+a[j])%2==0 && (b[i]+b[j])%2==0 && f[(a[i]+a[j])/2][(b[i]+b[j])/2]==true) s++;
        }
    cout<<s;
    return 0;
}
