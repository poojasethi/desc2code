#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
#define MAXN 3003

pair<int,int> q[MAXN];
int ans,N;

int main()
{
	cin>>N;
	for(int i=0;i<N;i++)
		scanf("%d%d",&q[i].first,&q[i].second);
	sort(q,q+N);
	for(int i=0;i<N;i++) for(int j=i+1;j<N;j++)
	{
		int tmpx=q[i].first+q[j].first;
		int tmpy=q[i].second+q[j].second;
		if(tmpx%2==0&&tmpy%2==0)
			if(binary_search(q,q+N,make_pair(tmpx/2,tmpy/2))) ans++;
	}
	cout<<ans<<endl;
	return 0;
}
