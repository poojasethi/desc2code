#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;

int n,x[3001],y[3001];
int i,j,a,b,ans;
bool fix[2001][2001];

main()
{cin>>n;
 for(i=1;i<=n;i++){
  cin>>x[i]>>y[i];
  x[i]+=1000;
  y[i]+=1000;
  
  fix[x[i]][y[i]]=1;}
  
 for(i=1;i<n;i++)
  for(j=i+1;j<=n;j++)
   if((abs(x[i]-x[j]))%2==0 &&
      (abs(y[i]-y[j]))%2==0){
     
     a=(abs(x[i]+x[j])/2);
     b=(abs(y[i]+y[j])/2);
     if(fix[a][b])ans++;} 
      
      
 cout<<ans<<endl;
// system("pause");
}
