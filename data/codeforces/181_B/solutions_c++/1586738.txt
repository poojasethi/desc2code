#include <iostream>
using namespace std;
typedef long long ll;
const int MAXN=3000+10;
bool mark[2111][2111];
ll x[MAXN],y[MAXN];
ll n,ans;
int main()
{
	cin >> n;
	for(int i=0;i<n;i++)
	{
		cin >> x[i] >> y[i];
		x[i]+=1000;
		y[i]+=1000;
		mark[x[i]][y[i]]=1;
	}
	for(int i=0;i<n;i++)
		for(int j=0;j<i;j++)
		{
			if((x[i]+x[j])%2||(y[i]+y[j])%2)
				continue;
			if(mark[(x[i]+x[j])/2][(y[i]+y[j])/2])
				ans++;
		}
	cout << ans << endl;
	return 0;
}
