#include<iostream>
#include<stdio.h>
using namespace std;
int x[5000],y[5000];
bool m[3000][3000];
int main(){
	int n,q=0,mx,my;
	scanf("%d",&n);
	for(int i=0;i<n;++i){
		scanf("%d%d",&x[i],&y[i]);
		m[x[i]+=1000][y[i]+=1000]=true;
	}
	for(int i=0;i<n;++i){
		for(int j=0;j<i;++j){
			mx=x[i]+x[j];
			my=y[i]+y[j];
			q+=(mx%2==0 && my%2==0 && m[mx/2][my/2]);
		}
	}
	printf("%d\n",q);
	return(0);
}
