#include<iostream>
#include<cstdio>
using namespace std;
int n,i,j,ans,x[3001],y[3001];
bool f[2001][2001];
int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	{
		scanf("%d%d",&x[i],&y[i]);
		x[i]+=1000;
		y[i]+=1000;
		f[x[i]][y[i]]=true;
	}
	for (i=1;i<n;i++)
	for (j=i+1;j<=n;j++)
	if ((x[i]+x[j])%2==0&&(y[i]+y[j])%2==0&&f[(x[i]+x[j])/2][(y[i]+y[j])/2])
	ans++;
	printf("%d",ans);
}
