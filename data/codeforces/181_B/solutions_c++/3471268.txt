#include<iostream>
#include<cstdio>
using namespace std;
bool hash[2000][2000];
int x[3000];
int y[3000];
int main()
{
	int n;
	cin>>n;
	int i,j;
	int AC=0;
	for(i=0;i<n;i++)
	{
		scanf("%d%d",x+i,y+i);
		hash[x[i]+1000][y[i]+1000]=true;
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<i;j++)
		{
			if((~(x[i]^x[j])&1)&&(~(y[i]^y[j])&1)&&(hash[(x[i]+x[j])/2+1000][(y[i]+y[j])/2+1000]))
			{
				AC++;
			}
		}
	}
	cout<<AC;
	return 0;
}
