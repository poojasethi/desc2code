#include <stdio.h>
int n,ans,x[3010],y[3010];
bool pt[2010][2010];
int main()
{
	int tx,ty;
	scanf("%d",&n);
	for (int i=0; i<n; i++) {
		scanf("%d%d",&tx,&ty);
		x[i]=tx+1000,y[i]=ty+1000;
		pt[x[i]][y[i]]=true;
	}
	for (int i=0; i<n; i++) {
		for (int j=i+1; j<n; j++) {
			if (!((x[i]+x[j])&1)&&!((y[i]+y[j])&1)&&pt[(x[i]+x[j])/2][(y[i]+y[j])/2])
				ans++;
		}
	}
	printf("%d\n",ans);
	return 0;
}
