#include<cstdio>
#include<cstdlib>
using namespace std;
int a[2001][2001];
int main()
{
	int n;
	scanf("%d",&n);
	int x[n],y[n];
	for(int i=0;i<n;i++)
	{
		scanf("%d%d",&y[i],&x[i]);
		y[i]+=1000;
		x[i]+=1000;
		a[y[i]][x[i]]++;
	}
	long ans=0;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
			if((x[i]-x[j])%2==0&&(y[i]-y[j])%2==0)
			{
				int mx,my;
				mx=x[i]+(x[j]-x[i])/2;
				my=y[i]+(y[j]-y[i])/2;
				ans+=a[my][mx];
			}
	printf("%d",ans);
}
