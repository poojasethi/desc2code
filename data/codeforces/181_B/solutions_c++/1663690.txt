#include <iostream>

using namespace std;

int x[3010],y[3010];
bool map[2010][2010];

int main()
{
    int T,ans=0;
	cin >> T;
	for (int i=0;i<T;i++)
	{
		cin >> x[i] >> y[i];
		map[x[i]+1000][y[i]+1000]=1;
	}
	for (int i=0;i<T;i++)
		for (int j=i+1;j<T;j++)
			if (i!=j&&(x[i]+x[j])%2==0&&(y[i]+y[j])%2==0 && map[(x[i]+x[j])/2+1000][(y[i]+y[j])/2+1000]==1)
				ans++;
	cout << ans << endl;
	return 0;
}
