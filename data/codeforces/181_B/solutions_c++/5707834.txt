#include <stdio.h>
#include <iostream>
using namespace std;
int n,x[3005],y[3005],a,b,s=0;
bool f[2500][2500]={false};
int main( ){
	cin>>n;
	for (int i=1;i<=n;i++){
		cin>>x[i]>>y[i];
		x[i]+=1200;
		y[i]+=1200;
		f[x[i]][y[i]]=true;
	}
	for (int i=1;i<n;i++)
		for (int j=i+1;j<=n;j++){
			a=x[i]+x[j];
			b=y[i]+y[j];
			if (a%2==0 && b%2==0 && f[a/2][b/2])
				s++;
		}
	cout<<s<<endl;
	return 0;
}
