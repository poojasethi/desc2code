#include <cstdio>

int N,x[5000],y[5000],Tot[5000][5000],Ans;

inline bool odd(int x){return x%2==1;}

int main()
{
    scanf("%d",&N);
    for (int i=1;i<=N;++i){
        scanf("%d%d",&x[i],&y[i]);
        Tot[x[i]+1001][y[i]+1001]=1;
    }
    for (int i=1;i<N;++i)
    for (int j=i+1;j<=N;++j)
    {
        if ((odd(x[i]+x[j]+3000))||(odd(y[i]+y[j]+3000)))continue;
        Ans+=Tot[(x[i]+x[j])/2+1001][(y[i]+y[j])/2+1001];   
    }
    printf("%d\n",Ans);
    return 0;
}
