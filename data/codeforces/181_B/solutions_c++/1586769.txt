#include <iostream>
#include <cstdio>
using namespace std;
typedef long long ll;
const int N=3010;
bool f[N][N];
int x[N],y[N];
int main()
{
    int n,ans=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>x[i]>>y[i];
        x[i]+=1000;
        y[i]+=1000;
        f[x[i]][y[i]]=true;
    }
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
            if ((x[i]+x[j])%2==0 && (y[i]+y[j])%2==0 && f[(x[i]+x[j])/2][(y[i]+y[j])/2])
                ans++;
    cout<<ans;
    return 0;
}
