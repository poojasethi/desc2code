#include<stdio.h>
int n,x[3000],y[3000],map[3001][3001]={},i,j,cnt=0;
main(){
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d%d",&x[i],&y[i]);
        x[i]+=1000,y[i]+=1000;
        map[x[i]][y[i]]=1;
    }
    for(i=0;i<n;i++)
        for(j=0;j<n;j++)
            if(i!=j&&(x[i]+x[j])%2==0&&(y[i]+y[j])%2==0&&map[(x[i]+x[j])/2][(y[i]+y[j])/2])cnt++;
    printf("%d\n",cnt/2);
}
