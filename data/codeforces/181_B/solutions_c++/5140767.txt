#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
struct point{
	int x,y;
};
point a[3000],t;
int n,kol=0;
bool use[3000][3000];
int main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("%d%d",&a[i].x,&a[i].y);
		a[i].x+=1000;
		a[i].y+=1000;
		use[a[i].x][a[i].y]=1;
	}
	for (int i=0;i<n;i++)
		for (int k=i+1;k<n;k++){
			t.x=(a[i].x+a[k].x);
			t.y=(a[i].y+a[k].y);
			if ((t.x%2==0 && t.y%2==0) && use[t.x/2][t.y/2])
				kol++;
			}
	cout<<kol;
	return 0;
}
