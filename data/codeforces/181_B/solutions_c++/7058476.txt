#include<cstdio>
int x[3001],y[3001];
bool a[2010][2010]={0};
int main()
{
	int n,i,j,cnt=0;
	scanf("%d",&n);
	for(i=0;i<n;++i){
		scanf("%d%d",x+i,y+i);
		a[x[i]+=1000][y[i]+=1000]=true;
	}
	for(i=0;i<n;++i)
	for(j=0;j<n;++j){
		int mx=x[i]+x[j],my=y[i]+y[j];
		if(mx>=0&&mx<=4000&&!(mx&1)&&my>=0&&my<=4000&&!(my&1)&&a[mx/2][my/2])++cnt;
	}
	printf("%d\n",(cnt-n)/2);
	return 0;
}
