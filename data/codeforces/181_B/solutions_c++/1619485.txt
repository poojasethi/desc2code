#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int a[2005][2005];
int x[3005], y[3005];

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d%d", &x[i], &y[i]);
		a[x[i]+1000][y[i]+1000]=1;
	}
	int cnt=0;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++) {
			if((x[i]+x[j])%2==0 && (y[i]+y[j])%2==0) {
				int nx = (x[i]+x[j])/2, ny = (y[i]+y[j])/2;
				if(a[nx+1000][ny+1000])
					++cnt;
			}
		}
	printf("%d\n", cnt);
	return 0;
}
