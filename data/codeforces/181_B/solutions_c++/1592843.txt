#include <stdio.h>
#include <algorithm>
char pt[2001][2001]={0};
int n, xx[3000], yy[3000];
int main()
{
	int i, j, x, y, ans=0;
	//freopen("input.txt", "r", stdin);
	scanf("%d", &n);
	for(i=0; i<n; i++) { scanf("%d %d", &xx[i], &yy[i]); xx[i]+=1000; yy[i]+=1000;pt[xx[i]][yy[i]]=1;}
	for(i=0; i<n; i++)
		for(j=i+1; j<n; j++) {
			x=xx[i]+xx[j]; y=yy[i]+yy[j];
			if(x%2==0 && y%2==0) if(pt[x/2][y/2]) ans++;
		}
	printf("%d\n", ans);
}
