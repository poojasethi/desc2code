#include <iostream>
using namespace std;
bool p[2001][2001];
int main(){
    int n;
    cin>>n;
    int V[n][2];
    for (int i = 0; i < n; i++)
        cin>>V[i][0]>>V[i][1], p[V[i][0]+1000][V[i][1]+1000] = true;
    long long c = 0;
    for (int i = 0; i < n; i++)
        for (int j = i+1; j < n; j++){
            if ((V[i][0]+V[j][0])%2==0 && (V[i][1]+V[j][1])%2==0 && p[(V[i][0]+V[j][0])/2+1000][(V[i][1]+V[j][1])/2+1000])
                c++;
        }
    cout<<c<<endl;
    //system("PAUSE");
    return 0;
}
