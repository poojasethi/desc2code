#include<iostream>
using namespace std;
bool a[2001][2001]={0};
int main()
{
	long int t=0;
	int x,y,n,i,j,l,p;
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>x>>y;
		a[x+1000][y+1000]=1;
	}
	for(i=0;i<=2000;i++)
		for(j=0;j<=2000;j++)
			if(a[i][j]==1)
				for(l=i;l<=2000;l+=2)
					for(p=j%2;p<=2000;p+=2)
						if((l==i && p>j) || l!=i)
							if(a[l][p]==1 && a[(l+i)/2][(p+j)/2]==1)
								t++;
	cout<<t;
}
