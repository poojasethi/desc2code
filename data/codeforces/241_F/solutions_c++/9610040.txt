#include<cstdio>
using namespace std;
int n,m,k,dx,dy,L;
int x[35],y[35];
char s[1005],a[105][105];
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++)
	{
		scanf("%s",a[i]+1);
		for(int j=1;j<=m;j++)
			if(a[i][j]>96&&a[i][j]<123)
				x[a[i][j]-97]=i,
				y[a[i][j]-97]=j,
				a[i][j]=49;
	}
	scanf("%d%d",&dx,&dy);
	scanf("%s",s+1);
	scanf("%d%d",&x[26],&y[26]);
	for(L=1;s[L];L++)
		s[L]-=97;
	s[L]=26;
	for(int i=1;i<=L;i++)
		while(dx!=x[s[i]]||dy!=y[s[i]])
		{
			k-=a[dx][dy]-48;
			if(k<0)break;
			if(dx>x[s[i]])dx--;
			if(dx<x[s[i]])dx++;
			if(dy>y[s[i]])dy--;
			if(dy<y[s[i]])dy++;
		}
	printf("%d %d\n",dx,dy);
	return 0;
}
