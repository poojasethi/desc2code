#include <iostream>
#include <fstream>
#include <queue>
#include <map>
#include <set>
#include <cstring>
#include <iomanip>
#include <algorithm>

#define pb push_back
#define mp make_pair
#define xx first
#define yy second

using namespace std;
const long long maxn = 1000 +  10;

typedef pair <int, int> ii;

int Arr[maxn][maxn];
ii POS[maxn];

int main () {
	ios::sync_with_stdio (0);
	int T=1;
	while(T--)
	{
		int N,M,K;
		cin >> N >> M >> K;
		for(int i=1;i<=N;i++)
				for(int j=1;j<=M;j++)
				{
					char t;cin >> t;
					if(t=='#')
						Arr[i][j]=-1;
					else if(t>='1' && t<='9')
						Arr[i][j]=t-'0';
					else
					{
						Arr[i][j]=1;
						POS[t-'a']=make_pair(i,j);
					}
				}
		ii A,B;
		string S;
		cin >> A.xx >> A.yy >> S >> B.xx >> B.yy;
		int ind=0;
		while(true)
		{
			if(A==B && ind==((int)S.size()))
				break;
			if(Arr[A.xx][A.yy]>K)
					break;
			K-=Arr[A.xx][A.yy];
			if(ind!=(int)S.size() && A==POS[S[ind]-'a'])
					ind++;
			ii Nxt;
			if(ind!=(int)S.size())
				Nxt=POS[S[ind]-'a'];
			else
				Nxt=B;
			if(Nxt.xx==A.xx && Nxt.yy>A.yy)
					A.yy++;
			else if(Nxt.xx==A.xx && Nxt.yy<A.yy)
					A.yy--;
			else if(Nxt.xx>A.xx && Nxt.yy==A.yy)
					A.xx++;
			else if(Nxt.xx<A.xx && Nxt.yy==A.yy)
					A.xx--;
		}
		cout<<A.xx<<" "<<A.yy<<endl;
	}
	return 0;
}
