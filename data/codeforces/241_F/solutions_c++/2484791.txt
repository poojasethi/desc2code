#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <algorithm>
#include <queue>
using namespace std;

const int N = 100;

char b[N][N+1];
int ji[26], jj[26];

int signum(int x) { if (x==0) return 0; if (x > 0) return 1; return -1; }

void go(int i1, int j1, int i2, int j2, int& k) {
  
//  printf("go %d %d -> %d %d, k = %d\n", i1,j1,i2,j2,k);
  int di = signum(i2 - i1);
  int dj = signum(j2 - j1);
  while (i1 != i2 || j1 != j2) {
    int d = b[i1][j1] - '0';
  //  printf("i,j = %d, %d, d = %d, k: %d\n", i1,j1,d,k);

    if (k < d) {
      printf("%d %d\n", 1+i1,1+j1);
      exit(0);
    } else {
      k -= d;
      i1 += di;
      j1 += dj;
    }
  }
}

int main() {
  int n,m,k;
  scanf("%d%d%d",&n,&m,&k);
  for (int i=0;i<n;i++) scanf("%s",b[i]);
  for (int i=0;i<26;i++) ji[i]=jj[i]=-1;
  for (int i=0;i<n;i++)
    for (int j=0;j<m;j++)
      if (b[i][j] >= 'a' && b[i][j] <= 'z') {
        ji[b[i][j]-'a'] = i;
        jj[b[i][j]-'a'] = j;
        b[i][j] = '1';
      }
  int si,sj,ti,tj;
  char s[1001];
  scanf("%d%d",&si,&sj);
  scanf("%s",s);
  int l = strlen(s);
  for (int i=0;i<l;i++) s[i] -= 'a';
  scanf("%d%d",&ti,&tj);
  go(si-1,sj-1,ji[s[0]],jj[s[0]],k);
  for (int i=0;i<l-1;i++)
    go(ji[s[i]],jj[s[i]],ji[s[i+1]],jj[s[i+1]],k);
  go(ji[s[l-1]],jj[s[l-1]],ti-1,tj-1,k);
  printf("%d %d\n", ti, tj);
  return 0;
}
