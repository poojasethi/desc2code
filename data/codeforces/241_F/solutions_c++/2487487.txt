
#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <vector>
#include <queue>
#define FOR(it,c) for ( __typeof((c).begin()) it=(c).begin(); it!=(c).end(); it++ )

typedef long long int LL;

using namespace std;

int n,m,k,l;
int stx,sty,edx,edy;
char d[110][110];
char s[1100];
char t[30][2];

void input(){
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;i++){
		scanf("%s",d[i]);
	}
	scanf("%d%d %s %d%d",&stx,&sty,s,&edx,&edy);
	stx--,sty--,edx--,edy--;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			if('a'<=d[i][j] && d[i][j]<='z'){
				t[d[i][j]-'a'][0]=i;
				t[d[i][j]-'a'][1]=j;
				d[i][j]='1';
			}
		}
	}
	l=strlen(s);
}

void solve(){
	int x=stx,y=sty,tx,ty;
	int time=0;
	for(int i=0;i<l;i++){
		tx=t[s[i]-'a'][0],ty=t[s[i]-'a'][1];
		//printf("-%d %d, %d %d\n",x,y,tx,ty);
		if(tx<x)     for(;x!=tx;x--){time+=d[x][y]-'0';if(time>k)break;}
		else if(tx>x)for(;x!=tx;x++){time+=d[x][y]-'0';if(time>k)break;}
		else if(ty<y)for(;y!=ty;y--){time+=d[x][y]-'0';if(time>k)break;}
		else if(ty>y)for(;y!=ty;y++){time+=d[x][y]-'0';if(time>k)break;}	
		if(time>=k){
			printf("%d %d\n",x+1,y+1);
			return;
		}
	}
	tx=edx,ty=edy;
	if(tx<x)     for(;x!=tx;x--){time+=d[x][y]-'0';if(time>k)break;}
	else if(tx>x)for(;x!=tx;x++){time+=d[x][y]-'0';if(time>k)break;}
	else if(ty<y)for(;y!=ty;y--){time+=d[x][y]-'0';if(time>k)break;}
	else if(ty>y)for(;y!=ty;y++){time+=d[x][y]-'0';if(time>k)break;}	
	
	printf("%d %d\n",x+1,y+1);
	return;
}

int main(){
	input();
	solve();
	return 0;
}
