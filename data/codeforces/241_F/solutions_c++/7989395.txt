#include<cstdio>
#include<cstring>
#include<algorithm>
#include<queue>
#include<vector>
#include<string>
#define LL long long
#define MP make_pair
#define PII pair<int,int>
using namespace std;
PII p[1010],Po[30];
int n,m,K,tot,ti[110][110],xx[4]={1,-1,0,0},yy[4]={0,0,1,-1};
char s[110][110],S[1010];
void read(int &x)
{
	char ch=getchar();int mark=1;for(;ch!='-'&&(ch<'0'||ch>'9');ch=getchar());if (ch=='-') mark=-1,ch=getchar();
	for(x=0;ch>='0'&&ch<='9';ch=getchar()) x=x*10+ch-48;x*=mark;
}
PII Get(int K)
{
	PII now=p[1];
	int time=0;
	for(int i=2;i<=tot;i++)
	{
		int d=3;
		if (now.first<p[i].first) d=0;
		if (now.first>p[i].first) d=1;
		if (now.second<p[i].second) d=2;
		for(;now.first!=p[i].first||now.second!=p[i].second;)
		{
			time+=ti[now.first][now.second];
			if (time>K) return now;
			now.first+=xx[d];
			now.second+=yy[d];
		}
	}
	return now;
}
int main()
{
	read(n);read(m);read(K);
	for(int i=1;i<=n;i++) 
	{
		scanf("%s",s[i]+1);
		for(int j=1;j<=m;j++)
			if (s[i][j]>='a'&&s[i][j]<='z') Po[s[i][j]-'a']=MP(i,j),ti[i][j]=1;
			else if (s[i][j]>='1'&&s[i][j]<='9') ti[i][j]=s[i][j]-'0';
	}
	read(p[1].first);read(p[1].second);tot=1;
	scanf("%s",S);
	int l=strlen(S);
	for(int i=0;i<l;i++) p[++tot]=Po[S[i]-'a'];
	read(p[++tot].first);read(p[tot].second);
	PII ans=Get(K);
	printf("%d %d\n",ans.first,ans.second);
	return 0;
}
