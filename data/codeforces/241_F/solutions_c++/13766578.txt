#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
#define N 105
int n,m,p,Sx,Sy,Tx,Ty,wz[N*3][2];char b[N][N],c[N*10];
void Move(int x,int y)
 {
    while (x!=Sx||y!=Sy)
     {
        if ((p-=b[Sx][Sy]-'0')<0)
         {cout <<Sx<<' '<<Sy<<endl;exit(0);}
        Sx+=(x>Sx)-(x<Sx);
        Sy+=(y>Sy)-(y<Sy);
     }
 }
int main()
 {
    cin >>n>>m>>p;
    for (int i=1;i<=n;i++)
     {
        scanf("%s",b[i]+1);
        for (int j=1;j<=m;j++)
         if (b[i][j]>='a'&&b[i][j]<='z')
           wz[b[i][j]-'a'][0]=i,wz[b[i][j]-'a'][1]=j,
           b[i][j]='1';
     }
    scanf("%d%d%s%d%d",&Sx,&Sy,c,&Tx,&Ty);
    int len=strlen(c);
    for (int i=0;i<len;i++)
      Move(wz[c[i]-'a'][0],wz[c[i]-'a'][1]);
    Move(Tx,Ty);
    cout <<Sx<<' '<<Sy<<endl;
    return 0;
 }
