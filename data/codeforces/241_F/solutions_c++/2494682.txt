# define PII pair<int,int>
# define FF first
# define SS second
# define LL long long
# define PB push_back
using namespace std;
const int MAX_N=100+10;
int Arr[MAX_N][MAX_N];
PII POS[MAX_N];
int main()
{
	ios::sync_with_stdio(false);
	int T=1;
	while(T--)
	{
		int N,M,K;
		cin >> N >> M >> K;
		for(int i=1;i<=N;i++)
				for(int j=1;j<=M;j++)
				{
					char t;cin >> t;
					if(t=='#')
						Arr[i][j]=-1;
					else if(t>='1' && t<='9')
						Arr[i][j]=t-'0';
					else
					{
						Arr[i][j]=1;
						POS[t-'a']=make_pair(i,j);
					}
				}
		PII A,B;
		string S;
		cin >> A.FF >> A.SS >> S >> B.FF >> B.SS;
		int ind=0;
		while(true)
		{
			if(A==B && ind==((int)S.size()))
				break;
			if(Arr[A.FF][A.SS]>K)
					break;
			K-=Arr[A.FF][A.SS];
			if(ind!=(int)S.size() && A==POS[S[ind]-'a'])
					ind++;
			PII Nxt;
			if(ind!=(int)S.size())
				Nxt=POS[S[ind]-'a'];
			else
				Nxt=B;
			if(Nxt.FF==A.FF && Nxt.SS>A.SS)
					A.SS++;
			else if(Nxt.FF==A.FF && Nxt.SS<A.SS)
					A.SS--;
			else if(Nxt.FF>A.FF && Nxt.SS==A.SS)
					A.FF++;
			else if(Nxt.FF<A.FF && Nxt.SS==A.SS)
					A.FF--;
		}
		cout<<A.FF<<" "<<A.SS<<endl;
	}
	return 0;
}
