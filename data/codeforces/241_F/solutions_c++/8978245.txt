#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cstdlib>
#define rep(i,l,r) for (int i=l; i<=r; ++i)
using namespace std;

int n,m,k,x[26],y[26],sx,sy,tx,ty;
char cmd[1234],mp[120][120];

void walk(int tx,int ty){
	while (sx!=tx || sy!=ty){
		if (k-=mp[sx][sy]-'0',k<0){
			printf("%d %d\n",sx,sy);
			exit(0);
		}
		sx+=(tx>sx)-(tx<sx);
		sy+=(ty>sy)-(ty<sy);
	}
}
int main(){
	scanf("%d%d%d",&n,&m,&k);
	rep(i,1,n) scanf("%s",mp[i]+1);
	rep(i,1,n) rep(j,1,m) if ('a'<=mp[i][j] && mp[i][j]<='z'){
		int id=mp[i][j]-'a';
		x[id]=i,y[id]=j,mp[i][j]='1';
	}
	scanf("%d%d%s%d%d",&sx,&sy,cmd,&tx,&ty);
	int cmd_n=strlen(cmd),cmd_q=0;
	for (;;){
		if (cmd_q==cmd_n){
			walk(tx,ty);
			break;
		} else{
			int id=cmd[cmd_q++]-'a';
			walk(x[id],y[id]);
		}
	}
	printf("%d %d\n",sx,sy);
}
