#include<cstdio>
#include<cstring>
using namespace std;

const int N = 105;
const int M = 1005;

int n, m, k, x, y;
int ti[N][N];
int X[N], Y[N], S[M];

int rd() { char c; while(c=getchar(), c<=' '); return c; }
void rnd(int *S)
{
	char c; while(c=getchar(), c<=' ');
	S[ S[0]=1 ] = c - 'a';
	while(c=getchar(), c>='a') S[++S[0]] = c - 'a';
}

int main()
{
	scanf("%d%d%d", &n, &m, &k);
	for(int i=1; i<=n; i++) for(int j=1; j<=m; j++)
	{
		ti[i][j] = rd();
		if(ti[i][j] >= 'a') X[ti[i][j]-'a'] = i, Y[ti[i][j]-'a'] = j, ti[i][j] = 1;
		else ti[i][j] -= '0';
	}
	scanf("%d%d", &x, &y), rnd(S);
	S[++S[0]] = 27, scanf("%d%d", &X[27], &Y[27]);
	
	for(int i=1; i<=S[0] && k>0; i++)
	{
		while(!(x==X[S[i]] && y==Y[S[i]]))
		{
			if((k-=ti[x][y])<0) break;
			x += (X[S[i]] > x) - (x > X[S[i]]);
			y += (Y[S[i]] > y) - (y > Y[S[i]]);
		}
	}
	printf("%d %d\n", x, y);
	
	return 0;
}
