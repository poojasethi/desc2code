#include <cstdio>
#include <cstring>
#include <queue>
#include <map>
const int N = 100 + 9,d[4][2] = {{0,1},{0,-1},{1,0},{-1,0}};
int n,m,K,cnt,time;
char map[N][N],S[2000];
struct point
{
	int x,y;
	point(const int _x,const int _y):
		x(_x),y(_y){}
	point():x(0),y(0){}
}pos[N],p[2000],s;
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("241F.in","r",stdin);
	freopen("241F.out","w",stdout);
	#endif
	time = 0;
	memset(map,'#',sizeof map);
	scanf("%d%d%d\n",&n,&m,&K);
	for (int i = 1; i <= n; ++i,scanf("\n"))
		for (int j = 1; j <= m; ++j) {
			scanf("%c",&map[i][j]);
			if (map[i][j] >= 'a' && map[i][j] <= 'z') {
				pos[map[i][j] - 'a'] = point(i,j);
				map[i][j] = '1';
			}
		}
	scanf("%d%d%s",&p[1].x,&p[1].y,S);
	cnt = 1;
	for (char *c = S; *c; ++c)
		p[++cnt] = pos[*c - 'a'];
	++cnt;
	scanf("%d%d",&p[cnt].x,&p[cnt].y);
	s = p[1];
	for (int i = 2; i <= cnt; ++i)
		while (!(s.x == p[i].x && s.y == p[i].y)) {
			time += map[s.x][s.y] - '0';
			if (time > K) {printf("%d %d\n",s.x,s.y);return 0;}
			if (s.x < p[i].x) ++s.x;
			if (s.x > p[i].x) --s.x;
			if (s.y < p[i].y) ++s.y;
			if (s.y > p[i].y) --s.y;
		}
	printf("%d %d\n",p[cnt].x,p[cnt].y);
}
