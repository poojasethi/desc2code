#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <algorithm>
#include <cstring>
#include <cmath>
#include <string>
using namespace std;
const int Maxn=105;
int m,n,k,len;
char s[Maxn][Maxn],p[1005];
int px[30],py[30],curx,cury;

int main() {
	scanf("%d%d%d",&m,&n,&k);
	for (int i=1;i<=m;i++) {
		scanf(" %s",s[i]+1);
		for (int j=1;j<=n;j++)
			if (s[i][j]>='a'&&s[i][j]<='z')
				px[s[i][j]-'a']=i,py[s[i][j]-'a']=j,s[i][j]=1;
			else s[i][j]-='0';
	}
	scanf("%d%d %s%d%d",&curx,&cury,p,px+26,py+26);
	len=strlen(p);
	for (int i=0;i<len;i++) p[i]-='a'; p[len]=26;
	for (int i=0;i<=len&&k>0;i++)
		while ((curx!=px[p[i]]||cury!=py[p[i]])&&(k>=s[curx][cury])) {
			k-=s[curx][cury];
			if (curx<px[p[i]]) ++curx;
			if (curx>px[p[i]]) --curx;
			if (cury<py[p[i]]) ++cury;
			if (cury>py[p[i]]) --cury;
		}
	cout<<curx<<' '<<cury<<endl;
	return 0;
}
