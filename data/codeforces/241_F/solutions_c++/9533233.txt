#include<bits/stdc++.h>
using namespace std;
const int dx[]={0,0,-1,1},dy[]={-1,1,0,0};
char mp[101][101],s[101];
int n,m,k,cost[101][101],px[100001],py[100001],len;
int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=n;i++)
	{
		scanf("%s",mp[i]+1);
		for(int j=1;j<=m;j++)
		if(mp[i][j]>='a'&&mp[i][j]<='z')
		cost[i][j]=1;
		else if(mp[i][j]>='1'&&mp[i][j]<='9')
		cost[i][j]=mp[i][j]-'0';
	}
	scanf("%d%d%s",&px[0],&py[0],s+1);
	len=strlen(s+1);
	scanf("%d%d",&px[len+1],&py[len+1]);
	for(int t=1;t<=len;t++)
		for(int i=1;i<=n;i++)
			for(int j=1;j<=m;j++)
				if(mp[i][j]==s[t])
					px[t]=i,py[t]=j;
	int tx=px[0],ty=py[0];
	for(int i=0,dir;i<=len;i++)
	{
		if(px[i]==px[i+1])dir=py[i+1]>py[i]?1:0;
		else dir=px[i+1]>px[i]?3:2;
		while(tx!=px[i+1]||ty!=py[i+1])
		{
			if(k<cost[tx][ty])goto end;
			k-=cost[tx][ty];
			tx+=dx[dir];ty+=dy[dir];
		}
	}
end:printf("%d %d\n",tx,ty);
}
