#include<iostream>
#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
#define LEN 1005
#define nMax 105
int mp[nMax][nMax],px[28],py[28];
char str[nMax][nMax],tour[LEN];


inline int sgn(int n){	return (n>0)-(n<0);	}
void doit(int &x1,int &y1,int x2,int y2,int &T)
{
	int	dx,dy;
	while(x1!=x2 || y1!=y2){
		dx = sgn(x2-x1), dy = sgn(y2-y1);
		if(T>=mp[x1][y1]){
			T -= mp[x1][y1];
			x1 += dx;	y1 += dy;
		}else	break;
	}

}

int main()
{
	int n,m,K,x1,x2,y1,y2,i,j,typ,sz;
	scanf("%d%d%d",&n,&m,&K);
	for(i=0; i<n; i++)	scanf("%s",str[i]);
	for(i=0; i<n; i++)
		for(j=0; j<m; j++)
			if(str[i][j] >= 'a' && str[i][j] <= 'z'){
				typ = str[i][j] - 'a' + 1;
				px[typ] = i;	py[typ] = j;
				mp[i][j] = 1;
			}else	mp[i][j] = str[i][j] - '0';
	scanf("%d%d%s%d%d",&x1,&y1,tour,&x2,&y2);
	x1--;	y1--;	x2--;	y2--;
	for(i=0,sz=strlen(tour); i<sz; i++)		doit(x1,y1,px[tour[i] - 'a' + 1],py[tour[i] - 'a' + 1],K);

	doit(x1,y1,x2,y2,K);
	printf("%d %d\n",x1+1,y1+1);
	return 0;
}
