import sys

#sys.stdin = open('in', 'r')

n = input()
a = map(int, raw_input().split())

if sum(a) == 0:
  print 0
  exit()

res, i = 0, 0
while i < n:
  if a[i] == 1:
    res += 1
    i += 1
  else:
    res += 1
    while i < n and a[i] == 0:
      i += 1

res -= (a[0] == 0) + (a[n - 1] == 0)
print res
