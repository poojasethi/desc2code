#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
int main()
{
	int a,b,c,d,e;double d1,d2,d3;
	scanf("%d%d%d%d%d",&a,&b,&c,&d,&e);
	d1=(double)c*a*b/(b-a);
	d2=2.0*a/(b-a)+1;
	d3=d/(1-d2)*a*b/(b-a);
	d3=(e-d3)/(d1-d3);
	printf("%d\n",max(0,(int)(log(d3)/log(d2)+0.999999)));
}
