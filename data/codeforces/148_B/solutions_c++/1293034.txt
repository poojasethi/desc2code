#include <iostream>

using namespace std;

int main() {
	int vp,vd,t,f,c;
	cin >> vp >> vd >> t >> f >> c;
	if (vd <= vp) { cout << "0\n"; return 0; }
	double a = vp*t;
	a += a/(vd-vp)*vp;
	int res=0;
	while (a < c - 1e-9) {
		res++;
		a += (a/vd+f)*vp;
		a += a/(vd-vp)*vp;
	}
	cout << res << endl;
	return 0;
}
