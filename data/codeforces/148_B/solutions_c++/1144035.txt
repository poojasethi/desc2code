#include <stdio.h>
double vp,vd,t,f,c,Xp;
int b=0;
int main(){
  scanf("%lf%lf%lf%lf%lf", &vp, &vd, &t, &f, &c);
  if(vd<vp){ puts("0"); return 0; }
  Xp=vp*t;
  Xp=Xp+vp*(Xp/((vd-vp)));
  while(Xp<=c-1e-9){
    Xp=Xp+vp*(Xp/vd)+f*vp;
    Xp=Xp+vp*(Xp/((vd-vp)));
    b++;
  }
  
  printf("%d\n", b);
  return 0;
}
