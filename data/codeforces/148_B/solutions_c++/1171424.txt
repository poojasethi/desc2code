#include <stdio.h>
int main()
{
	double v,d,t,f,c,s;
	scanf("%lf%lf%lf%lf%lf",&v,&d,&t,&f,&c);if(d<=v)printf("0\n");
	else
	{
		int i;
		for(i=1;i<=1000;i++)
		{
			s=v*t/(d-v);
			if(s>=((c/v)-t))
			{
				printf("%d\n",i-1);
				break;
			}
			t=t+2*s+f;
		}
	}
	return 0;
}
