#include <stdio.h>
int main()
{
	int vp,vd,t,f,c;
	float d;
	int i;

	while(scanf("%d %d %d %d %d",&vp,&vd,&t,&f,&c)!=EOF)
	{
		i=0;
		if(vd>vp)
			d=1.0*t*vp*vd/(vd-vp);
		else
			d=c;
		while(d<c)
		{
			d=(d*vd+d*vp+f*vp*vd)/(vd-vp);
			++i;
		}
		printf("%d\n",i);
	}
	return 0;
}
