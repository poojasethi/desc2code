#include<stdio.h>
#include<stdlib.h>

int main()
{
	int vp,vd,t,f,c;
	scanf("%d %d %d %d %d",&vp,&vd,&t,&f,&c);
	int ans = 0;
	double l = t * vp;
	if(vp>=vd) ans = 0;
	else
	 while(l<c)
	 {
		l += l/(vd-vp)*vp;
		if(l<c) ans++;
		l += (l/vd+f)*vp;
	 }
	printf("%d",ans);
	return 0;
}
