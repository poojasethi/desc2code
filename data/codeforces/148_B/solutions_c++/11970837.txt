#include<stdio.h>
int main()
{
	int vp,vd,t,f,c;
	scanf("%d%d%d%d%d",&vp,&vd,&t,&f,&c);
	double s=vp*t;
	int bijou=0;
	if(vd>vp)
	{
		while(s<c){
			double t0=s/(vd-vp);//?
			s+=t0*vp;
			if(s>=c)	break;
			bijou++;
			t0+=f;
			s+=t0*vp;
		}
	}
	printf("%d",bijou);
	return 0;
}
