eps = 1e-9

vp, vd, t, f, c = [int(raw_input()) for i in range(5)]
cur = vp * t * 1.0
res = 0
while True:
    if vd == vp:
        break
    time = cur / (vd - vp)
    if time < 0:
        break
    cur += vp * time
    if cur + eps > c:
        break
    res += 1
    cur += vp * f + (cur / vd) * vp
print res
