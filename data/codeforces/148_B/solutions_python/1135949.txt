import sys
vp,vd,t,f,c = [int(sys.stdin.readline().strip()) for i in range(5)]
def solve(x):
    #print "x", x
    if c-x < 1e-9:
        return 0
    t1 = x/float(vd-vp)
    x2 = t1*vd
    if c-x2 < 1e-9:
        return 0
    #print "x2", x2
    t2 = x2/float(vd) + f
    return solve(x2 + vp*t2) + 1
