vp = int(input())
vd = int(input())
t = int(input())
f = int(input())
c = int(input())

if vd <= vp :
    print 0
else :
    total = vp * (vd * t) / (vd - vp)
    count = 0

    while total < c :
        count += 1
        t = total / float(vd) + f
        total += vp * t
        t = total / float(vd - vp)
        total += vp * t
    print count
