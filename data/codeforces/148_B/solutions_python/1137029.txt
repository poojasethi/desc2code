vp = int(raw_input())
vd = int(raw_input())
t = int(raw_input())
f = int(raw_input())
c = int(raw_input())

if vd <= vp:
    ans = 0
else:
    ans = 0
    s = vp * t
    while s < c:
        x = (s + 0.0) / (vd - vp)
        s += vp * x
        if s < c:
            ans += 1
            s += x * vp
            s += f * vp
print ans
