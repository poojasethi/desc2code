def readA():
    return int(raw_input())

vp = readA()
vd = readA()
t = readA()
f = readA()
c = readA()

n = 0
i = t * vp * 1.0;
if vp < vd:
    while i < c:
        ni = i + i * vp / (vd - vp)
        if (ni < c):
            n += 1
            i = ni + vp * (ni / vd + f)
        else:
            i = ni
print n;
