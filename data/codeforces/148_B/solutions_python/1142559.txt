import sys

vp = input()
vd = input()
t = input()
f = input()
c = input()

if vd<=vp:
    print 0
    sys.exit(0)

def catchTime(v, d):
    return float(d)/v

s = t*vp
tt = catchTime(vd-vp, s)
s += tt*vp
count = 0
while s<c:
    count += 1
    dt = float(s)/vd
    dt += f
    s += dt*vp
    tt = catchTime(vd-vp, s)
    s += tt*vp

print count
