from sys import stdin
fil = stdin
vp = float(fil.readline())
vd = float(fil.readline())
t = float(fil.readline())
f = float(fil.readline())
c = float(fil.readline())
dist = t * vp
minim = 0
if vp < vd:
    while dist < c:
        time = dist / (vd - vp)
        if time * vp + dist >= c:
            break
        time += f + time
        minim += 1
        dist += time * vp
print minim
