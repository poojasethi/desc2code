import sys

def main():
    vp, vd, t, f, c = map(int, sys.stdin.readlines())
    cp = 0
    if vp >=vd:
        return 0
    count = 0
    while cp < c:
        cp += t*vp
        td = float(cp)/(vd-vp)
        if cp + td*vp >=c:
            return count
        count += 1
        t = 2.0*td + f
    return count

if __name__ == "__main__":
    print main()
