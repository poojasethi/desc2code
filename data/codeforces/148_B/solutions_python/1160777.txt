import sys
import traceback
  
def solve():
  vp = float(sys.stdin.readline())
  vd = float(sys.stdin.readline())
  t = float(sys.stdin.readline())
  f = float(sys.stdin.readline())
  c = float(sys.stdin.readline())
  if not (vd > vp): return 0
  bijou = 0
  dv = vd - vp
  while True:
    dt = t * vp / dv
    tp = t + dt
    if c <= (tp * vp): return bijou
    bijou += 1
    t = tp + dt + f

if __name__=="__main__":
  print solve()
