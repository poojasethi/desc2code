def main():
	vp=int(raw_input())
	vd=int(raw_input())
	t=int(raw_input())
	f=int(raw_input())
	c=int(raw_input())
	if(vd-vp<=0):
		print 0
		return
	bij=0
	ploc=(vp*t)
	while(ploc<c):
		tchase=float(ploc)/(vd-vp)
		ploc+=vp*tchase
		if(ploc>=c):
			break
		bij+=1
		ploc+=(tchase+f)*vp
	print bij
if __name__=="__main__":
	main()
