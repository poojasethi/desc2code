vp = input()
vd = input()
t = input()
f = input()
c = input()

count = 0
CET = 1.0e-8
length = t * vp * 1.0

while (vp < vd and length < c - CET):
    t1 = length / (vd - vp)
    length += t1 * vp
    if length > c - CET:
        break
    count += 1
    t1 = f + length / vd
    length += t1 * vp

print count
