vp = int(raw_input())
vd = int(raw_input())
t = int(raw_input())
f = int(raw_input())
c = int(raw_input())
cnt = 0
pp = t * vp * 1.0
while vp < vd and pp < c - 1.0e-6:
    t1 = pp / (vd - vp)
    pp += t1 * vp
    if pp > c - 1.0e-6: break
    cnt += 1
    t1 = f + pp / vd
    pp += t1 * vp
print cnt
