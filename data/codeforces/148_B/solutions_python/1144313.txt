#!/usr/bin/env python

eps = 0.00001

if __name__ == "__main__":
    v_p = int(raw_input())
    v_d = int(raw_input())
    t = int(raw_input())
    f = int(raw_input())
    c = int(raw_input())
    if v_p >= v_d:
        print(0)
    else:
        p_dist = float(v_p * t)
        d_dist = 0.
        count = 0
        while c - p_dist >= eps:
            t_meet = p_dist / (v_d - v_p)
            p_dist += t_meet * v_p
            if c - p_dist < eps:
                break
            count += 1
            time = p_dist / v_d + f
            p_dist += time * v_p
        print(count)
