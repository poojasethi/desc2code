from fractions import Fraction
vp = int(raw_input())
vd = int(raw_input())
t = int(raw_input())
f = int(raw_input())
c = Fraction(raw_input())

vdiff = vd-vp
if vdiff<=0:
  print 0
  exit()

wait = t
dist = Fraction(0)
pos = Fraction(0)
ans = 0

while True:
  time = (vd*wait+dist)/vdiff
  move = vp*time
  pos = pos + move
  if pos>=c:
    print ans
    break
  ans += 1
  wait = f
  dist = pos*2
