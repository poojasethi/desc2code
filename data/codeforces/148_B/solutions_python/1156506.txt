def solve(vp, vd, t, f, c):
    if vd <= vp:
        return 0
    ppos = float(t*vp)
    ppos += vp*(ppos/(vd-vp))
    res = 0
    while ppos < c:
        res += 1
        ppos += vp * (f + ppos/vd)
        ppos += vp*(ppos/(vd-vp))
    return res
