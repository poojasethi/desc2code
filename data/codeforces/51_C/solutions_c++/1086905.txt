#include<cstdio>
#include<algorithm>
using namespace std;
int f[200005],n;
bool check(double d)
{
    int i=0;
    int pos=f[i];
    for(int j=0;j<3;j++)
    {
        int pos=f[i];
        for(;i<n;i++) if (f[i]-pos>2*d) break;
        if (i==n) return 1;
    }
    return 0;
}
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",f+i);
    sort(f,f+n);
    double l=0,r=f[n-1]-f[0];
    for(int i=0;i<100;i++)
    {
        double m=(l+r)/2;
        if (check(m))
        {
            r=m;
        }
        else
        {
            l=m;
        }
    }
    printf("%.6f\n",r);
    int i=0;
    for(int j=0;j<3;j++)
    {
        if (i==n)
        {
            printf("%.6f ",f[0]);
            continue;
        }
        int pos=f[i];
        printf("%.6f ",pos+r);
        for(;i<n;i++) if (f[i]-pos>2*r) break;
    }
    puts("");
}
