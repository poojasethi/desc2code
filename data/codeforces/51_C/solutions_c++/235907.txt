#include <stdio.h>
#include <cmath>
#include <string>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<(li)to;i++)
#define pb(a)		push_back(a)
#define sz(v)		((li)v.size())
int main(){
	int n;
	double x[200001],eps=1e-8;
	scanf("%d",&n);
	rep(i,n)	scanf("%lf",&x[i]);
	sort(x,x+n);
	double low=0.0,high=1e10;
	int i,j,k;
	while(true){
		double mid=(low+high)/2.0;
		if(low>high-eps)	mid=high;
		for(i=0;i<n;i++){
			if(x[i]-x[0]>mid)	break;
		}
		for(j=min(i,n-1);j<n;j++){
			if(x[j]-x[min(i,n-1)]>mid)	break;
		}
		for(k=min(j,n-1);k<n;k++){
			if(x[k]-x[min(j,n-1)]>mid)	break;
		}
		if(low>high-eps)	break;
		if(k==n)	high=mid;
		else		low=mid;
	}
	printf("%.6f\n",high/2.0);
	printf("%.6f %.6f %.6f\n",x[0]+(x[i-1]-x[0])/2.0,x[min(i,n-1)]+(x[j-1]-x[min(i,n-1)])/2.0,x[min(j,n-1)]+(x[n-1]-x[min(j,n-1)])/2.0);
}
