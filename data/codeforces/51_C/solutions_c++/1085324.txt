#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

const double veryfew=1e-6;
const int maxn=300000;

double p[4],z[maxn];

int n;

bool check(double now)
{
	int ll=10000000,rr=-1;
	p[1]=z[0]+now;
	p[3]=z[n-1]-now;
	double l=z[0]+2*now,r=z[n-1]-2*now;
	int m=0;
	for (int a=0;a<n;a++)
		if ((z[a]>l+veryfew) && (z[a]+veryfew<r))
		{
			m++;
			ll=min(ll,a);
			rr=max(rr,a);
		}
	if (m==0) return true;
	if (z[rr]-z[ll]>2*now+veryfew) return false;
	p[2]=(z[rr]+z[ll])*0.5;
	return true;
}

int main()
{
	//freopen("d.in","r",stdin);
	//freopen("d.out","w",stdout);

	scanf("%d",&n);
	for (int a=0;a<n;a++)
		scanf("%lf",&z[a]);
	sort(z,z+n);
	double l=0,r=z[n-1]-z[0];
	for (int a=1;a<=100;a++)
	{
		double m=(l+r) * 0.5;
		if (check(m)) r=m;
		else l=m;
	}	
	double ans=(l+r) * 0.5;
	printf("%.6lf\n",ans);
	check(ans);
	printf("%.6lf %.6lf %.6lf\n",p[1],p[2],p[3]);

	return 0;
}
