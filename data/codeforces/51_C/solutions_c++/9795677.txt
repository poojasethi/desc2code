#include <cstdio>
#include <algorithm>
int n,a[200010],p,q;
double l,r,m,x;
int main(){
	scanf("%d",&n);
	for (int i = 1;i <= n;i++) scanf("%d",&a[i]);
	std::sort(a + 1,a + n + 1);
	l = 0;
	r = (a[n] - a[1] + 1) / 6.0;
	p = 1;
	q = n;
	while (r - l > 1e-7){
		m = (l + r) / 2;
		if (a[p] - a[1] > 2 * m){
			while (a[p] - a[1] > 2 * m) p--;
			p++;
		}else{
			while (p < n && a[p + 1] - a[1] <= 2 * m) p++;
			p++;
		}
		if (a[n] - a[q] > 2 * m){
			while (a[n] - a[q] > 2 * m) q++;
			q--;
		}else{
			while (q > 1 && a[n] - a[q - 1] <= 2 * m) q--;
			q--;
		}
		if (p < q && a[q] - a[p] > 2 * m) l = m;else r = m;
	}
	m = (l + r) / 2;
	p = 1;
	x = a[1] + m * 2;
	while (p < n && a[p + 1] < x + 1e-4) p++;
	x = a[n] - m * 2;
	q = n;
	while (q > 1 && a[q - 1] > x - 1e-4) q--;
	p++;
	q--;
	if (p <= n && q >= p) x = (a[p] * 1.0 + a[q]) / 2;else x = 0;
	printf("%.6f\n%.6f %.6f %.6f\n",m,a[1] + m,x,a[n] - m);
}
