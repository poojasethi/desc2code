#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
using namespace std;
const int mm=2e5+9;
const double oo=1e49;
double dis[mm];int n;
double a,b,c,len=oo;
void ok(int x,int y)
{ if(y<0)y=0;
  double z;
  z=dis[x]-dis[0];
  z=max(dis[y]-dis[min(x+1,n-1)],z);
  z=max(dis[n-1]-dis[min(y+1,n-1)],z);
  if(z<len)
  {
    len=z;a=dis[x]+dis[0];b=dis[y]+dis[min(x+1,n-1)];c=dis[n-1]+dis[min(y+1,n-1)];
  }
}
int main()
{

  while(cin>>n)
  { len=oo;
    for(int i=0;i<n;i++)
      cin>>dis[i];
    sort(dis,dis+n);
    for(int i=0,j=0;i<n;i++)///j只跑一次，枚举分段
    { 
      while(j<n-1&&(dis[n-1]-dis[j+1]>dis[j]-dis[i+1]))
        ++j;
      ok(i,j);ok(i,j-1);
    }
    len/=2.0;a/=2.0;b/=2.0;c/=2.0;
   printf("%.6lf\n%.6lf %.6lf %.6lf\n",len,a,b,c);
  }
}
