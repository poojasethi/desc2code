#include <string>
#include <vector>
#include <algorithm>
#include <iostream>
#include <stdio.h>

using namespace std;

long long H[300100];


bool can(long long d, int n) {

	int last = 0;
	int cnt = 1;

	for(int i=0 ; i<n ; i++) {
		if(H[i]-H[last] > d ){
			cnt++;
			last = i;
		}
	}
	return cnt<=3;
}

int main() {

	int n;
	cin >> n;
	for(int i=0 ; i<n ; i++) cin >> H[i];
	
	sort(H, H+n);
	
	long long l=0, r=2000000001LL;
	while(l<r) {
		long long mid = (l+r)/2LL;
		if( can(mid, n) ) r = mid;
		else l = mid+1;
	}

	printf("%.6lf\n", (double)l/2.);
		
	int last = 0;
	int cnt = 0;

	for(int i=0 ; i<n ; i++) {
		if(H[i]-H[last] > l ){
			printf("%.6lf ", ((double)H[i-1]+H[last])/2.);
			cnt++;
			last = i;
		}
	}

	while(cnt < 3) {
		printf("%.6lf ", ((double)H[n-1]+H[last])/2.);
		cnt++;
	}
	printf("\n");

	return 0;
}
