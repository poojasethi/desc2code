int n,a[200000]; double t[3],ans[3];
bool ok(double d){
  int cur=0;
  rep(i,3){
    t[i]=a[cur]+d;
    int pos=lower_bound(a,a+n,a[cur]+2*d+1e-9)-a;
    cur=pos;
    if(cur>=n)break;
  }
  if(cur>=n)rep(i,3)ans[i]=t[i];
  return cur>=n;
}
void run(){

  rep(i,n)cin>>a[i];
