#include <iostream>
#include <algorithm>
#include <iomanip>
#define int long long

using namespace std;

const int E = 1000ll*1000,MAXN=400000,maxn=1000ll*1000*1100;
int n;
int x[MAXN];
int check1[3];
int check2[3];

bool check(int d){
	int y=0;
	for(int i=0; i<3 ; i++){
		if(y<n){
			check1[i]=x[y]+d;
			int q=check1[i]+d;
			y=upper_bound(x,x+n,q)-x;
		}
	}
	if(y>=n){
		for(int i=0; i<3; i++)
			check2[i]=check1[i];
		return 1;
	}
	else
		return 0;
}

main(){
	cout<<fixed<<setprecision(6);
	cin>>n;
	for(int i=0 ; i<n; i++)
		cin>>x[i],
		x[i]*=E;
	sort(x,x+n);
	int s=-1,e=E*maxn-1; 
	while(e-s>1){
		int m=(e+s)/2+(e+s)%2;
		if(!check(m))
			s=m;
		else
			e=m;
	}	
	cout<<(double(e)/(double)E)<<endl;
	for(int i=0 ; i<3 ; i++)
		cout<<(double)check2[i]/(double)E<<endl; 
	//system("pause");
	return 0;
}
