#include <bits/stdc++.h>
#define ld  double
using namespace std;
const int MAXN=1000*201;
ld a[MAXN];
int n;
ld b[MAXN];
int check(ld x)
{
    b[0]=a[0]+x;
    int j=0;
 //  cout<<b[0]<<endl;
    for(int i=1;i<n;i++)
    {
        if(a[i]>b[j]+x)
        {
            b[j+1]=a[i]+x;
           // cout<<b[j-1]<<endl;
            j++;
        }
            if(j>3)
            return 0;
    }
    if(j>2)
        return 0;
    return 1;
}
main()
{
    ios::sync_with_stdio(0);
    cin.tie();
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    ld l=0,r=(1e+8)*3,mid;
   // cout<<check(0)<<endl;
    for(int i=0;i<100;i++)
    {
        for(int j=0;j<5;j++)
            b[j]=0;
        mid=(r+l)/2;
        if(check(mid))
            r=mid;
        else
            l=mid;
    }
   cout<<fixed<<setprecision(10);
    check(r);
    cout<<r<<'\n';
    for(int i=0;i<3;i++)
        cout<<b[i]<<' ';
}
