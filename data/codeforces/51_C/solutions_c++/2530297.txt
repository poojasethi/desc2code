#include<iostream>
#include<algorithm>
#include<iomanip>
using namespace std;
int ted;
int a[200000+5];
const double eps=1e-15;
int f(double adad){
	int po1=upper_bound(a,a+ted,a[0]+(2*adad))-a;
	if(po1==ted)
		return 1;
	int po2=upper_bound(a,a+ted,a[po1]+(2*adad))-a;
	if(po2==ted)
		return 1;
	int po3=upper_bound(a,a+ted,a[po2]+(2*adad))-a;
	if(po3==ted)
		return 1;
	return 0;
}
double maxi=double(1000*1000*1000)/3;
double mini=0;
int main(){
	cin>>ted;
	for(int i=0;i<ted;i++){
		cin>>a[i];
	}
	sort(a,a+ted);
	for(int i=0;i<1000;i++){
		int bo=f((maxi+mini)/2);
		if(bo==1){
			maxi=(mini+maxi)/2;
		}
		if(bo==0){
			mini=(maxi+mini)/2;
		}
	}
	cout<<fixed<<setprecision(6)<<maxi<<endl;
	cout<<(a[0]+maxi)<<" ";
	int u=upper_bound(a,a+ted,a[0]+(2*maxi)+eps)-a;
	cout<<fixed<<setprecision(6)<<a[u]+maxi<<" ";
	cout<<fixed<<setprecision(6)<<a[(upper_bound(a,a+ted,a[u]+(2*maxi)+eps)-a)]+maxi<<endl;
}
