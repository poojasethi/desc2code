#include<iostream>
#include<algorithm>
#include<iomanip>
using namespace std;
int h[1001000];
int main(){
   cout<<fixed<<setprecision(6); 
   int n,i,j,help,helpp,chek=0;
   double x,y,d,ans;
   cin>>n;
   for(i=0;i<n;i++) cin>>h[i];
   sort(h,h+n);
   x=h[0],y=h[n-1];
   while(y-x>0.00000002||chek==0){
       d=(x+y);
       d/=2;
       d-=h[0];
       help=upper_bound(h,h+n,2*d+h[0])-h;
       if(h[help]==2*d+h[0]) help++;
       helpp=upper_bound(h,h+n,2*d+h[help])-h;
       if(h[helpp]-h[help]==2*d) helpp++;
       if(h[n-1]-h[helpp]<=2*d||helpp>=n) chek=1;
       else chek=0;
       (chek==0)?(x=d+h[0]):(y=d+h[0]);  
   }
   cout<<d<<endl;
   h[n]=h[n-1]; 
   cout<<h[0]+d<<" "<<h[help]+d<<" "<<h[helpp]+d<<endl;
   return 0; 
}
