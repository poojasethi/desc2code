#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int mm=2e5+5;
const int oo=1e9;
int d[mm],n;
double len=oo,a,b,c;
void jud(int x,int y)
{
    if(y<0)
        y=0;
    int z;
    z=d[x]-d[0];
    z=max(d[y]-d[min(x+1,n-1)],z);
    z=max(d[n-1]-d[min(y+1,n-1)],z);
    if(z<len)
    {
        len=z;
        a=d[x]+d[0];
        b=d[y]+d[min(x+1,n-1)];
        c=d[n-1]+d[min(y+1,n-1)];
    }
}
int main()
{
    int i,j;
    while(scanf("%d",&n)!=EOF)
    {
        len=oo;
        for(i=0;i<n;i++)
            scanf("%d",&d[i]);
        sort(d,d+n);
        for(i=0,j=0;i<n;i++)
        {
            while(j<n-1&&d[n-1]-d[j+1]>d[j]-d[i+1])
                ++j;
            jud(i,j);
            jud(i,j-1);
        }
        printf("%.6lf\n%.6lf %.6lf %.6lf",len/2.0,a/2.0,b/2.0,c/2.0);
    }
    return 0;
}
