#include<cstdio>
#include<algorithm>
using namespace std;
const int N=200002;
int a[N];
int main(){
	int n,l,r,m,i,p;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++)scanf("%d",a+i);
		sort(a,a+n);
		l=0,r=a[n-1]-a[0];
		while(l<r){
			m=(l+r)>>1;
			p=0;
			for(i=0;i<3;i++)p=upper_bound(a+p,a+n,a[p]+m)-a;
			if(p<n)l=m+1;
			else r=m;
		}
		printf("%d.%c00000\n",r/2,r&1?'5':'0');
		p=0;
		for(i=0;i<3;i++){
			printf("%d.%c00000 ",(p<n?a[p]:0)+r/2,r&1?'5':'0');
			p=upper_bound(a+p,a+n,a[p]+r)-a;
		}
	}
	return 0;
}
