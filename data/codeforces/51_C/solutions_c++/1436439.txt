#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int n,q[200005];


int  ok(double t)
{
     int k=0;
    for(int i=0;i<n;)
    {
           k++;
           double j=q[i]+2*t;
           for(;i<n&&(double)q[i]<=j;i++);
    }
    if(k<=3) return true;
    return false;
}

int main()
{
    cin>>n;
    for(int i=0;i<n;i++) cin>>q[i];

    sort(q,q+n);
    double l=0,r=1e10,mid;
    for(int i=0;i<100;i++)
    {
        mid=(l+r)*0.5;
        if(ok(mid)) r=mid;
        else l=mid;
    }
    
    printf("%.6lf\n",r);

     int s=0;
     double t[4];
    for(int i=0;i<n;)
    {
         t[s++]=q[i]+r;
        double j=q[i]+2*r;
        for(;i<n&&(double)q[i]<=j;++i) ;
    }
    while(s<3) t[s++]=q[n-1];
    printf("%.6lf %.6lf %.6lf\n",t[0],t[1],t[2]);

}
