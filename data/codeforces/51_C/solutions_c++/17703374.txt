#include "cstdio"
#include "algorithm"
using namespace std;
int v[200000+5];
int n;
bool c(double d)
{
	int i,l=v[0],cnt=3;
	for(i=1;i<n;i++)
	{
		if(v[i]<=l+2*d) continue;
		cnt--;
		l=v[i];
		if(cnt<=0) break;
	}
	if(i<n) return false;
	else return true;
}

int main()
{
	int i;
	while(scanf("%d",&n)!=EOF)
	{
		for(i=0;i<n;i++)
		{
			scanf("%d",&v[i]);
		}
		sort(v,v+n);
		double l=0,r=2e9,mid;
		for(i=0;i<100;i++)
		{
			mid=(l+r)/2;
			if(c(mid)){
				r=mid;
			}else{
				l=mid;
			}
		}
		if(r<=1e-6&&r>=-1e-6){
			printf("0\n");
		}else{
			printf("%f\n",r);
		}	
		int cnt=2;
		double pos=v[0]+r;
		printf("%f",pos);
		l=v[0];
		for(i=1;i<n;i++)
		{
			if(v[i]<=l+2*r+1e-6) continue;
			l=v[i];
			pos=l+r;
			printf(" %f",pos);
			cnt--;
		}	
		while(cnt--){
			printf(" %f",pos);
		}
		printf("\n");
	}
	return 0;
}
