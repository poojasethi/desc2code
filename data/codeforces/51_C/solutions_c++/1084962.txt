#include <cstdio>
#include <algorithm>
using namespace std;
const int N_MAX = 222222;
double pos[N_MAX];
int x[N_MAX];
int n;
int main() {
	//freopen("t.in", "r", stdin);
	scanf("%d", &n);
	for (int i = 1; i <= n; i ++)
		scanf("%d", &x[i]);
	sort(x + 1, x + n + 1);

	int lb = -1, rb = 2000000020;
	for (int mid = (lb + rb) / 2; lb < rb - 1;
			mid = (lb + rb) / 2) {
		int cnt = 0;
		int cur = 1;
		while (cur <= n) {
			cnt ++;
			int i = cur;
			for (; i < n && x[i + 1] - x[cur] <= mid; i ++);
			pos[cnt] = x[cur] + (double)(mid) / 2;
			cur = i + 1;
		}
		if (cnt <= 3)
			rb = mid;
		else
			lb = mid;
	}
	int cnt = 0;
	int cur = 1;
	while (cur <= n) {
		cnt ++;
		int i = cur;
		for (; i < n && x[i + 1] - x[cur] <= rb; i ++);
		pos[cnt] = x[cur] + (double)(rb) / 2;
		cur = i + 1;
	}
	printf("%.6lf\n", (double)rb / 2);
	for (int i = 1; i <= 3; i ++)
		printf("%.6lf ", pos[i]);
}
