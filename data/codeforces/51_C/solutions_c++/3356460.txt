#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>

using namespace std;

const int N = 2*10e5 + 5;
int a[N],n,ans[4];

bool judge(int m)
{
    ans[0] = a[0] + m;
    ans[1] = ans[2] = 2000000000;
    int i,t = 0;
    for(i=0;i<n;i++)
    {
        if(t>=3)
        return false;
        else if(ans[t] < a[i])
        ans[++t] = a[i] + m;
    }
    return t<3;
}

int main()
{
    int i;
    while(~scanf("%d",&n))
    {
        for(i=0;i<n;i++)
        scanf("%d",a+i);
        sort(a,a+n);
        int l = -1,r = a[n-1]+1;
        while(l+1<r)
        {
            int m = (l+r)>>1;
            if(judge(m))
            r = m;
            else
            l = m;
        }
        judge(r);
        printf("%.7lf\n%.7lf %.7lf %.7lf\n",r*0.5,ans[0]-r*0.5,ans[1]-r*0.5,ans[2]-r*0.5);
    }
    return 0;
}
