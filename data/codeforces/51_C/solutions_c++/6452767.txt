#include <iostream>
#include <iomanip>
#include <algorithm>
#include <cmath>
using namespace std;

int a[200001], n;
double p[5];

bool func(double val){
	double x = a[0]+val;
	int k = 0;
	p[0] = x;
	for (int i=1; i<n; i++)
		if(a[i] > x+val){
			x = a[i]+val;
			k++;
			if (k == 3)
				return false;
			p[k] = x;
		}
	return true;
}

int main(){	
	cin >> n;
	for (int i=0; i<n; i++)
		cin >> a[i];
	sort(a, a+n);
	double l = 0, r = 1000000000;
	for (int i=0; i<100; i++){
		if (func((r+l)/2) == true)
			r = (r+l)/2;
		else l = (r+l)/2;
	}
	cout << fixed << setprecision(6) << r << endl;
	cout << fixed << setprecision(6) << p[0] << " " << p[1] << " " << p[2] << endl;
	return 0;
}
