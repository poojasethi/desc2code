#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"

int n,i,j;
double mn,s1,s2,a,b,c;
int A[200001];

int main()
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
    scanf("%d",&A[i]); 
  sort(A,A+n);	
	if(n<=3)
	{
	  printf("%.6lf\n",0);
		for(i=0;i<3;i++)
		  printf("%.6lf ",(double)A[min(i,n-1)]);
		return 0;
	}	
	i=0;
	j=n-1;
	mn=A[n-1]-A[0]+1;
	while(i<j)
	{
		if(max(max(s1,s2),(double)A[j-1]-A[i+1])<mn)
		{
		  mn=max(max(s1,s2),(double)A[j-1]-A[i+1]);
			a=(A[0]+A[i])/2.;
			b=(A[i+1]+A[j-1])/2.;
			c=(A[j]+A[n-1])/2.;
		}
	  if(A[i+1]-A[0]<=A[n-1]-A[j-1])
		  s1=A[++i]-A[0];
		else
		  s2=A[n-1]-A[--j];
	}
	printf("%.6lf\n%.6lf %.6lf %.6lf ",mn/2,a,b,c);
	return 0;
}
