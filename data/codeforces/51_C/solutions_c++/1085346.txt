#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int a[200010];
bool cmp(int a,int b)
{
	return a<b;
}
int bs(double v,int n)
{
	int l=0,r=n-1;
	while(r-l>1)
	{
		int x=(r+l)/2;
		if(v>a[x])
			l=x;
		else
			r=x;
	}
	return l;
}
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n,cmp);
	double r=(a[n-1]-a[n]+1);
	double l=0;
	int mi,ma;
	while(r-l>1e-9)
	{
		double x=(r+l)/2;
		int i=bs(a[0]+2*x,n);
		if(a[i+1]>a[0]+2*x)
			mi=i+1;
		else
			mi=i+2;
		int j=bs(a[n-1]-2*x,n);
		if(a[j]<a[n-1]-2*x)
			ma=j;
		else
			ma=j-1;
		if(a[ma]-a[mi]<2*x)
			r=x;
		else
			l=x;
	}
	int i=bs(a[0]+2*r,n);
		if(a[i+1]>a[0]+2*r)
			mi=i+1;
		else
			mi=i+2;
		int j=bs(a[n-1]-2*r,n);
		if(a[j]<a[n-1]-2*r)
			ma=j;
		else
			ma=j-1;
	printf("%.6lf\n",r);
	printf("%.6lf %.6lf %.6lf\n",(a[0]+a[mi-1])/2.0,(a[mi]+a[ma])/2.0,(a[ma+1]+a[n-1])/2.0);
}
