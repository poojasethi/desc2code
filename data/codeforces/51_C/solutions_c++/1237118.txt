#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int a[200005];
int len=1e9;
int x, y, z, n;

void go(int i, int j) {
	if(j<0) j=0;
	int d = max(a[i]-a[0], max(a[j]-a[min(i+1,n-1)], a[n-1]-a[min(j+1,n-1)]));
	if(d<len) {
		len=d;
		x=a[0]+a[i], y=a[min(i+1,n-1)]+a[j], z=a[min(j+1,n-1)]+a[n-1];
	}
}

int main(void) {
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%d", &a[i]);
	sort(a, a+n);
	for(int i=0,j=0;i<n;i++) {
		while(j<n-1 && a[n-1]-a[j+1]>a[j]-a[i+1]) ++j;
		go(i,j);
		go(i,j-1);
	}
	printf("%.6f\n", len/2.);
	printf("%.6f %.6f %.6f\n", x/2., y/2.,z/2.);
	return 0;
}
