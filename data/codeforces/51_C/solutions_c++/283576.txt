#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#define MAXN 200000
#define MAXX 2000000000
using namespace std;

int X[MAXN];

int main()
{
	int n;
	scanf("%d", &n);
	for(int i = 0; i < n; i++)
	{
		scanf("%d", X+i);
		X[i]*=2;
	}
	sort(X, X+n);

	int l = 0, r = MAXX;
	while(l < r)
	{
		int m = (l+r)/2;
		int i, j = 0, c = 1;
		for(i = 0; i < n; i++)
		{
			if(X[i]-X[j] > 2*m)
			{
				c++;
				j = i;
			}
		}
		if(c <= 3)
			r = m;
		else
			l = m+1;
	}
	printf("%lf\n", (double)l/2);


	int j = 0, c = 1;
	for(int i = 0; i < n; i++)
	{
		if(X[i]-X[j] > 2*l)
		{
			if(c > 1)
				printf(" ");
			printf("%0.6lf", (double)(X[j]+l)/2);
			c++;
			j = i;
		}
	}
	while(c <= 3)
	{
		if(c > 1)
			printf(" ");
		printf("%.6lf", (double)(X[j]+l)/2);
		c++;
	}
	return 0;
}
