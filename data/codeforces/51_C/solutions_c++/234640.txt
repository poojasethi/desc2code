#include<stdio.h>
#include<cmath>
#include<algorithm>
#include<iostream>
using namespace std;
long long arr[200003];
double ans[10];
int main()
{
    long long n,i,j,c1,sp;
    long long lo,hi,mid;
    cin>>n;
    for (i=0;i<n;i++) cin>>arr[i];
    sort(arr,arr+n);
    lo=0;
    hi=2000000000;
    while (lo<hi)
    {
        mid=(lo+hi)/2;
        c1=0;
        for (i=0;i<n;i++)
        {
            for (j=i;j<n&&arr[j]<=arr[i]+mid;j++) ;
            i=j-1;
            c1++;
        }
        if (c1>3) lo=mid+1;
        else hi=mid;
    }
    printf("%.6lf\n",lo/2.0);
    sp=0;
    for (i=0;i<n;i++)
    {
        for (j=i;j<n&&arr[j]<=arr[i]+lo;j++) ;
        ans[sp++]=arr[i]+lo/2.0;
        i=j-1;
    }
    for (i=sp;i<3;i++) ans[i]=ans[i-1];
    printf("%.6lf %.6lf %.6lf\n",ans[0],ans[1],ans[2]);
    return 0;
}
