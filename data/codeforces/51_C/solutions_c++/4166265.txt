#include<cstdio>
#include<algorithm>
using namespace std;
int n;
int a[200009];

int main()
{
    while(scanf("%d",&n)!=EOF)
    {
        for(int i=0; i<n; i++)
        {
            scanf("%d",&a[i]);
        }
        sort(a,a+n);
        int l=0,r=a[n-1]-a[0];
        int p;
        while(l<r)
        {
            int mid=(r+l)/2;
            p=0;
            for(int i=0; i<3; i++)
                p=upper_bound(a+p,a+n,a[p]+mid)-a;
            if(p<n)
                l=mid+1;
            else r=mid;
        }
        printf("%d.%c00000\n",r/2,r&1?'5':'0');
        p=0;
        for(int i=0; i<3; i++)
        {
            printf("%d.%c00000 ",(p<n?a[p]:0)+r/2,r&1?'5':'0');
            p=upper_bound(a+p,a+n,a[p]+r)-a;
        }
    }
    return 0;
}
