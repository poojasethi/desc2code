#include<cstdio>
#include<algorithm>
using namespace std;
int n,x[200001];
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&x[i]), x[i] *= 2;
    sort(x,x+n);
    int f = 0,l = x[n-1];
    while(f <= l){
        int m = (f+l)/2,ct = 1,sv = x[0];
        for(int i=0;i<n && ct <= 3;i++){
            if(sv < x[i]-2*m)
                sv = x[i], ct++;
        }
        if(ct > 3) f = m+1;
        else l = m-1;
    }
    printf("%.6lf\n",0.5*f);
    int sv = x[0],ct = 1; 
    printf("%.6lf",0.5*(x[0]+f));
    for(int i=0;i<n;i++)
        if(sv < x[i]-2*f)
            sv = x[i],printf(" %.6lf",0.5*(sv+f)),ct++;
    while(ct < 3) 
        printf(" %.6lf",0.5*(sv+f)),ct++;
    return 0;
}
