int house[200020];
int n;
double a,b,c,len;
void _sort(int i,int j);
int main()
{

    while(cin>>n)
    {
       int i,y,x;
       for(i=0;i<n;i++)
       scanf("%d",&house[i]);
       sort(house,house+i);

       for(int i=0,j=0;i<n;i++)
       {

           while(j<n-1&&house[j]-house[i+1]<house[n-1]-house[j+1])
           j++;

       }
    printf("%.6lf\n%.6lf %.6lf %.6lf\n",len/2.0,a,b,c);

void _sort(int i,int j)
{
    if(j<0)
    j=0;
    int z=house[i]-house[0];
    z=max(house[j]-house[min(i+1,n-1)],z);
    z=max(house[n-1]-house[min(j+1,n-1)],z);
    if(z<len)
    {
        len=z;a=(house[i]+house[0])/2.0;b=(house[j]+house[min(i+1,n-1)])/2.0;c=(house[n-1]+house[min(j+1,n-1)])/2.0;
    }

}
