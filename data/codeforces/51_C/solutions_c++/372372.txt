#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;

int n,x[200005];
double mi,mid,ma,t[3],o[3];

bool check()
{
    t[0]=t[1]=t[2]=1000000000;
    for (int i=0,j=-1,k=0; i<n; i++)
    {
        if (x[i]>j)
        {
            j=x[i]+(int)(mid+mid+1e-8);
            if (k==3)
            {
                return false;
            }
            t[k++]=x[i]+mid;
        }
    }
    o[0]=t[0];
    o[1]=t[1];
    o[2]=t[2];
    return true;
}

int main()
{
    scanf("%d",&n);
    for (int i=0; i<n; i++)
    {
        scanf("%d",&x[i]);
    }
    sort(x,x+n);
    mi=0;
    ma=1000000000;
    while (mi+1e-7<ma)
    {
        mid=(mi+ma)/2;
        if (check())
        {
            ma=mid;
        }
        else
        {
            mi=mid;
        }
    }
    printf("%.6lf\n",mi);
    printf("%.6lf %.6lf %.6lf\n",o[0],o[1],o[2]);
    return 0;
}
