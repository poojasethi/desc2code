#include<bits/stdc++.h>
#define FI "station.i3"
#define FO "station.o3"
using namespace std;
const int NN = 200002;
int a[NN], n;

int main(){
    ios_base::sync_with_stdio(0), cin.tie(0);
    #ifndef ONLINE_JUDGE
    ifstream cin(FI);
    ofstream cout(FO);
    #endif // ONLINE_JUDGE
    cin >> n;
    for(int i = 0; i < n; i++) cin >> a[i];
	sort(a,a+n);
	int l = 0, r = a[n-1] - a[0], mid, pos;
    while(l < r){
        mid = (l + r) >> 1;
		pos = 0;
		for(int i = 0; i < 3; i++) pos = upper_bound(a + pos, a + n, a[pos] + mid) - a;
        if(pos < n) l = mid + 1;
		else r = mid;
	}
	cout << fixed << setprecision(6)<< (double)r/2 << endl;
	pos = 0;
	for(int i = 0; i < 3; i++){
        cout << fixed << setprecision(6) << (pos < n ? a[pos] : 0) + (double)r/2 << ' ';
		pos = upper_bound(a + pos, a + n, a[pos] + r) - a;
    }
	return 0;
}
