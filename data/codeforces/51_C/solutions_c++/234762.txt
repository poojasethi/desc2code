using namespace std;

typedef vector<long long> VI;
typedef vector<VI> VVI;

int main () {
//	freopen("input.txt","r",stdin);
	int n;
	cin >> n;
	VI a(n);
	for (int i=0;i<n;i++) cin >> a[i];
	sort(a.begin(),a.end());
	long long l=0,r=1000000000000;
	while (l<r)
	{
		long long m=(l+r)/2;
		int j=0;
		for (int i=0;i<3;i++)
		{
			if (j==n) break;
			long long st=a[j];
			j++;
			while (j<n && a[j]-st<=m) j++;
		}
		if (j==n) r=m; else l=m+1;
	}
	cout << fixed << setprecision(6) << l/2. << endl;
	int j=0;
	for (int i=0;i<3;i++)
	{
		if (j==n) {
			cout << 0. << ' ';
			continue;
		}
		long long st=a[j];
		j++;
		while (j<n && a[j]-st<=l) j++;
		cout << (st+a[j-1])/2. << ' ';
	}
}
