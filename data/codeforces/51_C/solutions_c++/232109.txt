#include <stdio.h>

#include <algorithm>

using namespace std;

int n;
int v[200005];
double sol1, sol2, sol3;

int exista (double dist)
{
	int i;
	sol1 = v[1] + dist;
	
	for (i = 1; i <= n; i ++)
		if (v[i] > sol1 + dist)
			break;
	sol2 = v[i] + dist;
	for (; i <= n; i ++)
		if (v[i] > sol2 + dist)
			break;
	sol3 = v[i] + dist;
	for (; i <= n; i ++)
		if (v[i] > sol3 + dist)
			return 0;
	return 1;
}

int main ()
{
#ifndef ONLINE_JUDGE
	freopen ("c.in", "r", stdin);
	freopen ("c.out", "w", stdout);
#endif

	scanf ("%d", &n);
	
	int i;
	
	for (i = 1; i <= n; i ++)
		scanf ("%d", &v[i]);
	sort (v + 1, v + n + 1);
	
	double st = 0, dr = v[n], m, sol;
	
	for (i = 1; i <= 100 && st <= dr; i ++)
	{
		m = (st + dr) * 0.5;
		if (exista (m))
		{
			dr = m;
			sol = m;
		}
		else
			st = m;
	}
	
	printf ("%.6lf\n", sol);
	printf ("%.6lf %.6lf %.6lf\n", sol1, sol2, sol3);
	
	return 0;
}
