#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
int n;
double a[200005];
double t[3],r[3];
bool solve(double mid)
{
	t[0]=a[1]+mid;
	r[0]=t[0]+mid;
	t[2]=a[n]-mid;
	r[2]=t[2]-mid;
	int st=1,ed=n;
	for(int i=1;i<=n;i++)
	{
		if(a[i]>r[0])
		{
			st=i;
			break;
		}
	}
	for(int i=n;i>=1;i--)
	{
		if(a[i]<r[2])
		{
			ed=i;
			break;
		}
	}
	if(ed<st)
	t[1]=t[2];
	else
	{
		if(a[ed]-a[st]>2*mid)
		return false;
		t[1]=a[st]+mid;
	}
	return true;
}
int main()
{
	while(~scanf("%d",&n))
	{
		for(int i=1;i<=n;i++)
		scanf("%lf",&a[i]);
		sort(a+1,a+1+n);
		double l=0,r=a[n]-a[1]+1;
		int lg=64;
		while(lg--)
		{
			//printf("%lf %lf\n",l,r);
			double mid=(l+r)*0.5;
			if(solve(mid))
			r=mid;
			else
		 	l=mid;
		}
		printf("%.6lf\n",l);
		for(int i=0;i<3;i++)
		printf("%.6lf\n",t[i]);
	}
	return 0;
}
