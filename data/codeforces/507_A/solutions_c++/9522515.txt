#include <bits/stdc++.h>
#define ll long long int
#define s(a) scanf("%lld",&a)

using namespace std;
vector<pair<ll,ll> >v;

int main()
{
	ll n,k,i,sum=0,j,x;
	s(n);s(k);
	for(i=1;i<=n;i++) {
		s(x);
		v.push_back(make_pair(x,i));
	}
	sort(v.begin(),v.end());
	for(i=0;i<n;i++) {
		sum+=v[i].first;
		if(sum>k) break;
	}
	cout<<i<<endl;
	for(j=0;j<i;j++) cout<<v[j].second<<" ";
	if(i!=0)
	cout<<endl;
	return 0;
}
