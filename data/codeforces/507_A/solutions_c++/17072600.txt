#include<iostream>
#include<algorithm>
using namespace std;
int n,k,i,ans,c;
pair <int,int> x[105];

int main(){
	cin>>n>>k;
	for(i=0; i<n; i++){
		cin>>x[i].first;
		x[i].second = i+1;
	}
	sort(x,x+n);
	i=0;
	for(i=0; i<n; i++){
		ans += x[i].first;
		if(ans<=k){
			c++;
		}
		else{
			break;
		}
	}
	cout<<c<<endl;
	for(i=0; i<c; i++){
		cout<<x[i].second<<" ";
	}
}
