#include<bits/stdc++.h>

using namespace std;



int main()
{
int n,k;
cin>>n>>k;
vector <pair <int,int> > x;
for(int i=1;i<=n;i++){int y;cin>>y;x.push_back(make_pair(y,i));}
sort(x.begin(),x.end());
vector <int> j;int v=0;
for(int i=0;i<n;i++){
    if(v+x[i].first <=k){v+=x[i].first;j.push_back(x[i].second);}

}
cout<<j.size()<<endl;
for(int i=0;i<j.size();i++){cout<<j[i]<<" ";}
}
