#include<iostream>
#include<cstdlib>
#include<algorithm>

using namespace std;

int main()
{
int i,ans=0,n,k;
cin >> n >> k;
pair<int,int> A[n];
for(i=0;i<n;i++)
	{ cin >> A[i].first;
     A[i].second = i+1;
	}
sort(A,A+n);
i=0;
while(k>=A[i].first && i<n)
	{ k = k-A[i].first;
     ans = ans+1;
     i=i+1;
	}

cout <<ans <<"\n";	
for(i=0;i<ans;i++)
	cout << A[i].second << " ";
cout << "\n";
return 0;
}
