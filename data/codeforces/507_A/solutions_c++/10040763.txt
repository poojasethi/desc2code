#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

int main() {
	int n,k,count=0;
	pair <int,int> a[100];
	cin>>n>>k;
	for(int i=0;i<n;i++) {
		cin>>a[i].first;
		a[i].second=i+1;
	}
	sort(a, a+n);
	for(int i=0;k>=a[i].first && i<n;i++) {
        k=k-a[i].first;
		count++;
	}
	cout<<count<<endl;
	for(int i=0;i<count;i++) {
        cout<<a[i].second<<' ';
	}
	return 0;
}
