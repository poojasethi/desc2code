#include<bits/stdc++.h>
using namespace std;
int main()
{
  pair<int,int> a[111];
  int n,d;
  cin>>n>>d;
  for(int i=0;i<n;i++)
   {  
    cin>>a[i].first;
    a[i].second=i+1;
   }
   sort(a,a + n);
  int c=0,s=0;
  for(int i=0;i<n;i++)
    {
      s=s+a[i].first;
      if(s<=d)
      c++;
     }
  cout<<c<<endl;
  for(int i=0;i<c;i++)
   cout<<a[i].second<<" ";
 cout<<endl;
}
