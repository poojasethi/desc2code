#include<bits/stdc++.h>

using namespace std;

int main()
{
    int n,k,x,s=0; pair<int,int>a[105];
    cin>>n>>k;
    for(int i=0;i<n;i++){
        cin>>a[i].first;a[i].second=i;
    }
    sort(a,a+n);
    int i=0;
    while (s<=k&&i<=n){
        s+=a[i].first;
        ++i;
    }
    cout<<i-1<<endl;
    for(int j=0;j<i-1;j++)
        cout<<a[j].second+1<<" ";
}
