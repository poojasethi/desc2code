#include<bits/stdc++.h>
using namespace std;
pair <int,int> a[111];
int main()
{
	int n,m; cin>>n>>m;
	for(int i=0 ; i<n ; i++)
	{
		cin>>a[i].first;
		a[i].second=i+1;
	}
	sort(a,a+n);
	long long sum=0,cnt=0;
	for(int i=0 ; i<n ; i++)
	{
		sum+=a[i].first;
		if(sum<=m )
		cnt++;
	}
	
	cout<<cnt<<endl;
	
	for(int i=0 ; i<cnt ; i++)
	cout<<a[i].second<<" ";
	

}
