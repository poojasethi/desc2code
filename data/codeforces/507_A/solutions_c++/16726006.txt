#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int n, k, c(0);
	cin >> n >> k;
	pair<int, int> a[n];
	for (int i = 0; i < n ; i++)
		cin >> a[i].first, a[i].second = i+1;
	
	sort(a, a + n);
	
	for (int i = 0; i < n && k >= a[i].first; i++)
		k-= a[i].first, c++;
	cout << c << endl;
	for (int i = 0; i < c ; i++)
		cout << a[i].second << " ";
	return 0;
}
