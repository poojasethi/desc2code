#include<bits/stdc++.h>
using namespace std;
int n,k,s,i;
pair<int,int>a[102];
int main(){
    cin>>n>>k;
    for(i=0;i<n;i++)
        cin>>a[i].first,a[i].second=i;
    sort(a,a+n);
    for(i=0;i<n;i++)
        if(s+a[i].first<=k)s+=a[i].first;
        else break;
    cout<<i<<'\n';
    for(int j=0;j<i;j++)
        cout<<a[j].second+1<<' ';
}
