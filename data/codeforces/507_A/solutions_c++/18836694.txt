#include <bits/stdc++.h>
using namespace std;
int main()
{
int n,k,i,j=1;
pair<int,int> a[101];
cin>>n>>k;
for(i=0;i<n;i++)
{
cin>>a[i].first;
a[i].second=j;
j++;
}
sort(a,a+n);
i=0;j=0;
while((k-a[i].first)>=0)
{if(i==n)
break;
k-=a[i].first;
i++;
}
if(i>0)
{
cout<<i<<endl;
for(j=0;j<i;j++)
cout<<a[j].second<<endl;
}
else
cout<<"0"<<endl;
return 0;
}
