using namespace std;
int main()
{
    int n,i,j,days;cin>>n>>days;pair<int,int>p[n];
    f(i,0,n){cin>>j;p[i]=make_pair(j,i+1);}
    sort(p,p+n); int sum=0;
    f(i,0,n){if(sum+p[i].first<=days)sum+=p[i].first;else break;}
    cout<<i<<endl;
    f(j,0,i)cout<<p[j].second<<" ";
}
