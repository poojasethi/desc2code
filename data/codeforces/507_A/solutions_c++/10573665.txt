#include<iostream>
#include<algorithm>
using namespace std;
int n,k,ans,i,j;
pair<int,int> x[150];
main()
{
    cin>>n>>k;
    for(i=0;i<n;++i)
    {cin>>x[i].first;
    x[i].second=i;}
    sort(x,x+n);
    for(i=0;i<n;++i)
    if(ans+x[i].first<=k)
    ans+=x[i].first;
       else
            break;
    cout<<i<<endl;
    for(j=0;j<i;j++)
        cout<<x[j].second+1<<' ';
}
