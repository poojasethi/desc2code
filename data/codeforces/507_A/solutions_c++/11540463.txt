#include<bits/stdc++.h>
#define f(j,n) for(int j=1 ; j<n ; j++)

using namespace std;


int main(){
long long n,k,ct=0,s=0;
cin>>n>>k;
pair <long,long> a[n];
for(long i=0 ; i< n ;i++){
	cin>>a[i].first;
	a[i].second=i+1;
}
sort(a,a+n);
for(long i=0 ; i<n && s+a[i].first<=k ; i++){
	ct++;
	s+=a[i].first;
}
cout<<ct<<endl;
for(long i=0 ;i<ct ; i++){
	cout<<a[i].second<<" ";
}



	return 0;
}
