#include<bits/stdc++.h>
using namespace std;
int main(){
int n;
long long sum=0, k,x;
vector<pair<long long,int> > arr;
cin>>n>>k;
for(int i=0;i<n;i++)
cin>>x,arr.push_back(make_pair(x,i+1));
sort(arr.begin(),arr.end());
int i=0;
for(i=0;i<n;i++){
sum+=arr[i].first;
if(sum>k)
break;
}
cout<<i<<endl;
for(int j=0;j<i;j++)
cout<<arr[j].second<<" ";
cout<<endl;
return 0;
}
