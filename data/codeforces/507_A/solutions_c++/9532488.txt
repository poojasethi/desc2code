#include<bits/stdc++.h>
using namespace std;
int n,k,dem,i;
pair<int,int>a[111];
int main()
{
    cin>>n>>k;
    for(i=0;i<n;++i)
        cin>>a[i].first,a[i].second=i;
    sort(a,a+n);
    for(i=0;i<n;++i)
        if(dem+a[i].first<=k)
            dem+=a[i].first;
        else
            break;
    cout<<i<<endl;
    for(int j=0;j<i;++j)
        cout<<a[j].second+1<<' ';
}
