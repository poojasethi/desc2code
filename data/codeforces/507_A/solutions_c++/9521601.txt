#include<cstdio>
#include<algorithm>
using namespace std;
pair<int,int> v[109];
int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++) scanf("%d",&v[i].first),v[i].second=i;
	sort(v,v+n);
	int rem=k,ans=0;
	for(int i=0;i<n;i++) if(rem-v[i].first>=0) ans++,rem-=v[i].first;
	printf("%d\n",ans);
	for(int i=0;i<ans;i++) printf("%d ",v[i].second+1);
}
