#include<iostream>
#include<bits/stdc++.h>
using namespace std;
pair<int,int>arr[110];
int main()
{
 int n,k,i,sum=0,j;
 cin>>n>>k;
 for(i=0;i<n;i++)cin>>arr[i].first,arr[i].second=i;
 sort(arr,arr+n);
 for(i=0;i<n;i++){
    if(arr[i].first+sum<=k)sum+=arr[i].first;
    else break;
 }
 cout<<i<<"\n";
 for(j=0;j<i;j++)cout<<arr[j].second+1<<" ";
    return 0;
}
