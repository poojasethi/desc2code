#include<iostream>
#include<utility>
#include<algorithm>

using namespace std;
	
int main(){
	int n,k,i,j,l,b=0;
	cin>>n>>k;
	pair<int,int>a[101];
	for(i=0;i<n;i++){
		cin>>a[i].first;
		a[i].second=i+1;		
	}
	sort(a,a+n);
	for(j=0;j<n;j++){
		if(b+a[j].first<=k)
		b=b+a[j].first;
		else
		break;
	}
	cout<<j<<endl;
	for(l=0;l<j;l++){
		cout<<a[l].second<<" ";
	}
	return 0;
}
