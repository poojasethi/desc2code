#include <bits/stdc++.h>
using namespace std;
pair <int,int>arr[110];
int main (){
	int n,m,i;
	cin>>n>>m;
	for(i=0;i<n;i++){
		cin>>arr[i].first;
		arr[i].second=i+1;}
	sort(arr,arr+n);
	for(i=0;i<n;i++){
		m-=arr[i].first;
		if(m<0)
			break;}
	cout<<i<<endl;
	while(i--){
		cout<<arr[i].second<<" ";}
	return 0;
}
