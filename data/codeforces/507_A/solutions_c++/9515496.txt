#include<bits/stdc++.h>
using namespace std;
vector<pair<int,int> >v;
int main()
{
	int n,k,i,x;
	cin>>n>>k;
	for(i=0;i<n;i++)
	{
		cin>>x;
		v.push_back(make_pair(x,i+1));
	}
	sort(v.begin(),v.end());
	i=0;x=k;
	while(i<n&&x>=v[i].first)
	{
		x-=v[i].first;
		i++;
	}
	cout<<i<<endl;
	i=0;x=k;
	while(i<n&&x>=v[i].first)
	{
		cout<<v[i].second<<" ";
		x-=v[i].first;
		i++;
	}
	return 0;
}
