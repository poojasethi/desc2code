#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n, m, sum=0;
	cin>>n>>m;
	pair<int,int> a[n];
	for(int i=0 ; i<n ; i++)
	{
		cin>>a[i].first;
		a[i].second = i+1;
	}
	sort(a,a+n);
	int j=0;
	while(sum<=m)
	{
		sum += a[j].first;
		j++;
	}
	cout<<--j<<endl;
	for(int k=0 ; k<j ; k++)
		cout<<a[k].second<<" ";
}
