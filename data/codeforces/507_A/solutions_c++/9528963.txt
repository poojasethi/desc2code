#include<iostream>
#include<algorithm>
using namespace std;

int main()
{
    pair<int,int> a[101];
    int n,k,i=0;
    cin>>n>>k;
    for(int i=0;i<n;i++)
    {
        cin>>a[i].first;a[i].second=i;
    }
    sort(a,a+n);
    i=0;
    while(k>0 &&i<n)
    {
        k-=a[i++].first;
    }
    if(k<0) i--;
    cout<<i<<endl;
    for(int j=0;j<i;j++)
        cout<<a[j].second+1<<" ";
}
