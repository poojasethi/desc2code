#include <iostream>
#include <algorithm>

using namespace std;

int n , k , d , ans;
pair <int , int> a[100];

int main() {
	cin>>n>>k;
	for (int i=0 ; i < n ; i++){
		cin>>a[i].first;
		a[i].second=i+1;
	}
	sort(a,a+n);
	for (int i=0 ; i < n ; i++)
		if (k >= d+a[i].first){
			d+=a[i].first;
			ans++;
		}
	cout<<ans<<'\n';
	for (int i=0 ; i < ans ; i++)
		cout<<a[i].second<<" ";
	return 0;
}
