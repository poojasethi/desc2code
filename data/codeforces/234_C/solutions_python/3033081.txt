import sys

sys.stdin = open('input.txt', 'r')
sys.stdout = open('output.txt', 'w')

n = int(input())
a = map(int, raw_input().split())
l, r = [0] * n, [0] *n
if a[0] >= 0: l[0] = 1
if a[n-1] <= 0: r[n-1] = 1
for i in range(1, n):
    l[i] = l[i-1]
    if a[i] >= 0: l[i] = l[i-1] + 1
    r[n-1-i] = r[n-i]
    if a[n-i-1] <= 0: r[n-1-i] = r[n-i] + 1

ans = 100000000
for i in range(0, n-1):
    ans = min(ans, l[i]+r[i+1])
print ans
