
F = open("input.txt")
o = open("output.txt", "w")
l = F.readline().split()
n = int(l[0])
l = F.readline()
a = map(int, l.split())
g = len([x for x in a if x <= 0])
f = 0
m = n + 1
for x in a[:-1]:
    if x <= 0:
        g -= 1
    if x >= 0:
        f += 1
    if (f + g < m):
        m = f + g         
o.write(str(m) + '\n')
o.close()
F.close()
