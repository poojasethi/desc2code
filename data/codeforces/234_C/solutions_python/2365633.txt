infile = open('input.txt')
outfile = open('output.txt', 'w')
def nums():
    return map(int, infile.readline().rstrip().split())
n = nums()[0]
ts = nums()
lf = [0] * n
rf = [0] * n
for i in xrange(1, n):
    lf[i] = lf[i - 1]
    if ts[i - 1] >= 0:
        lf[i] += 1
if ts[n - 1] <= 0:
    rf[n - 1] += 1
for i in xrange(n - 2, -1, -1):
    rf[i] = rf[i + 1]
    if ts[i] <= 0:
        rf[i] += 1
# print lf, rf
print >> outfile, min(x + y for x, y in zip(lf[1:], rf[1:]))
