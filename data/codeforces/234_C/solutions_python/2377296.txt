import sys
sys.stdin = open('input.txt', 'r')
sys.stdout = open('output.txt', 'w')
n = input()
a = map(int, raw_input().split())
sumL, sumR = [0], [0]
for i in xrange(1, n + 1):
    if a[i - 1] >= 0: sumL.append(sumL[-1] + 1)
    else: sumL.append(sumL[-1])
    if a[n - i] <= 0: sumR.append(sumR[-1] + 1)
    else: sumR.append(sumR[-1]);
ans = n + 1
for i in xrange(1, n):
    ans = min(ans, sumL[i] + sumR[n - i])
print ans
