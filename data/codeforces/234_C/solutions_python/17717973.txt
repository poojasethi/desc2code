f = open('input.txt', 'r').readlines()
n, t = int(f[0]), map(int, f[1].split())
nn, np = [0] * (n + 1), [0] * (n + 1)
for i in range(n):
  nn[i] = nn[i - 1] + (t[i] >= 0)
  np[n - i - 1] = np[n - i] + (t[n - i - 1] <= 0)
msf = n
for i in range(1, n):
  msf = min(msf, nn[i] + np[i] - (t[i] > 0))
f = open('output.txt', 'w')
f.write(str(msf) + '\n'); f.close()
