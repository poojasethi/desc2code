import sys
fi = open('input.txt', 'r')
fo = open('output.txt', 'w')
#fi = sys.stdin
#fo = sys.stdout
n = int(fi.readline())
seq = map(int, fi.readline().split(' '))
ans = 0
if seq[0] >= 0:
	ans = 1
for i in range(1, n):
	if seq[i] <= 0:
		ans = ans + 1

mans = ans
print 'curans', ans

for i in range(1, n - 1):
	if seq[i] < 0:
		ans = ans - 1
		print 'cur -1'
	elif seq[i] > 0:
		ans = ans + 1
		print 'cur +1'
	print 'curans', ans
	mans = min(mans, ans)

fo.write(str(mans) + '\n')
