import sys
sys.stdin = open("input.txt"); sys.stdout = open("output.txt", "w")
N = int(sys.stdin.readline().strip())
ts = map(int, sys.stdin.readline().strip().split())

zs = sum(1 for t in ts if t == 0)
ls = [0]*(N-1)
rs = [0]*(N-1)

ls[0] = int(ts[0] > 0)
for i in range(1,N-1):
  ls[i] = ls[i-1]
  t = ts[i]
  if t > 0:
    ls[i] += 1

rs[N-2] = int(ts[N-1] < 0)
for i in range(N-3, -1, -1):
  rs[i] = rs[i+1]
  t = ts[i+1]
  if t < 0:
    rs[i] += 1

print min(zs+l+r for l,r in zip(ls,rs))
