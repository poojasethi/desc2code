input_file = open("input.txt",'r');
n = int(input_file.readline())
list = map(int, input_file.readline().split(" "))
input_file.close()
outputfile = open("output.txt",'w')
neg=0
pos=0
for i in range(n):
	if list[i]<=0:
		neg = neg+1
ans = 100000
for i in range(n):
	if i==n-1:
		break
	if list[i]>=0:
		pos = pos+1
	if list[i]<=0:
		neg = neg-1;
	ans = min(pos+neg,ans)

outputfile.write(str(ans))
outputfile.close()
