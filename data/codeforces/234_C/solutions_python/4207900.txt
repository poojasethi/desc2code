ans = n = int(raw_input())
data = list(map(int,raw_input().split()))

x=0
for i in xrange(0,n):
    if data[i]<=0:
        x+=1
y = 0
for i in xrange(0,n):    
    if data[i] <= 0:
        x -= 1
    if data[i] >= 0:
        y += 1
    if i == n-1 and data[i]<=0:
        x+=1
    ans = min(ans,x+y)
