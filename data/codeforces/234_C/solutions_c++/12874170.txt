#include<stdio.h>
#include<iostream>
#include<fstream>
using namespace std;
int i,n,m,c1=0,c2=0,a[100005];
int main()
{
    freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
    cin>>n;
    m=n;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        if(a[i]<=0) c1++;
    }
    for(i=1;i<n;i++)
    {
        if(a[i]>=0) c2++;
        if(a[i]<=0) c1--;
        //cout<<c1<<" "<<c2<<"\n";
        m=min(m,c1+c2);
    }
    cout<<m;
}
