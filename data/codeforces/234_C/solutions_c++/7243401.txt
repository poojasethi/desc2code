#include <iostream>
#include <cmath>
#include <cstdio>

using namespace std;

int n;
int a[1000000];
int ri[1000000];
int le[1000000];

int main()
{
  freopen("input.txt","r",stdin);
  freopen("output.txt","w",stdout);
  cin>>n;
  for (int i=1;i<=n;i++)
    cin>>a[i];
  for (int i=n;i>1;i--)
    ri[i]=ri[i+1]+(a[i]<=0);
  int sol=1000000000;
  for (int i=1;i<n;i++) {
    le[i]=le[i-1]+(a[i]>=0);
    sol=min(sol,le[i]+ri[i+1]);
  }
  cout<<sol<<endl;
}
