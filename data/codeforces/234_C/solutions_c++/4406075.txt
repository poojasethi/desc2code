#include<iostream>
#include<cstdio>
#include<fstream>
#include<algorithm>
#define MAX 100005

using namespace std;

int ar[MAX],pos[MAX],neg[MAX];

int main(){
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int n;
	cin>>n;
	for(int i=0;i<n;++i) cin>>ar[i];
	for(int i=0;i<n;++i) pos[i+1]=pos[i]+(ar[i]>=0);
	for(int i=n-1;i>=0;--i) neg[i]=neg[i+1]+(ar[i]<=0);
	int res=100000000;
	for(int i=1;i<n;++i) res=min(res,pos[i]+neg[i]);
	cout<<res<<endl;
	return 0;
}
