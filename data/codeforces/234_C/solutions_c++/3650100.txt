#include<stdio.h>
int n,i,x[100010],c[100010],d[100010],min;
int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    min=n;
    for(i=1;i<=n;i++){
        scanf("%d",&x[i]);
        if(x[i]>=0)c[i]=c[i-1]+1;
        else c[i]=c[i-1];
    }
    for(i=n;i>1;i--){
        if(x[i]<=0)d[i]=d[i+1]+1;
        else d[i]=d[i+1];
        if(c[i-1]+d[i]<min)min=c[i-1]+d[i];
    }
    printf("%d",min);
}
