#include<bits/stdc++.h>

using namespace std;

int A[100005],P[100005],N[100005],n,i,mn = 100005;

int main(){

    freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	cin >> n;
	for(i=1;i<=n;i++){
		cin >> A[i];
	    P[i] = P[i-1] + (A[i] >= 0);
	}

	for(i=n; i ;i--)
		N[i] = N[i+1] + (A[i] <= 0);

	for(i=1;i<n;i++)
		mn = min(mn , P[i] + N[i+1]);
	
	cout << mn;
	return 0;
}
