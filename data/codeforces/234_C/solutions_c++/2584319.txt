#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main(){
int put,n,s[100002],cur;
cur=0;
freopen("input.txt", "r", stdin); freopen("output.txt", "w", stdout);
scanf("%d",&n);
for(int i=0;i<n;i++){
scanf("%d",&s[i]);
if(i==0 && s[i]>=0) cur++;
if(i!=0 && s[i]<=0) cur++;
}
put=cur;
for(int i=1;i<n-1;i++){
if(s[i]>0) cur++;
if(s[i]<0) cur--;
if(cur<put) put=cur;
}
printf("%d\n",put);
return 0;
}
