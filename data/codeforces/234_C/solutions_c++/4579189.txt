#include<bits/stdc++.h>
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,i,j,min,lc=0,rc=0;
	scanf("%d",&n);
	int arr[n];
	scanf("%d",arr);
	if (arr[0]>=0)
		lc++;
	for (i=1;i<n;i++)
	{
		scanf("%d",arr+i);
		if (arr[i]<=0)
			rc++;
	}
	min=lc+rc;
	for (i=2;i<n;i++)
	{
		if (arr[i-1]<0)
			rc--;
		else if (arr[i-1]==0)
		{
			lc++;
			rc--;
		}
		else
			lc++;
		if (lc+rc<min)
			min=lc+rc;
	}
	printf("%d",min);
	return 0;
}
