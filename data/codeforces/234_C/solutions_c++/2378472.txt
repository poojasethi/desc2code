#include <iostream>
#include<fstream>
using namespace std;
int a[100005];
int main()
{
    freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
    int n,i,neg=0,pos=0,m;
    cin>>n;
    m=n;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        if(a[i]<=0)
        neg++;
    }
    for(i=0;i<n-1;i++)
    {
        if(a[i]>=0)
        pos++;
        if(a[i]<=0)
        neg--;
        m=min(m,neg+pos);
    }
    cout<<m<<endl;
    return 0;
}
