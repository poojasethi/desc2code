#include<bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
ll n,i,a[100005],k=0,p=0,msd;
cin>>n;
msd=n;
for(i=1;i<=n;i++)
{
cin>>a[i];
if(a[i]<=0) k++;
}
for(i=1;i<n;i++)
{
if(a[i]>=0) p++;
if(a[i]<=0) k--;
msd=min(msd,p+k);
}
cout<<msd<<endl;
return 0;
}
