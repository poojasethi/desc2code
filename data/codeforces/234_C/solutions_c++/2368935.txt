#include <fstream>

using namespace std;

ifstream fin("input.txt");
ofstream fout("output.txt");

const int MAX_N=100010;
int n,a[MAX_N],l[MAX_N],r[MAX_N],ans=MAX_N;

int main(){
	fin >> n;
	l[0]=0,r[n+1]=0;
	for(int i=1;i!=n+1;i++){
		fin >> a[i];
		l[i]=l[i-1]+(a[i]>=0);
	}
	for(int i=n;i!=1;i--){
		r[i]=r[i+1]+(a[i]<=0);
		if (l[i-1]+r[i]<ans){
			ans=l[i-1]+r[i];
		}
	}
	fout << ans << endl;
	return 0;
}
