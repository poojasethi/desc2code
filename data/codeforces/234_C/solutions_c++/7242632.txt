#include <iostream>
#include <algorithm>
#include <fstream>
#include <string>
#include <vector>
using namespace std;

int n;
long long A[100006],T[100009];
int c=0,d=0;

int main()
{
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>A[i];
		if(A[i]>=0)
			c++;
		T[i]=c;
	}
	int r=200009;
	for(int i=n;i>1;i--)
	{
		if(A[i]<=0)
			d++;
		if(T[i-1]+d<r)
			r=T[i-1]+d;
	}

	cout<<r;
}
