#include<iostream>
#include<fstream>
using namespace std;
int main(){

    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int p=0,k=0,a[100005],n,m,i;
    scanf("%d",&n);
    m = n;
    for(i=0; i<n; i++){
        scanf("%d",&a[i]);
        if(a[i]<=0)k++;
    }
    for(i=0; i<n-1; i++){
        if(a[i]>=0)p++;
        if(a[i]<=0)k--;
        m = min(m, p+k);
    }
    printf("%d\n",m);
    return 0;
}
