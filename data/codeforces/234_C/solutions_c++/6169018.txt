#include<iostream>
#include<fstream>
#include<algorithm>
#include<cmath>
#include<string.h>
#include<iomanip>
#include<stdio.h>
#define LL long long 
using namespace std;


main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,k=0,ans=0,z=0;
	cin>>n;
	int a[n+1],i,j;
	for(i=1;i<=n;i++)
	{
		cin>>a[i];
		if(a[i]<=0) k++;
	}
	ans=n;
	for(i=1;i<n;i++)
	{
		if(a[i]>=0)z++;
		if(a[i]<=0)k--;
		ans=min(ans,z+k);
	}
	
	
	cout<<ans;
	
}
