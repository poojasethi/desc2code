#include <string>
#include <iostream>
#include <cstdio>

using namespace std;

int main(){
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,i,a[100005];
	cin>>n;
	int k=0,l=0,res=n;
	for(i=0;i<n;i++){
		cin>>a[i];
		if(a[i]<=0) k++;
	}
	for(i=0;i<n-1;i++){
		if(a[i]>=0){
			l++;
		}
		if(a[i]<=0){
			k--;
		}
		res=min(res,(l+k));
	}
	cout<<res;
	return 0;
}
