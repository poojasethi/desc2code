#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;

const int MAX=100000+10;

int n,a[MAX],p[MAX],d[MAX];

int main()
{
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
	scanf("%d",&n);
	int i,zero=0;
	for(i=1;i<=n;++i)
	{
		scanf("%d",&a[i]);
		if(a[i]==0)++zero;
		p[i]=p[i-1]+(a[i]<0);
		d[i]=d[i-1]+(a[i]>0);
	}
	int ans=n;
	for(i=2;i<=n;++i)
		ans=min(ans, d[i-1] + p[n]-p[i-1] );
	printf("%d\n",ans+zero);
	return 0;
}
