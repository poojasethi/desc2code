#include<stdio.h>
#include<iostream>
#include<fstream>
using namespace std;
int i,n,m,pos,neg,a[100005];
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    cin>>n;
    m=n;
    for(i=1;i<=n;i++)
    {
        cin>>a[i];
        if(a[i]<=0) neg++;
    }
    for(i=1;i<n;i++)
    {
        if(a[i]>=0) pos++;
        if(a[i]<=0) neg--;
        m=min(m,pos+neg);
    }
    cout<<m;
}
