#include <bits/stdc++.h>
using namespace std;
int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int mi,i,j,k=0,l,n,m,a[100005],b[100005]={0};
	cin>>n;
	mi = n;
	for(i=0;i<n;i++) {
		cin>>a[i];
		if(a[i] >= 0)
			b[i] = 1;
		if(i>0)
			b[i] = b[i-1]+b[i];
	}
	if(a[n-1]<1)
		k++;
	for(i=n-2;i>=0;i--) {
		mi = min(mi,b[i]+k);
		if(a[i]<=0)
			k++;
	}
	cout<<mi<<endl;
	return 0;
}
