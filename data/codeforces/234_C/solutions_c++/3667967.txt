#include <cstdio>
#include <algorithm>
using namespace std;

int n, m, x = 0, y = 0, i, a[100005];

int main() {
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
    scanf("%d", &n);
    m = n;
    for (i = 1; i <= n; i++) {
        scanf("%d", &a[i]);
        if (a[i] <= 0) y++;
    }
    for (i = 1; i < n; i++) {
        if (a[i] >= 0) x++;
        if (a[i] <= 0) y--;
        m = min(m, x + y);
    }
    printf("%d\n", m);
    return 0;
}
