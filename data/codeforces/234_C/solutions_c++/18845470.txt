#include<iostream>
#include<cstdio>
#include<cstring>
#define rep(i,a,b) for (int i=a;i<=b;i++)
using namespace std;
#define N 100005
int n,a[N];
int main()
{   int ans=1999999999,k=0,b=0;
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    rep(i,1,n){
        scanf("%d",&a[i]);
        if (a[i]<=0)k++;
    }
    rep(i,1,n-1)
    {     if (a[i]>=0)b++;
          if (a[i]<=0)k--;
          ans=min(ans,k+b);
    }
    printf("%d\n",ans);
}
