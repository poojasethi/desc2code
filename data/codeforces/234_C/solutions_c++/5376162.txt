#include<algorithm>
#include<iostream>
#include<cstdio>
using namespace std;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	for(long n,m,s,t;cin>>n;cout<<s<<'\n')
		for(cin>>m,s=t=-1<m;--n;)
			if(cin>>m,0<m)s=std::min(s,t++);
			else s=1+std::min(s,t),t+=!m;
	return 0;
}
