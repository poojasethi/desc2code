#include<iostream>
#include<fstream>
using namespace std;
int n,i,countN=0,countP=0,ans;
main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);

    cin>>n;ans=n;int a[n];
    for(i=1;i<=n;i++){
        cin>>a[i]; if(a[i]<=0)countN++;
    }
    for(i=1;i<n;i++){
        (a[i]==0)?countP++,countN--:((a[i]>0)?countP++:countN--);
        ans=min(ans,countN+countP);
    }
    cout<<ans;
}
