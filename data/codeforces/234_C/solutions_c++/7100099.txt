#include <stdio.h>
using namespace std;
int n,t[100001],d[100001],a,res=100000;
int main() 
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
    {
        scanf("%d",t+i);
        d[i]=d[i-1]+(t[i]>=0);
    }
    a=(t[n]<=0);
    for(int i=n-1;i>=1;i--)
    {
        if(res>=d[i]+a) res=d[i]+a;
        a+=(t[i]<=0);
    }
    printf("%d",res);
}
