#include <cstdio>
#include <algorithm>
using namespace std;
int t[100001];
int n,neg=0,pos=0,ans;
int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    scanf("%d",&n);
    ans=n;
    for(int i=0;i<n;i++){
      scanf("%d",&t[i]);
       if(t[i]<=0) neg++;
    }
    for(int i=0;i<n-1;i++){
        if(t[i]>=0)pos++;
        if(t[i]<=0) neg--;
        ans=min(ans,pos+neg);
    }
    printf("%d",ans);
}
