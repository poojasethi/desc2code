#include<cstdio>
#include<algorithm>
using namespace std;

int arr[100005];

int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int N;
	scanf("%d",&N);
	for(int i=0;i<N;i++)
		scanf("%d",&arr[i]);
		
	int n=0,p=0,ret=1<<30;
	
	for(int i=1;i<N;i++)
		if(arr[i]<=0)
			n++;
	
	if(arr[0]>=0)p=1;
	
	for(int i=1;i<N;i++)
	{
		ret=min(ret,n+p);
		if(arr[i]>=0)p++;
		if(arr[i]<=0)n--;
	}
	printf("%d\n",ret);
}
