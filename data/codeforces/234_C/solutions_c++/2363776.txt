#include<stdio.h>

const int MAXN=100000+9,INF=9999999;

int t[MAXN],l[MAXN],r[MAXN];
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,i,ans=INF;
	scanf("%d",&n);
	for(i=1;i<=n;++i)
		scanf("%d",t+i);
	for(i=1;i<=n;++i)
		l[i]=l[i-1]+(t[i]>=0);
	for(i=n;i;--i)
		r[i]=r[i+1]+(t[i]<=0);
	for(i=1;i<n;++i)
		if(l[i]+r[i+1]<ans)
			ans=l[i]+r[i+1];
	printf("%d\n",ans);
	return 0;
}
