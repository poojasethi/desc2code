#include <iostream>
#include<cstdio>

using namespace std;

int main(){
	int a[100005] , n , b , c , F;

	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	cin >> n;
	F = n ;
    b = c = 0;
	for (int i = 1 ; i <= n ; i++){
		cin>>a[i];
		if (a[i] <= 0) b++;
	}

	for (int i = 1 ; i < n ; i++){
		if (a[i] <= 0) b--;
		if (a[i] >= 0) c++;
		F = min(F , b + c);
	}
	cout << F << endl;
	return 0;
}
