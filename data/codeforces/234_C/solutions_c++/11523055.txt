#include<cstdio>
#include<algorithm>
using namespace std;

int n, l, r, ans;
int a[111111];

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%d", &a[i]);
	l = (a[0] >= 0);
	r = 0;
	for(int i=1;i<n;i++)
		r += (a[i] <= 0);
	
	ans = n;
	for(int i=1;i<n;i++) {
		ans = min(ans, l + r);
		l += (a[i] >= 0);
		r -= (a[i] <= 0);
	}
	printf("%d\n", ans);
	
	return 0;
}
