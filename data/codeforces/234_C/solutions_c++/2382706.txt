#include<cstdio>
int n = 0x7fffffff;
int A[100010];
int i,N,S1,S2,Ans=1<<30;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&N);
	for(i=1;i<=N;++i) scanf("%d",&A[i]);
	for(i=1;i<=N;++i) if(A[i]<=0) ++S2;
	for(i=1;i<N;++i)
	{
		if(A[i]<=0) --S2;
		if(A[i]>=0) ++S1;
		if(S1+S2<Ans) Ans=S1+S2;
	}
	printf("%d\n",Ans);
	return 0;
}
