#include <iostream>
#include <cstdio>
#include <fstream>
using namespace std;

const int maxn=1e5+10;

int n,t[maxn],l[maxn],r[maxn],ans=10000000;

int main(void)
{
	ifstream	fin("input.txt");
	ofstream	fout("output.txt");
	fin>>n;
	for(int i=1;i<=n;i++)
	{
		fin>>t[i];
		l[i]=l[i-1]+(t[i]>=0);
	}
	for(int i=n;i>1;i--)
	{
		r[i] += r[i+1]+(t[i]<=0);
		ans = min(ans,l[i-1]+r[i]);
	}

	fout<<ans<<endl;
	return 0;
}
