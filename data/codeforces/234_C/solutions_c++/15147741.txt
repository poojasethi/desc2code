#include <iostream>
#include <fstream>
#include <cstdio>

using namespace std;

	int n, x, Pos, Neg;

int main(){

	ifstream fin;
	ofstream fout;

	fin.open("input.txt");
	fout.open("output.txt");

	fin >> n >> x;
	Pos=Neg=(x>=0);
	for(int i=2; i<=n; i++){
		fin >> x;
		Pos=min(Pos, Neg);
		if(x>0)
			Neg++;
		else{
			Pos++;
			if(x==0)
				Neg++;
		}
	}

	fout << Pos;

 return 0;
}
