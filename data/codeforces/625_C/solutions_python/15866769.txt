l=lambda:map(int,raw_input().split())
n,k=l()
k -= 1
r=xrange(n)
a = [[0 for _ in r] for _ in r]
start = n * k
end = n * (k + 1)
t = n - k
for i in xrange(n * n):
  if i < start:
    a[i/k][i%k] = i + 1
  else:
    idx = i - start
    a[idx/t][idx%t + k] = i + 1
print reduce(lambda z,e:z+e[k],a,0)
print "\n".join([" ".join([`x` for x in e]) for e in a])
