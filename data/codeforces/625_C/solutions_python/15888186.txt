(N, k) = map(int,raw_input().strip().split())

sum = 0
max = N * N
lines = []
for i in xrange(N):
  line = []
  num = (max - (N - k)) - (N - k + 1) * i
  sum += num
  for j in xrange(1, k):
   line.append("%d"% (j + (k - 1) * i)),
  line.append("%d"% (num))
  for j in xrange(N - k):
   line.append("%d"% (num + j + 1)	)
  lines.append(" ".join(line))
  
print sum
for i in xrange(N):
  print lines[i]
