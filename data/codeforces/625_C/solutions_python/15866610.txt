
n,k = [int(i) for i in raw_input().split(' ')]

a=[[0]*n for i in range(n)]

mx=n**2
mn=1

for i in range(n):
        for j in range(n):
                if j<k-1:
                        a[i][j]=mn
                        mn+=1
                else:
                        a[i][k-1:]=range(mx-(n-k),mx+1)
                        mx-=(n-k+1)
                        break
s=0
for i in range(n):
        s+=a[i][k-1]
print s
for r in a:
        print " ".join([str(i) for i in r])
