n, k = map(int,raw_input().split())
a = []

l  = (n - k + 1) * n
start = range(1, n**2 - l + 1)
end = range(n**2 - l + 1, n**2 + 1)

for i in range(n):
    inn = []
    for x in range(i*(k - 1), (i + 1)*(k - 1)):
        inn.append(start[x])
    for y in range(i*(n - k + 1), (i + 1)*(n - k + 1)):
        inn.append(end[y])
    a.append(inn)

s = 0
for i in range(n):
    s += a[i][k - 1]

print s
for i in range(n):
    print ' '.join(map(str, a[i]))
