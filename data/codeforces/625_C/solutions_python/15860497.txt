#!/usr/bin/env python

n, k = map(int, raw_input().split())

nums = range(1,(n*n)+1)
topk = nums[-(n-k+1)*n:]
leftover = nums[:-(n-k+1)*n]

res = [[-1 for _ in range(n)] for __ in range(n)]
idx = 0
for i in range(n):
	for j in range(k-1, n):
		res[i][j] = topk[idx]
		idx += 1

idx = 0
for i in range(n):
	for j in range(k-1):
		res[i][j] = leftover[idx]
		idx += 1
sum = 0
for i in range(n):
	sum += res[i][k-1]

print sum
for r in res:
	print " ".join(map(str, r))
