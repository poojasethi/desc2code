#include<iostream>
using namespace std;
int main(){
	int N, K, i, j, s=0, aux=1;
	cin>>N>>K;

	for(i=1;i<=N;i++)
		s += (K-1)*N + (N-K+1)*i - (N-K); 
	cout<<s<<endl;

	for(i=1;i<=N;i++){
		for(j=1;j<=N;j++){
			if(j<K)
				cout<<(K-1)*(i-1) + j;
			else
				cout<<(K-1)*N + (j-K+1) + (N-K+1)*(i-1);
			cout<<" ";
		}
		cout<<endl;	
	}


	return 0;
}
