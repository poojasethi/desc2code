#include <bits/stdc++.h>
using namespace std;
int main() {
	ios::sync_with_stdio(false);
	int n, k, f, s, i, j, sum = 0;
	cin >> n >> k;
	stringstream out;
	f = 1;
	s = (k - 1) * n + 1;
	for(i = 0; i < n; i++) {
		for(j = 1; j < k; j++) {
			out << f << " ";
			f++;
		}
		sum += s;
		while(j <= n) {
			out << s << " ";
			s++;
			j++;
		}
		out << endl;
	}
	cout << sum << endl << out.str();
	return 0;
}
