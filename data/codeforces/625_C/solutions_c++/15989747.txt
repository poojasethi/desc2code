#include <stdio.h>

int main(void) {
	int n,k;
	int i,j;
	int l,m;
	int s = 0;
	scanf("%d %d",&n,&k);
	l = 1;
	m = n * ( k -1 ) + 1;
	
	for ( i = 0 ;i < n; i++)
	{
		s += n*(k-1)+1 + (n-k+1)*i;		
	}
	printf("%d\n", s);
	for ( i = 0 ;i < n; i++)
	{
		for ( j = 0 ;j < n; j++) {
			if ( j < (k -1) ) printf("%d ",l++); else printf("%d ",m++);			
		}
		printf("\n");
	}
	return 0;
}
