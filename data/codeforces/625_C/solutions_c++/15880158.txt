#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)

using namespace std;

int sum=0,n,k,s[505][505],p=0;
int main(){
  scanf("%d %d",&n,&k);
  FI(i,1,k-1) FI(j,1,n) s[j][i]=++p;
  FI(i,1,n) FI(j,k,n){
    s[i][j]=++p;
    if(j==k) sum+=p;
  }
  printf("%d\n",sum);
  FI(i,1,n) FI(j,1,n) printf("%d%c",s[i][j],j==n?'\n':' ');
  return 0;
}
