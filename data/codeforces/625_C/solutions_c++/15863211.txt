#include<cstdio>
using namespace std;

const int NUM=505;
int n,k,Ans[NUM][NUM];

int main(){
	int i,j,f,h;
	scanf("%d%d",&n,&k);f=1,h=n*n;
	printf("%d\n",n*n*n+n-(n-k+1)*n*(n+1)/2);
	for(i=1;i<=n;i++)
	{ for(j=1;j<k;j++)Ans[i][j]=f++;
	  for(j=n;j>=k;j--)Ans[i][j]=h--;
	}
	for(i=1;i<=n;i++)
	{ for(j=1;j<n;j++)printf("%d ",Ans[i][j]);
	  printf("%d\n",Ans[i][n]);
	}
	return 0;
}
