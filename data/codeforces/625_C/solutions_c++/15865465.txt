#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k;
	cin>>n>>k;
	int a[n][n];
	int i,j;
	int x=1;
	for(i=0;i<n;i++)
		for(j=0;j<k-1;j++)
			a[i][j]=x++;
	
	for(i=0;i<n;i++)
		for(j=k-1;j<n;j++)
			a[i][j]=x++;
	long long int ans=0;
	for(i=0;i<n;i++)
			ans+=a[i][k-1];
	
	cout<<ans<<endl;
	for(i=0;i<n;i++)
		{
		for(j=0;j<n;j++)
			cout<<a[i][j]<<" ";
		cout<<endl;
		}		
	return 0;
}
