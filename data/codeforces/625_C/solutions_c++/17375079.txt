#include <iostream>

using namespace std;
int n,k,a[600][600],stx,sty,sum;
int main(){
  cin>>n>>k;
  stx=n*(k-1)+1;
  sty=1;
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++){
      if(j<k-1)a[i][j]=sty++;
      else a[i][j]=stx++;
      if(j==k-1)sum+=a[i][j];
    }
  }
  cout<<sum<<endl;
  for(int i=0;i<n;i++){
    for(int j=0;j<n;j++)cout<<a[i][j]<<" ";
    cout<<endl;
  }

  return 0;
}
