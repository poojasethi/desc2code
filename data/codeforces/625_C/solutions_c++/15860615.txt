#include<iostream>
using namespace std;
int a[505][505],i,j,n,k,s;
int main(){
	cin>>n>>k;
	for(i=1;i<=n;i++){
		for(j=1;j<k;j++)
		a[i][j]=(i-1)*(k-1)+j;
		for(j=k;j<=n;j++)
		a[i][j]=(i-1)*(n-k+1)+n*(k-1)+j-k+1;
	}
	for(i=1;i<=n;i++)
	s+=a[i][k];
	cout<<s<<endl;
	for(i=1;i<=n;i++){
		for(j=1;j<=n;j++)
		cout<<a[i][j]<<" ";
		cout<<endl;
	}
}
