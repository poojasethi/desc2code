#include <iostream>

using namespace std;
string A,B;
int n,k,x,s,i,j,y;

int main()
{
    cin>>n>>k;
    x=n*k-n+1;y=1;
    s=n*x+(n-k+1)*n*(n-1)/2;
    cout<<s<<'\n';
    for(i=1;i<=n;i++)
    {
        for(j=1;j<k;j++,y++)
            cout<<y<<' ';
        for(j=k;j<=n;j++,x++)
            cout<<x<<' ';
        cout<<'\n';
    }
    return 0;
}
