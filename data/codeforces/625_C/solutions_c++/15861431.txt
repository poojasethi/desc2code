#include<cstdio>
int f[555][555]={};
int main()
{
	int n,k;
	int tot=0;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
		for(int j=1;j<k;j++)
			f[i][j]=++tot;
	for(int i=1;i<=n;i++)
		for(int j=k;j<=n;j++)
			f[i][j]=++tot;
	int sum=0;
	for(int i=1;i<=n;i++)
		sum+=f[i][k];
	printf("%d\n",sum);
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
			printf("%d ",f[i][j]);
		printf("\n");
	}
}
