#include<iostream>
using namespace std;
int a[501][501],ans=0,l=1,n,k,i,j;
int main()
 {
 cin>>n>>k;
 for(i=0;i<n;i++)
 {
  for(j=0;j<k-1;j++)
   a[i][j]=l++;
 }
 for(i=0;i<n;i++)
  for(j=k-1;j<n;j++)
   {
   a[i][j]=l++;

   }
   for(i=0;i<n;i++)
   ans+=a[i][k-1];
   cout<<ans<<endl;
  for(i=0;i<n;i++)
  {
  for(j=0;j<n;j++)
  cout<<a[i][j]<<" ";
  cout<<"\n";
  }
  return 0;
 }
