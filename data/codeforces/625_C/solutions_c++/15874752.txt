#include <iostream>
using namespace std;
int n,i,j,c,nr,sum,M[600][600];
int main()
{
    cin>>n>>c;
    for(i=1;i<=n;++i)
        for(j=1;j<c;++j) M[i][j]=++nr;
    for(i=1;i<=n;++i)
        for(j=c;j<=n;++j) M[i][j]=++nr;
    for(i=1;i<=n;++i) sum+=M[i][c];
    cout<<sum<<'\n';
    for(i=1;i<=n;++i,cout<<'\n')
        for(j=1;j<=n;++j) cout<<M[i][j]<<" ";
    return 0;
}
