#include<stdio.h>
int sb,shit;
int main()
{
	int n,a;
	scanf("%d%d",&n,&a);
	printf("%d\n",((n*(a-1)+1)+(n*(a-1)+1+(n-a+1)*(n-1)))*n/2);
	shit=n*(a-1);
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=n;j++)
		{
			if(j<a)
			printf("%d ",++sb);
			else 
			printf("%d ",++shit);
		}
		printf("\n");
	}
	return 0;
}
