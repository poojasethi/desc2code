#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,i,j,sum;
	int a[550][550];
	cin>>n>>k;
	int count1=1;
	for (i=0;i<n;i++)
	for (j=0;j<k-1;j++)
	a[i][j]=count1++;
	for (i=0;i<n;i++)
	for (j=k-1;j<n;j++)
	a[i][j]=count1++;
	sum=0;
	for (i=0;i<n;i++)
	sum+=a[i][k-1];
	cout<<sum<<endl;
	for (i=0;i<n;i++){
	for (j=0;j<n;j++)
	cout<<a[i][j]<<" ";
	cout<<endl;
	}
	return 0;
}
