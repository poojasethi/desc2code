#include<bits/stdc++.h>
using namespace std;
#define ll long long 
ll a[1000][1000];

int main()
{
ll n,k,i,j;
cin>>n>>k;
ll val=1;

for(i=1;i<=n;i++)
for(j=1;j<k;j++)
a[i][j]=val++;


for(i=1;i<=n;i++)
for(j=k;j<=n;j++)
a[i][j]=val++;

ll sum=0;

for(i=1;i<=n;i++)
sum+=a[i][k];

cout<<sum<<"\n";

for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)
cout<<a[i][j]<<" ";
cout<<"\n";
}


return 0;
}
