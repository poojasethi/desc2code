#include <iostream>

using namespace std;

int main()
{

int i,j,k,n;
cin>>n>>k;
int a[n+1][n+1];
int c=n*n;
for(i=n;i>=1;i--)
{
for(j=n;j>=k;j--)
{
a[i][j]=c;
c--;

}

}
for(i=n;i>=1;i--)
{
for(j=k-1;j>=1;j--)
{
a[i][j]=c;
c--;

}


}
int c1=0;
for(i=1;i<=n;i++)
{
c1=c1+a[i][k];

}
cout<<c1<<endl;
for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)
cout<<a[i][j]<<" ";
cout<<endl;

}

    return 0;
}
