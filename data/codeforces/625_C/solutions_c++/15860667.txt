#include <cstdio>

using namespace std;

int T[1000][1000];

int main(){
	int n,m;
	int i,j,p,k,s;
	scanf("%d%d", &n,&m);
	p = 1; k = n*n;
	s = 0;
	for(i=1; i<=n; i++){
		for(j=n; j>=m; j--) T[i][j] = k--;
		for(j=1; j<m; j++) T[i][j] = p++;
		s += T[i][m];
	}
	printf("%d\n", s);
	for(i=1; i<=n; i++){
		for(j=1; j<=n; j++) printf("%d ", T[i][j]);
		printf("\n");
	}
	return 0;
}
