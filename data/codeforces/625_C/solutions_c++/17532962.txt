#include<bits/stdc++.h>
using namespace std;

int main()
{
	int i,j,n,k,tot,step,l,r,ans = 0;
	scanf("%d%d",&n,&k);
	tot = n*n-n+k;step = (n-k+1);
	for(int i=1;i<=n;i++){
		ans += tot;
		tot -= step;
	}
	printf("%d\n",ans);
	l=1;r=n*n;
	for(int i=1;i<=n;i++){
		for(int j=1;j<k;j++) printf("%d ",l++);
		for(int j=k;j<=n;j++) printf("%d ",r-n+j);
		r-=step;
		printf("\n");
	}		
	return 0;
}
