#include <iostream>
using namespace std;
int main(){
	int n, k;
	cin>>n>>k;

	int a = 1, b = n*n - n + k;
	cout<<(n*(k+1+n*(k+n-2)))/2<<endl;
	for(int i=1;i<=n;++i){
		int c = b;
		for(int j=1;j<=n;++j){
			if(j < k) cout<<a++<<" ";
			else cout<<c++<<" ";
		}
		b = b - (n - k + 1);
		cout<<endl;
	}
}
