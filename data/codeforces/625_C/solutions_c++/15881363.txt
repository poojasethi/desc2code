#include <stdio.h>

int main()
{
    int a[550][550],n,k,i,j,p=1,sum=0;
    scanf("%d %d",&n,&k);

    for(i=1;i<=n;i++) for(j=1;j<k;j++) a[i][j]=p++;
    for(i=1;i<=n;i++) for(j=k;j<=n;j++) a[i][j]=p++;
    for(i=1;i<=n;i++) sum+=a[i][k];
    printf("%d\n",sum);
    for(i=1;i<=n;i++){
        for(j=1;j<=n;j++) printf("%d ",a[i][j]);
        printf("\n");
    }

    return 0;
}
