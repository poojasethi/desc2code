#include <bits/stdc++.h>
using namespace std;
int b[505][505];
int main(){
	int p=1,n,k,i,j;cin>>n>>k;
	for(j=1;j<=n;++j)
		for(i=1;i<k;++i)
			b[j][i]=p++;

	for(j=1;j<=n;++j)
	for(i=k;i<=n;++i)
		b[j][i]=p++;
	int sum=0;
	for(i=1;i<=n;++i)
		sum+=b[i][k];
	cout<<sum<<endl;
	for(i=1;i<=n;++i){
		for(j=1;j<=n;++j)
			cout<<b[i][j]<<" ";
		putchar('\n');
	}
}
