#include<cstdio>
int main(){
	int n,k;
	scanf("%d%d",&n,&k);
	int cnt1=1,cnt2=n*(k-1)+1,ans=0;
	for(int i=cnt2;i<=n*n;i+=n-k+1)
		ans+=i;
	printf("%d\n",ans);
	for(int i=1;i<=n;i++){
		for(int j=1;j<k;j++){
			printf("%d ",cnt1);
			cnt1++;
		}
		for(int j=k;j<n;j++){
			if(j==k) ans+=cnt2;
			printf("%d ",cnt2);
			cnt2++;
		}
		printf("%d\n",cnt2);cnt2++;
	}
	return 0;
}
