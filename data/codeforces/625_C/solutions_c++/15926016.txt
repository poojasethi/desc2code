#include<bits/stdc++.h>

using namespace std;
	
int n,k,s,t=1;
int table[501][501];

int main ()
{
	cin>>n>>k;
	for(int i=1; i<=n; i++)
		for(int j=1; j<k; j++)
			table[i][j]=t++;
	for(int i=1; i<=n; i++)
		for(int j=k; j<=n; j++)
			s+=(j==k)*(table[i][j]=t++);
	cout<<s<<endl;
	for(int i=1; i<=n; i++,puts(""))
		for(int j=1; j<=n; j++)
			cout<<table[i][j]<<" ";
}
