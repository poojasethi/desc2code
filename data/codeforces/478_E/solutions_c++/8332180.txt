#include <iostream>
#include <cstdlib>

using namespace std;

typedef long long int ll;

ll n,k;
int dn;
ll pn;

int cumple(ll x,int &act)
{
  if (x==0) return -1;
  if (x<10) return 1;
  if (x<100) {
    if (x%10==x/10%10) return -1;
    return 1;
  }
  int last=x%10;
  x/=10;
  act=x%10;
  while (x>=10) {
    x/=10;
    int next=x%10;
    if (next>=act and act>=last) return -1;
    if (next<=act and act<=last) return -1;
    last=act;
    act=next;
  }
  if (act>last) return 0;
  return 1;
}

int cuantos6=0;
int cuantos[1000000][10][2];

void genera(ll c,int numd)
{
  int daux;
  if (numd<dn) {
    ll x=n-(c*pn)%n;
    if (x==n) x=0;
    if (x<pn and 0<=cumple(c*pn+x,daux)) {
      if (k==1) {
	//cout<<"caso "<<c<<" "<<numd<<" "<<pn<<" "<<dn<<" "<<x<<endl;
	cout<<c*pn+x<<endl;
	exit(0);
      }
      k--;
    }
    return;
  }
  int d=c%10;
  if (numd==6) {
    if (c==0) {
      if (cuantos6<k) {
	k-=cuantos6;
	//cout<<"caso1 "<<c<<" "<<cuantos6<<endl;
	return;
      }
    } else {
      ll x=n-(c*1000000)%n;
      if (x==n) x=0;
      if (c<10) {
	int quita=0;
	if (d<9) quita+=cuantos[x][d+1][0];
	if (d>0) quita+=cuantos[x][d-1][1];
	if (quita<k) {
	  k-=quita;
	  //if (quita)
	    //cout<<"caso2 "<<c<<" "<<quita<<endl;
	  return;
	}
      } else {
	int dd=c%100/10;
	int quita=0;
	if (dd>d) quita=cuantos[x][d+1][0];
	else quita=cuantos[x][d-1][1];
	if (quita<k) {
	  k-=quita;
	  //if (quita)
	    //cout<<"caso3 "<<c<<" "<<quita<<endl;
	  return;
	}
      }
    }
  }
  if (c==0) {
    for (int dnext=0;dnext<=9;dnext++)
      genera(c*10+dnext,numd-1);
  } else if (c<10) {
    for (int dnext=0;dnext<=9;dnext++)
      if (dnext!=d or numd==1)
	genera(c*10+dnext,numd-1);
  } else {
    int dd=c%100/10;
    if (dd>d)
      for (int dnext=d+1;dnext<=9;dnext++)
	genera(c*10+dnext,numd-1);
    else
      for (int dnext=0;dnext<d;dnext++)
	genera(c*10+dnext,numd-1);
  }
}
