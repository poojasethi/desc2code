#define ll long long
using namespace std;
const ll maxn= 1e6+20,E4= 1e4,E5= 1e5,E6= 1e6,E7= 1e7,mo= 5201314;
ll head[mo+10],next[maxn],des[maxn];
ll a1[maxn],a2[maxn],cnt[mo+10];
ll N,K,ans= -1,t1,t2,i,j,k,yu,tot;
void dfs(ll x,ll o){
	if (o>7){
		if (x) a1[++t1]= x;
		if (x>E6||(x>E5 && x/E5>x/E4%10))
			a2[++t2]= x;
		return;
	}
	ll i;
	if (x<10){
		if (x==0){
			dfs(x,o+1);
			for (i= 1;i<=9;i++)
				dfs(i,o+1);
		}else for (i= 0;i<=9;i++)
				  if (i!=x) dfs(x*10+i,o+1);
	}else {
		ll a= x/10%10,b= x%10;
		if (a>b){
			for (i= b+1;i<=9;i++)
				dfs(x*10+i,o+1);
		}else {
			for (i= 0;i<=b-1;i++)
				dfs(x*10+i,o+1);
		}
	}
}
ll get(ll x,ll y,ll z){
	return (z*10+x)*10+y;
}
void update(ll o){
	ll K= get(o/E6,o/E5%10,o%N),i;
	for (i= head[K%mo];i;i= next[i])
		if (des[i]==K){
			cnt[i]++;
			return;
		}
	next[++tot]= head[K%mo];
	des[tot]= K;
	head[K%mo]= tot;
	cnt[tot]++;
}
ll val(ll j,ll k,ll v){
	ll K= get(j,k,v),i;
	for (i= head[K%mo];i;i= next[i])
		if (des[i]==K) return cnt[i];
	return 0;
}

							if (K>val(j,k,yu)) K-= val(j,k,yu);
