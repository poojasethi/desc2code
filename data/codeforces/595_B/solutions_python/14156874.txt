from math import *
I = lambda : map(int, raw_input().split())

MOD = 10**9+7
n, k = I()

a = I()
b = I()
count = 1
for i in xrange(len(a)):
    poss = ((10**k)-1) / a[i]
    poss += 1
    ll = b[i]*(10**(k-1))
    #print 'll',ll,
    rl = (b[i]+1)*(10**(k-1))-1
    #print 'rl',rl,
    tod = int(floor(float(rl)/a[i])) - int(ceil(float(ll)/a[i])) + 1
    #print 'tod', tod
    poss -= tod
    #print "pos",poss
    count *= poss%MOD
    count %= MOD
print count
