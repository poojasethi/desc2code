x = list(map(int, raw_input().split()))
a = list(map(int, raw_input().split()))
b = list(map(int, raw_input().split()))
n = x[0]
k = x[1]
tenk = 10**(k-1)
ans = 1
nk = len(a)
for i in range(nk):
    if b[i] != 0:
        left = b[i]*tenk - 1
        right = (b[i] + 1)*tenk - 1
        cur_ans = 1 + (tenk*10 - 1)/a[i] - right/a[i] + left/a[i]
    else:
        cur_ans = (tenk*10 - 1)/a[i] - (tenk - 1)/a[i]
    ans *= cur_ans

print ans % 1000000007
