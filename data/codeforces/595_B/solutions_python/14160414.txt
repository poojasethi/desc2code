'''input
8 2
1 22 3 44
5 4 3 2
'''
n, k = map(int, raw_input().split())
A = map(int, raw_input().split())
B = map(int, raw_input().split())

MOD = int(1e9 + 7)
ans = 1

def in_range (x, a, b):
	l = x * int(a / x)
	if l < a: l += x

	r = x * int(b / x)
	if r > b: r -= x

	return int(r / x) - int(l / x) + 1

for i in xrange(n / k):
	minb = B[i] * 10**(k - 1)
	maxb = (B[i] + 1) * 10**(k - 1) - 1
	div = in_range(A[i], 0, 10**k - 1) - in_range(A[i], minb, maxb)
	ans = (ans * div) % MOD

print ans
