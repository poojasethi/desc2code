n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())

result = 1
for i in xrange(n//k):
  tot = 10 ** k // a[i] + 1
  if (10 ** k) % a[i] == 0:
    tot -= 1
  
  lesser = ((10 ** (k-1)) * b[i]) // a[i]
  if ((10 ** (k-1)) * b[i]) % a[i] == 0:
    lesser -= 1
  greater = ((10 ** (k-1)) * (b[i] + 1)) // a[i]
  if ((10 ** (k-1)) * (b[i] + 1)) % a[i] == 0:
    greater -= 1
  tot -= (greater - lesser)
  result *= tot
print result % 1000000007
