MOD=1000000007
n,k=map(int,raw_input().split())
a=map(int,raw_input().split())
b=map(int,raw_input().split())
p=int(pow(10,k))-1
ans2=1
for i in range(0,n/k):
    ans=p/a[i]
    x=1
    for j in range(1,k+1):
        if(j==k):
            ans=ans-((b[i]+x-1)/a[i])+((b[i]-1)/a[i])+1
        b[i]=b[i]*10
        x=x*10
    ans2=ans2*(ans)
    if(ans2>=MOD):
        ans2=ans2%MOD
print ans2
