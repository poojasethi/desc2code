import sys
from operator import mul
n,k=map(int, sys.stdin.readline().split())
a=map(int, sys.stdin.readline().split())
b=map(int, sys.stdin.readline().split())
res=[]
for i in range(n/k):
  t=(10**k-1)/a[i]+1
  t-=((10**(k-1)-1) - (a[i]-b[i]*10**(k-1)%a[i])%a[i]) / a[i] + 1
  res.append(t)
r=1
M=10**9+7
for i in res:
  r=r*i%M
print r
