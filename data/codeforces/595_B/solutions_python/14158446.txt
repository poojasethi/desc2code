import math

n, k = [int(i) for i in raw_input().split(' ')]
ai = [int(i) for i in raw_input().split(' ')]
bi = [int(i) for i in raw_input().split(' ')]
ans=1

for i in xrange(n/k):
    t=0
    t += (10**k - 1)/ai[i]
    t -= ((bi[i]+1)*(10**(k-1)) - 1)/ai[i]
    t += (bi[i]*(10**(k-1)) - 1)/ai[i]
    ans *= (t+1)

print ans%(10**9 + 7)
