#!/usr/bin/env python
# -*- coding: utf-8 -*-

n,k = map(int,raw_input().split())
A = list(map(int,raw_input().split()))
B = list(map(int,raw_input().split()))
ans = 1

for i in xrange(n//k):
    count = 1 + (10**k - 1)//A[i]
    count -= (10**(k-1) * (B[i] + 1) - 1) // A[i] - 10**(k-1) * B[i] // A[i]
    # if B[i] % A[i] == 0:
    #     count -= 1
    if 10**(k-1)*B[i] % A[i] == 0:
        count -= 1

    ans *= count

print(ans % int(1e9+7))
