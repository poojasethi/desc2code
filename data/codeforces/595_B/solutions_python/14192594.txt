n, k = map(int, raw_input().split())
A = map(int, raw_input().split())
B = map(int, raw_input().split())

ans = 1
for i in xrange(n/k):
    a, b = A[i], B[i]
    cnt = 1 + ((10**k-1)/a) - ((10**(k-1)*(b+1)-1)/a - (10**(k-1)*(b)-1)/a)
    #     00    38, 76           (79/38) -> 2             (69/38) -> 1
    # # of n divisible a, 0-99         # of n divisible a, 70-79 
    ans *= cnt
    ans %= 10**9+7

print ans
