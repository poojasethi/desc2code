n,k = map(int,raw_input().split())
a = map(int,raw_input().split())
b = map(int,raw_input().split())
cnt = [0 for i in xrange(n/k)]
for i in xrange(n/k):
	if b[i] == 0:
		cnt[i] = (10**k-1) / a[i] - ((b[i]+1)*(10**(k-1)) - 1) / a[i]
	else:
		all_cnt = (10**k-1) / a[i] + 1
		lb = (b[i]*(10**(k-1)) - 1) / a[i]
		ub = ((b[i]+1)*(10**(k-1)) - 1) / a[i]
		cnt[i] = all_cnt - (ub - lb)
ans = 1
for i in xrange(n/k):
	ans *= cnt[i]
print ans % (10**9 + 7)
