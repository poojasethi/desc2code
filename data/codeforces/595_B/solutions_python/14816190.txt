from math import *
from Queue import *
from sys import *
from random import *





n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
cand = [0 for i in range(n/k)]
for i in range(n/k):
    L = b[i]*10**(k-1)
    U = (b[i]+1)*10**(k-1)
    cand[i] = (10**k + a[i] - 1)/a[i] - (U + a[i] - 1)/a[i] + (L + a[i] - 1)/a[i]
ret = 1
B = 10**9+7
for i in range(n/k):
    ret = (ret*cand[i])%B
print(ret)
