mod = 10**9+7


def count(k, a, b):
    n = (int(k*'9')/a) + 1
    l = (int(b+(k-1)*'9'))/a - (int(b+(k-1)*'0')-1)/a
    return n-l

n, k = (int(i) for i in raw_input().split())
a = raw_input().split()
b = raw_input().split()
m = 1
for i in xrange(n/k):
    m *= (count(k, int(a[i]), b[i])) % mod
print m % mod
