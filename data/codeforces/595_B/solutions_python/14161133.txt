import math
n,k=map(int,raw_input().split())
a=map(int,raw_input().split())
b=map(int,raw_input().split())
val=1
mod=int(10**9+7)
for i in range(n/k):
	ctr=0
	tmp=int(10**k)
	chk=a[i]
	mult=int(math.ceil((tmp*1.0)/chk))
	mval=(10**(k-1)*b[i])%a[i]
	mval=((a[i])-mval)%a[i]
	diff=int(math.ceil((tmp/10.0-mval)/(chk)))
	if mult>=diff:
		val=(val*(mult-diff))
		val%=mod
print val
