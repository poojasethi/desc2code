from sys import stdin

n, k = map(int, stdin.readline().rstrip().split(' '))
length = n/k
a = map(int, stdin.readline().rstrip().split(' '))
b = map(int, stdin.readline().rstrip().split(' '))
result = 1

for i in xrange(length):
    n_div_a = (10**k-1)/a[i] + 1
    n_starts_b = ((b[i]+1)*(10**(k-1))-1)/a[i] - (b[i]*(10**(k-1))-1)/a[i]
    block_valid = n_div_a - n_starts_b
    result *= block_valid

result %= (10**9 + 7)
print result
