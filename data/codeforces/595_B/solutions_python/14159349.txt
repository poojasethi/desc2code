#! /usr/bin/env python

mod=1000000007
n,k=map(int,raw_input().split())
a=map(int,raw_input().split())
b=map(int,raw_input().split())
val=n/k
res=1
x=10**(k-1)
for i in range(val):
    cnt=0
    if b[i]!=0:
        cnt=(x-1)/a[i]+1
    for j in range(1,10):
        if j!=b[i]:
            start=((j*x-1)/a[i])+1
            end=((j+1)*x-1)/a[i]
            cnt+=(end-start+1)%mod
    res=(res*cnt)%mod
print res
