n, k = [int(x) for x in raw_input().split()]

a = [int(x) for x in raw_input().split()]
b = [int(x) for x in raw_input().split()]

m = n/k
ss = 10**k - 1

s = 1
for i in xrange(m):
    z = ss // a[i] + 1
    #print z
    z += (b[i]*10**(k-1) - 1) // a[i]
    z -= ((b[i]+1)*10**(k-1) - 1) // a[i]
    s *= z
    s %= (10**9 + 7)

print s
