def getn(base, head, div):
    mod = base * head % div
    if mod == 0:
        return (base - 1) / div +1
    else:
        return (base + mod-1) / div

n, k = map(int, raw_input().split())
ret = 1
base = 10 ** (k - 1)
for a, b in zip(map(int, raw_input().split()), map(int, raw_input().split())):
#    print getn(base * 10, 0, a[i]) , getn(base, b[i], a[i])
    ret *= getn(base * 10, 0, a) - getn(base, b, a)
print ret % 1000000007
