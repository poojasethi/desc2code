#include <bits/stdc++.h>
using namespace std;
#define mod 1000000007
int n,k;
int a[100005],b[100005];
int main()
{
    //freopen("tt.in", "r", stdin);
    cin>>n>>k;
    for(int i=1;i<=n/k;i++)
        scanf("%d",&a[i]);
    for(int i=1;i<=n/k;i++)
        scanf("%d",&b[i]);
    long long ans=1,tmp=1;
    for(int i=1;i<=k;i++) tmp*=10;
    for(int i=1;i<=n/k;i++)
    {
        if(b[i]==0)
            ans=ans*((tmp-1)/a[i]-(tmp/10-1)/a[i])%mod;
        else
            ans=ans*((tmp-1)/a[i]+1-(b[i]*tmp/10+tmp/10-1)/a[i]+(b[i]*tmp/10-1)/a[i])%mod;
    }
    cout<<ans<<endl;
}
