#include <bits/stdc++.h>

using namespace std;
const int MaxN = 100005;
const int Mod = 1e9 + 7;
typedef long long ll;

int N, K;
int a[MaxN], b[MaxN];

int main(){
  scanf("%d%d", &N, &K);
  for (int i = 0; i < N/K; i++)
    scanf("%d", a + i);
  for (int i = 0; i < N/K; i++)
    scanf("%d", b + i);

  int r = 1;
  for (int i = 0; i < K - 1; i++)
    r *= 10;

  ll res = 1;
  for (int i = 0; i < N/K; i++){
    ll p = (10*r - 1) / a[i]  -  ((b[i]+1)*r - 1) / a[i]  +  (b[i]*r - 1) / a[i] + 1;
    if (b[i] == 0 && a[i] != 1) p--;
    res *= p;
    res %= Mod;
  }

  if (res < 0) res += Mod;
  cout<<res<<endl;
  return 0;
}
