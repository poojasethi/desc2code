#include <bits/stdc++.h>
using namespace std;
const int N = 100000;
int a[N],b[N],ret[N];
const int mod = 1e9+7;
int main(){
	int i,j,n,k,x;cin>>n>>k;x=n/k;
	for(i=0;i<x;++i)cin>>a[i];
	for(i=0;i<x;++i)cin>>b[i];
	int v,cur,ten = 1;for(i=0;i<k-1;++i)ten*=10;
	for(i=0;i<x;++i){
		cur = ten-1;v=0;
		for(j=0;j<=9;++j){
			if(j!=b[i])
				ret[i]+=(cur/a[i])-v;
			v = cur/a[i];
			cur+=ten;
		}
		++ret[i];
		if(b[i]==0)--ret[i];
	}
	long long ans = 1;
	for(i=0;i<x;++i)ans = (ans*ret[i])%mod;
	cout<<ans<<endl;
}
