#include<bits/stdc++.h>
using namespace std;
#define D 1000000007
long long a[100010],b[100010];
int main()
{

int n,k,i;
cin>>n>>k;
long long uri=n/k;

for(i=0;i<uri;i++)
cin>>a[i];
for(i=0;i<uri;i++)
cin>>b[i];

int lastx=1;
for(i=0;i<k;i++)
lastx*=10;


long long  prr=1,div,low,high,yy2,yy1;
prr = lastx/10;  // 10^(k-1)

lastx--;

long long ug=lastx;

long long mainans=1;

for(i=0;i<uri;i++)
{

div = lastx/a[i];

div++;

low = prr*b[i];

high = low+ lastx%prr;


yy2 = high/a[i];

yy1 = (low+a[i]-1)/a[i];

div-=(yy2-yy1+1);

mainans = (mainans*div)%D;

}

cout<<mainans;





}
