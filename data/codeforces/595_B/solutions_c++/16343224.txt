#include<cstdio>

#define mo 1000000007

int n, k, ans;
int a[111111];
int b[111111];

int mul(int a, int b) { return (long long)a*b%mo; }

int main() {
	scanf("%d%d", &n, &k);
	ans = 1;
	for(int i=n/k;i>0;i--)
		scanf("%d", &a[i]);
	for(int i=n/k;i>0;i--)
		scanf("%d", &b[i]);
	for(int i=n/k;i>0;i--) {
		int x = 1;
		for(int j=0;j<k;j++) x*=10;
		
		int v = (x-1)/a[i] - ((b[i]+1)*(x/10)-1)/a[i];
		if(b[i]) v+=(b[i]*(x/10)-1)/a[i]+1;
		
		ans = mul(ans, v);
	}
	printf("%d\n", ans);
	
	return 0;
}
