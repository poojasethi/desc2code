#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long n,k,a[100000],b[100000],R = 10,C = 1;
    cin >> n >> k;
    for(int i = 1;i < k;i++){R *= 10;}
    for(int i = 0;i < n/k;i++)
    {
        cin >> a[i];
    }
    for(int i = 0;i < n/k;i++)
    {
        cin >> b[i];
    }
    for(int i = 0;i < n/k;i++)
    {
    C *= ceil(double(R)/double(a[i]))-
    (ceil(double(b[i]*(R/10)+(R/10))/double(a[i]))
    - ceil(double(b[i]*(R/10))/double(a[i])));
    C %= 1000000007;
    }
    cout << C;

return 0;}
