#include <cstdio>
#define MOD 1000000007
int a[100005], b[100005];
int main() {
  int n, k;
  long long ans = 1;
  scanf("%d %d", &n, &k);
  int d = 1;
  for (int i = 1; i < k; i++) {
    d *= 10;
  }
  int m = n / k;
  for (int i = 0; i < m; i++) {
    scanf("%d", &a[i]);
  }
  for (int i = 0; i < m; i++) {
    scanf("%d", &b[i]);
  }
  for (int i = 0; i < m; i++) {
    int x = (d * 10 - 1) / a[i];
    x -= (d * (b[i] + 1) - 1) / a[i];
    if (b[i] != 0) {
      x += (d * b[i] - 1) / a[i] + 1;
    }
    ans *= x;
    ans %= MOD;
  }
  printf("%I64d\n", ans);
  return 0;
}
