#include <iostream>
#include <fstream>
#define MOD 1000000007
#define MAXIM 100001
using namespace std;
int calc(long long a,long long b)
{
    return a/b + (a%b!=0);
}
int main()
{
    int n,k;
    long long a[MAXIM],b[MAXIM],nr,t=1;
    cin>>n>>k;
    for(int i=1;i<=n/k;i++)
        cin>>a[i];
    for(int i=1;i<=n/k;i++)
        cin>>b[i];
    long long p=1;
    for(int i=1;i<=k;i++)p=p*10;
    for(int i=1;i<=n/k;i++)
    {
        nr = calc(p,a[i])-calc((b[i]+1)*p/10,a[i])+calc(b[i]*p/10,a[i]);
        t = (t*nr)%MOD;
    }
    cout<<t;
    return 0;
}
