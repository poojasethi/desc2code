#include<iostream>
#include<bits/stdc++.h>
using namespace std;

#define md 1000000007
#define ll long long

int a[100007];
ll st[10];
int main(){
    int n,k,b;
    ll ans =1;
    scanf("%d %d", &n, &k);
    int x;
    int p; 
    p = n/k;
    st[0] = 1;
    for(int i = 1; i<10; i++)
    st[i] = st[i-1]*10;
    
    
    
    for(int i = 0; i<p; i++){
    cin>>a[i];
    
    }
    for(int i = 0; i<p; i++){
    cin>>b;
    
    ans = ans*((st[k]-1)/a[i] - ((b+1)*st[k-1]-1)/a[i] + (b*st[k-1] -1 +a[i])/a[i] )%md;
    }
    
    cout<<ans;



}
