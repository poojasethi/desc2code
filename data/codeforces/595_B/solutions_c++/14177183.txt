#include <iostream>
using namespace std;
const int mod=1e9+7;
const int N=100005;
long long ans,a[N],b[N],d,n,k;
int main(){
    cin>>n>>k;
    d=n/k;
    long long base=1,ans=1;
    for(int i=1;i<=k;i++)base*=10;
    for(int i=1;i<=d;i++)cin>>a[i];
    for(int i=1;i<=d;i++)cin>>b[i];
    for(int i=1;i<=d;i++){
        long long t;
        if(b[i]==0)t=((base-1)/a[i]-(base/10-1)/a[i]);
        else{
            t=(base-1)/a[i]-((base/10*(b[i]+1)-1)/a[i]-(base/10*b[i]-1)/a[i])+1;
        }
        ans=(ans*(t%mod))%mod;
    }
    cout<<ans<<'\n';
    return 0;
}
