#include<cmath>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
long long a[100005],b[100005],n,k,l,ans=1,m=1,c;
int main()
{
    cin>>n>>k;l=n/k;
    for(int i=0;i<k;i++)
        m*=10;
    for(int i=0;i<l;i++)
        cin>>a[i];
    for(int i=0;i<l;i++)
cin>>b[i];
    for(int i=0;i<l;i++)
    {
        c=(m-1)/a[i]-(m/10*(b[i]+1)-1)/a[i]+(b[i]!=0?(m/10*b[i]-1)/a[i]+1:0);
        ans=c*ans%1000000007;
    }
    cout<<ans<<endl;    
}
