#include <iostream>
using namespace std;
#define MOD 1000000007
long long nr[11];
int k;
long long calcul(long long x,long y){
long long a,inf,sup,b;
    a=(nr[k]-1)/x+(y!=0);
    inf=y*nr[k-1];
    sup=inf+nr[k-1]-1;
    b=(sup/x)-(inf/x);
    if(inf&&inf%x==0)
        b++;
    return a-b;
}
int main()
{long n;
cin>>n>>k;
int b[n/k+6],q=n/k;
long long a[n/k+6];
for(int i=1;i<=q;i++)
    cin>>a[i];
for(int i=1;i<=q;i++)
    cin>>b[i];
nr[0]=1;
for(int i=1;i<=k;i++)
    nr[i]=nr[i-1]*10;
long long sol=1;
for(int i=1;i<=q;i++)
    sol=(sol%MOD *(calcul(a[i],b[i])%MOD))%MOD;
cout<<sol;
return 0;
}
