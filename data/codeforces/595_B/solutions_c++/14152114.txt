#include<iostream>
using namespace std;
long long int n,k,a[100006],b[100006],c1=1,bank=1,e1,x1,t,c2;
const long long int mod=1000000007;
main()
{
    cin>>n>>k;
    int q=n/k;
    for(int i=0;i<q;i++)
        cin>>a[i];
    for(int i=0;i<q;i++)
        cin>>b[i];
    for(int i=1;i<k;i++)
        c1*=10;//10^(k-1)
    c2=c1*10;
    for(int i=0;i<q;i++)
    {
        t=(b[i]*c1)%a[i];
        t=(a[i]-t)%a[i];
        e1=(c1-1-t)/a[i];
        if(t<c1)
            e1++;
        //x1=(c2-1)-((c2-1)/a[i])-(c1-e1);
        x1=(c2-1)/a[i]+1-e1;
        bank=(bank*x1)%mod;
    }
    cout<<bank;
}
