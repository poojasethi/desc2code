#include <iostream>

using namespace std;
long long n,k,mod=1000000007,a[100010],b[100010],mu[10],res[100010];
long long ans;

int main()
{
    cin>>n>>k;
    for (long long i=1;i<=n/k;++i) cin>>a[i];
    for (long long i=1;i<=n/k;++i) cin>>b[i];
    mu[0]=1;
    for (long long i=1;i<10;++i) mu[i]=mu[i-1]*10;
    ans=1;
    for (long long i=1;i<=n/k;++i)
    {
        res[i]=(mu[k]-1)/a[i]-((b[i]+1)*mu[k-1]-1)/a[i] +max(0ll,(b[i]*mu[k-1]-1)/a[i]);
        if (b[i]!=0) res[i]++;
        res[i]%=mod;
        ans*=res[i];
        ans%=mod;
    }
    ans%=mod;
    cout<<ans;
    return 0;
}
