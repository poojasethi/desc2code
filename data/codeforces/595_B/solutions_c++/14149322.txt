#include <stdio.h>
int main(){
    int n,k;
    scanf("%d%d",&n,&k);
    long long d=1;
    for (int i=1; i<k; i++)
        d*=10;

    int a[100000], b[100000];
    for (int i=0; i<n/k; i++)
        scanf("%d",&a[i]);
    for (int i=0; i<n/k; i++)
        scanf("%d",&b[i]);

    long long ans=1;
    for (int i=0; i<n/k; i++){
        long long c1=(1ll*b[i]+1)*d-1, c2=(1ll*b[i])*d-1;
        long long qans=(1ll*10*d-1)/a[i];
        qans-= c1/a[i];
        if (b[i]!=0)
           qans+= c2/a[i];
        if (b[i]!=0)
           qans++;
        ans=(ans*qans)%(1000*1000*1000+7);
    }
    printf("%d",ans);
}
