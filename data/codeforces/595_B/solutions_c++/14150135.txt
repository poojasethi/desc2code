#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL mod=1000000007LL;
LL ans=1,t;
int n,k;
int a[100005],b[100005];
LL f(int a,int b)
{
    LL tt,ans=t/a+1,p=t/10+1;
    tt=p*b;
    ans-=(tt+p-1)/a;
    if(tt)ans+=(tt-1)/a;
    if(!tt)--ans;
    return ans;
}
int main()
{
    scanf("%d%d",&n,&k);
    int i,j=n/k;
    t=9LL;
    for(i=1;i<k;++i)
        t=t*10+9;
    for(i=1;i<=j;++i)
        scanf("%d",a+i);
    for(i=1;i<=j;++i)
        scanf("%d",b+i);
    for(i=1;i<=j;++i)
        ans=ans*f(a[i],b[i])%mod;
    printf("%I64d\n",ans);
    return 0;
}
