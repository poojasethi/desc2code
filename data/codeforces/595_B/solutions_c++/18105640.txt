#include<stdio.h>
#define MOD 1000000007

long long num[11];

void fun()
{
	int i;
	num[0]=1;
	for(i=1;i<=10;i++) num[i]=num[i-1]*10;
}

int main()
{
	int n,k,i;
	long long a[100005],b[100005],ans=1,num1,num2,num3;
	fun();
	scanf("%d%d",&n,&k);
	for(i=0;i<n/k;i++) scanf("%lld",&a[i]);
	for(i=0;i<n/k;i++) scanf("%lld",&b[i]);
	for(i=0;i<n/k;i++)
	{
		num1=(num[k]-1)/a[i]+1;
		num2=(num[k-1]-1)/a[i]+1;
		num3=(num[k-1]*(b[i]+1)-1)/a[i]-(num[k-1]*b[i]-1)/a[i];
		if(b[i]) ans*=(num1-num3);
		else ans*=(num1-num2);
		ans%=MOD;
	}
	printf("%lld\n",ans);
	return 0;
}
