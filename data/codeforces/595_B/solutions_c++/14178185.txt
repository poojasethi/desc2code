#include<iostream>
using namespace std;
int main()
{
	long long int n,k,m=1000000007LL;
	cin>>n>>k;
	long long int z=n/k;
	long long int a[z],b[z],i,c=1,d=1,e,f,y;
	for(i=0;i<k;i++)
	{
		d=d*10;
	}
	for(i=0;i<z;i++)
	{
		cin>>a[i];
	}
	for(i=0;i<z;i++)
	{
		cin>>b[i];
	}
	for(i=0;i<z;i++)
	{
		e=(d+a[i]-1)/a[i];
		y=(-1)*(((d/10)*b[i])%a[i]);
		while(y<0)
		{
			y=y+a[i];
		}
		f=((d/10)-1-y+a[i])/a[i];
		c=(c*(e-f))%m;
	}
	cout<<c<<endl;
	return 0;
}
