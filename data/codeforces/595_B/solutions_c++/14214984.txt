#include<cstdio>
using namespace std;
#define ll long long
#define md 1000000007
ll p[10];int a1[100000];
int main()
{
    p[0]=1;for(int i=1;i<10;++i)p[i]=p[i-1]*10;
    int n,k,m,a;ll o=1;
    scanf("%d%d",&n,&k);m=n/k;
    for(int i=0;i<m;++i)scanf("%d",a1+i);
    for(int i=0;i<m;++i)
    {
        scanf("%d",&a);
        o=o*((p[k]-1)/a1[i]-((a+1)*p[k-1]-1)/a1[i]+(a*p[k-1]+a1[i]-1)/a1[i])%md;
    }
    printf("%d\n",o);
}
