#include <bits/stdc++.h>
using namespace std;
#define N 100005
#define M 1000000007
typedef long long int LL;

int main() {
	int n, k, i, nk;
	LL a[N], b[N], ans = 1, dp[N], pk = 1, nn, pn;
	cin >> n >> k;
	nk = n/k;
	for(i = 0; i < k; i++) pk *= 10;
	nn = pk - 1;
	pk /= 10;
	for(i = 0; i < nk; i++) cin >> a[i];
	for(i = 0; i < nk; i++) cin >> b[i];
	for(i = 0; i < nk; i++) {
		pn = (nn / a[i] + 1);
		if(b[i] == 0) pn = pn - ((pk - 1) / a[i] + 1);
		else pn = pn - ((pk * (b[i] + 1) - 1) / a[i] - (pk * b[i] - 1) / a[i]);
		ans = ans * pn % M;
	}
	cout << ans << endl;
}
