#include <cstdio>
using namespace std;

int main() {
    const int mod = 1e9+7;
    int n, k, a[111111], b[111111], t;
    long long c = 1;
    scanf("%d%d", &n, &k);
    for(int i=0; i*k<n; ++i) scanf("%d", &a[i]);
    for(int i=0; i*k<n; ++i) scanf("%d", &b[i]);

    t = 1; for(int i=1; i<k; ++i) t = t*10;
    for(int i=0; i*k<n; ++i) {
        if (b[i] != 0) c = (c * ((t*10-1)/a[i] - ((b[i]+1)*t-1)/a[i] + (b[i]*t-1)/a[i] + 1)) % mod;
        else c  = (c * ((t*10-1)/a[i] - (t-1)/a[i])) % mod;
    }
    printf("%lld\n", c);
}
