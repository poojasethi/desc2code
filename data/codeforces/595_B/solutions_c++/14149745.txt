#include<stdio.h>
long long n,k,a[100010],b[100010],s=1;
long long t,ans=1,dap,p,q;
int main()
{
    long long i;

    scanf("%lld %lld",&n,&k);
    for(i=1;i<=k;i++) s*=10;
    for(i=1;i<=n/k;i++) scanf("%lld",&a[i]);
    for(i=1;i<=n/k;i++) scanf("%lld",&b[i]);
    for(i=1;i<=n/k;i++)
    {
        dap=(s-1)/a[i]+1; q=0;
        p=(s/10)*b[i];
        q=a[i]*(p/a[i]);
        if(p%a[i]!=0) q+=a[i];

        t=p+(s/10)-1-q;
        if(t>=0){dap--; dap-=(t/a[i]);}
        dap%=1000000007;
        ans*=dap;
        ans%=1000000007;
    }
    printf("%lld",ans);
    return 0;
}
