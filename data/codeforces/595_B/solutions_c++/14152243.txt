#include<bits/stdc++.h>
using namespace std;
int main(){
	long long int i,j,mod=1000000007,e,dbg=0;
	long long int ans=1,n,k,maxe=0,maxi=1,a[100005],b[100005],ansa[100005];
	cin>>n>>k;
	for(i=0;i<n/k;i++)
		cin>>a[i];
	for(j=0;j<n/k;j++)
		cin>>b[j];
	for(i=0;i<k-1;i++)
		maxi*=10;
	for(i=0;i<n/k;i++){
		if(b[i]) {
			e=b[i]*maxi;
			e--;
			ansa[i]=e/a[i];
		}
		if(b[i]!=9) {
			e=(b[i]+1)*maxi;
			e--;
			ansa[i]=e/a[i]-ansa[i];
			e=maxi*10;
			e--;
			ansa[i]=e/a[i]-ansa[i];
		}
		if(b[i])
			ansa[i]++;
	}
	for(i=0;i<n/k;i++)
		ans=(ans*ansa[i])%mod;
	cout<<ans<<endl;
	return 0;
}
