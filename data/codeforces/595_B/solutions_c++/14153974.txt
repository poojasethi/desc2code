#include<cstdio>

long long ans=1,n,k;
long long a[100001],b[100001],mi[10];

long long make(long long ai,long long bi,long long tiao)
{
	long long ans=(tiao-1)/ai+1,l=bi*(tiao/10),r=(bi+1)*(tiao/10)-1,tot=l/ai*ai;
	if(tot<l) tot+=((l-tot)/ai+1)*ai;
	if(tot>=l&&tot<=r) ans-=(r-tot)/ai+1;
	return ans;
}

int main()
{
	scanf("%I64d%I64d",&n,&k);
	for(int i=1;i<=n/k;i++) scanf("%I64d",&a[i]);
	for(int i=1;i<=n/k;i++) scanf("%I64d",&b[i]);
	mi[0]=1;
	for(int i=1;i<=9;i++) mi[i]=mi[i-1]*10;
	for(int i=1;i<=n/k;i++) ans=(ans*(make(a[i],b[i],mi[k])%1000000007))%1000000007;
	printf("%I64d",ans);
	return 0;
}
