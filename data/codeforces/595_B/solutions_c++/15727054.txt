#include<bits/stdc++.h>
using namespace std;
int a[100000],b;
int main()
{
    int i,n,k;
    scanf("%d%d",&n,&k);
    for(i=0; i<n/k; i++)
        scanf("%d",&a[i]);
    __int64 gg,ans=1;
    for(i=0; i<n/k; i++)
    {
        scanf("%d",&b);
        int sign=k,flag=1;
        while(sign--) flag*=10;
        gg=(flag-1)/a[i];
        if(b!=0)
        {
            gg++;
            gg-=((b+1)*(flag/10)-1)/a[i]-(b*(flag/10)-1)/a[i];
        }
        else gg-=(flag/10-1)/a[i];
        ans=(ans*gg)%1000000007;
    }
    cout<<ans<<endl;
    return 0;
}
