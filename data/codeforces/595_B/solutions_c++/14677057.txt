#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int c[10]={1,10,100,1000,10000,100000,1000000,10000000,100000000,1000000000};
    long long int count=1,n,k,M=1000000007;
    cin>>n>>k;
    long long int m,r,x,i,j,a[n],p,q;
    for(i=0;i<n/k;i++)
        cin>>a[i];
    for(i=0;i<n/k;i++)
    {
        m=0;
        cin>>x;
        p=(c[k]-1)/a[i];
        q=(x*c[k-1] + c[k-1] -1)/a[i];
        m=p-q;
        r=(x*c[k-1] -1)/a[i];
        if(r>0)
            m+=r;
        if(x)
            m++;
        count=((count%M)*(m%M))%M;
    }
    cout<<count;
    return 0;
}
