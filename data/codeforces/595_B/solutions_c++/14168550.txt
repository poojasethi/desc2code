#include <iostream>
#define md 1000000007
using namespace std;
typedef __int64 ull;
ull power(ull  n);

int main() {
    ull n, k, m, a[100000], b[100000], i, s = 1, p, d, t;
    cin >> n >> k;
    m = n / k;
    for(i = 0; i < m; i++) cin >> a[i];
    for(i = 0; i < m; i++) cin >> b[i];
    d = power(k - 1);
	t = d * 10 - 1;
    for(i = 0; i < m; i++) {
		p = t / a[i] - (d * (b[i] + 1) - 1) / a[i];
		if(b[i] != 0) p = p + (d * (b[i]) - 1) / a[i] + 1;
		s = s * p;
		s %= md;    
	}
    cout << s;
}

ull power(ull n) {
    if(n == 0) return 1;
    return power(n - 1) * 10;
}
