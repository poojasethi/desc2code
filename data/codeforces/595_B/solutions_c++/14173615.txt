#include <bits/stdc++.h>

using namespace std;

const int base = 1e9+7;
int a[100010], b[100010];

int main() {
	//freopen("input.txt","r",stdin);
	int n, k;
	cin>>n>>k;
	for (int i=0; i<n/k; i++) {
		cin>>a[i];
	}
	for (int i=0; i<n/k; i++) {
		cin>>b[i];
	}
	long long res=1;
	for (int i=0; i<n/k; i++) {
		int x=9, y=b[i], z=b[i]-1;
		for (int j=1; j<k; j++) {
			x=x*10+9;
			y=y*10+9;
			z=z*10+9;
		}
		if (z==-1) res=res*(x/a[i]-y/a[i])%base;
		else res=res*(x/a[i]+1-y/a[i]+z/a[i])%base;
	}
	cout<<res<<endl;
	return 0;
}
