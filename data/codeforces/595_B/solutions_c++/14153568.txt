#include<bits/stdc++.h>

using namespace std;

#define mp make_pair
#define pb push_back 
#define nd second
#define st first

typedef long long int lli;
const int MAX = 1e5+7;
const int mod = 1e9+7;

lli ans=1;
lli n,k;
lli a[MAX],b[MAX];

int main ()
{
	cin>>n>>k;
	for(int i=1; i<=n/k; i++)
		scanf("%lld",a+i);

	for(int i=1; i<=n/k; i++)
		scanf("%lld",b+i);
	
	lli us=1;
	for(int i=1; i<=k; i++)
		us*=10;
	
	for(int i=1; i<=n/k; i++)
	 {
		 lli s = ((us-1)/a[i]) + 1;
		 s -= (((us/10))*(b[i]+1)-1) / a[i] + 1;
		 if(b[i])s += ((us/10)*b[i]-1)/a[i] +1;
		 ans *= (s) %mod;
		 ans %= mod;
	 }
	cout<<ans;
}
