#include <iostream>
using namespace std;
long long n,k,m,a[100005],b[100005],ten[10];
long long ans=1;
int main() {
    cin>>n>>k;
    m=n/k;
    for(int i=0;i<m;i++){
        cin>>a[i];
    }
    for(int i=0;i<m;i++){
        cin>>b[i];
    }
    ten[0]=1;
    for(int i=1;i<10;i++)ten[i]=ten[i-1]*10;
    for(int i=0;i<m;i++){
        long long tmp = (ten[k]-1)/a[i]-(ten[k-1]*(b[i]+1)-1)/a[i]+(max(ten[k-1]*(b[i])-1,(long long)0))/a[i];
        if(b[i]!=0)tmp++;
        ans=(ans*tmp)%1000000007;
    }
    cout<<ans;
}
