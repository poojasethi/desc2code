#include<stdio.h>
#define mod 1000000007
#define maxn 100005
int a[maxn],b[maxn];
int main(){
    int n,k;
    scanf("%d%d",&n,&k);
    int m=n/k;
    for(int i=1;i<=m;i++)scanf("%d",&a[i]);
    for(int i=1;i<=m;i++)scanf("%d",&b[i]);
    int lim=1;
    for(int i=1;i<k;i++)lim*=10;
    long long ans=1;
    for(int i=1;i<=m;i++){ 
        int tot=(lim*10-1)/a[i]+1;
        if(b[i]){
            int sb=((b[i]+1)*lim-1)/a[i]-(b[i]*lim-1)/a[i];
            ans=(ans*(tot-sb)%mod);
        }
        else ans=(ans*(tot-1-(lim-1)/a[i]))%mod;
    }
    printf("%I64d\n",ans);
    return 0;
}
