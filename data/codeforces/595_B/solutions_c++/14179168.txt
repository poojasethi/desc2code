#include <bits/stdc++.h>
using namespace std;

const int mod = 1e9+7;
int n, k, m, a[100005], b, total, bless, cur;
long long ans = 1;

int ten[] = {1e0,1e1,1e2,1e3,1e4,1e5,1e6,1e7,1e8,1e9};
inline int ndiv(int n, int d) { return n / d + 1; }

int main() {
  scanf("%d%d", &n, &k);
  m = n/k;
  for (int i = 0; i < m; i++) scanf("%d", &a[i]);
  for (int i = 0; i < m; i++) {
    scanf("%d", &b);
    total = ndiv(ten[k]-1, a[i]);
    bless = ndiv((b+1)*ten[k-1]-1, a[i]);
    cur = total - bless;
    if (b) cur += ndiv(b*ten[k-1]-1, a[i]);
    ans = (ans*cur) % mod;
  }
  printf("%d\n", (int)ans);
}
