#include <cstdio>
#define MAX 100010
#define MOD 1000000007
long long A[MAX];
long long P[MAX];
int main(){
    long long a,b,c,d,e,res=1;
    P[0]=1;
    for(a=1;a<MAX;a++){
        P[a]=(P[a-1]*10)%MOD;
    }
    scanf("%lld %lld",&a,&b);
    for(c=0;c<a/b;c++){
        scanf("%lld",&A[c]);
    }
    for(res=1,c=0;c<a/b;c++){
        scanf("%lld",&d);
        e=(P[b]-1)/A[c]+1;
        e-=(P[b-1]*(d+1)-1)/A[c]+1;
        if(d!=0){
            e+=(P[b-1]*d-1)/A[c]+1;
        }
        e=e%MOD;
        res=(res*e)%MOD;
    }
    printf("%lld\n",res);
}
