#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll n,k,a[100009],b[100009],ans=1;
#define mod 1000000007
int main()
{
	cin>>n>>k;
	for(int i=1;i<=n/k;i++)cin>>a[i];
	for(int i=1;i<=n/k;i++)cin>>b[i];
	for(int i=1;i<=n/k;i++)
	{
		ll p=1;
		for(int j=1;j<=k;j++)p*=10;
		ans*=((p-1)/a[i]-((b[i]+1)*(p/10)-1)/a[i]+(0,b[i]*(p/10)-(a[i]==1&&b[i]==0?0:1))/a[i]+(b[i]==0?0:1))%mod;
		ans%=mod;
		//cout<<"p="<<p<<" ans="<<ans<<endl;
	}
	cout<<ans;
	return 0;
}
