#include<bits/stdc++.h>
using namespace std;
typedef long long int lint;
#define MOD 1000000007
lint onk=1,t,k,n,a[1000000],bi,ans=1,cnt;
int main(){
	scanf("%lld %lld",&n,&k);
	t=k;
	while(t){
		onk*=10;
		t--;
	}
	for(lint i=1 ; i<=n/k ; i++){
		scanf("%lld",&a[i]);
	}
	for(lint i=1 ; i<=n/k ; i++){
		scanf("%lld",&bi);
		cnt=(onk-1)/a[i];
	//	printf("i-->%lld %lld\n",i,cnt);
		cnt-=((onk/10)*bi+(onk/10)-1)/a[i]-((bi==0) ? 0 : (((onk/10)*bi-1)/a[i]));
	//	printf("i-->%lld %lld\n",i,cnt);
		if(bi)
			cnt++;
		ans*=cnt;
		ans%=MOD;
	}
	printf("%lld\n",ans);
	return 0;
}
