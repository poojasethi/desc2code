#include<iostream>
#include<queue>
#include<algorithm>
using namespace std;
typedef long long ll;
ll x[1005],y[1005];
ll a[1000005],b[1000005];
priority_queue<ll> q,qempty;
ll inf=0x3f3f3f3f;
int main()
{
	ll ans=-inf*inf;
	ll n,m,k,p,r;
	cin>>n>>m>>k>>p;
	for(ll i=1;i<=n;i++)
	for(ll j=1;j<=m;j++)
	{
		cin>>r;
		x[i]+=r;
		y[j]+=r;
	}
	for(ll i=1;i<=n;i++)
	q.push(x[i]);
	for(ll i=1;i<=k;i++)
	{
		ll o=q.top();
		q.pop();
		a[i]=a[i-1]+o;
		q.push(o-p*m);
	}
	q=qempty;
	for (ll i=1;i<=m;i++)
	q.push(y[i]);
	for (ll i=1;i<=k;i++)
	{
		ll o=q.top();
		q.pop();
		b[i]=b[i-1]+o;
		q.push(o-p*n);
	}
	for(ll i=0;i<=k;i++)
	ans=max(a[i]+b[k-i]-(k-i)*i*p,ans);
	cout<<ans<<endl;
}
