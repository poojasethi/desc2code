#!/usr/bin/python

import sys

H , M = map(int, raw_input().split(":"))

while True:
	M += 1
	if M == 60:
		H = (H + 1) % 24
		M = 0
	if H < 10:
		HH ='0' +  str(H)
	else:
		HH = str(H)
	if M < 10:
		MM ='0' +  str(M)
	else:
		MM = str(M)
	if HH == MM[::-1]:
		print "%02s:%02s" % (HH,MM)
		sys.exit(0)
