#!/usr/bin/env python
# -*- coding: UTF-8 -*-
__author__ = 'firago'
time = raw_input()
hh, mm = time.split(':')
HH = hh
while True:
    rhh = hh[::-1]
    if int(rhh) < 60:
        if HH != hh or int(mm) < int(rhh):
            print '%s:%s' % (hh,rhh)
            break
    nhh = int(hh) + 1
    if nhh > 23:
        nhh = 0
    hh = '%02d' % nhh
