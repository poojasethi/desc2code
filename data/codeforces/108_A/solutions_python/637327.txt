tm = raw_input()


h = int(tm[:tm.find(':')])
m = int(tm[tm.find(':')+1:])

while 1:
    m = (m+1)%60
    if m == 0:
        h = (h+1)%24
    
    if h%10 == m/10 and h/10 == m%10:
        break
    

if len(str(h))<2:
    h = '0'+str(h)
if len(str(m))<2:
    m = '0'+str(m)

print str(h)+':'+str(m)
