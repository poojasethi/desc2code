def main():
	h, m = map(int, raw_input().split(":"))
	if 23 == h and m >= 32: print "00:00"
	else:
		for i in xrange(h, 24):
			for j in xrange(0 if i > h else m + 1, 60):
				a, b = i / 10, i % 10
				c, d = j / 10, j % 10
				if a == d and b == c:
					print "%d%d:%d%d" %(a, b, c, d)
					return	
 
if __name__ == "__main__": main()
