st = raw_input().split(':')
h = int(st[0])
m = int(st[1])
while True:
   m += 1
   if m >=60:
      h += 1
      m = 0
   if h >=24:
     h = 0
   mm = str(m)
   hh = str(h)
   if len(mm)==1:
     mm = "0"+mm
   if len(hh)==1:
     hh = "0"+hh
   m2=mm[::-1]
   if(hh == m2):
      break
print hh+":"+mm
