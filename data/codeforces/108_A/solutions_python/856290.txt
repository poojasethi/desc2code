h, m = raw_input().split(":")
rh = h[::-1]
if int(rh) < 60 and int(m) < int(rh):
    print "%02d:%02d" % (int(h), int(rh))
else:
    while 1:
        h = str((int(h)+1)%24)
        h = "0" + h if len(h) == 1 else h
        nm = h[::-1]
        if int(nm) < 60:
            break
    print "%02d:%02d" % (int(nm[::-1]), int(nm))
