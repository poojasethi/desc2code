s = raw_input()
hh = s[:2]
mm = s[3:]
hh = int(hh)
mm = int(mm)

sh = ''
sm = ''
while True:
  mm += 1
  if mm >= 60:
    mm = 0
    hh += 1
  if hh >= 24:
    hh = 0

  sh = str(hh)
  sm = str(mm)
  if len(sh) < 2: sh = '0' + sh
  if len(sm) < 2: sm = '0' + sm

  if sh == sm[::-1]:
    break

ans = sh + ':' + sm
print ans
