#include <iostream>
#include <cstdio>

using namespace std;

int main(){
	int hh, mm, t;
	char c;
	cin >> hh >> c >> mm;
	
	t = hh * 60 + mm;
	
	do {
                t = (t + 1) % (24*60);
        } while(t/60/10 != t%60%10 || t/60%10 != t%60/10);
	
	printf("%02d:%02d\n", t/60, t%60);
	
	return 0;
}
