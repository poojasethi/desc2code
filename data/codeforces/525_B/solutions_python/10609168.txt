s = raw_input()
m = raw_input()
tmp = [int(i) for i in raw_input().split()]

s = [s[i] for i in range(len(s))]
revs = [0]*len(s)

for i in tmp:
    revs[i-1] += 1

for j in range(len(s)/2):
    if revs[j]%2 == 1:
        s[j], s[len(s)-j-1] = s[len(s)-j-1], s[j]
    revs[j+1] += revs[j]%2
    
print ''.join(s)
