s=list(raw_input())
n=input()
c=0
d=sorted(map(int,raw_input().split()))
from collections import defaultdict as dd
C=dd(int)
for i in d:
    C[i]+=1
    
R = set(map(lambda (k,v):k, filter(lambda (k,v):v %2, C.items())))

L=len(s)
for i in range(len(s)/2):
    if i+1 in R:
        c = 1 - c
    if c:
        s[i],s[L- i-1]=s[L- i-1],s[i]
print ''.join(s)
