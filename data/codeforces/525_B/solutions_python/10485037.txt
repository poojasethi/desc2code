s = raw_input()
m = input()
a = map(int, raw_input().split())
b = []
for i in xrange(0, len(s)):
    b.append(0)
for i in xrange(0, m):
    b[a[i]-1]+=1
su = 0
for i in xrange(0, len(s)//2):
    su += b[i]
    b[i] = su
for i in xrange(0, len(s)//2):
    if b[i]%2:
        st1 = s[i]
        st2 = s[len(s)-i-1]
        s = s[:i]+st2+s[i+1:len(s)-i-1]+st1+s[len(s)-i:]
print s
