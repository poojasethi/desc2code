present = list(raw_input())

m = int(raw_input())
a = map(int,raw_input().split())

counts = [0]*(len(present)/2)
for num in a:
    counts[num - 1] += 1 
su = 0
for i in range(len(counts)):
    su += counts[i]
    if su & 1:
        present[i], present[-(i + 1)] = present[-(i + 1)], present[i] 

print "".join(present)
