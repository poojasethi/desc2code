s = list(raw_input())

m = input()

A = map(int, raw_input().split())
n = len(s)/2
dp = [False for _ in xrange(n)]

for num in A:
    dp[num-1] = not dp[num-1]

for i in xrange(1, n):
    if dp[i-1]:
        dp[i] = not dp[i]

for i in range(n):
    if dp[i]:
        s[i], s[len(s)-i-1] = s[len(s)-i-1], s[i]

print "".join(s)
