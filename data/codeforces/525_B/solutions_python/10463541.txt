s = raw_input()
s = list(s)
l = len(s)
m = input()
m = map(int, raw_input().split())

from collections import Counter
co = Counter(m)

flip = False
out = [''] * l

for k in xrange(l // 2):
    if co[k+1] % 2 == 1:
        flip = not flip
    if flip:
        out[k] = s[l-k-1]
        out[l-k-1] = s[k]
    else:
        out[k] = s[k]
        out[l-k-1] = s[l-k-1]
if l % 2 == 1:
    out[l//2] = s[l//2]
print ''.join(out)
