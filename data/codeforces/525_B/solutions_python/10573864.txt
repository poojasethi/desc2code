pstr = raw_input()
chars = len(pstr)
m = int(raw_input())
freq = dict((i, 0) for i in range(chars/2))
posns = [int(n) for n in raw_input().split()]
for i in posns:
    freq[i - 1] += 1

swaps = 0
pstr = list(pstr)
for i in range(chars/ 2):
    if (swaps + freq[i]) % 2 == 1:
        pstr[i], pstr[chars-i-1] = pstr[chars-i-1], pstr[i]
    swaps += freq[i]

print "".join(pstr)
