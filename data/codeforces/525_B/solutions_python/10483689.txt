S=raw_input()
m=int(raw_input())
revs=map(int,raw_input().split())
N=len(S)
counts=[False]*((N+1)/2)
for i in range(m):
	counts[revs[i]-1]=not counts[revs[i]-1]

res=[None]*N
curr=False
for i in range((N+1)/2):
	if counts[i]:
		curr=not curr
	res[i]=S[i] if not curr else S[N-1-i]

curr=False
for i in range(N-1,(N+1)/2-1,-1):
	if counts[N-i-1]:
		curr=not curr
	res[i]=S[i] if not curr else S[N-1-i]

print "".join(res)
