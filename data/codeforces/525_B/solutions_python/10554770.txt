line = raw_input()
no = input()
indexes = map(int, raw_input().split())
l = len(line)
line = list(line)
indexes = sorted(indexes)
d = [0] * 100000
for n in indexes:
    d[n-1]+= 1
sum = 0
for n in range(0, l // 2):
    sum += d[n]
    if sum % 2 == 1:
        left = n
        right = l - n - 1
        line[left], line[right] = line[right], line[left]



print "".join(line)
