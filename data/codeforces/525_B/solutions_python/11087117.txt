import sys
s=raw_input()
n=input()
a=[]
c=[]
cnt=[0]*100005
a=raw_input().split(' ')
for i in range(0,n):
	x=int(a[i])-1
	cnt[x]+=1
l=len(s)
for i in range(0,l):
	c.append(s[i])
for i in range(0,l/2):
	cnt[i+1]+=cnt[i]
for i in range(0,l/2):
	if(cnt[i]&1):
		t=c[i]
		c[i]=c[l-i-1]
		c[l-i-1]=t
for i in range(0,l):
	sys.stdout.write("%c" % c[i])
