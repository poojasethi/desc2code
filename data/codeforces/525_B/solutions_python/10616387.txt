import numpy as np

s = [c for c in raw_input()]
m = int(raw_input())
a = raw_input().split()

l1 = len(s)
l = l1 / 2
l1 -= 1
tally = np.zeros(l)


for i in xrange(m):
    tally[int(a[i])-1] += 1

flips = 0

for j in xrange(l):
    flips += tally[j]
    if flips % 2 == 1:
        s[j],s[l1-j] = s[l1-j], s[j]

print ''.join(s)
