s = list(raw_input())
n = int(raw_input())
ops = map(int,raw_input().split())

p = [0]*(len(s)+5)

for x in ops:
    p[x-1]+=1
    p[len(s)-x+1]-=1

sp = [p[0]%2]
for y in p[1:]:
    sp.append((sp[-1]+y)%2)

for i in range(len(s)//2):
    if sp[i]==1:
        s[i],s[len(s)-i-1] = s[len(s)-i-1],s[i]

print ''.join(s)
