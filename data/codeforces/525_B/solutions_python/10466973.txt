s=list(raw_input())
m=input()
a=list(map(int,raw_input().split()))
f=[0]*(len(s)/2)
sw=[0]*(len(s)/2)
for i in xrange(m):
    f[a[i]-1]^=1
sw[0]=f[0]
for i in xrange(1,len(sw)):
    sw[i]=f[i]^sw[i-1]
for i in xrange(len(sw)):
    if(sw[i]&1):
        s[i],s[len(s)-1-i]=s[len(s)-1-i],s[i]
print "".join(s)
