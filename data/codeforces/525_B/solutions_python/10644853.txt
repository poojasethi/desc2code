string = raw_input()

days=int(raw_input())
trans=raw_input().split()
trans=[int(x) for x in trans]

slist=[x for x in string]

l=len(slist)/2

tlist=[0 for x in xrange(l)]

for x in trans:
    tlist[x-1]+=1

sum=0

for x in xrange(l):
    sum+=tlist[x]
    if sum%2==1:
        slist[x],slist[-x-1]=slist[-x-1],slist[x]
    
print ''.join(slist)
