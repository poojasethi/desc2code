S = [c for c in raw_input()]
n = int(raw_input())
k = map(int, raw_input().split())
k = sorted(k)

rev = [False] * len(S)
for c in k:
	c -= 1
	rev[c] = not rev[c]
stat = [False] * len(S)
current = False
for i in xrange(len(S) / 2):
	if rev[i]:
		current = not current
	stat[i] = current


for i in xrange(len(S) / 2):
	if stat[i]:
		last = len(S) - i - 1

		S[i], S[last] = S[last], S[i]
print "".join(S)
