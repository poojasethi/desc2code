a=raw_input().strip()
b=[0 for i in xrange(len(a)+5)]
ans=['' for i in xrange(len(a)+5)]
k=len(a)
raw_input()
for i in map(int,raw_input().split()):
    b[i]+=1
    b[k-i+2]-=1
for i in xrange(len(a)):
    b[i+1]+=b[i]
    if b[i+1]%2:
        ans[k-i-1]=a[i]
    else:
        ans[i]=a[i]
print ''.join(ans)
