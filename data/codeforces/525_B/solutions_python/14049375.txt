s = list(raw_input())
m = int(raw_input())
a = sorted(map(int, raw_input().split()))
swaps = len(s) * [0]
for i in xrange(m):
    swaps[a[i] - 1] += 1
for i in xrange(1, len(s) / 2):
    swaps[i] += swaps[i - 1]
for i in xrange(len(s) / 2):
    if swaps[i] % 2:
        s[i], s[-i - 1] = s[-i - 1],s[i]
print "".join(s)
