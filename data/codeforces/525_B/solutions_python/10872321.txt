s = list(raw_input())
n = int(raw_input())
index_list = map(int, raw_input().split(' '))
index_list.sort()
length = len(s)

s = list(s)
pre_cnt, now_cnt = 0, 0
i, j = 0, 1 
while j <= length/2:
    now_cnt = pre_cnt
    while i < n and index_list[i] == j:
        now_cnt += 1
        i += 1
    if now_cnt&1:
        s[j-1], s[length-j] = s[length-j], s[j-1]
    j += 1
    pre_cnt = now_cnt


print ''.join(s)
