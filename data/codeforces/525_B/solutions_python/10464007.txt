s = list(str(raw_input()))
m = int(raw_input())
As = map(int, raw_input().split())
indx_prt = [0]*200000
for i in As:
    indx_prt[i] = indx_prt[i] ^ 1

for i in range(1, 200000):
    indx_prt[i] = indx_prt[i] ^ indx_prt[i-1]

for i in range((len(s)-2)/2 + 1):
    if indx_prt[i+1]:
        p = s[i]
        q = s[len(s)-i-1]
        s[len(s)-i-1] = p
        s[i] = q

print "".join(s)
