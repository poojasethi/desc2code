import sys

from collections import Counter


s = list(sys.stdin.next().strip())
n = len(s)

m = int(sys.stdin.next())

ops = Counter(map(lambda x: int(x) - 1, sys.stdin.next().split(" ")))

parity = (n / 2) * [0]
c = 0

for i in range(len(parity)):
    if i in ops:
        c += ops[i]
    parity[i] = c

new_s1 = ''
new_s2 = ''

for i, p in enumerate(parity):
    if p % 2 == 1:
        s[i], s[n - i - 1] = s[n - i - 1], s[i]

print("".join(s))
