from sys import stdin, stdout 
def main():
    s = stdin.readline().strip()
    n = int(stdin.readline())
    a = map(int, stdin.readline().split())
    l = len(s)
    t = [0] * (l + 1)
    for x in a:
        t[x-1] += 1
        t[l-x+1] -= 1
    k = 0
    ans = []
    for i in xrange(l):
        k += t[i]
        if k % 2:
            ans.append(s[-i-1])
        else:
            ans.append(s[i])
    stdout.write(''.join(ans))
main()
