#include<iostream>
#include<string.h>
#include<algorithm>
using namespace std;
int b[100010];
int main()
{
	char a[200010];
	cin>>a;
	int m;
	cin>>m;
	for (int i=1;i<=m;i++)
		cin>>b[i];
	sort(b,b+m+1);
	int t=0;
	for (int i=0;i<strlen(a)/2;i++)
		{
			while (b[t+1]<=i+1&&t+1<=m) t++;
			if (t%2) swap(a[i],a[strlen(a)-i-1]);
		}
	cout<<a;
}
