#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
bool b1[200001];
char s[200001];
int main()
{
	int n,m,a;bool bl=0;
	scanf("%s%d",s,&m);
	n=strlen(s);
	while(m--)
	{
		scanf("%d",&a);--a;
		b1[a]^=1;b1[n-a]^=1;
	}
	for(int i=0;2*i<n;++i)
	{
		bl^=b1[i];
		if(bl)swap(s[i],s[n-1-i]);
	}
	puts(s);
}
