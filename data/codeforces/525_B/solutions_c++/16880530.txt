#include <bits/stdc++.h>
using namespace std;
char a[200005];
int cnt[100005];
int n,m,x;
int main()
{
	cin>>a+1;
	n = strlen(a+1);
	cin>>m;
	while(m--)
	{
		cin>>x;
		cnt[x]++;
	}
	cnt[0] = 0;
	for(int i=1;2*i<=n;i++)
		cnt[i] += cnt[i-1];
	for(int i=1;2*i<=n;i++){
		if(cnt[i]&1)swap(a[i],a[n-i+1]);
	}
	cout<<a+1;
	return 0;
}
