
#include <iostream>

using namespace std;

int main()
{
	int n;
	string s;
	int a[200005] = {};
	
	cin>>s>>n;
	for (int i=0,t ; i<n ; i++)
		{ cin>>t; a[t-1]++; }
	
	
	int sl = s.length();
	for (int i=1 ; i<sl/2 ; i++)
		a[i]+=a[i-1];
	for (int i=0 ; i<sl/2 ; i++)
		if (a[i]%2) swap(s[i],s[sl-i-1]);
	
	cout<<s;
	return 0;
}
