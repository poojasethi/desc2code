#include <iostream>
using namespace std;
int a[100010];
int main(){
	string s;
	cin>>s;
	int l=s.length();
	int m,h;
	cin>>m;
	for(int i=1;i<=m;i++){
		cin>>h;
		a[h]++;
	}
	for(int i=1;i<=l/2;i++)
		a[i]=(a[i-1]+a[i])%2;
	for(int i=1;i<=l/2;i++)
		if (a[i]) swap(s[i-1],s[l-i]);
	cout<<s;
	return 0;
}
