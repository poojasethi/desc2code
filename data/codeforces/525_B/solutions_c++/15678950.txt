#include<iostream>
using namespace std;

const int oo = 1e5 + 10;
int n, x, a[oo];
string s;

int main(){
  cin >> s >> n;
  for(int i = 0; i < n; i++)
    cin >> x, a[x] = 1 - a[x];
  for(int i = 1; i <= s.length() / 2; i++){
    a[i] += a[i - 1];
    if(a[i] % 2)
      swap(s[i - 1], s[s.length() - i]);
  }
  cout << s;
  return 0;
}
