#include<bits/stdc++.h>
using namespace std;
string s;
long int m,a,sum=0,i;
int ar[100500]={0};
int main()
{	
	   cin>>s>>m;
	   while(m--)
	   {
	        cin>>a;
		ar[a]++;
	   }
	   for( i=1;i<=s.size()/2;i++)
	   {
	        sum+=ar[i];
	        if(sum%2)
		swap(s[i-1],s[s.size()-i]);
	   }
	   cout<<s<<endl;
	   return 0;
}
