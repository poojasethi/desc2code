#include <bits/stdc++.h>
using namespace std;

int a[100001];

int main()
{
	string s; cin >> s;

	int n; cin >> n;

	for(int i=0; i<n; i++)
	{
		int tmp;
		cin >> tmp;
		
		a[tmp-1] = a[tmp-1] ^ 1;
	}
	
	for(int i=1; i<s.size()/2; i++) a[i] = a[i-1] ^ a[i];
	for(int i=0; i<s.size()/2; i++) if(a[i]) swap(s[i], s[s.size()-1-i]);

	cout << s << '\n';
	return 0;
}
