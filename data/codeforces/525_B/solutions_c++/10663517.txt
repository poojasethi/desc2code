#include<bits/stdc++.h>
using namespace std;
int a[100000];
int main()
{
 string s;
 int m,l,x,sum=0;
 cin>>s>>m;
 l=s.size();
 while(m--)
 {
  scanf("%d",&x);
  a[x]++;
 }
 for(int i=1;i<=l/2;i++)
 {
             sum=sum+a[i];
             if((sum%2)==1)
             swap(s[i-1],s[l-i]);

}

     cout<<s;

}
