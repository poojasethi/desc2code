#include <bits/stdc++.h>
using namespace std;

#define sd(x) scanf("%d",&x)

char s[200000];

int main(){
	int m,l;
	cin>>s;
	l=strlen(s);
	int a[100000]={0},tmp;
	sd(m);
	for(int i=0;i<m;i++){
		sd(tmp);
		tmp--;
		a[tmp]++;
	}
	tmp=0;
	for(int i=0;i<l/2;i++){
		tmp+=a[i];
		if(tmp&1)swap(s[i],s[l-i-1]);
	}
	cout<<s;
	return 0;
}
