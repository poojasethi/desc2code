#include<iostream>
#include<string>
using namespace std;
int main(){
	int n,a,size,d[200005]={};
	string str;
	cin>>str>>n;
	size = str.length();
	for(int i=0;i<n;i++){
		cin>>a;
		a--;
		d[a]++;
		d[size-a]--;
	}
	for(int j=0,k=0;j<size;j++){
		k+=d[j];
		if(k%2==1)
			cout<<str[size-1-j];
		else
			cout<<str[j];
	}

	return 0;
}
