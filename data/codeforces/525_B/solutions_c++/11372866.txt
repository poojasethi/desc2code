#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j,len;
    string s;
    int a[300001];
    cin>>s>>n;

    len=s.size();
    for(i=0;i<n;i++)cin>>j,a[j-1]++,a[len-j+1]--;

    j=0;
    for(i=0;i<len;i++){
        j+=a[i];
        cout<<s[(j%2)?len-i-1:i];
    }
    return 0;
}
