#include <bits/stdc++.h>
using namespace std;

string s;
int n, a, len, sum, c[100005];
int main() {
	int i;
//	freopen("input.txt", "r", stdin);
	cin >> s >> n;
	len = s.size();
	for (i = 0; i < n; i++) {
		scanf("%d", &a);
		c[a]++;
	}
	for (i = 0; i < len / 2; i++) {
		sum += c[i + 1];
		if (sum % 2) swap(s[i], s[len - i - 1]);
	}
	cout << s;
}
