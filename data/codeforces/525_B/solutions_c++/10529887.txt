#include <iostream>
#include <string.h>
using namespace std;

char s[200005] = {0};
int t[100005] = {0};
int main()
{
	int i,n,d,f=0,l;
	cin>>s>>n;
	for(i=0;i<n;i++)
	{
		cin>>d;
		t[d]++;
	}
	l=strlen(s);
	for(i=0;i<l/2;i++)
	{
		f+=t[i+1];
		if(f&1)
		{
			swap(s[i],s[l-i-1]);
		}
	}
	cout<<s<<endl;
	return 0;
}
