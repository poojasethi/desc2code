#include <iostream>
#include <cstdio>
using namespace std;

bool b[1001001],c;

int main()
{
  string r;
  cin >> r;
  int q,w,l=r.size();
  cin >> q;
  while (q--) {
    cin >> w;
    b[w-1]=!b[w-1];
    b[l-w+1]=!b[l-w+1];
  }
  for (int i=0;i<l;i++) {
    c^=b[i];//cout<<b[i]<<c;
    cout << r[c?(l-i-1):i];
  }
  cout << '\n';
  return 0;
}
