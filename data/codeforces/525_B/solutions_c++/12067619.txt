#include <iostream>
using namespace std;

int A[100100], m, x;
string s;
int main() {
    cin >> s;
    for (cin >> m; m--; cin >> x, A[x-1]++);
    for (int i = 1; i < s.length()/2; i++) A[i] += A[i-1];
    for (int i = 0; i < s.length()/2; i++) {
        if (A[i] & 1) swap(s[i], s[s.length() - 1 - i]);
    }
    cout << s;
}
