#include<cstdio>
#include<cstring>
int v2[200002];
char v[200002];
int main ()
{int n,i,m,x;
scanf("%s",&v);
n=strlen(v);
scanf("%d",&m);
for(i=1;i<=m;i++)
	{scanf("%d",&x);
	v2[x]++;
	v2[n-x+2]--;
	}
for(i=1;i<=n;i++)
	v2[i]+=v2[i-1];
for(i=1;i<=n;i++)
	if(v2[i]%2==0)
		printf("%c",v[i-1]);
	else
		printf("%c",v[n-i]);
return 0;
}
