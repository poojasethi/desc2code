#include <bits/stdc++.h>
using namespace std;
int a[1000000];
string s;
int main()
{
	cin>>s;
	int n=s.length();
	int x;
	scanf("%d",&x);
	for(int i=0;i<x;i++)
    {
		int n;
		scanf("%d",&n);
		a[n]++;
	}
	x=0;
	for(int i=1;i<=n/2;i++)
	{
		x+=a[i];
		if(x%2)
		{
			swap(s[i-1],s[n-i]);
		}
	}
	cout<<s;
	return 0;
}
