
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,q,t;
string s;
cin>>s;
n=s.size();

vector <LL> v(n/2);
cin>>q;

for(i=0;i<q;i++)
	{cin>>t;
	v[t-1]++;}

t=0;
for(i=0;i<n/2;i++)
	{t=(t+v[i])%2;
	v[i]=t;
	}


for(i=0;i<n/2;i++)
	if(v[i])
		swap(s[i],s[n-i-1]);
	
	

cout<<s<<endl;

}
