#include <bits/stdc++.h>
using namespace std;
 
int main() {
    string s;
    int n, m, x, u = 0, c[101010];
    cin >> s >> m;
    n = s.size();
    while (m--) {
        cin >> x;
        c[x]++;
    }
    for (int i = 1; i <= n / 2; i++) {
        u += c[i];
        if (u % 2) swap(s[i - 1], s[n - i]);
    }
    cout << s;
    return 0;
}
