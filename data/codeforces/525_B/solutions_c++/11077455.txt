#include <bits/stdc++.h>
using namespace std;
int a[100005];
int main()
{
    string s;
    int m,i,k,sum=0;
    cin>>s>>m;
    for(i=0;i<m;i++){
        cin>>k;
        a[k]++;
    }
    for(i=1;i<=s.size()/2;i++){
        sum+=a[i];
        if(sum%2==1)swap(s[i-1],s[s.size()-i]);
    }
    cout<<s<<endl;
}
