#include <bits/stdc++.h>
using namespace std;
int a[1000000]={0};
int main()
{
    string str;
    int n,i,num;
    cin>>str>>n;
    for(i=0;i<n && cin>>num;a[num]++,i++);
    for (i=1,num=0;i<=str.size()/2;i++)
    {
        num+=a[i];
        if(num%2)
            swap(str[i-1],str[str.size()-i]);
    }
    cout<<str;
}
