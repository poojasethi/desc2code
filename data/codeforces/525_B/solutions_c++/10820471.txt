#include <cstdio>

int i,j,m,a[100001];
char s[200001];
bool w[200001];

int main(){
	scanf("%s",s);
	for(j=0;s[j];j++);
	scanf("%d",&m);
	for(i=0;i<m;i++){
		scanf("%d",&a[i]);
		w[a[i]-1]^=1;
		w[j-a[i]+1]^=1;
	}
	for(i=0;i<j;i++){
			printf("%c",w[i]?s[j-i-1]:s[i]);
			w[i+1]^=w[i];
	}
	printf("\n");
	return 0;
}
