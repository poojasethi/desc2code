
 #include<stdio.h>
typedef unsigned u;
char S[222222],c,B[222222];
int main()
{
    u n,x,l;
    while((c=getchar())<'a');
    for(S[l=0]=c;(c=getchar())>='a';)S[++l]=c;
    for(scanf("%u",&n);n--;B[--x]^=1)scanf("%u",&x);
    for(n=0,x=-1,++l;--l>++x;)if(n^=B[x]^B[l]){c=S[x];S[x]=S[l];S[l]=c;}
    printf("%s\n",S);
    return 0;
}
