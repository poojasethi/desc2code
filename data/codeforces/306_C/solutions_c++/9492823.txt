#include <stdio.h>

#define mo 1000000009

int n,w,b;

int qpow(int a,int b,int p)
{
	while (p) {
		if (p&1) a=1LL*a*b%mo;
		b=1LL*b*b%mo;
		p>>=1;
	}
	return a;
}

int getinv(int x)
{
	return qpow(1,x,mo-2);
}

int fw[4005],fb[4005];
int f[4005],f1[4005];

int C(int a,int b)
{
	return 1LL*f[a]*f1[b]%mo*f1[a-b]%mo;
}

int main()
{
	scanf("%d%d%d",&n,&w,&b);
	int i;
	f[0]=f1[0]=1;
	for (i=1;i<=w || i<=b;i++) f[i]=1LL*f[i-1]*i%mo,f1[i]=getinv(f[i]);
	for (i=1;i<=w;i++) fw[i]=1LL*f[w]*C(w-1,i-1)%mo;
	for (i=1;i<=b;i++) fb[i]=1LL*f[b]*C(b-1,i-1)%mo;
	int ans=0;
	for (i=1;i<=w && i<n;i++) ans=(ans+1LL*fw[i]*fb[n-i]%mo*(i-1))%mo;
	printf("%d\n",ans);
}
