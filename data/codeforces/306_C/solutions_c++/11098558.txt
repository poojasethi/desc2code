#include <cstdio>
#define p 1000000009
#define d(n,m) (1ll*c[m-1][n-1]*f[m]%p)
using namespace std;
int n,w,b,c[4005][4005],f[4004];
int main(){
	scanf("%d%d%d",&n,&w,&b);
	f[0]=1;
	int ans=0;
	for(int i=1;i<=4000;i++) f[i]=1ll*f[i-1]*i%p;
	for(int i=0;i<=4000;i++) for(int j=0;j<=i;j++){
		c[i][j]=!j?1:(c[i-1][j]+c[i-1][j-1])%p;
	}
	for(int i=1;i<n-1;i++) (ans+=d(i,b)*d(n-i,w)%p*(n-i-1)%p)%=p;
	printf("%d\n",ans);
	return 0;
}
