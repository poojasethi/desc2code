#include <iostream>
#include <cstdio>

using namespace std;

typedef long long ll;

const int MOD = 1000000009;
const int N = 4000;

ll dp[N + 1][N + 1];

int main() {
    int n, w, b;
    cin >> n >> w >> b;
    dp[0][0] = 1;
    for (int i = 1; i <= N; i++)
        for (int j = 1; j <= N; j++)
            dp[i][j] = (dp[i][j - 1] * (j - 1 + i) + dp[i - 1][j - 1] * i) % MOD;
    ll ans = 0;
    for (int i = 1; i <= min(b, n - 2); i++)
        ans = (ans + ((dp[i][b] * dp[n - i][w]) % MOD) * (n - 1 - i)) % MOD;
    cout << ans;
    return 0;
}
