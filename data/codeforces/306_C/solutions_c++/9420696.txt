#include <cstdio>
using namespace std;
typedef long long ll;
const int N=4004,P=1000000009;
int n,w,b,fac[N],ifac[N];
int qpower(int n,int k)
{
	ll ret=1,p=n;
	for (;k;p=p*p%P,k/=2) if (k%2) ret=ret*p%P;
	return (int)ret;
}
ll C(int n,int m) {return (ll)ifac[n-m]*ifac[m]%P*(ll)fac[n]%P;}
int main()
{
	int i; scanf("%d%d%d",&n,&w,&b);
	fac[0]=ifac[0]=1;
	for (i=1;i<=4000;i++) fac[i]=(ll)fac[i-1]*i%P,ifac[i]=qpower(fac[i],P-2);
	int ans=0;
	for (i=1;(i<=b)&&(i<n-1);i++)
	{
		if (n-i>w) continue;
		ans=(ans+C(w-1,n-i-1)*fac[w]%P*(ll)(n-i-1)%P*C(b-1,i-1)%P*(ll)fac[b])%P;
	}
	printf("%d\n",ans);
	return 0;
}
