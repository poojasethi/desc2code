#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long LL;

const int N = 4005;
const int MOD = int(1e9 + 9);

int f[N], c[N][N];
int n, w, b;

int main()
{
	scanf("%d%d%d", &n, &w, &b);
	
	for(int i=0; i<N; i++)
	{
		c[i][0] = c[i][i] = 1;
		for(int j=1; j<i; j++) c[i][j] = (c[i-1][j-1] + c[i-1][j]) % MOD;
	}
	
	f[0] = 1;
	for(int i=1; i<N; i++) f[i] = (LL(f[i-1]) * i) % MOD;
	
	LL ans(0);
	for(int i=1; i<n-1; i++)
	{
		LL temp = ((LL(c[b-1][i-1]) * c[w-1][n-i-1]) % MOD * (n-i-1)) % MOD;
		(ans += temp) %= MOD;
	}
	ans = ((LL(ans) * f[b]) % MOD * f[w]) % MOD;
	printf("%d\n", int(ans));
	
	
	return 0;
}
