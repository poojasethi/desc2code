#include <bits/stdc++.h>
#define P 1000000009
#define F(n,m) (1ll*c[m-1][n-1]*f[m]%P)
using namespace std;

int N,W,B,c[4005][4005],f[4005];

void doit()
{
	scanf("%d%d%d",&N,&W,&B),f[0]=1;
	int ans=0;
	for (int i=1; i<=4000; i++) f[i]=1ll*f[i-1]*i%P;
	for (int i=0; i<=4000; i++)
		for (int j=0; j<=i; j++) c[i][j]=!j?1:(c[i-1][j]+c[i-1][j-1])%P;
	for (int i=1; i<N-1; i++) (ans+=F(i,B)*F(N-i,W)%P*(N-i-1)%P)%=P;
	printf("%d\n",ans);
}

int main()
{
	doit();
	return 0;
}
