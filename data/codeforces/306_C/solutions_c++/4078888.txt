#include<iostream>
#include<cstdio>
#define mod 1000000009
#define lint long long
#define N 4000
using namespace std;
lint ans,C[N+1][N+1],A[N+1];
void Init(){
	A[1]=C[0][0]=C[1][0]=C[1][1]=1;
	for(int i=2;i<=4000;i++){
		A[i]=A[i-1]*i%mod;
		C[i][0]=C[i][i]=1;
		for(int j=1;j<i;j++)
		C[i][j]=(C[i-1][j-1]+C[i-1][j])%mod;
	}
}
int main(){
	Init();
	int n,w,b;
	cin>>n>>w>>b;
	int bin=max(1,n-w);
	int end=min(n-2,b);
	for(int i=bin;i<=end;i++){
		int k=n-i;//k段白色 
		ans=(ans+C[b-1][i-1]*C[w-1][k-1]%mod*(k-1)%mod)%mod;		
	}
	ans=A[b]*A[w]%mod*ans%mod;
	cout<<ans<<endl;
}
