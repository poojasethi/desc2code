#include<iostream>
#include<cstdio>
#define mod 1000000009
#define lint long long
#define N 4000
using namespace std;
lint ans,C1[N+1],C2[N+1];
void Init(lint *C,int m){
	C[0]=C[1]=1;
	for(int i=2;i<m;i++){
		C[i]=1;
		for(int j=i-1;j>0;j--)
		C[j]=(C[j-1]+C[j])%mod;
	}
}
int main(){
	int n,w,b;
	cin>>n>>w>>b;
	Init(C1,b);Init(C2,w);
	int bin=max(1,n-w);
	int end=min(n-2,b);
	for(int i=bin;i<=end;i++){
		int k=n-i;//k段白色 
		ans=(ans+C1[i-1]*C2[k-1]%mod*(k-1)%mod)%mod;		
	}
	int i;
	for(i=1;i<=b&&i<=w;i++)
	ans=ans*i%mod*i%mod;
	while(i<=b)ans=ans*(i++)%mod;
	while(i<=w)ans=ans*(i++)%mod; 
	cout<<ans<<endl;
}
