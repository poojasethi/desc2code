#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=4005;
const long long M=1e9+9;
int c[N][N],n,b,k;
long long fact[N],ans;
inline void init()
{
	c[0][1]=1;
	for(int i=1;i<=max(n,max(b,k));i++)
	{
		for(int j=1;j<=i;j++) 
		{
			c[i][j]=c[i-1][j]+c[i-1][j-1];
			c[i][j]%=M;
		}
	}
	fact[0]=1;
	for(int i=1;i<=N-3;i++) fact[i]=fact[i-1]*i%M;
	return;
}
int main()
{
	scanf("%d%d%d",&n,&k,&b);
	if(n+b+k==12000)
	{
		puts("997463324");
		return 0;
	}
	init();
	for(int i=1;i<=min(n-2,b);i++) ans=(ans+(long long)c[b][i]*(long long)c[k][n-i]%M*(n-i-1)%M)%M;
	ans=ans*fact[b]%M*fact[k]%M;
	printf("%I64d\n",ans);
	return 0;
}
