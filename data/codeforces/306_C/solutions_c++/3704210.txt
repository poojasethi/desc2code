#include <cstdio>

using namespace std;

const int modulo = 1000000009;

long long A(int events, int days)
{
	if (events < days) return 0;
	long long result = 1;
	for (int e = events - 1; e >= days; --e) {
		result *= e;
		result %= modulo;
	}
	for (int i = 0; i < days; i++) {
		result *= events;
		result %= modulo;
		--events;
	}
	return result;
}

int main()
{
	int n, w, b;
	scanf("%i%i%i", &n, &w, &b);
	long long result = 0;
	for (int i = 2; i < n; i++) {
		result += ((i - 1) * A(w, i)) % modulo * A(b, n - i);
		result %= modulo;
	}
	printf("%i\n", (int) result);
	return 0;
}
