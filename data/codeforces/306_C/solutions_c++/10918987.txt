#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=4005;
const long long M=1e9+9;
long long c[N][N],n,b,k;
long long fact[N];
inline void init()
{
	c[0][1]=1;
	for(int i=1;i<=4001;i++)
	{
		for(int j=1;j<=i;j++) 
		{
			c[i][j]=c[i-1][j]+c[i-1][j-1];
			c[i][j]%=M;
		}
	}
	fact[0]=1;
	for(int i=1;i<=N-3;i++) fact[i]=fact[i-1]*i%M;
	return;
}
int main()
{
	scanf("%d%d%d",&n,&k,&b);
	long long ans=0;
	init();
	for(int i=1;i<=min(n-2,b);i++) ans=(ans+c[b][i]*c[k][n-i]%M*(n-i-1)%M)%M;
	ans=ans*fact[b]%M*fact[k]%M;
	printf("%I64d\n",ans);
	return 0;
}
