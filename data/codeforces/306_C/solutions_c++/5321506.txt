

#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,a,b) for (int i=a;i<b;i++)
const int N=4005,oo=1000000009;
long long p[N],c[N][N],res,ans;
int n,w,b;
int main()
{
    p[0]=1; rep(i,1,N-2) p[i]=p[i-1]*i%oo;
    rep(i,0,N-2) rep(j,0,i+1) c[i][j]=i&&j?(c[i-1][j-1]+c[i-1][j])%oo:1;
    scanf("%d%d%d",&n,&w,&b);
    rep(i,max(n-w,1),b+1) if (i<n-1) ans=(c[b-1][i-1]*c[w-1][n-i-1]%oo*(n-1-i)+ans)%oo;
    printf("%d\n",p[w]*p[b]%oo*ans%oo); return 0;
}
