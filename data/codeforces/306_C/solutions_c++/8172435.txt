#include<iostream>
#include<cstring>
using namespace std;
const long long mod = 1e9+9;
int n,w,b;
long long fac[4001],ans,rem[4001][4001];
long long calc(int n,int m)
{
	if(n==0&&m==0)return 1;
	if(n==0||m==0)return 0;
	if(rem[n][m]>=0)return rem[n][m];
	return rem[n][m]=(calc(n-1,m)+calc(n-1,m-1))%mod;
}
int main()
{
	cin>>n>>w>>b;
	fac[0]=1;memset(rem,-1,sizeof(rem));
	for(int i=1;i<=4000;i++)fac[i]=fac[i-1]*i%mod;
	for(int i=1;i<=n-2;i++)
	ans+=(n-i-1)*fac[b]%mod*fac[w]%mod*calc(b,i)%mod*calc(w,n-i)%mod;
	cout<<ans%mod<<endl;
}
