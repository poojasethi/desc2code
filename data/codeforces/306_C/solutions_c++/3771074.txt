#include<stdio.h>
const long long p=1000000009;
long long Pow(long long,long long);
long long inv(long long);
long long Pow(long long a,long long n)
{
	if(n==0)return 1;
	long long ans=Pow(a,n/2);
	if(n%2==0)return (ans*ans)%p;
	else return ((ans*ans)%p*a)%p;
}
long long inv(long long a)
{
	return Pow(a,p-2);
}
int main()
{
	long long n,w,b,fac[10000],i,ans=0;
	fac[0]=1;
	for(i=1;i<10000;i++)
		fac[i]=(i*fac[i-1])%p;
	scanf("%I64d%I64d%I64d",&n,&w,&b);
	ans=((fac[w]*fac[b])%p*(w-1))%p;
	ans=(ans*fac[w+b-3])%p;
	ans=(ans*inv(fac[w+b-n]))%p;
	ans=(ans*inv(fac[n-3]))%p;
	printf("%I64d",ans);
	return 0;
}
