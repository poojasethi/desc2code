#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define int64 long long
using namespace std;
int64 fac[12000],ans,C[5100][5100];
const int mo=1000000009;
int i,j,k,l,n,m,a,b,x;
int main(){
	for(i=0;i<=4000;++i)
		for(j=0;j<=i;++j)C[i][j]=j==0?1:(C[i-1][j-1]+C[i-1][j])%mo;
	fac[0]=1;
	for(i=1;i<=4000;++i)fac[i]=(fac[i-1]*i)%mo;
	scanf("%d%d%d",&n,&a,&b);
	ans=0;
	for(x=1;x<n;++x)ans=(ans+fac[b]*C[b-1][x-1]%mo*fac[a]%mo*C[a-1][n-x-1]%mo*(n-x-1)%mo)%mo;
	printf("%I64d\n",ans);
}
