#include <iostream>
#include <cstdio>
#include <cstring>
#include <climits>
typedef long long LL;
using namespace std;
const int oo=INT_MAX>>1;

const int Mo=1000000009;
LL C[4100][4100];
LL fac[4100];

int main(){
    int i,j;
    fac[0]=C[0][0]=1;
    for (i=1;i<=4000;i++){
        fac[i]=fac[i-1]*i%Mo;
        C[i][0]=1;
        for (j=1;j<=i;j++) C[i][j]=(C[i-1][j-1]+C[i-1][j])%Mo;
    }
    int N,w,b;
    cin >> N >> w >> b;
    LL ans=0;
    int l=max(1,N-w),r=min(b,N);
    for (i=l;i<=r;i++) ans=(ans+C[b-1][i-1]*fac[b]%Mo*C[w-1][N-i-1]%Mo*fac[w]%Mo*(N-i-1)%Mo)%Mo;
    cout << ans << endl;
}
