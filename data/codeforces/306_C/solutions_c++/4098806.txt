#include <stdio.h>
#include <iostream>
#include <string.h>
#define ll long long
#define clr(a,b) memset(a,b,sizeof(a))
using namespace std;

const int mod=1000000009;

int n,a,b;
ll fact[8005],inv_fact[8005];

ll pow(ll a,ll b){
	ll ret=1;
	while(b){
		if(b&1)ret=ret*a%mod;
		a=a*a%mod;
		b>>=1;
	}return ret;
}

ll calinv(ll a){
	return pow(a,mod-2);
}

ll choose(ll a,ll b){
	return fact[a]*inv_fact[b]%mod*inv_fact[a-b]%mod;
}

int main(){
	cin>>n>>a>>b;
	fact[0]=inv_fact[0]=1;
	for(int i=1;i<=a+b;i++){
		fact[i]=fact[i-1]*i%mod;
		inv_fact[i]=calinv(fact[i]);
	}
	cout<<fact[a]*fact[b]%mod*choose(a+b-3,n-3)%mod*(a-1)%mod<<endl;
}
