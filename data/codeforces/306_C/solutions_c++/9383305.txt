#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long s64;

const int MOD = 1e9 + 9;

int n, w, b, C[4004][4004], fact[4004];

int divide(int n, int m)
{
	if (n < m) return 0;
	return (s64) C[n - 1][m - 1] * fact[n] % MOD;
}
int main()
{
	scanf("%d %d %d", &n, &w, &b);
	for (int i = 0; i <= 4000; ++i)
	{
		fact[i] = !i ? 1 : (s64) fact[i - 1] * i % MOD;
		for (int j = 0; j <= i; ++j)
		{
			C[i][j] = !j ? 1 : (C[i - 1][j - 1] + C[i - 1][j]) % MOD;
		}
	}
	int ans = 0;
	for (int m = 1; m <= n - 2; ++m)
	{
		ans += (s64) (n - 1 - m) * divide(b, m) % MOD * divide(w, n - m) % MOD;
		ans %= MOD;
	}
	printf("%d\n", ans);
}
