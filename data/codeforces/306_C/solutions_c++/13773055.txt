using namespace std;
#define N 4001
#define P 1000000009
#define ll long long
ll JC[N],ans;int n,C[N][N],m,p;
inline int Add(int x)
 {return x<P?x:x-P;}
int main()
 {
    cin >>n>>m>>p;
    for (int i=0;i<N;i++) C[i][0]=1;
    for (int i=1;i<N;i++)
     for (int j=1;j<=i;j++)
       C[i][j]=Add(C[i-1][j-1]+C[i-1][j]);
    JC[0]=1;
    for (int i=1;i<N;i++) JC[i]=JC[i-1]*i%P;
    for (int i=2;i<n;i++)
      ans=(ans+C[m-1][i-1]*JC[m]%P*C[p-1][n-i-1]%P*JC[p]%P*(i-1)%P)%P;

 }
