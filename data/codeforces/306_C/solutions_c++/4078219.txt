#include <stdio.h>
#include <string.h>
#define mod 1000000009
__int64 C[4001][4001];
void JC()
{
	int i,j;
	memset(C,0,sizeof(C));
	C[0][0]=1;
	for(i=0;i<4001;i++) C[i][0] = 1;
	for(i=1;i<4001;i++)
	{
		for(j=1;j<=i;j++)
			C[i][j]=(C[i-1][j]+C[i-1][j-1])%mod;
	}
	return;
}
__int64 A[4001];
void mul()
{
	A[0]=1;
	for(int i=1;i<4001;i++)
	A[i]=A[i-1]*i%mod;
}
int main()
{
	JC();
	mul();
	int n,w,b;
	__int64 ans;
	while(scanf("%d%d%d",&n,&w,&b)!=EOF)
	{
		ans=0;
		for(int i=1;i<=b;i++)
		{
			int k=n-i;
			if(k>w) continue;
			if(k<2) break;
			ans=(ans+C[b-1][i-1]*C[w-1][k-1]%mod*(k-1)%mod+mod)%mod;
		}
		ans=(ans*A[b]%mod*A[w]%mod+mod)%mod;
		printf("%I64d\n",ans);
	}
}
