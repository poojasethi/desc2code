#include<iostream>
#include<cstdio>
#define mod 1000000009
#define lint long long
#define N 4000
using namespace std;
lint ans,C[N+1][N+1],A;
int i,n,w,b;
void Init(){
	C[0][0]=C[1][0]=C[1][1]=1;
	for(int i=2;i<=4000;i++){
		C[i][0]=C[i][i]=1;
		for(int j=1;j<i;j++)
		C[i][j]=(C[i-1][j-1]+C[i-1][j])%mod;
	}
}
int main(){
	cin>>n>>w>>b;
	Init();
	int bin=max(1,n-w);
	int end=min(n-2,b);
	for(i=bin;i<=end;i++){
		int k=n-i;
		ans=(ans+C[b-1][i-1]*C[w-1][k-1]%mod*(k-1)%mod)%mod;		
	}
	for(i=1;i<=b&&i<=w;i++)
	ans=ans*i%mod*i%mod;
	while(i<=b)ans=ans*(i++)%mod;
	while(i<=w)ans=ans*(i++)%mod; 
	cout<<ans<<endl;
}
