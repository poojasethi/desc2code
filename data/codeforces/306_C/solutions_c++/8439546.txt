#include<cstdio>
long long ans;
long long jc[4010];
long long c[4010][4010];
int w,n,b;
int main()
{
    int i,j;
    for(i=0;i<=4000;++i)
    {
        c[i][0]=1;
        for(j=1;j<=i;++j)
        {
            c[i][j]=c[i-1][j]+c[i-1][j-1];
            if(c[i][j]>=1000000009)
                c[i][j]-=1000000009;
        }
    }
    jc[0]=1;
    for(i=1;i<=4000;++i)
        jc[i]=jc[i-1]*i%1000000009;
    scanf("%d%d%d",&n,&w,&b);
    for(i=2;n-i;++i)
        ans=(ans+c[w-1][i-1]*c[b-1][n-i-1]%1000000009*(i-1))%1000000009;
    ans=ans*jc[w]%1000000009*jc[b]%1000000009;
    printf("%d\n",(int)ans);
    return 0;
}
