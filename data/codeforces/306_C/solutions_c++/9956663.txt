#include<stdio.h>
#define mo 1000000009
long long  C[4005][4005],ans;
void yanghui()
{
	int i,j;
	C[0][0]=C[1][0]=C[1][1]=1;
	for(i=2;i<=4000;i++)
	{
		C[i][0]=C[i][i]=1;
		for(j=1;j<i;j++)
			C[i][j]=(C[i-1][j-1]+C[i-1][j])%mo;
	}
}
int main()
{
	yanghui();
	int n,w,b;
	long long A[4005];
	A[1]=1;
	for(int i=2;i<=4000;i++)A[i]=A[i-1]*i%mo;
	scanf("%d%d%d",&n,&w,&b);
	ans=0;
	for(int i=1;i<=b;i++)
	{
		int k=n-i;//需要k段白色的 ！ 
		if(k>w)continue;
		if(k<2)break;;
		ans=(ans+C[b-1][i-1]*C[w-1][k-1]%mo*(k-1)%mo)%mo;
	}
	ans=ans*A[b]%mo*A[w]%mo; 
	printf("%lld\n",ans);
}
