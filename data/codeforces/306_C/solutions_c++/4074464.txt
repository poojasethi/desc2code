#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<math.h>
#include<iostream>
using namespace std;
const int md=1000000009;
__int64 c[4005][4005];
void getc()
{
	c[0][0]=1;
	c[1][0]=1;
	c[1][1]=1;
	for(int i=2;i<=4000;i++)
	{
		for(int j=1;j<i;j++)
		{
			c[i][j]=(c[i-1][j]+c[i-1][j-1])%md;
		}
		c[i][0]=1;
		c[i][i]=1;
	}
}
int main()
{
	int n,w,b,i;
	getc();
	__int64 ans=0;	
	scanf("%d%d%d",&n,&w,&b);
	for(i=1;i<n-1;i++)
	{
		if(i>b)break;
		if(n-i>w)continue;
		if(n-i<2)continue;
		ans=(ans+(c[b-1][i-1]*c[w-1][n-i-1]%md)*(n-i-1)%md)%md;
	}
	for(i=1;i<=b;i++)
	{
		ans=(ans*i)%md;
	}
	for(i=1;i<=w;i++)
	{
		ans=(ans*i)%md;
	}
	printf("%I64d\n",ans);
}
