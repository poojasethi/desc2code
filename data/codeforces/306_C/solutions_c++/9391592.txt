#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define modd 1000000009
#define N 4001
using namespace std;
long long ans,dp[N][N],n,w,b,hehe;
int main()
{
    cin>>n>>w>>b;
    for (int i=0;i<=max(w,b);i++) dp[0][i]=1;
    for (int i=1;i<=n;i++)
    for (int j=1;j<=max(w,b);j++)
    dp[i][j]=(dp[i][j-1]+dp[i-1][j-1])%modd;//,cout<<dp[i][j]<<endl;
    for (int i=1;i<=n-2;i++)
    ans=(ans+(dp[i][b]-dp[i][b-1]+modd)%modd*(dp[n-i][w]-dp[n-i][w-1]+modd)%modd*(n-i-1)%modd)%modd;
    for (int i=1;i<=w;i++) ans=ans*i%modd;
    for (int i=1;i<=b;i++) ans=ans*i%modd;
    cout<<ans<<endl;
//  system("pause");
    return 0;
}
