#include <stdio.h>
#include <memory.h>
#include <algorithm>
using namespace std;
#define M 101000
#define N 105

int d[M],h[M],t[M],s[M],x[N],y[N],id[N][M];
__int64 f[M][N],sum[M];

int main(){
//    freopen("in.txt","r",stdin);
    int i,j,k,n,m,p;
    scanf("%d%d%d",&n,&m,&p);
    for (i=2;i<=n;++i) scanf("%d",&d[i]),d[i]+=d[i-1];
    for (i=1;i<=m;++i) scanf("%d%d",&h[i],&t[i]),s[i]=t[i]-d[h[i]];
    sort(s+1,s+m+1);
    for (i=1;i<=m;++i) sum[i]=sum[i-1]+s[i];
    for (i=1;i<=m;++i) for (j=1;j<=p;++j){
        while(x[j-1]<y[j-1] && f[id[j-1][x[j-1]]][j-1]+sum[id[j-1][x[j-1]]]-f[id[j-1][x[j-1]+1]][j-1]-sum[id[j-1][x[j-1]+1]] > (__int64)(id[j-1][x[j-1]]-id[j-1][x[j-1]+1]) * s[i])
            ++x[j-1];
        k=id[j-1][x[j-1]];
        f[i][j]=f[k][j-1]+(__int64)(i-k)*s[i]-(__int64)(sum[i]-sum[k]);
        while (x[j]<y[j] &&
            (__int64)(f[id[j][y[j]-1]][j]+sum[id[j][y[j]-1]]-f[id[j][y[j]]][j]-sum[id[j][y[j]]]) * (id[j][y[j]]-i) >=
            (__int64)(f[id[j][y[j]]][j]+sum[id[j][y[j]]]-f[i][j]-sum[i]) * (id[j][y[j]-1]-id[j][y[j]]))
                --y[j];
        id[j][++y[j]]=i;
    }
    printf("%I64d\n",f[m][p]);
    return 0;
}
