#include<cstdio>
#include<algorithm>
using namespace std;
#define rep(i,a,b) for (int i=a;i<=b;i++)
#define ll long long
#define N 100005
int n,m,x,p,d[N],id[N],o[N],a[N],q[N]; ll s[N],f[N],g[N];
bool cmp(const int i,const int j){return o[i]<o[j];}
#define cal(x) f[x]+1LL*a[i]*(i-x)-s[i]+s[x]
int main()
{
    scanf("%d%d%d",&n,&m,&p);
    rep(i,2,n) scanf("%d",d+i),d[i]+=d[i-1];
    rep(i,1,m) scanf("%d%d",&x,o+i),o[i]-=d[x],id[i]=i;
    sort(id+1,id+m+1,cmp); rep(i,1,m) a[i]=o[id[i]],s[i]=s[i-1]+a[i];
    rep(i,1,m) f[i]=1LL*a[i]*i-s[i];
    rep(j,2,p){
        int h=1,t=1; q[1]=0; rep(i,1,m) g[i]=1LL<<50;
        rep(i,1,m){
            while (h<=t && (f[i]-(cal(q[t])))*(q[t]-q[t-1])<=
                ((cal(q[t]))-(cal(q[t-1])))*(i-q[t])) t--;
            q[++t]=i; while (h<t && cal(q[h])>=cal(q[h+1])) h++; g[i]=min(f[i],cal(q[h]));
            }
        rep(i,1,m) f[i]=g[i];
        }
    printf("%I64d\n",f[m]); return 0;
}
