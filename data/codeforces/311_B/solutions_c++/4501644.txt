#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define db double
#define int64 long long
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,p,d[N],l,r;
int64 t[N],s[N],f[110][N],inf,g[N],an;
inline int64 cal(int x) {
	return f[i-1][x]-t[j]*x+s[x]+t[j]*j-s[j];
}
inline db G(int x,int y) {
	return (f[i-1][x]+s[x]-f[i-1][y]-s[y])/(-y-(-x));
}
int main() {
	scanf("%d%d%d",&n,&m,&p);
	For(i,2,n) scanf("%d",&d[i]),d[i]+=d[i-1];
	For(i,1,m) {
		scanf("%d%I64d",&k,&t[i]);
		t[i]-=d[k];
	}
	sort(t+1,t+m+1);
	For(i,1,m) s[i]=s[i-1]+t[i];
	memset(f,120,sizeof(f)); inf=an=f[0][0]; f[0][0]=0;
	For(i,1,p) {
		l=r=1; g[1]=0;
		For(j,0,m) {
			for (;l<r&&cal(g[l])>=cal(g[l+1]);l++);
			f[i][j]=cal(g[l]);
			if (f[i-1][j]<inf) {
				for (;l<r&&G(g[r-1],g[r])>=G(g[r],j);r--);
				g[++r]=j;
			}
		} an=min(an,f[i][m]);
	}
	printf("%I64d\n",an);
	//for(;;);
	return 0;
}
