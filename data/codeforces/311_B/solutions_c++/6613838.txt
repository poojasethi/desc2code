#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100000
#define int64 long long
#define db double
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int64 i,j,k,n,m,p,x,y,l,r,b;
int64 d[N],g[N],a[N];
int64 s[N],f[N],an,B[N],K[N],G[N];
db L[N];
inline db jiao(db x0,db y0,db x1,db y1) {
    if (x1==x0) return y0<y1?-1e15:1e15;
    else return (y1-y0)/(x0-x1);
}
int main() {
    scanf("%I64d%I64d%I64d",&n,&m,&p);
    For(i,2,n) scanf("%I64d",&k),d[i]=d[i-1]+k;
    For(i,1,m) scanf("%I64d%I64d",&x,&y),a[i]=y-d[x];
    sort(a+1,a+m+1);
    For(i,1,m) s[i]=s[i-1]+a[i];
    memset(f,3,sizeof(f)); f[0]=0;
    an=(int64)1e15;
    For(i,1,p) {
        l=1,r=0;
        For(j,0,m) {
            k=-j; b=f[j]+s[j];
            for (;l<=r&&jiao(k,b,K[r],B[r])<L[r];r--);
            K[++r]=k,B[r]=b;
            if (l==r) L[r]=-1e10;
            else L[r]=jiao(k,b,K[r-1],B[r-1]);
            for (;l<r&&L[l+1]<=a[j+1];l++);
            G[j+1]=a[j+1]*(j+1)-s[j+1]+B[l]+K[l]*a[j+1];
        }
        For(j,0,m) f[j]=G[j];
        an=min(an,f[m]);
    } printf("%I64d",an); return 0;
}
