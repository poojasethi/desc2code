#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define db double
#define int64 long long
#define N 100010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,p,d[N],l,r;
int64 t[N],s[N],f[N],g[N],h[N];
inline int64 cal(int x) {
	return f[x]-t[j]*x+s[x]+t[j]*j-s[j];
}
inline db G(int x,int y) {
	return (f[x]+s[x]-f[y]-s[y])/(0.-y-(-x));
}
int main() {
	scanf("%d%d%d",&n,&m,&p);
	For(i,2,n) scanf("%d",&d[i]),d[i]+=d[i-1];
	For(i,1,m) {
		scanf("%d%I64d",&k,&t[i]);
		t[i]-=d[k];
	}
	sort(t+1,t+m+1);
	For(i,1,m) s[i]=s[i-1]+t[i];
	For(i,1,m) f[i]=t[i]*i-s[i];
	For(i,2,p) {
		l=1,r=0;
		For(j,1,m) {
			for (;l<r&&G(g[r-1],g[r])>=G(g[r],j);r--);
			g[++r]=j;
			for (;l<r&&cal(g[l])>=cal(g[l+1]);l++);
			h[j]=cal(g[l]);
		}
		For(j,1,m) f[j]=min(f[j],h[j]);
	}
	printf("%I64d\n",f[m]);
	//for(;;);
	return 0;
}
