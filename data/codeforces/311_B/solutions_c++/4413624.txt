#include<stdio.h>
#include<queue>
#include<algorithm>
#define LL long long
using namespace std;
struct node
{
  LL x,y;
};
node q[100005];
int a[100005];
LL s[100005];
LL dp[2][100005];
int d[100005];
const long long inf = 1000000000000000000LL;
int main ()
{
   int i,n,j,m,p,t,h;
   scanf("%d %d %d",&n,&m,&p); 
   for(i=2;i<=n;i++)
   {
      scanf("%d",&d[i]);
      d[i]+=d[i-1];
   }
   for(i=1;i<=m;i++)
   {
      scanf("%d %d",&h,&t);
      a[i] = t-d[h];
   }
   sort(a+1,a+m+1);
   for(i=1;i<=m;i++)
   {
   s[i]=s[i-1]+a[i];
   dp[0][i] =inf;
   }
   dp[0][0] = 0;
   int f,r;
   node is;
   for(i=1,t=1;i<=p;i++,t^=1)
   {
      f  = r = 0;
      dp[t][0] = 0;
      q[r].x = 0;
      q[r].y = 0;
      r++;
      for(j = 1;j<=m;j++)
      {
        while(r-f>1 && (q[f+1].y-q[f].y)<(LL)a[j]*(q[f+1].x-q[f].x))
        f++;
        dp[t][j] = q[f].y +(j- q[f].x)*(LL)a[j]-s[j];
        is.y = dp[t^1][j]+s[j];
        is.x = j;
        while(r-f>1 &&(is.y-q[r-1].y)*(q[r-1].x-q[r-2].x)<(is.x-q[r-1].x)*(q[r-1].y-q[r-2].y))
        r--;
        q[r] = is;
        r++;
      }
   }
   printf("%I64d",dp[t^1][m]);
   //scanf("%d",&n);
   return 0;
}
