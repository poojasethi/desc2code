#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define MAXN 100005
#define INF 0x7fffffffffffLL
using namespace std ;
typedef long long LL ;

int N, M, P ;
LL tmd[MAXN], ans ;
LL dis[MAXN] ;
LL dp[MAXN], sum[MAXN], X[MAXN] ;
int q[MAXN], f1, r1 ;

int main()
{
    int i, j, at ;
    scanf("%d %d %d", &N, &M, &P) ;
    for(i = 2; i <= N; i ++)
        scanf("%I64d", &dis[i]), dis[i] += dis[i-1] ;
    for(i = 1; i <= M; i ++)
    {
        scanf("%I64d %I64d", &at, &tmd[i]) ;
        tmd[i] -= dis[at] ;
    }
    sort(tmd+1, tmd+1+M) ;
    for(i = 0; i <= M; i ++)
        dp[i] = INF, sum[i] = sum[i-1]+tmd[i] ;
    ans = dp[0], dp[0] = 0 ;
    for(at = 1; at <= P; at ++)
    {
        f1 = 1, r1 = 1, q[f1] = 0 ;
        for(i = 1; i <= M; i ++)
        {
            while(f1 < r1 && X[q[f1+1]]-X[q[f1]] <= tmd[i]*(q[f1+1]-q[f1])) f1 ++ ;
            X[i] = dp[i]+sum[i], dp[i] = X[q[f1]]+tmd[i]*(i-q[f1])-sum[i] ;
            while(f1 < r1 && (X[q[r1]]-X[q[r1-1]])*(i-q[r1]) >= (X[i]-X[q[r1]])*(q[r1]-q[r1-1])) 
                r1 -- ;
            q[++r1] = i ;
        }
        ans = min(ans, dp[M]) ;
    }
    printf("%I64d\n", ans) ;
    //system("pause") ;
    return 0 ;
    
}
