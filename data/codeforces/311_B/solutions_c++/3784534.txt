#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 210000
#define int64 long long
#define foru(i,a,b) for(int i=a;i<=b;++i)

using namespace std;
int64 dis[N],sum,x[N],f[N],g[N],s[N];
int v[N],n,m,p,a,b,l,r;
bool compare(int a,int b,int64 c)
{
	return g[a]-a*c>g[b]-b*c;
}
int64 getx(int a,int b)
{
	return (g[b]-g[a]+b-a-1)/(b-a);
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	scanf("%d%d%d",&n,&m,&p);
	foru(i,2,n) scanf("%d",&dis[i]),dis[i]+=dis[i-1];
	foru(i,1,m) scanf("%d%d",&a,&b),x[i]=b-dis[a],sum-=x[i];
	sort(x+1,x+m+1); 
	foru(i,1,m) s[i]=s[i-1]+x[i];
	foru(i,1,m) g[i]=(int64)i*100000000000; g[0]=0;
	foru(k,1,p)
	{
		l=1; r=1; v[1]=k-1;
		foru(i,k,m)
		{
		  	while(l<r&&compare(v[l],v[l+1],x[i])) ++l;
		  	f[i]=g[v[l]]+(i-v[l])*x[i];
		  	while(r>l&&getx(v[r],i)<getx(v[r-1],v[r])) --r;
		  	v[++r]=i;
		}
		foru(i,1,m) g[i]=f[i];
	}
	printf("%I64d\n",g[m]+sum);
	return 0;
}
