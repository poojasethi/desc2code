#include<cstdio>
#include<cstring>
#include<algorithm>
#define LL long long
using namespace std ;
const int maxn = 100100 ;
LL n , m , p , l , r , d[maxn] , h[maxn] , t[maxn] , s[maxn] , q[maxn];
LL f[maxn] , y[maxn] , f1[maxn];
void in() {
    scanf("%I64d%I64d%I64d",&n,&m,&p) ;
    for(int i = 2 ; i <= n ; i++) scanf("%I64d",&d[i]) ;
    for(int i = 1 ; i <= m ; i++) scanf("%I64d%I64d",&h[i],&t[i]) ;
    for(int i = 2 ; i <= n ; i++) d[i] += d[i-1] ;
    for(int i = 1 ; i <= m ; i++) t[i] -= d[h[i]];
    sort(t+1 , t+1+m) ;
    for(int i = 1 ; i <= m ; i++) s[i] = s[i-1] + t[i] ;
}

void work() {
    for(int i = 1 ; i <= m ; i++) f[i] = i*(LL)t[i] - s[i] ;
    p-- ;
    while(p--) {
        l = r = 0 ;
        for(int i = 1 ; i <= m ; i++) {
            while(l<r-1 && ((f[q[r-1]]+s[q[r-1]]-f[q[r-2]]-s[q[r-2]])*(i-q[r-1])
                        >= (f[i]+s[i]-f[q[r-1]]-s[q[r-1]])*(q[r-1]-q[r-2]))) r-- ;
            q[r++] = i ;
            while(l<r-1 && (f[q[l+1]]+s[q[l+1]]-f[q[l]]-s[q[l]]) < t[i]*(q[l+1]-q[l])) l++ ;
            int j = q[l] ;
            y[i] = f[j]+s[j]+t[i]*(i-j)-s[i] ;
        }
        for(int i = 1 ; i <= m ; i++) f[i] = y[i] ;
    }
    printf("%I64d\n",f[m]) ;
}

int main() {
    in() ;
    work() ;
    return 0 ;
}
