#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

#define N 100005
#define P 105

typedef long long int64;

int Q[N], h, t;
int64 f[N], S[N], T[N], V[N];
int n, m, p, x, y;

int main(){
	
	scanf("%d%d%d", &n, &m, &p);
	for(int i=1; i<n; i++) {scanf("%d", &x); S[i+1]=S[i]+x;}
	for(int i=1; i<=m; i++) {scanf("%d%d", &x, &y); T[i]=y-S[x];}
	
	sort(T+1, T+m+1);
	for(int i=m; i>=1; i--) T[i] -= T[1];
	for(int i=1; i<=m; i++) S[i] = S[i-1] + T[i];
	for(int i=1; i<=m; i++) f[i] = T[i]*i-S[i];
	
	for(int j=2; j<=p; j++){
		for(int i=1; i<=m; i++) V[i] = f[i]+S[i];
		h=0; t=-1;
		for(int i=1; i<=m; i++){
			while(h<t && (V[Q[h+1]]-V[Q[h]])<T[i]*(Q[h+1]-Q[h])) ++h;
			while(h<t && (V[Q[t]]-V[Q[t-1]])*(i-Q[t])>(V[i]-V[Q[t]])*(Q[t]-Q[t-1])) --t;
			Q[++t] = i;
			f[i] = V[Q[h]] + T[i]*(i-Q[h]) - S[i];
		}
	}
	
	printf("%I64d\n", f[m]);
	
	return 0;
}
