n,m,k = map(int,raw_input().split())
rf=gf=False
ra = []
u = [0]*10
for i in xrange(n):
    s = raw_input()
    g,r = map(s.find,'GR')
    if r<0 or g<0:
        if r<0 and g<0 or '-' not in s: continue        
        if r<0: gf=True
        else: rf=True
        continue
    for i,v in enumerate(bin(abs(r-g)-1)[:1:-1]): u[i]+=int(v)
if gf and rf: print "Draw"
else:
    if any(i%(k+1) for i in u) or gf: print "First"
    else: print "Second"
