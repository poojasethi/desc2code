#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std; 

int	n,m,k; 
char map[105][105]; 
int	ans[100];

int main()
{
	scanf("%d%d%d" , &n , &m , &k); 
	for (int i = 0; i<n; i++) scanf("%s" , map[i] ); 
	if ( m == 1 ) { printf("Second\n"); return 0; }
	int g ,h , t = 0, now = 0; 
	for (int i = 0; i<n; i++)
	{
		g = h = -1;
		for (int j = 0; j<m; j++)
			if ( map[i][j] != '-' ) 
			{
				if ( g == -1 ) g = j; 
				else h = j; 
			}
		if ( g == -1 ) continue; 
		if ( h == -1 )
		{
			if ( map[i][g] == 'G' ) now = now | 1; 
			else now = now | 2; 
			continue; 
		}
		if ( map[i][g] == map[i][h] )
		{
			if ( m == 2 ) continue;
			if ( map[i][g] == 'G' ) now = now | 1; 
			else now = now | 2; 
			continue; 
		}
		else
		{
			int l = abs(g-h)-1 , i = 0; 
			while ( l > 0 )
			{
				ans[i] += l%2; 
				l = l/2; 
				i++; 
			}
		}
	}
	if ( now == 2 ) printf("Second\n"); 
	else if ( now == 1 ) printf("First\n"); 
	else if ( now == 3 ) printf("Draw\n");
	else 
	{
		bool flag = false;
		for (int i = 0; i<100; i++)
			if ( ans[i]%(k+1) != 0 ) { flag = true; break; }
		if ( flag ) printf("First\n"); 
		else printf("Second\n"); 
	}

	return 0; 
}
