using namespace std;
int n,m,k;
char s[103];
vector<int> a;
int get()
{
    int f=0,v=0;char ch;
    while(!isdigit(ch=getchar()))if(ch=='-')break;
    if(ch=='-')f=1;else v=ch-48;
    while(isdigit(ch=getchar()))v=v*10+ch-48;
    if(f==1)return -v;else return v;
}
int main()
{
    scanf("%d%d%d\n",&n,&m,&k);
    bool f1=0,f2=0;
    a.clear();
    for(int i=1;i<=n;i++)
    {
        scanf("%s",s);
        int x=-1,y=-1,cnt=0;
        for(int j=0;j<m;j++)
        {
            if(s[j]=='G')x=j;
            if(s[j]=='R')y=j;
            if(s[j]!='-')cnt++;
        }
        if(x!=-1&&y==-1&&cnt<m)f1=1;
        if(x==-1&&y!=-1&&cnt<m)f2=1;
        if(x!=-1&&y!=-1)a.push_back(abs(x-y)-1);
    }
    if(f1&&!f2)puts("First");else
    if(!f1&&f2)puts("Second");else
    if(f1&&f2)puts("Draw");else
    {
        for(int i=0;i<10;i++)
        {
            int tp=0;
            for(int j=0;j<(int)a.size();j++)
            {
                tp+=a[j]%2;
                a[j]/=2;
            }
            if(tp%(k+1)!=0)
            {
                puts("First");
                return 0;
            }
        }
        puts("Second");
    }
    return 0;
}
