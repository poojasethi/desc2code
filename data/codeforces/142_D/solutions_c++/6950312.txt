#include<cstdio>
#include<iostream>
#include<cmath>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<queue>
#define mk make_pair
using namespace std;
int i,j,k,l,n,m,s,an,sum,tot,len,K,r,MA,MI,M,s1,s2,L,R;
const int N=110;
char a[N][N];
int g[N];
int main()
{
    scanf("%d%d%d",&n,&m,&K);
    if (m==1) {printf("Second\n");return 0;}
    for (int i=1;i<=n;i++)
    scanf("%s",a[i]+1);
    s=0;
    bool ans1=false,ans2=false;
    for (int i=1;i<=n;i++)
    {
        int x1=0,x2=0;
        int s1=0,s2=0;
        for (int j=1;j<=m;j++)
        if (a[i][j]=='G') x1=j,s1++;
        else if (a[i][j]=='R') x2=j,s2++;
        if ((!x1)&&(!x2)) continue;
        if (!x2)
        {
            if((s1==2)&&(m==2)) continue;
            ans1=true;
            continue;
        }
        else
        if (!x1)
        {
            if((s2==2)&&(m==2)) continue;
            ans2=true;
            continue;
        }
        //if ((!x1)||(!x2)) {printf("Draw\n");return 0;}
        if (abs(x1-x2)!=1)
        s++,g[++tot]=abs(x1-x2)-1;
    }
    if (ans1&&ans2){printf("Draw\n");return 0;}
    if (ans1) {printf("First\n");return 0;}
    if (ans2) {printf("Second\n");return 0;}
    for (int i=0;i<20;i++)
    {
        int s=0;
        for (int j=1;j<=tot;j++)
        s+=((g[j]&(1<<i))>0);
        if (s%(K+1)!=0) {printf("First\n");return 0;}
    }
    printf("Second\n");
    return 0;
}
