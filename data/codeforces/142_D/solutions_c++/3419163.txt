#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;

const int N = 105;

int f[N], n, m, k;
char str[N][N];

int main () {
#ifdef LOCAL
	freopen("in", "r", stdin);
#endif
	int isoR = 0, isoG = 0, status = 1, hasPair = 0;
	// status = 0 : First wins; 1 : otherwise
	scanf("%d%d%d", &n, &m, &k);
	for (int i = 1; i <= n; ++i) {
		scanf("%s", str[i] + 1);
		int rpos = -1, gpos = -1, hasBlank = false;
		for (int j = 1; j <= m; ++j)
			if (str[i][j] == 'R') rpos = j;
			else if (str[i][j] == 'G') gpos = j;
			else hasBlank = true;
		if (!hasBlank) {
			continue;
		}
		if (rpos == -1 && gpos >= 0) isoG = 1;
		if (gpos == -1 && rpos >= 0) isoR = 1;
		if (rpos > 0 && gpos > 0) f[i] = abs(rpos - gpos) - 1, hasPair = 1;
	}
	if (! hasPair) {
		if (isoG && isoR) puts("Draw");
		else puts(isoG ? "First" : "Second");
		return 0;
	}
	for (int i = 9; i >= 0; --i) {
		int bsum = 0;
		for (int j = 1; j <= n; ++j) bsum += (f[j] >> i) & 1;
		if (bsum % (k + 1) != 0) { status = 0; break; }
	}
	if (! status) {
		puts(isoR ? "Draw" : "First");
	}
	else {
		puts(isoG ? "Draw" : "Second");
	}
	return 0;
}
