#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cctype>
#include <algorithm>
using namespace std;

char map[128][128];
int stone[128];

int main() {

    // freopen("D.in","r",stdin);

    int n, m, k;
    scanf("%d%d%d",&n,&m,&k);

    for (int i = 0; i < n; i++)
        scanf("%s",map[i]);

    int countG = 0, countR = 0;
    for (int i = 0; i < n; i++) {
        int ag = 0;
        int ar = 0;
        int pg,pr;
        for (int j = 0; j < m; j++)
            if (map[i][j]=='G')
                ag++,pg = j;
            else 
                if (map[i][j]=='R')
                    ar++,pr = j;
        if (ag && ar) {
            stone[i] = abs(pg-pr) - 1;
        } else if (ag > 0) {
            countG += ag<m;
        } else if (ar > 0) {
            countR += ar<m;
        }
    }

    if (countG > 0 && countR > 0)
        return puts("Draw")&&0;
    if (countG > 0) 
        return puts("First")&&0;
    if (countR > 0) 
        return puts("Second")&&0;

    for (int i = 0; i < 31; i++) {
        int s = 0;
        for (int j = 0; j < n; ++j) {
            s += (stone[j] >> i) & 1;
        }
        if (s % (k + 1) != 0) {
            puts("First");
            return 0;
        }
    }

    puts("Second");return 0;
}
