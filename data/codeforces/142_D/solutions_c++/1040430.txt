using namespace std;

char a[200][200];
int ans[10];

int main()
{
	int i,j,m,n,k,b1,b2,g1,g2,em,tmp;
	scanf("%d%d%d",&m,&n,&k);
	for (i=0;i<m;i++)
		scanf("%s",a[i]);
	g1=0;g2=0;
	memset(ans,0,sizeof(ans));
	for (i=0;i<m;i++)
	{
		b1=-1;b2=-1;em=-1;
		for (j=0;j<n;j++)
		{
			if (a[i][j]=='G')
			{
				b1=j;
			}
			if (a[i][j]=='R')
			{
				b2=j;
			}
			if (a[i][j]=='-')
			{
				em=j;
			}
		}
		if ((b2!=-1)&&(b1==-1)&&(em!=-1)) g2=1;
		if ((b1!=-1)&&(b2==-1)&&(em!=-1)) g1=1;
		if ((b1!=-1)&&(b2!=-1))
		{
			tmp=abs(b1-b2)-1;
			for (j=0;j<10;j++)
			{
				ans[j]+=tmp%2;
				ans[j]%=(k+1);
				tmp/=2;
			}
		}
	}
	if ((g1==1)&&(g2==1))
	{
		printf("Draw\n");
		return 0;
	}
	if (g1==1)
	{
		printf("First\n");
		return 0;
	}
	if (g2==1)
	{
		printf("Second\n");
		return 0;
	}
	for (i=0;i<10;i++)
		if (ans[i]!=0) break;
	if (i==10)
	{
		printf("Second\n");
	}
	else printf("First\n");
	return 0;
}
