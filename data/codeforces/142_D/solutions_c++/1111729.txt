#include<cstdio>
#define FOR(i,a,b) for(int i=a;i<=b;i++)
using namespace std;
int c[10],i,j,n,p1,p2,k,m;
char ch,c1,c2;
bool kr,kg;


int main(){
  scanf("%d%d%d",&n,&m,&k);
  scanf("%c",&ch);
  kr=kg=false;
  FOR(i,1,n){
    p1=p2=0;
    FOR(j,1,m){
      scanf("%c",&ch);
      if((ch=='R')||(ch=='G')){
        if(p1==0)p1=j,c1=ch;
        else p2=j,c2=ch;
      }
    }
    scanf("%c",&ch);
    if(p1!=0){
      if(((p2==0)&&(m>1))||((p2!=0)&&(m>2)&&(c1==c2))){
        if(c1=='G')kg=true;else kr=true;
      }else{
        int x=p2-p1-1;
        FOR(h,1,10)c[h]=(c[h]+(x%2))%(k+1),x=x/2;
      }
    }
  }
  if(kg&&kr)puts("Draw");else
  if(kg)puts("First");else
  if(kr)puts("Second");else{
    kg=false;
    FOR(i,1,10)if(c[i])kg=true;
    if(kg)puts("First");else puts("Second");
  }
}
