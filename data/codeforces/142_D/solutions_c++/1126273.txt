#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#include <cmath>

using namespace std;

const int MAXN = 105;

int i, j, k, N, M, K, sum, stone[MAXN];
bool win1 = false, win2 = false;
char c;

void init()
 {
     scanf("%d%d%d", &N, &M, &K);
    for (i = 0; i < N; i ++)
     {
           int x = -1, y = -1, b = 0;
           for (j = 0; j < M; j ++)
            {
                  while (c = getchar(), (c != '-') && (c != 'G') && (c != 'R'));
                  if (c == 'G') x = j; if (c == 'R') y = j; if (c == '-') ++ b;
            }
           if (x == -1 && y >= 0 && b > 0) win2 = true;
           if (y == -1 && x >= 0 && b > 0) win1 = true;
           if ((x >= 0) && (y >= 0)) stone[++ stone[0]] = abs(x-y)-1;
     }
 }

void work()
 {
    if (win1 && win2) puts("Draw");
    else if (win1) puts("First");
    else if (win2) puts("Second");
    else
     {
         for (k = 0; k < 9; k ++)
          {
             sum = 0;
             for (i = 1; i <= stone[0]; i ++)
              {
                    sum += (stone[i] & 1);
                    stone[i] >>= 1;
              }
             if (sum % (K+1) != 0)
              {
                puts("First");
                return;
              }
          }
         puts("Second");
     }
 }
  
int main()
 {
    init();
    work();
    return 0;
 }
