#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 110
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,K,g,r;
int a[10];
char p[N];
inline void into(int x) {
	int i;
	For(i,0,9) a[i]+=x>>i&1;
}
int main() {
	scanf("%d%d%d",&n,&m,&K);
	For(i,1,n) {
		scanf("%s",p+1);
		int gg,rr,A,B;
		gg=rr=A=B=0;
		For(j,1,m) {
			gg|=p[j]=='G';
			rr|=p[j]=='R';
			if (p[j]!='-') {
				if (!A) A=j;
				else B=j;
			}
		}
		if (!B&&m>1) g|=gg,r|=rr;
		if (B&&m>2&&(!gg||!rr)) g|=gg,r|=rr;
		if (B&&gg&&rr) into(B-A-1);
	}
	if (g&&r) printf("Draw\n");
	else if (g) printf("First\n");
	else if (r) printf("Second\n");
	else {
		For(i,0,9) if (a[i]%(K+1)) {
			printf("First\n");
			break;
		}
		if (i==10) printf("Second\n");
	}
	//for(;;);
	return 0;
}
