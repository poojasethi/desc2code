const int N=105;

int n,m,k;

int main()
{
//	freopen("input.txt","r",stdin);
//	freopen("output.txt","w",stdout);

  cin>>n>>m>>k;
  scanf("\n");
  if (m==1)
  {
    puts("Second");
    return 0;
  }
  int A=0,B=0;
  for (int i=1;i<=n;++i)
  {
    int x=0,y=0,xx=0,yy=0;
    for (int j=1;j<=m;++j)
    {
      char c;
      scanf("%c",&c);
      if (c=='G') x=j,++xx;
      if (c=='R') y=j,++yy;
    }
    scanf("\n");
    if (x==0&&y!=0&&yy!=m) B=1;
    if (x!=0&&y==0&&xx!=m) A=1;
    if (!x||!y) continue;
    a[i]=abs(x-y)-1;
  }
  if (A&&B) 
  {
    puts("Draw");
    return 0;
  }
  if (A&&!B)
  {
    puts("First");
    return 0;
  }
  if (!A&&B)
  {
    puts("Second");
    return 0;
  }
  int t=0;
  for (int i=1;i<=10;++i)
  {
    int s=0;
    for (int j=1;j<=n;++j) s+=a[j]%2,a[j]/=2;
    s%=(k+1);
    t+=!s;
  }
  if (t==10)
    puts("Second");
  else
    puts("First");

	return 0;
}
