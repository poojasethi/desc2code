#include<cstdio>
char tmp[110];
int ans[20];
int main()
{
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    int fg=0,fb=0;
    int cnt=0;
    for(int i=0;i<n;i++)
    {
        scanf("%s",tmp);
        int l=-1,r;
        int mg=0,mb=0;
        for(int j=0;j<m;j++)
        {
            if(tmp[j]=='G')
            {
                mg++;
            }
            if(tmp[j]=='R')
            {
                mb++;
            }
            if(tmp[j]!='-')
            {
                if(l==-1)l=j;
                else r=j;
                //printf("%d %d\n",l,r);
            }
        }
        //printf("%d %d %d %d\n",l,r,mg,mb);
        if(mb==0&&mg<m&&mg>0)fg=1;
        if(mg==0&&mb<m&&mb>0)fb=1;
        if(mg==1&&mb==1&&r-l-1>0)
        {
            int tu=r-l-1;
            for(int i=0,mark=1;mark<=tu;i++,mark<<=1)if(tu&mark)
            {
                ans[i]++;
            }
        }
    }
    int ff=0;
    for(int i=0;i<20;i++)if(ans[i]%(k+1))ff=1;
    if(fg&&fb)printf("Draw\n");
    else if(fg)printf("First\n");
    else if(fb)printf("Second\n");
    else printf("%s\n",(ff)?"First":"Second");
}
