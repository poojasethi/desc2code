#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <vector>
using namespace std;

int n,m,k;
char s[105][105];
vector<int> stone;
bool freeRow[2]={false,false};

int main()
{
	scanf("%d%d%d",&n,&m,&k);
	for (int i=0;i<n;++i){
		scanf("%s",s[i]);
		vector<int> p[2];
		for (int j=0;j<m;++j){
			if (s[i][j]=='G'){
				p[0].push_back(j);
			}else if (s[i][j]=='R'){
				p[1].push_back(j);
			}
		}
		if (p[0].size()+p[1].size()==0){
			continue;
		}
		if (p[0].size()==0 && p[1].size()){
			if (p[1].size()<m){
				freeRow[1]=true;
			}
		}else if (p[1].size()==0 && p[0].size()){
			if (p[0].size()<m){
				freeRow[0]=true;
			}
		}else{
			//1 + 1
			int x=abs(p[0][0]-p[1][0])-1;
			if (x>0){
//printf("stone %d\n",x);
				stone.push_back(x);
			}
		}
	}
	if (freeRow[0]){
		if (freeRow[1]){
			puts("Draw");
			return 0;
		}else{
			puts("First");
			return 0;
		}
	}
	if (freeRow[1]){
		puts("Second");
		return 0;
	}
	
	for (int bit=0;bit<20;++bit){
		int sum=0;
		for (int i=0;i<stone.size();++i){
			sum+=stone[i]>>bit&1;
		}
//printf("bit %d, sum= %d\n",bit,sum);
		if (sum%(k+1)){
			puts("First");
			return 0;
		}
	}
	puts("Second");
	return 0;
}
