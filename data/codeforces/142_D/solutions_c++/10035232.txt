#include <cstdio>
#include <cstring>
#include <vector>
#include <cmath>
using namespace std;

const int N = 100+5, LOG = 11;

char t[N];
vector<int> gap;
int sum[LOG],n,m,k;

inline void add(int v){
    for (int i=0;i<LOG;i++)
        if (v>>i&1)
            sum[i]++, sum[i]%=(k+1);
}

int main(){
    bool aonly = 0, bonly = 0;
    scanf("%d%d%d",&n,&m,&k);
    for (int r=0;r<n;r++){
        scanf("%s",t);
        int a = -1, b = -1;
        int na = 0, nb = 0;
        for (int i=0;i<m;i++){
            if (t[i]=='G'){
                a = i;
                na++;
            }
            if (t[i]=='R'){
                b = i;
                nb++;
            }
        }
        if (a==-1 && b==-1)
            continue;
        if (a==-1 && nb!=m)
            bonly = 1;
        else if (b==-1 && na!=m)
            aonly = 1;
        else  if (a!=-1 && b!=-1) add(abs(a-b)-1);
    }

    if (!aonly && bonly){
        printf("Second\n");
        return 0;
    }
    if (aonly && !bonly){
        printf("First\n");
        return 0;
    }
    if (aonly && bonly){
        printf("Draw\n");
        return 0;
    }

    for (int i=0;i<LOG;i++)
        if (sum[i]){
            printf("First\n");
            return 0;
        }
    printf("Second\n");

    return 0;
}
