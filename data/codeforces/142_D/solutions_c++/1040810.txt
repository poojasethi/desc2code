#define sqr(x) ((x) * (x))
using namespace std;


typedef	long long	int64;


char	a[105][105];
int	b[105], M, N, K;
int	Rapp = 0, Gapp = 0, Rdraw = 0, Gdraw = 0, First = 0;


void	Return(int x)
{
	if (x == 0) puts("Draw");
	if (x == 1) puts("First");
	if (x == 2) puts("Second");
	exit(0);
}

int	main()
{
//	freopen("D.in", "r", stdin);
//	freopen("D.out", "w", stdout);
	
	scanf("%d%d%d", &M, &N, &K);
	
	if (N == 1)
	{
		Return(2);
		exit(0);
	}
	
	for (int i = 1; i <= M; ++ i)
	{
		scanf("%s", a[i] + 1);
		
		int	R = 0, G = 0;
		for (int j = 1; j <= N; ++ j)
		{
			R += (a[i][j] == 'R');
			G += (a[i][j] == 'G');
		}
		
		Rapp |= (R > 0);
		Gapp |= (G > 0);
		
		if (!R && !G) continue; else
		if (!G) Rdraw |= (R != N); else
		if (!R) Gdraw |= (G != N); else
		{
			b[i] = N;
			for (int j = 1; a[i][j] == '-'; ++ j) -- b[i];
			for (int j = N; a[i][j] == '-'; -- j) -- b[i];
			b[i] -= 2;
		}
	}
	
	if (!Rapp && Gapp) Return(2);
//	if (!Gapp && Rapp) Return(1);
//	if (!Rapp && !Gapp) Return(2);
	
	if (Rdraw && Gdraw) Return(0);
	if (Rdraw) Return(2);
	if (Gdraw) Return(1);
	
	for (int k = 0; k < 10; ++ k)
	{
		int	C = 0;
		for (int i = 1; i <= M; ++ i) C += (b[i] & (1 << k)) > 0;
		if (C % (K + 1)) First = 1;
	}
	
	if (First) Return(1); else Return(2);
	
	return 0;
}
