#include<cstdio>
#include<cstring>
int F(int a){if(a>0)return a;return -a;}
int main()
{
	int cnt[8]={0};
	int i,j;
	int n,m,k,d;
	int g,r,g2,r2,G,R;
	char s[101];
	scanf("%d%d%d",&n,&m,&k);
	G=R=0;
	for(i=0;i<n;i++)
	{
		scanf("%s",s);
		g=r=-1;
		g2=r2=0;
		for(j=0;j<m;j++)
		{
			if(s[j]=='G')
			{
				g2=1;
				if(g!=-1)
					g2=2;
			     g=j;
			}
			if(s[j]=='R')
			{
				r2=1;
				if(r!=-1)
					r2=2;
				r=j;
			}
		}
		if(g2==m||r2==m)
			continue;
		if(g==-1&&r==-1)
			continue;
		if(r==-1)
			G=1;
		if(g==-1)
			R=1;
		if(G==1||R==1)
			continue;
		d=F(g-r)-1;
		int l=0;
		while(d)
		{
			if(d&1)cnt[l]++;
			d>>=1;
			l++;
		}
	}
	if(G==1&&R==1)
		printf("Draw\n");
	else if(G==1)
		printf("First\n");
	else if(R==1)
		printf("Second\n");
	else
	{
		for(i=0;i<8;i++)
		{
			if(cnt[i]%(k+1)!=0)
			{
				printf("First\n");
				return 0;
			}
		}
		printf("Second\n");
	}
	return 0;
}
