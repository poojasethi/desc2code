#include <iostream>
#define LL long long
using namespace std;

LL C(LL n, LL k)
{
	LL result=1;
	for(LL i=1; i<=k; i++)
	{
		result*=(n+1-i);
		result/=i;
	}
	return result;
}

int main()
{
	LL n, m, t;
	cin >> n >> m >> t;

	LL result=0;
	for(int i=1; i<=t-4; i++) result+=C(n,t-i)*C(m,i);
	cout << result;
	return 0;
}
