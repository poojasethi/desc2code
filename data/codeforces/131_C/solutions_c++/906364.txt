#include <iostream>
#define LL  long long
using namespace std;
LL n,m,t,ans;
LL c(LL b,LL a){
    LL r=1;
    for(LL i=1;i<=a;i++)
    {
        r*=b-i+1;
        r/=i;
    }
    return r;
}
int main(){
    cin>>n>>m>>t;
    for(LL i=4;i<t;i++)
        ans+=c(n,i)*c(m,t-i);
    cout<<ans;
}
