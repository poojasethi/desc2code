#include <iostream>

using namespace std;
long long n,m,t,i,j,sol,c[62][62];
int main()
{
    cin>>n>>m>>t;
    for(i=0;i<=30;i++)c[i][0]=c[i][i]=1;
    for(i=2;i<=30;i++)for(j=1;j<i;j++)c[i][j]=c[i-1][j-1]+c[i-1][j];
    for(i=1,j=t-1;j>=4;i++,j--)
        sol+=c[n][j]*c[m][i];
    cout<<sol;
    return 0;
}
