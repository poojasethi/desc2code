#include <iostream>
using namespace std;


long long C(int n,int m)  //claculates combination C(n,m)
{
 if(m==1) return n;
long long ans=1; 
int i;
for(i=1;i<=m;i++)
ans=(ans*(n-i+1))/i;
return ans;
}

int main()
{
int n,m,t,i,j;

{
 k=C(n,i);

}
