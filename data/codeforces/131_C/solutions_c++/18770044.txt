#include<bits/stdc++.h>
using namespace std;
long long C(int n,int m){
    long long res=1;
    for(int i=0;i<m;i++){
        res=res*(n-i)/(i+1);
    }
    return res;
}
int main(){
long long n,m,k;cin>>n>>m>>k;
    long long sum=0;
    for(int i=4;i<k;i++){
        sum+=C(m,k-i)*C(n,i);
    }
cout<<sum<<endl;

return 0;}
