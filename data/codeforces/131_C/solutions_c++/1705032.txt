#include <stdio.h>
__int64 c(__int64 m, __int64 n)
{
	if (n>m) return 0;
	__int64 i,j,ans=1;
	for (i=1, j=m; i<=m-n; i++, j--)
		ans=ans*j/i;
	return ans;
}
int main()
{
	__int64 n,m,t,i,ans=0;
	scanf("%I64d%I64d%I64d",&n,&m,&t);
	for (i=4; t-i>0; i++)
		ans+=c(n,i)*c(m,t-i);
	printf("%I64d\n",ans);
	return 0;	
}
