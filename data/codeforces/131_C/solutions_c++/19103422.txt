#include <iostream>
typedef __int64 ll;
using namespace std;
ll c(ll a, ll b)
{
	ll ans=1, i;
	for(i=1;i<=b;i++)
	{
		ans*=a-i+1;
		ans /=i;
	}
	return ans;
}
int main()
{
	ll t,b,g;
	ll ans=0;
	cin>>b>>g>>t;
	for(ll i=4;i<t;i++)
	{
		ans+= c(b,i)*c(g, t-i);
	}
	cout<<ans <<endl;
}
