#include<iostream>
using namespace std;
int n,m,t,i,j;
long long ans,zh1,zh2;
int main()
{
	cin>>n>>m>>t;
	for(i=4;i<t;i++)
	{
		zh1=1LL;
		for(j=n;j>=n+1-i;j--)zh1=zh1*(1LL*j)/(1LL*(n+1-j));
		zh2=1LL;
		for(j=m;j>=m+1-t+i;j--)zh2=zh2*(1LL*j)/(1LL*(m+1-j));
		ans=ans+zh1*zh2;
	}
	cout<<ans;
	return 0;
}
