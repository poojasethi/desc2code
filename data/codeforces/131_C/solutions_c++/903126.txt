#include<iostream>
using namespace std;

__int64 Orz(int n,int i)
{
    __int64 sum=1;
    for(int j=1;j<=i;j++)
    {
        sum*=n-j+1;
        sum/=j;
    }
    return sum;
}


int main()
{
    int a,b,c;
    cin>>a>>b>>c;
    __int64 sum=0;
    for(int i=4;i<c;i++)
    sum+=Orz(a,i)*Orz(b,c-i);
    cout<<sum<<endl;
}
