#include <cstdio>
using namespace std;
#define ll long long
ll f(ll a,ll b)
{
    ll ans=1;
    for(int i=1;i<=b;++i)
        ans=ans*(a-i+1)/i;
    return ans;
}
int main()
{
    int n,m,t;ll ans=0;
    scanf("%d%d%d",&n,&m,&t);
    for(int i=4;i<t;++i)
        ans+=f(n,i)*f(m,t-i);
    printf("%I64d\n",ans);
}
