#include<iostream>

using namespace std;

long long chs(int a , int b)
{
	long long ans = 1;
	for(int i = 0; i < b; i++)
	{
		ans = ans * (a-i)/(i+1);
	}
	return ans;
}

int n , m , t;

int main()
{
	cin >> n >> m >> t;
	cout << chs(n+m,t)-chs(n,t)-chs(m,t)-chs(n,1)*chs(m,t-1)-chs(n,2)*chs(m,t-2)-chs(n,3)*chs(m,t-3);
}
