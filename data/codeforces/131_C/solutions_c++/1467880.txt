#include<cstdio>
__int64 C(__int64 n, __int64 k )
{
	__int64 i,cb=1;
	if(n-k<k) k=n-k;
	for(i=1;i<=k;i++) cb=cb*(n-k+i)/i;
	return cb;
}
int main()
{
	__int64 i,n,m,t,s=0;
	scanf("%I64d%I64d%I64d",&n,&m,&t);
	for(i=4;i<t&&i<=n;i++)
	{
		if(t-i<=m) s+=C(n,i)*C(m,t-i);
	}
	printf("%I64d\n",s);
	return 0;
}
