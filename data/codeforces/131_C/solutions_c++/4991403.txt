#include<iostream>
using namespace std;
long long c[61][61],n,m,t;
int main()
{
	c[0][0]=1;cin>>n>>m>>t;
	for(int i=1;i<=n+m;i++)
	{
		c[i][0]=1;
		for(int j=1;j<=n+m;j++)
		c[i][j]=c[i-1][j]+c[i-1][j-1];
	}
	cout<<c[n+m][t]-c[n][t]*c[m][0]-c[n][3]*c[m][t-3]-c[n][2]*c[m][t-2]-c[n][1]*c[m][t-1]-c[n][0]*c[m][t]<<endl;
	return 0;
}
