#include<iostream>
using namespace std;
typedef long long ll;
ll n,m,t,ans;
ll C(ll a,ll b);
int main()
{
	while(cin>>n>>m>>t)
	{
		ans=0;
		for(int i=4;i<=n;i++)
		{
			int j=t-i;
			if(j<1)
				break;
			ans=ans+C(n,i)*C(m,j);
		}
		cout<<ans<<endl;
	}
	return 0;
}
ll C(ll a,ll b)
{
	return b==0?1:a*C(a-1,b-1)/b;
}
