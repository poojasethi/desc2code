#include<stdio.h>
#define LL __int64
LL C(int n,int m){
	LL ans=1;
	for(int i=0;i<m;i++){
		ans*=n-i;
		ans/=i+1;
	}
	return ans;
}
int main(){
	int  n,m,t,i;
	while(scanf("%d %d %d",&n,&m,&t)!=EOF){
		LL ans=0,tmp=1;
		for(i=4;i<=n&&i<t;i++){
			ans+=tmp*C(n,i)*C(m,t-i);
		}
		printf("%I64d\n",ans);
	}
	return 0;
}
