#include <bits/stdc++.h>
using namespace std;
#define LL long long
LL C[32][32];
int main()
{
	for(int i = 0;i<=30;i++)
	{
		C[i][0] = 1;
		for(int j = 1;j<=i;j++)
		{
			C[i][j] = C[i-1][j-1] + C[i-1][j];
		}
	}
	int N,M,T;
	cin>>N>>M>>T;
	LL Ans = 0;
	for(int i = 4;i<=N;i++)
	{
		if(T-i>=1&&T-i<=M)
			Ans+=C[N][i]*C[M][T-i];
	}
	cout<<Ans<<endl;
}
