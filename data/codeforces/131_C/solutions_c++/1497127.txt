#include<iostream>
using namespace std;
int n,m,t;
long long Cal(int n,int k){
	long long ret=1;
	for(int i=0;i<k;++i){ret*=n-i;ret/=i+1;}
	return ret;
}
int main()
{
    cin>>n>>m>>t;
	long long ans=0;
	for(int i=4;i<=n;i++)
	  for(int j=1;j<=m;j++)
	    if(i+j==t){ans+=Cal(n,i)*Cal(m,j);}
	cout<<ans<<endl;
    return 0;
}
