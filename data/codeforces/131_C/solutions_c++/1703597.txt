#include<stdio.h>
__int64 zuhe(__int64 x,__int64 y){
__int64 bc=1,i,k;
for( i=y-x+1, k=1;i<=y;i++,k++)bc=bc*i/k;
return bc;
}
int main(){
__int64 num=0,i, n,m,t;
scanf("%I64d%I64d%I64d",&n,&m,&t);
for( i=1;i<=t-4;i++)
if((i<=m)&&(t-i<=n))
num+=zuhe(i,m)*zuhe(t-i,n);
printf("%I64d\n",num);

return 0;
}
