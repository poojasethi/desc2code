#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
typedef long long ll;
using namespace std;
ll c(ll a,ll b){
	ll ans=1,i;
	for(i=1;i<=b;i++){
		ans*=a-i+1;
		ans/=i;
	}
	return ans;
}
int main(){
	ll n,b,g,ans=0,i;
	cin>>b>>g>>n;
	for(i=4;i<n;i++)
	{
		ans+=c(b,i)*c(g, n-i);
	}
	cout<<ans<<endl;
}
