#include<iostream>
using namespace std;
typedef long long ll;

ll c(ll a, ll b)
{
	ll ans=1;
	for(ll i=1; i<=b; i++)
	{
		ans*=a-i+1;
		ans/=i;
	}
	return ans;
}
int main()
{
	ll t,n,m;
	while(cin>>n>>m>>t)
	{
		ll ans=0;
		for(ll i=4;i<t;i++)
			ans+=c(n,i)*c(m, t-i);
		cout<<ans<<endl;
	}
	return 0;
}
