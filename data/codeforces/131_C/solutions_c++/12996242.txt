#include <iostream>
using namespace std;
long long int C(int n,int s){
    long long res=1;
    for(int i=0;i<s;i++)res*=n-i,res/=i+1;
    return res;
}
int main(){
    long long int ans=0,n,m,t;
    cin>>n>>m>>t;
    for(int i=4;i<=n && i<t;i++){
        ans+=C(n,i)*C(m,t-i);
    }
    cout<<ans;

    return 0;
}
