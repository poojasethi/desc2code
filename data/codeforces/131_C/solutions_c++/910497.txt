#include <iostream>
using namespace std;

long long qweqwe(int n,int m){
long long rez=1;
for(int i=1;i<=m;i++){
rez*=n-i+1;
rez/=i;
}
return rez;
}
int main ()
{
    int m,n,t;
    long long summ=0;
    cin>>n>>m>>t;
    for(int i=4;i<t;i++){
    summ+=qweqwe(n,i)*qweqwe(m,t-i);
    }
    cout<<summ; 
    return 0;
}
