#include <cstdio>

int main()
{
  long long C[62][62];
  C[0][0] = 1;
  for(int n=1; n<=60; n++)
  {
    C[n][0] = C[n][n] = 1;
    for(int i=1; i<n; i++)
      C[n][i] = C[n-1][i-1]+C[n-1][i];
  }
  int n,m,t;
  scanf("%d%d%d",&n,&m,&t);
  printf("%I64d\n",C[n+m][t]-C[n][0]*C[m][t]-C[n][1]*C[m][t-1]-C[n][2]*C[m][t-2]-C[n][3]*C[m][t-3]-C[n][t]);
}
