#include <iostream>

using namespace std;

long long c[64][64];
 

int main() {
	int n,m,t;
	cin >> n >> m >> t;
	for(int i=0;i<=30;i++) {
		c[i][0]=1;
		for(int j=1;j<=i;j++) c[i][j]=c[i-1][j-1]+c[i-1][j];
	}
	long long res=0;
	for(int i=4;i<t;i++) res += c[n][i]*c[m][t-i];
	cout << res << endl;
	return 0;
}
