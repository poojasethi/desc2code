def c(a,b):
	if(b>a):
		return 0;
	if(b==0):
		return 1;
	sum=1;
	d=a-b
	while(a>0):
		sum*=a;
		a=a-1;
	while(b>0):
		sum=sum/b;
		b=b-1;
	while(d>0):
		sum=sum/d;
		d=d-1;  
	return sum;

n,m,t=map(int,raw_input().split());
i=4;
ans=0;
while(i<=n and t-i>=1):
	ans=ans+c(n,i)*c(m,t-i);
	i=i+1;
print ans
