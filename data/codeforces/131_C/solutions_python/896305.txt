import operator
def C(n, k):
	if n < k:
		return 0
	if k == 0:
		return 1
	return reduce(operator.mul, xrange(n-k+1, n+1), 1)/reduce(operator.mul, xrange(1,k+1), 1)

n, m, t = map(int, raw_input().split())
sumv = 0
if n >= 4 and m >= 1 and m+n >= t:
	for i in xrange(4, t):
		sumv += C(n, i) * C(m, t - i)
print sumv
