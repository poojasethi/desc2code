import sys
comb = [[1] * 40 for x in xrange(40)]
for x in xrange(2,40):
    for y in xrange(1,x):
        comb[x][y] = comb[x-1][y-1] + comb[x-1][y]
n, m, t = map(int, sys.stdin.readline().split())
ans = 0
for b in range(4, t):
    g = t - b
    if b > n or g > m: continue
    ans += comb[n][b] * comb[m][g]
print ans
