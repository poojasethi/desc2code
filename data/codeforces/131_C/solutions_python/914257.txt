def c(kk, nn):
    r = 1
    for j in range(1, kk+1):
        r *= (nn - kk + j)
    for j in range(1, kk+1):
        r /= j
    return r
        

n, m, k = map(int,raw_input().split())
a = 0
for i in range(1, k+1):
    if i <= m and k - i <= n and k - i >= 4:
        a += c(k - i, n) * c(i, m)
print a
