s = raw_input().split()
m = int(s[0])
n = int(s[1])
t = int(s[2])

def factor(k):
    s=1;
    for i in range(1,k+1):
        s*=i
    return s

def cc(n,k):
    return factor(n)/(factor(k)*factor(n-k))

result =0;
for i in range(max(4,t-n),t):
    j = t-i
    result += cc(m,i)*cc(n,j)
print result
