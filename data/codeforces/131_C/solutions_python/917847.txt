from math import factorial

combination = lambda n, m: factorial(n) / factorial(m) / factorial(n - m)

n, m, t = map(int, raw_input().split())
ans = 0
for i in range(4, n + 1):
    for j in range(1, m + 1):
        if i + j != t: continue
        ans += combination(n, i) * combination(m, j)
print ans
