su=1
ftab=[1]
for i in range(2,80):
  ftab+=[su];
  su*=i
def c(m,n):
  return ftab[n]/(ftab[m]*ftab[n-m]);
inp=raw_input().split(" ");
n=int(inp[0]);m=int(inp[1]);t=int(inp[2]);
cnt=0;
for i in range(4,n+1):
  j=t-i;
  if(j<1 or j>m):continue;
  cnt+=c(i,n)*c(j,m);
print "%d"%(cnt)
