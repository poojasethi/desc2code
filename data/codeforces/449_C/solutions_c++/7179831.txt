#include <iostream>
#include <vector>

using namespace std;

typedef pair<int, int> pii;
const int MAXN = 100010;
vector<pii> ans;
bool marked[MAXN], used[MAXN];
vector<int> v;
int main()
{
	int n;
	cin >> n;
	for (int i = 2; i <= n; i++)
		if (!marked[i])
			for (int j = i * 2; j <= n; j += i)
				marked[j] = true;
	for (int i = n; i >= 2; i--)
		if (!marked[i])
		{
			v.clear();
			for (int j = i; j <= n; j += i)
				if (!used[j])
					v.push_back(j);
			if (v.size() == 1)
				continue;
			if (v.size() % 2 == 1)
				v.erase(v.begin() + 1); // 2 * i
			for (int i = 0; i < v.size(); i += 2)
			{
				ans.push_back(pii(v[i], v[i + 1]));
				used[v[i]] = used[v[i + 1]] = true;
			}
		}
	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i].first << " " << ans[i].second << endl;
}
