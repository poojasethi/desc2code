#include <bits/stdc++.h>
using namespace std;

const int MAXN = 100005;
int n, pr[MAXN] = {};
bool use[MAXN] = {};
vector<int> v;

int main(){
  cin >> n;
  for(int i = 2; i <= n; i++) pr[i] = 1;
  for(int i = 2; i <= n; i++){
    if(!pr[i]) continue;
    for(int j = i + i; j <= n; j += i) pr[j] = 0;
  }
  for(int i = n; i >= 2; i--){
    if(!pr[i] || 2 * i > n) continue;
    vector<int> u;
    for(int j = i; j <= n; j += i){
      if(!use[j]){
        u.push_back(j);
        use[j] = 1;
      }
    }
    if(u.size() % 2 == 1){
      use[u[1]] = 0;
      swap(u[1], u[u.size() - 1]);
      u.pop_back();
    }
    for(int j = 0; j < u.size(); j++) v.push_back(u[j]);
  }
  cout << v.size() / 2 << '\n';
  for(int i = 0; i < v.size(); i += 2) cout << v[i] << ' ' << v[i + 1] << '\n';
}
