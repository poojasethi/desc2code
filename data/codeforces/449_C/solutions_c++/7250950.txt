#define LL long long
using namespace std;
bool mark[1000010];
int ans[1000010][2];
int f[1000010],p[1000010];
int N,last,len,l;
bool f1,f2;
int main(){
	scanf("%d",&N);
	for (int i=2;i<=N/2;i++){
		if (!f[i]){f[i]=i;p[++l]=i;}
		for (int j=1;j<=l&&p[j]<=f[i]&&p[j]*i<=N/2;j++)
		f[p[j]*i]=p[j];
	}
	for (int i=l;i;i--){
		bool f1=false;
		for (int j=1;j<=N/p[i];j++)
		if (!mark[j*p[i]])f1=f1^1;
		for (int j=1;j<=N/p[i];j++)
		if (!mark[j*p[i]]&&(!f1||j!=2||i==1)){
			if (last){
				++len;
				ans[len][0]=last;
				ans[len][1]=j*p[i];
				mark[last]=true;
				mark[j*p[i]]=true;
				last=0;
			}else last=j*p[i];
		}
	}
	printf("%d\n",len);
	for (int i=1;i<=len;i++)
	printf("%d %d\n",ans[i][0],ans[i][1]);
}
