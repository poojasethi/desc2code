#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

bool check[100005];
int n,ans[50005][2],num,a[50005],tot,t[50005],cnt2;
int main() {
	scanf("%d",&n);
	for (int i=3;i<=n/2;i+=2) if (!check[i]) {
		tot=0;
		for (int j=i;j<=n;j+=i) if (!check[j]) a[++tot]=j;
		if (tot%2==1) {
			for (int j=2;j<tot;j++) a[j]=a[j+1];
			tot--;
		}
		for (int j=1;j<=tot;j+=2) {
			num++;
			ans[num][0]=a[j]; ans[num][1]=a[j+1];
			check[a[j]]=check[a[j+1]]=true;
		}
	}
	for (int i=2;i<=n;i+=2) if (!check[i]) t[++cnt2]=i;
	if (cnt2&1) cnt2--;
	for (int i=1;i<=cnt2;i+=2) {
		num++;
		ans[num][0]=t[i]; ans[num][1]=t[i+1];
	}
	printf("%d\n",num);
	for (int i=1;i<=num;i++) printf("%d %d\n",ans[i][0],ans[i][1]);
	return 0;
}
