#include <bits/stdc++.h>
using namespace std;
#define N 100005
bool f[N], p[N];
vector < pair<int, int> > sol;
vector <int> tmp1, tmp2;
int main() {
	int n;
	scanf("%d", &n);
	for(int i = 2; i <= n; i++) if(!p[i]) for(int j = 2 * i; j <= n; j += i) p[j] = 1;
	for(int i = n; i >= 2; i--) if(!p[i]) {
		for(int j = i; j <= n; j += i) {
			if(!f[j]) {
				tmp1.push_back(j);
				if(j != 2 * i) tmp2.push_back(j);
			}
		}
		int u = tmp1.size();
		int v = tmp2.size();
		if((u & 1) == 0) {
			for(int j = 1; j < u; j += 2) {
				sol.push_back(make_pair(tmp1[j - 1], tmp1[j]));
				f[tmp1[j - 1]] = f[tmp1[j]] = 1;
			}
		} else {
			for(int j = 1; j < v; j += 2) {
				sol.push_back(make_pair(tmp2[j - 1], tmp2[j]));
				f[tmp2[j - 1]] = f[tmp2[j]] = 1;
			}
		}
		tmp1.clear(); tmp2.clear();
	}
	printf("%ld\n", sol.size());
	for(int i = 0; i < sol.size(); i++) {
		printf("%d %d\n", sol[i].first, sol[i].second);
	}
	return 0;
}
