#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long LL;
#define rep(i,n) for (int i=1;i<=n;++i)
const int N=100005;
int n,L,u[N],v[N],p[N]; bool b[N];
int main()
{
	scanf("%d",&n),p[1]=1;
	rep(i,n) if (!p[i]) for (int j=i;j<=n;j+=i) if (!p[j]) p[j]=i;
	for (int i=n;i>1;--i) if (p[i]==i){
		int x=-1;
		for (int j=i;j<=n;j+=i) if (!b[j] && p[j]==i)
			if (x<0) x=j; else b[x]=b[j]=1,u[++L]=x,v[L]=j,x=-1;
		if (x>0) for (int j=i;j<=n;j+=i) if (j!=x && !b[j]){b[x]=b[j]=1,u[++L]=x,v[L]=j; break;}
	}
	printf("%d\n",L);
	rep(i,L) printf("%d %d\n",u[i],v[i]);
	return 0;
}
