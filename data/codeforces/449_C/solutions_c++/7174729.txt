#include <iostream>
#include <vector>
using namespace std;

int main()
{
	int n;
	bool v[100005];
	vector<pair<int, int> > res;
	vector<int> t;

	cin >> n;
	for (int i = 2; i <= n; ++i)
		v[i] = false;

	for (int i = 3; i < n; i += 2)
	{
		if (!v[i])
		{
			t.clear();
			for (int j = i; j <= n; j += i * 2)
			{
				if (!v[j])
					t.push_back(j);
			}
			if ((t.size() % 2 == 1) && (i * 2 <= n))
				t.push_back(i * 2);
			for (int j = 0; j + 1 < t.size(); j += 2)
			{
				res.push_back(make_pair(t[j], t[j+1]));
				v[t[j]] = true;
				v[t[j+1]] = true;
			}
		}
	}

	t.clear();
	for (int i = 2; i <= n; i += 2)
		if (!v[i])
			t.push_back(i);
	for (int i = 0; i + 1 < t.size(); i += 2)
		res.push_back(make_pair(t[i], t[i+1]));

	cout << res.size() << endl;
	for (int i = 0; i < res.size(); ++i)
		cout << res[i].first << " " << res[i].second << endl;

	return 0;
}
