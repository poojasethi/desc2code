#include <stdio.h>
#include <vector>
using namespace std;
vector<int>res1;
vector<int>res2;
const int maxn = 100001;
int col[maxn], l;
bool tub[maxn];
bool ok[maxn];
int main(){
	tub[0] = tub[1] = true;
	for(int i = 2; i < maxn; i ++)
	   if(1ll * i * i < maxn)
	   	  for(int j = i * i; j < maxn; j += i) tub[j] = true;
	int n,i,j,c;
	scanf("%i",&n);
	for(i = n; i > 1; i --){
		if(!tub[i]){
			l = 0;
			for(j = i; j <= n; j += i) if(!ok[j]) col[l ++] = j;
			if(l % 2 == 1){
				for(c = 0; c < l; c ++) if(col[c] % 2 == 0) break;
				col[c] = col[--l];
			}
			for(j = 1; j < l; j += 2){
				res1.push_back(col[j]);
				res2.push_back(col[j-1]);
				ok[col[j]] = ok[col[j-1]] = true;
			}
		}
	}
	printf("%i\n",res1.size());
	for(i = 0; i < res1.size(); i ++) printf("%i %i\n",res1[i],res2[i]);
	return 0;
}
