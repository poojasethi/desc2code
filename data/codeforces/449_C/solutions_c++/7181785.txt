#include <cstdio>
#include <cstring>

using namespace std;

const int MXN = 128*1024;
int used[MXN];
int prime[MXN];
int an;
int ans[MXN];

void solve(int n)
{
  an = 0;
  memset(used, 0, sizeof(used));
  memset(prime, 1, sizeof(prime));
  for (int i=2; i<1024; i++) {
    if (!prime[i])continue;
    for (int k=i*i; k<MXN; k+= i) {
      prime[k] = 0;
    }
  }
  for (int i=3; i<=n; i++) {
    if (!prime[i])continue;
    int cn=0;
    for (int k=i; k<=n; k+=i) {
      if (!used[k])cn++;
    }
    if (cn == 1)continue;
    for (int k=i; k<=n; k+=i) {
      if (k==2*i && (cn&1)) continue;
      if (used[k])continue;
      ans[an++] = k;
      used[k] = 1;
    }
  }
  //fprintf(stderr, "XX %d\n", an);
  for (int i=2; i<=n ; i+=2) if (!used[i])ans[an++] = i;
  int ca = an/2;
  printf("%d\n", ca);
  for (int i=0; i<ca; i++) printf("%d %d\n", ans[2*i], ans[2*i+1]);
}

int main()
{
  int n;
  while (scanf("%d", &n) == 1)
  {
    solve(n);
  }
}
