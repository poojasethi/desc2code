#include <iostream>
using namespace std;
const int maxn = 100000 + 10;

int index,v[maxn],ans[maxn];
int main()
{
    int N, i, j;
    cin >> N;
    for(i = 3; i <= N / 2; i++)
    {
        int k = 0;
        for(j = i; j <= N; j += i) if( !v[j] ) k++;
        if( k & 1  && k > 1) 
            for(j = i; j <= N; j += i )
            {
                if(j == i * 2) continue;
                if(!v[j]) ans[index++] = j,v[j] = 1;
            }
        else 
        {
            if(k == 1) continue; 
            for(j = i; j <= N; j += i )
            if(!v[j])
            {
                ans[index++] = j;
                v[j] = 1;
            }
        }
    }
    for(i = 2; i <= N; i += 2) if(!v[i]) ans[index++] = i;
    cout << index / 2 << endl;
    for(i = 0; i < index / 2; i++ )
    {
        cout << ans[2 * i] << " ";
        cout << ans[2 * i + 1] << endl;
    }
}
