#include<iostream>
#include<cstring>
#include<vector>
#include<queue>
#include<algorithm>
using namespace std;
int re[100005];
vector< int >p[100005];
vector< pair<int,int> >o;
int main()
{
		ios_base::sync_with_stdio(0);
		cin.tie(0);
		int n,i,j,k;
		cin>>n;
		for(i=2;i<=n;i++)
		{
				if(!re[i])
				{
						for(j=i;j<=n;j+=i)
						{
								re[j]=1;
								p[i].push_back(j);
						}
				}
		}
		re[1]=1;
		int sz,br;
		for(i=n;i>=2;i--)
		{
				sz=p[i].size();
				if(sz<2)continue;
				br=0;
				for(j=0;j<sz;j++)
						br+=re[p[i][j]];
				j=0;
				if(br&1)re[i*2]=0;
				for(;j<sz;j++)
				{
						if(!re[p[i][j]])continue;
						for(k=j+1;k<sz;k++)
						{
								if(!re[p[i][k]])continue;
								o.push_back(make_pair(p[i][j],p[i][k]));
								re[p[i][j]]=re[p[i][k]]=0;
								break;
						}
						j=k;
				}
				if(br&1)re[i*2]=1;
		}
		sz=o.size();
		cout<<sz<<'\n';
		for(i=0;i<sz;i++)
				cout<<o[i].first<<" "<<o[i].second<<'\n';
}
