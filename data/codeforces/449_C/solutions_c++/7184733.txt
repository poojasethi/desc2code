#include <iostream>
#include <vector>

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 1e5 + 10;
bool prime[MAXN], use[MAXN];
vector<pii> ans;
vector<int> cur;

int main() {
	int n;
	cin >> n;
	prime[1] = prime[0] = 1;
	for (ll i = 2; i < MAXN; i++)
		if (!prime[i])
			for (ll j = i * i; j < MAXN; j += i)
				prime[j] = true;
	for (int i = n; i >= 2; i--) {
		if (prime[i])
			continue;
		cur.clear();
		for (int j = i; j <= n; j += i)
			if (!use[j])
				cur.push_back(j);
		if (cur.size() == 1)
			continue;
		if (cur.size() % 2)
			cur.erase(cur.begin() + 1);
		for (int i = 1; i < cur.size(); i += 2)
			ans.push_back(pii(cur[i], cur[i - 1])), use[cur[i]] = use[cur[i - 1]] = true;
	}
	cout << ans.size() << endl;
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i].first << " " << ans[i].second << endl;
}
