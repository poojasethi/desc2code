#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>

using namespace std;
int n,flag[200000],ans[200000],ansn;
int main()
{
	scanf("%d",&n);
	memset(flag,0,sizeof(flag));
	for (int i=3;i<=n/2;i+=2)
	{
		int k=0;
		for (int j=1;j*i<=n;++j) if (flag[j*i]==0) k++;
		if (k%2==0)
		{
			for (int j=1;j*i<=n;++j)
			{
				if (flag[j*i]) continue;
				flag[j*i]=1;
				++ansn;
				ans[ansn]=j*i;
			}
		}
		else
		{
			for (int j=1;j*i<=n;++j)
			{
				if (j==2) continue;
				if (flag[j*i]) continue;
				flag[j*i]=1;
				++ansn;
				ans[ansn]=j*i;
			}
		}
	}
	for (int i=2;i<=n;i+=2)
	{
		if (flag[i]==0)
		{
			++ansn;
			ans[ansn]=i;
		}
	}
	printf("%d\n",ansn/2);
	for (int i=1;i<=ansn/2;++i) printf("%d %d\n",ans[2*i-1],ans[2*i]);
	return 0;
}
