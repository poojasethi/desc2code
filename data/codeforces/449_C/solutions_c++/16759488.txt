#include <bits/stdc++.h>
using namespace std;
bool u[100005],p[100005];
int n;
int main(){
	cin >> n;
	for(int i=2;i<=100000;i++){
		if(p[i]) continue;
		for(int j=i+i;j<=100000;j+=i){
			p[j] = 1;
		}
	}
	vector<int>a,b;
	for(int i=n;i>=2;i--){
		if(p[i]) continue;
		if(i*2 > n) continue;
		vector<int>vi; vi.push_back(i*2);
		for(int j=1;i*j<=n;j++){
			if(j!=2 && !u[i*j]){
				vi.push_back(i*j);
			}
		}
		int beg = 0;
		if(vi.size()%2 == 1) beg = 1;
		for(int i=beg;i<vi.size();i+=2){
			a.push_back(vi[i]);
			b.push_back(vi[i+1]);
			u[vi[i]] = 1; u[vi[i+1]] = 1;
		}
	}
	cout << a.size() << endl;
	for(int i=0;i<a.size();i++) cout << a[i] << " " << b[i] << endl;
}
