#include <bits/stdc++.h>
#define pb push_back
using namespace std;
const int maxn = 1e5+5;
int n, p[maxn], ans[maxn][2], tot;
void f(int i){
	if(p[i]) return;
	set<int> s;
	// cout << i << ": ";
	for(int j=1;j<=n/i;j++){
		if(p[i*j]) continue;
		s.insert(i*j); 
		// cout << i*j << " ";
	}
	// cout << endl;
	if(s.size() > 1 && s.size()%2 == 1) s.erase(2*i);
	vector<int> a(s.begin(),s.end());
	for(int j=1;j<a.size();j+=2){
		ans[tot][0] = a[j-1]; ans[tot][1] = a[j];
		tot++; p[a[j-1]] = p[a[j]] = 1;
	}
}
int main(){
	ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
	cin >> n;
	for(int i=3;i<=n;i++){
		f(i);
	}
	f(2);
	cout << tot << "\n";
	for(int i=0;i<tot;i++) cout << ans[i][0] << " " << ans[i][1] << "\n";
	return 0;
}
