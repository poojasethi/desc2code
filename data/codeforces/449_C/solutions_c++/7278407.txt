#include<cstdio>
#define N 200000
int n,i,j,k,ans,tot;
int f[N][2],a[N],c[N],flag[N];
int main()
{
	scanf("%d",&n);
	for (i=2;i<=n;i++)
	if (c[i]==0)
	for (j=i;j<=n/i;j++)
	c[i*j]=1;
	
	for (i=2;i<=n;i++)
	if (c[i]==0)
	{
		tot++;
		a[tot]=i;
	}
	
	for (j=1;j<=tot;j++)
	if ((a[j]>2)&&(a[j]<=n/2))
	{
	for (k=3;k<=n/a[j];k++)
	if (flag[a[j]*k]==0)
	{
		flag[a[j]*k]=1;
		if (f[ans][0]==0)
		{
			f[ans][0]=a[j]*k;
		}
		else
		{
			f[ans][1]=a[j]*k;
			ans++;
		}
	}
	
		if (f[ans][0]==0)
		{
			flag[a[j]]=1;flag[a[j]*2]=1;
			f[ans][0]=a[j];
			f[ans][1]=a[j]*2;
			ans++;
		}
		else
		{
			flag[a[j]]=1;
			f[ans][1]=a[j];
			ans++;
		}
	}
	for (i=1;i<=n;i++)
	if ((flag[i]==0)&&(i%2==0))
	{
		if (f[ans][0]==0)
		f[ans][0]=i;
		else
		{
			f[ans][1]=i;
			ans++;
		}
	}
	printf("%d\n",ans);
	for (i=0;i<ans;i++)
	printf("%d %d\n",f[i][0],f[i][1]);
}
