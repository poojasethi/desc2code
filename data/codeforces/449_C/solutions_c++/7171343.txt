#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <stdio.h>
using namespace std;
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t=0,d[100000],f[100000];
  bool a[100001],s[100001];
  cin>>q;
  for(w=2;w<=q;w++)
   a[w]=s[w]=1;
  for(w=2;w<=q;w++)
   if(a[w])
    for(e=w<<1;e<=q;e+=w)
     a[e]=0;
  for(w=q;w>=2;w--)
   if(a[w])
    if(w*2<=q){
     r=w;
     for(e=q-(q%w);e>=2;e-=w)
      if(s[e])
       if(r>0){
        d[t]=r;
        f[t]=e;
        t++;
        s[r]=s[e]=0;
        r=0;}
       else
        r=e;}
  printf("%d\n",t);
  for(w=0;w<t;w++)
   printf("%d %d\n",d[w],f[w]);
  return 0;}
