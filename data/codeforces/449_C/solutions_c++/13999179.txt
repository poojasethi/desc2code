#include <stdio.h>
#define N 100009
typedef long long ll;
int s[N],c[N],t[N],v[N],vn;
void prepare() {
	s[0]=s[1]=1;
	for (ll i=2;i<N;i++) if (!s[i])
		for (ll j=i*i;j<N;j+=i) s[j]=1;
	for (int i=1;i<N;i++) {
		c[i]=c[i-1]+(!s[i]);
	}
}
main() {
	prepare();
	int n;
	scanf("%d",&n);
	int ans=(n-1-(c[n]-c[n/2]))/2;
	printf("%d\n",ans);
	for (int i=n/2;i>=2;i--) if (!s[i]) {
		vn=0;
		for (int j=i;j<=n;j+=i) if (!t[j]) v[vn++]=j;
		for (int j=vn-1;j>0;j-=2) {
			int vv=v[j-1];
			if (j==2) vv=v[0];
			printf("%d %d\n",v[j],vv);
			t[v[j]]=1;t[vv]=1;
		}
	}
}
