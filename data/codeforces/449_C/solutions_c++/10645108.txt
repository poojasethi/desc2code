#include <bits/stdc++.h>
using namespace std;

const int N=1e5+10;
int vis[N], a[N];
vector<pair<int, int> > ans;

int main() {
    int n;
    scanf("%d", &n);
    for(int i=4; i<N; i+=2) a[i]=1;
    for(int i=3; i<N; i+=2) if(!a[i]) for(int j=2*i; j<N; j+=i) a[j]=1;
    for(int i=n/2; i>=2; i--) {
        if(a[i]) continue;
        vector<int> aux;
        for(int j=i; j<=n; j+=i) {
            if(vis[j]) continue;
            aux.push_back(j);
        }
        if((int)aux.size()%2==1) aux.erase(aux.begin()+1);
        for(int i=0; i<aux.size(); i+=2) {
            ans.push_back(make_pair(aux[i], aux[i+1]));
            vis[aux[i]]=vis[aux[i+1]]=1;
        }
    }
    printf("%d\n", (int)ans.size());
    for(int i=0; i<ans.size(); i++) printf("%d %d\n", ans[i].first, ans[i].second);
    return 0;
}
