#include <stdio.h>

int main() {
	int n, m, a[1000], b[1000], i, c, j;

	scanf("%d%d%d", &n, &m, &c);

	for(i = 0; i<n; i++) scanf("%d", &a[i]);
	for(i = 0; i<m; i++) scanf("%d", &b[i]);

	for(i = 0; i<=n-m; i++) {
		for(j = 0; j<m; j++) {
			a[i+j] += b[j];
			a[i+j] %= c;
		}
	}

	for(i = 0; i<n; i++) {
		printf("%d ", a[i]);
	}


	return 0;
}
