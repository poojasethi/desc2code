#include<iostream>


using namespace std;
const int maxN=1e3+100;
int a[maxN],b[maxN];
int main()
{
  int m,n,c;

  cin>>n>>m>>c;
  for(int i=1;i<=n;i++)
    cin>>a[i];
  for(int i=1;i<=m;i++)
    cin>>b[i];

  for(int i=1;i<=n-m+1;i++)
    for(int j=1;j<=m;j++){
      a[i+j-1]+=b[j];
      a[i+j-1]%=c;
    }

  for(int i=1;i<=n;i++)
    cout<<a[i]<<" ";

  return 0;
}
