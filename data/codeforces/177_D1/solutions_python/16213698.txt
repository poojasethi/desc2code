#include<bits/stdc++.h>

using namespace std;

#define tm (tl + tr >> 1)
#define N 100005

int A[N],P[N],n,m,md,i;

int main(){

	cin >> n >> m >> md;

	for(i=1;i<=n;i++)
		cin >> A[i];

	for(i=1;i<=m;i++){
	    cin >> P[i];
		P[i] = ( P[i] + P[i-1] ) % md;
	}

	for(i=1;i<=n;i++)
		cout << ( ( A[i] + P[min(i , m)] - P[max( min(i , m) - min( n-i+1 , n-m+1 ) , 0)] ) % md + md ) % md << " ";

	return 0;
}
