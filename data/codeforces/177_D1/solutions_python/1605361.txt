#include <iostream>
#include <cstdio>

#define N 100050

int a[N];
int b[N];

int main()
{
	int n,m,c;
	int i,j,k;
	scanf("%d%d%d",&n,&m,&c);
	for(i=0;i<n;i++) scanf("%d",&a[i]);
	for(j=0;j<m;j++) scanf("%d",&b[j]);
	for(k=0;k<n-m+1;k++) {
		for(i=0;i<m;i++) 
			a[i+k] = (a[i+k]+b[i])%c;
	}
	for(i=0;i<n;i++) printf("%d ",a[i]);
	printf("\n");

}
