#include <stdio.h>

int main()
{
	int i,j,n,m,c,a[100001]={0},b[100001]={0},st=0;
	scanf("%d%d%d",&n,&m,&c);
	st=n-m+1;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(i=1;i<=m;i++)
	{
		scanf("%d",&b[i]);
	}
	for(i=1;i<=st;i++)
	{
		for(j=i;j<=m+i-1;j++)
		{
			a[j]+=b[j-i+1];
		}
	}

	for(i=1;i<=n;i++)
	{
		a[i]%=c;
		printf("%d ",a[i]);
	}

	return 0;
}
