#include <iostream>
#include <cstdio>
#include <vector>

using namespace std;

int n,m,c;
vector<int> a,b;

int main(){
	scanf("%d%d%d",&n,&m,&c);
	a.resize(n);
	b.resize(m);
	for (int i=0; i<n; i++)
		scanf("%d",&a[i]);
	for (int i=0; i<m; i++)
		scanf("%d",&b[i]);
	for (int i=0; i<=n-m; i++)
		for (int j=0; j<m; j++)
			a[i+j]=(a[i+j]+b[j])%c;
	for (int i=0; i<n; i++)
		printf("%d ",a[i]);
	return 0;
}
