#include<cstdio>
int main()
{
	int n,m,c;
	scanf("%d %d %d",&n,&m,&c);
	int *a=new int[n];
	int *b=new int[m];
	for(int i=0;i<n;i++)scanf("%d",&a[i]);
	for(int i=0;i<m;i++)scanf("%d",&b[i]);
	int s=0;
	for(int i=0;i<n;i++)
	{
		if(i<m){s+=b[i];}
		if(i+m>n){s-=b[i-n+m-1];}
		a[i]+=s;a[i]%=c;
		
	}
	printf("%d",a[0]);
	for(int i=1;i<n;i++)	
		printf(" %d",a[i]);
}
