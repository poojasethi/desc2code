# include <stdio.h>

int a,b,c,q,x;
int a1[100009];
int b1[100009];

int main()
{
	scanf("%d %d %d",&a,&b,&c);
	
	for(int i=0;i<a;i++)
	scanf("%d",&a1[i]);
	
	for(int i=0;i<b;i++)
	{
		scanf("%d",&b1[i]);
		
		if(i!=0)
		b1[i]=b1[i-1]+b1[i];
	}
	
	for(int i=0;i<a;i++)
	{
		x=i;
		
		if(x>=b)
		x=b-1;
		
		q=i-(a-b);
		
		if(i<a-b)
		q=0;
		
		printf("%d\n",(a1[i]+(b1[x]-b1[q-1]))%c);
	}
}
