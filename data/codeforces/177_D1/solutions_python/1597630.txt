#include <iostream>
using namespace std;
int main()
{
	long m, n, c;
	cin >> n >> m >> c;
	long a[100001];
	long b[100001];
	for(int i=0;i<n;++i)
		cin >> a[i];
	for(int i=0;i<m;++i)
		cin >> b[i];
	int k=n-m+1;
	int j=0;
	while(k--)
	{
		for(int i=j, f=0;i<m+j;++i, ++f)
		{
			a[i]=(a[i]+b[f])%c;
		}
		j++;
	}
	for(int i=0;i<n;++i)
		cout << a[i] << ' ';
	
	
return 0;
}
