#include<cstdio>
#include<queue>
int n,m,c,a[100001],b[100001],x,y,t,i;
std::queue<int> s;
int main(){
	scanf("%d%d%d",&n,&m,&c);
	for(i=1;i<=n;++i)scanf("%d",a+i);
	for(i=1;i<=m;++i)scanf("%d",b+i);
	for(i=1;i<m;++i)(b[i+1]+=b[i])%=c;
	for(i=1;i<=n;++i){
		if(i+m<=n+1)s.push(i);
		if(s.front()+m<=i)s.pop();
		x=(i-s.back())%m;y=x+s.size();
		t=(y>m?b[m]+b[y-m]-b[x]:b[y]-b[x]);
		printf("%d ",(a[i]+t+c)%c);
	}
	return 0;
}
