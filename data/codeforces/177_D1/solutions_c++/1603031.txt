#include <iostream>
#include <cstdio>
using namespace std;
int a[10000],b[10000];
int main(){
	int n , m ,c;
	cin >> n >> m >> c;;
	for (int i=1;i<=n;i++){
		cin >> a[i];
	}
	for (int j=1;j<=m;j++){
		cin >> b[j];
	}
	for (int i=0;i<=n-m;i++){
		for (int j=1;j<=m;j++)
			a[i+j] += b[j];
	}
	for (int i=1;i<=n;i++)
		cout <<a[i]%c << ' ';
	return 0;
}
