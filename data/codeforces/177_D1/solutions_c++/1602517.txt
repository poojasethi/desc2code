#include <iostream>
using namespace std;
const int MAXN=1000+10;
int a[MAXN],b[MAXN],n,m,c;
void add(int x)
{
	for(int i=x;i<x+m;i++)
		a[i]+=b[i-x];
}
int main()
{
	cin >> n >> m >> c;
	for(int i=0;i<n;i++)
		cin >> a[i];
	for(int i=0;i<m;i++)
		cin >> b[i];
	for(int i=0;i<=n-m;i++)
		add(i);
	for(int i=0;i<n;i++)
		cout << a[i]%c << " ";
	return 0;
}
