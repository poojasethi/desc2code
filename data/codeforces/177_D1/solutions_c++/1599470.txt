#include <iostream>
using namespace std;

int a[100100]={0},b[100100]={0},n,m,c;

int main()
{
   int i,j,k;
   
   cin>>n>>m>>c;
   
   for (i=0;i<n;i++)
      cin>>a[i];
   
   for (i=0;i<m;i++)
      cin>>b[i];
   
   
   for (i=0;i<n-m+1;i++)
      for (j=i;j-i<m && j<n;j++)
         a[j]=(a[j]+b[j-i])%c;
   
   cout<<a[0];
   for (i=1;i<n;i++)
      cout<<" "<<a[i];
   cout<<endl;
   
   return 0;
}
