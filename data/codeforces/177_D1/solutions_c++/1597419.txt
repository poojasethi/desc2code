#include <cstdio>
#include <cstring>
#include <iostream>
int main()
{
	long n,m,c;
	std::cin >> n >> m >> c;
	int* text = new int[n];
	int* k = new int[m];
	for (int i=0; i<n; i++)
		std::cin >> text[i];
	for (int i=0; i<m; i++)
		std::cin >> k[i];
	for (int i=0; i<n-m+1; i++)
	{
		for (int j=i; j<i+m; j++)
			text[j] = (text[j]+k[j-i])%c;
	}
	for (int i=0; i<n; i++)
		std::cout << text[i] << ' ';
	return 0;
}
