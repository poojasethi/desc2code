#include <stdio.h>
main() {
  int n,m,c;
  short a[100100],b[100100];
  scanf("%d%d%d",&n,&m,&c);
  for (int i=1; i<=n; i++)
    scanf("%d",&a[i]);
  for (int i=1; i<=m; i++)
    scanf("%d",&b[i]);
  for (int i=1,e=n-m+1; i<=e; i++)
    for (int j=1; j<=m; j++)
      a[i+j-1]=(a[i+j-1]+b[j])%c;
  for (int i=1; i<=n; i++)
    printf("%d ",a[i]);
  return 0;
  }
