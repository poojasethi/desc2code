#include<iostream>
using namespace std;
int main()
{
	int n,m,c,i,j,k;
	int a[100000],b[100000];
	while(cin>>n>>m>>c)
	{
		for(i=0;i<n;i++)
			cin>>a[i];
		for(i=0;i<m;i++)
			cin>>b[i];
		for(i=0;i<n-m+1;i++)
			for(j=0,k=i;j<m;j++,k++)
				a[k]=a[k]+b[j];
			for(i=0;i<n;i++)
				a[i]=a[i]%c;
			for(i=0;i<n;i++){
				if(i)
					cout<<" "<<a[i];
				else 
					cout<<a[i];
			}
			cout<<endl;
	}
	return 0;
}
