#include<cstdio>
using namespace std;
#define MaxN 1005
int a[MaxN], b[MaxN], c, i, j, n, m;
int main()
{
    scanf("%d %d %d", &n, &m, &c);
    for(i=1; i<=n; i++)
        scanf("%d", &a[i]);
    for(i=1; i<=m; i++)
        scanf("%d", &b[i]);
    for(i=1; i<=n-m+1; i++)
    {
        for(j=i; j<=i+m-1; j++)
            a[j]=(a[j]+b[j-i+1])%c;
    }
    for(i=1; i<=n; i++)
        printf("%d ", a[i]);
}
