#include<iostream>
using namespace std;
 int main()
 {
 	int n, m ,c;
 	cin>>n>>m>>c;
 	int a[n], b[m],i,j,nos = n-m +1;
 	for(i=0;i<n;i++)
 	cin>>a[i];
 	for(i=0;i<m;i++)
 	cin>>b[i];
 	for(i=0;i<nos;i++)
 	{
	 	for(j=0;j<m;j++)
	 	a[i+j] = (a[i+j]+ b[j])%c;
	 }
 	for(i=0;i<n;i++)
 	cout<<a[i]<< " ";
 	cout<< endl;
 	return 0;
 }
