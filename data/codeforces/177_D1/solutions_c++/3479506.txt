#include <iostream>
using namespace std;
#define MX 100010
int s[MX], key[MX], n,m,c, i,j;

int main()
{   cin>>n>>m>>c;
    for (i=1; i<=n; i++) cin>>s[i];
    for (i=1; i<=m; i++) cin>>key[i];

    for (i=1; i<=n-m+1; i++)
        for (j=1; j<=m; j++ )
            s[j+i-1]=(s[j+i-1]+key[j])%c;

    for (i=1; i<=n; i++ ) cout<<s[i]<<" ";

    return 0;
}
