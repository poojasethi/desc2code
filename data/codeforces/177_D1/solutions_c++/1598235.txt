#include<stdio.h>

int main()
{
    int n,m,c;
    int x1[11111],x2[11111];
    scanf("%d %d %d",&n,&m,&c);
    for(int i=0;i<n;i++)scanf("%d",&x1[i]);
    for(int i=0;i<m;i++)scanf("%d",&x2[i]);
    for(int i=0;i<n-m+1;i++)
    {
        for(int j=i;j<i+m;j++)
        {
            x1[j]=(x1[j]+x2[j-i])%c;
        }
    }
    for(int i=0;i<n-1;i++)printf("%d ",x1[i]);
    printf("%d\n",x1[n-1]);
}
