#include<iostream>
using namespace std;

int m,n,c;
int main()
{
	cin>>n>>m>>c;
	int a[n],b[m];
	for(int i=0;i<n;i++) cin>>a[i];
	for(int i=0;i<m;i++) cin>>b[i];
	
	for(int i=0;i<n-m+1;i++)
	{
		int k=0;
		for(int j=i;j<i+m;j++)
		{
			a[j]=(a[j]+b[k])%c;
			k++;
		}
	}
	for(int i=0;i<n;i++)
	{
		if(i!=(n-1))cout<<a[i]<<" ";
		else cout<<a[i]<<endl;
	}
	return 0;
}
