#include<iostream>
using namespace std;

int n,m,c;
int a[1010];
int b[1010];

int main()
{
    cin>>n>>m>>c;
    for(int i=1;i<=n;i++)
       cin>>a[i];
    for(int i=1;i<=m;i++)
       cin>>b[i];
    for(int i=1;i+m-1<=n;i++)
       for(int j=1;j<=m;j++)
       {
          a[i+j-1]+=b[j];
          a[i+j-1]%=c;
       }
    for(int i=1;i<n;i++)
       cout<<a[i]<<" ";
    cout<<a[n]<<endl;
    return 0;
}
