#include <iostream>
#include <vector>

using namespace std;

int main()
{
	int n, m, c;
	cin >> n >> m >> c;
	vector<int> a(n);
	vector<int> b(n);
	
	for (int i = 0; i < n; i++) {
		cin >> a[i];
	}
	for (int i = 0; i < m; i++) {
		cin >> b[i];
	}

	for (int i = 0; i < n - m + 1; i++) {
		for (int j = 0; j < m; j++) {
			a[i + j] += b[j];
		}
	}

	for (int i = 0; i < n; i++) {
		cout << a[i] % c << " ";
	}
	
	return 0;
}
