#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
int a[100009],b[100009];
int main()
{
	int n,i,j,m,c;
	cin>>n>>m>>c;	
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<m;i++)
		cin>>b[i];
	for(i=0;i<n-m+1;i++)
		for(j=0;j<m;j++)
			a[i+j]=(a[i+j]+b[j])%c;
	for(i=0;i<n;i++)
		cout<<a[i]<<" ";
	return 0;
}
