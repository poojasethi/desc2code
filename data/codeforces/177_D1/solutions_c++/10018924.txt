#include <iostream>
#define loop(i,s,e) for(int  i = s; i < e;i++)
#define lop(i,e) for(int  i = 0; i < e;i++)
using namespace std;

int main(){
   int n,m,c;
   int a[100000],b[100000];
   cin >>n>>m>>c;
   lop(i,n)cin >> a[i];
   lop(i,m) cin>>b[i];
int index =  0;
lop(i,n-m+1){
int k = index;
lop(j,m){
a[k] = (a[k]+b[j])%c;
k++;
}
index++;
}
lop(i,n) cout<<a[i]<<" ";
    return 0;
}
