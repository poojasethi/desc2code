#include <cstdio>
int a[100001],b[100001];
int main(){
    int n,m,c,k=0;
    scanf("%d%d%d",&n,&m,&c);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<m;i++) scanf("%d",&b[i]);
    for(int i=0;i<n;i++){
        if(i<m) k+=b[i];
        if(i+m+1>=n) k-=b[i+m-n-1];
        printf("%d ",(a[i]+k)%c);
    }
   return 0;
}
