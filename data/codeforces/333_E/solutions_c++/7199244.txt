#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<bitset>
using namespace std;
const int lim=1000000000;
struct pt{
	int x,y;
}a[3011];
struct data{
	int a,b,d;
}q[3011*3011/2+2];
bitset<3011> w[3011];
int ans,i,j,k,n,m,tot;
int sqr(int x){
	return x*x;
}
bool cmp(data a,data b){
	return a.d>b.d;
}
int main(){
	scanf("%d",&n);tot=0;
	for (i=1;i<=n;i++) scanf("%d%d",&a[i].x,&a[i].y);
	for (i=1;i<n;i++)
	 for (j=i+1;j<=n;j++) k=sqr(a[i].x-a[j].x)+sqr(a[i].y-a[j].y),q[++tot].a=i,q[tot].b=j,q[tot].d=k;
	sort(q+1,q+1+tot,cmp);
	for (i=1;i<=tot;i++){
		if ((w[q[i].a]&w[q[i].b]).any()){
			ans=q[i].d;break;
		}
		w[q[i].a][q[i].b]=w[q[i].b][q[i].a]=1;
	}
	printf("%.9lf\n",sqrt(ans)/2);
	return 0;
}
