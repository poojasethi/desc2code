#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<bitset>
#include<cmath>
using namespace std;
bitset<3000>b[3000];
struct S{double x,y;}a[3000];
struct Node{int x,y;double z;}f[5000000];
double dist(int i,int j)
{
    return (a[i].x-a[j].x)*(a[i].x-a[j].x)+(a[i].y-a[j].y)*(a[i].y-a[j].y);
}
bool operator<(Node a,Node b)
{
    return a.z<b.z;
}
int main()
{
    int n,m=0;
    cin>>n;
    for(int i=0;i<n;++i)cin>>a[i].x>>a[i].y;
    for(int i=0;i<n;++i)
        for(int j=i+1;j<n;++j)
        f[++m].x=i,f[m].y=j,f[m].z=sqrt(dist(i,j))/2;
    sort(f+1,f+m+1);
    for(int i=m;;--i)
    {
        if((b[f[i].x]&b[f[i].y])!=0)
        {
            printf("%.10lf\n",f[i].z);
            break;
        }
        b[f[i].x][f[i].y]=1;
        b[f[i].y][f[i].x]=1;
    }
}
