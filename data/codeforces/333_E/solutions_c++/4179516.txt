#include <cstdio>
#include <algorithm>
#include <vector>
#include <bitset>
#include <cmath>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
#define pb push_back
#define mp make_pair
int n,x[4040],y[4040];
vector<pair<int,pair<int,int> > > vec;
bitset<3000> bit[4040];
int main() {
	scanf("%d",&n);
	rep(i,0,n) scanf("%d%d",x+i,y+i);
	rep(i,0,n) rep(j,i+1,n) vec.pb(mp((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]),mp(i,j)));
	sort(vec.rbegin(),vec.rend());
	rep(i,0,vec.size()) {
		int u=vec[i].second.first,v=vec[i].second.second;
		if ((bit[u]&bit[v]).any()) { printf("%.9f\n",sqrt(vec[i].first)/2);return 0;}
		bit[u][v]=bit[v][u]=1;
	}
}
