#include<cstdio>
#include<iostream>
#include<cstring>
#include<cmath>
#include<cstdlib>
#include<bitset>
#define For(i,n) for (int i=0; i<n; i++)
using namespace std;

struct dot
{
	double len;
	int x,y;
} d[9010000];

int n,m,x[3010],y[3010];
bitset<3010>b[3010];

void Qsort(int l,int r)
{
	double k=d[(l+r)/2].len; int i=l,j=r;
	while (i<=j)
	{
		while (d[i].len>k) i++;
		while (d[j].len<k) j--;
		if (i<=j)
		{
			dot tt=d[i]; d[i]=d[j]; d[j]=tt;
			i++; j--;
		}
	}
	if (l<j) Qsort(l,j);
	if (i<r) Qsort(i,r);
}

int main()
{
	scanf("%d",&n);
	For (i,n)
	{
		scanf("%d %d",&x[i],&y[i]);
		For (j,i)
		{
			d[++m].len=(x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
			d[m].len=sqrt(d[m].len); d[m].len/=2; d[m].x=i; d[m].y=j;
		}
	} Qsort(1,m);
	For (k,m)
	{
		int i=d[k+1].x,j=d[k+1].y;
		if ((b[i]&b[j])!=0)
		{
			printf("%.20lf",d[k+1].len);
			return 0;
		} b[i][j]=1; b[j][i]=1;
	}
}
