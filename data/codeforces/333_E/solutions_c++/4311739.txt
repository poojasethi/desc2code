#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#include <bitset>
#define db double
#define N 3010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
    int x,y,a;
} a[N*N];
int i,j,k,n,m,s;
int x[N],y[N];
bitset<N> h[N],t;
inline int dis(int i,int j) {
    return (x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
}
inline bool cc1(const ww &a,const ww &b) {
    return a.a>b.a;
}
int main() {
    scanf("%d",&n);
    For(i,1,n) scanf("%d%d",&x[i],&y[i]);
    For(i,1,n)For(j,i+1,n) a[++s].x=i,a[s].y=j,a[s].a=dis(i,j);
    sort(a+1,a+s+1,cc1);
    //printf("%d\n",clock()); return 0;
    For(i,1,s) {
        int A=a[i].x,B=a[i].y;
        t=h[A]&h[B];
        if (t.any()) {
            printf("%.10lf\n",sqrt(0.+a[i].a)/2); return 0;
        }
        h[A][B]=h[B][A]=1;
    }
    return 0;
}
