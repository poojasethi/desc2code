#include<bits/stdc++.h>
#define mp make_pair
#define ll int
using namespace std;
const ll N=3001;
pair<double, pair<ll,ll> >d[N*N];
ll n,q,x[N],y[N];
double dis[N][N];
bitset<N> bt[N];
bool ck(ll a,ll b)
{
	return (bt[a]&bt[b]).any();
}
int main()
{
	cin>>n;
	for(ll i=0;i<n;i++)
        cin>>x[i]>>y[i];
	for(ll i=0;i<n;i++)
		for(ll j=i+1;j<n;j++)
        {
			dis[i][j]=dis[j][i]=sqrt(pow(x[i]-x[j],2)+pow(y[i]-y[j],2))/2;
			d[q]=mp(dis[i][j],mp(i,j));
            q++;
		}
	sort(d,d+q);
	reverse(d,d+q);
	for(ll i=0;i<q;i++)
    {
		ll a=d[i].second.first,b=d[i].second.second;
		if(ck(a,b))
		{
			cout<<setprecision(10)<<fixed<<d[i].first<<endl;
			return 0;
		}
		bt[a][b]=bt[b][a]=1;
	}
	return 0;
}
