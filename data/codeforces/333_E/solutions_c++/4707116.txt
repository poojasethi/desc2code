#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
#define PI acos(-1.0)
int n,m;
double EPS = 1e-12;
double x[3001][2],ans,dx,dy,hi,lo,df;
pair <double,double> P[3001];
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%lf %lf",&x[i][0],&x[i][1]);
	for(int i=0;i<n;i++){
		m = 0;
		for(int j=0;j<n;j++)
			if(i != j){
				dx = x[j][0] - x[i][0], dy = x[j][1] - x[i][1];
				P[m++] = make_pair( sqrt(dx*dx + dy*dy) , atan2(dy,dx) );
			}
		sort(P,P+m);
		hi = lo = 0.0;
		for(int j=m-2;j>=0;j--){
			df = P[j].second - P[m-1].second;
			if(df > PI) df -= 2*PI;
			if(df < -PI) df += 2*PI;
			hi = max(hi,df);
			lo = min(lo,df);
			if(hi - lo > PI/3.0-EPS){
				ans = max(ans,P[j].first);
				break;
			}
		}
	}
	printf("%.11lf\n",ans/2);
	return 0;
}
