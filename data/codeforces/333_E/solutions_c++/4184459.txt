#include<iostream>
#include<iomanip>
#include<cstring>
#include<algorithm>
#include<cmath>
#include<bitset>
using namespace std;
bitset<3000> b[3000];
struct{double x,y;}a[3000];
struct rec{int x,y;double z;}f[5000010];
int n,m,i,j;

double dist(int i,int j)
{
	return (a[i].x-a[j].x)*(a[i].x-a[j].x)+(a[i].y-a[j].y)*(a[i].y-a[j].y);
}

bool operator <(rec a,rec b)
{
	return a.z<b.z;
}

int main()
{
	cin>>n;
	for(i=0;i<n;i++) cin>>a[i].x>>a[i].y;
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++) f[++m].x=i,f[m].y=j,f[m].z=sqrt(dist(i,j))/2;
	sort(f+1,f+m+1);
	for(i=m;i;i--)
	{
		if((b[f[i].x]&b[f[i].y])!=0) break;
		b[f[i].x][f[i].y]=1;
		b[f[i].y][f[i].x]=1;
	}
	cout<<setprecision(10)<<fixed<<f[i].z<<endl;
}
