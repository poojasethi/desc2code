#include <cstdio>
#include <cmath>
#include <utility>
#include <algorithm>
#define X first
#define Y second
#define DO double
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

pair<DO,DO> a[3030];
int n;
DO ans;

inline DO Dist(int x,int y) {
    return ((a[x].X-a[y].X)*(a[x].X-a[y].X)+(a[x].Y-a[y].Y)*(a[x].Y-a[y].Y));
    return 0;
}

int main() {
    scanf("%d",&n);
    REP(i,n)
        scanf("%lf%lf",&a[i].X,&a[i].Y);
    sort(a+1,a+n+1);
    REP(i,1000000) {
        int x=rand()%n+1;
        int y=rand()%n+1;
        int z=rand()%n+1;
        ans=max(ans,min(Dist(x,y),min(Dist(x,z),Dist(y,z))));
    }
    for (int x=n;x;--x)
        REP(y,x-1)
            if (Dist(x,y)>ans) REP(z,y/7)
                ans=max(ans,min(Dist(x,y),min(Dist(x,z),Dist(y,z))));
    printf("%.20lf\n",sqrt(ans)/2);
    return 0;
}
