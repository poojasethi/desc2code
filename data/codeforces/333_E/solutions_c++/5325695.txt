#include <bits/stdc++.h>

using namespace std;

int main ()
{
	vector < pair < double, pair < int, int > > > pts;
	int X[3000], Y[3000], n;
	cin >> n;
	
	for (int i = 0; i < n; i++)
		cin >> X[i] >> Y[i];

	for (int i = 0; i < n; i++)
		for (int j = i + 1; j < n; j++)
		{
			double d = sqrt ((X[i] - X[j]) * (X[i] - X[j]) + (Y[i] - Y[j]) * (Y[i] - Y[j]));
			pts.push_back (make_pair (d, make_pair (i, j)));
		}

	sort (pts.begin (), pts.end ());
	reverse (pts.begin (), pts.end ());
	
	bitset < 3000 > visited[3000];

	int sz = pts.size ();
	double ret;
	for (int i = 0; i < sz; i++)
	{
		int x = pts[i].second.first, y = pts[i].second.second;
		ret = pts[i].first;
		if ((visited[x] & visited[y]).any ())
			break;
		visited[x][y] = visited[y][x] = true;
	}

	printf ("%.9lf\n", ret / 2.0);

	return 0;
}
