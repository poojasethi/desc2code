#include <cstdio>
#include <algorithm>
#include <cmath>
#include <bitset>

using namespace std;

struct lines {
    long long l;
    int a, b;
    bool operator< (const lines & y) const {
        return l > y.l;
    }
}l[5000000];
bitset<3010> s[3010];
int x[3010], y[3010];
int tot;

int main()
{
    int n;
    scanf("%d", &n);
    for (int i = 1; i <= n; ++i) scanf("%d%d", &x[i], &y[i]);
    for (int i = 1; i < n; ++i)
        for (int j = i+1; j <= n; ++j)
            l[tot].a = i, l[tot].b= j, l[tot++].l = (x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
    sort(l, l+tot);
    for (int i = 0; i < tot; ++i) {
        if ((s[l[i].a] & s[l[i].b]) != 0) {
            printf("%.20lf\n", (double)sqrt(l[i].l)/2.0);
            break;
        }
        s[l[i].a][l[i].b] = true;
        s[l[i].b][l[i].a] = true;
    }
    return 0;
}
