#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
typedef double db;
const int N=3005;
const db pi=acos(-1);
int n,m;
struct vec{
	int x,y;
	void read(){
		scanf("%d%d",&x,&y);
	}
} a[N];
typedef pair<int,db> data;
data b[N];

inline int calc(){
	db l=0,r=0;
	for(int i=m-2;i>=0;i--){
		db theta=b[i].second-b[m-1].second;
		while(theta<-pi) theta+=2*pi;
		while(theta>pi) theta-=2*pi;
		l=min(l,theta);
		r=max(r,theta);
		if(r-l>pi/3) return b[i].first;
	}
	return 0;
}

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++) a[i].read();
	int ans=0;
	for(int i=1;i<=n;i++){
		m=0;
		for(int j=1;j<=n;j++) if(j!=i){
			db dx=a[j].x-a[i].x,dy=a[j].y-a[i].y;
			b[m++]=data(dx*dx+dy*dy,atan2(dy,dx));
		}
		sort(b,b+m);
		ans=max(ans,calc());
	}
	printf("%.10lf\n",sqrt(ans)/2);
}
