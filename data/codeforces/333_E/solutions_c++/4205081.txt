#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
#define ran 3333
int n;
double x[3333],y[3333];
int la;
pair<double,pair<int,int> > a[5000000];
unsigned int nex[3333][95];
int T = 0;
double check(){   
    for(int i=la-1; i>=0; i--){
        int A=a[i].second.first,B=a[i].second.second;
        for(int j=0; j<T; j++)
            if(nex[A][j]&nex[B][j])
                return sqrt(a[i].first);
        nex[A][B>>5]|=1u<<(B&31);
        nex[B][A>>5]|=1u<<(A&31);
    }
    return .0;
}
double dist(double x,double y){
    return x*x+y*y;
}
int main(){
    scanf("%d",&n);
    T = (n-1)/32+1;
    for(int i=0; i<n; i++)
        scanf("%lf%lf",&x[i],&y[i]);
    for(int i=0; i<n; i++)
        for(int j=i+1; j<n; j++)
            a[la++]=make_pair(dist(x[i]-x[j],y[i]-y[j]),make_pair(i,j));
    sort(a,a+la);
    printf("%.8lf\n",check()/2.);
    return 0;
}
