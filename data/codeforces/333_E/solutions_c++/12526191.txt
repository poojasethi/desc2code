#include <bits/stdc++.h>
using namespace std;

int n;
const int max_n = 3010;

int x[max_n],y[max_n];
int dis[max_n][max_n];
vector<pair<double ,pair<int ,int > > >vec;
bitset<max_n>bit[max_n],c;

int main()
{
	cin >> n;
	for(int i=0;i<n;i++) cin >> x[i] >> y[i];
	for(int i=0;i<n;i++)
		for(int j=0;j<i;j++)
		{
			dis[i][j] = dis[j][i] = (x[i] - x[j])*(x[i] - x[j]) + (y[i] - y[j])*(y[i] - y[j]);
			vec.push_back(make_pair(dis[i][j],make_pair(i,j)));
		}
	sort(vec.begin(),vec.end());
	reverse(vec.begin(),vec.end());
	cout << setprecision(6) << fixed;
	for(int i=0;i<vec.size();i++)
	{
		int u  = vec[i].second.first;
		int v = vec[i].second.second;
		long double d = vec[i].first;
		bit[u][v] = bit[v][u] = 1;
		c = (bit[u] & bit[v]);
		if(c.count() > 0) return cout << sqrt(d)/2. << endl,0;
	}
	return 0;
}
