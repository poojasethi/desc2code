#include <iostream>
#include <cstdio>
#include <algorithm>
#include <bitset>
#include <cmath>

#define Sqr(x) (x)*(x)

using namespace std;

const int N=3005;

int n;
int x[N],y[N];

bitset<N> V[N];

struct Cnode{
    int x,y,v;
}G[N*N];

double Calc(int i,int j){
    return Sqr(x[i]-x[j])+Sqr(y[i]-y[j]);
}

bool Cmp(Cnode a,Cnode b){
    return a.v>b.v;
}

int main(){
    scanf("%d",&n);
    for(int i=0;i!=n;i++){
        int X,Y;
        scanf("%d%d",&X,&Y);
        x[i]=X,y[i]=Y;
    }
    int top=0;
    for(int i=0;i!=n;i++){
        for(int j=i+1;j!=n;j++){
            G[top++]={i,j,Calc(i,j)};
        }
    }
    sort(G,G+top,Cmp);
    for(int x,y,i=0;i!=top;i++){
        x=G[i].x,y=G[i].y;
        if ((V[x]&V[y]).any()){
            printf("%.10lf\n",sqrt((double)G[i].v)/2);
            return 0;
        }
        V[x].set(y);
        V[y].set(x);
    }
    return 0;
}
