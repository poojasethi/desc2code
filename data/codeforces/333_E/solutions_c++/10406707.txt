#include<cstdio>
#include<algorithm>
#include<cmath>
#define N 5000010
using namespace std;
int n,tot,l[N],r[N],x[N],y[N],i,j;
int f[3010][110];
struct typ{
       int l,r,w;       
}a[N];
int dis(int a,int b)
{
    return (x[a]-x[b])*(x[a]-x[b])+(y[a]-y[b])*(y[a]-y[b]);
}
bool cmp(typ x,typ y)
{
     return x.w>y.w;
}
int main()
{
    scanf("%d",&n);
    for (i=0;i<n;i++)
    scanf("%d%d",&x[i],&y[i]);
    for (i=0;i<n;i++)
    for (j=i+1;j<n;j++)
    {
        tot++;a[tot].w=dis(i,j);
        a[tot].l=i;a[tot].r=j;
    }
    sort(a+1,a+1+tot,cmp);
    for (i=1;i<=tot;i++)
    {
        f[a[i].l][a[i].r/30]+=(1<<(a[i].r%30));
        f[a[i].r][a[i].l/30]+=(1<<(a[i].l%30));
        for (j=0;j<=n/30;j++)
        if (f[a[i].l][j]&f[a[i].r][j]) 
        {
                printf("%.12lf",sqrt(a[i].w)/2);        
                return 0;             
        }
    }
}
