#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<bitset>
#include<cmath>
using namespace std;
bitset<3000>b[3000];

int xx[3333], yy[3333];
struct Node{
    int x,y;
    double z;
}f[5000000];
double dis(int i,int j){
    return (double)((xx[i]-xx[j])*(xx[i]-xx[j])+(yy[i]-yy[j])*(yy[i]-yy[j]));
}

bool operator<(Node a,Node b)
{
    return a.z<b.z;
}

int main()
{
    int n,m=0;
    scanf("%d", &n);
    for(int i=0;i<n;++i)scanf("%d%d", &xx[i], &yy[i]);
    for(int i=0;i<n;++i)
        for(int j=i+1;j<n;++j){
            f[++m].x=i;
            f[m].y=j;
            f[m].z=sqrt(dis(i,j))/2;
        }
    sort(f+1,f+m+1);
    for(int i=m;;--i)
    {
        if((b[f[i].x]&b[f[i].y])!=0)
        {
            printf("%.10lf\n",f[i].z);
            break;
        }
        b[f[i].x][f[i].y]=1;
        b[f[i].y][f[i].x]=1;
    }
    return 0;
}
