#include<stdio.h>
#include<iostream>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<bitset>
#include<math.h>
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
int xx[3333];
int yy[3333];
double d[3333][3333];
struct E
{
	int x,y;
	double z;
	E(){}
	E(int x,int y,double z):x(x),y(y),z(z){}
}e[9000020];
bitset<3012> b[3012];
int n;
bool operator<(const E&a,const E&b)
{
	return a.z>b.z;
}
int main(void)
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d %d",&xx[i],&yy[i]);
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			e[i*n+j]=E(i,j,((double)xx[i]-xx[j])*(xx[i]-xx[j])+((double)yy[i]-yy[j])*(yy[i]-yy[j]));
	sort(e,e+n*n);
	for(int i=0;i<n*n;i++)
	{
		if((b[e[i].x]&b[e[i].y]).any())
		{
			printf("%.10f",sqrt(e[i].z)/2);
			return 0;
		}
		b[e[i].x][e[i].y]=1;
		b[e[i].y][e[i].x]=1;
	}
	return 0;
}
