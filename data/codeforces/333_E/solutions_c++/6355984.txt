#include <cmath>
#include <bitset>
#include <iomanip>
#include <iostream>
#include <algorithm>
using namespace std;
typedef pair<int, int> pii;
typedef pair<int, pii> piii;
const int N = 3e3 + 3;

int n, m, x[N], y[N];
piii a[N * N];
bitset<N> adj[N];

int main() {
	ios_base::sync_with_stdio(false);
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> x[i] >> y[i];
		for (int j = 0; j < i; j++)
			a[m++] = piii((x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j]), pii(i, j));
	}
	sort(a, a + m);
	for (int i = m - 1; ;i--) {
		int u = a[i].second.first, v = a[i].second.second;
		if ((adj[u] & adj[v]).any()) {
			cout << fixed << setprecision(10) << sqrt((double) a[i].first) / 2 << '\n';
			break;
		}
		adj[u][v] = adj[v][u] = 1;
	}
	return 0;
}
