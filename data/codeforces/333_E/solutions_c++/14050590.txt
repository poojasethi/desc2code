#include<stdio.h>
#include<algorithm>
#include<bitset>
#include<math.h>
#define X first
#define Y second
using namespace std;
typedef pair<int,int>ii;
typedef pair<int,ii>iii;
int x[3020],y[3020],ee,n;
iii e[4500020];
bitset<3012>b[3012];
int D(int i,int j)
{
	return (x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
}
int main(void)
{
	scanf("%d",&n);
	for(int i=0,j;i<n;i++)
		for(scanf("%d%d",&x[i],&y[i]),j=0;j<i;j++)
			e[ee++]=iii(D(i,j),ii(i,j));
	sort(e,e+ee);
	for(int i=ee;i--;)
	{
		if((b[e[i].Y.X]&b[e[i].Y.Y])!=0)
			return printf("%.9f",sqrt(e[i].X)/2),0;
		b[e[i].Y.X][e[i].Y.Y]=1;
		b[e[i].Y.Y][e[i].Y.X]=1;
	}
}
