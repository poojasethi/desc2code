#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <bitset>
#include <utility>
#include <functional>

using namespace std;

const int MaxN = 3010;

typedef pair<int,int> PII;
typedef pair<int,PII> PIP;

int n;
int x[MaxN],y[MaxN];
int a[MaxN][MaxN],cnt;
PIP c[MaxN * MaxN];
bitset<MaxN> b[MaxN];

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d%d",&x[i],&y[i]);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++) if(i!=j)
			a[i][j]=(x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]);
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			c[++cnt]=PIP(a[i][j],PII(i,j));
	sort(c+1,c+1+cnt,greater<PIP>());
	int ans=0;
	for(int l=1;l<=cnt;l++){
			int i=c[l].second.first,j=c[l].second.second;
			if((b[i]&b[j]).any()) {ans=a[i][j];break;}
			b[i][j]=b[j][i]=1;
		}
	printf("%.10lf\n",sqrt(ans)/2);
	return 0;
}
