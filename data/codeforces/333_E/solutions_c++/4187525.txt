#include <cstdio>
#include <cmath>
#include <ctime>
#include <utility>
#include <algorithm>
#define X first
#define Y second
#define DO double
#define REP(i,n) for (int i=1;i<=n;++i)
using namespace std;

int n;
pair<DO,DO> a[3030];
DO d[3030][3030];
DO ans;

inline DO Dist(int x,int y) {
	return sqrt((a[x].X-a[y].X)*(a[x].X-a[y].X)+(a[x].Y-a[y].Y)*(a[x].Y-a[y].Y));
	return 0;
}

int main() {
	srand(time(0));
	scanf("%d",&n);
	REP(i,n)
		scanf("%lf%lf",&a[i].X,&a[i].Y);
	sort(a+1,a+n+1);
	REP(i,n) REP(j,n)
		d[i][j]=Dist(i,j);
	REP(i,10000000) {
		int x=rand()%n+1;
		int y=rand()%n+1;
		int z=rand()%n+1;
		ans=max(ans,min(d[x][y],min(d[x][z],d[y][z])));
	}
	for (int x=n;x;--x)
		REP(y,x-1)
			if (d[x][y]>ans)
				REP(z,y/7)
					ans=max(ans,min(d[x][y],min(d[x][z],d[y][z])));
	printf("%.20lf\n",ans/2);
	return 0;
}
