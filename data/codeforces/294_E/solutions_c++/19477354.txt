#include "bits/stdc++.h"
#define MAXN 5009
#define INF 1000000007
#define mp(x,y) make_pair(x,y)
#define all(v) v.begin(),v.end()
#define pb(x) push_back(x)
#define wr cout<<"----------------"<<endl;
#define ppb() pop_back()
#define tr(ii,c) for(typeof((c).begin()) ii=(c).begin();ii!=(c).end();ii++)
#define ff first
#define ss second

using namespace std;

typedef long long ll;
typedef pair<ll,ll> PII;
template<class T>bool umin(T& a,T b){if(a>b){a=b;return 1;}return 0;}
template<class T>bool umax(T& a,T b){if(a<b){a=b;return 1;}return 0;}
ll sub[MAXN];
vector<PII>adj[MAXN];
void dfs(ll nd,ll pr){
	sub[nd]=1;
	tr(it,adj[nd])
		if(it->ff!=pr){
			dfs(it->ff,nd);
			sub[nd]+=sub[it->ff];
		}
}
ll dfs1(ll nd,ll pr){
	ll res=0;
	tr(it,adj[nd])
		if(it->ff!=pr){
			res+=dfs1(it->ff,nd);
			res+=sub[it->ff]*(it->ss);
		}
	return res;	
}
ll ans,n;
ll dfs2(ll nd,ll pr,ll sz,ll cost){
	ans+=cost;ll res=cost;
	tr(it,adj[nd])
		if(it->ff!=pr){
			ll par=sz-sub[it->ff];
			if(par>=sub[it->ff])
				umin(res,dfs2(it->ff,nd,sz,cost+(par-sub[it->ff])*it->ss));
			else
				umin(res,dfs2(it->ff,nd,sz,cost-(sub[it->ff]-par)*it->ss));
		}
	return res;	
}
ll solve(ll x,ll y,ll z){
	dfs(x,y);dfs(y,x);ans=0;
	ll dis1=dfs2(x,y,sub[x],dfs1(x,y));
	ll dis2=dfs2(y,x,sub[y],dfs1(y,x));
	return ans/2+(sub[x]*dis2)+(sub[y]*dis1)+(sub[x]*sub[y]*z);
}
pair<PII,ll>edge[MAXN];
int main(){
	ll mn=ll(1e18);
	#ifndef ONLINE_JUDGE
    freopen("file.in", "r", stdin);
	#endif	
	scanf("%lld",&n);
	for(ll i=1;i<n;i++){
		ll u,v,w;
		scanf("%lld%lld%lld",&u,&v,&w);
		adj[u].pb(mp(v,w));
		adj[v].pb(mp(u,w));
		edge[i]=mp(mp(u,v),w);
	}	
	for(ll i=1;i<n;i++)
		umin(mn,solve(edge[i].ff.ff,edge[i].ff.ss,edge[i].ss));
	printf("%lld\n",mn);	
	return 0;
}
