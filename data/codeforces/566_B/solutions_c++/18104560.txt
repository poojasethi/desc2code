#include <algorithm>
#include <stdio.h>
#include <memory.h>
#include <string.h>
#include <vector>
#include <queue>
#include <set>
#include <map>
#include <string>

using namespace std;

typedef pair<int, int> pii;
typedef long long INT;

#define NN 121000

int cnt[NN], vst[NN];
int a[NN], b[NN], c[NN];
int qu[NN][10];

set<pii> s;

int ans[NN];

int main() {
	int N;
	scanf("%d", &N);
	for (int i=0; i<N*4; i++) {
		scanf("%d%d%d", &a[i], &b[i], &c[i]);
		a[i]--, b[i]--, c[i]--;
		qu[b[i]][cnt[b[i]]++]=i;
		qu[c[i]][cnt[c[i]]++]=i;
	}
	
	for (int i=0; i<N; i++) s.insert(pii(cnt[i], i));
	
	for (int i=0; i<4*N; i++) {
		int u=s.rbegin()->second;
		for (int j=0; j<8; j++) {
			int k=qu[u][j];
			
			if (vst[k]) continue;
			
			vst[k]=1; ans[i]=k;
			
			s.erase(pii(cnt[c[k]], c[k]));
			cnt[c[k]]--;
			s.insert(pii(cnt[c[k]], c[k]));
			
			s.erase(pii(cnt[b[k]], b[k]));
			cnt[b[k]]--;
			s.insert(pii(cnt[b[k]], b[k]));
			
			s.erase(pii(cnt[a[k]], a[k]));
			cnt[a[k]]++;
			s.insert(pii(cnt[a[k]], a[k]));
			
			break;
		}
	}
	
	reverse(ans, ans+N*4);
	
	puts("YES");
	for (int i=0; i<4*N; i++) {
		if (i) printf(" ");
		printf("%d", ans[i]+1);
	}
	puts("");
	
	return 0;
}
