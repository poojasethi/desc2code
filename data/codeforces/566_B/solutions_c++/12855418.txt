#include<bits/stdc++.h>
#define ll long long int
#define F first
#define S second
#define rep(i,b,a) for(i=a;i<b;i++)
#define f(i,n) rep(i,n,0)
#define ii pair<int,int>
#define vi vector<int>
#define mp make_pair
#define pb push_back
using namespace std;
#define maxn 100010

/*The ans can never be NO :)*/
int main()
{
	vector< pair<int,ii> > a;
	int n,ans[maxn];
	set<int> visited;
	vi final;
	int i,x,y,z;
	cin>>n;
	a.pb(mp(0,mp(0,0)));
	rep(i,4*n+1,1) {cin>>x>>y>>z;a.pb(mp(x,mp(y,z)));}
	rep(i,4*n+1,1) visited.insert(i);
	rep(i,n+1,1) ans[i]=4;
	__typeof(visited.begin()) it=visited.begin();
	while(visited.size()>0)
	{
		i=(*it);
		x=a[i].F;
		y=a[i].S.F;
		z=a[i].S.S;
		if((ans[y]<9 and z!=y and x==z) or
			(ans[z]<9 and z!=y and x==y) or
			(ans[y]<9 and ans[z]<9 and z!=y and y!=x and z!=x) or
			(ans[y]<9 and z==y and z==x) or
			(ans[y]<8 and z==y and z!=x))
		{
			ans[y]++;
			ans[z]++;
			ans[x]--;
			final.pb(i);
			visited.erase(it++);
		}
		else it++;
		if(it==visited.end()) it=visited.begin();
	}
	cout<<"YES"<<endl;
	f(i,final.size()) cout<<final[i]<<" ";
	
}
