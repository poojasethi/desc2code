using namespace std;
int n,n4;
int cnt[30005],a[30005][10],b[30005][10],id[30005][10],p[30005],ans[200005],l;
priority_queue <pair<int,int> >q;

int main()
{
	cin>>n;
	n4=n*4;
	for (int i=1;i<=n4;i++)
	{
		int x,y,z;
		cin>>x>>y>>z;
		p[x]++;
		a[x][p[x]]=y;
		b[x][p[x]]=z;
		id[x][p[x]]=i;
	}
	for (int i=1;i<=n;i++)
	{
		cnt[i]=8;
		q.push(make_pair(-8,i));
		p[i]=1;
	}
	while (!q.empty())
	{
		int i=q.top().second;
		q.pop();
		if (cnt[i]==9 || p[i]>4) continue;
		int j=a[i][p[i]];
		int k=b[i][p[i]];
		cnt[i]++;
		cnt[j]--;
		cnt[k]--;
		q.push(make_pair(-cnt[i],i));
		q.push(make_pair(-cnt[j],j));
		q.push(make_pair(-cnt[k],k));
		ans[++l]=id[i][p[i]];
		p[i]++;
	}
	cout<<"YES"<<endl;
	for (int i=l;i>=1;i--)
	{
		cout<<ans[i]<<" ";
	}
	return 0;
}
