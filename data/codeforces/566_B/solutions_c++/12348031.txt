#include <iostream>
#include <vector>
#include <cstdio>
using namespace std;
int N;
int Count[4*30005];
struct Elem{
int a,b,c,poz;
};
vector <Elem> V;
void Read()
{
    cin>>N;
    for(int i=1;i<=4*N;i++)
    {
        Elem p;
        cin>>p.a>>p.b>>p.c;
        p.poz=i;
        V.push_back(p);
        Count[i]=4;
    }
}

void Solve()
{
    cout<<"YES\n";
    int i=0;
    while(!V.empty())
    {
        if(i==V.size())
            i=0;
        int a=V[i].a;
        int b=V[i].b;
        int c=V[i].c;
        int poz=V[i].poz;
        Count[a]--;Count[b]++;Count[c]++;
        if(Count[a]>=0 && Count[b]<=9 && Count[c]<=9)
        {
            V[i]=V.back();
            V.pop_back();
            printf("%d ",poz);
        }
        else
        {
            Count[a]++;
            Count[b]--;
            Count[c]--;
            ++i;
        }
    }
}
int main()
{
    Read();
    Solve();
    return 0;
}
