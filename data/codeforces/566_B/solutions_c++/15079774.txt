#include <bits/stdc++.h>
using namespace std;
const int MAXN = 120010;
#define FI "socialnet.i1"
#define FO "socialnet.o1"

int n;
vector<int> A, B, C, cnt, ans, used;

bool ok(int i){
	bool res = false;
	cnt[A[i]]--, cnt[B[i]]++, cnt[C[i]]++;
	if(cnt[B[i]] <= 9 && cnt[C[i]] <=9 ) res = true;
	cnt[A[i]]++, cnt[B[i]]--, cnt[C[i]]--;
	return res;
}

int main(){
    #ifndef ONLINE_JUDGE
    ifstream cin(FI);
    ofstream cout(FO);
    #endif // ONLINE_JUDGE
	cin >> n; cnt.assign(n, 4);
	n <<= 2;
	A.resize(n), B.resize(n), C.resize(n), used.resize(n);
	for(int a, b, c, i = 0; i < n; i++){
        cin >> a >> b >> c; a--, b--, c--;
        A[i] = a, B[i] = b, C[i] = c; used[i] = 0;
	}
	for(int j = 0; j < n; )	{
		for(int i = 0; i < n; i++)
			if(!used[i] && ok(i)){
				cnt[A[i]]--, cnt[B[i]]++, cnt[C[i]]++;
				used[i] = true;
				ans.push_back(i+1);
				j++;
			}
	}
	cout<< "YES\n";
	for(int i = 0; i < (int)ans.size(); i++) cout << ans[i] <<' ';
	return 0;
}
