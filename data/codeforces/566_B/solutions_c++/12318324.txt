#include <iostream>
#include <cmath>
#include <cstdio>
#include <algorithm>
#include <vector>

#define MAXN 120200

using namespace std;

int cnt , n , A[MAXN] , B[MAXN] , C[MAXN] , now[MAXN >> 2] , vis[MAXN];

inline void check(int idx)
{
    if(!vis[idx])
    {
        --now[A[idx]];++now[B[idx]];++now[C[idx]];
        if(now[A[idx]] < 10 and now[B[idx]] < 10 and now[C[idx]] < 10)
        {
            ++cnt;
            vis[idx] = true;
            printf("%d " , idx);
            return;
        }
        ++now[A[idx]];--now[B[idx]];--now[C[idx]];
    }
    return;
}

int main()
{
    scanf("%d" , &n);
    for(int i = 1;i < n + 1;++i)
        now[i] = 4;

    n <<= 2;
    for(int i = 1;i < n + 1;++i)
    {
        int a , b , c;
        scanf("%d%d%d" , A + i , B + i , C + i);
    }
    printf("YES\n");
    while(cnt < n)
        for(int i = 1;i < n + 1;++i)
            check(i);

    return 0;
}
