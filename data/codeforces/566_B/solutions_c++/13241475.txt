#include<bits/stdc++.h>

using namespace std;

int const MAXN = 3e4+10;

int n, pt[MAXN];

typedef pair<int,int>pii;
typedef pair<int,pii>tri;

multiset<tri> p[MAXN];
multiset<tri>::iterator it;

set<pii> s;
set<pii>::iterator ii, tt;

int tam[MAXN], ans[4*MAXN];

int main(){
    cin >> n;
    for(int i = 1; i <= n; i++) tam[i] = 8;
    for(int i = 1; i <= 4*n; i++){
        int a, b, c;
        cin >> a >> b >> c;
        p[a].insert( tri( i, pii(b,c) ) );
    }   
    int id = 0;
    for(int i = 1; i <= n; i++) s.insert(pii(tam[i],i));
    while(id != 4*n){
            ii = s.begin();
            int next = ii->second;
            if(ii != s.end() )s.erase(ii);
            if(pt[next]==4)continue;                    
            it = p[next].begin();
            int ind = it->first, b = it->second.first, c = it->second.second;
            if(next!=b){ tt = s.find(pii(tam[b],b)); if(tt != s.end() ) s.erase(tt); }
            if(next!=c && b!=c){ tt = s.find(pii(tam[c],c)); if(tt != s.end() ) s.erase(tt); }
            tam[next]++, tam[b]--, tam[c]--;
            s.insert(pii(tam[b],b));
            s.insert(pii(tam[c],c));
            s.insert(pii(tam[next],next));
            pt[next]++;
            if(it != p[next].end()) p[next].erase(it);
            ans[++id] = ind;       
    }

    cout << "YES" << '\n';
    for(int i = 4*n; i >= 1; i--) cout << ans[i] << ' ';
    cout << '\n';

    return 0;
}
