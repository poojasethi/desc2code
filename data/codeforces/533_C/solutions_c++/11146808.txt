#include<stdio.h>
#include<algorithm>
using namespace std;
int xp,yp,xv,yv;

int main()
{
    scanf("%d%d%d%d",&xp,&yp,&xv,&yv);
    int bestp = xp+yp;
    int bestv = max(xv,yv);
    if(bestp<=bestv) {printf("Polycarp"); return 0;}
    if(xp<=xv && yp<=yv) {printf("Polycarp"); return 0;}//Block
    else printf("Vasiliy");
    return 0;
}
