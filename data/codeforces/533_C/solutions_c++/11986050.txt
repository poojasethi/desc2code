#include<iostream>
#include<algorithm>
using namespace std;
#define OFF ios::sync_with_stdio(false)
int main()
{
    OFF;
    int xp,yp,xv,yv;
    cin>>xp>>yp>>xv>>yv;
    if(((xp+yp)<=max(xv,yv))||((xp<=xv)&&(yp<=yv)))
        cout<<"Polycarp"<<endl;
    else
        cout<<"Vasiliy"<<endl;
    return 0;
}
