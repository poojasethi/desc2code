#include<stdio.h>

#define MAX(x,y) (x>y?x:y)

int main()
{
    int xp,yp,xv,yv;
    int distp,distv;

    scanf("%d %d %d %d",&xp,&yp,&xv,&yv);

    distp=(xp+yp);
    distv=MAX(xv,yv);

    if((xv>=xp&&yv>=yp)||distp<=distv)
        printf("Polycarp");

    else
        printf("Vasiliy");


    return 0;
}
