#include<bits/stdc++.h>

using namespace std;

int xp, yp, xv, yv, ansp, ansv;

int main()
{
    cin >> xp >> yp >> xv >> yv;
    if (xp+yp <= max(xv, yv))
        cout << "Polycarp";
    else
    {
        if (xp > xv || yp > yv)
            cout << "Vasiliy";
        else
            cout << "Polycarp";
    }
    return 0;
}
