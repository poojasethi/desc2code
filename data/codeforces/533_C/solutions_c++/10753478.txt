#include <iostream>
using namespace std;

int main()
{
    int px,py,cx,cy;
	cin>>px>>py>>cx>>cy;
	if ((px<=cx) && (py<=cy)) {
		cout<<"Polycarp"<<endl;
		return 0;
	}
	if ((px>=cx) && (py>=cy))
	{
		cout<<"Vasiliy"<<endl;
		return 0;
	}
	if (px+py<=max(cx,cy)) cout<<"Polycarp"<<endl;
	else cout<<"Vasiliy"<<endl;
	return 0;
}
