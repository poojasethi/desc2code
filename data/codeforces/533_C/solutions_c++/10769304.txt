#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int x1,y1,x2,y2;
    scanf("%d%d%d%d",&x1,&y1,&x2,&y2);
    int flag=0;
    int s1=x1+y1;
    int s2=max(x2,y2);
    if(s1<=s2)flag=1;
    if(x1>=0&&x1<=x2&&y1>=0&&y1<=y2)flag=1;
    if(flag)printf("Polycarp\n");
    else printf("Vasiliy\n");
}
