import sys

nm = raw_input()
nm = nm.split()
n = int(nm[0])
m = int(nm[1])

tags = raw_input()
tags = tags.split(" ")
fruits = []
fruitcount = []

for i in range(0, m):
	fruit = raw_input()
	if fruit not in fruits:
		fruits.append(fruit)
		fruitcount.append(1)
	else:
		index = fruits.index(fruit)
		fruitcount[index] = fruitcount[index]+1

min = 0
max = 0
newtags = [int(i) for i in tags]
newtags.sort()
mintags = newtags[:]
newtags.reverse()
maxtags = newtags[:]

fruitcount.sort()
fruitcount.reverse()
for i in range(0, len(fruitcount)):
	min = min+(fruitcount[i]*mintags[i])
	max = max+(fruitcount[i]*maxtags[i])

print min, max
