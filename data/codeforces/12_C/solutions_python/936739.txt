number = raw_input().split(" ")
prices = [int(x) for x in raw_input().split(" ")]

prices = sorted(prices)

names = dict()
for i in range(0,int(number[1])):
    s = str(raw_input())    
    if ( s in names ):
        names[s]+=1
    else:
        names[s]=1

frecs =[]
for k,v in names.iteritems():
    frecs.append(v) 

frecs = sorted(frecs)
frecs.reverse()

# print prices
# print frecs

smallest = 0
for i in range(0,len(frecs)):
    smallest+= prices[i]*frecs[i]

prices.reverse()

biggest = 0
for i in range(0,len(frecs)):
    biggest+= prices[i]*frecs[i]
    
print str(smallest) + " "+str(biggest)
