N,M=raw_input().strip().split()
N,M=int(N),int(M)
arr=list(map(int,raw_input().strip().split()))
arr.sort()
fruits={}
for i in xrange(M):
	fruit=raw_input().strip()
	if fruit in fruits:
		fruits[fruit]+=1
	else:
		fruits[fruit]=1
count=fruits.values()
count.sort()
mins,maxs=0,0
j=0
for i in count[::-1]:
	mins+=i*arr[j]
	j+=1
j=N-1
for i in count[::-1]:
	maxs+=i*arr[j]
	j-=1
print mins,maxs
