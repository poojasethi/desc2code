def main():
    n, m = map(int, raw_input().split())
    prices = sorted(map(int, raw_input().split()))
    data = {}
    for i in xrange(m):
        key = raw_input()
        data[key] = data.get(key, 0) + 1

    values = sorted(data.itervalues())
    a, b = 0, 0
    N = len(values)
    for i in xrange(N):
        a += values[N - i - 1] * prices[i]
    for i in xrange(N):
        b += values[N - i - 1] * prices[n - i - 1]

    print "%s %s" % (a, b)


if __name__ == '__main__':
    main()
