import sys

lines = sys.stdin.readlines()
p = [int(i) for i in lines[1].rstrip().split(" ")]
items= [i.rstrip() for i in lines[2:]]
counts = []
keys = []
for i in items:
    if i not in keys:
        counts.append(items.count(i))
        keys.append(i)
    
p.sort()
counts.sort()

small = 0
big = 0
p_size = len(p)
c_size = len(counts)
for i in range(c_size):
    small += counts[c_size - 1 - i] * p[i]
    big += counts[c_size -1 - i] * p[p_size - 1 - i]

print small, big
