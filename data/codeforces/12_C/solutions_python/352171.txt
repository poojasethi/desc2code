import operator

N, M = [int(x) for x in raw_input().split()]
tags = [int(x) for x in raw_input().split()]
tags.sort()
count = {}
for m in range(M):
    name = raw_input()
    count[name] = count.get(name, 0) + 1
wants = count.values()
wants.sort(reverse = True)
mi = sum( [x * y for x, y in zip(tags, wants) ])
tags.reverse()
ma = sum( [x * y for x, y in zip(tags, wants) ])
print mi, ma
