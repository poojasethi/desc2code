n,m = map(int,raw_input().split())
price = sorted(map(int,raw_input().split()))

f = {}
for i in xrange(m):
    s = raw_input()
    if s not in f.keys():
        f[s] = 1
    else:
        f[s] += 1
f1 = f.values()
minp,maxp = 0,0

for i,j in enumerate(sorted(f1,reverse=True)):
    minp += price[i]*j
for i,j in enumerate(sorted(f1,reverse=True)):
    maxp += price[len(price)-i-1]*j
print minp,maxp
