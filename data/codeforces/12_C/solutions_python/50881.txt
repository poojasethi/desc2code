from sys import *

n,m = map(int, stdin.readline().split())
cost = map(int,stdin.readline().split())
cost.sort()

items = dict()
for _ in xrange(m):
    item = stdin.readline()[:-1]
    if item in items:
        items[item] += 1
    else:
        items[item] = 1

vals = items.values()
vals.sort()
vals.reverse()

min, max = 0,0
for i in xrange(len(vals)): min += cost[i]*vals[i]

cost.reverse()
for i in xrange(len(vals)): max += cost[i]*vals[i]

print min,max
