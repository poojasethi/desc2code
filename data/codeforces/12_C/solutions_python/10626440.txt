
from itertools import groupby

n, m = map(int, raw_input().split()) 

P = sorted(map(int, raw_input().split()))

F = sorted([raw_input() for _ in range(m)])
C = []

for k, g in groupby(F): 
    C.append(len(list(g)))

t = min(n, len(C))

print sum(map(lambda p, c: p * c, P[:t], sorted(C, reverse=True))[:t]), 
print sum(map(lambda p, c: p * c, 
    P[-1:-t-1:-1], sorted(C, reverse=True)[:t]))
