def stdin():
   return map(int, raw_input().split());


n, m = stdin()
cost = sorted(stdin())

map = {}

for i in range(m):
   s = raw_input()
   if s not in map:
      map[s] = 0
   map[s] += 1
   
count = sorted(map.values())

min_cost, max_cost = 0, 0

m = len(count)

for i in range(m):
   min_cost += count[i] * cost[m - i - 1]
   max_cost += count[m - i - 1] * cost[n - i - 1]
   
print min_cost, max_cost
