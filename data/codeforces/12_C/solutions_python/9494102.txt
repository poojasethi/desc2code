n,m=map(int,raw_input().split())
l=map(int,raw_input().split())
l.sort()
w={}
while m:
    m-=1
    s=raw_input()
    if w.has_key(s):
        w[s]+=1
    else:
        w[s]=1
w2=sorted(w.items(), key=lambda x:x[1])
p=len(w2)
a=b=chk=0
for i in range(-1,-p-1,-1):
    a+=l[chk]*w2[i][1]
    chk+=1

for j in range(-1,-p-1,-1):
    b+=l[j]*w2[j][1]
print a,b
