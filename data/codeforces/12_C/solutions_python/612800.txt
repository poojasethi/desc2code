n, m = map(int, raw_input().split())
a = map(int, raw_input().split());
mp = {}
for i in range(m):
    x = raw_input()
    if(x in mp):
        mp[x] += 1
    else:
        mp[x] = 1
b = mp.values()
b.sort()
b.reverse()
a.sort()
x = 0
y = 0
for i in range(len(b)):
    x += a[i] * b[i]
    y += a[n - i - 1] * b[i]
print x, y
