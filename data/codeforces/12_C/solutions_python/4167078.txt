n,m = tuple(map(int,raw_input().split()))
price = list(map(int,raw_input().split()))
cn = {}
for i in xrange(0,m):
    tmp = raw_input()
    if tmp not in cn:
        cn[tmp]=1
    else:
        cn[tmp]+=1

price.sort()
cns = cn.values()
cns.sort()
#print price
#print cns
minans = maxans = 0

i = 0
j = len(cns)-1
while j>=0:
    minans += price[i]*cns[j]
    j-=1
    i+=1
i = len(price)-1
j = len(cns)-1
while j>=0:
    maxans += price[i]*cns[j]
    j-=1
    i-=1

print minans, maxans
