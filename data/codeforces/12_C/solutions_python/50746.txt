n, m = map(int, raw_input().split())
c = map(int, raw_input().split())

names = {}
for i in range(m):
    s = raw_input()
    if s not in names: names[s] = 0
    names[s] += 1

a = []
for s in names: a.append(names[s])

c.sort()
a.sort()

sum = 0
for i in range(len(a)):
    sum += a[i]*c[len(a)-1-i]
sum1 = sum

sum = 0
for i in range(len(a)):
    sum += a[i]*c[len(c)-len(a)+i]
print sum1, sum
