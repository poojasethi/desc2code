# from itertools import *
from collections import defaultdict

k = defaultdict(list)
temp, T = map(int, raw_input().split())
cost = map(int, raw_input().split())
data2 = [ raw_input() for i in xrange(T) ]
cost = sorted(cost)
t_cost = cost[:]
t_cost.reverse()
for i in data2:
    if i in k:
        k[i] += 1
    else:
        k[i] = 1
        
wannabe = []

for j in k:
    wannabe.append(k[j])
    
wannabe = sorted(wannabe)
wannabe.reverse()

max_cost = min_cost = 0
for i in wannabe:
    max_cost += i*cost.pop()
    min_cost += i*t_cost.pop()
print min_cost, max_cost
