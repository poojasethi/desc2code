n,m = map(int, raw_input().split())
price = sorted(map(int, raw_input().split()))
fruit = []
rep = [1]
mi = 0
ma = 0

for i in xrange(m):
    fruit.append(str(raw_input()))
    
fruit = sorted(fruit)

for j in xrange(1,m):
    if fruit[j] == fruit[j-1]:
        rep[len(rep)-1] += 1
    else:
        rep.append(1)

rep = sorted(rep)
rep.reverse()

for a,b in zip(rep, price):
    mi += a*b
price.reverse()
for a,b in zip(rep,price):
    ma += a*b

print "%s %s" % (mi, ma)
