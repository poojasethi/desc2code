a,b=map(int,raw_input().split())
c=map(int,raw_input().split())
c.sort()
maxsort=c[::-1]
d={}
for i in range(b):
    s=raw_input()
    if s in d:
        d[s]+=1
    else:
        d[s]=1
s=[]
for k in d.values():
    s.append(k)
s.sort()
s=s[::-1]
min,max=0,0
for i in range(len(s)):
    min=min+int(s[i])*int(c[i])

for i in range(len(s)):
    max=max+int(s[i])*int(maxsort[i])
print min,max
