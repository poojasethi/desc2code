n,m = map(int, raw_input().split())
p = sorted(map(int, raw_input().split()))
dic = {}
for _ in xrange(m):
    s = raw_input()
    dic[s] = dic.get(s, 0)+1
fru = sorted(dic, key=lambda x: dic[x], reverse=True)

minn,maxn = 0,0
for i,s in enumerate(fru):
    minn += p[i]*dic[s]
    maxn += p[-i-1]*dic[s]
print minn, maxn
