n, m = map(int, raw_input().split())
p = sorted(map(int, raw_input().split()))
d = {}
for _ in xrange(m):
    s = raw_input()
    if s not in d:
        d[s] = 0
    d[s] += 1
d = sorted(d.values())
m = len(d)
a, b = 0, 0
for i in xrange(m):
    a += p[i] * d[m - 1 - i]
    b += p[n - 1 - i] * d[m - 1 - i]
print a, b
