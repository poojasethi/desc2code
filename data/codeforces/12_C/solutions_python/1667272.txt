#!/usr/bin/python
import sys

def calc(fruits, prices):
	sum = 0
	for i in range(len(fruits)):
		sum += fruits[i] * prices[i]
	return sum

(a, b) = map(lambda x: int(x), sys.stdin.readline().split())
prices = sys.stdin.readline().split()
prices = map(lambda x: int(x), prices)
prices.sort()

dic = {}
for i in range(b):
	line = sys.stdin.readline().strip()
	if dic.has_key(line):
		dic[line] += 1
	else:
		dic[line] = 1

items = map(lambda x: dic[x], dic.keys())
items.sort()
items.reverse()

total = calc(items, prices)
print total,
prices.reverse()
total = calc(items, prices)
print total
