n,m = map(int, raw_input().split())
a = map(int, raw_input().split())
b = {}
for i in xrange(m):
	s = raw_input()
	if(b.has_key(s)):
		b[s] += 1
	else:
		b[s] = 1
v = b.values()
v = sorted(v, reverse = True)
a.sort()
mini = 0
maxi = 0
l = len(v)
for i in xrange(l):
	mini += v[i] * a[i]
a.reverse()
for i in xrange(l):
	maxi += v[i] * a[i]
print mini,maxi
