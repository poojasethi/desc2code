def numeric_compare(x,y):
    return x - y

def reverse_compare(x,y):
    return y - x

n,m = map(int,raw_input().split())
prices = map(int,raw_input().split())
fruit = {}

p1 = sorted(prices, cmp=numeric_compare)
p2 = sorted(prices, cmp=reverse_compare)

for i in xrange(m):
    s = raw_input()
    if s in fruit:
        fruit[s] += 1
    else:
        fruit[s] = 1

l = []
for k in fruit:
    l.append(fruit[k])
l = sorted(l, cmp=reverse_compare)
low = 0
high = 0
for i, li in enumerate(l):
    low += li * p1[i]
    high += li * p2[i]
print low, high
