n,m = map(int,raw_input().split())
prices = map(int,raw_input().split())
juices = dict()
for i in range(m):
    ss = raw_input()
    if juices.has_key(ss): juices[ss] += 1
    else: juices[ss] = 1

num = juices.values()
num.sort()
prices.sort()


a,b = (0,0)
lnum = len(num)
for i in range(lnum):
    a += num[lnum-1-i]*prices[i]
    b += num[lnum-1-i]*prices[n-1-i]
print a,b
