import sys

n, m = map(int, sys.stdin.readline().split())
fruits = {}

x = map(int, sys.stdin.readline().split())
x.sort()

for i in xrange(m):
    fruit = sys.stdin.readline()
    y = fruits.get(fruit, 0)
    fruits[fruit] = y + 1

z = sorted(fruits.values(), reverse=True)

print str(sum([a*b for a,b in zip(x, z)])) + ' ' + str(sum([a*b for a,b in zip(reversed(x), z)]))
