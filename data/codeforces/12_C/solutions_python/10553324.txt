n, m = map(int, raw_input().split())
prices = map(int, raw_input().split())

fruits = {}
for x in xrange(m):
	fruit = raw_input()
	if fruit in fruits:
		fruits[fruit] += 1
	else:
		fruits[fruit] = 1

prices.sort()

min_index = 0
max_index = -1
min_value = 0
max_value = 0

numbers = fruits.values()
numbers.sort()
numbers = numbers[::-1]

for number_wanted in numbers:
	min_value += prices[min_index] * number_wanted
	max_value += prices[max_index] * number_wanted
	min_index += 1
	max_index -= 1

print '{0} {1}'.format(min_value, max_value)
