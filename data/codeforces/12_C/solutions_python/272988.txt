from itertools import *
[n, m], price = map (int, raw_input ().split ()), map (int, raw_input ().split ())
fruits = [raw_input () for _ in xrange (m)]

price = sorted (price)
fruits = sorted ([fruits.count (i) for i in set (fruits)], reverse=True)
a = reduce (lambda x, y: x + price[y] * fruits[y], range (len (fruits)), 0)

price = sorted (price, reverse=True)
b = reduce (lambda x, y: x + price[y] * fruits[y], range (len (fruits)), 0)
print a, b
