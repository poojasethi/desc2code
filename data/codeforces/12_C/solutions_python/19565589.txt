#!/usr/bin/python

from collections import deque
from collections import defaultdict as dd

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n, m = ia()
a = ia(); mi = sorted(a); ma = sorted(a, reverse=True)

d = dd(int)
for i in range(m):
    c = raw_input()
    d[c] += 1

d = d.values()
d.sort(reverse=True)

p_mi, p_ma = 0, 0
for i in range(len(d)):
    p_mi += d[i]*mi[i]
    p_ma += d[i]*ma[i]

print p_mi, p_ma
