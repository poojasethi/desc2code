#! /usr/bin/python

from sys import stdin

if __name__=='__main__':
	n,m=map(int, stdin.readline().split())
	values=map(int,stdin.readline().split())
	values.sort()
	names=[line.strip() for line in stdin]
	counts={}
	for name in names:
		cnt=counts.get(name,0)
		counts[name]=cnt+1
	counted=[counts[k] for k in counts]
	counted.sort()
	minimum=0
	for index,item in enumerate(reversed(counted)):
		minimum+=item*values[index]
	maximum=0
	for index,item in enumerate(reversed(counted)):
		maximum+=item*values[-index-1]
	print "%d %d"%(minimum, maximum)
