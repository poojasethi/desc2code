#include<bits/stdc++.h>
using namespace std;
int a[110] , mn , mx;
map<string,int> mp;
vector<int> v;
int main(){
	int n,m; cin >> n >> m;
	for(int i = 0 ; i < n ; i++) cin >> a[i];
	for(int j = 0 ; j < m ; j++){
		string s; cin >> s;
		mp[s]++;
	}
	for(map<string,int>::iterator it = mp.begin() ; it != mp.end() ; it++)v.push_back(it->second);
	sort(v.rbegin(),v.rend()); sort(a,a+n);
	for(int i = 0 ; i < v.size() ; i++)mn += v[i]*a[i];
	reverse(a,a+n);
	for(int i = 0 ; i < v.size() ; i++)mx += v[i]*a[i];
	cout << mn << " " << mx << endl;
}
