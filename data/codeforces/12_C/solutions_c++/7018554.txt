#include <iostream>
#include <map>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	int n, m, ans1=0, ans2=0;
	cin>>n>>m;
	int a[n];
	for (int i=0; i<n; i++) cin>>a[i];
	sort(a, a+n); 
	map <string, int> b;
	while (m--) {
		string x;
		cin>>x;
		b[x]++;
	}
	int k=b.size(), c[k];
	int i=0;
	for (map<string,int>::iterator it=b.begin(); it!=b.end(); it++) c[i++]=it->second;
	sort(c, c+k);
	for (int i=0; i<k; i++) {
		ans1+=c[k-i-1]*a[i];
		ans2+=c[k-i-1]*a[n-1-i];
	}
	cout<<ans1<<" "<<ans2<<endl;
	return 0;
}
