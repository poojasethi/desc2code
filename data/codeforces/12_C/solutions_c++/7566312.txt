#include<iostream>
#include<algorithm>
#include<cstdio>
#include<map>
#include<cstring>
#include<string>
using namespace std;
#define MAX 101
map<string,int> mp;
	int a[MAX];
	int b[MAX];
map<string,int>::iterator it;
int main()
{
	int i,n,m;

	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	string s;
	for(i=0;i<m;i++)
	{
		cin>>s;
		if(!mp[s])
			mp[s]=1;
		else
			mp[s]+=1;
	}
	int cnt=0;
	for(it=mp.begin();it!=mp.end();it++)
	{
		b[cnt++]=it->second;
	}
	sort(b,b+cnt);
	int minm=0,maxm=0;
	for(i=0;i<cnt;i++)
	{
		minm+=a[i]*b[cnt-1-i];
		maxm+=a[n-1-i]*b[cnt-1-i];
	}
	cout<<minm<<" "<<maxm<<endl;
	return 0;
}
