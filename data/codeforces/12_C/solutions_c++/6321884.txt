#include <iostream>
#include <cstdio>
#include <cstring>
#include <map>
#include <string>
#include <algorithm>
using namespace std;

int a[110],b[110];

int main()
{
	int n,m;
	cin>>n>>m;
	map<string,int> M;
	for(int i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	for(int i=0;i<m;i++){
		string s;
		cin>>s;
		if(!M[s]) M[s] = 1;
		else M[s] ++;
	}
	int num = 0;
	map<string,int>::iterator it;
	for(it = M.begin(); it!=M.end();it++){
		b[num++] = it->second;
	}
	sort(b,b+num);
	int Max = 0,Min = 0;
	for(int i=0;i<num;i++){
		Min += a[i]*b[num-1-i];
		Max += b[num-1-i]*a[n-1-i];
	}
	cout<<Min<<" "<<Max<<endl;
	return 0;
}
