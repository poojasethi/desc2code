#include<iostream>
#include<map>
#include<algorithm>
#include<string>
#define fe(i,a) for(__typeof(a.begin()) i=a.begin(); i!=a.end(); i++)
using namespace std;
int n,m,a[110],b[110],tot;
map<string,int> h;

int main()
{
	cin>>n>>m;
	for (int i=0; i<n; i++) cin>>a[i];
	sort(a,a+n);
	for (int i=0; i<m; i++) 
	{
		string s;
		cin>>s;
		if (h.count(s)) h[s]++; else h[s]=1;
	}
	int a1(0), a2(0);
	fe(i,h) 
	{
		b[tot++]=i->second;
	}
	sort(b,b+tot);
	for (int i=tot-1, j=0; i>=0; i--, j++)
		a1+=b[i]*a[j];
	for (int i=tot-1, j=n-1; i>=0; i--, j--)
		a2+=b[i]*a[j];
	cout<<a1<<" "<<a2<<endl;
}
