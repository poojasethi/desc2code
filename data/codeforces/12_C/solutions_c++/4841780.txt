#include <iostream>
#include <map>
#include <algorithm>
using namespace std;

int n, m, a[101], b[101], bcount;
string name;
map<string, int> f;

int main()
{
	cin >> n >> m;
	for(int  i = 0; i < n; i++)	cin >> a[i];
	for(int i = 0; i < m; i++)
	{
		cin >> name;
		f[name]++;
	}
	for(map<string, int>::iterator it = f.begin(); it != f.end(); it++)
	{
		b[bcount++] = it->second;
	}
	sort(a, a + n);
	sort(b, b + bcount);

	int sum1 = 0, sum2 = 0;
	for(int i  = 0; i < bcount; ++i)
	{
		sum1 += a[i] * b[bcount - 1 - i];
		sum2 += a[n- 1 - i] * b[bcount - 1 - i];
	}
	cout << sum1 << " " << sum2 << endl;
	return 0;
}
