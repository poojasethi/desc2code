//In the Name of God

#include<iostream>
#include<algorithm>
#include<vector>

using namespace std;

int n,m,p[110],c,mini,Maxi;
string s[110];
vector<int> v;
 
int main()
{
  cin>>m>>n;
  for(int i=0;i<m;i++)
    cin>>p[i];
  for(int i=0;i<n;i++)
    cin>>s[i];
  sort(p,p+m);
  sort(s,s+n);
  c=1;
  for(int i=1;i<n;i++)
    if(s[i]==s[i-1])
      c++;
    else
      {
	v.push_back(c);
	c=1;
      }
  v.push_back(c);
  sort(v.begin(),v.end());
  for(int i=0;i<v.size();i++)
    mini+=v[v.size()-i-1]*p[i],Maxi+=v[v.size()-i-1]*p[m-i-1];
  cout<<mini<<" "<<Maxi<<endl;
  return 0;
}
