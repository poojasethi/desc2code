#include <iostream>
#include <map>
#include <string>
#include <algorithm>
using namespace std;
int A[400],S[300];
int N,M;
map <string,int> m;
string K[300];
int main()
{
	string str;
	int k=0;
	cin >> N >> M ;
	for(int i=1;i<=N;i++)
		cin >> A[i];
	sort(A+1,A+N+1);
	for(int i=1;i<=M;i++)
	{
		cin>>str;
		if(!m[str]) K[++k]=str;
		m[str]++;
	}
	for(int i=1;i<=k;i++)
		S[i]=m[K[i]];
	sort(S+1,S+1+k);
	reverse(S+1,S+1+k);
	int sum=0;
	for(int i=1;i<=k;i++)
		sum+=S[i]*A[i];
	cout << sum << " ";
	sum=0;
	reverse(A+1,A+1+N);
	for(int i=1;i<=k;i++)
		sum+=S[i]*A[i];
	cout << sum << endl;
	return 0;
}
