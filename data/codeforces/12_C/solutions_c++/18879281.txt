#include <bits/stdc++.h>
using namespace std;
map<string,int>mp;
int a[110],b[110];
bool cmp(int a,int b){
	return a>b;
}
int main() {
	int n,m;
	cin>>n>>m;
	string s;
	for(int i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	for(int i=0;i<m;i++){
		cin>>s;
		mp[s]++;
	}
	int cnt=0;
	for(map<string,int>::iterator it=mp.begin();it!=mp.end();it++)
		b[cnt++]=(*it).second;
	sort(b,b+n,cmp);
	int mn=0,mx=0;
	for(int i=0;i<m;i++) mn+=b[i]*a[i];
	for(int i=0;i<m;i++) mx+=b[i]*a[n-i-1];
	cout<<mn<<" "<<mx<<endl;
	return 0;
}
