#include<bits/stdc++.h>

using namespace std;

int i,j,k,price[110],food[110],minm,maxm;
map<string,int>mp;
set<string> s;
string temp;

int main(void)
{
	int n,m;
	cin>>n>>m;
	for(i=0;i<n;i++)cin>>price[i];
	sort(price,price+n);
	for(i=0;i<m;i++)
	{
		cin>>temp;
		s.insert(temp);
		mp[temp]++;
	}
	set<string>::iterator it;
	int size=s.size();
	for(it=s.begin();it!=s.end();it++)
	{
		food[j]=mp[*it];
		j++;
	}
	sort(food,food+j,greater<int>());
	
	for(i=0;i<size;i++)
	{
		minm+=food[i]*price[i];
	}
	for(i=0;i<size;i++)
	{
		maxm+=food[i]*price[n-1-i];
	}
	cout<<minm<<' '<<maxm;
	return 0;
}
