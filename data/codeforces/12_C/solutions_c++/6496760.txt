#include<cstdio>
#include<string>
#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
    int i,m,n,p[1000],k=0,U[1000],mini=0,i1,maxi=0;
    string fr;
    map<string,int>o;
    scanf("%d%d",&n,&m);
    for(i=0;i<n;i++)
        scanf("%d",&p[i]);
    sort(p,p+i);
    while(m--){cin>>fr;o[fr]++;}
    map<string,int>::iterator j;
    for(j=o.begin();j!=o.end();j++)
        U[k++]=(*j).second;
    sort(U,U+k);
    for(i=k-1,i1=0;i>=0,i1<k;i--,i1++)
        mini+=U[i]*p[i1];
    for(i=k-1,i1=n-1;i>=0;i--,i1--)
        maxi+=U[i]*p[i1];
    printf("%d %d\n",mini,maxi);
    
}
