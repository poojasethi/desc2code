#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<map>
using namespace std;
int n,m,sz=1,ar[101],ans,ans2,b[101];
string a[101];
bool cmp(int a,int b){
  return a>b;
}
map<string,int> d,used;
int main(){
  scanf("%d%d",&n,&m);
  for(int i=1;i<=n;++i)
    scanf("%d",&b[i]);
  for(int i=1;i<=m;++i){
    cin>>a[i];
    d[a[i]]++;
  }
  for(int i=1;i<=m;++i)
    if(used[a[i]]==0){
      used[a[i]]=1;
      ar[sz++]=d[a[i]];
    }
  sort(b+1,b+n+1);
  sort(ar+1,ar+sz,cmp);
  for(int i=1;i<=sz;++i){
    ans+=ar[i]*b[i];
    ans2+=ar[i]*b[n--];
  }
  printf("%d %d",ans,ans2);
}
