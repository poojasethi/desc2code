#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
    int a,b;
    string g;
    cin>>a>>b;
    map<string,int>d;
    int v[150],v1[150];
    for(int i=0;i<a;i++) cin>>v[i];
    for(int i=0;i<b;i++) {cin>>g;d[g]++;}
    sort(v,v+a);
    map<string,int>::iterator it=d.begin();
    int r=0;
    for(;it!=d.end();it++) {v1[r]=(it->second); r++;}
    sort(v1,v1+r);
    //for(int i=0;i<r;i++) cout<<v1[i]<<" "<<endl;
    int m=0,M=0;
    for(int i=0;i<r;i++) m+=v[i]*v1[r-1-i];
    for(int i=0;i<r;i++) M+=v[a-i-1]*v1[r-i-1];
    cout<<m<<" "<<M<<endl;
}
