#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

int main(void)
{
  map<string,int> list;
  ll n,m;
  ll N[106];
  cin >> n >> m;
  for(int i=0;i<n;i++)
  {
    cin >> N[i];
  }
  string S;
  for(int i=0;i<m;i++)
  {
    cin >> S;
    list[S]++;
  }
  ll F[106];
  int i=0;
  for(map<string,int>::iterator iter=list.begin();iter!=list.end();iter++)
  {
    F[i++]=iter->second;
  }

  sort(N,N+n);

  sort(F,F+i);

  int max=0,min=0;
  for(int x=0;x<i;x++)
  {
    min+=N[x]*F[i-x-1];
    max+=N[n-x-1]*F[i-x-1];
  }
  cout << min<<" "<< max <<endl;
  return 0;
}
