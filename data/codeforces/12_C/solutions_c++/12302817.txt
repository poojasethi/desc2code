#include <bits/stdc++.h>
using namespace std;

int main()
{
int n,m,mn=0,mx=0,x;
string s;
map<string,int> v;
vector<int> vv;
cin >> n >> m;
vector<int> price;
for(int i=0;i<n;i++) {cin >> x;price.push_back(x);}
sort(price.begin(),price.end());
for(int i=0;i<m;i++)
{
    cin >> s ;
    v[s]++;
}
for (map<string,int>::iterator it = v.begin(); it != v.end(); ++it)
{
    vv.push_back(it->second);
}
sort(vv.rbegin(),vv.rend());
for(int i=0;i<vv.size();i++) mn+=price[i]*vv[i];
sort(price.rbegin(),price.rend());
for(int i=0;i<vv.size();i++) mx+=price[i]*vv[i];
cout << mn << " " << mx;




}
