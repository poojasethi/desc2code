#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,m;
	string s;
	map<string,int> mp;
	cin >> n >> m;
	vector<int> prices(n);
	while(n--)
		cin >> prices[n];
	while(m--)
	{
		cin >> s;
		mp[s]++;
	}
	vector<int> qty;
	for(map<string,int>::iterator it=mp.begin();it!=mp.end();++it)
		qty.push_back(it->second);
	sort(qty.begin(),qty.end());
	sort(prices.begin(),prices.end());
	int a,b;
	a = b = 0;
	for(int i=qty.size()-1,j=0;i>=0;--i,++j)
		a += qty[i]*prices[j];
	for(int i=qty.size()-1,j=prices.size()-1;i>=0;--i,--j)
		b += qty[i]*prices[j];
	cout << a << " " << b;
	return 0;
}
