#include <bits/stdc++.h>
#define MIN 0
#define MAX 500005
using namespace std;
map< string,int > m;
string s[105];
int a[105];
int main()
{
    int n,m,i,mn=0,mx=0;
    cin>>n>>m;
    for(i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(i=0;i<m;i++)
    {
        cin>>s[i];
    }
    sort(s,s+m);
    int j=0,c[n];
    memset(c,0,sizeof(c));
    for(i=0;i<m;)
    {
        int e=i;
        while(e<m&&s[i]==s[e])
            e++;
        c[j++]=e-i;
        i=e;
    }
    sort(c,c+n);
    for(i=0;i<n;i++)
    {
        mx+=c[i]*a[i];
        mn+=c[n-1-i]*a[i];
    }
    cout<<mn<<" "<<mx;
    return 0;
}
