#include<iostream>
#include<algorithm>
#include<stdio.h>
#include<map>
using namespace std;

int main()
{//freopen("i.txt","r",stdin);
int n,m; cin>>n>>m;
int c[101]={},kol[101]={};
string s,name[101];
for (int i=0;i<n;i++) cin>>c[i];
sort(c,c+n);
reverse(c,c+n);
int j=0;
for (int i=0;i<m;i++)
 {cin>>s;
 int k=0;
 for (;k<i;k++)  if (s==name[k]) {kol[k]++;break;};
 if (k==i) {name[j]=s; kol[j]=1; j++;}
 }
sort(kol,kol+j);
reverse(kol,kol+j);
int ans=0, ans1=0;
for (int i=0;i<j;i++)
 {ans+=kol[i]*c[i];
  ans1+=kol[i]*c[n-i-1];
 }
cout<<ans1<<" "<<ans;
return 0;
}
