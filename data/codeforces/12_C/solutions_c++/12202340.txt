#include<iostream>
#include<map>
#include<vector>
#include<algorithm>
#define f(i,a,b) for(int i=a;i<b;i++)
using namespace std;
int main(){
	vector<int>a;
	int n,m,t;
	cin>>n>>m;
	f(i,0,n){
		cin>>t;
		a.push_back(t);
	}
	string s;
	map<string,int>mp;
	f(i,0,m){
		cin>>s;
		mp[s]++;
	}
	vector<int>num;
	map< string , int > :: iterator it;
	for(it=mp.begin();it!=mp.end();it++)
		num.push_back(it->second);
	sort(num.begin(),num.end());
	sort(a.begin(),a.end());
	int mi,ma;
	mi=ma=0;
	int k=num.size();
	f(i,0,k){
		mi+=num[k-1-i]*a[i];
		ma+=num[k-1-i]*a[a.size()-1-i];
	}
	cout<<mi<<" "<<ma<<endl;
	return 0;
}
