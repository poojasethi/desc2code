#include<bits/stdc++.h>
#include<map>
using namespace std;

int main()
{
   int n,m,a[101],b[101];
   string s;
   map<string ,int> mp;
   map<string ,int> ::iterator it;
   cin>>n>>m;
   for(int i=0;i<n;i++) cin>>a[i];
   for(int i=0;i<m;i++) {cin>>s; mp[s]++;}
   int d=0;
   for(it=mp.begin();it!=mp.end();++it) b[d++]=it->second;
   sort(a,a+n);
   sort(b,b+d);
   int mn=0,mx=0;
   for(int i=0,j=d-1;j>=0;i++,j--)
   mn+=a[i]*b[j];
   for(int j=d-1,i=n-1;j>=0;j--,i--)
   mx+=a[i]*b[j];
   cout<<mn<<" "<<mx<<endl;
    return 0;
}
