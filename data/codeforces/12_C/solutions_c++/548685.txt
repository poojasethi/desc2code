#include <iostream>
#include <algorithm>
#include <map>

using namespace std;

int N,M,Min,Max;
int A[150],Me[150],syc;
map<string,int> mp;



void Solve()
{
	syc=1;
	for(int i=M;i>=1;i--) Min+=Me[i]*A[syc++];
	syc=N;
	for(int i=M;i>=1;i--) Max+=Me[i]*A[syc--];
	cout<<Min<<" "<<Max<<endl;
}

void Read()
{
	syc=1;
	string s;
	cin >> N >> M;
	for(int i=1;i<=N;i++) cin >> A[i];
	for(int i=1;i<=M;i++)
	{
		cin >> s;
		if(!mp[s])
		{
			mp[s]=syc;
			Me[syc]++;
			syc++;
		}
		else Me[mp[s]]++;
	}
	sort(A+1,A+N+1);
	M=syc-1;
	sort(Me+1,Me+M+1);
}

int main()
{
	Read();
	Solve();
}
