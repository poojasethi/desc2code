#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<iostream>
#include<algorithm>
using namespace std;
struct T
{
    int i;
    char a[105];
}f[105];
int cmp(T a,T b)
{
  return a.i>b.i;
}
int main()
{
 int n,m,i,j;
 char c[105];
 int b[105];

 scanf("%d%d",&n,&m);
for(i=0;i<n;i++)
scanf("%d",&b[i]);
for(i=0;i<m;i++)
{
    f[i].i=0;
scanf("%s",c);
for(j=0;j<i;j++)
if(!strcmp(c,f[j].a))break;
f[j].i++;
if(j==i) strcpy(f[i].a,c);
}
sort(b,b+n);
sort(f,f+m,cmp);
int sum1=0,sum2=0;
for(i=0;f[i].i;i++)
{
    sum1+=b[i]*f[i].i;
    sum2+=b[n-1-i]*f[i].i;
}
printf("%d %d\n" ,sum1,sum2);




return 0;
}
