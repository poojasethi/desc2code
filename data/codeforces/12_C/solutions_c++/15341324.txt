#include<bits/stdc++.h>
using namespace std;
vector<int>v;
map<string,int>m;
map<string,int>::iterator it;
int a[100];
int main()
{
    int n,p;
    string s;
    cin>>n>>p;
    for(int i=0;i<n;i++)
        cin>>a[i];
    for(int i=0;i<p;i++)
    {
        cin>>s;
        m[s]++;
    }
    for(it=m.begin();it!=m.end();it++)
        v.push_back(it->second);
    sort(v.begin(),v.end());
    sort(a,a+n);
    int j=0;
    int i=n-1;
    int MIN=0,MAX=0;
    while(j<v.size())
    {
        MIN+=v[v.size()-j-1]*a[j];
        MAX+=v[v.size()-j-1]*a[i];
        i--,j++;
    }
    cout<<MIN<<" "<<MAX;
}
