#include <iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<map>
using namespace std;
map<string,int >mp;
string s[111];
int a[111],b[111];
int main()
{
	int i,j,m,n,k,t,mn,mx;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	getchar();
	mp.clear();
	sort(a,a+n);
	for(i=0;i<m;i++)
	{
		cin>>s[i];
		mp[s[i]]++;
	}
	for(t=i=0;i<m;i++)
		if(mp[s[i]])
		{
			b[t++]=mp[s[i]];
			mp[s[i]]=0;
		}
	sort(b,b+t);
	for(i=t-1,j=mn=0;i>=0;i--,j++)
		mn+=a[j]*b[i];
	for(i=t-1,j=n-1,mx=0;i>=0;i--,j--)
		mx+=a[j]*b[i];
	printf("%d %d\n",mn,mx);
}
