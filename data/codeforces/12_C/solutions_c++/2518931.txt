#include<string>
#include<iostream>
#include<algorithm>
#include<map>
using namespace std;
int main()
{
int n,m;
map<string,int> ma;
cin>>n>>m;

int p[n];
for(int i=0;i<n;i++) {
cin>>p[i];
}
sort(p,p+n);
int s1=0,s2=0;
for(int i = 0;i<m;i++ ){
string temp;
cin>>temp;
ma[temp]++;
}
int y=0,z=n-1;
map<string,int>::iterator it,temp;
int max=0,t=0;

for(int i=0;i<ma.size();) {
    max=0;
for( it=ma.begin();it!=ma.end();it++) {
if((*it).second>max) {
max=(*it).second;
temp=it;
}
}
s1=s1+(max)*(p[y]);
y++;
s2=s2+(max)*(p[z]);
z--;
ma.erase(temp);
}
cout<<s1<<" "<<s2;
}
