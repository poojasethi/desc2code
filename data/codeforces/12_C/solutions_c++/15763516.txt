#include <iostream>
#include <map>
#include <algorithm>

using namespace std;

const int MAXN = 1000;
int cnt[MAXN];
int arr[MAXN];
map <string, int> mp;

int32_t main()
{
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> arr[i];
	sort(arr, arr + n);
	int pt = 0;
	for (int i = 0; i < m; i++)
	{
		string s;
		cin >> s;
		if (mp.find(s) == mp.end())
			mp[s] = pt++;
		cnt[mp[s]]++;
	}
	sort(cnt, cnt + pt);
	int sum = 0;
	for (int i = pt - 1; i >= 0; i--)
		sum += arr[pt - i - 1] * cnt[i];
	cout << sum << ' ';
	sum = 0;
	for (int i = pt - 1; i >= 0; i--)
		sum += arr[n - pt + i] * cnt[i];
	cout << sum << endl;
}
