#include<cstdio>
#include<cstring>
#include<map>
#include<string>
#include<algorithm>
using namespace std;
map<string,int>mp;
int a[101],b[101];
bool cmp(int x,int y){
	return x>y;
}
int main(){
	int n,m;
	char s[33];
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n);
	while(m--){
		scanf("%s",s);
		mp[s]++;
	}
	int lb=0;
	for(map<string,int>::iterator it=mp.begin();it!=mp.end();++it)
		b[lb++]=it->second;
	sort(b,b+lb,cmp);
	int ans1=0,ans2=0;
	for(int i=0;i<lb;i++)
		ans1+=a[i]*b[i];
	sort(a,a+n,cmp);
	for(int i=0;i<lb;i++)
		ans2+=a[i]*b[i];
	printf("%d %d",ans1, ans2);
	return 0;
}
