 #include<iostream>
#include<algorithm>
#include<cmath>
#include<map>
#include<string>
using namespace std;
int n,m,k,l,r,a[1001],ma,mi;
string s,t[101],q,w,e;
map<string,int>v;
int main()
{
    cin>>n>>m;
    for(int i=1;i<=n;i++)
    cin>>a[i];
    sort(a+1,a+n+1);
    for(int i=1;i<=m;i++)
    {
            cin>>s,v[s]++;
            if(v[s]==1)
            k++,t[k]=s;
    }
    for(int i=1;i<=m;i++)
    for(int j=1;j<m;j++)
    if(v[t[j]]<v[t[j+1]])
    swap(t[j],t[j+1]);
    for(int i=1;i<=m;i++)
    mi+=a[i]*v[t[i]];
    for(int i=1;i<=m;i++)
    ma+=a[n-i+1]*v[t[i]];
    cout<<mi<<" "<<ma;
}
