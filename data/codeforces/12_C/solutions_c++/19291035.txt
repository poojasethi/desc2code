#include<cstdio>
#include<cstring>
#include<algorithm>
#include<string>
#include<map>
using namespace std;
int main()
{
	int n,m,cnt=1;
	int p[105]={0},count[105]={0};
	map<string,int> num;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;++i)
	{
		scanf("%d",&p[i]);
	}
	sort(p,p+n);
	char f[105]={0};string s;
	for(int i=0;i<m;++i)
	{
		scanf("%s",f);
		s=f;
		if(!num[s])
		{
			num[s]=cnt++;
		}
		++count[num[s]];
	}
	sort(count+1,count+cnt);
	int minans=0,maxans=0;
	for(int i=cnt,j=0;i>=1;--i,++j)
	{
		minans+=count[i-1]*p[j];
		maxans+=count[i-1]*p[n-j-1];
	}
	printf("%d %d\n",minans,maxans);
	return 0;
}
