A=raw_input()
B=raw_input()
n=len(B)
A='#'+A
B='#'+B
a=[False]*(n+5)
b=[False]*(n+5)
a[0]=True
for i in xrange(1,n+1): a[i]=a[i-1] and (A[i]==B[i])
b[n+1]=True
for i in reversed(xrange(1,n+1)): b[i]=b[i+1] and (A[i+1]==B[i])
s=filter(lambda i:a[i-1] and b[i],xrange(1,n+2))
print len(s)
if s:print ' '.join(map(str,s))
