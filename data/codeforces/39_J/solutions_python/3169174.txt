s1 = raw_input()
s2 = raw_input()
n, m = len(s1), len(s2)
l, r = 0, n - 1
for l in xrange(1, m + 1):
  if s1[l - 1] != s2[l - 1]:
    l -= 1
    break
for r in xrange(1, m + 1):
  if s1[n - r] != s2[m - r]:
    r -= 1
    break
if l + r < m:
  print 0
else:
  if l + r == m:
    r = range(l + 1, n - r + 1)
  else:
    r = range(n - r, l + 2)
  print len(r)
  print ' '.join(map(str, r))
