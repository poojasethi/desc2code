#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;

#define MAX 1000003

char a[MAX], b[MAX];

int main() {
	// your code goes here
	int na, nb, left, right = 0;
	scanf("%s %s", a, b);
	na = strlen(a);
	nb = strlen(b);
	left = na;
	int i = 0;
	while(a[i] == b[i] && i < nb) i++;
	right = i;
	i = na - 1;
	while(a[i] == b[i - 1] && i > 0) i--;
	left = i;
	if (right < left)
		printf("0");
	else
	{
		printf ("%d \n", right - left + 1);
		for (int j = left; j <= right; j++)
			printf("%d ", j + 1);
	}
	return 0;
}
