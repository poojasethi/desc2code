#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
  string s,t;cin >> s >> t;

  int posf = 0;
  while(posf < (int) t.size() && s[posf] == t[posf]) posf++;

  int posb = t.size() - 1;
  while(posb >= 0 && s[posb+1] == t[posb]) posb--;
  //  cout << posf << " " << posb << endl;

  if(posf - posb < 0 ){
    cout << 0 << endl;
  }else {
    cout << posf - posb <<endl;
    for(int i = posb + 1 ;i <= posf ;i++){
      cout << i + 1 << " " ;
    }
  }

  return 0;

}
