#include<iostream>
#include<string>
using namespace std;

string a,b;

int main()
{
	cin>>a>>b;
	int n=(int)a.size();
	int m=(int)b.size();
	int f=n-1;
	for(int i=0;i<m;i++)
		if(a[i]!=b[i])
		{
			f=i;
			break;
		}
	if(b==a.substr(0,f)+a.substr(f+1,n-(f+1)))
	{
		int t=f;
		while(t>0 && a[t]==a[t-1])
			t--;
		cout<<f-t+1<<endl;
		for(int i=t;i<=f;i++)
			cout<<i+1<<' ';
		cout<<endl;
	}
	else
		cout<<0<<endl;
	return 0;
}
