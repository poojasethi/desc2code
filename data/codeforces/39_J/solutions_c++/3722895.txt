#include <cstdio>
#include <cstring>
using namespace std;
#define N 1000004
char s[N], ss[N];
int main() {
    scanf(" %s %s", s, ss);
    int l, r;
    for (l=0; s[l]==ss[l]; l++);
    for (r=strlen(s)-1; r && s[r]==ss[r-1]; r--);
    printf("%d\n", l>=r? (l-r+1) : 0);
    for (int i=r; i<=l; i++) printf("%d ", i+1);

    return 0;
}
