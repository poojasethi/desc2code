#include<iostream>
#include<string>
using namespace std;
int main()
{
	string a,b;
	cin>>a>>b;
	int l,r,n = b.length();
	
	l = 0;
	while(l < n && a[l] == b[l]) l++;
	l--;
	
	r = n - 1;
	while(r >= 0 && a[r + 1] == b[r]) r--;
	r++;

	if (r - l > 1) cout<<0; else
	{
		cout<<l - r + 2<<endl;
		for(int i = r; i <= l + 1; i++)
			cout<<i + 1<<" ";
	}
	return 0;
}
