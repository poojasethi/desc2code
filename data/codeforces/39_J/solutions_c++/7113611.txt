#include<cstdio>
#include<cstring>
#include<algorithm>

using namespace std;

const int MAX_N = 1000007;

char s1[MAX_N], s2[MAX_N];
int l1, l2, ans, t1, t2;

int main() {
	scanf("%s %s", s1, s2);
	l1 = strlen(s1); l2 = strlen(s2);
	t1 = 0; t2 = l2;
	while (s1[t1] == s2[t1]) ++t1;
	while (s1[t2 + 1] == s2[t2] && t2 >= 0) --t2;
	printf("%d\n", t1 > t2 ? t1 - t2 : 0);
	for (int i = t2 + 1; i <= t1; ++i) 
	  printf("%d ", i + 1);
	puts("");
	return 0;
}
