#include <iostream>
#include <cstring>
#include <vector>
using namespace std;
#define P 31
#define ull long long
ull h[1000005],hesh;
vector <int> ans;
string s;
int main()
{
  cin>>s;
  int len=s.size();
  ull p=P;
  h[0]=s[0]-'a'+1;
  for(int i=1;i<len;i++,p*=P)h[i]=h[i-1]+(s[i]-'a'+1)*p;
  p=1;
  cin>>s;
  for(int i=0;i<len-1;i++,p*=P)hesh+=(s[i]-'a'+1)*p;
  for(int i=0;i<len;i++)
    if ((i?h[i-1]:0)*P+(h[len-1]-h[i])==hesh*P)ans.push_back(i+1);
  cout<<ans.size()<<"\n";
  for(int i=0;i<ans.size();i++)cout<<ans[i]<<" ";
  return 0;
}
