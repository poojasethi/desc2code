#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<queue>
#include<map>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;

int n, m, l, r;
string s, t;
int main() {
	cin >> s >> t;
	n = s.length();
	m = n - 1;
	for (l = 0; l < m && s[l] == t[l]; l++);
	for (r = 0; r < m && s[n - 1 - r] == t[m - 1 - r]; r++);
	if (l + r < m ) {
		puts("0");
		return 0;
	}
	r = n - r;
	cout << l - r + 2 << endl;
	for (int i = r - 1; i <= l; i++)
		cout << i+1 << ' ';
	cout << endl;
}
