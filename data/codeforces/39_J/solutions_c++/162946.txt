#include<stdio.h>
#include<string.h>
#define N 1000007

char a[N], s[N];
int main(void)
{
  scanf("%s", a);
  scanf("%s", s);
  int n = strlen(a);
  int p1 = 0, p2 = n;
  while(p1 < n-1 && a[p1] == s[p1])
    p1++;
  while( p2 > 0 && a[p2] == s[p2 - 1])
    p2--;
  if(p1 < p2)
    printf("0\n");
  else{
    while(p2 > 0 && a[p2] == a[p2-1])
      p2--;
    while(p1 < n && a[p1] == a[p1+1])
      p1++;
    printf("%d\n", p1-p2+1);
    for(int i = p2; i <= p1; i++)
      printf("%d ", i + 1);
  }

  return 0;
}
