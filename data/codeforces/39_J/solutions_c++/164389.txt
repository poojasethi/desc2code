#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

#define MAXN 1000001

char s1[MAXN];
char s2[MAXN];

int main()
{
    int i,j;
    cin>>s1>>s2;
    int len=strlen(s2); 
    for(i=0;s1[i]==s2[i];i++);
    for(j=len;s1[j]==s2[j-1] && j>0 ;j--);
    if(i>=j)
    {
        cout<<i-j+1<<endl;
        for(;j<=i;j++)
            cout<<j+1<<' ';
    }
    else cout<<0;
    cout<<endl;
    return 0;
}
