#include <iostream>
#include <algorithm>
#include <vector>
#include <string>

using namespace std;

int main() {
  string a, b;
  cin >> a >> b;
  int x, y;
  for (x = 0; x < b.size(); x++)
    if (a[x] != b[x]) break;
  for (y = b.size()-1; y >= 0; y--)
    if (a[y+1] != b[y]) break;
  //cout << x << ' '  << y << endl;
  if (x < y) {
    cout << 0 << endl;
  } else {
    cout << x - y << endl;
    while (y++ < x)
      cout << y+1 << ' ';
  }
  return 0;
}
