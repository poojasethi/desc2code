#include<cstdio>
#include<cstring>
using namespace std;
char s1[1000010],s2[1000010];
int main()
{
	int i,l,ans=0,j,p;
	gets(s1);
	gets(s2);
	l=strlen(s1);
	for(i=0;i<l;++i)
	{
		if(s1[i]!=s2[i])
		{
			for(p=i;p&&s1[p-1]==s1[p];--p);
			for(j=i+1;s1[j]==s1[j-1];++j);
			ans=j-p;
			break;
		}
	}
	for(;i<l;++i)
	{
		if(s1[i+1]!=s2[i])
		{
			puts("0");
			return 0;
		}
	}
	printf("%d\n%d",ans,++p);
	for(;p<j;++p)
	{
		printf(" %d",p+1);
	}
	return 0;
}
