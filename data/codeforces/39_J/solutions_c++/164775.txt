#include <cstdlib>
#include <iostream>
#include <string.h>
using namespace std;

int main(int argc, char *argv[])
{
    int i,l,r,n;
    char a1[1000000], a2[1000000];
    cin>>a1>>a2;
    n=strlen(a2);
    l=0;
    while((l<n)&&(a1[l]==a2[l])) l++;
    l--;
    r=n-1;
    while((r>=0)&&(a1[r+1]==a2[r])) r--;
    r++;
    
    if (r - l > 1) cout<<0; else
        {
                cout<<l - r + 2<<endl;
                for(i = r; i <= l + 1; i++)
                        cout<<i + 1<<" ";
        }
    //system("PAUSE");
    return EXIT_SUCCESS;
}
