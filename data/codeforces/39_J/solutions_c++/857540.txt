#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define N 1000010
char a[N],b[N];
using namespace std;
int main()
{
        cin >> a >> b;
        int n=strlen(a);
        int L=0,R=n-1;
        while ( a[L]==b[L] ) L++;
        while ( R>0 && a[R]==b[R-1] ) R--;
        if ( L<R ) puts("0");
        else {
                cout << L-R+1 << '\n';
                for ( int i=R; i<=L; i++ ) printf("%d%c",i+1,i==L?'\n':' ');
        }
        return 0;
}
