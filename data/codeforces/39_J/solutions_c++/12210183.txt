#include<cstdio>
#include<cstring>
#define N 1000010
char s1[N],s2[N];
int len1,len2,f[N],g[N],ans,o[N],i;
int main()
{
    scanf("%s",s1);
    scanf("%s",s2);
    len1=strlen(s1);
    len2=strlen(s2);
    f[0]=1;g[len2+1]=1;
    for (i=1;i<=len2;i++)
    if (s1[i-1]==s2[i-1])
         f[i]=(f[i-1]&1);       
     
    for (i=len2;i>=1;i--)
    if (s1[i]==s2[i-1])
         g[i]=(g[i+1]&1);
    
    for (i=0;i<=len2;i++)
    if (f[i]&&g[i+1]) 
    {
       ans++;
       o[ans]=i+1;
    }
    printf("%d\n",ans);
    for (i=1;i<=ans;i++)
    printf("%d ",o[i]);
}
