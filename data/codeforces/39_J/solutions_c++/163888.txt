#include<cstdio>
#include<string.h>
#define MAXN 1100000
using namespace std;
char c1[MAXN],c2[MAXN];
int i,j;
int main()
{
    scanf("%s%s",c1,c2);
    int len=strlen(c2);
    for(;c1[i]==c2[i];i++);
    for(j=len;j>0&&c1[j]==c2[j-1];j--);
    if(i-j>=0)
    {
        printf("%d\n",i-j+1);
        for(;j<i+1;j++)
            printf("%d ",j+1);
    }
    else
        printf("0\n");
}
