#include <stdio.h>
#include <string.h>
char a[1000005], b[1000005];
int v[1000005];
int la, lb;
int main() {
    scanf("%s%s", a, b);
    la = strlen(a);
    lb = strlen(b);
    int i, j;
    for(i = 0; i != lb && a[i] == b[i]; ++i); 
    for(j = 0; j != lb && a[la - j - 1] == b[lb - j - 1]; ++j);
    j = la - j - 1;
    if(i < j) puts("0");
    else {
        printf("%d\n", i - j + 1);
        for(;j <= i; ++j) printf("%d ", j + 1);
    }
    return 0;
}
