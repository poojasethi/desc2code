#include<bits/stdc++.h>
#define pb push_back
using namespace std;

string a,b;
int se[1000005],sb[1000005];
vector<int> out;

int main()
{
    int i,j,k,n;
    cin>>a>>b;
    n=a.size();
    se[n]=1;
    for(i=n-1;i>0;i--) se[i]=se[i+1] && a[i]==b[i-1];
    sb[0]=a[0]==b[0];
    for(i=1;i<n-1;i++) sb[i]=sb[i-1] && a[i]==b[i];
    for(i=0;i<a.size();i++)
    {
        if(i==0 && se[i+1]) out.pb(i+1);
        else if(sb[i-1] && se[i+1]) out.pb(i+1);
    }
    cout<<out.size()<<endl;
    for(i=0;i<out.size();i++) cout<<out[i]<<" ";
    return 0;
}
