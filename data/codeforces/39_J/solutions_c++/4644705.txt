#include <iostream>
#include <string>
using namespace std;

int main() {
  string a, b;
  cin >> a >> b;
  int i, j, k = b.length();

  for (i = 0; i < k && a[i] == b[i]; i++);
  i--;
    
  for (j  = k - 1; j >= 0 && a[j + 1] == b[j]; j--);
  j++;
    
  if (j - i > 1) {
    cout << 0;
  } else {
    cout << i - j + 2 << endl;

    for (int p = j; p <= i + 1; p++)
      cout << p + 1 << endl;
  }

  return 0;
}
