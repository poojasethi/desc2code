#include<iostream>
#include<vector>
#include<string>
using namespace std;


string sres, s;
int p1, p2, res;


int main()
{
   cin >> s >> sres;
   int i = 0;
   while(s[i] == sres[i]  && i < sres.length())
   {
       p1++;
       i++;
   }
   i = sres.length() - 1;
   while(s[i + 1] == sres[i]  && i >= 0)
   {
       p2++;
       i--;
   }
   res = (p1 + p2) - sres.length() + 1;
   if(res < 0) res = 0;
   cout << res;
   if(res)
   {
       cout << endl;
       for(int i = sres.length() - p2 + 1; i <= p1 + 1; i++)
           cout << i << ' ';
   }
   return 0;
}
