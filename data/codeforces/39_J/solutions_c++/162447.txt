// xabcdefgggghw
// yabcdefggggh
// q      p    q
#include<cstdio>

char A[1000005], B[1000005];

int main(void)
{
    int n, p, q;
    scanf("%s", A);
    scanf("%s", B);
    
    q=0; while( A[q]==B[q] ) q++;
    for(n=q+1; A[n] && A[n]==B[n-1]; n++);
    if (A[n]) { printf("0\n"); return 0; }
    
    p=q; while( p-1>=0 && A[p-1]==A[p] ) p--;
    printf("%d\n", q-p+1);
    while (p<=q) printf("%d ", ++p);
    return 0;
}
