#include <iostream>
#include <string>

using namespace std;

int main()
{
    string s1,s2;
    cin>>s1>>s2;
    int i=0;
    int l=s2.length();
    while(i<l&&s1[i]==s2[i])i++;
    int start=i;
    while(start>=0&&s1[start]==s1[i])start--;
    start++;
    int end=i;
    while(end<l+1&&s1[end]==s1[i])end++;
    end--;
    while(i<l)
        if(s1[i+1]!=s2[i])
        {
            cout<<0;
            return 0;
        }
        else
            i++;
    cout<<end-start+1<<endl;
    end++;
    start++;
    for(int i=start;i<=end;i++)
        cout<<i<<' ';
    return 0;
}
