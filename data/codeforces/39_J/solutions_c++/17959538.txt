#include <bits/stdc++.h>
using namespace std;
#define N 1001001

vector <int> v;
char s[N], t[N];

int main() {
//	freopen("j.in", "r", stdin);
	scanf("%s %s", s, t);
	int len = strlen(s), a = -1, b = len;
	for (int i = 0; i < len - 1; i ++) {
		if (s[i] == t[i]) a = i;
		else break;
	}
	for (int i = len - 1; i > 0; i --) {
		if (s[i] == t[i-1]) b = i;
		else break;
	}
	for (int i = 0; i < len; i ++) if (a >= i - 1 && b <= i + 1) v.push_back(i);
	printf("%d\n", v.size());
	for (int i = 0; i < v.size(); i ++) printf("%d ", v[i] + 1); puts("");
	return 0;
}
