#include <cstdio>
#include <vector>
#include <cstring>
using namespace std;

const int N = 1000000 + 10;
char word[N], dic[N];
bool issame[N];
vector<int> ret;
int main()
{
	scanf("%s%s", word, dic);
	int len = strlen(dic);
	issame[0] = true;
	for (int i = 0; i < len; i++)
	{
		issame[i + 1] = issame[i] && (word[i] == dic[i]);
	}
	for (int i = len; i >= 0; i--)
	{
		if (issame[i])
		{
			ret.push_back(i + 1);
		}
		if (i > 0 && word[i] != dic[i - 1])
			break;
	}
	printf("%d\n", ret.size());
	for (int i = ret.size() - 1; i >= 0; i--)
		printf("%d%c", ret[i], i == 0 ? '\n' : ' ');
}
