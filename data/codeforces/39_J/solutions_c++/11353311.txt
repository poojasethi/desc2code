#include<bits/stdc++.h>
using namespace std;
char s[1000010],t[1000010];
int main()
{
    scanf("%s%s",s,t);
    int len=strlen(t);
    int pre=0,suf=len-1;
    for(int i=0;i<len;i++)
    {
        if(s[i]==t[i]) pre++;
        else break;
    }
    for(int i=len-1;i>=0;i--)
    {
        if(t[i]==s[i+1]) suf--;
        else break;
    }
    if(pre-suf<=0) printf("0");
    else
    {
        printf("%d\n",pre-suf);
        for(int i=suf+1;i<=pre;i++)
            printf("%d ",i+1);
    }
    printf("\n");
    return 0;
}
