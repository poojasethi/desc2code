#include <bits/stdc++.h>
using namespace std;

string a,b,la,lb,ra,rb;
int m,n,l[1000002],r[1000002],sl=0;

int main() {
    getline(cin,a); getline(cin,b);
    m=a.size(); n=b.size();
    
    l[0]=1;
    for (int i=0; i<n; i++)
        l[i+1]=(l[i]*(a[i]==b[i]));
    r[n]=1;
    for (int i=n-1; i>=0; i--)
        r[i]=(r[i+1]*(a[i+1]==b[i]));
        
    for (int i=0; i<=n; i++)
        if ((l[i]==1)&&(r[i]==1)) sl++;
    
    cout << sl << endl;
    for (int i=0; i<=n; i++)
        if ((l[i]==1)&&(r[i]==1)) cout << i+1 << " ";
}
