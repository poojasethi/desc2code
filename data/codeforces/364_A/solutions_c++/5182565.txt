#include <iostream>
#include <cstdio>
#include <cstring>

#define For(i, a, b) for(int i=a; i<b; ++i)

using namespace std;

char ar[4005];
long long rep[36005];

int main()
{
	int T, N;
	long long res = 0;

	scanf("%d\n", &T);
	scanf("%s", ar);

	N = strlen(ar);

	For(i, 0, N)
	{
		int sum = 0;
		For(j, i, N)
		{
			sum += ar[j]-'0';
			rep[sum]++;
		}
	}

	if (T == 0)
		For(i, 0, 36001)
			res += rep[0]*rep[i];

	//cout<<res<<endl;

	For(i, 1, 36001)
		if (rep[i] > 0 and T%i == 0 and T/i <= 36000)
		{
			//cout<<rep[i]<<endl;
			res += rep[i]*rep[T/i];
		}
	cout<<res<<endl;

	return 0;
}
