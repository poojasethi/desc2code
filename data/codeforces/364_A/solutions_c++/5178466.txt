#include <iostream>
#include <algorithm>
#include <string>
using namespace std;
 
typedef long long i8;
 
int t;
string s;
i8 cma[4444*9];
 
main() {
	cin >>t >>s;
	
	int sz=s.size();
	for (int i=0; i<sz; i++) {
		int su=0;
		for (int j=i; j<sz; j++) {
			su += s[j]-(int)'0';
			++cma[su];
		}
	}
	
	i8 re=t?0:cma[0]*cma[0], ms=sz*9;
	for (int su=1; su<=ms; su++) {
		if (t) {
			if (t%su) continue;
			int ot=t/su;
			if (ot<=ms)
				re += cma[su]*cma[ot];
		} else {
			re += 2*cma[0]*cma[su];
		}
	}
	
	cout <<re;
}
