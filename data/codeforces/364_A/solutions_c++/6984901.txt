#include<stdio.h>
#include<string.h>
typedef long long ll;
int sum[50000],j,b[50000],flag[50000],i,n,a;
ll cnt;
char s[50000];
int main(){
	//freopen("in.txt","r",stdin);
	scanf("%d",&a);
	scanf("%s",s);
	n=strlen(s);
	for(i=0;i<n;i++) b[i+1]=s[i]-'0';
	sum[1]=b[1];
	for(i=2;i<=n;i++) sum[i]=sum[i-1]+b[i];
	for(i=1;i<=n;i++){
		for(j=i;j<=n;j++){
			flag[sum[j]-sum[i-1]]++;
		}
	}

	cnt=0;
	for(i=1;i<=n*9;i++){
		if(a%i==0&&(a/i)<=n*9) cnt+=1ll*flag[i]*flag[a/i];
	}
	if(a==0) cnt=1ll*2*cnt+1ll*flag[0]*flag[0];
	printf("%I64d\n",cnt);
	return 0;
}
