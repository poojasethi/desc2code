#include <iostream>
#include <string>
using namespace std;
int sum[4001];
long long hoge[36001];
int main(){
	int a;
	string s;
	cin >> a >> s;
	int n=s.size();
	for(int i=0;i<n;i++) sum[i+1]=sum[i]+(s[i]-'0');
	for(int i=1;i<=n;i++){
		for(int j=0;j<i;j++){
			hoge[sum[i]-sum[j]]++;
		}
	}
	long long ans=0;
	if(a==0){
		cout << hoge[0]*n*(n+1)-hoge[0]*hoge[0];
		return 0;
	}
	for(int i=1;i<=sum[n];i++){
		if(a%i!=0) continue;
		if(a/i>sum[n]) continue;
		ans+=hoge[i]*hoge[a/i];
	}
	cout << ans;
	return 0;
}
