#include <bits/stdc++.h>
#define MAX 36009
using namespace std;

typedef long long ll;

int main(){
  int n, arr[MAX];
  ll ret = 0;
  string str;
  cin >> n >> str;
  memset(arr, 0, sizeof(arr));
  for (int i = 0; i < str.size(); i++){
    int sum = 0;
    for (int j = i; j < str.size(); j++)
      sum += (str[j] - '0'), ++arr[sum];
  }
  for (int i = 1; i < MAX; i++)
    if (n % i == 0 && n / i < MAX)
      ret += (ll)arr[i] * arr[n / i];
  if (!n)
    ret *= 2, ret += (ll)arr[0] * arr[0];
  cout << ret << endl;
}
