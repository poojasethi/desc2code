#include<cmath>
#include<cstdio>
#include<cstring>
using namespace std;
long long ans,n,q,a[4005],b[40005];
char s[4005];
int main()
{
	scanf("%d",&q);
	scanf("%s",s+1);
	n=strlen(s+1);
	for(int i=1;i<=n;i++)
		a[i]=a[i-1]+s[i]-48;
	for(int i=1;i<=n;i++)
		for(int j=0;j<i;j++)
			b[a[i]-a[j]]++;
	for(int i=sqrt(q);i;i--)
		if(q%i==0&&i<40005&&q/i<40005)
			ans+=b[i]*b[q/i]*((q/i!=i)+1);
	if(!q)ans=b[0]*n*(n+1)-b[0]*b[0];
	printf("%I64d\n",ans);
	return 0;
}
