#include <iostream>
#include<string>
using namespace std;

long long sum[4001];
long long  field[40000];
int main() {
	
	long long a;
	string p;
	cin>>a>>p;
	int len = p.length();
	long long i,j;
	for(i=0;i<len;i++)
		sum[i+1] = sum[i]+p[i]-'0'; 
	for(i=1;i<=len;i++) 
		for(j=0;j<i;j++)
			field[sum[i]-sum[j]]++;
	long long ans=0;
	if(a==0) 
		for(i=0;i<=sum[len];i++) ans+=field[0]*field[i];  
	for(i=1;i<=sum[len];i++)  
		if(a%i==0&&a/i<=sum[len])
			ans+=field[i]*field[a/i];
	cout<<ans<<endl;
	return 0;
}
