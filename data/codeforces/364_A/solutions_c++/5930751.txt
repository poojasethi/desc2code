#include<cstdio>
#include<cstring>
typedef long long ll;
#define rep(i,n) for (int i=0;i<n;++i)
const int N=40005;
int n,m; char a[N]; ll ans,sum,s[N],c[N];
int main()
{
	scanf("%d%s",&m,a),n=strlen(a);
	rep(i,n) s[i+1]=s[i]+a[i]-48;
	rep(j,n+1) rep(i,j) ++c[s[j]-s[i]],++sum;
	if (!m) ans=sum*sum-(sum-c[0])*(sum-c[0]);
	else for (int i=1;i<=n*9 && i<=m;++i){
		int j=m/i;
		if (i*j==m && j<=n*9) ans+=c[i]*c[j];
	}
	printf("%I64d\n",ans);
	return 0;
}
