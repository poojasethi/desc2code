#include <bits/stdc++.h>
using namespace std;
const int N=10005;
char s[N];
typedef long long ll;
ll cnt[N*10], sum[N], a, ans;
int main() {
	ios::sync_with_stdio(false);
	cin >> a >> s+1;
	int n=strlen(s+1);
	for(int i=1; i<=n; ++i) sum[i]=sum[i-1]+s[i]-'0';
	for(int i=1; i<=n; ++i) for(int j=1; j<=i; ++j) cnt[sum[i]-sum[j-1]]++;
	if(a==0) {
		for(int i=0; i<=sum[n]; ++i) ans+=cnt[0]*cnt[i]*2;
		cout << ans-cnt[0]*cnt[0];
		return 0;
	}
	for(int i=1; i<=sum[n]; ++i) 
		if(a%i==0 && a/i<=sum[n]) ans+=cnt[i]*cnt[a/i];
	cout << ans;
	return 0;
}
