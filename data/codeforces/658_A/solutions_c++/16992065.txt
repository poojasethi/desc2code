#include <iostream>
#include <cstdio>
using namespace std;

int n, c, sa, sb, ta, tb, p[1005], t[1005];
int main(){
	int i;
	cin>>n>>c;
	for(i=0; i<n; i++){
		scanf("%d", &p[i]);
	}
	for(i=0; i<n; i++){
		scanf("%d", &t[i]);
	}
	for(i=0; i<n; i++){
		ta += t[i];
		sa += max(0, p[i]-c*ta);
	}
	for(i=n-1; i>=0; i--){
		tb += t[i];
		sb += max(0, p[i]-c*tb);
	}
	if(sa > sb) cout<<"Limak";
	else if(sb > sa) cout<<"Radewoosh";
	else cout<<"Tie";
	return 0;
}
