#include<stdio.h>

int main()
{
	int p[55],t[55];
	int n,c;
	int i,j,k;
	while(scanf("%d%d",&n,&c)!=EOF)
	{
		int a=0,b=0;
		for(i=0;i<n;i++) scanf("%d",&p[i]);
		for(i=0;i<n;i++) scanf("%d",&t[i]);
		int tt=0;
		for(i=0;i<n;i++)
		{
			tt+=t[i];
			if(p[i]-c*tt>0) a+=p[i]-c*tt;
		}
		tt=0;
		for(i=n-1;i>=0;i--)
		{
			tt+=t[i];
			if(p[i]-c*tt>0) b+=p[i]-c*tt;
		}
		if(a<b) puts("Radewoosh");
		else if(a>b) puts("Limak");
		else puts("Tie");
	}
	return 0;
}
