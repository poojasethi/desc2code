#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,c;
	cin>>n>>c;
	int s[n+2];
	int t[n+2];
	for(int i=0;i<n;i++){
		cin>>s[i];
	}
	for(int i=0;i<n;i++){
		cin>>t[i];
	}
	int li=0,ra=0;
	int con=0;
	for(int i=0;i<n;i++){
		con+=t[i];
		li+=max(0,s[i]-(c*con));
	}
	con=0;
	for(int i=n-1;i>=0;i--){
		con+=t[i];
		ra+=max(0,s[i]-(c*con));
	}
	if(ra==li){
		cout<<"Tie"<<endl;
	}else{
		if(ra>li){
			cout<<"Radewoosh"<<endl;
		}else{
			cout<<"Limak"<<endl;
		}
	}
	return 0;
}
