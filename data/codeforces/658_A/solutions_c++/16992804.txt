#include <iostream>
using namespace std;

main () {
	int n,c,i,p[1000],t[1000],s1=0,s2=0,t1=0,t2=0,a;
	cin >>n>>c;
	for (i=0;i<=n-1;i++) {
		cin >>p[i];
	}
	for (i=0;i<=n-1;i++) {
		cin >>t[i];
	}
	for (i=0;i<=n-1;i++) {
		t1+=t[i];
		a=p[i]-c*t1;
		if (a<0) a=0;
		s1+=a;
	}
	for (i=n-1;i>=0;i--) {
		t2+=t[i];
		a=p[i]-c*t2;
		if (a<0) a=0;
		s2+=a;
	}
	if (s1>s2) cout <<"Limak";
	else if (s1<s2) cout <<"Radewoosh";
	else cout <<"Tie";
	return 0;
}
