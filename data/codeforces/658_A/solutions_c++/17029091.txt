#include<bits/stdc++.h>
using namespace std;
int p[1003],t[1003];
int i;
int main()
{
int n,c;
cin >> n >> c;
for(i=0;i<n;i++)
cin >> p[i];
for(i=0;i<n;i++)
cin >> t[i];
int temp1=0,temp2=0;
int x1=0,x2=0;
for(i=0;i<n;i++)
{
	x1=x1+t[i];
	temp1=temp1+max(0,p[i]-x1*c);
	
}
for(i=n-1;i>=0;i--)
{
	x2=x2+t[i];
	temp2=temp2+max(0,p[i]-c*x2);
}

if(temp1>temp2)
cout << "Limak";
if(temp1<temp2)
cout << "Radewoosh";
if(temp1==temp2)
cout << "Tie";
return 0;
}
