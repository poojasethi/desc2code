#include<bits/stdc++.h>
using namespace std;

int n,c,p[1009],t[1009];

int main()
{
	scanf("%d %d", &n, &c);
	for(int i=0; i<n; i++)
	{
		scanf("%d", &p[i]);
	}
	for(int i=0; i<n; i++)
	{
		scanf("%d", &t[i]);
	}
	
	int L=0,R=0;
	
	int t1=0,t2=0;
	for(int i=0; i<n; i++)
	{
		t1=t1+t[i];
		t2=t2+t[n-i-1];
		
		L=L+max(0, p[i]-t1*c);
		R=R+max(0, p[n-i-1]-t2*c);
	}
	
	if(L>R)	printf("Limak\n");
	else if(L<R)	printf("Radewoosh\n");
	else printf("Tie\n");
	
	return 0;
}
