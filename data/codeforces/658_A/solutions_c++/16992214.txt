#include <iostream>

using namespace std;

int main() {
	int n, c;
	int l = 0, r = 0;
	int tl = 0, tr = 0;
	cin >> n >> c;
	int p[n], t[n];
	for (int i=0; i<n; i++) cin >> p[i];
	for (int i=0; i<n; i++) cin >> t[i];
	for (int i=0; i<n; i++) {
		tl += t[i];
		l += max(0,p[i]-tl*c);
		tr += t[n-1-i];
		r += max(0,p[n-1-i]-tr*c);
	}
	if (l > r) cout << "Limak" << endl;
	else if (l < r) cout << "Radewoosh" << endl;
	else cout << "Tie" << endl;

}
