#include<stdio.h>
int main()
{
long long int n,c,p[2000]={0},t[2000]={0},x=0,A=0,B=0,g=0,i;
scanf("%lld %lld",&n,&c);
for(i=0;i<n;i++)
{
scanf("%lld",&p[i]);
}
for(i=0;i<n;i++)
{
scanf("%lld",&t[i]);
}
for(i=0;i<n;i++)
{
x+=t[i];
long long int g=0;
g=p[i]-c*x;
if(g>0)A+=g;
}
x=0;
for(i=n-1;i>=0;i--)
{
x+=t[i];
long long int g=0;
g=p[i]-c*x;
if(g>0)B+=g;
}
if(A>B)printf("Limak\n");
else if(B>A)printf("Radewoosh\n");
else printf("Tie\n");
return 0; 
}
