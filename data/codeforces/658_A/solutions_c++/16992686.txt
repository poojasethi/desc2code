#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,c;cin>>n>>c;
	int p[n],t[n];
	for(int i=0;i<n;i++) cin>>p[i];
	for(int i=0;i<n;i++) cin>>t[i];

	int ans1=0,ans2=0,t1=0,t2=0;

	for(int i=0;i<n;i++){
		t1+=t[i];
		t2+=t[n-i-1];
		ans1 += max(0,p[i]-c*t1);
		ans2 += max(0,p[n-i-1]-c*t2);
	}
	//cout<<ans1<<endl<<ans2<<endl;
	if(ans1>ans2) cout<<"Limak";
	else if(ans1<ans2) cout<<"Radewoosh";
	else cout<<"Tie";

	return 0;
}
