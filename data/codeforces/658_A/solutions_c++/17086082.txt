#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,c;
	cin>>n>>c;
	int p[n],t[n];
	for(int i=0;i<n;i++)
	{
		cin>>p[i];
	}
	for(int i=0;i<n;i++)
	{
		cin>>t[i];
	}
	int f=0,b=0,p1=0,p2=0;
	for(int i=0;i<n;i++)
	{   f+=t[i];
		p1+=max(0,(p[i]-(f*c)));
		b+=t[n-i-1];
		p2+=max(0,(p[n-1-i]-(b*c)));
	}
	if(p1>p2)
	{
		cout<<"Limak"<<endl;
	}
	else
	{
		if(p1<p2)
		{
			cout<<"Radewoosh"<<endl;
		}
		else
		{
			cout<<"Tie"<<endl;
		}
	}
}
