#include<iostream>
#include<cmath>
using namespace std;
int main ()
{
 int n,c;
 int p[51],t[51];
 cin>>n>>c;
 int i;
 for(i=0;i<n;i++) cin>>p[i];
 for(i=0;i<n;i++) cin>>t[i];
 int resL=0;
 int resR=0;
 int ts=0;
 for(i=0;i<n;i++) {ts+=t[i];resL+=(max(0,p[i]-c*ts));}
 ts=0;
 for(i=n-1;i>=0;i--) {ts+=t[i];resR+=(max(0,p[i]-c*ts));}
 if(resL>resR) cout<<"Limak"<<endl;
 else if(resR>resL) cout<<"Radewoosh"<<endl;
 else if(resL==resR) cout<<"Tie"<<endl;
 return 0;
}
