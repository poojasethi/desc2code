#include<stdio.h>
#include<algorithm>
using namespace std;
int main(){
int n,c;
scanf("%d %d",&n,&c);
int p[n],t[n];
int limak=0,radewoosh=0;
int x=0;
for(int i=0;i<n;i++)scanf("%d",p+i);
for(int i=0;i<n;i++)scanf("%d",t+i);
for(int i=0;i<n;i++)
{
x+=t[i];
limak+=max(0,p[i]-c*x);

}
x=0;
for(int i=n-1;i>=0;i--)
{
x+=t[i];
radewoosh+=max(0,p[i]-c*x);
}
if(limak>radewoosh)printf("Limak");
else if(radewoosh>limak)printf("Radewoosh");
else printf("Tie");
}
