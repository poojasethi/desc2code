#include<bits/stdc++.h>

using namespace std;

int Sc[200],T[200],c,n,sum1=0,sum2=0,ans1=0,ans2=0;

int main(){
	cin>>n>>c;

	for(int i= 0; i<n;i++){
		cin>>Sc[i];
	}

//cout<<"Nosg\n"<<endl;
	for(int i= 0; i<n;i++){
		cin>>T[i];
		sum1+= T[i];
		ans1+= max(Sc[i]- c*sum1,0);
	}


	for(int i= n-1; i>=0;i--){
		
		sum2+= T[i];
		ans2+= max(Sc[i]- c*sum2,0);
	}

	if(ans1>ans2)cout<<"Limak"<<endl;
	else if(ans1==ans2) cout<<"Tie"<<endl;
	else cout<<"Radewoosh"<<endl;

	return 0;
}
