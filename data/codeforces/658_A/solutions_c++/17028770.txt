#include<bits/stdc++.h>
using namespace std;

int main(){
	int n,c,p[1005],t[1005];
	int i,j,tl=0,tr=0,sl=0,sr=0,temp=0;

	cin>>n>>c;
	for(i=0;i<n;i++)
		cin>>p[i];
	for(i=0;i<n;i++)
		cin>>t[i];
	for(i=0;i<n;i++){
		tl+=t[i];
		temp=p[i]-c*tl;
		if(temp>0){
			sl+=temp;
			}
		}
	for(i=n-1;i>=0;i--){
		tr+=t[i];
		temp=p[i]-c*tr;
		if(temp>0){
			sr+=temp;
			}
		}
	if(sl>sr){
		cout<<"Limak\n";
		}
	else if(sl<sr){
		cout<<"Radewoosh\n";
		}
	else{
		cout<<"Tie\n";
		}
return 0;	
}
