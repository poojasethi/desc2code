#include <bits/stdc++.h>
using namespace std;

 
 
int main()
{   int n,c;
    cin>>n>>c;
    long long t[n],p[n];
    for(int i=0;i<n;i++)cin>>t[i];
    for(int i=0;i<n;i++)cin>>p[i];
        long long a1=0,a2=0;
    long long x=0;
    for(int i=0;i<n;i++)x+=p[i],a1+=max(0LL,t[i]-c*x);
    x=0;
    for(int i=n-1;i>=0;i--)x+=p[i],a2+=max(0LL,t[i]-c*x);
    if(a1>a2)cout<<"Limak";
    else if(a1==a2)cout<<"Tie";
    else cout<<"Radewoosh";
 
    return 0;
}
