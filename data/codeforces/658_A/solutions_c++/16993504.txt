#include<iostream>
using namespace std;
int main()
{
	int p[1000],t[10000],n,c,i,sum1=0,sum2=0,time=0,sum;
	cin>>n>>c;
	for(i=0;i<n;i++)
	cin>>p[i];
	for(i=0;i<n;i++)
	cin>>t[i];
	for(i=0;i<n;i++)
	{
		time+=t[i];
		sum=p[i]-c*time;
		if(sum>0)
		sum1+=sum;
	}
	time=0;
	for(i=n-1;i>=0;i--)
	{
		time+=t[i];
		sum=p[i]-c*time;
		if(sum>0)
		sum2+=sum;
	}
	if(sum1>sum2)
	cout<<"Limak";
	else if(sum2>sum1)
	cout<<"Radewoosh";
	else
	cout<<"Tie";
	return 0;
}
