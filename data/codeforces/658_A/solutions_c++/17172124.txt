#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<iostream>
using namespace std;
int main(){
	int i,j,n,c,p[10000],t[10000],L=0,R=0,ts=0;
	cin>>n;
	cin>>c;
	for(i=0;i<n;i++){
		cin>>p[i];
	}
	for(i=0;i<n;i++){
		cin>>t[i];
	}
	for(i=0;i<n;i++){
		ts+=t[i];
		if(0<p[i]-ts*c) L+=p[i]-ts*c;
	}
	ts=0;
	for(i=n-1;i>-1;i--){
		ts+=t[i];
		if(0<p[i]-ts*c) R+=p[i]-ts*c;
	}
	if(L>R) cout<<"Limak";
	else
	if(L<R) cout<<"Radewoosh";
	else cout<<"Tie";
}
