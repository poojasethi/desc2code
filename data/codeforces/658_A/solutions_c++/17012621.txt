#include<iostream>
using namespace std;
int main()
{
	int n,c,d=0,e=0,f=0,g=0,i;
	cin>>n>>c;
	int a[n],b[n];
	for(i=0;i<n;i++){
		cin>>a[i];
	}
	for(i=0;i<n;i++){
		cin>>b[i];
	}
	for(i=0;i<n;i++){
		d=b[i]+d;
		e=max(0,a[i]-c*d)+e;
	}
	for(i=n-1;i>=0;i--){
		f=b[i]+f;
		g=max(0,a[i]-c*f)+g;
	}
	if(e>g)
	cout<<"Limak";
	else if(g>e)
	cout<<"Radewoosh";
	else
	cout<<"Tie";
}
