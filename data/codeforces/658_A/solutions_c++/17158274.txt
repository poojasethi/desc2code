#include<bits/stdc++.h>
using namespace std;
int main()
{
    int a,b;
    cin>>a>>b;
    int ans1=0,ans2=0;
    int c[1002],d[1004];
    int t=0;
    for(int i=1;i<=a;i++) cin>>c[i];
    for(int i=1;i<=a;i++)  cin>>d[i];
    for(int i=1;i<=a;i++) t+=d[i],ans1+=max(0,c[i]-t*b);
    t=0;
    for(int i=a;i>=1;i--) t+=d[i],ans2+=max(0,c[i]-t*b);
    if (ans1==ans2) cout<<"Tie";
    else if (ans1>ans2) cout<<"Limak";
    else cout<<"Radewoosh";
}
