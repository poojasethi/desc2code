#include <iostream>
using namespace std;
int p[1001],t[1001];
int main(){
	int n,c;
	cin>>n>>c;
	for(int i=0;i<n;i++){
		cin>>p[i];
	}
	for(int i=0;i<n;i++){
		cin>>t[i];
	}
	int l=0,r=0,ans=0,ans1=0;
	for(int i=n-1;i>=0;i--){
		l=l+t[i];
		ans+=max(0,p[i]-c*l);	
	}
	for(int i=0;i<n;i++){
		r=r+t[i];
		ans1+=max(0,p[i]-c*r);	
	}
	if(ans1>ans){
		cout<<"Limak";
	}
	else if(ans>ans1){
		cout<<"Radewoosh";
	}
	else{
		cout<<"Tie";
	}
	
	
}
