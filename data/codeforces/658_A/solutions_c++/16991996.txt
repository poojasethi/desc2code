#include<bits/stdc++.h>
using namespace std;
typedef long int ll;
int main()
{
	ll i,n,c,a[100],t[100],tme1=0,ans1=0,tme2=0,ans2=0;
	cin>>n>>c;
	for(i=0;i<n;i++)cin>>a[i];
	for(i=0;i<n;i++)cin>>t[i];
	for(i=0;i<n;i++)
	{
		tme1+=t[i];
		if((a[i]-(tme1*c))>0)ans1+=a[i]-(tme1*c);
	}
	for(i=n-1;i>=0;i--)
	{
		tme2+=t[i];
		if((a[i]-(tme2*c))>0)ans2+=a[i]+-(tme2*c);
	}
	if(ans1>ans2)cout<<"Limak"<<endl;
	else if(ans2>ans1)cout<<"Radewoosh"<<endl;
	else cout<<"Tie"<<endl;
	return 0;
}
