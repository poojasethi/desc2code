#include <iostream>
#include <sstream>
#include <algorithm>

using namespace std;

int main() {
	int n,c;
	cin>>n>>c;
	int p[n], t[n];
	for(int i = 0; i < n; i++){
		cin>>p[i];
	}
	for(int i = 0; i < n; i++){
		cin>>t[i];
	}
	int l = 0, r = 0, tl = 0, tr = 0;
	for(int i=0;i<n;i++){
		tl+=t[i];
		l+=max(0, p[i]-c*tl);
		tr+=t[n-i-1];
		r+=max(0,p[n-i-1]-c*tr);
	}
	if(l>r){
		cout<<"Limak"<<endl;
	}
	else if(r>l){
		cout<<"Radewoosh"<<endl;
	}
	else{
		cout<<"Tie"<<endl;
	}
}
