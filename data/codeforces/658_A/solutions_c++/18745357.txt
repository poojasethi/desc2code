#include <iostream>
using namespace std;

int main() {
	int n,c,i,l=0,r=0;
	cin>>n>>c;
	int p[n],t[n];
	for(i=0;i<n;i++)
	cin>>p[i];
	for(i=0;i<n;i++)
	cin>>t[i];
	
	int t1=0;
	for(i=0;i<n;i++)
	{
		t1+=t[i];
		l+=max(0,p[i]-c*t1);
	}
	int t2=0;
	for(i=n-1;i>=0;i--)
	{
		t2+=t[i];
		r+=max(0,p[i]-c*t2);
	}
	if(l>r)
	cout<<"Limak\n";
	else if(r>l)
	cout<<"Radewoosh\n";
	else
	cout<<"Tie\n";
	return 0;
}
