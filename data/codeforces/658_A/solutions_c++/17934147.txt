#include<cstdio>
#include<algorithm>

int n,c;
int p[60];
int t[60],T;
int A,B;

int main()
{
	scanf("%d%d",&n,&c);
	for(int i=0;i<n;i++) scanf("%d",p+i);
	for(int i=0;i<n;i++) scanf("%d",t+i);

	for(int i=0;i<n;i++)
	{
		T+=t[i];
		A+=std::max(0,p[i]-c*T);
	}
	T=0;
	for(int i=n-1;i>=0;i--)
	{
		T+=t[i];
		B+=std::max(0,p[i]-c*T);
	}

	if(A==B) printf("Tie\n");
	else if(A>B) printf("Limak\n");
	else printf("Radewoosh\n");
	return 0;
}
