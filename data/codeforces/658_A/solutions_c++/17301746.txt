#include<iostream>
using namespace std;
int main()
{	
	int n,c,*p,*t,s1,s2,t1,t2,i;
	while(cin>>n>>c)
	{	
		p=new int[n];
		t=new int[n];
		for(i=0;i<n;i++) cin>>p[i];
		for(i=0;i<n;i++) cin>>t[i];
  		for(s1=0,s2=0,t1=0,t2=0,i=0;i<n;i++)
  		{
  			t1+=t[i];
  			t2+=t[n-1-i];
  			s1+=p[i]-c*t1>0?p[i]-c*t1:0;
  			s2+=p[n-1-i]-c*t2>0?p[n-1-i]-c*t2:0;
  		}
  		cout<<(s1-s2==0?"Tie":(s1-s2>0?"Limak":"Radewoosh"))<<endl;
		delete[] p;
		delete[] t;
	}
	return 0;
}
