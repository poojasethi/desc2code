#include <iostream>
#define NMAX 52
using namespace std;

int main(){
   int n, c, i, p[NMAX], t[NMAX], x1, x2, s1, s2;
   cin>>n>>c;
   x1=0;
   x2=0;
   s1=0;
   s2=0;
   for(i=1; i<=n; i++)cin>>p[i];
   for(i=1; i<=n; i++)cin>>t[i];
   for(i=1; i<=n; i++){
      x1+=t[i];
      x2+=t[1+n-i];
      s1+=max(0, p[i]-c*x1);
      s2+=max(0, p[1+n-i]-c*x2);
   }
   if(s1>s2){
      cout<<"Limak";
   }else if(s2>s1){
      cout<<"Radewoosh";
   }else{
      cout<<"Tie";
   }
   return 0;
}
