#include <iostream>
using namespace std;

int main(){
	int n,c;
	cin>>n>>c;
	int p[n];
	int t[n];
	
	for(int i=0;i<n;i++){
		cin>>p[i];
	}
	
	for(int i=0;i<n;i++){
		cin>>t[i];
	}
	
	int l=0,r=0;
	int time=0;
	for(int i=0;i<n;i++){
		time+=t[i];
		l+=max(0,p[i]-c*time);
	}
	
	time=0;
	
	for(int i=n-1;i>=0;i--){
		time+=t[i];
		r+=max(0,p[i]-c*time);
	}
	
	if(l==r){
		cout<<"Tie";
	}
	else if(l>r){
		cout<<"Limak";
	}
	else
	cout<<"Radewoosh";
}
