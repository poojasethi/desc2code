#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,c,p[100],t[100];
	cin>>n>>c;
	for(int i = 1; i<=n;i++)
		cin>>p[i];
	for(int i = 1; i<=n;i++)
		cin>>t[i];

	int t1 = 0, t2 = 0, sum1 = 0, sum2 = 0;
	for(int i = 1; i<=n;i++)
	{
		t1 += t[i];
		sum1 += max(0, p[i] - c*t1);

		t2 += t[n-i+1];
		sum2 += max(0, p[n-i+1] - c*t2);
	}
	if(sum1 > sum2)
	{
		cout<<"Limak";
	}
	else if(sum1 < sum2)
	{
		cout<<"Radewoosh";
	}
	else
	{
		cout<<"Tie";
	}
}
