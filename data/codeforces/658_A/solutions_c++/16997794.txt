#include <iostream>
using namespace std;
int main()
{
int n,x,c,R,L,k;
int p[51];
int t[51];
while(cin>>n>>c)
{
R=0;L=0;k=0;
for(int i=0;i<n;i++)
{
cin>>p[i];
}

for(int i=0;i<n;i++)
{
cin>>t[i];
}
x=0;
for(int i=0;i<n;i++)
{
x+=t[i];
k=p[i]-c*x;
if(k<=0)
    k=0;   
L+=k;
}
x=0;
for(int i=n-1;i>=0;i--)
{
x+=t[i];
k=p[i]-c*x;
if(k<=0)
    k=0;
R+=k;
}
if(R == L)
    cout<<"Tie"<<endl;
else if(R < L)
    cout<<"Limak"<<endl;
else
    cout<<"Radewoosh"<<endl;
}
return 0;
}
