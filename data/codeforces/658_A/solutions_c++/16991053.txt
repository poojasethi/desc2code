#include "bits/stdc++.h"
using namespace std;

int main() {
	// your code goes here
	int n,c,a[55],b[55],x=0,sum=0,summ=0;
	cin>>n>>c;
	for(int i=0;i<n;i++)
	cin>>a[i];
	for(int i=0;i<n;i++)
	cin>>b[i];
	for(int i=0;i<n;i++)
	{
		x+=b[i];
		sum+=max(0,a[i]-(c*x));
	}
	x=0;
	for(int i=n-1;i>=0;i--)
	{
		x+=b[i];
		summ+=max(0,a[i]-(c*x));
	}
	if(sum>summ)
	cout<<"Limak";
	else if(sum<summ)
	cout<<"Radewoosh";
	else
	cout<<"Tie";
	return 0;
}
