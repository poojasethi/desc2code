#include<iostream>
#include<cmath>
#include<algorithm>
using namespace std;
long long n,c,r,l,a[1010],b[1010],t;
int main(){
cin>>n>>c;
for( int i = 0; i < n; i++){
    cin>>a[i];
}
for( int i = 0; i < n; i++){
    cin>>b[i];
}   
for( int i = 0; i < n; i++){
    t+=b[i];
    l+=max(1LL*0,(a[i]-c*t));
}  
t=0;
for ( int i = n-1; i >= 0; i--){
    t+=b[i];
    r+=max(1LL*0,(a[i]-c*t));
}
if(r>l)cout<<"Radewoosh";
else if(l>r)cout<<"Limak";else if(l==r)cout<<"Tie";
}
