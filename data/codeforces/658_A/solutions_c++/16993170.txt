#include<bits/stdc++.h>
using namespace std;

int main()
{int n,c;
	cin>>n>>c;
	int p[n];
	int t[n];
	for(int i=0;i<n;i++)
	cin>>p[i];
	
	for(int i=0;i<n;i++)
	cin>>t[i];
	
	int time=0;
	int sl=0;
	for(int i=0;i<n;i++)
	{
		time+=t[i];
		sl=sl+max(0,p[i]-(c*time));
	}
	
	 time=0;
	int sr=0;
	for(int i=n-1;i>=0;i--)
	{
		time+=t[i];
		sr=sr+max(0,p[i]-(c*time));
	}
	
	if(sl>sr)
	cout<<"Limak";
	else if(sr>sl)
	cout<<"Radewoosh";
	else
	cout<<"Tie";
}
