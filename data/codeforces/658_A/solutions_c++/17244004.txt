#include<bits/stdc++.h>
using namespace std;
int a,b,arr[51],arr2[51],l,r,t;
int main(){
 cin>>a>>b;
 for(int i=0;i<a;i++){
  cin>>arr[i];
 }
 for(int i=0;i<a;i++){
  cin>>arr2[i];
 }
 for(int i=0;i<a;i++){
  t+=arr2[i];
  l+=max(0,arr[i]-b*t);
 }
 t=0;
 for(int i=a-1;i>=0;i--){
  t+=arr2[i];
  r+=max(0,arr[i]-b*t);
 }
 //cout<<l<<" "<<r<<endl;
 if(l>r)cout<<"Limak"<<endl;
 else if(r>l)cout<<"Radewoosh"<<endl;
 else cout<<"Tie"<<endl;
}
