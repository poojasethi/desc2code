#include<bits/stdc++.h>
using namespace std;
#define maxn 1005
int n,c,p[maxn],t[maxn];
int main(){
	scanf("%d%d",&n,&c);
	for(int i=1;i<=n;i++)
		scanf("%d",&p[i]);
	for(int i=1;i<=n;i++)
		scanf("%d",&t[i]);
	int ans1=0,t1=0;
	for(int i=1;i<=n;i++){
		t1+=t[i];
		ans1+=max(0,p[i]-c*t1);
	}
	int ans2=0,t2=0;
	for(int i=n;i;i--){
		t2+=t[i];
		ans2+=max(0,p[i]-c*t2);
	}
	if(ans1>ans2)puts("Limak");
	else if(ans1<ans2)puts("Radewoosh");
	else puts("Tie");
	return 0;
}
