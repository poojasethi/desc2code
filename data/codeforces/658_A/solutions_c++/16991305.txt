#include <iostream>

using namespace std;

long long n,a[2000],c,b[2000];

int main()
{
    cin >> n >> c;
    long long dem=0;
    for (int i=1;i<=n;i++)
        cin >> a[i];
    long long t=0;
    for (int i=1;i<=n;i++)
        cin >> b[i],dem+=max(1LL*0,a[i]-c*t-c*b[i]),t+=b[i];
    t=0;
    for (int i=n;i>=1;i--)
        dem-=max(1LL*0,a[i]-c*t-c*b[i]),t+=b[i];
    if (dem<0) cout <<"Radewoosh";
    if (dem>0) cout << "Limak";
    if (dem==0) cout << "Tie";
}
