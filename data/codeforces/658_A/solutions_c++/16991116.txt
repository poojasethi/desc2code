
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,m,i;

cin>>n>>m;

vector <LL> v(n),t(n);

for(i=0;i<n;i++)
	cin>>v[i];


for(i=0;i<n;i++)
	cin>>t[i];

//Limak
LL l=0,ti=0;
for(i=0;i<n;i++)
	{ti+=t[i];
	l+=max((LL)0,v[i]-m*ti);
	}
 
//Radewoosh
LL r=0;
ti=0;
for(i=n-1;i>=0;i--)
	{ti+=t[i];
	r+=max((LL)0,v[i]-m*ti);
	}

if(l>r)
		cout<<"Limak"<<endl;
else if(l<r)
		cout<<"Radewoosh"<<endl;
else
		cout<<"Tie"<<endl;

}
