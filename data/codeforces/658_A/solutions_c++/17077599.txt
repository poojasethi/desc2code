#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,c,p[55],t[55];
	long int ans,tot=0,pt1=0,pt2=0;
	cin>>n>>c;
	for(i=1;i<=n;i++) cin>>p[i];
	for(i=1;i<=n;i++) cin>>t[i];
	for(i=1;i<=n;i++) 
	{
		tot+=t[i];
		ans=(p[i]-c*tot);
		pt1+=max(0,(int)ans);
	}
	tot=0;
	for(i=n;i>0;i--) 
	{
		tot+=t[i];
		ans=(p[i]-c*tot);
		pt2+=max(0,(int)ans);
	}
	if(pt1>pt2) cout<<"Limak\n";
	else if(pt2>pt1) cout<<"Radewoosh\n";
	else cout<<"Tie\n";
	return 0;
}
