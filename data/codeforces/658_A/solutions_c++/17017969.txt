#include<bits/stdc++.h>

using namespace std;

int main(){
	
	int i,n,p[100],t[100],c,L=0,R=0;
	
	cin>>n>>c;
	
	for(i=0;i<n;i++){
		cin>>p[i];
	}
	
	for(i=0;i<n;i++){
		cin>>t[i];
	}
	
	int time=0;
	for(i=0;i<n;i++){
		time+=t[i];
		L+=max(0,p[i]-c*time);
	}
	
	time=0;
	for(i=n-1;i>=0;i--){
		time+=t[i];
		R+=max(0,p[i]-c*time);
	}
	
	if(L>R){
		cout<<"Limak";
	}
	else if(L<R){
		cout<<"Radewoosh";
	}
	else{
		cout<<"Tie";
	}
	
	return 0;
}
