#include<bits/stdc++.h>
using namespace std;
int p[1010],t[1010];
int main()
{
int n,c,i,j;
cin>>n>>c;


for(i=0;i<n;i++)
  cin>>p[i];

  for(i=0;i<n;i++)
    cin>>t[i];

long long sc1=0,sc2=0,tt=0;

for(i=0;i<n;i++){
    tt+=t[i];
  sc1+=max(0LL,p[i]-c*tt);
  }
  tt=0;

for(i=n-1;i>=0;i--)
{
 tt+=t[i];
  sc2+=max(0LL,p[i]-c*tt);
}


if(sc1>sc2)
  cout<<"Limak";
else if(sc1<sc2)
  cout<<"Radewoosh";
else
cout<<"Tie";




}
