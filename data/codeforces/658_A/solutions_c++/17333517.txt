#include <iostream>
#include <cstdio>
#include <cstdlib>
using namespace std;
int main()
{
	int n,c,p[55],t[55],s1=0,s2=0,ti=0;
	cin>>n>>c;
	for(int i=0;i<n;i++)
		cin>>p[i];
	for(int i=0;i<n;i++)
		cin>>t[i];
	for(int i=0;i<n;i++)
	{
		ti+=t[i];
		s1+=max(0,p[i]-c*ti);
	}
	ti=0;
	for(int i=n-1;i>=0;i--)
	{
		ti+=t[i];
		s2+=max(0,p[i]-c*ti);
	}
	if(s1>s2)
		cout<<"Limak\n";
	else if(s1<s2)
		cout<<"Radewoosh\n";
	else
		cout<<"Tie\n";
	return 0;
}
