#include<iostream>
using namespace std;
int msk(int x,int y){
	return x>y?x:y;
}
int main(){
	int n,c,i,a[55],b[55],sum1=0,sum2=0,tim1=0,tim2=0;
	cin>>n>>c;
	for(i=1;i<=n;i++)cin>>a[i];
	for(i=1;i<=n;i++)cin>>b[i];
	for(i=1;i<=n;i++){
		tim1+=b[i];
		tim2+=b[n-i+1];
		sum1+=msk(0,a[i]-c*tim1);
		sum2+=msk(0,a[n-i+1]-c*tim2);
	}
	if(sum1>sum2)cout<<"Limak"<<endl;
	else if(sum1<sum2)cout<<"Radewoosh"<<endl;
	else cout<<"Tie"<<endl;
	return 0;
}
