# -*-coding:utf-8 -*-
n,c=map(int,raw_input().split(' '))
p=map(int,raw_input().split(' '))
t=map(int,raw_input().split(' '))
Limak=0
time_Limak=0
Radewoosh=0
time_Radewoosh=0
for i in range(n):
    time_Limak+=t[i]
    Limak+=max(0,p[i]-c*time_Limak)
for i in range(n-1,-1,-1):
    time_Radewoosh+=t[i]
    Radewoosh+=max(0,p[i]-c*time_Radewoosh)
if Limak>Radewoosh:
    print "Limak"
elif Limak==Radewoosh:
    print "Tie"
else:
    print "Radewoosh"
