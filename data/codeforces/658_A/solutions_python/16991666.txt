l = raw_input().split(' ')
n = int(l[0])
c = int(l[1])
l = raw_input().split(' ')
P = [int(x) for x in l]
l = raw_input().split(' ')
T = [int(x) for x in l]
Ls = 0
Rs = 0
Lt = 0
Rt = 0

for i in range(0, n):
    Lt += T[i]
    Ls += max(0, P[i]-c*Lt)
    
    Rt += T[n-i-1]
    Rs += max(0, P[n-i-1] - c*Rt)
    
if(Ls > Rs):
    print "Limak"
elif(Ls < Rs):
    print "Radewoosh"
else:
    print "Tie"
