line1 = raw_input().split(' ')
radp = map(int,raw_input().split(' '))
radt = map(int,raw_input().split(' '))

limp = radp[::-1]
limt = radt[::-1]

n = int(line1[0])
c = int(line1[1])

lim = 0
rad = 0

for i in range(0,n):
    rad += max(0, (radp[i] - (sum(radt[0:i+1]) * c)))
    lim += max(0, (limp[i] - (sum(limt[0:i+1]) * c)))
    
if lim > rad:
    print('Radewoosh')
elif lim < rad:
    print('Limak')
else:
    print('Tie')
