n, c = map(int, raw_input().split())
p = map(int, raw_input().split())
t = map(int, raw_input().split())

s1, s2 = 0, 0
x1, x2 = 0, 0
for i in xrange(n):
    x1 += t[i]
    x2 += t[n-i-1]
    s1 += max(0, p[i] - c*x1)
    s2 += max(0, p[n-i-1] - c*x2)
if s1 > s2: print "Limak"
elif s1 == s2: print "Tie"
else: print "Radewoosh"
