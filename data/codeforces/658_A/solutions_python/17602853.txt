n,c = raw_input().split(' ')
n=int(n)
c=int(c)

marks = [int(x) for x in raw_input().split()]
time = [int(x) for x in raw_input().split()]

t=0
q=0
r=0
# print time

for i in range(int(n)):
	t= t+time[i]
	q = q+max(0,marks[i]-t*c)
	
t = 0

for i in range(int(n-1),-1,-1):
	t= t+time[i]
	r = r+max(0,marks[i]-t*c)

if q>r :
	print "Limak"
elif q<r :
	print "Radewoosh"
else :
	print "Tie"
