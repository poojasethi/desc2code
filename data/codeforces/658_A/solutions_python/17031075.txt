import sys, getopt
from sys import stdin

limak = 0
rade = 0

def solve(score,constant,time,reverse):
    global limak,rade
    minus_time = 0
    if reverse == 0:
        for item in score:
            minus_time = minus_time + int(time[score.index(item)])
            tmp = (int(item) - int(constant) * int(minus_time))
            limak = limak + max(0,tmp)
    else:
        for item in reversed(score):
            minus_time = minus_time + int(time[score.index(item)])
            tmp = (int(item) - int(constant) * int(minus_time))
            rade = rade + max(0,tmp)

def main(argv):
        global limak,rade

        shit = stdin.readline().split()
        n = shit[0]
        c = shit[1]
        scores = stdin.readline().split()
        time = stdin.readline().split()

        solve(scores,c,time,0)
        solve(scores,c,time,1)
        
        if limak == rade:
                print 'Tie'
        elif limak > rade:
                print 'Limak'
        else:
                print 'Radewoosh'
  
        
if __name__ == "__main__":
   main(sys.argv[1:])
