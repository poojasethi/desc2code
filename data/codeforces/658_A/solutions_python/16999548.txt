n,c=raw_input().split()
n,c=int(n),int(c)
p=raw_input().split()
t=raw_input().split()
xl=0
ml=0
xr=0
mr=0
for i in range (n) :
    p[i]=int(p[i])
    t[i]=int(t[i])
for i in range (n) :
    xl+=t[i]
    xr+=t[n-i-1]
    if p[i]-c*xl>0 :
        ml+=p[i]-c*xl
    if p[n-i-1]-c*xr>0 :
        mr+=p[n-i-1]-c*xr
if ml>mr :
    print "Limak"
if ml<mr :
    print "Radewoosh"
if ml==mr :
    print "Tie"
