import math

n, c = map(int, raw_input().split(' '))
a = map(int, raw_input().split(' '))
b = map(int, raw_input().split(' '))

sa, sb, t = 0, 0, 0
for i in range(0,n):
    t += b[i]
    sa += max(0,a[i]-c*t)
t = 0
for i in range(n-1,-1,-1):
    t += b[i]
    sb += max(0,a[i]-c*t)

if sa > sb:
    print 'Limak'
elif sa == sb:
    print 'Tie'
else:
    print 'Radewoosh'
