n, c = map(int, raw_input().split())
p = map(int, raw_input().split())
t = map(int, raw_input().split())

l_s = 0
l_t = 0
for i in xrange(n):
	l_t += t[i]
	l_s += max(0, p[i] - c * l_t)
r_s = 0
r_t = 0
for i in xrange(n-1, -1, -1):
	r_t += t[i]
	r_s += max(0, p[i] - c * r_t)

if l_s > r_s:
	print "Limak"
elif l_s < r_s:
	print "Radewoosh"
else:
	print "Tie"
