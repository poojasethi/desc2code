n, c = map(int, raw_input().split())
ps = map(int, raw_input().split())
ts = map(int, raw_input().split())
c_t = s_l = s_r = 0
for p, t in map(None, ps, ts):
	c_t += t
	s_l += max(0, p - c*c_t)
c_t = 0
for p, t in map(None, ps[::-1], ts[::-1]):
	c_t += t
	s_r += max(0, p - c*c_t)
print 'Limak' if s_l > s_r else 'Radewoosh' if s_r > s_l else 'Tie'
