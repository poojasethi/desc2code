#!/usr/bin/python

import sys
from math import *

def Nil(): return map(int, sys.stdin.readline().split())

n, c = Nil()
p = Nil()
t = Nil()

limak = 0
time = 0
for i in range(n):
    time += t[i]
    limak += max(0, p[i] - c * time)

#print limak

rad = 0
time = 0
for _i in range(n):
    i = n - 1 - _i
    time += t[i]
    rad += max(0, p[i] - c * time)

#print rad

if rad < limak:
    print "Limak"
elif rad > limak:
    print "Radewoosh"
else:
    print "Tie"
