(n,c)  = map(int,raw_input().split(' '))
masp  = map(int,raw_input().split(' '))
mast = map(int,raw_input().split(' '))

pl = int()
pr = int()
tl = int()
tr = int()

for i in range(n):
    tl += mast[i]
    pl += max(0,masp[i]-c*tl)
    tr += mast[-(i+1)]
    pr += max(0,masp[-(i+1)]-c*tr)

if pl>pr:
    print "Limak"
elif pr>pl:
    print("Radewoosh")
else:
    print("Tie")
