n, c = map(int, raw_input().split())
px = map(int, raw_input().split())
xx = map(int, raw_input().split())
pp = 0
t = 0

for p, x in zip(px, xx):
    pp += max(0, p - c*(x+t))
    t += x
q = 0
t = 0
for p, x in zip(px, xx)[::-1]:
    q += max(0, p - c*(x+t))
    t += x
if pp > q:
    print 'Limak'
elif q > pp:
    print 'Radewoosh'
else:
    print 'Tie'
