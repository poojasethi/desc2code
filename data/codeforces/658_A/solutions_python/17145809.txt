n,c = map(int,raw_input().split())

score = map(int,raw_input().split())
time = map(int,raw_input().split())



L = 0
R = 0

Ltime = 0

for i in range(n):
    Ltime += time[i]
    L += max(0,score[i]-c * Ltime)

Rtime = 0

for i in range(n-1,-1,-1):
    Rtime += time[i]
    R += max(0,score[i]-c * Rtime)


if L > R:
    print "Limak"
elif L < R:
    print "Radewoosh"
else:
    print "Tie"
