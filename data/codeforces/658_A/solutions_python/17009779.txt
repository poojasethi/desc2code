def maxi(a,b): 
    if a>b: return a
    else: return b
n,c = raw_input().split(' ')
n,c = int(n), int(c)
p = [int(x) for x in raw_input().split(' ')]
t = [int(x) for x in raw_input().split(' ')]
L = 0
TL = 0
i = 0
for x in p:
    TL +=t[i]
    L += maxi(x - c*TL, 0)
    i+=1
i-=1
R = 0
TR = 0
for x in xrange(0,len(p)):
    TR += t[i]
    R += maxi(0, p[i] - c*TR)
    i -=1 
if L==R: print 'Tie'
elif L>R: print 'Limak'
else: print 'Radewoosh'
