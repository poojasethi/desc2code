def score(p, t, c):
    curr = ans = 0
    for i, x in enumerate(p):
        curr += t[i]
        ans += max(0, x - c * curr)
    return ans


def solve(p, t, c):
    s1 = score(p, t, c)
    s2 = score(p[::-1], t[::-1], c)
    if s1 > s2:
        return 'Limak'
    elif s1 < s2:
        return 'Radewoosh'
    else:
        return 'Tie'


n, c = map(int, raw_input().split())
p = map(int, raw_input().split())
t = map(int, raw_input().split())
print solve(p, t, c)
