nc,p,t = raw_input().split(),raw_input().split(),raw_input().split()
n, c = int(nc[0]), int(nc[1])

def getTotal(r):
    tot = 0
    score = 0
    for i in r:
        tot += int(t[i])
        score += max(0, int(p[i]) - (c * tot))

    return score

limak, rade = getTotal(range(n)), getTotal(reversed(range(n)))

#print limak, rade

if limak > rade:
    print "Limak"
elif limak < rade:
    print "Radewoosh"
else:
    print "Tie"
