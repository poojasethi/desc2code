n,c=map(int,raw_input().split())
N=map(int,raw_input().split())
T=map(int,raw_input().split())
t=0
l=0
add=0
diff=0
for i in range(n):
    t+=T[i]
    add+=max(N[i]-c*t,0)
    l+=T[n-i-1]
    diff+=max(N[n-i-1]-c*l,0)
if add>diff:
    print "Limak"
elif  add<diff:
    print "Radewoosh"
else:
    print "Tie"
