N,c = map(int,raw_input().split())
p = map (int,raw_input().split())
T = map (int,raw_input().split())

l = 0
r = 0
t = 0

for i in xrange (N):
	if p[i]-c*(T[i]+t) > 0 :
		l += p[i]-c*(T[i]+t)
	t += T[i]

t = 0
i = N-1
while i >=0 :
	if p[i]-c*(T[i]+t) > 0 :
		r += p[i]-c*(T[i]+t)
	
	t += T[i]
	i-=1;

if l > r:
	print "Limak"
elif l < r:
	print "Radewoosh"
else :
	print "Tie"
