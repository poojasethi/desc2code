#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
#define rep(i,n) for(int i=1;i<=n;i++)
vector<int> adj[100100];
int ans[100100],used[100100],n;
bool con(int a,int b)
{
	int i;
	if(a==0||b==0)return true;
	for(i=0;i<4;i++)
		if(adj[a][i]==b)
			return true;
	return false;
}

  
  
 void dfs(int p)
{
	int i;
	if(p==n)
	{
		if(!(con(ans[n-1],ans[1]) && con(ans[n],ans[1])&& con(ans[n],ans[2])) )
			return;
		for(i=1;i<=n;i++)
			printf("%d ",ans[i]);
		exit(0);
	}
	for(i=0;i<4;i++)
	{
		ans[p+1]=adj[ans[p]][i];
		if(used[ans[p+1]] || !con(ans[p+1],ans[p-1]) )
			continue;
		used[ans[p+1]]=1;
		dfs(p+1);
		used[ans[p+1]]=0;
	}
}
 
 
    
  
int main()
{
  //int n;
  cin>>n;
  rep(i,2*n)
  {
    int x,y;
    cin>>x>>y;
    adj[x].push_back(y);
    adj[y].push_back(x);
  }
  rep(i,n)
  if(adj[i].size()!=4)
  {cout<<"-1";return 0;}
  ans[1]=1;  
  used[1]++;
  dfs(1);//cout<<endl<<x<<endl;
  
  cout<<"-1 ";
  return 0;
}
