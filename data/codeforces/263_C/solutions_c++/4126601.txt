#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#include <set>
#include <map>
#include <queue>
#define pb(a) push_back(a)
#define mp(a,b) make_pair(a,b)
using namespace std;
const int N = 100005;
int n,vis[N]={0};
vector<int>a[N];
map<int,bool>m[N];
vector<int>ans;
int main(){
    scanf("%d",&n);
    for(int i=0;i<2*n;i++){
        int u,v;
        scanf("%d%d",&u,&v);
        a[u].pb(v);
        a[v].pb(u);
        m[u][v]=m[v][u]=true;
    }
    for(int i=1;i<=n;i++){
        if(a[i].size()!=4){
            puts("-1");
            return 0;
        }
    }
    if(n==5){
        printf("1 2 3 4 5\n");
        return 0;
    }
    int now=1,last=1;
    for(int t=1;t<n;t++){
        bool ok=false;
        ans.pb(now);
        vis[now]=1;
        for(int i=0;i<4;i++){
            int cur=a[now][i],cnt=0;
            if(vis[cur]) continue;
            for(int j=0;j<4;j++){
                if(m[now][a[cur][j]])
                    cnt++;
            }
            if(cnt>1&&!vis[cur]&&m[last][cur]){
                ok=true;last=now;now=cur;break;
            }
        }
        if(!ok){
            puts("-1");
            return 0;
        }
    }
    ans.pb(now);
    for(int i=0;i<ans.size();i++)
        printf("%d ",ans[i]);
    puts("");
    return 0;
}
