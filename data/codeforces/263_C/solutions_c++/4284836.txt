#define pb push_back
int n;

vector<int>g[100010];
int vis[100010];
int ans[100010];
bool dfs(int a,int b,int f=0,int k=1){
  ans[k-1]=a,ans[k]=b;
  if(k<n){
    vis[b]=1;
    for(int i=0;i<4;++i){
      int c=g[b][i];
      if(c==f||vis[c])continue;
      if((g[a][0]==c||g[a][1]==c||g[a][2]==c||g[a][3]==c)&&dfs(b,c,a,k+1))return 1;
    }
    vis[b]=0;
    return 0;
  }
  else return 1==b;
}


int main()
{
    cin>>n;
    for(int i=0;i<2*n;i++)
    {
        int x,y;cin>>x>>y;
        g[x].pb(y);
        g[y].pb(x);
    }
    for(int i=1;i<=n;i++)if(g[i].size()!=4){cout<<"-1";return 0;}else vis[i]=0;

    // cout<<"dfs";
    //ans[0]=1;vis[1]=1;
    for(int i=0;i<4;i++)
    if(dfs(1,g[1][i]))
    {
        for(int i=0;i<n;i++)cout<<ans[i]<<" ";return 0;
    }
    cout<<"-1";
  return 0;
}
