#include <cstdio>
#include <cstring>
#include<stdio.h>
#include<iostream>
using namespace std;
int n,g[100100][5],d[100100]={0},v[100100];
bool in[100100];
bool dfs(int a,int b,int k=1){
  v[k-1]=a,v[k]=b;
  if(k<n){
    in[b]=1;
//    cout<<"f = "<<f<<endl;
    for(int i=0;i<4;++i){
      int c=g[b][i];
     //if(c==f||in[c])continue;
      if(in[c])continue;

      if((g[a][0]==c||g[a][1]==c||g[a][2]==c||g[a][3]==c)&&dfs(b,c,k+1))return 1;
    }
    in[b]=0;
    return 0;
  }
  else return 1==b;
}
int main(void) {
  scanf("%d",&n);
  for(int i=n*2,a,b;i-->0&&2==scanf("%d%d",&a,&b);){
  //  cout<<a<<"  "<<d[a]<<"  "<<b<<"    "<<d[b]<<endl;
    g[a][d[a]++]=b,g[b][d[b]++]=a;

    if(d[a]>4||d[b]>4){puts("-1");return 0;}
  }

//cout<<g[1][0]<<"  "<<g[1][1]<<"  "<<g[1][2]<<"  "<<g[1][3]<<endl;

  for(int i=0;i<4;++i){
    if(dfs(1,g[1][i])){
      for(int j=0;j<n;++j)printf("%d ",v[j]);puts("");
      return 0;
    }
  }
  puts("-1");
}
