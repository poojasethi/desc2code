#include <stdio.h>
#include <vector>
#include <queue>
#include <stack>
#include <map>
#include <cmath>
#include <cstring>
#include <algorithm>

using namespace std;

map <int,bool> num[100005];
vector <int> num2[100005];
vector <int> resp;
int vis[100005];

int main()
{
	int i, j, n,comum,a,b;
	scanf(" %d", &n);
	for(i=0;i<2*n;i++)
	{
		scanf(" %d %d", &a, &b);
		num2[a].push_back(b);
		num2[b].push_back(a);
		num[a][b]=true;
		num[b][a]=true;
	}
	for(i=1;i<=n;i++)
		if(num2[i].size()!=4)
		{
			printf("-1");
			return 0;
		}
	int last = 1, last2=1;
	for(i=0;i<n;i++)
	{
		bool mudou = false;
		vis[last]=1;
		resp.push_back(last);
		for(j=0;j<4;j++)
		{
			comum = 0;
			int q = num2[last][j];
			for(int k=0;k<4;k++)
				if(num[last][num2[q][k]])
					comum++;
			if(comum>1 && !vis[q] && num[last2][q])
			{
				last2=last;
				last=q;
				mudou = true;
				break;
			}
		}
		if(!mudou && i+1!=n)
		{
			printf("-1");
			return 0;
		}
	}
	for(i=0;i<n;i++)
		printf("%d ", resp[i]);
	return 0;
}
