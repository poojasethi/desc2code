#include <bits/stdc++.h>

using namespace std;

typedef pair<int,int> pii;
typedef vector<int> vi;
typedef map<int,int> mii;

#define IOS ios::sync_with_stdio(false);
#define pb push_back
#define po pop_back
#define F first
#define S second

const int MX = 1e5 + 2;

bool mark[MX],ok;
map < pii , bool > id;
vi v[MX],ans;
int n;

void dfs ( int x , int p , int t)
{
	mark[x] = true;
	ans.pb ( x );
    for (int i=0;i<v[x].size();i++){
    	int h = v[x][i];
    	if ( !mark[h] && id[ pii ( h , p ) ] ){
    		dfs ( h , x , t+1 );
    	}
    	else if ( id[ pii ( h , p ) ] && h == 1 && t == n-1 ){
			for (int j=0;j<ans.size();j++)
				cout << ans[j] << " ";
		    exit(0);
    	}
    }
    ans.po();
    mark[x] = false;
}

int main(){
	IOS
	cin >> n;
	for (int i=1;i<=2*n;i++){
		int x,y;
		cin >> x >> y;
		id[ pii ( x , y ) ] = true;
		id[ pii ( y , x ) ] = true;
		v[x].pb ( y );
		v[y].pb ( x );
	}
	for (int i=1;i<=n;i++)
		if ( v[i].size() > 4 ){
			cout << -1;
			exit(0);
		}
	ans.pb (1);
	mark[1] = true;
	for (int i=0;i<v[1].size();i++){
		dfs ( v[1][i] , 1 , 1 );
	}
    cout << -1;
    return 0;
}
