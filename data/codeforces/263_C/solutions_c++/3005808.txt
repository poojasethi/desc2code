#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int n,e[100005][4],ce[100005],v[100005];

int main() {
	scanf("%d",&n);
	memset(ce,0,sizeof(ce));
	for (int i=0; i<2*n; i++) {
		int a,b;
		scanf("%d%d",&a,&b);
		if (ce[a] == 4 || ce[b] == 4) break;
		e[a][ce[a]++] = b;
		e[b][ce[b]++] = a;
	}

	for (int i=1; i<=n; i++)
		if (ce[i] != 4) {
			puts("-1");
			return 0;
		}

	int cnt,at;
	for (int i=0; i<4; i++)
		for (int j=0; j<4; j++)
			if (i != j) {
				at = 1;
				cnt = 1;
				memset(v,0,sizeof(v));
				v[at] = 1; ce[0] = at;
				v[e[1][i]] = 1; ce[cnt++] = e[1][i];
				v[e[1][j]] = 1; ce[cnt++] = e[1][j];
				at = e[1][j];
				while (cnt < n) {
					int ti = 0,x;
					for (x=0; x<4; x++) {
						if (v[e[at][x]] != 0) continue;
						ti = 0;
						for (int y=0; y<4; y++)
							if (e[e[at][x]][y] == ce[cnt-1] || e[e[at][x]][y] == ce[cnt-2]) {
								ti++;
							}
						if (ti == 2) {
							at = e[at][x];
							v[at] = 1; ce[cnt++] = at;
							break;
						}
					}
					if (x == 4) break;
				}

				if (cnt == n) {
					for (int i=0; i<n; i++) {
						if (i > 0) printf(" ");
						printf("%d",ce[i]);
					}
					puts("");
					return 0;
				}
			}
	puts("-1");
	return 0;
}
