#include<iostream>
using namespace std;
#define Max 210005
#define ll long long int
ll a[Max];
ll ed[Max][4];
ll fuck[Max];
ll n;
ll ans[Max];
bool vis[Max];
bool dfs(ll x,ll loc)
{
    ans[loc] = x;
    if (loc == n - 1)return true;
    vis[x] = true;
    ll i, j;
    for (i = 0; i < 4; ++i)
    {
        if (ed[x][i] == 0)
            continue;
        if (loc == 0)
        {
            if (dfs(ed[x][i], loc + 1))
                return true;
        }
        else if (!vis[ed[x][i]])
            for (j = 0; j < 4; ++j)
            {
                if (ed[x][i] == ed[ans[loc - 1]][j])
                {
                    if (dfs(ed[x][i], loc + 1))
                        return true;
                    break;
                }
                    
            }
    }
    vis[x] = false;
    return false;
    
}
int main()
{
    ios_base::sync_with_stdio(0);
    cin >> n;
    ll i, j;
    ll x, y;
    for (i = 0; i < 2 * n; ++i)
    {
        cin >> x >> y;
        x--;
        y--;
        ed[x][fuck[x]++] = y;
        ed[y][fuck[y]++] = x;
    }
    if (!dfs(0, 0))
    {
        cout << -1 << endl;
        return 0;
    }
    for (i = 0; i < n; ++i)
        cout << ans[i] + 1 << " ";
    cout << endl;
    return 0;
}
