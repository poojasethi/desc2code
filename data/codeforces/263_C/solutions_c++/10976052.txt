#include <cstdio>
#include <iostream>
#include <vector> 
#include <cstring>
using namespace std;
const int maxn=100500;
int n;
vector<int> e[maxn];
int ans[maxn];
bool vis[maxn];
bool dfs(int u,int v,int deep)
{
	if(deep==n)
	{
		return true;
	}
	if(vis[v]) return false;
	vis[v]=true;
	ans[deep]=v;
	if(u==-1)
	for(int i=0;i<4;i++)
	{
		if(dfs(v,e[v][i],deep+1))
		return true;
	}
	else
	{
		for(int i=0;i<4;i++)
		{
			int t=e[v][i];
			bool ok=false;
			for(int j=0;j<e[t].size();j++)
			{
			if(e[t][j]==u)
			ok=true;
			}
			if(ok)
			{
				if(dfs(v,t,deep+1)) return true;
			}
		}
	}
	vis[v]=false;
	return false;
}
int main()
{
	//freopen("111.txt","r",stdin);
	cin>>n;
	memset(vis,false,sizeof(vis));
	for(int i=0;i<2*n;i++)
	{
		int num1,num2;
		cin>>num1>>num2;
		e[num1].push_back(num2);
		e[num2].push_back(num1);
	}
	for(int i=1;i<=n;i++)
	{
		if(e[i].size()!=4)
		{
			cout<<"-1"<<endl;
			return 0;
		}
	}
	if(dfs(-1,1,0))
	{
		for(int i=0;i<n;i++)
		cout<<ans[i]<<" ";
	}
	else
	cout<<"-1"<<endl;
	return 0;
}
