#include <iostream>
#include <vector>
#include <map>
using namespace std;

const int MAXN=1000*100+10;
map <int, bool> adj[MAXN];
vector <int> v[MAXN];
vector <int> ans;
bool mark[MAXN];

int main(){
    int n;
    cin>>n;
    for (int i=0; i<2*n; i++){
        int a, b;
        cin>>a>>b;
        v[a].push_back(b);
        v[b].push_back(a);
        adj[a][b]=true;
        adj[b][a]=true;
    }
    for (int i=1; i<=n; i++)
        if (v[i].size()!=4){
            cout<<-1<<endl;
            return 0;
        }
    int x=1, y=1;
    for (int i=0; i<n; i++){
        bool flag=false;
        mark[x]=true;
        ans.push_back(x);
        for (int j=0; j<4; j++){
            int num=0;
            int cur=v[x][j];
            for (int k=0; k<4; k++)
                if (adj[x][v[cur][k]])
                    num++;
            if (num>1 && !mark[cur] && adj[y][cur]){
                y=x;
                x=cur;
                flag=true;
                break;
            }
        }
        if (!flag && i+1!=n){
            cout<<-1<<endl;
            return 0;
        }
    }
    for (int i=0; i<n; i++)
        cout<<ans[i]<<" ";
    cout<<endl;
    return 0;
}
