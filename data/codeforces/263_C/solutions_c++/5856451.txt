#include <stdio.h>
#include <iostream>
#include <cstring>
using namespace std;
struct node{
	int v,next;
}a[500000];
int h[100005],s=0,n,p[100005],l[100005]={0},m,u,v;
bool e,f[100005]={false},g=true,q,w;
bool back(int u,int c,int d){
	if (d==n) return true;
	if (!f[u]){
		f[u]=true;
		p[d]=u;
		if (c==-1)
			for (int i=h[u];i!=-1;i=a[i].next){
				v=a[i].v;
				if (back(v,u,d+1)) return true;
			}
		else
			for (int i=h[u];i!=-1;i=a[i].next){
				v=a[i].v;
				if (v==c) continue ;
				q=false,w=false;
				for (int j=h[v];j!=-1;j=a[j].next)
					if (a[j].v==u) q=true;
					else if (a[j].v==c) w=true;
				if (q && w)
					if (back(v,u,d+1))
						return true;
			}
		f[u]=false;
	}
    return false;
}
int main( ){
	scanf("%d",&n);
	m=n*2;
	memset(h,-1,sizeof(h));
	for (int i=0;i<m;i++){
		scanf("%d %d",&u,&v);
		a[s].v=v;
		a[s].next=h[u];
		h[u]=s++;
		l[u]++;
		a[s].v=u;
		a[s].next=h[v];
		h[v]=s++;
		l[v]++;
	}
	for (int i=1;i<=n;i++)
		if (l[i]!=4){
			g=false;
			break;
		}
	e=false;
	if (g) e=back(1,-1,0);
	if (e){
		for (int i=0;i<n;i++){
			if (i>0) cout<<' ';
			cout<<p[i];
		}
		cout<<endl;
	}else cout<<-1<<endl;
	return 0;
}
