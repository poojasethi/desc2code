#include<cstdio>
#define FOR(i,s,e)for (int i = (s); i < (e); i++)
#define N 1000005
int n, a, b;
int Q[N][4],Qnum[N];
bool AC=0,u[N],tmp;
bool ck(int a, int b){
     bool tmp = 0;
     FOR(i,0,4)
        if (Q[a][i] == b)tmp = 1;
     return tmp;
}
bool go(int pr, int v, int k){
     u[v] = 1;
     if (AC) return 1;
     if (k == n){
        if ((ck(pr,1)) && (ck(v,1))) {
           AC = 1;
           printf("%d ",v);
           return 1;
        }
        return 0;
     }
     
          FOR(i,0,4)
             if ((ck(pr,Q[v][i])) && (!u[Q[v][i]])  )
                if (go(v,Q[v][i],k+1)){
                   printf("%d ",v);
                   return 1;
                }
     
     u[v] = 0;
     return 0;
}
int main(){
    scanf("%d",&n);
    FOR(i,1,n+1)Qnum[i] = 0;
    FOR(i,0,2*n){
       scanf("%d%d",&a,&b);
       Q[a][Qnum[a]++] = b;
       Q[b][Qnum[b]++] = a;
    }
    FOR(i,2,n+1)u[i] = 0; u[1] = 1;
    FOR(i,0,4)
       if (!AC)
          tmp = go(1,Q[1][i],2);
    if (AC)printf("1\n");
    else printf("-1\n");
    return 0;
}
