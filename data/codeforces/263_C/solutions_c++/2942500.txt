#include<iostream>
#include<vector>
using namespace std;

const int N = 100100;

int n, n1, n2, nr, x[N];
vector<int> v[N];
bool ver[N];

bool sol() {
	int i, j;
	
	for(i = 1; i<=n; ++i)
		ver[i] = 0;
	ver[n1] = 1;
	ver[n2] = 1;
	nr = 0;
	x[++nr] = 1;
	x[++nr] = n1;
	int vvv = 0;
	
	while(n2 != 1 &&!vvv) {
		if(n2 == x[nr])
			return 0;
		
		x[++nr] = n2;
		
		if(nr > n)
			return 0;
		int vver = 0;
		ver[n2] = 1;
		for(i = 0; i < 4; ++i) if(!vver && !ver[v[n1][i]])
			for(j = 0; j < 4; ++j)  if(!ver[v[n2][j]] && v[n1][i] == v[n2][j]) {
				
				if(v[n2][j] == 1) {
					if(nr == n) {
						vver = 1;
						vvv = 1;
						break;
					}
					else
						continue;
				}
				
				n1 = n2;
				n2 = v[n2][j];
				
				vver = 1;
				break;
			}
	}
	
	for(i = 1; i<=nr; ++i)
		cout << x[i] << " ";
	return 1;
}

int main() {
	int i, a, b;
	
	//freopen("ttt", "r", stdin);
	
	cin >> n;
	
	for(i = 1; i<=2 * n; ++i) {
		cin >> a >> b;
		v[a].push_back(b);
		v[b].push_back(a);
	}
	
	for(i = 1; i<=n; ++i)
		if(v[i].size() != 4) {
			cout << -1;
			return 0;
		}
	
	for(a = 0; a < 4; ++a)
		for(b = 0; b < 4; ++b) if(b != a) {
			n1 = v[1][a];
			n2 = v[1][b];
			if(sol())
				return 0;
		}
	cout << "-1";
	
	return 0;
}
