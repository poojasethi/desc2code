#define MAX 100002

int ans[MAX];
vector<int>v[100002];
int vis[MAX];

int n;
bool con(int a,int b){
	fr(i,4){
		if(v[a][i]==b){
			return true;
		}
	}
	return false;
}

void dfs(int t,int ind){
	vis[t]=1;

	if(ind==n){
		fr1(i,n){
			cout<<ans[i]<<" ";
		}
		flag=true;
		return;
	}
	fr(i,v[t].size()){
		if(!vis[v[t][i]]&&con(v[t][i],ans[ind-1])){
			ans[ind+1]=v[t][i];
			dfs(v[t][i],ind+1);
		}
	}
	vis[t]=0;
}
