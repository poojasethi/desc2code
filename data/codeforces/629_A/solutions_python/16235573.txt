import fileinput

# input
source = fileinput.input()

# source = open('data.txt', 'r').readlines()

data = []
for line in source:
	data.append(line.strip())
data = data[1:]

h = 0
for i in range(0, len(data)):
	cs = 0
	for c in data[i]:
		if c == 'C':
			cs += 1
	h += cs * (cs - 1) / 2

for i in range(0, len(data)):
	cs = 0
	for j in range(0, len(data)):		
		if data[j][i] == 'C':
			cs += 1
	
	h += cs * (cs - 1) / 2


print h
