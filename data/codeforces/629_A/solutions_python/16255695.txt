#!/usr/bin/python
import sys

n = input()
ans = 0

cols = []
for i in xrange(n):
    cols += [0]

for line in sys.stdin:
    k = line[:-1].count("C")
    ans += k * (k - 1) / 2
    for i in xrange(n):
        cols[i] += int(line[i] == "C")
        
for i in xrange(n):
    ans += cols[i] * (cols[i] - 1) / 2

print ans
