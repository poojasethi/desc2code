#include<bits/stdc++.h>
using namespace std;
int col[200],r[200];
int main()
{
    int n;char x;cin >> n;
    for(int i = 0;i<n;i++)
        for(int j = 0;j<n;j++)
    {cin >> x;
    if(x=='C')r[i]++,col[j]++;}
    long long ans = 0;
    for(int i = 0;i<n;i++)
        ans+=r[i]*(r[i]-1)/2;
    for(int i = 0;i<n;i++)
        ans+=col[i]*(col[i]-1)/2;
    cout << ans << endl;
}
