#include <cstdio>

using namespace std;

int n,x[100],y[100],i,j,ans;
int main()
{
    scanf("%d ",&n);
    for(i=0;i<n;++i)
    {
        for(j=0;j<n;++j)
        {
            if(getchar()=='C')
            {
                ans+=x[i]++;
                ans+=y[j]++;
            }
        }
        getchar();
    }
    printf("%d\n",ans);
    return 0;
}
