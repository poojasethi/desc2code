#include<stdio.h>
#define M 100
using namespace std;
char s[M + 5][M + 5];
int n , ans;
main(){
	scanf("%d",&n);
	for(int i = 0 ; i < n ; i++)	scanf("%s",s[i]);
	for(int i = 0 ; i < n ; i++)
		for(int j = 0 ; j < n ; j++)
			for(int k = j + 1 ; k < n ; k++)
				ans += ( (s[i][j] == 'C' && s[i][k] == 'C') + (s[j][i] == 'C' &&s[k][i] == 'C') );
	printf("%d",ans); 
}
