#include <iostream>

using namespace std;

string S;
int A[105],B[105],N;
long long rs = 0;
int main(){
	cin >> N;
	getline(cin,S);
	for(int i = 0;i<N;i++){
		getline(cin,S);
		for(int j = 0;j<int(S.length());j++) if(S[j] == 'C') A[i]++,B[j]++; 
	}
	for(int i = 0;i<N;i++)
		rs+=(A[i]*(A[i]-1))/2 + (B[i]*(B[i]-1))/2;
		
	cout << rs;
	return 0;
}
