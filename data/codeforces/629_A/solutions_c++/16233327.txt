#include <iostream>

using namespace std;

int n;
int col[200];
int fil[200];

int computa(int k)
{
  return k*(k-1)/2;
}

int main()
{
  cin>>n;
  for (int i=1;i<=n;i++) {
    for (int j=1;j<=n;j++) {
      char c;
      cin>>c;
      fil[i]+=c=='C';
      col[j]+=c=='C';
    }
  }
  int sol=0;
  for (int i=1;i<=n;i++)
    sol+=computa(fil[i])+computa(col[i]);
  cout<<sol<<endl;
}
