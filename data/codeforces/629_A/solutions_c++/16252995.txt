#include<iostream>
#include<cmath>
using namespace std;
int x,i,j,o,k=0;
char m[200][200];
int main(){
    cin>>x;
	for(i=0;i<x;i++){
		o=0;
		for(j=0;j<x;j++){
			cin>>m[i][j];
			if(m[i][j]=='C')o++;
		}
		k+=o*(o-1)/2;
	}
	for(i=0;i<x;i++){
		o=0;
		for(j=0;j<x;j++){
			if(m[j][i]=='C')o++;
		}
		k+=o*(o-1)/2;
	}	
	cout<<k;
	return 0;
}
