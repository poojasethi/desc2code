#include <bits/stdc++.h>
using namespace std;
int main()
{
int n,i,j,sum=0,k=0;
char a[100][100];
cin>>n;
for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
{
cin>>a[i][j];
if(a[i][j]=='C')
sum++;
}
k+=sum*(sum-1)/2;
sum=0;
}
sum=0;
for(i=0;i<n;i++)
{
for(j=0;j<n;j++)
{
if(a[j][i]=='C')
sum++;
}
k+=sum*(sum-1)/2;
sum=0;
}
cout<<k<<endl;
return 0;}
