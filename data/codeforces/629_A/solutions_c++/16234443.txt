#include<bits/stdc++.h>
using namespace std;
int main(){
int i,j,n,ct;
cin>>n;
char a[100][100];
unsigned long int ans=0;
for(i=0;i<n;i++){ct=0;
	for(j=0;j<n;j++)
		{cin>>a[i][j]; if(a[i][j]=='C') ct++;}
	ans+=((ct-1)*ct)/2;
}
for(j=0;j<n;j++){ct=0;
	for(i=0;i<n;i++)
		{if(a[i][j]=='C') ct++;}
	ans+=((ct-1)*ct)/2;
}
cout<<ans<<"\n";
return 0;
}
