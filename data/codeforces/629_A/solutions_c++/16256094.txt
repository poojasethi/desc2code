#include<cstdio>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	char tab[n][n];
	for(int i=0;i<n;i++)
	{
		scanf(" ");
		for(int j=0;j<n;j++)
			scanf("%c",&tab[i][j]);
	}
	int w=0;
	for(int i=0;i<n;i++)
	{
		int s1=0;
		int s2=0;
		for(int j=0;j<n;j++)
		{
			s1+=(tab[i][j]=='C');
			s2+=(tab[j][i]=='C');
		}
		w+=(s1*(s1-1))/2+(s2*(s2-1))/2;
	}
	printf("%d\n",w);
	return 0;
}
