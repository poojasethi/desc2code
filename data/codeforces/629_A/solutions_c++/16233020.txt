#include <iostream>

using namespace std;

char T[1000][1000];
int W[1000];

int main(){
	int n,res,x,i,j;
	res = 0;
	cin >> n;
	for(i=0; i<n; i++){
		x = 0;
		for(j=0; j<n; j++) {
			cin >> T[i][j];
			if(T[i][j] == 'C') {
				x++;
				W[j]++;
			}
		}
		res += (x*(x-1))/2;
	}
	for(j=0; j<n; j++) res += (W[j]*(W[j]-1))/2;
	cout << res << "\n";
	return 0;
}
