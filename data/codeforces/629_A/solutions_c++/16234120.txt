#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n;
	cin>>n;
	char a;
	int r[n]={0};
	int c[n]={0};
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>a;
			if(a=='C')
			{
				r[i]++;
				c[j]++;
			}
		}
	}
	int sum=0;
	for(int i=0;i<n;i++)
	{
		sum+=(r[i]*(r[i]-1)/2)+(c[i]*(c[i]-1)/2);
	}
	cout<<sum;
}
