#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll row[200],col[200];

int main()
{
ll n,i,j,ans=0;
string s;
char c;
cin>>n;

for(i=1;i<=n;i++)
{
cin>>s;
for(j=0;j<n;j++)
{
c=s[j];
if(c=='C')
{
row[i]++;
col[j+1]++;
}
}
}

for(i=1;i<=n;i++)
{
ans=ans + (row[i]*(row[i]-1))/2;
ans=ans+(col[i]*(col[i]-1))/2;
}
cout<<ans;
return 0;
}
