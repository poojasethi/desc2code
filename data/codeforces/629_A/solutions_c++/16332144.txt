#include<iostream>
using namespace std;
char c[110][110];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			cin>>c[i][j];
		}
	}
	long long ans=0;
	for(int i=0;i<n;i++)
	{
		int s1=0,s2=0;
		for(int j=0;j<n;j++)
		{
			if(c[i][j]=='C') s1++;
			if(c[j][i]=='C') s2++;
		}
		ans+=s1*(s1-1)/2;
		ans+=s2*(s2-1)/2;
	}
	cout<<ans;
}
