#include <bits/stdc++.h>
using namespace std;
int main()
{
	int i,j,k,n,p=0;
	char s[105][105];
	cin>>n;
	for(i=0;i<n;i++)
		cin>>s[i];
	for(i=0;i<n;i++)
	{
		k=0;
		for(j=0;j<n;j++)
			if(s[i][j]=='C')
				k++;
		p+=(k-1)*k/2;
	}
	//cout<<p<<endl;
	for(i=0;i<n;i++)
	{
		k=0;
		for(j=0;j<n;j++)
			if(s[j][i]=='C')
				k++;
		p+=(k-1)*k/2;
	}
	cout<<p<<endl;
	return 0;
}
