#include <bits/stdc++.h>
using namespace std;
int row[120], col[120];
int main()
{
	char c;
	int n;
	scanf("%d", &n);
	for(int i = 0; i<n; i++)
		for(int j = 0; j<n; j++)
		{
			scanf(" %c", &c);
			if(c=='C')
			{
				row[i]++;
				col[j]++;
			}
		}
	int sol = 0;
	for(int i = 0; i<n; i++)
	{
		sol += (row[i]*(row[i] - 1))/2;
		sol += (col[i]*(col[i] - 1))/2;
	}
	printf("%d\n", sol);
}
