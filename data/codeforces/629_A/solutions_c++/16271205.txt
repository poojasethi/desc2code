#include<cstdio>
int a[105];
int main()
{
	int n,i,j;
	char ch;
	scanf("%d",&n);
	int ans=0,cnt;
	for(i=1;i<=n;i++)
	{
		cnt=0;
		for(j=1;j<=n;j++)
		{
			scanf(" %c",&ch);
			if(ch=='C')
			{
				cnt++;
				a[j]++;
			}
		}
		ans+=(cnt-1)*cnt;
	}
	for(i=1;i<=n;i++)
		ans+=a[i]*(a[i]-1);
	printf("%d\n",ans/2);
	return 0;
}
