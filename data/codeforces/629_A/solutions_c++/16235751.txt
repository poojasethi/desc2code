#include<cstdio>
#include<cstring>
int main()
{
	int T,i,j,s=0,s1;
	scanf("%d",&T);
	char a[T][T+1];
	for(i=0;i<T;i++)
		scanf("%s",a[i]);
	for(i=0;i<T;i++){
		s1=0;
		for(j=0;j<T;j++){
			if(a[i][j]=='C') s1++;
		}
		if(s1>1) s+=s1*(s1-1)/2;
	}
	for(i=0;i<T;i++){
		s1=0;
		for(j=0;j<T;j++)
			if(a[j][i]=='C') s1++;
		if(s1>1) s+=s1*(s1-1)/2;
	}
	printf("%d",s);
}
