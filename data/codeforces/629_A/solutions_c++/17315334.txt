#include<bits/stdc++.h>
using namespace std;
long long  n,i,j,a[105],b[105],ans;
char q;
int main()
{
    cin >>n;
    for (i=1;i<=n;++i)
    {
        for (j=1;j<=n;++j)
        {
            cin >>q;
            if (q=='C'){a[i]++;b[j]++;}else continue;
            //cout <<q;
            ans+=a[i]-1;
            ans+=b[j]-1;
        }
    }
    cout <<ans;
}
