#include<iostream>
using namespace std;
long ans,b[1000],c[1000],i,j,n;
char a;
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)
	for(j=1;j<=n;j++)
	{
		cin>>a;
		if(a=='C')
		{
			b[i]++;
			c[j]++;
		}
	}
	ans=0;
	for(i=1;i<=n;i++)
	ans+=b[i]*(b[i]-1);
	for(i=1;i<=n;i++)
	ans+=c[i]*(c[i]-1);
	cout<<ans/2;
}
