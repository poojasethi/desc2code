#include <cstdio>
int a[105],b[105];
int main()
{
	int n;
	char c;
	scanf("%d",&n);
	c=getchar();
	for (int i=0;i<n;i++)
	{
		for (int j=0;j<n;j++)
		{
			c=getchar();
			if (c=='C')
			{
				a[i]++;
				b[j]++; 
			}
		}
		c=getchar();
	}
	int ans=0;
	for (int i=0;i<n;i++)
	{
		ans+=a[i]*(a[i]-1)/2;
		ans+=b[i]*(b[i]-1)/2;
	}
	printf("%d",ans);
}
