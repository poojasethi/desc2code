#include<stdio.h>
int n,z,i,j,x;
char A[110][110];
int main()
{
	scanf("%d",&x);
	for(i=1;i<=x;i++)
	{
		for(j=1;j<=x;j++)
		{
			scanf(" %c",&A[i][j]);
			if(A[i][j]=='C') z++;
		}
		n+=z*(z-1)/2;
		z=0;
	}
	for(i=1;i<=x;i++)
	{
		for(j=1;j<=x;j++)
		{
			if(A[j][i]=='C') z++;
		}
		n+=z*(z-1)/2;
		z=0;
	}
	printf("%d",n);
	
	
}
