
#include <iostream>

using namespace std;
long long n,S;
long a[1000][100];
char c;
int main()
{   cin>>n;
S=0;
for (int i=0; i<n; i++){
    for (int j=0; j<n; j++){
        cin>>c;
        if (c=='C') {if (a[i][1]>0) S+=a[i][1];  a[i][1]++;
        if (a[j][2]>0)  S+=a[j][2]; a[j][2]++; }
    }

}

    cout << S << endl;
    return 0;
}
