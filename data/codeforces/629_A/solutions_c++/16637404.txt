#include <iostream>

using namespace std;

int main()
{
    int n,r,c,ris=0;
    char l[100][100];
    cin >> n;
    for(int i=0;i<n;i++) cin >> l[i];
    for(int i=0;i<n;ris+=(c*(c-1))/2+(r*(r-1))/2,r=0,c=0,++i)
        for(int j=0;j<n;r+=(l[i][j]=='C'),c+=(l[j][i]=='C'),++j);
    cout << ris;
    return 0;
}
