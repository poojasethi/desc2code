#include<iostream>
using namespace std;
	char a[102][102];
	int n,r[102],c[102],res=0;
//.......................................
main(){
	cin>>n;
	for(int i=0;i<n;i++)
	 for(int j=0;j<n;j++)
	  	{
	  		cin>>a[i][j];
	  		if (a[i][j]=='C'){
	  			r[i]++;
	  			c[j]++;
	  		}
	  	}
	for(int i=0;i<n;i++) res=res+(r[i]*(r[i]-1))/2+(c[i]*(c[i]-1))/2;
	cout<<res;
}
