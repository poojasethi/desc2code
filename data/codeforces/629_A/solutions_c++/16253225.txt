#include<iostream>
using namespace std;
char w[111][111];
int n,row[111]={},col[111]={};
int main()
{
	cin>>n;
	for (int i=1;i<=n;i++)
	{
		for (int j=1;j<=n;j++)
		{
			cin>>w[i][j];
			if (w[i][j]=='.') continue;
			row[i]++;
			col[j]++;
		}
		
	}
	int ans=0;
	for (int i=1;i<=n;i++)
	{
		ans+=row[i]*(row[i]-1);
		ans+=col[i]*(col[i]-1);
	}
	cout<<ans/2;
}
