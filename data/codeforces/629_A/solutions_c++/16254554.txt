#include<iostream>
using namespace std;
int main(){
	int n,i,j,ans = 0;
	char x;
	cin>>n;
	int a[n],b[n];
	for(i = 0 ; i<n; i++)
		a[i] = 0,b[i] = 0;
	for(i = 0; i<n; i++){
		for(j = 0; j<n; j++){
			cin>>x;
			if(x == 'C')
				a[i]++,b[j]++;
		}
	}
	for(i = 0; i<n;i++)
		ans+=((a[i]*(a[i]-1)/2) + (b[i]*(b[i]-1))/2 );
	cout<<ans;
}
