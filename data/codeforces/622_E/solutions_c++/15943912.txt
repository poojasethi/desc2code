#include <stdio.h>
#include <vector>
#include <algorithm>
std::vector<int> V[500010];
int check[500010],x[500010],s;
void func(int k, int dist)
{
	int i;
	check[k]=1;
	if(V[k].size()==1) x[s++] = dist;
	else
	{
		for(i=0;i<V[k].size();i++)
		{
			if(check[V[k][i]]==1) continue;
			func(V[k][i],dist+1);
		}
	}
	check[k]=0;
}
int main()
{
	int max=0;
	int a,b,c,i,j;
	scanf("%d",&a);
	for(i=1;i<a;i++)
	{
		scanf("%d%d",&b,&c);
		V[b].push_back(c);
		V[c].push_back(b);
	}
	check[1]=1;
	for(i=0;i<V[1].size();i++)
	{
		s=1;
		b = V[1][i];
		func(b,0);
		std::sort(x+1,x+s);
		for(j=2;j<s;j++) if(x[j-1]>=x[j]) x[j]=x[j-1]+1;
		max = max>x[s-1]+1?max:x[s-1]+1;
	}
	printf("%d",max);
}
