#include <stdio.h>
#include <algorithm>
#include <string.h>
#include <memory.h>

#define NN 500010

using namespace std;

int head[NN], nxt[2*NN], to[2*NN], vst[NN], dis[NN], tmp[NN], cnt, ans;

void dfs(int u, int dp){
	dis[u]=dp;
	vst[u]=1;
	int fl=0;
	for(int edge=head[u]; edge!=-1; edge=nxt[edge]){
		int v=to[edge];
		if(vst[v]) continue;
		fl=1;
		dfs(v,dp+1);
		if(u==1){
			sort(tmp,tmp+cnt);
			for(int i=1; i<cnt; i++){
				tmp[i]=max(tmp[i],tmp[i-1]+1);
			}
			ans=max(ans,tmp[cnt-1]+1);
			cnt=0;
		}
	}
	if(!fl) tmp[cnt++]=dis[u]-1;
}

int main (){
	
	int N,i,u,v;
	scanf("%d",&N);
	int edge=0;
	memset(head,-1,sizeof(head));
	
	for(i=1; i<N; i++){
		scanf("%d %d",&u,&v);
		nxt[edge]=head[u], to[edge]=v, head[u]=edge++;
		nxt[edge]=head[v], to[edge]=u, head[v]=edge++;
	}
	
	ans=1;
	dfs(1,0);
	
	printf("%d\n",ans);
	return 0;
}
