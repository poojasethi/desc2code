#include<bits/stdc++.h>
using namespace std;

const int N = int(5e5)+5;
vector<int> ver[N];
vector<int> aa[N];
int idx=0;


void dfs(int v,int p,int l){
	for(int i=0;i<ver[v].size();i++){
		if(ver[v][i]!=p){
			dfs(ver[v][i],v,l+1);
			if(v==1)
				idx++;
		}
	}
	if(ver[v].size()==1 && v!=1)
		aa[idx].push_back(l);
}





int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n-1;i++){
		int t,T;
		scanf("%d%d",&t,&T);
		ver[t].push_back(T);
		ver[T].push_back(t);
	}
	dfs(1,0,0);
	int maxi=0;
	for(int i=0;i<idx;i++){
		sort(aa[i].begin(),aa[i].end());
		int ans=aa[i][0];
		for(int ii=1;ii<aa[i].size();ii++){
			ans=max(ans+1,aa[i][ii]);
		}
		maxi=max(ans,maxi);
	}
	cout<<maxi<<endl;











	return 0;
}
