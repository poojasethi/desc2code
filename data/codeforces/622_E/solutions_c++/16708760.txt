#include <bits/stdc++.h>
using namespace std;
const int maxn = 5e5+10;
vector<int> e[maxn];
int n,dep[maxn],m,now,ans;
pair<int,int> a[maxn];
void dfs(int rx,int x){
    int cnt=0;
    for (vector<int>:: iterator it= e[x].begin(); it!= e[x].end(); it++){
        if (*it!=rx){
            dep[*it]=dep[x]+1;
            cnt++;
            dfs(x,*it);
        }
    }
    if (cnt==0) a[++m]=make_pair(now,dep[x]);
}
int main(){
    scanf("%d",&n);
    for (int u,v,i=1; i<n; i++){
        scanf("%d%d",&u,&v);
        e[u].push_back(v);
        e[v].push_back(u);
    }
    for (vector<int>:: iterator it= e[1].begin(); it!= e[1].end(); it++) now=*it,dfs(1,*it);
    sort(a+1,a+m+1);a[0].first=a[1].first+1;
    for (int i=1; i<=m; i++){
        if (a[i].first==a[i-1].first) a[i].second=max(a[i].second,a[i-1].second+1);
        ans=max(ans,a[i].second+1);
    }
    cout<<ans;
}
