#include <bits/stdc++.h>

#define int long long
#define mp make_pair
#define pb push_back
#define L first
#define R second

using namespace std;

const int maxn = 5e5+100;

vector <int> v[maxn] , ans;
bool mark[maxn];
int h , Ans;

void dfs(int x){
	mark[x] = true;
	for(int i = 0;i < v[x].size();i++){
		int child = v[x][i];
		if(mark[child] == false){
			h++;
			dfs(child);
			h--;
		}
	}
	if(v[x].size() == 1) ans.pb(h);
}

main(){
	ios_base::sync_with_stdio(false); cin.tie(false); cout.tie(false);
	int n; cin >> n;
	for(int i = 1;i < n;i++){
		int x , y; cin >> x >> y;
		v[x].pb(y) , v[y].pb(x);
	} 
	mark[1] = true;
	for(int i = 0;i < v[1].size();i++){
		int child = v[1][i];
		h = 1 , ans.clear();
		dfs(child);
		sort(ans.begin() , ans.end()); reverse(ans.begin() , ans.end());
		for(int j = 0;j < ans.size();j++) Ans = max(Ans , ans[j] + j);
	}
	cout << Ans << endl;
}
