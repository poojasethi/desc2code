#include<bits/stdc++.h>
using namespace std;
int n,ans;
vector<int> v[500009];
vector<int> z;
void dfs(int id,int par,int h)
{
	if(v[id].size()==1)
	{
		z.push_back(h);return;
	}
	for(int i=0;i<v[id].size();i++)
		if(v[id][i]!=par)
			dfs(v[id][i],id,h+1);
}
int solve(int id)
{
	z.clear();
	dfs(id,1,0);
	sort(z.begin(),z.end());
	for(int i=1;i<z.size();i++)
		z[i]=max(z[i],z[i-1]+1);
	return z.back();
}
int main()
{
	cin>>n;
	for(int i=1;i<n;i++)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		v[x].push_back(y);
		v[y].push_back(x);
	}
	for(int i=0;i<v[1].size();i++)
	{
		ans=max(ans,solve(v[1][i])+1);
	}
	cout<<ans;
	return 0;
}
