#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
#define rep(i,a,b) for(int i=a;i<=b;++i)
#define erp(i,a,b) for(int i=a;i>=b;--i)
#define mem(a) memset(a,0,sizeof a)
#define LL long long
using namespace std;
const int MAXN = 500005;
int N;
vector<int> g[MAXN], r;

void dfs(int u, int fa, int dep)
{
	if (g[u].size()==1) { r.push_back(dep); return; }
	for (int i = 0; i<(int)g[u].size(); ++i)
		if (g[u][i]!=fa) dfs(g[u][i], u, dep+1);
}

int main()
{
	int u, v;
	scanf("%d", &N);
	rep(i, 2, N)
	{
		scanf("%d%d", &u, &v);
		g[u].push_back(v);
		g[v].push_back(u);
	}
	int ans = 0;
	for (int i=0; i<int(g[1].size()); ++i)
	{
		r.clear();
		dfs(g[1][i], 1, 1);
		sort(r.begin(), r.end());
		int pos = r[0];
		for (int j = 1; j<(int)r.size(); ++j)
		{
			if (r[j]<=pos) pos ++;
			else pos = r[j];
		}
		ans = max(ans, pos);
	}
	printf("%d\n", ans);
	return 0;
}
