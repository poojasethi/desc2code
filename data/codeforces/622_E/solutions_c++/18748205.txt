#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<vector>
#define maxn 500010

using namespace std;

vector<int> tmp;
int head[maxn],to[2*maxn],next[2*maxn];
int n,m,num,ans;

void addedge(int x,int y)
{
	num++;to[num]=y;next[num]=head[x];head[x]=num;
}

void dfs(int x,int fa,int dep)
{
	bool flag=0;
	for (int p=head[x];p;p=next[p])
	  if (to[p]!=fa) dfs(to[p],x,dep+1),flag=1;
	if (!flag) tmp.push_back(dep);
}

int solve(int x)
{
	tmp.clear();
	dfs(x,1,0);
	sort(tmp.begin(),tmp.end());
	for (int i=1;i<tmp.size();i++) tmp[i]=max(tmp[i],tmp[i-1]+1);
	return tmp[tmp.size()-1];
}

int main()
{
	scanf("%d",&n);
	for (int i=1;i<n;i++)
	{
		int x,y;
		scanf("%d%d",&x,&y);
		addedge(x,y);addedge(y,x);
	}
	for (int p=head[1];p;p=next[p]) ans=max(ans,solve(to[p]));
	printf("%d\n",ans+1);
	return 0;
}
