#include <bits/stdc++.h>
using namespace std;
const int N = 5e5+5;
vector<int> adj[N],R;
void dfs(int root,int par,int h){
	if(adj[root].size()==1 && adj[root][0]==par){
		R.push_back(h);
		return;
	}
	int i,v;
	for(i=0;i<adj[root].size();++i){
		v= adj[root][i];
		if(par!=v){
			dfs(v,root,h+1);
		}
	}
}
int main(){
	int n,i,j,a,b;
	cin>>n;
	for(i=0;i<n-1;++i){
		scanf("%d%d",&a,&b);
		adj[a].push_back(b);adj[b].push_back(a);
	}
	int ret=0;
	for(i=0;i<adj[1].size();++i){
		dfs(adj[1][i],1,1);
		sort(R.begin(),R.end());
		int pos= R[0];
		for(j=1;j<R.size();++j){
			if(R[j]<=pos)pos++;
			else
				pos= R[j];
		}
		R.clear();
		ret= max(ret,pos);
	}
	cout<<ret<<endl;
}
