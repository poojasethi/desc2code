#include<bits/stdc++.h>
#define N 1001000
using namespace std;
int n,i,j,x,y,t,tot,ans,fir[N],la[N],ne[N],du[N],d[N],q[N];
void ins(int x,int y){la[++tot]=y;ne[tot]=fir[x];fir[x]=tot;du[x]++;}
void dfs(int x,int fa){
	if(du[x]==1)q[++t]=d[x];
	for(int i=fir[x];i;i=ne[i])if(la[i]!=fa)d[la[i]]=d[x]+1,dfs(la[i],x);
}
int main(){
	scanf("%d",&n);
	for(i=1;i<n;i++)scanf("%d%d",&x,&y),ins(x,y),ins(y,x);
	ans=1;dfs(1,0);
	for(i=fir[1];i;i=ne[i]){
		t=0;d[la[i]]=0;dfs(la[i],1);sort(q+1,q+t+1);
		for(j=1;j<=t;j++)ans=max(ans,j+q[t-j+1]);
	}
	printf("%d",ans);
}
