#include <iostream>
#include <vector>
#include <queue>

using namespace std;

vector <vector <int> >g;
int n, m;
pair <int,int>p;

int main(){
    int x, y;
    cin >> n;
    g=vector <vector <int> >(n);
    for(int i=0;i<n-1;++i){
      cin >> x >> y;
      g[x-1].push_back(y-1);
      g[y-1].push_back(x-1);
    }
    vector <bool>b(n,false);
    m=0;
    b[0]=true;
    for(int i=0;i<g[0].size();++i){
      queue <pair <int,int> >q;
      q.push(pair <int,int>(g[0][i],1));
      vector <int>v;
      while(!q.empty()){
	p=q.front();
	q.pop();
	if(!b[p.first]){
	  b[p.first]=true;
	  if(g[p.first].size()==1)v.push_back(p.second);
	  else{
	    for(int i=0;i<g[p.first].size();++i){
	      if(!b[g[p.first][i]])q.push(pair <int,int>(g[p.first][i],p.second+1));
	    }
	  }
	}
      }
      for(int i=1;i<v.size();++i){
	  if(v[i]<=v[i-1])v[i]=v[i-1]+1;
      }
      m=max(m,v[v.size()-1]);
    }
    cout << m << endl;
}
