#include <bits/stdc++.h>
using namespace std;

#define mp(x, y) make_pair((x), (y))

typedef long long ll;

int n;
vector<int> next[500005];
int l[500005];
int maxl;
int ans;

void dfs(int v, int f, int le)
{
	maxl=max(maxl, le);
	if(next[v].size()==1) l[le]++;
	for(int i=0; i<next[v].size(); i++) {
		int s=next[v][i];
		if(s==f) continue;
		dfs(s, v, le+1);
	}
}

int main()
{
	scanf("%d", &n);
	for(int i=0; i<n-1; i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		x--; y--;
		next[x].push_back(y);
		next[y].push_back(x);
	}
	for(int i=0; i<next[0].size(); i++) {
		int s=next[0][i];
		dfs(s, 0, 0);
		int cur=0;
		for(int j=0; j<=maxl; j++) {
			if(l[j]==0) cur=max(0, cur-1);
			else cur+=l[j]-1;
			l[j]=0;
		}
		ans=max(ans, cur+maxl+1);
	}
	printf("%d\n", ans);

	return 0;
}
