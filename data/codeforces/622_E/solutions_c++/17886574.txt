#include<cstdio>
#include<algorithm>
#include<vector>
using namespace std;
vector<int>g[500010];
int f[500010],cnt,vis[500010],n,m,ans;
void dfs(int u,int fa,int dep){
	vis[u]=1;if(g[u].size()==1){++cnt;f[cnt]=dep;return;}
	for(int i=0;i<g[u].size();++i)if(fa!=g[u][i])dfs(g[u][i],u,dep+1);
}
int main(){
	scanf("%d",&n);
	for(int i=1,u,v;i<n;++i)scanf("%d%d",&u,&v),g[u].push_back(v),g[v].push_back(u);
	for(int i=0;i<g[1].size();++i){
		cnt=0,dfs(g[1][i],1,1),sort(f+1,f+1+cnt);
		f[0]=0;for(int j=1;j<=cnt;++j)f[j]=max(f[j-1]+1,f[j]);
		ans=max(ans,f[cnt]);
	}
	printf("%d\n",ans);
}
