#include <iostream>
#include <stdio.h>

using namespace std;

const int N = 1e6 + 7;

long long a[N],dp[N],up,dn;
int n;

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("I.in","r",stdin);
	#endif
	cin>>n;
	for(int i=0;i<n;i++) scanf("%I64d",&a[i]);
	up = a[0];
	dn = -a[0];
	for(int i=0;i<n;i++)
	{
		dp[i+1] = max(up-a[i],dn+a[i]);
		up = max(up,dp[i+1]+a[i+1]);
		dn = max(dn,dp[i+1]-a[i+1]);
	}
	cout<<dp[n]<<endl;
	return 0;
}
