#include<cstdio>
#define N 1100000
int i,n,a[N];
long long f[N],c1,c2;
long long max(long long a,long long b)
{
	if (a>b) return a;return b;
}
int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	scanf("%d",&a[i]);
	//c1=0x37373737;c1=c1*c1;c1=-c1;c2=c1;
	c1=-a[1];c2=a[1];
	for (i=1;i<=n;i++)
	{
		f[i]=max(0,max(c1+a[i],c2-a[i]));
		c1=max(c1,f[i]-a[i+1]);
		c2=max(c2,f[i]+a[i+1]); 
	}
	printf("%lld\n",f[n]);
}
