#include <iostream>
#include <algorithm>
#define Nmax 1000005
#define INF (1LL<<62)

using namespace std;

long long a[Nmax],dp[Nmax];

int main()
{
    int i,n;
    long long cmin=-INF,cmax=-INF;
    cin>>n;
    for(i=1;i<=n;++i) cin>>a[i];
    for(i=1;i<=n;++i)
    {
        dp[i]=dp[i-1];
        dp[i]=max(dp[i],a[i]+cmin);
        dp[i]=max(dp[i],cmax-a[i]);
        cmin=max(cmin,dp[i-1]-a[i]);
        cmax=max(cmax,dp[i-1]+a[i]);
    }
    cout<<dp[n];
    return 0;
}
