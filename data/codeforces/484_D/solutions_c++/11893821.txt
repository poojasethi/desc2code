#include <cstdio>

#define N 1000100
#define max(a, b) ((a) > (b) ? (a) : (b))

template <class T> void chkmax(T &a, T b) { if (a < b) a = b; }

typedef __int64 LL;

main() {
    int n, v;
    scanf("%d", &n);
    LL rlt = 0, a, b;
    scanf("%I64d", &b), n --, a = -b;
    while (n --) {
        scanf("%d", &v);
        chkmax(a, rlt - v);
        chkmax(b, rlt + v);
        rlt = max(a + v, b - v);
    }
    printf("%I64d\n", rlt);
}
