#include <cstdio>
#include <algorithm>
#define ll long long
using namespace std;
int const MIN = -2147483646;
int const MAX = 1e6 + 10;
ll num[MAX], ans;
int main()
{
    int n;
    ll ans = 0, a = MIN, b = MIN, val;
    scanf("%d", &n);
    for(int i = 1; i <= n; i++)
    {
        scanf("%I64d", &val);
        a = max(a, ans + val);
        b = max(b, ans - val);
        ans = max(ans, a - val);
        ans = max(ans, b + val);
    }
    printf("%I64d\n",ans);
}
