#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
typedef long long ll;
ll num[1000010],ans;
int main()
{
    int T,t,n,m,i,j,k;
    ll ans=0,a=-1e9,b=-1e9,val;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        scanf("%I64d",&val);
        a=max(a,ans+val);
        b=max(b,ans-val);
        ans=max(ans,a-val);
        ans=max(ans,b+val);
    }
    printf("%I64d\n",ans);
}
