#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
typedef long long LL;
const int N=3e6;
LL a[N],f[N],s1[N],s2[N],n;
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
	s1[0]=a[1];
	s2[0]=-a[1];
	for(int i=1;i<=n;i++)
	{
		f[i]=max(s1[i-1] - a[i],s2[i-1] + a[i]);
		s1[i]=max(s1[i-1],f[i]+a[i+1]);
		s2[i]=max(s2[i-1],f[i]-a[i+1]);
	}
	cout<<f[n]<<endl;
}
