#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll ans, cx, cy, n,x;
int main() {
	ios::sync_with_stdio (0);
	cin >> n;
	for (int i = 0; i < n; ++i) {
              cin>>x;
		if (!i || ans - x > cx) cx = ans - x;
		if (!i || ans + x > cy) cy = ans + x;
		ans = max (ans, cx + x);
		ans = max (ans, cy - x);
	}
	cout << ans;
	return 0;
}
