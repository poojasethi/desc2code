#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

int n;
long long a[1000010],ans,g0,g1;

int main()
{
    scanf("%d",&n);
    for (int i=1;i<=n;i++) scanf("%I64d",&a[i]);
    for (int i=1;i<=n;i++)
    {
        if (i==1) 
        {
            g0=-a[i]; 
            g1=a[i];
        } else {
            g0=max(g0,ans-a[i]);
            g1=max(g1,ans+a[i]);
        }
        ans=max(g0+a[i],g1-a[i]);
    }
    printf("%I64d\n",ans);
}
