#include <bits/stdc++.h>

using namespace std;

typedef long long LL;

const int maxn = int(1e6)+10;
const int INF = int(1e9)+1;

int n;
int a[maxn];
LL f[maxn];

int main() {
	scanf("%d",&n);
	for (int i = 1;i <= n; i++) scanf("%d",&a[i]);
	f[0] = f[1] = 0;
	LL tad = f[0]+a[1],tmi = f[0]-a[1];
	for (int i = 2;i <= n; i++) {
		f[i] = max(tad-a[i],tmi+a[i]);
		f[i] = max(f[i-1],f[i]);
		tad = max(tad,f[i-1]+a[i]);
		tmi = max(tmi,f[i-1]-a[i]);	
	}	
	LL ans = f[n];
	cout<<ans<<endl;	
	return 0;
}
