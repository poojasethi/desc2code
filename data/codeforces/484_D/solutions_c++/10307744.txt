#include <iostream>
#include <algorithm>
#include <cstdio>
#define LL long long
using namespace std;
LL ans,s,l;
int main()
{
	int n;
    scanf("%d",&n);
	LL x;
	scanf("%I64d",&x);
	ans=0,s=-x,l=x;
	for (int i=1;i<n;i++)
	{
	    scanf("%I64d",&x);
		if (ans-x>s)
			s=ans-x;
		if (ans+x>l)
			l=ans+x;
		ans=max(ans,max(s+x,l-x));
	}
	cout<<ans<<endl;
	return 0;
}
