#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
#define rep(i,a,b) for(int i=a;i<=b;i++)
using namespace std;
const int maxn=1e6+5;
long long n,a[maxn+1],d[maxn+1];
int main()
{
	scanf("%lld",&n);
	rep(i,1,n)
	{
		scanf("%lld",a+i);
	}
	d[1]=0;
	d[2]=abs(a[2]-a[1]);
	rep(i,3,n)
	{
		if((a[i]-a[i-1])*(a[i-1]-a[i-2])>=0)
		{
			d[i]=d[i-1]+abs(a[i]-a[i-1]);
		}
		else
		{
			d[i]=max(d[i-1],d[i-2]+abs(a[i]-a[i-1]));
		}
	}
	printf("%lld\n",d[n]);
}
