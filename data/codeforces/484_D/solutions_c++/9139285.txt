#include<iostream>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#include<cstdio>
#define LL long long
int main()
{
    int n,t;
    scanf("%d",&n);
    LL ans=0,t1=0,t2=0;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&t);
        if (!i||ans+t>t1)t1=ans+t;
        if (!i||ans-t>t2)t2=ans-t;
        ans=t1-t>t2+t?t1-t:t2+t;
    }
    printf("%I64d\n", ans);
    return 0;
}
