#include <cstdio>
#include <algorithm>

using namespace std;

const int SIZE = 1e6 + 7;
const long long inf = 1e17;

int n;
int a[SIZE];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	long long answer = 0;
	long long maxVal1 = -inf, maxVal2 = -inf;
	for (int i = 0; i < n; i++) {
		maxVal1 = max(maxVal1, answer - a[i]);
		maxVal2 = max(maxVal2, answer + a[i]);
		answer = max(max(a[i] + maxVal1, -a[i] + maxVal2), answer);
	}
	printf("%I64d\n", answer);
	return 0;
}
