#include <bits/stdc++.h>
using namespace std;
int n,a[1000005];
long long f[1000005];
int main(){
    ios::sync_with_stdio(false);
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i];
    int j=1;
    f[2]=abs(a[2]-a[1]);
    for(int i=3;i<=n;i++){
        if((a[i-2]>=a[i-1]&&a[i-1]<=a[i])||(a[i-2]<=a[i-1]&&a[i-1]>=a[i])) j=i-1;
        f[i]=max(f[j-1]+abs(a[j]-a[i]),f[j]+abs(a[j+1]-a[i]));
    }
    cout<<f[n];
}
