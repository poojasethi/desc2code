#include <bits/stdc++.h>
using namespace std;

#define sz(v) int(v.size())
#define endl '\n'
typedef long long ll;
typedef pair<int,int> pii;

const int MAXn = 1e6+10;
ll a[MAXn];
ll ans[MAXn];

int main() {
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> a[i];
	ll pb = a[n-1], nb=-a[n-1];
	for(int i = n-2; i >= 0; i--) {
		pb = max(pb, a[i]+ans[i+1]);
		nb = max(nb, -a[i]+ans[i+1]);
		ans[i] = max(pb-a[i], nb+a[i]);
	}
	cout << ans[0] << endl;
	return 0;
}
