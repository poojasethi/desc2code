#include <iostream>

using namespace std;

int main()
{
  int n;
  cin>>n;
  n--;
  long long int i;
  cin>>i;
  long long int sol=0,maximo=i,minimo=-i;
  while(n--)
  {
    cin>>i;
    if(sol+i>maximo)
      maximo=sol+i;
    if(sol-i>minimo)
      minimo=sol-i;
    sol=max(sol,max(minimo+i,maximo-i));
  }
  cout<<sol<<endl;
  return 0;
}
