#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>
#define ll long long
using namespace std;
ll a[1000004],n,ans=0;
ll f,g;
int main(){
    scanf("%I64d",&n);
    for(int i=1;i<=n;i++)scanf("%I64d",&a[i]);
    f=-a[1];g=a[1];
	for(int i=2;i<=n;i++){
		    f=max(f,ans-a[i]);
		    g=max(g,ans+a[i]);
		    ans=max(f+a[i],g-a[i]);
	}
	cout<<ans<<endl;
	//system("pause"); 
	return 0;
}
