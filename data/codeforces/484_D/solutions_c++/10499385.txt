#include <cstdio>
#include <algorithm>
using std::abs;
using std::max;
int main(){
	int n;
	scanf("%d",&n);
	int prev;
	scanf("%d",&prev);
	long long yes=0,no=0;
	bool inc=false;
	for(int i=1;i<n;i++){
		int cur;
		scanf("%d",&cur);
		long long nyes=no+abs(cur-prev);
		if((inc&&cur>prev)||(!inc&&cur<=prev)){
			nyes=max(nyes,yes+abs(cur-prev));
		}
		no=max(yes,no);
		yes=nyes;
		inc=cur>prev;
		prev=cur;
	}
	printf("%lld\n",max(yes,no));
	return 0;
}
