#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define LL long long
LL a[1000001],n,ans,f,g;
int main()
{
	scanf("%I64d",&n);for(int i=1;i<=n;i++)scanf("%I64d",&a[i]);f=-a[1],g=a[1];
	for(int i=2;i<=n;i++)f=max(f,ans-a[i]),g=max(g,ans+a[i]),ans=max(f+a[i],g-a[i]);
	printf("%I64d\n",ans);return 0;
}
