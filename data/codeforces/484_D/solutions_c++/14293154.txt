#include <bits/stdc++.h>
using namespace std;
int ar[1000000];
#define ll long long int
int main(){	
	int n,i=0,j;for(scanf("%d",&n);i<n;++i)scanf("%d",&ar[i]);
	ll ret=0,A=-ar[0],B=ar[0];
	for(int i=1;i<n;++i){
		A = max(ret-ar[i],A);
		B = max(ret+ar[i],B);
		ret= max(A+ar[i],B-ar[i]);
	}
	cout<<ret<<endl;
}
