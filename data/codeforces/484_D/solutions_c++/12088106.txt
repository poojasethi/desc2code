#include <bits/stdc++.h>
#define sd(x) scanf("%d",&x)
using namespace std;
int n, prev, a[1000100], u = 1, d = 1;long long dp[1000100];int main(){sd(n);for(int i=0;i<n;i++)sd(a[i]);for(int i=1;i<n;i++){if(a[i]==a[i-1])u=1,d=1,prev=i-1;if(u && a[i-1]>a[i])d=1,u=0,prev=i-1;if(d && a[i-1]<a[i])d=0,u=1,prev=i-1;dp[i+1]=max(dp[prev]+abs(a[i]-a[prev]),dp[prev+1]+abs(a[i]-a[prev+1]));}printf("%I64d\n", dp[n]);return 0;}
