#include<cstdio>
#include<algorithm>
using namespace std;
inline void readi(int &x)
{char c;bool fl;
while(c=getchar(),c!='-'&&(c>'9'||c<'0'));if(c=='-')fl=1,x=0;else fl=0,x=c^'0';
while(c=getchar(),c>='0'&&c<='9')x=x*10+(c^'0');if(fl)x=-x;}
int n,t;
long long ans,u,d;

int main()
{
	readi(n);
	for(int i=0;i<n;i++)
	{
		readi(t);
		if(!i||ans-t>u)u=ans-t;
		if(!i||ans+t>d)d=ans+t;
		ans=max(u+t,d-t);
	}
	printf("%I64d\n",ans);
	return 0;
}
