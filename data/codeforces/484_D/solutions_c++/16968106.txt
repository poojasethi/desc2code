#include<iostream>
#include<cstdio>
using namespace std;
typedef long long ll;

int main()
{
    int n,a;
    scanf("%d",&n);
    ll ans=0,t1=0,t2=0;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a);
        if(!i||ans+a>t1)
            t1=ans+a;
        if(!i||ans-a>t2)
            t2=ans-a;
        ans=t1-a>t2+a?t1-a:t2+a;
    }
    printf("%I64d\n",ans);
    return 0;
}
