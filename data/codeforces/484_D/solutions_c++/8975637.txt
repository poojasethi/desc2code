#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 2000000;

int n, a[N];

int main() {

    scanf("%d", &n);

    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }

    long long x = a[0];
    long long y = -x;
    long long res = 0;

    for (int i = 0; i < n; i++) {
        res = max(x - a[i], y + a[i]);
        x = max(x, res + a[i + 1]);
        y = max(y, res - a[i + 1]);
    }

    printf("%I64d\n", res);

    return 0;

}
