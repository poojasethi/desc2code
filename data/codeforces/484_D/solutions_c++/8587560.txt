#include<iostream>

using namespace std;

int main() {

    int i, n, x;
    long long rez, a, b;

    cin >> n;

    cin >> x;
    a = x;
    b = -x;
    rez = 0;

    for(i = 2; i <= n; ++i) {
        cin >> x;

        long long lastr = rez;

        rez = max(rez, max(a - x, b + x));

        a = max(a, lastr + x);
        b = max(b, lastr - x);
    }

    cout << rez;

    return 0;
}
