#include <cmath>
#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define rep(i,j,k)  for (int i = (j); i <= (k); i++)
#define dep(i,j,k)  for (int i = (j); i >= (k); i--)
using namespace std;
typedef long long ll;
int n, t;
ll t1, t2, ans;
int main()
{
	cin >> n;
	rep(i,1,n)
	{
		scanf("%d", &t);
		if (i==1 || ans+t>t1)	t1 = ans+t;
		if (i==1 || ans-t>t2)	t2 = ans-t;
		ans = t1-t>t2+t?t1-t:t2+t;
	}
	cout << ans << endl;
}
