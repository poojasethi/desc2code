#include <cstdio>
#include <algorithm>
using namespace std;
int n, las, cur;
long long lasD0=0,curD0,lasD1=0,curD1;
int main(){
	scanf("%d%d", &n, &las);
	while (--n){
		scanf("%d", &cur);
		if (cur > las){
			curD0 = max(lasD0 + cur - las, lasD1);
			curD1 = max(lasD1, lasD0);
		}else{
			curD1 = max(lasD1 + las - cur, lasD0);
			curD0 = max(lasD0, lasD1);
		}
		lasD0 = curD0; lasD1 = curD1; las = cur;
	}
	printf("%I64d", max(curD0, curD1));
	return 0;
}
