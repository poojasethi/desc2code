#include <cstdio>
#include <algorithm>

using namespace std;

const int N = 1000010;

int main() {
    int n;
    long long g0, g1, cur;
    scanf("%d", &n);
    for (int i = 1, x; i <= n; ++i) {
        scanf("%d", &x);
        if (i == 1) {
            cur = 0;
            g0 = -x, g1 = x;
        } else {
            g0 = max(g0, cur - x);
            g1 = max(g1, cur + x);
            cur = max(g0 + x, g1 - x);
        }
    }
    printf("%I64d\n", cur);
}
