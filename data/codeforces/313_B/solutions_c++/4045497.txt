#include<cstdio>
#define N 100005
char A[N];
int  V[N];
int main()
{
	int n,a,b,i,j,s;
	scanf("%s",A);
	for(s = i=0; A[i]; i++)
	{
		if(A[i]==A[i+1])
			s++;
		V[i+1] = s;
	}
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d%d",&a,&b);
		printf("%d\n",V[b-1]-V[a-1]);
	}
	return 0;
}
