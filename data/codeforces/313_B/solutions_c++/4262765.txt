#include<iostream>
#include<string>

using namespace std;

int main(){

string s;
int a, b,c,i,j;
cin>>s;
cin>>c;
int n=s.size();
int v[n];
v[0]=0;
for(i=1; i<n; i++){
	if(s[i]==s[i-1])v[i]=v[i-1]+1;
	else v[i]=v[i-1];
}
for(j=1;j<=c;j++)
{
	cin>>a;
	cin>>b;
	cout<<v[b-1]-v[a-1]<<endl;
}
return 0;

}
