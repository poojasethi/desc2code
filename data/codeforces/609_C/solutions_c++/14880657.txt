#include <bits/stdc++.h>
using namespace std;
int main() {
	long long int i,j,k,l,n,m,a[100005],b[20005]={0};
	cin>>n;
	long long int sum=0;
	for(i=0;i<n;i++) {
		cin>>a[i];
		sum+=a[i];
	}
	k = sum/n;
	long long int cnt=sum%n,ans=0;
	for(i=0;i<n;i++) {
		if(a[i] > k) {
			if(cnt) {
				cnt--;
				ans += a[i]-k-1;
			}
			else
				ans+=a[i]-k;
		}
	}
	cout<<ans<<endl;
	return 0;
}
