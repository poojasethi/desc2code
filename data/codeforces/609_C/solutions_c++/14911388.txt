#include <iostream>
#include <vector>
using namespace std;
typedef unsigned long long ll;
int main()
{
ll n,m,r,q,s=0; vector <ll> t;
cin>>n;
for (ll i=0; i<n;i++)
{
    cin>>m; t.push_back(m); s+=m;
}
q=s/n; r=s%n;
ll occ=0;
for (ll i=0;i<n;i++)
{
    if (t[i]>q)
    {
        if (r) {occ+=t[i]-q-1; r--;}
        else {occ+=t[i]-q;}
    }
}
cout<<occ<<endl;
}
