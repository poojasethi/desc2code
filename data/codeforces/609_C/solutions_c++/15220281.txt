#include<bits/stdc++.h>
using namespace std;
const int MAX=1e5+9;
int a[MAX],sum,t1,t2,ans,n;
int main()
{
    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i],sum+=a[i];
    t2=sum%n,t1=n-t2;
    sort(a,a+n);
    for (int i=0;i<t1;i++) ans+=abs(sum/n-a[i]);
    for (int i=t1;i<n;i++) ans+=abs(a[i]-sum/n-1);
    cout<<ans/2;
}
