#include<iostream>
#include<algorithm>
using namespace std;
int a[100005];
int main()
{
    int n,s=0,x=0,y=0,c=0; 
    cin>>n;
    for(int i=1;i<=n;i++) cin>>a[i],s+=a[i];
    x=s/n; y=s%n;
    s=0;
    for(int i=1;i<=n;i++) if(a[i] > x) s+=a[i]-x,c++;
    cout<<s-min(y,c);
    //system("pause"); 
    return 0;
}
