#include <iostream>
using namespace std;
main()
{
	int n;
	int upper=0, lower=0;
	double sm=0;
	cin>>n;
	int *m=new int[n];
	for(int i=0; i<n; i++)
	{
		cin>>m[i];
		sm+=m[i]; //sum
	}
	sm/=n; //mean
	for(int i=0; i<n; i++)
	{
		if(m[i]>sm && m[i]-sm>=1)
			upper+=(int)(m[i]-sm);
		else if(m[i]<sm && sm-m[i]>=1)
			lower+=(int)(sm-m[i]);
	}
	cout<<max(upper,lower);
}
