#include<algorithm>
#include<iostream>
using namespace std;

int A[100050];
int n,m,a,s,rs;
int main()
{
    cin>>n;
    for(int i=0; i<n; ++i)
    {
        cin>>A[i];
        s+=A[i];
    }
    int rest=s%n;
    s=s/n;
    sort(A, A+n);
    reverse(A, A+n);
    for(int i=0; i<n; ++i)
        rs+=abs(A[i]-s-((i<rest)?1:0));
    cout<<rs/2<<endl;
    return 0;
}
