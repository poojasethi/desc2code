#include<iostream>
using namespace std;

int main()
{
		int n, cnt = 0;
		cin >> n;
		int f[n], sum = 0;
		for(int i=1;i<=n;i++)
		{
			cin >> f[i];
			sum += f[i];
		}
		int thg = sum / n;
		int mod = sum % n;
		for(int i=1;i<=n;i++)
		{
			if(f[i] > thg)
			(mod > 0) ? ( (cnt += f[i] - thg - 1) && mod --) : (cnt += f[i] - thg) ;
		}
		cout << cnt;
}
