#include <bits/stdc++.h>

using namespace std;
typedef long long ull ;

int main()
{
ull n ,s=0,q,c=0 ,c1=0; cin>> n ;
ull t[n] ;
for ( ull k=0 ; k<n ;k++) {cin>>t[k] ; s+=t[k] ;}
q=s/n ;

for ( ull k=0 ; k<n ;k++)
{
    if ( t[k]<q) c+=q-t[k] ;
    else if ( t[k]> q) c1+=t[k]-q-1;
}

cout<< max ( c , c1 ) ;


    return 0 ;
}
