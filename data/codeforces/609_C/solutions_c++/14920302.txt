#include <iostream>
using namespace std;

int main()
{
    long n,a[100001],ans,s=0,res1=0,res2=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        s+=a[i];
    }
    s=s/n;
    for(int i=0;i<n;i++)
    {
        if(a[i]>s)
            res1+=a[i]-s-1;
        else
            if(a[i]<s)
            res2+=s-a[i];
    }
    ans=max(res1,res2);
    cout<<ans;
}
