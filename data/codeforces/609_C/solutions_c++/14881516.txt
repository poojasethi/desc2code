#include <iostream>
using namespace std;

int arr[100001];

int main(){
    int n;  cin>>n;
    long long sum = 0;
    for(int i=0;i<n;i++)    {   cin>>arr[i] ; sum+=arr[i];  }
    int av = (sum/n);
    int ex = sum % n;
    long long res = 0;
    for(int i=0;i<n;i++)    if(arr[i]>av)   {
        if(ex)  res+=arr[i]-av-1,ex--;
        else res+=arr[i]-av;
    }
    cout<<res;
}
