#include<bits/stdc++.h>
using namespace std;
const int mxmx=1e5+10;
int n;
int a[mxmx];
long long jam=0;
int t0,t1;
int x,y;

long long ANS=0;
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>a[i];
		jam+=a[i];
	}
	sort(a,a+n);
	t1=jam%n;
	t0=n-t1;
	x=jam/n;
	y=x+1;
	for(int i=0;i<t0;i++){
		ANS+=abs(a[i]-x);
	}
	for(int i=t0;i<n;i++){
		ANS+=abs(a[i]-y);
	}
	cout<<ANS/2;
}
