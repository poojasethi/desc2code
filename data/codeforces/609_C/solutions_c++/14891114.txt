#include "stdio.h"
#include "stdlib.h"
#include <algorithm>
 int main(){
 	int n, a[100000],s=0;;
 	double m;
 	scanf("%d",&n);
 	for(int i=0;i<n;i++){
 		scanf("%d",&a[i]);
 		s+=a[i];
 	}
 	m=1.0*s/n;

 	s=0;int b=0;
 	for(int i=0;i<n;i++){
 		if(a[i]<m) s+=abs(a[i]-m); else b+=abs(a[i]-m);
 	}


 	printf("%d",std::max(s,b)); 

 return 0;
 }
