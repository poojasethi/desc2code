#include <bits/stdc++.h>
using namespace std;

int main()
{
long long n,S = 0,C = 0;
vector<int> V;
cin >> n;
V.resize(n);
for(int i = 0;i < n;i++)
{
    cin >> V[i];
    C += V[i];
}
sort(V.begin(),V.end());
long long Q = n-(C%n);C /= n;
for(int i = 0;i < n;i++)
{
    if(Q > 0){S += abs(V[i] - C);Q--;}
    else{S += abs(V[i] - (C+1));}
}
cout << S/2;
return 0;
}
