#include<cstdio>
using namespace std;

int n,m[100005],s=0,t=0,r;

int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&m[i]);
		s+=m[i];
		}
	r=s%n;
	s/=n;
	for(int i=0;i<n;i++){
		if(m[i]>s){
			t+=m[i]-s;
			if(r)
				r--,t--;
			}
		}
	printf("%d\n",t);
	return 0;
	}
