#include <iostream>

using namespace std;

int main(){
	ios_base::sync_with_stdio(0);
	long long n,s=0;
	cin>>n;
	int T[n];
	for(int i=0;i<n;i++){
		cin>>T[i];
		s += T[i];
	}
	long long avg = s/n, x=s%n, cnt=0;
	s=0;
	for(int i=0;i<n;i++)
		if(T[i]>avg){
			s += T[i]-avg;
			cnt++;
		}
	cout<<s-min(x,cnt)<<endl;
	return 0;
}
