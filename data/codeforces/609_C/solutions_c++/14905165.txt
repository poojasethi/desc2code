#include<bits/stdc++.h>
using namespace std;
int x[100000],low,lim,sum;
int main()
{
    int n; cin >> n; for(int i=0;i<n;i++) scanf("%d",&x[i]),sum+=x[i];
    int r1 = sum/n , r2 = r1+min(sum%n,1);
    for(int i=0;i<n;i++)
        low+=max(r1-x[i],0),lim+=max(x[i]-r2,0);
    cout << max(low,lim);
}
