#include<bits/stdc++.h>
using namespace std;
long long a[1000005],i,j,m,n,s,p;
int main()
{
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        s+=a[i];
    }
    m=s/n;
    j=s%n;
    for(i=0;i<n;i++)
    {
        if(a[i]>m)
        {
          p+=(a[i]-m);
          if(j)
          {
            p--;j--;
          }
        }
    }
    cout<<p;
}
