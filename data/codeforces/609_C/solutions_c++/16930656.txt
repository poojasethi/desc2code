#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	int i,j,n,l=0,r=0,s=0,avg;
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
	{
		cin>>a[i];
		s+=a[i];
	}
	avg=s/n;
	
	sort(a,a+n);
	for(i=0;i<n;i++)
	{
		if(a[i]<=avg)
			l+=(avg-a[i]);
		else
			r+=(a[i]-(avg+1));// eg--for the case 5 5 7 whose answer state is 5 6 6 
	}
	cout<<max(l,r)<<endl;
}
