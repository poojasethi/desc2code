#include<bits/stdc++.h>

using namespace std;

int r[100007];

int main()
{
  int n,sum=0,ans=0;
  cin>>n;
  for(int i=0;i<n;i++)
    cin>>r[i],sum+=r[i];
  int mid=sum/n;
  sort(r,r+n);
  reverse(r,r+n);
  int cnt=sum%n;
  for(int i=0;i<cnt;i++)
    ans+=abs(mid+1-r[i]);
  for(int i=cnt;i<n;i++)
    ans+=abs(mid-r[i]);
  cout<<ans/2<<endl;
  return 0;
}
