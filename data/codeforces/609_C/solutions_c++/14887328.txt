#include <bits/stdc++.h>
using namespace std;

const int maxn = 100050;

int a[maxn],n,ans,esp,q;

int main(){
	cin.sync_with_stdio(false);
	cin>>n;
	int i,j;
	for(i=0;i<n;i++){cin>>a[i];esp+=a[i];}
	sort(a,a+n);
	q=esp%n;
	esp/=n;q=n-q;
	for(i=0;i<n;i++){
	  if(i>=q)ans+=abs(a[i]-esp-1);
	  else ans+=abs(a[i]-esp);
    }
    cout<<ans/2;
    return 0;
}
