#include <cstdio>
#include <algorithm>

using namespace std;

int g[200000];
int i,n,s,p1,p2,t1,t2;

int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++)
	    scanf("%d",&g[i]),s=s+g[i];
	p1=p2=s/n;
	if (s%n!=0)
	    p2++;
	for (i=1;i<=n;i++)
	{
	    if (p1>g[i])
	        t1=t1+p1-g[i];
	    if (p2<g[i])
	        t2=t2+g[i]-p2;
	}
	printf("%d",max(t1,t2));
	return 0;
}
