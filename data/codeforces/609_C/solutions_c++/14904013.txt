#include <bits/stdc++.h>
using namespace std;

int main() {
    int i,n;cin>>n;
    int a[n];
    long long sum=0,ans=0;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
        sum+=a[i];
    }
    int temp=sum%n,j=n-temp;
    sort(a,a+n);
    for(i=0;i<j;i++)
        ans+=(abs(sum/n-a[i]));
    for(i=j;i<n;i++)
        ans+=(abs(a[i]-sum/n-1));
    cout<<ans/2<<'\n';
    return 0;
}
