#include<bits/stdc++.h>
using namespace std;

int n,m,a[100005],sum,te,co;
long long re,re1,re2;

int main(){
cin>>n;sum=0;
for(int i=0;i<n;i++){
    cin>>a[i];
    sum+=a[i];
}
te=sum/n;re1=(long long)0;re2=(long long)0;
for(int i=0;i<n;i++){
    if(a[i]<=te) re1+=(long long)(te-a[i]);
    else re2+=(long long)(a[i]-te-1);
}
cout<<max(re1,re2)<<endl;
return 0;
}
