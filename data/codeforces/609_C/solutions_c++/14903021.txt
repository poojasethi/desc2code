#include <iostream>
#include <algorithm>
using namespace std;
const int N = 1e5 + 1;
int ar[N];
int main(){
	int n,i,sum=0; cin >> n;
	for (i = 0; i < n; ++i)cin >> ar[i],sum+=ar[i];
	int mid = sum / n, big = sum - n*mid, ret = 0;
	for (i = 0; i < n; ++i){
		if (ar[i]>mid){
			if (big>0)
				--big, ret += ar[i] - mid -1;
			else
				ret += ar[i] - mid;
		}
	}
	cout << ret << endl;
}
