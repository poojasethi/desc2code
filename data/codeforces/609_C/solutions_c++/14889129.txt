#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int n,s0=0,s1=0,s=0;
    cin>>n;
    vector<long long int> a(n);
    for(long long int i=0;i<n;i++){
        cin>>a[i];
        s=s+a[i];
    }
    s=s/n;
    for(long long int i=0;i<n;i++){
        if(s>a[i]) s1=s1+s-a[i];
        else if(s<a[i]) s0=s0+a[i]-s-1;
    }
    cout<<max(s0,s1);
    return 0;
}
