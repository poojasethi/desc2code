#include <bits/stdc++.h>
using namespace std;

#define N 100005
int arr[N];
int main()
{
	int n;
	cin >> n;
	int avg = 0;

	for(int i = 0; i<n; i++)
	{
		scanf("%d", &arr[i]);
		avg+=arr[i];
	}

	sort(arr, arr+n);
	int l = avg%n;
	avg/=n;
	int ct = 0;
	for(int i = 0; i<n-l; i++)
	{
		ct+=abs(avg-arr[i]);
	}

	for(int i = n-l; i<n; i++)
	{
		ct+=abs(avg+1-arr[i]);
	}

	cout << ct/2 << endl;
}
