#include <bits/stdc++.h>
using namespace std;
int n,a[100005],ans,k,sum;
int main(){
	scanf("%d",&n);
	for(int x=0;x<n;x++){
		scanf("%d",&a[x]);
		sum+=a[x];
	}
	sort(a,a+n);
	for(int x=0;x<n-sum%n;x++) ans+=abs(sum/n-a[x]);
	for(int x=n-sum%n;x<n;x++) ans+=abs(sum/n+1-a[x]);
	printf("%d\n",ans/2);
	return 0;
}
