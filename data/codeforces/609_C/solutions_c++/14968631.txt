#include <bits/stdc++.h>
using namespace std;

int n, m[100005], r, s, t, u;

int main () {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &m[i]);
		s += m[i];
	}
	r = s%n;
	s /= n;
	for (int i = 0; i < n; i++) {
		if (m[i] > s) {
			t += m[i]-s;
			u++;
		}
	}
	printf("%d\n", t-min(r,u));
	return 0;
}
