#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long int n;
	cin>>n;
	long long int a[n+1];
	long long int S=0,ans=0;
	for(long long int i=1;i<=n;i++)
	{
		cin>>a[i];
		S+=a[i];
	}
	long long temp=S;
	S/=n;
	sort(a+1,a+n+1);
	long long int j=temp-S*n;
	for(long long int i=n;i>0;i--,j--)
	{
		if(j==0)
			S--;
		ans+=max((long long)0,a[i]-S-1);
	}
	cout<<ans<<endl;
	return 0;
}
