#include <iostream>
#include <algorithm>
#include <cstdlib>
using namespace std;
int main (){
int n,sum=0,ans=0;
cin>>n;
int a[n+1],b[n+1];
for(int k=1;k<=n;k++)
cin>>a[k];
sort(a+1,a+n+1);
for(int k=1;k<=n;k++)
sum+=a[k];
for(int k=1;k<=n;k++)
b[k]=sum/n;
for(int k=n;k>n-(sum%n);k--)
b[k]++;
for(int k=1;k<=n;k++)
ans+=abs(a[k]-b[k]);
cout<<ans/2;
}
