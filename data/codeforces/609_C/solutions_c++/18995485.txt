#include<bits/stdc++.h>
using namespace std;
#define f(i,a,b) for(int i=a;i<=b;i++)
int A[100000];
int main(){
	int n,sum=0,b,c1=0,c2=0,cnt;
	cin>>n;
	f(i,1,n){
		cin>>A[i];
		sum+=A[i];
		b=sum/n;
	}
	f(i,1,n){
		if(A[i]<b) 
		c1+=b-A[i];
		if(A[i]>b+1)
		c2+=A[i]-b-1;
	}
	cnt=max(c1,c2);
	cout<<cnt;
}
