#include<bits/stdc++.h>
using namespace std;

const int N = (int)1e5+11;
int a[N];

int main()
{
	int n;
	cin>>n;
	int sum=0;
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		sum+=a[i];
	}
	sum=sum/n;
	int ans=0;
	int tmp=0;
	for(int i=0;i<n;i++){
		if(a[i]<sum)
			ans+=sum-a[i];
		else if(a[i]>sum){
			tmp+=a[i]-sum-1;
		}
	}
	tmp=tmp-ans;
	if(tmp>0)
		ans+=tmp;
	cout<<ans<<endl;
	return 0;
}
