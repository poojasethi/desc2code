#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

#define mp(x, y) make_pair((x), (y))

int n;
int a[100005];
int s;
int ans;

int main()
{
	scanf("%d", &n);
	for(int i=0; i<n; i++) scanf("%d", &a[i]);
	for(int i=0; i<n; i++) s+=a[i];
	sort(a, a+n);
	ans=0;
	for(int i=0; i<n; i++) {
		ans+=abs(a[i]-(s/n+(i>=n-s%n)));
	}
	printf("%d\n", ans/2);

	return 0;
}
