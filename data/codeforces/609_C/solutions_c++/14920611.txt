#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,a[100005],ans,s;
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]),s+=a[i];
	sort(a+1,a+n+1);
	for (int i=1;i<=n-s%n;i++) ans+=abs(a[i]-s/n);
	for (int i=n-s%n+1;i<=n;i++) ans+=abs(a[i]-s/n-1);
	printf("%d",ans/2);
}
