#include <iostream>
#include <algorithm>
#include<math.h>
using namespace std;
const int MAX=1e5+9;
int a[MAX],sum,careful,t2,ans,be;
int main()
{
    cin>>be;
    for(int i=0;i<be;i++) cin>>a[i],sum+=a[i];
    t2=sum%be,careful=be-t2;
    sort(a,a+be);
    for (int i=0;i<careful;i++) ans+=abs(sum/be-a[i]);
    for (int i=careful;i<be;i++) ans+=abs(a[i]-sum/be-1);
    cout<<ans/2;
}
