import math

n= int(raw_input())

arr= map(int, raw_input().split())

arr.sort()

avg=(sum(arr) )/len(arr)
prod= avg*n
diff= sum(arr)-prod

ret=0

for i in range(0, n):
	ret+= abs(arr[i]-avg)

for i in xrange(len(arr)-1, -1, -1):
	if diff<=0:
		break
	diff-=1
	if arr[i]>avg:
		ret-=1
	else:
		ret+=1

print ret/2
