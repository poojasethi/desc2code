def main():
    serversCount = int(raw_input())
    serverLoadsList = sorted(map(int, raw_input().split()))

    serversJobs = sum(serverLoadsList)
    remainder = serversJobs % serversCount
    serversWithMinLoad = serversCount - remainder

    minServerLoad = serversJobs / serversCount
    maxServerLoad = minServerLoad
    if remainder != 0:
        maxServerLoad += 1

    moves = 0
    serverIdx = 0
    while serverIdx < serversCount:
        serverLoad = serverLoadsList[serverIdx]
        if serverIdx < serversWithMinLoad:
            if serverLoad > minServerLoad:
                moves += serverLoad - minServerLoad
        else:
            if serverLoad > maxServerLoad:
                moves += serverLoad - maxServerLoad

        serverIdx += 1

    print moves


if __name__ == "__main__":
    main()
