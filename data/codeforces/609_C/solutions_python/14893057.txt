n = input()
list = map(int,raw_input().split())
sum=0
outover=0
outunder=0
for item in list:
    sum+=item

if sum % n ==0:
    aver=sum/n
    value=False
else:
    aver=sum/n
    value=True

if not value:
    for item in list:
        if item <aver:
            outunder=outunder+aver-item

        elif item > aver:
            outover=outover+item-aver

else:
    for item in list:
        if item < aver:
            outunder=outunder+aver-item
        elif item > aver+1:
            outover=outover+item-aver-1

if outover>outunder:
    print outover
else: print outunder
