__author__ = 'aste'


def main():
    n = int(raw_input())
    m = map(int, raw_input().split())

    m.sort()

    s = sum(m)

    a = s/n
    b = s%n
    res = 0
    for i in range(0, n):
        if i < n - b:
            res += abs(m[i] - a)
        else:
            res += abs(m[i] - (a + 1))
    print res/2

main()
