n = int(raw_input())
num = map(int, raw_input().split())
s = 0
for i in num:
	s += i
low = s/n 
temp = s - low*n 
high = low+1
highsum = temp
lowsum = n - highsum
ans = 0
for i in num :
	if i <= low:
		if lowsum >= 1:
			ans += low - i 
			lowsum -= 1
		else :
			ans += high - i 
			highsum -= 1
	elif i >= high :
		if highsum >= 1:
			ans += i - high 
			highsum -= 1
		else:
			ans += i - low 
			lowsum -= 1

print ans/2
