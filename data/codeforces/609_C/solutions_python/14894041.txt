n = int(raw_input())
tasks = [int(i) for i in raw_input().split()]
tasks_sum = sum(tasks)
tasks.sort()
task_int = tasks_sum/n
task_frac = tasks_sum % n
balanced_tasks = [task_int for i in range(n)]
for i in range(task_frac):
    balanced_tasks[n - i - 1] += 1
result = 0
for i in range(n):
    result += abs(balanced_tasks[i] - tasks[i])
print result/2
