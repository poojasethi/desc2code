def loadBalancing(tasks, serversCnt):
	total = sum(tasks)

	# Consider the server we need to ressign tasks to (it means t < balanced task),
	# and count the tasks needed.
	if(total % serversCnt == 0):
		cnt = 0
		balancedTask = total / serversCnt
		for t in tasks:
			if(t < balancedTask):
				cnt += balancedTask - t
		# print(balancedTask)
		return cnt
	else:
		balancedTask = int(total / serversCnt)
		more = total % serversCnt
		# Calculate the wrong less!!!!!!!!!
		less = serversCnt - more
		# print(total, more, less)
		lessCnt, moreCnt = 0, 0
		for t in tasks:
			if(t <= balancedTask):
				lessCnt += balancedTask - t
				if(less > 0):
					less -= 1
				else:
					more -= 1
					lessCnt += 1
			# print(less, more, balancedTask)

			# else:
			# 	moreCnt += t - (balancedTask + 1)
			# 	if(more > 0):
			# 		more -= 1
			# 	else:
			# 		less -= 1
			# 		moreCnt += 1
		# return (lessCnt + moreCnt) / 2
		return lessCnt
