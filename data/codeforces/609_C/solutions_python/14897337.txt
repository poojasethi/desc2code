n = int(raw_input())

a = map(int, raw_input().split())

b = []

s = sum(a)

div = s / n

rem = s % n

for i in range(n):
    b.append(div)

for i in range(n):
    if rem:
        b[i] += 1
        rem -= 1

a = sorted(a)
b = sorted(b) 

ans = 0

for i in range(n):
    ans += abs(a[i] - b[i])

ans /= 2

print ans
