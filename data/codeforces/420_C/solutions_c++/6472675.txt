#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
typedef pair <int, int> PII;
const int MN=4e5;
int n,m;
map <PII,LL> mp;
map <PII,LL> ::iterator ii;
int t[MN],w[MN];

int main(){
    scanf("%d%d",&n,&m);
    int i,j,k;
    for (i=1; i<=n; i++){
        int x,y;
        scanf("%d%d",&x,&y);
        if (x>y) swap(x,y);
        mp[PII(x,y)]++;
        t[x]++; t[y]++;
        w[x]++; w[y]++;
    }
    sort(t+1,t+n+1);
    LL sum=0;
    for (i=n; i>=1; i--){
        j=lower_bound(t+1,t+i,m-t[i])-t;
        sum+=i-j;
    }
    for (ii=mp.begin(); ii!=mp.end(); ii++){
        LL tmp=w[ii->first.first]+w[ii->first.second];
        if (tmp>=m && tmp-(ii->second)<m) sum--;
    }
    cout << sum << endl;
    return 0;
}
