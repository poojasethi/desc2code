#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 300010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,t,x,y,v;
int s[N],a[N*2],next[N*2],head[N],tr[N];
int64 an;
inline void add(int x,int y) {
	a[++t]=y,next[t]=head[x],head[x]=t;
	a[++t]=x,next[t]=head[y],head[y]=t;
}
inline void inc(int x,int y) {
	x=n+1-x;
	for (;x<=n+1;x+=x&-x) tr[x]+=y;
}
inline int cal(int x) {
	x=max(x,0);
	x=n+1-x;
	int an=0;
	for (;x;x-=x&-x) an+=tr[x];
	return an;
}
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n) {
		scanf("%d%d",&x,&y);
		s[x]++,s[y]++;
		add(x,y);
	}
	For(i,1,n) inc(s[i],1);
	For(i,1,n) {
		inc(s[i],-1);
		for (v=head[i];v;v=next[v]) {
			int A=a[v];
			inc(s[A],-1);
			s[A]--;
			inc(s[A],1);
		}
		an+=cal(m-s[i]);
		inc(s[i],1);
		for (v=head[i];v;v=next[v]) {
			int A=a[v];
			inc(s[A],-1);
			s[A]++;
			inc(s[A],1);
		}
	}
	printf("%I64d\n",an/2);
	//for(;;);
	return 0;
}
