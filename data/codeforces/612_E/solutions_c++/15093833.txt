#include<cstdio>
#include<vector>
#include<algorithm>
#include<utility>
#include<ctime>
using namespace std;

int n,a[1000005],ans[1000005],k=0;
pair<int,int> s[1000005];
vector<int> v[1000005];
bool vis[1000005];

int main(){
scanf("%d",&n);
	for(int i=1; i<=n; i++)scanf("%d",a+i);
	for(int i=1; i<=n; i++){
		if(vis[i])continue;
		int last=i;
		while(!vis[last]){
			vis[last]=1;
			v[k].push_back(last);
			last=a[last];
		}
		if(v[k].size()%2==1){
			int mod = v[k].size();
			for(int j=0; j<mod; j++)ans[v[k][j]]=v[k][(j+1+mod/2)%mod];
			v[k].clear();
		}else k++;
	}
	for(int i=0; i<k; i++)s[i]=make_pair(v[i].size(),i);
	sort(s,s+k);
//	for(int i=0; i<k; i++)printf("%d %d %d\n",i,s[i].first,s[i].second);
	vis[0]=1;
	for(int i=0; i<k; i++){
		if(!vis[i])continue;
		if(i==k)return 0*printf("-1\n");
		if(s[i].first==s[i+1].first){
	//		printf("######\n");
			int mod=s[i].first,l=s[i].second,r=s[i+1].second;
			for(int j=0; j<mod; j++)ans[v[l][j]]=v[r][j];
			for(int j=0; j<mod; j++)ans[v[r][j]]=v[l][(j+1)%mod];
			vis[i]=vis[i+1]=0;
		}
		else return 0*printf("-1\n");
	}
	for(int i=1; i<=n; i++)printf("%d ",ans[i]); printf("\n");
//	for(int i=0; i<k; i++)for(int j=0; j<v[i].size(); j++)printf("%d ",v[i][j]); printf("\n");	
	return 0;
}
