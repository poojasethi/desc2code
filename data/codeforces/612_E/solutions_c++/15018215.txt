#include<bits/stdc++.h>
using namespace std;
typedef long long int uli;
const int mx=1e6+10;
bool vis[mx];
vector<int>cycle;
int p[mx],q[mx],c;
vector<int>even[mx];
bool ok=true;

void ev(){
   if(even[c].empty()){
      even[c]=cycle;
      return;
   }
   int x=0,y=0;
   for(int i=0;i<c+c;i++){
      if(i%2==0){
         q[cycle[x]]=even[c][y];
         x++;
      }
      else{
         q[even[c][y]]=cycle[x];
         y++;
      }
      x%=c,y%=c;
   }
   even[c].clear();
}
void od(){
   if(c==1){
      q[cycle[0]]=cycle[0];
      return;
   }
   int x=0;
   int y=(c+1)/2;
   for(int j=0;j<c;j++){
      if(j%2==0){
         q[cycle[x]]=cycle[y];
         x++;
      }
      else{
         q[cycle[y]]=cycle[x];
         y++;
      }
      x%=c,y%=c;
   }
}
int main(){
   int n;
   scanf("%d",&n);
   for(int i=0;i<n;i++){
      scanf("%d",p+i);
      p[i]--;
   }
   bool ok=true;
   for(int i=0;i<n && ok;i++)if(!vis[i]){
      int u=i;
      cycle.clear();
      while(!vis[u]){
         vis[u]=true;
         cycle.push_back(u);
         u=p[u];
      }
      c=cycle.size();
      if(c%2==0)ev();
      else od();
   }
   for(int i=0;i<mx && ok;i++)
      if(!even[i].empty())ok=false;

   if(!ok){
      puts("-1");
      return 0;
   }
   for(int i=0;i<n;i++){
      if(i!=0)printf(" ");
      printf("%d",1+q[i]);
   }
   printf("\n");
   return 0;
}
