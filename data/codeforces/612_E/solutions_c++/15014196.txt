#include <bits/stdc++.h>
using namespace std;
int n,p[1000005],q[1000005],cnt=0,a2[1000005];
bool vis[1000005];
vector <int> a[1000005];
typedef pair<int,int> ii;
vector <ii> v2;
int main(){
	scanf("%d",&n);
	for(int x=1;x<=n;x++) scanf("%d",&p[x]);
	for(int x=1;x<=n;x++){
		if(vis[x]) continue;
		int v=x;
		do{
			vis[v]=1;
			a[cnt].push_back(v);
			v=p[v];
		}while(v!=x);
		v2.push_back(ii(a[cnt].size(),cnt));
		cnt++;
	}
	sort(v2.begin(),v2.end());
	for(int x=0;x<v2.size();x++){
		if(v2[x].first&1){
			int k=v2[x].second;
			for(int y=0;y<=a[k].size()/2;y++) a2[y*2]=a[k][y];
			for(int y=a[k].size()/2+1;y<a[k].size();y++) a2[(y-a[k].size()/2-1)*2+1]=a[k][y];
			for(int y=0;y<a[k].size()-1;y++) q[a2[y]]=a2[y+1];
			q[a2[a[k].size()-1]]=a2[0];
		}
		else{
			if(x==v2.size()-1||v2[x+1].first!=v2[x].first){
				printf("-1\n");
				return 0;
			}
			int k1=v2[x].second,k2=v2[x+1].second;
			for(int y=0;y<a[k1].size()-1;y++){
				q[a[k1][y]]=a[k2][y];
				q[a[k2][y]]=a[k1][y+1];
			}
			q[a[k1][a[k1].size()-1]]=a[k2][a[k2].size()-1];
			q[a[k2][a[k2].size()-1]]=a[k1][0];
			x++;
		}
	}
	for(int x=1;x<n;x++) printf("%d ",q[x]);
	printf("%d\n",q[n]);
	return 0;
}
