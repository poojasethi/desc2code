#include <iostream>
#include <cstdio>

using namespace std;

int n;
int p[2000000];
int q[2000000];
int visto[2000000];
int inicial[2000000];

int longitud(int u)
{
  if (visto[u]) return 0;
  visto[u]=1;
  return 1+longitud(p[u]);
}

void combina(int u,int v)
{
  if (q[u]) return;
  q[u]=v;
  combina(v,p[u]);
}

int main()
{
  scanf("%d",&n);
  for (int i=1;i<=n;i++)
    scanf("%d",p+i);
  int count=0;
  for (int u=1;u<=n;u++) {
    if (not visto[u]) {
      int len=longitud(u);
      if (len%2) {
	count+=len;
	int v=u;
	for (int i=0;i<=len/2;i++)
	  v=p[v];
	combina(u,v);
      } else {
	if (not inicial[len]) inicial[len]=u;
	else {
	  count+=2*len;
	  int v=inicial[len];
	  inicial[len]=0;
	  combina(u,v);
	}
      }
    }
  }
  if (count<n) cout<<-1<<endl;
  else {
    for (int i=1;i<=n;i++) {
      if (i>1) printf(" ");
      printf("%d",q[i]);
    }
    printf("\n");
  }
}
