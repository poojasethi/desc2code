#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>

using namespace std;
vector<int> next(1000000+10,-1);
vector<bool> vis(1000000+10,false);
vector<int> next2(1000000+10,-1);
vector<int> cycle;
vector<vector<int> > length(1000000+10);
int dfs(int x){
	vis[x]=true;
	cycle.push_back(x);
	if(!vis[next[x]]){
		return 1+dfs(next[x]);
	}else{
		return 1;
	}
}
int main(){
	int n;
	cin>>n;
	for(int i=1;i<=n;i++){
		scanf("%d",&next[i]);
	}
	for(int i=1;i<=n;i++){
		if(!vis[i]){
			cycle.clear();
			int size=dfs(i);
			if(size%2){
				for(int i=0;i<(int)cycle.size();i++){
					next2[cycle[i]]=cycle[(i+(size+1)/2)%size];
				}
			}else{
				if(length[size].size()==0){
					length[size]=cycle;
				}else{
					for(int i=0;i<size;i++){
						next2[cycle[i]]=length[size][i];
						next2[length[size][i]]=cycle[(i+1)%size];
					}
					length[size].clear();
				}
			}
		}
	}
	for(int i=1;i<=1000000;i++){
		if(length[i].size()!=0){
			cout<<"-1"<<endl;
			return 0;
		}
	}
	for(int i=1;i<=n;i++){
		if(i>1) cout<<" ";
		cout<<next2[i];
	}
	cout<<endl;
	return 0;
}
