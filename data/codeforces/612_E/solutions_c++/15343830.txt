#include<iostream>
#include<stdio.h>
using namespace std;
int n,p[1000050],q[1000050],vis[1000050],h[1000050];
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    scanf("%d",&p[i]);
    for(int i=1;i<=n;i++)
    {
        if(!vis[i])
        {
            int len=0,now=i;
            while(!vis[now])
            {
                vis[now]=1;
                now=p[now];
                len++;
            }
            if(len&1)
            {
            int u=i,v=i;
            for(int j=1;j<=(len+1)/2;j++)
                v=p[v];
            q[i]=v;
            for(int j=1;j<=len;j++)
            {
                q[v]=p[u];
                v=q[v];
                u=q[u];
            }
            }
            else if(h[len])
            {
                int t=h[len],v=i,u=i;
                for(int j=1;j<=len;j++)
                {
                    q[u]=t;
                    u=q[u];
                    v=p[v];
                    q[u]=v;
                    u=v;
                    t=p[t];
                }
                h[len]=0;
            }
            else h[len]=i;
        }
    }
    for(int i=1;i<=n;i++)
    {
        if(h[i])
        {
            cout<<-1<<endl;
            return 0;
        }
    }
    for(int i=1;i<=n;i++)
        printf("%d%c",q[i],i==n?'\n':' ');

}
