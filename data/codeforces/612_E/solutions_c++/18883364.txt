#include<bits/stdc++.h>
using namespace std;
const int MAXN = 1e6+10;
int B[MAXN],vis[MAXN],ans[MAXN],id[MAXN];
vector<int> v[MAXN];
#define pb push_back
bool cmp(const vector<int> &a,const vector<int> &b)
{
    return a.size() < b.size();
}
int main()
{
    int i,n,cnt = 0,tot = 0;
    cin>>n;
    for(i = 1;i <= n;i++)
        scanf("%d",B+i);
    for(i = 1;i <= n;i++)if(vis[i] != 1){
        int tmp = i;
        ++cnt;
        do{
            vis[tmp] = 1;
            v[cnt].pb(tmp);
            tmp = B[tmp];
        }while(tmp != i);
    }
    sort(v+1,v+1+cnt,cmp);
    for(i = 1;i <= cnt;i++){
        int sz = v[i].size();
        if(sz & 1){
            for(int k = 0;k < sz;k++)
                id[k*2%sz] = v[i][k];
            for(int k = 0;k < sz;k++)
                ans[id[k]] = id[(k+1)%sz];
        }
        else if(sz == v[i+1].size()){   
            for(int k = 0;k < sz;k++){
                ans[v[i][k]] = v[i+1][k];
                ans[v[i+1][k]] = v[i][(k+1)%sz];
            }
            i++;
        }
        else{
            return puts("-1"),0;
        }
    }
    for(i = 1;i <= n;i++){
        printf("%d ",ans[i]);
    }
}
