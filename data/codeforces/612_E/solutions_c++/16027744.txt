#include <bits/stdc++.h>
using namespace std;
#define M 1001001
typedef pair<int, int> pii;

int n;
int q[M];
int p[M];
bool flag[M];

int m;
pii a[M];
bool ok() {
	sort(a + 1, a + m + 1);
	for (int i = 1; i <= m; i++) if (a[i].first % 2 == 0) {
		if (i == m || a[i].first < a[i+1].first) return 0;
		++i;
	}
	return 1;
}

int x[M<<1], nx;
int y[M<<1], ny;
void get(int *x, int &nx, int t) {
	nx = 0;
	while (flag[t]) {
		x[++nx] = t;
		flag[t] = 0;
		t = p[t];
	}
	for (int i = 1; i <= nx; i++) x[i+nx] = x[i];
}

void print() {
	for (int i = 1; i <= m; i++) {
		if (a[i].first % 2 == 0) {
			get(x, nx, a[i].second);
			get(y, ny, a[++i].second);
            assert(nx == ny);
			for (int j = 1; j <= nx; j++) q[x[j]] = y[j];
			for (int j = 1; j <= nx; j++) q[y[j]] = x[j+1];
		} else {
			get(x, nx, a[i].second);
            for (int j = 1; j <= nx; j++) q[x[j]] = x[j + (nx+1)/2];
		}
	}
	for (int i = 1; i <= n; i++) printf("%d%c", q[i], i == n ? '\n' : ' ');
}

int main() {
	//freopen("Edu4_E4.in", "r", stdin);
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%d", p + i);
	for (int i = 1; i <= n; i++) if (!flag[i]) {
		int g = 0;
		for (int x = i; !flag[x]; x = p[x]) {
			++g;
			flag[x] = 1;
		}
		a[++m] = pii(g, i);
	}
    if (!ok()) puts("-1");
    else print();
	return 0;
}
