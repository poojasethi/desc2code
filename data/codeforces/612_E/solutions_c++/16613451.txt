#include <cstring>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN=1000000;

int n,p[MAXN+10],q[MAXN+10];

int st[MAXN+10],top;

bool vis[MAXN+10];

int line[MAXN+10],lt;
int sta[MAXN+10];

int a[MAXN+10];

void init()
{
	cin>>n;
	for (int i=1;i<=n;i++) scanf("%d",&p[i]);
	for (int i=1;i<=n;i++)
		if (!vis[i]) {
			vis[i]=true; st[top=1]=i;
			int x=p[i];
			while (!vis[x]) { st[++top]=x; vis[x]=true; x=p[x]; }
			if (top&1) {
				int y=(top>>1)+2;
				for (int i=1;i<=(top>>1);i++) {
					q[st[i]]=st[y]; q[st[y++]]=st[i+1]; }
				q[st[(top>>1)+1]]=st[1]; }
			else {
				if (a[top]) {
					for (int i=1;i<=top;i++) {
						q[line[sta[top]+i]]=st[i]; q[st[i]]=line[sta[top]+i+1]; }
					q[st[top]]=line[sta[top]+1]; }
				else sta[top]=lt;
				a[top]^=1;
				for (int i=1;i<=top;i++) line[++lt]=st[i]; } }
	for (int i=1;i<=n;i++)
		if (a[i]) { puts("-1"); return ; }
	for (int i=1;i<=n;i++)
		printf("%d ",q[i]);
	return ;
}

int main()
{
	init();
	return 0;
}
