#include <cstdio>
#include <iostream>
using namespace std;
int n,a[1010000],ans[1010000],st[1010000],b[1010000];

int main()
{
	cin >> n;
	for (int i = 1;i <= n;i++) scanf("%d",&a[i]);
	for (int i = 1;i <= n;i++)
		if (!b[i])
		{
			int cnt = 0;
			for (int j = i;b[j]==0;j=a[j])
				b[j] = 1,cnt++;
			if (cnt % 2 == 1)
			{
				ans[i] = i;
				for (int j = 1;j <= (cnt+1)/2;j++) ans[i] = a[ans[i]];
					b[i] = 2;
				for (int j = i;b[ans[j]]==1;j=ans[j])
					ans[ans[j]] = a[j],b[j]=2;
			}
			else 
			{
				if (st[cnt]==0) st[cnt] = i;
				else 
				{
					ans[i] = st[cnt];
					b[i] = 2;
					for (int j = i;b[ans[j]]==1;j=ans[j])
						ans[ans[j]] = a[j],b[j]=2;
					st[cnt] = 0;
				}
			}
		}
	for (int i = 1;i <= n;i++)
		if (st[i])
		{
			printf("-1\n");return 0;
		}
	for (int i = 1;i <= n;i++)
		printf("%d ",ans[i]);
}
