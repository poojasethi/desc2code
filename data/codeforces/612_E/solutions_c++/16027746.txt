#include <cstdio>
#include <vector>

using namespace std;

#define N 1001000

int n, p[N], vis[N], m, q[N], f[N], g[N];
int c[N];
vector <int> v[N];

bool run() {
    for(int i = 1; i <= n; i ++) if(!vis[i]) {
        int j = p[i];
        v[m].push_back(i);
        vis[i] = 1;
        while(j != i) {
            v[m].push_back(j);
            vis[j] = 1;
            j =  p[j];
        }
        c[m] = v[m].size();
        f[c[m]] ++; m ++;
    }
    for(int i = 1; i <= n; i ++) if(!(i&1) && (f[i] & 1)) return 0;
    for(int i = 0; i < m; i ++) {
        if(c[i] & 1) {
            int k = (c[i] + 1) / 2;
            for(int j = 0; j < c[i]; j ++) q[v[i][j]] = v[i][(j + k) %c[i]];
        } else {
            if(f[c[i]] & 1) {
                int r = g[c[i]];
                for(int j = 0; j < c[i]; j ++) {
                    q[v[i][j]] = v[r][j];
                    q[v[r][j]] = v[i][(j + 1) %c[i]];
                }
            } else g[c[i]] = i;
            f[c[i]] --;
        }
    }
}

int main() {
    scanf("%d", &n);
    for(int i = 1; i <= n; i ++) scanf("%d", p + i);
    if(run()){for(int i = 1; i <= n; i ++) printf("%d ", q[i]); puts("");}
    else puts("-1");
    return 0;
}
