#include "bits/stdc++.h"
using namespace std;

const int maxn = 1000000 + 10;

int n;
int p[maxn], q[maxn], h[maxn];
bool vis[maxn];

int main()
{
    scanf("%d", &n);
    for(int i = 1; i <= n; i++) scanf("%d", p + i);
    for(int i = 1; i <= n; i++) if(!vis[i]) {
            int len = 0;
            for(int j = p[i]; !vis[j]; j = p[j]) {
                vis[j] = true;
                len++;
            }
            if(len & 1) {
                int t = i, u = i;
                for(int j = 0; j < (len + 1) / 2; j++) t = p[t];
                q[i] = t;
                for(int j = 1; j < len; j++) {
                    q[t] = p[u];
                    t = q[t];
                    u = q[u];
                }
            } else if(h[len]) {
                int t = i, u = i, v = h[len];
                for(int j = 0; j < len; j++) {
                    q[t] = v;
                    t = q[t];
                    u = p[u];
                    q[t] = u;
                    t = q[t];
                    v = p[v];
                }
                h[len] = 0;
            } else h[len] = i;
        }

    for(int i = 1; i <= n; i++) if(h[i]) {
            printf("-1\n"); return 0;
        }
    for(int i = 1; i <= n; i++) printf("%d ", q[i]);
    printf("\n");

    return 0;
}
