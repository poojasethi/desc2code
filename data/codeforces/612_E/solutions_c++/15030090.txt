#include <bits/stdc++.h>

using namespace std;

int a[1000005],b[1000005],visited[1000005],mat[1000005];

int visit(int i){
    int ln = 0;
    if(!visited[i]){
        ln++;int j = a[i];
        visited[i] = 1;
        while(j!=i){
            visited[j] = 1;
            ln++;
            j = a[j];
        }
    }
    return ln;
}


int main(){
    int n;scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    
    memset(mat,0,sizeof(mat));
    for(int i=1;i<=n;i++)
        mat[i] = visit(i);
    for(int i=1;i<=n;i++){
        int ln = mat[i];
        if(ln){
            if(ln&1){
            int c = (ln+1)/2 , d=i;
            for(int j=0;j<c;j++)
                d = a[d];
            for(int j=i,x=0;x<ln;x++){
                b[j] = d;
                d = a[d];
                j = a[j];
            }
            }
            else{
                int j;
                for(j=i+1;j<=n;j++)
                    if(mat[j]==ln)
                        break;
                
                if(j==n+1){
                    printf("%d\n",-1);return 0;
                }
                mat[j] = 0;
                int h = i;
                for(int x=0;x<ln;x++){
                    b[h] = j;
                    b[j] = a[h];
                    j = a[j];
                    h = a[h];
                }
            }
        }

    }

    for(int i=1;i<=n;i++)
        printf("%d ",b[i]);
    printf("\n");
}
