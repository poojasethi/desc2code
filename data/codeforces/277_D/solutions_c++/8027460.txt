// head

const ll e6=1000000;
struct task {
	ll ss,sl,pr;
	int ts,tl;
}t[1010];
bool cmp(const task &a,const task &b) {
	return a.tl*a.pr*(e6-b.pr)<b.tl*b.pr*(e6-a.pr);
}
int n,T,ss,sl;
double pr;
pair<ll,double> dp[1010][1600];
inline void upd(int i,int j,pair<ll,double> c) { if (j<=T) dp[i][j]=min(dp[i][j],c);}
int main() {
	scanf("%d%d",&n,&T);
	rep(i,0,n) {
		scanf("%d%d%d%d%lf",&ss,&sl,&t[i].ts,&t[i].tl,&pr);
		t[i].ss=ss*e6; t[i].pr=pr*e6+0.1; t[i].sl=sl*(e6-t[i].pr);
	}
	sort(t,t+n,cmp);
	rep(i,0,n) rep(j,0,T+1) {
		upd(i+1,j,dp[i][j]);
		upd(i+1,j+t[i].ts,mp(dp[i][j].fi-t[i].ss,dp[i][j].se+t[i].ts));
		upd(i+1,j+t[i].ts+t[i].tl,mp(dp[i][j].fi-t[i].ss-t[i].sl,
			(j+t[i].tl)*(1-1e-6*t[i].pr)+1e-6*t[i].pr*dp[i][j].se+t[i].ts));
	}
	pair<ll,double> ret=mp(0,0);
	rep(j,0,T+1) ret=min(ret,dp[n][j]);
	printf("%.10f %.10f\n",-1e-6*ret.fi,ret.se);
}
