#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
#define eps 1e-9
int n,m;
struct rec1{
       int s1,s2,t1,t2;
       double p;
       }a[1100];
bool cmp(rec1 a,rec1 b){
     return a.t2*a.p/(1-a.p)<b.t2*b.p/(1-b.p);
     }
struct rec2{
       long double s,t;
}f[1100][1600];
void Max(int i,int j,long double s,long double t){
     if(j>m)return;
     if(s>f[i][j].s+eps)f[i][j].s=s,f[i][j].t=t;
     else{
       if(fabs(s-f[i][j].s)<eps)f[i][j].t=min(f[i][j].t,t);     
     }
}
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)cin>>a[i].s1>>a[i].s2>>a[i].t1>>a[i].t2>>a[i].p;
    sort(a+1,a+n+1,cmp);
    for(int i=0;i<n;i++)
      for(int j=0;j<=m;j++){
        Max(i+1,j,f[i][j].s,f[i][j].t);
        Max(i+1,j+a[i+1].t1,f[i][j].s+a[i+1].s1,f[i][j].t+a[i+1].t1);
        Max(i+1,j+a[i+1].t1+a[i+1].t2,f[i][j].s+a[i+1].s1+a[i+1].s2*(1-a[i+1].p),
            (j+a[i+1].t1+a[i+1].t2)*(1-a[i+1].p)+(f[i][j].t+a[i+1].t1)*a[i+1].p);
      }
    rec2 ans=f[n][0];
    for(int i=1;i<=m;i++)
      if(f[n][i].s>ans.s)ans=f[n][i];
      else 
        if(fabs(f[n][i].s-ans.s)<eps)ans.t=min(ans.t,f[n][i].t);
    printf("%.10lf %.10lf\n",(double)ans.s,(double)ans.t);
    return 0;
}
