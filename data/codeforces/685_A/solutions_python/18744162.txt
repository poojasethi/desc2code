import itertools

def w7(t):
  x, w = 1, 1
  while x * 7 < t:
    x *= 7
    w += 1
  return w


n, m = map(int, raw_input().split())
a, b = w7(n), w7(m)
z = 0
if a + b <= 7:
  for s in itertools.permutations('0123456', a + b):
    if int(''.join(s[:a]), 7) < n and int(''.join(s[a:]), 7) < m:
      z += 1
print z
