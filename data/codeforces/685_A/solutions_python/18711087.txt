def f(x):
    for i in xrange(1, 8):
        if 7 ** i >= x:
            return i
    return 8

def g(x, l):
    ret = []
    for i in range(l):
        ret.append(x % 7)
        x = x / 7
    return set(ret)

(n, m) = map(int, raw_input().split())
(l1, l2) = (f(n), f(m))

if l1 + l2 > 7:
    print 0
    exit()

s1 = [g(i, l1) for i in xrange(0, n)]
s2 = [g(j, l2) for j in xrange(0, m)]
ans = len([1 for u in s1 for v in s2 if len(u.union(v)) == l1 + l2])

print ans
