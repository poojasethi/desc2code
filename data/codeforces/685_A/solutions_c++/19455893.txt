#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;
int vis[10];
bool judge(int i,int j,int x,int y)
{
	memset(vis,0,sizeof(vis));
	while(x--)
	{
		int h=i%7;
		i/=7;
		if(!vis[h])
		{
			vis[h]=1;continue;
		}
		return false;
	}
	while(y--)
	{
		int t=j%7;
		j/=7;
		if(!vis[t])
		{
			vis[t]=1;continue;
		}
		return false;
	}
	return true;
}
int main()
{
	int m,n,hd=0,md=0;
	long long ans=0;
	cin>>n>>m;
	int x=n-1,y=m-1;
	if(x==0) hd=1;
	if(y==0) md=1;
	while(x>0)
	{
		x/=7;
		hd++;
	}
	while(y>0)
	{
		y/=7;
		md++;
	}
	if(hd+md>7){
		cout<<"0"<<endl;
		return 0;
	}
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			if(judge(i,j,hd,md)) ans++;
		}
	 }
	 cout<<ans<<endl;
	 return 0; 
}
