#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n,m1;
  cin>>n>>m1;
  int len1=1,len2=1;
  for(int i=7;i<n;i*=7)
    len1++;
  for(int i=7;i<m1;i*=7)
    len2++;
  int ans=0;
  if(len1+len2<=7)
  {
    for(int i=0;i<n;i++)
    {
      for(int j=0;j<m1;j++)
      {
        vector<int>m(7,0);
        for(int a=i,k=0;k!=len1;k++,a=a/7)
        {
          m[a%7]++;
        }
        for(int a=j,k=0;k!=len2;k++,a=a/7)
        {
          m[a%7]++;
        }
      vector<int>::iterator q=max_element(m.begin(),m.end());
        if(*q<=1)
          ans++;
      }
    }
  }
  cout<<ans;
}
