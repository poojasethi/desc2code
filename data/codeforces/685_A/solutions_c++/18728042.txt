#include <cstdio>
#include <algorithm>
using namespace std;
const int ws[8]={1,7,49,343,2401,16807,117649,823543};
bool u[7];
int nw,mw,n,m,ans=0;

void dfs2(int x,int sum)
{
	if (sum*ws[x]>m) return;
	if (!x) {ans++; return;}
	for (int i=0;i<=6;i++)
	{
		if (u[i]) continue;
		u[i]=true;
		dfs2(x-1,sum*7+i);
		u[i]=false;
	}
}
void dfs1(int x,int sum)
{
	if (sum*ws[x]>n) return;
	if (!x) {dfs2(mw,0); return;}
	for (int i=0;i<=6;i++)
	{
		if (u[i]) continue;
		u[i]=true;
		dfs1(x-1,sum*7+i);
		u[i]=false;
	}
}

int main()
{
	scanf("%d%d",&n,&m);
	n--; m--;
	nw=upper_bound(ws,ws+7,n)-ws;
	mw=upper_bound(ws,ws+7,m)-ws;
	if (nw+mw>7 || !n && !m) {puts("0"); return 0;}
	if (!n || !m) u[0]=true;
	dfs1(nw,0);
	printf("%d",ans);
}
