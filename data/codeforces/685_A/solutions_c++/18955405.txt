#include <bits/stdc++.h>
using namespace std;

int n,m,cnt,f[7];

int calc(int x)
{
	int cnt=0;
	for(;x;x/=7) cnt++;
	return max(cnt,1);
}

int main()
{
	cin>>n>>m;
	int x=calc(n-1),y=calc(m-1);
	if(x+y>7)
	{
		puts("0");
		return 0;
	}

	for(int i=0;i<n;i++)
		for(int j=0;j<m;j++)
		{
			cnt++,memset(f,0,sizeof(f));
			for(int k=0,u=i;k<x;k++,u/=7) f[u%7]++;
            for(int k=0,u=j;k<y;k++,u/=7) f[u%7]++;
            for(int k=0;k<7;k++)
            	if(f[k]>1)
				{
					cnt--;
					break;
				}
		}
	cout<<cnt<<endl;
	return 0;
}
