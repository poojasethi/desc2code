#include<bits/stdc++.h>
using namespace std;
int n,m,dn,dm;
int arr[10];
int dig(int x)
{
	if(x==0)
		return 1;
	int ans=0;
	while(x)
		++ans,x/=7;
	return ans;
}
int state(int x,int d,int arr[])
{
	int s=0;
	for(int i=1;i<=d;++i,x/=7)
	{
		if(arr[x%7]==1)
			return -1;
		else
			arr[x%7]=1;	
	}
	return 0;
}
int calc(int n,int m)
{
	int ans=0;
	for(int i=0;i<=n;++i)
		for(int j=0;j<=m;++j)
		{
			memset(arr,0,sizeof(arr));
			int s1=state(i,dn,arr),s2=state(j,dm,arr);
			if(s1!=-1 && s2!=-1 && (s1&s2)==0)
				++ans;	
		}
	return ans;
}
int main()
{
	cin>>n>>m;
	--n,--m;
	dn=dig(n);
	dm=dig(m);
	if(dn+dm>7)
		cout<<0<<endl;
	else
		cout<<calc(n,m)<<endl;
	return 0;
}
