#include <bits/stdc++.h>

using namespace std;

int n, m;
int a, b, vis[10], mod, ans;

void doit(int d, int res) {
	if(d == a + b) {
		int x = res / mod;
		int y = res % mod;
		if(x < n && y < m) ans ++;
		return ;
	}
	for(int i = 0; i < 7; i ++) if(!vis[i]) {
		vis[i] = 1;
		doit(d + 1, res * 7 + i);
		vis[i] = 0;
	}
}

int run() {
	if(a + b > 7) return 0;
	mod = 1;
	for(int i = 0; i < b; i ++) mod *= 7;
	doit(0, 0);
	return ans;
}

int main() {
	scanf("%d%d", &n, &m);
	int x = n - 1;
	while(x) {
		x /= 7; a ++;
	}
	if(!a) a = 1;
	x = m - 1;
	while(x) {
		x /= 7; b ++;
	}
	if(!b) b = 1;
	printf("%d\n", run());
	return 0;
}
