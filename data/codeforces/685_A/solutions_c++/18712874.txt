#include<cstdio>
int n,m,c,nk,mk,u[7],nt[7],mt[7],ans;
void dfsm(int x)
{
	int i,nn=0,mm=0,p;
	if(x==mk)
	{
		for(i=0,p=1;i<nk;i++,p*=7)nn+=nt[i]*p;
		for(i=0,p=1;i<mk;i++,p*=7)mm+=mt[i]*p;
		if(nn<=n&&mm<=m)ans++;
		return; 
	}
	for(i=0;i<7;i++)if(!u[i])
	{
		mt[x]=i;
		u[i]=1;
		dfsm(x+1);
		u[i]=0; 
	}
}
void dfsn(int x)
{
	if(x==nk){dfsm(0);return;}
	for(int i=0;i<7;i++)if(!u[i])
	{
		nt[x]=i;
		u[i]=1;
		dfsn(x+1);
		u[i]=0; 
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	n--;m--;
	for(c=n;c;c/=7)nk++;
	for(c=m;c;c/=7)mk++;
	if(!n)nk++;if(!m)mk++; 
	dfsn(0);
	printf("%d",ans);
}
