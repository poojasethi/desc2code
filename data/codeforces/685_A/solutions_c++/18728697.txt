#include<stdio.h>
#include<algorithm>
using namespace std;
int n,m,ln,lm,res;
int flag[7];
int getl(int n){
	int l=0;
	while(n){
		l++;
		n/=7;
	}
	return l;
}
void dfs(int deep,int max,int value,int type){
	if(deep==max){
		if(type){
			if(value<m)res++;
		}
		else{
			if(value<n){
				dfs(0,lm,0,1);
			}
		}
		
		
		return ;
	}
	
	for(int i=0;i<7;i++){
		if(flag[i]==0){
			flag[i]=1;
			dfs(deep+1,max,value*7+i,type);
			flag[i]=0;
		}
		
	}
	
}



int main(){
	int i,j,k;
	while(~scanf("%d%d",&n,&m)){
		ln=max(getl(n-1),1);	
		lm=max(getl(m-1),1);	
	//	printf("%d %d\n",ln,lm);
		if(ln+lm>7){
			printf("0\n");
			continue;
		}
		res=0;
		dfs(0,ln,0,0);
		
		printf("%d\n",res);
		
	}	
	
	
	return 0;
}
