#include<cstdio>
#include<iostream>
using namespace std;
int ans=0,LIM,l_a=1,l_b=1,n,m;
bool v[10];
void dfs(int now,int val)
{
	if(now==l_a+l_b+1)
	{
		if(val>m) return;
		ans++; return;
	}
	if(now==l_a+1)
	{
		if(val>n) return;
		for(int i=0; i<7; i++) if(!v[i])
		{
			v[i]=true;
			dfs(now+1,i);
			v[i]=false;
		}
		return;
	}
	for(int i=0; i<7; i++) if(!v[i])
	{
		v[i]=true;
		dfs(now+1,val*7+i);
		v[i]=false;
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	n--,m--;
	long long cur=7;
	while(cur<=n)
	{
		cur*=7ll;
		l_a++;
	}
	cur=7;
	while(cur<=m)
	{
		cur*=7ll;
		l_b++;
	}
	if(l_a+l_b>7) {puts("0"); return 0;}
	dfs(1,0);
	printf("%d\n",ans);
	return 0;
}
