#include <bits/stdc++.h>
typedef long long LL;
using namespace std;
int n,m,a,b,ans;
int pow7[15]={1};
bool vis[7];
void dfs(int x,int cur)
{
    if(x<=a&&cur>n)return;
    if(x>a&&cur%pow7[x-a]>m)return;
    if(x==a+b)
    {
        ++ans;
        return;
    }
    for(int i=0;i<7;++i)
        if(!vis[i])
        {
            vis[i]=1;
            dfs(x+1,cur*7+i);
            vis[i]=0;
        }
}
int main()
{
    for(int i=1;i<=10;++i)
        pow7[i]=pow7[i-1]*7;
    scanf("%d%d",&n,&m);
    --n;--m;
    for(int t=n;t;t/=7,++a);
    a+=!n;
    for(int t=m;t;t/=7,++b);
    b+=!m;
    if(a+b>7)
    {
        puts("0");
        return 0;
    }
    dfs(0,0);
    printf("%d\n",ans);
    return 0;
}
