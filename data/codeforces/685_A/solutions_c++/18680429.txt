#include <bits/stdc++.h>

using namespace std;

long long alln,allm,n,m,N,M,a[102];
map < pair < int , int > , bool > f;
int i,ans;

int main()
 {

  cin>>n>>m;
  N=n; M=m; n--; m--;
  while (n > 0)
   {
    n/=7;
    alln++;
   }
   if (alln == 0) alln=1;
  while (m > 0)
   {
    m/=7;
    allm++;
   }
   if (allm == 0) allm=1;
  for (i=1;i<=7;i++)
   a[i]=i;
   if (allm+alln > 7){ cout<<0<<endl; return 0; }
  while (true)
   {
    n=0; m=0;
    for (i=1;i<=alln;i++)
     n=n*7+a[i]-1;
    for (i=alln+1;i<=alln+allm;i++)
     m=m*7+a[i]-1;
    if (n < N && m < M && !f[make_pair(n,m)])
        { ans++; f[make_pair(n,m)]=1; }
    if (!next_permutation(a+1,a+7+1)) break;
   }
  cout<<ans<<endl;
 }
