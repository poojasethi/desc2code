#include <bits/stdc++.h>
using namespace std;

long long n=1,k=1,u,v,p,a,b,x=7,y,i,j;
set <int> s;
main(){
	cin>>a>>b;
	
	while(a>x){
		x*=7;
		n++;
	}x=7;
	
	while(b>x){
		x*=7;
		k++;
	}
	
	if(n+k>7){cout<<0;return 0;}
	
	while(i<a){
		while(j<b){
			x=i;y=j;
			while(x>0||u<n){
				s.insert(x%7);
				x/=7;u++;
			}u=0;if(s.size()!=n){s.clear();break;}
			while(y>0||v<k){
				s.insert(y%7);
				y/=7;v++;
			}v=0;
			if(s.size()==n+k){p++;}
			s.clear();
			j++;
		}j=0;i++;
	}
	cout<<p;
}
