#include <bits/stdc++.h>

using namespace std;

int main()
{
    long long int n,m,l1=1,l2=1,ans=0;
    cin>>n>>m;
    for(long long int i=7;i<n;i*=7)
        l1++;
    for(long long int i=7;i<m;i*=7)
        l2++;
    if(l1+l2<=7)
    {
        for(long long int i=0;i<n;i++)
        {
            for(long long int j=0;j<m;j++)
            {
                vector<long long int> vec(7,0);
                for(long long int a=i,k=0;k<l1;a/=7,k++)
                    vec[a%7]+=1;
                for(long long int b=j,k=0;k<l2;b/=7,k++)
                    vec[b%7]+=1;
                if(*max_element(vec.begin(),vec.end())<=1)
                    ans++;
            }
        }
    }
    cout<<ans;
    return 0;
}
