#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int cn=1,cm=1, prem[7] = {0,1,2,3,4,5,6};
map<pair<int,int> , bool> mp;
ll ans;
int main(){
	ll n,m,tn=7,tm=7; cin >> n >> m;
	while(tn < n) tn*=7,cn++;
	while(tm < m) tm*=7,cm++;
	if(cn+cm > 7) return puts("0");
	do{
		ll ab7 = 0 , bb7 = 0;
		for(int i = 0 ; i < cn ; i++) ab7 = ab7 * 7 + prem[i];
		for(int i = cn ; i < cn+cm ; i++) bb7 = bb7 * 7 + prem[i];
		if(ab7 < n && bb7 < m && !mp[make_pair(ab7,bb7)]){
			ans++;
			mp[make_pair(ab7,bb7)] = 1;
		}
		//cout << ab7 << " " << bb7 << endl;
	}while(next_permutation(prem,prem+7));
	cout << ans << endl;
}
