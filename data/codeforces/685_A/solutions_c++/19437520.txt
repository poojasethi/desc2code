#include<bits/stdc++.h>
using namespace std;

int num[10],ans;
void check(int x,int y,int n,int m)
{
    for(int i=0;i<10;i++)num[i]=0;
    if(!n)num[0]++;
    if(!m)num[0]++;
    while(n){
        num[x%7]++;
        x/=7;
        n/=7;
    }
    while(m){
        num[y%7]++;
        y/=7;
        m/=7;
    }
    for(int i=0;i<10;i++)
        if(num[i]>1)return;
    ans++;
}
int n,m;    
int main()
{
    scanf("%d%d",&n,&m);
    if(1ll*n*m>1e7){
        puts("0");
        return 0;
    }
    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++)
            check(i,j,n-1,m-1);
    cout<<ans<<endl;
}
