#define problem "6850"
#include <cstdio>
#include <iostream>
#include <memory.h>
using namespace std;
int N, M, A[10];

int main(){

    cin >> N >> M;
    int cn(0), cm(0);
    for(int tmp = N - 1; tmp; tmp /= 7) cn++;
    for(int tmp = M - 1; tmp; tmp /= 7) cm++;
	if(cn + cm > 7){
		cout << "0";
		return 0;
	}
	cn = max(cn, 1);
	cm = max(cm, 1);
	int ans(0);
	for(int i = 0; i < N; i++) for(int j = 0; j < M; j++){
		bool ok(true);
		memset(A, 0, sizeof(A));
		for(int tmp = i, k = 0; k < cn; k++, tmp /= 7) if(A[tmp % 7]++) ok = false;
		for(int tmp = j, k = 0; k < cm; k++, tmp /= 7) if(A[tmp % 7]++) ok = false;
		ans += ok;
	}
	cout << ans;
}
