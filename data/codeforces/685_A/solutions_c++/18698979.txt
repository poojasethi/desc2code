#include<bits/stdc++.h>
using namespace std;
int n,m,dn,dm;
int dig(int x)
{
	if(x==0)
		return 1;
	int ans=0;
	while(x)
		++ans,x/=7;
	return ans;
}
int state(int x,int d)
{
	int s=0;
	for(int i=1;i<=d;++i,x/=7)
	{
		if(s&(1<<(x%7)))
			return -1;
		s|=(1<<(x%7));
	}
	return s;
}
int calc(int n,int m)
{
	int ans=0;
	for(int i=0;i<=n;++i)
		for(int j=0;j<=m;++j)
		{
			int s1=state(i,dn),s2=state(j,dm);
			if(s1!=-1 && s2!=-1 && (s1&s2)==0)
			{
				
				++ans;
			}
		}
	return ans;
}
int main()
{
	cin>>n>>m;
	--n,--m;
	dn=dig(n);
	dm=dig(m);
	if(dn+dm>7)
		cout<<0<<endl;
	else
		cout<<calc(n,m)<<endl;
	return 0;
}
