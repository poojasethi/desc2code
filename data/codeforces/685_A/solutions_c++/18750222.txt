#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

int num[10];
inline int check(int x, int y, int n, int m) {
	memset(num, 0, sizeof(num));
	if (!n) ++(*num);
	if (!m) ++(*num);
	while (n) {
		++num[x % 7]; x/= 7; n /= 7;
	}
	while (m) {
		++num[y % 7]; y/= 7; m /= 7;
	}
	for (int i = 0; i < 7; ++i)
		if (num[i] > 1) return 0;
	return 1;
}

int main() {
	int n,m;
	scanf("%d%d", &n, &m);
	if ((ll)n * m > 1e7) {
		puts("0"); return 0;
	}
	int ans = 0;
	for (int i = 0; i < n; ++i) 
		for (int j = 0; j < m; ++j)
			ans += check(i, j, n - 1, m - 1);
	printf("%d", ans);
	return 0;
}
