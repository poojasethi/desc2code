#include <iostream>
#include <cstdlib>

using namespace std;

int n,m;

int visto[8];

bool mirar(int x,int pos)
{
  for (int i=0;i<pos;i++) {
    int ind=x%7;
    if (visto[ind]) return false;
    visto[ind]++;
    x/=7;
  }
  return true;
}

int posiciones(int val)
{
  if (val==0) return 1;
  int pos=0;
  while (val) {
    pos++;
    val/=7;
  }
  return pos;
}

int main()
{
  cin>>n>>m;
  int posn=posiciones(n-1);
  int posm=posiciones(m-1);
  //cout<<posn<<" "<<posm<<endl;
  if (posn+posm>7) {
    cout<<0<<endl;
    exit(0);
  }
  int sol=0;
  for (int xn=0;xn<n;xn++) {
    for (int ym=0;ym<m;ym++) {
      for (int i=0;i<8;i++)
	visto[i]=0;
      sol+=mirar(xn,posn) and mirar(ym,posm);
    }
  }
  cout<<sol<<endl;
}
