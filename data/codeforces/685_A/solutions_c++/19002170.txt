#include<bits/stdc++.h>
using namespace std;

int n,m;
int res;
bool u[10];

void input()
{
	cin>>n>>m;
}

int add(int a,int b,int la,int lb)
{
	int i;
	for(i=0;i<la;i++,a/=7)
	{
		if(u[a%7]) return 0;
		else u[a%7]=1;
	}
	for(i=0;i<lb;i++,b/=7)
	{
		if(u[b%7]) return 0;
		else u[b%7]=1;
	}
	return 1;
}

void solve()
{
	int bit1=1,bit2=1,bit,cnt_n,cnt_m,part1,part2,part,i,j;
	for(i=7;i<n;i*=7,bit1++);
	for(i=7;i<m;i*=7,bit2++);
	bit = bit1 + bit2;
	if(bit>7) res=0;
	else
	{
		res = 0;
		for(i=0;i<n;i++){
			for(j=0;j<m;j++)
			{
				memset(u,0,sizeof(u));
				res += add(i,j,bit1,bit2);
			}
		}
		
	}
}

void output()
{
	cout<<res<<endl;
}

int main()
{
	input();
	solve();
	output();
	return 0;
}
