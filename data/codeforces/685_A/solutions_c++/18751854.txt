#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

int n,m,n0,m0,t,val[10],Ans;
bool vis[10];

void work(int u,int v,int id)
{
	if (u>=n||v>=m) return;
	if (id==n0+m0) {Ans++; return;}
	for (int i=0;i<7;i++) if (!vis[i])
	{
		if (id<n0) u+=val[id]*i;
		else v+=val[id-n0]*i;
		vis[i]=1;
		work(u,v,id+1);
		vis[i]=0;
		if (id<n0) u-=val[id]*i;
		else v-=val[id-n0]*i;
	}
}

int main()
{
	scanf("%d%d",&n,&m);
	t=n-1?n-1:1; while (t) n0++,t/=7;
	t=m-1?m-1:1; while (t) m0++,t/=7;
	if (n0+m0>7) {printf("0\n"); return 0;}
	val[0]=1;
	for (int i=1;i<=7;i++) val[i]=val[i-1]*7;
	work(0,0,0);
	printf("%d\n",Ans);
}
