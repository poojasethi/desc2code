#include <iostream>
#include<cstdio>

using namespace std;

int a[30],b1,b2,n,m;

void getbit(){
	int t = n;
	b1 = 0;
	if(!t) b1 = 1;
	else for(;t;t/=7) b1++;
	b2 = 0;
	t = m;
	if(!t) b2 = 1;
	else for(;t;t/=7) b2++;
}

bool check(int x,int y){
	int i,len = 0;
	for(i = 0;i < 7;i++) a[i] = 0;
	for(i = 0;i < b1;i++,x/=7) a[x%7]++;
	for(i = 0;i < b2;i++,y/=7) a[y%7]++;
	for(i = 0;i < 7;i++) if(a[i]>1) return false;
	return true;
}

int sol(){
	int i,j,res = 0;
	for(i = 0;i <= n;i++){
		for(j = 0;j <= m;j++){
			res += check(i,j);
		}
	}
	return res;
}

int main() {
	// your code goes here
	int i,j;
	scanf("%d%d",&n,&m);n--,m--;
	getbit();
	printf("%d\n",b1+b2<=7?sol():0);
	return 0;
}
