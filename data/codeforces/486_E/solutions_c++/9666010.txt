#include<queue>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#define MAXN 100000

using namespace std;
int a[MAXN+80],d[2][MAXN+80],cnt[MAXN+80],g[MAXN+80];
bool ans[MAXN+80];
int main()
{
	int n,i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]);
	for(i=1;i<=n;i++)
	{
		j=lower_bound(g+1,g+g[0]+1,a[i])-g;
		d[0][i]=j;
		g[j]=a[i];
		if(g[0]<j)
			g[0]=j;
	}
	for(i=n;i;i--)
	{
		j=lower_bound(g+1,g+g[0]+1,-a[i])-g;
		d[1][i]=j;
		g[j]=-a[i];
		if(g[0]<j)
			g[0]=j;
	}
	for(i=1;i<=n;i++)
		if(d[0][i]+d[1][i]-1!=g[0])
			ans[i]=1;
		else
			cnt[d[0][i]]++;
	for(i=1;i<=n;i++)
		if(ans[i])
			putchar('1');
		else
			putchar('2'+(cnt[d[0][i]]==1));
	return 0;
}
