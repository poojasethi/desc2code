#include <iostream>
#include <algorithm>

using namespace std;

int a[100002], dp[100001], x[100005], out[100005], MAX[100005], cnt[100005];

int main()
{
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	int ans = 0;
	for(int i = 0; i < n; i++)
	{
		int lo = -1, hi = ans;
		while(lo < hi - 1)
		{
			int mid = (lo + hi) / 2;
			if(x[mid] >= a[i])
				hi = mid;
			else
				lo = mid;
		}
		dp[i] = hi + 1;
		if(hi == ans)
			x[ans++] = a[i];
		else
			x[hi] = min(x[hi], a[i]);
	}
	fill(MAX, MAX + n, -1);
	for(int i = n - 1; i >= 0; i--)
	{
		if(dp[i] == ans)
		{
			MAX[ans] = max(MAX[ans], a[i]);
			out[i] = 2;
			continue;
		}
		int x = MAX[dp[i] + 1];
		if(x == -1 || x <= a[i])
			out[i] = 1;
		else
			out[i] = 2;
		if(out[i] == 2)
			MAX[dp[i]] = max(MAX[dp[i]], a[i]);
	}
	for(int i = 0; i < n; i++)
		if(out[i] == 2)
			cnt[dp[i]]++;
	for(int i = 0; i < n; i++)
		if(out[i] == 2 && cnt[dp[i]] == 1)
			cout << 3;
		else
			cout << out[i];
	cout << endl;
	return 0;
}
