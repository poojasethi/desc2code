#include<cstdio>
#include<iostream>
#include<cstring>
#include<string>
#include<algorithm>
#include<map>
#include<set>
#include<vector>
#include<queue>
#include<cstdlib>
#include<ctime>
#include<cmath>
using namespace std;
typedef long long LL;
#define N 100010

int n;
int a[N], lis[N], st[N], top, in[N], big[N], num[N];

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
		scanf("%d", &a[i]);
	top = 1;
	for (int i = 1; i <= n; i++) {
		if (st[top - 1] < a[i]) {
			st[top] = a[i];
			lis[i] = top;
			top++;
		} else {
			int pos = lower_bound(st + 1, st + top, a[i]) - st;
			st[pos] = a[i];
			lis[i] = pos;
		}
	}
	top--;
	for (int i = n; i >= 1; i--) {
		if (lis[i] == top || big[lis[i] + 1] > a[i]) {
			in[i] = 1;
			num[lis[i]]++;
			big[lis[i]] = max(big[lis[i]], a[i]);
		}
	}
	for (int i = 1; i <= n; i++) {
		if (in[i]) {
			if (num[lis[i]] > 1)
				putchar('2');
			else
				putchar('3');
		} else
			putchar('1');
	}
	return 0;
}
