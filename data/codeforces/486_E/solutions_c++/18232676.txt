#include<cstdio>
#include<algorithm>
using namespace std;
const int N=1e5+5,LEN=1e5+5;
int num[N],tmp[N],dp[N],mx[LEN]={0},ok[LEN]={0},ans[N];
inline void Max(int &a,int b){
	if(b>a)a=b;
}
int main(){
	int n,lis=1;
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
		scanf("%d",&num[i]);
	tmp[0]=num[1];
	dp[1]=1;
	for(int i=2;i<=n;++i){
		int k=lower_bound(tmp,tmp+lis,num[i])-tmp;
		tmp[k]=num[i];
		if(k==lis)++lis;
		dp[i]=k+1;
	}
	for(int i=n;i;--i)
		if(dp[i]==lis||mx[dp[i]+1]>num[i]){
			ans[i]=dp[i];
			++ok[dp[i]];
			Max(mx[dp[i]],num[i]);
		}
		else ans[i]=0;
	for(int i=1;i<=n;++i)
		if(!ans[i])putchar('1');
		else if(ok[ans[i]]==1)putchar('3');
		else putchar('2');
	puts("");
	return 0;
}
