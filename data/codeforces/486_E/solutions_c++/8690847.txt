#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;
const int MaxN = 100000 + 5, INF = 0x3f3f3f3f;
int n,a[MaxN],g[MaxN],f[MaxN],h[MaxN],tot[MaxN],MaxLen;bool no[MaxN];
int erf(int x)
{
	int st=0,en=n+1,mid;
	while(st+1<en)
	{
		mid=(st+en)>>1;
		if(g[mid]>=x) en=mid;
		else st=mid;
	}
	return en;
}
void solve()
{
	for(int i=1;i<=n;i++) g[i]=INF;
	int uu;
	for(int i=1;i<=n;i++)
	{
		uu=erf(a[i]);
		g[uu]=a[i];
		f[i]=uu;
		MaxLen=max(MaxLen,uu);
	}
	for(int i=n;i>=1;i--)
		if(f[i]==MaxLen||h[f[i]+1]>a[i])
		{
			h[f[i]]=max(h[f[i]],a[i]);
			tot[f[i]]++;
		}
		else no[i]=1;
	for(int i=1;i<=n;i++)
		if(no[i]) printf("1");
		else if(tot[f[i]]==1) printf("3");
		else printf("2");
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	solve();
	return 0;
}
