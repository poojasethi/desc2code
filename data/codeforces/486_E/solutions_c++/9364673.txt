#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[200000],d[2][200000];
int s[200000],check[200000],sz,lis;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int i=0;i<n;i++){
        int p=lower_bound(s,s+sz,a[i])-s;
        s[p]=a[i];
        d[0][i]=p+1;
        if(p==sz)sz++;
        a[i]=-a[i];
        lis=max(lis,d[0][i]);
    }
    sz=0;
    for(int i=n-1;i>=0;i--){
        int p=lower_bound(s,s+sz,a[i])-s;
        s[p]=a[i];
        d[1][i]=p+1;
        if(p==sz)sz++;
        if(d[0][i]+d[1][i]==lis+1)check[d[0][i]]++;
    }
    for(int i=0;i<n;i++){
        if(d[0][i]+d[1][i]!=lis+1)printf("1");
        else if(check[d[0][i]]>1)printf("2");
        else printf("3");
    }
}
