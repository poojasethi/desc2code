#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

const int maxn = 100000 + 5;

int a[maxn], s[maxn], f[maxn], cnt[maxn], tail[maxn], ans[maxn];

int main() {
    int n;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        scanf("%d", &a[i]);
    }
    int an = 0;
    for (int i = 0; i < n; i++) {
        int l = lower_bound(s, s + an, a[i]) - s;
        if (l == an) an++;
        s[l] = a[i];
        f[i] = l;
        ans[i] = 1;
    }
    for (int i = n - 1; i >= 0; i--)
        if (f[i] + 1 == an || a[i] < tail[f[i] + 1]) {
            ans[i] = 3;
            tail[f[i]] = a[i];
            cnt[f[i]]++;
        }
    for (int i = 0; i < n; i++) {
        if (ans[i] == 3 && cnt[f[i]] > 1) ans[i] = 2;
        printf("%d", ans[i]);
    }
    return 0;
}
