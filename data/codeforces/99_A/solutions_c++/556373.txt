#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main()
{
	string str;
	cin >> str;

	int p = str.find('.');
	if (str[p-1] == '9')
		cout << "GOTO Vasilisa." << endl;
	else
	{
		if (str[p+1]-'0' >= 5)
			str[p-1]++;
		cout << string(str.begin(), str.begin()+p) << endl;
	}
}
