#include <cstdio>
using namespace std;
int main() {
  char str[3000];
  int i;
  scanf("%s", str);
  for (i = 0; str[i] != '.'; i++);
  if (str[i - 1] == '9')
    printf("GOTO Vasilisa.\n");
  else {
    str[i] = 0;
    if (str[i + 1] >= '5')
      str[i - 1]++;
    printf("%s\n", str);
  }
  return 0;
}
