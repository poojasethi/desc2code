#include<cstdio>
#include<string.h>
#include<algorithm>
using namespace std;
int n,m;
int abs(int a){return a>0?a:-a;}
struct e
{
    int x,y;
}nodes[505];
int len[505];
int used[505];
int main()
{
    scanf("%d%d",&n,&m);
    int i,j,l;
    for(i=1;i<=n;i++)
    {
        scanf("%d%d",&nodes[i].x,&nodes[i].y);
    }
    nodes[0]=nodes[n];
    nodes[n+1]=nodes[1];
    for(i=1;i<=n;i++)
    {
        len[i]=abs(nodes[i-1].x-nodes[i].x)
        +abs(nodes[i-1].y-nodes[i].y)
        +abs(nodes[i+1].x-nodes[i].x)
        +abs(nodes[i+1].y-nodes[i].y);
    }
    for(i=1;i<=m;i++)
    {
        scanf("%d",&l);
        for(j=1;j<=n;j+=2)
        {
            if(used[j])continue;
            if(len[j]!=l)continue;
            used[j]=i;
            break;
        }
        for(j=2;j<=n;j+=2)
        {
            if(used[j])continue;
            if(len[j]!=l)continue;
            used[j]=i;
            break;
        }
    }
    for(i=1;i<=n;i+=2)
    {
        if(!used[i])break;
    }
    if(i>n)
    {
        puts("YES");
        for(i=1;i<=n;i+=2)
        printf("%d -1 ",used[i]);
        return 0;
    }
    ///////////////////
    for(i=2;i<=n;i+=2)
    {
        if(!used[i])break;
    }
    if(i>n)
    {
        puts("YES");
        for(i=2;i<=n;i+=2)
        printf("-1 %d ",used[i]);
        return 0;
    }
    puts("NO");
    return 0;
}
