#include <iostream>
#include <cstdio>
#include <vector>
#include <cstring>
using namespace std;

#define FOR(i, n) for(int i = 0; i < (n); i++)
#define FORB(i, a, b) for(int i = a; i < (b); i++)
#define PB push_back
#define SZ(A) ((int)(A).size())

const int MAXn = 500 +10;

int coor[MAXn][2];
int dis[MAXn];
int rod[MAXn];
bool mark[MAXn];
int ans[MAXn];
int n, m;

inline int abs(int a){
	if ( a < 0 )	return -a;
	return a;
}


void input(){
	cin >> n >> m;
	FOR(i, n)
		cin >> coor[i][0] >> coor[i][1];
	FOR(i, m)
		cin >> rod[i];
	FOR(i, n)
	dis[i] = abs(coor[i][0] - coor[(i+1)%n][0]) + abs(coor[i][1] - coor[(i+1)%n][1]);
}
bool find(int a){
	int nxt = (a+1)%n;
	FOR(i, m)
		if (rod[i] == (dis[a] + dis[nxt]) && !mark[i]){
			mark[i] = true;
			ans[nxt] = i;
			return true;
		}
	return false;
}
bool good(int start){
	memset(mark, 0, sizeof(mark));
	memset(ans, -1, sizeof(ans));
	FOR(i, n/2){
		int now = start + 2*i;
		if (find(now) == false)
			return false;
	}
	return true;
}
void output(){
	cout << "YES" << endl;
	FOR(i, n){
		if (ans[i] == -1)
			cout << -1 << " ";
		else
			cout << ans[i]+1 << " ";
	}
	cout << endl;
}

int main(){
	input();
	if (good(0))
		output();
	else{
		if(good(1))
			output();
		else
			cout << "NO" << endl;
	}
	return 0;
}
