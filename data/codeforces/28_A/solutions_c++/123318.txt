#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <utility>
using namespace std;
#define rep(i, n) for(int i=0; i<(int)(n); i++)

int n, m;
int x[500], y[500], d[500], r[500];
int u[500];
vector<pair<int, int> > rr;

int can(int o) {
    rep(i, n) u[i] = -1;
    vector<pair<int, int> > v;
    rep(i, n/2) {
        int cur=(2*i+o)%n;
        v.push_back(make_pair(d[cur]+d[(cur+n-1)%n], cur));
    }
    sort(v.begin(), v.end());
    int k=0;
    rep(i, v.size()) {
        while(k<m && v[i].first>rr[k].first) k++;
        if(k==m || v[i].first<rr[k].first) return false;
        u[v[i].second] = rr[k].second;
        k++;
    }
    return true;
}

int main() {
    cin >> n >> m;
    rep(i, n) cin >> x[i] >> y[i];
    rep(i, n) d[i] = abs(x[i]-x[(i+1)%n])+abs(y[i]-y[(i+1)%n]);
    rep(i, m) cin >> r[i];
    rep(i, m) rr.push_back(make_pair(r[i], i+1));
    sort(rr.begin(), rr.end());
    if(can(0) || can(1)) {
        cout << "YES" << endl;
        rep(i, n-1) cout << u[i] << " ";
        cout << u[n-1] << endl;
    }
    else {
        cout << "NO" << endl;
    }
    return 0;
}
