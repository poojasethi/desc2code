#include<iostream>

using namespace std;

struct point
{
    int x, y;
};

int abs(int a)
{
    return a < 0 ? -a : a;
}

int main()
{
    int n, m;

    cin >> n >> m;

    point g[n];
    int l[m], ans[n];
    bool lu[m], lg[n];

    for(int i = 0; i < m; ++i)
        lu[i] = false;

    for(int i = 0; i < n; ++i)
    {
        lg[i] = false;
        ans[i] = -1;
    }

    for(int i = 0; i < n; ++i)
        cin >> g[i].x >> g[i].y;

    for(int i = 0; i < m; ++i)
    	cin >> l[i];

    for(int i = 0; i < n; ++i)
    {
        if(lg[i]) continue;

        for(int j = 0; j < m; ++j)
        {
            if(!lu[j] && ((abs(g[(i+1)%n].x - g[i].x) + abs(g[(i-1+n)%n].y - g[i].y) == l[j]) ||  (abs(g[(i-1+n)%n].x - g[i].x) + abs(g[(i+1)%n].y - g[i].y) == l[j])))
            {
                lu[j] = true;
                lg[i] = true;
                lg[(i+1)%n] = true;
                lg[(i-1+n)%n] = true;
                ans[i] = j+1;
                break;
            }
        }
    }           

    bool can = true;

    for(int i = 0; i < n; ++i)
        if(!lg[i]) can = false;

   //    for(int i = 0; i < n; ++i)
   //        cout << ans[i] << ' ';


    if(!can)
       cout << "NO";
    else
    {
       cout << "YES" << endl;
       for(int i = 0; i < n; ++i)
           cout << ans[i] << ' ';
    }

    return 0;
}
