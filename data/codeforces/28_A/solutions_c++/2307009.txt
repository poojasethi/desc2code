#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <iostream>
#include <vector>

const int N=200010;

using namespace std;

#define pb push_back
#define sz(x) ((x).size())

int n,m,c[N],d[N],x[N],y[N],vis[N],l[N];;

int dis(int x1,int y1,int x2,int y2)
{
	return abs(x1-x2)+abs(y1-y2);
}

bool solve(int n,int s,vector<int> &ret)
{
	memset(vis,0,sizeof(vis));
	memset(d,0,sizeof(d));
	for(int i=s;i<n+s;i+=2)
	{
		int a=i%n,b=(i+1)%n,C=(i+2)%n;
		int k=dis(x[a],y[a],x[b],y[b])+dis(x[b],y[b],x[C],y[C]);
		if(c[k]-d[k]) d[k]++;
		else return 0;
		for(int j=0;j<m;j++)
		{
			if(vis[j]) continue;
			if(l[j]-k) continue;
			vis[j]=1;
			ret.pb(-1);
			ret.pb(j+1);
			break;
		}
	}
	return 1;
}

int main()
{
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%d%d",x+i,y+i);
	for(int i=0;i<m;i++)
	{
		scanf("%d",l+i);
		c[l[i]]++;
	}
	vector<int> ret;
	if(solve(n,0,ret))
	{
		cout<<"YES\n";
		for(int i=0;i<sz(ret);i++)
			cout<<ret[i]<<" ";cout<<endl;
		return 0;
	}
	ret.clear();
	if(solve(n,1,ret))
	{
		cout<<"YES\n";
		cout<<ret[sz(ret)-1]<<" ";
		for(int i=0;i<sz(ret)-1;i++)
			cout<<ret[i]<<" ";cout<<endl;
		return 0;
	}
	cout<<"NO\n";
	return 0;
}
