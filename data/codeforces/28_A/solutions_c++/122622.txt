#include <iostream>
#include <cstdio>
#include <cmath>
#include <vector>
#include <set>
#include <map>
#include <iomanip>
#include <algorithm>
#include <string>

using namespace std;

const int MAXN=500+50;
int n,m;
int a[MAXN],b[MAXN];
int c[MAXN];
int ans[MAXN];
bool mark[MAXN];

bool hal(int x){
	for(int i=x;i<n;i+=2){
		int q=abs(a[(i+1)%n]-a[i])+abs(b[(i+1)%n]-b[i])+abs(a[(i-1+n)%n]-a[i])+abs(b[(i-1+n)%n]-b[i]);
		bool z=false;
//		cout << q << endl;
		for(int j=1;j<=m;j++)
			if(!mark[j] && c[j]==q){
				z=true;
				mark[j]=true;
				ans[i]=j;
				break;
			}
		if(!z)
			return false;
	}
//	cout << "Salam " << endl;
	ans[(x-1+n)%n]=-1;
	for(int i=(x+1)%n;i!=(x-1+n)%n;i=(i+2)%n)
		ans[i]=-1;
	return true;
}

int main(){
	cin >> n >> m;
	for(int i=0;i<n;i++)
		cin >> a[i] >> b[i];
	for(int i=1;i<=m;i++)
		cin >> c[i];
	if(n%2==1){
		cout << "NO" << endl;
		return 0;
	}
	if(!hal(0)){
		for(int i=0;i<=m;i++)
			mark[i]=false;
		if(!hal(1)){
			cout<< "NO" << endl;
			return 0;
		}

	}
	cout << "YES" <<endl;
	for(int i=0;i<n;i++)
		cout << ans[i] << " ";
	cout << endl;
}
