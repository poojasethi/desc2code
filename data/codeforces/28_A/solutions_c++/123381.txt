using namespace std;

const int INF=(1<<30)-1;
const double EPS=1e-9, PI=(2*acos(0.0));
typedef long long LL;

#define sz 505
int x[sz],y[sz],r[sz],f[sz];

int main()
{
	//freopen("A.txt","r",stdin);
	int n,m,i,j,k,l,xx,yy;
	cin>>n>>m;
	for(i=0;i<n;i++) cin>>x[i]>>y[i];
	
	for(l=1;l<=m;l++) cin>>r[l];
	
	memset(f,255,sizeof(f));
	for(i=0;i<n;i+=2) {
		j=(i-1+n)%n;
		k=(i+1+n)%n;
		int d=abs(x[i]-x[j])+abs(x[i]-x[k])+abs(y[i]-y[j])+abs(y[i]-y[k]);
		
		for(l=1;l<=m;l++)
			if(d==r[l]) {
				f[i]=l;
				r[l]=-1;
				break;
			}
		if(l>m) break;
	}
	if(i>=n) {
		cout<<"YES"<<endl;
		for(i=0;i<n;i++) cout<<f[i]<<' ';
		return 0;
	}
	
	memset(f,255,sizeof(f));
	for(i=1;i<n;i+=2) {
		j=(i-1+n)%n;
		k=(i+1+n)%n;
		int d=abs(x[i]-x[j])+abs(x[i]-x[k])+abs(y[i]-y[j])+abs(y[i]-y[k]);
		
		for(l=1;l<=m;l++)
			if(d==r[l]) {
				f[i]=l;
				r[l]=-1;
				break;
			}
		if(l>m) break;
	}
	if(i>=n) {
		cout<<"YES"<<endl;
		for(i=0;i<n;i++) cout<<f[i]<<' ';
		return 0;
	}
		
	cout<<"NO";
	return 0;
}
