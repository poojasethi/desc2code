#include<iostream>
#include<vector>
using namespace std;
const int MN=210000;
vector<int> a[MN],b[MN];
int pt[MN][2],v[MN],n,m;
int abs(int x){

	if(x<0){
		return x*-1;
	}return x;
}
int dis(int i,int v){
	int j=i+v;
	j+=n;
	j%=n;
	if(pt[i][0]==pt[j][0]){
		return abs(pt[i][1]-pt[j][1]);
	}else{
		return abs(pt[j][0]-pt[i][0]);
	}
}

int main(){
	cin>>n>>m;
	for(int i=0;i<n;i++){
		int x,y;
		cin>>x>>y;
		pt[i][0]=x;
		pt[i][1]=y;
	}
	for(int i=0;i<m;i++){
		int lf;
		cin>>lf;
		a[lf].push_back(i+1);
	}
	for(int i=0;i<MN;i++){
		b[i]=a[i];
	}
	bool t=1;
	for(int i=1;i<n;i+=2){
		int ds=dis(i,-1)+dis(i,1);
		if(b[ds].size()==0){
			t=0;
			break;
		}else{
			v[i]=b[ds][b[ds].size()-1];
			b[ds].pop_back();
		}
	}
	if(t){
		cout<<"YES"<<endl;
		for(int i=0;i<n;i++){
			if(i%2==0){
				cout<<"-1"<<" ";
			}else{
				cout<<v[i]<<" ";
			}
		}cout<<endl;
		return 0;
	}
	t=1;
	for(int i=0;i<MN;i++){
		b[i]=a[i];
	}
	for(int i=0;i<n;i+=2){
		int ds=dis(i,-1)+dis(i,1);
		if(b[ds].size()==0){
			t=0;
			break;
		}else{
			v[i]=b[ds][b[ds].size()-1];
			b[ds].pop_back();
		}
	}
	if(t){
		cout<<"YES"<<endl;
		for(int i=0;i<n;i++){
			if(i%2==1){
				cout<<"-1"<<" ";
			}else{
				cout<<v[i]<<" ";
			}
		}cout<<endl;
		return 0;
	}
	cout<<"NO"<<endl;
}
