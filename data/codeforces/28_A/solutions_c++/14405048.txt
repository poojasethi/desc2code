#include <cmath>
#include <iostream>
using namespace std;

const int MAXN = 500 + 10;
int d[MAXN], ans[MAXN], p[MAXN];
pair <int, int> xy[MAXN];
bool chck[MAXN];

int main()
{
	int n, m;
	cin>> n >> m;

	for(int i = 0; i < n; i++)
		cin>> xy[i].first >> xy[i].second;

	for(int i = 1; i < n; i++)
	{
		if(xy[i].first == xy[i - 1].first)
			d[i] += abs(xy[i - 1].second - xy[i].second);
		else
			d[i] += abs(xy[i - 1].first - xy[i].first);
	}
	for(int i = 0; i < n - 1; i++)
	{
		if(xy[i].first == xy[i + 1].first)
			d[i] += abs(xy[i + 1].second - xy[i].second);
		else
			d[i] += abs(xy[i + 1].first - xy[i].first);
	}
	int a;
	if(xy[0].first == xy[n - 1].first)
		a = abs(xy[n - 1].second - xy[0].second);
	else
		a = abs(xy[n - 1].first - xy[0].first);

	d[0] += a;
	d[n - 1] += a;
	for(int i = 0; i < m; i++)
		cin>> p[i];


	for(int j = 0; j < 2; j++)
	{
		//cerr<< j << ": ";

		for(int i = 0; i < n; i ++)
			ans[i] = -1;
		
		for(int i = 0; i < m; i ++)
			chck[i] = false;

		bool ch = true;
		for(int i = 0 + j; i < n; i += 2)
		{
			for(int k = 0; k < m; k ++)
				if(d[i] == p[k] && chck[k] == false)
				{
					chck[k] = true;
					ans[i] = k + 1;
					break;
				}

			if(ans[i] == -1)
			{
				ch = false;
				break;
			}
		}
		
		if(ch)
		{
			cout<< "YES\n";
			for(int i = 0; i < n; i++)
				cout<< ans[i] << " ";
			cout<< endl;
			return 0;
		}
		//cerr<< endl;
	}

	cout<< "NO\n";
	return 0;


}
