#include <iostream>
#include <vector>
#include <map>
#include <algorithm>
using namespace std;

int dist(pair<int,int> &a, pair<int,int> &b)
{
  return abs(a.first-b.first)+abs(a.second-b.second);
}

int main()
{
  int n, m; cin>>n>>m;

  vector<pair<int,int> > cs;
  for (int i=0; i<n; i++){
    int x, y; cin>>x>>y;
    cs.push_back(make_pair(x, y));
  }

  vector<int> ls(m);
  for (int i=0; i<m; i++) cin>>ls[i];

  for (int ofs=0; ofs<2; ofs++){
    multimap<int, int> mm;
    for (int i=0; i<m; i++) mm.insert(make_pair(ls[i], i+1));

    vector<int> sol(n, -1);

    for (int i=0; i<n; i+=2){
      int len =
	dist(cs[(i+ofs-1+n)%n], cs[(i+ofs-0+n)%n]) +
	dist(cs[(i+ofs-0+n)%n], cs[(i+ofs+1+n)%n]);
      if (mm.count(len)==0) goto _next;
      multimap<int, int>::iterator p=mm.find(len);
      sol[i+ofs]=p->second;
      mm.erase(p);
    }

    cout<<"YES"<<endl;
    for (int i=0; i<n; i++){
      if (i>0) cout<<" ";
      cout<<sol[i];
    }
    cout<<endl;
    return 0;

  _next:;
  }

  cout<<"NO"<<endl;

  return 0;
}
