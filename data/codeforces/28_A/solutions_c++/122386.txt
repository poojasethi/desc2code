#include <cstdlib>
#include <cstdio>
#include <cstring>
#include <climits>
#include <limits>
#include <iostream>
#include <algorithm>
#include <vector>
#include <map>
#include <set>
#include <list>
#include <cmath>

using namespace std;

int n, m;
int x[500], y[500];
int used[500] = {0};
int l[500];
int conn[500];

bool solve(int i, int start)
{
    int i1 = (i + 1) % n;
    if (i1 == start) return false;
    int i2 = (i + 2) % n;
    
    int len = abs(x[i] - x[i1]) + abs(y[i] - y[i1]) + 
            abs(x[i1] - x[i2]) + abs(y[i1] - y[i2]);
    
    for (int j = 0; j < m; j++)
        if (!used[j] && l[j] == len)
        {
            used[j] = true;
            bool ok = (i2 == start) || solve(i2, start);
            if (ok) 
            {
                conn[i] = conn[i2] = -1;
                conn[i1] = j + 1;
            }
            used[j] = false;
            return ok;
        }
    
    return false;
}

int main(void)
{
#ifdef LOCAL_DEBUG
    freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
#endif
    
    cin >> n >> m;
    for (int i = 0; i < n; i++) cin >> x[i] >> y[i];
    for (int i = 0; i < m; i++) cin >> l[i];
    
    if (!solve(0, 0) && !solve(1, 1)) cout << "NO";
    else
    {
        cout << "YES" << endl;
        for (int i = 0; i < n; i++) cout << conn[i] << ' ';
    }
    
    return 0;
}
