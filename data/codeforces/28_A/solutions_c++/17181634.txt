#include <cstdio>
#include <cstring>
#include <cmath>
#include <cctype>
#include <iostream>
#include <cstdlib>
using namespace std;
int ret[510], x[510], y[510], c[200100], d[200100], l[510], vis[510];
int n, m;

int dis(int a, int b)
{
    return abs(x[a] - x[b]) + abs(y[a] - y[b]);
}
bool solve(int s)
{
	int i, j;
	for (i = s; i < n; i += 2)
	{
		int a = (i + n - 1) % n, b = (i + 1) % n;
		int len = dis(a, i) + dis(i, b);
		if (d[len] == c[len]) return 0;
		d[len]++;
		for (j = 0; j < m; ++ j)
		{
			if (vis[j]) continue;
			if (l[j] != len) continue;
			ret[i] = j + 1;
			vis[j] = true;
			break;
		}
	}
	return true;
}

int main()
{
    int i, j;
	scanf("%d%d", &n, &m);
	for (i = 0; i < n; ++ i)
	{
		scanf("%d%d", &x[i], &y[i]);
	}
	for (i = 0; i < m; ++ i)
	{
		scanf("%d", &l[i]);
		c[l[i]]++;
	}
	memset(ret, -1, sizeof(ret));
	if (solve(0))
	{
		puts("YES");
		for (i = 0; i < n; ++ i)  printf("%d ", ret[i]);
		printf("\n");
		return 0;
	}
	memset(ret, -1, sizeof(ret));
	memset(d, 0, sizeof(d));
	if (solve(1))
	{
		puts("YES");
		for (i = 0; i < n; ++ i)  printf("%d ", ret[i]);
		printf("\n");
		return 0;
	}
	puts("NO");
	return 0;
}
