#include <iostream>
#include <cstdlib>
#include <cstring>
#include <stdio.h>

using namespace std;

int n,len[555],was[555],m,flag,t,b[555];
struct tp {
	int x,y;
}a[555];


void print()
{
	puts("YES");
	for ( int i = 1; i <= n; i++ )
		printf("%d ",(was[i]) ? was[i] : -1);
}
int main()
{
	scanf("%d%d",&n,&m);
	for ( int i = 1; i <= n; i++ ) 
		scanf("%d%d",&a[i].x,&a[i].y);
	for ( int i = 1; i <= m; i++ )
		scanf("%d",&len[i]);
	
	a[0].x = a[n].x,a[0].y = a[n].y,a[n+1].x = a[1].x,a[n+1].y = a[1].y;

	for ( int i = 1; i <= n; i+=2 ) {
		flag = 0;
		for ( int j = 1; j <= m; j++ ) {
			if (  !b[j] && len[j] == (abs(a[i-1].x - a[i].x)+abs(a[i-1].y - a[i].y))+((abs(a[i+1].x - a[i].x)+(abs(a[i+1].y-a[i].y)))) )  { flag=1,was[i] = j; b[j] = 1; break; }
		}
	if ( flag == 0 ) { t = 1; break;  }
	}
	if ( !t ) print();
	else {
		memset(was,0,sizeof(was));
		memset(b,0,sizeof(b));
		for ( int i = 2; i <= n; i+=2 ) {
			flag = 0;
			for ( int j = 1; j <= m; j++ ) {
			if ( !b[j] && len[j] == (abs(a[i-1].x - a[i].x)+abs(a[i-1].y - a[i].y))+((abs(a[i+1].x - a[i].x)+(abs(a[i+1].y-a[i].y)))) )  { flag=1,was[i] = j; b[j] = 1; break; }
			}
		if ( !flag ) { puts("NO"); return 0; }
		}
	print();
	}
}
