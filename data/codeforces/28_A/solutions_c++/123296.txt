#include <iostream>
#include <vector>
#include <map>
#include <utility>
#include <algorithm>
#include <string>
#include <set>
#include <stdio.h>
#include <sstream>
#include <math.h>

using namespace std;

int main() {
    cout << fixed;
    cout.precision(9);

    int n, m;
    cin >> n >> m;
    vector<int> x(n), y(n);
    for (int i = 0; i < n; i++)
        cin >> x[i] >> y[i];
    vector<int> p(m);
    for (int i = 0; i < m; i++)
        cin >> p[i];
    vector<int> ans(m, -1);
    
    for (int start = 0; start <= 1; start++) {
        bool ok = true;
        ans.assign(m, -2);
        for (int i = start, k = 0; k < n/2; i = (i+2)%n, k++) {
            int prev = (i+n-1) % n, next = (i+n+1)%n;
            int len = abs(x[i] - x[prev]) + abs(y[i] -y[prev]) + abs(x[i] - x[next]) + abs(y[i] -y[next]);
            int j  =0;
            for (; j < m; j++) if (ans[j] < 0 && p[j] == len) {
                ans[j] = i;
                break;
            }
            if (j == m) {
                ok = false;
                break;
            }
        }
        if (ok) {
            cout << "YES\n";
            vector<int> res(n, -1);
            for (int i = 0; i < m; i++) if (ans[i] >= 0) {
                res[ans[i]] = i+1;
            }
            for (int i = 0; i < n; i++) {
                cout << res[i] << " ";
            }
            return 0;
        }
    }
    cout << "NO\n";
    return 0;
}
