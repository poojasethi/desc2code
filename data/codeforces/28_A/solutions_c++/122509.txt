#include <iostream>
#include <vector>
#include <cmath>
#include <cstdio>
using namespace std;
vector<pair<int,int> > g;
vector<int> p;
vector<bool> used;
int n,m;
void inc(int &a)
{
	a++;
	if (a==n)
		a=0;
}
int scan(int s)
{
	for (int i=0;i<m;i++)
	{
		if (!used[i]&&p[i]==s)
			return i;
	}
	return -1;
}
int main()
{
//	freopen("in.in","r",stdin);
	cin>>n>>m;
	g.resize(n);
	used.resize(m);
	p.resize(m);
	for (int i=0;i<n;i++)
	{
		int a,b;
		cin>>a>>b;
		g[i]=make_pair(a,b);
		
	}
	for (int i=0;i<m;i++)
	{
		used[i]=0;
		cin>>p[i];
	}
	vector<int> par(n,-1);
for (int ss=0;ss<n;ss++)
{
	int f1=ss;
	bool ok=1;
	for (int i=0;i<n;i++)
	{
		int f2=f1;
		inc(f2);
		int f3=f2;
		inc(f3);
		int rasst=sqrt((g[f1].first-g[f2].first)*(g[f1].first-g[f2].first) +(g[f1].second-g[f2].second)*(g[f1].second-g[f2].second)+.0);
		rasst+=sqrt((g[f2].first-g[f3].first)*(g[f2].first-g[f3].first)+(g[f2].second-g[f3].second)*(g[f2].second-g[f3].second)+.0);
		int l=scan(rasst);
		if (l==-1)
		{
			ok=0;
			break;
		}
		else
		{
			used[l]=1;
			par[f2]=l+1;
			inc(f1);
			inc(f1);
			i++;
		}
	}
	if (ok)
	{
		printf("YES\n");
		for (int i=0;i<n;i++)
			printf("%d ",par[i]);
		return 0;
	}
	for (int i=0;i<n;i++)
		par[i]=-1;
	
		for (int i=0;i<m;i++)
			used[i]=0;
}
	printf("NO");

	


	return 0;
}
