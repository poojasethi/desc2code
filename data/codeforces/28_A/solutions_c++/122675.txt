#include<stdio.h>
#include<memory.h>
#include<stdlib.h>
#define N 222222

int n,m,x[N],y[N],d[N],w[N],r[N],ok=0,was[N],ans[N];


int main(void){
//	freopen("in","r",stdin);
//	freopen("out","w",stdout);
	scanf("%d%d",&n,&m);
	for(int i=1;i<=n;i++) scanf("%d%d",&x[i],&y[i]);
	x[0]=x[n]; y[0]=y[n];
	x[n+1]=x[1]; y[n+1]=y[1];

	for(int i=0;i<m;i++){
		int xx=0;
		scanf("%d",&xx);
		d[xx]++;
		r[i]=xx;
	}
	
	memcpy(w,d,sizeof(d));
	memset(ans,-1,sizeof(ans));

	for(int it=0;it<2;it++){
		memcpy(d,w,sizeof(w));
		memset(was,0,sizeof(was));
		ok=1;
		for(int i=it+1;i<=n;i+=2){
			int dd=abs(x[i]-x[i-1]) + abs(y[i]-y[i-1]) + abs(y[i]-y[i+1]) + abs(x[i]-x[i+1]);
			if(d[dd]==0){
				ok=0;
				break;
			}
			d[dd]--;
		}
		if(ok){
			for(int i=it+1;i<=n;i+=2){
				int dd=abs(x[i]-x[i-1]) + abs(y[i]-y[i-1]) + abs(y[i]-y[i+1]) + abs(x[i]-x[i+1]);
				for(int j=0;j<m;j++) if(!was[j] && dd==r[j]){
					ans[i]=j+1;
					was[j]=1;
					break;
				}
			}
			puts("YES");
			for(int i=1;i<n;i++) printf("%d ",ans[i]);
			printf("%d\n",ans[n]);
			return 0;
		}
	}
	puts("NO");
	return 0;
}
