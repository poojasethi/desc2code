#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;
#define MAXN 550
int n,m,x[MAXN],y[MAXN],len[MAXN],idx[MAXN];
bool vst[MAXN];
bool sol(int s,int e)
{
    memset(vst,0,sizeof(vst));
    memset(idx,-1,sizeof(idx));
    for(int i=s;i<e;i+=2)
    {
        int j=i-1;
        if(j<s) j=e-1;
        int tmp=abs(x[j]-x[i])+abs(y[i]-y[j])
                +abs(x[i+1]-x[i])+abs(y[i+1]-y[i]);
        for(int j=1;j<=m;j++)
        {
            if(!vst[j]&&len[j]==tmp)
            {
                vst[j]=true;
                idx[i%n]=j;
                break;
            }
            if(j==m) return false;
        }
    }
    return true;
}
int main()
{
    while(scanf("%d%d",&n,&m)==2)
    {
        for(int i=0;i<n;i++) scanf("%d%d",x+i,y+i);
        for(int i=1;i<=m;i++) scanf("%d",len+i);
        x[n]=x[0],y[n]=y[0];
        x[n+1]=x[1],y[n+1]=y[1];
        if(sol(0,n)||sol(1,n+1))
        {
            printf("YES\n");
            for(int i=0;i<n;i++)
            {
                printf("%d",idx[i]);
                if(i<n-1) printf(" ");
                else printf("\n");
            }
        }
        else printf("NO\n");
    }
    return 0;
}
