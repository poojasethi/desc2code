#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>
const int N = 500 + 10;
typedef std::pair<int, int> pii;

pii a[N];
int len[N], op[N];
bool vis[N];

int dis(int x, int y) {
	return abs(a[x].first - a[y].first) + abs(a[x].second - a[y].second);
}

int main() {
	int n, m, f;
	scanf("%d%d",&n, &m);
	
	for ( int i = 1; i <= n; ++i) 
		scanf("%d%d", &a[i].first, &a[i].second);
	for ( int i = 1; i <= m; ++i)
		scanf("%d", &len[i]);
	
	a[0] = a[n];
	a[n + 1] = a[1];
	f = true;
	
	for (int i = 1; i <= n; i += 2) {
		bool done = false;
		for (int j = 1; j <= m; ++j) 
			if (!vis[j] && len[j] == dis(i - 1, i) + dis(i + 1, i)) { 
				done = 1;
				op[i] = j; 
				vis[j] = true; 
				break; 
			}
		
		if (!done) { 
			f = false; 
			break;  
		}
	}
	if (!f) {
		f = true;
		memset(op, 0, sizeof op);
		memset(vis, 0, sizeof vis);
		for ( int i = 2; i <= n; i+=2) {
			bool done = false;
			for (int j = 1; j <= m; ++j) 
				if (!vis[j] && len[j] == dis(i - 1, i) + dis(i + 1, i)) { 
					done = 1;
					op[i] = j; 
					vis[j] = true; 
					break; 
				}
			
			if (!done) { 
				f = false; 
				break;  
			}
		}
	}
	if (!f)
		puts("NO");
	else {
		puts("YES");
		for ( int i = 1; i <= n; i++ )
			printf("%d ",(op[i]) ? op[i] : -1);
	}
	return 0;
}
