#include <iostream>
#include <string.h>
using namespace std;

bool flag[1000];
int a[20][20];
int find(int k)
{
	for (int i=k;;i++)
		if (flag[i]==false) return i;
}

bool check(int j,int k)
{
	for (int i=0;i<j;i++)
		if (flag[a[i][0]+k]==true)
			return false;
	return true;
}	

int main()
{
	memset(flag,0,sizeof(flag));
	int n;
	cin >> n;
	int p=0;
	for (int i=0;i<n;i++)
	{
		if (i!=0)
		p=0;
		while(!check(i,p))
			p=find(p+1);
		p++;
		for (int j=0;j<i;j++)
		{
			a[i][j]=a[j][i]=(a[j][0]-a[j-1][0]+p);
			flag[a[i][j]-1]=true;
			p=a[i][j];
		}
		a[i][i]=0;
	}
	for (int i=0;i<n;i++)
	{
		for (int j=0;j<n;j++)
		cout << a[i][j] << " ";
		cout << "\n";
	}
	return 0;
}
