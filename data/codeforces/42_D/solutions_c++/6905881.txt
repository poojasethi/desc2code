#include <iostream>
using namespace std;

#define REP(i,n) for((i)=0;(i)<(int)(n);(i)++)

int x[] = {1,2,3,5,8,13,21,30,39,53,74,95,128,152,182,212,258,316,374,413};

int main(void){
    int N,i,j;
    cin >> N;
    REP(i,N) REP(j,N) cout << ((i == j) ? 0 : x[i] + x[j]) << ((j == N-1) ? "\n" : " ");
    return 0;
}
