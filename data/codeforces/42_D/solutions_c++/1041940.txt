#include <iostream>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

int a[30],b[1100],n,i,j;

int main(){
        n=1;
        REP(i,1,1001){
                if (n==20) break;
                REP(j,0,n) if (b[i+a[j]]) break;
                if (j!=n) continue;
                REP(j,0,n) b[i+a[j]]=1;
                a[n++]=i;
        }
        //REP(i,0,20) cout<<a[i]<<' ';cout<<endl;
        cin>>n;
        REP(i,0,n)REP(j,0,n){
                if (i==j) cout<<0;else cout<<a[i]+a[j];
                if (j==n-1) cout<<endl;else cout<<' ';
        }
}
