#include <cstdio>
int a[21][21], i, j, o, w[9999], c, n;
int main ()
{
	scanf("%d", &n);
	a[1][2] = a[2][1] = 1;
	a[1][3] = a[3][1] = 2;
	a[2][3] = a[3][2] = 3;
	w[1] = w[2] = w[3] = 1;
	for (i = 4; i <= n; i++) {
		c = 1;
		for (;;) {
			o = 1;
			for (j = 1; j < i; j++) {
				if (w[a[1][j] + c]) {
					o = 0;
					break;
				}
			}
			if (o) break;
			c++;
		}
		for (j = 1; j < i; j++) {
			a[i][j] = a[1][j] + c;
			a[j][i] = a[i][j];
			w[a[i][j]] = 1;
		}
	}
	for (i = 1; i <= n; i++) {
		for (j = 1; j <= n; j++)
			printf("%d ", a[i][j]);
		printf("\n");
	}
}
