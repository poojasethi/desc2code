#include <cstdio>

bool flag[1111];
int f[23];

int main() {
	f[1] = 1, f[2] = 2, flag[1] = true;

	int n; scanf("%d", &n);
	for (int i = 3; i <= n; ++i) {
		for (int k = f[i - 1] + 1; ; ++k) {
			bool now = true;
			for (int j = 1; j < i; ++j)
				if (flag[k + f[j]]) now = false;

			if (now) {
				f[i] = k;
				for (int j = 1; j < i; ++j)
					flag[k + f[j]] = true;
				break;
			}
		}
	}

	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j < i; ++j) printf("%d ", f[i] + f[j]);
		printf("%d%c", 0, i == n ? '\n' : ' ');
		for (int j = i + 1; j <= n; ++j) printf("%d%c", f[j] + f[i], j == n ? '\n' : ' ');
	}

	return 0;
}
