#include <cstdio>
#include <algorithm>
using namespace std;


int main()
{
    int n;
    scanf("%d", &n);
    int a[23];
    a[0] = 0;
    a[1] = 1;
    a[2] = 2;
    for(int p = 3; p < n; ++p) {
        bool q[1003] = {};
        for(int i = 0; i < p; ++i)
            for(int j = 0; j < p; ++j)
                for(int k = j+1; k < p; ++k) {
                    int x = a[j]+a[k]-a[i];
                    if(x > 0) q[x] =  true;
                }
        
        int x = 1;
        while(q[x])
            ++x;
        a[p] = x;
    }
    
    for(int i = 0; i < n; ++i) {
        for(int j = 0; j < n; ++j) {
            if(j) printf(" ");
            printf("%d", i != j ? a[i]+a[j] : 0);
        }
        printf("\n");
    }
    return 0;
}
