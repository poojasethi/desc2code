#include<cstdio>  
#include<cstring>  
#include<cstdlib>  
#include<iostream>  
#include<algorithm>  
using namespace std;
#define fo(i,a,b) for(i=a;i<=b;++i)
const int mn=100;
int a[mn],n,i,j,k;
bool b[2100];
int main()
{
    scanf("%d",&n);
    k=0;
    fo(i,0,1000)
    {
        fo(j,1,k)if(b[i+a[j]])break;
        if(b[i+a[j]])continue;
        fo(j,1,k)b[i+a[j]]=1;
        a[++k]=i;
        if(k==n)break;
    }
    fo(i,1,n)
    {
        fo(j,1,n-1)printf("%d ",i==j?0:a[i]+a[j]);
        printf("%d\n",i==n?0:a[i]+a[n]);
    }
    return 0;
}
