#include<cstdio>
int a[21] = {1, 2, 3}, n;
int main()
{
	scanf("%d", &n);
	for(int i = 3; i < n; ++i)
		for(int j = a[i-1]+1; j < 1001; ++j)
		{			
			bool g = 1;
			for(int u = 0; u < i && g; ++u)
				for(int v = 0; v < i && g; ++v)
					for(int q = 0; q < i && g; ++q)
						if(a[u] + a[v] - a[q] == j)
							g = 0;
			if(g)
			{
				a[i] = j;
				break;
			}
		}	
	for(int i = 0; i < n; ++i, printf("\n"))
		for(int j = 0; j < n; ++j)
			printf("%d ",(i==j?0:a[i]+a[j]));
}
