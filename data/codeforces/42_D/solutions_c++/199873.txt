#include <iostream>

using namespace std;

#define REP(i,n) for((i)=0;(i)<(int)(n);(i)++)

int x[] = {51,438,137,20,250,498,181,423,489,404,339,282,177,220,272,362,255,410,491,157};

int main(void){
    int N,i,j;
    cin >> N;
    REP(i,N) REP(j,N) cout << ((i == j) ? 0 : x[i] + x[j]) << ((j == N-1) ? "\n" : " ");
    return 0;
}
