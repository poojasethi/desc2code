#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int ans[20],vis[1001];
int main()
{
	int n;
	while(~scanf("%d",&n))
	{
		if(n==3)
		{
			printf("0 3 4\n3 0 5\n4 5 0\n");
			continue;
		}
		memset(vis,0,sizeof(vis));
		int cnt=1;
		for(int i=0;i<n;i++)
		{
			while(vis[cnt]) cnt++;
			int mark=0;
			for(int j=0;j<i;j++)
			{
				int ret=cnt+ans[j];
				for(int p=0;p<i;p++)
				for(int q=p+1;q<i;q++)
				if(ans[p]+ans[q]==ret)
				mark=1;
			}
			if(!mark) 
			{	
				ans[i]=cnt;
				vis[cnt]=1;
			}
			else 
			{
				i--;
				cnt++;
			}
		}
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{
				if(i==j) printf("0 ");
				else printf("%d ",ans[i]+ans[j]);
			}
			printf("\n");
		}
	} 
	return 0; 
}
