#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int a[22],now,n;
bool u[1200];

int main()
{
//	freopen("42D.out","w",stdout);
	cin.sync_with_stdio(false);
    cin>>n;
    memset(u,false,sizeof(u));

    for (int i=1;i<n;i++)
    {
        for (int j=a[i-1]+1;j<=1000;j++)
        {
  			bool fl=true;
       		for (int k=0;k<i;k++) if (u[a[k]+j]) { fl=false;break; }
       		if (fl) { a[i]=j;for (int k=0;k<i;k++) u[a[k]+a[i]]=true; break; }
        }
    }
    for (int i=0;i<n;i++) for (int j=0;j<n;j++)
    {
  		if (i==j) cout<<0; else cout<<a[i]+a[j];
  		if (j==n-1) cout<<endl;else cout<<" ";
    }

    return 0;
}
