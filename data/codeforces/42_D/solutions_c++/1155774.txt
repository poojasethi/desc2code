#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<queue>
using namespace std;
int g[25][25],flag[1005],a[25];
int n;
bool ok(int x,int y)
{
	for(int i=1;i<x;i++)
		for(int j=1;j<x;j++)
			for(int k=1;k<x;k++)
				if(a[i]+a[j]-a[k]==y)
					return false;
	return true;
}
int main()
{
	int i,j,k,l;
	while(scanf("%d",&n)!=EOF)
	{
		memset(g,0,sizeof(g));
		memset(flag,0,sizeof(flag));
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=500;j++)
				if(ok(i,j))
				{
					flag[j]=1;
					a[i]=j;
				}
		}
		for(i=1;i<=n;i++)
		{
			for(j=1;j<=n;j++)
			{
				printf("%d%c",(i==j?0:a[i]+a[j]),j==n?'\n':' ');
			}
		}
	}
	return 0;
}
