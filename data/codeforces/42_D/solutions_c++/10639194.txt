#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int a[233],f[2333];

int main()
{
    int i,j,k,m,n;
    scanf("%d",&n);
    m=0;
    memset(f,true,sizeof(f));
    for(i=1;m<n;i++)
    {
        for(j=1;j<=m;j++)
        {
            if(f[i+a[j]]==false)break;
        }
        if(j>m)
        {
            for(j=1;j<=m;j++)
            {
                f[i+a[j]]=false;
            }
            m++;
            a[m]=i;
        }
    }
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
            if(i==j)printf("0 ");
            else printf("%d ",a[i]+a[j]);
        }
        printf("\n");
    }
    return 0;
}
