#include <stdio.h>
int check[1005];
int main()
{
	int pot[20]={0,};
	pot[0]=1;
	for (int q=1;q<20;++q)
	{
		for (int i=pot[q-1]+1;;++i)
		{
			int good = 1;
			for (int w=0;w<q;++w)
				if (check[ pot[w] + i])
				{
					good = 0;
					break;
				}
			if (good)
			{
				for (int w=0;w<q;++w) check[pot[w]+i]=1;
				pot[q] = i;
				break;
			}
		}
	}

	int n;
	scanf("%d",&n);
	for (int q=0;q<n;++q,printf("\n")) for (int w=0;w<n;++w)
	{
		if (q==w) printf("0 ");
		else printf("%d ",pot[q]+pot[w]);
	}
	return 0;
}
