#include <cstdio>
#include <cstring>

const int ANS[] = {0,1,2,4,7,12,20,29,38,52,73,94,127,151,181,211,257,315,373,412};

int n;

int main()
{
   scanf("%d", &n);
   int i, j;
   for (i = 0; i < n; ++i)
   {
      for (j = 0; j < n; ++j)
      if (i == j) printf("0 ");
      else printf("%d ", ANS[i] + ANS[j]);
      printf("\n");
   }
   return 0;
}
