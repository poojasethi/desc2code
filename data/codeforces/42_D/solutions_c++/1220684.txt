#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<vector>
using namespace std;

int a[21];
int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);

	int n;
	cin >> n;
	a[0] = 1, a[1] = 2, a[2] = 3;
	for(int i = 3; i < n; ++i)
		for(int j = a[i - 1] + 1; j <= 1000; ++j)
		{			
			bool good = true;
			for(int i1 = 0; i1 < i && good; ++i1)
				for(int i2 = 0; i2 < i && good; ++i2)
					for(int i3 = 0; i3 < i && good; ++i3)
						if(a[i1] + a[i2] - a[i3] == j)
							good = false;
			if(good)
			{
				a[i] = j;
				break;
			}
		}	
	for(int i = 0; i < n; ++i)
	{
		for(int j = 0; j < n; ++j)
			cout << (i == j ? 0 : a[i] + a[j]) << ' ';
		cout << endl;
	}
	return 0;
}
