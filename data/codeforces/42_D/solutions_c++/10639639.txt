#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

bool f[1005];
int n,m,a[25];
int main() {
	scanf("%d",&n);
	for (int i=1;i<=1000;i++) {
		bool flag=true;
		for (int j=1;j<=m;j++) if (f[i+a[j]]) {
			flag=false; break;
		}
		if (flag) {
			a[++m]=i;
			for (int j=1;j<=m;j++) f[i+a[j]]=true;
		}
		if (m==n) break;
	}
	for (int i=1;i<=n;i++)
		for (int j=1;j<=n;j++) {
			if (i==j) printf("0"); else printf("%d",a[i]+a[j]);
			if (j<n) printf(" "); else printf("\n");
		}
	return 0;
}
