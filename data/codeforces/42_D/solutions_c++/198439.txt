#include <cstdlib>
#include <iostream>

using namespace std;

int main(int argc, char *argv[])
{
    int n;
    cin>>n;
    int a[20]={0, 1, 2, 4, 7, 12, 20, 29, 38, 52, 73, 94, 127, 151, 181, 211, 257, 315, 373, 412};
    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(i==j) cout<<0<<" ";
            else cout<<a[i]+a[j]<<" ";
        }
        cout<<endl;
    }
    /*system("PAUSE");

}
