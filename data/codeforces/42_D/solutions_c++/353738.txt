/* Greedy Algorithm */

#include<cstdio>

#define	rep(i,n)	for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n;	scanf("%d",&n);
	int mark[20]={0,1};
	for(int i=2;i<n;i++){
		int p;
		for(p=2;;p++){
			rep(a,i)rep(b,a)rep(c,i) if(mark[a]+mark[b]==mark[c]+p) goto WRONG;
			break;

			WRONG:;
		}
		mark[i]=p;
	}

	rep(u,n)rep(v,n)	printf("%d%c",u==v?0:mark[u]+mark[v],v<n-1?' ':'\n');

	return 0;
}
