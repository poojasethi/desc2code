/* Prewritten code begins */
#define REP(i,n)    for(int i=0; i<(n); ++i)
/* Prewritten code ends */

const int maxN = 20;
int a[maxN], n;
int main() {
	cin >> n;
	REP(i,n) {
		set<int> aux;
		REP(j,i) REP(k,i) REP(l,i) aux.insert(a[j]+a[k]-a[l]);
		for(int j=1;;j++) if(!aux.count(j)) { a[i] = j; break; }
	}
	REP(i,n) REP(j,n) cout << (i==j ? 0 : a[i]+a[j]) << (j+1==n ? '\n' : ' ');
	return 0;
}
