#include<iostream>  
using namespace std;

int a[21],n,i,j,k;
bool bj[2001];

int main()
{
    cin>>n;
    k=0;
    for(i=0;i<=1000;i++)
    {
        for(j=1;j<=k;j++) if(bj[i+a[j]]) break;
        if (bj[i+a[j]]) continue;
        for(j=1;j<=k;j++) bj[i+a[j]]=true;
        a[++k]=i;
        if(k==n) break;
    }
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n;j++)
        {
          cout<<(i==j?0:a[i]+a[j]);
          if (j<n) cout<<" ";
        }
        cout<<endl;
    }
    return 0;
}
