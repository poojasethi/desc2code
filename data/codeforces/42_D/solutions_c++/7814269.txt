#include <stdio.h>
#include <cstring>
#include <cmath>
#include <cstring>
#include <iostream>

using namespace std;

int a[25][25];

int main()
{
    int n,cnt=3;
    scanf("%d",&n);
    int c[]={1,2,3,5,8,13,21,30,39,53,74,95,128,152,182,212,258,316,374,413};
    for(int i=0;i<n;i++){
        for(int j=0;j<i;j++){
            a[i][j]=a[j][i]=c[i]+c[j];
        }
        a[i][i]=0;
    }

    for(int i=0;i<n;i++){
        for(int j=0;j<n;j++){
            if(j!=n-1) printf("%d ",a[i][j]);
            else printf("%d\n",a[i][j]);
        }
    }
    return 0;
}
