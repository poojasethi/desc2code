#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int n,i,j,a[120],bo[12000],k;
int main(){
	scanf("%d",&n);
	a[1]=0; a[2]=1; a[3]=2;
	bo[a[1]+a[2]]=bo[a[1]+a[3]]=bo[a[2]+a[3]]=1;
	for(i=4;i<=n;++i){
		for(j=a[3];j<=1000;++j){
			for(k=1;k<i;++k)if(bo[j+a[k]])break;
			if(k==i){
				a[i]=j;
				for(k=1;k<i;++k)bo[j+a[k]]=1;
				break;
			}
		}
	}
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)printf("%d%c",i==j?0:a[i]+a[j],j==n?'\n':' ');
}
