n = int(raw_input())

w = [False for i in xrange(2000)]
w[0] = True
f = [0]
for i in xrange(n):
	for j in xrange(1, 1001):
		ok = True
		for v in f:
			if w[v + j]:
				ok = False
				break
		if ok:
			for v in f:
				w[v + j] = True
			f.append(j)
			break
f = f[1:]
ans = [[0 for i in xrange(n)] for j in xrange(n)]
for i in xrange(n):
	for j in xrange(n):
		ans[i][j] = f[i] + f[j]
	ans[i][i] = 0

	for j in xrange(n):
