N = input()

Ai = [1]

i = 2
edges = set()

while (len(Ai) != N):

    flag = True
    for a in Ai:
        if (a + i) in edges:
            flag = False
            break

    if (flag):
        for a in Ai:
            edges.add(a + i)
        Ai.append(i)

    i += 1

for ai in Ai:
    tmp = ""
    for aj in Ai:
        if (ai == aj):
            tmp += "0 "
        else:
            tmp += str(ai + aj) + " "
    print tmp
