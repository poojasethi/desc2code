#!/usr/bin/python

import sys
from math import floor, ceil

def read_obj(cls):
    return cls(sys.stdin.readline().strip())

def read_obj_list(cls):
    return map(cls, sys.stdin.readline().strip().split())

def solve():
    n = read_obj(int)
    nums = read_obj_list(float)

    sl = sr = 0.0
    zl = zr = 0

    for i, num in enumerate(nums):
        if i % 2:
            if abs(num - round(num)) < 10e-5:
                zr += 1
            else:
                sr += ceil(num) - num
        else:
            if abs(num - round(num)) < 10e-5:
                zl += 1
            else:
                sl += num - floor(num)

    diff = abs(sl - sr)

    if sl > sr:
        zeroes = zr
    else:
        zeroes = zl

    if diff > zeroes:
        res = diff - zeroes
    else:
        res = min(diff - floor(diff), ceil(diff) - diff)

    return "%.3f" % res

if __name__ == "__main__":
    print solve()
