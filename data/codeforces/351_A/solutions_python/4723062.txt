#!/usr/bin/env python

n = int(raw_input())
a = map(lambda x: float(x) - int(float(x)), raw_input().split())
num_int = sum(1 for x in a if x == 0)

sum_a = sum(a)
min_diff = 100000
for num_round_up in xrange(max(0, n - num_int), min(n, 2 * n - num_int) + 1):
    cur_min_diff = abs(sum_a - num_round_up)
    if cur_min_diff < min_diff:
        min_diff = cur_min_diff
print '{0:.3f}'.format(min_diff)
