n = int(raw_input())
a = map(float, raw_input().split())

def frac(x):
    return x - int(x)

def minNum(s, m, M):
    res = abs(s-m)
    for i in xrange(m+1, M+1):
        res = min(abs(s-i), res)
    return res

nI = len(filter(lambda i: int(i) == i, a))
s = sum(map(frac, a))

if nI > n:
    print "%.3f" % minNum(s, 0, 2*n - nI)
else:
    print "%.3f" %minNum(s, n-nI, n)
