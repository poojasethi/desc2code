n = int(raw_input())
a = map(float,raw_input().split(' '))
import math
c = 0
down = 0.0
for num in a:
    if math.ceil(num) == num:
        c += 1
    down += num - math.floor(num)

ans = 2000000000
for i in xrange(c+1):
    if i > n:
        break
    ans = min(ans,abs(down - (n-i)))

print '%.3lf'%ans
