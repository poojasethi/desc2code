n = input()
v = [x - int(x) for x in list(map(float, raw_input().split()))]

total = sum(v)
#dp[i][j] = diferenta minima (memorata fara modul) dintre N si o submultime de j valori din primele i

k = v.count(0.0)
r = 2*n - k

smax = min(r, n)
smin = 0 if n >= r else r - n

print '%.3f' % min([abs(float(x) - total) for x in range(smin, smax+1)])
