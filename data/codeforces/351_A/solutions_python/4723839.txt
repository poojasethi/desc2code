n = input()
num = sorted(map(lambda x: int(x.split('.')[1]), raw_input().split()))
zero_num = sum([1 if number == 0 else 0 for number in num])
ans = abs((n - zero_num) * 1000 - sum(num))
for pair_num in xrange(zero_num):
    ans = min(ans, abs((n - pair_num) * 1000 - sum(num)))
print '%d.%03d' % (ans / 1000, ans % 1000)
