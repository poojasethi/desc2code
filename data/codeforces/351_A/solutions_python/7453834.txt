a = raw_input().split()[0]
a = int(a)
res = 0.0
lst = map(float, raw_input().split())

sm = 0.0
tot = 0
tot2 = 0

for i in range(a*2):
    temp = int(lst[i] * 1000 + 0.5) % 1000
    if temp > 0 :
        #print "temp : ", temp
        sm += temp
        tot2 += 1   
    else:
        tot += 1

i = max(0, a-tot)

ans = 1000000009.0
while i < min(a, tot2) + 1:
    ans = min(ans, abs(sm - i * 1000))
    i+=1

print "%.3f" % (ans/1000)
