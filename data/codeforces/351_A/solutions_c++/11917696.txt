#include<bits/stdc++.h>
using namespace std;
int r=1e9,z=0,s=0,a,b,n,i;
main()
{
    scanf("%d",&n);
    for (i=1;i<=2*n;i++)
    {
        scanf("%d.%d",&a,&b);
        s+=b;
        if (b) z++;
    }
    for (i=max(0,z-n);i<=min(n,z);i++)
        r=min(r,abs(s-1000*i));
    printf("%d.%03d",r/1000,r%1000);
}
