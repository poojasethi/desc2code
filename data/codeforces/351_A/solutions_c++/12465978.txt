#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>


using namespace std;



int main(){
	int n, kol = 0;
	double b, ans, tmp = 0;
	scanf("%d", &n);
	for (int i = 1; i <= n + n; i++){
		cin >> b;
		b -= int(b);
		tmp += b;
		if ( b > 0.0001)
			kol++;
	}
	ans = 1000000000;
	for (int i = kol - n; i <= n; i++)
		ans = min(ans, fabs(i - tmp));
	printf("%.3lf\n", ans);
	return 0;
}
