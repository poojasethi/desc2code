#include <cstdio>
#include <iostream>
#include <cstdlib>
using namespace std;
int main() {
	int n;
	scanf("%d",&n);
	int s=0;
	int z=0;
	for(int i=0; i<2*n; ++i) {
		int a,b;
		scanf("%d.%d", &a,&b);
		s += b;
		if (b) ++z;
	}
	int r=1e9;
	for(int i=max(0,z-n); i<=min(z,n); ++i)
		r = min(r, abs(s-1000*i));
	printf("%d.%03d\n",r/1000,r%1000);
}
