#include<cstdio>
#include<cstring>
#include<algorithm>
#include<cstdlib>
using namespace std;
int i,n,ans,a[12000],x,cnt,sum;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n+n;++i){
		double v;
		scanf("%lf",&v);
		x=(int)(v*1000+1e-9);
		a[i]=x%1000;
		if(a[i]==0)cnt++;
		sum-=a[i];
	}
	ans=100000000;
	for(i=max(0,n-cnt);i<=min(n,2*n-cnt);++i)ans=min(ans,abs(sum+i*1000));
	printf("%.3lf\n",ans/1000.0);
}
