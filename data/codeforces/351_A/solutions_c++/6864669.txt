#include<bits/stdc++.h>
using namespace std;
int main()
{
	//freopen("input.txt", "r",stdin);
	int n;
	cin>>n;
	int frac=0;
	int a,b,res=0;
	for(int i=0;i<2*n;i++)
	{
		scanf("%d.%d",&a,&b);
		res+=b;
		if(b>0)
		{
			frac++;
		}
	}
	int ans=100000000;
	for(int i=max(frac-n,0);i<=min(frac,n);i++)
	{
		ans=min(ans,abs(res-i*1000));
	}
	printf("%d.%03d",ans/1000,ans%1000);
}
