#include <bits/stdc++.h>
#define INF 1000000009
using namespace std;

int n, x, ints, sum, ans;

int main(){

  scanf("%d", &n);

  sum = ints = 0;
  for(int i = 0; i < (2 * n); i ++){
    scanf("%*d.%d", &x);
    if(x == 0) ints ++;
    sum += x;
  }

  ans = INF;
  for(int i = max(0, n - ints); i <= min(2 * n - ints, n); i ++)
    ans = min(ans, abs(sum - 1000 * i));

  printf("%d.%03d\n", ans / 1000, ans % 1000);

  return 0;
}
