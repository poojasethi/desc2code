#include<bits/stdc++.h>
using namespace std;

int main()
{
	int n,tmp,a,sum,sum0,i,ans;
	char ch;
	while(cin>>n)
	{
		sum=sum0=0;
		for(i=1;i<=2*n;i++)
		{
			scanf("%d%c%d",&tmp,&ch,&a);
			if(a==0)
				sum0++;
			sum+=a;
		}
		ans=1<<29;
		for(i=n;i>=n-sum0;i--)
		{
			ans=min(ans,abs(sum-i*1000));
		}
		printf("%d.%03d",ans/1000,ans%1000);
	}
}
