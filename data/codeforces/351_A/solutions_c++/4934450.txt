#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

int N,M;

double ans=1e9,sum;

int main()
{
    cin>>N;
    for(int i=1;i<=2*N;++i)
    {
        double t;
        cin>>t;
        if(ceil(t)!=t)  ++M;
        sum+=ceil(t)-t;
    }
    for(int i=max(0,M-N);i<=min(N,M);++i) ans=min(ans,fabs(sum-i));
    printf("%.3lf\n",ans);
    return 0;
}
