#include<bits/stdc++.h>
//#define mod 1000000007
using namespace std;
#define ll long long int
int main(){
double n;
cin>>n;
double a[4110];
double nz=0;
double tot=0;
for(int i=0;i<2*n;i++){
	cin>>a[i];
	a[i]-=(int)a[i];
	if(a[i]!=0) nz++;
	tot+=a[i];
}
double ans=10000;
	for(int i=nz;i>=0;i--){
     if( (n-i<0) || (nz-i<0) || (n-nz+i<0) ) continue;
     ans=min(ans,fabs(tot-i));
	}
printf("%.3f",ans);
}
