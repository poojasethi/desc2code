#include<iostream>
#include<cstring>
#include<cstdio>
#include<cmath>
using namespace std;

int main()
{
    int n,i,j,k;
    scanf("%d",&n);
    int cnt=0,sum=0,ans=1000000000;
    for(i=0;i<2*n;i++)
    {
        scanf("%d.%d",&j,&k);
        sum+=k; cnt+=(k==0);
    }
    for(i=n-cnt;i<=n;i++)
        ans=min(ans,int(fabs(sum-i*1000)));
    printf("%.3f\n",double(ans)/1000);
    return 0;
}
