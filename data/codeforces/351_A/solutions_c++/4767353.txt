#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
	int n,ln=0;
	double x,ans=0,a,b;
	scanf("%d",&n);
	for(int i=2*n;i;--i)
	{
		scanf("%lf",&x);
		x-=(int)x;
		if(x)
		{++ln;ans+=x;}
	}
	a=ans-max(0,ln-n);
	b=ans-min(n,ln);
	if(b>0)printf("%.3lf\n",b);
	else if(a<0)printf("%.3lf\n",-a);
	else
	{
		a-=(int)a;
		if(a>0.5)a=1-a;
		printf("%.3lf\n",a);
	}
}
