#include <bits/stdc++.h>
using namespace std;

int n, z;
double a[4000], sum, ans;

int main() {
	scanf("%d", &n);
	for (int i = 0; i < 2*n; i++) {
		scanf("%lf", a + i);
		a[i] -= floor(a[i]);
		if (a[i] > 0)
			z++;
		sum += a[i];
	}
	double ans = 1e9;
	for (int q = max(z-n, 0); q <= min(n, z); q++)
		ans = min(ans, abs(sum - q));
	printf("%.3lf", ans);
}
