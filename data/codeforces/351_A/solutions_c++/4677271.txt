#include <cstdio>
#include <algorithm>
using namespace std;

int main() {
	int n, sum = 0, one = 0;
	scanf("%d", &n);
	for(int a, i = 0; i < n+n; ++i) {
		scanf("%*d%*c%d", &a);
		if(a) sum += a, one++;
	}
	int mi = max(0, one-n);
	int ma = min(n, one);
	int ans = n*2000;
	for(int i = mi; i <= ma; ++i) {
		ans = min(ans, abs(sum-i*1000));
	}
	int p = ans / 1000;
	int q = ans % 1000;
	printf("%d.%03d\n", p, q);
}
