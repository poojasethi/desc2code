#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;

int n,c,a=10000000,m=0,v,s=0;

int main(){
    scanf("%d",&n);
    for(int i=0; i<2*n; i++){
        scanf("%d.%d",&v,&c);
        s+=c;
        m+=c==0;
    }
    for(int i=max(0,n-m); i<=min(2*n-m,n); i++){
        a=min(abs(a),abs(s-1000*i));
    }
    printf("%d.%03d\n",a/1000,a%1000);
    return 0;
}
