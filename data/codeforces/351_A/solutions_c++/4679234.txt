#include <cstdio>
#include <iostream>
#include <cmath>
using namespace std;
int main(){
    int n,i,c=0,lb,ub;
    double a,s=0;
    scanf("%d",&n);
    for (i=1;i<=2*n;i++){
        scanf("%lf",&a);
        if (a!=floor(a)) c++;
        s+=a-floor(a);
    }
    lb=max(c-n,0);
    ub=min(c,n);
    s-=lb;
    i=lb;
    while (i<ub&&abs(s)>abs(s-1)) i++,s--;
    printf("%.3lf\n",abs(s));
    return 0;
}
