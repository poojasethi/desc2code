#include<cstdio>
#include<algorithm>
using namespace std;
int n,s,x,y,ct,ans;
int main(){
    scanf("%d",&n);
    for(int i=0;i<2*n;i++){
        scanf("%d.%d",&x,&y);
        if(y) ct++;
        s += y;
    } 
    ans = 1e9; 
    for(int i=max(0,ct-n);i<=min(n,ct);i++)
        ans = min(ans,abs(s-1000*i));
    printf("%d.%03d",ans/1000,ans%1000);
    return 0;
}
