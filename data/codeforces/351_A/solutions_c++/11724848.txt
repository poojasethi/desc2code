#include <bits/stdc++.h>
using namespace std;
int main(){
    int frac=0,sum=0,n,a,b,ans=10000007;
    scanf("%d",&n);
    for(int i=0;i<2*n;i++){
        scanf("%d.%d",&a,&b);
        sum=sum+b;
        if(b>0) frac++;
    }
    for(int i=max(frac-n,0);i<=min(frac,n);i++){
        ans=min(ans,abs(sum-i*1000));
    }
    printf("%d.%03d\n",ans/1000,ans%1000);
    return 0;
}
