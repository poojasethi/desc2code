#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
int n,a,b,i,ans,sum,num,zero;
int main()
{
  scanf("%d",&n);ans=999999999;
  for (i=1;i<=2*n;i++)
  {
    scanf("%d.%d",&a,&b);sum+=b;
    if (b) num++;else zero++;
  }
  int end=(num<=zero)?num:n;
  for (i=(num<=zero)?0:n-zero;i<=end;i++)
    ans=min(ans,abs(sum-i*1000));
  printf("%.3lf",ans/1000.0);
  return 0;
}
