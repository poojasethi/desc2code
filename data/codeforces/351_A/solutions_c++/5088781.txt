#include <bits/stdc++.h>

using namespace std;

int n, x, y, s, ret = 22222222, cnt;

int main()
{
  scanf("%d", &n);
  for (int i = 0; i < 2 * n; i++)
  {
    scanf("%d.%d", &x, &y);
    s += y;
    if (y)
    {
      cnt++;
    }
  }
  for (int i = max(0, cnt - n); i <= min(n, cnt); i++)
  {
    ret = min(ret, abs(s - 1000 * i));
  }
  printf("%d.%03d\n", ret / 1000, ret % 1000);
}
