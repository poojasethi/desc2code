#include<cstdio>
#include<algorithm>
#include<cstdlib>
#define M 1000
using namespace std;

int main()
{
	int n;scanf("%d",&n);
	int s = 0, z = 0;
	for(int i=0;i<2*n;++i)
	{
		int u,v; scanf("%d.%d",&u,&v);
		s += v;
		z += v>0;
	}
	
	// m number of closing
	int ans = 1e9;
	for(int i=max(0,z-n); i <= min(z,n) ;++i)
	{
		ans = min(ans, abs(s-M*i) );
	}
    
	printf("%d.%03d\n",ans/M,ans%M);
	return 0;
}
