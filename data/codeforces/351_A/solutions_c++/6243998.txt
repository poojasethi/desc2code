#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<cmath>
using namespace std;

int n , x=0 ;
double y , z , tot=0.0 , ans=1000000.0 ; 

int main()
{
    scanf("%d",&n) ; x=2*n ;
    for(int i=1 ; i<=2*n && scanf("%lf",&z) ; i++) if(z-(int)z>1e-6) tot+=z-(int)z , x-- ;
    for(int i=min(n,x) ; i>=max(0,x-n) ; i--) { y=abs(tot-n+i) ; ans=min(ans,y) ; }    
    printf("%.3lf\n",ans) ;

 return 0 ; 
}
