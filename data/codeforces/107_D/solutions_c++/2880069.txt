#include <cstdio>
#include <algorithm>
#include <iostream>
#include <cstring>
using namespace std;
const int mo=12345,nn=1001;
int a[27],v[27],p[nn],num[nn],c[123][123];
char str[15];
int s=1,w[123][123],b[123][123];
void matrix(long long a)
{
	int i,j,k;
	if (!a) for (i=0;i<s;i++) c[i][i]=1;else
	{
		matrix(a>>1);memset(b,0,sizeof(b));
		for (i=0;i<s;i++) for (j=0;j<s;j++) for (k=0;k<s;k++)
		b[i][j]=(c[i][k]*c[k][j]+b[i][j])%mo;
		for (i=0;i<s;i++) for (j=0;j<s;j++) c[i][j]=b[i][j];
		if (a&1)
		{
			memset(b,0,sizeof(b));
			for (i=0;i<s;i++) for (j=0;j<s;j++) for (k=0;k<s;k++)
			b[i][j]=(c[i][k]*w[k][j]+b[i][j])%mo;
			for (i=0;i<s;i++) for (j=0;j<s;j++)
			c[i][j]=b[i][j];
		}
	}
}
int main()
{
	long long n;int i,j,k,m,t,ans=0;
	scanf("%I64d%d",&n,&m);if (m) getchar();
	for (i=1;i<=m;i++)
	{
		gets(str+1);p[i]=str[1]-'A'+1;v[str[1]-'A'+1]=1;
		for (j=3;str[j]>='0'&&str[j]<='9';j++) num[i]=num[i]*10+str[j]-'0';
		s*=num[i];
	}
	for (i=1;i<=m;i++) v[p[i]]*=num[i];
	for (i=0;i<s;i++)
	{
		t=i;
		for (j=1;j<=26;j++) if (v[j])
		{
			a[j]=t%v[j];t/=v[j];
		}
		t=1;
		for (j=1;j<=26;j++) if (v[j])
		{
			if (a[j]==v[j]-1) w[i][i-a[j]*t]++;
			else w[i][i+t]++;
			t*=v[j];
		}
	}
	matrix(n);
	for (i=0;i<s;i++)
	{
		t=i;
		for (j=1;j<=26;j++)	if (v[j])
		{
			a[j]=t%v[j];t/=v[j];
			for (k=1;k<=m&&(p[k]!=j||a[j]%num[k]);k++);
			if (k>m) break;
		}
		if (j>26) ans=(ans+c[0][i])%mo;
	}
	printf("%d\n",ans);
}
