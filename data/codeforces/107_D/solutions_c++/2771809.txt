#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[125][125],f[125][125],type[30],mul[30],b[30],v[30][125];
int m,t,tot,i,j,k,temp,x,cnt,ans;
long long n;
char ch;

void calc(int a[125][125],int b[125][125])
{
	int c[125][125],i,j,k;
	memset(c,0,sizeof(c));
	for(i=0;i<tot;i++)
		for(j=0;j<tot;j++)
			for(k=0;k<tot;k++)
				c[i][j]=(c[i][j]+a[i][k]*b[k][j])%12345;
	memcpy(a,c,sizeof(c));
}

int main()
{
	cin>>n>>m;
	for(i=1;i<=m;i++)
	{
		cin>>ch>>x;
		v[ch-'A'][x]=1;
	}
	for(i=0;i<26;i++)
		if(v[i][1]) v[i][0]=1,cnt++;
		else{
			for(j=2;j<=123;j++)
				if(v[i][j]) break;
			v[i][0]=j>123;
		}
	for(i=0,tot=1;i<26;i++)
		if(!v[i][0])
		{
			mul[++t]=1,type[t]=i;
			for(j=1;j<=123;j++)
				if(v[i][j]) mul[t]*=j;
			tot*=mul[t];
		}
	for(i=0;i<tot;i++)
	{
		a[i][i]=cnt;
		memset(b,0,sizeof(b));
		for(j=1,k=i;j<=t;k/=mul[j],j++) b[j]=k%mul[j];
		for(j=1;j<=t;j++)
		{
			b[j]=(b[j]+1)%mul[j];
			for(k=t,temp=0;k;k--)
				temp=temp*mul[k]+b[k];
			a[i][temp]=1;
			b[j]=(b[j]-1+mul[j])%mul[j];
		}
	}
	f[0][0]=1;
	for(;n;n>>=1)
	{
		if(n&1) calc(f,a);
		calc(a,a);
	}
	for(i=0;i<tot;i++)
	{
		for(j=1,k=i;j<=t;k/=mul[j],j++) b[j]=k%mul[j];
		for(j=1,temp=0;j<=t;j++)
			for(k=2;k<=123;k++)
				if(v[type[j]][k]&&b[j]%k==0) temp|=1<<j-1;
		if(temp==(1<<t)-1) ans+=f[0][i];
	}
	cout<<ans%12345<<endl;
	return 0;
}
