n = int(raw_input())
p = [0] * (n+1)

lo, hi = 1, n

possible = True

while hi - lo > 2:
  if not p[hi-1]: p[hi-1] = lo
  if not p[lo]:   p[lo]   = lo+1
  if not p[hi]:   p[hi]   = hi-1
  if not p[lo+1]: p[lo+1] = hi

  lo += 2
  hi -= 2
  n -= 4

# lo is the first zero
if n == 1:
  p[lo] = lo
elif n != 0:
  possible = False

if possible:
  print ' '.join(map(str, p[1:]))
else:
  print -1
