
n = int(raw_input())

if n % 4 == 2 or n % 4 == 3:
    print "-1"

else:
    perm = [0 for i in range(n)]
    d = n/4
    for i in range(d):
        perm[2*i] = 2*i + 2
        perm[2*i + 1] = n - 2*i
        perm[n - 2*i - 1] = n - 2*i - 1
        perm[n - 2*i - 2] = 2*i + 1
        
    if n % 2 == 1:
        perm[n / 2] = n / 2 + 1

    print " ".join([str(x) for x in perm])
