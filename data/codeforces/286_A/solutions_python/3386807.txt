N = int(raw_input())

if N%4 >= 2:
    print -1
else:
    perm = [0 for i in xrange(N)]
    for i in xrange(0,N/2,2):
        perm[i] = i+1
        perm[i+1] = N-i-1
        perm[-i-1] = N-perm[i]-1
        perm[-i-2] = i

    if N%2 == 1:
        perm[N/2] = N/2

    for p in perm:
        print p+1,
    print
