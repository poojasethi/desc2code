#include<cstdio>
using namespace std;
int n,i,c,sum;
int main()
{
  scanf("%d",&n);
  if (n%4==2||n%4==3) {puts("-1");return 0;}
  sum=n+2;
  for (i=1;i<=n/2/2;i++)
    c+=2,printf("%d %d ",c,sum-c);
  if (n&1) printf("%d ",n/2+1);
  sum-=2;
  c=n/2+1;
  for (i=1;i<=n/2/2;i++)
    c-=2,printf("%d %d ",c,sum-c);
  return 0;
}
