#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,p[100010],i;
int main()
{
	cin>>n;
	if(n%4==2||n%4==3) {puts("-1"); return 0;}
	for(i=1;i<n/2;i+=2) p[i]=i+1,p[i+1]=n-i+1,p[n-i+1]=n-i,p[n-i]=i;
	if(n%4==1) p[n/2+1]=n/2+1;
	for(i=1;i<=n;i++) cout<<p[i]<<' ';
	puts("");
	return 0;
}
