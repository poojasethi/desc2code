#include<iostream>
#include<cstring>
using namespace std;

int a[100001];

int main()
{
	int i,n;
	cin>>n;
	if(n%4==2 ||n%4==3){
		cout<<-1;
		return 0;
	}
	if(n%2==1)  a[(n+1)/2]=(n+1)/2;
	for(int i=1;i<=n/2;i+=2)
	{
		a[i]=i+1;
		a[i+1]=n-i+1;
		a[n-i]=i;
		a[n-i+1]=n-i;
	}
	cout<<a[1];
	for(int i=2;i<=n;i++)
	   cout<<" "<<a[i];
	return 0;
}
