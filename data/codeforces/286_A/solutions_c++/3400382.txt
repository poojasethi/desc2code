#include <cstdio>
#include <iostream>

using namespace std;

int n;
int a[100010];

int main(){
	scanf("%d",&n);
	if	(n%4>1)	puts("-1");
	else{
		for	(int i=1;i<n/2;i++){
			if	(a[i])	continue;
			a[i]=i+1;
			a[i+1]=n-i+1;
			a[n-i+1]=n-i;
			a[n-i]=i;
		}
		if	(n&1)	a[n/2+1]=n/2+1;
		for	(int i=1;i<=n;i++)
			cout <<a[i] <<" ";
		cout <<endl;
	}
}
