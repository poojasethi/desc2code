#include <cstdio>
using namespace std;

const int MAXN = 1e5;

int n, p[MAXN+1];


int main()
{
	scanf("%d", &n);
	if (n%4!=0 && n%4!=1) {
		puts("-1");
		return 0;
	}
	
	if (n%4 == 1) p[n/2+1] = n/2+1;
	for (int i=1; 2*i<=n; i+=2) {
		p[i] = i+1;
		p[i+1] = n-i+1;
		p[n-i+1] = n-i;
		p[n-i] = i;
	}
	
	for (int i=1; i<=n; ++i) printf("%d ", p[i]);
	putchar('\n');
	return 0;
}
