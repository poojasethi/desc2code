#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int i,p[120000],n;
int main(){
	scanf("%d",&n);
	if(n%4!=0 && n%4!=1)return puts("-1"),0;
	for(i=1;i+i<=n;i+=2){
		p[i]=i+1;
		p[i+1]=n-i+1;
		p[n-i+1]=n-i;
		p[n-i]=i;
	}
	if(n%4==1)p[n/2+1]=n/2+1;
	for(i=1;i<=n;++i)printf("%d ",p[i]);
}
