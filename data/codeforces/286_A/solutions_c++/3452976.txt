#include<iostream>
using namespace std;
int p[100001];
int main()
{
int n,i=1,k=2,m;
cin>>n;
if (n==1) {cout<<1;return 0;}
if (n%4==2 || n%4==3) {cout<<-1;return 0;}
m=n;
while (!p[i])
{
if (p[i+1]>0) {p[i]=i;break;}
p[i]=k;
p[i+1]=n-i+1;
p[m-1]=p[i]-1;
p[m]=p[i+1]-1;
k+=2; i+=2; m-=2;
}
for (i=1; i<=n; i++)
cout<<p[i]<<" ";
return 0;
}
