#include<iostream>

using namespace std;

int n;
int nums[100001];

int main()
{
	cin>>n;
	if(n%4==0||n%4==1)
	{
		for(int i=1;i<=n/2;i+=2)
		{
			nums[i]=i+1;
			nums[i+1]=n-i+1;
			nums[n-i+1]=n-i;
			nums[n-i]=i;
		}
		if(n%4==1)
			nums[n/2+1]=n/2+1;
		for(int i=1;i<n;++i)
			cout<<nums[i]<<' ';
		cout<<nums[n]<<endl;
	}
	else
		cout<<-1<<endl;
}
