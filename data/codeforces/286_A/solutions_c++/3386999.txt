#include <iostream>
#include <algorithm>
#include <string>

using namespace std;
int n,i,j;
int A[100001];
int main(void)
{
	cin>>n;
	if (n%4>=2)
		cout<<-1;
	else {
		for (i=1,j=n;j-i>=3;i+=2,j-=2)
			A[i]=n-i,A[i+1]=i,A[j]=i+1,A[j-1]=j;
		if (i==j)
			A[i]=i;
		for (i=1;i<=n;i++)
			cout<<A[i]<<" ";
	}
	return 0;
}
