#include<stdio.h>
int a[200000];
int main(){
	int n;
	scanf("%d",&n);
	if( n%4>1 ){ printf("-1\n"); return 0; }
	int pos = 0;
	for( ; pos < n/2 ; ){
		a[pos] = pos+1;
		a[pos+1] = n-1-pos;
		a[n-1-pos] = n-2-pos;
		a[n-2-pos] = pos;
		pos+=2;
	}
	if( n%4 == 1) a[n/2] = n/2;
	for(int i = 0 ; i < n; i++ ) printf("%d ",a[i]+1);
	printf("\n");
}
