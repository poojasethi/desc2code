#include<stdio.h>
int a[100010];
int main() {
	int i,k,n;
  	scanf("%d",&n);
  	if(n%4>1){puts("-1");return 0;}
  	for(i=k=0;k+4<=n;i+=2,k+=4){
    	a[i+1]=i+2;
		a[i+2]=n-i;
		a[n-i]=n-i-1;
		a[n-i-1]=i+1;
 	}
  	if(k+1==n)a[i+1]=i+1;
  	for(i=1;i<=n;i++)printf("%d ",a[i]);
  	puts("");
  	return 0;
}
