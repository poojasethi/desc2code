#include <cstdio>

int n;

int tab[100005];

int main()
{
	scanf("%d",&n);
	if((n&3)==2 || (n&3)==3)
	{
		printf("-1\n");
		return 0;
	}
	for(int i=1;i<n/2;i+=2)
	{
		tab[i]=i+1;
		tab[i+1]=n+1-i;
		tab[n-i]=i;
		tab[n+1-i]=n-i;
	}
	if(n&1) tab[(n+1)>>1]=(n+1)>>1;
	for(int i=1;i<=n;i++) printf("%d ",tab[i]);
	printf("\n");
	return 0;
}
