#include<cstdio>
using namespace std;

int n,i,a[100001];

void doit(int q,int h)
{
	if (q>h)	return ;
	if (q==h)	a[q]=q;
	else	if (h-q+1>=4)
	{
		a[q]=q+1;
		a[q+1]=h;
		a[h-1]=q;
		a[h]=h-1;
		doit(q+2,h-2);
	}
}
int main()
{
	scanf("%d",&n);
	if (n%4==2||n%4==3)	printf("-1\n");
	else
	{
		doit(1,n);
		for (i=1;i<=n;i++)	printf("%d ",a[i]);
	}
}
