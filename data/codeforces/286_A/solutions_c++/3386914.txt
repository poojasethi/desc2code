#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int n,i,j,k;
int p[100011];
int main(){
	cin>>n;
	if (n%4==2||n%4==3) return cout<<-1,0;
	if (n&1) p[(n+1)/2]=(n+1)/2;
	for (i=1;i<=n/2;i+=2){
		int x=i,y=i+1;
		p[x]=y;
		p[n-y+1]=x;
		p[n-x+1]=n-y+1;
		p[y]=n-x+1;
	}
	for (i=1;i<=n;i++)
			printf("%d ",p[i]);
	return 0;
}
