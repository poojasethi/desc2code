#include <iostream>
using namespace std;
int main()
{
	int n, a[100100];
	cin >> n;
	if((n/2)%2)
	{
		cout << "-1";
		return 0;
	}
	for(int i = 1; i <= n / 2; i+=2)
	{
		a[i] = i+1;
		a[i+1] = n - i + 1;
		a[n-i] = i;
		a[n-i+1] = n-i;
	}
	if(n % 2)
		a[n/2+1] = n/2+1;
	for(int i = 1; i <= n; i++)
		cout << a[i] << " ";
	return 0;
}
