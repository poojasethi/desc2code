#include<cstdio>

int n;
int ans[111111];

int main(){
	scanf("%d",&n);
	if(n%4==2||n%4==3){
		printf("-1\n");
		return 0;
	}
	if(n%2==1)
		ans[n/2+1]=n/2+1;
	int l=1,r=n;
	while(l<r){
		ans[l]=l+1;
		ans[l+1]=r;
		ans[r-1]=l;
		ans[r]=r-1;
		l+=2;r-=2;
	}
	for(int i=1;i<=n;i++)
		printf("%d%c",ans[i],i<n?' ':'\n');

	return 0;
}
