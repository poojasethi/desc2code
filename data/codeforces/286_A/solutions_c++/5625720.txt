#include <iostream>
using namespace std;
int arr[100010];
int main(){
    int n;
    cin >>n;
    if((n/2)&1) cout << -1;
    else{
    for(int i=1;i<=n/2;i+=2)
        arr[i]=i+1,arr[i+1]=n-i+1,arr[n-i+1]=n-i,arr[n-i]=i;
    if(n&1) arr[n/2+1]=n/2+1;
    for(int i=1;i<=n;i++)
        cout << arr[i] << " ";
    }
   return 0;
}
