#include <iostream>
using namespace std;

int n,i,j,a[100100],b[100100];

int main()
{
    cin>>n;
    for (i=1;i<=n/2;i+=2)
    {
      a[i]=i+1; 
      a[i+1]=n-i+1; 
      a[n-i+1]=n-i;
      a[n-i]=i;
    }
    if (n%2) a[n/2+1]=n/2+1;
    for (i=1;i<=n;i++) if (b[a[i]]++) j=1;
    
    if (j) cout<<-1; else
    for (i=1;i<=n;i++) cout<<a[i]<<" ";
}
