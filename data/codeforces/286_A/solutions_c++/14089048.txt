#include <iostream>
using namespace std;
int main(){
    int n;
    cin>>n;
    if((n/2)%2>0){
        cout<<-1;
        return 0;
    }
    for(int i=1;i<=n/4;i++){
        cout<<i*2<<' '<<n-(i-1)*2<<' ';
    }
    if(n%2!=0) cout<<(n+1)/2<<' ';
    for(int i=n/4;i>=1;i--){
        cout<<i*2-1<<' '<<n-(i-1)*2-1<<' ';
    }

}
