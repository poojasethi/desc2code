#include<iostream>
using namespace std;
int num[100005];
int main(){
	int n;
	cin >> n;
	if(n % 4 == 0 || n% 4 == 1){
		for(int i = 1; i <= n/2; i+=2){
			num[n-i+1] = n - i;
			num[i] = i+1;
			num[n-i] = i;
			num[i+1] = n-i+1;
		}
		if(n%4) num[n/2+1] = n/2 + 1;
		for(int i = 1; i <= n; i++)
			cout<< num[i]<<" ";cout<<endl;
	} else cout<< -1 << endl;
}
