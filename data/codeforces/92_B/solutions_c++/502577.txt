#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
using namespace std;

int main()
{
    char c;
    int k = 0, k0 = 0, kk = 0;
    while (cin >> c)
    {
        k++;
        if (c == '0') k0++;
        else kk = k;
    }
    if (kk == 1) cout << k - 1;
    else cout << kk + 1 + k0;
    return 0;
}
