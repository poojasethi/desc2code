#include <iostream>
#include <string>
using namespace std;

int main(){
	string a;
	getline(cin,a);
	int c(0),s(0);
	for(int i=a.size()-1;i>=1;i--){
		
		if(s==1){
			if (a[i]=='1'){c++;}
			else {c+=2;}
		}
		else{
			if (a[i]=='1'){c+=2; s=1;}
			else {c++;}
		}
	}
	if (s==1){c++;}
	cout<<c;
	
}
