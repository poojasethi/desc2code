#include<cstring>
#include<cstdio>
int sayac;
char a[1000001];
int main()
{
	scanf("%s",a);
	int len=strlen(a);
	for(int i=len-1;i>=1;i--)
	{
		if(a[i]=='1')
		{
			sayac++;
			while(i>=0 && a[i]=='1')
			a[i]='0',i--;
			if(i==-1)
			sayac++;
			if(i>=0) a[i]='1',i++;
		}
	}
	printf("%d",sayac+len-1);
}
