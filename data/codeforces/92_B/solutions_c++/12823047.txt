#include<bits/stdc++.h>
using namespace std;

string s;
int x, c, b;

main() {
    ios_base::sync_with_stdio(0);
    cin >> s;
    int l=s.length();
    for(int i=1; i<l; i++) {
        if (s[i]=='1') {
            x+=c;
            c=0;
            b=1;
        } else c++;
    }
    cout << l-1+x+2*b << '\n';
}
