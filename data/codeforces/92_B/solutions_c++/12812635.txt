#include<iostream>
using namespace std;

int main()
{
	string lol;
	string s="0";
	cin>>lol;
	s+=lol;
	int idx=s.length()-1;
	int tot=0;
	if(idx==1) tot=-2;
	int c=0;
	while(idx>=0)
	{
		if(s[idx]=='0')
		{	
			tot+=c+1;
			if(c) c=1;
		}
		else c++;
		idx--;
		if(idx==1 && c==0) tot-=2;
	}
	cout<<tot<<endl;
}
