#include <iostream>

using namespace std;

int main()
{
	string s;
	cin >> s;
	int ans = 0, i = s.size() - 1;
	while (i > 0)
	{
		if (s[i] == '0')
			i--;
		else
		{
			while (s[i] == '1' && i >= 0)
			{
				ans++;
				i--;
			}
			if (i >= 0)
				s[i] ='1';
		}
		ans++;
	}
	cout << ans << endl;
	return 0;
}
