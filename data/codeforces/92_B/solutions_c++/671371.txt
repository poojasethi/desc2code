#include <cstdio>
#include <cstring>
char s[1111111]; int l, c, a;
int main() {
    gets(s); l = strlen(s) - 1; c = a = 0;
    while (l) {
        if (a) if (s[l] == '1') c++; else c += 2;
        else if (s[l] == '1') { c += 2; a = 1; } else c++;
        l--;
    }
    if (a) c++; printf("%d\n", c);
}
