//O(length of number)
#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string s;
    int res;
    cin >> s;
    res = s.rfind('1');
    if (res != 0)
        res = count(s.begin(), s.begin() + res, '0') + 2;
    res += s.size() - 1;
    cout << res;
    return 0;
}
