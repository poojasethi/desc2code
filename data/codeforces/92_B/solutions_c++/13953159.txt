#include<bits/stdc++.h>

using namespace std ;

int main(){

    string s ;
    cin >> s ;
    int x = s.size() , y = 0 , counter = 0 ;
    counter = s.rfind("1") ;
    //cout<<counter;
    y = count(s.begin(),s.end(),'1') ;
    //cout<<" "<<y;
    if(y>1)
        y-=2 ;
    cout << x+counter-y ;
return 0 ;
}
