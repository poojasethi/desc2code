#include <bits/stdc++.h>
using namespace std;

int main(){
	string s;
	cin>>s;
	int i=s.size()-1;
	int r=0;
	while(i>=1)
	{
	//	printf("%d\n",i);
		while(i>=0 and s[i]=='0')
			i--,r++;
		if(i>=1 and s[i]=='1')
		{
			r++;
			while(i>=0 and s[i]=='1')
				i--,r++;
			if(i>=0)
				s[i]='1';
		}
	}
	printf("%d\n",r);
	
	return 0;
}
