#include <cstdio>
int i,j,cnt,tail,a[1000010]; char c;
int main() {
  while (scanf("%c",&c)>=0)
    if (c=='0'||c=='1')
      a[++tail]=(int)(c=='1');

  while (!a[tail]) tail--,cnt++;
  while (tail>1) {
    for (i=tail;a[i];i--);
    cnt+=tail-i+1;
    a[tail=i]=1;
  }
  printf("%d\n",cnt);
  return 0;
}
