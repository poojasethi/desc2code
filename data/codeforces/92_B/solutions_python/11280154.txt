#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
from operator import getitem
input = sys.stdin
output = sys.stdout

def solve(A):
    Ar = reversed(A[1:])
    if len(A)==1:
        return 0
    n = 0
    have_one = False
    d = 0
    for c in Ar:
        d = 0
        if c=='1':
            d = 1
            
        #print d, have_one

        if have_one:
            if d==0:
                d = 1
                have_one = False
            else:
                d = 0
        
        if d==0:
            n += 1
        else:
            n += 2
            have_one = True

    if have_one:
        n += 1

    return n
    
A = input.readline().strip()
assert A[0]=='1' and len(A)<=10**6

n = solve(A)
output.write('%s\n' % n)
