x = raw_input()
x = list(x)

count = 0
while len(x) != 1:
	if x[-1] == '1':
		x[-1] = '0'
		i = -2
		while True:			
			if x[i] == '1':
				x[i] = '0'
				if i + len(x) == 0:
					x.insert(0, '1')
					break;
				i -= 1
			else:
				x[i] = '1'
				break;
		count += 1
	else:
		x.pop()
		count += 1
		
print count
