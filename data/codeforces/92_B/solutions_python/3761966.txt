def main():
    st = raw_input()
    carry = False
    pow_2 = 0
    count = 0
    for i in st[-1::-1]:
        if i == '0':
            if carry:
                count += 2
            else:
                count += 1
        else:
            if carry:
                count += 1
            else:
                carry = True
                count += 2
            if pow_2 < 2:
                pow_2 += 1
    if pow_2 < 2:
        count -= 2
    print count
main()
