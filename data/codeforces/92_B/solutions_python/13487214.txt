i = raw_input()

ops = 0
foundOne = False
n = len(i) - 1
for c in i[::-1]:
    if c == '0' and not foundOne:
        ops += 1
    elif c == '0':
        ops += 2
    elif c == '1' and n != 0:
        ops += 1
        foundOne = True
    elif c == '1' and foundOne:
        ops += 2
    n -= 1

print ops
