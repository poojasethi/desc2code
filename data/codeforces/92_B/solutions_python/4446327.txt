number = list(raw_input())

actions = 0
index = len(number) - 1
while index > 0:
    if number[index] == "1":
        i = index
        while number[i] == "1" and i > 0:
            number[i] = "0"
            i -= 1
        if i == 0:
            number.insert(1, "0")
            index += 1
        else:
            number[i] = "1"
        index -= 1
        actions += 1
    else:
        index -= 1
    actions += 1

print actions
