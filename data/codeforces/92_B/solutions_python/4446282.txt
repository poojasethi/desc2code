n = list(raw_input())[::-1]

l = len(n)
index = 0
count = 0
# 1001001

while l != 1:
    if n[index] == '1':
        l -= 1
        i = index + 1
        while i < len(n) and n[i] == '1':
            n[i] = 0
            i += 1
        if i < len(n):
            n[i] = '1'
        else:
            n.append('1')
            l += 1
        count += 2
        index += 1
    else:
        l -= 1
        count += 1
        index += 1
