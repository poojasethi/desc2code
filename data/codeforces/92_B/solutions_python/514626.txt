a = raw_input()
a = list(a)
ln = len(a) -1
t = 0

while ln > 0 :
    if(a[ln] == '0'):
        t+=1
        ln-=1
    else:
        cnt = 0
        tmp = ln
        while tmp>=0 and a[tmp] == '1':
            tmp-=1
            cnt+=1
        ln = tmp;
        a[ln] ='1'
        t+= cnt + 1
    
print t
