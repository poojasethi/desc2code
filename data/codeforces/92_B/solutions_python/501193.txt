#!/usr/bin/env python
# -*- coding: utf-8 -*-

x = [ord(c)-ord('0') for c in raw_input()]
c = 0
while len(x) != 1 or x[0] != 1:
    if len(x) != 1:
        if x[-1] == 0:
            x.pop(len(x)-1)
            c += 1
        elif x[-1] == 1:
            x.pop(len(x)-1)
            x[-1] += 1
            c += 2
        else:
            x[-2] += x[-1] / 2
            x[-1] = x[-1] % 2
    else:
        if x[0] == 0:
            x[0] = 1
            c += 1
        else:
            x.insert(0, x[0] / 2)
            x[1] = x[1] % 2
print c
