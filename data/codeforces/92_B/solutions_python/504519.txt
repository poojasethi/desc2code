
s = raw_input()[::-1]
if list(s).count('1') == 1:
  print len(s) - 1
else:
  ans = 0
  carry = 0
  for i in xrange(len(s)):
    if s[i] == '0':
      now = 0
    else:
      now = 1

    now += carry
    carry = 0

    if now == 0:
      ans += 1
    elif now == 1:
      ans += 2
      carry = 1
    elif now == 2:
      ans += 1
      carry = 1

  print ans
