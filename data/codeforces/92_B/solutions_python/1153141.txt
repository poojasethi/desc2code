a = [2,1]
def solve():
    line = raw_input().strip()
    if line.count('1')==1:
        print len(line)-1
        return
    i=0
    total = 0
    flag = True
    while i< len(line):
        i+=1
        x = int(line[-i])
        if flag:
            total += a[x^1]
            if x == 1:
                flag = False
        else:
            total += a[x]
    print total

solve()
