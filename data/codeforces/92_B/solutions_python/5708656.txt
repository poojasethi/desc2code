import sys
from collections import deque

val = deque(sys.stdin.readline().strip())
answ = 0

while len(val) != 1 or val[0] != '1':
    #    print(val)
    e_o = val[-1]
    if e_o == '0':
        val.pop()
        answ += 1
    else:
        answ += 2
        val.pop()
        while val and val[-1] == '1':
            val.pop()
            answ += 1
        if val:
            val[-1] = '1'
        else:
            val = ['1']

print(answ)
