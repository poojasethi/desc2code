L = raw_input()

ans = 0
c = 0

for i in range(len(L) - 1, -1, -1):
    if i == 0:
        if c == 1:
            ans += 1
    else:
        if L[i] == '1':
            if c == 1:
                ans += 1
            else:
                c = 1
                ans += 2
        else:
            if c == 1:
                ans += 2
            else:
                ans += 1
print ans
