import sys

def fangy():
    n = map(int, list(sys.stdin.readline().strip()))
    n.reverse()
    size = len(n)
    k = 0
    retenue = 0
    for i in range(size -1):
        if (n[i] + retenue) != 1:
            k +=1
        else:
            retenue = 1
            k+=2
        # print ((n + retenue) %2)
        # print n, retenue, k
    print k+retenue

fangy()
