s = raw_input().strip()

flag = True
n = 0
for i in range(len(s) - 1, -1, -1):
    c = s[i]
    if flag:
        if c == "0":
            n += 1
        else:
            if i != 0:
                n += 2
                flag = False
    else:
        if c == "0":
            n += 2
        else:
            n += 1
print n
