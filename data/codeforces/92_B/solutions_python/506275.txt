import sys
import time

s = list(raw_input())[::-1]
ans = 0
i = 0
while True:
  while s[i] == '0':
    ans += 1
    i += 1

  if i + 1 == len(s):
    break

  j = i + 1
  carry = 1
  while carry > 0:
    if j == len(s):
      s.append('0')
    if s[j] == '1':
      s[j] = '0'
      carry = 1
    else:
      s[j] = '1'
      carry = 0
    j += 1
  ans += 2
  i += 1

print ans
