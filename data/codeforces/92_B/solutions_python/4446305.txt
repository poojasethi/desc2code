n = list(raw_input())
moves = 0
r = len(n) - 1

while r != 0:
  if n[r] == '0':
    moves += 1

  else:
    moves += 2
    x = r
    while 1:
      x -= 1
      if x == -1:
        x = 0
        moves += 1
        n.insert(0, '1')
        break

      if n[x] == '0':
        n[x] = '1'
        break
      else:
        n[x] = '0'
  r -= 1

print moves
