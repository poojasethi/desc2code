#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int n, b;   cin >> n >> b;
    int a[2000];
    for (int i=0; i<n; i++)
        cin >> a[i];
    int m = b;
    for (int i=0; i<n-1; i++)
        for (int j=i+1; j<n; j++)
            m = max(m, b/a[i] * a[j] + b%a[i]);
    cout << m;
    return 0;
}
