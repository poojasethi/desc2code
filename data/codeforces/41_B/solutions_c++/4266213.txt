#include <iostream>
#include <cstdio>
using namespace std;
int n,b,i,a[2001],s,ans,j,kans,m;
int main()
{
cin>>n>>b;
for (i=1;i<=n;i++)
cin>>a[i];
for (i=1;i<=n;i++)
{
	s=b/a[i];
	ans=b%a[i];
	for (j=i+1;j<=n;j++)
	{
	kans=ans;
	kans+=s*a[j];
		if (kans>m)
		m=kans;		
	}
}
cout<<max(m,b);



return 0;
}
