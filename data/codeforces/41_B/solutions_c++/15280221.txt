#include<stdio.h> 

int main()
{
	int n,b,a[2000];
	scanf("%d%d",&n,&b);
	int max=b;
	for(int i=0;i<n;++i)
		scanf("%d",&a[i]);
	for(int i=0;i<n-1;++i)
	{
		for(int j=i+1;j<n;++j)
		{
			max=max>b%a[i]+b/a[i]*a[j]?max:b%a[i]+b/a[i]*a[j];
		}
	}
	printf("%d\n",max);
	return 0;
}
