#include<iostream>
#include<string>
using namespace std;
int main()
{
int x,max,i,j,n,m,a[10000],o;
cin>>n>>m;
for(i=0;i<n;i++)
   cin>>a[i];
max=m;
for(i=0;i<n;i++)
   for(j=i+1;j<n;j++)
   {
      o=m%a[i];
      x=m/a[i];
      x=(x*a[j])+o;
      if(x>max)
         max=x;
}     
cout<<max<<endl;
return 0;
}
