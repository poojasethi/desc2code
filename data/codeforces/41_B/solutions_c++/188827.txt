#include<cstdio>

int main(){
  
  int n,b;
  scanf("%d%d",&n,&b);
  int cost[n];
  for(int i=0;i<n;i++)
    scanf("%d",&cost[i]);

  int tmp=0;
  int res=b;
  for(int i=0;i<n;i++)
    {
      for(int j=i+1;j<n;j++){

	tmp=cost[j]*(b/cost[i])+b%cost[i];
	res =(res>tmp)?res:tmp;
      }
      
    }
  printf("%d\n",res);
  return 0;
}
