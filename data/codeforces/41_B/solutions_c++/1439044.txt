#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int a[10000];
int n,b;
int mn;
int main(){
  mn=1;
  cin>>n>>b;
  mn=b;
  for(int i=0;i<n;i++)cin>>a[i];
  for(int i=0;i<n;i++){
    for(int j=i+1;j<n;j++){
      mn=max(mn,(b/a[i])*a[j]+b%a[i]);
    }
  }
  cout<<mn<<endl;
  
  return 0;
}
