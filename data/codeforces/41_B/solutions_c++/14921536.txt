#include<bits/stdc++.h>
using namespace std;

int n, b, x, res, a[2005];

int main(int argc, char **argv) {
	scanf("%d%d", &n, &b);
	for(int i=0 ; i<n ; ++i) scanf("%d", &a[i]);
	for(int i=0 ; i<n ; ++i)
		for(int j=i+1 ; j<n ; ++j)
			res = max(res, (b/a[i]*(a[j] - a[i])));
	printf("%d\n", b+res);
	return 0;
}
