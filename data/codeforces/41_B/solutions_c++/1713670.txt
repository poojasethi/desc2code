#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std ;

#define fo(i,a,b) for(int i=a;i<=b;i++)

int N , B , Ret ;
int A[2009] ;

int main() {
	cin >> N >> B ; Ret = B ;
	fo(i,1,N) cin >> A[i] ;
	fo(i,1,N) fo(j,i+1,N) Ret = max( Ret , B / A[i] * A[j] + B % A[i] ) ;
	cout << Ret << "\n" ;
}
