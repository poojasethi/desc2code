#include<cstdio>
#include<algorithm>
#include<functional>
using namespace std;

int main(){
	int b,n,mini=2005;
	scanf("%d %d",&n,&b);

	int ret=b;
	for(int i=0;i<n;++i){
		int a;
		scanf("%d",&a);
		if(a<mini){
			mini = a;
		}
		int cand = (b/mini)*a + (b%mini);
		ret = max(ret, cand);
	}
	printf("%d\n",ret);
	return 0;
}
