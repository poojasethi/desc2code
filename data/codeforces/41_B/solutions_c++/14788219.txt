#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,b,a[2016]={0},ma,aux,t1,t2;
	cin>>n>>b;
	for(int i=0;i<n;i++){
		cin>>a[i];
	}
	ma=b;
	aux=a[n-1];
	for(int i=n-2;i>=0;i--){
		if(a[i]>=aux){
			aux=a[i];
		}
		else{
			t1=b/a[i];
			t2=t1*aux+b%a[i];
			if(t2>ma)ma=t2;
		}
	}
	cout<<ma<<endl;
}
