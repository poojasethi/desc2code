#include <iostream>
#include <string>
#include <cstdio>

using namespace std;

int n,i,r,a[2000],j,m;
int main(){
    cin >> n >> r;
    for(i = 0; i < n; i++)cin >> a[i];
    m=r;
    for(i = 0; i < n - 1; i++)
        for(j = i + 1; j < n; j++)
            m = max(m,(r / a[i]) * a[j] + (r % a[i]));
    cout << m; 
}
