#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define MAX 2001

int n, a, b, ans, mn;

int main() {
	scanf("%d %d %d", &n, &b, &mn);
	ans = b;
	for (int i = 1; i < n; i++) {
		scanf("%d", &a);
		ans = max(ans, b%mn + b/mn*a);
		mn = min(mn, a);
	}
	printf("%d", ans);
	return 0;
}
