#include <iostream>
#include <cstdlib>
using namespace std;

int main() {
	int n, b, a[2048];
	int md = 0;
	
	cin >> n >> b;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	
	for (int i = 0; i < n-1; i++)
		for (int j = i+1; j < n; j++)
			md = max(md, (a[j] - a[i]) * (b / a[i]));
	
	cout << b + md << endl;	
		
	return 0;
}
