#include <stdio.h>
#include <algorithm>
using namespace std;

int main(void) {
	int a[2005];
	int n, b;
	scanf("%d%d", &n, &b);
	for(int i=0;i<n;i++)
		scanf("%d", &a[i]);
	int ans = b;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
			if(a[i] < a[j]) 
				ans = max(ans, b%a[i]+b/a[i]*a[j]);
	printf("%d\n", ans);
	return 0;
}
