#include <stdio.h>
#include <string.h>
int main()
{
	int n,i,b,mi,ma,a;
	while(scanf("%d%d",&n,&b)!=EOF)
	{
		scanf("%d",&mi);
		ma=0;
		for(i=2;i<=n;i++)
		{
			scanf("%d",&a);
			if(a<mi)  mi=a;
			if(b/mi*a+b%mi>ma)  ma=b/mi*a+b%mi;
		}
		if(ma>b) b=ma;
		printf("%d\n",b);
	}
	return 0;
}
