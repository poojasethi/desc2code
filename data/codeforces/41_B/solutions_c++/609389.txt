#include<iostream>
using namespace std;

const int maxn = 1000*3;
int p[maxn];
int n,b;

int main()
{
	cin >> n >> b;
	for(int i=0;i<n;i++)
		cin >> p[i];
	int ans = b;
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
		{
			int x = b/p[i];
			int y = b%p[i];
			if(x*(p[j]) + y > ans)
				ans = x*p[j] + y;
		}
	cout << ans << endl;
	return 0;
}
