#include <cstdio>
#include <algorithm>

int main() {
  int b, n;
  scanf("%d %d", &n, &b);

  int a[n];
  for(int i = 0; i < n; i++)
    scanf("%d", &a[i]);

  int ans = b;
  for(int i = 0; i < n; i++)
    for(int j = i+1; j < n; j++)
      ans = std::max(b%a[i] + b/a[i] * a[j], ans);

  printf("%d\n", ans);
}
