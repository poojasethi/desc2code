#include <iostream>

using namespace std;

const int MAXN=2000+10;
int n,m;
int a[MAXN];

int main(){
	cin >> n >> m;
	for(int i=1;i<=n;i++)
		cin >> a[i];
	int ans=m;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			ans=max(ans,m%a[i]+(m/a[i])*a[j]);;
	cout << ans << endl;
}
