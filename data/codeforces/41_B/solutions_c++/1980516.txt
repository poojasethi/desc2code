#include<stdio.h>
int x[999999];
main()
{
    int n,b,c,max=0,i;
    scanf("%d%d",&n,&b);
    for(c=0;c<n;c++)
    scanf("%d",&x[c]);
    for(c=0;c<=n-1;c++)
    {
       for(i=c;i<=n;i++)
       {
           if((b/x[c]*x[i]+b%x[c])>max)
           max=(b/x[c]*x[i]+b%x[c]);
       }
    }
    printf("%d",max);

}
