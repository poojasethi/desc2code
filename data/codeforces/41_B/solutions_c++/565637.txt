#include <iostream>

using namespace std;

int main(int argc, char* argv[])
{
	int n, b;
	cin >> n >> b;
		
	int a[2000], m = b;
		
	for (int i = 0; i < n; i++)
	{		
		cin >> a[i];		
		
		for (int j = 0; j < i; j++)
			m = max(m, (b / a[j]) * a[i] + b % a[j]);
	}
	
	cout << m << endl;
	
	return 0;
}
