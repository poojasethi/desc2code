#include<iostream>
using namespace std;

int main()
{
	long long n,b,a[2000];
	cin >> n >> b;
	int max=b;
	for(int i=0;i<n;i++)
	{
		cin >> a[i];
	}
	for(int i=0;i<n-1;i++)
	{
		for(int j=i+1;j<n;j++)
		{
			max=max>b%a[i]+b/a[i]*a[j]?max:b%a[i]+b/a[i]*a[j];
		}
	}
	cout << max << endl;
	return 0;
}
