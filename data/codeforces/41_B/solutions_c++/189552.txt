#include <iostream>
using namespace std;
int i,j,n,b,bb,x[10000],l;

int main()
{

    cin>>n>>b;
    for(i=1;i<=n;i++)cin>>x[i];

    bb=b;

    for(i=1;i<=n;i++)
        for(j=i+1;j<=n;j++)
        {
            l=b/x[i];
            if(b%x[i]+l*x[j]>bb)bb=b%x[i]+l*x[j];
        }
    cout<<bb<<endl;
};
