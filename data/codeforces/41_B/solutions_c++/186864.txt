#include<stdio.h>
#include<stdlib.h>

int main(){
	int i,j,k,n,b,a[2001],l,f,max;
	scanf("%d%d",&n,&b);
	max=b;
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	for(i=0;i<n-1;i++){
		k=b/a[i];l=b%a[i];
		for(j=i+1;j<n;j++){
			f=k*a[j]+l;
			if(f>max)
				max=f;
		}
	}
	printf("%d\n",max);
	

	return 0;
}
