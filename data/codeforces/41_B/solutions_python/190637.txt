ini = int(nums[1])
instr = raw_input()
nums = instr.split()
num = []
for idx in range(day):
    num.append(int(nums[idx]))
ans = ini
for idxi in range(day):
    for idxj in range(idxi):
        if num[idxj] < num[idxi]:
            ans = max(ans, ini % num[idxj] + ini / num[idxj] * num[idxi])
print ans
