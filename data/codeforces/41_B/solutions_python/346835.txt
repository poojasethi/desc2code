import sys

def reverse(s):
    chars = [i for i in s]
    chars.reverse()
    return "".join(chars)

lines = [i.rstrip() for i in sys.stdin.readlines()]
b = int(lines[0].split(" ")[1])
nums = [int(i) for i in lines[1].split(" ")]
ans = 0
for i in range(1, len(nums)):
    t1 = min(nums[:i])
    t2 = max(nums[i:])
    if (t1 < t2) and (b >= t1):
        d = b / t1
        ans = max(ans, t2 * d + b - t1 * d)
if ans == 0:
    ans = b
    
print ans
