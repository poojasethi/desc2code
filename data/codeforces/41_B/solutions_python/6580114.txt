
def run():
    n, d = map(int, raw_input().split());
    A = map(int, raw_input().split());

    respuesta = d;
    minimo = 5000;
    maximo = -5000;

    for a in A:
        if a < minimo:
            minimo = a;
            maximo = a;
        if a > maximo:
            maximo = a;
        respuesta = max(respuesta, (d / minimo) * maximo + d % minimo);
    print respuesta;


if __name__ == "__main__":
    run();
