import time
import sys, io

#start = time.clock()
(n,b)=map(int, raw_input().split())
l=[int(x) for x in raw_input().split()]
ans=b
for i in range(n-1):
    start = b
    kari = start/l[i]
    start -= (kari*l[i])
    ans=max(ans,start+(kari*max(l[i+1:])))
print ans
