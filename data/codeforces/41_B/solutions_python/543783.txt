s = raw_input().split()

p = []
q = []
for x in s:
	p.append(int(x))
	q.append(int(x))

for i in range(1,len(p)):
	if p[i-1] < p[i]:
		p[i] = p[i-1]
	if q[-i] > q[-i-1]:
		q[-i-1] = q[-i]
ans = b
for i in range(1,len(p)):
	ans = max(ans, b%p[i-1]+(b/p[i-1]*q[i]))
print ans
