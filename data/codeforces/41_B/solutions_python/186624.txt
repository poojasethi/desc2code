from sys import stdin
#x = [ '4 10', '4 2 3 1' ]

x = stdin.readlines()
nb, ai = map( lambda x: x. strip(), [ l for l in x ] )
n, b = map(int, nb.split(' ') )
ai = map(int, ai.split(' ') )[::-1]
maxs = [0] * len(ai)
mx = 0
for i in range( len(ai) ) :
    if ai[i] > mx :
        mx = ai[i]
    maxs[i] = mx

ai = ai[::-1]
maxs = maxs[::-1]
maxr = b

for a, s in zip( ai, maxs ) :
    r = ( b / a ) * s + b - ( b / a ) * a
    if maxr < r :
        maxr = r

print maxr
