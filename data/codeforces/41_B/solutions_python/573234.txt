import sys
f = sys.stdin
lines = f.readlines();
n,b = map(int, lines[0].split());
ar = map(int, lines[1].split());
mx = 0
for i in xrange(n):
    for u in xrange(i,n):
        mn = ar[i];
        sell = b // mn;
        lft = b - b // mn * mn;
        lft += sell * ar[u];
        mx = max(mx, lft)
        
print mx
