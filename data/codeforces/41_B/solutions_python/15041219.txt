from math import *
from Queue import *
from sys import *
from random import *






n, b = map(int, raw_input().split())
a = map(int, raw_input().split())
m = [0 for i in range(n)]
m[-1] = a[-1]
for i in range(n-2, -1, -1):
    m[i] = max(m[i+1], a[i])
res = b
for i in range(n-1):
    if m[i+1] > a[i]:
        res = max(res, b%a[i] + (b/a[i])*m[i+1])
print(res)
