n, b = raw_input().split()
n, b = int(n), int(b)

a = [int(x) for x in raw_input().split()]
  
curmax = 0
maxi = []
for x in reversed(a):
  curmax = max(curmax, x)
  maxi.append(curmax)
  
maxi.reverse()
maxi.append(0)

best = b
for i, x in enumerate(a):
  buy = b / x
  res = b % x
  res += buy * maxi[i + 1]
  best = max(best, res)
  
print best
