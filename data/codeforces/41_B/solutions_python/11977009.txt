case=raw_input()
prices = raw_input()
n,b=case.split()
n=int(n)
b=int(b)
ans=0
price=prices.split()
price = [int(x) for x in price]
for i in range(n):
    buy=b/price[i]
    money=b-buy*price[i]
    for j in range(i,n):
        sell=buy*price[j]+money
        if sell>ans:
            ans=sell
print ans
