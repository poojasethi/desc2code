#include<stdio.h>
#include<algorithm>
#include<utility>
using namespace std;
int n,m,la,lb;
pair<int,int> a[1010],b[1010];
int main(){
	int x,y;
	long long int ans=0;
	scanf("%d%d",&n,&m);
	for(int i=0; i<n; i++){
		scanf("%d%d",&x,&y);
		if(y==1)
			a[la++]=make_pair(x,i+1);
		else
			b[lb++]=make_pair(x,i+1);
		ans+=x*2LL;
	}
	sort(a,a+la);
	sort(b,b+lb);
	if(la>=m){
		for(int i=la-1; i>la-m; i--)
			ans-=a[i].first;
		ans-=min(a[0].first,lb?b[0].first:1000000000);
		printf("%I64d.%d\n",ans/2,ans%2?5:0);
		for(int i=la-1; i>la-m; i--)
			printf("1 %d\n",a[i].second);
		printf("%d",la-m+1+lb);
		for(int i=la-m; i>=0; i--)
			printf(" %d",a[i].second);
		for(int i=0; i<lb; i++)
			printf(" %d",b[i].second);
		puts("");
	}else{
		for(int i=la-1; i>=0; i--)
			ans-=a[i].first;
		printf("%I64d.%d\n",ans/2,ans%2?5:0);
		for(int i=la-1; i>=0; i--)
			printf("1 %d\n",a[i].second);
		m-=la;

		for(int i=0; i<m-1; i++)
			printf("1 %d\n",b[i].second);

		printf("%d",lb-(m-1));
		for(int i=m-1; i<lb; i++)
			printf(" %d",b[i].second);
		puts("");
	}
	return 0;
}
