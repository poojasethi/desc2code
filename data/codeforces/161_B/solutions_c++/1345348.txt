#include <cstdio>
#include <iostream>
#include <vector>
#include <algorithm>

#define fi first
#define se second
#define SIZE(a) ((int)a.size())
#define pb push_back
using namespace std;

typedef pair<int,int> ii;
typedef pair<ii,int> iii;
typedef long long LL;

vector<ii> ar[3];

const int MAXN = 1010;

vector<iii> pack[MAXN];

int main(){
	
	int n,k,a,b;
	double res=0;
	
	scanf("%d %d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%d %d",&a,&b);
		ar[b].pb(ii(a,i+1));
		res+=a;
	}
	
	sort(ar[1].begin(),ar[1].end());
	reverse(ar[1].begin(),ar[1].end());
	sort(ar[2].begin(),ar[2].end());
	reverse(ar[2].begin(),ar[2].end());
	for(int i=0;i<SIZE(ar[1]);i++)
		pack[min(i,k-1)].pb(iii(ar[1][i],1));
	for(int i=0;i<SIZE(ar[2]);i++)
		pack[min(i+SIZE(ar[1]),k-1)].pb(iii(ar[2][i],2));
	double t;
	for(int i=0;i<k;i++)
	{
		t=1e10;
		int fl=0;
		for(int j=0;j<SIZE(pack[i]);j++)
		{
			t=min(t,(double)pack[i][j].fi.fi);
			if(pack[i][j].se==1)
				fl=1;
		}
		if(fl)
			res-=t/2.0;
	}
	printf("%.1lf\n",res);
	for(int i=0;i<k;i++)
	{
		printf("%d",SIZE(pack[i]));
		for(int j=0;j<SIZE(pack[i]);j++)
			printf(" %d",pack[i][j].fi.se);
		puts("");
	}
	return 0;
	
}
