f = open('input.txt','r')
out = open('output.txt','w')
r = lambda: f.readline().strip()

n,k = map(int,r().split())
jars = map(int,r().split())

piglet = 0
loops = 0
while loops < 3:
    piglet += sum(j for j in jars if j<k)
    jars = [j-k for j in jars if j>=k]
    loops += 1
piglet += sum(jars)
out.write(str(piglet))
