#!env python

with open('input.txt') as f:
    n, k = f.readline().split(' ')
    n = int(n)
    k = int(k)
    pots = f.readline().split(' ')

for i in xrange(len(pots)):
    pots[i] = int(pots[i])

for i in xrange(len(pots)):
    for j in xrange(3):
        if pots[i] < k:
            break
        else:
            pots[i] -= k

with open('output.txt', 'w') as f:
    f.write(str(sum(pots)))
