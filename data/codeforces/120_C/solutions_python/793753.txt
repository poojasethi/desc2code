import sys

sys.stdin = open("input.txt", "r")
sys.stdout = open("output.txt", "w")

n, k = map(int, sys.stdin.readline().split())
a = map(int, sys.stdin.readline().split())

sum = 0
for item in a:
	if item - 3 * k > 0:
		sum += item - 3 * k
	else:
		sum += item % k

print sum
