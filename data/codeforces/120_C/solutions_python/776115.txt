f=open("input.txt", 'r')
l=open("output.txt", 'w')
a=[]
b=[]
a=f.readline().split()
b=f.readline().split()
summ=0
for i in range(int(a[0])):
    if int(b[i])>=(3*int(a[1])):
        summ+=int(b[i])-(3*int(a[1]))
    elif int(b[i])>=(2*int(a[1])):
        summ+=int(b[i])-(2*int(a[1]))
    elif int(b[i])>=int(a[1]):
        summ+=int(b[i])-int(a[1])
    else:
        summ+=int(b[i])
l.write(str(summ))
f.close()
l.close()
