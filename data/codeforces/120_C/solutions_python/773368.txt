source = open('input.txt', 'r')
result = open('output.txt', 'w')
n, k = [int(i) for i in source.readline().split()]
honey = [[int(i), 0] for i in source.readline().split()]
honey.sort()
piglet = 0
for i in range(n - 1, -1, -1):
	while honey[i][0] >= k and honey[i][1] < 3:
		honey[i][0] -= k
		honey[i][1] += 1
	piglet += honey[i][0]
result.write(str(piglet))
source.close()
result.close()
