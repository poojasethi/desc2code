filein = open("input.txt","r")
line1 = filein.readline().rstrip()
line2 = filein.readline().rstrip()
ans = ""
n,k = line1.split()
n = int(n)
k = int(k)
sev = line2.split()
ans = 0
for i in range(len(sev)):
    num = int(sev[i])
    if num > 3*k:
        num -= 3*k
    else:
        num %= k
    ans += num
print ans
fileout = open("output.txt","w")
fileout.write(str(ans))
fileout.flush()
