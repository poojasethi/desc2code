import sys
f1 = open('input.txt', 'r')
f2 = open('output.txt', 'w')
sys.stdin = f1
sys.stdout = f2

n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
b = [0] * n

ans = 0
while sum(a) != 0:
    m = max(a)
    i = a.index(m)
    if m < k:
        ans += m
        a[i] = 0
    elif b[i] == 3:
       ans += a[i]
       a[i] = 0
    else:
        b[i] += 1
        a[i] -= k
print ans
