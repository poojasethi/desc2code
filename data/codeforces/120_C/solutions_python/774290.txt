fi = open("input.txt", 'r')
fo = open("output.txt", 'w')
def main():
	[n,k] = [int(x) for x in fi.readline().split()]
	m = [int(x) for x in fi.readline().split()]
	m.sort()
	sum = 0
	for i in xrange(len(m)-1,-1,-1):
		if m[i] - k*3 > 0:
			sum += m[i] - k*3
		else:
			sum += m[i] % k
	return sum		
	
fo.write(str(main()))
