#include<iostream>
#include<fstream>
using namespace std;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	long long m,n,con=0;
	cin>>m>>n;
	for(int i=0;i<m;i++)
	{
		long long a,b;
		cin>>a;
		b=a/n;
		if(b>=3)
			con+=(a-(n*3));
		else
			con+=(a-(n*b));
	}	
	cout<<con<<endl;
	

}
