#include <bits/stdc++.h>

using namespace std;

const int N = 1e5 + 9;

int n, k, ans, i, a;

int main() {

	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	cin >> n >> k;

	while(n--) {
		cin >> a;
		ans += (a >= 3 * k) ? (a - 3 * k) : (a % k);
	}

	cout << ans << endl;

	return 0;
}
