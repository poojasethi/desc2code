#include <iostream>
#include <fstream>

using namespace std;

int main () {
	ifstream fin ("input.txt");
	ofstream fout ("output.txt");

	int n, k; fin >> n >> k;
	
	int honeyLeft = 0, aux;
	for (int i = 0; i < n; i++) {
		fin >> aux;
		honeyLeft += (max (aux - 3*k, aux%k));
	}
	fout << honeyLeft <<  endl;
	return 0;
}
