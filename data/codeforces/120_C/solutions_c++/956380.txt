#include <iostream>
#include <stdio.h>

using namespace std;
int main()
 {int a[1002];
int n,k,ans=0;
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>n>>k;
for (int i=0; i<n; i++)
{
    cin>>a[i];
    if (a[i]>=k*3) ans+=a[i]-k*3;
    else ans+=a[i]%k;
}
cout<<ans;
return 0;
}
