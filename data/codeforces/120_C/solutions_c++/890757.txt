#include <fstream>
#include <iostream>
using namespace std;

ifstream fin ("input.txt");
ofstream fout ("output.txt");

int N,K;
int sum;

int main(){
  fin >> N >> K;
  int a,b;
  for(int i=0;i<N;i++){
    fin >> a;
    b=min(3,a/K);
    a-=b*K;
    sum+=a;
  }
  fout << sum << endl;
  return 0;
}
