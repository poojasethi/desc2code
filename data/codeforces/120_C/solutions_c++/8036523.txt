
#include<iostream>
#include<stdio.h>
using namespace std;
int n,k,x, ans=0;
int main(){
    freopen ("input.txt", "r", stdin) ;
    freopen ("output.txt", "w", stdout);
    cin>>n>>k;
    while(n--){
        cin>>x;
        if(x>=3*k) ans+=x-3*k;
        else   ans+=x%k;
    }
    cout<<ans<<endl;
}
