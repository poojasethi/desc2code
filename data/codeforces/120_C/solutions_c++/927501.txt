#include <iostream>
#include <fstream>

using namespace std;
ifstream fin ("input.txt");
ofstream fout ("output.txt");
int main()
{
    int n,k,sum=0,jar;
    fin>>n>>k;
    while(n--)
    {
        fin>>jar;
        if(jar>=3*k) sum+=jar-3*k;
        else sum+=(jar%k);
    }
    fout<<sum;
return 0;
}
