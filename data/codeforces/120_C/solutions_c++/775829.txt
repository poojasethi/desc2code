#include<fstream>
#include<iostream>
using namespace std;

int main() {
	ifstream fin("input.txt");
	ofstream fout("output.txt");

	int n,k; fin>>n>>k;
	int ans=0;
	while(n--) {
		int curr; fin>>curr;
		for (int i=0; i<3; i++) {
			if(curr<k) break;
			curr-=k;
		}
		ans+=curr;
	}
	fout << ans << endl;
	return 0;
}
