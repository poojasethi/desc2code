#include <iostream>
#include <vector>
using namespace std;


int main () {
	int n,k,c, temp, res=0;
	vector<int> hols(n, 0);
	cin >> n >> k >> c;
	for (int i=0; i<c; i++) {
		cin >> temp;
		hols[temp]=1;
	}
	int timer=0;
	for (int i=1; i<=n; i++){
		timer++;
		if (timer == k || hols[i] == 1){
			res++;
			timer=0;
		}
		
	}
	cout << res << endl;
}
