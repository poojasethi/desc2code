#include <iostream>

using namespace std;

int main()
{

    int answer=0,i,n,c,k,x,lx=0;

    cin >> n >> k >> c;

    for(i=0;i<c;i++)
    {
        cin >> x;
        answer++;
        answer+=(x-lx-1)/k;
        lx = x;
    }

    answer += (n-lx)/k;



    cout << answer << endl;

    return 0;
}
