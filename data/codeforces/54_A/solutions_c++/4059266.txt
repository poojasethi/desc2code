#include<cstdio>
int main()
{
    int a[400],n,k,x;
    scanf("%d %d",&n,&k);
    scanf("%d",&x);
    for (int i=1; i<=x; i++) scanf("%d",&a[i]);
    int now=0,ans=-1,hol=1;
    while (now<=n)
    {
        ans++;
        while (a[hol]<=now+k && hol<=x)
            ans++,now=a[hol],hol++;
        now=now+k;
    } printf("%d",ans);
    return 0;
}
