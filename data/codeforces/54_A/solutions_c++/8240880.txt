#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    int n, k, c, x=0, x2=0, res = 0;
    scanf("%d%d%d", &n, &k, &c);
    for (int i = 0; i < c; i++)
    {
        scanf("%d", &x);
        res += (x - x2 - 1) / k;
        x2 = x;
    }
    res += (n - x)/k;
    printf("%d", res + c);
}
