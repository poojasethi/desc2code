#include<iostream>
using namespace std;
long long int n,k,c,d[1111],a[1111],dem,vem;
int main(){
	cin>>n>>k;
	cin>>c;
	for(int i=1; i<=c; i++){
	cin>>d[i];
	}
	for(int i=1; i<=c; i++){
		a[d[i]]++;
	}
	for(int i=1; i<=n; i++){
		vem++;
		if(vem==k || a[i]>0){
		vem=0;
		dem++;
		}
	}
	cout<<dem;	
	return 0;
}
