#include <cstdio>
#include <cstdlib>
#include <cstring>
void init(){   
int N, K, C, aa = 0, bb, ans = 0;  
scanf("%d%d%d", &N, &K, &C);  
for (int i = 1; i <= C; ++i)    {    
++ans;        
scanf("%d", &bb);   
ans += (bb - 1 - aa) / K;      
aa = bb;   
}   
ans += (N - aa) / K; 
printf("%d\n", ans);
}
int main(){ 
init();    
return 0;
}
