#include <iostream>
using namespace std;
int main()
{
    unsigned long long sum=0;
    int n,k,hol,temp,prev=0;
    cin>>n>>k>>hol;
    while(hol--)
    {
        cin>>temp;
        prev=temp-1-prev;
        sum+=prev/k+1;
        prev=temp;
    }
    prev=n-prev;
    sum+=prev/k;
    cout<<sum;
    return 0;
}
