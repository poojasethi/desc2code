#include<iostream>
using namespace std;
int res = 0, t, n, k, c, i, x;
int main(){
    cin>>n>>k;
    cin>>c;
    t = 0;
    for(i=0;i<c;i++){
      cin>>x;
      while (t + k < x) {res++; t+=k;}
      t = x;
      res++;
    }
    while (t + k <= n){res++;t+=k;}
    cout<<res;
    //system("pause");
}
