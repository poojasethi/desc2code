#include<cstdio>
#include<cstring>
using namespace std;
const int maxn=366;
bool v[maxn];
int n,k,c;

int main(){
  int i,j;
  scanf("%d%d%d",&n,&k,&c);
  while (c--){
    scanf("%d",&i);
    v[i]=1;
  }
  j=0;
  for (i=1;i<=n;i++){
    j++;
    if (v[i] || j==k){
      j=0;
      c++;
    }
  }
  printf("%d",c+1);
  return 0;
}
