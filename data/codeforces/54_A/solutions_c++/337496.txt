#include <iostream>
using namespace std;

int main() {
	int n, k;
	cin >> n >> k;

	int c;
	cin >> c;
	c++;

	int t[367];
	t[0] = 0;
	for (int i = 1; i < c; i++) {
		cin >> t[i];
	}
	t[c++] = n;

	int tot = 0;
	for (int i = 0; i < c-1; i++) {
		int dt = t[i+1] - t[i];
		tot += dt/k + (i != c-2 && dt%k != 0);
	}

	cout << tot << endl;
	return 0;
}
