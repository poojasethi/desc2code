#include<iostream>
using namespace std;

int main(){
  int n,k,c,g[1000],cnt=0,i,j,now,m;
  cin>>n>>k>>c;
  for(i=0;i<366;i++)g[i]=0;
  for(i=0;i<c;i++){
    cin>>m;
    g[m]=1;
 }
  now=0;
  for(i=1;i<=n;i++){
    now++;
 
    if(g[i]){now=0;cnt++;}
    if(now>=k){now=0;cnt++;}
}
  cout<<cnt<<endl;
  return 0;
  
}
