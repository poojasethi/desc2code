#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int n,k,m,u[10000];

int main()
{
	cin>>n>>k;
	cin>>m;
	for (int i = 0; i < m; ++i) {
		int x;
		cin>>x;
		u[x] = true;
	}
	
	int prev = 0, cnt =0;
	for (int i = 1; i <= n; i++)
	if (i - prev == k || u[i]) {
		cnt++;
		prev = i;
	}
	cout<<cnt<<endl;
	return 0;
}
