#include "iostream"
#include "algorithm"

using namespace std;

int main(int argc, char const *argv[])
{
	int n , k , c ;
	int h[ 400 ];
	h[ 0 ] = 0;
	cin >> n >> k;
	cin >> c;
	int ans = 0;
	for (int i = 0; i < c; ++i)
	{
		cin >> h[ i + 1 ];
		ans += ((h[ i + 1 ] - h[ i ] - 1)/k) + 1;
	}
	ans += (n - h[ c ])/k;
	cout << ans << endl;
	return 0;
}
