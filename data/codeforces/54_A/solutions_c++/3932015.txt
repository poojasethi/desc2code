#include<stdio.h>
int n,k,c,arr[400];
int main()
{
      int i,ix=0,cnt=0;
      scanf("%d %d %d",&n,&k,&c);
      arr[ix++]=0;
      for(i=0;i<c;i++) scanf(" %d",&arr[ix++]);
      arr[ix]=n+1;
      for(i=0;i<ix;i++)
      {
            cnt+=(arr[i+1]-arr[i]-1)/k;
      }
      printf("%d\n",cnt+c);
      return 0;
}
