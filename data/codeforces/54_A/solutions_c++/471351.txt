#include <stdio.h>

int main(void) {
	int n, k, a[366]={}, cnt=0, i, j, x, c;
	scanf("%d%d", &n, &k);
	scanf("%d", &c);
	for(i=0;i<c;i++) {
		scanf("%d", &x);
		a[x] = 1;
		++cnt;
	}
	a[0]=1;
	for(i=k;i<=n;i++) {
		for(j=i-1;j>=i-k+1;j--)
			if (a[j]==1) break;
		if(j<i-k+1 && a[i]==0) {
			a[i] = 1;
			++cnt;
		}
	}
	printf("%d\n", cnt);
	return 0;
}
