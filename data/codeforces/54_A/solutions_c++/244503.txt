#include <iostream>
#include <vector>

using namespace std;

int main() {
  int n, k, c;
  cin >> n >> k >> c;
  vector<int> h(n+1);
  for (int i=0; i < c; i++) {
    int d;
    cin >> d;
    h[d] = 1;
  }

  int last = 0;
  int res = 0;
  for (int i=1; i <= n; i++)
    if (h[i] || i - last >= k) {
      res++;
      last = i;
    }

  cout << res << endl;
}
