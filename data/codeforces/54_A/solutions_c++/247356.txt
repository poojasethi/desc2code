#include<iostream>
using namespace std;

int n,k,c,ans,holiday[370],i,j,pre;
int main()
{
	cin>>n>>k>>c;
	for(i=0;i<c;i++)cin>>holiday[i];
	ans=0;
	pre=k;
	j=0;
	while(pre <= n)
	{
		if(j<c && holiday[j]<=pre)
		{
			ans++;
			pre=holiday[j]+k;
			j++;
			continue;
		}
		ans++;
		pre+=k;
	}
	ans+=c-j;
	cout<<ans;



}
