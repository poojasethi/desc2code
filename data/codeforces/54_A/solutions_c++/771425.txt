#include<iostream>
#include<algorithm>
using namespace std;
int a[366]={0};
int main()
{
	int n,k,c;
	cin>>n>>k;
	cin>>c;
	int s;
	while(c--)
	{
		cin>>s;
		a[s]=1;
	}
	int res=0;
	int ct=0;
	for(int j=1;j<=n;j++)
	{
		if(a[j]==1 ||j-res==k)
		{
			res=j;
			ct++;
		}		
	}
	cout<<ct<<"\n";
	return 0;
}
