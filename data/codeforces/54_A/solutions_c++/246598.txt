#include <iostream>
#include <cmath>
using namespace std;
int main(){
  int n,k,c;
  cin >> n >> k;
  cin >> c;
  int ans=0,last=0,l;
  for(int i=0;i<c;i++){
    cin >> l;
    ans += (l-last)/k;
    if((l-last) % k) ans++;
    last = l;
  };
  ans += (n-last) /k;
  cout << ans << endl;
  return 0;
}
