#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>

using namespace std;

int n, k, a, b, c, wynik;

int main()
{
	scanf("%d %d %d", &n, &k, &c);
	for(int i=0; i<c; i++)
	{
	    scanf("%d", &b);
	    wynik += (b-a-1)/k + 1;
	    a = b;
	}

	wynik += (n-b)/k;

	printf("%d\n", wynik);

	return 0;
}
