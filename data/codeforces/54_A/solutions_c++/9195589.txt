#include<iostream>
using namespace std;
int main()
{
	int n,k,c;
	cin>>n>>k>>c;
	int a[c+1];
	for(int i=1;i<=c;i++)
	{
		cin>>a[i];
	}
	a[c+1]=366;
	int gift=0;
	int i=k;
	int j=1;
	while((i<=n))
	{
		if(i>=a[j])
		{
			i=a[j];
			j++;
		}
		gift++;
		i=i+k;
		if((i>n)&&(a[j]<=n))
		{
		i=a[j];
	    }
	}
	cout<<gift;
	return 0;
}
