#include <iostream>
using namespace std;

int j,n,i,a[1001],s,c,k,t[1001],z;

int main(){
 	cin>>n>>k>>c;
	for(i=0;i<c;++i) cin>>a[i];
	i=0;
	while(i<=n){
		if (i>0)
		if (j<c && i>=a[j] && !t[j]){ s++; i=a[j]; t[j]=1; j++;}
		else s++;
		i+=k;
		if (i>n && j<c) s+=(c-j);
	}
	cout<<s;
	return 0;

}
