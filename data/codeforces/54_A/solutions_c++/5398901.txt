#include <iostream>
using namespace std;
bool a[400];
int main()
{
    int p,n,k,kol=0,c;
    cin>>n>>k>>c;
    for(int i=0;i<c;i++)
    {
        cin>>p;
        a[p]=1;
    }
    p=0;
    for(int i=1;i<=n;i++)
    {
        if((a[i])||(p+k==i))
        {
            kol++;
            p=i;
        }
    }
    cout<<kol;
    return 0;
}
