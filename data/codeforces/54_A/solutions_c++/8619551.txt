#include<stdio.h>
int n,k,c,num;
int a[400];
int main()
{
	scanf("%d%d",&n,&k);
	scanf("%d",&c);
	for(int i=1;i<=c;i++)
	scanf("%d",&a[i]);
	num=0;
	for(int i=1;i<=c;i++)
	{
	    num+=(a[i]-a[i-1])/k;
		if((a[i]-a[i-1])%k)
		num++;
	}
	num+=(n-a[c])/k;
	printf("%d\n",num);
	return 0;
}
