# -*- coding: utf-8 -*-
from __future__ import division

def readint():
    return map(int, raw_input().split(' '))
    
    
n,k = readint()
holidays = set(readint()[1:])
dwp = 0

pcnt = 0

for i in range(n):
    if (i+1) in holidays:
        pcnt += 1
        dwp = 0
    elif dwp == k-1:
        pcnt += 1
        dwp = 0
    else:
        dwp += 1

print pcnt
