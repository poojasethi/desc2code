N , K = eval("[" + raw_input().replace(' ' , ',') + "]")
Holiday = eval("[" + raw_input().replace(' ' , ',') + "]")

isHoliday = [False for i in range(0 , N + 2)]
last = 0
answer = 0

for i in range(1 , Holiday[0] + 1):
    isHoliday[Holiday[i]] = True
    
for i in range(1 , N + 1):
    if isHoliday[i] == True or i == last + K:
        answer += 1
        last = i
        
print(answer)
