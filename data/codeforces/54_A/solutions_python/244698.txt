import sys

n, k = map(int, sys.stdin.readline().split())
l = map(int, sys.stdin.readline().split())
c = l[0]
holidays = l[1:]

t = [0 for i in xrange(400)]

for i in xrange(c):
    t[holidays[i]] = 1
t[0] = 1

result = c
last = 0;
for i in xrange(1, n+1):
    if t[i] == 1:
        last = i
    else:
        if (i-last == k):
            last = i
            result += 1

print result
