N, K = map(int, raw_input().split())
holidays = map(int, raw_input().split())[1:]

days = 0
presents = 0
for i in xrange(1, N+1):
    days += 1
    if i in holidays:
        presents += 1
        days = 0
    if days == K:
        presents += 1
        days = 0
