#!/usr/bin/env python
from string import split

# Input
N, K = map(int, split(raw_input()))
C = map(int, split(raw_input()))
hd = C[1:]
C = C[0]

# Mark days
days = [False] * (N + 1)
for i in range(C):
	days[hd[i]] = True

row = 0
r = 0
for i in range(1, N + 1):
	if days[i] == True:
		row = 0
		r += 1
	else:
		row += 1
		if row >= K:
			row = 0
			r += 1
print r
