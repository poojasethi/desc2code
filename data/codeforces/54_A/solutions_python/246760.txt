N, K = map(int, raw_input().split())
mas = map(int, raw_input().split())
C = mas.pop(0)
next_day = K
sum = 0
for i in xrange(N + 1):
    if i == next_day:
        sum += 1
        next_day += K
        continue
    if i in mas:
        sum += 1
        next_day = i + K
        continue
