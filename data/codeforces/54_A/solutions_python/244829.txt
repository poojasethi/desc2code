
( N, K ) = [ int( k ) for k in raw_input().split() ]
hdays = raw_input().split()
C = int( hdays[ 0 ] )
hdays = set( [ int( k ) for k in hdays[ 1 : ] ] )
# print N, K, C, hdays

l = 0
gifts = 0
for d in xrange( 1, N + 1 ) :
	if d in hdays :
		gifts += 1
		l = d
	else :
		if d - l >= K :
			gifts += 1
			l = d
print gifts
