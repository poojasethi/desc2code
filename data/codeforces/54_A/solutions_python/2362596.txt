#!/usr/bin/python
import math

N, K = [int(x) for x in raw_input().split()]
holidays = [int(x) for x in raw_input().split()]
C = holidays[0]
holidays[0] = 0

result = C
for i in range(1, len(holidays)):
	result += int(math.ceil(float(holidays[i] - holidays[i-1]) / K)) - 1

print result + (N - holidays[-1]) // K
