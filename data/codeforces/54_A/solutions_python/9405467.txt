N,K = map(int, raw_input().split())
holidays = map(int, raw_input().split())
numHoliday = holidays[0]
nextHoliday = 1
prevDay = 0
numPresents = 0

for day in range(N + 1):
    if nextHoliday <= numHoliday and holidays[nextHoliday] == day:
        nextHoliday += 1
        numPresents += 1
        prevDay = day
    elif day - prevDay >= K:
        numPresents += 1
        prevDay = day

print numPresents
