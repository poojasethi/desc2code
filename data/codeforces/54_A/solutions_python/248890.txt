import sys
N, K = map(int,sys.stdin.readline().split(' '))
LC   = map(int, sys.stdin.readline().split(' '))
C    = LC[0]
H    = LC[1:]
ish = [False]*(N+1)
for h in H:
    ish[h] = True

last = 0
res = 0
for i in range(1,N+1):
    if (i-last)==K or ish[i]:
        res += 1
        last = i
print res
