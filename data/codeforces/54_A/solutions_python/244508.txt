from sys import stdin

in_lines = stdin.readlines()
#in_lines = [ '5 2', '1 3' ]

def int_values(line) :
    return map( int, line.split(' ') )

lines = map( int_values, map( lambda x:x.strip(), in_lines  ) )

n, k = lines[0]
days = set( lines[1][1:] )
d = 0
p = 0
for i in range( 1, n + 1) :
    d += 1
    if i in days :
        d = 0
        p += 1
    if d >= k :
        d = 0
        p += 1
print p
