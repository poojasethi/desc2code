N,K = map(int,raw_input().split())
holidays = map(int,raw_input().split())[1:]

if 0 in holidays:
	holidays = holidays[1:]

numPresents = 0
curDay = 0
curHoliday = 0

while curDay <= N:
	numPresents += 1

	if curHoliday >= len(holidays):
		curDay += K
	else:
		if holidays[curHoliday] <= curDay + K:
			curDay = holidays[curHoliday]
			curHoliday += 1
		else:
			curDay += K

print numPresents-1
