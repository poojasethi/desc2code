typedef long long LL;
using namespace std;

char s[110000],c[510][10];
int sum[510][110000],a[510],cnt[510],p[510],q[510],l[510],r[510],L,R,d;
LL ans;

void add(int x){
    if (a[x]==L-1) ++d;
    if (a[x]==R) --d;
    ++a[x];
}   

void sub(int x){
    if (a[x]==R+1) ++d;
    if (a[x]==L) --d;
    --a[x];
}

int main(){
    scanf("%s",s+1);
    int N=strlen(s+1),M;
    scanf("%d%d%d",&M,&L,&R);
    int i,j;
    for (i=1;i<=M;i++)scanf("%s%d%d",c[i],&l[i],&r[i]);
    //memset(a,-1,sizeof(a));
    for (j=1;j<=N;j++){
        d+=!L;
        //add(j);
        for (i=1;i<=M;i++){
            sum[i][j]=sum[i][j-1]+(s[j]==*c[i]);
            int &p=::p[i],&q=::q[i];
            if (sum[i][j]>=l[i]){
                if (!p && !q){
                    p=q=1; add(1);
                }
                while (sum[i][j]-sum[i][p-1]>r[i]) sub(p++);
                while (q<j && sum[i][j]-sum[i][q]>=l[i]) add(++q);
            }
        }
        ans+=d;
        //cout << j << ": " << ans << endl;
        //cout << j << ":" << endl;
        //for (i=1;i<=M;i++) cout << p[i] << ' ' << q[i] << endl;
        //cout << endl;
        //for (i=1;i<=N;i++) cout << a[i] << ' ';
        //cout << endl;
        //for (i=L;i<=R;i++) ans+=cnt[i];
    }
    printf("%I64d\n",ans);
}
