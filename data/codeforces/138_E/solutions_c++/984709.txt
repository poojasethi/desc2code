#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<iostream>
#define MAXN 100050
#define MAXM 505
using namespace std;
int n,L,R,m,l[MAXM],r[MAXM],a[MAXN],b[MAXM];
int lp[MAXM],rp[MAXM],cnt_l[MAXM],cnt_r[MAXM],cnt[MAXN];
bool judge(int x)
{
	return x>=L&&x<=R;
}
long long cal()
{
	long long ans=0;
	int w,now=0;
	memset(cnt,0,sizeof(cnt));
	for(int i=1;i<=n;i++){
		if(judge(0))
			now++;
		for(int j=1;j<=m;j++){
			w=(a[i]==b[j]);
			cnt_l[j]+=w;
			cnt_r[j]+=w;
			while(rp[j]<i&&cnt_r[j]>=l[j]){
				rp[j]++;
				w=(a[rp[j]]==b[j]);
				cnt_r[j]-=w;
				cnt[rp[j]]++;
				if(!judge(cnt[rp[j]]-1)&&judge(cnt[rp[j]]))
					now++;
				else if(judge(cnt[rp[j]]-1)&&!judge(cnt[rp[j]]))
					now--;
			}
			while(lp[j]<rp[j]&&cnt_l[j]>r[j]){
				lp[j]++;
				w=(a[lp[j]]==b[j]);
				cnt_l[j]-=w;
				cnt[lp[j]]--;
				if(!judge(cnt[lp[j]]+1)&&judge(cnt[lp[j]]))
					now++;
				else if(judge(cnt[lp[j]]+1)&&!judge(cnt[lp[j]]))
					now--;
			}
		}
		ans+=now;
	}
	return ans;
}
int main()
{
	//freopen("input.txt","r",stdin);
	char str[MAXN];
	while(scanf("%s",str)!=EOF){
		n=strlen(str);
		for(int i=0;i<n;i++)
			a[i+1]=str[i]-'a';
		scanf("%d%d%d",&m,&L,&R);
		for(int i=1;i<=m;i++){
			scanf("%s%d%d",str,l+i,r+i);
			b[i]=str[0]-'a';
			cnt_l[i]=cnt_r[i]=lp[i]=rp[i]=0;
		}
		printf("%I64d\n",cal());
	}
	return 0;
}
