typedef long long LL;
typedef vector< string > VS;

#define p_b push_back
#define m_p make_pair

#define Size(a) ((int)a.size())
#define sqr(x) ((x)*(x))

#define oo 1000000000
#define eps 1e-9
const double pi = acos(-1.0);

#define maxn 100000 + 10
#define maxm 500 + 10

int n,m,L,R;
LL ans = 0;

char str[maxn],ch;
int sum[26][maxn];
int c[maxm],l[maxm],r[maxm];

int s[maxn],x[maxm],y[maxm];

int main()
{
    int i,j,t = 0;
    
    scanf("%s",str+1);
    n = strlen(str+1);
    scanf("%d%d%d",&m,&L,&R);
    for( i = 1 ; i <= m ; ++i )
        {
            scanf(" %c%d%d",&ch,&l[i],&r[i]);
            c[i] = ch-'a';
            x[i] = 1;
        }
    for( i = 0 ; i < 26 ; ++i )
        for( j = 1 ; j <= n ; ++j )
            sum[i][j] = sum[i][j-1] + (str[j]-'a' == i);
    
    for( i = 1 ; i <= n ; ++i )
        {
            if( !L )
                ++t;
            for( j = 1 ; j <= m ; ++j )
                {
                    while( sum[c[j]][i]-sum[c[j]][x[j]-1] > r[j] )
                        {
                            if( s[x[j]] == L )
                                --t;
                            --s[x[j]];
                            if( s[x[j]] == R )
                                ++t;
                            ++x[j];
                        }
                    while( y[j] < i && sum[c[j]][i]-sum[c[j]][y[j]] >= l[j] )
                        {
                            ++y[j];
                            if( s[y[j]] == R )
                                --t;
                            ++s[y[j]];
                            if( s[y[j]] == L )
                                ++t;
                        }
                }
            ans += t;
        }
    cout<<ans<<endl;
        
    return 0;
}
