[n, k] = map(int, raw_input().split())
res = []
res.append(1)
res.append(k + 1)
i = 2
delta = k
while i < k + 1:
    delta -= 1
    if i & 1 == 1:
        res.append(res[i - 1] + delta)
    else:
        res.append(res[i - 1] - delta)
    i += 1
i = k + 1
while i < n:
    res.append(i + 1)
    i += 1
print ' '.join(map(str,res))
