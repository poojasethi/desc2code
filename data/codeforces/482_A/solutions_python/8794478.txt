line = raw_input().strip()
n, k = int(line.split()[0]), int(line.split()[1])
i, j = n, 1

print i,
i -= 1
curr = False

while k > 1:
	if curr:
		print i,
		i -= 1
	else:
		print j,
		j += 1
	curr = not curr
	k -= 1

if curr:
	while j <= i:
		print j,
		j += 1
else:
	while i >= j:
		print i,
		i -= 1
