def solve(n, k):
    result = [x for x in xrange(1, n+1)]

    m = k+1

    for x in xrange(1, (m+1)/2 + 1):
        result[2*x-2] = x

    for x in xrange((m+1)/2+1, m+1):
        result[(m-x)*2+1] = x

    return result


def main():
    inputs = raw_input().split()
    n, k = int(inputs[0]), int(inputs[1])

    print ' '.join([str(x) for x in solve(n, k)])

if __name__ == '__main__':
    main()
