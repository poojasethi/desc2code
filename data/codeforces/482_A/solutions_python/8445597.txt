def solve(n, k):
    res = [1]
    unused = set(range(2, n+1))
    while k > 0:
        x = res[-1]-k
        if not x in unused:
            x = res[-1]+k
        unused.remove(x)
        k -= 1
        res.append(x)
    for x in unused:
        res.append(x)
    return res

n, k = map(int, raw_input().split())
print ' '.join(map(str, solve(n, k)))
