n, k = map(int, raw_input().split())
l, r = 1, n;
ans = []
for i in xrange(k):
    if i & 1:
        ans.append(r)
        r -= 1
    else:
        ans.append(l)
        l += 1

if k & 1:
    ans.extend(range(l, r + 1))
else:
    ans.extend(range(r, l - 1, -1))
print " ".join(map(str, ans))
