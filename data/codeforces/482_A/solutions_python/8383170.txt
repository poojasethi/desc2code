n, k = map(int, raw_input().split())
u = True
d = n - 1
l = 1
print l,
k -= 1
p = n - 1
while k > 0:
    if u:
        l += d
        u = False
    else:
        l -= d
        u = True
    d -= 1
    k -= 1
    print l,
    p -= 1
if u:
    while p > 0:
        l += 1
        print l,
        p -= 1
else:
    while p > 0:
        l -= 1
        print l,
        p -= 1
