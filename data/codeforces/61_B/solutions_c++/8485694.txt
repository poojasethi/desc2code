#include <iostream>

using namespace std;
string cl(string s)
{
 string st;
 for (int i=0; i<s.size(); i++)
  {
    if (s[i]>='a'&&s[i]<='z') st+=s[i];
    if (s[i]>='A'&&s[i]<='Z') st+=(s[i]-'A'+'a');
  }

 return st;
}
int main()
{ string s1,s2,s3,s;
  int n;
  cin>>s1>>s2>>s3;
 s1=cl(s1);
 s2=cl(s2);
 s3=cl(s3);
 cin>>n;
 for (int i=0; i<n;i++)
  {
   cin>>s;
   s=cl(s);
   if ((s==s1+s2+s3)||(s==s1+s3+s2)||(s==s2+s1+s3)||(s==s2+s3+s1)||(s==s3+s1+s2)||(s==s3+s2+s1)) cout<<"ACC"<<endl;
   else cout<<"WA"<<endl ;
  }

 return 0;
}
