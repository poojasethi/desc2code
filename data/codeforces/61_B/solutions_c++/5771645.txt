#include <iostream>
using namespace std;
int n;
string x,y,z,x1,y1,z1,s;
int main()
{
    cin >> x >> y >> z;
    cin >> n;
    for (int i=0; i<x.size(); i++)if (isalpha(x[i]))x1+=tolower(x[i]);
    for (int i=0; i<y.size(); i++)if (isalpha(y[i]))y1+=tolower(y[i]);
    for (int i=0; i<z.size(); i++)if (isalpha(z[i]))z1+=tolower(z[i]);
    for (int i=0; i<n; i++)
    {
        cin >> s;
        string s1;
        for (int i=0; i<s.size(); i++)if (isalpha(s[i]))s1+=tolower(s[i]);
        if (s1==(x1+y1+z1) || s1==(x1+z1+y1) || s1==(y1+x1+z1) || s1==(y1+z1+x1) || s1==(z1+x1+y1) || s1==(z1+y1+x1))cout << "ACC\n";
        else cout << "WA\n";
    }

    return 0;
}
