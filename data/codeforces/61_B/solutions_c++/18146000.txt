#include <iostream>
#include <string>
using namespace std;

string process(string s){
	string ans;
	long long i;
	
	for(i=0;i<s.size();i++){
		if(s[i]>='a' and s[i]<='z') ans+=s[i];
		else if(s[i]>='A' and s[i]<='Z') ans += s[i]-'A'+'a';
	}
	return ans;
}

int main(){
	string a, b, c, inp, arr[6];
	long long n, i, f;
	
	cin >> inp;
	a = process(inp);
	cin >> inp;
	b = process(inp);
	cin >> inp;
	c = process(inp);
	
	arr[0] = a+b+c;
	arr[1] = a+c+b;
	arr[2] = b+a+c;
	arr[3] = b+c+a;
	arr[4] = c+a+b;
	arr[5] = c+b+a;
	
	cin >> n;
	while(n--){
		cin >> inp;
		inp = process(inp);
		f=0;
		for(i=0;i<6;i++){
			if(inp==arr[i]) f=1;
		}
		if(f) cout << "ACC\n";
		else cout << "WA\n";
	}
	return 0;
}
