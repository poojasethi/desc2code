#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <cmath>
#include <queue>

using namespace std;

void sredi(string &s)
{
    string s1;
    for(int i=0;i<s.size();i++)
    {
        if(s[i]>='a' && s[i]<='z')s1+=s[i];
        if(s[i]>='A' && s[i]<='Z')s1+=char(s[i]-('A'-'a'));
    }
    s=s1;
}

int main()
{
    string a, b, c;
    cin>>a>>b>>c;
    int n;
    cin>>n;
    sredi(a);
    sredi(b);
    sredi(c);
    for(int i=0;i<n;i++)
    {
        string tmp;
        cin>>tmp;
        sredi(tmp);
        if(tmp==a+b+c || tmp==a+c+b || tmp==b+a+c || tmp==b+c+a || tmp==c+a+b || tmp==c+b+a)
            cout<<"ACC"<<endl;
        else cout<<"WA"<<endl;
    }
    return 0;
}
