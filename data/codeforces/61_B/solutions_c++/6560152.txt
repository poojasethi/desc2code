#include <iostream>
#include <cstdio>

using namespace std;

string process(string a){
	string x;
	for(int i = 0; i < a.size(); i++){
		if(isalpha(a[i])){
			x += tolower(a[i]);
		}
	}
	return x;
}

int main() {
	string a, b, c, d, e, f;
	string x, y, z;
	cin >> x >> y >> z;
	x = process(x);
	y = process(y);
	z = process(z);
	int n;
	a = x + y + z;
	b = x + z + y;
	c = y + x + z;
	d = y + z + x;
	e = z + x + y;
	f = z + y + x;
	cin >> n;
	for(int i = 0; i < n; i++){
		cin >> x;
		x = process(x);
		if(x == a || x == b || x == c || x == d || x == e || x == f){
			cout << "ACC\n";
		}
		else{
			cout << "WA\n";
		}
	}
	
	return 0;
}
