#include<iostream>
#include<string>
using namespace std;

string s,r,t,a[6],b;
int n;
bool ch;

string f(string s){
  string r;
  for(int i=0;i<s.length();i++){
    if(s[i]>='A' && s[i]<='Z')
      s[i]+='a'-'A';
    if(s[i]>='a' && s[i]<='z')
      r+=s[i];
  }
  return r;
}

int main(){
  cin>>s>>r>>t;
  s=f(s); r=f(r); t=f(t);
  a[0]=s+r+t; a[1]=s+t+r; a[2]=r+s+t;
  a[3]=r+t+s; a[4]=t+s+r; a[5]=t+r+s;
  cin>>n;
  for(int i=0;i<n;i++){
    cin>>b;
    ch=false;
    b=f(b);
    for(int j=0;j<6;j++)
      if(b==a[j])
        ch=true, cout<<"ACC"<<endl;
    if(!ch)
      cout<<"WA"<<endl;
  }
  return 0;
}
