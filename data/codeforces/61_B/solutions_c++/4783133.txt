#include <bits/stdc++.h>

using namespace std;

int main() {

  int n;
  string a[3];
  for(int i = 0; i < 3; ++i) {
    char c;
    while((c = getchar()) <= ' ');
    if(isalpha(c)) a[i] += tolower(c);
    while((c = getchar()) > ' ') if(isalpha(c)) a[i] += tolower(c);
  }
  sort(a, a + 3);
  set<string> valid;
  do
  {
    string x;
    for(int i = 0; i < 3; ++i) x += a[i];
    valid.insert(x);
  }while(next_permutation(a, a + 3));
  scanf("%d", &n);

  while(n--) {
    char c;
    string s;
    while((c = getchar()) <= ' ');
    if(isalpha(c)) s += tolower(c);
    while((c = getchar()) > ' ') if(isalpha(c)) s += tolower(c);
    if(valid.count(s)) {
      puts("ACC");
    }else {
      puts("WA");
    }
  }

  return 0;
}
