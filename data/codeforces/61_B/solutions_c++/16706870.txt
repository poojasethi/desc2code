#include<bits/stdc++.h>
using namespace std;
inline void splay(int &v){
	v=0;char c=0;int p=1;
	while(c<'0' || c>'9'){if(c=='-')p=-1;c=getchar();}
	while(c>='0' && c<='9'){v=(v<<3)+(v<<1)+c-'0';c=getchar();}
	v*=p;
}
string a,b,c;
string change(string a){
	string tmp;
	for(int i=0;i<a.length();i++){
		if(a[i]>='A' && a[i]<='Z')a[i]-='A'-'a';
		if((a[i]>='a'&&a[i]<='z')||(a[i]>='A'&&a[i]<='Z')){
			tmp+=a[i];
		}
	}
	return tmp;
}
int main(){

cin>>a>>b>>c;a=change(a),b=change(b),c=change(c);
	int x;splay(x);
	for(;x--;){
		string t;cin>>t;t=change(t);
		if(t==a+b+c||t==a+c+b||t==b+a+c||t==b+c+a||t==c+a+b||t==c+b+a)cout<<"ACC\n";
		else cout<<"WA\n";
	}
}
