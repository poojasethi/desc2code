#include <iostream>
#include <string>
#include <cstdio>
using namespace std;
string x,y,z,k;
string change(string s)
{
    string str="";
    for(int i=0;i<s.length();i++)
        if(s[i]>='a'&&s[i]<='z') str+=s[i];
    else if(s[i]>='A'&&s[i]<='Z') str+=s[i]-'A'+'a';
    return str;
}
int main(){
    int n;
    cin>>x>>y>>z>>n;
    x=change(x);
    y=change(y);
    z=change(z);
    for(int i=0;i<n;i++)
    {
       string k;
       cin>>k;
       k=change(k);
       if(k==x+y+z||k==x+z+y||k==y+x+z||k==y+z+x||k==z+x+y||k==z+y+x) cout<<"ACC"<<endl;
       else
        cout<<"WA"<<endl;
    }
    return 0;
}
