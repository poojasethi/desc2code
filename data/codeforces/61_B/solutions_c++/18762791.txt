#include<bits/stdc++.h>

using namespace std;

string convert(string s)
{
    int i;
    string tmp;
    for( i=0; i<s.length(); i++)
    {
        if(s[i]>='a' && s[i]<='z') tmp+=s[i];
        if(s[i]>='A' && s[i]<='Z') tmp+=s[i]-'A'+'a';
    }
    return tmp;
}

int main()
{
    string x,y,z,str,t;
    int i,n;
    cin>>x>>y>>z>>n;
    x=convert(x);
    y=convert(y);
    z=convert(z);
    for(i=1; i<=n; i++)
    {
        cin>>str;
        str=convert(str);
        if(str==x+y+z || str==x+z+y || str==y+x+z || str==y+z+x || str==z+y+x || str==z+x+y)
            cout<<"ACC";
        else
            cout<<"WA";
        cout<<endl;
    }
    return 0;
}
