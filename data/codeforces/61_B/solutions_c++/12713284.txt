#include <bits/stdc++.h>

using namespace std;

string a,b,c,cur;

string prep(string s)
{
    string res="";
    int i;
    for(i=0;i<s.size();i++)
    {
        if(s[i]>='A'&&s[i]<='Z') res+=s[i]-'A'+'a';
        else if(s[i]>='a'&&s[i]<='z') res+=s[i];
    }
    return res;
}

int main()
{

    cin >> a >> b >> c;
    a=prep(a);b=prep(b);c=prep(c);
    int n;
    cin >> n;
    int i;
    for(i=0;i<n;i++)
    {
        cin >> cur;
        cur=prep(cur);
        if(a+b+c==cur||a+c+b==cur||b+a+c==cur||b+c+a==cur||c+a+b==cur||c+b+a==cur)
        {
            cout << "ACC" << endl;
        }
        else
        {
            cout << "WA" << endl;
        }
    }
}
