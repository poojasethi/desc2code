#include <iostream>
using namespace std;

string r(string a){
	string q = "";
	for(int i=0; i<a.size(); i++){
		if(a[i]>='a' && a[i]<='z'){
			q+=a[i];
	   	}
	   	if(a[i]>='A' && a[i]<='Z'){
	   		q+=(a[i] - 'A' + 'a');
	   	}
  	}
  	return q;
}

int main(){
	string a, b, c, ans1="",ans2="",ans3="",ans4="",ans5="",ans6="";
	cin >> a >> b >> c;
	a = r(a); b = r(b); c = r(c);
	ans1=a+b+c;ans2=b+c+a;ans3=c+a+b;ans4=c+b+a;ans5=b+a+c;ans6=a+c+b;
	int n;
	cin >> n;
	for(int i=0; i<n; i++){
		string t;
		cin >> t;
		t = r(t);
		if(t==ans1 || t==ans2 || t==ans3 || t==ans4 || t==ans5 || t==ans6){
			cout << "ACC" << endl;
		}
		else{
			cout << "WA" << endl;
		}
	}
	return 0;
}
