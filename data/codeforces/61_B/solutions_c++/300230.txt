#include <iostream>
#include <algorithm>

using namespace std;

string get() {
    string s,r;
    cin>>s;
    for (unsigned int i=0; i<s.size(); i++)
        if (isalpha(s[i]))
            r.push_back(isupper(s[i])?tolower(s[i]):s[i]);
    return r;
}

int main() {
    ios::sync_with_stdio(false);
    string a[3],p[6];
    a[0]=get();
    a[1]=get();
    a[2]=get();
    sort(a,a+3);
    for (int i=0; i<6; i++) {
        p[i]=a[0]+a[1]+a[2];
        next_permutation(a,a+3);
    }
    int n;
    cin>>n;
    while (n--)
        cout<<((find(p,p+6,get())!=(p+6))?"ACC":"WA")<<endl;
    return 0;
}
