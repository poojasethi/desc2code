#include<iostream>
using namespace std;
string C(string s){
	string t;
	for(int i=0;i<s.length();i++){
		if(s[i]>='a'&&s[i]<='z')t+=s[i];
		if(s[i]>='A'&&s[i]<='Z')t+=s[i]-'A'+'a';
	}
	return t;
}
main(){
	string x,y,z,s,t;
	int i,n;
	cin>>x>>y>>z>>n;
	x=C(x);
	y=C(y);
	z=C(z);
	for(i=1;i<=n;i++){
		cin>>s;
		s=C(s);
		if(s==x+y+z||s==x+z+y||s==y+x+z||s==y+z+x||s==z+y+x||s==z+x+y)cout<<"ACC";
		else cout<<"WA";
		cout<<endl;
	}
}
