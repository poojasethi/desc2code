#include<bits/stdc++.h>

using namespace std;

string convert(string s)
{
    int i;
    string st;
    for( i=0; i<s.length(); i++)
    {
        if(s[i]>='a' && s[i]<='z') st+=s[i];
        if(s[i]>='A' && s[i]<='Z') st+=s[i]-'A'+'a';
    }
    return st;
}

int main()
{
    string x,y,z,s,t;
    int i,n;
    cin>>x>>y>>z>>n;
    x=convert(x);
    y=convert(y);
    z=convert(z);
    for(i=1; i<=n; i++)
    {
        cin>>s;
        s=convert(s);
        if(s==x+y+z || s==x+z+y || s==y+x+z || s==y+z+x || s==z+y+x || s==z+x+y)
            cout<<"ACC\n";
        else
            cout<<"WA\n";
    }
    return 0;
}
