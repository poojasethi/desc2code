#include<bits/stdc++.h>
using namespace std;

string Fix(string x)
{
    for(int i=0;i<x.size();i++)
        if(x[i]>='A'&&x[i]<='Z')
            x[i] = x[i] - 'A' + 'a';
        else if(x[i]=='-'||x[i]=='_'||x[i]==';'){
            x.erase(x.begin()+i);
            i--;
        }
    return x;
}
int main()
{
    int n;  string a,b,c; cin >> a >> b >> c; cin >> n;
    a = Fix(a); b = Fix(b); c = Fix(c);
    for(int i=0;i<n;i++)
    {
        string x;
        cin >> x;
        x = Fix(x);
        if(x==a+b+c||x==a+c+b||x==b+a+c||x==b+c+a||x==c+a+b||x==c+b+a)
            cout << "ACC" << endl;
        else
            cout << "WA" << endl;
    }
}
