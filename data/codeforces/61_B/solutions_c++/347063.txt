 string make(string a)
 {
      string b;
      int n=a.size();
      for(int i=0;i<n;i++){
      if(a[i]>=65 && a[i]<=90)
      b+=a[i]+32;
      else if(a[i]>=97 && a[i]<=122)
      b+=a[i];}
      return b;
 }
