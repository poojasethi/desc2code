#!/usr/bin python
from itertools import permutations
def remove(s):
    return ''.join((chr(ord(t)|0x20) for t in s if t.isalpha()))

def getcat():
    return permutations(remove(raw_input()) for _ in xrange(3)) 

lex = set(''.join(f) for f in getcat())
ans = ['WA', 'ACC']
print '\n'.join(ans[remove(raw_input()) in lex] for _ in xrange(int(raw_input())))
