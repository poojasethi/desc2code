from string import ascii_letters
a=filter(lambda x: x in ascii_letters,raw_input()).lower()
b=filter(lambda x: x in ascii_letters,raw_input()).lower()
c=filter(lambda x: x in ascii_letters,raw_input()).lower()
n=int(raw_input())
for i in range(n):
	s=filter(lambda x: x in ascii_letters,raw_input()).lower()
	if s==a+b+c or s==c+a+b or s==b+c+a or s==c+b+a or s==a+c+b or s==b+a+c:
		print 'ACC'
	else:
		print 'WA'
