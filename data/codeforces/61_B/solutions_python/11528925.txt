p = raw_input()
p = p.replace('_','').replace('-','').replace(';','').lower()

q = raw_input()
q = q.replace('_','').replace('-','').replace(';','').lower()

r = raw_input()
r = r.replace('_','').replace('-','').replace(';','').lower()

l = [p+q+r, p+r+q, q+p+r, q+r+p, r+p+q, r+q+p]

n = int(raw_input())

for x in range(0, n) :

    s = raw_input()

    s = s.replace('_','').replace('-','').replace(';','').lower()

    if s in l :

        print 'ACC'

    else :

        print 'WA'
