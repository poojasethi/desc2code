#!/usr/bin python
from itertools import permutations

def remove(s):
    return ''.join((t.lower() for t in s if t.isalpha()))

def getcat():
    return permutations(remove(raw_input()) for _ in xrange(3)) 

lex = {}.fromkeys(''.join(f) for f in getcat())
ans = ['WA', 'ACC']
print ' '.join(ans[remove(raw_input()) in lex] for _ in xrange(int(raw_input())))
