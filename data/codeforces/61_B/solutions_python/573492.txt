def conv(s):
    t = ""
    for k in s.lower():
        if k >= 'a' and k <= 'z':
            t += k
    return t

a = []
for i in range(3):
    a.append(conv(raw_input()))
b = []
for i in range(3):
    for j in range(3):
        for k in range(3):
            if i != j != k:
                b.append(a[i] + a[j] + a[k])

n = int(raw_input())
for i in range(n):
    if conv(raw_input()) in b:
        print "ACC"
    else:
        print "WA"
