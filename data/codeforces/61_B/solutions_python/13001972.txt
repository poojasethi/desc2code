import re,itertools

def convert(t):
    t = t.lower()
    return "".join(re.findall("[a-z]+",t))

st = []
for i in xrange(3):
    st.append(raw_input())

st = map(convert,st)
poss = set("".join(x) for x in itertools.permutations(st))

n = int(raw_input())
for _ in xrange(n):
    st1 = convert(raw_input())
    print "ACC" if st1 in poss else "WA"
