from itertools import *

def remove(S):
	S=S.replace(';','')
	S=S.replace('-','')
	S=S.replace('_','')
	return S.lower()

def Main():
	D=[remove(raw_input().strip()) for i in range(3)]
	P=[''.join(x) for x in permutations(D)]
	for N in range(int(raw_input())):
		I=remove(raw_input().strip())
		if I in P:print "ACC"
		else:print "WA"

Main()
