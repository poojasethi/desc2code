def read():
    return raw_input().replace('-', '').replace(';', '').replace('_', '').lower()

s1 = read()
s2 = read()
s3 = read()
good = set([s1+s2+s3, s1+s3+s2, s2+s1+s3, s2+s3+s1, s3+s1+s2, s3+s2+s1])
for _ in range(int(raw_input())):
    if read() in good:
        print('ACC')
    else:
        print('WA')
