def norm(s):
    s=s.lower();
    s=s.replace("_","")
    s=s.replace(";","")
    s=s.replace("-","")
    return s

a=norm(raw_input())
b=norm(raw_input())
c=norm(raw_input())
n=(int)(raw_input())

alist=[a+b+c,a+c+b,b+a+c,b+c+a,c+a+b,c+b+a]

for i in xrange(n):
    s=norm(raw_input())
    if s in alist:
        print("ACC")
    else:
        print("WA")
