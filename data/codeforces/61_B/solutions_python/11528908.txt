p = raw_input().lower()
p = p.replace('_','').replace('-','').replace(';','')

q = raw_input().lower()
q = q.replace('_','').replace('-','').replace(';','')

r = raw_input().lower()
r = r.replace('_','').replace('-','').replace(';','')

l = [p+q+r, p+r+q, q+p+r, q+r+p, r+p+q, r+q+p]

n = int(raw_input())

for x in range(0, n) :

    s = raw_input().lower()

    s = s.replace('_','').replace('-','').replace(';','')

    if s in l :

        print 'ACC'

    else :

        print 'WA'
