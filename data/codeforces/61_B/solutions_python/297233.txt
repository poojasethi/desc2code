#! /usr/bin/env python

from sys import stdin
import re

def clean(s):
	s=s.strip().lower()
	s=re.sub(r"[^a-z]+","",s)
	return s

if __name__=='__main__':
	(A,B,C)=map(clean,[stdin.readline() for i in xrange(3)])
	cache=set([A+B+C,A+C+B,B+A+C,B+C+A,C+A+B,C+B+A])
	
	n=int(stdin.readline())
	for case in xrange(n):
		if clean(stdin.readline()) in cache:
			print "ACC"
		else:
			print "WA"
