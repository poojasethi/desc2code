#include <bits/stdc++.h>
#define PII pair<int,int>
#define f first
#define s second
#define VI vector<int>
#define MP make_pair
#define PB push_back
#define ALL(V) V.begin(),V.end()
#define REP(x, n)     for(int x=0;x<(n);x++)
using namespace std;
int n,m,a,b;
VI V[200004];
int pop[200004];
vector<PII> powr[200004];
int odw[200004];
int odl[200004];
VI idz(int a,int b)
  {
  VI ret;
  ret.PB(a);
  while(a!=b)
    {
    a=pop[a];
    ret.PB(a);
    }
  return ret;
  }
void pisz(VI V)
  {
  printf("%d ",(int)V.size());
  REP(i,V.size())printf("%d ",V[i]);
  puts("");
  }
void koncz(PII a,PII b,int x)
  {
  if(odl[a.s]<odl[b.s])swap(a,b);
  int x1=a.f;
  int y1=a.s;
  int x2=b.f;
  int y2=b.s;
  VI s1=idz(x,y1);
  VI s2=idz(x1,x);
  reverse(ALL(s2));
  s2.PB(y1);
  
  VI s3=idz(x2,x);
  reverse(ALL(s3));
  VI temp=idz(y1,y2);
  reverse(ALL(temp));
  REP(i,temp.size())s3.PB(temp[i]);
  puts("YES");
  pisz(s1);
  pisz(s2);
  pisz(s3);
  exit(0);
  }
void dfs(int x)
  {
  odw[x]=1;
  REP(i,V[x].size())
    {
    if(V[x][i]==pop[x])continue;
    if(odw[V[x][i]])
      {
      if(odl[V[x][i]]<odl[x])
        {
        powr[x].PB(MP(x,V[x][i]));
        }
      continue;
      }
    odl[V[x][i]]=odl[x]+1;
    pop[V[x][i]]=x;
    dfs(V[x][i]);
    REP(j,powr[V[x][i]].size())
      {
      if(powr[V[x][i]][j].s!=x)powr[x].PB(powr[V[x][i]][j]);
      }
    }
  if(powr[x].size()>1)
    {
    koncz(powr[x][0],powr[x][1],x); 
    }
  }
main()
{
scanf("%d%d",&n,&m);
REP(i,m)
  {
  scanf("%d%d",&a,&b);
  V[a].PB(b);
  V[b].PB(a);
  }
REP(i,n)
  {
  if(!odw[i+1])dfs(i+1);
  }
puts("NO");
}
