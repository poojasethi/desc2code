#include<bits/stdc++.h>
using namespace std;
int dp[5001],w[5001],h[5001],dir[5001],n;
int rec(int t)
{
	if(dp[t])return dp[t];
	dp[t]=1;
	for(int i=0;i<=n;i++)
	{
		if(w[i]>w[t] && h[i]>h[t])
		{
			if(1+rec(i)>dp[t])
			{
				dp[t]=1+dp[i];
				dir[t]=i;
			}
		}
	}
	return dp[t];
}
int main()
{
	scanf("%d",&n);
	for(int i=0;i<=n;i++)
	{
		dir[i]=-1;
		scanf("%d%d",w+i,h+i);
	}
	rec(0);
	printf("%d\n",dp[0]-1);
	for(int i=dir[0]; ; )
	{
		if(i==-1)break;
		printf("%d ",i);
		i=dir[i];
	}
	return 0;
}
