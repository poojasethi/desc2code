#include <iostream>
#include <vector>
using namespace std;

int dfs(int v);
bool u[6000];
int p[6000], d[6000], n, i;
vector< pair<int, int> > x;

int main() {
	cin >> n;
	n++;
	vector< pair<int, int> > a(n);
	for(i = 0; i < n; i++) cin >> a[i].first >> a[i].second, p[i] = -1;
	x = a;
	dfs(0);
	cout << d[0] << endl;
	for(i = p[0]; i != -1; i = p[i]) cout << i << " ";	
}

int dfs(int v) {
	if(u[v]) return d[v];
	u[v] = true;	
	for(int i = 0; i < n; ++i) {
		if(x[v].first < x[i].first && x[v].second < x[i].second && dfs(i) + 1 > d[v]) {
			d[v] = d[i] + 1;
			p[v] = i;
		}
	}
	return d[v];		
}
