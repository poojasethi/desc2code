#include<bits/stdc++.h>
using namespace std;
int a[5001],b[5001],d[5001],index[5001],n;
int dp(int k)
{
    if(d[k])
    return d[k];
    int i;
    d[k]=1;
    for(i=0;i<=n;++i)
        if(a[i]>a[k] && b[i]>b[k])
            if((dp(i)+1)>d[k])
        index[k]=i,d[k]=d[i]+1;
    return d[k];
}
int main()
{
    int x,y,i;
    scanf("%d",&n);
    for(i=0;i<=n;++i)
        index[i]=-1,scanf("%d%d",(a+i),(b+i));
    dp(0);
    printf("%d\n",d[0]-1);
    for(i=index[0];i!=-1;i=index[i])printf("%d ",i);
    return 0;
}
