#include <iostream>
#include <vector>
using namespace std;
#define SZ 5005
int w[SZ],h[SZ],dp[SZ],go[SZ],n;
int dfs(int v)
{
  if (dp[v]!=0)return dp[v];
  int maxi=0,t;
  for(int j=1;j<n;j++)
  if (w[v]<w[j] && h[v]<h[j])
  {
    t=dfs(j);
    if (t>maxi)maxi=t,go[v]=j;
  }
  return dp[v]=maxi+1;
}
int main()
{
  cin>>n>>w[0]>>h[0];
  n++;
  for(int i=1;i<n;i++)cin>>w[i]>>h[i];
  dfs(0);
  cout<<dp[0]-1<<endl;
  int v=0;
  while (go[v]!=0)
  {
    cout<<go[v]<<" ";
    v=go[v];
  }
  return 0;
}
