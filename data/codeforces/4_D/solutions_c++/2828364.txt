#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
#define maxn 5010
int n,w[maxn],h[maxn],d[maxn],pre[maxn];
int dfs(int v)
{
	int i;
	if(d[v])return d[v];
	d[v]=1;
	for(i=1;i<=n;i++)
		if(w[i]>w[v]&&h[i]>h[v])
			if(dfs(i)+1>d[v])
				pre[v]=i,d[v]=d[i]+1;
	return d[v];
}
int main()
{
    int i,ans;
    scanf("%d",&n);
    for(i=0;i<=n;i++)
    	scanf("%d%d",w+i,h+i);
    ans=dfs(0);
    ans--;
    printf("%d\n",ans);
    for(i=pre[0];i;i=pre[i])
    	printf("%d ",i);
    puts("");
	return 0;
}
