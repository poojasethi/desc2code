#include <stdio.h>
#include <string.h>
int p[5001],w[5001],h[5001],d[5001],n;
int dfs(int v){
    int i;
    if(d[v] != 0)
        return d[v];
    d[v] = 1;
    for(i = 0; i <= n; i++)
        if(w[i] > w[v] && h[i] > h[v])
            if(dfs(i) + 1 > d[v])
                p[v] = i, d[v] = d[i] + 1;
    return d[v];
}
int main(){
    int i;
    scanf("%d",&n);
    memset(p, -1, sizeof(p));
    for(i = 0; i <= n; i++)
        scanf("%d%d",&w[i],&h[i]);
    printf("%d\n",dfs(0)-1);
    for(i = p[0]; i != -1; i = p[i])
        printf("%d ",i);
}
