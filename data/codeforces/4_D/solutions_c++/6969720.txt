#include <iostream>
using namespace std;
int n,w[5005],h[5005];
bool was[5005];
int best[5005],p[5005];
void fb(int v){
	if (was[v]) return;
	for(int i=0;i<=n;i++)
		if (w[v]<w[i]&&h[v]<h[i]){
			fb(i);
			if (best[i]+1>best[v]){
				best[v]=best[i]+1;
				p[v]=i;
			}
		}
	was[v]=true;
}
void pr(int v){
	cout<<v;
	if (p[v]){
		cout<<" ";
		pr(p[v]);
	}
}
int main(){
	cin>>n>>w[0]>>h[0];
	for(int i=1;i<=n;i++)
		cin>>w[i]>>h[i];
	fb(0);
	cout<<best[0]<<endl;
	if (best[0]) pr(p[0]);
	return 0;
}
