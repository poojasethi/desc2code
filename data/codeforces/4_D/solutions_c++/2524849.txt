#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
const int MX = 5001;
int w[MX],h[MX],to[MX],d[MX];
int n;
int dfs(int v)
{
    if(d[v])return d[v];
    d[v]=1;
    for(int i = 0 ; i <=n;i++)
    if(w[i]>w[v]&&h[v]<h[i])
       if(dfs(i)+1 >d[v])
        {to[v]=i;d[v]=d[i]+1;}
    return d[v];
    }
int main()
{
 cin>>n;
 for(int i = 0;i<=n;i++)
 {cin>>w[i]>>h[i];to[i]=-1;}
 
 dfs(0);
 
 cout<<d[0]-1<<"\n";
for(int i=to[0];i!=-1;i=to[i])
cout<<i<<" ";
 
 //system("PAUSE");
 return 0;   
}
