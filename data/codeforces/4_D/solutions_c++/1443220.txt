#include<iostream>
#include<cstring>
using namespace std;
int dfs(int);
int w[5001],h[5001],d[5001],p[5001],n;
main()
{
	cin>>n;
	for(int i=0;i<=n;i++)
	cin>>w[i]>>h[i];
	memset(d,0,sizeof(d));
	int k=dfs(0);
	cout<<k<<"\n";
	int i=0;
	while(p[i])
	{
		cout<<p[i]<<" ";
		i=p[i];
	}



	}
int dfs(int x)
{
	
	if(d[x])return d[x];
	for(int i=0;i<=n;i++)
	{
		if(w[i]>w[x] && h[i]>h[x])
		{
			int t=dfs(i);
			if(d[x]<t+1)
			{
				d[x]=t+1;
				p[x]=i;
			}
		}
	}
	return d[x];
}
