#include<stdio.h>

int w[5001], h[5001], d[5001], to[5001];
int n;
int dfs (int v)
{
  int i, t;
  if (d[v]) return d[v];
  d[v] = 1;
  for (i = 1; i <= n; i++)
    if (w[i] > w[v] && h[i] > h[v])
      if ((t = dfs(i) + 1) > d[v]) {
	d[v]  = t; to[v] = i;
      }
  return d[v];
}
int main ()
{
  int i, env;
  scanf ("%d", &n);
  for (i = 0; i <= n; i++)
    scanf ("%d %d", &w[i], &h[i]);

  env = dfs (0);
  env--;
  printf ("%d\n", env);
  if (to[0]) {
    for (i = to[0]; i; i = to[i])
      printf ("%d ", i);
  }
  printf ("\n");
  return 0;
}
