#include <cstdio>
#include <algorithm>
using namespace std;
int n,w[5001],h[5001],p[5001],d[5001];
int dfs(int i)
{
	if(d[i]!=0)
	{
		return d[i];
	}
	int max=0;
	for(int j=0;j<n;j++)
	{
		if(w[j]>w[i]&&h[j]>h[i])
		{
			int t;
			t=dfs(j);
			if(t>max)
			{
				max=t;
				p[i]=j;
			}
		}
	}
	return d[i]=max+1;
}
int main()
{
	scanf("%d%d%d",&n,&w[0],&h[0]);
	n++;
	for(int i=1;i<n;i++)
	{
		scanf("%d%d",&w[i],&h[i]);
	}
	dfs(0);
	printf("%d\n",d[0]-1);
	int v=0;
	while(p[v]!=0)
	{
		printf("%d%c",p[v],' ');
		v=p[v];
	}
}
