#include<bits/stdc++.h>
using namespace std;
int w[5005],h[5005],dp[5005],ans[5005],n;
int fun(int v)
{
	if(dp[v]!= 0) return dp[v];
	int maxx =0,k;
	for(int i =1  ;i <= n; i++)
	{
		if(w[v] < w[i] && h[v] < h[i])
		{
			k = fun(i);
			if(k > maxx){
				maxx = k;
				ans[v] = i;
			}
		}
	}
	return dp[v] = maxx+1;
}
int main()
{
	scanf("%d",&n);
	for(int i = 0; i <= n; i++)
	{
		scanf("%d%d",&w[i],&h[i]);
	}
	fun(0);
	cout << dp[0]-1<<endl;
	for(int i = 0; ans[i]; i = ans[i])
	{
		cout<<ans[i]<<" ";
	}
	
}
