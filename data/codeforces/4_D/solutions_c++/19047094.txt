#include <algorithm>
#include <cstdio>
int f[5010],g[5010],i,j,n,x,y,ans;
struct P{int x,y,id;}a[5010];
bool operator <(P a,P b){return a.x<b.x || (a.x==b.x && a.y<b.y);}
void Pr(int x)
{
	if (!x) return;
	Pr(g[x]);
	printf("%d ",a[x].id);
}
int main()
{
	scanf("%d%d%d\n",&n,&x,&y);
	for (i=1;i<=n;++i) scanf("%d%d\n",&a[i].x,&a[i].y),a[i].id=i;
	std::sort(a+1,a+n+1);
	for (i=1;i<=n;++i)
	if (a[i].x>x && a[i].y>y)
	{
		f[i]=1;g[i]=0;
		for (j=1;j<i;++j)
			if (a[i].x>a[j].x && a[i].y>a[j].y && f[j]>=f[i])
				f[i]=f[j]+1,g[i]=j;
		if (f[i]>f[ans]) ans=i;
	}
	printf("%d\n",f[ans]);
	Pr(ans);
}
