#include <iostream>
using namespace std;
int w[5010],h[5010],dp[5010],p[5010];
int n;
int dfs(int v)
{
    if(dp[v]!=0)return dp[v];
    int maxn=0,k;
    for(int j=1;j<=n;j++)
    if(w[v]<w[j] && h[v]<h[j])
    {
        k=dfs(j);
        if(k>maxn)maxn=k,p[v]=j;
    }
    return dp[v]=maxn+1;
}

int main()
{
    cin>>n>>w[0]>>h[0];
    for(int i=1;i<n+1;i++){cin>>w[i]>>h[i];}
    dfs(0);
    cout <<dp[0]-1<<endl;
    int t=0;
    while (p[t]!=0)
    {
          cout <<p[t]<<" ";
          t=p[t];
    }
    //system("pause");
    return 0;
}
