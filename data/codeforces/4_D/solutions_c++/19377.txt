#include<cstdio>
#define N 5555
int n,a[N],b[N],dp[N],next[N],sta[N];
void dfs(int i){
    if(dp[i])return;
    dp[i]=1;
    int j;
    for(j=n;j>=0;--j){
        if(a[i]<a[j]&&b[i]<b[j]){
            dfs(j);
            if(dp[i]<dp[j]+1){
                dp[i]=dp[j]+1;
                next[i]=j;
            }
        }
    }
}
int main(){
    scanf("%d%d%d",&n,a,b);
    int i;
    for(i=1;i<=n;++i)scanf("%d%d",a+i,b+i);
    dfs(0);
    printf("%d\n",--*dp);
    i=0;
    while(dp[0]--){
        i=next[i];
        printf("%d ",i);
    }
}
