#include<bits/stdc++.h>
#define MAX 5003
using namespace std;

int dp[MAX],to[MAX],w[MAX],h[MAX],n;

void dfs(int v)
{
	if(dp[v])
		return;
	dp[v]=1;
	for(int i=0;i<=n;i++)
	{
		if(w[i]>w[v] && h[i]>h[v])
		{
			dfs(i);
			if(dp[i]+1>dp[v])
			{
				dp[v]=dp[i]+1;
				to[v]=i;
			}
		}
	}
}

int main()
{
	scanf("%d %d %d",&n,w,h);
	for(int i=1;i<=n;i++)
		scanf("%d %d",w+i,h+i);
	dfs(0);
	printf("%d\n",--dp[0]);
	int i=0;
	while(dp[0]--)
	{
		i=to[i];
		printf("%d ",i);
	}
	return 0;
}
