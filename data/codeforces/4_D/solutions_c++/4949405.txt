#include <stdio.h>
#include <string.h>

int n,w[5005],h[5005],path[5005],dp[5005];

void dfs(int v)
{
	if(dp[v])
	{
	   return;
	}
	dp[v]=1;
	int i;
	for(i=0;i<=n;i++)
	{
		if(w[i]>w[v] && h[i]>h[v])
		{
			dfs(i);
			if(dp[i]+1>dp[v])
			{
				dp[v]=dp[i]+1;
				path[v]=i;
			}
		}
	}
}

int main()
{
int i;
while(scanf("%d",&n)==1)
{
	for(i=0;i<=n;i++)
	{
		scanf("%d%d",&w[i],&h[i]);
		path[i]=-1;
	}
	dfs(0);
	printf("%d\n",dp[0]-1);
	for(i=path[0];i!=-1;i=path[i])
	{
	   printf("%d ",i);
    }
	printf("\n");
}
return 0;
}
