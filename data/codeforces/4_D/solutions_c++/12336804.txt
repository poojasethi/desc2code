#include <bits/stdc++.h>

using namespace std;

const int N = 5010;

int n, w[N], h[N], to[N], d[N];

int dp(int v) {
  if (d[v]) return d[v];
  d[v] = 1;
  for (int i = 0; i <= n; ++i) {
    if (w[i] > w[v] && h[i] > h[v]) {
      if (dp(i) + 1 > d[v]) {
        to[v] = i;
        d[v] = d[i] + 1;
      }
    }
  }
  return d[v];
}

int main() {
  scanf("%d", &n);
  for (int i = 0; i <= n; ++i) {
    to[i] = -1;
    scanf("%d%d", w + i, h + i);
  }
  dp(0);
  printf("%d\n", --d[0]);
  for (int i = to[0]; i != -1; i = to[i]) {
    printf("%d ", i);
  }
  return 0;
}
