#include<cmath>
#include<iostream>
#include<vector>
using namespace std;
int main(){
    long long n;
    vector<long long>v;
    cin>>n;
    if(n%3)
        cout<<0;
    else{
        n/=3;
        int ans=0,a,b,c;
        for(long long i=2;i*i<=n;i++)
            if(n%i==0)
                v.push_back(i);
        for(int i=0;i<v.size();i++){
            a=v[i];
            int f=0;
            for(int j=0;j<v.size();j++)
                if((n/v[i])%v[j]==0){
                    b=v[j],c=n/v[i]/v[j];
                    if((a+b-c)%2==0&&(b+c-a)%2==0&&(c+a-b)%2==0&&(a+b-c)>0&&(b+c-a)>0&&(c+a-b)>0)
                        ans++;         
                }            
        }
        cout<<ans;
    }
}
