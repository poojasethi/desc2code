#include<iostream>
using namespace std;
#define LL long long
LL n,a[10000000],cnt,ans;
int main() {
	cin>>n;
	if(n%3) {cout<<0;return 0;}
	n/=3;
	for(LL i=1;i*i<=n;i++) if(n%i==0) a[++cnt]=i;
	for(int i=1;i<=cnt;i++)
		for(int j=cnt;j>=1;j--) {
			LL x=a[i],y=a[j],z=n/x/y;
			if(x*y*z!=n||((x+y+z)&1)||y>z) continue;
			if(x>y||x+y<=z||y+z<=x) break;
			if(x==y&&y==z) ans++;
			else if(x==y||y==z||x==z) ans+=3;
			else ans+=6;
		}
	cout<<ans<<endl;
	return 0;
}
