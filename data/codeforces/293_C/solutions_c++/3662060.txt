#include<iostream>
#include<cstdio>
#include<vector>
#include<cmath>

using namespace std;
typedef long long ll;

vector<ll> v;
ll n;

int main(){

   ll i,j,k;
   cin>>n;
   if(n%3){
    cout<<0;
    return 0;
   }
   n/=3ll;
   for(i=2;i*i<=n;++i)
    if(n%i==0)v.push_back(i);

    int ans=0;

   for(i=0;i<v.size();++i){
    ll d=n/v[i];
    for(j=v.size()-1;j>=0;--j){
        if(d%v[j])continue;
        ll vv=d/v[j];
        if(v[i]+v[j]<=vv)break;
        if((v[i]+v[j]+vv)%2==0 && v[i]+vv>v[j] && vv+v[j]>v[i])
            ans++;
    }
   }

    cout<<ans;

    return 0;
}
