#include<iostream>
#include<stdio.h>
#include<algorithm>
#include<vector>
#include<cstring>
#define MAX(a,b) (a>b?a:b)
#define MIN(a,b) (a<b?a:b)
#define pb push_back
#define mp make_pair
#define F first
#define S second
using namespace std;
long long n,i,j,q,A,B,C,ans;
vector<long long> v;

main()
{cin>>n;
 if(n%3!=0){cout<<0<<endl;return 0;}
 n/=3;
 // (a+b)(a+c)(b+c)
 for(i=2;i*i<=n;i++)
     if(n%i==0)v.pb(i);
 for(i=0;i<v.size();i++){
     q=n/v[i];
     A=v[i];
     for(j=i;j<v.size() && v[j]*v[j]<=q;j++){
     B=v[j];
     if(q%B!=0 || q/B==1)continue;
     C=q/B;
     if(A*B*C==n && A+B>C && B+C>A && A+C>B)
     if((A+B+C)%2==0){
       if(A==B && B==C)ans+=1;else
       if(A==B)ans+=3;else
       if(B==C)ans+=3;else
       ans+=6;
     }
     }
 }
 cout<<ans<<endl;
}
