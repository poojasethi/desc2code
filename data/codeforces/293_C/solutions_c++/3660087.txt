# include <iostream>
#include <math.h>
using namespace std;

int main ()
{
	long long pr[100000];
	int kol=0;
	long long n;
	cin>>n;
	if (n % 6 != 0) { cout<<0; return 0;}  
	long long ans=0;
	n=n/3;
	for (long long i=2;i*i<=n;i++)
	{
		if (n % i ==0) pr[kol++]=i;
	}
	for (long long q=0;q<kol && pr[q]*pr[q]*pr[q]<=n;q++)
	{
		for (long long a=q;pr[q]*pr[a]*pr[a]<=n && a<kol;a++)
		{
			long long k=n/pr[q]/pr[a];
			long long i = pr[q];
			long long j = pr[a];
			if ( i*j*k==n && (i+j+k) % 2 ==0 && (i+j-k>0))
			{
				if (i==j && i==k && j==k) ans++;
				else 
					if (i!=j && j!=k && k!=i) ans +=6;
					else ans+=3;
			}
		}
		
	}
	cout<<ans;
	return 0;
}
