#include <iostream>
#include <cstdio>
#include <cmath>
#include <set>
using namespace std;
#define LL long long
#define LD long double
int main(){
    LL N;
    cin >> N;
    if (N%3){
        cout << 0 << endl;
        return 0;
    }
    N/=3;
    LL ans=0;
    int lim1=pow(N,(LD)1.0/3),i,j;
    for (i=2;i<=lim1;i++){
        if (N%i) continue;
        int lim2=sqrt((LD)N/i);
        for (j=lim2;j>=i;j--)
            if (N%(i*j)==0){
                LL k=N/i/j;
                if ((i+j+k)&1) continue;
                if (i+j<=k) break;
                if (i==j){
                    if (j==k) ans++;
                    else ans+=3;
                }else{
                    if (j==k) ans+=3;
                    else ans+=6;
                }
            }
    }
    cout << ans << endl;
}
