#include <cstdio>

int t;
long long n, x, y, z, v, res;
long long q[10000000];

int main()
{
	scanf("%I64d", &n);
	if (n % 3)
	{
	 	puts("0");
	 	return 0;
	}

	n /= 3;

	for (x = 2; x*x <= n; x++) if (n % x == 0) q[t++] = x;
	for (int ix=0; ix < t; ix++)
	{
	 	x = q[ix];
	 	if (x*x*x > n) break;

	 	v = n/x;
		for (int iy = ix; iy < t; iy++)
		{
		 	y = q[iy];
		 	if (y*y > v) break;

			if (v % y) continue;
			z = v/y;

			if (!((z+y+x)&1) && z<y+x)
			{
				if (x == z)
					res++;
				else
				if (x!=y&&x!=z&&y!=z)
					res += 6;
				else
					res += 3;
			}
		}
	}

	printf("%I64d\n", res);

	return 0;
}
