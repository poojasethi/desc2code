#include <iostream>
#include <vector>

using namespace std;

typedef long long ll;

ll d[1000000], ds = 0, n, q, ans = 0;

int main () {
	cin >> n;
	if (n % 3 == 0) {
		n /= 3;
		for (q = 1; q * q <= n; ++q)
			if (n % q == 0)
				d[ds++] = q;
		for (int i = 0; i < ds; ++i) {
			ll a = d[i], p = n / a;
			for (int j = 0; j < ds && d[j] <= p; ++j) {
				ll b = d[j], c = p / d[j];
				if (!(p % d[j] || (a + b + c) % 2) && a + b > c && a + c > b && b + c > a)
					ans++;
			}
		}
	}
	cout << ans << endl;
}
