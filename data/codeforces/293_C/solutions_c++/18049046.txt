#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
typedef long long ll;
ll i,j,k,n,t;
int ans;
int cal(int x,int y,int z)
{
	if((x+y-z&1)||(x+z-y&1)||(y+z-x&1))return 0;
	if(x+y<=z||x+z<=y||y+z<=x)return 0;
	if(x==y&&y==z)return 1;
	if(x==y||y==z||x==z)return 3;
	return 6;
}
int main()
{
	scanf("%I64d",&n);
	if(n%3)return puts("0"),0;
	for(n/=3,i=2;i*i*i<=n;i++)if(n%i==0)
		for(t=n/i,j=max(i,ll((sqrt(i*i+4*t)-i)/2));j*j<=t;j++)
			if(t%j==0)ans+=cal(i,j,t/j);
	printf("%d",ans);
}
