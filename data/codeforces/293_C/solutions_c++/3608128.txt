#include <iostream>
#include <vector>

using namespace std;

#define forn(i,n) for (int i = 0; i < int(n); i++)

typedef long long ll;

vector<ll> d;

int main () {
	ll n;
	cin >> n;
	if (n % 3) {
		cout << 0 << endl;
		return 0;
	}
	n /= 3;
	for (ll q = 1; q * q <= n; ++q) {
		if (n % q == 0) {
			d.push_back(q);
		}
	}
	ll ans = 0;
	forn (i, d.size()) {
		ll a = d[i];
		ll p = n / a;
		for (int i = 0; i < d.size() && d[i] * d[i] <= p; ++i) {
			if (p % d[i]) {
				continue;
			}
			ll b = d[i];
			ll c = p / d[i];
			if ((a + b + c) % 2) {
				continue;
			}
			if (a + b <= c || a + c <= b || b + c <= a) {
				continue;
			}
			
			ans++;
			if (d[i] * d[i] < p) {
				ans++;
			}
		}
	}
	cout << ans << endl;
	return 0;
}
