#include <iostream>
#include <cmath>

using namespace std;

typedef long long ll;

ll n, rt;

int tot = 0;
int check (ll a, ll b, ll c)
{ 
  // cout << a << " " << b << " " << c << endl;
  if ((a + b + c) % 2 != 0) return 0;
  if (a > b || b > c) return 0;
  if (a + b + c <= 2 * c) return 0;
  if (a == b && b == c) return 1;
  if (a == b || b == c) return 3;
  return 6;
}

void process()
{
  for (int i = 1; i <= rt; i++)
  {
    if (n % i != 0) continue;
    ll m = n / i;
    int sq = (int) sqrt (m + .5);
    for (int j = (sq - i / 2); j <= sq; j++)
    {
      if (m % j != 0) continue;
      tot += check (i, j, m / j);
    }
  }
}

int main()
{
  cin >> n;
  if (n % 6 == 0)
  {
    n /= 3;
    rt = (int) pow (n + .5, 1.0/3.0);
    process();
  }
  cout << tot << endl;
}
