#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cstring>
using namespace std ;
typedef long long LL ;

LL d0, d1, N ;
LL ans ;
LL D[100005] ;

int main()
{
	LL i, j, tmp ;
	LL a, b, c ;
	scanf("%I64d", &N) ;
	if(N%6) {puts("0\n"); return 0 ;}
	N /= 3 ; 
	if(N < 8) {puts("0\n"); return 0 ;}
	for(i = 2; i*i <= N; i ++)
		if(N%i == 0) D[++d0] = i ;
	for(i = 1; i <= d0; i ++)
		for(j = 1; j <= d0; j ++)
		{
			a = D[i], b = D[j] ;
			c = N/a/b ;
			if(N%(a*b) || (a+b+c)%2 || a+b <= c || a+c <= b || b+c <= a) ;
			else ans ++ ;
		}
	printf("%I64d\n", ans) ;
	//system("pause") ;
	return 0 ;
}
