#include<cstdio>
#include<cmath>
#include<iostream>
using namespace std;
typedef __int64 i64;
i64 gao(i64 k1,i64 k2)
{
    i64 k=k1*k1-k2*4;
    if(k==0)    return 1;
    i64 kk=sqrt(k+0.5);
    if(kk*kk!=k)    return 0;
    return 2;
}
i64 half(i64 m,i64 k)
{
    i64 kk=k*k,x=1,y=(int(sqrt(m+0.5)));
    while(y>x)
    {
        i64 i=x+(y-x)/2;
        if((m-i*(i+k))*4<=kk)
            y=i;
        else
            x=i+1;
    }
    return y;
}
i64 ask(i64 m,i64 k)
{
    i64 res=0;
    for(i64 i=half(m,k);i*(i+k)<m;++i)
        res+=gao(k,m-i*(i+k));
    return res;
}
i64 cal(i64 n)
{
    if(n%3) return 0;
    n/=3;
    i64 res=0;
    for(i64 i=2;i*i<n;++i)if(n%i==0)
        res+=ask(n/i,i);
    return res;
}
int main()
{
    i64 n;
    cin>>n;
    cout<<cal(n)<<endl;
    return 0;
}
