#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <cmath>
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int64 i,j,k,n,m,an;
inline int work(int a,int b,int c) {
	return a==b&&b==c?1:a==b||b==c?3:6;
}
int main() {
	scanf("%I64d",&n);
	if (n%3) {
		printf("0\n"); return 0;
	}
	n/=3;
	int A=(int)(pow(n,1./3)+1e-9);
	For(i,2,A) if (n%i==0) {
		int B=int(sqrt(n/i)+1e-9);
		for(j=B;j>=i;j--) if (n/i%j==0) {
			int C=n/i/j;
			if (i+j<=C) break;
			if (j<=C&&(i+j+C)%2==0) {
				int D=(i+j+C)/2;
				an+=work(D-C,D-j,D-i);
			}
		}
	}
	printf("%I64d\n",an);
	//for(;;);
	return 0;
}
