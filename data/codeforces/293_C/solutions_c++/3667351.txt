#include <cstdio>
long long l[1111111], n, ans, i, j, a, b, c;
int tot;
int main(){
	scanf("%I64d", &n);
	if (n % 3 != 0){
		puts("0");
		return 0;
	}
	n /= 3;
	for (i = 2; i * i <= n; i++)
		if (n % i == 0)
			l[++tot] = i;
	for (i = 1; i <= tot; i++)
		for (j = tot; j; j--){
			a = l[i]; b = l[j]; c = n / a / b;
			if (a * b * c != n || b > c || (a + b + c) & 1) continue;
			if (a > b || a + b <= c || b + c <= a) break;
			if (a == b && b == c && c == a) ans += 1;
			else if (a == b || b == c || c == a) ans += 3;
			else ans += 6;
		}
	printf("%I64d\n", ans);
}
