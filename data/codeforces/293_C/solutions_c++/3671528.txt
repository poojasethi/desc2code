#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int ans,tot;
long long i,j,k,n,x,a,b,c,p[12000];
int main(){
	scanf("%I64d",&n);
	if(n%3){
		puts("0");
		return 0;
	}
	n/=3;
	for(i=1;i*i<=n;++i)if(n%i==0)p[++tot]=i;
	for(i=1;i<=tot && p[i]*p[i]*p[i]<=n;++i){
		for(j=i;j<=tot && p[i]*p[j]*p[j]<=n;++j)if((n/p[i])%p[j]==0){
			k=n/p[i]/p[j];
			if((p[i]+p[j]+k)&1)continue;
			x=(p[i]+p[j]+k)/2;
			a=x-k;
			b=x-p[j];
			c=x-p[i];
			//printf("%I64d %I64d %I64d\n",a,b,c);
			if(a<=0)continue;
			if(a==b && a==c)ans++;
			else if(a==b || a==c || b==c)ans+=3;
			else ans+=6;
		}
	}
	printf("%d\n",ans);
//	system("pause");
}
