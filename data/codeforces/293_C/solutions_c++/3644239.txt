//program 293-C
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long LL;
#define N 1001000
LL l[N],tot,n,ans;

int main()
{
	cin>>n;
	if (n%3) return cout<<0,0;
	n/=3;
	for (LL i=2;i*i<=n;i++)
		if (n%i==0)
			l[tot++]=i;
	for (LL i=0;i<tot;i++)
		for (LL j=tot-1;j>=0;j--)
			{
				LL a=l[i], b=l[j], c=n/l[i]/l[j];
				if ((a+b+c)&1||a*b*c!=n||b>c) continue;
				if (a>b||a+b<=c||b+c<=a) break;
				if (a==b&&a==c) ans++;
				else if (a==b||a==c||b==c) ans+=3;
				else ans+=6;
			}
	cout<<ans<<endl;
}
