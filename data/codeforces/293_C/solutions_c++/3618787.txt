#include <iostream>
#include <cstring>
#include <cstdio>
#include <vector>
#include <cmath>
using namespace std;
typedef long long ll;
ll n;
vector<ll> fac;
void getFac()
{
	for (ll i = 1; i * i * i <= n; ++i)
		if (n % i == 0) fac.push_back(i);
}

int main()
{
	cin >> n;
	if (n % 6 != 0) puts("0");
	else
	{
		n /= 3;
		getFac();
		int ans = 0;
		for (int i = 0; i < fac.size(); ++i)
		{
			ll x = fac[i];
			ll m = n / x;
			ll k = sqrt(m + 0.5);
			for (ll y = k; (y + x) * y > m && x <= y; --y)
			{
				if (m % y) continue;
				ll z = m / y;
				if (x + y - z & 1) continue;
				ll a= (x + y - z) / 2, b = (x + z - y) / 2, c = (y + z - x) / 2;
				if (a == b && b == c) ans++;
				else if (a == b || b == c) ans += 3;
				else ans += 6;
			}
		}
		cout << ans << endl;
	}
	return 0;
}
