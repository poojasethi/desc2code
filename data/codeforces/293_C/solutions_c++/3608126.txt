#include <stdio.h>
#include <math.h>
long long n,cuantos,falta,i,j,k;
int main() {
    scanf("%I64d",&n);
    if(n%3>0) {
        printf("0");
        return 0;
    }
    n/=3;
    for(i=1; i*i*i<=n; i++) {
        if(n%i!=0) continue;
        falta=n/i;
        for(j=sqrt(falta); j>0; j--) {
            if(((i+j)<(falta/j))||(i>j)) break;
            if(falta%j!=0) continue;
            k=falta/j;
            if(((((i&1)+(j&1)+(k&1))&1)==0)&&(i+j)>k){
                if((i==j)&&(j==k))
                    cuantos++;
                else if(i == j || j == k || i == k)
                    cuantos+=3;
                else
                    cuantos+=6;
            }
        }
    }
    printf("%I64d",cuantos);
}
