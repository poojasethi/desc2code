#!/usr/bin/python
import sys

n = int(raw_input())
a = [sys.stdin.read(1) for i in range(2*n)]
a = map(lambda x: int(x), a)
a.reverse()

d = [[0 for i in range(n+1)] for j in range(n+1)]
for i in range(1, n+1):
	d[i][0] = a[i-1]*pow(10, i-1) + d[i-1][0]
	d[0][i] = a[i-1]*pow(10, i-1) + d[0][i-1]

for i in range(1, n+1):
	for j in range(1, n+1):
		d[i][j] = max(a[i+j-1]*pow(10, i-1) + d[i-1][j],
				      a[i+j-1]*pow(10, j-1) + d[i][j-1])

i = n
j = n
s = ''
while i > 0 and j > 0:
	if d[i][j] == a[i+j-1]*pow(10, i-1) + d[i-1][j]:
		s += 'H'
		i -= 1
	else:
		s += 'M'
		j -= 1

if i == 0:
	s += 'M' * j
else:
	s += 'H' * i
print s
