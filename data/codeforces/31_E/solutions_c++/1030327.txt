#define LL long long
#define mp(x, y) make_pair(x, y)
#define pb(x) push_back(x)
#define size(S) S.size()
#define PII pair<int, int>
#define PID pair<int, double>

using namespace std;

int n, m;
LL f[20][20], e[20];
string s, p[20][20];

int main(){
	cin >> n >> s;
	e[0] = 1;
	for (int i = 1; i <= n; i++) e[i] = e[i - 1] * 10;
	for (int i = 0; i <= n; i++)
		for (int j = 0; j <= n; j++){
			if (i < n && f[i][j] + (s[i + j] - '0') * e[n - i - 1] >= f[i + 1][j]){
				f[i + 1][j] = f[i][j] + (s[i + j] - '0') * e[n - i - 1];
				p[i + 1][j] = p[i][j] + 'H';
			}
			if (j < n && f[i][j] + (s[i + j] - '0') * e[n - j - 1] >= f[i][j + 1]){
				f[i][j + 1] = f[i][j] + (s[i + j] - '0') * e[n - j - 1];
				p[i][j + 1] = p[i][j] + 'M';
			}
		}
	cout << p[n][n] << endl;
}
