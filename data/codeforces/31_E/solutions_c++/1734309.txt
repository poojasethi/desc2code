#include<cstdio>
#include<cstdlib>
int n,tag[40][40],n_s=0;
long long dp[40][40],base10[20];
char s[51],ans[81];
int main(){
    scanf("%d",&n);
    scanf("%s",&s);
    base10[0] = 1;
    for(int i=1;i<=18;i++) base10[i] = base10[i-1] * 10;
    for(int i=1;i<=2*n;i++)
       for(int j=0;j<=18&&j<=i;j++){
          int k = i-j;
          if(j != 0 && dp[j][k] < dp[j-1][k] + (long long)(s[2*n-i]) * base10[j-1])
              dp[j][k] = dp[j-1][k] + (long long)(s[2*n-i]) * base10[j-1],tag[j][k] = 1;
          if(k != 0 && dp[j][k] < dp[j][k-1] + (long long)(s[2*n-i]) * base10[k-1])
              dp[j][k] = dp[j][k-1] + (long long)(s[2*n-i]) * base10[k-1],tag[j][k] = 2;
          //printf("%d %d %I64d\n",j,k,dp[j][k]);
       }
    int now1 = n,now2 = n;
    while(now1!=0||now2!=0){
        if(tag[now1][now2] == 1) ans[n_s++] = 'H',now1--;
        else ans[n_s++] = 'M',now2--; 
    }
    printf("%s\n",ans);
    //system("pause");
    return 0;
}
