#include <cstdio>
#include <algorithm>
#include <cmath>
#include <iostream>
#include <map>
#include <set>
#include <vector>

using namespace std;
char s[40];
long long w[19],f[19][19];
int n;

int main()
{
    scanf("%d%s",&n,s);
	w[1] = 1;
	for (int i = 2; i <= n; ++i) w[i] = w[i-1]*10;
	for (int i = 0; i <= n; ++i)
		for (int j = 0; j <= n; ++j) {
			int x=s[n*2-i-j]-48;
			if (i) f[i][j]=f[i-1][j]+x*w[i];
			if (j) f[i][j]=max(f[i][j],f[i][j-1]+x*w[j]);
		}
    int i = n, j = n;
	while (i > 0 || j > 0) {
		int x=s[n*2-i-j]-48;
		if (i && f[i-1][j]+x*w[i]==f[i][j]) --i,printf("H");
		else --j, printf("M");
	}
	return 0;
}
