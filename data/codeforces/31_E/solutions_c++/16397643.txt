#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
const int N = 20;
ll dp[N][N],pm[N];
int n;
string s;

int main()
{
    ios_base::sync_with_stdio(0); cin.tie(0);
    #ifndef ONLINE_JUDGE
    freopen("TVGame.inp","r",stdin);
    freopen("TVGame.out","w",stdout);
    #endif // ONLINE_JUDGE
    cin >> n >> s;
    pm[1]=1;
    for (int i=2; i<=n; i++) pm[i]=pm[i-1]*10;
    for (int i=0; i<=n; i++)
        for (int j=0; j<=n; j++){
            int num = s[2*n-i-j]-'0';
            if (i) dp[i][j] = max(dp[i][j],dp[i-1][j]+num*pm[i]);
            if (j) dp[i][j] = max(dp[i][j],dp[i][j-1]+num*pm[j]);
        }
    int i=n, j=n;
    //cout << dp[i][j] << '\n';
    while (i || j) {
        int num = s[2*n-i-j] - '0';
        if (i && dp[i-1][j] + num*pm[i] == dp[i][j]) {cout << 'H'; i--; continue;}
        if (j && dp[i][j-1] + num*pm[j] == dp[i][j]) {cout << 'M'; j--;}
    }
    return 0;
}
