#include <cstdio>
#include <string>
#include <iostream>
#include <cstring>

typedef long long LL;

LL f[21][21];
bool r[21][21];
LL Pow10[21];
char s[21];

int main() {
	int n; scanf("%d%s", &n, s);
	Pow10[1] = 1ll;
	for (int i = 2; i <= n; ++i) Pow10[i] = Pow10[i - 1] * 10ll;

	memset(f, -1, sizeof(f)); f[0][0] = 0ll;
	for (int i = 0; i <= n; ++i)
		for (int j = 0; j <= n; ++j) {
			LL now = (LL) (s[i + j] - '0');

			if (i < n && f[i + 1][j] < f[i][j] + now * Pow10[n - i])
				f[i + 1][j] = f[i][j] + now * Pow10[n - i], r[i + 1][j] = true;
			if (j < n && f[i][j + 1] < f[i][j] + now * Pow10[n - j])
				f[i][j + 1] = f[i][j] + now * Pow10[n - j], r[i][j + 1] = false;
		}

	std::string fuck = "";

	int i = n, j = n;
	while (i + j) {
		if (r[i][j]) fuck = "M" + fuck, i--;
		else fuck = "H" + fuck, j--;
	}

	std::cout << fuck;

	return 0;
}
