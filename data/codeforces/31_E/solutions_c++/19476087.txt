#include <bits/stdc++.h>
using namespace std;
#define int long long

const int maxn = 20;

int dp[maxn][maxn];
int n;
int p[maxn];

string s;
string ans;

void input() {
	cin >> n >> s;
}

void setp() {
	p[0] = 1;
	for (int i = 1; i < maxn; i++)
		p[i] = 10 * p[i-1];
}

void DP() {
	for (int i = 0; i <= n; i++)
		for (int j = 0; j <= n; j++) {
			if (i)
				dp[i][j] = dp[i-1][j] + (s[2*n-i-j]-'0') * p[i-1];
			
			if (j)
				dp[i][j] = max(dp[i][j], dp[i][j-1] + (s[2*n-i-j]-'0')*p[j-1]);
	}
}

void print() {

	for (int i = n, j = n; i || j; ) {
		if (i && dp[i-1][j] + (s[2*n-i-j]-'0')*p[i-1] == dp[i][j]) {
			ans += 'H';
			i--;
		}
		else if (j) {
			ans += 'M';
			j--;
		}
	}
	cout << ans << endl;
}

main() {
	input();
	setp();
	DP();
	print();
	return 0;
}
