#include <cstdio>
#include <cstring>
#include <string>
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long i64;

int M;
char S[40];
i64 DP[20][20], P[20][20], pow10[20];

int main() {

    scanf("%d %s", &M, S);
    memset(DP, -1, sizeof(DP));
    pow10[M-1] = 1;
    for (int i = M-2; i >= 0; i--) pow10[i] = pow10[i+1] * 10;
    DP[0][0] = 0;
    i64 c;
    for (int i = 0; i <= M; i++) for (int j = 0; j <= M; j++) {
        if (i < M && DP[i+1][j] < (c = DP[i][j] + pow10[i]*(S[i+j]-'0')))
            DP[i+1][j] = c, P[i+1][j] = 'M';
        if (j < M && DP[i][j+1] < (c = DP[i][j] + pow10[j]*(S[i+j]-'0')))
            DP[i][j+1] = c, P[i][j+1] = 'H';
    }
    string ret;
    int i = M, j = M;
    while (i || j) {
        ret = string(1,P[i][j]) + ret;
        if (P[i][j]=='M') i--;
        else j--;
    }
    cout << ret << endl;

    return 0;
}
