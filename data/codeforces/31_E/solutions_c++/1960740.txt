#include <iostream>

using namespace std;

const int MAXn=20;
long long int d[MAXn][MAXn],base[MAXn];
string s;

int main()
{
    base[1]=1;
    int n;
    cin>>n>>s;
    for(int i=2;i<=n;i++)
        base[i]=base[i-1]*10;
    for(int i=0;i<=n;i++)
        for(int j=0;j<=n;j++)
        {
            int a=s[2*n-i-j]-'0';
            if(i-1>=0)
                d[i][j]=d[i-1][j]+a*base[i];
            if(j-1>=0)
                d[i][j]=max(d[i][j],d[i][j-1]+a*base[j]);
        }
    int i,j;
    i=j=n;
    while(i>=1 || j>=1)
    {
        int a=s[2*n-i-j]-'0';
        if(d[i][j]==d[i-1][j]+base[i]*a && i>=1)
        {    
            cout<<"H";
            i--;
        }
        else
        {    
            cout<<"M";
            j--;
        }
    }
    //system("pause");
}
