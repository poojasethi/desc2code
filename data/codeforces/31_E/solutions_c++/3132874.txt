#include<iostream>
#include<cstring>
using namespace std;
const int mm=100;
long long f[mm][mm],w[mm];
int main()
{
  int m;char s[mm];
  while(cin>>m>>s)
  { w[1]=1;
    for(int i=2;i<=18;i++)
      w[i]=w[i-1]*10;
    memset(f,0,sizeof(f));
    for(int i=0;i<=m;i++)
      for(int j=0;j<=m;j++)
    { long long z=s[m+m-i-j]-'0';
      if(i)f[i][j]=f[i-1][j]+w[i]*z;
      if(j)f[i][j]=f[i][j]>f[i][j-1]+w[j]*z?f[i][j]:f[i][j-1]+w[j]*z;
    }
    for(int i=m,j=m;i>0||j>0;)
    { long long z=s[m+m-i-j]-'0';
      if(i&&f[i][j]-f[i-1][j]==z*w[i])
        {--i;cout<<"H";
        }
      else {--j;cout<<"M";}
    }
    cout<<"\n";
  }
}
