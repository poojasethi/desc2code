#include<iostream>
#include<algorithm>
#include<string.h>
using namespace std;
char num[40];
long long  dp[20][20];
long long w[20];
int main()
{
	w[1]=1;
	for(int i=2;i<=18;i++)
	{
	    w[i]=w[i-1]*10;
	}
	int n;
	while(cin>>n)
	{
		memset(dp,0,sizeof(dp));
		for(int i=0;i<2*n;i++)
		{
			cin>>num[i];
		}
		for(int i=0;i<=n;i++)
		{
			for(int j=0;j<=n;j++)
			{
				long long k=num[n+n-i-j]-'0';
				if(i>0)
				{
					dp[i][j]=dp[i-1][j]+w[i]*k;
				}
				if(j>0)
				{
					dp[i][j]=max(dp[i][j],dp[i][j-1]+w[j]*k);
				}
			}
		}
		//cout<<dp[n][n]<<endl;
		for(int i=n,j=n;i>0||j>0;)
		{
			long long k=num[n+n-i-j]-'0';
			if(i&&dp[i][j]-dp[i-1][j]==w[i]*k)
			{
				cout<<"H";
				i--;
			}
			else
			{
				cout<<"M";
				j--;
			}
		}
		cout<<endl;
	}
}
