#include <bits/stdc++.h>
 
#define fi first
#define se second
#define pb push_back
#define mp make_pair
#define pii pair<int,int>
#define MOD 4001
#define ll long long int
#define MAXINT 100000000
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

int n;
ll dp[50][50][50];
char P[50][50][50];
char s[100];
ll ten[20];

ll work(int p,int x,int y) {
	ll q1=-1,q2=-1;
	if(dp[p][x][y]!=-1)	return dp[p][x][y];
	if(p==2*n)			return dp[p][x][y]=0;
	if(x)
		q1=work(p+1,x-1,y)+(s[p]-'0')*ten[x-1];
	if(y)
		q2=work(p+1,x,y-1)+(s[p]-'0')*ten[y-1];
	if(q1>q2) {
		P[p][x][y]='A';
		dp[p][x][y]=q1;
	}
	else {
		P[p][x][y]='B';
		dp[p][x][y]=q2;
	}
	return dp[p][x][y];
}

void print(int p,int x,int y) {
	if(p==2*n)	return;
	if(P[p][x][y]=='A') {
		cout<<"M";
		print(p+1,x-1,y);
	}
	else {
		cout<<"H";
		print(p+1,x,y-1);
	}
}

int main () {
	int i;
	cin>>n;
	cin>>s;
	ten[0]=1;
	for(i=1;i<20;i++)	ten[i]=ten[i-1]*10;
	SET(dp,-1);
	work(0,n,n);
	print(0,n,n);
	cout<<endl;
	return 0;
}
