#include<stdio.h>
#include<string.h>

char a[42],tr[21][21];
long long re[21][21],mul[21];

int main()
{
    int i,n,j;

    while(scanf("%d\n",&n)==1)
    {
        gets(a);

        mul[n-1] = 1;
        for(i=n-2;i>-1;i--) mul[i] = mul[i+1]*10;
        memset(re,-1,sizeof re);
        re[0][0]=0;
        for(i=0;i<=n;i++)   for(j=0;j<=n;j++) if(re[i][j]!=-1)
        {
            if(i<n && re[i][j]+mul[i]*(a[i+j]-'0')>re[i+1][j] )
            {
                re[i+1][j] = re[i][j]+mul[i]*(a[i+j]-'0');
                tr[i+1][j] = 'H';
            }
            if(j<n && re[i][j]+mul[j]*(a[i+j]-'0')>re[i][j+1] )
            {
                re[i][j+1] = re[i][j]+mul[j]*(a[i+j]-'0');
                tr[i][j+1] = 'M';
            }
        }
        //printf("%I64d\n",re[n][n]);
        for(i=j=n;i||j; tr[i][j]=='M'?j--:i--)    a[i+j-1]=(tr[i][j]=='M')?'H':'M';
        printf("%s\n",a);
    }

    return 0;
}
