#include <iostream>
#include <string>
#include <utility>
using namespace std;
typedef long long ll;
const int MN=18+10;
ll d[MN][MN];
ll pt[MN];
int par[MN][MN];
bool mark[MN][MN];
int n;
string no,res;
ll find(int a,int b)
{
	if (a==0 && b==0)
		return 0;
	if (mark[a][b])
		return d[a][b];
	mark[a][b]=1;
	ll & res=d[a][b];
	res=-1;
	if (a>0)
	{
		ll t1=find(a-1,b);
		if (t1+(no[2*n-a-b]-'0')*pt[a-1]>res)
		{
			res=t1+(no[2*n-a-b]-'0')*pt[a-1];
			par[a][b]=1;
		}
	}
		if (b>0)
	{
		ll t1=find(a,b-1);
		if (t1+(no[2*n-a-b]-'0')*pt[b-1]>res)
		{
			res=t1+(no[2*n-a-b]-'0')*pt[b-1];
			par[a][b]=2;
		}
	}
		return res;
}
void print(int a,int b)
{
	if (a==0 && b==0)
		return ;
	if (par[a][b]==1)
	{
		res[2*n-a-b]='H';
		print(a-1,b);
		return ;
	}
	res[2*n-a-b]='M';
	print(a,b-1);
}
int main()
{
	ios::sync_with_stdio(false);
	cin>>n>>no;
	pt[0]=1;
	for (int i=1;i<=18;i++)
		pt[i]=10*pt[i-1];
	find(n,n);
	res=no;
	print(n,n);
	cout<<res<<endl;
	return 0;
}
