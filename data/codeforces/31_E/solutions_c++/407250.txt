#include<stdio.h>
#include<string.h>
#include<vector>
using namespace std;

int main(){
	int i, j, k, n;
	long long dp[19][19] = {0}, e[18];
	vector<char> ans[19][19];
	char num[50];
	e[0] = 1;
	for(i = 1; i < 18; i++) e[i] = e[i - 1] * 10;
	scanf("%d", &n);
	scanf("%s", num + 1);
	for(i = 1; num[i]; i++) num[i] -= '0';
	for(i = 1; i <= n; i++){
		dp[i][0] = dp[i - 1][0] + e[n - i] * num[i];
		ans[i][0] = ans[i - 1][0];
		ans[i][0].push_back('H');
		ans[0][i] = ans[0][i - 1];
		ans[0][i].push_back('M');
		for(j = 1; j <= n; j++){
			dp[i][j] = dp[i][j-1]+e[n-j]*num[i+j];
			ans[i][j] = ans[i][j-1];
			ans[i][j].push_back('M');
			if(dp[i-1][j]+e[n-i]*num[i+j] > dp[i][j]){
				dp[i][j] = dp[i-1][j]+e[n-i]*num[i+j];
				ans[i][j] = ans[i-1][j];
				ans[i][j].push_back('H');
			}
			//printf("dp[%d][%d] = %lld\n", i, j, dp[i][j]);
		}
	}
	for(i = 0; i < ans[n][n].size(); i++) putchar(ans[n][n][i]); puts("");
	scanf("%*d");
}
