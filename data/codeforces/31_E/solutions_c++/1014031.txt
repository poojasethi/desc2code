#include <iostream>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

typedef long long LL;
LL f[20][20],p[20];
int n;
string s,g[20][20];

int main(){
        cin>>n>>s;
        int i,j;
        p[1]=1;REP(i,1,n)p[i+1]=p[i]*10LL;
        f[0][0]=0;g[0][0]="";
        REP(i,0,n+1)REP(j,0,n+1){
                if (i<n&&f[i][j]+p[n-i]*(s[i+j]-'0')>=f[i+1][j]){
                        f[i+1][j]=f[i][j]+p[n-i]*(s[i+j]-'0');
                        g[i+1][j]=g[i][j]+'H';
                }
                if (j<n&&f[i][j]+p[n-j]*(s[i+j]-'0')>=f[i][j+1]){
                        f[i][j+1]=f[i][j]+p[n-j]*(s[i+j]-'0');
                        g[i][j+1]=g[i][j]+'M';
                }
        }
        //cout<<f[n][n]<<endl;
        cout<<g[n][n]<<endl;
}
