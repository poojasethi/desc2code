#include <bits/stdc++.h>
using namespace std;
int n;
char str[55];
char op[55], temp[55];
long long mark[25][25];
long long A[25];
void init()
{
	A[0] = 1; 
	for(int i = 1; i < 18; i++) A[i] = A[i-1]*10;
}
void dfs(int cur, int a, int b, long long val)
{
//	cout<<cur<<" "<<a<<" "<<b<<" "<<val<<" "<<mark[a][b]<<endl;
//	if(cur > 2*n) return;
	if(a < 0 || b < 0) return;
	if(mark[a][b] == -1) mark[a][b] = val;
	else if(mark[a][b] < val) mark[a][b] = val;
	else return;
	
	if(cur == 2*n)
	{
	//	cout<<"11111"<<endl;
		for(int i = 0; i < 2*n; i++) op[i] = temp[i];
		return;
	}

	temp[cur] = 'H';
	dfs(cur+1, a-1, b, val + (str[cur]-'0')*A[a-1]);
	temp[cur] = 'M';
	dfs(cur+1, a, b-1, val + (str[cur]-'0')*A[b-1]);
}
int main()
{
	init();
	while(~scanf("%d", &n))
	{
		scanf("%s", str);
		memset(mark, -1, sizeof(mark));
		dfs(0, n, n, 0);
		for(int i = 0; i < 2*n; i++) cout<<op[i];
		cout<<endl;
	}
	return 0;
}
