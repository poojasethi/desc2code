#include <iostream>
#include <cstring>
using namespace std;

typedef long long tal;

string number;
int n;

tal powten[] = {
1,
10,
100,
1000,
10000,
100000,
1000000,
10000000,
100000000,
1000000000,
10000000000,
100000000000,
1000000000000,
10000000000000,
100000000000000,
1000000000000000,
10000000000000000,
100000000000000000,
1000000000000000000
};
tal f[19][19];
bool choice[19][19];

tal F(int x, int y)
{
	if(x+y==0)
		return 0;
	if(f[x][y] != -1)
		return f[x][y];
	tal A = -1,
		B = -1;
	int d = int(number[2*n-x-y]) - 48;
	if(x-1>=0)
		A = d*powten[x-1] + F(x-1,y);
	if(y-1>=0)
		B = d*powten[y-1] + F(x,y-1);
	choice[x][y] = (A>B);
	return f[x][y] = max(A,B);	
}

int main()
{
	cin >> n >> number;
	for(int i =0;i<=n;++i)for(int j=0;j<=n;++j)f[i][j]=-1;
//cout << "score:" <<	F(n,n)<<endl;
	F(n,n);
	int x = n;
	int y = n;
	while(x+y>0)
		if(choice[x][y])
		{
			cout << "M";
			--x;
		}
		else
		{
			cout << "H";
			--y;
		}
	cout << endl;
	return 0;
}
