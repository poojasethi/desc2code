#include<bits/stdc++.h>
#define ll long long int
#define SET(x,y) memset(x,y,sizeof(x));

using namespace std;

ll dp[100][100][100];
ll ten[50];
ll flag[100][100][100];
char A[100],C[100];

ll fun(ll nh,ll nm,ll n){
    ll wh=-1,wm=-1;

	if(n<0) return 0;

	if(dp[nh][nm][n]!=-1) return dp[nh][nm][n];
    
    if(nh) wh=(A[n]-'0')*ten[nh-1]+fun(nh-1,nm,n-1);

    if(nm) wm=(A[n]-'0')*ten[nm-1]+fun(nh,nm-1,n-1);

	if(wh>wm){
		flag[nh][nm][n]=0;
		return dp[nh][nm][n]=wh;
	}
	else{
		flag[nh][nm][n]=1;
		return dp[nh][nm][n]=wm;
	}
}

int main(){
    ll sum,i,j,k,n;
	
	SET(dp,-1);
	SET(flag,-1);

   
    cin>>n;
    cin>>A;
    ten[n-1]=1;
    for(i=n-2;i>=0;i--)
    	ten[i]=ten[i+1]*10;
    
    fun(n,n,2*n-1);
    j=k=n;
    i=2*n-1;
    while(i>=0){

    	if(flag[j][k][i]==0){
    		C[i]='H';
    		i--;
    		j--;
    	}
    	else{
    	    C[i]='M';
    		i--;
    		k--;
    	}
    }
    
    cout<<C<<endl;

	return 0;
}
