#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int mm=101;
long long f[mm][mm],w[mm];
char s[mm];
int main()
{
  w[1]=1;
  for(int i=2;i<=18;i++)w[i]=w[i-1]*10;
  int m;
  while(cin>>m>>s)
  { memset(f,0,sizeof(f));
    for(int i=0;i<=m;i++)
      for(int j=0;j<=m;j++)
    { long long z=s[m+m-i-j]-'0';
      if(i)f[i][j]=f[i-1][j]+z*w[i];
      if(j)f[i][j]=f[i][j]>f[i][j-1]+z*w[j]?f[i][j]:f[i][j-1]+z*w[j];
    }
    for(int i=m,j=m;i>0||j>0;)
    {
      int z=s[m+m-i-j]-'0';
      if(i&&f[i-1][j]+z*w[i]==f[i][j])
      {
        i--;cout<<"H";
      }
      else {j--;cout<<"M";}
    }
    cout<<"\n";
  }
}
