#include<cstdio>
#include<algorithm>
using namespace std;

const int N = 20;
char s[N+N], res[N+N];
long long f[N][N], pow[N];
int n;
#ifdef ONLINE_JUDGE
    const char p1 = 'H', p2 = 'M';
#else
    const char p1 = 'C', p2 = 'K';
#endif

int main() {
#ifndef ONLINE_JUDGE
    freopen("07.inp", "r", stdin);
    freopen("07.out", "w", stdout);
#endif
    scanf("%d%s",&n,s);
    pow[0] = 1; for(int i = 1; i <= n; ++i) pow[i] = pow[i-1] * 10;
    for(int i = 0; i <= n; ++i)
        for(int j = 0; j <= n; ++j) {
            int digit = s[n+n-i-j] - 0x30;
            f[i][j] = max(i ? f[i-1][j] + digit*pow[i-1] : 0, j ? f[i][j-1] + digit*pow[j-1] : 0);
        }
    for(int i = n, j = n; i || j; ) {
        int digit = s[n+n-i-j] - 0x30;
        if(i == 0 || i && f[i][j] != f[i-1][j] + digit*pow[i-1]) 
            res[n+n-i-j] = p2, --j;
        else res[n+n-i-j] = p1, --i;
    }
    puts(res);
    return 0;
}
