#include <iostream>
#include <cstdio>
#include <string>
#include <cstring>
using namespace std;

int N,t1,t2;
unsigned long long int ans,S[40],dp[21][21][45],D[21];
char next[21][21][45];
string s;
bool visit[21][21][45];

unsigned long long int cal(int a,int b,int x){
  if(visit[a][b][x]) return dp[a][b][x];
  visit[a][b][x] = 1;
  if(x < 2*N){
    if(a<N && cal(a+1,b,x+1)+D[a]*S[x] >= dp[a][b][x]){
		dp[a][b][x] = dp[a+1][b][x+1]+D[a]*S[x];
		next[a][b][x] = 'H';
	}
    if(b<N && cal(a,b+1,x+1)+D[b]*S[x] >= dp[a][b][x]){
		dp[a][b][x] = dp[a][b+1][x+1]+D[b]*S[x];
		next[a][b][x] = 'M';
	}
  }
  return dp[a][b][x];
}

int main(){
    scanf("%d",&N);
    cin>>s;
    D[N-1] = 1;
    for(int i=N-2;i>=0;--i) D[i] = 10llu*D[i+1];
    for(int i=0;i<s.length();++i) S[i] = s[i]-'0';
    cal(0,0,0);
    for(int i=0;i<2*N;++i){
		printf("%c",next[t1][t2][i]);
		if(next[t1][t2][i] == 'H') ++t1;
		else ++t2;
	}
	printf("\n");
//	system("pause");
   // return 0;
}
