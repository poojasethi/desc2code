#include <cstdio>
#include <string>
#include <cstring>
#include <algorithm>
using namespace std;
typedef pair<long long,int> PLL;
int main(){
    int n,a[40];
    scanf("%d",&n);
    PLL dp[20][20];
    for(int i=1;i<=2*n;i++) scanf("%1d",&a[i]);
    long long e[20]={1};
    for(int i=1;i<=n;i++) e[i]=e[i-1]*10;
//  for(int i=0;i<=5;++i)printf("%I64d ",e[i]);puts("");
    memset(dp,0,sizeof(dp));
    for(int i=0;i<=n;i++) 
        for(int j=0;j<=n;j++){
            dp[i][j]={(long long)0,-1};//pair compare 是比first 關鍵字,,seceond=laststep
            if(i) dp[i][j]=max(dp[i][j],PLL(dp[i-1][j].first+a[i+j]*e[n-i],1));//用二維限制都選nsteps
            if(j) dp[i][j]=max(dp[i][j],PLL(dp[i][j-1].first+a[i+j]*e[n-j],0));
        }
    string ans;
    for(int x=n,y=n,c;x|y;(c?x:y)--) ans=((c=dp[x][y].second)?'M':'H')+ans;
    puts(ans.c_str());
}
