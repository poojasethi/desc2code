#include <iostream>
#include <cstdlib>
#include <algorithm>
#include <vector>
#include <set>
#include <map>
#include <cstring>
#include <cmath>
using namespace std;
#define mp make_pair
#define int long long
int dp[40][40],p[40],a[40],n,par[40][40];
void print(int x,int y)
{
  if(x==2*n)
    return;
  if(par[x][y])
    cout<<'M',y--;
  else
    cout<<'H';
  print(x+1,y);
}

main()
{
  ios::sync_with_stdio(false);
  p[0]=1;
  for(int i=1;i<=18;i++)
    p[i]=p[i-1]*10ll;
  cin>>n;
  for(int i=0;i<2*n;i++)
    {
      char c;
      cin>>c;
      a[i]=c-'0';
    }
  for(int i=2*n-1;i>=0;i--)
    for(int j=0;j<=min(2*n-i,18ll);j++)
      {
	if(j!=0)
	  {
	    dp[i][j]=dp[i+1][j-1]+p[j-1]*a[i];
	    par[i][j]=1;
	  }
	int s=2*n-i-j,x=dp[i+1][j]+p[s-1]*a[i];
	if(dp[i][j]<x)
	  dp[i][j]=x,par[i][j]=0;
      }
  cerr<<dp[0][n]<<endl;
  print(0,n);
  return 0;
}
