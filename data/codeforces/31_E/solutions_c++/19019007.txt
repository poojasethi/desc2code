#include <iostream>
#include <map>
#include <algorithm>
#include <cmath>
#include <string.h>
#include <set>
#include <cstdio>
#include <queue>
#include <cstdio>
#include <vector>
#define N 25
using namespace std;

int n;
string s;
long long f[N][N],w[N];
int main()
{
	while(cin>>n)
	{
		cin>>s;
		memset(f,0,sizeof(f));
		w[1] = 1;
		for(int i = 2; i <= 18; i++)
		{
			w[i] = w[i-1]*10;
		}
		for(int i = 0; i <= n; i++)
		{
			for(int j = 0; j <= n; j++)
			{
				long long t = s[n*2-i-j] - '0';
				if(i) f[i][j] = f[i-1][j] + t*w[i];
				if(j) f[i][j] = max(f[i][j-1] + t*w[j],f[i][j]);
			}
		}
		for(int i = n,j = n; i > 0 || j > 0;)
		{
			long long t = s[n*2-i-j] - '0';
			if(i && f[i][j] - f[i-1][j] == t*w[i])
			{
				cout<<"H";
				i--;
			}
			else
			{
				cout<<"M";
				j--;
			}
		}
		cout<<endl;
	}

}
