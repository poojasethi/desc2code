#include <cstdio>
using namespace std;
#define N 20
#define mmax(n1,n2) (n1)>(n2)?(n1):(n2);
long long f[N][N],w[N];
char s[N<<1];
int main()
{
    int n;
    scanf("%d",&n);
    scanf("%s",s);
    w[1]=1;
    for(int i=2;i<=n;i++)
        w[i]=w[i-1]*10;
    for(int i=0;i<=n;i++)
        for(int j=0;j<=n;j++)
        {
            long long t = s[n*2-i-j]-'0';
            if(i)
                f[i][j]=f[i-1][j]+w[i]*t;
            if(j)
                f[i][j]=mmax(f[i][j-1]+w[j]*t,f[i][j]);
        }
    for(int i=n,j=n;i>0||j>0;)
    {
        long long t= s[n*2-i-j]-'0';
        if(i && f[i][j]-f[i-1][j] == t*w[i])
            i--,printf("H");
        else
            j--,printf("M");
    }
    printf("\n");
    return 0;
}
