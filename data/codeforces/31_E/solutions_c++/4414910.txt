#include <stdio.h>
#include <algorithm>
using namespace std;
int n;
char str[37];
long long dp[37][19][19],pow10[19]={1};
bool vis[37][19][19];
long long calc(int i,int na,int nb){
    if(i==2*n)
        return 0;
    if(!vis[i][na][nb]){
        vis[i][na][nb]=true;
        dp[i][na][nb]=-(1ll<<60);
        if(na)
            dp[i][na][nb]=max(dp[i][na][nb],pow10[na-1]*(str[i]-'0')+calc(i+1,na-1,nb));
        if(nb)
            dp[i][na][nb]=max(dp[i][na][nb],pow10[nb-1]*(str[i]-'0')+calc(i+1,na,nb-1));
    }
    return dp[i][na][nb];
}
void print(int i,int na,int nb){
    if(i==2*n)
        return;
    if(na && dp[i][na][nb]==pow10[na-1]*(str[i]-'0')+calc(i+1,na-1,nb)){
        printf("H");
        print(i+1,na-1,nb);
        return;
    }
    printf("M");
    print(i+1,na,nb-1);
}
int main()
{
    scanf("%d%s",&n,str);
    for(int i=1;i<19;++i)
        pow10[i]=pow10[i-1]*10;
    calc(0,n,n);
    print(0,n,n);
    return 0;
}
