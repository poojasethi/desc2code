#include <cstdio>
#include <cstring>

#define ll long long int

using namespace std;

ll dezna[20];
char s[40];
int n;
int pai[20][20][40];
ll PD[20][20][40];

ll calc(int h, int m, int u) {
    if (u == 2*n) return 0;
    ll &pd = PD[h][m][u];
    if (pd != -1) return pd;
    ll opc1 = -32;
    if (h) opc1 = (s[u]-'0')*dezna[h-1] + calc(h-1,m,u+1);
    ll opc2 = -32;
    if (m) opc2 = (s[u]-'0')*dezna[m-1] + calc(h,m-1,u+1);
    if (opc1 > opc2) {
        pai[h][m][u] = 1;
        return pd = opc1;
    }
    pai[h][m][u] = 2;
    return pd = opc2;
}

int main() {

    memset(PD,0xff,sizeof(PD));
    dezna[0] = 1ll;
    for (int i=1;i<20;i++)
        dezna[i] = dezna[i-1]*10ll;

    scanf("%d",&n);
    scanf("%s",s);
    ll tt = calc(n,n,0);

    int h=n, m=n, u=0;
    while (u < 2*n) {
        int p = pai[h][m][u];
        printf("%c",p==1?'H':'M');
        if (p==1)
            h--;
        else
            m--;
        u++;
    }
    printf("\n");

    return 0;
}
