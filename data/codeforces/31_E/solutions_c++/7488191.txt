#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;
typedef long long ll;

ll dp[19][19];
char ans[40],str[40];
ll w[20];

int main()
{
    int n;
    w[1]=1;
    for(int i=2;i<=18;i++)
        w[i]=w[i-1]*10;

    scanf("%d",&n);
    scanf("%s",str);

    memset(dp,0,sizeof dp);

    for(int i=0;i<=n;i++)
    {
        for(int j=0;j<=n;j++)
        {
            long long z=str[n+n-i-j]-'0';
            if(i)dp[i][j]=dp[i-1][j]+z*w[i];
            if(j)dp[i][j]=max(dp[i][j],dp[i][j-1]+z*w[j]);
        }
    }
    /*for(int i=0;i<=n;i++)
    {
        for(int j=0;j<=n;j++)
        {
            printf("dp[%d][%d] = %I64d\n",i,j,dp[i][j]);
        }
    }*/
    //printf("--%I64d\n",dp[n][n]);
    for(int i=n,j=n;i>0 || j>0 ;)
    {
        long long z=str[n+n-i-j]-'0';
        if(i && dp[i-1][j]+z*w[i]==dp[i][j]){cout<<'H';i--;}
        else {
            cout<<'M';j--;
        }
    }

    return 0;
}
