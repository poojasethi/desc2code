#include <iostream>
using namespace std;

long long t[19], a[19][19], h, m;
char b[19][19];
int main()
{
	int n, i, j;
	string s;
	cin>>n>>s;
	for (t[n]=1,i=n-1; i>0; i--) t[i]=t[i+1]*10;
	for (i=1; i<=n; i++) b[0][i]='H', b[i][0]='M', a[0][i]=a[i][0]=a[0][i-1]+(s[i-1]-'0')*t[i];
	for (i=1; i<=n; i++) for (j=1; j<=n; j++) {
		h = a[i][j-1]+(s[i+j-1]-'0')*t[j];
		m = a[i-1][j]+(s[i+j-1]-'0')*t[i];
		if (h>=m) b[i][j]='H', a[i][j]=h;
		else b[i][j]='M', a[i][j]=m;
	}
	for (i=j=n; i+j>0;) {
		s[i+j-1]=b[i][j];
		if (b[i][j]=='H') j--;
		else i--;
	}
	cout<<s<<endl;
	return 0;
}
