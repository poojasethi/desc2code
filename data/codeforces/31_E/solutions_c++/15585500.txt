#include <bits/stdc++.h>
using namespace std;
#define MAXN 20
#define FI "digits.i1"
#define FO "digits.o1"

string s, ans = "";
long long p10[MAXN], dp[MAXN][MAXN];
int n;
int main(){
    ios_base::sync_with_stdio(0), cin.tie(0);
    #ifndef ONLINE_JUDGE
    ifstream cin(FI);
    ofstream cout(FO);
    #endif // ONLINE_JUDGE
    cin >> n >> s;
    //s += '0';
    p10[1] = 1;
    int x;
    for (int i = 2; i <= n; i++) p10[i] = p10[i-1]*10;
    for (int i = 0; i <= n; i++)
        for (int j = 0; j <= n; j++){
            x = s[n * 2 - i - j] - 48;
            if (i) dp[i][j] = dp[i-1][j] + x * p10[i];
            if (j) dp[i][j] = max(dp[i][j], dp[i][j - 1] + x * p10[j]);
        }
    for (int i = n, j = n; i || j;){
        x = s[n * 2 - i - j] - 48;
        if (i && dp[i-1][j] + x * p10[i] == dp[i][j]) --i, ans += 'H';
        else --j, ans += 'M';
    }
    cout << ans <<'\n';
    return 0;
}
