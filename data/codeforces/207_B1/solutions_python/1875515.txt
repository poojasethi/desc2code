#include<stdio.h>
#include<string.h>
#include<math.h>
#include<algorithm>
#include<stdlib.h>
#include<set>

#include<iostream>
#define ll long long
#define clr(x,y) memset(x,y,sizeof(x))
using namespace std;
const int inf=1000000000;

int i,j,n,m;
int a[1000],dp[1000];

int main(){
//    freopen("in.txt","r",stdin);
    scanf("%d",&n);
    for(i=1;i<=n;i++)
        scanf("%d",&a[i]);
    int ans=0;
    for(int l=1;l<=n;l++){
        for(i=2;i<=n;i++) dp[i]=inf;
        for(i=1;i<=n;i++){
            for(j=i+1;j<=n;j++){
                if(i>=j-a[j])
                    dp[j]=min(dp[j],dp[i]+1);
            }
        }
        ans+=dp[n];
        int tmp=a[n];
        for(i=n;i>=2;i--)
            a[i]=a[i-1];
        a[1]=tmp;
    }
    printf("%d\n",ans);
    return 0;
}
