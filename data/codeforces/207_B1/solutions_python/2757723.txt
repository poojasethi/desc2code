#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long ll;
const int kMaxN = 250000;
int n, s[kMaxN * 2 + 1];

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++ i) {
		scanf("%d", &s[i]); 
		s[i] = min(s[i], n - 1);
		s[i + n] = s[i];
	}
	ll ans = 0;
	for (int i = 1; i <= n; ++ i) {
		int pos = i + n, req = i + 1;
		while (true) {
			int lb = pos - s[pos], rb = pos;
			if (lb <= req) {
				++ ans;
				break;
			}
			int rec = pos;
			for (int j = lb; j < rb; ++ j)
				if (j - s[j] < rec - s[rec]) rec = j;
			++ ans;
			pos = rec;
		}
	}
	printf("%I64d\n", ans);
	return 0;
}
