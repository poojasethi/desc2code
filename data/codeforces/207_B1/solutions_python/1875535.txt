#define randomize srand((unsigned)time(NULL))
#define random(x) ((rand() * (rand() + rand())) % (x))
#define ms(x, y) memset(x, y, sizeof(x))
#define mc(x, y) memcpy(x, y, sizeof(x))

const int maxn = 300 + 10;

int n;
int a[maxn], f[maxn];

int main() {
	
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%d", &a[i]);
	}
	
	int ans = 0;
	for (int z = 0; z < n; ++z) {
		ms(f, 0x7F);
		f[n] = 0;
		for (int i = n; i > 1; --i)
			for (int j = i - 1; j >= i - a[i] && j >= 1; --j)
				if (f[i] + 1 < f[j])
					f[j] = f[i] + 1;
		//for (int i = 1; i <= n; ++i)
		//	printf("%d ", a[i]);
		//printf(" == %d\n", f[1]);
		ans += f[1];
		int tmp = a[1];
		memmove(a + 1, a + 2, sizeof(int) * (n - 1));
		a[n] = tmp;
	}
	
	printf("%d\n", ans);
	
	return 0;
}
