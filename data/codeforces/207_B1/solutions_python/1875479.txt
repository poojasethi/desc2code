#include <stdio.h>
#include <memory.h>

int const maxtanks = 300;

int dp[maxtanks][maxtanks*2];
int tankrad[maxtanks*2];
int tanknum;
int main(){
	//freopen("in.txt", "r", stdin);
	int i, j, k, tmp, x, y, result = 0;
	scanf("%d", &tanknum);
	for(i=0;i<tanknum;++i){
		scanf("%d", &tankrad[i]);
		tankrad[i+tanknum] = tankrad[i];
	}
	memset(dp, -1, sizeof(dp));
	for(i=tanknum, x=0;i<tanknum*2;++i, ++x){
		memset(dp, -1, sizeof(dp));
		dp[x][i] = 0;
		y = i-tanknum;
		for (j=i;j>y;--j){
			for(k=1;k<=tankrad[j];++k){
				if(j-k>y){
					tmp = dp[x][j] + 1;
					if(tmp < dp[x][j-k] || dp[x][j-k] == -1)
						dp[x][j-k] = tmp;
				}else
					break;
			}
		}
		result += dp[x][x+1];
	}
	printf("%d\n", result);
	return 0;
}
