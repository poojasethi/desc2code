#include <cstdio>
#include <iostream>
#include <cstring>
#include <cmath>

using namespace std;

const int N=612;
int n,a[N],ans=0;

int main()
  {
    scanf("%d",&n);
    for (int i=1;i<=n;++i) scanf("%d",&a[i]),a[i+n]=a[i];
    for (int i=n+1;i<=n*2;++i)
      {
        int t=i;
        while (t>i-n+1)
          {
            for (int j=t;j<=i;++j) t=min(t,j-a[j]);
            ++ans;
          }
      }  
    printf("%d\n",ans);
  }
