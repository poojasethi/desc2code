#include <cstdio>
#include <cmath>
#include <iostream>
#include <algorithm>
#include <string>
#include <memory.h>
#include <memory>

using namespace std;

int i , j , k , d , dd , n ;
int mas[ 20001 ] , ans[ 20001 ];
long long int result;
int main()
{
  cin >> n;
  for( i = 0 ; i < n ; i++ )
  {
    cin >> mas[ i ];
    mas[ n + i ] = mas[ i ];
  }
  for( i = n - 1 ; i < 2 * n - 1 ; i++ )
  {
        d = i - 1 ;
        ans[ i ] = 0;
        for( j = i ; j > i - n + 1 ; j-- )
        {
            dd = max( i - n + 1 , j - mas[ j ] );
            for( ; d >= dd ; d-- )
            ans[ d ] = ans[ j ] + 1;
        }
        result += ans[ i - n + 1 ];
  }
  cout << result << endl;
  return 0;
}
