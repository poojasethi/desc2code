#include <iostream>
#include<stdio.h>
#include<algorithm>

using namespace std;

typedef struct xxx
{
    int id;
    int ax;
};
xxx a[10000];
int n = 0,t = 0,s = 0;
void zhao(void)
{
    int sum = 0,i = 1,j = 1;
    int ad = 0;
    while(1)
    {
        for(j = i+1;j<=n;j++)
        {
            if(a[j].ax>=(j - i))
            {
                ad = j;
            }
        }
        i = ad;
        t++;
        if(i >= n) break;
    }
    return ;
}
void huan(void)
{
    int i = 0,t1 = 0,t2 = 0;
    t1 = a[n].ax;
    t2 = a[n].id;
    for(i = n;i>=2;i--)
    {
        a[i].ax = a[i - 1].ax;
        a[i].id = a[i - 1].id;

    }
    a[1].ax = t1;
    a[1].id = t2;
    return ;
}
int iff(void)
{
    int i = 0;
    for(i = 1;i<=n;i++)
    {
        if(a[i].id != i) break;
    }
    if(i >= n+1) return 1;
    return 0;
}
int main()
{
    int i = 0;
    cin>>n;
    for(i = 1;i<=n;i++)
    {
        a[i].id = i;
        scanf("%d",&a[i].ax);
    }
    while(1)
    {
        zhao();
        huan();
        if(iff())
        {
            cout<<t<<endl;
            break;
        }
    }
    return 0;
}
