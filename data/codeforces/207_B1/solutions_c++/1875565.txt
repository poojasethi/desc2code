#include <cstdio>
#include <iostream>
#include <algorithm>
#define FOR(i,s,e) for(int i=(s);i<(int)(e);i++)
#define FOE(i,s,e) for(int i=(s);i<=(int)(e);i++)
using namespace std;
typedef long long LL;

const int N = 1048576, INF = N;
int n;

int a[N];
int d[N];

int main() {
        cin >> n;
        FOR(i, 0, n) cin >> a[i];
        copy(a, a+n, a+n);

        LL ans = 0LL;
        FOR(i, 0, n) {
                int e = i+n-1;
                d[e] = 0;
                for (int j = e-1; j >= i; j--) {
                        d[j] = INF;
                        for (int k = j+1; k <= e; k++) {
                                if (j+a[k] >= k)
                                        d[j] = min(d[j], d[k]+1);
                        }
                }
                ans += d[i];
        }
        cout << ans << endl;
        return 0;
}
