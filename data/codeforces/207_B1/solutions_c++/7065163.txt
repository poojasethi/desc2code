#include<iostream>
#include<cstdio>
#include<cmath>
#include<vector>
#define inf 1000000000
#define mod 1000000007
using namespace std;
int a[250003];
int n;
int main()
{
    scanf("%d",&n);
    int i,j;
    for(i=0;i<n;i++)
        scanf("%d",&a[n-i-1]);
    int oldr,newr;
    long long ans=0;
    int tem;
    for(i=0;i<n;i++)
    {
        newr=oldr=0;
        while(newr<n-1)
        {
            tem=newr+1;
            for(j=newr;j>=oldr;j--)
                newr=max(newr,j+a[j]);
            oldr=tem;
            ans++;
        }
        tem=a[n-1];
        for(j=n-1;j>0;j--)
            a[j]=a[j-1];
        a[0]=tem;
    }
    printf("%I64d\n",ans);
    return 0;
}
