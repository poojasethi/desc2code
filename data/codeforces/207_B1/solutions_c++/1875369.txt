void swap(int* a, int n){
    int i, tp, tp2;
    int* b=new int[n];
    b[0]=a[n-1];
    for(i=1; i<n; ++i)
        b[i]=a[i-1];
    for(i=0; i<n; ++i)
        a[i]=b[i];
    return;
}
