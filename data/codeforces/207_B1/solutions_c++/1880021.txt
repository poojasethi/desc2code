#include<cstdio>
#include<cstring>
#include<deque>
#include<queue>

#define MAXX 301

short n,m,i;
int a,ans;
std::deque<int>rad;
bool done[MAXX];

struct node
{
    short id;
    int count;
    void print()
    {
        printf("id:%hd %d\n",id,count);
    }
}t,t1;

std::queue<node>q;

int main()
{
    scanf("%hd",&n);
    while(n--)
    {
        scanf("%d",&a);
        rad.push_back(a);
    }
    n=rad.size();
    while(n--)
    {
        memset(done,(bool)false,sizeof(done));
        while(!q.empty())
            q.pop();
        t.id=0;
        t.count=0;
        q.push(t);
        while(!q.empty())
        {
            t=q.front();
            q.pop();
            if(t.id>=rad.size()-1)
                break;
            ++t.count;
            for(i=t.id+1;i<rad.size();++i)
                if(!done[i] && rad[i]>=i-t.id)
                {
                    done[i]=true;
                    t1=t;
                    t1.id=i;
                    q.push(t1);
                }
        }
        ans+=t.count;
        rad.push_front(rad.back());
        rad.pop_back();
    }
    printf("%d\n",ans);
    return 0;
}
