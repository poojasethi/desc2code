#include <stdio.h>
#define MAX_N 500001

int n;
int r[MAX_N];

long long sol[MAX_N];

long long ret = 0LL;

int main()
{
    scanf("%d",&n);
    for (int i=0;i<n;i++)
    {
        scanf("%d",&r[i]);
    }
    sol[0] = 0LL;
    int next;
    for (int i=0;i<n;i++)
    {
        int curr = i;
        next = i;
        while (next != i+n-1)
        {
          for (int j=next+1;j<(i+n);j++)
          {
            if (r[j%n] >= j-curr) next=j;
          }
          ret++;
          curr = next;
          next = curr;
        }
    }
    printf("%I64d\n",ret);
    return 0;
}
