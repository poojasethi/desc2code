//字典序最小问题
#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
string s[50010];
bool cmp(const string& a,const string& b){
	return a + b < b + a;
}
int main(){
	int n,i,j;
	cin>>n;
	for(i=0;i<n;i++) cin>>s[i];
	sort(s,s+n,cmp);
	for(i=0;i<n;i++) cout<<s[i];
	cout<<endl;
	return 0;
	
}
