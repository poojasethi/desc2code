#include<iostream>
#include<cstring>
#include<string>
#include<algorithm>
using namespace std;
const int qq=50000+50;
string s[qq];
int cmp(string a,string b)
{
	return a+b<b+a;
}
int main()
{
	int n;cin >> n;
	for(int i=0;i<n;++i)
		cin >> s[i];
	sort(s,s+n,cmp);
	for(int i=0;i<n;++i)
		cout << s[i];
		cout << endl;
}
