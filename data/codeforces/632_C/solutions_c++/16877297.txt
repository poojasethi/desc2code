#include<bits/stdc++.h>

using namespace std;
string str[50007];

bool compare(string s1,string s2)
{
	if(s1+s2 < s2+s1)
	return 1;
	else return 0;
}
int main()
{
	int n,i;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>str[i];
	}
	sort(str,str+n,compare);
	for(i=0;i<n;i++)
	{
		cout<<str[i];
	}
	cout<<"\n";
	return 0;
}
