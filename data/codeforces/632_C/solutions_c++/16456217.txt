#include <bits/stdc++.h>
using namespace std;
bool cmpr(string &s, string &t){
	return t+s>s+t;
}

int main() {
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	int n;
	cin >> n;
	string s[n];
	for(int i=0 ; i<n ; i++)
		cin >> s[i];
	sort(s,s+n,cmpr);
	for(int i=0 ; i<n ; i++)
		cout << s[i] ;
	return 0;
}
