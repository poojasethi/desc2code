#include<bits/stdc++.h>
using namespace std;
int n;
bool cmp(string a,string b)
{
    if((a+b)<(b+a))
        return true;
    return false;
}
int main()
{
    scanf("%d",&n);
    string a[n+1];
    for(int i=1;i<=n;i++)
        cin>>a[i];
    sort(a+1,a+n+1,cmp);
    for(int i=1;i<=n;i++)
        cout<<a[i];
}
