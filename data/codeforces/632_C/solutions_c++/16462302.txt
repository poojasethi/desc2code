#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<iostream>
#include<cmath>
using namespace std;
int n,i;
string s[50001];
bool cmp(string a,string b)
{
	return a+b<b+a;
}

int main()
{
		scanf("%d\n",&n);
		for (i=1;i<=n;i++) cin>>s[i];
		sort(s+1,s+n+1,cmp);
		for (i=1;i<=n;i++) cout<<s[i];
}
