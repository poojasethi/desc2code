#include<bits/stdc++.h>
using namespace std;
#define N 50003
string s[N];
bool cmp(string a,string b)
{
    return a+b<b+a;
}
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++){
        cin>>s[i];
    }
    sort(s,s+n,cmp);
    for(int i=0;i<=n;i++){
        cout<<s[i];
    }
    printf("\n");
    return 0;
}
