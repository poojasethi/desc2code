#include<bits/stdc++.h>
using namespace std;
string ar[1000000];
bool comp(const string &a ,const string &b){
	return a+b < b+a;
}
int main(){
	int n;
	scanf("%d",&n);
	for(int i=1 ; i<=n ; i++){
		cin >> ar[i];
	}
	sort(ar+1,ar+n+1,comp);
	for(int i=1 ; i<=n ;i++){
		cout << ar[i];
	}
	puts("");
	return 0;
}
