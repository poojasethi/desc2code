#include <bits/stdc++.h>
using namespace std;

int n;
vector <string> s;
bool func(string a,string b) {
	return a+b<b+a;
}


int main() {

	scanf("%d",&n);
	s.resize(n+1);
	for (int i=1; i<=n; i++)
		cin>>s[i];
	
	sort (s.begin()+1,s.end(),func);
	
	for (int i=1; i<=n; i++)
		cout<<s[i];
	
	return 0;
}
