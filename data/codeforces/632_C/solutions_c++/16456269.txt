#include<iostream>
#include<algorithm>
using namespace std;


string strj[50005];
bool cmp(string a, string b)
{
    return (a+b)<(b+a);
}

int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>strj[i];
    sort(strj,strj+n,cmp);
    for(int i=0;i<n;i++)
        cout<<strj[i];
    return 0;
}
