#include <bits/stdc++.h>
using namespace std;

const int N = 5*1e6 + 100;
string str[N];

bool comp(string a,string b)
{
    return a+b < b+a;
}
int main()
{
    int n; cin>>n;
    for(int i=0;i<n;i++) cin>>str[i];
    sort(str,str+n,comp);
    for(int i=0;i<n;i++) cout<<str[i];
    cout<<endl;
    return 0;
}
