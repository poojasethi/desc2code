#include <bits/stdc++.h>

using namespace std;

bool compare ( string x , string y )
{
    return x+y < y+x;
}

string s[5*1000*10+2];

int main()
{
    int n;
    cin >> n;
    for (int i=1;i<=n;i++)
        cin >> s[i];
    sort( s+1 , s+1+n , compare );
    for (int i=1;i<=n;i++)
        cout << s[i];
}
