#include <bits/stdc++.h>
#define MAX 50000
using namespace std;
int n;
string v[MAX];

bool cmp(string a, string b)
{
    return a + b < b + a;
}

int main()
{
    scanf("%d", &n);
    for(int i = 0; i < n; i++) cin >> v[i];
    sort(v, v + n, cmp); 
    string ans = accumulate(v, v + n, string());
    puts(ans.c_str());
}
