#include <bits/stdc++.h>
using namespace std;

#define ll long long

bool cmp(string a, string b){
	return (a+b).compare(b+a)<0;
}

int main() {
	int n,i;
	scanf("%d",&n);
	vector<string> s(n);
	for(i=0;i<n;i++)
		cin>>s[i];
	sort(s.begin(),s.end(),cmp);
	for(i=0;i<n;i++)
		cout<<s[i];
	
	return 0;
}
