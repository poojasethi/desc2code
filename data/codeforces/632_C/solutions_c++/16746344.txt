#include <bits/stdc++.h>

using namespace std;

bool mysort(string a,string b){
	return a+b < b+a;
}

int main(){
	ios_base::sync_with_stdio(false);
	int n;
	cin>>n;
	vector<string> v(n);
	for(int i=0;i<n;i++){
		cin>>v[i];
	}
	sort(v.begin(),v.end(),mysort);
	for(int i=0;i<n;i++){
		cout<<v[i];
	}
	return 0;
}
