def cmp_concetanate(word1, word2):
    c1 = word1 + word2
    c2 = word2 + word1
    if c1 < c2:
        return -1
    elif c1 > c2:
        return 1
    else:
        return 0
n = int(raw_input())
lst = []
for i in range(n):
	s = raw_input()
	lst.insert(i,s)
lst.sort(cmp_concetanate)
my_lst = ''.join([str(i) for i in lst])
print my_lst
