#include<cstdio>
#include<cstring>
#define N 5010
char s1[N],s2[N];
int d1[N],d2[N],d[N][N];
int n,m,i,j,tmp,ans;
int max(int a,int b)
{
	if (a>b) return a;return b;
}
int min(int a,int b)
{
	if (a<b) return a;return b;
}
int main()
{
	scanf("%s",&s1);
	scanf("%s",&s2);
	n=strlen(s1);
	m=strlen(s2);
	

	memset(d,0,sizeof(d));
	for (i=n-1;i>=0;i--)
	for (j=n-1;j>=0;j--)
	if (s1[i]==s1[j])
	d[i][j]=d[i+1][j+1]+1;

	
	for (i=0;i<=n-1;i++)
	for (j=0;j<=n-1;j++)
	if (i!=j)
	d1[i]=max(d1[i],d[i][j]);
		memset(d,0,sizeof(d));

	for (i=m-1;i>=0;i--)
	for (j=m-1;j>=0;j--)
	if (s2[i]==s2[j])
	d[i][j]=d[i+1][j+1]+1;

	for (i=0;i<=m-1;i++)
	for (j=0;j<=m-1;j++)
	if (i!=j)
	d2[i]=max(d2[i],d[i][j]);
		memset(d,0,sizeof(d));
	
	for (i=n-1;i>=0;i--)
	for (j=m-1;j>=0;j--)
	if (s1[i]==s2[j])
	d[i][j]=d[i+1][j+1]+1;

	
	ans=0x37373737;
	for (i=0;i<=n-1;i++)
	for (j=0;j<=m-1;j++)
	{
	tmp=max(d1[i],d2[j]);
	if (tmp>=d[i][j]) continue;
	ans=min(ans,tmp+1);
	}
	
	if (ans!=0x37373737)
	printf("%d",ans);
	else
	printf("-1");
}
