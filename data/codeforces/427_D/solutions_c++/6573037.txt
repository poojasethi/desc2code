#include <iostream>
#include <cstring>
#include <cstdlib>
#include <ctime>
#include <cstdio>
#define f cin
#define g cout
#define NM 5010

using namespace std;

const int INF=0x3f3f3f3f;
int PI[NM];
int cnt1[NM],cnt2[NM];
char A[NM],B[NM];
int ans=INF;
int N,M;

int main()
{
	scanf("%s%s",A+1,B+1);
	N=strlen(A+1);
	M=strlen(B+1);
    for(int p=1;p<=N;p++){
		memset(PI,0,sizeof(PI));
		memset(cnt1,0,sizeof(cnt1));
		memset(cnt2,0,sizeof(cnt2));
		PI[1]=0;
		int q=0;
		for(int i=p+1;i<=N;i++){
			while(q!=0 && A[q+p]!=A[i]) q=PI[q];
			if(A[q+p]==A[i]) q++;
			PI[i-p+1]=q;
		}
		q=0;
		for(int i=1;i<=N;i++){
			while(q!=0 && A[q+p]!=A[i]) q=PI[q];
			if(A[q+p]==A[i]) q++;
			if(i==p && q==1) q=0;
			cnt1[i]=q;
		}
		q=0;
		for(int i=1;i<=M;i++){
			if(q!=0 && A[q+p]!=B[i]) cnt2[q]++;
			while(q!=0 && A[q+p]!=B[i]) q=PI[q];
			if(A[q+p]==B[i]) q++;
		}
		cnt2[q]++;
		for(int i=N;i>=1;i--) cnt2[i]+=cnt2[i+1];
		int maxi=0;
		for(int i=1;i<=N;i++) maxi=max(maxi,cnt1[i]);
		maxi++;
		for(int i=maxi;i<=N;i++) if(cnt2[i]==1) ans=min(ans,i);
    }
	if(ans==INF) ans=-1;
	printf("%d\n",ans);
	return 0;
}
