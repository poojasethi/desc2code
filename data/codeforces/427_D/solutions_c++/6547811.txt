#include<cstdio>
#include<cstring>
#include<algorithm>
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define dep(i,a,b) for (int i=a;i>=b;--i)
#define INF 0x7fffffff
#define N 5005
using namespace std;
int f11[N][N],f22[N][N],f12[N][N],f2[N],f1[N],n,m,res=INF;
char a[N],b[N];bool ok[N];
int main() {
    scanf("%s%s",a,b);m=strlen(a);n=strlen(b);
    dep(i,m-1,0) dep(j,n-1,0) f12[i][j]=a[i]==b[j]?f12[i+1][j+1]+1:0;
    dep(i,m-1,0) dep(j,m-1,0) f11[i][j]=a[i]==a[j]?f11[i+1][j+1]+1:0;
    dep(i,n-1,0) dep(j,n-1,0) f22[i][j]=b[i]==b[j]?f22[i+1][j+1]+1:0;
    rep(i,0,m-1) {memset(ok,0,sizeof ok);rep(j,0,m-1) if (i!=j) ok[f11[i][j]]=1;f1[i]=INF;for (int j=m-i;j&&!ok[j];--j) f1[i]=j;}
    rep(i,0,n-1) {memset(ok,0,sizeof ok);rep(j,0,n-1) if (i!=j) ok[f22[i][j]]=1;f2[i]=INF;for (int j=n-i;j&&!ok[j];--j) f2[i]=j;}
    rep(i,0,m-1) rep(j,0,n-1) if (f12[i][j]>=f1[i]&&f12[i][j]>=f2[j]) res=min(res,max(f1[i],f2[j]));
    printf("%d\n",res==INF?-1:res);
    return 0;
}
