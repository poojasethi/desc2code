#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 5010
#define M 12500010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,an,t,v,w;
int head[M],a[M],next[M],b[M],c[M],len[M];
char p[N],s[N];
int main() {
	scanf("%s%s",p+1,s+1);
	n=strlen(p+1);
	m=strlen(s+1);
	For(i,1,n) {
		w=0;
		For(j,i,n) {
			int A=p[j]-'a';
			for (v=head[w];v;v=next[v]) if (b[v]==A) break;
			if (!v) b[++t]=A,next[t]=head[w],head[w]=v=t,len[t]=len[w]+1;
			w=v;
			a[v]++;
		}
	}
	For(i,1,m) {
		w=0;
		For(j,i,m) {
			int A=s[j]-'a';
			for (v=head[w];v;v=next[v]) if (b[v]==A) break;
			if (!v) break;
			w=v;
			c[v]++;
		}
	}
	an=n+1;
	For(i,1,t) if (len[i]<an&&a[i]==1&&c[i]==1) an=len[i];
	printf("%d\n",an>n?-1:an);
	return 0;
}
