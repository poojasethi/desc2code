#include<cstdio>
using namespace std;

int n,m,c;
char ch1[5555],ch2[5555];
int dp[5555][5555];

int main(){
    cin>>ch1>>ch2;
    n=strlen(ch1);m=strlen(ch2);
    for(int i=n-1;i+1;i--){
    for(int j=m-1;j+1;j--)
    {
        if(ch1[i]==ch2[j])dp[i][j]=1+dp[i+1][j+1];
        else dp[i][j]=0;
    }}
    int ans=10000;int f=10000;
    for(int i=0;i<n;i++){
        int a=0,b=0,k;
        for(int j=0;j<m;j++)
        if(dp[i][j]>=a){k=j;b=a;a=dp[i][j];}
        else if(dp[i][j]>=b)b=dp[i][j];
        if(a==b)continue;
        int c=b;
        for(int z=0;z<n;z++)
        if(z!=i)c=max(c,dp[z][k]);
        if(c<a){
            if(ans>c+1){ans=c+1;}
        }
    }
    if(ans!=f)
    cout<<ans<<endl;
    else cout<<-1<<endl;
}
