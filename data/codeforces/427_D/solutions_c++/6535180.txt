#include <cstdio>
#include <iostream>

using namespace std;

char A[5005], B[5005];
int M[5005], M2[5005], P[5005][5005], PM[5005][5005], PM2[5005][5005], tam, tam2, mejor=-1;

int main()
{
    scanf("%s%s",A+1,B+1);
    for(tam=1; A[tam]!='\0'; tam++);
    --tam;
    for(tam2=1; B[tam2]!='\0'; tam2++);
    --tam2;
    for(int i=tam; i>0; i--){
        for(int e=tam; e>i; e--){
            if(A[i]==A[e])
                PM[i][e]=PM[i+1][e+1]+1;
            if(PM[i][e]>M[i])
                M[i]=PM[i][e];
            if(PM[i][e]>M[e])
                M[e]=PM[i][e];
        }
    }
    for(int i=tam2; i>0; i--){
        for(int e=tam2; e>i; e--){
            if(B[i]==B[e])
                PM2[i][e]=PM2[i+1][e+1]+1;
            if(PM2[i][e]>M2[i])
                M2[i]=PM2[i][e];
            if(PM2[i][e]>M2[e])
                M2[e]=PM2[i][e];
        }
    }
    for(int i=tam; i>0; i--){
        for(int e=tam2; e>0; e--){
            if(A[i]==B[e])
                P[i][e]=P[i+1][e+1]+1;
            if(P[i][e]>max(M[i], M2[e])){
                if(mejor==-1 or max(M[i], M2[e])+1<mejor)
                    mejor=max(M[i], M2[e])+1;
            }
        }
    }
    printf("%d\n",mejor);
    return 0;
}
