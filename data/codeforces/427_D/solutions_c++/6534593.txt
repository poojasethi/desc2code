
#include<bits/stdc++.h>
#define all(x) x.begin(), x.end()
#define pb(x) push_back(x)
#define MAXN 24000000


using namespace std;

int next[MAXN], top[MAXN], depth[MAXN], C[2][MAXN], nodes = 0;
short int car[MAXN];

char s[5005];

void add(int pos, int id){
    
    int cur = 0;
    for(int i = pos, j; s[i]; i++){
        
        for( j = top[cur]; j != 0; j = next[j] )
            if(car[j] == s[i] - 'a' + 1)break;
        
        if(!j && id == 0){
            
            if(nodes >= MAXN)while(true){}
            j = ++nodes;
            car[j] = s[i] - 'a' + 1;
            
            next[j] = top[cur];
            
            top[cur] = j;
            depth[j] = depth[cur] + 1;
            
        }
        else if( !j && id == 1)break;
        
        cur = j;
        C[id][cur]++;
        
    }
        
}

int main(){

    scanf("%s", s);
    for(int i = 0; s[i]; i++)add(i, 0);
        
    scanf("%s", s);
    for(int i = 0; s[i]; i++)add(i, 1);
        
    int mini = (1<<30);
    for(int i = 1; i <= nodes; i++)if(C[0][i] == 1 && C[1][i] == 1)mini = min(mini, depth[i]);
    
    if(mini == (1<<30))puts("-1");
    else cout<<mini<<endl;
        
}
