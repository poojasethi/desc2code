#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define MAXN 5050
char a[MAXN],b[MAXN];
int dpa[MAXN][MAXN],dpb[MAXN][MAXN],dpab[MAXN][MAXN];
int numa[MAXN],numb[MAXN];
void solve(char *a,char *b,int dp[][MAXN]){
	int n=strlen(a);
	int m=strlen(b);
	for(int i=1;i<=n;++i)
		for(int j=1;j<=m;++j){
			if(a[i-1]==b[j-1])
				dp[i][j]=dp[i-1][j-1]+1;
			else
				dp[i][j]=0;
		}
}
void getnum(int n,int dp[][MAXN],int *num){
	for(int i=1;i<=n;++i){
		num[i]=0;
		for(int j=1;j<=n;++j){
			if(i==j)
				continue;
			num[i]=max(num[i],dp[i][j]);
		}
		num[i]++;
	}
}
int main(){
	while(~scanf("%s%s",a,b)){
		int alen=strlen(a);
		int blen=strlen(b);
		solve(a,a,dpa);
		solve(b,b,dpb);
		solve(a,b,dpab);
		getnum(alen,dpa,numa);
		getnum(blen,dpb,numb);
		int ans=alen+1;
		for(int i=1;i<=alen;++i)
			for(int j=1;j<=blen;++j)
				if(dpab[i][j]>=numa[i] && dpab[i][j]>=numb[j]){
					ans=min(ans,max(numa[i],numb[j]));
				}
		if(ans>alen)
			ans=-1;
		cout<<ans<<endl;
	}
}
