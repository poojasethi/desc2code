#include<bits/stdc++.h>

#define FP( ii,aa,bb ) for( int ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb ) for( int ii=aa;ii>=bb;ii-- )

#define MAXN	5005

using namespace std;

int slen,plen,res=MAXN;
char s[MAXN],p[MAXN];

int dp1[MAXN][MAXN],dp2[MAXN][MAXN],dp12[MAXN][MAXN],pre1[MAXN],pre2[MAXN];

int main(){
	
	scanf("%s %s",s+1,p+1);
	slen = strlen( s+1 );
	plen = strlen( p+1 );
	
	FM( i,slen,1 )	FM( j,slen,1 )	dp1 [i][j] = ( s[i]==s[j] ? dp1 [i+1][j+1]+1 : 0 );
	FM( i,plen,1 )	FM( j,plen,1 )	dp2 [i][j] = ( p[i]==p[j] ? dp2 [i+1][j+1]+1 : 0 );
	FM( i,slen,1 )	FM( j,plen,1 )	dp12[i][j] = ( s[i]==p[j] ? dp12[i+1][j+1]+1 : 0 );
	
	FP( i,1,slen )	FP( j,1,slen )	if( i!=j )	pre1[i] = max( pre1[i],dp1[i][j] );
	FP( i,1,plen )	FP( j,1,plen )	if( i!=j )	pre2[i] = max( pre2[i],dp2[i][j] );
	
	FP( i,1,slen )	FP( j,1,plen )	if( pre1[i]<dp12[i][j] and pre2[j]<dp12[i][j] )	res = min( res,max( pre1[i],pre2[j] )+1 );

	printf("%d\n",(res==MAXN?-1:res));

}
