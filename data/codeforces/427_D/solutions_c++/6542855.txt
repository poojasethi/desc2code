#define N 5020
#include <stdio.h>
#include <string.h>
char s1[N],s2[N];
int dp11[N][N],dp22[N][N],dp12[N][N],Mx1[N],Mx2[N];
int Max(int x,int y)
{
	return x>y?x:y;
}
void dp(int f[N][N],char str1[N],char str2[N],int l1 , int l2)
{
	for (int i=l1-1;i>=0;i--)
		for (int j=l2-1;j>=0;j--)
			if (str1[i]==str2[j]) f[i][j] = f[i+1][j+1] + 1;
}
void work(int f[N],int g[N][N],int len)
{
	for (int i=0;i<len;i++)
	{
		for (int j=0;j<len;j++)
			if (j!=i&&g[i][j]>f[i])
				f[i] = g[i][j];
	}
}
int main()
{
	scanf("%s%s",&s1,&s2);
	int l1 = strlen(s1) , l2 = strlen(s2);
	dp(dp11,s1,s1,l1,l1);
	dp(dp22,s2,s2,l2,l2);
	dp(dp12,s1,s2,l1,l2);
	work(Mx1,dp11,l1);
	work(Mx2,dp22,l2);
	int tmp = 0;
	int ans = -1;
	for (int i=0;i<l1;i++)
		for (int j=0;j<l2;j++)
		{
			tmp = Max(Mx1[i],Mx2[j]) + 1;
			if (dp12[i][j]>=tmp&&(tmp<ans||ans==-1)) 
				ans = tmp;
		}
	printf("%d\n",ans);
	return 0;
}
