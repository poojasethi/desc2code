#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int ans,len[2],f[2][2][5001][5001],Max[2][2][5001];
char s[2][5001];

int main()
{
	for (int i=0;i<=1;i++) scanf("%s",s[i]+1),len[i]=strlen(s[i]+1);
	for (int ii=0;ii<=1;ii++)
		for (int jj=ii;jj<=1;jj++)
		{
			for (int i=1;i<=len[ii];i++)
				for (int j=1;j<=len[jj];j++)
					if (s[ii][i]==s[jj][j]) f[ii][jj][i][j]=f[ii][jj][i-1][j-1]+1;
			for (int i=1;i<=len[ii];i++)
				for (int j=1;j<=len[jj];j++) 
					if (i!=j) Max[ii][jj][i]=max(Max[ii][jj][i],f[ii][jj][i][j]);
		}	
	for (int i=1;i<=len[0];i++)
		for (int j=1;j<=len[1];j++)
			if (f[0][1][i][j]>Max[0][0][i]&&f[0][1][i][j]>Max[1][1][j])
			{
				int tmp=max(Max[0][0][i]+1,Max[1][1][j]+1);
				if (tmp<ans||!ans) ans=tmp;
			}
	if (!ans) ans--;
	printf("%d\n",ans);
}
