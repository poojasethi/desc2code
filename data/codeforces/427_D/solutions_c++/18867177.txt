#include <bits/stdc++.h>

using namespace std;

const int N = 5011;
int next[3][N][N], maxi[2][N];

void process(string a, string b, int ind) {
	for (int i = a.size() - 1; i > -1; i--)
		for (int j = b.size() - 1; j > -1; j--)
			if (a[i] == b[j])
				next[ind][i][j] = next[ind][i + 1][j + 1] + 1;
}

void compare(string a, int ind) {
	for (int i = 0; i < a.size(); i++)
		for (int j = 0; j < a.size(); j++)
			if (i - j)
				maxi[ind][i] = max(maxi[ind][i], next[ind][i][j]);
}

main() {
	string s, t;
	cin >> s >> t;
	process(s, s, 0); compare(s, 0);
	process(t, t, 1); compare(t, 1);
	process(s, t, 2);
	int ans = N * N;
	for (int i = 0; i < s.size(); i++)
		for (int j = 0; j < t.size(); j++)
			if (next[2][i][j] > max(maxi[0][i], maxi[1][j]))
				ans = min(ans, max(maxi[0][i], maxi[1][j]) + 1);
	if (ans == N * N)
		return cout << -1 << endl, 0;
	cout << ans << endl;
}
