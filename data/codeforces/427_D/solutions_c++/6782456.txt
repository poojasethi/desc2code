#include <bits/stdc++.h>
using namespace std;

const int N=5e3+5;

int f[N][N], g[N][N], h[N][N], x[N], y[N];

void dp(const string &a, const string &b, int F[][N]){
	int m=a.size(), n=b.size();
	for(int i=m-1; i>=0; i--)
		for(int j=n-1; j>=0;j--) if(a[i]==b[j]) F[i][j]=F[i+1][j+1]+1;
}

void maximize(int &a, int b){
	if(a<b) a=b;
}

void minimize(int &a, int b){
	if(a>b) a=b;
}

int main(){
	string a, b;
	//freopen("input.inp", "r", stdin);
	cin >> a >> b;
	dp(a, b, f);
	dp(a, a, g);
	dp(b, b, h);
	int m=a.size(), n=b.size();
	for(int i=0; i<m; i++)
		for(int j=0; j<m; j++) if(j!=i) maximize(x[i], g[i][j]);
	for(int i=0; i<n; i++)
		for(int j=0; j<n; j++) if(j!=i) maximize(y[i], h[i][j]);
	int ans=N;
	for(int i=0; i<m; i++)
		for(int j=0; j<n; j++) if(f[i][j]>=max(x[i], y[j])+1) minimize(ans,max(x[i], y[j])+1);
	if(ans==N) cout << -1;
	else cout << ans;
}
