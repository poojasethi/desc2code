#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
const int MAXN=5005;
int dp1[MAXN][MAXN],dp2[MAXN][MAXN],dp3[MAXN][MAXN],l1[MAXN],l2[MAXN];

int main(){
    string a,b;
    cin>>a>>b;
    int n=a.size(), m=b.size();
    for(int i=n-1;i>=0;i--)
        for(int j=m-1;j>=0;j--)
            if(a[i] == b[j]) dp1[i][j]=dp1[i+1][j+1]+1;
            
    for(int i=n-1;i>=0;i--)
        for(int j=n-1;j>=0;j--){
            if(i==j) continue;
            if(a[i]==a[j]) dp2[i][j]=dp2[i+1][j+1]+1;
        }

    for(int i=m-1;i>=0;i--)
        for(int j=m-1;j>=0;j--){
            if(i==j) continue;
            if(b[i]==b[j])
                dp3[i][j]=dp3[i+1][j+1]+1;
        }

    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++){
            if(i==j) continue;
            l1[i]=max(l1[i],dp2[i][j]);
        }

    for(int i=0;i<m;i++)
        for(int j=0;j<m;j++){
            if(i==j) continue;
            l2[i]=max(l2[i],dp3[i][j]);
        }
    int ans = 2147483647;

    for(int i=0;i<n;i++)
        for(int j=0;j<m;j++){
            if(dp1[i][j]>max(l1[i],l2[j]))
                ans=min(ans,max(l1[i],l2[j])+1);
        }

    printf("%d",ans==2147483647?-1:ans);
}
