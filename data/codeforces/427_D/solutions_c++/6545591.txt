#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N=5010;
char a[N],b[N];
int csa[N][N],csb[N][N],csab[N][N],da[N],db[N];
int lena,lenb;

void lca(int f[N][N],char a[N],char b[N])
{
    int lena=strlen(a),lenb=strlen(b);
    for (int i=lena;i>=0;i--)
    for (int j=lenb;j>=0;j--)
        if (i==lena||j==lenb||a[i]!=b[j]) f[i][j]=0;
        else f[i][j]=f[i+1][j+1]+1;
}

void calc(int d[N],int f[N][N],char a[N])
{
    int n=strlen(a);
    for (int i=0;i<n;i++) {
        d[i]=0;
        for (int j=0;j<n;j++)
        if (i!=j)
            d[i]=max(d[i],f[i][j]);
    }
}

int main()
{
   // freopen("1.txt","r",stdin);
    scanf("%s",a); lena=strlen(a);
    scanf("%s",b); lenb=strlen(b);
    lca(csa,a,a);
    lca(csb,b,b);
    lca(csab,a,b);
    calc(da,csa,a);
    calc(db,csb,b);
    int ans=-1;
    for (int i=0;i<lena;i++)
    for (int j=0;j<lenb;j++) {
        int m=max(da[i],db[j])+1;
        if (csab[i][j]>=m&&(ans==-1||m<ans)) ans=m;
    }
    printf("%d\n",ans);
    return 0;
}
