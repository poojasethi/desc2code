#include <iostream>
#include <cstdio>
#include <string>
#include <cstring>

using namespace std;

#define INF 0x7fffffff
#define MAXN 5005

int n,m;
int dp[MAXN][MAXN];
int a[MAXN],b[MAXN];

void solve(string& s1,string& s2,int l1,int l2)
{
    int i,j;
    memset(dp,0,sizeof(dp));
    for (i=l1-1;i>=0;i--)
    {
        for (j=l2-1;j>=0;j--)
        {
            if (s1[i]==s2[j]) dp[i][j]=dp[i+1][j+1]+1;
        }
    }
}

int main()
{
    string s1,s2;
    int i,j;
    int ans;
    cin>>s1>>s2;
    n=s1.size(); m=s2.size();
    solve(s1,s1,n,n);
    for (i=0;i<n;i++)
    {
        a[i]=0;
        for (j=0;j<n;j++) if (i!=j) a[i]=max(a[i],dp[i][j]);
        a[i]++;
    }
    solve(s2,s2,m,m);
    for (i=0;i<m;i++)
    {
        b[i]=0;
        for (j=0;j<m;j++) if (i!=j) b[i]=max(b[i],dp[i][j]);
        b[i]++;
    }

    solve(s1,s2,n,m);

    ans=INF;
    for (i=0;i<n;i++)
    {
        for (j=0;j<m;j++)
        {
            if (dp[i][j]>=max(a[i],b[j]))
            {
                ans=min(ans,max(a[i],b[j]));
            }
        }
    }
    if (ans!=INF) cout<<ans<<endl;
    else cout<<-1<<endl;
    return 0;
}
