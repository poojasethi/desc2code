#include <string>
#include <iostream>
#include <algorithm>
using namespace std;
string s, t;
int ss[5010][5010], tt[5010][5010], st[5010][5010];
int ls[5010], lt[5010];
int main() {
	cin >> s >> t;
	for (int i = 0; i < s.size(); i++) {
		for (int j = 0; j < s.size(); j++) {
			if (s[i] == s[j]) {
				ss[i+1][j+1] = ss[i][j] + 1;
			} else {
				ss[i+1][j+1] = 0;
			}
		}
	}
	for (int i = 0; i < s.size(); i++) {
		ls[i] = 0;
		for (int j = 0; j < s.size(); j++) {
			if (i != j) {
				ls[i] = max(ls[i], ss[i+1][j+1]);
			}
		}
	}
	for (int i = 0; i < t.size(); i++) {
		for (int j = 0; j < t.size(); j++) {
			if (t[i] == t[j]) {
				tt[i+1][j+1] = tt[i][j] + 1;
			} else {
				tt[i+1][j+1] = 0;
			}
		}
	}
	for (int i = 0; i < t.size(); i++) {
		lt[i] = 0;
		for (int j = 0; j < t.size(); j++) {
			if (i != j) {
				lt[i] = max(lt[i], tt[i+1][j+1]);
			}
		}
	}
	int ans = -1;
	for (int i = 0; i < s.size(); i++) {
		for (int j = 0; j < t.size(); j++) {
			if (s[i] == t[j]) {
				st[i+1][j+1] = st[i][j] + 1;
			} else {
				st[i+1][j+1] = 0;
			}
			int tmp = max(ls[i], lt[j]);
			if (st[i+1][j+1] > tmp) {
				if (ans == -1) {
					ans = tmp + 1;
				} else {
					ans = min(ans, tmp + 1);
				}
			}
		}
	}
	cout << ans << endl;
	return 0;
}
