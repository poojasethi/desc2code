#include<bits/stdc++.h>
using namespace std;
int lcp[5005][5005];
int ma[50005],mb[50005];
void LCP(string a,string b)
{
    int len1=a.size();
    int len2=b.size();
    for(int i=len1-1;i>=0;i--)
    {
        for(int j=len2-1;j>=0;j--)
        {
            if(a[i]==b[j])
                 lcp[i][j]=1+lcp[i+1][j+1];
            else lcp[i][j]=0;
        }
    }
}
int main()
{
   string a,b;
    cin>>a>>b;
    memset(lcp,0,sizeof lcp);
    LCP(a,a);
    for(int i=0;a[i];i++)
    {
        int mn=0;
        for(int j=0;a[j];j++){if(i==j)continue;
            mn=max(mn,lcp[i][j]);
        }
            ma[i]=mn+1;

    }
    memset(lcp,0,sizeof lcp);
    LCP(b,b);
     for(int i=0;b[i];i++)
    {
        int mn=0;
        for(int j=0;b[j];j++){
                if(i==j)continue;
            mn=max(mn,lcp[i][j]);
        }
            mb[i]=mn+1;

    }
    int ans=INT_MAX;
    memset(lcp,0,sizeof lcp);
    LCP(a,b);
    for(int i=0;a[i];i++)
    {
        for(int j=0;b[j];j++)
        {
            if(lcp[i][j]>=max(ma[i],mb[j]))
                 ans=min(ans,max(ma[i],mb[j]));
        }
    }
    if(ans==INT_MAX)ans=-1;
    cout<<ans<<endl;
  return 0;
}
