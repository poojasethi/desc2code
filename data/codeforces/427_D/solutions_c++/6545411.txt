#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int dp[5003][5003];
char str1[10000],str2[10000];
int map1[10000],map2[10000];

int main(){
	scanf("%s",str1+1);
	scanf("%s",str2+1);
	int n=strlen(str1+1),m=strlen(str2+1);
	for (int i=n; i; --i)
		for (int j=n; j;--j)
			if (str1[i]==str1[j]) dp[i][j]=dp[i+1][j+1]+1;
	for (int i=1; i<=n; ++i)
		for (int j=1; j<=n; ++j)
			if (i!=j) map1[i]=max(map1[i],dp[i][j]);
	memset(dp,0,sizeof(dp));
	for (int i=m; i; --i)
		for (int j=m; j; --j)
			if (str2[i]==str2[j]) dp[i][j]=dp[i+1][j+1]+1;
	for (int i=1; i<=m; ++i)
		for (int j=1; j<=m; ++j)
			if (i!=j) map2[i]=max(map2[i],dp[i][j]);
	memset(dp,0,sizeof(dp));
	for (int i=n; i; --i)
		for (int j=m; j; --j)
			if (str1[i]==str2[j]) dp[i][j]=dp[i+1][j+1]+1;
	int ans=100000000;
	for (int i=1; i<=n; ++i)
		for (int j=1; j<=m; ++j){
			int t=max(map1[i],map2[j]);
			if (t>=dp[i][j]) continue;
			ans=min(t+1,ans);
		}
	if (ans>n) ans=-1;
	printf("%d\n",ans);
}
