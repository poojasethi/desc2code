#include<bits/stdc++.h>

#define FOR(a,b,c) for(int a=b;a<=c;a++)
#define ROF(a,b,c) for(int a=b;a>=c;a--)
#define inf 2000000000
using namespace std;
const int n = 5000 + 5;
int N,M;

int dp1[n][n] , dp2[n][n] , dn[n][n] , pre1[n] , pre2[n];

char A[n],B[n];

int main()
{
	int res = inf;
	scanf("%s %s",A+1,B+1);
	N = strlen(A+1);
	M = strlen(B+1);
	ROF(i,N,1)	ROF(j,N,1)	dp1[i][j] = (A[i] == A[j]) * (dp1[i+1][j+1]+1);
	ROF(i,M,1) 	ROF(j,M,1) 	dp2[i][j] = (B[i] == B[j]) * (dp2[i+1][j+1]+1);
	ROF(i,N,1) 	ROF(j,M,1) 	 dn[i][j] = (A[i] == B[j]) *  (dn[i+1][j+1]+1); 
	
	FOR(i,1,N) 	FOR(j,1,N)	if(i!=j)  pre1[i] = max(pre1[i] , dp1[i][j]);
	
	FOR(i,1,M)	FOR(j,1,M)	if(i!=j)  pre2[i] = max(pre2[i] , dp2[i][j]);
	
	FOR(i,1,N)	
		FOR(j,1,M)  
			if(dn[i][j] > max(pre1[i] , pre2[j])) res = min(res, max(pre1[i],pre2[j])+1);
	
	cout << (res==inf ? -1 : res) << endl;
	return 0;
}
