#include<cstdio>
int n,m,k;
int main(){
	scanf("%d",&k);
	if((k&1)==0){
		printf("NO\n");
		return 0;
	}
	if(k==1){
		printf("YES\n2 1\n1 2\n");
		return 0;
	}
	n=2*k+4;
	m=k*(k+2);
	printf("YES\n%d %d\n1 2\n%d %d\n%d %d\n",n,m,k*2+1,k*2+3,k*2+2,k*2+4);
	
	for(int i=2;i<=k;i++){
		printf("1 %d\n2 %d\n",i*2-1,i*2);
		printf("%d %d\n%d %d\n",i*2-1,k*2+1,i*2-1,k*2+3);
		printf("%d %d\n%d %d\n",i*2,k*2+2,i*2,k*2+4);
		if(i&1)for(int j=i+1;j<=k;j++)printf("%d %d\n%d %d\n",i*2-1,j*2-1,i*2,j*2);
		else for(int j=i+2;j<=k;j++)printf("%d %d\n%d %d\n",i*2-1,j*2-1,i*2,j*2);
	}
	return 0;
}
