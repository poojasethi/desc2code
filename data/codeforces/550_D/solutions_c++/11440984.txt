#include <bits/stdc++.h>
using namespace std;
const int MAX = 1e5+10;
int k;
void aa(int e){
    for(int i=2; i<=k; i++){printf("%d %d\n",1+e,i+e);}
    for(int i=2; i<=k; i++){
        for(int j=k+1; j<2*k; j++){
            printf("%d %d\n",i+e,j+e);
        }
    }
    for(int i=k+1; i<2*k; i+=2){printf("%d %d\n",i+e,i+e+1);}
}

int main(){
    scanf("%d",&k);
    if(k%2==0){printf("NO\n"); return 0;}
    printf("YES\n");
    printf("%d %d\n",4*k-2,2*k*k-k);
    aa(0);
    aa(2*k-1);
    printf("%d %d\n",1,2*k);
    return 0;
}
