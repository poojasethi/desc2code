#include <iostream>

using namespace std;

int k, n, m;

int main()
{
	cin >> k;
	
	if (k%2 == 0)
	{
		cout << "NO" << endl;
		return 0;
	}
	
	cout << "YES" << endl;
	cout << 4*k-2 << ' ' << k*(2*k-1) << endl;
	
	cout << "1 " << 2*k << endl;
	
	for (int i = 2; i <= k; ++i)
	{
		cout << "1 " << i << endl;
		cout << 2*k << ' ' << 2*k+i-1 << endl;
	}
	
	for (int i = 2; i <= k; ++i)
	{
		for (int j = k+1; j < 2*k; ++j)
		{
			cout << i << ' ' << j << endl;
			cout << 2*k-1+i << ' ' << 2*k-1+j << endl;
		}
	}
	
	for (int i = k+1; i <= 2*k-2; i+=2)
	{
		cout << i << ' ' << i+1 << endl;
		cout << 2*k-1+i << ' ' << 2*k+i << endl;
	}
}
