#include<cstdio>
int main()
{
int k; scanf("%d",&k);
if(k%2==0) {printf("NO\n"); return 0;}

for(int i=3; i<2+k; i++) printf("%d %d\n",1,i);
for(int i=3; i<2+k; i++)
for(int j=k+2; j<k+1+k; j++)
printf("%d %d\n",i,j);
for(int j=k+2; j<k+1+k; j+=2) printf("%d %d\n",j,j+1);
printf("1 2\n");
for(int i=k+k+1; i<k+k+k; i++) printf("%d %d\n",2,i);

for(int j=k+k+k; j<k+k+k+k-1; j++)
printf("%d %d\n",i,j);

}
