#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define print(x) cout << #x << "=" << x << endl
using namespace std;

void work(int x,int k){
	
	for(int i=x+2;i<=x+k;++i){
		printf("%d %d\n",x+1,i);
	}
	for(int i=x+2;i<=x+k;++i){
		for(int j=x+k+1;j<=x+k*2-1;++j){
			printf("%d %d\n",i,j);
		}
	}
	for(int i=x+k+2;i<=x+k*2-1;i+=2){
		printf("%d %d\n",i-1,i);
	}
}
int main(){
	int k;
	scanf("%d",&k);
	if(k%2==0){
		puts("NO");
		return 0;
	}
	puts("YES");
	int n=((k-1)*2+1)*2;
	int m=(((k-1)*2+1)*k-1)/2*2+1;
	printf("%d %d\n",n,m);
	work(0,k);
	printf("%d %d\n",1,n/2+1);
	work(n/2,k);
	return 0;
}
