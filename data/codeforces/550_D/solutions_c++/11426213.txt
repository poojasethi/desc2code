#include<cstdio>

int n,m,x,p;
int main()
{
	scanf("%d",&x);
	if (x%2)
	{
		printf("YES\n");
		n=2+(x-1)*4;
		m=n*x/2;
		printf("%d %d\n",n,m);
		printf("%d %d\n",1,2);
		p=2;
		for (int i=1;i<x;i++) printf("1 %d\n",++p);
		for (int i=1;i<x;i++) p++;
		for (int i=3;i<x+2;i++)
			for (int j=x+2;j<=p;j++) printf("%d %d\n",i,j);
		for (int i=x+2;i<=p;i+=2) printf("%d %d\n",i,i+1);
		int st=p+1;
		for (int i=1;i<x;i++) printf("2 %d\n",++p);
		int ed=p+1;
		for (int i=1;i<x;i++) p++;
		for (int i=st;i<ed;i++)
			for (int j=ed;j<=p;j++) printf("%d %d\n",i,j);
		for (int i=ed;i<=p;i+=2) printf("%d %d\n",i,i+1);
	}
	else printf("NO\n");
	return 0;
}
