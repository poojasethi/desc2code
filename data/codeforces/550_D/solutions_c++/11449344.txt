//REVERSER
#include<bits/stdc++.h>
using namespace std;
int main()
{
  int k;
  cin>>k;
  if(k%2 == 0)
    cout<<"NO";
  else{
    cout<<"YES"<<endl;
    cout<<4*k-2<<" "<<k*(2*k-1)<<endl;
    for(int i=2;i<=k;i++)
      cout<<1<<" "<<i<<endl;
    for(int i=2;i<=k;i++)
      for(int j=k+2;j<=2*k;j++)
	cout<<i<<" "<<j<<endl;
    for(int i=k+2;i<=2*k-1;i+=2)
      cout<<i<<" "<<i+1<<endl;
    cout<<1<<" "<<k+1<<endl;
    for(int i=2*k+1;i<=3*k-1;i++)
      cout<<k+1<<" "<<i<<endl;
    for(int i=2*k+1;i<=3*k-1;i++)
      for(int j=3*k;j<=4*k-2;j++)
	cout<<i<<" "<<j<<endl;
    for(int i=3*k;i<=4*k-3;i+=2)
      cout<<i<<" "<<i+1<<endl;
  }
}
