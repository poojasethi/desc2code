#include<stdio.h>

int i,j,n;

int main(){
    scanf("%d",&n);
    if(n%2==0){
        printf("NO\n");
        return 0;
    }
    printf("YES\n");
    printf("%d %d\n",4*n-2,(2*n-1)*n);
    printf("1 2\n");
    for(i=2;i<=n;i++){
        printf("%d %d\n",2*i-1,1);
        for(j=2*n+1;j<=4*n-2;j+=2){
            printf("%d %d\n",2*i-1,j);
        }
        printf("%d %d\n",2*i,2);
        for(j=2*n+2;j<=4*n-2;j+=2){
            printf("%d %d\n",2*i,j);
        }
    }
    for(j=2*n+1;j<=4*n-2;j+=4)printf("%d %d\n",j,j+2);
    for(j=2*n+2;j<=4*n-2;j+=4)printf("%d %d\n",j,j+2);
}
