#include<bits/stdc++.h>
using namespace std;
int main() {
    int k,i,j,n,m,n2;
    scanf("%d",&k);
    if(k==1){printf("YES\n2 1\n2 1\n");return 0;}
    if(k%2==0){printf("NO\n");return 0;}
    n2=k+2;
    n=n2*2;
    m=k*k+k*2;
    printf("YES\n");
    printf("%d %d\n",n,m);
    printf("1 %d\n",n2+1);
    for(i=2;i<=k;i++)printf("1 %d\n%d %d\n",i,n2+1,i+n2);
    for(i=2;i<n2;i++)
    {
        for(j=i+1;j<=n2;j++)
        {
            if(i%2 || i>=k || j!=i+1)
            {
                printf("%d %d\n%d %d\n",i,j,i+n2,j+n2);
            }
        }
    }
    return 0;
}
