#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int k;
int main(){
  scanf("%d",&k);
  if(k==1) printf("YES\n2 1\n1 2\n");
  else if(k%2==0) printf("NO\n");
  else{
    printf("YES\n%d %d\n",(k+2)*2,(k+2)*k);
    int n=k+2;
    for(int i=1;i<=n;i++){
     for(int j=i+1;j<=n;j++){
      if(i==1 && j==n) continue;
      if(i==2 && j==n) continue;
      if(i%2==1 && i>1 && j==i+1) continue;
      printf("%d %d\n%d %d\n",i,j,n+n+1-i,n+n+1-j);
     }
    }
    printf("%d %d\n",n,n+1);
  }
  return 0;
}
