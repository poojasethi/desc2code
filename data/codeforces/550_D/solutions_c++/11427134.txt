#include <iostream>
#include <algorithm>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <cstdlib>
using namespace std;
struct edge
{
	int x,y;
}e[100000];
int k,tot;
int main()
{
    cin>>k;
	if (k%2==0) puts("NO");
	else
	{
		puts("YES");
		tot=0;
		for (int i=2;i<=k;i++)
		{
			e[++tot].x=1,e[tot].y=i;
			for (int j=k+1;j<=k+k-1;j++)
				e[++tot].x=i,e[tot].y=j;
		}
		for (int i=k+1;i<=k+k-1;i+=2)
			e[++tot].x=i,e[tot].y=i+1;
		cout<<k*4-2<<" "<<tot*2+1<<endl;
		for (int i=1;i<=tot;i++)
			printf("%d %d\n%d %d\n",e[i].x,e[i].y,e[i].x+k*2-1,e[i].y+k*2-1);
		cout<<1<<" "<<k*2<<endl;
	}
	return 0;
}
