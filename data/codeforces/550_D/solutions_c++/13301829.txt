#include <stdio.h>
#include <stdlib.h>

int k;
int main() {
    scanf("%d",&k);
    if(!(k&1)) {
        puts("NO");
        return 0;
    }
    puts("YES");
    printf("%d %d\n",k*k+1,(k*k+1)*k/2);
    puts("1 2");
    for(int i=0;i<k-1;++i) {
        int sp=(k+1)*i+3,ep=(k+1)*(i+1)+2;
        for(int j=sp;j<=ep;++j)
            for(int l=j+1;l<=ep;++l)
                if(j==sp&&l==j+1)
                    printf("%d %d\n%d %d\n",j,(i&1)+1,(i&1)+1,l);
                else
                    printf("%d %d\n",j,l);
    }
    return 0;
}
