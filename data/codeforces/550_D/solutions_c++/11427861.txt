#include <bits/stdc++.h>
using namespace std;


void get(int first,int t)
{
	for(int i=first;i<=first+t;i++)
	{
		for(int j=i+1;j<=first+t;j++)
			if(i!=first||j!=first+t)
				cout<<i<<" "<<j<<endl;
	}
}
int main()
{
	int k;
	cin>>k;
	if(k%2==0)
	{
		cout <<"NO"<<endl;
		return 0;
	}
	int n = 2;
	cout <<"YES"<<endl;
	int x = (k+1)*(k-1)+2;
	cout <<x<<" "<<x*k/2<<endl;
	for(int i=0;i<(k-1)/2;i++)
	{
		cout << 1 <<" "<< n<<endl;
		cout << 1 <<" "<<n+k<<endl;
		get(n,k);
		n+=k+1;
	}
	cout <<x<<" "<<1<<endl;
	for(int i=0;i<(k-1)/2;i++)
	{
		cout << x<<" "<<n<<endl;
		cout << x<<" "<<n+k<<endl;
		get(n,k);
		n+=k+1;
	}
	return 0;
}
