#include<cstdio>

int k, n;
int e[222][222];

void doit(int l, int r, int dest) {
	for(int i=l;i<=r;i++)
		for(int j=i+1;j<=r;j++)
			e[i][j] = true;
	for(int i=l;i<r-2;i+=2) {
		e[i][i+1] = false;
		e[i][dest] = e[i+1][dest] = true;
	}
}

int main() {
	scanf("%d", &k);
	if(k==1) {
		printf("YES\n2 1\n1 2\n");
		return 0;
	}
	if((k&1)==0) {
		printf("NO\n"); return 0;
	}
	n = (k+1)*2+2;
	printf("YES\n%d %d\n", n, n*k/2);
	
	doit(1, k+1, n-1);
	doit(k+2, n-2, n);
	e[n-1][n] = 1;
	for(int i=1;i<=n;i++)
		for(int j=i+1;j<=n;j++)
			if(e[i][j])
				printf("%d %d\n", i, j);
	
	return 0;
}
