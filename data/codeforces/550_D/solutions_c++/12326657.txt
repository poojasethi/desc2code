#include<stdio.h>
int n;
inline void add(int x,int y)
{
	printf("%d %d\n",x,y);
	printf("%d %d\n",n+x,n+y);
}
int main()
{
	int k;
	scanf("%d",&k);
	if (k==1)
	{
		printf("YES\n");
		printf("2 1\n");
		printf("1 2\n");
		return 0;
	}
	if (k%2==0)
	{
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	n=k+2;
	printf("%d %d\n",2*n,k*n);
	add(n-1,n-2);
	for (int i=1;i<=n-3;i++)
	{
		add(i,n);
		add(i,n-1);
		add(i,n-2);
	}
	for (int i=1;i<=n-3;i++)
	{
		int j;
		if (i&1) j=i+2;
		else j=i+1;
		while (j<=n-3)
		{
			add(i,j);
			j++;
		}
	}
	printf("%d %d\n",n,2*n);
	return 0;
}
