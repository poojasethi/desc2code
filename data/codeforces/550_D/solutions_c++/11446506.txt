#include <stdio.h>
#include <string.h>


int main()
{
    int i,j,m,n,k;
    scanf("%d",&k);
    if(k%2==0)
    {
         puts("NO");
         return 0;
    }
    puts("YES");
    if(k==1)
    {
        printf("2 1\n1 2\n");
        return 0;
    }
    printf("%d %d\n",2*k+4,(k+2)*k);
    for(i=1;i<=k+2;i++)
        for(j=i+1;j<=k+2;j++)
        {
            if(i==1 && j<=k)
                printf("%d %d\n%d %d\n",i,j,i+k+2,j+k+2);
            if(i!=1)
                if(j!=i+1 || (j==i+1 && (j>k || i%2==1)))
                    printf("%d %d\n%d %d\n",i,j,i+k+2,j+k+2);
        }
    printf("1 %d\n",k+3);
    return 0;
}
