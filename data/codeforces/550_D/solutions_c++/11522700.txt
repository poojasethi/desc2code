#include <bits/stdc++.h>
using namespace std;

int main() {
	int k;
	scanf("%d", &k);
	if(k % 2 == 0)
		printf("NO\n");
	else {
		printf("YES\n%d %d\n", 4*k-2, (2*k-1)*k);
		for(int i = 2; i<=k; i++)
			printf("%d %d\n", 1, i);
		for(int i = 2; i<=k; i++)
			for(int j = k+1; j < 2*k; j++)
				printf("%d %d\n", i, j);
		for(int i = k+1; i<2*k; i+= 2)
			printf("%d %d\n", i, i+1);

		printf("%d %d\n", 1, 2*k);

		for(int i = 2*k+1; i<=3*k-1; i++)
			printf("%d %d\n", 2*k, i);
		for(int i = 2*k+1; i<=3*k-1; i++)
			for(int j = 3*k; j < 4*k-1; j++)
				printf("%d %d\n", i, j);
		for(int i = 3*k; i<4*k-1; i+= 2)
			printf("%d %d\n", i, i+1);
	}
	return 0;
}
