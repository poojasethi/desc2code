#include<cstdio>
int main(){
    int n;
    scanf("%d", &n);
    if(n % 2 == 0){
        puts("NO");
        return 0;
    }
    puts("YES");
    printf("%d %d\n", 2 + 4*(n-1), 1 + 2 * (n-1 + (n-1)*(n-1) + (n-1)/2));
    puts("1 2");
    int d = 2 * (n-1);
    for(int i=1; i<n; i++){
        printf("1 %d\n", i+2);
        printf("2 %d\n", i+2+d);
        for(int j=1; j<n; j++){
            printf("%d %d\n", i+2, 1+j+n);
            printf("%d %d\n", i+2+d, 1+j+n+d);
        }
    }
    for(int i=1; i<n; i+=2){
        printf("%d %d\n", n+1+i, n+2+i);
        printf("%d %d\n", n+1+i+d, n+2+i+d);
    }
    return 0;
}
