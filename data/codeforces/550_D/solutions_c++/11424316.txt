int n;

void output(int k){
	for (int i=2;i<=n;i++) printf("%d %d\n",k+1,i+k);
	for (int i=2;i<=n;i++) printf("%d %d\n%d %d\n",i+k,k+n+1,i+k,k+n+2);
	printf("%d %d\n",n+1+k,n+2+k);
	for (int i=2;i<=n;i++)
		for (int j=i+1;j<=n;j++) 
			if (i%2==0 && j==i+1) continue;
				else printf("%d %d\n",i+k,j+k);
	
}

int main(){
	cin >> n;
	if (n==1){puts("YES");puts("2 1");puts("1 2");} else

	}else puts("NO");

}
