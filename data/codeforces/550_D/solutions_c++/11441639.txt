#include<stdio.h>

int n,v,e,i,j;

int main(){
    scanf("%d",&n);
    if(n%2==0){
        printf("NO\n");
    }
    else{
        printf("YES\n");
        v=4*n-2;
        e=(v/2)*n;
        printf("%d %d\n",v,e);
        printf("1 2\n");
        for(i=1;i<n;i++){
            printf("%d %d\n",1,2*i+1);
            printf("%d %d\n",2,2*i+2);
        }
        for(i=3;i<=2*n;i++){
            for(j=1;j<n;j++){
                printf("%d %d\n",i,2*n+2*j-(i%2));
            }
        }
        for(i=n;i<2*(n-1);i+=2){
            printf("%d %d\n",2*i+1,2*i+3);
            printf("%d %d\n",2*i+2,2*i+4);
        }
    }
}
