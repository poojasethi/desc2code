#include<stdio.h>
#define V 103
int deg[V];
bool graph[V][V]={0};

int main(){
	int i,j,k,sz;
	scanf("%d",&k);
	if(k&1){
		printf("YES\n");
		if(k==1)
			printf("2 1\n1 2\n");
		else{
			sz = k+2;
			printf("%d %d\n",2*sz,k*sz);
			for(i=2;i<=k;i++)
				graph[1][i] = true;
			for(i=2;i<=k+1;i++)
				for(j=i+1;j<=k+2;j++)
					graph[i][j] = true;
			for(i=3;i<=k;i+=2)
				graph[i-1][i] = false;
			for(i=1;i<=k+1;i++)
				for(j=i+1;j<=k+2;j++)
					if(graph[i][j])
						printf("%d %d\n%d %d\n",i,j,i+k+2,j+k+2);
			printf("%d %d\n",1,k+3);
		}
	}
	else
		printf("NO\n");
	return 0;
}
