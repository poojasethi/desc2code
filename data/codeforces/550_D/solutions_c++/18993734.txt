#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main(){
	int k;
	while(cin>>k){
		if(k&1){
			cout<<"YES\n";
			int n=4*k-2,m=2*k*k-k;
			cout<<n<<" "<<m<<endl;
			for(int i=1;i<k;++i)
				cout<<1<<" "<<i+1<<endl;
			for(int i=1;i<k;++i){
				if(i&1)
					cout<<k+i<<" "<<k+i+1<<endl;
				for(int j=1;j<k;++j)
					cout<<k+i<<" "<<j+1<<endl;
			}
			cout<<1<<" "<<2*k<<endl;
			for(int i=1;i<k;++i)
				cout<<2*k<<" "<<2*k+i<<endl;
			for(int i=1;i<k;++i){
				if(i&1)
					cout<<3*k+i-1<<" "<<3*k+i<<endl;
				for(int j=1;j<k;++j)
					cout<<3*k+i-1<<" "<<2*k+j<<endl;
			}
			continue;
		}
		else
			cout<<"NO\n";
	}
}
