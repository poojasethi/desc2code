#include <bits/stdc++.h>
using namespace std;
int main() {
  int k;
  scanf("%d", &k);
  if (k == 1) {
    printf("YES\n2 1\n1 2\n");
    return 0;
  }
  if (k % 2 == 0) {
    printf("NO\n");
    return 0;
  }
  int n = k * 2 + 4;
  int m = n * k / 2;
  printf("YES\n%d %d\n1 %d\n", n, m, n);
  for (int i = 1; i < k; i++) {
    printf("1 %d\n", i + 1);
    printf("%d %d\n", n, n - i);
  }
  for (int i = 2; i <= k + 2; i++) {
    for (int j = i + 1; j <= k + 2; j++) {
      if (i % 2 == 0 && j == i + 1 && j <= k) continue;
      printf("%d %d\n", i, j);
      printf("%d %d\n", n - i + 1, n - j + 1);
    }
  }
  return 0;
}
