#include <iostream>
using namespace std;
int k;
void prt(int a, int b){
    cout<<a<<" "<<b<<"\n";
    cout<<a+k+2<<" "<<b+k+2<<"\n";
}
int main() {
    cin>>k;
    if(k==1){
        cout<<"YES\n2 1\n1 2\n";
    }
    else if(k%2==0)cout<<"NO";
    else{
        cout<<"YES\n"<<k*2+4<<" "<<k*(k+2)<<"\n";
        cout<<"1 "<<k+3<<"\n";
        prt(k+1,k+2);
        for(int i=2;i<k;i+=2){
            prt(i,1);prt(i+1,1);
            for(int j=i+2;j<k+3;j++){
                prt(i,j);prt(i+1,j);
            }
        }
    }
}
