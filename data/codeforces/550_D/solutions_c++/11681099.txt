#include <bits/stdc++.h>
using namespace std;
void func(int k,int gap)
{
	int i,j;
	for(i=1;i<=k+1;i++)
			{
				for(j=i+1;j<=k+1;j++)
				{
					if(j-i==(k+1)/2&&i!=(k+1)/2)
						continue;
					printf("%d %d\n",i+1+gap,j+1+gap);

				}
			}	
			for(i=1;i<=k;i++)
			{
				if(i==(k+1)/2)
					continue;
				printf("%d %d\n",1+gap,i+1+gap);
			}
}
int main()
{
	int k;
	scanf("%d",&k);
	if(k&1)
	{
		if(k==1)
			printf("YES\n%d %d\n%d %d",2,1,1,2);
		else
		{
			printf("YES\n");
			printf("%d %d\n",2*k+4, k*(k+2));
			func(k,0);
			printf("%d %d\n",1,k+3);
			func(k,k+2);
		}
	}
	else
		printf("NO\n");
	return 0;
}
