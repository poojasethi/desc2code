#include<stdio.h>
int main()
{
	int k,i,j;
	scanf("%d",&k);
	if(k%2==0)
	{
		printf("NO");
		return 0;
	}
	else if(k==1)
	{
		printf("YES\n2 1\n1 2");
		return 0;
	}
	else
	{
		printf("YES\n");
		printf("%d %d\n",(2*k+4),(k+2)*k);
		for(i=2;i<=k;i++)
		printf("1 %d\n%d %d\n",i,k+3,i+k+2);
		for(i=2;i<=k-1;i++)
		{
			if(i%2==0)
			j=i+2;
			else
			j=i+1;
			for(;j<=k;j++)
			printf("%d %d\n%d %d\n",i,j,i+k+2,j+k+2);
		}
		for(i=2;i<=k;i++)
		printf("%d %d\n%d %d\n%d %d\n%d %d\n",i,k+1,i,k+2,i+k+2,2*k+3,i+k+2,2*k+4);
		printf("%d %d\n%d %d\n",k+1,k+2,k+1+k+2,k+1+k+3);
		printf("1 %d\n",k+3);
	}
}
