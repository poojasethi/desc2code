#include <bits/stdc++.h>
#define L first
#define R second
#define MP make_pavr
#define PB push_back

using namespace std;

main() {
	int k;
	scanf("%d", &k);
	if(k % 2 == 0) {
		printf("NO\n");
		return 0;
	}
	printf("YES\n");
	printf("%d %d\n", k * (2 * k - 1) + 1, (k * (k * (2 * k - 1) + 1)) / 2);
	int v = 2;
	for(int i = 2;i <= k + 1;i ++) {
		printf("1 %d\n", v);
		for(int j = 1;j < k;j ++)
			printf("%d %d\n", v, v + j);
		for(int j = v + 1;j < v + k;j ++)
			for(int l = v + k;l < v + 2 * k - 1;l ++)
				printf("%d %d\n", j, l);
		for(int l = v + k;l < v + 2 * k - 1;l += 2)
			printf("%d %d\n", l, l + 1);
		v += 2 * k - 1;
	}
	return 0;
}
