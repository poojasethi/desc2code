#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>
#include <deque>

using namespace std;

void EXEC()
{
	int n;
	scanf("%d", &n);
	if (n % 2 == 0) {
		printf("NO\n");
		return;
	}
	
	printf("YES\n");
	
	int m = n + 2;
	if (n == 1)
		m = 1;
	printf("%d %d\n", m * 2, m * n);
	
	printf("%d %d\n", 1, 1 + m);
	int s = m * n - 1;
	for (int i = 1; i < m; ++i) {
		int t = s / (m + 1 - i);
		s -= t;
		for (int j = 1; j <= t; ++j) {
			int t2 = m - ((s % (m - i) == 0) ? m - i : s % (m - i)) + 1;
			printf("%d %d\n", i, t2);
			printf("%d %d\n", i + m, t2 + m);
			--s;
		}
	}
}

int main()
{
	EXEC();
	return 0;
}
