#include <iostream>
using namespace std;

int main(){
	string a,lower,upper;
	cin>>a;
	lower = a;
	upper = a;
	int l=0,h=0;
	for(int i=0;i<a.length();i++){
		if(a[i]<91){
			h++;
			lower[i] = a[i]+32;
		} else {
			l++;
			upper[i] = a[i]-32;
		}
	}
	if(h>l) cout<<upper<<endl;
	else cout<<lower<<endl;

}
