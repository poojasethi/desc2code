#include<stdio.h>
#include<string.h>
int main()
{
    int flag=0,flagbig=0;
    char s[110];
    gets(s);
    for(int i=0;s[i];i++) 
        if(s[i]<='Z' && s[i]>='A') flagbig++;
        else flag++;
    if(flag>=flagbig) strlwr(s);
    else strupr(s);
    puts(s);
    return 0;
}
