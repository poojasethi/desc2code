#! /usr/bin/python

n = int(raw_input())

isPrime = [False, False] + [True for i in range(n - 1)]
ans = 0
questions = []

for i in range(n + 1):
	if isPrime[i]:
		j = 2
		while i*j <= n:
			isPrime[i*j] = False
			j += 1

		p = i
		while p <= n:
			ans += 1
			questions += [p]
			p *= i

print ans
