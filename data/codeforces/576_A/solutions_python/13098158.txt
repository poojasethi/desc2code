import math
prime=[2,3,5,7,11,13]
for i in xrange(7,1001):
    x=int(math.sqrt(i))
    for j in xrange(2,x+1):
        if not i%j:
            break
    else:
        prime.append(i)
n=input()
ans=[]
used=[0]*1001
for p in prime:
    x=p
    while(x<=n):
        if not used[x]:ans.append(x)
        used[x]=1
        x*=p
print len(ans)
print ' '.join(map(str,ans))
