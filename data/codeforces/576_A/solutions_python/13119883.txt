n = int(raw_input())
l = []

primes = [2]
for i in range(3, n + 1):
    for p in primes:
        if i % p == 0: break
        elif p * p > i: break
    if p * p > i:
        primes.append(i)
#print primes

for p in primes:
    cur = p
    while cur <= n:
        l.append(cur)
        cur *= p

print len(l)
print ' '.join(str(x) for x in l)
