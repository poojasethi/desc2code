import math
def notprime(i):
    for j in range(2,int(math.sqrt(i)+1)):
        if i%j==0:
            return j
    return 0

def power(i):
    factor = notprime(i)
    if factor==0:
        return 0
    while (i>0 and i%factor==0):
        i = i/factor
    if i==1:
        return 1
    return 0

n = input()
ans = []
for i in range(2,n+1):
    if notprime(i)==0:
        ans.append(i)
    if power(i):
        ans.append(i)

print len(ans)
