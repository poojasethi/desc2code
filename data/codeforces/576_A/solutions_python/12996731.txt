n = int(raw_input())

prime = []
u = [0 for i in range(n + 1)]

for i in range(2, n + 1) :
    if not u[i] : prime.append(i)
    for j in prime :
        if i * j > n : break
        u[i * j] = 1
        if i % j == 0 : break

A = []
for i in prime :
    A.append(i)
    while (A[len(A) - 1] * i <= n) :
        A.append(A[len(A) - 1] * i)

print len(A)
print ' '.join(map(str, A))
