#!/usr/bin/python

import math

def sieve(n):
	primos = [True]*(n+1)
	primos[0] = False
	primos[1] = False
	m = int(math.sqrt(n))
	for i in range(2, m + 1):
		if primos[i]:
			for k in range(i*i,n+1,i):
				primos[k] = False
	return primos	

n = int(raw_input())

primos = sieve(n)

r = []

for p in xrange(2, len(primos)):
	if primos[p]:
		temp = p
		entrou = False
		while temp <= n:
			r.append(temp)
			temp *= p
			entrou = True
		if not entrou and p <=n:
			r.append(p)

print len(r)
print " ".join(map(str,r))
