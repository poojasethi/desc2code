n = input()

def prime (x):
    for k in range (2, x):
        if x % k == 0:
            return 0
    return 1

ans = 0
ret = []
for k in range (2, n + 1):
    if prime (k) == 1:
        l = k
        while l <= n:
            ans = ans + 1
            ret.append(l)
            l = l * k

print ans
print ' '.join(map (str, ret))
