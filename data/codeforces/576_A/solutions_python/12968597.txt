from math import log

n = int(raw_input())


ans = []


prime = []
isPrime = [True for i in xrange(1000)]
isPrime[0] = isPrime[1] = False
for num in xrange(2,1000):
    if isPrime[num]:
        for j in xrange(num+num,1000,num):
            isPrime[j] = False
        prime.append(num)




for p in prime:
    if p > n:break
    for i in xrange(1,int(log(n)+10)):
        if p**i > n:
            break
        else:
            ans.append(p**i)
print len(ans)
