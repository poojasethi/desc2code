def is_prime(n):
	if n == 2:
		return True
	elif n % 2 == 0 or n == 1:
		return False
		
	for i in xrange(3, int(n**0.5)+1, 2):
		if n % i == 0:
			return False
	return True

n = int(raw_input())
l = []
nums = [False]*(n+1)


for x in xrange(2, n+1):
	if is_prime(x) and not nums[x]:
		for j in xrange(1, n):
			k = x**j
			if k <= n:
				if not nums[k]:
					l.append(k)
					nums[k] = True
					
			else:
				break
