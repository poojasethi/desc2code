n = input()

prime_check_list = [True] * (n + 1)

for i in range(2, n+1):
    if prime_check_list[i] is False:
        continue;
    else:
        for j in range(2, n):
            if i * j > n:
                break
            else:
                prime_check_list[i*j] = False

prime_list = []
result_list = []

for i in range(2, n+1):
    if prime_check_list[i] is True:
        prime_list.append(i)

for p in prime_list:
    for i in range(1, n):
        if pow(p, i) > n:
            break;
        else:
            result_list.append(pow(p, i))
