#include<bits/stdc++.h>
using namespace std;
int n,h[2000],s,cvp[2000];

int asal(int x){
	for(int i=2;i<=sqrt(x);i++)
		if(x%i == 0)
			return 0;
	return 1;
}
int main(){
	cin >> n ;
	for(int i=2;i<=n;i++){
		if(asal(i)){
			for(int j=i;j<=n;j*=i)
				if(!h[j]){
					cvp[++s] = j;
					h[j]=1;
				}
		}
	}
	cout << s << endl;
	for(int i=1;i<=s;i++)
		cout << cvp[i] << ' ';
}
