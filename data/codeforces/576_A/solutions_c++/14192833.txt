#include<cstdio>

int n, x, tot;
bool f[1111];
int ans[1111];

int main() {
	scanf("%d", &n);
	tot = 0;
	for(int i=2;i<=n;i++) {
		if (f[i]) continue;
		for(int j=i*i;j<=n;j+=i)
			f[j] = true;
		x = i;
		while(x<=n) {
			ans[tot++] = x;
			x*=i;
		}
	}
	printf("%d\n", tot);
	for(int i=0;i<tot;i++)
		printf("%d%c", ans[i], i<tot-1 ? ' ' : '\n');
	
	return 0;
}
