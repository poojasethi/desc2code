#include <iostream>
using namespace std;

int prime[1001];

int main()
{
	int n, wyn=0;
	cin >> n;
	for (int i=2; i<=n; i++)
		if (prime[i] == 0)
		{
			for (int j=i; j<=n; j+=i)
				prime[j] = -1;
			for (int j=i; j<=n; j*=i)
			{
				wyn++;
				prime[j] = 1;
			}
		}
	
	cout << wyn << "\n";
	for (int i=2; i<=n; i++)
		if (prime[i] == 1)
			cout << i << " ";
}
