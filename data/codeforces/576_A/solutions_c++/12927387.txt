#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
#define N 200000
int cnt=0;
int n,a[N];
int main(){
	scanf("%d",&n);
	for(int i=2;i<=n;i++){
		int w=1;
		for(int j=2;j<i;j++)
			if(i%j==0)w=0;
		if(w){
			int t=i;
			while(t<=n)cnt++,a[cnt]=t,t*=i;
		}
	}printf("%d\n",cnt);
	for(int i=1;i<=cnt;i++)printf("%d ",a[i]);
	puts("");
	return 0;
}
