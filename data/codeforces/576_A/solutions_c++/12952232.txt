#include<stdio.h>
using namespace std;
int i,j,k,l,s,m,n,ans,v[1005],f[1005];
int main() {
    scanf("%d",&n);
    for (i=2;i<=n;i++) {
        if (!f[i])
        for (j=1;j*i<=n;j++)
        f[i*j]++;
    }
    for (i=1;i<=n;i++) if (f[i]==1) v[i]=1,ans++;
    printf("%d\n",ans);
    for (i=1;i<=n;i++) if (v[i]) printf("%d ",i);
}
