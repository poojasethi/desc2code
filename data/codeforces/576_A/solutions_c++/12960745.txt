#include<stdio.h>

int main()
{
	int a;
	scanf("%d",&a);
	int buffer[1001],i,j,n=0,out[1001];
	for(i=2;i<=a;i++) buffer[i]=0;
	for(i=2;i<=a;i++){
		if(buffer[i]==0){
			for(j=i;j<=a;j*=i) {
				out[n]=j;
				n++;
			}
			for(j=1;j<=a/i;j++) buffer[i*j]=1;
		}
	}
	printf("%d\n",n);
	for(i=0;i<n;i++) printf("%d ",out[i]);
	return 0;
}
