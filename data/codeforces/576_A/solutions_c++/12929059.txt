#include <bits/stdc++.h>

using namespace std;

int N,ar[1230];
vector <int> v;

int main(){
	scanf("%d",&N);
	
	for(int i=2;i<=N;i++)
		if(!ar[i]){
			for(int j=i;j<=N && !ar[j];j*=i)
				v.push_back(j);
			for(int j=i;j<=N;j+=i)
				ar[j]++;
		}
	
	printf("%d\n",v.size());
	
	for(int i=0;i<v.size();i++)
		printf("%d ",v[i]);
	puts("");
	
	return 0;
}
