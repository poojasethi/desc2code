#include <cstdio>
#include <vector>
using namespace std;
const int maxn=1e3+1 ;
bool visit[maxn] ;
vector<int> V;
int main()
{
    int n;
    scanf("%d",&n) ;
    for(int i=2;i<n+1;i++)
	if(!visit[i])
	{
	    for(int j=i;j<n+1;j+=i)
		visit[j]=true ;
	    for(int j=i;j<n+1;j*=i)
		V.push_back(j) ;
	}
    printf("%d\n",V.size()) ;
    for(int i=0;i<V.size();i++)
	printf("%d ",V[i]) ;
}
