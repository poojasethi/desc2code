#include<cstdio>
#define N 100100
int n,i,tot,c[N],ans[N],j,t;
int main()
{
    scanf("%d",&n);
    for (i=2;i<=n;i++)
    if (c[i]==0)
    {
        for (j=1;j<=n/i;j++)
        c[i*j]=1;
        t=i;
        while (t<=n)
        {
            tot++;ans[tot]=t;
            t=t*i;
        }
    }
    printf("%d\n",tot);
    for (i=1;i<=tot;i++)
        printf("%d ",ans[i]);
}
