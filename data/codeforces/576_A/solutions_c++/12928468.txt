#include<bits/stdc++.h>

using namespace std;

bool simp(int a){
int r=1;
for(int i=2;i<a;i++) if (a%i==0)r=0;
return r==1;
}

vector<int> ans;
int main(){
int n; cin>>n;
for(int i=2;i<3000;i++)if (simp(i)){ int t=i;
while (t<=n){ans.push_back(t);t*=i;}
}
cout<<ans.size()<<endl;
for(int i=0;i<ans.size();i++)cout<<ans[i]<<' ';
return 0;
}
