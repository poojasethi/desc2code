#include <bits/stdc++.h>
using namespace std;
int f[1023],i,j,c,n;
main(){
    cin>>n;
    for(i=2;i<=n;++i){
        for(j=2;i*j<=n;++j)
            if(__gcd(i,j)==1) f[i*j]=1;
    }
    for(i=2;i<=n;++i)
        if(f[i]==0) c++;
    cout<<c<<'\n';
    for(i=2;i<=n;++i)
        if(f[i]==0) cout<<i<<' ';
}
