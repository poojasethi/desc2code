#include <iostream>
using namespace std;

int main() {
	int a,b,c[1001],d=0,e=0;
	cin>>a;
	
	
	for(int i=2;i<=a;i++)
	{
		
		for(int z=1;z<=i;z++)
		{
			if(i%z==0)
			e++;
		}
		if(e < 3)
		{
			for(int o=i;o<=a;o*=i){
			c[d]=o;
			d++;
			}
		}
		e=0;
			
		
		
	}
	cout<<d<<endl;
	for(int i=0;i<d;i++)
		cout<<c[i]<<" ";
	
	
	return 0;
}
