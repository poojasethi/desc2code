#include<iostream>
#include<vector>
using namespace std;
int a[1001],ans=0;
vector<int> an;
int main()
{
	int n,i;
	cin>>n;
	for (i=2;i<=n;i++)if (!a[i])for (int j=i;j<=n;j+=i){a[j]++;}
	for (i=2;i<=n;i++)if (a[i]<=1){ans++;an.push_back(i);}
	cout<<ans<<endl;
	for (i=0;i<an.size();i++)cout<<an[i]<<' ';
	return 0;
}
