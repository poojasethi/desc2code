#include<cstdio>
using namespace std;
int ans[100005],m,n,i,s;
int prime(int k){
	if (k<=2) return k==2;
	for (int i=2;i*i<=k;i++)
		if (k%i==0) return 0;
	return 1;
}
int main()
{
	scanf("%d",&n);m=n;
	for (i=2;i<=m;i++)
		if (prime(i)){
			for (s=i;s<=n;s*=i)
				ans[++*ans]=s;
	  }
	printf("%d\n",*ans);
	for (i=1;i<=*ans;i++)
		printf("%d ",ans[i]);
}
