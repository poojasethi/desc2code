#include<cstdio>
using namespace std;
int o[1000];bool b1[1001];
int main()
{
    int n,lo=0;
    scanf("%d",&n);
    for(int i=2;i<=n;++i)if(!b1[i])
    {
        for(int j=2*i;j<=n;j+=i)b1[j]=1;
        for(int j=i;j<=n;j*=i)o[lo++]=j;
    }
    printf("%d\n",lo);
    while(lo--)
    {
        printf("%d",o[lo]);
        putchar(lo?' ':'\n');
    }
}
