#include <cstdio>
bool notp[1001];
int num,lst[5000];
int main()
{
	int n;scanf("%d",&n);
	if(n==1){printf("0\n");return 0;}
	for(int i=2;i<=n;i++)
		for(int j=2;j<=n/i;j++)
			notp[i*j]=1;
	for(int i=2;i<=n;i++)
	{
		if(notp[i])continue;
		int t=i;
		while(t<=n)
		{
			lst[++num]=t;
			t*=i;
		}
	}
	printf("%d\n",num);
	for(int i=1;i<=num;i++)
		printf("%d ",lst[i]);
}
