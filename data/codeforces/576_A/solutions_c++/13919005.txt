#include<bits/stdc++.h>
using namespace std;
bool v[1001];
int main(){
   int n;
   int ans[1001],p=0;
   scanf("%d",&n);
   for(int i=2;i<=n;i++){
    if(v[i])
        continue;
    int j=i;
    while(j<=n)
        ans[p++]=j,j*=i;
    for(int j=2*i;j<=n;j+=i)
        v[j]=true;
   }
   printf("%d\n",p);
   for(int i=0;i<p;i++)
    printf("%d ",ans[i]);
   return 0;
}
