#include <cstdio>

#define N 1005

int sto[N], cnt = 0;

bool isprime(int x) {
	for (int i = 2; i < x; i ++) if (x % i == 0) return 0;
	return 1;
}

main() {
	int n;
	scanf("%d", &n);
	for (int i = 2; i <= n; i ++) if (isprime(i)) {
		for (int j = i; j <= n; j *= i) sto[cnt ++] = j;
	}
	printf("%d\n", cnt);
	for (int i = 0; i < cnt; i ++) printf("%d ", sto[i]);
	return 0;
}
