#include <stdio.h>
int a[200],ans[1010],t,s;
int main(){
    int i,j,n;
    scanf("%d",&n);
    for(i=2;i<=n;++i){
        for(j=1;j<=t;++j)if(i%a[j]==0)break;
        if(j>t)a[++t]=i;
    }
    for(i=1;i<=t;++i){
        for(j=a[i];j<=n;j*=a[i])ans[++s]=j;
    }
    printf("%d\n",s);
    for(i=1;i<=s;++i)printf("%d ",ans[i]);
    return 0;
}
