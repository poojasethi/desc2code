#include<bits/stdc++.h>
using namespace std;
int n,p[1005],t=0,a[1005];
int main(){
    scanf("%d",&n);
    for (int i=2; i<=n; i++) p[i]=1;
    p[1]=0;
    for (int i=2; i<=n; i++) if (p[i]){
        a[++t]=i;
        for (int j=i+i; j<=n; j+=i) p[j]=0;
        for (int j=i*i; j<=n; j*=i) a[++t]=j;
    }
    printf("%d\n",t);
    for (int i=1; i<=t; i++) printf("%d ",a[i]); 
}
