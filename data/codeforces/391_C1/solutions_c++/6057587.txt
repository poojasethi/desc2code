#include <iostream>
#include <fstream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
int n,k,p[20],e[20],ans=2147483600,score[20]={0},eff=0;
bool beat[20]={0};
void process(int i,int sum){
	if (i>n){
		int count=1;
		for (int j=1;j<=n;j++){
			if (score[j]>sum) count++;
			else {
				if (score[j]==sum&&(!beat[j]))
					count++;
			}
		}
		if (count<=k){
			if (eff<ans) ans=eff;
		}
		return;
	}
	score[i]=p[i]+1;
	beat[i]=0;
	process(i+1,sum);
	
	eff+=e[i];
	score[i]=p[i];
	beat[i]=1;
	process(i+1,sum+1);
	eff-=e[i];
}
int main(){
	cin>>n>>k;
	for (int i=1;i<=n;i++)
		cin>>p[i]>>e[i];
	process(1,0);
	cout<<(ans==2147483600?-1:ans);
    return 0;
}
