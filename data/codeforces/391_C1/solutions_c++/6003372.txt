int main()
{
   int n, k, p[20], w[20];
   cin >> n >> k;
   int ans=100000000;
   for (int i=0; i<n; i++)
      cin >> p[i] >> w[i];

   for (int i=0; i<33000; i++)
   {
      int p2[20];
      int e=0, pts=0;
      for (int j=0; j<n; j++)
         p2[j]=p[j];
      for (int j=0; j<n; j++)
      {
         if ((i/(1<<j))%2==0)
         {
            e+=w[j];
            pts++;
         }
         else
            p2[j]++;
      }
      vector <pair <int, int> > pr;
      for (int j=0; j<n; j++)
         pr.push_back(make_pair(p2[j],((i/(1<<j))%2)));
      sort (pr.begin(),pr.end());
      reverse (pr.begin(), pr.end());
      int ss=n+1;
      for (int j=0; j<n; j++)
      {
         if (pr[j].first==pts && pr[j].second==0)
         {
            ss=j+1;
            break;
         }
         else if (pr[j].first<pts)
         {
            ss=j+1;
            break;
         }
      }
      if (e<ans && ss<=k)
         ans=e;
   }
   if (ans==100000000)
      cout << -1;
   else
      cout << ans;
}
