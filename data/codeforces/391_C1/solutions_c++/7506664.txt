#include <bits/stdc++.h>
#define f0(i,n) for(__typeof(n) i=0;i<n;i++)
#define rep(i,a,b) for(__typeof(b) i=a ; i<=b ; i++)
#define read(s) freopen(s , "r" , stdin)
#define write(s) freopen(s , "w" , stdout)

using namespace std;
const int MAXN = 200007;

int N,k;
int p[MAXN],e[MAXN];
bool ck[MAXN];

int main(){

	scanf("%d %d" ,&N,&k);
	int maxx = 1 << N;
	int minn = INT_MAX;
	f0(i,N){
		scanf("%d %d" ,&p[i],&e[i]);
		minn = min ( minn , p[i] );
	}
	vector < int > tmp(p,p+N);
	tmp.push_back(N);
	sort(tmp.begin(), tmp.end() , greater < int > () );

	if (tmp[k-1]>N) return puts("-1") , 0;

	int res = INT_MAX;

	f0(mask , maxx){
		memset(ck , 0 , sizeof(ck));
		typedef pair < int , int > ii;
		vector < ii > P;
		f0(i,N) P.push_back(ii(p[i],i));
		P.push_back(ii(0,N));

		int cost = 0;

		f0(i,N) if(mask&(1<<i)) P[N].first++ , ck[i] = 1 , cost += e[i];
				else P[i].first++;

		int point = P[N].first;
		sort(P.begin(), P.end() , greater < ii > () );

		//if (mask == 28) rep(i,0,N) printf("%d %d\n",P[i].first,P[i].second );
		if(P[k-1].first> point) continue;

		rep(i,0,N) if(P[i].first == point){
			int r = 0;

			rep(j,i,N)

			if (P[j].first == point){
				if(P[j].second != N) if(!ck[P[j].second])
					r++;
			}
			else break;
			
			if (i+r+1<=k){
				//printf("mask = %d %d\n", mask,r);
				res = min ( res , cost );
			}

			break;
		}

	}

	if(res<INT_MAX) printf("%d\n", res);
	else puts("-1");

}
