#include <iostream>
#include <cmath>
#include <algorithm>
#include <string>
#include <cstdio>
#include <vector>
#include <queue>

using namespace std;

int n, k, a[30], b[30], c[30], st[30];
bool flag[30];

int main()
{
	cin >> n >> k;
	for (int i = 0; i < n; i++)
		cin >> a[i] >> b[i];
	st[0] = 1;
	for (int i = 1; i <= n; i++)
		st[i] = st[i - 1] * 2;
	int q = st[n];
	int ans1 = 1000000000;
	for (int i = 0; i < q; i++)
	{
		int kol = 0, ans = 0;
		for (int j = 0; j < n; j++)
		{
			c[j] = a[j];
			flag[j] = false;
        }
		for (int j = 0; j < n; j++)
		{
			int m = st[j];
			int d = m & i;
			if (d == m)
			{
				kol++;
				ans += b[j];
			}
			else
			{
				c[j]++;
				flag[j] = true;
			}
		}
		int h = 1;
		for (int j = 0; j < n; j++)
			if (kol	< c[j] || kol == c[j] && flag[j])
				h++;
		if (h <= k)
			ans1 = min(ans, ans1);
	}
	if (ans1 == 1000000000)
	{
		cout << -1;
		return 0;
	}
	cout << ans1 << endl;								
	return 0;
}
