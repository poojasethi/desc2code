#include <bits/stdc++.h>
using namespace std;

const int N = 16;
int n, k, p[N], e[N];


int main(){
	//freopen("input.in", "r", stdin);
	scanf("%d %d", &n, &k);
	for(int i=0; i<n; i++) scanf("%d %d", p+i, e+i);
	int res = 1e9;
	for(int i=0; i<1<<n; i++){
		int sum = 0, x = __builtin_popcount(i), ls = 0;
		for(int j=0; j<n; j++)
			if(i&1<<j){
				if(x < p[j]) ls++;
				sum += e[j];
			}
			else {
				if(x <= p[j]+1) ls++;
			}
		if(ls < k) res = min(res, sum);
	}
	if(res == 1e9) printf("-1");
	else printf("%d", res);
}
