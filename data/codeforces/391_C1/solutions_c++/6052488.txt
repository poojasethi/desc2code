#include <iostream>
#include <algorithm>
using namespace std;

int n, k;
int p[15], e[15], win[15];
int mine = 1000000000;

void fight(int idx, int total, int effort)
{
	if (idx < n)
	{
		win[idx]++;
		fight(idx + 1, total + 1, effort + e[idx]);
		win[idx]--;
		p[idx]++;
		fight(idx + 1, total, effort);
		p[idx]--;
	}
	else
	{
		int cnt = 0;
		for (int i = 0; i < n; i++)
			if (total < p[i] || (total == p[i] && win[i] == 0))
				cnt++;

		if (cnt < k)
			mine = min(mine, effort);
	}
}

int main()
{
	cin >> n >> k;
	for (int i = 0; i < n; i++)
		cin >> p[i] >> e[i];

	fight(0, 0, 0);

	if (mine == 1000000000)
		cout << -1 << "\n";
	else
		cout << mine << "\n";

	return 0;
}
