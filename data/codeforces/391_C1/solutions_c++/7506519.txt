#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <string>
#include <vector>
#include <cstring>
#define N 25

using namespace std;
const int INF = 1<<N;

int n, k, p[N], e[N], match[N], ans = INF;

void read() {
    cin>>n>>k;
    for (int i=0;i<n;i++)
        cin>>p[i]>>e[i];
}

void solve() {
    for (int i=0;i<(1<<n);i++) {
        int power = 0, point = 0, Rank = n+1;
        for (int j=0;j<n;j++)
            if (i & (1<<j)) {
                power += e[j];
                ++point;
                match[j]=0;
            } else match[j]=1;
        for (int j=0;j<n;j++) {
            if (point > p[j] + match[j] || (point == p[j] && (i & (1<<j))))
                --Rank;
            if (Rank <= k) ans = min(ans, power);
        }
    }
}

void result() {
    if (ans==INF) cout<<"-1"; else cout << ans;
}

int main() {
    read();
    solve();
    result();
}
