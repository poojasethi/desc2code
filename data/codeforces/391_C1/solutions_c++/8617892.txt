#include <bits/stdc++.h>
using namespace std;



int p[15] ;
int e[15] ;
int n ,k;
vector<pair<int,int> > test;
int main(){
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>p[i]>>e[i];
	}
	int res = 1<<30;
	for(int msk = 0 ; msk < (1<<n) ; msk++){
		test.clear() ;
		int E = 0;
		int po = 0;
		for(int i=0;i<n;i++){
			if((msk>>i)&1){
				E+=e[i];
				po++;
				test.push_back(make_pair( p[i] ,0 )) ;
			}else{
				test.push_back(make_pair(p[i]+1 , 2)) ;
			}
		}

		test.push_back(make_pair(po , 1)) ;
		sort(test.rbegin() , test.rend()) ;
		int pos ;
		for(int i=0;i<n+1;i++){
			if(test[i].second == 1)pos = i;
		}
		pos++;
		if(pos <= k) res = min(res , E);
	}
	if(res == (1<<30))puts("-1");
	else
	cout<<res<<endl;

}
