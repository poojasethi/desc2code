
#include <iostream>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;

long long mini = 100000000;
pair<int, int> person[100];

void check(string s, int k)
{

	int effort = 0, score = 0, rnk = 1;

	for (unsigned int i = 0; i < s.length(); i++) if (s[i] == '1') ++score;
	for (unsigned int i = 0; i < s.length(); i++) 
	{  
	
		if (s[i] == '0') 
		{
			
			if (score <= person[i].first + 1) ++rnk;

		}
		else 
		{ 
			
			if (score < person[i].first) ++rnk; 
			
			effort += person[i].second; 
		
		}
	
	}

	if (rnk <= k) if (effort < mini) mini = effort;

}

void pr(string s, int n, int k)
{

	if (s.length() == n)
	{
		
		check(s, k);

		return;

	}

	pr(s + '0', n, k);
	pr(s + '1', n, k);

}

int main()
{

	int n, k;
	
	cin>>n>>k;
	for (int i = 0; i < n; i++) cin>>person[i].first>>person[i].second;

	pr("", n, k);
	
	if (mini == 100000000) cout<<-1;
	else cout<<mini;

	return 0;

}
