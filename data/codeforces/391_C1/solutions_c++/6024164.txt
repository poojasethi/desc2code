#include<stdio.h>
#include<algorithm>
using namespace std;
int p[20]={0};
int p_co[20];
int e[20];
int main()
{ //  freopen("input.txt","r",stdin);
    int n,k;
    scanf("%d%d",&n,&k);
    for(int i=0;i<n;i++)    scanf("%d%d",p+i,e+i);
    for(int i=0;i<n;i++)
    {
        p_co[i]=p[i];
    }
    int all=1<<n;
    int fin=99999999;
    for(int i=0;i<all;i++)
    {
        int point=0,effort=0,rank=n+1;
        int sign[20]={0};
        for(int m=0;m<n;m++)
        {
            p_co[m]=p[m];
        }
        for(int j=0;j<n;j++)
        {
            if((i&(1<<j))==0)
            {

                p_co[j]++;
            }
            else
            {
                point++;
                effort+=e[j];
                sign[j]=1;
            }
        }
        for(int l=0;l<n;l++)
        {
            if((point>p_co[l])||(point==p_co[l]&&sign[l]==1))
                rank--;
        }
        if(rank<=k)
            {fin=min(fin,effort);
            }

    }
    if(fin<99999999)
    printf("%d",fin);
    else printf("-1");
}
