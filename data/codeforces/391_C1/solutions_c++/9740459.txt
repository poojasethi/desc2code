#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define inf 200000000

int n,k;
ll p[105],e[105];
ll DP[105][105][105];
int i;

ll solve(ll ind,ll rank,ll points) {
    if (rank > k)
        return inf;
    if (points > i)
        return inf;
    if (ind == n && points != i)
        return inf;
    if (ind == n && points == i )
        return 0;
    if (DP[ind][rank][points] != -1)
        return DP[ind][rank][points];
    if (p[ind] < i - 1) {
        DP[ind][rank][points] = min(e[ind] + solve(ind+1,rank,points+1),solve(ind+1,rank,points));
    }
    else if (p[ind] == i-1) {
        DP[ind][rank][points] = min(e[ind] + solve(ind+1,rank,points+1), solve(ind+1,rank+1,points));
    }
    else if (p[ind] == i)
        DP[ind][rank][points] = min(e[ind] + solve(ind+1,rank,points+1),solve(ind+1,rank+1,points));
    else
        DP[ind][rank][points] = min(e[ind] + solve(ind+1,rank+1,points+1),solve(ind+1,rank+1,points));
    return DP[ind][rank][points];
}
int main() {
    cin >> n >> k;
    ll res = inf;
    for (i = 0; i < n; i++)
        cin >> p[i] >> e[i];
    for (i = 0; i <= n; i++) {
        memset(DP,-1,sizeof DP);
        ll mm = solve(0,1,0);
        if (mm < res)
            res = mm;
    }
    if (res == inf)
        cout << -1;
    else
        cout << res;
    return 0;
}
