#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
typedef vector<ii> vii;
typedef vector<int> vi;

int conversion(string p){ int o; o=atoi(p.c_str()); return o; }
string toString(int h){ stringstream ss; ss<<h; return ss.str(); }
long long gcd(long long a,long long b) {return (b==0 ? a : gcd(b,a%b));}
long long lcm(long long a,long long b) {return (a*(b/gcd(a,b))); }
long long toi(string p){long long x;istringstream in(p); in>>x; return x;}

bool v[16];

int main()
{

    #ifndef ONLINE_JUDGE
    freopen("D:/input.txt","r",stdin);
    //freopen("D:/output.txt","w",stdout);
    #endif

    int n,k,x,y,res=INT_MAX,punt=0,ener=0,nm=0;
    cin>>n>>k;
    vii M;
    memset(v,false,sizeof v);
    for(int i=0;i<n;i++)
    {
        cin>>x>>y;
        M.push_back(make_pair(x,y));
    }
    if(k==n+1)
    {
        cout<<0<<endl;
        return 0;
    }
    for(int mask = 0 ; mask<(1<<n); mask++)
    {
        for(int i=0;i<n;i++)
        {
            if((mask&(1<<i))>0)
            {
                punt++;
                ener+=M[i].second;
                v[i]=true;
            }
        }
        for(int i=0;i<n;i++)
        {
            if(!v[i]&&punt>M[i].first+1)
                nm++;
            else if(v[i]&&punt>=M[i].first)
                nm++;
        }

        if((n-nm)+1<=k)
            res=min(ener,res);

        punt=ener=nm=0;
        memset(v,false,sizeof v);
    }
    if(res!=INT_MAX) cout<<res<<endl;
    else cout<<-1<<endl;
}
