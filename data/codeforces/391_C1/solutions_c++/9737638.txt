#include <bits/stdc++.h>

using namespace std;

#define ll long long
#define inf 200000000

int n,k;
int p[105],e[105];

int rank(int points,int state) {
    int res = 1;
    for (int i = 0; i < n; i++) {
        if (p[i] < points)
            continue;
        if (p[i] > points) {
            res++;
            continue;
        }
        if ((1 << i) & state)
            continue;
        res++;
    }
    return res;
}

int main() {
    cin >> n >> k;
    for (int i = 0; i < n; i++)
        cin >> p[i] >> e[i];
    ll res = inf;
    for (int i = 0; i < (1<<n); i++) {
        ll points = 0;
        ll effort = 0;
        for (int j = 0; j < n; j++) {
            if ((1 <<j) & i) {
                points++;
                effort+= e[j];
            } else {
                p[j]++;
            }
        }
        if (rank(points,i) <= k)
            if (effort < res)
                res = effort;
        for (int j = 0; j < n; j++) {
            if ((1 <<j) & i) {
            } else {
                p[j]--;
            }
        }
    }
    if (res == inf)
        cout << -1;
    else
        cout << res;
    return 0;
}
