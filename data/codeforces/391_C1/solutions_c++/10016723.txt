#include <bits/stdc++.h>
#define pb push_back
#define mp make_pair
using namespace std;
typedef long long ll;
typedef pair<int,int> pii;
const int MOD=1e9+7;
const int INF=2e9;
const int maxn=20;
pii a[maxn],help[maxn];
int bin[maxn];
bool vis[maxn];
int ans=INF;
void pr()
{
    for(int i=0;i<maxn;i++)
    {
        help[i]=a[i];
        bin[i]=0;
        vis[i]=false;
    }
}
int main()
{
    int n,m; cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        cin>>a[i].first>>a[i].second;
        a[i].first++;
    }
    for(int i=0;i<(1<<n);i++)
    {
        pr();
        int k=0;
        int l=i;
        while(l)
        {
            bin[k]=l%2;
            l=l/2;
            k++;
        }
        int need=0,point=0;
        for(int j=0;j<n;j++)
        {
            if(bin[j])
            {
                need+=help[j].second;
                vis[j]=true;
                help[j].first--;
                point++;
            }
        }
        int place=1;
        for(int j=0;j<n;j++)
        {
            if(help[j].first>point)
                place++;
            if(help[j].first==point && !vis[j])
                place++;
        }
        if(place<=m)
            ans=min(ans,need);
    }
    if(ans==INF)
        cout<<-1;
    else
        cout<<ans;
    return 0;
}
