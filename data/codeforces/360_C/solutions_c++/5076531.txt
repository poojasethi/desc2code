#include <cstdio>
#include <cstring>
#include <algorithm>
#define N 2010
typedef long long ll;
using namespace std;

char s[2010];
int dp[N][N][2];
const int mod=1000000007;

int main(){
	int n,k;
	scanf("%d %d%s",&n,&k,&s[1]);
	memset(dp,0,sizeof(dp));
	dp[0][0][0]=1;
	for(int i=1;i<=n;i++){
		for(int j=0;j<=k;j++){
			dp[i][j][1]=(dp[i-1][j][0]+dp[i-1][j][1])%mod;
			dp[i][j][0]=((ll)dp[i-1][j][0]+dp[i-1][j][1])*(s[i]-'a')%mod;
			for(int k=i-1;k>=0;k--){
				if((i-k)*(n-i+1)>j) break;
				dp[i][j][0]=((ll)dp[i][j][0]+(ll)dp[k][j-(i-k)*(n-i+1)][0]*('z'-s[i]))%mod;
			}
		}
	}
	printf("%d\n",(dp[n][k][0]+dp[n][k][1])%mod);
}
