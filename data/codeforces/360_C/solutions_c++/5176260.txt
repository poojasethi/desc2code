#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#define N 2005
#define mod 1000000007
using namespace std;
typedef long long LL;
LL a,b,c,d,e,f,g,h,i,j,k;
char w[N];
LL F[N][N];
LL tmp[N];
LL H[N];

int main()
{
    cin>>a>>b;
    scanf("%s",w+1);
    F[0][0]=1; H[0]=1;
    for(i=1;i<=a;i++)
        for(j=0;j<=b;j++)
        {
            for(k=i-1;k>=0 && (i-k)*(a-i+1)<=j;k--)
            {
                F[i][j]+=F[k][j-(i-k)*(a-i+1)]*('z'-w[i]);
                F[i][j]%=mod;
            }
            F[i][j]+=H[j]*(w[i]-'a'); F[i][j]%=mod;
            H[j]+=F[i][j]; H[j]%=mod;
        }
    cout<< H[b] <<endl;
    return 0;
}
