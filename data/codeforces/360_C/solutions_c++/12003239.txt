#include <cstdio>
#include <iostream>
using namespace std;
const int mo = 1e9 + 7;
int n, k;
long long dp[2005][2005], s[2005][2005];
char a[2005];

int main(){
    scanf("%d%d", &n, &k);
    scanf("%s", a + 1);
    dp[0][0] = s[0][0] = 1;
    for (int i = 1; i <= n; ++ i){
        for (int j = 0; j <= k; ++ j){
            int l = a[i] - 'a';
            int r = 'z' - a[i];
            dp[i][j] = s[i-1][j] * l % mo;
            for (int pre = 1; i >= pre && j >= pre * (n - i + 1); pre ++){
                dp[i][j] = (dp[i][j] + dp[i-pre][j-pre*(n-i+1)] * r) % mo;
            }
            s[i][j] = (s[i-1][j] + dp[i][j]) % mo;
        }
    }
    cout << s[n][k] << endl;
    return 0;
}
