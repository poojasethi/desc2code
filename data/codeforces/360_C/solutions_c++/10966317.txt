#include<iostream>
#include<string.h>
#include<stdio.h>
#define MOD 1000000007
using namespace std;
long long n,k;
char s[2100];
long long f[2100][2100];
long long sum[2100];
int main()
{
	long long i,q,j;
	long long t;
	scanf("%d%d",&n,&k);
	scanf("%s",s);
	f[n][0]=1;
	for(i=n-1;i>=0;i--)
	{
		f[i][0]=1;
		for(j=i;j<=n-1;j++)
		{
			t=(j-i+1)*(n-j);
			for(q=k;q>=t;q--)
				f[i][q]=(f[i][q]+(f[j+1][q-t]*('z'-s[j]))%MOD)%MOD;
		}	
		for(q=0;q<=k;q++)
		{
			sum[q]=(sum[q]+(f[i+1][q]*(s[i]-'a'))%MOD)%MOD;
			f[i][q]=(f[i][q]+sum[q])%MOD;
		}
	}
	cout<<f[0][k]<<endl;
  return 0;
}
