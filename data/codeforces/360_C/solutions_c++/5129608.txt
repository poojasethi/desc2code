#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#define N 2100
using namespace std;
const int mo=1000000007;
int n,v,t,f[N][N],k,x,y,i,j,g[N][N];
char c[N];
int main(){
	scanf("%d%d",&n,&k);
	scanf("%s",c+1);
	for(i=0;i<=n;++i)f[i][0]=1;
	for(i=n;i>=1;--i){
		v=n-i+1;
		t='z'-c[i];
		for(j=0;j<=k;++j){
			f[i][j]=(f[i][j]+g[i][j])%mo;
			g[i-1][j]=g[i][j];
			if(f[i][j]){
				for(x=i-1,y=j+v;x>=0 && y<=k;x--,y+=v)
					f[x][y]=(f[x][y]+(long long)f[i][j]*t)%mo;
				g[i-1][j]=(g[i-1][j]+(long long)f[i][j]*(26-t-1))%mo;
			}
		}
	}
	printf("%d\n",(f[0][k]+g[0][k])%mo);
}
