#include <cstdio>
using namespace std;
typedef long long LL;
const LL INF=1000000007LL;
const int NMax=2010;
int N,K;
char A[NMax];
LL dp[NMax][NMax],sum[NMax];
LL upd(LL a,LL b) {
	return (a+b)%INF;
}
LL mul(LL a,LL b)  {
	return (a*b)%INF;
}
int main()
{
	scanf("%d%d%s",&N,&K,A);
	for(int i=N;i>=1;i--) A[i]=A[i-1];
	dp[0][0]=1;
	sum[0]=dp[0][0];
	for(int i=1;i<=N;i++) {
		for(int j=0;j<=K;j++) {
			dp[i][j]=upd(dp[i][j],mul(sum[j],A[i]-'a'));
			for(int k=0;k<i&&(1+k)*(N-i+1)<=j;k++)
				dp[i][j]=upd(dp[i][j],mul(dp[i-k-1][j-(1+k)*(N-i+1)],'z'-A[i]));
			sum[j]=upd(sum[j],dp[i][j]);
		}
	}
	LL ret=0;
	for(int i=0;i<=N;i++) ret=upd(ret,dp[i][K]);
	printf("%I64d\n",ret);
	getchar();getchar();
	return 0;
}
