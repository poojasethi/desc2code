#include <iostream>
#include <cstring>
#include <cstdio>

#define MAXN 2005
#define MOD 1000000007

using namespace std;

int n, k, c[MAXN][MAXN], d[MAXN][MAXN];
char s[MAXN];

main()
{
  scanf("%d%d\n", &n, &k);
  gets(s + 1);
 
  c[0][0] = d[0][0] = 1;
  for (int i = 1; i <= n; i++)
  {
    for (int j = 0; j <= k; j++) 
      c[i][j] = 1LL * d[i - 1][j] * (s[i] - 'a') % MOD;
    int p = n - i + 1;
    for (int j = 1; j <= i; j++)
      for (int u = 0; u + p * j <= k; u++) 
        c[i][u + p * j] = (c[i][u + p * j] + 1LL * c[i - j][u] * ('z' - s[i]) % MOD) % MOD;
    for (int j = 0; j <= k; j++) d[i][j] = (d[i - 1][j] + c[i][j]) % MOD;
  }
  
  cout << d[n][k];
//  system("pause");
}
