#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=0;i<n;++i)
typedef long long LL;
const int N=2005,P=1000000007;
int n,m,f[N][N][2]; char s[N]; LL z;
void upd(int &x,int y){x+=y; if (x>=P) x-=P;}
int main()
{
	scanf("%d%d%s",&n,&m,s),f[0][0][1]=1;
	rep(i,n+1) rep(j,m+1) rep(t,2){
		if (t && i) for (int k=0;k<i && j+(n-i+1)*(k+1)<=m;++k)
			upd(f[i][j+(n-i+1)*(k+1)][1],f[i-k-1][j][1]*(122LL-s[i-1])%P);
		if (z=f[i][j][t])
			upd(f[i+1][j][1],z*(s[i]-97)%P),upd(f[i+1][j][0],z);
	}
	printf("%d\n",(f[n][m][0]+f[n][m][1])%P);
	return 0;
}
