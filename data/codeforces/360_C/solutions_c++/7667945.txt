#include <cstdio>
#include <cstdlib>
#define P 1000000007
using namespace std;

int n,K;
long long f[2001][2001],pre[2001][2001];
char s[2001];

int main()
{
	scanf("%d%d",&n,&K);
	scanf("%s",s+1);
	f[0][0]=pre[0][0]=1;
	for (int i=1;i<=n;i++)
	{
		for (int j=0;j<=K;j++)
		{
			f[i][j]=(f[i][j]+pre[i-1][j]*(s[i]-'a'))%P;
			for (int k=i;(k)&&((i-k+1)*(n-i+1)<=j);k--) f[i][j]=(f[i][j]+f[k-1][j-(i-k+1)*(n-i+1)]*('z'-s[i]))%P;
		}
		for (int j=0;j<=K;j++) pre[i][j]=(pre[i-1][j]+f[i][j])%P;
	}
	printf("%I64d\n",pre[n][K]);
}
