#include <cstdio>

#define MOD 1000000007

typedef long long int lli;

lli dp[2005][2005],sum[2005][2005];
char s[2005];

int main()
{
    int n,k,i,j,l,t;
    scanf("%d%d%s",&n,&k,s+1);
    dp[0][0] = sum[0][0] = 1;
    for(i=1;i<=n;i++)
        for(j=0;j<=k;j++) {
            dp[i][j] = (sum[i-1][j]*(s[i]-'a'))%MOD;
            for(l=0; l<i&&(t=(l+1)*(n-i+1))<=j;l++) dp[i][j] = (dp[i][j] + (dp[i-l-1][j-t]*('z'-s[i]))%MOD)%MOD;
            sum[i][j] = (sum[i-1][j] + dp[i][j])%MOD;
        }
    printf("%lld\n",sum[n][k]);
    return 0;
}
