#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#include<algorithm>
#define maxn 2010
#define mod 1000000007
using namespace std;
typedef long long LL;
int n,m;
char str[maxn];
int Dp()
{
	static LL dp[maxn][maxn];
	dp[0][0]=1;
	for(int i=1;i<=n;++i)
		for(int j=0;j<=m;++j)
		{
			dp[i][j]=dp[i-1][j]*(str[i]-'a'+1)%mod;
			for(int k=0;i-k>=2&&(n-i+1)*(k+1)<=j;++k)
			{
				int t=j-(k+1)*(n-i+1);
				dp[i][j]=(dp[i][j]+(dp[i-k-1][t]-dp[i-k-2][t]+mod)*('z'-str[i]))%mod;
			}
			if((n-i+1)*i==j)
				dp[i][j]=(dp[i][j]+'z'-str[i])%mod;
		}
	return dp[n][m];
}
int main()
{
	scanf("%d %d %s",&n,&m,str+1);
	printf("%d\n",Dp());
	return 0;
}
