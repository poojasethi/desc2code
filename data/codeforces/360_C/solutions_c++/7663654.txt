#include<cstdio>
#define P 1000000007
#define N 2006
int i,j,u,t,n,k,f[N][N],g[N][N];
char s[N];
int main(){
    while(~scanf("%d%d%s",&n,&k,s+1)){f[0][0]=g[0][0]=1;
        for(i=1;i<=n;++i)for(j=0;j<=k;++j){
            f[i][j]=1ll*g[i-1][j]*(s[i]-'a')%P;
            for(u=0;u<i&&(t=(u+1)*(n-i+1)) <=j;++u)
                (f[i][j]+=1ll*f[i-u-1][j-t]*('z'-s[i]) %P)%=P;
            g[i][j]=(g[i-1][j]+f[i][j])%P;
        }
        printf("%d\n",g[n][k]);
    }
}
