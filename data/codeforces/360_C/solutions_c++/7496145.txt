#include <bits/stdc++.h>
using namespace std;
#define LL long long
#define MOD 1000000007
LL dp[2010][2010];
LL S[2010];
int main()
{
	string str;
	int N,K;
	cin>>N>>K;
	cin>>str;
	dp[N][0] = 1;
	for(int i = N-1;i>=0;i--)
	{
		for(int j = i;j<N;j++)
		{
			int Prefix = j - i;
			int Extra = N - j;
			for(int k = Extra*(Prefix+1);k<=K;k++)
			{
				dp[i][k] += dp[j+1][k - (Prefix+1)*Extra] * ('z' - str[j]);
				dp[i][k]%=MOD;
			}
		}
		for(int j = 0;j<=K;j++)
		{
			dp[i][j] += dp[i+1][j]*(str[i] - 'a');
			dp[i][j] += S[j];
			S[j]+=dp[i+1][j]*(str[i] - 'a');
			if(j==0)
				dp[i][j]+=1;
			dp[i][j]%=MOD;
			S[j]%=MOD;
		}
	}
	cout<<dp[0][K]<<endl;
	return 0;
}
