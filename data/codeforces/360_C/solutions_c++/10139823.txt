#include <cstdio>
#include <iostream>
#define N 2002
#define P 1000000007
using namespace std;
int n,k;
char s[N];
long long num[N],f[N][N],sum[N];
int main(){
	scanf("%d%d%s",&n,&k,s+1);
	num[n+1]=1;
	for(int i=n;i;--i)num[i]=num[i+1]*(s[i]-'a'+1)%P;
	if(k==0){printf("%I64d",num[1]);return 0;}
	long long ans=0;
	f[0][0]=sum[0]=1;
	for(int i=1;i<=n;++i){
		for(int j=0;j<=k;++j)
			for(int l=0;l<i && (l+1)*(n-i+1)<=j;++l)
				f[i][j]=(f[i][j]+f[i-l-1][j-(l+1)*(n-i+1)]*('z'-s[i]))%P;
		ans=(ans+f[i][k]*num[i+1])%P;
		for(int j=0;j<=k;++j){
			f[i][j]=(f[i][j]+sum[j]*(s[i]-'a'))%P;
			sum[j]=(sum[j]+f[i][j])%P;
		}
	}
	printf("%I64d",ans);
}
