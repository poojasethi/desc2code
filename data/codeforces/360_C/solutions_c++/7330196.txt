#include<cstdio>
#include<algorithm>
#include<cstring>
#include<cmath>
#include <iostream>
#define LL long long
#define N 2020
#define MOD 1000000007
using namespace std;
LL f[N][N],sum[N][N];
int n,k;
char s[N];
int main() {
	scanf("%d%d",&n,&k);
	scanf(" %s",s+1);
	memset(f,0,sizeof(f));
	memset(sum,0,sizeof(sum));
	f[0][0]=1;
	sum[0][0]=1;
	for(int i=1;i<=n;++i)
	{
		for(int j=0;j<=k;++j)
		{
			f[i][j]=(sum[i-1][j]*(s[i]-'a'))%MOD;
			for(int x=i-1;x>=0&&(i-x)*(n-i+1)<=j;--x)
			{
				f[i][j]=(f[i][j]+f[x][j-(i-x)*(n-i+1)]*('z'-s[i]))%MOD;
			}
			sum[i][j]=(f[i][j]+sum[i-1][j])%MOD;
		}
	}
	printf("%I64d\n",sum[n][k]);
	return 0;
}
