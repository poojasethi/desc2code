#include <cstdio>
#include <cstring>
#include <algorithm>
#define maxn 2009
#define MOD 1000000007
using namespace std;
__int64 dp[maxn][maxn],sum[maxn];
char s[maxn];
int n,m;
int main()
{
	scanf("%d%d",&n,&m);
	scanf("%s",s+1);
	dp[0][0]=1;sum[0]++;
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<=m;j++)
		{
			dp[i][j]=sum[j]*(s[i]-'a')%MOD;
			for(int k=0;k<i&&(k+1)*(n-i+1)<=j;k++)
			{
				dp[i][j]=(dp[i][j]+dp[i-k-1][j-(k+1)*(n-i+1)]*('z'-s[i]))%MOD;
			}
			sum[j]+=dp[i][j];
			sum[j]%=MOD;
		}
	}
	__int64 ans=0;
	for(int i=0;i<=n;i++)
	{
		ans+=dp[i][m];
		ans%=MOD;
	}
	printf("%I64d\n",ans);
	//system("pause");
	return 0;
}
