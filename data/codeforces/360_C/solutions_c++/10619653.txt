#include <bits/stdc++.h>

#define mp make_pair
#define st first
#define nd second
#define pb push_back

using namespace std;

typedef long long ll;
typedef pair< ll,ll > pll;
typedef pair< int,int > pii;

const int inf = 1000000000, N = 2000, mod = 1000000007;

int n, m;
ll dp[N+5][N+5], sum[N+5][N+5];
char s[N+5];

int main()
{
	scanf("%d %d %s", &n, &m, s+1);

	dp[0][0] = 1;
	sum[0][0] = 1;

	for (int i = 1; i <= n; i++) {
		for (int j = 0; j <= m; j++) {
			for (int k = i; k >= 1; k--) {
				if (j - (n-i+1) * (i-k+1) < 0)
					break;

				dp[i][j] += ('z' - s[i]) * dp[k-1][j - (n-i+1) * (i-k+1)];
			}

			dp[i][j] += (s[i] - 'a') * sum[i-1][j];

			dp[i][j] %= mod;

			sum[i][j] = (sum[i-1][j] + dp[i][j]) % mod;
		}
	}
	
	cout << sum[n][m] << endl;

	return 0;
}
