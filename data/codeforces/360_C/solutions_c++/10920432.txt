#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
typedef long long LL;
const int maxn=2005,M=1000000007;
int n,k,ans;
char s[maxn];
int f[maxn][maxn],sf[maxn];
inline void up(int &x,const int &y)
{x+=y;if(x>=M)x-=M;}

int main()
{
	scanf("%d%d\n",&n,&k);
	if(k==0)ans=1;
	scanf("%s",s+1);
	f[0][0]=sf[0]=1;
	for(int i=1;i<=n;i++)
	{
		for(int j=0;j<=k;j++)
		{
			for(int p=0;p<i&&(p+1)*(n-i+1)<=j;p++)
				up(f[i][j],f[i-p-1][j-(p+1)*(n-i+1)]);
			f[i][j]=((LL)sf[j]*(s[i]-'a')+(LL)f[i][j]*('z'-s[i]))%M;
			up(sf[j],f[i][j]);
		}
		up(ans,f[i][k]);
	}
	printf("%d\n",ans);
	return 0;
}
