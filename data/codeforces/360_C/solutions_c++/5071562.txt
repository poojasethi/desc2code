#include <iostream>
#include <string>

using namespace std;

const int limite=3000;

typedef long long int ll;

const ll modulo=1e9+7;

int n,k;
string s;

ll maneras[limite][limite];
ll manerasmenorigual[limite][limite];

int main()
{
  cin>>n>>k>>s;
  maneras[n+1][0]=manerasmenorigual[n+1][0]=1;
  for (int i=n;i>=1;i--) {
    int val=s[i-1]-'a';
    for (int j=0;j<=k;j++)
      maneras[i][j]=manerasmenorigual[i][j]=(val*maneras[i+1][j]+manerasmenorigual[i+1][j])%modulo;
    for (int ii=i;ii<=n;ii++) {
      val=s[ii-1]-'a';
      int count=(ii-i+1)*(n-ii+1);
      for (int j=count;j<=k;j++) {
	maneras[i][j]+=(25-val)*maneras[ii+1][j-count];
	maneras[i][j]%=modulo;
      }
    }
  }
  cout<<maneras[1][k]<<endl;
}
