#include <cstdio>
#define ll long long
#define zhl 1000000007
ll dp[2010][2010],g[2010];
int n,K;
char s[2010];
int main()
{
	scanf("%d%d%s",&n,&K,s+1);
	dp[0][0] = g[0] = 1;
	for (int i = 1;i <= n;i++)
		 for (int j = 0;j <= K;j++)
		{
			dp[i][j] = g[j]*(s[i]-'a')%zhl+(i==n)*g[j];
			if (dp[i][j] > zhl) dp[i][j] -= zhl;
			for (int k = 1;k*(n-i+1)<=j&&k<=i;k++)
			{ 
				dp[i][j] = dp[i][j]+dp[i-k][j-k*(n-i+1)]*('z'-s[i])%zhl;
				if (dp[i][j] > zhl) dp[i][j] -= zhl;
			}
			g[j] += dp[i][j];
			if (g[j] > zhl) g[j] -= zhl;
		}
	printf("%I64d",dp[n][K]);
	return 0;
}
