#include <cstdio>
#include <algorithm>
#include <cstring>
typedef long long ll;
const int N = 2000 + 5;
const int mod = 1000000007;

char s[N];
ll d[N][N], sum[N];
int n, K;
int main() {
	scanf("%d%d", &n, &K);
	scanf("%s", s + 1);
	
	d[0][0] = sum[0] = 1;
	for (int i = 1; i <= n; ++i) {
		for (int j = 0; j <= K; ++j) {
			for (int k = i - 1; k >= 0; --k) {
				if (j - (i - k) * (n - i + 1) < 0)
					break;
				d[i][j] = (d[i][j] + d[k][j - (i - k) * (n - i + 1)]) % mod;
			}
			d[i][j] = (d[i][j] * ('z' - s[i])) % mod;
		}
		for (int j = 0; j <= K; ++j) {
			d[i][j] = (d[i][j] + sum[j] * (s[i] - 'a')) % mod;
			sum[j] = (sum[j] + d[i][j]) % mod;
		}
	}
	printf("%d\n", sum[K]);
	return 0;
}
