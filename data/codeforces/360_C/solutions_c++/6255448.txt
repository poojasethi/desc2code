#include<stdio.h>
#include<string.h>
    
typedef __int64 LL;
const int N=2002,mod=1e9+7;
LL dp[N][N],sum[N][N],ans;
char s[N];
int main(){
    int n,m,i,j,k,t,p;
    scanf("%d%d%s",&n,&m,s+1);
    for(sum[0][0]=dp[0][0]=i=1;i<=n;i++){
        for(j=0;j<=m;j++){
            t=j/(n-i+1)-1;
            for(k=0;k<=t&&k<i;k++) dp[i][j]=(dp[i][j]+dp[i-k-1][j-(1+k)*(n+1-i)]*('z'-s[i]))%mod;
            dp[i][j]=(dp[i][j]+sum[i-1][j]*(s[i]-'a'))%mod;
            sum[i][j]=(sum[i-1][j]+dp[i][j])%mod;
        }
    }
    for(k=0;k<n;k++) dp[n][m]=(dp[n][m]+dp[n-1-k][m])%mod;
    printf("%I64d\n",dp[n][m]);
    return 0;
}
