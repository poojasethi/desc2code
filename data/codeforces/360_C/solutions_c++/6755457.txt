#include <iostream>
using namespace std;
#define f(i,a,b) for(int i=a;i<=b;i++)
#define N 2024
#define M 1000000007
long long d[N][N], col[N];
int main() {
    int n, k; string s;
    cin >> n >> k >> s;
    d[0][0] = col[0] = 1;
    f(i, 1, n) f(j, 0, k) {
        d[i][j] = col[j]*(s[i-1]-'a' + (i==n));
        f(e, 0, i-1) if(j < (e+1)*(n-i+1)) break;
        else d[i][j] = (d[i][j] + d[i-e-1][j-(e+1)*(n-i+1)]*('z'-s[i-1])) % M;
        col[j] += ( d[i][j] %= M );
    }
    cout << d[n][k];
}
