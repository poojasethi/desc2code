#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;
const int MAXN = 2030,Mod = 1000000007;
int n,k;
char s[MAXN];
long long f[MAXN][MAXN],ret = 0,sum[MAXN];

int main(){
	scanf("%d%d",&n,&k);
	scanf("%s",s + 1);
	f[0][0] = 1; sum[0] = 1;
	if(k == 0) ret += f[0][0];
	for(int i = 1;i <= n;++ i){
		for(int j = 0;j <= k;++ j){
			f[i][j] = sum[j] * (s[i] - 'a') % Mod;
			for(int pre = 0;pre < i && (pre + 1) * (n - i + 1) <= j;++ pre)
				f[i][j] = (f[i][j] + f[i - pre - 1][j - (pre + 1) * (n - i + 1)] * ('z' - s[i]) % Mod) % Mod;
			sum[j] = (sum[j] + f[i][j]) % Mod;
		}
		ret = (ret + f[i][k]) % Mod;
	}
	printf("%d\n",int(ret));
	return 0;
}
