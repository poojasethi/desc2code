#include <cstdio>
#include <iostream>
#include <cmath>
#include <algorithm>
#include <cstring>

using namespace std;

typedef long long LL;

const LL mod=1e9+7;
LL f[2009][2009],g[2009][2009];
char s[2009];
int n,k;
int dq;
int main()
{
	memset(f,0,sizeof(f));
	memset(g,0,sizeof(g));
	scanf("%d%d",&n,&k);
	f[0][0]=1;g[0][0]=1;
	scanf("%s",s);
	for (int i=1;i<=n;++i)
		for (int j=0;j<=k;++j)
		{
			f[i][j]=(g[i-1][j]*(s[i-1]-'a'))%mod;
			for (int k=0;k<i&&((k+1)*(n-i+1)<=j);++k)
			{
				dq=(k+1)*(n-i+1);
				f[i][j]=(f[i][j]+(f[i-k-1][j-dq]*('z'-s[i-1]))%mod)%mod;
			}
			g[i][j]=(g[i-1][j]+f[i][j])%mod;
		}
	printf("%I64d\n",g[n][k]);
	return 0;
}
