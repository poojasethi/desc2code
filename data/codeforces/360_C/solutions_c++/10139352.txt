#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long LL;
const LL mod=1000000007;
int n,m,z,x,y;
LL s1[2005],f[2005][2005];
char c[2005];
int main(){
	//freopen("cf360C.in","r",stdin);freopen("cf360C.out","w",stdout);
	scanf("%d%d\n",&n,&m);
	scanf("%s",c);
	f[0][0]=1;s1[0]=1;
	for (int i=1;i<=n;++i)
	for (int j=0;j<=m;++j){
		f[i][j]=(s1[j]*(c[i-1]-'a'))%mod;
		for (int k=1;k<=j/(n-i+1) && k<=i;++k) 
		f[i][j]=(f[i][j]+f[i-k][j-k*(n-i+1)]*('z'-c[i-1])%mod)%mod;
		s1[j]=(s1[j]+f[i][j])%mod;
	}
	printf("%I64d",s1[m]);
	//fclose(stdin);fclose(stdout);
}
