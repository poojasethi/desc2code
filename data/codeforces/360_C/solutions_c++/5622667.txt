#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include <algorithm>
#include<iostream>

#define M 1000000007
using namespace std;
typedef __int64 LL;

int n,k;
int dp1[2010][2010];
int dp2[2010][2010];
char s[2010];

int main()
{
    int i,j,h;
    cin>>n>>k;
    scanf("%s",s+1);
    dp1[0][0]=1;
    dp2[0][0]=1;
    for(i=1;i<=n;i++)
        for(j=0;j<=k;j++)
    {
        int t=n-i+1;
        dp1[i][j]=(1ll*dp2[i-1][j]*(s[i]-'a'))%M;
        for(h=0;h<i&&(h+1)*t<=j;h++)
            (dp1[i][j]+=1ll*dp1[i-h-1][j-(h+1)*t]*('z'-s[i])%M)%=M;
        dp2[i][j]=(1ll*dp2[i-1][j]+dp1[i][j])%M;
    }

    cout<<dp2[n][k];
    return 0;
}
