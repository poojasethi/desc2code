#include<cstdio>
#include<vector>

using namespace std;

#define ll long long
#define MOD 1000000007ll

char str[2222];
int n,K;
ll dp[2222][2222];
ll sum[2222][2222];

int main() {
    scanf("%d%d%s",&n,&K,str+1);

    dp[0][0]=1,sum[0][0]=1;
    for(int i=1;i<=n;++i) {
        for(int j=0;j<=K;++j) {
            dp[i][j] = (dp[i][j] + sum[i-1][j] * (str[i]-'a')) % MOD;
            for(int k=i-1;k>=0;--k) {
                int prv=j-(i-k)*(n-i+1);
                if(prv<0) break;
                dp[i][j] = (dp[i][j] + dp[k][prv] * ('z' - str[i])) % MOD;
            } 
            sum[i][j]=(sum[i-1][j]+dp[i][j]) % MOD;
        }
    } printf("%I64d",sum[n][K]);
}
