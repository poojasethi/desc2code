#include<cstdio>
using namespace std;
const int NMAX = 2005;
const int MOD = 1000000007;
int N,K,i,j,k,SOL;
long long DP[NMAX][NMAX],Sum[NMAX][NMAX];
char S[NMAX];
int main()
{
    //freopen("test.in","r",stdin);
    //freopen("test.out","w",stdout);
    scanf("%d%d\n%s",&N,&K,S+1);
    DP[0][0]=Sum[0][0]=1;
    for(i=1;i<=N;i++)
        for(j=0;j<=K;j++)
        {
            DP[i][j]=(DP[i][j]+Sum[i-1][j]*(S[i]-'a'))%MOD;
            for(k=0;k<i && j-(k+1)*(N-i+1)>=0;k++)
                DP[i][j]=(DP[i][j]+DP[i-k-1][j-(k+1)*(N-i+1)]*('z'-S[i]))%MOD;
            Sum[i][j]=(Sum[i-1][j]+DP[i][j])%MOD;
        }
    for(i=0;i<=N;i++) SOL=(SOL+DP[i][K])%MOD;
    printf("%d\n",SOL);
    return 0;
}
