MAX=10**6+3
res=[0]*MAX
beacons=[-1]*MAX


n=int(raw_input())
for i in xrange(n):
    a,b=map(int, raw_input().split())
    beacons[a+1]=b
    
for i in reversed(xrange(MAX)):
    if beacons[i]==-1:
        res[i-1]=max(res[i-1], res[i])
    else:
        index=max(0, i-beacons[i]-1)
        res[index]=max(res[index], res[i]+1)
        
print (n-res[0])
