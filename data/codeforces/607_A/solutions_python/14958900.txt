from sys import stdin
def main():
    n = int(stdin.readline())
    a = [map(int, stdin.readline().split()) for _ in xrange(n)]
    dp = [0] * 1000010
    b = [0] * 1000010
    for p, x in a:
        b[p+1] = x
    for i in xrange(1, 1000010):
        dp[i] = dp[i-1]
        if b[i]:
            j = i - b[i] - 1
            if j < 0:
                j = 0
            dp[i] = dp[j] + 1
    print n - max(dp)
main()
