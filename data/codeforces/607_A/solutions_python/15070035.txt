n = input()
ar = [0]*(1000010)
for i in xrange(n):
    a,b = map(int,raw_input().split())
    ar[a+1] = b
dp = [0]*(1000010)       # stores no. of objects not destroyed till i
for i in xrange(1000010):
    if ar[i] == 0:
        dp[i] = dp[i-1]
    else:
        if i - ar[i] - 1 >= 0:
            dp[i] = dp[i-ar[i]-1] + 1
        else:
            dp[i] = 1
print n - max(dp)
