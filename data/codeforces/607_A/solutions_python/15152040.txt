#!/usr/bin/env python
import sys

bc = [0] * 1000005
dp = [0] * 1000005
sm = [0] * 1000005

n = int(sys.stdin.readline())
rz = 0
for i in xrange(n):
    a, b = map(int, sys.stdin.readline().split())
    bc[a + 1] = min(b, a)
    sm[a + 1] = 1
    rz = max(rz, a)

mn = n
for i in xrange(1, rz + 2):
    sm[i] += sm[i - 1]
    dp[i] = sm[i - 1] - sm[i - bc[i] - 1] + dp[i - bc[i] - 1]
    mn = min(mn, dp[i] + n - sm[i])

print mn
