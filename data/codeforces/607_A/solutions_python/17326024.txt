
n = int(raw_input())
pos = 1000001*[0]
Range = 1000001*[0]
dp = 1000001*[0]

last = -1
for i in xrange(n):
    x,y = map(int,raw_input().split())
    pos[x] = 1
    Range[x] = y
    last = max(last,x)

ans = 0
dp[0] = pos[0]

for i in range(1,1000001):
    if pos[i] :
        dp[i] = 1
        if i-Range[i]-1>=0 :
            dp[i] += dp[i-Range[i]-1]
    else :
        dp[i] = dp[i-1]

    ans = max(ans, dp[i])

print n- ans
