#include <iostream>
#include <cstring>

using namespace std;

int N,a,P[1000005],dp[1000005];
void se()
{
  if(P[0]!=0)dp[0]=1;
  int ma=0;
  for(int i=1;i<1000005;i++)
    {
      if(P[i]==0)dp[i]=dp[i-1];
      else
	{
	  if(P[i]>=i)dp[i]=1;
	  else dp[i]=dp[i-P[i]-1]+1;
	}
      ma=max(dp[i],ma);
    }
  cout << N-ma << '\n';
}

int main()
{
  cin >> N;
  for(int i=0;i<N;i++)
    {cin >> a;cin >> P[a];}
  se();return 0;
}
