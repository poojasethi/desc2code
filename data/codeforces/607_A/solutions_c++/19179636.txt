#include <iostream>
#include <bits/stdc++.h>
using namespace std;
int main()
{
	long long int N;
	cin>>N;
	long long int A[1000002]={0},d[1000002]={0};
	long long int a1,b1;
	for(int i=0;i<N;i++)
	{	cin>>a1>>b1;
		A[a1]=b1;

	}

	if(A[0]==0)	d[0]=0;
	else d[0]=1;
	
	long long int max=0;
	for(int i=0;i<1000002;i++)
	{	if(A[i]==0)
			d[i]=d[i-1];
		else if(i-A[i]-1>=0)
			d[i]=d[i-A[i]-1]+1;
		else d[i]=1;

		if(d[i]>max)
			max=d[i];

	}

	cout<<N-max<<endl;




	
}
