#include <stdio.h>
int dp[1000010]={0},b[1000010]={0};
int main()
{
	int n,a,i,far=0,max=1;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a);
		scanf("%d",&b[a]);
		if(a>far) far=a;
	}
	if(b[0]>0) dp[0]=1;
	for(i=1;i<=far;i++)
	{
		if(b[i]==0) dp[i]=dp[i-1];
		else
		{
			if(b[i]>=i) dp[i]=1;
			else if(b[i]<i) dp[i]=dp[i-b[i]-1]+1;
		}
		if(dp[i]>max) max=dp[i];
	}
	printf("%d",n-max);
	return 0;
}
