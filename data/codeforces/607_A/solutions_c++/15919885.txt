#include <iostream>

using namespace std;
long long a[1000005], dp[1000005];
long long maxim, n,  b;

int main()
{ cin>>n;
for(long long i=0; i<n; i++){
cin>>b; cin>>a[b]; }
maxim=0;
if (a[0]>0) dp[0]=1;
  for(long long x=1; x<1000005; x++){
       if (a[x]==0) dp[x]=dp[x-1]; else { if (a[x]>=x) dp[x]=1; else dp[x]=dp[x-a[x]-1]+1;}

    if (maxim<dp[x]) maxim=dp[x];
  }
  cout<<n-maxim;
    return 0;
}
