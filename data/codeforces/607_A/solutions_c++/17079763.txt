#include<stdlib.h>

int sv[1000005]={0},Max=-1,ans,n,m,i,a,b,h[1000005]={0};

{
   scanf("%d ",&n);
   for(i=1;i<=n;i++)

      h[a]=b;
   }
   for(i=0;i<=1000003;i++)

      Max=max(Max,sv[i]=((i-h[i]>0?sv[i-h[i]-1]:0)+(h[i]!=0)));
   }

}
