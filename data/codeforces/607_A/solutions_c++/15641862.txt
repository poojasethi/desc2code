#include<bits/stdc++.h>
using namespace std;

const int mxn=1e6+5;

int b[mxn], dp[mxn];

int main() {
  int n,a; cin>>n;
  for (int i=0; i<n; i++)
    cin>>a, cin>>b[a];

  if(b[0]>0) dp[0]=1;

  int mx=0;
  for (int i=1; i<mxn; i++) {
    if(b[i]==0) dp[i]=dp[i-1];
    else {
      if(b[i]>=i) dp[i]=1;
      else dp[i]=dp[i-b[i]-1]+1;
    }
    mx=max(mx,dp[i]);
  }
  cout<<n-mx<<endl;
  return 0;
}
