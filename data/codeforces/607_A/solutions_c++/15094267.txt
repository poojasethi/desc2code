#include<bits/stdc++.h>
const int  MAX = 1e6 + 7;
using namespace std;
int n;
int h[MAX];
int dp[MAX];
int a,b;
int main ()
{
	int mx=-1;
	cin>>n;
	for(int i=1; i<=n; i++)
		scanf("%d %d",&a,&b),
			h[a] = b;
	
	for(int i=0; i<MAX; i++)
		mx = max(mx,dp[i] = ((i-h[i]>0)?dp[i-h[i]-1]:0) + (h[i]!=0));
	
	

	cout<<n-mx;
}
