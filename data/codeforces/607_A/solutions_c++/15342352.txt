#include<iostream>

using namespace std;

int ary[1000001]={0};

int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;++i){
		int mid;
		cin>>mid;
		cin>>ary[mid];
	}
	int max=0;
	for(int i=0;i<=1000000;++i){
		if(ary[i]){
			if(i-ary[i]-1<0){
				ary[i]=1;
			}else{
				ary[i]=ary[i-ary[i]-1]+1;
			}
			if(max<ary[i])
                max=ary[i];
		}else{
			ary[i]=ary[i-1];
		}
	}
	cout<<n-max<<endl;
	return 0;
}
