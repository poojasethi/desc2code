#include<bits/stdc++.h>
using namespace std;
const int MAX=1e6+5;
int lefty[MAX],dp[MAX],ans=0;
int main(){
	int n,a,b;
	cin>>n;
//	int n;
	for(int i=0;i<n;i++){
	cin>>a>>b;
	lefty[a]=b;
	}
	for(int i=0;i<MAX;i++){
		ans=max(ans,dp[i]=(i-lefty[i]>0?dp[i-lefty[i]-1]:0)+(lefty[i]!=0) );
	}
	cout<<n-ans;
	return 0;
}
