 #include<stdio.h>
 #define max(a,b)(a>b)?a:b
 int dp[1000010],a[1000010];
 int main()
 {
 	int i,n,p,f=0;
 	scanf("%d",&n);
 	for(i=0;i<n;i++)
 	{
 		scanf("%d",&p);
 		f=max(f,p);
 		scanf("%d",&a[p]);
	 }
	 if(a[0]>0)dp[0]=1;
	 p=1;
	 for(i=1;i<=f;i++)
	 {
	 	if(a[i]==0)dp[i]=dp[i-1];
	 	else dp[i]=a[i]>i?1:(dp[i-a[i]-1]+1);
	 	p=max(p,dp[i]);
	 }
	 printf("%d\n",n-p);
 	return 0;
 }
