#include<stdio.h>
int power[1000010];
int s[1000010];
int main(){
	int n,t=0,i,p,q,max=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&p,&q);
		power[p]=q;
		if(max<p)
			max=p;
	}
	if(power[0]>0)
		s[0]=1;
	int w=1;
	for(i=1;i<=max;i++)
	{
		if(power[i]==0)
			s[i]=s[i-1];
		else
			s[i]=power[i]>=i?1:s[i-power[i]-1]+1;
		if(s[i]>w)
			w=s[i];
	//	printf("%d\n",s[1]);
	}
	printf("%d",n-w);
	return 0;
}
