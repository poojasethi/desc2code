#include<bits/stdc++.h>
using namespace std;
int p[1000005], f[1000005];
int main(){
	int n, a, ans;

	scanf("%d", &n);
	
	for(int i = 0; i < n; ++i)
	scanf("%d", &a), scanf("%d", &p[a]);
	
	if(p[0]) f[0]++; ans = f[0];
	
	for(int i = 1; i < 1000005; ++i)
	f[i] = !p[i]? f[i-1] : p[i] >= i? 1: f[i - p[i] - 1] +1, ans = max(ans, f[i]);
	
	printf("%d", n-ans);
	return 0;}
