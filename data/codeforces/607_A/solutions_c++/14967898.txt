#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
const int maxn=1000010;

int f[maxn],d[maxn];
int n,dis,ans;

int main()
{
	scanf("%d",&n);
	for (int i=1,a,b;i<=n;i++) scanf("%d%d",&a,&b),d[a+1]=b,dis=max(dis,a+1);
	for (int i=1;i<=dis;i++) if (!d[i]) f[i]=f[i-1];
	else f[i]=f[max(i-d[i]-1,0)]+1,ans=max(ans,f[i]);
	printf("%d\n",n-ans);
	return 0;
}
