#include "cstdio"
const int MAXN = 1e6 + 6;
int n, dp[MAXN], magic[MAXN], ans;
int main(int argc, char const *argv[])
{
	scanf("%d", &n);
	for(int i = 0; i < n; ++i) {
		int x, y;
		scanf("%d%d", &x, &y);
		magic[x] = y;
	}
	for(int i = 0; i <= 1e6; ++i) {
		int x = i - magic[i] - 1;
		if(x < 0) dp[i] = 0;
		else dp[i] = dp[x];
		if(magic[i]) dp[i]++;
		if(dp[i] > ans) ans = dp[i];
	}
	printf("%d\n", n - ans);
	return 0;
}
