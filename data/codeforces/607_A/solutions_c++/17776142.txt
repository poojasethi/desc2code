#include<stdio.h>
int a[1000001],f[1000001],n;
int main()
{
	long i,m,fin=0;
	scanf("%ld",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&m);
		if(m>fin)fin=m;
		scanf("%d",a+m);
	}	
	if(a[0])f[0]=1;
	for(i=1,m=1;i<=fin;i++)
	{
		if(!a[i])f[i]=f[i-1];
		else f[i] = a[i] > i ? 1:(f[i-a[i]-1]+1);
		if(f[i]>m)m=f[i];
	}
	printf("%d",n-m);
	return 0;
}
