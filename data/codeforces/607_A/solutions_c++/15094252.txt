#include<bits/stdc++.h>
const int  MAX = 1e6 + 7;
using namespace std;
int n;
int h[MAX];
int dp[MAX];
int main ()
{
	int mx=-1;
	cin>>n;
	for(int i=1; i<=n; i++)
	{
		int a,b;
		scanf("%d %d",&a,&b);
		h[a] = b;
	}
	
	for(int i=0; i<MAX; i++)
		dp[i] = ((i-h[i]>0)?dp[i-h[i]-1]:0) + (h[i]!=0);
	
	for(int i=0; i<MAX; i++)
		mx = max(mx,dp[i]);
	

	cout<<n-mx;
}
