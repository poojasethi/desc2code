#include <bits/stdc++.h>
using namespace std;

int N, x, a, c[1000010];

int main() {
    cin >> N;
    for (int i = 0; i < N; i++) cin >> x, cin >> c[x];
    for (int i = 0; i < 1000010; a = max(a, c[i++])) {
        if (c[i]) {
            if (i - c[i] >= 0) c[i] = c[i - c[i] - 1] + 1;
            else c[i] = 1;
        } else c[i] = c[i - 1];
    }
    cout << N - a;
}
