#include<bits/stdc++.h>
using namespace std;

int n, a[1000010], dp[1000010], x, y, z;

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i){
		scanf("%d %d", &x, &y);
		a[x+1] = y;
	}
	for(int i=1 ; i<=1000001 ; ++i) {
		if(!a[i]) dp[i] = dp[i-1];
		else dp[i] = 1 + dp[max(0, i-1-a[i])];
		z = max(z, dp[i]);
	}
	printf("%d\n", n - z);
	return 0;
}
