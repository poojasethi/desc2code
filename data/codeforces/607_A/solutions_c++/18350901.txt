#include<cstdio>
#include<algorithm>
#define N 1000010
using namespace std;

int f[N],b[N],n;

int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		int x,y;
		scanf("%d%d",&x,&y);
		b[x]=y;
	}
	for(int i=0;i<=1000000;i++){
		if(b[i]){
			if(i-1-b[i]<0) f[i]=1;
			else f[i]=f[i-1-b[i]]+1;
		}else{
			f[i]=f[i-1];
		}
	}
	int ans=0;
	for(int i=0;i<=1000000;i++) ans=max(ans,f[i]);
	printf("%d\n",n-ans);
	return 0;
}
