#include<bits/stdc++.h>
using namespace std;

int n,a[1000005],b[1000005],p;

main(){
	cin >> n;
	int mx=0,mn=INT_MAX,m=1; 
	for(int i=0; i<n; i++){
		cin >> p;
		a[p]=1;
		cin >> b[p];
		if(mx<p)
			mx=p;
		if(mn>p)
			mn=p;
	}
	a[mn]=1;
	for(int i=mn+1; i<=mx; i++){
		if(a[i]==0)
			a[i]=a[i-1];
		else{
			if((i-b[i]-1)>=0)
				a[i]=a[i-b[i]-1]+1;
			else
				a[i]=1;
		}
		if(a[i]>m)
			m=a[i];
	}
	cout << n-m << endl;
	return 0;
}
