#include<stdio.h>
int a[1000005];
int b[1000005];
int dp[1000005];
int c[1000005];

int main()
{
int n,i,x,ma=1000001;
scanf("%d",&n);
for(i=1;i<=n;i++)
{
scanf("%d",&a[i]);
scanf("%d",&b[i]);
a[i]++;
c[a[i]] = b[i];
}
for(i=1;i<=ma;i++)
{
if(c[i]==0)dp[i] = dp[i-1];
else {
x = i-c[i]-1;
if(x<0)x = 0;
dp[i] = dp[x] + 1;
}
}
int ans = 0;
for(i=0;i<=ma;i++)if(dp[i] > ans)ans = dp[i];
printf("%d\n",n-ans);
return 0;
}
