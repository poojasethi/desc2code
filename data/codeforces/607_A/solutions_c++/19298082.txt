
#include "stdio.h"

int n, x, y, range[1000005], dp[1000005];

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		scanf("%d%d", &x, &y);
		range[x] = y;
	}
	int ans = 0;
	for (int i = 0; i <= 1000000; i++)
	{
		int x = i - range[i] - 1;
		if (x < 0)
			dp[i] = 0;
		else
			dp[i] = dp[x];
		if (range[i])
			dp[i]++;
		if (dp[i] > ans)
			ans = dp[i];
	}
	printf("%d\n", n - ans);
    return 0;
}
