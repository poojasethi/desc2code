#include<cstdio>
#include<cstdlib>
#include<cstring>

#include<iostream>
#include<algorithm>
using namespace std;

const int _N=1000010;
const int N0=1000005;

int n,s[_N],f[_N],re;

int main()
{
    scanf("%d",&n);
    static int x,y;
    for(int i=1;i<=n;i++)
        scanf("%d%d",&x,&y),s[x]=y;
    for(int i=0;i<=N0;i++)
        f[i]=(i-s[i]>0?f[i-s[i]-1]:0)+(s[i]!=0),re=max(re,f[i]);
    printf("%d\n",n-re);
}
