#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

int n,ma,mi,dp[1000005],a[1000006];

int main(){
	scanf("%d",&n);
	for(int i=0; i<n; i++){
		int x,y;
		scanf("%d %d",&x,&y);
		a[x]=y;
		ma=max(ma,x);
	}
	mi=0;
	if(a[0]>0)dp[0]=mi=1;
	for(int i=1; i<=ma; i++){
		if(a[i]==0)dp[i]=dp[i-1];
		else if(a[i]>=i)dp[i]=1;
		else dp[i]=dp[i-a[i]-1]+1;
		mi=max(mi,dp[i]);
	}
	printf("%d\n",n-mi);
	return 0;
}
