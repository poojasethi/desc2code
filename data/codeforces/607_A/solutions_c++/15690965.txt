#include<stdio.h>
int b[1000001]={0},light[1000001]={0};
int main()
{
	int n,t,p,i,nt=0;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&t);
		nt=nt>t?nt:t;
		scanf("%d",&b[t]);
	}
	if(b[0])light[0]=1;
	p=1;
	for(i=1;i<=nt;i++)
	{
		if(!b[i])light[i]=light[i-1];
		else light[i]=b[i]<i?(1+light[i-b[i]-1]):1;
		p=p>light[i]?p:light[i];
	}
	printf("%d",n-p);
	return 0;
}
