#include<iostream>
using namespace std;
int main()
{
	int n,t,i;
	long long int x[100001],h[100001],lc;
	cin>>n;
	if(n>=2)
		t=2;
	else
		t=n;
	for(i=0;i<n;i++)
		cin>>x[i]>>h[i];
	lc=x[0];
	for(i=1;i<n-1;i++)
	{
		if(h[i]<x[i]-lc)
		{
			t++;
			lc=x[i];
		}
		else if(h[i]<x[i+1]-x[i])
		{
			t++;
			lc=x[i]+h[i];
		}
		else
			lc=x[i];
	}
	cout<<t<<endl;
	return 0;
}
