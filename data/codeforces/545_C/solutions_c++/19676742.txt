#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,k,i;
long long x[100000],h[100000];
cin>>n;
for(i=0;i<n;i++)
{
cin>>x[i]>>h[i];
}
if(n>=2)k=2;
else k=1;
for(i=1;i<n-1;i++)
{

if(x[i]-h[i]>x[i-1])
{
k++;}
else if(x[i]+h[i]<x[i+1])
{x[i]=x[i]+h[i];k++;}
else k=k;}
cout<<k<<endl;

return 0;}
