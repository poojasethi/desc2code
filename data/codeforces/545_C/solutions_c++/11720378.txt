#include <iostream>
using namespace std;

int main(){
	int n;
	cin>>n;
	int arrx[n], arrh[n], i,ans=1;
	for(i = 0; i < n; i++)
		cin>>arrx[i]>>arrh[i];
	for(i = 1; i < n-1; i++){
		if((arrx[i]-arrh[i]) > arrx[i-1]) ans++;
		else if(arrx[i]+arrh[i] < arrx[i+1]){
			ans++;
			arrx[i] = arrx[i]+arrh[i];
		}
	}
	if(n>1) ans++;
	cout<<ans<<endl;
}
