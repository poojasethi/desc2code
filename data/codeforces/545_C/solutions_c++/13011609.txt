#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	pair<int,int> a[n];
	for(int i=0;i<n;i++)
		cin>>a[i].first>>a[i].second;
	int ans=2;
	for(int i=1;i<n-1;i++)
	{
		if(a[i].first-a[i-1].first>a[i].second)
			ans++;
		else if(a[i+1].first-a[i].first>a[i].second)
		{
			ans++;
			a[i].first+=a[i].second;
		}
	}
	cout<<(n==1?1:ans)<<endl;
	return 0;
}
