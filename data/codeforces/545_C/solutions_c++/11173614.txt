#include <bits/stdc++.h>
int main()
{
	int i,n;
	scanf("%d",&n);
	int a[n][2];
	for(i=0;i<n;i++)
		scanf("%d%d",&a[i][0],&a[i][1]);
	if(n==1) printf("%d\n",n);
	else {
	int co=0;
	for(i=1;i<n-1;i++)
			if(a[i][0]-a[i][1] > a[i-1][0]) co++;
			else if(a[i][1]+a[i][0] < a[i+1][0]) {co++,a[i][0]+=a[i][1];}
	printf("%d\n",co+2);
	}
	return 0;
}
