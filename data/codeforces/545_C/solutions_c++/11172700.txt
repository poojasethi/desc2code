#include <bits/stdc++.h>
using namespace std;
int main(){
	int n;
	cin>>n;
	if(n==1){
		cout<<1<<endl;
		return 0;
	}
	int ans=2;
	int x[n],h[n];
	for (int i = 0; i < n; i++) cin>>x[i]>>h[i];
	long long sum = x[0];
	for (int i = 1; i < n-1; i++) {
	 	if(sum+h[i]<x[i]) sum=x[i],ans++;
	 	else if(x[i]+h[i]<x[i+1]) sum=x[i]+h[i],ans++;
	 	else sum=x[i];
	} 
	cout<<ans<<endl;
}
