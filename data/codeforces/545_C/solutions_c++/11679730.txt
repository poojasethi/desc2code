#include <cstdio>
int n, x[100100], h[100100], last, cnt;
int main(){
	scanf("%d", &n);
	for(int i=1; i<=n; i++)
		scanf("%d%d", &x[i], &h[i]);
	x[n+1]=2100000000;
	cnt=1;
	for(int i=2; i<=n; i++)
	{
		if(x[i-1]+last*h[i-1]<x[i]-h[i]) cnt++, last=0;
		else if(x[i]+h[i]<x[i+1]) cnt++, last=1;
		else last=0;
	}
	printf("%d\n", cnt);
}
