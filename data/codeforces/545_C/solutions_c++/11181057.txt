#include<bits/stdc++.h>
using namespace std;
int main()
{
long n,c=1;
cin>>n;
long x[n],h[n];
for(int i=0;i<n;i++)
{
cin>>x[i]>>h[i];
}
for(int i=1;i<n-1;i++)
{
 if(x[i]-x[i-1]>h[i])
 {
   c++;
   continue;
 }
 if(x[i+1]-x[i]>h[i])
 {
    c++;
    x[i]=x[i]+h[i];
 }
}
if(n==1)
{
cout<<"1";
}
else
cout<<c+1;
return 0;
}
