#include<bits/stdc++.h>
using namespace std;
int main()
{
	int a[200004][3],n,i;
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d %d",&a[i][0],&a[i][1]);
	int c;
	if(n==1)
		c=1;
	else
		c=2;
	for(i=1;i<n-1;i++)
	{
		if(a[i][0]-a[i-1][0]>a[i][1])
			c++;
		else if(a[i+1][0]-a[i][0]>a[i][1])
		{
			c++;
			a[i][0]=a[i][0]+a[i][1];
		}
	}
	printf("%d",c);
	return 0;
}
