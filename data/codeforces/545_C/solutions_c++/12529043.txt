#include <bits/stdc++.h>
using namespace std;
int a[100001],b[100001];
int main() {
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i]>>b[i];
	}
	int counter=2;
	for(int i=1;i<n-1;i++)
	{
		if(a[i]-b[i]>a[i-1])counter++;
		else if(a[i]+b[i]<a[i+1])
		{
			a[i]+=b[i];
			counter++;
		}
	}
	if(n!=1)cout<<counter<<endl;
	else cout<<"1"<<endl;
	return 0;
}
