#include<bits/stdc++.h>
using namespace std;
int main()
{
int n,i,ans=2;
cin>>n;
int a[n],b[n];
for(int i=0;i<n;i++)
scanf("%d %d",&a[i],&b[i]);

for(i=1;i<n-1;i++)
{
if(a[i]-a[i-1]>b[i])
ans++;
else if(a[i+1]-a[i]>b[i])
{
ans++;
a[i]=a[i]+b[i];
}

}
if(n==1)
cout<<"1"<<endl;
else
cout<<ans<<endl;
}
