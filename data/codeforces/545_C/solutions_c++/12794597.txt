#include <iostream>

using namespace std;
int x[100005],h[100005];

int main()
{
    int n,c;
    cin>>n;
    c=min(n,2);
    for(int i=0; i<n; i++){
        cin>>x[i]>>h[i];
    }
      for(int i=1; i<n-1; i++){
      if(x[i]-h[i]>x[i-1])c++;
      else if(x[i]+h[i] < x[i+1]){c++;x[i]+=h[i];}
      }
                cout<<c<<endl;
    return 0;
}
