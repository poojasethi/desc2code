#include<bits/stdc++.h>
using namespace std;
int ans=0,x[100005],h[100005],i,n;
main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++)
        scanf("%d%d",&x[i],&h[i]);
    x[0]=-1e9,h[0]=-1e9;x[n+1]=2e9+1;
    for (i=1;i<=n;i++)
        if (x[i]-h[i]>x[i-1])
            ans++;
        else if (x[i]+h[i]<x[i+1])
                ans++,x[i]+=h[i];
    printf("%d",ans);
}
