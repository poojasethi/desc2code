#include<iostream>
#include<stdio.h>
int a[100001],b[100001];
int n,i,k,l,h,x,y;
using namespace std;
int main()
{
	cin>>n;
	for(i=0;i<n;i++)cin>>a[i]>>b[i];
	if(n==1){cout<<1;return 0;}
	x=a[0];
	for(i=1;i<n-1;i++){
	 y=a[i];h=b[i];
	 if(x+1<=y-h){x=y;k++;}
	  else{
	  if(y+h+1<=a[i+1]){x=y+h;k++;}else x=y;
	  }
	}
	cout<<k+2;
 	return 0;
}
