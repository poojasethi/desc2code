#include<stdio.h>

int H[100005],x[100005],h[100005],i,j,ans,n,t=-1e9;

int main(){

	scanf("%d",&n);
	for(i = 1; i <= n; i++)
		scanf("%d %d",&x[i],&h[i]);

	x[i] = 2e9 + 123123;
	for(i = 1; i <= n; i++){
		
		if(x[i] - h[i] > t){
			t = x[i];
			ans++;
		}
		else if(x[i] + h[i] < x[i+1]){
			t = x[i]+h[i];
			ans++;
		}
		else
			t = x[i];		

	}

	printf("%d\n",ans);

	return 0;
}
