#include <cstdio>

int main(){
	int n;
	scanf("%d", &n);
	int x[n], h[n];
	for (int i = 0; i < n; i++) {
		scanf("%d%d", &x[i], &h[i]);
	}
	int left = x[0], m = n;
	for (int i = 1; i < n-1; i++) {
		if (left < x[i]-h[i]) {
			left = x[i];
		} else if (x[i]+h[i] < x[i+1]) {
			left = x[i]+h[i];
		} else {
			left = x[i];
			m--;
		}
	}
	printf("%d\n", m);
	return 0;
}
