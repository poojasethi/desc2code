#include<iostream>
using namespace std;

int main()
{
    int n;
    cin>>n;
    int a[n],h[n];
    for(int i=0;i<n;i++)
        cin>>a[i]>>h[i];
    int cnt=(n>2?2:n);
    for(int i=1;i<n-1;i++)
    {
        if(a[i]-a[i-1]>h[i]) {cnt++;}
        else if(a[i+1]-a[i]>h[i]) {cnt++;a[i]+=h[i];}
    }
    cout<<cnt<<endl;
}
