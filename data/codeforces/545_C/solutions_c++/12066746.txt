#include<iostream>
#include<algorithm>
using namespace std;
long long d=0,n, a[1000000],i,res=2,b[1000000];
int main()
{
    cin>>n;
    for(i=1;i<=n;i++)
    {
        cin>>a[i]>>b[i];      
    }
   for(i=2;i<n;i++)
    {
        if(a[i]-b[i]>a[i-1]) res++;
        else
            if( a[i]+b[i]<a[i+1]){ res++;a[i]+=b[i];}
    }
   cout<<(n==1?1:res);
   
}
