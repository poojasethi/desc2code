#include<bits/stdc++.h>

using namespace std;
int main()
{

	int n;
	cin>>n;

	int a[100005],h[100005];

	for(int i=0;i<n;i++)
		cin>>a[i]>>h[i];


	int count=1;
	for(int i=1;i<n-1;i++)
	{
		if(a[i]-h[i] > a[i-1])
			count++;

		else if(a[i]+h[i] < a[i+1])
		{
			count++;
			a[i]+=h[i];
		}
	
	}

	if(n>1)
		count++;

	cout<<count<<endl;



	return 0;
}
