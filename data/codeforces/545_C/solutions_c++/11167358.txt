#include<iostream>
using namespace std;
int main()
{
    int s = 2 , n , x[100001] , h[100001] , l;
    cin>>n;
    for(int i = 1 ; i<=n ; i++) cin>>x[i]>>h[i];
    for(int i = 2 ; i<n ; i++){
        if(x[i] - x[i-1] > h[i]) s++;
        else if(x[i+1] - x[i] > h[i]) s++ , x[i] += h[i]; 
    }
    if(n == 1) cout<<"1";
    else cout<<s<<endl;
}
