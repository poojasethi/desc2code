#include <iostream>
using namespace std;

int n, x[100005], h[100005];
int s;

int main()
{
	cin >> n;
	s = 1;
	x[n] = 2000000001;
	
	for(int i = 0; i < n; i++)
		cin >> x[i] >> h[i];
	
	for(int i = 1; i < n; i++)
	{
		if(x[i] - x[i-1] > h[i]) s++;
		else if(x[i+1] - x[i] > h[i])
		{
			x[i] = x[i] + h[i];
			s++;
		}
	}
	cout << s;
	return 0;
}
