#include <iostream>
#include <algorithm>

using namespace std;

int x[100000],h[100000];

int main()
{
    int n,cnt=2;
    cin >> n;
    for(int i=0;i<n;i++) cin >> x[i] >> h[i];
    for(int i=1;i<n-1;i++)
        if(x[i]-x[i-1]>h[i])cnt++;
        else if(x[i+1]-x[i]>h[i])
            x[i]+=h[i],cnt++;

    cout << (n==1?1:cnt);
    return 0;
}
