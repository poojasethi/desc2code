#include <bits/stdc++.h>
using namespace std;
const int N = 1e5+5;
int x[N],h[N];
int main(){
	int pos=0,i,j,k,n;
	cin>>n;
	for(i=0;i<n;++i)
		cin>>x[i]>>h[i];
	int ret=1;
	pos= x[0];
	x[n]= 2e9+15;
	for(i=1;i<n;++i){
		if(pos<x[i]-h[i])
			++ret,pos =x[i];
		else{
			if(x[i]+h[i]<x[i+1])
				pos = x[i]+h[i],++ret;
			else
				pos = x[i];
		}
	}
	cout<<ret<<endl;
}
