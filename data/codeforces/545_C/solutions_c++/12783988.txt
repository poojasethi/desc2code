#include<bits/stdc++.h>
using namespace std;

int n, a[100005], b[100005];
int s;

main() {
    ios_base::sync_with_stdio(0);
    cin >> n;
    s=2;
    for(int i=0; i<n; i++) cin >> a[i] >> b[i];
    for(int i=1; i<n-1; i++)
    if (a[i]-a[i-1]>b[i]) s++;
    else if (a[i+1]-a[i]>b[i]) {s++; a[i]+=b[i];}

    n==1 ? cout << "1\n" : cout << s << "\n";
}
