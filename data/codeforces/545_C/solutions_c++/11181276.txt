#include <bits/stdc++.h>
using namespace std;
int main()
{
	int x[100005];
	int h[100005];
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>x[i]>>h[i];
	}
	int ans=n>1?2:1;
	for(int i=1;i<(n-1);i++)
	{
		if((x[i]-x[i-1])>h[i])
			ans++;
		else if((x[i+1]-x[i])>h[i])
		{
			x[i]=x[i]+h[i];
			ans++;
		}
	}
	cout<<ans;
	return 0;
	
}
