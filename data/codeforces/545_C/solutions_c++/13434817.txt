#include <bits/stdc++.h>
using namespace std;
int n;
int ans=2;
int main(){
cin>>n;
long long a[n];
long long h[n];
for(int i=0; i<n; i++){
cin>>a[i]>>h[i];}
if(n==1){
cout<<1;
return 0;}
for(int i=1; i<n-1; i++){
if(a[i]-h[i]>a[i-1]){
ans++;

continue;}
if(a[i]+h[i]<a[i+1]){
a[i]=a[i]+h[i];

ans++;}
}
cout<<ans;
}
