#include <cstdio>

const int maxn = 100010;
int x[maxn], h[maxn];

int main() {
	int n, cnt=2;
	scanf("%d", &n);
	for(int i=0; i<n; i++) scanf("%d%d", &x[i], &h[i]);
	for(int i=1; i<n-1; i++) {
		if(x[i]-x[i-1]>h[i]) cnt++;
		else if(x[i+1]-x[i]>h[i]) {
			x[i] += h[i];
			cnt++;
		}
	}
	printf("%d\n", n-1?cnt:1);
	return 0;
}
