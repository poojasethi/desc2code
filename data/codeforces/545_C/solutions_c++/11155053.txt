#include<iostream>
using namespace std;
int main()
{
	int n,x[100001],h[100001],i,j,l,ans=0;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>x[i]>>h[i];
	}

	ans=1;
	if(n>=2)
		ans=2;
	for(i=1;i<n-1;i++)
	{
		if(x[i-1]<x[i]-h[i])
		{
			ans++;
		}
		else if(x[i+1]>x[i]+h[i])
		{
			ans++;
			x[i]=x[i]+h[i];
		}
	}

	cout<<ans<<endl;

	return 0;
}
