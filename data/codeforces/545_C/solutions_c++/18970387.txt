#include<bits/stdc++.h>
using namespace std;
int main()
{int n,a[100000],b[100000];
cin>>n;
for(int i=0;i<n;i++)
{cin>>a[i]>>b[i];
}
if(n==1)
{
    cout<<"1";
    return 0;
}
int count=2;
for(int i=1;i<n-1;i++)
{
if(a[i]-a[i-1]>b[i])
{
count++;
}
else if(a[i+1]-a[i]>b[i])
{
count++;
a[i]=a[i]+b[i];
}
}
cout<<count;
return 0;}
