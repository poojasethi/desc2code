#include <iostream>

using namespace std;

int main(){
	long long int n,x[100100]={0},h[100100]={0},s=2;
	cin>>n;
	if(n==1){cout<<1;return 0;}
	for(int i=0;i<n;i++)cin>>x[i]>>h[i];
	for(int i=1;i<n-1;i++){
		if(x[i]-x[i-1]>h[i])s++;
		else if(x[i+1]-x[i]>h[i]){s++;x[i]+=h[i];}
	}
	cout<<s<<"\n";

	return 0;
}
