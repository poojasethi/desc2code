#include<iostream>
using namespace std;
int main()
{
	int n,i;
	cin >> n;
	long long int a[n],h[n],count=0;
	for(i=0;i<n;i++)
	{
		cin >> a[i];
		cin >> h[i];
	}
	for(i=1;i<n-1;i++)
	{
		if(a[i-1]<a[i]-h[i])
		{
			count++;
		}
		else if(a[i+1]>a[i]+h[i])
		{
			count++;
			a[i]=a[i]+h[i];
		}
	}
	if(n>1)
		count+=2;
	else if(n==1)
		count+=1;
	cout << count << "\n";
	return 0;
}
