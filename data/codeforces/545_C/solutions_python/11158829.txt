n = int(raw_input())
arvores = []
count = 0

for i in xrange(n):
    arvores.append(map(int, raw_input().split()))

for i in xrange(n):
    if i == 0:
        count += 1

    elif i == n - 1:
        count += 1

    else:
        if arvores[i][0] - arvores[i][1] > arvores[i-1][0]:
            count += 1

        elif arvores[i][0] + arvores[i][1] < arvores[i+1][0]:
            arvores[i][0] = arvores[i][0] + arvores[i][1]
            count += 1

print count
