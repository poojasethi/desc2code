n = int(raw_input())
x = [0 for i in range(n)]
h = [0 for i in range(n)]
for i in range(n):
	x[i], h[i] = map(int,raw_input().split(" "))
if n <= 2:
	print n
else:
	cut = 2
	for i in range(1,n-1):
		if x[i] - h[i] > x[i-1]:
			cut += 1
		elif x[i] + h[i] < x[i+1]:
			x[i] += h[i]
			cut += 1
	print cut
