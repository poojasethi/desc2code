n = int(raw_input())
x = []
h = []
for _ in xrange(n):
	xi, hi = map(int, raw_input().split())
	x.append(xi)
	h.append(hi)
r = False
out = min(2, n)
for i in xrange(1, n - 1):
	if r:
		if h[i] + h[i - 1] < x[i] - x[i - 1]:
			out += 1
			r = False
		elif h[i] < x[i + 1] - x[i]:
			out += 1
			r = True
		else:
			r = False
	else:
		if h[i] < x[i] - x[i - 1]:
			out += 1
			r = False
		elif h[i] < x[i + 1] - x[i]:
			out += 1
			r = True
		else:
			r = False
print out
