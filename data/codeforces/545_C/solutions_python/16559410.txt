import sys
n = int(raw_input())
if n < 2:
    print 1
    sys.exit()

trees = [map(int, raw_input().split()) for i in range(n)]

result = 1  # first falls left
for i in range(1, n-1):
    prev = trees[i-1]
    curr = trees[i]
    nextt = trees[i+1]

    if prev[0] + curr[1] < curr[0]:  # fall left
        result += 1
    elif curr[0] + curr[1] < nextt[0]:  # fall right
        result += 1
        trees[i][0] += curr[1]
    else:
        continue

result += 1
print result
