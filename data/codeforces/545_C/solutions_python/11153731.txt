n = int(raw_input())
x = []
h = []
current_point = 0
ans = 0

for i in range(n):
	l = map(int, raw_input().split(' '))
	x.append(l[0])
	h.append(l[1])

for i in range(n):
	# Check Left
	if current_point == 0 or x[i] - h[i] > current_point:
		current_point = x[i]
		ans += 1
	# Check Right
	elif i == n-1 or x[i] + h[i] < x[i+1]:
		current_point = x[i] + h[i]
		ans += 1
	else:
		current_point = x[i]

print ans
