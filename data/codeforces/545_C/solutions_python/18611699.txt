n=int(raw_input())
arr=[]
count=0
for i in xrange(n):
    arr.append(map(int,raw_input().split()))
for i in xrange(n):
    if i==0:
        count+=1
    elif i==n-1:
        count+=1
    else:
        if arr[i][0]-arr[i][1]>arr[i-1][0]:
            count+=1
        elif arr[i][0]+arr[i][1]<arr[i+1][0]:
            arr[i][0]=arr[i][0]+arr[i][1]
            count+=1
print(count)
