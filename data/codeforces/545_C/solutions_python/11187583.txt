n=int(input())
lis=[]
for i in range(n):
	a,b=map(int,raw_input().split())
	lis.append([a,b])
lis.append([100000000000,0])
prev=-100000000000
ctr=0
for i in range(n):
	if(lis[i][0]-lis[i][1]>prev):
		ctr+=1
		prev=lis[i][0]
	elif (lis[i][0]+lis[i][1]<lis[i+1][0]):
		ctr+=1
		prev=lis[i][0]+lis[i][1]
	else:
		prev=lis[i][0]
print ctr
