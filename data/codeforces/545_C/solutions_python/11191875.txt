import sys, os, re

n = int(sys.stdin.readline())
lst = []
for i in xrange(n):
    items = [int(x) for x in sys.stdin.readline().strip().split()]
    lst.append([items[0], items[1]])

total = 1
for i in xrange(1,n-1):
    if lst[i][0]-lst[i-1][0]>lst[i][1]:
        total += 1
    elif lst[i+1][0]-lst[i][0]>lst[i][1]:
        total += 1
        lst[i][0] +=lst[i][1]
if n>1:
