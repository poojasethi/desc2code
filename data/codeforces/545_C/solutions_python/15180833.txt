n=input()
x=[]
h=[]
for i in xrange(n):
    t=map(int,raw_input().split())
    x+=[t[0]]
    h+=[t[1]]
if(n<2):
    l=1
else:
    l=2
x1=x[0]
for i in xrange(1,n-1):
    if(x[i]-x1>h[i]):
        l+=1
        x1=x[i]
    elif(x[i+1]-x[i]>h[i]):
        l+=1
        x1=x[i]+h[i]
    else:
        x1=x[i]
print l
