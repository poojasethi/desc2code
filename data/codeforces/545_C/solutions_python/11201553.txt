n = int(raw_input())
x,h=[0]*n,[0]*n
for i in xrange(n):
    x[i],h[i] = map(int,raw_input().split(' '))
ans,pre=0,-0x3fffffff
for i in xrange(n):
    if pre<x[i]-h[i]:
        pre=x[i]
        ans=ans+1
    elif i==n-1 or x[i]+h[i]<x[i+1]:
        pre=x[i]+h[i]
        ans=ans+1
    else:
        pre=x[i]
print ans
