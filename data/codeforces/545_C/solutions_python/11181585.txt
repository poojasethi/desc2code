n=int(raw_input())
x=[]
h=[]
for i in xrange(n):
    xi,hi=map(int,raw_input().split())
    x.append(xi)
    h.append(hi)
#print h
c_opt=[h[0]]
for j in xrange(1,n):
    c_opt.append(x[j]-x[j-1]-1)
c_opt.append(h[-1])
#print c_opt
a=0
for k in xrange(n):
    if h[k]<=c_opt[k]:
        a+=1
    else:
        if h[k]<=c_opt[k+1]:
            c_opt[k+1]-=h[k]
            a+=1
print a
