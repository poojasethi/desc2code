size = int(raw_input())
trees = []
for i in range(size):
    trees.append([int(i) for i in raw_input().split(' ')])

cut = min(size,2)
left = trees[0][0]
for i in range(1,len(trees)-1):
    if trees[i][0] - trees[i][1] > left:
        
        cut += 1
        left = trees[i][0]
    elif trees[i][0]+trees[i][1] < trees[i+1][0]:
        
        left = trees[i][0]+trees[i][1]
        cut += 1
    else:
        left = trees[i][0]

        
print cut
