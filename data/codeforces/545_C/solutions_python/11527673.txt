n = int(raw_input())
x = []
h = []
for i in xrange(n):
    xi, hi = map(int, raw_input().split())
    x.append(xi)
    h.append(hi)
count = 1
loc = x[0]
for i in xrange(1, n-1):
    if x[i]-h[i]>loc:
        count += 1
        loc = x[i]
    elif x[i]+h[i]<x[i+1]:
        count += 1
        loc = x[i]+h[i]
    else:
        loc = x[i]

if n == 1:
    print 1
else:
    print count+1
