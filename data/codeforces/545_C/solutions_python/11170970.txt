n = input()
m = []

for i in xrange(n):
    m.append(map(int, raw_input().split()))

dp = [[0, -9999999999999999999999999], ]

for i in range(n - 1):
    cx, ch = m[i]
    nx, nh = m[i + 1]

    if dp[-1][1] < (cx - ch):
        dp.append([dp[-1][0] + 1, cx])
    elif (cx + ch) < nx:
        dp.append([dp[-1][0] + 1, cx + ch])
    else:
        dp.append([dp[-1][0], cx])

print dp[-1][0] + 1
