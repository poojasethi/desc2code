__author__ = 'Azad'

n = int(raw_input())

s = [tuple(map(int,(raw_input().split()))) for x in xrange(n)]


if len(s)<3:
    print len(s)
    exit()

ma = 2

side = s[0][0]

for x in xrange(1,len(s)-1):

    pos , high = s[x]

    next_pos = s[x+1][0]
    if pos - high>side:
        ma +=1
        side = pos

    elif high + pos < next_pos:
        ma += 1
        side = high + pos

    else:

        side =pos


print ma
