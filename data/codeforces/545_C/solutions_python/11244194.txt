n = int(raw_input())
trees = []
height = []
for k in range(n):
    x,h = map(int, raw_input().split())
    trees.append(x)
    height.append(h)

def maxTrees(n,trees,height):
    if len(trees) > 2:
        refpoint = trees[0]
        felltrees = 2
        for k in range(1,n-1):
            if trees[k] - height[k] > refpoint :
                felltrees += 1
                refpoint = trees[k]
            elif trees[k] + height[k] < trees[k+1]:
                felltrees += 1
                refpoint = trees[k] + height[k]
            else:
                refpoint = trees[k]


        return felltrees
    else:
        return len(trees)

print maxTrees(n,trees,height)
