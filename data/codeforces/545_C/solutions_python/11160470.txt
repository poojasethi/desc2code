n=int(input())
v=[]
for i in range(0,n):
	x,h=map(int,raw_input().split())
	v.append([x,h])

v.sort()

count=1
hp=v[0][0]
#print v
for i in range(1,n-1):
	if(hp<v[i][0]-v[i][1]):
		count+=1
		hp=v[i][0]
	elif(v[i][0]+v[i][1]<v[i+1][0]):
		count+=1
		hp=v[i][0]+v[i][1]
	else:
		hp=v[i][0]
if(n!=1):
	count+=1	
print count
