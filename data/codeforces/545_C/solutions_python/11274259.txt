X=2000020000
n=input()
a=[[-X,0]]+[map(int,raw_input().split()) for _ in range(n)]+[[X,0]]
v=[[0,0,0] for _ in range(n+1)]
for i in range(n):
  x0,h0,x1,h1=a[i-1]+a[i]
  s,t=max(v[i-1][0],v[i-1][1]),v[i-1][2]
  if x1-h1>x0:
    v[i][0]=max(v[i][0],1+s)
  if x1-h1>x0+h0:
    v[i][0]=max(v[i][0],1+t)
  v[i][1]=max(v[i][1],max(s,t))
  if x1+h1<a[i+1][0]:
    v[i][2]=1+v[i][1]
z=0
for x in v:
  z=max(z,max(x))
print z
