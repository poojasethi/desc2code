n = input()
trees = []
cnt = 2
if(n<2):
    cnt = n
for _ in xrange(n):
    a,b = map(int,raw_input().split())
    trees+= [(a,b)]
start = trees[0][0]
for i in xrange(1,n-1):
    if(trees[i][0]-trees[i][1]>start):
        cnt+=1
        start = trees[i][0]
    elif(trees[i][0]+trees[i][1]<trees[i+1][0]):
        cnt+=1
        start = trees[i][0]+trees[i][1]
    else:
        start = trees[i][0]
print cnt
