def C545():
    n = input()
    w = [map(int,raw_input().split()) for _ in xrange(n)]
    ans = 2
    if n <= 2:
        print n
        exit(0)
    for i in xrange(1,n-1):
        x,h = w[i] 
        if x - h > w[i-1][0]:
            ans += 1
        elif x + h < w[i+1][0]:
            w[i][0] += w[i][1]
            ans += 1
    print ans    
    
if __name__=='__main__':
    C545()
