import sys

n = int(raw_input())
trees = []
for i in xrange(n):
    trees.append((map(int, raw_input().split())))

lfted = -sys.maxint
ans = 0
for i, (x, h) in enumerate(trees):
    if x - h > lfted:
        lfted = x
        ans += 1
    elif i == n - 1 or x + h < trees[i + 1][0]:
        lfted = x + h
        ans += 1
    else:
        lfted = x

print ans
