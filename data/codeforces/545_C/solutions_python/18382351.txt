n = int(raw_input())
trees = []
for i in range(n):
	trees.append([int(i) for i in raw_input().split()])
last_cut = trees[0][0]
max_trees = 1 if n == 1 else 2
for i in range(1, n - 1):
	if trees[i][0] - trees[i][1] > last_cut:
		max_trees += 1
		last_cut = trees[i][0]
	elif trees[i][0] + trees[i][1] < trees[i + 1][0]:
		max_trees += 1
		last_cut = trees[i][0] + trees[i][1]
	else:
		last_cut = trees[i][0]
print max_trees
