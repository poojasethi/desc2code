#!/usr/bin/python2
# -*- coding: utf-8 -*-

import sys
import os

def main():
    _ = sys.stdin.readline()
    data = map(int, sys.stdin.readline().strip().split())
    r = 0
    count = [0, 0]
    for i in data:
        count[i] += 1
    c0 = count[0]
    for v in data:
        if v == 0:
            c0 -= 1
        else:
            r += c0
    print r


if __name__ == '__main__':
    main()
