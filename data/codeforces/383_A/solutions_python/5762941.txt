n = int(raw_input())
a = map(int, raw_input().split())

c = 0
b = [False] * n

k = 0
j = 0
for i in range(n):
	if a[i] == 0 and not b[i]: j += 1
	if a[i] == 1 or i == n - 1:
		c += j * k
		b[i] = True
		j = 0
		k += 1
