from sys import stdin
def main():
    n = int(stdin.readline())
    a = stdin.readline().split()
    t = [0]
    ans = 0
    for i, x in enumerate(a):
        if x == '0':
            ans += t[-1]
            t.append(t[-1])
        else:
            t.append(t[-1] + 1)
    for i, x in enumerate(a):
        if x == '1':
            ans += (n - i - 1) - (t[-1] - t[i+1])
    print ans / 2
main()
