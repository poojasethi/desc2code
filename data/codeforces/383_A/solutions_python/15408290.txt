n = int(raw_input())
a = raw_input().split()

#left index 0
#right index 1
try: 
    firstOccuranceOfRight = a.index('1')
    counter = 0
    for i in xrange(firstOccuranceOfRight,n): #cow how many faces left from first occurance
        if a[i] == '0':
            counter+=1
    globalcounter=0
    for i in xrange(firstOccuranceOfRight,n):
        if a[i] == '1': #if faces right, add counter
            globalcounter+=counter
        else:
            counter-=1
        if counter == 0:
            break
    print globalcounter
except:
    print 0
