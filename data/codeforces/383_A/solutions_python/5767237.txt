import fileinput
a = []
for line in fileinput.input():
    a.append(line)
n = int(a[0][:-1])
s = a[1]
cows = [int(s[i*2:i*2+1]) for i in range(n)] 

min_sum = 0
mult = 0
count = 0
for i in range(n):
    if cows[i] == 1:
        min_sum += count*mult
        mult += 1
        count = 0
    else:
        count += 1
min_sum += count*mult
print min_sum
