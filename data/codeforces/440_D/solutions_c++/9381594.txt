#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
class sp{
  public:
  int q;
  sp *w;
  sp(int q1,sp *w1){
   q=q1;w=w1;}
};
sp *a[400];
int s[400][401],d[400],h[400];
void l(int q,int w){
  d[q]=1;
  sp *c;
  int v,e,r;
  s[q][0]=1;
  s[q][1]=0;
  for(e=2;e<401;e++)
   s[q][e]=999;
  for(c=a[q];c;c=c->w)
   if(c->q!=w){
    v=c->q;
    l(v,q);
    for(e=d[q];e>0;e--){
     for(r=d[v];r>0;r--)
      s[q][e+r]=min(s[q][e+r],s[q][e]+s[v][r]);
     s[q][e]++;}
    d[q]+=d[v];}
}
void l1(int q,int w,int i){
  d[q]=1;
  sp *c;
  int v,e,r,c1,d1[401];
  short int *f[401],*f1[401];
  for(e=0;e<401;e++){
   f[e]=new short int[e+1];
   f1[e]=new short int[e+1];}
  s[q][0]=1;
  s[q][1]=0;
  h[q]=1;
  d1[1]=0;
  for(e=2;e<401;e++)
   s[q][e]=999;
  for(c=a[q];c;c=c->w)
   if(c->q!=w){
    v=c->q;
    for(e=d[q];e>0;e--){
     for(r=d[v];r>0;r--)
      if(s[q][e+r]>s[q][e]+s[v][r]){
       s[q][e+r]=s[q][e]+s[v][r];
       for(c1=0;c1<d1[e];c1++){
        f[e+r][c1]=f[e][c1];
        f1[e+r][c1]=f1[e][c1];}
       f[e+r][d1[e]]=v;
       f1[e+r][d1[e]]=r;
       d1[e+r]=d1[e]+1;}
     s[q][e]++;}
    d[q]+=d[v];}
  for(e=0;e<d1[i];e++)
   l1(f[i][e],q,f1[i][e]);
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t=999,y=-1,i,f[400],f1[400];
  cin>>q>>i;
  for(w=0;w<q;w++){
   a[w]=0;
   h[w]=0;}
  for(w=1;w<q;w++){
   scanf("%d%d",&e,&r);
   e--;r--;
   a[e]=new sp(r,a[e]);
   a[r]=new sp(e,a[r]);
   f[w]=e;
   f1[w]=r;}
  for(w=0;w<q;w++){
   l(w,-1);
   if(t>s[w][i]){
    t=s[w][i];
    y=w;}}
  cout<<t<<"\n";
  l(y,-1);
  l1(y,-1,i);
  for(w=1;w<q;w++)
   if(h[f[w]]!=h[f1[w]])
    printf("%d ",w);
  return 0;}
