#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
using namespace std;
const int X=100010;
vector<int>e[X*2];
vector<pair<int,int> >event;
int as[X*2];
void dfs(int u,int color){
    as[u]=color;
    for(int i=0;i<e[u].size();i++)
        if(as[e[u][i]]==-1)
            dfs(e[u][i],color^1);
}
void addedge(int u,int v){
    e[u].push_back(v);
    e[v].push_back(u);
}
int main(){
    int i,j,n,l,r;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d%d",&l,&r);
        event.push_back(make_pair(l*2,i));
        event.push_back(make_pair(r*2+1,i+n));
        addedge(i,i+n);
    }
    memset(as,-1,sizeof(as));
    sort(event.begin(),event.end());
    for(i=0;i<n*2;i+=2)
        addedge(event[i].second,event[i+1].second);
    for(i=0;i<n;i++)
        if(as[i]==-1)
            dfs(i,0);
    for(i=0;i<n;i++)
        printf("%d ",as[i]);
    return 0;
}
