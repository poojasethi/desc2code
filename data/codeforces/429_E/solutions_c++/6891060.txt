#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
#include<vector>
#define MAXN 200005
using namespace std ;
typedef long long LL ;

int N, M ;
vector<int> out[MAXN] ;
int at[MAXN], rk[MAXN] ;
int col[MAXN] ;

bool cmp(int x, int y)
{
	return at[x] < at[y] ;
}

void Dfs(int x)
{
	int i, tmp ;
	for(i = 0; i < out[x].size(); i ++)
		if(!col[tmp=out[x][i]])
			col[tmp] = 3^col[x], Dfs(tmp) ;
}

int main()
{
	int i, j ;
	scanf("%d", &N) ;
	for(i = 1; i <= N; i ++)
	{
		scanf("%d %d", &at[i], &at[i+N]) ;
		++at[i+N] ;
		out[i].push_back(i+N), out[i+N].push_back(i) ;
		rk[i] = i, rk[i+N] = i+N ;
	}
	sort(rk+1, rk+1+N*2, cmp) ;
	for(i = 1; i <= 2*N; i += 2) 
		out[rk[i]].push_back(rk[i+1]), out[rk[i+1]].push_back(rk[i]) ;
	for(i = 1; i <= 2*N; i ++)
		if(col[i] == 0)
			col[i] = 1, Dfs(i) ;
	for(i = 1; i <= N; i ++)
		printf("%d ", col[i]-1) ;
	//system("pause") ;
	return 0 ;
}
