#include <vector>
#include <cstring>
#include <string>
#include <algorithm>
#include <iostream>
#include <cmath>
#include <map>
#include <cstdio>

using namespace std;

#define FOR(i, a, b) for(int i = a; i < b; i++)
#define REP(i, n) FOR(i, 0, n)
#define ll long long
#define mp make_pair
#define pb push_back

#define N 100005

int n, color[N+N];
bool vis[N+N];
vector<int> a[N+N];
vector< pair<int, int> > event;

void go(int u, int c = 0)
{
    vis[u] = 1;
    color[u] = c;
    REP(i, a[u].size())
    {
        int v = a[u][i];
        if(!vis[v]) go(v, c^1);
    }
}

int main()
{
    ios_base::sync_with_stdio(0);
    cin >> n;
    REP(i, n)
    {
        int l, r; cin >> l >> r;
        event.pb(mp(l<<1, i<<1));
        event.pb(mp(r<<1|1, i<<1|1));
        a[i<<1].pb(i<<1|1);
        a[i<<1|1].pb(i<<1);
    }
    sort(event.begin(), event.end());
    REP(i, n)
    {
        a[event[i<<1].second].pb(event[i<<1|1].second);
        a[event[i<<1|1].second].pb(event[i<<1].second);
    }
    REP(i, (n<<1)) if(!vis[i]) go(i);
    REP(i, n) cout << color[i<<1] << ' '; cout << endl;
    return 0;
}
