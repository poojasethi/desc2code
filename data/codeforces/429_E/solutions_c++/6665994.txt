using namespace std;


const int maxn=200010;

vector<int> e[maxn];
vector< pair<int,int> > events;
int col[maxn];

void add(int u,int v)
{
    e[u].push_back(v);
    e[v].push_back(u);
    return ;
}

void dfs(int u,int co)
{
    if (col[u]!=-1) return ;
    col[u]=co;
    for (vector<int>::iterator i=e[u].begin(); i!=e[u].end(); i++)
            dfs(*i,co^1);
    return ;
}

int main()
{
    int n;
    scanf("%d",&n);
    for (int i=0; i<n; i++) {
            int l,r;
            scanf("%d %d",&l,&r);
            r++;
            add(2*i,2*i+1);
            events.push_back(make_pair(l,2*i));
            events.push_back(make_pair(r,2*i+1));
    }
    sort(events.begin(),events.end());
    for (int i=0; i<n; i++) add(events[i*2].second,events[i*2+1].second);
    memset(col,-1,sizeof col);
    for (int i=0; i<2*n; i++)
            if (col[i]==-1) dfs(i,0);
    for (int i=0; i<n; i++) cout<<col[2*i]<<" ";
    return 0;
}
