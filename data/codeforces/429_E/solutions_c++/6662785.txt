#include<stdio.h>
#include<map>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=200050;
int n,l[maxn],r[maxn];
int p[maxn],ps=0;
map<int,int> q;
int de[maxn];
int he[maxn],to[maxn*2],ne[maxn*2];
int c[maxn];
void ins(int x,int y,int id)
{
	to[id]=y,ne[id]=he[x],he[x]=id;
}
void add(int x,int y,int id)
{
	ins(x,y,id*2);
	ins(y,x,id*2+1);
}
void dfs(int x)
{
	int i,j;
	for(i=he[x];i;i=ne[i])
		if(c[i>>1]<0)
			c[i>>1]=(to[i]>x),dfs(to[i]);
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,n0;
	scanf("%d",&n);
	n0=n;
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&l[i],&r[i]);
		r[i]++;
		p[i*2-2]=l[i];
		p[i*2-1]=r[i];
	}
	sort(p,p+n*2);
	ps=unique(p,p+n*2)-p;
	for(i=0;i<ps;i++)
		q[p[i]]=i+1;
	for(i=1;i<=n;i++)
	{
		l[i]=q[l[i]],r[i]=q[r[i]];
		de[l[i]]++,de[r[i]]++;
		add(l[i],r[i],i);
	}
	for(i=1;i<=ps;i++)
		if(de[i]&1)
		{
			for(j=i+1;j<=ps&&!(de[j]&1);j++);
			add(i,j,++n0);
			i=j;
		}
	for(i=1;i<=n0;i++)
		c[i]=-1;
	for(i=1;i<=ps;i++)
		dfs(i);
	for(i=1;i<=n;i++)
		printf("%d%c",c[i],i<n?' ':'\n');
	return 0;
}
