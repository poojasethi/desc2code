#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

vector<pair<int,int> > g[1000000];
int visto[1000000];
int sol[1000000];

int signo(int x)
{
  if (x>0) return 1;
  return -1;
}

int absolut(int x)
{
  if (x<0) return -x;
  return x;
}

void genera(int u,int val)
{
  if (visto[u]) return;
  visto[u]=1;
  sol[u]=val;
  vector<pair<int,int> > &ar=g[u];
  for (int i=0;i<int(ar.size());i++)
    genera(ar[i].first,val^ar[i].second);
}

int main()
{
  int n;
  cin>>n;
  vector<pair<int,int> > v;
  for (int i=1;i<=n;i++) {
    int l,r;
    cin>>l>>r;
    v.push_back(pair<int,int> (l,-i));
    v.push_back(pair<int,int> (r,i));
  }
  sort(v.begin(),v.end());
  for (int i=0;i<int(v.size());i+=2) {
    if (signo(v[i].second)==signo(v[i+1].second)) {
      g[absolut(v[i].second)].push_back(pair<int,int> (absolut(v[i+1].second),1));
      g[absolut(v[i+1].second)].push_back(pair<int,int> (absolut(v[i].second),1));
    } else {
      g[absolut(v[i].second)].push_back(pair<int,int> (absolut(v[i+1].second),0));
      g[absolut(v[i+1].second)].push_back(pair<int,int> (absolut(v[i].second),0));
    }
  }
  for (int u=1;u<=n;u++)
    genera(u,0);
  for (int u=1;u<=n;u++) {
    if (u>1) cout<<" ";
    cout<<sol[u];
  }
  cout<<endl;
}
