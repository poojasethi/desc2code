#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
using namespace std;

#define N 200010
#define pb push_back
#define pr pair <int,int>
#define mp make_pair
#define X first
#define Y second
vector <int> g[N];
pr a[N];
#define L (i * 2)
#define R (i * 2 + 1)
int n, an[N], vis[N] = {0};
void dfs(int x,int co){
	vis[x] = 1;
	an[x] = co;
	for(int i = 0;i < g[x].size();i ++)	if(!vis[g[x][i]])	dfs(g[x][i], co ^ 1);
}
int main(){
	scanf("%d", &n);
	int x,y;
	for(int i = 0;i < n;i ++){
		scanf("%d%d", &x, &y);
		g[L].pb(R);
		g[R].pb(L);
		a[L] = mp(y * 2 + 1, L);
		a[R] = mp(x * 2, R);
	}
	sort(a, a + 2 * n);
	for(int i = 0;i < n;i ++){
		x = a[L].Y, y = a[R].Y;
		g[x].pb(y);
		g[y].pb(x);
	}
	for(int i = 0;i < n * 2;i ++)	if(!vis[i])	dfs(i,0);
	for(int i = 0;i < n;i ++)	printf("%d ",an[L]);
	return 0;
}
