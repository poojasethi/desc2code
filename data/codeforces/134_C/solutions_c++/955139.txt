#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#include<vector>
#include<set>
#include<string>
#include<map>
#include<queue>
using namespace std;

struct Node{
	int id,n;
	Node(){}
	Node(int id, int n):id(id),n(n){}
	bool operator <(const Node &nd)const{
		return n<nd.n;
	}
};

int a[200005];
priority_queue<Node> pq;
int ans[200005][2];
Node nd[200005];
int main(){
	int i,j,k,m,n,s;
	Node tnd;
	bool f;
	while(cin>>n>>s){
		for(i=1;i<=n;++i){
			scanf("%d",&a[i]);
			if(a[i]>0)
				pq.push(Node(i,a[i]));	
		}
		m=0;
		f=false;
		while(true){
			tnd=pq.top();
			pq.pop();
			if(tnd.n>pq.size())	break;
			for(i=0;i<tnd.n;++i){
				nd[i]=pq.top();
				pq.pop();
				ans[m][0]=tnd.id;
				ans[m][1]=nd[i].id;
				++m;
				--nd[i].n;
			}
			for(i=0;i<tnd.n;++i){
				if(nd[i].n>0)	pq.push(nd[i]);
			}
			if(pq.empty()){
				f=true;break;
			}
		}
		
		if(f){
			puts("Yes");
			cout<<m<<endl;
			for(i=0;i<m;++i)	printf("%d %d\n",ans[i][0],ans[i][1]);
		}
		else	puts("No");
		
	}
}
