#include <iostream>
#include <utility>
#include <set>
#include <vector>
using namespace std;
typedef pair<int,int> pie;
#define endl '\n'
set <pie> me;
int n;
vector <pie> move;
vector <pie> tmp;
int main()
{
	ios_base::sync_with_stdio(false);
	cin>>n;
	int _;
	cin>>_;
	for (int i=0;i<n;i++)
	{
		int a;
		cin>>a;
		me.insert(pie(a,i+1));
	}
	while (!me.empty())
	{
		pie f=*me.rbegin();
		me.erase(f);
		if (f.first==0)
			break;
		tmp.clear();
		for (int i=0;i<f.first;i++)
		{
			if (me.empty())
			{
				cout<<"No"<<endl;
				return 0;
			}
			pie no=*me.rbegin();
			me.erase(no);
			if (no.first==0)
			{
				cout<<"No"<<endl;
				return 0;
			}
			no.first--;
			tmp.push_back(no);
			move.push_back(pie(f.second,no.second));
		}
		for (int i=0;i<tmp.size();i++)
			me.insert(tmp[i]);
	}
	cout<<"Yes"<<endl;
	cout<<move.size()<<endl;
	for (int i=0;i<move.size();i++)
		cout<<move[i].first<<" "<<move[i].second<<endl;
	return 0;
}
