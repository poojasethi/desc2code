#include <iostream>
#include <cstdio>
#include <vector>
#include <set>
#include <utility>
using namespace std;
typedef pair<int,int> pii;
int n,s;
int step[200010];
struct compare{
  bool operator()(const int& a,const int& b)const{return((step[a]!=step[b])?step[a]<step[b]:a<b);}
};
set<int,compare> U;
bool OK=true;
vector<pii> ans;
vector<int> add;
int main(){
  int i,j;
  scanf("%d%d",&n,&s);
  for(int i=0;i<n;++i)scanf("%d",&step[i]);
  for(int i=0;i<n;++i)U.insert(i);
  while(!U.empty() && OK){
   vector<int> T;
   int V=*(--U.end());
   U.erase(V);
   if(step[V]>U.size()){OK=false; break;}
   add.clear();
   while(step[V]--){
    int GO=*(--U.end()); U.erase(GO);
    if(step[GO]==0){OK=false; break;}
    --step[GO]; ans.push_back(pii(V+1,GO+1));
    add.push_back(GO);
                   }
   for(int i=0;i<add.size();++i)U.insert(add[i]);
                         }
  if(!OK)printf("No\n");
  else{
   printf("Yes\n%d\n",ans.size());
   for(int i=0;i<ans.size();++i)printf("%d %d\n",ans[i].first,ans[i].second);
      }
  //system("pause");
  return 0;
}
