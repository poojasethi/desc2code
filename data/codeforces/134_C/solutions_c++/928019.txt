#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <queue>
#include <vector>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define mp make_pair
#define f first
#define s second
#define st pair <int,int>
#define lim 200001

int n,s,i,z;
st x,y;
priority_queue <st> Q;
vector <st> V;
int X[lim],Y[lim];

int main()
{
  scanf("%d%d",&n,&s);
  for(i=0;i<n;i++)
  {
    scanf("%d",&z);  
    Q.push(mp(z,i+1));
  }
  z=0;
  while(!Q.empty())
  {
    x=Q.top();
    Q.pop();
	V.clear();
	while(x.f && !Q.empty())
	{
	  y=Q.top();
	  if(!y.f)
	    break;
	  X[z]=x.s;
	  Y[z++]=y.s;
	  y.f--;
	  V.push_back(y);
	  Q.pop();
	  x.f--;
	}
	if(x.f)
	{
	  printf("No");
	  return 0;
	}
	for(i=0;i<V.size();i++)
	  Q.push(V[i]);
  }
  printf("Yes\n%d\n",z);
  for(i=0;i<z;i++)
    printf("%d %d\n",X[i],Y[i]);
  return 0;
}
