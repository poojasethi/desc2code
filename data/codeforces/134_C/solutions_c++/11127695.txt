#include<iostream>
#include<cstdio>
#include<queue>
using namespace std;
#define N 200009              //怎么老喜欢给它加分号结尾呢
typedef pair <int ,int > pai;
pai ans[N];
int main()
{
    int n,s,i,a;
    int num=0;

    cin>>n>>s;

    priority_queue<pai>q,t;

    for(i=1;i<=n;i++)
    {
        scanf("%d",&a);
        q.push(make_pair(a,i));            //构成一对
    }
    while(!q.empty())
    {
        pai p=q.top();  q.pop();           //和栈一样是top，而不是front‘

        while(p.first!=0)
        {
            if(q.empty())
            {
                puts("No");
                return 0;
            }
            pai v=q.top(); q.pop();

            ans[++num]=make_pair(p.second,v.second);

            p.first--;
            v.first--;

            if(v.first)
                t.push(v);

        }
       while(!t.empty())
        {
            q.push(t.top());
            t.pop();
        }
    }
    puts("Yes");

    printf("%d\n",num);

    for(i=1;i<=num;i++)
        printf("%d %d\n",ans[i].first,ans[i].second);

    return 0;
}
