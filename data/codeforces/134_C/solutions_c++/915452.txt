#include <cstdio>
#include <cstring>
#include <queue>
#include <algorithm>
#define N 200005
using namespace std;

struct node{
	int s, p;
	node(){}
	node(int s, int p) : s(s), p(p){}
	bool operator < (node const &T) const{return s < T.s;}
};

int n, x, ok, cnt, rex[N], rey[N];
priority_queue<node> q;
node tmp[N], T;

int main(){
	scanf("%d%*d", &n);
	for (int i=0; i<n; i++){
		scanf("%d", &x);
		if (x) q.push(node(x, i+1));
	}
	
	ok = 1;
	while (!q.empty() && ok){
		T = q.top();
		q.pop();
		x = T.s;
		if (x > q.size()) ok = 0;
		else{
			for (int j=0; j<x; j++){
				tmp[j] = q.top();
				q.pop();
			}
			for (int j=0; j<x; j++){
				tmp[j].s--;
				if (tmp[j].s) q.push(tmp[j]);
				rex[cnt] = T.p;
				rey[cnt++] = tmp[j].p;
			}
		}
	}
	
	if (!ok) printf("No\n");
	else{
		printf("Yes\n");
		printf("%d\n", cnt);
		for (int i=0; i<cnt; i++) printf("%d %d\n", rex[i], rey[i]);
	}
	
	return 0;
}
