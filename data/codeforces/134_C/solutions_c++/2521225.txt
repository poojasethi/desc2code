#include<iostream>
#include<cstdio>
#include<cstring>
#include<set>
#include<vector>
#define rep(i,a,b) for (int i=a;i<=b;i++)
#define fi first
#define se second
using namespace std;
typedef pair<int,int> PII;
#define N 200010
int n,a[N],sum;
set<PII> b;
vector<PII> ans;
int main()
{
	   scanf("%d%d",&n,&sum);
	   rep(i,1,n){
	        scanf("%d",&a[i]);
	        b.insert(make_pair(a[i],i));
	   }
	   if (sum&1)printf("No\n");
	   else{
	   	    for(int i=n;i>=1;i--){
                PII x=*b.rbegin();b.erase(x);
               if(x.fi>=i||x.fi<0){printf("No\n");return 0;}
               vector<PII> c;
               for(int j=0;j<x.fi;j++)
               {
                  PII y=*b.rbegin();b.erase(y),
                  y.fi--,c.push_back(y),ans.push_back(make_pair(x.se,y.se));
                }
                for(int j=0;j<(int)c.size();j++)b.insert(c[j]);
	   	    }
	   	    printf("Yes\n");
	       printf("%d\n",(int)ans.size());
           for(int i=0;i<(int)ans.size();i++)
           printf("%d %d\n",ans[i].fi,ans[i].se);
		}
}
