#include<iostream>
#include<vector>
#include<set>

using namespace std;

typedef pair< int,int >pie;

const int maxn=200000;

int main()
{
	int n,s;
	cin>>n>>s;
	set< pie >se;
	for(int i=0; i<n; i++)
	{
		int a;
		cin>>a;
		se.insert(make_pair(a,i+1));
	}
	vector< pie >vex;
	while(se.size()>0)
	{
		vector< pie >vec;
		set<pie>::iterator it=se.end();
		it--;
		int x=it->first, y=it->second;
		if(x>=se.size())
		{
			cout<<"No"<<endl;
			return 0;
		}
		for(int i=0; i<x; i++)
		{
			it--;		
			if(it->first<1)
			{
				cout<<"No"<<endl;
				return 0;
			}
			vec.push_back(make_pair(it->first-1,it->second));	
		}
		se.erase(it,se.end());
		for(int i=0; i<vec.size(); i++)
		{
			se.insert(vec[i]);
			vex.push_back(make_pair(y,vec[i].second));
		}
	}
	cout<<"Yes"<<endl<<vex.size()<<endl;
	for(int i=0; i<vex.size(); i++)
		cout<<vex[i].first<<" "<<vex[i].second<<endl;
	return 0;
}
