#include <iostream>
#include <stdio.h>
#include <string>

#include <queue>
#include <cmath>
#define N 222222
#define ll __int64

using namespace std;

typedef pair <int ,int > pp;
pp ans[N];

int a,b;

int main()
{
	int n,s,i,x;
	int num=0;

	scanf("%d %d",&n,&s);

	priority_queue<pp>q,t;

	for(i=1;i<=n;i++)
	{
		scanf("%d",&x);
		q.push(make_pair(x,i));
	}

	while(!q.empty())
	{
		pp p=q.top();  q.pop();

		while(p.first!=0)
		{
			if(q.empty())
			{
				cout<<"No"<<endl;
				return 0;
			}
			pp v=q.top(); q.pop();

			ans[++num]=make_pair(p.second,v.second);

			p.first--;
			v.first--;

			if(v.first)
				t.push(v);

		}
	   while(!t.empty())
        {
            q.push(t.top());
            t.pop();
        }
	}

	cout<<"Yes"<<endl;

	printf("%d\n",num);

	for(i=1;i<=num;i++)
		printf("%d %d\n",ans[i].first,ans[i].second);

	return 0;
}
