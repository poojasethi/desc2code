#include<iostream>
#include<set>
#include<vector>
#include<map>
#define pb push_back
using namespace std;
const int maxn=300000;
set<pair<int,int>,greater<pair<int,int> > > st;
vector<int> adj[maxn];
vector<pair<int,int> > edge;
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++){
		int x;
		cin>>x;
		st.insert(make_pair(x,i));
	}
	for(int i=0;i<n;i++){
		if(st.begin()->first==0)
			break;
		vector<pair<int,int> > v;
		int d=st.begin()->first,x=st.begin()->second;
		st.erase(make_pair(d,x));
		if(d>st.size()){
			cout<<"No"<<endl;
			return 0;
		}
		for(int j=0;j<d;j++){
			v.pb(*st.begin());
			st.erase(*st.begin());
		}
		for(int j=0;j<d;j++){
			if(v[j].first==0){
				cout<<"No"<<endl;
				return 0;
			}
			edge.pb(make_pair(x,v[j].second));
			v[j].first--;
		}
		for(int j=0;j<v.size();j++)
			st.insert(v[j]);
	}
	cout<<"Yes"<<endl;
	cout<<m/2<<endl;
	for(int i=0;i<m/2;i++)
		cout<<(edge[i].first)+1<<' '<<(edge[i].second)+1<<endl;
	return 0;
}
