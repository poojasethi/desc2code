#include <cstdio>
#include <algorithm>
#include <queue>

using namespace std;

typedef pair<int,int> pnt;
typedef priority_queue<pnt> qnt;

qnt q,r;
pnt t[1000005];
pnt out[1000005];
int no;

int main()
{
    int n,s;
    int i;
    int a,b;
    pnt j,k;
    scanf("%d %d",&n,&s);
    for(i=0;i<n;i++) {
        scanf("%d",&a);
        q.push(pnt(a,i+1));
    }
    while(!q.empty()) {
        j = q.top(); q.pop();
        while(j.first!=0) {
            if(q.empty()) {
                printf("No\n");
                return 0;
            }
            k = q.top(); q.pop();
            out[no++] = pnt(j.second,k.second);
            k.first--;
            j.first--;
            if(k.first) r.push(k);
        }
        while(!r.empty()) {
            q.push(r.top());
            r.pop();
        }
    }
    printf("Yes\n%d\n",no);
    for(i=0;i<no;i++)
        printf("%d %d\n",out[i].first,out[i].second);
    return 0;
}
