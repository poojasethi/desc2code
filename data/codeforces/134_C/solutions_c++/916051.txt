#include<cstdio>
#include<queue>
using namespace std;
#define maxn 200005

struct node{
    int id,num;
    
    node(){}
    node(int id,int num):id(id),num(num){}
    bool operator<(const node &d)const{return num<d.num;}
} a[maxn];

int n,s,num,a1[maxn],a2[maxn];
priority_queue<node> q;

int main(){
    num=0;
    scanf("%d%d",&n,&s);
    for (int i=0;i<n;++i){
        int x;
        scanf("%d",&x);
        if (x) q.push(node(i,x));
    }
    
    bool ok=1;
    while (ok && !q.empty()){
        node tmp=q.top(); q.pop();
        if (tmp.num>q.size()) ok=0;
        else{
            int pos=0;
            for (int i=0;i<tmp.num;++i){
                a[pos++]=q.top(); q.pop();
            }
            for (int i=0;i<pos;++i){
                a1[num]=tmp.id; a2[num]=a[i].id; ++num;
                --a[i].num;
                if (a[i].num) q.push(a[i]);
            }
        }
    }
    if (!ok) printf("No\n");
    else{
        printf("Yes\n%d\n",num);
        for (int i=0;i<num;++i) printf("%d %d\n",a1[i]+1,a2[i]+1);
    }
    return 0;
}
