#include <cstdio>
#include <queue>
using namespace std;

struct node {
	int a, i;
	bool operator < (const node &t) const {
		return a < t.a;
	}
}temp, q[200005];
priority_queue<node> pq;

int n, s, a[200005], sol[200005][2], ans;

void adds(int x, int y) {
	sol[ans][0] = x;
	sol[ans][1] = y;
	ans++;
}

bool solve() {
	ans = 0;
	for (int i=1; i<=n; ++i) {
		temp = pq.top();
		pq.pop();
		if (temp.a == 0) return true;
		for (int j=0; j<temp.a; ++j) {
			if (pq.empty()) return false;
			q[j] = pq.top();
			pq.pop();
			if (q[j].a == 0) return false;
			q[j].a--;
			adds(temp.i, q[j].i);
		}
		for (int j=0; j<temp.a; ++j) {
			pq.push(q[j]);
		}
	}
	return false;
}

int main() {
	scanf("%d%d\n", &n, &s);
	for (int i=1; i<=n; ++i) {
		scanf("%d", &a[i]);
		temp.a = a[i];
		temp.i = i;
		pq.push(temp);
	}
	if (solve()) {
		printf("Yes\n%d\n", ans);
		for (int i=0; i<ans; ++i) {
			printf("%d %d\n", sol[i][0], sol[i][1]);
		}
	}
	else printf("No\n");
	return 0;
}
