#include <bits/stdc++.h>
using namespace std;
const int N = 200005;
int n,s;
typedef pair<int,int> pii;
#define fi first
#define se second
#define mp make_pair
int ar[N];
priority_queue<pii> sem, pq;
pii temport[N];
int main()
{
	scanf("%d%d",&n, &s);
	if(s&1)
	{
		puts("No");
		return 0;
	}
	int maks1=-1, maks2=-1;
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&ar[i]);
		if(ar[i]>maks1)
		{
			maks2 = maks1;
			maks1 = ar[i];
		}	
		else if(ar[i] > maks2)
		{
			maks2 = ar[i];
		}
	}
	for(int i=1;i<=n;i++) if(ar[i]) pq.push(mp(ar[i], i));
	int nows = 0;
	while(!pq.empty())
	{
		pii temp = pq.top();pq.pop();
		for(int i=0;i<temp.fi;i++)
		{
			if(pq.empty())
			{
				puts("No");
				return 0;
			}
			pii now = pq.top();
			pq.pop();
			temport[nows] = mp(temp.se, now.se);
			nows++;
			if(now.fi!=1) sem.push(mp(now.fi-1,now.se));
		}
		while(!sem.empty())
		{
			pq.push(sem.top());
			sem.pop();
		}
	}
	puts("Yes");
	printf("%d\n", s>>1);
	for(int i=0;i<s>>1;i++)
	{
		printf("%d %d\n", temport[i].fi, temport[i].se);
	}
	return 0;
}
