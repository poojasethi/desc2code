#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <vector>
#include <queue>
#include <stack>
using namespace std;

const int N = 200005;
#define mp make_pair

vector<pair<int, int> > vi; //ans
priority_queue<pair<int, int> > Q;
stack<pair<int, int> > st;

int main() {
	//freopen("in.txt", "r", stdin);
	//freopen("out.txt", "w", stdout);

	int n, S;
	cin >> n >> S;

	for (int i = 0; i < n; i++) {
		int a;
		scanf("%d", &a);
		if (a > 0) {
			Q.push(mp(a, i + 1));
		}
	}
	while (!Q.empty()) {
		pair<int, int> now = Q.top();
		Q.pop();
		while (!Q.empty() && now.first > 0) {
			pair<int, int> tmp = Q.top();
			Q.pop();
			now.first--;
			tmp.first--;
			vi.push_back(mp(now.second, tmp.second));
			if (tmp.first > 0) {
				st.push(tmp);
			}
		}
		if (now.first > 0) {
			printf("No\n");
			return 0;
		}
		while (!st.empty()) {
			Q.push(st.top());
			st.pop();
		}
	}
	printf("Yes\n%d\n", vi.size());
	for (int i = 0; i < vi.size(); i++) {
		printf("%d %d\n", vi[i].first, vi[i].second);
	}
}
