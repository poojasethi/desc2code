#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<cmath>
#include<queue>
#include<vector>
using namespace std;

#define mp make_pair
#define N 200010
typedef pair<int,int> P;
priority_queue<P> q;
P tmp[N],ans[N];
int s,n,a[N],cnt;

bool solve(){
	if(s&1) return false;
	for(int i=1;i<=n;i++)
	if(a[i]) q.push(mp(a[i],i));
	while(!q.empty()){
		P x=q.top(); q.pop();
		for(int i=1;i<=x.first;i++){
			if(q.empty()) return false;
			tmp[i]=q.top(); q.pop();
			tmp[i].first--;
			ans[++cnt]=mp(x.second,tmp[i].second);
		}
		for(int i=1;i<=x.first;i++) 
		if(tmp[i].first) q.push(tmp[i]);
	}
	return true;
}

int main(){
	scanf("%d%d",&n,&s);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	if(!solve()) puts("No");
	else {
		puts("Yes");
		printf("%d\n",cnt);
		for(int i=1;i<=cnt;i++) printf("%d %d\n",ans[i].first,ans[i].second);
	}
	return 0;
}
