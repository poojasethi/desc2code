#include<cstdio>
#include<queue>
using namespace std;
#define mp make_pair
typedef pair<int,int> pi;
priority_queue<pi> q;
const int N=200005;
pi t[N],ans[N];
int n,s,x,tot,num;
#define pause {puts("No");return 0;}
int main(){
	scanf("%d%d",&n,&s);
	for (int i=1;i<=n;i++) scanf("%d",&x),q.push(mp(x,i));
	if (s&1) pause;
	while (!q.empty()){
		pi p=q.top();q.pop();
		tot=0;
		while (p.first!=0){
			if (q.empty()) pause;
			pi v=q.top();q.pop();
			v.first--;
			p.first--;
			ans[++num]=mp(p.second,v.second);
			if (v.first) t[++tot]=v;
		}
		for (int i=1;i<=tot;i++) q.push(t[i]);
	}
	puts("Yes");
	printf("%d\n",s/2);
	for (int i=1;i<=num;i++) printf("%d %d\n",ans[i].first,ans[i].second);
	return 0;
}
