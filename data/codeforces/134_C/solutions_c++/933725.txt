#include <iostream>
#include <vector>
#include <cstdio>
#include <memory.h>
#include <string>
#include <queue>
#include <algorithm>
#define P pair <int, int>
#define MAX 5001
#define x first
#define y second
#include <queue>
using namespace std;
vector <P> v, m;
int main(){
	int n, s, a;
	cin >> n >> s;
	priority_queue <P> Q;
	for (int i = 0;i < n;++i){
		cin >> a;
		if (a) Q.push(P(a, i));
	}
	bool ok = true;
	while (!Q.empty() && ok){
		int j = Q.top().x,
			id = Q.top().y;
		Q.pop();
		vector <P> v; 
		for (int k = 0;k < j;++k){
			if (Q.empty()) {
				ok = false;
				break;
			}
			P tmp = Q.top();
			Q.pop();
			tmp.x--;
			m.push_back(P(id, tmp.y));
			if (tmp.x) v.push_back(tmp);
		}
		for (int i = 0;i < v.size();++i)
			Q.push(v[i]);
	}
	if (ok){
		cout << "Yes" << endl;
		cout << m.size() << endl;
		for (int i = 0;i < m.size();++i)
			cout << m[i].x + 1 << " " << m[i].y + 1 << endl;
	}
	else cout << "No" << endl;
	return 0;
}
