#include <iostream>
#include <set>
#include <vector>
using namespace std;
typedef pair<int, int> pii;

set<pii> opt;
int main() {
	int n, s;
	cin >> n >> s;
	for (int i = 0; i < n; ++i) {
		int t;
		cin >> t;
		if (t) {
			opt.insert(pii(t, i + 1));
		}
	}
	vector<pii> mvs;
	while (opt.size()) {
		pii t = *opt.rbegin();
		opt.erase(--opt.end());
		if (opt.size() < t.first) {
			cout << "No\n";
			return 0;
		}
		vector<pii> ret;
		for (int i = 0; i < t.first; ++i) {
			pii t2 = *opt.rbegin();
			opt.erase(--opt.end());
			mvs.push_back(pii(t.second, t2.second));
			if (t2.first > 1) {
				t2.first--;
				ret.push_back(t2);
			}
		}
		for (int i = 0; i < ret.size(); ++i) {
			opt.insert(ret[i]);
		}
	}
	cout << "Yes\n";
	cout << mvs.size() << '\n';
	for (int i = 0; i < mvs.size(); ++i) {
		cout << mvs[i].first << ' ' << mvs[i].second << '\n';
	}
}
