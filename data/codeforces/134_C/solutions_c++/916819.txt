#include<iostream>
#include<queue>
#include<algorithm>
#include<stdio.h>
#include<vector>
using namespace std;

int main(){
    priority_queue<pair<int,int> > q;
    int n,s;
    cin >> n >> s;
    for(int i=0;i<n;i++){
        int k;
        cin>>k;
        q.push(make_pair(k,i+1));
    }
    vector<pair<int,int> >ans;
    vector<pair<int,int> >v;
    while(!q.empty()){

        pair<int,int> p=q.top();
        q.pop();
        v.clear();
        for(int i=0;i<p.first;i++){
            if(q.empty()){
                printf("No\n");
                return 0;
            }
            pair<int,int> p2=q.top();
            q.pop();
            if(p2.first==0){
                printf("No\n");
                return 0;
            }
            ans.push_back(make_pair(p.second,p2.second));
            v.push_back(make_pair(p2.first-1,p2.second));
        }
        for(int i=0;i<v.size();i++)q.push(v[i]);
    }
    printf("Yes\n");
    printf("%d\n",ans.size());
    for(int i=0;i<ans.size();i++){
        printf("%d %d\n",ans[i].first,ans[i].second);
    }
}
