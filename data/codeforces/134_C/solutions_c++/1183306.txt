#include <cstdio>
#include <utility>
#include <cstring>
#include <vector>
using namespace std;
const int maxn = 202020;
vector<int> a[maxn];
vector<pair<int, int> > ans, tmp;
int n, s, num = 0, tar = 0;
int main() {
	scanf("%d%d", &n, &s);
	num = n;
	for (int i = 1; i <= n; ++i) {
		int x; scanf("%d", &x);
		if (x == 0) --num; 
			else a[x].push_back(i);
		tar += x;
	}
	for (int i = s; i > 0; ) if (!a[i].empty()) {
		if (--num < i) break;
		int here = a[i].back(); a[i].pop_back();
		for (int cnt = i, j = i; cnt > 0; -- cnt) {
			while (a[j].empty()) --j;
			int k = a[j].back();
			a[j].pop_back();
			ans.push_back(pair<int,int>(k, here));
			if (j > 1) tmp.push_back(pair<int,int>(j-1,k));
				else --num;
		}
		while (!tmp.empty()) {
			pair<int,int> t = tmp.back();
			tmp.pop_back();
			a[t.first].push_back(t.second);
		}	
	} else --i;
	if ((int)ans.size() * 2 == tar) {
		printf("Yes\n%d\n", tar / 2);
		for (size_t i = 0; i < ans.size(); ++i) 
			printf("%d %d\n", ans[i].first, ans[i].second);
	} else puts("No");
	return 0;
}
