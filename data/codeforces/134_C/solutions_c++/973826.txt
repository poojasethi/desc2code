#include<queue>
#include<cstdio>
#include<iostream>
#define fo(i,a,b) for(i=a;i<=b;++i)
using namespace std;
const int mn=201000;
struct Node{int p,d;}a[mn],x;
int ans[mn][2],n,i,s,d,len=0;
priority_queue<Node> q;
bool operator<(Node a,Node b)
{
    return a.d<b.d;
}
int main()
{
    scanf("%d%d",&n,&s);
    fo(i,1,n)
        if(scanf("%d",&d))
        {
            x.p=i,x.d=d;
            q.push(x);
        }
    while(!q.empty())
    {
        x=q.top(),q.pop();
        if(x.d>q.size())
        {
            printf("No\n");
            return 0;
        }
        fo(i,1,x.d)a[i]=q.top(),q.pop();
        fo(i,1,x.d)
        {
            ans[++len][0]=x.p,ans[len][1]=a[i].p;
            if(--a[i].d)q.push(a[i]);
        }
    }
    printf("Yes\n%d\n",len);
    fo(i,1,len)printf("%d %d\n",ans[i][0],ans[i][1]);
    cin.get();
    cin.get();
    return 0;
}
