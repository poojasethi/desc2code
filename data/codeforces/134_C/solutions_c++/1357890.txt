#include<iostream>
#include<queue>
#include<vector>
using namespace std;
#define PII pair<int,int>
#define F first
#define S second

const int MAX = 200*1000;
PII d[MAX+10];
PII out[MAX+10];
int head = 0;

int main()
{
	ios::sync_with_stdio(false);
	int n,s;
	cin >> n >> s;
	priority_queue<PII> heap;
	for(int i=0; i<n; i++)
	{
		cin >> d[i].F;
		d[i].S = i+1;
		if( d[i].F > 0 )
			heap.push(d[i]);
	}
	for(int i=0; i<n && !heap.empty() ; i++)
	{
		PII big = heap.top();
		heap.pop();
		int head2 = 0;
		for(int j=0; j<big.F; j++)
		{
			if(heap.empty())
			{
				cout << "No" << endl;
				return 0;
			}
			PII c = heap.top();
			heap.pop();
			c.F--;
			out[head].F = big.S;
			out[head++].S = c.S;
			if( c.F )
				d[head2++] = c;
		}
		for(int j=0; j<head2; j++)
			heap.push(d[j]);
	}
	cout << "Yes" << endl << head << endl;
	for(int i=0; i<head; i++)
		cout << out[i].F << " " << out[i].S << endl;
	return 0;
}
