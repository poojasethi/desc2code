#include <iostream>
#include <algorithm>
using namespace std;

const int MAXN = 1000 + 10;
pair <int, int> p[MAXN];

int main()
{
	int n;
	cin>> n;

	for(int i = 0; i < n; i ++)
	{
		cin>> p[i].first;
		p[i].second = i;
	}

	sort(p, p + n);

	cout<< p[n - 1].second + 1 << " " << p[n - 2].first << endl;
}
