def enBuyuk(L):
    enBuyuk=L[0]
    index=0
    for i in range(1, len(L)):
        if L[i]>enBuyuk:
            enBuyuk=L[i]
            index=i
    return index


n=raw_input()
n=int(n)
L=raw_input()
L=L.split(" ")
for i in range(0, n):
    L[i]=int(L[i])

index=enBuyuk(L)
R=L[:index]+L[index+1:]
miktar=R[enBuyuk(R)]
print index+1, miktar
