n = input()
prices = [int(i) for i in raw_input().split(" ")]
winer = 0
second_price = 0

for i in xrange(n):
    if(prices[i] > prices[winer]):
        winer = i

for i in xrange(n):
    if(i != winer):
        if(prices[i] > second_price):
            second_price = prices[i]

print winer + 1, second_price
