#A second-price auction codeforces.com problem
#!/usr/bin/python

x=int(input())
auctions=map(int,raw_input().split())

def second_price(auctions):
    x1=auctions.index(max(auctions))+1
    auctions.remove(max(auctions))
    x2=max(auctions)
    return str(x1)+" "+str(x2)

print second_price(auctions)
