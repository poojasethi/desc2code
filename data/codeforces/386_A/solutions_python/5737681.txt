I=lambda:map(int, raw_input().split())
n = input()
a = I()
if a[0] > a[1]: 
    x, y, xi, yi = a[0], a[1], 0, 1
else:
    x, y, xi, yi = a[1], a[0], 1, 0
    
for i in xrange(2, n):
    if a[i] >= x: 
        x, xi, y, yi = a[i], i, x, xi
    elif a[i] > y:
        y, yi = a[i], i
print xi+1, y
