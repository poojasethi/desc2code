#!/usr/bin/env python

import sys

(N, a, b, c) = map(int, sys.stdin.readline().rstrip('\n').split(' '))

a = a / 2

if N > a + b + c * 2:
    print 0
    exit(0)

x = a + b + c

A = [0] * 10010

for i in xrange (0, a + b + 1):
    A[i] = min(a, i) + min(b, i) - i + 1

sol = 0

for i in xrange(0, N + 1):
    if (N - i) % 2 == 0 and (N - i) / 2 <= c:
        sol += A[i]

print sol
