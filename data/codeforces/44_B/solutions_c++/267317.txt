#include <iostream>

using namespace std;


int main(){
  int n,a,b,c;cin >> n >> a >> b >>c;
  int ans(0);
  for(int i = 0;i <= c;i++){
    for(int j = 0;j <= b;j++){
      int remain;
      remain = n -  2*i - j;
      if(remain < 0) continue;
      if(remain * 2 <= a)  ans++;
    }
  }
  cout << ans << endl;
}
