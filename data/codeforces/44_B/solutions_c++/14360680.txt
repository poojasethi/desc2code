#include <bits/stdc++.h>
using namespace std;
int main ()
{
    int long long n,a,b,c,sum=0;
    cin>>n>>a>>b>>c;
    sum=0;
    for (int i=0;i<=a;i+=2){
        for (int j=0;j<=c;j++){
            int d=n-i/2-2*j;
            if (d<0 || d>b) {continue;}
            sum++;
        }
    }
    cout<<sum<<endl;
}
