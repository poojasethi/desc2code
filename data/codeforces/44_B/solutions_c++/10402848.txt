#include<iostream>
using namespace std;
int main()
{
	int n, a, b, c;
	while (cin >> n >> a >> b >> c)
	{
		int ans = 0;
		for (int i = 0; i <= c;i++)
		for (int j = 0; j <= b; j++)
		{
			int temp = n - i * 2 - j;
			if (temp >= 0 && a * 0.5 >= temp)
				ans++;
		}
		cout << ans << endl;
	}
}
