#include<stdio.h>
#include<string.h>

int main()
{
	int n,a,b,c,i,j;
	while(scanf("%d%d%d%d",&n,&a,&b,&c)!=EOF)
	{
		int count=0;
		for(i=0;i<=a;i++)
		{
			for(j=0;j<=b;j++)
			{
				int tt=2*n-i-2*j;
				if(tt%4==0&&tt>=0&&tt<=4*c)
					count++;
			}
		}
		printf("%d\n",count);
	}
	return 0;
}
