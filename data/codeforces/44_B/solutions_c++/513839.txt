#include<iostream>
using namespace std;
int main()
{
	int n,a,b,c,x,y,z,sum = 0;
	cin>>n>>a>>b>>c;
	n = 2*n;
	b = 2*b;
	c = 4*c;
	for (z = 0; z<=c; z+=4)
	{
		if (z> n)break;
		for (y = 0; y<=b; y+=2)
		{
			if (z+y>n)break;
			if (z+y+a>=n)sum++;
		}
	}
	cout<<sum<<endl;

}
