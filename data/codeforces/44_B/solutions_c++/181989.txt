#include <iostream>

using namespace std;

int n, a, b, c;

int main()
{
	cin>>n>>a>>b>>c;
	n*=2;

	int s=0;
	for (int i=0; i<=a && i<=n; i++)
		for (int j=0; j<=b && i+j*2<=n; j++)
		{
			int k=n-i-2*j;
			if (k%4)
				continue;
			k/=4;

			//cout << i<<j<<k<<endl;
			s+=(k<=c);
		}

	cout << s<<endl;

	return 0;
}
