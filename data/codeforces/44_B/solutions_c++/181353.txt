#include <iostream>
using namespace std;

int main()
{
	int n, a, b, c;
	cin >> n >> a >> b >> c;
	n *= 2;

	int answer = 0;
	for(int cc=0; cc<=c; ++cc)
		if( 4*cc<=n && n<=1*a+2*b+4*cc )
			for(int bb=0; bb<=b; ++bb)
				if( 2*bb+4*cc<=n && n<=1*a+2*bb+4*cc )
					++answer;
	cout << answer << endl;
}
