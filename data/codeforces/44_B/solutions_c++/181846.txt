#include <cstdio>

int main()
{
	int n, a, b, c;
	scanf("%i%i%i%i", &n, &a, &b, &c);
	n *= 2;
	int res = 0;
	for(int i = 0; i <= a; i++) {
		for(int j = 0; j <= b; j++) {
			if(i + 2 * j > n) break;
			int k = n - (i + 2 * j);
			if(k % 4 == 0 && k / 4 <= c) {
				res++;
			}
		}
	}
	printf("%i\n", res);
}
