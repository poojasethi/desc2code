#include <stdio.h>
int main(){
	int n,a,b,c;
	int i,j;
	int sum;
	while(scanf("%d%d%d%d",&n,&a,&b,&c)!=EOF){
		sum=0;
		n*=2;
		for(i=0;i<=c;i++){
			if(4*i>n)break;
			for(j=0;j<=b;j++){
				if(i*4+j*2>n)break;
				if(i*4+j*2+a>=n)sum++;
			}
		}
		printf("%d\n",sum);
	}
	return 0;
}
