#include <cstdio>
using namespace std;
int main() {
  int n, a, b, c, ans = 0, i, j;
  ans = 0;
  scanf("%d%d%d%d", &n, &a, &b, &c);
  for (i = 0; i <= c; ++i)
    for (j = 0; j <= b; ++j)
      if (i + i + j <= n && (n - i - i - j) * 2 <= a) ++ans; 
  //???
  //PROFIT!!!
  printf("%d\n", ans);
  return 0;
}
