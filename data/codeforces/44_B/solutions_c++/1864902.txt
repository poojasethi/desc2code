#include <cstdio>
#include <algorithm>

using namespace std;

int n, a, b, c, cnt;

int main() {
	scanf("%i%i%i%i", &n, &a, &b, &c);
	for(int i = min(n / 2, c); i >= 0; i--) {
		int m = n - 2 * i;
		for(int j = min(m, b); j >= 0; j--) {
			int k = m - j;
			if(2 * k <= a) cnt++;
		}
	}
	printf("%i\n", cnt);
}
