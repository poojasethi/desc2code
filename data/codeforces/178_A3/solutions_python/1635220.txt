#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
long long a[100005],ans=0;
int main()
{
	int n; scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<n;i++){
		int j; for (j=0;;j++) if (i+(1<<j)>n) break;
		j--; ans+=a[i]; a[i+(1<<j)]+=a[i];
		cout<<ans<<endl;
		}
	return 0;
}
