#include<stdio.h>
int num[20]={1};
__int64 a[100005]={0};
int main()
{
    int i,n,j,t;
    __int64 ans=0;
    for(i=1;i<20;i++)
    num[i]=num[i-1]<<1;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        scanf("%d",&t);
        a[i]+=t;
        if(i==n)break;
        ans=ans+a[i];
        printf("%I64d\n",ans);
        for(j=0;num[j]+i<=n;j++);
        a[num[j-1]+i]+=a[i];
    }

    return 0;
}
