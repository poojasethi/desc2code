#include <iostream>
#include <cstdio>

using namespace std;
const int N = 100009;

long long n, i, j, s, a[N];

int main () {
	cin >> n;
	for (i = 1; i <= n; i++) 
		cin >> a[i];
	for (i = 1; i < n; i++) {
		for (j = 0; i + (1 << j) <= n; j++);
		j--;
		a[i + (1 << j)] += a[i];
		s += a[i];
		cout << s << endl;                 
	}
}
