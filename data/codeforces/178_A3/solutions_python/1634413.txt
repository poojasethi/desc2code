#include <cstdio>
#include <algorithm>
using namespace std;

int n,tab[100000];
long long res;
int main() {
  scanf("%d", &n);
  for(int i=0; i<n; ++i) scanf("%d", tab+i);
  for(int i=0; i+1<n; ++i) {
    res+=tab[i];
    int r=1;
    while(i+2*r<n) r*=2;
    tab[i+r]+=tab[i];
    printf("%I64d\n", res); 
  }
}
