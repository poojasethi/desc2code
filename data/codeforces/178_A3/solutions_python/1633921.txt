#include <cstdio>
#include <cstring>

const int MAXN = 100001;

int n;
long long a[MAXN];

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%I64d", &a[i]);
	}
	for (int i = 1; i <= n; ++i) {
		int t;
		for (t = 1; i + t <= n; t <<= 1);
		t >>= 1;
		a[i + t] += a[i];
	}
	for (int i = 2; i <= n; ++i) {
		a[i] += a[i - 1];
	}
	for (int i = 1; i < n; ++i) {
		printf("%I64d\n", a[i]);
	}
	return 0;
}
