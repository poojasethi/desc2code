#include<stdio.h>
#include<iostream>
using namespace std;
const int lmax=100005;
typedef __int64 i64;

int main()
{
 int n,a[lmax],i,j;
 i64 sum;
 while(scanf("%d",&n)!=EOF)
 {
  for(i=0;i<n;i++) scanf("%d",a+i);
  for(sum=i=0;i<n-1;i++)
  {
   int x=1;
   while(1)
   {
    if(i+2*x>=n) break;
    x*=2;
   }
   sum+=a[i];
   a[i+x]+=a[i];
   a[i]=0;
   printf("%I64d\n",sum);
  }
 }
}
