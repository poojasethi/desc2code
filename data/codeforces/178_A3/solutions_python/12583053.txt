#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int a[100005];

int main()
{
	int n,i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	ll ans = 0;
	for(i=1;i<n;i++)
	{
		int j;
		for(j=20;j>=0;j--)
		{
			if(i + (1<<j)<=n)break;
		}
		ans+=a[i];
	//	printf("%d\n",ans);
	cout<<ans<<endl;
		a[i+(1<<j)]+=a[i];
	}
	
	}
