#include<stdio.h>
typedef long long int lnt;
lnt a[100100],n;
lnt maxbit(lnt k){
	lnt ans=1;
	for(;ans<=k;ans*=2);
	return ans/2;
	}
int main(){
	scanf("%I64d",&n);
	for(int i=1;i<=n;i++)
		scanf("%I64d",a+i);
	lnt ans=0;
	for(int i=1;i<n;i++){
		lnt p=maxbit(n-i);
		ans+=a[i];
		a[i+p]+=a[i];
		printf("%I64d\n",ans);
		}
	//while(getchar());
	return 0;
	}
