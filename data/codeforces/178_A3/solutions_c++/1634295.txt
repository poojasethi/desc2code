#include<stdio.h>
int a[1000000];
int main(){
    int n;
    scanf("%d",&n);
    for(int i = 0 ; i < n ; i++ ) scanf("%d",&a[i]);
    long long res = 0;
    for(int i = 0 ; i < n-1 ; i++ ){
        int pos = 1;
        for( ; i+pos < n ; pos<<=1);
        pos >>= 1;
        res += a[i];
        a[i+pos] += a[i];
        a[i] = 0;
        printf("%I64d\n",res);
    }
    //for(;;);
}
