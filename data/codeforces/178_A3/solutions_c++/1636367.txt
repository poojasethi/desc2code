#include <stdio.h>
main() {
  int n,a[100100],t[20]={1},k;
  long long s=0;
  scanf("%d",&n);
  for (int i=1; i<=n; i++)
    scanf("%d",&a[i]);
  for (int i=1; i<19; i++)
    t[i]=2*t[i-1];
  for (int i=1; i<n; i++) {
    k=0;
    while (i+t[k+1]<=n)
      k++;
    a[i+t[k]]+=a[i];
    s+=a[i];
    printf("%I64d\n",s);
  }
  return 0;
}
