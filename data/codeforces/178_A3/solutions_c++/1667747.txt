#include<cstdio>
#include<cstring>
#define ll long long

int main(){
	int n, v[100010];
	ll  s[100010];
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&v[i]);
	memset(s,0,sizeof(s));
	for(int i=0;i<n-1;i++){
		int j;
		for(j=1;i+j<n;j*=2);
		j/=2;
		s[i+j]+=v[i]+s[i];
	}
	ll r=0;
	for(int i=0;i<n-1;i++){
		r += (ll)v[i] + s[i];
		printf("%I64d\n",r);
	}
	return 0;
}
