#include <stdio.h>
#include <string.h>
#define N 100003
int main()
{
	long long n,a[N];
	scanf("%I64d",&n);
	for(int i=0;i<n;++i)
		scanf("%I64d",&a[i]);
	long long step=0;
	int t;
	for(int i=0;i<n-1;++i)
	{
		t=1<<17;
		if(a[i]!=0)
		{
			while(i+t>n-1)
				t>>=1;
			a[i+t]+=a[i];
			step+=a[i];
			a[i]=0;
		}
		printf("%I64d\n",step);
	}
	return 0;
}
