#include<iostream>
#include<algorithm>

using namespace std;

int a[100005];

int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> a[i];
	long long ans = 0;
	for(int i = 0; i < n - 1; i++)
	{
		int t = n - i - 1, p = 0;
		while((1 << p) <= t)
			p++;
		p--;
		a[i + (1 << p)] += a[i];
		ans += a[i];
		a[i] = 0;
		cout << ans << endl;
	}
	return 0;
}
