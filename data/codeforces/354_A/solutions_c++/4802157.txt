#include<stdio.h>
int n,l,r,Ql,Qr,w,sum[100001],En,Min=-1u/2;
main()
{
    int i;
    scanf("%d%d%d%d%d",&n,&l,&r,&Ql,&Qr);
    for (i=1;i<=n;i++)
    { scanf("%d",&w); sum[i]=sum[i-1]+w; }
    for (i=0;i<=n;i++) {
        En=sum[i]*l+(sum[n]-sum[i])*r;
        if (i > n-i)
            En+=Ql*(i-(n-i)-1);
        else if (i < n-i)
            En+=Qr*((n-i)-i-1);
        if (En<Min) Min=En;
    }
    printf("%d\n",Min);
}
