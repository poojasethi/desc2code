
 #include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,l,r,q1,q2,a[120000],res,ans,i,j;
int main(){
	ans=1<<30;
	scanf("%d%d%d%d%d",&n,&l,&r,&q1,&q2);a[0]=0;
	for(i=1;i<=n;++i)scanf("%d",&a[i]),a[i]+=a[i-1];
	for(i=0;i<=n;++i){
		j=n-i;
		res=a[i]*l+(a[n]-a[i])*r;

		if(i>j)res+=(i-j-1)*q1;
		if(i<j)res+=(j-i-1)*q2;
		ans=min(ans,res);
	}
	printf("%d\n",ans);
}
