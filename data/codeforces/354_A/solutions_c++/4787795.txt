#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

int a[111111],b[111111],t[111111];
int i,j,n,m,x;
int r,l,q1,q2;
int main()
{
   scanf("%d%d%d%d%d",&n,&l,&r,&q1,&q2);
   for (i=1;i<=n;i++)
   {
    scanf("%d",&a[i]);
    b[i]=b[i-1]+a[i];
   }
int ans=1e9;
for (i=0;i<=n;i++)
{
 int tp=b[i]*l+(b[n]-b[i])*r;
 if (i<n-i) tp+=(n-2*i-1)*q2;
 if (i>n-i) tp+=(2*i-n-1)*q1;
 ans=min(ans,tp);
}
printf("%d",ans);

   }
