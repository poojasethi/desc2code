#include <iostream>
#include <vector>
using namespace std;
int n,l,r,ql,qr;
vector<int> weight;
int main(){
    cin >> n >> l >> r >> ql >> qr;
    weight.resize(n+1);
    for(int i = 1; i <= n; i++){
        cin >> weight[i];
        if(i>0) weight[i]+=weight[i-1];
    }
    int cost = 2000000000;
    for(int i = 0; i <= n; i++){
        cost = min(cost,weight[i]*l+(weight[n]-weight[i])*r+max(0,2*i-n-1)*ql+max(0,n-2*i-1)*qr);
    }
    cout << cost << '\n';
}
