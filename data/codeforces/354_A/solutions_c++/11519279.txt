#include<bits/stdc++.h>
using namespace std;
int n,l,r,ql,qr,w[100003],ans=2e9;
int i,j,u=0;
main()
{
    scanf("%d%d%d%d%d",&n,&l,&r,&ql,&qr);
    for (i=1;i<=n;i++) scanf("%d",&w[i]),w[i]+=w[i-1];
    for (i=0;i<=n;i++)
    {
        int ens=w[i]*l+(w[n]-w[i])*r;
        if (i<n-i) ens+=qr*(n-2*i-1);
        else if (i>n-i) ens+=ql*(2*i-n-1);
        ans=min(ans,ens);
    }
    printf("%d\n",ans);
}
