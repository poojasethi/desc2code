#include <cstdio>
const int N=100000;
int w,sum[N],min=(1<<30),ans;
int n,l,r,ql,qr;
int main(){
	scanf("%d%d%d%d%d",&n,&l,&r,&ql,&qr);
	int i,j;
	sum[0]=0;
	for(i=1;i<=n;i++){
		scanf("%d",&w);
		sum[i]=sum[i-1]+w;
	}
	for(i=0;i<=n;i++){
		ans=sum[i]*l+(sum[n]-sum[i])*r;
		if(n-i-1==i||i-1==n-i||i==n-i)ans+=0;
		else if(n-i>i+1)ans+=(n-i-i-1)*qr;
		else if(n-i<i-1)ans+=(i-1-n+i)*ql;
		if(min>ans)min=ans;
	}
	printf("%d\n",min);
	return 0;
}
