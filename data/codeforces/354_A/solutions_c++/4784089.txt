#include<iostream>
#include <string.h>
#include<vector>
#include<list>
#include<cstdlib>
#include<fstream> 
using namespace std;
int main(){
	int n,r,l,Ql,Qr,i,hL=0,hR=0,sum,minS;
	cin>>n>>l>>r>>Ql>>Qr;
	int w[n];
	for(i=0;i<n;i++)
	cin>>w[i],hR+=w[i];
	for(i=0;i<=n;i++){
		sum=hL*l+hR*r;
		if(2*i<n) sum+=Qr*(n-2*i-1);
		if(2*i>n) sum+=Ql*(2*i-n-1);
		if(!i) minS=sum;
		else minS=min(minS,sum);
		hL+=w[i];
		hR-=w[i];
	}
	cout<<minS;
	return 0;
}
