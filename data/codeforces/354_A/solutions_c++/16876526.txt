#include<bits/stdc++.h>
#define ll long long int
using namespace std;
ll n,l,r,qr,ql,w[100100],suml[100100],sumr[100100],mn=1000000007,e;
int main(){
	cin>>n>>l>>r>>ql>>qr;
	for(int i=1;i<=n;i++)scanf("%I64d",&w[i]);
	for(int i=1;i<=n;i++)suml[i]=suml[i-1]+w[i]*l;
	for(int i=n;i;i--)sumr[i]=sumr[i+1]+w[i]*r;
	for(int i=0;i<=n;i++){
		e=suml[i]+sumr[i+1];
		if(i>n-i)e+=(2*i-n-1)*ql;
		else if(i<n-i)e+=(n-2*i-1)*qr;
		mn=min(mn,e);
	}
	cout<<mn<<endl;
}
