#include<stdio.h>
int main()
{
	__int64 n,l,r,ql,qr,sum,min,en,i,w[100005];
	while(scanf("%I64d%I64d%I64d%I64d%I64d",&n,&l,&r,&ql,&qr) != EOF)
	{
		w[0]=0;
		for(i=1;i<=n;i++)
		{
			scanf("%I64d",&en);
			w[i]=w[i-1]+en;
		}
		min=1000000000;
		for(i=0;i<=n;i++)
		{
			sum=w[i]*l+(w[n]-w[i])*r;
			if(i>n-i)
				sum+=ql*(2*i-n-1);
			else
				if(i<n-i)
					sum+=qr*(n-2*i-1);
			if(sum<min)
				min=sum;
		}
		printf("%I64d\n",min);
	}
	return 0;
}
