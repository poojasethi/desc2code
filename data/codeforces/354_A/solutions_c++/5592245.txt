#include<iostream>
using namespace std;
int main() {
	int n, l, r, Ql, Qr, w[100000];
	long long ecost=0, mincost=20000000000;
	cin >> n >> l >> r >> Ql >> Qr;
	for(int i=0;i<n;i++) {
		cin >> w[i];
		ecost += r*w[i];
	}
	for(int i=0;i<n;i++) {
		mincost = min(mincost, ecost + max(0, i-(n-i) - 1)*Ql + max(0, (n-i)-i-1)*Qr);
		ecost -= r*w[i];
		ecost += l*w[i];
	}
	mincost=min(mincost, ecost + (n-1)*Ql);
	cout << mincost << '\n';
	return 0;
}
