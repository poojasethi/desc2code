#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int a[100010];

int main()
{
	int n,l,r,q1,q2,sum=0,i,ans;
	cin>>n>>l>>r>>q1>>q2;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		sum+=a[i];
	}
	sum*=r;
	ans=sum+(n-1)*q2;
	for(i=1;i<=n;i++)
	{
		sum=sum-r*a[i]+l*a[i];
		if(i<n-i)	ans=min(ans,sum+(n-i-i-1)*q2);
		else if(i>n-i)	ans=min(ans,sum+(i-(n-i)-1)*q1);
		else	ans=min(ans,sum);
	}
	cout<<ans;
	return 0;
}
