#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long ll;

int n;
ll l,r,ql,qr,w[100005];

int main(){
	scanf("%d",&n);
	scanf("%I64d%I64d%I64d%I64d",&l,&r,&ql,&qr);
	for(int i=1;i<=n;i++) scanf("%I64d",w+i),w[i]+=w[i-1];
	ll ans=1e18;
	for(int i=0;i<=n;i++){
		ans=min(ans,w[i]*l+(w[n]-w[i])*r+(i>n-i?max(i-(n-i)-1,0)*ql:max(n-i-i-1,0)*qr));
	}
	printf("%I64d\n",ans);
}
