#include <math.h>
#include <stdio.h>
#include <algorithm>
using namespace std;
#define inf 2000000000
int n,l,r,Ql,Qr,d[1000001],a,ans,z,i,ot;
int main(){
  scanf("%d %d %d %d %d",&n,&l,&r,&Ql,&Qr);
  for(i=1;i<=n;i++) 
   scanf("%d",&a),
   d[i]=d[i-1]+a;
  ans=inf;
  for(i=0;i<=n;i++){
   z=d[i]*l+(d[n]-d[i])*r;
   if(i>n-i) z+=(2*i-n-1)*Ql;
   if(i<n-i) z+=(n-2*i-1)*Qr;
   ans=min(ans,z);
  }
  printf("%d",ans);
 return 0;
}
