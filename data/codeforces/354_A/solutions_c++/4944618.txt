#include<cstdio>
#include<cstring>
#include<algorithm>
const int inf=2100000000;
using namespace std;
int i,j,k,n,m,x,y,l,r,a[210001],s[210001],ans;
int main(){
	scanf("%d%d%d%d%d",&n,&x,&y,&l,&r);
	for (i=1,s[0]=0;i<=n;i++) scanf("%d",&a[i]),s[i]=s[i-1]+a[i];
	for (i=0,ans=inf;i<=n;i++){
		k=s[i]*x+(s[n]-s[i])*y;
		if (i<=n/2) k+=r*max(0,n-2*i-1);else k+=max(2*i-n-1,0)*l;
		ans=min(ans,k);
	}
	printf("%d\n",ans);
	return 0;
}
