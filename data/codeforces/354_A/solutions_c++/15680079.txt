#include<cstdlib>
#include<cstdio>
using namespace std;

int m,k,ans=0x7fffffff;
int n,l,r,ql,qr,w,s[100001];

int main()
{
	scanf("%d%d%d%d%d",&n,&l,&r,&ql,&qr);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&w);
		s[i]=s[i-1]+w;
	}
	for (int i=0;i<=n;i++)
	{
		m=l*s[i]+r*(s[n]-s[i]);
		k=i<<1;
		if (k>n) m+=(k-n-1)*ql;
		else if (k<n) m+=(n-k-1)*qr;
		if (m<ans) ans=m;
	}
	printf("%d",ans);
	return 0;
}
