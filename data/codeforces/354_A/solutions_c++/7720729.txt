#include <iostream>
using namespace std;
const int MAXN=100100;
int dp[MAXN];
int main()
{
  int n,l,r,ql,qr;
  cin>>n>>l>>r>>ql>>qr;
  for(int i=1;i<=n;i++)
    {
      int x;
      cin>>x;
      dp[i]=dp[i-1]+x;
    }
  long long ans=1e16;
  for(int i=0;i<=n;i++)
    {
      long long x=dp[i]*l+(dp[n]-dp[i])*r;
      if(i>n-i)
	x+=ql*(i-(n-i)-1);
      if(i<n-i)
	x+=qr*(n-i-i-1);
      ans=min(ans,x);
    }
  cout<<ans<<endl;
  return 0;
}
