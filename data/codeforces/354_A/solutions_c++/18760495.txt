#include<bits/stdc++.h>
using namespace std;
const int N = 100001;
int n, l, r, ql, qr, w[N], s1, s2, tmp, ans = 2e9+2e5;

int main(int argc, char **argv) {
	scanf("%d%d%d%d%d", &n, &l, &r, &ql, &qr);
	for(int i=0 ; i<n ; ++i) {
		scanf("%d", w+i);
		s2 += w[i];
	}
	for(int i=0 ; i<=n ; ++i) {
		tmp = s1 * l + s2 * r + ql * max(i+i-n-1, 0) + qr * max(n-1-i-i, 0);
		s1 += w[i]; s2 -= w[i];
		ans = min(ans, tmp);
	}
	cout << ans << endl;
	return 0;
}
