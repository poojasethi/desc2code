#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>
using namespace std;
int a[100000+3],n,l,r,Ql,Qr,ans;
int main()
{
scanf("%d%d%d%d%d",&n,&l,&r,&Ql,&Qr);
for (int i=1;i<=n;i++) scanf("%d",&a[i]),a[i]+=a[i-1];ans=2e9;
for (int i=0;i<=n;i++)
  {
  int ss=a[i]*l+(a[n]-a[i])*r;
  if (n-i>i) ss+=max(n-i-i-1,0)*Qr;
    else ss+=max(i-n+i-1,0)*Ql;
  ans=min(ans,ss);
  }
printf("%d",ans);
}
