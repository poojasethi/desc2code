#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=1;i<=n;i++)
const int N=100005;
int n,l,r,Ql,Qr,ans,a[N],s[N];
int main()
{
	scanf("%d%d%d%d%d",&n,&l,&r,&Ql,&Qr),ans=2147483647;
	rep(i,n) scanf("%d",a+i),s[i]=s[i-1]+a[i];
	for (int i=0;i<=n;++i)
		ans=min(ans,s[i]*l+(s[n]-s[i])*r+max(0,i-n+i-1)*Ql+max(0,n-i-i-1)*Qr);
	printf("%d\n",ans); return 0;
}
