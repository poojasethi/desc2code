#include <cstdio>
#include <algorithm>
int N,L,R,l,r,s[100005],ans=1e9;
int main()
{
	scanf("%d%d%d%d%d",&N,&l,&r,&L,&R);
	for (int i=1; i<=N; i++) scanf("%d",&s[i]),s[i]+=s[i-1];
	for (int i=0; i<=N; i++) ans=std::min(ans,s[i]*l+(s[N]-s[i])*r+(i>N-i+1?(2*i-N-1)*L:N-i>i+1?(N-2*i-1)*R:0));
	printf("%d\n",ans);
}
