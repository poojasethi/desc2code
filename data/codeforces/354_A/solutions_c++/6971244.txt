#include<bits/stdc++.h>
#define INF (long long)1e+18
typedef long long ll;
using namespace std;

int main()
{
    ll n,l,r,Ql,Qr,ans=INF;
    cin>>n>>l>>r>>Ql>>Qr;
    vector<ll> arr(n+1,0);
    for(ll i=1;i<=n;i++)
      {
        cin>>arr[i];
        arr[i] = arr[i-1] + arr[i];
      }

    for(ll i=0;i<=n;i++)
        ans = min(ans , ((arr[i]*l) + (arr[n]-arr[i])*r) + max(0ll,i-(n-i)-1)*Ql + max(0ll,(n-i)-i-1)*Qr);
    
    cout<<ans;
    return 0;
}
