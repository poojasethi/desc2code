n, l, r, ql, qr = map(int,raw_input().split())
w = map(int, raw_input().split())

ans = sum(w)*r + (n-1)*qr
sum_w = [0]*(n+1)
for i in range(n):
    sum_w[n-i-1] = w[n-i-1] + sum_w[n-i]
    
for i in xrange(n):
    temp = (sum_w[0] - sum_w[i+1]) * l + sum_w[i+1] * r
    if i+1 != n-i-1:
        temp += (i+i+2-n-1)*ql if n-i-1<= i+1 else (n-i-i-3)*qr
    ans = min(ans, temp)
print ans
