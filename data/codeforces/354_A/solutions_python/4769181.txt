from sys import stdin
def main():
    n, l, r, ql, qr = map(int, stdin.readline().split())
    w = map(int, stdin.readline().split())
    ans = 10 ** 9
    m = n / 2
    a = sum(w) * r
    for i, x in enumerate(w):
        t = i - (n - i)
        if t < -1:
            ans = min(ans, a + qr * (-t - 1))
        elif t > 1:
            ans = min(ans, a + ql * (t - 1))
        else:
            ans = min(ans, a)
        a += x * (l - r)
    ans = min(ans, a + ql * (n - 1))
    print ans
main()
