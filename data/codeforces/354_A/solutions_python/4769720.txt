n, l, r, Ql, Qr = map(int, raw_input().split())
w = map(int, raw_input().split())

s = w[:]
for i in range(1, n):
  s[i] += s[i - 1]

cost = [0 for i in range(n + 1)]

cost[0] = s[n - 1] * r + (n - 1)* Qr
for i in range(1, n + 1):
  cost[i] = s[i - 1] * l + (s[n - 1] - s[i - 1]) * r
  if n - i > i:
    cost[i] += (n - i - i - 1) * Qr
  elif n - i < i:
    cost[i] += (i - (n - i) - 1) * Ql
  

print min(cost)
