n, l, r, qL, qR = map(int, raw_input().split())
w = map(int, raw_input().split())
s = [0]
for x in w:
    s.append(s[-1] + x)
ans = -1
for k in xrange(0, n+1):
    tmp = l * s[k] + r * (s[n] - s[k])
    if k < n - k:
        tmp += (n - 2 * k - 1) * qR
    elif k > n - k:
        tmp += (2 * k - n - 1) * qL
    if ans == -1 or ans > tmp:
        ans = tmp
print ans
