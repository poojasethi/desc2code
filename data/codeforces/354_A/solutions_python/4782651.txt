#!/usr/bin/env python

n, l, r, ql, qr = map(int, raw_input().split())
w = map(int, raw_input().split())

s = [0]
for wi in w:
    s.append(s[-1] + wi)

ans = 10 ** 9
mid = (n + 1) / 2
for left in xrange(n + 1):
    sl = s[left] * l
    sr = (s[n] - s[left]) * r
    if abs(n - 2 * left) > 1:
        if left < mid:
            sr += (n - 2 * left - 1) * qr
        else:
            sl += (2 * left - n - 1) * ql
    ans = min(ans, sl + sr)
print ans
