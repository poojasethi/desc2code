n = map(int, raw_input().split())[0]
a = map(int, raw_input().split())
b = map(int, raw_input().split())
joy = 0

for i in xrange(n):
    note = a[i]
    min_note = b[i]
    if (2*note) < min_note:
        joy -= 1
    else:
        if min_note / 2 == 0:
            joy -= 1
        elif min_note % 2 == 0:
            joy +=((min_note)/2) * ((min_note/2))
        else:
            joy += ((min_note/2) * ((min_note/2) + 1))

print joy
