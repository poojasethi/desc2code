#!/usr/bin/env python

import math

if __name__ == '__main__':
    n = int(raw_input())
    v = map(int, raw_input().split())
    b = map(int, raw_input().split())

    joy = 0
    for i, bi in enumerate(b):
        if v[i] * 2 < bi or bi == 1:
            joy -= 1
        else:
            joy +=  int(math.floor(bi / 2.0) * math.ceil(bi / 2.0))
    if joy < 0:
        print '-1'
    else:
        print joy
