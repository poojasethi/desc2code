import sys
import math

def main():
  notes = int(sys.stdin.readline())
  a_in = sys.stdin.readline().split()
  b_in = sys.stdin.readline().split()
  total = 0

  for i in range(notes):
    a = int(a_in[i])
    b = int(b_in[i])
    if a * 2 < b:
      total -= 1
      continue
    elif b < 2:
      total -= 1
    elif b % 2 == 0:
      total += (b/2)*(b/2)
    else:
      total += ((b-1)/2)*((b-1)/2+1)

  print total

if __name__ == '__main__':
  main()
