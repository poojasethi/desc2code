def main():
    n = int(raw_input())

    a = map(int, raw_input().split())
    b = map(int, raw_input().split())

    joy = 0

    for i in xrange(n):
        if (b[i] > 2*a[i]) or b[i] == 1:
            joy -= 1
        elif b[i] % 2 == 0:
            joy += (b[i]/2)**2
        else:
            joy += (b[i]/2)*(b[i]/2 + 1)

    print joy


if __name__ == "__main__":
    main()
