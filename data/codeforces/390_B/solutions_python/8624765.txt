n = int( raw_input() )

ai = []
bi = []

st = raw_input().split()

for i in st:
    ai.append( int(i) )

st = raw_input().split()

for i in st:
    bi.append( int(i) )

tot = 0

for i in xrange( 0, n ):
    if ai[ i ]*2 >= bi[ i ]:
        x = bi[ i ]/2
        y = bi[ i ] - x
        if x > 0 and y > 0:
            tot += x*y
        else:
            tot -= 1
    else:
        tot -= 1

print tot
