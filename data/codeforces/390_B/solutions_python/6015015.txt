#!/usr/bin/env python


def read():
    raw_input()
    A = map(int, raw_input().split())
    B = map(int, raw_input().split())
    return (A, B)


def work((A, B)):
    ans = 0
    for i in range(len(A)):
        if A[i] * 2 >= B[i] >= 2:
            ans += (B[i] / 2) ** 2 if B[i] % 2 == 0 else (B[i] / 2) * ((B[i] + 1) / 2)
        else:
            ans -= 1
    print ans


if __name__ == "__main__":
    work(read())
