from sys import stdin
def main():
    n = int(stdin.readline())
    a = map(int, stdin.readline().split())
    b = map(int, stdin.readline().split())
    s = 0
    for i in xrange(n):
        if 2 <= b[i] <= a[i] + a[i]:
            x = b[i] / 2
            s += x * (b[i] - x)
        else:
            s += -1
    print s
main()
