#include"stdio.h"
long long v[111111];
int main()
{
	long long b,i,sum,n;
	scanf("%lld",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&v[i]);
	}
	sum=0;
	for(i=1;i<=n;i++)
	{
		scanf("%lld",&b);
		if(b==1||b>v[i]*2)sum--;
		else {
		if(b%2==0)sum+=(b*b/4);
		else sum+=((b/2)*((b/2)+1));
		}
	}
	printf("%lld\n",sum);
}
