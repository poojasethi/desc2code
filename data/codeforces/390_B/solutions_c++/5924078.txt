#include<bits/stdc++.h>
using namespace std;

int main()
{
    long long n, a[100005], b[100005], j=0, temp;

    cin>>n;
    for(int i=0;i<n;i++) cin>>a[i];
    for(int i=0;i<n;i++) cin>>b[i];
    for(int i=0;i<n;i++){
        if((b[i]<2)||(b[i]>(2*a[i]))) { j--; }
        else { temp=((int)(b[i]/2)); j+=(temp*(b[i]-temp)); }
    }

    cout<<j;
    return 0;
}
