#include<iostream>
using namespace std;
#define ll long long
int main()
{
	int n;
	cin>>n;
	ll v[n],b[n];
	for(int i=0;i<n;i++)
	cin>>v[i];
	for(int i=0;i<n;i++)
	cin>>b[i];
	ll ct = 0;
	for(int i=0;i<n;i++)
	{
		ll g,vi;
		g = vi = b[i]/2;
		if(b[i]&1)
		g++;
		if(vi==0)
		vi=1;
		//cout<<g<<" "<<vi<<" "<<g*vi<<endl;
		if(g<=v[i] && vi<=v[i] && (g+vi)==b[i])
		ct += g*vi;
		else
		ct--;
	}
	cout<<ct<<endl;
return 0;
}
