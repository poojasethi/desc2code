#include <bits/stdc++.h>
using namespace std;
int n;
int a[10+100000];
int main(){
    //freopen("A.inp","r",stdin);
    scanf("%d",&n);
    for(int i = 0;i < n;++i) scanf("%d",a+i);
    long long ans = 0;
    for(int i = 0;i < n;++i){
        int u;
        scanf("%d",&u);
        if(u > 2*a[i] || u == 1) {ans-=1;continue;}
        else ans+=1ll*(u/2)*(u-(u/2));
    }
    printf("%I64d\n",ans);
return 0;
}
