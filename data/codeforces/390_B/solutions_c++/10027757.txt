#include<iostream>
using namespace std;
int n;
long long ans;
long long a[100010],b[100010];
int main()
{
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int i=1;i<=n;i++)
		cin>>b[i];
	for(int i=1;i<=n;i++)
	{
		if(2ll*a[i]>=b[i]&&b[i]>1)
			ans+=(b[i]/2ll)*(b[i]-b[i]/2ll);
		else ans--;
	}
	cout<<ans<<endl;
	return 0;
}
