#include<iostream>
using namespace std;
int main()
{
    int N;
    cin>>N;
    long long int A[N],B[N],res=0;
    for(int i=0;i<N;i++)
        cin>>A[i];
    for(int i=0;i<N;i++)
        cin>>B[i];
    for(int i=0;i<N;i++)
    {
        if(((B[i]/2)+B[i]%2)<=A[i] && B[i]!=1)
            res+=(B[i]/2)*(B[i]-B[i]/2);
        else
            res-=1;
    }
    cout<<res;
    return 0;
}
