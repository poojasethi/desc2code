#include <cstdio>
#include <algorithm>
using namespace std;
long long a[100005],b[100005],c,d,e,f,g,h,i,j,k,l,m,n;
int main(){
	scanf("%I64d",&n);
	for(i=1;i<=n;i++)scanf("%I64d",&a[i]);
	for(i=1;i<=n;i++)scanf("%I64d",&b[i]);
	for(i=1;i<=n;i++){
		if(b[i]==1){h--;continue;}
		if(a[i]+a[i]<b[i]){h--;continue;}
		else h=h+(b[i]/2)*(b[i]-(b[i]/2));
	}
	printf("%I64d",h);
	return 0;
}
