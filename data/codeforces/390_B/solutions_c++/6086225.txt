#include<cstdio>
#include<iostream>
using namespace std;
typedef long long ll;
const int maxn=100001;
ll a[maxn]={0};
ll b[maxn]={0};
int n;
int main(){
ll ans=0;
scanf("%d",&n);
for(int i=1;i<=n;i++)cin>>a[i];
for(int i=1;i<=n;i++)cin>>b[i];
for(int i=1;i<=n;i++){
if(2*a[i]<b[i]||b[i]==1)ans--;
else {
ans+=(b[i]/2)*(b[i]-b[i]/2);
}
}
cout<<ans<<endl;
return 0;
}
