#include <cstdio>
#include <algorithm>
using namespace std;
int a[100010],b[100010],n;
long long ans;
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;i++) scanf("%d",&a[i]);
	for (int i=1;i<=n;i++) scanf("%d",&b[i]);
	ans = 0;
	for (int i=1;i<=n;i++){
		if (b[i]==1 || a[i]*2<b[i]) ans--;
		else {
			long long k = b[i]/2;
			ans+=k*(b[i]-k);
		}
	}
	printf("%I64d",ans);
}
