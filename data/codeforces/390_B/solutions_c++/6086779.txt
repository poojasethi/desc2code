#include<stdio.h>
int main()
{
   int n,x[100050];
   __int64 y[100050],i;
   __int64 sum=0;
	scanf("%d",&n);

	for(i=0;i<n;i++)
	   scanf("%d",&x[i]);
    for(i=0;i<n;i++)
       scanf("%I64d",&y[i]);
    for(i=0;i<n;i++)
    {
    	if(2*x[i]>=y[i]&&y[i]!=1)
    	  sum=sum+(y[i]/2)*(y[i]-y[i]/2);
  	    else sum--;
    }
    printf("%I64d\n",sum);
}
