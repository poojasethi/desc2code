#include"stdio.h"
__int64 a[100005],b[100005],ans;
main() {
	int i,n;
	while(~scanf("%d",&n)){
		for(i=0;i<n;i++)scanf("%I64d",&a[i]);
		for(i=0;i<n;i++)scanf("%I64d",&b[i]);
		ans=0;
		for(i=0;i<n;i++){
			if(b[i]>a[i]*2||b[i]==1)ans--;
			else {
				if(b[i]%2==0)ans+=(b[i]*b[i]/4);
				else ans+=((b[i]/2)*(1+b[i]/2));
			}
		}
		printf("%I64d\n",ans);
	}
}
