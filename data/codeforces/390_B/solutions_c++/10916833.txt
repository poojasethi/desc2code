#include <iostream>
using namespace std;
typedef long long ll;

const int N = 1e5 + 10;

int a[N], b[N];

int main()
{
  int n;
  cin >> n;
  for (int i = 0; i < n; ++i)
    cin >> a[i];
  for (int i = 0; i < n; ++i)
    cin >> b[i];
  ll ans = 0;
  for (int i = 0; i < n; ++i)
    {
      if (a[i] * 2 < b[i] || b[i] == 1)
	ans --;
      else
	ans += (ll) b[i] * b[i] / 4;
    }
  cout << ans << '\n';
  return 0;
}
