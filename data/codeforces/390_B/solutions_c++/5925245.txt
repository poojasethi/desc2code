#include<stdio.h>

int a[100100];
int b[100100];

int main() {
  int n;
  long long j=0;
  scanf("%d", &n);
  for (int i=0; i<n; i++) { scanf("%d", &a[i]); }
  for (int i=0; i<n; i++) { scanf("%d", &b[i]); }
  for (int i=0; i<n; i++) {
    if (2*a[i] < b[i] || b[i]<2) { j--; }
    else { j += (long long)(b[i]/2)*(b[i] - (b[i]/2)); }
  }
  printf("%I64d", j);
  return 0;
}
