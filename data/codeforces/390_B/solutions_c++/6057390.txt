#include <iostream>
#include <fstream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;
int main(){
	long long n,a[100001],b[100001],ans=0,i;
	cin>>n;
	for (i=1;i<=n;i++)
		cin>>a[i];
	for (i=1;i<=n;i++){
		cin>>b[i];
		if (a[i]*2>=b[i]&&b[i]>1){
			ans+=((b[i]/2)*(b[i]-b[i]/2));
		}
		else ans--;
	}
	cout<<ans;
    return 0;
}
