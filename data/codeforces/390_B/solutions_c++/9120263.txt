#include <iostream>

using namespace std;

long long a[100005], n, i, x, s;

int main()
{
    cin>>n;
    for (i=1; i<=n; ++i)
         cin>>a[i];
    for (i=1; i<=n; ++i)
    {
        cin>>x;

        if (a[i]*2>=x && x!=1)
            s+=(x/2)*(x-x/2);
        else --s;

    }
    cout<<s;


    return 0;
}
