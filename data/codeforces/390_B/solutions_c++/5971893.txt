#include <iostream>

using namespace std;

//

int N,A[100010],B[100010];
long long ret;

int main()
{
    cin >> N;
    for (int i=0;i<N;i++) cin >> A[i];
    for (int i=0;i<N;i++) cin >> B[i];
    for (int i=0;i<N;i++)
        if (A[i]*2<B[i]||B[i]/2==0) ret--;
        else ret+=(long long)B[i]/2*(B[i]-B[i]/2);
    cout << ret << endl;
    return 0;
}
