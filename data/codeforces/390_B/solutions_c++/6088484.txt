#include<stdio.h>
#define Max 100005
__int64 a[Max],b[Max];
int main()
{
	int n,i;
	__int64 x,max=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	  scanf("%I64d",&a[i]);
    for(i=0;i<n;i++)
      scanf("%I64d",&b[i]);
    for(i=0;i<n;i++)
    {
    	if(b[i]==1||a[i]*2<b[i])
    	  max--;
   	    else
	   	{   x=b[i]/2;
    	   	max+=x*(b[i]-x);
    	}
    	  
    }
    printf("%I64d\n",max);
}
