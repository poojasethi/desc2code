#include <iostream>
#include <cstdio>
#include <cstring>

int n,i;
long long a[100100],b[100100];
long long x,ans;

int main()
{
    scanf("%d",&n);
    for (i=1;i<=n;i++) scanf("%I64d",&a[i]);
    for (i=1;i<=n;i++) scanf("%I64d",&b[i]);
    ans=0;
    for (i=1;i<=n;i++)
    {
        if (2*a[i]<b[i]||b[i]==1) ans--;
        else ans+=b[i]/2*(b[i]-b[i]/2);
    }
    printf("%I64d\n",ans);
    return 0;
}
