#include<iostream>
using namespace std;
int main()
{
    long long int n,i,b,count=0;
    cin>>n;
    long long int arr[n];
    for(i=0;i<n;i++)cin>>arr[i];
    for(i=0;i<n;i++){
        cin>>b;
        if(b>2*arr[i] || b<2)count--;
        else count+=(b*b)/4;
    }
    cout<<count;
    return 0;
}
