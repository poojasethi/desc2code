#include <cstdio>
#include <cstdlib>
#include <algorithm>
#define N 1000005
using namespace std;
int a[N];

int main()
{
	long long ans=0;
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;++i) scanf("%d",&a[i]);
	for(int i=1,x;i<=n;++i)
	{
		scanf("%d",&x);
		if(a[i]*2<x){ans--;continue;}
		long long t=x/2;
		if(t<=0){ans--;continue;}
		ans+=t*(x-t);
	}
	printf("%I64d\n",ans);
	return 0;
}
