#include <bits/stdc++.h>
using namespace std;

int main(){
	long long int n, v, t;
	cin >> n;
	int a[n+100], b[n+100];
	for(int i=0; i<n; i++){
		cin >> a[i];
	}
	for(int i=0; i<n; i++){
		cin >> b[i];
	}
	long long int resp=0;
	for(int i=0; i<n; i++){
		if(b[i]>a[i]*2 || b[i]<2) resp--;
		else resp += pow(b[i], 2)/4;
	}
	cout << resp << endl;
	return 0;
}
