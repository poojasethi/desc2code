#include <iostream>
#include <cmath>
using namespace std;

int main() {
     int n, a[100000], b;
    long long cnt = 0;
    cin>>n;
    for(int i=0; i<n; i++) 
        cin>>a[i];
    for(int i=0; i<n; i++) {
        cin>>b;
        if(b>a[i]*2 || b<2) 
            cnt--;
        else 
            cnt+=pow(b, 2)/4;
    }
    cout<<cnt<<endl;
    return 0;
}
