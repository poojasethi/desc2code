#include <iostream>

using namespace std;

typedef long long int ll;

ll a[1000000],b[1000000];

int main()
{
  int n;
  cin>>n;
  ll sol=0;
  for (int i=0;i<n;i++)
    cin>>a[i];
  for (int i=0;i<n;i++)
    cin>>b[i];
  for (int i=0;i<n;i++) {
    if (b[i]==1 or 2*a[i]<b[i])
      sol--;
    else {
      ll c=b[i]/2;
      if (2*c<b[i]) sol+=c*(c+1);
      else sol+=c*c;
    }
  }
  cout<<sol<<endl;
}
