#include<iostream>
#include<cstdio>

using namespace std;

__int64 a[1000001],b[1000001];
int main()
{
	__int64 i,n,h=0;
	scanf("%I64d",&n);
	for(i=0;i<n;i++)
		scanf("%I64d",&a[i]);
	for(i=0;i<n;i++)
		scanf("%I64d",&b[i]);
	for(i=0;i<n;i++)
	{
		if(a[i]*2>=b[i]&&b[i]!=1)
			h+=(b[i]/2)*(b[i]-b[i]/2);
		else
			h--;
	}
	printf("%I64d\n",h);
	return 0;
}
