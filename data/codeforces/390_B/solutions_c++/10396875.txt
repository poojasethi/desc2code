#include<cstdio>
int n;
long long a[100001],b[100001];
int main()
{
	int i,j,k;
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	long long ans=0;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&b[i]);
		if(b[i]>2*a[i]||b[i]<=1)ans--;else
		{
			if(b[i]&1)ans+=(b[i]/2)*(b[i]/2+1);else ans+=(b[i]/2)*(b[i]/2);
		}
	}
	printf("%I64d\n",ans);
	return 0;
}
