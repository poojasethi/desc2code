#include<iostream>
#include<cstdio>
using namespace std;
long long a[1000001];
long long b[1000001];
long long ans;
int main()
{
	int i,j,n;
	cin>>n;
	for(i=1;i<=n;i++)
		scanf("%d",a+i);
	for(i=1;i<=n;i++)
		scanf("%d",b+i);
	for(i=1;i<=n;i++)
	{
		if(b[i]==1)
			ans--;
		else if(a[i]*2>=b[i])
		{
			ans+=b[i]/2*(b[i]-b[i]/2);
		}
		else
		{
			ans--;
		}
	}
	cout<<ans;
}
