#include <bits/stdc++.h>

using namespace std;

#define ll long long int

ll cal(ll num)
{
	if(num==1) return -1;
	if(num%2==0)
		return (num*num)/4;
	else return (num/2)*(num/2+1);
}

int main()
{
	ll n,i,j,k;
	cin>>n;
	ll A[n],B[n];
	for(i=0;i<n;i++) cin>>A[i];
	for(i=0;i<n;i++) cin>>B[i];
	ll ans=0;
	for(i=0;i<n;i++)
	{
		if((2*A[i])>=B[i]) ans+=cal(B[i]);
		else ans-=1;
	}
	cout<<ans<<endl;
	return 0;
}
