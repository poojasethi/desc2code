#include<bits/stdc++.h>
using namespace std;
int main()
{
  int n;
  cin>>n;
  long long int a[n],b[n];
  for(int i=0;i<n;i++)
  cin>>a[i];
  for(int i=0;i<n;i++)
    cin>>b[i];
  long long int joy=0;
  for(int i=0;i<n;i++)
  {
    if((2*a[i])<b[i] || b[i]==1)
      joy--;
    else
    {
      joy+=((b[i]/2)*(b[i]-(b[i]/2)));
    }
  }
  printf("%lld",joy);
  return 0;
}
