#include <iostream>

using namespace std;

int main()
{
    long long int ans=0,i,l,n;
    std::ios::sync_with_stdio(false);

    cin >> n;

    long long int a[n],b[n];
    for(i=0;i<n;i++) cin >> a[i];
    for(i=0;i<n;i++) cin >> b[i];

    for(i=0;i<n;i++)
    {
        if(b[i]>a[i]*2 || b[i]==1) ans--;
        else ans += (b[i]/2)*((b[i]+1)/2);
    }

    cout << ans << endl;

    return 0;
}
