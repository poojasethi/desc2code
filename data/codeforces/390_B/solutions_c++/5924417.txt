#include<iostream>
using namespace std;

const int oo = 100010;
long long n, a[oo], b[oo], ans;

int main(){
  cin >> n;
  for(int i = 0; i < n; i++)
    cin >> a[i];
  for(int i = 0; i < n; i++)
    cin >> b[i];
  for(int i = 0; i < n; i++){
    if(b[i] > 2 * a[i] || b[i] == 1)
      ans--;
    else
      ans += ((b[i] / 2) * (b[i] / 2 + b[i] % 2));
  }
  cout << ans;
  return 0;
}
