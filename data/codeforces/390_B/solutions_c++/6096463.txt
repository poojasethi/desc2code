#include<stdio.h>
#define ll __int64
ll n,i,j,tot,a[1000005],b[1000005];
int main(){
	scanf("%I64d",&n);
	for(i=1;i<=n;i++)scanf("%I64d",&a[i]);
	for(i=1;i<=n;i++)scanf("%I64d",&b[i]);
	tot=0;
	for(i=1;i<=n;i++){
		if(a[i]*2>=b[i]&&b[i]!=1){
			tot+=(b[i]/2)*(b[i]-b[i]/2);
		}
		else tot-=1;
	}
	printf("%I64d\n",tot);
}
