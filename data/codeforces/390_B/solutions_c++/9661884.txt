#include<stdio.h>
#include<math.h> 
int a[100000],b[100000];
int main()
{
	int n,i;
	long long int sum=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	scanf("%d",&a[i]);
	for(i=0;i<n;i++)
	scanf("%d",&b[i]);
	for(i=0;i<n;i++)
	{
		if(b[i]<=2*a[i]&&b[i]>=2)
		{
			long long int c=b[i]/2;
			sum+=c*(b[i]-c); 
		}
		else sum--;
	}
	printf("%lld\n",sum);
	return 0;
}
