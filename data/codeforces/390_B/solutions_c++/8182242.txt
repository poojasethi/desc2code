#include <iostream>
#include <set>
#include <algorithm>
using namespace std;
int main()
{
    int n;cin>>n;
    int a[n],b[n];
    for(int i=0;i<n;i++)cin>>a[i];
    for(int i=0;i<n;i++)cin>>b[i];

    long long joy=0;
    for(int i=0;i<n;i++)
    {
        if(b[i]-a[i]>a[i] || b[i]<=1)joy--;
        else joy+=(long long)b[i]/2*(b[i]-b[i]/2);
    }
    cout<<joy;
}
