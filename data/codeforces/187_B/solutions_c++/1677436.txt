#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int a[65][65],b[65][65],d[65],f[65],ans[65][65][65];
int n,m,t,i,j,k,l,s,x,y,z;
int main()
{
	cin>>n>>m>>t;
	memset(b,0x3f,sizeof(b));
	for(l=1;l<=m;l++)
	{
		for(i=1;i<=n;i++)
			for(j=1;j<=n;j++) scanf("%d",&a[i][j]);
		for(k=1;k<=n;k++)
			for(i=1;i<=n;i++)
				for(j=1;j<=n;j++)
					a[i][j]=min(a[i][j],a[i][k]+a[k][j]);
		for(i=1;i<=n;i++)
			for(j=1;j<=n;j++) b[i][j]=min(b[i][j],a[i][j]);
	}
	for(s=1;s<=n;s++)
	{
		memset(d,0x3f,sizeof(d));
		d[s]=0;
		for(k=0;k<=n;k++)
		{
			memcpy(ans[s][k],d,sizeof(d));
			memcpy(f,d,sizeof(d));
			for(i=1;i<=n;i++)
				for(j=1;j<=n;j++)
					f[j]=min(f[j],d[i]+b[i][j]);
			memcpy(d,f,sizeof(f));
		}
	}
	for(i=1;i<=t;i++)
	{
		scanf("%d%d%d",&x,&y,&z);
		z++,z=min(z,n);
		printf("%d\n",ans[x][z][y]);
	}
	return 0;
}
