#include<iostream>
#include<cstdio>
#include<string.h>
#include<string>
#include<math.h>
#include<algorithm>
#include<map>
using namespace std;

int adj[61][61][61];
int f[61][61][61];
int n,m,r;

int main(){
	cin >> n >> m >> r;
	memset(f,1,sizeof(f));
	for(int k=0;k<m;k++)
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				cin >> adj[k][i][j];
	for(int t=0;t<m;t++) for(int k=0;k<n;k++)
			for(int i=0;i<n;i++) for(int j=0;j<n;j++)
				adj[t][i][j] = min(adj[t][i][j],
						adj[t][i][k]+adj[t][k][j]);
	for(int t=0;t<m;t++)
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				f[0][i][j] = min(f[0][i][j],adj[t][i][j]);
	for(int t=1;t<=n;t++)
		for(int k=0;k<n;k++)
			for(int i=0;i<n;i++)
				for(int j=0;j<n;j++)
					f[t][i][j] = min(f[t][i][j],f[t-1][i][k]+f[0][k][j]);
	while(r--){
		int s,t,k;
		cin >> s >> t >> k;
		k = min(n-1,k);
		cout << f[k][s-1][t-1] << endl;
	}
	return 0;
}
