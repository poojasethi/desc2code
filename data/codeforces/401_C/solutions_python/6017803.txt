n0,n1 = map(int,raw_input().split())
cando = True
if n1 > 2*(n0+1) :
    cando = False
if n1 < n0 - 1 :
    cando = False
s = ""
if cando :
    if n1 == n0 - 1 :
        print "0%s" % ("10"*n1)
    else :
        n1 -= n0
        for i in range(n0) :
            if n1 > 0 :
                s += "%s10" % ("1")
                n1 -= 1
            else :
                s += "10"
        s += "1"*n1
        print s
else :
    print -1
