inp = raw_input().split()
n = int(inp[0])
m = int(inp[1])
if(n-1>m):
    print -1
elif(m>(2*(n+1))):
    print -1
else:
    ans = ["1"]
    m-=1
    while(n):
        ans +=["0","1"]
        n-=1
        m-=1
    x=0
    if(m==-1):
        print "".join(ans[1:])
    elif(m==-2):
        print "".join(ans[1:-1])
    else:
        while(m>0):
           ans[x]="11"
           x+=2
           m-=1
        print "".join(ans)
