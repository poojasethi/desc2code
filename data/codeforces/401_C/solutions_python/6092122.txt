n, m = map(int, raw_input().split())

def start_with_one(n, m):
	if n > m:
		return None
	buff = []
	while m > n and n > 0 and m > 0:
		buff += "110"
		m -= 2
		n -= 1
	if m == n:
		return ''.join(buff) + "10"* m
	if m <= 2:
		return ''.join(buff) + "1"* m
	else:
		return None

s = start_with_one(n, m)
if s is None:
	s = start_with_one(n-1, m)
	if s is None:
		print -1
	else:
		s = "0" + s
		print s
else:
	print s
