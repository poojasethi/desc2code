n,m=map(int,raw_input().split())
if (m<n-1 or m>2*n+2):
    print "-1"
else:
    if m==n-1:
        print "01"*m + "0"
    elif m==n:
        print "01"*n
    elif m<2*n+1 :
        print "10"*(2*n-m)+"110"*(m-n)
    elif m==2*n+1:
        print "110"*(n)+"1"
    elif m==2*n+2:
        print "110"*(n)+"11"
