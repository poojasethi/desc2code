zeros, ones = map(int, raw_input().split())

if ones < zeros-1 or ones > 2*(zeros+1):
	print -1
else:
	ans = []
	if ones == zeros-1:
		for i in xrange(ones): 
			ans.append(0)
			ans.append(1)
		ans.append(0)
	else:
		while zeros:
			if ones > zeros:
				ans.append(1)
				ones -= 1
			ans.append(1)
			ans.append(0)
			ones, zeros = ones-1, zeros-1

		for i in range(0,ones):
			ans.append(1)
	
	print ("".join(map(str,ans)))
