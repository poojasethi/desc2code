n, m = map(int, raw_input().split())

res = ''

if m<n-1 or m>2*n+2:
    print '-1'
else:
    x = m-n
    if x == -1:
        res = '0'+'10'*(n-1)
    elif x == 0:
        res = '10'*m
    elif x >= 1:
        while ( m>0 or n>0 ):
            if m-n>0 and m>1:
                res += '11'
                m -= 2
            else:
                res += '1'
                m -= 1
                
            if n>0:
                res += '0'
                n -= 1
                
    print res
