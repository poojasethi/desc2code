import sys
ri = lambda: raw_input().strip()

# n = int(ri())  # 0
# m = int(ri())  # 1
n, m = map(int, ri().split())

if m < n - 1 or m > 2 * n + 2:
    print -1

else:
    if n >= m:
        if n - m == 0:
            print "01"*m

        elif n - m == 1:
            print "01"*m + "0"


    else:
        if m - n == 1:
            print "10"*n + "1"

        elif 2*n >= m:
            print "110" * (m-n) + "10" * (2*n-m)

        else:
            print "110" * n + "1" * (m-2*n)
