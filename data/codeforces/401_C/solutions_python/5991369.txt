n, m = map(int, raw_input().split())
if m > 2 * (n + 1) or n > (m + 1):
	print -1;
else:
	# if need leading '0'
	l = 1 if n > m else 0
	n = n - l
	# '110' strings
	d = max((m - 1) - (n + 1), 0)
	n, m = n - d, m - (2 * d)
	# '10' strings
	s = max(min(m, n), 0)
	n, m = n - s, m - s
	# trailing '1's
	t = m
	m = m - t
	# output
	print ('0' * l) + ('110' * d) + ('10' * s) + ('1' * t)
