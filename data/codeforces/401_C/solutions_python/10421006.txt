import math
n,m=map(int,raw_input().split())
s=""
while n!=0 and m!=0:
    if (n-math.ceil(float(m)/3)) <= (m-n) and m>=2:
        s+=("110")
        n-=1
        m-=2
    else:
        s+=("10")
        n-=1
        m-=1
if m==1 or m==2:
    s+=("1"*m)
    print s
elif n==1:
    s="0"+s
    print s
elif m==0 and n==0:
    print s
else:
    print -1
