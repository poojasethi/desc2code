import sys
def iarr(): return map(int,raw_input().split())
[n,m] = iarr()
if(m < n-1 or m >2*n+2):
    print -1
else:
    m -= n-1
    d = min(2,m)
    sys.stdout.write(d*'1')
    m-=d
    for i in xrange(n-1):
        d = min(1,m)
        sys.stdout.write('01' + d*'1')
        m -= d
    sys.stdout.write('0' + m*'1')
    sys.stdout.close()
