zeros, ones = map(int, raw_input().split())

s = ""

if zeros > ones + 1 or 2*(zeros + 1) < ones:
    print "-1"
else:
    count_ones = 0
    while zeros > 0 or ones > 0:
        if ones >= zeros and count_ones < 2:
            s += "1"
            ones -= 1
            count_ones += 1
        else:
            s += "0"
            zeros -= 1
            count_ones = 0

    print s
