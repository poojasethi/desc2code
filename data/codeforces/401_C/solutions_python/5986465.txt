def arrange(n, m):
    if (n >= m+2):
        return -1
    if (m > 2*(n+1)):
        return -1
    if (n == m+1):
        return m*"01" + "0"
    if (n == m):
        return m*"10"
    if (n == m-1):
        return n*"10" + "1"
    if (m == 2*(n+1)):
        return n*"110" + "11"
    else:
        return (m-(n+1))*"110" + (2*n-m+1)*"10" + "1"

x = raw_input("").split(' ')
n = int(x[0])
m = int(x[1])
print arrange(n, m)
