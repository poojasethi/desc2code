def main():
    n,m = map(int,raw_input().split())
    if m < n-1 or m > 2*(n+1):
        print "-1"
        return
    else:
        out = ""
        for i in xrange(n):
            if m > n - i - 1:
                out += "1"
                m -= 1
                if m > n - i - 1:
                    out += "1"
                    m -= 1
            out += "0"
        out += "1"*m    
        print out
        return

if __name__ == "__main__":
    main()
