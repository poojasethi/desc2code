n, m  = map(int, raw_input().split())

s = ''

if n>m+1:
    s = '-1'
elif n==m+1:
    s = '0'+'10'*m
elif n==m:
    s = '10'*m
elif n+1==m:
    s = '1'+'10'*n
elif n+2==m:
    s = '1'+'10'*n+'1'
elif n+3==m:
    s = '1'+'10'*n+'11'
else:
    k = m-2*n
    if k < 0:
        s = '10'*(2*n-m)+'110'*(m-n)
    else:
        if k > 2:
            s = '-1'
        else:
            s = '110'*n+'1'*k

print s
