I=lambda:map(int, raw_input().split())
n1, m1 = I()
#2x+y=m,x+y=n
#m-1,m-2,   n-1
#110  1,11
#001  0
for m in [m1, m1-1, m1-2]:
    for n in [n1, n1-1]:
        x, y = m - n, 2*n-m
        if x >= 0 and y >= 0:
            if m1 - m > 0:
                s = '110'*x+'10'*y+'1'*(m1-m)+'0'*(n1-n)
            else:
                s = '011'*x+'01'*y+'0'*(n1-n)
            print s
