z,o=(int(i) for i in raw_input().split())
if((o>2*z+2) or (z>o+1)):
    print -1
else:
    if z==0:
        print "1"*o
    elif o==0:
        print 0
    elif z==o:
        print "01"*z
    elif z>o:
        print "0"+"10"*o
    elif o==2*z+2:
        print "110"*z+"11"
    elif o==2*z+1:
        print "110"*z+"1"
    else:
        print "110"*(o-z)+"10"*(2*z-o)
