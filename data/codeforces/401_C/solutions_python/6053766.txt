#! /usr/bin/env python
from sys import stdout

n, m = map(int, raw_input().split())
end1 = False

if (m < n - 1 or m > 2 * (n + 1)):
	print -1
	exit(0)

if (m == n - 1):
	stdout.write('0')
	n = n - 1
elif (m > 2 * n):
	stdout.write('1')
	m = m - 1
	if (m > 2 * n):
		end1 = True
		m = m - 1
x = m - n
y = n - x
for i in range(y):
	stdout.write('10')
for i in range(x):
	stdout.write('101')
if (end1):
	stdout.write('1')
