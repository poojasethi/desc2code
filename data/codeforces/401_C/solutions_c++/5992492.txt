#include<bits/stdc++.h>
using namespace std;

int n,m;

int main()
{
	cin>>n>>m;
	if(m<n-1||m>2*(n+1))
	  puts("-1");
	else
	{
		for(int j=0;j<m-2*n;m--)
		  printf("1");
		while(n)
		{
			printf("0");
			n--;
			if(m)
			{
				printf("1");
				m--;
			}
			if(m>n)
			  printf("1"),m--;
		}
	}
	return 0;
}
