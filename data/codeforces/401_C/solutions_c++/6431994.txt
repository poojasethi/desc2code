#include <cstdio>
int n,m,cnt,sum;
int main()
{
    scanf("%d %d\n",&n,&m); sum=n+m;
    if(m<sum/2 || m > 2*(sum+1)/3){ puts("-1"); return 0;}
    for(int i = 1; i <= sum; i++)
    {
       if(cnt==2 || n>m){printf("0");cnt=0;n--;}
       else if(cnt == 0 || cnt == 1){printf("1");cnt++;m--;}
    }
}
