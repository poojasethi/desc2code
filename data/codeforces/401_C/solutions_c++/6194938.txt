#include<stdio.h>
int main(){
	//freopen("in.txt","r",stdin);

	int m,n,i;
	scanf("%d %d\n",&m,&n);
	if(m>n+1 || n>2*(m+1))return (puts("-1"));
	if(m==n+1){
		for(i=0;i<n;i++)printf("01");puts("0");
		return 0;
	}
	while(m<n){
		if(m){printf("110");n-=2;m--;}
		else {printf("1");n--;}
	}
	for(i=0;i<m;i++)printf("10");
	puts("");
	return 0;
}
