#include<iostream>
#include<cstring>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	if(m<n-1||m>2*n+2){cout<<"-1";return 0;}
	else
	{
		if(m==n-1){cout<<"0";n--;}
		while(n)
		{
			if(m>n){m-=2;cout<<"11";}
			else if(m==n){m--;cout<<"1";}
			n--;cout<<"0";
		}
		if(m>0)for(int i=1;i<=m;i++)cout<<"1";
	}
	return 0;
}
