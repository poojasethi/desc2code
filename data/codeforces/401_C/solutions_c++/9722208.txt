#include<iostream>
using namespace std;
int main()
{
  int n,m,t=1,k=0; cin>>n>>m;
  if(n*2+2<m || n-m >=2){
    cout<<"-1";
    return 0;
  }
  while(n || m){
    if(!t || (m>n && k<2)){
      m--;
      k++;
      t=1;
      cout<<"1";
    }
    else{
      n--;
      t=k=0;
      cout<<"0";
    }
  }
  return 0;
}
