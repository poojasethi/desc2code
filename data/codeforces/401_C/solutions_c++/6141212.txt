#include <iostream>
#include <cstdio>

using namespace std;

int n, m, p, k;

main() {
	scanf("%d%d", &n, &m);
	if(n - 1 > m || m > 2 * (n + 1)) {
		printf("-1");
		return 0;
	}
	p = 1; k = 0;
	while(n || m) {
		if(!p || (k < 2 && m > n)) {
			++k, p = 1, --m;
		}
		else {
			k = 0, p = 0, --n;
		}
		printf("%d", p);
	}
	return 0;
}
