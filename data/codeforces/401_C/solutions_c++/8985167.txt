#include<iostream>
using namespace std;

int main()
{
    long long int one,zero;
    cin>>zero>>one;
    if(zero>one+1 || (zero<(one-1)/2)) {cout<<-1<<endl;return 0; }

    while(zero||one)
    {
        if(zero<=one) if(one>0) {cout<<1;one--; }
        if(one>zero) if(one>0) {cout<<1;one--; }
        if(zero>0) {cout<<0;zero--; }
    }
}
