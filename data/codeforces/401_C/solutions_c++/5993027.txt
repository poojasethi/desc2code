#include <iostream>
using namespace std;

int main()
{
  int n,m;
  cin >> n >> m;
  if(n * 2 + 2 < m || n > m+1){
	cout << "-1" << endl;
	return 0;
  }
  if(n > m){
	cout << "0";
	n--;
  }
  while(m > n && n != 0){
	cout << "110";
	m -= 2;
	n--;
  }
  while(n!=0){
	cout << "10";
	m--;
	n--;
  }
  while(m!=0){
	cout << "1";
	m--;
  }
  cout << endl;
  return 0;
}
