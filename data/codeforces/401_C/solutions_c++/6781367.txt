#include<iostream>
using namespace std;
int main()
{
	int x,y;
	cin>>x>>y;
	bool now=true;
	if(x>y+1||x*2+2<y)
	{
		cout<<-1;
		return 0;
	}
	else if(x==y+1)now=false;
	while(x>0||y>0)
	{
		if(now)
		{
			if(x+1<y)cout<<"11",y-=2;
			else cout<<"1",y--;
			now=!now;
		}
		else cout<<"0",x--,now=!now;
	}
}
