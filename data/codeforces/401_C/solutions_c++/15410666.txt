#include <cstdio>
int N,M;

int main()
{
	scanf("%d%d",&N,&M);
	if (M<N-1||M>2*N+2) return puts("-1"),0;
	if (M==N-1)
	{
		for (int i=1; i<=M; i++) printf("01");
		return puts("0"),0;
	}
	for (;N&&M>N;) printf("110"),M-=2,N--;
	for (;M||N;)
	{
		if (M) putchar('1'),M--;
		if (N) putchar('0'),N--;
	}
}
