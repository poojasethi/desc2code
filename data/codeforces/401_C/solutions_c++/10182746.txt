#include<stdio.h>

int main(){
    
	int s,b,k1 = 0,k2 = 0;
	scanf("%d %d",&s,&b);

	if(s > b+1 || b > s*2+2){
		printf("-1");
		return 0;
	}

	while(1){

		if(b >= s && b > 0 && k2 < 2) {
			printf("1"); 
			b--; 
			k2++;
			k1 = 0;
		}
		else{
			printf("0"); 
			s--; 
		   k1++;
			k2 = 0;
		}
		if(s == 0 && b == 0) 
				break;
	
	}

	return 0;
}
