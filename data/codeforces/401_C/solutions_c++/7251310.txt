#include<bits/stdc++.h>
using namespace std;
int main()
{
	int m,n,c=0,v=1;
	cin>>n>>m;
	if(n<=m+1 && m<=2*n+2)
	{
		while(m>0 || n>0)
		{
			if(!v || (m>n && c<2))
			{
				m--;
				v=1;
				c++;
			}
			else
			{
				n--;
				v=c=0;
			}
			putchar(v+'0');
		}
	}
	else
		cout<<"-1";
	cout<<endl;
	return 0;
}
