#include <iostream>
#include <cstdio>
using namespace std;

#define PUT1 if (m>0) {putchar('1'); m--;}

int main()
{
	int n, m;
	scanf("%d%d", &n, &m);
	if (m<n-1 || m>2*n+2) cout<<-1<<endl;
	else {
		m = m-n+1;
		PUT1; PUT1;
		while (--n) {
			putchar('0'); putchar('1');
			PUT1;
		}
		putchar('0'); PUT1; PUT1; putchar('\n');
	}
	return 0;
}
