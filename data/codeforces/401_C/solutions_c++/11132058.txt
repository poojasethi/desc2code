#include<cstdio>
using namespace std;
int main(){
	int n,m,k=0;
	scanf("%d%d",&n,&m);
	if(m>=n-1&&m<=2*(n+1)){
		while(m>n-1&&m>0){
			if(k!=2){
			printf("1");k++;m--;
		}else{
			printf("0");k=0;n--;
		}}
		if(m==n-1){
			for(int i=1;i<n;i++)printf("01");
			printf("0");}
	}else printf("-1");
	return 0;
}
