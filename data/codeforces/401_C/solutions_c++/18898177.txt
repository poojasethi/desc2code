#include<bits/stdc++.h>

using namespace std;
int main()
{
	int m,n,a=0,c=1;
	cin>>n>>m;
	if(n<=m+1 && m<=2*n+2)
	{
		while(m>0 || n>0)
		{
			if(!c || (m>n && a<2))
			{
				m--;
				c=1;
				a++;
			}
			else
			{
				n--;
				c=a=0;
			}
			cout<<c;
		}
	}
	else
		cout<<"-1";
	cout<<endl;
	return 0;
}
