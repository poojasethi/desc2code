#include<bits/stdc++.h>
using namespace std;
int main()
{
    int o,z;cin>>z>>o;
    if(z-1>o||o-2>z*2)
    {
        printf("-1");return 0;
    }
    if(z>o){printf("0");z--;}
    while(o>z&&z>0){printf("110");o-=2;z--;}
    while((z>0 && o>0)){printf("10");o--;z--;}
    while(o>0){printf("1");o--;}
}
