#include <iostream>
using namespace std;
int main() {
    int n, m;
    cin >> n >> m;
    if (n - 1 > m || m > 2 * (n + 1)) { cout << -1; return 0; }
    int p = 1, k = 0;
    while (n || m) {
        if (!p || (k < 2 && m > n))
            {k++  , p = 1, m--;}
        else 
            {k = 0; p = 0; n--;}
    cout << p;
    }
    return 0;
}
