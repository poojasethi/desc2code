#include <cstdio>
int n,m;
int main(){
    scanf("%d%d",&n,&m);
    if (m>2*(n+1) || m<n-1) printf("-1");
    else{
        int cnt=m-n;
        if (cnt==-1) {n--; printf("0");}
        for (int i=0; i<n; i++, cnt--) printf("%d",cnt>0 ? 110:10);
        for (int i=0; i<cnt; i++) printf("%d",1);
    }
    return 0;
}
