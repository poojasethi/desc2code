#include<cstdio>
using namespace std;
int main()
{
	int n,m;

	scanf("%d %d",&n,&m);
	if(n>m+1) printf("-1\n");
	else if(2*n+2<m) printf("-1\n");
	else {
		if(n==m+1) printf("0"),n--;
		while(m>n+2)
		{
			printf("110");
			m-=2,n--;
		}
		while(m>0||n>0)
		{
			if(m) printf("1"),m--;
			if(n) printf("0"),n--;
		}
		printf("\n");
	}
}
