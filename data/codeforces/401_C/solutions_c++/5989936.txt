#include<stdio.h>

inline bool check(int m, int n) {
  return m<=2*(n+1) && m>=n-1;
}

int main() {
  int n,m;
  scanf("%d%d", &n,&m);
  if (!check(m,n)) { printf("-1"); return 0; }
  int r1=0;
  while (m+n) {
    if (r1<2 && check(m-1, n)) { putchar('1'); r1++; m--;}
    else { putchar('0'); r1=0; n--; }
  }
}
