#include<cstdio>
#include<cstdlib>
#include<cstring>
using namespace std;
typedef long long ll;
const int mod=1000000007;
int a[3000],f[3000],v[3000];
int x,y,n;
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i)
	{
		scanf("%d",&a[i]);
		if (a[i]!=-1) v[a[i]]=1;
	}
	for (int i=1;i<=n;++i)
		if (a[i]==-1)
			if (v[i])
				++x;
			else ++y;
	f[0]=1;
	for (int i=1;i<=x;++i) f[0]=f[0]*(ll)i%mod;
	for (int i=1;i<=y;++i)
	{
		f[i]=(x+i-1)*(ll)f[i-1]%mod;
		if (i>1) f[i]=(f[i]+(i-1)*(ll)f[i-2]%mod)%mod;
	}
	printf("%d\n",f[y]);
}
