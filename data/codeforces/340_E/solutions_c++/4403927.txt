#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<vector>
#include<set>
#include<queue>
#include<cmath>
using namespace std;

const int maxn = 2100;
#define pb push_back
#define ll long long
#define FI first
#define SE second
#define mod 1000000007
ll dp[maxn][maxn];
int a[maxn];

int main(){
    int n,s1,s2;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	s1=s2=0;
	for(int i=1;i<=n;i++){
		s1+=(a[i]==-1);
		if(a[i]!=-1)
		s2+=(a[a[i]]==-1);
	}
	
	memset(dp,0,sizeof dp);
	dp[0][0]=1;
	for(int i=1;i<=s1;i++)dp[i][0]=(dp[i-1][0]*i)%mod;

	for(int i=1;i<=s1;i++){
		for(int j=1;j<=i;j++){
			dp[i][j]=(dp[i][j-1]-dp[i-1][j-1])%mod;
		}
	} 
    cout<<(dp[s1][s1-s2]+mod)%mod<<endl;
	return 0;
}
