#include <iostream>
using namespace std;

#define mod 1000000007
long long c[2001][2001];
int a[2010], n, k, t;

int main() {
    cin>>n;
    for (int i=0; i<n; i++) cin>>a[i];
    for (int i=0; i<n; i++) if (a[i]==-1) k++;
    t = k;
    for (int i=0; i<n; i++) if (a[i]!=-1 && a[a[i]-1]==-1) t--;
    c[0][0] = 1;
    for (int i=1; i<=n; i++) c[i][0] = (c[i-1][0]*i)%mod;
    for (int i=1; i<=n; i++) {
        for (int j=1; j<=i; j++) {
            c[i][j] = (c[i][j-1]-c[i-1][j-1]+mod)%mod;
        }
    }
    cout<<c[k][t]<<endl;
    return 0;
}
