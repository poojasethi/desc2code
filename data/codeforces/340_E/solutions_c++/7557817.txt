#include<bits/stdc++.h>
#define mod 1000000007
#define ll long long
using namespace std;
int a[2010];
ll f[2009][2009];
int main()
{
	int i,n,x=0,y=0;
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>a[i];
	for(i=1;i<=n;i++)
		if(a[i]==-1)
			x++;
	for(i=1;i<=n;i++)
		if(a[i]!=-1&&a[a[i]]==-1)
			y++;
	f[0][0]=1;
	for(i=1;i<=x;i++) f[i][0]=(f[i-1][0]*i)%mod;
	for(int i=1;i<=x;i++)
        for(int j=1;j<=i;j++)
            f[i][j]=(f[i][j-1]-f[i-1][j-1])%mod;
    printf("%d",(int)((f[x][x-y]+mod)%mod));
}
