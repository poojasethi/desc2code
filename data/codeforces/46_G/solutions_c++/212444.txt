int get(ii x) {
	re sqr(x.fi) + sqr(x.se);
}

bool l1(ii a, ii b) {
	re get(a) < get(b);
}

bool l2(int a, int b) {
	re v[a].fi > v[b].fi;
}

bool l3(int a, int b) {
	re v[a].se > v[b].se;
}

bool l4(int a, int b) {
	double tmp = atan2(v[a].fi, v[a].se) - atan2(v[b].fi, v[b].se);
	if (abs(tmp) > 1e-8) {
		re tmp > 0;
	}
	else
		re get(v[a]) > get(v[b]);
}
