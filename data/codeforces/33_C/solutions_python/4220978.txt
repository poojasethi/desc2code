n = int(raw_input())
data = list(map(int,raw_input().split()))
dp1 = [0]*(n+1)
n_total = 0
for i in xrange(1,n+1):
    n_total -= data[i-1]
    dp1[i] = max(dp1[i-1] + data[i-1], n_total)
n_total = 0
ans = (-1)*(10**4)
#print dp1
for i in xrange(n,0,-1):
    ans = max(n_total+dp1[i], ans)
    n_total -= data[i-1]
print ans
