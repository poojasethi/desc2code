n = input()
a1 = map(int, raw_input().split())

s1 = [0] * n
s1[0] = a1[0]
for i in xrange(1, n):
  s1[i] = s1[i - 1] + a1[i]

d1 = [0] * (n + 1)
d1[1] = abs(a1[0])
for i in xrange(1, n):
  d1[i + 1] = max(d1[i] + a1[i], -s1[i])
  
a2 = list(reversed(a1))

s2 = [0] * n
s2[0] = a2[0]
for i in xrange(1, n):
  s2[i] = s2[i - 1] + a2[i]

d2 = [0] * (n + 1)
d2[1] = abs(a2[0])
for i in xrange(1, n):
  d2[i + 1] = max(d2[i] + a2[i], -s2[i])
  
print max((x + y for x, y in zip(d1, reversed(d2))))
