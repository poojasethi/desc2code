def readint(): return int(raw_input())
def readarray(foo): return [foo(x) for x in raw_input().split()]

def run():
	n = readint()
	a = readarray(int)
	sums = [0]
	acc = 0
	res = 0
	for i in xrange(len(a)):
		sums.append(acc)
		acc += a[i]
	sums.append(acc)
	for i in xrange(1, len(sums)):
		sums[i] = min(sums[i], sums[i - 1])

	acc = 0
	for i in xrange(len(a), -1, -1):
		res = min(res, acc + (sums[i]))
		if i == 0: break
		acc += a[i - 1]
	return sum(a) - res - res
