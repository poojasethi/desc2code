from itertools import *

n = input()
a1 = map(int, raw_input().split())

d1 = [0] * (n + 1)
d2 = [0] * (n + 1)

s = a1[0]
d1[1] = abs(a1[0])
for i in xrange(1, n):
  s += a1[i]
  d1[i + 1] = max(d1[i] + a1[i], -s)
  
a2 = list(reversed(a1))

s = a2[0]
d2[1] = abs(a2[0])
for i in xrange(1, n):
  s += a2[i]
  d2[i + 1] = max(d2[i] + a2[i], -s)
  
print max((x + y for x, y in izip(d1, reversed(d2))))
