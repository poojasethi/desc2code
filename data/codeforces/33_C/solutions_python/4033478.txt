n = int(raw_input())
data = list(map(int, raw_input().split()))
table = [0]
mm = 0
total = 0
for i in xrange(0, n):
    total += data[i]
    mm = max(mm, (-2)*total)
    table.append(total+mm)

table2 = [0]*(n+1)
mm = 0
totalx = 0
for i in xrange(n-1, -1, -1):
    totalx += data[i]
    mm = max(mm, (-2)*totalx)
    table2[i] = totalx+mm
#print table
#print table2

for i in xrange(0,n+1):
    table[i] += table2[i]
