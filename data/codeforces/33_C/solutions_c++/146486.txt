#include <cstdio>
using namespace std;

int main() {
	int a, n, s, small, best = 0;
	scanf("%d %d", &n, &s);
	small = s;
	if (s > 0) {
		best = s;
		small = 0;
	}
	for (int i = 1; i < n; i++) {
		scanf("%d", &a);
		s += a;
		int diff = s - small;
		if (diff > best) best = diff;
		if (s < small) small = s;
	}
	
	int res = 2*best - s;
	printf("%d\n", res);
	
	return 0;
}
