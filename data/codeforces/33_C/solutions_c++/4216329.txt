#include<stdio.h>
int main()
{
	int N,i,number[100005],sum=0,max=0,sum1=0;
	scanf("%d",&N);
	for(i=0;i<N;i++)
	{
		scanf("%d",&number[i]);
		sum+=number[i];
	}
	
	for(i=0;i<N;i++)
	{
		sum1+=number[i];
		if(sum1<0)  sum1=0;
		if(sum1>max) max=sum1;
	}
	
	printf("%d\n",2*max-sum);
	return 0;
}
