#include<iostream>
using namespace std;
int a[100010];
int main(){
    int n;cin>>n;
    int sum=0;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        sum+=a[i];
    }
    int g=0;
    int ans=0;
    for(int i=1;i<=n;i++)
    {
        g+=a[i];
        if(g<0)g=0;
        else ans=max(ans,g);
    }
    cout<<2*ans-sum;
}
