#include <cstdio>
#define max(a,b) (a)>(b)?(a):(b)
int main(int argc, char const *argv[])
{
	int n, a;
	int s = 0, ps = 0, mps = 0;
	scanf("%d", &n);
	while (n--) {
		scanf("%d", &a);
		s += a;
		ps += a;
		if (mps < ps)
			mps = ps;
		if (ps < 0)
			ps = 0;
	}
	int r = -(s - mps) + mps;
	r = max(r, (s > 0)? s : -s);
	printf("%d\n", r);
	return 0;
}
