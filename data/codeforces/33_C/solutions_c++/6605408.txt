#include<iostream>
using namespace std;
long long ar[100009];

int main()
{
 long long n,i,min=0,res=0;

 cin>>n;

 for(i=1;i<=n;i++)
   {
    cin>>ar[i];
    ar[i]+=ar[i-1];
    if(ar[i]<min)
     min=ar[i];
    if(2*(ar[i]-min)>res)
     res=2*(ar[i]-min);
   }

 res-=ar[n];
 cout<<res<<endl;
}
