#include<cstdio>
int a[100010];
int main(){
	int n,tmpsum=0,csum=0,totsum=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
		//printf("%d\n",csum);
		totsum+=a[i];
		csum+=a[i];
		if(tmpsum<csum){
			tmpsum=csum;
		}
		if(csum<0){
			csum=0;
		}
	}
	printf("%d\n",2*tmpsum-totsum);
	return 0;
}
