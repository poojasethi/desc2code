#include <iostream>
#include <vector>

using namespace std;

int main(int argc, char* argv[])
{
	int n;
	cin >> n;
		
	vector<int> v(n);
	int s = 0;
	for (int i = 0; i < n; ++i)
	{
		cin >> v[i];
		s += v[i];
	}
	
	int c = 0, m = 0;
	for (int i = 0; i < n; ++i)
	{
		c = max(0, c + v[i]);
		m = max(m, c);
	}
	
	cout << 2 * m - s << endl;
	return 0;
}
