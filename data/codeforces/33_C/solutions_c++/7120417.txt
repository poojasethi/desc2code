#include<stdio.h>  
int main()  
{  
    int N,i,sum=0,number,max=0,summax=0;  
    scanf("%d",&N);  
    for(i=1;i<=N;i++)  
     {  
        scanf("%d",&number);  
        sum+=number;
                summax+=number;
        if(summax<0) summax=0;  
        if(summax>max) max=summax;  
     }  
     printf("%d\n",2*max-sum);  
}
