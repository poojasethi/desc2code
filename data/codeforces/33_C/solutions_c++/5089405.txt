#include<iostream>

using namespace std;

int main()
{
    int n,maxi=0,sum=0,sumkol=0;
    cin>>n;
    for(int i=0; i<n; i++)
    {
        int y;
        cin>>y;
        sumkol+=y;
        sum+=y;
        maxi=max(maxi,sum);
        if(sum<0)
            sum=0;
    }
   // cout<<maxi<<" "<<sumkol<<endl;
    cout<<-sumkol+2*maxi<<endl;
}
