#include <cstdio>

int main() {
    int n,ms1=0,s=0,sum=0,ai;
    scanf("%d",&n);
    for (int i=0; i<n; i++) {
        scanf("%d",&ai);
        s+=ai;
        sum+=ai;
        if (sum<0)
            sum=0;
        if (sum>ms1)
            ms1=sum;
    }
    printf("%d\n",ms1+ms1-s);
    return 0;
}
