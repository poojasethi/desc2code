#include<iostream>
#include<stdio.h>
#include<cstdio>
using namespace std;
int main()
{
	int n;
	scanf("%d",&n);
	int a[n];
	int i;
	long long int sum=0;
	long long int mid=0;
	long long int max=0;
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		sum+=a[i];
		mid+=a[i];
		if(mid>max)
			max=mid;
		if(mid<0)
			mid=0;
	}
	printf("%lld\n",sum-2*(sum-max));
	return 0;
}
