#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll cnt=0,sum=0,temp=0,x;
    int n;
    cin>>n;
    while(n--)
    {
        cin>>x;
        sum+=x;
        temp+=x;
        if(temp<0) temp=0;
        cnt=max(cnt,temp);
    }
    
    cout<<2*cnt-sum;
   
    
    
    return 0;
}
