#include <iostream>
#include <cstdio>
using namespace std;

int N,T,MIN=0,ANS=0,C;
int SUM;

int main(){
   // freopen("in.txt","r",stdin);
    cin >> N;
    while (N--){
        cin >> T;
        SUM+=T;
        C = max(SUM-MIN,C);
        MIN = min(SUM,MIN);
    }
    cout << -SUM+(C<<1) << endl;
}
