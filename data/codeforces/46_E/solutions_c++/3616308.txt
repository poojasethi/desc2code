#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define ll long long 
ll n,m;
int a[2000][2000];
ll sum[2000][2000];
ll dp[2000][2000];
int main(){
#ifdef LOCAL
	freopen("e.txt","r",stdin);
#endif
	cin>>n>>m;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			scanf("%d",a[i]+j);
		}
	}
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			sum[i][j]=sum[i][j-1]+a[i][j];
		}
	}
	for(int i=1;i<=n;i++){
		if(i%2){//从小到大
			ll mm=-2e18;
			for(int j=2;j<=m;j++){
				mm=max(mm,dp[i-1][j-1]);
				dp[i][j]=mm+sum[i][j];
			}
		}else{//从达到小
			ll mm=-2e18;
			for(int j=m-1;j>=1;j--){
				mm=max(mm,dp[i-1][j+1]);
				dp[i][j]=mm+sum[i][j];
			}
		}
	}
	ll ans=-2e18;
	if(n%2){
		for(int i=2;i<=m;i++){
			ans=max(ans,dp[n][i]);
		}
	}else{
		for(int i=1;i<m;i++){
			ans=max(ans,dp[n][i]);
		}
	}
	cout<<ans<<endl;
}
