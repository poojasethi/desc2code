#include "cstdio"
#include "iostream"
#include "cstdlib"
#include "cstring"

using namespace std;

typedef long long ll;

int n, m;
int a[1555][1555];
ll sum[1555][1555];
ll f[1555][1555], f1[1555][1555], f2[1555][1555];

int main() {
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= m; j++) {
			scanf("%d", &a[i][j]);
			sum[i][j] = sum[i][j-1] + a[i][j];
		}

	for(int i = 1; i <= n; i++) {
		for(int j = 1; j <= m; j++) 
			if (i&1) f[i][j] = f1[i-1][j-1] + sum[i][j];
			else f[i][j] = f2[i-1][j+1] + sum[i][j];
		f1[i][0] = -1e18;
		for(int j = 1; j <= m; j++)
			f1[i][j] = max(f1[i][j-1], f[i][j]);
		f2[i][m+1] = -1e18;
		for(int j = m; j >= 1; j--)
			f2[i][j] = max(f2[i][j+1], f[i][j]);
	}
		
	printf("%I64d", f1[n][m]);

	return 0;
}
