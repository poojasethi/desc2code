#include <bits/stdc++.h>
#define pb push_back
#define mp make_pair
using namespace std;
typedef vector<int> vi;
typedef pair<int,int> pii;

long long sum[1500+15][1500+15];
long long dp[1500+15][1500+15];

int main()
{
	ios_base::sync_with_stdio(false);
	int n,m,x,y;  scanf("%d%d",&n,&m);
	for( x=0;x<n;++x)
	{
		for( y=0;y<m;++y)
			{ scanf("%I64d",&sum[x][y]); if(y) sum[x][y]+=sum[x][y-1]; }
	}
	for( x=0;x<n;++x)
	{
		if(x&1) {
			for( y=0;y<m-1;++y)
			{
				dp[x][y]=sum[x][y]+dp[x-1][y+1];
				if(y&&dp[x][y-1]>dp[x][y]) dp[x][y]=dp[x][y-1];
			}
		}
		else{
			for( y=m-1;y;--y)
			{
				dp[x][y]=sum[x][y];
				if(x)dp[x][y]+=dp[x-1][y-1];
				if(y<m-1&&dp[x][y+1]>dp[x][y]) dp[x][y]=dp[x][y+1];
			}
		}
	}
	if(n&1) cout << dp[n-1][1] << endl;
	else cout << dp[n-1][m-2] << endl;
}
