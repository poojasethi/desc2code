#include<cstdio>
#include<cstring>
#include<iostream>
#include<string>

using namespace std;

typedef long long LL;

const int Mod =  1e9+7;
int n;
string s;
string t[111111];
LL add[11],shift[11];



int main()
{
    cin>>s;
    cin>>n;
    for (int i=0; i<10; i++)
    {
        add[i] = i;
        shift[i] = 10;
    }
    for (int i=1; i<=n; i++)
        cin>>t[i];
    for (int i=n; i>=1; i--)
    {
        LL ta = 0,  ts = 1;
        for (int j= t[i].size()-1; j>=3; j--)
        {
            ta  = ( ta + add[ t[i][j]-'0'] * ts ) % Mod;
            ts  = ( ts * shift[ t[i][j]-'0'] )  % Mod;
        }
        int x = t[i][0] - '0';
        add[x] = ta;
        shift[x] =  ts;
    }
    LL ans = 0;
    for(int i=0;i<s.size();++i)
        ans=(ans*shift[s[i]-'0']+add[s[i]-'0'])%Mod;
    cout<<ans<<endl;
    return 0;
}
