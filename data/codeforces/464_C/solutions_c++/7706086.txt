#include <bits/stdc++.h>
using namespace std;

#define sz(v) int(v.size())
#define endl '\n'
typedef long long ll;
typedef pair<int,int> pii;

const int MOD = 1000*1000*1000+7;
const int MAXn = 1e5+10;
ll mod[10], coef[10];
string q[MAXn];

int main() {
	ios::sync_with_stdio(false);
	string s;
	cin >> s;
	int n;
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> q[i];
	for(int i = 0; i < 10; i++) {
		coef[i] = 10;
		mod[i] = i;
	}
	for(int i = n-1; i >= 0; i--) {
		int d = q[i][0]-'0';
		ll m = 0, c = 1;
		for(int j = 3; j < (int)q[i].size(); j++) {
			int e = q[i][j]-'0';
			c = c * coef[e] % MOD;
			m = m * coef[e] % MOD + mod[e];
			m %= MOD;
		}
		mod[d] = m;
		coef[d] = c;
	}
	ll res = 0;
	for(int i = 0; i < (int)s.size(); i++) {
		int d = s[i]-'0';
		res = res * coef[d] % MOD + mod[d];
		res %= MOD;
	}
	cout << res << endl;
	return 0;
}
