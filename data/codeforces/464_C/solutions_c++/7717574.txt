#include <bits/stdc++.h>
using namespace std;
#define rep(i,a,n) for (int i=a;i<n;i++)
#define per(i,a,n) for (int i=n-1;i>=a;i--)
#define SZ(x) ((int)(x).size())
typedef long long ll;
const ll mod=1000000007;

const int N=101000;
string t[N];
char s[N];
ll pwd[20],val[20];
int n;
int main() {
	scanf("%s",s);
	scanf("%d",&n);
	rep(i,0,n) cin>>t[i];
	rep(i,0,10) pwd[i]=10,val[i]=i;
	per(i,0,n) {
		int l=t[i][0]-'0';
		ll pw=1,va=0;
		rep(j,3,SZ(t[i])) {
			int r=t[i][j]-'0';
			va=(va*pwd[r]+val[r])%mod;
			pw=pw*pwd[r]%mod;
		}
		pwd[l]=pw; val[l]=va;
	}
	ll va=0;
	int lenn=strlen(s);
	rep(i,0,lenn) {
		int r=s[i]-'0';
		va=(va*pwd[r]+val[r])%mod;
	}
	printf("%I64d\n",va);
}
