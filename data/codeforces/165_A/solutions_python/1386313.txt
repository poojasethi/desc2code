pt = [map(int, raw_input().split()) for i in range(input())]
res = 0
for i in pt:
	r = 0
	l = 0
	u = 0
	d = 0
	for j in pt:
		if i != j:
			if j[0] < i[0] and j[1] == i[1] : l += 1
			if j[0] > i[0] and j[1] == i[1] : r += 1
			if j[1] < i[1] and j[0] == i[0] : u += 1
			if j[1] > i[1] and j[0] == i[0] : d += 1
	if l > 0 and r > 0 and u > 0 and d > 0:
		res += 1
print res
