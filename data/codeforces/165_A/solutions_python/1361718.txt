n = input()
p = []
for i in range(n):
    p.append(map(int, raw_input().split()))
ans = 0
for x, y in p:
    f = 0
    for a, b in p:
        if a == x:
            if b < y: f |= 1
            if b > y: f |= 2
        if b == y:
            if a < x: f |= 4
            if a > x: f |= 8
    if f == 15:
        ans += 1
print ans
