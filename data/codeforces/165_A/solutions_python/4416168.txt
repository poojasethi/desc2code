n = int(raw_input())

xs = {}
ys = {}

xys = []

for i in xrange(n):
	x, y = map(lambda x: int(x), raw_input().split())
	xys += [[x, y]]
	
for i in xys:
	x, y = i
	if x not in xs:
		xs[x] = [y]
	else:
		xs[x] += [y]
		xs[x] = sorted(xs[x])
	if y not in ys:
		ys[y] = [x]
	else:
		ys[y] += [x]
		ys[y] = sorted(ys[y])

ans = 0

for i in xys:
	x, y = i
	if min(ys[y]) != x and max(ys[y]) != x and min(xs[x]) != y and max(xs[x]) != y:
	   	ans += 1

print ans
