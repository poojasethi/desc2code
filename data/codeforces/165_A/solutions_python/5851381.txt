import sys
lines=sys.stdin.readlines()
a=[]
cnt=0
res=0
for line in lines[1:]:
    a.append(tuple([int(i) for i in line.split(' ')]))
for (i,j) in a:
    cnt=0
    for (u,v) in a:
        if u<i and j==v:
            cnt|=(1<<0)
        if u>i and j==v:
            cnt|=(1<<1)
        if u==i and j<v:
            cnt|=(1<<2)
        if u==i and j>v:
            cnt|=(1<<3)
    if cnt==15:
        res+=1
print res
