n = int(raw_input())
r = [[int(c) for c in raw_input().split()] for i in xrange(n)]
ans = 0
for x,y in r:
	code = 0
	for X,Y in r:
		if x > X and y == Y:
			code |= 8
		if x < X and y == Y:
			code |= 4
		if x == X and y > Y:
			code |= 2
		if x == X and y < Y:
			code |= 1
	if code == 15:
		ans += 1
print ans
