n = int(raw_input())
data = []
for i in range(n):
    data.append(map(int,raw_input().split()))
ans=0
for i in data:
    up,down,left,right=[0,0,0,0]
    for j in data:
        if i[0]==j[0] and i[1]<j[1]: up=1
        elif i[0]==j[0] and i[1]>j[1]: down=1
        elif i[0]>j[0] and i[1]==j[1]: left=1
        elif i[0]<j[0] and i[1]==j[1]: right=1
    if up==1 and down==1 and left==1 and right==1:ans=ans+1
print ans;
