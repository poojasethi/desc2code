n=input()
lst=[]
for i in range(n):
    lst.append(map(int,raw_input().split()))

count=0
for i in lst:
    l=0
    r=0
    u=0
    d=0
    for j in lst:
        if j!=i:
            if j[1]==i[1] and j[0]<i[0]: l+=1
            if j[1]==i[1] and j[0]>i[0]: r+=1
            if j[0]==i[0] and j[1]>i[1]: d+=1
            if j[0]==i[0] and j[1]<i[1]: u+=1

    if l>0 and r>0 and u>0 and d>0:
        count+=1
print count
