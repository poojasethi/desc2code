n=int(raw_input())
a=[]
for i in xrange(0,n):
	x,y=map(int,raw_input().split())
	a.append([x,y])
c=0
for i in xrange(0,n):
	le=0
	ri=0
	lo=0
	up=0
	for j in xrange(0,n):
		if j==i:
			continue
		if a[i][0]==a[j][0]:
			if a[i][1]<a[j][1]:
				lo+=1
			else:
				up+=1
		elif a[i][1]==a[j][1]:
			if a[i][0]<a[j][0]:
				le+=1
			else:
				ri+=1
		#print a[i],a[j],lo,up,le,ri
		if lo>=1 and up>=1 and le>=1 and ri>=1:
			c+=1
			break
print c
