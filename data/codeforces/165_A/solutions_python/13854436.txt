maxx = 2001 * [0]
minx = 2001 * [2001]
maxy = 2001 * [0]
miny = 2001 * [2001]
n = int(raw_input())
a =[]
for _ in range(n):
    x, y = map(int, raw_input().split())
    x += 1000
    y += 1000
    a.append((x, y))
    maxx[y] = max(maxx[y], x)
    minx[y] = min(minx[y], x)
    maxy[x] = max(maxy[x], y)
    miny[x] = min(miny[x], y)
ans = 0
for x, y in a:
    if x < maxx[y] and x > minx[y] and y < maxy[x] and y > miny[x]:
        ans += 1
print ans
