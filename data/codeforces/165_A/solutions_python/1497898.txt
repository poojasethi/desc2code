n = int(raw_input())
tmp = [raw_input().split() for i in xrange(n)]
a = [(int(x), int(y)) for (x, y) in tmp]
ans = 0
for i in a:
	t = [0,0,0,0]
	for j in a:
		if i[0] == j[0]:
			if j[1] < i[1]: t[0] = 1
			elif j[1] > i[1]: t[1] = 1
		elif i[1] == j[1]:
			if j[0] < i[0]: t[2] = 1
			elif j[0] > i[0]: t[3] = 1
	if t == [1,1,1,1]:
		ans += 1
print(ans)
