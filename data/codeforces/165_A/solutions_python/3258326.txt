I=lambda:map(int, raw_input().split())
n=input()
l,r,u,d=1001,-1001,1001,-1001
a=[I() for _ in xrange(n)]
cnt=0
for x,y in a:
    l,r,u,d=0,0,0,0
    for x0,y0 in a:
            if y0<y and x==x0 : l += 1
            if y0>y and x==x0 : r += 1
            if x0<x and y==y0 : u += 1
            if x0>x and y==y0 : d += 1
    if l > 0 and r > 0 and u > 0 and d > 0:
        cnt+=1
print cnt
