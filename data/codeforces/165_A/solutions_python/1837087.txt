n = input()
s = [ map(int, raw_input().split()) for i in range(n) ]
res = 0
for x,y in s:
    a = 0; b = 0; c = 0; d = 0
    for anoth_x,anoth_y in s:
        if anoth_x > x and anoth_y == y: a = 1
        if anoth_x < x and anoth_y == y: b = 1
        if anoth_x == x and anoth_y < y: c = 1
        if anoth_x == x and anoth_y > y: d = 1
    if a and b and c and d: res += 1
print res
