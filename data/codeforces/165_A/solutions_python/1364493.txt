#!/usr/bin/env python

import sys,os

n=int(raw_input())
x=[]
y=[]
r=[]
count=0

for i in range(0,n):
  s=raw_input()
  x.append(int(s.split()[0]))
  y.append(int(s.split()[1]))
  r.append(0)
  
for i in range(0,n):
  for j in range(0,n):
    if i!=j:
      if x[i]==x[j]:
	if y[i]>y[j]:
	  r[i]|=1
	if y[i]<y[j]:
	  r[i]|=2
      if y[i]==y[j]:
	if x[i]>x[j]:
	  r[i]|=4
	if x[i]<x[j]:
	  r[i]|=8	
	
for i in range(0,n):
  if r[i]==15:
    count+=1
    
print count
