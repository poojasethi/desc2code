n = input()
minstr,maxstr,mincol,maxcol = {},{},{},{}

s = [map(int,raw_input().split()) for _ in range(n)]

for i in range(n):
	x = s[i][0]
	y = s[i][1]
	minstr[x]  = min(minstr.get(x,1001), y) 
	maxstr[x]  = max(maxstr.get(x,-1001), y)
	mincol[y]  = min(mincol.get(y,1001), x)
	maxcol[y]  = max(maxcol.get(y,-1001), x)
ans = 0	
for i in range(n):
	x = s[i][0]
	y = s[i][1]
	if minstr[x] < y < maxstr[x] and mincol[y]<x < maxcol[y] : ans +=1
	
print ans
