coor = [map(int, raw_input().split()) for i in [0]*input()]
num = 0
for i in coor:
    if filter(lambda x:x[0]>i[0] and x[1]==i[1], coor) and filter(lambda x:x[0]<i[0] and x[1]==i[1], coor) and filter(lambda x:x[0]==i[0] and x[1]<i[1], coor) and filter(lambda x:x[0]==i[0] and x[1]>i[1], coor):
        num += 1
print num
