n = input()
points = []
for i in range(n):
	p = map(int, raw_input().split(" "))
	points.append(p)

def check(n, points):
	count = 0
	for X,Y in points:
		if any([y<Y for x,y in points if x==X]) and any([y>Y for x,y in points if x==X]) and any([x<X for x,y in points if y==Y]) and any([x>X for x,y in points if y==Y]):
			count += 1
	return count

print check(n,points)
