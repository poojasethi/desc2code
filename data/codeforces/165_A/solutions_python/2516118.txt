def valid(p):
	global n, point
	cnt = 0
	for i in range(n):
		if p[0] == point[i][0]:
			if p[1] < point[i][1]:
				cnt |= 1
			elif p[1] > point[i][1]:
				cnt |= 2
		elif p[1] == point[i][1]:
			if p[0] < point[i][0]:
				cnt |= 4
			elif p[0] > point[i][0]:
				cnt |= 8
	return cnt == 15

n = input()
point = []
for i in range(n):
	point.append(map(int, raw_input().split()))
print len([1 for p in point if valid(p)])
