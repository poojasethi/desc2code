pts=[map(int,raw_input().split()) for _ in [0]*input()]
ans=0
for xi,yi in pts:
    l,r,u,d = [0]*4
    for xj,yj in pts:
        if xi == xj:
            if yi > yj:
                u = 1
            elif yi < yj:
                d = 1
        elif yi == yj:
            if xi > xj:
                l = 1
            elif xi < xj:
                r = 1
    ans = ans + 1 if (l+r+u+d) == 4 else ans
print ans
