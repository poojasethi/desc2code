n=input()
enter=[map(int,raw_input().split()) for i in xrange(n)]
summ=0
for x,y in enter:
    a=0;b=0;c=0;d=0
    for xx,yy in enter:
        if xx>x and yy==y: a=1
        if xx<x and yy==y: b=1
        if xx==x and yy<y: c=1
        if xx==x and yy>y: d=1
    if a and b and c and d: summ+=1
print summ
