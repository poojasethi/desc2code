n = int(raw_input())
x = []
y = []
for i in range(0,n):
	a,b = map(int,raw_input().split(' '))
	x.append(a)
	y.append(b)
ans = 0
for i in range(0,n):
	l=r=u=d=0
	for j in range(0,n):
		if x[j] > x[i] and y[j] == y[i]:
			r = 1
		if x[j] < x[i] and y[j] == y[i]:
			l = 1
		if y[j] > y[i] and x[j] == x[i]:
			u = 1
		if y[j] < y[i] and x[j] == x[i]:
			d = 1
	if l == r == u == d == 1:
		ans = ans+1
print ans
