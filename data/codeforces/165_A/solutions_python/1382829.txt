N = int(raw_input())
x = []
y = []

for i in range(N):
    u,v = map(int, raw_input().split())
    x.append(u)
    y.append(v)

k = 0;

for i in range(N) :
    a = b = c = d = 0
    for j in range(N):
        a |= (x[j] < x[i] and y[j] == y[i])
        b |= (x[j] > x[i] and y[j] == y[i])
        c |= (x[j] == x[i] and y[j] < y[i])
        d |= (x[j] == x[i] and y[j] > y[i])
    if a and b and c and d :
        k = k + 1

print k
