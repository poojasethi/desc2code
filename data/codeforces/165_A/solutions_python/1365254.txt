n = input()
X = []
for i in xrange(n):
    X.append(map(int, raw_input().split()))

def issuper(a):
    q = 0
    for x, y in X:
        if a[0] == x:
            if a[1]<y: q|=1
            if a[1]>y: q|=2
        if a[1] == y:
            if a[0]<x: q|=4
            if a[0]>x: q|=8
    return q == 15

print len([x for x in X if issuper(x)])
