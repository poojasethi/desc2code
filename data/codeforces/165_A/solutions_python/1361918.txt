n=input()
x=[]
y=[]
for i in range(n):
  u,v=map(int,raw_input().split())
  x.append(u)
  y.append(v)
ok = [[0,0,0,0] for i in range(n)]
for i in range(n):
  for j in range(n):
    if i!=j:
      ok[i][0] |= x[i] == x[j] and y[i] < y[j]
      ok[i][1] |= x[i] == x[j] and y[i] > y[j]
      ok[i][2] |= x[i] < x[j] and y[i] == y[j]
      ok[i][3] |= x[i] > x[j] and y[i] == y[j]
      
r=0
for i in range(n):
  r += sum(ok[i])==4
print r
