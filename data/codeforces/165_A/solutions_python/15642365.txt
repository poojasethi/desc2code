n = input()
x = []
y = []
for i in range(n):
	a, b = map(int, raw_input().split())
	x.append(a)
	y.append(b)
ans = 0
for i in range(n):
	cnt = [0, 0, 0, 0]
	for j in range(n):
		if x[i] == x[j] and y[i] < y[j]:
			cnt[0] = 1
		elif x[i] == x[j] and y[i] > y[j]:
			cnt[1] = 1
		elif x[i] < x[j] and y[i] == y[j]:
			cnt[2] = 1
		elif x[i] > x[j] and y[i] == y[j]:
			cnt[3] = 1
	if 4 == sum(cnt): ans += 1
print ans
