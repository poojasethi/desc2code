n=int(raw_input())
a=list()
for i in xrange(n):
    a.append(tuple(map(int,raw_input().split())))
f1=set()
f2=set()
f3=set()
f4=set()    
for i in a:
    for j in a:
        if (i[0]==j[0]) and (i[1]>j[1]):
            f1.add(i)
        if (i[0]==j[0]) and (i[1]<j[1]):
            f2.add(i)
        if (i[1]==j[1]) and (i[0]>j[0]):
            f3.add(i)
        if (i[1]==j[1]) and (i[0]<j[0]):
            f4.add(i)
print len(set.intersection(f1,f2,f3,f4))
