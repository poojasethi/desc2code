coor = [map(int, raw_input().split()) for i in [0]*input()]
right = lambda i:filter(lambda x:x[0]>i[0] and x[1]==i[1], coor)
left = lambda i:filter(lambda x:x[0]<i[0] and x[1]==i[1], coor)
lower = lambda i:filter(lambda x:x[0]==i[0] and x[1]<i[1], coor)
upper = lambda i:filter(lambda x:x[0]==i[0] and x[1]>i[1], coor)
print len([i for i in coor if right(i) and left(i) and lower(i) and upper(i)])
