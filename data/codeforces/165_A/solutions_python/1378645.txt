n = int(raw_input())
data = []
for i in range(n):
    data.append(map(int,raw_input().split()))
ans=0
for a,b in data:
    up,down,left,right=[0,0,0,0]
    for c,d in data:
        if a==c and b<d: up=1
        elif a==c and b>d: down=1
        elif a>c and b==d: left=1
        elif a<c and b==d: right=1
    if up==1 and down==1 and left==1 and right==1:ans=ans+1
print ans;
