from itertools import *

n = input()
a = [map(int, raw_input().split()) for i in xrange(n)]
b = [0] * n
for i, j in product(range(n), repeat = 2):
    if a[i] == a[j]: continue
    if a[i][0] == a[j][0]:
        b[i] |= 1 << (a[i][1] < a[j][1])
    if a[i][1] == a[j][1]:
        b[i] |= 1 << ((a[i][0] < a[j][0]) + 2)
print sum(x == 15 for x in b)
