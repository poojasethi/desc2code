coor = [map(int, raw_input().split()) for i in [0]*input()]
f = lambda cond:lambda i:filter(cond, coor)
right = f(lambda x:x[0]>i[0] and x[1]==i[1])
left = f(lambda x:x[0]<i[0] and x[1]==i[1])
lower = f(lambda x:x[0]==i[0] and x[1]<i[1])
upper = f(lambda x:x[0]==i[0] and x[1]>i[1])
print len([i for i in coor if right(i) and left(i) and lower(i) and upper(i)])
