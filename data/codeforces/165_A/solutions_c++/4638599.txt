#include<iostream>
using namespace std;
int main()
{
 int n,i,j,r,l,u,d,x[200],y[200],c=0;
 cin>>n;
 for(i=0;i<n;i++)
  cin>>x[i]>>y[i];
 for(i=0;i<n;i++)
 {r=0;l=0;u=0;d=0;
  for(j=0;j<n;j++)
   {
    if(x[i]<x[j]&&y[i]==y[j])
     r++;
    if(x[i]>x[j]&&y[i]==y[j])
    l++;
    if(x[i]==x[j]&&y[i]<y[j])
    u++;
    if(x[i]==x[j]&&y[i]>y[j])
    d++;
   }
   if(r>0&&l>0&&u>0&&d>0)
    c++;
 }
 cout<<c;
 return(0);
}
