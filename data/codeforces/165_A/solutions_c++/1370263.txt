#include<stdio.h>
int main(){
    int n,count=0,i,j,x[200],y[200],a,b,c,d;
    scanf("%d",&n);
    for(i=0;i<n;i++)
      scanf("%d%d",x+i,y+i);
    for(i=0;i<n;i++){
      a=b=c=d=0;
      for(j=0;j<n;j++){
        if(x[j]>x[i]&&y[j]==y[i]) a=1;
        if(x[j]<x[i]&&y[j]==y[i]) b=1;
        if(x[i]==x[j]&&y[i]>y[j]) c=1;
        if(x[i]==x[j]&&y[i]<y[j]) d=1;
      }
      if(a&&b&&c&&d) count++;
    }
    printf("%d",count);
    return 0;
}
