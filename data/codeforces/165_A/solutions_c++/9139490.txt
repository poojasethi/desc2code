#include <iostream>
using namespace std;
int main(){
	int n,i,j,t=0;
	int a,b,c,d;
	int x[201],y[201];
	cin>>n;
	for(i=0;i<n;++i)
		cin>>x[i]>>y[i];
	for(i=0;i<n;i++){
		a=0;b=0;c=0;d=0;
		for(j=0;j<n;j++){
			if(x[i]==x[j]&&y[i]>y[j])
				a=1;
			else if(x[i]==x[j]&&y[i]<y[j])
				b=1;
			else if(x[i]>x[j]&&y[i]==y[j])
				c=1;
			else if(x[i]<x[j]&&y[i]==y[j])
				d=1;}
			if(a+b+c+d>3)
				t++;
		}
	cout<<t;
	return 0;
}
