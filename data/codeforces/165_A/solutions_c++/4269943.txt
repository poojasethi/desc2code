#include<stdio.h>
int main () {
	int c=0,n,i,j,x[205],y[205],x1,x2,x3,x4;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d %d",&x[i],&y[i]);
	for(i=1;i<=n;i++)
	{
		x1=x2=x3=x4=0;
		for(j=1;j<=n;j++)
		{
			if(i==j)
				continue;
			if(x[i]==x[j] && y[i]<y[j])
				x1=1;
			if(x[i]==x[j] && y[i]>y[j])
				x2=1;
			if(x[i]<x[j] && y[i]==y[j])
				x3=1;
			if(x[i]>x[j] && y[i]==y[j])
				x4=1;
		}
		if(x1 && x2 && x3 && x4)
			c++;
	}
	printf("%d\n",c);
	return 0;
}
