#include <stdio.h>
main()
{
int x[250],y[250],n,k=0;
scanf("%d",&n);
for (int i=0; i<n; i++)
scanf("%d %d",&x[i],&y[i]);
for (int i=0; i<n; i++) {
  int d=0,s=0,u=0,b=0;
  for (int j=0; j<n; j++) {
    if (x[j]>x[i] && y[j]==y[i]) d=1;
    if (x[j]<x[i] && y[j]==y[i]) s=1;
    if (x[j]==x[i] && y[j]>y[i]) u=1;
    if (x[j]==x[i] && y[j]<y[i]) b=1;
    }
  if (d+s+u+b==4) k++;
  }
printf("%d",k);
return 0;
}
