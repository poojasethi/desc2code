#include<iostream>

using namespace std;
int x[200],y[200],u,o,r,l,n,res;
int main()
{
	cin>>n;
	for(int i=0;i<n;i++) cin>>x[i]>>y[i];
	for(int i=0;i<n;i++)
	{
		u=0;o=0;r=0;l=0;
		for(int j=0;j<n;j++)
		{
			if(x[i]<x[j]&&y[i]==y[j]) u=1;
			if(x[i]>x[j]&&y[i]==y[j]) o=1;
			if(x[i]==x[j]&&y[i]>y[j]) l=1;
			if(x[i]==x[j]&&y[i]<y[j]) r=1;
		}
		if(u&&o&&l&&r) res++;
	}
	cout<<res<<endl;
	return 0;
}
