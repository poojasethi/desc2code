
#include<cstdio>


long x[200],y[200],u,i,o,r,l,n,res;
int main()
{
	while(scanf("%ld",&n)==1)
	{
		for(i=0;i<n;i++)
			scanf("%ld%ld",&x[i],&y[i]);;
		for(i=0;i<n;i++)
		{
			u=0;o=0;r=0;l=0;
			for(int j=0;j<n;j++)
			{
			if(x[i]<x[j]&&y[i]==y[j]) u=1;
			if(x[i]>x[j]&&y[i]==y[j]) o=1;
			if(x[i]==x[j]&&y[i]>y[j]) l=1;
			if(x[i]==x[j]&&y[i]<y[j]) r=1;
			}
			if(u&&o&&l&&r) res++;
	}
	printf("%ld\n",res);
	}
	return 0;
}
