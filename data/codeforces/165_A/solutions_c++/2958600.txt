#include<iostream>
using namespace std;
int main()
{
	int n,a[200],b[200],i,j,c,d=0,e,f,g;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i]>>b[i];
	for(i=0;i<n;i++)
	{
		c=e=f=g=0;
		for(j=0;j<n;j++)
		{
			if((a[j]==a[i])&&(b[j]>b[i])){c=1;continue;}
			if((a[j]==a[i])&&(b[j]<b[i])){e=1;continue;}
			if((a[j]>a[i])&&(b[j]==b[i])){f=1;continue;}
			if((a[j]<a[i])&&(b[j]==b[i])){g=1;continue;}
		}
		if(c==1&&e==1&&f==1&&g==1)d+=1;
	}
	cout<<d<<endl;
	return 0;
}
