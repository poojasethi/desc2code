#include <cstdio>
int x[201],y[201],n;
int p[201][4];
int ans;
int main(){
scanf("%d",&n);
for(int i = 0; i < n;i++)
scanf("%d%d",&x[i],&y[i]);
for(int i=0;i<n;i++)
for(int j = 0; j<n;j++){
if(x[i]==x[j]&&y[i]<y[j])p[i][0]=1;
if(x[i]==x[j]&&y[i]>y[j])p[i][1]=1;
if(x[i]<x[j]&&y[i]==y[j])p[i][2]=1;
if(x[i]>x[j]&&y[i]==y[j])p[i][3]=1;
}
for(int i = 0;i<n;++i)
ans+=(p[i][0]+p[i][1]+p[i][2]+p[i][3])/4;
printf("%d\n",ans);
}
