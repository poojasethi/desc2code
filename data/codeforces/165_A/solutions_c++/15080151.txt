#include<iostream>
using namespace std;
int main()
{
	std::ios::sync_with_stdio(false);
	int j,n,a[205][2],i,q,w,e,r,ans;
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i][0]>>a[i][1];
	for(ans=i=0;i<n;i++)
	{
		for(q=w=e=r=j=0;j<n;j++)
		{
			a[i][0]>a[j][0]&&a[i][1]==a[j][1]?q++:q;
			a[i][0]<a[j][0]&&a[i][1]==a[j][1]?w++:w;
			a[i][1]>a[j][1]&&a[i][0]==a[j][0]?e++:e;
			a[i][1]<a[j][1]&&a[i][0]==a[j][0]?r++:r;
		}
		q&&w&&e&&r?ans++:ans;
	}
	cout<<ans<<'\n';
}
