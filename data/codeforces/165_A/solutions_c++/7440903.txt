#include<iostream>
using namespace std ;
int n , x[201] , y[201] , k , ans=0 ;
int main()
{
	cin >> n ;
	for(int i=0 ; i<n ; i++)
		cin >> x[i] >> y[i] ;
	for(int i=0 ; i<n ; i++)
	{
		k=0 ;
		for(int j=0 ; j<n ;j++)
		{
			if(x[i]==x[j] && y[i]<y[j])
				k|=1;
			if(x[i]==x[j] && y[j]<y[i])
				k|=2;
			if(x[i]<x[j] && y[i]==y[j])
				k|=4;
			if(x[j]<x[i] && y[i]==y[j])
				k|=8 ;
		}
		if(k==15)
			ans++ ;
	}
	cout << ans << endl ;
	return 0 ;
}
