#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,x[1000],y[1000],count=0;;
	cin>>n;
	for(int i=0;i<n;++i)
	cin>>x[i]>>y[i];
	for(int i=0;i<n;++i)
	{
		int b=15;
		for(int j=0;j<n&&b!=0;++j)
		{
			if(x[i]==x[j])
			{
				if(y[i]<y[j])
				b&=15-1;
				if(y[i]>y[j])
				b&=15-2;
			}
			if(y[i]==y[j])
			{
				if(x[i]<x[j])
				b&=15-4;
				if(x[i]>x[j])
				b&=15-8;
			}
		}
		if(b==0)
		count+=1;
	}
	cout<<count;
	return 0;
}
