#include<cstdio>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int main(){
	int n; scanf("%d",&n);
	int x[200],y[200];
	rep(i,n) scanf("%d%d",x+i,y+i);

	int ans=0;
	rep(i,n){
		bool b[4]={};
		rep(j,n){
			if(x[i]==x[j] && y[i]<y[j]) b[0]=true;
			if(x[i]==x[j] && y[i]>y[j]) b[1]=true;
			if(x[i]<x[j] && y[i]==y[j]) b[2]=true;
			if(x[i]>x[j] && y[i]==y[j]) b[3]=true;
		}
		if(b[0] && b[1] && b[2] && b[3]) ans++;
	}
	printf("%d\n",ans);

	return 0;
}
