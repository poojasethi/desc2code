#include<stdio.h>
int x[1001],y[1001];
int main()
{
	int n,i,j,ans=0;
	scanf("%d",&n);
	for(i=0;i<n;++i)
		scanf("%d%d",&x[i],&y[i]);
	for(i=0;i<n;++i)
	{
		int cnt=0;
		for(j=0;j<n;++j)
			if(j!=i)
			{
				if(x[j]==x[i]&&y[j]<y[i])
					cnt|=1;
				if(x[j]==x[i]&&y[j]>y[i])
					cnt|=2;
				if(y[j]==y[i]&&x[j]<x[i])
					cnt|=4;
				if(y[j]==y[i]&&x[j]>x[i])
					cnt|=8;
			}
		if(cnt==15)
			ans++;
	}
	printf("%d\n",ans);
	return 0;
}
