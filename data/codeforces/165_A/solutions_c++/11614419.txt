#include <iostream>
using namespace std;
int k,n,x[212],y[212],u,d,l,r,sum;
int main(){
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>x[i]>>y[i];
	for(int i=0;i<n;i++){
		u=d=l=r=0;
		for(int j=0;j<n;j++){
			if(x[i]==x[j] && y[i]>y[j])
				d++;
			if(x[i]==x[j] && y[i]<y[j])
				u++;
			if(x[i]>x[j] && y[i]==y[j])
				l++;
			if(x[i]<x[j] && y[i]==y[j])
				r++;
		}
		if(l && r && u && d)
			sum++;
	}
	cout<<sum<<endl;
}
