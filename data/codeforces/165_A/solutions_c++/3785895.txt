#include<iostream>
using namespace std;
#define rep(i, n) for(int i = 0; i < (int)n; ++i)
int x[210], y[210];
int main() {
int n;
cin >> n;
rep (i, n) {
cin >> x[i] >> y[i];
}
int res = 0;
rep (i, n) {
int bit = 0;
rep (j, n) {
if (x[i] < x[j] && y[i] == y[j]) bit |= 1;
if (x[i] > x[j] && y[i] == y[j]) bit |= 2;
if (x[i] == x[j] && y[i] < y[j]) bit |= 4;
if (x[i] == x[j] && y[i] > y[j]) bit |= 8;
}
if (bit == 15) ++res;
}
cout << res << endl;
return 0;
}
