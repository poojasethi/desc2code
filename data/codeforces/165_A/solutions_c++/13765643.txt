#include <cstdio>
#include <algorithm>
using namespace std;
int a,b,c,d,e,f,g,h,i,j,k,l,m,n;
int x[205],y[205];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d%d",&x[i],&y[i]);
	for(i=1;i<=n;i++){
		g=0;
		for(j=1;j<=n;j++){
			if(x[i]==x[j]&&y[i]<y[j])g|=1;
			if(x[i]==x[j]&&y[i]>y[j])g|=2;
			if(x[i]<x[j]&&y[i]==y[j])g|=4;
			if(x[i]>x[j]&&y[i]==y[j])g|=8;
		}
		if(g==15)h++;
	}
	printf("%d\n",h);
	return 0;
}
