#include<stdio.h>
int main()
{
	int x[300], y[300];
	int n, cou = 0;
	int i, j;
	scanf("%d",&n);
	for(i=1;i<=n;i++) scanf("%d %d",&x[i],&y[i]);
	for(i=1;i<=n;i++)
	{
		int a = 0,b  = 0, c = 0, d  = 0;
		for(j=1;j<=n;j++)
		{
			if(i==j) continue;
			if(x[i]==x[j]&&y[i]<y[j]) a++;
			if(x[i]==x[j]&&y[i]>y[j]) b++;
			if(x[i]<x[j]&&y[i]==y[j]) c++;
		    if(x[i]>x[j]&&y[i]==y[j]) d++;
		}
		if(a&&b&&c&&d) cou++;
	}
	printf("%d\n",cou);
	return 0;
}
