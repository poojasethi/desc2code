#include <iostream>
using namespace std;

int n,x[201],y[201],i,j,s=0,r,l,u,d;

int main()
{cin>>n;
for(i=1;i<=n;i++)
cin>>x[i]>>y[i];

for(i=1;i<=n;i++)
{r=0;
l=0;
u=0;
d=0;

for(j=1;j<=n;j++)
{
    if(y[i]==y[j]&&x[j]<x[i])
    l=1;

    if(y[i]==y[j]&&x[j]>x[i])
    r=1;

    if(x[i]==x[j]&&y[j]<y[i])
    d=1;

    if(x[i]==x[j]&&y[j]>y[i])
    u=1;
}

if(r==1&&l==1&&u==1&&d==1)
s++;

}

cout<<s;

    return 0;
}
