#include<stdio.h>
main(){
	int x[201],y[201],n,i,j,a,b,c,d,ans;
	while(scanf("%d",&n)==1){
		for(i=0;i<n;i++) scanf("%d %d",&x[i],&y[i]);
		ans=0;
		for(i=0;i<n;i++){
			a=b=c=d=0;
			for(j=0;j<n;j++){
				if(j==i) continue;
				if(x[j]==x[i]&&y[j]>y[i]) a=1;
				if(x[j]==x[i]&&y[j]<y[i]) b=1;
				if(x[j]>x[i]&&y[j]==y[i]) c=1;
				if(x[j]<x[i]&&y[j]==y[i]) d=1;
			}
			if(a+b+c+d==4) ans++;
		}
		printf("%d\n",ans);
	}
}
