#include <cstdio>
int x[201],y[201];
int main(){
    int n,u,d,l,r,ans=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d%d",&x[i],&y[i]);
    for(int i=0;i<n;i++){
        for(int j=u=d=l=r=0;j<n;j++){
            if(x[j]==x[i]&&y[j]>y[i]) u=1;
            if(x[j]==x[i]&&y[j]<y[i]) d=1;
            if(y[j]==y[i]&&x[j]<x[i]) l=1;
            if(y[j]==y[i]&&x[j]>x[i]) r=1;
        }
        if(u&d&l&r) ans++;
    }
    printf("%d",ans);
    return 0;
}
