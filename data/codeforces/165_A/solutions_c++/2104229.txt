#include<stdio.h>
int main()
{
int n,a[250],b[250],i,j,l1,l2,l3,l4,count=0;
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%d%d",&a[i],&b[i]);
for(i=0;i<n;i++)
{
l1=l2=l3=l4=0;
for(j=0;j<n;j++)
{
if(a[i]==a[j])
{
if(b[i]>b[j])
l1=1;
else if(b[i]<b[j])
l2=1;
}
if(b[i]==b[j])
{
if(a[i]>a[j])
l3=1;
else if(a[i]<a[j])
l4=1;
}
}
if(l1==1&&l2==1&&l3==1&&l4==1)
count=count+1;
}
printf("%d",count);
return 0;
}
