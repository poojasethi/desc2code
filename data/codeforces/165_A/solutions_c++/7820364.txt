#include<bits/stdc++.h>

using namespace std;

int n,a,b,c,d,i,k,cnt,x[210],y[210];
int main()
{
    //freopen("in.txt","r",stdin);
	scanf("%d",&n);
	for(i=0;i<n;i++)
        scanf("%d %d",&x[i],&y[i]);
	for(i=0;i<n;i++)
    {
		for(k=a=b=c=d=0;k<n;k++)
		{
			a=a|(x[i]==x[k]&&y[i]<y[k]);
			b=b|(x[i]==x[k]&&y[i]>y[k]);
			c=c|(y[i]==y[k]&&x[i]<x[k]);
			d=d|(y[i]==y[k]&&x[i]>x[k]);
		}
		if(a&b&c&d)++cnt;
	}
	printf("%d\n",cnt);
	return 0;
}
