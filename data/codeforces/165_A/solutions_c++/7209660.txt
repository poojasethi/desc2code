#include <cstdio>

int n,i,j,a,b,c,d,ans;
int x[400],y[400];

int main ()
{
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d %d",&x[i],&y[i]);
	for(i=1;i<=n;i++)
	{
		a=b=c=d=0;
		for(j=1;j<=n;j++)
		if(j!=i)
		{
			if(y[i]==y[j])
				if(x[i]>x[j])a++;
				else b++;
			if(x[i]==x[j])
				if(y[i]>y[j])c++;
				else d++;
		}
		if(a && b && c && d)ans++;
	}
	printf("%d\n",ans);

	return 0;
}
