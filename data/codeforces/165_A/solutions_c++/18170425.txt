#include <stdio.h>
main()
{int i,t=0,j,n,d,f,g,c=0,a[200],b[200];
scanf("%d",&n);
for(i=0;i<n;i++)
scanf("%d %d",&a[i],&b[i]);
for(i=0;i<n;i++)
    {c=0;d=0;f=0;g=0;for(j=0;j<n;j++)
{if(a[i]>a[j] && b[i]==b[j] )
 c=1;
else if(a[i]<a[j] && b[i]==b[j])
 d=1;
 else if(a[i]==a[j] && b[i]<b[j])
 f=1;
 else if(a[i]==a[j] && b[i]>b[j])
 g=1;}
 if(c+f+g+d>3)
 t++;}
 printf("%d",t);
return 0;}
