#include <iostream>
using namespace std;

int main(){
	int n,k=0,a,b,c,d;
	cin>>n;
	int s[n],h[n];
	for(int i = 0;i<n;i++){
		cin>>s[i];
		cin>>h[i];
	}
	for(int i = 0;i<n;i++){
		a=0,b=0,c=0,d=0;
		for(int j = 0;j<n;j++){
			if(s[i]==s[j] && h[i]>h[j])a=1;
			if(s[i]==s[j] && h[i]<h[j])b=1;
			if(s[i]<s[j] && h[i]==h[j])c=1;
			if(s[i]>s[j] && h[i]==h[j])d=1;
		}
		if(a&b&c&d)k++;
	}
	cout<<k<<endl;
    return 0;
}
