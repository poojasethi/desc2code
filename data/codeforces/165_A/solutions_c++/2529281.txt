#include<iostream>
#include<map>
using namespace std;
int main()
{
int n;
cin>>n;
int x[n],y[n];
for(int i=0;i<n;i++)
cin>>x[i]>>y[i];
int t=0;
for(int i=0;i<n;i++) {
map<char,int> m;
for(int j=0;j<n;j++ ) {
if(j==i)
continue;
if(x[j]>x[i]&& y[i]==y[j])
m['r']++;
else if(x[j]<x[i]&&y[i]==y[j])
m['l']++;
else if(x[i]==x[j] && y[i]>y[j])
m['p']++;
else if(x[i]==x[j]&& y[i]<y[j])
m['o']++;
}
if(m.size()==4)
t++;
}
cout<<t;
}
