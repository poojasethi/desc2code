#include<stdio.h>
int main()
{
	int a[222],b[222],i,j,n,l=0,r,s,t,u;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d %d",&a[i],&b[i]);
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<n;j++)
		{
			if(a[i]==a[j]&&b[i]>b[j]) r=1;
			if(a[i]==a[j]&&b[i]<b[j]) s=1;
			if(a[i]>a[j]&&b[i]==b[j]) t=1;
			if(a[i]<a[j]&&b[i]==b[j]) u=1;
			
		}
		if(1==r&&1==s&&1==t&&1==u) 
			l+=1;
		r=s=t=u=0;
	}
	printf("%d\n",l);
	return 0;
}
