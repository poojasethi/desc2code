#include<iostream>
using namespace std;
int main()
{
int n,count=0,f1,f2,f3,f4,i,j;
cin>>n;
int x[n],y[n];
for(i=0;i<n;i++)
cin>>x[i]>>y[i];
for(i=0;i<n;i++)
 {
 f1=0,f2=0,f3=0,f4=0;
 for(j=0;j<n;j++)
 {
 if(x[j]<x[i]  && y[i]==y[j])
 f1=1;
 if(x[j]>x[i]  && y[i]==y[j])
 f2=1;
 if(x[j]==x[i]  && y[i]>y[j])
 f3=1;
 if(x[j]==x[i]  && y[i]<y[j])
 f4=1;

 }
if(f1 && f2 && f3 && f4)
 count++ ;
 }
 cout<<count;

}
