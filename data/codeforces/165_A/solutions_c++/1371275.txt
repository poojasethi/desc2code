#include <iostream>

using namespace std;

int N,X[210],Y[210],result;

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
		cin >> X[i] >> Y[i];
	for (int i=0;i<N;i++)
	{
		int right=0,left=0,lower=0,upper=0;
		for (int j=0;j<N;j++)
		{
			right=right||(X[j]>X[i]&&Y[j]==Y[i]);
			left=left||(X[j]<X[i]&&Y[j]==Y[i]);
			lower=lower||(X[j]==X[i]&&Y[j]<Y[i]);
			upper=upper||(X[j]==X[i]&&Y[j]>Y[i]);
		}
		result+=right&&left&&lower&&upper;
	}
	cout << result << endl;
	return 0;
}
