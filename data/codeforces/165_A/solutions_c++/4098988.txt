#include <stdio.h>
int main()
{
    int n,x[200],y[200],c=0;
    bool l,r,u,d;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
        scanf("%d%d",&x[i],&y[i]);
    for(int i=0;i<n;++i){
        for(int j=l=r=u=d=0;j<n;++j){
            u|=x[i]==x[j] && y[i]<y[j];
            d|=x[i]==x[j] && y[i]>y[j];
            l|=y[i]==y[j] && x[i]>x[j];
            r|=y[i]==y[j] && x[i]<x[j];
        }
        c+=u&d&r&l;
    }
    printf("%d",c);
    return 0;
}
