import re
import sys

cnt = []

def calc_cells(tags, start):
  r = 0
  while start < len(tags):
    if tags[start] == "table":
      start = calc_cells(tags, start + 1)
    elif tags[start] == "/table":
      cnt.append(r)
      return start
    elif tags[start] == "td":
      r += 1
    start += 1

s = "".join(x.strip() for x in sys.stdin.readlines())
calc_cells([x[1:-1] for x in re.findall("<.*?>", s)], 0)
print " ".join(str(x) for x in sorted(cnt))
