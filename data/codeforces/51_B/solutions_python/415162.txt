#!/usr/bin/env python
# -*- coding: utf-8 -*-

import re, sys

def count(l, shift, ans):
    sum = 0
    while (shift < len(l)):
        if l[shift] == "td":
            sum += 1
        elif l[shift] == "table":
            shift = count(l, shift+1, ans)
        elif l[shift] == "/table":
            ans.append(sum)
            return shift
        shift += 1

def main():
    ans = []
    l = re.findall("<(/?table|td)>", "".join(sys.stdin.readlines()).replace("\n", ""))
    count(l, 1, ans)
    ans.sort()
    for i in ans:
        sys.stdout.write("%d " % i)
    sys.stdout.write("\n");
    return 0

if __name__ == '__main__':
    main()
