from __future__ import print_function
import sys
html = ''.join (line for line in sys.stdin).replace ('\n', '').replace ('<table>', '{').replace ('</table>', '}').replace ('<td>', '[').replace ('</td>', ']').replace ('<tr>', '').replace ('</tr>', '')
stack, ans = list (), list ()
for i in html:
    if i == '{':
        stack.append (0)
    elif i == '}':
        ans.append (stack.pop ())
    elif i == '[':
        stack[-1] += 1
print (*sorted (ans))
