#!/usr/bin/env python

import fileinput

data = ''
for line in fileinput.input():
    data += (line.strip())

tk = data.split('>')

ftsizes = []
tsizes = []

for t in tk:
    if t == '<table':
        tsizes.append(0)
    elif t == '</table':
        ftsizes.append(tsizes.pop())
    elif t == '</td':
        tsizes[-1] += 1

ftsizes.sort()

print ' '.join(str(x) for x in ftsizes)
