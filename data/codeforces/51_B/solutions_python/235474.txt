import sys

freq = []
stk = []
for tok in sys.stdin.read().rstrip().replace('\n','').replace('<tr>','').replace('</tr>','').replace('</td>','').replace('><','> <').split():
#   print >> sys.stderr, tok,
    if tok == '<table>':
        stk.append(0)
    elif tok == '</table>':
        freq.append( stk[-1] )
        stk.pop()
    elif tok == '<td>':
        stk[-1] += 1
    else:
        assert(False)
freq.sort()
print ' '.join( map( str, freq ) )
