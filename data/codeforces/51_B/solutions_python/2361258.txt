import sys
s=''.join(l.strip() for l in sys.stdin).replace('<table>','{')\
   .replace('</table>','}').replace('<tr>','').replace('</tr>','')\
   .replace('<td>','<').replace('</td>','')
v,t=[],[]
for c in s:
  if '{'==c:
    t+=[0]
  elif '}'==c:
    v+=[t.pop()]
  else:
    t[-1]+=1
print ' '.join(map(str,sorted(v)))
