t = ""

while 1:
    try:
        s = raw_input()
    except:
        break

    t += s.strip('\n')


t = t.replace("<", " ").replace(">", " ").split()

ans = []
s = []
cnt = 0
for a in t:
    if a == "table":
        ans.append(0)
        s.append(cnt)
        cnt += 1
    if a == "/table":
        s.pop()
    if a == "td":
        ans[s[-1]] += 1
ans.sort()

print " ".join(map(str, ans))
