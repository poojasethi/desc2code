import re
import sys
s = "".join(map(lambda x: x.strip(), sys.stdin))
tags = re.findall(r"<[^>]*>",s)

stack=[]
res=[]
cur=0
for t in tags:
	if t == '<td>':
		cur += 1
	elif t == '<table>':
		stack.append(cur)
		cur = 0
	elif t == '</table>':
		res.append(cur)
		cur = stack.pop()
print " ".join(map(str,sorted(res)))
