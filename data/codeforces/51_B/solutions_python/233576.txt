#! /usr/bin/env python

from sys import stdin
import re

def tokenize(data):
	data=data.replace("<"," <").replace(">","> ")
	data=data.strip()
	return data.split()

def count_cells(data):
	ret=[]
	stk=[]
	cur=0
	for token in data:
		#print stk
		if token=='<table>':
			stk.append(cur)
			cur=0
		elif token=='<td>':
			cur+=1
		elif token=='</table>':
			ret.append(cur)
			cur=stk.pop()
	ret.sort()
	return ret

if __name__=='__main__':
	data="".join(stdin.readlines())
	data="".join([c for c in data if c in "abcdefghijklmnopqrstuvwxyz<>/"])
	data=tokenize(data)
	#print data
	print " ".join(map(str,count_cells(data)))
