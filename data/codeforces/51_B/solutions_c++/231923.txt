	string s,ss;
	while (cin >> ss) s+=ss;

	int a[6000];
	int o = 0;
	vector<int> ans;

	for (int i = 2; i < s.size(); i++)
	{
		if (s[i] == 'a') a[o++] = 0;
		else
		if (s[i] == 'd') a[o-1]++;
		else
		if (s[i-1] == '/')
		{
			i++;
			if (s[i] == 'a') { o--; ans.push_back(a[o]); }
		}
	}

	sort(ans.begin(),ans.end());

	for (int i = 0; i < ans.size(); i++)
		cout << ans[i] << " ";

//	in.getline(s);

	return 0;	
}
