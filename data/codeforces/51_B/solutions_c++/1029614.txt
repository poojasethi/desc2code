#include <iostream>
#include <cmath>
#include <vector>
#include <string>
#include <stack>
#include <algorithm>
#define rep(i,n) for(int i=0;i<n;i++)
using namespace std;

int n,k,r,v[5000];
string s,t;
stack<int> st,cl;
const string d[]={"<table>","</table>","<tr>","</tr>","<td>","</td>"};

int main(){

  s="";
  while(cin>>t)s+=t;

  n=s.size();

  rep(ii,n){
    if(s.substr(ii,7)==d[0]){
      st.push(k++);
      cl.push(r);
      r=0;
    }
    if(s.substr(ii,8)==d[1]){
      v[st.top()]=r;
      st.pop();
      r=cl.top();
      cl.pop();
    }
    if(s.substr(ii,5)==d[5])r++;
  }
  sort(v,v+k);
  rep(i,k)cout<<v[i]<<" ";cout<<endl;
  
  return 0;

}
