#include<string>
#include<iostream>
#include<algorithm>
using namespace std;
bool T=1;
int k;
int cnt[23333];
void rec()
{
	int ans=0;
	
	char c;
	while(1)
	{
	string s="";
	while(1)
	{
		cin>>c;
		if(c=='>') break;
		s=s+c;
	}
	if(T) 
	{T=0;
		continue;
		
	}
	s.erase(0,1);if(s=="table") rec();
	if(s=="/td") ans++;
	if(s=="/table") {
		cnt[k]=ans;
		k++;
		break ;
		return ;
	}
}
}
int main()
{
	rec();
	sort(cnt,cnt+k);
	for(int i=0;i<k;i++)
	{
		cout<<cnt[i]<<" ";
	}
}
