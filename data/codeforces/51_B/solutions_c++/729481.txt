#include <iostream>
#include <algorithm>
#include <vector>

using namespace std;

string s;

vector<int> r;
int p;
void parsetable();
int parserow();
int parsecell() {
	p+=4;
	if(s[p+2] == 'a') parsetable();
	p+=5;
	return 1;
}
int parserow() {
	p+=4;
	int res = 0;
	while(s[p+2] == 'd') res+=parsecell();
	p+=5;
	return res;
}
void parsetable() {
	p+=7;
	int res = 0;
	while(s[p+2] == 'r') res+=parserow();
	p+=8;
	r.push_back(res);
}

int main () {
	string str;
	while(cin >> str) s = s+str;
	p = 0;
	parsetable();
	sort(r.begin(),r.end());
	cout << r[0];
	for(int i = 1; i < r.size(); i++) cout << " " << r[i];
	cout << endl;
	return 0;
}
