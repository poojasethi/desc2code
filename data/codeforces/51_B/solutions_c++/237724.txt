#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

int ans[5000] = {0}, all = 0;
char c;

int Found(int wh){

    while (cin >> c){
        if (c == 'd'){
            ans[wh]++;
        }
        else if (c == 'a'){
            all++;
            Found(all - 1);
        }
        else if (c == '/'){
            cin >> c >> c;
            if (c == 'a') return 0;
        }
    }
    return 0;
}

int main() {

//    freopen("in", "r", stdin);
    while (cin >> c){
        if (c == 'a'){
            all++;
            Found(all - 1);
        }
    }
    sort(ans, ans + all);
    for (int i = 0; i < all; i++)
        printf("%d%c", ans[i], i == all - 1 ? 10 : 32);
    return 0;
}
