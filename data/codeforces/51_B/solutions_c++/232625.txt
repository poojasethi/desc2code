#include <cstdio>
#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

bool qs (int a,int b) {return (a<b);}
int main()
{
	vector <int> stack,tables;
	int is;
	string s="",c="";
	while (!cin.eof())
	{
		getline(cin,c);
		s=s+c;
	}
	int i=0;
	while (i<s.length())
	{
		int l=s.find('<',i);
		int r=s.find('>',l);
		i=r+1;
		string teg=s.substr(l,r-l+1);
		if (teg=="<table>")
		{
			tables.push_back(0);
			stack.push_back(tables.size()-1);
			continue;
		}
		if (teg=="</table>")
		{
			stack.pop_back();
			continue;
		}
		if (teg=="<td>")
		{
			tables[stack[stack.size()-1]]++;
			continue;
		}
	}
	sort(tables.begin(),tables.end(),qs);
	for (int i=0;i<tables.size();i++)
		cout << tables[i] << " " ;
	return 0;
}
