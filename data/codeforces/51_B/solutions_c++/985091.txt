#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int read() {
	int c;
	while((c=getchar()) != EOF) {
		if(c=='<') break;
	}
	c=getchar();
	if(c == '/') return 0;
	return 1;
}

vector<int> ans;
void TABLE();

int TD() {
	while(read()) { TABLE(); }
	return 1;
}

int TR() {
	int cell=0;
	while(read()) { cell += TD(); }
	return cell;
}

void TABLE() {
	int cell=0;
	while(read()) { cell += TR(); }
	ans.push_back(cell);
}

int main(void) {
	read();
	TABLE();
	sort(ans.begin(), ans.end());
	for(int i=0;i<(int)ans.size();i++)
		printf("%d ", ans[i]);
	puts("");
	return 0;
}
