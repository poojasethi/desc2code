#include <cstdio>
#include <iostream>
#include <algorithm>
#include <string>
#include <stack>

using namespace std;

string s, dum;

int idx = 0;
int A[5000];

stack <int> S;

int main() {
	while (cin >> dum) {
		s += dum;
	}
	//cout << s << endl;
	for (int i = 0; i < s.length(); i++) {
		// s[i] == '<'
		dum = "";
		while (s[i] != '>') {
			dum += s[i];
			i++;
		}
		dum += ">";
		// cout << dum << endl;
		
		if (dum == "<table>") {
			S.push(idx);
			idx++;
		} else if (dum == "</table>") {
			S.pop();
		} else if (dum == "<td>") {
			A[S.top()]++;
		}
	}
	
	sort(A, A + idx);
	
	for (int i = 0; i < idx; i++) {
		if (i > 0) printf(" ");
		printf("%d", A[i]);
	}
	printf("\n");
	
	// system("pause");
	
	return 0;
}
