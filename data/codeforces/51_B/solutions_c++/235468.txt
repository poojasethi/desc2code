#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

char a[55555];
int s[11111],st;
int b[11111],bt;
int
main()
{
    int x=0;
    while(1){
        if(scanf("%s",a+x)==EOF)break;
        x=strlen(a);
    }
    for(int i=0; a[i]; i++)
        if(0==strncmp("<ta",a+i,3)){
            s[++st]=0;
        }else if(0==strncmp("<td",a+i,3)){
            s[st]++;
        }else if(0==strncmp("</ta",a+i,4)){
            b[bt++]=s[st--];
        }
    sort(b,b+bt);
    for(int i=0; i<bt; i++)
        printf("%d ",b[i]);
    return 0;
}
