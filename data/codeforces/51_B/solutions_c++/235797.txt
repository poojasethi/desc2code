#include <iostream>
#include <algorithm>
#include <stack>
#include <vector>

using namespace std;

string teg;
char c = '.';
stack <int> table;
vector <int> ans;

int main()
{
    while ( cin>>c )
    {
        if (c == '>' || c == '<')
        {
            if ( teg == "table" )
                table.push(0);
            if ( teg == "/table" )
            {
                ans.push_back(table.top());
                table.pop();
            }
            if ( teg == "td" )
            {
                int i = table.top()+1;
                table.pop();
                table.push(i);
            }
            teg.clear();
        }
        else
            if (c!='\n')
                teg+=c;
    }
    sort(ans.begin(), ans.end());
    for ( int i = 0; i < ans.size(); ++i )
        cout << ans[i] << " ";
    return 0;
}
