using namespace std;

#define f(i, a, b) for(int i = a; i < b; i++)
#define rep(i, n)  f(i, 0, n)

int cur, num;
int size[5000], res[5000];

int main(){

    string s, t;
    while(cin >> t) s += t;

    rep(i, s.length()) if(s[i] == '<' || s[i] == '>') s[i] = ' ';

    num = 0;
    cur = -1;

    stringstream ss(s);
    while(ss >> s) {
        if(s == "table") {
            cur++;
            size[cur] = 0;
        }
        else if(s == "td")
            size[cur]++;
        else if(s == "/table") {
            res[num++] = size[cur];
            cur--;
        }
    }

    sort(res, res + num);
    rep(i, num) cout << res[i] << ' ';
    cout << endl;

}
