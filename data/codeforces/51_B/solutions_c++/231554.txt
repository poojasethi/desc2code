using namespace std;

typedef vector<int> VI;
typedef vector<VI> VVI;

int main () {
	string text,s;
	VI res,st;
	while (cin>>s)
	{
		text+=s;
	}
	int i=0;
	while (i<text.size())
	{
		s="";
		while (i<text.size() && text[i]!='<') i++;
		if (i==text.size()) break;
		i++;
		while (text[i]!='>') {
			s+=text[i];
			i++;
		}
		if (s=="table") st.push_back(0);
		if (s=="/table") {
			res.push_back(st.back());
			st.pop_back();
		}
		if (s=="td") st.back()++;
	}
	sort(res.begin(),res.end());
	for (int i=0;i<res.size();i++) cout << res[i] << ' ';
	return 0;
}
