#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <string>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 5001

int i,n,t,r,a;
string s,b;
char str[lim];
int R[lim],C[lim],T[lim],S[lim],A[lim];

int main()
{
  while(gets(str)) 
    s+=str;	
	for(i=0,n=s.size();i<n;i++)
	{
	  b=s.substr(i,4);
	  if(b=="<tr>")
		  R[r++]=i;
		else if(b=="</tr")
		  r--;
		else if(b=="<td>")
		  C[R[r-1]]++;
	}
	for(i=0;i<n;i++)
	{
	  b=s.substr(i,4);
	  if(b=="<tab")
		{
		  S[t]=0;
		  T[t++]=i;
		}
		else if(b=="</ta")
		{
		  t--;
		  A[a++]=S[t];
		}
		else if(b=="<tr>")
		  S[t-1]+=C[i];
	}
	sort(A,A+a);
	for(i=0;i<a;i++)
	  printf("%d ",A[i]);
  return 0;
}
