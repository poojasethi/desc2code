#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <string>

using namespace std;

char txt[5010],buf[5010];
int tb[5010];
int stk[5010];

int main()
{
	int i,nt=0;
	int p=-1;
	int l;
	txt[0]='\0';
	memset(tb,0,sizeof(tb));
	while(scanf("%[^ \n]%*c",buf)>0)	strcat(txt,buf);
	l=0;
	while(sscanf(txt+l,"%[^>]%*c",buf)>0)
	{
		l+=strlen(buf)+1;
		if(strcmp(buf,"<table")==0)
		{
			p++;
			stk[p]=nt;
			nt++;
		}
		if(strcmp(buf,"<td")==0)
		{
			tb[stk[p]]++;
		}
		if(strcmp(buf,"</table")==0)	p--;
	}
	sort(tb,tb+nt);
	for(i=0;i<nt;i++)
	{
		if(!i)	printf("%d",tb[i]);
		else	printf(" %d",tb[i]);
	}
	printf("\n");
	return 0;
}
