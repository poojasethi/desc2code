#include<string>
#include<stack>
#include<vector>
#include<algorithm>
#include<set>
#include<iostream>
#include<cmath>


using namespace std;

int main() {

	string t;
	string s;
	while(cin >> s) t+=s;
	
	vector<int> mem(600, 0);
	
	stack<int> padre;
	int ind = 0;
	int tablas = 0;
	
	while(t.size() > 0) {
		string tag = "";
		
		while(t[0]!= '>') {
			tag += t[0];
			t.erase(t.begin());
		}
		tag += t[0];
		t.erase(t.begin());

		if(tag == "<table>") {
			tablas++;
			padre.push(ind);
			ind = tablas-1;
		} else if(tag == "<td>") {
			mem[ind] ++;
		} else if(tag == "</table>") {
			ind = padre.top();
			padre.pop();
		}
	}

	sort(mem.begin(), mem.begin()+tablas);
	
	for(int i=0 ; i<tablas ; i++) {
		if(i) cout << " ";
		cout << mem[i];
	}
	cout << endl;
	
	return 0;
}
