#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main()
{
	char c[100100];
	char a[100100];
	char d[10010][10];
	int i,j,k,l,kos=0,num,zk,sta;
	int stak[100100],table[100010];
	memset(table,0,sizeof(table));
     memset(stak,0,sizeof(stak));
	while(~scanf("%s",a))
     { 
       strcat(c,a);
     }
     sta=-1;
	l=strlen(c);
	j=zk=sta=num=0;
	for(i=0;i<l;i++)
	{
	    if(c[i]=='<')
	    {
	     k=0;
	     for(;c[i]!='>';i++)
	     d[j][k++]=c[i];
		d[j][k]='>';
		d[j][k+1]='\0';
		j++;
	    }
	}
	for(i=0;i<j;i++)
	{
	 if(strcmp("</table>",d[i])==0){zk=stak[--sta];}
	 if(strcmp("<table>",d[i])==0){stak[++sta]=num;zk=num;++num;}
	 if(strcmp("<td>",d[i])==0){++table[zk];}
	}
	sort(table,table+num);
	for(i=0;i<num-1;i++)
	printf("%d ",table[i]);
	printf("%d\n",table[num-1]);
	
}
