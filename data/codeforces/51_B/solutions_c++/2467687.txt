#include <iostream>
#include <cstring>
#include <stack>
#include <algorithm>
using namespace std;

stack<int> stiva;
char x[5001], xfin[60001];
int i, tabele, sz[1001];
bool finished;
int main()
{
	while(cin.getline(x, 5001))
	{
		strcpy(&xfin[strlen(xfin)], x);
	}
	int p=strlen(xfin);
	for(i=0; i<p; i++)
	{
		if(xfin[i]=='t' && xfin[i+1]=='a' && xfin[i-1]=='<')
		{
			tabele++;
			stiva.push(tabele);
		}
		if(xfin[i]=='t' && xfin[i+1]=='d' && xfin[i-1]=='<')
		{
			sz[stiva.top()]++;
		}
		if(xfin[i]=='t' && xfin[i+1]=='a' && xfin[i-1]=='/')
		{
			stiva.pop();
		}
	}
	sort(sz+1, sz+tabele+1);
	for(i=1; i<=tabele; i++)
	{
		cout<<sz[i]<<" ";
	}
}
