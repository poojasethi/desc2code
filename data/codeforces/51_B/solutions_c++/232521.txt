#include<stdio.h>
#include<stdlib.h>
#include<string.h>

int cmp(const void *x, const void * y){
	return *(int*)x-*(int*)y;
}

int main(){
	int i,j,k,l,m;
	char s[5003],s1[5003],c;
	int a[5000]={0},b[5000]={0};
	s[0]='\0';
	while(scanf("%[^\n]",s1)!=EOF){
		scanf("\n");
		strcat(s,s1);
	}
	l=strlen(s);
	j=0;
	k=-1;m=0;
	for(i=1;i<l;i++){
		if(s[i]=='>'){
			s1[j]='\0';
			j=0;
			if(strcmp(s1,"table")==0){
				k++;
				a[k]=0;
			}
			else if(strcmp(s1,"/table")==0){
				b[m++]=a[k];
				k--;
			}
			else if(strcmp(s1,"td")==0){
				a[k]++;
			}
		}
		else if(s[i]!='<')
			s1[j++]=s[i];
	}
	qsort(b,m,sizeof(int),cmp);
	for(i=0;i<m;i++)
		printf("%d ",b[i]);
	printf("\n");
	return 0;
}
