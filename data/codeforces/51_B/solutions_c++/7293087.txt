#include <iostream>
#include <cstring>
#include <cstdio>
#include <stack>
#include <queue>

using namespace std;

char s[6000];
stack<int> stk;
priority_queue<int, vector<int>, greater<int> > pri;

int main()
{
    string p = "";

    while( scanf("%s", s) != EOF )
    {
        for(int si = 0, sn = strlen(s); si < sn; si++)
        {
            p += s[si];

            if( s[si] == '>' )
            {
                if( p == "<table>" ) stk.push(0);
                if( p == "</table>" ) pri.push(stk.top()), stk.pop();
                if( p == "</td>" ) stk.top()++;
                p = "";
            }
        }
    }

    bool sp = false;

    while( !pri.empty() )
    {
        if( sp ) printf(" ");
        sp = true;

        printf("%d", pri.top());
        pri.pop();
    }

    puts("");
}
