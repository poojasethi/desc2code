#include <algorithm>
#include <sstream>
#include <vector>
#include <iostream>
#include <stack>
#include <string>

using namespace std;
int main(){
	vector<int> vec;
	int n;
	stack<int> s;
	string str="",tmp,res;
	stringstream ss;
	while(cin>>tmp)	str+=tmp;
	for(int i=0;i<str.size();i++){
		if(str.substr(i,7)=="<table>"){
			s.push(0);
		}
		if(str.substr(i,8)=="</table>"){
			vec.push_back(s.top());
			s.pop();
		}
		if(str.substr(i,5)=="</td>"){
			int t=s.top();
			s.pop();
			s.push(t+1);
		}
	}
	sort(vec.begin(),vec.end());
	for(int i=0;i<vec.size();i++)	ss<<vec[i]<<" ";
	cout<<ss.str().substr(0,ss.str().size()-1)<<endl;
}
