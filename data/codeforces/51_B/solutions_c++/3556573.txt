#include <stdio.h>
#include <string>
#include <stack>
#include <iostream>
#include <algorithm>
using namespace std;
string s,sum;
int main(){
	int i,j,flag1[10009]={0},l=0;
	while(cin>>s){
		sum.insert(sum.length(),s);
	}
	int count=0;
	stack <int> s;
	for(i=0;sum[i]!='\0';i++){
		if(sum[i]=='<'&&sum[i+1]=='t'&&sum[i+2]=='a'&&sum[i+3]=='b'){
			s.push(0);
			
		}
		else if(sum[i]=='<'&&sum[i+1]=='t'&&sum[i+2]=='d' &&sum[i+3] =='>'){
			s.top()++;	
		
		}
		else if(sum[i]=='<'  &&sum[i+1]=='/'
			&&sum[i+2]=='t'&&sum[i+3]=='a'){
			
			flag1[count++]=s.top();
			if(!s.empty()) s.pop();
		}
	}
	sort(flag1,flag1+count);
	for(i=0;i<count-1;i++)
		printf("%d ",flag1[i]);
	printf("%d\n",flag1[i]);
	return 0;
}
