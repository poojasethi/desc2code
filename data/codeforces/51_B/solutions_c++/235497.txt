#include <algorithm>
#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

int ans[5000], st[5000], tos;
char s[5010];

int main()
{
	int i = 0, k = 0, sz = 0, n = 0;
	
	while (cin >> s[n]) n++; 

//	printf("%s\n", s);

	while (i < n) {
		if (s[i] == '<'  && s[i + 2] == 'a') st[tos++] = sz, i += 7, sz++;
		else 
		if (s[i] == '<' && s[i + 1] == '/' && s[i + 3] == 'a') tos--, i += 8;
		else 
		if (s[i] == '<' && s[i + 2] == 'd') ans[st[tos - 1]]++, i += 4;
		else 
		if (s[i] == '<' && s[i + 2] == 'r') i += 4;
		else i += 5;
	}

	sort(ans, ans + sz);

	for (int i = 0; i < sz; ++i)
		printf("%d ", ans[i]);

	return 0;
}
