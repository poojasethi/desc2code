#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<stack>
#include<string>
#include<algorithm>
using namespace std;

int z,tot[5000];
string s,st;
stack<int> tag;

int main()
{
	st = "";
	while (getline(cin,s))
	{
		st += s;
	}
	z = 0;
	memset(tot,0,sizeof(tot));
	for (int i = 0; i < st.size()-2; i++)
		if ((st[i] == '<') && (st[i+1] == 't') && (st[i+2] == 'a'))
		{
			tag.push(z);
			z++;
		}
		else if ((st[i] == '/') && (st[i+1] == 't') && (st[i+2] == 'a'))
			tag.pop();
		else if ((st[i] == '<') && (st[i+1] == 't') && (st[i+2] == 'd'))
			tot[tag.top()]++;
	sort(tot,tot+z);
	for (int i = 0; i < z; i++)
		if (i < z-1)
			printf("%d ",tot[i]);
		else
			printf("%d\n",tot[i]);
}
