#include<iostream>
#include<vector>
#include<algorithm>
#include<sstream>
#include<math.h>
#include<set>
#include<map>
#include<cstring>
#include<cstdlib>
#include<stdio.h>
#define pb push_back
#define mp make_pair
#define f first
#define s second
#define ll long long
#define all(x) x.begin(),x.end()
using namespace std;
int main()
{
    vector<int> v,p;p.pb(0);
    string in,s="",x="";
    while(cin>>in)s+=in;
    for(int i=0;i<s.size();)
    {
        int j=i;
        while(s[j]!='>')
        if(++j>=s.size())break;
        ++j;    
        x=s.substr(i,j-i);
        if(x=="<table>")v.pb(0),p.pb(v.size()-1);
        else if(x=="</table>")p.pop_back();
        else if(x=="<td>")v[p[p.size()-1]]+=1;
        i=j;
    }
    sort(all(v));
    for(int i=0;i<v.size()-1;i++)cout<<v[i]<<" ";
    cout<<v[v.size()-1];
}
