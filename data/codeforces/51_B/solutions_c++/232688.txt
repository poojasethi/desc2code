#include <stdio.h>

#include <algorithm>

using namespace std;

int nr, sol[50002], pred[50002];
char s[50002];

int main ()
{
#ifndef ONLINE_JUDGE
	freopen ("b.in", "r", stdin);
	freopen ("b.out", "w", stdout);
#endif
	
	int i, ind;
	char ch;

	while (scanf ("%c", &ch) == 1)
	{
		if (ch != '\n')
			s[++ nr] = ch;
	}
	
	for (i = 1; i <= nr; i ++)
	{
		i ++;
		if (s[i] == '/')
		{
			if (s[i + 2] == 'a')
				ind = pred[ind];
		}
		else
		{
			if (s[i + 1] == 'a')
			{
				sol[0] ++;
				pred[sol[0]] = ind;
				ind = sol[0];
			}
			if (s[i + 1] == 'd')
				sol[ind] ++;
		}
		while (s[i] != '>')
			i ++;
	}
	
	sort (sol + 1, sol + sol[0] + 1);
	for (i = 1; i <= sol[0]; i ++)
		printf ("%d ", sol[i]);
	return 0;
}
