#include <cstdio>
#include <cstring>
#include <string>
#include <vector>
#include <sstream>
#include <algorithm>

using namespace std;

string s;
vector <int> res;

int main() {
	s = "";
	char str[5005];
	while (scanf("%s", str) != EOF) {
		s += str;
	}
	vector <string> vs;
	replace(s.begin(), s.end(), '<', ' ');
	replace(s.begin(), s.end(), '>', ' ');
	istringstream ist(s);
	string t;
	while (ist >> t) {
		vs.push_back(t);
	}
	vector <int> v;
	for (size_t i = 0; i < vs.size(); i++) {
		if (vs[i] == "table") {
			v.push_back(0);
		} else if (vs[i] == "/table") {
			res.push_back(v.back());
			v.pop_back();
		} else if (vs[i] == "td") {
			v.back()++;
		}
	}
	sort(res.begin(), res.end());
	for (size_t i = 0; i < res.size(); i++) {
		printf("%d%c", res[i], i + 1 == res.size() ? '\n' : ' ');
	}
	return 0;
}
