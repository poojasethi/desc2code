#include<bits/stdc++.h>
using namespace std;

string Scanner = "";
string buf;
int curr =0;
string getNextToken(){
	string ans;
	for(int i = curr; i<Scanner.size(); i++){
		if(Scanner[i] =='>'){
			ans = Scanner.substr(curr+1, i-curr-1);
			curr = i+1;
			return ans;
		}
	}
}
stack<string> parser;
priority_queue<int, vector<int>, greater<int> > ans;
int main(){
	while(getline(cin,buf)){
		Scanner+=buf;
	}

	while(curr<Scanner.size()){
		parser.push(getNextToken());
		if(parser.top() == "/table"){
			int count = 0;
			while(parser.top()!="table"){
				if(parser.top()=="td") count++;
				parser.pop();
			}
			parser.pop();
			ans.push(count);
		}
	}
	bool first = true;
	while(!ans.empty()){
		if(first) first = false;
		else cout<<" ";
		cout<<ans.top();
		ans.pop();
	}
	cout<<endl;
	return 0;
}
