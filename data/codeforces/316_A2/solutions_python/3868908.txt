#include<iostream>
#include<vector>
#include<algorithm>
#include<cmath>
using namespace std;
int main()
{
	string s;
	cin>>s;
	long long int ans=1,tens=0;
	vector <int> letters(10,0);
	int x=10;
	if((s[0]>='A')&&(s[0]<='J'))
	{
		letters[s[0]-'A']=1;
		ans*=9;
		x=9;
	}
	if(s[0]=='?')
	ans*=9;
	for(int i=1;i<s.size();i++)
	{
		if(s[i]=='?')
		tens++;
		if((s[i]>='A')&&(s[i]<='J'))
		{
			if(letters[s[i]-'A']==0)
			{
				letters[s[i]-'A']=1;
				ans*=x;
				x--;
			}
		}
	}
	cout<<ans;
	for(int i=0;i<tens;i++)
	cout<<"0";
	return 0;
}
