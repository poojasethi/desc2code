#include <cstdio>
#include <cctype>

char s[100001];
int n(0), m(1), o(10);
bool c['K'];

int main() {
  scanf("%s", s);
  if (s[0] == '?') {
    m *= 9;
  } else if (isalpha(s[0]) && !c[s[0]]) {
    c[s[0]] = true;
    m *= 9;
    --o;
  }
  for (int i(1); s[i]; ++i)
    if (s[i] == '?') {
      ++n;
    } else if (isalpha(s[i]) && !c[s[i]]) {
      c[s[i]] = true;
      m *= o;
      --o;
    }
  printf("%d", m);
  while (n--)
    putchar('0');
  putchar('\n');
}
