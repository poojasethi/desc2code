#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

string s;
bool used[1000];

int main() {
	cin >> s;
	int n = s.size();
	for (int i = 0; i < n; i++)
		if (s[i] >= 'A' && s[i] <= 'J')
			used[s[i]] = true;
	long long ans = 1;
	int fuck = 10;
	for (char c = 'A'; c <= 'J'; c++)
		if (used[c])
			ans *= fuck, fuck--;
	if (s[0] >= 'A' && s[0] <= 'J')
		ans = ans / 10 * 9;
	if (s[0] == '?')
		ans *= 9;
	cout << ans;
	for (int i = 1; i < n; i++)
		if (s[i] == '?')
			cout << 0;
	cout << endl;
	return 0;
}
