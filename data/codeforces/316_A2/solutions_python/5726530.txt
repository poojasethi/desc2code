#include <iostream>
#include <set>
using namespace std;

string s;
set<char> letters;
long long int ans = 1, letters_remaining = 10;

int main() {
	cin >> s;
	for(int i = 0; i < s.length(); i++) {
		if('A' <= s[i] && s[i] <= 'J') {
			if(letters.count(s[i]) != 0)
				continue;
			letters.insert(s[i]);
			if(i == 0)
				ans *= 9;
			else
				ans *= letters_remaining;
			letters_remaining--;
		}
	}	

	if(s[0] == '?')
		ans *= 9;

	cout << ans;
	for(int i = 1; i < s.length(); i++)
		if(s[i] == '?')
			cout << "0";
	cout << endl;
}
