#include <stdio.h>
#include <string.h>
char str[100101];
int freq[256];
int main()
{
	scanf("%s",str);
	for(int i=0;str[i];++i)
		++freq[str[i]];
	unsigned long long r=1;
	int tens=freq['?']?freq['?']-1:0;
	if(str[0]=='?')
		r*=9;
	else if(freq['?'])
		++tens;
	int h=10;
	if(str[0]>='A' && str[0]<='J'){
		h=9;
		freq[str[0]]=0;
		r*=9;
	}
	for(int i='A';i<='J';++i){
		if(!freq[i])
			continue;
		if(h==10)
			++tens;
		else
			r*=h;
		--h;
	}
	printf("%llu",r);
	for(int i=0;i<tens;++i)
		putchar('0');
	return 0;
}
