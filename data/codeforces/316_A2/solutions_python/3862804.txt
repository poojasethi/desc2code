#include<cstdio>
#include<vector>
#include<iostream>
#define X 200010
using namespace std;
typedef long long ll;
vector<int>e[X];
int a[X];
bool f[X],ff[X];
char s[X]; 
int main(){
	scanf("%s",s);
	int cnt=10,qu=0;
	long long sum=1;
	if(s[0]=='?')sum=9;
	if(s[0]>='A'&&s[0]<='Z')sum=9,f[s[0]-'A']=1,cnt--;
	for(int i=1;s[i];i++){
		if(s[i]=='?')qu++;
		if(s[i]>='A'&&s[i]<='Z'&&!f[s[i]-'A']){
			sum*=cnt--;
			f[s[i]-'A']=1;
		}
	}
	printf("%I64d",sum);
	while(qu--)putchar('0');puts("");
	return 0;
}
