# include <bits/stdc++.h>

using namespace std;

typedef long long int LL;

int main(){
	string s;
	cin >> s;
	int cnt = 0;
	set <char> st;
	for (int i = 0; i < s.size(); i++){
		if(s[i] == '?')
			cnt++;
		if('A' <= s[i] && s[i] <= 'J')
			st.insert(s[i]);
	}
	int zer = 0;
	LL ans = 1;
	while(cnt--)
		zer++;
	for (int i = 0; i < st.size(); i++)
		ans *=  (10 - i);

	if(s[0] == '?' || ('A' <= s[0] && s[0] <= 'J')){
		if(ans % 10 == 0)
			ans /= 10, ans *= 9;
		else
			zer--, ans *= 9;
	}

	cout << ans;
	while(zer--)
		cout << "0";
	cout << endl;

	return 0;
}
