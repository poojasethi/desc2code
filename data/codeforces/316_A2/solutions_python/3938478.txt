#include <iostream>
#include<string>
#include<string.h>
using namespace std;


int main()
{
    string s;cin>>s;
    int vis[27],i,c=10,k=0;
    memset(vis,0,sizeof(vis));
    long long ans=1;
    for(i=0;i<s.size();i++)
    {
        if(s[i]=='?'){
        if(i) k++;
        else ans*=9;
        }
        else if(s[i]>='A'&&s[i]<='Z'&&vis[s[i]-'A']==0) {
         vis[s[i]-'A']=1;
         if(i) ans*=c;
         else ans*=9;
         c--;
        }
    }
    cout<<ans;
    for(i=0;i<k;i++) cout<<0;
    return 0;
}
