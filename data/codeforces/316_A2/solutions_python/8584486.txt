#include<stdio.h>
#include<stdlib.h>
#include<string.h>
#include<algorithm>
using namespace std;
int an,c;
int bs[10005];
int main(){
	char ch=getchar();
	int now=9,p=0;an=1;
	while(ch!='\n'){
		if(ch>='0'&&ch<='9');else
		if(ch=='?'){
			if(now<10)an*=now;else ++c;
		}else
		if(ch>='A' &&ch<='Z'){
			if(!bs[ch-'A']){
				if(now-p<0) return puts("0"),0;
				an*=now-p;
				bs[ch-'A']=1;
				++p;
			}
		}
		now=10;
		ch=getchar();
	}
	printf("%d",an);
	for(int i=1;i<=c;i++)printf("0");
	return 0;
}
