# include <stdio.h>
# include <string.h>

long long ans=1;

char a[100050];

bool vis[500];

int main()
{
	scanf("%s",a);
	
	int s=strlen(a);
	
	int k=10,sh=0;
	
	for (int i=0;i<s;i++)
	{
		if(a[i] != '?' && a[i] > 60 && vis[a[i]] == 0)
		{
			if(i == 0) ans*=(k-1);
			else ans*=k;
			vis[a[i]]=1;
			k--;
		}
		
		if(a[i] == '?') sh++;
	}
	
	if(a[0] == '?') ans*=9,sh--;
	
	printf("%I64d",ans);
	
	for (int i=0;i<sh;i++) printf("0");
	
	getchar();getchar();
}
