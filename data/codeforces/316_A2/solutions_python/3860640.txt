#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int i,j,k,l,n,m,o,num,bo[1200],ans;
char c[120000];
int main(){
	scanf("%s",c+1);
	n=strlen(c+1);
	ans=1;
	for(i=1;i<=n;++i){
		if(c[i]=='?')num++;
		else if(c[i]>='A' && c[i]<='Z')
			if(!bo[c[i]])bo[c[i]]=1,o++;
	}
	for(i=0;i<o;++i)ans*=10-i;
	if(c[1]>='A' && c[1]<='Z')ans=ans/10*9;
	else if(c[1]=='?')ans*=9,num--;
	printf("%d",ans);
	for(i=1;i<=num;++i)printf("0");
}
