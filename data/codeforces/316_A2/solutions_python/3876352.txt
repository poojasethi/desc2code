#include<stdio.h>
#include<iostream>
#include<string.h>
using namespace std;
int main(){
	char s[100004];
	scanf("%s",s);
	long long int r=1;
	int count=0;
	int mult=0,flag[11]={0};
	for(int i=0;i<strlen(s);i++){
		if(i==0 && s[i]>='A'){
			r=r*9;
			mult++;
			flag[s[i]-'A']++;
		}
		if(i>0 && s[i]>='A'){
			if(flag[s[i]-'A']==0){
				flag[s[i]-'A']++;
				r=r*(10-mult);
				mult++;
			}
		}
		if(i==0 && s[i]=='?'){
			r=r*9;
		}
		if(i!=0 && s[i]=='?'){
			count++;
		}
	}
	cout<<r;
	for(int i=0;i<count;i++)printf("0");
	printf("\n");
	return 0;
}
