#include <iostream>
#include <string>
#define long long long
using namespace std;

string s;
bool b[20];
long n=10,nol=0;

int main()
{
    long ans=1;
cin>>s;
if (s[0]=='?')
    ans=9;
if (s[0]>='A' && s[0]<='J'){
    ans=9;
    b[s[0]-'A']=1;
    n--;
}

for (long i=1;i<s.length();i++){
if (s[i]=='?')
    nol++;
if (s[i]>='A' && s[i]<='J')
    if (!b[s[i]-'A'])
        b[s[i]-'A']=1,ans*=n,n--;
}
cout<<ans;
for (long i=0;i<nol;i++)
    cout<<0;
    return 0;
}
