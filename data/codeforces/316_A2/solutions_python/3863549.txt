#include <cstdio>
#include <iostream>
using namespace std;

char buf[100005];
bool was[256];

int main() {
	scanf("%s", buf);
	int zeros = 0, ans = 1, left = 10;
	for (int i = 0; buf[i]; ++i) {
		if (buf[i] >= 'A' && buf[i] <= 'Z') {
			if (!was[buf[i]]) {
				ans *= left - (i == 0);
				--left;
				was[buf[i]] = true;
			}
		} else if (buf[i] == '?') {
			if (i != 0) {
				++zeros;
			} else {
				ans *= 9;
			}
		}
	}
	cout << ans;
	for (int i = 0; i < zeros; ++i) {
		cout << '0';
	}
	cout << endl;
	return 0;
}
