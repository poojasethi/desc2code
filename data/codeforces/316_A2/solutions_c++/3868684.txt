#include <iostream>
using namespace std;
typedef long long int LLI;

char S[100002];
int d, l, p1, p2, U[256];
LLI w = 1;

int main() {
	ios_base::sync_with_stdio(false);
	cin >> (S + 1);
	for (int i = 1; S[i]; i++, d++) {
		if ('A' <= S[i] && S[i] <= 'J') {
			U[S[i]] = 1;
			if (i == 1) p1 = 1;
		} else if (S[i] == '?') {
			l++;
			if (i == 1) p2 = 1;
		}
	}
	for (int i = 'A', j = 10; i <= 'J'; i++) {
		if (U[i]) w *= j--;
	}
	if (p1) {
		w = w / 10 * 9;
	}
	if (p2) {
		w *= 9;
		l--;
	}
	cout << w;
	while (l--) cout << '0';
	cout << endl;
	return 0;
}
