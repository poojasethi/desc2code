#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,i,m,t,p;
__int64 ans;
char s[100010],v[200];
int main()
{
scanf("%s",s+1);
n=strlen(s+1);
for(i=1;i<=n;i++)
if(s[i]>='A'&&s[i]<='J'&&!v[s[i]])
v[s[i]]=1,m++;
else
if(s[i]=='?')
t++;
ans=1;
if(s[1]>='A'&&s[1]<='J')
p++,ans=9;
if(s[1]=='?')
t--,ans=9;
for(;p<m;p++)
ans*=10-p;
cout<<ans;
while(t--)
putchar('0');
cout<<endl;
}
