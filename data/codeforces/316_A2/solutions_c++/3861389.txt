#include <stdio.h>

char a[100000];
char b[255];
int ans,ans0,p,p2;
int main()
{
    scanf("%s",a);
    if(a[0]<='9'&&a[0]>='1')ans=1;
    if(a[0]=='?')ans=9;
    if(ans==0){ans=9;p-=1;p2=1;b[a[0]]=1;}
    for(int i=1;a[i]!=0;i++){
        if(a[i]=='?')ans0++;
        else if(a[i]>'9'||a[i]<'0')b[a[i]]=1;
    }
    for(int i='A';i<='J';i++)if(b[i])p++;
    for(int i=0;i<p;i++)ans*=(10-i-p2);
    printf("%d",ans);
    for(int i=0;i<ans0;i++)printf("0");
    return 0;

}
