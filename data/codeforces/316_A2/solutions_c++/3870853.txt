#include<cstdio>
#include<cstring>
#include<algorithm>
#define rep(i,n) 
char s[100005]; int n,c,ans,cnt; bool b[255];
int main()
{
	scanf("%s",s),n=strlen(s),c=10;
	if (*s=='?') ans=9;
	else if (*s>='A' && *s<='J') ans=9,c=9,b[*s]=1;
	else ans=1;
	for (int i=1;i<n;i++){
		if (s[i]=='?') cnt++;
		else if (s[i]>='A' && s[i]<='J')
			if (!b[s[i]]) ans*=c--,b[s[i]]=1;
		}
	printf("%d",ans);
	while (cnt--) putchar('0'); puts("");
	return 0;
}
