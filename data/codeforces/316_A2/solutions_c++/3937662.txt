#include<iostream>
#include<string>
using namespace std;
int d[1000];
int main()
{
	string s;
	int i;
	unsigned long long ans=1;
	cin>>s;
int x=0;
	for(i=0;i<s.length();i++)
	{
		if(s[i]>='A'&&s[i]<='J'&&d[s[i]-'A']==0)
		{
			if(i==0) ans*=9;
			else ans*=10-x;
			x++;
			d[s[i]-'A']=-1;
		}
	}
int bb=0;
	for(i=0;i<s.length();i++)
	{
		if(s[i]=='?') {if(i==0) ans*=9; else bb++;}
	}
	
	cout<<ans;
for(i=0;i<bb;i++)cout<<0;
	return 0;
}
