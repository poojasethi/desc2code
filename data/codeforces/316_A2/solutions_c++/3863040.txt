#include <map>
#include <cctype>
#include <iostream>

using namespace std;

long long ret = 1, cnt;
bool mark[333];
string s;

int main()
{
  cin >> s;
  for (int i = 0; i < s.size(); i++)
    if (s[i] == '?')
    {
      if (i == 0) ret *= 9;
      else cnt++;
    }
    else if (isalpha(s[i])) mark[s[i]] = true;
  for (int i = 'A', cnt = 0; i <= 'J'; i++) if (mark[i]) ret *= 10 - cnt++;
  if (isalpha(s[0])) ret *= 9, ret /= 10;
  cout << ret;
  for (int i = 0; i < cnt; i++) cout << 0;
  return 0;
}
