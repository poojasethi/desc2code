#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

string s;
int k, c[26];
long long res;

long long ff(char x)
{
  if (x == '?') return 10;
  if (x >= 'A' && !c[x - 65]) 
  {
    c[x - 65] = 1;
    return k--;
  }
  return 1;
}

main()
{
  cin >> s;
  k = 10;
  res = min(ff(s[0]), 9LL);
  
  int l = 0;
  for (int i = 1; i < s.size(); i++) 
    if (s[i] == '?') l++;
    else res *= ff(s[i]);
  cout << res;
  while (l--) cout << 0;
 // system("pause");
}
