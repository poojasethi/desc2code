n = input()
s = str(raw_input())
arr = []
arr1 = []
arr2 = []
ans = 0
b = 0
for _ in s:
    arr.append(int(_))

for _ in range(0,n-1,2):
    arr1.append(1)
    arr1.append(0)
    arr2.append(0)
    arr2.append(1)
if n%2 == 1:
    arr1.append(1)
    arr2.append(0)

for i in range(0,n):
    if arr[i] != arr1[i]:
        ans = ans +1
    if arr[i] != arr2[i]:
        b = b +1
    
print min(ans,b)
