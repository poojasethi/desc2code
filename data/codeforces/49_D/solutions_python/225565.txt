n = int(raw_input())
stripe = raw_input()
d1 = 0
for i in range(n):
    if i % 2:
        if stripe[i] == '1':
            d1 += 1
    else:
        if stripe[i] == '0':
            d1 += 1
d2 = 0
for i in range(n):
    if i % 2:
        if stripe[i] == '0':
            d2 += 1
    else:
        if stripe[i] == '1':
            d2 += 1
print(min(d1,d2))
