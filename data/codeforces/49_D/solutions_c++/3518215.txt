#include<stdio.h>
int main()
{
	int n;
	char str[1100],ss[1100];
	scanf("%d%s",&n,str);
	int i;
	for(i=0;i<n;i++)
	{
		if(i%2==0) ss[i]='0';
		else ss[i]='1';
	}
	int ans=0;
	for(i=0;i<n;i++)
	{
		if(ss[i]!=str[i]) ans++;
	}
	if(ans*2>n) ans=n-ans;
	printf("%d\n",ans);
	return 0;
}
