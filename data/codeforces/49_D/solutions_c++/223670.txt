#include <stdio.h>
#define N 1005
int main() {
	int n, res = N;
	char pat[N];
	scanf("%d%s", &n, pat);
	//printf("%d %s\n", n, pat);
	for(int i = 0; i != 2; ++i) {
		int cnt = 0;
		for(int j = 0; j != n; ++j)
			if(pat[j] - '0' != (j + i) % 2) cnt++;
		if(res > cnt) res = cnt;
	}
	printf("%d", res);
	return 0;
}
