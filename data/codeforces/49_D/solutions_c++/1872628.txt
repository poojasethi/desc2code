#include<stdio.h>
int n;
char s[20000];
int main(){
	scanf("%d%s",&n,s);
	int ans,tmp,i;
	for(i=ans=0;i<n;i++){
		if((i%2)== ((s[i]-'0')%2))
			ans++;
	}
	for(i=tmp=0;i<n;i++){
		if((i%2)!= ((s[i]-'0')%2))
			tmp++;
	}
	if(ans>tmp)
		ans=tmp;
	printf("%d\n",ans);
	return 0;
}
