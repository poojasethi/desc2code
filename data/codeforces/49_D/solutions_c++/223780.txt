#include<iostream>
#include<algorithm>
#include<string>
using namespace std;

int main() {
	int sz; string s; cin>>sz>>s;
	int A=0,B=0;
	for (int i=0; i<sz; i++) {
		if(i%2==0) {
			if(s[i]!='0') A++;
			else B++;
		} else {
			if(s[i]!='1') A++;
			else B++;
		}
	}
	cout << min(A,B) << endl;
	
	return 0;
}
