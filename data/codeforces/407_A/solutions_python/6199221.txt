a, b = map(int, raw_input().split())

m, M = min(a, b), max(a, b)

lol = True
for x in xrange(-1, -m, -1):
    y2 = m ** 2 - x ** 2
    y = int(y2 ** 0.5)
    
    if y ** 2 == y2:
        xx = (y * M) // m
        yy = (-x * M) // m
        
        if y != yy and xx ** 2 + yy ** 2 == M ** 2 and x != xx:
            print 'YES\n0 0\n{} {}\n{} {}'.format(x, y, xx, yy)
            lol = False
            break
    
if lol:
    print 'NO'
