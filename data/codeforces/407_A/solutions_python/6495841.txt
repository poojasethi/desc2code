def catets(h):
    h2 = h * h
    res =[]
    for i in xrange(1, h):
        i2 = i * i
        c2 = h2 - i2
        c = int(round(math.sqrt(c2)))
        
        if c2 == c * c:
           # if c2 < i2 :break
            res.append((i,c))
        
    return res
