#include <stdio.h>
#include <math.h>

int a, b;

void Run() {
	for (int x=1; x<a; x++) {
		int y=sqrt(a*a-x*x);
		if (x*x+y*y!=a*a) continue;
		for (int xx=1; xx<b; xx++) {
			int yy=sqrt(b*b-xx*xx);
			if (xx*xx+yy*yy!=b*b) continue;
			yy=-yy;
			if (x==xx) continue;
			if (x*xx+y*yy==0) {
				puts("YES");
				puts("0 0");
				printf("%d %d\n", x, y);
				printf("%d %d\n", xx, yy);
				return;
			}
		}
	}
	puts("NO");
}

int main() {
	scanf("%d %d", &a, &b);
	Run();
}
