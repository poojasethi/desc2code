#include<bits/stdc++.h>
std::map<int,int> f;
int main()
{
	int i,a,b;
	scanf("%d%d",&a,&b);
	for(i=1;i<=1000;i++) f[i*i]=i;
	for(i=1;i<=a;i++){
		int k=a*a-i*i;
		if(f[k]&&i*b%a==0){
			int t=i*b/a;
			int q=b*b-t*t;
			if(f[q]&&f[k]!=t){
				printf("YES\n0 0\n%d %d\n%d %d\n",-i,f[k],f[q],t);
				return 0;
			}
		}
	}
	puts("NO");
}
