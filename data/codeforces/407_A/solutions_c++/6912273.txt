#include<stdio.h>
#include<algorithm>
#include<math.h>
using namespace std;
int a,b;
void process(){
	int i,j,k,x,y;
	int ok=0;
	for(i=1;i<a;i++){
		j=sqrt(a*a-i*i);
		if((i*i+j*j)!=a*a)continue;
		x=(b*j)/a;
		y=(b*i)/a;
		if((x*x+y*y)!=b*b)continue;
		if(y==j)continue;
		ok=1;
		break;
	}
	if(!ok)puts("NO");
	else{
		puts("YES");
		printf("%d %d\n%d %d\n%d %d\n",0,0,i,j,-x,y);
	}
	return;
}
int main(){
	scanf("%d %d",&a,&b);
	process();
	return 0;
}
