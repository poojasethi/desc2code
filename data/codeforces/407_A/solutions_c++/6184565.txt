#include <cstdio>

const int N = 1000005;

int sqr[N];

int main() {
	for (int i = 1; i * i < N; ++i) {
		sqr[i * i] = i;
	}
	int a, b;
	scanf("%d%d", &a, &b);
	for (int x0 = 1; x0 < a; ++x0) {
		int y0 = sqr[a * a - x0 * x0];
		if (!y0) continue;
		for (int x1 = 1; x1 < b; ++x1) {
			int y1 = sqr[b * b - x1 * x1];
			if (!y1 || y1 == y0) continue;
			if (x0 * x1 == y0 * y1) {
				printf("YES\n0 0\n%d %d\n%d %d\n", x0, y0, -x1, y1);
				return 0;
			}
		}
	}
	puts("NO");
}
