#include <cstdio>
using namespace std;
int gcd(int x,int y)
{
	if (!y) return x; return gcd(y,x%y);
}
int main()
{
	int a,b,x=0,y,i,j;
	scanf("%d%d",&a,&b);
	int g=gcd(a,b);
	for (i=1;i<g;i++)
	    for (j=1;j<g;j++) if ((i*a!=j*b)&&(i*i+j*j==g*g)) x=i,y=j;
	if (!x) {printf("NO\n"); return 0;}
	printf("YES\n");
	printf("1 1\n%d %d\n%d %d\n",1+a/g*x,1+a/g*y,1+b/g*y,1-b/g*x);
	return 0;
}
