#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int x, y;
int main(){
	int a,b; cin >> a >> b;
	for (int i = 1; i < a; i++){
		for (int j = 1; j < b; j++){
			x = sqrt(a * a - i * i);
			y = sqrt(b * b - j * j);
			if (i * j == x * y && i * i + x * x == a * a && j * j + y * y == b * b && x != y){
				cout << "YES\n0 0\n" << i << " " << x << endl << -j << " " << y << endl;
				return 0;
			}
		}
	}
	printf("NO\n");
}
