#include<stdio.h>
#include<algorithm>
using namespace std;
int a,b;
int gcd(int u,int v)
{
	if(!v) return u;
	return gcd(v,u%v);
}
int main()
{
	int i,j;
	scanf("%d%d",&a,&b);
	for(i=1;i<=a;i++)
		for(j=1;j<=i;j++)
		{
			int g=gcd(i,j);
			if(i*i+j*j==a*a&&!(b*g%a))
			{
				printf("YES\n");
				if(j==i*b/a)
					printf("0 0\n%d %d\n%d %d",i*b/a,j*b/a,-j,i);
				else
					printf("0 0\n%d %d\n%d %d\n",i,j,-j*b/a,i*b/a);
				return 0;
			}
		}
	printf("NO\n");
	return 0;
}
