#include <cstring>
#include <cmath>

using namespace std;

int a, b;

int main() {
	scanf("%d%d", &a, &b);
	for (int i = 1; i < b; ++i) if (i * a % b == 0) {
		int j = sqrt(b * b - i * i);
		if (j * j + i * i < b * b) ++j;
		if (j * j + i * i == b * b && j != i * a / b) {
			int x = i * a / b, y = j * a / b;
			printf("YES\n%d %d\n%d %d\n%d %d\n", -y, x, 0, 0, i, j);
			return 0;
		}
	}
	printf("NO\n");
	return 0;
}
