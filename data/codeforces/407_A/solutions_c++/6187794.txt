#include<iostream>
using namespace std;
int a,b;
int gcd(int x, int y) {
	return (y?gcd(y,x%y):x);
}	
int main(){
	cin>>a>>b;
	int d=gcd(a,b);
	for(int k=1;k<d;k++) {
		for(int l=1;l<d;l++) {
			if(k*k+l*l==d*d) {
				cout<<"YES\n";
				cout<<"0 0\n";
				cout<<a/d*k<<" "<<a/d*l<<"\n";
				if(b/d*l!=a/d*k)cout<<b/d*l<<" "<<-b/d*k<<"\n";
				else cout<<-b/d*l<<" "<<b/d*k<<"\n";
				return 0;
			}
		}			
	}
	cout<<"NO";
	return 0;
}
