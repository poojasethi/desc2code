#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int a, b;
	cin >> a >> b;
	for (int i = 1; i < a; i++){
		int u = sqrt(a*a - i*i);
		if (i*i + u*u != a*a) continue;
		for (int j = 1; j < b; j++){
			int q = sqrt(b*b - j*j);
			if (q*q + j*j != b*b) continue;
			q *= (-1);
			if (i == j) continue;
			if (i*j + q*u == 0){
				cout << "YES\n" << "0 0\n" << i << ' ' << u << '\n'
					<< j << ' ' << q;
				return 0;
			}
		}
	}
	cout << "NO";
	return 0;
}
