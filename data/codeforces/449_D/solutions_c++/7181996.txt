#include<cstdio>
using namespace std;

const int l = 20, M = 1e9+7, maxn = 1e6;
int n, a, ans, cnt[1<<l], pow[maxn+1];
int main()
{
  scanf("%d", &n);
  for (int i = 0; i < n; i++) {
    scanf("%d", &a);
    cnt[a]++;
  }
  pow[0] = 1;
  for (int i = 1; i <= n; i++) pow[i] = (pow[i-1]<<1)%M;
  for (int i = 0; i < l; i++)
    for (int u = (1<<l)-1; u > 0; u--)
      if (((u>>i)&1) == 1) cnt[u^(1<<i)] += cnt[u];
  for (int u = 0; u < (1<<l); u++) {
    int sum = 0;
    for (int i = 0; i < l; i++)
      if (((u>>i)&1) == 1) sum++;
    if (sum&1) {
      ans -= pow[cnt[u]];
    } else {
      ans += pow[cnt[u]];
    }
    ans %= M;
  }
  printf("%d\n", (ans+M)%M);
}
