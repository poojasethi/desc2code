#include <cstdio>

#define N 2000000
#define mod 1000000007

using namespace std;

int n,a[N+10],b[N+10],p[N+10],c[N+10];
long long ans,t;

int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%d",&a[i]),b[a[i]]++;
	p[0]=1;
	for (int i=1;i<=n;++i) p[i]=p[i-1]*2%mod;
	for (int i=0;i<20;++i)
		for (int j=0;j<=1048575;++j)
			if (j&(1<<i)) b[j-(1<<i)]+=b[j],++c[j];
	for (int i=0;i<=1048575;++i){
		t=(p[b[i]]+mod-1)%mod;
		if (c[i]&1) ans=(ans+mod-t)%mod;
		else ans=(ans+t)%mod;
	}
	printf("%I64d\n",ans);
	return 0;
}
