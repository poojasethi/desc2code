#include <bits/stdc++.h>
using namespace std;
typedef long long LL;
const LL mod = 1e9 + 7;
int n;
int cnt[1 << 20];
LL pow_mod(LL a,LL b,LL p){LL t=1;for(;b;b>>=1,a=a*a%p)if(b&1)t=t*a%p;return t;}
int main()
{
	cin >> n;
	for(int i = 1; i <= n; i++)
	{
		int x;
		scanf("%d", &x);
		cnt[x]++;
	}
	for(int j = 0; j < 20; j++)
	for(int i = 0; i < (1 << 20); i++)
		if((i >> j & 1) == 0) cnt[i] += cnt[i ^ (1 << j)];
	LL ans = 0;
	for(int s = 0; s < (1 << 20); s++)
	{
		LL tmp = pow_mod(2, cnt[s], mod);
		if(__builtin_popcount(s) & 1) ans -= tmp;
		else ans += tmp;
		ans %= mod;
	}
	cout << (ans % mod + mod) % mod << endl;
	return 0;
}
