#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
#define V 1048576
#define MOD 1000000007
long long f[V],pow2[V+1];
int cnt[V];
int main()
{
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		int t;scanf("%d",&t);
		f[t]++;
	}
	pow2[0]=1;
	for(int i=1;i<=V;i++)
		pow2[i]=pow2[i-1]*2%MOD;
		
	for(int i=0;i<20;i++)
	for(int j=0;j<V;j++)
	{
		if((j>>i)&1)
		{
			cnt[j]++;
			f[j-(1<<i)]=(f[j-(1<<i)]+f[j])%MOD;
		}
	}
	long long ans=0;
	for(int i=0;i<V;i++)
	{
		if(cnt[i]%2==0)ans=(ans+pow2[f[i]])%MOD;
		else ans=(ans+MOD-pow2[f[i]])%MOD;
	}
	cout<<ans<<endl;
}
