#include<iostream>
using namespace std;
const int M=20 , N = (1<<M) , MOD=1e9 +7;
long long cnt[N] , dp[N] , pw[N];

int main(){
	int n;cin >> n;
	pw[0] = 1;
	for(int i=0 ; i<n ; i++){
		int d;cin >> d;
		cnt[d] ++;
		pw[i+1] = (2*pw[i])%MOD;
	}
	for(int i=0 ; i<M ; i++)
		for(int j=(1<<M)-1 ; j>=0 ; j--)
			if( j&(1<<i) )
				cnt[ j^(1<<i) ] += cnt[j];
	long long ans=0;
	for(int i=(1<<M)-1 ; i>=0 ; i--){
		ans = ans + ( __builtin_popcount(i)&1 ? -pw[ cnt[i] ] : pw[ cnt[i] ] );
		ans += MOD;ans%=MOD;
	}
	cout << ans << "\n";
}
