#include <cstdio>
#include <iostream>
using namespace std;
const int N = 2000000+10;
const int MOD = 1000000007;
int f[N],c[N];
long long ans = 0;
long long p[N];
int main(){
	int n,t;
	scanf("%d",&n);
	for(int i = 0;i < n;i++){
		scanf("%d",&t);
		f[t]++;
	}
	p[0] = 1;
	for(int i = 1;i <= n;i++) p[i] = p[i-1]*2%MOD;
	for(int i = 0;i < 20;i++)
		for(int j = 0;j <= 1048575 ;j++)
			if( j & (1 << i)) 
				f[j - (1<<i)] += f[j],++c[j];
	for(int i = 0;i < (1<<20);i++){
		long long t = p[f[i]];
		if(c[i] % 2 ==  1) 
			ans = (ans + MOD - t) % MOD;
		else
			ans = (ans + t) %  MOD;
	}
	cout << ans << endl;
	return 0;
}
