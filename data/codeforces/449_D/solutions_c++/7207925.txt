#include<stdio.h>
using namespace std;
#define M 1000100
#define mod 1000000007
#define LL __int64
int f[1 << 21], num[1 << 21], g[M], n, x;
LL ans;

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", &x), f[x]++;
    g[0] = 1;
    for (int i = 1; i <= n; i++) g[i] = g[i - 1]*2 % mod;
    for (int i = 0; i < 20; i++)
        for (int j = 0; j < (1 << 20); j++)
            if (!(j >> i & 1)) f[j] += f[j | (1 << i)];
            else num[j]++;
    for (int i = 0; i < (1 << 20); i++) {
        LL tmp = g[f[i]] - 1;
        if (num[i]&1) ans = (ans - tmp + mod) % mod;
        else ans = (ans + tmp) % mod;
    }
    printf("%I64d\n", ans);
}
