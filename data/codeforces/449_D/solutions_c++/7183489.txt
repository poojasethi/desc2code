#include <stdio.h>
#include <string.h>
#define MOD 1000000007
#define N 1000020
int cnt[1<<20],n;
long long pw2[1<<20];
int main()
{
	scanf("%d",&n);
	pw2[0] = 1;
	for (int i=1;i<=n;i++) pw2[i] = pw2[i-1] * 2 % MOD;
	for (int i=1;i<=n;i++) 
	{
		int x;
		scanf("%d",&x);
		cnt[x]++;
	}
	for (int i=0;i<20;i++)
		for (int j=0;j<(1<<20);j++)
			if ((j>>i)&1) cnt[j-(1<<i)] += cnt[j];
	long long ans = pw2[n];
	for (int i=1;i<(1<<20);i++)
	{
		int tot = 0;
		for (int j=0;j<20;j++)
			if ((i>>j)&1) tot++;
		if (tot % 2 ==1) 
			ans += MOD - pw2[cnt[i]];
		else ans += pw2[cnt[i]];
		if (ans>=MOD) ans -= MOD;
	}
	printf("%I64d\n",ans);
	return 0;
}
