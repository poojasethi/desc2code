#include <iostream>
#include <cstdio>

using namespace std;

const int N = 1000000 + 10;
const int B = 20;
const int MOD = 1000000007;

int a[N], p2[N];
int n, ans;

int main()
{
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		static int x;
		scanf("%d", &x);
		++a[x];
	}
	for (int j = 0; j < B; ++j)
		for (int i = N - 1; i >= 0; --i)
			if (i >> j & 1) a[i ^ (1 << j)] += a[i];
	p2[0] = 1;
	for (int i = 1; i < N; ++i) p2[i] = (p2[i - 1] + p2[i - 1]) % MOD;
	ans = p2[n] - 1;
	for (int i = 1; i < N; ++i) {
		int sgn = 1;
		for (int j = 0; j < B; ++j)
			if (i >> j & 1) sgn = -sgn;
		ans = (ans + sgn * (p2[a[i]] - 1)) % MOD;
	}
	cout << (ans + MOD) % MOD << endl;
	return 0;
}
