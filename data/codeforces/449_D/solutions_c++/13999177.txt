#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N (1<<20)
#define mo 1000000007
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,an;
int f[N],h[N];
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%d",&k),f[k]++;
	m=N-1;
	For(i,0,19)For(j,0,m) if (!(j&(1<<i))) f[j]+=f[j|(1<<i)];
	h[0]=1; For(i,1,N-1) h[i]=h[i-1]*2%mo;
	For(i,0,m) {
		int s=0;
		For(j,0,19) s+=i>>j&1;
		if (f[i]) an=((s&1?-1:1)*(h[f[i]]-1)+an)%mo;
	}
	if (an<0) an+=mo;
	printf("%d\n",an);
	return 0;
}
