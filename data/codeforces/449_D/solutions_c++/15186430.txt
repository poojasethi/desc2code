# include <cstdio>

# define REP(i, n) for (int i = 1; i <= n; ++ i)
# define REP_0(i, n) for (int i = 0; i < n; ++ i)
# define p 1000000007
const int L = 20, M = 1 << 20;
int pw[1001000], a[1 << 20], d[1 << 20], n;
int main () {
    scanf ("%d", &n), pw[0] = 1; int ta; d[0] = 1;
    REP (i, n) pw[i] = (pw[i - 1] << 1) % p;
    REP (i, n) scanf ("%d", &ta), ++ a[M - ta - 1];
    REP_0 (i, L) REP_0 (s, M) if (s >> i & 1) a[s] += a[s ^ 1 << i];
    int ans = 0;
    REP_0 (s, M) d[s] = (s & 1) ? (-d[s >> 1]) : d[s >> 1], ans = (ans + 1ll * d[s] * (pw[a[M - s - 1]] + p - 1)) % p;
    printf ("%d\n", ans);
    return 0;
}
