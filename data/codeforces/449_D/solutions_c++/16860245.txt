#include <iostream>
using namespace std;

const int MAXLG = 21;
const int MAXN = 1e6;
const int MOD = 1e9+7;

int n, ans, x, dp[MAXN+5], p2[MAXN+5];

int main(){
    ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);
	p2[0] = 0;
	for (int i = 1; i < MAXN+2; i++)
		p2[i] = (p2[i-1]*2+1)%MOD;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> x, dp[x]++;
	for (int i = 0; i < MAXLG+2; i++)
		for (int j = 0; j < MAXN+2; j++)
			if ((j>>i)&1)
				dp[j^(1<<i)] += dp[j];
	for (int i = 0; i < MAXN+2; i++)
		ans = ((ans+p2[dp[i]]*((__builtin_popcount(i)&1) ? -1 : 1))%MOD+MOD)%MOD;
	cout << ans << "\n";
    return 0;
}
