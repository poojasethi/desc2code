#include<cstdio>
using namespace std;
typedef long long LL;
const LL P=1000000007;
const LL U=1048575;
LL n,i,x,j,opt,ans=0,M;
LL f[U+5],c[U+5],pow[U+5];
int main()
{
  scanf("%I64d",&n);
  for (i=1;i<=n;i++)
    scanf("%I64d",&x),f[x]++;
  for (j=0;j<20;j++)
    for (i=0;i<=U;i++)
      if ((1<<j)&i)
        f[i-(1<<j)]+=f[i],c[i]++;
  pow[0]=1;for (i=1;i<=n;i++) pow[i]=pow[i-1]*2%P;
  for (i=0;i<=U;i++)
  {
    opt=(c[i]&1)?-1:1;
    ans=(ans+((pow[f[i]]-1+P)%P*opt)+P)%P;
  }
  printf("%I64d",ans);
  return 0;
}
