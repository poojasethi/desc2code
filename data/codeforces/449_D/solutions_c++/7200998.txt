#include<iostream>
using namespace std;
const int MAXN = (1 << 20);
const long long MOD = 1e9 + 7;
int a[MAXN] , f[MAXN] , pow[MAXN];
int main()
{
	
	int n;
	cin >> n;

	for(int i = 0;i < n;i ++)
	{
		cin >> a[i];
		f[(1 << 20) - 1 - a[i]] ++;
	}
	
	for(int i = 20;i >= 0;i --)
		for(int j = (1 << 20);j >= 0;j --)
			if(j & (1 << i))
				f[j] += f[j ^ (1 << i)];
	
	pow[0] = 1ll;
	for(int i = 1;i <= n;i ++)
		pow[i] = (pow[i - 1] * 2) % MOD;
	
	long long ans = 0;
	for(int i = 0;i < (1 << 20);i ++)
	{
		int q = __builtin_popcount(i);
		if(q % 2)
			ans = (ans - pow[f[i]] + MOD) % MOD;
		else
			ans = (ans + pow[f[i]]) % MOD;
	}
	cout << ans << endl;
	return 0;
}
