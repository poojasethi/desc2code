#include<cstdio>
using namespace std;
long long P=1000000007ll,ans,f[20][1100010],p2[1100010],now;
int n,i,x,l,j;
int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;i++) {scanf("%d",&x);f[0][x]++;if (x&1) f[0][x-1]++;}
	p2[0]=1;
	for (i=1;i<=n;i++) p2[i]=p2[i-1]*2%P;
	for (l=1;l<=19;l++)
	  for (i=0;i<(1<<20);i++)
	    if (i&(1<<l))   f[l][i]=f[l-1][i];
	    else f[l][i]=(f[l-1][i]+f[l-1][i+(1<<l)])%P;
	for (i=0;i<(1<<20);i++)
	{
		now=1;
		for (j=0;j<=19;j++)
		  if (i&(1<<j)) now=-now;
		ans=(ans+now*p2[f[19][i]]+P)%P;
	}
	printf("%I64d\n",ans);
	return 0;
}
