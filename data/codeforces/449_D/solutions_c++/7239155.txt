#include <iostream>

using namespace std;

const int MAXN = 1e6 + 10, MAXLG = 21, MOD = 1e9 + 7;
int dp[MAXN], p2[MAXN];

int main() {
	p2[0] = 0;
	for (int i = 1; i < MAXN; i++)
		p2[i] = (p2[i - 1] * 2 + 1) % MOD;
	int n, x;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> x, dp[x]++;
	for (int j = 0; j < MAXLG; j++)
		for (int i = 0; i < MAXN; i++)
			if ((i >> j) & 1)
				dp[i ^ (1 << j)] += dp[i];
	int ans = 0;
	for (int i = 0; i < MAXN; i++)
		ans = ((ans + p2[dp[i]] * (__builtin_popcount(i) & 1 ? -1 : 1)) % MOD + MOD) % MOD;
	cout << ans << endl;
}
