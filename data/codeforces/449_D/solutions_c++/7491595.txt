#include<cstdio>
#include<cstdlib>
#define MOD 1000000007
int f[5000001],g[5000001],ans=0;
long long power(long long a,int b){
	long long res=1;
	while (b){
		if(b&1) res=res*a%MOD;
		b>>=1; a=a*a%MOD;
	}
	return res;
}
int main(){
	int n,N,x;
	scanf("%d",&n);
	for (int i=1;i<=n;i++){
		scanf("%d",&x);
		f[x]++;
	}
	N=1048575;
	for (int i=1;i<=N;i<<=1){
		for (int j=0;j<=N;j++)
		if(j&i){
			f[j-i]=(f[j]+f[j-i])%MOD;
			g[j]++;
		}
	}
//	printf("%d %d %d %d\n",f[0],f[1],f[2],f[3]);
	for (int i=0;i<=N;i++){
		if(g[i]&1) ans=(ans+MOD-power(2,f[i]))%MOD;
		else ans=(ans+power(2,f[i]))%MOD;
	}
	printf("%d\n",ans);
}
