#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int f[2000000],pow[1000005];
int main()
{
    int n,t;
    cin>>n;
    int ans=0,M=1e9+7;
    memset(f,0,sizeof(f));
    pow[0]=1;
    for(int i=1;i<=1e6;i++) pow[i]=(2*pow[i-1])%M;
    for(int i=0;i<n;i++)
    {
        scanf("%d",&t);
        f[t]++;
    }
    for(int i=0;i<20;i++) for(int j=0;j<=1e6;j++)
        if(j&(1<<i)) f[j^(1<<i)]+=f[j];
    for(int i=0;i<=1e6;i++)
    {
        int count=1;
        for(int j=0;j<20;j++) if(i&(1<<j)) count=-count;
        ans= ((ans+ count*(pow[f[i]]-1) )%M+M)%M;
    }
    printf("%d\n",ans);
    return 0;
}
