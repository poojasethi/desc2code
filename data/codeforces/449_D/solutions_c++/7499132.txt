#include <cstdio>
#include <cstring>
#include <cstdlib>
const int MX=(1<<20)-1;
typedef long long ll;
ll mod=1000000007;
ll f[MX+1];
ll qp(ll&exp)
{
	ll b=2,r=1;
	for(;exp;exp>>=1)
	{
		if(exp&1)r=r*b%mod;
		b=b*b%mod;
	}
	return r;
}
int N;
int make(int x)
{
	int r=0;
	for(;x;x-=x&-x)r++;
	return r&1?-1:1;
}
int main()
{
	scanf("%d",&N);
	for(int t,i=1;i<=N;i++)
	{
		scanf("%d",&t);
		f[t]++;
	}
	for(int i=1; i<=20; i++)
		for(int j=MX; j>=0; j--)
			if(((j>>i-1)&1)^1) f[j]=(f[j]+f[j+(1<<i-1)])%mod;
	ll ans=0;
	for(int i=0;i<=MX;i++)
		ans=(ans+qp(f[i])*make(i))%mod;
	ans=(ans%mod+mod)%mod;
	printf("%d\n",ans);	
	return 0;
}
