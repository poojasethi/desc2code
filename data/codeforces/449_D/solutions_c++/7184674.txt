#include<stdio.h>
#include<string.h>
#define mod 1000000007
int n,f[2000000],i,t,jz[2000000],j,ans,temp;
main()
{
	scanf("%d",&n);
	memset(f,0,sizeof(f));
	for(i=1;i<=n;i++)
	{
		scanf("%d",&t);
		f[t]++;
	}
	jz[0]=1;
	for(i=1;i<1<<20;i++) jz[i]=jz[i-1]*2%mod;
	for(i=0;i<20;i++)
		for(j=1;j<1<<20;j++)
			if(j>>i&1)
				f[j-(1<<i)]+=f[j];
	ans=0;
	for(i=0;i<1<<20;i++)
	{
		temp=0;
		for(j=0;j<20;j++)
			if(i>>j&1) temp++;
		if(temp%2==0) ans=(ans+jz[f[i]])%mod;
			else ans=(ans+mod-jz[f[i]])%mod;
	}
	printf("%d\n",ans);
}
