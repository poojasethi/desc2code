#include <cstdio>
#include <algorithm>
using namespace std;
#define MOD 1000000007

int n;
int memo[1<<20];
int pw[1000005];

int main(){
	pw[0] = 1;
	for(int i = 1; i < 1000005; i++ ) pw[i] = (pw[i-1]<<1)%MOD;
	scanf("%d",&n);
	for(int i = 0; i < n; i++ ){
		int x; scanf("%d",&x);
		memo[x]++;
	}
	for(int j = 1; j <= 20; j++ ){
		for(int i = 0; i < (1<<20); i++ ){
			if(!(i&(1<<(j-1)))) memo[i] += memo[i|(1<<(j-1))];
		}
	}
	int ans = 0;
	for(int i = 0; i < (1<<20); i++ ){
		if(__builtin_popcount(i)&1) ans -= pw[memo[i]];
		else ans += pw[memo[i]];
		ans %= MOD;
	}
	if(ans<0) ans += MOD;
	printf("%d\n",ans);
	return 0;
}
