#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const ll mod=1000000007;
ll powmod(ll a,ll b) {ll res=1;a%=mod;for(;b;b>>=1){if(b&1)res=res*a%mod;a=a*a%mod;}return res;}
const int N=1<<20;
int n,x;
ll cnt[N+10];
int main() {
	scanf("%d",&n);
	for (int i=0;i<n;i++) { scanf("%d",&x);cnt[x]++;}
	for (int j=0;j<20;j++) for (int i=0;i<N;i++) if (!(i&(1<<j)))
		cnt[i]+=cnt[i+(1<<j)];
	for (int i=0;i<N;i++) cnt[i]=powmod(2,cnt[i]);
	for (int j=0;j<20;j++) for (int i=0;i<N;i++) if (!(i&(1<<j)))
		cnt[i]-=cnt[i+(1<<j)];
	cnt[0]%=mod;
	if (cnt[0]<0) cnt[0]+=mod;
	printf("%I64d\n",cnt[0]);
}
