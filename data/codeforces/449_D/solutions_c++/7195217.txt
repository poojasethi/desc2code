#include <cstdio>

#define maxn (1<<20)+10
#define mod 1000000007

int f[maxn];
bool g[maxn];
int pow[maxn];

inline void inc(int &a,int b)
{
	a+=b;
	while (a >= mod)
		a-=mod;
}

int main()
{
	pow[0]=1;
	for (int i=1;i<=(1<<20);i++)
		inc(pow[i],pow[i-1]+pow[i-1]);
	for (int i=0;i<(1<<20);i++)
	{
		for (int j=i;j;j-=j&-j)
			g[i]=!g[i];
	}
	int n;
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		f[x]++;
	}
	for (int i=1;i<=(1<<19);i<<=1)
		for (int j=0;j<(1<<20);j++)
			if (!(i&j))
				(f[j]+=f[j|i])%=mod;
	int ans=0;
	for (int i=0;i<(1<<20);i++)
		if (g[i])
			inc(ans,mod-pow[f[i]]);
		else
			inc(ans,pow[f[i]]);
	printf("%d\n",ans);
	return 0;
}
