#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
typedef long long ll;
const ll mod=(ll)1e9+7;
int cnt[1<<20],x,n;
ll power[1<<20],ans=0;
int main(){
    scanf("%d",&n);
    power[0]=0;
    for (int i=1;i<(1<<20);i++) power[i]=power[i-1]*2+1,power[i]%=mod;
    for (int i=0;i<n;i++) scanf("%d",&x),cnt[x]++;
    for (int i=0;i<20;i++){
        for (int j=0;j<(1<<20);j++)
            if ((j>>i)&1) cnt[j-(1<<i)]+=cnt[j];
    }
    for (int i=0;i<(1<<20);i++){
        int tot=0;for (int j=0;j<20;j++) if ((i>>j)&1) tot++;
        if (tot&1)ans-=power[cnt[i]];
        else ans+=power[cnt[i]];
        if (ans<0||ans>=mod) ans=(ans+mod)%mod;
    }
    cout<<ans<<endl;
}
