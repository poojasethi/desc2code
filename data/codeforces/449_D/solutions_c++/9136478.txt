#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
int p[2000005],A,n,i,j,Q,X;
long long ans,bit[1000005];
const int MOD=1000000007;
int main()
{
    scanf("%d",&n);
    for (i=1; i<=n; i++)
    {
        scanf("%d",&A);
        p[A]++;
    }
    for (j=0; j<20; j++)
      for (i=1000000; i>=0; i--)
        if (!(i&(1<<j))) p[i]+=p[i+(1<<j)];
    bit[0]=1; for (i=1; i<=n; i++) bit[i]=bit[i-1]*2%MOD;
    ans=bit[n]-1;
    for (i=1; i<=1000000; i++)
    {
        Q=i; X=0; while (Q) {X^=(Q%2); Q/=2;}
        if (X==0) X=-1;
        ans=(ans-(bit[p[i]]-1+MOD)%MOD*X+MOD)%MOD;
    }
    cout<<ans;
    return 0;

}
