#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn=(1<<21)+5;
const int mod=1e9+7;
int dp[22][maxn];
int ted[maxn],n;
ll t2[maxn],ans;
int main(){
	t2[0]=1;
	for(int i=1 ; i<(1<<21) ; i++)
		t2[i]=(t2[i-1]*2)%mod;
	cin>>n;
	for(int i=1 ; i<=n ; i++){
		int a;cin>>a;
		ted[a]++;
	}
	for(int i=0 ; i<(1<<21) ; i++)
		dp[0][i]=ted[i];
	for(ll k=1 ; k<=20 ; k++)
		for(ll i=0 ; i<(1<<21) ; i++){
			if((i>>(k-1))&1)
				dp[k][i]=dp[k-1][i];
			else
				dp[k][i]=dp[k-1][i]+dp[k-1][i+(1<<(k-1))];
		}
	for(int i=0 ; i<(1<<21) ; i++){
		if((__builtin_popcount(i))%2)
			ans=(ans-t2[dp[20][i]])%mod;
		else
			ans=(ans+t2[dp[20][i]])%mod;
	}
	ans=(ans+mod)%mod;
	cout<<ans<<endl;
	return 0;
}
