#include<iostream>
using namespace std;
const int MOD=1e9+7;
int f[(1<<20)],pow2[1000100];
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie();
	int n;
	cin>>n;
	pow2[0]=1;
	for(int i=1;i<1000100;i++)
        pow2[i]=(pow2[i-1]*2)%MOD;
	for(int i=0;i<n;i++)
	{
		int val;
		cin>>val;
		f[val]++;
	}
	for(int p=0;p<20;p++)
		for(int mask=0;mask<(1<<20);mask++)
			if (!((mask>>p)&1))
				f[mask]+=f[mask^(1<<p)];
	int ans=0;
	for(int mask=0;mask<(1<<20);mask++)
	{
		if (__builtin_popcount(mask)&1)
			ans-=pow2[f[mask]];
		else
			ans+=pow2[f[mask]];
		ans%=MOD;
        ans+=MOD;
        ans%=MOD;
	}
	cout<<ans<<endl;
	return 0;
}
