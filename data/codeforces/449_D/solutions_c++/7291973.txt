#include<iostream>
using namespace std;
long long n,m,x,ans,i,j,md,f[1000002],d[1000002];
int main()
{
    cin>>n;
    md=1000000007;
    m=1000000;
    f[0]=1;
    for (i=1;i<=m;i++)
        f[i]=(f[i-1]*2)%md;
    for (i=1;i<=n;i++)
    {
        cin>>x;
        d[x]++;
    }
    for (i=20;i>=0;i--)
        for(j=m;j>=0;j--)
         if(j & (1<<(i))) d[j ^ (1<<i)]+=d[j];
    for(int j=0;j<=m;j++)
        if(d[j])
    {
        if(__builtin_popcount(j)%2)
             ans=(ans-f[d[j]]+1+md)%md;
        else ans=(ans+f[d[j]]-1+md)%md;
    }
    cout<<(ans+md)%md;
}
