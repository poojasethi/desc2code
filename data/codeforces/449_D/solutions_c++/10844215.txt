#include<cstdio>
#include<iostream>
#include<algorithm>
#define N 1000010
#define mo 1000000007
using namespace std;
int n, f[N + N], s[N], two[N], a[N], ans, i, j;
int main()
{
    scanf("%d", &n), two[0] = 1;
    for (i = 1; i <= 1000000; i++) two[i] = (two[i - 1] << 1) % mo;
    for (i = 1; i <= n; i++) scanf("%d", &a[i]), f[a[i]]++;
    for (j = 0; j < 20; j++)
        for (i = 0; i <= 1000000; i++)
            if (!(i & (1 << j))) f[i] += f[i ^ (1 << j)]; else s[i]++;
    for (i = 0; i <= 1000000; i++)
        if (!(s[i] & 1)) (ans += two[f[i]]) %= mo; else (ans -= two[f[i]]) %= mo;
    cout << (ans + mo + 1) % mo;
    return 0;
}
