#include<bits/stdc++.h>
using namespace std;
#define M 1000000007
#define MX 1000002
#define si(x) scanf("%d", &x)
int f[MX],p[MX];
int main() {
	int n, tmp; si(n); f[0] = 0, p[0] = 1;
	for(int i=1;i<MX;i++) f[i] = 0, p[i] = (p[i-1]*2)%M;
	for(int i=0;i<n;i++)
		si(tmp),f[tmp]++,p[i]--;
	p[n]--;
	for(int j=0;j<=20;j++)for(int i=MX-1;i>=0;i--)if(i&(1<<j)) f[i^(1<<j)]+=f[i];
	long long ans = 0;
	for(int i=0;i<MX;i++) 
		if(__builtin_popcount(i)%2)
	        ans = (ans-p[f[i]]+M)%M;
		else
		    ans = (ans+p[f[i]])%M;
	cout<<ans<<endl;
    return 0;
}
