#include <cstdio>
int a[1000100];
const int mod = 1000000007;
int b[1 << 20];
int p[1000100];
int main() {
	int n;
	scanf("%d", &n);
	int C = (1 << 20) - 1;
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
		b[a[i]]++;
	}
	for (int i = 0; i < 20; i++) {
		for (int j = 0; j < 1 << 20; j++) {
			if ((j >> i & 1) == 0) {
				b[j] += b[j | (1 << i)];
			}
		}
	}
	int ans = 0;
	p[0] = 1;
	for (int i = 0; i < n; i++) {
		p[i+1] = (p[i] << 1) % mod;
	}
	for (int i = 0; i < 1 << 20; i++) {
		int t = p[b[i]];
		for (int j = 0; j < 20; j++) {
			if (i >> j & 1) t = -t;
		}
		if (t < 0) {
			t += mod;
		}
		ans += t;
		if (ans >= mod) {
			ans -= mod;
		}
	}
	printf("%d\n", ans);
	return 0;
}
