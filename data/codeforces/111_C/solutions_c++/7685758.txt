#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 45
#define rep(i,n) for(int i=0;i<n;i++)
#define REP(i,n) for(int i=1;i<=n;i++)
#define Rep(i,n) for(int i=0;i<=n;i++)
#define lowbit(x) x&-x
#define inf 100000
using namespace std;
int n,m,lim,ans;
int f[N][1<<6][1<<6],get[1<<6];
int calc(int x){
	int r=0;
	while (x) r++,x-=lowbit(x);
	return m-r;
}
void init(){
	scanf("%d%d",&n,&m);
	if(n<m) swap(n,m);
	lim=1<<m;
	Rep(i,n) Rep(j,lim) Rep(k,lim) f[i][j][k]=-inf;
	rep(i,lim) f[0][0][i]=0;
	rep(i,lim) get[i]=calc(i);
}
bool check(int a,int b,int c){
	return ((b|(b<<1)|(b>>1)|a|c)&(lim-1))==(lim-1);
}
int main(){
	init();
	rep(i,n)
	rep(j,lim) rep(k,lim)
	if(f[i][j][k]!=-inf)
	rep(l,lim)
	if(check(j,k,l)) f[i+1][k][l]=max(f[i+1][k][l],f[i][j][k]+get[k]);
	rep(i,lim) ans=max(ans,f[n][i][0]);
	printf("%d\n",ans);
}
