#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N (1<<6)
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,nn,x,u,an;
int a[50][N][N],b[N];
int main() {
	scanf("%d%d",&n,&m);
	if (n<m) swap(n,m);
	memset(a,130,sizeof(a));
	a[0][0][0]=0; nn=(1<<m)-1;
	For(i,0,nn) {
		b[i]=m;
		For(j,0,m-1) b[i]-=i>>j&1;
	}
	For(i,0,n)For(j,0,nn)For(k,0,nn) if ((x=a[i][j][k])>=0) {
		int s=0;
		if (i) {
			For(u,0,m-1) {
				int A=(k>>u&1)|(j>>u&1);
				if (u) A|=k>>u-1&1;
				if (u+1<m) A|=k>>u+1&1;
				if (!A) s|=1<<u;
			}
			if (i==n&&!s) an=max(an,x);
		}
		if (i==n) continue;
		For(u,0,nn) if ((u&s)==s) a[i+1][k][u]=max(a[i+1][k][u],x+b[u]);
	}
	printf("%d\n",an);
	return 0;
}
