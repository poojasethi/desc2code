#include <stdio.h>
#include <iostream>
using namespace std;
int n,m,s=0;
void make(int a,int b){
	int n=a,m=b;
	if (n>m) swap(n,m);
	if (n<=6){
		if (n==1) s+=m*n-(m+2)/(4-n);
		if (n==2) s+=m*n-(m+2)/(4-n);
		if (n==3) s+=m*n-m/4*3-(m%=4)-(m==0);
		if (n==4) s+=m*n-m-(m==5 || m==6 || m==9);
		if (n==5) s+=m*n-m/5*6+(m==7)-(m%=5)-1-(m>1);
		if (n==6) s+=m*n-10;
		return ;
	}
	if (n%2==1){
		s+=m/3*2;
		n--;
	}
	if (m%3==1){
		if (n>2) s+=n-2;
		m=m/3*3;
	}
	if (m%3==2){
		make(2,n);
		m=m/3*3;
	}
	s+=n*m/3*2;
}
int main( ){
	cin>>n>>m;
	make(n,m);
	cout<<s<<endl;
	return 0;
}
