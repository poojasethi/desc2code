#include<iostream>
#include<algorithm>
#define INF 1000009
using namespace std;

int n,m,tot,ls,plusx[100],pow[]={1,2,4,8,16,32,64};
int dp[41][100][100];
bool check(int i,int j,int k){
	return ((i|j|k|(j*2)|(j/2))&(ls-1))==(ls-1);
}

int main()
{
	cin>>n>>m;
	if(m>n) swap(m,n);
	ls=pow[m];
	for(int i=0;i<=40;i++)
	   for(int j=0;j<ls;j++)
	      for(int k=0;k<ls;k++)
	         dp[i][j][k]=-INF;
	for(int i=0;i<ls;i++){
		plusx[i]=m;
		for(int j=0;j<m;j++)
			if(i&pow[j]) plusx[i]--;
		dp[0][0][i]=0;
	}
	for(int i=0;i<n;i++)
	   for(int j=0;j<ls;j++)
	      for(int k=0;k<ls;k++)
	         for(int p=0;p<ls;p++)
	         	if(check(j,k,p)) dp[i+1][k][p]=max(dp[i+1][k][p],dp[i][j][k]+plusx[k]);
	for(int i=0;i<ls;i++)
	    tot=max(tot,dp[n][i][0]);
	cout<<tot<<endl;
	return 0;
}
