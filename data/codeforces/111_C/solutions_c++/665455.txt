#include<algorithm>
#define S(x)	switch(x)
#define C(x)	case x:
#define A(x)	return printf("%d\n",m*n-(x)),0;
int main(){
    int n,m;
	scanf("%d%d",&n,&m);
	if(n>m)std::swap(n,m);
	S(n){
		C(1)A((m+2)/3)
		C(2)A(m/2+1)
		C(3)S(m%4){
			C(0)C(1)A(m/4*3+1)
			C(3)A((m+1)/4*3)
			C(2)A(m/4*3+2)
		}
		C(4)S(m){
			C(4)A(4)
			C(5)A(6)
			C(6)C(7)A(7)
			C(8)A(8)
			C(9)C(10)A(10)
		}
		C(5)A(m+1+m/4)

	}
}
