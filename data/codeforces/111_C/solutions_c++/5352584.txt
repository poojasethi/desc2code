#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int n,m,s,bit[70],dp[45][70][70],ret;
int main()
{
	scanf("%d%d",&n,&m);
	if (m>n) swap(n,m);
	s=(1<<m)-1;
	for (int i=0; i<=s; i++)
	{
		for (int j=i; j; j-=j&-j) bit[i]++;
	}
	memset(dp,0x3f,sizeof(dp));
	ret=(1<<30);
	dp[0][0][0]=0;
	for (int i=0; i<n; i++)
	{
		for (int j=0; j<=s; j++)
		{
			for (int k=0; k<=s; k++)
			{
				for (int l=0; l<=s; l++)
				{
					if (i && ((j|k|(k>>1)|(k<<1)|l)&s)!=s) continue;
					dp[i+1][k][l]=min(dp[i+1][k][l],dp[i][j][k]+bit[l]);
					if (i==n-1 && ((k|l|(l<<1)|(l>>1))&s)==s) ret=min(ret,dp[i+1][k][l]);
				}
			}
		}
	}
	printf("%d\n", n*m-ret);
	return 0;
}
