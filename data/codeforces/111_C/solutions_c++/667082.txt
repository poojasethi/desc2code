#include <cstdio>
int main() {
	int n, m, tmp, ret;
	
	scanf("%d%d", &n, &m);
	if (n > m) { tmp = n; n = m; m = tmp; }
	if (n == 1) ret = (m - 1) / 3 + 1;
	else if (n == 2) ret = m / 2 + 1;
	else if (n == 3) ret = m - (m - 1) / 4;
	else if (n == 4) ret = m + ((m + 1) & 2) / 2 - m / 10;
	else if (n == 5) ret = m + (m + 1) / 3;
	else ret = 10;
	printf("%d\n", n * m - ret);
	return 0;
}
