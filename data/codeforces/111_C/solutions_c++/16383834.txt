#include <iostream>

using namespace std;

int main()
{
	int n, m;
	cin >> n >> m;
	if (n > m)
		swap(n, m);
	if (n == 1)
		cout << m - (m + 2) / 3 << endl;
	else if (n == 2)
		cout << 2 * m - (m + 2) / 2 << endl;
	else if (n == 3)
	{
		int ans = 3 * m - m / 4 * 3 - (m % 4);
		if (m % 4 == 0)
			ans--;
		cout << ans << endl;
	}
	else if (n == 4)
	{
		int ans = 4 * m - m;
		if (m == 5 || m == 6 || m == 9)
			ans--;
		cout << ans << endl;
	}
	else if (n == 5)
	{
		int ans = 5 * m - m / 5 * 6 - (m % 5) - 1;
		if (m == 7)
			ans++;
		if (m % 5 > 1)
			ans--;
		cout << ans << endl;
	}
	else if (n == 6)
		cout << 6 * m - 10 << endl;
	return 0;
}
