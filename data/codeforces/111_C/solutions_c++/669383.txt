#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main(){
    int m,n;
    cin>>m>>n;
    if(m>n) swap(m,n);
    if(m==1){if(n==1)cout<<0<<endl;else cout<<(2*n)/3<<endl;}
    else if(m==2)cout<<m*n-(n+2)/2<<endl;
    else if(m==3) cout<<2*n+(n-1)/4<<endl;
    else if(m==4)cout<<n*3-1+(n==4 || n==7 || n==8 || n==10)<<endl;
    else if(m==5) cout<<4*n-1-(n/4)<<endl;
    else    cout<<n*m-10<<endl;
    return 0;
}
