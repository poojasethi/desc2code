#include<iostream>
    #include<algorithm>

    using namespace std;

    int main(){

        int m, n, tot;
       
        cin>>m>>n;
        tot = m*n;
        if(m>n)    swap(m, n);
        if(m == 1)
                cout<<tot-(n/3 + (bool)(n%3));
        if(m == 2)
            cout<<tot-(n/2 + 1);
        if(m == 3)
            cout<<tot-(n/4*3+(n %=4 )+(n == 0));
        if(m == 4)
            cout<<tot-(n+(bool)((n == 5)||(n == 6)||(n == 9)));
        if(m == 5)
            cout<<(n*4-2-(n == 8));
        if(m == 6)
            cout<<tot-10;

        return 0;
       
    }
