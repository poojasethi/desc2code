#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int i,j,ans[500001],arr[500001],n,le,ri,maxi=0,pos=0;
int main()
{
	//ios_base::sync_with_stdio(false);cin.tie(0);
	cin>>n;
	for(i=0;i<n;i++)
	{
		scanf("%d",&arr[i]);
		//left[i]=0;
		//right[i]=0;
	}
	i=0;
	while(i<n)
	{
		le=i;
		ri=i;
		while(ri<=n-2 && arr[ri+1]%arr[i]==0)
			ri++;
		while(le>0 && arr[le-1]%arr[i]==0)
			le--;
		
		i=ri+1;
		int temp=ri-le;
		if(temp>maxi)
		{
			maxi=temp;
			pos=0;
			ans[pos++]=le+1;
		}
		else if(temp==maxi)
		{
			ans[pos++]=le+1;
		}
	}
	//cout<<pos<<" "<<maxi<<"\n";
	printf("%d %d\n",pos,maxi);
	for(i=0;i<pos;i++)
		printf("%d ",ans[i]);
}
