#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
using namespace std;
#define N 600000
int a[N],n,L[N],ans,can[N],R[N];
int main(){
	scanf("%d",&n);ans=-1;
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]),L[i]=R[i]=i;
	for(int i=2;i<=n;i++)
		while(L[i]>1&&a[L[i]-1]%a[i]==0)L[i]=L[L[i]-1];
	for(int i=n-1;i;i--)
		while(R[i]<n&&a[R[i]+1]%a[i]==0)R[i]=R[R[i]+1];
	for(int i=1;i<=n;i++)ans=max(ans,R[i]-L[i]);int tmp=0;
	for(int i=1;i<=n;i++)if(R[i]-L[i]==ans)can[L[i]]=1;
	for(int i=1;i<=n;i++)tmp+=can[i];
	printf("%d %d\n",tmp,ans);
	bool flag=1;
	for(int i=1;i<=n;i++)
		if(can[i])
		{
			if(!flag)putchar(' ');
			else flag=0;
			printf("%d",i);
		}
	puts("");
}
