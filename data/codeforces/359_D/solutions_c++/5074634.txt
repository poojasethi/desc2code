#include "stdio.h"
#include "iostream"
#include "vector"
using namespace std;
const int MAX = 300005;
int data[MAX];
int main()
{
    int n;
    cin>>n;
    for(int i=0;i<n;i++)cin>>data[i];
    vector<int> v;
    int max = 0;
    for(int i=0;i<n;i){
        int l = i, r = i;
        while(l>=0&&data[l]%data[i]==0)l--;
        while(r<n&&data[r]%data[i]==0)r++;
        int t = r-l-1;
        i = r;
        if(t>max){
            v.clear();
            v.push_back(l+2);
            max = t;
        }else if(t == max){
            v.push_back(l+2);
        }
    }
    cout<<v.size()<<" "<<max-1<<endl;
    for(int i=0;i<v.size();i++){
        cout<<v[i]<<" ";
    }
}
