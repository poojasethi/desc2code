#include <stdio.h>
#include <algorithm>

const int N = 300300;

int a[N], L[N], R[N];
int b[N];
int main() {
	int n, i;
	scanf("%d", &n);
	for(i=1;i<=n;++i) scanf("%d", &a[i]);
	L[1] = 1;
	for(i=2;i<=n;++i) {
		int p = i-1;
		while( p && a[p] % a[i] == 0 )
			p = L[p] - 1;
		L[i] = p+1;
	}
	R[n] = n;
	for(i=n-1;i>=1;--i) {
		int p = i+1;
		while( p!=n+1 && a[p] % a[i] == 0 )
			p = R[p] + 1;
		R[i] = p-1;
	}
	int _max = 0, cnt = 0;
	for(i=1;i<=n;++i) if(R[i]-L[i]>_max) _max=R[i]-L[i];
	for(i=1;i<=n;++i) if(R[i]-L[i]==_max) b[cnt ++] = L[i];
	std::sort(b, b+cnt);
	cnt = std::unique(b, b+cnt) - b;
	printf("%d %d\n", cnt, _max);
	for(i=0;i<cnt;++i) printf("%d ", b[i]);
	printf("\n");
	return 0;
}
