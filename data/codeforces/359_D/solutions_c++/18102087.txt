#include <bits/stdc++.h>

using namespace std;

int a[300010],ans[300010];
int main()
{
    int n,l,r,i,j,b=0,c=0;

    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>a[i];
    }

    for(i=0;i<n;)
    {
        l=r=i;
        while(l && a[l-1]%a[i]==0) l--;
        while(r<n-1 && a[r+1]%a[i]==0) r++;

        i=r+1; r-=l;

        if(r>b) c=0,b=r;
        if(r==b) ++c,ans[c]=l+1;
    }

    cout<<c<<" "<<b<<endl;

    for(i=1;i<=c;i++)
        cout<<ans[i]<<endl;


}
