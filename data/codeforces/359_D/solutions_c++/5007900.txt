#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int a[333333],x[333333];
int n;
int main()
{
    int ans=0;
    int ansnum=0;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)scanf("%d",&a[i]);
    for(int i=1;i<=n;)
    {
        int l,r;
        l=r=i;
        while(l>1&&a[l-1]%a[i]==0)l--;
        while(r<n&&a[r+1]%a[i]==0)r++;
        if(r-l>ans)
        {
            ans=r-l;
            ansnum=0;
        }
        if(r-l==ans)x[++ansnum]=l;
        i=r+1;
    }
    cout<<ansnum<<" "<<ans<<endl;
    cout<<x[1];
    for(int i=2;i<=ansnum;i++)cout<<' '<<x[i];
    cout<<endl;
    return 0;
}
