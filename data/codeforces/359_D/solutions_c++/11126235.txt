#include<iostream>
#include<cstring>
#include<cstdio>
using namespace std;
#define M 300050
int pri[M],ans[M];
int main()
{
    int n,l,r;
    while(scanf("%d",&n)!=EOF){
        for(int i=1;i<=n;i++){
            scanf("%d",&pri[i]);
        }
        int cnt=0,tmp=0;
        for(int i=1;i<=n;){
            l=r=i;
            while(l&&pri[l]%pri[i]==0)l--;
            while(r<=n&&pri[r]%pri[i]==0)r++;
            i=r;r=r-l-2;
            if(r>tmp)cnt=0,tmp=r;
            if(r==tmp)ans[cnt++]=l+1;
        }
        printf("%d %d\n",cnt,tmp);
        for(int i=0;i<cnt;i++)
        printf("%d ",ans[i]);
        printf("\n");
    }
    return 0;
}
