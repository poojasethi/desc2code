#include<stdio.h>
#define N 300010
int a[N],ans[N];
int main()
{
    int n,i,j;
    while(scanf("%d",&n)!=EOF)
    {
        for(i=1;i<=n;i++)scanf("%d",&a[i]);
        int t=0,l,r,len=0,cnt=0;
        for(i=1;i<=n;)
        {
            l=r=i;
            while(l&&a[l]%a[i]==0)l--;
            while(r<=n&&a[r]%a[i]==0)r++;
            i=r;
            len=r-l-2;
            if(len>t){t=len;cnt=0;}
            if(len==t)ans[cnt++]=l+1;
        }
        printf("%d %d\n",cnt,t);
        for(i=0;i<cnt;i++)printf("%d ",ans[i]);
        printf("\n");
    }
}
