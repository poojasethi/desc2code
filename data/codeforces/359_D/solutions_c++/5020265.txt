#include<stdio.h>
#include<vector>
#include<iostream>
#define M 500005
using namespace std;
int A[M];
vector<int>ans;
int main(){
	int n,i,j,k;
	while(scanf("%d",&n)!=EOF){
		for(i=1;i<=n;i++)
			scanf("%d",&A[i]);
		ans.clear();
		int res=0;
		k=1;
		while(k<=n){
			int l=k,r=k;
			while(l>0&&A[l]%A[k]==0)l--;
			while(r<=n&&A[r]%A[k]==0)r++;
			if(r-l-2>res){
				res=r-l-2;
				ans.clear();ans.push_back(l+1);
			}else if(r-l-2==res)
				ans.push_back(l+1);
			k=r;
		}
		printf("%d %d\n",ans.size(),res);
		for(i=0;i<ans.size();i++)
			printf("%d ",ans[i]);
		puts("");
	}
	return 0;
}
