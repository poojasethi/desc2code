#include <cstdio>
#include <cstring>
using namespace std;
int arr[500005];
int ans[500005];
int main(){
    int n;
	scanf("%d",&n);
	int maxlen=0;
	int num=0;
    for(int i=0;i<n;i++) scanf("%d",&arr[i]);
	for(int i=0;i<n;){
		int count=0;
		int countleft=0;
		for(int j=i-1;j>=0;j--){
			if(arr[j]%arr[i]==0) countleft++;
			else break;
		}
		for(int j=i;j<n;j++){
			if(arr[j]%arr[i]==0) count++;
			else break;
		}
		if(countleft+count>maxlen){
			maxlen=count+countleft;
			num=0;
			ans[num++]=i-countleft;
		}else if(count+countleft==maxlen)
			ans[num++]=i-countleft;
		i+=count;
	}

	printf("%d %d\n",num,maxlen-1);
	for(int i=0;i<num;i++)
		printf("%d ",ans[i]+1);

    return 0;
}
