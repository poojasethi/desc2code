#include <cstdio>
using namespace std;

const int MAXN = 300000;
int a[MAXN+1];
int numn[MAXN+1];
int n;

int main()
{
	int l,r;
	int cnt = 0;
	int b = 0;
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	int i = 0;
	while( i < n )
	{
		l = r = i;
		while(l&&a[l-1]%a[i]==0) l--;
		while(r<n-1&&a[r+1]%a[i]==0) r++;
		i = r+1; r-=l;
		if(r>b)
		{
			cnt = 0;
			b = r;
		}
		if(r==b)
			numn[cnt++] = l;
	}
	printf("%d %d\n",cnt,b);
	for(int i=0;i<cnt;i++)
		printf(i==cnt-1?"%d\n":"%d ",numn[i]+1);
	return 0;
}
