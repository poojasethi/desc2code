#include <cstdio>

const int maxn = 300010;
int n, l, r, cur=0, res=-1, a[maxn], ans[maxn];

int main() {
	scanf("%d", &n);
	for(int i=0; i<n; i++) scanf("%d", &a[i]);
	for(int i=0; i<n; i=r+1) {
		l=r=i;
		while(l && a[l-1]%a[i] == 0) l--;
		while(r+1<n && a[r+1]%a[i] == 0) r++;
		if(r-l>res) {
			res=r-l;
			cur=0;
		}
		if(r-l == res) ans[cur++]=l;
	}
	printf("%d %d\n", cur, res);
	for(int i=0; i<cur; i++) {
		if(i) putchar(' ');
		printf("%d", ans[i]+1);
	}
	puts("");
	return 0;
}
