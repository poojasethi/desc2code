#include <iostream>
#include <queue>
#include <vector>

using namespace std;

bool cmd(pair<int, int> a, pair<int, int> b)
{
	return a.first<b.first;
}

int main()
{
	int n, a[500005];
	cin>>n;
	for(int i=0; i<n; i++)
	{
		cin>>a[i];
	}
	int maxlen=0;
	vector<int> v;
	for(int i=0; i<n; i++)
	{
		if(i>0 && a[i]==a[i-1]) continue;
		int len=0, t=i;
		while(t<n && a[t]%a[i]==0) t++, len++;
		t=i-1;
		while(t>=0 && a[t]%a[i]==0) 
		{
			t--, len++;
			if(a[t]==a[i]) break;
		}
		t++;
		if(len>maxlen)
		{
			maxlen=len;
			v.clear(); v.push_back(t);
		}
		else if(len==maxlen)
		{
			v.push_back(t);
		}
	}
	cout<<v.size()<<" "<<maxlen-1<<endl;
	for(int i=0; i<v.size(); i++) cout<<v[i]+1<<" "; cout<<endl;
}
