#include<iostream>
#include<vector>
using namespace std;
int a[3*1000*100+15];
int main(){
    int n,maxi=-1;
    cin>>n;
    vector<int> ans;
    for(int i=0;i<n;i++)cin>>a[i];
    int l,r=0,i=0;
    while(i<n){
        l=i;
        while(a[r]%a[i]==0 && r<n)r++;
        while(a[l]%a[i]==0 && l>=0)l--;
        if(r-l-2>maxi){
            maxi=r-l-2;
            ans.clear();
            ans.push_back(l+1);
        }
        else if(r-l-2==maxi){
            ans.push_back(l+1);
        }
        i=r;
    }
    cout<<ans.size()<<" "<<maxi<<endl;
    for(int i=0;i<ans.size();i++)cout<<ans[i]+1<<" ";
}
