#include <iostream>
#include <vector>

using namespace std;

int n, num[300000], ans[300000];

int main() {
	cin >> n;

	for (int i = 0; i < n; ++i) {
		cin >> num[i];
	}

	int max = -1;
	int amt = 0;
	int idx = 0;

	while (idx < n) {
		int l = idx;
		int r = idx;

		while (l - 1 >= 0 && num[l - 1] % num[idx] == 0) --l;
		while (r + 1 < n && num[r + 1] % num[idx] == 0) ++r;

		if (r - l > max) {
			amt = 0;
			ans[amt++] = l;
			max = r - l;
		} else if (r - l == max) {
			ans[amt++] = l;
		}

		idx = r + 1;
	}

	cout << amt << " " << max << endl;
	cout << ans[0] + 1;
	for (int i = 1; i < amt; ++i) {
		cout << " " << ans[i] + 1;
	}
	cout << endl;

	return 0;
}
