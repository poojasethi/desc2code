#include<iostream>
using namespace std;
int a[300010],ans[300010],n,ma=-1,c,i;
int main()
{
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    for(i=0;i<n;)
    {
        int l=i,r=i;
        while(l&& a[l-1]%a[i]==0)l--;
        while(r<n-1 && a[r+1]%a[i]==0)r++;
        //cout<<l<<r<<endl;
        i=r+1;
        if(r-l>ma)ma=r-l,c=0,ans[c]=l+1;
        else if(r-l==ma)ans[++c]=l+1;
    }
    cout<<c+1<<" "<<ma<<endl;
    for(i=0;i<=c;i++)cout<<ans[i]<<" ";
}
