#include <cstdio>

int a[1234567], ans[1234567], alen;
int main() {
    int n, l, r, bd=-1;

    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", a+i);
    for (int i = 0; i < n;) {
        for (l = i; l > 0 && a[l-1]%a[i] == 0; l--);
        for (r = i; r < n-1 && a[r+1]%a[i] == 0; r++);
        if (r-l > bd) {
            ans[0] = l; alen = 1; bd = r-l;
        }
        else if (r-l == bd) {
            ans[alen++] = l;
        }
        i = r+1;
    }
    printf("%d %d\n", alen, bd);
    for (int i = 0; i < alen; i++) printf("%d%c", ans[i]+1, i==alen-1 ? '\n' : ' ');

    return 0;
}
