#include<bits/stdc++.h>
using namespace std;
int dat[333333],n;
int maxs=0,cnt=0;
int L[333333],r_l[333333];
int main()
{
	cin>>n;
	for(int i=1;i<=n;++i)
		cin>>dat[i];
	for(int i=1;i<=n;++i)
	{
		int ii=i;
		L[i]=i;
		for(int j=i-1;j>=0;--j)
			if(dat[j]%dat[i]||!j)
			{
				L[i]=j+1;
				break;
			}
		for(int j=i+1;j<=n+1;++j)
			if(dat[j]%dat[i]||j==n+1)
			{
				r_l[i]=j-1-L[i];
				i=j-1;
				break;
			}		
		if(r_l[ii]>maxs)
			cnt=1,maxs=r_l[ii];
		else if(r_l[ii]==maxs)
			cnt++;	
	}	
	cout<<cnt<<' '<<maxs<<endl;
	for(int i=1;i<=n;++i)
		if(r_l[i]==maxs)
			cout<<L[i]<<' ';
}
