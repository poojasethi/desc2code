#include <iostream>
#include <vector>
using namespace std;
const int MAX = 300005;
int a[MAX];
vector<int> answer;
int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	int ans = 0;
	for (int i = 0; i < n ;)
	{
		int r = i, l = i;
		while (l && a[l - 1] % a[i] == 0)
			l--;
		while (r < n - 1 && a[r + 1] % a[i] == 0)
			r++;
		i = r + 1;
		r -= l;
		if (ans < r)
		{
			ans = r;
			answer.clear();
		}
		if (ans == r)
			answer.push_back(l + 1);
	}
	cout << answer.size() << " " << ans << endl;
	for (int i = 0; i < answer.size(); i++)
		cout << answer[i] << " ";
	return 0;
}
