#include<cstdio>
using namespace std;

int a[300008];
int w[300008];

int main()
{
    int n;
    int c,l,r,b;
    while (1==scanf("%d",&n))
    {
        for (int i=0;i<n;i++)
        {
            scanf("%d",&a[i]);
        }
        c=b=0;
        for (int i=0;i<n;){
            r=l=i;
            while (l>0 && a[l-1]%a[i]==0) l--;
            while (r<n-1 && a[r+1]%a[i]==0) r++;
            i=r+1;
            r-=l;
            if (r>b) c=0,b=r;
            if (r==b) w[c++]=l+1;
        }
        printf("%d %d\n",c,b);
        for(int i=0;i<c;i++) printf("%d ",w[i]);
        printf("\n");
    }
    return 0;
}
