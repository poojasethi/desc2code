#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int a[300008], w[300008];
int mx, Count, n, r, l;

int main ( )
{
	int i ;
	mx = 0;
	for( scanf( "%d", &n ), i = 0; i < n; ++i ){
		scanf( "%d", a + i );
	}
	for( i = 0; i < n;  ){
		l = r = i;
		while( l && a[l - 1] % a[i] == 0)
			--l;
		while( r < n - 1 && a[r + 1] % a[i] == 0 )
			++r;
		i = r + 1;
		r -= l;
		if( r  > mx){
			Count = 0;
			mx = r ;
		}
		if( r  == mx )
			w[Count++] = l + 1;
	}
	printf( "%d %d\n", Count, mx );
	for( i = 0; i < Count; ++i ){
		printf( "%d%c", w[i], i + 1 == Count? '\n' : ' ' );
	}
	return 0;
}
