#include<stdio.h>
#include<vector>
using namespace std;
vector<int>v;
#define maxn 300005
int a[maxn];
int main() {
	int l, r, n, ans = 0;
	scanf("%d", &n);
	for(int i = 1; i <= n; i++) scanf("%d", &a[i]);
	for(int i = 1; i <= n; i = r + 1) {
		if(a[i] == 1) {
			printf("%d %d\n1\n", 1, n - 1);
			return 0;
		}
		l = r = i;
		while(l > 1 && a[l - 1] % a[i] == 0) l--;
		while(r < n && a[r + 1] % a[i] == 0) r++;
		if(r - l > ans) {
			v.clear();
			ans = r - l;
		}
		if(r - l == ans) v.push_back(l);
	}
	printf("%ld %d\n", v.size(), ans);
	for(int i = 0; i < v.size(); i++) printf("%d ", v[i]);
	printf("\n");
	return 0;
}
