#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;

const int MAXN=300005;
int a[MAXN];
int ans[MAXN];

int main()
{
    int n;
    scanf("%d",&n);
    for(int i=1;i<=n;i++) scanf("%d",&a[i]);
    int l,r;
    int len=0,cnt=0;
    for(int i=1;i<=n;)
    {
        l=r=i;
        while(a[l]%a[i]==0&&l>0) l--;
        while(a[r]%a[i]==0&&r<=n) r++;
        i=r;
        if(r-l-2>len){
            cnt=0; len=r-l-2;
        }
        if(r-l-2==len) ans[cnt++]=l+1;
    }
    printf("%d %d\n",cnt,len);
    for(int i=0;i<cnt;i++)
        printf("%d%c",ans[i],(i==cnt-1)?'\n':' ');
    return 0;
}
