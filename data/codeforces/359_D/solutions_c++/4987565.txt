#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

const int INF = 1000000007;

int a[300005];

int main(){
    int n,l,r,ans=0;
    scanf("%d",&n);
    a[0]=a[n+1]=INF;
    vector<int> u;
    for(int i=1;i<=n;i++) scanf("%d",a+i);
    for(int i=1;i<=n;i=r+1){
        if(a[i]==1) return printf("%d %d\n1\n",1,n-1),0;
        l=r=i;
        while(a[l-1]%a[i]==0) l--;
        while(a[r+1]%a[i]==0) r++;
        if(r-l> ans) u.clear(),ans=r-l;
        if(r-l==ans) u.push_back(l);
    }
    printf("%d %d\n",u.size(),ans);
    for(size_t i=0;i+1<u.size();i++) printf("%d ",u[i]);
    printf("%d\n",u.back());
}
