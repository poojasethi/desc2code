#include <bits/stdc++.h>

#define o ((b+e)/2)

using namespace std;

int N;
int ar[350000],l[350000],r[350000];
set <int> v[350000];

int main(){
	
	cin >> N;
	
	for( int i=1 ; i<=N ; i++ ) 	scanf(" %d",&ar[i]);
	
	for( int i=1 ; i<=N ; i++ ){
		int j;
		for( j=i-1 ; j>=1 && ar[j]%ar[i] == 0 ; j=l[j] );
		l[i]=j;
	}
	
	for( int i=N ; i>=1 ; i-- ){
		int j;
		for( j=i+1 ; j<=N && ar[j]%ar[i] == 0 ; j=r[j] );
		r[i]=j;
	}
	
	int res=0;
	for( int i=1 ; i<=N ; i++ ){
		v[r[i]-l[i]-2].insert( l[i]+1 );
		if( r[i]-l[i]-2 > res  ) 	res=r[i]-l[i]-2;
	}
	
	set<int>::iterator it;
	printf("%d %d\n",(int)v[res].size(),res);
	for( it=v[res].begin() ; it!=v[res].end() ; it++ ) 	printf("%d ",*it);
	
	return 0;
	
}
