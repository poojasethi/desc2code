#include <bits/stdc++.h>
using namespace std;
int ans[3000005];
int mxL;
int n,cnt;
int a[300005];

int main()
{
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=0;i<n;)
	{
		int r=i,l=i;
		while(r<n-1 && a[r+1]%a[i]==0)r++;
		while(l && a[l-1]%a[i]==0)l--;
		i = r+1;
		
		if(r-l > mxL)
		{
			mxL = r-l;
			cnt = 0;
			ans[cnt++] = l+1;
		}
		else if(r-l==mxL)
			ans[cnt++] = l+1;
			
	}
	cout<<cnt<<" "<<mxL<<"\n";
	for(int i=0;i<cnt;i++)printf("%d ",ans[i]);
	return 0;
}
