#include <bits/stdc++.h>

using namespace std;
bool is_prime(int i){//int flag=1;
    for(int j=2;j<=sqrt(i);++j){
        if(i%j==0){
            return false;
        }
    }
    return true;
}
//bool A[100000];
int dp[5005][13000];
int ans[5005];
int par[5005];
int C[5005];
pair<int,int> A[5005];
int main(){ios::sync_with_stdio(false);
    int n;
    cin >> n;

    for(int i=1;i<=n;++i){
        cin >> A[i].first;
        A[i].second=i;
    }
    sort(A+1,A+n+1);

    par[1]=0;



    for(int i=1;i<n;++i){
        int a,b,c;
        cin >> a >> b >> c;



        if(par[max(a,b)]==0)
        par[max(a,b)]=min(a,b),C[max(a,b)]=c;
        else
            par[min(a,b)]=max(a,b),C[min(a,b)]=c;

    }
    for(int i=1;i<=n;++i){
        int t=0;
        int c=A[i].second;

        while(c!=1){
            int a=par[c];
            if(dp[c][t]<C[c]){

                dp[c][t]++;
                ++t;
                c=a;
            }
            else{
                ++t;

            }
        }
        ans[A[i].second]=t;
    }
    for(int i=1;i<=n;++i)
        cout << ans[i] << " ";
    cout << endl;
    return 0;



}
