#include <algorithm>
#include <iostream>

#include <cstdio>
#include <cstring>

using namespace std;

int n, prev[5100], gr[5100][5100], used[5100], ans[5100];
pair<int, int> divs[5100];
int busy[5100][5100];

void dfs(int v)
{
	used[v] = 1;
	for (int i = 0; i < n; i++)
		if (gr[v][i] && !used[i]) prev[i] = v, dfs(i);
}

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
#endif
	memset(used, 0, sizeof(used));
	memset(busy, 0, sizeof(busy));
	memset(gr, 0, sizeof(gr));
	cin >> n;
	int a, b, c;
	for (int i = 0; i < n; i++)	cin >> divs[i].first, divs[i].second = i;
	for (int i = 0; i < n - 1; i++)
		cin >> a >> b >> c, gr[--a][--b] = c, gr[b][a] = c;
	prev[0] = -1, dfs(0);
	sort(divs, divs + n);
	for (int i = 0; i < n; i++)
	{
		int id = divs[i].second;
		ans[id] = 0;
		for (int v = id; v != 0; v = prev[v])
		{
			while (busy[v][ans[id]] == gr[v][prev[v]]) ans[id]++;
			busy[v][ans[id]]++, ans[id]++;
		}
	}
	for (int i = 0; i < n; i++)
		cout << ans[i] << " ";
	return 0;
}
