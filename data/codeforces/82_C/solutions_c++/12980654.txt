#include<iostream>
#include<vector>
#include<algorithm>
#define pb push_back

using namespace std;

const int maxn=5050;
int n,c[maxn],d[maxn],ans[maxn],in[maxn][maxn];
pair<int,int> a[maxn];
bool vis[maxn];
vector<int> g[maxn],h[maxn];

void dfs(int v)
{
  vis[v]=true;
  for(int i=0;i<g[v].size();i++)
    if(!vis[g[v][i]])
      d[g[v][i]]=v,c[g[v][i]]=h[v][i],dfs(g[v][i]);
}

int gfs(int v,int t)
{
  if(v==1)
    return t;
  if(in[v][t]<c[v])
    {
      in[v][t]++;
      return gfs(d[v],t+1);
    }
  return gfs(v,t+1); 
}

int main()
{
  cin>>n;
  for(int i=0;i<n;i++)
    {
      cin>>a[i].first;
      a[i].second=i+1;
    }
  for(int i=1,v,u,c;i<n;i++)
    {
      cin>>v>>u>>c;
      g[v].pb(u);
      g[u].pb(v);
      h[v].pb(c);
      h[u].pb(c);
    }
  dfs(1);
  sort(a,a+n);
  for(int i=0;i<n;i++)
    ans[a[i].second]=gfs(a[i].second,0);
  for(int i=1;i<=n;i++)
    cout<<ans[i]<<endl;
  return 0;
}
