#include<bits/stdc++.h>
using namespace std;
#define LL long long
#define rep(i,n) for(int i=0;i<n;++i)
#define Rep(i,n) for(int i=1;i<=n;++i)

pair<int, int>p[5050];
int fa[5050], first[5050], nxt[10100], vv[10100], cap[5050], c[5050], pos[5050], temp[5050], ans[5050];

void add(int u, int v) {
	static int i = 0;
	vv[i] = v;
	nxt[i] = first[u];
	first[u] = i++;
}

void dfs(int u, int pa) {
	fa[u] = pa;
	for(int i = first[u]; ~i; i = nxt[i]) {
		int v = vv[i];
		if(v == pa) continue;
		cap[v] = c[i/2];
		dfs(v, u);
	}
}

int main()
{
	int n;
	cin >> n;
	Rep(i, n) cin >> p[i].first, p[i].second = i;
	sort(p+1, p+n+1);
	int u, v;
	memset(first, -1, sizeof(first));
	rep(i, n-1) {
		scanf("%d%d%d", &u, &v, c+i);
		add(u, v);
		add(v, u);
	}
	dfs(1, 1);
	memset(ans, -1, sizeof(ans));
	ans[1] = 0;
	Rep(i, n) pos[i] = i;
	Rep(t, n) {
		memcpy(temp, cap, sizeof(temp));
		Rep(k, n) {
			int i = p[k].second;
			if(ans[i] >= 0) continue;
			int& pi = pos[i];
			if(temp[pi]) --temp[pi], pi = fa[pi];
			if(pi == 1) ans[i] = t;
		}
	}
	cout << 0;
	for(int i = 2; i <= n; ++i) printf(" %d", ans[i]);
	puts("");
	return 0;
}
