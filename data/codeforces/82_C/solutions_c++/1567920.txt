#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
const int N = 5005;
#define mp(x,y) make_pair(x,y)
typedef pair<int,int> per;
vector<per>E[N];
int p[N],d[N],fa[N],pos[N],tmp[N],tc[N],r[N];
bool u[N];
void get(int x)
{
	u[x]=true;
	for(int i=E[x].size()-1;i>=0;i--)
	{
		int j=E[x][i].first;
		if(!u[j])
		{
			fa[j]=x;
			tc[j]=E[x][i].second;
			get(j);
		}
	}
}
bool cmp(int x,int y){return p[x]<p[y];}
int main()
{
	int n,i,a,b,c;
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",p+i);
	for(i=1;i<=n;i++)pos[i]=d[i]=i;
	sort(d+1,d+1+n,cmp);
	for(i=1;i<n;i++)
	{
		scanf("%d%d%d",&a,&b,&c);
		E[a].push_back(mp(b,c));
		E[b].push_back(mp(a,c));
	}
	get(1);
	int cnt=1;
	for(int day=0;cnt<n;day++)
	{
		for(i=1;i<=n;i++)tmp[i]=tc[i];
		for(i=1;i<=n;i++)
		{
			int j=d[i];
			if(pos[j]!=1&&tmp[pos[j]])
			{
				tmp[pos[j]]--;
				pos[j]=fa[pos[j]];
				if(pos[j]==1){r[j]=day+1;cnt++;}
			}
		}
	}
	for(i=1;i<n;i++)printf("%d ",r[i]);printf("%d\n",r[i]);
	return 0;
}
