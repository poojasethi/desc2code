#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

typedef pair<int,int> pnt;
typedef vector<pnt> vnt;

pnt arm[5005];
int tm[5005];
int par[5005];
int cap[5005];
int crm[5005];
vnt way[5005];
int n;

int read_input()
{
	int i;
	int a,b,c;
	scanf("%d",&n);
	for(i=0;i<n;i++) {
		scanf("%d",&a);
		arm[i] = pnt(a,i);
	}
	for(i=1;i<n;i++) {
		scanf("%d %d %d",&a,&b,&c); a--; b--;
		way[a].push_back(pnt(b,c));
		way[b].push_back(pnt(a,c));
	}
	return 0;
}

int dfs(int p,int u,int c)
{
	vnt::iterator i;
	par[u] = p;
	cap[u] = c;
	for(i=way[u].begin();i!=way[u].end();i++) {
		if((*i).first==p) continue;
		dfs(u,(*i).first,(*i).second);
	}
	return 0;
}
		

int main()
{
	int i,j,k;
	int ab,c;
	read_input();
	sort(arm,arm+n);
	for(i=0;i<n;i++) {
		arm[i].first = arm[i].second;
		tm[i] = -1;
	} tm[0] = 0;
	dfs(-1,0,0);
	for(c=1;true;c++) {
		for(i=0;i<n;i++) crm[i] = cap[i];
		for(i=0,ab=1;i<n;i++) {
			if(arm[i].first==0) continue;
			else ab = 0;
			if(crm[arm[i].first]==0) continue;
			crm[arm[i].first]--;
			arm[i].first = par[arm[i].first];
			if(arm[i].first==0) tm[arm[i].second] = c;			
		}
		if(ab) break;
	}
	for(i=0;i<n;i++) printf("%d ",tm[i]);
	printf("\n");
	return 0;
}
