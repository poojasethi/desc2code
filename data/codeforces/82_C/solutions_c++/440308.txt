#include <iostream>
#include <stdio.h>

using namespace std;

int p[5005],pr[5005],ans[5005],n,a[5005],uu[5005];
short b[5005][10000],m[5005][5005];
bool u[5005];

void send(int q)
{
    int c,i;
    c=q;
    i=0;
    while (c!=1)
    {
        while (b[c][i]>=pr[c])
        {
            i++;
        }
        b[c][i]++;
        c=p[c];
        i++;
    }
    ans[q]=i;
}

void dfs(int q)
{
    u[q]=true;
    for (int i=1;i<=n;i++)
    {
        if ((!u[i])&&m[q][i])
        {
            pr[i]=m[q][i];
            p[i]=q;
            dfs(i);
        }
    }
}

int main()
{
    //freopen("in.txt","r",stdin);
    int i,j,aa,bb,c;
    scanf("%d",&n);
    for (i=1;i<=n;i++)
    {
        scanf("%d",&a[i]);
        uu[i]=i;
        u[i]=false;
        for (j=1;j<=n;j++)
          m[i][j]=0;
        for (j=0;j<=n+n*2/3+15;j++)
          b[i][j]=0;
    }
    for (i=1;i<=n-1;i++)
    {
        scanf("%d %d %d",&aa,&bb,&c);
        m[aa][bb]=c;
        m[bb][aa]=c;
    }
    dfs(1);
    for (i=1;i<=n;i++)
      for (j=i+1;j<=n;j++)
        if (a[i]>a[j])
        {
            swap(a[i],a[j]);
            swap(uu[i],uu[j]);
        }
    for (i=1;i<=n;i++)
    {
        send(uu[i]);
    }
    for (i=1;i<=n;i++)
      printf("%d ",ans[i]);
}
