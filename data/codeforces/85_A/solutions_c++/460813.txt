#include <stdio.h>

char a[6][110];

int main()
{
	int n, i, ch, k;
	scanf("%d", &n);
	ch=0;
	if (n==1)
	{
		a[0][0]=a[1][0]='a';
		a[2][0]=a[3][0]='b';
	}
	else 
	{
		a[0][0]=a[0][1]='a';
		a[1][0]=a[1][1]='b';
		a[2][0]=a[3][0]='c';
		ch=3;
	}
	for (i=1; i<n; i++)
	{
		for (k=0; k<4; k++)
		{
			if (a[k][i]!=0) continue;
			if (i==n-1)
			{
				a[k][i]=a[k+1][i]='a'+ch;
				ch=(ch+1)%26;
			}
			else 
			{
				a[k][i]=a[k][i+1]='a'+ch;
				ch=(ch+1)%26;
			}
		}
	}
	for (i=0; i<4; i++)
	  printf("%s\n", a[i]);
	return 0;
}
