#include<cstdio>

#define NMAX 100

int n;
char ans[4][NMAX];

void solve() {
	int j,i,cnt=0; char c;
	ans[0][0]='x';
	ans[1][0]='x';
	for(j=0;j<n;j++) {
		for(i=0;i<4;i++) {
			if(ans[i][j])continue;
			if(j+1<n) {
				c="abcdefgh"[cnt];
				cnt=((++cnt)&7);
				ans[i][j]=c;
				ans[i][j+1]=c;
			} else {
				ans[i][j]='y';
				ans[i+1][j]='y';
			}
		}
	}
}

void output() {
	int i,j;
	for(i=0;i<4;i++) {
		for(j=0;j<n;j++)putchar(ans[i][j]);
		puts("");
	}
}

int main() {
	scanf("%d",&n);
	solve();
	output();
	return 0;
}
