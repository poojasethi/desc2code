#include <stdio.h>
#include <math.h>
#include <vector>
#include <string>
#include <string.h>
#include <algorithm>
#include <map>
#include <set>
#include <queue>
#include <iostream>
#include <sstream>
using namespace std;

int main()
{
  int n;
  cin >> n;
  for (int i=0; i<4; i++) {
    if (i<2) printf("a");
    int m = n - (i<2);
    for (int j=0; j<m/2; j++) {
      char ch = 'c' + i*5 + (j%2);
      printf("%c%c", ch, ch);
    }
    if (m%2) printf("b");
    printf("\n");
  }
  return 0;
}
