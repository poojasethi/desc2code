#include <cstdio>

char ans[4][150];
int n;

int main() {
  scanf("%d", &n);
  if (n == 1) {
    puts("a");
    puts("a");
    puts("b");
    puts("b");
    return 0;
  }

  int color = 4;
  ans[0][0] = ans[0][1] = 'a';
  ans[1][0] = ans[1][1] = 'b';
  ans[2][0] = ans[3][0] = 'c';
  int ch = 1;
  for (int i = 1; i < n; i++) {
    if (i + 1 == n) {
      ans[2 * ch][i] = ans[2 * ch + 1][i] = 'z';
    } else {
      ans[2 * ch][i] = ans[2 * ch][i + 1] = 'a' + (color++ % 25);
      ans[2 * ch + 1][i] = ans[2 * ch + 1][i + 1] = 'a' + (color++ % 25);
      ch ^= 1;
    }
  }

  for (int i = 0; i < 4; i++) {
    ans[i][n] = 0;
    puts(ans[i]);
  }

  return 0;
}
