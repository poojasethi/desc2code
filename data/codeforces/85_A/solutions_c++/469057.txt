#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string b = "aa", a[3] = {"deed", "fuuf", "shhs"};
	for(int i = 0; i < n - 1; i ++)
		b += a[i%3];
	b += "zz";
	for(int i = 0; i < 4; i ++)
	{
		for(int j = 0; j < n; j ++)
			cout<<b[(j&1)?(j*4+i):(j*4+3-i)];
		cout<<endl;
	}
	//system("pause");
	return 0;
}
