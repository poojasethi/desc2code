#include <iostream>
using namespace std;

void domino(int n, char *xx, char *yy) {
  for (int i = 0; i < n; ++i)
    cout << ((i % 2) ? xx : yy);
}

int main(void) {
  int n;
  cin >> n;
  if (n % 2 == 0) {
    domino(n / 2, "xx", "yy"); cout << "\n";
    cout << "z"; domino(n / 2 - 1, "aa", "bb"); cout << "c\n";
    cout << "z"; domino(n / 2 - 1, "vv", "dd"); cout << "c\n";
    domino(n / 2, "xx", "yy"); cout << "\n";
  } else {
    cout << "a"; domino(n / 2, "xx", "yy"); cout << "\n";
    cout << "a"; domino(n / 2, "zz", "ww"); cout << "\n";
    domino(n / 2, "qq", "pp"); cout << "b\n";
    domino(n / 2, "ss", "tt"); cout << "b\n";
  }
  return 0;
}
