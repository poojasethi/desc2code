#include <iostream>
using namespace std;
int main(){
    int n;
    cin>>n;
    
    for (int i = 0; i < n; i++)
        cout<<(char)('a'+((i/2)%2));
    cout<<endl;
    for (int i = 0; i < n; i++)
        cout<<(char)('c'+((i/2)%2)-((i==n-(n%2))?2:0));
    cout<<endl;
    for (int i = 1; i <= n; i++)
        cout<<(char)('f'+((i/2)%2));
    cout<<endl;
    for (int i = 1; i <= n; i++)
        cout<<(char)('d'+((i>1)?0:2)+((i/2)%2)+((i==n && n%2==0)?2:0));
    cout<<endl;

    //system("PAUSE");
    return 0;
}
