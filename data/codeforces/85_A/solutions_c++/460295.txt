#include <iostream>
#include <string>

using namespace std;

int n;

void solve()
{
	string s1, s2, s3, s4;

	for (int i = 0; i < n/2; ++i)
	{
		if (i%2 == 0) { s1 += "aa"; s2 += "bb"; }
		else { s1 += "bb"; s2 += "aa"; }
	}

	if (n%2 == 1) { s1 += "c"; s2 += "c"; }

	s3 += "z"; s4 += "z";

	for (int i = 0; i < (n-1)/2; ++i)
	{
		if (i%2 == 0) { s3 += "xx"; s4 += "yy"; }
		else { s3 += "yy"; s4 += "xx"; }
	}

	if (n%2 == 0) { s3 += "w"; s4 += "w"; }

	cout << s1 << endl << s2 << endl << s3 << endl << s4 << endl;
}

int main()
{
	cin >> n;
	solve();

	return 0;
}
