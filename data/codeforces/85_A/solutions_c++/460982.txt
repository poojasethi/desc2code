#include <cstdio>

using namespace std;

char nc(char &a)
{
    a = (a+1)%26;
    return a+'a';
}

int main()
{
    int n;
    scanf("%d",&n);
    char ans[4][150]={},a=0;
    ans[0][0]=ans[1][0]=nc(a);
    for (int i=0;i<n;i+=2)
    {
        ans[0][i+1]=ans[0][i+2]=nc(a);
        ans[1][i+1]=ans[1][i+2]=nc(a);
        ans[2][i  ]=ans[2][i+1]=nc(a);
        ans[3][i  ]=ans[3][i+1]=nc(a);
    }
    if (n%2==0)
        ans[1][n-1]=ans[0][n-1];
    else
        ans[3][n-1]=ans[2][n-1];
    for (int i=0;i<4;++i)
    {
        ans[i][n]=0;
        printf("%s\n",ans[i]);
    }
    return 0;
}
