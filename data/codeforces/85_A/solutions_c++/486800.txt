#include <cstdio>
using namespace std;

char mat[5][101];
int main ()
{
    int N;
    scanf("%d",&N);
    int n=3;
    for (int i=3;i<=N;i+=2,n+=4) mat[1][i]=mat[1][i-1]=n%26,mat[2][i]=mat[2][i-1]=(n+1)%26;
    n=1;
    for (int i=2;i<=N;i+=2,n+=4) mat[3][i]=mat[3][i-1]=n%26,mat[4][i]=mat[4][i-1]=(n+1)%26;
    if (N%2==0) mat[1][N]=mat[2][N]=(n+24)%26;
    else mat[3][N]=mat[4][N]=n%26;
    for (int i=1;i<=4;++i) {
        for (int j=1;j<=N;++j) printf("%c",mat[i][j]+'a');
        printf("\n");
    }
    return 0;
}
