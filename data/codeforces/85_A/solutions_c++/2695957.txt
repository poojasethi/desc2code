#include <cstdio>
#include <cstring>

using namespace std;

char pre[][5]={"a","b","c","c"};
char mid[][5]={"abba","baab","ddee","eedd"};
char suf[][4][5]={{"abb","baa","ddf","eef"},{"abbf","baaf","ddee","eedd"},{"a","b","f","f"},{"af","bf","dd","ee"}};

int main() {
	int n;
	scanf("%d",&n);
	if (n==1) { printf("a\na\nb\nb\n"); return 0; }
	for(int i=0;i<4;i++) {
		printf(pre[i]);
		for(int j=0;j<n-5;j+=4) printf(mid[i]);
		printf("%s\n",suf[n%4][i]);
	}
	return 0;
}
