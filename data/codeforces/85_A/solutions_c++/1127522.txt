#include <iostream>
using namespace std;

void print(int n, char c)
{
	for(int i = 0; i < n; i++)
		cout<<char(c+(i/2)%2);
}

int main()
{
	int n;
	cin >> n;
	if(n%2==0)
	{
		print(n,'a'); cout << endl;
		cout << 'z'; print(n-2,'c'); cout << 'x' << endl;
		cout << 'z'; print(n-2,'e'); cout << 'x' << endl;
		print(n,'g'); cout << endl;
	}
	else
	{
		print(n-1,'a'); cout << 'z' << endl;
		print(n-1,'c'); cout << 'z' << endl;
		cout<<'x';print(n-1,'e');cout<<endl;
		cout<<'x';print(n-1,'g');cout<<endl;
	}
	return 0;
}
