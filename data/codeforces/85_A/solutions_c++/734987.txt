#include <stdio.h>

int main()
{
    int n, m, x;

    scanf("%d", &n);

    for(int i=0;i<4;i++)
    {
        if(i<2)
            printf("a");

        m=n-(i<2);

        for(int j=0;j<m/2;j++)
        {
            x = 'a'+(i+j)%24+1;
            printf("%c%c", x, x);
        }

        if(m%2)
            printf("z");

        printf("\n");
    }

    return 0;
}
