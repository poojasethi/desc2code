#include <iostream>

using namespace std;

const int MaxN= 150;
char g[4][MaxN];

int main(){
	int n;
	cin >> n;
	g[2][0]= g[3][0]= 'a';
	char now= 'b';
	int r= 0;
	for (int i=0 ; i<n ; i++){
		if (i==n-1){
			g[r][i]= g[r+1][i]= now;
			break;
		}
		g[r][i]= g[r][i+1]= now;
		now++;
		if (now>'z')
			now= 'a';
		g[r+1][i]= g[r+1][i+1]= now;
		r+= 2;
		r%= 4;
		now++;
		if (now>'z')
			now= 'a';
	}
	for (int i=0 ; i<4 ; i++){
		for (int j=0 ; j<n ; j++)
			cout << g[i][j];
		cout << endl;
	}
	return 0;
}
