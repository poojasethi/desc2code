#include <iostream>
using namespace std;

char field[4][101] = {0};

int main()
{
	int i, n;
	bool odd = true;
	cin >> n;
	for (i=0; i+2<=n; i+=2)
	{
		field[0][i] = field[0][i+1] = (odd)?'a':'f';
		field[1][i] = field[1][i+1] = (odd)?'b':'g';
		odd = !odd;
	}
	if (i<n) field[0][i]=field[1][i]='k';
	field[2][0]=field[3][0]='c';
	for (i=1; i+2<=n; i+=2)
	{
		field[2][i] = field[2][i+1] = (odd)?'d':'h';
		field[3][i] = field[3][i+1] = (odd)?'e':'i';
		odd = !odd;
	}
	if (i<n) field[2][i]=field[3][i]='m';
	cout << field[0] << endl;
	cout << field[1] << endl;
	cout << field[2] << endl;
	cout << field[3] << endl;
	return 0;
}
