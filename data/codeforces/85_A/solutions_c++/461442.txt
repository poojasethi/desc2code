#include <iostream>
#include <stdio.h>
#include <vector>
using namespace std;
char res[103][4];

int main()
{
	int n;
	scanf("%d",&n);
	if(n==1)
	{
		printf("%c\n%c\n%c\n%c\n",'a','a','b','b');
		return 0;
	}
	res[0][0]='a';
	res[0][1]='a';
	res[0][2]='l';
	res[0][3]='m';
	for(int i=1;i<n;i+=2)
	{
		res[i+1][0]=res[i][0]='b'+(i%4);
		res[i+1][1]=res[i][1]='c'+(i%4);
		res[i][2]='k'+1-(i%4)/2;
		res[i][3]='l'+1-(i%4)/2;
		res[i+1][2]='k'+(i%4)/2;
		res[i+1][3]='l'+(i%4)/2;
	}
	if(n%2==0)
	{
		res[n-1][0]='x';
		res[n-1][1]='x';
	}
	else
	{
		res[n-1][2]='x';
		res[n-1][3]='x';
	}

	for(int i=0;i<4;i++)
	{
		for(int j=0;j<n;j++)
			printf("%c",res[j][i]);
		printf("\n");
	}
}
