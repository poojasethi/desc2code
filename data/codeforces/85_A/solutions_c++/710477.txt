#include<iostream>
#include<string.h>
using namespace std;
int main()
{
	int n;

	cin>>n;

	if(n==1)
	{
		cout<<"a\na\nb\nb\n";
		return 0;
	}

	for(int k = 0 ; k < 2 ; ++k )
	{
		for(int i = 0 ; i < n-1 ; i += 2 )
		{
				if(i%4)	cout<<char('a'+k)<<char('a'+k);
				else	cout<<char('c'+k)<<char('c'+k);
		}
		if(n%2==1)	cout<<'z';
		cout<<endl;
	}

	for(int k = 0 ; k < 2 ; ++k )
	{
		cout<<'k';	
		for(int i = 1 ; i < n-1 ; i+=2 )
		{
			if(i%4==1)	cout<<char('e'+k)<<char('e'+k);
			else    cout<<char('g'+k)<<char('g'+k);
		}
		if(n%2==0)cout<<'x';
		cout<<endl;
	}
			
}
