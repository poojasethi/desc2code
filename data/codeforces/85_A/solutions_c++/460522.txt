#include <stdio.h>
int i,n;
char mas[10][200];
int main()
{
scanf("%d",&n);
for(i=0;i<n;i++)
   {
   mas[0][i] = 'b' + (i/2)%26;
   if(mas[0][i] > 'z') mas[0][i] -= 26;
   mas[1][i] = 'a' + (i/2)%26;
   if(mas[1][i] > 'z') mas[1][i] -= 26;
   if(i == n-1 && (n & 1))
      mas[1][i] = mas[0][i];
   }
mas[2][0] = 'd';
mas[3][0] = 'd';
for(i=1;i<n;i++)
   {
   mas[2][i] = 'f' + ((i-1)/2)%26;
   if(mas[2][i] > 'z') mas[2][i] -= 26;
   mas[3][i] = 'e' + ((i-1)/2)%26;
   if(mas[3][i] > 'z') mas[3][i] -= 26;
   if(i == n-1 && !(n & 1))
      mas[3][i] = mas[2][i];
   }
printf("%s\n%s\n%s\n%s\n",mas[0],mas[1],mas[2],mas[3]);
return 0;
}
