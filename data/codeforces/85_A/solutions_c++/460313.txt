#include <iostream>

using namespace std;

int main (int argc, char const *argv[])
{
	int n;
	cin >> n;
	int p = n / 2;
	for(int i = 0; i < p; i++) {
		if (i%2) cout << "aa";
		else cout << "bb";
	}
	if (n%2) cout << "c";
	cout << endl;
	
	for(int i = 0; i < p; i++) {
		if (i%2) cout << "bb";
		else cout << "aa";
	}
	if (n%2) cout << "c";
	cout << endl;
	
	cout << "d";
	p = (n-1) / 2;
	for(int i = 0; i < p; i++) {
		if (i%2) cout << "ee";
		else cout << "ff";
	}
	if (n%2 == 0) cout << "g";
	cout << endl;
	
	cout << "d";
	for(int i = 0; i < p; i++) {
		if (i%2) cout << "ff";
		else cout << "ee";
	}
	if (n%2 == 0) cout << "g";
	cout << endl;
	return 0;
}
