int main(){
	char c[4][110];
	int n;
	scanf("%d",&n);
	for(int i=0;i<n;i+=2){
		c[0][i]=c[0][i+1]='a'+i%4;
		c[1][i]=c[1][i+1]='a'+4+i%4;
	}
	c[2][0]=c[3][0]='y';
	for(int i=1;i<n;i+=2){
		c[2][i]=c[2][i+1]='a'+8+i%4;
		c[3][i]=c[3][i+1]='a'+12+i%4;
	}
	if(n&1) c[0][n-1]=c[1][n-1]='z';
	else c[2][n-1]=c[3][n-1]='x';
	for(int i=0;i<4;i++)c[i][n]=0;
	for(int i=0;i<4;i++)printf("%s\n",c[i]);
	return 0;
}
