#include <stdio.h>
#include <iostream>
#include <map>
using namespace std;


int main() {
    char dom[4][101]={};
    int n;
    cin>>n;
/*  if (n==1) {
        printf("-1\n");
        return 0;
    }*/
    
    for (int x=0;x<n;x++) {
        dom[0][x]='a'+   (x/2)%6;
        dom[1][x]='a'+6+ (x/2)%6;
        dom[2][x]='a'+12+((x+1)/2)%6;
        dom[3][x]='a'+18+((x+1)/2)%6;
    }
    dom[3][0]=dom[2][0];
    if (n%2==0) {
        dom[3][n-1]=dom[2][n-1];
    } else {
        dom[1][n-1]=dom[0][n-1];
    }
    for (int i=0;i<4;i++)
        printf("%s\n", dom[i]);
    return 0;
}
