#include <cstdio>
int N;
char cover[4][101];
int main(){
      scanf("%d", &N);
      cover[0][0] = 'a';
      cover[1][0] = 'a';
      cover[2][0] = 'b';
      cover[3][0] = 'b';
      int c = 1;
      int toUse = 0;
      for(int i = 1; i<N; i++){
        cover[2-toUse][i] = cover[2-toUse][i-1] = 'a'+c;
        cover[3-toUse][i-1] = cover[3-toUse][i] = 'a'+c+1;
        cover[0+toUse][i] = cover[1+toUse][i] = 'a'+c+2;
        c+=2; c%=24;
        toUse ^= 2;
      }
      for(int i = 0; i<4; i++){
        cover[i][N] = '\0';
        printf("%s\n", cover[i]);
      }
}
