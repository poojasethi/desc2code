#include <iostream>

using namespace std;

void print(char c, int n, char pre = 0, char post = 0)
{
	if (pre != 0)
		cout << pre;

	for (int i = 0; i < n; i++)
	{
		cout << c << c;
		c++;
		c = (c - 'a' - 2) % 24 + 'a' + 2;
	}

	if (post != 0)
		cout << post;

	cout << endl;
}

int main(int argc, char* argv[])
{
	int n;
	cin >> n;

	if (n % 2 == 0)
	{
		print('c', n / 2);
		print('e', n / 2 - 1, 'a', 'b');
		print('c', n / 2 - 1, 'a', 'b');
		print('d', n / 2);
	}
	
	if (n % 2 == 1)
	{
		print('c', n / 2, 'a');
		print('e', n / 2, 'a');
		print('c', n / 2, 0, 'b');
		print('d', n / 2, 0, 'b');
	}
}
