#include<iostream>
#include<cstdio>
#include<string.h>
using namespace std;
int n;
int main(){
    cin >> n;
    for(int i=0;i<4;i++){
        if(i<2) printf("a");
        int m=n-(i<2);
        for(int j=0;j<m/2;j++) printf("%c%c",'a'+(i+j)%24+1,'a'+(i+j)%24+1);
        if(m%2) printf("z");
        printf("\n");
    }
    return 0;  
}
