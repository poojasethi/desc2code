#include <iostream>
#include <string>
using namespace std;
int n,i;
string a,b,c,d,x,y,z,q;
int main()
{
 cin>>n;
 x="bb";
 y="cc";
 z="ee";
 q="ff";
 for(;i<(n-1)/2;++i)
 {
  a+=x;
  b+=y;
  swap(x,y);
  c+=z;
  d+=q;
  swap(z,q);
 }
 if(n%2)
  cout<<'a'+a<<'\n'<<'a'+b<<'\n'<<c+'d'<<'\n'<<d+'d'<<'\n';
 else
  cout<<'a'+a+'d'<<'\n'<<'a'+b+'d'<<'\n'<<"ff"+c<<'\n'<<"ee"+d<<'\n'; 
}
