n = input()
def func(color, fb):
    s = ""
    lp = n-len(fb)
    for i in xrange(lp):
        s += color[i/2%3]
    if fb == "f":
        s = "x" + s
    elif fb == "b":
        s += "x"
    elif fb == "fb":
        s = "x" + s + "y"
    print s
if n == 1:
    print "a\na\nb\nb\n"

elif n % 2 == 0:
    func("rgb", "")
    func("lmn", "fb")
    func("stu", "fb")
    func("rgb", "")
else:
    func("rgb", "b")
    func("lmn", "b")
    func("stu", "f")
    func("rgb", "f")
