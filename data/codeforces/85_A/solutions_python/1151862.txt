alphabets = 'abcdefghijklmn'

num = int(raw_input())
line = ['z','z','','']
i=0
while i<num-1:
    x = 2*i % 14
    y = (2*i+1) % 14
    line[0] += alphabets[x]*2
    line[1] += alphabets[y]*2
    line[2] += alphabets[x]*2
    line[3] += alphabets[y]*2
    i += 2
if num%2==0:
    line[0] = (line[0])[:-2]
    line[1] = (line[1])[:-1]
    line[0] += (line[1])[-1]
else:
    line[2] += 'x'
    line[3] += 'x'
