n=input()
m=n/2
o=n%2

C=lambda c: chr(ord('a') + c%26)
l=lambda a,p: ''.join(C(i%26)*2 for i in range(a,p))
print l(0,m)+ (C(m) if o else '')
print l(4,m+4)+ (C(m) if o else '')
print C(11) + l(10,(n-1)/2+10) + (C((n-1)/2+6) if not o else '')
print C(11) + l(4,(n-1)/2+4) + (C((n-1)/2+6) if not o else '')
