import sys

N = int(raw_input())

def fill(s, p):
    while len(s) + len(p) <= N:
        s += p
    return s

r = ['']*4

r[0] = fill(r[0], 'aaff')
r[0] = fill(r[0], 'aa')

r[1] = fill(r[1], 'bbgg')
r[1] = fill(r[1], 'bb')
if len(r[0]) < N:
    r[0] += 'x'
    r[1] += 'x'

r[2] = 'y'
r[3] = 'y'

r[2] = fill(r[2], 'ddee')
r[2] = fill(r[2], 'dd')

r[3] = fill(r[3], 'hhii')
r[3] = fill(r[3], 'hh')

if len(r[2]) < N:
    r[2] += 'z'
    r[3] += 'z'

for out in r:
    print out
