n = int(raw_input())

k0 = n % 2
t0 = (n - 1) % 4
k3 = (n - 1) // 4
k1 = k3 + (t0 + 1) // 4
k2 = 1 if t0 in (1, 2) else 0
k4 = (t0 + 2) // 4

print "aabb" * k1 + "aa" * k2 + "e" * k0
print "ccdd" * k1 + "cc" * k2 + "e" * k0
print "j" + "ffgg" * k3 + "ff" * k4 + "k" * (1 - k0)
print "j" + "hhii" * k3 + "hh" * k4 + "k" * (1 - k0)
