#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

from itertools import cycle

H1 = ['a','b','c']
H2 = ['d','e','f']

def solve(n):
    if n==1:
        return ['a','a','b','b']
    if n==2:
        return ['aa','bb','cc','dd']

    h1 = cycle(H1)
    h2 = cycle(H2)
    if n % 2 == 0:
        Ro1 = []
        Ro2 = []
        for _ in range(n/2):
            Ro1.extend([h1.next()]*2)
            Ro2.extend([h2.next()]*2)
        Ri1 = ['x']
        Ri2 = ['x']
        for _ in range(n/2-1):
            Ri1.extend([h2.next()]*2)
            Ri2.extend([h1.next()]*2)
        Ri1.append('y')
        Ri2.append('y')
        return [Ro1,Ri1,Ri2,Ro2]
    else:
        R1 = []
        R2 = []
        for _ in range((n-1)/2):
            R1.extend([h1.next()]*2)
            R2.extend([h2.next()]*2)
        R1.append('y')
        R2.append('y')
        R3 = ['x'] 
        R4 = ['x']
        for _ in range((n-1)/2):
            R3.extend([h1.next()]*2)
            R4.extend([h2.next()]*2)
        return [R1,R2,R3,R4]

n = int(input.readline())

M = solve(n)
for row in M:
    output.write('%s\n' % ''.join(row))
