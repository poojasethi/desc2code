#! /usr/bin/env python

import sys

n = int(input())
if n == 1:
    print 'a\na\nb\nb\n'
    sys.exit(0)
if n & 1:
    print ('c' + 'aabb' * n)[:n]
    print ('c' + 'bbaa' * n)[:n]
    print ('ddff' * n)[:n - 1] + 'e'
    print ('ffdd' * n)[:n - 1] + 'e'
else:
    print ('c' + 'aabb' * n)[:n - 1] + 'd'
    print ('c' + 'bbaa' * n)[:n - 1] + 'd'
    print ('eeff' * n)[:n]
    print ('ffee' * n)[:n]
