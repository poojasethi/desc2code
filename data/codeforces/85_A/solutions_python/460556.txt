import sys
n = input()
if n==1: 
    print 'a\na\nb\nb'
    sys.exit(0)
f = lambda s,n: list((s*n)[:n])
l1 = f('aabb',n)
l2 = f('ddee',n)
l3 = ['z']+f('ffgg',n-1)
l4 = ['z']+f('hhii',n-1)
if n&1: l2[-1]=l1[-1] = 'c'
else: l4[-1]=l3[-1] = 'x'
print ''.join(l1)
print ''.join(l2)
if n==2:
    print 'ff\ngg'
    sys.exit(0)
print ''.join(l3)
print ''.join(l4)
