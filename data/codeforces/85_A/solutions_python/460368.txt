n = int(raw_input())
res = [['.' for i in xrange(n)] for j in xrange(4)]
a = 'qwertyuiopasdfghjklzxcvbnm'

o = 0
def place(c, r, h):
	global o, res
	o += 1
	o %= 26
	res[r][c] = a[o]
	if h: c += 1
	else: r += 1
	res[r][c] = a[o]
	
place(0, 0, False)
for i in xrange(n - 1):
	if i % 2 == 0:
		place(i, 2, True)
		place(i, 3, True)
	else:
		place(i, 0, True)
		place(i, 1, True)
if n % 2 == 1:
	place(n - 1, 2, False)
else:
	place(n - 1, 0, False)

print "\n".join(["".join(x) for x in res])
