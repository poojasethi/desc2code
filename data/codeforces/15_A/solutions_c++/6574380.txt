#include <bits/stdc++.h>

using namespace std;

const int maxN = 100 * 1000 + 321;
pair<double , double> a[maxN];
int main()
{
  int n, t;
  cin >> n >> t;
  for(int i = 0 ; i < n; i ++)
    cin >> a[i].first >> a[i].second;
  
  sort( a , a + n);
  int ans = 2;
  for(int i = 0 ; i < n - 1; i ++)
    {
      double cur = a[i + 1].first - a[i].first;
      cur -= a[i].second / 2 + a[i + 1].second / 2;
      if(cur > t)
	ans += 2;
      if(cur == t)
	ans ++;
    }
  cout << ans << endl;
  return 0;
}
