#include<cstdio>
#include<algorithm>
using namespace std;

int main(void){
 int n,t;
 int x[1000],a[1000];
 int count;
 int i,j,k;

 scanf("%i%i",&n,&t);
 for(i=0;i<n;i++)
  scanf("%i%i",&x[i],&a[i]);
 
 for(i=0;i<n;i++)
  for(j=0;j<n-i-1;j++)
   if(x[j]>x[j+1]){
    k=x[j+1];x[j+1]=x[j];x[j]=k;
    k=a[j+1];a[j+1]=a[j];a[j]=k;
   }
 count=2;

 for(i=0;i<n-1;i++){
  if(2*x[i+1]-a[i+1]-2*x[i]-a[i]>2*t) count+=2;
  if(2*x[i+1]-a[i+1]-2*x[i]-a[i]==2*t) count++;
 }

 printf("%i\n",count);
 return 0;
}
