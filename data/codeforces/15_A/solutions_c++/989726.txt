#include<iostream>
#include<utility>
#include<algorithm>
using namespace std;
const int maxn=2000;
pair<float,float> arr[maxn];
int main(){
	int n,s,ans=2;
	cin>>n>>s;
	for(int i=0;i<n;i++){
		float a,b;
		cin>>a>>b;
		arr[i].first=(float)a-b/2;
		arr[i].second=(float)a+b/2;
	}
	sort(arr,arr+n);
	for(int i=1;i<n;i++){
		float dist=arr[i].first-arr[i-1].second;
		if(dist==s)
			ans++;
		else if(dist>s)
			ans+=2;
	}
	cout<<ans<<endl;
}
