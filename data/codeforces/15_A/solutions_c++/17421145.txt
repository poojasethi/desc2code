#include<stdio.h>
#include<algorithm>

struct house{
  int x, a;
  bool operator<(const house b) const{
    return x < b.x;
  }
};

int main(int agrc, char*argv[]){
  int N, T; scanf("%d %d", &N, &T);
  house h[1000];
  for(int i=0; i<N; i++){
    scanf("%d %d", &(h[i].x), &(h[i].a));
  }
  std::sort(h, h+N);
  int res = 2;
  for(int i=0; i<N-1; i++){
    int d2 = (2*h[i+1].x - h[i+1].a) - (2*h[i].x + h[i].a);
    if(d2==2*T) res++;
    if(d2>2*T) res+=2;    
  }
  printf("%d\n", res);
  return 0;
}
