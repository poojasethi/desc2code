#include <stdio.h>
#include <algorithm>
using namespace std;

int x[1005], a[1005];

int main(void) {
	int j, i, n, t;
	scanf("%d%d", &n, &t);
	for(i=0;i<n;i++)
		scanf("%d%d", &x[i], &a[i]);
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++)
			if(x[i]>x[j])
				swap(x[i], x[j]), swap(a[i], a[j]);
	int ans = 2;
	for(i=1;i<n;i++)
		ans += (2*(x[i]-x[i-1])-a[i]-a[i-1] >= t * 2)
			+ (2*(x[i]-x[i-1])-a[i]-a[i-1] > t * 2);
	printf("%d\n", ans);
	return 0;
}
