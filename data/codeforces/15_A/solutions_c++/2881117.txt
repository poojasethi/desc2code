#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
 int n,t,x,a,ans=2,d=0;
float r;
 float b[10000];
  scanf("%d%d",&n,&t);
  r=t;
  for(int i=0;i<n;i++)
  {
         scanf("%d%d",&x,&a);
        b[d]=x-(float)a/2;
        b[d+1]=x+(float)a/2;
        d=d+2;
  }
 sort(b+0,b+n*2);
  for(int j=1;j<n*2-1;j=j+2)
  {
    float s=b[j+1]-b[j];
    if(s==r)
    ans=ans+1;
    if(s>r)
    ans=ans+2;
  }
 cout<<ans;
return 0;
}
