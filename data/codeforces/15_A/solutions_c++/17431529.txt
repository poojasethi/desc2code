#include<cstdio>
#include<algorithm>
using namespace std;
#define N 2100

int a[N], x, y, n, t;

int main(){
    int ans = 2;
    scanf("%d %d", &n, &t);
    for(int i = 0; i < n; i ++){
        scanf("%d %d", &x, &y);
        a[i*2] = 2 * x - y;
        a[i*2+1] = 2 * x + y;
    }
    sort(a, a + 2 * n);
    for(int i = 1; i < n; i ++){
        x = a[i*2-1] + t;
        y = a[i*2] - t;
        if(x == y) ans ++;
        else if(x < y) ans += 2;
    }
    printf("%d\n", ans);
}
