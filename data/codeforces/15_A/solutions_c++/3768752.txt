#include<stdio.h>
double A[1005];
int B[5005];
int n,t;
int main()
{
	double st,en;
	int i,t1,t2,j=0,s=0;
	scanf("%d %d",&n,&t);
	

	for(i=0;i<n;i++)
	{
		scanf("%d %d",&t1,&t2);
		B[t1+1000]=t2;
	}
	for(i=-1000;i<=1000;i++)
	{
		if(B[i+1000])
		{
			st=i-B[i+1000]/2.0;
			if(i+1000)
				A[j++]=st-en;
			en=i+B[i+1000]/2.0;
		}
	}
	for(i=0;i<j;i++)
	{
		if(A[i]==t) s++;
		else if(A[i]>t) s+=2;
	}
	printf("%d\n",s+2);
	return 0;
}
