#include <iostream>
#include <algorithm>
using namespace std;
double t;
pair<double,double> x[1000];
int main()
{
    int n;
    cin>>n>>t;
    for(int i=0;i<n;++i)
        cin>>x[i].first>>x[i].second;
    int r=2;
    sort(x,x+n);
    for(int i=1;i<n;++i){
        if((x[i].first-x[i].second/2)-(x[i-1].first+x[i-1].second/2)==t)
            ++r;
        else if((x[i].first-x[i].second/2)-(x[i-1].first+x[i-1].second/2)>t)
            r+=2;
    }
    cout<<r;
    return 0;
}
