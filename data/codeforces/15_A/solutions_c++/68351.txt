#include <iostream>
#include <algorithm>
using namespace std;

struct P {
	int a,b;
} p[1002];

bool comp(P x,P y)
{
	return (x.a<y.a);
}

int main()
{
	int n,t,i,k,x,l;
	while( cin>>n>>t )
	{
		t*=10;
		for(i=0;i<n;i++)
		{
			cin>>x>>l;
			x*=10;
			l*=10;
			p[i].a=x-l/2;
			p[i].b=x+l/2;
		}
		sort(&p[0],&p[n],comp);

		k=2;
		for(i=1;i<n;i++)
		{
			if(p[i].a-p[i-1].b > t) k+=2;
			else if(p[i].a-p[i-1].b == t) k+=1;
		}
		cout<<k<<endl;
	}
	return 0;
}
