#include<cstdio>
#include<algorithm>
using namespace std;
struct po{
	int x,a;
	bool operator<(const po&b)const{
		return x<b.x;
	}
}p[1001];
int main(){
	int n,t,cnt=2;
	scanf("%d%d",&n,&t);
	for(int i=0;i<n;i++)
		scanf("%d%d",&p[i].x,&p[i].a);
	sort(p,p+n);
	for(int i=0;i<n-1;i++){
		if((p[i+1].x-p[i].x)*2-p[i].a-p[i+1].a>2*t)
			cnt+=2;
		else if((p[i+1].x-p[i].x)*2-p[i].a-p[i+1].a==2*t)
			cnt++;
	}
	printf("%d",cnt);
	return 0;
}
