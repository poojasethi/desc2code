#include <iostream>
#include <algorithm>
using namespace std;
int main(){
    int n, t, c = 2, x, y;double a[1003], b[1003]; cin >> n >> t;
    for (int i = 0; i < n; i++){
        cin >> x >> y;
        a[i] = (x + y / 2.0);
        b[i] = (x - y / 2.0);
    }
    sort(a, a + n);
    sort(b, b + n);
    for (int i = 0; i < n - 1; i++){
        if (b[i + 1] > a[i] + t)c += 2;
        else if (b[i + 1] == a[i] + t)c++;
    }cout << c << endl;
    return 0;
}
