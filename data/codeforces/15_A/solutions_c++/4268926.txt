#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,t,A[2005],i,j,c;
	cin>>n>>t;
	for(i=1;i<=n;i++)
	{
		cin>>j>>c;
		A[i*2-1]=j*2+c;
		A[i*2]=j*2-c;
	}
	sort(A+1,A+2*n+1);
	n*=2;
	t*=2;
	c=2;
	for(i=2;i<n;i+=2)
		if(A[i+1]-A[i]>t)
			c+=2;
		else if(A[i+1]-A[i]==t)
			c++;	
	cout<<c;
	return 0;		
}
