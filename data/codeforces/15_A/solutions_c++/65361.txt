#include <iostream>
#include <algorithm>

using namespace std;

int main ()
{
	int n, t, arr[3000];
	cin >> n >> t;
	for(int i = 0; i < n; i++)
	{
		int x, a;
		cin >> x >> a;
		arr[2 * i] = 2 * x - a;
		arr[2 * i + 1] = 2 * x + a;
	}
	sort(&arr[0], &arr[2 * n]);
	int res = 0;
	for(int i = 0; i < 2 * n; i += 2)
	{
		if(i == 0) res++;
		else if(arr[i] - arr[i - 1] > 2 * t) res += 2;
		else if(arr[i] - arr[i - 1] == 2 * t) res += 1;
	}
	res++;
	cout << res << endl;
}
