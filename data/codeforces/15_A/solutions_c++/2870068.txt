# include<iostream>
# include<cstdio>
# include<cstring>
# include<algorithm>
using namespace std;
pair<int,int>h[1111];

int main()
{
    int n,t;
    cin>>n>>t;
    for(int i=0;i<n;++i) cin>>h[i].first>>h[i].second;
    sort(h,h+n);
    int ans = 2;
    for(int i=0;i<n-1;++i)
    {
        double d = h[i+1].first-h[i+1].second/2.0-h[i].first-h[i].second/2.0;
        if(d == t) ans++;
        else if(d>t) ans+=2;
    }
    cout<<ans<<endl;
    return 0;
}
