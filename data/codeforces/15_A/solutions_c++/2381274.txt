#include<iostream>
#include<algorithm>
using namespace std;
int x[2048];
int main()
{
  int n, t;
  cin>>n>>t;
  for (int i = 1; i <= n; i++) {
    int xi, w;
    cin>>xi>>w;
    x[2*i] = 2*xi+w;
    x[2*i-1] = 2*xi-w;
  }
  n *= 2;
  sort (x+1, x+n+1);
  t *= 2;
  x[0] = -1000000;
  x[n+1] = 1000000;
  int ans = 0;
  for (int i = 1; i <= n; i += 2) {
    int p = x[i]-t;
    if (p > x[i-1]) ans++;
    p = x[i+1] + t;
    if (p <= x[i+2]) ans++;
  }
  cout<<ans<<endl;
}
