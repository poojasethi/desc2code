#include <bits/stdc++.h>
#define mp make_pair
using namespace std;

typedef pair<int,int> pii;

int main(){
  int N,t,ans = 2;;
  double d;
  cin >> N >> t;
  pii houses[N];
  for(int i = 0;i < N;i++) cin >> houses[i].first >> houses[i].second;
  sort(houses,houses + N);
  for(int i = 1;i < N;i++){
    d = houses[i].first - houses[i - 1].first - houses[i].second/2.0 - houses[i - 1].second/2.0;
    if(d == t)
      ans++;
    else if(d > t)
      ans += 2;
  }
  cout << ans << "\n";
  return 0;
}
