#include <iostream>
#include<algorithm>
using namespace std;
double a[3000];
int main()
{
    int n,t,i,j=0,x,c=0;
    double l;
    cin>>n>>t;
    for(i=1;i<=n;i++)
    {
        cin>>x>>l;
        a[j++]=x-l/2;
        a[j++]=x+l/2;
    }
    sort(a,a+j);
    for(i=1;i<j;i=i+2)
    {
        if(a[i+1]-a[i]==t)
        c++;
        else if(a[i+1]-a[i]>t)
        c=c+2;
    }
    cout<<c+2;
    return 0;
}
