#include<algorithm>
#include<iostream>
using namespace std;

#define MAXN 1005

int a1[MAXN] , b1[MAXN];

int main()
{
  
int n , t;
cin>>n>>t;
for(int i = 0 ; i < n ; i++)
{
    int x , a;
    cin>>x>>a;
    a1[i] = 2*x - a;
    b1[i] = 2*x + a;

}
  sort( a1 , a1 + n );
  sort( b1 , b1 + n );
  int ret = 2; 
  for(int i = 0 ; i < n - 1; i++)
  {
    int sep = a1[i+1] - b1[i];
    if( sep == 2*t )
      ret+=1;
    else if( sep > 2*t )
      ret+=2;
  }
  cout<<ret<<endl;
}
