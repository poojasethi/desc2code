#include <algorithm>
#include <iostream>
using namespace std;

int main(void)
{
	int n, t, data[1000*2+1];

	cin >> n >> t;
	for (int i=1; i<=n; ++i)	{
		int x, a;
		cin >> x >> a;
		data[i*2-1]=x*2+a;
		data[i*2]=x*2-a;
	}
	sort ( data+1,data+n*2+1 );
	int answer=2;
	for (int i=1; i<=n*2; i+=2)	{
		if (data[i]-data[i-1] > t*2)	answer+=2;
		else if (data[i]-data[i-1] == t*2)	++answer;
	}
	cout << answer << endl;
}
