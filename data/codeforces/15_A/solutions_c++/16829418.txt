#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n, t;
	cin>>n>>t;
	pair <int,int> A[n];
	for (int i=0;i<n;i++){
		cin>>A[i].first>>A[i].second;
	}
	sort(A,A+n);
	float k;
	int ans=2;
	for (int i=0;i<n-1;i++){
		k=A[i+1].first-A[i].first;
		k=k-A[i+1].second*1.0/2-A[i].second*1.0/2;
		if (k>t) ans=ans+2;
		else if (k==t) ans=ans+1;
	}
	cout<<ans;
	return 0;
}
