def readint(): return int(raw_input())
def readfloat(): return float(raw_input())
def readarray(N, foo):
        res = []
        for _ in xrange(N):
                res.append(foo())
        return res
def readlinearray(foo): return map(foo, raw_input().split())

def dist((x1, a1), (x2, a2)):
    return x2 - (x1 + a1)

def get_npos(n, t, houses):
    r = 2
    
    for i in xrange(len(houses)-1):
        chouse = houses[i]
        nhouse = houses[i+1]
        d = dist(chouse, nhouse)
        if d >= t:
            r += 1
            if d > t:
                r += 1
    return r

if __name__ == '__main__':
    n, t = readlinearray(int)
    
    houses = []
    for _ in xrange(n):
        xi, ai = readlinearray(float)
        house = (xi - ai / 2.0, ai)
        houses.append(house)
    houses.sort()
    print get_npos(n, t, houses)
