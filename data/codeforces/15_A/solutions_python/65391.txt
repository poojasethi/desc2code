#! /usr/bin/python
import sys

#nacteni 2 cisel
numbers = sys.stdin.readline()[:-1].split()
cases = int(numbers[0])
t = int(numbers[1])
actual_case = 0

list = []
while actual_case < cases:
    # reading and so
    actual_case += 1
    
    #nacteni 2 cisel
    numbers = sys.stdin.readline()[:-1].split()
    centre = int(numbers[0])
    length = int(numbers[1])

    # zacatek
    list.append((centre - length/2.0,1))
    # konec
    list.append((centre + length/2.0,0))

list.sort()
possible = 1
for i in range(len(list)-1):
    if list[i][1] == 0:
	if list[i+1][0] -list[i][0] > t:
	    possible +=2
	elif list[i+1][0] -list[i][0] == t:
            possible +=1

possible += 1
print possible
