class Casa:
    def __init__(self, X, A):

        self.A = A

def resolver(n, t, c):
    r = 0

    for a in range(n-1):

        p1 = c[a].X + (c[a].A / 2)
        p2 = c[a + 1].X - (c[a + 1].A / 2)

        d = p2-p1

        if d > t: r += 2
        if d == t: r += 1
    r += 2
    return r

if __name__ == "__main__":
    n, t = map(float, raw_input().split())

        casa = Casa(x, a)
