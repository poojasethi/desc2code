from operator import itemgetter

n, t = map(int, raw_input().split())
t += t
dataList = []
for i in xrange(n):
    x, a = map(int, raw_input().split())
    x += x
    dataList.append((x-a, x+a))
dataList.sort(key=itemgetter(0))
ans = 2
for i in xrange(1, len(dataList)):
    pre = dataList[i-1]
    cur = dataList[i]
    if (cur[0] > pre[1]+t):
        ans += 2
    elif (cur[0] == pre[1]+t):
        ans += 1
print ans
