n, t = map(int, raw_input().split())
a = [ map(int, raw_input().split()) for _ in range(n) ]
a.sort()
ans = 2
for i in range(1, n):
    if a[i][0] - a[i - 1][0] - (a[i][1] + a[i - 1][1]) * 0.5 == t:
        ans += 1
    elif a[i][0] - a[i - 1][0] - (a[i][1] + a[i - 1][1]) * 0.5 > t:
        ans += 2
print ans
