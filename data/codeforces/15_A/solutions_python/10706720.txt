from decimal import Decimal
from operator import itemgetter

n, t = map(int, raw_input().split())

houses = []

for x in xrange(n):
	center, width = map(Decimal, raw_input().split())
	houses.append({
		'left': center - width / 2,
		'right': center + width / 2
	})

houses.sort(key=itemgetter('left'))

places = 2 #can always put on left and right side

for x in xrange(len(houses) - 1):
	if houses[x]['right'] + t < houses[x + 1]['left']:
		places += 2 # can put on both right of house x and left of house x + 1
	elif houses[x]['right'] + t == houses[x + 1]['left']:
		places += 1

print places
