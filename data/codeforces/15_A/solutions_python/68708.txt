n, t = [int(x) for x in raw_input().strip().split()]
p = []
for i in xrange(n):
    x, a = [int(y) for y in raw_input().strip().split()]
    p.append(x - a / 2.0)
    p.append(x + a / 2.0)
p.sort()
s = 2
for i in xrange(1, 2 * n - 1, 2):
    l = p[i + 1] - p[i]
    if l > t:
        s += 2
    elif l == t:
        s += 1
print s
