from itertools import *
import sys
[n, t], ans = map (int, raw_input ().split ()), 2
houses = sorted ([map (int, raw_input ().split ()) for _ in range (n)])
for l, r in zip (houses[:-1], houses[1:]):
    ans += 2 if l[0] + 1. * l[1] / 2 + t < r[0] - 1. * r[1] / 2 else l[0] + 1. * l[1] / 2 + t == r[0] - 1. * r[1] / 2
print ans
