n, t = map(int, raw_input().split())
houses = [map(int, raw_input().split()) for _ in range(n)]
houses.sort(key=lambda (x, y): x)
count = 2; # can always put houses at ends
for i in range(1,n):
	room = (houses[i][0] - houses[i][1] / float(2)) - (houses[i - 1][0] + (houses[i - 1][1] / float(2)))
	if room > t:
		count += 2
	elif room == t:
		count += 1
print count
