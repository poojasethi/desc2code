from __future__ import division

a, t = map(int, raw_input().split())
l = [map(int, raw_input().split()) for i in range(a)]
s = []
ans = 0
for i in range(a):
    s.append([l[i][0] - l[i][1] / 2, l[i][0] + l[i][1] / 2])
s.sort()
for i in range(a - 1):
    #print abs(abs(s[i+1][0]) - abs(s[i][1]))
    if (s[i+1][0] - s[i][1]) == t:
        ans += 1
    elif (s[i+1][0] - s[i][1]) > t:
        ans += 2

ans += 2
print ans
