from operator import itemgetter
n,t = [int(x) for x in raw_input().split()]
lst = [[int(x) for x in raw_input().split()] for i in xrange(n)]
lst = sorted(lst, key=itemgetter(0))
count = 0
last = None
for x,a in lst:
    # x, a = [int(x) for x in raw_input().split()]
    b = x - (a/2.0)
    e = x + (a/2.0)
    if last == None or last < b:
        count += 1
    elif last > b:
        count -= 1
    last = e + t
    count += 1
print count
