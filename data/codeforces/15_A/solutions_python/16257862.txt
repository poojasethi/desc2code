def fun():
    n,t=map(int,raw_input().split())
    xi=[]
    ai={}
    for i in xrange(0,n):
        a,b=map(int,raw_input().split())
        xi.append(a)
        ai[a]=(b)
    xi.sort()
    ans=2
    for i in xrange(0,n-1):
        #print xi[i],ai[xi[i]]
        juli=xi[i+1]-xi[i]-(ai[xi[i+1]]+ai[xi[i]])/2.0-t
        if juli==0:
            #print i
            ans=ans+1
        elif juli>0:
            #print i
            ans=ans+2
    print ans
if __name__=='__main__':
    fun()
