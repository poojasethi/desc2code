n,t = [int(x) for x in raw_input().split()]
h = []
for i in range(n):
	h += [[int(x) for x in raw_input().split()]]
h.sort()

c = 0
for i in range(n):
	if i == 0 or 2*(h[i][0]-h[i-1][0]) >= (h[i-1][1]+h[i][1])+2*t:
		c += 1
	if i == n-1 or 2*(h[i+1][0]-h[i][0]) > (h[i][1]+h[i+1][1])+2*t:
		c += 1
print c
