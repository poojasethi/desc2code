#!/usr/bin/env python2
# -*- coding: utf-8 -*-
import sys
input = sys.stdin
output = sys.stdout

def solve(H,t):
    N = 0
    H = sorted(H, key=lambda o: o[0])
    n = len(H)
    for i in range(1,n):
        h = H[i]
        p = H[i-1]
        d = (2*h[0]-h[1]) - (2*p[0]+p[1]) 
        if d>2*t:
            N += 2
        elif d==2*t:
            N += 1
    return N+2

def numbers_from_line(d=' '):
    return [int(s) for s in input.readline().strip().split(d) if len(s.strip())>0]

A = numbers_from_line()
n,t = tuple(A[:2])

H = []
for _ in range(n):
    A = numbers_from_line()
    x,a = tuple(A[:2])
    H.append((x,a))

a = solve(H,t)
output.write('%d\n' % a)
