#!/usr/bin/python

import sys

(n, t) = map(lambda x: int(x), sys.stdin.readline().split())
houses = []
for i in range(n):
	(x, a) = map(lambda x: int(x), sys.stdin.readline().split())
	houses.append((x, a))

houses = sorted(houses, key=lambda x: x[0])
result = 2
for i in range(n-1):
	delta = houses[i+1][0] - houses[i][0] - (houses[i+1][1] + houses[i][1])/2.0
	if delta > t:
		result += 2
	elif delta == t:
		result += 1

print result
