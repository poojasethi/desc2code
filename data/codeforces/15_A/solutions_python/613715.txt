n, t = map(int, raw_input().split())
x = [[]] * n
for i in range(n):
    x[i] = map(int, raw_input().split())
x.sort()
ans = 2
for i in range(n - 1):
    dis = (x[i + 1][0] - x[i][0]) * 2 - x[i + 1][1] - x[i][1]
    if(dis == t * 2):
        ans += 1
    elif(dis > t * 2):
        ans += 2
print ans
