n, t = map(int, raw_input().split())
a = []
for i in range(n):
    x, l = map(int, raw_input().split())
    a.append((x - l / 2.0, x + l / 2.0))
a.sort()
res = 0
for i in range(1, n):
    if a[i][0] - a[i - 1][1] > t:
        res += 2
    elif a[i][0] - a[i - 1][1] == t:
        res += 1

print res + 2
