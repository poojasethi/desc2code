n, m = map(int, raw_input().split())
k, d = n * m, 2 * m
c = {'11': 0, '01': 0, '10': 0}
for i in range(n):
    t = raw_input()
    for i in ('11', '01', '10'): c[i] += t.count(i)
a, b = c['11'], c['10'] + c['01']
t = ['11'] * a + (b / 2) * ['10', '01'] + ['10'] * (b & 1) + ['00'] * (k - a - b)
for i in range(0, k, d):
    print ' '.join(t[i: i + m])
    print ' '.join(t[i + d - 1: i + m - 1: -1])
