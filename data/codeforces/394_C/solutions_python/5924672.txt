#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

n, m = map(int, raw_input().split())
nums = [0] * 3

for i in range(n) :  
	ds = raw_input().split()
	for d in ds : 
		if d == '11' : 
			nums[0] += 1
		elif d == '10' or d == '01' : 
			nums[1] += 1
		else : 
			nums[2] += 1

pattern = [['00' for j in range(m+1)] for i in range(n+1)]

index = 1
def getType(index, nums) : 
	for i in range(1, 3) : 
		if index <= sum(nums[:i]) : 
			return i

for i in range(1, n+1) : 
	if i % 2 == 0 : 
		sj, ej, dj = m, 0, -1
	else : 
		sj, ej, dj = 1, m+1, 1

	for j in range(sj, ej, dj) : 
		t = getType(index, nums) 
		if t == 1 : 
			pattern[i][j] = '11'
		elif t == 2 : 
			if pattern[i-1][j] == '10' : 
				pattern[i][j] = '01'
			else : 
				pattern[i][j] = '10'
		else : 
			pattern[i][j] = '00'

		index += 1

for i in range(1, n+1) : 
	print ' '.join(pattern[i][1:])
