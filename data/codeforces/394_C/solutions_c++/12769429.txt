
#include <bits/stdc++.h>
using namespace std;
int n,m,k;
int main() {
#ifndef ONLINE_JUDGE
    freopen("d.in", "r", stdin);
#endif
    int c1=0,c2=0,c3=0;
scanf("%d%d",&n,&m);
for(int i=0;i<n;++i)
    for(int j=0;j<m;++j){
         cin>>k;
         if(k==0)
             c1++;
         else if(k==11)
             c2++;
         else
             c3++;
    }
vector<vector<int> >v(n,vector<int>(m));
for(int i=0;i<n;++i)
    for(int j=0;j<m;++j){
        if(c2>0){
            if(i%2==0)
            v[i][j]=11;
            else
                v[i][m-1-j]=11;
            --c2;
        }
        else if(c3>0){
            if(i%2==0)
                v[i][j]=10;
            else{
                v[i][m-1-j]=1;
            }
            --c3;
        }
        else if(c1>0){
            if(i%2==0)
            v[i][j]=0;
            else
                v[i][m-1-j]=0;
            --c1;
        }
    }
for(int i=0;i<n;++i){
    for(int j=0;j<m;++j)
        printf("%.2d ",v[i][j]);
    printf("\n");
}
        return 0;
}
