#include <iostream>

using namespace std;
int n,m,s11=0,s00=0,s10=0;

{

   string s;

   for(int j=0;j<m;j++){
    cin>>s;
    if(s=="11"){s11++;}else{
        if(s=="00"){s00++;}else{s10++;}
    }
   }
   }
   for(int i=0;i<n;i++){
    for(int j=0;j<m;j++){
      if(s11){cout<<"11 ";s11--;}else{
        if(m-j<=s10&&s10){s10--;if(i%2){cout<<"10 ";}else{cout<<"01 ";}}else{if(s00){cout<<"00 ";s00--;}}

      }

    }


    cout<<endl;
   }

    return 0;
}
