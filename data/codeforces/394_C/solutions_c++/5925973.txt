#include <iostream>
#include <cstdio>
#include <cmath>
#include <algorithm>
#include <queue>
#include <string>
#include <vector>
#include <iso646.h>
using namespace std;
int main(){
	//freopen("input", "r", stdin);
	int n, m;
	cin>>n>>m;
	int a[3];
	for(int i = 0; i < 3; i++){
		a[i] = 0;
	}
	string s;
	for(int i = 0; i < n; i++){
		for(int j = 0; j < m; j++){
			cin>>s;
			a[s[0] + s[1] - 2 * '0']++;
		}
	}
	//cout<<a[0]<<" "<<a[1]<<" "<<a[2]<<endl;
	int board[1000][2000];
	for(int i = 0; i < 1000; i++){
		for(int j = 0; j < 2000; j++){
			board[i][j] = 0;
		}
	}
	for(int i = 0; i < n; i++){
		for(int j = 0; j < 2 * m; j += 2){
			if(board[i][j]) continue;
			if(a[2]){
				board[i][j] = 2;
				board[i][j + 1] = 2;
				a[2]--;
			}
			else if(a[1]){
				board[i][j] = 2;
				board[i][j + 1] = 1;
				a[1]--;
				if(i != n-1 && a[1]){
					board[i + 1][j] = 1;
					board[i + 1][j + 1] = 2;
					a[1]--;
				}
			}
			else{
				board[i][j] = 1;
				board[i][j + 1] = 1;
			}
		}
	}
	for(int i = 0; i < n; i++){
		for(int j = 0; j < 2 * m; j += 2){
			if(board[i][j] == 2){
				cout<<'1';
			}
			else{
				cout<<'0';
			}
			if(board[i][j + 1] == 2){
				cout<<'1';
			}
			else{
				cout<<'0';
			}
			cout<<' ';
		}
		cout<<endl;
	}
	return 0;
}
