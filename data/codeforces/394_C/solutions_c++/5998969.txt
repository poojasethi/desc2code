#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main()
{
    int n, m;
	int a[3];
	string s;

	cin >> n >> m;
	a[0] = a[1] = a[2] = 0;
	for (int i = 0; i < n; ++i)
		for (int j = 0; j < m; ++j)
		{
			cin >> s;
			if (s == "11")
				a[0]++;
			else if ((s == "01") || (s == "10"))
				a[1]++;
			else
				a[2]++;
		}

	for (int i = 0; i < n; ++i)
	{
		for (int j = 0; j < m; ++j)
		{
			if (j)
				cout << " ";
			if (a[0])
			{
				cout << "11";
				a[0]--;
			}
			else if (a[1] >= m - j)
			{
				a[1]--;
				if (i % 2)
					cout << "01";
				else
					cout << "10";
			}
			else
			{
				cout << "00";
			}
		}
		cout << endl;
	}
    //system("pause");
	return 0;
}
