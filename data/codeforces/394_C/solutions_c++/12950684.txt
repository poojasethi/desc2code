#include<iostream>
#include<algorithm>
#include<string>
#include<vector>
using namespace std;
string arr[1000];
int main(){
	int n, m, c11 = 0, c10 = 0, c00 = 0, c, x;
	string s, s1 = "", s2 = "";
	cin >> n >> m;
	int n1 = n;
	while (n--){
		for (int i = 0; i < m; i++){
			cin >> s;
			if (s == "11")c11++;
			else if (s == "01" || s == "10")c10++;
			else c00++;
		}
	}
	c = c10 / 2;
	x = c10 - c;
	
	for (int i = 0; i < n1; i++){
		for (int j = 0; j < m; j++)
		{
			if (c11){ cout << "11" << " "; c11--; continue; }
			if (x){ cout << "10" << " "; x--; continue; }
			if (c00){ cout << "00" << " "; c00--; continue; }
			if (c){ cout << "01"<< " "; c--; continue; }

		}
		cout << endl;
	}
	return 0;
}
