#include<bits/stdc++.h>
using namespace std;

int d[1000005], mat[1005][1005];

int main()
{
    int n, m, x=-1, y=0, l=0; bool flag=true;
    cin>>n>>m; for(int i=0;i<n;i++) for(int j=0;j<m;j++) cin>>d[l++];
    sort(d,d+l); memset(mat,0,sizeof(mat)); l--;

    for(int i=0;((i<n)&&(d[l]==11));i++) for(int j=0;((j<m)&&(d[l]==11));j++) { mat[i][j]=d[l--]; y=i; x=j; }
    for(int j=m-1;((j>x)&&(d[l]!=0));j--) { mat[y][j]=1; l--; }
    for(int i=y+1;((i<n)&&(d[l]!=0));i++) {
        flag=(!flag);
        for(int j=m-1;((j>=0)&&(d[l]!=0));j--) { if(flag) { mat[i][j]=1; l--; } else { mat[i][j]=10; l--; } }
    }

    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            if(mat[i][j]==11) { cout<<"11 "; }
            else if(mat[i][j]==10) { cout<<"10 "; }
            else if(mat[i][j]==01) { cout<<"01 "; }
            else { cout<<"00 "; }
        }
        cout<<endl;
    }

    return 0;
}
