#include <iostream>
#include <string>
using namespace std;

bool last[1005];

int main()
{
	int n, m;
	cin >> n >> m;

	int num[3] = {0, 0, 0};
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
		{
			string dom;
			cin >> dom;
			if (dom == "00")
				num[0]++;
			else if (dom == "11")
				num[2]++;
			else
				num[1]++;
		}

	int idx = 0;
	int grid[1005][1005];

	for (int i = 0; i < n; i++)
	{
		if (i % 2 == 0)
		{
			for (int j = 0; j < m; j++)
			{
				while (num[idx] == 0)
					idx++;

				grid[i][j] = idx;
				num[idx]--;
			}
		}
		else
		{
			for (int j = m - 1; j >= 0; j--)
			{
				while (num[idx] == 0)
					idx++;

				grid[i][j] = idx;
				num[idx]--;
			}
		}
	}

	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
		{
			if (grid[i][j] == 0)
				cout << "00 ";
			else if (grid[i][j] == 2)
				cout << "11 ";
			else
			{
				if (last[j])
					cout << "10 ";
				else
					cout << "01 ";
				last[j] = !last[j];
			}
		}
		cout << "\n";
	}

	return 0;
}
