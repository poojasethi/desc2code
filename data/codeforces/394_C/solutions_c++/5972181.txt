#include <iostream>
#include <fstream>
#include <algorithm>
#include <cstring>
#include <cmath>
using namespace std;
#define N 1005

int Map[N][N],n,m,x,y;
int sum[3];

void put(int v,bool flag)
{
    while (Map[x][y]>0) {
        if (y<m-1) y++;
        else { x++;y=0; }
    }
    Map[x][y]=v;
    if (v==10 && flag)
        Map[x+1][y]=1;

    if (y<m-1) y++;
    else { x++;y=0; }
}

int main()
{
    //freopen("1.in","r",stdin);
    scanf("%d%d",&n,&m);
    for (int i=0;i<n;i++)
        for (int j=0,k;j<m;j++)
        {
            scanf("%d",&k);
            if (k==0) sum[0]++;
            else if (k==11) sum[2]++;
            else sum[1]++;
        }

    //build map
    x=y=0;
    for (int k=0;k<sum[2];k++)
        put(11,false);
    int k = 0;
    while (k<sum[1]){
        bool b =(sum[1]-k>1 && x<n-1)?true:false;
        put(10,b);
        if (b) k++;
        k++;
    }

    //print;
    for (int i=0;i<n;i++)
        for (int j=0;j<m;j++){
            if (Map[i][j]==1) printf("01");
            else if (Map[i][j]==0) printf("00");
            else printf("%d",Map[i][j]);
            //format;
            if (j<m-1) printf(" ");
            else printf("\n");
        }

    return 0;
}
