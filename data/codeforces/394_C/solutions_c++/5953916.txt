#include <iostream>
#include <vector>

using namespace std;

int cuantos[3];

int main()
{
  ios::sync_with_stdio(false);
  int n,m;
  cin>>n>>m;
  int total=n*m;
  for (int i=0;i<total;i++) {
    int x;
    cin>>x;
    if (x==10) x=1;
    else if (x==11) x=2;
    cuantos[x]++;
  }
  vector<vector<int> > v(m,vector<int> (3));
  {
    int iv=0;
    while (cuantos[1]>0) {
      v[iv][1]++;
      cuantos[1]--;
      if (cuantos[1]>0 and v[iv][1]<n) {
	v[iv][1]++;
	cuantos[1]--;
      }
      iv++;
      if (iv==m) iv=0;
    }
    while (cuantos[2]>0) {
      if (v[iv][1]+v[iv][2]<n) {
	v[iv][2]++;
	cuantos[2]--;
      }
      iv++;
      if (iv==m) iv=0;
    }
  }
  for (int i=0;i<n;i++) {
    for (int j=0;j<m;j++) {
      if (j>0) cout<<" ";
      if (v[j][2]) {
	v[j][2]--;
	cout<<"11";
      } else if (v[j][1]) {
	v[j][1]--;
	if (i%2)
	  cout<<"01";
	else
	  cout<<"10";
      } else {
	cout<<"00";
      }
    }
    cout<<endl;
  }
}
