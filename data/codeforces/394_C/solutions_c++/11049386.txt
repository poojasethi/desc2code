#include <bits/stdc++.h>
using namespace std;

string res[1005][1005];
int main(){

  int n,m;
  cin>>n>>m;

   string dom;
   int all=0,none=0,half=0;
   for(int i=0;i<n;i++){
      for(int j=0;j<m;j++){
         cin>>dom;
         if(dom=="00")none++;
         if(dom=="11")all++;
         if(dom=="01" || dom=="10")half++;
      }
   }
      for(int i=0;i<n;i++){

          if(i%2==0){
             for(int j=0;j<m;j++){
                 if(all)res[i][j]="11",all--;
                 else if(half)res[i][j]="01",half--;
                 else if(none)res[i][j]="00",none--;

             }
          }
          else{
             for(int j=m-1;j>=0;j--){
                 if(all)res[i][j]="11",all--;
                 else if(half)res[i][j]="10",half--;
                 else if(none)res[i][j]="00",none--;

             }
          }

      }

      for(int i=0;i<n;i++)
          for(int j=0;j<m;j++){
             cout<<res[i][j];
             if(j==m-1)cout<<"\n";
             else cout<<" ";
          }

}
