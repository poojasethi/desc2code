#include <bits/stdc++.h>

using namespace std;

#define MAX 1005

int sum[3], ans[MAX][MAX];

int main(){
    int n, m;
    cin>>n>>m;

    for(int i=0 ; i<n ; i++){
        for(int j=0 ; j<m ; j++){
            string s;
            cin>>s;

            sum[s[0] + s[1] - 2 * '0']++;
        }
    }

    int i = 0, j = 0;

    while(sum[2]){
        ans[j][i] = 2;
        i++;
        if(i == m)
            i = 0, j++;
        sum[2]--;
    }

    for( ; i<m ; i++){
        if(sum[1] == 0)
            break;
        ans[j][i] = 1;
        sum[1]--;
    }

    if(i == m){
        i = 0;
        j++;
    }

    while(sum[1]){
        ans[j][m-i-1] = 1;
        sum[1]--;
        i++;
        if(i == m)
            i = 0, j++;
    }

    for(int i=0 ; i<n ; i++){
        for(int j=0 ; j<m ; j++){
            if(ans[i][j] == 2){
                cout<<"11 ";
            } else if(ans[i][j] == 1){
                if(i%2){
                    cout<<"01 ";
                } else {
                    cout<<"10 ";
                }
            } else {
                cout<<"00 ";
            }
        }
        cout<<endl;
    }

    return 0;
}
