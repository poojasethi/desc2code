#include<iostream>
#include<cstdio>
using namespace std;
const int MaxN = 1000 + 10;
char s[5];
int sl[3], map[MaxN][MaxN], n, m, v;
bool vis[MaxN][MaxN];
void solve()
{
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= m; ++j)
			if(sl[2])
			{
				map[i][j] = 2;
				--sl[2];
				vis[i][j] = 1;
			}
	for(int i = n; i >= 1; i -= 2)
		for(int j = m; j >= 1; --j)
			if(sl[1])
			{
				map[i][j] = 1;
				vis[i][j] = 1;
				--sl[1];
				if(!vis[i - 1][j] && sl[1])
				{
					map[i - 1][j] = -1;
					vis[i - 1][j] = 1;
					--sl[1];
				}
			}
	for(int i = 1; i <= n; ++i)
	{
		for(int j = 1; j <= m; ++j)
			if(map[i][j] == 0) printf("00 ");
			else if(map[i][j] == 1) printf("10 ");
			else if(map[i][j] == -1) printf("01 ");
			else printf("11 ");
		printf("\n");
	}
}
int main()
{
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= m; ++j)
		{
			scanf("%s", s + 1);
			v = s[1] + s[2] - ('0' << 1);
			++sl[v];
		}
	solve();
	return 0;
}
