#include<stdio.h>
#include<iostream>

using namespace std;
int ans[1004][2004];
int main()
{
	int i,j,n,m,a=0,b=0,c=0,x=0,y=0;
	string s;
	cin >> n >> m;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			cin >> s;
			if(s[0]=='0')	x++;
			if(s[0]=='1')	y++;
			if(s[1]=='0')	x++;
			if(s[1]=='1')	y++;

			if(x==2)	c++;
			else if(y==2)	b++;
			else if(x==1 && y==1)	a++;
			x=0;y=0;

		}
	}
//	cout << a << b << c << endl;
	for(i=0;i<n;i++)
	{
		for(j=0;j<2*m;j++)
		{
			ans[i][j]=-1;
		}
	}
	x=a/2;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			if(b)
			{
				if(!j)	cout << "11";
				else	cout << " 11";
				b--;
			}
			else if(x)
			{
				if(!j)	cout << "10";
				else	cout << " 10";
				x--;
				a--;
			}
			else if(c)
			{
				if(!j)	cout << "00";
				else	cout << " 00";
				c--;
			}
			else if(a)
			{
				if(!j)	cout << "01";
				else	cout << " 01";
				a--;
			}
		}
		cout << endl;
	}
	return 0;
}
