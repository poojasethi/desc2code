#include<cstdio>
#include<cstring>
#include<algorithm>
#define mem(name,value) memset(name,value,sizeof(name))
#define FOR(i,n) for(int i=1;i<=n;i++)
using namespace std;
const int maxn = 1000+10;
int maze[maxn][maxn];
char s[5];
int cnt[5];
void print_ans(int cnt){
    switch(cnt){
        case 0: printf("00"); break;
        case 1: printf("01"); break;
        case 2: printf("10"); break;
        case 3: printf("11"); break;
    }
}
int main(){
    mem(maze,0); mem(cnt,0);
    int n,m;
    scanf("%d%d",&n,&m);
    FOR(i,n) FOR(j,m){
        scanf("%s",s);
        if(strcmp(s,"11")==0) cnt[3]++;
        else if(strcmp(s,"10")==0) cnt[2]++;
        else if(strcmp(s,"01")==0) cnt[1]++;
    }
    int sum = cnt[1] + cnt[2];
    cnt[1] = sum/2, cnt[2] = sum-cnt[1];
    int r = 1, c = 1;
    while(cnt[3]--){
        maze[r][c] = 3;
        c++;
        if(c>m) r++,c=1;
    }
    while(cnt[2]--){
        maze[r][c] = 2;
        c++;
        if(c>m) r++,c=1;
    }
    r = n, c = m;
    while(cnt[1]--){
        maze[r][c] = 1;
        c--;
        if(c<1) r--,c=m;
    }
    for(int i=1;i<=n;i++){
        for(int j=1;j<=m;j++){
            if(j!=1) printf(" ");
            print_ans(maze[i][j]);
        }
        printf("\n");
    }
    return 0;
}
