#include<bits/stdc++.h>

using namespace std;
#define CLR(a,val) memset(a, val, sizeof(a))
#define SZ(V) (LL)V.size()
#define ALL(V) V.begin(),V.end()
#define RALL(V) V.rbegin(),V.rend()
#define FORN(i, n) for(LL i = 0; i < n; i++)
#define FORAB(i, a, b) for(LL i = a; i <= b; i++)
#define MOD 1000000007LL
#define PB push_back
#define MP make_pair
#define F first
#define S second

typedef long long LL;
typedef pair<LL,LL> pll;


LL a[3];
string as[1001][1001];
LL rc[2001];
int main()
{
	LL n,m;
	cin >> n >> m;
	CLR(a,0);
	FORN(i,n) FORN(j,m){
		string s;
		cin >> s;
		a[s[0]-'0' + s[1]-'0']++;
	}
	CLR(rc,0);
	FORN(i,n) FORN(j,m){
		if(a[2]>0 && rc[j]<n){
			as[rc[j]++][j]="11";
			a[2]--;
		}
		else if(a[1]>0 && rc[j]<n){
			as[rc[j]++][j]="10";
			a[1]--;
			if(a[1]>0 && rc[j]<n){
				as[rc[j]++][j]="01";
				a[1]--;
			}
		}
		else if(a[0]>0 && rc[j]<n){
			as[rc[j]++][j]="00";
			a[0]--;
		}
	}
	FORN(i,n){ FORN(j,m) cout << as[i][j] << " ";cout << endl; } 
	return 0;
}
