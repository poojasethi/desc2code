using namespace std;

typedef unsigned long long uLL;

typedef pair<int,int> pii;

int main()	
{
ios_base::sync_with_stdio(0);
	
	int n,m;
	cin>>n>>m;
	string s;
	int c11=0 , c10=0;
	for (int i=0 ; i<n ; i++)
	{
		for (int j=0 ; j<m ; j++)
		{
			cin>>s;
			if ((s[0]-'0' + s[1]-'0')==2) c11++;
			else if ((s[0]-'0' + s[1]-'0')==1) c10++;
		}
	}
	
	int u = (c11/m)+1;
	for (int i=1 ; i<=n ; i++)
	{
		for (int j=1 ; j<=m ; j++)
		{
			if (c11) { c11--; cout<<"11 "; }
			else if (c10)
			{
				if ((i-u)%2)
				{ 
					if (c10>=(m-j+1)) { c10--; cout<<"01 "; } 
					else cout<<"00 ";
				}
				else
				{
					c10--;
					cout<<"10 ";
				}
			}
			else
				cout<<"00 ";
		}
		
		cout<<"\n";
	}
	
	return 0;
}
