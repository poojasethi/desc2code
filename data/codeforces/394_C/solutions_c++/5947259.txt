#include <cstdio>
#include <algorithm>
bool cmp(const int &a,const int &b){return a>b;}
int n,m;
int a[1010101];
int fw[1050];
int d[1050][1050];
int main(){
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i)
	 for (int j=1;j<=m;++j){
		int t;scanf("%d",&t);
		if (t==11) a[(i-1)*m+j]=2;
		if (t==10||t==1) a[(i-1)*m+j]=1;
		if (t==0) a[(i-1)*m+j]=0;
		}
	std::sort(a+1,a+n*m+1,cmp);
	int p=0;
	for (int i=1;i<=n;++i){
		if (i&1){
			for (int j=1;j<=m;++j) d[i][j]=a[++p];
			}else
			for (int j=m;j>=1;--j) d[i][j]=a[++p];
		}
	for (int i=1;i<=n;++i){
		for (int j=1;j<m;++j){
			if (d[i][j]==2) {fw[j]=3;printf("11 ");}
			if (d[i][j]==1) {
				if (fw[j]==1) printf("01 ");
				if (fw[j]==2) printf("10 ");
				if (fw[j]==3||fw[j]==0){printf("01 ");fw[j]=2;}
					else fw[j]=3-fw[j];
				}
		if (d[i][j]==0) printf("00 ");
		}
		
		if (d[i][m]==2) {fw[m]=3;printf("11");}
		if (d[i][m]==1) {
			if (fw[m]==1) printf("01");
			if (fw[m]==2) printf("10");
			if (fw[m]==3||fw[m]==0){printf("01");fw[m]=2;}
				else fw[m]=3-fw[m];
			}
		if (d[i][m]==0) printf("00");
		printf("\n");
		}
}
