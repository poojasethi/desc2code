#include <iostream>

using namespace std;

const int MAXN = 1001;

string ar[MAXN][MAXN];

int main() {
  int n, m;
  cin >> n >> m;
  int c11 = 0;
  int c00 = 0;
  int co = 0;
  for (int i = 0; i < n; ++i) {
    for (int k = 0; k < m; ++k) {
      string in; cin >> in;
      if (in == "11") {
        ++c11;
      } else if (in == "00") {
        ++c00;
      } else {
        ++co;
      }
      ar[i][k] = "00";
    }
  }
  int q = c11 / m;
  int r = c11 % m;
  for (int i = 0; i < q; ++i) {
    for (int k = 0; k < m; ++k) {
      ar[i][k] = "11";
    }
  }
  for (int i = 0; i < r; ++i) {
    ar[q][i] = "11";
  }
  string tmp = "01";
  if (co > 0) {
    for (int i = r; i < m; ++i) {
      ar[q][i] = tmp;
      --co;
      if (co == 0) break;
    }
  }
  if (co > 0) {
    tmp = "10";
    for (int i = q + 1; i < n; ++i) {
      for (int k = m - 1; k >= r; --k) {
        ar[i][k] = tmp;
        --co;
        if (co == 0) break;
      }
      if (co == 0) break;
      for (int k = 0; k < r; ++k) {
        ar[i][k] = tmp;
        --co;
        if (co == 0) break;
      }
      if (co == 0) break;
      if (tmp == "01") {
        tmp = "10";
      } else {
        tmp = "01";
      }
    }
  }
  for (int i = 0; i < n; ++i) {
    for (int k = 0; k < m; ++k) {
      cout << ar[i][k] << ' ';
    }
    cout << '\n';
  }
  return 0;
}
