#include <iostream>
#include <cstdio>
#include <string>
using namespace std;

int a[3];
int main()
{
    int n,m,i,j;
    string s;
    cin >> n >> m;
    for (i = 0; i < n; i++) {
        for (j = 0; j < m; j++) {
            cin >> s;
            if (s == "11") a[0]++;
            else if (s == "00") a[2]++;
            else a[1]++;
        }    
    }
    
    for (i = 0; i < n; i++) {
        for (j = 0; j < m; j++) {
            if (a[0]) {
               cout << "11 ";
               a[0]--;          
            } else if (a[1] >= m-j) {
              if (i%2) cout << "01 ";
              else cout << "10 "; 
              a[1]--;      
            } else cout << "00 ";
        }    
        cout << endl;
    }
    return 0;
}
