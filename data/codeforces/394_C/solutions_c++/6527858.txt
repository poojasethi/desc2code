#include <iostream>
using namespace std;
int main()
{
	int n,m,a=0,b=0,c=0,i,j,d1,d2;
	cin>>n>>m;
	int ch[n][2*m];
	char a1,a2;
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			cin>>a1>>a2;
			if(a1=='1'&&a2=='1')
				a++;
			else if((a1=='0'&&a2=='1')||(a1=='1'&&a2=='0'))
				b++;
			else c++;
		}
	}
	bool flag=true,f1=true;
	for(i=0;i<n;i++)
	{
		if(flag)
		{
			for(j=0;j<m;j++)
			{
				if(a)
					{ch[i][2*j]=ch[i][2*j+1]=1;a--;}
				else if(b)
				{
					if(f1)
						{ch[i][2*j]=1;ch[i][2*j+1]=0;b--;f1=false;}
					else
						{ch[i][2*j]=0;ch[i][2*j+1]=1;b--;f1=true;}
				}
				else
					ch[i][2*j]=ch[i][2*j+1]=0;
			}
			flag=false;
		}
		else
		{
			for(j=m-1;j>=0;j--)
			{
				if(a)
					{ch[i][2*j]=ch[i][2*j+1]=1;a--;}
				else if(b)
				{
					if(f1)
						{ch[i][2*j]=1;ch[i][2*j+1]=0;b--;f1=false;}
					else
						{ch[i][2*j]=0;ch[i][2*j+1]=1;b--;f1=true;}
				}
				else
					ch[i][2*j]=ch[i][2*j+1]=0;
			}
			flag=true;
		}
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
			cout<<ch[i][2*j]<<ch[i][2*j+1]<<" ";
		cout<<endl;
	}
}
