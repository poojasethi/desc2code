#include <iostream>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <utility>
#include <algorithm>
#include <cmath>

using namespace std;

int n,m;
int a,b;
int ans[1010][1010];
int d[1010];

int main() {
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		for (int j = 0,x; j < m; j++) {
			cin >> x;
			if (x == 11) b++;
			else if (x != 0) a++;
		}
	}
	int i;
	for (i = 0; i < n && b; i++) {
		for (int j = 0; j < m; j++) {
			if (b) {
				ans[i][j] = 3;
				b--;
			}
			else if (a) {
				ans[i][j] = d[j] + 1;
				d[j] = !d[j];
				a--;
			}
		}
	}
	for (; i < n && a; i++) {
		for (int j = m-1; j >= 0 && a; j--,a--) {
			ans[i][j] = d[j] + 1;
			d[j] = !d[j];
		}
	}
	for (i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) cout << ans[i][j]/2 << ans[i][j]%2 << ' ';
		cout << '\n';
	}
}
