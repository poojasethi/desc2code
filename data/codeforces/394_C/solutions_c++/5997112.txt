#include <cstdio>

using namespace std;

int n, m;
int a[3]; // 00= 0, 11= 1, 01= 10= 2
int res[1005][1005]; // 00= 0, 11= 1, 01= 2, 10= 3
bool sign[1005][1005];

int main(){
    int i, j, x, y;
    char c[5];
    scanf("%d%d", &n, &m);
    for(i=1;i<=n;i++){
        for(j=1;j<=m;j++){
            scanf("%s", c);
            if(c[0]=='0' && c[1]=='0') a[0]++;
            else if(c[0]=='1' && c[1]=='1') a[1]++;
            else a[2]++;
        }
    }

    x= 1; y= 1;
    for(i=1;i<=a[1];i++){
        res[x][y]= 1;
        sign[x][y]= true;
        if(i<a[1]){
            if(y==m){x++; y= 1;}
            else y++;
        }
    }
    x= n; y= 1;
    while(a[0]>0){
        if(sign[x][y]==false){
            sign[x][y]= true; a[0]--;
            res[x][y]= 0;
        }
        if(y==m){
            x--; y= 1;
        }
        else y++;
    }

    x= 0;
    for(i=1;i<=n;i++){
        for(j=1;j<=m;j++){
            if(!sign[i][j]){
                res[i][j]= x+2;
            }
        }
        x= (x+1)%2;
    }

    for(i=1;i<=n;i++){
        for(j=1;j<=m;j++){
            if(res[i][j]==0) printf("00 ");
            if(res[i][j]==1) printf("11 ");
            if(res[i][j]==2) printf("01 ");
            if(res[i][j]==3) printf("10 ");
        }
        printf("\n");
    }

    return 0;
}
