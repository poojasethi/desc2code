#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,A,B,C;
int f[N][N];
char p[2];
int main() {
	scanf("%d%d",&n,&m);
	For(i,1,n)For(j,1,m) {
		scanf("%s",p);
		if (p[0]>p[1]) swap(p[0],p[1]);
		if (p[0]=='1') A++;
		else if (p[1]=='0') B++;
		else C++;
	}
	For(i,1,n)For(j,1,m) {
		if (!A) goto yy1;
		f[i][j]=1; A--;
	}
	yy1:;
	for (;i<=n;i++) {
		for (;j<=m;j++) if (!f[i][j]) {
			if (!C) goto yy2;
			f[i][j]=2; C--;
			if (!C) goto yy2;
			if (i<n) f[i+1][j]=3,C--;
		}
		j=1;
	}
	yy2:;
	For(i,1,n){
		For(j,1,m) printf("%s ",f[i][j]==1?"11":f[i][j]==2?"01":f[i][j]==3?"10":"00");
		printf("\n");
	}
	return 0;
}
