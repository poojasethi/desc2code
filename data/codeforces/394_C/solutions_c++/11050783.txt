#include<bits/stdc++.h>
using namespace std;
int arr[1009][2*1009],n,m,d[1099]={0};
int main()
{
    int c1=0,c2=0,c3=0,c0=0;
    string str;cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        int a,b,c,d;
        for(int l=0,k=0;l<m;l++)
        {
            cin>>a;
            if(a==0)arr[i][k++]=0,arr[i][k++]=0,c0++;
            if(a==1)arr[i][k++]=0,arr[i][k++]=1,c1++;
            if(a==10)arr[i][k++]=1,arr[i][k++]=0,c2++;
            if(a==11)arr[i][k++]=1,arr[i][k++]=1,c3++;
        }
    }
    string a[n][m];
    int c21=c2+c1;
        for(int i=0;i<n;i++)
        {
            if(i%2==0)
                {
                    for(int l=0;l<m;l++){
                    if(c3)c3--,a[i][l]+="11 ";
                    else if(c21)c21--,a[i][l]+="10 ";
                    else if(c0)c0--,a[i][l]+="00 ";}

                }
                else
                    {
                        for(int l=m-1;l>=0;l--){
                        if(c3)c3--,a[i][l]+="11 ";
                        else if(c21)c21--,a[i][l]+="01 ";
                        else if(c0)c0--,a[i][l]+="00 ";}

                    }
        }

    for(int i=0;i<n;i++)
    {
        for(int l=0;l<m;l++)
            cout<<a[i][l]<<" ";

    }
}
