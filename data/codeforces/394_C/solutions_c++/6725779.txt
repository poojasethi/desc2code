#include <bits\stdc++.h>
using namespace std;

char a[1001][1001];

int sum[1001][2];

int main(){
    int n,m;
    while(scanf("%d%d",&n,&m)>0){
        int c,two=0,one=0;
        memset(a,0,sizeof(a));
        memset(sum,0,sizeof(sum));
        for(int i=1; i<=n; i++) for(int j=1; j<=m; j++){
            scanf("%d",&c);
            if(c==10||c==1) one++;
            else if(c==11) two++;
        }
        int x=1,y;
        for(int cnt=0; x<=n&&cnt<two; x++) for(y=1; y<=m&&cnt<two; y++,cnt++){
            a[x][y]=11;
            sum[y][0]++; sum[y][1]++;
        }
        if(x>1) x--;
        for(int cnt=0; x<=n&&cnt<one; x++) for(y=m; y&&cnt<one; y--) if(a[x][y]==0){
            if(sum[y][0]>sum[y][1]){
                a[x][y]=01; sum[y][1]++;
            }else{
                a[x][y]=10; sum[y][0]++;
            }
            cnt++;
        }
        for(x=1; x<=n; x++){
            for(y=1; y<=m; y++){
                if(a[x][y]<10) printf("0");
                printf("%d ",a[x][y]);
            }
            printf("\n");
        }
    }
    return 0;
}
