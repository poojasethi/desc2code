#include <bits/stdc++.h>
#define pp pair<int, int>
#define mp make_pair
#define pb push_back
#define MAX 1001

using namespace std;

typedef long long ll;

int ans[MAX][MAX];
string conv[4]={"11","10","01","00"};

int main() {
	int n, m;
	cin >> n >> m;
	int num1=0, num2=0, num3=0;
	for(int i=0; i<n; i++) {
		for(int j=0; j<m; j++) {
			string str;
			cin >> str;
			if(str=="11")
				num1++;
			else if(str=="10" || str=="01")
				num2++;
			else
				num3++;
		}
	}
	for(int i=0; i<n; i++) {
		for(int j=0; j<m; j++) {
			if(ans[i][j]!=0){
				cout << conv[ans[i][j]-1] << " ";
				continue;
			}
			if(num1!=0) {
				ans[i][j]=1;
				num1--;
			}
			else if(num2!=0) {
				if(i!=n-1 && num2>=2) {
					ans[i][j]=2;
					ans[i+1][j]=3;
					num2-=2;
				}
				else {
					ans[i][j]=2;
					num2--;
				}
			}
			else {
				ans[i][j]=4;
				num3--;
			}
			cout << conv[ans[i][j]-1] << " ";
		}
		cout << endl;
	}
}
