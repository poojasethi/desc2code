#include<bits/stdc++.h>
using namespace std;
//#define IN 1
int ans[1005][2010] = {0};
int main()
{
#ifdef IN
   freopen("input.txt","r",stdin);
   freopen("output.txt","w",stdout);
#endif // IN
    int zeros=0,one=0,dup = 0, nums=0;
    int n,m;
    cin >> n >> m;
    for (int i = 0; i<n; i++)
        for (int j = 0; j<m; j++)
    {
        string d;
        cin >> d;
        if (d[0]==d[1] && d[0]=='1') {nums+=2; dup++;}
        else
        if (d[0]==d[1] && d[0]=='0') { zeros++;}
        else {one++; nums++;}
    }

    int ads = m - dup % m; if (ads == m) ads = 0;
    bool sign = true;

    for (int i = 0; i<n; i++)
    {
        sign = !sign;
        if (ads)
        {
            for (int j = 1; j<=ads && one; j++)
                {if (sign) ans[i][j*2] = 1; else ans[i][j*2-1]=1; one--;}
            for (int j = ads+1; j<=m && dup; j++)
                {ans[i][j*2]=ans[i][j*2-1]=1; dup--;}
            ads = 0;
        }
        else if (one)
        {for (int j = 1; j<=m; j++)
        if (one) {if (sign) ans[i][j*2] = 1; else ans[i][j*2-1]=1; one--;}
        }
        else if (dup)
            for (int j = 1; j<=m; j++)
            if (dup) {ans[i][j*2]=ans[i][j*2-1]=1; dup--;}
    }


    for (int i = 0; i<n; i++)
    {
        for (int j = 1; j<=m; j++)
        cout << ans[i][j*2]<<ans[i][j*2-1]<<' ';
        cout << endl;
    }
    return 0;
}
