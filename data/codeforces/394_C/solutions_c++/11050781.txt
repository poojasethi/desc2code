#include <bits/stdc++.h>
using namespace std;

int n, m, cnt1, cnt2, num;
string s;

int main()
{
   //freopen("r.txt","r",stdin);
   scanf("%d %d",&n, &m);
   for(int i=0;i<n;i++)
   {
       for(int j=0;j<m;j++)
       {
           cin>>s;
           if(s=="11")
              cnt1++;
           if(s=="10" || s=="01")
             cnt2++;
       }
   }
   for(int i=0;i<n;i++)
   {
       string pr;
       if(i%2==0)
       {
           for(int j=0;j<m;j++)
           {
               if(cnt1>0)
                   cnt1--, pr+="11 ";
               else if(cnt2>0)
                   cnt2--, pr+="10 ";
               else
                  pr+="00 ";
           }
       }
       else
       {
           for(int j=m-1;j>=0;j--)
           {
               if(cnt1>0)
                   cnt1--, pr+="11 ";
               else if(cnt2>0)
                   cnt2--, pr+="10 ";
               else
                  pr+="00 ";
           }
           reverse(pr.begin(), pr.end()-1);
           /*for(int x=0;x<pr.size();x++)
               if( (pr[x]=='0' && pr[x+1]=='1') || (pr[x]=='1' && pr[x+1]=='0') )
                   swap(pr[x], pr[x+1]);*/
       }
       cout<<pr<<"\n";
   }

    return 0;
}
