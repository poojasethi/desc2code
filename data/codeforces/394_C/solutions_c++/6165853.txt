using namespace std;

typedef long long INT;
typedef pair <int, int> pii;
typedef pair <pii, int> p3i;

int ans[1010][1010];

int main() {
 	int i, j, n, m, k, a, b, c;
	char str[10];

	scanf("%d %d ", &n, &m);
	a=b=c=0;
	for (i=0; i<n; i++) for (j=0; j<m; j++) {
		scanf("%s ", str);
		if (str[0]=='1' && str[1]=='1') c++;
		else if (str[0]=='0' && str[1]=='0') a++;
		else b++;
	}
	
	for (k=2, i=0; i<n; i++, k=3-k) {
		if (i&1) {
			for (j=0; j<m; j++) {
				if (c) ans[i][j]=3, c--;
				else if (b) ans[i][j]=k, b--;
				else ans[i][j]=0;
			}
		} else {
			for (j=m-1; j>=0; j--) {
				if (c) ans[i][j]=3, c--;
				else if (b) ans[i][j]=k, b--;
				else ans[i][j]=0;
			}
		}
	}
	
	for (i=0; i<n; i++) {
		for (j=0; j<m; j++) {
			str[0]=ans[i][j]/2+'0';
			str[1]=ans[i][j]%2+'0';
			str[2]=0;
			printf("%s%c", str, j<m-1?' ':'\n');
			k++;
		}
	}
	
    return 0;
}
