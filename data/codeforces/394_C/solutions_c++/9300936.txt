#include <algorithm>
#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

const int MAXN = 1E3 + 10;

int N, M;
int repetitions[3];
int array[MAXN][MAXN];

int main(void) {
    char a, b;
    scanf("%d%d", &N, &M);
    for (int i = 0; i < N; i++) {
	for (int j = 0; j < M; j++) {
	    scanf(" %c%c", &a, &b);
	    //printf("%d\n", (a - '0') + (b - '0'));
	    repetitions[(a - '0') + (b - '0')]++;
	}
    }

    int p1 = 0, p2 = 0;
    for (int k = 2; k > 0; k--) {
	for (int i = 0; i < repetitions[k]; i++) {
	    if (p2 >= M) {
		p1++;
		p2 = M - 1;
	    }
	    if (p2 < 0) {
		p1++;
		p2 = 0;
	    }

	    //printf("(%d %d) -> %d\n", p1, p2, k);

	    array[p1][p2] = k;
	    if ((p1 % 2) == 0){ 
		p2++;
	    } else {
		p2--;
	    }
	}
    }

    for (int i = 0; i < N; i++) {
	for (int j = 0; j < M; j++) {
	    switch (array[i][j]) {
		case 2:
		    printf("11");
		    break;
		case 1:
		    if ((i % 2) == 0) {
			printf("10");
		    } else {
			printf("01");
		    }
		    break;
		case 0:
		    printf("00");
	    }
	    if (j == M - 1) {
		printf("\n");
	    } else {
		printf(" ");
	    }
	}
    }
    return 0;
}
