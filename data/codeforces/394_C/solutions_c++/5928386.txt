#include<stdio.h>
const int maxn=1050;
int n,m,s[5]={0},cs;
int cur[maxn]={0},ans[maxn][maxn];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,c;
	char a,b;
	scanf("%d%d\n",&n,&m);
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
		{
			scanf("%c%c",&a,&b);
			s[a+b-96]++;
			getchar();
		}
	cs=(s[2]+s[1]/2)/m;
	for(i=0;i<cs&&(s[1]||s[2]);i++)
		for(j=0;j<m;j++)
			if(s[2]) ans[cur[j]++][j]=3,s[2]--;
			else if(s[1]) ans[cur[j]++][j]=1,ans[cur[j]++][j]=2,s[1]-=2;
			else break;
	if(s[2])
		for(j%=m;s[2];j++,j%=m)
			ans[cur[j]++][j]=3,s[2]--;
	if(s[1])
		for(c=1;s[1];j++)
		{
			if(j>=m) j%=m,c^=3;
			ans[cur[j]++][j]=c,s[1]--;
			if(s[1]&&cur[j]<n) ans[cur[j]++][j]=c^3,s[1]--;
		}
	for(i=0;i<n;i++)
	{
		printf("%d%d",ans[i][0]>>1,ans[i][0]&1);
		for(j=1;j<m;j++)
			printf(" %d%d",ans[i][j]>>1,ans[i][j]&1);
		printf("\n");
	}
	return 0;
}
