#include<bits/stdc++.h>
using namespace std;
int n, m, one, two, O;
char s[3];

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d", &n, &m);
	for(int i=0 ; i<n ; ++i)
		for(int j=0 ; j<m ; ++j){
			scanf("%s", s);
			if(strcmp(s, "11") == 0) ++two;
			else if(strcmp(s, "00") == 0) ++O;
			else ++one;
		}
	for(int i=0 ; i<n ; ++i){
		for(int j=0 ; j<m ; ++j){
			if(j) putchar(' ');
			if(two) printf("11"), --two;
			else if(one+j>=m && (i&1)) printf("01"), --one;
			else if(one+j>=m) printf("10"), --one;
			else if(O) printf("00"), --O;
		}
		puts("");
	}
	return 0;
}
