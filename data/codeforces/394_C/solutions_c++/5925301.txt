#include<stdio.h>
#include<algorithm>

using namespace std;

int tab[1010][1010];

int main() {
  int m,n;
  int x;
  int a=0,b=0,c=0;
  scanf("%d%d", &n,&m);
  for (int i=0; i<n; i++) {
    for (int j=0; j<m; j++) {
      scanf("%d", &x);
      //printf("scan %d \n", x);
      tab[i][j] = -1;
      if      (x==11) { a++; }
      else if (x==10) { b++; }
      else if (x==01) { b++; }
      else if (x==00) { c++; }
    }
  }
  //printf("%d %d %d\n", a,b,c);
  int i=0;
  int j=0;
  while (i+1 < n && a>0 && c>0) {
    tab[i][j]   = 11;
    tab[i+1][j] = 00;
    a--; c--;
    j++; if (j>=m) { j=0; i+=2; }
  }
  while (i+1 < n && b>1) {
    tab[i][j]   = 10;
    tab[i+1][j] = 01;
    b-=2;
    j++; if (j>=m) { j=0; i+=2; }
  }



  while (b>0) {
    if (tab[i][j] == -1) {
      tab[i][j]   = 10;
      b--;
    }
    j++; if (j>=m) { j=0; i++; }
  }
  while (a>0) {
    if (tab[i][j] == -1) {
      tab[i][j] = 11;
      a--;
    }
    j++; if (j>=m) { j=0; i++; }
  }
  while (c>0) {
    if (tab[i][j] == -1) {
      tab[i][j]   = 00;
      c--;
    }
    j++; if (j>=m) { j=0; i++; }
  }

  for (int i=0; i<n; i++) {
    for (int j=0; j<m; j++) {
      printf("%02d ", tab[i][j]);
    }
    printf("\n");
  }


  return 0;
}
