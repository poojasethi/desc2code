#include <iostream>
using namespace std;

const int MAXN = 1005;
int n, m, a = 0, b = 0, c = 0;
int r[MAXN][MAXN] = {}, x = 0, y = 0;

void advance(){
  while(r[x][y]){
    y++;
    if(y == m){
      x++;
      y = 0;
    }
  }
}

int main(){
  ios::sync_with_stdio(false);
  cin >> n >> m;
  for(int i = 0; i < n; i++){
    for(int j = 0; j < m; j++){
      string s;
      cin >> s;
      if(s == "00") a++;
      else if(s == "11") b++;
      else c++;
    }
  }
  for(int i = 0; i < c / 2; i++){
    advance();
    r[x][y] = 1;
    r[x + 1][y] = 2;
  }
  for(int i = 0; i < b; i++){
    advance();
    r[x][y] = 3;
  }
  if(c % 2){
    advance();
    r[x][y] = 1;
  }
  for(int i = 0; i < n; i++){
    for(int j = 0; j < m; j++){
      cout << r[i][j] / 2 << r[i][j] % 2 << " ";
    }
    cout << '\n';
  }
}
