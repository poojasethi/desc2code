#include <iostream>

using namespace std;

const int MAXN = 1001;
int cnt[3];
string ans[MAXN][MAXN];

int main()
{
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
		{
			int x; cin >> x;
			if (x == 1 || x == 10) cnt[1]++;
			else if (x == 11) cnt[2]++;
			else cnt[0]++;
		}
	int cur = 0;
	for (int i = 0; i < n; i++)
	{
		if (i % 2)
		{
			for (int j = 0; j < m; j++)
			{
				if (cnt[2])
				{
					cnt[2]--;
					ans[i][j] = "11";
				}
				else if (cnt[1])
				{
					cnt[1]--;
					ans[i][j] = "01";
				}
				else
					ans[i][j] = "00";
			}
		}
		else
		{
			for (int j = m - 1; j >= 0; j--)
			{
				if (cnt[2])
				{
					cnt[2]--;
					ans[i][j] = "11";
				}
				else if (cnt[1])
				{
					cnt[1]--;
					ans[i][j] = "10";
				}
				else
					ans[i][j] = "00";
			}
		}
	}
	for (int i = 0; i < n; i++)
	{
		for (int j = 0; j < m; j++)
			cout << ans[i][j] << " ";
		cout << endl;
	}
	return 0;
}
