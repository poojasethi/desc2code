#include <iostream>
#include <string>

using namespace std;
int main(){
	int n,m;
	cin>>n>>m;
	int zo=0,zz=0,oo=0;
	string tmp;
	int abc=1;
	while(abc<=n*m){
		cin>>tmp;
		if(tmp=="10"||tmp=="01"){
			zo++;
		}else if(tmp=="11"){
			oo++;
		}else if(tmp=="00"){
			zz++;
		}
		abc++;
	}
	string array[n][m];
	if(n==1){
		for(int i=1;i<=zo;i++){
			cout<<"10 ";
		}
		for(int i=1;i<=oo;i++){
			cout<<"11 ";
		}
		for(int i=1;i<=zz;i++){
			cout<<"00 ";
		}
		cout<<endl;
		return 0;
	}
	for(int i=0;i<(n/2)*2;i){
		for(int j=0;j<m;j++){
			if(zz>0&&oo>0){
				zz--;oo--;
				array[i][j]="00";
				array[i+1][j]="11";
			}else if(zo>=2){
				array[i][j]="01";
				array[i+1][j]="10";
				zo-=2;
			}else{
				if(zz>0){array[i][j]="00";zz--;}
				else if(zo>0){array[i][j]="01";zo--;}
				else if(oo>0){array[i][j]="11";oo--;}
				if(zz>0){array[i+1][j]="00";zz--;}
				else if(zo>0){array[i+1][j]="01";zo--;}
				else if(oo>0){array[i+1][j]="11";oo--;}
			}
		}
		i+=2;
	}
	if(n%2==1){
		for(int j=0;j<m;j++){
			if(zz>0){array[n-1][j]="00";zz--;}
			else if(zo>0){array[n-1][j]="01";zo--;}
			else if(oo>0){array[n-1][j]="11";oo--;}
		}
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			cout<<array[i][j]<<" ";
		}
		cout<<endl;
	}
	return 0;
}
