#include<cstdio>
#include<cstring>
#include<cmath>
#include<iostream>
#include<algorithm>
#include<queue>
#include<vector>
using namespace std;
int a[1010][1010];
int main(){
    int n,m,i,j,c1,c2,c;
    while(cin>>n>>m)
    {
//        memset(a,0,sizeof(a));
        c1=c2=0;
        for(i=1;i<=n;i++)
            for(j=1;j<=m;j++){
                scanf("%d",&c);

                if(c==11)c2++;
                else if(c)c1++;
            }
        int x=1,y=1;
        for(i=1;i<=c2;i++)
        {
            a[x][y]=1;
            y++;
            if(y>m)y=1,x++;
        }
        y=m;
//        cout<<c1<<endl;
        for(i=1;i<=c1;i++)
        {
            if(a[x-1][y]==2)a[x][y]=3;
            else a[x][y]=2;
            y--;
            if(y<1)y=m,x++;
            if(a[x][y])y=m,x++;
        }
        for(i=1;i<=n;i++){
            for(j=1;j<=m;j++)
            {
                if(j!=1)cout<<' ';
//                cout<<a[i][j];
                if(a[i][j]==1)cout<<"11";
                else if(a[i][j]==2)cout<<"01";
                else if(a[i][j]==3)cout<<"10";
                else cout<<"00";
            }
            cout<<endl;
        }
    }
}
