#include <iostream>
#include <vector>
#include <algorithm>
#define MP make_pair
#define X first
#define Y second
#define PB push_back
#define F(i,a) for(int i=0;i<int((a).size());++i)
using namespace std;

int f(int n, int l1, int r1, int l2, int r2)
{
	if (l1==l2)
		return min(r2,r1)-l1+1;
	if (r1==r2)
		return r1-max(l1,l2)+1;
	vector<int> h;
	vector< pair<int,int> > v,u;
	int m=1<<(n-1);
	if (l1<=m && r1>=m && 
		l2<=m && r2>=m)
		h.PB(min(r1,r2)-max(l1,l2)+1);
	if (l1<m)
		v.PB(MP(l1,min(r1,m-1)));
	if (l2<m)
		u.PB(MP(l2,min(r2,m-1)));
	if (r1>m)
		v.PB(MP(max(l1,m+1)-m,r1-m));
	if (r2>m)
		u.PB(MP(max(l2,m+1)-m,r2-m));
	F(i,v) F(j,u) 
		h.PB(f(n-1,v[i].X,v[i].Y,u[j].X,u[j].Y));
	sort(h.begin(),h.end());
	return h.empty()?0:h.back();
}

int main()
{
	int l1,r1,l2,r2;
	cin >> l1 >> r1 >> l2 >> r2;
	cout << f(30,l1,r1,l2,r2) << endl;
	return 0;
}
