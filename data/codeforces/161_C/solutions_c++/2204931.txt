#include <cstdio>
#include <iostream>
#define  MAXN 30
using namespace std;

int A,B,X,Y,res;
int S1,S2,S3,S4;

int main()
{
	int i,len,tmp;
	
	scanf(" %d %d %d %d",&A,&B,&X,&Y);
	
	for(i=1;i<=MAXN;i++)
	{
		if((1<<i)<min(B-A,X-Y) && (A%(1<<i)<B%(1<<i) || X%(1<<i)<Y%(1<<i)))
			break;
			
		S1=((A-1+(1<<(i-1)))/(1<<i))*(1<<i)+(1<<(i-1));
		S2=S1+(1<<i);
		S3=((X-1+(1<<(i-1)))/(1<<i))*(1<<i)+(1<<(i-1));
		S4=S3+(1<<i);
		
		
		len=(1<<(i-1))-1;
		tmp=min(len,min(S1-A,S3-X))+min(len,min(B-S1,Y-S3));
		if(S2<=B)
			tmp=max(tmp,min(len,min(S2-A,S3-X))+min(len,min(B-S2,Y-S3)));
		if(S4<=Y)
			tmp=max(tmp,min(len,min(S1-A,S4-X))+min(len,min(B-S1,Y-S4)));
		if(S2<=B && S4<=Y)
			tmp=max(tmp,min(len,min(S2-A,S4-X))+min(len,min(B-S2,Y-S4)));
		res=max(res,tmp+1);
	}
	
	printf("%d\n",res);
	return 0;
}
