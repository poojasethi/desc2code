#include <cstdio>
#include <cstdlib>
#include <vector>
#include <algorithm>

using namespace std;

int l1, r1, l2, r2;
int res;

void find_min(const int l1, const int r1, const int l2, const int r2, const int c=1<<29)
{
    if (r1 - l1 + 1 <= res || r2 - l2 + 1 <= res)
        return ;
    if (c == 1)
    {
        res = max(res, 1);
        return;
    }
    res = max(res, min(r1, r2) - max(l1, l2) + 1);
    //printf(">>> l1:%d r1:%d l2:%d r2:%d res:%d c:%d\n", l1, r1, l2, r2, res, c);
    
    find_min(l1, min(c - 1, r1), l2, min(c - 1, r2), c / 2);
    find_min(max(1, l1 - c), r1 - c, max(1, l2 - c), r2 - c, c / 2);
    find_min(max(1, l1 - c), r1 - c, l2, min(c - 1, r2), c / 2);
    find_min(l1, min(c - 1, r1), max(1, l2 - c), r2 - c, c / 2);
}

int main(int argc, char *argv[])
{
    scanf("%d%d%d%d", &l1, &r1, &l2, &r2);
    res = 0;
    find_min(l1, r1, l2, r2);
    printf("%d\n", res);
    return 0;
}
