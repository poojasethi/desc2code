#include <cstdio>
#include <cstring>
inline void swap(int &a,int &b)
{int tmp=a;a=b;b=tmp;}
inline int max(int a,int b){return a>b?a:b;}
int calc(int l1,int r1,int l2,int r2,int k)
{
	//printf("%d %d %d %d %d\n",l1,r1,l2,r2,k);
	int mid=1<<(k-1);
	if(l1>r1||l2>r2)return 0;
	if(l1>l2)
	{
		swap(l1,l2);
		swap(r1,r2);
	}
	if(r1>=r2)return r2-l2+1;
	if((l1<=mid&&r1>=mid)||(l2<=mid&&r2>=mid))
	{
		if((l1<=mid&&r1>=mid)&&(l2<=mid&&r2>=mid))
		{
			return max(r1-l2+1,calc(l1,mid-1,1,r2-mid,k-1));
		}
		else if(l1<=mid&&r1>=mid)
		{
			if(l2>mid)
			{l2-=mid;r2-=mid;}
			return max(calc(l1,mid-1,l2,r2,k-1),calc(1,r1-mid,l2,r2,k-1));
		}
		else if(l2<=mid&&r2>=mid)
		{
			return max(calc(l2,mid-1,l1,r1,k-1),calc(1,r2-mid,l1,r1,k-1));
		}
	}
	else
	{
		if(l2>mid)
		{l2-=mid;r2-=mid;}
		if(l1>mid)
		{l1-=mid;r1-=mid;}
		return calc(l1,r1,l2,r2,k-1);
	}
}
int main()
{
	int l1,r1,l2,r2;
	scanf("%d%d%d%d",&l1,&r1,&l2,&r2);
	printf("%d\n",calc(l1,r1,l2,r2,30));
	scanf(" ");
}
