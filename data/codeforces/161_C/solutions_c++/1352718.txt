#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;
int fuck(int x,int y,int l,int r,int k)
{
	if(x>y || l>r) return 0;
	int mid=1<<k;
	if(l<x || (l==x && r>y)) return fuck(l,r,x,y,k);
	if(y>=r) return r-l+1;
	if(l>mid) return fuck(x,y,l-mid,r-mid,k);
	if(r<mid && y<mid) return fuck(x,y,l,r,k-1);	
	return max(y-l+1,max(fuck(x,y,l,mid-1,k),fuck(1,r-mid,x,y,k)));
}
int main()
{
	int x,y,l,r;
	scanf("%d %d %d %d",&x,&y,&l,&r);
	printf("%d\n",fuck(x,y,l,r,30));
//	system("pause");
	return 0;	
}
