#include <iostream>
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <string>
#include <map>

#include <vector>

#define fa a
#define fb (b >= mid ? mid - 1 : b)
#define fc c
#define fd (d >= mid ? mid - 1 : d)
#define ra (a <= mid ? 1 : a - mid)
#define rb b - mid
#define rc (c <= mid ? 1 : c - mid)
#define rd d - mid
using namespace std;

int a, b, c, d;
int len = (1 << 30) - 1, mid;
int ans;

void solve(int a, int b, int c, int d, int len){
	int mid = len / 2 + 1;
	if (b - a + 1 <= ans || d - c + 1 <= ans) return;
	ans = max(ans, min(b, d) - max(a, c) + 1);
	solve(fa, fb, fc, fd, len / 2);
	solve(fa, fb, rc, rd, len / 2);
	solve(ra, rb, fc, fd, len / 2);
	solve(ra, rb, rc, rd, len / 2);
}

int main(){
	cin >> a >> b >> c >> d;
	solve(a, b, c, d, len);
	cout << ans << endl;
}
