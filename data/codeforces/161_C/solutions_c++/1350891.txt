int rec(int x, int y, int l, int r, int k)
{
    int m=1<<k;
    //cout<<x<<" "<<y<<" "<<l<<" "<<r<<" "<<k<<endl;
    if ( l>r||x>y)
        return 0;
    if ( (l<x)||((l==x)&&(r>y)) )
        return rec(l,r,x,y,k);
    if ( y>=r)
        return r-l+1;
    if ( l > m)
        return rec(x,y,l-m,r-m,k);
    if ( y<m&&r<m)
        return rec(x,y,l,r,k-1);
    return max(y-l+1, max( rec(x,y,l,m-1,k), rec(1,r-m,x,y,k) ) );
}
