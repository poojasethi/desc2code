#include<cstdio>
#include<iostream>
using namespace std;
int ans=0;

void dfs(int l1,int r1,int l2,int r2,int dep){
    //printf("%d\n",dep);
    if(r1-l1+1<=ans||r2-l2+1<=ans)return ;
    int x=1<<(dep-1);
    if(l1>x)l1-=x,r1-=x;
    if(l2>x)l2-=x,r2-=x;
    int l=max(l1,l2);
    int r=min(r1,r2);
    if(r>=l)ans=max(ans,r-l+1);

    dfs(l1,min(r1,x-1),l2,min(r2,x-1),dep-1);
    if(r1>x)dfs(1,r1-x,l2,min(r2,x-1),dep-1);
    if(r2>x)dfs(l1,min(r1,x-1),1,r2-x,dep-1);
    if(r1>x&&r2>x)dfs(1,r1-x,1,r2-x,dep-1);

}
int main(){
    #ifdef DouBi
    freopen("in.cpp","r",stdin);
    #endif // DouBi
    int l1,r1,l2,r2;while(scanf("%d%d%d%d",&l1,&r1,&l2,&r2)!=EOF){
        ans=0;
        dfs(l1,r1,l2,r2,30);
        printf("%d\n",ans);
    }
    return 0;
}
