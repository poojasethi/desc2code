#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int rec(int a,int b,int c,int d,int m)
{
    if(a>c || (a==c && b>d)) return rec(c,d,a,b,m);
    if(b>=d) return d-c+1;
    if(a==c) return b-a+1;
    if(a>m) return rec(a-m,b-m,c-m,d-m,m>>1);
    if(c>m) return rec(a,b,c-m,d-m,m);
    if(d<m) return rec(a,b,c,d,m>>1);
    if(b<m) return max(rec(a,b,c,m-1,m>>1),rec(a,b,1,d-m,m>>1));
    return max(b-c+1,rec(a,m-1,1,d-m,m>>1));
}

int main()
{
    int a,b,c,d; const int inf=(1<<30);
    scanf("%d%d%d%d",&a,&b,&c,&d);
    printf("%d\n",rec(a,b,c,d,inf));
    return 0;
}
