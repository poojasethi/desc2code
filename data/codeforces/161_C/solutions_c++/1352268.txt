#include <iostream>
#include <stdio.h>

using namespace std;

int sol(int l1, int r1, int l2, int r2, int k)
{
    int m = 1 << k;
    if (l1 > r1 || l2 > r2) return 0;
    if (l2 < l1 || l2 == l1 && r2 > r1) return sol(l2, r2, l1, r1, k);
    if (r1 >= r2) return r2 - l2 + 1;
    if (l2 > m) return sol(l1, r1, l2 - m, r2 - m, k);
    if (r1 < m && r2 < m) return sol(l1, r1, l2, r2, k - 1);
    return max(max(sol(l1, r1, l2, m - 1, k), sol(1, r2 - m, l1, r1, k)), r1 - l2 + 1);
}

int main()
{
    int l1, r1, l2, r2;
    scanf("%d%d%d%d", &l1, &r1, &l2, &r2);
    printf("%d\n", sol(l1, r1, l2, r2, 29));
    return 0;
}
