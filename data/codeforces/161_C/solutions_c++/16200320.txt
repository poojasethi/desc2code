#include<bits/stdc++.h>
using namespace std;
int len;
void solve(int l1,int r1,int l2,int r2,int dep){
	if(l1>r1||l2>r2)return;
	if(r1-l1+1<=len||r2-l2+1<=len)return;
	int mid;
	while((1<<dep)>max(r1,r2))dep--;
	mid=1<<dep;
	if(l1>mid)l1-=mid,r1-=mid;
	if(l2>mid)l2-=mid,r2-=mid;
	int dl=max(l1,l2);
	int dr=min(r1,r2);
	if(dr-dl+1>len)len=dr-dl+1;
	if(max(r1,r2)<mid)solve(l1,r1,l2,r2,dep-1);
	else if(r1<mid&&r2>=mid)solve(l1,r1,l2,mid-1,dep-1),solve(l1,r1,1,r2-mid,dep-1);
	else if(r2<mid&&r1>=mid)solve(l1,mid-1,l2,r2,dep-1),solve(1,r1-mid,l2,r2,dep-1);
	else{
		solve(l1,mid-1,l2,mid-1,dep-1);
		solve(l1,mid-1,1,r2-mid,dep-1);
		solve(1,r1-mid,l2,mid-1,dep-1);
		solve(1,r1-mid,1,r2-mid,dep-1);
	}
}
int main(){
	int l1,r1,l2,r2;
	scanf("%d%d%d%d",&l1,&r1,&l2,&r2);solve(l1,r1,l2,r2,30);
	printf("%d",len);
	return 0;
}
