#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>
#include <string>
#include <map>

using namespace std;
typedef long long ll;



int a, b, c, d;

int calc(int a, int b, int c, int d, int k) {
    if (k < 0) return 0;

    if (a > b || c > d) return 0;
    int mid = 1 << k;
    int res = 0;
    res = max(res, min(b, d) - max(a, c) + 1);
    //if (k == 1) cout << a << b << c << d << endl;
    if (b - a < res || d - c < res) return res;
    res = max(res, calc(a, min(mid - 1, b), c, min(mid - 1, d), k - 1));
    res = max(res, calc(a, min(mid - 1, b), max(c - mid, 1), d - mid, k - 1));
    res = max(res, calc(max(a - mid, 1), b - mid, max(c - mid, 1), d - mid, k - 1));
    res = max(res, calc(max(a - mid, 1), b - mid, c, min(mid - 1, d), k - 1));
    return res;
}

int main() {
   // freopen("in", "r", stdin);
   // freopen("out", "w", stdout);
    while (cin >> a >> b >> c >> d) {
        cout << calc(a, b, c, d, 30) << endl;
    }
}
