#include <cstdio>
#include <iostream>
using namespace std;

int A,B,C,D;
int Res;

int main(){
	scanf("%d%d%d%d",&A,&B,&C,&D);
	A--; B--; C--; D--;
	
	for(int i=0,Ta,Tc,us;i<=30;i++){
		us=(1<<i);
		Ta=((A/us)+1)*us-1;
		Ta+=(Ta%(us*2)==us*2-1)*us;
		Tc=((C/us)+1)*us-1;
		Tc+=(Tc%(us*2)==us*2-1)*us;
		
		for(int ta=Ta,j=1;ta<=B && j<=2;ta+=us)
			if(!(ta%(us*2)==us*2-1)){
				for(int tc=Tc,k=1;tc<=D && k<=2;tc+=us)
					if(!(tc%(us*2)==us*2-1)){
						Res=max(Res,min(us,min(ta-A+1,tc-C+1))+min(us-1,min(B-ta,D-tc)));
						/*if(Res==379149396){
							printf("%d %d %d\n",i,ta,tc);
						}*/
						k++;
					}
				j++;
			}
	}

	printf("%d\n",Res);
	return 0;
}
