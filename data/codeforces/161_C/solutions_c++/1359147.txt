#include<iostream>
#include<algorithm>
using namespace std;

int a,b,c,d;

int work(int a,int b,int c,int d,int n){
    
    if(a>b||c>d) return 0;
    
    if(a>c||(a==c&&b<d)) return work(c,d,a,b,n);
    
    if(b>=d) return d-c+1;
    
    int k=(1<<n);
    
    if(c>k) return work(a,b,c-k,d-k,n);
    
    if(b<k&&d<k) return work(a,b,c,d,n-1);
    return max(b-c+1,max(work(a,b,c,k-1,n),work(1,d-k,a,b,n)));
}
int main(){
    
    cin>>a>>b>>c>>d;
    cout<<work(a,b,c,d,30)<<endl;
   // while(1);
}
