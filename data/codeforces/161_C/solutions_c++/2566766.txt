#include <stdio.h>
#include <iostream>
using namespace std;

int abacaba(int l1, int r1, int l2, int r2, int cnt) {
	if (r1 < l1 || r2 < l2) 
		return 0;

	int i = 1 << cnt;
	if (l1 > l2 || l1 == l2 && r1 < r2)
		return abacaba(l2, r2, l1, r1, cnt);
	if (r2 <= r1)
		return r2 - l2 + 1;
	if (l1 > i)
		return abacaba(l1 - i, r1 - i, l2 - i, r2 - i, cnt - 1);
	if (r2 < i)
		return abacaba(l1, r1, l2, r2, cnt - 1);
	if (l2 > i)
		return abacaba(l1, r1, l2 - i, r2 - i, cnt);
	return max(r1 - l2 + 1, max(abacaba(l1, r1, l2, i - 1, cnt), abacaba(l1, r1, 1, r2 - i, cnt)));

}
int main() {
	int l1, r1, l2, r2;

	scanf("%d %d %d %d", &l1, &r1, &l2, &r2);

	int m = abacaba(l1, r1, l2, r2, 30);
	printf("%d\n", m);
	return 0;
}
