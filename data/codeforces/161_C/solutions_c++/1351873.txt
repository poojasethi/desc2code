#include<cstdio>
#include<algorithm>
using namespace std;
#define	U	unsigned
#define	X(t) return f=max(f,t);
U f,l,r,a,b;
U p(U l,U r,U a,U b,U m){
	if(l+f>r||a+f>b)X(f);
	do{	for(;r<m&&b<m;)m>>=1;
		if(l>m){l-=m;r-=m;}
		if(a>m){a-=m;b-=m;}
	}while(!(l<=m&&m<=r||a<=m&&m<=b));
	if(l==a)X(min(r,b)-a+1);
	if(r==b)X(b-max(a,l)+1);
	if(b>r)swap(a,l),swap(b,r);
	if(b>=l)f=max(f,b-max(a,l)+1);
	if(r>=m&&b>=m){
		f=max(min(r,b)-max(a,l)+1,f);
		X(max(max(p(l,m-1,a,m-1,m),p(m+1,r,m+1,b,m)),max(p(l,m-1,m+1,b,m),p(m+1,r,a,m-1,m))));
	}
	X(max(p(a,b,l,m-1,m),p(a,b,m+1,r,m)));
}
int main(){
	scanf("%u%u%u%u",&l,&r,&a,&b);
	p(l,r,a,b,1<<30);
	return printf("%u\n",f),0;
}
