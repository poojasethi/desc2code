#include <stdio.h>
using namespace std;

#include <algorithm>
#include <assert.h>
int lo(int a1, int b1, int a2, int b2, int k) {
	//printf("%d %d %d %d %d\n", a1, b1, a2, b2, k);
	if (a2 < a1)
		return lo(a2,b2,a1,b1,k);
	if (b1 >= b2)
		return b2-a2+1;
	int m = (1<<k)-1;
	//if (a1 > m)
		//return lo(a1-m-1,b1-m-1,a2,b2,k);
	if (a2 > m)
		return lo(a1,b1,a2-m-1,b2-m-1,k);
	if (b1 < m && b2 < m)
		return lo(a1,b1,a2,b2,k-1);
	if (b1 >= m)
		return max(max(lo(a1,m-1,a2,b2,k), lo(m+1,b1,a2,b2,k)), min(b1,b2)-a2+1);
	if (b2 >= m)
		return max(max(lo(a2,m-1,a1,b1,k), lo(m+1,b2,a1,b1,k)), min(b1,b2)-a2+1);
	assert(0);
}

int main() {
	int a1,b1,a2,b2;
	scanf("%d %d %d %d", &a1,&b1,&a2,&b2);
	a1--;b1--;a2--;b2--;
	printf("%d\n", lo(a1,b1,a2,b2,30));
	return 0;
}
