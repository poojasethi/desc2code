typedef long long LL;

int go(int x, int y, int l, int r, int k) {
    int m = (1 << k);
    if (y < x || r < l) return 0;
    if (l < x || l == x && r > y) return go(l, r, x, y, k);
    if (r <= y) return r - l + 1;
    if (r < m) return go(x, y, l, r, k - 1);
    if (l > m) return go(x, y, l - m, r - m, k);
    return max(y - l + 1, max(go(x, y, l, m - 1, k), go(x, y, m + 1, r, k)));

}

int main() {

    //freopen("test.in", "r", stdin);

    int x, y, l, r;
    scanf("%d %d %d %d", &x, &y, &l, &r);
    printf("%d", go(x, y, l, r, 30));

    return 0;

}
