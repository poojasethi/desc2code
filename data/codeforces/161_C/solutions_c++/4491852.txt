#include <cstdio>

int l1,l2,r1,r2,n,r;
int rec( int a, int b, int c, int d, int lim )
{
    if( a>c || ( a==c && b>d ) ) return rec( c, d, a, b, lim );
    if( b >= d ) return ( d - c ) + 1 ;
    if( a == c ) return ( b - a ) + 1;
    if( a > lim ) return rec( a - lim, b - lim, c - lim, d - lim, lim / 2 );
    if( c > lim ) return rec( a, b, c - lim, d - lim, lim );
    if( d < lim ) return rec( a, b, c, d, lim/2 );
    if( b < lim ) return max( rec( a, b, c, lim - 1, lim/2 ), rec( a, b, 1, d - lim, lim/2 ) );
    return max( ( b - c ) + 1, rec( a, lim - 1, 1, d - lim, lim / 2 ) );
}
