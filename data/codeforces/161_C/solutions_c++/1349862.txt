#include<cstdlib>
#include<iostream>
using namespace std;

int l1,r1,l2,r2;

void decrease(int &l,int &r)
{
	for (int x=30;x>=0;x--)
	if ((1<<x)<l && r<(1<<(x+1))) l-=1<<x,r-=1<<x;
}

int solve(int l1,int r1,int l2,int r2)
{
	if (l1>r1 || l2>r2) return 0;
	decrease(l1,r1); decrease(l2,r2);
	if (l1<=l2 && r2<=r1) return r2-l2+1;
	if (l2<=l1 && r1<=r2) return r1-l1+1;
	int m1,m2,tmp=0,ret=0;
	for (m1=1;(m1<<1)<=r1;m1<<=1);
	for (m2=1;(m2<<1)<=r2;m2<<=1);
	if (m1==m2)
	{
		ret=min(r1,r2)-max(l1,l2)+1;
		if (l1<l2 && r1<r2) tmp=solve(l1,m1-1,m1+1,r2);
		else if (l2<l1 && r2<r1) tmp=solve(l2,m1-1,m1+1,r1);
		return max(ret,tmp);
	}
	if (m1<m2) return max(solve(l1,r1,l2,m2-1),solve(l1,r1,m2+1,r2));
	if (m1>m2) return max(solve(l1,m1-1,l2,r2),solve(m1+1,r1,l2,r2));
}

int main()
{	
	cin>> l1 >> r1 >> l2 >> r2;
	cout<< solve(l1,r1,l2,r2) <<endl;
}
