#include <iostream>
#include <algorithm>
using namespace std;

int main()
{
  long long i=1,c=0,l1,l2,r1,r2,t1,t2,minim=-1,maxim=0,k1=0,k2=0;
  cin >> l1 >> r1 >> l2 >> r2;
  while(c<30){
  if((r1/i)%2==1)
    t1=(r1/i)*i;
  else
    t1=(r1/i-1)*i;
  if((r2/i)%2==1)
    t2=(r2/i)*i;
  else
    t2=(r2/i-1)*i;
  k1=((t1/i)-2)*i;
  if(k1<l1)
    k1=t1;
  k2=((t2/i)-2)*i;
  if(k2<l2)
    k2=t2;
  minim= min(min(r1-t1,i-1),min(r2-t2,i-1))+1+min(min(t1-l1,i-1),min(t2-l2,i-1));
  minim=max(minim,min(min(r1-t1,i-1),min(r2-k2,i-1))+1+min(min(t1-l1,i-1),min(k2-l2,i-1)));
  minim=max(minim,min(min(r1-k1,i-1),min(r2-k2,i-1))+1+min(min(k1-l1,i-1),min(k2-l2,i-1)));
  minim=max(minim,min(min(r1-k1,i-1),min(r2-t2,i-1))+1+min(min(k1-l1,i-1),min(t2-l2,i-1)));
  maxim= max(minim,maxim);
  i*=2;c++;
  }
  cout << maxim;
  return 0;
}
