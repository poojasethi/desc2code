#include <algorithm>
#include <iostream>
#include <cstring>
#include <vector>
#include <cstdio>
#include <math.h>

using namespace std;

int solve(int l1, int r1, int l2, int r2, int k){
    if(l1 > r1 || l2 > r2)return 0;

    if( (l1 <= l2 && r2 <= r1) || (l2 <= l1 && r1 <= r2) )
       return min(r2-l2+1, r1-l1+1);

    int t = (1<<k) - 1;
    int m = (t>>1) + 1;
    int o = min(r1, r2)-max(l1, l2)+1;

    if( l1 <= m && m <= r1)
       return max(o, max(solve(l1, m-1, l2, r2, k), solve(m+1, r1, l2, r2, k)));
    if( l2 <= m && m <= r2)
       return max(o, max(solve(l1, r1, l2, m-1, k), solve(l1, r1, m+1, r2, k)));

    if(l1 > m) l1 -= m, r1 -= m;
    if(l2 > m) l2 -= m, r2 -= m;
    return solve(l1, r1, l2, r2, k-1);
}


int main(){

    int a, b, c, d;
    scanf("%d%d%d%d", &a, &b, &c, &d);
    printf("%d\n", solve(a, b, c, d, 30) );
    return 0;
}
