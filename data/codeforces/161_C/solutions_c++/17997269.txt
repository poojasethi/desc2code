#include<bits/stdc++.h>
using namespace std;


int cos(int x,int y,int l,int r,int k)
{
    int m=1<<k;
    if(l>r||x>y)
    {
        return 0;
    }
    if(l<x||l==x&&r>y)
    {
        return cos(l,r,x,y,k);
    }
    if(y>=r)
    {
        return r-l+1;
    }
    if(l>m)
    {
        return cos(x,y,l-m,r-m,k);
    }
    if(y<m&&r<m)
    {
        return cos(x,y,l,r,k-1);
    }
    return max(max(cos(x,y,l,m-1,k),cos(1,r-m,x,y,k)),y-l+1);

}
int main()
{
    int a,b,c,d,x;
    cin>>a>>b>>c>>d;
    x=cos(a,b,c,d,30);
    cout<<x;
    return 0;
}
