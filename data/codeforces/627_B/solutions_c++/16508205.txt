#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int w[200010],c[200010][2];

inline int lowbit(int x) {return x&(-x);}

void change(int x,int y,int z,int n)
{
	while (x<=n)
	{
		c[x][y]+=z;x+=lowbit(x);
	}
}

int query(int x,int y)
{
	int z=0;
	while (x)
	{
		z+=c[x][y];x-=lowbit(x);
	}
	return z;
}

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n,k,a,b,q;scanf("%d%d%d%d%d",&n,&k,&a,&b,&q);
	memset(w,0,sizeof(w));
	memset(c,0,sizeof(c));
	while (q--)
	{
		int op,x,y;scanf("%d%d",&op,&x);
		if (op==1)
		{
			scanf("%d",&y);
			change(x,0,max(0,min(w[x]+y,b)-w[x]),n);
			change(x,1,max(0,min(w[x]+y,a)-w[x]),n);
			w[x]=min(w[x]+y,a);
		}
		else
			printf("%d\n",query(x-1,0)+query(n,1)-query(x+k-1,1));
	}
	return 0;
}
