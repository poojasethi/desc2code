#include<bits/stdc++.h>

using namespace std;

#define sgmn int v, int tl, int tr
#define tm  (tl + tr >> 1)
#define sol v*2, tl, tm
#define sag v*2+1, tm+1, tr
#define N 300005

int s[4*N][2],d,n,k,a,b,w,h,x,asd,l,r;

void up(sgmn){
	if(tl == tr)
	{  s[v][0] = min(a , s[v][0] + x);  s[v][1] = min(b , s[v][1] + x);  return;  }
	if(r <= tm)
		up(sol);
	else
		up(sag);
	s[v][0] = s[v*2][0] + s[v*2+1][0];
	s[v][1] = s[v*2][1] + s[v*2+1][1];
}

int qry(sgmn){
	if(l > r || tl > r || tr < l)
	   return 0;
	if(tl >= l && tr <= r)
	   return s[v][h];
	return qry(sol) + qry(sag);
}

int main(){

	scanf("%d%d%d%d%d",&n,&k,&a,&b,&w);

	while(w--){

		scanf("%d%d",&d,&r);

		if(d == 1){
			scanf("%d",&x);
			up(1,1,n);
		}

		else{

			h = 1;
			l = 1;
			r--;
			asd = qry(1,1,n);

			h = 0;
			l = r+k+1;
			r = n;

			printf("%d\n", asd + qry(1,1,n) );

		}
	}

	return 0;
}
