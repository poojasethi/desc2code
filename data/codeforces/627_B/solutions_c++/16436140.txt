#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 2000000 + 10;

int n, k, a, b, q;
int cnt[2][N];
int value[N];

void modify(int who, int p, int x)
{
	for( ; p <= n; p += p & -p) cnt[who][p] += x;
}

int query(int who, int p)
{
	int ret = 0;
	for( ; p; p -= p & -p) ret += cnt[who][p];
	return ret;
}

void solve()
{
	cin >> n >> k >> a >> b >> q;
	for( ; q --; ) {
		int opt;
		scanf("%d", &opt);
		if (opt == 1) {
			int d, x;
			scanf("%d%d", &d, &x);
			modify(0, d, -min(value[d], b));
			modify(1, n + 1 - d, -min(value[d], a));
			value[d] += x;
			modify(0, d, min(value[d], b));
			modify(1, n + 1 - d, min(value[d], a));
		} else {
			int d;
			scanf("%d", &d);
			printf("%d\n", query(0, d - 1) + query(1, n + 1 - (d + k)));
		}
	}
}

int main()
{
	solve();
	return 0;
}
