#include <iostream>
#include <cmath>

using namespace std;

typedef long long int ll;

const int primero=1<<19;
const int tope=1<<20;

void anyade(int pos,ll val,ll cota,ll suma[tope])
{
  pos+=primero;
  suma[pos]=min(cota,suma[pos]+val);
  while (pos>1) {
    pos/=2;
    suma[pos]=suma[2*pos]+suma[2*pos+1];
  }
}

ll sumale(int pos,ll suma[tope])
{
  pos+=primero;
  ll s=0;
  while (pos>1) {
    if (pos%2==1) s+=suma[pos-1];
    pos/=2;
  }
  return s;
}

ll sumari(int pos,ll suma[tope])
{
  pos+=primero;
  ll s=suma[pos];
  while (pos>1) {
    if (pos%2==0) s+=suma[pos+1];
    pos/=2;
  }
  return s;
}

ll cota[2];
ll suma[2][tope];
int n,k,q;

int main()
{
  ios::sync_with_stdio(false);
  cin>>n>>k>>cota[1]>>cota[0]>>q;
  for (int cas=0;cas<q;cas++) {
    int t;
    cin>>t;
    if (t==1) {
      int d,val;
      cin>>d>>val;
      anyade(d,val,cota[0],suma[0]);
      anyade(d,val,cota[1],suma[1]);
    } else {
      int d;
      cin>>d;
      cout<<sumale(d,suma[0])+sumari(d+k,suma[1])<<endl;
    }
  }
}
