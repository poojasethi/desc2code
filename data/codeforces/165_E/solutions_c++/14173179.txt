#include <cstdio>
#include <algorithm>
#define m 23
using namespace std;
int a[1000005],b,c,d,e,f[10000005],g,h,i,j,k,l,n;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++){
		f[a[i]^((1<<m)-1)]=a[i];
	}
	for(i=(1<<m)-1;i>=0;i--){
		for(j=0;j<m;j++)if(f[i|(1<<j)])f[i]=f[i|(1<<j)];
	}
	for(i=1;i<=n;i++)if(f[a[i]])printf("%d ",f[a[i]]);else printf("-1 ");
	return 0;
}
