#include <cstdio>

using namespace std;

const int maxl = 22;
int F[1 << maxl], A[1 << maxl];

int main()
{
  int n; scanf("%d", &n);
  
  for (int i = 0; i < n; ++i) {
    scanf("%d", &A[i]);
    F[A[i]] = A[i];
  }
  
  int e = (1 << maxl) - 1;
  
  for (int i = 1; i <= e; ++i) {
    for (int j = 0; j < maxl; ++j) if ((i & (1 << j)) && F[i - (1 << j)]) {
      F[i] = F[i - (1 << j)];
    }
  }
  
  for (int i = 0; i < n; ++i) {
    printf("%d ", F[e ^ A[i]] ? F[e ^ A[i]] : -1);
  }
}
