#include <cstdio>

int lim=1<<22,n,a[1000005],vis[4500005]; 

int main(){
    scanf("%d",&n);
    for(int i=1;i<=n;++i)scanf("%d",a+i),vis[a[i]]=a[i];
    for(int i=0;i<lim;++i)
    for(int j=1;j<lim;j<<=1)
    if(i&j && vis[i^j])vis[i]=vis[i^j];
    --lim;
    for(int i=1;i<=n;++i)
    if(vis[lim^a[i]])printf("%d ",vis[lim^a[i]]);else printf("-1 ");
}
