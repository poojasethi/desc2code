#include <cstdio>
#include <cstring>
using namespace std;
const int M=1<<22;
int n,a[M],v[M];
int main(void) {
  scanf("%d",&n);
  memset(v,0xff,sizeof(v));
  for(int i=0;i<n&&1==scanf("%d",&a[i]);++i)v[(M-1)&~a[i]]=a[i];
  for(int i=M;i-->0;)if(v[i]>=0)for(int j=i,k;j>0;j-=k)v[i^(k=j&-j)]=v[i];
  for(int i=0;i<n;++i)printf(i>0?" %d":"%d",v[a[i]]);puts("");
  return 0;
}
