#include <iostream>
#define pw(x) (1ull<<(x))
using namespace std;

int n,a[1111111],f[5555555],e;


int main(){
  cin >> n;
  for(int i=0;i<n;i++) cin >> a[i];
  for(int i=0;i<n;i++) f[a[i]]=a[i];
  e = pw(22)-1;
  for(int o=1;o<=e;o++)
    for(int i=0;i<22;i++)
      if((o&pw(i)) && f[o-pw(i)]) f[o]=f[o-pw(i)];
  
  for(int i=0;i<n;i++){
    if(f[e^a[i]]) cout << f[e^a[i]]; else cout << -1;
    if( i<(n-1) )cout << " "; cout << endl;
  }

  return 0;
}
