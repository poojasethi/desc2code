#include<cstdio>
#include<algorithm>
const int m=(1<<22)-1;
int a[m],f[m+1],n,i,j;
int main()
{
    for(scanf("%d",&n),i=1;i<=n;i++){scanf("%d",a+i);f[a[i]]=a[i];}
    for(i=0;i<=m;i++){f[i]=(f[i]==0)?-1:f[i];for(j=i;j;j-=j&(-j))f[i]=std::max(f[i],f[i-(j&(-j))]);}
    for(i=1;i<=n;i++)printf("%d ",f[m^a[i]]);
}
