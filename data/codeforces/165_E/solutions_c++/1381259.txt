#include <cstdio>
#define N 1000005
#define M (1 << 22)
#define FI(a, b, c) for(int a = b; a <= c; a++)

int n, a[N], d[M + 1];

int main(){
	FI(i, 0, M) d[i] = -1;
	
	scanf("%d", &n);
	FI(i, 1, n) scanf("%d", &a[i]), d[a[i]] = a[i];
	
	FI(i, 1, M) if(d[i] == -1){
		FI(j, 0, 22) if((i & 1 << j) && d[i ^ 1 << j] != -1){
			d[i] = d[i ^ 1 << j];
			break;
		}
	}
	
	FI(i, 1, n) printf("%d%c", d[M - 1 ^ a[i]], i == n ? '\n' : ' ');
}
