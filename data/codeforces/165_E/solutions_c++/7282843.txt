#include <cstdio>
#include <cstring>

using namespace std;

int n,a[1000010],f[1<<22];

int main()
{
	scanf("%d",&n);a[0]=-1;
	for(int i=1;i<=n;i++) scanf("%d",&a[i]),f[a[i]]=i;
	for(int i=1;i<(1<<22);i++)
		for(int j=0;j<22 && !f[i];j++) if((i&(1<<j)) && f[i^(1<<j)])
			f[i]=f[i^(1<<j)];
	for(int i=1;i<=n;i++) printf("%d ",a[f[((1<<22)-1)&~a[i]]]);
	return puts(""),0;
}
