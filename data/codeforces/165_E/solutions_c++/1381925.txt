#include<iostream>
#define maxn 10010000
using namespace std;

int n,i,j,a[maxn],d[maxn*4];

int main()
{
	cin >> n;
	for(i=1;i<=n;++i)
	{
		cin >> a[i];
		d[a[i]^((1<<22)-1)]=a[i];
	}
	for(i=(1<<22)-1;i>=0;--i)
		if (!d[i])
			for(j=0;j<22;++j)
				if (d[i|(1<<j)])
				{
					d[i]=d[i|(1<<j)];
					break;
				}
	for(i=1;i<=n;++i)
		if (d[a[i]])
			cout << d[a[i]] << ' ';
		else
			cout << "-1 ";
	cout << endl;
	return 0;
}
