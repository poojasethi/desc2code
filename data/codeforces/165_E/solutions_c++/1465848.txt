#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 1<<22;

int n, a[N], dp[N];

int main() {
	memset(dp, 0xFF, sizeof(dp));
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", a+i);
		dp[a[i]] = a[i];
	}
	for (int i = 0; i < N; ++i)
		for (int j = 1; j < N; j<<=1)
			if (i&j) dp[i] = max(dp[i], dp[i^j]);
	for (int i = 0; i < n; ++i)
		printf("%d%c", dp[N-1^a[i]], i<n-1?' ':'\n');
}
