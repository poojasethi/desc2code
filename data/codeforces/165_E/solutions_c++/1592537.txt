#include <stdio.h>
#include <string.h>
const int MAXN=1<<20;
const int BITS=22;
int a[MAXN], dp[(1<<BITS)+10],n;
int main()
{
	memset(dp,-1,sizeof(dp));
	scanf("%d",&n);
	for (int i=0; i<n; i++) {
		scanf("%d",a+i);
		dp[a[i]]=a[i];
	}
	for (int i=1; i<(1<<BITS); i++) {
		if (dp[i]>0) {
			for (int j=0; j<BITS; j++)
				if (dp[i|(1<<j)]==-1)
					dp[i|(1<<j)]=dp[i];
		}
	}
	for (int i=0; i<n; i++) {
		int t=((1<<BITS)-1)^a[i];
		printf("%d ",dp[t]);
	}
	return 0;
}
