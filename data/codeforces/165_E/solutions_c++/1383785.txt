#include <cstdio>
#include <cstring>
using namespace std;

int n;
int a[1000000];
int f[1<<22];

int main() {
	scanf("%d",&n);
	memset(f,-1,sizeof(f));
	for (int i=0;i<n;i++) {
		scanf("%d",&a[i]);
		f[a[i]]=i;
	}
	for (int x=1;x<(1<<22);x++) {
		for (int i=0;i<22;i++) {
			if (f[x]==-1) f[x]=f[x&~(1<<i)];
		}
	}
	for (int i=0;i<n;i++) {
		int x=a[i]^((1<<22)-1);
		if (f[x]==-1) printf("-1 ");
		else printf("%d ",a[f[x]]);
	}
	printf("\n");
    return 0;
}
