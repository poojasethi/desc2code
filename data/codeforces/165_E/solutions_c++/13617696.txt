#include <stdio.h>
#include <iostream>
#include <algorithm>
#include <memory.h>
using namespace std;
int dp[1<<22];
const int m = (1<<22)-1;
int main(){
	int n,i,j,y;
	cin>>n;
	int ar[1000000];
	memset(dp,-1,sizeof(dp));
	for(i=0;i<n;++i){
		scanf("%d",&ar[i]);
		dp[ar[i]]= ar[i];
	}
	for(i=1;i<=m;++i)
		if(dp[i]!=-1)
		for(y= 1;y<=m;y<<=1){
			if(((i ^y ) & dp[i])==dp[i]){
				dp[i^y]= dp[i];
			}
		}
	for(i=0;i<n;++i){
		y = (~ar[i])&m;
		printf("%d ",dp[y]);
	}
}
