#include <cstdio>
#include <cstring>

using namespace std;

int f[1<<22];
int n,a[1000010];

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]),f[a[i]]=a[i];
	for(int i=0;i<(1<<22);i++) if(f[i])
		for(int j=0;j<22;j++) if(!(i&(1<<j)) && !f[i|(1<<j)])
			f[i|(1<<j)]=f[i];
	for(int i=1;i<=n;i++){
		int t=~a[i] & ((1<<22)-1);
		if(!f[t]) printf("-1 ");
		else printf("%d ",f[t]);
	}
	puts("");
	return 0;
}
