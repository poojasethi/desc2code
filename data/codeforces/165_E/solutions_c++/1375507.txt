#include <cstdio>
#include <cstring>
using namespace std;

const int N = 1 << 24;
int f[N], a[N];

int main() {	
	int n;

	memset(f, -1, sizeof(f));
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d", &a[i]);
		f[(N - 1) & ~a[i]] = a[i];
	}
	for (int i = N - 1; i > 0; i--) {
		if (~f[i]) {
			for (int j = i, k; j > 0; j -= k) {
				k = j & (-j);
				f[i ^ k] = f[i];
			}
		}
	}
	for (int i = 0; i < n; i++) {
		printf("%d ", f[a[i]]);
	}

	return 0;
}
