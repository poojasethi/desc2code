#include <stdio.h>
#include <vector>
#include <memory.h>
#include <bits/stdc++.h>
using namespace std;
int a[1000000],dp[1<<24];
const int M = (1<<22)-1;
int main(){
	int n;scanf("%d",&n);memset(dp,-1,sizeof(dp));
	for(int i=0;i<n;++i){
		scanf("%d",&a[i]);
		dp[~a[i]&M]=a[i];
	}
	for(int i=M;i>=0;--i)
		if(dp[i]!=-1){
			for(int j=0;j<=22;++j){
				if(  (i&(1<<j) ) !=0 ){
					dp[i^(1<<j)]= dp[i];
				}
			}
		}
	for(int i=0;i<n;++i)
		cout<<dp[a[i]]<<" ";
	cout<<endl;
}
