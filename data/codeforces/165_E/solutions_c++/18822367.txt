#include <iostream>
#include <cstdio>
int a[1000010],f[1<<22],i,j,n,num,ans;
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d\n",&n);num=(1<<22)-1;
	for (i=1;i<=n;++i) scanf("%d",&a[i]),f[a[i]]=1;
	for (i=1;i<=num;++i)
		for (j=i;j;j-=j&-j) f[i]|=f[j&-j^i];
	for (i=1;i<=n;++i)
	{
		ans=num^a[i];
		if (!f[ans]) ans=-1;
		else
		for (j=ans;j;j-=j&-j)
			if (f[j&-j^ans]) ans^=j&-j;
		printf("%d ",ans);
	}
}
