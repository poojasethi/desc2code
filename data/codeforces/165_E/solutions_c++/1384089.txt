#include<iostream>
#include<algorithm>
using namespace std;
#define FOR(i,j,k) for(i=j;i<=k;i++)
const int m=(1<<22)-1;
int a[m],f[m+1],n,i,j;
int main()
{
    FOR(i,0,m)f[i]=-1;
    cin>>n;
    FOR(i,1,n){cin>>a[i];f[a[i]]=a[i];}
    FOR(i,0,m)for(j=i;j;j-=j&(-j))f[i]=max(f[i],f[i-(j&(-j))]);
    FOR(i,1,n)cout<<f[m^a[i]]<<" ";
}
