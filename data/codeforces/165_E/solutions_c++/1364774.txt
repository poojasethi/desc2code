#include<stdio.h>
int f[5000001],a[1000001];
int main()
{
	int n,i,j;
	scanf("%d",&n);
	for(i=1;i<=n;++i)
	{
		scanf("%d",&a[i]);
		f[a[i]^((1<<22)-1)]=a[i];
	}
	for(i=(1<<22)-1;i>=0;--i)
		if(!f[i])
			for(j=0;j<22;++j)
				if(f[i|(1<<j)])
				{
					f[i]=f[i|(1<<j)];
					break;
				}
	for(i=1;i<=n;++i)
		if(f[a[i]])
			printf("%d ",f[a[i]]);
		else
			printf("-1 ");
	return 0;
}
