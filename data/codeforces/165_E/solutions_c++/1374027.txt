#include <iostream>
#include <memory.h>
using namespace std;
int dp[1 << 22];
int in[1000001];
int main(){
	int n;
	cin >> n;
	memset(dp, -1, sizeof dp);
	for (int i = 0;i < n;++i){
		cin >> in[i];
		dp[in[i]] = in[i];
	}
	for (int j = 0;j < 1 << 22;++j){	
		for (int i = 0;i < 22;++i){
			if (j & (1 << i) &&  dp[j - (1 << i)] != -1)
				dp[j] = dp[j - (1 << i)];
		}
	}
	for (int i = 0;i < n;++i)
		cout << dp[in[i] ^ ((1 << 22) - 1)] << " ";
	cout << endl;
	return 0;
}
