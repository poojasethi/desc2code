#include<stdio.h>
#include<iostream>
#include<string.h>
#define N 1<<22
using namespace std;
int dp[1<<22],num[1000005];
int main()
{
int n,i,j;
scanf("%d",&n);
for(i=0;i<N;i++) dp[i]=-1;
for(i=1;i<=n;i++)
   {
     scanf("%d",&num[i]);
      dp[num[i]]=num[i];
    }
for(i=0;i< 1<<22;i++)
  for(j=0;j<22;j++)
     if(i&(1<<j)&&dp[i-(1<<j)]!=-1)
       dp[i]=dp[i-(1<<j)];
for(i=1;i<=n;i++)
  cout<<dp[num[i]^((1<<22)-1)]<<" ";
cout<<endl;
return 0;
}
