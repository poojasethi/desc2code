#include<stdio.h>
#define MAX (1<<22)
int i,j,n,ans[MAX],t[1000000],p;
main(){
	scanf("%d",&n);
	for(i=0;i<MAX;i++)ans[i]=-1;
	for(i=0;i<n;i++){
		scanf("%d",&t[i]);
		ans[t[i]]=t[i];
	}
	for(i=0;i<MAX;i++)
		if(ans[i]<0)
			for(j=0;j<22;j++)
				if(((i>>j)&1)&&ans[i^(1<<j)]>=0)ans[i]=ans[i^(1<<j)];
	for(i=0;i<n;i++)printf("%d ",ans[MAX-t[i]-1]);
}
