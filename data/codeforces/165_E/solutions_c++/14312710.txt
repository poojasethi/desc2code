#include <bits/stdc++.h>

using namespace std;

const int MAXN = 1e6+123;
const int asd = 4194303;

int N,ar[MAXN],inc[MAXN*10];

int main(){
	
	scanf(" %d",&N);
	for(int i=1;i<=N;i++){
		scanf(" %d",&ar[i]);
		inc[ar[i]]=i;
	}
	
	for(int i=0;i<(1<<22);i++)
		for(int j=0;j<22;j++)
			if(!inc[i|(1<<j)])
				inc[i|(1<<j)]=inc[i];
	
	for(int i=1;i<=N;i++)
		if(inc[ar[i]^asd]) printf("%d ",ar[inc[ar[i]^asd]]);
		else printf("-1 ");
	printf("\n");
	
	return 0;
}
