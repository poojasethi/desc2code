#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int dp[1<<22];
int a[1000001];

int main(){

  memset(dp,-1,sizeof(dp));
  int n;
  cin>>n;
  for(int i=0;i<n;i++){
    cin>>a[i];
    dp[a[i]]=a[i];
  }
  for(int i=1;i<(1<<22);i++)
    for(int j=0;j<22;j++)
      if(((i>>j)&1)&&(dp[i^(1<<j)]>=0))dp[i]=dp[i^(1<<j)];
  for(int i=0;i<n;i++){
    cout<<dp[(1<<22)-1-a[i]];
    if(i==n-1)cout<<endl;
    else cout<<" ";
  }

  return 0;
}
