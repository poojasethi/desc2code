#include<bits/stdc++.h>

using namespace std;

#define N 100005

// I wish I could solve this problem on my own :(

int nxt[N],stp[N],jmp[N],n,w,i,h,x,y,asd,kok = 333;

void up(int x){
	if(nxt[x] > n || nxt[x] / kok > x / kok){
		jmp[x] = nxt[x];
		stp[x] = 1;
	}
	else{
		jmp[x] = jmp[ nxt[x] ];
		stp[x] = stp[ nxt[x] ] + 1;
	}
}

int main(){

	cin >> n >> w;

	for(i=1;i<=n;i++){
		scanf("%d",&nxt[i]);
		nxt[i] += i;
	}

	for(i=n; i ; i--)
		up(i);

	while(w--){

		scanf("%d%d",&h,&x);

		if(h){
			for(asd = 0;jmp[x]<=n;x = jmp[x])
				asd += stp[x];

			asd += stp[x];

			for(;nxt[x] <= n;x = nxt[x]);

			printf("%d %d\n",x,asd); 
		}

		else{
			scanf("%d",&y);
			nxt[x] = x+y;
			i = x / kok * kok;
			for(;x >= i; x--)
				up(x);
		}
	}

	return 0;
}
