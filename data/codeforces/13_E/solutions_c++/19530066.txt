#include <bits/stdc++.h>

#define fi first
#define se second

using namespace std;

const int N = (int)1e5+9;

int n,m,sqrtt,i;
int a[N],las[N],f[N],type,x,y,last,cou;

int main(){
	//ios_base::sync_with_stdio(false);cin.tie(NULL);
	//freopen("spring.inp","r",stdin);
	scanf("%d%d", &n, &m);
	sqrtt=sqrt(n);
	for(i=0;i<n;i++) scanf("%d", &a[i]);
	for(i=n-1;i>=0;i--){
		if(i+a[i]>=n || i/sqrtt!=(i+a[i])/sqrtt) las[i]=i,f[i]=0;
		else las[i]=las[i+a[i]],f[i]=f[i+a[i]]+1;
	}
	while(m--){
		scanf("%d", &type);
		if(type){
			scanf("%d", &x);--x;
			cou=0;
			for(;x<n;last=las[x],x=last+a[last]) cou+=f[x]+1;
			printf("%d %d\n", last+1,cou);
		}else{
			scanf("%d%d", &x, &y);--x;
			a[x]=y;
			for(i=x;i>=0 && i/sqrtt==x/sqrtt;i--){
				if(i+a[i]>=n || i/sqrtt!=(i+a[i])/sqrtt) las[i]=i,f[i]=0;
				else las[i]=las[i+a[i]],f[i]=f[i+a[i]]+1;
			}
		}
	}
}
