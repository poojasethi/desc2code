#include <iostream>
#include <stdio.h>

using namespace std;

int N, M, a[100010], b[100010][2], t, K;
void fix(int i){
	if(i + a[i] >= N)b[i][0] = 0, b[i][1] = i;
	else
		if(i / K != (i + a[i]) / K) b[i][0] = 1, b[i][1] = i + a[i];
		else b[i][0] = 1 + b[i + a[i]][0], b[i][1] = b[i + a[i]][1];
}
int main(){
	scanf("%d %d", &N, &M);

	for(int i = 0; i < N; i++)
		scanf("%d", &a[i]);

	for(K = 1; (K + 1) * (K + 1) <= N; K++);

	for(int i = N - 1; i >= 0; i--)
		fix(i);

	for(int k = 0; k < M; k++){
		scanf("%d", &t);

		if(t == 0){
			int s, v;
			scanf("%d %d", &s, &v);
			s--;

			a[s] = v;

			for(int i = min(N - 1, s / K * K + K - 1); i >= s / K * K; i--)
				fix(i);

		}else{
			int s, cnt = 0;
			scanf("%d", &s);
			s--;

			while(b[s][1] != s) cnt += b[s][0], s = b[s][1];

			printf("%d %d\n", s + 1, cnt + 1);
		}
	}

	return 0;
}
