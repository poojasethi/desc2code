#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e5+10;
int n,a[maxn],sn,r[maxn],c[maxn],q,id,u,v;
pair<int,int> ans;
void update(int x,int y){
    a[x]=y;int z=(x-1)/sn;
    for (int k,i=x; i>0&&(i-1)/sn==z; i--){
        k=i+a[i];
        if (k>n||(k-1)/sn!=z){
            r[i]=i;c[i]=1;
        } else{
            r[i]=r[k];
            c[i]=c[k]+1;
        }
    }
}
pair<int,int> get(int x){
    pair<int,int> ans=make_pair(r[x],c[x]);
    x=r[x];x+=a[x];
    if (x<=n){
        pair<int,int> tmp;
        tmp=get(x);
        ans=make_pair(tmp.first,tmp.second+ans.second);
    }
    return ans;
}
int main(){
    scanf("%d%d",&n,&q);sn=350;
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    for (int i=n; i>=1; i--) update(i,a[i]);
    while (q--){
        scanf("%d",&id);
        if (id){
            scanf("%d",&u);
            ans=get(u);
            printf("%d %d\n",ans.first,ans.second);
        } else{
            scanf("%d%d",&u,&v);
            update(u,v);
        }
    }
}
