#include <cstdio>

#define maxn 100010

int n,m,i,j,L[maxn],R[maxn],a[maxn],toEnd[maxn],last[maxn],pn,prt[maxn],o,x,y,ans,lt;

void upd(int idx){
	for(j=R[idx];j>=L[idx];j--)if(j+a[j]>R[idx])toEnd[j]=1,last[j]=j;else toEnd[j]=1+toEnd[j+a[j]],last[j]=last[j+a[j]];
}

int main (int argc, char * const argv[]) {
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++){if(i%232==1)L[++pn]=i;R[pn]=i;prt[i]=pn;}
	for(i=1;i<=pn;i++)upd(i);
	for(i=1;i<=m;i++){
		scanf("%d",&o);
		if(o==0){scanf("%d%d",&x,&y);a[x]=y;upd(prt[x]);}else{
			scanf("%d",&x);
			j=x;ans=0;
			while(j<=n){
				ans+=toEnd[j];
				lt=last[j];
				j=last[j]+a[last[j]];
			}
			printf("%d %d\n",lt,ans);
		}
	}
    return 0;
}
