#define N (1<<17)
#include <bits/stdc++.h>
using namespace std;

int n,m,g,t,x,y,a[N],b[N],c[N];

int main()
{
	cin>>n>>m,g=sqrt(n);
	for(int i=1;i<=n;i++)
		scanf("%d",a+i);

	for(int i=1;i<=n;i++)
	{
		int j=i;
		for(;j/g==i/g && j<=n;j+=a[j])
			b[i]++;
		c[i]=j;
	}

	for(int k=0;k<m;k++)
	{
		scanf("%d",&t);
		if(t==0)
		{
			scanf("%d%d",&x,&y),a[x]=y;
			for(int i=x;i/g==x/g && i>=1;i--)
				if(i+a[i]>n || (i+a[i])/g>i/g) c[i]=i+a[i],b[i]=1;
				else c[i]=c[i+a[i]],b[i]=b[i+a[i]]+1;
		}
		else
		{
			scanf("%d",&x);
			int res=0,cur=x;
			for(;c[cur]<=n;cur=c[cur])
				res+=b[cur];
			for(;cur+a[cur]<=n;cur+=a[cur])
				res++;
			printf("%d %d\n",cur,res+1);
		}
	}
}
