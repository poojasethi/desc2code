#include <bits/stdc++.h>
#include <cstdio>
using namespace std;
typedef int ll;
const int maxn = 100000 + 100;

ll power[maxn],countt[maxn],next[maxn],Block[maxn];
ll N , K , M ;


void DO(ll a)
{
  ll h = power[a] +a;
  if(h > N)
    next[a]=a,countt[a]=0;
  else if (Block[ a] != Block[ h ] )
    next[a]=h , countt[a]=1;
  else
    next[a] = next[ h ] , countt [a] = 1 + countt[ h] ;
}


void change(ll a , ll b)
{
  power[a] = b;
  for(ll i = a; Block[a] == Block[i] ; i--)
    DO(i);
}

void put(ll a)
{
  ll ans=1;
  while( next[a] != a)
    ans += countt[a], a = next[a];
  printf("%d %d",a,ans);
  printf("\n");
}

int main()
{
  cin>>N>>M;
  K = sqrt(N);
  Block[0] = -100000;
  for(ll i=1; i<=N; i++){
    Block[i]= (i-1) / K;
    scanf("%d", &power[i] );
  }
  for(ll i=N; i>0; i--)
    DO(i);
  for(ll i=0; i<M; i++){
    ll t,a,b;
    scanf("%d%d",&t,&a);
    if(t){
      put(a);
    }
    else{
      scanf("%d",&b);
      change(a,b);
    }
  }
  return 0;
}
