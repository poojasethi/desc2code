#include <cstdio>
#include <math.h>

int f[100*100*10];
int p[100*100*10];
int r[100*100*10];
int n,m,k;

void fun(int i)
{
	if( i+f[i]>=n || i/k != (i + f[i])/k)  {
		p[i]=i;
		r[i]=0;
	}
	else {
		p[i] = p[i + f[i]];
		r[i] = r[i + f[i]] + 1;
	}
}

int main()
{ 

	scanf("%d%d", &n,&m);
	k=(int)sqrt((long double)n);

	for(int i=0;i<n;i++)
		scanf("%d",&f[i]);
	
	for(int i=n-1;i>=0;i--)
		fun(i);
	
	for(int i=0;i<m;i++)
	{
		int t,a,b;
		scanf("%d%d",&t,&a);a--;
		if(t==0)   {
			scanf("%d",&b);

			f[a]=b;
			for(int j=a;j>=0 && j/k==a/k;j--) 
				fun(j);
		}
		else   {
			int res=0,res2=0;
			for(;a<n;res2=p[a] , a=res2+f[res2])
				res+=r[a]+1;

			printf("%d %d\n",res2+1,res);
		}
	}
	return 0;
}
