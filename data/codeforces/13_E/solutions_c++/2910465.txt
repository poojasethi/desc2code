#include<iostream>
#include<cmath>
using namespace std;
const int mm=100100;
int f[mm],num[mm],last[mm];
int k,m,n;
void fen(int x)///区间值更新
{
  if(x+f[x]>m||x/k!=(x+f[x])/k)///挑出区间了
    last[x]=x,num[x]=0;
  else last[x]=last[f[x]+x],num[x]=num[x+f[x]]+1;///在去间内正常模拟
}
int main()
{
  int a,b,c;
  while(cin>>m>>n)
  {
    for(int i=1;i<=m;i++)
      cin>>f[i];
    for(int i=1;i*i<m;i++)///分sqrt(m)个区间
      k=i;
    for(int i=m;i>0;i--)///预处理
      fen(i);
    for(int i=0;i<n;i++)
    {
      cin>>a>>b;
      if(a)
      {
       int _num=0,_last;
        while(b<=m)///模拟区间跳
        { _last=last[b];
          _num+=num[b]+1;
          b=_last+f[_last];
        }
        cout<<_last<<" "<<_num<<"\n";
      }
      else
      {
        cin>>c;f[b]=c;
        for(int i=b;i>0&&i/k==b/k;i--)///区间更新
          fen(i);
      }
    }
  }
}
