#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
using namespace std;
int n,m,op,x,y,t,p,block,times[100010],pos[100010],a[100010],pre[100010];
int main()
{
	scanf("%d%d",&n,&m);
	block=sqrt(n);
	for(int i=0;i<n;++i)
		scanf("%d",&a[i]);
	for(int i=n-1;~i;--i)
	{
		int tmp=i+a[i];
		if(tmp>=n)pos[i]=-1,times[i]=1,pre[i]=i;
		else if(tmp>=(i/block+1)*block)pos[i]=tmp,times[i]=1,pre[i]=i;
		else pos[i]=pos[tmp],times[i]=times[tmp]+1,pre[i]=pre[tmp];
	}
	while(m--)
	{
		scanf("%d",&op);
		if(op)
		{
			scanf("%d",&x);
			--x;
			t=p=0;
			for(int i=x;~i;i=pos[i])
				t+=times[i],p=pre[i];
			printf("%d %d\n",p+1,t);
		}
		else
		{
			scanf("%d%d",&x,&y);
			--x;
			a[x]=y;
			for(int i=x;i>=x/block*block;--i)
			{
				int tmp=i+a[i];
				if(tmp>=n)pos[i]=-1,times[i]=1,pre[i]=i;
				else if(tmp>=(i/block+1)*block)pos[i]=tmp,times[i]=1,pre[i]=i;
				else pos[i]=pos[tmp],times[i]=times[tmp]+1,pre[i]=pre[tmp];
			}
		}
	}
}
