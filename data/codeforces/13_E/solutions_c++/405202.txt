#include <cstdio>
#include <math.h>
using namespace std;

const int maxn = 100000+5;
int n,m,k,a[maxn];
int ne[maxn],p[maxn],num[maxn];

void updata(int u)
{
	if (u+a[u]>n) {
		ne[u]=u;
		num[u]=1;
	} else {
		if (p[u]==p[u+a[u]]) {
			ne[u]=ne[u+a[u]];
			num[u]=num[u+a[u]]+1;
		} else {
			ne[u]=u+a[u];
			num[u]=1;
		}
	}
}

void work()
{
	for (int i = n; i >= 1; --i) 
		updata(i);

	int c,x,y,ans;
	for (int i = 0; i != m; ++i) {
		scanf("%d",&c);
		if (c==0) {
			scanf("%d%d",&x,&y);
			a[x]=y;
			for (int j = x; j >= 1; --j) {
				if (p[j]!=p[x]) break;
				updata(j);
			}
		} else {
			scanf("%d",&x);
			ans = 0;
			while (x<=n) {
				ans+=num[x];				
				if (x==ne[x]) break;
				if (p[x]==p[ne[x]]) ans--;
				x=ne[x];
			}
			printf("%d %d\n",x,ans);
		}
	}
	return;
}

void init()
{
	scanf("%d%d",&n,&m);
	k=(int)(sqrt(double(n)));
	for (long i = 1; i <= n; ++i) {
		scanf("%d",&a[i]);
		p[i]=i/k+1;	
	}
	return;
}

int main()
{
	init();
	work();

	return 0;
}
