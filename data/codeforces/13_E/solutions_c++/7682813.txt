const int N=131072,P=256;
int n,m,qy,bc=0,i,j,x,y,ans,stop,pt[N],L[N],R[N],last[N],a[N],toNext[N];
//void update(int id){

//		toNext[j]=toNext[j+a[j]]+1,last[j]=last[j+a[j]];

void update(int idx){
	for(j=R[idx];j>=L[idx];j--)if(j+a[j]>R[idx])toNext[j]=1,last[j]=j;else toNext[j]=1+toNext[j+a[j]],last[j]=last[j+a[j]];
}
int main(){
	//puts("OK");
	//freopen("input.txt","r",stdin);
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	for(i=1;i<=n;i++){if(i%334==1)L[++bc]=i;R[bc]=i;pt[i]=bc;}
	for(i=1;i<=bc;i++)update(i);
	for(i=1;i<=m;i++){
		scanf("%d",&qy);
		if(qy==0){scanf("%d%d",&x,&y);a[x]=y;update(pt[x]);}else{
			scanf("%d",&x);
			j=x;ans=0;
			while(j<=n){
				ans+=toNext[j];
				stop=last[j];
				j=last[j]+a[last[j]];
			}
			printf("%d %d\n",stop,ans);
		}
	}
	return 0;
}
