#include<bits/stdc++.h>
#define MP make_pair
#define R second
#define L first
using namespace std;
const int maxn=200*1000+1000, sq=500;
int n;
int p[maxn];
int nx[maxn];
int t[maxn];
pair<int, int> get(int x){
	if(p[nx[x]+p[nx[x]]]==0)
		return MP(nx[x], t[x]);
	pair<int, int> ret=get(nx[x]+p[nx[x]]);
	ret.R+=t[x];
	return ret;
}

void modify(int a, int b){
	int l=a/sq*sq, r=l+sq;
	p[a]=b;
	for(int i=min(r-1, n-1);i>=l;i--)
		if(i+p[i]>=min(r, n))
			nx[i]=i, t[i]=1;
		else
			t[i]=t[i+p[i]]+1, nx[i]=nx[i+p[i]];
}

int main(){
	ios_base::sync_with_stdio(false);
	for(int i=0;i<maxn;i++)
		nx[i]=i;
	int m;
	cin >> n >> m;
	for(int i=0;i<n;i++)
		cin >> p[i];
	for(int l=0;l<n;l+=sq)
		for(int i=min(l+sq	-1, n-1);i>=l;i--){
			if(i+p[i]>=min(l+sq, n))
				nx[i]=i, t[i]=1;
			else
				t[i]=t[i+p[i]]+1, nx[i]=nx[i+p[i]];
		}
	while(m--){
		int q;
		cin >> q;
		if(q==0){
			int a, b;
			cin >> a >> b;
			modify(--a, b);
		}else{
			int a;
			cin >> a;
			pair<int, int> ret=get(--a);
			cout << ret.L+1 << " "<< ret.R << endl;
		}
	}
}
