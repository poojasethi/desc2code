#include<iostream>
#include<cstdio>
#include<cmath>
#include<cstring>
#define N 100005
using namespace std;
int nxt[N],step[N],a[N];
int n,m;
void solve()
{
	int i,j,k,c,d;
	k=(int)sqrt(1.0*n);
	for(i=n-1;i>=0;i--)
	{
		nxt[i]=a[i]+i;
		step[i]=1;
		if(nxt[i]<n&&nxt[nxt[i]]<n&&i/k==nxt[i]/k)
			step[i]+=step[nxt[i]],nxt[i]=nxt[nxt[i]];
	}
	while(m--)
	{
		scanf("%d",&i);
		if(i==1)
		{
			scanf("%d",&c);
			c--;
			d=step[c];
			while(nxt[c]<n)
			{
				c=nxt[c];
				d+=step[c];
			}
			printf("%d %d\n",c+1,d);
		} 
		else
		{
			scanf("%d%d",&c,&d);
			c--;
			a[c]=d;
			for(j=c;;j--)
			{
				nxt[j]=j+a[j];
				step[j]=1;
				if(nxt[j]<n&&nxt[nxt[j]]<n&&j/k==nxt[j]/k)
					step[j]+=step[nxt[j]],nxt[j]=nxt[nxt[j]];
				if(j%k==0)
					break;
			}
		}
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	solve();
	return 0;
}
