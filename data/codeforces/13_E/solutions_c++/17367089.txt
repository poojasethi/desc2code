#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
const int N=100001;
int a[N],sp[N],nx[N];
int main(){
	int n,m,sz;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)
		scanf("%d",&a[i]);
	sz=sqrt(1.0*n)+1;
	int nxt;
	for(int i=n-1;~i;--i){
		nxt=i+a[i];
		if(nxt>=n){
			nx[i]=i;
			sp[i]=1;
		}
		else{
			if(i/sz==nxt/sz){
				sp[i]=1+sp[nxt];
				nx[i]=nx[nxt];
			}
			else{
				sp[i]=1;
				nx[i]=nxt;//careful
			}
		}
	}
	int u,v,step;
	while(m--){
		scanf("%d",&u);
		if(u){
			scanf("%d",&v);
			--v;
			step=0;
			u=v;
			while(v<n){
				step+=sp[v];
				if(v==nx[v]){
					if(u/sz==v/sz&&u!=v)
						--step;
					break;
				}
				u=v;
				v=nx[v];
			}
			printf("%d %d\n",v + 1, step);
			continue;
		}
		scanf("%d%d",&u,&v);
		a[--u]=v;
		for(int i=u;i>=u/sz*sz;--i){
			nxt=i+a[i];
			if(nxt>=n){
				nx[i]=i;
				sp[i]=1;
			}
			else{
				if(i/sz==nxt/sz){
					sp[i]=1+sp[nxt];
					nx[i]=nx[nxt];
				}
				else{
					sp[i]=1;
					nx[i]=nxt;
				}
			}
		}
		
	}
	return 0;
}
