#include<iostream>
#include<cstdio>
#include<cmath>
using namespace std;
#define N 100010

int a[N];
int next[N];
int step[N];
int n, m, c;

void update(int l,int r)
{
	for (int i=r-1; i>=l; i--)
		if (i+a[i]<r) next[i]=next[i+a[i]], step[i]=step[i+a[i]]+1;
		else next[i]=i, step[i]=0;
}

void prepare()
{
	c = int (sqrt(n)+0.5);
	for (int l=0,r=l+c; l<n; l=r,r=min(n,l+c)) 
		update(l,r);
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=0; i<n; i++) scanf("%d",a+i);
	
	prepare();
	
	while (m--)
	{
		int ctrl, x, y;
		scanf("%d",&ctrl);
		if (! ctrl) 
		{
			scanf("%d%d",&x,&y);
			a[x-1]=y;
			int l=(x-1)/c*c, r=min(n,l+c);
			update(l,r);
		} else 
		{
			scanf("%d",&x);
			int ans = 0, last = 0;
			for (int i=x-1; i<n; )
				if (next[i]==i) ans++, last=i, i=i+a[i];
				else ans+=step[i], i=next[i];
			printf("%d %d\n", last+1, ans);
		}
	}
}
