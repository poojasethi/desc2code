#include<cstdio>
#include<cmath>
const int N=100010;
int n,m;
int a[N];
int J[N];
int len[N];

int main(){
	scanf("%d %d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	J[n]=n;
	len[n]=0;
	int d=sqrt(n);
	for(int i=n-1;i>=0;i--){
		int next=(i +d)/d*d;
		if(i+a[i]>=n){
			J[i]=n;
			len[i]=1;
		}
		else if(i+a[i]>=next){
			J[i]=i+a[i];
			len[i]=1;
		}
		else J[i]=J[i+a[i]],len[i]=1+len[i+a[i]];
	}
	int op;
	int x,y;
	for(int i=0;i<m;i++){
		scanf("%d",&op);
		if(op==0){
			scanf("%d %d",&x,&y);x--;
			a[x]=y;
			int low=x/d*d;
			for(int j=x;j>=low;j--){
				int next=(j +d)/d*d;
				if(j+a[j]>=n){
					J[j]=n;
					len[j]=1;
				}
				else if(j+a[j]>=next){
					J[j]=j+a[j];
					len[j]=1;
				}
				else J[j]=J[j+a[j]],len[j]=1+len[j+a[j]];
			
			}
		}
		else{
			scanf("%d",&x);x--;
			int ans=0;int last;
			while(x<n){
				//printf("x:%d %d\n",x,len[x]);
				ans+=len[x];
				if(J[x]==n){
					while(x<n){
						last=x;
						x+=a[x];
					}
				}
				if(x>n)x=n;
				x=J[x];
			}
			printf("%d %d\n",last+1,ans);
		}
	}
}
