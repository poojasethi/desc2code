#include <stdio.h>

const int K = 500;
int n;
int a[100005];
int to[100005];
int c[100005];
int g[100005];

void handle(int i) {
	if(!a[i])
		to[i] = i, c[i] = 0;
	else if(g[a[i]] != g[i]) {
		to[i] = a[i];
		c[i] = 1;
	} else {
		to[i] = to[a[i]];
		c[i] = c[a[i]]+1;
	}
}

void change(int x, int y) {
	a[x] = (x+y>n? 0: x+y);
	for(int i=x;i>0 && g[i]==g[x];--i)
		handle(i);
}

void query(int x) {
	int ans = 1;
	while(to[x]!=x) {
		ans += c[x];
		x = to[x];
	}
	printf("%d %d\n", x, ans);
}

int main(void) {
	int m, x, i;
	scanf("%d%d", &n, &m);
	for(i=1;i<=n;i++) {
		scanf("%d", &x);
		a[i] = (i+x>n?0:i+x);
	}
	for(i=1;i<=n;i++)
		g[i] = (i-1)/K;
	for(i=n;i>=1;--i)
		handle(i);
	while(m--) {
		int typ, y;
		scanf("%d%d", &typ, &x);
		if(typ==0) {
			scanf("%d", &y);
			change(x, y);
		} else {
			query(x);
		}
	}
	return 0;
}
