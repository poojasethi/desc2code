#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;
const int Maxn = 100 * 1000 + 10;
int arr[Maxn], cnt[Maxn], nb[Maxn], sqr,n,m;

void fill(int num)
{
	int s = num*sqr;
	int t = min(n,(num+1)*sqr);
	for(int i = t-1; i >= s; i--)
	{
		cnt[i] = 0;
		if(i+arr[i] >= t)
			nb[i] = i;
		else
			nb[i] = nb[i+arr[i]];
		if(i+arr[i] >= t)
			cnt[i] = 1;
		else 
			cnt[i] = cnt[i+arr[i]]+1;
	}
}
int main()
{
	scanf("%d%d",&n,&m);
	for(int i = 0; i < n; i++)
		scanf("%d",&arr[i]);
	sqr = 0;
	while(sqr*sqr < n)
		sqr++;
	for(int i = 0; i < sqr; i++)
		fill(i);
	for(int i = 0; i < m; i++)
	{
		int type,a,b;
		scanf("%d",&type);
		if(type)
		{
			scanf("%d",&a);
			int k = --a;
			int ans = cnt[k];
			while(nb[k] + arr[nb[k]] < n)
			{
				k = nb[k] + arr[nb[k]];
				ans += cnt[k];
			}
			printf("%d %d\n",nb[k]+1,ans);
		}
		else
		{
			scanf("%d%d",&a,&b);
			arr[a-1] = b;
			fill(((a-1)/sqr));
		}
	}
	return 0;
}
