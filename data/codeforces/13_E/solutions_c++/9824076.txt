#include<iostream>
#include<cstdio>

using namespace std;

const int sq = 300, MAXN = 2e5 + 5;
const int end = 2e5 - 100;
int a[MAXN], cnt[MAXN], nxt[MAXN];
int n, m;

inline void update(int x)
{
	int d = x + a[x];
	if(d >= n)
		d = end;
	if(x / sq == d / sq)
		cnt[x] = cnt[d] + 1, nxt[x] = nxt[d];
	else
		cnt[x] = 1, nxt[x] = d;
}

int main()
{
	scanf("%d%d", &n, &m);
	for(int i = 0; i < n; i++)
		scanf("%d", &a[i]);
	for(int i = n - 1; i >= 0; i--)
		update(i);
	while(m--)
	{
		int t, x, y;
		scanf("%d%d", &t, &x);
		x--;
		if(t == 0)
		{
			scanf("%d", &a[x]);
			for(int j = x; j >= 0 && j / sq == x / sq; j--)
				update(j);
		}
		else
		{
			int ans = 0, last = x;
			while(x != end)
			{
				ans += cnt[x];
				last = x;
				x = nxt[x];
			}
			x = last;
			while(x != end)
			{
				last = x;
				x = x + a[x];
				if(x >= n)
					x = end;
			}
			printf("%d %d\n", last + 1, ans);
		}
	}
	return 0;
}
