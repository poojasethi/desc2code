#include<cstdio>
#include<cstring>
const int N = 100005;
int jmp[N],end[N],cnt[N],next[N],n;
int get(int x)
{
	if(jmp[x]==N)
	{
		printf("%d ",end[x]);
		return cnt[x];
	}
	return cnt[x]+get(jmp[x]);
}
void up(int i,int j)
{
	if(j>n)
	{
		end[i]=i;
		jmp[i]=N;
		cnt[i]=1;
	}
	else
	{
		end[i]=end[j];
		if(j/300!=i/300)
		{
			jmp[i]=j;
			cnt[i]=1;
		}
		else
		{
			jmp[i]=jmp[j];
			cnt[i]=cnt[j]+1;
		}
	}
}
int main()
{
	int m,i,j,a,b,c;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%d",next+i);
	for(i=n;i>0;i--)
	{
		up(i,i+next[i]);
	}
	while(m--)
	{
		scanf("%d",&a);
		if(!a)
		{
			scanf("%d%d",&b,&c);
			next[b]=c;
			up(b,b+c);
			j=b/300*300;
			for(i=b-1;i>=j;i--)if(i+next[i]<=b)up(i,i+next[i]);
		}
		else
		{
			scanf("%d",&b);
			printf("%d\n",get(b));
		}
	}
	return 0;
}
