# include <bits/stdc++.h>

using namespace std;

const int maxn = 1e5 + 100;
const int SQRT = 300;

int p[maxn], next[maxn], move[maxn];

int main(){
	ios_base::sync_with_stdio(false);
	cin.tie(0); cout.tie(0);
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> p[i];
	
	next[n] = n;
	for (int i = n-1; i >= 0; i--){
		int nxt = min(i + p[i], n);
		if(i / SQRT == nxt / SQRT)
			next[i] = next[nxt], move[i] = move[nxt] + 1;
		else
			next[i] = nxt, move[i] = 1;
	}

	while(m--){
		int t;
		cin >> t;
		if(t == 0){
			int a, b;
			cin >> a >> b; a--;
			p[a] = b;
			for (int i = a; i >= 0; i--)
				if(i / SQRT == a / SQRT){
					int nxt = min(i + p[i], n);
					if(i / SQRT == nxt / SQRT)
						next[i] = next[nxt], move[i] = move[nxt] + 1;
					else
						next[i] = nxt, move[i] = 1;
				}
				else
					break;
		}
		else{
			int a;
			cin >> a; a--;
			int ans = 0;
			while(next[a] != n)
				ans += move[a], a = next[a];
			while(a + p[a] < n)
				a += p[a], ans++;
			cout << a + 1 << " " << ans + 1 << '\n';
		}
	}
	return 0;
}
