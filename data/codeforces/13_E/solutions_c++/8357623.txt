							#include <bits/stdc++.h>
#define MP make_pair
#define L first
#define R second
using namespace std ;
const int seg = 500, maxn=2*100*1000+100 ;
int n , m;
int p[maxn], out[maxn], t[maxn];

void input(){
	cin >> n >> m ;
	for ( int i = 0 ; i < n ; ++i )
		cin >> p[i];
}

pair<int ,int> get(int x){
	if(out[x]+p[out[x]]>=n)
		return MP(out[x], t[x]);
	pair<int, int> ret=get(out[x]+p[out[x]]);
	ret.R+=t[x];
	return ret;
}

void change( int segnum ){
	int l=segnum*seg, r=min(l+seg, n);
	for(int i=r-1;i>=l;i--){
		if(i+p[i]>=r)
			out[i]=i, t[i]=1;
		else
			out[i]=out[i+p[i]], t[i]=t[i+p[i]]+1;
	}
}

void pre_process(){
	for(int i=0;i<=(n-1)/seg;i++)
		change(i);
}

void give_query(){
	while(m--){
		int x ;
		cin >> x ;
		if ( x == 1 ){
			int a;
			cin >> a ;
			pair<int, int> ret=get(--a);
			cout << ret.L+1 << " " << ret.R << endl ;
		}
		else{
			int a, b ;
			cin >> a >> b ;
			p[--a]=b;
			change(a/seg) ;
		}
	}
}

main(){
	ios_base::sync_with_stdio(false);
	input();
	pre_process();
	give_query();
}
