#include<stdio.h>
#define M 100000
#include<iostream>
#include<algorithm>
#define ull unsigned long long
using namespace std;
void minimize(ull &a,ull b){
	if(a > b) a = b;
}
ull ans = 2e19;
int m,n;
int a[M + 20] , b[M + 20];
ull sA = 0 , sB = 0;
ull tA = 0 , tB = 0;
main(){
//	freopen("test.inp","r",stdin);
	scanf("%d %d",&m,&n);
	for(int i = 1 ; i <= m ; i++) 	scanf("%d",&a[i]) , sA+=a[i];
	for(int i = 1 ; i <= n ; i++)	scanf("%d",&b[i]) , sB+=b[i];
	sort(a + 1 , a + 1 + m);
	sort(b + 1 , b + 1 + n);
	for(int i = 0 ; i < m ; i++){
		tA+=a[i];
		minimize(ans , tA + sB*(m - i));	
	}
	for(int i = 0 ; i < n ; i++){
		tB+=b[i];
		minimize(ans , tB + sA*(n - i));
	}
	cout<<ans;
}
