#include <cstdio>
#include <algorithm>
#include <cmath>
#include <iostream>
using namespace std;
long long n,m,x[100005],y[100005],sum1,sum2;
int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)scanf("%I64d",&x[i]),sum1+=x[i];
	for(int i=0;i<m;i++)scanf("%I64d",&y[i]),sum2+=y[i];
	sort(x,x+n);
	sort(y,y+m);
	long long ts1=sum1,ts2=sum2;
	long long ans=1e16;
	for(long long i=n-1;i>=0;i--)
	{
		ts1-=x[i];
		if((ans-ts1)/(n-i)>=sum2-1)
		ans=min(ans,ts1+(sum2)*(n-i));
	}
	for(long long i=m-1;i>=0;i--)
	{
		ts2-=y[i];
		if((ans-ts2)/(m-i)>=sum1-1)
		ans=min(ans,ts2+(sum1)*(m-i));
	}
	cout<<ans<<endl;
	return 0;
}
