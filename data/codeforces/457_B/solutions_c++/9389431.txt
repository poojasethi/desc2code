#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
#define N 111111
using namespace std;
int n,m;
long long a[N],b[N],ans1,ans2;
int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)scanf("%I64d",&a[i]),a[0]+=a[i];
    for(int i=1;i<=m;i++)scanf("%I64d",&b[i]),b[0]+=b[i];
    sort(a+1,a+n+1);
    sort(b+1,b+m+1);
    ans1=b[0];for(int i=1;i<n;i++)if(a[i]<b[0])ans1+=a[i];else ans1+=b[0];
    ans2=a[0];for(int i=1;i<m;i++)if(b[i]<a[0])ans2+=b[i];else ans2+=a[0];
    cout<<min(ans1,ans2)<<endl;
    return 0;
}
