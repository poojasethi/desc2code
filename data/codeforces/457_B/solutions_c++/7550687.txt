#include<bits/stdc++.h>
using namespace std;
int a[100005],b[100005],tmp[100005];
int main()
{
    //freopen("b.txt","r",stdin);
    int n,m,i;
    while(~scanf("%d%d",&n,&m)){
        long long sa=0,sb=0;
        for(i=0;i<n;i++)
            scanf("%d",&a[i]),sa+=a[i];
        for(i=0;i<m;i++)
            scanf("%d",&b[i]),sb+=b[i];
        sort(a,a+n,greater<int>());
        sort(b,b+m,greater<int>());
        if(a[0]<b[0])  memcpy(tmp,a,sizeof(a)),memcpy(a,b,sizeof(b)),memcpy(b,tmp,sizeof(tmp)),swap(sa,sb),swap(n,m);
        long long ans=sb;
        for(i=1;i<n;i++)
            ans+=a[i]>sb?sb:a[i];
        cout<<ans<<endl;
    }
    return 0;
}
