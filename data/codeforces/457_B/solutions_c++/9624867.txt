#include<stdio.h>
int a[100000];
int b[100000];

long long min(long long a,long long b){return a<b?a:b;}

int main(){
    int n,m;
    scanf("%d %d",&n,&m);
    int i;
    long long s=0,t=0;
    int M1=0,M2=0;
    for(i=0;i<n;i++){
        scanf("%d",&a[i]);
        s+=a[i];
        if(a[i]>a[M1])M1=i;
    }
    for(i=0;i<m;i++){
        scanf("%d",&b[i]);
        t+=b[i];
        if(b[i]>b[M2])M2=i;
    }
    long long res1=t,res2=s;
    for(i=0;i<n;i++)if(i!=M1)res1+=min(a[i],t);
    for(i=0;i<m;i++)if(i!=M2)res2+=min(b[i],s);
    printf("%I64d\n",min(res1,res2));
    return 0;
}
