#include <bits/stdc++.h>
#define maxn 100009
using namespace std;
__int64 a[maxn],b[maxn];
int n,m;
unsigned long long ans=1e19+1e15;
int main()
{
	scanf("%d%d",&n,&m);
	unsigned long long sum1=0,sum2=0;
	for(int i=1;i<=n;i++)
		scanf("%I64d",&a[i]),sum1+=a[i];
	for(int i=1;i<=m;i++)
		scanf("%I64d",&b[i]),sum2+=b[i];
	sort(a+1,a+1+n);
	sort(b+1,b+1+m);
	unsigned long long cur=0;
	for(int i=1;i<=n;i++)
	{
		ans=min(ans,sum2*(n+1-i)+cur);
		cur+=a[i];
	}
	cur=0;
	for(int i=1;i<=m;i++)
	{
		ans=min(ans,sum1*(m+1-i)+cur);
		cur+=b[i];
	}
	cout<<ans<<endl;
	//system("pause");
	return 0;
}
