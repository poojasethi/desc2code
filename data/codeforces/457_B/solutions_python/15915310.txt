m, n = map(int, raw_input().split())
a = map(int, raw_input().split())
b = map(int, raw_input().split())
a.sort()
b.sort()
a.reverse()
b.reverse()
sa, sb = [a[0]], [b[0]]
for i in range(1, m):
	sa.append(sa[i-1] + a[i])
for i in range(1, n):
	sb.append(sb[i-1] + b[i])
aa, bb = sa[m-1], sb[n-1]
ans = 9999999999999999999999999999;
for i in range(m):
	ans = min(ans, aa - sa[i] + bb *(i + 1) );
for i in range(n):
	ans = min(ans, bb - sb[i] + aa * (i + 1));
print ans
