#include <iostream>
using namespace std;
int get(int a, int b) {
    if (a > b) swap(a, b);
    if (a == 1) return b - 1;
    if (b % a == 0) return 1 << 29;
    return get(a, b % a) +  b / a;
}
int main() {
    int n;
    cin >> n;
    
    int best = 1 << 29;
    for (int i = 1; i <= n; ++i) {
        best = min(best, get(i, n));
    }
    cout << best;
}
