#include <cstdio>

int gcd(int a, int b, int& s)
{
	if (!b) return a;
	int c = a/b;
	s += c;
	return gcd(b, a%b, s);
}

int main()
{
	int n, res;
	scanf("%d", &n);
	res = n;
	for (int i = 1; i <= n; ++i)
	{
		int s = 0;
		int g = gcd(n, i, s);
		if (g != 1) continue;
		if (s < res) res = s;
	}
	printf("%d\n", res-1);
}
