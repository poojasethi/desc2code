#include<stdio.h>
#include<stdlib.h>
int get(int a,int b)
{
	if (b<=0) return 1000000; 
	if(b==1) return a-1;
	else return 
	  get(b,a-a/b*b)+a/b; 
}

int main(void)
{
	int n,ans,i,t;
    scanf("%d",&n);
    ans=0; 
	for (i=1; i<=n-1; i++)
	{
		t=get(n,i);
		if (ans==0) ans=t; 
		else if (ans>t) ans=t; 
	}
	printf("%d",ans); 
	i=1; 
}
