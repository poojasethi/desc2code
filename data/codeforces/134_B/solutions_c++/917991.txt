#include<cstdio>
int inf=0x7ffffff;
int pan(int i,int n)
{
    if(n%i==0)
    if(i==1)
    return n-1;
    else
    return inf;
    return pan(n%i,i)+n/i;
}
int main()
{
    int min=inf,n,i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
        int aa=pan(i,n);
        if(aa<min)
        min=aa;
    }
    printf("%d\n",min);
    return 0;
}
