#include <cstdio>
#include <algorithm>
#define  inf 200000000
using namespace std;
int cal(int n,int m)
{
    if(m==1) return n-1;
    if(m==0) return inf;
    return n/m+cal(m,n%m);
}
int main()
{
    int ct=0;
    int n;
    int ans=inf;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        ans=min(ans,cal(n,i));
    }
    printf("%d\n",ans);
}
