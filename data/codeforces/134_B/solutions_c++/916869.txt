#include<cstdio>
int f(int a,int b)
{
    if (a%b==0) 
        if (b==1) return a-1;
        else return 1<<30;
    else
        return f(b,a%b)+a/b;
}
int main()
{
    int n;
    scanf("%d",&n);
    int ans=1<<30;
    for(int i=1;i<=n;i++) 
    {
        int tmp=f(n,i);
        if (ans>tmp) ans=tmp;
    }
    printf("%d\n",ans);
}
