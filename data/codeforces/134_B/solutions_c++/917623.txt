#include<iostream>
#include<cstdio>
using namespace std;
const int inf = ~0u>>2;
int dfs(int a,int b){
	if(b==1) return a;
	if(a%b==0) return inf;
	return a/b+dfs(b,a%b);
}
int main(){
	int n;
	cin>>n;
	if(n==1) {puts("0"); return 0;}
	int ans = inf;
	for(int i=1;i*2<=n;i++){
		int s = dfs(i,n-i);
//		cout<<i<<" "<<s<<endl;
		if(s<ans) ans = s;
	}
	cout<<ans<<endl;
}
