#include <iostream>
using namespace std;

const int INF = 1e9;
int c;
void gcd(int a, int b)
{
	if(b == 0)
	{
		if(a != 1)
			c = INF;
		return;
	}
	c += a/ b;
	gcd(b, a % b);
}

int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	int mn = INF;
	for(int i = 1; i <= n; i++)
	{
		c = 0;
		gcd(n, i);
		mn = min(mn, c-1);
	}
	cout << mn << endl;
	return 0;
}
