#include <stdio.h>
#include <stdlib.h>
int ans=99999999;
int min(int m,int n)
{
    if (n==1)
        return(m-1);
    if (m%n==0) return(9999999);
    return(min(n,m%n)+m/n);
}
int main()
{
    int n,i,k;
    scanf("%d",&n);
    for (i=1;i<=(n/2)+1;i++)
    {
        k=min(n,i);
        if (k<ans) ans=k;
    }
    printf("%d",ans);
}
