#include <cstdio>
#include <algorithm>
using namespace std;

const int maxn = 1e6;

int ex_gcd(int a, int b) { // a > b && b !| a
	if (b == 0) {
		if (a == 1) return 0;
		else return maxn;
	}
	return ex_gcd(b, a % b) + (a / b);
}

int main() {
	int n;
	scanf("%d", &n);
	int res = n;
	for (int i = 1; i < n; i++)
		res = min(res, ex_gcd(n, i));
	printf("%d\n", res - 1);
}
