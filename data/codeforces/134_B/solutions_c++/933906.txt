#include <iostream>
#include<algorithm>
using namespace std;
int sol(int n,int i)
{
    if(n%i==0)
    {
        if(i==1)return n-1;
        return 1<<30;
    }
    return sol(i,n%i)+n/i;
}
int main()
{
    int n;
    cin>>n;
    int ans=(1<<30);
    for(int i=1;i<=n/2+1;i++)
    ans=min(ans,sol(n,i));
    cout<<ans<<endl;
    return 0;
}
