#include <cstdio>
inline int min(int a,int b){return a<b?a:b;}
int calc(int a,int b)
{
    if(b==1)return a-1;
    if(a%b==0)return 99999;
    return a/b+calc(b,a%b);
}
int main()
{
    int n,ans=99999;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        ans=min(ans,calc(n,i));
    printf("%d\n",ans);
    scanf(" ");
}
