#include <iostream>
using namespace std;
const int INF=1000*1000*10+10;
int n,mini=INF;
int f(int x,int y)
{
	if(x==1&&y==1)
		return 0;
	if(x==y)
		return INF;
	if(x>y)
		return f(x-y,y)+1;
	else
		return f(x,y-x)+1;
}
int main()
{
	cin >> n;
	for(int i=1;i<=n;i++)
		mini=min(mini,f(n,i));
	cout << mini << endl;
	return 0;
}
