#include <iostream>
using namespace std;

int calc(int a, int b)
{
    if(a<1 || b<1)
        return 9999999;
    if(a==b && b==1)
        return 0;
    if(a>b)
        swap(a, b);
    return calc(b-a, a)+1;
}

int main()
{
    int n, ans;
    cin >> n;
    ans = n-1;
    for(int i=1;i<n;i++)
        ans = min(ans, calc(n, i));
    cout << ans << "\n";
}
