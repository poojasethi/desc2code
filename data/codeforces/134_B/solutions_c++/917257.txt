#include <iostream>
#include <cstdio>
#define inf 1<<30
using namespace std;

int n, ans;

int calc(int x, int y)
{
	if (x==1 && y==1) return 0;
	if (x == y) return inf;
	if (x > y) return calc(x-y, y) + 1;
	else return calc(x, y-x) + 1;

}
int main()
{
	cin >>n;
	ans = n-1;
	for (int i=1; i<=n; i++)
	  ans = min(ans, calc(n, i));

	cout <<ans <<endl;


	return 0;
}
