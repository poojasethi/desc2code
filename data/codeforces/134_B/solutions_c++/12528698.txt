#include <iostream>
using namespace std;
#define INF 1000000000

int n,ans=INF;
int t(int c,int d){
    if(d<=0)return INF;
    if(c>1&&d==1)return c-1;
    return t(d,c-c/d*d)+c/d;
}
int main(){
  cin>>n;
  for(int i=1;i<=n/2;i++){
    ans=min(ans,t(n,i));
  }
  if(n==0 || n==1){cout<<0;return 0;}
  cout<<ans<<endl;
}
