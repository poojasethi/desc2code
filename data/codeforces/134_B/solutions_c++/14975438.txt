#include<bits/stdc++.h>
using namespace std;
int f(int a,int b)
{
    if(a==1) return 0;
    if(a==b) return 1e9;
    if(a%b!=0)
    return a/b+f(b,a%b);
    else return a/b-1+f(b,b);
}
int main()
{
    int ans=1e9,n,i;
    scanf("%d",&n);
    for(i=1;i<=n;i++) ans=min(ans,f(n,i));
    cout<<ans;
}
