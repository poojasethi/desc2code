#include <cstdio>

#define INF 0x7fffff


int n;

int cal(int a, int b){
	if (a == 1 && b == 1)
		return 0;
	if (a == b)
		return INF;
	if (a > b)
		return cal(a-b,b)+1;
	else
		return cal(a,b-a)+1;
}


int main(){


	while (scanf("%d",&n) != EOF){
		int ans = INF;
		for (int i = 1 ; i <= n ; i++){
			if(ans > cal(i,n))
				ans = cal(i,n);
		}
		printf("%d\n",ans);
	}

}
