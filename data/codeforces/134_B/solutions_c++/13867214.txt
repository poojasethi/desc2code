#include <bits/stdc++.h>
using namespace std;

const int maxn = 5e5+100;

int f(int x , int y){
	if(x == 1 and y == 1) return 0;
	if(x < y) return f(x , y - x) + 1;
	return f(x - y , y) + 1;
}

int main(){
	int n; cin >> n;
	int ans = 100000000;
	for(int i = 1;i <= n;i++){
		int g = __gcd(i , n);
		if(g == 1) ans = min(ans , f(i , n));
	}
	return(cout << ans << endl , 0);
}
