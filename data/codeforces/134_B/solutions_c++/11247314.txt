
#include <iostream>
using namespace std;
const int INF=1000000000;
inline int Min(int x,int y){
  return x<y?x:y;
}
int dfs(int b,int s){
  if(s==1)
    return b-1;
  if(b%s==0)
    return  INF;
  return b/s+dfs(s,b%s);
}
int main(){
  int n,i;
  int ans=INF;
  cin>>n;
  for(i=1;i<=n;i++){
    ans=Min(ans,dfs(n,i));
  }
  cout<<ans;
 return 0;
}
