#include <iostream>
using namespace std;

int f(int x,int y)
{
    if(x==1&&y==1) return 0;
    if(x==y) return 10000000;
    if(x>y) return f(x-y,y)+1;
    return f(x,y-x)+1;
}

int main()
{
    int n,mini=1000000;
    cin>>n;
    for(int i=1;i<=n;i++)
        mini = min(mini,f(i,n));
    cout << mini << '\n';
    

    return 0;
}
