//In the Name of God

#include<iostream>

using namespace std;

const int inf=1e9+7;
int n,ans;

int dis(int x,int y)
{
  if(y==0)
    {
      if(x==1)
	return -1;
      else
	return inf;
    }
  return dis(y,x%y)+x/y;
}

int main()
{
  cin>>n;
  ans=inf;
  for(int i=1;i<=n;i++)
    ans=min(ans,dis(n,i));
  cout<<ans<<endl;
  return 0;
}
