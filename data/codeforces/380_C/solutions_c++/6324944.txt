#include <iostream>
#include <cstdio>
#include <vector>
#include <string>
#include <cstring>
using namespace std;

const int nmax=1e6+10;
int f[nmax][22],g[nmax],h[nmax],n;
string s;
void dp(){
	for(int i=0;i<n;i++)
		f[i][0]=g[i];
	for(int i=1;i<22;i++)
		for(int j=0;j<n;j++){
			if(j+(1<<i)-1>=n) break;
			f[j][i]=min(f[j][i-1],f[j+(1<<(i-1))][i-1]);
		}
}

int main(){
	//freopen("input.inp","r",stdin);
	ios::sync_with_stdio(false);
	
	cin >> s;
	n=s.size();
	h[0]=1;g[0]=-1;
	if(s[0]=='(') {h[0]--;g[0]=1;}
	for(int i=1;i<n;i++){
		h[i]=h[i-1]+1; 
		g[i]=g[i-1];
		if(s[i]=='('){
			g[i]++;
			h[i]--;
		}
		else g[i]--;
	}
	dp();
	int m;
	cin >> m;
	while(m--){
		int l,r,lg=0;
		cin >> l >> r;
		while((1<<lg)<=r-l+1)lg++;
		lg--;
		l--;r--;
		int t=min(f[l][lg],f[r-(1<<lg)+1][lg]);
		if(l){
			t-=g[l-1];
			if(t>0) t=0;
			cout<<(h[r]-h[l-1]+t)*2;
		}
		else {
			if(t>0) t=0;
			cout<<(h[r]+t)*2;
		}
		cout<<endl;
	}
}
