#include <stdio.h>
#include <string.h>
#include <math.h>
#define min(a, b) a<b?a:b
#define M 1000001

int dp[21][M];
char tmp[M];

int main(){
    ///freopen("3.in", "r", stdin);
    int i, j, n;
    scanf("%s", tmp);
    n=strlen(tmp);
    dp[0][0] = 0;
    for(i=0; i<n; i++)
        dp[0][i+1]=dp[0][i]+(tmp[i]=='('?1:-1);

    for(i=1; (1<<i)<=n+1; i++) for(j=0; j+(1<<i)-1<=n; j++)
        dp[i][j] = min(dp[i-1][j], dp[i-1][j + (1<<(i-1))]);

    int q;
    scanf("%d", &q);
    for(i=0; i<q; i++){
        int l, r, k, aa;
        scanf("%d %d",&l, &r);
        l--, k=log2(r-l+1);
        aa=min(dp[k][l], dp[k][r-(1<<k)+1]);
        printf("%d\n", (r-l)-(dp[0][l]+dp[0][r]-aa*2));
    }
    return 0;
}
