#include <iostream>
#include <set>
using namespace std;
#define N 100010
#define rep(i,a,b) for(int i=a; i<=b; i++)
#define repr(i,a,b) for(int i=a; i>=b; i--)
typedef long long ll;
int main(){
   ll n, a[N], b[N];
   ll m=0, s=0;
   set<ll> r;
   cin>>n;
   rep(i,0,n-1)cin>>a[i];
   rep(i,0,n-1)cin>>b[i];
   rep(i,1,n){
      s+=a[i-1]-b[i-1];
      m=min(s,m);
   }
   rep(i,0,n-1){
      if(m>=0)r.insert(i+1);
      m-=(a[i]-b[i]);
   }
   s=m=0;
   repr(i,n-1,0){
      s+=a[(i+1)%n]-b[i];
      m=min(s,m);
   }
   if(m>=0)r.insert(1);
   repr(i,n-1,1){
      m-=(a[(i+1)%n]-b[i]);
      if(m>=0)r.insert(i+1);
   }
   cout<<r.size()<<endl;
   for(set<ll>::iterator it=r.begin(); it!=r.end(); it++)cout<<(*it)<<" ";
   cout<<endl;
}
