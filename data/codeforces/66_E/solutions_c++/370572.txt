#include <iostream>
#include <set>
using namespace std;

int main() {
	int n; cin >> n;
	int a[n], b[n];
	for (int i = 0; i< n; ++i) cin >> a[i];
	for (int i = 0; i < n; ++i) cin >> b[i];
	int d[n];
	set<int> sol;
	
	d[0] = a[0] - b[0]; int D= d[0];
	for (int i = 1; i < n; ++i) D = min(d[i] = d[i-1] + a[i] - b[i], D);
	if (D >= 0) sol.insert(1);
	for (int i = 1; i < n; ++i) if (D - d[i-1] >= 0) sol.insert(i+1);
	
	int aux = b[n-1];
	for (int i = n-1; i >= 1; --i) b[i] = b[i-1]; b[0] = aux;
	
	d[n-1] = a[n-1] - b[n-1]; D = d[n-1];
	for (int i = n-2; i >= 0; --i) D = min(d[i] = d[i+1] + a[i] - b[i], D);
	if (D >= 0) sol.insert(n);
	for (int i = n-2; i >= 0; --i) if (D - d[i+1] >= 0) sol.insert(i+1);
	
	cout << sol.size() << endl;
	bool p = 1;
	for (set<int>::iterator i = sol.begin(); i != sol.end(); ++i) {
		(p)?p=0:cout<<" ";
		cout << *i; 
	} cout << endl;
}
