#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <cmath>
#include <cstring>
#include <algorithm>
#include <memory.h>
#define sqr(x) ((x)*(x))
#define min(a,b) ((a)<(b)?(a):(b))
#define max(a,b) ((a)>(b)?(a):(b))
#define swap(a,b) {int ttt=a;a=b;b=ttt;}
#define y1 stupid_math
#define pi 3.14159265359
typedef long long int64;
using namespace std;

int64 p[100005],m,c,a[100005], b[100005];
int n,i,j,k,l;
bool u[100005];

int main(){
    //freopen("input.txt","r",stdin);
   // freopen("output.txt","w",stdout);
	
	cin>>n;
	for(i=0;i<n;++i) cin>>a[i];
	for(i=0;i<n;++i) cin>>b[i];
	a[n]=a[0]; b[n]=b[0];
	
	m=0;
	c=0;
	p[0]=0;
	for(i=1;i<n;++i){
		p[i]=p[i-1]+a[i-1]-b[i-1];
		if(p[i]<m) m=p[i]; 
	}
	for(i=0;i<n;++i){
		if(m+c>=0) u[i]=1; 
		c-=(a[i]-b[i]);
	}
	
	m=0;
	c=0;
	p[n-1]=0;
	for(i=n-2;i>=0;--i){
		p[i]=p[i+1]+a[i+1]-b[i];
		if(p[i]<m) m=p[i]; 
	}
	for(i=n-1;i>=0;--i){
		if(m+c>=0) u[i]=1; 
		c-=(a[i]-b[(i-1+n)%n]);
	}
	
	//for(i=0;i<n;++i) cout<<p[i]<<' '; cout<<endl;
	
	
	m=0;
	for(i=0;i<n;++i) if(u[i]) p[m++]=i+1;
	
	cout<<m<<endl;
	for(i=0;i<m;++i) cout<<p[i]<<' ';
	
    return 0;
}
