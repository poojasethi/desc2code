
#include <cstdio>
#include <cstring>

const int N = 100010;
int a[N],b[N],c[N];
bool ok[N];
int ct,n;

void LR(int i)
{
	if(ok[i+1])return;
	ok[i+1]=true;
	ct++;
}

void RL(int i)
{
	if(ok[n-i])return ;
	ok[n-i]=true;
	ct++;
}

void Doit( void (*f)(int) )
{
	int min=c[0];	
	int sum=c[0];
	for(int i=1;i<n;++i){
		sum+=c[i];
		if(min>sum)min=sum;
	}
	if(min>=0) f(0);

	for(int i=1;i<n;++i){
		min -= c[i-1];
		if(min>0) min = 0;
		if(min>=0) f(i);
	}

}

int main()
{
	while(scanf("%d",&n)==1){
		ct=0;
		memset(ok,0,sizeof ok);
		for(int i=0;i<n;++i)scanf("%d",a+i);
		for(int i=0;i<n;++i)scanf("%d",b+i);
	
		for(int i=0;i<n;++i)c[i]=a[i]-b[i];
		Doit(LR);
		for(int i=0;i<n;++i)c[i]=a[ n-i-1 ]-b[ (n-i-2+n)%n ];
		Doit(RL);
		
		printf("%d\n",ct);
		for(int i=1;i<=n;++i){
			if(ok[i]){
				if(ct>1){
					ct--;
					printf("%d ",i);
				}
				else printf("%d\n",i);
			}
		}
	
	}

	return 0;
}
