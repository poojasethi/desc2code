#include <stdio.h>
#include <algorithm>

const int kMaxN = 100000;

int a[kMaxN], b[kMaxN], sum[kMaxN];
bool ans[kMaxN];
int n;
bool op;

inline int ida(int id) { return op ? n - 1 - id : id; }
inline int idb(int id) { return op ? (n - 2 - id + n) % n : id; }

void Check() {
  sum[0] = a[ida(0)] - b[idb(0)];
  int min = sum[0];
  for (int i = 1; i < n; ++i) {
    sum[i] = sum[i - 1] + a[ida(i)] - b[idb(i)];
    min = std::min(min, sum[i]);
  }
  if (min >= 0) ans[ida(0)] = true;
  for (int i = 1; i < n; ++i)
    if (sum[i - 1] == min) ans[ida(i)] = true;
}

int main() {
  scanf("%d", &n);
  for (int i = 0; i < n; ++i) scanf("%d", &a[i]);
  for (int i = 0; i < n; ++i) scanf("%d", &b[i]);
  op = false;
  Check();
  op = true;
  Check();
  int cnt = 0;
  for (int i = 0; i < n; ++i) if (ans[i]) ++cnt;
  printf("%d\n", cnt);
  bool printed = false;
  for (int i = 0; i < n; ++i)
    if (ans[i]) {
      if (printed) printf(" ");
      printed = true;
      printf("%d", i + 1);
    }
  puts("");
  return 0;
}
