#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
const int mm=1e5+9;
const int oo=1e9;
int a[mm],b[mm];
bool vis[mm];
int main()
{
  int n;
  while(cin>>n)
  { memset(vis,0,sizeof(vis));
    for(int i=0;i<n;++i)cin>>a[i];
    for(int i=0;i<n;++i)cin>>b[i];
    int mi=oo,z=0,mmi=oo,zz=0;
    for(int i=0;i<n;++i)
    {z+=a[i]-b[i];zz+=a[n-i-1]-b[(n-i-2+n)%n];
     mi=min(mi,z);mmi=min(mmi,zz);
    }
    int num=0;
    for(int i=0;i<n;++i)
    {
      if(mi>=0&&!vis[i+1])vis[i+1]=1,++num;
      if(mmi>=0&&!vis[n-i])vis[n-i]=1,++num;
      mi-=a[i]-b[i];
      mmi-=a[n-i-1]-b[(n-i-2+n)%n];
    }
    cout<<num<<"\n";
    for(int i=0;i<=n;++i)
      if(vis[i])
      cout<<i<<" ";
    cout<<"\n";
  }
}
