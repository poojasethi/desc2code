#include <cstdlib>
#define FOR(i,a,n) for(int i=(a);i<(n);i++)
#define REP(i,n) FOR(i, 0, n)
#define mp make_pair
#define pb push_back
#define all(v) v.begin(),v.end()
#define lint long long
#define PII pair <int,int>
#define PDD pair <double,double>
#define VI vector <int>
#define VII vector < PII >
#define X first
#define Y second
#define sz(a) (int)(a.size())
using namespace std;

const int DIM = 110*1000;
int n;
int a[DIM], b[DIM];
lint ch[DIM];
char us[2][DIM];
void solve(int it){
	REP(i, n-1){
		ch[i+1] = ch[i]+a[i]-b[i];
	}
	lint m = *min_element(ch, ch+n);
	REP(i, n) if (ch[i] == m) us[it][i] = 1;
}
int main(){
	scanf("%d", &n);
	REP(i, n) scanf("%d", &a[i]);
	REP(i, n) scanf("%d", &b[i]);
	solve(0);
	reverse(a, a+n);
	reverse(b, b+n);
	rotate(b, b+1, b+n);
	solve(1);
	VI v;
	REP(i, n) if (us[0][i] || us[1][n-1-i]) v.pb(i+1);
	printf("%d\n", sz(v));
	REP(i, sz(v)) printf(" %d"+!i, v[i]);
	if (sz(v)) printf("\n");
	return 0;
}
