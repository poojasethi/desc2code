#include<iostream>
#include<set>
#include<algorithm>

using namespace std;

int n,a[100000],b[100000];
set<int>  s;

void g(int k)
{
	int mini,i,pred;

	mini=pred=a[0]-b[0];
	for(i=1;i<n;i++)
	{
		pred+=a[i]-b[i];
		mini=min(pred,mini);
	}

	for(i=0;i<n;i++)
	{      	
		if(mini>=0) 
		{
			if(k>0)  s.insert(i+1);
			else s.insert(n-i);
		}

		mini-=a[i]-b[i];
	}
}

int main()
{
	int i;

	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
		cin>>b[i];
		
	g(1);

	reverse(a,a+n);
	reverse(b,b+n-1);

	g(-1);

	cout<<s.size()<<"\n";
	while(!s.empty())
	{
		cout<<*(s.begin())<<" ";
		s.erase(s.begin());
	}

	return 0; 
}
