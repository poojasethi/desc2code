#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int a[1<<17];
int b[1<<17];

int main()
{
    int n;
    cin >> n;

    for (int i=0; i<n; i++)
        cin >> a[i];

    for (int i=0; i<n; i++)
        cin >> b[i];

    a[n] = a[0];
    b[n] = b[0];

    vector<int> res1, res2;
    long long m = 0;
    long long c = 0;

    for (int i=0; i<n; i++)
    {
        if (c < m)
        {
            res1.clear();
            c = m;
        }

        if (c == m)
            res1.push_back(i);

        c += a[i] - b[i];
    }

    c = 0;
    m = 0;

    for (int i=n; i; i--)
    {
        if (c < m)
        {
            res2.clear();
            c = m;
        }

        if (c == m)
            res2.push_back(i%n);

        c += a[i] - b[i-1];
    }

    for (int i=0; i<res2.size(); i++)
        res1.push_back(res2[i]);

    sort(res1.begin(), res1.end());
    res1.resize(unique(res1.begin(), res1.end()) - res1.begin());

    cout << res1.size() << endl;
    for (int i=0; i<res1.size(); i++)
        cout << res1[i]+1 << " ";
    cout << endl;

    return 0;
}
