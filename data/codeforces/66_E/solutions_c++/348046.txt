#include<iostream>
#include<algorithm>
#define msize 100005

using namespace std;

int a[msize];
int b[msize];
int d[msize]={0,};
int m[msize];
int can[msize]={0,};
int index[msize];

int n;
int ans=0;

void calc(){
    m[1]=a[1]-b[1];
    for(int i=1;i<=n;i++){
        d[i]=d[i-1]+a[i]-b[i];
        if(m[1]>d[i]) m[1]=d[i];
    }
    
    for(int i=2;i<=n;i++) m[i]=m[i-1]-(a[i-1]-b[i-1]);
    for(int i=1;i<=n;i++) if(m[i]>=0) can[index[i]]=1;

}

void rebirth(){
    int t;

    for(int i=1;i<=n/2;i++){
        swap(a[i],a[n-i+1]);
        swap(b[i],b[n-i+1]);
        swap(index[i],index[n-i+1]);
    }
    t=b[1];
    for(int i=1;i<n;i++) b[i]=b[i+1];
    b[n]=t;

    

}

int main(){

    cin>>n; 
    for(int i=1;i<=n;i++) index[i]=i;
    for(int i=1;i<=n;i++) cin>>a[i];
    for(int i=1;i<=n;i++) cin>>b[i];
    calc();
    rebirth();
    calc();

    for(int i=1;i<=n;i++) if(can[i]) ans++;
    cout<<ans<<endl;
    for(int i=1;i<=n;i++) if(can[i]) cout<<i<<' ';
    

    return 0;

}
