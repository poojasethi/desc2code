#include <iostream>
#include <vector>
#include <cmath>
#include <set>
#include <algorithm>

using namespace std;

void calculaminimos(int n,vector<int> &a,vector<int> &b,set<int> &sol,
		    bool invertir)
{
  vector<int> c(n);
  c[0]=0;
  int minimo=0;
  for (int i=1;i<n;i++) {
    c[i]=c[i-1]+a[i-1]-b[i-1];
    minimo=min(minimo,c[i]);
  }
  for (int i=0;i<n;i++) {
    if (c[i]==minimo) {
      if (not invertir) 
	sol.insert(i+1);
      else
	sol.insert((n-1-i)+1);
    }
  }
}

int main()
{
  int n;
  cin>>n;
  vector<int> a(n);
  for (int i=0;i<n;i++)
    cin>>a[i];
  vector<int> b(n);
  for (int i=0;i<n;i++)
    cin>>b[i];
  set<int> sol;
  calculaminimos(n,a,b,sol,false);


  reverse(a.begin(),a.end());
  reverse(b.begin(),b.end());
  {
    vector<int> baux(n);
    for (int i=0;i<n-1;i++)
      baux[i]=b[i+1];
    baux[n-1]=b[0];
    b=baux;
  }
  calculaminimos(n,a,b,sol,true);


  cout<<int(sol.size())<<endl;
  bool primer=true;
  for (set<int>::iterator it=sol.begin();it!=sol.end();it++) {
    if (not primer) cout<<" ";
    primer=false;
    cout<<*it;
  }
  cout<<endl;
}
