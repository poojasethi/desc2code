#include<set>
#include<cstdio>

#include<cstring>
#include<algorithm>
using namespace std;
int n,a[100003],b[100003],c[100003];
set<int> h;
bool ans[100003];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;++i)scanf("%d",a+i);
	for(int i=1;i<=n;++i)scanf("%d",b+i);
	for(int i=1;i<=n;++i)c[i]=a[i]-b[i];
	h.clear();
	int sum=0;
	for(int i=1;i<=n;++i){
		sum+=c[i];
		h.insert(sum);
	}
	int v=0;
	for(int i=1;i<=n;++i){
		if(*(h.begin())-v>=0)ans[i]=1;
		v+=c[i];
		h.erase(v);
		h.insert(v+sum);
	}
	h.clear();c[n+1]=sum=0;
	rotate(b+1,b+n,b+n+1);
	for(int i=1;i<=n;++i)c[i]=a[i]-b[i];
	for(int i=n;i;--i){
		sum+=c[i];
		h.insert(sum);
	}
	v=0;
	for(int i=n;i;--i){
		if(*h.begin()-v>=0)ans[i]=1;
		v+=c[i];
		h.erase(v);
		h.insert(v+sum);
	}
	int cnt=0;
	for(int i=1;i<=n;++i)
		cnt+=ans[i];
	printf("%d\n",cnt);
	for(int i=1;i<=n;++i)
		if(ans[i]){
			printf("%d",i);
			if(--cnt==0)puts("");
			else putchar(' ');
		}
	return 0;
}
