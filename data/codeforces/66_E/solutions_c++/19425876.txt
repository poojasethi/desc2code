#include <iostream>
#include <cstdio>
using namespace std;

const int maxn=111111;
const int OO=1e9;

int gas[maxn],dis[maxn];
bool v[maxn];
int n,Dk,sum,ans;

int main() {
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&gas[i]);
    for(int i=0;i<n;i++) scanf("%d",&dis[i]);

    Dk=OO;
    sum=0;
    for(int i=0;i<n;i++)
    {
        sum+=gas[i]-dis[i];
        Dk=min(Dk,sum);
    }

    ans=0;
    for(int i=0;i<n;i++)
    {
        if(Dk>=0 && !v[i])
        {
            v[i] = true;
            ans++;
        }
        Dk-=gas[i]-dis[i];
    }

    Dk=OO;
    sum=0;
    for(int i=n-1;i>=0;i--)
    {
        sum+=gas[i]-dis[(i-1+n)%n];
        Dk=min(Dk,sum);
    }

    for(int i=n-1;i>=0;i--)
    {
        if(Dk >= 0 && !v[i])
        {
            v[i]=true;
            ans++;
        }
        Dk-=gas[i]-dis[(i-1+n)%n];
    }

    printf("%d\n", ans);
    for(int i=0;i<n;i++)
    {
        if(v[i])
        {
            printf("%d ", i+1);
        }
    }
    printf("\n");
    return 0;
}
