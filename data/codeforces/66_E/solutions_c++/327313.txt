#include<iostream>
#include<deque>
#include<vector>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;

#define REP(i,n) for(int i=0;i<(int)(n);++i)
#define SZ(a) ((int)((a).size()))
#define REPSZ(i,v) REP(i,SZ(v))
#define ALL(a) (a).begin(),(a).end()
template<class T>void pv(T a,T b) {for(T i = a;i!=b;++i)cerr<<*i<<' ';cout<<endl;}
typedef long long Int;
typedef vector<Int> Vi;

int ans[100000 + 10];

int main(){
  for(int N; ~scanf("%d", &N); ) {
    memset(ans, 0, sizeof(ans));
    vector<int> a(N), b(N);
    REP(i, N) scanf("%d", &a[i]);
    REP(i, N) scanf("%d", &b[i]);

    Vi c(N);
    REP(i, N) c[i] = a[i] - b[i];
    for(int i = 1; i < N; i++)
      c[i] += c[i - 1];
    int small = *min_element(ALL(c));
    REP(i, N) if ( c[i] == small ) ans[(i + 1) % N] = 1;

    REP(i, N) c[i] = a[i] - b[(i - 1 + N) % N];
    for(int i = N - 2; i >= 0; i--)
      c[i] += c[i + 1];
    small = *min_element(ALL(c));
    REP(i, N) if ( c[i] == small ) ans[(i - 1 + N) % N] = 1;

    cout << count(ans, ans + N, 1) << endl;
    REP(i, N) if( ans[i] ) cout << i + 1 << ' ';
    cout << endl;
  }
  return 0;
}
