#include<cstdio>
#include<algorithm>
using namespace std;

const int N = (int) 1e5 + 5;
int a[N], b[N];
bool mark[N];

int main() {
#ifndef ONLINE_JUDGE
    freopen("20.inp", "r", stdin);
    freopen("20.out", "w", stdout);
#endif
    int n; scanf("%d",&n);
    for(int i = 0; i < n; ++i) scanf("%d", a+i);
    for(int i = 0; i < n; ++i) scanf("%d", b+i);
    int minn = 1 << 30;
    for(int i = 0, sum = 0; i < n; ++i) minn = min(minn, sum += a[i] - b[i]);
    int res = 0;
    for(int i = 0; i < n; ++i) {
        if(minn >= 0 && !mark[i]) mark[i] = 1, ++res; 
        minn -= a[i] - b[i];
    }
    minn = 1 << 30;
    for(int i = n-1, sum = 0; i >= 0; --i) minn = min(minn, sum += a[i] - b[(i-1+n)%n]);
    for(int i = n-1; i >= 0; --i) {
        if(minn >= 0 && !mark[i]) mark[i] = 1, ++res;
        minn -= a[i] - b[(i-1+n)%n];
    }
    printf("%d\n", res);
    for(int i = 0, cnt = 0; i < n; ++i) if(mark[i]) {
        if(cnt++) printf(" ");
        printf("%d", i+1);
    }
    printf("\n");
    return 0;
}
