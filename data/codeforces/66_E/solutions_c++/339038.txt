#include <iostream>
#include <cmath>
#include <stdio.h>
#define FOR(a,b,c) for(long a=b;a<=c;a++)
#define FORR(a,b,c) for(long a=b;a>=c;a--)
using namespace std;
long long a[100010],l[100100],r[100010];
long long b[100010],lb[100010],rb[100010];
long n;
int main(){
scanf("%ld",&n);
FOR(i,1,n)scanf("%ld",&a[i]);
long old=0;
FOR(i,1,n){
           long u;
           scanf("%ld",&u);
           b[i]=a[i]-old;
           a[i]-=u;
           a[i]+=a[i-1];
           old=u;
           }
b[1]-=old;
FORR(i,n,1)b[i]+=b[i+1];
l[1]=a[1];
FOR(i,2,n)l[i]=min(a[i],l[i-1]);
r[n]=a[n];
FORR(i,n-1,1)r[i]=min(a[i],r[i+1]);
lb[1]=b[1];
FOR(i,2,n)lb[i]=min(b[i],lb[i-1]);
rb[n]=b[n];
FORR(i,n-1,1)rb[i]=min(b[i],rb[i+1]);
long dem=0;
FOR(i,1,n)dem+=((r[i]-a[i-1]>=0 && l[i]+a[n]-a[i-1]>=0) || (lb[i]-b[i+1]>=0 && rb[i]+b[1]-b[i+1]>=0)  );
printf("%ld \n",dem);
FOR(i,1,n)
          if ((r[i]-a[i-1]>=0 && l[i]+a[n]-a[i-1]>=0) || (lb[i]-b[i+1]>=0 && rb[i]+b[1]-b[i+1]>=0)  )printf("%ld ",i);
return 0;
}
