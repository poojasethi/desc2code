using namespace std ;

typedef pair<int,int> PII ;
typedef pair<int,PII> PIP ;
typedef pair<PII,int> PPI ;

const int MAXN = (int)1e5 + 10 ;

int a[MAXN] , b[MAXN] , n ;
int ans[MAXN] ;

void gao(){
    int s = 0 , m ;
    for (int i=0;i<n;i++)
    {
        s += a[i]-b[i];
        m = min(s , m) ;
    }
    for (int i=n-1;i>=0;i--){
        m += a[i]-b[i];
        if (a[i]-b[i]>=0 && m >= 0)
            ans[i] = true ;
    }
}

int main(){
    scanf("%d",&n);
    for (int i=0;i<n;i++) scanf("%d",&a[i]);
    for (int i=0;i<n;i++) scanf("%d",&b[i]);
    memset(ans,0,sizeof(ans)) ;
    gao() ;
    reverse(a,a+n);
    reverse(b,b+n-1);
    reverse(ans,ans+n);
    gao();
    reverse(ans,ans+n);
    vector<int> lis ;
    for (int i=0;i<n;i++) if (ans[i]) lis.push_back(i);
    printf("%d\n",lis.size());
    for (int i=0;i<lis.size();i++)
        printf("%d ",lis[i]+1);
    printf("\n");
    return 0 ;
}
