#include<stdio.h>
#include<string.h>
const int N = 100005;
int a[N],b[N],aa[N],bb[N],r;
__int64 D,d[N];
bool v[N];
int main()
{
	int n,i;
	scanf("%d",&n);
	memset(v,false,sizeof(v));
	r=0;
	for(i=1;i<=n;i++)
		scanf("%d",a+i);
	for(i=1;i<=n;i++)
		scanf("%d",b+i);
	//˳ʱ��
	for(D=1000000001,i=1;i<=n;i++)
	{
		d[i]=d[i-1]+a[i]-b[i];
		//printf("i:%d a:%d b:%d d:%d \n",i,a[i],b[i],d[i]);
		if(d[i]<D)D=d[i];
	}
	for(i=1;i<=n;i++)
	{
		D-=a[i-1]-b[i-1];
		//printf("i:%d D:% d\n",i,D);
		if(D>=0){r++;v[i]=true;}
	}
	//��ʱ��
	for(i=1;i<=n;i++)
	{
		bb[i]=b[n+1-i];
		aa[i]=a[(n+1-i)%n+1];
	}
	for(D=1000000001,i=1;i<=n;i++)
	{
		d[i]=d[i-1]+aa[i]-bb[i];
		//printf("i:%d a:%d b:%d d:%d \n",i,a[i],bb[i],d[i]);
		if(d[i]<D)D=d[i];
	}
	for(i=1;i<=n;i++)
	{
		D-=aa[i-1]-bb[i-1];
		//printf("i:%d D:% d\n",i,D);
		if(D>=0&&!v[(n+1-i)%n+1]){r++;v[(n+1-i)%n+1]=true;}
	}
	//
	printf("%d\n",r);
	int f=0;
	for(i=1;i<=n;i++)
		if(v[i]){if(f)putchar(' ');f=1;printf("%d",i);}
	puts("");
}
