#include <stdio.h>
#include <iostream>
#include <algorithm>
using namespace std;

int a[100000];
int d[100000];
int b1[100000];
int b2[100000];

int main()
{
//	freopen("in.txt","r",stdin);
	int n,i;
	scanf("%d",&n);
	for (i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	for (i=0;i<n;i++)
	{
		scanf("%d",&d[i]);
	}
	b1[0]=0;
	int min1=1000000000;
	for (i=1;i<n;i++)
	{
		b1[i]=b1[i-1]+a[i-1]-d[i-1];
		if (b1[i]<min1) min1=b1[i];
	}
	if (min1>0) min1=0;
	b2[n-1]=0;
	int min2=1000000000;
	for (i=n-2;i>=0;i--)
	{
		b2[i]=b2[i+1]+a[i+1]-d[i];
		if (b2[i]<min2) min2=b2[i];
	}
	if (min2>0) min2=0;
	int rn=0;
	for (i=0;i<n;i++)
	{
		if (b1[i]==min1 || b2[i]==min2) rn++;
	}
	/*
	for (i=0;i<n;i++)
	{
		printf("%d ",b1[i]);
	}
	printf("\n");
	for (i=0;i<n;i++)
	{
		printf("%d ",b2[i]);
	}
	printf("\n");
	*/

	printf("%d\n",rn);

	int count=0;
	for (i=0;i<n;i++)
	{
		if (b1[i]==min1 || b2[i]==min2) 
		{
			if (count==rn-1) break;
			printf("%d ",i+1);
			count++;
		}
	}
	printf("%d\n",i+1);
	return 0;
}
