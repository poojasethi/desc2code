#include<cstdio>
#include<iostream>
#include<vector>
#include<algorithm>
#include<cstring>
#include<cmath>
#include<map>
#include<set>
using namespace std;

int n,a[100005],b[100005],rec[100005],tot=0;

int main() {
	scanf("%d",&n);
	for (int i=0; i<n; i++)
		scanf("%d",&a[i]);
	for (int i=0; i<n; i++)
		scanf("%d",&b[i]);
	memset(rec,0,sizeof(rec));
	int mi=2147483647,sum=0;
	for (int i=0; i<n; i++) {
		sum+=a[i]-b[i];
		mi=min(mi,sum);
	}
	sum=0;
	for (int i=0; i<n; i++) {
		sum+=a[i]-b[i];
		if (sum==mi && rec[(i+1)%n]==0) {
			rec[(i+1)%n]=1; tot++;
		}
	}
	mi=2147483647; sum=0;
	for (int i=n-1; i>=0; i--) {
		sum+=a[i]-b[(i+n-1)%n];
		mi=min(mi,sum);
	}
	sum=0;
	for (int i=n-1; i>=0; i--) {
		sum+=a[i]-b[(i+n-1)%n];
		if (sum==mi && rec[(i+n-1)%n]==0) {
			rec[(i+n-1)%n]=1; tot++;
		}
	}
	printf("%d\n",tot);
	for (int i=0; i<n; i++) {
		if (rec[i]) {
			printf("%d",i+1);
			tot--;
			if (tot) printf(" ");
		}
	}
	printf("\n");
}
