using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define pb push_back

int n,i,x;
bool M[100001];

void f(vector <int> A,vector <int> B)
{
  int i,mn=1,s,x;
  vector <int> V;

  for(i=x=0;i<n;i++)
  {
    A[i]-=B[i];
    if(x<mn)
    {
      mn=x;
      V.clear();
    } 
    if(x==mn)
      V.pb(i);
    x+=A[i];
  }
  for(i=0;i<V.size();i++)
    M[V[i]]++;
}
int main()
{
  scanf("%d",&n);
  vector <int> A(n),B(n);
  for(i=0;i<n;i++)
    scanf("%d",&A[i]);
  for(i=0;i<n;i++)
    scanf("%d",&B[i]);
  f(A,B);
  for(i=0;i<n-1-i;i++)
    swap(M[i],M[n-1-i]);
  x=B[n-1];
  B.pop_back();
  B.insert(B.begin(),x);
  reverse(A.begin(),A.end());
  reverse(B.begin(),B.end());
  f(A,B);
  for(i=x=0;i<n;i++)
    if(M[i])
      x++;
  printf("%d\n",x);
  reverse(M,M+n);
  for(i=0;i<n;i++)
    if(M[i])
      printf("%d ",i+1);

}
