using namespace std;
#define li		int
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb		push_back
#define sz(v)		((li)v.size())


int main(){
	li n;
	cin>>n;
	vector<li> a(n),b(n);
	rep(i,n) cin>>a[i];
	rep(i,n) cin>>b[i];
	vector<bool> left(n,true),right(n,true);
	li rem=0;
	rep(i,2){
		rep(j,n){
			li k=(j-1+n)%n;
			rem+=a[j];
			rem-=b[k];
			if(rem<0) left[j]=false;
			else rem=0;
		}
	}
	rem=0;
	rep(i,2){
		for(int j=n-1;j>=0;j--){
			rem+=a[j];
			rem-=b[j];
			if(rem<0) right[j]=false;
			else rem=0;
		}
	}
	vector<int> res;
	rep(i,n)if(left[i] || right[i]) res.pb(i);
	cout<<sz(res)<<endl;
	rep(i,sz(res)) cout<<res[i]+1<<" "; cout<<endl;
			
}
