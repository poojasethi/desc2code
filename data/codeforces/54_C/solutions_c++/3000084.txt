#define LL long long
const double pi=acos(-1.0),e=exp(1.0);
using namespace std;

double f[1100],a[1100];

LL calc(LL x){
    LL res=0,t=1,X=x;
    while (x){
        if (x/10 || x%10>1) res+=t;
        else if (x/10==0 && x%10==1) res+=X-t+1;
        x/=10; t*=10;
    }
    return res;
}

int main(){
    int N,K,i,j;
    LL L,R;
    cin >> N;
    for (i=1;i<=N;i++){
        cin >> L >> R;
        a[i]=(double)(calc(R)-calc(L-1))/(R-L+1);
    }
    cin >> K;
    f[0]=1;
    for (i=1;i<=N;i++){
        for (j=N;j>=1;j--) f[j]=f[j-1]*a[i]+f[j]*(1.0-a[i]);
        f[0]=f[0]*(1-a[i]);
    }
    double ans=0;
    for (j=0;j<=N;j++)
        if (j*100>=K*N) ans+=f[j];
    printf("%.15f\n",ans);
}
