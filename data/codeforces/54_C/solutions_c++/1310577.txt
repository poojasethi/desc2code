#include <iostream>
#include <iomanip>
#include <algorithm>
using namespace std;

typedef unsigned long long llu;
typedef long double ld;

llu f(llu a) {
    llu cnt = 0, pow = 1;
    for (int i = 0; pow <= a && i <= 18; i++) {
        llu hi = pow * 2 - 1;
        cnt += min(hi, a) - pow + 1;
        pow *= 10;
    }
    return cnt;
}
ld dp[1000];
int main() {
    int n;
    cin >> n;
    dp[0] = 1;
    for (int i = 0; i < n; i++) {
        llu l, r;
        cin >> l >> r;
        llu poss = f(r) - f(l - 1);
        ld prob = ld(poss) / (r - l + 1), comp = 1 - prob;
        for (int j = i + 1; j >= 0; j--) {
            dp[j] *= comp;
            if (j) dp[j] += dp[j - 1] * prob;
        }
    }
    int perc;
    cin >> perc;
    int lim = (n * perc + 99) / 100;
    ld sum = 0;
    for (int i = lim; i <= n; i++) {
        sum += dp[i];
    }
    cout << fixed << setprecision(12) << sum << endl;
}
