#include <iostream>
#include <math.h>
#include <stdio.h>
using namespace std;
typedef long long ll;
#define N 1010
ll f(ll y){
   ll c=1, r=0, x=y;
   while(x>1){
      r+=c;
      c*=10;
      x/=10;
   }
   if(x)r+=(y-c+1);
   return r;
}
int msb(ll y){
   while(y>1){
      y/=10;
   }
   return y;
}
int main(){
   ll l[N], r[N], n, k;
   ll i, j, c1, c2, c;
   double p[N], dp[N][N];
   cin>>n;
   i=-1;
   while(++i < n){
      cin>>l[i]>>r[i];
      p[i]=(f(r[i])-f(l[i])+msb(l[i]))*1.0/(r[i]-l[i]+1);
   }
   cin>>k;
   k=ceil(k*n*1.0/100);
   for(i=0; i<k; i++)dp[0][i]=0.0;
   dp[0][0]=1.0-p[0];
   dp[0][1]=p[0];
   for(i=1; i<n; i++){
      for(j=0; j<k; j++){
         dp[i][j]=(1.0-p[i])*dp[i-1][j]+((j>0)?(p[i]*dp[i-1][j-1]):0);
      }
   }
   double ret=0;
   for(i=0; i<k ;i++)ret+=dp[n-1][i];
   printf("%.15f\n",1.0-ret);
   return 0;
}
