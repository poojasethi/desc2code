#include<bits/stdc++.h>
using namespace std;

typedef long long ll ;

int n , K ;
ll l , r ;
double p[1001][1001] , ans=0.0 ;

ll totNum(ll n){
  ll ten=1 , tmp=n , ans=0 ;
  while(tmp){
    ans += ten ;
    ten *= 10 ; tmp /= 10 ;
  } 
  ten /= 10 ;
  if( ten && n/ten == 1 ) ans += (n%ten)+1-ten ;
  return ans ;       
}

int main(){
  cin >> n ;
  
  p[0][0]=1.0 ;
  for(int i=1 ; i<=n ; i++){
    cin >> l >> r ;
    double prob = (double)(totNum(r)-totNum(l-1))/(r-l+1) ;
    for(int j=0 ; j<=i ; j++){
       p[i][j] = p[i-1][j]*(1-prob) ;
       if(j)  p[i][j] +=  prob*p[i-1][j-1] ;
    }    
  }
  
  cin >> K ;
  for(int j=0 ; j<=n ; j++)  if((1.0*j*100/n)>=K) ans += p[n][j] ;
  
  printf("%.9lf\n",ans) ;
  
  return 0;
}
