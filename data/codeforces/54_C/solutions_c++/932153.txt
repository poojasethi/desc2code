#include<cstdio>
#include<vector>
#include<string>
#include<climits>
#include<cstdlib>
#include<string.h>
#include<iostream>
#include<algorithm>
#define LL long long
#define _max(a, b) ((a) > (b) ? (a) : (b))
#define _min(a, b) ((a) < (b) ? (a) : (b))
using namespace std;
#define N 1111

LL x, y, l, r;
int k, n;
double p[N], f[N][N], ans = 0.0;
int main()
{
	cin>> n;
	for (int i = 0; i < n; i++)
	{
		k = 1;
		cin>> l>> r;
		double prob = 0.0;
		for (LL q = 1; k <= 19; q *= 10ll, k++)
		{
			x = _max(l, q);
			y = _min(r, 2ll * q - 1);
			if (x <= y) prob += y - x + 1;
		}
		p[i] = prob / (r - l + 1);
	}
	cin>> k;
	f[0][0] = 1;
	for (int i = 0; i < n; i++)
		for (int j = 0; j <= i; j++)
		{
			f[i + 1][j + 1] += f[i][j] * p[i];
			f[i + 1][j] += f[i][j] * (1.0 - p[i]);
		}
	for (int i = 0; i <= n; i++) if (i * 100 >= k * n) ans += f[n][i];
	cout.precision(13);	
	cout<< fixed<< ans<< endl;
	return 0;
}
