#include <iostream>
#include <iomanip>

using namespace std;

long double p[1100];
long double d[1100][1100];
long double ans;

int main()
{
	int n, per;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		unsigned long long l, r, f = 1ull, tmp = 0ull;
		cin >> l >> r;
		for (int j = 0; j < 19; j++)
		{
			unsigned long long f2 = f * 2 - 1, f1 = f;
			f1 = max(f1, l);
			f2 = min(f2, r);
			if (f2 >= f1)
				tmp += f2 - f1 + 1;
			f *= 10;
		}
		p[i] = (long double)tmp / (long double)(r - l + 1);
	}
	cin >> per;
	d[0][0] = 1 - p[0];
	d[0][1] = p[0];
	for (int i = 1; i < n; i++)
		for (int j = 0; j <= i + 1; j++)
		{
			d[i][j] = d[i - 1][j] * (1 - p[i]);
			if (j > 0)
				d[i][j] += d[i - 1][j - 1] * p[i];
		}
	for (int i = 0; i <= n; i++)
		if (per * n <= i * 100)
			ans += d[n - 1][i];
	cout << setprecision(10) << fixed << ans << endl;
	return 0;
}
