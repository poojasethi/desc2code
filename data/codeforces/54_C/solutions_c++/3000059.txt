#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;
long long cal(long long n)
{
    if(n==0)return 0;
    long long tmp=n;
    int len=0;
    long long ans=0;
    long long po=1;
    while(n>9)len++,n/=10;len++;
    if(n==1)
    {
        for(int i=0;i<=len-2;i++)ans+=po,po*=10;
        ans+=tmp-po+1;
    }
    else
    {
        for(int i=0;i<=len-1;i++)ans+=po,po*=10;
    }
    return ans;
}
const int maxn=1010;
double p[maxn];
double dp[maxn][maxn];
int n;
int main()
{
    while(~scanf("%d",&n))
    {
        for(int i=1;i<=n;i++)
        {
            long long l,r;
            scanf("%lld %lld",&l,&r);
            p[i]=(cal(r)-cal(l-1))*1.0/(r-l+1);
        }
        dp[0][0]=1;
        for(int i=1;i<=n;i++)
        for(int j=0;j<=i;j++)
        dp[i][j]=(dp[i-1][j]*(1-p[i])+dp[i-1][j-1]*p[i]);
        double ans=0;
        int K;
        scanf("%d",&K);
        for(int i=0;i<=n;i++)if(100*i>=K*n)ans+=dp[n][i];
        printf("%.10f\n",ans);
    }
}
