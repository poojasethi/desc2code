using namespace std;
double dp[1005][1005];
long long ones(long long x) {
	// how many 1 thingies from 1..x?
	// 1, 10-19, 100-199, 1000-1999, ...
	long long ret = 0;
	for(unsigned long long i=1;i<=x;i*=10) {
		if(i*2-1 <= x) ret += i;
		else ret += (x-i+1);
	}
	return ret;
}
int main() {
	int N,K;
	double ans = 1;
	scanf("%d",&N);
	for(int i=0;i<N;++i) {
		long long l,r;
		cin >> l >> r;
		// count how many 1 thingies
		long long o = ones(r) - ones(l-1);
		long long ran = r-l+1;
		double p = o/double(ran);
		dp[i][0] = 1;
		if(!i) dp[i][1] = p;
		else {
			for(int j=1;j<=i+1;++j)
				dp[i][j] = dp[i-1][j] * (1-p) + dp[i-1][j-1] * p;
		}
	}
	scanf("%d",&K);
	K = (N*K+99)/100;
	printf("%.9lf\n",dp[N-1][K]);
}
