#include <stdio.h>
#include <iostream>
using namespace std;
int n, i, j, k, kk;
long long l, r, x, len, kol;
double ans[1500][1500], res;
int main()
{
	 //freopen("input.txt", "r", stdin);
	 //freopen("output.txt", "w", stdout);
	 cin >> n;
	 ans[0][0] = 1;
	 for (i = 1; i <= n; i++)
	 {
		 cin >> l >> r;
		 x = 1; kol = 0;
		 for (j = 1; j <= 19; j++)
		 {
			 len = min(r, x * 2 - 1) - max(l, x) + 1;
			 if (len > 0) kol += len;
			 x = x * 10;
		 }
		 len = r - l + 1;
		 ans[i][0] = (ans[i - 1][0] * (len - kol)) / len;
		 for (j = 1; j <= n; j++)
		 {
			 ans[i][j] = (ans[i - 1][j] * (len - kol) + ans[i - 1][j - 1] * kol) / len;
		 }
	 }
	 cin >> k;
	 kk = (n * k) / 100;
	 if ((n * k) % 100 != 0) kk++;
	 res = 0;
	 for (j = kk; j <= n; j++)
	 {
		 res += ans[n][j];
	 }
	 cout.precision(30);
	 cout << res;
	 return 0;	 
}
