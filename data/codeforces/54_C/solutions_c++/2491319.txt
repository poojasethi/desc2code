#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<cmath>
#define fou(i,j,k) for (int i=j;i<=k;i++)
#define fod(i,j,k) for (int i=j;i>=k;i--)
using namespace std;

typedef unsigned long long ULL;

const int maxn=1010;

int n,K;
double p[maxn],f[maxn][maxn];

ULL cal(ULL x)
{
	if (x==0) return 0;
	ULL t=1,tot=0;
	while (t*10<=x) t*=10;
	if (x>=t*2) x=t*2-1;
	tot+=x-t+1;
	t/=10;
	while (t>0){
		tot+=t;
		t/=10;
	}
	return tot;
}

void init()
{
	ULL li,ri,tmp;
	scanf("%d",&n);
	fou(i,1,n){
		cin>>li>>ri;
		tmp=cal(ri)-cal(li-1);
		p[i]=tmp/double(ri-li+1);
	}
	scanf("%d",&K);
}

void solve()
{
	int nk=(int)ceil(n*K/100.0);
	f[0][0]=1;
	fou(i,1,n){
		f[i][0]=f[i-1][0]*(1-p[i]);
		fou(j,1,i)
			f[i][j]=f[i-1][j-1]*p[i]+f[i-1][j]*(1-p[i]);
	}
	double ans=0;
	fou(i,nk,n) ans+=f[n][i];
	printf("%.12lf\n",ans);
}

int main()
{
	init();
	solve();
	return 0;
}
