#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
double f[1100][1100],ans;
int n,K;
long long l,r,a,b,c[1100],d[1100];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;++i)
	{
		scanf("%I64d%I64d",&l,&r);
		for (long long j=1,k=1;k<=19;j*=10,++k)
		{
			a=max(l,j);
			b=min(r,j*2-1);
			if (a<=b) c[i]+=b-a+1;
		}
		d[i]=r-l+1-c[i];
	}
	scanf("%d",&K);
	f[0][0]=1;
	for (int i=1;i<=n;++i)
		for (int j=0;j<=i;++j)
		{
			f[i][j]=(double)f[i-1][j]*d[i]/(d[i]+c[i]);
			if (j) f[i][j]+=(double)f[i-1][j-1]*c[i]/(d[i]+c[i]);
		}
	for (int j=0;j<=n;++j)
		if (j*100>=K*n)
			ans+=f[n][j];
	printf("%.15lf\n",ans);
	//system("pause");
}
