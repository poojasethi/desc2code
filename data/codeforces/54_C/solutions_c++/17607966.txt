#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string.h>
#include<vector>
#include<map>
using namespace std;
typedef long long ll;
long double dp[1010][1010];
double ans;
ll N, K, L[1010], R[1010];
long double calc(long double x)
{
	long double i;
	long double res = 0;
	for(i = 1; i * 10 <= x; i *= 10)
		res += i;
	if(x >= 2 * i)
		res += i;
	else if(x >= i)
		res += x - i + 1;
	return res;
} 
int main()
{
//	freopen("input.txt", "r", stdin);
	scanf("%I64d", &N);
	for(int i = 1; i <= N; i++)
		scanf("%I64d%I64d", &L[i], &R[i]);
	dp[0][0] = 1.0;
	for(int i = 1; i <= N; i++){
		
		long double P = (long double)(calc(R[i]) - calc(L[i] - 1)) / (R[i] - L[i] + 1);
		for(int j = 0; j <= i; j++)
			dp[i][j] = dp[i - 1][j] * (1.0 - P) + dp[i - 1][j - 1] * P;
	}
	scanf("%I64d", &K);
	for(int i = 0; i <= N; i++){
		if(i * 100 >= N * K)
			ans += dp[N][i];
	}
	printf("%.15f", ans);
	return 0;
}
