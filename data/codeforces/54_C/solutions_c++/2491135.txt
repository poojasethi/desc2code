#include<cstdio>
#include<cstring>
#include<cmath>
#define maxn 1010
typedef long long LL;
int n,m;
double p[maxn],f[2][maxn],ans;
bool now;
LL max(LL a,LL b){return a>b?a:b;}
LL min(LL a,LL b){return a<b?a:b;}
int main(){
    int i,k,j,t;
    LL l,r,sum,rec,xx,yy,jj;
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%I64d%I64d",&l,&r);
        sum=r-l+1;
        rec=0;
        for(jj=t=1;t<=19&&jj<=r;t++,jj*=10){
            xx=max(jj,l);
            yy=min(jj*2-1,r);
            if(xx<=yy)rec+=yy-xx+1;
        }
        p[i]=rec*1.0/sum;
    }
    scanf("%d",&k);
    m=(int)ceil(n*(k/100.0));
    f[now=false][0]=1;
    for(i=1;i<=n;i++){
        now^=true;
        for(j=0;j<=i;j++)
            f[now][j]=f[!now][j]*(1-p[i])+f[!now][j-1]*p[i];
    }
    for(i=m;i<=n;i++)
        ans+=f[now][i];
    printf("%.10lf\n",ans);
}
