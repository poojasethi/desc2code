#include<cmath>
#include<cstdio>
#include<string>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<queue>
using namespace std;
const int N=10100;
typedef long long ll;
typedef long double lf;
lf dp[N];
ll l,r,s10[20];
char s[20];
int n,k;
ll sum(ll x){
	if(x<2) return x;
	sprintf(s,"%I64d",x);
	if(s[0]=='1'){
		sscanf(s+1,"%I64d",&x);
		return x+1+s10[strlen(s)-2];
	}
	else return s10[strlen(s)-1];
}
int main()
{
	s10[0]=1;
	for(int i=1;i<20;i++)s10[i]=s10[i-1]*10+1;
	while (~scanf("%d",&n))
	{
		for(int i=1;i<=n;i++)dp[i]=0.0;
		dp[0]=1.0;
		for(int i=1;i<=n;i++){
			scanf("%I64d%I64d",&l,&r);
			ll num=sum(r) - sum(l-1);
			lf tmp=(lf)num/lf(r-l+1);
			for(int j=i;j>0;j--)
				dp[j]=dp[j]*(1.0 - tmp) + dp[j-1]*tmp;
			dp[0]*=(1.0 - tmp);
		}
		scanf("%d",&k);
		k=ceil(n/100.0 * k);
		lf ans=0.0;
		for(int i=k;i<=n;i++)ans+=dp[i];
		printf("%.10lf\n",(double)ans);
	}
	return 0;
}
