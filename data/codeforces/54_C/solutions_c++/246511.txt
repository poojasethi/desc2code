#include<cstdio>
int n;
double p[1001],f[1001][1001];
long long calc(long long x)
{
    if(!x)return 0;
    long long i;
    long long ret=0;
    for(i=1;i<=x>>1&&i>0;i*=10)
        ret+=i;
    if(x>=i)ret+=(x-i+1);
    return ret;
}
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
    {
        long long l,r;
        scanf("%I64d%I64d",&l,&r);
        p[i]=(calc(r)-calc(l-1))/double(r-l+1);
    }
    int k;
    scanf("%d",&k);
    f[0][0]=1;
    for(int i=0;i<=n;++i)
        for(int j=0;j<n;++j)
            {
                f[i+1][j+1]+=f[i][j]*p[i+1];
                f[i+1][j]+=f[i][j]*(1-p[i+1]);
            }
    double ans=0;
    for(int i=0;i<=n;++i)
        if(i/double(n)>=k/100.0)
            ans+=f[n][i];
    printf("%.12lf\n",ans);
    return 0;
}
