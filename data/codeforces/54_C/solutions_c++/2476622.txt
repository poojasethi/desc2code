#include<iostream>
#include<cstdio>

using namespace std;

const int maxn = 1010;

int n , K;
double f[maxn][maxn];

long long caln_2( long long p ) {
	long long ret = 0 , i = 1;
	while ( p / 10 >= i ) {
		ret += i;
		i *= 10;
	}
	if ( p >= 2 * i ) ret += i;
		else ret += p - i + 1;
	return ret;
}

double caln() {
	long long l , r; cin >> l >> r;
	long long a = caln_2( r ) - caln_2( l - 1 );
	long long b = r - l + 1;
	return ( double ) a / b;
}

int main() {
	scanf( "%d" , &n );
	f[0][0] = 1;
	for ( int i = 1 ; i <= n ; i++ ) {
		double p = caln();
		f[i][0] = f[i - 1][0] * ( 1 - p );
		for ( int j = 1 ; j <= i ; j++ )
			f[i][j] = f[i - 1][j - 1] * p + f[i - 1][j] * ( 1 - p );
	}
	
	scanf( "%d" , &K );
	double ans = 0;
	for ( int i = 0 ; i <= n ; i++ )
		if ( i * 100 >= n * K ) ans += f[n][i];
	printf( "%.10lf\n" , ans );
	return 0;
}
