#include<iostream>
#include<stdio.h>
using namespace std;

typedef long double ll;
ll read()
{
    ll x=0,f=1;char c=getchar();
    while(c<'0'||c>'9'){if(c=='-')f=-1;c=getchar();}
    while(c>='0'&&c<='9'){x=x*10+c-'0';c=getchar();}
    return x*f;
}
int n;
ll f[1010][1010],p[1010];

ll calc (ll x)
{
	ll i,t=0;
	for (i=1;i*10<=x;i*=10)
	  t+=i;
	if (x>=i*2)
	  t+=i;
	else if (x>=i)
	  t+=x-i+1;
	return t;
}

void doing ()
{
	ll l,r,k;
	cin>>n;
	for (int i=1;i<=n;i++)
	{
		l=read();r=read();
		p[i]=(calc(r)-calc(l-1))/(r-l+1);
	}
	//cout<<p[0]<<" !!!!!"<<endl;
	//cout<<p[1]<<" !!!!!"<<endl;
	f[0][0]=1.0;
	for (int i=1;i<=n;i++)
	  for (int j=0;j<=i;j++)
	  {
	    f[i][j]+=f[i-1][j]*(1-p[i]);
	  	f[i][j+1]+=f[i-1][j]*p[i];
	  }
	//cout<<f[0][0]<<" "<<f[0][1]<<endl;
	//cout<<f[1][0]<<" "<<f[1][1]<<endl;
	cin>>k;
	double ans=0;
	for (int i=0;i<=n;i++) 
		if (100*i>=n*k)
		  ans+=f[n][i];
	printf ("%.15lf\n",ans);
}

int main ()
{
	doing ();
	return 0;
}
