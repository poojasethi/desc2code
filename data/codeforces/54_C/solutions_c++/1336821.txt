#include <iostream>
#include <cstdio>
#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl
#define ll long long
using namespace std;

ll l,r;
double dp[1010][1010],p[1010],res;
int n,m,k;

double f(ll a,ll b){
  int ka=1,kb=1;double r;
  ll ta=a,tb=b,sa=2,sb=2,c,cc;

  while(ta>=10){ka++;ta/=10;sa*=10;}
  while(tb>=10){kb++;tb/=10;sb*=10;}

  if(ka==kb){
    if(ta>=2)r=0;
    else if(tb<2)r=1;
    else r=(double)(sa-a)/(b-a+1);
  }

  else {
    cc=b-sb/2+1+sa*5-a;
    c=(b-a+1-cc)/9+((tb==1)?(b-sb/2+1):sb/2)+((ta==1)?(sa-a):0);
    r=(double)c/(b-a+1);
  }

  return r;

}


int main(){


  cin>>n;

  rep(i,n){cin>>l>>r;p[i]=f(l,r);}

  rep(i,n)rep(j,n+1){
    if(i==0)dp[i][j]=(j==0?(1.0-p[i]):(j==1?p[i]:0));
    else dp[i][j]=dp[i-1][j]*(1-p[i])+(j==0?0:dp[i-1][j-1]*p[i]);
  }

  cin>>k;

  m=(n*k%100==0)?(n*k/100):(int)(n*k/100)+1;
  
  for(int i=m;i<=n;i++)res+=dp[n-1][i];
  
  printf("%.10f\n",res);
  return 0;
}
