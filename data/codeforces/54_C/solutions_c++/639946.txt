#include<iostream>
#include<cstdio>
using namespace std;

long long m10[20], g[20],l,r,k,i,j,n;
double p[1010], f[1010][1010], res = 0;

int chuanbi(){
    m10[0] = 1;
    g[0] = 0;
    for(int i=1; i<=19; i++) {
            m10[i] = m10[i-1]*10;
            g[i] = g[i - 1] * 10 + 1;
    }
}

long long tinh(long long x){
   int i;
   if (x == 0) return 0;
   if (x < 10) return 1;
   for(i=0;i<=19;i++) if(m10[i]>x/10)break;
   return g[i]+min(x-m10[i]+1, m10[i]);
}

int main(){
    chuanbi();
    cin>>n;
    for(i = 1; i<= n;i++){
          cin>>l>>r;
          p[i] = (double) (tinh(r) - tinh(l-1))/(r - l + 1);
    }
    cin>>k;
    f[0][0]=1;
    for(i = 1; i <= n; i++){
       f[i][0] = f[i - 1][0] * (1 - p[i]);
       for(j = 0; j <= i;j++) f[i][j] = f[i - 1][j]*(1-p[i]) + f[i - 1][j - 1]*p[i];
    }
    for (i = 0; i <= n; i++) if(i*100<k*n) res += f[n][i]; else break;
    res = 1 - res;
    printf("%.9f",res);
    //system("pause");
}
