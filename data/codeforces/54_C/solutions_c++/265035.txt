using namespace std;
# define FOR(i,a,n) for(int i=a;i<n;++i)
# define REP(i,n) FOR(i,0,n)
typedef long long LL;
int n,k;
LL one[2000],all[2000];
double dp[2000][2000];
LL calc(LL num)
{
	if(!num) return 0;
	LL tmp=num;
	int len=0;
	while(num)
	{
		num/=10;
		++len;
	}
	LL m=1,sum=0;
	REP(i,len)
	{
		sum+=m;
		if(i<len-1) m*=10;
	}
	if(tmp/m==1)
	{
		sum-=m-1-(tmp%m);
	}
	return sum;
}
double go(int pos,int t)
{
	if(pos==n) return t*100>=k*n;
	double &r=dp[pos][t];
	if(r!=-1) return r;
	double p=one[pos]*1.0/all[pos];
	r=p*go(pos+1,t+1)+(1.0-p)*go(pos+1,t);
	return r;
}
	
int main()
{
	cin>>n;
	int cnt=0;
	REP(i,n)
	{
		LL a,b;
		cin>>a>>b;
		one[cnt]=calc(b)-calc(a-1);
		all[cnt]=b-a+1;
		++cnt;
	}
	cin>>k;
	REP(i,n) REP(j,n+1) dp[i][j]=-1;
	printf("%.9f",go(0,0));
}
