N = input()
T = [1]+[0]*N
for i in range(N):
	L,R = map(int,raw_input().split())
	c = 1
	n = 0
	while c<=10**18:
		n += max(0,min(R,2*c-1)-max(L,c)+1)
		c *= 10
	p = float(n)/(R-L+1)
	
	B = T
	T = [B[0]*(1-p)]+[0]*N
	for j in range(1,N+1):
		T[j] = B[j]*(1-p)+B[j-1]*p

K = input()
print sum(T[(N*K+99)/100:])
