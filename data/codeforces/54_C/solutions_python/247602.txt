N = input()
D = [1] + [0]*N
for i in range(N):
    L,R = map(int, raw_input().split())
    c = 1
    count = 0
    while c <= R:
        c1 = 2*c - 1
        if c1 >= L:
            count += min(R, c1) - max(L, c) + 1
        c *= 10
    p = count/float(R-L+1)
       
    Dprev = D
    D = [Dprev[0] * (1 - p)]
    for j in range(1, N+1):
        D.append(Dprev[j - 1]*p + Dprev[j] * (1 - p))

K = input()

index = N*K/100
if N*K % 100 != 0:
    index += 1
print sum(D[index:])
