#!/usr/bin/python
import sys
MAXP = 20
n = int (sys.stdin.readline ())
l, r = [], []
for i in range (n):
	cl, cr = [int (x) for x in sys.stdin.readline ().split ()]
	l += [cl]
	r += [cr]
k = int (sys.stdin.readline ())
#print '%.20f' % 1.2345678901234567890123
p, q, g = [], [], []
for i in range (n):
	res = 0
	total = r[i] - l[i] + 1
	for u in range (0, MAXP):
		v = 10 ** u
		lo = v
		hi = v * 2 - 1
		lo = max (lo, l[i])
		hi = min (hi, r[i])
		if lo <= hi:
			res += hi - lo + 1
	p += [res]
	q += [total]
	g += [float (res) / total]
#for i in range (n):
#	print p[i], q[i], g[i]

f = [[0.0 for i in range (n + 1)] for j in range (n + 1)]
f[0][0] = 1.0
for i in range (n):
	for j in range (i + 1):
#		print 'f[%d][%d] = %.20lf' % (i, j, f[i][j])
		f[i + 1][j + 1] += f[i][j] * g[i]
		f[i + 1][j] += f[i][j] * (1.0 - g[i])

res = 0.0
for i in range (n + 1):
	if i * 100 >= n * k:
		res += f[n][i]
print '%.20f' % res
