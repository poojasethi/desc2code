m = input()
q = raw_input().split(' ')
for i in xrange(m):
	q[i] = int(q[i])

n = input()
a = raw_input().split(' ')
for i in xrange(n):
	a[i] = int(a[i])

best = min(q)
a = sorted(a, reverse = True)

answer = 0
for i in xrange(n):
	if ((i + 1) % (best + 2) == 0) or ((i + 2) % (best + 2) == 0):
		continue
	answer += a[i]

print answer
