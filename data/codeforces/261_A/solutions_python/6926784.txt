m = int(raw_input())
mn = min(int(x) for x in raw_input().split())
n = int(raw_input())
arr = [int(x) for x in raw_input().split()]
arr.sort(reverse=True)
ans = 0
limit = len(arr)
i = 0
while i < limit:
    if limit - i > mn:
        for j in range(mn):
            ans += arr[i]
            i += 1
        i += min(limit-i,2)
    else:
        ans += arr[i]
        i += 1

print ans
