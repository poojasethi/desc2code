range = xrange
m = int(raw_input())
qs = map(int, raw_input().split())
n = int(raw_input())
ps = map(int, raw_input().split())
ps.sort()
cartSize = min(qs)
x = n/(cartSize + 2)
s = n % (cartSize + 2)
if s < cartSize:
    ans = sum(ps[i] for i in range(n%(cartSize+2)))
else:
    ans = sum(ps[i] for i in range(s-cartSize, s))
for i in range(n%(cartSize+2), n, cartSize+2):
    ans += sum(ps[j] for j in range(i+2, min(n, i + cartSize + 2)))
print(ans)
