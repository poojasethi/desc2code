NAME = "P1"
try:
    inFile = open(NAME+".txt")
except:
    pass

def read():
    try:
        return inFile.readline().strip()
    except:
        return raw_input().strip()

m = int(read())
dis = map(int,read().split())
n = int(read())
items = map(int,read().split())
d = min(dis)
items.sort()
#items = items[:n]

ans = 0
i = n-1
while i >= 0:
    for _ in xrange(d):
        if i < 0:
            break
        ans += items[i]
        i -= 1
    i -= 2
print ans
