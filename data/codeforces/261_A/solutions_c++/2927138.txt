#include<iostream>
#include<algorithm>
using namespace std;
int q[100001],s[100001];
int main(){
	int m,n;
	cin >>m;
	for(int i=0;i<m;i++)
		cin >>q[i];
	cin >>n;
	for(int i=0;i<n;i++)
		cin >>s[i];
	sort(q,q+m);
	sort(s,s+n);
	int sum=0;
	n--;
	while(n>=0){
		for(int i=0;i<q[0];i++){
			sum+=s[n];
			n--;
			if(n<0)
				break;
		}
		n-=2;
	}
	cout <<sum<<endl;
}
