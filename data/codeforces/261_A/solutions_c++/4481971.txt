#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int x,i,a[120000],v,j,ans,m,n;
int main(){
	scanf("%d",&m);
	v=1e8;
	for(i=1;i<=m;++i)scanf("%d",&x),v=min(v,x);
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d",a+i);
	sort(a+1,a+n+1);
	j=n+1;
	for(i=n;i>=1;--i){
		ans+=a[i];
		if(j-i==v){
			i-=2;
			j=i;
		}
	}
	printf("%d\n",ans);
}
