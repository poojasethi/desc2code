#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int m,q[100001],n,a[100001],s=0;
    cin>>m;
    for(int i=0;i<m;i++)
        cin>>q[i];
    sort(q,q+m);
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(int i=n-1;i>=0;i-=q[0]+2)
        for(int j=i;j>i-q[0]&&j>=0;j--)
            s+=a[j];
    cout<<s;
}
