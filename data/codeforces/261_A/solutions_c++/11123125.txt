#include<bits/stdc++.h>
using namespace std;
int main()
{
	int i,n,m,ans=0,j,k;
	cin>>m;
	int d[m];
	for(i=0;i<m;i++)
		scanf("%d",&d[i]);
	sort(d,d+m);
	cin>>n;
	int a[n];
	for(i=0;i<n;i++)
		scanf("%d",&a[i]);
	sort(a,a+n,greater<int>());
	j=0;k=0;
	for(i=0;i<n;)
	{
		if(k!=d[j])
		{
			ans+=a[i];
			k++;
			i++;
		}	
		else
		{
			i+=2;
			k=0;
		}
		//cout<<i<<endl;	
	}	
	cout<<ans<<endl;	
	return 0;
}
