#include<stdio.h>
#include<algorithm>
#define N 100005
using namespace std;

int q[N], a[N];
int main(){
	int i,m,n,dap=0,j;
	scanf("%d",&m);
	for(i=0;i<m;i++)
	{
		scanf("%d",&q[i]);
	}
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	sort(a,a+n);
	sort(q,q+m);

	for(i=n-1;i>=0;i--)
	{
		for(j=0;j<q[0] && i>=0;j++)
		{
			dap += a[i];
			i--;
		}
		i--;
	}
	printf("%d",dap);
}
