#include<cstdio>
#include<algorithm>
using namespace std;
int q[100005],a[100005];
int main(){
    int m,n,count,ans=0;scanf("%d",&m);
    for(int i=0;i<m;i++)scanf("%d",&q[i]);
    sort(q,q+m);scanf("%d",&n);
    for(int i=0;i<n;i++)scanf("%d",&a[i]);
    sort(a,a+n);reverse(a,a+n);
    for(int i=0;i<n;i++){
        if(i%(q[0]+2)<q[0])ans+=a[i];
    }
    printf("%d",ans);
    return 0;
}
