#include<bits/stdc++.h>
using namespace std;
int main()
{
	int m;
	int dis=10e6;
	scanf("%d",&m);
	int i,data;
	for(i=0;i<m;i++)
		scanf("%d",&data),dis= min(dis,data);
	int n;
	scanf("%d",&n);
	int cost[n];
	for(i=0;i<n;i++)
		scanf("%d",&cost[i]);
	sort(cost,cost+n);
	int totalcost = 0,items=0;
	for(i=n-1;i>=0;i--)
	{	
		totalcost += cost[i];
		items++;
		if(items==dis)
			i-=2,items=0;
	}
	cout << totalcost << endl;
	return 0;
}
