#include <cstdio>
#include <algorithm>
using namespace std;
const int N=(int)1e5+5;
int n,m,x=N,ans,q[N],a[N];
int main(){
	scanf("%d",&m);
	for(int i=1;i<=m;i++) scanf("%d",q+i),x=min(x,q[i]);
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",a+i);
	sort(a+1,a+n+1);
	for(int i=n;i>=1;i--)
		if((n-i)%(x+2)<x) ans+=a[i];
	printf("%d\n",ans);
	scanf("\n");
}
