#include <stdio.h>
#include <algorithm>
#define MAX 100005
using namespace std;
int n,m,x=MAX,ans,q[MAX],a[MAX];
int main()
{
	scanf("%d",&m);
	for(int i=1;i<=m;i++)
	{
		scanf("%d",q+i);
		x=min(x,q[i]);
	}
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",a+i);
	sort(a+1,a+n+1);
	for(int i=n;i>=1;i--)
		if((n-i)%(x+2)<x) ans+=a[i];
	printf("%d\n",ans);
	scanf("\n");
}
