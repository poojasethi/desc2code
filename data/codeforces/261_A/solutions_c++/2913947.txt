#include<iostream>
#include<algorithm>
using namespace std;
int main(){
	int m,c=1e9,v;
	static int num[100005];
	cin >> m;
	for(int i = 0; i < m; i++){
		cin >> v;
		c = min(c,v);
	}
	int n; cin >> n;
	for(int i = 0; i < n; i++)
		cin >> num[i];
	sort(num,num+n);
	int ans = 0;
	for(int i = n-1; i>=0;){
		for(int j = 0; j< c && i>=0;j++,i--)
			ans += num[i];
		i-=2;
	}
	cout<< ans << endl;
}
