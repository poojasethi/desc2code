#include<bits/stdc++.h>
using namespace std;
int main()
{
    int m,mn=1e6,q;
    cin>>m;
    while(m--)
    {
        cin>>q;
        mn=min(mn,q);
    }
    int n,a[100015];
    q=mn;
    cin>>n;
    for(int i=0;i<n;++i)
    cin>>a[i];
    sort(a,a+n);
    int sum=0;
    for(int i=n-1;i>=0;)
    {
        for(int j=0;j<q&&i>=0;++j,--i)sum+=a[i];
        i-=2;
    }
    cout<<sum;

}
