#include<cstdio>
#include<algorithm>

using namespace std;

int m,n;
int q[100005],a[100005];

int main() {
	scanf("%d",&m);
	for(int i=0;i<m;++i) scanf("%d",q+i);
	sort(q,q+m);
	scanf("%d",&n);
	for(int i=0;i<n;++i) scanf("%d",a+i);
	sort(a,a+n);

	int ans = 0;
	for(int i=n-1;i>=0;) {
		for(int j=0;j<q[0];++j) {
			if(i-j<0) break;
			ans += a[i-j];
		} i-=q[0]+2;
	} printf("%d\n",ans);
}
