#include <iostream>
#include<bits/stdc++.h>
using namespace std;

int main() 
{
	int m,n,i,sum=0,x;
	cin>>m;
	int a[m];
	for(i=0;i<m;i++)
	cin>>a[i];
	sort(a,a+m);
	x=a[0];
	cin>>n;
	int b[n];
	for(i=0;i<n;i++)
	cin>>b[i];
	sort(b,b+n);
	reverse(b,b+n);
	for(i=0;i<n;i++)
	{
		if(i%(x+2)<x)
		sum+=b[i];
	}
	cout<<sum;
	return 0;
}
