#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int m,n,min=2000000,sum=0,count=0,i;
	cin>>m;
	int q[m];
	for(i=0;i<m;i++){cin>>q[i];if(q[i]<min)min=q[i];}
	cin>>n;int a[n];
	for(i=0;i<n;i++)cin>>a[i];
	sort(a,a+n);i=n-1;
	while(i>=0)
	{
		for(int j=1;j<=min;j++)
		{
		   sum+=a[i];
		   i--;
		   if(i<0)break;
	    }
		i-=2;
	}
	cout<<sum;
}
