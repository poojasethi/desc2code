#include <bits/stdc++.h>

using namespace std;

int p[100000+10], s;
int main() 
{
    int m,q=1000000, a; cin>>m;
    for(int i=0; i<m; i++) {
        cin >>a;
        q = min(q, a);
    }
    int n; cin>>n; 
    for(int i=0; i<n; i++) {
        cin>>p[i];
    }

    sort(p, p+n);
    
    for(int i=n-1; i>=0; i-=2) {
        for(int j=0; j<q&&i>=0; j++, i--){
            s += p[i];
        }
    }
    cout <<s;
	return 0;
}
