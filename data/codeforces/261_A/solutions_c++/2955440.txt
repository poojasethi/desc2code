#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
int m;
cin>>m;
int q[m],i,j,min=100000,sum=0;
for(i=0;i<m;i++)
{cin>>q[i];
if(min>q[i])
min=q[i];
}
int n;
cin>>n;
int arr[n];
for(i=0;i<n;i++)
cin>>arr[i];
sort(arr,arr+n);
i=n-1;
while(true)
{
for(j=0;j<min;j++)
{if(i-j<0)
break;
sum+=arr[i-j];
}
if(i-j<0)
break;
i=i-min-2;
}
cout<<sum;
return 0;
}
