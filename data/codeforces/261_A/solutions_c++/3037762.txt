#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int d;
	cin>>d;
	int dtype[d],sum=0;
	for(int i=0;i<d;i++)
		cin>>dtype[i];
		int n;
		cin>>n;
		int price[n];
	for(int i=0;i<n;i++)
		cin>>price[i];
	sort(dtype,dtype+d);
	sort(price,price+n,greater<int>());
	
		
		int i=0;
	for(int i=0;i<n;i++)
	{	
			if(i%(dtype[0]+2)<dtype[0])
			{
			sum+=price[i];
			}
	}
	cout<<sum<<endl;	
	return 0;
}
