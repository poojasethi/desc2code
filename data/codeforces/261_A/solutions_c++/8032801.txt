#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int main()
{
	int m;
	cin>>m;
	int d[m];
	for(int i=0;i<m;i++)
		cin>>d[i];
	int n;
	cin>>n;
	ll p[n];
	for(int i=0;i<n;i++)
		cin>>p[i];
	sort(p,p+n);
	sort(d,d+m);
	int inc = d[0]+2;
	ll answer = 0;
	for(int i=n-1;i>=0;i-=inc)
	{
	//	cout<<i<<endl;
		for(int j=i;j>=max(0,i-d[0]+1);j--)
			answer+=p[j];
	}
	cout<<answer<<endl;
	return 0;
}
