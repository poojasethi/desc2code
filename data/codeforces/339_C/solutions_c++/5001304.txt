#include <cstdio>
char s[11];
int ans[1111],m;
bool dfs(int now,int del,int pre)
{
    if(now>m)
    {
        puts("YES");
        for(int i=1;i<=m;++i)printf("%d ",ans[i]);
        puts("");
        return true;
    }
    for(int i=del+1;i<=10;++i)
    {
        if(s[i]=='0' || i==pre)continue;
        ans[now]=i;
        if(dfs(now+1,i-del,i))return true;
    }
    return false;
}
int main()
{
    scanf("%s",s+1);
    scanf("%d",&m);    
    if(!dfs(1,0,0))puts("NO");
    return 0;
}
