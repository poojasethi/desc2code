#include <stdio.h>

char s[20];
int m, a[1010];

bool dfs(int x,int y)
{
    if (x>m)
    {
        printf("YES\n");
        for (int i=1;i<=m;++i)
            printf("%d ",a[i]);
        return 1;
    }

    for(int i=y+1;i<=10;++i)
        if (i!=a[x-1] && s[i-1]=='1')
        {
            a[x]=i;
            if (dfs(x+1,i-y)==1)
                return 1;
        }
    return 0;
}

int main()
{
    scanf("%s",s);
    scanf("%d",&m);
    if(dfs(1,0)==0)
        printf("NO");
    return 0;
}
