#include<bits/stdc++.h>
using namespace std;
string s;
int m;
int  a[1001];
bool dfs(int cnt,int w) {
	if(cnt>m) {
		cout << "YES" << endl;
		cout << a[1];
		for(int i=2;i<=m;i++)
			cout << " " << a[i];
		cout << endl;
		return 1;
	}
	for(int i=w+1;i<=10;i++) 
		if(s[i-1]=='1' && i!=a[cnt-1])
		{
			a[cnt]=i;
			if(dfs(cnt+1,i-w))
				return 1;
		}
			return 0;
	
	return 0;
}
	

int main() {
	cin >> s;
	cin >> m;
	a[0]=0;
	if(!dfs(1,0))
		cout << "NO" << endl;
}
