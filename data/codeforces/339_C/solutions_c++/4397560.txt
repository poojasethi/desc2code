#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
long long n;
char s[11];
long long a[4000];
bool dfs(long long now,long long b,long long c)
{
	if(now==n)return true;
	for(int i=1;i<11;i++)if(s[i]=='1'&&i!=b&&i>c)
	{
		a[now]=i;
		if(dfs(now+1,i,i-c))return true;
		a[now]=0;
	}
	return false;
}
int main()
{
	for(long long i=1;i<11;i++)cin>>s[i];
	cin>>n;
	if(dfs(0,0,0))
	{
		cout<<"YES\n";
		for(long long i=0;i<n;i++)cout<<a[i]<<" ";
		cout<<endl;
	}
	else cout<<"NO\n";
	return 0;
}
