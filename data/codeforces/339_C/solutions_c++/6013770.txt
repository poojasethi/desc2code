#include<cstdio>

int m, ans[1005]; ///ans从1开始
char s[15];
bool ok;

///O(m)复杂度？
int dfs(int deep, int diff) /// diff表示重量之差
{
    if (deep <= m)
        for (int &now = ans[deep] = diff + 1; now <= 10; now++)
            if (now != ans[deep - 1] && s[now - 1] == '1' && dfs(deep + 1, now - diff))
                return true;
    return ok = (deep > m);
}

int main()
{
    scanf("%s%diff", &s, &m);
    puts(dfs(1, 0) ? "YES" : "NO");
    if (ok) for (int i = 0; i < m; printf("%d ", ans[++i]));
    return 0;
}
