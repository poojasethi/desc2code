#include<bits/stdc++.h>
using namespace std;
string s;int m;int ans[1006];
bool dfs(int r,int lvl)
{
    if(lvl==m+1)
    {
        cout<<"YES"<<endl;
        for(int i=1;i<=m;i++)
        cout<<ans[i]<<" ";
        return 1;
        
    }
    for(int i=1;i<=10;i++)
    {
        if(s[i-1]-'0'&&i>r && i!=ans[lvl-1])
        {
            ans[lvl]=i;
            if(dfs(i-r,lvl+1))return 1;
        }
    }
    return 0;
}
int main()
{

    cin>>s>>m;


    if(!dfs(0,1))cout<<"NO";
}
