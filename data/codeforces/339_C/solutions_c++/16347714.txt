#include<bits/stdc++.h>
using namespace std;
int m;
bool h[11];
vector < int > v;
int dfs(int last,int b,int cnt)
{
	if(cnt==m) return 1;
	for(int i=1; i<=10; i++)
		if(h[i] and i != last and i>abs(b))
		{
			v.push_back(i);
			if(dfs(i,b>0?b-i:b+i,cnt+1)) return 1;
			v.pop_back();
		}
	return 0;
}
int main ()
{
	for(int i=1; i<=10; i++)
		h[i]=getchar()-'0';
	cin>>m;
	if(dfs(0,0,0))
	{
		cout<<"YES"<<endl;
		for(int i=0; i<v.size(); i++)
			cout<<v[i]<<" ";
	}
	else
		cout<<"NO";
}
