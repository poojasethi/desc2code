#include <stdio.h>
char cade[16];
int M;
int arr[1024];

int dfs(int n, int dif){
    if(n>M) return 1;
    
    for(int d=dif+1; d<=10; d++)
        if(arr[n-1]!=d && cade[d-1]=='1'){
            arr[n]=d;
            if(dfs(n+1,d-dif))
                return 1;
        }
    return 0;
}

int main(){
    scanf("%s",cade);
    scanf("%d",&M);
    
    if(dfs(1,0)){ 
        printf("YES\n");
        for(int i=1; i<=M; i++)
            printf("%d ",arr[i]);
        printf("\n");
    }else{
        printf("NO\n");
    }
    
    return 0;
}
