#include<cstdio>
#include<cstdlib>
using namespace std;
int m,p[1555],a[15];
char ch;
void dfs(int last,int step,int weight){
    if(step==m){
        puts("YES");
        for(int i=1;i<m;++i)printf("%d ",p[i]);printf("%d\n",p[m]);
        exit(0);
    }
    for(int i=weight+1;i<=10;++i){
        if(last==i||(!a[i]))continue;
        p[step+1]=i;
        dfs(i,step+1,i-weight);
    }
}
int main(){
    for(int i=1;i<=10;++i)ch=getchar(),a[i]=ch-48;
    scanf("%d",&m);
    dfs(0,0,0);
    puts("NO");
}
