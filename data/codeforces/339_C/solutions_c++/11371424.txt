#include<stdio.h>
int ans[1005] , i , m;
char s[15];
int f( int d , int g)
{
  if( d <= m )
  {
    for( int i = g + 1 ; i <= 10 ; i++ )
    {
      ans[d] = i;
      if( i != ans[d-1] && s[i-1] == '1' && f(d+1 , i-g) )
      {
        return 1;
      }
    }
  }
  if( d > m )
    return 1;
  return 0;
}
int main()
{
  scanf("%s %d" , s , &m);
  if( f(1,0) )
  {
    puts("YES");
    for(i=1;i<=m;i++)
      printf("%d " , ans[i] );
  }
  else puts("NO");
  return 0;
}
