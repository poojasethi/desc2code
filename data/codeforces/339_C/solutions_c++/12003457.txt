#include<bits/stdc++.h>
using namespace std;
char ch[12];
int m,a[1003];
bool dfs(int x,int s)
{
    if (x>m)
    {
        printf("YES\n");
        for (int i=1;i<=m;i++)
            printf("%d ",a[i]);
        return 1;
    }
    for (int i=s+1;i<=10;i++)
        if (i!=a[x-1]&&ch[i-1]!='0')
    {
        a[x]=i;
        if (dfs(x+1,i-s))
            return 1;
    }
    return 0;
}
main()
{
    scanf("%s", ch);
    scanf("%d", &m);
    a[0]=0;
    if (!dfs(1,0)) printf("NO");
}
