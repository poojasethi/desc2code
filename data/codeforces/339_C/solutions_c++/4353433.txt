#include <iostream>
#include <string>
using namespace std;
int i, m, q[1024];
string s;

bool rec(int d, int l, int n) {
	for(int &w=q[n]=d+1; n<m && w<11; w++)
		if (w!=l && s[w-1]=='1' && rec(w-d,w,n+1)) return true;
	return (n==m);
}

int main() {
	cin >> s >> m;
	if (!rec(0,0,0)) cout<<"NO";
  else for (cout<<"YES\n";i<m;cout<<q[i++]<<" ");
  cout<<endl;
}
