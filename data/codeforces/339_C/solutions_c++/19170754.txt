#include <bits/stdc++.h>

using namespace std;

char s[15];
int m;
bool f=false;
vector<int> ans(1005);

void solve (int prev, int w, int tot) {
	if (f) return;
	if (tot==m){
		f=true;
		cout << "YES\n";
		for(int i=1 ; ans[i] ; i++)
			cout << ans[i] << " ";
		return ;
	}
	for (int i=w+1 ; i <= 10 ; i++) {
		if (s[i] == '1' && prev != i) {
			ans[tot+1]=i;
			//cout<<"SADA"<<i<<i-w<<tot<<endl;
			solve(i,i-w,tot+1);
		}
	}
}
int main() {
	scanf("%s",s+1);
	cin >> m;
	solve(0,0,0);
	if(!f) cout << "NO";
	return 0;
}
