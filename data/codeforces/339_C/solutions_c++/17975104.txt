#include<iostream>
#include<vector>
using namespace std;
bool f[11]={0};
int m;
bool dfs(int d,int l,int n,vector<int>path){
	if(n==m){
		cout<<"YES\n";
		for(int i=0;i<path.size();i++){
			cout<<path[i]<<' ';
		}
		return true;
	}
	for(int i=d+1;i<=10;i++)
		if(f[i]&&i!=l){
			path.push_back(i);
			if(dfs(i-d,i,n+1,path))return true;
			path.pop_back();
		}
	return false;
}
int main()
{
	char tmp;
	for(int i=1;i<=10;i++){
		cin>>tmp;
		if(tmp=='1')f[i]=true;
	}
	cin>>m;
	vector<int>list;
	if(!dfs(0,0,0,list))cout<<"NO";
	return 0;
}
