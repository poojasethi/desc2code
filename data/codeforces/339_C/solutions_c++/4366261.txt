#include<cstdio>
#include<algorithm>

char tmp[20];

int m, tab[1005];

bool dfs(int b, int c)
{
	if(b > m)
	{
		puts("YES");
		printf("%d", tab[1]);
		for(int i = 2; i <= m; i++)
			printf(" %d", tab[i]);
		printf("\n");
		return true;
	}
	for(int i = c+1; i <= 10; i++)
		if(i != tab[b-1] && tmp[i-1]-'0')
		{
			tab[b] = i;
			if(dfs(b+1, i-c))
				return 1;
		}
		return 0;
}

int main()
{
	scanf("%s", tmp);
	scanf("%d", &m);
	
	if(!dfs(1, 0))
		puts("NO");
	
	
	
	return 0;
}
