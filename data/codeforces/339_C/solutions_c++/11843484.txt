#include <iostream>
#include <vector>

using namespace std;

string x;
int m,q[1024];

int rec(int n ,int d){
    if(n <= m){
        for(int w = d + 1 ; w <= 10 ; w++){
            q[n] = w;
            if(w != q[n - 1] && x[w - 1] > '0' && rec(n + 1,w - d)) return 1;
        }
    }
    return n > m;   
}
int main(){
    cin >> x >> m;
    bool r = rec(1,0);
    cout << (r ?"YES":"NO") << endl;
    if(r)
    for(int i = 1 ; i <= m ; i++){
        cout << q[i] << " ";
    }
    cout << endl;
    return 0;
}
