#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

int m,ans[1010];

char s[12];

bool dfs(int n,int d)
{
	if (n>m)
	{
		printf("YES\n");
		for (int a=1;a<=m;a++)
			printf("%d ",ans[a]);
		printf("\n");
		return true;
	}
	for (ans[n]=1;ans[n]<=10;ans[n]++)
		if (s[ans[n]]=='1' && d+ans[n]>0 && ans[n]!=ans[n-1] && dfs(n+1,-d-ans[n])) return true;
	return false;	
}

int main()
{
	scanf("%s",s+1);
	scanf("%d",&m);
	if (!dfs(1,0)) printf("NO\n");

	return 0;
}
