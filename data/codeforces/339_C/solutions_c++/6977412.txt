#include <cstdio>

using namespace std;
char sir[14];
int m,sol[1014];
int dfs(int x,int s){
    if(x>m){
        printf("YES\n");
        printf("%d",sol[1]);
        for(int i=2;i<=m;++i)printf(" %d",sol[i]);
        return 1;
    }
    for(int i=s+1;i<=10;++i)
        if(i!=sol[x-1] and sir[i]-'0'==1){
            sol[x]=i;
            if(dfs(x+1,i-s))return 1;
        }
    return 0;
}
int main( )
{
    //freopen("c.in","r",stdin);
    gets(sir+1);
    scanf("%d",&m);
    sol[0]=0;
    if(!dfs(1,0))printf("NO");
}
