#include<bits/stdc++.h>

using namespace std;

int A[1003],t[2],m;

string s;

void f(int j){

    int i;

    if(j == m+1) { for(cout << "YES\n",i=1; i<=m ;i++) cout << A[i] << " "; exit(0); }

    for(i=1;i<=10;i++)
        if(s[i-1] == '1' && i != A[j-1] && t[j&1] + i > t[!(j&1)]){
           t[j&1] += A[j] = i;
           f(j+1);
           t[j&1] -= i;
    }
}
int main(){

    cin >> s >> m;

    f(1);

    cout << "NO";
    return 0;
}
