#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char ch[100];
int n,ans[1009];
int dfs(int k,int y)
{
    if(k>n)return 1;
    for(int i=y+1;i<=10;i++)if(ch[i]=='1' && i!=ans[k-1])
    {
        ans[k]=i;
        if(dfs(k+1,i-y))return 1;
    }
    return 0;
}
int main()
{
    scanf("%s%d",ch+1,&n);
    if(dfs(1,0))
    {
        puts("YES");
        for(int i=1;i<=n;i++)printf("%d ",ans[i]);
    }
    else puts("NO");
    return 0;
}
