#include<cstdio>
#include<cstring>
using namespace std;

int ans[1000],m;
bool h[11],flag=true;

void dfs(int n,int ch,int prev)
{
if(n>m)
{
flag=false;
printf("YES\n%d",ans[0]);
for(int i=1;i<n;++i)
printf(" %d",ans[i]);
printf("\n");
return;
}

for(int i=ch+1; flag&&i<=10; ++i)if(h[i]&&i!=prev)
{
    ans[n]=i;
    dfs(n+1,i-ch,i);
}
}

int main()
{
char c;
for(int i=1;i<11;++i)
{
scanf("%c",&c);
if(c=='1')h[i]=true;
}
