#include <bits/stdc++.h>
using namespace std;

vector<int > ans;	
char s[15];
int m;

bool dfs(int now, int prev, int bal)
{
	if (now > m)
	{
		puts("YES");
		for (int i = 1; i <= m; i++)
			cout << ans[i] << " ";
		puts("");
		return true;
	}
	for (int i = bal + 1; i <= 10; i++)
	{
		if (s[i] == '0' or i == prev)
			continue;
		ans[now] = i;
		if (dfs(now + 1, i, i - bal))
			return true;
	}
	return false;
}

int main()
{
	scanf("%s", s + 1);
	cin >> m;
	ans.resize(1234);
	if (!dfs(1, 0, 0))
		puts("NO");
	return 0;
}
