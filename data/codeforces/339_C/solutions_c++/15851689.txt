#include<bits/stdc++.h>
using namespace std;
long long m,a[1024];
char s[12];
bool dfs(int n,int diff)
{
    if(n>m)return 1;
    for(int d = diff+1;d<=10;d++){
        if(a[n-1]!=d&&s[d-1]=='1'){
            a[n]=d;
            if(dfs(n+1,d-diff))
                return 1;
        }
    }
    return 0;
}
int main()
{
    scanf("%s %d",s,&m);
    if(dfs(1,0)){
        printf("YES\n");
        for(int i=1;i<=m;i++){
            printf("%d ",a[i]);
        }
    }
    else{
        printf("NO\n");
    }
}
