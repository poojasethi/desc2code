#include<iostream>
using namespace std;
string str;
int ans[1010],m;
bool ftw(int b,int sm,int p){
	if((sm<=0||sm>10)&&b>0)
		return 0;
	if(b==m){
		cout<<"YES\n";
		for(int i=0;i<b;i++){
			cout<<ans[i]<<" ";
		}
		return 1;
	}
	//cout<<b<<endl;
	for(int i=0;i<10;i++){
		if(str[i]=='1'&&p!=i){
			ans[b]=i+1;
			if(ftw(b+1,i+1-sm,i))
				return 1;
		}
	}
	return 0;
}
int main(){
	
	cin>>str;
	
	cin>>m;
	if(!ftw(0,0,10))
		cout<<"NO";
	return 0;
}
