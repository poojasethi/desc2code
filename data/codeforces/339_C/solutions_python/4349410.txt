

import sys
sys.setrecursionlimit(1500)


    
def step(weight,stack,left,right,needed):
  if needed == len(stack):
    return stack
  if not stack:
    for i in weight:
      stack.append(i)
      result = step(weight,stack,i,0,needed)
      if result:
        return result
      stack.pop()
  else:
    if len(stack) % 2:
      for i in weight:
        if i != stack[-1] and right+i>left:
          stack.append(i)
          result = step(weight,stack,left,right+i,needed)
          if result:
            return result
          stack.pop()
    else:
      for i in weight:
        if i != stack[-1] and left+i>right:
          stack.append(i)
          result = step(weight,stack,left+i,right,needed)
          if result:
            return result
          stack.pop()
        
      

def main():
  str1 = raw_input()
  m = input()
  weight = [i+1 for i in range(len(str1)) if str1[i]=='1']
  result = step(weight,[],0,0,m)
  if result:
    print 'YES'
    print ' '.join([str(i) for i in result])
  else:
    print 'NO'




main()
