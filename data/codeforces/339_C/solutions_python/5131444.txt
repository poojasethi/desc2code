def xenia_and_weights ():

    weights = []
    for weight, char in enumerate(raw_input()):
        if char == '1':
            weights.append(weight + 1)

    nr_weights = int(raw_input())

                    continue
                for weight in weights:
                    if ( weight != prev ) and ( weight > diff ):
                        memo[num + 1][weight - diff][weight] = prev
                        if ((num + 1) == nr_weights):
                            ok = True
                            diff = weight - diff
                            break
                if ok == True:
                    break
            if ok == True:
                break
        if ok == True:
            break
