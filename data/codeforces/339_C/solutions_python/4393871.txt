import sys
sys.setrecursionlimit(10000)


w = [bool(int(x)) for x in raw_input()]
weight = []
a=0
for x in w:
	a+=1
	if x:
		weight.append(a)

m = int(raw_input())

chain = []
def tryit(left, right, last, m):
	if(left <= right):
		return False
	if(m == 0):
		return True
 
	for i in weight:
		if i!= last and tryit(right + i, left, i, m - 1):
			chain.append(i)
			return True

	return False


for i in weight:
	if tryit(i, 0, i, m-1):
		chain.append(i)
