import copy

def readInp():

    weight_idx = raw_input()
    m = int(raw_input())
    
    return weight_idx, m


def getWeightList(weight_idx):
    weight_list = []
    for idx, w_idx in enumerate(weight_idx, start=1):
        if int(w_idx) == 1:
            weight_list.append(idx)
    return weight_list


def checkWeight(scale_1, scale_2, weight_list_tmp):
    weight_candidates = []

    for weight in weight_list_tmp:
        if weight + scale_2 > scale_1:
            weight_candidates.append(weight)

    return weight_candidates
