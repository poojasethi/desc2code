#! /usr/bin/env python
import sys
sys.setrecursionlimit(1500)
def solve(cnt, diff, last):
	if cnt==m:
		print 'YES'
		print ' '.join(map(str,ans[1:]))
		sys.exit(0)
	for i in range(1,11):
		if opt[i-1]=='1' and i>diff and i!=last:
			if (visited[cnt+1][abs(i-diff)][i]==0):
				visited[cnt+1][abs(i-diff)][i]=1
				ans[cnt+1] = i
				solve(cnt+1, abs(i-diff), i)

opt = raw_input()
m = input()
ans = [0] * (m+1)
visited = [[[0 for k in range(11)] for j in range(21)] for i in range(1011)]
visited[0][0][0] = 1
solve(0,0,0)
print 'NO'
