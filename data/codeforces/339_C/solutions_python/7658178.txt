import sys

sys.setrecursionlimit(10000)

weights = sys.stdin.readline()
m = int(sys.stdin.readline())

backtrack = [0] * 1001

def place(now, bef, last, placed):
    if placed == m:
        return True

    solved = False
    for w, have in enumerate(weights):
        weight = w + 1
        have = (have == '1')

        if have and weight + now > bef and weight != last:
            backtrack[placed] = weight
            solved = solved or place(bef, weight + now, weight, placed + 1)

        if solved:
            return solved

    return solved

result = place(0, 0, 0, 0)
print 'YES' if result else 'NO'
if result:
    print ' '.join(map(str, backtrack[:m]))
