#include<cstdio>
#include<iostream>
#include<algorithm>
#define maxn 100000
using namespace std;
int num[2*maxn+10],tot;
int find(int x){
	int l=1,r=tot;
	for(;;){
		int m=l+r>>1;
		if(num[m]==x)return m;
		if(num[m]<x)l=m+1;
		else r=m-1;
	}
}
typedef long long LL;
LL ans[maxn+10];
int pr[2*maxn+10],cnt[2*maxn+10];
struct event{
	int r,c,f;
	bool operator<(const event &a)const{return r<a.r;}	
}q[2*maxn+10];
int main(){
	#ifndef ONLINE_JUDGE
	freopen("D.in","r",stdin);
	#endif
	int n,k;
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++){
		int r,c;
		scanf("%d%d",&r,&c);
		num[2*i-1]=c;
		num[2*i]=c+k;
		q[2*i-1]=(event){r,c,1};
		q[2*i]=(event){r+k,c,-1};
	}
	sort(num+1,num+2*n+1);
	tot=1;
	for(int i=2;i<=2*n;i++)if(num[i]!=num[i-1])num[++tot]=num[i];
	sort(q+1,q+2*n+1);
	int cur=1;
	while(cur<=2*n){
		LL cr=q[cur].r;
		while(cur<=2*n&&q[cur].r==cr){
			int l=find(q[cur].c),r=find(q[cur].c+k);
			for(int i=l;i<r;i++)ans[cnt[i]]+=(cr-pr[i])*(num[i+1]-num[i]),cnt[i]+=q[cur].f,pr[i]=cr;
			cur++;
		}
	}
	for(int i=1;i<n;i++)printf("%I64d ",ans[i]);
	printf("%I64d\n",ans[n]);
	return 0;
}
