#define F1(x,y,z) for(int x=y;x<z;x++)
#define F2(x,y,z) for(int x=y;x<=z;x++)
#define F3(x,y,z) for(int x=y;x>z;x--)
#define F4(x,y,z) for(int x=y;x>=z;x--)
#define mp make_pair
#define pb push_back
#define LL long long
#define co complex<double>

#define MAX 100005
#define AMAX 1500
#define MOD 1000000007

#define f(c,d) ((1<<(c))*(d))

using namespace std;
using namespace __gnu_cxx;

int n,k,ta,ia[MAX],ib[MAX];
LL ans[MAX];
pair<int,int> x[MAX];
vector<pair<int,int> > ya;
vector<pair<int,pair<int,int> > > yb;
vector<pair<int,int> > xx;

int main(){
	scanf("%d%d",&n,&k);
	F1(a,0,n){
		scanf("%d%d",&x[a].first,&x[a].second);
		xx.pb(mp(x[a].first,a*2));
		xx.pb(mp(x[a].first+k,a*2+1));
		ya.pb(mp(x[a].second,a*2));
		ya.pb(mp(x[a].second+k,a*2+1));
	}
	sort(xx.begin(),xx.end());
	sort(ya.begin(),ya.end());
	yb.pb(mp(INT_MIN,mp(0,0)));
	F1(a,0,ya.size()){
		if(ya[a].first!=yb.back().first)yb.pb(mp(ya[a].first,mp(0,0)));
		if(ya[a].second%2==0)ia[ya[a].second/2]=yb.size()-1;
		else ib[ya[a].second/2]=yb.size()-1;
	}
	yb.pb(mp(INT_MAX,mp(0,0)));
	F1(a,0,xx.size()){
		ta=xx[a].second/2;
		//printf(" %d %d   %d\n",ta,xx[a].second%2,xx[a].first);
		if(xx[a].second%2==0){
			F1(b,ia[ta],ib[ta]){
				ans[yb[b].second.first]+=(xx[a].first-yb[b].second.second)*1LL*(yb[b+1].first-yb[b].first);
				yb[b].second.first++;
				yb[b].second.second=xx[a].first;
			}
		}else{
			F1(b,ia[ta],ib[ta]){
				ans[yb[b].second.first]+=(xx[a].first-yb[b].second.second)*1LL*(yb[b+1].first-yb[b].first);
				yb[b].second.first--;
				yb[b].second.second=xx[a].first;
			}
		}
	}
	F2(a,1,n)printf("%I64d%c",ans[a],a==n?'\n':' ');
	#ifdef LOCAL_PROJECT
	system("pause");
	#endif
	return 0;
}
