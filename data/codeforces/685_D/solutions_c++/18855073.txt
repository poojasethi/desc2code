#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <iostream>
#include <vector>
#include <utility>
#include <queue>
#include <algorithm>
#include <string>

#include <set>

#define sqr(x) (x)*(x)
#define eps 1e-10
#define Max(a,b) ((a)>(b)?(a):(b))
#define Min(a,b) ((a)<(b)?(a):(b))
#define mp(a,b) make_pair(a,b)
#define fi first
#define se second
using namespace std;
typedef long long LL;

const int N = 1e5+10;
int n,k,x[N],y[N],dx[2*N],dy[2*N],tx,ty,g[2*N],h[2*N];
map<int,int> _x,_y;
vector<int> in[2*N],out[2*N];
LL f[N];

int main(){
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;i++) scanf("%d%d",x+i,y+i);
	for(int i=0;i<n;i++) {
		dx[2*i]=x[i]+1;
		dx[2*i+1]=x[i]-k+1;
		dy[2*i]=y[i]+1;
		dy[2*i+1]=y[i]-k+1;
	}
	sort(dx,dx+2*n);
	sort(dy,dy+2*n);
	for(int i=0;i<2*n;i++){
		if(i==0 || dx[i]!=dx[i-1]){
			dx[tx++]=dx[i];
		}
		if(i==0 || dy[i]!=dy[i-1]){
			dy[ty++]=dy[i];
		}
	}
	for(int i=0;i<tx;i++) _x[dx[i]] = i;
	for(int i=0;i<ty;i++) _y[dy[i]] = i;
	for(int i=0;i<n;i++){
		in[_x[x[i]-k+1]].push_back(y[i]);
		out[_x[x[i]+1]].push_back(y[i]);
	}
	for(int i=0;i<tx;i++){
		for(int j=0;j<out[i].size();j++){
			int t1 = _y[out[i][j]-k+1], t2 = _y[out[i][j]+1];
			for(int p=t1;p<t2;p++){
				f[h[p]]+=(LL)(dy[p+1]-dy[p])*(dx[i]-dx[g[p]]);
				g[p]=i;
			}
		}
		for(int j=0;j<in[i].size();j++){
			int t1 =  _y[in[i][j]-k+1], t2 = _y[in[i][j]+1];
			for(int p=t1;p<t2;p++) {
				f[h[p]]+=(LL)(dy[p+1]-dy[p])*(dx[i]-dx[g[p]]);
				g[p]=i;
			}
		}
		for(int j=0;j<out[i].size();j++){
			int t1 = _y[out[i][j]-k+1], t2 = _y[out[i][j]+1];
			for(int p=t1;p<t2;p++) h[p]--;
		}
		for(int j=0;j<in[i].size();j++){
			int t1 =  _y[in[i][j]-k+1], t2 = _y[in[i][j]+1];
			for(int p=t1;p<t2;p++) h[p]++;
		}
	}
	for(int i=1;i<=n;i++) printf("%I64d ",f[i]);
	printf("\n");
	
	return 0;
}
