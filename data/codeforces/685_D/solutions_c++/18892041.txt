#include<cstdio>
#include<iostream>
#include<string>
#include<cstring>
#include<algorithm>
using namespace std;

#define ll long long

struct point
{int x,y;
};
struct point a[200010];

struct event
{int l,r,y,z;
};
struct event b[400010];

int n,m,N,tot;
int sa[400010],pre[400010],val[400010];
long long ans[200010];

bool operator <(event A,event B)
{return A.y<B.y;
}

void change(int i)
{int j;
 
 for(j=b[i].l;j<=b[i].r;j++)
 {	if(val[j]) ans[val[j]]+=(ll)(b[i].y-pre[j])*(sa[j+1]-sa[j]);
 	val[j]+=b[i].z;
 	pre[j]=b[i].y;
 }
}

int main()
{int i,j,x,y;
 
 scanf("%d%d",&n,&m);
 for(i=1;i<=n;i++)
 {	scanf("%d%d",&a[i].x,&a[i].y);
 	sa[++tot]=a[i].x-m+1;
 	sa[++tot]=a[i].x+1;
 }
 sort(sa+1,sa+tot+1);
 tot=unique(sa+1,sa+tot+1)-sa-1;
 for(i=1;i<=n;i++)
 {	x=lower_bound(sa+1,sa+tot+1,a[i].x-m+1)-sa;
 	y=lower_bound(sa+1,sa+tot+1,a[i].x+1)-sa-1;
 	b[++N]=(event){x,y,a[i].y-m,1};
 	b[++N]=(event){x,y,a[i].y,-1};
 }
 sort(b+1,b+N+1);
 for(i=1;i<=N;i++) change(i);
 for(i=1;i<=n;i++)
 { 	printf("%I64d",ans[i]);
 	printf("%c",i!=n?' ':'\n');
 }
 return 0;
}
