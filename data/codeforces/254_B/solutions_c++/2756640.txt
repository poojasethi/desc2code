#include<stdio.h>
using namespace std;
int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int n,A[13],i,m,d,p,t,day,max;
	int X[480]={0};
	scanf("%d",&n);
	A[1]=31;A[2]=28;A[3]=31;A[4]=30;A[5]=31;A[6]=30;A[7]=31;A[8]=31;A[9]=30;A[10]=31;A[11]=30;A[12]=31;
	for(i=2;i<=12;i++) A[i]+=A[i-1] ;
	A[0]=0;
	while(n--) {
		scanf("%d%d%d%d",&m,&d,&p,&t);
		day=A[m-1]+d+100;
		for(i=1;i<=t;i++) X[day-i]+=p;
	}
	max=0;
	for(i=1;i<=465;i++) if(X[i] > max) max=X[i];
	printf("%d\n",max);
	return 0;
}
