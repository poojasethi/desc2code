#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
int D[12]={31,28,31,30,31,30,31,31,30,31,30,31};
int kol[500]={0};
int n,m,d,p,t;
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>n;
for (int i=0;i<n;i++)
 {
  cin>>m>>d>>p>>t;
  int td=d;
  for (int j=0;j<m-1;j++) td+=D[j];
  for (int j=td-t-1+100;j<td-1+100;j++) kol[j]+=p;
 }
int i=0;
for (int j=1;j<500;j++) if (kol[j]>kol[i]) i=j;
cout<<kol[i];
return 0;
}
