#include <iostream>
#include <fstream>
#include <cstring>
#include <cstdio>
#include <cmath>
using namespace std;
int f[]={0,0,31,59,90,120,151,181,212,243,273,304,334};
int cnt[1000],n;

int main(){
	ifstream fin("input.txt");
	ofstream fout("output.txt");
	while(fin>>n){
		memset(cnt,0,sizeof(cnt));
		while(n--){
			int m,d,p,t;
			fin>>m>>d>>p>>t;
			int j=400+f[m]+d;
			while(t--){
				cnt[j--]+=p;
			}
		}
		int maxx=0;
		for(int i=0;i<1000;i++)
			maxx=max(maxx,cnt[i]);
		fout<<maxx<<endl;
	}
	return 0;
}
