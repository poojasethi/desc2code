#include<stdio.h>
#include<string.h>
#define N 100005
int cnt[N];
int mon[13]={31,28,31,30,31,30,31,31,30,31,30,31};
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	int i,n,a,b,c,d;
	scanf("%d",&n);
	while(n--)
	{
		scanf("%d%d%d%d",&a,&b,&c,&d);
		int day=366;
		for(i=0;i<a-1;i++)day+=mon[i];
		day+=b;
		for(i=0;i<d;i++)cnt[day-i]+=c;
	}
	int ans=0;
	for(i=0;i<=1000;i++)if(ans<cnt[i])ans=cnt[i];
	printf("%d\n",ans);
	return 0;
}
