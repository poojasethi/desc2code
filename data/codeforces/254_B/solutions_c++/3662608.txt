#include<fstream>
using namespace std;
int main()
{
    ifstream in("input.txt");
    ofstream out("output.txt");
    int month[12] = {0,31,59,90,120,151,181,212,243,273,304,334};
    int tests,MAX=0,hobo[600]={0};
    int m,d,p,t,d1;
    in>>tests;
    
    while(tests--)
    {

       in>>m>>d>>p>>t;
        m--;
        d1=100+month[m]+d;
for(int j = 1; j <= t; ++j) {
            hobo[d1- j] += p;
            if(hobo[d1- j] > MAX)
                MAX = hobo[d1-j];
        }

}

    out<<MAX;  
    return 0;
}
