#include <algorithm>
#include <fstream>
using namespace std;

long long arr[1000], day1[13]={0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int main(){
	
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	long long n, m, d, p, t, i, j, k, ans;
	cin >> n;
	for(i=0;i<=12;i++) day1[i] += day1[i-1];
	for(i=0;i<n;i++){
		cin >> m >> d >> p >> t;
		for(j=500+day1[m-1]+d, k=0;k<t;k++, j--) arr[j]+=p;
	}
	ans = 0;
	for(i=0;i<1000;i++){
		ans = max(ans, arr[i]);
	}
	cout << ans << '\n';
	return 0;
}
