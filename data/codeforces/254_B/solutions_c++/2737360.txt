#include<iostream>
#include<fstream>

using namespace std;

int mnth[]={0,0,31,59,90,120,151,181,212,243,273,304,334};
int m,n,c,t,cnt[1010],d;

int main()
{
  ifstream fid("input.txt");
  ofstream fout("output.txt");
  
  fid>>n;
  
  for(int i=0;i<n;i++)
  {
    fid>>m>>d>>c>>t;
    for(int j=t; j>0;j--)
      cnt[mnth[m]+d+400 - j] += c;
  }
  
  int mn=0;
  for(int i=0;i<1000;i++)
    if(mn<cnt[i])mn=cnt[i];
    
  fout<<mn<<"\n";
  
  return 0;
}
