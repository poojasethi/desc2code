#include<stdio.h>#include<string.h>#include<algorithm>using namespace std;int month[15]={100,31,28,31,30,31,30,31,31,30,31,30,31};int sum[599];int main(){	freopen("input.txt", "r", stdin);	freopen("output.txt", "w", stdout);	int i,n,j,a,b,c,d;	for(i=1;i<13;i++) month[i]+=month[i-1];	scanf("%d",&n);	for(i=0;i<n;i++)	{		scanf("%d%d%d%d",&a,&b,&c,&d);		int ll=month[a-1]+b-d;		int rr=month[a-1]+b-1;		for(j=ll;j<=rr;j++)            sum[j]+=c;	}	int maxn=-1;;	for(i=0;i<566;i++)		if(sum[i]>maxn)			maxn=sum[i];	printf("%d\n",maxn);	return 0;}
