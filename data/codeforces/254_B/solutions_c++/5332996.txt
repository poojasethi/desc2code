#include<fstream>
#include<iostream>
using namespace std;
int main()
{
    ifstream in("input.txt");
    ofstream out("output.txt");
    int mt[12]={0,31,59,90,120,151,181,212,243,273,304,334};
    int a[500]={},Max=0;
    int n;
    in>>n;
    for(int i=0;i<n;i++)
    {
        int m,d,p,t;
        in>>m>>d>>p>>t;
        m--;
        for(int j=0;j<t;j++)
            Max=max(Max,a[mt[m]+d+100-j]+=p);
    }
        out<<Max;
    return 0;
}
