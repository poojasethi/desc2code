#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{
int D[12]={31,28,31,30,31,30,31,31,30,31,30,31};
int kol[500]={0};
int n,m,d,p,t;
freopen("input.txt","r",stdin);
freopen("output.txt","w",stdout);
cin>>n;
for (int i=0;i<n;i++)
 {
  cin>>m>>d>>p>>t;
  int td=d;
  for (int j=0;j<m-1;j++) td+=D[j];
  for (int j=td-t-1;j<td-1;j++) kol[j+100]+=p;
 }
int ans=0;
for (int i=1;i<500;i++) ans=max(ans,kol[i]);
cout<<ans;
return 0;
}
