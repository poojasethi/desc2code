
#include <iostream>
#include <cstdio>
#include <cstring>
#include <map>
#include <algorithm>

using namespace std;

const int day[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
int n, m, d, p, t, r;
map<int, int> M;

int main() {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		scanf("%d%d%d%d", &m, &d, &p, &t);
		while (--m) d += day[m];
		while (t--) r = max(r, M[--d] += p);
	}
	printf("%d\n", r);
	return 0;
}
