#include<bits/stdc++.h>
using namespace std;
int main(){
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    int n,a[466]={0},m,d,p,t,mo[12]={0,31,28,31,30,31,30,31,31,30,31,30};
    for(int i=1;i<12;i++){mo[i]+=mo[i-1];/*cout<<mo[i]<<" ";*/}
    cin>>n;
    for(int i=1;i<=n;i++){
        cin>>m>>d>>p>>t;
        int indx=mo[m-1]+d+100;
        for(int j=indx-1;j>=indx-t;j--){
            a[j]+=p;
        }
    }
    int mx=a[0];
    for(int i=1;i<466;i++)if(a[i]>mx)mx=a[i];
    cout<<mx<<endl;
}
