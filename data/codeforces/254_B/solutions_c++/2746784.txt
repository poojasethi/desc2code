#include<iostream>
#include<fstream>

using namespace std;
int n,m,d,p,t,cal[566] = {0},d1;
int month[13] = {0,0,31,59,90,120,151,181,212,243,273,304,334};

int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);

	cin >> n;
	int max = 0;
	for(int i = 0; i < n; ++i) {
		cin >> m >> d >> p >> t;
		d1 = 200 + month[m] + d;
		for(int j = 1; j <= t; ++j) {
			cal[d1 - j] += p;
			if(cal[d1 - j] > max)
				max = cal[d1-j];
		}
	}

	cout << max << endl;
		
	return(0);
}
