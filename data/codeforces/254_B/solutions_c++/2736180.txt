#include <iostream>
#include <cstdio>

using namespace std;

int m[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int ans, x, y, p, t, a[999], i, j, n;

int main () {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);

	cin >> n;
	for (i = 0; i < n; i++) {
		cin >> x >> y >> p >> t;
		for (j = 1; j < x; j++)
			y += m[j];
		for (j = 200 + y - 1; t > 0; t--, j--)
			a[j] += p;
	}
	for (i = 0; i < 999; i++)
		ans = max(ans, a[i]);
	cout << ans;
}
