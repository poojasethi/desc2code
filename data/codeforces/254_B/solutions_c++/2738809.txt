#include <fstream>
#include <vector>
using namespace std;

#define f(x,y) for (int x = 0; x < y; ++x)

ifstream in("input.txt");
ofstream out("output.txt");

typedef vector<int> VI;

const int ds[] = {31, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

int main() {
  int n; in >> n;
  VI v(500); int M = 0;
  f(i,n) {
    int m, d, p, t;
    in >> m >> d >> p >> t;
    int w = 105; for (int j = 1; j < m; ++j) w += ds[j]; w += d;
    f(j,t) {
      v[--w] += p;
      M = max(M, v[w]);
    }
  }
  out << M << endl;
}
