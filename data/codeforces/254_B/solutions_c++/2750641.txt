#include <iostream>
#include <fstream>
using namespace std;

int a[13]={400,431,459,490,520,551,581,612,643,673,704,734,765};
int n,i,j,answ,p,d,t,m,st,fn;
int b[1000];
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    cin>>n;
    for (i=1;i<=n;i++)
    {
        cin>>m>>d>>p>>t;
        st=a[m-1]+d-t;
        fn=a[m-1]+d-1;
        for (j=st;j<=fn;j++)
        {
            b[j]=b[j]+p;
            if (b[j]>answ)answ=b[j];
            }
    }
    cout<<answ;
    return 0;
}
