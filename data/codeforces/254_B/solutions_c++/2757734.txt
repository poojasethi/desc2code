#include<stdio.h>
#include<string.h>
int M[]={0,31,28,31,30,31,30,31,31,30,31,30,31};
int member[800];
int main()
{
	int n,i,j,m,d,c,t;
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	while(scanf("%d",&n)!=EOF)
	{
		memset(member,0,sizeof(member));
		for(i=1;i<=n;i++)
		{
			scanf("%d%d%d%d",&m,&d,&c,&t);
			for(j=1;j<m;j++)d+=M[j];
			for(j=1;j<=t;j++)
				member[d-j-1+365]+=c;
		}
		for(i=1,c=0;i<800;i++)if(member[i]>c)c=member[i];
		printf("%d\n",c);
	}
	return 0;
}
