#include<iostream>
#include<fstream>
using namespace std;
int luni[14]={0,101,132,160,191,221,252,282,313,344,374,405,435};
int x[472],m,d,p,t,i,j;
int main()
{
	int n,max;
	ifstream f("input.txt");
	ofstream g("output.txt");
	f>>n;
	for(i=0;i<n;++i)
	{
		f>>m>>d>>p>>t;
		for(j=luni[m]+d-t;j<luni[m]+d;++j)
		{
			x[j]+=p;
		}
	}
	max=0;
	for(i=0;i<470;++i)
		if(x[i]>max)
			max=x[i];
	g<<max;
	return 0;
}
