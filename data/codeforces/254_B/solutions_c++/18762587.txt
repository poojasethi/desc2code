#include<bits/stdc++.h>

using namespace std;
int month[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
int arr[999];


int main () {
	freopen("input.txt", "r", stdin);
	freopen("output.txt", "w", stdout);
int ans=0, m, d, p, t, i, j, n;
	cin >> n;
	for (i = 0; i < n; i++) {
		cin >> m >> d >> p >> t;

		for (j = 1; j < m; j++)
			d += month[j];

		for (j =100+ d - 1; t > 0; t--, j--)
			arr[j] += p;
	}
	for (i = 0; i < 999; i++)
		ans = max(ans, arr[i]);
	cout << ans;
}
