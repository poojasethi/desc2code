#include<cstdio>

int a[466]={0},k[12]={100,131,159,190,220,251,281,312,343,373,404,434},n,m,d,p,t,s=0;

int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);

    for(scanf("%d",&n);n>0;n--)
    {
        scanf("%d%d%d%d",&m,&d,&p,&t);

        for(d+=k[m-1];t>0;t--,d--)
            a[d]+=p;
    }

    for(d=0;d<466;d++)
        s=a[d]>s?a[d]:s;

    printf("%d",s);

    return 0;
}
