#include<stdio.h>
#include<stdlib.h>
#include<algorithm>
#define foru(i,a,b) for(int i=a;i<=b;++i)

using namespace std;
const int s[13]={0,0,31,59,90,120,151,181,212,243,273,304,334};
int n,a,b,c,d,ans,x[510],now;
int main()
{
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&n);
	foru(i,1,n) scanf("%d%d%d%d",&a,&b,&d,&c),x[s[a]+b+100-c]+=d,x[s[a]+b+100]-=d;
	foru(i,1,500) now+=x[i],ans=max(now,ans);
	printf("%d\n",ans);
	return 0;
}
