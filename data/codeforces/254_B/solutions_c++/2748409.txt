#include <iostream>
#include <fstream>

using namespace std;

int main()
{
    ifstream in("input.txt");
    ofstream out("output.txt");

    int mtds[12] = {1,32,60,91,121,152,182,213,244,274,305,335}, a[470] = {}, n, Max = 0;
    in >> n;
    for (int q = 0; q < n; ++q)
	{
		int m, d, w, t;
		in >> m >> d >> w >> t;
		--m;
		for (int i = 0; i < t; ++i)
			Max = max(Max, a[mtds[m] + d + 101 - i] += w);
	}
	out << Max;

    return 0;
}
