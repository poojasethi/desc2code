#include<iostream>
#include<fstream>
#include<algorithm>
using namespace std;
int arr[12]={0,31,59,90,120,151,181,212,243,273,304,334};
int a[500]={};
int main()
{
	ifstream cin("input.txt");
	ofstream cout("output.txt");
	int n,days=0;
	cin>>n;
	int m,d,p,t;
	for(int i=0;i<n;i++)
	{
		cin>>m>>d>>p>>t;
		for(int j=0;j<t;j++)
			days=max(days,a[100+arr[m-1]+d-j]+=p);
	}
	cout<<days<<endl;
	return 0;
}
