#include<bits/stdc++.h>

int month[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
int a[999];


int main ()
{
	freopen("input.txt", "r", stdin);
    freopen("output.txt", "w", stdout);
int c=0, m, d, p, t, i, j, n;
	cin >> n;
	for (i = 0; i < n; i++)
    {
		cin >> m >> d >> p >> t;
		for (j = 1; j < m; j++)
			d += month[j];//cout<<d<<" ";
		for (j =100+ d - 1; t > 0; t--, j--)
			{a[j] += p;//cout<<a[j]<<" ";}
	}
}

		c = max(c, a[i]);

}
