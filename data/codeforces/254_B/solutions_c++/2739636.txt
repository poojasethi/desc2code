#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
const int shift = 200;
int req[700];
main( ) {
	freopen( "input.txt", "r", stdin);
	freopen( "output.txt", "w", stdout);
	int dm[] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31}, n, r = 0;
	int m, d, p, t, cno;
	scanf ("%d", &n);
	while (n--) {
		scanf ("%d%d%d%d", &m, &d, &p, &t);
		cno = shift + d;
		for (int i = 0; i < m; i++) cno += dm[i];
		for (int i = 1; i <= t; i++) req[cno - i]+=p;
	}
	for (int i = 0; i <= 700; i++) r = max(r, req[i]);
	printf ("%d\n", r);
	return 0;
}
