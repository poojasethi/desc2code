#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int f[1000];
int n,m,d,p,t;
int mon[12]={0,31,59,90,120,151,181,212,243,273,304,334};
int main()
{
    freopen("input.txt","r",stdin);
    freopen("output.txt","w",stdout);
    memset(f,0,sizeof(f));
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d%d%d%d",&m,&d,&p,&t);
        for(int j=200+mon[m-1]+d;t;t--,j--)f[j]+=p;
    }
    int s=0;
    for(int i=0;i<800;i++)
    {
        if(f[i]>s)s=f[i];
    }
    cout<<s<<endl;
    return 0;
}
