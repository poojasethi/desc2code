import sys
sys.stdin = open('input.txt','r')
sys.stdout = open('output.txt','w')
cnt = 1
mp = {}
for m,k in enumerate([31,28,31,30,31,30,31,31,30,31,30,31]):
	for d in range(k):
		mp[(m+1,d+1)] = cnt
		cnt = cnt + 1
n = input()
flag = [0]*400
ans = 0
for ch in sys.stdin:
	m,d,p,t=map(int,ch.split())
	day = mp[(m,d)]
	for i in range(max(day-t,0),day):
		flag[i] = flag[i] + p
		ans = max(ans,flag[i])
print ans
