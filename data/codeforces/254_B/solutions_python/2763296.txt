from datetime import datetime, timedelta
infile = open('input.txt', 'r')
n = int(infile.readline())
olympiad = []
for i in range(n):
    tmp = map(int, infile.readline().split())
    day = datetime(2013, tmp[0], tmp[1])
    olympiad.append((day-timedelta(days=tmp[3]), tmp[2]))
    olympiad.append((day, -tmp[2]))
infile.close()
ans = 0
olympiad.sort()
now = 0
outfile=open('output.txt','w')
for e in olympiad:
    now += e[1]
    if now>ans:
        ans=now
print >>outfile, ans
outfile.close()
