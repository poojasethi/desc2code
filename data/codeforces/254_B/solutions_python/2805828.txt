import sys
import datetime
import copy
def RedirectIO():
    sys.stdin=open("input.txt","r")
    sys.stdout=open("output.txt","w")

class MyDate:
    def __init__(self,year,month,day,person,begin=False):
        self.date=datetime.datetime(year,month,day)
        self.person=person
        self.begin=begin
    def __sub__(self,time):
        temp=copy.deepcopy(self)
        temp.date-=time
        return temp
    def __cmp__(self,other):
        if self.date<other.date:
            return -1
        if self.date==other.date:
            if self.begin!=other.begin:
                if self.begin==False:
                    return -1
                else:
                    return 1
            else:
                return 0
        return 1
    def __str__(self):
        return "%s %d %r" %(self.date,self.person,self.begin)
    def setBegin(self,begin):
        self.begin=begin
        return self
def main():
    RedirectIO()
    n=input()
    
    date=[]
    for i in range(n):
        data=map(int,raw_input().split())
        days=MyDate(2013,data[0],data[1],data[2])
        date.append(days)
        
        time=datetime.timedelta(data[3])
        date.append((days-time).setBegin(True))
    
    date.sort()
    ans=0
    for time in date:
        if time.begin==True:
            if ans<time.person:
                ans=time.person
            ans-=time.person
        else:
            ans+=time.person
    print ans
main()
