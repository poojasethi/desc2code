import sys
Month = (0,31,28,31,30,31,30,31,31,30,31,30,31)
def change(month, day):
  ans = 0
  for i in range(1,month):
    ans += Month[i]
  return ans + day

sys.stdin = open('input.txt','r')
sys.stdout = open('output.txt','w')
n = int(raw_input())
count = [0 for i in range(1000)]
for i in range(n):
  (m,d,p,t) = map(int,raw_input().split(' '))
  time = change(m,d)
  for j in range(time - t,time) :
    count[j+200] += p
print max(count)
