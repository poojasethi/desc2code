import datetime

inp = open('input.txt', 'r')
outp = open('output.txt', 'w')

n = int(inp.readline())

count = []
for i in range(1000):
        count.append(0)

for i in range(n):
        (m, d, p, t) = [int(j) for j in inp.readline().split()]
        date = datetime.date(2013, m, d) - datetime.timedelta(t)
        index = date - datetime.date(2012, 1, 1)
        for j in range(t):
                count[index.days + j] += p

outp.write(str(max(count)) + '\n')
