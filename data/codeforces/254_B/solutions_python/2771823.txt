#! /usr/bin/env python2
import datetime

inFile = open("input.txt", 'r')
outFile = open("output.txt", 'w')

c = {}

n = int(inFile.readline())

for line in inFile.readlines()[:n]:
    m, d, p, t = map(int, line.strip().split(' '))
    y = 2013
    delta = datetime.timedelta(1)
    now = datetime.date(y, m, d)
    for i in xrange(0, t):
        now -= delta
        c.setdefault(now, 0)
        c[now] += p
outFile.write("{:d}".format(max(c.values())))
