import datetime

def main():
    f = open('input.txt')
    w = open('output.txt','w')
    n = int(f.readline())
#    n = int(raw_input())

    res = [0] *1000
    for i in range(n):
        m,d,p,t = map(int, f.readline().strip('\n').split())
#        m,d,p,t = map(int, raw_input().split())
        day = datetime.date(2013,m,d) - datetime.date(2012,1,1)
        for j in range(t):
            res[day.days - j] += p
#    print(max(res))
    w.write(str(max(res)))
    w.close()
main()
