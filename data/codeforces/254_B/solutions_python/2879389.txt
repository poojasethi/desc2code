import sys
sys.stdin=open("input.txt","r")
sys.stdout=open("output.txt","w")
DaysByMounth = (31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31)
dayIndex = lambda m, d: sum(DaysByMounth[: m - 1]) + d
n = int(raw_input())
N = 365
timeline = [0] * (2 * N + 1)
for _ in xrange(n):
    m, d, p, t = map(int, raw_input().split())
    olympDay = dayIndex(m, d)
    for i in xrange(olympDay - t, olympDay):
        timeline[i + N] += p

print max(timeline)
