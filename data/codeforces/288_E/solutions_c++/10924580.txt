#include<cstring>
#include<cstdio>
#include<cstdlib>
using namespace std;

typedef long long ll;
const int A = 4,B = 7,rhl = 1000000007;
#define maxn (100010)
char s[maxn]; ll pow10[maxn],low[maxn]; int ans,n;

inline ll qua(ll a) { return a*a%rhl; }

inline void update(ll &ret,ll f,ll g,ll h,int i)
{
	ret += qua(pow10[n-i+1])*f%rhl;
	if (ret >= rhl) ret -= rhl;
	ret += pow10[n-i+1]*(pow10[n-i]+low[n-i])%rhl*(A+B)%rhl*g%rhl;
	if (ret >= rhl) ret -= rhl;
	ret += ((qua(pow10[n-i])+qua(low[n-i]%rhl))*A*B%rhl+(qua(A)+qua(B))*pow10[n-i]%rhl*low[n-i])%rhl*h%rhl;
	if (ret >= rhl) ret -= rhl;
}

inline int work()
{
	scanf("%s",s+1); n = strlen(s+1);
	ll f = 0,g = 0,h = 0,tmp = 0,ret = 0;
	for (int i = 1;i <= n;++i)
	{
		update(ret,f,g,h,i);
		ll tf = f,tg = g,th = h;
		f = 0,g = 0,h = 0;
		f = (200*tf%rhl+(20*(A+B))*tg%rhl+(qua(A)+qua(B))*th%rhl)%rhl;
		g = (tg*20%rhl+th*(A+B))%rhl;
		h = th+th; if (h >= rhl) h -= rhl;
		if (s[i] == B+'0')
		{
			update(ret,tmp*tmp%rhl,tmp,1,i);
			f += qua((tmp*10+A)%rhl); if (f >= rhl) f -= rhl;
			g += (tmp*10+A)%rhl; if (g >= rhl) g -= rhl;
			++h; if (h >= rhl) h -= rhl;
		}
		tmp = (tmp*10+s[i]-'0')%rhl;
	}
	return ret;
}

int main()
{
	pow10[0] = 1;
	for (int i = 1;i <= 100000;++i) pow10[i] = 10*pow10[i-1]%rhl,low[i] = (10*low[i-1]+1LL)%rhl;
	ans -= work(); ans += work();
	if (ans >= rhl) ans -= rhl; if (ans < 0) ans += rhl;
	printf("%d",ans);
	return 0;
}
