#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <string>
using namespace std;
long long dp1[100005][2],dp2[100005][2],dp3[100005][2],sum1[100005],sum2[100005],ans,len[100005];
int i;
char s[100005];
const int MOD=1000000007;
void work()
{
    int p,i;
    p=strlen(s);
    dp1[0][1]=1; dp2[0][1]=0; dp3[0][1]=0;
    for (i=1; i<=p; i++)
    {
        dp1[i][1]=dp1[i-1][1];
        dp1[i][0]=dp1[i-1][0]*2;
        if (s[i-1]=='7') dp1[i][0]+=dp1[i-1][1];
        dp1[i][0]%=MOD;
        dp2[i][1]=(dp2[i-1][1]*10+s[i-1]-'0')%MOD;
        dp2[i][0]=(dp2[i-1][0]*20+11*dp1[i-1][0])%MOD;
        if (s[i-1]=='7') dp2[i][0]=(dp2[i][0]+dp2[i-1][1]*10+4*dp1[i-1][1])%MOD;
        dp3[i][1]=(dp3[i-1][1]*100+(s[i-1]-'0')*(s[i-1]-'0')+2*(s[i-1]-'0')*dp2[i-1][1]*10)%MOD;
        dp3[i][0]=(dp3[i-1][0]*200+65*dp1[i-1][0]+22*dp2[i-1][0]*10)%MOD;
        if (s[i-1]=='7') dp3[i][0]=(dp3[i][0]+dp3[i-1][1]*100+16+8*dp2[i-1][1]*10)%MOD;
    }
    for (i=1; i<=p; i++)
    {
        if (s[i-1]=='7')
        ans=(ans+dp3[i-1][1]*len[p-i+1]%MOD*len[p-i+1]+dp2[i-1][1]*len[p-i+1]%MOD*(sum1[p-i+1]+sum2[p-i+1])+dp1[i-1][1]*sum1[p-i+1]%MOD*sum2[p-i+1])%MOD;
        ans=(ans+dp3[i-1][0]*len[p-i+1]%MOD*len[p-i+1]+dp2[i-1][0]*len[p-i+1]%MOD*(sum1[p-i+1]+sum2[p-i+1])+dp1[i-1][0]*sum1[p-i+1]%MOD*sum2[p-i+1])%MOD;
    }
}
int main()
{
    scanf("%s",s);
    len[0]=1;
    for (i=1; i<=100000; i++) len[i]=len[i-1]*10%MOD;
    sum1[1]=4; sum2[1]=7;
    for (i=2; i<=100000; i++) sum1[i]=(sum1[i-1]*10+7)%MOD,sum2[i]=(sum2[i-1]*10+4)%MOD;
    work();
    ans=(MOD-ans)%MOD;
    scanf("%s",s);
    work();
    printf("%d\n",ans);
    return 0;
}
