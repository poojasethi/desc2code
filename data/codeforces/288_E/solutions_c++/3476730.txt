#include <cstdio>
#include <iostream>
#include <cstring>
#define fr(i,a,b) for(int i=a;i<b;i++)
#define rep(i,b) fr(i,0,b)
#define MOD 1000000007LL

using namespace std;
typedef long long ll;


#define N 100100
char s1[N], s2[N];
ll ini;

ll mod(ll a){ return ((a%MOD)+MOD)%MOD; }
ll func(char* s, int t, bool som){
    ll r=0;
    ll ul=0;
    
    ll soma=0,ls=ini;
    ll qt = 0LL;
    
    bool tem=false;
    
    ll sum, lst, val, bas, qtd;
    sum = lst = ini;
    val = 0;    
    bas = 3LL;
    qtd = 1LL;
    for(int i=t-1,j=0;i>=0;i--,j++){
        ul = (ul*10 + s[j]-'0')%MOD;
        ll bas2 = (bas*bas)%MOD;
        if(s[i]=='7'){
            if(qt>0) r = (r+((qt-1)*bas2)%MOD + (bas*(2*soma-ls-ini))%MOD + (lst*(ini+bas))%MOD)%MOD;
            r = (r+val)%MOD;
            soma = (soma + sum + qt*bas)%MOD;
            qt = (qt+qtd)%MOD;
            
            if(tem) ls = (ls+bas)%MOD;
            tem = true;
        }else if(!tem) ls = (ls+bas)%MOD;       
        
        val = (2*val + (lst*(bas+ini))%MOD + (bas2*(qtd-1))%MOD + (bas*(2*sum-lst-ini))%MOD)%MOD;
        sum = (2*sum + qtd*bas)%MOD;
        lst = (lst + bas)%MOD;
        bas = (bas*10)%MOD;
        qtd = (qtd*2)%MOD;
    }   
    if(tem) return (r+ul*ls)%MOD;
    return r;
}

int main(){
    int tam;
    scanf("%s %s",s1,s2);
    tam = strlen(s1);
    
    ini = 0;
    rep(i,tam) ini = (ini*10 + 4LL)%MOD;
    
    cout << mod(func(s2,tam,true)-func(s1,tam,false)) << endl;
    
    return 0;
}
