#include<cstdio>
#include<cstring>
typedef long long ll;
char l[111111],r[111111];
int n,f[2][2][111111],g[2][2][111111],h[2][2][111111];
int p4[111111],p7[111111],p10[111111];
#define MOD 1000000007
#define UPT(x,y) {(x)+=(y);if((x)>=MOD)(x)-=MOD;}
#define F(i,n) for(int i=0;i<n;++i)
#define FD(i,n) for(int i=n-1;i>=0;--i)
int main(){
    scanf("%s%s",l,r);n=strlen(l);
    p4[0]=p7[0]=0,p10[0]=1;
    F(i,n){
        p4[i+1]=((ll)p4[i]*10+4)%MOD;
        p7[i+1]=((ll)p7[i]*10+7)%MOD;
        p10[i+1]=(ll)p10[i]*10%MOD;
    }
    F(lo,2)F(hi,2){
        int a=lo?l[n-1]-'0':4,b=hi?r[n-1]-'0':7;
        if(a==4&&b==7){
            f[lo][hi][n-1]=1;
            g[lo][hi][n-1]=11;
            h[lo][hi][n-1]=28;
        }
    }
    FD(i,n-1)F(lo,2)F(hi,2){
        int a=lo?l[i]-'0':4,b=hi?r[i]-'0':7;
        if(a==4&&b==7){
            int p=((ll)p10[n-i-1]*4+p7[n-i-1])%MOD,q=((ll)p10[n-i-1]*7+p4[n-i-1])%MOD;
            UPT(f[lo][hi][i],1);
            UPT(g[lo][hi][i],(p+q)%MOD);
            UPT(h[lo][hi][i],(ll)p*q%MOD);
        }
        for(int _=4;_<=7;_+=3)if(a<=_&&_<=b){
            int t=(ll)p10[n-i-1]*_%MOD;
            int lo_=lo&(_==a),hi_=hi&&(_==b);
            UPT(f[lo][hi][i],f[lo_][hi_][i+1]);
            UPT(g[lo][hi][i],g[lo_][hi_][i+1]) UPT(g[lo][hi][i],(ll)t*2%MOD*f[lo_][hi_][i+1]%MOD);
            UPT(h[lo][hi][i],h[lo_][hi_][i+1]) UPT(h[lo][hi][i],(ll)t*g[lo_][hi_][i+1]%MOD) UPT(h[lo][hi][i],(ll)t*t%MOD*f[lo_][hi_][i+1]%MOD);
        }
    }
    printf("%d\n",h[1][1][0]);
    return 0;
}
