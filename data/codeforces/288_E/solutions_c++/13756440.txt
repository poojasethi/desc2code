#include <iostream>
#include <cstring>
#include <cstdio>
#include <cmath>
#include <algorithm>
using namespace std;
#define N 200050
#define P 1000000007
#define ll long long
ll ans,a2[N],a0[N],a4[N],a7[N],mul[N],s[N],n,m;char a[N],b[N];
ll Calc(ll x,int y)
 {
    if (y==1) return (x+4)*(x+7)%P;
    x=(x+4*a0[y-1])%P;ll tot=(mul[y-1]+a4[y-1]*a7[y-1]%P)%P;
    tot=(tot+x*2*s[y-1]%P)%P;
    tot=(tot-x*a7[y-1]%P+P)%P;
    tot=(tot+(x+3*a0[y-1])%P*a7[y-1]%P)%P;
    tot=(tot+x*x%P*(a2[y-1]+P-1)%P+x*(x+3*a0[y-1])%P)%P;
    return tot;
 }
ll Solve()
 {
    ll tot=0,now=0;
    for (int i=0;i<n;i++)
     {
        if (a[i]=='7') tot=(tot+Calc(now,n-i))%P;
        now=(now+(a[i]-'0')*a0[n-i-1]%P)%P;
     }
    return tot;
 }
int main()
 {
    gets(a);n=strlen(a);
    a0[0]=a2[0]=1;
    for (int i=1;i<=2*n;i++)
      a0[i]=a0[i-1]*10%P,a2[i]=a2[i-1]*2%P,
      a4[i]=(a4[i-1]*10+4)%P,a7[i]=(a7[i-1]*10+7)%P;
    mul[1]=28;s[1]=11;
    for (int i=2;i<=n;i++)
      s[i]=(s[i-1]*2+11*a0[i-1]%P*a2[i-1]%P)%P,
      mul[i]=(2*mul[i-1]+a4[i-1]*a7[i-1]%P+
        11*a0[i-1]%P*s[i-1]*2%P+
        65*(a2[i-1]+P-1)%P*a0[2*i-2]%P+28*a0[2*i-2]%P)%P,
      mul[i]=(mul[i]-4*a0[i-1]%P*a7[i-1]%P-7*a0[i-1]%P*a4[i-1]%P+2*P)%P;
    ans=Solve();gets(a);
    ans=(Solve()-ans+P)%P;
    cout <<ans<<endl;
 }
