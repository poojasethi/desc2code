#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 100010
#define mo 1000000007
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int64 i,j,k,n,m;
int64 s[N][10],sum[N][10],f[N][10],a[11][N],a1,a2,a3,x,an;
//s:ge sum:he f:ans
char p[N],v[N];
inline int64 work(char p[N]) {
	int i; int64 he=0,re=0;
	for (i=n;i;i--) {
		if (p[n-i+1]=='7') {
			x=a[10][i];
			int64 A=s[i][4];
			int64 B=sum[i][4];
			a1=f[i][4];
			a2=he*he%mo*x%mo*x%mo*(A-1)%mo;
			a3=B*2-a[4][i]-a[7][i-1]-4*a[10][i-1];
			a3%=mo;
			a3=he*x%mo*a3%mo;
			re=(re+a2+a3+a1)%mo;
			if (re<0) re+=mo;
			a1=(he*x+4*a[10][i-1]+a[7][i-1])%mo;
			a2=(he*x+7*a[10][i-1]+a[4][i-1])%mo;
			re=(re+a1*a2)%mo;
		}
		he=(he*10+p[n-i+1]-48)%mo;
	}
	return re;
}
int main() {
	//freopen("288e.in","r",stdin);
	scanf("%s%s",v+1,p+1);
	n=strlen(p+1);
	For(i,1,n) a[4][i]=(a[4][i-1]*10+4)%mo,a[7][i]=(a[7][i-1]*10+7)%mo;
	a[10][0]=1;
	For(i,1,n) a[10][i]=a[10][i-1]*10%mo;
	s[1][4]=s[1][7]=1; sum[1][4]=4; sum[1][7]=7;
	For(i,2,n)For(j,4,7) if (j==4||j==7) {
		x=a[10][i-1];
		int64 A=(s[i-1][4]+s[i-1][7])%mo;
		int64 B=(sum[i-1][4]+sum[i-1][7]+j*x%mo*A%mo)%mo;
		a1=(4*a[10][i-2]+a[7][i-2])%mo*((7*a[10][i-2]+a[4][i-2])%mo)%mo+f[i-1][4]+f[i-1][7];
		a1%=mo;
		a2=j*j*x%mo*x%mo;
		a2=a2*(A-1)%mo;
		a3=2*(sum[i-1][4]+sum[i-1][7])-a[4][i-1]-a[7][i-1];
		a3%=mo;
		a3=j*x%mo*a3%mo;
		int64 C=(a2+a3+a1)%mo;
		s[i][j]=A,sum[i][j]=B,f[i][j]=C;
	}
	//printf("%d\n%d\n",work(v),work(p));
	an=(work(p)-work(v))%mo;
	if (an<0) an+=mo;
	printf("%I64d\n",an);
	return 0;
}
