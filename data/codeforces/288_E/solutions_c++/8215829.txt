#include <bits/stdc++.h>
#define P 1000000007
#define maxn 100005
#define ll long long
using namespace std;

int N;
char L[maxn],R[maxn];
ll p2[maxn],p10[maxn],p11[maxn],f[maxn],g[maxn];

ll G(ll x,int i) {return x*p10[i]%P;}
ll H(ll x,int i) {return x*p11[i]%P;}
ll F(ll x,int i)
{
	x=G(x,i);
	return (f[i]+x*x%P*(p2[i]-1)%P+(g[i]*2-H(4+7,i))%P*x)%P;
}

void work()
{
	p2[0]=p10[0]=1;
	for (int i=0; i<=N; i++)
	{
		p2[i+1]=p2[i]*2%P,p10[i+1]=p10[i]*10%P,p11[i+1]=(p11[i]*10+1)%P;
		f[i+1]=(F(4,i)+F(7,i)+((G(4,i)+H(7,i))%P)*((G(7,i)+H(4,i))%P))%P;
		g[i+1]=(G(4+7,i)*p2[i]+g[i]*2)%P;
	}
}

ll calc(char*s)
{
	ll x=0,y,res=0;
	for (int i=1,j; i<=N; i++) if (s[i]=='4') x=(x*10+4)%P;
	else
	{
		y=(x*10+4)%P,x=(x*10+7)%P,j=N-i;
		(res+=F(y,j)+((G(y,j)+H(7,j))%P)*((G(x,j)+H(4,j))%P))%=P;
	}
	return res;
}

void doit()
{
	scanf("%s%s",L+1,R+1),N=strlen(L+1);
	work();
	printf("%d\n",(calc(R)-calc(L)+P)%P);
}

int main()
{
	doit();
	return 0;
}
