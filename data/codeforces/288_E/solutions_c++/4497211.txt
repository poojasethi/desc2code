#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<cstring>
#define int64 long long
#define N 1200000
using namespace std;
const int mo=1000000007;
int64 pow10[N],pow2[N],low[N],s[N],sum[N];
char l[N],r[N];
int64 he(int64 x,int i){
	int64 y=x*pow10[i+1]%mo;
	return (y*y%mo*(pow2[i]-1) + (s[i]*2-11*low[i])%mo*y + sum[i])%mo;
}
void pret(){
	pow10[1]=1; pow2[0]=1; pow2[1]=2; low[1]=1;
	for(int i=1;i<=110000;++i){
		pow10[i+1]=pow10[i]*10%mo;
		pow2[i+1]=pow2[i]*2%mo;
		low[i+1]=(10*low[i]+1)%mo;
	}
	sum[0]=s[0]=0;
	for(int i=1;i<=100000;++i){
		sum[i]=(he(4,i-1) +he(7,i-1) + (4*pow10[i]+7*low[i-1])%mo*((7*pow10[i]+4*low[i-1])%mo))%mo;
		s[i]=(11*pow10[i]%mo*pow2[i-1]+s[i-1]*2)%mo;
	}
}
int64 work(char c[]){
	int n=strlen(c+1),i,j;
	int64 res=0,x=0,y;
	for(i=1;i<n;++i)
		res=(res+sum[i] + 7*low[i]*4%mo*low[i+1])%mo;
	for(i=1;i<=n;++i){
		if(c[i]=='7'){
			y=(x*10+4)%mo;
			res=(res+he(y,n-i) )%mo;
			res=(res+(y*pow10[n-i+1]+7*low[n-i])%mo*( ((y+3)*pow10[n-i+1]+4*low[n-i]) %mo))%mo;
			x=(x*10+7)%mo;
		}else x=(x*10+4)%mo;
	}
	return res;
}
int main(){
	pret();
	scanf("%s%s",l+1,r+1);
	printf("%d\n",(work(r)-work(l)+mo)%mo);
}
