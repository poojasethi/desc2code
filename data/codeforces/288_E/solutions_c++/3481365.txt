#include <iostream>
#include <cstdio>
#include <cstring>
#define mod 1000000007
using namespace std;
long long n,m,i,j,k,L;
long long f[100005],g[100005],u[100005],t[100005],o[100005];
char a[100005],b[100005];

long long Add(long long i,long long A){
  return (f[i]%mod+(A*g[i])%mod+((o[i]-1)%mod*((A*A)%mod))%mod)%mod;
}

void Re(){
 long long i,A1,A2;
 
  t[0]=1; o[0]=1; u[0]=0;
  for(i=1;i<=L;i++)
  { t[i]=(t[i-1]*10)%mod; 
	 o[i]=(o[i-1]<<1)%mod; 
	 u[i]=(u[i-1]*10+1)%mod; 
  }
  
//
  f[1]=28; g[1]=11; f[0]=g[0]=0;
  for(i=2;i<=L;i++)
  { 
	 A1=(4*t[i-1])%mod; A2=(7*t[i-1])%mod;
	 f[i]=(Add(i-1,A1)+Add(i-1,A2))%mod;
	 f[i]=(f[i]+(A1+(7*u[i-1])%mod)%mod*(A2+(4*u[i-1])%mod)%mod)%mod;
	 
	 g[i]=(2*g[i-1])%mod+((o[i]-2)%mod*(A1+A2)%mod)%mod;
	 g[i]=(g[i]+(A1+(7*u[i-1])%mod)%mod+(A2+(4*u[i-1])%mod)%mod)%mod;
  }
}

long long Getans(char *a){ 
 long long i,pre,now,ans;
	
 pre=0; ans=0;
 for(i=0;i<L;i++)
 { 
	pre=((pre*10)%mod+(a[i]-'0'))%mod;
	if(a[i]=='7')
	{ 
	   now=((pre-3)*t[L-i-1])%mod;
	   ans=(ans+(Add(L-i-1,now)+((((now+(7*u[L-i-1])%mod)%mod)*((now+(3*t[L-i-1])%mod+(4*u[L-i-1])%mod)%mod))%mod))%mod)%mod;
	}
 }
 return ans%mod;
}

int main()
{
  scanf("%s",&a); scanf("%s",&b);
  L=strlen(a);
	
//  题目已经明确 l < r
  if(L==1){ printf("28\n"); return 0; }
  
//
  Re();
  printf("%I64d\n",((Getans(b)%mod)-(Getans(a)%mod)+mod)%mod);  //先不加mod试一试
  
//
  return 0;
}
