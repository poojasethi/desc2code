#include <iostream>
#include <string>
using namespace std;
int main(){
    string s;
    bool minus=false;
    cin>>s;
    if(s[0]=='-') minus=true;
    int i,j;
    for(i=0;i<s.size();i++)
        if(s[i]=='.')
           break;
    if(minus) cout<<"(";
    cout<<"$";
    s=s+"000";
    for(j=0;j<i;j++){
        if(s[j]=='-')
            continue;
        if((i-j)%3==0 && (j-minus)!=0)
            cout<<",";
        cout<<s[j];
    }
    cout<<".";
    cout<<s[i+1]<<s[i+2];
    if(minus) cout<<")";
}
