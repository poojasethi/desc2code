#include <string>
#include <iostream>
using namespace std;

int main() {
	string x;
	bool isneg;
	size_t d;
	int i;

	cin >> x;
	isneg = (x[0] == '-');
	if (isneg)
		x.erase(0, 1);
	d = x.find('.');
	if (d == x.npos)
		x += ".00";
	else if (d == x.size() - 2)
		x += '0';
	else
		x.erase(d + 3, x.size() - (d + 3));
	i = x.find('.') - 3;
	while (i > 0) {
		x.insert(i, 1, ',');
		i -= 3;
	}
	x = '$' + x;
	if (isneg)
		x = '(' + x + ')';
	cout << x;
	return 0;
}
