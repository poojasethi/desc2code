#include<bits/stdc++.h>

using namespace std;

int main()
{
	char str[200]={};
	int t,i;
	gets(str);
	int len=strlen(str);
	bool P=*str!='-';
	printf("($"+P);
	if(!P)
		for(i=1;str[i-1];i++)
			str[i-1]=str[i];
	for(t=0;str[t];t++)
		if(str[t]=='.')
			break;

	for(i=0;i<t;i++)
		printf(",%c"+(!i || (t-i)%3),str[i]);
	printf(".");
	putchar(str[t+1]?str[t+1]:'0');
	putchar(str[t+2]?str[t+2]:'0');
	printf(")\n"+P);
    return 0;
}
