# include <iostream>
# include <string>
using namespace std;
int main ()
{
	string s,s1=".",s2=",",s3=".00",s4="0",s5="(",s6=")",s7="$";
	int i,l,kol=0,k;
	cin >> s;
	l=s.length();
	k=s.find(s1);
	if (k==-1)
		s+=s3;
	else
		if (k!=l-3)
			if (k<l-3)
				s.erase(k+3,l-k-3);
			else
				s+=s4;
	if (s[0]=='-')
	{
		s=s5+s7+s+s6;
		s.erase(2,1);
	}
	else
		s=s7+s;
	k=s.find(s1);
	for (i=k-1;i>=0 && s[i]!='$';i--)
	{
		kol++;
		if (kol==3 && s[i]!='(' && s[i-1]!='$')
		{
			s.insert(i,s2);
			kol=0;
		}
	}
	cout << s << endl;
	return 0;
}
