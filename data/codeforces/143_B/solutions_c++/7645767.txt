#include<iostream>
using namespace std;
int main()
{
	string s,s1="",s2="",s3="";
	cin>>s;
	int a,b,c,d,de,l=s.length();
	int i=0;
	a=b=c=d=0;
	while(i<l && s[i]!='.')
	{
		i++;
	}
	de=i;
	i--;
	while(i>=0 && s[i]!='-')
	{
		if(d==3)
		{
			s1=","+s1;
			d=0;
		}
			
		s1=s[i]+s1;
		d++;
		i--;
	}
	i=de+1;
	while(i<l)
	{
		if(s2.length()==2)
			break;
		s2=s2+s[i];
		i++;
	}
	while(s2.length()<2)
	{
		s2=s2+"0";
	}
	if(s[0]=='-')
	{
		cout<<"($"<<s1<<"."<<s2<<")";
	}
	else
		cout<<"$"<<s1<<"."<<s2;
	
	return 0;
}
