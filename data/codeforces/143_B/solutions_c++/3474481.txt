#include <iostream>
#include <cstring>
using namespace std;

int main()
{
	char c;
	string s;
	c=cin.peek();

	if(c=='-') cin.ignore(1);
	bool mi=c=='-';
	cin >> s;

	int n=s.length();
	int dot_pos=0;
	while(dot_pos<n && s[dot_pos]!='.') dot_pos++;

	if(mi) cout << '(';
	cout << '$';

	for(int i=0; i<dot_pos; i++)
	{
		if( (dot_pos-i)%3==1 && i!=dot_pos-1 ) cout << s[i] << ',';
		else cout << s[i];
	}
	cout << '.';
	for(int i=dot_pos+1; i<=dot_pos+2; i++)
	{
		if(i<n) cout << s[i];
		else cout << '0';
	}
	if(mi) cout << ')';
	return 0;
}
