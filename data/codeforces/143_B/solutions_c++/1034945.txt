#include<stdio.h>
#include<string.h>
#include<iostream>
using namespace std;
int main(){
	char a[105];
	int i,k,t;
	while(scanf("%s",a)!=EOF)
	{
		int len = strlen(a);
		for(k=0;k<len;k++)
			if(a[k]=='.') break;
		if(a[0]=='-') t=1;
		else t=0;
		if(t==1) printf("(");
		printf("$");
		for(i=t;i<k;i++)
		{
			if((k-i)%3==0 && i!=t)
			{
				printf(",");
			}
			printf("%c",a[i]);
		}
		if(k==len){
			printf(".00");
		}
		else if(k==len-2)
			printf(".%c0",a[k+1]);
		else
		{
			printf(".%c%c",a[k+1],a[k+2]);
		}
		if(t==1) printf(")");
		printf("\n");
		
	}
	return 0;
}
