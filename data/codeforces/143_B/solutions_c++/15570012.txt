#include<cstdio>
main(){
	char s[200]={};
	int t,i;
	gets(s);
	bool P=*s!='-';
	printf("($"+P);
	if(!P)
		for(i=1;s[i-1];i++)
			s[i-1]=s[i];
	for(t=0;s[t];t++)
		if(s[t]=='.')
			break;
	for(i=0;i<t;i++)
		printf(",%c"+(!i || (t-i)%3),s[i]);
	printf(".");
	putchar(s[t+1]?s[t+1]:'0');
	putchar(s[t+2]?s[t+2]:'0');
	printf(")\n"+P);
}
