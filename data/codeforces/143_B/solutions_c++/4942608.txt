#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
char str[109];
int main()
{
	scanf("%s",str);
	int i=0,p=0,l=strlen(str);
	while (str[p] && str[p]!='.')p++;
	if (!str[p])
	{
		p=l;
		str[p]='.';
		l++;
	}
	if (str[0]=='-')
	{
		printf("(");
		i++;
	}
	printf("$");
	for (;i<=p+2;i++)
	{
		if (i>l-1)
		{
			putchar('0');
			continue;
		}
		putchar(str[i]);
		if (i<p-1 && (p-i-1)%3==0)
			putchar(',');
		if (p && i==p+2)
			break;
	}
	if (str[0]=='-')
		printf(")");
	return 0;
}
