#include <cstring>
using namespace std;

string fr(string s2)
{
 int n=s2.size();
 for (int i=n-3; i>0;i-=3) s2.insert(i,",");
return s2;
}

{

 if (s[0]=='-') {s1='(';s.erase(0,1); f=true;}
 s1+='$';
 int i=s.find('.');
 if (i==-1) s1=s1+fr(s)+".00";
  else
