#include<iostream>
using namespace std;
int main()
{
string s;
int i,m=0,j=0;
cin>>s;
if(s[0]=='-') m=j=1;
if(m) cout<<"(";
cout<<"$";
for(i=0;i<s.length();i++)
{
if(s[i]=='.') break;
}
if(i==s.length()) s+=".00";
else s+="00";
for(i=0;i<s.length();i++)
{
if(s[i]=='.')
{
for(;j<i;j++)
{
if((i-j)%3==0 && (j>1 || j==1 && s[0]!='-')) cout<<",";
cout<<s[j];
}
cout<<".";
cout<<s[i+1]<<s[i+2];
if(m) cout<<")";
break;
}
}
return 0;
}
