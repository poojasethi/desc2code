#include <iostream>
#include <string>
using namespace std;
int i,j,o=0;
string a;
int main(){
    cin>>a;
    if(a[0]=='-')o=1,a=a.substr(1),cout<<'(';
	cout<<"$";
	for(i=0;i<a.length();i++)if(a[i]=='.')break;
	for(j=0;j<i;j++){
		if((i-j)%3==0 && j!=0 && j!=i-1)cout<<',';
		cout<<a[j];
	}
	if(i==a.length())cout<<".00";
	if(i+2==a.length() && a[i]=='.')cout<<"."<<a[i+1]<<"0";	
	if(i+3<=a.length() && a[i]=='.')cout<<"."<<a[i+1]<<a[i+2];	
	if(o==1)cout<<')';	
    return 0;
}
