#include <iostream>
#include <string>
using namespace std;
int i=0,j=0,x;
string s;
int main()
{
	cin>>s;
	if (s[0]=='-')
	{
		cout<<"(";
		i=1;
		j=1;
	}
	cout<<"$";
	while ((i<s.size())&&(s[i]!='.')) i++;
	x=i;
	if (s[0]=='-') x--;
	x%=3;
	if (x==0) x=3;
	while (j<i)
	{
		if (x==0)
		{
			cout<<',';
			x=3;
		}
		cout<<s[j];
		x--;
		j++;
	}
	if (i<s.size())
	{
		cout<<"."<<s[i+1];
		if (i+2<s.size())
			cout<<s[i+2];
		else cout<<"0";
	}
	else cout<<".00";
	if (s[0]=='-')
		cout<<")";
	return 0;
}
