#include <iostream>
#include <string>
using namespace std;
int main(){
    int i,t,sign=1;
    string s;
    cin >> s;
    if(s[0]=='-'){ sign=-1; s.erase(0,1);}
    t=s.find('.');
    if(t==-1) {t=s.length();s+=".00";}
    else if(s.length()-t==2) s+='0';
    else { s=s.substr(0,t+3);}
    for(i=t-3;i>0;i-=3)
        s.insert(i,",");
    s='$'+s;
    if(sign==-1) s='('+s+')';
    cout << s << endl;
    return 0;
}
