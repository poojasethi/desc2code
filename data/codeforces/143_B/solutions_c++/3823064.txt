#include <stdio.h>
#include <string>
using namespace std;
char str[128];
int main()
{
	scanf("%[^. \0\n]",str);
	string A=str,B;
	if(scanf(".%s",str)){
		B=str;
	}else
		B="00";
	B.resize(2,'0');
	bool f=A[0]=='-';
	if(f){
		A=A.substr(1);
		printf("(");
	}
	printf("$");
	for(int i=0;i<A.size();++i){
		printf("%c",A[i]);
		if(((signed)A.size()-i-1)%3==0 && i+1<A.size())
			printf(",");
	}
	printf(".%s",B.c_str());
	if(f)
		printf(")");
	return 0;
}
