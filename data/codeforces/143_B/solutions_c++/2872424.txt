# include <stdio.h>
# include <string.h>

int k,st,t;

char a[109];

int main()
{
	scanf("%s",a);
	
	st=strlen(a);
	
	a[st+1]='0';
	a[st+2]='0';
	a[st]='0';
	
	if(a[0]=='-')
	{
		printf("(");t=1;
		
		for(int i=1;i<st;i++)
		a[i-1]=a[i];
		
		st--;
		a[st]='0';
	}
	
	printf("$");
	
	
	for(int i=0;i<st;i++)
	if(a[i]=='.')
	k=i;
	
	if(k==0)
	k=st;
	
	for(int i=0;i<k;i++)
	{
		if((k-i)%3==0 && i!=0)
		printf(",");
		printf("%c",a[i]);
	}
	
	printf(".%c%c",a[k+1],a[k+2]);
	
	if(t==1)
	printf(")");
	
	getchar();
	getchar();
}
