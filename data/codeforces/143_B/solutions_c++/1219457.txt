#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
char s[500]={};
int main(void) {
	scanf("%s", s);
	int p=0;
	for(p=0;s[p]!='.' && s[p];p++);
	int flag=0, st=0;
	if(s[0]=='-') flag=st=1;
	if(flag) printf("(");
	printf("$");
	for(int i=st;i<p;i++) {
		printf("%c", s[i]);
		if(p-i>1 && (p-i)%3==1) printf(",");
	}
	printf(".");
	if(!s[p+1]) s[p+1]='0';
	if(!s[p+2]) s[p+2]='0';
	printf("%c%c", s[p+1], s[p+2]);
	if(flag) printf(")");
	printf("\n");
	return 0;
}
