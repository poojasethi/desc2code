#include <stdio.h>
#include <string.h>

char s[110];

int main()
{
	int n,i,flag=0,c=0;
	scanf("%s",s);
	n=strlen(s);
	if (s[0]=='-'){flag=1;printf("(");}
	printf("$");
	for (c=flag;c<n&&s[c]!='.';c++);
	for (i=flag;i<c;i++)
	{
		if (i!=flag&&(c-i)%3==0)printf(",");
		printf("%c",s[i]);
	}
	printf(".");
	if (s[c]=='.'&&c+1<n)printf("%c",s[c+1]);
	else printf("0");
	if (s[c]=='.'&&c+2<n)printf("%c",s[c+2]);
	else printf("0");
	if (flag)printf(")");
	printf("\n");
}
