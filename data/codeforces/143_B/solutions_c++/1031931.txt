#include <cstdio>

char s[105];
int d, c;

int main(){
	scanf("%s", s);
	for(int i = s[0] == '-'; s[i] && s[i] != '.'; i++, d++);
	if(s[0] == '-') printf("(");
	printf("$");
	for(int i = s[0] == '-', j = 0; s[i] && s[i] != '.'; i++, j++){
		printf("%c", s[i]);
		if((d - j - 1) % 3 == 0) printf(d == j + 1 ? "." : ",");
	}
	for(int i = (s[0] == '-') + d + 1; s[i] && c < 2; i++, c++) printf("%c", s[i]);
	for(; c < 2; c++) printf("0");
	if(s[0] == '-') printf(")");
	printf("\n");
}
