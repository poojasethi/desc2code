#include <iostream>
#include <cstdio>
using namespace std;
#define MAX 101

char s[MAX];

int main() {
	scanf("%s", s);
	int k = 0;
	if (s[0] == '-') {
		printf("(");
		k++;
	}
	printf("$");
	int i;
	for (i = 1; s[i]; i++) {
		if (s[i] == '.')
			break;
	};
	while (k < i) {
		printf("%c", s[k]);
		if(i-k == 1) printf(".");
			else
		if((i-k) % 3 == 1){
			printf(",");
		}
		k++;
	}
	if (s[k + 1])
		printf("%c", s[k + 1]);
	else
		printf("0");
	if (s[k + 2])
		printf("%c", s[k + 2]);
	else
		printf("0");
	if (s[0] == '-')
		printf(")");
	return 0;
}
