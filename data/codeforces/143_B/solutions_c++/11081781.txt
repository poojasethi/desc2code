#include<bits/stdc++.h>
using namespace std;
int main(){string s;bool f=0;char a='0',b='0';cin>>s;if(s[0]=='-')f=1,s.erase(0,1);int i=(f)?1:0;while(s[i]!='.'&&i<s.size())i++;int l=i+1;if(l<s.size()){a=s[l];l++;if(l<s.size())b=s[l];}string tmp;i--;for(int k=i-3;k>=0;k-=3,i++){tmp=s.substr(0,k+1);tmp+=',';tmp+=s.substr(k+1,i-k);s=tmp;}s+='.',s+=a,s+=b; l=0;while(l<s.size()&&s[l]!='.')l++;l+=3;if(l<s.size())s.erase(l,s.size()-l);if(f)cout<<"($"<<s<<")";else cout<<"$"<<s;
}
