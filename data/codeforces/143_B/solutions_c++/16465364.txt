#include<cstdio>
#include<cstring>
int d;
char s[101];
void print()
{
	int i,m;
	char a,b;
	for(i=0;i<d;i++)
	{
		if(s[i]=='.')
			break;
	}
	if(i+1<d)a=s[i+1];
	else a='0';
	if(i+2<d)b=s[i+2];
	else b='0';
	d=i;
	m=3-d%3;
	for(i=0;i<d;i++)
	{
		printf("%c",s[i]);
		if((m+i+1)%3==0&&i!=d-1)
			printf(",");
	}
	printf(".%c%c",a,b);
}
int main()
{
	int i;
	scanf("%s",s);
	d=strlen(s);
	if(s[0]=='-')
	{
		for(i=0;i<d-1;i++)
			s[i]=s[i+1];
		d--;
		printf("($");
		print();
		printf(")");
	}
	else
	{
		printf("$");
		print();
	}
	return 0;
}
