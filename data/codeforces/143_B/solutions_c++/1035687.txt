
#include<cstdio>

using namespace std;
int main(){
	string s;
	cin>>s;
	int l=s.size(),i,f=0,j,f1=0;
	for(i=0;i<l;i++){
		if(s[i]=='.'){
			f=1;
			break;
		}
	}
	if(s[0]=='-'){
		f1=1;
		cout<<"(";
	}
	cout<<"$";
	s=s+"000";
	for(j=0;j<i;j++){
		if(s[j]=='-')
			continue;
		if((i-j)%3==0 && (j-f1)!=0)
			cout<<",";
		cout<<s[j];
	}
//	if(f==1){
		cout<<".";
//	}
	cout<<s[i+1]<<s[i+2];
	if(s[0]=='-')
		cout<<")";
	cout<<endl;
	return 0;
}
