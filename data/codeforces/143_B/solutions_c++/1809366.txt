#include <cstdio>

int main() {
    char s[110];
    scanf("%s",s);
    int d=0;
    for(;s[d]&&s[d]!='.';d++);
    int i=0;
    bool neg=false;
    if (s[0]=='-') i++, neg=true;
    if (neg) printf("(");
    printf("$");
    for(;i<d;i++) {
        printf("%c",s[i]);
        if (d-i>1 && (d-i)%3==1) printf(",");
    }
    printf(".");
    if (!s[d]) printf("00");
    else if (!s[d+2]) printf("%c0",s[d+1]);
    else printf("%c%c",s[d+1],s[d+2]);
    if (neg) printf(")");
    printf("\n");
    return 0;
}
