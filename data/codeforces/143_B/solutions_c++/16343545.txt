#include<iostream>
using namespace std;

string s;

void prt(string s) {
	cout<<'$';
	int p = -1;
	for(int i=0;i<s.length();i++)
		if(s[i]=='.') p=i;
	if(p==-1) {
		p = s.length();
		s=s+".00";
	} else {
		s = s+"00";
		s = s.substr(0, p+3);
	}
	p-=3;
	while(p>0) {
		s = s.substr(0, p) + ","+s.substr(p);
		p-=3;
	}
	cout<<s;
}

int main() {
	cin>>s;
	if(s[0]=='-') {
		cout<<'(';
		prt(s.substr(1));
		cout<<')';
	} else prt(s);
	cout<<endl;
	
	return 0;
}
