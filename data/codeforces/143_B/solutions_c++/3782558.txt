#include<iostream>
#include<string>
using namespace std;

int main()
{
	string input,frac;
	getline(cin,input);
	int point=input.find(".");
	if(point!=string::npos)
	{
		frac=input.substr(point+1,2);
		input.erase(point);
		while(frac.size()<2)
		frac+="0";
	}
	else frac="00";
	int sign;
	if(input[0]=='-')
	{
		sign=-1;
		input.erase(0,1);
	}
	else sign=1;
	for(int i=input.size()-3;i>0;i-=3)
	input.insert(i,",");
	if(sign==-1)
	cout<<"($"<<input<<"."<<frac<<")";
	else cout<<"$"<<input<<"."<<frac;
	return 0;
}
