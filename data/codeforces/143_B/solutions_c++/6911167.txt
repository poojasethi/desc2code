# include <stdio.h>
# include <string.h>

int  l;
int  i;
int  j;
char s[100];
bool b;

int main()
{
    scanf ( "%s" ,s );
    
    l=strlen(s);
    
    if(s[0]=='-')
    b=1,printf ( "(" );
    printf ( "$" );
    
    for ( j=0; j<l && s[j]!='.' ; j++ );
    
    for ( i=(b==1); i<j ; i++ )
    {
        printf ( "%c" ,s[i] );
        if ( j-i-1!=0 && (j-i-1)%3==0 )
        printf ( "," );
    }
    
    printf ( ".%c%c" ,(j+1<l?s[j+1]:'0'),(j+2<l?s[j+2]:'0') );
    
    if ( b==1 )
    printf ( ")" );
    
    return 0;
}
