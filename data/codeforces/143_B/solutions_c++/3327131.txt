#include <iostream>
#include <cstdio>
#include <string>
#include <cctype>

using namespace std;

int main()
{
    string s, ans;
    cin >> s;
    s += ".00";
    int p = s.find('.');
    for(int i = p-1; i >= 0 && isdigit(s[i]); i--){
        if( i != p-1 && (p - i)%3 == 1) ans = ',' + ans;
        ans = s[i] + ans;
    }
    ans += s.substr(p,2);
    ans += isdigit(s[p+2])?s[p+2]:'0';
    ans = '$' + ans;
    if( !isdigit(s[0]) ) ans = "(" + ans + ")";
    cout << ans << endl;
}
