#include<iostream>
using namespace std;

int main() {
	string line;
	cin>>line;
	int neg = 0;
	string dec="00";
	if(line[0]=='-') {
		neg = 1;
		line = line.substr(1,line.size());
	}
	if(line.find('.')!=string::npos) {
		int ind = line.find('.');
		dec=line.substr(ind+1,2);
		line=line.substr(0,ind);
		if(dec.size()==1)
			dec+="0";
	}
	if(neg)
		cout<<"(";
	cout<<"$";
	int mod = line.size()%3;
	for(int i=0;i<line.size();i++) {
		if((i-mod)%3==0 && i!=0)
			cout<<',';
		cout<<line[i];
	}
	cout<<'.'<<dec;
	if(neg)
		cout<<")";
	cout<<endl;
}
