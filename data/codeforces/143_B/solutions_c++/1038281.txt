#include <iostream>
#include <algorithm>
#include <fstream>
using namespace std;

int main(){
string t,s="",d="";
int i,l;
bool ok=false;
cin>>t;
if(t[0]=='-')l=1;
else l=0;
for(i=l;i<t.length();i++){
if(t[i]=='.'){ok=true;goto l;}
if(!ok)s+=t[i];
else   d+=t[i];
l:;
}
string Q="";
int E=0;
if(l==1){
cout<<"(";
}
cout<<"$";

for(i=s.length()-1;i>=0;i--){E++;Q+=s[i];if(i!=0 && E%3==0){Q+=',';}}
reverse(Q.begin(),Q.end());
cout<<Q;
cout<<'.';
if(d.length()>=2){cout<<d[0]<<d[1];}
else if(d.length()==1)cout<<d[0]<<"0";
else if(d=="")cout<<"00";
if(l==1){cout<<")";}


return 0;
}
