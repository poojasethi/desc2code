#include <iostream>
#include <vector>
#include <string>
#include <algorithm>



using namespace std;


int main()
{
  string x;
  cin>>x;
  int d=0;
  while(x[d]!='.' && d<x.size()) d++;
  int i=0;
  if(x[0]=='-')
    {
      cout<<"(";
      i=1;
    }
  cout<<"$";
  for(;i<d;i++)
    {
      cout<<x[i];
      if((d-i)%3==1 && d-i!=1)
	cout<<",";
    }
  cout<<".";
  d++;
  if(d<x.size())
    cout<<x[d];
  else
    cout<<0;

  d++;
  if(d<x.size())
    cout<<x[d];
  else
    cout<<0;
  if(x[0]=='-')
    cout<<")";
  cout<<endl;
  return 0;
}
