#include<cstdio>

int a[200], c, d, e;
bool b = false;

int main() {
	for (int i = 0 ; a[i ? i - 1 : i] != '\n' ; )
		scanf("%c", &a[i++]);
	if (a[0] == '-')
		b = true;
	while (a[c] != '\n' && a[c] != '.')
		c++;
	if (a[c] == '\n')
		a[c] = '.';
	if (b)
		printf("($");
	else
		printf("$");
	for (int i = b ? 1 : 0 ; i < c + 3 ; i++) {
		if (i != c && i != (b ? 1 : 0) && (c - i) % 3 == 0)
			printf(",");
		if (!a[i] || a[i] == '\n')
			printf("0");
		else
			printf("%c", a[i]);
	}
	if (b)
		printf(")\n");
	else
		printf("\n");
	return 0;
}
