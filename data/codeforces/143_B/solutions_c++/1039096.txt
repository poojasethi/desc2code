#include<iostream>
#include<string>

using namespace std;

int main()
{
	string s;
	cin>>s;
	string sp="";
	int x=s.size()-1;
	while(x>=0 && s[x]!='.')
		x--;
	if(x<0)
	{
		x=s.size();
		s+=".00";		
	}	
	else
	{
		while(int(s.size())-x<3)
			s+='0';
		while(int(s.size())-x>3)
			s.resize(s.size()-1);
	}	
	for(int i=int(s.size())-1; i>=x; i--)
		sp=s[i]+sp;
	for(int i=x-1; i>=0; i--)
		if(i!=0 || s[0]!='-')
		{
			sp=s[i]+sp;
			if((i%3)==(x%3) && i>0 && s[i-1]!='-')
				sp=","+sp;
		}
	sp="$"+sp;
	if(s[0]=='-')
	{
		sp='('+sp;
		sp+=')';
	}
	cout<<sp<<endl;
	return 0;
}
