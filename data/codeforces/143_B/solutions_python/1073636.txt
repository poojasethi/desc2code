s = raw_input()
inte = ""
frac = ""
flag = 0
if "." in s:
	flag = 1
	temp = s[0]
	i = 1
	while temp != ".":
		inte += temp
		temp = s[i]
		i += 1
	frac = s[i:]
else:
	inte = s
if "-" in inte:
	inte = inte[1:]
num = len(inte) / 3
l = len(inte) - 3
for i in range(num):
	if l > 0:
		inte = inte[:l] + "," + inte[l:]
	l -= 3
if flag:
	if len(frac) == 1:
		frac += "0"
	else:
		frac = frac[:2]
else:
	frac = "00"
if s[0] != "-":
	print "$" + inte + "." + frac
else:
	print "($" + inte + "." + frac + ")"
