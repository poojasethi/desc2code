import sys

dec,fr = (sys.stdin.readline().strip('\n\r ').split('.') + ['00'])[:2]
fr = (fr+'00')[:2]
if dec[:1]=='-':
  ps='()'
  dec=dec[1:]
else:
  ps=''

if len(dec)==0: dec='0'

groups=[]
while dec:
  groups += [ dec[-3:] ]
  dec = dec[:-3]

groups.reverse()

print( ps[:1] + '$' + ','.join(groups) + '.' + fr + ps[-1:])
