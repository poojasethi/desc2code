a=raw_input()
try:
    i=a.index('.')
    a=a[:i+3]
    while len(a) < i+3:
        a=a+'0'
except ValueError:
    a=a+'.00'
a=list(a[::-1])
x=a.index('.')+4
while x < len(a):
    a.insert(x,',')
    x+=4
if a[-1] == '-' and a[-2]==',':
    del a[-2]
a=''.join(a[::-1])
if a[0]=='-':
    print '($'+a[1:]+')'
else:
    print '$'+a
