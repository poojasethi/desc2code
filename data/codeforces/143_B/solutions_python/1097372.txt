from sys import stdin, stdout

inp = stdin.read().strip() + ".000"
inp = inp.split(".");

a, b =  inp[0], inp[1] + "00";

minus = (a[0] == '-')
if minus:
    a = a[1:]

x = []
i = 0;
t = ""

for c in a[::-1]:
    t += c
    i += 1
    if (i == 3):
        x += [t[::-1]]
        t = ""
        i = 0

if (t!=""):
    x += [t[::-1]]

r = '$' + ','.join(x[::-1]) + '.' + b[:2]
if (minus):
    r = '(' + r + ')'

print r
