s = raw_input()
if s.find('.')!=-1:
    integer, fraction = s.split('.')
else:
    integer, fraction = s, '00'

if integer[0] == '-':
    bracket = True
    integer = integer[1:]
else:
    bracket = False

il = []
index = len(integer)
while index > 0:
    il.append(integer[max(0, index-3):index])
    index -= 3

if len(fraction) < 2:
    fraction = fraction + '0' * (2 - len(fraction))

il.reverse()
num = '$'+','.join(il)+'.'+fraction[:2]
if bracket:
    print '('+num+')'
else:
    print num
