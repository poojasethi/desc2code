# r102 B-div2
inp = raw_input()
ans = []
if inp[0] == '-': 
    ans.append('(')
    inp = inp[1:]
ans.append('$')

if '.' not in inp:
    inp += '.00'
inp += '00'

fp = inp[inp.index('.'):][:3]

ip = inp[:inp.index('.')]
triples = [ip[i:i+3] for i in range(len(ip)%3, len(ip), 3)]
if len(ip)%3 != 0: triples.insert(0, ip[0:len(ip)%3])

ans.extend(','.join(triples))
ans.extend(fp)
if ans[0] == '(': ans.extend(')')
print ''.join(ans)
