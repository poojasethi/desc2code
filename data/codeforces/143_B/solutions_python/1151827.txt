neg_flag = False
line = raw_input().strip()
if '.' in line:
    line = line.split('.')
else:
    line = [line, '0']
line[1] += '00'
dec = (line[1])[:2]
num = list(line[0])
if(num[0] == '-'):
    neg_flag = True
    num.pop(0)
i = len(num)
while i-3>0:
    i -= 3
    num.insert(i,',')
line[0] = '$' + ''.join(num)
line[1] = dec
x = '.'.join(line)
if neg_flag:
    print '('+x+')'
else:
    print x
