from __future__ import division
import itertools as it
from sys import stdin

def solve(s):
    neg = s[0] == '-'
    if neg: s = s[1:]
    if '.' in s:
        s, f = s.split('.')
        f = f.ljust(2, '0')[:2]
    else:
        f = '00'

    s = s[::-1]
    s = ','.join(s[i:i+3] for i in xrange(0, len(s), 3))
    s = s[::-1]
    s = '$%s.%s' % (s, f)
    if neg: s = '(%s)' % s
    return s

def main():
    for s in stdin:
        print solve(s.strip())

if __name__ == '__main__':
    main()
