s = raw_input()
p = s.find(".")
ans = ""
if p == -1:
    ans = ".00"
    p = len(s)-1
else:
    s += "00"
    ans = s[p:p+3]
    p -= 1
cnt = 0
while p >= 0 and s[p] != "-":
    if cnt > 0 and cnt % 3 == 0:
        ans = "," + ans
    ans = s[p] + ans
    p -= 1
    cnt += 1
if s[0] == "-":
    ans = "($" + ans + ")"
else:
    ans = "$" + ans
print ans
