n = raw_input()
if('.' not in n):
    n+=".00"
else:
    n+="00"
n = n.split('.')
frac = n[1][:2]
brac = False
if(n[0][0]=='-'):
    num = n[0][1:]
    brac = True
else:
    num = n[0]
num = num[-1::-1]
abc = []
for i in xrange(0,len(num),3):
    abc += [num[i:i+3][-1::-1]]
abc=abc[-1::-1]
ans=""
if(brac):
    ans="($"
else:
    ans="$"
for i in abc[:len(abc)-1]:
    ans+=i
    ans+=','
ans+=abc[-1]
ans+='.'
ans+=frac
if(brac):
    ans+=')'
print ans
