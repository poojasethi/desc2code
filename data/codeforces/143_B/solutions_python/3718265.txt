#!/usr/bin/env python
from sys import stdin
from re import sub

revfil = lambda k: k[1::-1].zfill(2)[::-1][:2]

def splitthousand(num):
    return sub(r'(\d{3})(?=\d)', r'\1,', num[::-1])[::-1]
    
def p_last(num):
    return revfil(num[1]) if len(num) == 2 else "00"
    

num = stdin.readline().strip().split(".")
print "%s.%s" % [ ('$'+splitthousand(num[0]), p_last(num)) , ('($'+splitthousand(num[0])[1:], p_last(num)+')') ][ num[0][0] == '-']
