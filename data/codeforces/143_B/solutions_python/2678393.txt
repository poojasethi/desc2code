num = raw_input().strip().split(".")
front = abs(int(num[0]))
neg = (num[0][0] == '-')

if len(num) == 1:
  back = "00"
else:
  if front == 0 and int(num[1]) == 0: neg = False
  back = num[1][:2]
  back += "0"*(2-len(back))

def ff(num):
  digs = []
  i = 0
  for c in reversed(str(num)):
    if i > 0 and i % 3 == 0:
      digs.append(",")
    digs.append(c)
    i += 1
  return "".join(reversed(digs))

if neg:
  print "($%s.%s)" % (ff(front), back)
else:
  print "$%s.%s" % (ff(front), back)
