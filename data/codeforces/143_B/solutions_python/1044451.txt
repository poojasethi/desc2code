s = raw_input()
if s.find('.') != -1:
        s += '00'
        s = s[:s.find('.')+3]
else:
        s += '.00'
minus = 0
if s[0] == '-':
        minus = 1
        s = s[1:]
cur = s.find('.')-3
while cur > 0:
        s = s[:cur]+','+s[cur:]
        cur -= 3
if minus == 1: s = '($'+s+')'
else: s = '$'+s
print s
