inp = raw_input()
ans = []
if inp[0] == '-':
    ans.extend(')')
    inp = inp[1:]
    
L = list(inp)
if '.' not in L:
    L.extend('.00')
L.extend('00')
ind = L.index('.')
frac = L[ind:][:3]
ans.extend(reversed(frac))
inte = L[0:ind]
while inte:
    ans.extend(reversed(inte[-3:]))
    ans.extend(',')
    inte = inte[0:-3]
ans = ans[:-1]
ans.extend('$')
if ans[0] == ')': ans.extend('(')
ans.reverse()
print ''.join(ans)
