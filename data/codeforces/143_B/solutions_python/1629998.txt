from sys import stdin as I

s = I.readline()[:-1]
neg = float(s)<0
if neg:
    s = s[1:]
    
dot = s.find('.')
if dot == -1:
    s += '.00'
else:
    s += '00'
    s = s[:dot+3]
    
n = s[:-3]
s = s[-3:]
while len(n) >= 3:
    part = n[-3:]
    s = part + s
    n = n[:-3]
    if n:
        s = ',' + (3-len(part))*'0' + s

if n:
    s = n + s
    
if s[0] == '.':
    s = '0' + s
s = '$' + s
if neg:
    s = '(' + s + ')'
    
print s
