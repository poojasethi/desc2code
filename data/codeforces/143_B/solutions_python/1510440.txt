s = raw_input()
minus = 0
if s[0] == '-':
    minus = 1
    s = s[1:]
d = s.find('.')
if d == -1:
    a = s
    b = '00'
else:
    a, b = s.split('.')
    if len(b) > 2:
        b = b[0:2]
    else:
        for i in range(len(b), 2):
            b += '0'
a = a[::-1]
c = ""
for i in range(len(a)-1):
    c += a[i]
    if i % 3 == 2:
        c += ','
c += a[-1]
c = c[::-1]

res = '$' + c + '.' + b

if minus:
    print '(' + res + ')'
else:
    print res
