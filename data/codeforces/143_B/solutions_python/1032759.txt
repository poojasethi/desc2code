import sys
s = sys.stdin.readline().strip()
neg = s[0] == '-'
if neg:
    s = s[1:]
if "." in s:
    s, d = s.split(".")
    d = d[:2]
    while len(d) != 2:
        d += "0"
else:
    d = "00"
for i in range(3,len(s),3):
    i += (i//3)-1
    s = s[:len(s)-i]+","+s[len(s)-i:]
s = "$"+s+"."+d
if neg:
    s = "("+s+")"
print s
