data = raw_input()

a, _, b = data.partition('.')

if b == '':
    b = '00'
else:
    b += '00'
b = b[:2]

if a[0] == '-':
    negative = True
    a = a[1:]
else:
    negative = False

i = 1
a = list(a)
a.reverse()
a1 = []
for it in a:
    a1.append(it)
    if not i % 3:
        a1.append(',')
    i += 1
a1.reverse()
a = ''.join(a1)
if a[0] == ',':
    a = a[1:]

if negative:
    print '($%s.%s)' %(a, b)
else:
    print '$%s.%s' % (a, b)
