s = raw_input()
inte = ""
frac = ""
j = 0
i = ""
while j != len(s) and i != ".":
	i = s[j]
	inte += i
	j += 1
frac = s[j:]
if len(frac) == 1:
	frac += "0"
elif len(frac) == 0:
	frac = "00"
else:
	frac = frac[:2]
if "." in inte:
	inte = inte[:len(inte) - 1]
if "-" in inte:
	inte = inte[1:]
j = len(inte) / 3
i = len(inte)
for k in range(j):
	i -= 3
	if i > 0:
		inte = inte[:i] + "," + inte[i:]
inte += "." + frac
if s[0] == "-":
	print "($" + inte + ")"
else:
	print "$" + inte
