def main():
	s = raw_input()
	dot = s.find('.')
	if -1 == dot: dot = len(s)
	ans = ['$']
	for i in xrange(0 if s[0] != '-' else 1, dot):
		ans.append(s[i])
		if 1 == (dot - i) % 3 and dot != i + 1: ans.append(',')
	if dot == len(s): ans.append(".00")
	elif len(s) - dot < 3: ans.append(s[dot:] + '0' * (3 - len(s) + dot))
	else: ans.append(s[dot:dot + 3])
	if s[0] != '-': print "".join(ans)
	else: print "(" + "".join(ans) + ")"
 
if __name__ == "__main__": main()
