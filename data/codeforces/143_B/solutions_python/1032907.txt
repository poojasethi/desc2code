import re
s=raw_input().rstrip('\r')
if s.find(".")==-1:
    s=s+"."
s=re.sub(r"\.(.*)$",lambda x: "."+(x.group(1)+"0"*(2-len(x.group(1))) if len(x.group(1))<2 else x.group(1)[:2]),s)
h=0
if s[0]=='-':
    h=1
    s=s[1:]
ind=s.find(".")
while ind%3!=0 and ind>=3:
    s="0"+s
    ind+=1
s=re.sub(r"(\d{3})(?!\.)",lambda x: x.group(1)+",",s)
s=re.sub(r"^0+(?=[^0.])","",s)
s="$"+s
if h:
    s="("+s+")"
print s
