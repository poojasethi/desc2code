s=raw_input()
a,b='',''
if s.find('.')!=-1: a,b=s.split('.')
else:a=s
neg=False
if a[0]=='-':neg=True;a=a[1:]
n=len(a)
j=n%3
x=a[:j]
y=','.join([a[i:i+3] for i in xrange(j,n,3)])
if x and y: x+=','+y
elif y: x=y
z=''
if len(b) >=2:z=b[:2]
else:z=['00',b+'0',b[:2]][len(b)]
r='$'+x+'.'+z
if neg:r='('+r+')'
print r
