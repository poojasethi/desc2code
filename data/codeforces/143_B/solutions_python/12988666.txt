n = raw_input()
temp = n.split('.')

integer = temp[0]
minus = False
if integer[0] == "-":
    minus = True
    integer = integer[1:]

fraction = "00"
if len(temp) == 2:
    fraction = temp[1][0:2]
    if len(fraction) == 1:
        fraction =fraction + "0"

p = []
r_int = integer[::-1]
for i in xrange(0,len(r_int),3):
    p.append(r_int[i:i+3])

integer = (",".join(p))[::-1]

if minus:
    print "($%s.%s)"%(integer,fraction)
else:
    print "$%s.%s"%(integer,fraction)
