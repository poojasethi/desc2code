import re

s = raw_input()
sgn = 1
if s[0] == '-':
	  sgn = -1
	  s = s[1:]
integer, decimal = s, '00'
if '.' in s:
	integer, decimal = s.split('.')
	decimal = decimal[:2]
	if len(decimal) == 1: decimal += '0'
integer = re.sub('(\d\d\d)', '\\1,', integer[::-1])[::-1]
if integer[0] == ',': integer = integer[1:]
s = '$' + integer + '.' + decimal
if sgn == -1:
	  s = '(' + s + ')'
print s
