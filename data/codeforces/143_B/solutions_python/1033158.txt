s = raw_input()

K = s.split('.')

num = K[0]

ind = True
if num[0] == '-':
    ind = False
    num = num[1:]
i = len(num) - 1
ans = ''
while i >= 3:
    ans = ',' + num[i - 2:i + 1] + ans
    i -= 3
ans = num[:i + 1] + ans

if len(K) == 1:
    ans += '.00'
else:
    num = K[1]
    if len(num) > 2:
        ans += '.' + num[:2]
    else:
        ans += '.' + num + '0' * (2 - len(num))
if not ind:
    ans = '($' + ans + ')'
else:
    ans = '$' + ans
print ans
