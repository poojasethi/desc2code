s=raw_input().split(".")
if s[0][0]=="-":
	s[0]=s[0][1:]
	sign=True
else:
	sign=False
if len(s)==1: s.append("00")
if len(s[1])==1: s[1]+="0"
l=[]
while s[0]!="":
	l.append(s[0][-3:])
	s[0]=s[0][:-3]
l=l[::-1]
s[0]=",".join(l)
s[1]=s[1][:2]
f="(${0})" if sign else "${0}"
print f.format(".".join(s))
