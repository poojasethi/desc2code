x=raw_input()
if '.' in x:
    f=x.index('.')
    st=x[f+1:]
    if len(st)==1:
        back='.'+x[f+1]+str(0)
    else:
        back='.'+x[f+1:f+3]
else:
    back=".00"
    f=len(x)

flag=0
if x[0]=='-':
    flag=1
    st1=x[1:f]
else:
    flag=0
    st1=x[0:f]
st2=""
st1=st1[::-1]
while(len(st1)>3):
    st2=st2+st1[0:3]+','
    st1=st1[3:]
if st1=="":
    st2=st2[:-1]
else:
    st2=st2+st1
st2=st2[::-1]
if flag==1:
    print "($"+st2+back+')'
else:
