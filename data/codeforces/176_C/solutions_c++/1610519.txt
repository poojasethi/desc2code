#include<cstdio>
int n,m,x1,y1,x2,y2,t;
const int a[]={4,4,4,3,2};
int abs(int a)
{
	if(a<0) return -a;
	return a;
}
int main()
{
	scanf("%d%d%d%d%d%d",&n,&m,&x1,&y1,&x2,&y2);
	n=abs(x1-x2); m=abs(y1-y2);
	if(n>m) t=n,n=m,m=t;
	if(n<=4&&m<=a[n]) printf("First\n");
	else printf("Second\n");
	return 0;
}
