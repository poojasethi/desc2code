#include <iostream>
#include <cmath>
#include <cstdio>

using namespace std;

int n,m,x1,Y1,x2,y2;

int main()
{
    scanf("%d%d%d%d%d%d",&n,&m,&x1,&Y1,&x2,&y2);
    if(abs(x1-x2)>=5 || abs(Y1-y2)>=5 || abs(x1-x2)+abs(Y1-y2)>=7)printf("Second\n");
    else printf("First\n");
    return 0;
}
