#include <iostream>

using namespace std;

long abs(long x)
{
	if (x>0) return x;
	return -x;
}

main()
{
	long n,m,x1,y1,x2,y2;
	cin >> n >> m >> x1 >> y1 >> x2 >> y2;
	long dx,dy;
	dx=abs(x1-x2);dy=abs(y1-y2);
	if (((dx<=3)&&(dy<=3))||((dx<=2)&&(dy<=4))||((dx<=4)&&(dy<=2)))
		cout << "First" << endl;
	else
		cout << "Second" << endl;
}
