#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
int n, m, xa, xb, ya, yb;

int main()
{
	scanf("%d%d%d%d%d%d", &n, &m, &xa, &ya, &xb, &yb);
	int x = xa - xb, y = ya - yb;
	if (x < 0) x = -x;
	if (y < 0) y = -y;
	if (x > y)
	{
		int t = x;
		x = y;
		y = t;
	}
	if (x <= 2 && y >= 5 || x == 3 && y >= 4 || x >= 4)
		puts("Second");
	else puts("First");
	return 0;
}
