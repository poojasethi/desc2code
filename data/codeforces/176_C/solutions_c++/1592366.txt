#include <cstdio>
#include <iostream>
#include <cmath>
using namespace std;

int main()
{
    int n,m,a,b,c,d;
    
    cin>>n>>m>>a>>b>>c>>d;
    
    int dd1=abs(a-c),dd2=abs(b-d);
    if(max(dd1,dd2)>4 || (max(dd1,dd2)==4 && dd1+dd2>6))printf("Second\n");
    else printf("First\n");
    return 0;
}
