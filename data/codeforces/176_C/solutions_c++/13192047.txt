#include <bits/stdc++.h>
using namespace std;

int main()
{
    int n, m, x, y, a, b;
    scanf("%d %d %d %d %d %d", &n, &m, &x, &y, &a, &b);
    n = (int)abs(x - a);
    m = (int)abs(y - b);
    if(n < m) swap(n, m);
    if(n > 4 || (n == 4 && (m == 3 || m == 4))) puts("Second");
    else puts("First");
    return 0;
}
