#include <bits/stdc++.h>
using namespace std;

int main()
{
    int x1,x2,y1,y2;
    scanf("%*d %*d %d %d %d %d",&x1,&y1,&x2,&y2);
    int x = abs(x1-x2); int y = abs(y1-y2);
    int mx = max(x,y); int mn = min(x,y);
    if(mx<4 and mn < 4)puts("First");
    else if(mn<3 and mx < 5)puts("First");
    else puts("Second");
    return 0;
}
