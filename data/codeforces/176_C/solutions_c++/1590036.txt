//176C Playing with Superglue
#include <cstdio>
#include <algorithm>
using namespace std;

int N, M, X1, Y1, X2, Y2;

int main ()
{
  scanf("%d %d %d %d %d %d", &N, &M, &X1, &Y1, &X2, &Y2);
  int DX = abs(X2 - X1);
  int DY = abs(Y2 - Y1);
  if ((max(DX, DY) >= 5) || ((min(DX, DY) >= 3) && (max(DX, DY) > 3)))
    printf("Second\n");
  else
    printf("First\n");
  return 0;
}
