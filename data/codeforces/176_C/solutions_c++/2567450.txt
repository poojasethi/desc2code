#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

int m, n, x1, y1, x2, y2;

int main() {
	scanf("%d%d%d%d%d%d", &m, &n, &x1, &y1, &x2, &y2);
	int dx = abs(x1-x2), dy = abs(y1-y2);
	if ((dx <= 3 && dy <= 3) || (dx == 4 && dy <= 2) || (dx <= 2 && dy == 4)) printf("First\n");
	else printf("Second\n");
	return 0;
}
