#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int main(void) {
	int n, m, ax, ay, bx, by;
	scanf("%d%d%d%d%d%d", &n, &m, &ax, &ay, &bx, &by);
	int dx=abs(ax-bx), dy=abs(ay-by);
	if(dx>dy) swap(dx, dy);
	if(dy<=4 && dx<4 && !(dx==3 && dy==4)) { puts("First"); return 0; }
	else puts("Second");
	return 0;
}
