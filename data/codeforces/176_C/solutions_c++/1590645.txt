#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int main() {
	int x1, y1, x2, y2;
	scanf("%*d%*d%d%d%d%d", &x1, &y1, &x2, &y2);
	int dx = abs(x1 - x2);
	int dy = abs(y1 - y2);
	if (dx > dy) {
		swap(dx, dy);
	}
	puts((dx < 3 && dy <= 4) || (dx == 3 && dy <= 3) ? "First" : "Second");
	return 0;
}
