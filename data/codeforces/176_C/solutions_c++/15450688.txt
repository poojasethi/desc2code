#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int m, n, x1, x2, y1, y2;

int main() {
	scanf("%d%d%d%d%d%d",&n,&m,&x1,&y1,&x2,&y2);
	n = abs(x1-x2)+1;
	m = abs(y1-y2)+1;
	if(n > m) swap(n, m);
	puts((m<5 || (n <= 3 && m == 5))?"First":"Second");
	return 0;
}
