#include <cstdio>
#include <algorithm>
using namespace std;
int main(){
  int n,m,x1,x2,y1,y2;
  scanf("%d %d %d %d %d %d",&n,&m,&x1,&y1,&x2,&y2);
  bool fail=false;
  int d1=abs(x1-x2);
  int d2=abs(y1-y2);
  if(d1>d2) swap(d1,d2);
  if(d1>4) fail=true;
  if(d2>4) fail=true;
  if(d1>2 && d2==4) fail=true;
  if(fail) printf("Second");
  else printf("First");
  return 0;
}
