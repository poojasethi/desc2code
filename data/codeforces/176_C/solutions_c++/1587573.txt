#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,m,x1,y1,x2,y2;
const int a[]={0,5,5,5,4,3};
int main()
{
	scanf("%d%d%d%d%d%d",&n,&m,&x1,&y1,&x2,&y2);
	n=abs(x1-x2)+1; m=abs(y1-y2)+1;
	if (n>m) {int t=n;n=m;m=t;}
	if (n<=5 && m<=a[n]) printf("First\n"); else printf("Second\n");
	return 0;
}
