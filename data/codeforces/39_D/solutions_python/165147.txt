#! /usr/bin/python

from string import *

def doMain ( ) :
    t1 = raw_input()
    t2 = raw_input()
    tt1 = t1.split()
    tt2 = t2.split()
    res = 0 
    for x in range(0,3):
        a = int(tt1[x])
        b = int(tt2[x])
        if a != b :
            res += 1 
    if res <3 :
        print "YES"
    else :
        print "NO"
      
if __name__=="__main__":
    doMain()
