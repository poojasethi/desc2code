
N=100050
edge=[[] for _ in xrange(N)]

def dfs(u,p):
    ans=0
    mx=0
    for v in edge[u]:
        if v[0]==p:
            continue
        tmp=dfs(v[0],u)
        mx=max(mx,tmp[1]-tmp[0]+v[1])
        ans+=tmp[1]+2*v[1]
    return (ans-mx,ans)

t=1
for _ in xrange(t):
    n=input()
    for i in xrange(n):
        edge[i+1]=[]
    for _ in xrange(n-1):
        u,v,d=map(int,raw_input().split())
        edge[u].append((v,d))
        edge[v].append((u,d))
    print dfs(1,1)[0]
