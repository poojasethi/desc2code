import sys

sys.setrecursionlimit(100000)

def dfs(v):
    global path
    for p in s[v]:
        if path[p[0]] == -1:
            path[p[0]] = path[v] + p[1]
            dfs(p[0])
            
n = int(raw_input())
s = [[] for i in range(n)]
sum = 0
for i in range(n-1):
    a,b,c = map(int,raw_input().split())
    sum += 2*c
    s[a-1]+=[(b-1,c)]
    s[b-1]+=[(a-1,c)]
path = [-1]*n
path[0]=0
dfs(0)
print(sum - max(path))
