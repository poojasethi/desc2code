s, n = 0, int(raw_input())
p, r = [[] for i in range(n + 1)], [0] * (n + 1)
for i in range(n - 1):
    x, y, d = map(int, raw_input().split())
    p[x].append((y, d))
    p[y].append((x, d))
    s += d
t = [x for x in range(1, n + 1) if len(p[x]) == 1]
while t:
    y = t.pop()
    if y == 1: continue
    x, d = p[y][0]
    r[x] = max(r[x], d + r[y])
    p[x].remove((y, d))
    if len(p[x]) == 1: t.append(x)
print 2 * s - r[1]
