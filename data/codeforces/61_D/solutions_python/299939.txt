n = int(raw_input())
g = [ [] for i in xrange(n+1) ]

def calc(u, pre):
    res = 0
    for [v, w] in g[u]:
        if v != pre:
            res = max(res, calc(v, u)+w)
    return res

def main():
    ans = 0
    for i in xrange(n-1):
        [u, v, w] = map(int, raw_input().split())
        g[u].append([v, w]);
        g[v].append([u, w]);
        ans += 2*w;
    print ans-calc(1, 0)

if __name__ == '__main__':
    main()
