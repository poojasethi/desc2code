n = input ()
g = [[] for _ in xrange (n)]
s = 0
for _ in xrange (n - 1):
    u, v, w = map (int, raw_input ().split ())
    u -= 1
    v -= 1
    g[u].append ([v, w])
    g[v].append ([u, w])
    s += w

def dfs (u, p, W):
    res = W    
    for v, w in g[u]:
        if v != p:
            res = max (res, dfs (v, u, W + w))
    return res

print 2 * s - dfs (0, -1, 0)
