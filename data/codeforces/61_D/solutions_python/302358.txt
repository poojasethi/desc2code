g,s,n = {0: []}, 0, int(raw_input())
for i in range(n-1):
    x,y,w = map(int, raw_input().split())
    g.setdefault(x-1, []).append((y-1, w))
    g.setdefault(y-1, []).append((x-1, w))
    s += w

d,q = {0:0}, [0]
while q != []:
    u = q.pop(0)
    for v,r in g[u]:
        if v not in d:
            d[v] = d[u] + r
            q.append(v)

print (2*s - max(d.values()))
