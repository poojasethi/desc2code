from collections import deque
n = input(); s = 0; g = [[] for i in range(n+1)]
for i in range(n-1):
    x,y,w = map(int, raw_input().split())
    g[x].append((y, w))
    g[y].append((x, w))
    s += w
d = [-1,0]+[-1]*(n-1); q = deque([1])
while q :
    u = q.popleft()
    for v,r in g[u]:
        if d[v]==-1:
            d[v] = d[u] + r
            q.append(v)
print (2*s - max(d))
