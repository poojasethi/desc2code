#include <cstdio>
#include <cstring>
#include <vector>
using namespace std;
#define MAXN 100010
#define max(a,b) (a>b?a:b)
#define MP make_pair
typedef long long ll;

ll path[MAXN];
vector<pair<int,int> >mp[MAXN];
void dfs(int x,int fa)
{
	path[x]=0;
	for(int i=0;i<mp[x].size();i++)
	{
		int v=mp[x][i].first;
		if(v!=fa)
		{
			dfs(v,x);
			path[x]=max(path[x],path[v]+mp[x][i].second);
		}
	}
}

int main()
{
	int n;
	while(scanf("%d",&n)!=EOF)
	{
		int a,b,c;
		ll sum=0;
		for(int i=1;i<n;i++)
		{
			scanf("%d%d%d",&a,&b,&c);
			mp[a].push_back(MP(b,c));
			mp[b].push_back(MP(a,c));
			sum+=2*c;
		}
		dfs(1,0);
		printf("%I64d\n",sum-path[1]);
	}
}
