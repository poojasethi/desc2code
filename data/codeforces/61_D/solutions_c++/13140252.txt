#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<pair<int,int> >v[100000];
long long s=0,larg=0;

int dfs(int i,int n)
{
	long long l=0;
	if(v[i].size()==1)
	larg=max(larg,s);
	for(int j=0;j<v[i].size();j++)
	{
		if(v[i][j].first==n)
		continue;
		s+=v[i][j].second;
		long long d=dfs(v[i][j].first,i);
		s-=v[i][j].second;
		d+=v[i][j].second;
		l+=d;
	}
	return l;
}
int main()
{
	int n;
	cin>>n;
	int x,y,w;
	for(int i=0;i<n-1;i++)
	{
		cin>>x>>y>>w;
		x--;
		y--;
		v[x].push_back({y,w});
		v[y].push_back({x,w});
	}
	long long u=dfs(0,-1);
	cout<<(2*u)-larg;
}
