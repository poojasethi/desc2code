# include <bits/stdc++.h>

using namespace std;

typedef long long int ll;

const int maxn = 1e5 + 10;
ll n , ans;
ll dis[maxn];
bool mark[maxn];
vector <pair<int,ll> > adj[maxn];

void dfs(int v){
	mark[v] = true;
	for(int i=0 ; i<adj[v].size() ; i++){
		int u = adj[v][i].first;
		if(!mark[u]){
			dis[u] = dis[v] + adj[v][i].second;
			dfs(u);
		}
	}
}

int main(){
	cin >> n;
	for(int i=1 ; i<n ; i++){
		int fi, se, w;
		cin >> fi >> se >> w;
		adj[fi].push_back(make_pair(se , w));
		adj[se].push_back(make_pair(fi , w));
		ans += w;
	}

	dfs(1);
	ll maxx = 0;
	for(int i=1 ; i<=n ; i++)
		maxx = max(maxx , dis[i]);
	
	cout << 2*ans - maxx << endl;

	return 0;
}
