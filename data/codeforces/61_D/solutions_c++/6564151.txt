#include<bits/stdc++.h>
using namespace std;

vector < pair<int,int>  >adj[100001];
long long dfs(int chiled,int parent)
{ long long ret=0;
for(int i=0;i<adj[chiled].size();i++)
{
    pair<int,int>v=adj[chiled][i];
    if(v.first!=parent)
        ret=max(ret,dfs(v.first,chiled)+v.second);
}
    return ret;
}
int main()
{
    long long result=0;
    int x; cin>>x;
    int begining,end,cost;
    for(int i=0;i<x-1;i++)
    {
        cin>>begining>>end>>cost;
        begining--,end--;
        adj[begining].push_back(make_pair(end,cost));
        adj[end].push_back(make_pair(begining,cost));
        result+=2*cost;

    }
cout<<result-dfs(0,0)<<endl;
    return 0;
}
