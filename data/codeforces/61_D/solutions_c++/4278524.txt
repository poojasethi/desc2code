#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
vector <pair<int,long long> > g[100001];
int a,b,i,c,n;
long long d,sum;
long long dfs(int a,int b,long long d)
{
long long ret=d;
	for (int i=0;i<g[a].size();i++)
	if (g[a][i].first!=b)
	{
		ret=max(ret,dfs(g[a][i].first,a,d+g[a][i].second));
	}
	return ret;	
}
int main()
{
cin>>n;
for (i=1;i<n;i++)
{
cin>>a>>b>>c;
g[a-1].push_back(make_pair(b-1,c));
g[b-1].push_back(make_pair(a-1,c));
sum+=c;
}
cout<<2*sum-dfs(0,-1,d); 
return 0;
}
