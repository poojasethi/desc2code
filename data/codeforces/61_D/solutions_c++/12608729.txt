#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
vector<pair<ll,ll> > v[100001];
ll reach[100001],dp[100001],n;
ll dfs(ll node,ll pa)
{
	ll sum=0,i;
	for(i=0;i<v[node].size();i++)
	{
		if(v[node][i].first!=pa)
		{
			sum=max(sum,v[node][i].second + dfs(v[node][i].first,node));
		}
	}
	return sum;
}
int main()
{
	ios_base::sync_with_stdio(false);cin.tie(0);
	cin>>n;
	ll i,a,b,c,pa,temp,t,j,maxi,pos,sum=0;
	memset(reach,0,sizeof(reach));
	for(i=0;i<n-1;i++)
	{
		cin>>a>>b>>c;
		v[a].push_back(make_pair(b,c));
		v[b].push_back(make_pair(a,c));
		sum+=2*c;
	}
	
	cout<<sum-dfs(1,-1);
	
}
