#include <cstdio>
#include <iostream>
#include <vector>

#define fi first
#define se second
#define pb push_back

using namespace std;

typedef pair<int,int> ii;
typedef long long int Lint;

const int MAXN=100100;

int N;

Lint res,mx;

vector<ii> way[MAXN];

void dfs( int nd , Lint cur , int dad ){
	
	mx=max(mx,cur);
	
	for( vector<ii>::iterator it=way[nd].begin() ; it!=way[nd].end() ; it++ )
		if( it->fi!=dad ){
			res+=it->se*2;
			dfs(it->fi,cur+it->se,nd);
		}
}

int main(){
	
	scanf(" %d",&N);
	
	for( int a,b,c,i=1 ; i<N ; i++ ){
		scanf(" %d %d %d",&a,&b,&c);
		way[a].pb(ii(b,c));
		way[b].pb(ii(a,c));
	}
	
	dfs(1,0LL,0);
	printf("%I64d\n",res-mx);
	return 0;
}
