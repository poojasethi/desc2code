#include <iostream>
#include <cstring>
#include <cstdio>
#include <vector>
using namespace std;

typedef long long LL;
vector<pair<int, LL> > a[100005];

LL ans = 0;

LL go(int x, int from = -1){
	LL r = 0;
	for(int i = 0; i < a[x].size(); i++){
		if(a[x][i].first == from) continue;
		r = max(r, go(a[x][i].first, x) + a[x][i].second);
	}
	return r;
}

int main(void){
	//freopen("input.txt", "r", stdin);
	int n;
	scanf("%d", &n);
	for(int i = 1; i < n; i++){
		int x, y;
		LL w;
		scanf("%d%d%I64d", &x, &y, &w);
		a[x].push_back(make_pair(y,w));
		a[y].push_back(make_pair(x,w));
		ans += w+w;
	}

	ans -= go(1);
	printf("%I64d", ans);
	return 0;
}
