#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;

const int maxn=100000+10;
vector< pair<int,int> > e[maxn];
long long d[maxn];
long long res;
int n;

void dfs(int u,int f)
{
	for (int v=0;v<e[u].size();v++)
	if (e[u][v].first!=f)
	{
		dfs(e[u][v].first,u);
		d[u]=max(d[u],d[e[u][v].first]+e[u][v].second);
	}
}

int main()
{
	scanf("%d",&n);
	for (int i=1;i<n;i++)
	{
		int u,v,w;
		scanf("%d%d%d",&u,&v,&w);
		e[u].push_back(make_pair(v,w));
		e[v].push_back(make_pair(u,w));
		res+=(long long)w*2;
	}
	dfs(1,0);
	printf("%I64d\n",res-d[1]);
}
