#include <iostream>
#include <vector>

using namespace std;
#define lol long long
#define mp make_pair
const int MAXN=1e5+10;
lol dis[MAXN];
vector <pair <int,int> > a[MAXN];

void dfs(int v,int p,int d)
{
  dis[v]=d;
  for(int i=0;i<a[v].size();i++)
    {
      int u=a[v][i].first,w=a[v][i].second;
      if(u!=p)
	dfs(u,v,d+w);
    }
}

int main()
{
  lol n,ans=0;
  cin>>n;
  for(int i=1;i<n;i++)
    {
      int v,u,w;
      cin>>v>>u>>w;
      a[v].push_back(mp(u,w));
      a[u].push_back(mp(v,w));
      ans+=w;
    }
  ans*=2;
  dfs(1,0,0);
  lol Max=0;
  for(int i=1;i<=n;i++)
    Max=max(Max,dis[i]);
  cout<<ans-Max<<endl;
  return 0;
}
