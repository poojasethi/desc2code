#include <bits/stdc++.h>
using namespace std;
#define int long long
vector<int> v[100005];
vector<int> cost[100005];

int dfs(int a , int b){
	int ans=0;
	for(int i=0 ; i<v[a].size() ; i++){
		if(v[a][i] == b) continue;
		ans=max(ans , cost[a][i]+dfs(v[a][i] , a));
	}
	return ans;
}

main() {
	int n , a , b , c, total=0;
	cin>>n;
	for(int i=0 ; i<n-1 ; i++){
		cin>>a>>b>>c;
		v[a].push_back(b);
		v[b].push_back(a);
		cost[a].push_back(c);
		cost[b].push_back(c);
		total+=c;
	}
	
	cout<<2*total-dfs(1 , 0);

}
