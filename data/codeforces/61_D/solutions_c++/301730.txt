#include <algorithm>
#include <cstdio>
#include <vector>
using namespace std;
typedef pair<int, int> PII;
vector<PII> adj[111111];

int dfs(int i, int p)
{
	int maxx = 0;
	for (int k = 0; k < adj[i].size(); ++k)
	{
		int j = adj[i][k].first;
		int w = adj[i][k].second;
		if (j != p)
		{
			maxx = max(maxx, dfs(j, i) + w);
		}
	}
	return maxx;
}

int main()
{
	int n;
	scanf("%d", &n);
	int sum = 0;
	while (--n)
	{
		int a, b, c;
		scanf("%d%d%d", &a, &b, &c);
		sum += c;
		--a;
		--b;
		adj[a].push_back(PII(b, c));
		adj[b].push_back(PII(a, c));
	}
	printf("%I64d", (long long)sum * 2 - dfs(0, -1));
	return 0;
}
