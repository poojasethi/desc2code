#include <cstdio>
#include <vector>
#include <iostream>
using namespace std;

vector<pair<int, int> > A[101000];
int N;

typedef long long ll;
bool used[101000];

ll go(int i) {
	ll ret = 0, tmp;
	used[i] = 1;
	for (int j=0; j<A[i].size(); ++j) {
		int k = A[i][j].first;
		int w = A[i][j].second;
		if (!used[k]) {
			tmp = w + go(k);
			if (tmp > ret) ret = tmp;
		}
	}
	return ret;
}

int main() {
	scanf("%d", &N);
	ll ret = 0;
	for (int i=1; i<N; ++i) {
		int x, y, w;
		scanf("%d%d%d", &x, &y, &w);
		--x, --y;
		A[x].push_back(make_pair(y, w));
		A[y].push_back(make_pair(x, w));
		ret += w + w;
	}
	ret -= go(0);
	cout << ret << endl;
	return 0;

}
