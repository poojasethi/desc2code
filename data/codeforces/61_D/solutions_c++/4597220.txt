#include<stdio.h>
#include<iostream>
#include<string.h>
#include<algorithm>
#include<vector>
#include<math.h>

#define LL long long

using namespace std;

vector< pair<int,LL> > g [100001];

int f[100001];

LL dfs(LL n)
{
	int i;
	f[n]=1;
	int l = g[n].size();
	LL mx=0;
	for(i=0;i<l;i++)
	{
		int y = g[n][i].first;
		if(f[y]==0)
			mx= max(mx,dfs(y)+ g[n][i].second);
	}
	return mx;
}

int main()
{
	int n;
	cin>>n;

	int i,j,k;
	LL ans = 0;
	int x,y,z;
	for(i=0;i<n-1;i++)
	{
		cin>>x>>y>>z;
		g[x].push_back(make_pair(y,z));
		g[y].push_back(make_pair(x,z));
		ans+=z+z;
	}
	cout<<ans - dfs(1)<<'\n';
	return 0;
}
