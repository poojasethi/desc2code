#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
vector<pair<ll,ll> > v[100001];
ll n;
 int visited[100001];
ll dfs(ll node)
{
    visited[node]=1;
	ll sum=0,i;
	for(i=0;i<v[node].size();i++)
	{
		if(!visited[v[node][i].first])
		{
		    //cout<<node<<" "<<v[node][i].first<<" "<<v[node][i].second<<endl;
			sum=max(sum,v[node][i].second + dfs(v[node][i].first));
			//cout<<sum<<endl;
			//visited[v[node][i].first]=1;
		}
	}
	return sum;
}
int main()
{
	cin>>n;
	ll i,a,b,c,pa,temp,t,j,maxi,pos,sum=0;
	for(i=0;i<n-1;i++)
	{
		cin>>a>>b>>c;
		v[a].push_back({b,c});
		v[b].push_back({a,c});
		sum+=2*c;
	}
    visited[1]=1;
	cout<<sum-dfs(1);
}
