#include<bits/stdc++.h>
using namespace std;
const int mxmx=101000;
int n,dp[mxmx],p[mxmx];
vector <int> v[mxmx][2];
void hesab(int I){
	int maX=0;
	for(int i=0;i<v[I][1].size();i++){
		if(p[I]!=v[I][0][i]){
			p[v[I][0][i]]=I;
			hesab(v[I][0][i]);
			maX=max(dp[v[I][0][i]]+v[I][1][i],maX);
		}
	}
	dp[I]=maX;
}
int main(){
	ios::sync_with_stdio(0);cin.tie(0);
	cin>>n;
	long long int ANS=0;
	for(int i=1,a,b,w;i<n;i++){
		cin>>a>>b>>w;
		ANS+=2*w;
		v[a][0].push_back(b);
		v[b][0].push_back(a);
		v[a][1].push_back(w);
		v[b][1].push_back(w);
	}
	hesab(1);
	cout<<ANS-dp[1];
}
