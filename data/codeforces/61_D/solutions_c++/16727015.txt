#include<bits/stdc++.h>
#define ll long long int
using namespace std;

ll ans=0,dp[100005]={0};
vector<pair<int,int> >*adj;

void dfs(int u,int p)
{
	for(int i=0;i<adj[u].size();i++)
		if(adj[u][i].first!=p)
		{
			dfs(adj[u][i].first,u);
			dp[u]=max(dp[u],dp[adj[u][i].first]+adj[u][i].second);
		}
}

int main()
{
	int n,u,v,w;
	scanf("%d",&n);
	adj=new vector<pair<int,int> >[n+1];
	for(int i=1;i<n;i++)
	{
		scanf("%d %d %d",&u,&v,&w);
		adj[u].push_back(make_pair(v,w));
		adj[v].push_back(make_pair(u,w));
		ans+=(2*w);
	}
	dfs(1,0);
	ans-=dp[1];
	printf("%I64d\n",ans);
	return 0;
}
