#include <bits/stdc++.h>
#define MAXN 100000
#define mp make_pair
#define ri(n) scanf("%d",&n)
#define riii(a,b,c) scanf("%d %d %d",&a,&b,&c)
#define FOR(i,l,r) for(int i=l;i<r;i++)
using namespace std;

int N, A[MAXN];
long long ans=0;
vector<pair<int,int> > T[MAXN];

void dfs(int v) {
	A[v]=0;
	FOR(i,0,T[v].size()) 
		if(A[T[v][i].first]==-1) {
			dfs(T[v][i].first);
			A[v]=max(A[v],A[T[v][i].first]+T[v][i].second);
	}
}

int main() {
	ri(N); memset(A,-1,sizeof(A));
	for(int v,u,w,i=0;i<N-1;i++) {
		riii(v,u,w); --v; --u;
		T[v].push_back(mp(u,w));
		T[u].push_back(mp(v,w));
		ans+=w;
	}
	dfs(0);
	printf("%I64d\n",2*ans-A[0]);
}
