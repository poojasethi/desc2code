#include <cstdio>
#include <algorithm>
#include <vector>
#include <iostream>
using namespace std;
#define FOR(_i,_n) for(int _i=0;_i<_n;++_i)
int n;
typedef pair<int,int> PII;
typedef vector<PII> VP;
typedef long long LL;
VP g[100100];
LL dfs(int a=1,int f=0){
  LL r=0;
  for(VP::const_iterator it=g[a].begin();it!=g[a].end();++it)if(it->first!=f)r=max(r,it->second+dfs(it->first,a));
  return r;
}
int main(void) {
  scanf("%d",&n);
  LL s=0;
  for(int i=1,x,y,z;i<n&&3==scanf("%d %d %d",&x,&y,&z);++i)g[x].push_back(PII(y,z)),g[y].push_back(PII(x,z)),s+=z;
  cout<<(s*2-dfs())<<endl;
  return 0;
}
