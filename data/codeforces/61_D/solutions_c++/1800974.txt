#include <cstdio>
#include <algorithm>
#include <vector>

using namespace std;

vector<int> a[100001],g[100001];
int f[100001];
long long m,d[100001];

void dfs(int x)
{
  m=max(m,d[x]);
  for(int z=0;z<(int)a[x].size();++z)
  {
    int y=a[x][z];
    if(y==f[x])
      continue;
    d[y]=d[x]+g[x][z];
    f[y]=x;
    dfs(y);
  }
}

int main(void)
{
  int n;
  long long ans=0;
  scanf("%d",&n);
  for(int i=1;i<n;++i)
  {
    int j,k,l;
    scanf("%d %d %d",&j,&k,&l);
    a[j].push_back(k);
    g[j].push_back(l);
    a[k].push_back(j);
    g[k].push_back(l);
    ans+=l;
  }
  dfs(1);
  ans=ans*2-m;
  printf("%I64d\n",ans);
  return 0;
}
