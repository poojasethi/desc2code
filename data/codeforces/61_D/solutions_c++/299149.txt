#include <iostream>
#include <fstream>
#include <vector>

#define int long long

using namespace std;

const int MAXV = 100 * 1000 + 10;

vector<pair<int, int> > e[MAXV];
bool m[MAXV];
bool p[MAXV];
int l[MAXV];

int dfs(int a)
{
	int t = 0;
	m[a] = true;
	for(int i = 0; i < e[a].size(); i++)
		if(!m[e[a][i].first])
		{
			l[e[a][i].first] = dfs(e[a][i].first);
			t = max(t, l[e[a][i].first] + e[a][i].second);
		}
	return t;
}

main()
{
	int n, s = 0;
	cin >> n;
	for(int i = 1, a, b, c; i < n; i++)
		cin >> a >> b >> c, 
		e[a - 1].push_back(pair<int, int>(b - 1, c)), e[b - 1].push_back(pair<int, int>(a - 1, c)), 
		s += 2 * c;

//	cerr << dfs(0) << endl;
	cout << s - dfs(0) << endl;
	return 0;
}
