#include<iostream>
#include<vector>

using namespace std;
#define N 100006
#define pb push_back 
#define ll long long
vector<ll> a[N];
vector<ll> w[N];
ll visited[N];


ll dfs(ll i,ll f){
visited[i] = 1;
ll p = f;
    for(ll j = 0; j<a[i].size(); j++)
        if(!visited[a[i][j]])
        p = max(p,dfs(a[i][j],f+w[i][j]));  
    return p;
}



int main(){
ll  n, x, y, z,ans = 0;
cin>>n;
for(ll i = 0; i<n-1; i++){
    cin>>x>>y>>z;
    a[x].pb(y),a[y].pb(x),w[x].pb(z),w[y].pb(z);
    ans+=2*z;
    
    
}
ans-=dfs(1,0);
cout<<ans;

}
