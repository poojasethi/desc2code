#include <cstdio>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;

typedef long long int ll;

vector<pair<int,int> >graf[100005];

int n;

ll wyn=0LL;

ll m=0LL;

void dfs(int w,int o,ll g)
{
	m=max(m,g);
	for(int i=0;i<graf[w].size();i++)
	{
		if(graf[w][i].first==o) continue;
		dfs(graf[w][i].first,w,g+graf[w][i].second);
	}
}

int main()
{
	scanf("%d",&n);
	for(int i=1;i<n;i++)
	{
		int a,b,c;
		scanf("%d%d%d",&a,&b,&c);
		graf[a].push_back(make_pair(b,c));
		graf[b].push_back(make_pair(a,c));
		wyn+=c*2;
	}
	dfs(1,-1,0LL);
	printf("%I64d\n",wyn-m);
	return 0;
}
