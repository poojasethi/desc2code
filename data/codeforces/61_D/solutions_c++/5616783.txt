#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
#define MP make_pair
#define PB push_back
#define REP(i,n) for(int i=0;i<n;i++)

vector<pair<int,int> >adj[100009];
ll jj;
bool vis[100009];
ll dfs(int cn,ll sum)
{
    if(sum>jj)jj=sum;
    vis[cn]=1;
    REP(i,adj[cn].size())
    {
        if(!vis[adj[cn][i].first])dfs(adj[cn][i].first,adj[cn][i].second+sum);
    }
}
int main()
{
    int u,v,w;
    ll sum=0;
    int N;
    cin >> N;
    REP(i,N-1)
    {
        cin >> u >> v >> w;
        adj[u].PB(make_pair(v,w));
        adj[v].PB(make_pair(u,w));
        sum+=(w);
    }
    sum<<=1;
    dfs(1,0);
    cout << sum-jj << endl;
    return 0;
}
