#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
#define pb push_back
#define ll long long
ll maxi=0;
vector <int> g[100005],w[100005];
void dfs(int v,int pr,ll sum)
{
  if (g[v].size()==1 && v!=0)
  {
    maxi=max(maxi,sum);
    return;
  }
  for(int i=0;i<g[v].size();i++)
    if (g[v][i]!=pr)dfs(g[v][i],v,sum+w[v][i]);
}
int main()
{
  ll sum=0;
  int n,x,y,wes;
  cin>>n;
  for(int i=0;i<n-1;i++)
  {
    scanf("%d%d%d",&x,&y,&wes);
    w[x-1].pb(wes);
    w[y-1].pb(wes);
    g[x-1].pb(y-1);
    g[y-1].pb(x-1);
    sum+=wes;
  }
  dfs(0,-1,0);
  cout<<2*sum-maxi;
  return 0;
}
