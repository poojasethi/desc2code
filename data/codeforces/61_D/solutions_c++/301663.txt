#include <iostream>
#include <vector>
using namespace std;
const int MN = 100*1000+10;
typedef long long ll;
#define int ll
vector<int> e[MN];
vector<int> cost[MN];
int N;

int dfs(int n, int f) {
	int r=0;
	for(size_t i=0; i<e[n].size(); ++i) {
		int t = e[n][i];
		if (t==f) continue;
		int c = cost[n][i];
		r = max(r, c + dfs(t, n));
	}
	return r;
}

main()
{
	cin>>N;
	int s=0;
	for(int i=0; i<N-1; ++i) {
		int a,b,c;
		cin>>a>>b>>c;
		e[a].push_back(b);
		cost[a].push_back(c);
		e[b].push_back(a);
		cost[b].push_back(c);
		s+=c;
	}
	cout<<2*s-dfs(1,-1)<<'\n';
}
