#include<bits/stdc++.h>
#define mp make_pair
#define f first 
#define s second
using namespace std;
const int maxn=1e5+5;
long long ans,cnt=1,n,mark[maxn],dp[maxn];
vector<pair <long long,long long> >v[maxn];
void dfs(int a)
{
	mark[a]=1;
	for(int i=0;i<v[a].size();i++)
	{	
		int child=v[a][i].s;
		if(!mark[child])
		{
			dp[child]=dp[a]+v[a][i].f;
			dfs(child);
		}
	}
}	
int main()
{
	int a,b,w;cin>>n;
	for(int i=1;i<n;i++)
	{
		cin>>a>>b>>w;
		ans+=w;
		v[a].push_back(mp(w,b));
		v[b].push_back(mp(w,a));
	}
	dfs(1);
	long long Max=0;
	for(int i=1;i<=n;i++)
		Max=max(Max,dp[i]);
	cout<<2*ans-Max;
}
