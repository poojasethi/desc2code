#include<bits/stdc++.h>
using namespace std;
list<pair<int,int> > li[1000000];
int dfs(int start,int p)
 {
 	int ret=0;
 	 list<pair<int,int> >:: iterator it;
 	 for(it=li[start].begin();it!=li[start].end();it++)
 	  {
 	  	  if(it->first!=p)
 	  	   {
 	  	   	
 	  	   	 ret=max(ret,dfs(it->first,start)+it->second);
			  }
	   }
	   return ret;
 }
int main()
 {
 	 int n;
 	  cin>>n;
 	  long long int maxi=0;
 	  for(int i=0;i<n-1;i++)
 	   {
 	   	 int a,b,c;
 	   	 scanf("%d %d %d",&a,&b,&c);
 	   	 li[a].push_back({b,c});
 	   	 li[b].push_back({a,c});
 	   	 maxi+=2*c;
		}
 	cout<<maxi-dfs(1,0);
 }
