#include <bits/stdc++.h>
using namespace std;

typedef long long ll;

ll n,k;

ll ft[13][100013];
void update(ll i,ll j,ll v)
{
	while(j<=n+1)
	{
		ft[i][j]+=v;
		j+=j&(-j);
		//cout<<j<<endl;
	}
}
ll query(ll i,ll j)
{
//	cout<<i<<"  "<<j<<endl;
	if(j==0)
	return 0;
	return ft[i][j]+query(i,(j-(j&(-j))));	
}
int main() {
	scanf("%lld%lld",&n,&k);
	k++;
	update(0,1,1);
	ll l,r,i,j;
	for(i=0;i<n;i++)
	{
		scanf("%lld",&l);
		l++;
		for(j=k;j>0;j--)
		{
//			cout<<j<<endl;
			r=query(j-1,l);
			update(j,l,r);	
		}	
	}	
	cout<<query(k,n+1);
	return 0;
}
