#include <bits/stdc++.h>
using namespace std;
int const maxn = 5e5+100;
long long f[12][maxn];

void add(long long * f, int x, long long y)
  {
    for(int i = x; i < maxn; i |= i + 1) f[i] += y;
  }

long long get(long long * f, int x)
  {
    long long ret = 0;
    for(int i = x; i >= 0; i = (i & (i + 1)) - 1) ret += f[i];
    return ret;
  }
int main()
  {
    int n , k ;cin >> n >> k;
    add(f[0], 0, 1);
    for(int i = 0; i < n; i++)
	{
        int x ; cin >> x;
        for(int j = k + 1; j > 0; j--)
		{
            long long got = get(f[j - 1], x - 1);
            add(f[j], x, got);
        }
    }
    cout << get(f[k + 1], maxn - 1) << endl;
  }
