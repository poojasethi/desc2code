#include <bits/stdc++.h>

using namespace std;
#define ll long long int
#define N 100000

ll tree[12][100010];

void st(ll k,ll idx,ll val)
{
    while(idx<=N)
    {
        tree[k][idx]+=val;
        idx += (idx & -idx);
    }
}

ll get(ll k,ll x)
{
    ll ans=0;
    while(x>0)
    {
        ans+=tree[k][x];
        x -= (x & -x);
    }
    return ans;
}

int main()
{
    ll n,i,j,k,a;
    cin>>n>>k;
    k++;
    for(i=0;i<n;i++)
    {
        cin>>a;
        st(1,a,1);
        if(a>1)
        {
            for(j=2;j<=k;j++)
                st(j,a,get(j-1,a-1));
        }
    }
    cout<<get(k,N)<<endl;
    return 0;
}
