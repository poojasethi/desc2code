#include <bits/stdc++.h>
using namespace std;
#define ll long long
ll f[100010][15],a[100010],ans;
ll T[100010][15],n,k;
void update(int x, int i, ll u){
	while(x<=n){
		T[x][i] += u;
		x += x&-x;
	}
}
ll get_sum(int x, int i){
	ll ret = 0;
	while(x){
		ret += T[x][i];
		x -= x&-x;
	}
	return ret;
}
int main(){
	//freopen("in.in","r",stdin);
	cin >> n >> k; for(int i = 1; i <= n; i++) cin >> a[i];
	for(int i = 1; i <= n; i++){
		for(int j = k+1; j >= 2; j--){
			f[i][j] += get_sum(a[i],j-1);
			update(a[i],j,f[i][j]);
		}
		f[i][1] = 1;
		update(a[i],1,f[i][1]);
	}
	for(int i = 1; i <= n; i++){
		ans += f[i][k+1];
	}
	cout << ans;
}
