#include<bits/stdc++.h>

using namespace std;

#define sgmn int v, int tl, int tr
#define tm  (tl + tr >> 1)
#define sol v*2, tl, tm
#define sag v*2+1, tm+1, tr
#define N 100005

long long F[11][N],n,i,j,k,x,t,asd;

void up(int x){
	for(; x <= n ; x += x & -x)
		F[j][x] += t;
}

void qry(int x){
	for(; x ; x -= x & -x)
		t += F[j][x];
}

int main(){

	scanf("%lld%lld",&n,&k);

	for(i=1;i<=n;i++){
		
		scanf("%lld",&x);

		for(j=0;j<=k;j++){

			if(j) {  t = 0;  j--;  qry(x - 1);  j++;  }

			else t = 1;

			up(x);

		}

		asd += t;

	}

	printf("%lld",asd);

	return 0;
}
