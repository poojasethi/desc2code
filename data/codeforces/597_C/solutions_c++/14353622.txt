#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
inline int lowbit(int x){return x&(-x);}
int n,k,a;
ll s[100010][12];
inline ll getSum(int x,int y){
    ll res=0;
    for(;x>0;x-=lowbit(x))res+=s[x][y];
    return res;
}
void change(int x,int y,ll d){
    for(;x<=100005;x+=lowbit(x))s[x][y]+=d;
}
int main(){
    cin>>n>>k;
    for(int i=1;i<=n;i++){
        cin>>a;
        change(a,1,1);
        if(a==1)continue;
        for(int j=2;j<=k+1;j++){
            ll tt=getSum(a-1,j-1);
            change(a,j,tt);
        }
    }
    cout<<getSum(100005,k+1);
    return 0;
}
