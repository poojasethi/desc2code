#include <bits/stdc++.h>

using namespace std;

typedef long long ll;

#define pii pair<int,int>
#define pll pair<ll,ll>
#define PB push_back
#define MP make_pair
#define X first
#define Y second

void update(ll *T, int i, int n, ll val){
	if(i>n);
	else{
		T[i]+=val;
		update(T,i+(i&-i),n,val);
	}
}

ll query(ll *T, int i){
	if(i==0)
		return T[i];
	else
		return T[i]+query(T,i-(i&-i));
}

int main(){
	int n,m,k,i,j,x;
	ll ans=0;

	scanf("%d%d", &n, &k);

	ll **T;
	T=(ll**)calloc(k+1,sizeof(ll*));

	for(i=0;i<k+1;i++)
		T[i]=(ll*)calloc(n+1,sizeof(ll));

	T[0][0]=1;

	m=n;

	while(m--){
		scanf("%d", &x);

		for(i=0;i<k;i++)
			update(T[i+1],x,n,query(T[i],x-1));

		ans+=query(T[k],x-1);
	}

	cout << ans << '\n';

	return 0;
}
