#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define maxn 100005
#define int long long
using namespace std;
int d[maxn],pre[maxn],dp[maxn],n,m,tree[maxn],i,j,k,pre2[maxn];
void update(int x,int val){
	 for(;x<=n;x+=x&-x)
	 tree[x]+=val;
}
int query(int x){
	int res=0;
	for(;x;x-=x&-x)
	res+=tree[x];
	return res;
}
main(){
	scanf("%d %d",&n,&k);
	for(i=1;i<=n;i++)
	scanf("%d",&d[i]);
	for(i=1;i<=n;i++)
	pre[i]=1;
	for(i=1;i<=k;i++){
		for(j=1;j<=n;j++){
		pre2[d[j]]=query(d[j]-1);
		update(d[j],pre[d[j]]);
		}
		for(j=1;j<=n;j++)
		pre[j]=pre2[j];
		memset(tree,0,sizeof tree);
	}
	int res=0;
	for(i=1;i<=n;i++)
	res+=pre[i];
	printf("%lld\n",res);
}
