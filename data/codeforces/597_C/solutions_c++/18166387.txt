#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const int N = 100010;
int n, k, a[N];
ll x[12][N], tmp;

void update(int idx, ll v, int i) {
	while(idx <= n){
		x[i][idx] += v;
		idx += idx & -idx;
	}
}

ll get(int idx, int i) {
	ll s = 0;
	while(idx > 0){
		s += x[i][idx];
		idx -= idx & -idx;
	}
	return s;
}

int main(int argc, char **argv) {
#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
#endif
	scanf("%d%d", &n, &k); ++k;
	for(int i=1 ; i<=n ; ++i)
		scanf("%d", a+i);
	for(int i=1 ; i<=n ; ++i) {
		update(a[i], 1, 1);
		for(int j=2 ; j<=k ; ++j){
			tmp = get(a[i]-1, j-1);
			update(a[i], tmp, j);
		}
	}
	cout << get(n, k) << endl;
	return 0;
}
