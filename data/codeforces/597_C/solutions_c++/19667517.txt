#include<cstdio>
#define lb(x) (x&-x)
using namespace std;
const int N = 101010;
int n,i,k,j,a[N];
long long ans,f[N][15],c[N];
void cc(int x,long long w)
{
	while (x<=n)
	{
		c[x]+=w;
		x+=lb(x);
	}
}
long long sum(int x)
{
	long long ans=0;
	while (x)
	{
		ans+=c[x];
		x-=lb(x);
	}
	return ans;
}
int main()
{
	scanf("%d%d",&n,&k);
	k++;
	for (i=1;i<=n;i++)
	scanf("%d",&a[i]);
	for (i=1;i<=n;i++)
	{
		f[i][1]=1;
		if (k==1) ans++;
	}
	for (j=2;j<=k;j++)
	{
	for (i=1;i<=n;i++)
	c[i]=0;
	for (i=1;i<=n;i++)
	{
		f[i][j]=sum(a[i]-1);
		if (j==k) ans+=f[i][j];
		cc(a[i],f[i][j-1]);
	}
	}
	printf("%I64d\n",ans);
}
