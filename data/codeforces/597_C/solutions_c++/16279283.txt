#include <iostream>
#include <cstdio>
using namespace std;

typedef unsigned long long ll;
const int maxn = 100050;
ll C[12][maxn];
int n, k;

ll sum(int len, int i)
{
   ll ret = 0;
   while(i > 0) {
      ret += C[len][i];
      i -= i & -i;
   }
   return ret;
}

void add(int len, int i, ll x)
{
   while(i <= n) {
      C[len][i] += x;
      i += i & -i;
   }
   return;
}

int main(void)
{
   scanf("%d%d", &n, &k);
   for(int i = 0; i < n; i++) {
      int x;
      scanf("%d", &x);
      add(1, x, 1);
      for(int len = 2; len <= k + 1; len++) {
         ll temp = sum(len - 1, x - 1);
         add(len, x, temp);
      }
   }
   cout << sum(k + 1, n) << endl;
   return 0;
}
