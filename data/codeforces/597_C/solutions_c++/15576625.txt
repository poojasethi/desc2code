#include <cstdio>

int const N = 100100;
int const K = 15;

long long tree[K][N + K];

void add(int id, int pos, long long value) {
    //pos++;
	while (pos < N) {
		tree[id][pos] += value;
		pos += pos & -pos;
	}
}

long long get(int id, int to) {
	long long ret = 0;
	while(to > 0) {
		ret += tree[id][to];
		to -= to & -to;
	}
	return ret;
}

int main() {
	int n, k;
	scanf("%d%d", &n, &k);
	++k;
	add(0, 1, 1);
	int LOOP = n;
	while(LOOP--) {
		int x;
		scanf("%d", &x);
		for (int t = k; t > 0; --t)
			add(t, x+1, get(t - 1, x));
	}
	printf("%I64d\n", get(k, N));
}
