#include <bits/stdc++.h>
using namespace std;
int ar[100000],n,i,k,j;
long long dp[12][100000],temp[100001];
void add(int x,long long v){
	int t;
	while(x<=n){
		temp[x]+=v;
		t = x&(-x);
		x+=t;
	}
}
long long sum(int x){
	long long ret=0;int tt= n;
	while(tt>0){
		ret+= temp[tt];
		tt= tt&(tt-1);
	}
	while(x>0){
		ret -= temp[x];
		x = x&(x-1);
	}
	return ret;
}
int main(){
	cin>>n>>k;for(i=0;i<n;++i)cin>>ar[i],dp[0][i]=1;
	for(int x=1;x<=k;++x){
		memset(temp,0,sizeof(temp));
		for(i=n-1;i>=0;--i){
			dp[x][i] = sum(ar[i]);
			add(ar[i],dp[x-1][i]);
		}
	}
	long long ret=0;
	for(i=0;i<n;++i)
		ret+=dp[k][i];
	cout<<ret<<endl;
}
