#include <bits/stdc++.h>
using namespace std;
const int maxn = 1e5+10;
int n,a[maxn],k;
long long f[15][maxn];
long long get(int x,int y){
    if (x==0) return 1;
    long long ans=0;
    while (y){
        ans+=f[x][y];
        y-=y&-y;
    }
    return ans;
}
long long update(int x,int y,long long v){
    while (y<=n){
        f[x][y]+=v;
        y+=y&-y;
    }
}
int main(){
    scanf("%d%d",&n,&k);k++;
    for (int i=1; i<=n; i++) scanf("%d",&a[i]);
    for (int i=1; i<=n; i++){
        for (int j=k; j>0; j--){
            update(j,a[i],get(j-1,a[i]-1));
        }
    }
    printf("%I64d\n",get(k,n));
}
