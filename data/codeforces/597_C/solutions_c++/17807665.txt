#include <bits/stdc++.h>
typedef long long ll;
using namespace std;

ll f[int(1e5+5)][12];
int k,n;

void updateBIT(int i, int j, ll val){
    while (i<=n+1){
        f[i][j]+=val;
        i+=i&(-i);
    }
}

ll getBIT(int i, int j){
    ll sum=0;
    while (i>0){
        sum+=f[i][j];
        i-=i&(-i);
    }
    return sum;
}

int main(){
//    memset(f,0,sizeof(f));
    ios::sync_with_stdio(false);
    cin>>n>>k;
    updateBIT(1,0,1);
    for (int i=0; i<n; i++){
        int inp;
        cin>>inp;
        for (int j=k+1; j>0; j--){
            updateBIT(inp+1, j, getBIT(inp+1, j-1));
        }
    }
    cout<<getBIT(n+1,k+1)<<endl;
    return 0;
}
