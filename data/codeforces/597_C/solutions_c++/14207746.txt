#include<bits/stdc++.h>

using namespace std;

#define ll long long
#define MAXX 100000

int n,k;
int a[MAXX+5];
ll bit[15][MAXX+5];


void update(int turn, int pos, ll num){
	for(;pos<=MAXX;pos+=pos&-pos) bit[turn][pos]+=num;
}

ll query(int turn, int pos){
	ll ret = 0;
	for(;pos;pos-=pos&-pos) ret += bit[turn][pos];
	return ret;
}

int main(){
	//ll ans = 0;
	scanf("%d %d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]);
	}
	
	for(int i=0;i<n;i++){
		update(0,a[i],1LL);
		for(int j=1;j<=k;j++){
			ll tmp = query(j-1,a[i]-1);
		//	cout<<i<<' '<<tmp<<endl;
			update(j,a[i],tmp);
		}
		//ans+=query(k,a[i]);
	}
//	ans = query(k,MAXX)
	cout<<query(k,MAXX)<<endl;
}
