#include <iostream>
using namespace std;
typedef long long ll;

const int MAXN = 100005;
const int MAXK = 15;
int N, K;
int A[MAXN];
ll BIT[MAXK][MAXN];

void add(int k, int idx, ll val)
{
	for (; idx < MAXN; idx += idx&-idx)
		BIT[k][idx] += val;
}

ll get(int k, int idx)
{
	ll ret = 0;
	for (; idx > 0; idx -= idx&-idx)
		ret += BIT[k][idx];
	return ret;
}

int main()
{
	ios::sync_with_stdio(0);

	cin >> N >> K;
	K++;
	for (int i = 1; i <= N; i++)
	{
		cin >> A[i];
		A[i]++;
	}

	add(0, 1, 1);
	for (int i = 1; i <= N; i++)
		for (int k = 1; k <= K; k++)
		{
			ll cnt = get(k - 1, A[i] - 1);
			add(k, A[i], cnt);
		}

	cout << get(K, N + 1) << "\n";

	return 0;
}
