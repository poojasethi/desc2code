#include <bits/stdc++.h>
using namespace std;
int const N = 123456;
void update(long long *f,int x,long long y)
{int i;
   for(i=x;i<N;i|=i+1)
    f[i]+=y;
    return;
}
long long query(long long *f,int x) 
    {int i;
        long long sum=0;
        for(i=x;i>=0;i=(i&(i+1))-1)
        sum+=f[i];
        return sum;
    }
    long long bit[12][N];
int main() {
    int n,k;
    scanf("%d %d",&n,&k);

    update(bit[0],0,1);
    for(int i=1;i<=n;i++)
    {
        int x;
        scanf("%d",&x);
        for(int j=k+1;j>=1;j--)
        {
            long long s=query(bit[j-1],x-1);
            update(bit[j],x,s);
        }
    }
    printf("%lld",query(bit[k+1],N-1));
    return 0;
}
