#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
const int m = 100005;
ll f[12][ m ];
int n, k;
void update( int i, int j, ll delta )
{
	while( j <= n + 1 ){ f[ i ][ j ] += delta; j += j & -j; }
}
ll get( int i, int j )
{
	return j > 0 ? f[ i ][ j ] + get( i, j - ( j & ( -j ) ) ) : 0;
}

int main()
{
	cin >> n;
	cin >> k;
	// update( 0, 1, 1 );
	for(int i = 0;i < n;i++)
	{
		int a;
		cin >> a;
		for( int j = k + 1; j > 0; j-- )
		{
			update( j, a, j == 1 ? 1 : get( j - 1, a - 1 ) );
		}
	}
	cout << get( k + 1, n + 1 );
	return 0;
}
