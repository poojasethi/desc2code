#include<bits/stdc++.h>
using namespace std;

int arr[100005];

long long bit[12][100005];

void update(int ind,int st,long long val)
{
    int x=st;
	while(x<=100005)
	{
		bit[ind][x]+=val;
		x+=(x&(-x));
	}
}
long long query(int ind,int st)
{
	long long ans=0;
	int x=st-1;
	while(x)
	{
		ans+=bit[ind][x];
		x-=(x&(-x));
	}
	return ans;
}

int main()
{
	int n,k;
	scanf("%d %d",&n,&k);
	for(int i=0;i<n;i++)
	{
		scanf("%d",&arr[i]);
	}
	long long temp;
	for(int i=0;i<n;i++)
	{
		update(1,arr[i],1);
		for(int j=1;j<=k;j++)
		{
			temp=query(j,arr[i]);
			update(j+1,arr[i],temp);
		}
	}
	printf("%lld",query(k+1,100001));
}
