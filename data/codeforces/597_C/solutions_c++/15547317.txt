// indxing 1 to n
void update(int pos,long long int add,int n,int k)
{
	int ind=pos;
	while(ind<=n)
	{
		bit[ind][k]+=add;
			ind=ind+(ind&(-ind));
	}
}

long long int query(int pos,int k)
{
	int ind=pos ;
	long long int sum=0;
	while(ind!=0)
	{
		sum+=bit[ind][k];
		ind=ind-(ind&(-ind));
	}
	return sum;
}
