#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
vector<vector<long long> > fenw;
long long get(int r, int k){
	long long res = 0;
	for (; r >= 0; r = (r & (r + 1)) - 1)
		res += fenw[k][r];
	return res;
}
void add(int r, long long x, int k){
	for (; r < fenw[k].size(); r |= r + 1)
		fenw[k][r] += x;
}
int main()
{
	long long n, k, ans = 0;
	cin >> n >> k;
	k++;
	fenw.resize(k + 1, vector<long long>(n + 1));
	add(0, 1, 0);
	for (int i = 0; i < n; i++)
	{
		int buffer;
		cin >> buffer;
		for (int j = 1; j <= k; j++)
		{
			long long x = get(buffer - 1, j - 1);
			add(buffer, x, j);
			if (j == k) ans += x;
		}
	}
	cout << ans;
	return 0;
}
