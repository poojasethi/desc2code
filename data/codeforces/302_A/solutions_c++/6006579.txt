#include <stdio.h>

int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	int arr[2]={0};
	for(int i=0;i<n;++i) {
		int x;
		scanf("%d",&x);
		++arr[x<0];
	}
	while(m--){
		int a,b;
		scanf("%d%d",&a,&b);
		if((b-a)%2==0){
			puts("0");
		}else if((b-a+1)>arr[0]*2||(b-a+1)>arr[1]*2){
			puts("0");
		}else puts("1");
	}
}
