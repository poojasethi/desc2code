#include<stdio.h>
long i,j,u,c,n,m;
int main(){
 scanf("%ld %ld",&n,&m);
 for(i=0;i<n;i++){
  scanf("%ld",&j);
  if(j<0)
    c++;
  else
    u++;
 }
 if(c>u)
    c=u;
 c=c*2;
 for(i=0;i<m;i++){
  scanf("%ld %ld",&u,&j);
  u=j-u+1;
  if((u>c)||(u%2==1))
    printf("0\n");
  else
    printf("1\n");
 }
}
