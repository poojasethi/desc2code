#include <iostream>
using namespace std;

long n,m,a[1000000],l,r,u,d,i;
main()
{
	
	cin>>n>>m;
	 
	 for(i=1;i<=n;i++)
	   {cin>>a[i];
	   if (a[i]>0) d++; else u++;
	   }
	   
	   for (i=1;i<=m;i++)
	    { cin>>l>>r;
	     l=r-l+1;
	     if (l%2==0 && l/2<=u && l/2<=d) cout<<1<<endl; else cout<<0<<endl;
	    }
}
