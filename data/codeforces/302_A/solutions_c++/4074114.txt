#include<stdio.h>
int main()
{
	int n,m,c[200000],a,b,i,x,y;
	scanf("%d%d",&n,&m);
	for(i=0,a=0,b=0;i<n;i++)
	{
		scanf("%d",&c[i]);
		if(c[i]==1)
			a++;
		else
			b++;
	}
	while(m--)
	{
		scanf("%d%d",&x,&y);
		i=y-x+1;
		if(i%2==0 && i/2<=a && i/2<=b)
			printf("1\n");
		else
			printf("0\n");
	}
	return 0;
}
