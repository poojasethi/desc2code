#include <iostream>
using namespace std;
const int N=200005;
int s,n,m;
int main(){
    cin>>n>>m;
    for(int i=1;i<=n;i++){
        int t;cin>>t;if(t<0) s++;
    }
    if(s>n-s) s=n-s;
    while(m--){
        int l,r;
        cin>>l>>r;
        r-=l-1;
        cout<<((r%2==0&&r/2<=s)?1:0)<<endl;
    }
}
