#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
int n,m,p,q;cin>>n>>m;
int neg=0,pos=0;
for(int i=0;i<n;i++)
{scanf("%d",&p);if(p>0)pos++;else neg++;}
while(m--)
{
scanf("%d%d",&p,&q);
int t=q-p;t++;
if(t%2)printf("0\n");
else if(neg>=t/2&&pos>=t/2)printf("1\n");
else printf("0\n");
}
}
