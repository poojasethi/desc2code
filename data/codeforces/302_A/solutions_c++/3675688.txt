#include<iostream>
using namespace std;
int a[1000000][2];
int main()
{
	int n,m,i,x,l,r,o=0,mi=0;
	cin>>n>>m;
	for(i=1;i<=n;i++){
		cin>>x;
		if(x==1)o++;
		else mi++;
	}
	for(i=0;i<m;i++){
		cin>>l>>r;
		if((r-l+1)%2==0 && o>=(r-l+1)/2 && mi>=(r-l+1)/2)cout<<1<<endl;
		else cout<<0<<endl;
	}
	return 0;
}
