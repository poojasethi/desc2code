#include<stdio.h>
int main()
{
   int n,m,i,k,l1 = 0,l2 = 0;
   scanf("%i %i",&n,&m);
   for(i=0;i<n;i++){
      scanf("%i",&k);
      if(k==1)l1 ++;
      else l2 ++;
   }
   for(i=0;i<m;i++){
      scanf("%i %i",&n,&k);
      printf("%i\n",(k-n)&1 && l1<<1 >= k-n && l2<<1 >= k-n);
   }
   return 0;
}
