#include <iostream>
using namespace std;
int n,x,m,l,r,a,b;
main(){
    cin>>n>>m;
    while(n--){
        cin>>x;
        x==1?a++:b++;
    }
    while(m--){
        cin>>l>>r;
        if((r-l+1) & 1)     cout<<"0\n";
        else if((r-l+1)/2<=a && (r-l+1)/2<=b) cout<<"1\n";
        else cout<<"0\n";
    }
}
