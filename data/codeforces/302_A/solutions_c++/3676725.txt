#include<cstdio>
int main () {
	int n,m,i,a,b,x=0;
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++) { scanf("%d",&a); if(a<0) x++;}
	for(i=0;i<m;i++) {
		scanf("%d%d",&a,&b);
		if((b-a+1)%2==0) {
			if(x>=(b-a+1)/2&&(n-x)>=(b-a+1)/2)
				printf("1\n");
			else printf("0\n");
		}
		else printf("0\n");
	}
	return 0;
}
