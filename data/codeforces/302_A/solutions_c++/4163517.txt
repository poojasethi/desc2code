#include <cstdio>
int a[2];
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++){
		int x;
		scanf("%d",&x);
		a[x==1?0:1]++;
	}
	for (int i=0;i<m;i++){
		int l,r;
		scanf("%d%d",&l,&r);
		if ((r-l+1)%2==0 && (r-l+1)/2 <= a[0] &&(r-l+1)/2 <= a[1] )printf("1\n");else printf("0\n");
	}
	return 0;
}
