#include <cstring>
#include <cstdio>
#include <iostream>

using namespace std;

int main()
{
	int x=0, y=0, n, m, l, r;
	cin>>n>>m;
	for(int i=1; i<=n; i++)
	{
		int a;
		cin>>a;
		a>0?x++:y++;
	}
	for(int i=0; i<m; i++)
	{
		cin>>l>>r;
		if((r-l+1)%2==0 && (r-l+1)/2<=x && (r-l+1)/2<=y) cout<<1<<endl;
		else cout<<0<<endl;
	}
	return 0;
}
