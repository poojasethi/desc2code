#include <iostream>
#include<stdio.h>
using namespace std;
int main(){	
	int n,m,x,y,t=0,s=0,num;
	scanf("%d %d",&n,&m);
	for(int i=0;i<n;i++){
		cin>>num;
		if(num==1){
			s++;
		}
		else{
			t++;
		}
	}
	for(int i=0;i<m;i++){
		scanf("%d %d",&x,&y);
		cout<<((y-x+1)/2<=min(s,t) && ((y-x+1)%2==0))<<endl;
		
	}
	
}
