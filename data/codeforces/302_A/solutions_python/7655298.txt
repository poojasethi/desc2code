n,m=map(int,raw_input().split())
l=[int(x) for x in raw_input().split()]
s=l.count(-1)
b=l.count(1)
for i in xrange(m):
    l,r=map(int,raw_input().split())
    chk=(r+1)-l
    if chk%2:
        print 0
    else:
        if chk/2<=b and chk/2<=s:
            print 1
        else:
            print 0
