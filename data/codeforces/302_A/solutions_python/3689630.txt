n, m = map(int, raw_input().split())
arr = map(int, raw_input().split())
neg = arr.count(-1)
pos = arr.count(1)

for i in xrange(m):
    l, r = map(int, raw_input().split())
    res = r - l + 1
    half = res / 2
    if res % 2 != 0 or half > neg or half > pos:
        print('0')
    else:
        print('1')
