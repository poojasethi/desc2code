#include <iostream>

using namespace std;

int a[300010];
int pos[300010];

int main()
{
    ios::sync_with_stdio(0);
    int n;
    cin >> n;
    for(int i = 0 ; i < n ; ++i)
    {cin >> a[i];--a[i];pos[a[i]] = i;}
    int q;
    cin >> q;
    for(int i = 0 ; i < q ; ++i)
    {
        int t, x, y;
        cin >> t >> x >> y;
        --x, --y;
        if(t == 1)
        {
            int res = 1, P = pos[x];
            for(int j = x+1 ; j <= y ; ++j)
            {
                if(pos[j] < P) ++res;
                P = pos[j];
            }
            cout << res << "\n";
        }
        else
        {swap(pos[a[x]], pos[a[y]]);swap(a[x], a[y]);}
    }
	return 0;
}
