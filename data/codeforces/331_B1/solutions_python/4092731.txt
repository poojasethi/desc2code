#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#define maxn 110
using namespace std;
int i,j,k,n,m,x,y,z,tmp,pre,ans;
int a[maxn],b[maxn];
int main()
{
    scanf("%d",&n);
    for(i=1;i<=n;i++)
     scanf("%d",&a[i]);
    scanf("%d",&m);
    for(i=0;i<m;i++)
    {
        scanf("%d%d%d",&x,&y,&z);
        if (x==2)
        {
            tmp=a[y];a[y]=a[z];a[z]=tmp;
        }
        else
        {
            ans=1;
            for(j=1;j<=n;j++)
             b[a[j]]=j;
             pre=b[y];
            for(j=y+1;j<=z;j++)
             {
                if (b[j]<pre) ans++;
                pre=b[j];
            }
            printf("%d\n",ans);
                
        }
    }   
    return 0;
}
