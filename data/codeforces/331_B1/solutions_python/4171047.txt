#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
    int a[101],b[101],n,q,x,y,p,check,session;
    scanf("%d",&n);
    for(int i=1;i<=n;i++){
        scanf("%d",&a[i]);
        b[a[i]]=i;
    }
    scanf("%d",&q);
    while(q--){
        scanf("%d %d %d",&p,&x,&y);
        if(p==2){
            swap(b[a[x]],b[a[y]]);
            swap(a[x],a[y]);
        }
        else{
                session=1;
                check=b[x];
                for(int i=x+1;i<=y;i++){
                    if(check>b[i])
                        session++;
                    check=b[i];
                }

            printf("%d\n",session);
        }

    }
    return 0;
}
