
#include <cstdio>
#include <cstring>

int d[110];
int p[110];
int n;

void init()
{
    memset(p,-1,sizeof(p));
    for(int i=0;i<n;++i){
	int j = d[i]-1;
	if(p[ j ]  == -1)
	    p[ d[i] ] = d[i];
	else
	    p[ d[i] ] = p[j];
    }

    /*
    for(int i=1;i<=n;++i)
	printf("%d ",d[i-1]);
    puts("");
    for(int i=1;i<=n;++i)
	printf("i=%d -> %d\n",i,p[i]);
	*/
}

int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;++i)scanf("%d",d+i);
    init();


    int q;
    scanf("%d",&q);
    while(q--){
    
	int t,x,y;
	scanf("%d%d%d",&t,&x,&y);
	if(t==1){
	    int ans=0;
	    while( y >= x ){
		ans ++;
		y = p[y]-1;
	    }
	    printf("%d\n",ans);
	}
	if(t==2){
	    x--;y--;
	    d[x]^=d[y];
	    d[y]^=d[x];
	    d[x]^=d[y];
	    init();
	}
    }


    return 0;
}
