#include<iostream>
#include<set>
#include<cstring>
#include<algorithm>
using namespace std;

const int MAX = 100+7;
int a[MAX];
int n, q;
set<int> tn;
bool vis[MAX];
int wh[MAX];

int main() {
	cin >> n;
	for (int i=0; i<n; i++) {
		cin >> a[i];
		wh[a[i]] = i;
	}

	cin >> q;
	for (int i=0; i<q; i++) {
		int pi = 0, x,y;
		cin >> pi >> x >> y;
		if (pi == 2) {
			x--, y--;
			swap(a[x], a[y]);
			swap(wh[a[x]], wh[a[y]]);
		}
		else {
			int ret = 0;
			memset(vis, 0, sizeof vis);
			for (int t=0; t<n; t++) {
				if (!vis[a[t]-1] && a[t] >= x && a[t] <= y)
					ret++;
				if (a[t] >= x && a[t] <= y)
					vis[a[t]] = 1;
			}
			cout << ret << endl;
		}
	}
	return 0;
}
