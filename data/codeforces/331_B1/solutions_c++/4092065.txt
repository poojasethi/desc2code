#include <iostream>
#include <cstdio>

using namespace std;

int a[200],n,q;

int main()
{
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	cin >> q;
	for (int i = 0; i < q; i++)
	{
		int p,x,y;
		scanf("%d%d%d",&p,&x,&y);
		//printf("&&&&\n");
		if (p == 2)
		{
			x--;y--;
			int t = a[x];
			a[x] = a[y];
			a[y] = t;
		}
		else
		{
			int a_ans = 0;
			int k = x;
			while (k <= y)
			{
				int j = 0;
				while (j < n)
					if (a[j++] == k)
						k++;
				a_ans++;
			}
			printf("%d\n",a_ans);
		}
	}
	return 0;
}
