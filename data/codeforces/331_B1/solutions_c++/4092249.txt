#include <iostream>
using namespace std;

long a[200],b[200],n,t;
long x,y;

int main()
{
    cin>>n;
    for (long i=1;i<=n;i++)
        cin>>a[i];

    for (long i=1;i<=n;i++)
        b[a[i]]=i;

    cin>>t;
    long tp;
    for (long j=0;j<t;j++){
    cin>>tp;
    if (tp==1){
    cin>>x>>y;
        long sch=1;
        long tek=b[x];
        for (long k=x+1;k<=y;k++){
            if (b[k]<tek)
                sch++;
                tek=b[k];}
        cout<<sch<<endl;


    }
    if (tp==2){
    cin>>x>>y;
    swap(a[x],a[y]);
    swap(b[a[x]],b[a[y]]);

    }

    }
    return 0;
}
