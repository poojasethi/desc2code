#include<iostream>
using namespace std;
const int Maxn = 105;
int a[Maxn];
bool mark[Maxn];
int n;
int F(int y,int z)
{
	for(int i=1;i<=n;i++)
		mark[i]=false;
	int c=0;
	for(int i=1;i<=n;i++)
	{
		if(mark[i]==false)
		{
			if((a[i]<=z)&&(a[i]>=y))
			{
				c++;
				mark[i]=true;
				int root = i;
				for(int j=i+1;j<=n;j++)
				{
					if(a[j]==(a[root]+1))
					{
						if(a[j]<=z)
						{
							mark[j]=true;
							root = j;
						}
					}
				}
			}
		}
	}
	return c;
}
int main()
{
	int q,x,y,z;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	cin>>q;
	for(int i=1;i<=q;i++)
	{
		cin>>x>>y>>z;
		if(x==1)
		{
			cout<<F(y,z)<<endl;
		}
		else
		{
			swap(a[y],a[z]);
		}
	}
}
