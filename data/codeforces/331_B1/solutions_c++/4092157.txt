#include <cstdio>
#include <map>
#define maxn 105
std::map<int,int>b;
int n,q,a[maxn];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
    {
        int x;
        scanf("%d",a+i);
        b[a[i]]=i;
    }
    scanf("%d",&q);
    for(int i=0;i<q;++i)
    {
        int t,l,r;
        scanf("%d%d%d",&t,&l,&r);
        if(t==1)
        {
            int num=1,tmp=b[l];
            for(int j=l+1;j<=r;++j)
            {
                if(b[j]<tmp)
                    ++num;
                tmp=b[j];
            }
                
            printf("%d\n",num);
        }
        else
        {
            int tmp=a[l];
            a[l]=a[r];
            a[r]=tmp;
            b[a[l]]=l;
            b[a[r]]=r;
        }
    }
    return 0;
}
