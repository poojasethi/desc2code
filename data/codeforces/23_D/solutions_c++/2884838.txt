#include <cstdio>
#include <utility>
#include <vector>
#include <cmath>
#include <algorithm>
#define eps 1e-8
using namespace std;
typedef pair<double,double> pr;
vector<pr> vec;
double x[10],y[10];
int cross(pr a,pr b,pr c)
{
	double re = (a.first-c.first)*(b.second-c.second);
	re -= (a.second-c.second)*(b.first-c.first);
	if(fabs(re)<eps) return 0;
	return re>0?1:-1;
}
bool work()
{
	double x0,y0,d,xb,yb,xc,yc,xa,ya,xd,yd; int i;
	x0 = x[1]*2-x[2]; y0 = y[1]*2-y[2];
	d = x0*(y[1]-y[3])+x[1]*(y[3]-y0)+x[3]*(y0-y[1]);
	d *= 2;
	xb = (x0*x0+y0*y0)*(y[1]-y[3]);
	xb += (x[1]*x[1]+y[1]*y[1])*(y[3]-y0);
	xb += (x[3]*x[3]+y[3]*y[3])*(y0-y[1]);
	xb /= d;
	yb = (x0*x0+y0*y0)*(x[3]-x[1]);
	yb += (x[1]*x[1]+y[1]*y[1])*(x0-x[3]);
	yb += (x[3]*x[3]+y[3]*y[3])*(x[1]-x0);
	yb /= d;
	xc = 2*x[1]-xb; yc = 2*y[1]-yb;
	xa = 2*x[3]-xb; ya = 2*y[3]-yb;
	xd = 2*x[2]-xc; yd = 2*y[2]-yc;
	vec.clear();
	vec.push_back(make_pair(xa,ya));
	vec.push_back(make_pair(xd,yd));
	vec.push_back(make_pair(xc,yc));
	vec.push_back(make_pair(xb,yb));
	for(i=0;i<4;i++)
		if(cross(vec[i],vec[(i+1)%4],vec[(i+2)%4])*cross(vec[i],vec[(i+1)%4],vec[(i+3)%4])<=0)
		return false;
	printf("YES\n");
	for(i=0;i<4;i++)
		printf("%.10f %.10f ",vec[i].first,vec[i].second);
	printf("\n");
	return true;
}
int main()
{
	//freopen("input.txt","r",stdin);
	int T,i;
	scanf("%d",&T);
	while(T--)
	{
		for(i=1;i<=3;i++)
			scanf("%lf%lf",&x[i],&y[i]);
		if((x[1]-x[3])*(y[2]-y[3])==(y[1]-y[3])*(x[2]-x[3]))
			{printf("NO\n\n"); continue;}
		if(work()) continue;
		swap(x[1],x[2]); swap(y[1],y[2]);
		if(work()) continue;
		swap(x[1],x[3]); swap(y[1],y[3]);
		if(work()) continue;
		printf("NO\n\n");
	}
}
