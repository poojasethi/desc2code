#include <cstdio>
#include <cstring>
#include <cstdlib>

using namespace std;

double solC(int xa,int ya,int xb,int yb)
{
  return ((xa - xb) * (xa + xb) + (ya - yb) * (ya + yb)) / 2.0;
}

double det(double x1,double y1,double x2,double y2)
{
  return x1 * y2 - x2 * y1;
}

bool solve(int xa,int ya,int xb,int yb,int xc,int yc)
{
  double C1 = solC(xa,ya,xb,yb);
  double C2 = solC(xa,ya,xc,yc) + (xc - xa) * xa * 2 + (yc - ya) * ya * 2;
  int A1 = xa - xb,B1 = ya - yb;
  int A2 = xc - xa,B2 = yc - ya;
  double x0 = det(C1,C2,B1,B2) / det(A1,A2,B1,B2);
  double y0 = det(C1,C2,A1,A2) / det(B1,B2,A1,A2);
  double x1 = 2 * xa - x0,y1 = 2 * ya - y0;
  double x2 = 2 * xc - x1,y2 = 2 * yc - y1;
  double x3 = 2 * xb - x0,y3 = 2 * yb - y0;
  if ((det(x1 - x0,y1 - y0,x2 - x1,y2 - y1) < 0 && det(x2 - x1,y2 - y1,x3 - x2,y3 - y2) < 0 && det(x3 - x2,y3 - y2,x0 - x3,y0 - y3) < 0) && det(x0 - x3,y0 - y3,x1 - x0,y1 - y0) < 0)
    {
      puts("YES");
      printf("%.9f %.9f %.9f %.9f %.9f %.9f %.9f %.9f\n",x0,y0,x1,y1,x2,y2,x3,y3);
      return true;
    }
  return false;
}

int main()
{
  int t;
  scanf("%d",&t);
  while(t-- > 0)
    {
      int xa,ya,xb,yb,xc,yc;
      scanf("%d%d%d%d%d%d",&xa,&ya,&xb,&yb,&xc,&yc);
      if ((xb - xa) * (yc - yb) - (xc - xb) * (yb - ya) == 0  || !(solve(xa,ya,xb,yb,xc,yc) ||  solve(xb,yb,xa,ya,xc,yc) || solve(xc,yc,xa,ya,xb,yb) || solve(xa,ya,xc,yc,xb,yb) || solve(xb,yb,xc,yc,xa,ya) || solve(xc,yc,xb,yb,xa,ya)))
	{
	  puts("NO");
	  puts("");
	}
    }
  return 0;
}
