#include <cstdio>
#include <cstring>
#include <iostream>
#include <ctime>
#include <cmath>
#include <algorithm>
using namespace std;
const double eps=1e-8;
struct yqh
{
    double x,y;
    void get()
    {
        scanf("%lf%lf",&x,&y);
    }
}a,b,c;
int T;
double chaji(yqh a,yqh b,yqh c)
{
    return (b.x-a.x)*(c.y-a.y)-(b.y-a.y)*(c.x-a.x);
}
yqh cross(yqh a,yqh b,yqh c,yqh d)
{
    double t1=chaji(a,c,b),t2=chaji(a,b,d);
    yqh e; 
    e.x=(c.x*t2+d.x*t1)/(t1+t2);e.y=(c.y*t2+d.y*t1)/(t1+t2); 
    return e;
}
void get_mid(yqh a,yqh b,yqh &c,yqh &d)
{
    c.x=(a.x+b.x)/2;c.y=(a.y+b.y)/2;
    d.x=c.x+b.y-a.y;d.y=c.y+a.x-b.x;
}
bool get_ans(yqh a,yqh b,yqh c)
{
    yqh A,B,C,D,d,e,na,nb,nc,nd;
    get_mid(a,b,na,nb);
    get_mid(b,c,nc,nd);
    d.x=b.x+nd.x-nc.x;
    d.y=b.y+nd.y-nc.y;
    d=cross(b,d,na,nb);
    e=cross(na,nb,nc,nd);
    A.x=2*d.x-e.x;A.y=2*d.y-e.y;
    B.x=2*b.x-A.x;B.y=2*b.y-A.y;
    C.x=2*c.x-B.x;C.y=2*c.y-B.y;
    D.x=2*a.x-A.x;D.y=2*a.y-A.y;
    if(chaji(A,B,C)*chaji(B,C,D)>eps&&chaji(B,C,D)*chaji(C,D,A)>eps&&chaji(C,D,A)*chaji(D,A,B)>eps)
    {
        printf("YES\n");
        printf("%.9lf %.9lf %.9lf %.9lf %.9lf %.9lf %.9lf %.9lf\n",A.x,A.y,B.x,B.y,C.x,C.y,D.x,D.y);
        return true;
    }else return false;
}   
int main()
{
    scanf("%d",&T);
    while(T--)
    {
        a.get();
        b.get();
        c.get();
        if(fabs(chaji(a,b,c))>eps&&(get_ans(a,b,c)||get_ans(b,a,c)||get_ans(a,c,b)));else printf("NO\n\n");
    }
    return 0;
}
