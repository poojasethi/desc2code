#include <cstdio>
int t,x1,y1,x2,y2,x3,y3,a1,b1,c1,a2,b2,c2;
double X1,Y1,X2,Y2,X3,Y3,X4,Y4,v1,v2,v3,v4;
double cj(double x0,double y0,double x1,double y1,double x2,double y2)
{
	return (x1-x0)*(y2-y0)-(x2-x0)*(y1-y0);
}
int sqr(int x){return x*x;}
bool js(int x1,int y1,int x2,int y2,int x3,int y3)
{
	a1=(-x1+x2)*2;b1=(-y1+y2)*2;c1=sqr(2*x1-x2)+sqr(2*y1-y2)-sqr(x1)-sqr(y1);
	a2=(x3-2*x2+x1)*2;b2=(y3-2*y2+y1)*2;c2=sqr(x1)+sqr(y1)-sqr(x3-2*x2+2*x1)-sqr(y3-2*y2+2*y1);
	if (a1*b2==a2*b1) return false;
	Y1=double(c2*a1-c1*a2)/(b1*a2-b2*a1);X1=double(c2*b1-c1*b2)/(a1*b2-a2*b1);
	X2=2*x1-X1;Y2=2*y1-Y1;X3=2*x2-2*x1+X1;Y3=2*y2-2*y1+Y1;X4=2*x3-2*x2+2*x1-X1;Y4=2*y3-2*y2+2*y1-Y1;
	v1=cj(X1,Y1,X2,Y2,X3,Y3);
	v2=cj(X2,Y2,X3,Y3,X4,Y4);
	v3=cj(X3,Y3,X4,Y4,X1,Y1);
	v4=cj(X4,Y4,X1,Y1,X2,Y2);
	if ((v1<0)&&(v2<0)&&(v3<0)&&(v4<0)||(v1>0)&&(v2>0)&&(v3>0)&&(v4>0))
	{
		printf("YES\n%.9lf %.9lf %.9lf %.9lf %.9lf %.9lf %.9lf %.9lf\n",X1,Y1,X2,Y2,X3,Y3,X4,Y4);return true;
	}
	return false;
}
int main()
{
	scanf("%d",&t);
	for (;t>0;--t)
	{
		scanf("%d%d%d%d%d%d",&x1,&y1,&x2,&y2,&x3,&y3);
		if (js(x1,y1,x2,y2,x3,y3)||js(x1,y1,x3,y3,x2,y2)||js(x2,y2,x1,y1,x3,y3));else printf("NO\n\n");
	}
}
