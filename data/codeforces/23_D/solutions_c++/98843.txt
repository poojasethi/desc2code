#include <stdio.h>
#include <string.h>
#include <math.h>

int ccw(double x1, double y1, double x2, double y2, double x3, double y3)
{
  double dx1 = x2-x1;
  double dy1 = y2-y1;
  double dx2 = x3-x2;
  double dy2 = y3-y2;
  double diff = dy1*dx2 - dy2*dx1;
  if (fabs(diff) < 1e-7) return 0;
  if (diff < 0) return 1;
  else return -1;
}

int find(int x1, int y1, int x2, int y2, int x3, int y3)
{
  if ((x2-x1)*(y3-y2) == (y2-y1)*(x3-x2)) return 0;
  int ax = x1, ay = y1;
  int bx = -x2+2*x1, by = -y2+2*y1;
  int cx = x3-2*x2+2*x1, cy = y3-2*y2+2*y1;
//  printf("%d %d %d %d %d %d\n", ax, ay, bx, by, cx, cy);
  int d = 2*(ax*(by-cy)+bx*(cy-ay)+cx*(ay-by));
  double p0x = ((ay*ay+ax*ax)*(by-cy) + (by*by+bx*bx)*(cy-ay) + (cy*cy+cx*cx)*(ay-by)) / (double)d;
  double p0y = ((ay*ay+ax*ax)*(cx-bx) + (by*by+bx*bx)*(ax-cx) + (cy*cy+cx*cx)*(bx-ax)) / (double)d;
  double p1x = 2*x1-p0x;
  double p1y = 2*y1-p0y;
  double p2x = 2*x2-p1x;
  double p2y = 2*y2-p1y;
  double p3x = 2*x3-p2x;
  double p3y = 2*y3-p2y;

  // check convex
  double ccw1 = ccw(p0x, p0y, p1x, p1y, p2x, p2y);
  double ccw2 = ccw(p1x, p1y, p2x, p2y, p3x, p3y);
  double ccw3 = ccw(p2x, p2y, p3x, p3y, p0x, p0y);
  double ccw4 = ccw(p3x, p3y, p0x, p0y, p1x, p1y);
  if (ccw1==0 || ccw2==0 || ccw3==0 || ccw4==0) return 0;
  if (ccw1 != ccw2 || ccw2 != ccw3 || ccw3 != ccw4) return 0;
  printf("YES\n");
  printf("%.9f %.9f %.9f %.9f %.9f %.9f %.9f %.9f\n", p0x, p0y, p1x, p1y, p2x, p2y, p3x, p3y);
  return 1;
}

int main()
{
  int T;
  scanf("%d", &T);
  for (int i=0; i<T; i++) {
    int x1, y1, x2, y2, x3, y3;
    scanf("%d %d %d %d %d %d", &x1, &y1, &x2, &y2, &x3, &y3);
    if (!find(x1, y1, x2, y2, x3, y3))
      if (!find(x1, y1, x3, y3, x2, y2))
        if (!find(x2, y2, x1, y1, x3, y3))
          printf("NO\n\n");
  }
  return 0;
}
