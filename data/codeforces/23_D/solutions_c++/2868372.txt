#include<cstdio>
struct poi{
	double x,y;
	poi(){x=y=0;}
	poi(double a,double b){x=a,y=b;}
	double operator*(const poi b) const{
		return x*b.y-y*b.x;
	}
	void scan(){
		scanf("%lf%lf",&x,&y);
	}
	void print(bool p){
		printf("%.9lf %.9lf",x,y);
		printf(p?"\n":" ");
	}
} p[3],s[5];
poi operator -(const poi a,const poi b){
	return poi(a.x-b.x,a.y-b.y);
}
poi operator +(const poi a,const poi b){
	return poi(a.x+b.x,a.y+b.y);
}
poi operator *(double a,const poi b){
	return poi(a*b.x,a*b.y);
}
bool ol(const poi a,const poi b,const poi c){
	return (b-a)*(c-a)==0;
}
poi rot(const poi a){
	return poi(a.y,-a.x);
}
bool ck(const poi a,const poi b,const poi c){
	poi d=rot(a-b),e=rot(b-c),f=b-0.5*(a+c);
	double v=(f*e)/(d*e);
	s[0]=0.5*(a+b)+v*d;
	s[3]=2*a-s[0],s[2]=2*c-(s[1]=2*b-s[0]),s[4]=s[0];
	if (ol(s[0],s[1],s[2]) || ol(s[0],s[1],s[3]) || ol(s[0],s[2],s[3]) || ol(s[1],s[2],s[3]))
		return false;
	int i;
	bool pd;
	for (pd=(s[0]-s[3])*(s[1]-s[0])>0,i=1;i<=3 && ((s[i]-s[i-1])*(s[i+1]-s[i])>0)==pd;i++);
	if (i==4){
		puts("YES");
		for (i=0;i<4;i++) s[i].print(i==3);
		return true;
	} else return false;
}
int main(){
	int n,i;
//	freopen("23d.in","r",stdin);
//	freopen("23d.out","w",stdout);
	for (scanf("%d",&n);n;n--){
		for (i=0;i<3;i++) p[i].scan();
		if (ol(p[0],p[1],p[2]) || !ck(p[0],p[1],p[2]) && 
			!ck(p[1],p[0],p[2]) && !ck(p[0],p[2],p[1]))
			puts("NO\n");
	}
	return 0;
}
