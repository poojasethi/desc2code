rl = lambda: raw_input().strip()

cases = int(rl())
for cc in xrange(cases):
    n, k, d1, d2 = map(int, rl().split())
    def test(d1, d2):
        b3 = k - d1 - d2
        if b3 % 3 > 0: return False
        b = b3 / 3
        a = b + d1
        c = b + d2
        return min([a, b, c]) >= 0 and max([a, b, c]) <= n / 3
    def possible():
        if n % 3 > 0: return False
        return any([test(d1, d2), test(-d1, d2), test(d1, -d2), test(-d1, -d2)])

    print 'yes' if possible() else 'no'
