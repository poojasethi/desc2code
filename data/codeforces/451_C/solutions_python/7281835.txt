for t in xrange(input()):
	n, k, d1, d2 = map(int, raw_input().split())
	def p(n,k,d1,d2):
		if n % 3 == 0:
			x2 = (k-d1+d2)/3
			if (k-d1+d2) % 3 == 0 and x2 >= 0 and x2 <= k:
				x1, x3 = d1+x2, x2-d2
				if x1 >= 0 and x1 <= k and x3 >= 0 and x3 <= k:
					if x1 <= n/3 and x2 <= n/3 and x3 <= n/3:
						return 1
		return 0
	print "yes" if any([p(n,k,d1,d2),p(n,k,d1,-d2),p(n,k,-d1,d2),p(n,k,-d1,-d2)]) else "no"
