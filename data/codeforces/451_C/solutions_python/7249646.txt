#!/usr/bin/env python


def read():
    n = int(raw_input())
    ret = []
    for i in range(n):
        ret.append(map(int, raw_input().split()))
    return ret


def work(vvList):
    
    
    for (n, k, d1, d2) in vvList:

        def judge():
            if n % 3 != 0:
                print "no"
                return
            
            for mul1 in range(-1, 2, 2):
                for mul2 in range(-1, 2, 2):
                    t = k - mul1 * d1 - mul1 * d1 - mul2 * d2
                    
                    if t % 3 == 0 and t >= 0:
                        x1 = t / 3
                        x2 = t / 3 + mul1 * d1
                        x3 = t / 3 + mul1 * d1 + mul2 * d2
                        
                        if 0 <= min(x1, x2, x3) and max(x1, x2, x3) <= n / 3:
                            print "yes"
                            return
                    
                    t = k - mul1 * d1 - mul2 * d2
                    
                    if t % 3 == 0 and t >= 0:
                        x1 = t / 3 + mul1 * d1
                        x2 = t / 3
                        x3 = t / 3 + mul2 * d2
                        
                        if 0 <= min(x1, x2, x3) and max(x1, x2, x3) <= n / 3:
                            print "yes"
                            return
            
            print "no"
        
        judge()
    

if __name__ == "__main__":
    work(read())
