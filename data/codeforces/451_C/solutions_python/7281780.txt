for t in xrange(input()):
	n, k, d1, d2 = map(int, raw_input().split())
	if n % 3: print "no"
	else:
		p = 0
		for d1 in [-d1,d1]:
			for d2 in [-d2,d2]:
				x2 = (k-d1+d2)/3
				if (k-d1+d2) % 3: continue
				if x2 >= 0 and x2 <= k:
					x1 = d1 + x2
					x3 = x2 - d2
					if x1 >= 0 and x1 <= k and x3 >= 0 and x3 <= k:
						if x1 <= n/3 and x2 <= n/3 and x3 <= n/3:
							p = 1
		if p: print "yes"
		else: print "no"
