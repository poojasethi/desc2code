#include<algorithm>
#include<iostream>
using namespace std;
bool ck(long long n,long long k,long long a,long long b)
{
	k-=a+b;
	if(k%3)return false;
	k/=3;
	if(k<0||k+a<0||k+b<0)return false;
	return k<=n&&k+a<=n&&k+b<=n;
}
int main()
{
	int t;
	cin>>t;
	while(t--)
	{
		long long n,k,a,b;
		cin>>n>>k>>a>>b;
		if(n%3){cout<<"no\n";continue;}
		n/=3;
		if(ck(n,k,a,a+b)||ck(n,k,a,a-b)||ck(n,k,-a,b-a)||ck(n,k,-a,-a-b))cout<<"yes\n";
		else cout<<"no\n";
	}
	return 0;
}
