#include<iostream>
#define ll long long
using namespace std;
bool check(ll n,ll k,ll d1,ll d2)
{
    return n-k>=d1 && (n-k-d1)%3==0 && k>=d2 && (k-d2)%3==0;
}
int main()
{
    ll t;
    cin>>t;
    while(t--)
    {
        ll n,k,d1,d2;
        cin>>n>>k>>d1>>d2;
        if(check(n,k,d1+d2*2,d1*2+d2) || check(n,k,d1*2+d2,d1+d2*2) || check(n,k,d1+d2,max(d1,d2)*2-min(d1,d2)) || check(n,k,max(d1,d2)*2-min(d1,d2),d1+d2))
            cout<<"yes\n";
        else
            cout<<"no\n";
    }
    return 0;
}
