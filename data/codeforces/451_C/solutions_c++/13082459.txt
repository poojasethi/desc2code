#include<cstdio>
#include<algorithm>
using namespace std;
#define ll long long
ll n,k;
bool f(ll a,ll b)
{
	ll c=k-a-b*2;
	if(c<0||c%3)return 0;
	return n/3>=a+b+c/3;
}
int main()
{
	int t;ll d1,d2,a,b;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%I64d%I64d%I64d%I64d",&n,&k,&d1,&d2);
		if(n%3){puts("no");continue;}
		if(f(d1,d2)||f(abs(d1-d2),min(d1,d2))||
			f(d2,d1)||f(min(d1,d2),abs(d1-d2)))
			puts("yes");
		else puts("no");
	}
}
