#include <iostream>

using namespace std;

int main()
{
    long long n,k,d1,d2,t,flag;
    cin>>t;
    while(t--){
    cin>>n>>k>>d1>>d2;
    if(d2>d1)
		d1^=d2,
		d2^=d1,
		d1^=d2;
    n-=k;
    flag=0;
    if(((n-d1-d2)>=0&&(n-d1-d2)%3==0&&k>=2*d1-d2&&(k-2*d1+d2)%3==0)||
	   (n-d1-d2-d2>=0&&(n-d1-d2-d2)%3==0&&k>=d1+d2+d1&&(k-d1-d1-d2)%3==0)
	   ||(n-d1-d1-d2>=0&&(n-d1-d1-d2)%3==0&&k>=d1+d2+d2&&(k-d1-d2-d2)%3==0)
	   ||((n+d2-d1-d1)>=0&&(n+d2-d1-d1)%3==0&&k>=d1+d2&&(k-d1-d2)%3==0))
		cout<<"yes"<<endl;
	else
		cout<<"no"<<endl;
    }
    return 0;
}
