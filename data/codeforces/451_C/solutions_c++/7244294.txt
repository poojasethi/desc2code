#include<stdio.h>
#include<iostream>
#define LL long long
using namespace std;
LL n,k;
bool can(LL d1,LL d2)
{
    LL tmp = 2*d1+d2+k;
    LL m   =n/3;
    if(tmp%3) return false;
    LL a = tmp/3;
    LL b = a-d1;
    LL c = k-(a+b);
    if(a<0||b<0||c<0||a>m||b>m||c>m) return false;
    return true;
}
int main ()
{
    int t;
    LL d1,d2;
    scanf("%d",&t);
    while(t--){
        cin >> n>>k>>d1>>d2;
        if(n%3==0&&(can(d1,d2)||can(-d1,d2)||can(d1,-d2)||can(-d1,-d2)))
            printf("yes\n");
        else
            printf("no\n");
    }
    return 0;
}
