{

{

bool f=0;
if (n%3==0)
{ n/=3;
for (i=-1;i<=1;i+=2)
for (j=-1;j<=1;j+=2)
{
a=k+d2*i+d1*2*j;
if ((a%3==0)&&(a>=0)&&(a/3-d1*j>=0)&&(a/3-d1*j-d2*i>=0)) 
{
a=a/3;
if ((n>=a)&&(n>=a-d1*j)&&(n>=a-d2*i-d1*j))
	{
	f=1;
	}
}
}
}

}
}
