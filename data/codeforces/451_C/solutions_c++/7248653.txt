#include <bits/stdc++.h>
using namespace std;

long long t, a, b, c, d;

bool go(long long c, long long d){
  if((b + c + c + d) % 3 != 0) return 0;
  long long x = (b + c + c + d) / 3, y = x - c, z = b - x - y;
  return x >= 0 && y >= 0 && z >= 0 && x <= a / 3 && y <= a / 3 && z <= a / 3;
}

int main(){
  ios::sync_with_stdio(0);
  cin >> t;
  while(t--){
    cin >> a >> b >> c >> d;
    cout << (a % 3 == 0 && (go(c, d) || go(c, -d) || go(-c, d) || go(-c, -d)) ? "yes\n" : "no\n");
  }
}
