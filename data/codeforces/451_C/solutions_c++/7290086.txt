#include <iostream>
using namespace std;
typedef long long LL;

bool f(LL x){
  return x>=0 && x%3==0;
}

int main(){
  int t;
  cin >> t;
  while(t--){
    LL n,k,d1,d2;
    cin >> n >> k >> d1 >> d2;
    if(f(k+d2-2*d1)&&f(k+d1-2*d2)&&f(n-k-d1-d2))
      cout << "yes" << endl;
    else if(f(k-d1-d2)&&(((d1>=d2)&&f(n-k-2*d1+d2))||((d1<d2)&&f(n-k-2*d2+d1))))
      cout << "yes" << endl;
    else if(f(k-d1-2*d2)&&f(n-k-2*d1-d2))
      cout << "yes" << endl;
    else if(f(k-d2-2*d1)&&f(n-k-d1-2*d2))
      cout << "yes" << endl;
    else
      cout << "no" << endl;
  }
  return 0;
}
