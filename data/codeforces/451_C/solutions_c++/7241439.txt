#include<stdio.h>
int f(long long a)
{
return (a>=0&&(double)a/3==a/3);
}
int main()
{
int t;
long long n,k,d1,d2;
while(scanf("%d",&t)==1)
while(t--)
{
scanf("%I64d%I64d%I64d%I64d",&n,&k,&d1,&d2);
if(f(k+d2-2*d1)&&f(k+d1-2*d2)&&f(n-k-d1-d2))
printf("yes\n");
else if(f(k-d1-d2)&&(((d1>=d2)&&f(n-k-2*d1+d2))||((d1<d2)&&f(n-k-2*d2+d1))))
printf("yes\n");
else if(f(k-d1-2*d2)&&f(n-k-2*d1-d2))
printf("yes\n");
else if(f(k-d2-2*d1)&&f(n-k-d1-2*d2))
printf("yes\n");
else
printf("no\n");
}
return 0;
}
