#include<bits/stdc++.h>
using namespace std;

long long n,k;

inline bool j(long long s1,long long s2,long long s3)
{
	if(k-s1-s2-s3<0||(k-s1-s2-s3)%3!=0)
		return 0;
	long long mx=max(max(s1,s2),s3);
	long long x=n-(k+mx-s1+mx-s2+mx-s3);
	if(x<0||x%3!=0)
		return 0;
	return 1;
}

int main()
{
	int T;
	long long d1,d2,d;
	for(scanf("%d",&T);T--;)
	{
		scanf("%I64d%I64d%I64d%I64d",&n,&k,&d1,&d2);
		d=max(d1,d2);
		puts(j(0ll,d1,d1+d2)||j(d1+d2,d2,0ll)||j(d1,0ll,d2)||j(d-d1,d,d-d2)?"yes":"no");
	}
	return 0;
}
