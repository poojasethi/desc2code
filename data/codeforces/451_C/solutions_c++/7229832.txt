#include <iostream>
#include <algorithm>
#define ll long long
using namespace std;
ll n,k;
bool b(ll f,ll s,ll t){
	if(n%3) return false;
	ll z = n/3;
	if(f<0 || f>z || s <0 || s>z || t<0 || t>z) return false;
	return true;
}
bool a(ll d1,ll d2){
	if((k-d1-d2)%3) return false;
	ll s = (k-d1-d2)/3;
	ll f = s+d1, t= s+d2;
	return b(f,s,t);
}

int main(){
	ll t,d1,d2;
	cin >> t;
	while(t--){
		cin >> n >> k >> d1 >> d2;
		if(a(d1,d2)||a(d1,-d2)||a(-d1,d2)||a(-d1,-d2)) cout << "yes\n";
		else cout << "no\n";
	}
	return 0;
}
