#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
#define LL long long
LL t,n,k,d1,d2;
bool _(LL a,LL b){return (a/3>=b && a/3*3==a && a<=n);}
bool x()
{
	if(n%3)return false;
	if(_(n-k+d1+d2*2,d1+d2))return true;
	if(_(n-k+d2+d1*2,d1+d2))return true;
	if(_(n-k+d1+d2,max(d1,d2)))return true;
	if(_(n-k-min(d1,d2)+max(d1,d2)*2,max(d1,d2)))return true;
	return false;
}
int main(){scanf("%I64d",&t);for(LL i=0;i<t;i++)cin>>n>>k>>d1>>d2,printf("%s\n",x()?"yes":"no");return 0;}
