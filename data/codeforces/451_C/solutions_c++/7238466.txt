#include <iostream>
#include <algorithm>
#include <stdlib.h>
using namespace std;
typedef long long ll;
bool solve(ll n,ll k,ll d1,ll d2)
{
    ll v=(n-k)-(2*d1+d2);
    ll u=k-(d1+d2*2);
    if(v>=0&&v%3==0&&u>=0&&u%3==0)
        return 1;
        return 0;

}
int main()
{
    ll n,k,d1,d2;
    int T;
    cin>>T;
    while(T--)
    {
    cin>>n>>k>>d1>>d2;
    if(d1>d2)
        swap(d1,d2);
    if(solve(n,k,d1,d2)||solve(n,k,d2,d1)||solve(n,k,d2-d1,d1)||solve(n,k,d1,d2-d1))
        cout<<"yes"<<endl;
    else
        cout<<"no"<<endl;
    }
}
