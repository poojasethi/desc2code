#include<stdio.h>
#include<string.h>
#include<stdlib.h>

int main()
{
	int t;
	long long n,k,i,d1,d2,sum;
	scanf("%d",&t);
	while(t--)
	{
		scanf("%lld%lld%lld%lld",&n,&k,&d1,&d2);
		sum=d2>d1?2*d2-d1:2*d1-d2;
		if(((n-k>=sum)&&(n-k-sum)%3==0)&&(k-d1-d2)>=0&&(k-d1-d2)%3==0||((n-k>=2*d1+d2)&&(n-k-d2-2*d1)%3==0)&&(k-d1-2*d2)>=0&&(k-d1-2*d2)%3==0||((n-k>=d1+2*d2)&&(n-k-d1-2*d2)%3==0)&&(k-2*d1-d2)>=0&&(k-2*d1-d2)%3==0||(n-k>=d1+d2)&&(n-k-d1-d2)%3==0&&(k-sum)>=0&&(k-sum)%3==0) printf("yes\n");
		else printf("no\n");
	}
}
