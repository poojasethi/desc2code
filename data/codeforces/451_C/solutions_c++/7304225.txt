#include <bits/stdc++.h>
#define l long long
using namespace std;
l _,n,k,d1,d2,t,x,y;
bool V(l i,l j,l k,l n,l m){
    l u=m-i-j-k,g=-u/3;
    if (u%3||i<g||j<g||k<g) return 0;
    y=(n-m)-max(i,max(j,k))*3+i+j+k;
    return y>=0;
}
