def calculate(number, _sum, _max):
	if number == 0:
		return []
	ret = [_max] * number
	total = _max * number
	pos = 0
	while total > _sum:
		ret[pos] -= 1
		total -= 1
		pos = (pos + 1) % number
	return ret

n, k, l, r, s_all, s_k = map(int, raw_input().split())

first = calculate(k, s_k, r)
last = calculate(n-k, s_all - s_k, min(first))

print ' '.join(map(str, first + last))
