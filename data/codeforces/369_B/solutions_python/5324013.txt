from __future__ import division
import sys

n, k, l, r, sall, sk = map(int, sys.stdin.readline().split())

avgk = sk // k
remk = sk % k

if n > k:
    avgall = (sall - sk) // (n - k)
    remall = (sall - sk) % (n - k)

results = []
for i in range(k):
    results.append(avgk + (1 if i < remk else 0))
for i in range(n - k):
    results.append(avgall + (1 if i < remall else 0))

print " ".join(map(str, results))
