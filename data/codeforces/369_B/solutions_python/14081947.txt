n,k,l,r,sa,sk = map(int,raw_input().split())
la= ''

x = sk / k
resto = n - k
sr = sa -sk
if resto == 0:
    h = 0
else:
    h = sr / resto


for i in range(resto):
    if i < sr-(h*resto):
        la += str(h+1)+' '
    else:
        la += str(h)+' '
for j in range(k):
    if j < sk - (x*k):
        la += str(x+1)+' '
    else:
        la += str(x)+' '
print la
