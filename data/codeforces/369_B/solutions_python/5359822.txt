n, k, l, r, sa, sk = map(int, raw_input().split())
a, tk = [l for i in range(n)], sk - (k * l) 
def fill(a, tK):
  N = len(a)
  if N <= 0: return a
  for i in range(N):
    a[i] += tK / N
  for i in range(tK % N):
    a[i] += 1
  return a
a[:k] = fill(a[:k], sk - (k * l))
a[k:n] = fill(a[k:n], sa - sum(a))
print ' '.join(map(str, a))
