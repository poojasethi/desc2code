ri = lambda: raw_input().strip()

n, k, l, r, sall, sk = map(int, ri().split())

maxScore = sk / k
remain = sk % k

for x in xrange(0, remain):
    print maxScore+1,

for x in xrange(0, k-remain):
    print maxScore,

if n-k != 0:
    remainValue = (sall-sk) / (n - k)
    remain = (sall-sk) % (n - k)

    for x in xrange(0, remain):
        print remainValue+1,

    for x in xrange(0, n-k-remain):
        print remainValue,
