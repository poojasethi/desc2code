n,k,l,r,sall,sk = map(int, raw_input().strip().split(' '))
sk_left = sk-(sk/k)*k
answer_lis = [sk/k + 1]*sk_left + [sk/k]*(k-sk_left)
score_left = sall-sk
if n - k != 0:
    ssk_left = score_left - (score_left/(n-k))*(n-k)
    answer_lis += [score_left/(n-k) + 1]*ssk_left + [score_left/(n-k)]*(n-k-ssk_left)

print " ".join(str(i) for i in answer_lis)
