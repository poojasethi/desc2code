ss = raw_input()
n, k, l, r, s_all, s_k = ss.split()
n = int(n)
k = int(k)
l = int(l)
r = int(r)
s_all = int(s_all)
s_k = int(s_k)

left_part = [s_k // k for i in xrange(k)]
tmp = s_k % k
for i in xrange(tmp):
	left_part[i] += 1

if n > k:
	right_part = [(s_all - s_k) // (n - k) for i in xrange(n - k)]
	remain = (s_all - s_k) % (n - k)
	for i in xrange(remain):
		right_part[i] += 1
else :
	right_part = []
left_part.extend(right_part)
print ' '.join([str(itm) for itm in left_part])
