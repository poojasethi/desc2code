__author__ = 'leonardo'

n, k, l, r, s_all, s_k = map(int, raw_input().split(' '))

points = [0] * n

rest = s_k % k
point = s_k / k

for i in xrange(k):
    points[i] = point

for i in xrange(rest):
    points[i] += 1

reminder = (s_all - s_k)

if n > k:
    rest = reminder % (n - k)
    point = reminder / (n -k)

    for i in xrange(k,n):
        points[i] = point

    for i in xrange(rest):
        points[k+i] += 1

exit = ""
for p in xrange(n):
    exit += "%s " % points[p]

print exit[:-1]
