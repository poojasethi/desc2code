n, k, l, r, sa, sk = map(int, raw_input().split())
a = [l] * n
sa -= l * n
sk -= l * k
sa -= sk

for i in xrange(r - l + 1, 0, -1):
    if i * k > sk and (i - 1) * k <= sk:
        break
mn = i - 1
for i in xrange(0, k, 1):
    a[i] += mn
sk -= mn * k
i = 0
while sk > 0:
    a[i] += 1
    sk -= 1
    i += 1

for i in xrange(k, n, 1):
    if sa == 0:
        break
    if mn < sa:
        a[i] += mn
        sa -= mn
    else:
        a[i] += sa
        sa = 0
