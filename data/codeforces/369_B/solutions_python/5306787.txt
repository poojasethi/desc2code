data = map(int, raw_input().split())

n, k, l, r, s, ks = data

def getScores(total, n):
    if n == 0:
        return []
    more = total % n
    base = total / n

    return [base]*(n - more) + [base + 1]*more

scores = getScores(ks, k)
scores.extend(getScores(s - ks, n - k))

print " ".join(map(str, scores))
