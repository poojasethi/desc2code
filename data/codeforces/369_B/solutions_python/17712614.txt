def read_int_tuple():
	l = raw_input().strip().split()
	return tuple(map(int, l))

def main():
	n, k, l, r, s_all, s_k = read_int_tuple()
	scores = list()
	for i in xrange(k):
		scores.append(s_k/k)
	for i in xrange(s_k % k):
	    scores[i] += 1
	s_all -= s_k
	l = n-k
	if n > k:
		for i in xrange(l):
			scores.append(s_all/l)
		for i in xrange(s_all %l):
			scores[i+k] += 1
	
	print " ".join(map(str,scores))

if __name__ == "__main__":
    main()
