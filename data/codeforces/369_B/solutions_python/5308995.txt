n,k,l,r,sall,sk=map(int,raw_input().split(' '))

if n!=k:
    remain=n-k
    sremain=sall-sk

    highdiv=sk/k
    highmod=sk%k
    lowdiv=sremain/remain
    lowmod=sremain%remain

#print highdiv,highmod,lowdiv,lowmod

    print (str(highdiv+1)+' ')*highmod + (str(highdiv)+' ')*(k-highmod) +  (str(lowdiv+1)+' ')*lowmod + (str(lowdiv)+' ')*(remain-lowmod)

else:
    divi=sall/n
    modi=sall%n
    print (str(divi+1)+' ')*modi + (str(divi)+' ')*(n-modi)
