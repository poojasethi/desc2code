#include<cstdio>
int main()
{
    int n,k,l,r,sall,sk,i,x;
    scanf("%d %d %d %d %d %d",&n,&k,&l,&r,&sall,&sk);
    x=sk-l*k;
    for(i=0;i<k;i++)
    {
        printf("%d ",l+x/k+(i<x%k?1:0));
    }
    x=sall-sk-l*(n-k);
    for(;i<n;i++)
    {
        printf("%d ",l+x/(n-k)+((i-k)<x%(n-k)?1:0));
    }
}
