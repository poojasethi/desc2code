#include <iostream>

using namespace std;

int main()
{
    int n,k,l,r,sa,sk ;
    cin >>n>>k>>l>>r>>sa>>sk ;
    int temp = n-k ;
    for (int i =0 ; i < k ; i++){
        cout << (sk+i)/k << " " ;
    }
    for (int i =0  ; i < temp ; i++){
        cout << (sa-sk +i ) / temp  << " " ;
    }
    return 0;
}
