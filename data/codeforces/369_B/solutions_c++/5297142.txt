#include <cstdio>

int n, k, l, r, s, sk, p;

int main()
{
	scanf("%d%d%d%d%d%d", &n, &k, &l, &r, &s, &sk);

	while (k)
	{
		p = sk / k;
		if (sk % k)
			++ p;
		sk -= p;
		s -= p;
		-- n;
		-- k;
		printf("%d ", p);
	}

	while (n)
	{
		p = s / n;
		if (s % n)
			++ p;
		s -= p;
		-- n;
		printf("%d ", p);
	}

	return 0;
}
