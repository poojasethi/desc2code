#include<stdio.h>

using namespace std;

int main () {
  int n,k,l,r,sall,sk;

  scanf("%d %d %d %d %d %d", &n, &k, &l, &r, &sall, &sk);

  sall -= sk;
  n -= k;
  while (k>0) {
    printf("%d ", sk/k);
    sk -= sk/k;
    k--;
  }

  while (n>0) {
    printf("%d ", sall/n);
    sall -= sall/n;
    n--;
  }


  return  0;
}
