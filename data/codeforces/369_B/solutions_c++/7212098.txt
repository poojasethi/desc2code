#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,k,l,r,tot,s,i,rem,a[1010];
	cin>>n>>k>>l>>r>>tot>>s;
	rem=tot-s;
	for(i=0;i<k;i++)
	{
		a[i]=s/k;
		if(s%k!=0)
		{
			s--;
			a[i]++;
		}
	}
	for(i=k;i<n;i++)
	{
		a[i]=rem/(n-k);
		if(rem%(n-k)!=0)
		{
			rem--;
			a[i]++;
		}
	}
	for(i=0;i<n;i++)
		cout<<a[i]<<' ';
	cout<<endl;
	return 0;
}
