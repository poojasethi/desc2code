#include <stdio.h>

int main()
{
int n,k,l,r,Sall,Sk; 
while(scanf("%d%d%d%d%d%d",&n,&k,&l,&r,&Sall,&Sk)==6)
{
   int i=0;
   Sall-=Sk;
   n-=k;
   for(i=k;i>=1;i--)
   {
      printf("%d ",Sk/i);
      Sk=Sk-(Sk/i);
   }
   for(i=n;i>=1;i--)
   {
      printf("%d ",Sall/i);
      Sall=Sall-(Sall/i);
   }
}
return 0;  
}
