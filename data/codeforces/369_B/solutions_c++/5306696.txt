#include<iostream>
using namespace std;
int main()
{
    int n,m,k,l,r,sall,sk;
    cin>>n>>k>>l>>r>>sall>>sk;
    sall=sall-sk;
    n=n-k;
    for(int i=k;i>0;i--)
    {
        m=sk/i;
        cout<<m<<" ";
        sk=sk-m;
    }
    
    for(int i=n;i>0;i--)
    {
        m=sall/i;
        cout<<m<<" ";
        sall=sall-m;
    }
}
