#include <iostream>

using namespace std;

int main()
{
    int n,k,l,r,s,sk,i;

    cin >> n >> k >> l >> r >> s >> sk;


    s-=sk;
    n-=k;

    for(i=0; i<k; i++){
    cout << sk/(k-i) << " ";
    sk=sk-(sk/(k-i));}

    for(i=0; i<n; i++){
    cout << s/(n-i)<< " ";
    s=s-s/(n-i);}
    return 0;
}
