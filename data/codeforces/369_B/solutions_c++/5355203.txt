#include<iostream>
using namespace std;
int main() {
    int n, k, l, r, sall, sk;
    cin >> n >> k >> l >> r >> sall >> sk;
    for(int i=0;i<k;i++)
        cout << sk/k + (i < sk % k) << ' ';
    for(int i=0;i<n-k;i++)
        cout << (sall-sk)/(n-k) + (i < (sall-sk)%(n-k)) << ' ';
    return 0;
}
