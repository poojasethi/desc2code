#include <bits/stdc++.h>

using namespace std;
int main()
{
   int n,k,l,r,sall,sk;
   cin>>n>>k>>l>>r>>sall>>sk;
   sall-=sk;
   for (int i=0;i<k;i++)
   {
       cout<<sk/(k-i)<<" ";
       sk-=sk/(k-i);
   }
   for (int i=k;i<n;i++)
   {
       cout<<sall/(n-i)<<" ";
       sall-=sall/(n-i);
   }
        return 0;
}
