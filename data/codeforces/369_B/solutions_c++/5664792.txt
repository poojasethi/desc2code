#include <cstdio>

int main(){
    int n,k,l,r,sall,sk;
    scanf("%d %d %d %d %d %d",&n,&k,&l,&r,&sall,&sk);
    n=n-k,sall=sall-sk;
    while(k>0)
    {
        printf("%d ",sk/k);
        sk=sk-sk/k;
        k--;
    }
    while(n>0)
    {
        printf("%d ",sall/n);
        sall=sall-sall/n;
        n--;
    }
    return 0;
}
