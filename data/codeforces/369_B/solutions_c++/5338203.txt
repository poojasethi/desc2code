
#include <iostream>

using namespace std;

long n, k, l, r, sAll, sk;
int main()
{
	
	int temp = 0;
	cin>>n>>k>>l>>r>>sk>>sAll;
	sk -= sAll;
	n -= k;
	for(int i = 0; i < k; i++)
	{
		temp = sAll / (k - i);
		cout<<temp<<" ";
		sAll -= temp;
	}
	for(int i = 0; i < n; i++)
	{
		temp = sk / (n - i);
		cout<<temp<<" ";
		sk -= temp;
	}
}
