#include<bits/stdc++.h>

using namespace std;

int main()
{
    int n,m,k,l,r,sall,sk,i;
    cin>>n>>k>>l>>r>>sall>>sk;
    sall-=sk;
    n-=k;
    for(i=k;i>0;i--)
    {
        m=sk/i;
        cout<<m<<" ";
        sk-=m;
    }
    for(i=n;i>0;i--)
    {
        m=sall/i;
        cout<<m<<" ";
        sall-=m;
    }
}
