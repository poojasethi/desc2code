#include<iostream>
using namespace std;



int main()
{

    int n,k,l,r,sall,sk;
    cin>>n>>k>>l>>r>>sall>>sk;
    n=n-k,sall=sall-sk;
    while(k>0)
    {
        cout<<sk/k<<" ";
        sk=sk-sk/k;
        k--;
    }
    while(n>0)
    {
        cout<<sall/n<<" ";
        sall=sall-sall/n;
        n--;
    }


    return 0;
}
