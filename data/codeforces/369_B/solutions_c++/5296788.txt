#include <bits/stdc++.h>
using namespace std;

int main() {
	int n,k,l,r,sall,sk;
	cin>>n>>k>>l>>r>>sall>>sk;
	for(int i=0;i<n-k;i++)
	{
		int t = (sall-sk)/(n-k);
		if(i<((sall-sk)%(n-k)))t++;
		cout<<t<<" ";
	}
	for(int i=n-k;i<n;i++)
	{
		int t = (sk)/(k);
		if((i-n+k)<((sk)%(k)))t++;
		cout<<t<<" ";
	}
	cout<<endl;
	return 0;
}
