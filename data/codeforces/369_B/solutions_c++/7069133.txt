#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int a[1005]={0};
int main()
{  
    int  n,k,l,r,sall,sk;
    cin>>n>>k>>l>>r>>sall>>sk;
    for(int i=0;i<sk;i++)  a[i%k]++;
    for(int i=0;i<sall-sk;i++) a[i%(n-k)+k]++;
    for(int i=0;i<n;i++)    cout<<a[i]<<" ";
    cout<<endl;
}
