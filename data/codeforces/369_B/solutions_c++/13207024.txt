#include <iostream>
using namespace std;
int main ()
{
    int n , k , l , r , s_all , s_k;
    cin>>n>>k>>l>>r>>s_all>>s_k;
    for (int i = 0; i < k; ++i)
        cout<<(s_k + i) / k<<" ";
    for (int i = 0; i < n - k; ++i)
        cout<<(s_all - s_k + i) / (n - k)<<" ";
    cout<<endl;
    return 0;
}
