#include<iostream>

using namespace std;

int n,k,l,r,sum,sumk;

int main(){

  cin >> n >> k >> l >> r >> sum >> sumk;

  sum -= sumk;

  for( int i = k ; i ; i-- ){
    cout << (sumk/i) << " ";
    sumk -= sumk/i;
  }

  for( int i = n-k ; i ; i-- ){
    cout << (sum/i) << " ";
    sum -= sum/i;
  }

  cout << endl;

}
