#include <iostream>

using namespace std;

int main() {
    long long x,y,z,k;
    cin>>x>>y>>z>>k;
    while (((x-1)+(y-1)+(z-1))>k) {
        int m=max(x,max(y,z));
        if (m==x)
            x--;
        else if (m==y)
            y--;
        else
            z--;
    }
    cout<<x*y*z<<endl;
    return 0;
}
