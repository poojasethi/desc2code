#include <iostream>
#include <algorithm>
using namespace std;

long long x[3];
long long ans[3];
long long cuts;
int main()
{
	cin >> x[0] >> x[1] >> x[2] >> cuts;
	sort(x,x+3);

	for(int i=0;i<3;i++)
	{
		x[i]--;
		long long w = cuts/(3-i);
		if(cuts%(3-i)) w++;
		ans[i] = min(x[i],w);
		cuts-=ans[i];		
	}
		cout << (ans[0]+1) * (ans[1]+1) * (ans[2]+1) << endl;
	
	return 0;
}
