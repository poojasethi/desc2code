#include <iostream>
#include <stdio.h>


using namespace std;

int main()
{
    long long x,y,z,s,k,a,b;
    cin>>x>>y>>z>>k;
    if (x>y)
      swap(x,y);
      if (x>z)
        swap(x,z);
        if (y>z)
          swap(y,z);
    s=min(k / 3,x-1);
    a=s;
    k-=s;
    s=0;
    s=min(k / 2,y-1);
    b=s;
    k-=s;
    s=min(k,z-1);
    cout<<(a+1)*(b+1)*(s+1);
    return 0;
}
