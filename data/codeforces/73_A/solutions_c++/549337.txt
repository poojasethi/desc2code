#include <iostream>

using namespace std;

int main(int argc, char* argv[])
{
	int s[3], k, d[3];
	cin >> s[0] >> s[1] >> s[2] >> k;

	d[0] = d[1] = d[2] = 1;

	int i = 0;
	while (k != 0)
	{
		if (d[i] < s[i])
		{
			d[i]++;
			k--;
		}
		i = (i + 1) % 3;

		if (d[0] + d[1] + d[2] == s[0] + s[1] + s[2])
			break;
	}

	cout << (long long)(d[0]) * d[1] * d[2] << endl;
}
