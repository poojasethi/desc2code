#include<iostream>
#include<algorithm>
using namespace std;

int a[3], b[3], c[3], k;
long long ans;

int main() {
	cin>>a[0]>>a[1]>>a[2]>>k;
	sort(a,a+3);
	
	ans = 1LL;
	b[0] = k/3;
	c[0] = min(b[0], a[0]-1);
	ans *= c[0]+1;
	k -= c[0];
	
	b[1] = k/2;
	c[1] = min(b[1], a[1]-1);
	ans *= c[1]+1;
	k -= c[1];
	
	ans *= min(k, a[2]-1) + 1;
	cout<<ans<<endl;
	
	return 0;
}
