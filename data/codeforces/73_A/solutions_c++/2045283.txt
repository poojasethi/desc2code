#include <iostream>
#include <algorithm>

using namespace std;

int main() {
  int a[3], b[3], k;
  cin >> b[0] >> b[1] >> b[2] >> k;
  sort(b, b + 3);
  
  a[0] = min(k / 3, b[0] - 1);
  
  k -= a[0]++;
  unsigned long long x = a[0];
  
  a[1] = min(k / 2, b[1] - 1);
  
  k -= a[1]++;
  x *= a[1];
  
  a[2] = min(k, b[2] - 1);
  
  x *= ++a[2];
  
  cout << x;
  return 0;
}
