#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
    long long a[3],n;
    cin >> a[0] >> a[1] >> a[2] >> n;
    sort(a,a+3);
    a[0]=min(a[0]-1,n/3);
    n-=a[0];
    a[0]++;
    a[1]=min(a[1]-1,n/2);
    n-=a[1];
    a[1]++;
    a[2]=min(a[2]-1,n);
    a[2]++;
    n=a[0]*a[1]*a[2];
    cout << n << endl;
    return 0;
}
