#include<iostream>
#include<algorithm>
using namespace std;

int arr[3];

int main(){
	int k;
	long long f=1;
	cin>>arr[0]>>arr[1]>>arr[2]>>k;
	sort(arr,arr+3);
	for(int i=0;i<3;i++){
		if(arr[i]>1 && k/(3-i)){
			if((arr[i]-1)>k/(3-i)){
				f*=k/(3-i)+1;
				k-=k/(3-i);
			}
			else{
				f*=arr[i];
				k-=arr[i]-1;
			}
		}
	}
	cout<<f<<endl;
	return 0;
}
