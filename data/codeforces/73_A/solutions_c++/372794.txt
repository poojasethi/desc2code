#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

int x[3],n,y[3];

int main()
{
    scanf("%d%d%d%d",&x[0],&x[1],&x[2],&n);
    sort(x,x+3);
    y[0]=min(x[0]-1,n/3);
    y[1]=min(x[1]-1,(n-y[0])/2);
    y[2]=min(x[2]-1,(n-y[0]-y[1]));
    cout<<(y[0]+1)*(long long)(y[1]+1)*(long long)(y[2]+1)<<endl;
    return 0;
}
