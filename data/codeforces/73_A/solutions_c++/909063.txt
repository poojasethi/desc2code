#include <iostream>
using namespace std;
typedef long long ll;
ll x,y,z,k;
int main()
{
	ios_base::sync_with_stdio(false);
	cin>>x>>y>>z>>k;
	while ((x-1)+(y-1)+(z-1)>k)
	{
		ll t=max(x,max(y,z));
		if (t==x)
			x--;
		else if (t==y)
			y--;
		else if (t==z)
			z--;
	}
	cout<<x*y*z<<endl;
	return 0;
}
