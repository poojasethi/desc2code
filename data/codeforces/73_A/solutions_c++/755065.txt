#include <stdio.h>
#include <algorithm>
using namespace std;

int main(void) {
	int a[3], k;
	long long ans=0;
	scanf("%d%d%d%d", &a[0], &a[1], &a[2], &k);
	sort(a, a+3);
	int p = min(a[0]-1, k/3);
	int q = min(a[1]-1, (k-p)/2);
	int r = min(a[2]-1, k-p-q);
	if(r<0) r=0;
	ans = max(ans, 1LL*(p+1)*(q+1)*(r+1));
	printf("%I64d\n", ans);
	return 0;
}
