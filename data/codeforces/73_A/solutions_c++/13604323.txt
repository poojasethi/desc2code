#include<bits/stdc++.h>
using namespace std;
#define ll long long
ll minl(ll a,ll b){
    return a<b?a:b;
}
int main(){
    ll m,n,i,j,k,x,y,z;
    while(cin>>x>>y>>z>>k){
        if(x>y) swap(x,y);
        if(x>z) swap(x,z);
        if(y>z) swap(y,z);
        x=minl(k/3,x-1);
        y=minl((k-x)/2,y-1);
        z=minl(k-x-y,z-1);
        ll ans=(x+1)*(y+1)*(z+1);
        cout<<ans<<endl;
    }
}
