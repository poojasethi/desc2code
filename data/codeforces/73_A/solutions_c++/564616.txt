#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;

int main(void)
{
    long long x,y,z,k;
    scanf("%I64d%I64d%I64d%I64d", &x,&y,&z,&k);
    while ( x+y+z-3>k){
          if ( x > y && x > z) x--;
          else if (y > z) y--;
          else z--;
    }
    printf("%I64d", x*y*z);
}
