#include <cstdio>

using namespace std;

int main()
{
	__int64 x[4], k, n;
	int i, f;
	
	x[3] = -1;

	scanf("%I64d %I64d %I64d %I64d", &x[0], &x[1], &x[2], &k);

	if(x[0]+x[1]+x[2]-3>k) {
		while(k<x[0]+x[1]+x[2]-3){
			f = 3;
			for(i=0; i<3; i++)
				if(x[i]>x[f]) f = i;
			x[f]--;
		}
	}
	n = x[0]*x[1]*x[2];
	printf("%I64d\n", n);

	return 0;
}
