#include <iostream>
#include <cstdio>
using namespace std;
long long x,y,z,k,l,a,b,c,ans;
int main()
{
cin>>x>>y>>z>>k;
k=min(x+y+z-3,k);
for (a=0;a<=k && a<x;a++)
{
b=(k-a)/2;
c=k-a-b;
if (b>=y)
c+=b-y+1,b=y-1;
if (c>=z)
b+=c-z+1,c=z-1;
if (b>=y)
	b=y-1;
if (c>=z)
	c=z-1;
ans=max((a+1)*(b+1)*(c+1),ans);
}
cout<<ans;
return 0;
}
