#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

long long x[3],k;
bool u[3];
int main()
{
	cin >> x[0] >> x[1] >> x[2] >> k;
	sort(x,x+3);
	memset(u,0,sizeof(u));
	long long f=k/3,u,o;	
	long long ans;
	f=min(x[0]-1,f);
	u=(k-f)/2;
	u=min(x[1]-1,u);
	o=k-f-u;
	o=min(x[2]-1,o);
	ans=(f+1)*(u+1)*(o+1);
	cout << ans << endl;
	return 0;
}
