#include <iostream>

using namespace std;

int main()
{
	long long x,y,z,k,max,buff,xud,yud,zud;
	cin>>x;
	cin>>y;
	cin>>z;
	cin>>k;
	xud=yud=zud=1;
	while (k!=0)
	{
		if (xud<x && k>0) 
		{
			xud++;
			k--;
		}
		if (yud<y && k>0)
		{
			yud++;
			k--;
		}
		if (zud<z && k>0) 
		{
			zud++;
			k--;
		}
		if (x==xud && y==yud && z==zud) break;
	}
	max=xud*yud*zud;
	if (max<=x*y*z)
		cout<<max;
	else 
		cout<<x*y*z;
}
