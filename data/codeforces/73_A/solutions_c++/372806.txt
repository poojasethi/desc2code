#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;
int main()
{
	long long a,b,c,k,s,aa,bb,cc;
	cin>>a>>b>>c>>k;
	if ((a-1)+(b-1)+(c-1)<k)
		s=a*b*c;
	else
	{
		if (b<a)
			swap(a,b);
		if (c<a)
			swap(c,a);
		if (c<b)
			swap(b,c);
		aa=k/3;
		if (aa>a-1)
			aa=a-1;
		bb=(k-aa)/2;
		if (bb>b-1)
			bb=b-1;
		cc=k-aa-bb;
		if (cc>c-1)
			cc=c-1;
		s=(aa+1)*(bb+1)*(cc+1);
	}
	cout<<s<<endl;
	return 0;
}
