#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int V[3], A[3], k;
    cin >> V[0] >> V[1] >> V[2] >> k;
    sort( V, V + 3);
    k += 3;
    for(int i = 0; i < 3; i++){
        A[i] = min(V[i],k/(3-i));
        k -= A[i];
    }
    cout << (long long)A[0]*A[1]*A[2] << endl;
}
