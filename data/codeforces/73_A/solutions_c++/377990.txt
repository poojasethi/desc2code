#include <iostream>
#include <algorithm>
using namespace std;
typedef long long ll;
ll C[3],BR;
ll cut[3];
int main(){
  cin>>C[0]>>C[1]>>C[2]>>BR;
  sort(C,C+3);
  cut[0]=min(BR/3,C[0]-1);
  BR-=cut[0];
  cut[1]=min(BR/2,C[1]-1);
  BR-=cut[1];
  cut[2]=min(BR,C[2]-1);
  cout<<(cut[0]+1)*(cut[1]+1)*(cut[2]+1)<<'\n';
  return 0;
}
