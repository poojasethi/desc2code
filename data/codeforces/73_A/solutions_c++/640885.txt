#include <iostream>
using namespace std;
long long x, y, z, k;
int main(){
	cin >> x >> y >> z >> k;
	while( x+y+z-3 > k ){
		if( x > y && x > z ){
			x -= 1;
		}else if( y > z ){
			y -= 1;
		}else{
			z -= 1;
		}
	}
	cout << x * y * z << endl;
	return 0;
}
