#!/usr/bin/python

import sys, os
sys.setrecursionlimit(10000)

def readline():
	return sys.stdin.readline().strip()
def readrow():
	return readline().split(' ')

(x, y, z, k) = map(int, readrow())

def removeall(l, v):
	while True:
		try:
			l.remove(v)
		except ValueError:
			break

xyz = [x-1, y-1, z-1]
cut = [1, 1, 1]
while k > 0:
	removeall(xyz, 0)
	if len(xyz) == 0: break
	n = len(xyz)
	if k//n > 0:
		mi = min((min(xyz), k // n))
		xyz = map(lambda x: x-mi, xyz)
		for i in range(n):
			cut[i] += mi
		k -= mi*n
	else:
		for i in range(k%n):
			cut[i] += 1
		break
print(reduce(lambda x,y: x*y, cut))
