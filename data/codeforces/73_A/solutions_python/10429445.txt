import sys

MIN = -1000000000
x, y, z, k = [int(x) for x in sys.stdin.readline().split()]
x, y, z = sorted([x, y, z])
x1, y1, z1 = 1, 1, 1
pd = k / 3
if pd != 0:
    prevx1, prevy1, prevz1 = x1, y1, z1
    x1, y1, z1 = min(x, x1+pd), min(y, y1+pd), min(z, z1+pd)
    k -= (x1-prevx1 + y1-prevy1 + z1-prevz1)
pd = k / 2
if pd != 0:
    prevy1, prevz1 = y1, z1
    y1, z1 = min(y, y1+pd), min(z, z1+pd)
    k -= (y1-prevy1 + z1-prevz1)
z1 = min(z, z1+k)
#print(x1,y1,z1)
print(x1*y1*z1)
