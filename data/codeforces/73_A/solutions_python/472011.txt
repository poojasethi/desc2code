L = map(int,raw_input().strip().split())
x = L[0] - 1
y = L[1] - 1
z = L[2] - 1
a = min(x,y,z)
c = max(x,y,z)
b = x + y + z - a - c
k = L[3]
if(k >= a + b + c):
    print (a+1)*(b+1)*(c+1)
else:
    if 3*a <= k:
        t1 = a
    else:
        t1 = k/3
    k = k - t1
    if 2*b <= k:
        t2 = b
    else:
        t2 = k/2
    k = k - t2
    if c <= k:
        t3 = c
    else:
        t3 = k
    print (t1+1)*(t2+1)*(t3+1)
