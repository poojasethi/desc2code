x,y,z,k=map(int,raw_input().split())
x,y,z=sorted([x,y,z])
a,b,c=0,0,0
if k<=3*(x-1):
    a=k/3
    b=k/3
    c=k/3
    k-=3*(k/3)
    if k: c+=1; k-=1
    if k: b+=1; k-=1
    print (a+1)*(b+1)*(c+1)
else:
    a,b,c=x-1,x-1,x-1
    k-=3*(x-1)
    if k<=2*(y-x):
        b+=k/2
        c+=k-k/2
        print (a+1)*(b+1)*(c+1)
    else:
        b,c=y-1,y-1
        k-=2*(y-x)
        c=min(c+k,z-1)
        print (a+1)*(b+1)*(c+1)
