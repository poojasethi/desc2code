n, m, x, y = map(int, raw_input().split())
c = [0]
O = n * m
for v in [r.count('#') for r in zip(*[raw_input() for _ in range(n)])]:
    c.append(c[-1] + v)
a, b = [0], [0]
for i in range(m):
    a.append(min([b[j] + c[i + 1] - c[j] for j in range(max(i - y + 1, 0), i - x + 2)] + [O]))
    b.append(min([a[j] + n * (i - j + 1) - c[i + 1] + c[j] for j in range(max(i - y + 1, 0), i - x + 2)] + [O]))
print min(a[-1], b[-1])
