#include <iostream>
using namespace std;
int n,m,x,y,b[1010][2],c[1010][2];
char a[1010][1010];
int main(){
	cin>>n>>m>>x>>y;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++){
			if (a[i-1][j-1]=='#') b[j][1]++;
			if (a[i-1][j-1]=='.') b[j][2]++;
		}
	for(int i=1;i<=m;i++){
		b[i][1]+=b[i-1][1];
		b[i][2]+=b[i-1][2];
		c[i][1]=100000000;
		c[i][2]=100000000;
	}
	for(int i=0;i<m;i++){
		for(int j=x;j<=y;j++){
			c[i+j][1]=min(c[i+j][1],c[i][2]-b[i][1]+b[i+j][1]);
			c[i+j][2]=min(c[i+j][2],c[i][1]-b[i][2]+b[i+j][2]);
		}
	}
	cout<<min(c[m][1],c[m][2]);
}
