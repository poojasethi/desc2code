#include <iostream>
#include <algorithm>
using namespace std;
int color[1005][2];
int F[1005][2];
int main(){
	int n, m, x, y;
	cin >> n >> m >> x >> y;
	for(int i=1;i<=n;i++){
		for(int j=1;j<=m;j++){
			char c;
			cin >> c;
			if(c == '#') color[j][0]++;
			else color[j][1]++;
		}	
	}
	
	for(int i=1;i<=m;i++){
		color[i][0]+=color[i-1][0];
		color[i][1]+=color[i-1][1];
		F[i][0] = 10000000;
		F[i][1] = 10000000;
	}
	
	for(int i=0;i<=m;i++){
		for(int j=x;j<=y;j++){
			F[i+j][0] = min(F[i+j][0], F[i][1] + color[i+j][0] - color[i][0]);
			F[i+j][1] = min(F[i+j][1], F[i][0] + color[i+j][1] - color[i][1]);
		}
	}
	
	cout << min(F[m][0], F[m][1]) << endl;
}
