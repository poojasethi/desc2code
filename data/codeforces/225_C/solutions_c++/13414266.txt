#include<iostream>
using namespace std;
int n,m,i,j,k,x,y,A[1001][2],F[1001][2];
char B[1001][1001];
int main()
{
    cin>>n>>m>>x>>y;
    for(i=1;i<=n;i++)
        for(j=1;j<=m;j++){
            cin>>B[i][j];
            if(B[i][j]=='#')A[j][0]++;
            else A[j][1]++;
        }
    for(j=1;j<=m;j++)
        A[j][0]+=A[j-1][0],A[j][1]+=A[j-1][1],F[j][0]=10000000,F[j][1]=10000000;
    for(i=0;i<m;i++)
        for(j=x;j<=y;j++){
            F[i+j][0]=min(F[i+j][0],F[i][1]+A[i+j][0]-A[i][0]);
            F[i+j][1]=min(F[i+j][1],F[i][0]+A[i+j][1]-A[i][1]);
        }
    cout<<min(F[m][0],F[m][1]);
}
