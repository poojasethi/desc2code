#include<iostream>
#include<stdio.h>
#include<string>
#define INVALID 2000000000

using namespace std;

int dp[1010][3];

int main(){
	int n,m,x,y,i,j;
	string s;
	
	cin>>n>>m>>x>>y;
	
	for(i=1;i<=n;i++){
		cin>>s;
		for(j=1;j<=m;j++)
			dp[j][0]+=(s[j-1]=='#');
	}
	
	for(j=1;j<=m;j++){		
		dp[j][1]=dp[j][2]=INVALID;
		
		if(j<x)	continue;		
		
		int b=0,w=0;
		
		for(i=j;i>(j-x+1);i--)
			b+=dp[i][0],w+=(n-dp[i][0]);				
		
		for(i--;i>=max(0,j-y);i--){
			b+=dp[i+1][0];
			w+=(n-dp[i+1][0]);
			dp[j][1]=min(dp[j][1], b+dp[i][2]);
			dp[j][2]=min(dp[j][2], w+dp[i][1]);
		}
		
	}		
	
	cout<<endl<<min(dp[m][1],dp[m][2]);
	return 0;	
}
