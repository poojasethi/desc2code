#include<bits/stdc++.h>
using namespace std;
int dp[2][2004],a[2][2004]={0},i,j,n,m,x,y;
char b[1004][1004];
main()
{
    scanf("%d%d%d%d",&n,&m,&x,&y);
    for (i=1;i<=n;i++)
        for (j=1;j<=m;j++)
    {
        scanf(" %c",&b[i][j]);
        if (b[i][j]=='.') a[0][j]++;
        else a[1][j]++;
    }
    for (i=1;i<=m;i++)
        a[0][i]+=a[0][i-1],a[1][i]+=a[1][i-1],dp[0][i]=1e7,dp[1][i]=1e7;
    dp[0][0]=0;
    dp[1][0]=0;
    for (i=0;i<m;i++)
        for (j=x;j<=y;j++)
        {
                dp[0][i+j]=min(dp[0][i+j],dp[1][i]+a[0][i+j]-a[0][i]);
                dp[1][i+j]=min(dp[1][i+j],dp[0][i]+a[1][i+j]-a[1][i]);
        }
    printf("%d",min(dp[0][m],dp[1][m]));
}
