//In the Name of God

#include<iostream>

using namespace std;

const int inf=1e9+7;
int n,m,x,y,ans,a[1010],b[1010],w[1010],dp[1010][1010][2];
string s[1010];

int main()
{
  cin>>n>>m>>x>>y;
  for(int i=0;i<n;i++)
    cin>>s[i];
  for(int i=0;i<n;i++)
    for(int j=0;j<m;j++)
      a[j+1]+=(s[i][j]=='#');
  for(int i=1;i<=m;i++)
    a[i]+=a[i-1];
  w[1]=b[1]=inf;
  for(int i=1;i<=m;i++,w[i]=b[i]=inf)
    for(int j=x;j<=i && j<=y;j++)
      {
	dp[i][j][0]=w[i-j]+a[i]-a[i-j];
	dp[i][j][1]=b[i-j]+(j*n)-a[i]+a[i-j];
	b[i]=min(b[i],dp[i][j][0]);
	w[i]=min(w[i],dp[i][j][1]);
      }
  ans=min(b[m],w[m]);
  cout<<ans<<endl;
  return 0;
}
