#include <stdio.h>
#include <iostream>
#include <string>
using namespace std;
int n,m,x,y,p[1200][2],f[1200][2];
string t;
int main( ){
	cin>>n>>m>>x>>y;
	for (int i=1;i<=n;i++){
		cin>>t;
		for (int j=1;j<=m;j++)
			if (t[j-1]=='#') p[j][0]++; else p[j][1]++;
	}
	for (int j=1;j<=m;j++){
		p[j][0]+=p[j-1][0];
		p[j][1]+=p[j-1][1];
		f[j][0]=100000000;
		f[j][1]=100000000;
	}
	for (int i=0;i<m;i++)
		for (int j=x;j<=y;j++){
			if (f[i+j][0]>f[i][1]+p[i+j][0]-p[i][0])
				f[i+j][0]=f[i][1]+p[i+j][0]-p[i][0];
			if (f[i+j][1]>f[i][0]+p[i+j][1]-p[i][1])
				f[i+j][1]=f[i][0]+p[i+j][1]-p[i][1];
		}
	if (f[m][0]<f[m][1])
		cout<<f[m][0]<<endl;
	else
		cout<<f[m][1]<<endl;
	return 0;
}
