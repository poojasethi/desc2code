#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
string a[1005];
int b[1005],f[1005][2],t,maxn;
int main()
{
	maxn=111111111;
	int m,n,x,y;
	cin>>m>>n>>x>>y;
	for(int i=1;i<=m;i++)
	{
		cin>>a[i];
		a[i]=" "+a[i];
	} 
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)
		{
			if(a[j][i]=='#')
			{
				b[i]++;
			} 
		}
	} 
	for(int i=1;i<=n;i++)
	{
		t=0;
		f[i][0]=f[i][1]=maxn;
		for(int j=i;j>=1;j--)
		{
			t+=b[j];
			if(i-j+1<x) continue;
            if(i-j+1>y) break;	
			f[i][0]=min(f[i][0],f[j-1][1]+m*(i-j+1)-t);
			f[i][1]=min(f[i][1],f[j-1][0]+t);	
		} 
	}
	cout<<min(f[n][0],f[n][1]);
	return 0;
}
