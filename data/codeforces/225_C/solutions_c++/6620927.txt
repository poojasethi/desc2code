#include<bits/stdc++.h>
#define inf 2000000000
using namespace std;
int m,n,lt,mt;
char arr[1004][1004];
int coun[1005];
int dp[1005][2][1003];
int f(int x,int k,int num){
	if(x==n+1){
		if(num>lt || num==1) return 0; 
		return inf;
	}
	int krk = (k)? m-coun[x]: coun[x]; 
	int &ret=dp[x][k][num];
	if(ret!=-1) return ret;
	ret=inf;
	if(num>=lt)
	 	ret=min(ret,f(x+1,!k,1)+krk);
	if(num<mt)
		ret=min(ret,f(x+1,k,num+1)+krk);
	return ret;
}
int main(){
	int i,j;
	cin>>m>>n>>lt>>mt;
	for(i=1;i<=m;i++)
		for(j=1;j<=n;j++) 
			cin>>arr[i][j];
	for(i=1;i<=n;i++)
		for(j=1;j<=m;j++)
			coun[i]+=(arr[j][i]=='#');
	memset(dp,-1,sizeof dp);
	cout<<min(f(1,0,1),f(1,1,1));
	return 0;
}
