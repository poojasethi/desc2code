#include<iostream>
#include<algorithm>
#include<cstring>
using namespace std;

const int N = 1020;

int n, m, x, y, b[1020], w[2010], d[1020][2];
char xx[1020][1020];

int main() {
	int i, j;
	
	cin >> n >> m >> x >> y;
	
	for(i = 1; i<=n; ++i) {
		
		for(j = 1; j<=m; ++j) {
			d[j][0] = d[j][1] = 10000000;
			cin >> xx[i][j];
			if(xx[i][j] == '.')
				b[j]++;
			else
				w[j]++;
		}
	}
	for(i = 1; i<=m; ++i) {
		b[i] += b[i-1];
		w[i] += w[i-1];
	}
	
	for(i = x; i<=m; ++i) {
		
		for(j = i - x; j>=0 && j>=i - y; --j) {
			d[i][0] = min(d[i][0], d[j][1] + w[i] - w[j]);
			d[i][1] = min(d[i][1], d[j][0] + b[i] - b[j]);
		}
	}
	
	cout << min(d[m][0], d[m][1]);
	
	return 0;
}
