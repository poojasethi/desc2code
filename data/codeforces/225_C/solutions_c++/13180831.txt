#include<bits/stdc++.h>
using namespace std;
string s[2000];
int dp[2000][2]={0},a[2000][2]={0};
int main ()
{
    int x,y,l,r;cin >> x >> y >> l >> r;
    for(int i=0;i<x;i++)cin >> s[i];
    for(int i=0;i<x;i++)
        for(int j=0;j<y;j++)
            if(s[i][j]=='.')a[j+1][1]++;
            else a[j+1][0]++;
    for(int i=1;i<=y;i++)
        a[i][0]+=a[i-1][0],a[i][1]+=a[i-1][1],dp[i][0]=100000000,dp[i][1]=100000000;
    for(int i=0;i<y;i++)
        for(int j=l;j<=r;j++)
        {
            dp[i+j][0]=min(dp[i+j][0],dp[i][1]+a[i+j][0]-a[i][0]);
            dp[i+j][1]=min(dp[i+j][1],dp[i][0]+a[i+j][1]-a[i][1]);
        }
    cout << min(dp[y][0],dp[y][1])<<endl;
}
