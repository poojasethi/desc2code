#include<iostream>
#include<cstring>
#include<cmath>
#include<stdio.h>
#include<queue>

using namespace std;
int main()
{
	int n=0,m,ans=0,x,y,a[1010]={0},b[1010]={0},f[2010][3],g[1010][3];
	string s;
	cin >> n >> m >> x >> y;
	for (int i=1;i<=n;i++){
		cin >> s;
		for (int j=0;j<m;j++) if (s[j]=='.') a[j+1]++;
	}
	int t=0;
	for (int i=1;i<=m;i++){
		t=t+a[i]; b[i]=t; f[i][1]=1000001; f[i][2]=10000001;
	}
	f[0][1]=0; f[0][2]=0;
	for (int i=0;i<=m;i++){
		for (int j=x;j<=y;j++){
			f[i+j][1]=min(f[i+j][1],f[i][2]+b[i+j]-b[i]);
			f[i+j][2]=min(f[i+j][2],f[i][1]-b[i+j]+b[i]+j*n);
		}
	}
	cout << min(f[m][1],f[m][2]) << endl;
}
