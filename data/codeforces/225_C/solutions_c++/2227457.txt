#include <iostream>
using namespace std;

int main()
{
	int n,m,x,y;
	cin>>n>>m>>x>>y;
	int wcnt[1010]={0},bcnt[1010]={0};
	char buf[1010];
	for(int i=0;i<n;i++)
	{
		cin>>buf;
		for(int j=0;j<m;j++)
			if(buf[j]=='#')bcnt[j+1]++;
			else wcnt[j+1]++;
	}
	for(int i=1;i<=m;i++)
	{
		bcnt[i]+=bcnt[i-1];
		wcnt[i]+=wcnt[i-1];
	}
	int dp[2][1010]={0};
	for(int j=1;j<=m;j++)
		for(int i=0;i<2;i++)
		{
			int Min=1<<30;
			for(int k=x;k<=y&&j-k>=0;k++)
			{
				int tmp;
				if(i==0)
					tmp=dp[1][j-k]+bcnt[j]-bcnt[j-k];
				else
					tmp=dp[0][j-k]+wcnt[j]-wcnt[j-k];
				if(tmp<Min)
					Min=tmp;
			}
			dp[i][j]=Min;
		}
	if(dp[0][m]>dp[1][m])
		dp[0][m]=dp[1][m];
	cout<<dp[0][m]<<endl;
	return 0;
}
