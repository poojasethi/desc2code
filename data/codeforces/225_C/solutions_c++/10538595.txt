#include <iostream>
#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	ll n,m,x,y;
	char s;
	cin>>n>>m>>x>>y;
	ll arr[m+1];
	memset(arr,0,sizeof(arr));
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<m;j++)
		{
			cin>>s;
			if(s=='.')
			{
				arr[j+1]++;
			}
		}
		getchar();
	}
	for(int j=1;j<=m;j++)
	{
		arr[j]=arr[j]+arr[j-1];
	}
	ll dp[m+1][2];
	for(int i=0;i<m;i++)
	{
		dp[i][0]=INT_MAX;
		dp[i][1]=INT_MAX;
	}
	dp[0][1]=0;
	dp[0][0]=0;
	for(int i=1;i<m+1;i++)
	{
		for(int j=x;j<=y;j++)
		{
			if(i>=j)
			{
			dp[i][1]=min(dp[i][1],(dp[i-j][0]+arr[i]-arr[i-j]));
			dp[i][0]=min(dp[i][0],(dp[i-j][1]+n*j-arr[i]+arr[i-j]));
			}
			
	}
}
cout<<min(dp[m][0],dp[m][1]);

}
