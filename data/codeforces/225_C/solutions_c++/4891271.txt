#include<stdio.h>
#define MIN(x,y) ( x>=y ? y: x )
int main(){
	int n,m,x,y;
	while( scanf("%d%d%d%d",&n,&m,&x,&y)==4 ){
		int A[1001]={0},B[1001]={0};
		int DP[2][1001]={0};
		char S[1001];
		for(int i=1;i<=n;i++){
			scanf("%s",S);
			for(int j=0;j<m;j++)
				if( S[j]=='#' )
					A[j+1]++;
				else
					B[j+1]++;
		}
		for(int i=0;i<=m;i++)
			DP[0][i]=DP[1][i]=2e9;
		DP[0][0]=DP[1][0]=0;
		
		for(int i=1;i<=m;i++)
			for(int j=x;j<=y&&i>=j;j++){
				int tmp1,tmp2;
				tmp1=tmp2=0;
				for(int k=i-j+1;k<=i&&k>0;k++)
					tmp1+=A[k],tmp2+=B[k];
				DP[0][i]=MIN(DP[0][i],DP[1][i-j]+tmp1);
				DP[1][i]=MIN(DP[1][i],DP[0][i-j]+tmp2);
			}
		printf("%d\n",MIN(DP[0][m],DP[1][m]));
	}
	return 0;
}
