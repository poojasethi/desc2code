#include<iostream>
using namespace std;
#define INF 1e7
char map[1050][1050];
int num[1050];
int dp[1050][3];
int pre[1050];
int main(){
	int n,m,x,y;
	while(cin>>n>>m>>x>>y){
		for(int i=1;i<=n;++i){
			for(int j=0;j<m;++j)
				cin>>map[i][j];
		}
		for(int j=1;j<=m;++j)
			for(int i=1;i<=n;++i)
				if(map[i][j-1]=='#')num[j]++;
		for(int i=1;i<=m;++i){
			pre[i]=pre[i-1]+num[i];
		}
		for(int i=0;i<=m;++i){
			dp[i][0]=INF;
			dp[i][1]=INF;
		}dp[0][0]=0;dp[0][1]=0;
		for(int i=x;i<=m;++i){
			for(int k=x;i-k>=0&&k<=y;++k){
				int j=i-k;
				dp[i][0]=min(dp[i][0],dp[j][1]+(n*k)-(pre[i]-pre[j]));
				dp[i][1]=min(dp[i][1],dp[j][0]+pre[i]-pre[j]);
			}
		}
		cout<<min(dp[m][0],dp[m][1])<<endl;
	}
	return 0;
}
