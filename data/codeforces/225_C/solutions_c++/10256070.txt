#include<bits/stdc++.h>
using namespace std;

const int N=1010;

int b[N];
int dp[N][2];
string s[N];

int main() {
  int n,m,x,y;
  scanf("%d%d%d%d",&n,&m,&x,&y);
  for(int i=0;i<n;i++) {
    cin>>s[i];
    for(int j=0;j<m;j++) {
      if(s[i][j]=='#') {
        b[j+1]++;
      }
    }
  }
  for(int i=1;i<m;i++) {
    b[i+1]+=b[i];
  }
  for(int i=1;i<=m;i++) {
    dp[i][0]=dp[i][1]=m*n;
  }
  for(int i=0;i<m;i++) {
    for(int j=x;j<=y&&i+j<=m;j++) {
      dp[i+j][0]=min(dp[i+j][0],dp[i][1]+(b[i+j]-b[i]));
      dp[i+j][1]=min(dp[i+j][1],dp[i][0]+j*n-(b[i+j]-b[i]));
    }
  }
  int a=min(dp[m][0],dp[m][1]);
  printf("%d\012",a);
  return 0;
}
