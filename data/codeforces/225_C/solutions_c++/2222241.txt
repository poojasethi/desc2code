#include<iostream>
using namespace std;
char c[1001][1001];
int n,m,i,j,k,x,y,a[1001][2],D[1001][2];
main()
{
    cin>>n>>m>>x>>y;
    for (i=1;i<=n;i++)
    for (j=1;j<=m;j++)
    cin>>c[i][j];
    for (i=1;i<=n;i++)
    for (j=1;j<=m;j++)
    if (c[i][j]=='#') a[j][0]++; else a[j][1]++;
    for (j=1;j<=m;j++)
    {a[j][0]+=a[j-1][0];a[j][1]+=a[j-1][1];}
    for (i=1;i<=m;i++) {D[i][0]=10000000;D[i][1]=10000000;}
    for (i=0;i<m;i++)
    for (j=x;j<=y;j++)
    {
        D[i+j][0]=min(D[i+j][0],D[i][1]+a[i+j][0]-a[i][0]);
        D[i+j][1]=min(D[i+j][1],D[i][0]+a[i+j][1]-a[i][1]);
    }
    cout<<min(D[m][0],D[m][1])<<endl;
}
