#include<iostream>
using namespace std;
int n,m,x,y,gs[1001][2],s,d[1001][2];
char a[1001][1001];
int main()
{
	cin>>n>>m>>x>>y;
	for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=m;j++)cin>>a[i][j];
	}
	for(int j=1;j<=m;j++)
	{
		for(int i=1;i<=n;i++)if(a[i][j]=='#')gs[j][1]++;else gs[j][0]++;
		gs[j][1]+=gs[j-1][1];gs[j][0]+=gs[j-1][0];
	}
	for(int j=1;j<=m;j++)d[j][0]=1000000,d[j][1]=1000000;
	for(int i=0;i<m;i++)
	{
		for(int j=x;j<=y;j++)
		{
			if(i+j>m)continue;
			d[i+j][0]=min(d[i+j][0],d[i][1]+gs[i+j][0]-gs[i][0]);
			d[i+j][1]=min(d[i+j][1],d[i][0]+gs[i+j][1]-gs[i][1]);
		}
	}
	cout<<min(d[m][0],d[m][1]);
	return 0;
}
