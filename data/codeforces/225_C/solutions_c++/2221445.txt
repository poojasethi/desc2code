#include <algorithm>
#include <cstdio>
#include <cstring>

using namespace std;

char g[1024][1024];
int t[1024][2];
int main(void) {
  int n,m,x,y,b[2][1024]={{0}},v=123456789;
  scanf("%d %d %d %d",&n,&m,&x,&y);
  for(int i=0;i<n;++i)scanf("%s",g[i]);
  for(int i=0;i<m;++i)for(int j=0;j<n;++j)++b['#'==g[j][i]][i+1];
  for(int i=0;i<2;++i)for(int j=0;j<m;++j)b[i][j+1]+=b[i][j];
  memset(t,0x3f,sizeof(t)),t[0][0]=t[0][1]=0;
  for(int i=0;i<m;++i)
    for(int j=x;j<=y&&i+j<=m;++j)
      t[i+j][1]=min(t[i+j][1],t[i][0]+(b[0][i+j]-b[0][i])),
      t[i+j][0]=min(t[i+j][0],t[i][1]+(b[1][i+j]-b[1][i]));
  printf("%d\n",min(t[m][0],t[m][1]));
  return 0;
}
