#include <cstdio>
#include <algorithm>
using namespace std;
int main()
{
	int n,m,x,y,black[1+1000]={0},dp[1+1000][2];
	char image[1000][1000+1];
	scanf("%d%d%d%d",&n,&m,&x,&y);
	for(int i=0;i<n;i++)
	{
		scanf("%s",image[i]);
		for(int j=0;j<m;j++)
			if(image[i][j]=='#')
				black[j+1]++;
	}
	for(int i=1;i<m;i++)
		black[i+1]+=black[i];
	for(int i=1;i<=m;i++)
		dp[i][1]=dp[i][0]=m*n;
	dp[0][1]=dp[0][0]=0;
	for(int i=0;i<m;i++)
		for(int j=x;j<=y&&i+j<=m;j++)
		{
			dp[i+j][0]=min(dp[i+j][0],dp[i][1]+(black[i+j]-black[i]));
			dp[i+j][1]=min(dp[i+j][1],dp[i][0]+j*n-(black[i+j]-black[i]));
		}
	printf("%d\n",min(dp[m][0],dp[m][1]));
	return 0;
}
