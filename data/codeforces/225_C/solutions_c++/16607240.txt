#include<bits/stdc++.h>
using namespace std;
int n,m,i,j,k,x,y,a[1001][2],f[1001][2];
char b[1001][1001];
int main(){
 cin>>n>>m>>x>>y;
 for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			cin>>b[i][j];
			if(b[i][j]=='#')a[j][0]++;
			else a[j][1]++;
		    }
    }
    for(j=1;j<=m;j++){
	     	a[j][0]+=a[j-1][0];
		     a[j][1]+=a[j-1][1];
		     f[j][0]=10000000;
		     f[j][1]=10000000;
    }
 for(i=0;i<m;i++){
		for(j=x;j<=y;j++){
			f[i+j][0]=min(f[i+j][0],f[i][1]+a[i+j][0]-a[i][0]);
			f[i+j][1]=min(f[i+j][1],f[i][0]+a[i+j][1]-a[i][1]);
		}
 }
	cout<<min(f[m][0],f[m][1])<<endl;
}
