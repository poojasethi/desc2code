#include <iostream>
#include <algorithm>
using namespace std;
const int maxn=1001;
int b[maxn];
int dp[maxn][2];
string c[maxn];
int main()
{
	int n,m,x,y;
	cin>>n>>m>>x>>y;
	for(int i=0;i<n;i++)
	{
		cin>>c[i];
		for(int j=0;j<m;j++)
			if(c[i][j]=='#')
				b[j+1]++;
	}
	for(int i=1;i<m;i++)
		b[i+1]+=b[i];
	for(int i=1;i<=m;i++)
		dp[i][1]=dp[i][0]=m*n;
	dp[0][1]=dp[0][0]=0;
	for(int i=0;i<m;i++)
		for(int j=x;j<=y && i+j<=m;j++)
		{
			dp[i+j][0]=min(dp[i+j][0],dp[i][1]+(b[i+j]-b[i]));
			dp[i+j][1]=min(dp[i+j][1],dp[i][0]+j*n-(b[i+j]-b[i]));
		}
	int a=min(dp[m][0],dp[m][1]);
	cout<<a;
	return 0;
}
