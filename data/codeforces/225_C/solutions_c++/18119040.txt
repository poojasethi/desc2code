
#include<bits/stdc++.h>
using namespace std;
const int MAX=1009;
int n,m,x,y,dp[MAX][2],sum[MAX];
int main()
{
	ios_base::sync_with_stdio(false),cin.tie(0),cout.tie(0);
	cin>>n>>m>>x>>y;
	for (int i=0;i<n;i++)
	{
		string s;
		cin>>s;
		for (int j=0;j<m;j++) if (s[j]=='#') sum[j]++;
	}
	for (int i=0;i<m;i++)
	{
		dp[i][0]=dp[i][1]=1e9;
		int s=0;
		for (int j=i;j>=0;j--) 
		{
			s+=sum[j];
			if (i-j+1>=x && i-j+1<=y)
				dp[i][0]=min(dp[i][0],(j?dp[j-1][1]:0)+s),dp[i][1]=min(dp[i][1],(j?dp[j-1][0]:0)+((i-j+1)*n-s));
		}
	}
	cout<<min(dp[m-1][0],dp[m-1][1]);
}
