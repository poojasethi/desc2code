#include <stdio.h>
const int MAXN = 1100, MAXSUM = 10100, MOD = 1e9+7;

int dp[MAXN][2*MAXSUM];

int main(){
    int n, a, resp = 0;

    scanf("%d", &n);
    dp[0][10000] = 1;
    for(int i = 1; i <= n; i++){
        scanf("%d", &a);
        for(int k = 0; k <= 20000; k++){
            dp[i][k] = (dp[i-1][k+a]+dp[i-1][k-a])%MOD;
        }
        resp = (resp+dp[i][10000]++)%MOD;
    }
    printf("%d\n", resp);

    return 0;
}
