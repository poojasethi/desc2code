#include <cstdio>
int n, a, res;
int d[1005][20005];
const int o = 10000;
const int mod = 1000000007;
int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) {
		scanf("%d", &a);
		for (int j = 0; j <= 2 * o; j++) {
			if (j - a >= 0) d[i][j] = (d[i][j] + d[i - 1][j - a]) % mod;
			if (j + a <= 2 * o) d[i][j] = (d[i][j] + d[i - 1][j + a]) % mod;
		}
		d[i][o + a] = (d[i][o + a] + 1) % mod;
		d[i][o - a] = (d[i][o - a] + 1) % mod;
		res = (res + d[i][o]) % mod;
	}
	printf("%d", res);
}
