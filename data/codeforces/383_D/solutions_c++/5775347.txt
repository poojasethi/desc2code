#include <cstdio>
#include <memory.h>
using namespace std;

int n, x, ans, dp[21000], dp1[21000];

void up(int &x, int y) {
	x = (x + y) % 1000000007;
}

int main() {
	scanf("%d", &n);
	while (n--) {
		scanf("%d", &x);
		memset(dp, 0, sizeof dp);
		up(dp1[10000], 1);
		for (int i = 0; i <= 20000 - x; i++)	up(dp[i + x], dp1[i]);
		for (int i = x; i <= 20000; i++)	up(dp[i - x], dp1[i]);
		up(ans, dp[10000]);
		memcpy(dp1, dp, sizeof dp);
	}
	printf("%d\n", ans);
}
