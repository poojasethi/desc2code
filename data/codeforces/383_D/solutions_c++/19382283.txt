#include<cstdio>
#include<algorithm>
#define N 10000
#define MOD 1000000007
using namespace std;
long long a[N],f[1005][N*2+5],n,Ans;
int main(){
	scanf("%d",&n);f[0][N]=1;
	for(int i=1;i<=n;i++)
		scanf("%d",&a[i]),f[i][N]=1;
	for(int i=1;i<=n;i++)
		for(int j=0;j<=2*N;j++){
			if(j+a[i]<=2*N) f[i][j+a[i]]=(f[i][j+a[i]]+f[i-1][j])%MOD;
			if(j-a[i]>=0) f[i][j-a[i]]=(f[i][j-a[i]]+f[i-1][j])%MOD;
		}
	for(int i=1;i<=n;i++)
		Ans=(Ans+f[i][N]-1+MOD)%MOD;
	printf("%I64d\n",Ans);
}
