#include <stdio.h>

const int mod=1000000007;

int n,dp[30000][2];

int main(){
	int at,resp=0;
	scanf("%d %d",&n,&at);
	dp[15000+at][0]=1;
	dp[15000-at][0]=1;
	for(int i=1;i<n;i++){
		scanf("%d",&at);
		for(int j=at;j+at<30000;j++){
			dp[j][i%2]=(dp[j-at][(i+1)%2]+dp[j+at][(i+1)%2])%mod;
		}
		dp[15000+at][i%2]=(dp[15000+at][i%2]+1)%mod;
		dp[15000-at][i%2]=(dp[15000-at][i%2]+1)%mod;
		resp=(resp+dp[15000][i%2])%mod;
	}
	printf("%d\n",resp);
	return 0;
}
