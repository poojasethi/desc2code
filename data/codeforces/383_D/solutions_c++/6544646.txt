#include<cstdio>
#include<cstring>
#include<algorithm>
#define mod 1000000007
#define Mn 1050
#define M 10000
using namespace std;
int n,ans;
int F[M*3],G[M*3],i,j,x,*f=F,*g=G;
void inc(int &a,int b) {a+=b; if (a>=mod) a-=mod;}
int main()
{
	scanf("%d",&n);
	f[M]=1;
	for(i=1;i<=n;i++)
		{
			scanf("%d",&x);
			for(j=0;j<=2*M;j++)
				if (f[j]) inc(g[j+x],f[j]),inc(g[j-x],f[j]),f[j]=0;
			inc(ans,g[M]); swap(f,g); f[M]++;
		}
	printf("%d",ans);
	return 0;
}
