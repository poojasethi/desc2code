#include <iostream>
using namespace std;
int a[1000],f[1000][20001];
int main()
{
	int n,i,j,ans=0;
	cin>>n;
	for (i=0;i<n;i++) cin>>a[i];
	f[0][10000-a[0]]=f[0][10000+a[0]]=1;
	for (i=1;i<n;i++)
	{
		f[i-1][10000]=(f[i-1][10000]+1)%1000000007;
		for (j=0;j<=20000;j++)
			f[i][j]=((j-a[i]>=0?f[i-1][j-a[i]]:0)+(j+a[i]<=20000?f[i-1][j+a[i]]:0))%1000000007;
	}
	for (i=0;i<n-1;i++) ans=(ans+f[i][10000]-1)%1000000007;
	ans=(ans+f[n-1][10000])%1000000007;
	cout<<ans<<endl;
	return 0;
}
