#include<bits/stdc++.h>
using namespace std;

const int mod=1e9+7;

int ans,n,i;
int A[1005];
int DP[1005][30005];

int f(int x,int top){

	int &r=DP[x][top+10000];

	if(r!=-1)
		return r;

	r=0;

	r += (!top);

	if(x==n+1)
		return r;

	r += f(x+1, top + A[x]);
	r += f(x+1, top - A[x]);

	return r%=mod;
}

int main(){
	
	scanf("%d",&n);

	for(i=1 ; i<=n ; i++)
		scanf("%d",A+i);

	memset(DP,-1,sizeof DP);

	for(i=1 ; i<=n ; i++){		
	
		ans+=f(i,0);
		
		ans %= mod;
	}

	cout<<ans-n;
	
	return 0;
}
