#include <stdio.h>
#define mod 1000000007;
int i, j, dp[2][22005], ans;
int n, a, pre;
int main() {
    scanf("%d", &n);
    for (i = 0; n--; i = pre) {
        scanf("%d", &a); pre = !i;
        for (j = 1000; j <= 22000; j++)
            dp[i][j] = (dp[pre][j - a] + dp[pre][j + a]) % mod;
        dp[i][11000 + a]++; dp[i][11000 - a]++;
        ans = (ans + dp[i][11000]) % mod;
    }
    printf("%d\n", ans);
    return 0;
}
