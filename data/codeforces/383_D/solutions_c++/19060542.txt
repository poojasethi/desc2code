#include "cstdio"
#include "algorithm"
#include "math.h"
#include "iostream"
using namespace std;
long long dp[2][22001];
const int mod=1000000007;
int main(int argc, char* argv[])
{
	int n,now=1,a;
	long long ans=0;
	scanf("%d",&n);
	while(n--)
	{
		now=1-now;
		scanf("%d",&a);
		for(int i=1000;i<=21000;i++)
			dp[now][i]=(dp[1-now][i+a]+dp[1-now][i-a])%mod;
		dp[now][11000-a]++;
		dp[now][11000+a]++;
		ans=(ans+dp[now][11000])%mod;
	}
	printf("%lld\n",ans);
	return 0;
}
