#include<bits/stdc++.h>

using namespace std;

long long M[1003][30004],A[1003],n,i,asd,md = 1000000007;

long long f(int i, int t){

	long long &r = M[i][t+10000];

	if(r != -1)
		return r;

    r = (!t);
	if(i>n) return r;

	r = (r + f(i+1,t+A[i]))%md;
	r = (r + f(i+1,t-A[i]))%md;

	return r%=md;
}

int main(){

	cin >> n;

	for(i=1;i<=n;i++)
		cin >> A[i]; 

	memset(M , -1 , sizeof M);

    for(i=1;i<=n;i++)
		asd = (asd + f(i,0)) % md;

	cout << (asd-n+md)%md <<endl;
	return 0;
}
