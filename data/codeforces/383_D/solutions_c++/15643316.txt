#include<cstdio>
#include<cstdlib>
#include<iostream>
using namespace std;
const long long Mod=1000000007;
long long dp[2][21010],ans=0;
   long long n,a;
int main()
{
   
   scanf("%I64d",&n);
   for(int i=0;n--;i^=1)
   {
     scanf("%I64d",&a);
     for(int j=1000;j<=21000;j++)
     {
       dp[i][j]=(dp[!i][j-a]+dp[!i][j+a])%Mod;
     }
     dp[i][11000+a]++;dp[i][11000-a]++;
     ans=(dp[i][11000]+ans)%Mod;
   }
   cout<<ans;return 0;
   
}
