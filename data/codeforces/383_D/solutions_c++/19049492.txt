#include <cstdio>
const int st=1000,base=11000,en=21000,mod=1e9+7;
long long ans,dp[2][22010];
int n,x;
int main(){
    scanf("%d",&n);
    for(int i=0;n--;i^=1){
        scanf("%d",&x);
        for(int j=st;j<=en;j++)dp[i][j]=(dp[!i][j-x]+dp[!i][j+x])%mod;
        dp[i][base+x]++;dp[i][base-x]++;
        ans=(ans+dp[i][base])%mod;
    }return printf("%I64d\n",ans),0;
}
