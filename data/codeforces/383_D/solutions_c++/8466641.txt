#include<cstdio>
#include<cstring>
#define LL long long
#define FOR(a,b,c) for(int a=b;a<=c;a++)
int a,N,A[1005],mod=1000000007;
LL dp[1005][20005];
LL F(int x,int c){
	if(x == N + 1) return !c;
	LL &r = dp[x][c+10000];
	if(r!=-1) return r;
	return r = (F(x+1,c-A[x]) + F(x+1,c+A[x]) + !c)%mod;
}
main()
{
	scanf("%d",&N);
	FOR(i,1,N)	scanf("%d",&A[i]);
	memset(dp,-1,sizeof dp);
	FOR(i,1,N){
		a+=F(i,0);
		a %= mod;
	}
	printf("%d\n",a-N);
}
