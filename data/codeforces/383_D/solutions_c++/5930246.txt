#include<cstdio>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=0;i<n;++i)
const int M=10005,P=1000000007;
int n,x,ans,F[M*2],G[M*2],*f=F,*g=G;
void add(int &x,int y){x+=y; if (x>=P) x-=P;}
int main()
{
	scanf("%d",&n),f[M]=1;
	rep(i,n){
		scanf("%d",&x);
		rep(i,M*2) if (f[i])
			add(g[i+x],f[i]),add(g[i-x],f[i]),f[i]=0;
		add(ans,g[M]),add(g[M],1),swap(f,g);
	}
	printf("%d\n",ans);
	return 0;
}
