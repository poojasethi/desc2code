#include<stdio.h>
const int mod=1000000007;
int f[2][20001]={0};
int main()
{
	int n,i,j=0,a,ans=0;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a);
		for(j=a;j<=20000-a;j++)
			f[i&1][j]=(f[~i&1][j-a]+f[~i&1][j+a])%mod;
		f[i&1][10000-a]++;
		f[i&1][10000+a]++;
		ans=(ans+f[i&1][10000])%mod;
	}
	printf("%d",ans);
	return 0;
}
