#include <iostream>
using namespace std;

int n, a;
long long ans, d[2][22001];

int main() {
	cin >> n;
	for (int f = 0; n--; f ^= 1) {
		cin >> a;
		for (int j = 1000; j <= 21000; ++j)
			d[f][j] = (d[!f][j - a] + d[!f][j + a]) % 1000000007;
		++d[f][11000 + a];
		++d[f][11000 - a];
		ans = (ans + d[f][11000]) % 1000000007;
	}
	cout << ans << endl;
}
