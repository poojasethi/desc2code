#include <stdio.h>
#include <string.h>
#include <stdlib.h>

long long ans, dp[2][22222];
int n, a;

int main(){
    int i, j;
    scanf("%d", &n);
    for(i = 0; n > 0; i ^= 1, n--){
        scanf("%d", &a);
        for(j = 1000; j <= 21000; j++){
            dp[i][j] = (dp[!i][j - a] + dp[!i][j + a]) % 1000000007;
        }
        dp[i][11000 - a]++;
        dp[i][11000 + a]++;
        ans = (ans + dp[i][11000]) % 1000000007;
    }
    printf("%d\n", ans);
    return 0;
}
