#include<stdio.h>

long long a[2][10001];

int main()
{
	int s=0,e=1,i;
	int n,tmp,ans=0;
    scanf("%d",&n);
    for(i=0;i<10001;i++){
    	a[0][i]=0;
    	a[1][i]=0;
	}
    while(n--)
    {
        scanf("%d",&tmp);
        for (i=0;i<10001;i++)
            	a[s][i]=(a[e][(i>tmp)?i-tmp:tmp-i]+a[e][i+tmp])%1000000007;
        a[s][tmp]++;
        ans=(ans+a[s][0])%1000000007;
        e=s;
        s=(s+1)%2;
    }
    printf("%d\n",ans);
    return 0;
}
