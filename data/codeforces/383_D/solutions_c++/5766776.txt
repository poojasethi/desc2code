#include <iostream>

using namespace std;

const int MOD = 1000 * 1000 * 1000 + 7, MAXN = 1010, SHIFT = 11010;
int a[MAXN], dp[MAXN][2 * SHIFT], n, sum, ans;

int main()
{
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i], sum += a[i];
	dp[0][SHIFT] = 1;
	for (int i = 1; i <= n; i++)
	{
		for (int j = -sum; j <= sum; j++)
			dp[i][j + SHIFT] = (dp[i - 1][j + SHIFT - a[i - 1]] + dp[i - 1][j + SHIFT + a[i - 1]]) % MOD;
		ans = (ans + dp[i][SHIFT]) % MOD;
		dp[i][SHIFT]++;
	}
	cout << ans << endl;
	return 0;
}
