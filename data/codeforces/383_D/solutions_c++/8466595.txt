#include<bits/stdc++.h>
#define ll long long
#define FOR(a,b,c) for(int a=b;a<=c;a++)
using namespace std;
int N,M,sum,mod=1000000007;
int A[1005];
ll dp[1005][20005];
ll F(int x,int c){
	if(x == N + 1) return !c;
	ll &r = dp[x][c+10000];
	if(r!=-1) return r;
	return r = (F(x+1,c-A[x]) + F(x+1,c+A[x]) + !c)%mod;
}
main()
{
	ll a(0);
	scanf("%d",&N);
	FOR(i,1,N)	scanf("%d",&A[i]);
	memset(dp,-1,sizeof dp);
	FOR(i,1,N){
		a+=F(i,0);
		a %= mod;
	}
	printf("%d\n",a-N);
}
