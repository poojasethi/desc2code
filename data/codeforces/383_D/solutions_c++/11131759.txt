#include <bits/stdc++.h>
using namespace std;
const int maxn = 11111, N = 1111, mod = 1e9+7;
int n, a[maxn], dp[N][3*maxn], ans;
int main() {
    int n; cin >> n;
    for(int i = 1; i <= n; i++)
        cin >> a[i];
    dp[0][maxn] = 1;
    for(int i = 1; i <= n; i++) {
        for (int j = -maxn+1; j < maxn; j++)
            dp[i][j+maxn] = (dp[i-1][j+maxn+a[i]] + dp[i-1][j+maxn-a[i]]) % mod;
        ans = (ans + dp[i][maxn]++)%mod;
    }
    cout << ans;
}
