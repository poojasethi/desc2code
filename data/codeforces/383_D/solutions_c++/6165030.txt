#include <cstdio>
const int N = 1000,M = 11000,p = 1000000007;
long long ans,f[2][2*M+1];
int n,k,x;
int main() {
    scanf("%I64d",&n);
    while (n--) {
        scanf("%d",&x);
        for (int j = N;j+N <= 2*M;j++)
            f[k][j] = (f[k^1][j-x]+f[k^1][j+x])%p;
        f[k][M+x]++;
        f[k][M-x]++;
        (ans += f[k][M]) %= p;
        k ^= 1;
    }
    printf("%I64d\n",ans);
    return 0;
}
