#include<iostream>
using namespace std;

int dp[2010][30010];
int n, k=0;
int main()
{
	
	dp[0][10000]=1;
	cin>>n;
	for(int i=1; i<=n; i++)
	{
		int a;
		cin>>a;
		
		k=(k+dp[i-1][10000])%1000000007;
		dp[i][10000]=1;
		for(int j=0; j<20000; j++)
		{
			if(j-a>=0)
			dp[i][j-a]= (dp[i][j-a] + dp[i-1][j])%1000000007;
			
			if(j+a<=20000)
			dp[i][j+a]= (dp[i][j+a] + dp[i-1][j])%1000000007;
		}
		
	}
	
	cout<<(dp[n][10000]+k-n-1)%1000000007;
}
