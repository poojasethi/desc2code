#include <iostream>

using namespace std;

int dp[2000][30000]={0};

main()
{
	int N;
	int sum=0;
	cin >> N;
	dp[0][10000]=1;
	for (int i=1;i<=N;i++)
	{
		int a;
		cin >> a;
		sum=(sum+dp[i-1][10000])%1000000007;
		dp[i][10000]=1;
		for (int j=0;j<=20000;j++)
		{
			if (j-a>=0) dp[i][j-a]=(dp[i][j-a]+dp[i-1][j])%1000000007;
			if (j+a<=20000) dp[i][j+a]=(dp[i][j+a]+dp[i-1][j])%1000000007;
		}
	}
	cout << (dp[N][10000]+sum-N-1)%1000000007 << endl;
}
