#include<cstdio>
const int maxn=20000+5000;
const int mod=1000000007;
long long d[2][maxn];
int main()
{
	int n,a;
	long long ans=0;
	scanf("%d",&n);
	for (int i=0;n--;i^=1)
	{
		scanf("%d",&a);
		for (int j=1000;j<=21000;j++)
			d[i][j]=(d[!i][j+a]+d[!i][j-a])%mod;
		d[i][11000-a]++;
		d[i][11000+a]++;
		ans=(ans+d[i][11000])%mod;
	}
	printf("%I64d\n",ans); 
	return 0;
}
