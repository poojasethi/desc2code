#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
const int S=10100,M=1000000007;;
int n,w[2][S]={{1}},*p=w[0],*q=w[1],t;
int main(){
  cin>>n;
  for(int a;n-->0&&cin>>a;swap(p,q)){
    memset(q,0,sizeof(w[0]));
    for(int i=0;i<S;++i)
      if(p[i]){
        (q[i+a]+=p[i])%=M;
        (q[abs(i-a)]+=p[i])%=M;
      }
    (t+=q[0])%=M;
    (++q[0])%=M;
  }
  cout<<t<<endl;
}
