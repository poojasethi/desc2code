#include<iostream>
#include<stdio.h>

using namespace std;

long long ans,a,n,i,j,DP[1001][20001];

main()
 {
  cin>>n;
  DP[0][10000]=1;
  for (i=1;i<=n;i++)
   {
    cin>>a;
    for (j=0;j<=20000;j++)
     DP[i][j]=(((j-a)>0?DP[i-1][j-a]:0)+((j+a)<20000?DP[i-1][j+a]:0))%1000000007;
    ans=(ans+DP[i][10000])%1000000007;
    DP[i][10000]++;
   }
  cout<<ans<<endl;
 }
