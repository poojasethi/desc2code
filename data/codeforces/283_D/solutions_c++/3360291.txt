#include<stdio.h>
#include<string.h>
#include<stdlib.h>
#include<algorithm>
#include<vector>
#include<queue>
#include<math.h>
#include<string>
#include<map>
#include<set>
#include<iostream>
using namespace std;
typedef long long ll;
const int inf=1<<29;

ll a[5010];
int dp[5010];
int main()
{
	int n,ans=0;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%I64d",&a[i]);
		dp[i]=1;
		ll y=a[i];
		for(int j=i-1;j>=1;j--){
			if(y%2&&a[j]%y==0)
				dp[i]=max(dp[j]+1,dp[i]);
			if(y%2==0&&a[j]%y==y/2)
				dp[i]=max(dp[j]+1,dp[i]);
			y=y&1?y:y>>1;
		}
		ans=max(ans,dp[i]);
	}
	printf("%d\n",n-ans);
}
