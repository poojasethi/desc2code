#include<cstdio>//dp,记录满足条件最多多少个数即可 
#include<iostream>
#include<algorithm>
using namespace std;
const int N=5005;
long long a[N],f[N];
int main()
{
      int i,j,n;
      scanf("%d",&n);
      for(i=1;i<=n;++i)  scanf("%I64d",&a[i]);
      for(i=1;i<=n;++i)
      {
         long long x=a[i];
         f[i]=1;
         for(j=i-1;j>=1;--j)
         {
           if((x&1 && a[j]%x==0)|| (!(x&1) && a[j]%x==x/2)) f[i]=max(f[i],f[j]+1);
           if(!(x&1)) x=x>>1;
         }
      }
      long long ans=1000000;
      for(i=1;i<=n;++i)
       ans=min(ans,n-f[i]);
      printf("%I64d",ans);
//      system("pause");
      return 0;
}
