#include <iostream>
using namespace std;
long long array[5003];


bool test(long long a,long long b){
    if(b&1)
        return a%b==0;
    return b>>1==a%b;
}

int dyn[5003];

int main(void){
    int N;
    cin>>N;
    for(int i=0;i<N;++i){
        cin>>array[i];
        dyn[i]=1;
    }
    int mx=-1;
    long long tmp;
    for(int i=1;i<N;++i){
        tmp=array[i];
        for(int j=i-1;j>=0;--j){
            if(test(array[j],tmp))
                dyn[i]=max(dyn[i],dyn[j]+1);
            tmp&1?0:tmp>>=1;
        }
    }
    //for(int i=0;i<N;++i)
        //cout<<dyn[i]<<"   ";
    //cout<<endl;
    for(int i=0;i<N;++i)
        mx=max(mx,dyn[i]);
    cout<<N-mx<<endl;
    return 0;
}
