
#include <cstdio>
#include <algorithm>

typedef long long LL;
int n;
LL a[5010];
int cnt[5010];
int dp[5010];

int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;++i){
        scanf("%I64d",a+i);
        while(a[i]%2==0) cnt[i]++,a[i]/=2;
        dp[i]=1;
    }
    
    int ans=0;
    for(int i=1;i<n;++i){
        int j=i;
        while(j--){
            if( a[j]%a[i]==0 && ( cnt[i]==cnt[j]+i-j || cnt[i]<i-j ) )
                dp[i] = std::max(dp[i],dp[j]+1);
        }
        if(ans < dp[i]) ans = dp[i];
    }
    printf("%d\n",n-ans);

    return 0;
}
