#include <cstdio>
#include <iostream>
using namespace std;
typedef	long long	ll;
const	ll	maxn = 5500;
int	n, dp[maxn];
ll	a[maxn], p[maxn], q[maxn];
int	main()
{
	scanf("%d", &n);
	for (int i=1; i<=n; i++)
	{
		scanf("%I64d", &a[i]);
		for (p[i] = 0, q[i] = a[i]; !(q[i] & 1); q[i] >>= 1) p[i] ++;
	}
	for (int i=n; i; i--)
	{
		dp[i] = 1;
		for (int j=i+1; j<=n; j++)
			if ((dp[j] + 1 > dp[i]) && (!(q[i] % q[j])) && ((p[j] - p[i] == ll(j - i)) || (p[j] < ll(j - i))))
				dp[i] = dp[j] + 1;
	}
	int ans = 0;
	for(int i=1;i<=n;i++)ans = max(ans, dp[i]);
	printf("%d\n", n - ans);
	return 0;
}
