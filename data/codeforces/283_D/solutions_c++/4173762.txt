#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
typedef long long ll;
#define N 5050
ll a[N],b[N];int f[N],n,c[N],S;
int main()
{
    cin>>n;
    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
        b[i]=a[i];
        while(b[i]%2==0)b[i]/=2,c[i]++;
    }
    for(int i=1;i<=n;i++)
    {
        f[i]=1;
        for(int j=1;j<i;j++)
            if(b[j]%b[i]==0&&(i-j>c[i]||i-j==c[i]-c[j]))f[i]=max(f[i],f[j]+1);
    }
    for(int i=1;i<=n;i++)S=max(S,f[i]);
    printf("%d\n",n-S);
    return 0;
}
