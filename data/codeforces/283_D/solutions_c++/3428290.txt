#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<map>
#include<vector>
#include<queue>
#include<cmath>
#include<stack>
using namespace std;
#define ll long long
const int N = 5005;
int n;
ll a[N];
int f[N];

int main()
{
    int i,j,ans;
    ll tmp;
    scanf("%d",&n);
    for(i=1; i<=n; i++) scanf("%I64d",a+i);
    for(i=1; i<=n; i++){
        f[i]=1; tmp=a[i];
        for(j=i-1;j;j--){
            if( (tmp&1) ? (a[j]%tmp==0) : (a[j]%tmp == (tmp/2)) )    f[i]=max(f[i],f[j]+1);
            if((tmp&1)==0)    tmp >>= 1;
        }
    }
    ans=0;
    for(i=1; i<=n; i++) ans=max(ans,f[i]);
    printf("%d\n",n-ans);
    return 0;
}
