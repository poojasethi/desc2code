#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <string>
#include <algorithm>
#include <set>
#include <map>
using namespace std;

int n, i, j, dp[5100], ans, t[5100];
long long a[5100];

int main() {
	scanf("%d", &n);
	for (i = 1; i <= n; i++)	scanf("%I64d", &a[i]);
	for (i = 1; i <= n; i++)
		while (a[i] % 2 == 0)	t[i]++, a[i] /= 2;
	for (i = 1; i <= n; i++)	dp[i] = i - 1;
	for (i = 2; i <= n; i++)
		for (j = 1; j < i; j++)
			if (a[j] % a[i] == 0) {
				if (i - t[i] > j || t[i] - i + j == t[j])
				dp[i] = min(dp[i], dp[j] + i - j - 1);
			}
	ans = n;
	for (i = 1; i <= n; i++)	ans = min(ans, dp[i] + n - i);
	printf("%d\n", ans);
}
