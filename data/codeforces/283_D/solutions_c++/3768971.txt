#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;

#define oo (1<<28)

long long a[5000];
int p[5000];
int n;

int main(void)
{
	scanf("%i",&n);
	int M = 0;
	for (int i = 0; i < n; i++)
	{
		long long x;
		cin >> x;
		int cnt = 0;
		while ( (x & 1ll) == 0)
			cnt++, x >>= 1;
		a[i] = x;
		p[i] = cnt;
		M = max(M,cnt);
	}
	int dp[n];
	dp[0] = 0;
	for (int k = 1; k < n; k++)
	{
		dp[k] = k;
		long long r = a[k];
		int e = p[k];
		for (int j = 0; j < k; j++)
			if ( (a[j] % r) == 0 && (e < k - j || e == p[j] + k - j) )
				dp[k] = min(dp[k],dp[j] + (k-j-1) );
	}
	int ans = 100000;
	for (int i = 0; i < n; i++)
		ans = min(ans,dp[i]+n-1-i);
	cout << ans << endl;
	return 0;
}
