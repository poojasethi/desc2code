#include <iostream>
#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;
int n,f[100000],i,ans;
long long k,s[100000];
int main()
{
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
		scanf("%I64d",&s[i]);
	for (int i=1;i<=n;i++)
		{
			k=s[i];
			for (int j=i-1;j;j--)
				{
					if (k%2==0) {if (s[j]%k==k/2) f[i]=max(f[i],f[j]); k/=2;} else
					if (s[j]%k==0) f[i]=max(f[i],f[j]);												
				}
			f[i]=f[i]+1;
		}
	ans=0;
	for (int i=1;i<=n;i++)
		if (f[i]>ans) ans=f[i];
	printf("%d\n",n-ans);									 
	return 0;
}
