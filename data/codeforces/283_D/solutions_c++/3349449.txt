#include<iostream>
#include<cstring>
#include<cstdlib>
#include<cstdio>
using namespace std;
#define LL long long
const int N=5010;
int n,b[N],ans,f[N];
LL a[N];
inline bool ok(int x,int y) {
    if(a[x]%a[y]!=0) return false;
    if(b[x]+y-x==b[y]||b[y]<=y-x-1) return true;
    return false;
}
int main() {
    scanf("%d",&n);
    for(int i=1;i<=n;i++) {
        scanf("%I64d",&a[i]);
        while(a[i]%2==0) b[i]++,a[i]/=2;
    }
    for(int i=1;i<=n;i++) {
        f[i]=1;
        for(int j=1;j<i;j++)
            if(ok(j,i))
                f[i]=max(f[i],f[j]+1);
    }
    for(int i=1;i<=n;i++) ans=max(ans,f[i]);
    printf("%d\n",n-ans);
    return 0;
}
