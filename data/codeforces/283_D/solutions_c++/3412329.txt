#include <iostream>
#include <algorithm>
#define MAXN 5005

using namespace std;

int n, res, c[MAXN], v[MAXN];
long long a[MAXN];

main()
{
  cin >> n;
  for (int i = 0; i < n; i++) cin >> a[i];
  
  for (int i = 0; i < n; i++) 
    {
      //long long p = a[i];
      while (a[i] && !(a[i] & 1)) 
        {
          a[i] >>= 1;
          v[i]++;
        }
        
      c[i] = 1;
      for (int j = 0; j < i; j++)
        if (!(a[j] % a[i]) && (v[i] < i - j ||  v[i] - v[j] == i - j)) c[i] = max(c[i], c[j] + 1);
    }
    
  res = n - *max_element(c, c + n);
  cout << res;
  
//am  system("pause");
}
