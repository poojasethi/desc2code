#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int N=6100;
int i,j,k,l,n,m,num[N],ans,f[N];
long long a[N];
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i){
		scanf("%I64d",&a[i]);
		while(a[i]%2==0){
			a[i]/=2;
			num[i]++;
		}
	}
	for(i=1;i<=n;++i){
		f[i]=1;
		for(j=1;j<i;++j)if(a[j]%a[i]==0 && (num[j]+(i-j)==num[i] || (i-j-1)>=num[i]))
			f[i]=max(f[i],f[j]+1);
		ans=max(ans,f[i]);
	}
	printf("%d\n",n-ans);
}
