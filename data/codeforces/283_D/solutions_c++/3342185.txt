#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#define LL long long

using namespace std;

int n;
int f[6000];
LL a[6000];

int main() {
    scanf("%d",&n);
    for (int i=n-1;i>=0;--i)
        cin >> a[i];
    for (int i=0;i<n;++i) {
        f[i]=max(f[i],1);
        LL k=a[i];
        for (int j=i+1;j<n;++j) {
            if (k & 1) {
                if (a[j] % k==0) f[j]=max(f[i]+1,f[j]);
            } else {
                k/=2;
                if (a[j] % k==0 && a[j]/k % 2==1)
                    f[j]=max(f[j],f[i]+1);
            }
        }
    }
    int ans=0;
    for (int i=0;i<n;++i)
        ans=max(ans,f[i]);
    printf("%d\n",n-ans);
}
