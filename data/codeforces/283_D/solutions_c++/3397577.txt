#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

typedef long long LL;
const int N=5010;
LL a[N];
int p[N],dp[N];
int main(){
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        LL x;
        scanf("%I64d",&x);
        p[i]=0;
        while(x%2==0){
            p[i]++;
            x/=2;
        }
        a[i]=x;
    }
    memset(dp,0,sizeof(dp));
    for(int i=0;i<n;i++)
        for(int j=0;j<i;j++){
            if(a[j]%a[i]==0 && (p[i]==p[j]+i-j || p[i]<i-j))
                dp[i]=max(dp[i],dp[j]+1);
        }
    int ans=0;
    for(int i=0;i<n;i++) ans=max(ans,dp[i]);
    printf("%d\n",n-1-ans);
    return 0;
}
