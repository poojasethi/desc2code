#include<cstdio>
#include<algorithm>
using namespace std;
long long pow2[10000];
long long dp[10000];
long long a[10000];
long long n,ans;
inline bool check(long long x,long long y,long long l)
{
	if(a[x]%a[y]!=0)return false;
	if(pow2[y]-pow2[x]==l)return true;
	if(pow2[y]<l)return true;
	return false;
}
int main()
{
	scanf("%I64d",&n);
	ans=n;
	for (long long i=1;i<=n;i++)
	{
		scanf("%I64d",&a[i]);
		while (a[i]%2==0)
		{
			pow2[i]++;
			a[i]/=2;
		}
		dp[i]=1;
		for (long long j=1;j<i;j++)
		{
			if(check(j,i,i-j))
			{
				dp[i]=max(dp[i],dp[j]+1);
			}
		}
		ans=min(ans,n-dp[i]);
	}printf("%I64d\n",ans);
	return 0;
}
