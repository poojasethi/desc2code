#include <algorithm>
#include <iostream>

using namespace std;

int main() {
  int n;
  cin >> n;
  long long a[n];
  int p[n];
  for (int i = 0; i < n; i++) {
    cin >> a[i];
    for (p[i] = 0; a[i] % 2 == 0; a[i] /= 2)
      p[i]++;
  }
  int d[n];
  for (int i = 0; i < n; i++)
    d[i] = i;
  for (int i = 0; i < n; i++)
    for (int j = i + 1; j < n; j++) {
      if (a[i] % a[j] != 0)
        continue;
      int k = j - i;
      if (p[j] >= k && p[j] != p[i] + k)
        continue;
      d[j] = min(d[j], d[i] + k - 1);
    }
  int ans = n;
  for (int i = 0; i < n; i++)
    ans = min(ans, d[i] + (n - i - 1));
  cout << ans << endl;
  return 0;
}
