#include <iostream>
#include <map>

using namespace std;

const int MAXN = 1e6 + 10;

map <int, int> l, r;
int n;
int a[MAXN], rright[MAXN], lleft[MAXN], fen[MAXN];
long long ans;

void add(int x, int val)
{
	for (x++; x < n; x += (x & -x))
		fen[x] += val;
}

int get(int x, int ret = 0)
{
	for (; x; x -= (x & -x))
		ret += fen[x];
	return ret;
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0);
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> a[i];
	for (int i = 0; i < n; i++)
		lleft[i] = ++l[a[i]];
	for (int i = n - 1; i > -1; i--)
		rright[i] = ++r[a[i]];
	for (int i = n - 1; i > -1; i--)
	{
		ans += get(lleft[i]);
		add(rright[i], 1);
	}
	cout << ans << endl;
	return 0;
}
