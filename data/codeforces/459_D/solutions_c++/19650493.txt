#include<bits/stdc++.h>
using namespace std;
const int maxn=1e6+6;
map<int,int>M;
int pf[maxn],sf[maxn],a[maxn];
int d[maxn];
void add(int x){
	while(x<maxn)
		d[x]++,
		x+=x&-x;
}
int get(int x){
	int o=0;
	while(x)
		o+=d[x],
		x-=x&-x;
	return o;
}
int n;
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i]);
	}
	
	for(int i=1;i<=n;i++){
		M[a[i]]++;
		pf[i]=M[a[i]];
//		printf("%d%d",pf[i]," \n"[i==n]);
	}

	
	M.clear();
	for(int i=n;i>=1;i--){
	
	
		M[a[i]]++;
		sf[i]=M[a[i]];
//		printf("%d%d",sf[i]," \n"[i==1]);
	}
	
	
		
		
	long long ans=0;
	for(int i=n;i>=1;i--){
		ans+=get(pf[i]-1);
		add(sf[i]);
	}
	
	cout<<ans<<endl;
	return 0;
}
