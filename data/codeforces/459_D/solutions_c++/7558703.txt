#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <map>
#include <set>
#include <algorithm>
using namespace std;
#define MAXN 1000010

int n, a[MAXN], f[MAXN], c[MAXN];
map<int, int> ma;

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i += 1) {
        scanf("%d", a + i);
        f[i] = ++ma[a[i]];
    }
    long long ans = 0;
    for (int i = 0; i < n; i += 1) {
        int t = ma[a[i]] - f[i];
        ans += i;
        for (int x = t + 1; x; x -= x & (-x)) ans -= c[x];
        for (int x = f[i]; x <= n; x += x & (-x)) c[x]++;
    }
    printf("%I64d\n", ans);
}
