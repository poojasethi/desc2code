#include <iostream>
#include <vector>
#include <set>
#include <map>
#include <algorithm>

using namespace std;
const int MAXN = 1000 * 1000 + 10;
int a[MAXN], d[MAXN], fen[MAXN];
map<int, int> c;

int main()
{
    ios_base::sync_with_stdio(false);
    int n;
    long long ans = 0;
    cin >> n;
    for(int i = 1 ; i < n + 1 ; i++)
    {
        cin >> a[i];
        d[i] = ++c[a[i]];
    }
    for(int i = n ; i > 0 ; i--)
    {
        for(int j = d[i] - 1 ; j > 0 ; j -= j & -j)
            ans += fen[j];
        for(int j = c[a[i]] - (d[i] - 1) ; j < n + 1 ; j += j & -j)
            fen[j]++;
    }
    cout << ans << endl;
}
