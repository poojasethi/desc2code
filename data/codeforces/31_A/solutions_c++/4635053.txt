#include <iostream>
using namespace std;

int main(){
	int n;
	cin>>n;
	int v[n];
	for(int i = 0;i<n;i++){cin>>v[i];}
	for(int i = 0;i<n;i++){	
		for(int j = 0;j<n;j++){
			for(int k = 0;k<n,k!=j;k++){
				if(v[i]==v[j]+v[k]){cout<<i+1<<" "<<j+1<<" "<<k+1;return 0;}				
			}
		}				
	}
	cout<<-1;
	return 0;
}
