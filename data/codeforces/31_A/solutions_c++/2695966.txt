# include <iostream>
# include <algorithm>
using namespace std;
int A[105],N,i,j,k;
int main()
{
	cin>>N;
	for(i=0;i<N;i++)	cin>>A[i];
	for(i=0;i<N;i++) for(j=0;j<N;j++) for(k=0;k<N;k++) if(i!=j && j!=k && k!=i && A[k]==A[i]+A[j]) {cout<<k+1<<" "<<j+1<<" "<<i+1<<endl; exit(0);}
	cout<<-1<<endl;
	exit(0);
}
