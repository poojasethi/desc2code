#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,a[101];cin>>n;for(int i=0;i<n;i++) cin>>a[i];
    for(int i=0;i<n;i++)
        for(int j=0;j<n;j++)
            for(int k=0;k<n;k++) if(a[i]+a[j]==a[k] && i!=j && j!=k && k!=j) {cout<<k+1<<" "<<j+1<<" "<<i+1;return 0;}
    cout<<-1;
    return 0;
}
