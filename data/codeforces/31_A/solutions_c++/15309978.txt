#include<cstdio>

int n;
int v[111];

int main() {
	scanf("%d", &n);
	for(int i=1;i<=n;i++) scanf("%d", &v[i]);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=n;j++)
			for(int k=j+1;k<=n;k++)
				if(i!=j&&i!=k&&v[i]==v[j]+v[k]) {
					printf("%d %d %d\n", i, j, k);
					return 0;
				}
	printf("-1\n");
	
	return 0;
}
