#include <stdio.h>

int main(void) {
	int n;
	scanf("%d", &n);
	int a[105], flag=1;
	for(int i=0;i<n;i++) scanf("%d", &a[i]);
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++)
			for(int k=0;k<n;k++)
				if(i!=j&&j!=k&&k!=i&&a[i]==a[j]+a[k] &&flag)
					printf("%d %d %d\n", i+1, j+1, k+1), flag=0;
	if(flag) puts("-1");
	return 0;
}
