#include<iostream>
using namespace std;
int main(){
	int n,i,a[101],j,k;
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>a[i];
	}
	for(i=1;i<=n;i++){
			for(j=1;j<=n;j++){
				for(k=1;k<=n;k++){
					if(a[i]==a[k]+a[j] and k!=j){
						cout<<i<<" "<<k<<" "<<j;
						return 0;
					}
				}
			}	
	}
	cout<<"-1";
	
	return 0;
}
