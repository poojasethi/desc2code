#include<iostream>
using namespace std;
int a[110];
int main()
{
	int n;
	cin>>n;
	for(int i=1;i<=n;i++)
		cin>>a[i];
	for(int v=1;v<=n;v++)
		for(int i=1;i<=n;i++)
			for(int j=1;j<=n;j++)
				if(i!=j&&a[v]==a[i]+a[j])
				{
					cout<<v<<" "<<i<<" "<<j;
					return 0;
				}
	cout<<-1;
	return 0;
}
