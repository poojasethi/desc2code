#include<iostream>
using namespace std;
long i,j,k,n;
long T[105];
int main()
{
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>T[i];
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			for(k=1;k<=n;k++)
				if((j!=i)&&(i!=k)&&(k!=j)&&(T[j]+T[k]==T[i]))
				{
					cout<<i<<" "<<j<<" "<<k;
					return 0;
				}
	cout<<-1;
	return 0;	
}
