#include<iostream>
using namespace std;

int n,a[100+10];

int main(){
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>a[i];
	}
	for(int k=1;k<=n;k++){
		for(int i=1;i<=n;i++){
			for(int j=i+1;j<=n;j++){
				if(a[i]+a[j]==a[k]){
						cout<<k<<" "<<j<<" "<<i;
						return 0;
						}
				}
			}
		}
	cout<<"-1";
	return 0;
}
