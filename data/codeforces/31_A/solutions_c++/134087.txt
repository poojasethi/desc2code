#include <stdio.h>
long a[200];
long i,j,k,n,m;
int main(){
  scanf("%d",&n);
  for(i=1;i<=n;i++)
    scanf("%d",&a[i]);
  for(i=1;i<=n;i++){
    for(j=1;j<=n;j++)
      for(k=1;k<=n;k++)
        if(j!=k&& a[i]==a[j]+a[k]) {
         printf("%d %d %d",i,j,k);
         return 0;
        }
  }
  printf("-1");
  return 0;
}
