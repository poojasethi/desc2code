#include<stdio.h>

int  main()
{
  int n, i, j, q, a[101];
  scanf("%d", &n);
  for(i=1; i<=n; i++)
    scanf("%d", &a[i]);
  for(i=1; i<=n; i++)
    for(j=1; j<=n; j++)
       for(q=1; q<=n; q++)
         if(i!=j && i!=q && j!=q && (a[i]==a[j]+a[q])){printf("%d %d %d\n", i, j, q); return 0;}
         printf("-1\n"); 
}
