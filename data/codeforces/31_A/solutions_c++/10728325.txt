#include<stdio.h>

int i,j,n,t,arr[1005];

int main(){

	scanf("%d",&n);
	for(i = 1; i <= n; i++)
		scanf("%d",&arr[i]);

	for(i = 1; i <= n; i++)
		for(j = 1; j <= n; j++)
			for(t = 1; t <= n; t++)
				if(i != j && j != t && arr[t]+arr[j] == arr[i]){
					printf("%d %d %d",i,j,t);
					return 0;
				}

	puts("-1");

	return 0;
}
