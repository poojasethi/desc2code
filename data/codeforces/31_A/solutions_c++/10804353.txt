#include<bits/stdc++.h>
using namespace std;
int v[1003];
int main()
{int n,x,m=0;cin>>n;
    for(int i=0; i<n; i++){cin>>x;v[m++]=x;}
    for(int i=0; i<m; i++)for(int l=0; l<m; l++){if(i==l)continue;for(int k=0;k<m;k++){if(k==i||k==l)continue;if(v[i]==v[l]+v[k]){cout<<i+1<<" "<<l+1<<" "<<k+1;return 0;}}}
    cout<<-1;

}
