int n,arr[1000],i,j,t;

	scanf("%d ",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&arr[i]);
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			for(t=1;t<=n;t++)
				if(i!=j && j!=t && i!=t && arr[i] == arr[j] + arr[t]){
					printf("%d %d %d",i,j,t);
					return 0;
				}
			printf("-1");
	return 0;
}
