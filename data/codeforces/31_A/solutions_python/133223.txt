import sys

def main():
    n = int(sys.stdin.readline())
    l = [int(x) for x in sys.stdin.readline().split(' ')]
    dic = {}
    for (i,x) in enumerate(l):
        dic[x] = i
    for i in range(0,n):
        for j in range(i+1,n):
            t = l[i] + l[j]
            if t in dic:
                print "%d %d %d" % (dic[t]+1, i+1, j+1)
                return
    print "-1"

if __name__ == "__main__":
    main()
