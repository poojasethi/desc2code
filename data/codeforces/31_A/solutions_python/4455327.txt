I=lambda:map(int, raw_input().split())
n = input()
a = I()
b = []
for i in xrange(n):
    b.append((a[i], i+1))
b = sorted(b, key = lambda x: x[0])

for i in xrange(n-1, -1, -1):
    j, k = 0, i - 1
    while j < k:
        x = b[j][0] + b[k][0]
        if x == b[i][0]: 
            print b[i][1],b[j][1], b[k][1] 
            exit()
        elif x < b[i][0]:
            j += 1
        else:
            k -= 1
print -1
