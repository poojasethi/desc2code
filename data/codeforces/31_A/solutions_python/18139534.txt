#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = ia()

def solve():
    for i in range(n):
        for j in range(i+1, n):
            for k in range(j, n):
                if a[i]+a[j]==a[k]: print k+1, j+1, i+1; return
                if a[i]+a[k]==a[j]: print j+1, i+1, k+1; return
                if a[j]+a[k]==a[i]: print i+1, j+1, k+1; return
    print -1

solve()
