from sys import stdin

n = int(stdin.readline())
a = map(int, stdin.readline().split(' '))

for i in range(n):
  for j in range(n):
    if i == j:
      continue
    for k in range(n):
      if k == i or k == j:
        continue
      if a[i] == a[j] + a[k]:
        print 1+i, 1+j, 1+k
        quit()

print -1
