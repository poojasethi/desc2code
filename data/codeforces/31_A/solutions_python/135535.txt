#In the name of ALLAH
n = int (raw_input())
array = map(int, raw_input().split())
a = "-1"
for i in range(0, n):
    for j in range(0, n):
        for z in range(0, n):
            if array[i] == array[j] + array[z] and i != j and i != z and j != z:
                a = "{0} {1} {2}".format(i+1, j+1, z+1)
print a
