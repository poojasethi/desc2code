#!/usr/bin/env python

def main():
    raw_input()
    l = map(lambda x: int(x), raw_input().split())
    for i in xrange(0, len(l)):
        for j in xrange(0, len(l)):
            if (l[j] >= l[i] or i == j):
                continue
            for k in xrange(0, len(l)):
                if (j != k and i != k and l[i] == l[j] + l[k]):
                    print "%d %d %d" % (i+1, j+1, k+1)
                    return 0
    print "-1"
    return 0

if __name__ == '__main__':
    main()
