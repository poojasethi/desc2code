import sys
n = int(raw_input())
a = list(map(int,raw_input().split()))
for i in range(n):
    for j in range(i):
        for k in range(j):
            if a[i]==a[j]+a[k]:
                print i+1,j+1,k+1 
                sys.exit()
            if a[j]==a[k]+a[i]:
                print j+1,k+1,i+1 
                sys.exit()
            if a[k]==a[j]+a[i]:
                print k+1,j+1,i+1 
                sys.exit()
print -1
