import sys

n = int(raw_input())
f = raw_input().split(' ')
f = map(lambda x:int(x), f)


for i in range(n):
    for j in range(i+1,n):
        for k in range(j+1,n):
            if f[i]==f[j]+f[k]:
                print i+1,j+1,k+1
                sys.exit(0)
            if f[j]==f[i]+f[k]:
                print j+1,i+1,k+1
                sys.exit(0)
            if f[k]==f[j]+f[i]:
                print k+1,j+1,i+1
                sys.exit(0)

print -1
