import sys
import math

file = sys.stdin
cnt = int(file.readline().rstrip())
p = map(int, file.readline().rstrip().split())
s = [0]*2005

for i, j in [(i, j) for i in range(cnt) for j in range(cnt) if i != j]:
    s[p[i] + p[j]] = (i, j)

for i in range(len(p)):
    t = p[i]
    if s[t] != 0:
        print i + 1, s[t][0] + 1, s[t][1] + 1
        break
else:
    print "-1"
