forms = [ int(x) for x in raw_input().split() ]
def do():
    for i in range(len(forms)):
        for k in range(i + 1, len(forms)):
            s = forms[i] + forms[k]
            for j in range(len(forms)):
                if j == i or j == k: continue
                if s == forms[j]:
                    print j+1, i+1, k+1
                    return
    print "-1"
