a=map(int,raw_input().split())
aa={}
for i in range(len(a)):
    if not a[i] in aa:
        aa[a[i]]=1
found=False
for i in range(len(a)-1):
    for j in range(i+1,len(a)):
        if a[i]+a[j] in aa and not found:
            print a.index(a[i]+a[j])+1,i+1,j+1
            found=True
if not found:
    print -1
