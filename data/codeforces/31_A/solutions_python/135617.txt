import sys
n = int(sys.stdin.readline());
a = map(int, sys.stdin.readline().split(' '));
done = False
try:
    for i in xrange(1,n+1):
        for j in xrange(1,n+1):
            for k in xrange(1,n+1):
                if i==j or j==k or k==i:
                    continue
                if a[k-1]+a[j-1]==a[i-1]:
                    done = True
                    print i,j,k
                    raise StopIteration()
except StopIteration:
    pass
        
if done == False:
    print '-1'
