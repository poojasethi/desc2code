#include <stdio.h>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;

struct el { long long i,x; };
bool operator < ( el A,el B) { return (A.x>B.x); };
el a[200000];

long long n,k,b,i,res,j,s;
main()
{
 scanf("%lld%lld",&n,&k);
 scanf("%lld",&b);
 for ( i=0;i<n;i++ )
  {
   scanf("%lld",&a[i].x);
   a[i].i=i;
  }
 n--;
 sort(a,a+n);

 res=n;
 for ( i=0;i<k-1;i++ )
  s+=a[i].x;
 
 for ( i=0;i<k-1;i++ )
  if ( s+a[k-1].x>b )
   res=min(res,a[i].i);
 
 for ( i=k-1;i<n;i++ )
  if ( s+a[i].x>b )
   res=min(res,a[i].i);
 
 printf("%lld\n",res+1LL);
// system("B.exe");
}
