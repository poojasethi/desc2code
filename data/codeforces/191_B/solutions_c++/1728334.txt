#include<iostream>
#include<cstring>
#include<algorithm>
using namespace std;

const int maxn = 1e5+10;

typedef pair<int,int> pii;

pii in[maxn];
pii in2[maxn];
int mark[maxn];


int main() {
	int n,k;
	cin>>n>>k;
	long long b;
	cin>>b;
	for(int i=0;i<n;i++) {
		cin>>in[i].first;
		in[i].second = i;
	}
	memcpy(in2,in,sizeof in2);
	sort(in,in+n-1,greater<pii>());
	long long sum = 0;
	for(int i=0;i<k-1;i++) {
		if(in[i].first!=in[k-1].first)
			mark[in[i].second] = 1;
		sum+=in[i].first;
	}
	int ans = n;
	for(int i=0;i<n;i++) {
		if(!mark[i] && sum+in2[i].first>b) {
			ans = i+1;
			break;
		} else if(mark[i] && sum+in[k-1].first>b) {
			ans = i+1;
			break;
		}
	}
	cout<<ans<<endl;

}
