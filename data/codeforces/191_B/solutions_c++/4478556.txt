#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
#include<map>
#include<set>
#include<string>
#include<queue>
#include<cmath>

#define N 100005
#define mod 1000000007
#define ll long long

using namespace std;

int n,m,a[N];
ll b,sum=0;

int main(){
    scanf("%d%d",&n,&m);
    scanf("%d",&b);
    for(int i=1;i<=n;i++) scanf("%d",a+i);
    int ans=n;
    priority_queue<int,vector<int>,greater<int> > Q; 
    for(int i = n - 1; i >0 ; i--)
    {
        if(a[i]+sum>b) ans=i;
        sum+=a[i];
        Q.push(a[i]);
        if(Q.size()==m) sum-=Q.top(),Q.pop();
    }
    printf("%d\n",ans);
    return 0;
}
