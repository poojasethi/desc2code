#include<stdio.h>
#include<algorithm>
#include<queue>
#include<functional>
#include<vector>
using namespace std;
priority_queue< long long , vector<long long> ,greater< long long > > heap;
long long a[100000];
int n,k;long long b;

int main()
{
    scanf("%d%d%I64d",&n,&k,&b);
    for (int i=0;i<n;i++) scanf("%I64d",&a[i]);
    int ans=n--;
    long long sum=0;
    for (int i=n-k+1;i<n;i++)
    if (i>=0)
    {
        sum+=a[i];
        heap.push(a[i]);
    }
    for (int i=n-k;i>=0;i--)
    {
        sum+=a[i];
        if (sum>b) {ans=i+1;}
        heap.push(a[i]);
        sum-=heap.top();
        heap.pop();
    }
    printf("%d\n",ans);
    return 0;
}
