#include<iostream>
#include<algorithm>
#include<utility>
using namespace std;
int a[100000];
pair<int,int> x[100000];
bool used[100000];
long long n,k,b;
int main(){
	cin>>n>>k>>b;
	for(int i=0;i<n;i++) cin>>a[i];
	for(int i=0;i<n-1;i++) x[i]=make_pair(a[i],i);
	sort(x,x+n-1);
	for(int i=0;i<k-1;i++){
		b-=x[n-i-2].first;
		used[x[n-i-2].second]=true;
	}
	int i=0;
	while(i<n-1 && (a[i]<=b || used[i])) i++;
	if(i==n-1) cout<<n<<endl;
	else{
		int l=0;
		while(l<n && !used[l]) l++;
		cout<<((l<i)?l:i)+1<<endl;
	}
}
