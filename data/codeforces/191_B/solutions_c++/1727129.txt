#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;

pair <int, int> A[100000];

int main () {
  int n, k;
  long long b = 0;
  scanf ("%d%d", &n, &k);
  scanf ("%I64d", &b);
  for (int i = 0; i < n - 1; i++) {
    scanf ("%d", &A[i].first);
    A[i].second = i;
  }
  sort (A, A + n - 1);
  long long sum = 0;
  int mn = n;
  for (int i = n - 2; i >= n - k - 1; i--) {
    sum += A[i].first;
    mn = min (mn, A[i].second);
  }
  if (sum <= b) {
    printf ("%d\n", n);
  } else {
    for (int i = n - k - 2; i >= 0; i--) {
      sum -= A[i + 1].first;
      sum += A[i].first;
      if (sum > b) mn = min (mn, A[i].second);
    }
    printf ("%d\n", mn + 1);
  }
  return 0;
}
