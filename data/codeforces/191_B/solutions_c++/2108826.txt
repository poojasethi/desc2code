#include <cstdio>
#include <queue>
#include <vector>
#include <algorithm>
#include <functional>
using namespace std;

typedef long long LL;

priority_queue< int, vector<int>, greater<int> > pque;
vector <LL> a;

int main ()
{
#ifdef LOCAL
	freopen("in", "r", stdin);
#endif
	LL n, m, b; scanf("%I64d%I64d%I64d", &n, &m, &b);
	a.resize(n);
	for (int i = 0; i < n; ++i) scanf("%I64d", &a[i]);
	LL cs = 0; int res(n - 1);
	for (int i = n - 2; i >= 0; --i)
	{
		cs += a[i]; pque.push(a[i]);
		if (cs > b) res = i;
		if (pque.size() >= m) cs -= pque.top(), pque.pop();
	}
	printf("%d\n", ++res);
}
