#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct rec{int x,y;}a[100010];
int v[100010],p[100010],n,k,i;
long long m,sum; 
bool cmp(rec a,rec b)
{
	return a.x>b.x;
}
int main()
{
	cin>>n>>k>>m;
	for(i=1;i<=n;i++) scanf("%d",&a[i].x),a[i].y=i;
	sort(a+1,a+n,cmp);
	for(i=1;i<=k;i++) sum+=a[i].x,v[a[i].y]=i;
	for(i=1;i<=n;i++) p[a[i].y]=i;
	if(sum<=m) {cout<<n<<endl; return 0;}
	for(i=1;i<n;i++)
		if(v[i]||sum-a[k].x+a[p[i]].x>m) break;
	cout<<i<<endl;
	return 0;
}
