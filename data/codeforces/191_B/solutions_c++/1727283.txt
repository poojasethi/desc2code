#include<stdio.h>
#include<algorithm>
using namespace std;

struct fp
{
	int data,pos;
	friend bool operator < (fp a,fp b)
	{
		return a.data>b.data;
	}
}px[100001];

int a[100001],hash[100001];
int main()
{
	int n,k,i,j;
	long long b,s=0;
	scanf("%d%d%I64d",&n,&k,&b);
	for(i=0;i<n-1;++i)
	{
		scanf("%d",&px[i].data);
		a[i]=px[i].data;
		px[i].pos=i;
	}
	sort(px,px+n-1);
	
	for(i=0;i<k;++i)
	{
		hash[px[i].pos]=1;
		s+=px[i].data;
	}
	if(s<=b)
	{
		printf("%d\n",n);
		return 0;
	}
	
	for(i=0;i<n-1;++i)
		if(hash[i]||s-px[k-1].data+a[i]>b)
			break;
	printf("%d\n",i+1);
	return 0;
}
