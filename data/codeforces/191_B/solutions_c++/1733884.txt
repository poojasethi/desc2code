#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

long a[200005]={0};
long b[200005]={0};
long long s[200005]={0};

int main()
{
 long n,k,i,j;
 long long m;
 
 scanf("%ld%ld",&n,&k);
 scanf("%I64d",&m);
 for(i=1;i<=n;i++)
   {
    scanf("%ld",&a[i]);
    b[i]=-a[i];
   }
 b[n]=0,a[n]=0;
 sort(b+1,b+n+1);
 for(i=1;i<=n;i++)
   {
    b[i]=-b[i];
    s[i]=b[i]+s[i-1];
   }
 for(i=1;i<n;i++)
   {
    if(a[i]>=b[k]&&s[k]>m||s[k-1]+a[i]>m&&a[i]<b[k])
     break;
   }
 printf("%ld\n",i);
 
 return 0;
}
