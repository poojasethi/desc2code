#include "cstdio"
#include "algorithm"
using namespace std;
typedef long long ll;
struct node
{
	int cost,id;
	bool operator <(const node &other) const
	{
		return other.cost<cost;
	}
};
node s[100000+5];
int c[100000+5];
int main()
{
	//freopen("in.txt","r",stdin);
	//freopen("out.txt","w",stdout);
	int n,k;
	ll m,sum=0;
	scanf("%d%d",&n,&k);
	int i;
	scanf("%I64d",&m);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&c[i]);
		s[i].cost=c[i];
		s[i].id=i;
	}
	sort(s+1,s+n);
	for(i=1;i<=k;i++)
	{
		sum+=s[i].cost;
	}
	if(sum<=m) printf("%d\n",n);
	else{
		for(i=1;i<=n;i++)
		{
			if(sum-s[k].cost+c[i]>m) break;
		}
		printf("%d\n",i);
	}
	return 0;
}
