#include <iostream>
#include <algorithm>
using namespace std;

const int MAXn=1e5+10;
typedef long long int64;

int a[MAXn],tmp[MAXn];
int k, n;
int64 b;

int main()
{
	ios::sync_with_stdio(false);
	cin >> n >> k >> b;
	n--;
	for(int i = 0; i < n; i++)
		cin >> a[i];
	for(int i = 0; i < n; i++)
		tmp[i]=a[i];
	sort(tmp,tmp+n);
	int64 sum=0;
	for(int i = 0; i < k; i++)
		sum+=tmp[n-1-i];
	if(sum<=b)
	{
		cout << n+1 << endl;
		return 0;
	}
	for(int i = 0; i < n; i++)
		if(a[i]>=tmp[n-k])
		{
			cout << i+1 << endl;
			return 0;
		}
		else if(sum-tmp[n-k]+a[i]>b)
		{
			cout << i+1 << endl;
			return 0;
		}
	return 0;
}
