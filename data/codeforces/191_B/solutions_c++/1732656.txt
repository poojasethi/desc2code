#include <iostream>
#include <algorithm>
#include <cstdlib>

using namespace std;

int a[100005];
int ord[100005];

typedef long long int lint;

int main()
{
  int n,k;
  cin>>n>>k;
  lint b;
  cin>>b;
  for (int i=0;i<n;i++) {
    cin>>a[i];
    ord[i]=a[i];
  }
  sort(ord,ord+(n-1));
  lint sumakmenos1=0;
  for (int i=0;i<k-1;i++)
    sumakmenos1+=ord[n-2-i];
  int valk=ord[n-2-(k-1)];
  for (int i=0;i<n-1;i++) {
    if (a[i]<=valk) {
      if (a[i]+sumakmenos1>b) {
	cout<<i+1<<endl;
	exit(0);
      }	
    } else {
      if (sumakmenos1+valk>b) {
	cout<<i+1<<endl;
	exit(0);
      }
    }
  }
  cout<<n<<endl;
}
