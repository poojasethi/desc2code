
#include <iostream>
#include <algorithm>
#include <cstdio>

using namespace std;

const int maxn=110000;

typedef long long ll;

int n,k;
int a[maxn];
int b[maxn];
ll boss;


int main () {
  cin>>n>>k;
  cin>>boss;
  for (int i=0;i<n;i++) {
	cin>>a[i];
	b[i]=a[i];
  }
  sort(b,b+n-1);
  reverse(b,b+n-1);

  ll ksum=0;

  for (int i=0;i<k;i++)
	ksum+=b[i];

  int best=n;
  for (int i=0;i<n;i++) {
	if (a[i]>=b[k-1]) {
	  if (ksum>boss) {
		best=i+1; break;
	  }
	} else {
	  ll sum=ksum-b[k-1]+a[i];
	  if (sum>boss) {
		best=i+1; break;
	  }
	}
  }
  cout<<best<<endl;

  return 0;
}
