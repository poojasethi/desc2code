#include <iostream>
#include <algorithm>
#include <functional>

using namespace std;

int N, K;
int A[100002], C[100002];
long long B;
int result;

int main()
{
	cin >> N >> K >> B;
	for (int i = 1; i <= N; ++i)
	{
		cin >> A[i];
		C[i] = A[i];
	}
	
	sort(C + 1, C + N, greater<int>());
	
	long long sumnow = 0, lim = C[1];
	for (int i = 1; i <= K; ++i)
	{
		sumnow += C[i];
		lim = min(lim, 1LL * C[i]);
	}
	
	for (int i = 1; i <= N; ++i)
	{
		long long auxsum = sumnow;
		if (A[i] < lim)
		{
			auxsum -= lim;
			auxsum += A[i];
		}
		
		if (B - auxsum < 0)
		{
			result = i;
			break;
		}
	}
	if (result == 0) result = N;
	
	cout << result << '\n';
}
