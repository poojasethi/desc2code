#include<cstring>
#include<cstdio>
#include<iostream>
#include<algorithm>

using namespace std;

int const N=100010;


__int64 a[N],b[N],m,sum=0;
int n,k;
int c[N];
int ans;

int cmp(int i,int j)
{
	return a[i]>a[j];
}

int main()
{
	scanf("%d%d%I64d",&n,&k,&m);
	for(int i=0;i<n;i++)
	{
		scanf("%I64d",&a[i]);
		b[i]=i;c[i]=0;
	}
	sort(b,b+n-1,cmp);
	for(int i=0;i<k;i++)
	{
		sum+=a[b[i]];c[b[i]]=1;
	}
	if(sum<=m){printf("%d\n",n);return 0;} 
	for(int i=0;i<n;i++)
	{
		if(c[i]||sum-a[b[k-1]]+a[i]>m){printf("%d\n",i+1);return 0;}
	}
	printf("%d\n",n);
	return 0;
}
