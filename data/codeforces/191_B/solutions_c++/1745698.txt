#include <iostream>
#include <cstring>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const int Maxn=1e5+10;
int n,k;
ll b,sum;
ll a[Maxn],aa[Maxn];
int solve()
{
    for(int i=0;i<n-1;++i)
    {
        ll bb=b-a[i];
        if(bb<sum-max(aa[n-k-1],a[i]))return i+1;
    }
    return n;
}
int main()
{
   // freopen("in.txt","r",stdin);
    scanf("%d%d",&n,&k);
    scanf("%I64d",&b);
    for(int i=0;i<n;++i)
    {
        scanf("%I64d",&a[i]);
        aa[i]=a[i];
    }
    sort(aa,aa+n-1);
    sum=0;
    for(int i=n-2;i>=n-k-1;--i)
        sum+=aa[i];
    printf("%d\n",solve());
    return 0;
}
