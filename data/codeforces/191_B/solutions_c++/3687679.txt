#include<stdio.h>
#include<algorithm>
#include<map>
using namespace std;
typedef pair<int,int> P;
P ar[100000];
int min(int a,int b){return a<b?a:b;}
int main(){
    int n,k,i;
    scanf("%d %d",&n,&k);
    long long b;
    scanf("%I64d",&b);
    for(i=0;i<n;i++){
        int x;
        scanf("%d",&x);
        ar[i]=P(x,i);
    }
    sort(ar,ar+n-1);
    reverse(ar,ar+n-1);
    long long s=0;
    int m=n-1;
    for(i=0;i<k-1;i++){
        s+=ar[i].first;
        m=min(m,ar[i].second);
    }
    if(s+ar[i].first<=b){
        printf("%d\n",n);
        return 0;
    }
    while(i<n-1&&s+ar[i].first>b)m=min(m,ar[i++].second);
    printf("%d\n",m+1);
    return 0;
}
