#include <cstdio>
#include <algorithm>
#define N 100010
#define LL long long
using namespace std;

bool vst[N]; int n;
LL a[N], k, sum, mon;
struct pp{ LL v, i; }b[N];
bool comp(pp x, pp y){ return x.v > y.v; }

int main (){
	scanf("%d%I64d%I64d", &n, &k, &mon);
	for(int i = 0; i < n; i++){
		scanf("%I64d", a + i);
		b[i].v = a[i]; b[i].i = i;
	}
	sort(b, b + n - 1, comp);
	for(int i = 0; i < k; i++){
		sum += b[i].v;
		vst[b[i].i] = 1;
	}
	LL mn = b[k - 1].v;
	for(int i = 0; i < n; i++){
		LL val = sum;
		if(!vst[i]) val = val - mn + a[i];
		if(val > mon) return printf("%d\n", i+1), 0;
	}
	printf("%d\n", n);
}
