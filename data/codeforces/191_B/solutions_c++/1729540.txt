#include <cstdio>
#include <algorithm>
#define N 100005
#define LL long long
#define fi(a, b, c) for(int a = (b); a < (c); a++)
using namespace std;

int n, k, A, B;
LL b, s;
struct odp{
	int x, i;
	bool operator < (const odp &T) const{ return x > T.x;}
} r[N];

int main(){
	scanf("%d %d %I64d", &n, &k, &b);
	fi(i, 0, n) scanf("%d", &r[i].x), r[i].i = i + 1;
	
	A = B = n;
	sort(r, r + n - 1);
	fi(i, 0, k - 1) s += r[i].x, A = min(A, r[i].i);
	
	fi(i, k - 1, n - 1) if(s + r[i].x > b) B = min(B, min(A, r[i].i));
	
	printf("%d\n", B);
	scanf("\n");
}
