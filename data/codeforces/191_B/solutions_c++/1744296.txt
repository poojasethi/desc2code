#include<iostream>
#include<algorithm>
using namespace std;

int main() {
	int n, k;
	long long b;
	pair<int, int> a[100001];
	cin >> n >> k >> b;
	for(int i = 0; i < n; ++i){
		cin >> a[i].first;
		a[i].second = i+1;
	}
	sort(a, a+n-1);
	reverse(a, a+n-1);
	long long t = 0;
	for(int i = 0; i < k-1; ++i) t += a[i].first;
	int ans = n;
	for(int i = 0; i < n-1; ++i){
		int cur = a[max(k-1, i)].first;
		if(b-t<cur && ans > a[i].second) 
				ans = a[i].second;
	}
	cout << ans << endl;
	return 0;
}
