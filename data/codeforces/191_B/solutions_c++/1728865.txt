#include <cstdio>
#include <cstring>
#include <algorithm>
#define FOR(i,s,e) for (int i=(s); i<(e); i++)
#define LLD long long
#define N 100005
using namespace std;

int n, m, nxt, ret;
int a[N], b[N], v[N];
LLD tot, sum, cur;

int cmp(int x, int y){return a[x] > a[y];}

int main(){
	scanf("%d%d%I64d", &n, &m, &tot);
	FOR(i,0,n) scanf("%d", &a[i]), b[i] = i;

	sort(b, b+n-1, cmp);
	FOR(i,0,m-1) sum += a[b[i]], v[b[i]] = 1;
	if (n != m) nxt = a[b[m-1]];
	
	ret = n-1;
	FOR(i,0,n-1){
		cur = sum;
		if (v[i]) cur = sum - a[i] + nxt;
		if (tot - cur < a[i]) ret = min(ret, i);
	}
	
	printf("%d\n", ret + 1);
	return 0;
}
