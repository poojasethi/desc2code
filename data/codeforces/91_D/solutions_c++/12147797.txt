#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
bool s1[100000];
int d2[50000][2],d3[33333][3],h[50000],*h1[50000],*h2[50000];
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t,t2=0,t3=0,y=0,a[100000],s[100000];
  scanf("%d",&q);
  for(w=0;w<q;w++){
   scanf("%d",&a[w]);
   a[w]--;
   s1[w]=1;}
  for(w=0;w<q;w++)
   if(s1[w]){
    t=1;
    s1[w]=0;
    s[0]=w;
    while(a[s[t-1]]!=s[0]){
     s[t]=a[s[t-1]];
     s1[s[t]]=0;
     t++;}
    while(t>5){
     h[y]=5;
     h1[y]=new int[5];
     h2[y]=new int[5];
     for(e=1;e<5;e++)
      h2[y][e-1]=h1[y][e]=s[t-5+e];
     h2[y][4]=h1[y][0]=s[t-5];
     y++;
     t-=4;}
    if(t==5){
     h[y]=5;
     h1[y]=new int[5];
     h2[y]=new int[5];
     for(e=1;e<5;e++)
      h2[y][e-1]=h1[y][e]=s[t-5+e];
     h2[y][4]=h1[y][0]=s[t-5];
     y++;
     continue;}
    if(t==4){
     h[y]=4;
     h1[y]=new int[4];
     h2[y]=new int[4];
     for(e=1;e<4;e++)
      h2[y][e-1]=h1[y][e]=s[t-4+e];
     h2[y][3]=h1[y][0]=s[t-4];
     y++;
     continue;}
    if(t==3){
     for(e=0;e<3;e++)
      d3[t3][e]=s[e];
     t3++;}
    if(t==2){
     for(e=0;e<2;e++)
      d2[t2][e]=s[e];
     t2++;}}
  while(t3){
   t3--;
   h[y]=3;
   h1[y]=new int[5];
   h2[y]=new int[5];
   for(e=1;e<3;e++)
    h2[y][e-1]=h1[y][e]=d3[t3][e];
   h2[y][2]=h1[y][0]=d3[t3][0];
   if(t2){
    h[y]=5;
    t2--;
    h1[y][3]=h2[y][4]=d2[t2][0];
    h1[y][4]=h2[y][3]=d2[t2][1];
    y++;
    continue;}
   if(t3){
    h[y]=5;
    t3--;
    h1[y][3]=h2[y][4]=d3[t3][1];
    h1[y][4]=h2[y][3]=d3[t3][2];
    for(e=0;e<2;e++)
     d2[t2][e]=d3[t3][e];
    t2++;}
   y++;}
  while(t2){
   t2--;
   h[y]=2;
   h1[y]=new int[4];
   h2[y]=new int[4];
   h1[y][0]=h2[y][1]=d2[t2][0];
   h1[y][1]=h2[y][0]=d2[t2][1];
   if(t2){
    h[y]=4;
    t2--;
    h1[y][2]=h2[y][3]=d2[t2][0];
    h1[y][3]=h2[y][2]=d2[t2][1];}
   y++;}
  printf("%d\n",y);
  for(w=0;w<y;w++){
   printf("%d\n",h[w]);
   for(e=0;e<h[w];e++)
    printf("%d ",h1[w][e]+1);
   printf("\n");
   for(e=0;e<h[w];e++)
    printf("%d ",h2[w][e]+1);
   printf("\n");}
  return 0;}
