#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
#define N 105
using namespace std;
int head[N],next[N*N*2],e[N*N*2],c[N*N*2],len,dis[N],bfs[N*N*N],st[N],
    i,j,k,s,t,v[N*N*2],flag[N],ans,n,x;
void addedge(int x,int y,int k,int o){
    next[len]=head[x]; head[x]=len; e[len]=y; c[len]=k; v[len]=o; len++;
    next[len]=head[y]; head[y]=len; e[len]=x; c[len]=0; v[len]=-o; len++;
}
void Mincostflow(){
    int l,r,i,x,flow;
    for(;;){
        for(i=s;i<=t;++i)dis[i]=1<<30;
        bfs[l=r=1]=s;
        dis[s]=0;
        while(l<=r){
            x=bfs[l++]; flag[x]=0;
            for(i=head[x];~i;i=next[i])if(c[i] && dis[x]+v[i]<dis[e[i]]){
                dis[e[i]]=dis[x]+v[i];
                st[e[i]]=i;
                if(!flag[e[i]])bfs[++r]=e[i],flag[e[i]]=1;
            }
        }
        if(dis[t]==1<<30)return;
        flow=1<<30;
        for(i=t;i!=s;i=e[st[i]^1])flow=min(flow,c[st[i]]);
        for(i=t;i!=s;i=e[st[i]^1]){
            c[st[i]]-=flow;
            c[st[i]^1]+=flow;
        }
        if((long long)flow*dis[t]<=k)ans+=flow,k-=flow*dis[t];
        else {
            ans+=k/dis[t];
            return;
        }
    }
}
int main(){
    scanf("%d%d",&n,&k);
    memset (head, -1, sizeof head);
    for(i=1;i<=n;++i)
        for(j=1;j<=n;++j){
            scanf("%d",&x);
            if(!x)continue;
            addedge(i,j,x,0);
            addedge(i,j,1<<30,1);
        }
    s=1; t=n;
    Mincostflow();
    printf("%d\n",ans);
}
