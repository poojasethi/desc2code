#include<iostream>
using namespace std;
main(){
string s;
int k;
cin>>k;
cin>>s;
int ile[26];
for(int i=0;i<26;i++)ile[i]=0;
for(int i=0;i<s.size();i++)ile[s[i]-'a']++;
for(int i=0;i<26;i++){if(ile[i]%k!=0){cout<<"-1\n";return 0;}}
for(int i=0;i<k;i++){
for(int j=0;j<26;j++){
for(int q=0;q<ile[j]/k;q++)cout<<char('a'+j);  
}
}
cout<<"\n";
}
