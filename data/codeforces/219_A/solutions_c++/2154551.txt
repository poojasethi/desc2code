#include<iostream>
#include<string>
using namespace std;
string x;
int k;
int n[27];
int main(){
	cin>>k;
	cin>>x;
	for(int i=0;i<x.size();i++)
		n[x[i]-'a'+1]++;
	for(int i=1;i<=26;i++)
	if(n[i]%k!=0){
		cout<<-1;
		return 0;
	}
	for(int q=1;q<=k;q++)
		for(int i=1;i<=26;i++)
			for(int j=1;j<=n[i]/k;j++)
		    	cout<<char(i+'a'-1);
}
