#include <bits/stdc++.h>
using namespace std;
int main(){
	int a[30] = {} , k;
	string s = "" , ans = "";
	cin >> k >> s;
	for(int i = 0 ; i < s.size() ; i++)
		a[s[i]-'a']++;
	for(int i = 0 ; i < 26 ; i++){
		if(a[i] % k != 0){
			cout << -1;
			return 0;
		}
		for(int j = 0 ; j < a[i]/k ; j++)
			ans += (i + 'a');
	}
	for(int i = 0 ; i < k ; i++)
		cout << ans;
}
