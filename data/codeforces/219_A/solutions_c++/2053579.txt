#include <stdio.h>
#include <string.h>
int main() {
	int k,d[26]={};
	char s[1000];
	scanf("%d",&k);
	scanf("%s",s);
	for(int i=strlen(s)-1;i>=0;--i){
		++d[s[i]-'a'];
	}
	int i=25;
	for(;i>=0;--i){
		if(d[i]%k!=0) break;
	}
	if(i==-1)
		for(int j=0;j<k;++j)
			for(int i=0;i<26;++i){
				for(int l=d[i]/k;l>0;--l)
					printf("%c",'a'+i);
			}
	else
		printf("-1");
	return 0;
}
