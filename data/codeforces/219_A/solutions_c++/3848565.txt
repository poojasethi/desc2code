#include<iostream>
#include<cstring>
using namespace std;

int main(){
string s;
int fr[26], k;
memset(fr,0,sizeof(fr));
cin>>k>>s;
bool ok =1;
for(int i=0; i<s.size(); i++) fr[s[i]-'a']++;
for(int I=0; I<26; I++){ok&=(!(fr[I]%k)); fr[I]/=k;}
if(!ok) cout<<-1;
else for(int I =0; I<k;I++) for(int j =0;j<26; j++) for(int m=0;m<fr[j];m++) cout << (char)(j+'a');
}
