
 #include <iostream>
using namespace std;
int a[26],k;
string s,t;
int main(){
	cin >> k >> s;
	for(int i=0 ; i<s.size() ; i++)
		a[s[i]-'a']++;
	for(int i=0 ; i<26 ; i++){
		if(a[i]%k!=0){
			cout << -1 << endl;
			return 0;
		}
		for(int j=0 ; j<a[i]/k ; j++)
			t+=i+'a';
	}
	for(int i=0 ; i<k ; i++)
		cout << t;
	cout << endl;
	return 0;
}
