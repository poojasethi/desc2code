#include <iostream>

using namespace std;
char s[1010],*p;
int i,j,k,v[30];
int main()
{
    cin>>k>>s;
    for(p=s;*p;p++)v[*p-'a']++;
    p=s;
    for(i=0;i<='z'-'a';i++)
    {
        if(v[i]%k){cout<<-1;return 0;}
        v[i]/=k;
        for(j=v[i];j;j--,p++)*p='a'+i;
    }
    *p=0;
    for(;k;k--)cout<<s;
    return 0;
}
