#include <string>
#include <iostream>
using namespace std;

int main()
{
	int i, n, count[26]={0};
	string in, out;
	cin>>n>>in;
	for (i=0; i<in.length(); i++)
		count[in[i]-'a']++;
	for (i=0; i<26; i++)
		if (count[i]%n == 0)
			out.append(count[i]/n, i+'a');
		else
		{
			cout<<-1<<endl;
			return 0;
		}
	for (i=0; i<n; i++)
		cout<<out;
	cout<<endl;
	return 0;
}
