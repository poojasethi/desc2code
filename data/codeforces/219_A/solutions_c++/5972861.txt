#include <iostream>
using namespace std;
#include <string>
int main()
{
	int k,l,a[124]={0};
	string s;
	cin>>k>>s;
	l=s.size();
	for(int i=0;i<l;i++)
		a[s[i]]++;
	for(int i='a';i<='z';i++)
		if(a[i]%k)
		{
			cout<<-1<<endl;
			return 0;
		}
	for(int i=0;i<k;i++)
		for(char j='a';j<='z';j++)
			for(int m=0;m<a[j]/k;m++)
				cout<<j;
}
