#include<bits/stdc++.h>
using namespace std;
int cnt[26];
int main(){
	int k;
	scanf("%d",&k);
	char c;
	getchar();
	while(isalpha(c=getchar())) cnt[c-'a']++;
	for(int i=0;i<26;i++){
		if(cnt[i]%k){
			puts("-1");
			return 0;
		}
	}
	for(int i=0;i<k;i++){
		for(int j=0;j<26;j++){
			for(int l=0;l<cnt[j]/k;l++) putchar('a'+j);
		}
	}
	return 0;
}
