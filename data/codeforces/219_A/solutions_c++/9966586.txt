#include<iostream>
#include<string>
using namespace std;
int main()
{
	int k,junr[26] {};
	string c;cin>>k>>c;
	for(int i=0;i<c.length();i++)junr[c[i]-'a']++;
	for(int i=0;i<26;i++){if(junr[i]%k){cout<<-1<<endl;return 0;}}
	for(int i=0;i<k;i++)
	{
		for(int j=0;j<26;j++)
		if(junr[j])
		{
			for(int l=1;l<=junr[j]/k;l++)
			cout<<char(j+'a');
		}
	}
}
