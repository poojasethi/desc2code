#include <cstdio>
#include <cstdlib>
#define N 1005
using namespace std;

int main()
{
	int k,f[26]={0};
	char s[N];
	scanf("%d%s",&k,s);
	for(register int i=0;s[i];++i) f[s[i]-'a']++;
	for(register int i=0;i<26;++i)
		if(f[i]%k) return puts("-1"),0;
	for(int i=1;i<=k;++i)
		for(int i=0;i<26;++i)
			for(int j=f[i]/k;j;--j) putchar('a'+i);
	return 0;
}
