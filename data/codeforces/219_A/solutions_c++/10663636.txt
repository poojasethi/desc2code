#include <iostream>
#include <string>
using namespace std;
string s,s1;
int k,arr[130];
int main()
{
    cin >> k;
    cin >> s;
    for (int i=0; i<s.length(); i++) arr[(int)s[i]]++;
    for (int i=97; i<125; i++) if (arr[i]%k!=0) {cout << -1; return 0;}
    for (int i=97; i<125; i++) for (int j=0; j<arr[i]/k; j++) s1+=(char)i;
    for (int i=0; i<k; i++) cout << s1;
}
