#include<iostream>
#include<string>
using namespace std;
int main(){
	int n,a[26]={0},l,i; string s;
	cin >> n >> s;
	l = s.size();
	for(i=0;i<l;i++) a[s[i]-'a']++;
	for(i=0;i<26;i++)if(a[i]%n!=0){cout<<-1<<endl;return 0;}
	s = "";
	for(i=0;i<26;i++) if(a[i]>0)for(int j=0;j<a[i]/n;j++) s+= i+'a';
	for(i=0;i<n;i++) cout << s ;
	cout << endl;
	return 0;
}
