#include <iostream>

using namespace std;

int k,a[27];
string s;

int main(){
  cin>>k>>s;
  for(int i=0;i<s.size();i++)
    a[s[i]-'a']++;
  for(int i=0;i<26;i++){
    if(a[i]%k){
      cout<<-1<<endl;
      return 0;
    }
  }
  for(int i=0;i<k;i++){
    for(int j=0;j<26;j++){
      for(int q=0;q<(a[j]/k);q++){
	cout<<(char)(j+'a');
      }
    }
  }
  cout<<endl;
}
