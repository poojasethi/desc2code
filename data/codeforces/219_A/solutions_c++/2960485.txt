#include<iostream>
#include<string>
using namespace std;
int main()
{
  string s,b="";
  int a[256]={0},i,n,k,j,t;
  cin>>n>>s;
  k=s.size();
  if (k%n) {cout<<-1; return 0;}
  t=k/n;
  for(i=0;i<k;i++) a[s[i]]++;
  for(i=0;i<256;i++)
  {
    if (a[i]%n) {cout<<-1; return 0;}
    for(j=0;j<a[i]/n;j++) b+=i;
  }
  for(i=0;i<n;i++) cout<<b;

  return 0;
}
