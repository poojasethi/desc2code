from sys import stdin

inn = int(stdin.next())
a,b = map(int, stdin.next().split())
m = map(int, stdin.next().split())

def calc(mb, ms, ns):
	sg = -1 if mb < ms else 1

	mi = sorted([ (e, i * sg) for i,e in enumerate(m) ])

	outIdx = [ (i*sg, ms) for e,i in mi[:ns] ]
	outIdx.extend([ (i*sg, mb) for e,i in mi[ns:] ])

	return [ e for i,e in sorted(outIdx) ]

if a > b:
	out = calc(2, 1, a)
elif a < b:
	out = calc(1, 2, b)
else:
	out = [1] * a + [2] * b

print ' '.join(map(str, out))
