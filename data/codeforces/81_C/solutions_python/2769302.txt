n = input()
a, b = map(int, raw_input().split())
e = [[] for _ in range(6)]
for i, x in enumerate(map(int, raw_input().split())):
	e[x].append(i)

def f(e, a):
	r = [0] * n
	for x in e:
		for y in x:
			if a:
				r[y] = 1
				a -= 1
			else:
				r[y] = 2
	return r

if a > b:
	r = f(e, a)
elif a == b:
	r = [1] * a + [2] * b
else:
	r = f(e[ : : -1], a)
print ' '.join(map(str, r))
