#!/usr/bin/python

import sys

data = iter(sys.stdin.read().split())

n = int(data.next())
a = int(data.next())
b = int(data.next())

lix = []
liy = []
ans = [1] * n

for i in xrange(n):
	x = int(data.next())
	lix.append((x, i))
	liy.append((x, -i))

if a == b:
	for i in xrange(b):
		ans[i + a] = 2
elif a < b:
	liy.sort()
	for i in xrange(b):
		ans[-liy[i][1]] = 2
else:
	lix.sort()
	for i in xrange(b):
		ans[lix[n - 1 - i][1]] = 2

for i in ans:
	print i,
