n = input()
a, b = map(int, raw_input().split())
t = zip(map(int, raw_input().split()), range(n))
if a == b:
	ans = [1] * a + [2] * b
elif a < b:
	t.sort(key = lambda x: [-x[0], x[1]])
	ans = [2] * n
	for i in t[:a]:
		ans[i[1]] = 1
else:
	t.sort(key = lambda x: [-x[0], -x[1]])
	ans = [1] * n
	for i in t[:b]:
		ans[i[1]] = 2
print ' '.join(map(str, ans))
