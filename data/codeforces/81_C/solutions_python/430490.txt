#!/usr/bin/python
# -*- coding: utf-8 -*- 

n = int(raw_input())
a,b = map(int, raw_input().split())
if a == b:
    print ( "1 " * a) + ( "2 " * a )
    exit( 0 )
t = [ int(x) for x in raw_input().split()]

s = []
for ( i, x ) in enumerate( t ):
    s.append( ( x,i if a > b else -i ) )
s.sort()


if a < b:
    res = [ 1 ] * n
    for i in xrange( b ):
        res[ -s[ i ][ 1 ] ] = 2
else:
    res = [ 2 ] * n
    for i in xrange( a ):
        res[ s[ i ][ 1 ] ]  = 1

print " ".join( map( str, res ) )
