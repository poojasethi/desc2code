n = input()
a, b = map(int, raw_input().split())
e = [[] for _ in range(6)]
for i, x in enumerate(map(int, raw_input().split())):
	e[x].append(i)
r = [0] * n
if a > b:
	for x in e:
		for y in x:
			if a:
				r[y] = 1
				a -= 1
			else:
				r[y] = 2
elif a == b:
	r = [1] * a + [2] * b
else:
	for x in e[ : : -1]:
		for y in x:
			if a:
				r[y] = 1
				a -= 1
			else:
				r[y] = 2
print ' '.join(map(str, r))
