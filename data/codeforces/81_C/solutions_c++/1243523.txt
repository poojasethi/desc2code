#include <cstdio>
#include <algorithm>
using namespace std;
int a[100000], b[100000], note[6];
int main()
{
  int n, na, nb;
  scanf("%d%d%d", &n, &na, &nb);
  for (int i = 0; i != n; i++)
  {
    scanf("%d", &a[i]);
    b[i] = a[i];
  }
  if (na == nb)
  {
    for (int i = 0; i != n; i++)
      if (i < na) printf("1 ");else printf("2 "); 
    return 0;  
  }
  sort(a, a+n);
  if (na < nb) for (int i = n-1; i >= n-na; i--) note[a[i]]++; 
    else for (int i = 0; i != na; i++) note[a[i]]++;
  for (int i = 0; i != n; i++)
    if (note[b[i]] > 0)
    {
      note[b[i]]--;
      printf("1 ");               
    }
    else printf("2 ");
}
