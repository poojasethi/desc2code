#include<iostream>
#include<algorithm>

using namespace std;

pair<int,int> arr[100001]={};
bool f[100001]={};

int main()
{
	int a,b,n;

	cin>>n;

	cin>>a>>b;

	for(int i = 0 ; i < n ; ++i)
	{
		cin>>arr[i].first;
		arr[i].second = i;

		if(a<b)	arr[i].first *= -1;
	}

	if(a!=b)	sort(arr,arr+n);


	for(int i = 0 ; i < a ; ++i )
	{
		f[arr[i].second] = true;
	}

	for(int i = 0 ; i < n ; ++i )
	{
		if(i)	cout<<' ';
		if(f[i])	cout<<1;
		else	cout<<2;
	}cout<<endl;
	
}
