#include <iostream>
#include <utility>
#include <algorithm>
using namespace std;

const int maxlen = 200000;

int d[maxlen], res[maxlen];
pair<int, int> g[maxlen];
int n;
int a, b;

bool func(const pair<int, int> l, const pair<int, int> r) {
	return (l.first == r.first) ? l.second < r.second : l.first > r.first;
}

int main() {
	cin >> n >> a >> b;
	for (int i = 1; i <= n; i++) cin >> d[i];
	for (int i = 1; i <= n; i++) g[i].first = d[i], g[i].second = i;
	if (a < b) {
	    sort(g + 1, g + 1 + n, func);
	} else {
		if (a > b)
		sort(g + 1, g + 1 + n);
	}
	for (int i = 1; i <= a; i++) res[g[i].second] = 1;
	for (int i = a + 1; i <= n; i++) res[g[i].second] = 2;
	for (int i = 1; i <= n; i++) cout << res[i] << " ";
}
