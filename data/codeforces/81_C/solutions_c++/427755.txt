#include<iostream>
#include<stdio.h>
#include<stack>
#include<algorithm>
#include<queue>
#include<vector>
#include<cstring>
#include<string.h>
using namespace std;

int A[100001];
int C[100001]={0};
int main()
{
	int n,a,b;
	scanf("%d",&n);
	scanf("%d %d",&a,&b);
	for(int i=0;i<n;i++)scanf("%d",&A[i]);
//	cout<<"sad";    
	int take[5];
	for(int i=0;i<5;i++)if(a>b)take[i]=i+1;
	else {
		take[i]=5-i;
	}
	if(a==b)
	{
		for(int i=0;i<a;i++)cout<<"1 ";
		for(int i=0;i<b;i++)cout<<"2 ";
		return 0;
	}
	int cont=0;
	while(cont<a)
	{
		
		for(int k=0;k<5;k++)
		{
			for(int i=0;i<n;i++)
			{
				if(A[i]==take[k])
				{
					C[i]=1;
					cont++;
					if(cont==a)
					{
						k=i=1<<30;
					}
				}
			}
		}
	}
	//  cout<<"s";
	for(int i=0;i<n;i++)printf("%d ",2-C[i]);

	return 0;
}
