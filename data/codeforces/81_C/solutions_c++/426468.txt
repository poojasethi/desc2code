#include <iostream>
#include <algorithm>
using namespace std;

int n, a, b, t[100010], p[100010], f[100010];

bool cmp1(int x, int y) {return t[x] < t[y] || t[x] == t[y] && x < y;}

bool cmp2(int x, int y) {return t[x] > t[y] || t[x] == t[y] && x < y;}

int main()
{
	//freopen("input.txt", "r", stdin);
	//freopen("output.txt", "w", stdout);
	cin >> n >> a >> b;
	if (a == b) {
		for (int i = 0; i < a; ++i) cout << "1 ";
		for (int i = 0; i < a; ++i) cout << "2 ";
		return 0;
	}
	for (int i = 0; i < n; p[i] = i++)
		cin >> t[i];
	if (a > b) sort(p, p + n, cmp1);
	else sort(p, p + n, cmp2);
	for (int i = 0; i < a; ++i)
		f[p[i]] = 1;
	for (int i = 0; i < n; ++i)
		cout << 2 - f[i] << ' ';
	return 0;
}
