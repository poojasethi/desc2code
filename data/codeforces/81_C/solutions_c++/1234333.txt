#include <cstdio>
#include <cmath>
#include <cstring>
#include <algorithm>
using namespace std;
const int N = 100005;
struct node
{
    int s, id;
    bool operator < (const node &T) const
    {return s < T.s || (s == T.s && id < T.id);}
}p[N];

int main ()
{
    int n, a, b, i, ans[N];
    for (i = 0; i < N; i ++) p[i].id = i;
    while (3 == scanf ("%d%d%d", &n, &a, &b))
    {
        for (i = 0; i < n; i ++) {scanf ("%d", &p[i].s);if (a < b) p[i].s *= -1;}

        if (a != b) sort (p, p + n);
        for (i = 0; i < a; i ++) ans[p[i].id] = 1;
        for (i = a; i < n; i ++) ans[p[i].id] = 2;
        for (i = 0; i < n; i ++) printf ("%d ", ans[i]);
        puts ("");
    }
    return 0;
}
