#include <stdio.h>
#include <algorithm>
#include <set>
#include <map>
#include <vector>
#include <string>
#include <queue>
#include <math.h>
#define N 100100
using namespace std;
int m[N], u[N];
bool cmpa(int i, int j) { return m[i]>m[j] || (m[i]==m[j] && i<j); }
bool cmpb(int i, int j) { return m[i]<m[j] || (m[i]==m[j] && i<j); }
int main()
{
    int n, a, b, i;
    for(scanf("%d%d%d", &n, &a, &b), i=0; i<n; scanf("%d", &m[i]), u[i]=i, i++);
    if(a<b) sort(u, u+n, cmpa);
    else if(a>b) sort(u, u+n, cmpb);
    for(i=0; i<n; m[u[i]]=1+(i>=a), i++);
    for(i=0; i<n-1; printf("%d ", m[i]), i++);
    printf("%d\n", m[i]);
    return 0;
}
