#include <iostream>
#include <map>
#include <vector>
#include <algorithm>

using namespace std;

typedef pair <int,int> P;
int main()
{
    int n,f,s,count=0,sum=0;
    cin>>n>>f>>s;
    vector<P> A(n);
    vector<int> B(n,2);
    for(int i = 0; i < n; i++)
    {
     cin>>A[i].first;
     A[i].second=i; 
     if(f<s)
     {
      A[i].first*=-1;  
     }
    }
    if(f != s)
     sort(A.begin(),A.end());
    for(int i=0;i<f;i++)
     B[A[i].second]=1;
    for(int i=0;i<n;i++)
     cout<<B[i]<<' ';
    //system("pause");
    return 0;
}
