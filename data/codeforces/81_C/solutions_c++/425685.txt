#include <stdio.h>
#include <algorithm>
using namespace std;

int N,a,b,num[100000],who[100000];

struct P{
    int sc,id;
}p[100000];

bool cmp1(P a,P b){
    if(a.sc != b.sc) return a.sc > b.sc;
    return a.id < b.id;
}

bool cmp2(P a,P b){
    if(a.sc != b.sc) return a.sc < b.sc;
    return a.id < b.id;
}

int main(){
    scanf("%d%d%d",&N,&a,&b);
    for(int i = 0;i < N;++i) scanf("%d",&num[i]);
    for(int i = 0;i < N;++i) p[i].sc = num[i],p[i].id = i;
    if(a < b) sort(p,p + N,cmp1);
    else if(a > b) sort(p,p + N,cmp2);
    for(int i = 0;i < a;++i) who[p[i].id] = 1;
    for(int i = a;i < N;++i) who[p[i].id] = 2;
    for(int i = 0;i < N;++i) printf("%d ",who[i]);
    printf("\n");
    return 0;
}
