#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main () {
	int n,a,b;
	cin >> n >> a >> b;
	vector<int> m(n);
	if (a==b) {
		for (int i=0;i<a;i++) cout << "1 ";
		for (int i=0;i<b;i++) cout << "2 ";
		return 0;
	}
	int cnt[6];
	cnt[1]=cnt[2]=cnt[3]=cnt[4]=cnt[5]=0;
	for (int i=0;i<n;i++) {
		cin >> m[i];
		cnt[m[i]]++;
	}
	
	if (a<b) {
		for (int i=5;i>=1;i--) {
			int ber=min(a,cnt[i]);
			a-=ber;
			cnt[i]=ber;
		}
	} else {
		for (int i=1;i<=5;i++) {
			int ber=min(a,cnt[i]);
			a-=ber;
			cnt[i]=ber;
		}
	}
	for (int i=0;i<n;i++) {
		if (cnt[m[i]]) {
			cout << "1 ";
			cnt[m[i]]--;
		} else cout << "2 ";
	}
}
