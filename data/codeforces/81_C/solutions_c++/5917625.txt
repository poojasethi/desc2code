#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
struct P
{
    int v,id;
}s[201000];
int cmp1(P a,P b) { if(a.v==b.v) return a.id<b.id;return a.v<b.v;}
int cmp2(P a,P b) { if(a.v==b.v) return a.id<b.id;return a.v>b.v;}
int ans[200100];
int main()
{

    int n,a,b;
    scanf("%d%d%d",&n,&a,&b);
    for(int i=0;i<n;i++){
        scanf("%d",&s[i].v);
        s[i].id=i;
    }
    if(a>b) sort(s,s+n,cmp1);
    else if(a<b) sort(s,s+n,cmp2);
    for(int i=0;i<a;i++){
        ans[s[i].id]=1;
    }
    for(int i=a;i<n;i++)
        ans[s[i].id]=2;
    printf("%d",ans[0]);
    for(int i=1;i<n;i++)
        printf(" %d",ans[i]);
    puts("");
}
