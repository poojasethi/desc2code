#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

struct person
{
  int x, id;
};

bool operator < (person a, person b) {return a.x < b.x || (a.x == b.x && a.id < b.id);}

const int maxn = int(2e5);

int n, a, b, x;
person inp[maxn];
int answer[maxn];

int main()
{
  scanf("%d%d%d", &n, &a, &b);
  for (int i = 0; i < n; i++)
  {
    scanf("%d", &x);
    inp[i].x = x;
    inp[i].x *= (a - b);
    inp[i].id = i;
    answer[i] = 1;
  }
  stable_sort(inp, inp + n);
  for (int i = 0; i < b; i++)
    answer[inp[n - i - 1].id] = 2;
  for (int i = 0; i < n; i++)
    printf("%d ", answer[i]);
  printf("\n");
  return 0;
}
