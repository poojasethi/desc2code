#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;

int srt[100005];
int t[100005];
vector<int> p[10];
int w[100005];

int main(void) {
	int n, a, b, i, j;
	scanf("%d", &n);
	scanf("%d%d", &a, &b);
	for(i=0;i<n;i++) {
		scanf("%d", &t[i]);
		p[t[i]].push_back(i);
	}
	if(a==b) {
		for(i=0;i<a;i++)
			w[i] = 1;
		for(;i<n;i++)
			w[i] = 2;
	} else if (a > b) {
		int k=0;
		for(i=1;i<=5;i++)
			for(j=0;j<p[i].size();j++)
				srt[k++] = p[i][j];
	} else if (a < b) {
		int k = 0;
		for(i=5;i>=1;--i)
			for(j=0;j<p[i].size();j++)
				srt[k++] = p[i][j];
	}
	if(a != b) {
		for(i=0;i<a;i++)
			w[srt[i]] = 1;
		for(;i<n;i++)
			w[srt[i]] = 2;
	}
	for(i=0;i<n;i++)
		printf("%d%c", w[i], i==n-1?'\n':' ');
	return 0;
}
