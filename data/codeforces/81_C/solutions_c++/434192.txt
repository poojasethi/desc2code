#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;

int a, b, n;
struct mark
{
  int x, num;
} ;

mark m[100100];
int ans[100100];

bool Less(mark a, mark b)
{
  return a.x < b.x || (a.x == b.x && a.num < b.num);
};

int main()
{
  //freopen("a.in", "r", stdin);
  //freopen("a.out", "w", stdout);

  scanf("%d %d %d", &n, &a, &b);
  for (int i = 0; i < n; i++)
  {
    scanf("%d", &m[i].x);
    m[i].num = i;
    m[i].x *= (a - b);
  }

  stable_sort(m, m + n, Less);

  {
  	for (int i = 0; i < a; i++)
    	ans[m[i].num] = 1;
  	for (int i = a; i < n; i++)
    	ans[m[i].num] = 2;
  }
  for (int i = 0; i < n; i++)
    printf("%d ", ans[i]);

  return 0;
}
