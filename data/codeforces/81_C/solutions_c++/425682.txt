#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;

const int MAXN=100050;
int n,a,b;
int t[MAXN],tt[MAXN];
int ans[MAXN];
int main()
{
	scanf("%d",&n);
	scanf("%d%d",&a,&b);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&t[i]);				
		tt[i]=t[i];
	}
	sort(tt+1,tt+n+1);
	memset(ans,0,sizeof(ans));
	int l=1,r=5;
	if (a<b)	
	{
		l=tt[n-a+1];
		for (int i=1;i<=n;i++) if (t[i]>l) {ans[i]=1;a--;}
	}
	else
	if (a>b)
	{
		r=tt[a];
		for (int i=1;i<=n;i++) if (t[i]<r) {ans[i]=1;a--;}
	}


	for (int i=1;i<=n;i++)
	{
		if (ans[i]==0)
		{
		ans[i]=2;
		if (l<=t[i] && t[i]<=r && a>0)
		{
			ans[i]=1;
			a--;
		}
		}
		printf("%d",ans[i]);
		printf(" ");
	}	
}
