#include<algorithm>
#include<cstdio>
using namespace std;

const int mx=100010;

int n,p,q;
int a[mx];
int id[mx];
int pos[mx];

bool cmp(int w,int e)
{
	return a[w]>a[e]||a[w]==a[e]&&w<e;
}

bool cmp2(int w,int e)
{
	return a[w]<a[e]||a[w]==a[e]&&w<e;
}

int main()
{
	int i;
	scanf("%d%d%d",&n,&p,&q);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
		id[i]=i;
	}
	if(p<q)
	{
		sort(id,id+n,cmp);
	}
	else if(p>q)
	{
		sort(id,id+n,cmp2);
	}
	for(i=0;i<p;i++)
		pos[id[i]]=1;
	for(;i<n;i++)
		pos[id[i]]=2;
	printf("%d",pos[0]);
	for(i=1;i<n;i++)
		printf(" %d",pos[i]);
	printf("\n");

	return 0;
}
