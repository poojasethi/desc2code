#include <string>
#include <iostream>
#include <vector>

using namespace std;

int main() {
	int n, a, b;
	cin >> n >> a >> b;
	vector<int> v(6), x(n);
	for (int i = 0; i < n; ++i) {
		cin >> x[i];
		v[x[i]]++;
	}
	if (b == a) { // any
		for (int i = 0; i < a; ++i) {
			cout << 1 << ' ';
		}
		for (int i = 0; i < b; ++i) {
			cout << 2 << ' ';
		}	
	}
	else {
		int al = a; // as left
		for (int i = 0; i < n; ++i) {
			int sum = 0;
			if (b > a) { // max a to 1
				for (int j = x[i]+1; j <= 5; ++j) {
					sum += v[j];
				}
			}
			else { // min a to 1
				for (int j = x[i]-1; j >= 0; --j) {
					sum += v[j];
				}				
			}
			if (sum < al) {
				cout << 1 << ' ';
				al--;
			}
			else {
				cout << 2 << ' ';
			}
			v[x[i]]--;
		}
	}
	
}
