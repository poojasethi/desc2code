#include <cstdio>
#include <vector>
#include <algorithm>

using namespace std;


int main (void) {
  int n, a, b;
  scanf ("%d%d%d", &n, &a, &b);
  
  vector <pair <int, int> > v(n);

  for (int i = 0; i < n; i++) {
    scanf ("%d", &v[i].first);
    v[i].second = i;
    if (a < b) {
      v[i].first = -v[i].first;
    } 
  }
  if (a != b) {
    sort (v.begin(), v.end());
  }

  vector <int> r(n, 2);
  for (int i = 0; i < a; i++) {
    r[v[i].second] = 1;
  }
  for (int i = 0; i  < n; i++) {
    printf ("%d%c", r[i], " \n"[i + 1 == n]);
  }

  return 0;
}
