#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

typedef pair<int, int> pi;
int n, a, b;
bool d[100000];
pi s[100000];

bool comp(pi x, pi y) {
	if (x.first != y.first) return x.first < y.first;
	else return (a > b) ? x.second < y.second : x.second > y.second;
}

int main() {
	int i;
	scanf("%d%d%d", &n, &a, &b);
	for (i = 0; i < n; s[i].second = i++) scanf("%d", &s[i].first);
	if (a == b)
		for (i = 0; i < n; ++i) printf("%d%c", (i < a) ? 1 : 2, (i < n - 1) ? ' ' : '\n');
	else {
		sort(s, s + n, comp);
		if (a > b) for (i = 0; i < n; ++i) d[s[i].second] = i < a;
		else for (i = 0; i < n; ++i) d[s[i].second] = i >= b;
		for (i = 0; i < n; ++i) printf("%d%c", d[i] ? 1 : 2, (i < n - 1) ? ' ' : '\n');
	}
	return 0;
}
