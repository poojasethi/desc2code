#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;

const int MAXN = 100*1000+10;

int num[MAXN], srt[MAXN];

bool cmpG(int i, int j)
{
	return num[i] != num[j] ? num[i] > num[j] : i < j;
}

bool cmpE(int i, int j)
{
	return i < j;
}

bool cmpL(int i, int j)
{
	return num[i] != num[j] ? num[i] < num[j] : i < j;
}

int main()
{
	ios::sync_with_stdio(false);
	int n, a, b;
	cin >> n;
	cin >> a >> b;
	for(int i=0; i<n; i++)
	{
		cin >> num[i];
		srt[i] = i;
	}
	if(a < b)
		sort(srt, srt+n, cmpG);
	else if(a == b)
		sort(srt, srt+n, cmpE);
	else
		sort(srt, srt+n, cmpL);
	for(int i=0; i<n; i++)
		num[srt[i]] = i<a ? 1 : 2;
	for(int i=0; i<n; i++)
		cout << num[i] << " ";
	cout << endl;
	return 0;
}
