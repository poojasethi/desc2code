#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n, a, b, ret[100005], s[100005], p[100005];

bool cmp1(int a, int b){return s[a]==s[b]?a>b:s[a]>s[b];}
bool cmp2(int a, int b){return s[a]==s[b]?a>b:s[a]<s[b];}

int main(){

	scanf("%d%d%d", &n, &a, &b);
	for (int i=0; i<n; i++) scanf("%d", &s[i]), p[i] = i;

	for (int i=0; i<n; i++) ret[i] = 1;

	if (a>b) sort(p, p+n, cmp1);
	else if (b>a) sort(p, p+n, cmp2);
	else{
		for (int i=0; i<b; i++) p[i] = n-i-1;
	}
	
	for (int i=0; i<b; i++) ret[p[i]] = 2;
	for (int i=0; i<n; i++) printf("%d%c", ret[i], i==n-1?'\n':' ');

	return 0;
}
