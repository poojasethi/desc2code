#include <iostream>
#include <vector>
#include <algorithm>
#include <utility>

using namespace std;
typedef pair<int, int> pii;

int main(){
	int n, a, b;
	cin >> n >> a >> b;
	vector<int> t(n);
	vector<pii> s(n);
	for(int i = 0; i < n; ++i){
		cin >> t[i];
		s[i] = pii(t[i], a >= b ? i : n - i);
	}
	vector<int> f(n, a >= b ? 1 : 2);
	if(a == b){
		for(int i = 0; i < n; ++i){
			f[i] = (i < a) ? 1 : 2;
		}
	}else{
		sort(s.begin(), s.end());
		for(int i = n - min(a, b); i < n; ++i){
			if(a >= b){
				f[s[i].second] = 2;
			}else{
				f[n - s[i].second] = 1;
			}
		}
	}
	for(int i = 0; i < n; ++i){
		cout << f[i] << (i == n - 1 ? '\n' : ' ');
	}
	return 0;
}
