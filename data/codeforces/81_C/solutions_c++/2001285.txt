#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
int main ( ) {
        int n, a, b, i;
        cin >> n >> a >> b;
        vector < pair < int, int > > v ( n );
        for ( i = 0; i < n; ++i ) {
                cin >> v[i].first;
                v[i].second = i;
                if ( a < b ) v[i].first *= -1;
        }
        if ( a != b ) sort ( v.begin ( ), v.end ( ) );
        vector < int > u ( n, 2 );
        for ( i = 0; i < a; ++i ) u[v[i].second] = 1;
        for ( i = 0; i < n; ++i ) cout << u[i] << ' ';
        cout << '\n';
}
