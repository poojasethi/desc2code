#include <cstdio>
#include <algorithm>
#include <functional>
#define MAXN 100005
using namespace std;
int main() {
	int n;
	while(scanf("%d", &n) != EOF) {
		int a, b, x[MAXN], ans[MAXN];
		pair<int, int> sorted[MAXN];
		scanf("%d%d", &a, &b);
		for(int i = 0; i < n; ++i)
			scanf("%d", &x[i]);
		if(2 * a < n)
			transform(x, x + n, x, negate<int>());
		else if(2 * a == n)
			fill(x, x + n, 0);
		for(int i = 0; i < n; ++i)
			sorted[i] = make_pair(x[i], i);
		partial_sort(sorted, sorted + a, sorted + n);
		fill(ans, ans + n, 2);
		for(int i = 0; i < a; ++i)
			ans[sorted[i].second] = 1;
		for(int i = 0; i < n; ++i)
			printf("%d%c", ans[i], (i == n - 1 ? '\n' : ' '));
	}
	return 0;
}
