#include <cstdio>
#include <algorithm>
#define MAX 100005

using namespace std;

int data[ MAX ];
int p[ MAX ];
int ans[ MAX ];

int cmp( int i , int j ){
	if( data[i] != data[j] ) return data[i] < data[j];
	return i < j ;
}

int main(){
	int a , b , n;
	scanf("%d%d%d",&n,&a,&b);
	for( int i = 0 ; i <  n ; i++ ) scanf("%d", data + i );
	if( a < b ) for( int i = 0 ; i < n ; i++ ) data[i] = -data[i] ;
	else if( a == b ) for( int i = 0 ; i < n ; i++ ) data[i] = 0;
	for( int i = 0 ; i < n ; i++ )	p[i] = i;
	sort( p , p + n , cmp );
	for( int i = 0 ; i < a ; i++ )	ans[ p[i] ] = 1;
	for( int i = a ; i < n ; i++ )	ans[ p[i] ] = 2;
	for( int i = 0 ; i < n ; i++ )	printf("%d%c", ans[i] , i == n-1 ?'\n':' ');
	return 0;
}
