#include <iostream>
#include <vector>
using namespace std;
#define f first
#define s second
#define mp make_pair
#define pb push_back
vector <int> t,f;
int n,a,b;
void go(int x,int col)
{
  for(int j=5;j>=1 && x>0;j--)
    if (col==1)
    {
      for(int i=0;i<n && x>0;i++)
        if (t[i]==j)f[i]=col,x--;
    } else
      for(int i=n-1;i>=0 && x>0;i--)
        if (t[i]==j)f[i]=col,x--;
}
int main()
{
  cin>>n>>a>>b;
  t.resize(n);
  f.resize(n);
  int oth;
  for(int i=0;i<n;i++)cin>>t[i];
  if (a==b)
  {
    for(int i=1;i<=a;i++)cout<<1<<" ";
    for(int i=1;i<=b;i++)cout<<2<<" ";
    return 0;
  }
  if (a<=b)go(a,1),oth=2;
  else go(b,2),oth=1;
  for(int i=0;i<n;i++)
    if (f[i]==0)cout<<oth<<" ";
    else cout<<f[i]<<" ";
  return 0;
}
