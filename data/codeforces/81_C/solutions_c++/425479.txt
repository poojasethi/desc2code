#include<stdio.h>
#include<algorithm>
#include<utility>
using namespace std;
int n,A,B;
pair<int,int> a[100000];
int ans[100000];
int main(){
	scanf("%d%d%d",&n,&A,&B);
	for(int i=0; i<n; i++){
		scanf("%d",&a[i].first);
		a[i].first*=-1;
	}
	if(A<B){
		for(int i=0; i<n; i++){
			a[i].second=i;
			ans[i]=2;
		}
		sort(a,a+n);
		for(int i=0; i<A; i++)
			ans[a[i].second]=1;
		for(int i=0; i<n; i++)
			printf("%d%c",ans[i],i<n-1?' ':'\n');
	}else
	if(A>B){
		for(int i=0; i<n; i++){
			a[i].second=-i;
			ans[i]=1;
		}
		sort(a,a+n);
		for(int i=0; i<B; i++)
			ans[-a[i].second]=2;
		for(int i=0; i<n; i++)
			printf("%d%c",ans[i],i<n-1?' ':'\n');
	}
	else
		for(int i=0; i<n; i++)
			printf("%d%c",i<A?1:2,i<n-1?' ':'\n');
	return 0;
}
