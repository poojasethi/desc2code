#include<cstdio>
#include<algorithm>
using namespace std;

int n, c[5],s[100000],f[100000];
int a,ac[5];
int b,bc[5];

void solve() {
	int i;
	if(a==b) {
		for(i=0;i<a;i++)f[i]=1;
		for(;i<n;i++)f[i]=2;
		return;
	}
	int ar=a,br=b,at,bt;
	for(i=4;i>=0;i--) {
		if(a<b) {
			at=min(ar,c[i]);
			bt=c[i]-at;
		} else {
			bt=min(br,c[i]);
			at=c[i]-bt;
		}
		ac[i]+=at;ar-=at;
		bc[i]+=bt;br-=bt;
	}
	for(i=0;i<n;i++) {
		if(ac[s[i]]) {
			ac[s[i]]--;
			f[i]=1;
		} else {
			bc[s[i]]--;
			f[i]=2;
		}
	}
}

int main() {
	int i;
	scanf("%d%d%d",&n,&a,&b);
	for(i=0;i<n;i++) {
		scanf("%d",s+i);
		c[--s[i]]++;
	}
	solve();
	for(i=0;i<n;i++) {
		if(i)putchar(' ');
		printf("%d",f[i]);
	}
	puts("");
	return 0;
}
