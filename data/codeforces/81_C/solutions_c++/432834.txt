#include<iostream>
#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
bool cmp(pair<int, int> a, pair<int, int> b)
{
	return a.first == b.first ? a.second < b.second : a.first > b.first;
}
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	#endif
	int n,a,b;
	scanf("%d%d%d",&n,&a,&b);
	vector< pair<int, int> > v(n);
	vector<int> p(n);
	for(int i = 0; i < n; i++)
	{
		scanf("%d",&v[i].first);
		v[i].second = i;
	}
	if (a > b) sort(v.begin(),v.end()); else 
	if (a < b) sort(v.begin(),v.end(),cmp);
	for(int i = 0; i < a; i++)
		p[v[i].second] = 1;
	for(int i = a; i < n; i++)
		p[v[i].second] = 2;
	for(int i = 0; i < n; i++)
		printf("%d ",p[i]);
	return 0;
}
