#include<cstdio>
#include<algorithm>
//        Last Change:  2011-05-04 14:02:22
using namespace std;
int n,A,B,a[100003],p[100003],ans[100003];
bool cmp(int x,int y)
{
	if(a[x]!=a[y]&&A!=B)
		if(A>B)
			return a[x]<a[y];
		else return a[x]>a[y];
	else
		return x<y;
}
int main()
{
	scanf("%d%d%d",&n,&A,&B);
	for(int i=1;i<=n;++i)
	{
		scanf("%d",a+i);
		p[i]=i;
	}
	sort(p+1,p+n+1,cmp);
	for(int i=1;i<=A;++i)
		ans[p[i]]=1;
	for(int i=1;i<=n;++i)
		if(ans[i])
		{
			putchar('1');
			if(i!=n)putchar(' ');
		}
		else
		{
			putchar('2');
			if(i!=n)putchar(' ');
		}
	puts("");
	return 0;
}
