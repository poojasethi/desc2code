#include<cstdio>
#include<algorithm>
#include<utility>
using namespace std;
pair<int,int>A[100100];
int n,a,b,o[100100],i;
main(){
	scanf("%d%d%d",&n,&a,&b);
	for(i=0;i<n;i++){
		scanf("%d",&A[i].first);
		A[i].second=i;
		if(a<b)
			A[i].first*=-1;
	}
	if(a!=b)
		sort(A,A+n);
	for(i=0;i<a;i++)
		o[A[i].second]=1;
	for(i=0;i<n;i++)
		printf("%d ",o[i]?1:2);
}
