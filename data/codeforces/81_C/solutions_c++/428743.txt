#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;

#define MAX 100001

int x[MAX],c[6],c1[6],c2[6];

int main(){
	int i,n,a,b,d,one,two;
	cin>>n>>a>>b;
	for (i=0; i<n; i++) cin>>x[i];
	if (a==b){
		cout<<1;
		for (i=1; i<a; i++) cout<<" 1";
		for (i=0; i<b; i++) cout<<" 2";
	}
	else{
		for (i=0; i<n; i++) c[x[i]]++;
		d=max(a,b);
		for (i=1; i<6; i++){
			if (a>b){
				c1[i]=min(d,c[i]);
				c2[i]=max(0,c[i]-d);
			}
			else{
				c2[i]=min(d,c[i]);
				c1[i]=max(0,c[i]-d);
			}
			d=max(0,d-c[i]);
		}
		for (i=0; i<n; i++){
			if (i) cout<<" ";
			if (c1[x[i]]){
				cout<<1;
				c1[x[i]]--;
			}
			else{
				cout<<2;
				c2[x[i]]--;
			}
		}
	}
	cout<<endl;
	return 0;
}
