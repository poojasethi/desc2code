#include <iostream>
#include <algorithm>

using namespace std;

typedef long long ll;
typedef pair<int, int> P;

P r[100100];
int res[100100];

int main()
{

	int n, a, b;
	cin >> n >> a >> b;

	for (int i = 0; i < n; i++ ){
		cin >> r[i].first;
		r[i].second = i;
		if (a < b )
			r[i].first *= -1;
	} // end for

	if (a != b )
		sort (r, r + n );

	for (int i = 0; i < a; i++ )
		res[r[i].second] = 1;

	for (int i = 0; i < n ; i++ )
		cout << (res[i] ? 1 : 2 ) << endl;
 	cout << endl;

	return 0;
}
