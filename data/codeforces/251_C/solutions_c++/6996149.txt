#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long LL;
const int M = 360360;
int d[M<<1],k;
LL a,b;

int solve(int a,int b){
    d[0] = 0;
    for(int i = 1;i <= a-b;i++){
        d[i] = d[i-1]+1;
        for(int j = 2;j <= k;j++)if((b+i)%j <= i){
            d[i] = min(d[i],d[i-(b+i)%j]+1);
        }
    }
    return d[a-b];
}

int main(){
    scanf("%I64d%I64d%d",&a,&b,&k);
    int u = a%M,v = b%M;
    LL cnt = a/M - b/M;
    if(u < v)   cnt--,u += M;
    LL ans = cnt*solve(M,0) + solve(u,v);
    printf("%I64d\n",ans);
    return 0;
}
