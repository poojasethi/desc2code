#include <cstdio>
#include <cmath>
#include <cstring>
#include <cstdlib>
#include <iostream>
using namespace std;

typedef long long LL;

LL A,B,K;

LL tot=1;

LL gcd(LL a,LL b){
	if(b==0)	return a;
	return gcd(b,a%b);
}

const int MAXV = 5e5;

int dn[MAXV];

int go(int a,int b){
	
	memset( dn,127,sizeof dn );
	
	dn[a]=0;
	for(int i=a+1;i<=b;i++){
		dn[i]=dn[i-1]+1;
		for(int j=2;j<=K;j++)
			if(i%j)
				dn[i]=min(dn[i],dn[i-i%j]+1);
	}
	return dn[b];
	
}

int main(){
	
	cin >> A >> B >> K;
	
	for(int i=2;i<=K;i++)
		tot=(tot*i)/gcd(tot,i);
	
	if(A/tot == B/tot)
		cout << go(B%tot,A%tot) << endl;
	else
		cout << go(B%tot,tot-1)+1 + go(0,A%tot) + (go(0,tot-1)+1)*abs(A/tot-(B+tot-1)/tot) << endl;
	
	return 0;
	
}
