#include<cstdio>  
#include<algorithm>
#define INF 2100000000
using namespace std;
typedef long long LL;
int LCM(int a,int b){return a/__gcd(a,b)*b;}
LL a,b,lcm=1,ans;int k;
int f[1<<20];
int slove(int u,int v)
{
    for (int i=0;i<=lcm;i++) f[i]=INF;
    f[u]=0;
    for(int i=u;i>v;i--)
    {
        f[i-1]=min(f[i-1],f[i]+1);
        for (int j=2;j<=k;j++)
            if((i-i%j)>=v)
                f[i-i%j]=min(f[i-i%j],f[i]+1);
    }
    return f[v];
}
int main()
{
    scanf("%I64d%I64d%d",&a,&b,&k);
    for(int i=2;i<=k;i++) lcm=LCM(lcm,i);
    if (a/lcm==b/lcm) ans=slove(a%lcm,b%lcm);
    else ans=slove(a%lcm,0)+(a/lcm-b/lcm-1)*slove(lcm,0)+slove(lcm,b%lcm);
    printf("%I64d",ans);
    return 0;
}
