#include<iostream>
#include<cstdio>

using namespace std;

typedef long long LL;
const LL lcm=360360;

LL a,b,k;
LL res=0;

LL transform(LL x, LL y)
{
	LL ret=0;
	
	LL cx;
	while(x!=y)
	{
		cx=x-1;
		for(int p=2; p<=k; ++p)
		{
			if((x/p)*p>=y)
				cx=min(cx,(x/p)*p);
		}
		x=cx;
		++ret;
	}
	
	return ret;
}

int main()
{
	cin>>a>>b>>k;
	
	if((a/lcm)==(b/lcm))
		res = transform((a%lcm),(b%lcm));
	else
	{
		res = transform((a%lcm),0) + transform(lcm,(b%lcm)) + ((a/lcm)-(b/lcm)-1)*transform(lcm,0) ;
		//cout<<transform((a%lcm),0)<<" "<<transform(lcm,(b%lcm))<<" "<<((a/lcm)-(b/lcm)+1)<<" "<<transform(lcm,0)<<endl;
	}
	
	cout<<res<<endl;
	
	return 0;
}
