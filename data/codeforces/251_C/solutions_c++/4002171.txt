#include<stdio.h>
#define M 360360
#define min(a,b)(a<b?a:b)
__int64 a,b;
int k,cnt[M+1];
int main(){
//	freopen("4.in","r",stdin);
//	freopen("1.out","w",stdout);
	int i,j,g,h;
	scanf("%I64d %I64d %d\n",&a,&b,&k);
	for(i=1;i<=M;i++){
		g=i-1;
		for(j=2;j<=k;j++)h=i/j*j,g=min(g,h);
		cnt[i]=cnt[g]+1;
	}
	__int64 ans=0,aa=a/M*M,bb=(b+M-1)/M*M;
	if(aa>=bb){
		ans+=cnt[a-aa],a=aa;
		ans+=(a-bb)/M*cnt[M],a=bb;
	}
	while(a>b){
		aa=a-1;
		for(j=2;j<=k;j++)if((bb=a/j*j)>=b)aa=min(aa,bb);
		ans++,a=aa;
	}
	printf("%I64d\n",ans);
}
