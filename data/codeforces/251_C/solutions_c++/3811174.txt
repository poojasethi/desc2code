#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
const int p=360360;
long long a,b;
int k,dis[p+10],bfs[p+10];
int Dis(int s,int t){
	int i,j,x,y,l,r;
	for(i=0;i<=p;++i)dis[i]=-1;
	dis[s]=0;
	bfs[l=r=1]=s;
	while(l<=r){
		x=bfs[l++];
		for(i=2;i<=k;++i)if(dis[y=(x-x%i)]==-1){
			dis[y]=dis[x]+1;
			bfs[++r]=y;
		}
		if(dis[x-1]==-1)dis[x-1]=dis[x]+1,bfs[++r]=x-1;
		if(dis[t]!=-1)return dis[t];
	}
}
int main(){
	scanf("%I64d%I64d%d",&a,&b,&k);
	if(a/p==b/p)printf("%d\n",Dis(a%p,b%p));
	else printf("%I64d\n",Dis(a%p,0)+(a/p -((b-1)/p+1))*Dis(p,0) +Dis(p,b%p));
}
