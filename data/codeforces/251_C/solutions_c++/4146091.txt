#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;

typedef long long LL;
const int LCM = 360360;
LL f[LCM+100];

LL calc(LL a, LL b, int k){
	memset(f, 0x3f, sizeof(f));
	f[a] = 0;
	if(a < b) swap(a, b);
	for(LL i = a; i >= b; i --){
		f[i-1] = min(f[i-1], f[i] + 1);
		for(int j = 2; j <= k ; j ++){
			if(i - i%j >= b)
				f[i - i%j] = min(f[i -i%j], f[i] + 1);
		}
	}
	return f[b];
}

int main(){
	int k;
	//freopen("test.in", "r", stdin);
	for(LL a, b; scanf("%I64d%I64d%d", &a, &b, &k)!=EOF; ){
		if(a/LCM == b/LCM){
			printf("%I64d\n", calc(a%LCM, b%LCM, k));
			continue;
		}
		printf("%I64d\n", calc(a%LCM, 0, k) + calc(LCM-1, b%LCM, k) + 1 +
			(calc(LCM - 1, 0, k)+1) * (a/LCM - b/LCM - 1));
	}
	return 0;
}
