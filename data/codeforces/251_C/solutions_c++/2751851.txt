#include<cstdio>
typedef long long lint;

lint a, b, k, m = 2, q;
lint gcd(lint a, lint b){
    if(!b) return a;
    return gcd(b,a%b);
}
inline lint lcm(lint a,lint b){
    return a*b/gcd(a,b);
}
inline lint count(lint a, lint b){
    lint tot=0;
    for(lint x=a,mx,i;x>b;x-=mx,tot++)
        for(i=k,mx=1;i>=2 && mx<i-1;i--)
            if(x-x%i>=b) mx = (mx<x%i?x%i:mx);
    return tot;
}
int main(){
    scanf("%I64d %I64d %I64d",&a,&b,&k);
    for(lint i=3;i<=k;i++) m = lcm(m,i); q = (a-b)/m;
    printf("%I64d\n",q*count(m,0) + count(a-q*m,b));
    return 0;
}
