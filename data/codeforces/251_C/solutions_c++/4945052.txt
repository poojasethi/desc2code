#include <algorithm>
#include <stdio.h>
#define  maxn      400000
using    namespace std;
typedef  long long LL;

LL a,b,k,lcm=1;
LL dp[maxn];

LL gcd(LL x , LL y)
{
  if(!y) return x;
  return gcd(y,x%y);
}

LL f(LL x , LL y)
{
  for(LL i=0 ; i<maxn ; i++) dp[i]=maxn;
  dp[x]=0;
  for(LL i=x+1 ; i<=y ; i++)
  {
    dp[i]=dp[i-1]+1;
    for(LL j=2 ; j<=k ; j++)
      dp[i]=min(dp[i],dp[i-(i%j)]+1);
  }
  return dp[y];
}

int main()
{
  scanf("%I64d%I64d%I64d",&a,&b,&k);
  for(LL i=2 ; i<=k ; i++)
    lcm=lcm*i/gcd(lcm,i);
  if(a/lcm==b/lcm) printf("%I64d\n",f(b%lcm,a%lcm));
  else             printf("%I64d\n",f(b%lcm,lcm)+f(0,a%lcm)+f(0,lcm)*(a/lcm-(b+lcm-1)/lcm));
  return 0;
}
