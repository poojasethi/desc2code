#define FOR(i,a,b) for(int i=(a);i<=(b);i++)
#define FORD(i,a,b) for(int i=(a);i>=(b);i--)
#define MS0(x) memset(x,0,sizeof(x))

using namespace std;
typedef long long LL;

LL a,b,k,la,aa;

int main(){
  cin>>a>>b>>k;
  la=-1;
  LL ans=0;
  while(a!=b){
    LL t=1;
    FOR(i,2,k){
      if(a%i>t && a-a%i>=b)t=a%i;
    }
    ans++;
    a-=t;
    if(a%360360==0){
      if(la==-1)la=ans;
      else{
        aa=ans-la;
        ans+=(a-b)/360360*aa;
        a=a-(a-b)/360360*360360;
      }
    }

  }
  
  cout<<ans<<endl;
}
