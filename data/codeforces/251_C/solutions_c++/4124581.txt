#include <iostream>
using namespace std;
typedef long long ll;
#define rep(i,s,e) for (int i=(s);i<(e);++i)

ll a,b,k,dp[1000000],lcm=360360;
ll solve(ll a, ll b) {
  dp[0]=0;
  rep(i,1,a-b+1) {
    dp[i] = 1 + dp[i-1];
    rep(x,2,k+1) {
      int diff = (b+i)%x;
      if (i-diff < 0) continue;
      dp[i] = min(dp[i], 1+dp[i-diff]);
    }
  }
  return dp[a-b];
}

int main() {
  cin >> a >> b >> k;
  if (a-b < 1000000) {
    cout << solve(a,b) << endl;
    return 0;
  }
  ll a1 = (b+lcm-1)/lcm*lcm;
  ll a2 = a/lcm*lcm;
  cout << solve(a1,b) + solve(a,a2) + (a2-a1)/lcm*solve(lcm,0) << endl;
}
