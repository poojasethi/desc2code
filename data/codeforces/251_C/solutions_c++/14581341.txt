#include <bits/stdc++.h>

using namespace std;
using namespace std;
typedef pair<int,int> ii;
typedef long long ll;
const int N = 1000010;
ll a, b; int k;

int GCD(int x,int y) {
    if (y == 0) return x;
    return GCD(y,x%y);
}

int LCM(int x,int y) {
    return x*y/GCD(x,y);
}

ll solve(ll a,ll b) {
    ll res = 0, t; int i;
    for (;a>b;a-=t,res++)
    for (i=k    ,t=1;i>=2 && t<i-1;--i)
        if (a-a%i>=b) t = max(t,a%i);
    return res;
}

int main()
{
    int _LCM = 2;
    cin >> a >> b >> k;
    for (int i=3;i<=k;++i) _LCM = LCM(_LCM,i);
    ll d = (a-b)/_LCM;
    cout << d*solve(_LCM,0) + solve(a-_LCM*d,b);
    return 0;
}
