#include <algorithm>
#include <iostream>
#include <cstdio>

using namespace std;

typedef long long ll;

const int MOD=360360;

ll A, B, a, b, ans;

int k, f[MOD];

int go(int a, int b) {
    f[a]=0;
    for(int i=b; i<a; ++i) f[i]=MOD;
    for(int i=a; i>b; --i) {
        f[i-1]=min(f[i-1], f[i]+1);
        for(int t=2, d=i&1; t<=k; d=i%++t)
            f[i-d]=min(f[i-d], f[i]+1);
    }
    return f[b];
}

int main() {
    //freopen("transform.inp", "r", stdin);
    //freopen("transform.out", "w", stdout);
    //Input
    cin >> A >> B >> k;

    //Solve
    a=A/MOD, b=B/MOD;
    if(a==b) ans=go(A%MOD, B%MOD); else {
        ans=go(A%MOD, 0);
        ans+=(ll)(a-b-1)*(1+go(MOD-1, 0));
        ans+=1+go(MOD-1, B%MOD);
    }

    //Output
    cout << ans;
}
