#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
long long a,b,ans,x;
int k,dp[360365],i,j;
int main()
{
    cin>>a>>b>>k;
    dp[1]=1;
    for (i=2; i<=360360; i++)
    {
        dp[i]=dp[i-1]+1;
        for (j=2; j<=k; j++)
          dp[i]=min(dp[i],dp[i-i%j]+1);
    }
    x=b/360360; b-=x*360360; a-=x*360360;
    if (a>360360) {ans=dp[a%360360]; a=a-a%360360;}
    if (a>360360) ans+=dp[360360]*(a/360360-1); a=min(a,(long long)360360);
    dp[b]=0;
    for (i=b+1; i<=a; i++)
    {
        dp[i]=dp[i-1]+1;
        for (j=2; j<=k; j++)
          if (i-i%j>=b)
            dp[i]=min(dp[i],dp[i-i%j]+1);
    }
    cout<<ans+dp[a];
    return 0;
}
