#include<cstdio>
#include<algorithm>
#define N 1000100
#define M 360360
#define INF 999999999
#define LL long long
using namespace std;
LL k, dp[N];
LL solve(LL x, LL y){
	fill(dp, dp+N, INF);
	dp[x] = 0;
	for(LL i = x; i >= y; i--){
		for(LL j = 2; j <= k; j++)
			dp[i/j*j] = min(dp[i/j*j], dp[i]+1);
		dp[i-1] = min(dp[i-1], dp[i]+1);
	}
	return dp[y];
}
int main(){
	LL a, b;
	scanf("%I64d%I64d%I64d", &a, &b, &k);
	LL val = solve(M, 0);
	LL u = a%M, v = b%M, cnt = (a/M) - (b/M);
	if(u < v) u += M, cnt--;
	printf("%I64d\n", cnt * val + solve(u, v));
}
