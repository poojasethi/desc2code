#include <stdio.h>
#include <algorithm>
using namespace std;
int k, m = 360360;
int dp[400000];
int pd(int a, int b){
    for (int i = 0; i < a; i++) dp[i] = 400000;
    dp[a] = 0;
    for (int i = a; i > b; i--){
        dp[i - 1] = min(dp[i - 1], dp[i] + 1);
        for (int j = 2; j <= k; j++){
            int x = i % j;
            dp[i - x] = min(dp[i - x], dp[i] + 1);
        }
    }
    return dp[b];
}
int main(){
    long long a, b;
    scanf("%I64d%I64d%d", &a, &b, &k);
    if (a / m == b / m){
        printf("%d\n", pd(a % m, b % m));
    }else{
        printf("%I64d\n", pd(a % m, 0) + 1 + pd(m - 1, b % m) + (pd(m - 1, 0) + 1) * (a / m - b / m - 1));
    }
    return 0;
}
