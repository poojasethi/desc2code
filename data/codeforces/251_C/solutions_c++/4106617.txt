#include<stdio.h>
__int64 f[400000],k;
__int64 gcd(__int64 a,__int64 b){
    if(a%b==0)return b;
    else return gcd(b,a%b);
}
__int64 dp(int l,int b){
	__int64 i,j,s;
	f[0]=0;
	for(i=1;i<=l;i++){
		f[i]=f[i-1]+1;
		for(j=2;j<=k;j++){
			s=(b+i)%j;
			if(i-s>=0&&f[i-s]+1<f[i])f[i]=f[i-s]+1;
		}
	}
	return f[l];
}
int main() {
    __int64 a,b,ans;
    __int64 i,j,c;
    while (scanf("%I64d%I64d%I64d",&a,&b,&k)!=EOF) {
    	if(a==b)printf("0\n");
		else if(a-1==b)printf("1\n");
		else{
	    	c=1;
        	for(i=2;i<=k;++i)c=c*i/gcd(c,i);
			printf("%I64d\n",(a/c-b/c-1)*dp(c,0)+dp(a%c,0)+dp(c-b%c,b%c));
	    }
    }
}
