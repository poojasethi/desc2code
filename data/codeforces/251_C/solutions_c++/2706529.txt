#include<iostream>
using namespace std;
int main(){
	
	long long int lcm[16] = {-1,-1,2,6,12,60,60,420,840,2520,2520,27720,27720,360360,360360,360360};
	long long int a,b,k;
	cin>>a>>b>>k;
	long long int MAX = lcm[k];
	long long int to[MAX+1];
	long long int to_b[MAX+1];
	to[0] = 0;
	for (int i=1;i<=MAX;i++){
		int min = i-1;
		for (int j=2;j<=k;j++) if (i-(i%j)<=min) min = i-(i%j);
		to[i] = to[min]+1;
	}
	to_b[b%MAX] = 0;
	for (int i=(b%MAX)+1;i<=MAX;i++){
		int min = i-1;
		for (int j=2;j<=k;j++) if (i-(i%j)<=min && i-(i%j)>=b%MAX) min = i-(i%j);
		to_b[i] = to_b[min]+1;
	}
	long long int cycles = (a/MAX)-(b/MAX)-1;
	if (cycles==-1){
		cout<<to_b[a%MAX];
		return 0;
	}
	long long int ans = cycles*to[MAX] + to[a%MAX] + to_b[MAX];
	cout<<ans;
}
