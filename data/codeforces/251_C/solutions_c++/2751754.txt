#include<iostream>
using namespace::std;

typedef long long lint;

lint gcd(lint a, lint b){
    if(!b) return a;
    return gcd(b,a%b);
}
inline lint lcm(lint a,lint b){
    return a*b/gcd(a,b);
}

int main(){
    lint a, b, k, m = 2, step = 1, tot, q;
    cin >> a >> b >> k;
    for(lint i=3;i<=k;i++) m = lcm(m,i);
    for(lint x=m-1,mx,i;x>0;x-=mx,step++)
        for(i=k,mx=0;i>=2 && mx<i-1;i--) mx = max(mx,x%i);
    q = (a-b)/m;
    tot = q*step;
    for(lint x=a-q*m,mx,i;x>b;x-=(mx?mx:1),tot++)
        for(i=k,mx=0;i>=2 && mx<i-1;i--) if(x-x%i>=b) mx = max(mx,x%i);
    cout << tot << endl;
    return 0;
}
