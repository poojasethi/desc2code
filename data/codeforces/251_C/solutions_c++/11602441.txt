#include <iostream>
#include <algorithm>

using namespace std;

const long long m = 360360;

long long k;
long long ans[1000001];

long long func(long long x, long long y) {
	for (long long i = 0; i < 1e6; i++)
		ans[i] = 1e9;
	ans[x] = 0;
	
	for (long long i = 999999; i >= 1; i--)
		if (ans[i] < 1e9) {
			ans[i - 1] = min(ans[i - 1], ans[i] + 1);
			for (long long j = 2; j <= k; j++)
				ans[i - i % j] = min(ans[i - i % j], ans[i] + 1);
		}
	
	return ans[y];
}

int main() {
	long long a, b; cin >> a >> b >> k;
	
	if (a - a % m == b - b % m)
		cout << func(a % m, b % m) << endl;
	else
		cout << func(m - 1, b % m) + func(a % m, 0) + 1 + func(m - 1, 0) * (a / m - b / m - 1) + a / m - b / m - 1 << endl;
	
	return 0;
}
