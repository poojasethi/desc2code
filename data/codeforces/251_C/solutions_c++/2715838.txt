#include <iostream>
#include <algorithm>

using std::cin;
using std::cout;
using std::endl;
using std::min;

long long a, b, k;
const long long c = 360360;
long long f[c + 1];

int foo(long long a, long long b)
{
	f[b] = 0;
	for (int i = b + 1; i <= a; ++i)
	{
		f[i] = f[i - 1] + 1;
		for (int j = 2; j <= k; ++j)
			if (i - i % j >= b)
				f[i] = min(f[i], f[i - i % j] + 1);
	}
	return f[a];
}

int main()
{
	cin >> a >> b >> k;
	if (a / c == b / c) cout << foo(a % c, b % c) << endl;
	else cout << foo(a % c, 0) +
		     foo(c, 0) * (a / c - (b + c - 1) / c) +
		     foo(c, b % c) << endl;
}
