#include<iostream>
using namespace std;
typedef __int64 i64;
i64 a,b,k;
int d[410000];
const i64 c = 360360 ;
int ask(i64 a,i64 b)
{
    d[0]=0;
    for(int i=1;i<=a-b;++i)
    {
        d[i]=d[i-1]+1;
        for(int j=2;j<=k;++j)if((b+i)%j<=i)
            d[i]=min(d[i],d[i-(b+i)%j]+1);
    }
    return d[a-b];
}
int main()
{
    cin>>a>>b>>k;
    if(a/c==b/c)
        cout<<ask(a,b)<<endl;
    else
        cout<<ask(a%c,0)+ask(c,b%c)+ask(c,0)*(a/c-(b+c-1)/c)<<endl;
    return 0;
}
