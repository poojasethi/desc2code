#include<iostream>
#include<stdio.h>
#include<stdlib.h>
#define inf 0x7fffffff
using namespace std;
typedef long long LL;
LL a,b,ans;
int k,l;
int dp[1<<20];
int gcd(int x,int y){
	return y==0?x:gcd(y,x%y);
}
int solve(int x,int y){
	for(int i=0;i<=l;i++) dp[i]=inf;
	dp[x]=0;
	for(int i=x;i>y;i--){
		dp[i-1]=min(dp[i-1],dp[i]+1);
		for(int j=2;j<=k;j++)
			if(i-i%j>=y)
				dp[i-i%j]=min(dp[i-i%j],dp[i]+1);
	}
	return dp[y];
}
int main(){
	cin>>a>>b>>k;
	l=1;
	for(int i=2;i<=k;i++)
		l=l/gcd(l,i)*i;
	if(a/l==b/l)
		ans=solve(a%l,b%l);
	else
		ans=solve(a%l,0)+(a/l-b/l-1)*solve(l,0)+solve(l,b%l);
	cout<<ans<<endl;
	return 0;
}
