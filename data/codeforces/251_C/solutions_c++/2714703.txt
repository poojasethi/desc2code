#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
typedef long long LL;
using namespace std;

LL a,b,k,d[410000];
const LL c = 360360 ;
LL ask(LL a,LL b)
{
    d[0]=0;
    for(int i=1;i<=a-b;++i)
    {
        d[i]=d[i-1]+1;
        for(int j=2;j<=k;++j)if((b+i)%j<=i)
            d[i]=min(d[i],d[i-(b+i)%j]+1);
    }
    return d[a-b];
}

int main()
{
    cin>>a>>b>>k;
    if(a/c==b/c)
        cout<<ask(a,b)<<endl;
    else
        cout<<ask(a%c,0)+ask(c,b%c)+ask(c,0)*(a/c-(b+c-1)/c)<<endl;
    return 0;
}
