#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int N=360360;
long long a,b,ans;
int x,y,S,T,k,f[N+5],d[N+5];
int bfs(int S,int T)
{
	int h=0,t=1; f[1]=S; memset(d,0,sizeof(d)),d[S]=1;
	while (h<t){
		x=f[++h];
		for (int i=1;i<=k;i++){
			y=x-(x%i)-(i==1); if (y<0) y+=N;
			if (!d[y]) d[y]=d[x]+1,f[++t]=y;
			}
		}
	return d[T];
}
int main()
{
	cin>>a>>b>>k,S=a%N,T=b%N,a/=N,b/=N;
	if (a==b) ans=bfs(S,T)-1; else ans=(a-b-1)*(bfs(N-1,0))+bfs(S,0)+bfs(N-1,T)-1;
	cout<<ans<<endl; return 0;
}
