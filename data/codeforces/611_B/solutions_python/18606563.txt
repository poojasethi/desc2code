#2^x-2^y-1=2^y(2^(x-y))-1, where 0<=y<x$
import math

lower, upper = [int(i) for i in raw_input().split()]
all_ones = [2**x-1 for x in xrange(2,int(math.log(10**19,2)))]
#x is the number of digits
count = 0
for i in xrange(len(all_ones)):
    for j in xrange(i+1):
        candidate=all_ones[i]-(1<<j)
        if candidate>=lower and candidate<=upper:
            count+=1
print count
