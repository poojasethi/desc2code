a, b = raw_input().split(' ')

candidates = []

for l in xrange(1, 64):
    ones = '1' * l
    for zero in xrange(1, len(ones)):
        candidates.append(ones[:zero] + '0' + ones[zero + 1:63])

a = int(a)
b = int(b)
count = 0
for i in candidates:
    br = int(i, 2)

    if br >= a and br <= b:
        count += 1

print count
