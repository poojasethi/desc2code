if __name__ == '__main__':
    a, b = map(int, raw_input().strip().split(' '))
    
    ans = 0
    for i in range(61):
        s = '1'*i
        for j in range(1, i):
            k = list(s)
            k[j] = '0'
            num = int(''.join(k), 2)
            if a <= num <= b:
                ans += 1
    print ans
