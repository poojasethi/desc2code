def lg(x):
    l=0
    while(x):
        x/=2
        l+=1
    return l

b=lg(U)
x=2**a-1
y=2**b
l=0
while(x<=y):
    i=0
    while(1):
        if(x-2**i>=L):
            if(x-2**i<=U):
                l+=1
