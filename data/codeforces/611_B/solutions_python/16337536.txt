a, b = map(int, raw_input().split())

n_a = len(bin(a)[2:])
n_b = len(bin(b)[2:])

answer = (n_a + n_b - 2) * (n_b - n_a + 1) / 2

for i in range(1, n_a):        
    temp = int("1" * i + "0" + "1" * (n_a - i - 1), 2)
    if temp >= a:
        break
    answer -= 1

for i in range(n_b - 1):
    temp = int("1" * (n_b - i - 1) + "0" + "1" * i, 2)
    if temp <= b:
        break
    answer -= 1

print answer
