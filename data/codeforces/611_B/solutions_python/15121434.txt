b = map(int, raw_input().split())
counter = 0
bawah = len(bin(b[0]))-2
atas = len(bin(b[1]))-2
binaray = []
for i in range(bawah,atas+1):
    binari='1'*i
    for j in range(1,i):
        intbin=binari[:j]+'0'+binari[j+1:]
        binaray.append(int(intbin,2))
for i in range(len(binaray)):
    if (binaray[i]>=b[0]) & (binaray[i]<=b[1]):
        counter+=1
print counter
