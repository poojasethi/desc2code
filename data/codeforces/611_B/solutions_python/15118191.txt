import sys

es = sys.stdin.readline().split()
a, b = map(int, es)

def digits(x):
    d = 0
    while x > 0:
        x /= 2
        d += 1
    return d

def f(x):
    d = digits(x)
    res = (d-1)*(d-2)/2+1

    m = 0
    while True:
        if x+1 >= 2**d - 2**m:
            break
        else:
            m += 1
    res += (d-m)-1
    
    return res

print f(b) - f(a-1)
