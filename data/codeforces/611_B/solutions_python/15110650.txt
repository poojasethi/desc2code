from math import log
a,b=map(int,raw_input().split())
k=4
pow2=[1]
ans=[]
while pow2[-1]<=int(1e19):
    pow2.append(2*pow2[-1])
for i in pow2:
    if i>=4:
        u=i/2
        for j in pow2:
            if j<u:
                ans.append(i-1-j)
co=0
ans.sort()
for i in ans:
    if a<=i and i<=b:
        co+=1
print co
