from sys import stdin
a,b = map(int,stdin.readline().strip().split())
ans = []
for i in xrange(1,65):
 hi = 1<<i 
 hi-=1
 for j in xrange(0,i-1):
  ans.append(hi^(1<<j))
s = set(ans)
ans = list(ans)
#print len(ans)
ans.sort()
#print ans[:10]
res = 0
for i in ans:
 if i>=a and i<=b:
  res+=1
print res
