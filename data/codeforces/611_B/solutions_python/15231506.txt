a, b = map(int, raw_input().split())
count = 0
i = 0x7fffffffffffffff
j = 0x2000000000000000
l = []
while i > 0:
    k = j
    while k > 0:
      l.append(i^k)
      k >>= 1
    j >>= 1
    i >>= 1
l = sorted(l)
for v in l:
    if v >= a and v <= b:
        count += 1
print count
