def calc(x):
    ans = 0
    for i in xrange(64):
        t = (1 << i) - 1
        j = i
        while True:
            j += 1
            t += 1 << j
            if t <= x:
                ans += 1
            else:
                break
    return ans

if __name__ == '__main__':
    l, r = map(int, raw_input().split())
    print calc(r) - calc(l - 1)
