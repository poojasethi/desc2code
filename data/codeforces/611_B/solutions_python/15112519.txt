from sys import stdin
line = stdin.readline()
nums = line.split()
a = int(nums[0])
b = int(nums[1])
nums = []
for base in xrange(0,64):
    num = 0;
    for i in xrange(0,base):
        num = (num | (1 << i))
    
    #print "%d %d" %(base, num)
    for j in xrange(base+1,64):
        num = (num | (1 << j))
        if num<a or num>b: continue
        nums.append(num)

nums = set(nums)
print(str(len(nums)))
