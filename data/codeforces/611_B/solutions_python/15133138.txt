a, b = raw_input().split()
a = int(a)
b = int(b)
length = len(bin(b)) - 2
ret = 0
for cnt in range(length, 0, -1):
    list = ['0']
    for i in range(cnt - 1):
        list.append('1')
    for i in range(cnt - 1):
        list[i] = '1'
        list[i + 1] = '0'
        string = ''
        for i in range(cnt):
            string += list[i]
        if int(string, 2) >= a and int(string, 2) <= b:
            ret += 1        
print ret
