#coding: utf-8

a, b = map(int,raw_input().split())



x = 1
cont = 0
flag = True
while(flag):
	dg1 = 2**x -1
	for n in xrange(x,-1,-1):
		c = 2**n
		numero = dg1 - c
		if(numero > b):
			flag = False
			break
		elif(numero < a):
			continue
		else:
			binario = bin(numero)
			if(binario.count('0') == 2):
				cont+=1
	x+=1

print cont
