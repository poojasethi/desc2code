def solveA(x):
	if x<2:
		return 0
	p  = 1
	count  = 0
	while p <= 65:
		p1s  = (2**p) - 1
		if p1s > x:
			break
		count += (p-1)
		p += 1
	p1s    = 2**p - 1
	d0     = p-1
	num    = p1s - (2**(d0-1))
	while d0>0 and num <= x:
		count += 1
		d0    -= 1
		num    = p1s - (2**(d0-1))
	return count

def main():
	x,y  = map(int,raw_input().split())
	print solveA(y) - solveA(x-1)

main()
