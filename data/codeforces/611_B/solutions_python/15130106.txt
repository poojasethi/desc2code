a,b=(int(i) for i in raw_input().split())
count=0
for length in range(1,61):
    for pos in range(1,length):
        s=""
        for j in range(0,length):
            if j==pos:
                s+='0'
            else:
                s+='1'
        x=int(s,2)
        if x>=a and x<=b:
            #print x,s
            count+=1
print count
