a,b=map(int, raw_input().split())

all=[]

for i in range(1,66):
    for j in range(1,i):
        s='1'*i
        s=s[:j] + '0' + s[j+1:]
        s='0'+s
        all.append(int(s,2))
all.sort()
aa=0
for i in all:
    if i >= a:
        break
    aa+=1

bb=0
for i in all:
    if i > b:
        break
    bb+=1
print bb-aa
