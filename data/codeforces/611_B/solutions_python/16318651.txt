n, m = map(int, raw_input().split())

s = ['1']*(len(bin(n)) - 2)

i = 1
if(len(s) == 1):
    s.append('1')

s[i] = '0'
counter = 0

while int("".join(s), 2) <= m:

    number = int("".join(s), 2)
    if(number >= n and number <= m):
        counter += 1
    s[i] = '1'
    i += 1
    if(i == len(s)):
        s.append('1')
        i = 1

    s[i] = '0'


print counter
