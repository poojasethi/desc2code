#include <bits/stdc++.h>

using namespace std;

int main()
{
	long long i,j,a,b,p;
	int	Ans=0;

	scanf("%lld%lld",&a,&b);
	for(j=0;j<=63;++j)
	{
		p=0;
		for(i=0;i<=j;++i)
			p|=1LL<<i;
		for(i=0;i<j;++i)
		{
			long long t=p&~(1LL<<i);
			if(t>=a && t<=b)Ans++;
		}
	}
	printf("%d\n",Ans);
	return 0;
}
