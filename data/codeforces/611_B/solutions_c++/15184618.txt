#include<bits/stdc++.h>
using namespace std;
typedef long long int lint;
lint n,m,res;
lint f(lint k,lint cnt){
	if(k>m)
		return 0;
	if(k<=m && k>=n && cnt)
		res++;
	if(!cnt)
		f(2*k,cnt+1);
	f(2*k+1,cnt);
} 
int main(){
	scanf("%lld %lld",&n,&m);
	res=0;
	f(1,0);
	printf("%lld\n",res);
	return 0;
}
