#define per(i,a,n) for (int i=n-1;i>=a;i--)
#define pb push_back
#define mp make_pair
#define all(x) (x).begin(),(x).end()
#define SZ(x) ((int)(x).size())
#define fi first
#define se second
typedef vector<int> VI;
typedef long long ll;
typedef pair<int,int> PII;
const ll mod=1000000007;
ll powmod(ll a,ll b) {ll res=1;a%=mod;for(;b;b>>=1){if(b&1)res=res*a%mod;a=a*a%mod;}return res;}
// head

const int N=50100;
struct task { int t,s,p,id;}tk[N];
bool cmp(const task &a,const task &b) { return a.t<b.t;}
PII kt[N];
ll mt[N],T;
int n,sp;
set<int> hs;
priority_queue<PII> st;
ll check(int p) {
	rep(i,0,n) {
		ll re=(i==n-1)?1ll<<60:(tk[i+1].t-tk[i].t),ct=tk[i].t;
		int pr=tk[i].p==-1?p:tk[i].p;
		kt[i]=mp(tk[i].s,tk[i].id);
		st.push(mp(pr,i));
		while (!st.empty()&&re) {
			int pr=st.top().fi,id=st.top().se; st.pop();
			if (kt[id].fi<=re) {
				re-=kt[id].fi;
				ct+=kt[id].fi;
				mt[kt[id].se]=ct;
			} else {
				kt[id].fi-=re;
				st.push(mp(pr,id));
				re=0;
			}
		}
	}
	return mt[sp];
}
int main() {
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
	scanf("%d",&n);
	rep(i,0,n) {
		scanf("%d%d%d",&tk[i].t,&tk[i].s,&tk[i].p); tk[i].id=i;
		if (tk[i].p==-1) sp=i; else hs.insert(tk[i].p),tk[i].p*=2;
	}
	sort(tk,tk+n,cmp);
	scanf("%I64d",&T);
	int l=0,r=1000000010;
	while (l+1<r) {
		int md=(l+r)>>1;
		if (check(2*md+1)<=T) r=md;
		else l=md;
	}
	while (hs.count(r)) ++r; 
	check(2*r);
	printf("%d\n",r);
	rep(i,0,n) printf("%I64d ",mt[i]);
}
