#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

int main()
{
  int N;scanf("%d",&N);
  static int A[400];
  for(int i=0;i<N;i++)scanf("%d",&A[i]);
  int Ans=-1;
  for(int i=0;i<N;i++)
    for(int j=i+1;j<N;j++)
      {
        int M=A[i]-1,Cost_0=1;
        for(int k=i+1;k<=j;k++)
          {Cost_0+=M/A[k];M%=A[k];}
        int T=M=A[i]-1-M+A[j],Cost_1=0;
        for(int k=0;k<N;k++)
          {Cost_1+=M/A[k];M%=A[k];}
        if(Cost_0<Cost_1&&(Ans==-1||T<Ans))Ans=T;
      }
  printf("%d\n",Ans);
  return 0;
}
