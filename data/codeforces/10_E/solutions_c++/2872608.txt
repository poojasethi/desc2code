#include <iostream>
#include <cstdio>

int a[400], n;

int main() {
    scanf("%d", &n);
    for (int i = 0; i < n; i++) scanf("%d", a + i);
    int ans = -1;
    for (int i = 0; i < n; i++) for (int j = i + 1; j < n; j++) {
        int p = a[i] - 1, cnt0 = 0, cnt1 = 0;
        for (int k = 0; k <= j; k++) cnt0 += p / a[k], p %= a[k];
        p = a[i] - 1 - p + a[j];
        int q = p;
        for (int k = 0; k < n; k++) cnt1 += p / a[k], p %= a[k];
        if (cnt0 + 1 < cnt1 && (ans == -1 || ans > q)) ans = q;
    }
    printf("%d\n", ans);
    return 0;
}
