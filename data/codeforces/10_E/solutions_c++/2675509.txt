
// Codeforces Beta Round #10 E Greedy Change

#include <cstdio>

using namespace std;

int a[400];

int main() {
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i ++) scanf("%d", &a[i]);
	
	int ans = -1;
	for (int i = 0; i < n; i ++)
		for (int j = i + 1; j < n; j ++) {
			int cur = a[i] - 1, cnt1 = 0;
			for (int k = i + 1; k <= j; k ++) {
				cnt1 += cur / a[k];
				cur %= a[k];
			}
			int tmp = a[i] - 1 - cur + a[j];
			cur = tmp;
			int cnt2 = 0;
			for (int k = 0; k < n; k ++) {
				cnt2 += cur / a[k];
				cur %= a[k];
			}
			if (cnt1 + 1 < cnt2 && (ans == -1 || tmp < ans)) ans = tmp;
		}
	
	printf("%d\n", ans);
	
	return 0;
}
