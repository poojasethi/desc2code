#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
int n,c[1000],ans=-1;
int main() {
    int i,j,k;
    scanf("%d",&n);
    fo(i,1,n)scanf("%d",c+i);
    fo(i,2,n)
        fo(j,i,n)
        {
            int t,w,v=c[i-1]-1,sv=0,sw=0;
            fo(k,i,j)sv+=v/c[k],v%=c[k];
            t=w=c[i-1]-1-v+c[j];
            fo(k,1,n)sw+=w/c[k],w%=c[k];
            if(sv+1<sw&&(ans==-1||t<ans))ans=t;
        }
    printf("%d\n",ans);
    return 0;
}
