#include <cstdio>
int c[1000];
int main()
{
	//freopen("input.txt","r",stdin);
	int tmp,i,n,j,c1,c2,a,b,k,ans=1000000000;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d",&c[i]);
	for(i=1;i<=n;i++)
		for(j=i;j<=n;j++)
		{
			a = c[j]; b = c[i]-1;
			for(c1=1,k=1;k<=j;k++)
			{
				c1 += b/c[k];
				a += b/c[k]*c[k];
				b %= c[k];
			}
			tmp = a;
			for(c2=0,k=1;k<=n;k++)
			{
				c2 += a/c[k];
				a %= c[k];
			}
			if(a < ans && c1 < c2)
				ans = tmp;
		}
	printf("%d\n",ans<1000000000?ans:-1);
}
