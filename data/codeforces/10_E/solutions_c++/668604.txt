#include <iostream>
using namespace std;
#define fo(i,a,b) for (int i=a; i<=b; ++i)
#define min(a,b) ((a)<(b)?(a):(b))

const int mn=410;

int c[mn];

int main(){
	int n,Ans=2147483647;
	cin>>n;
	fo (i,1,n) cin>>c[i];
	fo (i,2,n) fo (j,i,n){
		int S1=c[i-1]-1,M=1;
		fo (k,i,j) M+=S1/c[k],S1%=c[k];
		int S2=c[i-1]-1-S1+c[j],G=0;
		fo (k,1,n) G+=S2/c[k],S2%=c[k];
		if (M<G) Ans=min(Ans,c[i-1]-1-S1+c[j]);
	}
	if (Ans==2147483647) Ans=-1;
	cout<<Ans<<endl;
}
