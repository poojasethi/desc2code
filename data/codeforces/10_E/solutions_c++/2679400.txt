#include<cstdio>
int i,j,k,x,y,cnt,t,ans=-1,n,c[405];
int main()
{
	scanf("%d",&n); for (i=0;i<n;i++) scanf("%d",&c[i]);
	for (i=0;i<n;i++) for (j=i;j<n;j++){
		cnt=1,x=c[j],y=c[i]-1;
		for (k=0;k<=j;k++) t=y/c[k],cnt+=t,x+=t*c[k],y%=c[k];
		for (k=0,y=x;k<n;k++) cnt-=y/c[k],y%=c[k];
		if (cnt<0 && (ans<0 || x<ans)) ans=x;
		}
	printf("%d\n",ans); return 0;
}
