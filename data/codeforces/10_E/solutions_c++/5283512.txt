#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int n , ans , a[401];

int main(){

	scanf("%d",&n);
	for(int i = 1 ; i <= n ; i++) 
		scanf("%d", &a[i]);

	ans = 1<<30;

	for(int i = 2 ; i <= n ; i++){
		for(int j = i ; j <= n ; j++){
			int x = a[i-1]-1;
			int y = 0;
			for (int k = i ; k <= j ; k++){
				y += x/a[k];
			    x %= a[k];
			}
			x = a[i-1]-1-x+a[j];
			y++;
			int x1 = x;
			if(x1 < ans){
				for(int k = 1 ; k <= n ; k++){
					y -= x/a[k];
			        x %= a[k];
				}
		        if(y < 0) 
			    	ans = x1;
			}
		}
	}
	if(ans == 1<<30) 
		puts("-1");
	else 
		printf("%d\n",ans);
	return 0;
}
