#include<cstdio>
#include<iostream>
using namespace std;

int n;
long long ans=-1;
int a[510];

int main()
{int i,j,k;
 long long s,t,cnt1,cnt2;
 
 scanf("%d",&n);
 
 for(i=1;i<=n;i++) 
   scanf("%d",&a[i]);
 
 for(i=1;i<=n;i++)
   for(j=i+1;j<=n;j++)
   {s=a[i]-1;
    cnt1=1;
    for(k=i;k<=j;k++)
    {cnt1+=s/a[k];
     s-=s/a[k]*a[k];
    }
    s=a[i]-1-s+a[j];
    t=s;
    cnt2=0;
    for(k=1;k<=n;k++)
    {cnt2+=t/a[k];
     t-=t/a[k]*a[k];
    }
    //cout<<i<<" "<<j<<" "<<s<<" "<<cnt1<<" "<<cnt2<<" "<<ans<<endl;
    if(cnt1<cnt2 && (ans==-1 || s<ans))
      ans=s;
   }
 
 cout<<ans<<endl;
 
 return 0;
}
