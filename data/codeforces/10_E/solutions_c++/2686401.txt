#include <iostream>

using namespace std;

#define  FOR(i, a, b)   for (int i = a; i <= b; ++ i)

typedef  long long  LL;

const  int  maxn  =  401;

int n;
LL c[maxn], g[maxn], w = -1;

LL greedy(LL x) {
	LL ret = 0;
	FOR(i, 1, n) {
		g[i] = x / c[i];
		x %= c[i];
		ret += g[i];
	}
	return ret;
}

int main() {
	cin >> n;
	FOR(i, 1, n)  cin >> c[i];
	FOR(i, 2, n) 
		FOR(j, i, n) {
			greedy(c[i-1] - 1);
			LL x = 0, s = 0;
			FOR(k, i, j)  x += g[k], s += g[k] * c[k];
			s += c[j],  ++ x;
			if (x < greedy(s)) 
				if (w == -1 || w > s)  w = s;
		}
	cout << w << endl;
	return 0;
}
