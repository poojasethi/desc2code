#include <iostream>

using namespace std;

int v[401];

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i)
		cin >> v[i];
	int ans = -1;
	for (int i = 0; i < n; ++i)
		for (int j = i; j < n; ++j) {
			int p = v[i] - 1, sum = v[j], count = 1;
			for (int k = i + 1; k <= j; ++k) {
				count += p / v[k];
				sum += p / v[k] * v[k];
				p %= v[k];
			}
			p = sum;
			for (int k = 0; k < n; ++k) {
				count -= p / v[k];
				p %= v[k];
			}
			if (count < 0 && (ans == -1 || sum < ans))
				ans = sum;
		}
	cout << ans << endl;
	return 0;
}
