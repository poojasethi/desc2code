#include <cstdio>
#include <cstring>
int a[401], ans, n, x, y, z, w;
int main()
{
//    freopen("cf10e.in", "r", stdin);
//    freopen("cf10e.out", "w", stdout);
    scanf("%d", &n);
    ans = 1e9;
    for (int i = 1; i <= n; i++) scanf("%d", &a[i]);
    for (int i = 1; i <= n; i++) for (int j = i; j <= n; j++)
    {
        x = a[i] - 1; y = 1; z = a[j];
        for (int k = 1; k <= j; k++){y += x / a[k]; z += x / a[k] * a[k]; x %= a[k];}
        x = z; w = 0;
        for (int k = 1; k <= n; k++){w += x / a[k]; x %= a[k];}
        if (w > y && z < ans) ans = z;
    }
    if (ans == 1e9) ans = -1;
    printf("%d\n", ans);
    return 0;
}
