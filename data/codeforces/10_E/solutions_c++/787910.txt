#include<stdio.h>

const int maxn=410;

int n;

int data[maxn];

void init()
{
	scanf("%d",&n);
	
	for (int i=1;i<=n;i++)
		scanf("%d",&data[i]);
}

void doit()
{
	int ans=-1;
	
	for (int i=2;i<=n;i++)
		for (int j=i;j<=n;j++)
		{
			int rem=data[i-1]-1,cnt_1=0,cnt_2=0;
			for (int k=i;k<=j;k++)
				cnt_1+=rem/data[k] , rem%=data[k];
			int tmp=rem=data[i-1]-1-rem+data[j];
			for (int k=1;k<=n;k++)
				cnt_2+=rem/data[k] , rem%=data[k];
			if (cnt_1<cnt_2-1 && (ans==-1 || tmp<ans)) ans=tmp;
		}
		
	printf("%d\n",ans);
}

int main()
{
	//freopen("10E.in","r",stdin);
	//freopen("10E.out","w",stdout);
	
	init();
	doit();
	
	return 0;
}
