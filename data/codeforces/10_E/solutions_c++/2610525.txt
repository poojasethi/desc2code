#include<cstdio>
int i,j,k,N,res,M,Ans,Cost0,Cost1;
int A[405];
int main()
{
	scanf("%d",&N);
	for(i=1;i<=N;++i) scanf("%d",&A[i]);
	Ans=-1;
	for(i=1;i<=N;++i)
		for(j=i+1;j<=N;++j)
		{
			M=A[i]-1,Cost0=1;
			for(k=i+1;k<=j;++k)
			{
				Cost0+=M/A[k];
				M%=A[k];
			}
			res=M=A[i]-1-M+A[j],Cost1=0;
			for(k=1;k<=N;++k)
			{
				Cost1+=M/A[k];
				M%=A[k];
			}
			if(Cost0<Cost1&&(Ans==-1||res<Ans)) 
				Ans=res;
      }
  printf("%d\n",Ans);
  return 0;
}
