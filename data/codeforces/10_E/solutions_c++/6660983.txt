#include<iostream>
#define Nmax 400

int main(){

    int v[Nmax + 1],n,k,pp,tmp,c;
	std::cin>>n;

	int i,j;
	for(i=0;i<n;++i)
		std::cin>>v[i];
    int sol=-1;
	for(i=0;i<n;++i)
		for(j=i;j<n;++j){
			pp=v[i]-1;
			tmp=v[j];
			c=1;
			for(k=0;k<=j;++k){
                c+=pp/v[k];
                tmp+=pp/v[k]*v[k];
                pp%=v[k];
			}
			for(pp=tmp,k=0;k<n;++k){
			    c-=pp/v[k];
			    pp%=v[k];
			}
			if(c<0&&(sol<0||sol>tmp))
                sol=tmp;
		}
	std::cout<<sol;
}
