#include <iostream>
#include <cstdio>
using namespace std;
int n,x,a[410],b[410],c[410],ans=2100000000;

void grd(int x,int b[410])
{
	for (int i=1;i<=n;i++)
	{
		b[i]=x/a[i];
		x=x%a[i];
		b[i]+=b[i-1];
	}
}
int main()
{
	
	cin>>n;
	int i;
	for (i=1;i<=n;i++)
	{
		cin>>a[i];
	}
	for (i=2;i<=n;i++)
	{
		grd(a[i-1]-1,b);
		for (int j=i;j<=n;j++)
		{
			x=a[j];
			for (int k=i;k<=j;k++)
			{
				x+=(b[k]-b[k-1])*a[k];
			}
			grd(x,c);
			if(c[n]>b[j]+1 && x<ans)
			{
				ans=x;
			}
		}
	}
	if (ans<2100000000)
		cout<<ans<<endl;
	else
		cout<<-1<<endl;
}
