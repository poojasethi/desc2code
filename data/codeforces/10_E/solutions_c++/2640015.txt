#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int ans,tt;
int n,sw,ww,a[500],i,j,k;

int main()
{
	//freopen("1.in","r",stdin);
	//freopen("1.out","w",stdout);
	scanf("%d",&n);
	ans=1e9;
	for(i=0;i<=n;i++)
	scanf("%d",&a[i]);
	for(i=1;i<=n-1;i++)
		for(j=1;j<=i;j++)
		{
			sw=1;
			tt=a[j-1]-1;
			for(k=j;k<=i;k++)
			{
				sw+=tt/a[k];
				tt%=a[k];
			}
			ww=tt=a[j-1]-1-tt+a[i];
			for(k=0;k<n;k++)
			sw-=tt/a[k],tt%=a[k];
			if (sw<0&&ww<ans)ans=ww;
			//cout<<i<<' '<<j<<' '<<sw<<endl;
		}
		if (ans==1e9) ans=-1;
		cout<<ans;
}
