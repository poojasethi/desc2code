#include<cstdio>
#define rep(i,n) for(int i=0;i<n;i++)
#define For(i,n) for(int i=1;i<=n;i++)
#define FOR(i,a,b) for(int i=a;i<=b;i++)
typedef long long ll;
const ll inf=~0ULL>>1;
ll t,ans=inf;
int n,cnt,tmp,a[401];
int main()
{
	//freopen("CF-10-E.out","w",stdout);
	scanf("%d",&n);
	rep(i,n) scanf("%d",&a[i]);
	For(j,n-1)
	    For(i,j)
	    {
			t=a[i-1]-1;cnt=1;
			FOR(k,i,j) cnt+=t/a[k],t%=a[k];
			tmp=t=a[i-1]-1-t+a[j];
			rep(k,n) cnt-=t/a[k],t%=a[k];
			if (cnt<0 && tmp<ans) ans=tmp;
		}
	if (ans==inf) ans=-1;
	printf("%I64d",ans);
	return 0;
}
