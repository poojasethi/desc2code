#include <bits/stdc++.h>
using namespace std;
int n;
int a[410];

int main()
{
    int i, j, k;
    int ans = -1;
    scanf("%d", &n);
    for (i = 1; i <= n; i++)  scanf("%d", &a[i]);
    for (i = 1; i < n; i++)
     for (j = i + 1; j <= n; j++)
    {
        int M = a[i] - 1, cost0 = 1;
        for (k = i + 1; k <= j; k++)
            cost0 += M / a[k], M %= a[k];
        int T = M = a[i] - 1 - M + a[j], cost1 = 0;
        for (int k = 1; k <= n; k++)
        {
            cost1 += M / a[k];
            M %= a[k];
        }
        if (cost0 < cost1 && (ans == -1  || T < ans))   ans = T;
    }
    printf("%d\n", ans);
     return 0;
}
