#include <cstdio>
using namespace std;
const int MAXN = 450,oo = ~0U>>2;
int n,a[MAXN],ans,cnt,sum,candi;

int main(){
	scanf("%d",&n);
	ans = oo;
	for(int i = 0;i < n;++ i) scanf("%d",&a[i]);
	for(int i = 1;i < n;++ i)
		for(int j = i;j < n;++ j){
			cnt = 0; sum = a[i - 1] - 1;
			for(int k = i,tmp;k <= j;++ k){
				tmp = sum / a[k];
				sum %= a[k];
				cnt += tmp;
			}
			candi = a[i - 1] - 1 - sum + a[j];
			++ cnt;
			if(candi >= ans) continue;
			sum = candi;
			for(int k = 0,tmp;k < n;++ k){
				tmp = sum / a[k];
				sum %= a[k];
				cnt -= tmp;
			}
			if(cnt < 0) ans = candi;
		}
	printf("%d\n",ans < oo ? ans : -1);
	return 0;
}
