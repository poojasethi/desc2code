#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[401],ans;
int main(){
  scanf("%d",&n);
  for(int i=0;i<n;i++) scanf("%d",&a[i]);
  ans = -1;
  for(int i=0;i<n;i++)
    for(int j=i+1;j<n;j++){
      int p = a[i]-1, c1 = 1, c2 = 0, sv = a[j];
      for(int k=i+1;k<=j;k++) {
    c1 += p/a[k]; sv += p/a[k]*a[k]; p %= a[k];
      }
      p = sv;
      for(int k=0;k<n;k++){
    c2 += p/a[k]; p %= a[k];
      }
      if(c2 > c1 && (ans > sv || ans == -1)) ans = sv;
    }
  printf("%d\n",ans);
}
