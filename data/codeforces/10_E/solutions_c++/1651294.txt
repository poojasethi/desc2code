#include <iostream>
using namespace std;
int n,i,a[1111],j,l,ans=-1,s,k,ds;
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++)
		{
			int s=a[i]-1,k1=1,k2=0;
			for(l=0;l<=j;l++)
			{
				k1+=s/a[l];
				s%=a[l];
			}
			s=a[i]+a[j]-1-s;
			ds=s;
			for(l=0;l<n;l++)
			{
				k2+=s/a[l];
				s%=a[l];
			}
			if ((k2>k1)&&((ans==-1)||(ds<ans)))
				ans=ds;
		}
	cout<<ans<<endl;
	return 0;
}
