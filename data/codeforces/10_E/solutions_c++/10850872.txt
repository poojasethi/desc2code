#include<bits/stdc++.h>

using namespace std;

int a[402],n,i,j,k,z,nor;
int ans=2000000000,xar,tmp;

main()
{
 cin>>n;
 for (i=0;i<n;i++) cin>>a[i];
 for (i=1;i<n;i++)
  for (k=i;k<n;k++)
   {
    z=a[i-1]-1; nor=1; tmp=a[k];
    for (j=0;j<=k;j++) tmp+=(z/a[j])*a[j],nor+=z/a[j],z%=a[j];
     z=tmp; xar=0;
     for (j=0;j<n;j++) xar+=z/a[j],z%=a[j];
     if (xar > nor) ans=min(ans,tmp);
    }
 if (ans >= 2000000000) ans=-1;
 cout<<ans<<endl;
}
