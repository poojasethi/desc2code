#include<iostream>
#include<cstdio>
#include<cmath> 
#include<cstdlib>
#include<algorithm>
using namespace std;
int a[100010],n,k,x,y,i,now;
double ans;
double dist(int z)
{
	return sqrt((x-z)*1.0*(x-z)+y*1.0*y);
}
double go(int l,int r)
{
	return a[r]-a[l]+min(dist(a[l]),dist(a[r]));
}
double back(int l,int r)
{
	return a[r]-a[l]+min(dist(a[r])+abs(now-a[l]),dist(a[l])+abs(now-a[r]));
}
int main()
{
	cin>>n>>k;
	for(i=1;i<=n;i++) scanf("%d",&a[i]);
	cin>>x>>y;
	now=a[k];
	sort(a+1,a+n+1);
	if(k==n+1) ans=go(1,n);
	else{
		ans=back(1,n);
		for(i=1;i<=n;i++)
		{
			if(i>1) ans=min(ans,back(1,i-1)+go(i,n));
			if(i<n) ans=min(ans,go(1,i)+back(i+1,n));
		}
	}
	printf("%.20f\n",ans);
	return 0;
}
