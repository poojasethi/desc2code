#include<cstdio>
#include<cmath>
#include<algorithm>
using namespace std;
const int maxn=100000+10;
typedef double lf;
lf a[maxn],x,y,mx,ans;
lf dis(int i){
	return sqrt((x-a[i])*(x-a[i])+y*y);
}
lf cas1(int l,int r){
	return a[r]-a[l]+min(dis(l),dis(r));
}
lf cas2(int l,int r){
	return a[r]-a[l]+min(dis(l)+fabs(mx-a[r]),dis(r)+fabs(mx-a[l]));
}
int n,k;
int main(){
//	freopen("picture.in","r",stdin);
//	freopen("picture.out","w",stdout);
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
		scanf("%lf",&a[i]);
	scanf("%lf%lf",&x,&y);
	mx=a[k];
	sort(a+1,a+n+1);
	if(k==n+1)
		ans=cas1(1,n);
	else{
		ans=cas2(1,n);
		for(int i=1;i<n;i++){
			ans=min(ans,min(cas1(1,i)+cas2(i+1,n),cas2(1,i)+cas1(i+1,n)));
		}
	}
	printf("%.10lf\n",ans);
	return 0;
}
