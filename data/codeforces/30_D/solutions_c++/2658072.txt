#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int i,n,sx,sy,k,a[200000],kx;
double ans;
double tr(int x)
{
	double xx=a[x]-sx,yy=sy;
	return(sqrt(xx*xx+yy*yy));
}
double work1(int x,int y)
{
	return(a[y]-a[x]+min(tr(x),tr(y)));
}
double work2(int x,int y)
{
	return(a[y]-a[x]+min(tr(x)+abs(a[y]-kx),tr(y)+abs(kx-a[x])));
}
int main()
{
//	freopen("1.in","r",stdin);
//	freopen("1.out","w",stdout);
	scanf("%d%d",&n,&k);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	kx=a[k]; 
	scanf("%d%d",&sx,&sy);
	sort(a+1,a+1+n);
	if (k==n+1) ans=work1(1,n);
	else
	{
		ans=work2(1,n);
		//printf("%.8lf\n",ans);
		for(i=1;i<=n;i++)
		{
		if (i!=1) ans=min(ans,work1(1,i-1)+work2(i,n));
		//	printf("%.8lf\n",ans);
		if (i!=n) ans=min(ans,work2(1,i)+work1(i+1,n));
		//	printf("%.8lf\n",ans);
		}
		//cout<<ans;
	} 
	printf("%.9lf",ans);
}
