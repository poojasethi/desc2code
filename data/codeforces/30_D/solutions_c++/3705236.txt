#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<iostream>
#include<algorithm>
#define MAXN 100005
using namespace std;
int n,k;
double x[MAXN],y,t;
double dis(int d)
{
    return sqrt((x[n+1]-x[d])*(x[n+1]-x[d])+y*y);
}
double roadA(int l,int r)
{
    return x[r]-x[l]+min(dis(r),dis(l));
}
double roadB(int l,int r)
{
    return x[r]-x[l]+min(dis(r)+fabs(t-x[l]),dis(l)+fabs(x[r]-t));
}
int main()
{
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n+1;i++)
        scanf("%lf",&x[i]);
    t=x[k];
    scanf("%lf",&y);
    sort(x+1,x+1+n);
    if(k==n+1)
    {
        printf("%.20lf\n",roadA(1,n));
        return 0;
    }
    double ans=roadB(1,n);
    for(int i=1;i<n;i++)
        ans=min(ans,min(roadA(1,i)+roadB(i+1,n),roadB(1,i)+roadA(i+1,n)));
    printf("%.20lf\n",ans);
    return 0;
}
