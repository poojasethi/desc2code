#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;

const int maxn = 100005;
double a[maxn], tx, x, y, ans;
int n, k;

double dis(double w) { return hypot(x - w, y); }

double walkA(int l, int r) {
    return a[r] - a[l] + min(dis(a[l]), dis(a[r]));
}

double walkB(int l, int r) {
    return a[r] - a[l] + min(dis(a[l]) + fabs(tx - a[r]), dis(a[r]) + fabs(tx - a[l]));
}

int main() {
    int i;
    scanf("%d%d", &n, &k);
    for (i = 1; i <= n; ++i)
        scanf("%lf", a + i);
    scanf("%lf%lf", &x, &y);
    tx = a[k];
    sort(a + 1, a + n + 1);
    if (k > n) ans = walkA(1, n);
    else {
        ans = walkB(1, n);
        for (i = 2; i <= n; ++i)
            ans = min(ans, min(walkA(1, i - 1) + walkB(i, n), walkB(1, i - 1) + walkA(i, n)));
    }
    printf("%.10lf\n", ans);
    return 0;
}
