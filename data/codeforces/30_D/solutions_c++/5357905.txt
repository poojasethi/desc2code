#include <cstdio>
#include <cstring>
#include <algorithm>
#include <cmath>
using namespace std;
int a[200001];
int i,j,n,m,k,x,y,p,q,data;
double ans,ans1,ans2;
double dist(int z)
{
    return sqrt((x-z)*1.0*(x-z)+y*1.0*y);
}
double go(int l,int r)
{
    return a[r]-a[l]+min(dist(a[l]),dist(a[r]));
}
double back(int l,int r)
{
    return a[r]-a[l]+min(dist(a[r])+abs(data-a[l]),dist(a[l])+abs(data-a[r]));
}
int main()
{
 scanf("%d %d",&n,&k);
 for (i=1;i<=n;i++)scanf("%d",&a[i]);
 scanf("%d",&x);scanf("%d",&y);
 data=a[k];
 sort(a+1,a+n+1);
 if (k==n+1)
  {
   ans=a[n]-a[1]+sqrt(min((x-a[1])*(x-a[1]),(x-a[n])*(x-a[n]))+y*y);
   printf("%lf\n",ans); 
   return 0;
  }
 ans=back(1,n);
 for (i=1;i<=n;i++)
  {
   if (i>1)
    {
     ans=min(ans,back(1,i-1)+go(i,n));
    }
   if (i<n)
    {
     ans=min(ans,go(1,i)+back(i+1,n));
    } 
  }
 printf("%lf\n",ans);
 return 0;
}
