#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,m,i,j,k,l,o,p;
double x[100010],xx,yy,ans,x1;
double sqr(double x){ return x*x;}
double get(int i)
{
    return sqrt(sqr(x[i]-xx)+sqr(yy));
}
double did(int l,int r)
{
    if(r<l) return 0;
    double ans=x[r]-x[l];
    if(l<=m && m<=r) ans+=min(x[m]-x[l]+get(r),x[r]-x[m]+get(l));
    else ans+=min(get(l),get(r));
    return ans;
}
int main()
{
    cin >> n >> m;
    m--;
    for(i=0;i<n;i++) scanf("%lf",x+i);
    x1=x[m];
    sort(x,x+n);
    cin >> xx >> yy;
    if(m==n)
    {
        ans=x[n-1]-x[0]+min(get(n-1),get(0));
        printf("%0.8lf\n",ans);
        return 0;
    }
    else
    {
        ans=1e99;
        for(i=0;i<n;i++)
            if(x[i]==x1) m=i;
        for(i=0;i<n;i++) ans=min(ans,did(0,i-1)+did(i,n-1));
    }
    printf("%0.8lf\n",ans);
    cin >> n;
    return 0;
}
