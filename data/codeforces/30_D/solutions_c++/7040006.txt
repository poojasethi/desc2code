#include<cstdio>
#include<cmath>
#include<algorithm>
#include<cstring>
using namespace std;

double ans, x[100010], xp, yp, xk;

double C(int z){
    return hypot(x[z]-xp, yp); // 计算 直角三角形的斜边长
}
double A(int l,int r){
    return x[r]-x[l]+min(C(l),C(r));
}

double B(int l,int r){
    return x[r]-x[l]+min(abs(xk-x[l])+C(r), abs(xk-x[r])+C(l));
}

int main() {
    int n, k;
    scanf("%d%d", &n, &k);
    for(int i = 1; i <= n; i ++) scanf("%lf", &x[i]);
    scanf("%lf%lf",&xp, &yp);
    xk = x[k];
    sort(x+1, x+n+1);
    if (k <= n) {
        ans = B(1, n);
        for(int i = 2; i <= n; i ++) {
            double t=min(A(1, i-1)+B(i, n), A(i, n)+B(1, i-1));
            if(t < ans) ans=t;
        }
    }
    else ans=A(1, n);
    printf("%.10lf\n", ans);
    return 0;
}
