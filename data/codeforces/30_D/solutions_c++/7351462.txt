#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,k,a[100001],x,y;
double dc(int i){ return sqrt(1.0*(x-a[i])*(x-a[i]) + 1.0*y*y); }
int main(){
  scanf("%d%d",&n,&k); k--;
  for(int i=0;i<n;i++) scanf("%d",&a[i]); 
  scanf("%d%d",&x,&y);
  if(k == n){
    sort(a,a+n);
    printf("%.12lf\n",min(dc(0),dc(n-1)) + a[n-1] - a[0]);
  }
  else{
    k = a[k]; sort(a,a+n);
    double ans = min(dc(n-1) + k-a[0], dc(0) + a[n-1]-k) + a[n-1]-a[0];
    for(int i=1;i<n;i++){
      ans = min(ans , min( min(k-a[0] + a[i-1]-a[0] + dc(i-1) , abs(k-a[i-1]) + a[i-1]-a[0] + dc(0)) + min(dc(i),dc(n-1)) + a[n-1]-a[i] , min(abs(k-a[i]) + a[n-1]-a[i] + dc(n-1) , a[n-1]-k + a[n-1]-a[i] + dc(i)) + min(dc(0),dc(i-1)) + a[i-1]-a[0]) );
    }
    printf("%.12lf\n",ans);
  }
  return 0;
}
