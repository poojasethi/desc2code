#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int n,k;
double x[100005],y[100005];
double dist(int i,int j){
  return sqrt((x[i]-x[j])*(x[i]-x[j])+(y[i]-y[j])*(y[i]-y[j]));
}
double mindist(){
  if (k==n+1) return min(dist(0,1),dist(0,n))+dist(1,n);
  double res=min(dist(0,1)+dist(1,n)+dist(n,n+1),dist(0,n)+dist(1,n)+dist(1,n+1));
  for (int i=1;i<n;i++) res=min(res,min(dist(0,1)+dist(1,i)+dist(i,n+1),dist(0,i)+dist(1,i)+dist(1,n+1))+min(dist(i+1,n+1),dist(n,n+1))+dist(i+1,n));
  for (int i=2;i<=n;i++) res=min(res,min(dist(0,i)+dist(i,n)+dist(n,n+1),dist(0,n)+dist(i,n)+dist(i,n+1))+min(dist(1,n+1),dist(i-1,n+1))+dist(1,i-1));
  return res;
}
int main(){
  scanf("%d%d",&n,&k);
  for (int i=1;i<=n+1;i++) scanf("%lf",&x[i]);
  scanf("%lf",&y[n+1]);
  x[0]=x[k];
  y[0]=y[k];
  sort(x+1,x+n+1);
  printf("%.10lf\n",mindist());
}
