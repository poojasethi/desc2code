#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
#include <cmath>
using namespace std;
const int MAXN = 100005;
int n,k,x[MAXN],ex,ey;
double ans;

double dist(int p){
	return sqrt(double(ex - p) * double(ex - p) + double(ey) * double(ey));
}

double after(int l,int r){
	return min(dist(x[l]),dist(x[r])) + x[r] - x[l];
}

double before(int l,int r){
	return min(fabs(k - x[l]) + dist(x[r]),fabs(k - x[r]) + dist(x[l])) + x[r] - x[l];
}

int main(){
	scanf("%d%d",&n,&k);
	for(int i = 0;i < n;++ i) scanf("%d",&x[i]);
	scanf("%d%d",&ex,&ey);
	if(k == n + 1){
		sort(x,x + n);
		ans = after(0,n - 1);
	}else{
		k = x[k - 1]; sort(x,x + n);
		ans = before(0,n - 1);
		for(int i = 0;i < n;++ i){
			if(i && x[i] <= k) ans = min(ans,before(i,n - 1) + after(0,i - 1));
			if(i < n - 1 && x[i] >= k) ans = min(ans,before(0,i) + after(i + 1,n - 1));
		}
	}
	printf("%.15lf\n",ans);
	return 0;
}
