#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cmath>
#include <algorithm>
#define il inline
#define db double
#define min(_,__) ((_)<(__) ? (_):(__))
using namespace std;

const int N=100000+5;

int n,k;
db x[N],y,w;

il db Dis(int d) {
    return sqrt((x[n+1]-x[d])*(x[n+1]-x[d])+y*y);
}
il db Ask1(int l,int r) {
    return x[r]-x[l]+min(Dis(r),Dis(l));
}
il db Ask2(int l,int r) {
    return x[r]-x[l]+min(Dis(r)+fabs(w-x[l]),Dis(l)+fabs(x[r]-w));
}
int main() {
    scanf("%d %d",&n,&k);
    for(int i=1;i<=n+1;i++)
        scanf("%lf",&x[i]);
    w=x[k];
    scanf("%lf",&y);
    sort(x+1,x+n+1);
    if(k==n+1) {
        printf("%.6lf\n",Ask1(1,n));
        return 0;
    }
    db ans=Ask2(1,n);
    for(int i=1;i<n;i++)
        ans=min(ans,min(Ask1(1,i)+Ask2(i+1,n),Ask2(1,i)+Ask1(i+1,n)));
    printf("%.6lf\n",ans);
    return 0;
}
