#include<cmath>
#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define For(i,n) for(int i=1;i<=n;i++)
#define N 100010
using namespace std;
int n,m,k,a[N],x,y;
double ans,tmp;
inline double len(int p)
{
	return sqrt(1.*(a[p]-x)*(a[p]-x) + 1.*y*y);
}
int main()
{
	scanf("%d%d",&n,&k);
	For(i,n) scanf("%d",&a[i]);
	scanf("%d%d",&x,&y);
	m=a[k];
	sort(a+1,a+n+1);
	
	if (k>n) ans=a[n]-a[1]+min(len(1),len(n));
	else
	{
		For(i,n) if (a[i]==m) {k=i;break;}
		ans=a[n]-a[1]+min(a[n]-m+len(1),m-a[1]+len(n));
		For(i,n-1)
		{
			if (k<=i)
			{
				tmp=a[i]-a[1]+min( a[i]-m+len(1) , m-a[1]+len(i));
				tmp+=a[n]-a[i+1]+min(len(i+1),len(n));
				if (tmp<ans) ans=tmp;
			}
			else
			{
				tmp=a[n]-a[i+1]+min( a[n]-m+len(i+1) , m-a[i+1]+len(n));
				tmp+=a[i]-a[1]+min(len(1),len(i));
				if (tmp<ans) ans=tmp;
			}
		}
	}
	printf("%.10f",ans);
	return 0;
}
