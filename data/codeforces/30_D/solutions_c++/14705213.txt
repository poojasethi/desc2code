#include<cstdio>
#include<algorithm>
#include<cmath>
int n,k; double ans,all;
struct note{int x,y;} a[100005],tmp;
bool cmp(const note i,const note j){return i.x<j.x;}
double dist(double x,double y){return sqrt(x*x+y*y);}
double D(int i,int j){return dist(a[i].x-a[j].x,a[i].y-a[j].y);}
inline double min(double a,double b){return a<b?a:b;}
double st(int i,int j){return D(i,j)+min(D(k,i)+D(j,0),D(k,j)+D(i,0));}
double ed(int i,int j){return D(i,j)+min(D(i,0),D(j,0));}
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) scanf("%d",&a[i].x);
	scanf("%d%d",&a[0].x,&a[0].y),tmp=a[k];
	std::sort(a+1,a+n+1,cmp); if (k<=n) k=-1;
	for (int i=1;i<=n && k<0;i++) if (a[i].x==tmp.x && a[i].y==tmp.y) k=i;
	ans=st(1,n); if (k>n) ans=ed(1,n); else
	for (int i=1;i<n;i++) ans=min(ans,min(st(1,i)+ed(i+1,n),ed(1,i)+st(i+1,n)));
	printf("%.9lf\n",ans); return 0;
}
