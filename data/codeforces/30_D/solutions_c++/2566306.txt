#include<iostream>
#include<cstdio>
#include<cmath> 
#include<cstdlib>
#include<algorithm>
using namespace std;
double a[100010],x,y,ans;
int n,k;
double dis(int i){
    return sqrt(y*y+(x-a[i])*(x-a[i]));
}
double go(int l,int r){
    return a[r]-a[l]+min(fabs(a[r]-a[k])+dis(l),fabs(a[k]-a[l])+dis(r));
}
double back(int l,int r){
    return a[r]-a[l]+min(dis(l),dis(r));
}
int main(){
    //freopen("AC.in","r",stdin);
    //freopen("AC.out","w",stdout);
    scanf("%d%d",&n,&k);
    for(int i=1;i<=n;i++)scanf("%lf",&a[i]);
    double ram=a[k];
    scanf("%lf%lf",&x,&y);
    sort(a+1,a+n+1);
    if(k==n+1)ans=back(1,n);
    else{
        for(int i=1;i<=n;i++)if(ram==a[i])k=i;
        ans=go(1,n);
        //
        for(int i=1;i<=n;i++){
            if(i!=1)ans=min(ans,go(1,i-1)+back(i,n));
            if(i!=n)ans=min(ans,go(i+1,n)+back(1,i));
        }
    }
    printf("%.12lf\n",ans);
    return 0;
}
