#include<iostream>
#include<string>

#define FOR(i,a,b) for(i=a;i<=b;i++)
#define ROF(i,a,b) for(i=a;i>=b;i--)

using namespace std;

string a,b,c;
int A[300],B[300],C[300],rb=1e6,x,y,rx,ry,maxx,i;
int main()
{
     cin>>a>>b>>c;
     for (i=0;i<a.size();i++) A[a[i]]++;
     for (i=0;i<b.size();i++) B[b[i]]++;
     for (i=0;i<c.size();i++) C[c[i]]++;
     for (i='a';i<='z';i++)if (B[i])rb=min(rb,A[i]/B[i]);
     for (x=0;x<=rb;x++)
     {
          int y=1e6;
          for (i='a';i<='z';i++)
               if (C[i])
          {
               int z=A[i]-x*B[i];
               y=min(y,z/C[i]);
          }
          if (x+y>=maxx)
               maxx=x+y,rx=x,ry=y;
     }
     for (i=1;i<=rx;i++)cout<<b;
     for (i=1;i<=ry;i++)cout<<c;
     for (i='a';i<='z';i++)
     {
          A[i]-=rx*B[i]+ry*C[i];
          while (A[i]>0)cout<<char(i),A[i]--;
     }
     return 0;
}
