#include<iostream>
#include<string>
#include<cmath>
using namespace std;
int main(){
    string s,a,b;
    cin>>s>>a>>b;
    int l=s.length(),m=a.length(),n=b.length();
    int fs[26]={0},fa[26]={0},fb[26]={0};
    for(int i=0;i<l;i++)fs[s[i]-'a']++;
    for(int i=0;i<m;i++)fa[a[i]-'a']++;
    for(int i=0;i<n;i++)fb[b[i]-'a']++;
    int x=l;
    //cout<<"ghuk\n";
    for(int i=0;i<26 ;i++){if( fa[i]!=0 ) x=min(x,fs[i]/fa[i]);}
   // int y=l;
   int res=0,a1=0,b1=0;
   for(int i = x;i>=0;i--){
        int y=l;
        for(int j=0;j<26 ;j++){if( fb[j]!=0) y=min(y,(fs[j]-i*fa[j])/fb[j]);}
        if((i+y)>res)
        res=i+y,a1=i,b1=y;
    }
    for(int i=0;i<26;i++)fs[i]-=(a1*fa[i]+b1*fb[i]);
    while(a1--)cout<<a;
    while(b1--)cout<<b;
    for(int i=0;i<26;i++) {while(fs[i]--)cout<<char('a'+i);}
    //cout<<res<<endl;
}
