#include <iostream>
#include <algorithm>
using namespace std;
int main(){
    string s,a,b;
    cin>>s>>a>>b;
    int ss[30]={0},aa[30]={0},bb[30]={0},i,j,sum1=1000000,ans1=0,ans2=0;
    for (i=0;i<s.size();i++) ss[s[i]-'a']++;
    for (i=0;i<a.size();i++) aa[a[i]-'a']++;
    for (i=0;i<b.size();i++) bb[b[i]-'a']++;
    for (i=0;i<26;i++){
        if (aa[i]!=0){
            sum1=min(sum1,ss[i]/aa[i]);
        }
    }
    for (i=0;i<=sum1;i++){
        int res=1000000;
        for (j=0;j<26;j++){
            if (bb[j]!=0){
                res=min(res,(ss[j]-i*aa[j])/bb[j]);
            }
        }
        if (i+res>ans1+ans2) {
            ans1=i;
            ans2=res;
        }
    }
    for (i=0;i<ans1;i++) cout<<a;
    for (i=0;i<ans2;i++) cout<<b;
    for (i=0;i<26;i++){
        for (j=0;j<ss[i]-ans1*aa[i]-ans2*bb[i];j++){
            cout<<char(i+'a');
        }
    }
}
