#include <bits/stdc++.h>
using namespace std;
int main()
{
	char a[100005],b[100005],c[100005];
	int i;
	scanf("%s%s%s",a,b,c);
	int d1[26],d2[26],d3[26];
	for(i=0;i<26;i++)
		d1[i]=d2[i]=d3[i]=0;
	for(i=0;a[i]!='\0';i++)
		d1[a[i]-'a']++;
	for(i=0;b[i]!='\0';i++)
		d2[b[i]-'a']++;
	for(i=0;c[i]!='\0';i++)
		d3[c[i]-'a']++;
	int min,val,flag=1,k=0,k1=0,max=0;
	while(flag)
	{
		min=100000;
		for(i=0;i<26;i++)
		{
			val=d1[i]-k*d2[i];
				if(val<0)
					{flag=0;
				break;}
			if(d3[i]!=0)
			{
				if(val/d3[i]<min)
					min=val/d3[i];
			}
		}
		if(flag&&(k+min)>max)
			max=k+min,k1=k;
	k++;}
	//printf("%d %d %d",max,k,k1);
	for(i=0;i<k1;i++)
		printf("%s",b);
	for(i=0;i<max-k1;i++)
		printf("%s",c);
	for(i=0;i<26;i++)
		d1[i]-=(k1*d2[i]+(max-k1)*d3[i]);
	for(i=0;i<26;i++)
	{
		while(d1[i]>0)
			printf("%c",(i+97)),d1[i]--;
	}
	return 0;
}
