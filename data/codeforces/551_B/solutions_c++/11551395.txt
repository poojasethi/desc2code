#include<cstdio>
#include<algorithm>
#include<cstring>
#define MAXN 100005
using namespace std;
char a[MAXN],b[MAXN],c[MAXN];
int len,sa[26],sb[26],sc[26];
int kb,kc,ansb,ansc;
int main()
{
	scanf("%s%s%s",a+1,b+1,c+1);
	len=strlen(a+1);
	for (int i=1;i<=len;i++)sa[a[i]-'a']++;
	len=strlen(b+1);
	for (int i=1;i<=len;i++)sb[b[i]-'a']++;
	len=strlen(c+1);
	for (int i=1;i<=len;i++)sc[c[i]-'a']++;
	kb=0x7fffffff;
	for (int i=0;i<26;i++)
	if (sb[i])kb=min(kb,sa[i]/sb[i]);
	for (int i=0;i<=kb;i++)
	{
		kc=0x7fffffff;
		for (int j=0;j<26;j++)
		if (sc[j])kc=min(kc,(sa[j]-sb[j]*i)/sc[j]);
		if (kc+i>ansb+ansc)ansb=i,ansc=kc;
	}
	for (int i=0;i<26;i++)
	{
		sa[i]-=ansb*sb[i];
		sa[i]-=ansc*sc[i];
	}
	for (int i=1;i<=ansb;i++)printf("%s",b+1);
	for (int i=1;i<=ansc;i++)printf("%s",c+1);
	for (int i=0;i<26;i++)
	for (int j=1;j<=sa[i];j++)
	putchar('a'+i);
}
