#include <stdio.h>

char s[3][100001];

inline int min(int a, int b){
	if(a>b)
		return b;
	else
		return a;
}

int main(){
	int l,m,n,i,j,k;
	int a[3][26];

	for(i=0;i<3;i++)
		for(j=0;j<26;j++)
			a[i][j]=0;

	for(j=0;j<3;j++){
		scanf("%s", s[j]);

		for(i=0;s[j][i]!='\0';i++)
			a[j][s[j][i]-'a']++;
	}

	for(l=100000,i=0;i<26;i++)
		if(a[1][i]==0);
		else
			l=min(l,a[0][i]/a[1][i]);

	for(i=0,m=0,k=0;i<l+1;i++){
		for(j=0,n=100000;j<26;j++)
			if(a[2][j]==0);
			else
				n=min(n,(a[0][j]-i*a[1][j])/a[2][j]);

		if(i+n>m){
			m=i+n;
			k=i;
		}
	}

	m-=k;

	for(i=0;i<26;i++)
		a[0][i]-=(k*a[1][i]+m*a[2][i]);

	while(k--)
		printf("%s", s[1]);

	while(m--)
		printf("%s", s[2]);

	for(i=0;i<26;i++)
		while(a[0][i]--)
			putchar('a'+i);

	putchar('\n');

	return 0;
}
