#include<iostream>
#include<stdio.h>
#include<string.h>
#include<string> 
using namespace std;

int main()
{
	int bans=0,cans=0;
	string a,b,c;
	cin>>a>>b>>c;
	int an['z'+1]={},bn['z'+1]={},cn['z'+1]={},t['z'+1];
	for(int i=0;i<a.length();i++)
		an[a[i]]++;
	for(int i=0;i<b.length();i++)
		bn[b[i]]++;
	for(int i=0;i<c.length();i++)
		cn[c[i]]++;
	memcpy(t,an,sizeof(an));
	for(int bnum=0,over=false;!over;bnum++)
	{
		int cnum=100000;
		for(int i='a';i<='z';i++)
			if(cn[i])
				cnum=min(cnum,an[i]/cn[i]);
		if(bnum+cnum>bans+cans)
		{
			bans=bnum;
			cans=cnum;
		}
		for(int i='a';i<='z';i++)
		{
			an[i]-=bn[i];
			if(an[i]<0)
				over=true;
		}
	}
	for(int i=0;i<bans;i++)
		cout<<b;
	for(int i=0;i<cans;i++)
		cout<<c;
	for(int i='a';i<='z';i++)
		for(int j=0;j<t[i]-bans*bn[i]-cans*cn[i];j++)
			cout<<char(i);
	return 0;
}
