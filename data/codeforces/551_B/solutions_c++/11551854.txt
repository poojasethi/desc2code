//in the name of God

#include <iostream>
#include <string>
using namespace std;

typedef long long ll;
ll a[333], b[333], c[333], t[333];
string s1, s2, s3;

int main()
{
	cin >> s1;
	for(ll i=0; i<s1.size(); i++)
		a[s1[i]]++;
	
	cin >> s2;
	for(ll i=0; i<s2.size(); i++)
		b[s2[i]]++;
	
	cin >> s3;
	for(ll i=0; i<s3.size(); i++)
		c[s3[i]]++;

	ll bi = 0, bk = 0;

	for(ll i=0; i<=1e5; i++)
	{
		bool flag = 1;
		ll cc = 1e7;
		for(char j='a'; flag && j <= 'z'; j++)
		{
			t[j] = a[j]-i*b[j];
			if(t[j] < 0)
			{
				flag = 0;
				continue;
			}

			ll k = 1e5;
			if(c[j])
				k = t[j]/c[j];

			cc = min(cc, k);
		}

		if(!flag)
			continue;
		if(cc + i > bi+bk)
			bi = i, bk = cc;
	}


	for(ll i=0; i<bi; i++)
		cout << s2;
	for(ll i=0; i<bk; i++)
		cout << s3;

	for(char i='a'; i <= 'z'; i++)
	{
		a[i] -= bi*b[i] + bk*c[i];

		while(a[i] > 0)
		{
			cout << i;
			a[i]--;
		}
	}
	cout << endl;

	return 0;
}
