//In the Name of God

#include<iostream>

using namespace std;

string a,b,c,ans="";
int A[100],B[100],C[100];
int mb=10000000,mc=10000000,maxb,maxc,m;

int main()
{
  cin>>a>>b>>c;
  for(int i=0;i<a.size();i++)
    A[a[i]-'a']++;
  for(int i=0;i<b.size();i++)
    B[b[i]-'a']++;
  for(int i=0;i<c.size();i++)
    C[c[i]-'a']++;
  for(int i=0;i<26;i++)
    if(B[i]!=0)
      mb=min(mb,A[i]/B[i]);
  for(int i=0;i<=mb;i++)
    {
          mc=10000000;
      for(int j=0;j<26;j++)
	if(C[j]!=0)
	  mc=min(mc,(A[j]-(B[j]*i))/C[j]);
      if(i+mc>m)
	{
	  m=i+mc;
	  maxb=i;
	  maxc=mc;
	}
    }
  for(int i=0;i<maxb;i++)
    ans+=b;
  for(int i=0;i<maxc;i++)
    ans+=c;
  for(int i=0;i<26;i++)
    for(int j=0;j<A[i]-(B[i]*maxb)-(C[i]*maxc);j++)
      ans+=char(i+'a');
  cout<<ans<<endl;
  return 0;
}
