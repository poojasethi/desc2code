#include<bits/stdc++.h>
using namespace std;
typedef vector<int> vi;
#define for(i,n) for(i=0;i<n;i++)
#define for1(i,a,b) for(i=a;i<=b;i++)
int i,n,x,ma=1000000,mx=0,k,j,xa,xb,mb;
string s,a,b;
vi wc(26,0),wa(26,0),wb(26,0);
int main(){
    cin >> s >> a >> b;
    for(i,s.size()){wc[s[i]-'a']++;}
    for(i,a.size()){wa[a[i]-'a']++;}
    for(i,b.size()){wb[b[i]-'a']++;}
    for(i,26) if(wa[i]) ma = min(ma,wc[i]/wa[i]);
    for(j,ma+1){
        mb=1000000;
        for(i,26) if(wb[i])  mb = min(mb,(wc[i]-j*wa[i])/wb[i]);
        if(j+mb > mx){
            xb=mb;
            xa=j;
            mx = j+mb;
        }
    }
    for(i,xa) cout << a;
	for(i,xb) cout << b;
	for(i,26){
        int rn = wc[i]-xa*wa[i]-xb*wb[i];
        for(k,rn) printf("%c",'a'+i);
    }
    return 0;
}
