#include <bits/stdc++.h>
using namespace std;
int ac[26],bc[26],cc[26],temp[26];
int main(){
	string a,b,c;
	int i,j;cin>>a>>b>>c;
	for(i=0;i<a.size();++i)
		ac[a[i]-'a']++;
	for(i=0;i<b.size();++i)
		bc[b[i]-'a']++;
	for(i=0;i<c.size();++i)
		cc[c[i]-'a']++;
	int bb=1e9,ck,ret=0,x=0,y=0;
	for(i=0;i<26;++i)
		if(bc[i]!=0)
			bb= min(bb,ac[i]/bc[i]);
	for(i=0;i<26;++i)
		temp[i]= ac[i]-bc[i]*bb;

	int bt = bb;
	for(i=0;i<=bb;++i){
		ck= 1e9;
		for(j=0;j<26;++j)
		if(cc[j]!=0)
			ck= min(ck,temp[j]/cc[j]);
		if(bt+ck>ret){
			ret= bt+ck,x=bt,y=ck;
		}
		for(j=0;j<26;++j)
			temp[j]+=bc[j];
		--bt;
	}
	for(i=0;i<26;++i)
		ac[i]-=bc[i]*x+cc[i]*y;
	for(i=0;i<x;++i)cout<<b;
	for(i=0;i<y;++i)cout<<c;
	for(i=0;i<26;++i)
		for(j=0;j<ac[i];++j)putchar('a'+i);
	putchar('\n');
}
