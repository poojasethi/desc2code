#include<bits/stdc++.h>
#define N 1<<17
using namespace std;
int cnt[26],cnt1[26],cnt2[26];
int main(){
	string a,b,c;
	cin>>a>>b>>c;
	for(int i=0;i<26;i++) cnt[i]=cnt1[i]=cnt2[i]=0;
	for(int i=0;i<a.size();i++){
		cnt[a[i]-'a']++;
	}
	for(int i=0;i<b.size();i++) cnt1[b[i]-'a']++;
	for(int i=0;i<c.size();i++) cnt2[c[i]-'a']++;
	int x=0,y=0,z=0;
	for(int i=0;;i++)
	{
		bool flag=1;
		for(int j=0;j<26;j++)
			if(cnt1[j]*i>cnt[j]) flag=0;
		if(!flag) break;

		int t=N;
		for(int j=0;j<26;j++)
			if(cnt2[j]) t=min(t,(cnt[j]-cnt1[j]*i)/cnt2[j]);

		if(i+t>z)
			z=i+t,x=i,y=t;
	}
	//cout<<cnt_fs<<" "<<cnt_ss<<endl;
	for(int i=0;i<x;i++) cout<<b;
	for(int i=0;i<y;i++) cout<<c;
	for(int i=0;i<26;i++) {
		cnt[i]-=(cnt1[i]*x+cnt2[i]*y);
		for(int j=0;j<cnt[i];j++) printf("%c",'a'+i);
	}
}
