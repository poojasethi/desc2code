#include <iostream>
#include <cstdio>
#define N 100005
using namespace std;
char a[N] , b[N] , c[N] ;
int ca[26] , cb[26] , cc[26] ;
void cnt(char s[],int cnt[]) {
	for(int i=0 ; s[i] ; i++)
		cnt[s[i]-'a']++ ;
}
int main () {
	while(scanf("%s%s%s",a,b,c)==3) {
		int i , j , cnta=1e9 , cntb , ansa=0 , ansb=0 ;
		cnt(a,ca) ;
		cnt(b,cb) ;
		cnt(c,cc) ;
		for(i=0 ; i<26 ; i++)
			if(cb[i]!=0)
				cnta = min(cnta,ca[i]/cb[i]) ;
		for(i=0 ; i<=cnta ; i++) {
			cntb=1e9 ;
			for(j=0 ; j<26 ; j++)
				if(cc[j]!=0)
					cntb = min((ca[j]-i*cb[j])/cc[j],cntb) ;
			if(i+cntb>ansa+ansb) {
				ansa = i ;
				ansb = cntb ;
			}
		}
		for(i=0 ; i<ansa ; i++)
			printf("%s",b) ;
		for(i=0 ; i<ansb ; i++)
			printf("%s",c) ;
		for(i=0 ; i<26 ; i++) {
			ca[i] -= ansa*cb[i] + ansb*cc[i] ;
			for(j=0 ; j<ca[i] ; j++)
				printf("%c",char('a'+i)) ;
		}
		puts("") ;
	}
}
