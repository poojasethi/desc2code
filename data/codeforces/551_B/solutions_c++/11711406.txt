#define N (1<<17)
#include <bits/stdc++.h>
using namespace std;

char a[N],b[N],c[N];
int c1[N],c2[N],c3[N];

void Count(char* a,int* cnt)
{
	int l=strlen(a);
	for(int i=0;i<l;i++)
		cnt[a[i]-'a']++;
}

int main()
{
	cin>>a>>b>>c;
	Count(a,c1),Count(b,c2),Count(c,c3);

	int x=0,y=0,z=0;
	for(int i=0;;i++)
	{
		bool flag=1;
		for(int j=0;j<26;j++)
			if(c2[j]*i>c1[j]) flag=0;
		if(!flag) break;

		int t=N;
		for(int j=0;j<26;j++)
			if(c3[j]) t=min(t,(c1[j]-c2[j]*i)/c3[j]);

		if(i+t>z)
			z=i+t,x=i,y=t;
	}

	for(int i=0;i<x;i++)
		printf("%s",b);
	for(int i=0;i<y;i++)
		printf("%s",c);
	for(int i=0;i<26;i++)
	{
		int cnt=c1[i]-c2[i]*x-c3[i]*y;
		for(int j=0;j<cnt;j++)
			putchar(i+'a');
	}
	cout<<endl;
	return 0;
}
