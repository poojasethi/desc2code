#include<bits/stdc++.h>
using namespace std;
string s1,s2,s3;
int i,n=1e9,a[260],b[260],c[260],ansa,ansb;
main()
{
    cin>>s1>>s2>>s3;
    for (i=0;i<s1.size();i++) a[s1[i]]++;
    for (i=0;i<s2.size();i++) b[s2[i]]++;
    for (i=0;i<s3.size();i++) c[s3[i]]++;
    for (i=97;i<=122;i++)
        if (b[i]) n=min(n,a[i]/b[i]);
    for (i=0;i<=n;i++)
    {
        int p=1e6;
        for (int j=97;j<=122;j++) if (c[j])  p=min(p,(a[j]-b[j]*i)/c[j]);
        if (i+p>ansa+ansb) ansa=i,ansb=p;
    }
    for (i=0;i<ansa;i++) cout<<s2;
    for (i=0;i<ansb;i++) cout<<s3;
    for (i=97;i<=122;i++)
        for (int j=1;j<=a[i]-ansa*b[i]-ansb*c[i];j++)
    {
        char c=i;
        printf("%c",c);
    }
}
