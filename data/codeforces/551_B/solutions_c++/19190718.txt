#include<bits/stdc++.h>
using namespace std;

const int maxn=100010;

int na,nb,nc;
char A[maxn],B[maxn],C[maxn];
int a[30],b[30],c[30];

int main()
{
	//freopen("B.in","r",stdin);
	//freopen(".out","w",stdout);
	scanf("%s%s%s",A,B,C);
	na=strlen(A);nb=strlen(B);nc=strlen(C);
	for(int i=0;i<na;i++) a[A[i]-'a']++;
	for(int i=0;i<nb;i++) b[B[i]-'a']++;
	for(int i=0;i<nc;i++) c[C[i]-'a']++;
	int res=0,resb=0,resc=0,cur,ok=1;
	for(int n=0;ok;n++)
	{
		cur=maxn;
		for(int i=0;i<26&&ok;i++)
		{
			if(a[i]<n*b[i]) ok=0;
			if(c[i]) cur=min(cur,(a[i]-n*b[i])/c[i]);
		}
		if(res<n+cur&&ok)
		{
			resb=n;
			resc=cur;
			res=n+cur;
		}
	}
	for(int i=0;i<resb;i++) printf("%s",B);
	for(int i=0;i<resc;i++) printf("%s",C);
	for(int i=0;i<26;i++) for(int j=0;j<(a[i]-resb*b[i]-resc*c[i]);j++)
		printf("%c",'a'+i);
	printf("\n");
	return 0;
}
