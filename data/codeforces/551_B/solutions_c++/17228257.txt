#include<iostream>
#include<string>
using namespace std;
int t[26],ta[26],tb[26],aa=0,ab=0;
int main()
{
    string s,a,b;
    cin>>s>>a>>b;
    for(int i=0;i<s.size();i++)
		t[s[i]-'a']++;
    for(int i=0;i<a.size();i++)
		ta[a[i]-'a']++;
    for(int i=0;i<b.size();i++)
		tb[b[i]-'a']++;
    int sum=100010;
    for(int i=0;i<26;i++)
		if(ta[i])
			sum=min(sum,t[i]/ta[i]);
    for(int i=0;i<=sum;i++)
    {
        int res=100010;
        for(int j=0;j<26;j++)
			if(tb[j])
				res=min(res,(t[j]-ta[j]*i)/tb[j]);
        if(i+res>aa+ab)
		{
			aa=i;
		    ab=res;
		}
    }
    for(int i=0;i<aa;i++)
		cout<<a;
    for(int i=0;i<ab;i++)
		cout<<b;
    for(int i=0;i<26;i++)
		for(int j=0;j<t[i]-ta[i]*aa-tb[i]*ab;j++)
        cout<<char(i+'a');
}
