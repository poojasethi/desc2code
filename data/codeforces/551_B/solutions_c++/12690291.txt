#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

const int N=100100;
char a[N],b[N],c[N];
int cnta[50],cntb[50],cntc[50];
int i,j,X,Y,msum,resx,resy;

int main()
{
	using namespace std;
	scanf("%s%s%s",a,b,c);
	for (i=0;a[i];i++) cnta[a[i]-'a'+1]++;
	for (i=0;b[i];i++) cntb[b[i]-'a'+1]++;
	for (i=0;c[i];i++) cntc[c[i]-'a'+1]++;
	X=0;
	msum=-1;
	while (1)
	{
		Y=1<<29;
		int tmp=0;
		for (i=1;i<=26;i++)
		{
			tmp=cnta[i]-X*cntb[i];
			if (tmp<0) break;
			if (cntc[i])
				Y=min(tmp/cntc[i],Y);
		}
		if (tmp<0) break;
		if (X+Y>msum)
		{
			msum=X+Y;
			resx=X,resy=Y;
		}
		X++;
	}
	X=Y=0;
	if (msum>0)
	{
		X=resx;Y=resy;
		while (resx--) printf("%s",b);
		while (resy--) printf("%s",c);
	}
	for (i=1;i<=26;i++)
	{
		for (j=1;j<=cnta[i]-X*cntb[i]-Y*cntc[i];j++)
			printf("%c",i+'a'-1);
	}
	puts("");
	return 0;
}
