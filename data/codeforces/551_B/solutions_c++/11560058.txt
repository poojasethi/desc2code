#include<stdio.h>
#include<algorithm>
int n,al,bl,cl,a[200],b[200],c[200],d[200],x,y,i,j,m=999999,k,max,s;
char t[100001],p[100001],q[100001];
int main(){
	scanf("%s%s%s",t,p,q);
	while(t[al]!=0)al++;
	while(p[bl]!=0)bl++;
	while(q[cl]!=0)cl++;
	for(i=0;i<al;i++)
		a[t[i]]++;
	for(i=0;i<bl;i++)
		b[p[i]]++;
	for(i=0;i<cl;i++)
		c[q[i]]++;
	for(i=97;i<123;i++)
		if(b[i]!=0)m=(m<a[i]/b[i]?m:a[i]/b[i]);
	for(i=0;i<=m;i++){
		k=999999;
		for(j=97;j<123;j++)
			d[j]=a[j]-i*b[j];
		for(j=97;j<123;j++)
			if(c[j]!=0)k=(k<d[j]/c[j]?k:d[j]/c[j]);
		if(max<i+k){max=i+k;s=i;}
	}
	y=max-s;
	for(i=0;i<s;i++)
		printf("%s",p);
	for(i=0;i<y;i++)
		printf("%s",q);
	for(i=97;i<123;i++)
		for(j=0;j<a[i]-s*b[i]-y*c[i];j++)
			printf("%c",i);

}
