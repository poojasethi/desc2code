#include<cstdio>
#include<iostream>
#include<cstdio>

using namespace std;


char a[100005],b[100005],c[100005];
int aa[30],bb[30],cc[30];
int main(){
	gets(a);gets(b);gets(c);
	for(char *p=a;*p;p++)
		aa[*p-'a']++;
	for(char *p=b;*p;p++)
		bb[*p-'a']++;
	for(char *p=c;*p;p++)
		cc[*p-'a']++;
	int ansb=0,ansc=0;
	for(int cb=0;;cb++){
		for(int i=0;i<26;i++)
			if(aa[i]<bb[i]*cb) goto done;
		int co_c=10e7;
		for(int i=0;i<26;i++)
			if(cc[i]!=0)
				co_c=min(co_c,(aa[i]-bb[i]*cb)/cc[i]);
		if(cb+co_c>ansb+ansc)
			ansb=cb,ansc=co_c;
	}
	done:
	for(int i=1;i<=ansb;i++)
		printf("%s",b);
	for(int i=1;i<=ansc;i++)
		printf("%s",c);
	for(int i=0;i<26;i++)
		for(int j=1;j<=aa[i]-bb[i]*ansb-cc[i]*ansc;j++)
			printf("%c",i+'a');
	return 0;
}
