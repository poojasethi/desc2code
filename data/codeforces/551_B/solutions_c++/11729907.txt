#include<bits/stdc++.h>
using namespace std;

string a,b, c;
int A[205],B[205],C[205],res,nc;
main()
{
    cin >> a >> b >> c;
    for(int i=0; i<a.size(); i++)
        A[int(a[i])]++;

    for(int i=0; i<b.size(); i++)
        B[int(b[i])]++;

    for(int i=0; i<c.size(); i++)
        C[int(c[i])]++;
    res = 0;
    for(int i=0; i<=a.size()/b.size(); i++)
    {
        int k = 1e9;
            for(int j=97; j<97+26; j++)
        {
            if (C[j])
                if (A[j] - i*B[j]>=0) k = min(k,(A[j] - i*B[j])/C[j]);
            if (A[j] - i*B[j]<0) {k=1e9;break;}
        }
        if (k!=1e9)
        if (res < i+k) res = i + k, nc = k;
    }
    for(int i=0; i<res - nc; i++) cout << b;
    for(int i=0; i<nc; i++) cout << c;
    for(int i=97; i<97 + 26; i++)
        for(int j=1; j<=A[i] - (res - nc) * B[i] - nc * C[i]; j++) cout << char(i);
}
