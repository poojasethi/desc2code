#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;

const int N=1e5;

string a,b,c;
int aa[50],bb[50],cc[50];

int mxb()
{
int p=1e6;

for(int i=0;i<50;i++)
{
if(bb[i]!=0)
p=min(p,aa[i]/bb[i]);
}

return p;
}


int mxc()
{
int p=1e6;

for(int i=0;i<50;i++)
{
if(cc[i]!=0)
p=min(p,aa[i]/cc[i]);
}

return p;
}


int main()
{
cin >>a>>b>>c;
for(int i=0;i<a.size();i++)
aa[a[i]-'a']++;

for(int i=0;i<b.size();i++)
bb[b[i]-'a']++;

for(int i=0;i<c.size();i++)
cc[c[i]-'a']++;



int z=0,x=0;



for(int i=0;i<=mxb();i++)
{
for(int j=0;j<50;j++)
aa[j]-=i*bb[j];

if(mxc()+i>z+x)
{
z=i;
x=mxc();
}

for(int j=0;j<50;j++)
aa[j]+=i*bb[j];
}
for(int i=1;i<=z;i++)
cout<<b;
for(int i=1;i<=x;i++)
cout<<c;

for(int i=0;i<50;i++)
{
aa[i]-=z*bb[i];
aa[i]-=x*cc[i];

}
for(int i=0;i<50;i++)
for(int j=1;j<=aa[i];j++)
cout<<char(i+'a');

return 0;

}
