#include <iostream>
#define FOR(i, to, n) for (int i=to; i<n; i++ )
using namespace std;

int A[26]={};
int B[26]={};
int C[26]={};
string a, b, c;
int main () {
	cin >> a >> b >> c;
	int la, lb, lc, nb, nc;
	la=a.length(); lb=b.length(); lc=c.length();
	FOR(i, 0, la)
		A[a[i]-'a']++;
	FOR(i, 0, lb)
		B[b[i]-'a']++;
	FOR(i, 0, lc)
		C[c[i]-'a']++;
	nb=100001;
	FOR(i, 0, 26) {
		if (B[i])
			nb=min(nb, A[i]/B[i]);
	}
	if (nb>100000) nb=0;
	int M=0;
	int b1=0, c1=0;
	FOR(j, 0, nb+1) {
		nc=100001;
		FOR(i, 0, 26) {
			if (C[i])
				nc=min(nc, (A[i]-j*B[i])/C[i]);
		}
		if (nc>100000) nc=0;
		if (M<j+nc) {
			M=j+nc;
			b1=j;
			c1=nc;
		}
	}
//	cout << b1 << ' ' << c1 << endl;
	FOR(i, 0, b1)
		cout << b;
	FOR(i, 0, c1)
		cout << c;
	FOR(k, 0, 26)
		FOR(i, 0, A[k]-b1*B[k]-c1*C[k])
			cout << (char) (k+'a');
	
	cout << endl;
	
}
