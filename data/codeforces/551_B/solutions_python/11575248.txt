from collections import Counter
import copy
import sys

A = raw_input()
b = raw_input()
c = raw_input()

Z = Counter(A)
one = Counter(b)
two = Counter(c)
guarda = copy.deepcopy(Z)

best = 0
uno = 0
dos = 0
va = 0

while True:
    chico = 100005
    for i in two:
        if i not in Z.keys():
            chico = 0
        else:
            chico = min(chico, Z[i] / two[i])
    if chico + va > best:
        best = chico + va
        uno = va
        dos = chico
        guarda = copy.deepcopy(Z)


    para = False
    for i in b:
        if i not in Z.keys():
            para = True
        else:
            Z[i] -= 1
            if Z[i]<0:
                para = True

    if para:
        break
    va += 1

Z = guarda

sys.stdout.write(b * uno)
sys.stdout.write(c * dos)

if dos > 0:
    for i in c:
        Z[i] -= dos

for i in Z.keys():
    sys.stdout.write(i * Z[i])
