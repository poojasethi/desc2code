#!/bin/sh
# -*- coding: utf-8 -*-
''''which python2 >/dev/null && exec python2 "$0" "$@" # '''
''''which python  >/dev/null && exec python  "$0" "$@" # '''

a = raw_input()
b = raw_input()
c = raw_input()
maxn = 10**9

import string
import sys

ac = map(a.count, string.lowercase)
bc = map(b.count, string.lowercase)
cc = map(c.count, string.lowercase)


cnt1 = 0
cnt2 = 0

for x in xrange( min( map(lambda i: ac[i]/bc[i] if bc[i] else maxn, xrange(26)) )+1 ):

    t = min( map(lambda i: (ac[i]-(x*bc[i])) /cc[i] if cc[i] else maxn, xrange(26)) )

    if x + t > cnt1+cnt2:
        cnt1 = x
        cnt2 = t


sys.stdout.write(b*cnt1)
sys.stdout.write(c*cnt2)

for i in xrange(26):
    if ac[i]-(cnt1*bc[i]+cnt2*cc[i]) > 0:
        sys.stdout.write( string.lowercase[i] * (ac[i] - (cnt1*bc[i]+cnt2*cc[i])) )
