#!/usr/bin/python

import sys
import math
import string
from collections import defaultdict
from string import ascii_lowercase

def solve(A,B,C):
	fa = defaultdict(int)
	fb = defaultdict(int)
	fc = defaultdict(int)
	for a in A:
		fa[a] += 1
	for b in B:
		fb[b] += 1
	for c in C:
		fc[c] += 1

	maxB = 1000000
	for c in ascii_lowercase:
		if fb[c] != 0:
			maxB = min(maxB, fa[c]/fb[c])

	best = 0
	bestB = 0
	bestC = 0
	for nB in range(maxB+1):
		maxC = 1000000
		for c in ascii_lowercase:
			if fc[c] != 0:
				maxC = min(maxC, (fa[c] - fb[c] * nB) / fc[c])
		if nB + maxC > best:
			best = nB + maxC
			bestB = nB
			bestC = maxC

	ans = ''.join([B for _ in range(bestB)]) + ''.join([C for _ in range(bestC)])
	for c in ascii_lowercase:
		ans += ''.join([c for _ in range(fa[c] - bestB * fb[c] - bestC * fc[c])])

	return ans


if __name__ == "__main__":	
	A = sys.stdin.readline().split()[0]
	B = sys.stdin.readline().split()[0]
	C = sys.stdin.readline().split()[0]
	print solve(A,B,C)
