s=[raw_input() for _ in range(3)]
cnt=[[0]*26 for _ in range(3)]
for i in range(3):
	for x in s[i]:
		cnt[i][ord(x)-97]+=1
c=d=0
for x in range(1+min((cnt[0][i]/cnt[1][i] if cnt[1][i] else 10**5) for i in range(26))):
	y=min(((cnt[0][i]-x*cnt[1][i])/cnt[2][i] if cnt[2][i] else 10**5) for i in range(26))
	if c<x+y: c,d=x+y,x
print d*s[1]+(c-d)*s[2]+''.join([chr(97+i)*(cnt[0][i]-cnt[1][i]*d-cnt[2][i]*(c-d)) for i in range(26)])
