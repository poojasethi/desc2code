from string import ascii_lowercase
from collections import Counter

a = raw_input()
b = raw_input()
c = raw_input()

acount, bcount, ccount = map(Counter, [a, b, c])

best, bbest = 0, 0
bcur = 0
while min(acount.values()) >= 0:
	res = 1000000
	for i in ccount.keys():
		#print i
		res = min(res, acount[i] / ccount[i])
	if bcur + res > best:
		best = bcur + res
		bbest = bcur
	
	bcur = bcur + 1
	acount.subtract(bcount)

#print b * bbest

acount = Counter(a)
for i in range(bbest):
	acount.subtract(bcount)
for i in range(best - bbest):
	acount.subtract(ccount)

res = ''.join([b * bbest, c * (best - bbest)])
res = res +  ''.join(list(acount.elements()))
print res
