#include<cstdio>
#include<iostream>
using namespace std;
typedef long long u64;
#define mod 1000000007
u64 d[210][55][55], c[210][210];
int a1, a2;

void gao(int m) {
    int i, j;
    for (i = 0; i <= m; ++i) {
        c[i][0] = 1;
        for (j = 1; j <= i; ++j) {
            c[i][j] = (c[i - 1][j - 1] + c[i - 1][j]) % mod;
        }
    }
}

int main() {
    gao(200);
    int n, i, k, j, a, b, A, B;
    scanf("%d%d", &n, &k);
    k /= 50;
    while (n--) {
        scanf("%d", &i);
        if (i == 50)
            a1++;
        else
            a2++;
    }
    d[0][a1][a2] = 1;
    for (i = 1; i <= 200; ++i) {
        for (a = 0; a <= a1; ++a) {
            for (b = 0; b <= a2; ++b) {
                for (A = 0; A <= a; ++A) {
                    for (B = 0; B <= b; ++B) {
                        if (A + B == 0)
                            continue;
                        if (A + 2 * B > k)
                            break;
                        d[i][a][b] = (d[i][a][b] + d[i - 1][a1 - a + A][a2 - b + B] * c[a1 - a + A][A] * c[a2 - b + B][B]) % mod;
                    }
                }
            }
        }
        if ((i & 1) && d[i][a1][a2])
            break;
    }
    if (i == 201) {
        printf("-1\n0\n");
    } else {
        printf("%d\n", i);
        cout << d[i][a1][a2] << endl;
    }
    return 0;
}
