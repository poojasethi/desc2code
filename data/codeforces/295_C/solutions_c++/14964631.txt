#include<bits/stdc++.h>
#define ll long long
using namespace std;
const int mod=1000*1000*1000+7;
int a,b,n,k,dis[55][55][5];
ll t[55][55][5],c[55][55];
queue <pair<pair<int,int>,bool> > qe;
int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    cin>>n>>k;
    for(int i=0; i<n; i++)
    {
        int w;
        cin>>w;
        if(w==100)
            a++;
        else
            b++;
    }
    for(int i=0; i<55; i++)
        c[i][0]=1;
    for(int i=1; i<55; i++)
        for(int j=1; j<55; j++)
            c[i][j]=(c[i-1][j]+c[i-1][j-1])%mod;
    qe.push(make_pair(make_pair(a,b),0));
    dis[a][b][0]=1;
    t[a][b][0]=1;
    while(qe.size())
    {
        int a2=qe.front().first.first,b2=qe.front().first.second,a3=a-a2,b3=b-b2,f=qe.front().second;
        qe.pop();
        for(int i=0; i<=a2 && i*100<=k; i++)
            for(int j=0; j<=b2 && i*100+j*50<=k; j++)
                if((i!=0 || j!=0) && dis[a3+i][b3+j][1-f]==0)
                {
                    dis[a3+i][b3+j][1-f]=dis[a2][b2][f]+1;
                    t[a3+i][b3+j][1-f]=(t[a2][b2][f]*c[a2][i]*c[b2][j])%mod;
                    qe.push(make_pair(make_pair(a3+i,b3+j),1-f));
                }
                else if(dis[a3+i][b3+j][1-f]==dis[a2][b2][f]+1)
                    t[a3+i][b3+j][1-f]=(t[a3+i][b3+j][1-f]+t[a2][b2][f]*c[a2][i]*c[b2][j])%mod;
    }
    cout<<dis[a][b][1]-1<<'\n'<<t[a][b][1]<<'\n';
    return 0;
}
