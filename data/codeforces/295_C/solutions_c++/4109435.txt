#include <stdio.h>
#include <iostream>
#include <string.h>
#define ll long long
#define clr(a,b) memset(a,b,sizeof(a))
using namespace std;

const int mod=1000000007;
int n,m;
int a[2];
ll dp[55][55][210];
ll choose[55][55];

void gao(int x,int y,int k){
	ll g=dp[x][y][k];
	for(int i=0;i<=x;i++){
		if(i*50>m)break;
		for(int j=0;j<=y;j++){
			if(i*50+j*100>m)break;
			if(i+j==0)continue;
			int a=::a[0]-x+i,b=::a[1]-y+j;
			dp[a][b][k+1]=(dp[a][b][k+1]+g*choose[x][i]%mod*choose[y][j]%mod)%mod;
		}
	}
}

int main(){
	choose[0][0]=1;
	for(int i=1;i<=50;i++){
		choose[i][0]=1;
		for(int j=1;j<=i;j++)
			choose[i][j]=(choose[i-1][j-1]+choose[i-1][j])%mod;
	}
	cin>>n>>m;
	for(int i=0;i<n;i++){
		int u;cin>>u;
		a[u!=50]++;
	}
	clr(dp,0);
	dp[a[0]][a[1]][0]=1;
	ll ans=-1;
	for(int k=0;k<=4*n;k++)
		for(int i=0;i<=a[0];i++)
			for(int j=0;j<=a[1];j++){
				//cout<<i<<" "<<j<<" "<<k<<" "<<dp[i][j][k]<<endl;
				if(k&1){
					if(i==a[0]&&j==a[1]&&dp[i][j][k]){
						cout<<k<<endl<<dp[i][j][k]<<endl;
						return 0;
					}
					if(dp[i][j][k]==0)continue;
					gao(i,j,k);
				}else{
					if(dp[i][j][k]==0)continue;
					gao(i,j,k);
				}
			}
	cout<<"-1\n0"<<endl;
}
