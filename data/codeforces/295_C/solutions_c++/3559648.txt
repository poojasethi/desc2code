#include<iostream>
#include<cstdio>

#define STOP puts("Stop here~");
#define FF(i,n) for(int i=0; i<(n); ++i)
#define FOR(i,s,t) for(int i=(s); i<(t); ++i)
#define mid ((left+right)>>1)
#define lson rt<<1, left, m
#define rson rt<<1|1, m+1, right
#define ll long long
#define N 10005
#define M 1000000007
ll dp[507][55][55],c[55][55];
int main()
{
 //freopen("in.txt","r",stdin);
 //freopen("input.txt","w",stdout);
 int n,m,i,j,k,q;
 c[0][0]=1;
 for(i=1;i<=50;i++)
 {
  c[i][0]=c[i][i]=1;
  for(j=1;j<i;j++)
  c[i][j]=(c[i-1][j-1]+c[i-1][j])%M;
 }
 while(~scanf("%d%d",&n,&k))
 {
  k/=50;
  int small=0,big=0,a;
  for(i=0;i<n;i++)
  scanf("%d",&a),(a==100?big:small)++;
  memset(dp,0,sizeof(dp));
  dp[0][0][0]=1;
  for(int f=0;f<500;f++)
  for(i=0;i<=small;i++)
  for(j=0;j<=big;j++)
  if(dp[f][i][j])
  {
    if(i==small&&j==big)
    {printf("%d\n%I64d\n",f,dp[f][i][j]);
    goto p;}
    int s=(f&1?i:small-i);
    int b=(f&1?j:big-j);
    for(int ss=0;ss<=s;ss++)
    for(int bb=0;bb<=b;bb++)
    if(ss+bb>0&&ss+2*bb<=k)
    {
     int ns=i+ss*(f&1?-1:1);
     int nb=j+bb*(f&1?-1:1);
     ll res=dp[f][i][j];
     res*=c[s][ss];res%=M;
     res*=c[b][bb];res%=M;
     dp[f+1][ns][nb]+=res;dp[f+1][ns][nb]%=M;
    }
  }
  printf("-1\n0\n");
  p:continue;
 }
 return 0;
}
