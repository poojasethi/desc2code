#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define mo 1000000007
#define Long long long
#define For(i,l,r) for(int i=l;i<=r;i++)
using namespace std;
int n,K,A,B;
int f[105][55][55],C[55][55];
void update(int &a,Long b){a=(a+b)%mo;}
int main(){
    scanf("%d%d",&n,&K);
    For(i,1,n){
        int x; scanf("%d",&x);
        if(x==50) A++;
        else B++;
    }
    
//  printf("K A B %d %d %d\n",K,A,B);
    C[0][0]=1;
    For(i,1,n){
        C[i][0]=1;
        For(j,1,i) C[i][j]=(C[i-1][j-1]+C[i-1][j])%mo;
    }
    
    f[0][0][0]=1;
    For(i,0,2*n){
        if(f[i][A][B]){printf("%d\n%d\n",2*i-1,f[i][A][B]); return 0;}
        For(a,0,A) For(b,0,B) if(f[i][a][b]){
//          printf("i %d a b %d %d f-> %d\n",i,a,b,f[i][a][b]);
            for(int na=0;a+na<=A&&na*50<=K;na++)
            for(int nb=0;b+nb<=B&&na*50+nb*100<=K;nb++)
            if(na || nb){
                if(a+na==A && b+nb==B) update(f[i+1][a+na][b+nb],f[i][a][b]);
                for(int ra=0;ra<=a+na&&ra*50<=K;ra++)
                for(int rb=0;rb<=b+nb&&ra*50+rb*100<=K;rb++){
                    if(ra>=na&&rb>=nb) break;
                    if(ra||rb) update(f[i+1][a+na-ra][b+nb-rb],1LL*f[i][a][b]*C[A-a][na]%mo*C[B-b][nb]%mo*C[a+na][ra]%mo*C[b+nb][rb]);
                }
            }
        }
    }
    
    printf("-1\n0\n");
}
