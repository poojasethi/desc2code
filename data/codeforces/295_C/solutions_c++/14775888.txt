#include <bits/stdc++.h>

#define ll long long
#define ld long double

using namespace std;

const int MAXN=50+5,mod=1000000007;
int dis[MAXN][MAXN][2],dp[MAXN][MAXN][2];
queue<int>q;
ll choose[MAXN][MAXN];


int main(){
	int n,w,l;
	int a=0,b=0;
	cin>>n>>w;
	for(int i=0;i<MAXN;i++)
		choose[0][i]=1;
	for(int i=1;i<MAXN;i++)
		for(int j=1;j<MAXN;j++)
			choose[i][j]=(choose[i][j-1]+choose[i-1][j-1])%mod;
	for(int i=0;i<n;i++){
		cin>>l;
		if(l==50)a++;
		else b++;
	}
	for(int i=0;i<MAXN;i++)
		for(int j=0;j<MAXN;j++)
			dis[i][j][1]=dis[i][j][0]=-1;
	q.push(a);
	q.push(b);
	q.push(0);
	dis[a][b][0]=0;
	dp[a][b][0]=1;
	while(q.size()){
		int x=q.front();
		q.pop();
		int y=q.front();
		q.pop();
		int c=q.front();
		q.pop();
		for(int i=0;i<=x;i++)
			for(int j=0;j<=y;j++){
				if(i!=0 or j!=0)
					if(i*50+j*100<=w and (dis[x][y][c]+1<=dis[a-(x-i)][b-(y-j)][1-c] or dis[a-(x-i)][b-(y-j)][1-c]==-1)){
						if(dis[a-(x-i)][b-(y-j)][1-c]==-1)
							q.push(a-(x-i)),q.push(b-(y-j)),q.push(1-c);
						dis[a-(x-i)][b-(y-j)][1-c]=dis[x][y][c]+1;
						dp[a-(x-i)][b-(y-j)][1-c]+=(dp[x][y][c]*((choose[i][x]*choose[j][y])%mod))%mod;
						dp[a-(x-i)][b-(y-j)][1-c]=dp[a-(x-i)][b-(y-j)][1-c]%mod;
					}
			}
	}
	cout<<dis[a][b][1]<<endl<<dp[a][b][1];
	return 0;
}
