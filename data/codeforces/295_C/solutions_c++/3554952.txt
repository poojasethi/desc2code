#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;
const int mo=int(1e9+7);
int num[120],dis[61][61][2],g[61][61][2],x,y,i,j,C[61][61],z,bfs[120000][3],
	f[61][61][2],r,n,m,l;
void plus(int &x,int y){
	x+=y;
	if(x>=mo)x-=mo;
}
void renew(int x,int y,int b,int v,int o){//50 x,100 y,b,dist,����
	if(!dis[x][y][b]){
		dis[x][y][b]=v;
		f[x][y][b]=o;
		bfs[++r][0]=x; bfs[r][1]=y; bfs[r][2]=b;
	}else if(dis[x][y][b]==v){
		plus(f[x][y][b],o);
	}
}	
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;++i)scanf("%d",&x),num[x]++;
	for(i=0;i<=n;++i)
		for(j=0;j<=i;++j)C[i][j]=j?(C[i-1][j]+C[i-1][j-1])%mo:1;
	renew(num[50],num[100],0,0,1);
	while(l<=r){
		x=bfs[l][0]; y=bfs[l][1]; z=bfs[l][2];
		l++;
		for(i=0;i<=x && i*50<=m;++i)
			for(j=0;j<=y && i*50+j*100<=m;++j)if(i+j)
				renew(num[50]-x+i,num[100]-y+j,1-z,dis[x][y][z]+1,
					  (long long)C[x][i]*C[y][j]%mo*f[x][y][z]%mo);
	}
	x=num[50]; y=num[100];
	if(dis[x][y][1]==0)printf("-1\n0\n");
	else printf("%d\n%d\n",dis[x][y][1],f[x][y][1]);
}
