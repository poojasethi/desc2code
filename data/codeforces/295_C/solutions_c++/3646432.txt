#include<cstdio>
#include<cstring>
int n,m,x,oo=1000000007;
int a[55][55][2],b[55][55][2],s[3],e[10005],f[10005],g[10005],c[55][55];
bool update(int &x,int &y,int x1,int y1)
{if (x<0) return x=x1,y=y1,1; if (x1==x) y=(y+y1)%oo; return 0;}
int main()
{
	scanf("%d%d",&n,&m),m/=50; memset(a,-1,sizeof(a));
	for (int i=0;i<=52;i++) for (int j=0;j<=i;j++) c[i][j]=i&&j?(c[i-1][j]+c[i-1][j-1])%oo:1;
	for (int i=1;i<=n;i++) scanf("%d",&x),s[x/50]++;
	int h=0,t=1; f[1]=s[1],g[1]=s[2],e[1]=1; a[s[1]][s[2]][1]=0,b[s[1]][s[2]][1]=1;
	while (h<t){
		int x=f[++h],y=g[h],z=e[h];
		if (z){
			for (int i=0;i<=x;i++)
				for (int j=0;j<=y;j++) if (i+j && i+j*2<=m)
					if (update(a[x-i][y-j][0],b[x-i][y-j][0],a[x][y][1]+1,1LL*b[x][y][1]*c[x][i]%oo*c[y][j]%oo))
						f[++t]=x-i,g[t]=y-j,e[t]=0;
			}
		else{
			for (int i=0;i+x<=s[1];i++)
				for (int j=0;j+y<=s[2];j++) if (i+j && i+j*2<=m)
					if (update(a[x+i][y+j][1],b[x+i][y+j][1],a[x][y][0]+1,1LL*b[x][y][0]*c[s[1]-x][i]%oo*c[s[2]-y][j]%oo))
						f[++t]=x+i,g[t]=y+j,e[t]=1;
			}
		}
	printf("%d\n%d\n",a[0][0][0],b[0][0][0]); return 0;
}
