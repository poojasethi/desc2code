#include <cstdio>
const int N=55,mo=(int)1e9+7;
int n,k,cx,cy,d[N][N][2],f[N][N][2],C[N][N];
struct data{
	int x,y,s;
} q[N*N*2],*head=q,*tail=q;
void renew(int x,int y,int s,int dis,int val){
	if(f[x][y][s]){
		if(d[x][y][s]==dis) (f[x][y][s]+=val)%=mo;
		return;
	}
	f[x][y][s]=val;
	d[x][y][s]=dis;
	*tail++=(data){x,y,s};
}
int main(){
	scanf("%d%d",&n,&k);
	for(int i=0;i<=n;i++)
		for(int j=0;j<=i;j++)
			C[i][j]=j?(C[i-1][j-1]+C[i-1][j])%mo:1;
	while(n--){
		int x; scanf("%d",&x);
		if(x==50) cx++; else cy++;
	}
	renew(cx,cy,0,0,1);
	while(head<tail){
		for(int i=0;i<=head->x&&i*50<=k;i++)
			for(int j=0;j<=head->y&&i*50+j*100<=k;j++) if(i||j)
				renew(cx-head->x+i,cy-head->y+j,head->s^1,d[head->x][head->y][head->s]+1,
					1LL*f[head->x][head->y][head->s]*C[head->x][i]%mo*C[head->y][j]%mo);
		head++;
	}
	printf("%d\n%d\n",f[cx][cy][1]?d[cx][cy][1]:-1,f[cx][cy][1]);
}
