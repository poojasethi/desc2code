#include <bits/stdc++.h>
using namespace std;
#define pf pop_front
#define pb push_back
#define ulli unsigned long long int
#define FOR(i,begin,end) for (LL i=begin;i<=end;i++)
#define rep(i,t) for (LL i=0;i<t;i++)
#define LL __int64
#define pii pair<LL,LL>
const LL MAX=1e7,N=57,MOD=(1e9)+7;
LL n,k,a,b,d[N][N],x,o[N][N][3],ans=1,e[N][N][4];
deque<pii>q;
deque<bool>qp;
bool v[N][N][3];
void BFS()
{
	q.pb(pii(a,b));
	qp.pb(0);
	while(q.size())
	{
		LL hq,hx,hy,px,py;
		hx=q[0].first;
		hy=q[0].second;
		hq=qp[0];
		v[hx][hy][hq]=true;
		FOR(i,0,hx)
		{
			FOR(j,0,hy)
			{
				if(i+j && (50*(i+(2*j)))<=k && !v[a-hx+i][b-hy+j][1-hq])
					{
						q.pb(pii(a-hx+i,b-hy+j));
						qp.pb(1-hq);
						o[a-hx+i][b-hy+j][1-hq]=(o[hx][hy][hq]+1);
						v[a-hx+i][b-hy+j][1-hq]=true;
					}
					else if(i+j && (50*(i+(2*j)))<=k && v[a-hx+i][b-hy+j][1-hq])
					{
						e[hx][hy][hq]+=(e[a-hx+i][b-hy+j][1-hq]*d[i][a-hx+i]*d[j][b-hy+j]);
						e[hx][hy][hq]%=MOD;
					}
			}
		}
		if(hx==a && hy==b && hq)
		{
			return ;
		}
		q.pf();
		qp.pf();
	}
}
int main()
{
	cin>>n>>k;
	FOR(i,1,n)
	{
		cin>>x;
		if(x==50)
		{
			a++;
		}
		else
		{
			b++;
		}
	}
	e[a][b][0]=1;
	FOR(i,0,50)
		{
			d[1][i]=i;
			d[0][i]=1;
		}
		FOR(i,2,50)
		{
			FOR(j,2,50)
			{
				d[i][j]=(d[i-1][j-1]+d[i][j-1]);
			}
		}
	BFS();
	if(v[a][b][1])
	{
		cout<<o[a][b][1]<<"\n";
		return cout<<e[a][b][1]<<"\n",0;
	}
	else
	{
		cout<<"-1\n";
		return cout<<"0\n",0;
	}
}
