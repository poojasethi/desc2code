#include <iostream>
#include <stdio.h>
#define  maxn 55
#define  mod  1000000007
using namespace std;
int c[maxn][maxn],n,k,a,b,x;
long long f[maxn<<2][maxn][maxn];
int main(){
    cin>>n>>k;
    k/=50;
    for(int i=0;i<=n;i++)
       for(int j=0;j<=i;j++)
          c[i][j]=j?(c[i-1][j-1]+c[i-1][j])%mod:1;
    for(int i=1;i<=n;i++){
       cin>>x;
       if(x==50)a++;else b++;
    }
    f[0][a][b]=1;
    for(int x=0;x<=n<<2;x++){
       for(int i=a;i>=0;i--)
          for(int j=b;j>=0;j--){
             if(!f[x][i][j])continue;
             for(int ii=0;ii<=i;ii++)
             for(int jj=0;jj<=j;jj++){
                if((!ii)&&(!jj))continue;
                if(ii+jj*2>k)continue;
                f[x+1][a-i+ii][b-j+jj]+=(f[x][i][j]*c[i][ii]%mod*c[j][jj]%mod);
                f[x+1][a-i+ii][b-j+jj]%=mod;
             }
          }
       if(!(x&1)&&f[x+1][a][b]){
          printf("%d\n%I64d\n",x+1,f[x+1][a][b]);
          return 0;
       }
    }
    printf("-1\n0\n");
    return 0;
}
