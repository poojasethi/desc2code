#include <bits/stdc++.h>
#define MOD 1000000007
using namespace std;
typedef pair<int, int> pii;
typedef pair<int, pii> piii;

const int N=55;
int c[N][N], cnt[2], vis[2][N][N], dp[2][N][N];

int main() {
	int n, k;
	memset(vis, -1, sizeof(vis));
	for(int i=0; i<N; i++) {
		c[i][0]=1;
		for(int j=1; j<=i; j++) c[i][j]=(c[i-1][j]+c[i-1][j-1])%MOD;
	}
	scanf("%d %d", &n, &k);
	int q0=k/50, q1=k/100;
	for(int i=0; i<n; i++) {
		int w;
		scanf("%d", &w);
		cnt[w/50-1]++;
	}
	queue<piii> q;
	vis[0][0][0]=0;
	dp[0][0][0]=1;
	q.push(piii(0, pii(0, 0)));
	while(!q.empty()) {
		int t=q.front().first;
		int c0=q.front().second.first;
		int c1=q.front().second.second;
		q.pop();
		int endi, endj;
		if(t) endi=c0, endj=c1;
		else endi=cnt[0]-c0, endj=cnt[1]-c1;
		for(int i=0; i<=min(endi, q0); i++) for(int j=0; j<=min(endj, q1); j++) if((i || j) && (50*i+100*j<=k)) {
			int s=t ? -1 : 1;
			if(vis[1-t][c0+s*i][c1+s*j]==-1) {
				vis[1-t][c0+s*i][c1+s*j]=vis[t][c0][c1]+1;
				dp[1-t][c0+s*i][c1+s*j]=(((1ll*dp[t][c0][c1]*c[endi][i])%MOD)*c[endj][j])%MOD;
				q.push(piii(1-t, pii(c0+s*i, c1+s*j)));
			}
			else if(vis[1-t][c0+s*i][c1+s*j]==vis[t][c0][c1]+1)
				dp[1-t][c0+s*i][c1+s*j]=(dp[1-t][c0+s*i][c1+s*j]+(((1ll*dp[t][c0][c1]*c[endi][i])%MOD)*c[endj][j])%MOD)%MOD;
		}
	}
	printf("%d\n%d\n", vis[1][cnt[0]][cnt[1]], dp[1][cnt[0]][cnt[1]]);
	return 0;
}
