#include <stdio.h>
#include <string.h>
typedef long long ll;
ll sum,nn,ma;
int main(){
	ll i,j,t,c,x,m,n;
	scanf("%I64d %I64d",&n,&m);
	if(n==0){
		printf("%I64d\n",-m*m);
		for(i=1;i<=m;i++) printf("x");
	}
	else if(m==0){
		printf("%I64d\n",n*n);
		for(i=1;i<=n;i++) printf("o");
	}
	else if(n&&m){
		ma=n*n-m*m;x=1;
		for(i=2;i<=m;i++){
			if(n<i-1) break;
			t=m/i;
			c=m%i;
			sum=t*t*i+2*c*t+c;
			nn=i-2+(n-i+2)*(n-i+2);
			if(nn-sum>ma) {ma=nn-sum;x=i;}
		}
		printf("%I64d\n",ma);
		t=m/x;c=m%x;
		for(i=1;i<=x-1;i++){
			if(i<=c){
				for(j=1;j<=t+1;j++) printf("x");printf("o");
			}
			else{
				for(j=1;j<=t;j++) printf("x");printf("o");
			}
		}
		for(i=1;i<=n-x+1;i++) printf("o");for(i=1;i<=t;i++) printf("x");
	}
	return 0;
}
