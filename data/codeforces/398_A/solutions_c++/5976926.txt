#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
typedef long long ll;
ll n,m,t1,t2,ans,tmp;
int i,k;
ll calc(int x){
	ll k,res,t;
	t=m/x;k=m%x;
	if (x==1) res=n*n;else res=(x-2)+(n-(x-2))*(n-(x-2));
	res=res-k*(t+1)*(t+1)-(x-k)*t*t;
	return res;
}
void outp(int x){
	ll t,k;
	int i,j;
	t=m/x;k=m%x;
	if (x==1){
		for (i=1;i<=n;i++) putchar('o');
		for (i=1;i<=m;i++) putchar('x');
		puts("");return;
	}
	for (i=1;i<x;i++){
		for (j=1;j<=((i<=k)?t+1:t);j++) putchar('x');
		for (j=1;j<=((i==1)?n-(x-2):1);j++) putchar('o');
	}
	for (j=1;j<=t;j++) putchar('x');
	puts("");
}
int main(){
	cin>>n>>m;
	if (!m){
		cout<<n*n<<endl;
		for (i=1;i<=n;i++) putchar('o');
		puts("");
		return 0;
	}
	for (i=1,ans=-(1LL<<60);i<=min(m,n+1);i++){
		tmp=calc(i);
		if (tmp>ans) ans=tmp,k=i;
	}
	cout<<ans<<endl;
	outp(k);
	return 0;
}
