#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
typedef long long LL;
using namespace std;
LL a,b,ans,dq,x,y,ansn;
int main()
{
	scanf("%I64d%I64d",&a,&b);
	ans=1LL<<63;
	ans=-ans;
	if (a==0)
	{
		ans=-b*b;  
		printf("%I64d\n",ans);
		for (int i=1;i<=b;++i) printf("x");
		return 0;
	}
	for (LL i=1;i<=a;++i)
	{
		dq=0;
		dq=i-1+(a-i+1)*(a-i+1);
		x=b/(i+1);
		y=b-x*(i+1);
		dq=dq-(x+1)*(x+1)*y;
		dq=dq-(i+1-y)*x*x;
		if (dq>ans)
		{
			ans=dq;ansn=i;
		}
	}
	printf("%I64d\n",ans);
	x=b/(ansn+1);y=b-x*(ansn+1);
	for (int i=1;i<=ansn;++i)
	{
		if (y>0) for (int i=1;i<=x+1;++i) printf("x");
		else   for (int i=1;i<=x;++i) printf("x");
		printf("o");y--;
	}
	for (int i=1;i<=a-ansn;++i) printf("o");
	for (int i=1;i<=x;++i) printf("x");
	return 0;
}
