#include<cstdio>
#include<iostream>
using namespace std;
typedef long long ll;
int x,y;
ll ans=-20000000000000000;
int bj;
void solve(int num){
ll result=(ll)(num-1)+(ll)(x-num+1)*(ll)(x-num+1);
int len=y/(num+1);
int res=y-len*(num+1);
result-=(ll)(num+1-res)*(ll)len*(ll)len;
result-=(ll)res*(ll)(len+1)*(ll)(len+1);
if(result>ans){ans=result;bj=num;}
}
int main(){
scanf("%d%d",&x,&y);
if(!x){ans=-y*y;cout<<ans<<endl;for(int i=1;i<=y;i++)printf("x");printf("\n");return 0;}
//枚举把x分成几块
for(int i=1;i<=x;i++)solve(i);
int num=bj;
int len=y/(num+1);
int res=y-len*(num+1);
cout<<ans<<endl;
for(int i=1;i<=len;i++)printf("x");
for(int i=1;i<=x-num+1;i++)printf("o");
for(int i=1;i<num;i++){
if(res){res--;for(int i=1;i<=len+1;i++)printf("x");}
else {for(int i=1;i<=len;i++)printf("x");}
printf("o");
}
if(res){res--;for(int i=1;i<=len+1;i++)printf("x");}
else {for(int i=1;i<=len;i++)printf("x");}
printf("\n");
}
