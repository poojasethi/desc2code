#include<cstdio>
using namespace std;
typedef long long ll;
int a, b, t;
ll Sum(int t) {
  if (t == 1) return 1LL*a*a-1LL*b*b;
  ll k = 1LL*(a-t+2)*(a-t+2)+t-2;
  k -= 1LL*(b%t)*(b/t+1)*(b/t+1);
  k -= 1LL*(t-b%t)*(b/t)*(b/t);
  return k;
}
int main()
{
  scanf("%d%d", &a, &b);
  t = 1;
  for (int i = 1; i <= a+1; i++)
    if (Sum(i) > Sum(t)) t = i;
  printf("%I64d\n", Sum(t));
  for (int i = 0; i < b%t; i++) {
    for (int j = 0; j < b/t+1; j++) printf("x");
    if (a-- > 0) printf("o");
  }
  for (int i = b%t; i < t; i++) {
    for (; i==t-1 && a-->0; printf("o"));
    for (int j = 0; j < b/t; j++) printf("x");
    if (a-- > 0) printf("o");
  }
  printf("\n"); 
}
