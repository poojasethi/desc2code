#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
using namespace std;
long long maxx,res;
int ans,n,m,i,tt,xx,j;
int main(){
	scanf("%d%d",&n,&m);
	ans=0; maxx=-1ll<<55;
	for(i=0;i<=n;++i){
		int t=m/(i+2),
			x=m-t*(i+2);
		if(i==n){
			t=m/(i+1);
			x=m-t*(i+1);
		}
		res=1ll*(n-i)*(n-i)+i-1ll*(t+1)*(t+1)*x-1ll*t*t*(i+1+(i!=n)-x);
		if(res>maxx)maxx=res,ans=i,tt=t,xx=x;
	}
	printf("%I64d\n",maxx);
	if(ans!=n)for(i=1;i<=tt;++i)printf("x");
	for(i=1;i<=n-ans;++i)printf("o");
	for(i=1;i<=ans+1;++i){
		for(j=1;j<=tt;++j)printf("x");
		if(i<=xx)printf("x");
		if(i!=ans+1)printf("o");
	}
}
