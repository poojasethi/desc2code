#include <iostream>
#include <cmath>
#include <queue>
#include <vector>
#include <cstring>
#include <cstdio>
using namespace std;
typedef long long LL;
LL a,b;

int main(){
    //freopen("in","r",stdin);
    cin>>a>>b;
    if (a==0){
        cout<<-b*b<<endl;
        for (int i=0;i<b;i++)
            cout<<"x";
        return 0;
    }
    LL ans=-(1LL<<62),k;
    for (LL i=2;i<=a+1;i++){
        LL tmp=(a-(i-2))*(a-(i-2))+i-2-(b/i+1)*(b/i+1)*(b%i)-(b/i)*(b/i)*(i-b%i);
        if (tmp>ans) {ans=tmp; k=i;}
    }
    cout<<ans<<endl;
    int tmp=b%k;
    for (int i=0;i<k-1;i++){
        for (int j=0;j<(b/k);j++) cout<<"x";
        if (tmp) cout<<"x",tmp--;
        cout<<"o";
    }
    for (int i=1;i<a-(k-2);i++)
        cout<<"o";
    for (int i=0;i<b/k+tmp;i++)
        cout<<"x";
    return 0;
}
