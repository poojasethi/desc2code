#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=200050;
int A[maxn];
long long n,a,b,ans=-0x7fffffffff;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,j2,l;
	long long k,t,ans0;
	scanf("%I64d%I64d",&a,&b);
	n=a+b;
	if(!a)
	{
		ans=-b*b;
		for(i=0;i<b;i++)
			A[i]=1;
	}
	else
		for(i=2;i<=a+1;i++)
		{
			k=b/i;
			t=b%i;
			ans0=i-2+(a-i+2)*(a-i+2)-t*(k+1)*(k+1)-(i-t)*k*k;
			if(ans0>ans)
			{
				ans=ans0;
				l=0;
				for(j=0;j<i-t&&j<i-1;j++)
				{
					for(j2=0;j2<k;j2++)
						A[l++]=1;
					if(j<i-2) A[l++]=0;
				}
				for(;j<i-1;j++)
				{
					for(j2=0;j2<k+1;j2++)
						A[l++]=1;
					if(j<i-2) A[l++]=0;
				}
				for(j=i-2;j<a;j++)
					A[l++]=0;
				for(j=0;j<k+(t?1:0);j++)
					A[l++]=1;
			}
		}
	printf("%I64d\n",ans);
	for(i=0;i<n;i++) 
		printf("%c",A[i]?'x':'o');
	return 0;
}
