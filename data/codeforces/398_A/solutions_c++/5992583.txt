#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
typedef __int64 ll;
const ll INF=-1999999999;
ll v,x,ans,a,b,k;
ll slove(int c){
	x=b/c;
	v=b%c;
	ll temp1=(a-c+2)*(a-c+2)+c-2;
	ll temp2=(x+1)*(x+1)*v+x*x*(c-v);
	//printf("%I64d\n",temp1-temp2);
	return  temp1-temp2;
}
void print(){
	x=b/k;
	v=b%k;
//	printf("%I64d\n",k);
	printf("%I64d\n",ans);
	int i,j,vis=0;
	for(i=0;i<k-1;i++){
		if(vis) printf("o");
		for(j=0;j<x;j++) printf("x");
		if(i<v) printf("x");
		vis=1;
	}
	for(i=0;i<a-k+2;i++) printf("o");
	for(i=0;i<x;i++) printf("x");
	printf("\n");
}
int main(){
	int i,j;
	while(scanf("%I64d%I64d",&a,&b)!=EOF){
		if(a==0){
			printf("-%I64d\n",b*b);
			for(i=1;i<=b;i++) printf("x");
			printf("\n");
			continue;
		}
		ans=INF;
		k=1;
		for(i=2;i<=a+1;i++){
			ll t=slove(i);
			if(t>ans){
				ans=t;
				k=i;	
			}
			//printf("%I64d %I64d\n",ans,k);	
		}
		print();
		//printf("%I64d\n",ans);
	}
}
