#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
typedef __int64 ll;
#define N 1000011
char s[N];
int main() {
	//freopen("1.in","r",stdin);
	int a,b;
	scanf("%d %d",&a,&b);
	if (a<=0 || b<=1) {
		printf("%I64d\n",ll(a)*a-ll(b)*b);
		int l=0;
		for (int i=0;i<a;i++) s[l++]='o';
		for (int i=0;i<b;i++) s[l++]='x';
		s[l]=0;
		puts(s);
		return 0;
	}
	ll ans=-(1ll<<60),tmp;
	int j;
	for (int i=1;i<=a;i++) if (b>=i+1) {
		tmp=ll(a-i+1)*(a-i+1)+(i-1)-ll(b/(i+1)+1)*(b/(i+1)+1)*(b%(i+1))-ll(b/(i+1))*(b/(i+1))*(i+1-b%(i+1));
		if (tmp>ans) ans=tmp,j=i;
	}
	printf("%I64d\n",ans);
	int l=0;
	for (int i=0;i<=j;i++) {
		if (i<j+1-b%(j+1)) for (int k=0;k<b/(j+1);k++) s[l++]='x';
		else for (int k=0;k<b/(j+1)+1;k++) s[l++]='x';
		if (i<j-1) s[l++]='o';
		else if (i==j-1) for (int k=0;k<=a-j;k++) s[l++]='o';
		else s[l]=0;
	}
	puts(s);
}
