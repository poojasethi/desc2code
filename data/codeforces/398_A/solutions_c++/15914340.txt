#include<cstdio>
#define rep(i,n) for (int i=0;i<n;++i)
typedef long long ll;
ll n,m,x,y,s,t,res,ans=-1LL<<60;
ll get(ll t){x=m/t,y=m%t; return (t-y)*x*x+y*(x+1)*(x+1);}
int main()
{
scanf("%I64d%I64d",&n,&m);
if (!n){printf("%I64d\n",-m*m); rep(i,m) printf("x"); return 0;}
rep(i,n){res=(n-i)*(n-i)+i-get(i+2); if (res>ans) ans=res,s=i;}
printf("%I64d\n",ans),t=s+2; x=m/t,y=m%t;
rep(j,s+2){
if (j) if (j>1) printf("o"); else rep(i,n-s) printf("o");
rep(i,x+(j<y)) printf("x");
}
return 0;
}
