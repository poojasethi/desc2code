#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<algorithm>
#define ll long long
#define LiangJiaJun main
using namespace std;
ll a,b,ans;
int LiangJiaJun(){
    cin>>a>>b;
    if(a==0){
        cout<<-b*b<<endl;
        for(int i=1;i<=b;i++)cout<<"x";cout<<endl;
        return 0;
    }
    ll ans=-(1LL<<62),k;
    for (ll i=2;i<=a+1;i++){
        ll tmp=(a-(i-2))*(a-(i-2))+i-2-(b/i+1)*(b/i+1)*(b%i)-(b/i)*(b/i)*(i-b%i);
        if (tmp>ans){ans=tmp; k=i;}
    }
    cout<<ans<<endl;
    int tmp=b%k;
    for (int i=0;i<k-1;i++){
        for (int j=0;j<(b/k);j++) cout<<"x";
        if (tmp) cout<<"x",tmp--;
        cout<<"o";
    }
    for(int i=1;i<a-(k-2);i++)cout<<"o";
    for(int i=0;i<b/k+tmp;i++)cout<<"x";
    cout<<endl;
    //system("pause");
    return 0;
}
