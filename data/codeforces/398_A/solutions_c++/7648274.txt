#include <iostream>
using namespace std;
#define ll long long
#define N 20000
int a, b;
int main() {
	cin >> a >> b;
	ll maxs = (ll) a * a - (ll) b * b;
	int maxn = 1;
	for (int n = 1; n <= min(b, a+1); n++) {
		ll s = 0;
		s -= (ll) (b%n) * (b/n+1) * (b/n+1);
		s -= (ll) (n-b%n) * (b/n) * (b/n);
		int m = max(n-2, 0);
		s += m;
		s += (ll) (a-m) * (a-m);
		if (s >= maxs) {
			maxs = s;
			maxn = n;
		}
	}
	cout << maxs << endl;
	int n = maxn;
	if (n == 0) {
		for (int i = 0; i < a; i++)
			cout << 'o';
		return 0;
	}
	if (n == 1) {
		for (int i = 0; i < a; i++)
			cout << 'o';
		for (int i = 0; i < b; i++)
			cout << 'x';
		return 0;
	}
	int ar[2*n-1];
	for (int i = 0; i < n; i++) {
		ar[2*i] = b/n;
		if (i < b%n)
			ar[2*i]++;
	}
	ar[1] = a - max(n-2, 0);
	for (int i = 1; i < n-1; i++)
		ar[2*i+1] = 1;
	char c = 'x';
	for (int i = 0; i < 2*n-1; i++) {
		for (int j = 0; j < ar[i]; j++)
			cout << c;
		c = c == 'x' ? 'o' : 'x';
	}
}
