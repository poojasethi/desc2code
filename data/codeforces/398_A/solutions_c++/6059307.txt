#include <iostream>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#include <algorithm>
using namespace std;
long long maxx,res;
int ans,n,m,i,tt,xx,j;

int main(){
    cin>>n>>m;
    ans=0; maxx=-1ll<<55;
    for(i=0; i<=n; i++){
        int t=m/(i+2),
            x=m-t*(i+2);
        if(i==n){
            t=m/(i+1);
            x=m-t*(i+1);
        }
        res=1ll*(n-i)*(n-i)+i-1ll*(t+1)*(t+1)*x-1ll*t*t*(i+1+(i!=n)-x);
        if(res>maxx)maxx=res,ans=i,tt=t,xx=x;
    }
    cout<<maxx<<endl;
    if(ans!=n)
        for(i=1;i<=tt;++i)
            cout<<"x";
    for(i=1;i<=n-ans;++i)
        cout<<"o";
    for(i=1;i<=ans+1;++i){
        for(j=1;j<=tt;++j)
            cout<<"x";
    if(i<=xx)
        cout<<"x";
    if(i!=ans+1)
        cout<<"o";
    }
}
