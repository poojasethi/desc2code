#include <cstdio>
#include <cstring>

using namespace std;
typedef long long LL;
LL a,b;
LL C(LL t){LL x=b/t,y=b%t;return (t-y)*x*x+y*(x+1)*(x+1);}
int main()
{
	scanf("%I64d%I64d",&a,&b);
	if(!a){printf("%I64d\n",-b*b);for(int i=1;i<=b;i++) putchar('x');puts("");return 0;}
	LL ans=-0x7fffffffffffffffLL,s=0;
	for(int i=0;i<a;i++){LL r=(a-i)*(a-i)+i-C(i+2);if(r>ans) ans=r,s=i;}
	printf("%I64d\n",ans);LL x=b/(s+2),y=b%(s+2);
	for(int j=0;j<=s+1;j++){
		if(j){
			if(j>1) putchar('o');
			else for(int i=0;i<a-s;i++) putchar('o');
		}
		for(int i=0;i<x+(j<y);i++) putchar('x');
	}
	puts("");return 0;
}
