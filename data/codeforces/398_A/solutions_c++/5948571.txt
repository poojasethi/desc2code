#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int64 i,j,k,n,m,an,re,A,ans;
int main() {
	scanf("%I64d%I64d",&n,&m);
	if (!n||m<=1) {
		printf("%I64d\n",n*n-m*m);
		For(i,1,n) printf("o");
		For(i,1,m) printf("x");
		printf("\n");
		//for(;;);
		return 0;
	}
	an=(int64)-1e18;
	For(i,1,n) {
		if (m<i+1) break;
		k=m/(i+1);
		A=m%(i+1);
		re=(n-i+1)*(n-i+1)+i-1-k*k*(i+1-A)-(k+1)*(k+1)*A;
		if (re>an) an=re,ans=i;
	}
	printf("%I64d\n",an);
	k=m/(ans+1);
	A=m%(ans+1);
	For(i,1,ans) {
		For(j,1,(k+(i<=A))) printf("x");
		For(j,1,(i==1?n-ans+1:1)) printf("o");
	}
	For(i,1,k) printf("x");
	printf("\n");
	//for(;;);
	return 0;
}
