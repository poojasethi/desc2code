#include<stdio.h>
#include<string.h>
#include<stdlib.h>  
long long a,b,ans,min,p,mod,now;
long long maxk,i,j;
void pc(int num,char c)
{
  int i;
  for (i=1;i<=num;i++)printf("%c",c);
}
void print()
{
  int i;
  if (maxk==1)
  {
    pc(a,'o');
    pc(b,'x');
  }
  else
  {
  p=b/maxk;
  mod=b%maxk;
  pc(p,'x');
  int l=maxk-mod-1,r=mod,oo=a-maxk+2;
  for (i=1;i<maxk;i++)
  {
      if (i==1)pc(oo,'o');else pc(1,'o');
      if (l) {pc(p,'x');l--;}else pc(p+1,'x');
  }
  }
  printf("\n");
}
int main()
{ 
  ans=0;
  scanf("%I64d%I64d",&a,&b);
  if (a<b)min=a+1;else min=b;
  ans=a*a-b*b;
  maxk=1;
  for (i=2;i<=min;i++)
  {
    now=(a-i+2)*(a-i+2)+i-2;
    p=b/i;
    mod=b%i;
    now-=p*p*(i-mod)+(p+1)*(p+1)*mod;
    if (now>ans){ans=now;maxk=i;}
  }
  printf("%I64d\n",ans);
  print();
  //system("pause");
  return 0;
}
