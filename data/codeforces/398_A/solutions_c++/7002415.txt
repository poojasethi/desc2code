#include <stdio.h>
typedef long long ll;
#define Min(a,b)(a<b?a:b)
ll a,b,ans,num;
void calc(ll x){
	ll res=(a-x+1)*(a-x+1)+x-1;
	x++;
	ll r=b%x;
	res-=(b/x)*(b/x)*(x-r);
	res-=r*(b/x+1)*(b/x+1);
	if(ans<res) {ans=res;num=x-1;}
}
void prnt(ll n){
	ll i,j,r=0;
	if(!n){
		for(i=0;i<a;i++) printf("o");
		for(i=0;i<b;i++) printf("x");
		return;
	}
	for(i=0;i<2*n+1;i++){
		if((i&1)){
			if(i<2*n-1) printf("o");
			else for(j=0;j<a-n+1;j++) printf("o");
		}
		else {
			if(r>=(b%(n+1))) for(j=0;j<b/(n+1);j++) printf("x");
			else {for(j=0;j<=b/(n+1);j++) printf("x");r++;}
		}
	}
	return;
}

int main(){
	ll i;
	scanf("%I64d %I64d",&a,&b);
	ll m=Min(a,b-1);
	ans=a*a-b*b;num=0;
	for(i=1;i<=m;i++) calc(i);
	printf("%I64d\n",ans);
	prnt(num);
	return 0;
}
