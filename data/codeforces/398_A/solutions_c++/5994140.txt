#include <stdio.h>
#include <vector>
#include <iterator>
#include <string>
#include <iostream>

#define pow2(x) ((x)*(x))

using namespace std;
typedef long long ll;

void print(char ch, int n) {
	while (n--)
	cout<<ch;
}

int main() {
	ll a, b;
	cin>>a>>b;
	if (!a) {
		cout<<-b * b<<endl;
		print('x', b);
		cout<<endl;
		return 0;
	}
	if (!b) {
		cout<<a * a<<endl;
		print('o', a);
		cout<<endl;
		return 0;
	}

	ll maxVal, maxX = -1;
	for (int x = 0; x <= a; ++x) {
		int n = x + (x < a) + 1;
		ll val = pow2(a - x) + x - (b % n) * pow2(b / n + 1) - (n - b % n) * pow2(b / n);
		if (maxX == -1 || maxVal < val) {
			maxVal = val;
			maxX = x;
		}
	}

	cout<<maxVal<<endl;
	int n = maxX + (maxX < a) + 1;
	//printf("n = %d, maxX = %d\n", n, maxX);
	for (int i = 0; i < n; ++i) {
		print('x', b / n + (i < (b % n)));
		if (i < n -1)
			print('o', !i ? int(a - maxX) : 1);
	}
	cout<<endl;
		
	return 0;
}
