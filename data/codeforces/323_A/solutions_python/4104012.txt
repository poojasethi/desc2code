import sys
def solve(n):
    if n % 2 == 1:
        print '-1'
        return

    for i in xrange(n):
        al = []
        for i2 in xrange(n):
            strr = []
            for i3 in xrange(n):
                if (i+i2/2+i3/2) % 2 == 0:
                    strr.append('b')
                else:
                    strr.append('w')
            al.append(''.join(strr))
        print '\n'.join(al)
        print '\n'

    
n = int(sys.stdin.readline())
solve(n)
