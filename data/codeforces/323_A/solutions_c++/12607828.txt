#include <iostream>
using namespace std;
int main () {
	int N;
	cin >> N;
	if (N & 1) {
		cout << -1 << endl;
		return 0;
	}
	for (int i = 0; i < N; i++) {
		cout<<endl;
		for (int j = 0; j < N; j++, cout << endl) {
			cout<<endl;
			for (int k = 0; k < N; k++) {
				int n = min (min (j,k),min (N-1-j, N-1-k));
				if ((i % 2) ^ (n % 2)){
						cout << "b";
					}
					else{
						cout << "w";
				}		
			}
		}
	}
	return 0;
}
