#include<stdio.h>
int n;
int main()
{
    scanf("%d",&n);
    if(n%2)
        printf("-1");
    else
    {
        int i, j, k;
        for(i=0; i<n; i++,printf("\n"))
            for(j=0;j<n;j++,printf("\n"))
                for(k=0;k<n;k++)
                {
                    if((i%2)^(((j/2)%2)^((k/2)%2)))
                        printf("b");
                    else
                        printf("w");
                }
    }
}
