#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <vector>
using namespace std;

#define FOR(it, c) for(__typeof((c).begin()) it = (c).begin(); it != (c).end(); it++)
#define SZ(c) ((int)(c).size())

int main(void) {
	int n;
	scanf("%d", &n);
	if(n%2) puts("-1");
	else {
		for(int i=0;i<n;i++ ,puts(""))
			for(int j=0;j<n;j++,puts(""))
				for(int k=0;k<n;k++)
					if((i%2)^(min(min(j,k),min(n-1-j,n-1-k))%2)) putchar('b');
					else putchar('w');
	}
	return 0;
}
