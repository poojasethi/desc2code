#include<cstdio>
using namespace std;
int main()
{
    int n;scanf("%d",&n);
    if(n%2==1)printf("-1\n");
    else
    {
        for(int h=0;h<n;h++)
        {

            for(int i=0;i<n;i++)
            {
                for(int j=0;j<n/2;j++)
                {
                    if((h+i/2+j)%2==0)printf("ww");else printf("bb");
                }
                printf("\n");
            }
            printf("\n");
        }

    }
    return 0;
}
