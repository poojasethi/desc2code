#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int n,i,j,k,t;
int main(){
	scanf("%d",&n);
	if(n&1)return puts("-1"),0;
	for(i=0;i<n;i++){
		for(j=0;j<n;j++){
			for(k=0;k<n;++k){
				t=i/2&1 ^ j/2&1 ^ k&1;
				if(t)printf("w");
				else printf("b");
			}
			puts("");
		}
		puts("");
	}
}
