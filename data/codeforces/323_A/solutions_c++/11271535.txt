#include <cstdio>
#include <algorithm>
#include <cstdlib>
using namespace std;

const int MAXN = 110;

int k;

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt", "r", stdin), freopen("output.txt", "w", stdout);
#endif
	scanf("%d", &k);
	if (k & 1)
		puts("-1"), exit(0);
	for (int i = 1; i <= k; ++i)
	{
		for (int j = 1; j <= k; ++j, puts(""))
			for (int p = 1; p <= k; ++p)
			{
				int tmp = min(min(j, p), min(k - j + 1, k - p + 1));
				tmp ^= (i & 1);
				if (tmp & 1)
					printf("b");
				else
				printf("w");
			}
		puts("");
	}
	fclose(stdin);
	fclose(stdout);
	return 0;
}
