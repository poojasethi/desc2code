#include <iostream>
#include <cstdio>

using namespace std;

int main() {
    int n;
    cin>>n;
    if (n % 2) {
        cout<<-1<<endl;
        return 0;
    }
    for(int k=0; k<n; ++k) {
        for(int i=0; i<n; ++i) {
            for(int j=0; j<n; ++j) {
//                if ((i/2%2) ^ (j/2%2) ^ (k%2))
//                    printf("w");
//                else
//                    printf("b");
                printf("%c", (((i/2%2) ^ (j/2%2) ^ (k%2)) == 1 ? 'b' : 'w'));
            }
            printf("\n");
        }
        printf("\n");
    }
    return 0;
}
