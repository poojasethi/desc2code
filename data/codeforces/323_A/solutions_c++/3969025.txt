#include <iostream>

using namespace std;

int n;

int main()
{
	cin>>n;
	if(n%2)
		{cout<<-1<<endl;return 0;}
	for(int i=0;i<n;++i)
	{
		for(int j=0;j<n;++j)
		{
			for(int l=0;l<n;++l)
				cout<<((min(min(j,n-j-1),min(l,n-l-1))+i)%2==0 ? "b" : "w"); 
			cout<<endl;
		}
		cout<<endl;
	}
	return 0;
}
