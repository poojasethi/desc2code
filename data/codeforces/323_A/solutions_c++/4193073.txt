#include<stdio.h>
char res[100][100][100];
int min(int a,int b){return a<b?a:b;}

int main(){
    int n;
    scanf("%d",&n);
    if(n%2==1)puts("-1");
    else{
        for(int i=0;i<n;i++){
            for(int j=0;j<n;j++){
                for(int k=0;k<n;k++){
                    int x=(min(min(j,n-j-1),min(k,n-k-1))+i)%2;
                    printf("%c",x==0?'w':'b');
                }
                puts("");
            }
            puts("");
        }
    }
    return 0;
}
