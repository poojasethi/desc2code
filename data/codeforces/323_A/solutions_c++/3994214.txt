#include<cstdio>
#include<cstdlib>

using namespace std;

int n;
int main()
{
    scanf("%d",&n);
    if (n % 2) printf("-1\n");
    else{
             for (int i = 0; i < n; i++)
             {
               for (int j = 0; j < n; j++)
               {
                 for (int k = 0; k < n; k++)
                   
                   printf("%c",  (i/2+k/2+j) % 2 ? 'w' : 'b' );
                 
                 printf("\n");
               }
              
               printf("\n");
               
             }
        }
    //system("pause");
    return 0;
}
