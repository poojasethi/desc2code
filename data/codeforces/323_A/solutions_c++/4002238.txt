#include<iostream>
using namespace std;

int main()
{
	int n;
	char c[2]={'w','b'};
	cin>>n;
	if(n%2==1) cout<<"-1";
	else{
       for(int i=1;i<=n;i++){
           for(int j=0;j<n;j++){
              for(int k=0;k<n;k++)
              {
              	if((j/2)%2==0)
              		if((k/2)%2==0) cout<<c[i%2];else cout<<c[!(i%2)];
              	if((j/2)%2==1) 
              	    if((k/2)%2==0) cout<<c[!(i%2)];else cout<<c[i%2];
              }
              cout<<endl;
          }
          cout<<endl;
      }
  }
  return 0;
}
