#include <iostream>
#include <cmath>
#include <string>

using namespace std;

int main()
{
	int k;
	cin>>k;
	if (k%2 == 1)
	{
		cout<<-1<<endl;
		return 0;
	}
	for (int i = 0; i<k; i++)
	{
		if (i>0)
			cout<<endl;
		for (int j = 0; j<k; j++)
		{
			for (int l = 0; l<k; l++)
			{
				int layer = min(j, min(l, min(k-j-1, k-l-1)));
				if (layer%2 == i%2)
					cout<<'w';
				else
					cout<<'b';
			}
			cout<<endl;
		}
	}
	return 0;
}
