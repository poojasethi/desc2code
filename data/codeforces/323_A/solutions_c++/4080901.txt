#include<stdio.h>
int main(){
    int n,i,j,k,f[105][105];
    scanf("%d",&n);
    if(n%2==1)printf("-1\n");
    else {
        for(i=1;i<=n/2;i++)
            for(j=i;j<=n-i+1;j++)
                for(k=i;k<=n-i+1;k++)
                    f[j][k]=i%2;
        for(k=1;k<=n;k++){
            for(i=1;i<=n;i++){
                for(j=1;j<=n;j++){
                    if(f[i][j]==1)printf("b");
                    else printf("w");
                    f[i][j]=1-f[i][j];
                }
                printf("\n");
            }               
        }
    }
}
