#include <iostream>
#include <stdio.h>
using namespace std;

int n;
bool a[100][100] = {};

int main()
{
#ifdef Svyat
  freopen("input.txt", "r", stdin);
#endif

  cin >> n;
  if (n & 1)
    return !printf("-1\n");

  for (int i = 0; i < n; ++i)
    for (int j = 0; j < n; ++j)
      a[i][j] = (i >> 1 & 1) ^ (j >> 1 & 1);

  for (int i = 0; i < n; ++i, putchar('\n'))
    for (int j = 0; j < n; ++j, putchar('\n'))
      for (int k = 0; k < n; ++k)
    if (i & 1)
      putchar(a[j][k] ? 'b' : 'w');
    else putchar(!a[j][k] ? 'b' : 'w');

  return 0;
}
