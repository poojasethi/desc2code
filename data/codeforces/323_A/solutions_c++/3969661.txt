#include <cstdio>

char thing (int a) {
  if (a % 2 == 0) return 'b';
  return 'w';
}

int main() {
  int k;
  scanf ("%d", &k);
  if (k % 2 == 1) printf ("%d\n", -1);
  else {
    for (int i = 0; i < k; i++) {
      for (int j = 0; j < k; j++) {
	for (int a = 0; a < k; a++) {
	  printf ("%c", thing (a /2 + j / 2 + i));
	}
	printf ("\n");
      }
      printf ("\n");
    }
  }
}
