#include<iostream>
using namespace std;
int main()
{
    int n;cin>>n;
    if(n%2==1)cout<<"-1"<<endl;
    else
    {
        for(int h=0;h<n;h++)
        {

            for(int i=0;i<n;i++)
            {
                for(int j=0;j<n/2;j++)
                {
                    if((h+i/2+j)%2==0)cout<<"ww";else cout<<"bb";
                }
                cout<<endl;
            }
            cout<<endl;
        }

    }
    return 0;
}
