#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;


int main()
{
	int k;
	cin >> k;

	if (k % 2 != 0)
	{
		cout << -1 << endl;
		return 0;
	}

	for (int z = 0; z < k; ++z)
	{
		for (int y = 0; y < k; ++y)
		{
			for (int x = 0; x < k; ++x)
			{
				if (((x >> 1) ^ y ^ (z >> 1)) % 2)
					cout << 'w';
				else
					cout << 'b';
			}
			cout << endl;
		}
		cout << endl;
	}

	return 0;
}
