#include <iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	if (n % 2) {
		cout << -1 << endl;
		return 0;
	}
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			for (int k = 0; k < n; k++)
				cout << ((j / 2 + k / 2 + i) % 2 == 0 ? 'b' : 'w');
			cout << endl;
		}
		cout << endl;
	}
	return 0;
}
