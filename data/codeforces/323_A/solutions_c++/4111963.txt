#include <iostream>

using namespace std;

const int max_dim = 112;

int k, a[max_dim][max_dim];

int main()
{
  for (int i = 0; i < max_dim; i += 2)
    for (int j = 0; j < max_dim; j += 2)
      a[i][j] = a[i + 1][j] = a[i][j + 1] = a[i + 1][j + 1] = (((i / 2) & 1) + ((j / 2) & 1)) & 1;
  cin >> k;
  if (k & 1) cout << -1;
  else
    for (int l = 0; l < k; l++)
    {
      for (int i = 0; i < k; i++)
      {
        for (int j = 0; j < k; j++)
          cout << (a[i][j] ^ (l & 1) ? 'b' : 'w');
        cout << endl;
      }
      cout << endl;
    }
  return 0;
}
