#include <iostream>
using namespace std;
int main()
{
    int p;
    cin >>p;
    if (p%2==1) {cout <<-1 <<endl; return 0;}
    else
    {
        for (int i=0; i<p; i++)
        {
            for (int j=0; j<p; j++)
            {
                for (int k=0; k<p; k++)
                {
                    if ((k/2)%2==(j/2)%2==(i%2==0))
                    {
                        cout <<"w";
                    }
                    else cout <<"b";
                }
                cout <<endl;
            }
            cout <<endl;
        }
    }
    return 0;
}
