#include <iostream>
using namespace std;

int main() {
    int k;
    cin>>k;
    if(k%2==1){cout<<"-1";return 0;}
    for(int i=0;i<k;i++){
        for(int j=0;j<k;j++){
            for(int p=0;p<k;p++){
                if((i+j/2+p/2)%2)
                    cout<<"w";
                else cout<<"b";
            }

            cout<<endl;
        }

        cout<<endl;
    }

    return 0;
}
