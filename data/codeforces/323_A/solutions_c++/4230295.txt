#include<cstdio>
#include<cstring>
using namespace std;
char col[2]={'b','w'};
char map[105][105];
int main(){
    int n;
    scanf("%d",&n);
    if(n&1){
        puts("-1");
        return 0;
    }
    int r=0,p=0,x,si;
    for(int i=0;i<n;i++){
        p=r;
        r^=1;
        for(x=0,si=n;si>=2;si-=2,x++){
            for(int j=0;j<si;j++){
                map[x][x+j]=map[x+si-1][x+j]=map[x+j][x]=map[x+j][x+si-1]=col[p];
            }
            p^=1;
        }
        for(int j=0;j<n;j++){
            map[j][n]='\0';
            puts(map[j]);
        }
        puts("");
    }
    return 0;
}
