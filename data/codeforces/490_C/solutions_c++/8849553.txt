#include <stdio.h>
#include <string.h>
const int maxlen = 1000001;
char c[maxlen];
int q[maxlen];
int main(){
	int a, b, l, d = 1, s = 0, get = 0;
	gets(c);
	scanf("%d %d",&a,&b);
	l = strlen(c);
	for(int i = l - 1; i >= 0; i --){
		s = (s + d * (c[i] - '0')) % b;
		q[i] = s;
		d = (d * 10) % b;
	}
	s = 0;
	for(int i = 0; i < l - 1; i ++){
		s = (s * 10 + c[i] - '0') % a;
		if(s == 0 && q[i + 1] == 0 && c[i + 1] != '0'){
			get = i + 1;
			break;
		}
	}
	if(get){
		puts("YES");
		s = c[get];
		c[get] = '\0';
		puts(c);
		c[get] = s;
		puts(c + get);
	}else puts("NO");
}
