#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
ll ar[1000009];

int main()
{
string s;
ll p=1,l,r,ss=0;

cin>>s>>l>>r;
int n=s.size();

for(int x=n-1; x>0; x--){
    ar[x]=(((s[x]-'0')*p)%r+ar[x+1])%r;
    p*=10;
    p%=r;
}

p=1;
for(int i=0; i<n-1; i++){

     ss=((ss*10)%l)+(s[i]-'0');
     ss%=l;
      if(ss==0 && ar[i+1]==0 && s[i+1]!='0'){cout<<"YES\n"<<s.substr(0,i+1)<<endl<<s.substr(i+1,n-i-1)<<endl;return 0;}

}
cout<<"NO\n";
    return 0;
}
