#include <cstdio>
#include <cstring>
char in[1048576], vis[1048576];
int main() {
    int a, b, n, t(0), base(1), i; scanf("%s%d%d", in, &a, &b);
    n = strlen(in);
    for (i = 1; i < n; ++i) {
        t = (t * 10 + in[i - 1] - '0') % a;
        vis[i] = !t;
    }
    t = 0;
    for (i = n - 1; i > 0; --i) {
        (t += base * (in[i] - '0')) %= b;
        if ((in[i] - '0') && !t && vis[i]) break;
        (base *= 10) %= b;
    }
    if (!i) puts("NO");
    else {
        puts("YES");
        for (int j = 0; j < i; ++j) putchar(in[j]);
        putchar('\n'); puts(in + i);
    }
}
