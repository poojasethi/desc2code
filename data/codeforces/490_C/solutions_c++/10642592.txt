#include <bits/stdc++.h>
using namespace std;
int a,b,n,p=1, i, d[1000005];
int main() 
{   
    char s[1000005];
    cin>>s;
    n = strlen(s);
    cin>>a>>b;
    for (i=n-1; i>=0; i--) 
    {
        d[i]=((s[i]-'0')*p+d[i+1])%b;
        p=(p*10)%b;
    }
    p=0;
    for(i=1;i<n;i++) 
    {
        p=(p*10+s[i-1]-'0')%a;
        if (!p && !d[i] && s[i] != '0') 
        {
            cout<<"YES"<<endl;
            printf("%.*s\n%s",i,s,s+i);
            return 0;
        }
    }
    cout<<"NO"<<endl;
    return 0;
}
