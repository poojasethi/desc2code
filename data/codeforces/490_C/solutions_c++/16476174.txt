#include <bits/stdc++.h>
using namespace std;
int s[1000001];
int main() {
    ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
    int num=0, mult=1, a, b;
    string t;
    cin >> t >> a >> b;
    for(int i=t.size()-1 ; i>0 ; i--){
    	num=(num+((t[i]-'0')*mult))%b;
    	mult=(mult*10)%b;
    	if(!num && t[i]!='0')
    		s[i]=1;
    }
    num=0;
    for(unsigned int i=0 ; i<t.size() ; i++){
    	num=(num*10+(t[i]-'0'))%a;
    	if(!num && s[i+1]){
    		cout << "YES\n" << t.substr(0,i+1) << '\n' << t.substr(i+1,t.size()-i);
    		return 0;
    	}
    }
    cout << "NO";
	return 0;
}
