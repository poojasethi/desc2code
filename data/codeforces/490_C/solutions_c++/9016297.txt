#include <bits/stdc++.h>
#define S 1000005
using namespace std;
char m[S];
int a, b, l, p = 1, r[S];

int main() {
//    freopen("input.txt", "r", stdin);
    int i;
    cin >> m;
    l = strlen(m);
    cin >> a >> b;
    for (i = l - 1; i >= 0; i--) {
        r[i] = ((m[i] - '0') * p + r[i + 1]) % b;
        p = p * 10 % b;
    }
    p = 0;
    for (i = 0; i < l- 1; i++) {
        p = (p * 10 + m[i] - '0') % a;
        if (p == 0 && r[i + 1] == 0 && m[i + 1] != '0') {
            puts("YES");
            printf("%.*s\n%s", i + 1, m, m + i + 1);
            return 0;
        }
    }
    puts("NO");
}
