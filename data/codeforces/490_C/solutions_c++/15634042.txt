#include<iostream>
#include<string>
using namespace std;
int main()
{
	string c;
	cin>>c;
	int a,b,n=c.length(),rem=0,k=1;
	cin>>a>>b;
	int d[1000001]={0};
	for(int i=0;i<n-1;i++)
	{
		rem=(rem*10+(c[i]-'0'))%a;
		if(!rem)d[i]++;
	}rem=0;
	for(int i=n-1;i>0;i--)
	{
		rem=(rem+((c[i]-'0')*k)%b)%b;
		if(c[i]!='0'&&rem==0)d[i-1]++;
		k=(k*10)%b;
	}
	for(int i=0;i<n;i++)
	{
		if(d[i]==2)
		{
			cout<<"YES"<<endl<<c.substr(0,i+1)<<endl<<c.substr(i+1);
			return 0;
		}
	}
	cout<<"NO";
}
