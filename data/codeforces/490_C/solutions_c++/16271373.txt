#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
const int N=1000001;
char str[N];
int r[N];
int main()
{
    int a,b,len,d=1,s,t,i;
	gets(str);
	scanf("%d %d",&a,&b);
	len=strlen(str);
	s=0;
	for(i=len-1;i>=0;i--)
    {
		s=(s+d*(str[i]-'0')) % b;
		r[i]=s;
		d=(d*10) % b;
	}
	s=t=0;
	for(i=0;i<len-1;i++)
    {
		s=(s*10+str[i]-'0') % a;
		if(s==0 && r[i+1] == 0 && str[i+1] != '0')
		{
			t=i+1;
			break;
		}
	}
	if(t)
    {
		puts("YES");
		s=str[t];
		str[t]='\0';
		puts(str);
		str[t]=s;
		puts(str+t);
	}
	else
        puts("NO");
}
