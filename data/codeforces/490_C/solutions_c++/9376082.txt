#include <iostream>
#include <string>
using namespace std;
int main() {
	string s; cin >> s; int size = s.size();
	int n, m, ar[size]; cin >> n >> m; ar[0] = 0;
	for(int i = 0; i < size - 1; ++i)
		ar[i] = (ar[!i ? 0 : i - 1] * 10 + (int)(s[i] - '0')) % n;
	for(int i = size - 1, o = 0, k = 1; i > 0; --i, k = (k * 10) % m){
		o = (k * (int)(s[i] - '0') + o) % m; 
		if(!o && !ar[i - 1] && s[i] != '0') {
			cout << "YES" << endl;
			for(int j = 0; j < s.size(); ++j){
				cout << s[j];
				if(j + 1 == i) cout << endl;
			}
			return 0;
		}
	}
	cout << "NO";
}
