#include<bits/stdc++.h>
using namespace std;
int a, b, n, p=1, i, rem[1000005]={0};
int main()
{
    string s;
    cin>>s;
    n = s.length();
    cin>>a>>b;
    for(int i=n-1; i>=0; i--)
    {
        rem[i]=((s[i]-'0')*p + rem[i+1])%b;
        p=(p*10)%b;
    }
    p=0;
    for(int i=1; i<n; i++)
    {
        p=(p*10 + s[i-1]-'0')%a;
        if(!p && !rem[i] && s[i]!='0')
        {
            cout<<"YES"<<endl;
            cout<<s.substr(0,i)<<endl<<s.substr(i,-1);
            return 0;
        }
    }
    cout<<"NO";
    return 0;
}
