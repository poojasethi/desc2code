#include<cstdio>
#include<cstring>

static char s[1000000+1];
static int a,b,n;
static int right[1000000];

static void output(int l,int r) {
	while(l<=r)putchar(s[l++]);
	putchar('\n');
}

static bool ker(int l,int r) {
	return s[l]!='0';
}

int main() {
	scanf("%s%d%d",s,&a,&b);
	n=strlen(s);
	{
		int x=0,p=1;
		for(int i=n-1;i>=0;i--) {
			x=(x+(s[i]-'0')*p)%b;
			right[i]=x;
			p=p*10%b;
		}
	}
	{
		int x=0;
		for(int i=0;i+1<n;i++) {
			x=(x*10+(s[i]-'0'))%a;
			if(x==0&&right[i+1]==0&&ker(0,i)&&ker(i+1,n-1)) {
				puts("YES");
				output(0,i);
				output(i+1,n-1);
				return 0;
			}
		}
	}
	puts("NO");
	return 0;
}
