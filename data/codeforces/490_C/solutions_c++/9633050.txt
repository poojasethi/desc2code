#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char str[1000001];
int r[1000001];
int main()
{
    int a,b,l,d=1,s,t,i;
	gets(str);
	scanf("%d %d",&a,&b);
	l=strlen(str);
	s=0;
	for(i=l-1; i>=0; i--)
    {
		s=(s+ d*(str[i]-'0')) % b;
		r[i]=s;
		d=(d*10) % b;
	}
	s=t=0;
	for(i=0; i<l-1; i++)
    {
		s=(s*10 + str[i] - '0') % a;
		if(s==0 && r[i+1] == 0 && str[i+1] != '0')
		{
			t=i+1;
			break;
		}
	}
	if(t)
    {
		puts("YES");
		s=str[t];
		str[t]='\0';
		puts(str);
		str[t]=s;
		puts(str+t);
	}
	else
        puts("NO");
}
