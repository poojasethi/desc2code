#include<cstdio>
#include<cstring>
#define ll __int64
const int N=1000005;
char str[N];
ll mod[N];
ll a,b;
int main()
{
	scanf(" %s",str+1);
	scanf("%I64d%I64d",&a,&b);
	int len=strlen(str+1);
	for(int i=1;i<len;i++){
		mod[i]=(mod[i-1]*10+str[i]-'0')%a;
	}
	ll n=1,t=0;
	int k,flag=0;;
	for(int i=len;i>1;i--){
		t=((str[i]-'0')*n+t)%b;
		n=n*10%b;
		if(t==0&&mod[i-1]==0&&str[i]!='0')
			flag=1,k=i-1;
	}
	if(flag){
		puts("YES");
		for(int i=1;i<=k;i++)
			printf("%c",str[i]);
		puts("");
		puts(str+k+1);
	}else
		puts("NO");
	return 0;
}
