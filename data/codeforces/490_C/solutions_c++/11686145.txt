#include <cstdio>
#include <cstring>
#include <iostream>

int a , b , c , p = 1 , d[2000010];
char m[2000010];

int main(){
    scanf("%s",&m);
    int n = strlen(m);
    scanf("%d%d",&a,&b);
    for(int i = n - 1; i >= 0 ; i--){
        d[i] = ((m[i] - '0') * p + d[i + 1]) % b;
        p = p * 10 % b;
    }
    for(int i = 1 ; i < n ; i++){
        c = (c * 10 + m[i - 1] - '0') % a;
        if(!c && !d[i] && m[i] != '0'){
            printf("YES\n%.*s\n%s",i,m,m + i);
            return 0;
        }
    }
    printf("NO");
    return 0;   
}
