#include<cstdio>
#include<cstring>

int a, b, len, x, y;
char s[1111111];
int v[1111111];

int main() {
	scanf("%s%d%d", s, &a, &b);
	len = strlen(s);
	x=0;
	for(int i=0;i<len;i++) {
		x=(x*10 + s[i]-'0') % a;
		v[i] = x;
	}
	x=0;y=1;
	for(int i=len-1;i>0;i--) {
		x=(x+y*(s[i]-'0')) % b;
		if(v[i-1]==0 && s[i]!='0'&&!x) {
			printf("YES\n");
			for(int j=0;j<i;j++) printf("%c", s[j]);
			printf("\n");
			for(int j=i;j<len;j++) printf("%c", s[j]);
			printf("\n");
			return 0;
		}
		y = y*10 % b;
	}
	printf("NO\n");
	
	return 0;
}
