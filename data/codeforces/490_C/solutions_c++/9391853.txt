#include<iostream>
#include<string>
using namespace std;
int main()
{
	string s;
	long long int i,j,l;
	cin>>s;
	j=s.length();
	long long int p,a,b,num,rem[j],rem2[j];
	cin>>a>>b;
	num=0;
	for(i=0;i<j;i++)
	{
		num=((num*10)%a+(s[i]-'0'))%a;
		rem[i]=num%a;
	}
	p=1;
	num=0;
	for(i=j-1;i>=0;i--)
	{
		num=(num+(p*(s[i]-'0')%b)%b)%b;
		p=(p*10)%b;
		rem2[i]=num;
	}
	for(i=0;i<j-1;i++)
	{
		if(rem[i]==0 && rem2[i+1]==0 && s[i+1]!='0')
		{
			cout<<"YES\n";
			cout<<s.substr(0,i+1)<<endl<<s.substr(i+1,j);
			return 0;
		}
	}
		cout<<"NO"<<endl;
		return 0;
}
