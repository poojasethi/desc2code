#import<bits/stdc++.h>
int a, b, c, n, p=1, i, d[1<<20];
char m[1<<20];
main() {
	gets(m);
	n = strlen(m);
	scanf("%d%d", &a, &b);
	for (i=n-1; i>=0; i--) {
		d[i] = ((m[i] - '0') * p + d[i + 1]) % b;
		p = p * 10 % b;
	}
	for (i=1; i<n; i++) {
		c = (c * 10 + m[i - 1] - '0') % a;
		if (!c && !d[i] && m[i] != '0') {
			puts("YES");
			printf("%.*s\n%s", i, m, m + i);
			return 0;
		}
	}
	puts("NO");
}
