#include <bits/stdc++.h>
using namespace std;

int main(){
	ios::sync_with_stdio(false);
	string s;
	cin>>s;
	int a , b , l , i, j , flag=0 , rem1=0 , rem1_arr[1000005] ,  rem2=0;
	cin>>a>>b;
	l=s.size();
	for( i=0 ; i<(l-1) ; i++){
		rem1=(rem1*10+(s[i]-'0'))%a;
		rem1_arr[i]=rem1;
	}
	int power=1;
	for(j=l-1 ; j>0 ; j--){
		rem2=(rem2+(s[j]-'0')*power)%b;
		power=(power*10)%b;
		if(rem2==0 && rem1_arr[j-1]==0 && s[j]!='0') { flag=1; break; }
	}
	if(flag){
		cout<<"YES\n";
		cout<<s.substr(0 , j)<<"\n";
		cout<<s.substr(j , l-j)<<"\n";
	}
	else cout<<"NO\n";
	return 0;
}
