#include<stdio.h>

#define maxn 1001000

char p[maxn];

int q[maxn];

int i,j,n,m,a,b;

int main() {
  scanf("%s%n",p,&n);
  scanf("%d %d",&a,&b);
  j=1;
  for(i=n-1;i>=0;i--){
    p[i]-='0';
    q[i]=(q[i+1] + j*p[i]) %  b;
    j=(j*10)%b;
  }
  int now=0;
  for(i=0;i<n-1;i++){
    now = (now*10+p[i])%a;
    if(now==0 && q[i+1] ==0 &&p[i+1])
      break;
  }
  if (i==n-1) {
    printf("NO\n");
    return 0;
  }
  printf("YES\n");
  j=i;
  for(i=0;i<=j;i++)
    printf("%d",p[i]);
  printf("\n");
  while(i<n){
    printf("%d",p[i]);
    i++;
  }
  printf("\n");
  return 0;
}
