#include <bits/stdc++.h>

using namespace std;

char S[1000000+10];
int a, b, n;
set <int> candinate;

int main(){
	scanf("%s", S);
	scanf("%d %d", &a, &b);

	n = strlen(S);

	int p = 0;
	for(int i=0; i<n-1; ++i){
		p = p*10 + S[i] -'0';
		p %= a;
		if(p == 0 && S[i+1] > '0'){
			candinate.insert(i);
		}
	}
		
	p = 0;
	for(long long i=n-1, j=1; i>0; --i){
		p = p + (S[i] - '0')*j;
		p %= b;
		j = j*10%b;
		if(p == 0 && candinate.find(i-1) != candinate.end()){
			printf("YES\n");
			for(int k=0; k<i; ++k) printf("%c", S[k]);
			printf("\n");
			printf("%s\n", S+i);
			return 0;
		}
	}

	printf("NO\n");
	return 0;
}
