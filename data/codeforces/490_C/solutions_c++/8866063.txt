#include <stdio.h>
#include <string.h>

#define N 1000001
int x[N],y[N];
char s[N],c;

int main(){
  int a,b,l,p,t,i,k;
  scanf("%s%d%d",s,&a,&b);
  l = strlen(s);
  p = 0;
  for(i=0;i+1<l;i++){
    t = (s[i]-'0');
    p = (p*10 + t)%a;
    if(p==0&&s[i+1]!='0') 
      x[i+1]=1;
  }
  p = 0;k=1;
  for(i=l-1;i>0;i--){
    t = (s[i]-'0');
    p = (p + t*k)%b;
    k = (k*10)%b;
    if(p==0)
      y[i]=1;
  }
  for(i=1;i<l;i++)
    if(x[i]==1&&y[i]==1)
      break;
  if(i!=l){
    c = s[i];s[i]=0;
    printf("YES\n%s\n",s);
    s[i]=c;
    printf("%s\n",s+i);
  }
  else
    printf("NO\n");
  return 0;
}
