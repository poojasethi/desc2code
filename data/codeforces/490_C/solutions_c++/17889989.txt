#include<iostream>
#include<string>
#include<string.h>
using namespace std;
int A,B,Test[1000001];

int main(){

string x;
int a,b,k=1;
cin >> x >> a >> b;
int n = x.length();

for(int i=0;i<n-1;i++)
{
A=(A*10+(x[i]-'0'))%a;
if(A==0)Test[i]++;
}

for(int j=n-1;j>=0;j--)
{
B=(B+((x[j]-'0')*k)%b)%b;
if(B==0 && x[j]!='0')
Test[j-1]++;
k=(k*10)%b;
}

for(int i=0;i<n;i++)
	{
		if(Test[i]==2)
		{
			cout<<"YES"<<endl<<x.substr(0,i+1)<<endl<<x.substr(i+1);
			return 0;
		}
	}
cout << "NO";

}
