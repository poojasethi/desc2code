#include<bits/stdc++.h>
const int Maxn = 2000000 + 10;
int d[Maxn];
char s[Maxn];
int a,b,c,p=1;

int main()
{
    int n;
    gets(s);
    n=strlen(s);
    scanf("%d%d",&a,&b);
    for(int i=n-1;i>=0;i--)
    {
        d[i]=(d[i+1] + p*(s[i]-'0'))%b;
        p=p*10%b;
    }
    for(int i=1;i<n;i++)
    {
        c=(c*10 + s[i-1] - '0')%a;
        if(!c && !d[i] && s[i]!='0')
        {
            printf("YES\n%.*s\n%s",i,s,s+i);
            return 0;
        }
    }
    puts("NO");
    return 0;
}
