#include <iostream>
#include <algorithm>
using namespace std;

struct gp
{
int tedad;
int money;
int nobat;
};
gp a[1010];
pair<int , int> c[1010];
int b[1010];

bool fac (gp a , gp b)
{
return a.money > b.money ;
}

int main()
{
int n;
cin >>n;
for(int i=0;i<n;i++)
{
cin >> a[i].tedad >> a[i].money;
a[i].nobat = i+1;
}
int m;
cin >> m;
for(int i=0;i<m;i++)
cin >> b[i];

sort( a , a+n , fac) ;

long long int sum =0;
int miz = 0;

for(int i=0;i< n;i++)
{
int s=0;
int Min = 1000000;
int d=0;
for(int j=0;j<m;j++)
{
if(b[j] >= a[i].tedad)
{
s++;
int z = b[j]- a[i].tedad;
Min=min(Min , z);
if(Min == z)
{
c[miz].second = j+1;
d = j;
}
}
}
if(s > 0)
{
c[miz].first = a[i].nobat;
miz++;
sum +=a[i].money;
b[d] = 0;
}

}
cout << miz <<" "<< sum<<endl;
for(int i =0;i<miz;i++)
cout << c[i].first << " " << c[i].second<<endl;
}
