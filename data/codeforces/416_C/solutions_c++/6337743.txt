#include <iostream>
#include <algorithm>
using namespace std;
int n,m,c[2000],r[2000],p[2000],o[2000];
bool used[2000];
int ans1[2000],ans2[2000],ans,totalSum;
inline bool comp(int a, int b){ return p[a]>p[b]; }
int main(){
	cin>>n;
	for(int i=0;i<n;i++) {
		cin>>c[i]>>p[i];
		o[i]=i;
	}
	cin>>m;
	for(int i=0;i<m;i++) {
		cin>>r[i];
	}
	sort(o,o+n,comp);
	for(int i=0;i<n;i++) {
		int minR=10000;
		int minJ=-1;
		for(int j=0;j<m;j++) {
			if(!used[j] && r[j]>=c[o[i]] && r[j]<minR) {
				minR=r[j];
				minJ=j;
			}	
		}
		if(minJ>-1) {
			used[minJ]=1;
			ans1[ans]=o[i]+1;
			ans2[ans]=minJ+1;
			ans++;
			totalSum+=p[o[i]];
		}	
	}
	cout<<ans<<" "<<totalSum<<endl;
	for(int i=0;i<ans;i++) cout<<ans1[i]<<" "<<ans2[i]<<endl;
	return 0;
}
