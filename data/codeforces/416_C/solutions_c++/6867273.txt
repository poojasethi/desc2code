#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;
const int MAXN=1100;
pair <int,int> a[MAXN],r[MAXN];
int ans[MAXN];
bool mark[MAXN];
#define X first
#define Y second

int main()
{
  int n;
  cin>>n;
  for(int i=1;i<=n;i++)
    cin>>a[i].X>>a[i].Y;
  int k;
  cin>>k;
  for(int i=1;i<=k;i++)
    cin>>r[i].X,r[i].Y=i;
    sort(r+1,r+1+k);
  long long sum=0,num=0;
  for(int i=1;i<=k;i++)
    {
      int Max=0;
      for(int j=1;j<=n;j++)
	if(a[j].X<=r[i].X && a[j].Y>a[Max].Y && mark[j]==false)
	  {
	    //if(i==6)
	    //cerr<<"6::"<<Max<<endl;
	    Max=j;
	  }
      if(Max!=0)
	{
	  ans[r[i].Y]=Max;
	  sum+=a[Max].Y,num++;
	  mark[Max]=true;
	}
    }
  cout<<num<<" "<<sum<<endl;
  for(int i=1;i<=k;i++)
    if(ans[i]!=0)
      cout<<ans[i]<<" "<<i<<endl;
  return 0;
}
