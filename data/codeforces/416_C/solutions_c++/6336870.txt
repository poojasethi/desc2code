#include<cstdio>
#include<cstring>

#include<queue>
#include<iomanip>
#include<vector>

#include<cstdio>
using namespace std;

int n;
struct node{
int c,p,id;
}a[1111];
bool cmp(node a,node s){return a.p>s.p;}
int vis[1111],r[1111],k;
int res[1111];

int main(){
cin>>n;
memset(res,-1,sizeof(res));
for(int i=0;i<n;i++)cin>>a[i].c>>a[i].p,a[i].id=i+1;
sort(a,a+n,cmp);
cin>>k;
for(int i=0;i<k;i++)cin>>r[i];
int ans=0,num=0;
for(int i=0;i<n;i++){
int j=-1;
for(int s=0;s<k;s++)
if(vis[s]==0&&r[s]>=a[i].c){
if(j==-1)j=s;
else if(r[s]<=r[j])j=s;
}
if(j==-1)continue;
vis[j]=1;
ans+=a[i].p;
res[a[i].id]=j+1;
num++;
}
cout<<num<<' '<<ans<<endl;
for(int i=1;i<=n;i++)
if(res[i]!=-1)cout<<i<<' '<<res[i]<<endl;
}
