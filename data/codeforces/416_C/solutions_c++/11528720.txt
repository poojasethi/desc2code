#include<bits/stdc++.h>
using namespace std;
int check[1010];
vector<pair<pair<int,int>,int> >v;
vector<pair<int,int> >ans;
vector<pair<int,int> >table;
int main()
{
	int i,j,n,x,y,k;
	ios_base::sync_with_stdio(false);
	cin>>n;
	for(i=1;i<=n;i++)
	{
		cin>>x>>y;
		v.push_back(make_pair(make_pair(y,x),i));
	}
	sort(v.begin(),v.end());
	cin>>k;
	for(i=1;i<=k;i++)
	{
		cin>>x;
		table.push_back(make_pair(x,i));
	}
	sort(table.begin(),table.end());
	x=0;
	for(i=v.size()-1;i>=0;i--)
	{
		for(j=0;j<table.size();j++)
		{
			if(table[j].first>=v[i].first.second&&!check[table[j].second])
			{
				x+=v[i].first.first;
				check[table[j].second]=1;
				ans.push_back(make_pair(v[i].second,table[j].second));
				break;
			}
		}
	}	
	cout<<ans.size()<<" "<<x<<endl;
	for(i=0;i<ans.size();i++)
		cout<<ans[i].first<<" "<<ans[i].second<<endl;
	return 0;
}
