#include<iostream>
#include<algorithm>
#define inf 10000

using namespace std ;

const int maxn=2000 ;
int a[maxn] , b[maxn] , c[maxn] , p[maxn] , t[maxn] ;
int n , m , d , k , i , j , ans=0 , cnt=0 ; 

int main()
{
	cin >> n ;
	for(i=1; i<=n ; i++)
		cin >> c[i] >> p[i] ;
 	cin >> m ;
 	fill(t,t+maxn, inf) ;
	for(i=1 ; i<=m ; i++)
		cin >>t[i] ;
	for(i=1 ; i<=n ; i++)
	{
		k=0 , d=0 ;
		for(j=1 ; j<=n ; j++)
			if(!a[j] && p[k]<p[j])
				k=j ;
		if(k)
		{
			for(j=1 ; j<=m ; j++)
				if(!b[j] && t[j]>=c[k] && t[d]>t[j])
					d=j ;
			if(d)
			{
				a[k]=d ;
				b[d]=k ;
				ans+=p[k] ;
				cnt++ ;
			}
			else
				a[k]= inf ;
		} 
		
	}
	cout << cnt << " " << ans << endl ;
	for(i=1 ; i<=n ; i++)
		if(a[i]>0 && a[i]<inf)
			cout << i << " " << a[i] << endl ;
	return 0 ;
}
