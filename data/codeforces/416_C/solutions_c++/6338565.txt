#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
int c[2100],p[2100],hasput[2100];
struct T
{
	int pos,size;
	friend bool operator<(T a,T b)
	{
		return a.size<b.size;
	}
}a[2100];
int main(int argc, char *argv[])
{
	int n;scanf("%d",&n);
	for(int i=1;i<=n;i++)scanf("%d%d",&c[i],&p[i]);
	int m;scanf("%d",&m);
	for(int i=1;i<=m;i++)scanf("%d",&a[i].size),a[i].pos=i;
	sort(a+1,a+m+1);
	for(int i=1;i<=m;i++)
	{
		
		int wc=0;
		for(int j=1;j<=n;j++)
			if(!hasput[j]&&c[j]<=a[i].size&&p[j]>p[wc])wc=j;
		if(wc)hasput[wc]=a[i].pos;
	}
	int tmp=0,tans=0;
	for(int i=1;i<=n;i++)
		if(hasput[i])tmp++,tans+=p[i];
	printf("%d %d\n",tmp,tans);
	for(int i=1;i<=n;i++)
		if(hasput[i])printf("%d %d\n",i,hasput[i]);
	return 0;
}
