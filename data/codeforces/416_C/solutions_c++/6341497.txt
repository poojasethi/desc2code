#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int n,k;
pair<int,pair<int,int> > a[2000];
pair<int,int> r[2000];
vector< pair<int , int> > v;
bool visited[2000];
int main()
{
	cin>>n;
	for(int i=0;i<n;i++)
	{
		cin>>a[i].second.first>>a[i].first;
		a[i].second.second=i+1;
	}
	cin>>k;

	for(int i=0;i<k	;i++)
	{
		cin>>r[i].first;
		r[i].second=i+1;
	}
	int sum=0;
	sort(a,a+n);
	reverse(a,a+n);
	sort(r,r+k);
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<k;j++)
		{
		if(r[j].first>=a[i].second.first&&!visited[j])
			{	
				v.push_back(make_pair(a[i].second.second,r[j].second));
				sum+=a[i].first;
				visited[j]=true;
				break;
			}
		}
	}
	cout<<v.size()<<" "<<sum<<endl;
	for(int i=0;i<v.size();i++)
	{
		cout<<v[i].first<<" "<<v[i].second<<endl;
	}
}
