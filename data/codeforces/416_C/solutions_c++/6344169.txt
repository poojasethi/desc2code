#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define maxn 20010

struct Fvck
{
    int c,p,pos;
    friend bool operator < (Fvck a, Fvck b) { if (b.p==a.p) return a.c<b.c; return a.p>b.p; }
}q[maxn], a[maxn];
int cnt, ans[maxn], sum, ch[maxn];
int n, m;
bool f[maxn];

inline void print()
{
    printf("%d %d\n",cnt, sum); for (int i=1;i<=cnt;i++) printf("%d %d\n",ans[i],ch[i]);
}

int main ()
{
    scanf("%d", &n); for (int i=1; i<=n; i++) { scanf("%d%d",&q[i].c, &q[i].p); q[i].pos=i; } sort(q+1,q+n+1);
    scanf("%d", &m); for (int i=1; i<=m; i++) scanf("%d",&a[i].c), a[i].pos=i, a[i].p=1; sort(a+1, a+m+1);
    for (int i=1; i<=n; i++)
        for (int j=1; j<=m; j++)
            if (!f[j])
                if (a[j].c>=q[i].c) { f[j]=1; cnt++; ans[cnt]=q[i].pos; ch[cnt]=a[j].pos; sum+=q[i].p; break; }
    print();    
}
