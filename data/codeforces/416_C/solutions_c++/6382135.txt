#include <iostream>
#include <algorithm>

using namespace std;

typedef pair<int, int> pii;

const int MAXN = 1e5;
pii all[MAXN], r[MAXN];
int ans[MAXN], c[MAXN];
bool mark[MAXN];

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
		cin >> c[i] >> all[i].first, all[i].second = i;
	sort(all, all + n);
	reverse(all, all + n);
	int k;
	cin >> k;
	for (int i = 0; i < k; i++)
		cin >> r[i].first, r[i].second = i;
	sort(r, r + k);
	int mon = 0, num = 0;
	for (int i = 0; i < k; i++) {
		for (int j = 0; j < n; j++)
			if (!mark[j] && c[all[j].second] <= r[i].first) {
				ans[r[i].second] = all[j].second + 1;
				mark[j] = true;
				mon += all[j].first;
				num++;
				break;
			}
	}
	cout << num << " " << mon << endl;
	for (int i = 0; i < k; i++)
		if (ans[i] > 0)
			cout << ans[i] << " " << i + 1 << endl;
	return 0;
}
