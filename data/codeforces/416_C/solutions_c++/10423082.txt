#include <bits/stdc++.h>

using namespace std;
#define For(i,po,ki) for(int i=po;i<=ki;i++)
#define Down(i,po,ki) for(int i=po;i>=ki;i--)
#define ll long long
#define P first
#define C second

int n,m,t,k,kil,sum,ans[1001];
pair <int,int>z[1001];
multiset <pair<int, int> > r;
multiset <pair<int, int> > :: iterator it;
int main()
{ cin>>n;
  For(i,1,n) {cin>>z[i].C>>z[i].P; z[i].P=-z[i].P; z[i].C+=i*10000;}
  sort(z+1,z+1+n);
  cin>>k;
  For(i,1,k) {cin>>t; r.insert(make_pair(t,i));}
  For(i,1,n) {it=r.upper_bound(make_pair(z[i].C%10000,0));
   if (it!=r.end()) {kil++;sum-=z[i].P; ans[z[i].C/10000]=it->second;r.erase(it);}}
  cout<<kil<<" "<<sum<<endl;
  For(i,1,n) if (ans[i]>0) cout<<i<<" "<<ans[i]<<endl;
 }
