#include<stdio.h>
int main()
{
	int n,c[1000],p[1000],q[1000],t;
	int k,r[1000],g[1000],m=0,s=0,mi[1000],mj[1000];
	int i,j;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&c[i],&p[i]);
		q[i]=i;
	}
	for(i=1;i<n;i++)
		for(j=0;j<n-i;j++)
			if(p[j]<p[j+1])
			{
				t=p[j];
				p[j]=p[j+1];
				p[j+1]=t;
				t=c[j];
				c[j]=c[j+1];
				c[j+1]=t;
				t=q[j];
				q[j]=q[j+1];
				q[j+1]=t;
			}
	scanf("%d",&k);
	for(i=0;i<k;i++)
	{
		scanf("%d",&r[i]);
		g[i]=i;
	}
	for(i=1;i<k;i++)
		for(j=0;j<k-i;j++)
			if(r[j]>r[j+1])
			{
				t=r[j];
				r[j]=r[j+1];
				r[j+1]=t;
				t=g[j];
				g[j]=g[j+1];
				g[j+1]=t;
			}
	for(i=0;i<n;i++)
		for(j=0;j<k;j++)
			if(r[j]>=c[i])
			{
				s+=p[i];
				r[j]=0;
				mi[m]=q[i]+1;
				mj[m]=g[j]+1;
				m++;
				break;
			}
	printf("%d %d\n",m,s);
	for(i=0;i<m;i++)
		printf("%d %d\n",mi[i],mj[i]);
	return 0;
}
