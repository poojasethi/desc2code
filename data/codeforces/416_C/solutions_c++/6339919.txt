#include<stdio.h>
#include<iostream>
#include<vector>
#include<algorithm>

using namespace std;
int used[1004];

int main()
{
	int i,j,n,m,k,c,p,money=0;
	string s;
	scanf("%d",&n);
	vector<pair<pair<int,int>,int> > v;
	vector<pair<int,int> > t;
	vector<pair<int,int> > ans;
	for(i=1;i<=n;i++)
	{
		cin >> c >> p;
		v.push_back(make_pair(make_pair(p,c),i));
	}
	cin >> m;
	for(i=1;i<=m;i++)
	{
		cin >> c;
		t.push_back(make_pair(c,i));
	}
	sort(v.begin(),v.end());
	sort(t.begin(),t.end());
	for(i=n-1;i>=0;i--)
	{
		for(j=0;j<m;j++)
		{
			if(used[j])	continue;
			if(v[i].first.second<=t[j].first)
			{
				used[j]=1;
				money+=v[i].first.first;
				ans.push_back(make_pair(v[i].second,t[j].second));
				break;
			}
		}
	}
	printf("%d %d\n",ans.size(),money);
	for(i=0;i<ans.size();i++)
	{
		printf("%d %d\n",ans[i].first,ans[i].second);
	}
	printf("");
	return 0;
}
