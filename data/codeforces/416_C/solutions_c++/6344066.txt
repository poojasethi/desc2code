#include<iostream>
#include<algorithm>
using namespace std;
int n,k,c[1000],p[1000],r[1000],a[1000],b[1000],u[1000],sum,sumn;
bool cmpa(int x,int y){
	if(r[x]<r[y]) return 1;
	return 0;
}
int main(){
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>c[i]>>p[i];
	}
	cin>>k;
	for(int i=0;i<k;i++){
		cin>>r[i];
		a[i]=i;
	}
	sort(a,a+k,cmpa);
	int maxi;
	for(int i=0;i<1000;i++){
	}
	for(int i=0;i<k;i++){
		maxi=0;
		for(int j=0;j<n;j++){
			if(r[a[i]]>=c[j])if(maxi<p[j]){
				b[a[i]]=j;
				maxi=p[j];
				u[a[i]]=1;
			}
		}
		sumn+=u[a[i]];
		sum+=maxi;
		if(u[a[i]]!=0)c[b[a[i]]]=1002;
	}
	cout<<sumn<<" "<<sum<<endl;
	for(int i=0;i<k;i++){
		if(u[a[i]]!=0)cout<<b[a[i]]+1<<" "<<a[i]+1<<endl;
	}
}
