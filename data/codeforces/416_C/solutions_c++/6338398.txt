#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
struct node {int x,y,z;} t[1005],a[1005];
bool v[1005];
int ANS,ans,i,p[1005],n,m,j;
int CMP(node i,node j) {return i.x<j.x;}
int cmp(node i,node j) {return i.y>j.y;}
int main()
{
	scanf("%d",&n);
	for (i=1; i<=n; i++)
	{
		scanf("%d%d",&t[i].x,&t[i].y);
		t[i].z=i;
	}
	scanf("%d",&m);
	for (i=1; i<=m; i++) scanf("%d",&a[i].x),v[i]=false;
	for (i=1; i<=m; i++) a[i].y=i;
	sort(a+1,a+m+1,CMP);
	sort(t+1,t+n+1,cmp);
	for (i=1; i<=n; i++)
	{
		for(j=1; j<=m; j++)
		  if (a[j].x>=t[i].x && !v[j])
		  {
				p[t[i].z]=a[j].y;
				v[j]=true;
				ans+=t[i].y;
				ANS++;
				break;
	      }
	}
	cout<<ANS<<' '<<ans<<endl;
	for (i=1; i<=n; i++)
	  if (p[i]) cout<<i<<' '<<p[i]<<endl;
	return 0;
}
