#include<iostream>
#include<cstdio>
#include<cstring>
#include<set>
#include<algorithm>
using namespace std;

typedef pair<int,pair<int,int> > people;
typedef pair<int,int> table;

set<table> c;

people a[10000];

table ans1[10000];

int main(){
	int n,m;
	scanf("%d",&n);
	for (int i=1; i<=n; ++i){
		scanf("%d%d",&a[i].second.first,&a[i].first);
		a[i].second.second=i;
	}
	sort(a+1,a+1+n);
	int ans=0,M=0;
	scanf("%d",&m);
	for (int i=1; i<=m; ++i){
		int x;
		scanf("%d",&x);
		c.insert(make_pair(x,i));
	}
	for (int i=n; i; --i){
		set<table>::iterator IT;
		IT=c.lower_bound(make_pair(a[i].second.first,0));
		if (IT==c.end()) continue;
		ans+=a[i].first;
		ans1[++M]=make_pair(a[i].second.second,IT->second);
		c.erase(IT);
	}
	sort(ans1+1,ans1+1+M);
	printf("%d %d\n",M,ans);
	for (int i=1; i<=M; ++i) 
		printf("%d %d\n",ans1[i].first,ans1[i].second);
	return 0;
}
