#include <algorithm>
#include <cstdio>
#include <iostream>
using namespace std;
int c[1001],v[1001];
int loc[1001][2];
int ans[1001];
struct node
{	
	int x;
	int lca;

}a[1001];
bool cmp(node x,node y)
{
	return x.x < y.x;
}
int main(void)
{
	//freopen("in.txt","r",stdin);
	int n;cin >> n;
	for(int i =1;i<=n;i++)
		scanf("%d%d",&c[i],&v[i]);
	int m;cin>>m;
	for(int i = 1;i<=m;i++)
		scanf("%d",&a[i].x),a[i].lca = i;
	sort(a+1,a+1+m,cmp);
	int sum = 0;
	int p = 0;
	for(int i = 1;i<=m;i++)
	{
		int val=0,flag = 0,cnt = 0;
		for(int j =1;j <= n;j++)
		{
			if(c[j]<=a[i].x && !ans[j] && v[j] > val)
			{
				flag = 1,
				val = v[j];
				cnt = j;
			}
		}
		if(flag)
		{
			ans[cnt]=1;
			p++;
			sum += val;
			loc[p][0]=cnt;
			loc[p][1]=a[i].lca;
		}
	}
	printf("%d %d\n",p,sum);
	for(int i = 1;i<=p;i++)
	{
		printf("%d %d\n",loc[i][0],loc[i][1]);
	}
	return 0;
}
