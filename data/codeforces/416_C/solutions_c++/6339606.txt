#include <iostream>#include <set>#include <algorithm>#include <vector>using namespace std;typedef pair<int, int> pie;#define L first#define R second#define MP make_pair#define PB push_backvector<pie> v;set<pie> s;pair<pie, int> a[100000];int main(){	int n;	cin>>n;	for(int i=1;i<=n;i++){		cin>>a[i].L.R>>a[i].L.L;		a[i].R = i;	}	sort(a+1, a+n+1);	int k;	cin>>k;	for(int i=1;i<=k;i++){		int cur;		cin>>cur;		s.insert(MP(cur, i));	}	int ans = 0, ted = 0;	for(int i=n;i>=1;i--){		if(s.lower_bound(MP(a[i].L.R, 0)) == s.end())			continue;		ted++;		ans += a[i].L.L;		v.PB(MP(a[i].R, (*s.lower_bound(MP(a[i].L.R, 0))).R));		s.erase(s.lower_bound(MP(a[i].L.R, 0)));	}	cout<<ted<<" "<<ans<<endl;	for(int i=0;i<v.size();i++)		cout<<v[i].L<<" "<<v[i].R<<endl;}
