#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;
int n,k;
struct table{
	int ord,r;
	bool operator < (const table b) const{
		return r<b.r;
	}
} b[1005];
bool flag[1005];
struct data{
	int ord,c,p;
	bool operator < (const data b) const{
		return p>b.p||p==b.p&&c<b.c;
	}
} a[1005];
int m,s;
struct ans{
	int a,b;
} seq[1005];

int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d%d",&a[i].c,&a[i].p);
		a[i].ord=i;
	}
	sort(a,a+n);
	scanf("%d",&k);
	for(int i=0;i<k;i++){
		scanf("%d",&b[i].r);
		b[i].ord=i;
	}
	sort(b,b+k);
	for(int i=0;i<n;i++){
		for(int j=0;j<k;j++) if(!flag[j]&&b[j].r>=a[i].c){
			flag[j]=true;
			s+=a[i].p;
			seq[m++]=(ans){a[i].ord,b[j].ord};
			break;
		}
	}
	printf("%d %d\n",m,s);
	for(int i=0;i<m;i++)
		printf("%d %d\n",seq[i].a+1,seq[i].b+1);
	scanf("\n");
}
