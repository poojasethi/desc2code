#define Down(i,po,ki) for(int i=po;i>=ki;i--)

pair <int,int> z[1001];

int main()
{ cin>>n;
  for (int i=1;i<=n;i++) {cin>>z[i].S>>z[i].F; z[i].F=-z[i].F; z[i].S+=i*10000;}

  cin>>k;
  for (int i=1;i<=k;i++) {cin>>t; r.insert(make_pair(t,i));}
  for (int i=1;i<=n;i++) {it=r.upper_bound(make_pair(z[i].S%10000,0));
   if (it!=r.end()) {kil++;sum-=z[i].F; ans[z[i].S/10000]=it->second;r.erase(it);}}
  cout<<kil<<" "<<sum<<endl;
  for (int i=1;i<=n;i++) if (ans[i]>0) cout<<i<<" "<<ans[i]<<endl;
 }
