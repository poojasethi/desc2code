#include<bits/stdc++.h>
using namespace std;
typedef struct{
    int x,y,z;
}ada;
ada a[1001],f[1001];
bool used[1001];
int n,m,dp[1001];
int ans=0,cnt=0;
bool cmp(ada a,ada b) { return a.y>b.y; }
bool CMP(ada a,ada b){ return a.x<b.x; }
int main(){
  //  freopen("chaythu.cpp","r",stdin);
    cin>>n;
    for(int i=1;i<=n;++i){
        cin>>a[i].x>>a[i].y;   // x la can nang, y la gia tri
        a[i].z=i;
    }
    cin>>m;
    for(int i=1;i<=m;++i) cin>>f[i].x, f[i].y=i, used[i]=0;  // x la suc chua
    sort(a+1,a+n+1,cmp);
    sort(f+1,f+m+1,CMP);
    for(int i=1;i<=n;++i)
        for(int j=1;j<=m;++j)
    if(f[j].x>=a[i].x && not used[j] ){
        used[j]=1;
        ans+=a[i].y;
        cnt++;
        dp[a[i].z]=f[j].y;
        break;
    }
    cout<<cnt<<" "<<ans<<endl;
    for(int i=1;i<=n;++i)  if(dp[i])  cout<<i<<" "<<dp[i]<<endl;
}
