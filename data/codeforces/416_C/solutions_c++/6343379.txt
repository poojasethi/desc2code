#include <iostream>
#include <vector>
#include <algorithm>
#define N 1000
struct D
{
	int c,p,z;
	bool m;
	D ( ){
		c = 0, p = 0, z = 0, m = 0;
	}
};
bool operator< (D q,D w)
{
	if (q.p<w.p)
		return true;
	if (q.p==w.p)
		if (q.c<w.c)
			return true;
	return false;
}
using namespace std;
int main()
{
	int n,h,d=0,t=0;;
	D a[1001],r[1001];
	cin>>n;
	for (int i=0;i<n;i++)
	{
		cin>>a[i].c>>a[i].p;
		a[i].z=i;
	}
	sort(&a[0],&a[n]);
	cin>>h;
	for (int i=0;i<h;i++)
	{
		cin>>r[i].c;
		r[i].z=i;
	}
	sort(&r[0],&r[h]);
		for (int j=n-1;j>=0;j--)
	for (int i=0;i<h;i++)
		{
			//cout<<endl<<r[i].c<<" "<<a[j].c<<" ";
			if (a[j].c<=r[i].c && !a[j].m && !r[i].m)
			{
				//cout<<".";
				t++;
				d+=a[j].p;
				a[j].m=true;
				r[i].m=true;
				r[i].p=a[j].z;
			}
		}
	cout<<t<<" "<<d<<endl;
	for (int i=0;i<h;i++)
		if (r[i].m)
			cout<<r[i].p+1<<" "<<r[i].z+1<<endl;
}
