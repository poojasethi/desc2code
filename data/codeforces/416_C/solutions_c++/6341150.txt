#include <iostream>#include <set>#include <algorithm>#include <vector>#define L first#define R second#define Joft make_pair#define Shuf push_backusing namespace std;typedef pair<int, int> PII;vector<PII> Anine;set<PII> s;pair<PII, int> arr[100000];int main(){	int n;	cin>>n;	for(int i=1;i<=n;i++)	{		cin>>arr[i].L.R>>arr[i].L.L;		arr[i].R = i;	}	sort(arr+1, arr+n+1);	int k;	cin>>k;	for(int i=1;i<=k;i++)	{		int Input;		cin>>Input;		s.insert(Joft(Input, i));	}	int Answer = 0, Result = 0;	for(int i=n;i>=1;i--)	{		if(s.lower_bound(Joft(arr[i].L.R, 0)) == s.end())			continue;		Result++;		Answer += arr[i].L.L;		Anine.Shuf(Joft(arr[i].R, (*s.lower_bound(Joft(arr[i].L.R, 0))).R));		s.erase(s.lower_bound(Joft(arr[i].L.R, 0)));	}	cout<<Result<<" "<<Answer<<endl;	for(int i=0;i<Anine.size();i++)		cout<<Anine[i].L<<" "<<Anine[i].R<<endl;}
