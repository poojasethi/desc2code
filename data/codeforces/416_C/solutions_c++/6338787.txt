#include <cstdio>
#include <cstdlib>
#include <algorithm>
int N,K;
struct group{int c,p;}G[1000+7];
int res[1000+7];
int used[1000+7],id[1000+7];
bool compareid(const int&a,const int&b){return res[a]<res[b];}
int main()
{
	scanf("%d", &N);
	for(int i=1;i<=N;i++)scanf("%d %d",&G[i].c,&G[i].p);
	scanf("%d",&K);
	for(int i=1;i<=K;i++)scanf("%d",res+i),id[i]=i;
	std::sort(id+1,id+K+1,compareid);
	int ans=0,accept=0;
	for(int i=1;i<=K;i++)
	{
		int r=res[id[i]];
		int p=-1, ch=0;
		for (int j=1; j<=N;j++)
		{
			if (!used[j]&&G[j].c<=r)
			{
				if (p==-1||p<G[j].p)
				{
					ch=j;
					p=G[j].p;
				} 
			}
		}
		if (p!=-1)
		{accept++;
			used[ch]=id[i];
			ans+=p;
		}
	}
	printf("%d %d\n",accept,ans);
	for(int i=1;i<=N;i++)
		if (used[i]) printf("%d %d\n", i, used[i]);
	return 0;
}
