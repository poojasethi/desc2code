#include <iostream>
#include <stdio.h>
using namespace std;
int main()
{//freopen("i.txt","r",stdin);
int n; cin>>n;
int c[1001]={},p[1001]={};
for (int i=1;i<=n;i++) cin>>c[i]>>p[i];
int k; cin>>k;
int t[1001]; t[0]=10000000;
for (int i=1;i<=k;i++) cin>>t[i];
int count=0, s=0;
int a[1001]={}; bool f[1001]={};
for (int i=1;i<=n;i++)
{int r=0;
 for (int j=1;j<=n;j++)
   if (a[j]==0&&p[j]>p[r]) r=j;
 if (r==0) break;
 int st=0;
 for (int j=1;j<=k;j++)
  if (f[j]==0&& t[j]>=c[r]&&t[j]<t[st]) st=j;
 if (st!=0)
  {a[r]=st; f[st]=1; count++; s+=p[r];}
  else a[r]=-1;
}
cout<<count<<" "<<s<<endl;
for (int i=1;i<=n;i++)
if (a[i]>0) cout<<i<<" "<<a[i]<<endl;
return 0;
}
