#include<stdio.h>
#include<iostream>
#include<algorithm>
#include<stdlib.h>
#define test
#include<string.h>
#define fr(i,m,n) for(int i=m; i<=n; i++)
#define dfr(i,m,n) for(int i=m; i>=n;i--)
using namespace std;

typedef long long ll;
const int N=1005;

int a[N], b[N], c[N], p[N], t[N];

int main()
{
	int n,m; scanf("%d", &n); t[0]=10000;
	fr(i,1,n) scanf("%d%d", c+i, p+i);
	scanf("%d",&m); fr(i,1,m) scanf("%d", t+i); int cnt=0, sum=0;
	fr(i,1,n)
	{
		int r=0;
		fr(j,1,n) if (p[r]<p[j] && !a[j]) r=j;
		if (r==0) continue;
		int s=0;
		fr(j,1,m) if (t[s]>t[j] && c[r]<=t[j] && !b[j]) s=j;
		if (!s) a[r]=10000;
		else b[s]=1, a[r]=s, sum+=p[r], cnt++;
	}
	printf("%d %d\n", cnt, sum);
	fr(i,1,n)
	{
		if (a[i]>0 && a[i]<10000) printf("%d %d\n", i, a[i]);
	}
	return 0;
}
