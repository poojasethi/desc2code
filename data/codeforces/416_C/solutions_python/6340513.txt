#!/usr/bin/env python

import sys
import logging
import traceback

n = input()
visitors = zip(range(1, n+1), 
        [map(int, raw_input().split()) for _ in range(n)])

flag = [False] * (n+1)

k = input()

tables = zip(range(1, k+1), map(int, raw_input().split()))

visitors.sort(key=lambda v: v[1][0])
tables.sort(key=lambda t: t[1])

V = 0
R = 0
A = []

for t in tables: 
    tid = t[0]
    cap = t[1]
    m_value = 0
    assign = (0, 0)
    for v in visitors: 
        vid = v[0]
        num = v[1][0]
        value = v[1][1]
        if flag[vid] == True: 
            continue
        if cap < num: 
            break
        if value > m_value: 
            m_value = value
            assign = (vid, tid)

    if m_value > 0: 
        flag[assign[0]] = True
        V += m_value
        R += 1
        A.append(assign)

print R, V
for a in A: 
    print ' '.join(map(str, a))
