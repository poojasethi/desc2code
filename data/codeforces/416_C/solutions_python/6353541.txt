nreq = int(raw_input())
reqs = []
taken = []
for x in range(nreq):
  reqs.append([int(x) for x in raw_input().strip().split()])
  taken.append(False)

ntable = int(raw_input())
tables = [[int(x),idx+1]  for idx, x in enumerate(raw_input().strip().split())]
tables.sort()

solution = []
total = 0

for cap,idxTable in tables:
  bestval = 0
  bestidx = 0
  bestReq = None
  for x in range(len(reqs)):
    nperson, nmoney = reqs[x]
    if not taken[x] and cap >= nperson and bestval < nmoney:
      bestReq = [nperson, nmoney]
      bestval = nmoney
      bestidx = x

  if bestval > 0: 
    solution.append([bestidx+1, idxTable])
    taken[bestidx] = True
    total += bestval

print len(solution), total
for x, y in solution:
  print x, y
