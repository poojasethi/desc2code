import operator
import math

def search(tables, size):
    start = 0
    end = len(tables) - 1
    while start < end:
        mid = int(math.ceil((start + end) / 2.0))
        if tables[mid][1] == size:
            return mid
        if tables[mid][1] > size > tables[mid-1][1]:
            return mid
        if tables[mid][1] > size:
            end = mid - 1
        else:
            start = mid
    if start == end and tables[start][1] >= size:
        return start
    return -1

n = int(raw_input())
groups = []
for i in xrange(n):
    groups.append([i + 1] + map(int, raw_input().split()))

k = int(raw_input())
tables = zip(range(1, k+1), map(int, raw_input().split()))

groups.sort(key=operator.itemgetter(1))
groups.sort(key=operator.itemgetter(2), reverse= True)
tables.sort(key=operator.itemgetter(1))

occupied, sales = [], 0
for group in groups:
    ID, size, sale = group
    index = search(tables, size)
    if index == -1:
        continue
    sales += sale
    table = tables.pop(index)
    occupied.append(str(ID) + " " + str(table[0]))

print len(occupied), sales
print "\n".join(occupied)
