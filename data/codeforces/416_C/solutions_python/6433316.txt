nreq = int(raw_input())
reqs = []
for i in range(nreq):
  nvisitor, nmoney = [int(x) for x in raw_input().strip().split()]
  reqs.append([nvisitor, nmoney, True])

ntable = int(raw_input())
tables = [[int(x),idx] for idx, x in enumerate(raw_input().strip().split())]
tables.sort()

acceptedReq, acceptedMoney, acceptedDetails = 0, 0, []

for capacity, tableIdx in tables:
  bestx = -1
  bestval = 0
  for i in range(len(reqs)):
    nvisitor, nmoney, haventAccepted = reqs[i]
    if haventAccepted and bestval < nmoney and capacity >= nvisitor:
      bestx = i
      bestval = nmoney

  if bestx > -1:
    acceptedReq += 1
    acceptedMoney += bestval
    acceptedDetails.append([bestx+1, tableIdx+1])
    reqs[bestx][2] = False

print acceptedReq, acceptedMoney
for detail in acceptedDetails:
  print " ".join(map(str,detail))
