n = int(raw_input())
r = enumerate([map(int, raw_input().split()) for _ in range(n)], 1)
k, t = int(raw_input()), enumerate(map(int, raw_input().split()), 1)
r = sorted(r, key=lambda (z, (x, y)): y, reverse=True)
t = sorted(t, key=lambda (x, y): y)
f, m, o = 0, 0, list()
for i, (c, p) in r:
	for j, s in t:
		if c <= s:
			t.remove((j, s))
			f, m, k = f + 1, m + p, k - 1
			o.append((i, j))
			break
print f, m
for a, b in o:
	print a, b
