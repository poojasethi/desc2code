#! python2

from operator import itemgetter

n = int(raw_input())

edsgers = [[(_ + 1)] + map(int, raw_input().split()) for _ in xrange(n)]
edsgers.sort(key=itemgetter(1))
edsgers.sort(key=itemgetter(2), reverse=True)

k = int(raw_input())

r = sorted(zip(xrange(1, k + 1), map(int, raw_input().split())), key=itemgetter(1))

matched = [0] * (k + 1)


s = 0
m = 0
for i, c, p in edsgers:
    for t, rk in r:
        if not matched[t] and c <= rk:
            matched[t] = i
            m += 1
            s += p
            break
            
print m, s
for t in xrange(1, k + 1):
    if matched[t]:
        print matched[t], t
