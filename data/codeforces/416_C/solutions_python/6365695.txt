input = raw_input
n = int(input())
PeopleList = []
for i in range(n):
    c, p = map(int, input().split())
    PeopleList.append((i+1, c, p))
k = int(input())
r = map(int, input().split())
TableList = []
for i in range(k):
    TableList.append((i+1, r[i]))
TableReserv = [0 for j in range(k)]

TableList = sorted(TableList, key=lambda place: place[1])
PeopleList = sorted(PeopleList, key=lambda money: money[2], reverse=1)

for people in PeopleList:
    for table in TableList:
        if not TableReserv[table[0]-1] and table[1] >= people[1]:
            TableReserv[table[0]-1] = people[0]
            break

sum = 0
count = 0
for table in TableReserv:
    if table:
        for people in PeopleList:
            if people[0]==table:
                sum += people[2]
                count += 1

print(format('%d %d' % (count, sum)))

for i in range(k):
    if TableReserv[i]:
        print(format('%d %d' % (TableReserv[i], i+1)))
