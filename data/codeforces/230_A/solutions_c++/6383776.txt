#include <bits/stdc++.h>
using namespace std;

int main() {
	int s, n;
	pair<int,int> a[1005];
	cin >> s >> n;
	for (int i = 0; i < n; ++i)
		cin >> a[i].first >> a[i].second;
	sort(a, a + n);
	for (int i = 0; i < n; ++i) {
		if (s <= a[i].first) return puts("NO"), 0;
		s += a[i].second;
	}
	puts("YES");
}
