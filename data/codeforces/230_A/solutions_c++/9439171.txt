#include<bits/stdc++.h>
using namespace std;
pair<int,int>x[100000];
int main()
{
	int s,n;
	cin>>s>>n;
	for(int i=0;i<n;++i)
	{
		cin>>x[i].first>>x[i].second;
	}
	sort(x,x+n);
	for(int i=0;i<n;++i)
	{
		if(x[i].first<s)
		{
			s+=x[i].second;
		}
		else 
		{
			cout<<"NO";
		return 0;
		}
	}
	cout<<"YES"<<endl;
	return 0;
}
