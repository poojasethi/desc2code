#include <algorithm>
#include <iostream>

using namespace std;

int s,n,x,y;
pair <int,int> p[1001];

int main()
{   cin>>s>>n; for(int i=1;i<=n;i++) {cin>>x>>y; p[i]=make_pair(x,y);}
    sort(p+1,p+n+1);
    for(int i=1;i<=n;i++) {if (s>p[i].first) s+=p[i].second; else s=-1;}
    if (s>=0) cout<<"YES\n"; else cout<<"NO\n";
}
