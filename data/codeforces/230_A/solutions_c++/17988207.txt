#include<algorithm>
#include<iostream>
using namespace std;
int n,s;
pair<int,int> p[12345];
int main()
{
	cin>>s>>n;
	for(int i=0;i<n;i++)
	{
		cin>>p[i].first>>p[i].second;
	}
	sort(p,p+n);
	int i;
	for(i=0;i<n;i++)
	{
		if(s>p[i].first) s+=p[i].second;
		else break;
	}
	if(i==n) cout<<"YES";
	else cout<<"NO";
}
