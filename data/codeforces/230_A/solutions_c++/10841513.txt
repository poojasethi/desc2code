#include<iostream>
using namespace std;
#include<algorithm>
#include<utility>
const int maxt=1000;
pair<int, int> a[maxt];
int main()
{
long k,n,i;
cin>>k>>n;
for(i=0;i<n;i++)
cin>>a[i].first>>a[i].second;
sort(a,a+n);
for(i=0;i<n;i++)
{
if(k>a[i].first)
k=k+a[i].second;
else
break;
}
if(i==n)
cout<<"YES";
else
cout<<"NO";
return 0;
}
