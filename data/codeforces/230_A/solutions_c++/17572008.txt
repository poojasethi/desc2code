#include<iostream>
using namespace std;
int main(){
int s,n;
cin>>s>>n;
int a[n],b[n];
for(int i=0;i<n;i++)cin>>a[i]>>b[i];
for(int i=0;i<n;i++){for(int j=i;j<n;j++)if(a[i]>a[j]){swap(a[i],a[j]);swap(b[i],b[j]);}}
for(int i=0;i<n;i++){if(a[i]<s){s=s+b[i];}
else{cout<<"NO"<<endl;return 0;}}
cout<<"YES"<<endl;;}
