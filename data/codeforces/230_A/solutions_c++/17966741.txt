#include <iostream>
#include <algorithm>
using namespace std;
int a,b,c,d,e;
pair <int,int> k[1005];
main()
{
	 cin>>a>>b;
	 for(c=0;c<b;c++)
	 {
	 	cin>>k[c].first>>k[c].second;
	 }
	sort(k,k+b);
	for(e=0;e<b;e++)
	{
		if(a>k[e].first) {a=a+k[e].second;}
		else {cout<<"NO"; return 0;}
	}
	cout<<"YES";
}
