#include<bits/stdc++.h>
using namespace std;
int main()
{
	int s, n, i;
	cin>>s>>n;
	pair<int, int> a[n];
	for( i=0 ; i<n ; i++)
		cin>>a[i].first>>a[i].second;
	sort(a,a+n);
	for( i=0 ; i<n ; i++)
	{
		if(a[i].first < s)
			s += a[i].second;
		else
			break;
	}
	if(i == n)
		cout<<"YES";
	else
		cout<<"NO";
}
