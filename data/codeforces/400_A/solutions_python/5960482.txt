n=int(raw_input())

for i in xrange(n):
	x=raw_input()
	ans=""
	cnt=0
	for j in xrange(13):
		if j==0:continue
		if 12%j==0:
			k=12/j
			flag=0
			for g in xrange(k):
				ff=0
				for h in xrange(j):
					if(x[h*k+g]=='O'):
						ff=1;break
				if(ff==0):flag=1;break
			if(flag==1):
				ans+=" "+str(j)+"x"+str(k)
				cnt+=1
	print str(cnt)+ans
