n = input()
for x in xrange(n):
    line = raw_input()
    lst = []
    for y in xrange(1,13):
        if 12%y != 0:
            continue
        for z in xrange(0,12/y):
            chk = True
            for zz in xrange(z,12,12/y):
                if line[zz] == 'O':
                    chk = False
                    break
            if chk:
                lst.append(y)
                break
    print len(lst),
    for y in lst:
        print str(y)+'x'+str(12/y),
    print ""
