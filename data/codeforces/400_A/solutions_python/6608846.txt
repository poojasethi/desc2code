d = [1, 2, 3, 4, 6, 12]
t = int(raw_input())
while t > 0:
	s = raw_input()
	res = []
	for i in d:
		n = 12 / i
		temp = [0] * n
		for j in range (0, i):
			x = s[j * n: j * n + n]
			#print(i, x)
			for k in range (0, n):
				if x[k] == "X":
					temp[k] += 1
		#print(i, n, temp)
		if max(temp) == i:
			res.append(i)
	ss = ""
	for i in res:
		ss += " " + str(i) + "x" + str(12 / i)
	print(str(len(res))  + ss)
	t -= 1
