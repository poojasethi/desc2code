n = input()

for _ in xrange(n):
    xs = map(lambda x: True if x == 'X' else False, raw_input())
    assert 12 == len(xs)

    ds = []
    for d in [1, 2, 3, 4, 6, 12]:
        step = 12 / d
        if any(map(lambda offset: all(xs[offset::step]), range(step))):
            ds.append(d)

    print len(ds), ' '.join(map(lambda d:'{0}x{1}'.format(d, 12 / d), ds))
