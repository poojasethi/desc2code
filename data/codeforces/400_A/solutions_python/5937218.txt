t = input()
while t > 0:
	s = raw_input()
	a = []
	for row in [1, 2, 3, 4, 6, 12]:
		col = 12 / row
		for j in range(col):
			flag = True
			for i in range(row):
				if s[i * col + j] == 'O': flag = False
			if flag:
				a.append(str(row) + 'x' + str(col))
				break

	print len(a) ,
	print ' '.join(x for x in a)
	t -= 1
