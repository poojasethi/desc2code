T = int(raw_input())

P = [12, 6, 4, 3, 2, 1]

for t in range(T):
  line = raw_input()
  ans = []
  for p in P:
    f = False
    for i in range(p):
      fb = True
      for j in range(12/p):
        if line[i+j*p] == 'O':
          fb = False
          break
      if fb:
        f = True
        break
    if f:
      ans.append(p)
  n = len(ans)
  if n == 0:
    print 0
  else:
    print n, ' '.join(map(lambda x: str(12/x) + 'x' + str(x), ans))
