#!/usr/bin/python

cases = int(raw_input())
bs = [1,2,3,4,6,12][::-1]
for case in range(cases):
    table = raw_input()
    rlt = []
    for b in bs:
        a = 12/b
        for i in range(b):
            flag = True
            for j in range(i,12,b):
                if table[j] != 'X':
                    flag = False
            if flag:
                rlt.append('%dx%d'%(a,b))
                break
    rlt.insert(0,str(len(rlt)))
    print ' '.join(rlt)
