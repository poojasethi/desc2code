a = [1, 2, 3, 4, 6, 12]
pre = ["1x12", "2x6", "3x4", "4x3", "6x2", "12x1"]
for _ in xrange(input()):
	string = raw_input()
	result = []
	for i in xrange(len(a)):
		for j in xrange(12/a[i]):
			if string[j::12/a[i]] == 'X'*a[i]:
				result.append(pre[i])
				break
	print len(result), ' '.join(result)
