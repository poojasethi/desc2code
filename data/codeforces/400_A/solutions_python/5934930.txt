def AllX(n, col):
  for x in range(n):
    if cards[x*12/n + col] != 'X': return False
  return True

nt = int(raw_input())
for t in range(nt):
  cards = raw_input()
  ans = []

  for x in range(1,13):
    if 12%x != 0: continue;
    for y in range(12/x):
      if AllX(x, y):
        ans.append(str(x) + 'x' + str(12/x))
        break

  print len(ans), " ".join(ans)
