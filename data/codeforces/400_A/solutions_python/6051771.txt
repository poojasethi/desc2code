t = int(raw_input())
for _ in xrange(t):
    X = raw_input()
    ans = []
    for i in xrange(1, 13):
        if 12 % i != 0: continue
        j = 12 / i
        w = False
        for k in xrange(j):
            a = True
            for l in xrange(i):
                if X[l*j+k] != 'X': a = False
            if a: w = True
        if w: ans.append("%dx%d" % (i, j))
    if len(ans) == 0: print 0
    else: print "%d %s" % (len(ans), ' '.join(ans))
