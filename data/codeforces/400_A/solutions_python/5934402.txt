#!/usr/bin/env python2


def main():
    case_num = int(raw_input())
    for case in xrange(case_num):
        line = raw_input().strip()
        solutions = []
        for a in (12, 6, 4, 3, 2, 1):
            if any(all(line[i] == 'X' for i in xrange(j, 12, a))
                   for j in xrange(0, a)):
                solutions.append(12 / a)
        if not solutions:
            print 0
        else:
            print len(solutions), ' '.join('%sx%s' % (a, 12 / a) for a in solutions)
                


if __name__ == '__main__':
    main()
