def f(a, b, s):
	for i in xrange(b):
		col = True
		for j in xrange(a):
			if s[i + b*j] != 'X':
				col = False
				break
		if col: return True
 	return False

t = int(raw_input())

for qq in xrange(t):
	s = raw_input()
	works = []
	for a in [1, 2, 3, 4, 6, 12]:
		b = 12 / a
		if f(a, b, s): works.append("{}x{}".format(a, b))
	
	output = str(len(works))
	for i in xrange(len(works)):
		output = output + " " + works[i]
	print output
