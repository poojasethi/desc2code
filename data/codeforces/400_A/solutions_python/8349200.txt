def solve(s):
  res = []
  for b in [12,6,4,3,2,1]:
    a = 12/b
    f = False
    for i in range(b):
      if s[i::b] == 'X'*a:
        f = True
        break
    if f:
      res.append('%dx%d' % (a, b))
  return res

t = int(raw_input())
for i in range(t):
  res = solve(raw_input())
  print len(res), ' '.join(res)
