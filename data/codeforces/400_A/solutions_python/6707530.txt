import sys
lines = sys.stdin.readlines()
n = int(lines[0])
opts = {1:12, 2:6, 3:4, 4:3, 6:2, 12:1}
for line in lines[1:]:
  sols = []
  for a,b in opts.items():
    for x in range(len(line)/a):
      es = True
      for i in range(a):
       if line[x+ i*b] != 'X':
          es = False
      if es:
        sols.append('%dx%d' % (a,b))
        break
  print len(sols), ' '.join(sols)
