#!/usr/bin/env python

import sys
import logging
import traceback

logging.basicConfig(level = logging.INFO)

T = input()
d = [1, 2, 3, 4, 6, 12]

for _ in range(T) : 
	s = raw_input()
	ret = []
	for a in d : 
		b = 12/a
		for i in range(b) : 
			seg = [s[i+j*b] for j in range(a)]
			if all(map(lambda x: x == 'X', seg)) : 
				ret.append(a)
				break

	rtnStr = []
	rtnStr.append(str(len(ret)))
	for a in ret : 
		rtnStr.append('%dx%d' % (a, 12/a))

	print ' '.join(rtnStr)
