for i in xrange(input()):
	s = raw_input().strip()
	ans = []
	if 'X' in s: ans+=['1x12']
	for j in xrange(6):
		if s[j]==s[j+6]=='X': ans+=['2x6']; break
	for j in xrange(4):
		if s[j]==s[j+4]==s[j+8]=='X': ans+=['3x4']; break
	for j in xrange(3):
		if s[j]==s[j+3]==s[j+6]==s[j+9]=='X': ans+=['4x3']; break
	for j in xrange(2):
		if s[j]==s[j+2]==s[j+4]==s[j+6]==s[j+8]==s[j+10]=='X': ans+=['6x2']; break
	if s=='X'*12: ans+=['12x1']
	print len(ans),
	for x in ans:
		print x,
	print '\n',
