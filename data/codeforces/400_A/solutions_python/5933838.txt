n = input()

for _ in range(n):
  s = raw_input()

  ans = []
  for a in range(1, 13):
    if 12 % a: continue

    b = 12 / a
    f = [True for i in range(b)]

    for i in range(12):
      if s[i] == "O":
        f[i % b] = False

    if any(f):
      ans.append((a, b))

  print len(ans),
  print " ".join(map(lambda x: "%dx%d" % (x[0], x[1]), ans))
