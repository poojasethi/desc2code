nt = int (input ())
n = 12
for t in range (nt):
	s = raw_input ()
	ans = []
	for a in range (1, n + 1):
		b = n // a
		if a * b == n:
			ok = False
			for j in range (b):
				ok |= (s[j::b].find ('O') == -1)
			if ok:
				ans += [(a, b)]
	print (' '.join ([str (len (ans))] + \
		['{0}x{1}'.format (*x) for x in ans]))
