n = input()
def solve():
    s = raw_input()
    ans = []
    for i in [1, 2, 3, 4, 6, 12]:
        j = 12 / i
        for k in xrange(j):
            #print i, j, k
            if all(s[k+j*z] =='X' for z in xrange(i)):
                ans.append(str(i)+'x'+str(j))
                break
    return ans

for _ in xrange(n):
    ans = solve()
