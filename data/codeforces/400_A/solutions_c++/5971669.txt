#include<cstdio>
int main(){
	int T,i,j,k,a[120]={0},sum=0,t[60]={12,6,4,3,2,1},ans[60]={0};
	char c[120];
	scanf("%d\n",&T);
	while(T--){
		sum=0;
		gets(c);
		for(i=0;i<12;i++){
			if(c[i]=='O') a[i]=0;
			else if(c[i]=='X') a[i]=1;
		}
		for(k=0;k<6;k++)
			for(j=0;j<t[k];j++){
				int temp=0;
				for(i=j;i<12;i+=t[k])
					temp+=a[i];
				if(temp==(12/t[k])){
					ans[k]=1;
					sum++;
					break;
				}
			}
		printf("%d ",sum);
		for(i=0;i<6;i++) if(ans[i]==1) printf("%dx%d ",(12/t[i]),t[i]);
		printf("\n");
		for(i=0;i<6;i++) ans[i]=0;
		
	}
	return 0;
}
