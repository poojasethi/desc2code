#include<stdio.h>

int main(void){
	int n,i,j,ans=0,u,b[13],judge;
	char a[13];
	scanf("%d",&n);
	getchar();
	while (n--){
		scanf("%s",a);
		getchar();
		for (u=1;u<=12;u++)
		if (12 % u==0){
			for (j=0;j<12/u;j++){
				judge=1;
				for (i=0;i<u;i++){
					if (a[j+(12/u)*i]=='O') {
						judge=0;
						break;
					}
				}
				if (judge) {
					ans++;
					b[ans]=u;
					break;
				}
			}
		}
		printf("%d",ans);
		for (i=1;i<=ans;i++) printf(" %dx%d",b[i],12/b[i]);
		ans=0;
		printf("\n");
	}
	return 0;
}
