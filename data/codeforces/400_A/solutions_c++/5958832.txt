//CODEFORCES
#include<bits/stdc++.h>
using namespace std;
void solve()
{
	vector< pair<int,int> > v;
	int i,bas,now;
	char arr[12];
	for (i=0;i<12;i++)
		cin>>arr[i];
	for (i=1;i<=12;i++)
	{
		if (12%i)
			continue;
		for (bas=0;bas<12/i;bas++)
		{
			for (now=bas;now<12;now+=12/i)
				if (arr[now]!='X')
					break;
			if (now>=12)
			{
				v.push_back(make_pair(i,12/i));
				break;
			}
		}
	}
	cout<<v.size()<<' ';
	for (i=0;i<v.size();i++)
		cout<<v[i].first<<'x'<<v[i].second<<' ';
	cout<<endl;
}
int main()
{
	int n;
	cin>>n;
	while (n--)
		solve();
	return 0;
}
