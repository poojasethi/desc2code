#include <iostream>
#include <string>

using namespace std;

int main(){
  int t;
  cin >> t;
  for(int i=0;i<t;i++){
    int nrez=0;
    int rez[12];
    string text;
    cin >> text;
    for(int j=1;j<=12;j++){
      if(12%j!=0){continue;}
      bool ok=0;
      for(int k=0;k<j;k++){
	bool kul=1;
	for(int m=k;m<12;m+=j){
	  if(text.at(m)!='X'){kul=0;}
	}
	if(kul){ok=1;break;}
      }
      if(ok){rez[nrez]=j;nrez++;}
    }
    cout << nrez;
    for(int a=nrez-1;a>=0;a--){cout << ' ' << 12/rez[a] << 'x' << rez[a];}
    cout << '\n';
  }
  return 0;
}
