#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
int main(){
	int i,j,k,n,a[7],b[7]={0,12,6,4,3,2,1};
	char s[20];
	scanf("%d",&n);
	while(n--){
		scanf("%s",s);
		memset(a,0,sizeof(a));
		for(i=1;i<=6;i++){
			for(j=0;j<b[i];j++){
				for(k=j;k<12;k+=b[i]){
					if(s[k]=='O') break;
				}
				if(k>=12) a[i]=1;
			}
		}
		for(j=0,i=1;i<=6;i++)
			j+=a[i];
		printf("%d",j);
		for(i=1;i<7;i++)
			if(a[i]) printf(" %dx%d",12/b[i],b[i]);
		printf("\n");
	}
	return 0;
}
