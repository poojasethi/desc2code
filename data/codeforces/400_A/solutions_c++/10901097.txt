#include<cstdio>
#include<cstring>
using namespace std;
char s[13];int ans[15];
int main(){
	int n,k=0,f;
	scanf("%d",&n);
	while(n--){
		scanf("%s",s);k=0;memset(ans,0,sizeof(ans));
		if(strchr(s,'X')!=NULL){ans[1]=1;k++;}
		for(int i=2;i<=12;i++){
			if(12%i==0){
				for(int j=0;j<12/i;j++){
					f=0;
					for(int m=j;m<12;m=m+12/i){
						if(s[m]=='O'){f=1;break;}}
					if(f==0){ans[i]=1;k++;break;}}
			}
		}
		printf("%d",k);
		for(int i=1;i<=12;i++){
			if(ans[i]==1)printf(" %dx%d",i,12/i);
		}printf("\n");
	}
	return 0;
}
