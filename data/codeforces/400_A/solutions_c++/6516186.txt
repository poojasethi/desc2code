#include<cstring>

scanf("%d\n",&N);
while(N--){
gets(s);
int cnt=0,ok[13];
for(int i=1;i<=12;i++) ok[i]=0;
for(int i=1;i<=12;i++){
if(12%i==0){
for(int j=0;j<(12/i);j++){
int kk=1;
for(int k=j;k<12;k+=12/i){
if(s[k]=='O') kk=0;
}
if(kk==1) ok[i]=1;
}
if(ok[i]==1) cnt++;
}
}
printf("%d",cnt);
for(int i=1;i<=12;i++)
if(ok[i]==1) printf(" %dx%d",i,12/i);

}

}
