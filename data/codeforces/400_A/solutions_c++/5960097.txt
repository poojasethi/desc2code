#include <cstdio>
#include <cstdlib>
#include <cstring>

char s[15];
int q[15];
bool b[15];

void doit()
{
	scanf("%s",s),q[0]=0;
	for (int i=1,j,ok; j=12/i,i<=12; i++) if (12%i==0)
	{
		memset(b,1,sizeof(b)),ok=0;
		for (int k=0; k<12; k++) if (s[k]=='O') b[k%j]=0;
		for (int k=0; !ok&&k<j; k++) if (b[k]) ok=1;
		if (ok) q[++q[0]]=i;
	}
	printf("%d",q[0]);
	for (int i=1; i<=q[0]; i++) printf(" %dx%d",q[i],12/q[i]);
	puts("");
}

int main()
{
	int T;
	for (scanf("%d",&T); T; T--) doit();
	return 0;
}
