#include<cstdio>
#include<cstring>
#define N 100000
int n,i,s[N],j,tot,ans[N];
char f[N];
int main()
{
	scanf("%d",&n);
	while (n)
	{
		getchar();
		tot=0;
		n--;
		for (i=1;i<=12;i++)
		scanf("%c",&f[i]);
		
		for (i=1;i<=12;i++)
		if (12%i==0)
		{
		memset(s,0,sizeof(s));	
		
		for (j=1;j<=12;j++)
		if (f[j]=='X')
		s[j%(12/i)]++;
		
		for (j=0;j<(12/i);j++)
		if (s[j]==i)
		{
			tot++;ans[tot]=i;
			break;
		}
		}
		
		printf("%d",tot);
		for (i=1;i<=tot;i++)
		printf(" %dx%d",ans[i],12/ans[i]);
		printf("\n");
	}
}
