#include <stdio.h>

char str[100];

int main(){

	int T;
	scanf("%d ",&T);
	for(int t=0;t<T;++t){
		gets(str);
		int as[]={1,2,3,4,6,12};
		bool ok[6];int ct=0;
		for(int i=0;i<6;++i){
			int a=as[i];
			int b=12/a;	
			ok[i]=false;
			for(int c=0;c<b;++c){
				bool okcol = true;
				for(int r=0;r<a;++r){
					if(str[r*b+c]!='X')
						okcol=false;
				}
				ok[i]|=okcol;
			}
			if(ok[i])++ct;
		}
		printf("%d",ct);
		for(int i=0;i<6;++i){	
			if(ok[i]){
				printf(" %dx%d", as[i],12/as[i]);
			}
		}
		putchar('\n');
	}


}
