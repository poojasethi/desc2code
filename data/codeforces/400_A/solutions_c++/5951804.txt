#include <cstdio>

int t;
char strs[18];
int oset[18];
int solls[18];
int main() {
	scanf("%d",&t);
	for(int i=0;i<t;i++) {
		scanf("%s",strs);
		int sol=0;
		for(int j=1;j<13;j++) {
			solls[j]=0;
			if(12%j==0) {
				for(int k=0;k<j;k++) oset[k]=0;
				for(int k=0;k<12;k++) if(strs[k]=='O') oset[k%j]=1;
				for(int k=0;k<j;k++) if(oset[k]==0) solls[j]=1;
			}
		}
		for(int j=1;j<13;j++) if(solls[j]) sol++;
		printf("%d",sol);
		for(int j=12;j>0;j--) if(solls[j]) printf(" %dx%d",12/j,j);
		printf("\n");
	}
	return 0;
}
