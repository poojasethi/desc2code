#include<cstdio>
char str[15];
void solve() {
	int A=0;
	int ans[10];
	for(int i=1;i<=12;i++) {
		if(12%i==0) {
			int x=12/i,y;
			for(int k=0;k<x;k++) {
				y=1;
				for(int j=k;j<12;j+=x) 
					if(str[j]=='O') y=0;
				if(y) {
					ans[A++]=i;
					break;
				}
			}
		}
	}
	printf("%d ",A);
	for(int i=0;i<A;i++) 
		printf("%dx%d ",ans[i],12/ans[i]);
}
int main () {
	int t,T;
	scanf("%d",&T);
	for(int t=0;t<T;t++) {
		scanf("%s",str);
		solve();
		printf("\n");
	}
}
