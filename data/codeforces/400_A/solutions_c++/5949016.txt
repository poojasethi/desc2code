#include<bits/stdc++.h>
using namespace std;

int main() {
	int nTest; cin >> nTest;
	while(nTest-- > 0) {
		string s; cin >> s;
		vector<int> sol;
		for(int x = 1; x <= 12; ++x) if(12 % x == 0) {
			bool win = false;
			int y = 12 / x;
			for(int col = 0; col < y; ++col) {
				bool allX = true;
				for(int row = 0; row < x; ++row)
					allX &= s[row * y + col] == 'X';
				win |= allX;
			}
			if(win) sol.push_back(x);
		}
		cout << sol.size();
		for(unsigned i = 0; i < sol.size(); ++i)
			cout << ' ' << sol[i] << 'x' << 12 / sol[i];
		cout << '\n';
	}
	return 0;
}
