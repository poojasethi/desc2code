#include <bits/stdc++.h>
using namespace std;
int main()
{
    string s;
	int i,j,k,t,a[13],dem,x,h;
	cin >> t;

	for (h=1;h<=t;h++)
    {
		cin>>s;
		for (dem=0,i=1; i<=12; i++)
        {
			a[i]=0;
			if (12%i==0)
			{
				for (j=0; j<12/i; j++)
                {
					for (x=1,k=j; k<=12; k+=12/i)
					if(s[k]=='O') x=0;

					if (x==1) a[i]++;
				}
			}
			if (a[i]>0) dem++;
		}
		cout<<dem;
		for (i=1; i<=12; i++) if (a[i]) cout<<" "<<i<<"x"<<12/i;
		cout<<endl;
	}
}
