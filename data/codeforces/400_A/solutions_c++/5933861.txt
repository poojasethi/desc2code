# include <stdio.h>
# include <algorithm>
using namespace std;
int a,an,z;
int t[209];
char b[109][19];
int main()
{
	scanf("%d",&a);
	for(int i=0;i<a;i++)
	scanf("%s",b[i]);
	for(int i=0;i<a;i++)
	{
		an=0;
		for(int j=1;j<=12;j++)
		if(12%j==0)
		{
			for(int k=0;k<j;k++)
			{
				z=0;
				for(int l=k;l<12;l+=j)
				if(b[i][l]=='O')
				z=1;
				
				if(z==0)
				{t[an]=j;an++;break;}
			}
		}
		printf("%d",an);
		for(int j=an-1;j>=0;j--)
		printf(" %dx%d",12/t[j],t[j]);
		printf("\n");
	}
	getchar();
	getchar();
}
