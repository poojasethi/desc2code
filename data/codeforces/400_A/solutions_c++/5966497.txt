#include <stdio.h>
int main()
{
	char s[14];
	int t;
	int dx[]={1,2,3,4,6,12};
	int dy[]={12,6,4,3,2,1};
	scanf("%d",&t);
	while(t--)
	{
		int g=0;
		int vis[6]={0};
		scanf("%s",s);
		for(int i=0;i<6;i++)
		{
			bool bl=false;
			for(int k=0;k<dx[i];k++)
			{
				bool bbl=true;
				for(int j=0;j<dy[i];j++)
					if(s[j*dx[i]+k]!='X')
						bbl=false;
				if(bbl)
					bl=true;
			}
			if(bl)
			{
				vis[i]=true;
				g++;
			}
		}
		printf("%d",g);
		for(int i=5;i>=0;i--)
			if(vis[i])
				printf(" %dx%d",dy[i],dx[i]);
		puts("");
	}
	return 0;
}
