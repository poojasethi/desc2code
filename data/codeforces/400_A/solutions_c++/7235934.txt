#include<iostream>
using namespace std;
int main(){
	int test;
	cin >> test;
	while(test--){
	string s;
	cin >> s;
	int i,j,n,k,ans=0,cnt[13][13]={{0}},factors[6]={1,2,3,4,6,12};
	for(i=0;i<6;i++){
		n=factors[i];
		for(j=0;j<12;j++){
			if(s[j]=='X') cnt[n][j%n]++;
		}
		for(j=0;j<12;j++)
			if(cnt[n][j]==12/n) {cnt[n][0]=n+12;}
	}
	for(i=0;i<6;i++){
		n = factors[i];
		if(cnt[n][0]==n+12) ans++;
		else factors[i]=-1;
	}
	cout << ans << " ";
	for(i=5;i>=0;i--){
		if(factors[i]!=-1) cout<<12/factors[i] << "x" <<factors[i] <<" ";
	}
	cout << endl;
	}
	return 0;
}
