#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=15;
int T,n=12,r,c,anss,ans[maxn];
char s[maxn];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j,k,z;
	scanf("%d\n",&T);
	while(T--)
	{
		anss=0;
		scanf("%s\n",s);
		for(r=1;r<=n;r++)
			if(!(n%r))
			{
				c=n/r;
				for(j=0;j<c;j++)
				{
					z=0;
					for(k=j;k<n;k+=c)
						if(s[k]!='X') z=1;
					if(!z) { ans[anss++]=r;break; }
				}
			}
		printf("%d",anss);
		for(i=0;i<anss;i++)
			printf(" %dx%d",ans[i],n/ans[i]);
		printf("\n");
	}
	return 0;
}
