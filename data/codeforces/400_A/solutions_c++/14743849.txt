#include<iostream>
using namespace std;
int main()
{char c[12];
int t;
int a[6]={1,2,3,4,6,12};
int b[6];
cin>>t;
while(t--)
{cin>>c;int top=-1;
for(int i=0;i<6;i++)
{
for(int j=0;j<12/a[i];j++)
{int count=0;
for(int k=j;k<12;k+=12/a[i])if(c[k]=='X')count++;
if(count==a[i])
{b[++top]=a[i];break;
}
}}
cout<<top+1<<" ";
for(int i=0;i<=top;i++)cout<<b[i]<<"x"<<12/b[i]<<" ";
cout<<endl;
}
}
