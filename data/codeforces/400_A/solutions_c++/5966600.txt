string s;

bool win(int a,int b){

for(int i=1;i<=b;i++){
	int j=i;
	bool posible=true;
	while(j<=12){
		if(s[j-1]!='X') posible=false;	
		j+=b;
	}
	if(posible) return true;
}
return false;	
}
