#include <iostream>
#include <string>
#include <vector>
using namespace std;

int main(){
	ios::sync_with_stdio(false);
	int t;
	string s;
	cin>>t;
	while(t--){
		vector<bool> c(13,0);
		cin>>s;
		int ans=0;
		for(int i=1;i<=12;i++)
			if(12%i==0){
				for(int k=0;k<i;k++){
					bool co=1;
					for(int j=k;j<12;j+=i)if(s[j]!='X'){co=0;break;}
					if(co){
						ans++;
						c[i]=1;
						break;
					}
				}
			}
		cout<<ans;
		for(int i=12;i>=1;i--)
			if(c[i]) cout<<" "<<12/i<<"x"<<i;
		cout<<endl;
	}
}
