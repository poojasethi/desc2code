#include<cstdio>
int main()
{
    char s[13];
    int t,r,n,i,j,x,y,f[6]={1,2,3,4,6,12},a[6];
    for(scanf("%d",&t);t--;)
    {
        scanf("%s",s);
        for(r=n=0;n<6;n++)
        {
            j=12/(i=f[n]);a[n]=0;
            for(x=0;x<j;x++)
            {
                for(y=x;y<12;y+=j)if(s[y]!='X')goto fin;
                r++;a[n]=1;break;
                fin:;
            }
        }
        printf("%u",r);
        for(n=0;n<6;n++)if(a[n])printf(" %dx%d",f[n],12/f[n]);
        putchar('\n');
    }
    return 0;
}
