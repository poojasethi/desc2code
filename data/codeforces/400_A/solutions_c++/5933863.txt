#include <cstdio>

int aa[9], t, k, a, b, i, j;
char s[15];
bool f, c;

int main()
{
	scanf("%d", &t);

	while (t--)
	{
		scanf("%s", s);
		k = 0;
		for (a = 1; a <= 12; a++)
			if (12 % a == 0)
			{
				b = 12 / a;
				f = 0;
				for (j = 0; j < b; j++)
				{
					c = 1;
					for (i = 0; c && i < a; i++)
						if (s[i * b + j] != 'X')
							c = 0;
					f |= c;
				}
				if (f)
					aa[k++] = a;
			}
		printf("%d", k);
		for (i = 0; i < k; i++)
			printf(" %dx%d", aa[i], 12 / aa[i]);
		printf("\n");
	}

	return 0;
}
