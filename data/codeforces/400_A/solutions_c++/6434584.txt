#include<stdio.h>
#include<string.h>
int main()
{
int s[6]={
12,6,4,3,2,1
};
int i,n,j,k,c[100],z;
char a[100];
scanf("%d%*c",&n);
while(n--)
{
z=0;
gets(a);
memset(c,0,sizeof(c));
for(i=0;i<12;i++)
if(a[i]=='X')break;
if(i==12)printf("0\n");
else
{
for(i=1;i<6;i++)
for(j=0;j<s[i];j++)
{
for(k=0;k<12/s[i];k++)
{
if(a[j]!=a[j+k*s[i]]||a[j]=='O'&&a[j+k*s[i]]=='O')break;
}
if(k==(12/s[i])){ z++;c[i]=1;break;}
}
printf("%d 1x12",z+1);
for(i=1;i<6;i++)
if(c[i]==1)
printf(" %dx%d",12/s[i],s[i]);
printf("\n");  
}
}
}
