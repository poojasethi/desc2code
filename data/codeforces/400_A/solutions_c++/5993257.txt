#include <cstdio>
#include <cstring>
#include <algorithm>
char c[200];
int ans[100],top;
bool num[100];
void check(int x){
	std::fill(num,num+20,true);
	for (int i=0;i<=11;++i)
		if (c[i]=='O')num[i%x]=false;
	for (int i=0;i<x;++i) if (num[i]){
		ans[++top]=x;
		return;
		}
}
int main(){
	int T;scanf("%d",&T);
	while (T--){
		scanf("%s",c);top=0;
		for (int i=1;i<=12;++i)
			if (12%i==0) check(i);
		printf("%d",top);
		for (int i=top;i>=1;--i)
			printf(" %dx%d",12/ans[i],ans[i]);
		printf("\n");
		}
}
