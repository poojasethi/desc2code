#include<iostream>
#include<string>
using namespace std;
pair<int, int> ans[12];
int t, ae;
string s;
int main() {
	cin >> t;
	for(int i=0;i<t;i++) {
		cin >> s;
		ae = 0;
		for(int i=12;i>0;i--)
			if(12%i == 0) {
				for(int j=0;j < i;j++) {
					bool valid=true;
					for(int k=0;k*i < 12;k++)
						if(s[i*k + j] == 'O')
							valid = false;
					if(valid) {
						ans[ae++] = make_pair(12/i, i);
						break;
					}
				}
			}
		cout << ae;
		for(int i=0;i<ae;i++)
			cout << ' ' << ans[i].first<< 'x' << ans[i].second;
		cout << '\n';
	}
	return 0;
}
