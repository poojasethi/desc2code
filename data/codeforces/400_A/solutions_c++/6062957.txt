#include <bits/stdc++.h>

using namespace std;

int a;
char ar[20];

int main() {
	scanf("%d",&a);
	
	for(int i=1;i<=a;i++) {
		scanf(" %s",ar+1);
		vector <int> w;
		for(int j=1;j<=12;j++) {
			if( 12 % j ) continue;
			int m = 0;
			for(int k=1;k<=12/j;k++) {
				int n = 0;
				for(int l=k;l<=12;l+=12/j)
					if( ar[l] == 'O' ) { n = 1; break; }
				if( !n ) { m=1; break; }
			}
			if( m ) w.push_back( j );
		}
		printf("%d ",w.size());
		for(int j=0;j<w.size();j++)
			printf("%dx%d ",w[j],12/w[j]);
		printf("\n");
	}
	
	
	
	return 0;
}
