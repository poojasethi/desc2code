#include <iostream>

using namespace std;

int main()
{
	char s[15];
	int t, p, x[20], q;
	bool l = 1;
	for(cin >> t;t;--t)
	{
		p = 0;
		cin >> s;
		for(int i = 1;i <= 12;++i)
		{
			if (12 % i == 0)
			for(int j = 0;j < i;++j)
			{
				l = 1;
				for(int k = j;k < 12;k += i)
				if (s[k] != 'X')
				{
					l = 0;
					break;
				}
				if (l)
				{
					x[p++] = i;
					break;
				}
			}
		}
		//cout << endl;
		cout << p << ' ';
		for(--p;p >= 0;--p)
		{
			cout << 12 / x[p] << 'x' << x[p] << ' ';
		}
		cout << endl;
	}
	return 0;
}
