#include <cstdio>
#include <cstring>
int a[]={12, 6, 4, 3, 2, 1};
bool ok[6];
int main() {
	int t;
	char str[15];
	bool xd;
	scanf("%d", &t);
	while( t-- ) {
		int ans=0;
		memset(ok, false, sizeof(ok));
		scanf("%s", str);
		for(int i=0; i<6; ++i) {
			for(int j=0; j<a[i]; ++j) {
				xd=true;
				for(int k=j; k<12; k+=a[i])
					if( str[k]=='O' ) xd=false;
				if( xd ) {
					++ans, ok[i]=true;
					break;
				}
			}
		}
		printf("%d", ans);
		for(int i=0; i<6; ++i)
			if( ok[i] ) printf(" %dx%d", 12/a[i], a[i]);
		puts("");
	}
}
