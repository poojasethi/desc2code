#include <cstdio>
#include <vector>
using namespace std;

vector<int> sol;
int t; bool b;
char str[20];

int main(void) {
	scanf("%d",&t);
	for(int i=0;i<t;++i) {
		scanf("%s",str);
		sol.clear();
		for(int j=1;j<=12;++j) {
			if(12%j==0)
				for(int k=0;k<12/j;++k) {
					b = true;
					for(int l=k;l<12;l+=12/j)
						if(str[l] != 'X') { b = false; break; }
					if(b) { sol.push_back(j); break; }
				}
		}
		printf("%d ",sol.size());
		for(int j=0;j<sol.size();++j)
			printf("%dx%d ",sol[j],12/sol[j]);
		printf("\n");
	}
	return 0;
}
