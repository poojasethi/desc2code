#include<iostream>
using namespace std;

int b(char s[],int l,int n)
{
	int i,j,k,f=0;
	k=l/n;
	for(i=0;i<k&&f==0;i++)
	{
		for(j=0;j<n;j++)
			if(s[j*k+i]!='X')
				break;
		if(j>=n)
			f=1;
	}
	if(f==1)
		return 1;
	else
		return 0;
}

int main()
{
	int t,m,i,j,a[15],c[6]={1,2,3,4,6,12};
	char s[15];
	cin>>t;
	while(t--)
	{
		cin>>s;
		for(i=0,m=0;i<6;i++)
			if(b(s,12,c[i]))
			{
				a[2*m]=c[i];
				a[2*m+1]=12/c[i];
				m++;
			}
		cout<<m;
		for(i=0;i<m;i++)
			cout<<" "<<a[2*i]<<"x"<<a[2*i+1];
		cout<<endl;
	}
	return 0;
}
