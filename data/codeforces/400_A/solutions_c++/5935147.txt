#include <cstdio>

using namespace std;

	int t,div[6] = {1,2,3,4,6,12},num;
	char line[12];
	bool b,f,res[6];

int main() {
	
	scanf("%d\n",&t);
	while (t--) {
		num = 0;
		scanf("%s\n",line);
		for (int i=0; i<6; i++) {
			f = 0;
			for (int k=0; k<div[i]; k++) {
				b = 1;
				for (int j=k; j<12; j+=div[i]) b = b && (line[j] == 'X');
				f = f || b;
			}
			res[i] = f;
			num += f;
		}
		printf("%d",num);
		for (int i=5; i>=0; i--) {
			if (res[i]) printf(" %dx%d",12/div[i],div[i]);
		}
		printf("\n");
	}

	return 0;
}
