#include<stdio.h>
int t;
char s[111];
int a[8]={1,2,3,4,6,12};
bool abc(int d){
	for(int i=0;i<d;i++){
		bool f=true;
		for(int j=i;j<12;j+=d){
			if(s[j]!='X'){
				f=false;
				break;
			}
		}
		if(f)return true;
	}
	return false;
}
int main(){
	int i,n;
	while(~scanf("%d",&t)){
		while(t--){
			n=0;
			scanf("%s",s);
			for(i=0;i<6;i++){
				if(abc(12/a[i])){
					n++;
				}
			}
			printf("%d",n);
			for(i=0;i<6;i++){
				if(abc(12/a[i]))printf(" %dx%d",a[i],12/a[i]);
				
			}printf("\n");
		}
	}
	return 0;
}
