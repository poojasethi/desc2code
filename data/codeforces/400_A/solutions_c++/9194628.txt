#include<stdio.h>

int main(){
	int n,i,j,k,t,sum,a[]={1,2,3,4,6,12},b[6]={0},total;
	char str[13];

	scanf("%d", &n);
	while(n --){
		total = 0;
		scanf("%s", str);
		for(i=0;i<6;i++){
			for(k=0;k<12/a[i];k++){ 
				sum = 0;
				t = 0;
				j = 0;
				for(j=k;j<12;j=t*12/a[i]+k){
					if(str[j] == 'X')
						sum++;
					t++;
				}
				if(sum == a[i]){
					b[total++] = a[i];
					break;
				}
			}
		}

		printf("%d", total);
		for(i=0;i<total;i++){
			printf(" %dx%d", b[i], 12/b[i]);
		}
		printf("\n");
	}

	return 0;
}
