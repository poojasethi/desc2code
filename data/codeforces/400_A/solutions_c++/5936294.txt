#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cmath>
using namespace std;
int main(){
int t,a,b,n,i,j,k,l,p[3000],sz=0,q[3000],y;
string s;
cin>>t;
for(i=1;i<=t;i++){
cin>>s;
sz=0;
for(l=1;l<=12;l++){
if(12%l!=0)continue;
a=l;
b=12/l;
int h=0;
for(j=1;j<=12;j++){
h=0;
for(k=1;k<=a;k++){
if(s[j-1]!='X' || s[((k*b)+j-1)%12]!='X'){h=1;break;}
}
if(h==0){sz++;p[sz]=a;q[sz]=b;break;}
}
}
cout<<sz<<" ";
for(y=1;y<=sz;y++){
cout<<p[y]<<"x"<<q[y]<<" ";
}
cout<<endl;
}
return 0;
}
