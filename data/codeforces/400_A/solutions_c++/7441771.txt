#include <stdio.h>

int main(){

	char s[20];
	int n;
	scanf("%d",&n);
	int res[10];
	
	while (n--){
		scanf("%s",s);
		int num = 0;
		for (int a=1;a<=12;a++)
		if (12%a==0){
			int b = 12/a;

			for (int j=0;j<b;j++){
				
				int p = 1;
				for (int i=0;i<a;i++)
				if (s[i*b+j]=='O'){
					p = 0;
					break;
				}
				if (p==1){
					res[++num] = a;
					break;
				}
			}			
		}
		
		printf("%d ",num);
		for (int i=1;i<=num;i++){
			printf("%dx%d ",res[i],12/res[i]);
		}printf("\n");

	}
	
	return 0;
}
