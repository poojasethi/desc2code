#include <bits/stdc++.h>
using namespace std;
char a[20];
bool isOk(int n){
	if(12 % n) return false;
	int m = 12 / n;
	for(int j = 0; j < m; j ++){
		bool f = true;
		for(int i = 0; i < n; i ++) if(a[i * m + j] != 'X') f = false;
		if(f) return true;
	}
	return false;
}
void solve(){
	scanf("%s",a);
	int col = 0;
	for(int i = 1; i <= 12; i ++) if(isOk(i)) col ++;
	printf("%d",col);
	for(int i = 1; i <= 12; i ++) if(isOk(i)) printf(" %dx%d", i, 12 / i);
	puts("");
}
int main(){
	int n;
	scanf("%d",&n);
	while(n --) solve();
}
