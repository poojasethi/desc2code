#include <cstdio>

using namespace std;

int f[701][701][4];

int main(void)
{
  int n,m;
  scanf("%d %d",&n,&m);
  for(int i=1;i<=m;++i)
  {
    int j,k;
    scanf("%d %d",&j,&k);
    f[j][k][1]=1;
    f[k][j][1]=1;
  }
  for(int l=1;l<3;++l)
  {
    for(int i=1;i<n;++i)
    {
      for(int j=i+1;j<=n;++j)
      {
        for(int k=1;k<=n;++k)
          if((k!=i)&&(f[k][j][1]>0))
            f[i][j][l+1]+=f[i][k][l]-f[i][j][l-1];
        f[j][i][l+1]=f[i][j][l+1];
      }
    }
  }
  long long ans=0;
  for(int i=1;i<n;++i)
    for(int j=i+1;j<=n;++j)
      for(int k=1;k<=n;++k)
        if((f[i][k][1]>0)&&(f[k][j][1]>0))
          ans+=f[i][j][3]-f[i][k][2]-f[k][j][2]+f[i][j][1]+f[j][i][1];
  ans/=5;
  printf("%I64d\n",ans);
  return 0;
}
