#include<cstdio>
#include<cmath>
#include<cstring>
#include<algorithm>
#include<iostream>
#define MAXN 705
using namespace std;
typedef long long LL;
int n;
void mul(LL a[][MAXN],LL b[][MAXN],LL c[][MAXN])
{
	for(int i=0;i<n;i++)
		for(int j=0;j<n;j++){
			c[i][j]=0;
			for(int k=0;k<n;k++)
				c[i][j]+=a[i][k]*b[k][j];
		}
}
LL deg[MAXN],x[MAXN][MAXN],y[MAXN][MAXN],z[MAXN][MAXN];
int main()
{
	//freopen("input.txt","r",stdin);
	int u,v,m;
	while(cin>>n>>m){
		memset(x,0,sizeof(x));
		memset(deg,false,sizeof(deg));
		while(m--){
			scanf("%d%d",&u,&v);
			u--;
			v--;
			deg[u]++;
			deg[v]++;
			x[u][v]++;
			x[v][u]++;
		}
		mul(x,x,y);
		mul(x,y,z);
		LL ans=0;
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				ans+=y[i][j]*z[j][i];
		ans/=10;
		for(int i=0;i<n;i++)
			for(int j=i+1;j<n;j++)
				for(int k=j+1;k<n;k++){
					if(x[i][j]&&x[j][k]&&x[k][i])
						ans-=deg[i]+deg[j]+deg[k]-3;
				}
		cout<<ans<<endl;
	}
	return 0;
}
