#include<cstdio>
#include<iostream>
using namespace std;
#define maxn 701
#define f(i,l,r) for (i=l;i<=r;i++)

long long g[6][maxn][maxn];
long long tmp,ans;
int i,j,k,n,m,u,v;
int d[maxn];

void mult(int x,int y,int z)
{
	f(k,1,n) f(i,1,n)
	if (g[y][i][k]) f(j,1,n) g[x][i][j]+=(long long)g[y][i][k]*g[z][k][j];
}

int main()
{	
	scanf("%d %d",&n,&m);
	f(i,1,m)
	{
		scanf("%d %d",&u,&v);
		d[u]++; d[v]++;
		g[1][u][v]=g[1][v][u]=1;
	}
	mult(2,1,1);
	mult(3,1,2);
	mult(5,2,3);
	f(i,1,n) ans+=(long long)g[5][i][i];

	ans/=10;
	f(i,1,n) f(j,i+1,n)
	if (g[1][i][j])
	{
		f(k,j+1,n)
		if (g[1][i][k]&g[1][k][j]) ans-=d[i]+d[k]+d[j]-3;
	}
	cout<< ans <<endl;
}
