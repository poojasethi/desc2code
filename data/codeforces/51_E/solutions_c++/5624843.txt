#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
#define N 710
using namespace std;
bool a[N][N];
int n,m,x,y,a2[N][N],a3[N][N],cnt[N],i,j,k;
long long ans;
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;++i){
		scanf("%d%d",&x,&y);
		a[x][y]=a[y][x]=1;
		cnt[x]++; cnt[y]++;
	}
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)if(a[i][j] && i!=j){
			int *v=a2[i];
			for(k=1;k<=n;++k)if(a[j][k] && k!=i && k!=j)v[k]++;
		}
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)if(a2[i][j] && i!=j){
			int *v=a3[i];
			for(k=1;k<=n;++k)if(a[j][k] && k!=i && k!=j)v[k]+=a2[i][j];
		}
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)if(a[i][j])a3[i][j]-=cnt[j]-1;
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)ans+=1ll*a2[i][j]*a3[i][j];
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)if(i!=j)ans-=2ll*a2[i][j]*a[i][j]*(cnt[i]-2);
	printf("%I64d\n",ans/10);
}
