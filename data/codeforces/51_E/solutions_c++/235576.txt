#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <iostream>
using namespace std;
typedef long long ll;
int edge[700][700];
int adj[700];
int forgot[700];
int common[700][700];
int n,m;
int main()
{
	scanf("%d %d",&n,&m);
	for (int q=0;q<m;++q)
	{
		int u,v;
		scanf("%d %d",&u,&v); u--;v--;
		edge[u][v] = edge[v][u] = 1;
	}

	for (int q=0;q<n;++q) for (int w=0;w<n;++w)
		adj[q] += edge[q][w];
	for (int q=0;q<n;++q) 
		for (int w=q+1;w<n;++w) 
			for (int e=w+1;e<n;++e)
			{
				if (edge[q][w] && edge[q][e]) common[w][e]++,common[e][w]++;
				if (edge[w][q] && edge[w][e]) common[q][e]++,common[e][q]++;
				if (edge[e][q] && edge[e][w]) common[q][w]++,common[w][q]++;
			}
	ll ret = 0;
	for (int q=0;q<n;++q) 
	{
		for (int w=0;w<n;++w) for (int e=w+1;e<n && w!=q;++e)
		{
			if (e!=q && edge[w][e])
			{
				int add = common[q][w] * common[q][e];
				if (edge[q][w]) add -= common[q][w];
				if (edge[q][e]) add -= common[q][e];
				if (edge[q][w] && edge[q][e]) 
				{
					add++;
					int del = adj[q]-2;
					ret -= del;
				}
				ret += add;
				
			}
		}
	}
	cout << ret/5 << endl;
}
