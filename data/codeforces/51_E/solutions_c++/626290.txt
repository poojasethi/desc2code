//#define local
#include <fstream>
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <string>
using namespace std;

#ifdef local
    ifstream inf("Input.txt");
    ofstream outf("Output.txt");
#else
    #define inf cin
    #define outf cout
#endif

typedef pair<int, int> pii;
typedef long long ll;

const int MAXN = 700;

int n, m;
int d[MAXN][MAXN];
int st[MAXN];
int Q[MAXN][MAXN];

int main()
{
    inf >> n >> m;
    for (int i = 0; i < m; i++) {
        int a, b; inf >> a >> b;
        --a; --b;
        d[a][b] = 1; d[b][a] = 1;
        ++st[a]; ++st[b];
    }

    for (int i = 0; i < n-1; i++)
     for (int j = i+1; j < n; j++)
      for (int k = 0; k < n; k++)
        if (d[i][k] == 1 && d[j][k] == 1) ++Q[i][j], ++Q[j][i];

    ll res = 0;
    for (int i = 0; i < n-1; i++)
     for (int j = i+1; j < n; j++) if (d[i][j] == 1)
      for (int k = 0; k < n; k++) if (k != i && k != j) {
          res += (Q[i][k]-d[j][k])*(Q[j][k]-d[i][k]);
          if (d[i][k] == 1 && d[j][k] == 1) res -= (st[k]-2);
      }

    outf << res/5 << endl;
}
