#include <cstdio>
#include <cstring>

using namespace std;

const int maxn = 710;
int n, m, d[maxn], s[maxn][maxn];
bool e[maxn][maxn];

int main()
{
	scanf("%d%d", &n, &m);
	memset(e, 0, sizeof e);
	memset(d, 0, sizeof d);
	for (int i = 0; i < m; i++)
	{
		int a, b;
		scanf("%d%d", &a, &b);
		a--, b--;
		e[a][b] = e[b][a] = true;
		d[a]++, d[b]++;
	}
	memset(s, 0, sizeof s);
	for (int i = 0; i < n; i++)
		for (int j = i + 1; j < n; j++)
		{
			for (int k = 0; k < n; k++)
				if (e[i][k] && e[k][j])
					s[i][j]++;
			s[j][i] = s[i][j];
		}
	long long ans = 0;
	for (int i = 0; i < n; i++)
		for (int j = i + 1; j < n; j++)
			if (e[i][j])
			{
				for (int k = 0; k < n; k++)
				{
					long long dd = (long long)(s[i][k] - e[j][k]) * (s[j][k] - e[i][k]);
					if (dd > 0) ans += dd;
					if (e[i][k] && e[j][k])
						ans -= d[k] - 2;
				}
			}
	printf("%I64d\n", ans / 5);
	
	return 0;
}
