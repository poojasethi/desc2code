#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <cmath>
#include <complex>
#include <queue>
#include <stack>
#include <set>
#include <map>

using namespace std;
int sum=0,res=0;

bool q[700][700];
int a[700][700];
int st[700];
long long n,m;
int main()
{
	cin>>n>>m;
	while (m--)
	{
		int a,b;
		cin>>a>>b;
		a--,b--;
		q[a][b]=1;
		st[a]++;
		st[b]++;
		q[b][a]=1;
	}
	for(int k=0;k<n;k++)
	{
		for(int i=0;i<n;i++)
		{
			for(int j=0;j<n;j++)
			{

				if (i!=k&&j!=k&&j!=i&&q[i][k]&&q[k][j])
					a[i][j]++;
			}
		}
	}
	long long res=0;
	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			for(int l=0;l<n;l++)
			{
				if(i!=j&&i!=l&&j!=l&&q[i][j])
				{
					int ta=a[i][l]-q[j][l];
					int tb=a[j][l]-q[i][l];
					res+=ta*tb;
					if (q[i][l]&&q[j][l])
						res-=st[l]-2;
				}
			}
		}
	}
	cout<<res/10;



	return 0;
}
