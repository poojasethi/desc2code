#include <iostream>
using namespace std;
#define REP(i,a,b) for(i=a;i<b;i++)

int g[3][800][800];

int main(){
        int i,x,y,z,n,m;
        cin>>n>>m;
        REP(i,0,m){
                //cout<<i<<' '<<m<<endl;
                cin>>x>>y;
                x--;y--;
                g[0][x][y]++;g[0][y][x]++;
        }
        REP(i,0,2)REP(x,0,n)REP(y,0,n)if(g[i][x][y])
        REP(z,0,n)if(g[0][y][z]) g[i+1][x][z]+=g[i][x][y];

        long long ans=0;
        REP(x,0,n)REP(y,x+1,n)REP(z,0,n)if(g[2][x][y]&&g[0][x][z]&&g[0][y][z]){
                ans+=g[2][x][y];
                ans-=g[1][x][z]+g[1][y][z];
                if (g[0][x][y]) ans-=g[1][x][x]+g[1][y][y]-3;
                //cout<<x<<' '<<y<<' '<<z<<' '<<ans<<endl;
        }
        //cout<<ans<<endl;
        cout<<ans/5<<endl;
}
