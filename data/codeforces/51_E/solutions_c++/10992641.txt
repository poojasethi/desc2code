#include<cstdio>
#define REP(i, n) for (int i = 0, _n = (n); i < _n; ++i)
#define FOR(i, a, b) for (int i = (a), _n = (b); i <= _n; ++i)
int x,y,n,m,i,k,G[700][700],P2[700][700],P3[700][700];
long long res;
void multiply(int (*A)[700],int (*B)[700],int (*C)[700]){
	REP(i,n)REP(k,n)FOR(j,0,i)C[i][j]+=A[i][k]*B[k][j];
	REP(i,n)FOR(j,i+1,n-1)C[i][j]=C[j][i];
}
int main(){
	scanf("%d%d",&n,&m);
	REP(i,m){
		scanf("%d%d",&x,&y);
		x--; y--;
		G[x][y]=G[y][x]=1;
	}
	multiply(G,G,P2);
	multiply(P2,G,P3);
	REP(i,n)for(k=i+1;k<n;++k)res+=1LL*P2[i][k]*(P3[i][k]+G[i][k]*(5-2*(P2[i][i]+P2[k][k])));
	printf("%I64d\n",res/5);
	return 0;
}
