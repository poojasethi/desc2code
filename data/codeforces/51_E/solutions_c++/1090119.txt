#include<cstdio>
long long a[710][710][6],ans;
int d[710],n,m,x,y;
int main()
{
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;i++)
    {
        scanf("%d%d",&x,&y);
        a[x][y][1]=a[y][x][1]=1;
        d[x]++;
        d[y]++;
    }
    int i,j,k;
    for(i=1;i<=n;i++)
    for(j=i;j<=n;j++)
    {
        for(k=1;k<=n;k++)
        if(k!=i&&k!=j)
        a[i][j][2]+=a[i][k][1]*a[k][j][1];
        a[j][i][2]=a[i][j][2];
    }
    for(i=1;i<=n;i++)
    for(j=i;j<=n;j++)
    {
        for(k=1;k<=n;k++)
        if(k!=i&&k!=j)
        a[i][j][3]+=(a[i][k][2]-a[i][j][1]*a[j][k][1])*a[k][j][1];
        a[j][i][2]=a[i][j][2];
    }
    for(i=1;i<=n;i++)
    for(j=i+1;j<=n;j++)
    ans+=a[i][j][3]*a[i][j][2]*2;
    for(i=1;i<=n;i++)
    ans-=(d[i]-2)*a[i][i][3]*2;
    printf("%I64d\n",ans/10);
}
