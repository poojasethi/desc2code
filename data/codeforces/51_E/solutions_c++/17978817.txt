#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
#define maxn 710
typedef long long ll;
using namespace std;
int n, m, u, v;
ll ans, d[maxn][maxn], t[maxn][maxn], deg[maxn], g[maxn][maxn];
int main()
{
	scanf("%d%d", &n, &m);
	for (int i = 1; i <= m; ++ i){
		scanf("%d%d", &u, &v);
		g[u][v] = g[v][u] = 1;
		++ deg[u]; ++ deg[v];
	}
	for (int i = 1; i <= n; ++ i)
		for (int j = i + 1; j <= n; ++ j){
			for (int k = 1; k <= n; ++ k)
				d[i][j] += g[i][k] * g[k][j];
			d[j][i] = d[i][j];
		}
	for (int i = 1; i <= n; ++ i)
		for (int j = 1; j <= n; ++ j)
			for (int k = 1; k <= n; ++ k)
				if (j != i && i != k) t[i][j] += d[i][k] * g[k][j];
	for (int i = 1; i <= n; ++ i)
		for (int j = 1; j <= n; ++ j)
			ans += d[i][j] * t[i][j];
	for (int i = 1; i <= n; ++ i)
		for (int j = i + 1; j <= n; ++ j)
			if (g[i][j])
				for (int k = j + 1; k <= n; ++ k)
					if (g[j][k] && g[i][k])
						ans -= (deg[i] + deg[j] + deg[k] - 6ll) * 6ll + 6ll;
	ans /= 10ll;
	printf("%I64d\n", ans);
	return 0;
}
