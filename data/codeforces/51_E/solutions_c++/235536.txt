#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

const int maxn = 701;
int g[maxn][maxn][3];
int n,m;

int main(){
    scanf("%d%d",&n,&m);
    memset(g,0,sizeof(g));
    int x,y,z;
    for(int i=0;i<m;i++){
        scanf("%d%d",&x,&y);
        x--,y--;
        g[x][y][0] = g[y][x][0] = 1;
    }
    for(int i = 0;i<2;i++)
        for(x = 0;x<n;x++)
            for(y = 0;y<n;y++)
                if(g[x][y][i])
                    for(z = 0;z<n;z++)
                        if(g[y][z][0])g[x][z][i + 1] += g[x][y][i];
    long long ans = 0;
    for(x = 0;x<n;x++)
        for(y = x + 1;y<n;y++)
            for(z = 0;z<n;z++)if(g[z][x][0] && g[z][y][0]){
                ans += g[x][y][2] - g[x][z][1] - g[y][z][1];
                if(g[x][y][0])ans -= g[x][x][1] + g[y][y][1] - 3;
            }
    cout<<ans/5<<endl;
    return 0;
}
