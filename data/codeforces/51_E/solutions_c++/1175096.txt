#include <cstdio>
#include <vector>
using namespace std;

typedef int Mat[700][700];
typedef long long LL;
int n,m,x,y,cnt[700];
Mat a,b,c;

void gao(int n, const Mat a, const Mat b, Mat c){
    for(int i=0;i<n;i++) for(int j=i;j<n;j++){
        for(int k=0;k<n;k++) c[i][j]+=a[i][k]*b[j][k];
        c[j][i]=c[i][j];
    }
}

int main(){
    scanf("%d%d",&n,&m);
    for(int i=0;i<m;i++){
        scanf("%d%d",&x,&y);
        cnt[--x]++;
        cnt[--y]++;
        a[x][y]=a[y][x]=1;
    }
    LL ans=0;
    gao(n,a,a,b);
    gao(n,a,b,c);
    for(int i=0;i<n;i++) for(int j=0;j<n;j++)
        ans+=LL(b[i][j])*c[i][j];
    ans/=10;
    for(int i=0;i<n;i++) for(int j=i+1;j<n;j++) if(a[i][j])
        for(int k=j+1;k<n;k++) if(a[i][k] && a[j][k])
            ans-=cnt[i]+cnt[j]+cnt[k]-3;
    printf("%I64d\n",ans);
}
