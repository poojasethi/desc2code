#include <iostream>
#include <cstring>
#include <cmath>
#include <algorithm>
#include <set>
#include <vector>
#include <cstdio>
using namespace std;

#define FOR(i,m,n) for(int i=m; i<n; i++)
#define ms(obj, val) memset(obj, val, sizeof(obj))
#define mp make_pair
#define pb push_back
#define ri(a) scanf("%d",&a)
#define rii(a,b) scanf("%d%d",&a,&b)

typedef long long ll;
typedef vector<int> vi;

const int MAXN=705;
ll G[MAXN][MAXN]={}, G2[MAXN][MAXN], G3[MAXN][MAXN];


int main(){
	int n, m, u, v;
	ll res=0;
	rii(n, m);
	FOR(i, 0, m){
		rii(u, v); u--; v--;
		G[u][v]=1;
		G[v][u]=1;
	}
	FOR(i, 0, n) FOR(j, 0, n) FOR(k, 0, n) G2[i][j]+=G[i][k]*G[k][j];
	FOR(i, 0, n) FOR(j, 0, n) FOR(k, 0, n) G3[i][j]+=G2[i][k]*G[k][j];
	FOR(i, 0, n) {
		res-=(5*G3[i][i]*(G2[i][i]-1));
		FOR(k, 0, n) res+=G2[i][k]*G3[k][i];
	}
	printf("%I64d\n", res/10);
}
