#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
const int mm=777;
int arc[4][mm][mm];
int main()
{
    int n,m;
    long long ans;
    while(scanf("%d%d",&n,&m)!=EOF)
    {
        ans=0;
        memset(arc,0,sizeof(arc));
        int i,j,k,l,a,b;
        for(i=0;i<m;i++)
        {
            scanf("%d%d",&a,&b);
            arc[1][a][b]=1;
            arc[1][b][a]=1;
        }
        for(l=1;l<3;l++)
            for(i=1;i<=n;i++)
                for(j=1;j<=n;j++)
                    if(arc[l][i][j])
                        for(k=1;k<=n;k++)
                            if(arc[1][j][k])
                                arc[l+1][i][k]+=arc[l][i][j];
        for(i=1;i<=n;i++)
            for(j=i+1;j<=n;j++)
                for(k=1;k<=n;k++)
                    if(arc[3][i][j]&&arc[1][i][k]&&arc[1][j][k])
                    {
                        ans+=arc[3][i][j]-arc[2][i][k]-arc[2][j][k];
                        if(arc[1][i][j])
                            ans-=arc[2][i][i]+arc[2][j][j]-3;
                    }
    printf("%I64d\n",ans/5);
    }
    return 0;
}
