#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <cstring>
#include <string>
#include <cctype>
#include <algorithm>
#define maxn 705

using namespace std;

typedef long long LL;

int n,m;LL f[4][maxn][maxn],ans;

void init()
 {
  int x,y,i;
  scanf("%d%d",&n,&m);
  for (i=1;i<=m;i++)
   {
    scanf("%d%d",&x,&y);
    f[1][x][y]=f[1][y][x]=1;
   }
 }
 
void work()
 {
  int i,j,k,l;LL t,sum;
  for (l=1;l<=2;l++)
   for (k=1;k<=n;k++)
    for (i=1;i<=n;i++)if (f[l][i][k])
     for (j=1;j<=n;j++)if (f[1][k][j]) 
       f[l+1][i][j]+=f[l][i][k]*f[1][k][j];
  for (i=1;i<=n;i++)
   {
     for (j=1;j<=n;j++)if (f[2][i][j] && f[3][j][i] && i!=j)
      {
       t=0;
       sum=f[3][j][i];
       if (f[1][i][j])sum-=f[2][i][i]+f[2][j][j]-1;
       for (k=1;k<=n;k++) if (f[1][i][k] && f[1][k][j])
        {
         t+=sum-f[2][k][i]-f[2][j][k];
         if (f[1][i][j])t+=2;
        }
       ans+=t;
      }
   }
  printf("%I64d\n",ans/10);
 }
 
int main()
 {
  init();
  work();
  return 0;
 }
