(n,r1,r2)=map(int,raw_input().split())
a=map(int,raw_input().split())
up=map(apply,[list]*(n+1))
for i in xrange(n-1):
  j=i+1
  if r1<=j:
    j+=1
  up[j].append(a[i])
  up[a[i]].append(j)


queue=[r2]
visited=[0]*(n+1)
res=[0]*(n+1)
visited[r2]=1
for i in xrange(n):
  for j in up[queue[i]]:
    if visited[j] is 0:
      queue.append(j)
      res[j]=queue[i]
      visited[j]=1

res=filter(lambda x:x!=0,res)

for result in res:
  print result,
