import sys
import os
import string
(n,r1,r2)=[int(x) for x in raw_input().split()]
line=iter([int(x) for x in raw_input().split()])
fa=[0]*(n+1)
for i in xrange(1,n+1):
    if (i==r1):
        continue;
    fa[i]=line.next();
ans=fa[:]
p=r2
while p!=r1:
    ans[fa[p]]=p
    p=fa[p]
ans[r2]=-1
print " ".join([str(x) for x in ans if x>0])
