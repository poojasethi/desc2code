from __future__ import print_function
n, r1, r2 = map (int, raw_input ().split ())
g = [list () for _ in xrange (n + 1)]
p = [0 for _ in xrange (n + 1)]
for u, v in zip (range (1, n), (map (int, raw_input ().split ()))):
    g[u + (u >= r1)].append (v)
    g[v].append (u + (u >= r1))

stack = [r2]
while stack:
    u = stack.pop ()
    for v in g[u]:
        if (not p[v]) and v != r2:
            p[v] = u
            stack.append (v)

print (*filter (lambda x: x, p))
