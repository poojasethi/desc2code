#!/usr/bin/env python
# -*- coding: utf-8 -*-

n, r1, r2 = map(int, raw_input().split())
l = map(int, raw_input().split())
sg = [[] for _ in xrange(n+1)]
ans = [None]*n
used = [False]*(n+1)

for i in xrange(1, r1):
    sg[i].append(l[i-1])
    sg[l[i-1]].append(i)
for i in xrange(r1+1, n+1):
    sg[i].append(l[i-2])
    sg[l[i-2]].append(i)

q = [r2]
while q:
    v = q.pop()
    for x in sg[v]:
        if not used[x]:
            ans[x-1] = v
            q.append(x)
    used[v] = True

for x in ans[:r2-1] + ans[r2:]:
