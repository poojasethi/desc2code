from collections import deque

n, r1, r2 = map(int, raw_input().split())

e = [[] for _ in xrange(n + 1)]
it = iter(map(int, raw_input().split()))
for i in xrange(1, n + 1):
	if i != r1:
		j = it.next()
		e[i].append(j)
		e[j].append(i)

ans = [-1] * (n + 1)

q = deque([r2])
ans[r2] = -2
while q:
	u = q.popleft()
	for v in e[u]:
		if ans[v] == -1:
			ans[v] = u
			q.append(v)
