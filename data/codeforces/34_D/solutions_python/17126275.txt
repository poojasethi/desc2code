def main():
	n, r1, r2 = map(int, raw_input().split())
	fa = map(int, raw_input().split())
	fa = [0] + fa[0:r1 -1] + [-1] + fa[r1 - 1:]
	ans = fa[:]
	p = r2
	while p != r1:
		ans[fa[p]] = p
		p = fa[p]
	ans = ans[1:r2] + ans[r2 + 1:]
	print " ".join(map(str, ans))


if __name__ == "__main__": main()
