#include<iostream>
#include<vector>

using namespace std;

int road[50001];
int t,c,f;

void rec(int city)
{
 if(city!=c) rec(road[city]);
  road[road[city]]=city;
}

int main()
{
    
    cin>>t>>c>>f;
    for(int x=1;x<=t;x++)
        if(x!=c) 
                 cin>>road[x];
    
    rec(f);
    cout<<"\n";
    for(int i=1;i<=t;i++)
            if(i!=f)
                    cout<<road[i]<<" ";
    
    return 0;
    }
