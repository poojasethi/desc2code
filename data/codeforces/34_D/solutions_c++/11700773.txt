#include <stdio.h>
int main() {
	int n,r2,r1,p[50001]={},at,ans[50001];
	scanf("%d%d%d",&n,&r1,&r2);
	for(int i=1;i<=n;++i){
		if(i==r1) continue;
		scanf("%d",&p[i]);
		ans[i]=p[i];
	}
	at=r2;
	do{
		ans[p[at]]=at;
		at=p[at];
	}while(at!=0);
	for(int i=1;i<=n;++i){
		if(i==r2) continue;
		printf("%d ",ans[i]);
	}
	return 0;
}
