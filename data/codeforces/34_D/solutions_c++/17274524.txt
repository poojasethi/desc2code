#include<bits/stdc++.h>
using namespace std;
int ans[100001];
std::vector<int> adj[100001];
void dfs(int cur,int pr){
	ans[cur]=pr;
	for(int i=0;i<adj[cur].size();i++){
		if(adj[cur][i]!=pr)
			dfs(adj[cur][i],cur);
	}
}
int main(){
	int n,r1,r2;
	scanf("%d %d %d",&n,&r1,&r2);
	for(int i=1;i<=n;i++){
		if(i==r1)
			continue;
		int a;
		scanf("%d",&a);
		adj[a].push_back(i);
		adj[i].push_back(a);
	}
	dfs(r2,-1);
	for(int i=1;i<=n;i++){
		if(i==r2)
			continue;
		printf("%d ",ans[i]);
	}
	printf("\n");
	return 0;
}
