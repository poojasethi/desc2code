#include<iostream>
using namespace std;
int n,r1,r2,p[50004];
int main()
{
	cin>>n>>r1>>r2;
	for(int i = 1; i <= n; i++)
		if (i != r1) cin>>p[i];
          else p[r1] = 0;
     int prev = r2, x = p[r2], y;
     do
     {
     	y = p[x];
     	p[x] = prev;
     	prev = x;
     	x = y;
     } while(x != 0);
     for(int i = 1; i <= n; i++)
     	if (i != r2) cout<<p[i]<<" ";
	return 0;
}
