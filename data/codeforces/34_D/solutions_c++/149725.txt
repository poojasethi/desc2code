#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>

int n, a, b;
int num[500005];
int ans[500005];


int main()
{
    scanf("%d %d %d", &n, &a, &b);
    for (int i = 1; i <= n; ++i)
    {
        if (i != a) scanf("%d", num + i);
        ans[i] = num[i];
    }
    int tmp = b;
    for (int pre = num[b]; pre != 0; pre = num[pre])
    {
        ans[pre] = tmp;
        tmp = pre;
    }
    for (int i = 1;i <= n; ++i)
    {
        if (i != b) printf("%d ", ans[i]);
    }
    return 0;
}
