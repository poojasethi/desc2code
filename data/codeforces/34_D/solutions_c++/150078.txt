#include<iostream>
#include<vector>
using namespace std;
vector<int> a[50005];
int v[50005],r1,r2,n,t[50005];
void dfs(int k,int tt)
{int i;
 t[k]=tt;
 v[k]=1;
 for(i=0;i<a[k].size();++i)
	 if(!v[a[k][i]])
		 dfs(a[k][i],k);
}
int main()
{cin>>n>>r1>>r2;
 int i;
  for(i=1;i<=n-1;++i)
   if(i>=r1)
    cin>>t[i+1];
     else
       cin>>t[i];
	
 for(i=1;i<=n;++i)
  if(t[i])
	{ 
	   a[i].push_back(t[i]);
	   a[t[i]].push_back(i);
	 }
 dfs(r2,0);
 for(i=1;i<=n;++i)
	 if(t[i])
		 cout<<t[i]<<" ";

 return 0;
}
