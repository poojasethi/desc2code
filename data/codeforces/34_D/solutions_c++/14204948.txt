#include<stdio.h>
int n,cap,cap2,i,arr[500000],a;

int main(){
    scanf("%d %d %d",&n,&cap,&cap2);

    for(i=1;i<=n;i++){
        if(i==cap)
            continue;
        scanf("%d",&a);
        arr[i]=a;   
    }
    int node=arr[cap2],x = cap2;
    while(node){
        int temp=arr[node];
        arr[node] = x;
        x=node; 
        node = temp;
    }
    
    for(i=1;i<=n;i++)
        if(i!=cap2)
            printf("%d ",arr[i]);
    return 0;
}
