#include <iostream>
using namespace std;

int n, r1, r2, par[50005];

int main(){

    cin >> n >> r1 >> r2;
    r1--; r2--;
    for(int i = 0; i<n-1; i++){
        cin >> par[i+(i>=r1)];
        par[i+(i>=r1)]--;
    }
    par[r1] = -1;
    int cur = r2, prev = -1;
    while(cur!=-1){
        int temp = par[cur];
        par[cur] = prev;
        prev = cur;
        cur = temp;
    }
    for(int i = 0; i<n; i++) if(i!=r2) cout << par[i]+1 << " ";
    cout << endl;

}
