#include<bits/stdc++.h>
int main()
{
	int n,r1,r2;
	scanf("%d %d %d",&n,&r1,&r2);
	int p[50005];
	for(int i=1;i<=n;i++)
	{
		if(i==r1)
		{
			p[i]=i;
			continue;
		}
		int parent;
		scanf("%d",&parent);
		p[i]=parent;
	}
	int prev=r2;
	int up=r2;
	while(up!=r1)
	{
		int newp=p[up];
		p[up]=prev;
		prev=up;
		up=newp;
	}
	p[r1]=prev;
	for(int i=1;i<=n;i++)
	{
		if(i!=r2)
			printf("%d ",p[i] );
	}
	printf("\n");
	return 0;
}
