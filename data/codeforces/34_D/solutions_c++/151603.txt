#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

int pre[50005];

int main(){
	int n, r1, r2;
	cin >> n >> r1 >> r2;
	memset(pre, 0, sizeof(pre));

	for(int i = 1; i <= n; i++){
		if(i == r1)
			continue;
		cin >> pre[i];
	}

	int a = r2, b = pre[a];
	while(b != 0){
		int tmp = pre[b];
		pre[b] = a;
		a = b;
		b = tmp;
	}

	pre[r2] = 0;
	for(int i = 1; i <= n; i++){
		if(i == r2)
			continue;
		cout << pre[i] << " ";
	}
	cout << endl;


	return 0;
}
