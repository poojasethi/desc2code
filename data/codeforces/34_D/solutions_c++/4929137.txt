#include<stdio.h>
#include<string.h>

int main()
{
int a[50010];
int n,r1,r2;	
int i,j,k,now;
while(scanf("%d%d%d",&n,&r1,&r2)==3)
{
	k=0;
	a[r1]=0;
	for(i=1;i<=n;i++)
	{
	   if(i!=r1)
	   {
          scanf("%d",&a[i]);
       } 
	}
	j=a[r2];
	a[r2]=0;
	now=r2;
	while(j!=0)
	{
		k=a[j];
		a[j]=now;
		now=j;
		j=k;
	}
	k=0;
	for(i=1;i<=n;i++)
	{
		if(i!=r2)
		{
			k++;
			printf("%d%c",a[i],k==n-1?'\n':' ');
		}
	}
}
return 0;
}
