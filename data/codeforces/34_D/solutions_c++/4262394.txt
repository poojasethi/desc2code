#include <iostream>
#include <cstdio>
#include <vector>
using namespace std;
int n,r1,r2,i,p[50001],a,b;
vector <int> g[50001];
void dfs(int v,int par=-1)
{
p[v]=par;
for (int i=0;i<g[v].size();i++)
	{
	if (g[v][i]!=par)
		dfs(g[v][i],v);
	}

}
int main()
{
cin>>n>>r1>>r2;
for (i=1;i<=n;i++)
{
 if (i==r1)
	continue;
	cin>>p[i];	
	g[i].push_back(p[i]);
	g[p[i]].push_back(i); 	
}
dfs(r2);
for (i=1;i<=n;i++)
{
	if(i==r2)
		continue;
	cout<<p[i]<<" ";
}




return 0;
}
