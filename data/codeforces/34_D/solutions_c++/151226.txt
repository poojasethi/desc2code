#include <stdio.h>
#define F(i, n) for(i = 0; i < n; ++i)
int a[50000], te;
void rec(int i, int p) {
  int te = a[i];
  a[i] = p;
  if(te < 0) return;
  rec(te - 1, i + 1);
}
int main() {
  int n, r1, r2, i;
  scanf("%d %d %d", &n, &r1, &r2);
  F(i, n) 
    if(i != r1 - 1) scanf("%d", &a[i]);
  a[r1 - 1] = -1;
  rec(r2 - 1, -1);
  F(i, n)
    if(i != r2 - 1)
      printf("%d ", a[i]);
}
