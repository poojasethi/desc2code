#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[50001], n, r1, r2, i;

void gotop(int x, int y)
{
	if (a[x]!=x) gotop(a[x], x);
	a[x]=y;
}

int main()
{
	cin>>n>>r1>>r2;
	for (i=1; i<=n; i++) if (i!=r1) cin>>a[i];
	a[r1]=r1;
	gotop(r2, a[r2]);
	for (i=1; i<=n; i++) if (i!=r2) cout<<a[i]<<" ";
	cout<<endl;
	return 0;
}
