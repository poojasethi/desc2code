#include <iostream>
#include <deque>
using namespace std;


int main()
{
	int n,r1,r2,i,a[60000],b[60000],c[60000],q,w;

	cin>>n>>r1>>r2;

	for(i=1;i<=n-1;i++)
	{
		cin>>a[i];
		if(i < r1)
		{
			b[i]=a[i];
		}
		else
		{
			b[i+1]=a[i];
		}
	}

	b[r1]=0;


	for(i=1;i<=n;i++)
		c[i]=b[i];

	q=b[r2];
	w=r2;
	while(q != r1)
	{
		c[q]=w;
		w=q;
		q=b[q];
	}
	c[q]=w;

	for(i=1;i<=n;i++)
	{
		if(i != r2)
			cout<<c[i]<<" ";
	}
	return 0;
}
