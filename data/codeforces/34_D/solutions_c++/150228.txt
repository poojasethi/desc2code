#include <iostream>
using namespace std;

int st[50001];
int pre[50001];
int old,ne;

void gao(int cur)
{
	int p=pre[cur];
	if(p!=old)
		gao(p);
	pre[p]=cur;
}
int main()
{
	int n;
	while(cin>>n>>old>>ne)
	{
		for(int i=1;i<=n;i++)
			if(i!=old)
				cin>>pre[i];
		if(old!=ne)
			gao(ne);
		for(int i=1;i<=n;i++)
			if(i!=ne)
				cout<<pre[i]<<" ";
		cout<<endl;
	}
	return 0;
}
