#include<iostream>
#include<vector>
using namespace std;
int n,r1,r2,d[50001];
vector<int> f[50001];
bool p=false;
void dfs(int a)
{
	if(a==r2)
	{
		p=true;
		return ;
	}
	int l=f[a].size();
	for(int i=0;i<l;i++)
	{
		dfs(f[a][i]);
		if(p)
		{
			d[a]=f[a][i];
			return ;
		}
	}
}
int main()
{
	cin>>n>>r1>>r2;
	for(int i=1;i<=n;i++)
	{
		if(i==r1)
			continue;
		cin>>d[i];
		f[d[i]].push_back(i);
	}
	dfs(r1);
	for(int i=1;i<=n;i++)
	{
		if(i==r2)
			continue;
		cout<<d[i]<<" ";
	}
	cout<<endl;
	return 0;
}
