#include <stdio.h>

int n, r1, r2;
int p[50005];
int main(void) {
	scanf("%d%d%d", &n, &r1, &r2);
	for(int i=1;i<=n;i++) {
		if(i==r1) {p[i] = r1;}
		else scanf("%d", &p[i]);
	}
	for(int from=r2, next=r2, i=r2; ; i=next) {
		next = p[i];
		p[i] = from;
		from = i;
		if(i == r1) break;
	}
	for(int i=1;i<=n;i++)
		if(i==r2) continue;
		else printf("%d ", p[i]);
	puts("");
	return 0;
}
