#include<bits/stdc++.h>

using namespace std;

vector<int> E[50005];
int A[50005];

void dfs(int v) {
	int i;
	for(i = 0; i<E[v].size(); ++i){
		if(A[E[v][i]] == -1) {
			A[E[v][i]] = v;
			dfs(E[v][i]);
		}
	}
}

int main() {
	int n, r1, r2, i, v;
	cin>>n>>r1>>r2;

	for(i = 1; i<=n; ++i) { 
		A[i] = -1;
		if(i != r1) {
			cin>>v;
			E[v].push_back(i);
			E[i].push_back(v);
		}
	}

	A[r2] = 0;
	dfs(r2);

	for(i = 1; i<=n; ++i)
		if(i != r2)
			cout<<A[i]<<' ';
	cout<<endl;	
}
