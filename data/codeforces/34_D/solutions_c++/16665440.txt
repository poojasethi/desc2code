#include<iostream>
#include<vector>
using namespace std;
int pm[50005]={0};
int s[50005]={0};
int n,r1,r2;
void mb(int hm)
{	
	if(pm[hm]!=r1)
		mb(pm[hm]);
	pm[pm[hm]]=hm;
}
int main()
{
	int x;
	cin>>n>>r1>>r2;
	for(int i=1;i<=n;i++)
	{
		if(i==r1) continue;
		cin>>x;
		pm[i]=x;
	}
	mb(r2);
	for(int i=1;i<=n;i++)
	{
		if(i==r2) continue;
		cout<<pm[i]<<' ';
	}
	return 0;
}
