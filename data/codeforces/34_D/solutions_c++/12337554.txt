#include<iostream>
#include<vector>
#include<queue>
using namespace std;
int main()
{
	int n,r1,r2;
	cin>>n>>r1>>r2;
	int p[n+1];
	for(int i=1;i<=n;i++)
	{
		if(i==r1)
		continue;
	 cin>>p[i];
    }
    queue<int>q;
    q.push(r2);
    int b=q.front();
    while(q.front()!=r1)
    {
    	q.push(p[q.front()]);
    	p[q.front()]=b;
    	b=q.front();
    	q.pop();
    }
    p[r1]=b;
    for(int i=1;i<=n;i++)
    {
    	if(i==r2)
    	continue;
    	cout<<p[i]<<" ";
    }
}
