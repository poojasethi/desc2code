#include<stdio.h>

int n,r1,r2,arr[100000],i,a;

void dfs(int x){
  if(x == r1) return;
  dfs(arr[x]);
  arr[arr[x]] = x;
}

int main(){

    scanf("%d %d %d",&n,&r1,&r2);
    for(i = 1; i <= n; i++)
        if(i != r1)
            scanf("%d",&arr[i]);

    dfs(r2);

    for(i = 1; i <= n; i++)
        if(i != r2)
            printf("%d ",arr[i]);

    return 0;
}
