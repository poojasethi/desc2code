#include<cstdio>
#include<cstdlib>

using namespace std;

int a[60000];
int p[60000],nw[60000];

void rotate(int x){
	if(p[x] != 1e9) {
		nw[p[x]] = x;
		rotate(p[x]);
	}
}

int main(){
	int n,r1,r2;
	for(int i=0;i<60000;i++) p[i] = 1e9;
	scanf("%d%d%d", &n,&r1,&r2);
	for(int i=1;i<=n;i++){
		if(i==r1) continue;
		scanf("%d", &p[i]);
		nw[i] = p[i];
	}
	rotate(r2);
	nw[r2] = 1e9;
	for(int i=1;i<=n;i++){
		if(i==r2) continue;
		printf("%d ", nw[i]);
	}
	return 0;
}
