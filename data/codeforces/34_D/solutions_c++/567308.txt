#include <cstdio>
using namespace std;

int n,s,t,a[50001];

void gao(int x){
    if(x!=s) gao(a[x]);
    a[a[x]]=x;
}

int main(){
    scanf("%d%d%d",&n,&s,&t);
    for(int i=1;i<=n;i++)
        if(i!=s) scanf("%d",&a[i]);
    gao(t);
    for(int i=1;i<=n;i++)
        if(i!=t) printf("%d ",a[i]);
    puts("");
}
