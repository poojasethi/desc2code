#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int n,r1,r2;
int prev[100000];
int main()
{
	//freopen("inp","r",stdin);
	scanf("%d%d%d",&n,&r1,&r2);
	for(int i=1;i<=n;i++)
	{
		if(i==r1) continue;
		int a;
		scanf("%d",&a);
		prev[i]=a;
	}
	int pr=0,ne=0;
	for(int i=r2;i!=r1;i=pr)
	{
		pr=prev[i];
		prev[i]=ne;	
		ne=i;		
	}
		pr=prev[r1];
		prev[r1]=ne;	
		ne=r1;	
	for(int i=1;i<=n;i++)
	{
		if(i==r2) continue;
		printf("%d ",prev[i]);
	}
	return 0;	
}
