#include<iostream>
using namespace std;

int main(){
	int n,r1,r2,i,j,a[50005];
	cin>>n>>r1>>r2;
	a[r1]=0;
	for(i=1;i<=n;i++){
		if(i==r1) continue;
		cin>>a[i];
	}
	int x=a[r2],y=r2;
	a[r2]=0;
	while(x!=0){
		int temp=a[x];
		a[x]=y;
		y=x;
		x=temp;
	}
	for(i=1,j=n-1;i<=n;i++){
		if(i==r2) continue;
		cout<<a[i];
		if(--j) cout<<" ";
		else cout<<endl;
	}
	return 0;
}
