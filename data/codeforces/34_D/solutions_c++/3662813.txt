#include <iostream>
using namespace std;

int n, r1, r2, p[(int)5e4 + 2], b[(int)5e4 + 2];

int main() {
	ios::sync_with_stdio(false);
	cin >> n >> r1 >> r2;
	for (int i = 1; i <= n; i++)
		if (i != r1)
			cin >> p[i];
	int u = r2;
	while (u != r1) {
		int v = p[u];
		b[v] = u;
		u = v;
	}
	for (int i = 1; i <= n; i++)
		if (i != r2)
			if (b[i])
				cout << b[i] << ' ';
			else
				cout << p[i] << ' ';
	cout << '\n';
	return 0;
}
