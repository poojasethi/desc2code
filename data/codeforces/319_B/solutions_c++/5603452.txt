#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#define INF 1000000
using namespace std;
int main(){
	int n,stk[100006][2],p=0,in,cnt,ans=0;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&in);
		cnt=1;
		while(p>0){
			if(stk[p-1][0]>in) break;
			p--;
			cnt=max(cnt,stk[p][1]+1);
		}
		if(p==0) cnt=INF;
		if(cnt<INF) ans=max(ans,cnt);
		stk[p][0]=in; stk[p++][1]=cnt;
	}
	printf("%d\n",ans);
	return 0;
}
