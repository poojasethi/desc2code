#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int n,ans,t,f[100010],a[100010],q[100010];
int main(){
    scanf("%d",&n);
    for(int i=n;i>=1;--i)scanf("%d",&a[i]);
    for(int i=1;i<=n;++i){
        for(;t&&a[q[t]]<a[i];--t)f[i]=max(f[i]+1,f[q[t]]);
        q[++t]=i;
        ans=max(ans,f[i]);
    }
    printf("%d\n",ans);
    return 0;
}
