#include<stdio.h>
#include<string.h>
#include<math.h>
#include<iostream>
#include<algorithm>
#include<vector>
#define M 100005
using namespace std;
int a[M],st[M],b[M];
int main()
{
	int i,n,t,ans,k;
	while(~scanf("%d",&n))
	{
		for(i=0;i<n;i++)
		{
			scanf("%d",&a[n-i-1]);
			b[i]=0;
		}
		k=ans=0;
		for(i=0;i<n;i++)
		{
			while(k&&a[st[k-1]]<a[i])
			{
				b[i]=max(b[i]+1,b[st[k-1]]);
				k--;
			}
			st[k++]=i;
			ans=max(b[i],ans);
		}
		printf("%d\n",ans);
	}
}
