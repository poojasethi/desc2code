#include <iostream>
#include <algorithm>

using namespace std;

int n;
int a[1000001];
int x[1000001];
int f[1000001];

int main() {
	cin >> n;
	for (int i = 1; i <= n; i++)
		cin >> a[i];
	
	int last = 1, ans = 0;
	x[last] = 1;
	for (int i = 2; i <= n; i++) {
		f[i] = 1;
		while (last && a[x[last]] < a[i]) {
			f[i] = max(f[i], f[x[last]] + 1);
			last--;
		}
		if (!last)
			f[i] = 0;
		last++;
		x[last] = i;
		if (ans < f[i])
			ans = f[i];
	}
	cout << ans << endl;
	return 0;
}
