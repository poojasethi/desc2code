#include<stdio.h>
int stack[100005],a[100005],time[100005];
int main()
{
	int n,i,j,sum;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	scanf("%d",&a[i]);
	int top=0;
	int ans=0;
	for(i=n;i>=1;i--)
	{
		while(top&&a[stack[top-1]]<a[i])
		{
			top--;
			time[i]=time[i]+1>time[stack[top]]?time[i]+1:time[stack[top]];
		}
		stack[top++]=i;
		ans=ans>time[i]?ans:time[i];
	}
	printf("%d\n",ans);
}
