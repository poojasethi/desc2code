#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,count=0;
    cin>>n;
    int a[n];
    int b[n];
    stack<int> S;
    for(int i=0; i<n; i++)  
    {	
    	cin>>a[i];
    }	
    for(int i=n-1; i>=0; i--)
    {
        b[i]=0;
        while(!S.empty()  && a[i]>a[S.top()])
        {
    		b[i]=max(b[i]+1,b[S.top()]);
            S.pop();
        }	
        S.push(i);
        count=max(count,b[i]);
    }

   cout<<count<<endl;
	return 0;
}
