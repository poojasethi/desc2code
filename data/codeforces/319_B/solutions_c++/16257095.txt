#include <bits/stdc++.h>
using namespace std;
int a[100005];
int st[100005];
int cnt[100005];
int top;
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	for(int i=n-1;i>=0;i--)
	{
		int t = 0;
		while(top && a[st[top-1]] < a[i])
			cnt[i] = t = max(t+1,cnt[st[--top]]);
		st[top++]=i;
	}
	cout<<*max_element(cnt,cnt+n);
	return 0;
}
