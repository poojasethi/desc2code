#include <stdio.h>
int max(int a,int b){
	return a>b?a:b;
}
int main(){
	int a[100005],line[100005],map[100005],n,m;
	scanf("%d",&n);
	m=0;
	int ans=0;
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=n;i>=1;i--){
		line[m+1]=a[i];
		map[m+1]=0;
		m++;
	
		int x=0;

		while(m>1&&line[m]>line[m-1]){
			line[m-1]=line[m];
			map[m-1]=max(map[m-1],map[m]+1);
			m--;
			ans=max(ans,map[m]);
			
		}
		if(ans<x)ans=x;	
	}
	printf("%d\n",ans);
	return 0;
	
}
