#include<stdio.h>
#define MAXN 1000000007
int n,i,maxn,t,f[100005],h[100005],a[100005];
int max(int x,int y){	
	return x>y?x:y;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;i++)scanf("%d",&a[i]);
	f[n]=0;h[n]=n+1;a[n+1]=MAXN;maxn=0;
	for(i=n-1;i>=1;i--){
		t=i+1;
		while(a[i]>a[t]){
			f[i]=max(f[i]+1,f[t]);
			t=h[t];
		}
		h[i]=t;
		maxn=max(f[i],maxn);
	}
	printf("%d\n",maxn);
	
}
