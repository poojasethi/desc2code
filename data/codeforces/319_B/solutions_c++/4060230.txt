#include<iostream>
#include<stack>
using namespace std;
int n,a[100010],k,i,s;
bool m[100010];
stack<int> t;
int main (){
    cin>>n;
    for(;i<n;i++){
        cin>>a[i];
        if(a[i]>k){
            k=a[i];
            m[i]=true;
        }
    }
    for(i--;i>=0;--i)
        if(!m[i]){
            while(t.size()&&t.top()<a[i])t.pop();
            t.push(a[i]);
            s=max(s,(int)t.size());
        }else
            while(t.size())t.pop();
    cout<<s;
}
