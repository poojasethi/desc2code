#include <iostream>
using namespace std;
const int MAX = 100005;
int a[MAX], st[MAX], val[MAX];
int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	int t = 0, ans = 0;
	for (int i = 0; i < n; i++)
	{
		cin >> a[i];
		while (t > 0 && a[st[t - 1]] < a[i])
		{
			val[i] = max(val[i], val[st[t - 1]]);
			t--;
		}
		if (t > 0)
			ans = max(ans, ++val[i]);
		else
			val[i] = 1e9;
		st[t++] = i;
	}
	cout << ans << endl;
	return 0;
}
