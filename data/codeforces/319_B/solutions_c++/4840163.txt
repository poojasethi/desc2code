#include <iostream>
#include <stack>
#include <algorithm>
using namespace std;

#define MAXN 100005
int n, ans;
stack <pair <int, int > > stk;
int a[MAXN];

int main() {
	cin >> n;
	for (int i = 0; i < n; i ++) cin >> a[i];
	for (int i = n-1; i >= 0; i --) {
		int t = 0;
		while (!stk.empty() && stk.top().first < a[i]) {
			t = max(t+1, stk.top().second);
			stk.pop();
		}
		ans = max(t, ans);
		stk.push(make_pair (a[i], t));
	}
	cout << ans << endl;
}
