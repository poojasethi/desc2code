#include <stdio.h>
int a[100010],b[100010],st[100010];
int main()
{
	int ans,n,i,top;
		scanf("%d",&n);
		ans=0;
		for(i=1;i<=n;i++)
		{
			scanf("%d",a+i);
			b[i]=0;
		}
		a[0]=0;
		for(top=0,i=n;i>0;i--)
		{
			while(top!=0&&a[i]>a[st[top-1]])
			{
				b[i]++;
				if(b[i]<b[st[top-1]])
				b[i]=b[st[top-1]];
				top--;
			}
			st[top++]=i;
		}
		for(i=1;i<=n;i++)
		{
			if(ans<b[i])
			ans=b[i];
		}
		printf("%d\n",ans);
}
