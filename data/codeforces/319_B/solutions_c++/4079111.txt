#include <stdio.h>
int a[100010],b[100010],st[100010];
int main()
{
	int ans,n,i,top;
	while(scanf("%d",&n)!=EOF)
	{
		ans=0;
		for(i=1;i<=n;i++)
		{
			scanf("%d",a+i);
			b[i]=0;
		}
		for(top=0,i=n-1;i>0;i--)
		{
			top++;
			st[top]=i+1;
			while(top!=0&&a[i]>a[st[top]])
			{
				b[i]++;
				if(b[i]<b[st[top]])
				b[i]=b[st[top]];
				top--;
			}
		}
		for(i=1;i<=n;i++)
		{
			if(ans<b[i])
			ans=b[i];
		}
		printf("%d\n",ans);
	}
}
