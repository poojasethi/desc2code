#include <stdio.h>
int x[3],y[3];
int main(){
    int n,i;
    scanf("%d",&n);

    int t,xt,yt;
    for(i=0;i<n;i++){
    scanf("%d%d%d",&t,&xt,&yt);

    x[t]+=xt;
    y[t]+=yt;
    }

    for(i=1;i<3;i++){
    printf("%s\n",x[i]>=y[i]?"LIVE":"DEAD");
    }

    return 0;
    
}
