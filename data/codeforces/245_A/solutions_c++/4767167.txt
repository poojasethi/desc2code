#include <stdio.h>

int n;
int x[2], y[2];

int main()
{
  scanf("%d", &n);
  for (int i = 1; i <= n; ++i) {
    int ti, xi, yi;
    scanf("%d%d%d", &ti, &xi, &yi);
    x[ti!=1] += xi;
    y[ti!=1] += yi;
  }
  printf("%s\n%s\n", x[0]>=y[0]?"LIVE":"DEAD", x[1]>=y[1]?"LIVE":"DEAD" );
  
	return 0;
}
