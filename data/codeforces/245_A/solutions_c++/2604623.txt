#include <iostream>
#include <cstdio>

using namespace std;

int main(){
	int n,t,x,y;
	while(~scanf("%d",&n)){
		int s[3]={0},f[3]={0};
		for(int i=0;i<n;i++){
			scanf("%d%d%d",&t,&x,&y);
			s[t]+=x; f[t]+=y;
		}
		printf("%s\n",s[1]>=f[1]?"LIVE":"DEAD");
		printf("%s\n",s[2]>=f[2]?"LIVE":"DEAD");
	}
	return 0;
}
