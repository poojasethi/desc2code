#include <cstdio>

int main()
{
	int n, a = 0, b = 0;
	scanf("%d", &n);
	
	while (n--)
	{
		int z, x, y;
		scanf("%d%d%d", &z, &x, &y);
		if (z == 1) a += (x-y);
		else b += (x-y);
	}
	
	if (a >= 0) printf("LIVE\n");
	else printf("DEAD\n");
	if (b >= 0) printf("LIVE\n");
	else printf("DEAD\n");
	
	return 0;
}
