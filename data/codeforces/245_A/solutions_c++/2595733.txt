#include <iostream>
#include <fstream>

using namespace std;

int N, A[2];

int main(){
    A[0] = A[1] = 0;
    cin >> N;
    for(int i=0; i<N; i++){
	int a,b,c; cin >> a >> b >> c;
	A[a-1] += b-c;
    }
    for(int i=0; i<2; i++){
	if (A[i] >= 0){
	    cout << "LIVE\n";
	}else{
	    cout << "DEAD\n";
	}
    }
}
