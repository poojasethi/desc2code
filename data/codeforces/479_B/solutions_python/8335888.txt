
n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
r = []

steps = 0
while(steps < k and max(a) - min(a) != 0):
	maxindex, minindex = 0, 0
	for i in range(0, n):
		if(a[i] >= a[maxindex]):
			maxindex = i
		if(a[i] <= a[minindex]):
			minindex = i
	a[minindex] += 1
	a[maxindex] -= 1
	r += [(maxindex + 1, minindex + 1)]
	steps += 1
print max(a) - min(a), steps
for i in r:
	print i[0], i[1]
