import sys
n,k=map(int,sys.stdin.readline().strip('\n').split())
m=map(int,sys.stdin.readline().strip('\n').split())
def maxindex(m):
	t=max(m)
	return m.index(t)
def minindex(m):
	t=min(m)
	return m.index(t)
s=max(m)-min(m)
procedure=[]
path=0
while path<k and s>1:
	a=maxindex(m)
	b=minindex(m)
	m[a]-=1
	m[b]+=1
	procedure.append([a,b])
	path+=1
	s=max(m)-min(m)
print s,path
for x in procedure:
	print x[0]+1,x[1]+1
