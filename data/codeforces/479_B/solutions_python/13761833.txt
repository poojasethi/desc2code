n,k=map(int,raw_input().split())
l=map(int,raw_input().split())
p=[]
c=0
for i in range(k):
	maxi=max(l)
	mini=min(l)
	maxi_index=l.index(maxi)
	mini_index=l.index(mini)
	if(maxi_index!=mini_index and maxi!=0):
		p.append([maxi_index+1,mini_index+1])
		l[maxi_index]-=1
		l[mini_index]+=1
		c+=1
	else:
		break
print max(l)-min(l),c
for i in p:
	print i[0],i[1]
