
n, k = map(int, raw_input().split())
a = map(int, raw_input().split())
r = []

steps = 0
while(steps < k and max(a) - min(a) != 0):
	ma, mi = a.index(max(a)), a.index(min(a))
	a[mi] += 1
	a[ma] -= 1
	r += [(ma + 1, mi + 1)]
	steps += 1
print max(a) - min(a), steps
for i in r:
	print i[0], i[1]
