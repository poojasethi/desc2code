n, k = [int(x) for x in raw_input().split()]


a = [int(x) for x in raw_input().split()]

j = 1

for i in a:
	a[j-1] = [j, i]
	j+=1

a.sort(key=lambda x: x[1], reverse=True)

r = []	
i = 0
while (i<k and a[0][1]-a[-1][1]>1):
	a[0][1]-=1
	a[-1][1]+=1
	r.append([a[0][0], a[-1][0]])
	a.sort(key=lambda x: x[1], reverse=True)
	i+=1
