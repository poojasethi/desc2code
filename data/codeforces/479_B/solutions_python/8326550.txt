R=lambda:map(int,raw_input().split())
n,k=R()
a=zip(R(),range(n))
o=[]
while k:
  x,y=a.index(max(a)),a.index(min(a))
  d=a[x][0]-a[y][0]
  if d:
    a[x]=(a[x][0]-1,a[x][1])
    a[y]=(a[y][0]+1,a[y][1])
    o+=[(x,y)]
  else:
    break
  k-=1
print max(a)[0]-min(a)[0],len(o)
for x,y in o:
  print x+1,y+1
