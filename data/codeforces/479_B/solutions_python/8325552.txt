n,k = map(int,raw_input().split())
l = map(int,raw_input().split())
ans = []
i = 0
while i<k:
    if(max(l) - min(l)>0):
        a,b = l.index(max(l)),l.index(min(l))
        l[a]-=1
        l[b]+=1
        i+=1
        ans.append([a+1,b+1])
    else:
        break
print max(l)-min(l),len(ans)
for i in ans:
    print i[0],i[1]
