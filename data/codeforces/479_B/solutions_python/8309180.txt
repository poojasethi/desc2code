n,k = map(int,raw_input().split())
a = map(int,raw_input().split())
ops = []
b = [(a[i],i) for i in xrange(n)]
while k > 0:
    b.sort()
    if b[n-1][0]-b[0][0] == 0:
        break
    else:
        b[n-1]=(b[n-1][0]-1,b[n-1][1])
        b[0]=(b[0][0]+1,b[0][1])
        ops.append([b[n-1][1],b[0][1]])
    k-=1
b.sort()
print b[n-1][0]-b[0][0],len(ops)
for i in range(len(ops)):
    print ops[i][0]+1,ops[i][1]+1
