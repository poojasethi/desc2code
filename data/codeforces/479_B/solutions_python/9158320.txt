n, pb = map(int, raw_input().split())
data = map(int, raw_input().split())

p = 0
ans = []
while p<pb:
    ma = max(data)
    mi = min(data)
    if ma-mi<2:
        break
    mad = data.index(ma)
    mid = data.index(mi)
    data[mad] -= 1
    data[mid] += 1
    ans.append((mad+1, mid+1))
    p+=1
print max(data)-min(data), p
for a1, a2 in ans:
    print a1, a2
