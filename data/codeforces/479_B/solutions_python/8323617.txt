n,k = map(int,raw_input().split())
a = map(int,raw_input().split())
arr = sorted([list(x) for x in enumerate(a)],key = lambda x:x[1])
m = 0
ans = []
while min(arr,key=lambda x:x[1])[1] + 1 < max(arr,key = lambda x:x[1])[1] and m < k:
    m += 1
    arr = sorted(arr,key=lambda x:x[1])
    ans.append((arr[-1][0]+1,arr[0][0]+1))
    arr[-1][1] -= 1
    arr[0][1] += 1
arr = sorted(arr,key=lambda x:x[1])
print arr[-1][1] - arr[0][1], m
for i,j in ans:
    print i,j
