n, k = raw_input().split()
n, k = int(n), int(k)

a = []
i = 1
for x in raw_input().split():
    a.append([int(x),i])
    i += 1

dem = 0

res = []
for i in range(k):
    a.sort()
    if a[-1][0] - a[0][0] >= 2:
        a[-1][0] -= 1
        a[0][0] += 1
        dem += 1
        res.append((a[-1][1],a[0][1]))
    else:
        break

a.sort()
print a[-1][0] - a[0][0], dem
for i in range(dem):
    print res[i][0], res[i][1]
