#include<bits/stdc++.h>
using namespace std ;
int main() 
{
	int n,k;
	cin>>n>>k ;
	vector<pair<int,int> >arr(n),ans;
	for(int i=0;i<n;i++) 
	{
		cin>>arr[i].first;
		arr[i].second=i+1;
	}
	sort(arr.begin(),arr.end());
	for(int i=0;i<k;i++) 
	{
		if(arr[0].first==arr[n-1].first)
			break;
		arr[0].first++;
		arr[n-1].first--;
		ans.push_back(make_pair(arr[n-1].second,arr[0].second)) ;
		sort(arr.begin(),arr.end());
	}
	cout<<arr[n-1].first-arr[0].first<< " " <<ans.size()<<endl ;
	for(int i=0;i<ans.size();i++)
		cout<<ans[i].first<<" "<<ans[i].second<<endl ;
}
