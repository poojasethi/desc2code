#include<cstdio>
#include<vector>
#include<algorithm>
using namespace std;
struct POINT{
	int num,tow;
};
int cmp(POINT a,POINT b){
	return a.tow<b.tow;
}
int main(){
	int n,k;
	vector<int>q;
	POINT a[110];
	scanf("%d %d",&n,&k);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i].tow);
		a[i].num=i+1;
	}
	while(k--){
		sort(a,a+n,cmp);
		if(a[n-1].tow==a[0].tow)break;
		a[n-1].tow--;
		a[0].tow++;
		q.push_back(a[n-1].num);
		q.push_back(a[0].num);
	}
	sort(a,a+n,cmp);
	printf("%d %d\n",a[n-1].tow-a[0].tow,q.size()/2);
	for(int i=0;i<q.size();++i,i++)
		printf("%d %d\n",q[i],q[i+1]);
}
