#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main(){
    int n,m,a,b,i,j,t=0,s;
    pair<int,int> x[101];
    pair<int,int> y[1001];
    cin>>n>>m;
    for(i=0;i<n;++i){
        cin>>x[i].first;
        x[i].second=i+1;}
    sort(x,x+n);
    while(x[n-1].first-x[0].first>1&&t<m){
        y[t].first=x[n-1].second;
        y[t].second=x[0].second;
        x[n-1].first--;
        x[0].first++;
        sort(x,x+n);
        t++;}
    cout<<x[n-1].first-x[0].first<<" "<<t<<endl;
    for(i=0;i<t;++i)
        cout<<y[i].first<<" "<<y[i].second<<endl;
    return 0;
}
