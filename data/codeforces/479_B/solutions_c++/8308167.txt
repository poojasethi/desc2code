#include<iostream>
using namespace std;
int n,k,x[101],i,j,st,dr,m;
struct {int a,b;} schimb[1001];
int main()
{
	cin>>n>>k;
	for(i=1;i<=n;i++) cin>>x[i]; 
	while(k)
	{
		st=n; dr=1;
		for(i=1;i<=n;i++)  
		{
			if(x[i]<x[st]) st=i; 
			if(x[i]>x[dr]) dr=i;
		}
		if(x[dr]-x[st]<=1) break;
		else 
		{ 
			schimb[++m].a=dr; 
			schimb[m].b=st; 
			x[st]++; 
			x[dr]--;
			k--;
		}
	}
	for(i=1;i<=n;i++)  
	{
		if(x[i]<x[st]) st=i; 
		if(x[i]>x[dr]) dr=i;
	}
	cout<<x[dr]-x[st]<<' '<<m<<'\n';
	for(i=1;i<=m;i++)
		cout<<schimb[i].a<<' '<<schimb[i].b<<'\n';
	return 0;
}
