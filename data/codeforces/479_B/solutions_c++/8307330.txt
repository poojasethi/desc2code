#include <iostream>
#include <algorithm>
using namespace std;
int n, k, a[101], i, j, ans[2000], m, pM, pN;
int main()
{
	cin>>n>>k;
	for (i=1; i<=n; i++) cin>>a[i];
	while (k--) {
		pM=pN=1;
		for (i=2; i<=n; i++)
			if (a[i]>a[pM]) pM=i;
			else if (a[i]<a[pN]) pN=i;
		if (a[pM]-a[pN]>1) {
			a[pM]--; a[pN]++; ans[m++]=pM; ans[m++]=pN;
		}
		else break;
	}
	pM=pN=1;
	for (i=2; i<=n; i++)
		if (a[i]>a[pM]) pM=i;
		else if (a[i]<a[pN]) pN=i;
	cout<<a[pM]-a[pN]<<" "<<m/2<<endl;
	for (i=0; i<m; i+=2) cout<<ans[i]<<" "<<ans[i+1]<<endl;
	return 0;
}
