#include <iostream>
#include <algorithm>
#include <utility>
#include <vector>
using namespace std;
typedef pair<int,int> pii;
vector<pii > v; 
int a[105];
int main() {
	int n,k;
	cin>>n>>k;
	for(int i=0;i<n;i++) cin>>a[i];
	for(int i=0;i<k;i++) {
		int p=max_element(a,a+n)-a;
		int q=min_element(a,a+n)-a;
		if(a[p]>a[q]) {
			v.push_back(make_pair(p+1,q+1));
			a[p]--,a[q]++;
		}
		else break;
	}
	int s=*max_element(a,a+n)-*min_element(a,a+n);
	cout<<s<<' '<<v.size()<<endl;
	for(int i=0;i<v.size();i++) cout<<v[i].first<<' '<<v[i].second<<endl;
	return 0;
}
