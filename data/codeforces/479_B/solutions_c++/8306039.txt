#include<iostream>
using namespace std;
int main()
{
	int n,m;
	cin>>n>>m;
	int a[n],maxa=0,mina=0,now=0;
	int ans1[m],ans2[m],ansl=0;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		if(a[i]>a[maxa])maxa=i;
		if(a[i]<a[mina])mina=i;
	}
	while(a[maxa]-a[mina]>1&&ansl<m)
	{
		ans1[ansl]=maxa;
		ans2[ansl]=mina;
		a[maxa]--;
		a[mina]++;
		ansl++;
		for(int i=0;i<n;i++)
		{
			if(a[i]>a[maxa])maxa=i;
			if(a[i]<a[mina])mina=i;
		}
	}
	cout<<a[maxa]-a[mina]<<" "<<ansl<<endl;
	for(int i=0;i<ansl;i++)cout<<ans1[i]+1<<" "<<ans2[i]+1<<endl;
}
