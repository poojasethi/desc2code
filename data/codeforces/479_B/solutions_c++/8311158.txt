#include<iostream>
using namespace std;
int main(){
	int n, k, i, a[100], x=0, y=0, ans=0, b[1001], c[1001];
	cin >> n >> k;
	for(i=0; i<n; i++){
		cin >> a[i];
	}
	x=y=0;
	for(i=1; i<n; i++){
		if(a[i]>a[x]){
			x=i;
		}
		if(a[i]<a[y]){
			y=i;
		}
	}
	while(a[x]-a[y]>=2 && k--){
		b[ans]=x+1;
		c[ans]=y+1;
		ans++;
		a[x]--;
		a[y]++;
		x=y=0;
		for(i=1; i<n; i++){
			if(a[i]>a[x]){
				x=i;
			}
			if(a[i]<a[y]){
				y=i;
			}
		}
	}
	cout << a[x]-a[y] << " " << ans << endl;
	for(i=0; i<ans; i++){
		cout << b[i] << " " << c[i] << endl;
	}
	return 0;
}
