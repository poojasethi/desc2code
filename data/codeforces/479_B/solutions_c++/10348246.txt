#include<cstdio>
#include<algorithm>
using namespace std;
struct stt{int a,i;}st[100];
bool cmp(stt st1,stt st2)
{return st1.a<st2.a;}
int o[1000][2];
int main()
{
	int n,k,a,b,c,d,e,f,g;
	scanf("%d%d",&n,&k);
	for(int i=0;i<n;++i)
	{scanf("%d",&st[i].a);st[i].i=i+1;}
	sort(st,st+n,cmp);
	for(a=0;a<k&&st[n-1].a-st[0].a>1;++a)
	{
		b=0;while(st[b+1].a==st[0].a)++b;
		c=n-1;while(st[c-1].a==st[n-1].a)--c;
		++st[b].a,--st[c].a;
		o[a][0]=st[b].i,o[a][1]=st[c].i;
	}
	printf("%d %d\n",st[n-1].a-st[0].a,a);
	for(int i=0;i<a;++i)
		printf("%d %d\n",o[i][1],o[i][0]);
}
