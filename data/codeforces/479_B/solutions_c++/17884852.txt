#include<bits/stdc++.h>
using namespace std;
int main(){
int n,k;cin>>n>>k;
vector<pair<int,int> >V(n);
for(int i=0;i<n;i++){
    cin>>V[i].first;
    V[i].second=i;
}
sort(V.begin(),V.end());
vector<pair<int,int> >Ops;
while(k--){
    if(V[0].first>=V[n-1].first-1)
        break;
    V[0].first++;
    V[n-1].first--;
    Ops.push_back(make_pair(V[n-1].second,V[0].second));
    sort(V.begin(),V.end());
}
cout<<V[n-1].first-V[0].first<<" "<<Ops.size()<<endl;
for(int i=0;i<Ops.size();i++){
    cout<<Ops[i].first+1<<" "<<Ops[i].second+1<<endl;
}
}
