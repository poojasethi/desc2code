#include<iostream>
#include<algorithm>	
int x[1001],y[1001];
using namespace std;
struct node{
	int h,id;
}s[101];
bool cmp(node a,node b)
{
	return a.h>b.h;
}
int main()
{
	int n,k,i,c;
	cin>>n>>k;
	for(i=1;i<=n;i++)
	{
		cin>>s[i].h;
		s[i].id=i;
	}
	for(i=1;i<=k;i++)
	{
		sort(s+1,s+n+1,cmp);
		if(s[1].h==s[n].h) 
		{
			k=i-1;
			break; 
		}
		s[1].h--;
		s[n].h++;
		x[i]=s[1].id;
		y[i]=s[n].id;
	}
	sort(s+1,s+n+1,cmp);
	cout<<s[1].h-s[n].h<<" "<<k<<endl;
	for(i=1;i<=k;i++)
		cout<<x[i]<<" "<<y[i]<<endl;
	return 0;
}
