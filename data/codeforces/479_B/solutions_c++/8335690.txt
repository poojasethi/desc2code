#include <iostream>
#include <algorithm>
using namespace::std;
pair<int,int> t[101],o[1001];

int main()
{
	int i,j,n,k;
	cin >> n >> k;
	for(i=0; i<n; i++)
	{
		cin >> t[i].first;
		t[i].second=i;
	}
	sort(t,t+n);
	for(i=0; i<k; i++)
	{
		if( (t[0].first == t[n-1].first) || ((t[0].first+1 == t[n-1].first)
		&& (t[n-1].first-1 == t[0].first)) ) break;
		t[n-1].first--;
		t[0].first++;
		o[i].first=t[n-1].second+1;
		o[i].second=t[0].second+1;
		sort(t,t+n);
	}
	cout << t[n-1].first-t[0].first << " " << i << "\n";
	for(j=0; j<i; j++) cout << o[j].first << " " <<o[j].second << "\n";
}
