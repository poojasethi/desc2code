#include<cstdio>
#include<algorithm>
using namespace std;

int n,m,k;
int a[105],ans1[1005],ans2[1005];
int MAX,MIN;

int main()
{
	while(~scanf("%d%d",&n,&m))
	{
		for(int i=0;i<n;i++)
		{
			scanf("%d",&a[i]);
		}
		k=0;
		while(k<m)
		{
			MAX=max_element(a,a+n)-a;
			MIN=min_element(a,a+n)-a;
			if((a[MAX]-a[MIN])<=1) break;
			a[MAX]--;
			a[MIN]++;
			ans1[k]=MAX;
			ans2[k]=MIN;
			k++;
		}
		MAX=max_element(a,a+n)-a;
		MIN=min_element(a,a+n)-a;
		printf("%d %d\n",a[MAX]-a[MIN],k);
		for(int i=0;i<k;i++)
		{
			printf("%d %d\n",ans1[i]+1,ans2[i]+1);
		}
	}
	return 0;
}
