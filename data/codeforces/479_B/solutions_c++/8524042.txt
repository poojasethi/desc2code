#include<iostream>
#include<algorithm>

using namespace std;

int ar[1000];
int x[100000],y[100000];


int main(){

    int n,k,p=0;
    cin>>n>>k;
    for(int c=0;c<n;c++)cin>>ar[c];

    while(*max_element(ar,ar+n)-*min_element(ar,ar+n)>1&&p<k){
        x[p]=max_element(ar,ar+n)-ar+1;
        y[p]=min_element(ar,ar+n)-ar+1;
        ar[max_element(ar,ar+n)-ar]--;
        ar[min_element(ar,ar+n)-ar]++;
        p++;
    }
    cout<<*max_element(ar,ar+n)-*min_element(ar,ar+n)<<" "<<p<<endl;
    for(int c=0;c<p;c++)cout<<x[c]<<" "<<y[c]<<endl;


}
