#include<stdio.h>
int main()
{
	int a[100],n,m,k,min,max,l=0;
	struct{int from,to;} b[1000];
	scanf("%i %i",&n,&m);
	for(int i=0;i<n;i++) scanf("%i",&a[i]);
	k=m;
	while(k--)
	{
		max=0;
		min=0;
		for(int i=1;i<n;i++)
		{
			if(a[i]<a[min]) min=i;
			if(a[i]>a[max]) max=i;
		}
		if(a[max]-a[min]<2) break;
		b[l++]={max,min};
		a[max]--;a[min]++;
	}
	max=0;
	min=0;
	for(int i=1;i<n;i++)
	{
		if(a[i]<a[min]) min=i;
		if(a[i]>a[max]) max=i;
	}
	printf("%i %i\n",a[max]-a[min],m-k-1);
	for(int i=0;i<l;i++) printf("%i %i\n",b[i].from+1,b[i].to+1);
}
