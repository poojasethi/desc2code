#include<cstdio>

int n, m, v[1111], a[1111], b[1111], tot;

void record(int x, int y) {
	a[tot] = x+1;
	b[tot] = y+1;
	tot++;
}

int main() {
	scanf("%d%d", &n, &m);
	for(int i=0;i<n;i++) scanf("%d", &v[i]);
	
	tot=0;
	for(int i=0;i<m;i++) {
		int x=0,y=0;
		for(int j=0;j<n;j++){
			if(v[x]>v[j])x=j;
			if(v[y]<v[j])y=j;
		}
		if(v[x]>=v[y]-1) break;
		record(y, x);
		v[x]++;v[y]--;
	}
	
	int x=0,y=0;
	for(int j=0;j<n;j++){
		if(v[x]>v[j])x=j;
		if(v[y]<v[j])y=j;
	}
	printf("%d %d\n", v[y]-v[x], tot);
	for(int i=0;i<tot;i++)
		printf("%d %d\n", a[i], b[i]);

	return 0;
}
