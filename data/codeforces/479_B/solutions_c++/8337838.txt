#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int n,k,cnt=0;
    cin>>n>>k;
    int a[n];
    int x[k],y[k];
    for(int i=0;i<n;i++)
    cin>>a[i];
    while( *max_element(a,a+n)-*min_element(a,a+n)>1 && cnt<k)
    {
        x[cnt]=max_element(a,a+n)-a+1;
        y[cnt]=min_element(a,a+n)-a+1;
        a[max_element(a,a+n)-a]--;
        a[min_element(a,a+n)-a]++;
        cnt++;
    }
    cout<<*max_element(a,a+n)-*min_element(a,a+n)<<" "<<cnt<<endl;
    for(int i=0;i<cnt;i++)
    cout<<x[i]<<" "<<y[i]<<endl;
    return 0;
}
