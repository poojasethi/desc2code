#include<iostream>
#include<algorithm>
#include<stdio.h>
using namespace std;
int n , k , a[105] , x , r1[1005], r2[1005];
int main()
{
    cin>>n>>k;
    for(int i=1; i<=n; i++)
        cin>>a[i];
    while(*max_element(a+1,a+n+1)- * min_element(a+1,a+n+1)>1 && x<k)
    {
        x++;
        r1[x]=max_element(a+1,a+n+1)-a;
        r2[x]=min_element(a+1,a+n+1)-a;
        a[r1[x]]--;
        a[r2[x]]++;
    }
    cout<<*max_element(a+1,a+n+1)- * min_element(a+1,a+n+1)<<" "<<x<<endl;
    for(int i=1; i<=x; i++)
        cout<<r1[i]<<" "<<r2[i]<<endl;

}
