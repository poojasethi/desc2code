#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int n,k,a[105];
vector<pair<int, int> > ans;

int main()
{
    cin>>n>>k;
    for (int i=0;i<n;++i)
        cin>>a[i];
    while ( k-- )
    {
        int is = min_element(a, a + n) - a;
        int ib = max_element(a, a + n) - a;
        if (is == ib)
            break;
        a[is]++;
        a[ib]--;
        ans.push_back(make_pair(ib+1,is+1));
    }
    sort(a,a+n);
    cout<<a[n-1]-a[0]<<' '<<ans.size()<<'\n';
    for (int i=0;i<ans.size();++i)
        cout<<ans[i].first<< ' ' <<ans[i].second<<endl;
}
