#include<stdlib.h>
#include<stdio.h>
#include<math.h>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n,k,m,s[1001][2];
struct DeaphetS{int a,b;}_[101];
bool cmp(DeaphetS a,DeaphetS b){return a.a<b.a;}
int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
	  scanf("%d",&_[i].a),_[i].b=i;
	sort(_+1,_+n+1,cmp);
	while(_[n].a-_[1].a>1 && m<k)
	  {
	  _[n].a--,_[1].a++,s[++m][0]=_[n].b,s[m][1]=_[1].b;
	  sort(_+1,_+n+1,cmp);
	  }
	printf("%d %d\n",_[n].a-_[1].a,m);
	for(int i=1;i<=m;i++)
	  printf("%d %d\n",s[i][0],s[i][1]);
	return 0;
}
