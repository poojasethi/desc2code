#include <bits/stdc++.h>
#define rep(i,j,k)	for (int i = j; i <= k; i++)
using namespace std;
typedef long long ll;
int n, k;
int ans, sum, t1[3000], t2[3000];
struct zcc{
	int a, id;
}a[3000];
bool cmp(zcc z, zcc c)
{
	return z.a < c.a;
}
int main()
{
	cin >> n >> k;
	rep(i,1,n)	scanf("%d", &a[i].a), a[i].id = i;
	sort(a+1,a+1+n,cmp);
	while (a[1].a < a[n].a-1 && ans<k)
	{
		ans++;
		t1[ans] = a[n].id, t2[ans] = a[1].id;
		a[n].a--, a[1].a++;sort(a+1,a+1+n,cmp);
	}
	cout << a[n].a-a[1].a << ' ' << ans << endl;
	rep(i,1,ans)	cout << t1[i]<<' ' << t2[i] << endl;
}
