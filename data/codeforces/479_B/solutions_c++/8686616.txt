#include<bits/stdc++.h>
using namespace std;
int i[1000],t[10000],s[10000];
main()
{
	int a,b,x=0;
	cin>>a>>b;
	for(int u=0;u<a;u++)
	cin>>i[u];
	while(*max_element(i,i+a)-*min_element(i,i+a)>1&&x<b)
	{
		t[x]=max_element(i,i+a)-i+1;
		s[x]=min_element(i,i+a)-i+1;
		i[max_element(i,i+a)-i]--;
		i[min_element(i,i+a)-i]++;
		x++;
	}
	cout<<*max_element(i,i+a)-*min_element(i,i+a)<<" "<<x<<endl;
	for(int u=0;u<x;u++) cout<<t[u]<<" "<<s[u]<<endl;
}
