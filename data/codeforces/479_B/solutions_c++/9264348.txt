#include<iostream>
#include<algorithm>
using namespace std;
int a[1005],c[1005],d[1005],n,k,i,l=0;
int main(){
    cin>>n>>k;
    for(i=0;i<n;i++)
        cin>>a[i];
    while(l<k){
        if(*min_element(a,a+n)<*max_element(a,a+n)-1){
            c[l]=max_element(a,a+n)-a+1;
            d[l]=min_element(a,a+n)-a+1;
            a[max_element(a,a+n)-a]--;
            a[min_element(a,a+n)-a]++;
            l++;
        }
        else
            break;
    }
    cout<<*max_element(a,a+n)-*min_element(a,a+n)<<" "<<l<<endl;
    for(i=0;i<l;i++)
        cout<<c[i]<<" "<<d[i]<<endl;
    return 0;
}
