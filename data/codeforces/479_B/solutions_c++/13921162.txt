#include <cstdio>
#include <algorithm>
using namespace std;

int main()
{
	int n,k; scanf("%d%d",&n,&k);
	int arr[1000]; for(int i=0;i<n;i++) scanf("%d",&arr[i]);
	int oo[1000],o[1000],j=0;
	
	int temp=0;
	while(temp<k)
	{
		int max=max_element(arr,arr+n)-arr,min=min_element(arr,arr+n)-arr;
		if(arr[max]-arr[min]==0) goto print;
		arr[max]--;
		arr[min]++;
		o[j]=max; oo[j]=min;
		temp++;
		j++;
	}
print:
	{
		printf("%d %d\n",(*max_element(arr,arr+n))-(*min_element(arr,arr+n)),temp);
		for(int i=0;i<j;i++) printf("%d %d\n",o[i]+1,oo[i]+1);
	}
	return 0;
}
