#include<iostream>
using namespace std;

const int maxn=110;
int n,k,a[maxn];
int st[1100][2];

int main()
{
	cin>>n>>k;
	for(int i=1;i<=n;i++) cin>>a[i];
	int m=0;
	for(;k>0;k--,m++){
		int mx=1,mn=1;
		for(int i=2;i<=n;i++)
		{
			if(a[mx]<a[i]) mx=i;
			if(a[mn]>a[i]) mn=i;
		}
		if(a[mn]==a[mx])break;
		++a[mn],--a[mx];
		st[m][0]=mx; st[m][1]=mn;
	}
	int mx=a[1],mn=a[1];
	for(int i=2;i<=n;i++)
	{
		if(mx<a[i])mx=a[i];
		if(mn>a[i])mn=a[i];
	}
	cout<<mx-mn<<' '<<m<<endl;
	for(int i=0;i<m;i++)cout<<st[i][0]<<' '<<st[i][1]<<endl;
	return 0;
}
