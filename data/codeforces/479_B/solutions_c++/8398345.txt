#include<stdio.h>
#include<algorithm>
using namespace std;
int n, k;
int H[100];
int R[1000][2];
main(){
	int d;
	scanf("%d%d", &n,&k);
	for(int i=0; i<n; ++i) scanf("%d", &H[i]);
	for(int i=0; i<k; ++i){
		int *pMax, *pMin;
		pMax = max_element(H, H+n);
		pMin = min_element(H, H+n);
		if((*pMax) == (*pMin)){
			k = i;
			break;
		}
		(*pMax)--;
		(*pMin)++;
		R[i][0] = pMax - H;
		R[i][1] = pMin - H;	
	}
	d = *max_element(H, H+n) - *min_element(H, H+n);
	printf("%d %d\n", d, k);
	for(int i=0; i<k; i++){
		printf("%d %d\n", R[i][0]+1, R[i][1]+1);
	}
}
