#include<iostream>
#include<stdio.h>
using namespace std;
int main()
{
int n,k,x,c,p,l;
cin>>n>>k;
int a[n+2],g[k+1][2];
for(int i=0;i<n;i++)
cin>>a[i];int z=0;
do
{  x=a[0];c=a[0];p=l=0;



    for(int j=0;j<n;j++)
       {

        if(a[j]<x)
        {x=a[j];p=j;}
        if(a[j]>c)
            {c=a[j];l=j;}
       }
             g[z][0]=l+1;g[z++][1]=p+1;
            if(x==c || c-x==1)
                break;
  a[l]--;a[p]++;
}while(k--);
cout<<c-x<<" "<<z-1<<"\n";
for(int i=0;i<z-1;i++)
    cout<<g[i][0]<<" "<<g[i][1]<<"\n";


}
