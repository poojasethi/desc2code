#include<cstdio>
#include<algorithm>
using namespace std;
const int maxn=1010;
int n,k,cnt=0;
struct point{int x,p;}a[maxn];
int ans[1010][2];

bool cmp(point u,point v){
	return u.x<v.x;
}

int main(){
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i].x);
		a[i].p=i;
	}
	sort(a+1,a+n+1,cmp);
	while(k--){
		if (a[1].x<a[n].x-1) a[1].x++,a[n].x--;
		else break;
		ans[cnt][0]=a[n].p,ans[cnt][1]=a[1].p;
		cnt++;
		sort(a+1,a+n+1,cmp);
	}
	printf("%d %d\n",a[n].x-a[1].x,cnt);
	for(int i=0;i<cnt;i++)
		printf("%d %d\n",ans[i][0],ans[i][1]);
}
