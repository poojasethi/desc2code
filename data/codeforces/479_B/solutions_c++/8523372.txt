#include<stdio.h>
int main()
{int n,k,i,fun(int *,int,int,int);
scanf("%d %d",&n,&k);
int a[n];
for(i=0;i<n;i++)
{scanf("%d",&a[i]);
}
fun(&a[0],n,k,0);
return 0;
}

int fun(int *a,int n,int k,int c)
{ int i,p1,p2,max1,min1;
max1=a[0];
min1=a[0];
for(i=0;i<n;i++)
{if(max1<=a[i])
{max1=a[i];
p1=i;}
if(min1>=a[i])
{min1=a[i];
p2=i;}
}
if((k==0)||((a[p1]-a[p2])==0)||((a[p1]-a[p2])==1))
{printf("%d %d\n",a[p1]-a[p2],c);
return 0;
}
k--;
c++;
a[p1]--;
a[p2]++;
fun(&a[0],n,k,c);
printf("%d %d\n",p1+1,p2+1);
return 0;
}
