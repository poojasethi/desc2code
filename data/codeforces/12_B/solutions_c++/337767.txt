#include <iostream>
#include <algorithm>
using namespace std;

string f(string n) {
	if (n == "0") {
		return n;
	}

	sort(n.begin(), n.end());
	swap(n[0], n[ n.find_first_not_of("0") ]);
	return n;
}

int main() {
	string n, m;
	cin >> n >> m;

	cout << (f(n) == m ? "OK" : "WRONG_ANSWER") << endl;
	return 0;
}
