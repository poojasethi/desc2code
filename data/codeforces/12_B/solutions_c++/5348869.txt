#include<iostream>
#include<algorithm>
#include<string>

using namespace std;

int main()
{
    string a,b;
    cin>>a>>b;
    sort(a.begin(),a.end());
    for(int i=0;i<a.size();i++){
        if(a[i]!='0'){swap(a[i],a[0]);break;}
    }
    if(a==b)cout<<"OK"<<endl;else cout<<"WRONG_ANSWER"<<endl;
    return 0;
}
