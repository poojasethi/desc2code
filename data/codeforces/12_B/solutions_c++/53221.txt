#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main ()
{
	string n, m;
	cin >> n >> m;
	
	sort(n.begin(), n.end());
	for(int i = 0; i < n.size(); i++)
		if(n[i] != '0')
		{
			swap(n[0], n[i]);
			break;
		}
	
	cout << (n == m ? "OK" : "WRONG_ANSWER") << endl;
	
}
