#include <iostream>
#include <string>
#include <algorithm>
using namespace std;

int main() {
    string a, b;
    cin >> a >> b;
    sort(a.begin(), a.end());
    int i=0;
    while(i<(int)a.size() && a[i]=='0') i++;
    if(i<(int)a.size()) swap(a[0], a[i]);
    cout << (a==b ? "OK" : "WRONG_ANSWER") << endl;
    return 0;
}
