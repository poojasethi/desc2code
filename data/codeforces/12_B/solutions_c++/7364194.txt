#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
string change(string x){
	sort(x.begin(), x.end());
	int idx=0; while(x.size()>1 && x[idx]=='0') idx++;
	swap(x[0], x[idx]);
	return x;
}
int main(){
	string x, y;
	cin>>x;
	cin>>y;
	printf(change(x)==y ? "OK\n" : "WRONG_ANSWER\n");
}
