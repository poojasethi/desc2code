#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	string n, m;
	cin >> n >> m;
	sort(n.begin(), n.end());
	int i = 0;
	while (n[i] == '0')
		i++;
	if (i != (int)n.size())
		swap(n[0], n[i]);
	if (n == m)
		cout << "OK" << endl;
	else
		cout << "WRONG_ANSWER" << endl;
	return 0;
}
