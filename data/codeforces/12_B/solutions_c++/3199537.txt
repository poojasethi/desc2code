#include <algorithm>
#include <iostream>
#include <string>
using namespace std;

int main(void)
{
	string n, m;

	cin >> n >> m;
	sort ( n.begin(),n.end() );
	int pos=0;
	while (pos<n.size() && n[pos] == '0')
		++pos;
	if ( pos < n.size() )	swap (n[0],n[pos]);
	cout << (n==m?	"OK" : "WRONG_ANSWER") << endl;
	return 0;
}
