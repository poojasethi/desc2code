#include<cstdio>
#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main()
{
    string x,ans;
    int i=0;
    cin>>x>>ans;
    sort(x.begin(),x.end());
    while(x[i]=='0')i++;
    if(i!=x.size())swap(x[0],x[i]);
    if(x==ans)printf("OK\n");
    else printf("WRONG_ANSWER\n");
}
