#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
	string n,m;
	cin >> n >> m;

	sort (n.begin(), n.end() );
	if (n[0] == '0' )
		for (int i = 1; i < n.length(); i++ )
			if (n[i] != '0' ){
				swap (n[0], n[i] );
				break;
			} // en dif

	cout <<  (n == m ? "OK" : "WRONG_ANSWER") << endl;
		
	return 0;
}
