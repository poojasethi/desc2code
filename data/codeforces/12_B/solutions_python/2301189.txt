n = raw_input()
ans = raw_input()
st = list(n)
st.sort()
min_val = '9'
min_idx = 0
for i in range(len(st)):
    s = st[i]
    if s < min_val and s != '0':
        min_val = s
        min_idx = i
st[0], st[min_idx] = st[min_idx], st[0]
myans = '%s'*len(st) % tuple(st)
print 'OK' if myans == ans else 'WRONG_ANSWER'
