s = raw_input();
t = raw_input();

flag = 1;
for i in range(len(s)):
	if s.count(s[i]) != t.count(s[i]):
		flag = 0; break;
#print flag
pos = 0

if t[0] == '0' and len(t) > 1:
	flag = 0;
if flag:
	for i in range(pos+1, len(t)):
		if t[i] == '0' and i == pos+1:
			continue;
		if t[i] < t[i-1]:
			flag = 0; break;
print "OK" if flag else "WRONG_ANSWER";
