
n = input()
m = raw_input()

tab = sorted(list(str(n)))

if sorted(list(m)) != tab:
    print 'WRONG_ANSWER\n'
    tab = []

for i in range(len(tab)):
    if tab[i] != '0':
        if ''.join(list(tab[i])+tab[:i]+tab[i+1:]) == m:
            print 'OK\n'
        else:
            print 'WRONG_ANSWER\n'
        tab = []
        break

if len(tab):
    print 'OK\n'
