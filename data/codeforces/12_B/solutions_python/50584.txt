from sys import *

def solve(s):
    ans = list(s)
    ans.sort()
    for i in xrange(len(ans)):
        if ans[i] != '0':
            ans[0],ans[i] = ans[i],ans[0]
            return ans
    return ans

s = stdin.readline()[:-1]
t = stdin.readline()[:-1]
if solve(s) == list(t):
    print "OK"
else:
    print "WRONG_ANSWER"
