n=input()
m=raw_input()
k=int(m)
a=map(int,str(n))
b=sorted(a)
count=0
for i in range(len(b)):
    if (b[i]==0):
        count=count+1
    else:
        break
if(len(b)>1):   
    temp=b[0]
    b[0]=b[count]
    b[count]=temp

num = int(''.join(map(str,b)))

if ((num==k) & (len(b)==len(m))):
    print "OK"
else:
    print "WRONG_ANSWER"
