import sys
from itertools import product

file = sys.stdin
#file = open("test", "r")
p = map(int, list(file.readline().rstrip()))
p.sort()
for i in range(len(p)):
    if p[i] != 0:
        temp = p.pop(i)
        p.insert(0, temp)
        break

ans =  "".join(map(str, p))
bro = file.readline().rstrip()
if ans == bro:
    print "OK"
else:
    print "WRONG_ANSWER"
