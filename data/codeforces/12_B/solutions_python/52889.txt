# -*- coding: utf-8 -*-

rev = lambda s:''.join(list(reversed( s )))
s  = (rev(sorted(raw_input())))
sa = raw_input()
try:
    r = s.index('0')
    s = list(s)
    s[r-1], s[-1] = s[-1], s[r-1]
except ValueError:
    pass
s = rev(s)
if s==sa or (int(s)==0 and sa=='0'):
    print 'OK'
else:
    print 'WRONG_ANSWER'
