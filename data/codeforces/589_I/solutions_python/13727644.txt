# Lottery

n, k = raw_input().split(' ')
balls = raw_input().split(' ')

n, k = int(n), int(k)

for i in range(len(balls)):
	balls[i] = int(balls[i])


pt = [0] * k

ans = 0

for i in balls:
	pt[i - 1] += 1
	if pt[i - 1] > n / k:
		ans += 1
#ans /= 2

print(ans)
