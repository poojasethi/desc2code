(n,k) = map(int,raw_input().split())
#arr = collections.Counter(map(int,raw_input().split()))
userBalls = {v:0 for v in range(1,k+1)}
for c in map(int,raw_input().split()):
    userBalls[c] += 1
normal = n/k
answer = 0
