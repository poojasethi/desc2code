first_string = raw_input()
second_string = raw_input()

BallNumber, ParticipantNumber = map(int, first_string.split())
BallType =  map(int, second_string.split())

BallNumber_per_Person = BallNumber/ParticipantNumber

def solve():
	ans = 0
	for num in range(ParticipantNumber):
		if BallType.count(num + 1) > BallNumber_per_Person:
			ans += BallType.count(num + 1) - BallNumber_per_Person
	print ans

solve()
