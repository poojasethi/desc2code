
if __name__ == '__main__':
    n, k = [int(x) for x in raw_input().split()]
    c = [int(x) for x in raw_input().split()]

    count = [0] * (k+1)
    for ci in c:
        count[ci] += 1
    n_k = n / k
    ret = 0
    for counti in count:
        if counti > n_k:
            ret += (counti - n_k)
    print ret
