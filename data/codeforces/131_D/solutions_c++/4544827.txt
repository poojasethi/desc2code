#include<iostream>
#include<stdlib.h>
#include<stdio.h>
#include <vector>
using    namespace std;
int n;
int sz[3005];
int dist[3005];
vector<int>adj[3005];
int DFS(int node , int par , int depth){
	dist[node]=depth;
	for(int i=0 ; i<sz[node] ; i++){
		int next=adj[node][i];
		if(next==par)   continue;
		if(dist[next]) return dist[next];
		int k=DFS(next,node,depth+1);
		if(k) return k;
	}
	return 0;
}
int main(){
	scanf("%d",&n);
	for(int i=1,x,y ; i<=n ; i++){
		scanf("%d %d",&x,&y);
		adj[x].push_back(y);
		adj[y].push_back(x);
		sz[x]++,sz[y]++;
	}
	for(int i=1 ; i<=n ; i++){
		for(int j=1 ; j<=n ; j++)
			dist[j]=0;
		printf("%d ",DFS(i,-1,1)-1);
	}
	printf("\n");
	return 0;
}
