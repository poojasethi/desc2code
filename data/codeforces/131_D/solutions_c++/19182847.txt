#include<bits/stdc++.h>
using namespace std;
const int maxn = 3000 + 10;
int d[maxn];
vector<int>g[maxn];
int dfs(int k,int fa,int dep){
    d[k] = dep;
    int len = g[k].size();
    for (int i = 0; i < len; ++i){
        int u = g[k][i];
        if (u == fa)continue;
        if (d[u])return d[u];
        int kk = dfs(u,k,dep+1);
        if (kk)return kk;
    }
    return 0;
}
int main(){
    int n;
    scanf("%d",&n);
    for (int i = 0 ;i < n; ++i){
        int u,v;
        scanf("%d %d",&u,&v);
        g[u].push_back(v);
        g[v].push_back(u);
    }
    for (int i = 1; i <= n; ++i){
        memset(d,0,sizeof d);
        if (i > 1)printf(" ");
        printf("%d",dfs(i,-1,1)-1);
    }
    return 0;
}
