 #include <cstdio>
#include <string>
#include <cstring>
#include <iostream>
#include <algorithm>
const int inf = 1000000007;

using namespace std;

int i, j, k, m, n, t, deg[3005], x[3005], q[3005], qi, qn, p[3005], ans[3005];

void get(int i) {
    if (ans[i] == -1) {
        get(p[i]);
        ans[i] = ans[p[i]]+1;
    }
}

int main()
{
   
    cin >> n;
    for (k = 0; k < n; k++) {
        cin >> i >> j;
        deg[i]++;
        deg[j]++;
        x[i] ^= j;
        x[j] ^= i;
    }
    for (i = 1; i <= n; i++)
        if (deg[i] == 1) q[qn++] = i;
    while (qi < qn) {
        i = q[qi++];
        ans[i] = -1;
        j = x[i];
        p[i] = j;
        x[j] ^= i;
        deg[j]--;
        if (deg[j] == 1) {
            q[qn++] = j;
        }
    }
    for (i = 1; i <= n; i++) {
        get(i);
        cout << ans[i] << " ";
    }
    return 0;
}
