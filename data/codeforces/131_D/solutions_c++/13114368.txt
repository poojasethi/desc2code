//In the Name of God

#include<iostream>
#include<vector>

using namespace std;

const int maxn=3030;
int n,d[maxn],dis[maxn];
vector<int> g[maxn];
bool vis[maxn],mark[maxn];

void dfs(int v,int D)
{
  vis[v]=true;
  d[v]=D;
  for(int i=0;i<g[v].size();i++)
    if(g[v][i]!=D)
      if(!vis[g[v][i]])
	dfs(g[v][i],v);
      else if(!mark[v])
	for(int j=v;j!=d[g[v][i]];j=d[j])
	  mark[j]=true;
}

void bfs(int v)
{
  vis[v]=true;
  for(int i=0;i<g[v].size();i++)
    if(!vis[g[v][i]] && !mark[g[v][i]])
      dis[g[v][i]]=dis[v]+1,bfs(g[v][i]);
}

int main()
{
  cin>>n;
  for(int v,u,i=0;i<n;i++)
    {
      cin>>v>>u;
      g[v].push_back(u);
      g[u].push_back(v);
    }
  dfs(1,0);
  for(int i=1;i<=n;i++)
    vis[i]=false;
  for(int i=1;i<=n;i++)
    if(mark[i])
      bfs(i);
  for(int i=1;i<=n;i++)
    cout<<dis[i]<<" ";
  cout<<endl;
  return 0;
}
