#include <iostream>
#include <cstdio>
#include <algorithm>
#include <queue>
#include <string.h>
#include <limits.h>
#include <vector>

using namespace std;

vector<int> a[3010];
int N,s;
int b[3010];

int dfs(int f,int l,int deep)
{
    int temp=0;
    
    b[f]=deep;
    
    for(int i=0;i<a[f].size();i++)
        if(a[f][i]!=l)
        {
            if(b[a[f][i]])
                return b[a[f][i]];
            temp=dfs(a[f][i],f,deep+1);
            if(temp>0)
                return temp;
        }
    return -1;
}

int main()
{
    int x,y,i;
    cin>>N;
    
        for(i=0;i<N;i++)
        {
            scanf("%d%d",&x,&y);
            a[x].push_back(y);
            a[y].push_back(x);
        }
        
        for(i=1;i<=N;i++)
        {
            cout<<dfs(i,0,1)-1<<" ";
            memset(b, 0, sizeof(b));
        }
        cout<<endl;
    
    return 0;
}
