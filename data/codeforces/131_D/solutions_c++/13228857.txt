#include<bits/stdc++.h>
using namespace std;
vector <vector<int> >g(3005);
int color[3005];
bool vis[3005];
int dad[3005];
int dfs(int ver,int c)
{	
	int tmp=0;	
	color[ver]=c;
	for(int i=0;i<g[ver].size();i++)
	{
		
		if(g[ver][i]==ver || dad[ver]==g[ver][i])
			continue;
		if(color[g[ver][i]])
		{			
			return color[g[ver][i]];
		}		
		dad[g[ver][i]]=ver;		
		tmp=dfs(g[ver][i],c+1	);
		if(tmp>-1)return tmp;
	}
	return -1;
}
int main()
{
	int N;
	cin>>N;
	for(int i=1;i<=N;i++)
	{
		int x,y;
		cin>>x>>y;
		g[x].push_back(y);
		g[y].push_back(x);
	}
	for(int i=1;i<=N;i++)
	{
		cout<<dfs(i,1)-1<<" ";
		memset(dad,0,sizeof dad);
		memset(color,0,sizeof color);
	}
	return 0;
}
