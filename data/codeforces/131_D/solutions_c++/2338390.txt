#include <algorithm>
#include <stdio.h>
#include <vector>
#define  MAXN      3002
#define  pb        push_back
using    namespace std;

int n;
int sz[MAXN];
int dist[MAXN];
vector<int>adj[MAXN];

int DFS(int node , int par , int depth)
{
  dist[node]=depth;
  for(int i=0 ; i<sz[node] ; i++)
  {
    int next=adj[node][i];
    if(next==par)   continue;
    if(dist[next])  return dist[next];
    int k=DFS(next,node,depth+1);
    if(k) return k;
  }
  return 0;
}

int main()
{
  scanf("%d",&n);
  for(int i=1,x,y ; i<=n ; i++)
  {
    scanf("%d%d",&x,&y);
    adj[x].pb(y);
    adj[y].pb(x);
    sz[x]++,sz[y]++;
  }
  for(int i=1 ; i<=n ; i++)
  {
    for(int j=1 ; j<=n ; j++)
      dist[j]=0;
    printf("%d ",DFS(i,-1,1)-1);
  }
  printf("\n");
  return 0;
}
