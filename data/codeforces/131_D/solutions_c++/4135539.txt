#include <iostream>
using namespace std;

int p[3001], x[3001], dist[3001], deg[3001], q[3001];

void getDist(int i)
{
    if(dist[i]==-1)
    {
        getDist(p[i]);
        dist[i] = dist[p[i]]+1;
    }
}
int main ()
{
    int n, a, b;
    cin>>n;
    for(int i=0; i<n; ++i)
    {
        cin>>a>>b;
        ++deg[a];
        ++deg[b];
        x[a] ^= b;
        x[b] ^= a;
    }

    int ql=0, qr=0;
    for(int i=1; i<=n; ++i)
        if(deg[i]==1)
            q[qr++] = i;
    int e, v;
    while(ql<qr)
    {
        e = q[ql++];
        dist[e] = -1;
        v = x[e];
        p[e] = v;
        x[v] ^= e;
        --deg[v];
        if(deg[v]==1)
            q[qr++] = v;
    }

    for(int i=1; i<=n; ++i)
    {
        getDist(i);
        cout<<dist[i]<<" ";
    }

    return 0;
}
