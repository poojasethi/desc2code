#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn=3002;
int a[maxn*2][2],b[maxn];
int d[maxn],q[maxn],p[maxn];
bool v[maxn];
int n,s,h,t;

void add(int i,int j){
  a[++s][0]=b[i];
  b[i]=s;
  a[s][1]=j;
}

int main(){
  int i,j,k;
  scanf("%d",&n);
  for (k=0;k<n;k++){
    scanf("%d%d",&i,&j);
    add(i,j);
    add(j,i);
    d[i]++;
    d[j]++;
  }
  for (i=1;i<=n;i++)
    if (d[i]==1) q[++t]=i;
  while (h<t){
    k=q[++h];
    v[k]=1;
    for (i=b[k];i;i=a[i][0]){
      j=a[i][1];
      if (!v[j]){
        d[j]--;
        if (d[j]==1) q[++t]=j;
        p[k]=j;
      }
    }
  }
  memset(d,0,sizeof(d));
  for (i=t;i;i--) d[q[i]]=d[p[q[i]]]+1;
  for (i=1;i<=n;i++) printf("%d ",d[i]);
  puts("");
  return 0;
}
