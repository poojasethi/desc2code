#include<iostream>
#include<vector>
#include<deque>
#define pb push_back
using namespace std;
int n,a[3005];
vector<int> v[3005];
deque<int> d;
bool dfs(int i)
{
int j,l;
for(j=0;j<v[i].size();j++)
{
if(a[v[i][j]] && v[i][j]!=a[i])
{
l=v[i][j];
do{
j=a[i];
a[i]=-1;
i=j;
}while(i!=l);
a[i]=-1;
return 0;
}
if(a[v[i][j]]==0)
{
a[v[i][j]]=i;
if(!dfs(v[i][j])) return 0;
}
}
return 1;
}
int main()
{
int i,j,l;
cin>>n;
for(i=1;i<=n;i++)
{
cin>>j>>l;
v[j].pb(l);
v[l].pb(j);
}
for(i=1;i<=n;i++)
{
a[i]=0;
}
a[1]=n+1;
dfs(1);
for(i=1;i<=n;i++)
{
if(a[i]<0) d.pb(i),a[i]=1;
else a[i]=0;
}
while(d.size())
{
i=d[0];
d.pop_front();
for(j=0;j<v[i].size();j++)
{
if(!a[v[i][j]])
{
a[v[i][j]]=a[i]+1;
d.pb(v[i][j]);
}
}
}
for(i=1;i<=n;i++) cout<<a[i]-1<<" ";
return 0;
}
