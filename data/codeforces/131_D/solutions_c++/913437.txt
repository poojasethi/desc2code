#include <iostream>
#include <vector>
#include <queue>
#include <cstring>
#include <algorithm>
using namespace std;
#define pb(a) push_back(a)
int n;
vector<int> fo[3010];
int go[3010] , val[3010];
queue<int> Q;
int calc(int vr){
  if(val[vr]!=-1)return val[vr];
  return val[vr]=(go[vr]==-1?0:1+calc(go[vr]));
}
int main(){
  int a,b;
  cin>>n;
  for(int i=0;i<n;++i)cin>>a>>b , --a , --b , fo[a].pb(b) , fo[b].pb(a);
  for(int i=0;i<n;++i)if(fo[i].size()==1)Q.push(i);
  memset(go,-1,sizeof go); memset(val,-1,sizeof val);
  while(!Q.empty()){
   int V=Q.front(); Q.pop();
   int next=fo[V][0];
   go[V]=next;
   fo[next].erase( find(fo[next].begin() , fo[next].end() , V) );
   if(fo[next].size()==1)Q.push(next);
                   }
  for(int i=0;i<n;++i)cout<<calc(i)<<(i+1==n?'\n':' ');
  //system("pause");
  return 0;
}
