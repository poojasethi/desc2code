#include <iostream>
#include<vector>
#include<stack>
#include<queue>


using namespace std;

vector <int> a[3001];
int d[3001]={0};
int ans[3001];
int n;


int dfs(int x,int len)
{
 ans[x]=len;
   for(size_t i=0; i<a[x].size();i++)
    { int y=a[x][i];
    if (d[y]<0) { d[y]=0; dfs(y,len+1);}

    }


}

int main()
{
cin>>n;
for (int i=0; i<n;i++)
 { int x,y;
 cin>>x>>y;
 a[x].push_back(y); d[x]++;
 a[y].push_back(x); d[y]++;
 }
queue <int> q;
for (int i=1;i<=n;i++) if (d[i]==1) q.push(i);
while (q.size()!=0)
 {
   int x;
   x=q.front();
   q.pop();
   d[x]=-1;
    for(size_t i=0; i<a[x].size();i++)
     {
      int y=a[x][i];d[y]--; if (d[y]==1) q.push(y);
     }
 }

for (int i=1;i<=n;i++) if (d[i]>0) dfs(i,0);

for (int i=1;i<=n;i++) cout<<ans[i]<<"  ";

    return 0;
}
