#include <iostream>
#include <vector>
#include <queue>
using namespace std;
int n;
vector<int> e[3000];
int ans[3000];
int d[3000];

int main(){
	cin >> n;
	for(int i=0; i<n; i++){
		int a,b;
		cin >> a >> b;
		a--;
		b--;
		e[a].push_back(b);
		e[b].push_back(a);
		d[a]++;
		d[b]++;
	}
	queue<int> q;
	for(int i=0; i<n; i++){
		if(d[i]==1)
			q.push(i);
	}
	while(!q.empty()){
		int src=q.front();
		//cerr<< src << endl;
		q.pop();
		for(size_t i=0; i<e[src].size(); i++){
			int des=e[src][i];
			d[des]--;
			if(d[des]==1)
				q.push(des);
		}
	}
	for(int i=0; i<n; i++){
		if(d[i]>1){
			q.push(i);
			ans[i]=1;
		}
	}

	while(!q.empty()){
		int src=q.front();
		q.pop();
		for(size_t i=0; i<e[src].size(); i++){
			int des=e[src][i];
			if(ans[des])
				continue;
			ans[des]=ans[src]+1;
			q.push(des);
		}
	}
	for(int i=0; i<n; i++){
		if(i) cout<< " ";
		cout<<ans[i]-1;
	}
	cout<<endl;
}
