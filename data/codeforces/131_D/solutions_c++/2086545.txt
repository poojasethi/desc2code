#include <stdio.h>
int g[3001][3001]={},d[3001]={},ans[3001]={};
inline void cut(int a){
	int i=-1;
	while(g[a][++i]>0&&d[g[a][i]]==1);
	if(g[a][i]>0&&d[g[a][i]]--==2)
		cut(g[a][i]);
}
inline void dfs(int a,int b,int p){
	ans[a]=b++;
	int i=-1;
	while(g[a][++i]>0)
		if(g[a][i]!=p)
			dfs(g[a][i],b,a);
}
int main() {
	int n,a,b,tocut[3000],ntocut=0;
	scanf("%d",&n);
	for(int i=1;i<=n;++i){
		scanf("%d%d",&a,&b);
		g[a][d[a]++]=b;
		g[b][d[b]++]=a;
	}
	for(int i=1;i<=n;++i)
		if(d[i]==1)
			tocut[ntocut++]=i;
	for(int i=0;i<ntocut;++i)
		cut(tocut[i]);
	for(int i=1;i<=n;++i)
		if(d[i]>1){
			int j=-1;
			while(g[i][++j]>0){
				if(d[g[i][j]]==1)
					dfs(g[i][j],1,i);
			}
		}
	for(int i=1;i<=n;++i)
		printf("%d ",ans[i]);
	return 0;
}
