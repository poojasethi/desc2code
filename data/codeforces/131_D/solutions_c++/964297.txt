#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int deg[3005];
vector<int> a[3005];
int Q[3005], qe=0;
int u[3005], p[3005];

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		a[x].push_back(y);
		a[y].push_back(x);
		++deg[x];
		++deg[y];
	}
	for(int i=1;i<=n;i++) {
		if(deg[i]==1) {
			Q[qe++]=i;
		}
	}
	for(int i=0;i<qe;i++) {
		u[Q[i]]=1;
		for(int j=0;j<(int)a[Q[i]].size();j++) {
			int y=a[Q[i]][j];
			if(u[y]==0) { if(--deg[y]==1) Q[qe++]=y; p[Q[i]]=y;}
		}
	}
	int dis[3005]={};
	for(int i=qe-1;i>=0;--i)
		dis[Q[i]] = dis[p[Q[i]]]+1;
	for(int i=1;i<=n;i++)
		printf("%d ", dis[i]);
	puts("");
	return 0;
}
