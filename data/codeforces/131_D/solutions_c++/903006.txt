#include <cstdio>
#include <iostream>
#include <vector>
#define maxn 3010
using namespace std;

int N;
int deg[maxn];
int dist[maxn],tree[maxn];
int way[maxn][maxn];
int tway[maxn][maxn];
int st[maxn],push;

void dfs(int k,int t){
	dist[k]=t;
	
	for(int i=1;i<=N;i++)
		if(tree[i] && way[k][i] && !dist[i])
			dfs(i,t+1);
}

int main(){
	scanf("%d",&N);
	
	for(int i=1,a,b;i<=N;i++){
		scanf("%d%d",&a,&b);
		
		way[a][b]=1;
		tway[a][b]=1;
		way[b][a]=1;
		tway[b][a]=1;
		
		deg[a]++;
		deg[b]++;
	}
	
	for(int i=1;i<=N;i++)
		if(deg[i]==1)
			st[++push]=i;
	
	while(push){
		int tmp=st[push--];
		tree[tmp]=1;
		for(int i=1;i<=N;i++)
			if(tway[tmp][i]){
				tway[tmp][i]=tway[i][tmp]=0;
				deg[i]--;
				if(deg[i]==1)
					st[++push]=i;
				break;
			}
	}
	
	for(int i=1;i<=N;i++)
		if(!tree[i])
			dfs(i,1);
	
	for(int i=1;i<=N;i++)
		printf("%d ",dist[i]-1);
	
	return 0;
}
