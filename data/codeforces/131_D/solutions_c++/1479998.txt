#include <cstdio>
#include <vector>
using namespace std;
vector<int> graph[3001];
int daddy[3001];
bool visited[3001];
int dist[3001];
int N,c;
bool goon=true;
void dfs1(int n, int d){
	if(!goon) return;
	if(daddy[n]){
		c=n;
		int p=d;
		while(p!=n){
			dist[p]=1;
			p=daddy[p];
		}
		dist[p]=1;
		goon=false;
		return;
	}
	daddy[n]=d;
	for(int i=0;i<graph[n].size();i++)
		if(graph[n][i]!=d)
			dfs1(graph[n][i],n);
}
void dfs2(int n, int d){
	if(visited[n]) return;
	visited[n]=true;
	if(!dist[n]) dist[n]=dist[d]+1;
	for(int i=0;i<graph[n].size();i++) dfs2(graph[n][i],n);
}
int main(){
	scanf("%d",&N);
	for(int i=0;i<N;i++){
		int a,b;
		scanf("%d %d",&a,&b);
		graph[a].push_back(b);
		graph[b].push_back(a);
	}
	dfs1(1,1);
	dfs2(c,1);
	for(int i=1;i<=N;i++) printf("%d ",dist[i]-1);
	printf("\n");
}
