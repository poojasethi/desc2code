#define PI acos(-1.0)
using namespace std;
int n,l[3005];
vector<int> a[3005];
int dfs(int x,int p){
	l[x]=l[p]+1;
	for(int i=0; i<a[x].size(); ++i){
		int y=a[x][i];
		if (y==p) continue;
		if (l[y]>0) return l[y];
		int t=dfs(y,x);
		if (t!=-1) return t;
	}
	return -1;
}
int main(){
	cin>>n;
	for(int i=1; i<=n; ++i){
		int x,y;
		cin>>x>>y;
		a[x].push_back(y);
		a[y].push_back(x);
	}
	for(int i=1; i<=n; ++i){
		for(int j=1; j<=n; ++j) l[j]=0;
		cout<<dfs(i,0)-1;
		if (i<n) cout<<" ";
		else cout<<endl;
	}
	return 0;
}
