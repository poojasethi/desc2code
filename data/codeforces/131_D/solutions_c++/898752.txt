#include <vector>
#include <set>
#include <iostream>
using namespace std;

int main() {
	int n, x, y;
	set<int> g[3000];
	vector<int> c[3000];
	bool u[3000];
	int v;
	int d[3000];
	vector<int> q;
	size_t t;
	int z;

	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> x >> y;
		--x;
		--y;
		g[x].insert(y);
		g[y].insert(x);
	}
	for (int i = 0; i < n; ++i)
		u[i] = false;
	v = 0;
	while (v != -1) {
		v = -1;
		for (int i = 0; i < n; ++i)
			if (!u[i] && g[i].size() == 1) {
				v = i;
				break;
			}
		if (v != -1) {
			c[*g[v].begin()].push_back(v);
			g[*g[v].begin()].erase(v);
			u[v] = true;
		}
	}
	for (int i = 0; i < n; ++i)
		if (!u[i]) {
			d[i] = 0;
			t = 0;
			q.clear();
			q.push_back(i);
			while (t < q.size()) {
				z = q[t];
				++t;
				for (size_t j = 0; j < c[z].size(); ++j) {
					q.push_back(c[z][j]);
					d[c[z][j]] = d[z] + 1;
				}
			}
		}
	for (int i = 0; i < n; ++i)
		cout << d[i] << " ";
}
