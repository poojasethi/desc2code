#include <iostream>
#include <string>
#include <cstring>
#include <vector>
#include <algorithm>
#include <set>
#include <map>
#include <queue>
#include <cstdlib>
#include <cmath>
#include <iomanip>
#include <ctime>
using namespace std;
typedef long long int64;

int n, res;
int rarr[10][3];
int arr[10][3];
int cnt[10];

string reg(int idx) {
	string result = "eax";
	result[1] += idx;
	return result;
}

void bt(int idx) {
	if(idx - 1 >= res) return;
	if(cnt[idx - 1] == n) {
		res = idx - 1;
		memcpy(rarr, arr, sizeof arr);
	}
	if(idx == 6) {
		return;
	}
	// lea x, [k * y]
	for(int i = 0; i < idx; i ++) {
		for(int j = 2; j <= 8; j *= 2) {
			arr[idx][0] = i;
			arr[idx][1] = j;
			arr[idx][2] = -1;
			cnt[idx] = cnt[i] * j;
			bt(idx + 1);
		}
	}
	// lea x, [y + k * z]
	for(int i = 0; i < idx; i ++) {
		for(int j = 0; j < idx; j ++) {
			for(int k = 1; k <= 8; k *= 2) {
				arr[idx][0] = i;
				arr[idx][1] = j;
				arr[idx][2] = k;
				cnt[idx] = cnt[i] + cnt[j] * k;
				bt(idx + 1);
			}
		}
	}
}

int main() {
	cin >> n;
	res = 100;
	cnt[0] = 1;
	bt(1);
	cout << res << endl;
	for(int i = 1; i <= res; i ++) {
		if(rarr[i][2] == -1) {
			cout << "lea " << reg(i) << ", [" << rarr[i][1] << "*" << reg(rarr[i][0]) << "]" << endl;
		} else {
			cout << "lea " << reg(i) << ", [" << reg(rarr[i][0]) << " + " << rarr[i][2] << "*" << reg(rarr[i][1]) << "]" << endl;
		}
	}
	return 0;
}
