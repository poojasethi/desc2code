#include <cstdio>
#include <cstring>
using namespace std;
int reg[32];
int n,ans;
char str[32][64];
bool search(int d)
{
    if(reg[d]==n)
    {
        return true;
    }
    if(d>=ans||reg[d]>n) return false;
    d++;
    for(int i=0;i<d;i++)
    {
        for(int j=0;j<d;j++)
        {
            for(int k=1;k<=8;k<<=1)
            {
                reg[d]=reg[i]+reg[j]*k;
                if(search(d))
                {
                    sprintf(str[d],"lea e%cx, [e%cx + %d*e%cx]",d+'a',i+'a',k,j+'a');
                    return true;
                }
            }
        }
    }
    for(int i=0;i<d;i++)
    {
        for(int k=1;k<=8;k<<=1)
        {
            reg[d]=reg[i]*k;
            if(search(d))
            {
                sprintf(str[d],"lea e%cx, [%d*e%cx]",d+'a',k,i+'a');
                return true;
            }
        }
    }
    return false;
}
int main()
{
    scanf("%d",&n);
    memset(reg,0,sizeof(reg));
    reg[0]=1;
    for(ans=0;!search(0);ans++);
    printf("%d\n",ans);
    for(int i=1;i<=ans;i++) printf("%s\n",str[i]);
    return 0;
}
