#include<iostream>
#include<stdio.h>
#include<string>
using namespace std;
int n,x[26],t[] = {1,2,4,8},f[26][3],best = 7,F[26][3];
char tt[] = {'1','2','4','8'};
void dfs(int cur)
{
	if (cur >= best) return;
	for(int i = 0; i < cur; i++)
		for(int j = 0; j < cur; j++)
			for(int k = 0; k < 4; k++)
			{
				x[cur] = x[i] + x[j] * t[k];
				f[cur][0] = i;
				f[cur][1] = j;
				f[cur][2] = k;
				if (x[cur] == n)
				{
					best = cur;
					for(int i = 1; i <= cur; i++)
						for(int j = 0; j < 3; j++)
							F[i][j] = f[i][j];
					return;
				} 
				if (x[cur] > n) continue; 
				dfs(cur + 1);
			}
}
string make(int id)
{
	if (id == 0) id = 26;
	string s = "e";
	s = s + char('a' + id - 1);
	s = s + "x";
	return s;
}
int main()
{
	cin>>n;
	if (n == 1) cout<<0<<endl; else
	{
		x[1] = 1;
		dfs(2);
		cout<<best - 1<<endl;
		for(int i = 2; i <= best; i++)
			cout<<"lea "<<make(i)<<", ["<<make(F[i][0])<<" + "<<tt[F[i][2]]<<"*"<<make(F[i][1])<<"]"<<endl;
	}	
	return 0;
}
