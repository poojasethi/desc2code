#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;

int b[600];
int a[10],c[10],n;

void solve(int n,int x){
    for(int i=1;i<=n;++i)
        for(int j=2;j<=8;j*=2){
            int k=c[i]*j;
            if(k==x){printf("lea e%cx, [%d*e%cx]\n",'a'+n,j,'a'+i-1);return;}
        }
    for(int i=1;i<=n;++i)
        for(int j=1;j<=n;++j)
            for(int k=1;k<=8;k*=2){
                int l=c[i]+c[j]*k;
                if(l==x){
                    printf("lea e%cx, [e%cx + %d*e%cx]\n",'a'+n,'a'+i-1,k,'a'+j-1);
                    return;
                }
            }
}

void find(int cur){
    if(cur==6)return;
    for(int i=1;i<=cur;++i)
        for(int j=2;j<=8;j*=2){
            int k=a[i]*j;
            if(k>255)continue;
            if((!b[k])||(b[k]>cur)){b[k]=cur;if(k==n){memcpy(c,a,sizeof(a));c[cur+1]=n;}}
            if(k>a[cur]){a[cur+1]=k;find(cur+1);}
        }
    for(int i=1;i<=cur;++i)
        for(int j=1;j<=cur;++j)
            for(int k=1;k<=8;k*=2){
                int l=a[i]+a[j]*k;
                if(l>255)continue;
                if((!b[l])||(b[l]>cur)){b[l]=cur;if(l==n){memcpy(c,a,sizeof(a));c[cur+1]=n;}}
                if(l>a[cur]){a[cur+1]=l;find(cur+1);}
            }
}

int main(){
    scanf("%d",&n);
    a[1]=1;
    find(1);
    printf("%d\n",b[n]);
    for(int i=2;i<=b[n]+1;++i)
        solve(i-1,c[i]);
    return 0;
}
