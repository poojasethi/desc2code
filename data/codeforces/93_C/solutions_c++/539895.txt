#include<cstdio>
#include<string>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

int n,reg[26],n_ans;
char ans[255][32];

bool solve(int next){
	if(reg[next-1]>n || next>n_ans) return false; // pruning

	if(reg[next-1]==n){ n_ans=next-1; return true; }

	bool ok=false;
	rep(i,next) rep(j,next) for(int k=1;k<=8;k*=2) {
		reg[next]=reg[i]+k*reg[j];
		if(solve(next+1)){
			sprintf(ans[next-1],"e%cx, [e%cx + %d*e%cx]",next+'a',i+'a',k,j+'a');
			ok=true;
		}
	}

	rep(i,next) for(int k=1;k<=8;k*=2) {
		reg[next]=k*reg[i];
		if(solve(next+1)){
			sprintf(ans[next-1],"e%cx, [%d*e%cx]",next+'a',k,i+'a');
			ok=true;
		}
	}

	return ok;
}

int main(){
	scanf("%d",&n);
	n_ans=6;
	reg[0]=1;

	solve(1);

	printf("%d\n",n_ans);
	rep(i,n_ans) printf("lea %s\n",ans[i]);

	return 0;
}
