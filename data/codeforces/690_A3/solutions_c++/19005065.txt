#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int main() {
    int T;
    scanf("%d",&T);
    for (; T--;) {
        int n,m;
        scanf("%d%d",&n,&m);
        int sum = 0;
        for (int i = 1; i < n; i++) {
            int x;
            scanf("%d",&x);
            sum = (sum + x - 1) % n;
        }
        sum = (m - sum + n) % n;
        printf("%d\n",sum + 1);
    }
    return 0;
}
