#include <stdio.h>
#include <iostream>
#include <string.h>
#include <algorithm>
using namespace std;

void one(){
	int n,r; scanf("%d%d",&n,&r); int sum=0; 
	for (int i=1;i<=n-1;i++) {
		int p; scanf("%d",&p); sum+=p;
	}
	sum=r-sum;
	sum=((sum%n)+n)%n;
	if (!sum) sum=n;
	printf("%d\n",sum);
}

int main(){
	int t;
	scanf("%d",&t);
	while (t--) one();
}
