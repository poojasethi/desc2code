#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>

using namespace std;


int main(){
    int t;
    cin>>t;
    while(t--){
        int n,r;
        cin>>n>>r;
        int s=0;
        for(int i=0;i<n-1;i++){
            int x;
            cin>>x;
            s+=x;
        }
        s%=n;
        r%=n;
        int ans = r-s;
        ans=(ans%n+n)%n;
        if(ans==0) ans = n;
        cout<<ans<<endl;
    }
    return 0;
}
