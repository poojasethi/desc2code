#include <iostream>
#include<cmath>
using namespace std;

int main()
{
	ios_base::sync_with_stdio(0);
	int t;
	cin >> t;
	while(t-->0)
	{
        int n, r;
        cin >> n >> r;
        int s = 0;
        r--;
        for (int i = 0; i < n - 1; i++)
		{
			int x;
			cin >> x;
			s += x - 1;
		}
		//cout << r << " " << s << "\n";
		cout << ((r - s) % n + n) % n + 1 << "\n";
	}

	return 0;
}
