#include<cstdio>

using namespace std;

int t,a,b,m,n,i;

int main(){
    scanf("%d",&t);
    while(t--){
        scanf("%d %d",&m,&n);
        a=0;
        for(i=1;i<m;i++){
            scanf("%d",&b);
            a+=b-1;
        }
        a%=m;
        b=1;
        while(a!=n-1){
            a++;
            a%=m;
            b++;
            b%=m;
        }
        if(b==0)b=m;
        printf("%d\n",b);
    }
}
