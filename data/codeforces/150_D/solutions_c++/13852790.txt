#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;

typedef long long ll;
const int maxn  = 160;
const ll inf = 1e9;
ll dp[maxn][maxn][maxn];
char s[maxn];
ll a[maxn];
ll ans[maxn];

ll slove(int l,int r,int cn){
    if(dp[l][r][cn]!=-1) return dp[l][r][cn];
    dp[l][r][cn]=-inf;
    if(cn) dp[l][r][cn]=a[cn*2]+slove(l,r,0);
    if(l==r){
        dp[l][r][cn]=max(dp[l][r][cn],max(a[1]+a[cn*2],a[cn*2+1]));
        return dp[l][r][cn];
    }
    if(s[l]==s[r]){
        if(l==r-1) {
            dp[l][r][cn]=max(dp[l][r][cn],max(a[2]+a[cn*2],a[cn*2+2]));
            dp[l][r][cn]=max(dp[l][r][cn],max(a[1]+a[2*cn+1],2*a[1]+a[cn*2]));
        }
        else dp[l][r][cn]=max(dp[l][r][cn],slove(l+1,r-1,cn+1));
    }
    for(int i=l;i<r;i++){
        ll x=max(slove(l,i,0)+slove(i+1,r,cn),slove(l,i,cn)+slove(i+1,r,0));
        dp[l][r][cn]=max(dp[l][r][cn],x);
    }
    return dp[l][r][cn];
}

int main()
{
    //freopen("in.txt","r",stdin);
    int n;
    while(cin>>n){
        memset(dp,-1,sizeof(dp));
        memset(ans,0,sizeof(ans));
        for(int i=0;i<n;i++){
            scanf("%I64d",&a[i+1]);
            if(a[i]==-1) a[i]=-inf;
        }
        scanf("%s",s);
        for(int i=1;i<=n;i++){
            ans[i]=ans[i-1];
            for(int j=0;j<i;j++)
                ans[i]=max(ans[i],ans[j]+slove(j,i-1,0));
        }
        cout<<ans[n]<<endl;
    }
    return 0;
}
