#include<iostream>
#include<string>
#include<algorithm>
#include<cstring>
using namespace std;

string S;
const int inf=1061109567;
int n,a[151],f[151][151][76],ans[151];

int F(int x,int y,int z)
{
	if(x>y)return z?-inf:0;
	if(f[x][y][z]!=-1)return f[x][y][z];
	int ret=-inf;
	for(int i=x;i<y;++i)
	ret=max(ret,max(F(x,i,z)+F(i+1,y,0),F(x,i,0)+F(i+1,y,z)));
	if(S[x-1]==S[y-1])
	ret=max(ret,max(F(x+1,y-1,z+1),F(x+1,y-1,0)+a[z*2+(x<y)+1]));
	if(ret<0)ret=-inf;
	return f[x][y][z]=ret;
}


int main()
{
	#ifndef ONLINE_JUDGE
	freopen("cf.in","r",stdin);
	freopen("cf.out","w",stdout);
	#endif
	cin>>n;
	memset(f,-1,sizeof f);
	for(int i=1;i<=n;++i)
	{
		cin>>a[i];
		if(a[i]==-1)a[i]=-inf;
	}
	cin>>S;
	for(int i=1;i<=n;++i)
	{
		ans[i]=ans[i-1];
		for(int j=1;j<=i;++j)
			ans[i]=max(ans[i],ans[j-1]+F(j,i,0));
	}
	cout<<ans[n]<<endl;
	return 0;
}
