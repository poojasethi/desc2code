#include<iostream>
#include<string>
#include<string.h>
#include<algorithm>
#include<stdio.h>
using namespace std;
string s[60];
int dp[55][55][55][55];
int n;
int check(int x,int y)
{
	if(s[x][0]==s[y][0] || s[x][1]==s[y][1]) return 1;
	else return 0;
}
int dfs(int x,int a,int b,int c)
{
	if(!dp[x][a][b][c])
	{
		if(x==0) dp[x][a][b][c]=1;
		else
		{
			dp[x][a][b][c]=-1;
			if(x-3>=0 && check(x-3,c) && dfs(x-1,c,a,b) || check(b,c)&&dfs(x-1,max(x-3,0),a,c))
			{
				dp[x][a][b][c]=1;
			}
		}
	}
	return dp[x][a][b][c]==1;
}
int main()
{
	cin>>n;
	for(int i=0;i<n;i++) cin>>s[i];
	int flag=dfs(n-1,max(n-3,0),max(n-2,0),n-1);
	if(flag) cout<<"YES"<<endl;
	else cout<<"NO"<<endl;
}
