#include<set>
#include<vector>
#include<iostream>
using namespace std;
#define vs vector<string>
set<vs> conf; 
int n;
bool solve( vs c ){
	if( !conf.insert(c).second ) return 0;
	int q = c.size()-1;
	if( q == 0 ) return 1;
	bool ok = 0;
	if( q>0 && (c[q][0] == c[q-1][0] || c[q][1]==c[q-1][1]) ){ 
		vs nc( c.begin() , c.end()-1 );
		nc[q-1] = c[q];
		ok = solve( nc );
	}
	if( !ok && q>2 && (c[q][0] == c[q-3][0] || c[q][1]==c[q-3][1]) ) {
		vs nc( c.begin() , c.end()-1 );
		nc[q-3] = c[q];
		ok = solve( nc );
	} 
	return ok;
}
int main(){
	cin >> n; vs sc(n);
	for( int i  = 0; i <n; ++i )cin >> sc[i];

	if( solve( sc ) ) cout << "YES\n";
	else cout << "NO\n";
}
