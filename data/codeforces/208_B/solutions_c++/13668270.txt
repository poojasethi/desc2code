#include<bits/stdc++.h>
using namespace std;
map<vector<string>,int>dp;
vector<string>a;
string ans="NO";
void solve(vector<string>a)
{
      if(dp[a])
      return;
      dp[a]=1;
      int x=a.size();
      if(x==1)
      ans="YES";
      if(x>1 && (a[x-1][0]==a[x-2][0] || a[x-1][1]==a[x-2][1]))
      {
          vector<string>b=a;
            b[x-2]=a[x-1];
          b.erase(b.end()-1);
          solve(b);
      }
      if(x>3 && (a[x-4][0]==a[x-1][0] || a[x-4][1]==a[x-1][1]))
      {
          vector<string>b=a;
          b[x-4]=a[x-1];
          b.erase(b.end()-1);
          solve(b);
      }
}
int main()
{
     int n;
     cin>>n;
     a.resize(n);
      for(int i=0;i<n;i++)
       cin>>a[i];
       solve(a);
       cout<<ans<<endl;
   return 0;
}
