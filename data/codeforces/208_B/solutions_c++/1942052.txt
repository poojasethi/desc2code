#include<cstring>
#include<iostream>
using namespace std;

char get[53][2];
int bit[53][53][53][53];
int n,i;


bool rec(int a, int b, int c, int f){
if (a==1)return true;
if(bit[a][b][c][f])return false;
if(a>=0 && b>=0 && c>=0 && f>=0)bit[a][b][c][f]=true;
 if(a>1 && ( get[b][1]==get[c][1] || get[b][0]==get[c][0] ) )if( rec(a-1,b,f,a-3) )return true;
 if(a>2 && ( get[b][1]==get[a-3][1] || get[b][0]==get[a-3][0] ) )if( rec(a-1,c,f,b) ) return true;   
return false; 
}     

main(){
cin>>n;
 for(i=1;i<=n;i++)cin>>get[i]; 
 if (rec(n,n,n-1,n-2))cout<<"YES"<<endl;
 else cout<<"NO"<<endl;
}
