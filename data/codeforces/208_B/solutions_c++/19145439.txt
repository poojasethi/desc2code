#include<cstdio>
bool f[60][60],dp[60][60][60][60];
char s[60][5];
bool dfs(int n,int a,int b,int c)
{
	if(dp[n][a][b][c])return 0;
	if(n==3&&f[a][b]&&f[a][c])return 1;
	if(n<3)return 0;
	if(f[a][b])if(dfs(n-1,a,c,n-3))return 1;
	if(n>=4&&f[a][n-3])if(dfs(n-1,b,c,a))return 1;
	dp[n][a][b][c]=1;
	return 0;
}
int main()
{
	int i,j,n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%s",s[i]);
	for(i=1;i<=n;i++)
	{
		for(j=i+1;j<=n;j++)
		{
			if(s[i][0]==s[j][0]||s[i][1]==s[j][1])
				f[i][j]=f[j][i]=1;
		}
	}
	if(n==1||(n==2&&f[2][1])||dfs(n,n,n-1,n-2))
		printf("YES\n");
	else
		printf("NO\n");
	return 0;
}
