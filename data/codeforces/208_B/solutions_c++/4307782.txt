#include<iostream>
#include<cstring>
using namespace std;
char x[60],y[60];
int dp[60][60][60][60];
int f(int i,int j)
{
    if(x[i]==x[j]||y[i]==y[j])
        return 1;
    return 0;
}
int dfs(int n,int i,int j,int k)
{
    if(dp[n][i][j][k]==1)
        return 0;
    dp[n][i][j][k]=1;
    if(n==3&&f(i,j)&&f(i,k))
        return 1;
    if(n>1&&f(i,j)&&dfs(n-1,i,k,n-3))
        return 1;
    if(n>3&&f(i,n-3)&&dfs(n-1,j,k,i))
        return 1;
    return 0;
}
int main()
{
    memset(dp,0,sizeof(dp));
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
        cin>>x[i]>>y[i];
    if(n==1||(n==2&&f(1,2))||dfs(n,n,n-1,n-2))
        cout<<"YES\n";
    else
        cout<<"NO\n";
}
