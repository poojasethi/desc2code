#include <bits/stdc++.h>
using namespace std;
int a[60],b[60];
int dp[60][60][60][60];
bool is(int x,int y){
	return (a[x]==a[y]) || (b[x]==b[y]);
}
int sol(int n,int x,int y,int z){
	if(n==3)return 1;
	int &ret= dp[n][x][y][z];
	if(ret!=-1)return ret;
	ret=0;
	if(is(y,z))
		ret= max(ret,sol(n-1,n-3,x,z));
	if(ret==1)return ret;
	if(is(n-3,z)&&n>5)
		ret=max(ret,sol(n-1,z,x,y) );
	return ret;
}
int main(){
	int n;
	cin>>n;string s;
	for(int i=3;i<n+3;++i){
		cin>>s;a[i]= s[0],b[i]=s[1];
	}
	memset(dp,-1,sizeof(dp));
	if(n==1 || sol(n+2,n,n+1,n+2) )cout<<"YES\n";
	else cout<<"NO\n";
}
