#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>
using namespace std;

const int MAX=55;
bool flag[MAX][MAX][MAX][MAX];
string data[MAX];


bool cmp(int i, int j) {
    return data[i][0] == data[j][0] || data[i][1] == data[j][1];
}


bool f(int pos, int x, int y, int z) {
    if (pos == 0) return true;
    if (flag[pos][x][y][z]) return false;
    flag[pos][x][y][z] = true;

    if (pos >= 1 && cmp(x, y) && f(pos - 1, x, z, pos - 3)) return true;
    if (pos >= 3 && cmp(x, pos - 3) && f(pos - 1, y, z, x)) return true;
    return false;
}


int main() {
    int i, n;
    cin >> n;
    for (i = 0; i < n; i++) cin >> data[i];
    memset(flag, false, sizeof(flag));
    bool result = f(n - 1, n - 1, n - 2, n - 3);
    cout << (result ? "YES" : "NO") << endl;
}
