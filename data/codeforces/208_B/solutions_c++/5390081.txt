

#include<cstdio>
#include<cstring>
using namespace std;
int n,a[100],b[100],dp[60][60][60][60];

int Ok(int x,int y)
{
	return (a[x]==a[y]||b[x]==b[y]);
}

int Dfs(int i,int j,int k,int p)
{
	if (dp[i][j][k][p])return 0;
	if (i==3) if (Ok(j,p)&&Ok(k,p)) return 1;
	if (Ok(k,p)) if (Dfs(i-1,i-3,j,p)) return 1;
	if (Ok(i-3,p)) if (Dfs(i-1,p,j,k)) return 1;
	dp[i][j][k][p]=1;
	return 0;
}

int main()
{
	char c;
	scanf("%d\n",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%c",&c); a[i]=c;
		scanf("%c",&c); b[i]=c;
		getchar();
	}
	if (Dfs(n,n-2,n-1,n)||n==1) 
		puts("YES"); else puts("NO");
	return 0;
}
