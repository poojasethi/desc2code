#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<string>
#include<iostream>
using namespace std;
char a[60][2];
bool can(int x1,int x2)
{
    return a[x1][0]==a[x2][0]||a[x1][1]==a[x2][1];
}
int dp[60][60][60][60];
int dfs(int n,int x1,int x2,int x3)
{
    int &ans=dp[n][x1+3][x2+3][x3+3];
    if(ans!=-1)return ans;
    if(n==1)return ans=1;
    ans=0;
    if(can(x1,x2)&&dfs(n-1,x1,x3,n-3))return ans=1;
    if(n-3>=1&&can(x1,n-3)&&dfs(n-1,x2,x3,x1))return ans=1;
    return ans=0;
}
int main()
{
    int n;
    while(~scanf("%d",&n))
    {
        for(int i=1; i<=n; i++)
        {
            cin>>a[i];
        }
        memset(dp,-1,sizeof(dp));
        if(dfs(n,n,n-1,n-2))
            puts("YES");
        else
            puts("NO");
    }
}
