#include <iostream>
#include <cstdio>
#include <cstring>

using namespace std;
char str[111][5];
int f[55][55][55][55];
int n;

int dfs(int num,int a,int b,int c)
{
    int ret=0;
    if(num==1) return 1;
    if(f[num][a][b][c]!=-1) return f[num][a][b][c];
    if( num>=2 && (str[a][0]==str[b][0]||str[a][1]==str[b][1]) )
        ret=dfs(num-1,a,c,num-3);
    if( !ret&&num>=4 && (str[a][0]==str[num-3][0]||str[a][1]==str[num-3][1]) )
        ret=dfs(num-1,b,c,a);
    f[num][a][b][c]=ret;
    return ret;
}

int main()
{
    memset(f,-1,sizeof(f));
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%s",str[i]);
    if(n==1 || (n==2&&(str[1][0]==str[2][0]||str[1][1]==str[2][1])) || (n>=3&&dfs(n,n,n-1,n-2)) )
        printf("YES\n");
    else printf("NO\n");
    return 0;
}
