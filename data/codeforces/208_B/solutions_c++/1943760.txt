#include <stdio.h>
#include <vector>
#include <string>
#include <set>
#include <iostream>
#include <stdlib.h>
using namespace std;

set< vector<string> > s;
int n;

void go(vector<string> v){
    if(s.find(v)!=s.end())return;
    s.insert(v);
    int p=v.size()-1;
    if(!p)printf("YES"),exit(0);

    if(v[p][0]==v[p-1][0] || v[p][1]==v[p-1][1]){
        vector<string> vv(v.begin(),v.end()-1);
        vv[p-1]=v[p];
        go(vv);
    }
    if(p>2 && (v[p][0]==v[p-3][0] || v[p][1]==v[p-3][1])){
        vector<string> vv(v.begin(),v.end()-1);
        vv[p-3]=v[p];
        go(vv);
    }
    

}
int main(){
    scanf("%d",&n);
    vector<string> v(n); 
    for(int i=0;i<n;i++)cin>>v[i];
    go(v);
    printf("NO");
}
