#include <iostream>
#include <stdio.h>
#include <vector>
#include <map>
using namespace std;
string ans="NO";
int n;
typedef vector<string> vs;
vs a;
map<vs,int> v;

int rec(vs a)
{
 if (v[a]) return 0;
 v[a]=1;
 int n=a.size();
 if (n==1) ans="YES";

 if (n>=2&&(a[n-1][0]==a[n-2][0]||a[n-1][1]==a[n-2][1]) )
 {
  vs b=a;
  b[n-2]=b[n-1];
  b.erase(b.begin()+n-1);
  rec(b);
 }
if (n>=4&&(a[n-1][0]==a[n-4][0]||a[n-1][1]==a[n-4][1]) )
 {
  vs b=a;
  b[n-4]=b[n-1];
  b.erase(b.begin()+n-1);
  rec(b);
 }
}


int main()
{
  //freopen("i.txt","r",stdin);
cin>>n;
  a.resize(n);
   for (int i=0; i<n;i++) cin>>a[i];
   rec(a);
  cout<<ans;
 return 0;
}
