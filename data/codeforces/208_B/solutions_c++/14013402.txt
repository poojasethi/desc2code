#include<bits/stdc++.h>

using namespace std;

#define N 62

int M[N][N][N][N],T[N],n,i;

char A[N],B[N];

void f(int i){

    if(i == 1)
    { cout << "YES"; exit(0); }

    int j , t , &r = M[i][ T[i] ][ T[i-1] ][ T[i-2] ];

    if(r != -1)
        return;

    r = 0;

    for(t=1,j=2;j--;t=3)
        if(A[ T[i] ] == A[ T[i-t] ] || B[ T[i] ] == B[ T[i-t] ]){
            swap(T[i] , T[i-t]);
            f(i-1);
            swap(T[i] , T[i-t]);
        }
}
int main(){

    cin >> n;

    for(i=1;i<=n;i++){
        cin >> A[i] >> B[i];
        getchar();
        T[i] = i;
    }

    memset(M , -1 , sizeof M);
    f(n);
    cout << "NO";
    return 0;
}
