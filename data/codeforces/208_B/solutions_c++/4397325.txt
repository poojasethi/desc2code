#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
int d[55][55][55][55];
char s[55][3];
bool pd;
void dp(int i,int j,int k,int l)
{
    if(pd||(i==1))
    {
        pd=true;
        return ;
    }
    if(i>1&&(s[k][0]==s[l][0]||s[k][1]==s[l][1])&&d[i-1][i-3][j][l]==0)
    {
        d[i-1][i-3][j][l]=1;
        dp(i-1,i-3,j,l);
    }
    if(i>3&&(s[i-3][0]==s[l][0]||s[i-3][1]==s[l][1])&&d[i-1][l][j][k]==0)
    {
        d[i-1][l][j][k]=1;
        dp(i-1,l,j,k);
    }
}
int main()
{
    //freopen("in.txt","r",stdin);
    int n,i,j;
    while(scanf("%d",&n)!=-1)
    {
        for(i=1;i<=n;i++)
            scanf("%s",s[i]);
        pd=false;
        dp(n,n-2,n-1,n);
        if(pd)
            printf("YES\n");
        else
            printf("NO\n");
    }
    return 0;
}
