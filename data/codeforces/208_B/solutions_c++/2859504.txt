#include <iostream>
#include <map>
#include <vector>
#include <cstdlib>
#include <cstdio>

using namespace std;

vector<string> a;
map<vector<string>, int> w;

void rec (vector<string> a) {
	if (w[a]) return;
	w[a] =  1;
	int len = (int)a.size();
	if (len == 1) { cout << "YES"; exit(0);}
	if (a[len-1][0] == a[len-2][0] || a[len-1][1] == a[len-2][1]) {
		vector<string> b = a;
		b[len-2] = b[len-1];
		b.erase(b.end()-1);
		rec(b);	
	}
	if (len >= 4 && (a[len-1][0] == a[len-4][0] || a[len-4][1] == a[len-1][1])) {
		vector<string> b=a;
		b[len-4] = b[len-1];
		b.erase(b.end()-1);
		rec(b);
	}
}

int main () {
	int n;
	cin >> n;
	a.resize(n);
	for (int i = 0; i < n; i++) cin >> a[i];
	rec(a);
	cout << "NO";
}
