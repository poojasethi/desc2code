#include<cstdio>
#include<cstring>
using namespace std;
#define MAX 55
bool dp[MAX][MAX][MAX][MAX];
char num[MAX][5];
bool check(int x,int y)
{
    if(num[x][0]==num[y][0]||num[x][1]==num[y][1])
        return true;
    return false;
}
bool dfs(int pos,int x,int y,int z)
{
    if(pos==0) return true;
    if(z>=0)
    {
        if(dp[pos][x][y][z]) return false;
        dp[pos][x][y][z]=true;
    }
    if(check(x,y)&&dfs(pos-1,x,z,pos-3))
        return true;
    if(pos-3>=0&&check(x,pos-3)&&dfs(pos-1,y,z,x))
        return true;
    return false;
}
int main()
{
    int n;
    scanf("%d",&n);
    memset(dp,false,sizeof(dp));
    for(int i=0; i<n; ++i)
        scanf("%s",num[i]);
    if(dfs(n-1,n-1,n-2,n-3))
        puts("YES");
    else
        puts("NO");
    return 0;
}
