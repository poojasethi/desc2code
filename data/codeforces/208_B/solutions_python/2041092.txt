#!/usr/bin/env python
n=input()
c=raw_input().split()
import Queue
q= Queue.Queue()
q.put(c)
v=set([tuple(c)])
while not q.empty():
    c=q.get();
    if len(c)==1:
        print"YES"
        exit()
    for i in[-2,-4]:
        if len(c)>=-i and(c[-1][0]==c[i][0]or c[-1][1]==c[i][1]):
            d=list(c)
            d[i+1]=d.pop();
            if tuple(d)not in v:
                v.add(tuple(d))
                q.put(d)
print"NO"
