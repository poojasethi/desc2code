def check(cards):
	length = len(cards);
	if length == 1:
		return True
	else:
		lastCard = cards[length - 1];
		checkCard = cards[length - 2];

		if lastCard[0] == checkCard[0] or lastCard[1] == checkCard[1]:
			newCards = cards[0:length - 2];
			newCards.append(lastCard);
#			print "newCards", newCards
			newCardsString = ''.join(newCards);
			if newCardsString not in memory:
				ans = check(newCards);
				memory[newCardsString] = ans;
			else:
				ans = memory[newCardsString];
			if ans:
				return ans;

		if length >= 4:
			checkCard = cards[length - 4];
			if lastCard[0] == checkCard[0] or lastCard[1] == checkCard[1]:
				newCards = cards[0:length - 1];
				newCards.pop(length - 4);
				newCards.insert(length - 4, lastCard);
#				print "newCards", newCards
				newCardsString = ''.join(newCards);
				if newCardsString not in memory:
					ans = check(newCards);
					memory[newCardsString] = ans;
					return ans;
				else:
					ans = memory[newCardsString];
				return ans;
	
	return False;
