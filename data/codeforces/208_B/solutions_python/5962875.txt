n = input()
top = raw_input().split()
stack = [top]
st = set([tuple(top)])
found = False
while (len(stack)):
	top = stack.pop()
	if len(top)==1:
		found = True
		break;
	for i in [-2,-4]:
		if len(top)>=-i and (top[-1][0]==top[i][0] or top[-1][1]==top[i][1]):
			tmp = list(top)
			last = tmp.pop()
			tmp[i+1] = last
			if tuple(tmp) not in st:
				st.add(tuple(tmp))
				stack.append(tmp)
if found:
	print "YES"
else:
	print "NO"
