#! /usr/bin/python2

import sys
import math

n = int(raw_input())
s = ""
for i in raw_input().split():
    s = s + i
    
ans = {'':1}
def ok(a):
    if (len(a) == 2):
        return 1
    if (a in ans):
        return ans[a]
    if (a[-1] == a[-3] or a[-2] == a[-4]):
        if (ok(a[: -4] + a[-2: ])):
            ans[a] = 1
            return 1
    if (len(a) > 7 and (a[-1] == a[-7] or a[-2] == a[-8])):
        if (ok(a[: -8] + a[-2: ] + a[-6: -2])):
            ans[a] = 1
            return 1
    ans[a] = 0
    return 0

print(["NO", "YES"][ok(s)])
