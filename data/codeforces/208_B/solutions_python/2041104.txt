#!/usr/bin/env python
n=input()
card = raw_input().strip().split()
import Queue
q = Queue.Queue()
q.put(card)
s = set([tuple(card)])
res = "YES"
while not q.empty():
    c = q.get();
    if len(c)==1:
        break
    for i in(-2,-4):
        if len(c)>=-i and(c[-1][0]==c[i][0]or c[-1][1]==c[i][1]):
            d=list(c)
            d[i+1]=d.pop();
            if tuple(d) not in s:
                s.add(tuple(d))
                q.put(d)
else:
    res = "NO"
print res
