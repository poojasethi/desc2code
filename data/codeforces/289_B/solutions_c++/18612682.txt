#include<bits/stdc++.h>
using namespace std;
int a[10005],n,m,r,i,ans,d;
main()
{
    cin>>n>>m>>d;
    n*=m;
    for (i=0;i<n;i++)
        scanf("%d",&a[i]);
    r=a[0]%d;
    for (i=1;i<n;i++) if (a[i]%d!=r) return cout<<-1,0;
    sort(a,a+n);
    for (i=0;i<n;i++)
        ans+=abs(a[i]-a[n/2]);
    cout<<ans/d;
}
