#include <iostream>
#include <cmath>
#include <algorithm>

using namespace std;

int main(){
	int a[100010],n,m,ans=0,d,r,i;
	cin>>n>>m>>d;
	n*=m;
	cin>>a[0];
	r=a[0]%d;
	for(i=1;i<n;i++){
		cin>>a[i];
		if(a[i]%d!=r){
			cout<<-1;
			return 0;
		}
	}
	sort(a,a+n);
	r=a[n/2];
	for(i=0;i<n;i++)
		ans+=abs(a[i]-r)/d;
	cout<<ans;
	return 0;
}
