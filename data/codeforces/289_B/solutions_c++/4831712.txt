#include <cstdio>
#include <algorithm>
using namespace std;

int a[10010];
int n, m, k;

int main() {
	scanf("%d%d%d", &n, &m, &k);
	n *= m;
	for (int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
		if ((a[i] - a[0]) % k) {
			printf("-1\n");
			return 0;
		}
	}
	int ans = 0;
	sort(a, a + n);
	for (int i = 0; i < n; ++i) ans += abs(a[i] - a[n / 2]);
	printf("%d\n", ans / k);
	return 0;
}
