#include <iostream>
#include <cmath>
#include <algorithm>
using namespace std;
int main ()
{
	int n,m,d,x;
	cin>>n>>m>>d;
	n*=m;
	int a[n];
	cin>>a[0];
	for (int i=1;i<n;i++){
		cin>>a[i];
		if (a[i]%d!=a[0]%d){
			cout<<-1;
			return 0;
		}
	}
	long long ans=0;
	sort (a,a+n);
	int t=a[n/2];
	for (int i=0;i<n;i++){
		ans+=abs(a[i]-t)/d;
	}
	cout<<ans;
	return 0;
}
