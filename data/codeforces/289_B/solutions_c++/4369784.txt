#include<iostream>
#include<stdio.h>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
int i,n,m,d,a[100005],ans=0;
int main(){
	cin>>n>>m>>d;
	for(i=1;i<=n*m;i++)
		cin>>a[i];
	sort(a+1,a+1+n*m);
	for(i=2;i<=n*m;i++)
		if ((a[i]-a[i-1])%d!=0){
			cout<<-1;
			return 0;
		}
	for(i=1;i<=n*m;i++)
		ans+=abs(a[i]-a[(n*m+1)/2])/d;
	cout<<ans;
	return 0;
}
