#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int a[110*110];
int main(){
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=1;i<=n*m;i++){
		cin>>a[i];
		if(a[i]%k!=a[1]%k)
		{
			cout<<-1;
			return 0;
		}
	}
	int d=n*m;
	sort(a+1,a+(d)+1);
	int cnt=a[d/2+1];
	int sum=0;
	for(int i=1;i<=d;i++)
	{
		int l=abs(cnt-a[i])/k;
		sum+=l;
	}
	cout<<sum;
}
