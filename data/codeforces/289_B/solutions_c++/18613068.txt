#include<bits/stdc++.h>

using namespace std;

int main()
{long long int n,m,d,i,arr[10001],r,ans=0;
cin>>n>>m>>d;

n*=m;
    for (i=0;i<n;i++)
        scanf("%lld",&arr[i]);
    r=arr[0]%d;
    for (i=1;i<n;i++) if (arr[i]%d!=r)
    {cout<<"-1";
        return 0;}
    sort(arr,arr+n);
    for (i=0;i<n;i++)
        ans+=abs(arr[i]-arr[n/2]);
    cout<<ans/d;

return 0;
}
