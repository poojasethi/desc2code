#include<bits/stdc++.h>

using namespace std;
int main()
{
	int i,k,l,m,n,a,b,d;
	cin>>n>>m>>d;
	vector<int> vect(n*m,0);
	for(i=0;i<n*m;i++)
	{
		cin>>vect[i];
	}
	k=0;
	sort(vect.begin(),vect.end());
	for(i=0;i<n*m;i++)
	{
		if((vect[(n*m)/2]-vect[i])%d!=0)
		{
			cout<<-1;
			return 0;
		}
		else
		{
			k+=abs((vect[(m*n)/2]-vect[i])/d);
		}
	}
	cout<<k;
	return 0;
}
