#include<iostream>
#include<cstdlib>
#include<algorithm>
using namespace std;
int main()
{
	int i,n,m,d,a[10000];
	cin>>n>>m>>d;
	int total=n*m;
	cin>>a[0];
	int remainder=a[0]%d;
	for(i=1;i<total;i++)
	{
		cin>>a[i];
		if(a[i]%d != remainder) 
		{
			cout<<"-1";
			return 0;
		}
	}
	sort(a,a+total);
	int mid=total/2,diff=0;
	for(int i=0;i<total;i++)
	{
		diff+=abs(a[mid]-a[i]);
	}
	cout<<diff/d;

}
