#include <bits/stdc++.h>
typedef long long ll;
using namespace std;
int a[1000010];
int main()
{
	int n, m, k, r, w;
	cin>>n>>m>>k;
	n*=m;
	cin>>a[0];
	r=a[0]%k;
	for(int i=1;i<n;i++)
	{
		cin>>a[i];
		if(a[i]%k!=r)
		{
			cout<<-1;
			return 0;
		}
	}
	sort(a,a+n);
	r=a[n/2];
	w=0;
	for(int i=0;i<n;i++)
	{
		w+=abs(a[i]-r)/k;
	}
	cout<<w;
	return 0;
}
