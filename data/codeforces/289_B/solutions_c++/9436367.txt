#include<cstdio>
#include<algorithm>
using namespace std;
int a1[10000];
int main()
{
    int n,m,d,o=0;
    scanf("%d%d%d",&n,&m,&d);
	n*=m;m=n/2;
    for(int i=0;i<n;++i)
		scanf("%d",a1+i);
	sort(a1,a1+n);
	for(int i=0;i<n;++i)
	{
		if((a1[i]-a1[m])%d)
		{puts("-1");return 0;}
		o+=abs(a1[i]-a1[m])/d;
	}
	printf("%d\n",o);
}
