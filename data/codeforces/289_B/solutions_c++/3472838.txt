#include<stdio.h>
#include<iostream>
#include<limits.h>
#include<algorithm>
using namespace std;
int a[100000];
int n,m,d;

int main()
{
	cin>>n>>m>>d;
	int i;
	for(i=0;i<n*m;i++)
	{
		cin>>a[i];
	}
	sort(a,a+n*m);
	int tmp=a[(n*m+1)/2-1];
	int ans=0;
	for(i=0;i<n*m;i++)
	{
		if((a[i]-tmp)%d!=0)
		{
			printf("-1\n");
			return 0;
		}
		else
		{
			ans+=((abs(a[i]-tmp))/d);
		}
	}
	cout<<ans<<endl;
		



}
