n, m, d = map(int, raw_input().split())
mod = None               
a, elem = [None] * n, []
for i in range(n):
    a[i] = tuple(map(int, raw_input().split()))
    elem += a[i] 
    for j in a[i]:
        if mod is None:
            mod = j % d
        elif mod != j % d:
            print -1
            exit(0)
elem = sorted(elem)[n * m // 2]
mod = 0
for i in range(n):
    for j in range(m):
        mod += abs(a[i][j] - elem) // d
print mod
