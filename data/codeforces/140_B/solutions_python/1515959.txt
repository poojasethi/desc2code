n=input()
a=[map(int,raw_input().split()) for _ in range(n+1)]
v=[0]*n
for i in range(n):
    p=[0]*(n+1)
    for j in range(n): p[a[i][j]]=j
    b=12345
    u,t=0,12345
    for x in a[n]:
        if x != i+1 and x<b:
            if p[x]<t: u,t=x,p[x]
            b=x
    v[i]=u
print ' '.join(map(str,v))
