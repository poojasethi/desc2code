#!/usr/bin/python

import sys

n = int(sys.stdin.readline())
priority = []

for s in sys.stdin:
	buf = []
	s = s + ' '
	i = 0
	j = s.find(' ')
	while i < len(s):
		buf.append(int(s[i:j]))
		i = j + 1
		j = s.find(' ', i)
	priority.append(buf)

best = [1]
almost_best = [0]
for i in range(2, n + 1):
	if priority[-1].index(i) < priority[-1].index(best[-1]):
		best.append(i)
		almost_best.append(best[-2])
	else:
		best.append(best[-1])
		if i == 2 or priority[-1].index(i) < priority[-1].index(almost_best[-1]):
			almost_best.append(i)
		else:
			almost_best.append(almost_best[-1])
#print best
#print almost_best
for k, p in enumerate(priority[:-1]):
	r = [0] * (n + 1)
	for i, j in enumerate(p):
		r[j] = i
	def f(i):
		if best[i] == k + 1:
			return almost_best[i]
		else:
			return best[i]
	ans = 0
	if k == 0:
		ans = 1
	for i in xrange(n):
		if i != k and r[f(i)] < r[f(ans)]:
			ans = i
