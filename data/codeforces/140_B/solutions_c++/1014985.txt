#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define N 305

int n;
int a[N][N], r[N];
int f[N];

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i)
		for (int j = 1; j <= n; ++j) {
			int x;
			scanf("%d", &x);
			a[i][x] = j;
		}
	for (int i = 1; i <= n; ++i) {
		int x;
		scanf("%d", &x);
		r[x] = i;
	}
	for (int i = 1; i <= n; ++i) {
		f[i] = -1;
		int mi = -1;
		for (int j = 1; j <= n; ++j)
			if (j != i) {
				if (-1 == mi || r[j] < r[mi]) {
					mi = j;
					if (-1 == f[i] || a[i][mi] < a[i][ f[i] ])
						f[i] = mi;
				}
			}
	}
	for (int i = 1; i < n; ++i)
		printf("%d ", f[i]);
	printf("%d\n", f[n]);
	return 0;
}
