#include <cstdio>

int f[310][310];

int main()
{
	int n;
	scanf("%d",&n);
	for(int i = 1; i <= n; i++)
		for(int j = 1;j <= n; j++)
			scanf("%d", &f[i][j]);

	for(int i = 1; i <= n; i++)
		scanf("%d",&f[0][i]);
	
	for(int i = 1; i <= n; i++)
		for(int j = 1; j <= n; j++){
			if(i == f[i][j])
				continue;
				
			int k;
			for(k = 1; f[0][k] != f[i][j]; k++)
				if (f[0][k]!=i && f[0][k] < f[i][j])
					break;
					
			if(f[0][k] == f[i][j]){
				printf("%d ",f[0][k]);
				break;
			}
		}

	return 0;
}
