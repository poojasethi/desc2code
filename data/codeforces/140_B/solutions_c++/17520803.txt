#include<stdio.h>

int pre[310][310];
int n;

int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=n;j++)
        {
            int temp;
            scanf("%d",&temp);
            pre[i][temp]=j;
        }
    for(int i=1;i<=n;i++)
    {
        int temp;
        scanf("%d",&temp);
        pre[0][temp]=i;
    }
    for(int i=1;i<=n;i++)
    {
        int my=i==1?2:1;
        int you=my;
        for(int j=1;j<=n;j++)
        {
            if(i==j) continue;
            if(pre[0][j]<pre[0][my]) my=j;
            if(pre[i][my]<pre[i][you]) you=my;
        }
        printf("%d ",you);
    }
    puts("");
    return 0;
}
