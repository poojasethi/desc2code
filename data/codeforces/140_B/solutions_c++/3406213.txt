#include <stdio.h>
#include <iostream>
#include <math.h>
#include <string.h>
#include <algorithm>
#include <string>
#define inf 10000000
using namespace std;
int b[400], a[400][400];
int main()
{
  int n,i,j,c,res,temp,x;
  scanf("%d", &n);
  for (i=0; i<n; i++)
    for (j=0; j<n; j++)
      scanf("%d", &x), a[i][x-1]=j;
  for (i=0; i<n; i++)
    scanf("%d", &x), b[x-1]=i;
  for (i=0; i<n; i++)
  {
    res=inf, temp=-1, c=inf;
    for (j=0; j<n; j++)
    {
      if (i==j)
        continue;
      c=min(c,b[j]);
      if (c==b[j] && res>a[i][j])
        res=a[i][j], temp=j;
    }
    printf("%d ", temp+1);
  }
  printf("\n");
  return 0;
}
