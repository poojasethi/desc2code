#include <cstdio>

int n, a[305][305], b[305], s[305];

bool check(int k, int m) {
	for (int i = 1; i <= n && b[i] != m; ++i) {
		if (b[i] == k) continue;
		if (b[i] < m) return false;
	}
	return true;
}

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= n; ++j) {
			scanf("%d", &a[i][j]);
		}
	}
	for (int i = 1; i <= n; ++i) {
		scanf("%d", &b[i]);
	}

	for (int i = 1; i <= n; ++i) {
		for (int j = 1; j <= n; ++j) {
			if (a[i][j] == i) continue;
			if (check(i, a[i][j])) {
				s[i] = a[i][j];
				break;
			}
		}
	}
	for (int i = 1; i <= n; ++i) {
		printf("%d%c", s[i], i==n?'\n':' ');
	}
}
