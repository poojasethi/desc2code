#include<cstdio>
int n,f[303][303],my[303];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		for(int j=1;j<=n;j++){
			scanf("%d",&f[i][j]);
		}
	}
	for(int i=1;i<=n;i++)	scanf("%d",my+i);
	for(int i=1,j=1;i<=n;i++){
		for(j=1;j<=n;j++){
			if(i==f[i][j])	continue;
			bool flag=false;
			for(int k=1;my[k]!=f[i][j];k++){
				if(my[k]==i)	continue;
				if(my[k]<f[i][j]){
					flag=true;
					break;
				}
			}
			if(!flag)	break;
		}
		printf(i==n?"%d\n":"%d ",f[i][j]);
	}
	return 0;
}
