#include <cstdio>

int n,i,j,k,a[400][400],b[400];

int main()
{
	scanf("%d",&n);
	for (i=1;i<=n;++i)
		for (j=1;j<=n;++j)
			scanf("%d",&a[i][j]);
	for (i=1;i<=n;++i)
		scanf("%d",&b[i]);
	for (i=1;i<=n;++i)
		for (j=1;j<=n;++j)
		{
			if (i==a[i][j]) continue;
			for (k=1;b[k]!=a[i][j];++k)
				if (b[k]<a[i][j]&&b[k]!=i)
					break;
			if (b[k]==a[i][j])
			{
				printf("%d",a[i][j]);
				putchar(i==n?'\n':' ');
				break;
			}
		}
	return 0;
}
