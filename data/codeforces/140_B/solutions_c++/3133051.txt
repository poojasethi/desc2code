
#include <iostream>
#include <cstdlib>
#include <cstring>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <vector>
#include <queue>
#define FOR(it,c) for ( __typeof((c).begin()) it=(c).begin(); it!=(c).end(); it++ )

typedef long long int LL;

using namespace std;

int n;
int p[310][310],q[310][310];

int main(){

	scanf("%d",&n);
	for(int i=0;i<n+1;i++){
		for(int j=0;j<n;j++)scanf("%d",&p[i][j]);
		for(int j=0;j<n;j++)q[i][p[i][j]-1]=j;
		q[i][n]=n+4;
	}
	for(int i=0;i<n;i++){
		int ai=n, pi=n;
		for(int j=0;j<n;j++){
			if(q[n][pi]>q[n][j] && i!=j) pi=j;
			if(q[i][ai]>q[i][pi]) ai=pi;
		}	
		printf("%d%c",ai+1,i==n-1?'\n':' ');
	}
	return 0;
}
