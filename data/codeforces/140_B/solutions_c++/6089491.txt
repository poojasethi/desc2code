#include <stdio.h>
#define N 310
int f[N][N],ans[N][N],a[N];
int main()
{
	int n,i,j,x,y;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			scanf("%d",&x);
			f[i][x]=j;
		}
		f[i][n+1]=n+1;
	}
	for(i=1;i<=n;i++)
	{
		scanf("%d",&x);
		a[x]=i;
	}
	a[n+1]=n+1;
	for(i=1;i<=n;i++)
	{
		x=y=n+1;
		for(j=1;j<=i;j++)if(a[j]<a[x])x=j;
		for(j=1;j<=i;j++)if(x!=j&&a[j]<a[y])y=j;
		ans[i][x]=y;
		for(j=1;j<=n;j++)if(x-j)ans[i][j]=x;
	}
	for(i=1;i<=n;i++)
	{
		x=1;
		for(j=2;j<=n;j++)if(f[i][ans[j][i]]<f[i][ans[x][i]])x=j;
		printf("%d ",ans[x][i]);
	}
	printf("\n");
	return 0;
}
