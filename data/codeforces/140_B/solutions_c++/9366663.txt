#include <iostream>
#include <cstdio>

using namespace std;

int N;
int rk[500][500];

int main()
{
    scanf("%d", &N);

    for(int i = 1; i <= N; i++)
        for(int j = 1; j <= N; j++)
        {
            int t; scanf("%d", &t);
            rk[i][t] = j;
        }


    for(int j = 1; j <= N; j++)
    {
        int t; scanf("%d", &t);
        rk[0][t] = j;
    }

    for(int i = 1; i <= N; i++)
    {
        int now = ( i==1 ? 2 : 1 ), mx = now;

        for(int j = 1; j <= N; j++)
        {
            if( i == j ) continue;

            if( rk[0][j] < rk[0][now] ) now = j;
            if( rk[i][now] < rk[i][mx] ) mx = now;
        }

        printf("%d ", mx);
    }
    printf("\n");
}
