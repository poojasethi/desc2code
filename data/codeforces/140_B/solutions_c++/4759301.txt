#include<iostream>

using namespace std;

int main()
{
	int n,a[301][301], arr[301], p[301];
	bool can[301][301] = {0};
	cin>>n;
	for (int i=0;i<n;i++)
		for (int j=0;j<n;j++)
		{
			cin>>a[i][j];
			a[i][j]--;
		}
	for (int i=0;i<n;i++)
	{
		cin>>arr[i];
		p[arr[i] - 1] = i;
	}
	for (int i=0;i<n;i++)
	{
		for (int k=0;k<n;k++)
		{
			can[i][k] = 1;
			for (int j=0;j<i && can[i][k];j++)
			{
				if (j == k)
					continue;
				if (p[j] < p[i])
					can[i][k] = 0;
			}
		}
	}
	for (int i=0;i<n;i++)
	{
		int best = -1;
		for (int j=0;j<n && best == -1;j++)
		{
			if (a[i][j] == i)
				continue;
			if (can[a[i][j]][i] )
				best = a[i][j];
		}
		if (i)
			cout<<" ";
		cout<<best + 1;
	}
	return 0;
}
