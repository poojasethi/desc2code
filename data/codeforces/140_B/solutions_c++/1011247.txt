#include <iostream>
#include <vector>

using namespace std;

int mejor(vector<int> s,vector<int> v)
{
  for (int i=0;i<int(v.size());i++) {
    int x=v[i];
    if (s[x]) return x;
  }
  return -1;
}

int main()
{
  int n;
  cin>>n;
  vector<vector<int> > v(n,vector<int> (n));
  for (int i=0;i<n;i++) {
    for (int j=0;j<n;j++) {
      cin>>v[i][j];
    }
  }
  vector<int> w(n);
  for (int i=0;i<n;i++) cin>>w[i];
  for (int i=0;i<n;i++) {
    vector<int> saux(n+1,0),s(n+1,0);
    for (int j=0;j<n;j++) {
      if (j!=i) {
	saux[j+1]=1;
	s[mejor(saux,w)]=1;
      }
    }
    if (i>0) cout<<" ";
    cout<<mejor(s,v[i]);
  }
  cout<<endl;
}
