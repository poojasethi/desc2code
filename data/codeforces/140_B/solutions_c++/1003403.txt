#include <cstdio>
#include <algorithm>

using namespace std;

const int h = 333;
const int u = 1 << 30;

int n, x;
int p[h], b[h];
int f[h][h];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		f[i][0] = u;
		for (int j = 0; j < n; j++) {
			scanf("%d", &x);
			f[i][x] = j;
		}
	}
	for (int i = 0; i < n; i++) {
		scanf("%d", &x);
		p[x] = i;
	}
	p[0] = u;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < n; j++) {
			int cur = 0;
			for (int k = 1; k <= i+1; k++)
				if (k != j+1 && p[k] < p[cur])
					cur = k;
			if (f[j][cur] < f[j][b[j]])
				b[j] = cur;
		}
	}
	for (int i = 0; i < n; i++)
		printf("%d ", b[i]);
	puts("");
	return 0;
}
