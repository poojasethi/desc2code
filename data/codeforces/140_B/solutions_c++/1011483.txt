#include <iostream>
#include <stdio.h>
#define N 300

using namespace std;

int rank[N + 1], a[N + 1][N + 1];

int main()
{
    int n, i, j, x, ans, p, r;
    scanf("%d", &n);
    for (i = 1; i <= n; i++)
        for (j = 1; j <= n; j++) scanf("%d", &x), a[i][x] = j;
    for (i = 1; i <= n; i++)
        scanf("%d", &x), rank[x] = i;
    for (i = 1; i <= n; i++) {
        p = n + 1; r = n + 1;
        for (j = 1; j <= n; j++) {
            if (i == j) continue;
            if (rank[j] < p) p = rank[j];
            if (p == rank[j] && r > a[i][j])
                r = a[i][j], ans = j;
        }
        printf("%d ", ans);
    }
    return 0;
}
