#include<iostream>
using namespace std;
int main()
{
int n,i,j,l,a[305][305],b[305],x,y;
cin>>n;
for(i=1;i<=n;i++)
{
for(j=1;j<=n;j++)
{
cin>>a[i][j];
}
}
for(j=1;j<=n;j++) cin>>a[0][j],b[j]=1;
b[1]=2;
for(i=1;i<=n;i++)
{
x=y=0;
for(l=1;l<=n;l++)
{
if(i>=a[0][l] && x==0) x=a[0][l];
else if(i>=a[0][l] && y==0)
{
y=a[0][l];
break;
}
}
for(j=1;j<=n;j++)
{
if(j!=x)
{
for(l=1;l<=n;l++)
{
if(a[j][l]==x)
{
b[j]=x;
break;
}
if(a[j][l]==b[j]) break;
}
}
else
{
for(l=1;l<=n;l++)
{
if(a[j][l]==y)
{
b[j]=y;
break;
}
if(a[j][l]==b[j]) break;
}
}
}
}
for(i=1;i<=n;i++)
{
cout<<b[i]<<" ";
}
return 0;
}
