#include<iostream>
#include<vector>
#include<cstdio>
using namespace std;
#define inf 1000000000
int arr[301][301];
int B[300];
int main()
{
		int n;
		cin>>n;
		int temp;
		for(int i=0;i<n;i++)
			for(int j=0;j<n;j++)
				scanf("%d",&temp),arr[i][temp-1]=j;
		for(int i=0;i<n;i++)
			scanf("%d",&temp),B[temp-1]=i;

		for(int i=0;i<n;i++)
		{

			int id=-1,b=inf,res=inf;
			for(int j=0;j<n;j++)
			{
				if(i==j)
					continue;
				b=min(b,B[j]);
			if(b==B[j]&&res>arr[i][j])
				res=arr[i][j],id=j;

			}
			cout<<id+1<<" ";
		}
		return 0;
}
