#include <stdio.h>

int f[301][301];

int main()
{
	int n,i,j,k;
	scanf("%d",&n);
	for (i=1;i<=n;i++)
		for (f[i][0]=999,j=1;j<=n;j++)
			scanf("%d",&f[i][j]);
	for (i=1;i<=n;i++)
		scanf("%d",&f[0][i]);
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
		{
			for (f[0][0]=0,k=1;k<=n&&!f[0][0];k++)
				if (f[0][k]!=j&&f[0][k]<=i)
					f[0][0]=k;
			for (k=1;k<=n;k++)
				if (f[j][k]==f[0][f[0][0]])
					break;
			if (f[j][0]>k)f[j][0]=k;
		}
	}
	for (i=1;i<n;i++)
		printf("%d ",f[i][f[i][0]]);
	printf("%d\n",f[n][f[n][0]]);
}
