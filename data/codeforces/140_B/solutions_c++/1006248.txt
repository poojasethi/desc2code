#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <cstdio>
#include <map>
#include <queue>

using namespace std;

int main()  {
	int n;
	cin >> n;
	int pref[n+2][n+1];
	int ord[n];
	int x;
	for(int i=0;i<n;i++) {
		for(int j=0;j<n;j++) {
			cin >> x;
			pref[i][x] = j;
		}
	}
	
	for(int j=0;j<n;j++)
		cin >> pref[n][j];

		
	
	for(int f=0;f<n;f++) {
		if(f>0)
			cout << ' ';
		int mn=1000;
		for(int j=0;j<n;j++) {
			int k=0;
			while(k<n && (pref[n][k] > j+1 || pref[n][k]==f+1))
				k++;
			if(k==n)
				continue;
			if(pref[f][pref[n][k]]<mn) {
				mn=pref[f][pref[n][k]];
				ord[f]=j+1;
			}
		}
		cout << ord[f];
	}
	
	cout << endl;
	
	
	return 0;
}
