#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 310
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m;
int a[N][N],tim[N],mi[N];
int main() {
	scanf("%d",&n);
	For(i,1,n+1)For(j,1,n) scanf("%d",&a[i][j]);
	For(i,1,n) mi[i]=n+1;
	For(i,1,n) {
		For(j,1,n) {
			For(k,1,n) if (a[n+1][k]<=i&&a[n+1][k]!=j) break;
			if (k>n) continue;
			k=a[n+1][k];
			int u;
			For(u,1,n) if (a[j][u]==k) break;
			if (u<mi[j]) mi[j]=u,tim[j]=i;
		}
	}
	For(i,1,n) printf("%d ",tim[i]);
	printf("\n");
	return 0;
}
