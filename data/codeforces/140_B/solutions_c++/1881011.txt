#include<iostream>
#include<cstdio>

using namespace std;

int n , A[301][301] , B[301] , C[301];

int main() {
	scanf( "%d" , &n );
	for ( int i = 1 ; i <= n ; i++ )
		for ( int j = 1 ; j <= n ; j++ )
			scanf( "%d" , &A[i][j] );
	for ( int i = 1 ; i <= n  ;i++ ) {
		scanf( "%d" , &B[i] );
		C[B[i]] = i;
	}
	
	for ( int i = 1 ; i <= n ; i++ )
		for ( int j = 1 ; j <= n ; j++ ) {
			int k = A[i][j];
			if ( k == i ) continue;
			
			bool check = true;
			for ( int l = 1 ; l < k ; l++ )
				if ( l != i && C[l] < C[k] ) {
					check = false;
					break;
				}
			
			if ( check ) {
				printf( "%d\n" , k );
				break;
			}
		}
	
	return 0;
}
