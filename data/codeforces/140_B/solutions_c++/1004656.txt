#include <cstdio>
#include <algorithm>
using namespace std;

#define For(i,n) for (int i = 0; i < n; ++i)

int n, d[301][300], t[300];

int f(int x, int *p1, int *p2) {
        int nn = 0;
        For(i,n) if (p2[i] == x) continue;
        else if (!nn || p2[i] < t[nn - 1]) t[nn] = p2[i], ++nn;
//      printf("%d : ", nn);
//      For(i,nn) printf("%d ", t[i]);
//      puts("---");
        For(i,n) For(j,nn) if (p1[i] == t[j]) return p1[i];
}

int main() {
        scanf("%d", &n);
        For(i,n + 1) For(j,n) scanf("%d", d[i] + j);
        For(i,n) printf("%d%c", f(i + 1, d[i], d[n]), i + 1 == n ? '\n' : ' ');
        return 0;
}
