#include<stdio.h>
int main()
{
	int n,i,j,ii,a[400][400]={0},b[400]={0},c[400]={0},x,p,flag;
	scanf("%d",&n);
	for (i=1;i<=n;i++)
		for (j=1;j<=n;j++)
			scanf("%d",&a[i][j]);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&x);
		b[x]=n+1-i;
	}
	for(i=1;i<=n;i++)
	{
		p=b[i];
		b[i]=0;
		j=1;
		while(j<=n)
			{
				if (a[i][j]==i)
					j++;
				flag=1;
				for(ii=1;ii<a[i][j];ii++)
					if (b[a[i][j]]<b[ii])
						flag=0;
				if (flag>0)
				{
					c[i]=a[i][j];
					break;
				}
				j++;
			}
		b[i]=p;
	}
	for (i=1;i<n;i++)
		printf("%d ",c[i]);
	printf("%d\n",c[n]);
	return 0;
}
