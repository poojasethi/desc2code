#include<stdio.h>
#include<stdlib.h>
int a[300][300];
int a2[300][300];
int b[300];
int c[300];
int main(){

#ifdef _DEBUG
	freopen("in.txt","r",stdin);
	freopen("out.txt","w",stdout);
#endif

	int n,i,j;
scanf("%d",&n);
	for(i=0;i<n;i++)
	    for(j=0;j<n;j++){
		scanf("%d",a[i]+j);
		a[i][j]--;
		a2[i][a[i][j]]=j;
	    }
	for(i=0;i<n;i++){
	    scanf("%d",b+i);
	    b[i]--;
	    c[b[i]]=i;
	}

	for(i=0;i<n;i++){
	    int ri=n*2,res=0;
	    int mi=n*2,m=0;

	    for(j=0;j<n;j++){
		if(c[j]<mi && j!=i)
		{
		  mi=c[j];
		  m=j;
		if(a2[i][m]<ri){
		 ri=a2[i][m];
		 res=j;
		}
		}
	    }
	    printf("%d ",res+1);
	}
	return 0;
}
