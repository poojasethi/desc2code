#include <iostream>
#include <vector>
#include <set>
#include <cstdio>

using namespace std;

int mejor(set<int> &s,vector<int> &v)
{
  for (int i=0;i<int(v.size());i++) {
    int x=v[i];
    if (s.find(x)!=s.end()) return x;
  }
  return -1;
}

int main()
{
  int n;
  cin>>n;
  vector<vector<int> > v(n,vector<int> (n));
  for (int i=0;i<n;i++) {
    for (int j=0;j<n;j++) {
      scanf("%d",&(v[i][j]));
    }
  }
  vector<int> w(n);
  for (int i=0;i<n;i++) cin>>w[i];
  for (int i=0;i<n;i++) {
    set<int> saux,s;
    for (int j=0;j<n;j++) {
      if (j!=i) {
	saux.insert(j+1);
	s.insert(mejor(saux,w));
      }
    }
    if (i>0) cout<<" ";
    cout<<mejor(s,v[i]);
  }
  cout<<endl;
}
