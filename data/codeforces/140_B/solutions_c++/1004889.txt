#include<stdio.h>

int zu[400][400];
int wei[400];
int love[400];
int main()
{
	int n;
	scanf("%d",&n);

	for(int i=0;i<n;i++)
	{
		for(int j=0;j<n;j++)
		{
			int ran;
			scanf("%d",&ran);
			zu[i][ran-1]=j;
		}
	}

	for(int i=0;i<n;i++)
	{
		int ran;
		scanf("%d",&ran);
		wei[ran-1]=i;
	}

	for(int i=0;i<n;i++)
	{
		int best=-1;
		int lovest=-1;
		for(int j=0;j<n;j++)
		{
			if(j!=i && (lovest==-1||wei[j]<wei[lovest]))
			{
				lovest=j;
			}
			if(lovest!=-1 && (best==-1||zu[i][lovest]<zu[i][best]))
			{
				best=lovest;
			}
		}
		printf("%d",best+1);
		if(i==n-1)
		{
			printf("\n");
		}
		else
		{
			printf(" ");
		}
	}
}
