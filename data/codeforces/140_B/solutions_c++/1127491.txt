#include <iostream>
using namespace std;
int N;
int rating[400][400];
int A[400],ans[400];
int main(){
  cin>>N;
  for(int i=0;i<N;++i)
   for(int j=0;j<N;++j){int T; cin>>T; rating[i][T]=j;}
  for(int i=0;i<N;++i){int T; cin>>T; A[T]=i;}
  for(int i=0;i<N;++i){
   ans[i]=-1; int in=-1;
   for(int j=1;j<=N;++j){
    if(i+1==j)continue;
    if(in==-1 || A[j]<A[in])in=j;
    else continue;
    if(ans[i]==-1 || rating[i][ans[i]]>rating[i][in])ans[i]=in;
                        }
                      }
  for(int i=0;i<N;++i)cout<<ans[i]<<(i+1==N?'\n':' ');
  //system("pause");
  return 0;
}
