#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int i,j,k,last,a[400][400],n,id[400],x;
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)
		for(j=1;j<=n;++j)scanf("%d",&a[i][j]);
	for(i=1;i<=n;++i)scanf("%d",&x),id[x]=i;
	for(i=1;i<=n;++i){
		for(j=1;j<=n;++j)if(a[i][j]!=i){
			last=1000000;
			for(k=1;k<=n;++k)if(k!=i && id[k]<last){
				if(k==a[i][j])break;
				last=id[k];
			}
			if(k<=n){
				printf("%d ",a[i][j]);
				break;
			}
		}
	}
}
