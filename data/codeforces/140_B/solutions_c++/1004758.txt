#include<iostream>
#include<cstdio>
using namespace std;
main()
{
	int friends[301][301],me[301],ans[301],pos[301];
	int n,i,j,k;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
			scanf("%d",&friends[i][j]);
	}
	for(i=1;i<=n;i++)
	{
		scanf("%d",&me[i]);
		pos[me[i]] = i;
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=n;j++)
		{
			if(friends[i][j] == i)
				continue; //rule 1
			int flag = 0;
			for(k=1;k<=n;k++)
			{
				if (me[k] == friends[i][j])
					break;
				if(me[k] < friends[i][j] && me[k]!= i)
				{
					flag = 1; // CAn't violate rule 2
					break;
				}
			}
			if (flag ==0)
			{
				printf("%d ",friends[i][j]);
				break;
			}
		}
	}
}
