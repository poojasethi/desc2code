#include <iostream>

using namespace std;

int ans[305];
int data[305][305];
int my[305];
int main(){
	int n,buf;
	cin>>n;
	for(int i=1;i<=n;i++){
		ans[i]=1;
		for(int j=0;j<n;j++){
			cin>>buf;
			data[i][buf]=j;
		}
	}
	ans[1]=2;
	for(int j=0;j<n;j++)
		cin>>my[j];

	int fav;

	for(int i=1;i<=n;i++){
		fav=500;
		for(int j=0;j<n;j++){
			if(my[j]!=i&&fav>my[j]){
				fav=my[j];
				if(data[i][ans[i]]>data[i][fav])
					ans[i]=fav;
			}
		}
	}


	cout<<ans[1];
	for(int i=2;i<=n;i++)
		cout<<" "<<ans[i];
	cout<<endl;
	return 0;
}
