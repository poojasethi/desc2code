#include <iostream>
#include <stdio.h>
using namespace std;
const int maxn=505;
int a[maxn][maxn],b[maxn],n,d[maxn],s[maxn];
void init()
{
     scanf("%d",&n);
     int i,j;
     for(i=1;i<=n;i++)
     for(j=1;j<=n;j++)
     scanf("%d",&a[i][j]);
     for(i=1;i<=n;i++)
     scanf("%d",&b[i]);
     for(i=1;i<=n;i++)
     for(j=1;j<i;j++)
     if(b[j]<b[i]){++s[b[i]];d[b[i]]=b[j];}
} 
void work()
{
     int i,j,k;
     for(i=1;i<=n;i++)
     for(j=1;j<=n;j++)
     {
     if(a[i][j]==i)continue;
     k=a[i][j];
     if(s[k]==0||(d[k]==i&&s[k]==1)){printf("%d ",k);break;}
     }
}
int main(void)
{
    init();
    work();
    return 0;
}
