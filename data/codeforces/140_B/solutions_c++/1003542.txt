#include <stdio.h>
#include <string.h>
#define maxn 310
int a[maxn][maxn];
int b[maxn],p[maxn];
int tag[maxn][maxn];
int ans[maxn];
int n;
int main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++) 
		for (int j=0;j<n;j++) scanf("%d",&a[i][j]);
	for (int i=0;i<n;i++){
		scanf("%d",b+i);
		p[b[i]]=i;
	}
	memset(tag,0,sizeof(tag));
	for (int i=0;i<n;i++){
		for (int j=1;j<=n;j++){
			tag[i][j]=1;
			if ((i+1)==j) tag[i][j]=0;
			for (int k=0;k<p[j];k++)
				if (b[k]!=(i+1) && b[k]<j){
					tag[i][j]=0;
					break;
				}
		}
	}
	for (int i=0;i<n;i++){
		for (int j=0;j<n;j++)
			if (tag[i][a[i][j]]){
				ans[i]=a[i][j];
				break;
			}
	}
	for (int i=0;i<n;i++) printf("%d ",ans[i]);puts("");
	return 0;
}
