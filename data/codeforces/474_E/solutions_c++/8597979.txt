#include <iostream>
#include <vector>
#include <cstdio>
#include <algorithm>
#include <string>
#include <cmath>
const int maxn=100000+100;
using namespace std;
int main()
{
    int n, d;
    scanf("%d%d",&n,&d);
    long long h[maxn];
    int z[maxn],p[maxn];

    for(int i=0; i<n; i++)
        scanf("%I64d",&h[i]);
    for(int i=0; i<n; i++)
    {
        z[i]=1;
    }
    for(int i=1; i<n; i++)
    {
        for(int j=max(i-300,0); j<i; j++)
        {
            if( abs(h[i]-h[j])>=d && z[j]+1>z[i])
            {
                z[i]=z[j]+1;
                p[i]=j;
            }
        }
    }
    int max=-100,index;
    for(int i=0; i<n; i++)
    {
        if(z[i]>max)
        {
            max=z[i];
            index=i;
        }
    }
    printf("%d\n",max);
    vector <int> a;
    a.push_back(index);
    max--;
    while(max>0)
    {
        index=p[index];
        a.push_back(index);
        max--;
    }
    for(int i=a.size()-1; i>=0; i--)
    {
        printf("%d ",a[i]+1);
    }
}
