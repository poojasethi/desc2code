#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <queue>
#include <string>
#include <cmath>

using namespace std;

typedef long long ll;

int main() {
    ll n,d;
    cin >> n >> d;
    vector<ll> v(n),r,p;

    for(ll i=0; i<n; i++)
        cin >> v[i];

    r.resize(n,1);
    p.resize(n,-1);

    for(ll i=n-1; i>=0; i--)
        for(ll j=i+1; j<min((ll)i+700,n); j++){
            if(abs(v[i]-v[j])>=d){
                if(r[j]+1>r[i]){
                    r[i] = r[j]+1;
                    p[i] = j;
                }
            }
        }

    ll mm=-1;
    int indexx = -1;
    for(ll i=0; i<n; i++)
        if(mm<r[i]){
            mm = r[i];
            indexx = i;
        }

    vector<ll> list;
    ll par = p[indexx];
    ll x = indexx;
    while(par != -1){
        list.push_back(x+1);
        x = par;
        par = p[x];
    }
    list.push_back(x+1);
    cout << mm << endl;
    for(int i=0; i<(int)list.size(); i++)
        cout << list.at(i) << " ";
    cout << endl;

    return 0;
}
