#include <iostream>
#include <cstdio>
using namespace std;
typedef long long int ll;
const int N=100100;
int n,K;
ll h[N],sum[N],path[N],ans[N];
int max(int x,int y)
{
    return x>y?x:y;
}
ll myabs(ll x)
{
    return x>0?x:-x;
}
int main()
{
    cin>>n>>K;
    sum[1]=1;
    int maxl=1;
    for(int i=1;i<=n;i++)
    {
        scanf("%lld",&h[i]);
        sum[i]=1;
        for(int j=max(1,i-700);j<i;j++)
        {
            if(myabs(h[j]-h[i])>=K&&sum[j]+1>sum[i])
            {
                sum[i]=sum[j]+1;
                path[i]=j;
            }
            if(sum[i]>sum[maxl])
            {
                maxl=i;
            }
        }
    }
    cout<<sum[maxl]<<endl;
    int T=(int)path[maxl];
    int cnt=0;
    ans[cnt]=maxl;
    while(T)
    {
        ans[++cnt]=T;
        T=(int)path[T];
    }
    for(int i=cnt;i>=0;i--){
        cout<<ans[i]<<" ";
    }
    return 0;
}
