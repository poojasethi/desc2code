#include<iostream>
#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
long long int len[100001],h[100001],pre[100001];
stack<long long int>s;
void func(long long int end)
{
    s.push(end);
    if(pre[end]==-1)
    return;
    func(pre[end]);
}   

int main()
{
    long long int i,j,n,d,end,ans=-1,temp=1;
    memset(pre,-1,sizeof pre);
    cin>>n>>d;
    for(i=1;i<=n;i++)
    cin>>h[i];
    for(i=1;i<=n;i++)
    len[i]=1;
    for(i=1;i<=n;i++)
    {
        for(j=max(temp,i-500);j<i;j++)
        {
            if(abs(h[i]-h[j])>=d && len[i]<len[j]+1)
            {
                len[i]=len[j]+1;
                pre[i]=j;
            }
            if(len[i]>ans)
            {
                end=i;
                ans=len[i];
            }   
        }
    }
    cout<<ans<<endl;
    func(end);
    while(!s.empty())
    {
        cout<<s.top()<<" ";
        s.pop();
    }
}
