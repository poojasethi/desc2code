#include<iostream>
#include<stdio.h>
#include<string.h>
using namespace std;

long long a[100005];
int dp[100005];
int p[100005],ans[100005];

long long abs(long long x){
    if(x>0) return x;
    return -x;
}

int main(){
    long long n,d;
    int i,j;
    scanf("%I64d%I64d",&n,&d);
    for(i=1;i<=n;i++){
        scanf("%I64d",&a[i]);
    }
    dp[1]=1;
    for(i=2;i<=n;i++){
        dp[i]=1;
        for(j=max(1,i-300);j<i;j++){
            if(abs(a[j]-a[i])>=d){
                if(dp[i]<dp[j]+1){
                    dp[i]=dp[j]+1;
                    p[i]=j;
                }
            }
        }
    }
    int t=1,mm=0;
    for(i=1;i<=n;i++){
        if(mm<dp[i]){
            mm=dp[i];
            t=i;
        }
    }
    int cnt=0;
    while(t){
        cnt++;
        ans[cnt]=t;
        t=p[t];
    }
    printf("%d\n",cnt);
    for(i=cnt;i>0;i--){
        printf("%d ",ans[i]);
    }
    return 0;
}
