#include <stdio.h>
#include <iostream>
#include <cmath>
#include <utility>
#include <vector>
using namespace std;
#define ll long long
#define f first
#define s second
ll n,d,a[100005],mx=0,p=0;
pair <ll,ll> f[100005];
vector <ll> v;
int main( ){
	scanf("%I64d %I64d",&n,&d);
	for (int i=1;i<=n;i++)
		scanf("%I64d",&a[i]);
	f[1]=make_pair(1,-1);
	for (int i=2;i<=n;i++){
		f[i]=make_pair(1,-1);
		for (int j=i-1;j>0 && i-j<700;j--)
			if (fabs(a[j]-a[i])>=d && f[j].f+1>f[i].f)
				f[i].f=f[j].f+1,f[i].s=j;
	}
	for (int i=1;i<=n;i++)
		if (f[i].f>mx)
			mx=f[i].f,p=i;
	printf("%d\n",f[p].f);
	v.push_back(p);
	while (f[p].s!=-1){
		v.push_back(f[p].s);
		p=f[p].s;
	}
	for (int i=(int)v.size()-1;i>0;i--)
		printf("%d ",v[i]);
	printf("%d\n",v[0]);
	return 0;
}
