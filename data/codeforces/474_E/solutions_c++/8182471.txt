#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define rep(i,j,k)	for (int i = j; i <= k; i++)
#define dep(i,j,k)	for (int i = j; i >= k; i--)
using namespace std;
typedef long long ll;
const int N = 1e5+10;
inline void get(ll &x)
{
	char ch = getchar();	x = 0;
	while (ch < '0')	ch = getchar();
	while (ch >= '0')	x = x*10+ch-'0', ch = getchar();
}
inline void get(int &x)
{
	char ch = getchar();	x = 0;
	while (ch < '0')	ch = getchar();
	while (ch >= '0')	x = x*10+ch-'0', ch = getchar();
}
ll d, h[N];
int n, st, ans, f[N], nxt[N];
int main()
{
	cin >> n >> d;
	rep(i,1,n)	get(h[i]);
	dep(i,n,1)
	{
		f[i] = 1;
		dep(j,min(i+600,n),i+1)	if (abs(h[i]-h[j]) >= d)
		{
			if (f[j]+1 > f[i])	f[i] = f[j]+1, nxt[i] = j;
			if (f[i] > ans)	ans = f[i], st = i;
		}
	}
	printf("%d\n%d", ans, st);
	for (int i = nxt[st]; i; i = nxt[i])	printf(" %d", i);
}
