#include <iostream>
#include <algorithm>
#include <string>
#include <cmath>
#include <cstdio>
using namespace std;

#define LL long long
#define MAXN 100005

LL h[MAXN];
int pre[MAXN],len[MAXN];

void print(int y){
	if (~pre[y]){
		print(pre[y]);
		printf("%d ",pre[y]);
	}
}

int main(){
	int n,d;
	scanf("%d%d",&n,&d);
	for (int i=1;i<=n;i++){
		scanf("%I64d",&h[i]);
		len[i]=1;pre[i]=-1;
	}
	int ans=0,end;
	for (int i=1;i<=n;i++){
		for (int j=max(1,i-300);j<i;j++)
			if (abs(h[i]-h[j])>=d&&len[i]<len[j]+1){
				len[i]=len[j]+1;
				pre[i]=j;
			}
		if (len[i]>ans){
			ans=len[i];
			end=i;
		}
	}
	printf("%d\n",ans);
	print(end);printf("%d\n",end);
	return 0;
}
