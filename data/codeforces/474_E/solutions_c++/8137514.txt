#include <bits/stdc++.h>
#define file freopen("inp.txt", "r", stdin);freopen("out.txt", "w", stdout);
#define N 100010

using namespace std;

int n, tr[N], sl[N], imax;
long long d, a[N];
vector <int> luu;

int main()
{
//    file;
    cin >> n >> d;
    for (int i = 1; i <= n; i++) scanf("%I64d", &a[i]);
    memset(tr, 0, sizeof(tr));
    memset(sl, 0, sizeof(sl));
    sl[1] = 1;
    for (int i = 2; i <= n; i++)
    {
        sl[i] = 1;
        for (int j = max(1, i - 500); j < i; j++)
            if (abs(a[i] - a[j]) >= d && sl[j] + 1 > sl[i]) sl[i] = sl[j] + 1, tr[i] = j;
        if (sl[i] > sl[imax]) imax = i;
    }
    cout << sl[imax] << endl;
    while (imax)
    {
        luu.push_back(imax);
        imax = tr[imax];
    }
    while (luu.empty() == 0) cout << luu.back() << " ", luu.pop_back();
}
