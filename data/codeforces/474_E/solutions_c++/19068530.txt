#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
#include <vector>
#include <cmath>
#include <cstdlib>
using namespace std;
typedef long long int ll;
const int maxn=100100;
int n,k;
ll h[maxn],s[maxn],last[maxn],ans[maxn];
int main(){
    scanf("%d%d",&n,&k);
    s[1]=1;
    int maxl=1;
    for(int i=1;i<=n;i++){
        scanf("%lld",&h[i]);
        s[i]=1;
        for(int j=max(1,i-600);j<i;j++){
            if(abs(h[j]-h[i])>=k&&s[j]+1>s[i]){
                s[i]=s[j]+1;
                last[i]=j;
            }
            if(s[i]>s[maxl]) maxl=i;
        }
    }
    printf("%I64d\n",s[maxl]);
    int T=last[maxl];
    int cnt=0;
    ans[cnt]=maxl;
    while(T){
        ans[++cnt]=T;
        T=last[T];
    }
    for(int i=cnt;i>=0;i--){
        printf("%I64d%c",ans[i],i==0?'\n':' ');
    }
    return 0;
}
