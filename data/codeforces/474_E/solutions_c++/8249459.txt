#include<iostream>
#include<cstdlib>

using namespace std ;

const int maxn=1e5+5 ;
int n , d, ans=1 ;
long long h[maxn] , dp[maxn], next[maxn] ;



int main()
{
  cin >> n >> d ;
  for(int i=1 ; i<=n ; i++)
    {
      cin >> h[i] ;
      dp[i]=1 ;
      next[i]=-1 ;
    }
  int start=n ;
  for(int i=n ; i>=1 ; i--)
    {
      for(int j=i+1 ; j<=min(i+700,n); j++)
	{
	  if(abs(h[j]-h[i])>=d && dp[j]+1>dp[i])
	    {
	      next[i]=j ;
	      dp[i]=dp[j]+1 ;
	    }
	}
      if(dp[i]>ans)
	{
	  ans=dp[i] ;
	  start=i ;
	}
    }
  cout << ans << endl ;
  int i=start ;
  while(i!=-1)
    {
      cout << i << " " ;
      i=next[i] ;
    }
  cout << endl ;
  return 0 ;
}
