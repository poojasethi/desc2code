#include<bits/stdc++.h>
using namespace std;
int n, d, mx, start, next[100005], ans[100005]; long long h[100005];
int main()
{
    ios_base::sync_with_stdio(NULL), cin.tie(NULL), cout.tie(NULL);
    cin >> n >> d;
    for(int i = 1; i <= n; i++) cin >> h[i], next[i] = -1, ans[i] = 1;
    for(int i = n; i >= 1; i--) for(int j = i + 1; j <= min(i + 250, n); j++){
         if(abs(h[i] - h[j]) >= d && ans[i] < ans[j] + 1) next[i] = j, ans[i] = ans[j] + 1;
         if(ans[i] > mx) mx = ans[i], start = i;
    }
    cout << mx << endl;
    while(start != -1){ cout << start << " "; start = next[start]; }
    #ifndef ONLINE_JUDGE
        cout << endl << (double)clock() / CLOCKS_PER_SEC << endl;
    #endif
}
