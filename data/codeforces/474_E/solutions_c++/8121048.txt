#include <bits/stdc++.h> 
#define MOD 1000000007
using namespace std;

int main(){
    ios_base::sync_with_stdio(0);
    long long n,d;
    cin >> n >> d;
    long long * a = new long long[n];
    for(int i=0;i<n;i++) cin >> a[i];
    vector<pair<long long, long long> > dp(n);
    dp[0] = make_pair(1,-1);
    for(int i=1;i<n;i++){
        dp[i] = make_pair(1,-1);
        for(int j=(i-1);j>=0 && (i-j)<1000;j--){ 
            if(abs(a[j]-a[i])>=d && (dp[j].first+1)>dp[i].first){
                dp[i].first = dp[j].first+1;
                dp[i].second = j;
            }
        }
    }
    long long mm=0,ind = 0;
    for(long long i=0;i<n;i++){
        if(dp[i].first > mm){
            mm=dp[i].first;
            ind=i;
        }
    }
    cout << dp[ind].first << "\n";
    vector<long long> v;
    v.push_back(ind+1);
    while(dp[ind].second!=-1){
        v.push_back(dp[ind].second+1);
        ind = dp[ind].second;
    }
    for(long long i=v.size()-1;i>=0 ; i--){
        cout << v[i] << " ";
    }
    cout << "\n";
    return 0;
}
