#include<cstdio>
double x,y,f[1005][1005],g[1005][1005];
int w,b,i,j;
int main()
{
	scanf("%d%d",&w,&b);
 	for(i=1;i<=w;i++) f[i][0]=1;
 	for(i=1;i<=w;i++)
 		for(j=1;j<=b;j++)
 		{
    		f[i][j]=i*1.0/(i+j)+g[i][j-1]*(1-i*1.0/(i+j));
    		x=(f[i-1][j-1]*i/(i+j-1))+(f[i][j-2]*(j-1)*1.0/(i+j-1));
    		y=(1-(i*1.0/(i+j)));
    		g[i][j]=x*y;
		}
 	printf("%.9lf\n",f[w][b]);
 	return 0;
}
