#include <cstdio>
#define maxn 1001

bool u[maxn][maxn];
double dp[maxn][maxn];

double solve(int w,int b, int ph){
	if(w<0||b<0) return 0;
	if(w+b==0) return 0;
	if(!u[w][b]){
		u[w][b]=1;
		if(ph==2){
			dp[w][b]=double(w)/(w+b)*solve(w-1,b,0)+double(b)/(w+b)*solve(w,b-1,0);
		}
		else{
			dp[w][b]=double(w)/(w+b)*(ph==0)+double(b)/(w+b)*solve(w,b-1,ph+1);
		}
	}
	return dp[w][b];
}
int main(){
	int w,b;
	scanf("%d%d",&w,&b);
	printf("%.9lf\n",solve(w,b,0));
	return 0;
}
