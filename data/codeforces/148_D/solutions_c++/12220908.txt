#include <iostream>
#include <cstdio>
#define maxn  1009
using namespace std;
double dp[maxn][maxn][2];
int main()
{
    int w,h;
    scanf("%d%d",&w,&h);
    for(int i=1;i<=w;i++)dp[i][0][0] = 1.0;
    for(int i=1; i<=w;i++)
    {
        for(int j = 1; j<= h;j++)
        {
            dp[i][j][0] = (i+ j*dp[i][j-1][1])/(i+j);
            dp[i][j][1] = (i*dp[i-1][j-1][0]+(j-1)*dp[i][j-2][0])/(i+j-1)*j/(i+j);
        }
    }
    printf("%.9f\n",dp[w][h][0]);
    return 0;
}
