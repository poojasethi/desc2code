#include <cstdio>
#include <cstring>
using namespace std;

typedef double DB;
DB dp[1010][1010];

DB dfs(int i, int j)
{
	if (i<0 || j<=0) return 0;
	if (i+j <= 2) return (DB)j/(i+j);
	if (~*(long long *)&dp[i][j]) return dp[i][j];
	return dp[i][j] = (j + i*(i-1)*((i-2)*dfs(i-3,j) + j*dfs(i-2,j-1)) /(i+j-2) / (i+j-1)) /(i+j);
}

int main(void)
{
	int W, B;
	scanf("%d%d", &W, &B);
	memset(dp, -1, sizeof dp);
	printf("%.9f\n", dfs(B, W));
}
