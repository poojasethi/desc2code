#include<cstdio>
#include<cstring>
using namespace std;
double dp[1001][1001][2];
int main()
{
	int w,b;
	scanf("%d%d",&w,&b);
	for(int i=1;i<=w;++i)
	{dp[i][0][0]=1;dp[i][0][1]=0;}
	for(int i=0;i<=b;++i)
	{dp[0][i][0]=0;dp[0][i][1]=1;}
	for(int i=1;i<=w;++i)for(int j=1;j<=b;++j)
	{
		dp[i][j][0]=(double)i/(i+j)+(double)j/(i+j)*dp[i][j-1][1];
		dp[i][j][1]=(double)j/(i+j)*((double)i/(i+j-1)*dp[i-1][j-1][0]+(double)(j-1)/(i+j-1)*dp[i][j-2][0]);
	}
	printf("%.9lf\n",dp[w][b][0]);
}
