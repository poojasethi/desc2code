#include <cstdio>

double f[1010][1010], g[1010][1010];
int w, b;


int main()
{
    scanf("%d%d",&w,&b);
    for(int i = 1; i <= w; i++)
        f[i][0] = 1;
    for(int i = 1; i <= w; i++)
    {
        for(int j = 1; j <= b; j++)
        {
            f[i][j] = 1.0*i/(i+j) + g[i][j-1]*(1-1.0*i/(i+j));
            g[i][j] = (1-1.0*i/(i+j))*(1.0*i/(i+j-1)*f[i-1][j-1]+1.0*(j-1)/(i+j-1)*f[i][j-2]);
        }
    }
    printf("%.10lf\n",f[w][b]);
    return 0;
}
