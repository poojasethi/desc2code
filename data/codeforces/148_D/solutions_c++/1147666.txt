#include <iostream>
#include <cstdio>
using namespace std;

double p[1003][1003];

int main( void )
{
  int w,b;
  cin >> w >> b;
  for( int i=1; i<=w; i++ ) for( int j=0; j<=b; j++ ){
    double W=i,B=j;
    p[i][j] = W/(W+B);
    if( j > 1 ) p[i][j] += B/(W+B) * (B-1)/(W+B-1) *     W/(W+B-2) * p[i-1][j-2];
    if( j > 2 ) p[i][j] += B/(W+B) * (B-1)/(W+B-1) * (B-2)/(W+B-2) * p[i  ][j-3];
  }
  printf( "%1.9lf\n", p[w][b] );
  return 0;
}
