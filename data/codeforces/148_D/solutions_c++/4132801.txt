#include <cstdio>
#include <cstring>

int w,b;
double dp[1010][1010];

int main()
{
	int i,j;
	scanf("%d%d",&w,&b);
	memset(dp,0,sizeof(dp));
	for(i=1;i<=w;i++) dp[i][0]=1;
	for(i=1;i<=b;i++) dp[0][i]=0;
	for(i=1;i<=w;i++)
	{
		for(j=1;j<=b;j++)
		{
			dp[i][j]+=(double)i/(i+j);
			if(j>=2&&i>=1) dp[i][j]+=dp[i-1][j-2]*j/(i+j)*(j-1)/(i+j-1)*(i)/(i+j-2);
			if(j>=3) dp[i][j]+=dp[i][j-3]*j/(i+j)*(j-1)/(i+j-1)*(j-2)/(i+j-2);
		}
	}
	printf("%.9f\n",dp[w][b]);
	return 0;
}
