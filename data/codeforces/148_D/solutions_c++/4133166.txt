#include <stdio.h>
	double dp[1005][1005];
int main(){
	int n,m;

	for(int i=0;i<=1000;i++){
		dp[0][i]=0;
	}

	for(int i=1;i<=1000;i++)
	for(int j=0;j<=1000;j++){
		dp[i][j]=i/(0.+i+j);
		if(i+j==2)continue;
		if(j>=2)dp[i][j]+=dp[i-1][j-2]*j/(0.+i+j)*(j-1)/(0.+i+j-1)*(i)/(0.+i+j-2);
		if(j>=3)dp[i][j]+=dp[i][j-3]*j/(0.+i+j)*(j-1)/(0.+i+j-1)*(j-2)/(0.+i+j-2);
	}
	while(~scanf("%d%d",&n,&m)){
		printf("%.9f\n",dp[n][m]);
	}
	return 0;
}
