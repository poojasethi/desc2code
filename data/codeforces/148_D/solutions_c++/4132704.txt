#include<stdio.h>
double f[1005][1005][2];
double ans;
int w,b,i,j,k;
int main(){
	scanf("%d%d",&w,&b);
	f[w][b][0]=1;
	for(i=w;i>=0;i--)
		for(j=b;j>=0&&i+j>0;j--){
				if(i>=0)ans+=i*f[i][j][0]/(i+j);
				if(j>=1)f[i][j-1][1]+=j*f[i][j][0]/(i+j);
				if(j>=2)f[i][j-2][0]+=j*(j-1)*f[i][j][1]/((i+j)*(i+j-1));
				if(i+j>1&&i>0&&j>0)f[i-1][j-1][0]+=j*i*f[i][j][1]/((i+j)*(i+j-1));				
			}
	printf("%0.9lf\n",ans);
	
}
