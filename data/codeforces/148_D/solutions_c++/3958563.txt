#include<cstdio>
#include<iostream>
using namespace std;

int W,B;
double f[1010][1010];

int main()
{
    scanf("%d%d",&W,&B);
    for(int i=1;i<=W;++i)
        for(int j=0;j<=B;++j){
            double w=i,b=j;f[i][j]=w/(w+b);
            if(j>1) f[i][j]+=b/(w+b)*(b-1)/(w+b-1)*w/(w+b-2)*f[i-1][j-2];
            if(j>2) f[i][j]+=b/(w+b)*(b-1)/(w+b-1)*(b-2)/(w+b-2)*f[i][j-3];
        }
    printf("%.9lf\n",f[W][B]);
    return 0;
}
