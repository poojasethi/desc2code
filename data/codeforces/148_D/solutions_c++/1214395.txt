#include<stdio.h>
#include<string.h>
#define N 1005
double  p[N][N],d[N][N];
int w,b;
int main(){
    
    int i,j;
    scanf("%d%d",&w,&b);
    for(i=1;i<=w;i++)  p[i][0]=1.0;
    
    for(i=1;i<=w;i++)
     for(j=1;j<=b;j++){
            p[i][j]=1.0*i/(i+j)+d[i][j-1]*j*1.0/(i+j);
            d[i][j]=1.0*j/(i+j)*(p[i-1][j-1]*(i)*1.0/(i+j-1)+p[i][j-2]*1.0*(j-1)/(i+j-1));
        }
        printf("%.9lf\n",p[w][b]);
    //while(1);
    return 0;
}
