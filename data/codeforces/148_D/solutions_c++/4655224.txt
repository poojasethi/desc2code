#include <iostream>
#include <cstdio>
using namespace std;
double f[1010][1010];
double p(double w,double b)
{
    if(f[(int)w][(int)b]) return f[(int)w][(int)b];
    if(w==0) return 0;
    if(w+b<=3) return w/(w+b);
    if(b<=0) return 1;
    double t=p(w,b-3)*(b-2)/(w+b-2)+p(w-1,b-2)*w/(w+b-2);
    return f[(int)w][(int)b]=w/(w+b)+t*(b-1)/(w+b-1)*b/(w+b);
}
int main()
{
    double w,b;
    cin>>w>>b;
    printf("%.9lf\n",p(w,b));
    return 0;
}
