#include<stdio.h>
double dp[1001][1001]={0};
int n,m;
main(){
scanf("%d%d",&n,&m);
int i,j,a;
for(a=1;a<=n;a++)dp[a][0]=1.000000;
for(i=1;i<=n;i++)
for(j=1;j<=m;j++)
{
dp[i][j]+=((double)i)/((double)(i+j));
if(j>=3)dp[i][j]+=(double)j/(i+j)*((double)(j-1)/(i+j-1))*((double)(j-2)/(i+j-2))*dp[i][j-3];
if(j>=2)dp[i][j]+=((double)j/(i+j))*((double)(j-1)/(i+j-1))*((double)i/(i+j-2))*dp[i-1][j-2];
}
printf("%.9lf\n",dp[n][m]);
}
