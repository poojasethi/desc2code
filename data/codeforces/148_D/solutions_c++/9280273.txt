#include <cstdio>
#include <iostream>
#define p(x,y) (x)*1.0/(y)
using namespace std;
int w,b;
double f[1001][1001];
int main(){
	scanf("%d%d",&w,&b);
    for(int i=0;i<=w;++i)f[i][0]=1;
    for(int i=0;i<=b;++i)f[0][i]=0;
    for(int i=1;i<=w;++i)
        for(int j=1;j<=b;++j){
        	f[i][j]=p(i,i+j);
        	if(j>=2)f[i][j]+=p(j,i+j)*p(j-1,i+j-1)*p(i,i+j-2)*f[i-1][j-2];
        	if(j>=3)f[i][j]+=p(j,i+j)*p(j-1,i+j-1)*p(j-2,i+j-2)*f[i][j-3];
        }
    printf("%.9lf",f[w][b]);
}
