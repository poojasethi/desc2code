#include<iostream>
#include<iomanip>

using namespace std;

double f[1001][1001];

int main()
{
	int i,j,w,b;
	double t;
	for(i=0;i<1001;i++)
	{
		f[i][0]=1;
		f[0][i]=0;
	}
	for(i=1;i<=1000;i++)
	{
		for(j=1;j<=1000;j++)
		{
			f[i][j]=i*1.0/(i+j);
			t=(j-1)*j*1.0/(i+j)/(i+j-1);
			if(j>1) f[i][j]+=i*t*f[i-1][j-2]/(i+j-2);
			if(j>2) f[i][j]+=(j-2)*t*f[i][j-3]/(i+j-2);
		}
	}
	cin>>w>>b;
	cout<<fixed<<setprecision(9)<<f[w][b]<<endl;
	return 0;
}
