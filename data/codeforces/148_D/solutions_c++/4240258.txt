#include<stdio.h>
double f[1010][1010];
int main(){
    int i,j,n,m;
    double w,b;
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)
		for(j=0;j<=m;j++){
            w=i,b=j;
			f[i][j]=w/(w+b);
			if(j>1)
				f[i][j]+=b/(w+b)*(b-1)/(w+b-1)*w/(w+b-2)*f[i-1][j-2];
			if(j>2)
				f[i][j]+=b/(w+b)*(b-1)/(w+b-1)*(b-2)/(w+b-2)*f[i][j-3];
		}
	printf("%.12lf\n",f[n][m]);
	return 0;
}
