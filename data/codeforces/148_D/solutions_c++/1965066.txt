#include <iostream>
#include <stdio.h>
using namespace std;
double p[1005][1005],d[1005][1005];

int main()
{
    int i,j,w,b;
    scanf("%d%d",&w,&b);
    for(i=1;i<=w;i++) p[i][0]=1;
    for(i=1;i<=w;i++)
    for(j=1;j<=b;j++)
    {
        p[i][j]=i*1.0/(i+j)+j*1.0/(i+j)*d[i][j-1];
        d[i][j]=j*1.0/(i+j)*(i*1.0/(i+j-1)*p[i-1][j-1]+(j-1)*1.0/(i+j-1)*p[i][j-2]);//居然数组下标不超界，因为那时（j-1)=0吧？？;
    }
    printf("%.9lf\n",p[w][b]);
    return 0;
}
