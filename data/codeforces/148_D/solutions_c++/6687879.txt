#include <stdio.h>
#include <iostream>
using namespace std;
bool f[1005][1005]={false};
double dp[1005][1005]={0};
int n,m;
inline double back(int w,int b,int s){
	if (w<0 || b<0 || w+b==0) return 0;
	if (!f[w][b]){
		f[w][b]=1;
		if (s==2) dp[w][b]=w*1.0/(w+b)*back(w-1,b,0)+b*1.0/(w+b)*back(w,b-1,0);
		else dp[w][b]=w*1.0/(w+b)*(s==0)+b*1.0/(w+b)*back(w,b-1,s+1);
	}
	return dp[w][b];
}
int main( ){
	cin>>n>>m;
	printf("%.9f\n",back(n,m,0));
	return 0;
}
