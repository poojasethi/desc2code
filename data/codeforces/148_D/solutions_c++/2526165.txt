#include <stdio.h>
double dp[1005][1005];
int w,b;
int main(){
	


	for(int i=1;i<=1000;i++)dp[i][0]=1.0;
	for(int i=0;i<=1000;i++)dp[0][i]=0.0;

	for(int i=1;i<=1000;i++)
	for(int j=1;j<=1000;j++){
		dp[i][j]=1.0*i/(i+j);
		if(j>=2) dp[i][j]+=1.0*(j)*(j-1)*(i)/(i+j)/(i+j-1)/(i+j-2)*dp[i-1][j-2];
		if(j>=3) dp[i][j]+=1.0*(j)*(j-1)*(j-2)/(i+j)/(i+j-1)/(i+j-2)*dp[i][j-3];

	}	
	while(scanf("%d%d",&w,&b)!=EOF)
	printf("%.9f\n",dp[w][b]);
	return 0;
}
