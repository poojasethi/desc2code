#include <iostream>
#include <cstdio>
double f[1010][1010][2],x,y;
int i,j,n,m;
int main()
{
//	freopen("ex.in","r",stdin);
	scanf("%d%d\n",&n,&m);
	for (i=1;i<=n;++i) f[i][0][0]=1;
	for (i=1;i<=n;++i)
		for (j=1;j<=m;++j)
		{
			x=1.0*i/(i+j);y=1.0*i/(i+j-1);
			f[i][j][0]=x+(1-x)*f[i][j-1][1];
			f[i][j][1]=(1-x)*y*f[i-1][j-1][0];
			if (j>1) f[i][j][1]+=(1-x)*(1-y)*f[i][j-2][0];
		}
	printf("%.10f\n",f[n][m][0]);
}
