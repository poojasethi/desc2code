#include<cstdio>
const int N=1001;
double dp[N][N];
int main()
{
	int i,j,a,b;
	scanf("%d%d",&a,&b);
	for(i=1;i<=a;i++)
		dp[i][0]=1.0;
	for(i=1;i<=a;i++)
	{
		for(j=1;j<=b;j++)
		{
			dp[i][j]=(double)i/(i+j);
			if(j>=3)
				dp[i][j]+=dp[i][j-3]*j*(j-1)*(j-2)/(i+j)/(i+j-1)/(i+j-2);
			if(j>=2)
				dp[i][j]+=dp[i-1][j-2]*j*(j-1)*i/(i+j)/(i+j-1)/(i+j-2);
		}
	}
	printf("%.9lf\n",dp[a][b]);
	return 0;
}
