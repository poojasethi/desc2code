#include<stdio.h>
double dp[1020][1020],tw,tb;
int main()
{
     int w,b,i,j;
     scanf("%d%d",&w,&b);
     for(i=1;i<=w;i++)
     	for(j=0;j<=b;j++)
     	{
           tw=i,tb=j;
           dp[i][j]=tw/(tw+tb);
           if(j>1)
           dp[i][j]+=(tb/(tw+tb)*(tb-1)/(tw+tb-1)*tw/(tw+tb-2)*dp[i-1][j-2]);
           if(j>2)
           dp[i][j]+=(tb/(tw+tb)*(tb-1)/(tw+tb-1)*(tb-2)/(tw+tb-2)*dp[i][j-3]);
     	}
     printf("%0.9lf\n",dp[w][b]);
	return 0;
}
