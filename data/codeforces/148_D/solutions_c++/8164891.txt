#include<bits/stdc++.h>
#define maxx 1111
using namespace std;
double dp[maxx][maxx];
int main()
{
	int i,j,w,b;
	cin>>w>>b;
	for(i=1;i<=w;i++) dp[i][0]=1;
	for(i=1;i<=b;i++) dp[0][i]=0;
	for(i=1;i<=w;i++)
		for(j=1;j<=b;j++)
		{
			dp[i][j]=double(i)/(i+j);
			if(j>=3) dp[i][j]+=dp[i][j-3]*j/(i+j)*(j-1)/(i+j-1)*(j-2)/(i+j-2);
			if(j>=2) dp[i][j]+=dp[i-1][j-2]*j/(i+j)*(j-1)/(i+j-1)*i/(i+j-2);
		}
	printf("%.9f",dp[w][b]);
}
