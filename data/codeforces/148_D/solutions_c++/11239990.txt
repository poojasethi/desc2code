#include<cstdio>
#include<cstdlib>
#include<cstring>
double dp[1005][1005];
int main()
{
	int w,b,i,j;
	scanf("%d%d",&w,&b);
	memset(dp,0,sizeof(dp));
	for(i=1;i<=w;i++)
	dp[i][0]=1.0;
	for(i=1;i<=w;i++)
	for(j=1;j<=b;j++)
	{
		dp[i][j]+=(double)i/(i+j);
	    if(j>=3)
        dp[i][j]+=(double)j/(i+j)*(j-1)/(i+j-1)*(j-2)/(i+j-2)*dp[i][j-3];
        if(j>=2)
        dp[i][j]+=(double)j/(i+j)*(j-1)/(i+j-1)*i/(i+j-2)*dp[i-1][j-2];
	}
	printf("%.9lf\n",dp[w][b]);
	return 0;
}
