#include<cstdio>
using namespace std;
double dp[1001][1001][2];
int main()
{
	int w,b;
	scanf("%d%d",&w,&b);
	for(int i=1;i<=w;++i)dp[i][0][0]=1;
	for(int i=1;i<=w;++i)for(int j=1;j<=b;++j)
	{
		dp[i][j][0]=(i+j*dp[i][j-1][1])/(i+j);
		dp[i][j][1]=(i*dp[i-1][j-1][0]+(j-1)*dp[i][j-2][0])/(i+j-1)*j/(i+j);
	}
	printf("%.9lf\n",dp[w][b][0]);
}
