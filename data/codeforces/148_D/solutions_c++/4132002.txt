#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
double dp[1005][1005];
int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;++i)
        for(int j=0;j<=m;++j)
        {
            double w=i,b=j;
            dp[i][j]=w/(w+b);
            if(j>1)dp[i][j]+=b/(w+b)*(b-1)/(w+b-1)*w/(w+b-2)*dp[i-1][j-2];
            if(j>2)dp[i][j]+=b/(w+b)*(b-1)/(w+b-1)*(b-2)/(w+b-2)*dp[i][j-3];
        }
    printf("%.9lf\n",dp[n][m]);
}
