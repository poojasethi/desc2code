#include <bits/stdc++.h>
using namespace std;
const int maxn = 2e3;
double dp[maxn][maxn];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	//memset(dp[0],0,sizeof(dp[0]));
	for(int w = 1; w<= n; w++)
		for(int b = 0; b <= m; b++)
		{
			dp[w][b] = 1.0*w/(w+b);
			if(b > 1)
			{
				double tmp = w*1.0/(w+b-2)*dp[w-1][b-2];
				if(b > 2)
					tmp+= (b-2)*1.0/(w+b-2)*dp[w][b-3];
				dp[w][b] += b*1.0/(w+b)*(b-1)/(w+b-1)*tmp;
			}
		}
	printf("%.10f\n",dp[n][m]);
	return 0;
}
