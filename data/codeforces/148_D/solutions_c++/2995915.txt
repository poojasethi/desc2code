#include<stdio.h>
double dp[1010][1010];
int main()
{
    for(int i=1;i<=1000;i++)
        dp[i][0]=1;
    for(int j=0;j<=1000;j++)
        dp[0][j]=0;
    for(int i=1;i<=1000;i++)
    for(int j=1;j<=1000;j++)
    {
        dp[i][j]=(double)i/(i+j);
        if(j>1)
            dp[i][j]+=((double)j/(i+j)*(double)(j-1)/(i+j-1))*((double)i/(j-2+i)*dp[i-1][j-2]+(double)(j-2)/(j-2+i)*dp[i][j-3]);
    }
    int n,m;
    scanf("%d%d",&n,&m);
    printf("%.9f\n",dp[n][m]);
}
