#include<stdio.h>

double dp[1005][1005];

int main()
{
	int n,m,x,y;
	double i,j;
	for(x=0;x<=1000;x++)
	{
		dp[x][0]=1.0;
		dp[0][x]=0;
	}
	for(x=1;x<=1000;x++)
		for(y=1;y<=1000;y++)
		{
			i=x;j=y;
			dp[x][y]+=1.0*i/(i+j);
			if(j>=3) dp[x][y]+=1.0*j/(i+j)*(j-1)/(i+j-1)*(j-2)/(i+j-2)*dp[x][y-3];
			if(i>=1&&j>=2) dp[x][y]+=1.0*j/(i+j)*(j-1)/(i+j-1)*i/(i+j-2)*dp[x-1][y-2];  
		}
	while(scanf("%d%d",&n,&m)!=EOF)
		printf("%.9f\n",dp[n][m]);
	return 0;
}
