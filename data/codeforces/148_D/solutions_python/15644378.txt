N = 2000
a, b = map(int, raw_input().split())
p = [[0.0] * N for i in range (N)]
p[a][b] = 1.0
ans = 0.0
for i in range (a, -1, -1):
    for j in range (b, -1, -1):
        if(a + b - i - j) % 3 == 0:            
            p[i][j] += ((i+1) * p[i+1][j] + (j+1) * p[i][j+1]) / (i + j + 1)
            if(i + j > 0):
                ans += p[i][j] * i / (i + j)            
        else:
            p[i][j] += p[i][j+1] * (j+1) / (i+j+1)      

print ans
