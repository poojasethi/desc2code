w, b = map(float, raw_input().split())

dp = {}


def f(w, b):
    if w < 0 or b < 0:
        return 0.0

    if w == 0:
        return 0.0
    if b == 0:
        return 1.0


    if (w, b) in dp:
        return dp[(w, b)]

    ans = w/(w+b)

    if b >= 2:
        ans += b/(w+b) * (b-1)/(w+b-1) * ( w/(w+b-2) * f(w-1, b-2) + (b-2)/(w+b-2) * f(w, b-3))

    dp[ (w, b) ] = ans
    return ans


print f( w, b )
