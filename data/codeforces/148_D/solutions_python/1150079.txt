#!/usr/bin/python

probs={(1,1):0.5}

def prob(w, b):
	if w<=0 or b<0:
		return 0
	if b==0: #w>0
		return 1
	if probs.has_key((w, b)):
		return probs[(w, b)]
	ww=w
	bb=b
	w=float(w)
	b=float(b)
	p1 = w/(b+w-2)
	if p1 > 0:
		p1*=prob(ww-1, bb-2)
	p2 = (b-2)/(b+w-2)*prob(ww, bb-3)
	answer = w/(b+w)+(b*(b-1)/((b+w)*(b+w-1)))*(p1+p2)
	probs[(ww, bb)]=answer
	return answer

w, b=tuple(map (int, raw_input().split(" ")))

print "%.9f"%(prob(w, b))
