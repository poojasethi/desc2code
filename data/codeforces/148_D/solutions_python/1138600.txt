#!/usr/bin/env python

memory = {}

def get_res(w, b):
    if (w, b) in memory:
        return memory[(w, b)]
    if w == 0:
        return 0.
    elif b == 0:
        return 1.
    res = float(w) / (w + b)
    if b >= 3:
        b_r = get_res(w, b-3)
        res += float(b) / (w + b) * float(b - 1)/(w + b - 1) * float(b - 2) / (w + b - 2) * b_r
    if b >= 2:
        w_r = get_res(w-1, b-2)
        res += float(b) / (w + b) * float(b - 1)/(w + b - 1) * float(w) / (w + b - 2) * w_r
    memory[(w, b)] = res
    return res

if __name__ == "__main__":
    w, b = map(int, raw_input().strip().split())
    print(get_res(w, b))
