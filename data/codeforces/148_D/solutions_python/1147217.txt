A, B = map(int, raw_input().split())
dp = [(B + 1) * [1e-9] for _ in range(A + 1)]

def cal(w, b):
    if dp[w][b] != 1e-9: return dp[w][b]
    if w <= 0: return 0
    if b <= 0: return 1
    nxt = 0.0
    if b > 2:
        nxt += 1.0 * (b - 2) / (w + b - 2) * cal(w, b - 3)
    if b > 1:
        nxt += 1.0 * w / (w + b - 2) * cal(w - 1, b - 2)
    dp[w][b] = 1.0 * w / (w + b) + 1.0 * b / (w + b) * (b - 1) / (w + b - 1) * nxt
    return dp[w][b]

print "%.9f" % cal(A, B)
