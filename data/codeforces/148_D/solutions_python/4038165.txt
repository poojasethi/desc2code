dp = {}

def dfs(x,y):
    if x <= 0:
        return 0.0
    if y <= 0:
        return 1.0
    if x == 1 and y==1:
        return 0.5
    if (x,y) in dp:
        return dp[(x,y)]
    else:
        tmp = float(y)/float(x+y)*(float(y-1)/float(x+y-1))
        dp[(x,y)] = float(x)/float(x+y) + tmp*(float(x)*dfs(x-1,y-2) + float(y-2)*dfs(x,y-3))/float(x+y-2)
        return dp[(x,y)]
w,b = tuple(map(int, raw_input().split()))

print dfs(w,b)
