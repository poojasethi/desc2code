#!/usr/bin/python

dic = {}

def get_data():
  w, b = raw_input().split(' ')
  return int(w), int(b)


def sum_probs(w, b):
  if (w, b) in dic:
    return dic[(w, b)]
  if not w:
    dic[(w, b)] = 0
    return 0
  if not b:
    dic[(w, b)] = 1
    return 1
  prob = float(w)/(w+b)
  bad = float(b)/(w+b)
  if not b-1:
    dic[(w, b)] = prob
    return prob
  bad *= float(b-1)/(w+b-1)
  if not b-2:
    if w == 1:
      dic[(w, b)] = prob
      return prob
    dic[(w, b)] = prob + bad
    return prob + bad
  case1 = float(w)/(w+b-2)
  case2 = float(b-2)/(w+b-2)
  result = prob+bad*(case1*sum_probs(w-1, b-2) + case2*sum_probs(w, b-3))
  dic[(w, b)] = result
  return result


def main():
  w, b = get_data()
  print sum_probs(w, b)


if __name__ == '__main__':
  main()
