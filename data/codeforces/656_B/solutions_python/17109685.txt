from fractions import gcd
def lcm(a, b):
    return a * b // gcd(a, b)
n = int(input())
m = [int(x) for x in raw_input().split()]
r = [int(x) for x in raw_input().split()]
g = 1
for x in m:
    g = lcm(g, x)
days = [0] * g
for i in range(len(m)):
    for x in range(r[i], g, m[i]):
        days[x] = 1
# print days
print float(days.count(1)) / len(days)
