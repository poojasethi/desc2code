#!/usr/bin/python

import sys
from math import *

def Nil(): return map(int, sys.stdin.readline().split())

n = Nil()[0]
M = Nil()
R = Nil()

period = 16 * 9 * 5 * 7 * 11 * 13
wash = 0
for D in xrange(period):
    found = False
    for i in range(n):
        if D % M[i] == R[i]:
            found = True
            break

    if found:
        wash += 1

print wash/float(period)
