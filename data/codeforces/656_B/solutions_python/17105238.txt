N = int(raw_input())
M = map(int, raw_input().split())
R = map(int, raw_input().split())

def gcd(a, b):
    """Return greatest common divisor using Euclid's Algorithm."""
    while b:      
        a, b = b, a % b
    return a

def lcm(a, b):
    """Return lowest common multiple."""
    return a * b // gcd(a, b)

def lcmm(*args):
    """Return lcm of args."""   
    return reduce(lcm, args)

mylcm=lcmm(*M)
check=[0 for i in xrange(mylcm)]

for D in xrange(mylcm):
    for i in xrange(N):
        if D % M[i]==R[i]:
            check[D]=1
            break

print sum(check)/float(mylcm)
