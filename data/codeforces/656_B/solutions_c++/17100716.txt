#include <bits/stdc++.h>

using namespace std;

int main()
{
int a[100],b[100];
bool c[100100]={0};
int n ;
cin>>n;
int ans=0;

for(int i=0;i<n;i++){
cin>>a[i];
}
for(int i=0;i<n;i++){
cin>>b[i];
}

for(int i=0;i<n;i++){

   for( int x=b[i];x<100100;x+=a[i] )
        c[ x ]=1;
}

for(int i=0;i<100100;i++)
    if( c[i] )
    ans++;

  cout<<fixed<<setprecision(6)<< (1.0*ans)/(1.0*100100) <<endl;

    return 0;
}
