#include<iostream>
#include<iomanip>

using namespace std;

int n;
int m[100],r[100];
int cnt=0;

int main()
{
  
  cin>>n;
  for(int i=0;i<n;i++)
    cin>>m[i];
  for(int i=0;i<n;i++)
    cin>>r[i];

  int k=(int)(2e6)+10000;
  for(int i=0;i<=(int)(2e6)+10000;i++)
    for(int j=0;j<n;j++)
      if(i%m[j]==r[j]){
	cnt++;
	break;}

  double ans=(double)cnt/k;
  cout<<fixed<<setprecision(6);
  cout<<ans<<endl;

  return 0;
}
