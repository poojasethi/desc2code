#include <bits/stdc++.h>

using namespace std;

int	M[20],R[20];

int main()
{
	int	i,j,n;
	cin >> n;
	for(i=1;i<=n;++i) cin >> M[i];
	for(i=1;i<=n;++i) cin >> R[i];

	int	N=10000000,cnt=0;
	for(i=1;i<=N;++i)
	{
		for(j=1;j<=n;++j)
			if(i%M[j]==R[j])
			{cnt++;break;}
	}

	printf("%.8f\n",(double)cnt/N);
	return 0;
}
