#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <vector>
#include <string>


using namespace std;


int main() {
  int i,j,k,n;
  int M[16];
  int R[16];
  cin >> n;

  for(i=0;i<n;i++) {
    cin >> M[i];
  }
  for(i=0;i<n;i++) {
    cin >> R[i];
  }
  double sum=0;
  for(i=0;i< 8*9*5*7*11*13;i++) {
    for(j=0;j<n;j++) {
      if(i%M[j]==R[j]) {
	sum=sum+1;
	break;
      }
    }
  }
  cout << sum/(8*9*5*7*11*13)<<endl;
  return 0;
}
