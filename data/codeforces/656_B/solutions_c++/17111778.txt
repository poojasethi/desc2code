#include <iostream>

using namespace std;

int main()
{
    int i,j,M[1000],R[1000];
    double q=0,n=0;
    cin>>n;
    for (i=1;i<=n;i++) cin>>M[i];
    for (i=1;i<=n;i++) cin>>R[i];
    for (j=1;j<=1000000;j++)
        for (i=1;i<=n;i++)
    {
        if (j%M[i]==R[i]) {q++;break;}
    }
    cout<<q/1000000;
}
