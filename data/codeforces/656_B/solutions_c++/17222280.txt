#include<stdio.h>
int pp[360400]={0},n,m[20]={0},r[20]={0},i,j;
double c1=0,ans=0;
int main()
{
  scanf("%d",&n);
  for(i=1;i<=n;i++) scanf("%d",&m[i]);
  for(i=1;i<=n;i++) scanf("%d",&r[i]);
  for(i=1;i<=360360;i++)
  {
    for(j=1;j<=n;j++)
    {
      if(i%m[j]==r[j])
      {
        pp[i]=1;
        c1+=1;
        break;
      }
    }
  }
  printf("%.8lf",c1/360360);
return 0;
}
