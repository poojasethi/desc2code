#include <iostream>

using namespace std;

int N;
int M[16];
int R[16];

int main()
{
	cin >> N;

	for (int i=0; i<N; i++)
		cin >> M[i];

	for (int i=0; i<N; i++)
		cin >> R[i];

	int days=0;
	for (int i=0; i<720720; i++)
	{
		for (int j=0; j<N; j++)
		{
			if (i%M[j] == R[j])
			{
				days++;
				break;
			}
		}
	}
	
	cout << days/720720.0 << endl;
}
