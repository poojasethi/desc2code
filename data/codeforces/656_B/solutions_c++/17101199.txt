#include <cstdio>
bool vis[720730];
int n,m[20],r[20];
int main()
{
    scanf("%d",&n);
    for(int i=1;i<=n;++i)
        scanf("%d",m+i);
    for(int i=1;i<=n;++i)
        scanf("%d",r+i);
    for(int i=1;i<=n;++i)
    {
        for(int j=r[i];j<720720;j+=m[i])
            vis[j]=1;
    }
    int sum=0;
    for(int i=0;i<720720;++i)
        sum+=vis[i];
    printf("%lf\n",1.0*sum/720720);
    return 0;
}
