#include <iostream>

using namespace std;

bool a[10000000];
int main(){
    int n;
    cin >> n;
    int r[n],m[n];
    for(int i=0;i<n;i++) cin >> r[i];
    for(int i=0;i<n;i++) cin >> m[i];
    for(int i=0;i<n;i++){
        int x = m[i];
        while(x+r[i] < 10000000){
            a[x] = true;
            x += r[i];
        }
    }
    double d = 0;
    for(int i=0;i<10000000;i++){
        if(a[i]) d++;
    }
    cout << d / 10000000 << "\n";
}
