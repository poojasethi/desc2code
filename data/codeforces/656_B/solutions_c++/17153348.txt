#include<bits/stdc++.h>
using namespace std;
int m[20],r[20];
int main()
{
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&m[i]);
    }
    for(int i=0;i<n;i++){
        scanf("%d",&r[i]);
    }
    double ans=0;
    for(int i=0;i<=50000;i++){
        for(int j=0;j<n;j++){
            if(i%m[j]==r[j]){
                ans++;break;
            }
        }
    }
    printf("%lf\n",ans/50001);
}
