#include <iostream>
#include <cstdio>
using namespace std;

const int LIMIT = 720720;

int m[20];
int r[20];

int main() {
  int n;
  cin>>n;
  for(int i =0;i<n;++i) cin>>m[i];
  for(int i =0;i<n;++i) cin>>r[i];
  
  int result = 0;

  for(int i = 1;i<=LIMIT;++i) {
    for(int j=0;j<n;++j)
      if(i % m[j] == r[j]) {
        ++result;
        break;
      }
  }

  printf("%.6lf",((double) result)/LIMIT);
}
