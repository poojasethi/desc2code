#include<bits/stdc++.h>
using namespace std;

const int N = (int)1e5+11;

int a[N],b[N];

int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	for(int i=0;i<n;i++)
		cin>>b[i];
	int ct=0;
	for(int j=0;j<=1e6;j++)
		for(int i=0;i<n;i++){
			if(j%a[i]==b[i]){
				ct++;
				break;
			}
		}
	double ans = (double)ct / (double)(1e6+1);
	printf("%.5lf\n",ans);















	return 0;
}
