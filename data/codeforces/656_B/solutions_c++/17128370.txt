#include <bits/stdc++.h>

using namespace std;
long long M[17], R[17], N, da, cnt;
bool u;
int main(){
	cin >> N;
	for(int i = 1; i <= N; i++){
		cin >> M[i];
		cnt = max(cnt,M[i]);
	}
	for(int i = 1; i <= N; i++) cin >> R[i];
	
	for(int i = 0; i <= 1000000; i++){
			u = 0;
		for(int j = 1; j <= N; j++){
			if(i%M[j] == R[j] && u == 0) da++, u=1;
			}
		}
	cout << setprecision(6) << 1.0*da/(1.0*1000000);
	return 0;
}
