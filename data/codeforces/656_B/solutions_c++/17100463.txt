#include<stdio.h>

int n;
int m[22];
int r[22];
int lim = 16*9*5*7*11*13;

int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&m[i]);
    for(int i=0;i<n;i++) scanf("%d",&r[i]);
    double cnt = 0.0;
    for(int i=0;i<lim;i++)
    {
        for(int j=0;j<n;j++)
        {
            if(i%m[j]==r[j]) {cnt++; break;}
        }
    }
    cnt/=(double)lim;
    printf("%.9f\n",cnt);
    return 0;
}
