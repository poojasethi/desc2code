#include<bits/stdc++.h>
using namespace std;
int arr1[17];
int arr2[17];
int main(){
double T;
cin>>T;
for(int i=0;i<T;i++){
    cin>>arr1[i];
}
for(int i=0;i<T;i++){
    cin>>arr2[i];
}
double counter=0;
for(int i=0;i<10000000;i++){
    for(int j=0;j<T;j++){
        if(i%arr1[j]==arr2[j]){
            counter++;
            break;
        }
    }
}
cout<<fixed<<setprecision(6)<<counter/10000000<<endl;
return 0;}
