#include <bits/stdc++.h>
using namespace std;
int a[100009],b[10009];
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++)cin>>a[i];
    for(int i=0;i<n;i++)cin>>b[i];
    int num = 0;
    for(int i=0;i<1000000;i++){
        for(int j=0;j<n;j++){
            if(i%a[j] == b[j]){
                num++;
                break;
            }
        }
    }

    cout<<(double)num/1000000<<endl;
}
