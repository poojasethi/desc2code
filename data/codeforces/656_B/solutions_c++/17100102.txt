#include <cstdio>
using namespace std;

int n,m[17],r[17];
int main(){
	scanf("%d",&n);
	for (int i=1;i<=n;++i) scanf("%d",&m[i]);
	for (int i=1;i<=n;++i) scanf("%d",&r[i]);
	int cnt=0;
	for (int i=1;i<=5000000;++i)
		for (int j=1;j<=n;++j)
			if (i%m[j]==r[j]){
				++cnt; break;
			}
	printf("%.6f",1.0*cnt/5000000);
	return 0;
}
