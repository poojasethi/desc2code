#include<iostream>
using namespace std;
int main() {
    int num,me=0;
    cin >> num;
    int m[16],r[16];
    for (int i=0;i<num;i++) {
        cin >> m[i];
    }
    for (int i=0;i<num;i++) {
        cin >> r[i];
    }
    for (int i=0;i<720720;i++) {
        for (int j=0;j<num;j++) {
            if (i%m[j]==r[j]) {
                me++;
                break;
            }
        }
    }
    cout << (double)me/720720.0;
}
