#include <bits/stdc++.h>
using namespace std;
int r[20],m[20];
int main() {
     int n,i,c=0,d,j;
     float ans;
	cin>>n;
	for(i=0;i<n;i++) cin>>m[i];
		for(i=0;i<n;i++) cin>>r[i];
		
	for(d=0;d<1000000;d++)
	{
	    for(i=0;i<n;i++)
	    {
	        if(d%m[i]==r[i])
	        {
	            c++;
	            break;
	        }
	    }
	}
	ans=(1.0*c)/1000000.0;
	//ans=(1.0*c)/(16.0);
	printf("%f",ans);
//	cout<<ans;
	return 0;
}
