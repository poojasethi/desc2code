#include <iostream>
#include <math.h>
#include <vector>

using namespace std;

int main(){
	int n; cin >> n;
	
	vector<int> M(n);
	for(int i = 0; i < n; i++) cin >> M[i];
	
	vector<int> R(n);
	for(int i = 0; i < n; i++) cin >> R[i];
	
	int sum = 0;
	for(int i = 0; i < 365000; i++){
		for(int k = 0; k < n; k++){
			if(i % M[k] == R[k]){
				sum++;
				break;
			}
		}
	}
	
	cout << (double)sum / 365000;
}
