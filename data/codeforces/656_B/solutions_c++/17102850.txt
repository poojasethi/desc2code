#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
const int maxn=20,M=720720;
int n,a[maxn],b[maxn];
double ans=0;
int main()
{
	cin>>n;
	for (int i=1;i<=n;i++)
	    cin>>a[i];
	for (int i=1;i<=n;i++)
	    cin>>b[i];
	for (int i=0;i<M;i++)
	    for (int j=1;j<=n;j++)
	        if (i%a[j]==b[j])
	        {
			ans++;
			break;
		    }
	printf("%.6lf",ans/M);
}
