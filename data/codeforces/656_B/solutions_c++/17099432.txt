#include <cstdio>

bool f[720730];
int a[20], b[20];

int main()
{
    int n, i, r=0, j, k;
    scanf("%d", &n);
    for (i=0; i<n; i++) scanf ("%d", &a[i]);
    for (i=0; i<n; i++) scanf ("%d", &b[i]);
    for (i=0; i<n; i++) {
        k=b[i];
        while (k<720720) {
            f[k]=true;
            k+=a[i];
        }
    }
    for (i=0; i<720720; i++) if (f[i]) r++;
    printf("%f", (r+0.0)/720720);
    return 0;
}
