#include <iostream>

using namespace std;

int main()
{
	int n, me = 0;
	cin >> n;
	int m[n], r[n];
	for(int i = 0; i < n; i++) cin >> m[i];
	for(int i = 0; i < n; i++) cin >> r[i];
	for(int j = 0; j < 10000000; j++)
		for(int i = 0; i < n; i++)
			if(j % m[i] == r[i])
			{
				me++;
				break;
			}
	cout << me/10000000.0 << endl;
}
