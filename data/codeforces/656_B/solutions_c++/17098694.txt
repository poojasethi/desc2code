#include <bits/stdc++.h>
using namespace std;
int LIM=720720;
int main()
{
    int n;
    cin >> n;
    int M[n],R[n];
    for(int i=0;i<n;i++) cin >> M[i];
    for(int i=0;i<n;i++) cin >> R[i];
    int awash=0;
    for(int i=0;i<LIM;i++)
    {
        bool washes=false;
        for(int j=0;j<n;j++) {if(i%M[j]==R[j]) washes=true;}
        if(washes) awash++;
    }
    printf("%f\n",1.0*awash/LIM);
    return 0;
}
