#include <bits/stdc++.h>

using namespace std;

  int a[1001],b[1001],n,ans=0,i,j;
  float c;

int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
  for (i=1;i<=n;i++)
    scanf("%d",&b[i]);
  for (i=1;i<=100000*n;i++)
    for (j=1;j<=n;j++)
      if (i%a[j]==b[j])
 {
          ans++;
          break;
}
c=ans;
  printf("%.5lf",(c/(100000*n)));
    return 0;
}
