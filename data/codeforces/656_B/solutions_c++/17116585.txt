#include<stdio.h>
#include<iostream>
using namespace std;

int main()
{
    int m[19],r[19];
    int n;
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d",&m[i]);
    for(int i=0;i<n;i++) scanf("%d",&r[i]);
    double kol=0;
    for(int day=1;day<=720720;day++){
        bool fl=false;
        for(int i=0;i<n && !fl;i++) if(day%m[i]==r[i]) fl=true;
        if(fl) kol++;
    }
    cout<<kol/720720;
}
