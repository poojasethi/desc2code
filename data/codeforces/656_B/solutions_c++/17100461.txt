#include <iostream>
#include <cstdio>
using namespace std;

int main () {
	int n, m[20], r[20];
	cin >> n;
	for (int i = 0; i < n; ++i) cin >> m[i];
	for (int i = 0; i < n; ++i) cin >> r[i];

	double x = 0;
	long long t = 1;

	while (t < 10000000) {
		for (int i = 0; i < n; ++i) {
			if (t % m[i] == r[i]) {
				x += 1;
				break;
			}
		}
		++t;
	}

	printf("%.6f\n", x / 10000000);

	return 0;
}
