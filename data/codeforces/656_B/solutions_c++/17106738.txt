int main(){
  int n;
  cin>>n;
  int m[n],r[n];
  for(int i=0;i<n;i++)
    cin>>m[i];
  for(int i=0;i<n;i++)
    cin>>r[i];
  int k=0;
  for(int i=0;i<99999;i++){
    for(int j=0;j<n;j++){
      if(i%m[j]==r[j]){
        k++;
        break;
      }
    }
  }
  printf("%.6f",k/100000.0);
}
