#include<bits/stdc++.h>
using namespace std;
#define maxn 10000000
int n,M[20],R[20],vis[maxn+5];
int main(){
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		scanf("%d",&M[i]);
	for(int i=1;i<=n;i++)
		scanf("%d",&R[i]);
	for(int i=1;i<=n;i++)
		for(int j=1;j<=maxn;j++)
			if(j%M[i]==R[i])vis[j]=1;
	double ans=0;
	for(int i=1;i<=maxn;i++)
		ans+=vis[i];	
	printf("%.6lf\n",ans/maxn);
	return 0;
}
