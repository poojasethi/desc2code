#include<iostream>

using namespace std;

int main() {
	int n;
	cin >> n;
	int res = 0;
	int m[17];
	int r[17];
	for (int i = 0;i < n;++i) 
		cin >> m[i];

	for (int i = 0;i < n;++i)
		cin >> r[i];
	for (int i = 0;i < 10000000;++i) {

		for (int j = 0;j < n;++j) {
			if (i%m[j] == r[j]) { ++res;break; }
		}
	}
	cout << ((long double)res / 10000000);

}
