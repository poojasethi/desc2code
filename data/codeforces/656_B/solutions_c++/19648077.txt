#include <cstdio>
bool D[720720] = {0};
int N, M[16], R;
int cnt = 0;
int main (){
    scanf("%d", &N);
    for (int i = 0; i < N; i++)
        scanf("%d", &M[i]);
    for (int i = 0; i < N; i++){
        scanf("%d", &R);
        for (int j = R; j < 720720; j += M[i]){
            if (!D[j])
                cnt++;
            D[j] = true;
        }
    }
    printf("%.6lf\n", cnt / 720720.0);
    return 0;
}
