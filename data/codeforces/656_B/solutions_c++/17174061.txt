#include <bits/stdc++.h>
using namespace std;
const int N = 45 ;
int n , b[N] , a[N] ;
double m ;
int main(){ 
	cin >> n ;
	for(int i = 0 ; i < n ; i++) 
		cin >> b[i] ;
	for(int i = 0 ; i < n ; i++) 
		cin >> a[i] ;
	for(int d = 0 ; d < 10000000 ; d++){
		for(int i = 0 ; i < n ; i++)
			if (d % b[i] == a[i]){
				m += 1 ;
				break ;
			}
	}
	cout << m / 10000000 ;
	return 0 ;
}
