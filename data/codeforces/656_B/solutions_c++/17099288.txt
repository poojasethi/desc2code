#include<bits/stdc++.h>
using namespace std;
const double eps(1e-8);
const double pi(3.14159265358979);
const int V=720720;
bool b[V]={};
int main()
{	
	int n,m[20],r[20];
	cin>>n;
	for(int i=1;i<=n;++i)
		cin>>m[i];
	for(int i=1;i<=n;++i)
		cin>>r[i];
	for(int i=1;i<=n;++i)
		for(int v=0;v<V;v+=m[i])
			b[v+r[i]]=true;
	int sum=0;
	for(int i=0;i<V;++i)
		sum+=b[i];
	printf("%.6f\n",sum*1.0/V);
	return 0;
}
