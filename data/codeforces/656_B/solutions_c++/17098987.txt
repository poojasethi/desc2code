#include <iostream>
using namespace std;
int main(){
    int n;
    int M[17],R[17];
    cin>>n;
    for(int i=0;i<n;i++)cin>>M[i];
    for(int i=0;i<n;i++)cin>>R[i];
    double rj=0;
    for(int D=0;D<1e5;D++){
        for(int i=0;i<n;i++){
            if(D%M[i]==R[i]){rj++; break;}
        }
    }
    cout<<rj/1e5;
    return 0;
}
