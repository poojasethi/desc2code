#include <bits/stdc++.h>
using namespace std;

int main() 
{
	int n,i,d,m[20],r[20];
	float cnt=0.0;
	cin>>n;
	for(i=0;i<n;++i)
	cin>>m[i];
	for(i=0;i<n;++i)
	cin>>r[i];
	for(d=0;d<=100000;++d)
	{
		for(i=0;i<n;++i)
		{
			if(d%m[i]==r[i])
			{
				//cout<<d<<" ";
				cnt++;
				break;
			}
		}
	}
	float ans=cnt/100000;
	printf("%0.4f",ans);
	// your code goes here
	return 0;
}
