#include <iostream>
#include <algorithm>

using namespace std;

string s[(int)1e4], k;
bool u[(int)1e4];

int main() {
    int n, len = 0;
    cin >> n;
    for (int i = 0; i < n; i++)
        cin >> s[i], len += s[i].size();
    cin >> k;
    for (int i = 0; i < n; i++)
        s[i] += k;
    len = 2 * len / n + 2;
    sort(s, s + n);
    for (int i = 0; i < n; i++) {
        if (u[i]) continue;
        for (int j = i + 1; j < n; j++) {
            if (s[i].size() + s[j].size() == len && !u[j]) {
                s[j][s[j].size() - 1] = '\n';
                u[i] = u[j] = true;
                cout << s[i] << s[j];
                break;
            }
        }
    }
}
