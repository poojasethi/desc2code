#include <string>
#include <iostream>
#include <set>
#include <algorithm>
#include <cstdio>
using namespace std;

string arr[10001];
char d;
int n;

int charsPerLine;

void input()
{
	cin >> n;
	for(int i = 0; i < n; i++)
		cin >> arr[i];
	cin >> d;
}

void solve()
{
	int totChars = 0;
	for(int i = 0; i < n; i++)
	{	
		totChars += arr[i].length();
		arr[i].push_back(d);
	}
	
	charsPerLine = totChars/(n/2);
	
	sort(arr, arr+n);
	set <string> used;
	
	for(int i = 0; i < n; i++)
	{
		string curr = arr[i];
		if(used.find(arr[i]) != used.end())
			continue;
		
		cout << curr;
		
		int remChars = charsPerLine - (curr.length()-1);
		for(int j = i+1; j < n; j++)
		{
			if(arr[j].length()-1 == remChars && used.find(arr[j]) == used.end())
			{
				for(int k = 0, l = arr[j].length()-1; k < l; k++)
					printf("%c", arr[j][k]);
				printf("\n");
				used.insert(curr);
				used.insert(arr[j]);
				break;
			}
		}
	}
}

int main()
{
	input();
	solve();
}
