#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

vector <vector <string> > s;
vector <string> res;
string str;
int n, k;
string d;
int main()
{
    s.resize(11);
    cin >> n;
    for(int i = 0; i < n; i++)
    {
        cin >> str;
        s[str.length()].push_back(str);
    }
    cin >> d;
    for(int i = 0; i <= 10; i++)
    {
        k += i * s[i].size();
        sort(s[i].begin(), s[i].end());
    }
    k /= n / 2;
    for(int i = 1; i < k / 2 + (k % 2); i++)
        for(int j = 0; j < s[i].size(); j++)
            res.push_back(min(s[i][j] + d + s[k - i][j], s[k - i][j] + d + s[i][j]));
    if(k % 2 == 0)
    {
        for(int i = 0; i < s[k / 2].size(); i+=2)    res.push_back(s[k / 2][i] + d + s[k / 2][i + 1]);
    }
    sort(res.begin(), res.end());
    for(int i = 0; i < res.size(); i++)
        cout << res[i] << endl;
}
