#include<iostream>
#include<string>
#include<algorithm>
#include<cstdio>
#include<cstring>
using namespace std;
string s[10010];
int main()
{
    int n;
    cin >> n;
    int total=0;
    for(int i=0;i<n;i++)
    {
        cin >> s[i];
        total+=s[i].size();
    }
    char c;
    cin >> c;
    for(int i=0;i<n;i++)
        s[i]+=c;
    sort(s,s+n);
    total/=n/2;
    for(int i=0;i<n;i++)
        if(s[i] != " ")
            for(int j=i+1;j<n;j++)
                if(s[j] != " " && s[i].size()+s[j].size() == total+2)
                {
                    cout << s[i];
                    for(int k=0;k<s[j].size()-1;k++)
                        printf("%c",s[j][k]);
                    printf("\n");
                    s[i]=" ";
                    s[j]=" ";
                }
    return 0;
}
