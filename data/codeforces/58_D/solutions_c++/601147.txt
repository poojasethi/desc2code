#include <map>
#include <set>

#include <stack>
#include <queue>
#include <string>
#include <vector>

#include <stdio.h>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
#define li		long long
#define rep(i,to)	for(li i=0;i<((li)to);i++)
#define pb		push_back
#define sz(v)		((li)(v).size())


int main(){
	li n,sum=0;
	string seq;
	cin>>n;
	vector<string> vec[20],con(n);
	rep(i,n) cin>>con[i];
	rep(i,n) vec[sz(con[i])].pb(con[i]);
	rep(i,n) sum+=sz(con[i]);
	rep(i,20) sort(vec[i].begin(),vec[i].end());
	sum=(sum+n/2)/(n/2);
	cin>>seq;
	rep(i,n) con[i]+=seq;
	sort(con.begin(),con.end());
	map<string,int> cnt;
	rep(i,n) cnt[con[i]]++;
	li cnt0=0;
	vector<li> v(20,0);
	rep(i,n/2){
		while(cnt[con[cnt0]]==0) cnt0++;
		cnt[con[cnt0]]--;
		cout<<con[cnt0];
		li tmp=sum-sz(con[cnt0]);
		while(cnt[vec[tmp][v[tmp]]+seq]==0) v[tmp]++;
		cnt[vec[tmp][v[tmp]]+seq]--;
		cout<<vec[tmp][v[tmp]]<<endl;
	}
}
