#include<algorithm>
#include<iostream>
#include<cstdio>
#include<string>
#include<vector>

using namespace std;

const int maxn = 10010;
const int maxm = 15;

int n , tot[maxm] , sum;
string data[maxn] , d;
vector<string> V[maxm];

bool cmp( string A , string B ) {
	return ( A + d ) < ( B + d );
}

int main() {
	scanf( "%d" , &n );
	for ( int i = 0 ; i < n ; i++ ) {
		cin >> data[i];
		sum += data[i].length();
	}
	sum /= ( n / 2 );
	cin >> d;
	sort( data , data + n , cmp );
	
	for ( int i = 0 ; i < n ; i++ )
		V[data[i].length()].push_back( data[i] );
	
	for ( int i = 0 ; i < n ; i++ ) {
		int len = data[i].length();
		if ( tot[len] != ( int ) V[len].size() && V[len][tot[len]] == data[i] ) {
			cout << V[len][tot[len]++] << d;
			cout << V[sum - len][tot[sum - len]++] << endl;
		}
	}
	
	return 0;
}
