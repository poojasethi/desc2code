#include <iostream>
#include <string>
#include <cstring>
#include <algorithm>
#include <queue>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,m,i,j,k,len;
string input[10001];
string sep;
priority_queue<string,vector<string>,greater<string> > vp[12];


int main(){

  cin>>n;rep(i,n)cin>>input[i]; cin>>sep;
  rep(i,n){
    len+=input[i].size();
    string nstr=input[i]+sep;
    vp[nstr.size()].push(nstr);
  }

  m=n/2;len/=m;

  while(m--){
    int maxarg;
    string maxstr="???";

    for(int i=2;i<=11;i++){
      if(!(vp[i].empty())&&(maxstr=="???"||maxstr>vp[i].top())){
    maxstr=vp[i].top();
    maxarg=i;
      }
    }
    
    vp[maxarg].pop();
    string rstr=vp[len+2-maxarg].top();vp[len+2-maxarg].pop();

    cout<<maxstr<<rstr.substr(0,rstr.size()-1)<<endl;

  }
  
}
