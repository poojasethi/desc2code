using namespace std;
string a[10000];
bool f[10000];
char ch;

bool cmp(const string &a, const string &b) {
	return a+ch<b+ch;
}

int main() {
	int n,sum=0;
	scanf("%d",&n);
	for (int i=0; i<n; ++i) {
		cin>>a[i];
		sum+=a[i].size();
	}
	cin>>ch;
	sum/=(n/2);
	sort(a,a+n,cmp);
	memset(f,true,sizeof(f));
	for (int i=0; i<n; ++i) if (f[i]) {
		for (int j=i+1; j<n; ++j) if (f[j]&&a[i].size()+a[j].size()==sum) {
			f[i]=false;
			f[j]=false;
			cout<<a[i]<<ch<<a[j]<<endl;
			break;
		}
	}
	return 0;
}
