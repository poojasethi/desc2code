#include <iostream>
#include <algorithm>
using namespace std;
string s[10000], d;
int n, i, j, k, k0=10, k1=1;
int main() {
	cin>>n;
	for (i=0; i<n; i++) {
		cin>>s[i];
		k=s[i].size();
		k0=min(k0,k);
		k1=max(k1,k);
	}
	cin>>d;
	for(int i = 0; i < n; i++) s[i] += d;
	sort(s, s+n);
	for (i=0; i<n; i+=2) {
		for (j=0; j<n; j++) {
			if (s[j].size()) {
				cout<<s[j];
				k=k0+k1-s[j].size()+2;
				s[j]="";
				break;
			}
		}
		for (j++; j<n; j++) {
			if (s[j].size()==k) {
				s[j][s[j].size()-1]='\n';
				cout<<s[j];
				s[j]="";
				break;
			}
		}
	}
	return 0;
}
