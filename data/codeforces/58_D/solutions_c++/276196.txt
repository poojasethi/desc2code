#include<algorithm>
#include<iostream>
//       Last Change:  2011-02-06 21:18:15
#include<string>
using namespace std;
string s[10001];
bool use[10001];
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;++i)cin>>s[i];
	char ch;cin>>ch;
	int len=0;
	for(int i=0;i<n;++i){s[i]+=ch;len+=s[i].size();}
	sort(s,s+n);
	len=(len<<1)/n;
	for(int i=0;i<n;++i)
		if(!use[i])
		{
			int k=len-s[i].size();
			cout<<s[i];
			for(int j=i+1;j<n;++j)
				if(!use[j]&&s[j].size()==k)
				{
					use[j]=true;
					s[j][s[j].size()-1]='\n';
					cout<<s[j];
					break;
				}
		}
	return 0;
}
