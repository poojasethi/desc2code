#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
#include <vector>
#define MAX 10010
using namespace std;
bool U[MAX];
string d;

bool _cmp(string a,string b){
    return ((a + d + b) < (b + d + a));
}

int main(){
    int n;
    scanf("%d",&n); int l = 0;
    vector < string > V(n);
    for(int i = 0;i < n;i++) cin >> V[i], l += V[i].size();
    cin >> d;
    l /= (n/2);
    sort(V.begin(),V.end(),_cmp);
    memset(U,false,sizeof U);
    for(int i = 0;i < n;i++){
        if(U[i]) continue;
        U[i] = true;
        for(int j = i + 1;j < n;j++){
            if(U[j]) continue;
            if(V[i].size() + V[j].size() != l) continue;
            U[j] = true;
            cout << V[i] + d << V[j] << endl;
            break;
        }
    }
    
    return 0;
}
