#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <cmath>
#include <algorithm>
#include <cassert>
#include <iostream>
#include <string>

using namespace std;

#define MaxN 10000

string s[MaxN];
char d;

bool check(string a, string b)
{
  return (a + d + b) < (b + d + a);
}

int used[MaxN];

int main(void)
{
  int n;
  scanf("%d", &n);
  for (int i = 0; i < n; i++)
    cin >> s[i];
  scanf(" %c", &d);
  int len = 0;
  for (int i = 0; i < n; i++)
    len += s[i].length();
  len /= (n / 2);
  sort(s, s + n, check);
  for (int i = 0; i < n; i++)
    if (!used[i])
    {
      used[i] = 1;
      for (int j = i + 1; j < n; j++)
        if (!used[j] && s[i].length() + s[j].length() == len)
        {
          used[j] = 1;
          cout << s[i] << d << s[j] << endl; 
          break;
        }
    }
  return 0;
}
