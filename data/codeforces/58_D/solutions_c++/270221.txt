#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <string>
#include <vector>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"
#define lim 10001

int n,i,s,x;
char d;
string A[lim];
int S[lim],M[lim];
bool B[lim];
vector <int> V[lim];

int main()
{
  scanf("%d",&n);
  for(i=0;i<n;i++)
    cin>>A[i];
  cin>>d;
  for(i=0;i<n;i++)
	  A[i]+=d;
	sort(A,A+n);
	for(i=0;i<n;i++)
	{
	  S[i]=A[i].size()-1;
	  s+=S[i];
		V[S[i]].push_back(i);
	}
	s/=n/2;
	for(i=0;i<n;i++)
	  if(!B[i])
		{
		  x=s-S[i];
			M[S[i]]++;
		  printf("%s%s\n",A[i].c_str(),A[V[x][M[x]]].substr(0,x).c_str());
		  B[V[x][M[x]]]=1;
			M[x]++;
		}
  return 0;
}
