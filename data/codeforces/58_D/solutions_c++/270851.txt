#include<iostream>
#include<algorithm>
#include<string>
#include<vector>
#include<map>
using namespace std;
int a,len=0,l[10010];
bool is[10010];
string s[10010],p[2],ans[10010];
int main()
{   
	cin>>a;
	for(int i=0;i<=a;i++)
	{
		is[i]=false;
		if(i>0)
		{
			l[i-1]=s[i-1].length();
			len+=l[i-1];
		}
		cin>>s[i];
	}
	sort(s,s+a);
	for(int i=0;i<a;i++)
		l[i]=s[i].length();
	len/=(a/2);
	int na=0;
	for(int i=0;i<a;i++)
		{
			if(!is[i])
			{
				for(int j=i+1;j<a;j++)
				{
					if(!is[j] && (l[i]+l[j]==len))
					{
						p[0]=s[i]+s[a]+s[j];
						p[1]=s[j]+s[a]+s[i];
						if(p[0]<p[1])
						ans[na]=p[0];
						else
						ans[na]=p[1];
						is[i]=is[j]=true;
						na++;
						break;
					}
				}
			}
		}
	sort(ans,ans+na);
	for(int i=0;i<na;i++)
		cout<<ans[i]<<endl;
	return 0;
}
