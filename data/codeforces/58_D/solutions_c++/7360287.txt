#include <cstdio>
#include <map>
#include <cstring>
#include <algorithm>
#include <vector>
#include <set>
#include <iostream>

using namespace std;

vector<string> s[15] ;
int pos[15] ;
int main(){
    int n,tot=0; scanf("%d",&n);
    for (int i=0;i<n;i++){
        string x;
        cin>>x; s[x.length()].push_back(x);
        tot+=x.length();
    }
    string symbol;
    cin>>symbol;
    set<string> myset;
    for (int i=1;i<=10;i++){
        pos[i]=0;
        sort(s[i].begin(),s[i].end());
        for (int j=0;j<s[i].size();j++)
            myset.insert(s[i][j]+symbol);
    }
    int eachlen=(tot+n/2)/(n/2);
    for (int i=1;i<=n/2;i++){
        string x=(*myset.begin()); myset.erase(x);
        int len=x.length()-1;
        pos[len]++;
        int nextlen=eachlen-len-1;
        cout<<x<<s[nextlen][pos[nextlen]]<<endl;
        myset.erase(s[nextlen][pos[nextlen]++]+symbol);
    }
    return 0;
}
