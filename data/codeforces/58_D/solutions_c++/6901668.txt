#include<stdio.h>
#include<set>
#include<string.h>
#include<vector>
#include<string>
#include<iostream>
using namespace std;
#define maxn 111111
set<string> se,see[maxn];
vector<string> ve[maxn];
char str[maxn][12];
int len[maxn];
int main()
{
    int  i, j, n,sum = 0;
    char c;
    scanf("%d",&n);
    for(i=1;i<=n;i++)
    {
       scanf("%s",str[i]);
       len[i] = strlen(str[i]);
       sum+=len[i];
    }
    sum+=n/2;
    sum/=(n/2);
    cin>>c;
    for(i=1;i<=n;i++)
    {
        see[len[i]].insert(str[i]);
        str[i][len[i]] = c,str[i][len[i]+1]  = 0;
        se.insert(str[i]);
    }
    for(i=1;i<=n/2;i++)
    {
        cout<<*se.begin();
        int len = (*se.begin()).length();
        se.erase(*se.begin());
        see[len-1].erase(*see[len-1].begin());
        int temp = sum-len;
        cout<<*see[temp].begin()<<endl;
        se.erase((*see[temp].begin())+c);
        see[temp].erase(*see[temp].begin());
        
    }
    
    return 0;
}
