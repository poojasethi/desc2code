#include <cstdio>
#include <cstring>
#include <algorithm>
#include <string>
#include <vector>
#include <iostream>
using namespace std;

int n;
bool vis[11111];
vector<string> s, ans;
string sy;

int main() {
    scanf("%d", &n);
    s.resize(n);
    for(int i = 0; i < n; i++) cin >> s[i];
    cin >> sy;
    sort(s.begin(), s.end());
    int len = 0;
    for(int i = 0; i < n; i++) len += s[i].size();
    len /= n / 2;
    ans.clear();
    for(int i = 0; i < n; i++) if(!vis[i]) {
        for(int j = i + 1; j < n; j++) if(!vis[j]) {
            if(s[i].size() + s[j].size() == len) {
                vis[i] = vis[j] = 1;
                ans.push_back(min(s[i] + sy + s[j], s[j] + sy + s[i]));
                break;
            }
        }
    }
    sort(ans.begin(), ans.end());
    for(int i = 0; i < ans.size(); i++) cout << ans[i] << endl;
    return 0;
}
