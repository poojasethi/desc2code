#include <iostream>
#include <algorithm>
#include <stdio.h>
#include <string.h>
#include <set>
#include <map>

const int N=100010;

using namespace std;

#define sz(x) ((x).size())

string str[N],s;
bool vis[N];

int main()
{
	int n,sum=0;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>str[i],sum+=sz(str[i]);
	cin>>s;
	for(int i=0;i<n;i++)
		str[i]+=s;
	int l=sum/(n/2);
	sort(str,str+n);
	for(int i=0;i<n;i++)
	{
		if(vis[i]) continue;
		vis[i]=1;
		int k=l-sz(str[i])+1;
		for(int j=i+1;j<n;j++)
		{
			if(vis[j]) continue;
			if(sz(str[j])==k+1)
			{
				cout<<str[i];
				for(int z=0;z<sz(str[j])-1;z++)
					cout<<str[j][z];
				cout<<endl;
				vis[j]=1;
				break;
			}
		}
	}
	return 0;
}
