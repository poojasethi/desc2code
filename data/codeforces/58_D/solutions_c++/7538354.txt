#include<cstdio>
#include<set>
#include<iostream>
#include<string>
using namespace std;

set<string> S;
set<string> Sl[11]; 
set<string> res;
string min(string s1, string s2)
{
    return s1 < s2 ? s1 : s2;
}

int main()
{
    //freopen("data.in", "r", stdin);

    int n;
    scanf("%d", &n);
    int sum = 0;
    for(int i = 0; i < n; i++)
    {
        string s;
        cin >> s;
        int len = s.length();
        S.insert(s);
        Sl[len].insert(s);
        sum += len;
    }
    char c[3];
    scanf("%s", c);

    int len = 2 * sum / n;
    for(int i = 0; i < n / 2; i++)
    {
        string s1 = *(S.begin());
        S.erase(s1);
        Sl[s1.length()].erase(s1);

        string s2 = *(Sl[len - s1.length()].begin());
        res.insert(min(s1 + c[0] + s2, s2 + c[0] + s1));
        S.erase(s2);
        Sl[s2.length()].erase(s2);
    }
    set<string>::iterator it;
    for(it = res.begin(); it != res.end(); it++) 
        cout << *it << endl;
    return 0;
}
