#include <iostream>
#include <set>
using namespace std;
const int MN=100;
set<string> len[MN];
set<string>::iterator it;
set<string> me;
char space;
int n;
int cnt=0;
int main()
{
	cin>>n;
	for (int i=0;i<n;i++)
	{
		string a;
		cin>>a;
		len[a.size()].insert(a);
		cnt+=a.size();
	}
	cnt/=n/2;
	cin>>space;
	for (int i=1;i<=10;i++)
	{
		for (it=len[i].begin();it!=len[i].end();it++)
			me.insert(*it+space);
	}
	for (int i=0;i<n/2;i++)
	{
		cout<<*me.begin();
		string temp=*me.begin();
		me.erase(me.begin());
		len[temp.size()-1].erase(temp.substr(0,temp.size()-1));
		int remain=cnt-temp.size()+1;
		cout<<*len[remain].begin()<<endl;
		me.erase(*len[remain].begin()+space);
		len[remain].erase(len[remain].begin());
	}
	return 0;
}
