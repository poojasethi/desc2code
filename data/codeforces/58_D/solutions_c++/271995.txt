#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

string s [ 10000 ];

int main(){
    string c;
    int n, i, j;
    unsigned t ( 0 );
    cin >> n;
    for ( i = 0; i < n; ++ i ){
        cin >> s [ i ];
        t += s [ i ] . size();
    }
    t /= ( n / 2 );
    cin >> c;
    for ( i = 0; i < n; ++ i )
        s [ i ] += c;
    sort ( s, s + n );
    for ( i = 0; i < n; ++ i )
        if ( s [ i ] != "" )
            for ( j = i + 1; j < n; ++ j )
                if ( s [ j ] != "" && t == s [ i ].size()+s[j].size()-2 ){
                    cout << s [ i ] << s [ j ].substr(0, s[j].size()-1) << '\n';
                    s [ i ] = s [ j ] = "";
                }
    return 0;
}
