#include <iostream>
#include <algorithm>
#include <set>

using namespace std;

int main(int argc, char* argv[])
{
	int n;
	cin >> n;

	set<string> l[10];
	int minlen = 10, maxlen = 1;

	for (int i = 0; i < n; i++)
	{
		string s;
		cin >> s;

		l[s.size() - 1].insert(s);
		minlen = min(minlen, int(s.size()));
		maxlen = max(maxlen, int(s.size()));
	}

	int len = minlen + maxlen + 1;

	char d;
	cin >> d;

	set<string> a;
	for (int i = 0; i < 10; i++)
		for (set<string>::iterator j = l[i].begin(); j != l[i].end(); j++)
			a.insert((*j) + d);

	for (int i = 0; i < n / 2; i++)
	{
		string f = *a.begin(); 
		a.erase(f);
		l[f.size() - 2].erase(f.substr(0, f.size() - 1));

		string s = *l[len - f.size() - 1].begin();
		a.erase(s + d);
		l[s.size() - 1].erase(s);

		cout << f << s << endl;
	}

	return 0;
}
