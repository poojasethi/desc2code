#include <iostream>
#include <cstdio>
#include <string>
#include <algorithm>
#include <vector>
using namespace std;

int N,T,L,c1,c2;
string s,t1,t2;
char d;
vector <string> V[30],A;

int main(){
	scanf("%d",&N);
	for(int i=0;i<N;++i){
		cin>>s;
		T += s.length();
		V[s.length()].push_back(s);
	}
	L = 2*T/N;
	scanf(" %c",&d);
//	printf("T = %d L = %d\n",T,L);
	for(int i=1;i<L;++i) sort(V[i].begin(),V[i].end());
//	printf("done\n");
	for(int i=1;i<=L/2;++i){
	//	printf("V[%d].size() = %d\n",i,V[i].size());
		if(i == L-i)
			for(int j=0;j<V[i].size();j+=2){
				t1 = V[i][j]+d+V[i][j+1];
				A.push_back(t1);
			}
		else
			for(int j=0;j<V[i].size();++j){
				t1 = V[i][j]+d+V[L-i][j];
				t2 = V[L-i][j]+d+V[i][j];
				A.push_back(min(t1,t2));	
			}
	}
	sort(A.begin(),A.end());
//	printf("done\n");
	for(int i=0;i<A.size();++i) cout<<A[i]<<endl;
//	system("pause");
}
