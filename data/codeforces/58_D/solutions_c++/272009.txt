#include <iostream>
#include <algorithm>
#include <string>

using namespace std;

string s [ 10000 ];

int main(){
    string c;
    int n, i, j;
    unsigned t ( 0 ), s1, s2;
    cin >> n;
    for ( i = 0; i < n; ++ i ){
        cin >> s [ i ];
        t += s [ i ] . size();
    }
    t /= ( n / 2 );
    cin >> c;
    for ( i = 0; i < n; ++ i )
        s [ i ] += c;
    sort ( s, s + n );
    for ( i = 0; i < n; ++ i )
        if ( s [ i ] != "" ){
            s1=s[i].size();
            for ( j = i + 1; j < n; ++ j ){
                s2=s[j].size();
                if ( t == s1+s2-2 ){
                    cout << s [ i ] << s [ j ].substr(0, s2-1) << '\n';
                    s [ i ] = s [ j ] = "";
                    break;
                }
            }
        }
    return 0;
}
