#include<bits/stdc++.h>
#define MOD 1000000007
#define MX 100010
#define ll long long
#define sc(n) scanf("%d",&n)
#define pr(m) printf("%d\n",m)
#define pi acos(-1.0)

using namespace std ;

string str[10005];
int visited[10005];
int main()
{
    int n,i,len=0,j;
    sc(n);
    for(i=0;i<n;i++){
        cin>>str[i];

    }
    char d;
    cin>>d;
    for(i=0;i<n;i++){
            str[i]=str[i]+string(1,d);
            len+=str[i].size();
    }
    sort(str,str+n);
    len=(len*2)/n;
    memset(visited,0,sizeof(visited));
    for(i=0;i<n;i++){
        if(visited[i]==0){
            cout<<str[i];
            for(j=i+1;j<n;j++){
                if(str[i].size()+str[j].size()==len && visited[j]==0){
                    str[j][str[j].size()-1]='\n';
                    cout<<str[j];
                    visited[i]=1;
                    visited[j]=1;
                    break;
                }
            }

        }
    }

    return 0;
}
