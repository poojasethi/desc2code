#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <string>
#include <iostream>
using namespace std;

string s[10005];
vector<int> t[35];
int idt[10005];
int u[10005];
int main(void) {
	int n, len=0;
	cin >> n;
	for(int i=0;i<n;i++) {
		cin >> s[i];
		len += s[i].size();
	}
	string sep;
	cin >> sep;
	for(int i=0;i<n;i++)
		s[i] += sep;
	sort(s, s+n);
	for(int i=0;i<n;i++) t[s[i].size()-sep.size()].push_back(i);
	len /= (n/2);
	
	for(int i=0;i<n;i++) {
		if(u[i]) continue;
		u[i]=1;
		int x=len-(int)s[i].size()+(int)sep.size();
		while(u[t[x][idt[x]]]) ++idt[x];
		u[t[x][idt[x]]]=1;
		string &r = s[t[x][idt[x]++]];
		cout << s[i] << r.substr(0,r.size()-sep.size()) << endl;
	}
	return 0;
}
