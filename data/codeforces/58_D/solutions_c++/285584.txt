#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;
int n;
char d;
string ss[1<<16];
int len;

vector<string> vs[16];
size_t cur[16];

int main()
{
	cin>>n;
	for(int i=0; i<n; ++i) cin>>ss[i];
	cin>>d;
	for(int i=0; i<n; ++i) len+=ss[i].size();
	len += n/2;
	len /= n/2;

	for(int i=0; i<n; ++i) ss[i]+=d;
	for(int i=0; i<n; ++i)
		vs[ss[i].size()].push_back(ss[i]);
	for(int i=0; i<16; ++i)
		sort(vs[i].begin(), vs[i].end());

	for(int i=0; i<n/2; ++i) {
		int bj=-1;
		for(int j=0; j<16; ++j) {
			if (cur[j]==vs[j].size()) continue;
			if (bj<0 || vs[j][cur[j]] < vs[bj][cur[bj]])
				bj = j;
		}
		string a = vs[bj][cur[bj]];
		++cur[bj];

		int k = len - (a.size() - 1);
		string b = vs[k][cur[k]];
		++cur[k];
		b.erase(b.end()-1);
		cout<<a<<b<<'\n';
	}
}
