#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
using namespace std;

int main(void)
{
  int n;
  string d;
  cin >> n;
  vector<string> bank(n);
  for (int i = 0; i < n; ++i)
    cin >> bank[i];
  cin >> d;
  int len = 0;
  for (int i = 0; i < n; ++i) {
    len += bank[i].size();
    bank[i] += d;
  }
  sort(bank.begin(), bank.end());
  len = len / (n/2) + 2;

  for (int i = 0; i < n / 2; ++i) {
    vector<string>::iterator p = bank.begin();
    cout << *p;
    int L = len - p->size();
    for(p = bank.erase(p); p != bank.end(); ++p) {
      if (p->size() == L) {
        cout << p->substr(0, L - 1) << "\n";
        bank.erase(p);
        break;
      }
    }
  }
  return 0;
}
