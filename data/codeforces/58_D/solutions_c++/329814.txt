#include <iostream>
#include <algorithm>
#include <vector>
#include <string>
#include <cstring>
using namespace std;

int main()
{
	//freopen("input.in", "r", stdin);
	int n;
	char d;
	cin >> n;
	vector<string> v(n);
	int total = 0;
	for(int i = 0; i < n; i++)
	{
		cin >> v[i];
		total += v[i].size();
	}
	cin >> d;
	vector<bool> used(n, false);
	for(int i = 0; i < v.size(); i++)
		v[i] += d;
	sort(v.begin(), v.end());
	int l = (total * 2) / n;
	for(int i = 0; i < n; i++)
	{
		if(!used[i])
		{
			cout << v[i];
			used[i] = true;
			for(int j = i + 1; j < v.size(); j++)
			{
				if(!used[j] && v[i].size() + v[j].size() - 2 == l)
				{
					v[j][v[j].size() - 1] = '\n';
					cout << v[j];
					used[j] = true;
					break;
				}
			}
		}
	}
	return 0;
}
