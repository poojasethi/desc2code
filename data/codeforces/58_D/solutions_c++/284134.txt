#include<cstdio>
#include<algorithm>
#include<string>
#include<cstring>
#include<iostream>
using namespace std;

string s[11111];
char c;
int n,f[11111];

int main()
{
	scanf("%d",&n);
	int len=0;
	for (int i=0;i<n;i++) cin>>s[i],len+=s[i].size();
	scanf("%s",&c);
	for (int i=0;i<n;i++) s[i]=s[i]+c;
	sort(s,s+n);
	len = (len*2)/n;
	for (int i=0;i<n;i++) if (f[i]==0)
	{
		cout<<s[i]; 
		for (int j=i+1;j<n;j++) 
		if (s[i].size()+s[j].size()-2==len && f[j]==0)
		{
			s[j][s[j].size()-1] = '\n';
			cout<<s[j];
			f[i]=1; f[j]=1;
		 	break;
		}
	}
	return 0;
}
