n = input()
text = sorted([raw_input() for x in xrange(n)], reverse = True)
d = raw_input()
n /= 2
slen = (sum(map(len, text)) + n)/n - 1
res = []
for i in xrange(n):
    a = text.pop()
    for b in text[::-1]:
        if len(a)+ len(b) == slen:
            res.append(min(a+d+b,b+d+a))
            text.remove(b)
            break

print '\n'.join(sorted(res))
