from collections import deque
n = input()
text = deque(sorted([raw_input() for x in xrange(n)]))
d = raw_input()
n /= 2
slen = sum(map(len, text))/n
res = []
for i in xrange(n):
    a = text.popleft()
    for b in text:
        if len(a)+ len(b) == slen:
            res.append(min(a+d+b,b+d+a))
            text.remove(b)
            break

print '\n'.join(sorted(res))
