import sys
rl = sys.stdin.readline
n = input()
a = [rl().strip() for i in xrange(n)]
d = rl().strip()
for i in xrange(n): a[i]=a[i]+d
a.sort()
l = [[] for i in xrange(12)]
lc=[0]*12
q = 2*sum(map(len,a))/n
for s in a: l[len(s)].append(s)    
for s in a:
    ls = len(s)
    if lc[ls]>=len(l[ls]) or l[ls][lc[ls]]!=s: continue
    lc[ls]+=1
    lr = q-ls
    print s+l[lr][lc[lr]][:-1]
    lc[lr]+=1
