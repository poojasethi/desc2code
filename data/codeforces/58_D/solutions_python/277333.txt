n = input()
text = sorted([raw_input() for x in xrange(n)], reverse = True)
lens = dict(zip(text,map(len,text)))
slen = max(lens.values()) + min(lens.values())
d = raw_input()

res = []
for i in xrange(n/2):
    a = text.pop()
    for b in text[::-1]:
        if lens[a] + lens[b] == slen:
            res.append(min(a+d+b,b+d+a))
            text.remove(b)
            break

print '\n'.join(sorted(res))
