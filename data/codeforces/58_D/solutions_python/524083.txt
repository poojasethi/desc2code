#!/usr/bin/python

import sys
import math

if __name__=="__main__":
    data = sys.stdin.readlines()
    n = int(data[0])
    d = data[-1][:-1]
    a = [s[:-1]+d for s in data[1:-1]]
    length = sum([len(s) for s in a])/(n/2)
    a.sort()
    h = [True] * n
    for i, s in enumerate(a):
        if not h[i]:
            continue
        k = 0
        for j in range(i+1, len(a)):
            if not h[j]:
                continue
            if len(a[j]) + len(s) == length:
                k = j
                break
        h[k] = False
        print s+a[k][:-1]
