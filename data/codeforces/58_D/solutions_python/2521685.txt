r = lambda: raw_input()
s = sorted([r() for i in xrange(input())])
d = r()
n = len(s) / 2;
k = sum(map(len, s)) / n
ans = []
for i in xrange(n):
    now = s.pop(0)
    for item in s:
        if len(now) + len(item) == k:
            ans.append(min(item + d + now, now + d + item))
            s.remove(item)
            break
print '\n'.join(sorted(ans))
