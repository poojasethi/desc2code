n = int(raw_input())
cities = [raw_input() for i in xrange(0, n)]
d = raw_input()

width = (sum(map(len, cities)) + n / 2) / (n / 2)
cities.sort()

calendar = []
for i in xrange(n / 2):
	a = cities[0]
	cities = cities[1:]
	for b in cities:
		if len(b) + len(a) + 1 == width:
			if a + d + b < b + d + a:
				line = a + d + b
			else:
				line = b + d + a
			break
	cities.remove(b)
	calendar.append(line)

calendar.sort()
for i in calendar:
	print i
