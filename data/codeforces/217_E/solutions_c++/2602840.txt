#include<cstdio>
#define fo(i,a,b) for(i=a;i<=b;++i)
#define fd(i,a,b) for(i=a;i>=b;--i)
const int mn=10100,mm=3010000;
char st[mm],ans[mm];
int f[mm],s[mm],L[mn],R[mn];
int n,m,K,i,j,d,len,now,sum;
void del(int p)
{
    for(;p<=K;p+=p&(-p))--s[p];
}
int query(int v)
{
    int i=0,l=m>>1;
    for(;l;l>>=1)
        if(i+l<=K&&s[i^l]<v)v-=s[i^=l];
    return i+1;
}
int main()
{
    scanf("%s",st+1);
    scanf("%d",&K);
    for(m=1;m<K;m<<=1);
    sum=K;
    scanf("%d",&n);
    fo(i,1,n)scanf("%d%d",L+i,R+i);
    fo(i,1,K)s[i]=i&(-i);
    fd(i,n,1)
    {
        if(R[i]>=sum)continue;
        len=R[i]-L[i]+1,now=L[i]+1;
        fo(j,1,len)
        {
            if(R[i]>=sum)break;
            if(now>R[i])now=L[i];
            d=query(R[i]+1),del(d);
            --sum;
            f[d]=query(now);
            now+=2;
        }
    }
    j=1;
    fo(i,1,K)
        ans[i]=f[i]?ans[f[i]]:st[j++];
    puts(ans+1);
    return 0;
}
