#include <cstdio>
#include <iostream>
#include <algorithm>

using namespace std;


const int maxn = 100100;
const int maxs = 10001000;
int f[maxs], a[maxn], b[maxn], fa[maxs], fb[maxs];

int main() {
	for (int i = 2; i * i < maxs; ++i) {
		if (f[i] == 0) {
			for (int j = i * i; j < maxs; j += i) {
				f[j] = i;
			}
		}
	}
	for (int i = 2; i < maxs; ++i) {
		if (f[i] == 0) {
			f[i] = i;
		}
	}


	int n, m;
	scanf("%d%d", &n, &m);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
		for (int j = a[i]; j > 1; j /= f[j]) {
			fa[f[j]] += 1;
		}
	}
	for (int i = 0; i < m; ++i) {
		scanf("%d", &b[i]);
		for (int j = b[i]; j > 1; j /= f[j]) {
			fb[f[j]] += 1;
		}
	}

	printf("%d %d\n", n, m);
	int i, j, x;
	for (i = 0; i < n; ++i) {
		x = 1;
		for (j = a[i]; j > 1; j /= f[j]) {
			if (fb[f[j]] > 0) {
				--fb[f[j]];
			} else {
				x *= f[j];
			}
		}
		printf("%d%s", x, i == n - 1 ? "\n" : " ");
	}
	for (i = 0; i < m; ++i) {
		x = 1;
		for (j = b[i]; j > 1; j /= f[j]) {
			if (fa[f[j]] > 0) {
				--fa[f[j]];
			} else {
				x *= f[j];
			}
		}
		printf("%d%s", x, i == m - 1 ? "\n" : " ");
	}

	return 0;
}
