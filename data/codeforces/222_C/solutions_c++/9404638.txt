#include<bits/stdc++.h>
using namespace std;
#define rep(i,a,b) for(i=a;i<=b;i++)
#define mp make_pair
#define pb push_back
#define maxn 10000007

int n,m;
int a[10000005],b[10000005];
int x[10000005],y[10000005];
int p[10000007];
void factorization()
{
  int i ,j;
  for(i=2;i*i<maxn;i++)if(!p[i])for(j=i*i;j<maxn;j+=i)p[j]=i;
  for(i=2;i<=maxn;i++)if(!p[i])p[i]=i;
}
void seive(int n,int s[],int t[])
{
  int i,j;
  rep(i,1,n)
  {
     for(j=s[i];j>1;j/=p[j])t[p[j]]++;
  }
}
void nout(int n,int s[],int t[])
{
  int i,j;
  rep(i,1,n)
  {
    int x=1;
    for(int j=s[i];j>1;j/=p[j])
    {
        if(t[p[j]]>0)t[p[j]]--;
        else x*=p[j];
    }
    printf("%d%c",x,(i==n)?'\n':' ');
  }
}
int main()
{
  scanf("%d %d",&n,&m);
  int i,j;
  rep(i,1,n)scanf("%d",&a[i]);
  rep(i,1,m)scanf("%d",&b[i]);
  factorization();
  printf("%d %d\n",n,m);
  seive(n,a,x);seive(m,b,y);
  nout(n,a,y);
  nout(m,b,x);
  return 0;
}
