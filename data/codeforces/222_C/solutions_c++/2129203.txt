#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

#define N 10000001
#define M 100005
#define LL long long

int p[N];
int fa[N], fb[N];
int a[M], b[M];

void gao(int n, int x[], int g[]) {
    for (int i = 0; i < n; ++i) {
        scanf("%d", &x[i]);
        for (int y = x[i]; y > 1; y /= p[y])
            ++g[ p[y] ];
    }
}

void output(int x[], int n, int f[]) {
    for (int i = 0; i < n; ++i) {
        for (int y = x[i]; y > 1; y /= p[y])
            while (f[ p[y] ] > 0 && x[i] % p[y] == 0) {
                --f[ p[y] ];
                x[i] /= p[y];
            }
        printf("%d", x[i]);
        putchar( i == n - 1 ? '\n' : ' ' );
    }
}

int main() {
    for (int i = 2; i < N; ++i)
        if (!p[i]) {
            for (int j = i; j < N; j += i)
                p[j] = i;
        }
    int an, bn;
    scanf("%d %d", &an, &bn);
    gao(an, a, fa);
    gao(bn, b, fb);
    for (int i = 0; i < N; ++i)
        fa[i] = fb[i]= min(fa[i], fb[i]);
    printf("%d %d\n", an, bn);
    output(a, an, fa);
    output(b, bn, fb);
    return 0;
}
