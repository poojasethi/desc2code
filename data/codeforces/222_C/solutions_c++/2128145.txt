#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn = 100010;
const int N = 10000010;
int a[maxn],b[maxn];
int p[N],x[N],y[N];
void init()
{
    for(int i=2;i*i<N;i++) if(!p[i]) for(int j=i*i;j<N;j+=i) p[j]=i;
    for(int i=2;i<N;i++) if(!p[i]) p[i]=i;
}
void go(int n,int *num,int *cnt)
{
    for(int i=0;i<n;i++) for(int j=num[i];j>1;j/=p[j]) cnt[p[j]]++;
}
void solve(int n,int *num,int *cnt)
{
    for(int i=0;i<n;i++)
    {
        int x=1;
        for(int j=num[i];j>1;j/=p[j])
        {
            if(cnt[p[j]]) cnt[p[j]]--;
            else x*=p[j];
        }
        printf("%d%c",x,i==n-1 ? '\n' :' ');
    }
}
int main()
{
    int n,m;
    init();
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++) scanf("%d",&a[i]);
    for(int j=0;j<m;j++) scanf("%d",&b[j]);
    printf("%d %d\n",n,m);
    go(n,a,x);go(m,b,y);
    solve(n,a,y); solve(m,b,x);
    return 0;
}
