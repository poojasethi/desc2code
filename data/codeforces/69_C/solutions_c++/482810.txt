#include<cstdio>
#include<string>
#include<map>
#include<algorithm>
#include<vector>
#include<iostream>
using namespace std;

vector< pair<string,int> > p[1000];
string A[1000],B[1000];
map<string,int> f[1000];
int n,m,num,q;
char buf[1000000];

int main()
{
    //freopen("input.txt","r",stdin);

    cin >> num >> n >> m >> q;
    for (int i=0;i<n;i++) cin >> A[i];
    for (int i=0;i<m;i++)
    {
        do gets(buf); while (!buf[0]);
        int j=0;
        for (;buf[j]!=':';j++) B[i]+=buf[j];
        for (++j;buf[j];j++)
        {
            while (buf[j]==' ') j++;
            if (!buf[j]) break;
            string C="";int D=0;
            while (buf[j]!=' ') C+=buf[j++];j++;
            while (buf[j]>='0' && buf[j]<='9') D=D*10+buf[j++]-'0';
            p[i].push_back(make_pair(C,D));
            if (!buf[j]) break;
        }
    }
    for (int i=0;i<q;i++)
    for (int j=0;j<n;j++)
        f[i][A[j]]=0;
    for (int i=0;i<q;i++)
    {
        int C;string D;
        cin >> C >> D;
        f[C][D]++;

        for (int j=0;j<m;j++)
        {
            for (;;)
            {
                bool ok=1;
                for (int k=0;k<p[j].size();k++)
                    ok&=(f[C][p[j][k].first]>=p[j][k].second);
                if (ok)
                {
                    for (int k=0;k<p[j].size();k++)
                        f[C][p[j][k].first]-=p[j][k].second;
                    f[C][B[j]]++;
                } else break;
            }
        }

    }
    for (int i=1;i<=num;i++)
    {
        
        int cnt=0;
        for (map<string,int> :: iterator it=f[i].begin();it!=f[i].end();it++)
        if (it->second) cnt++;
        cout << cnt << endl;
        for (map<string,int> :: iterator it=f[i].begin();it!=f[i].end();it++)
        if (it->second)
            cout << it->first << " " << it->second << endl;
    }
}
