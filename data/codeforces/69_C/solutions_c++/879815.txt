int n,k,m,q,d[110][110],f[110][110],F[110][110],x;
string b[110],c[110],s;
map<string,int> M;

int main(){
	cin>>k>>n>>m>>q;
	rep(i,n)cin>>b[i],M[b[i]]=i;
	getline(cin,s);
	rep(i,m){
		getline(cin,s);
		rep(j,s.sz)if(s[j]==':'||s[j]==',')s[j]=' ';
		iss ss(s);
		ss>>c[i];
		while(ss>>s>>x)d[i][M[s]]=x;
	}
	rep(i,q){
		cin>>x>>s;
		x--;
		f[x][M[s]]++;
		rep(j,m){
			int ok=1;
			rep(l,n)if(f[x][l]<d[j][l])ok=0;
			if(ok){
				rep(l,n)f[x][l]-=d[j][l];
				F[x][j]++;
			}
		}
	}
	rep(i,k){
		int N=0;
		vc<pair<string,int> > v;
		rep(j,n)if(f[i][j])v.pb(mp(b[j],f[i][j]));
		rep(j,m)if(F[i][j])v.pb(mp(c[j],F[i][j]));
		cout<<v.sz<<endl;
		sort(v.begin(),v.end());
		rep(i,v.sz)cout<<v[i].fir<<" "<<v[i].sec<<endl;
	}
}
