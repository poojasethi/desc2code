#include <iostream>

using namespace std;

int main()
{
	string s,k;
	int a,b,c,d,e,f,g,sum,n,maxx=-100000;
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		sum=0;
		cin>>s>>a>>b>>c>>d>>e>>f>>g;
		sum+=c+d+e+f+g;
		sum+=a*100;
		sum-=b*50;
		if(sum>maxx)
		{
			maxx=sum;
			k=s;
		}
	}
	cout<<k;

	return 0;
}
