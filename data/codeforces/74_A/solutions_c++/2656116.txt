#include<iostream>
#include<cstdlib>
#include<cstdio>
using namespace std;
int main ()
{
    int n,m,p,i,a,b,c,d,e,z,max=-999999;
    string s,q;
    cin>>n;
    for(i=1;i<=n;i++)
    {
        z=0;
        cin>>s>>p>>m>>a>>b>>c>>d>>e;
        z+=100*p-50*m+a+b+c+d+e;
        if(z>max){max=z;q=s;}
    }
    cout<<q;
    return 0;
}
