#include <iostream>
using namespace std;

int main(){
    string s,luu;
    int n,p,m,a,b,c,d,e,Ma=-1000000;
    cin>>n;
    while (n--){
        cin>>s>>p>>m>>a>>b>>c>>d>>e;
        if ((p*100-m*50+a+b+c+d+e)>Ma){
            Ma=p*100-m*50+a+b+c+d+e;
            luu=s;
        }
    }
    cout<<luu;
}
