#include <iostream>
#include <string>
using namespace std;
int n,i,m=-2000000000,x,s0;
string s,ms;
int main()
{
	cin>>n;
	for(i=0;i<n;i++)
	{
		s0=0;
		cin>>s;
		cin>>x; s0+=100*x;
		cin>>x; s0-=50*x;
		for(int u=0;u<5;u++)
		{
			cin>>x;
			s0+=x;
		}
		if (s0>m)
		{
			ms=s;
			m=s0;
		}
	}
	cout<<ms<<endl;
	return 0;
}
