#include <iostream>
#include <string>
using namespace std;

int main()
{
  string ans, cur;
  int p,m,a,b,c,d,e,n,ls=-2147483640,cs;
  
  cin>>n;
  
  while(n--)
  {
    cs=0;
    
    cin>>cur>>p>>m>>a>>b>>c>>d>>e;
    
    cs = (100*p)-(50*m)+a+b+c+d+e;
    
    if(cs>ls)
    {
      ls = cs;
      ans = cur;
    }
  }
  
  cout<<ans<<endl;
}
