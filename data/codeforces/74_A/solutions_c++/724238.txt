#include <cstdio>
#include <cstring>

char rname[100][500],ch;
int gao,ans,tot,i,a,b,c,d,e,f,g,n;

int main(){
	scanf("%d",&n);
	ans=-999999999;
	for(i=1;i<=n;i++){
		scanf("%c",&ch);
		scanf("%s",rname[i]);
		scanf("%d%d%d%d%d%d%d",&a,&b,&c,&d,&e,&f,&g);
		int tot=a*100-b*50+c+d+e+f+g;
		if(tot>ans){
			ans=tot;
			gao=i;
		}
	}
	printf("%s\n",rname[gao]);
}
