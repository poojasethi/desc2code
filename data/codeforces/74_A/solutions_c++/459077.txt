#include <cstdio>

char name[50][32];

int main(){
  int N;
  scanf("%d",&N);

  int max=-1;
  int maxi=-1;
  for(int i=0;i<N;++i){
    int p,m,a,b,c,d,e;
    scanf("%s%d%d%d%d%d%d%d",name[i],&p,&m,&a,&b,&c,&d,&e);
    int s=a+b+c+d+e +100*p -50*m;
    if(s>max || maxi==-1){
      max=s;
      maxi=i;
    }
  }
  puts(name[maxi]);
  return 0;
}
