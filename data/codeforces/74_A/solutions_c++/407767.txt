#include<iostream>
#include<string>

using namespace std;
int main()
{
	string x="";
	int y=-200000000,n;
	for(cin>>n;n--;)
	{
		string t;
		int p,m,a,b,c,d,e;
		cin>>t>>p>>m>>a>>b>>c>>d>>e;
		int tmp=a+b+c+d+e+p*100-50*m;
		if(tmp>y)y=tmp,x=t;
	}
	cout<<x<<endl;
	return 0;
}
