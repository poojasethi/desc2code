#include<iostream>
#include<string>
using namespace std;
int n,p,m,a,b,c,d,e,ans;
string s,t;
int main(){
	cin>>n;
	cin>>s>>p>>m>>a>>b>>c>>d>>e;
	n-=1;
	ans=100*p-50*m+a+b+c+d+e;
	t=s;
	while(n--){
		cin>>s>>p>>m>>a>>b>>c>>d>>e;
		if(ans<100*p-50*m+a+b+c+d+e){
			ans=100*p-50*m+a+b+c+d+e;
			t=s;
		}
	}
	cout<<t<<endl;
	return 0;
}
