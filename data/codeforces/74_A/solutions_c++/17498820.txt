#include <iostream>
#include <bits/stdc++.h>
using namespace std;

int main()
{


string s,q;
int n,m=-9999999999,a,b,c,d,e,hs,hf,h=0;
cin>>n;
for(int i=0 ; i<n ; i++)
{
    cin>>s>>hs>>hf>>a>>b>>c>>d>>e;
    h=h+a+b+c+d+e+(hs*100)-(hf*50);
    if(h>m)
    {
        m=h;
        q=s;
    }
    h=0;
}
cout<<q;

    return 0;
}
