#include <iostream>
#include <string>
using namespace std;

int main()
{
	int n;
	cin>>n;
	string m;
	int    ms=-3000;

	for(int i = 0; i < n; i++){
		string ns;
		int sh, fh, a, b, c, d, e;
		cin>>ns>>sh>>fh>>a>>b>>c>>d>>e;
		int cs = sh*100-50*fh+a+b+c+d+e;
			if( cs > ms ){
				ms = cs;
				m = ns;
			}
	}
	cout<<m;
	return 0;
}
