#include<iostream>
#include<algorithm>
#include<string>
#include<vector>
using namespace std;
string s,name;
int p,m,n,a,b,c,d,e,res,ans=-10000;
int main()
{
    cin>>n;
    while(n--){
        cin>>name>>p>>m>>a>>b>>c>>d>>e;
        res=(a+b+c+d+e+100*p)-50*m;
        if(ans<res){ans=res; s=name;}
    }
    cout<<s<<endl;
}
