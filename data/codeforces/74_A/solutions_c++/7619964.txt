#include<iostream>
using namespace std;
int main()
{
    int N,a,b,c,d,e,H,U,A,B,C,D,E;
    a=-1000000000;
    string s1,s2;
    cin>>N;
    while(N--)
    {
        cin>>s1>>H>>U>>A>>B>>C>>D>>E;
        b=100*H-50*U+A+B+C+D+E;
        if(b>a)
        {
            a=b;
            s2=s1;
        }
    }
    cout<<s2;
    
    return 0;
}
