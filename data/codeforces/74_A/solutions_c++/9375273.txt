#include<iostream>
#include<string>
using namespace std;

int main(){
	int n,suc,unsuc,a,b,c,d,e,x;
	string s,leader;
	cin>>n;
	int max = -10000;
	for(int i=0 ; i<n ; i++){
		cin>>s>>suc>>unsuc>>a>>b>>c>>d>>e;
		x=100*suc-50*unsuc+a+b+c+d+e;
		if(x>max){
			max=x;
			leader=s;
		}
	}
	cout<<leader<<endl;
    return 0;
}
