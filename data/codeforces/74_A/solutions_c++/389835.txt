#include<iostream>
#include<string>
using namespace std;
main()
{
    string s[100];
    int a,b,c,d,e,h,f,max,n,i,sum,maxn;
    cin>>n;
    max=-1000000;
    for (i=1;i<=n;i++)
    {
        cin>>s[i]>>h>>f>>a>>b>>c>>d>>e;
        sum=100*h-50*f+a+b+c+d+e;
        if (sum>max) {max=sum; maxn=i;}
    }
    cout<<s[maxn];
}
