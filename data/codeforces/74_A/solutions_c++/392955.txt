#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int main()
{
 int t,i;
 char a[50],b[50];
 long long int u,v,w,x,y,z,s,j,maxi=-1000000000;
 cin>>t;
 for(i=1;i<=t;i++)
 {
  cin>>a;
  cin>>u>>v>>w>>x>>y>>z>>s;
  j=(u*100)-(v*50);
  j=j+w+x+y+z+s;
  if(maxi<j)
  {
   maxi=j;
   strcpy(b,a);
  }
 }
 cout<<b<<"\n";
 return(0);
}
