#include <iostream>
#include <vector>

using namespace std;
int main() {
int n;
cin >> n;
int ind = 0;
int ma = -100000000;
string s;
string name;
int p,m,a,b,c,d,e;
for(int i=0; i<n; ++i) {
cin >> s >> p >> m >> a >> b >> c >> d >> e;
p = 100*p - 50*m + a + b + c + d + e;
if(p > ma) {
ma = p;
name = s;
}
}

cout << name;
return 0;
}
