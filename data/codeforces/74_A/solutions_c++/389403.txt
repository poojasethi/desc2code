#include <iostream>
#include <stdio.h>
#include <stdlib.h>
char s[50][100];
int sc[50]={};
int main()
{
	int n,i,p,m,a,b,c,d,e,maxp,max=-1000000;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%s",s[i]);
		scanf("%d%d%d%d%d%d%d",&p,&m,&a,&b,&c,&d,&e);
		sc[i]=p*100-m*50+a+b+c+d+e;
		if(sc[i]>max)max=sc[i],maxp=i;
	}
	printf("%s\n",s[maxp]);
}
