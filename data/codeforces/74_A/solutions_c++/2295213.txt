#include <iostream>

using namespace std;

int n;
string nick, lead;
int p,m,a,b,c,d,e;
int maxx;

int main(){
	maxx = -10000;
	cin >> n;
	for(int i = 0;i<n;i++){
		cin >> nick>>p>>m>>a>>b>>c>>d>>e;
		int sum = p*100-m*50+a+b+c+d+e;
		if(sum > maxx){
			maxx = sum;
			lead = nick;
		}
	}
	cout << lead;

	return 0;
}
