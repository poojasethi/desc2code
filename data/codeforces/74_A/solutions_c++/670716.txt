#include<stdio.h>
#include<string.h>
int main(){
	int n,p,m,a,b,c,d,e,po,best;
	char na[1000],bna[1000];
	scanf("%d",&n);
	best=-9999999;
	for(;n>0;n--){
		scanf("%s%d%d%d%d%d%d%d",na,&p,&m,&a,&b,&c,&d,&e);
		po=100*p-50*m+a+b+c+d+e;
		if(po>best){
			best=po;
			strcpy(bna,na);
		}
	}
	printf("%s\n",bna);
}
