#include<iostream>
#include<string>
using namespace std;

int main(){
  int n;
  string ans="";
  int m=-100000000;
  cin>>n;
  while(n--){
    string name;
    int ok;
    int ng;
    int a,b,c,d,e;
    cin>>name>>ok>>ng>>a>>b>>c>>d>>e;
    int p = ok*100 - ng * 50 + a + b + c + d+ e;
    if(m < p){
      m = p;
      ans = name;
    }
  }
  cout<<ans<<endl;
}
