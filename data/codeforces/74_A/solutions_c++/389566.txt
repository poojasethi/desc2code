#include<cstdio>
#include<iostream>
#include<string>
using namespace std;
string S,res;
int main()
{
	int n,mmax=0;
	cin>>n;
	long long tmp,p=0,x,y;
	for(int i=0;i<n;i++)
	{
		p=0;
		cin>>S;
		cin>>x>>y;
		p+=x*100;
		p-=y*50;
		for(int j=0;j<5;j++)
		{
			cin>>tmp;
			p+=tmp;
		}
		if(p>mmax||i==0)
		{
			mmax=p;
			res=S;
		}
	}
	cout<<res<<endl;
}
