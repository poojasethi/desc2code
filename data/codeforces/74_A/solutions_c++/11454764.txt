#include<iostream>
#include<map>
#include<cstdio>
using namespace std;
int main()
{
  int p,m,a,b,c,d,e,n,total;
  string handle;
  map<int,string> mp;
  scanf("%d",&n);
  while(n--)
  {
   cin>>handle>>p>>m>>a>>b>>c>>d>>e;
   total = p*100-m*50+a+b+c+d+e;
   mp[total] = handle;
  }
  cout<<mp.rbegin()->second;
  return 0;
}
