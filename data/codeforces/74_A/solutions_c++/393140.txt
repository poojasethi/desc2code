#include<iostream>
#include<string>
using namespace std;
const int MAX = 1000000;
int main()
{
	string s,ans;
	int score,k;
	int p,m,a,b,c,d,e;
	int n;
	cin>>n;
	score = -MAX;
	while(n--)
	{
		cin>>s>>p>>m>>a>>b>>c>>d>>e;
		k = 100*p-50*m+a+b+c+d+e;
		if(score < k)
		{
			score = k;
			ans = s;
		}
	}
	cout<<ans<<endl;
	return 0;
}
