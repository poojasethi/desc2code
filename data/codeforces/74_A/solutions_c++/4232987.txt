#include<iostream>
#include<cstring>
#include<string>
#include<cstdio>
using namespace std;

int main(){
	int n,p,m,a,b,c,d,e,Max=-10000;
	string ans,s;
	cin>>n;
	while(n--){
		int temp=0;
		cin>>s>>p>>m>>a>>b>>c>>d>>e;
		temp=100*p-50*m+a+b+c+d+e;
		if(temp>Max){
			Max=temp;
			ans=s;
		}
	}
	cout<<ans<<endl;
	return 0;
}
