#include<iostream>
#include<string>
using namespace std;
int main()
{
    int n,sh,uh,a,b,c,d,e,m=0,max=-2501;
    string h,s;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    m=0;
    cin>>h>>sh>>uh>>a>>b>>c>>d>>e;
    m+=a+b+c+d+e+100*sh-50*uh;
    if(m>max)
    {
    max=m;
    s=h;
}
}
    cout<<s<<endl;
    return 0;
}
