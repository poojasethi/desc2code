#include <stdio.h>
int main()
{
    int max=0,a,b,c,d,e,x,y,i,n,s[51]={0};
    char name[51][21];
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%s %d %d %d %d %d %d %d",name[i],&x,&y,&a,&b,&c,&d,&e);
        s[i]=100*x-50*y+a+b+c+d+e;
        if(s[max]<s[i])
        max=i;
    }
    printf("%s\n",name[max]);
    return 0;
}
