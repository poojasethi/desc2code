#include<iostream>
#include<string>
using namespace std;
long i,j,k,a,b,c,d,e,maxn=-1<<28,n,ans;
string s,t;
int main()
{
 cin>>n;
 for(i=1;i<=n;i++)
 {
  cin>>s>>j>>k>>a>>b>>c>>d>>e;
  ans=a+b+c+d+e+j*100-k*50;
  if(ans>maxn){maxn=ans;t=s;}               
                  }   
                  
  cout<<t<<endl;
  return 0;                
 }
