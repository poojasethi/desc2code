#include <cstdio>
#include <algorithm>
#include <vector>
using namespace std;

const int N=100000;
vector<int> primes;
int n,x;
int a[N];

void gen_primes()
{
    primes.resize(1,2);
    for(int i=3;i<x && primes.size()<=n;i++)
    {
        bool is_prime=true;
        for(int j=0;primes[j]*primes[j]<=i;j++)
            if(i%primes[j]==0)
            {
                is_prime=false;
                break;
            }
        if(is_prime)
            primes.push_back(i);
    }
}

int main()
{
    //freopen("input.txt","r",stdin);
    //freopen("output2.txt","w",stdout);
    scanf("%d%d",&n,&x);
    if(x==2)
    {
        printf("0");
        return 0;
    }
    gen_primes();
    for(int i=0;i<n;i++)
        scanf("%d",a+i);
    sort(a,a+n);
    if(a[0]==1)
    {
        printf("1");
        return 0;
    }
    if(primes.size()>n)
    {
        printf("-1");
        return 0;
    }
    int t=0;
    for(int i=0;i<n && t<primes.size();i++)
        if(a[i]==primes[t])
            t++;
    if(t<primes.size())
        printf("-1");
    else
        printf("%d",primes.size());
    return 0;
}
