#include <cstdio>
#include <algorithm>

using namespace std;

const int MAX_X = 2000000;

bool prime[MAX_X];
bool vis[MAX_X];

int main() {
	fill(prime, prime + MAX_X, true);
	prime[0] = prime[1] = false;
	for (int i = 2; i < MAX_X; i++) if (prime[i])
		for (int j = i+i; j < MAX_X; j += i)
			prime[j] = false;
	
	int n, x; scanf("%d %d", &n, &x);
	if (x == 2) {
		printf("0\n");
		return 0;
	}

	for (int i = 0; i < n; i++) {
		int val; scanf("%d", &val);
		if (val == 1) {
			printf("1\n");
			return 0;
		}
		if (val < MAX_X)
			vis[val] = true;
	}

	if (x >= MAX_X)
		printf("-1\n");
	else if (x == 2)
		printf("0\n");
	else {
		int ans = 0;
		for (int i = 2; i < x; i++) if (prime[i]) {
			if (vis[i])
				ans++;
			else {
				ans = -1;
				break;
			}
		}
		printf("%d\n", ans);
	}

	return 0;
}
