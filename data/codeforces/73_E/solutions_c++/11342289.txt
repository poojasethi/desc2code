#include <cstdio>
#include <set>
//codeforces
//morrowindows
using namespace std;
const int mx=2000000;
int i,j,n,r,x,y;
bool a[mx];
set <int> s;
int main() {
  for (i=2; i<mx; i++) 
   if (!a[i]) for (j=i+i; j<mx; j+=i) a[j]=true;
   //only prime nos will be set to false, and all the rest 
   //composite nos will be set to true
   
  
  scanf("%d%d",&n,&x); //get the input
  
  if (x==2) { puts("0"); return 0; }
  
  //get the number of ai ie the lists of items that displayed on a page
  for (i=0; i<n; i++) {
    scanf("%d",&y);
    if (y==1) { puts("1"); return 0; }//if 1 on each page
    //then obviously, the number of pages is the ans...
    
    //else, insert it in set
    s.insert(y);
  }
  
  
  if (x>=mx) { puts("-1"); return 0; }//for overlimit
  
  
  for (i=2; i<x; i++) 
  if (!a[i]) { //if prime number encountered...
  
  //thus all the prime nos that are less than x, must be present in the set
  //and this will be the solution.
  	
    if (s.find(i)==s.end()){ puts("-1"); return 0; } 
	else r++;
  }
  
  
  printf("%d\n",r);
  return 0;
}
