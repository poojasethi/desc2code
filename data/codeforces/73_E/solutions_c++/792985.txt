#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

const int maxn=1500000+10;
int f[maxn],a[maxn/10],b[maxn/10];
int n,m,tt;

void solve(){
    if(m==2){printf("0\n");return;}
    for(int i=2;i<maxn;++i){
        if(f[i]==0){
            a[++tt]=i;
            if(tt==n+1)break;
        }
        for(int j=1;j<=tt;++j){
            if(i*a[j]>maxn)break;
            f[i*a[j]]=i;
            if(i%a[j]==0)break;
        }
    }
    sort(b,b+n);
    if(find(b,b+n,1)-b<n){printf("1\n");return;}
    if(m>a[tt]){printf("-1\n");return;}
    int i,j;
    for(i=0,j=1;i<n;++i)
        if(b[i]==a[j]){
            ++j;
            if((j>tt)||(a[j]>=m))break;
        }
    a[tt+1]=2147483647;
    if(a[j]>=m)
        printf("%d\n",j-1);
    else printf("-1\n");
}

int main(){
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;++i)scanf("%d",&b[i]);
    solve();
    return 0;
}
