#include <cstdio>
#include <set>

using namespace std;

const int maxp = 20 * 100 * 1000 + 5;
const int maxn = 2 * 100 * 1000;
bool p[maxp];
int primes[maxn], a[maxn];
bool used[maxn];

int main()
{	
	int n, x;
	scanf("%d%d", &n, &x);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &a[i]);
	}
	
	int m = 0;
	x = min(x, maxp);
	for (int i = 2; i < x; ++i) {
		if (!p[i]) {
			primes[m++] = i;
		}
		else {
			continue;
		}
		for (long long j = i * 1LL * i; j < x; j += i) {
			p[j] = 1;
		}
	}

	if (x == 2) {
		printf("0\n");
		return 0;
	}
	
	int cnt = 0;
	for (int i = 0; i < n; ++i) {
		int pos = lower_bound(primes, primes + m, a[i]) - primes;
		if (pos != m && primes[pos] == a[i] && !used[pos]) {
			used[pos] = 1;
			++cnt;
		}
		if (a[i] == 1) {
			printf("1\n");
			return 0;
		}
	}
	
	printf("%d\n", m == cnt ? m : -1);
	
	return 0;
}
