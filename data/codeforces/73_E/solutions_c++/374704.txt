using namespace std;

#define REP(i,n) for((i)=0;(i)<(int)(n);(i)++)
#define foreach(c,itr) for(__typeof((c).begin()) itr=(c).begin();itr!=(c).end();itr++)

#define M 1500000
bool prime[M+10];

bool a[M+10];

int func(int N, int x){
    int i,ans=0;
    if(x == 2) return 0;
    if(a[1]) return 1;
    if(x >= M) return -1;
    for(i=2;i<=x-1;i++) if(prime[i]) if(a[i]) ans++; else return -1;
    return ans;
}

int main(void){
    int i,j,N,x,tmp;
    
    for(i=2;i<M;i++) prime[i] = true;
    for(i=2;i*i<M;i++) if(prime[i]) for(j=i*i;j<M;j+=i) prime[j] = false;
    
    scanf("%d%d",&N,&x);
    REP(i,N){
        scanf("%d",&tmp);
        if(tmp < M) a[tmp] = true;
    }
    
    int ans = func(N,x);
    cout << ans << endl;
    
    return 0;
}
