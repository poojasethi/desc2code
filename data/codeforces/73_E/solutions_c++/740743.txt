#include<iostream>
#include<set>
#include<cstring>
#include<cstdio>
using namespace std;
const int maxn = 2000000;
bool ss[ maxn ] = { 1, 1 };
set< int > s;

int main(){
    int n, x;
    scanf("%d%d", &n, &x);
    for(int i = 0; i < n; ++ i) {
          int j;
          scanf("%d", &j);
          s.insert( j ) ;          
    }
    if(x == 2) puts("0");
    else if( s.count( 1) > 0) puts("1");
    else if(x >= maxn) puts("-1");
    else {
        int ct = 0; 
        for(int i = 2; i < x; ++ i) if(! ss[ i ]) {
           if(! s.count( i) ) {
               puts("-1");
               return 0;
           }
           ++ ct;
           for(int j = i; j < x; j += i) ss[ j ] = true;  
        }
        cout << ct << endl;    
    } 
    return 0;    
}
