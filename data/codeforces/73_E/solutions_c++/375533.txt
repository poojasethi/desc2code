#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int maxn=1300000;
int a[100005];
int p[100005],np;
bool dp[maxn];
int main()
{
	int n,x,i,j;
	scanf("%d %d",&n,&x);
	for (i=0;i<n;i++)
		scanf("%d",a+i);
	sort(a,a+n);
	if (x==2)
		printf("0\n");
	else if (a[0]==1)
		printf("1\n");
	else
	{
		n=unique(a,a+n)-a;
		np=0;
		for (i=2;i<maxn&&np<=n;i++)
		{
			if (dp[i]==0)
			{
				p[np]=i;
				np++;
				for (j=i+i;j<maxn;j+=i)
					dp[j]=1;
			}
		}
		i=0;
		for (j=0;j<np&&p[j]<x;j++)
		{
			while (i<n&&a[i]<p[j])
				i++;
			if (i==n||a[i]>p[j])
			{
				printf("-1\n");
				j=-1;
				break;
			}
		}
		if (j!=-1)
			printf("%d\n",j);
	}
	return 0;
}
