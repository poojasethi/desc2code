template <class T> inline void chkmin(T &a, T b) { if (a>b) a=b; }
template <class T> inline void chkmax(T &a, T b) { if (a<b) a=b; }

#define NN 2100000

int prime[NN], vst[NN];

int calc(int x) {
	int i, ans=0;
	if (x==2) return 0;
	if (vst[1]) return 1;
	for (i=2; i<x; i++) if (prime[i]) {
		if (vst[i]) ans++;
		else return -1;
	}
	return ans;
}

int main() {
	int i, j, N, x, a;
	
	prime[2]=1;
	for (i=3; i<NN; i+=2) prime[i]=1;
	for (i=3; i*i<NN; i++) if (prime[i]) {
		for (j=i*i; j<NN; j+=i) prime[j]=0;
	}
	
	scanf("%d%d", &N, &x);
	for (i=0; i<N; i++) {
		scanf("%d", &a);
		if (a>NN) continue;
		vst[a]=1;
	}
	
	printf("%d\n", calc(x));
	
	return 0;
}
