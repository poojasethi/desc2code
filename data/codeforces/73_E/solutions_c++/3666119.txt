#include <cstdio>
#include <cmath>
#include <set>
using namespace std;

#define N 2000001
#define LL long long

int comp[N], has[N];
set<int> pri;

void pre()
{
	for (LL i = 2; i < N; ++i)
	{
		if (comp[i]) continue;
		for (LL j = i*i; j < N; j += i)
		{
			comp[j] = 1;
		}
	}
	for (int i = 2; i < N; ++i) if (!comp[i])
		pri.insert(i);
}

int main()
{
	pre();
	//printf("%d\n", pri.size());
	//printf("100000 1299721\n");
	//for (int i = 2, c = 0; i < N && c < 100000; ++i)
	//{
	//	if (!comp[i]) printf("%d\n", i);
	//}
	//printf("\n");
	//return 0;
	int n, x;
	scanf("%d%d", &n, &x);
	if (x == 2)
	{
		printf("0\n");
		return 0;
	}
	for (int i = 0; i < n; ++i)
	{
		int a; scanf("%d", &a);
		if (a >= N) continue;
		has[a] = 1;
	}
	if (has[1])
	{
		printf("1\n");
		return 0;
	}
	if (x > *pri.rbegin())
	{
		printf("-1\n");
		return 0;
	}
	int r = 0, s = 2;
	while (s < x)
	{
		if (*pri.begin() == s)
		{
			if (!has[s]) 
			{
				r = -1;
				break;
			}
			++r;
			pri.erase(pri.begin());
		}
		++s;
	}
	printf("%d\n", r);
}
