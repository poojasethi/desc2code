#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

const int N = 3000000;
int a[3000005];
int d[3000005];
int p[100005], pn;
int r[100005];

int main(void) {
	int n, x;
	scanf("%d%d", &n, &x);
	for(int i=2;i<=N && pn<100001;i++) {
		if(!a[i]) {
			d[i] = pn;
			p[pn++] = i;
			for(int j=i+i;j<=N;j+=i)
				a[j]=1;
		}
	}
	int c=0, q=0;
	for(int i=0;i<n;i++) {
		scanf("%d", &r[i]);
		if(r[i]==1) q=1;
		if(r[i]<=N && a[r[i]]==0) {
			a[r[i]] = 1;
			if(r[i]<x) ++c;
		}
	}
	int f=0;
	for(int i=2;i<x;i++)
		if(!a[i]) {
			f=1;
			break;
		}
	if(x==2) puts("0");
	else if(q) puts("1");
	else if(f) puts("-1");
	else printf("%d\n", c);

	return 0;
}
