#include <cstdio>

using namespace std;

bool a[2000001],p[2000001];

int main(void)
{
  int n,m;
  scanf("%d %d",&n,&m);
  if(m==2)
  {
    printf("0\n");
    return 0;
  }
  for(int i=1;i<=n;++i)
  {
    int k;
    scanf("%d",&k);
    if((k<m)&&(k<20*n))
      a[k]=true;
  }
  if(a[1])
  {
    printf("1\n");
    return 0;
  }
  if(m>20*n)
  {
    printf("-1\n");
    return 0;
  }
  for(int i=2;i<m;++i)
    p[i]=true;
  bool flag=true;
  int ans=0;
  for(int i=2;i<m;++i)
  {
    if(!p[i])
      continue;
    if(!a[i])
    {
      flag=false;
      break;
    }
    ++ans;
    long long j=(long long)i*i;
    while(j<m)
    {
      p[j]=false;
      j+=i;
    }
  }
  if(flag)
    printf("%d\n",ans);
  else
    printf("-1\n");
  return 0;
}
