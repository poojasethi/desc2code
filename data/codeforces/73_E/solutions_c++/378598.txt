#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <vector>
#include <climits>
#define MAXN 2000000
using namespace std;

int N,X,bound = sqrt(INT_MAX);
int A[100005],t,L;
vector <int> S;
bool prime[MAXN+5];

int main(){
    scanf("%d%d",&N,&X);
    for(int i=0;i<N;++i) scanf("%d",&A[i]);
    sort(A,A+N);
    if(X == 2) printf("0\n");
    else if(A[0] == 1) printf("1\n");
    else if(X >= MAXN){
        printf("-1\n");
        return 0;
    }
    else{
        for(int i=2;i<=MAXN;++i)
            if(!prime[i]){
                S.push_back(i);
                if(i <= bound)
                    for(int j=i*i;j<=MAXN;j+=i) prime[j] = 1;
            }
        L = S.size();
        for(int i=0;i<L;++i){
            if(S[i] >= X){
                printf("%d\n",i);
                return 0;
            }
            while(t < N && A[t] != S[i]) ++t;
            if(t == N){
                printf("-1\n");
                return 0;
            }
        }
    }
}
