#include<cstdio>
#include<cstdlib>
#include<set>
const int  Mx=2000000;
using namespace std;
int n,x;
int f[Mx];
set<int> a;

int main(){
    f[1]=1; 
    for (int i=2;i<Mx;i++) 
        if (!f[i]) 
            for (int j=i+i;j<Mx;j+=i) f[j]=1;
    scanf("%d %d",&n,&x);
    if (x==2){
        printf("0");
        return 0;
    }
    for (int i=0;i<n;i++){
        int xx;
        scanf("%d",&xx);
        if (xx==1){
            printf("1");
            return 0;
        }
        a.insert(xx);
    }
    if (x>Mx) printf("-1");
    else{
        int r=0;
        for (int i=1;i<x;i++)
            if (!f[i]){
                if (a.find(i)!=a.end()) r++;
                else{
                    r=-1;
                    break;
                }
            }
        printf("%d\n",r);
    }
}
