#include <iostream>
#include <cstdio>
using namespace std;
// ERATOSTEN_FAST
#define SIZE 1300010
int lp[SIZE],PR[110000],br=1;
bool is_prime[SIZE];
void gen_primes(){
  int i,j,neo;
  for(i=2;i<SIZE;++i){
   if(!lp[i]){
    lp[i]=br;is_prime[i]=true;
    PR[br++]=i;
             }
   for(j=1;j<=lp[i];++j){
    neo=PR[j]*i;
    if(neo<SIZE)lp[neo]=j;
    else break;
                        }
                     }
}
int main(){
  bool PD=false,PD1=false;
  int i,j,n,x,ANS=0;
  //system("pause");
  gen_primes();
  scanf("%d%d",&n,&x);
  if(x==2){printf("0\n");return 0;}
  if(x>=SIZE)PD1=true;
  for(i=0;i<n;++i){
   scanf("%d",&j);
   if(j==1)PD=true;
   if(j<x && j<SIZE && is_prime[j]){++ANS;is_prime[j]=false;}
                  }
  if(PD){printf("1\n");return 0;}
  else if(PD1){printf("-1\n");return 0;}
  bool HAS_SOL=true;
  for(i=x-1;i>1;--i)
   if(is_prime[i]){HAS_SOL=false;break;}
  if(HAS_SOL)printf("%d\n",ANS);
  else printf("-1\n");
  //system("pause");
  return 0;
}
