#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdlib>

using namespace std;

const int limite=10000000;
int esprimo[limite];

int n,x;
int a[limite];

int main()
{
  for (int i=2;i<limite;i++)
    esprimo[i]=1;
  for (int i=2;i<limite;i++)
    if (esprimo[i])
      for (int j=i+i;j<limite;j+=i)
	esprimo[j]=0;
  cin>>n>>x;
  for (int i=0;i<n;i++)
    cin>>a[i];
  sort(a,a+n);
  if (a[0]==1) {
    if (x==2)
      cout<<0<<endl;
    else
      cout<<1<<endl;
    exit(0);
  }
  int amount=0;
  int ia=0;
  for (int i=2;i<x;i++) {
    if (esprimo[i]) {
      while (ia<n and a[ia]<i) ia++;
      if (ia==n or a[ia]>i) {
	cout<<-1<<endl;
	exit(0);
      }
      amount++;
    }
  }
  cout<<amount<<endl;
}
