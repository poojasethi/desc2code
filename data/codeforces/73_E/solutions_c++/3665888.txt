#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
#include <iostream>
using namespace std;
typedef long long ll;

int gcd(int a, int b){ return b ? gcd(b,a%b) : a; }


#define fr(i,a,b) for(int i=a;i<b;++i)
#define rep(i,b) fr(i,0,b)
#define N 2100000
bool mk[N+10];
int ar[400100];
int g[100100];
int main(){
    int cnt = 0;
    for(ll i=2;i<=N;i++){
        if(!mk[i]){
            ar[cnt++] = i;
            for(ll j=i*i;j<=N;j+=i) mk[j] = true;
        }
    }
    //printf("%d\n",cnt);
    int n, x;
    scanf("%d %d",&n,&x);
    rep(i,n) scanf("%d",&g[i]);
    sort(g,g+n);
    n = unique(g,g+n)-g;
    if(x==2) printf("0\n");
    else if(g[0]==1) printf("1\n");
    else if(x>=N) printf("-1\n");
    else {
        //printf(">>>\n");
        int p0=0;
        rep(i,n){
            if(ar[p0]>x-1) break;
            if(g[i]==ar[p0]) p0++;
        }
        //printf(" >> %d\n",p0);
        int pos = lower_bound(ar,ar+cnt,x-1)-ar;
        if(ar[pos]>x-1) pos--;
        if(p0 == pos+1) printf("%d\n",p0);
        else printf("-1\n");
    }
    return 0;
}
