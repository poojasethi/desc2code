#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 100010
#define maxm 1300000

using namespace std;

int tag[maxm];
int pr[maxn+10000],pn;
int a[maxn];
int n,x;
int work(){
	if (x==2) return 0;
	if (a[0]==1) return x==2 ? 0 : 1;
	int i=0,j,k;
	for (j=0;j<pn && pr[j]<x;j++){
		while (i<n && a[i]<pr[j]) i++;
		if (i==n || a[i]>pr[j]) return -1;
	}
	return j;
}
int main(){
	int i,j,k;
	pn=0;
	for (i=2;i<maxm;i++)
		if (!tag[i]){
			pr[pn++]=i;
			for (j=i+i;j<maxm;j+=i) tag[j]=1;
		}
	scanf("%d %d",&n,&x);
	for (i=0;i<n;i++) scanf("%d",a+i);
	sort(a,a+n);
	int ans=work();
	printf("%d\n",ans);
	return 0;
}
