n,x = map(int,raw_input().split())
a = set(map(int,raw_input().split()))
if 1 in a and x>2: print 1
elif x>1300000: print -1
else:
    pr = range(x)
    for i in xrange(2,x):
        if not pr[i]: continue
        ii=i*i
        if ii>x: break
        pr[ii::i]=[0]*len(pr[ii::i])     
    pr = set(filter(None,pr)[1:])
    print -1 if len(pr-a) else len(pr)
