n,x = map(int,raw_input().split())
a = set(map(int,raw_input().split()))
if x==2: print 0
elif 1 in a: print 1
elif x>1300000: print -1
else:
    pr = range(x)
    for i in xrange(2,x):
        if not pr[i]: continue
        ii=i*i
        if ii>x: break
        pr[ii::i]=[0]*len(pr[ii::i])     
    pr = filter(None,pr)[1:]
    if all(p in a for p in pr): print len(pr)
