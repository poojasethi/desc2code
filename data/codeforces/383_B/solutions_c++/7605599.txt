#include <cstdio>
#include <vector>
#include <algorithm>

#define reset() ++s,j=0,sz=l[s].size()
#define pb push_back
#define ll first
#define rr second

using namespace std;

typedef pair<int,int> pii;
int sz,n,m,s,j;
struct Point{int x,y;}p[100005];
vector<pii>l[200005];

inline bool cmp(const Point &i,const Point &j){return i.x^j.x?i.x<j.x:i.y<j.y;}

inline void ck(int L,int R){
    if(R<L)return;
    while(j<sz && l[s][j].rr<L)++j;
    if(j>=sz || l[s][j].rr<L || R<l[s][j].ll)return;
    L=max(L,l[s][j].ll);
    l[s+1].pb(pii(L,R));
}

int main(){
    scanf("%d%d",&n,&m);
    for(int i=1;i<=m;++i)scanf("%d%d",&p[i].x,&p[i].y);
    sort(p+1,p+m+1,cmp);
    l[s=sz=1].pb(pii(1,1));
    for(int i=1;i<=m;++i){
        if(p[i].x^p[i-1].x)
            if(p[i].x==p[i-1].x+1)
                ck(1,p[i].y-1);
            else ck(1,n),reset(),ck(1,p[i].y-1);
        else ck(p[i-1].y+1,p[i].y-1);
        if(p[i].x^p[i+1].x)ck(p[i].y+1,n),reset();
    }
    if(p[m].x<n && sz || p[m].x==n && sz && l[s][sz-1].rr==n)printf("%d\n",n*2-2);
    else puts("-1");
}
