#include<cstdio>
#include<vector>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

#define MAXN 100010

int n, m;
pair<int, int> a[MAXN];

int main()
{
	scanf("%d %d", &n, &m);
	for (int i = 0; i < m; ++i)
		scanf("%d %d", &a[i].first, &a[i].second);
	sort(a, a+m);
	vector< pair<int, int> > last;
	last.push_back(make_pair(1, 1));
	int lx = 0;
	for (int i = 0; i < m; )
	{
		int x = a[i].first;
		if (x != lx + 1)
		{
			int u = last.begin()->first;
			last.clear();
			last.push_back( make_pair(u, n) );
			}
		lx = x;
		vector<int> now;
		now.push_back(0);
		now.push_back(n+1);
		for (;i <= m && a[i].first == x; ++i)
			now.push_back(a[i].second);
		sort(now.begin(), now.end());
		int ll = 0;
		int j;
		vector< pair<int, int> > ne;
		for(j=0; j+1 < (int)now.size(); ++j)
		{
			int l=now[j]+1;
			int r=now[j+1]-1;
			if (l>r) continue;
			for(;ll<(int)last.size() && last[ll].second<l;++ll);
			if(ll==(int)last.size() || last[ll].first>r) continue; 	
			ne.push_back( make_pair(max( last[ll].first, l) , r) );
		}
		last = ne;
		if(!ne.size() || (x==n && (--ne.end())->second!=n))
		{
			printf("-1\n");
			return 0;
		}
		}
	printf("%d\n", 2*(n-1));
	return 0;
	}
