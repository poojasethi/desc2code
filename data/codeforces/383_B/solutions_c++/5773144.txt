#include <bits/stdc++.h>

#define st first
#define nd second
#define mp make_pair
#define pb push_back
#define lli long long int
#define FP( ii,aa,bb ) for( int ii=aa;ii<=bb;ii++ )
#define FM( ii,aa,bb ) for( int ii=aa;ii>=bb;ii-- )

using namespace std;

int n,m,x;

pair<int,int> p[400000];
vector< pair<int,int> >	a,b,c;

char ctrl( pair< int,int > x,pair< int,int > y ){
	return max( x.st,y.st )<=min( x.nd,y.nd );
}

int main(){

	int ind,k;

	cin >> n >> m;
	FP( i,1,m )
		cin >> p[i].st >> p[i].nd;

	sort( p+1,p+m+1 );
	
	ind = m;
	FP( i,1,m ){
		
		p[++ind] = mp( p[i].st,n+1 );
		p[++ind] = mp( p[i].st,0 );
		while( i<m and p[i+1].st==p[i].st )	i++;

	}

	m = ind;
	p[++m] = mp( n,0 );
	p[++m] = mp( n,n+1 );
	sort( p+1,p+m+1 );
	m = unique( p+1,p+m+1 )-p-1;

	a.pb( mp( 1,1 ) );

	FP( i,1,m ){
		
		if( p[i].st>x+1 ){
			x = a[0].st;
			a.clear();
			a.pb( mp( x,n ) );
		}

		x = p[i].st;
		
		b.clear();
		do{
			
			if( p[i].st==p[i+1].st and p[i].nd+1<p[i+1].nd ) b.pb( mp( p[i].nd+1,p[i+1].nd-1 ) );
			i++;

		}while( i<=m and p[i].st==p[i-1].st );
		i--;

		k = 0;
		c.clear();

		FP( j,0,(int)b.size()-1 ){
			while( k<a.size() and b[j].st>a[k].nd )	k++;
			if( k==a.size() )	break;
			if( ctrl( b[j],a[k] ) )	c.pb( mp( max( b[j].st,a[k].st ),b[j].nd ) );
		}
		
		a = c;
		if( a.size()==0 ){ puts("-1"); return 0; }

	}

	if( a[ a.size()-1 ].nd<n )	puts("-1");
	else	cout << 2*n-2 << endl;

}
