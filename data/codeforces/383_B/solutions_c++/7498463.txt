#include<cstdio>
#include<cstdlib>
#include<vector>
#include<algorithm>
using namespace std;
#define X first
#define Y second
#define pb push_back
#define rep(i,n) for(int i=1;i<=n;i++)
typedef pair<int,int> pii;
vector <pii> f,g;
int x[100005],y[100005],id[100005],j,lx,ly,m,n;
bool cmp(int i,int j){return x[i]==x[j]?y[i]<y[j]:x[i]<x[j];}
void ins(int &l,int r){
	while (j<f.size()&&f[j].Y<=l) j++;
	if(j<f.size()&&f[j].X<r&&l+1<=r-1) g.pb(pii(max(f[j].X,l+1),r-1)); l=r;
}
bool work(){
	f.pb(pii(1,1));
	lx=1; ly=0;
	rep(k,m){
		int i=id[k];
		if(x[i]==lx) ins(ly,y[i]);
		else{
			ins(ly,n+1); ly=0;
			if(g.empty()) return 0;
			if(x[i]==lx+1) f=g;
			else f.clear(),f.pb(pii(g[0].X,n));
			g.clear(); j=0; lx=x[i];
			ins(ly,y[i]);
		}
	}
	ins(ly,n+1); if(g.empty()) return 0;
	return lx==n?g[g.size()-1].Y==n:1;
}
int main(){
	scanf("%d%d",&n,&m);
	rep(i,m) scanf("%d%d",&x[i],&y[i]),id[i]=i;
	sort(id+1,id+1+m,cmp);
	printf("%d\n",work()?n*2-2:-1);
}
