#include<cstdio>
#include<algorithm>
#include<cstring>
#include<iostream>
#include<vector>
#include<map>
#include<queue>
using namespace std;
#define N 300000
int range,n,x[N],y[N],vis[N];
map<int,map<int,int> >s1,s2;
queue<int>Q;
int main(){
	scanf("%d%d",&range,&n);
	for(int i=1;i<=n;i++){
		scanf("%d%d",&x[i],&y[i]);
		if(y[i]==range||x[i]==1)Q.push(i),vis[i]=1;
		s1[x[i]][y[i]]=i;
		s2[y[i]][-x[i]]=i;
	}
	while(!Q.empty()){
		int id=Q.front();Q.pop();
		if(x[id]==range||y[id]==1)return puts("-1"),0;
		int px=x[id],py=y[id];
		if(s1.count(px+1)){
			map<int,int>::iterator it=s1[px+1].lower_bound(py-1);
			if(it!=s1[px+1].end()&&!vis[it->second])vis[it->second]=1,Q.push(it->second);
		}if(s2.count(py-1)){
			map<int,int>::iterator it=s2[py-1].lower_bound(-(px+1));
			if(it!=s2[py-1].end()&&!vis[it->second])vis[it->second]=1,Q.push(it->second);
		}
	}printf("%d\n",range*2-2);
}
