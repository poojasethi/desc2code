#define pb push_back
#define REP(i,l,r) for((i)=(l);(i)<=(r);++(i))
#define REP2(i,l,r) for((i)=(l);(i)!=(r);++(i))
using namespace std;

typedef long long LL;
typedef double ld;

const int MAX=100000+10;

int n,m;
pair<int,int> d[MAX];

int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);freopen("output.txt","w",stdout);
#endif
	int i;
	scanf("%d%d",&n,&m);
	REP(i,1,m)
		scanf("%d%d",&d[i].x,&d[i].y);
	sort(d+1,d+m+1);
	vector< pair<int,int> > last;
	last.pb( mp(1,1) );
	int lx=0;
	for(i=1;i<=m;)
	{
		int x=d[i].x;
		if(x!=lx+1)
		{
			int u=last.begin()->x;
			last.clear();
			last.pb(mp(u,n));
		}
		lx=x;
		vector<int> now;
		now.pb(0);
		now.pb(n+1);
		for(;i<=m && d[i].x==x;++i)
			now.pb(d[i].y);
		sort(now.begin(),now.end());
		int ll=0;
		int j;
		vector< pair<int,int> > ne;
		for(j=0;j+1<(int)now.size();++j)
		{
			int l=now[j]+1;
			int r=now[j+1]-1;
			if(l>r)
				continue;
			for(;ll<(int)last.size() && last[ll].y<l;++ll)
				;
			if(ll==(int)last.size() || last[ll].x>r)
				continue;
			ne.pb( mp(max( last[ll].x, l) , r) );
		}
		last=ne;
		if(!ne.size() || (x==n && (--ne.end())->y!=n))
		{
			printf("-1\n");
			return 0;
		}
	}
	printf("%d\n",n*2-2);
	return 0;
}
