#include <cstdio>

int f[1<<24];
char s[5];
int n;

int main() {
//  freopen("input.txt", "r", stdin);
//  freopen("output2.txt", "w", stdout);
    
    scanf("%d", &n);
    for (int i = 1; i <= n; ++ i) {
        scanf("%s", s);
        int x = 0; for (int k = 0; k < 3; ++ k) x |= 1 << s[k]-'a';
        ++ f[x];
    }
    int ALL = (1 << 24) - 1, res = 0;
    for (int k = 0; k < 24; ++ k)
        for (int S = 0; S <= ALL; ++ S) if (S>>k&1) {
            f[S] += f[S^1<<k];
        }
    
    for (int S = 0; S <= ALL; ++ S) res ^= (n-f[S]) * (n-f[S]);
    printf("%d\n", res);
    
//  for(;;);
    return 0;
}
