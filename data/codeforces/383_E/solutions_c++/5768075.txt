#include <iostream>
#include <string>

using namespace std;

int n, cnt[1<<24];

void input(){
	cin >> n;

	int i;
	for(i = 0; i<n; i++){
		string t;
		cin >> t;

		int j, bit=0;
		for(j = 0; j<3; j++)
			bit |= 1<<(t[j]-'a');
		cnt[bit]++;
	}
}

void init(){
	int i, j;
	for(i = 0; i<24; i++){
		for(j = 0; j<(1<<24); j++){
			if((j & (1<<i)) == 0){
				cnt[j | (1<<i)] += cnt[j];
			}
		}
	}
}

void solve(){
	int i, res = 0;
	for(i = 0; i<(1<<24); i++){
		int t = cnt[i];
		t = n-t;
		res ^= t*t;
	}

	cout << res;
}

int main(){
	input();

	init();
	solve();

	return 0;
}
