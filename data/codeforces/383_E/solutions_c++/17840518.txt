#include<cstdio>
#define rep(i,n) for (int i=0;i<n;++i)
int n,m=24,x,ans; int f[1<<24]; char s[5];
int main()
{
    #ifdef DouBi
    freopen("in.cpp","r",stdin);
    #endif // DouBi
	while(scanf("%d",&n)!=EOF){
        rep(i,n){
            scanf("%s",s),x=0;
            rep(j,3) x|=1<<s[j]-97; ++f[x];
        }
        rep(j,m){
            rep(i,1<<m) if(i&(1<<j)) f[i]+=f[i^(1<<j)];
            //rep(i,1<<m)printf("%d ",f[i]);printf("\n");
        }
        rep(i,1<<m) ans^=(n-f[i])*(n-f[i]);
        printf("%d\n",ans);
	}
	return 0;
}
