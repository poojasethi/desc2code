#include <cstdio>

#define ri register int

const int lim=1<<24;

int n,f[lim];

int main(){
    scanf("%d",&n);
    for(ri i=1;i<=n;++i){
        scanf("\n");
        ri a=1<<getchar()-97,b=1<<getchar()-97,c=1<<getchar()-97;
        ++f[a],++f[b],++f[c],--f[a|b],--f[a|c],--f[c|b],++f[a|b|c];
    }
    for(ri i=1;i<lim;i<<=1)
    for(ri j=0;j<lim;++j)
    if(j&i)f[j]+=f[j-i];
    ri ans=0;
    for(ri i=0;i<lim;++i)ans^=f[i]*f[i];
    printf("%d\n",ans);
}
