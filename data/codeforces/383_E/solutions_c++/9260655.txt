#include <stdio.h>

int dy[1<<24];
int n;
char ch[5];

int main(void){
	int i, j, x, answer;
	scanf("%d\n", &n);
	for (j=1;j<=n;j++){
		scanf("%s", &ch[1]);
		x=0;
		for (i=1;i<=3;i++) x|=(1<<(ch[i]-'a'));
		dy[x]++;
	}

	for (i=0;i<24;i++){
		for (j=0;j<(1<<24);j++){
			if (j&(1<<i)) dy[j]+=dy[j^(1<<i)];
		}
	}
	answer=0;
	for (i=0;i<(1<<24);i++) answer^=(n-dy[i])*(n-dy[i]);
	printf("%d\n", answer);
	return false;
}
