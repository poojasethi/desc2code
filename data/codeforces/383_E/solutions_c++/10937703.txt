#include<cstdio>
int n,f[1<<24],a;
char s[5];
int main()
{
	scanf("%d",&n);
	for(int i=n;i--;)
	{
		scanf("%s",s);
		int t=0;
		for(int j=0;j<3;j++)t|=1<<s[j]-'a';
		f[t]++;
	}
	for(int j=0;j<24;j++)	
		for(int i=0;i<1<<24;i++)
			if(i&1<<j)f[i]+=f[i^1<<j];
	for(int i=0;i<1<<24;i++)
		a^=(n-f[i])*(n-f[i]);
	printf("%d\n",a);
	return 0;
}
