#include <bits/stdc++.h>
//Damned if I don't take what I want.

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 1 << 24;
ll dp[MAXN];

int main(){
	ios::sync_with_stdio( false );
	cout.tie(0); cin.tie(0);

	int n; cin >> n;
	for (int i = 0; i < n; i++){
		string s; cin >> s;

		ll cur = 0;
		for (int j = 0; j < s.size(); j++)
			cur |= 1 << (s[j] - 'a');

		dp[ cur ]++;
	}


	for (int i = 0; i < 24; i++)
		for (int j = 0; j < MAXN; j++)
			if (j >> i & 1) dp[j] += dp[j - (1 << i)];
	
	ll ans = 0;
	for (int i = 0; i < MAXN; i++)
		ans ^= (n - dp[i]) * (n - dp[i]);

	cout << ans << endl;
}
