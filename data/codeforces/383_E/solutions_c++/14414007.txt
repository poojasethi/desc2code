#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>

using namespace std;

int n, f[1<<24];
char st[4];

int main() {
#ifndef ONLINE_JUDGE
	freopen("vowel.in","r",stdin);
	freopen("vowel.out","w",stdout);
#endif
	scanf("%d", &n);
	for (int i=1;i<=n;i++) {
		scanf("%s", st);
		int x = 1<<24; x--;
		for (int j=0;j<3;j++)
			x &= ((1<<24)-1)-(1<<(st[j]-'a'));
		f[x] ++;
	}
	for (int j=0;j<24;j++)
		for (int i=(1<<24)-1;i>=0;i--)
		 {
			if ((i&(1<<j))==0) {
				int x = i|(1<<j);
				f[i] += f[x];
			}
		}
	int ans = 0;
	for (int i=0;i<1<<24;i++) ans ^= (n - f[i]) * (n - f[i]);
	printf("%d\n", ans);
	return 0;
}
