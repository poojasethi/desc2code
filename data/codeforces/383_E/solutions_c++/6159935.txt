#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <ctime>
#include <string>
using namespace std;

#define CN 24
int bit[CN + 1], cnt[1 << CN];

main() {
//	freopen("in.txt", "r", stdin);
//	freopen("out.txt", "w", stdout);

	int n, i, j, x, rlt;
	char s[5];
	
	bit[0] = 1;
	for (i = 1; i <= CN; i++) bit[i] = bit[i - 1] * 2;
	
	scanf("%d", &n);
	for (i = 0; i < n; i++) {
		scanf("%s", s);
		for (x = j = 0; s[j]; j++) x |= bit[s[j] - 'a'];
		cnt[x]++;
	}
	rlt = 0;
	for (j = 0; j < CN; j++)
		for (i = 0; i < bit[CN]; i++)
			if (i & bit[j]) cnt[i] += cnt[i - bit[j]];
	for (i = 0; i < bit[CN]; i++) rlt ^= (n - cnt[i]) * (n - cnt[i]);
	printf("%d\n", rlt);
}
