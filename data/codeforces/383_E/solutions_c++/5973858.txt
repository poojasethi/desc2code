#include <cmath>
#include <cstdio>
#include <string>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <algorithm>
using namespace std;

int N,S,f[1<<24],g[1<<24];

void init()
{
	scanf("%d%d",&N),S=1<<24;
	char s[5];
	for (int i=1,x; i<=N; i++)
	{
		scanf("%s",s),x=0;
		for (int j=0; j<3; j++) x|=1<<(s[j]-97);
		g[x]++;
	}
}

void doit()
{
	for (int i=1; i<S; i++)
	{
		int x=i&-i,y=(i^x)&-(i^x),z=(i^x^y)&-(i^x^y);
		f[i]=g[x^y^z];
		if (x&&y) f[i]+=f[i^x]+f[i^y]-f[i^x^y];
		if (x&&y&&z) f[i]+=f[i^z]-f[i^x^z]-f[i^y^z]+f[i^x^y^z];
	}
	int ans=0;
	for (int i=0; i<S; i++) ans^=(N-f[i])*(N-f[i]);
	printf("%d\n",ans);
}

int main()
{
	init();
	doit();
	return 0;
}
