#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long LL;
const LL MOD = (LL)1e9 + 7;
int n;
LL dp[1 << 24];

int main() {
	//freopen( "a.in", "r", stdin );

	scanf( "%d", &n );
	for ( int i = 1; i <= n; i ++ ) {
		int mask = 0;
		char s[10];
		scanf( "%s", s );
		for ( int j = 0; j < 3; j ++ ) {
			mask |= 1 << ((int)s[j] - 'a');
		}
		dp[mask] ++;
	}

	for ( int i = 0; i < 24; i ++ ) {
		for ( int j = 0; j < (1 << 24); j ++ ) if ( (j >> i) & 1 ) {
			dp[j] += dp[j ^ (1 << i)];
		}
	}

	LL ret = 0;
	for ( int i = 0; i < (1 << 24); i ++ ) {
		ret ^= ( n - dp[i] ) * ( n - dp[i] );
	}

	cout <<ret<<endl;

	return 0;	
}
