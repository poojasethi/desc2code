#include<cstdio>
#define M 24
using namespace std;
int f[1<<M],x,n,ans; char st[4];
int main()
{  
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
		{
			scanf("%s",st); x=0;
			for(int j=0;j<3;j++)
				x|=(1<<(st[j]-'a'));
			f[x]++;
		}
	
	for(int i=0;i<M;i++)
		for(int j=0;j<(1<<M);j++)
			if ((j>>i)&1)
				f[j]+=f[j^(1<<i)];
	
	for(int i=0;i<(1<<M);i++)
		ans^=(n-f[i])*(n-f[i]);
	printf("%d",ans);
	return 0;
}
