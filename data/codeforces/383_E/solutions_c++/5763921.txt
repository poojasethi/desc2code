#include <bits/stdc++.h>
using namespace std;

int dp[1<<24];

int main() {
    int n, S = (1<<24) - 1;
    scanf("%d", &n);
    for (int i = 0; i < n; i++) {
        char s[10];
        scanf("%s", s);
        int ts = 0;
        for (int j = 0; s[j]; j++) {
            ts |= (1 << (s[j] - 'a'));
        }
        dp[ts]++;
    }
    int ans = 0;
    for (int i = 0; i < 24; i++) {
        for (int j = 0; j < (1 << 24); j++) {
            if (j & (1 << i)) dp[j] += dp[j ^ (1 << i)];
        }
    }
    for (int i = 0; i < (1 << 24); i++)
        ans ^= (n - dp[i]) * (n - dp[i]);
    printf("%d\n", ans);
    return 0;
}
