#include <iostream>
using namespace std;
#define REP(i,n) for(int i=0;i<(n);++i)
int a[(1<<24)+10], n;
void solve(int m, int t)
{
	if(t < 0) return ;
	solve(m, t-1); solve(m+(1<<t), t-1);
	REP(i, (1<<t)) a[m+(1<<t)+i] += a[m+i];
}
int main()
{
	ios_base::sync_with_stdio(false);
	while(cin >> n){
		REP(i, n){
			char s[4];
			cin >> s;
			int m = 0;
			REP(j, 3) m |= 1<<(s[j]-'a');
			a[m]++;
		}
		solve(0, 23);
		int ans = 0;
		REP(i, (1<<24)) {
			ans ^= (n-a[i])*(n-a[i]);
		}
		cout << ans << endl;
	}
	return 0;
}
