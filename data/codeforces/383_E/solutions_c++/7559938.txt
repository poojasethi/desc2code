#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int dp[1<<24];
int main()
{
    char s[5];
    int n;
    scanf("%d",&n);
    int tmp=n;
    while(n--)
    {
        int v=0;
        scanf("%s",s);
        for(int i=0; i<3; i++)
            v|=(1<<(s[i]-'a'));
        dp[v]++;
    }
    for(int i=0; i<24; i++)
        for(int j=0; j<(1<<24); j++)
            if(j&(1<<i))
                dp[j]+=dp[j^(1<<i)];
    int ret=0;
    for(int i=0; i<(1<<24); i++)
        ret^=(tmp-dp[i])*(tmp-dp[i]);
    cout<<ret<<endl;
    return 0;
}
