#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int> pie;

#define L first
#define R second
#define MP make_pair
#define PB push_back

int dp[(1 << 24) + 100], dd[(1 << 24) + 100];

int main(){
	ios_base::sync_with_stdio(false);
	int n;
	cin >> n;
	for(int i=1;i<=n;i++){
		int x = (1 << 24) - 1;
		for(int j=1;j<=3;j++){
			char c;
			cin >> c;
			int cur = c - 'a';
			if(x & (1 << cur))
				x ^= (1 << cur);
		}
		dp[x] ++;
	}
	for(int i=0;i<24;i++)
		for(int mask=(1 << 24) - 1;mask >= 0;mask--)
			if(mask & (1 << i))
				dp[mask ^ (1 << i)] += dp[mask];
	int ans = 0;
	for(int i=0;i<(1 << 24);i++){
		ans ^= (n - dp[i]) * (n - dp[i]);
	}
	cout << ans << endl;
}
