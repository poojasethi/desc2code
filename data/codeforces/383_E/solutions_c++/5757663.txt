#include <cstdio>
#include <vector>

using namespace std;

const int CN = 1 << 24;

int n;
int c[CN];
char s[4];
int sol;

inline int f(char q)
{
    return 1 << q - 'a';
}

int main()
{
    scanf("%d", &n);
    while(n--)
    {
        scanf("%s", s);
        ++c[f(s[0])];
        ++c[f(s[1])];
        ++c[f(s[2])];
        --c[f(s[0])|f(s[1])];
        --c[f(s[0])|f(s[2])];
        --c[f(s[1])|f(s[2])];
        ++c[f(s[0])|f(s[1])|f(s[2])];
    }

    for(int i = CN; i; i >>= 1)
        for(int j = 0; j < CN; j += i)
            for(int k = 0, l = i >> 1; k < l; ++k)
                c[j+k+l] += c[j+k];

    for(int i = 0; i < CN; ++i)
        sol ^= c[i] * c[i];
    printf("%d\n", sol);
    return 0;
}
