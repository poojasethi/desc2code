#include <iostream>
#include <cstdio>

using namespace std;

const int maxn = (1 << 24) + 10;
int cnt[maxn], n;
int arr[10010];

int main(){
	scanf("%d", &n);
	for(int i = 0; i < n; i++) {
		char str[10];
		scanf("%s", str);
		int rmask = 0;
		for(int j = 0; j < 3; j++)
			rmask |= 1 << (str[j] - 'a');
		rmask = (1 << 24) - 1 - rmask;
		arr[i] = rmask;
		cnt[rmask]++;
	}
	int ans = 0;
	for(int i = 23; i >= 0; i--) {
		int mask = 1 << 24;
		mask -= 1 + (1 << i);
		for(int j = mask; j >= 0; j = (j - 1) & mask) {
			cnt[j] += cnt[j + (1 << i)];
			if(j == 0) break;
		}
	}
	for(int i = (1 << 24) - 1; i >= 0; i--) ans ^= (n - cnt[i]) * (n - cnt[i]);
	cout << ans << endl;

    return 0;
}
