#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>
using namespace std;

int n, dp[1 << 24];
char s[5];
int main(){
    memset(dp,0,sizeof(dp));
    scanf("%d",&n);
    for (int i = 1; i <= n; i ++){
        scanf("%s",s);
        int word = 0;
        for (int j = 0; j < 3; j ++){
            word |= 1 << (s[j] - 'a');
        }
        dp[word] ++;
    }

    for (int i = 0; i < 24; i ++)
        for (int j = 0; j < 1<<24; j ++)
            if ((j >> i) & 1){
                dp[j] += dp[j - (1 << i)];
            }

    int res = 0;
    for (int i = 0; i < 1<<24; i ++) // calc 1<<24 - i - 1
        res ^= (n - dp[i]) * (n - dp[i]);
    printf("%d\n",res);
//    system("pause");
    return 0;
}
