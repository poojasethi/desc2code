#define FOR(i,a,b) for(int i=(a);i<=(b);i++)
#define FORD(i,a,b) for(int i=(a);i>=(b);i--)
#define MS0(x) memset(x,0,sizeof(x))

typedef long long LL;
using namespace std;

const int N=(1<<24)-1;

int c[N+10],ans,n,t;
char ss[4];

int main(){
    scanf("%d",&n);
    FOR(i,1,n){
        scanf("%s",ss);
        t=0;
        FOR(j,0,2)t|=1<<(ss[j]-'a');

        c[t]++;
    }
    
    FOR(i,0,23)FOR(j,0,N)if(j&1<<i)c[j]+=c[j^1<<i];
    FOR(i,0,N)ans^=(n-c[i])*(n-c[i]);
    cout<<ans;
}
