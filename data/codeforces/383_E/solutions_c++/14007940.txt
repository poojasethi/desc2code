#include<bits/stdc++.h>
using namespace std;
int n,dp[1<<25];
int main()
{
	ios_base::sync_with_stdio(false);
    cin>>n;
	for(int i=0;i<n;i++)
	{
        string s;
		cin>>s;
		int ss=0;
		for(int j=0;j<3;j++)
			ss|=(1<<(s[j]-'a'));
		dp[((1<<24)-1)^ss]++;
	}
	for(int i=0;i<24;i++)
		for(int mask=0;mask<(1<<24);mask++)
			if((mask>>i)&1)
				dp[mask^(1<<i)]+=dp[mask];
	int ans=0;
	for(int mask=0;mask<(1<<24);mask++)
		ans^=((n-dp[mask])*(n-dp[mask]));
	cout<<ans<<endl;
	return 0;
}
