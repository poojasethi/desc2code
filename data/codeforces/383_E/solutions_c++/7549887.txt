#include <iostream>
using namespace std;
int dp[1 << 24];
int main()
{
	ios::sync_with_stdio(false);
	int n;
	cin >> n;
	for (int i = 0; i < n; i++)
	{
		string s;
		cin >> s;
		int mask = (1 << 24) - 1;
		for (int i = 0; i < 3; i++)
			mask &= ~(1 << (s[i] - 'a'));
		dp[mask]++;
	}
	for (int p = 0; p < 24; p++)
		for (int mask = 0; mask < (1 << 24); mask++)
			if ((1 << p) & mask)
				dp[mask ^ (1 << p)] += dp[mask];
	int ans = 0;
	for (int mask = 0; mask < (1 << 24); mask++)
		ans ^= (n - dp[mask]) * (n - dp[mask]);
	cout << ans << endl;
	return 0;
}
