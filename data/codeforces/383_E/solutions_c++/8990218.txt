#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
#include <string>
#include <cstring>
using namespace std;
int f[1<<24],p,i,n,j,ans;
char s[105];
int main()
{
    scanf("%d",&n);
    for (i=1; i<=n; i++)
    {
        scanf("%s",s); p=0;
        for (j=0; j<3; j++) p|=(1<<s[j]-'a');
        f[p]++;
    }
    for (i=0; i<24;i++)
      for (j=0; j<(1<<24); j++) if (j&(1<<i)) f[j]+=f[j-(1<<i)];
    for (j=0; j<(1<<24); j++) ans^=(n-f[j])*(n-f[j]);
    cout<<ans;
    return 0;
}
