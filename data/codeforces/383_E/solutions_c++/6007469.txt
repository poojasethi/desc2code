#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std ;
int n ;
const int maxn = (1 << 24)+100 ;
char s[10] ;
int f[maxn] ;
int main() {
	scanf("%d",&n) ;
	for(int i = 1 ; i <= n ; i++) {
		scanf("%s",s) ;
		int tmp = 0 ;
		for(int j = 0 ; j < 3 ; j++) tmp |= 1<<(s[j]-'a') ;
		f[tmp]++ ;
	}
	int maxs = 1 << 24 ;
	for(int i = 0 ; i < 24 ; i++)
	for(int j = 0 ; j < maxs ; j++) if((1<<i)&j) {
		f[j] += f[j^(1<<i)] ;
	}
	int ans = 0 ;
	for(int i = 0 ; i < maxs ; i++) ans ^= (n-f[i]) * (n-f[i]) ;
	printf("%d\n",ans) ;
	return 0 ;
}
