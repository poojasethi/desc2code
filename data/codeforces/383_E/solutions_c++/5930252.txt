#include<cstdio>
#define rep(i,n) for (int i=0;i<n;++i)
int n,m=24,x,y,z,ans; int f[1<<24],g[1<<24]; char s[5];
int main()
{
	scanf("%d",&n);
	rep(i,n){
		scanf("%s",s),x=0;
		rep(j,3) x|=1<<s[j]-97; ++g[x];
	}
	rep(i,1<<m){
		x=i&-i,y=i^x,y=y&-y,z=i^x^y,z=z&-z;
		if (!y) f[i]=g[i];
		else if (!z) f[i]=f[i^x]+f[i^y]-f[i^x^y]+g[x^y];
		else f[i]=f[i^x]+f[i^y]+f[i^z]-f[i^x^y]-f[i^x^z]-f[i^y^z]+f[i^x^y^z]+g[x^y^z];
	}
	rep(i,1<<m) ans^=(n-f[i])*(n-f[i]);
	printf("%d\n",ans);
	return 0;
}
