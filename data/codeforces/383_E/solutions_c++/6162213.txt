#include <iostream>
#include <cstdio>
#include <cstring>
#include <map>
#include <algorithm>
#include <queue>
#include <cmath>
using namespace std;
int n;
int dp[1<<24+1];
int main()
{
	ios::sync_with_stdio(0);
	cin>>n;
	for(int i=1;i<=n;i++)
	{
		string a;cin>>a;
		int num=0;
		for(int j=0;j<3;j++) num|=(1<<(a[j]-'a'));
		dp[num]++;
	}
	for(int i=0;i<24;i++)
	{
		for(int j=0;j<(1<<24);j++)
		{
			if((j&(1<<i))) dp[j]+=dp[j-(1<<i)];
		}
	}
	int ans=0;
	for(int i=0;i<(1<<24);i++)
		ans^=(n-dp[i])*(n-dp[i]);
	cout<<ans<<endl;
}
