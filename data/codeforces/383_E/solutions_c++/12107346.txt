#include <iostream>
#include <vector>

using namespace std;

const int maxlg = 24;
const int maxl = (1 << maxlg);

int dp [ maxl ];
int n;

int ans;

void print ( int x, int k = 0 ){
	if ( k == maxlg )
		return;

	print ( x >> 1, k + 1 );

	cout << ( x & 1 );
}

int main (){
	cin >> n;
	for ( int i = 0 ; i < n ; i++ ){
		int x = 0, tmp = 3;
		char c;

		while ( tmp-- ){
			cin >> c;
			x |= (1 << (c - 'a'));
		}

		dp [ x ]++;
	}

	for ( int j = maxlg ; j > 0 ; j-- )
		for ( int i = 0 ; i < maxl ; i++ )
			if ( i & (1 << (j - 1)) )
				dp [ i ] += dp [ i ^ (1 << (j - 1)) ];

	for ( int i = 0 ; i < maxl ; i++ )
		dp [ i ] = n - dp [ i ];

	for ( int i = 0 ; i < maxl ; i++ )
		ans ^= (int)(dp [ i ]) * (int)(dp [ i ]);

	cout << ans << endl;
}
