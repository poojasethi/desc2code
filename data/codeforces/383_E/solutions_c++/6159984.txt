#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <ctime>
#include <string>

using namespace std;

int N;
int res[1 << 24];
char str[5];
main() {
	
//	freopen("in.txt", "r", stdin);
//	freopen("out.txt", "w", stdout);
	int i, j, k;
	scanf("%d", &N);
	for (i = 0; i < N; i++) {
		scanf("%s", str);
		for (k = j = 0; j < 3; j++) k |= 1 << (str[j] - 'a');
		res[k]++;
	}
	for (j = 0; j < 24; j++) for (i = 0; i < (1 << 24); i++)
	if (i & (1 << j)) res[i] += res[i - (1 << j)];
	int ans = 0;
	for (i = 0; i < (1 << 24); i++) ans ^= (N - res[i]) * (N - res[i]);
	printf("%d\n", ans);
}
