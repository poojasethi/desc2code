#include<cstdlib>
#include<cmath>
#include<cstdio>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;
int n;
int dp[(1<<24)+10];
char tem[5];
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%s",tem);
		int a=0;
		for(int j=0;j<3;j++)
		{
			a|=1<<tem[j]-'a';
		}
		dp[a]++;
	}
	int full=1<<24;
	for(int i=0;i<24;i++)
	{
		for(int j=0;j<full;j++)if(j&(1<<i))dp[j]+=dp[j^(1<<i)];
	}
	int ans=0;
	for(int j=0;j<full;j++)
	{
		ans^=(n-dp[j])*(n-dp[j]);
	}
	printf("%d\n",ans);
	return 0;
}
