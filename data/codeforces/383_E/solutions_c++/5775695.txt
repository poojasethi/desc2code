#include <cstdio>
#include <algorithm>
#include <memory.h>
using namespace std;

int n, ans, i, j, a, b, c;
short f[1 << 24], f1[1 << 24];
char S[5];

int main() {
	scanf("%d", &n);
	while (n--) {
		scanf("%s", S);
		a = S[0] - 'a';
		b = S[1] - 'a';
		c = S[2] - 'a';
		f[1 << a]++; f[1 << b]++; f[1 << c]++;
		f[(1 << a) | (1 << b)]--; f[(1 << b) | (1 << c)]--; f[(1 << a) | (1 << c)]--;
		f[(1 << a) | (1 << b) | (1 << c)]++;
	}

	for (i = 0; i < 24; i++) {
		for (j = 0; j < (1 << 24); j++)
			if (j & (1 << i))	f1[j] = f[j] + f[j - (1 << i)];
			else	f1[j] = f[j];
		memcpy(f, f1, sizeof f1);
	}
	for (i = 0; i < (1 << 24); i++)	ans ^= (int)f[i] * (int)f[i];
	printf("%d\n", ans);
}
