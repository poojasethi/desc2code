#include <cstdio>
#include <iostream>
using namespace std;
int f[1<<24];
int N;
char s[5];
int main() {
	ios::sync_with_stdio(0);
	cin>>N;
	int x;
	for(int i=0;i<N;i++) {
		cin>>s;
		x=0;
		for(int j=0;j<3;j++) x|=1<<(s[j]-'a');
		f[x]++;
	}
	for(int j=0;j<24;j++) {
		for(int i=1;i<1<<24;i++) if(i>>j&1) f[i]+=f[i^1<<j];
	}
	int ans=0;
	for(int i=0;i<(1<<24);i++) ans^=(N-f[i])*(N-f[i]);
	cout<<ans<<endl;
}
