#include <iostream>
#include <cstdio>
#include <cstring>
#include <cmath>
#include <algorithm>

using namespace std;

int n;
int dp[1 << 24];

int main()
{
	scanf("%d", &n);	
	for (int i = 1; i <= n; ++i)
	{
		char s[5];
		scanf("%s", s);
		int x = 0;
		for (int j = 0; j < 3; ++j) x |= (1 << (s[j] - 'a'));
		++dp[x];
	}
	int m = 1 << 24;
	for (int i = 0; i < 24; ++i)
		for (int j = 0; j < m; ++j)
			if (j & (1 << i))
				dp[j] += dp[j ^ (1 << i)];
	int ans = 0;
	for (int i = 0; i < m; ++i)
		ans ^= (n - dp[i]) * (n - dp[i]);
	printf("%d\n", ans);
	
	return 0;
}
