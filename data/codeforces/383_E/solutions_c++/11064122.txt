#include <bits/stdc++.h>
using namespace std;
const int Maxn = (1 << 24) + 9;
int dp[Maxn];
int main(){
    int n;
    cin >> n;
    for(int i = 0; i < n;i++){
        string S;
        cin >> S;
        int x = 0;
        for(int j = 0; j < 3;j++)
            x |= 1 << (S[j] - 'a');
        dp[x]++;
    }
    for(int i = 0; i < 24;i++)
        for(int j = 0; j < (1 << 24);j++)
            if((1 << i) & j)
                dp[j] += dp[j ^ (1 << i)];
    long long ans = 0;
    for(int i = 0; i < (1 << 24);i++)
        ans ^= (n - dp[i]) * (n - dp[i]);
    cout << ans << endl;
}
