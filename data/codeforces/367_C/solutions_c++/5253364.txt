#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#define LL long long
using namespace std;
LL n,m,s[100005],dumb;
int main(){
	scanf("%I64d %I64d",&n,&m);
	for(int i=0;i<m;i++) scanf("%I64d %I64d",&dumb,&s[i]);
	sort(s,s+m);
	LL ans=0;
	for(int i=m-1,j=1;i>=0;i--,j++){
		LL ttl=j*(j/2)+(j%2==1);
		if(ttl>n){printf("%I64d\n",ans);return 0;}
		ans+=s[i];
	}
	printf("%I64d\n",ans);
	return 0;
}
