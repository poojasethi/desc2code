#include<iostream>
#include<algorithm>
using namespace std;
long long maks(long long n){
if(n%2)return n*(n-1)/2+1;
return n*n/2;    
}
main(){
int n,m,mm=0;
long long suma=0;
cin>>n>>m;
int tab[m];
for(int i=0;i<m;i++)cin>>tab[i]>>tab[i];
sort(tab,tab+m);
while(maks(mm)<=n)mm++;
mm--;
if(mm>m)mm=m;
for(int i=0;i<mm;i++)suma+=tab[m-1-i];
cout<<suma<<"\n";
//system("pause");
}
