#include"stdio.h"
#include"algorithm"
using namespace std;
int a[111111],sum[111111];
int main()
{
	int ans,n,m,i,j;
	scanf("%d%d",&n,&m);
	for(i=1;i<=m;i++)
	{
		int d,b;
		scanf("%d%d",&d,&b);
		a[i]=b;
	}
	sort(a+1,a+1+m);
	for(i=m;i>0;i--)
		sum[i]=sum[i+1]+a[i];
	for(i=1;i<=m;i++)
	{
		int d;
		if(i%2==0)d=i*i/2;
		else d=(i*(i-1)/2)+1;
		if(d<=n)ans=i;
		else break;
	}
	ans=sum[m-ans+1];
	printf("%d\n",ans);
}
