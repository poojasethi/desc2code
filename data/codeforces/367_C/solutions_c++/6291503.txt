//In the name of God
#include <algorithm>
#include <iostream>
using namespace std;
const int N = 1e5 + 5;

int n, m, a[N], ans;

int main() {
	cin >> n >> m;
	for (int i = 0, t; i < m; i++)
		cin >> t >> a[i];
	sort(a, a + m, greater<int>());
	for (int i = 0, cur = 0; i < m; i++) {
		cur += a[i];
		long long cost = 1 + 1ll * (i + 1) * i / 2 + (i % 2? (i - 1) / 2: 0);
		if (cost <= n)
			ans = cur;
	}
	cout << ans << '\n';
	return 0;
}
