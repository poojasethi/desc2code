#include<cstdio>
#include<algorithm>

std::pair<int,int> a[100000];

int get(int x)
{
	if(x&1)return x*(x+1)/2+x/2;
	else return x*(x+1)/2;
}

int main()
{
	int i,n,m;
	long long r=0;
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++)scanf("%d%d",&a[i].second,&a[i].first);
	std::sort(a,a+m);
	for(i=0;get(i)<=n-1&&i<m;i++)r+=a[m-i-1].first;
	printf("%I64d",r);
	return 0;
}
