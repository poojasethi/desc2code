#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;

int w[100010];

int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for (int i=0;i<m;i++)
    {
        int q;
        scanf("%d%d",&q,&w[i]);
    }
    sort(w,w+m);
    long long ans=0;
    for (int i=1;i*(i/2)+i%2<=n && i<=m;i++)
        ans+=w[m-i];
    printf("%I64d\n",ans);
    return 0;
}
