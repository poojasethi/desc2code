#include <cstdio>
#include <algorithm>
int N,M;
long long w[100005];
int main()
{
	scanf("%d%d",&N,&M);
	for (int i=1,x; i<=M; i++) scanf("%d%I64d",&x,&w[i]);
	std::sort(w+1,w+M+1);
	for (int i=1; i<=M; i++) w[i]+=w[i-1];
	for (int i=M; i; i--) if (1ll*i/2*i+i%2<=N) {printf("%I64d\n",w[M]-w[M-i]); return 0;}
}
