#include <iostream>
#include <cmath>
#include <iostream>
#include <stdio.h>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int N,M;
long long W[100005];
int main()
{
    cin>>N>>M;
    for (int i=1,x; i<=M; i++) cin>>x>>W[i];
    sort(W+1,W+M+1);
    for (int i=1; i<=M; i++) W[i]+=W[i-1];
    for (int i=M; i; i--) if (1ll*i/2*i+i%2<=N) {cout<<W[M]-W[M-i]; return 0;}
}
