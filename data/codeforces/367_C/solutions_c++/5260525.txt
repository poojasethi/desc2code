#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;
int l[2005],w[100000];
int cmp(int a,int b){return a>b;}
int main()
{
    int n,m,i,j,q;
    for (i=1;i<2005;i++)
        if (i%2) l[i]=l[i-1]+(i+1)/2;
        else l[i]=l[i-1]+i/2+i-2;
    cin>>n>>m;
    for (i=0;i<m;i++) cin>>q>>w[i];
    sort(w,w+m,cmp);
    long long ans=0;
    for (i=0;l[i]<=n;i++);
    for (j=0;j<min(i-1,m);j++) ans+=w[j];
    cout<<ans<<endl;
    return 0;
}
