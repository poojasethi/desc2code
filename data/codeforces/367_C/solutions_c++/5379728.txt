#include<cstdio>
#include<algorithm>

using namespace std ;

const int MAXN = 1000000;

int a[MAXN],sol;
int main(){

    int n ,tmp,x ;

    scanf("%d %d",&x,&n);

    for(int i = 0 ; i < n ; i ++)
        scanf("%d %d",&tmp,&a[i]), a[i] = -a[i];

    sort(a,a+n);

    for(int i = 0 ;  (i+1)/2*(i+1)+(i+1)%2 <= x  and i < n; i ++) sol -= a[i];

    printf("%d\n",sol);
}
