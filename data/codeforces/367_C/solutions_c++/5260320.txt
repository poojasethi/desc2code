#include<cstdio>
#include<algorithm>
using namespace std;
int w[100010];
int cal(int k){
    if(k&1)return k*(k-1)/2+1;
    return k*k/2;
}
int main(){
    int i,j,n,m,k;
    scanf("%d%d",&n,&m);
    for(i=0;i<m;i++){
        scanf("%d%d",&j,&w[i]);
    }
    sort(w,w+m);
    reverse(w,w+m);
    for(i=0;cal(i)<=n;i++);
    k=i-1;
    long long sum=0;
    for(i=0;i<k;i++)
        sum+=w[i];
    printf("%I64d\n",sum);
    return 0;
}
