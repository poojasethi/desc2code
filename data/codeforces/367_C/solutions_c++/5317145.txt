#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);
	long long n,m,i,a[100101],b,s=0,c;
	cin>>n>>m;
	for(i=1;i<=m;i++)
        cin>>b>>a[i];
    sort(a+1,a+m+1);
    reverse(a+1,a+m+1);
    for(i=1;i<=m;i++)
    {
        c=i*(i-1)/2;
        if(i%2==0)c+=i/2;
        else c++;
        if(c<=n)s+=a[i];
    }
    cout<<s<<endl;
}
