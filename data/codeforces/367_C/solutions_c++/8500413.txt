#include <iostream>
#include <algorithm>

using namespace std;

long long n, m, w[100005];

int main()
{
	ios::sync_with_stdio(0);
	
	cin >> n >> m;
	int x;
	for(int i = 0; i < m; i++)
	{
		cin >> x >> w[i];
	}
	sort(w, w + m);

	long long ans = 0;

	long long t = m;
	while(((t*(t - 1))/2) + ((t % 2 == 0) ? ((t/2) - 1) : 0) + 1 > n) t--;

	for(int i = 0; i < t; i++)
	{
		ans += w[m - i - 1];
	}

	cout << ans;
}
