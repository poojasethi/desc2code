#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

int n,m,w[2000001];
long long ans;

int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=m;i++) scanf("%d%d",&w[i],&w[i]);
	sort(w+1,w+1+m);
	for (int i=m,j=1;i;i--,j++)
	{
		long long tmp=(long long)j*(j-1)/2+1;
		if ((j&1)==0) tmp+=(j-1)-j/2;
		if (tmp>n) break;
		ans+=w[i];
	}
	printf("%I64d\n",ans);
}
