def comb(n):
	if(n % 2 == 1):
		return (n*(n-1))/2 + 1
	return (n*(n-1))/2 +(n/2-1) + 1

[n,m] = [int(s) for s in raw_input().split()]

values = []
for v in xrange(m):
   values.append(int(raw_input().split()[1]))
values.sort(reverse=True)

diff = 1
while(comb(diff) <= n):
   diff += 1
diff -= 1

print sum(values[:diff])
