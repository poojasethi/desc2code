import itertools

n, m = map(int, raw_input().split())
Q = [0]*m
W = [0]*m
for i in range(m):
  Q[i], W[i] = map(int, raw_input().split())

k = 0
while True:
  k += 1
  l = 0
  if k%2 == 1:
    l = k*(k-1)/2+1
  else:
    l = k*k/2
  if l > n:
    break
k -= 1
W.sort()
W.reverse()
if k > m:
  print sum(W)
else:
  print sum(W[:k])
