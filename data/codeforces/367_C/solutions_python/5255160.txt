from sys import stdin
def main():
    n, m = map(int, stdin.readline().split())
    a = [map(int, stdin.readline().split())[1] for i in xrange(m)]
    a.sort(reverse=True)
    k = [1, 2]
    for i in xrange(3, m+2):
        if (i - 1) % 2 == 0:
            k.append(i * (i-1) / 2 + 1)
        else:
            k.append(i * i / 2)
    for i, x in enumerate(k):
        if x > n: break
    print sum(a[:i])
main()
