#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <string>
#include <vector>
#include <iostream>

#define min(a,b)	((a)<(b)?(a):(b))
#define max(a,b)	((a)>(b)?(a):(b))
#define abs(a)	((a)<0?-(a):(a))
#define inf 214748364
#define pi 3.141592653589793
#define maxm 3001
#define maxq 100001
#define maxn 100
using namespace std;
typedef long long ll;
int dirx[4]={-1,0,0,1};
int diry[4]={0,-1,1,0};
int qx[maxq],head,tail,qy[maxq];
int ena[maxq],ech[maxq];
int stx,sty,enx,eny;
int res;
char ress[maxm],s[maxm];
int n,m,k;
char g[maxn][maxn];
int ans[maxm];
int f[maxn][maxn];
int now;
int vis[maxn][maxn];
inline void work()
{
	now++;
	g[stx][sty]=ech[1];
	qx[head=tail=0]=enx,qy[0]=eny;
	memset(f,63,sizeof(f));
	f[enx][eny]=0;
	while(head<=tail)
	{
		int x=qx[head],y=qy[head++];
		for(int i=0;i<4;++i)
		{
			int xx=x+dirx[i],yy=y+diry[i];
			if(xx<0||yy<0||xx>=n||yy>=m||!ena[g[xx][yy]])
				continue;
			if(f[xx][yy]>f[x][y]+1)
			{
				f[xx][yy]=f[x][y]+1;
				qx[++tail]=xx,qy[tail]=yy;
			}
		}
	}
	if(f[stx][sty]>res)
		return;
	memset(ans,63,sizeof(ans));
	ans[f[stx][sty]]=ech[1];
	qx[head=tail=0]=stx,qy[0]=sty;
	while(head<=tail)
	{
		int x=qx[head],y=qy[head++];
		if(ans[f[x][y]]!=g[x][y])
			continue;
		for(int i=0;i<4;++i)
		{
			int xx=x+dirx[i],yy=y+diry[i];
			if(xx<0||yy<0||xx>=n||yy>=m||!ena[g[xx][yy]])
				continue;
			if(f[xx][yy]==f[x][y]-1&&vis[xx][yy]!=now)
			{
				ans[f[xx][yy]]=min(ans[f[xx][yy]],g[xx][yy]);
				qx[++tail]=xx,qy[tail]=yy;
				vis[xx][yy]=now;
			}
		}
	}
	int sn=0;
	for(int i=f[stx][sty]-1;i>0;--i)
		s[sn++]=ans[i]+'a';
	s[sn]=0;
	if(res>f[stx][sty]||res==f[stx][sty]&&strcmp(s,ress)==-1)
		res=f[stx][sty],memcpy(ress,s,sizeof(s));
}
inline void dfs(int p,int la)
{
	if(p>k)
	{
		work();
		return;
	}
	for(int i=la;i<26;++i)
	{
		ena[i]=1;
		ech[p]=i;
		dfs(p+1,i+1);
		ena[i]=0;
	}
}
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("a.in","r",stdin);
	freopen("a.out","w",stdout);
	#endif
	scanf("%d%d%d",&n,&m,&k);
	for(int i=0;i<n;++i)
	{
		scanf("%s",g[i]);
		for(int j=0;j<m;++j)
		if(g[i][j]=='S')
			stx=i,sty=j;else
		if(g[i][j]=='T')
			enx=i,eny=j;else
			g[i][j]-='a';
	}
	res=inf;
	dfs(1,0);
	if(res==inf)
		printf("-1\n");else
		printf("%s\n",ress);
	return 0;
}
