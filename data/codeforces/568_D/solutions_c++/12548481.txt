const int N=100005;
struct line{long long a,b,c;}a[N];//={{1,1,2},{1,1,1},{0,1,1}};
inline bool samek(const int &l1,const int &l2){
	return a[l1].a*a[l2].b==a[l2].a*a[l1].b;
}
inline bool inc(const int &l,const int &l1,const int &l2){
	return a[l].a*(a[l1].c*a[l2].b-a[l2].c*a[l1].b)+a[l].b*(a[l1].a*a[l2].c-a[l2].a*a[l1].c)
			==a[l].c*(a[l1].a*a[l2].b-a[l1].b*a[l2].a);
}
int n,k;vector<int> V;
struct point{int l1,l2;} ans[6];int ansn=0;
inline point newpoint(const int &A,const int &B){
	if(B==-1){point p={A,-1};return p;}
	point p={A,B};return p;
}
void print(){
	cout<<"YES\n"<<ansn<<endl;
	for(int i=1;i<=ansn;i++)
		cout<<ans[i].l1<<' '<<ans[i].l2<<endl;
	exit(0);
}
int checkA[]={1,2,7},checkB[]={5,3,11};
#define fd(A,C) for(int Z_=0;Z_<V.size();Z_++)if(V[Z_]==C){A=Z_;break;}
void work(){
	int A,B,i,ct;
	while(clock()<500&&k&&n){
		if(n<=k){
			for(i=0;i<n;i++)ans[++ansn]=newpoint(V[i],-1);print();
		}
		A=rand()%n;B=rand()%n;
		if(samek(V[A],V[B]))continue;
		for(i=ct=0;i<n;i++)if(i==A||i==B||inc(V[i],V[A],V[B])){ct++;}
		if(k*ct>=n){
			ans[++ansn]=newpoint(V[A],V[B]);
			for(i=n-1;i>=0;i--)if(V[i]==ans[ansn].l1||V[i]==ans[ansn].l2||
				inc(V[i],ans[ansn].l1,ans[ansn].l2)){swap(V[i],V[V.size()-1]);V.pop_back();}
			n-=ct;k--;continue;
		}
		//exit(0);
	}
	if(!n)print();
}
void init(){

}
