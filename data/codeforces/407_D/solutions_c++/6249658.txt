#include<cstdio>
#include<set>
#include<cstring>
#include<algorithm>
using namespace std;
int dp_now[410][410],dp_pre[410][410],a[410][410],last_ap[410][160010];
int main(){
	int n,m,l,i,j;
	int ans=0;
	memset(last_ap,-1,sizeof(last_ap));
	memset(dp_pre,-1,sizeof(dp_pre));
	scanf("%d%d",&n,&m);
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			scanf("%d",&a[i][j]);
		}
	}
	for(i=0;i<n;i++){
		for(l=0;l<m;l++){
			for(j=0;j+l<m;j++){
				if(l){
					last_ap[j][ a[i][j] ]=i;
					last_ap[j+l][ a[i][j+l] ]=i;
					dp_now[l][j]=max(dp_pre[l][j],
									max( max(dp_now[l-1][j],dp_now[l-1][j+1]),
										 max(last_ap[j][a[i][j+l]],last_ap[j+l][a[i][j]]) 
										) 
									);
				}
				else{
					dp_now[l][j]=max(dp_pre[l][j],last_ap[j][a[i][j]]);
					last_ap[j][a[i][j]]=i;
				}
				ans=max(ans,(l+1)*(i-dp_now[l][j]));
			}
		}
		for(l=0;l<m;l++){
			for(j=0;j+l<m;j++){
				dp_pre[l][j]=dp_now[l][j];
			}
		}
	}
	printf("%d\n",ans);
	return 0;
}
