#include<iostream>
#include<cstdio>
using namespace std;
const int MaxN = 400 + 2;
int n, m, s[MaxN][MaxN];
short f[MaxN][MaxN], up[MaxN][MaxN * MaxN];
void init()
{
	scanf("%d%d", &n, &m);
	for(int i = 1; i <= n; ++i)
		for(int j = 1; j <= m; ++j)
			scanf("%d", &s[i][j]);
}
void solve()
{
	int dw, len, l, r, ans = 0;
	short tp;
	for(dw = 1; dw <= n; ++dw)
	{
		for(l = 1; l <= m; ++l)
		{
			f[l][l] = max(f[l][l], up[l][s[dw][l]]);
			ans = max(ans, dw - f[l][l]);
		}
		for(len = 2; len <= m; ++len)
			for(l = 1, r = len; r <= m; ++l, ++r)
			{
				tp = f[l][r];
				tp = max(tp, max(f[l + 1][r], f[l][r - 1]));
				tp = max(tp, max(up[l][s[dw][r]], up[r][s[dw][l]]));
				if(s[dw][l] == s[dw][r]) tp = dw;
				f[l][r] = tp;
				ans = max(ans, (dw - tp) * len);
			}
		for(l = 1; l <= m; ++l)	up[l][s[dw][l]] = dw;
	}
	cout << ans << endl;
}
int main()
{
	init();
	solve();
	return 0;
}
