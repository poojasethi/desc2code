#include <cstdio>
#include <iostream>
#include <cstdlib>
#include <algorithm>
#include <cmath>
#include <vector>
#include <set>
#include <map>
using namespace std;

int n, m, i, j, l, r, ans, a[410][410], f[410][410], h[410][410];
unsigned short g[410][161000];

int main() {
	scanf("%d%d", &n, &m);
	for (i = 1; i <= n; i++)
		for (j = 1; j <= m; j++)	scanf("%d", &a[i][j]);
	for (i = 1; i <= n; i++) {
		for (l = m; l; l--)
			for (r = l; r <= m; r++) {
				if (l == r)	f[l][r] = g[l][a[i][l]];
				else {
					f[l][r] = max(f[l + 1][r], f[l][r - 1]);
					f[l][r] = max(f[l][r], (int) g[r][a[i][l]]);
					f[l][r] = max(f[l][r], (int) g[l][a[i][r]]);
					if (a[i][l] == a[i][r])	f[l][r] = i;
				}
				h[l][r] = max(h[l][r], f[l][r]);
				ans = max(ans, (r - l + 1) * (i - h[l][r]));
			}
		for (j = 1; j <= m; j++)	g[j][a[i][j]] = i;
	}
	printf("%d\n", ans);
}
