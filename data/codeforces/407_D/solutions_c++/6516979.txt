#include <iostream>
#include <stdio.h>
#include <string.h>
using namespace std;
#define maxn 405
int a[maxn][maxn];
int last[maxn*maxn][maxn];
int dp[maxn][maxn];

int main()
{
    int n,m,res=0;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
            scanf("%d",&a[i][j]);
    for(int i=1;i<=n;i++)
    {
        for(int j=1;j<=m;j++)
        {
            dp[j][j]=max(dp[j][j],last[a[i][j]][j]);
            last[a[i][j]][j]=i;
        }
        for(int l=1;l<=m;l++)
            for(int r=l+1;r<=m;r++)
                dp[l][r]=max(dp[l][r],max(last[a[i][l]][r],last[a[i][r]][l]));
        for(int l=m;l>=1;l--)
            for(int r=l+1;r<=m;r++)
                dp[l][r]=max(dp[l][r],max(dp[l][r-1],dp[l+1][r]));
        for(int l=1;l<=m;l++)
            for(int r=l;r<=m;r++)
                res = max(res,(r-l+1)*(i-dp[l][r]));
    }
    printf("%d\n",res);
    return 0;
}
