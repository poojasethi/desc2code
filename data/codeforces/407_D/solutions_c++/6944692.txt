#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int n,m;
int a[405][405],cnt[160010] = {0},fail = 0,ans = 0;
int f[405][405],up[160010][405];

int main(){
	scanf("%d%d",&n,&m);
	for(int i = 1;i <= n;i ++)
		for(int j = 1;j <= m;j ++)
			scanf("%d",&a[i][j]);
	for(int i = 1;i <= n;i ++){
		for(int j = 1;j <= m;j ++){
			f[j][j] = max(f[j][j],up[a[i][j]][j]);
			up[a[i][j]][j] = i;
		}
		for(int l = 1;l <= m;l ++)
			for(int r = l + 1;r <= m;r ++)
				f[l][r] = max(f[l][r],max(up[a[i][l]][r],up[a[i][r]][l]));
		for(int j = 2;j <= m;j ++)
			for(int k = 1;k + j - 1 <= m;k ++){
				int r = k + j - 1;
				f[k][r] = max(f[k][r],max(f[k + 1][r],f[k][r - 1]));
			}
		for(int j = 1;j <= m;j ++)
			for(int k = j;k <= m;k ++)
				ans = max(ans,(k - j + 1) * (i - f[j][k]));
	}
	printf("%d\n",ans);
	return 0;
}
