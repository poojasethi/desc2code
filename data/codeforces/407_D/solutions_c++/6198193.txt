
#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <algorithm>
using namespace std;

short f[410][410][410];
int tn,n,m;
int ans=0;
int a[410][410];
int hl[192000];
short ll[192000];
int hr[192000];
short rr[192000];

int main() {
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;i++)
		for (int j=1;j<=m;j++)
			scanf("%d",a[i]+j);
	for (int l=0;l<m;l++)
		for (int i=1;i+l<=m;i++) {
			int j=i+l;
			tn++;
			for (int k=1;k<=n;k++) {
				f[i][j][k]=k;
				if (l!=0) f[i][j][k]=min(f[i][j-1][k],f[i+1][j][k]);
				if (k!=1) f[i][j][k]=min((int)f[i][j][k],f[i][j][k-1]+1);
				if (hl[a[k][j]]==tn) f[i][j][k]=min((int)f[i][j][k],k-ll[a[k][j]]);
				if (hr[a[k][i]]==tn) f[i][j][k]=min((int)f[i][j][k],k-rr[a[k][i]]);
				if (l && a[k][i]==a[k][j]) f[i][j][k]=0;
				hl[a[k][i]]=tn;
				ll[a[k][i]]=k;
				hr[a[k][j]]=tn;
				rr[a[k][j]]=k;
				if ((int)f[i][j][k]*(l+1)>ans)
					ans=(int)f[i][j][k]*(l+1);
			}
		}
	printf("%d\n",ans);
	return 0;
}
