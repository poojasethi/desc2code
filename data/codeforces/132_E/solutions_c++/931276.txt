#include <cstdio>
#include <iostream>
#include <string>
#include <map>
#include <vector>
#include <sstream>
#include <cstring>
#include <algorithm>
using namespace std;

const int N=505;

int e,n,m,cost,s,t,ss;
int a[N],b[N],h[N],v[N],f[N],pre[N],fir[N],u[N];
int q[N*N*5],ne[N*N],po[N*N],w[N*N],c[N*N];

void link(int x,int y,int z,int cc)
{
    ++e;ne[e]=h[x];po[e]=y;w[e]=z;c[e]=cc;h[x]=e;
    ++e;ne[e]=h[y];po[e]=x;w[e]=0;c[e]=-cc;h[y]=e;
}

void imp()
{
    for (int i=t;i!=s;i=pre[i])
    {
        --w[fir[i]],++w[fir[i]^1];
//      cout<<i<<endl;
    }
//  cout<<endl;
    cost+=f[t];
}

bool spfa()
{
    memset(f,61,sizeof f);
    q[0]=s;f[s]=0;v[s]=1;
    for (int st=0,ed=1,i;i=q[st],st<ed;++st)
    {
        for (int j=h[i],k;k=po[j],j;j=ne[j]) if (w[j])
            if (f[i]+c[j]<f[k])
            {
                f[k]=f[i]+c[j];pre[k]=i;fir[k]=j;
                if (!v[k]) {v[k]=1;q[ed++]=k;}
            }
        v[i]=0;
    }
    return f[t]<1000000;
}

int main()
{
//  freopen("input.txt","r",stdin);
//  freopen("output.txt","w",stdout);

    e=1;
    scanf("%d%d",&n,&m);
    for (int i=1;i<=n;++i)
    {
        scanf("%d",&a[i]);
        for (int j=0;j<=30;++j) b[i]+=a[i]>>j&1;
    }
    s=n+n+1;ss=s+1;t=ss+1;
    link(ss,t,m,0);
    for (int i=1;i<=n;++i) link(s,i,1,0),link(i,ss,1,b[i]),link(i+n,t,1,0);
    for (int i=1;i<=n;++i)
        for (int j=1;j<i;++j)
            link(i,j+n,1,(a[i]!=a[j])*b[i]);
    while (spfa()) imp();
    int U=0,o=0;
    for (int i=1;i<=n;++i)
    {
        int k=0;
        for (int j=h[i];j;j=ne[j]) if (!w[j]) {k=po[j];break;}
      //  assert(k);
        if (k==ss) u[i]=++U,++o,v[i]=1; else u[i]=u[k-n],o+=(a[i]!=a[k-n]),v[i]=a[i]!=a[k-n];
    }
    printf("%d %d\n",o+n,cost);
    for (int i=1;i<=n;++i)
    {
        if (v[i])
            printf("%c=%d\n",'a'+u[i]-1,a[i]);
        printf("print(%c)\n",'a'+u[i]-1);
    }

    return 0;
}
