#include<cstdio>
#include<vector>
#include<algorithm>
#define pb push_back
#define fo(i,a,b) for(i=a;i<=b;++i)
#define fd(i,a,b) for(i=a;i>=b;--i)
using namespace std;
const int mn=601,inf=1<<30;
int n,m,s,t,s1,ans,l;
int f[mn][mn],c[mn][mn],a[mn],b[mn],u[mn],pre[mn],dis[mn];
int q[mn+5];
bool vis[mn];
int v[mn];
vector<int>e[mn];
void add(int x,int y,int u,int v)
{
    f[x][y]=u,c[x][y]=v,c[y][x]=-v;
    e[x].pb(y),e[y].pb(x);
}
int spfa()
{
    fill(dis,dis+t+1,inf);
    fill(vis,vis+t+1,0);
    fill(pre,pre+t+1,0);
    int i,j,p,r=0,h=1;
    dis[s]=0,q[r]=s,vis[s]=1;
    while(r!=h)
    {
        i=q[r];
        for(p=0;p<e[i].size();++p)
        {
            j=e[i][p];
            if(!f[i][j])continue;
            if(dis[i]+c[i][j]<dis[j])
            {
                dis[j]=dis[i]+c[i][j],pre[j]=i;
                if(!vis[j])
                    q[h]=j,h=(h+1)%mn,vis[j]=1;
            }
        }
        vis[i]=0;
        r=(r+1)%mn;;
    }
    return dis[t];
} 
void repair()
{
    for(int i=t,j=pre[i];i!=s;i=j,j=pre[i])
        --f[j][i],++f[i][j];
}
int main()
{
    scanf("%d%d",&n,&m);
    int i,j,k,p;
    fo(i,1,n)scanf("%d",a+i);
    fo(i,1,n)
        for(k=a[i];k;k/=2)b[i]+=k&1;
    s1=n+n+1,s=s1+1,t=s+1;
    add(s,s1,m,0);
    fo(i,1,n)add(s1,i,1,b[i]),add(i+n,t,1,0),add(i,i+n,1,-10000);
    fo(i,1,n)
        fo(j,i+1,n)
            add(i+n,j,1,(a[i]!=a[j])*b[j]);
    ans=n*10000,l=n;
    while(spfa()<0)
        ans+=dis[t],repair();
    j=0;
    fo(i,1,n)
        for(p=0;p<e[i].size();++p)
        {
            if((k=e[i][p])<=n)continue;;
            if(!f[k][i])
            {
                if(k==s1)u[i]=++j;
                else u[i]=u[k-n];
                if(k==s1||a[i]!=a[k-n])++l;
                break;
            }
        }
    printf("%d %d\n",l,ans);
    fo(i,1,n)
    {
        if(v[u[i]]!=a[i])printf("%c=%d\n",u[i]+96,v[u[i]]=a[i]);
        printf("print(%c)\n",u[i]+96);
    }
    return 0;
}
