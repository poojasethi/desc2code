#include <iostream>
#include <cstdio>
using namespace std;
long long a[1000010],b[1000010],c[1000010],d[1000010],
	i,j,n,x,y,bo,ans1,ans2,ans3,ans4;
long long find(long long x)
{
	long long l=1,r=n,mid;
	for (;l<=r;)
	{
		mid=(l+r)>>1;
		if (b[mid-1]<x && x<=b[mid]) return mid;
		if (b[mid]<x) l=mid+1;else r=mid-1;
	}
	return 0;
}
int main()
{
		scanf("%I64d\n",&n);
		for (i=1;i<=n;++i) scanf("%I64d",&a[i]);scanf("\n");
		for (i=1;i<=n;++i) scanf("%I64d",&b[i]);scanf("\n");
		for (i=2;i<=n;++i) a[i]+=a[i-1],b[i]+=b[i-1];
		if (a[n]>b[n]) for (bo=1,i=1;i<=n;++i) swap(a[i],b[i]);
		for (i=1;i<n;++i) c[i]=d[i]=-1;
		for (i=1;i<=n;++i)
		{
			y=b[x=find(a[i])]-a[i];
			if (c[y]>=0)
			{
				ans1=c[y];ans2=i;ans3=d[y];ans4=x;
				break;
			}
			c[y]=i;d[y]=x;
		}
		if (bo) swap(ans1,ans3),swap(ans2,ans4);
		printf("%I64d\n",ans2-ans1);
		for (i=ans1+1;i<=ans2;++i) printf("%I64d ",i);
		printf("\n%I64d\n",ans4-ans3);
		for (i=ans3+1;i<=ans4;++i) printf("%I64d ",i);
}
