#include <iostream>
#include <string.h>
#include <math.h>
#include <stdio.h>
#include <stdlib.h>
#include <algorithm>
#include <assert.h>
#define MXN 1000010
using namespace std;
typedef long long ll;
int a[MXN],b[MXN];
int vis[MXN][2];
int main ()
{
//	freopen("F.in","r",stdin);
//	freopen("F.out","w",stdout);
	int k1,k2,k3,k4;
	bool flag=0;
	int n;
	ll suma=0,sumb=0;
	memset(vis,-1,sizeof(vis));
	scanf("%d",&n);
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		suma+=1ll*a[i];
	}
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&b[i]);
		sumb+=1ll*b[i];
	}
	if (suma>sumb)
	{
		flag=1;
		for (int i=1;i<=n;i++)
			swap(a[i],b[i]);
	}
	suma=sumb=0;
	a[0]=b[0]=0;
	for (int i=0,j=0;i<=n;i++)
	{
		suma+=1ll*a[i];
		while (j<n&&sumb+1ll*b[j+1]<=suma)
 			sumb+=1ll*b[++j];
		if (vis[suma-sumb][0]==-1)
		{
			vis[suma-sumb][0]=i;
			vis[suma-sumb][1]=j;
		}
		else
		{
			k1=vis[suma-sumb][0]; k2=i;
			k3=vis[suma-sumb][1]; k4=j;
		}
	}
	if (flag)
		swap(k1,k3),swap(k2,k4);
	printf("%d\n",k2-k1);
	for (int i=k1+1;i<=k2;i++)
		printf("%d ",i);
	printf("\n%d\n",k4-k3);
	for (int i=k3+1;i<=k4;i++)
		printf("%d ",i);
//	fclose(stdin);fclose(stdout);
	return 0;
}
