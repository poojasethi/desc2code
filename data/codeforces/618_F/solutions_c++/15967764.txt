#include <bits/stdc++.h>
using namespace std;
#define M 1001001
typedef long long ll;

int n;
int bf;
ll s[M], t[M];
int x[M], y[M];
int ls, lt, rs, rt;

void input() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++) scanf("%I64d", s + i), s[i] += s[i-1];
	for (int i = 1; i <= n; i++) scanf("%I64d", t + i), t[i] += t[i-1];
	if (s[n] > t[n]) { for (int i = 1; i <= n; i++) swap(s[i], t[i]); bf = 1; }
}

void solve() {
	for (int j = 0, i = 1; i <= n; i++) {
		while (s[i] > t[j]) j++;
		if (s[i] == t[j]) {
			rs = i, rt = j;
			return;
		}
		int d = t[j] - s[i];
		if (x[d]) {
			ls = x[d], rs = i;
			lt = y[d], rt = j;
			return;
		}
        x[d] = i;
        y[d] = j;
	}
}

void print(int l, int r) {
	printf("%d\n", r-l);
	for (; ++l <= r; ) printf("%d%c", l, l == r ? '\n' : ' ');
}
void output() {
	if (bf) swap(ls, lt), swap(rs, rt);
	print(ls, rs);
	print(lt, rt);
}

int main() {
	//freopen("500_F2.in", "r", stdin);
	input();
	solve();
	output();
	return 0;
}
