#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef pair<int,int> P;

const int MAXN = 1000005;

typedef long long LL;

P Ref[MAXN];
LL S[MAXN],S1[MAXN];
int N;

int main()
{
	scanf("%d", &N);
	for(int i = 1;i <= N;i ++) scanf("%I64d", &S[i]),S[i] += S[i - 1];
	for(int i = 1;i <= N;i ++) scanf("%I64d", &S1[i]),S1[i] += S1[i - 1];
	bool f = 0;
	if (S[N] > S1[N]) swap(S,S1),f = 1;
	memset(Ref,255,sizeof Ref);
	for(int i = 0,j = 0;i <= N;i ++)
	{
		for(;j < N && S1[j + 1] <= S[i];j ++);
		if (Ref[S[i] - S1[j]].first != -1)
		{
			int u = Ref[S[i] - S1[j]].first,v = Ref[S[i] - S1[j]].second;
			if (f) swap(i,j),swap(u,v);
			printf("%d ", i - u);
			for(int k = u + 1;k <= i;k ++) printf("%d%c", k, k == i ? '\n' : ' ');
			printf("%d ", j - v);
			for(int k = v + 1;k <= j;k ++) printf("%d%c", k, k == j ? '\n' : ' ');
			break;
		}
		Ref[S[i] - S1[j]] = P(i,j);
	}
	return 0;
}
