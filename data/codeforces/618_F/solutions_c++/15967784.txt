#include <bits/stdc++.h>
using namespace std;
#define M 1000010

int a[M], b[M], n, vis[M], en[M], flag;
__int64 s[M], t[M];

void print(int x, int y, int xx, int yy) {
	if (flag) swap(x, xx), swap(y, yy);
	printf("%d\n", y - x); for (int i = x + 1; i <= y; i++) printf("%d ", i); puts("");
	printf("%d\n", yy - xx); for (int i = xx + 1; i <= yy; i++) printf("%d ", i); puts("");
}

int main() {
//	freopen("F.in", "r", stdin);
    int x;
    scanf("%d", &n);
    for (int i = 1; i <= n; i++) scanf("%d", &x), s[i] = s[i-1] + x, vis[i] = -1;
    for (int i = 1; i <= n; i++) scanf("%d", &x), t[i] = t[i-1] + x;
    if (s[n] > t[n]) {flag = 1; for (int i = 1; i <= n; i++) swap(t[i], s[i]);}
    int I = 1;
    for (int i = 0; i <= n; i++) {
    	while (s[i] > t[I]) I++;
    	if (s[i] == t[I]) {print(0, i, 0, I); break;}
    	x = t[I] - s[i];
    	if (vis[x] >= 0) {print(vis[x], i, en[vis[x]], I); break;}
    	vis[x] = i; en[i] = I;
    }
}
