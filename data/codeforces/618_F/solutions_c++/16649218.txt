int pow(int a,int b){
    int ans=1;
    while(b){
        if(b&1)ans=1LL*ans*a%mod;
        a=1LL*a*a%mod;b>>=1;
    }
    return ans;
}

ll a[2][N];
pii vis[N];
int main(){
    int i,j,k,m,n,T;
    scanf("%d",&n);
    int x1,y1,x2,y2,rev=0,p=0,q=1;
    for(k=0;k<2;k++)
    for(i=1;i<=n;i++)scanf("%d",&j),a[k][i]=a[k][i-1]+j;
    if(a[0][n]<a[1][n])rev=1,swap(p,q);
    for(i=0;i<n;i++)vis[i]=mp(-1,-1);
    vis[0]=mp(0,0);
    for(i=1,j=0;i<=n;i++){
        while(j+1<=n&&a[q][j+1]<=a[p][i])j++;
        k=a[p][i]-a[q][j];
        if(vis[k].x!=-1){
            x1=vis[k].x+1,y1=i;
            x2=vis[k].y+1,y2=j;
            break;
        }
        vis[k]=mp(i,j);
    }
    if(rev)swap(x1,x2),swap(y1,y2);
    printf("%d\n",y1-x1+1);
    for(i=x1;i<=y1;i++)printf("%d ",i);puts("");
    printf("%d\n",y2-x2+1);
    for(i=x2;i<=y2;i++)printf("%d ",i);puts("");
    return 0;
}
