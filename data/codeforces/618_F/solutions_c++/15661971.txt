#include <iostream>
#include <algorithm>
#include <vector>
#include <cstdio>
#include <map>
using namespace std;
vector<int> ma;
vector<int> mb;
int a[1111111];
int b[1111111];
map<long long,pair<int,int> > st;
int main(){
	//freopen("in","r",stdin);
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++){
		scanf("%d", &a[i]);
	}
	for(int j=0;j<n;j++){
		scanf("%d", &b[j]);
	}
	long long sa = 0;
	long long sb = 0;
	int kol = 0;
	int i=0;
	int j=0;
	st[0] = make_pair(0,0);
	int ansi = 0;
	int ansj = 0;
	bool ok = true;
	while(kol<2*n&&ok){
		while(sa<=sb&&i<n&&ok){
			//cout<<sa<<endl;
			sa += a[i];	
			kol++;
			i++;
			if(st.count(sa-sb)>0){
				ansi = i;
				ansj = j;
				ok = false;
			}
			else{
				st[sa-sb] = make_pair(i,j);
			}
		}
		while(sb<=sa&&j<n&&ok){
			//cout<<sb<<endl;
			sb += b[j];
			kol++;
			j++;
			if(st.count(sa-sb)>0){
				ansi = i;
				ansj = j;
				ok = false;
			}
			else{
				st[sa-sb] = make_pair(i,j);
			}
		}
	}
	printf("%d\n", ansi - st[sa-sb].first);
	for(int x = st[sa-sb].first;x<ansi;x++){
		printf("%d ", x + 1);
	}
	printf("\n%d\n", ansj - st[sa-sb].second);
	for(int x = st[sa-sb].second;x<ansj;x++){
		printf("%d ", x + 1);
	}
}
