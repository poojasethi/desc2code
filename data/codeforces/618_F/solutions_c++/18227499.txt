#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

typedef long long int64;
int64 a[1000005],b[1000005];
int n,num;
bool flag;
struct delta {
	int i,j;
	int64 d;
} p[1000005];
inline bool operator <(delta a,delta b) {
	return a.d==b.d?a.i<b.i:a.d<b.d;
}
inline void print(int l1,int r1,int l2,int r2) {
	if (flag) {
		swap(l1,l2); swap(r1,r2);
	}
	printf("%d\n",r1-l1+1);
	for (int i=l1;i<=r1;i++) printf("%d ",i);
	printf("\n");
	printf("%d\n",r2-l2+1);
	for (int i=l2;i<=r2;i++) printf("%d ",i);
	printf("\n");
}
int main() {
	scanf("%d",&n);
	for (int i=1;i<=n;i++) {
		scanf("%d",&a[i]);
		a[i]+=a[i-1];
	}
	for (int i=1;i<=n;i++) {
		scanf("%d",&b[i]);
		b[i]+=b[i-1];
	}
	if (a[n]>b[n]) {
		flag=true;
		for (int i=1;i<=n;i++) swap(a[i],b[i]);
	}
	p[num=1]=(delta){0,0,0};
	for (int i=1,j=0;i<=n;i++) {
		while (j<=n&&b[j]<=a[i]) j++;
		j--;
		p[++num]=(delta){i,j,a[i]-b[j]};
	}
	sort(p+1,p+num+1);
	for (int i=2;i<=num;i++) if (p[i].d==p[i-1].d) {
		print(p[i-1].i+1,p[i].i,p[i-1].j+1,p[i].j);
		return 0;
	}
	return 0;
}
