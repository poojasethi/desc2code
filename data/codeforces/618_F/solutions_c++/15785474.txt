#include<stdio.h>
#include<algorithm>
#include<cstring>
#define N 1000005
using namespace std;
int A[N],B[N],X[N],Y[N],*a=A,*b=B,diff,x,y,pre,i,k,n,flag;
long long SUMA[N],SUMB[N],*suma=SUMA,*sumb=SUMB;
int main(){
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d",&a[i]),suma[i]=suma[i-1]+a[i];
  for (i=1;i<=n;i++)
    scanf("%d",&b[i]),sumb[i]=sumb[i-1]+b[i];
  if (suma[n]>sumb[n]) swap(suma,sumb),swap(a,b),flag=1;
  memset(X,-1,sizeof(X));
  for (i=0;i<=n;i++){
    pre=upper_bound(sumb+1,sumb+n+1,suma[i])-(sumb+1);
    diff=suma[i]-sumb[pre];
    if (X[diff]!=-1){
      x=X[diff];y=Y[diff];
      if (flag) swap(i,pre),swap(x,y),swap(a,b);
      printf("%d\n",i-x);
      for (k=x+1;k<=i;k++)
        printf("%d ",k);puts("");
      printf("%d\n",pre-y);
      for (k=y+1;k<=pre;k++)
        printf("%d ",k);
      return 0;
    }X[diff]=i;Y[diff]=pre;
  }
  //suma[x]-sumb[y]=suma[i]-sumb[pre]
  //suma[i]-suma[x]=sumb[pre]-sumb[y]
}
