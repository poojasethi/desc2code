#include <cmath>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

typedef long long ll;

ll a[1000010],b[1000010];
int d[1000010],f[1000010];

int main()
{
	#ifndef ONLINE_JUDGE
		freopen("input.txt","r",stdin);
		freopen("output.txt","w",stdout);
	#endif
	int n;scanf("%d",&n);
	a[0]=0;for (int i=1;i<=n;i++) {int x;scanf("%d",&x);a[i]=a[i-1]+x;}
	b[0]=0;for (int i=1;i<=n;i++) {int x;scanf("%d",&x);b[i]=b[i-1]+x;}
	bool ok=0;
	if (a[n]>b[n]) {ok=1;for (int i=1;i<=n;i++) swap(a[i],b[i]);}
	for (int i=0;i<n;i++) d[i]=-1;
	int k=0;
	for (int i=0;i<=n;i++)
	{
		while (k<n&&b[k+1]<=a[i]) k++;
		ll x=a[i]-b[k];
		if (d[x]!=-1)
		{
			if (!ok)
			{
				printf("%d\n",i-d[x]);
				for (int j=d[x]+1;j<i;j++) printf("%d ",j);printf("%d\n",i);
				printf("%d\n",k-f[x]);
				for (int j=f[x]+1;j<k;j++) printf("%d ",j);printf("%d\n",k);
			}
			else
			{
				printf("%d\n",k-f[x]);
				for (int j=f[x]+1;j<k;j++) printf("%d ",j);printf("%d\n",k);
				printf("%d\n",i-d[x]);
				for (int j=d[x]+1;j<i;j++) printf("%d ",j);printf("%d\n",i);
			}
			return 0;
		}
		d[x]=i;f[x]=k;
	}
}
