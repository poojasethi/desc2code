#include <bits/stdc++.h>
#define fi first
#define se second
using namespace std;

typedef long long lint;
typedef pair < int , int > ii;

const int maxn = 1e6 + 5;

int n;
lint a[maxn],b[maxn];
ii diff[maxn];
bool used[maxn];

void print( int l , int r ){
	printf("%d\n",r - l);
	for( int i = l + 1 ; i <= r ; i++ ){
		if( i != l + 1 ) printf(" ");
		printf("%d",i);
	}
	puts("");
}

int main(){
	
	scanf("%d",&n);
	
	for( int i = 1 ; i <= n ; i++ ){
		scanf("%lld",&a[i]);
		a[i] += a[i - 1];
	}
	
	for( int i = 1 ; i <= n ; i++ ){
		scanf("%lld",&b[i]);
		b[i] += b[i - 1];
	}
	bool swapped = 0;
	if( a[n] > b[n] ){
		swapped = 1;
		swap(a , b);
	}
	
	diff[0] = ii(0 , 0);
	used[0] = 1;
	for( int i = 1 , j = 0 ; i <= n ; i++ ){
		while( j < n and b[j + 1] <= a[i] ) j++;
		lint f = a[i] - b[j];
		if( used[f] ){
			if( swapped ){
				print(diff[f].se , j);
				print(diff[f].fi , i);
			} else {
				print(diff[f].fi , i);
				print(diff[f].se , j);
			}
			return 0;
		}
		used[f] = 1;
		diff[f] = ii(i , j);
	}
	
	return 0;
}
