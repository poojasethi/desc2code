#include<bits/stdc++.h>
using namespace std;
const int N=1000011;
int n;
int a[N],b[N];
map<long long,pair<int,int> > hash;
int main()
{
	cin>>n;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
	}
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&b[i]);
	}
	long long sa=0ll,sb=0ll;
	int ar=0,br=0;
	bool flag=true,e;
	pair<int,int> ans;
	hash[0]=make_pair(0,0);
	while (flag&&(ar<=n||br<=n))
	{
		e=false;
		while (sa<=sb&&ar<n&&flag)
		{
			e=true;
			ar++;
			sa+=a[ar];
			if (hash.count(sa-sb)>0)
			{
				ans=hash[sa-sb];
				flag=false;
			}
			else
			{
				hash[sa-sb]=make_pair(ar,br);
			}
		}
		while (sb<=sa&&br<n&&flag)
		{
			e=true;
			br++;
			sb+=b[br];
			if (hash.count(sa-sb)>0)
			{
				ans=hash[sa-sb];
				flag=false;
			}
			else
			{
				hash[sa-sb]=make_pair(ar,br);
			}
		}
		if (e==false) break;
	}
	//cout<<ar<<" "<<br<<endl;
	if (flag==true)
	{
		cout<<"-1";
		return 0;
	}
	cout<<ar-ans.first<<endl;
	for (int i=ans.first+1;i<=ar;i++)
	{
		printf("%d ",i);
	}
	puts("");
	cout<<br-ans.second<<endl;
	for (int i=ans.second+1;i<=br;i++)
	{
		printf("%d ",i);
	}
	
	
}
