#include <bits/stdc++.h>
#define f first
#define s second
using namespace std;

int n, a[1000021], b[1000021];
long long s1, s2, sa, sb;
map<int, pair<int, int> > mp;
bool f;
vector<int> aa, bb;

int main()
{
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cin >> n;
    for(int i = 0; i < n; ++i)
    {
        cin >> a[i];
        sa += a[i];
    }
    for(int j = 0; j < n; ++j)
    {
        cin >> b[j];
        sb += b[j];
    }
    if(sa > sb)
    {
        for(int i = 0; i < n; ++i)
            swap(a[i], b[i]);
        f = true;
    }
    mp[0] = make_pair(-1, -1);
    int j = -1;
    for(int i = 0; i < n; ++i)
    {
        s1 += a[i];
        while(j+1 < n && s2 + b[j+1] <= s1)
        {
            ++j;
            s2 += b[j];
        }
        int x = s1-s2;
        //cout << '-' << x <<  endl;
        if(mp.count(x)!=0)
        {
            pair<int, int > p = mp[x];
            for(int z = p.f+1; z <= i; ++z)
                if(z>=0)aa.push_back(z);
            for(int z = p.s+1; z <= j; ++z)
                if(z>=0)bb.push_back(z);
            if(f == true)
                swap(aa, bb);
            cout << aa.size() << "\n";
            for(int z = 0; z < aa.size(); ++z)
                cout << aa[z]+1 << ' ';
            cout << "\n";
            cout << bb.size() << "\n";
            for(int z = 0; z < bb.size(); ++z)
                cout << bb[z]+1 << ' ';
            return 0;
        }
        mp[x] = make_pair(i, j);
    }
    cout << -1;
}
