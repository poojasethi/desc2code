#include<bits/stdc++.h>

using namespace std;

#define pii pair<int,int>
#define ll long long
#define N (int)(1e6+10)
#define mod 1000000007
#define mp make_pair
#define pb push_back
#define nd second
#define st first
#define inf mod
#define endl '\n'
#define sag (sol|1)
#define sol (root<<1)
#define ort ((bas+son)>>1)
#define bit(x,y) ((x>>y)&1)

int i,j,k,n,m,x,y,z;
int a[2][N],w;
ll sum;
pii H[N];

int main(){
	cin >> n;

	for(i=1 ; i<=n ; i++){
		scanf("%d",a[0]+i);
		sum += a[0][i];
	}
	
	for(i=1 ; i<=n ; i++){
		scanf("%d",a[1]+i);
		sum -= a[1][i];
	}

	if(sum<0)
		w=1;

	ll t1=0,t2=0;

	memset(H,-1,sizeof H);

	H[0].st = H[0].nd = 0;

	for(i=1 ; i<=n ; i++){
		t1 += a[w][i];
		while(j <= n and t2 < t1)
			t2 += a[!w][++j];

		if(H[t2-t1].st != -1){
			if(!w){
				cout << i-H[t2-t1].st << endl;
				for(k=H[t2-t1].st+1 ; k<=i ; k++)
					printf("%d ",k);
				puts("");
			}
			cout << j-H[t2-t1].nd << endl;
			for(k=H[t2-t1].nd+1 ; k<=j ; k++)
				printf("%d ",k);
			puts("");
			if(w){
				cout << i-H[t2-t1].st << endl;
				for(k=H[t2-t1].st+1 ; k<=i ; k++)
					printf("%d ",k);
				puts("");
			}
			
			return 0;
		}
		H[t2-t1].st = i;
		H[t2-t1].nd = j;
	}
}
