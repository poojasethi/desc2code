#include <cstdio>
#include <iostream>
#include <cstring>
#include <algorithm>
#include <cmath>
#include <vector>
#include <queue>
using namespace std;
typedef long long ll;
const int N=1E6+5;
int n,x,t[N],o[N],flag;
ll a[N],b[N];

void print(int l,int r)
{
	cout<<r-l+1<<endl;
	for (int i=l;i<=r;i++) printf("%d ",i);
	cout<<endl;
}

int main()
{
	cin>>n; 
	for (int i=1;i<=n;i++) scanf("%d",&x),a[i]=a[i-1]+x;
	for (int i=1;i<=n;i++) scanf("%d",&x),b[i]=b[i-1]+x;
	if (a[n]>b[n]) flag=1,swap(a,b);
	for (int i=1;i<=n;i++)
	{
		t[i]=lower_bound(b+1,b+n+1,a[i]+1)-b-1;
		x=a[i]-b[t[i]];
		if (!o[x]&&x) o[x]=i; else
		{
			if (!flag) print(o[x]+1,i);
			print(t[o[x]]+1,t[i]);
			if (flag) print(o[x]+1,i);
			break;
		}
	}
}
