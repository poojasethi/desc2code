#include<cstdio>
const int N=1000005;
long long n,j,_i,_j,a[N],b[N],s1[N],s2[N],pos[N],bo[N];
int main(){
	scanf("%I64d",&n);
	for (int i=1;i<=n;i++) scanf("%I64d",&a[i]),s1[i]=s1[i-1]+a[i];
	for (int i=1;i<=n;i++) scanf("%I64d",&b[i]),s2[i]=s2[i-1]+b[i];
	j=0;
	for (int i=0;i<=n;i++){
		while (s2[j]<s1[i] && j<=n) j++;
		if (bo[s2[j]-s1[i]]){
			_i=bo[s2[j]-s1[i]]-1;
			_j=pos[_i];
			printf("%I64d\n",i-_i);
			for (int k=_i+1;k<=i;k++) printf("%I64d ",k);
			puts("");
			printf("%I64d\n",j-_j);
			for (int k=_j+1;k<=j;k++) printf("%I64d ",k);
			return 0;
		}
		pos[i]=j;
		bo[s2[j]-s1[i]]=i+1;
	}
	return 0;
}
