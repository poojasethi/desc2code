#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>

using namespace std;
typedef pair<int,int> ii;
int main(){
	long long n;
	cin>>n;
	vector<long long> sa,sb;
	bool flag=false;
	sa.assign(n+10,0);
	sb.assign(n+10,0);
	for(long long i=1;i<=n;i++){
		long long x;
		scanf("%I64d",&x);
		sa[i]=sa[i-1]+x;
	}
	for(long long i=1;i<=n;i++){
		long long x;
		scanf("%I64d",&x);
		sb[i]=sb[i-1]+x;
	}
	if(sa[n]>sb[n]){
		swap(sa,sb);
		flag=true;
	}
	vector<ii> ass;
	ass.assign(n+10,ii(-1,-1));
	int j=0;
	int a1,a2,b1,b2;
	for(int i=0;i<=n;i++){
		while(j<=n&&sb[j]<=sa[i]) j++;
		j--;
		long long diff=sa[i]-sb[j];
		if(ass[diff].first==-1){
			ass[diff]=ii(i,j);
		}else{
			a1=ass[diff].first;
			b1=ass[diff].second;
			a2=i;
			b2=j;
			break;
		}
	}
	if(flag){
		swap(a1,b1);
		swap(a2,b2);
	}
	cout<<a2-a1<<endl;
	for(int i=a1+1;i<=a2;i++){
		cout<<i<<" ";
	}
	cout<<endl;
	cout<<b2-b1<<endl;
	for(int i=b1+1;i<=b2;i++){
		cout<<i<<" ";
	}
	cout<<endl;

	return 0;
}
