#include <cstdio>
#include <algorithm>

using namespace std;

typedef long long LL;
typedef pair <int, int> PII;

#define X first
#define Y second
#define N 1000005
#define rep(a, b) for (int i = a; i <= b; i ++)

int n;
LL a[N], b[N];
int p[N], pos[N];

pair <PII, PII> solve(LL a[], LL b[]) {
	rep(1, n) pos[i] = -1;
	for (int i = 1, j = 0; i <= n; i ++) {
		for ( ; b[j] < a[i]; j ++);
		p[i] = j;
		int tmp = b[j] - a[i];
		if (pos[tmp] < 0) pos[tmp] = i;
		else {
			int pr = pos[tmp];
			return make_pair(PII(pr + 1, i), PII(p[pr] + 1, p[i]));
		}
	}
}

main() {
	scanf("%d", &n);
	for (int i = 1, x; i <= n; i ++) {
		scanf("%d", &x);
		a[i] = a[i - 1] + x;
	}
	for (int i = 1, x; i <= n; i ++) {
		scanf("%d", &x);
		b[i] = b[i - 1] + x;
	}
	pair <PII, PII> R;
	if (a[n] <= b[n]) R = solve(a, b);
	else R = solve(b, a), swap(R.X, R.Y);
	printf("%d\n", R.X.Y - R.X.X + 1);
	rep(R.X.X, R.X.Y) printf("%d ", i);
	printf("\n%d\n", R.Y.Y - R.Y.X + 1);
	rep(R.Y.X, R.Y.Y) printf("%d ", i);
}
