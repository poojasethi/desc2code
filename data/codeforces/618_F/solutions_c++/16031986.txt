#include<iostream>
#include<cstdio>
#include<algorithm>
#define maxn 1001000
using namespace std;
int n;
long long a[maxn];
long long b[maxn];
int wh=0;
bool flag;
struct node
{
	int posa,posb;
	long long val;
}p[maxn];
//int wh;
bool cmp(node a,node b)
{
	return a.val<b.val;
}
int abss(int x)
{
	if(x>0)return x;
	return -x;
}
void print(node a,node b)
{
	if(a.posa>b.posa)swap(a,b);
	if(flag)
	{
	    printf("%d\n",abss(a.posb-b.posb));
		for(int i=min(a.posb,b.posb)+1;i<=max(b.posb,a.posb);i++)printf("%d ",i);
	    printf("\n%d\n",b.posa-a.posa);
	    for(int i=a.posa+1;i<=b.posa;i++)printf("%d ",i);
	    puts("");
	    return;
	}
	//cout<<a.posa<<" "<<a.posb<<" "<<b.posa<<" "<<b.posb<<endl;
	printf("%d\n",b.posa-a.posa);
	for(int i=a.posa+1;i<=b.posa;i++)printf("%d ",i);
	printf("\n%d\n",abss(a.posb-b.posb));
	for(int i=min(a.posb,b.posb)+1;i<=max(b.posb,a.posb);i++)printf("%d ",i);
	puts("");
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%I64d",&a[i]);
		a[i]=a[i-1]+a[i];
	}
	for(int i=1;i<=n;i++)
	{
		scanf("%I64d",&b[i]);
		b[i]=b[i-1]+b[i];
	}
	if(a[n]>b[n])
	{
		flag=true;
		for(int i=1;i<=n;i++)
		swap(a[i],b[i]);
	}
	p[0].posa=0;p[0].posb=0;p[0].val=0;
	for(int i=1;i<=n;i++)
	{
		while(a[i]>=b[wh])wh++;
		wh--;
		p[i].posa=i;p[i].posb=wh;p[i].val=a[i]-b[wh];
	}
	sort(p,p+n+1,cmp);
	for(int i=1;i<=n;i++)
	{
		if(p[i].val==p[i-1].val)
		{
			print(p[i-1],p[i]);
			return 0;
		}
	}
	puts("-1");
	return 0;
}
