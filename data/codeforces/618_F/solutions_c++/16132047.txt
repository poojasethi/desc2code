#include<bits/stdc++.h>

#define FI(i,a,b) for(int i=(a);i<=(b);i++)
#define FD(i,a,b) for(int i=(a);i>=(b);i--)

#define LL long long

using namespace std;

int n,s[1000005],t[1000005];
int bef[1000005][2];
LL ss,st,cs,ct;
bool sw;

int main(){
  memset(bef,-1,sizeof(bef));
  bef[0][0]=bef[0][1]=0;
  scanf("%d",&n);
  FI(i,1,n) scanf("%d",&s[i]),ss+=s[i];
  FI(i,1,n) scanf("%d",&t[i]),st+=t[i];
  if(ss<st){
    swap(ss,st);
    FI(i,1,n) swap(s[i],t[i]);
    sw=true;
  }
  //ss>=st
  int p=0;
  FI(i,1,n){
    ct+=t[i];
    while(p<n && cs+s[p+1]<=ct) cs+=s[++p];
    LL d=ct-cs;
    assert(d>=0 && d<n);
    if(bef[d][0]==-1){
      bef[d][0]=p;
      bef[d][1]=i;
    }else{
      if(sw){
        printf("%d\n",i-bef[d][1]);
        FI(j,bef[d][1]+1,i) printf("%d%c",j,j==i?'\n':' ');
      }
      printf("%d\n",p-bef[d][0]);
      FI(j,bef[d][0]+1,p) printf("%d%c",j,j==p?'\n':' ');
      if(!sw){
        printf("%d\n",i-bef[d][1]);
        FI(j,bef[d][1]+1,i) printf("%d%c",j,j==i?'\n':' ');
      }
      return 0;
    }
  }
}
