#include <bits/stdc++.h>

using namespace std;

#define FOR(i,x,y)	for(int i = x;i < y;++ i)
#define IFOR(i,x,y)	for(int i = x;i > y;-- i)
#define pb	push_back
#define mp	make_pair
#define fi	first
#define se	second

typedef long long LL;
typedef pair <int,int>	PII;

const int maxn = 1000010;

int n;
LL a[maxn],b[maxn];

void In(){
	scanf("%d",&n);
	int num;
	a[0] = b[0] = 0;
	FOR(i,1,n+1){
		scanf("%d",&num);
		a[i] = a[i-1]+num;
	}
	FOR(i,1,n+1){
		scanf("%d",&num);
		b[i] = b[i-1]+num;
	}
}

bool vis[maxn];
PII id[maxn];

typedef pair <PII,PII>	ANS;

ANS work(LL* a,LL* b){
	ANS ans;
	FOR(i,0,n+1)	vis[i] = false;
	FOR(i,0,n+1){
		int L = 0,R = n;
		int num = 0;
		while(L < R){
			num ++;
			int mid = (L+R+1)>>1;
			if(a[i] < b[mid])	R = mid-1;
			else	L = mid;
		}
		int diff = a[i] - b[L];
		if(vis[diff]){
			ans = mp(id[diff],mp(i,L));
			break;
		}
		vis[diff] = true;
		id[diff] = mp(i,L);
	}
	return ans;
}

int main(){
	In();
	if(a[n] > b[n]){
		ANS ans = work(b,a);
		printf("%d\n",ans.se.se-ans.fi.se);
		FOR(i,ans.fi.se+1,ans.se.se)	printf("%d ",i);
		printf("%d\n",ans.se.se);
		printf("%d\n",ans.se.fi-ans.fi.fi);
		FOR(i,ans.fi.fi+1,ans.se.fi)	printf("%d ",i);
		printf("%d\n",ans.se.fi);
	}
	else{
		ANS ans = work(a,b);
		printf("%d\n",ans.se.fi-ans.fi.fi);
		FOR(i,ans.fi.fi+1,ans.se.fi)	printf("%d ",i);
		printf("%d\n",ans.se.fi);
		printf("%d\n",ans.se.se-ans.fi.se);
		FOR(i,ans.fi.se+1,ans.se.se)	printf("%d ",i);
		printf("%d\n",ans.se.se);
	}
	return 0;
}
