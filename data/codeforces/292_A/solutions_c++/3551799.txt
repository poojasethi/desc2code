#include<iostream>
using namespace std;
int t[1005],c[1005];
int main(){
    int n,ans;
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>t[i]>>c[i];
    ans=c[0];
    for(int i=1;i<n;i++){
        c[0]=max(c[0]-t[i]+t[i-1],0);
        c[0]+=c[i];
        ans=max(ans,c[0]);
    }
    cout<<c[0]+t[n-1]<<" "<<ans;
}
