#include <stdio.h>
#include <stdlib.h>
#include "string.h"
#include "stdlib.h"
int n,i,t,c,l,r;
int max(int x,int y)
{
    return x>y?x:y;
}
int main()
{
    scanf("%d",&n);
    for (i=0; i<n; i++)
    {
        scanf("%d%d",&t,&c);
        l=max(t,l)+c;
        r=max(l-t,r);
    }
    printf("%d %d\n",l,r);
    return 0;
}
