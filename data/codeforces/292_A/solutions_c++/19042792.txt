#include<cstdio>
int main()
{
	int n,i,x,y;
	scanf("%d",&n);
	int now=0,mx=-1,p=-1;
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x,&y);
		if(p!=-1)
		{
			now-=(x-p);
			if(now<0)
				now=0;
		}
		now+=y;
		if(now>mx)
			mx=now;
		p=x;
	}
	printf("%d %d\n",now+p,mx);
	return 0;
}
