#include<stdio.h>
int main()
{
    int n,t,c;
    int ans1 = 0,ans2 = 0;
    int T = 0, C = 0;
    scanf("%i",&n);
    while(n--)
    {
       scanf("%i %i",&t,&c);
       C -= t-T;
       if(C<0)C = 0;
       C += c;
       if(C>ans2)ans2 = C;
       T = t;
    }
    printf("%i %i",T + C,ans2);
  
    return 0;
}
