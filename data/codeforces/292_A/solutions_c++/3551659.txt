#include <iostream>
using namespace std;
 
int main() {
    int n;
    int t = 0;
    cin >> n;
    int u = 0;
    for (int i = 0; i < n; i++) {
        int a, b;
        cin >> a >> b;
        t = max(a + 1, t + 1);
        t += b - 1;
        u = max(u, t - a);
    }
    cout << t << ' ' << u << endl;
}
