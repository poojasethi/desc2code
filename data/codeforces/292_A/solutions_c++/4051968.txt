#include<iostream>
using namespace std;
main()
{
    int i,j,n;
    cin>>n;
    long int a,max;
    long int t[2],c;
    a=0;
    t[0]=0;
    max=-1;
    for (i=0;i<n;i++)
    {
       cin>>t[1]>>c;
       a=a-t[1]+t[0];
       if(a<0) a=0;
       a=a+c; if(a>max) max=a;
       t[0]=t[1];
    }
    cout<<t[0]+a<<" "<<max;
}
