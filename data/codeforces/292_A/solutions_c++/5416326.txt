#include<iostream>
#include<algorithm>
using namespace std;
main(){
int n,t,ile,ost=0;
long long kolejka=0,makskol=0;
cin>>n;
for(int i=0;i<n;i++){
    cin>>t>>ile;
    kolejka-=t-ost;
    if(kolejka<0)kolejka=0;
    kolejka+=ile;
    if(kolejka>makskol)makskol=kolejka;
    ost=t;
}
cout<<t+kolejka<<" "<<makskol<<"\n";
}
