#include <cstdio>
#include <cstring>

int n,t,x;
int tl,ti,mq;

int main(){
	tl=mq=ti=0;
	scanf("%d",&n);
	scanf("%d%d",&t,&x);
	tl+=t;
	mq=ti=x;
	for (int i=1;i<n;++i){
        scanf("%d%d",&t,&x);
		int d=t-tl;
		ti = ((ti-d)>0)?ti-d:0;
		ti+=x;
		if (ti>mq)
			mq=ti;
		tl=t;
	}
	printf("%d %d",ti+tl,mq);
}
