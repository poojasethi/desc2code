#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>

using namespace std;

int main()
{
	int i,n,now=0,last=0,t,c,mm=0;
	scanf("%d",&n);
	for (i=1; i<=n; i++)
	{
		scanf("%d%d",&t,&c);
		now-=min(now,t-last);
		last=t; 
		now+=c;
		mm=max(mm,now);
	}
	printf("%d %d\n",now+last,mm);
}
