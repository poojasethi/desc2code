#include <iostream>
#include <cmath>

using namespace std;

int main()
{
  int n;
  cin>>n;
  int sol=0;
  int cuantos=0;
  int tiempo=0;
  for (int i=0;i<n;i++) {
    int t,c;
    cin>>t>>c;
    cuantos=max(0,cuantos-(t-tiempo));
    tiempo=t;
    cuantos+=c;
    sol=max(sol,cuantos);
  }
  cout<<tiempo+cuantos<<" "<<sol<<endl;
}
