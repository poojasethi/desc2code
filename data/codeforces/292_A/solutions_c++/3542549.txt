#include <iostream>

using namespace std;

int n, q, t, c, time, maxi;

int main()
{
  cin >> n;
  for (int i = 0, prev = -1; i < n; i++)
  {
    cin >> t >> c;
    if (q) q = max(0, q - t + prev);
    q += c;
    maxi = max(maxi, q);
    time = max(time, t + q);
    prev = t;
  }
  cout << time << " " << maxi;
  return 0;
}
