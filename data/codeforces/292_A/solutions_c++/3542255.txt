#include <cstdio>
#include <algorithm>
using namespace std;

int T, MAX;
int t, c;
int N;

int main(){

    int i;

    scanf("%d", &N);
    for( i = 0; i < N; i++){
        scanf("%d %d", &t, &c);

        if( T < t)T = t;
        T += c;

        MAX = max( MAX, T - t);
    }

    printf("%d %d\n", T, MAX);
}
