
#include <cstdio>
#include <algorithm>
using namespace std;

typedef long long i8;

int n, t, c;

main() {
	scanf("%d", &n);
	int q=0, pt=0, mq=0;
	for (int i=0; i<n; i++) {
		scanf("%d%d",&t,&c);
		int o=t-pt;
		q=max(0,q-o);
		q+=c;
		mq=max(mq,q);
		pt=t;
	}
	pt += q;
	printf("%d %d\n", pt, mq);
}
