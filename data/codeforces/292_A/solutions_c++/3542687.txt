#include<cstdio>
using namespace std;
const int M1 = 1000000;
int main()
{
    int N;
	scanf("%d",&N);
	int at=0,mx=0,lt=0,lc=0;
	for(int i=0;i!=N;++i)
	{   int t,c;
	    scanf("%d%d",&t,&c);
		int tmp = lc-(t-lt)>0?lc-(t-lt)+c:c;
		lc = tmp;
		lt = t;
		if(mx<lc) mx = lc;
	}
	at = lt+lc;
	printf("%d %d\n",at,mx);
    return 0;
}
