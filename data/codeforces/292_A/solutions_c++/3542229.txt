#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n,t,prevt=0,c,i,q=0,max_=0;
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        scanf("%d %d",&t,&c);
        q-=t-prevt;
        q=max(q,0);
        q+=c;
        max_=max(max_,q);
        prevt=t;
    }
    printf("%d %d\n",t+q,max_);
}
