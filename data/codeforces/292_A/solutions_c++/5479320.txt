#include <iostream>
#include <cstring>
#include <cstdio>
#include <cstdlib>

using namespace std;

int n,pret,ls,L;

int main()
{
scanf("%d",&n);
for (int i=1;i<=n;i++)
  {
  int t,x;scanf("%d%d",&t,&x);
  ls=max(ls-(t-pret),0);ls+=x;L=max(L,ls);
  pret=t;
  }
printf("%d %d",pret+ls,L);
}
