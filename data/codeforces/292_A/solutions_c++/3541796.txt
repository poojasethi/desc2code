#include<cstdio>
#include<iostream>
using namespace std;

int main()
{
    int n;
    scanf("%d",&n);
    int time=0,ans=0;
    for (int i=0;i<n;i++)
    {
        int t,c;
        scanf("%d%d",&t,&c);
        ans=max(ans,max(0,time-t)+c);
        time=max(t+c,time+c);
    }
    printf("%d %d\n",time,ans);
    return 0;
}
