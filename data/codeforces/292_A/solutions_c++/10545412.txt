#include <iostream>
using namespace std;
int main()
{
    int n, i, j, r, x, y, t, g;
    cin >> n;
    cin >> r >> t;
    g = t;
    for(i = 1; i < n; i ++)
    {
        cin >> x >> y;
        t -= min(x - r,t);
        t += y;
        r = x;
        g = max(g, t);
    }
    cout << r + t << " " << g;
}
