#include<iostream>
using namespace std;
int main()
{
   int n,i,t,c,prev=0;
   long long int last,maxa=0,current=0;
   cin>>n;
   while(n--)
   {
      cin>>t>>c;
      current=current-(t-prev);
      if(current<0)
      current=0;
      current+=c;
      maxa=max(maxa,current);
      prev=t;
   }
   last=t+current;
   cout<<last<<" "<<maxa;
}
