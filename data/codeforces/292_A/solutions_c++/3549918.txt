#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
int n,pos,ans,cnt;

int main() {
    int i,t,c;
    scanf("%d",&n);
    for (i=1;i<=n;i++) {
        scanf("%d%d",&t,&c);
        cnt=max(cnt-(t-pos),0);
        ans=max(ans,cnt+=c);
        pos=t;
    }
    printf("%d %d\n",pos+cnt,ans);
}
