#include <iostream>

using namespace std;

int main()
{
	int n;
	cin >> n;
	int q = 0;
	int t1;
	cin >> t1 >> q;
	int ans = q;
	for (int i = 1; i < n; i++)
	{
		int t, c;
		cin >> t >> c;
		q = max(q - (t - t1), 0);
		q += c;
		ans = max(ans, q);
		t1 = t;
	}
	cout << t1 + q << " " << ans << endl;
}
