#include <iostream>
#include <algorithm>
#include <vector>
#include <queue>
using namespace std;
const int maxn=1e3+10;
int main()
{
  int n;
  cin>>n;
  int ans=0,cur=0,time=0;
  for(int i=0;i<n;i++){
    int t,c;
    cin>>t>>c;
    cur=max(0,cur-t+time);
    cur+=c;
    ans=max(ans,cur);
    time=t;
  }
  cout<<time+cur<<" "<<ans<<endl;
  return 0;
}
