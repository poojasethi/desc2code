if __name__ == '__main__':
	# w = last time message received
	# x = total time elapsed
	# y = max total in queue
	# z = current total in queue
	w, x, y, z = 0, 0, 0, 0
	for n in range(int(raw_input())):
		t, c = map(int, raw_input().split())

		x += (t - w)
		z = max(0, z-(t-w))
		z += c
		y = max(y, z)
		w = t
	if z > 0:
		x += z
	print x, y
