from sys import stdin

n_tasks = int ( stdin.readline() )
tasks = []
for i_task in range(n_tasks):
	task = t, c = map( int, stdin.readline().split() )
	tasks.append(task)

task_in_time, task_n_sms = tasks[0]
max_n_queue = n_queue = task_n_sms
t_prev = task_in_time

for task in tasks[1:]:
	task_in_time, task_n_sms = task

	n_queue -= task_in_time - t_prev
	if n_queue < 0:
		n_queue = 0

	n_queue += task_n_sms
	t_prev = task_in_time

	max_n_queue = max(max_n_queue, n_queue)

t = t_prev + n_queue

print "%d %d" % (t, max_n_queue)
