from __future__ import print_function
import sys


n = int(sys.stdin.readline())

q = []
for i in xrange(n):
    (t, c) = map(int, sys.stdin.readline().split())
    q.append((t+1, c))

time = 0
num = 0
maxCount = 0
for (t, c) in q:
    if time < t:
        num -= t - time
        if num < 0:
            num = 0
    time = t
    num += c
    if maxCount < num:
        maxCount = num

time += num - 1

print(time, maxCount)
