in1 = lambda : int(raw_input())
in2 = lambda : map(int, raw_input().split())

def main():
	n = in1()
	nt, nc, ans = 1, 0, 0
	for _ in xrange(n):
		t, c = in2()
		if nt + nc - 1 <= t: nc = 0
		else: nc -= t - nt + 1
		nt = t + 1
		nc += c
		ans = max(ans, nc)
	print nt + nc - 1, ans
	return 0

main()
