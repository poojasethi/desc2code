def main():

  n = int(raw_input())
  maxsize = 0
  que = []
  posq = 0
  curtime = 0
  lasttime = 0
  itus = []
  for _ in range(n):
    itus.append(map(int, raw_input().split()))
  itus.append((1020000000000000, 0))
  for t, c in itus:
    while posq < len(que) and curtime < t:
      if not que[posq]:
        posq += 1
        continue
      timeskip = min([t - curtime, que[posq]])
      que[posq] -= timeskip
      curtime += timeskip
    assert curtime <= t
    if posq != len(que):
      assert curtime == t
    lasttime = curtime
    curtime = t
    que.append(c)
    maxsize = max([maxsize, sum(que)])
  print lasttime, maxsize

main()
