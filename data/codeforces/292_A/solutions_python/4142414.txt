I = lambda:map(int , raw_input().split())
n = input()
a = [0] * n
b = [0] * n
cur_t =  max_val =  cur_queue = 0
for i in range(n):
    a[i] , b[i]= I()
    cur_t = max(cur_t , a[i]) + b[i]
    if i: cur_queue = max(0 , cur_queue - (a[i] - a[i - 1]))
    cur_queue += b[i]
    max_val = max(max_val , cur_queue)

print cur_t , max_val
