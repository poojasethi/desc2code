n = input()

curr_time = 0
curr_size = 0
max_size = 0

for i in range(n):
  t,c = map(int, raw_input().split())
  curr_size += c - min(curr_size, max(t - curr_time,0))
  if t > curr_time:
    curr_time = t
  if curr_size > max_size:
    max_size = curr_size

print str(curr_time + curr_size) + ' ' + str(max_size)
