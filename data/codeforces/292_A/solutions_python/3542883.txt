def main():
    n = int(raw_input())
    ts = []
    cs = []
    for i in range(n):
        t, c = map(int, raw_input().split())
        ts.append(t)
        cs.append(c)

    size = 0
    time = 0
    max_size = 0
    for i in range(n):
        size = max(0, size - (ts[i] - time)) + cs[i]
        time = ts[i]
        max_size = max(max_size, size)

    print size + ts[-1], max_size


if __name__ == "__main__":
    main()
