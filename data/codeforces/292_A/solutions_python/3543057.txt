def characteristics(events):
    last_time = None
    current_size = 0
    max_size = 0
    previous_time = 0
    for event in events:
        add_time, add_size = event
        current_size = max(0, current_size - (add_time - previous_time))
        current_size = current_size + add_size
        max_size = max(current_size, max_size)
        last_time = add_time + current_size
        previous_time = add_time
    return last_time, max_size

def main():
    event_count = int(raw_input())
    events = []
    for index in range(event_count):
        events.append(map(int, raw_input().split()))
    last_message, max_size = characteristics(events)
    print last_message, max_size

if __name__ == '__main__':
    main()
