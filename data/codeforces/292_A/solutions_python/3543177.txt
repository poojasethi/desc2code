n = int(raw_input())
max_queue = 0
prev_t = 0
queue = 0
t, c = 0, 0
for i in xrange(n):
    t, c = map(int, raw_input().split())
    queue -= t - prev_t
    queue = queue if queue >=0 else 0
    queue += c
    max_queue = max(max_queue, queue)
    prev_t = t
    #print prev_t, max_queue
    
prev_t += queue
print prev_t, max_queue
