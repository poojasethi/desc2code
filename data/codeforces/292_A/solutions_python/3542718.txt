#!/usr/bin/env python
import sys

if __name__ == '__main__':
    n = int(sys.stdin.readline().strip())
    tasks = []

    for i in xrange(n):
        t, c = map(int, sys.stdin.readline().strip().split())
        tasks.append((t, c))

    tasks.sort()
    
    
    in_queue = tasks[0][1]
    time, max_size = tasks[0][0], tasks[0][1]
    for i in xrange(1, n):
        t, c = tasks[i]
        if t - time >= in_queue:
            in_queue = c
            time, max_size = t, max(max_size, in_queue)
        else:
            in_queue -= (t - time)
            in_queue += c
            time, max_size = t, max(max_size, in_queue)

    time += in_queue
    print time, max_size
