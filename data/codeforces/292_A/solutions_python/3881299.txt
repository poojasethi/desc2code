n=input()
r=[]
for x in range(0,n):
    r.append(map(int,raw_input().split(" ")))
#
q=maxm=r[0][1]
for x in range(1,n):
    a=(q-(r[x][0]-r[x-1][0]))
    if a>=0:
        q = a+r[x][1]
    else:
        q = r[x][1]    
        #queue negative nahin hogi... jahan 0 hogi aur time bacha hoga to 0 hi rahegi tab tak
    maxm=max(maxm,q)
last=r[n-1][0]+q
print last,maxm
