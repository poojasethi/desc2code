n = (int)(raw_input())

tasks = []

i = 0
while i < n:
    t, s = raw_input().split(" ")
    tasks.append(((int)(t), (int)(s)))
    i += 1

tm = 0
que = 0
que_max = 0

for i in tasks:
    dt = i[0] - tm
    if que > dt:
        que -= dt
    else:
        que = 0

    tm = i[0]
    que += i[1]
    if que > que_max:
        que_max = que

print str(tm + que) + " " + (str)(que_max)
