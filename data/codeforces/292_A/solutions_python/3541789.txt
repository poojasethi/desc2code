NAME = "P1"
try:
    inFile = open(NAME+".txt")
except:
    pass

def read():
    try:
        return inFile.readline().strip()
    except:
        return raw_input().strip()


n = int(read())
q_size = 0
max_q = 0
prev = 0
for _ in xrange(n):
    t,c = map(int,read().split())
    q_size = max(0,q_size-(t-prev))
    q_size += c
    max_q = max(max_q,q_size)
    prev = t
print prev + q_size, max_q
