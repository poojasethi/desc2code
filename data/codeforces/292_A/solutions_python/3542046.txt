if __name__ == "__main__":
    n = int(raw_input())
    cur_time = 0
    cur_len = 0
    max_len = 0
    for i in range(n):
        t, c = map(int, raw_input().split())
        cur_len = max(0, cur_len - (t - cur_time))
        cur_len += c
        cur_time = t
        max_len = max(max_len, cur_len)
    print "{0} {1}".format(cur_time + cur_len, max_len)
