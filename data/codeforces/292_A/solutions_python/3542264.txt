import sys
sys.setrecursionlimit(10001)

n = int(sys.stdin.readline().strip())

size = 0
last = 0
max_size = 0
for i in xrange(n):
	t, c = map(int, sys.stdin.readline().strip().split())
	if size > 0:
		size -= (t - last if size >= t - last else size)
	size += c
	max_size = max(max_size, size)
	last = t
print last + size, max_size
