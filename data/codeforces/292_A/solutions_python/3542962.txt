import sys

n = int(raw_input())

t, m = map(int, sys.stdin.readline().strip().split())

stack = 0
mx = 0
time = 1

while n:
    if stack: 
        stack = stack - 1

    if time==t:
        stack = stack + m
        mx = max(stack, mx)

        n = n - 1
        if n: 
            t, m = map(int, sys.stdin.readline().strip().split())
    
    time = time + 1

print time+stack-1, mx
