n = int(raw_input())

t0, s = [int(x) for x in raw_input().split()]
maxs = s
for i in range(n-1):
    t, c = [int(x) for x in raw_input().split()]
    if s > 0:
        s -= t-t0
        if s < 0:
            s = 0
    s += c
    maxs = max(maxs,s)
    t0 = t
t=t0
print t+s, maxs
