#include<iostream>
#include<map>
using namespace std;
map<string,int> s,g;
string str[1000];
int c[1000],m=0;
int main()
{
	int n;
	cin>>n;
	for(int i=0;i<n;i++)cin>>str[i]>>c[i],s[str[i]]+=c[i];
	for(int i=0;i<n;i++)if(m<s[str[i]])m=s[str[i]];
	int i=0;
	for(;s[str[i]]<m||(g[str[i]]+=c[i])<m;i++);
	cout<<str[i];
	return 0;
}
