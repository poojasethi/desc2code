#include<bits/stdc++.h>
using namespace std;
int main()
{
    int n,i,j,mx=0;
    map<string,int>m,p;
    cin>>n;
    char s[n][33];
    int r[n];
    for(i=0;i<n;i++)cin>>s[i]>>r[i],m[s[i]]+=r[i];
    for(i=0;i<n;i++)mx=max(mx,m[s[i]]);
    for(i=0;;i++)if(m[s[i]]>=mx&&(p[s[i]]+=r[i])>=mx)break;
    cout<<s[i];
    return 0;
}
