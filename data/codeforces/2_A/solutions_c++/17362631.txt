#include<iostream> 
#include<map>
using namespace std; 
map<string,int> f,g; 
string s[1001]; 
int c[1001],i, m=0; 
int main() 
    { 
    int n; 
    cin >> n; 
    for(i=0;i<n; i++){cin>>s[i]>>c[i];f[s[i]]+= c[i];} 
    for(i=0;i<n; i++) m=max(m,f[s[i]]); 
    i=0;
while(f[s[i]]<m  ||  (g[s[i]]+=c[i])<m)i++;

cout<<s[i]; }
