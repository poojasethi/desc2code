n = int(raw_input())
d = dict()
p = []
for i in xrange(n):
	s,k = raw_input().split()
	k = int(k)
	p.append([s,k])
	if s not in d:
		d[s] = 0
	d[s] += k
mx = max(d.itervalues())
u = [k for k in d if d[k] == mx]
d = dict()
for s,k in p:
	if s not in d:
		d[s] = 0
	d[s] += k
	if s in u and d[s] >= mx:
		print s
		break
