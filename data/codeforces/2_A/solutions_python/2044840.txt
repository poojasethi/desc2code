n = int(raw_input())
map = {}
list = []
for i in range(n) :
    s=(raw_input().split())
    if s[0] not in map:
        map[ s[0] ] = 0
    map[ s[0] ] += int(s[1])
    list.append((s[0],map[ s[0] ]))
ans = max(map.values())
for s in list:
    if ans == map[ s[0] ] and s[1] >= ans:
        p = s[0]
        break
print(p)
