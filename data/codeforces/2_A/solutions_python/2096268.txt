n = input()
table = {}
seq = []
for i in xrange(n):
    name, score = raw_input().split()
    score = int(score)
    table[name] = table.get(name, 0) + score
    seq.append((table[name], name))
max_val = max(table.values())
for t in seq:
    if t[0] >= max_val and table[t[1]]==max_val:
        print t[1]
        break
