from collections import Counter

n=int(raw_input())
cnt=Counter()
rec=[]

for i in xrange(n):
    name, score=raw_input().split()
    cnt[name]+=int(score)
    rec.append((name, cnt[name]))

max_score=max(cnt.values())
print [name for (name,score) in rec if score>=max_score and cnt[name]==max_score][0]
