n=input()
d={}
l=[]
for i in range(n):
    name,score=raw_input().split()
    score=int(score)
    if name in d.keys():
        d[name]+=score
    else:
        d[name]=score
    l.append((name,d[name]))
max_score=max(d.values())
for name,score in l:
    if score>=max_score and d[name]==max_score:
        print name
        break
