a={}
b=[]
num=int(raw_input())
for x in xrange(num):
	name,score = raw_input().split()
	score = int(score)
	if name in a.keys():
		a[name]+=score
	else:
		a[name]=score
	b.append((name,a[name]))

maxscore = max(a.values())

for name,score in b:
	if score>=maxscore and a[name]==maxscore:
		print name
		break
