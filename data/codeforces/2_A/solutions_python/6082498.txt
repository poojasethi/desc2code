n = int(raw_input())
names = {}
act = []
for i in range(0, n):
	name, p = raw_input().split(' ')
	p = int(p)
	if name in names:
		names[name] += p
	else:
		names[name] = p
	act.append([name, names[name]])
maxp = max(names.values())
for name, p in act:
	if p >= maxp and names[name] == maxp:
		print name
		break
