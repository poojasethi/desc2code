n=int(raw_input())
mp={}
tp={}
lst=[]
for i in range(n):
	s=raw_input().split()
	tp[s[0]]=0
	lst.append(s)
	if(mp.has_key(s[0])):
		mp[s[0]]+=int(s[1])
	else :
		mp[s[0]]=int(s[1])
ans=max(mp[key] for key in mp.keys())
for elem in lst:
	tp[elem[0]]+=int(elem[1])
	if(tp[elem[0]]>=ans and mp[elem[0]]==ans):
		print elem[0]
		break
