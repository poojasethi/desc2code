n=int(raw_input())
d={}
l=[]

for i in range(n):
    name,score=raw_input().split()
    score = int(score)
    if name in d:
        d[name]+=score
    else:
        d[name]=score
    l.append((name,d[name]))

mx = max(d.values())

for nn,ss in l:
    if ss>=mx and d[nn]>=mx:
        print nn
        break
