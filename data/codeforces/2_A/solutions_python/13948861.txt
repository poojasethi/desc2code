n = int(raw_input())
d = {}
h = []
mx = 0
for _ in range(n):
    p, s = raw_input().split()
    if p in d:
        d[p] += int(s)
    else:
        d[p] = int(s)
    h.append((d[p], p))
m = max(d.values())
ma = [x for (x, y) in d.items() if y == m]
res = [y for (x, y) in h if x >= m and y in ma]
print res[0]
