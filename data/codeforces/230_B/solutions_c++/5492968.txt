#include<bits/stdc++.h>
using namespace std;
long long v[1000005],i,n,j,x;
set<long long>s;
int main()
{
   for(i=2;i<=1000000;i++)
   if(!v[i])
   {
       s.insert(i*i);
       for(j=i*i;j<=1000000;j+=i)
       v[j]=1;
   }
   cin>>n;
   while(n--)
   {
       cin>>x;
       cout<<(s.count(x) ? "YES\n" : "NO\n");
   }
}
