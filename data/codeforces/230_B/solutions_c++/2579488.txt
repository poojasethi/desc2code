#include<iostream>
#include<algorithm>
#include<math.h>
using namespace std;
long long a,ted,b;
int j;
int main(){
	cin>>ted;
	for(int i=0;i<ted;i++){
		cin>>a;
		b=sqrt(a);
		for(j=2;j*j<=b;j++){
			if(b%j==0)
				break;
		}
		if(a>1&& j*j>b && b*b==a)   
			cout<<"YES"<<endl;
		else{
			cout<<"NO"<<endl;
		}
	}
}
