#include<bits/stdc++.h>
using namespace std;
int main()
{ int n; scanf("%i",&n);
  for(int u=0;u<n;u++)
  { long long i; scanf("%I64d",&i);
    long long k=sqrtl(i);
    int j;
    for(j=2;j*j <= k;j++)
     if(k%j==0) break;
    if(j*j > k && k*k == i && i>1) printf("YES\n");
    else printf("NO\n");
  }
}
