#include<cstdio>
#include<cmath>
using namespace std;
inline int pan(long long x){
	int s=(int)(sqrt(x));for(int i=2;i*i<=s;i++)if(s%i==0)return 0;
	return (long long)s*s==x;
}
int main(){
	int n;scanf("%d",&n);
	for(int i=1;i<=n;i++){
		long long x;scanf("%lld",&x);if((x!=1)&&pan(x))printf("YES\n");
		else printf("NO\n");
	}
}
