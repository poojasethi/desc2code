#include <cstdio>
#include <cmath>
int main(){
    long long d,j,k;
    int n;
    scanf("%d",&n);
    while(n--){
        scanf("%I64d",&d);
        j=sqrt(d);
        k=2;
        while(j%k!=0&&k*k<=j) k++;
        if(k*k > j && j*j==d && d!=1) printf("YES\n");
        else printf("NO\n");
    }
   return 0;
}
