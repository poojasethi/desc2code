#include<iostream>
#include<cmath>
using namespace std;

int main()
{int n;
 long long a,s;
cin>>n;
for (int i=1;i<=n;i++)
 {
  cin>>a;

  s=sqrt(a);
  if (s*s!=a||a==1) {cout<<"NO"<<endl;continue;}
  int j;
   for (j=2;j*j<=s;j++)
    if (s%j==0) break;
  if (j*j>s) cout<<"YES"<<endl; else cout<<"NO"<<endl;
 }
return 0;
}
