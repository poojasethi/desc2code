#include <stdio.h>
#include <math.h>
int main()
{
    int n, j;
    long long a, b;
    scanf("%d", &n);
    while(n--){
        scanf("%I64d", &a);
        b= sqrtl(a);
        for(j=2; j*j<= b; j++)
            if(b%j==0)break;
        if(j*j> b && b*b== a && a>1)printf("YES\n");
        else printf("NO\n");
    }
return 0;
}
