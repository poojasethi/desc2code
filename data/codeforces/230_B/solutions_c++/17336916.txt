#include<iostream>
#include<set>
using namespace std;
int v[1000020];
set<long long> s;
int main()
{
	long long i,j,n,t;
	for(i=2;i<=1000000;i++)
		if(!v[i]){
			s.insert(i*i);
			for(j=i*i;j<=1000000;j+=i) v[j]=1;
		}
	cin>>n;
	while(n--){
		cin>>t;
		if(s.count(t)) cout<<"YES\n";
		else cout<<"NO\n";
	}
}
