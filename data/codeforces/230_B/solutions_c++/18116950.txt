#include <iostream>
#include <cmath>
using namespace std;
long long int n,x;
bool a[10000000];
int main(){
	a[0]=a[1]=1;
	for(int i=0;i*i<1e7;i++)
		if(!a[i])
			for(int j=i*i;j<1e7;j+=i)
				a[j]=1;
	cin>>n;
	while(n--){
		cin>>x;
		long long int r=sqrt(x);
		cout<<(r*r==x && !a[r]?"YES":"NO")<<endl;
	}
}
