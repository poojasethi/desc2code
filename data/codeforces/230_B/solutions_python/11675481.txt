mx = 1000001
primes = [False] * mx
primes[1] = True

tprimes = set()

for i in range(1, mx):
    if not primes[i]:
        tprimes.add(i * i)

        for j in range(i * i, mx, i):
            primes[j] = True

input()

for x in map(int, raw_input().split()):
    print "YES" if x in tprimes else "NO"
