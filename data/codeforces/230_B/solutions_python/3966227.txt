from math import sqrt
n = input()
num_in = map(int,raw_input().split(' '))
arr = [1]*(1000001)
arr[1] = 0
for i in xrange(2,1001):
	if arr[i] :
		for j in xrange(i*i,1000001,i):
			arr[j] = 0
for m in num_in:
	root = sqrt(m)
	if root % 1 == 0:
		print['NO','YES'][arr[int(root)]]
	else : print 'NO'
