import sys
def main():
    RL = sys.stdin.readline
    n = int(RL())
    a = map(int, RL().split())
    d = set()
    b = [0] * (10 ** 6 + 10)
    for i in xrange(2, 10 ** 6 + 10):
        if not b[i]:
            d.add(i*i)
            for j in xrange(i + i, 10 ** 6 + 10, i):
                b[j] = 1
    for x in a:
        print "YES" if x in d else "NO"
main()
