import math

m = 1000001
s=[0]*2+[1]*m

for i in range(2,1001):
     if s[i]:
          for j in range(i*2,m,i):
               s[j]=0

n = input()
ch = map(int,raw_input().strip().split())

for i in range(n):
     b=int(math.sqrt(ch[i]))
     if b**2 == ch[i] and s[b]:
         print 'YES'
     else:
         print 'NO'
