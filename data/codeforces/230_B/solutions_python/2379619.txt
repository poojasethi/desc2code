n = int(raw_input())
a = map(int, raw_input().split())
e = [True] * 1000000
s = set()
s.add(4)
for x in xrange(3, 1000000, 2):
    if e[x]:
        for y in xrange(x + x, 1000000, x):
            e[y] = False
        s.add(x * x)
for x in a:
    if x in s:
        print 'YES'
    else:
        print 'NO'
