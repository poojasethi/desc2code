
def main():
    n = int(raw_input())
    a = map(int, raw_input().split())

    st = set()
    b = [0] * (10**6+1)
    for i in xrange(2, 10**6+1):
        if b[i] == 0:
            st.add(i*i)
            for j in xrange(2*i, 10**6+1, i):
                b[j] = 1
    for x in a:
        if x in st:
            print("YES")
        else:
            print("NO")

if __name__ == '__main__':
    main()
