import math
input()
a=map(int,raw_input().split())
limit=10**6
p=[True]*(limit+1)
p[1]=False
for i in xrange(2,limit+1):
    if not p[i]:
        continue
    j=2*i
    while j<=limit:
        p[j]=False
        j+=i
for n in a:
    m=int(math.sqrt(n))
    if m*m==n and p[m]:
        print('YES')
    else:
        print('NO')
