def primes(n):
    sieve = [True] * n
    for i in xrange(3,int(n**0.5)+1,2):
        if sieve[i]:
            sieve[i*i::2*i]=[False]*((n-i*i-1)/(2*i)+1)
    return [2] + [i for i in xrange(3,n,2) if sieve[i]]
p = primes(1000000)
for i in range(len(p)): p[i] = p[i] * p[i]
p = set(p)
n = int(raw_input())
a = map(int, raw_input().split())
for i in a: print "YES" if i in p else "NO"
