#!/usr/bin/python

dt = 1e-7
ma = int(1e6+dt)
rng = xrange(ma+1)
pri = [0 for i in rng]
for i in xrange(2, ma+1):
    if pri[i] == 0:
        for j in xrange(i, ma/i+1):
            pri[i*j] = 1;

n = raw_input()
for x in map(int, raw_input().split()):
    ch = int(x**0.5+dt)
    if ch>1 and ch**2==x and pri[ch]==0:
        print "YES"
    else:
        print "NO"
