MAXN = 10**6

sieve = [False]*(MAXN+1)
sieve[0] = sieve[1] = True
primes = []
for i in range(2, len(sieve)):
  if sieve[i]: continue;
  primes.append(i)
  for j in range(i+i, len(sieve), i):
    sieve[j] = True
tprimes = set([p*p for p in primes])

n = int(raw_input())
x = map(int, raw_input().split())
for a in x:
  print 'YES' if a in tprimes else 'NO'
