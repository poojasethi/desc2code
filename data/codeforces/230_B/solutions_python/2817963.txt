b = 1<<20
p = [False] * b
p[1]=True
def pre():
    for i in range(2, b):
        if p[i]==False:
            for j in range(i*i, b, i):
                p[j] = True
def ok(n):
    x=(int)(n**0.5+0.1)
    return x*x==n and x!=1 and p[x]==False
pre()
n = input()
a = map(int, raw_input().split())
for x in a:
    print ('NO', 'YES')[ok(x)]
