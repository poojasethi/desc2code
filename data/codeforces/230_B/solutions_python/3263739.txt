import math

def sieve(p,n):
    for i in range(n-1):
        p.append(True)
    for i in range(2,n):
        if p[i]:
            for j in range(i*i,n+1,i):
                p[j] = False
                
    
p = [False,False]
sieve(p,1000*1000)
input()
a = map(int, raw_input().split())
for i in a:
    b = int(pow(i,0.5))
    if b**2 == i and p[b]:
        print 'YES'
    else:
        print 'NO'
