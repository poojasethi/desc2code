n = input()
t = []
a = raw_input().split()
for i in range(n):
    t.append(int(a[i]))
g = set()
p = [False] * 1000001
p[1] = [True]
for i in range(1,1000001):
    if p[i] == False:
         g.add(i * i)
         for z in range(i * i, 1000001, i):
             p[z] = True
for i in range(n):
    if t[i] in g:
        print 'YES'
    else:
        print 'NO'
