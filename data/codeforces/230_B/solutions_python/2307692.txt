MAX=1000100
prim=set()
ok=[0]*MAX
def solve():
    for i in range(2,MAX):
        if(not ok[i]):
            for j in range(i+i,MAX,i):
                ok[j]=1
    for i in range(2,MAX):
        if(not ok[i]):
            prim.add(i*i)
solve()
n=int(raw_input())
a=map(int,raw_input().split())
for x in a:
    if x in prim:
        print "YES"
    else:
        print "NO"
