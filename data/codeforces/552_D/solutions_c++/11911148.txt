#include <cstdio>
int x[2000], y[2000];
int main()
{
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
		scanf("%d %d", x + i, y + i);
	long long ans = 1LL * n * (n - 1) * (n - 2) / 6;
	for (int i = 0; i < n; i++)
		for (int j = i + 1; j < n; j++)
			for (int k = j + 1; k < n; k++)
				ans -= ((x[j] - x[i]) * (y[k] - y[i]) == (y[j] - y[i]) * (x[k] - x[i]));
	printf("%I64d\n", ans);
	return 0;
}
