#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    int i,j,k,n,ans=0;
    int x[2010],y[2010];
    cin >> n;
    for (i=1; i<=n; ++i)
        cin >> x[i] >> y[i];
    for (i=1; i<=n-2; ++i)
        for (j=i+1; j<=n-1; ++j)
            for (k=j+1; k<=n; ++k)
    {
        if (x[k]*(y[j]-y[i])-x[i]*y[j]!=y[k]*(x[j]-x[i])-y[i]*x[j]) ans=ans+1;
    }
    cout << ans << endl;
    return 0;
}
