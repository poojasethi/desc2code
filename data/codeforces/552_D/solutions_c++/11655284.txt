#include<stdio.h>
int n,x[2001],y[2001],i,j,k,s,d;
int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++)
		scanf("%d%d",&x[i],&y[i]);
	for(i=0;i<n-2;i++)
		for(j=i+1;j<n-1;j++)
			for(k=j+1;k<n;k++){
				d=x[i]*y[j]+x[j]*y[k]+x[k]*y[i]-x[i]*y[k]-x[j]*y[i]-x[k]*y[j];
				if(d!=0)s++;
			}
	printf("%d",s);
}
