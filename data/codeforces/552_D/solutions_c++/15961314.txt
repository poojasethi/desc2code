#include <stdio.h>
int main(){
int num,i,j,k;
scanf("%d",&num);
int data_x[num];
int data_y[num];
for(i=0;i<num;i++)
scanf("%d %d",data_x+i,data_y+i);
int sum=0;
for(i=0;i<num-2;i++)
    for(j=i+1;j<num-1;j++)
        for(k=j+1;k<num;k++)
            if(((data_x[j]-data_x[i])*(data_y[k]-data_y[i]))-((data_y[j]-data_y[i])*(data_x[k]-data_x[i])))
                sum++;
printf("%d",sum);
return 0;
}
