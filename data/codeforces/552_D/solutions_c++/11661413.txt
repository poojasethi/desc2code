#include <stdio.h>

int X[2001];
int Y[2001];

int main()
{
	long n, count = 0;
	int i;
	scanf("%ld", &n);
	for(i = 1; i <= n; i++)
	{
		scanf("%d %d", &X[i], &Y[i]);
	}
	for(i = 1; i <= n; i++)
	{
		int j;
		for(j = i+1; j <= n; j++)
		{
			int k;
			int tx = X[i]-X[j];
			int ty = Y[i]-Y[j];
			for(k = j+1; k <= n; k++)
			{
				if(tx * (Y[j]-Y[k]) != (X[j]-X[k]) * ty) count++;
			}
		}
	}
	printf("%ld\n", count);
	return 0;
}
