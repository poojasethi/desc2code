#include<bits/stdc++.h>
using namespace std;

const int N = (int)1e5+11;
int x[N];
int y[N];

int main()
{
	int n;
	scanf("%d",&n);
	int ans  =0;
	for(int i=0;i<n;i++){
		scanf("%d%d",&x[i],&y[i]);
	}
	if(n<3){
		cout<<0<<endl;
		return 0;
	}
	for(int i=0;i<n;i++)
		for(int j=i+1;j<n;j++)
			for(int k=j+1;k<n;k++){
				if(x[i]*y[j]+x[j]*y[k]+x[k]*y[i]-x[j]*y[i]-x[k]*y[j]-x[i]*y[k])
					ans++;
			}
	cout<<ans<<endl;
















	return 0;
}
