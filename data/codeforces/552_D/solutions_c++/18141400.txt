#include <cstdio>
short px[2001],py[2001],n;int ans;
int main()
{
	scanf("%hd",&n);
	for(int i=0;i<n;++i)scanf("%hd%hd",px+i,py+i);
	for(int i=0;i<n-2;++i)for(int j=i+1;j<n-1;++j)for(int k=j+1;k<n;++k)
	if((*(px+i)-*(px+j))*(*(py+j)-*(py+k))-(*(py+i)-*(py+j))*(*(px+j)-*(px+k)))++ans;
	printf("%d\n",ans);return 0;
}
