#include <stdio.h>
#define FOR(i, to, n) for (int i=(to); i<n; i++ )
#define ri(x) scanf("%d", &x)

int X[2000];
int Y[2000];

int main () {
	int n, x1, y1, x2, y2, res=0;
	ri(n);
	FOR(i, 0, n) {
		ri(X[i]);
		ri(Y[i]);
	}
	
	FOR(i, 0, n)
		FOR(j, i+1, n)
			FOR(k, j+1, n) {
				x1=X[j]-X[i]; y1=Y[j]-Y[i];
				x2=X[k]-X[i]; y2=Y[k]-Y[i];
				if (x1*y2!=x2*y1) res++;
	}
	
	printf("%d\n", res);
		

	
	return 0;
}
