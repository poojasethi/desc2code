#include <stdio.h>
#define MAX 2005

int n, nr, x[MAX], y[MAX], i, j, k;

int main(){
	scanf("%d", &n);
	for(i = 0; i < n; i++)
		scanf("%d%d", &x[i], &y[i]);

	for(i = 0; i < n - 2; i++)
		for(j = i + 1; j < n - 1; j++)
			for(k = j + 1; k < n; k++)
				if(x[k] * y[j] + x[i] * y[k] + x[j] * y[i] - x[i] * y[j] - x[j] * y[k] - x[k] * y[i] != 0)
					nr++;

	printf("%d\n", nr);
	return 0;
}
