#include<bits/stdc++.h>
using namespace std;
int x[2003],y[2003],i,j,k,n,ans=0;
main()
{
    scanf("%d",&n);
    for (i=0;i<n;i++) scanf("%d%d",&x[i],&y[i]);
    for (i=0;i<n;i++)
        for (j=i+1;j<n;j++)
            for (k=j+1;k<n;k++)
                if ((x[k]-x[i])*(y[j]-y[i])-(x[j]-x[i])*(y[k]-y[i])!=0)
                    ans++;
        printf("%d",ans);
}
