#include <iostream>
#include <cstdio>
using namespace std;
int main()
{
    long long ans=0;
    int i,j,k,n;
    int x[2010],y[2010];
    scanf("%d",&n);
    for (i=0;i<n;i++)
    {
        scanf("%d%d",&x[i],&y[i]);
    }
    for (i=0;i<n;i++)
        for (j=i+1;j<n;j++)
            for (k=j+1;k<n;k++)
                if ((x[j]-x[i])*(y[k]-y[i])!=(x[k]-x[i])*(y[j]-y[i])) ans++;
    printf("%I64d\n",ans);
    return 0;
}
