#include<iostream>
using namespace std;
int a[2005],b[2005];
int tri(int i,int j,int k)
{
	int m,l;
	m=(b[k]-b[i])*(a[j]-a[i]);
	l=(b[j]-b[i])*(a[k]-a[i]);
	if(m==l)
		return 0;
	else
		return 1;
}
int main()
{
	int n,i,j,k,count=0;
	cin >> n;
	for(i=0;i<n;i++)
	{
		cin >> a[i];
		cin >> b[i];
	}
	for(i=0;i<n-2;i++)
	{
		for(j=i+1;j<n-1;j++)
		{
			for(k=j+1;k<n;k++)
				count+=tri(i,j,k);
		}
	}
	cout << count<< endl;
	return 0;
}
