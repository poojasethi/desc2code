#include<math.h>
#include<stdio.h>
int x[2009];
int y[2009];
int main()
{
	int i,j,k,n;
	while(~scanf("%d",&n))
	{
		for(i=1;i<=n;i++){
			scanf("%d%d",&x[i],&y[i]);
		}
		int count=0;
		for(i=1;i<=n;i++){
			for(j=i+1;j<=n;j++){
				for(k=j+1;k<=n;k++){
					if(!((y[i]-y[j])*(x[j]-x[k])==(x[i]-x[j])*(y[j]-y[k])))
						count++;
				}
			}
		}
		printf("%d\n",count);
	}
	return 0;
}
