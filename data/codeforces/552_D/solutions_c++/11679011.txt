#include<stdio.h>
int n;
int x[2015],y[2015];
long long ans;

int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++)    scanf("%d%d",&x[i],&y[i]);
    for(int i=0;i<n-2;i++)
        for(int j=i+1;j<n-1;j++)
            for(int k=j+1;k<n;k++)
    {
        if((y[i]-y[j])*(x[j]-x[k])!=(y[j]-y[k])*(x[i]-x[j]))    ans++;
    }
    printf("%I64d",ans);
    return 0;
}
