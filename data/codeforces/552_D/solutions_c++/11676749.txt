#include <iostream>
using namespace std;
int main()
{
	int n, x[2015],y[2015], ans=0;
	cin>>n;
	for (int i=0; i<n; i++) cin>>x[i]>>y[i];
	for (int i=0; i<n; i++) for (int j=i+1; j<n; j++) for (int k=j+1; k<n; k++) {
		int x1=x[j]-x[i], y1=y[j]-y[i], x2=x[k]-x[i], y2=y[k]-y[i];
		if (x1*y2!=x2*y1) ans++;
	}
	cout<<ans<<endl;
	return 0;
}
