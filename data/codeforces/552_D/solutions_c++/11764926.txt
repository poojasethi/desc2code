#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int mas[n][2];
    for (int i=0;i<n;i++)
        cin>>mas[i][0]>>mas[i][1];
    int kol=0;
    for(int i=0;i<n;i++)
        for(int j=i+1;j<n;j++)
        for(int k=j+1;k<n;k++)
    {
        if (((mas[j][0]-mas[i][0])*(mas[k][1]-mas[i][1])-(mas[j][1]-mas[i][1])*(mas[k][0]-mas[i][0]))!=0) kol++;
    }
    cout<<kol<<endl;
    return 0;
}
