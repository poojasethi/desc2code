#include <iostream>
using namespace std;
int x[2005],y[2005];
int main()
{
    int n;
    cin >>n;
    int num=0;
    for(int i=0;i<n;i++){
        cin>>x[i]>>y[i];
        if(i>1){
            for(int j=0;j<i;j++)
                for(int k=j+1;k<i;k++)
                    if((x[i]-x[j])*(y[i]-y[k])-(y[i]-y[j])*(x[i]-x[k]))
                        num++;
        }
    }
    cout <<num<<endl;
    return 0;
}
