#include <stdio.h>
int main()
{
	int n;
	scanf ("%d",&n);
	int x[3002];
	int y[3002];
	long long ans=0,i,k,j;
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x[i],&y[i]);
	}
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++)
			for(k=j+1;k<n;k++)
				if(((x[i]-x[j])*(y[j]-y[k]))!=(x[j]-x[k])*(y[i]-y[j]))
					ans++;
	printf ("%lld",ans);
}
