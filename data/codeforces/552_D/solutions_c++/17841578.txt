#include<stdio.h>

int main(){
	int n, ans = 0;
	scanf("%d", &n);
	int x[2000] = {};
	int y[2000] = {};
	for(int i = 0 ; i < n; i++)
		scanf("%d %d", &x[i], &y[i]);
	for(int i = 0; i < n; i++)
		for(int j = i + 1; j < n; j++)
			for(int k = j + 1; k < n; k++)
				if((x[k]-x[i])*(y[j]-y[i])-(y[k]-y[i])*(x[j]-x[i]) != 0)
					ans++;
	printf("%d\n", ans);
	return 0;
}
