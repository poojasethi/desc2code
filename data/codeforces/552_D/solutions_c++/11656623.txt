#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n ;
	cin >> n;
	int x[n] , y[n];
	for(int i = 0;i < n;i++)
	{
		scanf("%d%d",&x[i],&y[i]);		
	}
	long long int cnt = 0;
	for(int i = 0;i < n;i++)
	{
		for(int j = i + 1;j < n;j++)
		{
			for(int k = j+1;k < n;k++)
			{
				int temp = x[i]*(y[j] - y[k]) + x[j]*(y[k] - y[i]) + x[k]*(y[i] - y[j]);
				if(temp)
					cnt++;
			}
		}
	}
	cout << cnt;
}
