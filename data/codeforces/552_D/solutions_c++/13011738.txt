#include<iostream>
using namespace std;
int main()
{
	int n,ans=0,i,j,k;
	cin>>n;
	int a[n][2];
	for(i=0;i<n;i++)
	{
		cin>>a[i][0]>>a[i][1];
	}
	if(n<3)
	{
		cout<<"0";
		return 0;
	}
	for(i=0;i<n-2;i++)
	{
		for(j=i+1;j<n-1;j++)
		{
			for(k=j+1;k<n;k++)		
			{
				if((a[i][0]*(a[j][1]-a[k][1])+a[j][0]*(a[k][1]-a[i][1])+a[k][0]*(a[i][1]-a[j][1]))!=0)
				{
					ans++;
				}
			}
		}	
	}
	cout<<ans;
	return 0;
}
