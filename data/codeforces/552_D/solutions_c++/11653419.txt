#include <map>
#include <stdio.h>
using namespace std;
int x[2222], y[2222], res, n;
int main(){
	scanf("%d", &n);
	for(int i = 0;  i < n; ++ i){
		scanf("%d %d", x + i, y + i);
	}
	for(int i = 0; i < n; ++ i){
		for(int j = i + 1; j < n; ++ j){
			for(int k = j + 1;  k < n; ++ k){
				res += (x[i] * (y[j] - y[k]) + x[j] * (y[k] - y[i]) + x[k] * (y[i] - y[j])) != 0;
			}
		}
	}
	printf("%d", res);
	return 0;
}
