#include <iostream>
#include <algorithm>
using namespace std;
int main(){
    int n,x[2000],y[2000],c=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
    	cin>>x[i]>>y[i];
	}
	for(int i=0;i<n-2;i++)
	{
		for(int j=i+1;j<n-1;j++)
		{
			for(int h=j+1;h<n;h++)
			{
				if((x[j]-x[i])*(y[j]+y[i])+(x[i]-x[h])*(y[i]+y[h])+(x[h]-x[j])*(y[h]+y[j])!=0)c++;
			}
		}
	}
	cout<<c;
    return 0;
}
