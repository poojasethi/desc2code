#include <iostream>
#include <cstdio>
using namespace std;
#define lol long long
double a[2100],b[2100];


int main()
{
  ios::sync_with_stdio(false);
  int n;
  cin>>n;
  for(int i=1;i<=n;i++)
    cin>>a[i]>>b[i];
  int ans=0;
  for(int i=1;i<=n;i++)
    for(int j=1;j<i;j++)
      for(int k=1;k<j;k++)
	{
	  int tmp=(b[k]-b[i])*(a[j]-a[i])-(a[k]-a[i])*(b[j]-b[i]);
	  if(tmp)
	    ans++;
	}
  cout<<ans<<endl;
  return 0;
}
