#include <stdio.h>
int n, x[2000], y[2000], res;
int main(){
	scanf("%d", &n);
	for(int i = 0; i < n; i ++){
		scanf("%d %d", x+i, y+i);
	}
	for(int i = 0; i < n-2; i ++){
		for(int j = i+1; j < n-1; j ++){
			for(int k = j+1; k < n; k ++){
				int t = (x[i] - x[j]) * (y[i] + y[j]) + (x[j] - x[k]) * (y[j] + y[k]) + (x[k] - x[i]) * (y[k] + y[i]);
				res += t != 0;
			}
		}
	}
	printf("%d", res);
	return 0;
}
