#include <iostream>
using namespace std;
int main()
{
    long long sum=0;
    int n;
    cin>>n;
    int a[n][2];
    for(int i=1;i<=n;i++)
    cin>>a[i][1]>>a[i][2];
    for(int i=1;i<=n-2;i++)
    for(int j=i+1;j<=n-1;j++)
    for(int k=j+1;k<=n;k++)
    if((a[j][2]-a[i][2])*(a[k][1]-a[j][1])!=(a[k][2]-a[j][2])*(a[j][1]-a[i][1]))
    sum++;
    cout<<sum;
}
