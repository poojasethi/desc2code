#include<algorithm> 
#include<iostream>
#include<vector>
using namespace std;
int main()
{
	int n,ans=0;
	cin>>n;
	int a[n],b[n];
	vector<pair<int,int> >s;
	for(int i=0;i<n;i++)cin>>a[i]>>b[i];
	for(int i=0;i<n;i++)for(int j=i+1;j<n;j++)for(int k=j+1;k<n;k++)
	{
		int x1=a[i]-a[j],y1=b[i]-b[j];
		int x2=a[i]-a[k],y2=b[i]-b[k];
		if(x1*y2!=x2*y1)ans++;
	} 
	cout<<ans<<endl;
}
