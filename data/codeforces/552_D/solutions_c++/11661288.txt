#include<cstdio>
using namespace std;
main()
{
    int n; scanf("%d", &n);
    int x[n], y[n];
    int s=0;
    for(int i=0;i<n;i++)
        scanf("%d %d", x+i, y+i);
    for(int i=0;i<n-2;i++)
        for(int j=i+1;j<n-1;j++)
            for(int k=j+1;k<n;k++)
                if((y[j]-y[i])*(x[k]-x[i])!=(y[k]-y[i])*(x[j]-x[i]))
                    s++;
    printf("%d", s);
    
}
