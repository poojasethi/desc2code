#include<iostream>
#include<cstdlib>
using namespace std;
int main()
{
    int x[2001], y[2001], n, s = 0;
    cin >> n;
    for(int i = 1 ; i <= n ; i++) cin >> x[i] >> y[i];
    for(int i = 1 ; i <= n - 2; i++)
        for(int j = i + 1 ; j <= n -1 ; j++)
            for(int k = j + 1 ; k <= n ; k++)
                if( abs( x[i] * ( y[j] - y[k] ) + x[j] * ( y[k] - y[i] ) + x[k] * ( y[i] - y[j] ) ) ) s++;
    cout << s << endl;
}
