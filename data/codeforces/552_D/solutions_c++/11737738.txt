#include<iostream>
using namespace std;
int i,j,k,n,Count;
int v[2000][2];
int istriangle(int i,int j,int k)
{
	return (v[i][0]*(v[j][1]-v[k][1])+v[j][0]*(v[k][1]-v[i][1])+v[k][0]*(v[i][1]-v[j][1]));
			}

int main()
{	cin>>n;
	Count=0;
	for(i=0;i<n;i++)
		cin>>v[i][0]>>v[i][1];
	for(i=0;i<n;i++)
		for(j=i+1;j<n;j++)
			for(k=j+1;k<n;k++)
				if(istriangle(i,j,k)!=0)
					Count++;
	cout<<Count<<endl;
	return 0;
}
