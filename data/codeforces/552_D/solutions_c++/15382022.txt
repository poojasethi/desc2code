#include<cstdio>
using namespace std;
int n;
long long ans;
int x[2010] , y[2010];

inline bool ok(int a, int b, int c){
	return (y[b]-y[a])*(x[c]-x[b]) != (y[c]-y[b])*(x[b]-x[a]);
}

int main(){
	scanf("%d", &n);
	for(int i=0 ; i<n ; ++i)
		scanf("%d%d", &x[i], &y[i]);
	for(int i=0 ; i<n-2 ; ++i)
		for(int j=i+1 ; j<n-1 ; ++j)
			for(int k=j+1 ; k<n ; ++k)
				if(ok(i,j,k))
					++ans;
	printf("%lld", ans);
	return 0;
}
