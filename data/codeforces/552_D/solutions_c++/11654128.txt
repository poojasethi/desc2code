#include<iostream>

using namespace std;
struct coord{
	int x,y;
}a[2001];

bool s(coord r,coord b, coord c){
	double k;
	return 0.5*((b.x-r.x)*(c.y-r.y)-(c.x-r.x)*(b.y-r.y));

}

int main(){
int n,t=0;
cin>>n;
for(int i=1;i<=n;i++)
	cin>>a[i].x>>a[i].y;

for(int i=1;i<=n-2;i++)
	for(int j=1+i;j<=n-1;j++)
		for(int k=j+1;k<=n;k++){
			if(s(a[i],a[j],a[k]))
			t++;
		}
	
	cout<<t;
	
return 0;
}
