#include<iostream>
using namespace std;
int x[2010],y[2010];
int main()
{
    int n,i,j,k;
    long long c=0;
    cin>>n;
    for(i=0;i<n;i++)
       cin>>x[i]>>y[i]; 
    for(i=0;i<=n-3;i++)
        for(j=i+1;j<=n-2;j++)
            for(k=j+1;k<n;k++)
                if((((x[i]*(y[j]-y[k]))+(x[j]*(y[k]-y[i]))+(x[k]*(y[i]-y[j]))))!=0)
                    c++;
    cout<<c;
    return 0;
}
