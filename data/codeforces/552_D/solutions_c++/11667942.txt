#include<bits\stdc++.h>
using namespace std;

int main()
{
    long n;
    cin >> n;
    long x[n],y[n];
    for(long i = 0;i<n;i++)
        cin >> x[i] >>y[i];

    long ans = 0;
    for(long i=0;i<n-2;i++)
        for(long j=i+1;j<n-1;j++)
            for(long k = j+1;k<n;k++)
                ans+=(x[i]-x[j])*(y[k]-y[j])!=(y[i]-y[j])*(x[k]-x[j]);

    cout<<ans;
    return 0;
}
