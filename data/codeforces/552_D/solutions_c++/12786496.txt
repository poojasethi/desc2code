#include<bits/stdc++.h>
int n,s,x[2002],y[2002];
using namespace std;

int main()
{
    cin >> n;
    for(int i=0; i<n; i++) cin >> x[i] >> y[i];
    for(int i=0; i<n; i++) 
        for(int j=i+1; j<n; j++) 
            for(int k=j+1; k<n; k++)
                s += ( (y[j]+y[i])*(x[j]-x[i]) + (y[k]+y[j])*(x[k]-x[j]) + (y[i]+y[k])*(x[i]-x[k]) != 0 );
    cout << s << endl;
}
