#include<bits/stdc++.h>
#define MAXN 100009
using namespace std;
int x[MAXN],y[MAXN];
int main()
{
	int a,ans=0;
	cin>>a;
	for(int i=0;i<a;i++)
		cin>>x[i]>>y[i];
	for(int i=0;i<a-2;i++)
		for(int j=i+1;j<a-1;j++)
			for(int k=j+1;k<a;k++)
				if((y[k]-y[j])*(x[i]-x[j])!=(x[k]-x[j])*(y[i]-y[j]))
					ans++;
	cout<<ans<<endl;
}
