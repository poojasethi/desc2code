a=int(raw_input())
if a<=2:print 0
else:
    ans=a*(a-1)*(a-2)/6
    coor=[map(int,raw_input().split()) for i in xrange(a)]
    for i in xrange(len(coor)):
        seen={}
        x1,y1=coor[i]
        for j in xrange(i+1,len(coor)):
            x2,y2=coor[j]
            if x2-x1==0:m=1e999
            else:
                m=(y2-y1)*1.0/(x2-x1)
            if m not in seen:seen[m]=0
            ans-=seen[m]
            seen[m]+=1
    print ans
