#include<iostream>
#include<algorithm>
using namespace std;
__int64  n,sum,no,s,a[50000];
int main(){
    cin>>n;
    int i;
    for( i=0;i<n;i++){
        cin>>a[i];
        s+=a[i];
    }
    s/=n;
    for( i=0;i<n;i++){
        no+=a[i]-s;// 累加 差距，向后地推。
        sum+=abs(no);
    }
    cout<<sum<<endl;
}
