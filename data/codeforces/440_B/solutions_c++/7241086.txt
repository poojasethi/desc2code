#include <cmath>
#include <iostream>
using namespace std;
long long a[100000];
int main(){
    long long n,s=0,cnt=0,ihave=0;
    cin >> n;
    for(int i=0;i<n;i++){
        cin >> a[i];
        s+=a[i];
    }
    s/=n;
    for(int i=0;i<n;i++){
            ihave+=a[i]-s;
            cnt+=abs(ihave);
    }
    cout << cnt;
}
