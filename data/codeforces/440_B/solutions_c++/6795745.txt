#include<iostream>
using namespace std;
long long a[50005];
int main()
{int n; cin>>n;
long long s=0;
for(int i=0;i<n;i++) {cin>>a[i]; s+=a[i];}
long long k=s/n;
long long ans=0;
for(int i=0;i<n-1;i++)
if (a[i]>k) {ans+=(a[i]-k); a[i+1]+=(a[i]-k);}
   else {ans+=(k-a[i]);a[i+1]-=(k-a[i]);}
cout<<ans;
return 0;
}
