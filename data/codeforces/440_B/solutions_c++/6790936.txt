#include <bits/stdc++.h>
using namespace std;
long long  arr[50010],sum,x,ans;
int main(int argc, char **argv) {

    int n,i;
    for(cin >> n,i=0;i<n;cin >> arr[i],sum+=arr[i++]);
    sum/=n;
    for(i = 0 ;i < n ; ++i)
    {
        x+= sum - arr[i];
        ans+=abs(x);
    }
    cout << ans << endl;
    return 0;
}
