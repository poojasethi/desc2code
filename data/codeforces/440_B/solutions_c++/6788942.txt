#include <iostream>
using namespace std;

int main() {
 long long int sum=0,n,k,total=0,i,a[100001];
 cin>>n;
 for(i=0;i<n;i++)
 {cin>>a[i];
 sum+=a[i];
 }
 k=sum/n;
 for(i=0;i<n;i++)
 {
 if(a[i]>k){a[i+1]+=(a[i]-k);total+=(a[i]-k);a[i]=k;}
 else if(a[i]<k){a[i+1]-=(k-a[i]);total+=(k-a[i]);a[i]=k;}
 }
 cout<<total<<endl;
 return 0;
}
