#include <iostream>
#include <algorithm>

using namespace std;

int main()
{long long  n;
long long  k,sum=0;
cin>>n;
long long a[50000];
for(int i=0;i<n;i++)
{
    cin>>a[i];
    sum+=a[i];
}
k=sum/n;
long long m=0;
for(int l=0;l<n-1;l++)
{
   m=m+abs(a[l]-k) ;
   a[l+1]=a[l+1]+a[l]-k;
}
cout<<m;
    return 0;
}
