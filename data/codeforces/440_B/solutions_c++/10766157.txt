#include<cmath>
#include <iostream>
using namespace std;

int main() {
    long long int n,i,p=0,c=0,sum=0;
    cin>>n;
    long long int a[n];
    for(i=0;i<n;i++)
    {cin>>a[i];
    sum+=a[i];
    }
    sum=sum/n;
    
    for(i=0;i<n;i++)
    {
        c+=a[i]-sum;
        p+=abs(c);
        
    }
    cout<<p;
    return 0;
}
