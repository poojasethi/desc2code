#include<iostream>
#include<algorithm>
#include<map>
using namespace std;


long long arr[50005],sum=0,avg,ans=0;
int n,i;
int main(){
cin>>n;
for(i=1;i<=n;i++)
{
cin>>arr[i];
sum=sum+arr[i];
}
avg=sum/n;
for(i=1;i<=n;i++)
{
ans=ans+abs(arr[i]-avg);
arr[i+1]=arr[i+1]+(arr[i]-avg);
}
cout<<ans;
return 0;
}
