#include<bits/stdc++.h>
#define LL long long
using namespace std;
LL n;
LL a[50005];
int main()
{
	while(cin>>n)
	{
		LL sum=0;
		for(int i=1;i<=n;i++)
		{
			scanf("%d",&a[i]);sum+=a[i];
		}
		sum/=n;
		LL ans=0;
		for(int i=1;i<n;i++)
			{a[i+1]+=a[i]-sum;ans+=abs(a[i]-sum);}
		cout<<ans<<endl;
	}
	return 0;
}
