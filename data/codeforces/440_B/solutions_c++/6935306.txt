#include<cstdio>
#include<algorithm>
using namespace std;
#define LL long long
int a1[50000];
int main()
{
	int n;LL t=0,a=0,o=0;
	scanf("%d",&n);
	for(int i=0;i<n;++i)
	{
		scanf("%d",a1+i);
		t+=a1[i];
	}
	t/=n;a=0;--n;
	for(int i=0;i<n;++i)
	{
		a+=a1[i]-t;
		o+=abs(a);
	}
	printf("%I64d\n",o);
}
