#include "iostream"
#include "cmath"

using namespace std;

int n;
long long a[100000];
long long ans = 0;

int main() {
	cin >> n; long long tot = 0;
	for (int i = 1; i<= n;i++) { cin>> a[i]; tot += a[i]; }
		tot /= n;
	for (int i = 1; i < n ; i++) { ans += abs(a[i] - tot);a[i + 1] += a[i] - tot;}
		cout << ans << endl;
}
