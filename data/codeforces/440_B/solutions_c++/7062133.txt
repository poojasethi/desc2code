#include <iostream>
#include <cmath>

using namespace std;

#define ll long long

ll n;
ll a[50005];

int main()
{
	ll i,x,sum,ans,now;
	cin>>n;
	sum=0; ans=0;
	for (i=1;i<=n;i++)
	{
		cin>>a[i]; sum+=a[i];
	}
	sum/=n;
	for (i=1;i<n;i++)
	{
		now=sum-a[i];
		a[i+1]-=now;
		ans+=abs(now);
	}
	cout<<ans<<endl;
	return 0;
}
