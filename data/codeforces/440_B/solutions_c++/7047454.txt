#include <bits/stdc++.h>
using namespace std;
long long int i,n,top,t,arr[500000];
int main(){
    scanf("%lld",&n);
    for(i=0;i<n;i++){
        cin >>arr[i];
        t+=arr[i];
    }
    t/=n;
    for(i=0;i<n;i++){
        top+=abs(arr[i]-t);
        arr[i+1]+=arr[i]-t;
    }
    printf("%lld",top);
}
