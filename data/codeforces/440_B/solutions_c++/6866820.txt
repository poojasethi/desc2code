#include<cstdio>

int n;
long long s;
long long a[50500];
long long k;
int main(){
  scanf("%d",&n);
  for(int i=1;i<=n;i++)scanf("%I64d",&a[i]);
  for(int i=1;i<=n;i++)s+=a[i];
  k=s/n;
  long long ans=0;
  for(int i=1;i<n;i++){
    if(a[i]<k)ans+=k-a[i];
    else ans+=a[i]-k;
    a[i+1]=a[i+1]+a[i]-k;
  }
  printf("%I64d\n",ans);
}
