#include <bits/stdc++.h>
using namespace std;

int main ()
{
	long long n, sum = 0, m, s = 0;;
	cin >> n;
	long long *a = new long long [n];
	for (int i = 0; i < n; i++){
		cin >> a[i];
		s += a[i];
	}
	s /= n;

	for(int i = 0; i < n - 1; i++){
		m = s - a[i];
		sum += abs(m);
		a[i + 1] -= m;
	}
	cout << sum << endl;
	return 0;
}
