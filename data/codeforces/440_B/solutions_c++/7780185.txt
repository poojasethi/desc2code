#include<iostream>
#include<algorithm>

using namespace std ;

int n ;
long long s=0 , p , ans=0 , k=0 ;
long long a[50010] ;

int main()
{
	cin >> n ;
	for(int i=0 ; i<n ; i++)
	{
		cin >> a[i] ;
		s+=a[i] ;
	}
	p=s/n ;
	for(int i=0 ; i<n ; i++)
	{
		k+=a[i]-p ;
		ans+= k>0 ? k : -k ;
	}
	cout << ans << endl ;
}
