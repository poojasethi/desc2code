#include <iostream>
#include <algorithm>

using namespace std;

int main(){
	long long int n, A[50000], i, val = 0, sum = 0, s = 0;
	cin >> n;
	for(i = 0; i < n; i++){
		cin >> A[i];
		s += A[i];
	}
	s /= n;
	for(i = 0; i < n; i++){
		val += A[i] - s;
		sum += abs(val);
	}
	cout << sum;
	return 0;
}
