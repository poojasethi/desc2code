#include<bits/stdc++.h>

using namespace std;

int main()
{
    long long i,n,av=0,A[50005],tmp=0,sum=0;
    scanf("%lld",&n);
    for(i=1;i<=n;i++)
    {
        scanf("%lld",&A[i]);
        av=av+A[i];
    }
    av=av/n;
    for(i=1;i<=n;i++)
    {
		tmp+=A[i]-av;
		sum+=abs(tmp);
	}
    printf("%lld\n",sum);
    return 0;
}
