#include<iostream>
#include<cmath>
using namespace std;

int main(){
    long long int n, a[60000],m=0,u=0,p = 0,i;
    cin>>n;

    for(i=1; i<=n; i++){
        cin>>a[i];
        m = m + a[i];
    }
    m = m/n;

    for(i=1; i<=n; i++){
        u+=(a[i]-m);
        p = p+abs(u);
    }
    cout<<p;
    return 0;
}
