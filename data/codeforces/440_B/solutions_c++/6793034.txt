#include <iostream>
#include <algorithm>

using namespace std;

typedef long long ll;

ll N, A[50050];

int main()
{
	cin >> N;
	ll total = 0;
	for(int i = 0; i < N; i++)
	{
		cin >> A[i];
		total += A[i];
	}
	total /= N;
	ll ans = 0, cur = 0;
	for(int i = 0; i < N; i++)
	{
		ans += abs(cur);
		cur += (A[i] - total);
	}
	cout << ans;
}
