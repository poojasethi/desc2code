#include<iostream>
#include<algorithm>
using namespace std;
long long sum=0,m,rez=0,a[50100],N;

int main() {
    cin>>N;
    for(int i=1;i<=N;++i) {
        cin>>a[i];
        sum += a[i];
    }
    m = sum/N;

    for(int i=1;i<=N;++i) {
        rez = rez + max(m-a[i],a[i] - m);
        a[i+1] = a[i+1] + a[i] - m;
    }
    cout<<rez;
    return 0;
}
