#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long a[n],s=0,ans=0,now=0;
	for(int i=0;i<n;i++)
    {
        cin>>a[i];
        s+=a[i];
    }
	s/=n;
	for(int i=0;i<n;i++)
	{
	    now+=a[i]-s;
	    ans+=fabs(now);
	}
	cout<<ans<<endl;
	return 0;
}
