n = int(raw_input())
num = (raw_input()).split()
sum = 0
ans = 0
for key in num:
    sum=sum+int(key)
sum = sum//n
#print sum
if n > 1:
    for i in range(len(num)):
        if int(num[i]) < sum:
            ans = ans+sum-int(num[i])
            num[i+1] = str(int(num[i+1])-(sum-int(num[i])))
        elif sum < int(num[i]):
            ans = ans+int(num[i])-sum
            num[i+1]=str(int(num[i+1])+(int(num[i])-sum))

print ans
