__author__ = 'Yoones'

res = 0

n = int(raw_input())

lst = map(int, raw_input().split(' '))

targ = sum(lst) / n

for i in range(len(lst)):
    if lst[i] > targ:
        dif = lst[i] - targ
        lst[i] -= dif
        lst[i + 1] += dif
        res += dif
    elif lst[i] < targ:
        dif = targ - lst[i]
        lst[i] += dif
        lst[i + 1] -= dif
        res += dif

print res
