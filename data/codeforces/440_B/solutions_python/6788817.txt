n = int(raw_input())
array = map(int, raw_input().split())
total = sum(array)
one = total / n

result = 0

for i in range(n):
	if array[i] < one:
		result += one - array[i]
		array[i+1] -= one - array[i]
	elif array[i] > one:
		result += array[i] - one
		array[i+1] += array[i] - one
print result
