n=int(raw_input())
arr=map(int, raw_input().split())
div=sum(arr)/n
count=0
for i in xrange(n-1):
    if(arr[i]==div):
        continue
    if(arr[i]<div):
        count+=(div-arr[i])
        temp=arr[i]
        arr[i]+=(div-arr[i])
        arr[i+1]-=(div-temp)
    if(arr[i]>div):
        count+=(arr[i]-div)
        temp=arr[i]
        arr[i]-=(arr[i]-div)
        arr[i+1]+=(temp-div)
print count
