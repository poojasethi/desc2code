n = int(raw_input())
matches = map(int, raw_input().split())
target = sum(matches)/n
ans = 0

for i in xrange(n):
	j = i
	while matches[i] != target:
		if matches[i] < target:
			diff = min(matches[j]-target, target-matches[i])
			matches[i] += diff
			matches[j] -= diff
		else:
			diff = min(matches[i]-target, target-matches[j])
			matches[i] -= diff
			matches[j] += diff
		ans += diff*abs(i-j)
		j += 1
	
print ans
