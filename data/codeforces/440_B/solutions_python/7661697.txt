n,a=input(),map(int,raw_input().split())
ave=sum(a)/n
a=map(lambda x: [x[0],x[1]-ave], enumerate(a))
neg=filter(lambda x:x[1]<0,a)
pos=filter(lambda x:x[1]>0,a)
i = j = ans = 0
while i < len(neg):
	s=neg[i][1]+pos[j][1]
	if s<0: 
		ans += abs(neg[i][0]-pos[j][0])*pos[j][1] 
		neg[i][1]+=pos[j][1]
		j += 1
	else: 
		ans -= abs(neg[i][0]-pos[j][0])*neg[i][1]
		pos[j][1]+=neg[i][1]
		i += 1 
		if not s: j += 1
print ans
