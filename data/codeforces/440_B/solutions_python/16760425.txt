no = int(raw_input())
arr = map(int , raw_input().split())
total = sum(arr)
each = total/no
count = 0 
for  i in range(no - 1 ):
	if(arr[i] == each):
		continue
	elif(arr[i] <each):
		defi = (each - arr[i])
		count += defi
		arr[i+1] -= defi
		arr[i] = each
	else:
		extra = arr[i] - each
		count += extra
		arr[i+1] += extra
		arr[i] = each
print count
