#include<bits/stdc++.h>
using namespace std;
set<char>let[102];
char name[102];
int main()
{
	int n,m	;
	long long ans=1;
	cin>>n>>m;
	for(int i=0;i<n;i++)
	{
		scanf("%s",name);
		for(int x=0;x<m;x++)
			let[x].insert(name[x]);
	}
	for(int j=0;j<m;j++)
		ans=(ans*(let[j].size()))%1000000007;
	cout<<ans<<endl;
	return 0;
}
