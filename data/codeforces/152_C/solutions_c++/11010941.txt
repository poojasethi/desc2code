#include<cstdio>
int x=1,i,k,j,n,m,t;
char s[128][128];
int main(){
    scanf("%d%d ",&n,&m);
    for(;i<n;++i)gets(s[i]);
    for(i=0;i<m;++i){
        for(k=0,t=n;k<n;++k)
            for(j=0;j<k;++j)
                if(s[k][i]==s[j][i]){--t;break;}
        x=x*1ll*t%1000000007;
    }
    printf("%d\n",x);
    return 0;
}
