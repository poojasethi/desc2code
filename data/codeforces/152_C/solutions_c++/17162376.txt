#include<bits/stdc++.h>

using namespace std;

const int kMaxN = 128;
const int kMod = 1000000007;

string s[kMaxN];

int main() {
  int n, m;
  cin>>n>>m;
  for(int i=0; i<n; ++i) {
    cin>>s[i];
  }
  int ans = 1;
  for(int i=0; i<m; ++i) {
	set<char>st;
	for(int j=0; j<n; ++j) {
	  st.insert(s[j][i]);
	}
	ans = (1LL*ans*st.size())%kMod;
  }
  cout<<ans<<"\n";
}
