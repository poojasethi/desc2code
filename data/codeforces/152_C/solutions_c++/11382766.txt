#include<bits/stdc++.h>
using namespace std;
long long mod=1000000007;
int main()
{

int n,m;
cin>>n>>m;
string str[n];
int i,j;
for(i=0;i<n;i++)
{
cin>>str[i];
}
int arr[26];
long long result=1;int c;
for(i=0;i<m;i++)
{
c=0;
memset(arr,0,sizeof(arr));
for(j=0;j<n;j++)
{
if (arr[str[j][i]-'A']==0)
{
c++;
arr[str[j][i]-'A']=1;
}
}
result=(result%mod*c%mod)%mod;
}
cout<<result<<"\n";
return 0;
}
