#include <iostream>
#include <set>
using namespace std;
#define MOD 1000000007
int main()
{
unsigned long long n,m,ans=1;
string s[101];
cin>>n>>m;
for (int i=0;i<n; i++) cin>>s[i];

for (int j=0;j<m; j++)
 { set<char> a;
    for (int i=0;i<n; i++) a.insert(s[i][j]);
  ans=ans*a.size()%MOD;
  }
cout<<ans<<endl;
 return 0;
}
