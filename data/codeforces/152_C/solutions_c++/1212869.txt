#include<iostream>
#include<set>
#include<vector>
using namespace std;
int main(){
int n,k;
cin>>n>>k;
string x;
set <char> str;
vector<string> v;
for(int i=0;i<n;i++){
cin>>x;
v.push_back(x);
}
long long int ans=1;
for(int j=0;j<k;j++){
for(int i=0;i<n;i++){
string x = v[i];
str.insert(x[j]);
}
ans = ans * str.size();
ans = ans % 1000000007;
str.clear();
}
cout<<ans<<endl;
}
