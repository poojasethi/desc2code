#include<iostream>
using namespace std;
int v[102][30];int a[104];string s[104];
int main(){
	int n,m;
	cin>>n>>m;
	for(int i=0;i<n;i++){
		cin>>s[i];
	}
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			int g=s[i][j]-'A';
			if(v[j][g]==0){v[j][g]=1;a[j]++;
			}
		}
	}long long int q=1,u=1000000007;
	for(int i=0;i<m;i++){
		q=q*a[i];q=q%u;
	}
	cout<<q<<endl;
}
