#include <iostream>
#include <set>
using namespace std;
int main(){
int i,j,n,m;
long long rez = 1;
cin>>n>>m;
string tabela[110];
for(int i=0; i<n;i++){
    cin>>tabela[i];
}
for(int i=0; i<m;i++){
    set<char> tab;
    for(int j =0; j<n;j++)
        tab.insert(tabela[j][i]);
    rez = (rez*tab.size())%1000000007;
}


cout<<rez;
return 0;
}
