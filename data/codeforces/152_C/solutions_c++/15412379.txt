#include <iostream>
#include <set>
#define MOD 1000000007
using namespace std;
long long N,M;
char X[105];
set <char> F[105];
long long rez=1LL;

int main()
{
	cin>>N>>M;
	cin.getline(X,105);
	for(int i=1;i<=N;i++)
	{
		cin.getline(X,105);
		for(int j=0;j<M;j++)
			F[j].insert(X[j]);
	}
	for(int j=0;j<M;j++)
		rez=(rez*F[j].size())%MOD;
	cout<<rez<<'\n';
	return 0;
}
