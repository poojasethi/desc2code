#include<bits/stdc++.h>
using namespace std;
set<char>se[1000000];
int main()
{
    long long ans=1;
    int n,m;
    string s;
    cin>>n>>m;
    for(int i=0;i<n;i++)
    {
        cin>>s;
        for(int j=0;j<m;j++)
            se[j].insert(s[j]);
    }
    for(int i=0;i<m;i++)
    {
        ans=(ans*se[i].size())%1000000007;
    }

    cout<<ans;
}
