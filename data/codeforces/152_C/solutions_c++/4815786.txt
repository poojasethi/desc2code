#include <iostream>
using namespace std;
const long long mod=1000000007;

int main()
{
	int n,m;
	cin >> n >> m;
	bool a[101][26]={};
	for (int i=0;i<n;i++)
	{
		string s;
		cin >> s;
		for (int j=0;j<m;j++)
			a[j][s[j]-'A']=1;
	}
	long long ans=1;
	for (int i=0;i<m;i++)
	{
		int t=0;
		for (int j=0;j<26;j++)
			if (a[i][j]) t++;
		ans*=t;
		ans%=mod;
	}
	cout << ans << endl;
}
