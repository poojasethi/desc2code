#include<iostream>
using namespace std;
int main()
{
	long long int n,m,sum=1,count,i,j;
	cin>>n>>m;
	char c[n][m+1];
	for(i=0;i<n;i++)cin>>c[i];
	for(i=0;i<m;i++)
	{   long long int a[26]={0};count=0;
		for(j=0;j<n;j++)a[c[j][i]-'A']++;
		for(j=0;j<26;j++)if(a[j])count++;
		sum=(sum*count)%1000000007;
	}
	cout<<sum;
}
