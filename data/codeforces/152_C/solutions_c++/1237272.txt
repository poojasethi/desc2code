#include<iostream>
#include<set>
using namespace std;
int main()
{
    int n,m,i,j;long long ans=1;
    cin>>n>>m;string a[120];
    for(i=0;i<n;i++)cin>>a[i];
    for(i=0;i<m;i++)
    {
        set<char>st;
        for(j=0;j<n;j++)
            st.insert(a[j][i]);
        ans=(ans*st.size())%1000000007;
    }
    cout<<ans<<endl;

}
