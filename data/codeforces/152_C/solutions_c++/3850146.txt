#include<iostream>
using namespace std;
int main(){
	int n,m,c[100]={0},i,j,k;
	long long int ans=1;
	bool a[100][26]={0};
	char ch;
	cin>>n>>m;
	for(i=0;i<n;i++){
		for(j=0;j<m;j++){
			cin>>ch;
			k=ch;
			if(a[j][k-65]==0){
				a[j][k-65]=1;
				c[j]++;
			}
		}
	}
	for(k=0;k<m;k++){
		ans=(ans*c[k])%1000000007;
	}
	cout<<ans;
	return 0;
}
