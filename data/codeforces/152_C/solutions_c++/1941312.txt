#include <string>
#include <iostream>
#include <set>

using namespace std;

int n,m;
const int MaxN=103;
string name[MaxN];
long long ans=1;
const int Delta=1000000007;

int main(){
	cin>>n>>m;
	for(int i=0;i<n;++i) cin>>name[i];
	for(int i=0;i<m;++i){
		set<int>s;
		for(int j=0;j<n;++j)
			s.insert(name[j][i]);
		ans=(ans*s.size())%Delta;
	}
	cout<<ans<<endl;
	return 0;
}
