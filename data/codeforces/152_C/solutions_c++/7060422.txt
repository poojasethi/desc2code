#include<iostream>
#include<string>
#include<set>
using namespace std;

const long long M=110,D=1000LL*1000*1000+7;
string s[M];
long long ans=1;
int n,m;

int main()
{
	cin>>n>>m;
	for(int i=0;i<n;i++)
		cin>>s[i];
	for(int i=0;i<m;i++)
	{
		set<char> st;
		for(int j=0;j<n;j++)
			st.insert(s[j][i]);
		ans*=st.size()%D;
		ans%=D;
	}
	cout<<ans<<endl;
	return 0;
}
