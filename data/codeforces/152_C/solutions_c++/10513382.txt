#include <bits/stdc++.h>
#define LL long long
#define MOD 1000000007
using namespace std;
bool cnt[26][110];
LL arr[110];
int main()
{
	int n,m;
	cin>>n>>m;
	string s;
	for(int i=0;i<n;i++)
	{
		cin>>s;
		for(int j=0;j<m;j++)
		{
			if(cnt[s[j]-'A'][j]);
			else
			{
				cnt[s[j]-'A'][j]=true;
				arr[j]++;
			}
		}
	}
	LL ans=1;
	for(int i=0;i<m;i++){
		ans*=arr[i];
		ans%=MOD;
	}
	cout<<ans;
}
