#include <iostream>
#include <algorithm>
#include <set>
using namespace std;

string T[100];
int n, m;

int main()
{
	cin >> n >> m;
	for(int i = 0; i < n; ++i)
		cin >> T[i];
	
	long long res = 1;
	for(int col = 0; col < m; ++col)
	{
		set<char> S;
		for(int word = 0; word < n; ++word)
			S.insert(T[word][col]);
		res = (res * S.size()) % 1000000007;
	}
	cout << res << endl;
	return 0;
}
