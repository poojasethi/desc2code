#include<cstdio>
#include<cstring>
#define oo 1000000007
long long ans=1; int n,m,a[150],c;
char s[105][105];
int main()
{
	scanf("%d%d",&n,&m); gets(s[0]);
	for (int i=0;i<n;i++) gets(s[i]);
	for (int j=0;j<m;j++){
		memset(a,0,sizeof(a)); c=0;
		for (int i=0;i<n;i++) a[s[i][j]]=1;
		for (int i=0;i<128;i++) c+=a[i];
		ans=ans*c%oo;
		}
	printf("%d\n",ans); //for(;;);
}
