#include<bits/stdc++.h>
using namespace std;
int main()
{
long long int n,m,flag[26]={},count=0,ans=1;
char s[101][101];
cin >> n >> m;
long long int i,j;
for(i=0;i<n;i++)
cin >> s[i];
for(i=0;i<m;i++)
{

count =0;
for(j=0;j<n;j++)
{
if(flag[s[j][i]-'A']!=i+1)
flag[s[j][i]-'A']=i+1,count++;
}
ans=(ans*count)%1000000007;
}
cout << ans << endl;
return 0;
}
