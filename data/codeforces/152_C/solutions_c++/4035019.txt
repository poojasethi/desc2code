#include <iostream>
#include <string>
#include <set>
#include <vector>

using namespace std;

typedef long long int ll;

ll modulo=1000000007;

int main()
{
  int n,m;
  cin>>n>>m;
  vector<set<char> > v(m);
  for (int i=0;i<n;i++) {
    string s;
    cin>>s;
    for (int j=0;j<m;j++)
      v[j].insert(s[j]);
  }
  ll r=1;
  for (int j=0;j<m;j++)
    r=r*int(v[j].size())%modulo;
  cout<<r<<endl;
}
