#include<bits/stdc++.h>
#define mod 1000000007
using namespace std;
int main()
{

long long n,m;
cin>>n>>m;

vector<string>v(n);
for(long long i=0;i<n;i++)
    cin>>v[i];
long long ans=1;
for(long long j=0;j<m;j++)
{
    set<char>jarvis;
    for(long long i=0;i<n;i++)
        jarvis.insert(v[i][j]);
    ans=((ans)%mod*(jarvis.size())%mod)%mod;
}

cout<<ans<<endl;
    return 0;
}
