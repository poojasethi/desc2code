#include <iostream>
#include <set>

using namespace std;

int main()
{
	int n,m;
	cin >> n >> m;
	set<char> a[m];
	char c;
	for (int i=0;i<n;i++)
		for (int j=0;j<m;j++)
		{
			cin >> c;
			a[j].insert(c);
		}
	long long ans=1;
	for (int i=0;i<m;i++)
	{
		ans = ((ans%1000000007)*(int)a[i].size())%1000000007;
	}
	cout << ans;
	return 0;
}
