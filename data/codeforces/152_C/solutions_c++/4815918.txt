#include<iostream>

using namespace std;

int main()
{
	int n,m;
	cin>>n>>m;
	string str[n];
	for(int i = 0; i<n; i++)
		cin>>str[i];
		int res = 0;
	long long cnt= 1;
	for(int i = 0; i<m; i++)
	{
		res = 0;
		int ch[26]= {0};
		for(int j = 0; j <n; j++)
		{
			if(ch[str[j][i]-'A'] == 0)
			{
				ch[str[j][i]-'A'] = 1;
				res++;
			}
		}
		cnt = (cnt * res)%1000000007;
	}
	cout<<cnt;
	return 0;
}
