#include<string>
#include<iostream>
#include<vector>
#include<set>

#define M 1000000007LL
using namespace std;

int main()
{
int n,m;
long long int ans=1;
cin>>n>>m;
vector<string> ip(n);
for(int i=0;i<n;i++)cin>>ip[i];
for(int j=0;j<m;j++)
{set<char> no;
for(int i=0;i<n;i++)no.insert(ip[i][j]);
ans=(ans*no.size())%M;
}
cout<<ans<<"\n";
}
