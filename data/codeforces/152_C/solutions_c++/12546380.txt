#include<bits/stdc++.h>
using namespace std;
int main()
{	int a,b;
	int i,j;
	char c;
	cin>>a>>b;
	string str[a];
	set<char> s[b];
	for(i=0;i<a;i++)
	{cin>>str[i];
	}
	for(i=0;i<b;i++)
	for(j=0;j<a;j++)
	  {
	  	s[i].insert(str[j][i]);
	  }
	long long ans=1;
	for(j=0;j<b;j++)
	{
		ans=(ans*(s[j].size())%1000000007);
	}
	cout<<ans<<endl;
}
