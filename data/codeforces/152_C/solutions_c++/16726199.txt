#include<bits/stdc++.h>
using namespace std;


#define mod 1000000007
#define l long long

int main(){
    int n,m;cin>>n>>m;
    
    string str[n];
    for(int i=0;i<n;i++)cin>>str[i];
    
    l ans=1;
    for(int i=0;i<m;i++){
        set<char>temp;
        for(int j=0;j<n;j++)temp.insert(str[j][i]);
        ans*=temp.size();ans%=mod;
    }
    cout<<ans;
}
