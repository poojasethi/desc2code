//In the name of God
#include <algorithm>
#include <iostream>
using namespace std;

int n, m;
char a[200][200];

int main() {
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		for (int j = 0; j < m; j++)
			cin >> a[j][i];
	int ans = 1, mod = 1e9 + 7;
	for (int i = 0; i < m; i++) {
		sort(a[i], a[i] + n);
		ans = 1ll * ans * (unique(a[i], a[i] + n) - a[i]) % mod;
	}
	cout << ans << '\n';
	return 0;
}
