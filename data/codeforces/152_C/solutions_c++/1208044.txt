#include <set>
#include <string>
#include <iostream>
using namespace std;

#define M 1000000007

int main() {
	int n, m;
	string s[100];
	set<char> z;
	long long int r = 1;

	cin >> n >> m;
	for (int i = 0; i < n; ++i)
		cin >> s[i];
	for (int j = 0; j < m; ++j) {
		z.clear();
		for (int i = 0; i < n; ++i)
			z.insert(s[i][j]);
		r = (r*z.size())%M;
	}
	cout << r;
	return 0;
}
