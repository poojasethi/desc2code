#include<cstdio>
using namespace std;
bool tab[100][26];
long long t[100];
int main(){
	int n, m;
	scanf("%d%d", &n, &m);
	char c;
	for(int i=0;i<n;i++){
		scanf("\n");
		for(int j=0;j<m;j++){
			scanf("%c", &c);
			if(!tab[j][c-'A']){
				tab[j][c-'A']=true;
				t[j]++;
			}
		}
	}
	long long w=1;
	for(int i=0;i<m;i++){
		w*=t[i];
		w%=1000000007;
	}
	printf("%d\n", int(w));
	return 0;
}
