#include <bits/stdc++.h>
#define ll long long
using namespace std;
ll mod=1000000007;
int main()
{
	string s;
	ll n,m;
	cin>>n>>m;
	vector <set <char> > a(m);
	for(ll i=0;i<n;i++)
	{
		cin>>s;
		for(ll j=0;j<m;j++)
		{
			a[j].insert(s[j]);
		}
	}
	ll ans=1;
	for(ll i=0;i<a.size();i++)
	{
		//cout<<a[i].size()<<endl;
		ans=((ans%mod)*(a[i].size())%mod)%mod;
	}
	cout<<ans<<endl;
}
