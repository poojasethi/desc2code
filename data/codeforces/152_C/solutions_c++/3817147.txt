#include <stdio.h>
int cnt[128];
bool vis[128][26];
char str[128];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;++i){
		scanf("%s",str);
		for(int i=0;i<m;++i){
			cnt[i]+=!vis[i][str[i]-'A'];
			vis[i][str[i]-'A']=true;
		}
	}
	n=1;
	for(int i=0;i<m;++i)
		n=(long long)n*cnt[i]%1000000007;
	printf("%d",n);
	return 0;
}
