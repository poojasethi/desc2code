inp=raw_input()
m=0
n=0
for i in range(len(inp)):
	if inp[i]==' ':
		n=int(inp[0:i])
		m=int(inp[i+1:len(inp)])
		break
l=list()
for i in range(n):
	l.append(raw_input())
ll=list()
for i in range(m):
	kk=set(l[k][i] for k in range(n))
	ll.append(len(kk))
ans=1
for i in ll:
	ans*=i
print ans%1000000007
