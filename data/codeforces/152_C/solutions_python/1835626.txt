import pdb
import sys

file = sys.stdin
#file = open("test", "r")

n, m = map(int, file.readline().rstrip().split())
p = []
for _ in range(n):
    p.append(file.readline().rstrip())

r = 1
for i in range(m):
    s = set()
    for word in p:
        s.add(word[i])
    r *= len(s)
    r %= 1000000007

print r
