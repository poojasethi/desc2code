#include <iostream>
#include <cmath>
#define MOD 1000000007
using namespace std;
int n,h;
int main()
{
    cin>>n>>h;
    int last = 0;
    int tmp = 0;
    int x;
    long long res = 1;
    for (int i=1; i<=n; i++)
    {
        cin>>x;
        tmp = h - x;
        if (abs(tmp - last)>1) { cout<<0; return 0;}
        if (tmp<=last)
            res = (res*(tmp+1))%MOD;
        last = tmp;
    }
    if (last>=2) cout<<0;
    else
        cout<<res;
    return 0;
}
