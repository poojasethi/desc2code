#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
long long a[2010];
long long ans=1;
long long n,h;
int main()
{
	scanf("%I64d%I64d",&n,&h);
	a[0]=h;
	for(int i=1;i<=n+1;i++)
	{
		if(i<=n)scanf("%I64d",&a[i]);
		else a[i]=h;
		if(a[i]-a[i-1]>(long long)(1) || a[i-1]-a[i]>(long long)(1))
		{
			printf("0\n");
			return 0;
		}
		if(a[i]!=a[i-1]-1) ans=ans*(h-a[i]+(long long)(1))%1000000007;
	}
	cout<<ans<<endl;
	return 0;
}
