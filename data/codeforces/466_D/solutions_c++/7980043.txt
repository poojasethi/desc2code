#include <bits/stdc++.h>
#define maxn (2010) 

using namespace std;
typedef long long LL;
const int MOD = 1000000007;

int n, H;
LL ans(1);
int h[maxn];

int main() {
	ios::sync_with_stdio(false);
	cin >> n >> H;
	for(int i=1; i<=n; i++) {
		cin >> h[i];
		h[i] = H - h[i];
	}
	for(int i=1; i<=n+1; i++) {
		if(h[i]<0) ans = 0;
		else if(h[i] == h[i-1] + 1) ans *= 1;
		else if(h[i] == h[i-1]) ans = ans * (h[i-1] + 1) % MOD;
	   	else if(h[i] == h[i-1] - 1) ans = ans * h[i-1] % MOD;	
		else ans = 0;
	}
	cout << ans << endl;
	return 0;
}
