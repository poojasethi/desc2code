#include<bits/stdc++.h>
using namespace std;

long long a[2020],dp[2020];
long long mod=1e9+7;

int main()
{
	int n,h,i;
	while(cin>>n>>h)
	{
		for(i=1;i<=n;i++)
			scanf("%lld",&a[i]);
		for(i=1;i<=n;i++)
		{
			a[i]=h-a[i];
			if(a[i]<0)
				break;
		}
		if(i<=n||a[0]>1)
		{
			cout<<0<<endl;
			continue;
		}
		a[n+1]=0;
		dp[1]=1;
		for(i=2;i<=n+1;i++)
		{
			if(a[i]==a[i-1]+1)
				dp[i]=dp[i-1];
			if(a[i]==a[i-1])
				dp[i]=dp[i-1]*(a[i-1]+1)%mod;
			if(a[i]==a[i-1]-1)
				dp[i]=dp[i-1]*a[i-1]%mod;
		}
		printf("%lld\n",dp[n+1]);
	}
}
