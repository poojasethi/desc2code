#include<cstdio>
#define N 2100
#define P 1000000007
int n,i,w,m,a[N];
long long f[N][N];
int main()
{
    scanf("%d%d",&n,&m);
    for (i=1;i<=n;i++)
    scanf("%d",&a[i]);
    f[0][0]=1;
    for (i=1;i<=n;i++)
    {
        w=m-a[i];
        if (w<0)
        {
            printf("0");
            return 0;
        }
        if (w>0)
        f[i][w]=(f[i-1][w]+f[i-1][w-1])%P;
        else
        f[i][w]=f[i-1][w];
        
        if (w>0)
        f[i][w-1]=(f[i][w]*w)%P;
    }
    printf("%lld",f[n][0]);
}
