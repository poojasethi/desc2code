#include<cstdio>
#include<cmath>
using namespace std;
const int oo=1000000007;
long long ans,l;
int a,b,last,n,h,i;
int main()
{
	scanf("%d%d",&n,&h);
	ans=1;
	for (i=1;i<=n+1;i++)
	{
		if (i<=n)
		{
			scanf("%d",&a);
			a=h-a;
			b=a-last;
		}
		else b=-last;
		switch(b)
		{
			case 1:l++;break;
			case 0:ans=(ans*(l+1))%oo;break;
			case -1:ans=(ans*l)%oo;l--;break;
			default:ans=0;
		}
		if (ans<=0) break;
		last=a;
	}
	printf("%I64d\n",ans);
	return 0;
}
