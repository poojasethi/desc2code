#include<bits/stdc++.h>
#define MOD 1000000007
using namespace std;
int n,m;
int s[2005];
long long dp[2005][2005];
int main(){
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)scanf("%d",&s[i]);
    if(s[0]==m)dp[0][0]=1;
    else dp[0][0]=dp[0][1]=1;
    for(int i=1;i<n;i++){
        if(m>=s[i]){
            dp[i][m-s[i]]=dp[i-1][m-s[i]]%MOD;
        }
        if(m-s[i]-1>=0){
            dp[i][m-s[i]-1]=(dp[i-1][m-s[i]]*(m-s[i])+dp[i-1][m-s[i]-1]*(m-s[i]))%MOD;
            dp[i][m-s[i]]=(dp[i][m-s[i]]+dp[i-1][m-s[i]-1])%MOD;
        }
    }
    printf("%lld\n",dp[n-1][0]);
}
