#include<stdio.h>
#include<algorithm>
using namespace std;
const int maxn=2050,mod=1e9+7;
int n,h,a[maxn],ans;
long long f[maxn];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
	freopen("data.out","w",stdout);
#endif
	int i,j;
	scanf("%d%d",&n,&h);
	f[0]=1;
	for(i=1;i<=n;i++)
		scanf("%d",&a[i]),a[i]=h-a[i];
	for(i=1,j=0;i<=n+1;i++)
	{
		if(abs(a[i]-a[i-1])>1) f[i]=0;
		else if(a[i]-a[i-1]==1) f[i]=f[i-1],j++;
		else if(a[i]-a[i-1]==-1) f[i]=f[i-1]*(j--)%mod;
		else f[i]=f[i-1]*(j+1)%mod;
	}
	//for(i=1;i<=n+1;i++) printf("%I64d ",f[i]); printf("\n");
	ans=f[n+1];
	printf("%d\n",ans);
	return 0;
}
