#include <iostream>
#define MAXN 2005
#define MOD 1000000007

using namespace std;

int n,h,a;
long long dp[MAXN][MAXN];

int main() {
    cin >> n >> h;
    dp[0][0] = 1;
    for (int i = 1; i <= n; i++) {
        cin >> a;
        if (h - a < 0) {
            cout << 0 << '\n';
            return 0;
        }
        dp[i][h-a] = dp[i-1][h-a];
        if (h-a > 0) {
            dp[i][h-a] += dp[i-1][h-a-1];
            dp[i][h-a] %= MOD;
            dp[i][h-a-1] = dp[i][h-a] * (h-a);
        }
    }
    cout << dp[n][0] << '\n';
}
