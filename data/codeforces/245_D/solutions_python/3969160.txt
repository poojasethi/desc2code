import sys

def solve(lines, output):
    n = int(lines.readline())
    a = [0] * n
    for i in range(n):
        bi = [int(x) for x in lines.readline().split()]
        for j in range(n):
            if i != j:
                a[i] |= bi[j]
    output.write(' '.join(str(x) for x in a) + '\n')

if __name__ == '__main__':
    solve(sys.stdin, sys.stdout)
