#!/usr/bin/env python

if __name__ == "__main__":
    d = int(raw_input())
    row = 0
    lista = "" 
    for x in range(d):
        valores = map(int, raw_input().split())
        number = 0
        col = 0
        for v in valores:
            if row != col:
                number |= v
            col += 1 
        
        row += 1
        lista += str(number)+" "

    print lista[:-1]
