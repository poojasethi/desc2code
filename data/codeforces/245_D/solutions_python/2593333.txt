def set_bits(dest, value):
    return dest - (dest & value) + value

n = int(raw_input())
matrix = [ map(int, raw_input().split()) for i in range(n) ]
ret = [0 for i in range(n)]

for i in range(n):
    for j in range(i+1,n):
        ret[i] = set_bits(ret[i], matrix[i][j])
        ret[j] = set_bits(ret[j], matrix[i][j])
        
print ' '.join(map(str, ret))
