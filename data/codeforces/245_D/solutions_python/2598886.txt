from __future__ import print_function
n = input()
table = [list(map(int,raw_input().split())) for  i in range(n)]
ans =[]
for i in range(n):
    preans = 0
    for j in range(n):
        if table[i][j]!=-1:
            preans = preans|table[i][j]
    ans.append(preans)
for i in ans:
    print(i,end= ' ')
