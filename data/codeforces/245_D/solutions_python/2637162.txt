from math import *
from array import *
import sys
n=int(sys.stdin.readline())
a = [[]*n for x in xrange(n)]
for i in range(0,n):
    s=sys.stdin.readline().split()
    for j in range(0,n):
        a[i].append(int(s[j]))
 
ans=[0 for x in range(n)]
for i in range(0,n):
    temp=0
    for j in range(0,n):
        if i!=j :
            temp=temp|a[i][j]
    ans[i]=temp
print ' '.join([str(x) for x in ans])
