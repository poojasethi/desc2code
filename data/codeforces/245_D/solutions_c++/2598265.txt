#include<iostream>
using namespace std;

int main(){
	
	int rows;
	int num;

	while(cin >> rows){
		
		for(int i = 0; i < rows; i++){
				int temp = 0;
			for(int j = 0; j < rows; j++){
				cin >> num;
				if(num != -1){
					temp |= num;
				}
			}
			cout << temp << " ";
		}
		cout << '\n';
	}
	return 0;
}
