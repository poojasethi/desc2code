#include<stdio.h>
#include<string.h>

int b[105][105],a[105];

int main()
{
	int n,i,j;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		for(j=1;j<=n;j++)
			scanf("%d",&b[i][j]);
	for(i=1;i<=n;i++)
	{
		a[i]=0;
		for(j=1;j<=n;j++) if(i!=j)
			a[i]|=b[i][j];
	}
	for(i=1;i<n;i++) printf("%d ",a[i]);
	printf("%d\n",a[n]);
}
