#include<cstdio>
#include<map>
using namespace std;
typedef long long ll;
map<int,int> m;
int main()
{
    ll n;
    scanf("%I64d",&n);
    ll ans=0;
    while(n--)
    {
        ll x;
        scanf("%I64d",&x);
        for(int i=0; i<=31; i++) ans+=m[(1ll<<i)-x];
        m[x]++;
    }
    printf("%I64d\n",ans);
}
