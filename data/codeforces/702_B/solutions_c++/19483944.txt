#include <iostream>
#include <stdio.h>
#include <map>

using namespace std;

int n , a;
long long ans;
map < int , int > set;

int main()
{
	cin >> n;
	for( register int i = 1 ; i <= n ; i++ )
	{
		scanf( "%d" , &a );
		for( register unsigned int j = 1 ; j <= 2000000000 ; j <<= 1 )
			ans += set[ j - a ];
		set[a]++;
	}
	cout << ans << endl;
	return 0;
}
