#include <bits/stdc++.h>
using namespace std;

map<int, int> m;
int n;
long long ans;

int main(){
    cin>>n;
    for(int i=0; i<n; ++i){
        int a;
        cin>>a;
        for(int j=0; j<31; ++j){
            int t=1LL<<j;
            ans+=m[t-a];
        }
        ++m[a];
    }
    cout<<ans;
}
