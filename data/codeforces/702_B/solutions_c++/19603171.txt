#include<bits/stdc++.h>
#define INT64 "%lld"

using namespace std;


int main(){

	long long int n;
	while(scanf("%lld",&n)!=EOF){
		long long int a[n],r=0;
		for(int i=0;i<n;i++)
			scanf("%lld",&a[i]);
		map<int,int> m;
		for(int i=0;i<n;i++){
			for(int j=0;j<31;j++){
				r += m[(1 << j) - a[i]];
			}
			m[a[i]]++;
		}
		printf("%lld\n",r);
	}

	return 0;
}
