#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
map<ll,ll> cnt;
ll N,A[100005],ans=0;
int main(){
	scanf("%lld",&N);
	for(int i = 0; i<N; i++) scanf("%lld",&A[i]);
	for(int i = 0; i<N;i++){
		ll k  = 1ll<<32;
		while(k>0){
			ans+=cnt[k-A[i]];
			k>>=1;
		}
		cnt[A[i]]++;
	}
	printf("%lld\n",ans);
}
