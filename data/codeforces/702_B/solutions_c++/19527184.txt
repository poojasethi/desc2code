#include<bits/stdc++.h>
//#include<map>
//#include<iostream>
using namespace std;
map <int,int> M;
int main(){
	int n;
	cin>>n;
	int a[n],i=0;
	long long ans=0;
	for(i=0;i<n;i++) cin>>a[i];
	sort(a,a+n);
	for(i=0;i<n;i++){
		for(int j=0;j<32;j++) ans+=M[(1LL<<j)-a[i]];
		M[a[i]]++;
	}
	cout<<"\n"<<ans;
	return 0;
}
