#include<bits/stdc++.h>
using namespace std;
int main()
{
    map<long long,long long> mp;
    long long n,k,i,ans=0,t;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>k;
        for(t=0;t<31;t++)
        {
            if(mp.find((1<<t)-k)!=mp.end())
                ans+=mp[(1<<t)-k];
        }
        mp[k]++;
    }
    cout<<ans;
}
