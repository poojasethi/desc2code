#include <bits/stdc++.h>
using namespace std;

int main()
{
	int x, i, j, y;
	long long count = 0;
	map<int, int> m; 
	scanf("%d", &x);
	for (i=0;i<x;i++)
	{
		scanf("%d", &y);
		for (j=0;j<31;j++) if (m.find((1<<j)-y)!=m.end())count += m[(1<<j)-y];
		m[y]++;
	}
	printf("%lld\n", count);
	return 0;
}
