#include<bits/stdc++.h>
using namespace std;
const int N=2e5+5;
typedef long long LL;
map<int, int> cnt;

int main(){
	int n;
	while(cin >> n){
		LL ans = 0;
		for(int i = 0; i < n; i++){
			int x;
			cin >> x;
			for(LL p = 1; p <= 2e9; p*=2)
				if(p > x)
					ans += cnt[p-x];
			cnt[x]++;
		}
	cout << ans << endl;
	}
	return 0;
}
