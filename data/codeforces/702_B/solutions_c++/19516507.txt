#include <iostream>
#include <map>

using namespace std;

typedef long long int ll;

map<int,int> mapa;

int main()
{
  int n;
  cin>>n;
  ll sol=0;
  for (int i=0;i<n;i++) {
    int x;
    cin>>x;
    for (int b=0;b<31;b++) {
      int p=1<<b;
      if (p>x and mapa.count(p-x))
	sol+=mapa[p-x];
    }
    mapa[x]++;
  }
  cout<<sol<<endl;
}
