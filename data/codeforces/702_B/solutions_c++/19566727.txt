#include<bits/stdc++.h>
using namespace std;
map<int,int>mpp;
long long prm[40],knt=0,x,n;

int main()
{
    prm[0]=1;
    for(int i=1; i<40; i++)prm[i]=prm[i-1]*2;

    cin>>n;
    for(int i=0; i<n; i++)
    {
        cin>>x;
        for(int j=0; j<32; j++)
            knt+=mpp[prm[j]-x];
        mpp[x]++;
    }

    cout<<knt;
}
