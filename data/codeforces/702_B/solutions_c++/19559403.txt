#include <bits/stdc++.h>
using namespace std;

map<int, int> mp;

int main()
{
    int n, x;
    long long ans = 0;
    scanf("%d", &n);
    while (n--)
    {
        scanf("%d", &x);
        for (long long i = 2; i <= 2e9; i <<= 1)
        {
            ans += mp[i - x];
        }
        ++mp[x];
    }
    printf("%lld\n", ans);
    return 0;
}
