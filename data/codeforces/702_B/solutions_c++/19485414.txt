#include<bits/stdc++.h>
#define LL long long
using namespace std;
const int N=100011;
int n;LL a[N];
map<LL,int> H;
int main()
{
	cin>>n;
	LL ans=0;
	for (int i=1;i<=n;i++) cin>>a[i];
	for (int i=1;i<=n;i++)
	{
		for (LL j=2ll;j<=(1ll<<31);j=j*2)
		{
			ans+=H[j-a[i]];
		}
		H[a[i]]++;
	}
	cout<<ans;
}
