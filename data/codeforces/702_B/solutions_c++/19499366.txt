#include <iostream>
#include <cstdio>
#include <map>
using namespace std;
long long n,a[100005],ans=0;
map < long long,int > m;
long long k=1;
int main(){
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%I64d",&a[i]);
	}
	for(int i=0;i<n;i++){
		k=1;
		for(long long j=1;j<=32;j++){
			ans+=m[k-a[i]];
			k*=2;
		}
		m[a[i]]++;
	}
	printf("%I64d",ans);
}
