#include <bits/stdc++.h>
using namespace std;
long long n;
long long ans;
map<long long, long long> f;
int main(){
	scanf("%lld",&n);
	for(long long i = 0, x; i < n; i++){
		scanf("%lld",&x);
		for(long long j = 1; j <= 31; j++){
			if((1 << j) < x) continue;
			ans += f[(1 << j) - x];
		}
		f[x]++;
	}
	printf("%lld", ans);
}
