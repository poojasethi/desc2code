#include <iostream>
#include <vector>
#include <map>

using namespace std;

map <int,int> mp;

int main()
{
 int n;
 cin>>n;
 vector <int> v(n);
 for (int i=0;i<n;i++)
 {
   cin>>v[i];
   mp[v[i]]++;

 }
 long long ans=0;
 for (int i=0;i<n;i++)
 {
   mp[v[i]]--;
   for (long long j=2;j<2e9;j*=2)
    ans+=mp[j-v[i]];
 }
 cout<<ans;
    return 0;
}
