#include<iostream>
#include<map>
using namespace std;
map<int ,int>m;
int main()
{
    int n,a;
    long long num=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a;
        for(int j=1;j<31;j++)
        num+=m[(1<<j)-a];
        m[a]++;
    }
    cout<<num<<endl;
return 0;
}
