#include <iostream>
#include <map>
#include <math.h>
using namespace std;
int main()
{long long int i,j,k,n,s=0,q;
map<long long int,long long int >ab;
cin>>n;
for(i=0;i<n;i++){
        q=1;
    cin>>k;
    if(ab[k]!=0){
        s=s+ab[k];
    }
    for(j=0;j<=32;j++){
            ab[q-k]++;
        q*=2;
    }
}
cout<<s;
return 0;
}
