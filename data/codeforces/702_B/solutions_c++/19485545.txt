#include<bits/stdc++.h>
using namespace std;
main()
{
	int n,i;
	scanf("%d",&n);
	map<long long,long long> keep;
	long long sum,t,ans=0;
	for(i=0;i<n;i++)
	{
		scanf("%I64d",&t);
		for(long long j=1;j<=2*1e9;j*=2)
		{
			long long t1=j-t;
			if(t1>0)ans+=keep[t1];
		}
		keep[t]+=1;
	}
	printf("%I64d\n",ans);
}
