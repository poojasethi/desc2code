#include <iostream>
#include <map>
using namespace std;
map <long long,int> arr;
int main()
{
    int n,a;
    long long ans=0;
    cin >> n;
    while(n--)
    {
        cin >> a;
        for(int i=1; i<=31; i++)
            ans+= arr[(1ll<<i)-a];
        arr[a]++;
    }
    cout << ans;
    return 0;
}
