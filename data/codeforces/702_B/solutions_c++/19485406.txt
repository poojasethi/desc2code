#include <bits/stdc++.h>
using namespace std;;


map<long long ,long long> s;
int main() {
	int n;
	scanf("%d", &n);
	long long ans = 0;
	for (int i = 1; i <= n; ++i) {
		int u;
		cin >> u;
		for (long long j = 1; j <= (1LL<<34); j += j) {
			if (j > u) {
				ans += s[j - u];
			}
		}
		++s[u];
	}
	cout << ans << endl;
}
