#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n;
	cin>>n;
	long long int a[32];
	a[0]=2;
	for(int i=1;i<32;i++)
	{
		a[i]=2*a[i-1];
	}
	long long int sum=0;
	map<int,int> p;
	while(n!=0)
	{
		int k;
		cin>>k;
		for(int i=0;i<32;i++)
		{
			sum +=p[a[i]-k];
		}
		p[k]++;
		n--;
	}
	cout<<sum<<endl;
}
