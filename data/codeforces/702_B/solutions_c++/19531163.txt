#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<iostream>
#include<map>
using namespace std;


map<int,int> M;
long long int ans=0;
int main()
{
	int n;
	scanf("%d",&n);
	while(n--)
	{
		int a;
		cin>>a;
		for(int i=1;i<=31;i++) ans+=M[(1LL<<i)-a];
		M[a]++;
	}
	cout<<ans<<endl;
	return 0;
}
