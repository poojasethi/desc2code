#include <bits/stdc++.h>

using namespace std;
typedef long long ll;

int main(){
    map<int,int>mp;
    int n,x;
    cin >> n;
    ll ans = 0;
    for(int i = 0; i < n; ++i){
        scanf("%d", &x);
        for(int j = 1; j < 31; ++j){
            ans += mp[(1<<j)-x];
        }
        mp[x]++;
    }
    cout << ans << endl;
    return 0;
}
