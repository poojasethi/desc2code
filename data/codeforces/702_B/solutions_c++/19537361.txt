#include <cstdio>
#include <map>
using namespace std;
int main()
{
    map<int, int> m;
    int a, n;
    unsigned long long ans = 0;
    scanf("%d\n", &n);
    for(int i=0;i<n;i++) {
        scanf("%d", &a);
        for(int k=1;k<32;k++)
            ans+=m[(1<<k)-a];
        m[a]++;
    }
    printf("%I64d", ans);
    return 0;
}
