#include<bits/stdc++.h>
using namespace std;
#define ll long long
int main()
{
	int n,i;
	scanf("%d",&n);
	int arr[n];
	for(i=0;i<n;i++)
	scanf("%d",&arr[i]);
	map<ll,int> mp;
	ll j,ans=0;

	for(i=0;i<n;i++)
	{
		for(j=31;j>=0;j--)
		{
		if(((ll)1<<j)>=arr[i])
		ans+=mp[(ll)((ll)1<<j)-arr[i]];
		else break;
		}	
		mp[arr[i]]++;	
	}
		cout<<ans<<endl;
	return 0;
}
