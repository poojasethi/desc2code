x = 2
b = []
for i in xrange(50):
    b.append(x)
    x *= 2
n = input()
a = map(int,raw_input().split())
d = {}
ans = 0
for i in xrange(n) :
    d[a[i]] = d.get(a[i],0) + 1

for i in xrange(n):
    d[a[i]] -= 1
    for j in xrange(len(b)):
        if b[j] - a[i] >= 0 and d.get(b[j] - a[i],-1) != -1:
            ans += d[b[j] - a[i]]
print ans
