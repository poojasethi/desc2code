#!/usr/bin/python

from collections import deque

def ir():
    return int(raw_input())

def ia():
    line = raw_input()
    line = line.split()
    return map(int, line)

n = ir()
a = ia()

d = {}
for e in a:
    d[e] = d.get(e, 0) + 1

ans  = 0
for e in a:
    p = 1
    for k in range(0, 31):
        p *= 2
        rst = p - e
        if rst not in d: continue
        if rst != e: ans += d[p - e]
        else       : ans += d[p - e] - 1

print ans // 2
