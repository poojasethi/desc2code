n = int(raw_input())
a = map(int, raw_input().split())
poweroftwo = [1<<i for i in xrange(1,33)]
differencedict = dict()
paircount = 0
for i in xrange(n):
    paircount += differencedict.get(a[i],0)
    for twoindex in poweroftwo:
        target = twoindex - a[i]
        if target >0:
            if target not in differencedict:
                differencedict[target] = 1
            else:
                differencedict[target]+=1
print paircount
