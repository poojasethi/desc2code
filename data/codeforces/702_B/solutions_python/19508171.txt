n = input()
a = map(int, raw_input().split())
cnt = {}
for i in a:
	if i not in cnt: cnt[i] = 1
	else: cnt[i] += 1
ans = 0
for i in a:
	for p in range(1, 32): 
		cur = 2 ** p
		if cur - i in cnt: 
			#print 'cur=', cur, 'i=', i, 'cur-i=', cur - i
			if cur - i == i:
				ans += cnt[cur - i] - 1
			else: ans += cnt[cur - i]
	cnt[i] -= 1
#print cnt
print ans
