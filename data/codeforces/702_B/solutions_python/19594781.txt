#!/usr/bin/python

import sys

n = int(input())
lst=map(int, raw_input().split())

_dict = {}
for elem in lst: _dict[elem] = _dict.get(elem, 0) + 1

n = len(_dict)
keys = _dict.keys()

answer = 0
for key in _dict.keys():
	if _dict[key] > 1:
		x = key << 1;
		if x != 0 and (x & (~x + 1)) == x:
			answer += _dict[key] * (_dict[key] - 1) / 2

	vx = _dict[key]
	_dict[key] = 0

	for j in range(0, 31):
		v = _dict.get((1 << j) - key, 0)
		if v: answer += vx * v

print answer
