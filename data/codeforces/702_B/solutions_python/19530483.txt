quant = int(raw_input())

num_list = [int(x) for x in raw_input().split()]
num_map = {}
ans = 0
for num in num_list:
  for x in range(1, 31):
    pair = pow(2, x) - num
    if pair in num_map:
      ans += num_map[pair]
  if num in num_map:
    num_map[num] += 1
  else:
    num_map[num] = 1
print ans
