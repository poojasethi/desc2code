#!/usr/bin/python

import sys

n = int(input())
lst=map(int, raw_input().split())

answer = 0

_dict = {}
for elem in lst: _dict[elem] = _dict.get(elem, 0) + 1
n = len(_dict)
keys = _dict.keys()

p2 = {}
for i in range(0, 31):
	p2[i] = 2 ** i

i = 0
while i < n:
	x = keys[i] << 1;
	y = _dict[keys[i]]
	if y > 1 and x != 0 and (x & (~x + 1)) == x:
		answer += y * (y - 1) / 2

	x = keys[i];
	vx = _dict[x]
	_dict[x] = 0

	for j in range(0, 31):
		y = p2[j] - x
		v = _dict.get(y, 0)
		if v:
			answer += vx * v
	i += 1

print answer
