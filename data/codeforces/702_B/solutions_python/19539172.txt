raw_input()
# holding all input integers
array = map(int, raw_input().split())
# a dictionary: <value, occurance>, updated as position moves
d = {}
counter = 0

# pre-compute 2 to the power of 30
test = [2**i for i in xrange(32)]

for val in array:
    for pow2 in test:
        if pow2 >= val:
            if (pow2 - val) in d:
                counter += d[pow2 - val]
    if val in d:
        d[val] += 1
    else:
        d[val] = 1
        
print counter
