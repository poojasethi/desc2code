n=int(raw_input())
l=raw_input().split()
li=[int(i) for i in l]
arr=dict()
tel=0
for i in range(0,n):
    tel+=arr.get(li[i],0)
    for j in range(1,31):
        if (2**j-li[i]) > 0:
            if (2**j-li[i]) not in arr:
                arr[2**j-li[i]]=1
            else:
                arr[2**j-li[i]]+=1
print(tel)
