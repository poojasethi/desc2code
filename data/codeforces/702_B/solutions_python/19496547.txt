from collections import defaultdict
#size = int(raw_input())
n = input()
a = sorted([int(i) for i in raw_input().split(' ')])
d = defaultdict(list)
for i in range(n):
        d[a[i]].append(i)
r=0
for i in range(36):
        t=2**i
        for j in d:
                if  t-j in d:
                        if j==t-j:
                                r+=(len(d[j])-1)*len(d[j])
                        else:
                                r+=len(d[j])*len(d[t-j])
print r/2
