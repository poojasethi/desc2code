#include <bits/stdc++.h>
using namespace std;
int m,s,ar[120],t;
int main()
{
	scanf("%d %d",&m,&s);
	if(m*9<s||!s)
	{
		if(!s&&m==1)
		{
			printf("0 0\n");
		}
		else
		{
			printf("-1 -1\n");
		}
		return 0;
	}
	t=s-1;
	for(int i=m-1;i>=0;i--)
	{
		ar[i]=min(9,t),t-=min(9,t);
		ar[0]++;
	}
	for(int i=0;i<m;i++)
	{
		printf("%d",ar[i]);
	}
	printf(" ");
	for(int i=0;i<m;i++)
	{
		cout << min(9,s),s-=min(9,s);
	}
	printf("\n");
	return 0;
}
