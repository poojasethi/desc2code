#include <bits/stdc++.h>
using namespace std;

int m, s, b[105];
int main() {
	int i, k;
	cin >> m >> s;
	if (!s) {puts(m == 1 ? "0 0" : "-1 -1"); return 0;}
	for (i = 0; i < m; i++) {
		b[m - i - 1] = min(9, s);
		s -= min(9, s);
	}
	if (s) {puts("-1 -1"); return 0;}
	for (k = 0; !b[k]; k++);
	b[0]++; b[k]--;
	for (i = 0; i < m; i++) cout << b[i];
	cout << ' ';
	b[0]--; b[k]++;
	for (i = m - 1; i >= 0; i--) cout << b[i];
}
