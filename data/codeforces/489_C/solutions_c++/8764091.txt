#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
int m,s;
int main(){
	scanf("%d%d",&m,&s);
	if((s<1&&m>1)||s>9*m){
		puts("-1 -1");
		return 0;
	}
	for(int i=m-1,k=s;i>=0;--i){
		int j=max(0,k-i*9);
		if(!j&&i==m-1&&k)j=1;
		printf("%d",j);
		k-=j;
	}
	printf(" ");
	for(int i=m-1,k=s;i>=0;--i){
		int j=min(9,k);
		printf("%d",j);
		k-=j;
	}
	puts("");
	return 0;
}
