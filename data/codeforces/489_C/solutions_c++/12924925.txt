#include<bits/stdc++.h>
using namespace std;
int main()
{
	string x,y;
	int m,s,t;
	cin>>m>>s;
	if(s==0)
	{
		if(m==1)
			cout<<"0 0";
		else
			cout<<"-1 -1";		
	}
	else
	{
		for(int i=0;i<m;i++)
		{
			t=min(s,9);
			y+= t+'0';
			s-=t;
		}
		if(s>0)
		{
			cout<<"-1 -1";	
			return 0;
		}
		x=y;
		int i;
		reverse(x.begin(),x.end());
		for(i=0;x[i]=='0';i++);				
		x[i]--;
		x[0]++;
		cout<<x<<" "<<y;
	}
	return 0;
}
