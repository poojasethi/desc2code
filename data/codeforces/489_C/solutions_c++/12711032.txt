#include <bits/stdc++.h>
using namespace std;

int m, s;
string x, y;
int main() {
	int i;
	cin >> m >> s;
	if (!s) {puts(m == 1 ? "0 0" : "-1 -1"); return 0;}
	for (i = 0; i < m; i++) {
		y += min(9, s) + '0';
		s -= min(9, s);
	}
	if (s) {puts("-1 -1"); return 0;}
	for (i = m - 1; i >= 0; i--) x += y[i];
	for (i = 0; x[i] == '0'; i++);
	x[0]++; x[i]--;
	cout << x << ' ' << y;
}
