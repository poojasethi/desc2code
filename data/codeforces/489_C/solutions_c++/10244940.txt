#include<bits/stdc++.h>
using namespace std;

int main() {
  int m,s;
  scanf("%d%d",&m,&s);
  if(9*m<s||s==0&&m>1) {
    puts("-1 -1");
    return 0;
  }
  for(int i=m-1,k=s;i>=0;i--) {
    int val=max(0,k-i*9);
    if(i==m-1&&val==0&&k) {
      val=1;
    }
    printf("%d",val);
    k-=val;
  }
  printf(" ");
  for(int i=m-1,k=s;i>=0;i--) {
    int val=min(9,k);
    printf("%d",val);
    k-=val;
  }
  return 0;
}
