#include<bits/stdc++.h>
#define ll long long int
using namespace std;
int main() {
int m,s,i;
string x,y;
cin>>m>>s;
if(s==0){ m==1?cout<<"0 0":cout<<"-1 -1"; return 0;}
for(int i=0;i<m;i++){
int t=min(s,9);
y+=t+'0';
s-=t;
}
if (s>0) {cout<<"-1 -1"; return 0;}
for(i=m-1;i>=0;i--) x+=y[i];
for(i=0;x[i]=='0';i++);
x[i]--; x[0]++;
cout<<x<<" "<<y;
return 0;
}
