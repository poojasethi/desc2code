#include <bits/stdc++.h>
using namespace std;

int a[300],b[300];
int main(){
    int n,m; cin>>n>>m;
    if (m==0 && n==1) {cout<<"0 0";return 0;}
    if (m==0 || n*9<m) {cout<<"-1 -1";return 0;}

    int k=m-1;
    for (int i=n;i;i--){a[i]=min(9,k); k-=a[i];}
    a[1]++;
    for (int i=1;i<=n;i++) cout<<a[i];cout<<' ';

    for (int i=1;i<=n;i++){b[i]=min(9,m);m-=b[i];}
    for (int i=1;i<=n;i++) cout<<b[i];
}
