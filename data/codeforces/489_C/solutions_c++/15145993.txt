#include<bits/stdc++.h>
using namespace std;

int m,s;
string maxnum,minnum;

int main()
{
	cin>>m>>s;
	int t=m,k=s;
	if((s>9*m) || (!s && m-1)) {cout<<"-1"<<" "<<"-1";return 0;}
	while(t--)
		if(k>=9)	maxnum+='9',k-=9;
		else 		maxnum += k + '0',k=0;
	
	minnum = maxnum;
	reverse(minnum.begin(),minnum.end());
	int i;
	for(i=0; i<minnum.length(); i++)
		if(minnum[i]!='0')
			break;
	if(i != minnum.length())
		minnum[i]--,minnum[0]++;
	cout<<minnum<<" "<<maxnum;
}
