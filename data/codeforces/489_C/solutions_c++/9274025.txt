#include<cstdio>
using namespace std;
int main()
{
	int a,b,c;
	scanf("%d%d",&a,&b);
	if(!b&&a>1||9*a<b)
	{puts("-1 -1");return 0;}
	c=(b-1)/9;
	for(int i=c;i<a;++i)
		printf("%d",(i==a-1?(b-1)%9:0)+(i==c?1:0));
	for(int i=c;i;--i)
		putchar('9');
	putchar(' ');
	c=b/9;
	for(int i=c;i;--i)
		putchar('9');
	for(int i=c;i<a;++i)
		printf("%d",i!=c?0:b%9);
	puts("");
}
