#include<bits/stdc++.h>
using namespace std;
int main(){
	int m,s;
	cin>>m>>s;
	if((s<1 && m>1) || s>9*m){
		cout<<"-1 -1"<<endl;
		return 0;
	}
	int j=s;
	for(int i=m-1;i>=0;i--){
		int digit=max(0,j-i*9);
		if(digit==0 && i==m-1 && j)digit=1;
		cout<<digit;
		j-=digit;
	}
	cout<<" ";
	j=s;
	for(int i=m-1;i>=0;i--){
		int digit=min(9,j);
		cout<<digit;
		j-=digit;
	}
	cout<<endl;
	return 0;
}
