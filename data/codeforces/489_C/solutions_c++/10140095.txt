#include <bits/stdc++.h>
using namespace std;

int m, s, k;
string x, y;
int main() {
	cin >> m >> s;
	if (!s) {puts(m == 1 ? "0 0" : "-1 -1"); return 0;}
	while (m--) {
		y += min(9, s) + '0';
		s -= min(9, s);
	}
	if (s) {puts("-1 -1"); return 0;}
	x = y;
	reverse(x.begin(), x.end());
	for (; x[k] == '0'; k++);
	x[0]++; x[k]--;
	cout << x << ' ' << y;
}
