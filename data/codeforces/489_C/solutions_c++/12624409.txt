#include<bits/stdc++.h>

using namespace std;

int a,b,i;
string x,y;
int main(){
	cin >> a >> b;
	if(!b){
		if(a==1) cout << "0 0";
		else	cout << "-1 -1";
		return 0;
	}

	for( i=0;i<a;i++){
		y += min(b,9) + '0';
		b -= min(b,9);
	}
	if(b > 0){
		cout << "-1 -1 ";
		return 0;
	}
	for( i = a-1 ;i>=0 ;i--) x += y[i];
	for(i=0;i<a;i++)
		if(x[i] - '0')
			break;
	x[i]--;
	x[0]++;
	cout << x<< ' ' << y << endl;
}
