#include<bits/stdc++.h>
using namespace std;
int m,s,i,t;string x,y;
int main()
{
    cin>>m>>s; if(s==0){cout<<(m==1?"0 0":"-1 -1")<<endl;return 0;}
    for(i=0;i<m;i++)
    {
        t=min(s,9);
        x+=t+'0';
        s-=t;
    }
    if(s>0){cout<<"-1 -1"<<endl;return 0;}
    for(i=m-1;i>=0;i--)y+=x[i];
    for(i=0;y[i]=='0';i++);
    y[i]--,y[0]++;
    cout<<y<<" "<<x<<endl;
}
