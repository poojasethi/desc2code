#include<bits/stdc++.h>
using namespace std;
string big="",small="";
int cnt,sum,i,k;
int main(){
   cin>>cnt>>sum;
   if(sum==0) {puts(cnt==1?"0 0":"-1 -1");return 0;}
   for(i=0;i<cnt;i++){
      k=min(9,sum);
      big+=k+'0';
      sum-=k;
   }
   if(sum!=0) {cout<<"-1 -1\n";return 0;}
   for(i=cnt-1;i>=0;i--) small+=big[i];
   for(i=0;small[i]=='0';i++);
   small[i]--;small[0]++;
   cout<<small<<" "<<big<<endl;
}
