#include<iostream>
#include<cmath>
#include<cstdio>
using namespace std;
int main ()
{
	int m,s,i,k;
	scanf("%d%d",&m,&s);
	if(s<1&&m>1||s>m*9)
		printf("-1 -1\n");
	else 
	{
		for(i=m-1,k=s;i>=0;i--)
		{
			int j=max(0,k-9*i);
			if(j==0&&i==m-1&&k) j=1;
			printf("%d",j);
			k-=j;
		}
		printf(" ");
		for(i=m-1,k=s;i>=0;i--)
		{
			int j=min(9,k);
			printf("%d",j);
			k-=j;
		}
		printf("\n");
	}
	return 0;
}
