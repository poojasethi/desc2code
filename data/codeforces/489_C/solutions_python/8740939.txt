n,m=map(int,raw_input().split())
if n==1: print '-1 -1' if m>9 else str(m)+' '+ str(m)
elif m>9*n or not m: print -1,-1
else:
	def f(i,n,m):
		if i==n:return []
		t=max(m-9*(n-i-1), 0 if i else 1)
		return [t]+f(i+1,n,m-t)
	def g(i,n,m):
		if i==n:return []
		t=min(m, 9)
		return [t]+g(i+1,n,m-t)
	print ''.join(map(str,f(0,n,m))), ''.join(map(str,g(0,n,m)))
