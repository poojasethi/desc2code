m, s = map(int, raw_input().split())
if 9 * m < s or (m > 1 and s == 0):
    print -1, -1
elif m == 1:
    print s, s
else:
    x = 10 ** (m - 1)
    #print "x :"
    #print x
    for i in xrange(s - 1):
        x += 10 ** (i / 9)
        #print x
    y = 0
    #print "Y :"
    for i in xrange(s):
        y += 10 ** (m - 1- i / 9)
        #print y 

    print x, y
