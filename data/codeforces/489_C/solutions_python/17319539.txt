m, s = map(int, raw_input().split(' '))
a, b, p, q = s, s-1, [0] * m, [1] + [0] * (m-1)

if m == 1 and s == 0:
    print 0, 0
    exit()

for i in xrange(m):
    if a >= 9:
        p[i] = 9
        a -= 9
    elif a > 0:
        p[i] = a
        a = 0
    if b >= 9:
        q[-1-i] = 9
        b -= 9
    elif b > 0:
        q[-1-i] += b
        b = 0

p, q = map(str, p), map(str, q)

if a == 0 and b == 0:
    print ''.join(q), ''.join(p)
else:
    print -1, -1
