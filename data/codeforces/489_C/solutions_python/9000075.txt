#!/usr/bin/python
# coding: utf-8

m, s = map(int, raw_input().split())

a, b, c, d = (s - 1) / 9, (s - 1) % 9, s / 9, s % 9

if 9 * m < s or m > 1 and s < 1:
  print -1, -1
elif s < 1:
  print 0, 0
else:
  print 10 ** (m - 1) + (b + 1) * (10 ** a) - 1, ((10 ** c - 1) * 10 + d) * (10 ** (m - c)) / 10
