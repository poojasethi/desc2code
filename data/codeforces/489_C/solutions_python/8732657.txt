def qmax(s,m):
    Max = '9'*(s / 9)
    if len(Max) < m:
        Max += str(s % 9)
        Max += '0'*(m-len(Max))
    return Max
m,s = [int(i) for i in raw_input().split(' ')]
if (s == 0 and m != 1 or m*9 < s):
    print '-1 -1'
else:
    if s >= (m-1)*9:
        Min = str(s - (m-1)*9)
        Min += '9'*(m-1)
    else:
        Min = '1' + qmax(s-1,m-1)[::-1]
    Max = qmax(s,m);
    print Min, Max
