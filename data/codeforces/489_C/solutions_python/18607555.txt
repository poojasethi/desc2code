m,s=map(int,raw_input().split())
if (m>1 and s==0) or s>9*m:
    print -1,-1
else:
    l,h=[],[]
    x,y=m-1,s
    while y>9:
        h+=str(9),
        y-=9
        x-=1
    h+=[str(y)]+['0']*x
    x,y=m-1,s-1
    while y>9 and x>0:
        l+=str(9)
        y-=9
        x-=1
    if x==0:
        l+=str(1+y),
    else:
        l+=[str(y)]+['0']*(x-1)+['1']
    print ''.join(l[::-1]),''.join(h)
