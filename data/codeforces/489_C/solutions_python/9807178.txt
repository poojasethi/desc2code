m, s = map(int, raw_input().strip().split())

if 9*m < s or (m>1 and s == 0):
    print -1, -1
elif m == 1:
    print s, s
else:
    small = 10 ** (m-1)
    for i in xrange(s-1):
        small += 10 ** (i/9)
    large = 0
    for j in xrange(s):
        large += 10 ** (m - 1- j/9)
    print small, large
