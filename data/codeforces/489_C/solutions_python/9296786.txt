m,s=raw_input().split()
m,s=int(m),int(s)
l=[]
sm=[1]+[0]*(m-1)
left=s
while len(l)!=m:
    if left>9:
        l.append(9)
        left-=9
    else:
        l.append(left)
        left=0
i=m-1
while i>=0:
    if sum(sm)==s:
        break
    if sm[i]==9:
        i-=1
    else:
        sm[i]+=1
a1,a2="",""
for x in sm:
    a1+=str(x)
for y in l:
    a2+=(str(y))
if m==1 and s==0:
    print 0,0
elif s<1 or s>9*m:
    print -1,-1
else:
    print a1,a2
