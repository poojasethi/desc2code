m,s = [int(x) for x in raw_input().split()]


mi = 0
si = 0
for i in xrange(m):
    k = 1 if i == 0 and m!=1 else 0
    d = min(max(s-si-9*(m-i-1), k),9)
    mi = mi*10 + d
    si+=d
    if si>s:
        mi = -1
        break

if si <s:
    mi = -1
ma = 0
si = 0
for i in xrange(m):
    d = min(s-si, 9)
    if si == 0 and d==0 and m!=1:
        d = 1
    si+=d
    ma = ma*10+d
    if si>s:
        ma = -1
        break
        
if si <s:
    ma = -1
print "{0} {1}".format(mi,ma)
