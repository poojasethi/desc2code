from math import floor
n,s = map(int,raw_input().split())
if ((n==1) & (s==0)):
    print 0,0
elif ((n > 0) & (s < 1)):
    print -1,-1
elif s > n*9:
    print -1,-1
else:
    res1 = 10**(n-1)
    for i in range(s-1):
        res1+= 10**(i//9)
    res2 = 10**(n-1)
    for i in range(1,s):
        ss = int(floor(i//9))
        #print i,ss
        res2 += 10**(n-1-(ss))

    print res1,int(res2)
