import sys
m, s = map(int, raw_input().split())
b = ""
#http://codeforces.com/problemset/submit

if s == 0:

	if m == 1:
		print "0 0"
		sys.exit()
	else:

		print "-1 -1"
		sys.exit()


for i in range(m):

	temp = min(s, 9)
	b += str(temp)
	s -= temp

if s > 0:

	print "-1 -1"
	sys.exit()

a = list(b[::-1])
j = 0

for i in a:

	if i != "0":
		break
	j += 1

a[j] = str(int(a[j]) - 1)
a[0] = str(int(a[0]) + 1)
print ''.join(a), b
