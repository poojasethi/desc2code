m, s = map(int, raw_input().split(" "))
if m == 1 and s == 0:
  print 0, 0
  exit()
t = s
a, b = '', ''

for i in range(m):
  for j in range(9, -1, -1):
    if s - j >= 0:
      a += str(j)
      s -= j
      break

for i in range(m):
  for j in range(i == 0, 10):
    if (m - i - 1) * 9 >= t - j:
      b += str(j)
      t -= j
      break

if len(a) < m or len(b) < m or s != 0 or t != 0:
  print -1, -1
else:
  print b, a
