n,m = map(int,raw_input().split())



mmax = [9]*n

if sum(mmax) < m : 
	print -1,-1
	exit()
	
for i in range(n-1,-1,-1):
	if ( sum(mmax) > m ):
		mmax[i] -= min(mmax[i],sum(mmax) - m)

if n > 1 : mmin = [1] + [0]*(n-1)
else : mmin = [0]

if sum(mmin) > m : 
	print -1,-1
	exit()
	
for i in range(n-1,-1,-1):
	if ( sum(mmin) < m ):
		mmin[i] += min(9-mmin[i],m - sum(mmin))

		
print ''.join(map(str,mmin)), ''.join(map(str,mmax))
