m,s=map(int,raw_input().split())
import sys
if m==1 and s==0:
    print '0 0'
    sys.exit()
if m*9<s or s==0:
    print '-1 -1'
    sys.exit()
if s%9:
    c=`s%9`+'0'*(m-1-s/9)
else:
    c='0'*(m-s/9)
a='9'*(s/9)+c
b=a[::-1]
if b[0]=='0':
    i=1
    while b[i]=='0':
        i+=1
    b='1'+b[1:i]+`int(b[i])-1`+b[i+1:]
print b,a
