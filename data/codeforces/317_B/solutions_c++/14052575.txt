#include <cstdio>
#include <cstdlib>
const int N=65;
int t,x,y,a[N+N][N+N];
int main(){
	scanf("%d%d",&a[N][N],&t);
	bool flag=1;
	while(flag){
		flag=0;
		for(x=1;x<N+N;x++)
			for(y=1;y<N+N;y++) if(a[x][y]>=4){
				flag=1;
				a[x+1][y]+=a[x][y]/4;
				a[x-1][y]+=a[x][y]/4;
				a[x][y+1]+=a[x][y]/4;
				a[x][y-1]+=a[x][y]/4;
				a[x][y]%=4;
			}
	}
	while(t--){
		scanf("%d%d",&x,&y);
		printf("%d\n",abs(x)<N&&abs(y)<N?a[N+x][N+y]:0);
	}
}
