#include<cstdlib>
#include<cstdio>
#include<cmath>
#define sht 1000
using namespace std;
int jc[2002][2002];

void dfs(int x,int y,int v)
{
  int f=jc[x][y]+=v;
  if(f<4) return;
  jc[x][y]%=4;
  dfs(x,y-1,f/4);
  dfs(x,y+1,f/4);
  dfs(x-1,y,f/4);
  dfs(x+1,y,f/4);
}

int main()
{
   int n,t,a,b;
   scanf("%d %d",&n,&t);
   dfs(sht,sht,n);
   while(t--)
   {
     scanf("%d %d",&a,&b);
     if(abs(a)>sht||abs(b)>sht) printf("0\n");
     else printf("%d\n",jc[a+sht][b+sht]);
   }
   return 0;
}
