#include <cstdio>

int fld[299][299];

int n,t;
int main() {
	scanf("%d%d",&n,&t);
	fld[74][74]=n;
	for(int i=0;i<3000;i++) {
		for(int j=0;j<149;j++) {
			for(int k=0;k<149;k++) {
				if(fld[j][k]>3) {
					fld[j-1][k]+=fld[j][k]/4;
					fld[j+1][k]+=fld[j][k]/4;
					fld[j][k-1]+=fld[j][k]/4;
					fld[j][k+1]+=fld[j][k]/4;
					fld[j][k]%=4;
				}
			}
		}
	}
	for(int i=0;i<t;i++) {
		int vx,vy;
		scanf("%d%d",&vx,&vy);
		if(vx>-70&&vx<70&&vy>-70&&vy<70) {
			printf("%d\n",fld[74+vx][74+vy]);
		} else {
			printf("0\n");
		}
	}
	return 0;
}
