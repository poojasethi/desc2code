#include <algorithm>
#include <iostream>

using namespace std;

const int N=70;
const int di[]={0,0,-1,1},dj[]={1,-1,0,0};

int n,t;
int a[2*N][2*N];

void rec(int i,int j)
{
	int p=a[i][j]/4;
	a[i][j]%=4;
	for(int k=0;k<4;++k)
	{
		int x=i+di[k];
		int y=j+dj[k];
		a[x][y]+=p;
		if(a[x][y]>=4)
			rec(x,y);
	}
}

int main()
{
	ios::sync_with_stdio(0);
	cin>>n>>t;
	a[N][N]=n;
	rec(N,N);
	for(int i=0;i<t;++i)
	{
		int x,y;
		cin>>x>>y;
		if(x>-N && x<N && y>-N && y<N)
			cout<<a[N+x][N+y]<<endl;
		else
			cout<<0<<endl;
	}
// 	for(int i=0;i<2*N;++i)
// 	{
// 		for(int j=0;j<2*N;++j)
// 			cout<<a[i][j]<<" ";
// 		cout<<endl;
// 	}
	return 0;
}
