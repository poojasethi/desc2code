#include<stdio.h>
#define N 1000
int ar[2*N][2*N];
void dfs(int x,int y,int v)
{
    int p=ar[x][y]+=v;
    if(ar[x][y]<4)return;
    ar[x][y]%=4;
    dfs(x+1,y,p/4);
    dfs(x-1,y,p/4);
    dfs(x,y+1,p/4);
    dfs(x,y-1,p/4);
}
main()
{
    int tc,n,t,a,b,i,j;
    scanf("%d%d",&n,&tc);
    dfs(N,N,n);
    for(t=1;t<=tc;t++)
    {
        scanf("%d%d",&a,&b);
        if(a>N || a<-N || b>N || b<-N)printf("0\n");
        else printf("%d\n",ar[a+N][b+N]);
    }
}
