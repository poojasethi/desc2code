#include <iostream>
#include <map>
using namespace std;
#define mp make_pair

int m[1000][1000];

int put_ant(int x, int y)
{
    if(m[x][y]<3)
        m[x][y]++;
    else
    {
        m[x][y]=0;
        put_ant(x+1, y);
        put_ant(x, y+1);
        put_ant(x-1, y);
        put_ant(x, y-1);
    }
}
int main()
{
    int n, t;
    cin>> n>> t;
    for(int i=0; i<n; i++)
        put_ant(500, 500);

    int x, y;
    while(t--)
    {
        cin>> x>> y;
        if(x>=500 || x<-500 || y>=500 || y<-500)
            cout << "0"<< endl;
        else
            cout << m[x+500][y+500]<< endl;
    }
    return 0;
}
