#include<stdio.h>
#define gps 250
int ans[500][500];
void put (int x,int y)
{
   ++ans[x][y];
   while(ans[x][y]>3)
   {
      ans[x][y]-=4;
      put(x,y+1);
      put(x+1,y);
      put(x-1,y);
      put(x,y-1);
   }
}
int main ()
{
    int n,m,x,y;
    scanf("%d %d",&n,&m);
    while(n--)
    {
      put(gps,gps);
    }
    while(m--)
    {
       scanf("%d %d",&x,&y);
       x+=gps;
       y+=gps;
       if(x<500&&x>=0&&y<500&&y>=0)
       printf("%d\n",ans[x][y]);
       else
       printf("0\n");
    }
    return 0;
    
}
