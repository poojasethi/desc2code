#include <iostream>
#include <cstdio>
#include <cstring>
#include <string>
using namespace std;
int nAnt[1005][1005];
void move(int x,int y){
	if(nAnt[x][y] < 3)
	nAnt[x][y] ++;
	else{
		nAnt[x][y] = 0;
		move(x + 1,y);
		move(x,y + 1);
		move(x - 1,y);
		move(x,y - 1);
	}
}
int n,t,i,x,y;
int main(){
	scanf("%d%d",&n,&t);
	for(i = 0;i < n;i++)
	move(500,500);
	for(i = 0;i < t;i++){
		scanf("%d%d",&x,&y);
		if(x < (-400) || x > 400 || y < (-400) || y > 400)
            printf("0\n");
        else
            printf("%d\n",nAnt[x + 500][y + 500]);
	}
	return 0;
}
