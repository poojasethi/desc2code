#include<iostream>
#include<cstdio>
#include<cstring>
#include<queue>
using namespace std;
int G[200][200];
void dfs(int x,int y)
{
    if(G[x][y]<4)return ;
    int d=G[x][y]/4;
    G[x][y]%=4;
    G[x+1][y]+=d;dfs(x+1,y);
    G[x-1][y]+=d;dfs(x-1,y);
    G[x][y+1]+=d;dfs(x,y+1);
    G[x][y-1]+=d;dfs(x,y-1);
}
int main()
{
    int n,t;
    cin>>n>>t;
    G[100][100]=n;
    dfs(100,100);
    while(t--)
    {
        int x,y;
        cin>>x>>y;
        x+=100,y+=100;
        if(x<0||x>=200||y<0||y>=200)cout<<0<<endl;
        else cout<<G[x][y]<<endl;
    }
}
