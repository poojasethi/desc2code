#include <stdio.h>
const int M=1024,C=M/2,Q=(1<<16)-1;
int a[M][M],b[M][M],d,n[Q+1],q[Q+1]={(C<<12)+C},h,k,z,t,u;
inline void go(int x,int y,int w){
  if((a[x][y]+=w)>=4&&b[x][y]!=d)q[k++]=(x<<12)+y,b[x][y]=d,(++k)&=Q;
}
main(){
  scanf("%d%d",&a[C][C],&t);
  if(a[C][C]>3)k=1;
  while(h!=k){
    int w,x,y;
    for(z=h;z!=k;(++z)&=Q)n[z]=a[x=q[z]>>12][y=q[z]&0xfff],a[x][y]=0;
    ++d;
    while(h!=z){
      w=n[h],x=q[h]>>12,y=q[h]&0xfff,(++h)&=Q,u=w>>2;
      go(x-1,y,u),go(x,y+1,u),go(x+1,y,u),go(x,y-1,u),go(x,y,w-(u<<2));
    }
  }
  for(int x,y;t-->0&&2==scanf("%d%d",&x,&y);)printf("%d\n",x>-C&&x<C&&y>-C&&y<C?a[x+C][y+C]:0);
}
