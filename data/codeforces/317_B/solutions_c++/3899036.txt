#include <stdio.h>
int c[3001][3001];
int dx[4] = {0, 1, 0, -1};
int dy[4] = {1, 0, -1, 0};
void dfs(int x, int y){
    if (c[x][y] < 4) return;
    int d = c[x][y] / 4;
    c[x][y] %= 4;
    for (int i = 0; i < 4; i++){
        c[x + dx[i]][y + dy[i]] += d;
        dfs(x + dx[i], y + dy[i]);
    }
}
int main(){
    int n, m, x, y;
    scanf("%d%d", &n, &m);
    c[1500][1500] = n;
    dfs(1500, 1500);
    while (m--){
        scanf("%d%d", &x, &y);
        x += 1500;
        y += 1500;
        if (x < 0 || y < 0 || x > 3000 || y > 3000) printf("0\n");
        else printf("%d\n", c[x][y]);
    }
    return 0;
}
