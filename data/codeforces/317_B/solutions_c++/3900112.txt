#include <stdio.h>
const int M=1024,C=M/2,Q=(1<<16)-1;
int a[M][M],b[M][M],d,n[Q+1],p[Q+1]={C},q[Q+1]={C},h,k,z,t,u;
inline void go(int x,int y,int w){
  if((a[x][y]+=w)>3&&b[x][y]!=d)b[p[k]=x][q[k]=y]=d,(++k)&=Q;
}
main(){
  scanf("%d%d",&a[C][C],&t);
  if(a[C][C]>3)k=1;
  for(int w,x,y;h!=k;){
    for(z=h;z!=k;(++z)&=Q)n[z]=a[x=p[z]][y=q[z]],a[x][y]=0;
    ++d;
    while(h!=z)w=n[h],x=p[h],y=q[h],(++h)&=Q,u=w>>2,go(x-1,y,u),go(x,y+1,u),go(x+1,y,u),go(x,y-1,u),go(x,y,w-(u<<2));
  }
  for(int x,y;t-->0&&2==scanf("%d%d",&x,&y);)printf("%d\n",x>-C&&x<C&&y>-C&&y<C?a[x+C][y+C]:0);
}
