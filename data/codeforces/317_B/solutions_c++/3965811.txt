#include <iostream>
#define max 1000

using namespace std;

int grid[max*2][max*2];
int xx[]{0,1,0,-1};
int yy[]{-1,0,1,0};

void move(int x,int y,int n)
{
    n=grid[x][y]+=n;
    if(grid[x][y]<4)return;
    grid[x][y]%=4;
    for(int i=0;i<4;i++)move(x+xx[i],y+yy[i],n/4);
}

int main()
{
    int n,t;
    long long x,y;

    cin>>n>>t;
    move(max,max,n);

    while(t--){

        cin>>x>>y;
        if(x>max||x<-max||y>max||y<-max)cout<<0<<endl;
        else cout<<grid[max+x][max+y]<<endl;
    }

    return 0;
}
