#include<iostream>

using namespace std;

int arr[1000][1000];


void fill(int x,int y)
{
	if(arr[x][y]<3)
		arr[x][y]++;
	else
	{
		arr[x][y]=0;
		fill(x+1,y);
		fill(x-1,y);
		fill(x,y+1);
		fill(x,y-1);
	}
}

int main()
{
	int n,t;
	cin>>n>>t;
	for(int i=0;i<n;i++)
		fill(500,500);

	int x,y;
	for(int i=0;i<t;i++)
	{
		cin>>x>>y;
		if(x<(-400) || x>400 || y <(-400) || y>400)
			cout<<"0"<<endl;
		else
			cout<<arr[x+500][y+500]<<endl;
	}
	return 0;
}
