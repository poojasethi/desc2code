#include <cstdio>
const int N = 70;
const int dx[4] = {0, 0, 1, -1};
const int dy[4] = {1, -1, 0, 0};
int a[200][200], n, m, i, j, k;
bool ff;
int main(){
	scanf("%d%d", &a[N][N], &m);
	for (ff = 1; ff;){
		ff = 0;
		for (i = 0; i < 140; i++)
			for (j = 0; j < 140; j++)
				if (a[i][j] >= 4){
					ff = 1;
					for (k = 0; k < 4; k++)
						a[i + dx[k]][j + dy[k]] += a[i][j] / 4;
					a[i][j] %= 4;
				}
	}
	while (m--){
		scanf("%d%d", &i, &j);
		i += N; j += N;
		printf("%d\n", i >= 0 && i <= 140 && j >= 0 && j <= 140 ? a[i][j] : 0);
	}
}
