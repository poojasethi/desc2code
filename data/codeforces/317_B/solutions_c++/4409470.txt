#include<stdio.h>
#define gps 250
int ans[500][500];
void put (int x,int y,int s)
{
   ans[x][y]+=s;
   int q;
   if(ans[x][y]>=4)
   {
     q = ans[x][y]/4;
     ans[x][y]%=4;
     put(x+1,y,q);
     put(x,y+1,q);
     put(x-1,y,q);
     put(x,y-1,q);
   }
}
int main ()
{
    int n,m,x,y;
    scanf("%d %d",&n,&m);
    put(gps,gps,n);
    while(m--)
    {
       scanf("%d %d",&x,&y);
       x+=gps;
       y+=gps;
       if(x<500&&x>=0&&y<500&&y>=0)
       printf("%d\n",ans[x][y]);
       else
       printf("0\n");
    }
    return 0;
    
}
