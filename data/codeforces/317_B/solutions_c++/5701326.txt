#include <iostream>

using namespace std;


const int MAXN = 1010;
int a[MAXN][MAXN];

void dfs(int x, int y, int val)
{
	a[x][y] += val;
	if (a[x][y] >= 4)
	{
		int tmp = a[x][y] / 4;
		a[x][y] %= 4;
		dfs(x, y + 1, tmp);
		dfs(x + 1, y, tmp);
		dfs(x, y - 1, tmp);
		dfs(x - 1, y, tmp);
	}
}

int main()
{
	int n, t;
	cin >> n >> t;
	dfs(MAXN / 2, MAXN / 2, n);
	for (int i = 0; i < t; i++)
	{
		int x, y;
		cin >> x >> y;
		x += MAXN / 2;
		y += MAXN / 2;
		if (x < 0 || y < 0 || x >= MAXN || y >= MAXN)
			cout << 0 << endl;
		else
			cout << a[x][y] << endl;
	}
	return 0;
}
