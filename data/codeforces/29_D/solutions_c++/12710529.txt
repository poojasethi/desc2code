#include<bits/stdc++.h>

using namespace std;

vector < int > V[302];

int Pre[302],R[302],Asd[100005],x,y,i,j,k,d;

bool H[302];

void f(int x){

    int i = 0;

    for(; i < V[x].size(); i++)
        if(!Pre[ V[x][i] ] && V[x][i] != 1){
            Pre[ V[x][i] ] = x;
            f(   V[x][i] );
        }
    if(i < 2 && x > 1) k++;
}
void g(int x){

     Asd[d++] = x;

    H[x] = 0;

    int i, s = 0;

    if(x == 1) s--;

    while(s + 1 < V[x].size()){
        for(i = 0; i < V[x].size(); i++)
            if(H[ V[x][i] ] && V[x][i] != Pre[x]){
                g(V[x][i]   );
                s++;
                Asd[d++] =  x;
            }
        if(s + 1 < V[x].size()){
            k++;
            y = R[k];
            while(y != x){
                H[y] = 1;
                y = Pre[y];
                if(!y) { cout << -1; exit(0); }
            }
        }
    }
}
int main(){

    cin >> i;

    while(--i){
        cin >> x >> y;
        V[x].push_back(y);
        V[y].push_back(x);
    }
    f(1);

    for(i=1;i<=k;i++)
        cin >> R[i];

    k = 0;
    g(1);
    for(i=0;i<d;i++) cout << Asd[i] << " ";
    return 0;
}
