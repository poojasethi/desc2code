#include <iostream>
#include <vector>

using namespace std;

#define OO 1000000000
#define FOR(i,a,b) for (int i=a; i<b; i++)
#define pb push_back

int g[ 305 ][ 305 ];
int pa[ 305 ][ 305 ];
vector<int> leaf;

void print(int u, int v){
	if ( u==v ){
	}
	else{
		print( u, pa[u][v] );
		cout << pa[u][v] << " ";
	}
}

int main(){
	int N;
	int u, v;
	
	cin >> N;
	
	FOR(i,1,N+1){
		FOR(j,1,N+1){
			g[i][j]=OO;
			pa[i][j] = -1;
		}
		g[i][i] = 0;
		pa[i][i] = i;
	}
	
	FOR(i,1,N){
		cin >> u >> v;
		g[u][v] = g[v][u] = 1;
		pa[u][v] = u;
		pa[v][u] = v;
	}
	
	FOR(k,1,N+1)
		FOR(i,1,N+1)
			FOR(j,1,N+1){
				if ( g[i][k]+g[k][j] < g[i][j] ){
					g[i][j] = g[i][k]+g[k][j];
					pa[i][j] = pa[k][j];
					//pa[j][i] = pa[k][i];
				}
			}
	
	leaf.pb(1);
	while ( cin >> v ){
		leaf.pb( v );
	}
	leaf.pb(1);
	
	int nmov = 0;
	FOR(i,1,leaf.size()) nmov += g[ leaf[i-1] ][ leaf[i] ];
	if ( nmov>2*N-1 ) cout << -1 << endl;
	else{
		FOR(i,1,leaf.size()) print( leaf[i-1], leaf[i] );
		cout << 1 << endl;
	}
	
	return 0;
}
