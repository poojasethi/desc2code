#include <cstdio>
#include <iostream>
#include <vector>

#define pb push_back

using namespace std;

const int MAXN=303;

int N;

vector<int> wr;
vector<int> way[MAXN];

inline bool dfs( int nd , int tar , int dad ){
	
	if( nd==tar )
		return 1;
	
	for( vector<int>::iterator it=way[nd].begin() ; it!=way[nd].end() ; it++ )
		if( *it!=dad && dfs(*it,tar,nd) ){
			wr.pb(nd);
			return 1;
		}
	
	return 0;
}

int main(){
	
	scanf(" %d",&N);
	
	for( int a,b,i=1 ; i<N ; i++ ){
		scanf(" %d %d",&a,&b);
		way[a].pb(b);
		way[b].pb(a);
	}
	
	int a,b=1;
	wr.pb(1);
	
	while( scanf(" %d",&a)!=EOF ){
		dfs(a,b,0);
		b=a;	
	}
	
	dfs(1,b,0);
	
	if( (int)wr.size()!=2*N-1 ){
		printf("-1\n");
		return 0;
	}
	
	for( vector<int>::iterator it=wr.begin() ; it!=wr.end() ; it++ )
		printf("%d ",*it);
	
	printf("\n");
	return 0;
}
