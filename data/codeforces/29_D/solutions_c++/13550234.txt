#include <bits/stdc++.h>

using namespace std;

typedef vector <int> vi;

const int maxn=310;

int N,yol[maxn][maxn],son[maxn],ar[maxn],cur;
vi ans,komsu[maxn];

vi findguys(int node,int dad){
	vi guys,tmp;
	if(node!=1 && komsu[node].size()==1){
		guys.push_back(node);
		return guys;
	}
	
	for(int i=0;i<komsu[node].size();i++) if(komsu[node][i]!=dad){
		int x=komsu[node][i];
		
		tmp.clear();
		tmp=findguys(x,node);
		
		for(int j=0;j<tmp.size();j++){
			yol[node][tmp[j]]=i;
			guys.push_back(tmp[j]);
		}
	}
	
	son[node]=guys.size();
	
	return guys;
}

int dfs(int node){
	ans.push_back(node);
	if(node==ar[cur]){
		cur++;
		return 1;
	}
	
	int used=0;
	
	while(used<son[node]){
		if(yol[node][ar[cur]]==-1){
			printf("-1\n");
			exit(0);
		}
		
		used+=dfs(komsu[node][yol[node][ar[cur]]]);
		ans.push_back(node);
	}
	
	return used;
}

int main(){
	scanf("%d",&N);
	
	for(int i=1;i<N;i++){
		int a,b;
		scanf("%d %d",&a,&b);
		komsu[a].push_back(b);
		komsu[b].push_back(a);
	}
	
	memset(yol,-1,sizeof yol);
	findguys(1,0);
	
	for(int i=1;i<=son[1];i++)
		scanf("%d",&ar[i]);
	ar[son[1]+1]=1;
	cur=1;
	
	dfs(1);
	
	for(int i=0;i<ans.size();i++)
		printf("%d ",ans[i]);
	puts("");
	
	return 0;
}
