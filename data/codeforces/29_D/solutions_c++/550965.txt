#include<stdio.h>
#include<string.h>
#include<vector>
#include<algorithm>
using namespace std;
vector <int> cnt[500];
int vis[500];
int n;
int p[1000],res[1000];
int top,flag;
void dfs(int st,int ed)
{
	vis[st]=1;
	if(st==ed)
	{
		flag=1;
		return;
	}
	for(int i=0;i<cnt[st].size();i++)
	{
		if(vis[cnt[st][i]])continue;
		res[top++]=cnt[st][i];
		dfs(cnt[st][i],ed);
		if(flag)return;
		top--;
	}
} 


int main()
{
	int i,j;
	int a,b;
	//freopen("in.txt","r",stdin);
	scanf("%d",&n);
	for(i=1;i<n;i++)
	{
		scanf("%d%d",&a,&b);
		cnt[a].push_back(b);
		cnt[b].push_back(a);
	}
	p[0]=1;
	int k=1;
	while(~scanf("%d",&p[k++]));
	k--;
	p[k]=1;
	top=0;
	res[top++]=1;
	for(i=0;i<k;i++)
	{
		memset(vis,0,sizeof(vis));
		flag=0;
		dfs(p[i],p[i+1]);
	}
	//printf("%d\n",top);
	if(top!=2*n-1)
	printf("-1\n");
	else
	{
		printf("%d",res[0]);
		for(i=1;i<top;i++)
		printf(" %d",res[i]);
		puts("");
	}
}
