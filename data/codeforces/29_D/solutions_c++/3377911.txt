#include <iostream>
#include <algorithm>
#include <cmath>
#include <vector>

#define rep(i,n) for(int i=0;i<n;i++)

using namespace std;

int n,t1,t2,f,s;
vector<int> g[310],res;

bool dfs(int x,int y,int pre){
  if(x==y)return 1;
  rep(i,g[x].size()){
    if(g[x][i]==pre)continue;
    if(dfs(g[x][i],y,x)){
      res.push_back(x);
      return 1;
    }
  }
  return 0;
}


int main(){
  cin>>n;

  rep(i,n-1){
    cin>>t1>>t2;
    g[t1].push_back(t2);
    g[t2].push_back(t1);
  }
  f = 1; res.push_back(1);
  while(cin>>s){
    dfs(s,f,0);
    f = s;
  }
  dfs(1,f,0);
  if(res.size()!=2*n-1)cout<<-1<<endl;
  else {
    rep(i,res.size())cout<<res[i]<<" ";cout<<endl;
  }

}
