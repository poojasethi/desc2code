#include<cstdio>
using namespace std;
int p[300001],ans[300001];
int f(int x)
{
    if(!p[x])return x;
    return p[x]=f(p[x]);
}
int main()
{
    int n,m,a,b,c,x;
    scanf("%d%d",&n,&m);
    while(m--)
    {
        scanf("%d%d%d",&a,&b,&c);
        a=f(a);
        while(a<c)
        {ans[a]=c;p[a++]=a+1;a=f(a);}
        ++a;a=f(a);
        while(a<=b)
        {ans[a]=c;p[a++]=a+1;a=f(a);}
    }
    for(int i=1;i<=n;++i)
        printf("%d ",ans[i]);
}
