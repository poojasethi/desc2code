#include <iostream>
#include <set>

using namespace std;

const int MAXN = 300010;

int N, M, l, r, x;
int ans[MAXN];

int main() {
	cin >> N >> M;
	set<int> kn;
	for (int i=1; i<=N; i++) kn.insert(i);
	for (int i=0; i<M; i++) {
		cin >> l >> r >> x;
		set<int>::iterator a = kn.lower_bound(l);
		set<int>::iterator b = kn.upper_bound(r);
		for (set<int>::iterator j = a; j!=b; j++)
			if (*j != x) ans[*j] = x;
		kn.erase(a,b);
		kn.insert(x);
	}
	for (int i=1; i<=N; i++)
		cout << ans[i] << ' ';
}
