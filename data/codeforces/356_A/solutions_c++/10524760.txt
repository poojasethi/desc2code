#include <iostream>
#include <set>
using namespace std;
set<int> s;
int ans[300005];
int main(){
	int n,m;
	cin>>n>>m;
	for (int i=1;i<=n;i++){
		s.insert(i);
	}
	for (int i=1;i<=m;i++){
		int x,y,k;
		cin>>x>>y>>k;
		set<int>::iterator it=s.lower_bound(x);
		while(it!=s.end() && *it<=y){
			if (*it==k){
				it++;
				continue;
			}
			else{
				ans[*it]=k;
				s.erase(it++);
			}
		}
	}
	for (int i=1;i<=n;i++){
		cout<<ans[i]<<" ";
	}
	return 0;
}
