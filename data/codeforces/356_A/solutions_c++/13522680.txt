#include <iostream>
using namespace std;
int parent[301000], cur, v, n, m, i, l, r, x, ans[301000];
int find(int v)
{
	if(parent[v]==v) return v;
	return parent[v] = find(parent[v]);
}
int main(){
	cin>>n>>m;
	for(i=1;i<=n+1;++i)
	parent[i]=i;
	for(i=0;i<m;++i)
	{
		cin>>l>>r>>x;
		cur = find(l);
		while(cur<=r)
		{
			if(cur==x) 
			++cur;
			else 
			{ 
				ans[cur]=x;
				parent[cur] = cur+1;
			}
			cur = find(cur);
		}
	}
	for(i=1;i<=n;++i) 
	cout<<ans[i]<<" ";
}
