#include <stdio.h>
#define M 300015
long fa[M];
long win[M]={0};
long get(long i)
{
	if(fa[i]==i)return i;
	return fa[i]=get(fa[i]);
}
void ser(long q,long z,long x)
{
	long i;
	for(i=get(q);i<=z;i=fa[i]=get(i+1))
	win[i]=x;
}
int main()
{
	long i,n,m;
	long r,l,x;
	scanf("%ld%ld",&n,&m);
	for(i=0;i<n+10;i++)
	{
		fa[i]=i;
	}
	while(m--)
	{
		
		scanf("%ld%ld%ld",&r,&l,&x);
		ser(r,x-1,x);
		ser(x+1,l,x);
	}
	for(i=1;i<=n;i++)
	printf("%ld%c",win[i],i==n?'\n':' ');
	return 0;   
}
