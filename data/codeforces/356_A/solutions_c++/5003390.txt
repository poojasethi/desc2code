#include <cstdio>
#include <iostream>
#include <set>
#include <vector>
using namespace std;

int n, m, l, x, r;
set<int> s;
int ans[300005]={0};

int main()
{
	cin>>n>>m;
	for(int i=0; i<=n+1; i++) s.insert(i);
	while(m--)
	{
		cin>>l>>r>>x;
		vector<set<int>::iterator> v;
		for(set<int>::iterator i=s.lower_bound(l); i!=s.end() && *i<=r; i++)
		{
			if(*i!=x) ans[*i]=x, v.push_back(i);
		}
		for(int i=0; i<v.size(); i++) s.erase(v[i]);
	}
	for(int i=1; i<=n; i++) cout<<ans[i]<<" "; cout<<endl;
	return 0;
}
