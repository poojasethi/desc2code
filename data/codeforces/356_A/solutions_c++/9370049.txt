#include<bits/stdc++.h>
using namespace std;
int main()
{
set <int >a;
int n,m;
cin>>n>>m;
int res[n+1];
for(int i=1;i<=n;i++)a.insert(i);
for(int i=0;i<m;i++)
{
int l,r,x;
cin>>l>>r>>x;
set<int >::iterator u=a.lower_bound(l);
set<int >::iterator v=a.upper_bound(r);
for(set<int >::iterator k=u;k!=v;++k)
{
res[*k]=x;
}
a.erase(u,v);
a.insert(x);
}
for(int i=1;i<=n;i++)
{
    if(res[i]==i)cout<<"0 ";
    else cout<<res[i]<<" ";
}
}
