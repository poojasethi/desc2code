#include <cstdio>
#include <iostream>
#include <algorithm>
using namespace std;
int N,M;
int ans[300011],last[300011],next[300011];
int main(){
	scanf("%d%d", &N, &M);
	for (int i = 1;i <= N;i++)next[i] = i + 1;
	while (M--){
		int l,r,x;
		scanf("%d%d%d", &l, &r, &x);
		for (int i = l, k; i <= r;i = k){
			if (i!=x&&ans[i] == 0)ans[i] = x;
			k = next[i];
			if (i < x)next[i] = x;
			if (i > x)next[i] = r + 1;
		}
	}
	for (int i = 1;i <= N;i++)printf("%d ", ans[i]);
	return 0;
}
