#include <iostream>
#include <cstdio>
using namespace std;
int p[300100],ans[300100];
int FIND(int x)
{
	return (p[x]==x?x:p[x]=FIND(p[x]));
}
int main()
{
	int n,m,l,r,x;
	scanf("%d%d",&n,&m);
	for (int i=1; i<=n+1; i++)
	{
		p[i]=i;
	}
	for (int i=1; i<=m; i++)
	{
		scanf("%d%d%d",&l,&r,&x);
		for (int j=FIND(l); j<=r; j=FIND(j+1))
		{
			if (j!=x) ans[j]=x,p[j]=j+1;
		}
	}
	for (int i=1; i<=n; i++)
	{
		printf("%d ",ans[i]);
	}
	return 0;
}
