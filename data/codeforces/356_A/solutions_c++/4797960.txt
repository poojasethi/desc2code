#include <iostream>
#include <set>
using namespace std;
set<int> st;
set<int>::iterator it;
int ans[300001];
int main(){
	int n,m;
	cin >> n >> m;
	for(int i=1;i<=n;i++) st.insert(i);
	for(int i=0;i<m;i++){
		int l,r,x;
		cin >> l >> r >> x;
		it=st.lower_bound(l);
		while(it!=st.end()){
			if(*it>r) break;
			if(*it==x){
				it++;
				continue;
			}
			ans[*it]=x;
			st.erase(it++);
		}
	}
	for(int i=1;i<=n;i++) cout << ans[i] << " ";
	return 0;
}
