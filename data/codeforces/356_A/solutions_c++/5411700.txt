#include <iostream>

using namespace std;
int n,k;
int m[9000000]={0};
int ans[9000000]={0};
int main()
{
    cin >> n >> k;
    for (int i=0;i<k;i++)
    {
    int b,mi,e,to;
    cin  >> b >> e >> mi;
    for (int j=b;j<=e;j++)
    {
    if (j<mi) to=mi; else to=e;
    if (m[j]==0) {ans[j]=mi;m[j]=to;}
    else {int h=j;j=m[j]-1;m[h]=to;}
    }
  if (m[e]==e)  m[e]++;
  m[mi]=0;
  ans[mi]=0;
    }
for (int i=1;i<=n;i++)
cout << ans[i] << " ";
}
