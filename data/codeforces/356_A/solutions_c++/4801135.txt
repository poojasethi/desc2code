#include<stdio.h>
#include<string.h>
#include<set>

using namespace std;

int ans[310000];

int main(){
	int n,m,i;
	scanf("%d%d",&n,&m);
	int l,r,x;
	set<int> ST;
	set<int>::iterator L,R,t;
	for(i=1;i<=n;i++) ST.insert(i);
	for(i=1;i<=m;i++){
		scanf("%d%d%d",&l,&r,&x);
		L=ST.lower_bound(l);
		R=ST.upper_bound(r);
		while(L!=R){
			t=L;L++;
			if((*t)!=x) ans[*t]=x,ST.erase(t);
		}
	}
	for(i=1;i<n;i++) printf("%d ",ans[i]); printf("%d\n",ans[i]);
	return 0;
}
