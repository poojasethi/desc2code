#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>
using namespace std;
int i,j,fa[310000],l,r,x,n,m,ans[310000];
int get(int x){
	return (fa[x]==x?x:(fa[x]=get(fa[x])));
}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n+1;++i)fa[i]=i;
	for(i=1;i<=m;++i){
		scanf("%d%d%d",&l,&r,&x);
		for(j=get(l);j<=r;j=get(j)){
			if(j!=x){
				ans[j]=x;
				fa[j]=j+1;
			}else j++;
		}
	}
	for(i=1;i<=n;++i)if(fa[i]==i)printf("0 ");
	else printf("%d ",ans[i]);
}
