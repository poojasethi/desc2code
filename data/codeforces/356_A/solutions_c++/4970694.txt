#include <iostream>
using namespace std;
int Next[301000], cur, v, n, m, i, l, r, x, ans[301000];
int getNext(int v){
	if(Next[v]==v) return v;
	return Next[v] = getNext(Next[v]);
}
int main(){
	cin>>n>>m;
	for(i=1;i<=n+1;++i) Next[i]=i;
	for(i=0;i<m;++i){
		cin>>l>>r>>x;
		cur = getNext(l);
		while(cur<=r){
			if(cur==x) ++cur;
			else { ans[cur]=x; Next[cur] = cur+1; }
			cur = getNext(cur);
		}
	}
	for(i=1;i<=n;++i) cout<<ans[i]<<" ";
}
