#include <cstdio>
#include <set>

using namespace std;

set<int> S;
int N,M;
int sol[300005];

int main(){
  scanf("%d %d",&N,&M);
  for(int c = 1 ; c <= N ; c++)
    S.insert(c);
  for(int c = 1 ; c <= M ; c++){
    int x,y,z;
    scanf("%d %d %d",&x,&y,&z);
    for(set<int>::iterator it = S.lower_bound(x) ;
	it != S.end() and *it <= y ;){
      if(*it == z){
	it++;
	continue;
      }else{
	sol[*it] = z;
	S.erase(it++);
      }
    }
  }
  for(int c = 1 ; c <= N ; c++)
    printf("%d ",sol[c]);
  printf("\n");
}
