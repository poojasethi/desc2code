#include<stdio.h>
int n,m,r[300300],t[300300];
int rt(int x){return x==r[x]?x:(r[x]=rt(r[x]));}
main(){
  scanf("%d%d",&n,&m);
  for(int i=0;i<=n+1;++i)r[i]=i;
  for(int i=0,a,b,c;i<m&&3==scanf("%d%d%d",&a,&b,&c);++i){
    while((a=rt(a))<=b){
      if(a!=c)t[a]=c,r[a]=a+1;
      ++a;
    }
  }
  for(int i=1;i<=n;++i)printf("%d ",t[i]);
}
