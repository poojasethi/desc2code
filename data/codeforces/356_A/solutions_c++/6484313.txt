#include <cstdio>
int head[300001],ans[300001];
int findhead(int x){
   if(x==head[x]) return x;
   return head[x]=findhead(head[x]);
}
int main(){
    int n,m,l,r,x,u,v;
    scanf("%d%d",&n,&m);
    for(int i=1;i<=n+1;i++) head[i]=i;
    for(int i=0;i<m;i++){
         scanf("%d%d%d",&l,&r,&x);
         while((l=findhead(l))<=r){

            if(l!=x) {ans[l]=x,head[l]=l+1; }
            l++;
         }
    }
    for(int i=1;i<=n;i++)
         printf("%d ",ans[i]);
   return 0;
}
