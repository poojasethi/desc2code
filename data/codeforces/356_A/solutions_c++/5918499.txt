#include <iostream>
#include <set>

using namespace std;

int N,M,ret[300010];
set<int> K;

void update(int l,int r,int x)
{
	for (set<int>::iterator it=K.lower_bound(l);it!=K.upper_bound(r);)
	{
		if (*it!=x)
		{
			ret[*it]=x;
			K.erase(it++);
		}
		else it++;
	}
}

int main()
{
	cin >> N >> M;
	for (int i=1;i<=N;i++) K.insert(i);
	for (int i=0;i<M;i++)
	{
		int l,r,x;
		cin >> l >> r >> x;
		update(l,r,x);
	}
	for (int i=1;i<=N;i++) cout << ret[i] << " ";
	return 0;
}
