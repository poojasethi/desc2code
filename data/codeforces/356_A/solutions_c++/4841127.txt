#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <algorithm>
using namespace std;
int N,M,l,r,x;
int f[400000],s[400000];
int find(int x){if (f[x]==x)return x;return f[x]=find(f[x]);}
int main(){
	scanf("%d%d",&N,&M);
	for (int i=1;i<=N+1;i++)f[i]=i;
	for (int i=1;i<=M;i++){
		scanf("%d%d%d",&l,&r,&x);
		for (int j=find(l);j<x;j=find(j+1)){s[j]=x;f[j]=j+1;}
		for (int j=find(x+1);j<=r;j=find(j+1)){s[j]=x;f[j]=j+1;}
	}
	for (int i=1;i<=N;i++)printf("%d ",s[i]);
}
