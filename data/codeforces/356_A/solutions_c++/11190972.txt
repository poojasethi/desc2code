#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
#include <set>

using namespace std;

set<int> q;
int n,m,l,r,x;
int a[300005];
int main()
{
	scanf("%d%d",&n,&m);
	for (int i=1;i<=n;++i) q.insert(i);
	for (int i=1;i<=m;++i)
	{
		scanf("%d%d%d",&l,&r,&x);
		for (set<int>::iterator j=q.lower_bound(l),k;j!=q.end()&&*j<=r;j=k)
		{
			k=j;k++;
			if (*j!=x) a[*j]=x,q.erase(j);
		}
	}
	for (int i=1;i<=n;++i) cout<<a[i]<<" ";
	return 0;
}
