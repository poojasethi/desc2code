#include<iostream>
using namespace std;
int g[300100],res[300100];
int find(int x)
{
	return g[x]==x ? x : g[x]=find(g[x]);
}
int main()
{
	ios::sync_with_stdio(false);
	int n,m;
	cin>>n>>m;
	for (int i=1;i<=n+1;i++)
		g[i]=i;
	for (int i=1;i<=m;i++)
	{
		int l,r,x;
		cin>>l>>r>>x;
		for (int j=find(l);j<=r;j=find(j+1))
		{
			if (j!=x) 
			{
				res[j]=x;
				g[j]=j+1;
			}
		}
	}
	for(int i=1;i<=n;i++)
		cout<<res[i]<<' ';
        //system("pause");
	return 0;
}
