#include<iostream>
#include<set>
#include<algorithm>
using namespace std;

int ans[300001];
set<int> s;

int main()
{
	int n,m;
	cin>>n>>m;
	for(int i=1;i<=n+10;i++)
	  s.insert(i);
	set<int> :: iterator it,k;
	for(int i=1;i<=m;i++)
	{
		int x,y,z;
		cin>>x>>y>>z;
		for(it=s.lower_bound(x);(it!=s.end())&&(*it<=y);it=k){
			k=it;k++;
			if(*it!=z) {ans[*it]=z;s.erase(it);}
		}
	}
	for(int i=1;i<=n;i++)
	   cout<<ans[i]<<" ";
	return 0;
}
