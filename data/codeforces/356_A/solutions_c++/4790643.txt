#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=1;i<=n;i++)
const int N=300005;
int n,m,x,y,z,next,f[N],a[N];
int get(int x){return !f[x]?x:f[x]=get(f[x]);}
int main()
{
	scanf("%d%d",&n,&m);
	rep(j,m){
		scanf("%d%d%d",&x,&y,&z),next=get(y+1);
		for (int i=get(x);i<=y;i=get(i+1)){
			if (i!=z) a[i]=z;
			if (i<z) f[i]=z;
			if (i>z) f[i]=next;
			}
	}
	rep(i,n) printf("%d ",a[i]); puts("");
	return 0;
}
