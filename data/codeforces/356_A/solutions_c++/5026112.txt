#include<stdio.h>
int a[300005];
int ne[300005];
int getn(int c)
{
	if(ne[c]==c)
		return c;
	else 
		return ne[c]=getn(ne[c]);
}
int main()
{
	int n,m,l,r,x;
	scanf("%d %d",&n,&m);
	for(int i=1;i<=n;i++)
		ne[i]=i;
	ne[n+1]=n+1;
	for(int i=0;i<m;i++)
	{
		scanf("%d %d %d",&l,&r,&x);
		int c=getn(l);
		while(c<=r)
		{
			if(c==x)
				c++;
			else
			{
				a[c]=x;
				ne[c]=c+1;
			}
			c=getn(c);	
		}
	}
	for(int i=1;i<=n;i++)
		printf("%d ",a[i]);
	printf("\n");
	return 0;
}
