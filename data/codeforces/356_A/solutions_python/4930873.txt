I=lambda:map(int, raw_input().split())
n,m=I()
R=['0']*(n+1)
g=[i+1 for i in xrange(n+1)]
for _ in xrange(m):
    l,r,x=I()
    i=l
    while i <= r:
        if R[i]=='0' and i != x: R[i] = `x`
        t = g[i]
        if i < x:
            g[i] = x
        else:
            g[i] = r+1
        i=t
        
print ' '.join(R[1:])
