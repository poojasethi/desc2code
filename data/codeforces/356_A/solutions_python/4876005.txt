def main():
  R=lambda:map(int,raw_input().split())

  n,m = R()
  res = [0]*(n+1)
  t = [x+1 for x in xrange(0,n+2)]
  for _ in xrange(m):
    l,r,x = R()
    j = l
    while j <=r:
      if j!=x and res[j]==0: res[j]=x
      tmp = t[j]
      if j < x:
        t[j] = x
      else:
        t[j]=r+1
      j = tmp
  print ' '.join(map(str,res[1:]))

main()
