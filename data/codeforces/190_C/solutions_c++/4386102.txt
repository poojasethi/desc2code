#include <iostream>
#include <cstring>
using namespace std;
bool judge=true;
string ans;
int n;
void dfs()
{
	string str;
	if(cin>>str)
	{
		ans+=str;
		if(str=="pair")
		{
			ans+="<";
			dfs();
			ans+=",";
			dfs();
			ans+=">";
		}
	}
	else
		judge=false;
}
int main()
{
	cin>>n;
	dfs();
	string a;
	if(cin>>a || !judge)
		cout<<"Error occurred"<<endl;
	else
		cout<<ans<<endl;
	return 0;
}
