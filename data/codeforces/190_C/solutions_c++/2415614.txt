#include<iostream>
#include<string>

using namespace std;

string str;
string ans;

bool Fun()
{
	if(cin>>str)
	{
		ans+=str;
		if(str=="pair")
		{
			ans+='<';
			if(Fun()==false) return false;
			ans+=',';
			if(Fun()==false) return false;
			ans+='>';
		}
		return true;
	}
	return false;
}
int main()
{
	ans="";
	cin>>str;
	if(Fun()==false || cin>>str) cout<<"Error occurred"<<endl;
	else cout<<ans<<endl;
	return 0;
}
