import sys

n = int(raw_input())
tokens = raw_input().split()

sym = []
s = ''
cnt = 0
processed = 0
for token in tokens:
  processed += 1
  if token == 'pair':
    s += 'pair<'
    sym += ['>',',']
  else:
    s += 'int'
    cnt += 1
    while len(sym):
      x = sym.pop()
      s += x
      if x == ',':
        break
    if len(sym) == 0:
      break

print s if cnt == n and len(sym) == 0 and processed == len(tokens) else 'Error occurred'
