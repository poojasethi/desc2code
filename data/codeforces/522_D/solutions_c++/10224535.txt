map <int,int> m;
int n,test,r1,r2,a[500001],mi[500001],t[2000001],l,r,p;
void build (int a[], int v, int tl, int tr) {
	if (tl == tr)
		t[v] = a[tl];
	else {
		int tm = (tl + tr) / 2;
		build (a, v*2, tl, tm);
		build (a, v*2+1, tm+1, tr);
		t[v] =min( t[v*2],t[v*2+1]);
	}
}
int mii (int v, int tl, int tr, int l, int r) {
	if (l > r)
		return INT_MAX/3;
	if (l == tl && r == tr)
		return t[v];
	int tm = (tl + tr) / 2;
	return min(mii(v*2, tl, tm, l, min(r,tm)),mii(v*2+1, tm+1, tr, max(l,tm+1), r));
}
int main()

for (int i=1;i<=n;i++) {scanf("%d",&a[i]);if (m[a[i]]==0) mi[i]=INT_MAX/3;else mi[i]=i-m[a[i]]; m[a[i]]=i;}
build(mi,1,1,n);
for (int i=1;i<=test;i++) {scanf("%d %d",&l,&r);p=mii(1,1,n,l,r);
while (p!=mii(1,1,n,l+p,r) && p!=INT_MAX/3) {p=mii(1,1,n,l+p,r);} if (p==INT_MAX/3) printf("%d\n",-1); else printf("%d\n",p);}

    return 0;
}
