n = int(raw_input())
nums = [int(x) for x in raw_input().split()]

min_l = []
min_v = 1 << 31
for i, v in enumerate(reversed(nums)):
    if v < min_v:
        min_v = v
    min_l.append(min_v)

min_l.reverse()

for i, v in enumerate(nums):
    l = i + 1
    r = n - 1
    ans = -1
    while l <= r:
        mid = (l + r) >> 1
        if min_l[mid] < v:
            ans = mid - i - 1
            l = mid + 1
        else:
            r = mid - 1
    print ans,
