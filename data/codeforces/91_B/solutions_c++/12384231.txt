#include<bits/stdc++.h>
using namespace std;
pair<int,int>a[100005];
int k[100005];
int main(){
	int n;
	scanf("%d",&n);
	for(int i=1;i<=n;i++){
		scanf("%d",&a[i].first);
		a[i].second=i;
	}
	sort(a+1,a+n+1);
	int ff=a[1].second;
	k[a[1].second]=0;
	for(int i=2;i<=n;i++){
		if(a[i].second>ff){
			ff=a[i].second;
			continue;
		}
		else{
			k[a[i].second]=ff-a[i].second;
		}
	}
	printf("%d",k[1]-1);
	for(int i=2;i<=n;i++)
		printf(" %d",k[i]-1);
	printf("\n");
	return 0;
}
