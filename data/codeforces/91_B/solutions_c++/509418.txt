#include <iostream>
#include <algorithm>

using namespace std;

const int MAXN = 1e5 + 10;
pair<int, int> a[MAXN];
int b[MAXN];

int main() {
	int n;
	cin >> n;
	for (int i = 0; i < n; ++i) {
		cin >> a[i].first;
		a[i].second = i;
	}

	sort(a, a + n);
	int x = -1;
	for (int i = 0; i < n; ++i) {
		b[a[i].second] = max(-1, x - a[i].second - 1);
		x = max(x, a[i].second);
	}

	for (int i = 0; i < n; ++i) {
		cout << b[i] << ' ';
	}

	return 0;
}
