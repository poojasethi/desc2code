#include<iostream>
#include<algorithm>
using namespace std;
const int maxn = 100*1000+100;
int n;
pair<int,int> a[maxn];
int ans[maxn];

int main()
{
	cin >> n;
	for(int i=0;i<n;i++)
	{
		cin >> a[i].first;
		a[i].second = i;
	}
	sort(a,a+n);
	int x=-1;
	for(int i=0;i<n;i++)
	{
		ans[a[i].second] = max(-1,x-a[i].second-1);
		x = max(x,a[i].second);
	}
	
	for(int i=0;i<n;i++)
		cout << ans[i] << " ";
	cout << endl;
	return 0;
}
