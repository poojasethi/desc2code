#include <iostream>
#include <algorithm>
using namespace std;

int n, a[100100], ans[100100];
int b[100100];

bool cmp0(int a1, int b){
  return a[a1]<a[b];
}

int main(){
  cin>>n;
  for (int i=0; i<n; ++i)
    cin>>a[i], b[i]=i;
  int last=-1;
  int lastv=-1;
  int nlast=-1;
  sort(b, b+n, cmp0);
  for (int i=0; i<n; ++i){
    if (lastv!=a[b[i]]) lastv=a[b[i]], last=max(last, nlast);
    nlast=max(nlast, b[i]);
    ans[b[i]]=last-b[i]-1;
    if (ans[b[i]]<0) ans[b[i]]=-1;
  }
  for (int i=0; i<n; ++i)
    cout<<ans[i]<<' ';
}
