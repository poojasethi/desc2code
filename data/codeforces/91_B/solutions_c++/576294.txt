#include<iostream>
#include<algorithm>

using namespace std;

pair<int,int> arr[100001]={};
int ans[100001]={};

int main()
{
	int n,h = -1;

	cin>>n;

	for(int i = 0 ; i < n ; ++i )
	{
		cin>>arr[i].first;
		arr[i].second = i;
	}

	sort(arr,arr+n);
	for(int i = 0 ; i < n ; ++i )
	{
		ans[ arr[i].second ] = max(-1, h - arr[i].second-1);
		h = max(h , arr[i].second);

//		cout<<h<<' ';
	}//cout<<endl;


	for(int i = 0 ; i < n ; ++i )
	{
		cout<<ans[i]<<' ';
	}cout<<endl;
	
}
