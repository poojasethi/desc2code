#include <stdio.h>
#include <string.h>
#include <algorithm>
#define maxn 100010
#define INF 1000000100

using namespace std;

int a[maxn];
int minA[maxn];
int n;
int main(){
	scanf("%d",&n);
	for (int i=0;i<n;i++) scanf("%d",a+i);
	minA[n]=INF;
	for (int i=n-1;i>=0;i--) minA[i]=min(minA[i+1],a[i]);
	for (int i=0;i<n;i++){
		int k=-1;
		int L=i+1,R=n,mid;
		while (L<=R){
			mid=(L+R)>>1;
			if (minA[mid]<a[i]){
				L=mid+1;
				k=mid-i-1;
			}else R=mid-1;
		}
		printf("%d ",k);
	}
	return 0;
}
