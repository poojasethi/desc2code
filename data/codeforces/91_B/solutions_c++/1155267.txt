#include<iostream>
using namespace std;
int i,j,n;
int a[100001],f[100001];

int find(int p) {
int u,v,mid;
	u=p+1; v=n+1;
	while (u<v){
		mid=(u+v)/2;
		if (f[mid]>=a[p]) v=mid;
		else             u=mid+1;
	}
	if (f[u]<a[p])  return  u;
	else			return  u-1;
}
int main(){	
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>a[i];
		f[i]=-1;
	}
	f[n]=a[n];
	f[n+1]=1000000001;
	for (i=n-1;i>1;i--){
		if (a[i]<f[i+1]) 	f[i]=a[i];
		else				f[i]=f[i+1];
	}
	
	for (i=1;i<=n;i
	++)
		cout<<find(i)-i-1<<' ';
	return 0;
}
