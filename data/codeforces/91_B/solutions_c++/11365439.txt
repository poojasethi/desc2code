#include<stdio.h>
#define min(x,y) x<y?x:y
int a[100000];
int b[100000];
int n;
int solve(int z,int zz){
	int x=z,y=n-1;
	while(x<y){
		int t=y-(y-x)/2;
		if(b[t]<zz) x=t;
		else y=t-1;
	}return x;
}
int main(void){
	int m=1000000000;
	scanf("%d",&n);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i]); 
	}for(int i=n-1;i>=0;i--){
		m=min(m,a[i]);
		b[i]=m;
	}for(int i=0;i<n;i++){
		printf("%d%c",solve(i,a[i])-i-1,i==n-1?'\n':' ');
	}
	return 0;
}
