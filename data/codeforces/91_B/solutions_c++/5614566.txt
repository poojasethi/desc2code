#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
int n,i,a[100100],b[100100],t;
vector <int> v;
int main()
{
	cin>>n;
	for (i=0;i<n;i++) scanf("%d",&a[i]);
	b[n-1]=a[n-1];
	for (i=n-2;i>=0;i--) b[i]=min(b[i+1],a[i]);
	for (i=0;i<n;i++) v.push_back(b[i]);
	for (i=0;i<n-1;i++){
		t=lower_bound(v.begin(),v.end(),a[i])-v.begin();
//	cout<<t<<" ";
	if ((t-i-2)<0)
		printf("%d ",-1);
	else printf("%d ",t-i-2);
                    }
    printf("%d",-1);
return 0;
}
