#include <iostream>
#include <algorithm>
using namespace std;

const int INF=1000000001;
int n;
int a[100001];
int c[100001];

int main(){

  cin>>n;
  for(int i=0;i<n;i++){
    cin>>a[i];
    c[i]=INF;
  }
  c[n-1]=a[n-1];
  for(int i=n-2;i>=0;i--)
    c[i]=min(c[i+1],a[i]);
  for(int i=0;i<n-1;i++){
    int *it=lower_bound(c+i+1,c+n,a[i]);
    cout<<it-(c+i)-2<<" ";
  }
  cout<<-1<<endl;

  return 0;
}
