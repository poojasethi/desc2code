#include <iostream>
#include <algorithm>

using namespace std;

int main(int argc, char* argv[])
{
	int n;
	cin >> n;

	int q[100000], m[1000000];

	for (int i = 0; i < n; i++)
	{
		cin >> q[i];
	}

	int l = 0;

	for (int i = n - 1; i >= 0; i--)
	{
		if ((l == 0) || (q[i] < l))	
			l = q[i];
		m[i] = l;
	}

	for (int i = 0; i < n; i++)
	{
		int p = lower_bound(m, m + n, q[i]) - m;
		cout << max(-1, p - i - 2) << " ";
	}
	cout << endl;

	return 0;
}
