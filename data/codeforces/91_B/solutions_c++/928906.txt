#include<iostream>
#include<algorithm>
using namespace std;
const int maxn=100000;
int arr[maxn],mn[maxn];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>arr[i];
	mn[n-1]=arr[n-1];
	for(int i=n-2;i>=0;i--)
		mn[i]=min(mn[i+1],arr[i]);
	for(int i=0;i<n-1;i++)
		cout<<lower_bound(mn+i+1,mn+n,arr[i])-mn-i-2<<' ';
	cout<<-1<<endl;
}
