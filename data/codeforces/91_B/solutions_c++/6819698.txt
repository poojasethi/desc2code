#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;
int a[100005];
int b[100005];
int main(){
    int i,n,Min=0x7fffffff;
    scanf("%d",&n);
    for(i=0;i<n;i++)
        scanf("%d",&a[i]);
    for(i=n-1;i>=0;i--)
        b[i]=Min=min(Min,a[i]);
    for(i=0;i<n;i++)
        printf("%d ",max((lower_bound(b,b+n,a[i])-b-i-2),-1));
    printf("\n");
    return 0;
}
