#include<cstdio>
#include<algorithm>
using namespace std;
int main()
{
    int n,a[100005],b[100005];
    pair<int,int> c[100005];
    scanf("%d",&n);
    for(int i=0;i<n;++i)
        {scanf("%d",a+i);c[i]=make_pair(a[i],i);}
    sort(c,c+n);
    int last=0;
    for(int i=0;i<n;++i)
    {

        if(c[i].second>=last){last=c[i].second;b[last]=-1;}
        else b[c[i].second]=last-c[i].second-1;
    }
    for(int i=0;i<n;++i)printf("%d ",b[i]);

}
