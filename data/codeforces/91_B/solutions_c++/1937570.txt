#include <cstdio>
#include <algorithm>
using namespace std;
int n,age[100001],b[100001];

int main(){
    scanf("%d",&n);
    b[n]=1000000005;
    for(int i = 0;i < n;i++) scanf("%d",&age[i]);
    for(int i = n-1;i >= 0;i--) b[i] = min(b[i+1],age[i]);
    for(int i = 0;i < n;i++)
    {
        size_t t = lower_bound(b+i+1,b+n,age[i]) - b;
        printf("%d",t-i-2);
        if(i!=n-1)printf(" ");
    }
}
