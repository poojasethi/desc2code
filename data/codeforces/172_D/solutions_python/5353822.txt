F = {}

def f(k):
    if not k in F:
        s, i, j = 0, 4, 4
        while i <= k:
            s += i * f(k / i)
            i += j + 1
            j += 2
        F[k] = (k * (k + 1)) / 2 - s
    return F[k]

def g(k):
    s, i, j = 0, 4, 4
    while i <= k:
        s += (i - 1) * f(k / i)
        i += j + 1
        j += 2
    return (k * (k + 1)) / 2 - s

a, n = map(int, raw_input().split())
print g(a + n - 1) - g(a - 1)
