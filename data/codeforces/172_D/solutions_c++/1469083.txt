#include<stdio.h>
#include<iostream>
#define N 10000007
using namespace std;
int num[N];
int main()
{
    int n,a,i,j;
    long long sum=0;
    scanf("%d%d",&a,&n);
    for(i=0;i<=n+1+a;i++)num[i]=i;
    for(i=2;i*i<=n+a;i++)
        for(j=i*i;j<=n+a;j+=i*i)
            num[j]=j/(i*i);
      for(i=a;i<n+a;i++)
        sum+=num[i];
     cout<<sum<<endl;
return 0;
}
