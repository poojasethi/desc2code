#include <cstdio>
const int N=10000010;
int DP[N];
int main()
{
    for(int i=1;i<=3162;i++)
    {
        int a=i*i;
        for(int j=a;j<N;j+=a)
            DP[j]=a;
    }
    int a,n;
    scanf("%d%d",&a,&n);
    long long ans=0;
    for(int i=a;i<a+n;i++)
        ans+=i/DP[i];
    printf("%I64d\n",ans);
    scanf(" ");
}
