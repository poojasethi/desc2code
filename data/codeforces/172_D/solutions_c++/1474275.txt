#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
int last[10000003];
int main()
{
    long long i,k,j,a,n,ans;
    for (i=1;i<=3162;i++)
    {
        k=i*i;
        for (j=k;j<=10000000;j+=k) last[j]=k;
    }
    cin>>a>>n;
    ans=0;
    for (i=a;i<=a+n-1;i++) ans+=i/last[i];
    cout<<ans<<endl;
    return 0;
}
