#include <stdio.h>

int in[10000005];
int a, n;

int main(){
    scanf("%d %d",&a, &n);
    for(int i=1;i*i<a + n;i++){
        int days = i * i;
        for(int j=(a / days) * days;j<a + n;j+=days){
            in[j] = j / days;
        }
    }
    long long ans = 0;
    for(int i=a;i<a+n;i++) 
        ans += in[i];
    printf("%I64d\n",ans);
return 0;
}
