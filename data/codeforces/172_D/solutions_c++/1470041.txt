#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int s[10000005];

int main(void) {
	int a, n;
	scanf("%d%d", &a, &n);
	for(int i=1;i*i<=a+n-1;i++) {
		for(int j=i*i;j<=a+n-1;j+=i*i)
			s[j]=i*i;
	}
	long long ans=0;
	for(int i=a;i<=a+n-1;i++)
		ans += i/s[i];
	printf("%I64d\n", ans);
	return 0;
}
