#include <iostream>

#define MAX 10000001
using namespace std;

long long D[MAX];

int main()
{
    long long i,j,k,a,n;
    long long res=0;
    for(i=1;i*i<MAX;i++)
    {
        k=i*i;
        for(j=k;j<MAX;j+=k)
            D[j]=k;
    }
    cin >> a >> n;
    n+=a;
    for(i=a;i<n;i++)
        res+=i/D[i];
    cout << res << endl;
    return 0;
}
