#include <iostream>
#include <vector>

using namespace std;

int main(){ 
  unsigned long long num = 0;
  int a, n;
  cin >> a >> n;
  n += a;
  vector<int> Vect(n, 1);
  
  for (int i = 2; i * i < n; i++)
    for (int p = i * i, j = p; j < n; j += p)
      Vect[j] = p;
  
  for (int i = a; i < n; i++)
    num += i / Vect[i];
  
  cout << num;
  return 0;
}
