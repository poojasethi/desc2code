#include <cstdio>
#include <iostream>
#define MAXN 10000000
using namespace std;
long long res;
int a,n;
int ar[MAXN+1];
int main()
{
	int i;
	long long k;
	scanf(" %d %d",&a,&n);
	for(i=1;i*i<=10000000;i++)
	{
		k=i*i;
		while(k<=10000000)
		{
			ar[k]=i*i;
			k+=i*i;
		}
	}
	for(i=a;i<n+a;i++)
		res+=i/ar[i];
	cout << res << endl;
	return 0;
}
