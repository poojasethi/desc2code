#include<iostream>
#define For(i,x,y) for(i=x;i<=y;i++)
using namespace std;
int dp[20000000],m,n;
void bul(){
	int i,j,k;
	For(i,1,4500){
		k=i*i;
		for(j=k;j<=1.2e+7;j+=k){
			dp[j]=k;	
		}
	}
}
int main(){
	long long int i,t=0;
	bul();
	cin>>m>>n;
	For(i,m,m+n-1){
		t+=i/dp[i];	
	}
	cout<<t;
	return 0;
}
