#include<cstdio>
#include<cmath>

int i,j,a,n,f[11000000];

long long Ans;

int main()
{//	freopen("d.in","rb",stdin);
//	freopen("d.out","wb",stdout);
	scanf("%d%d",&a,&n);
	for(i=1;i*i<a+n;i++)
		for(j=i*i;j<a+n;j+=i*i)
			f[j]=i*i;
	for(i=a;i<a+n;i++)Ans+=i/f[i];
	printf("%I64d\n",Ans);
	return 0;
}
