#include<iostream>
typedef long long ll;
using namespace std;

int d[10000001];

int main(){
	for(int i = 2; i*i <= 10000000; i++){
		for(int j = 1; j*i*i <= 10000000; j++){
			d[j*i*i] = j;
		}
	}
	ll ans = 0;
	int a,n;
	cin>>a>>n;
	for(int i = 0; i < n; i++){
		ans += d[a+i]?d[a+i]:a+i;
	}
	cout<<ans<<endl;
	return 0;
}
