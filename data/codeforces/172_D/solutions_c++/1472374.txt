#include<iostream>
#include<algorithm>
#include<fstream>
#include<string>
using namespace std;
long long int a[10000008];
void doldur(){
	long long int i,j,x;
	for(i=1;i<=3162;i++){
		x=i*i;
		for(j=x;j<=10000000;j+=x)
			a[j]=x;
	}
}
int main(){
	doldur();
	long long int x,y,i,top=0;
	cin>>x>>y;
	for(i=x;i<x+y;i++)
		top+=(i/a[i]);
	cout<<top<<endl;
	return 0;
}
