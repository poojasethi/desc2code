#include <cstdio>

#define N 10001000
int f[N];

int main() {
    for(int i=1; i<N; i++) f[i] = 1;
    for(int i=2; i*i <= N; i++)
        for(int j=i*i; j<N; j+=i*i) if(f[j] < i*i) f[j] = i*i;

    int a,n;
    scanf("%d%d",&a,&n);
    long long ans = 0;
    n += a;
    while(a!=n) {
        ans += a/f[a];
        a++;
    }
    printf("%I64d\n",ans);
}
