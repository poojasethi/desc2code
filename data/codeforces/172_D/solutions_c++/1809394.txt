#include <iostream>

using namespace std;

int f[10000010];

int main(){
    int a, n;
    long long p = 0;
    cin >> a >> n;
    for (int i = 1; i * i < (a + n); i++)
        for (int j = 1; j * i * i < (a + n); j++)
            f[j * i * i] = j;

    for (int i = a; i < a + n; i++) p += f[i];

    cout << p;
}
