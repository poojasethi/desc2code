#include<cstdio>
#include<cmath>
#include<ctime>
#include<algorithm>
using namespace std;
const int X=1e7+10;
int s[X];
int main(){
    int i,j,a,n;
    for(i=1;i*i<X;i++)
        for(j=i*i;j<X;j+=i*i)
            s[j]=i*i;
    long long as=0;
    scanf("%d%d",&a,&n);
    for(i=0,j=a;i<n;j++,i++)
        as+=j/s[j];
    printf("%I64d\n",as);
    return 0;
}
