#include <cstdio>
#include <cstring>

int ans[10000001];

int main(void)
{
	int a, n;
	scanf("%d %d", &a, &n);
	int lim = a+n-1;
	for(int i=10000;i;i--)
	{
		for(int j=i*i;j<=10000000;j+=i*i)
		{
			if(ans[j]==0) ans[j]=j/i/i;
		}
	}

	printf("%I64d\n", sum);

}
