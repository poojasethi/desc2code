#include<iostream>
using namespace std;
#define LL long long
int main()
{
    int l,r;
    cin>>l>>r;
    r+=l-1;
    LL t[10000000+3];
    for(int i=1;i*i<=r;i++)
    for(int j=i*i;j<=r;j+=i*i)
     t[j]=i;
    LL ans;
    for(int i=l;i<=r;++i)
    {
     ans+=i/(t[i]*t[i]);
    }
    cout<<ans<<endl;
    return 0;
}
