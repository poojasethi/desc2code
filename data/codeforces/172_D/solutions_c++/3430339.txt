#include <iostream>
using namespace std;

int main() {
	long long a, n, p = 0;
	cin >> a >> n;
	long long best[10000001];
	long long tmp = 1;
	while(tmp*tmp <= 10000000) {
		long long i = tmp*tmp;
		long long mult = 1;
		while(i*mult <= 10000000) {
			best[i*mult] = mult;
			mult++;
		}
		tmp++;
	}
	for(long long i = 0; i < n; i++) p += best[a+i];
	cout << p << endl;
	return 0;
}
