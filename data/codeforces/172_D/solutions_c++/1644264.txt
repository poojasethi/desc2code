#include <iostream>

#define N 10000000

using namespace std;

int num[N+5];

int main(){
	long long a,n,i,j,k,ans=0;
	for(i=0;i<=N;i++)
		num[i]=1;
	for(i=2;i*i<=N;i++){
		k=i*i;
		for(j=k;j<=N;j+=k)
			num[j]=k;
	}
	while(cin>>a>>n){
		ans=0;
		for(i=a;i<a+n;i++)
			ans+=i/num[i];
		cout<<ans<<endl;
	}
	return 0;
}
