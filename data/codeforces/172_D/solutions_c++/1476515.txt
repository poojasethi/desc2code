#include<iostream>
using namespace std;
long long n, m, k, i, j, a[10000001], p;
main(){
       cin>>n>>m;
       for(i=1;i<=10000001;i++)a[i]=i;
       for(i=2;i<=3400;i++){
       k=0;
       for(j=i*i;j<=10000000;j+=i*i){
       k++;
       a[j]=k;
       }
                           }
       for(j=n;j<n+m;j++)p+=a[j];
       cout<<p;
       }
