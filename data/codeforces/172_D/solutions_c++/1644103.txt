#include <cstdio>

const int MaxN = 10000007;
int DP[MaxN];

int main() {
	long long ret = 0;
	int L, R;
	
	scanf("%d%d", &L, &R);
	R += L - 1;
	for (int i = 1; i * i <= R; i++)
		for (int j = i * i; j <= R; j += i * i)
			DP[j] = i * i;
	for (int i = L; i <= R; i++)
		ret += i / DP[i];
	printf("%I64d\n", ret);
	
	return 0;
}
