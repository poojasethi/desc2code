#include<iostream>
#include<cmath>
#include<cstring>
using namespace std;

#define MAX 10000000
int flag[MAX];

int main()
{
	int a,n;
	cin >> a >> n;
	for(int i=1; i*i <= MAX; ++i )
		for(int j=1; j*i*i <= MAX; ++j){
			if(i ==1 ) flag[j*i*i] = j;
			else flag[j*i*i] = j;
		}
	long long sum = 0;
	for(int i=0; i<n; ++i)
		sum += flag[a+i];
	cout << sum << endl;
}
