#include <iostream>
#include <algorithm>
#include <cstdio>
#define lli long long int
using namespace std;
lli cvp,a,n,A[10000005];
int main()
{
	int i,j,c;
	for(i=1;i<=3162;i++)
	{
		c=i*i;
		for(j=c;j<=10000005;j+=c)
			A[j]=c;
	}
	cin>>a>>n;
	i=a;
	while(i<=a+n-1)
	{
		cvp+=i/A[i];
		i++;
	}
	cout<<cvp<<endl;
	getchar(); getchar();
	return 0;
}
