#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

const int N=10000000;

long long i,j,n,k,pat[11000000],m,q;

int  main()
{

	for(i=1;i*i<=N;i++)
	{
		k=i*i;
		for(j=k;j<=N;j+=k)
		{
			pat[j]=j/k;
		}
	}

	cin>>n>>m;
	for(i=n;i<=n+m-1;i++)
	{
		q+=pat[i];
	}
	cout<<q<<endl;
	return 0;
}
