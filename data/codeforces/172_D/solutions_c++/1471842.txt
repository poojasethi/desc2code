#include <iostream>
#include <cstring>
using namespace std;

long long ar[10000100];

int main()
{
	long long a, b, i, x, j;
	cin >> a >> b;
	b = a + b - 1;
	i = 1;
	while (1)
	{
		x = i * i;
		if (x > b)
			break;
		j = x;
		while (j <= b)
		{
			ar[j] = j / x;
			j += x;
		}
		i++;
	}
	j = 0;
	for (i = a; i <= b; i++)
		j += ar[i];
	cout << j;
	return 0;
}
