#include <stdio.h>
#define max 10000001
int arr[max];
int main()
{
int n,m;
int i,j;
long long int sum;
for(i=1;i*i<max;i++)
{
   for(j=i*i;j<=max;j+=i*i)
   {
      arr[j]=i*i;
   }
}
while(scanf("%d%d",&n,&m)==2)
{
   sum=0;
   for(i=n;i<n+m;i++)
   {
      sum+=i/arr[i];
      //printf("%d ",arr[i]);
   }
   printf("%I64d\n",sum);
}	
return 0;
}
