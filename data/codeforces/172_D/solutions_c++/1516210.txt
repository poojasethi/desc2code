# include <iostream>
using namespace std;
const int MAX_N=1000*1000*10+5;
long long D[MAX_N];
int A,N;
int main()
{
    ios::sync_with_stdio(false);
    cin >> A >> N;
    for(int i=1;i*i<MAX_N;i++)
        for(int j=i*i;j<MAX_N;j+=(i*i))
            D[j]=i*i;
    long long Ans=0;
    for(int i=A;i<A+N;i++)
        Ans+=(i/D[i]);
    cout<<Ans<<endl;
    return  0;
}
