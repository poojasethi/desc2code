#include <iostream>
using namespace std;
const int N = 10000005;
long long D[N],ans, a, n;
int main(){
	cin >> a >> n;
	for(int i = 1; i * i < N; i ++)
		for(int j = i * i; j < N; j += i * i)
			D[j] = i*i;
	for(int i = a; i < a + n; i++)
		ans += (i/D[i]);
	cout << ans << endl;
	return 0;
}
