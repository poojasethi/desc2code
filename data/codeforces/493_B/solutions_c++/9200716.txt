#include<iostream>

using namespace std;
long long i,j,k,l,f,a,n,m,d[100005],s[100005],t,g,x;

main(){
cin>>n;
for(i=0;i<n;i++){
    cin>>a;
    if(a<0) {d[l]=a*(-1);f+=d[l];l++;} else {s[k]=a;g+=s[k];k++;}
}

if(f>g) {cout<<"second";return 0;}
if(f<g) {cout<<"first";return 0;}

for(i=0;i<min(l,k);i++){
    if(d[i]>s[i]) {cout<<"second";return 0;}
    if(s[i]>d[i]) {cout<<"first";return 0;}
}

    if(a>0) cout<<"first"; else cout<<"second";
}
