#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long n,s,x,t,chk;
    cin>>n;
    s=0;
    vector<int>a,b;
    while(n--)
    {
        cin>>x;
        t=x>0;
        s+=x;
        if(x>0)
           a.push_back(x);
        else
            b.push_back(-x);

    }
    chk=0;
    if(s!=0)
        chk=s>0;
    else if(a>b)
        chk=1;
    else if(a<b)
        chk=0;
    else
        chk=t;
    printf("%s",chk? "first" : "second");

}
