#include <bits/stdc++.h>
using namespace std;
int main() {
	int N, x;
	vector<int> p1, p2;
	long long s1 = 0, s2 = 0;
	cin >> N;
	string last;
	for (int i=0; i<N; ++i) {
		cin >> x;
		if (x > 0) {
			p1.push_back(x);
			s1 += x;
			last = "first";
		} else {
			p2.push_back(-x);
			s2 += (-x);
			last = "second";
		}
	}
	if (s1 != s2) {
		cout << ((s1 < s2) ? "second" : "first");
	} else {
		cout << ((p1 < p2) ? "second" : ((p1 == p2) ? last : "first"));
	}
	cout << "\n";
	return 0;
}
