#include<bits/stdc++.h>
using namespace std;
int n,x,k;
long long s[2];
list<long long> a[2];
int main() {
	cin>> n;
	while (n--) {
		scanf("%d", &x);
		k = x<0;
		x = abs(x);
		s[k] += x;
		a[k].push_back(x);
	}
	a[0].push_front(s[0]);
	a[1].push_front(s[1]);
	if (a[0] == a[1]) a[k].push_back(1);
	cout<< (a[0]>a[1]? "first":"second");
}
