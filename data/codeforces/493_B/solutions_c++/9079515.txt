#include <bits/stdc++.h>
using namespace std;

long long n, t, c, s[2];
vector<long long> a[2];
int main()
{
//	freopen ("input.txt", "r", stdin);
	cin >> n;
	a[0].push_back(0);
	a[1].push_back(0);
	while (n--) {
		scanf("%I64d", &c);
		t = c < 0;
		a[t].push_back(abs(c));
		a[t][0] += abs(c);
	}
	a[t].push_back(1);
	puts(a[0] > a[1] ? "first" : "second");
}
