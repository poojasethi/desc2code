#include<cstdio>
#include<vector>
using namespace std;

int n, x;
long long tot;
vector<int> v[2];

bool chk() {
	if (tot) return (tot > 0);
	for(int i=0;i<v[0].size();i++)
		if(v[0][i] != v[1][i])
			return v[0][i] > v[1][i];
	return (x>0);
}

int main() {
	scanf("%d", &n);
	tot = 0LL;
	for(int i=0;i<n;i++) {
		scanf("%d", &x);
		tot += x;
		if (x>0) v[0].push_back(x);
		else v[1].push_back(-x);
	}
	if (chk()) printf("first\n");
	else printf("second\n");
	
	return 0;
}
