#include <iostream>

using namespace std;

long long ans,la,lb,sa,sb,n,t,a[200002],b[200002];

int main()
{ cin>>n;
  for(int i=0;i<n;i++) {cin>>t; 
   if (t>0) {la++; sa+=t; a[la]=t; ans=1;} else {lb++; sb-=t; b[lb]=-t; ans=0;}
  }
  if (sa>sb) {cout<<"first"; return 0;}
  if (sb>sa) {cout<<"second"; return 0;}
  for(int i=1;i<=la;i++) 
	if (a[i]>b[i]) {cout<<"first"; return 0;} else if (a[i]<b[i]) {cout<<"second"; return 0;}
  if (ans) cout<<"first"; else cout<<"second";
}
