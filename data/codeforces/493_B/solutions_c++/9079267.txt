#include <bits/stdc++.h>
using namespace std;

long long n, t, c, s[2];
list<long long> a[2];
int main()
{
//	freopen ("input.txt", "r", stdin);
	cin >> n;
	while (n--) {
		cin >> c;
		t = c < 0;
		a[t].push_back(abs(c));
		s[t] += abs(c);
	}
	a[0].push_front(s[0]);
	a[1].push_front(s[1]);
	a[t].push_back(1);
	puts(a[0] > a[1] ? "first" : "second");
}
