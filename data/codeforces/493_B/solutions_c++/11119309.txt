#include <bits/stdc++.h>
using namespace std;

long long n , a_ = 0 , as(0) , bs(0);
bool f = 0;

int main(){
	vector<int> a[2];
	cin >> n;
	while (n--){
		long long x; cin >> x;
		f = x > 0;
		a[f].push_back(abs(x));
		a_ += x;
	}
	if (a_){
		f = a_ > 0;
		cout << (f ? "first" : "second");
		return 0;
	}
	if (a[0] != a[1]) f = a[1] > a[0];
	cout << (f ? "first" : "second");
}
