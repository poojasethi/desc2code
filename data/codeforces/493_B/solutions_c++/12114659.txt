#include<stdio.h>
int i,n,a[200001],b[200001],pa,pb;
long long t,sa,sb;
int main(){
	scanf("%d",&n);
	for(i=0;i<n;i++){
		scanf("%I64d",&t);
		if(t>0){a[pa++]=t;sa+=t;}
		else if(t<0){b[pb++]=-t;sb-=t;}
	}
	if(sa>sb){
		printf("first");
		return 0;
	}
	else if(sb>sa){
		printf("second");
		return 0;
	}
	else{
		for(i=0;i<200000;i++){
			if(a[i]>b[i]){printf("first");return 0;}
			else if(b[i]>a[i]){printf("second");return 0;}
		}
		printf("%s",t>0?"first":"second");
	}
}
