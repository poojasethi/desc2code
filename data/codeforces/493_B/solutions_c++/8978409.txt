#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
#include<string>
#include<queue>
#include<map>
#define CL(x,y) memset(x,y,sizeof(x))

using namespace std;

int N,M,t;
bool flag;
long long int a,b;
vector<int> va,vb;

int main(){
	scanf("%d",&N);
	for (int i=0;i<N;i++){
		scanf("%d",&t);
		if (t>0){
			a+=t;
			va.push_back(t);
		}else{
			b+=-t;
			vb.push_back(-t);
		}
	}
	if (a==b) flag = va==vb?(t>0):(va>vb);
	else flag = a>b;
	puts(flag?"first":"second");
}
