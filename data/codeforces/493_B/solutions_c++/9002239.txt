#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
main()
{
	ll b,c,n;
	list<ll> k[2];
	vector<ll> a(2,0);
	cin>>n;
	while(n--)
	{
		cin>>b;
		c = b<0;
		b=abs(b);
		a[c]+=b;
		k[c].push_back(b);
	}
	k[0].push_front(a[0]);
	k[1].push_front(a[1]);
	if(k[0]==k[1]) k[c].push_back(1);
	cout<<(k[0]>k[1]? "first":"second");
}
