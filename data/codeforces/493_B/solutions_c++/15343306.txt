#include<iostream>
#include<vector>
using namespace std;
int main()
{
	long long n,s = 0,t,f = 1;
	cin>>n;
	vector<long long > a,b;

	while(--n)
	{
		cin>>t;
		s+=t;
		(t>0)?a.push_back(t):b.push_back(-1*t);
	}

	cin>>t;
	s+=t;
	
	if(t>0)
		a.push_back(t);
	else{
		b.push_back(-1*t);
		f = 0;
	}

	if(s>0)
		cout<<"first";
	else if(s<0)
		cout<<"second";
	else if(a>b)
		cout<<"first";
	else if(b>a)
		cout<<"second";
	else if(f)
		cout<<"first";
	else
		cout<<"second";

	cout<<endl;
}
