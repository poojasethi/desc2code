#include<cstdio>
using namespace std;
int a1[2][200000],l[2];
long long tt[2];
int main()
{
	int m,a,b;
	scanf("%d",&m);
	while(m--)
	{
		scanf("%d",&a);
		if(a>0)b=0;
		else a=-a,b=1;
		tt[b]+=a;
		a1[b][l[b]++]=a;
	}
	if(tt[0]!=tt[1])
	{
		puts(tt[0]>tt[1]?"first":"second");
		return 0;
	}
	a1[!b][l[!b]++]=-1;
	for(int i=0;i<l[0]||i<l[1];++i)
	if(a1[0][i]!=a1[1][i])
	{
		puts(a1[0][i]>a1[1][i]?"first":"second");
		break;
	}
}
