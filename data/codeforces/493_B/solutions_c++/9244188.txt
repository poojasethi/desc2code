#include <bits/stdc++.h>

using namespace std;

list<long long> v[2];

int main()
{
    long long s[2]; 
    s[0] = s[1] = 0;
    int n,t,x;

    scanf("%d",&n);
    
    while(n--)
    {
        scanf("%d",&x);
        t = x < 0;
        x = abs(x);
        v[t].push_back(x);
        s[t] += x;
    }
    v[0].push_front(s[0]);
    v[1].push_front(s[1]);
    if(v[0]==v[1]) v[t].push_back(1);
    cout<<(v[0]>v[1]? "first":"second");
    
    
    return 0;
}
