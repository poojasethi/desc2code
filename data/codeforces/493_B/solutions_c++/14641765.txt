#include <bits/stdc++.h>

using namespace std;
vector<int> a, b;
int n, x, flag;
long long res = 0;

main() 
{
  
    cin >> n;
    while (n--) 
    {
        cin >> x;
        flag = ((x > 0)?1:0);
        if (x > 0) a.push_back(x);
        else b.push_back(-x);
        res += x;
    }
    if (res != 0) res = ((res > 0)?1:0);
    else if (a> b) res = 1;
    else if (a < b) res = 0;
    else res = flag;
    printf("%s\n", res? "first": "second");
}
