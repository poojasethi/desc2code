#include <bits/stdc++.h>
using namespace std;
long long n, s[2];
list<long long> a[2];

int main() {
//    freopen("input.txt", "r", stdin);
    int x;
    cin >> n;
    while (n--) {
        scanf("%d", &x);
        s[x < 0] += abs(x);
        a[x < 0].push_back(abs(x));
    }
    a[0].push_front(s[0]);
    a[1].push_front(s[1]);
    if (a[0] == a[1]) a[x < 0].push_back(1);
    puts(a[0] > a[1] ? "first" : "second");
}
