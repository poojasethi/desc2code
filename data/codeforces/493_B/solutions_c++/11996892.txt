#include <iostream>
#include <vector>
using namespace std;

vector<int> A[2];
long long ans;
int n, x;
int main() {
    for (cin >> n; n--;) {
        cin >> x;
        ans += x;
        A[x < 0].push_back(x > 0 ? x : -x);
    }
    if (ans) ans = ans > 0;
    else if (A[0] > A[1]) ans = 1;
    else if (A[0] < A[1]) ans = 0;
    else ans = x > 0;
    cout << (ans ? "first" : "second");
}
