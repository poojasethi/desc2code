n = input()
A = []
for i in range(n):
    A.append(input())
FIRST = filter(lambda x: x > 0, A)
SECOND = map(abs,filter(lambda x: x < 0, A))

if sum(A) != 0:
    if sum(FIRST) > sum(SECOND):
        print "first"
    else:
        print "second"
else:
    if FIRST > SECOND:
        print "first"
    elif SECOND > FIRST:
        print "second"
    else:
        if A[-1] > 0:
            print "first"
        else:
            print "second"
