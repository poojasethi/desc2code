from sys import stdin
n=int(stdin.readline().rstrip("\n"))
l1=[]
l2=[]
flag=True
for i in range(0,n):
	k=int(stdin.readline().rstrip("\n"))
	if k>0:
		l1.append(k)
		flag=True
	else:
		l2.append(-k)
		flag=False
f1=sum(l1)
f2=sum(l2)
if f1>f2:
	print "first"
elif f2>f1:
	print "second"
else:
	if l1>l2:
		print "first"
	elif l2>l1:
		print "second"
	else:
		if flag:
			print "first"
		else:
			print "second"
