n=int(raw_input())
a=[]
b=[]
for i in xrange(n):
	x=int(raw_input())
	if x<0:
		b.append(-x)
	else:
		a.append(x)
	last=x
sa=sum(a)
sb=sum(b)
if sa>sb: print "first"
elif sa<sb: print "second"
else:
	if a>b: print "first"
	elif a<b: print "second"
	else:
		if last<0: print "second"
		else: print "first"
