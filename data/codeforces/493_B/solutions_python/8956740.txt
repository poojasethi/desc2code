from sys import stdin
def main():
    n = int(stdin.readline())
    a = [[], []]
    w = 0
    for i in xrange(n):
        t = int(stdin.readline())
        w = t < 0
        a[w] += [-t if w else t]
    s = [[sum(a[0])] + a[0], [sum(a[1])] + a[1]]
    if s[0] < s[1] or (s[0] == s[1] and w):
        print "second"
    else:
        print "first"
main()
