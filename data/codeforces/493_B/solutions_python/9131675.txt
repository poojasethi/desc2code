N = int(raw_input(''))
A = []
B = []
last_a = 1000000000
last_b = 1000000000
for i in range(0, N):
    x = int(raw_input(''))
    if x > 0:
        A.append(x)
        last_a = i
    else:
        B.append(-x)
        last_b = i
        
first = "first"
second = "second"
        
if sum(A) > sum(B):
    print first
elif sum(A) < sum(B):
    print second
elif A > B:
    print first
elif A < B:
    print second
elif last_a > last_b:
    print first
else:
    print second
