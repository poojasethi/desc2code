

from sys import stdin

import fileinput


count = int(stdin.readline()[:-1])

inp = []
for i in xrange(count):
	inp.append(stdin.readline()[:-1])	

inp = map(int, inp)

first = []
second = []
for i in inp:
	if i > 0:
		first.append(i)
	elif i < 0:
		second.append(-1*i)

ptFirst = sum(first)
ptSecond = sum(second)
if ptFirst > ptSecond:
	print "first"
elif ptFirst < ptSecond:
	print "second"
else:
	done = False
	for i in xrange(min(len(first), len(second))):
		if first[i] > second[i]:
			print "first"
			done = True
			break
		if first[i] < second[i]:
			print "second"
			done = True
			break
	if not done:
		if inp[-1] > 0:
			print "first"
		else:
			print "second"
