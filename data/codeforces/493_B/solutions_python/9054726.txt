#!/usr/bin/python
#encoding=utf-8

n = int(raw_input())
tech_point = [ int(raw_input()) for i in xrange(n) ]
sum = reduce(lambda x, y: x + y, tech_point)
positive_point = filter(lambda x : x > 0, tech_point)
negative_point = filter(lambda x : x < 0, tech_point)
negative_point[:] = [ -x for x in negative_point ]
result = ('first', 'second')
if sum > 0:
    print 'first'
elif sum < 0:
    print 'second'
else:
    if positive_point > negative_point:
        print result[0]
    elif positive_point < negative_point:
        print result[1]
    else:
        if tech_point[-1] < 0:
            print 'second'
        else:
            print 'first'
