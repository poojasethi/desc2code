N = int(raw_input())
A = []
B = []
ultimo = "first"
for _ in range(N):
    p = int(raw_input())
    if p > 0:
        A.append(p)
        ultimo  = "first"
    else:
        B.append(-1*p)
        ultimo  = "second"

sA, sB = sum(A), sum(B)
if sA < sB:
    print "second"
elif sA > sB:
    print "first"
else:
    if A > B:
        print "first"
    elif A < B:
        print "second"
    else:
        print ultimo
