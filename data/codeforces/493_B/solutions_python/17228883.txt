def main():
	n = input()
	a, b = [], []
	last = 0
	for i in xrange(n):
		x = input()
		if x > 0: a.append(x)
		else: b.append(-x)
		if i == n - 1: last = 0 if x > 0 else 1
	s1, s2 = sum(a), sum(b)
	if s1 != s2: print ["first", "second"][s1 < s2]
	else: 
		if a != b: print ["first", "second"][a < b]
		else: print ["first", "second"][last]

main()
