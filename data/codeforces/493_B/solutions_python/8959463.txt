n = input()

i = 0
a = []
b = []
for k in range(0, n):
    i = input()
    if i > 0:
        a.append(i)
    elif i < 0:
        b.append(abs(i))

#print a, b, ta, tb

if sum(a) > sum(b):
    print "first"
elif sum(a) < sum(b):
    print "second"
elif a > b: print "first"
elif b > a: print "second"
elif i > 0: print "first"
else: print "second"
