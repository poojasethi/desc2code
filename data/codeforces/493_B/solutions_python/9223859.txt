#!/usr/bin/env python
#-*- coding:utf-8 -*-

n = int(raw_input())

a = []
b = []
sa = 0
sb = 0
last = None
for _ in xrange(n):
    v = int(raw_input())
    if v > 0:
        last = "a"
        sa += v
        a.append(v)
    else:
        last = "b"
        v = -v
        sb += v
        b.append(v)

if sa > sb:
    print "first"
elif sb > sa:
    print "second"
else:
    if a > b:
        print "first"
    elif b > a:
        print "second"
    else:
        if last == "a":
            print "first"
        else:
            print "second"
