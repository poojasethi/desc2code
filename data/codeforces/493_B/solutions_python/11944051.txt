n=input()

l=[]
r=[]
C=0
for i  in range(n):
   c=input()
   if c <0:
       r+=[-c]
       C=1
   else:
       l+=[c]
       C=0

if r == l:
   if C==1:
       print 'second'
   else:
       print 'first'
elif (r > l and sum(r) == sum(l) )or sum(r) > sum(l):       
   print 'second'
else:
   print 'first'
