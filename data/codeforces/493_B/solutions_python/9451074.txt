n = int(raw_input())
first = []
second = []
last = "none"
for i in xrange(n):
    c = int(raw_input())
    if c > 0:
        first.append(c)
        last = "first"
    else:
        second.append(-c)
        last = "second"
if sum(first) > sum(second) or (sum(first) == sum(second) and first > second):
    print "first"
else:
    if first == second: print last
    else: print "second"
