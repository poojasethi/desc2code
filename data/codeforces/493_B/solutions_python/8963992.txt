n = input()
L1 = [0]*200000
L2 = [0]*200000
last_1 = False
count_1 = 0
count_2 = 0
for i in range(n):
    j = int(raw_input())
    if j>0:
        L1[count_1] = j
        count_1 += 1;
    else:
        L2[count_2] = -j
        count_2 += 1;
    last_1 = j>0
s1=sum(L1[:count_1])
s2=sum(L2[:count_2])
if s1>s2: print "first"
elif s2>s1: print "second"
elif L1>L2: print "first"
elif L1<L2: print "second"
elif last_1: print "first"
else: print "second"
