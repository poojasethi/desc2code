n = int(raw_input())
a1, a2 = [], []
last = 0
for _ in range(n):
    p = int(raw_input())
    if p < 0:
        a2.append(-p)
    else:
        a1.append(p)
    last = p
s1 = sum(a1)
s2 = sum(a2)
if s1 < s2:
    print "second"
elif s1 > s2:
    print "first"
elif a1 < a2:
    print "second"
elif a2 < a1 or last > 0:
    print "first"
else:
    print "second"
