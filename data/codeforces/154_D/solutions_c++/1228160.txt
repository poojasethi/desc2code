#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;
typedef long long LL;
LL x1, x2, a, b;
void solve() {
	if(x2 >= x1+a && x2 <= x1+b) {
		printf("FIRST\n");
		printf("%I64d\n", x2);
		return;
	}
	if((x1<x2 && a<=0) || (x1>x2 && b>=0)) {
		printf("DRAW\n");
		return;
	}
	LL n = abs(x1-x2), m = abs(a)+abs(b);
	a=abs(a); b=abs(b);
	if(a>b) swap(a, b);
	if(n%m==0) printf("SECOND\n");
	else {
		if(n%m>=a && n%m<=b) {
			printf("FIRST\n");
			printf("%I64d\n", x1+(x1>x2? -(n%m): n%m));
		} else
			printf("DRAW\n");
	}
}

int main(void) {
	scanf("%I64d%I64d%I64d%I64d", &x1, &x2, &a, &b);
	solve();
	
	return 0;
}
