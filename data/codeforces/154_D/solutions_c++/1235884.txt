#include<iostream>
using namespace std;

#define D { cout << "DRAW" << endl; return 0; }
#define W(x) { cout << "FIRST" << endl << x << endl; return 0; }
#define L { cout << "SECOND" << endl; return 0; }

int main()
{
	int x1,x2;
	int a,b;
	cin >> x1 >> x2 >> a >> b;
	int d = x2-x1;
	if( x1 > x2 )
	{
		a = -a;
		b = -b;
		swap(a,b);
		//swap(x1,x2);
		d = x1-x2;
	}
	//int d = x2-x1;
	if( b < 0 )
		D
	if( a <= 0 )
	{
		if( d >= a && d <= b )
			W(x2)
		else
			D
	}
	d %= (a+b);
	if( d == 0 )
		L
	else if( d >= a && d <= b )
		W(x1+d*( x1<x2 ? 1 : -1 ))
	else
		D
	return 0;
}
