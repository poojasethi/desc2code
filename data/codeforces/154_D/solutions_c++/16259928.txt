#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;

int x,y,a,b,flag=1;
int solve(){
	if(b<0) return -1;
	else if(a<=y-x&&y-x<=b) return y-x;
	else if(a<=0) return -1;
	else{
		int k=(y-x)%(a+b);
		if(!k) return -2;
		else if(a<=k&&k<=b) return k;
		else return -1;
	}
}

int main(){
	scanf("%d%d%d%d",&x,&y,&a,&b);
	if(x>y) x*=-1,y*=-1,a*=-1,b*=-1,swap(a,b),flag=-1;
	
	int ans=solve();
	if(ans>=0) printf("FIRST\n%d\n",flag*(x+ans));
	else if(ans==-2) printf("SECOND\n");
	else printf("DRAW");
	
	return 0;
}
