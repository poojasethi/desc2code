#include <iostream>
#include <stdio.h>
#include <stdlib.h>
#include <math.h>
#include <ctype.h>
#include <string.h>

using namespace std;

int main(){

    int x1, x, a, b;

    cin >> x1 >> x >> a >> b;
    x -= x1;

    if ( ( (a <= 0) && (x > 0) ) || ( (b >= 0) && (x < 0) ) ){
        if (x > 0){
            if (x > b) cout << "DRAW";
            else cout << "FIRST\n" << x1 + x;
        }
        else{
            if (x < a) cout << "DRAW";
            else cout << "FIRST\n" << x1 + x;
        }
    }
    else{
        x %= a + b;
        if (x == 0) cout << "SECOND";
        else{
            if ( (x < a) || (x > b) ) cout << "DRAW";
            else cout << "FIRST\n" << x1 + x;
        }
    }

    return 0;
}
