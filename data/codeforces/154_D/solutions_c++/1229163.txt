#include <stdio.h>
#include <algorithm>
using namespace std;

int main() {
	int x1, x2, a, b;
	scanf("%d %d %d %d", &x1, &x2, &a, &b);
	if (a <= 0 && 0 <= b) {
		if (x1+a <= x2 && x2 <= x1+b) {
			printf("FIRST\n%d\n", x2);
		} else {
			printf("DRAW\n");
		}
	} else {
		bool neg = (b < 0);
		if (b < 0) {
			a = -a;
			b = -b;
			swap(a,b);
			x1 = -x1;
			x2 = -x2;
		}
		if (x2 < x1) {
			printf("DRAW\n");
		} else {
			int r = (x2-x1)%(a+b);
			if (r == 0)
				printf("SECOND\n");
			else {
				if (a <= r && r <= b) {
					printf("FIRST\n%d\n", (!neg) ? (x1+r) : (-x1-r));
				} else {
					printf("DRAW\n");
				}
			}
		}
	}
	return 0;
}
