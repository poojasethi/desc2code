#include <algorithm>
#include <cstdio>
#define LL long long int
using namespace std;

LL x1,x2,a,b,T,mark,P;

int main(){
	scanf("%I64d%I64d%I64d%I64d",&x1,&x2,&a,&b);
	T = x2 - x1;
	mark = 1;
	if (T < 0){
		mark = -1;
		T = -T;
		a = -a;
		b = -b;
	}
	if (min(a,b) <= T && T <= max(a,b)){
		// Checking One step
		printf("FIRST\n");
		printf("%I64d\n",x2);
	}
	else {
		if (min(a,b) <= 0)
			printf("DRAW\n"); // Stay or run away
		else {
			if (a > b) swap(a,b);
			P = (T)%(a+b);
			if (P == 0) printf("SECOND\n");
			else if (a <= P && P <= b){
				printf("FIRST\n");
				printf("%I64d\n",x1 + mark*P);
			}
			else printf("DRAW\n");
		}
	}
	return 0;
}
