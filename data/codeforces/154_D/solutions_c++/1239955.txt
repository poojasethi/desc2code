#include <iostream>
#include <vector>
#include <cstdlib>

using namespace std;

int main()
{
  int x1,x2,a,b;
  cin>>x1>>x2>>a>>b;
  bool reverso=false;
  if (x1>x2) {
    swap(x1,x2);
    a=-a;
    b=-b;
    swap(a,b);
    reverso=true;
  }
  if (x1+a<=x2 and x2<=x1+b) {
    cout<<"FIRST"<<endl;
    if (not reverso)
      cout<<x2<<endl;
    else
      cout<<x1<<endl;
    exit(0);
  }
  if (a<=0) {
    cout<<"DRAW"<<endl;
    exit(0);
  }
  int x=x2-x1;
  x=x%(a+b);
  if (x==0) {
    cout<<"SECOND"<<endl;
  } else if (x>=a and x<=b) {
    cout<<"FIRST"<<endl;
    if (not reverso)
      cout<<x1+x<<endl;
    else
      cout<<x2-x<<endl;
  } else {
    cout<<"DRAW"<<endl;
  }
}
