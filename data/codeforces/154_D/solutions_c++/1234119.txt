#include<cstdio>
int x1,x2,a,b,d,t;
int main()
{
	scanf("%d%d%d%d",&x1,&x2,&a,&b);
	if(x1+a<=x2&&x1+b>=x2)
	{
		printf("FIRST\n");
		printf("%d\n",x2);
		return 0;
	}
	d=x2-x1;
	if(b<0&&d>0||b>0&&d<0||a<=0&&b>=0)
	{
		printf("DRAW\n");
		return 0;
	}
	if(d<0) d=-d;
	if(a<0) a=-a;
	if(b<0) b=-b;
	if(a>b) t=a,a=b,b=t;
	d=d%(a+b);
	if(!d) printf("SECOND\n");
	else if(d>=a&&d<=b)
		{
			printf("FIRST\n");
			if(x1<x2) printf("%d\n",x1+d);
			else printf("%d\n",x1-d);
		}
		else printf("DRAW\n");
	return 0;
}
