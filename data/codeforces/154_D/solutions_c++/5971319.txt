#include <iostream>
#include <algorithm>
using namespace std;

int x, y, a, b, d;
string f = "FIRST\n", s = "SECOND\n", t = "DRAW\n";

int main(){
  cin >> x >> y >> a >> b;
  if(x > y){
    a = -a;
    b = -b;
    swap(a, b);
  }
  d = abs(y - x);
  if(a > 0){
    if(d < a + b){
      if(d < a){
        cout << t;
      } else if(d <= b){
        cout << f << y << '\n';
      } else {
        cout << t;
      }
    } else {
      d %= (a + b);
      if(d == 0){
        cout << s;
      } else if(d < a || d > b){
        cout << t;
      } else {
        cout << f << x + (y - x) % (a + b) << '\n';
      }
    }
  } else {
    if(d >= a && d <= b){
      cout << f << y << '\n';
    } else {
      cout << t;
    }
  }
}
