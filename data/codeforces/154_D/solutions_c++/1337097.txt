#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int x1,x2,a,b;
int main(){
  cin>>x1>>x2>>a>>b;
  if(a<=0 && 0<=b){
   if(x1+a<=x2 && x2<=x1+b)cout<<"FIRST\n"<<x2<<'\n';
   else cout<<"DRAW\n";
                  }
  else{
   int d=x2-x1,sa=a,sb=b;
   if(sa<0)d*=-1 , sa*=-1 , sb*=-1 , swap(sa,sb);
   if(d<0)cout<<"DRAW\n";
   else{
    if(d%(sa+sb)==0)cout<<"SECOND\n";
    else if(d%(sa+sb)>=sa && d%(sa+sb)<=sb){
          cout<<"FIRST\n";
          int dst=d%(sa+sb);
          if(a>0)cout<<x1+dst<<'\n';
          else cout<<x1-dst<<'\n';
                                           }
         else cout<<"DRAW\n";
       }
      }
  //system("pause");
  return 0;
}
