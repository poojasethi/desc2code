#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;

int main()
{
	int x,y,a,b,v,d;
	scanf("%d%d%d%d",&x,&y,&a,&b);
	d=y-x;
	v=1;
	if(a<=0 && b<=0)
	{
		swap(a,b);a=-a;b=-b;d=-d;v=-1;
	}
	if(a<=0)
	{
		if(d<a || d>b)printf("DRAW\n");
		else printf("FIRST\n%d\n",y);
	}
	else
	{
		if(d<0)printf("DRAW\n");
		else if(d%(a+b)==0)printf("SECOND\n");
		else if(d%(a+b)<a || d%(a+b)>b)printf("DRAW\n");
		else printf("FIRST\n%d\n",y-v*(d/(a+b)*(a+b)));
	}
}
