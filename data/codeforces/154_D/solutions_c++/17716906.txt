#include<cstdio>

int x,y,d,a,b;

int main()
{
	while(~scanf("%d%d%d%d",&x,&y,&a,&b))
	{
		d=y-x;
		if(a<0&&b<0)
		{
			if(x<y) printf("DRAW\n");
			else
			{
				d*=-1,a*=-1,b*=-1;
				if(d%(a+b)==0) printf("SECOND\n");
				else if(b<=d%(a+b)&&d%(a+b)<=a) printf("FIRST\n%d\n",x-d%(a+b));
				else printf("DRAW\n");
			}
		}
		else if(a>0&&b>0)
		{
			if(x<y)
			{
				if(d%(a+b)==0) printf("SECOND\n");
				else if(a<=d%(a+b)&&d%(a+b)<=b) printf("FIRST\n%d\n",x+d%(a+b));
				else printf("DRAW\n");
			}
			else printf("DRAW\n");
		}
		else if(a<=0&&b>=0)
		{
			if(x<y)
			{
				if(b>=d) printf("FIRST\n%d\n",y);
				else printf("DRAW\n");
			}
			else if(x>y)
			{
				d*=-1,a*=-1;
				if(a>=d) printf("FIRST\n%d\n",y);
				else printf("DRAW\n");
			}
		}
	}
	return 0;
}
