#include <iostream>
using namespace std;
int main()
{
	ios::sync_with_stdio(false);
	long long x1, x2, a, b;
	cin >> x1 >> x2 >> a >> b;
	long long d = x2 - x1;
	if (d == 0)
	{
		cout << "SECOND";
		return 0;
	}
	if (a <= d && d <= b)
	{
		cout << "FIRST" << endl << x2;
		return 0;
	}
	if (a * b <= 0)
	{
		cout << "DRAW";
		return 0;
	}
	bool is = false;
	if (b < 0)
	{
		a = -a;
		b = -b;
		d = -d;
		swap(a, b);
		is = true;
	}
	if (d < 0)
	{
		cout << "DRAW";
		return 0;
	}
	if (d % (a + b) == 0)
	{
		cout << "SECOND";
		return 0;
	}
	if (a <= d % (a + b) && d % (a + b) <= b)
	{
		if (is)
			cout << "FIRST" << endl << x1 - (d % (a + b));
		else
			cout << "FIRST" << endl << x1 + (d % (a + b));
		return 0;
	}
	cout << "DRAW";
	return 0;
}
