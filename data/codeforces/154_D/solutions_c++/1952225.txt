#include <iostream>
#include <string.h>
#include <algorithm>
#include <cstdio>
using namespace std;

int x1, x2, a, b;

void work()
{
    if (x1+a <= x2 && x1+b >= x2)
    {
        puts("FIRST");
        printf("%d\n", x2);
        return;
    }
    bool SWAP = false;
    if (x1 > x2) SWAP = true, swap(x1, x2), swap(a, b), a *= -1, b *= -1, dec;
    if (a <= 0) 
    {
        puts("DRAW");
        return;
    }
    int SG = (x2-x1)%(a+b);
    if (SG == 0)
    {
        puts("SECOND");
        return;
    }
    if (SG < a || SG > b)
    {
        puts("DRAW");
        return;
    }
    puts("FIRST");
    if (SWAP) printf("%d\n", x2-SG);
    else      printf("%d\n", x1+SG);
}

int main()
{
    while (scanf("%d%d%d%d", &x1, &x2, &a, &b) != EOF) work();
}
