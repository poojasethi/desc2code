#include <cstdio>
using namespace std;

int x1, x2;
int a, b;
int mod;
int is_reversed;

int main() {
    scanf("%d %d %d %d", &x1, &x2, &a, &b);
    if (x1 > x2) {
        int tmp;
        is_reversed = 1;
        x1 = -x1;
        x2 = -x2;
        tmp = a;
        a = -b;
        b = -tmp;
    }
    
    if (a <= x2-x1 && x2-x1 <= b) {
        puts("FIRST");
        printf("%d\n", is_reversed ? -x2 : x2);
        return 0;
    }

    if (a <= 0) {
        puts("DRAW");
        return 0;
    }

    mod = (x2-x1)%(a+b);
    if (mod == 0) {
        puts("SECOND");
        return 0;
    }

    if (a <= mod && mod <= b) {
        puts("FIRST");
        printf("%d\n", is_reversed ? (-x1-mod) : (x1+mod));
    } else {
        puts("DRAW");
    }
}
