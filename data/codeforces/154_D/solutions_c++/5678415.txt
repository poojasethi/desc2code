#include <iostream>

using namespace std;

int main() {
  int x1, x2, a, b;
  cin >> x1 >> x2 >> a >> b;
  int state = -1, dist = x2 - x1, move = -1;
  if (a <= 0 && b >= 0) {
    state = (a <= dist && dist <= b) ? 1 : 2;
    move = x2;
  } else {
    bool switched = false;
    if (b < 0) {
      int c = b;
      b = -a;
      a = -c;
      dist = -dist;
      switched = true;
    }
    int rem = dist % (a + b);
    if (dist < 0 || (1 <= rem && rem < a) || b < rem)
      state = 2;
    else if (a <= rem && rem <= b) {
      state = 1;
      move = switched ? x1 - rem : x1 + rem;
    } else
      state = 0;
  }
  const string ans[] = { "SECOND", "FIRST", "DRAW" };
  cout << ans[state] << endl;
  if (state == 1)
    cout << move << endl;
  return 0;
}
