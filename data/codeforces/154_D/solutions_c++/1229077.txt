#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int main(){
	int n,m,p,q;
	scanf("%d%d%d%d",&n,&m,&p,&q);
	if (q<0){
		if (n<m){
			puts("DRAW");
			return 0;
		}
		p*=-1;
		q*=-1;
		swap(p,q);
		int d=n-m;
		if (d%(p+q)>=p && d%(p+q)<=q){
			printf("FIRST\n%d\n",n-d%(p+q));
		}else{
			if (d%(p+q)==0) puts("SECOND");else puts("DRAW");
		}
	}else
	if (p<=0){
		if (n<m){
			if (m-n<=q){
				printf("FIRST\n%d\n",m);
			}else{
				puts("DRAW");
			}
		}else{
			if (n-m<=-p){
				printf("FIRST\n%d\n",m);
			}else{
				puts("DRAW");
			}
		}
	}else{
		if (n>m){
			puts("DRAW");
			return 0;
		}
		int d=m-n;
		if (d%(p+q)>=p && d%(p+q)<=q){
			printf("FIRST\n%d\n",n+d%(p+q));
		}else{
			if (d%(p+q)==0) puts("SECOND");else puts("DRAW");
		}
	}
	return 0;
}
