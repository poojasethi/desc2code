#include <stdio.h>
#include <string.h>
#define MOD 1000003
int main(){
	int a,b,n,m;
	scanf("%d%d%d%d",&n,&m,&a,&b);
	if(n>m){
		if(m>=n+a&&m<=n+b)printf("FIRST\n%d\n",m);
		else if(a<=0&&b>=0)printf("DRAW\n");
		else if(a>0)printf("DRAW\n");
		else if((n-m)%(-a-b)==0)printf("SECOND\n");
		else if((n-m)%(-a-b)>-a||(n-m)%(-a-b)<-b)printf("DRAW\n");//eee
		else printf("FIRST\n%d\n",m+((n-m)/(-a-b)*(-a-b)));//.
	}
	else{
		if(m>=n+a&&m<=n+b)printf("FIRST\n%d\n",m);
		else if(a<=0&&b>=0)printf("DRAW\n");
		else if(b<0)printf("DRAW\n");
		else if((m-n)%(a+b)==0)printf("SECOND\n");//.
		else if((m-n)%(a+b)<a||(m-n)%(a+b)>b)printf("DRAW\n");//.
		else printf("FIRST\n%d\n",m-((m-n)/(a+b)*(a+b)));//.
	}
}
