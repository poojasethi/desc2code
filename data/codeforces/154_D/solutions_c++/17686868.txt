# include<iostream>
# include<cstdio>
using namespace std;
int main()
{
	int x,y,a,b;int a1,b1;
	while(~scanf("%d%d%d%d",&x,&y,&a1,&b1))
	{
		int flag=0;
		if(y<x)
		{
			flag=1;
			swap(x,y);
			b = -a1;
			a = -b1;
		}
		else
		{
			a = a1;
			b = b1;
		}
		if((y-x)>=a&&(y-x)<=b)
		{
			puts("FIRST");
			printf("%d\n",flag?x:y);
			continue;
		}
		if(a<=0)
		{
			puts("DRAW");
			continue;
		}
		if((y-x)%(a+b) == 0)
		{
			printf("SECOND\n");
			continue;
		}
		if((y-x)%(a+b)>=a&&(y-x)%(a+b)<=b)
		{
			puts("FIRST");
		    printf("%d\n",flag?y-(y-x)%(a+b):x+(y-x)%(a+b));
			continue;
		}
		puts("DRAW");
		
	}	
	return 0;
}
