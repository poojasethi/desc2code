#include<iostream>
using namespace std;
typedef long long int lli;
int main(){
    lli aux,d,x1,x2,a,b, lol = 1;
    cin >> x1 >> x2 >> a >> b;
    d = x2-x1;
    if(a <= 0 && 0 <= b){
        if (a <= d && d <= b){
	    cout << "FIRST" << endl << x2 << endl;
	    return 0;
	}
	else {
	    cout << "DRAW" << endl;
	    return 0;
	}
    }
    else {
        if (a < 0) {aux = -a; a = -b ; b = aux ; d = -d; lol = -lol;}
        if (d > 0) {
	    aux = d%(a+b);
	    if(aux == 0) {
	        cout << "SECOND" << endl;
	    } 
	    else if(a <= aux && aux <= b) {
                if(lol == -1){
		    aux = -aux;
                }
	        cout << "FIRST" << endl << x1 + aux << endl;
	    }
	    else {
	        cout << "DRAW" << endl;
	    }
	}
	else {
	    cout << "DRAW" << endl;
	}
    }
}
