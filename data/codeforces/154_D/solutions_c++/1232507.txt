#include <algorithm>
#include <iostream>
#include <cstdio>

typedef long long ll;

using namespace std;

ll x1, x2, a, b;

int main ()
{
	cin >> x1 >> x2 >> a >> b;

	ll l = min (abs (a), abs (b)), r = max (abs (a), abs (b));
	ll cur = abs (x1-x2)%(l+r);

	if (x1+b>=x2 && x1+a<=x2)
	{
		puts ("FIRST");
		cout << x2 << endl;
	}
	else
	if ((a<=0 && b>=0) || (x1<x2&&b<0) || (x1>x2&&a>0))
		puts ("DRAW");
	else
	{
		if (!cur)
		 	puts ("SECOND");
		else
		if (cur >= l && cur <= r)
		{
			puts ("FIRST");

			if (x1 < x2)
			 	cout << x1 + cur << endl;
			else
				cout << x1 - cur << endl;
		}
		else
			puts ("DRAW");
	}

	return 0;
}
