#include<iostream>
#include<vector>
#include<string>
#include<sstream>
#include<queue>
#include<stdio.h>

using namespace std;
int X,Y,A,B;
int main()
{
   // freopen("input.txt","r",stdin);

    cin>>X>>Y>>A>>B;

    int x1=X;
    int d=1;
    X=Y-X;

    if(X<0)
    {
        X*=-1; A*=-1; B*=-1; swap(A,B); d*=-1;
    }

    if(X<=B&&X>=A)
    {
        cout<<"FIRST"<<endl<<x1+X*d<<endl;
        return 0;
    }
    else if(A<=0)
    {
        cout<<"DRAW"<<endl;
        return 0;
    }

    X%=(A+B);
    if(X==0) cout<<"SECOND"<<endl;
    else if(X<A) cout<<"DRAW"<<endl;
    else if(X<=B)cout<<"FIRST"<<endl<<x1+X*d<<endl;
    else cout<<"DRAW"<<endl;


    return 0;
}
