#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"

int x,y,a,b,r,d,s;

int main()
{
  scanf("%d%d%d%d",&x,&y,&a,&b);
  if(x+a<=y && x+b>=y)
  {
    printf("FIRST\n%d",y);
    return 0;
  }
  if(!a || !b || a/abs(a)*b/abs(b)<0)
  {
    printf("DRAW");
	return 0;
  }
  if(a<0)
  {
    swap(a,b);
    a=-a;
    b=-b;
	x=-x;
	y=-y;
	s=-1;
  }
  else
    s=1;
  d=y-x;
  if(d<0)
    printf("DRAW");
  else 
  {
    r=d%(a+b);
    if(!r)
      printf("SECOND");
    else if(r>=a && r<=b)
	  printf("FIRST\n%d",(x+r)*s);
	else
	  printf("DRAW");
  }
  return 0;
}
