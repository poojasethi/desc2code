#include<iostream>
#include<stdio.h>
#include<string>
#include<string.h>
#include<algorithm>
#include<vector>
#include<cmath>
using namespace std;

int x1,x2,a,b;
int d;

void pp (int x)
{
	if (x2<x1) cout<<x1-x<<endl;
	else cout<<x1+x<<endl;
}

void doing ()
{
	cin>>x1>>x2>>a>>b;
	d=x2-x1;
	if (x2<x1)
	{
		a=-a;b=-b;d=-d;
		swap (a,b);
	}
	if (a<=0)
	{
	  if (b<=0){
	    cout<<"DRAW"<<endl;
	    return ;
	  }
	  if (d<=b)
	  {
	  	cout<<"FIRST"<<endl;
	  	pp(d);return ;
	  }
	  else{
	  	cout<<"DRAW"<<endl;return ;
	  }
	}
	d=d%(a+b);
	if (a>0)
	{
	  if (d==0)
	  {
	  	cout<<"SECOND"<<endl;return;
      }
      if (a<=d && d<=b)
      {
      	cout<<"FIRST"<<endl;
		pp(d);return;
      }
      cout<<"DRAW"<<endl;
	}
}

int main ()
{
	doing ();
	return 0;
}
