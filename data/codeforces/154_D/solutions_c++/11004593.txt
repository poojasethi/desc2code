#include<cstdio>
#define R(t)    return t,0;
#define F(x)    R(printf("FIRST\n%d",x))
#define S   R(puts("SECOND"))
#define D   R(puts("DRAW"))
int p,q,a,b,t;
int main(){
    scanf("%d%d%d%d",&p,&q,&a,&b);
    if(p+a<=q&p+b>=q)F(q)
    if(a==-b|p>q^-a>b|a<1&b>=0)D
    if((t=(q-p)%(b+a))==0)S else if(t>=a&&t<=b)F(p+t) else D
}
