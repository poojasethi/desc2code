#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,x,y,a,b,F;
int main() {
	scanf("%d%d%d%d",&x,&y,&a,&b);
	if (x>y) x=-x,y=-y,a=-a,b=-b,swap(a,b),F=1;
	if (x+a<=y&&y<=x+b) {
		printf("FIRST\n%d\n",F?-y:y);
		//for(;;);
		return 0;
	}
	if (a<=0) {
		printf("DRAW\n");
		//for(;;);
		return 0;
	}
	y=y-x;
	y%=a+b;
	if (!y) printf("SECOND\n");
	else if (y>=a&&y<=b) printf("FIRST\n%d\n",F?-x-y:x+y);
	else printf("DRAW\n");
	//for(;;);
	return 0;
}
