#include <cstdio>
#include <cstring>
#include <iostream>

using namespace std;

long long x1, x2, x, a, b, res, sgn;

int gao() {
	if (x + a <= 0 && x + b >= 0) {
		res = x2;
		return 1;
	} else if (a + b >= 0) {
		return 0;
	} else if (b >= 0) {
		return 0;
	} else {
		a = -a;
		b = -b;
		long long r = x % (a + b);
		if (r == 0) {
			return -1;
		} else if (r >= b && r <= a) {
			res = x1 - r * sgn;
			return 1;
		} else {
			return 0;
		}
	}
}

int main() {
	cin >> x1 >> x2 >> a >> b;
	x = x1 - x2;
	sgn = x < 0 ? -1 : 1;
	if (x < 0) {
		x = -x;
		a = -a;
		b = -b;
		swap(a, b);
	}
	res = 0;
	switch (gao()) {
	case -1: cout << "SECOND" << endl; break;
	case 0: cout << "DRAW" << endl; break;
	case 1: cout << "FIRST" << endl << res << endl; break;
	}
	return 0;
}
