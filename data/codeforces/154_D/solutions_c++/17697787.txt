#include <cstdio>

int x1, x2, a, b;

int main() {
	scanf("%d%d%d%d", &x1, &x2, &a, &b);
	int dis = x2 - x1, k = 1;
	if (x1 > x2) {
		dis = -dis;
		int t = a;
		a = -b;
		b = -t;
		k = -k;
	}
	if (b <= 0) puts("DRAW");
	else if (a <= 0) {
		if (dis <= b) printf("FIRST\n%d\n", x2);
		else puts("DRAW");
	} else {
		int r = dis % (a + b);
		if (!r) puts("SECOND");
		else if (a <= r && r <= b) printf("FIRST\n%d\n", x2 - k * dis / (a + b) * (a + b));
		else puts("DRAW");
	}
	return 0;
}
