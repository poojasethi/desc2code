#include <iostream>
using namespace std;

int abs(int x) { return x<0?x*-1:x; }

int main()
{
	int x1, x2, a, b;
	cin >> x1 >> x2 >> a >> b;
	int l = abs(x1-x2);
	bool f=false;
	if(x1>x2)
	{
		a*=-1;
		b*=-1;
		swap(a,b);
		f=true;
	}
	if(a<=l&&l<=b)
	{
		cout << "FIRST" << endl;
		cout << x2 << endl;
	}
	else if(a<=0)
		cout << "DRAW" << endl;
	else
	{
		if(l%(a+b)==0)
			cout << "SECOND" << endl;
		else if(l%(a+b)>=a&&l%(a+b)<=b)
		{
			cout << "FIRST" << endl;
			if(f)
				cout << x1-l%(a+b) << endl;
			else
				cout << x1+l%(a+b) << endl;
		}
		else 
			cout << "DRAW" << endl;
	}
	return 0;
}
