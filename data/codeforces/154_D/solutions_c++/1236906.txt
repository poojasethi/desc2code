#include<iostream>
#include<cstdlib>
using namespace std;
int x,y,a,b;
int ret(int f,int s)
{
	if(f==0)cout<<"DRAW"<<endl;
	if(f==1)cout<<"FIRST"<<endl<<s<<endl;
	if(f==2)cout<<"SECOND"<<endl;
	return 0;
}
int main()
{
	cin>>x>>y>>a>>b;
	if(y>=x+a&&y<=x+b)return ret(1,y);
	if((long long)a*b<=0||x<y==a<0)return ret(0,0);
	if((x-y)%(a+b)==0)return ret(2,0);
	if((y-x)%(a+b)>=min(a,b)&&(y-x)%(a+b)<=max(a,b))return ret(1,x+(y-x)%(a+b));
	return ret(0,0);
}
