#include <cstdio>
#include <cstdlib>
#include <algorithm>
using namespace std;

int main()
{
    int x1, x2, a, b;
    scanf("%d%d%d%d", &x1, &x2, &a, &b);

    if (x1 + a <= x2 && x1 + b >= x2) 
        return printf("FIRST\n%d\n", x2)&&0;

    bool replaced = 0;
    if (x1 > x2) {
        swap(x1, x2);swap(a, b);
        a = -a, b = -b, replaced = !replaced;
    }
    if (a <= 0) return puts("DRAW")&&0;
    int ans = (x2 - x1) % (a + b);
    if (ans == 0) return puts("SECOND")&0;
    if (ans < a || ans > b) return puts("DRAW")&&0;
    printf("FIRST\n%d\n", replaced?(x2-ans):(x1+ans));
    return 0;
}
