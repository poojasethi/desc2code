#include <iostream>
#include <cstdlib>
using namespace std;

void draw(){
  cout << "DRAW" << endl;
  exit(0);
}
int main(){
  long long x1,x2,a,b;
  cin >> x1 >> x2 >> a >> b;

  if(x1+a <= x2 && x2 <= x1+b){
    cout << "FIRST" << endl;
    cout << x2 << endl;
    return 0;
  }
  if(a*b <= 0) draw();

  bool rev = false;
  if(a < 0){
    rev = true;
    a = -a; b=-b; x1=-x1; x2=-x2;
  }

  if(x1 > x2) draw();

  int c = (x2 - x1) % (a + b);
  if(c==0){
    cout << "SECOND" << endl;
    return 0;
  }

  if(min(a,b) <= c && c <= max(a,b)){
    cout << "FIRST" << endl;
    cout << (rev ? -(x1+c): x1+c) << endl;
    return 0;
  }

  draw();
  return 0;
}
