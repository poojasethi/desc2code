x1,x2,a,b=map(int,raw_input().split())
if x1+a<=x2<=x1+b: print 'FIRST'; print x2
elif a<=0<=b: print 'DRAW'
elif a>0:
    if x1>x2: print 'DRAW'
    else:
        s=(x2-x1)%(a+b)
        if s==0: print 'SECOND'
        elif a<=s<=b: print 'FIRST';print x1+s
        else: print 'DRAW'
else:
    if x1<x2: print 'DRAW'
    else:
        s=abs(x1-x2)%abs(a+b)
        if s==0: print 'SECOND'
        elif -b<=s<=-a: print 'FIRST';print x1-s
        else: print 'DRAW'
