#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 200010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int p,e;
} a[N];
int i,j,k,n,m;
int id[N],f[N];
int64 sum,an;
inline bool cc1(const ww &a,const ww &b) {
	return a.p>b.p;
}
inline bool cc2(const int &A,const int &B) {
	return a[A].e>a[B].e;
}
inline void work(int x) {
	int i,j,s=0; int64 ss=0;
	For(i,1,n) if (a[i].p>x) s++;
	i=0;
	For(j,1,n-x) {
		for (i++;i<=n;i++) {
			int k=id[i];
			int F=a[k].p==x||a[k].p+1==x;
			if (F&&s+1==m) continue;
			s+=F; break;
		}
		if (i>n) break;
		ss+=a[id[i]].e;
	}
	an=min(an,sum-ss);
}
int main() {
	scanf("%d%d",&n,&m);
	if (m>n) {
		printf("0\n"); return 0;
	}
	For(i,1,n) scanf("%d%d",&a[i].p,&a[i].e),sum+=a[i].e,id[i]=i;
	sort(a+1,a+n+1,cc1);
	sort(id+1,id+n+1,cc2);
	int A=a[m].p; an=sum;
	if (A>n) {
		printf("-1\n"); return 0;
	}
	if (A+2<=n) work(A+2);
	if (A+1<=n) work(A+1);
	work(A);
	printf("%I64d\n",an);
	return 0;
}
