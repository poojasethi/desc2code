#include "iostream"
#include "cstdio"
#include "cstring"
#include "math.h"
#include "algorithm"
#define FOR(a,n,i) for(i =a;i<=n;i++)
using namespace std;
typedef long long ll;

struct node {
ll x;
ll y;
bool operator<(node a)const
{
    return x>a.x;
}
}date[200010];
ll a[200010],ai;
ll b[200010],bi;
ll min(ll a,ll b){
    if(a==-1)return b;
    else if(b==-1)return a;
    return a<b?a:b;
}
int main()
{
    ll n,k,i,t,j,p,ans,mk,tp;
    cin>>n>>k;
    FOR(1,n,i)cin>>date[i].x>>date[i].y;
    sort(date+1,date+n+1);
    t = date[k].x;
    if(t>n){cout<<-1<<endl;return 0;}
    ans=-1;
    FOR(1,n,i)if(i < k && date[i].x == t)p++;
    FOR(t,t+2,i){
        ai=0;bi=0;p=0;
        memset(b,0x3f3f3f3f,sizeof(b));
        memset(a,0x3f3f3f3f,sizeof(a));
        FOR(1,n,j){
            if(date[j].x >= i-1 && date[j].x <= i){
                    if(j>=k)p++;
                    a[ai++]=date[j].y;
            }
            else b[bi++]=date[j].y;
        }sort(a,a+ai);
        mk = 0;
        tp =p;
        FOR(0,ai-1,j){
        if(tp>0){mk+=a[j];--tp;}
        else b[bi++]=a[j];}

        sort(b,b+bi);
        FOR(0,i-p-1,j)mk+=b[j];
        ans = min(ans,mk);
    }
    cout<<ans<<endl;
    return 0;
}
