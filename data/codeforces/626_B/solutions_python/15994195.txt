from collections import Counter


n=input()
s = raw_input()
c= Counter(s)
v=sorted(c.values())
if len(v)==1:
        a = c.keys()[0]
elif len(v)==2 and v[-1]==1:
        for l in 'RGB':
                if l not in s:
                        a = l
elif len(v)==2 and v[0]==1:
        a="BRG".replace(c.most_common(1)[0][0],'')
else:
        a= "BRG"
print "".join(sorted(a))
