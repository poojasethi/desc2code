from collections import Counter

n = int(raw_input())
colors = raw_input()
cc = Counter(colors)
cols = set('RGB')
last = set(colors)
if len(last) == 1:
    print last.pop()
elif len(last) == 2:
    if cc.most_common(1)[0][1] > 1:
        if cc.most_common(2)[1][1] > 1:
            print 'BGR'
        else:
            print ''.join(sorted(list(cols.difference([cc.most_common(1)[0][0]]))))
    else:
        print cols.difference(colors).pop()
else:
    print 'BGR'
