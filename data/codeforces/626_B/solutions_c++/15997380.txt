#include <iostream>
#include <cstdio>

int main() {
    int n;
    int r, g, b;
    char code[201];
    scanf ("%d", &n);
    scanf ("%s", code);
    r = g = b = 0;
    for (int i=0; i<n; i++) {
        switch(code[i]) {
            case 'R': r++; break;
            case 'G': g++; break;
            case 'B': b++; break;
        }
    }
    if (!((b==0 && (g==n || r==n)) || b==n-1))
        putchar('B');
    if (!((g==0 && (r==n || b==n)) || g==n-1))
        putchar('G');
    if (!((r==0 && (g==n || b==n)) || r==n-1))
        putchar('R');
    putchar('\n');
    return 0;
}
