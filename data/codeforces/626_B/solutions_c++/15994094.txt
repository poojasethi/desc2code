#include <bits/stdc++.h>
using namespace std;
int a,b,c;
int main()
{
    int n;char cc;
    scanf("%d",&n);
    while(n--)
    {
        scanf(" %c",&cc);
        if(cc=='B')
            ++a;
        else if(cc=='G')
            ++b;
        else ++c;
    }
    if(b>=1&&c>=1)
        putchar('B');
    else if(!b&&!c&&a||(b>1||c>1)&&a)
        putchar('B');
    if(a>=1&&c>=1)
        putchar('G');
    else if(!a&&!c&&b||(a>1||c>1)&&b)
        putchar('G');
    if(a>=1&&b>=1)
        putchar('R');
    else if(!a&&!b&&c||(a>1||b>1)&&c)
        putchar('R');
    return 0;
}
