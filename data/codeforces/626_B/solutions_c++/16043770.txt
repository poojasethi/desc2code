#include "cstdio"
using namespace std;
int main()
{
    int n, b=0, g=0, r=0;
    char x;
    scanf("%d\n", &n);
    for (int i=0; i<n; i++)
    {
        scanf("%c", &x);
        switch (x)
        {
            case 'B': ++b; break;
            case 'G': ++g; break;
            case 'R': ++r; break;
        }
    }
    if ((b&&!g&&!r)||(g&&r)||(b&&g==0&&r>1)||(b&&g>1&&r==0))
        printf("B");
    if ((!b&&g&&!r)||(b&&r)||(b==0&&g&&r>1)||(b>1&&g&&r==0))
        printf("G");
    if ((!b&&!g&&r)||(b&&g)||(b==0&&g>1&&r)||(b>1&&g==0&&r))
        printf("R");
    return 0;
}
