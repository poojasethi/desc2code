#include"stdio.h"
bool check(int a,int b,int c){//&&与&之间的区别 
	if((b&&c)||(a&&(b+c-1)))
		return true;
	return false;
}
int main(){
	int n=205;
	int length;
	scanf("%d",&length);
	char str[n];
	int a=0,b=0,c=0;
	scanf("%s",str);
	for(int i=0;i<length;i++){
		if(str[i]=='B')
			a++;
		else if(str[i]=='G')
			b++;
		else if(str[i]=='R')
			c++;
	}
	if(check(a,b,c))
		printf("B");
	if(check(b,c,a))
		printf("G");
	if(check(c,a,b))
		printf("R");
	printf("\n");
	return 0;
}
