#include <bits/stdc++.h>
using namespace std;

int N, b, g, r;

main()
{
    scanf("%d\n", &N);
    for (char color; N--; )
    {
        scanf("%c", &color);
        if (color == 'B') b++;
        if (color == 'G') g++;
        if (color == 'R') r++;
    }
    if ((g && r) || (g == 0 && r == 0) || (b && max(g, r) > 1)) printf("B");
    if ((b && r) || (b == 0 && r == 0) || (g && max(b, r) > 1)) printf("G");
    if ((b && g) || (b == 0 && g == 0) || (r && max(b, g) > 1)) printf("R");
}
