#include <bits/stdc++.h>
using namespace std;
int n,a,b,c;///ei statement ta main er vitor dile vul output ase...
int main()
{
    string s;
    cin >> n>>s;
    for(int i = 0; i < s.size(); i++)
    {
        if(s[i] == 'B') a++;
        if(s[i] == 'R') b++;
        if(s[i] == 'G') c++;
    }
    if((b > 0 and c > 0) or ((b > 1 or c > 1) and a > 0) or (b == 0 and c == 0)) cout << 'B';
    if((b > 0 and a > 0) or ((b > 1 or a > 1) and c > 0) or (a == 0 and b == 0)) cout << 'G';
    if((c > 0 and a > 0) or ((c > 1 or a > 1) and b > 0) or (c == 0 and a == 0)) cout << 'R';
}
