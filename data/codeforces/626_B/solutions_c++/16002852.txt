#include<bits/stdc++.h>
using namespace std;

int main(){
	int i, j, k;
	int n;
	string s;
	while(cin >> n >> s){
		i = j = k = 0;
		for(n=0;n<s.size();n++){
			if(s[n]=='R')i++;
			else if(s[n]=='G')j++;
			else k++;	
		}
		if((i&&j)||(i>=2&&j==0&&k)||(j>=2&&i==0&&k)||(k&&!i&&!j))cout << 'B';
		if((i&&k)||(i>=2&&k==0&&j)||(k>=2&&i==0&&j)||(!i&&!k&&j))cout << 'G';
		if((k&&j)||(k>=2&&j==0&&i)||(j>=2&&k==0&&i)||(!j&&!k&&i))cout << 'R';
	}
}
