typedef unsigned long long ll;

int main()
{
std::ios::sync_with_stdio(false);
int n;
cin >> n;
string s;
cin >> s;
int r=0, g=0, b=0;
for(int i=0; i<n; i++)
{
    if(s[i]=='R') r++;
    else if (s[i]=='B') b++;
    else g++;
}
if(!(r==n || g==n || b==n-1))
    cout << "B";
if(!(b==n || r==n || g==n-1))
    cout << "G";
if(!(b==n || g==n || r==n-1))
    cout << "R";
return 0;
}
