#include <iostream>
#include <string>
using namespace std;
int main(){
	int n,r=0,g=0,b=0;
	cin>>n;
	char a [n];
	for(int i=0;i<n;i++){
		cin>>a[i];
		if(a[i]=='G')
			g++;
		else if(a[i]=='B')
			b++;
		else if(a[i]=='R')
			r++;
	}
	bool pr=0,pg=0,pb=0;
	if((b!=0 && r!=0) || (b==0 && r==0) || (g>0 && (b>1 || r>1)))
		pg=1;
	if((g!=0 && r!=0) || (g==0 && r==0) || (b>0 && (g>1 || r>1)))
		pb=1;
	if((b!=0 && g!=0) || (b==0 && g==0) || (r>0 && (g>1 || b>1)))
		pr=1;
	if(pb==1)
		cout<<'B';
	if(pg==1)
		cout<<'G';
	if(pr==1)
		cout<<'R'<<endl;
}
