#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n,nb=0,ng=0,nr=0,b,g,r,i;
    string s;
    cin>>n>>s;
    for(i=0;i<n;i++)
    {
        if(s[i]=='B')nb++;
        else if(s[i]=='G')ng++;
        else nr++;
    }
    if((ng==0 && nr==0)||(ng>0 && nr>0)||(nb>0 && max(ng,nr)>1))cout<<'B';
    if((nr==0 && nb==0)||(nr>0 && nb>0)||(ng>0 && max(nb,nr)>1))cout<<'G';
    if((nb==0 && ng==0)||(nb>0 &&ng>0)||(nr>0 && max(nb,ng)>1))cout<<'R';
    return 0;
}
