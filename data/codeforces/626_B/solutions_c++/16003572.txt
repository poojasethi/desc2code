#include <iostream>
#include <string>

using namespace std;

int main()
{
    int n;
    cin >> n;
    string s;
    cin >> s;

    int b = 0, g = 0, r = 0;
    for (size_t i = 0; i < n; ++i){
        b += s[i] == 'B';
        g += s[i] == 'G';
        r += s[i] == 'R';
    }

    if (g * r != 0 || g + r == 0 || b > 0 && g + r > 1)
        cout << "B";
    if (b * r != 0 || b + r == 0 || g > 0 && b + r > 1)
        cout << "G";
    if (b * g != 0 || b + g == 0 || r > 0 && b + g > 1)
        cout << "R";
    cout << endl;

    return 0;
}
