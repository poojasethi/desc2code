#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int n,x,y,z,X,Y,Z;
char s[205];
bool f[205][205][205];
void dfs(int x,int y,int z)
{
	if (f[x][y][z]) return;
	f[x][y][z]=1;
	if (x+y+z==1) {X|=x,Y|=y,Z|=z; return;}
	if (x&&y) dfs(x-1,y-1,z+1);
	if (x&&z) dfs(x-1,y+1,z-1);
	if (y&&z) dfs(x+1,y-1,z-1);
	if (x>=2) dfs(x-1,y,z);
	if (y>=2) dfs(x,y-1,z);
	if (z>=2) dfs(x,y,z-1);
}
int main()
{
	scanf("%d",&n);
	scanf("%s",s+1);
	for (int i=1;i<=n;i++) 
		if (s[i]=='B') x++; else if (s[i]=='G') y++; else z++;
	dfs(x,y,z);
	if (X) putchar('B');
	if (Y) putchar('G');
	if (Z) putchar('R');
	return 0;
}
