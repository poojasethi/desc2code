#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;
char str[210];
char ch[3] = {'B', 'G', 'R'};
int cnt[3];
int main() {
	int n;
	scanf("%d%s", &n, str);
	cnt[0] = cnt[1] = cnt[2] = 0;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < 3; j++) {
			if (str[i] == ch[j]) cnt[j]++;
		}
	}
	if ((cnt[1] && cnt[2]) || (cnt[1] + cnt[2] == 0) || (cnt[0] && (cnt[1] > 1 || cnt[2] > 1)))  printf("B");
	if ((cnt[0] && cnt[2]) || (cnt[0] + cnt[2] == 0) || (cnt[1] && (cnt[0] > 1 || cnt[2] > 1)))  printf("G");
	if ((cnt[0] && cnt[1]) || (cnt[0] + cnt[1] == 0) || (cnt[2] && (cnt[0] > 1 || cnt[1] > 1)))  printf("R");
}
