#include <bits/stdc++.h>
#define fi first
using namespace std;
int main () {
    int n;
    cin >> n;
    string s; 
	cin >> s;
    int G = 0 , R = 0 , B = 0;
    for (int i = 0 ; i < n ; ++i) {
                if (s[i] == 'B') B ++;
                if (s[i] == 'G') G ++;
                if (s[i] == 'R') R ++;
    }

    if ((B && (!G && !R)||(G && R)) || (B && (R > 1 || G > 1))) cout << "B";
    if ((G && (!B && !R)||(B && R)) || (G && (R > 1 || B > 1) )) cout << "G";
    if ((R && (!G && !B)|| (B && G))|| (R && (B > 1 || G > 1))) cout << "R";
    cout <<"\n";

    return 0;
}
