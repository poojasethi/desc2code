#include <cstdio>
using namespace std;
int t[512] ;
char s[211] ;
int main()
{
	int n;
	scanf("%d",&n) ;
	scanf("%s",s) ;
	for(int i=0;i<n;i++)
		t[s[i]]++ ;
	if(n==1)
	{
		printf("%s",s) ;
		return 0 ;
	}
	if(t['G']==n || t['B']==n || t['R']==n)
	{
		printf("%c",s[0]) ;
		return 0 ;
	}
	if((t['G']>0 && t['R']>0)|| t['R']>1 || t['G']>1 )
		printf("B") ;
	if((t['B']>0 && t['R']>0)|| t['R']>1 || t['B']>1 )
		printf("G") ;
	if((t['G']>0 && t['B']>0)|| t['B']>1 || t['G']>1 )
		printf("R") ;
}
