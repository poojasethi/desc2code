#include<bits/stdc++.h>
		 
using namespace std;
#define FOR(ii,aa,bb) for(int ii=aa;ii<=bb;ii++)

int n,r,g,b;
string str;
bool f(int a,int b,int c){
	if(!b and !c) return true;
	if(b and c) return true;
	c = max(b,c);
	if(!a) return false;
	if(c > 1) return true;
	return false;
} 
int main(){
	cin >> n >> str;
	FOR(i,0,n-1){
		if(str[i] == 'R') r++;
		if(str[i] == 'G') g++;
		if(str[i] == 'B') b++;
	}
	if(f(b,g,r)) printf("B");
	if(f(g,r,b)) printf("G");
	if(f(r,g,b)) printf("R");
}
