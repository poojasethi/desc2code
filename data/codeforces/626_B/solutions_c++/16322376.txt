#include<iostream>
#include<cstdio>
#include<vector>
#include<algorithm>
#include<cmath>
#include<iomanip>
#include<string>

using namespace std;
int main(){
	int r=0,g=0,b=0;
	int n;
	cin>>n;
	string s;
	cin>>s;
	for(int i=0;i<n;i++){
		if(s[i]=='R') r++;
		if(s[i]=='G') g++;
		if(s[i]=='B') b++;
	}
	if((g>0&&r>0)||(b>0&&(g+r!=1))) cout<<"B";
	if((r>0&&b>0)||(g>0&&(r+b!=1))) cout<<"G";
	if((g>0&&b>0)||(r>0&&(g+b!=1))) cout<<"R";
	return 0;
}
