#include <iostream>
#include <string>
#include <string.h>
#include <stdio.h>
using namespace std;

int main()
{ int n,r=0,g=0,b=0,i;
cin>>n;
char s[n];
cin>>s;
for(i=0;i<strlen(s);i++)
{
    if(s[i]=='R')
        r++;
    else if(s[i]=='B')
        b++;
       else if(s[i]=='G')
        g++;
}
if(r&&g || !r&&!g || r>1&&b || g>1&&b)
    cout<<"B";
    if(r&&b || !r&&!b || r>1&&g || b>1&&g)
    cout<<"G";
    if(b&&g || !b&&!g || g>1&&r || b>1&&r)
    cout<<"R";



    return 0;
}
