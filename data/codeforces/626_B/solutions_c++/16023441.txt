#include<bits/stdc++.h>
using namespace std;

int main()
{
    int n, r, g, b, i;
    string s;

    cin >> n >> s;
    r = g = b = i = 0;
    while (s[i] != '\0') {
        if (s[i] == 'R') r++;
        if (s[i] == 'G') g++;
        if (s[i] == 'B') b++;
        i++;
    }
    if ((b == n) || (g && r) || (g >= 2 && r == 0 && b) || (r >= 2 && g == 0 && b)) cout << "B";
    if ((g == n) || (r && b) || (r >= 2 && b == 0 && g) || (b >= 2 && r == 0 && g)) cout << "G";
    if ((r == n) || (g && b) || (g >= 2 && b == 0 && r) || (b >= 2 && g == 0 && r)) cout << "R";

    return 0;
}
