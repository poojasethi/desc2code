#include <iostream>
using namespace std;
int n, R, G, B;
int main() {

	char c;
	cin >> n;
	for (int i = 0; i < n; ++ i) {
		cin >> c;
		if (c == 'R') ++ R;
		else if (c == 'G') ++ G;
		else if (c == 'B') ++ B;
	}
	if ((R && G && B) || (R > 1 && B > 1) || (R > 1 && G > 1) || (G > 1 && B > 1))
		cout << "BGR\n";
	else if ((R == 1 && B == 1) || (G && (B + R) == 0))
		cout << 'G';
	else if ((R == 1 && G == 1) || (B && (G + R) == 0))
		cout << 'B';
	else if ((B == 1 && G == 1) || (R && (G + B) == 0))
		cout << 'R';
	else if (B > 1)
		cout << "GR";
	else if (G > 1)
		cout << "BR";
	else if (R > 1)
		cout << "BG";
}
