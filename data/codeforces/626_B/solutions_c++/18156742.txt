#include<bits/stdc++.h>
using namespace std;
#define MAXN 210
int main()
{
	int n,r=0,b=0,g=0;
	scanf("%d",&n);
	char s[MAXN];
	scanf("%s",s);
	for(int i=0;i<n;i++){
		if(s[i]=='R') r++;
		if(s[i]=='G') g++;
		if(s[i]=='B') b++;
	}
	if ((r&&g)||(!r&&!g)||(r>1&&b)||(g>1&&b)) cout<<'B';
	if ((r&&b)||(!r&&!b)||(r>1&&g)||(b>1&&g)) cout<<'G';
	if ((b&&g)||(!b&&!g)||(b>1&&r)||(g>1&&r)) cout<<'R';
	return 0;
}
