#include<iostream>

using namespace std;

int main() {

	int n,r=0,g=0,b=0;
	string v;

	cin>>n>>v;
	for(int i=0;i<n;i++) {

		v[i]=='R' ? r++ : r;
		v[i]=='B' ? b++ : b;
		v[i]=='G' ? g++ : g;
	}
	if ((r>0 && g>0) || (r>1 && b>0) || (g>1 && b>0) || (!r>0 &&  !g>0))
        cout<<'B';
	if ((r>0 && b>0) || (r>1 && g>0) || (b>1 && g>0) || (!r>0 &&  !b>0))
        cout<<'G';
	if ((b>0 && g>0) || (b>1 && r>0) || (g>1 && r>0) || (!b>0 &&  !g>0))
        cout<<'R';
}
