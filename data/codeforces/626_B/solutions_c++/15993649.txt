#include <iostream>
using namespace std;

int main()
{
	int n;
	string s;
	cin >> n >> s;

	int b = 0, g = 0, r = 0;
	for(int i = 0; i < n; ++i) {
		if (s[i] == 'B') ++b;
		if (s[i] == 'G') ++g;
		if (s[i] == 'R') ++r;
	}

	if((r >= 1 && g >= 1) || ((g > 1 || r > 1) && b >= 1) || (g == 0 && r == 0)) cout << 'B';
	if((r >= 1 && b >= 1) || ((r > 1 || b > 1) && g >= 1) || (r == 0 && b == 0)) cout << 'G';
	if((g >= 1 && b >= 1) || ((g > 1 || b > 1) && r >= 1) || (g == 0 && b == 0)) cout << 'R';
}
