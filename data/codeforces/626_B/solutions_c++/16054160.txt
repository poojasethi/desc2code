#include <stdio.h>
#include <stdlib.h>
#include <iostream>

using namespace std;

int N;
int r = 0, b = 0, g = 0;

bool good(int a, int b, int c)
{
	if( b == 0 && c == 0) return true;
	if(b > 0 && c > 0) return true;
	if(a > 0){
		if(b == 0 && c > 1) return true;
		if(c == 0 && b > 1) return true;
	}
	return false;
}

int main()
{
	cin>>N;
	for(int i=0;i<N;i++){
		char c;
		cin>>c;
		if(c == 'R') r++;
		if(c == 'B') b++;
		if(c == 'G') g++;
	}
	if(good(b, r, g)) printf("B");
	if(good(g, b, r)) printf("G");
	if(good(r, b, g)) printf("R");
	printf("\n");
}
