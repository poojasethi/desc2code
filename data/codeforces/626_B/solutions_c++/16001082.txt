#include <bits/stdc++.h>

using namespace std;
typedef long long ll;
#define N 100005

int main() {
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	int n,r=0,b=0,g=0;
	cin>>n;
	string s,ans="";
	cin>>s;
	for(int i = 0;i<n;i++){
		if(s[i] == 'R')
		r++;
		else if(s[i] == 'B')
		b++;
		else
		g++;
	}
	
	if(!((g == 0 &&!(b>0&&r>1))||(r == 0 &&!(b>0&&g>1))) || (r+g == 0))
	ans+="B";
	if(!((b == 0 &&!(g>0&&r>1))||(r == 0 &&!(g>0&&b>1))) || (r+b) == 0)
	ans+="G";
	if(!((g == 0 &&!(r>0&&b>1))||(b == 0 &&!(r>0&&g>1))) || (b+g) == 0)
	ans+="R";
	cout<<ans<<endl;
    return 0;
}
