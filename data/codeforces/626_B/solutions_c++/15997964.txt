#include <iostream>
using namespace std;

int main()
{
	int n, R = 0, G = 0, B = 0;
	string s, w;
	cin>> n >> s;

	for(int i = 0; i < n; i ++)	
		if(s[i] == 'R')
			R ++;
		else if(s[i] == 'G')
			G ++;
		else
			B ++;

	bool r = 0, g = 0, b = 0;
	if(R > 0)
		r = 1;
	if(G > 0)
		g = 1;
	if(B > 0)
		b = 1;

	if((r && g) || (!r && b && !g) || (b && (G > 1 || R > 1)))
		w += "B";
	if((r && b) || (!r && !b && g) || (g && (B > 1 || R > 1)))
		w += "G";
	if((b && g) || (r && !b && !g) || (r && (G > 1 || B > 1)))
		w += "R";

	cout<< w << endl;


}
