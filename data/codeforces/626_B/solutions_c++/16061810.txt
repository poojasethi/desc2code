#include <iostream>
using namespace std;

int main() {
	string str;
	int n;
	cin>>n;
	cin>>str;
	int ans=0;
	int r=0,g=0,b=0;
	for(int i=0;i<n;++i){
		if(str[i]=='R') ++r;
		if(str[i]=='G') ++g;
		if(str[i]=='B') ++b;
	}
	if(r&&g&&b) cout<<"BGR";
	else if(r==0&&g==0) cout<<"B";
	else if(g==0&&b==0) cout<<"R";
	else if(b==0&&r==0) cout<<"G";
	else if(r==1&&g==1) cout<<"B";
	else if(g==1&&b==1) cout<<"R";
	else if(b==1&&r==1) cout<<"G";
	else if(g+b==1) cout<<"BG";
	else if(b+r==1) cout<<"BR";
	else if(r+g==1) cout<<"GR";
	else cout<<"BGR";
	return 0;
}
