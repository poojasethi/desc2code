#include <bits/stdc++.h>

using namespace std;

int r,g,b;
bool R,G,B;

int main(){
	int n;
	scanf("%d",&n);
	string s;
	cin >> s;
	for(int i=0;i<n;i++){
		if(s.at(i)=='B')b++;
		if(s.at(i)=='G')g++;
		if(s.at(i)=='R')r++;
	}
	if(b && g && r)R=G=B=true;
	if(r>1 && (g || b))B=G=true;
	if(g>1 && (b || r))B=R=true;
	if(b>1 && (g||r))G=R=true;
	if(r && b)G=true;
	if(r && g)B=true;
	if(g && b)R=true;
	if(r && g<1 && b<1)R=true;
	if(g&& r<1 && b<1)G=true;
	if(b&& g<1 && r<1)B=true;
	if(B)printf("B");
	if(G)printf("G");
	if(R)printf("R");
	return 0;
}
