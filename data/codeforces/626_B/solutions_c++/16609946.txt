#include <cstdio>
#include <cstring>
#include <algorithm>
#define MAXN 210
using namespace std;

char str[MAXN],ans[3]={'R','G','B'};

int main()
{
	int n;
	scanf("%d",&n);
	scanf("%*c%s",str);
	int r=0,b=0,g=0;
	for(int i=0;i<n;i++){
		if(str[i]=='R') r++;
		else if(str[i]=='B') b++;
		else if(str[i]=='G') g++; 
	}
	if(b+g==1||(!r&&!b)||(!r&&!g)) ans[0]='\0';
	if(r+b==1||(!g&&!r)||(!g&&!b)) ans[1]='\0';
	if(r+g==1||(!b&&!r)||(!b&&!g)) ans[2]='\0';
	for(int i=2;i>=0;i--)
		if(ans[i])
		printf("%c",ans[i]);
	return 0;
}
