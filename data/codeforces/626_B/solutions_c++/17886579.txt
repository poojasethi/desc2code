#include <bits/stdc++.h>

using namespace std;

map<char,int>m;
string res;

void solve()
{
	if((m['R'] + m['G'] != 1 && m['B']) || (!m['B'] && m['R'] && m['G']))
		res += 'B';	
	if((m['R'] + m['B'] != 1 && m['G']) || (!m['G'] && m['R'] && m['B']))
		res += 'G';
	if((m['G'] + m['B'] != 1 && m['R']) || (!m['R'] && m['G'] && m['B']))
		res += 'R';
}

void input()
{
	int n; 
	cin >> n;
	char c;
	for(int i=0;i<n;++i){
		cin >> c;
		++m[c];
	}
}

int main()
{
	input();
	solve();
	cout << res << endl;
	return 0;
}
