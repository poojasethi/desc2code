#include<stdio.h>

int main()
{
    int n;
    scanf("%d",&n);
    char str[201];
    scanf("%s",str);
	int i,r=0,g=0,b=0;
	for(i=0;i<n;i++)
	{
		if(str[i]=='R') r++;
		else if(str[i]=='G') g++;
		else if(str[i]=='B') b++;
	}
	if(r+g==0) puts("B");
	else if(r+b==0) puts("G");
	else if(g+b==0) puts("R");
	else if(n>2 && r+g==1) puts("GR");
	else if(n>2 && r+b==1) puts("BR");
	else if(n>2 && g+b==1) puts("BG");
    else if(n==2 && r==1 && g==1) puts("B");
    else if(n==2 && r==1 && b==1) puts("G");
    else if(n==2 && g==1 && b==1) puts("R");
    else puts("BGR");

	return 0;
}
