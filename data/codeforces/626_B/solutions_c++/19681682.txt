#include<bits/stdc++.h>
using namespace std;
int main(){
    int n;
    string s;
    cin>>n>>s;
    int r,g,b;
    r = g = b = 0;
    for(int i=0;i<n;i++){
        if (s[i] == 'R') r++;
        if (s[i] == 'G') g++;
        if (s[i] == 'B') b++;
    }
    if(r and g or (!r and !g) or (b and (r >= 2 or g >= 2))){
        cout<<'B';
    }
    if (r and b or (!r and !b) or (g and (b >= 2 or r >= 2))){
        cout<<'G';
    }
    if (b and g or (!b and !g) or (r and (b >= 2 or g >= 2))){
        cout<<'R';
    }
}
