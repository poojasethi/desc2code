#include<iostream>
#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
int main()
{
	int n;
	char s[205];
	cin>>n>>s;
	int b=0,g=0,r=0;
	for(int i=0;i<n;i++)
	{
		if(s[i]=='B') b++;
		else if(s[i]=='R') r++;
		else if(s[i]=='G') g++;
	}
	if(b>0&&r>0&&g>0) 
	{
		cout<<"BGR"<<endl;return 0;
	}
	if((r>=1&&g>=1)||((r>1||g>1)&&b>=1)||(r==0&&g==0)) cout<<'B';
	if((r>=1&&b>=1)||((r>1||b>1)&&g>=1)||(r==0&&b==0)) cout<<'G';
	if((b>=1&&g>=1)||((b>1||g>1)&&r>=1)||(b==0&&g==0)) cout<<'R';
	return 0;
}
