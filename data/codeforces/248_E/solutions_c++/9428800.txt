#include <iostream>
#include <fstream>
#include <cstring>
#include <string>
#include <cstdlib>
#include <cstdio>
#include <algorithm>
#include <cmath>
#include <bitset>
#include <ctime>
#include <map>
#include <queue>
#include <set>
using namespace std;

const int maxn=100000+15;
const int maxe=100+10;
int n,x,sum[maxn],m,y,z;
double f[maxn][maxe];
int main()
{
	scanf("%d",&n);
	double ans=0;
	for (int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		sum[i]=x;
		f[i][x]=1;
        ans+=f[i][0];
	}
	scanf("%d",&m);
	while (m--)
	{
		scanf("%d%d%d",&x,&y,&z);
	    ans-=f[x][0];
	    for (int i=1,tmp=sum[x];i<=z;i++,tmp--)
	     for (int j=0;j<=100 && j<=tmp;j++)
	      f[x][j]=f[x][j]*(tmp-j)/tmp+f[x][j+1]*(j+1)/tmp;
		sum[y]+=z;
		sum[x]-=z;
		ans+=f[x][0];
		printf("%.11lf\n",ans);
	}
	return 0;
}
