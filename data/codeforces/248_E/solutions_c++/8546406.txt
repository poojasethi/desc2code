#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
double p[100010][110];
int a[100010],b[100010];
int n,q;
int main()
{
	double ans;
	int i,j,k,u,v,x,t;
	scanf("%d",&n);
	ans=0;
	for (i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		p[i][a[i]]=1;
		if (a[i]==0) ans++;
	}
	memcpy(b,a,sizeof(a));
	scanf("%d",&q);
	for (i=1;i<=q;i++)
	{
		scanf("%d%d%d",&u,&v,&x);
		ans-=p[u][0]; t=a[u];
		for (j=1;j<=x;j++)
		{
			for (k=0;k<=b[u];k++) p[u][k]=p[u][k]*(t-k)/t+p[u][k+1]*(k+1)/t;
			t--;
		}
		b[u]-=max(x-(a[u]-b[u]),0);
		a[u]-=x; a[v]+=x; ans+=p[u][0];
		printf("%.10lf\n",ans);
	}
	return 0;
}
