#include<cstdio>
#include<algorithm>
#define maxn 309
#define INF 1LL<<61
using namespace std;
__int64 dp[maxn][maxn],w[maxn][maxn];
int main()
{
    int n,m,K;
    scanf("%d%d%d",&n,&m,&K);
    for(int i=0;i<=n;i++)
        for(int j=0;j<=n;j++)
            w[i][j]=dp[i][j]=INF;
    for(int i=1;i<=m;i++)
    {
        int l,r;
        __int64 c;
        scanf("%d%d%I64d",&l,&r,&c);
        for(int j=l;j<=r;j++)
            w[j][r]=min(w[j][r],c);
    }
    dp[0][0]=0;
    for(int i=1;i<=n;i++)
        for(int j=0;j<=i;j++)
    {
        dp[i][j]=dp[i-1][j];
        for(int k=i-j+1;k<=i;k++)
           dp[i][j]=min(dp[i][j],dp[k-1][j-i+k-1]+w[k][i]);
    }
    __int64 ans=INF;
    for(int i=K;i<=n;i++)
        ans=min(dp[n][i],ans);
    if(ans<INF)
    printf("%I64d",ans);
    else
        printf("-1");
}
