#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
__int64 a[400][400];
__int64 f[400][400];
int main()
{
int n,m,k;
scanf("%d%d%d",&n,&m,&k);
memset(a,-1,sizeof(a));
memset(f,-1,sizeof(f));
int i,j,u;
for(i=0;i<m;i++)
  {
	int l,r;
	__int64 cost;
	scanf("%d%d%I64d",&l,&r,&cost);
	for(j=l;j<=r;j++)
	if(a[l][j]==-1||a[l][j]>cost)
	a[l][j]=cost;
  }
  f[0][0]=0;
  for(i=1;i<=n;i++)
  for(j=0;j<=i;j++)
  {

  	f[i][j]=f[i-1][j];
  	for(u=0;u<j;u++)
  	if(f[i-u-1][j-u-1]!=-1&&a[i-u][i]!=-1)
    if(f[i][j]==-1||f[i][j]>f[i-u-1][j-u-1]+a[i-u][i])
    f[i][j]=f[i-u-1][j-u-1]+a[i-u][i];
  }
  __int64 ans=1ll<<60;
  for(i=k;i<=n;i++)
  if(f[n][i]!=-1)
  ans=min(f[n][i],ans);
  if(ans==1ll<<60)printf("-1\n");
  else 
  printf("%I64d\n",ans);
}
