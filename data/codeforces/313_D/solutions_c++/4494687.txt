#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N1 310
#define N2 100010
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,lim;
int l[N2],r[N2],c[N2],h[N1][N1];
int64 f[N1][N1],inf,an;
int main() {
	scanf("%d%d%d",&n,&m,&lim);
	For(i,1,m) scanf("%d%d%d",&l[i],&r[i],&c[i]);
	For(i,1,n) {
		For(j,1,m) if (l[j]<=i&&r[j]>=i)
			if (!h[i][r[j]]||h[i][r[j]]>c[j]) h[i][r[j]]=c[j];
	}
	memset(f,120,sizeof(f));
	inf=f[0][0];
	For(i,0,n) f[i][0]=0;
	For(i,0,n)For(j,0,i) {
		f[i][j]=min(f[i][j],f[i-1][j]);
		if (f[i][j]<inf) {
			For(k,i+1,n) if (h[i+1][k]) f[k][j+k-i]=min(f[k][j+k-i],f[i][j]+h[i+1][k]);
		}
	}
	an=inf;
	For(j,lim,n)For(i,0,n) an=min(an,f[i][j]);
	printf("%I64d\n",an==inf?-1:an);
	//for(;;);
	return 0;
}
