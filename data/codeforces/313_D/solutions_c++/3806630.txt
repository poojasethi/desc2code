#include<cstdio>
#include<iostream>
using namespace std;
#define LL __int64
const LL inf=(LL)10000000*10000000;
LL dp[320][320];
LL d[320][320];
LL minz(LL a,LL b){return a<b?a:b;}
int main(){
    int n,m,k,a,b,c,i,j,o;
    cin>>n>>m>>k;
    for(i=0;i<=n;i++)for(j=0;j<=n;j++)d[i][j]=dp[i][j]=inf;
    for(i=0;i<m;i++){
        scanf("%d%d%d",&a,&b,&c);
        for(j=a;j<=b;j++)d[j][b]=minz(d[j][b],c);
        for(j=1;j<=b-a+1;j++)dp[b][j]=minz(dp[b][j],c);
    }
    for(i=1;i<=n;i++)
        for(j=1;j<=i;j++)
            for(o=1;o<i;o++){
                dp[i][j]=minz(dp[o][j],dp[i][j]);
                dp[i][j]=minz(dp[i][j],dp[o][j-i+o]+d[o+1][i]);
            }
    LL minn=inf;
    for(i=k;i<=n;i++)if(dp[n][i]<minn)minn=dp[n][i];
    if(minn==inf)printf("-1");else
    printf("%I64d",minn);
    return 0;
}
