# include <iostream>
# include <cstring>

using namespace std;

# define ll long long

const ll INF = 1LL<<60;

ll dp[301][301];
ll dist[301][301];

int n;

ll rec(int t,int filled){
	ll& res = dp[t][filled];
	if(res != -1) return res;
	if(filled <= 0) return 0;
	if(t == n) return INF;
	res = 0;
	res = rec(t+1,filled);
	for (int i = t; i < n; ++i){
		if(dist[t][i] != INF)
			res = min(res,dist[t][i] + rec(i+1,filled-(i-t+1)));
	}
	return res;
}

int init(){
	memset(dp,-1,sizeof(dp));
	for (int i = 0; i < 301; ++i){
		for (int j = 0; j < 301; ++j){
			dist[i][j] = INF;
		}
	}
	return 0;
}


int main(){
	int m,k;
	init();
	cin>>n>>m>>k;
	for (int i = 0; i < m; ++i){
		int l,r,c;
		cin>>l>>r>>c;
		l--,r--;
		for (int j = l; j <= r; ++j){
			dist[l][j] = min(dist[l][j],(ll) c);
		}
	}
	ll res = rec(0,k);
	if(res == INF) res = -1;
	cout<<res<<endl;
	return 0;
}
