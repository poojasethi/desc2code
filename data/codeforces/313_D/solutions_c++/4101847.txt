
#include <stdio.h>
#include <string.h>
#define ll long long
const ll inf=0x3f3f3f3f3f3f3f3fLL;
ll d[310][310],cost[310][310];
int main()
{
    int n,m,k,l,r;
    ll c;
    scanf("%d%d%d",&n,&m,&k);
    memset(d,0x3f,sizeof(d));
    memset(cost,0x3f,sizeof(cost));
    while(m--)
    {
        scanf("%d%d%I64d",&l,&r,&c);
        for(int i=l;i<=r;i++)
        if(c<cost[i][r])cost[i][r]=c;
    }
    d[0][0]=0;

    for(int i=1;i<=n;i++)
    {
        for(int j=0;j<=i;j++)
        {
            if(d[i-1][j]<d[i][j])
            d[i][j]=d[i-1][j];
            for(int p=1;p<=j;p++)
            {
                if(d[i-p][j-p]+cost[i-p+1][i]<d[i][j])
                d[i][j]=d[i-p][j-p]+cost[i-p+1][i];
            }
        }
    }
    if(d[n][k]==inf)puts("-1");
    else printf("%I64d\n",d[n][k]);
    return 0;
}
