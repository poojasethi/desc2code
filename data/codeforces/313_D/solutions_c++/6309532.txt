#include<iostream>
#include<cstring>
#include<cstdio>
#include<algorithm>
#include<queue>
#define LL __int64
#define N 310
#define INF 500000000000LL
#define MIN(x,y) ((x)<(y)?(x):(y))
using namespace std;
LL cost[N][N],dp[N][N];
bool cmp(int x,int y)
{return x>y;}
int main()
{
	int i,j,m,n,k,t,x,y,z;
	scanf("%d%d%d",&n,&m,&t);
	for(i=0;i<=n;i++)
		for(j=0;j<=n;j++)
			dp[i][j]=cost[i][j]=INF;
	for(i=0;i<m;i++)
	{
		scanf("%d%d%d",&x,&y,&z);
		for(j=x;j<=y;j++)
			cost[x][j]=MIN(cost[x][j],z);
	}
	dp[0][0]=0;
	for(i=1;i<=n;i++)
		for(j=0;j<=i;j++)
		{
			dp[i][j]=dp[i-1][j];
			for(k=1;k<=j;k++)
				dp[i][j]=MIN(dp[i][j],dp[i-k][j-k]+cost[i-k+1][i]);
		}
	if(dp[n][t]>=INF) dp[n][t]=-1;
	printf("%I64d\n",dp[n][t]);
}
