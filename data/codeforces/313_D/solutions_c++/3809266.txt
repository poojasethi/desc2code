#include<cstdio>
#include<cstring>
#include<string>
#include<algorithm>
#include<iostream>
using namespace std;

long long n, k, m, t, g[606][606], f[606][606], x, y, z, ans;

const long long lim=10000000000000000;


int main()
{
	cin >> n >> m >> k;
	for(int i = 0; i <= n; i++)
		for(int j = 0; j <= n; j++)
			g[i][j] = f[i][j] = lim;

	for(int i = 0; i < m; i++)
	{
		cin >> x >> y >> z;
		for(int j = x; j <= y; j++)
			g[j][y] = min(g[j][y], 1LL*z);
	}
	f[0][0] = 0;
	for(int i = 1; i <= n; i++)
		for(int j = 0; j <= n; j++)
		{
			f[i][j] = min(f[i][j], f[i-1][j]);
			for(int k = 1; k <= i; k++)
			{
				t = j+i-k+1;
				f[i][t] = min(f[i][t], f[k-1][j]+g[k][i]);
			}
		}
	ans = lim;
	for(int i = k; i <= n; i++)
		ans = min(ans, f[n][i]);
	if(ans == lim)
		printf("-1\n");
	else
		cout << ans << "\n";
		

	return 0;
}
