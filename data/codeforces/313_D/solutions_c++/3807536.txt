#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int N = 310;
typedef long long ll;
const ll inf=1LL<<60;
template<class T>void checkmin(T &a,T b){
	if(a>b)a=b;
}
ll cost[N][N];
ll dp[N][N];
int main(){
	int n,m,k;
	cin>>n>>m>>k;
	for(int i=0;i<N;i++)
		for(int j=0;j<N;j++)
			cost[i][j]=dp[i][j]=inf;
	for(int i=0;i<m;i++){
		int l,r;
		ll c;
		cin>>l>>r>>c;
		for(int j=l;j<=r;j++)
			checkmin(cost[j][r],c);
	}
/*	for(int i=1;i<=n;i++)
		for(int j=i;j<=n;j++)
			printf("cost[%d][%d] = %lld\n",i,j,cost[i][j]);
*/	dp[0][0]=0;
	for(int i=0;i<n;i++){
		dp[i][0]=0;
		for(int j=0;j<=i;j++){
			if(dp[i][j]==inf)continue;
			checkmin(dp[i+1][j],dp[i][j]);
			for(int k=1;i+k<=n;k++)
				checkmin(dp[i+k][j+k],dp[i][j]+cost[i+1][i+k]);
		}
	}
	ll ans=dp[n][k];
	if(ans==inf)ans=-1;
	cout<<ans<<endl;
	return 0;
}
