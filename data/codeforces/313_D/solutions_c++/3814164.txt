#include<cstdio>
#include<iostream>
#include<cstring>
#define oo 99999999999999
typedef long long ll;
using namespace std;

int n,m,k;
ll f[310][310],v[310][310],l,r,c,ans=oo;

int main()
{
	scanf("%d %d %d",&n,&m,&k);
	for (int i=1; i<=n; i++)
		for (int j=1; j<=n; j++)
			v[i][j]=oo;
	for (int i=1; i<=m; i++)
	{
		scanf("%I64d %I64d %I64d",&l,&r,&c);
		for (int g=l; g<=r; g++)
			for (int h=g; h<=r; h++)
				v[g][h]=min(v[g][h],c);
	} f[0][0]=0;
	for (int i=0; i<=n; i++)
		for (int j=i+1; j<=n; j++)
			f[i][j]=oo;
	for (int i=1; i<=n; i++)
	{
		for (int j=0; j<=i; j++)
		{
			f[i][j]=f[i-1][j];
			for (int g=i-j; g<i; g++)
				f[i][j]=min(f[i][j],f[g][j-i+g]+v[g+1][i]);
			ans=j<k?ans:min(ans,f[i][j]);
		}
	}
	ans=ans==oo?-1:ans;
	printf("%I64d",ans);
	return 0;
}
