#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#define MA 300000000005
using namespace std;
__int64 dp[305][305],f[305][305],c;
__int64 mmin(__int64 a,__int64 b)
{return a>b?b:a;}
int main() 
{
    int n,m,k,a,b;
    cin>>n>>m>>k;
    for(int i=0;i<=n;i++)
	for(int j=0;j<=n;j++)
		dp[i][j]=MA;
    for(int i=0;i<305;i++)
    for(int j=0;j<305;j++)
    f[i][j]=MA;
    while(m--) 
	{
        cin>>a>>b>>c;
        for(int i=a;i<=b;i++) 
		{
            f[a][i]=mmin(f[a][i],c);
        }
    }
    dp[0][0]=0;
    for(int i=1;i<=n;i++) 
	{
        for(int j=0;j<=i;j++) 
		{
            dp[i][j]=dp[i-1][j];
            for(int l=1;l<=j;l++) 
                dp[i][j]=mmin(dp[i][j],dp[i-l][j-l]+f[i-l+1][i]);
        }
    }
    if(dp[n][k]==MA) 
	{cout<<"-1"<<endl;return 0;}
    cout<<dp[n][k]<<endl;
}
