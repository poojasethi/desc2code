#include<iostream>
#include<cstring>
#include<cstdio>
#include<cstdlib>
using namespace std;
#define LL long long
const int N=310;
const LL inf=0x3f3f3f3f3f3f3f3fll;
int n,m,K;
LL f[N][N],w[N][N];
int main() {
	scanf("%d%d%d",&n,&m,&K);
	memset(w,0x3f,sizeof w);
	for(int i=1,x,y,z;i<=m;i++) {
		scanf("%d%d%d",&x,&y,&z);
		for(int j=x;j<=y;j++)
			w[j][y]=min(w[j][y],(LL)z);
	}
	memset(f,0x3f,sizeof f);
	f[0][0]=0;
	for(int i=1;i<=n;i++)
		for(int j=0;j<=min(i,K);j++) {
			f[i][j]=f[i-1][j];
			for(int k=0;k<i;k++)
				if(j>=i-k)
					f[i][j]=min(f[i][j],f[k][j-(i-k)]+w[k+1][i]);
		}
	if(f[n][K]==inf) f[n][K]=-1;
	printf("%I64d\n",f[n][K]);
	return 0;
}
