#include<iostream>
#include<cstdio>
#include<string>
#include<cmath>

using namespace std;
typedef long long ll;

const ll inf=1000000000000000ll;
const int max_n=310;
int n,m,K;
ll d[max_n][max_n];
ll dp[max_n][max_n];

int main()
{	
	scanf("%d%d%d",&n,&m,&K);
	for(int i=0; i<=n; i++)
		for(int j=0; j<=n; j++)
		{
			d[i][j]=inf;
			dp[i][j]=inf;
		}
	
	int l,r,c;
	for(int i=0; i<m; i++)
	{
		scanf("%d%d%d",&l,&r,&c);
		l--;r--;
		for(int j=l; j<=r; j++)
			d[j][r]=min(d[j][r],(ll)c);
	}
	
	for(int i=0; i<n; i++)
		for(int j=0; j<=i; j++)
		{
		
			dp[i][0]=0;
			dp[i+1][j]=min(dp[i+1][j],dp[i][j]);
			for(int k=i; k<n; k++)
				dp[k+1][j+k-i+1]=min(dp[k+1][j+k-i+1],dp[i][j]+d[i][k]);
		}
		
	ll res=inf;
	for(int i=K; i<=n; i++)
		res=min(res,dp[n][i]);
	if(res==inf)
		res=-1ll;
	cout<<res<<"\n";
	
	return 0;
}
