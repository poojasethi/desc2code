#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstring>
#include<cmath>
#define inf 1e15
#define min(x,y) ((x)<(y)?(x):(y))
using namespace std;
__int64 d[310][310],cost[310][310];
int main()
{
	int i,j,k,n,m,c,l,r,K;
	scanf("%d%d%d",&n,&m,&K);
	for(i=0;i<=n;i++)
		for(j=0;j<=n;j++)
		{
			d[i][j]=cost[i][j]=inf;
		}
		for(i=0;i<m;i++)
		{
			scanf("%d%d%d",&l,&r,&c);
			for(j=l;j<=r;j++)
			{
				//if(cost[j][r]>c)
					//cost[j][r]=c;
				if(cost[l][j]>c)
					cost[l][j]=c;
			}
		}
		d[0][0]=0;
		for(i=1;i<=n;i++)
		{
			for(j=0;j<=i;j++)
			{
				d[i][j]=d[i-1][j];
				for(k=1;k<=j;k++)
				{
					d[i][j]=min(d[i][j],d[i-k][j-k]+cost[i-k+1][i]);
				}
			}
		}
		if(d[n][K]>=inf)printf("-1\n");
		else
			printf("%I64d\n",d[n][K]);
}
