#include <bits/stdc++.h>

#define Max 309
#define INF 1000000000000000007
using namespace std;

typedef long long lld;


long long dp[Max][Max],mn[Max][Max];
int n,m,k;
int main(){

	cin >> n >> m  >> k;
	
	for( int i=0;i<Max;i++)
		for( int j=0;j<Max;j++)
			dp[i][j]=mn[i][j]=INF;
			
	for( int i=0;i<m;i++){
		int a,b;
		lld cs;
		cin >> a >> b >> cs;	
		for( int j=a;j<=b;j++)
			mn[a][j] = min( mn[a][j] , cs);
	}
	dp[0][0]=0;
	for( int i=1;i<=n;i++){	
		for( int j=0;j<=k;j++){
			
			dp[i][j]=min( dp[i][j] , dp[i-1][j] );
					
			for( int h=0;h<j;h++ ){
				if( i-h-1 >= 0 && j-h-1 >= 0 )
					dp[i][j] = min( dp[i][j] , dp[i-h-1][j-h-1] + mn[i-h][i]);
			}
		}
	}
	long long ans=dp[n][k];
	if( ans == INF )ans=-1;
	printf("%lld\n" ,ans);
	return 0;
}
