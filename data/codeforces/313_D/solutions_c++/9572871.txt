#include <bits/stdc++.h>
using namespace std;

const int MAXN = 300;
const int MAXM = 100000;
int jump[MAXN+5][MAXN+5];
long long dp[MAXN+5][MAXN+5];
const long long oops = 1e18;
int main(){
	 int n,m,k,s,e,c;
	 
	 cin>>n>>m>>k; 
	 memset(jump,-1,sizeof jump);
	 for(int a=0;a<m;a++){
		 cin>>s>>e>>c;
		 s--; e--;
		 for(int i=s;i<=e;i++){
			 if(jump[i][e]==-1 || jump[i][e]>c)
				jump[i][e]=c;
		 }
	 }
	 for(int i=0;i<=MAXN;i++)
		for(int j=0;j<=MAXN;j++)
			dp[i][j]=oops;
	dp[n][0]=0;
	
	 for(int holes=0;holes<=k;holes++)
		for(int i=n-1;i>=0;i--){
			
			long long &ret=dp[i][holes];
			
			ret=dp[i+1][holes];
			
			for(int j=i;j<n;j++)
				if(jump[i][j]!=-1 )
					ret=min(ret,dp[j+1][max(holes-(j-i+1),0)]+jump[i][j]);
		}
	 if(dp[0][k]==oops)dp[0][k]=-1;
	 cout<<dp[0][k]<<"\n";
}
