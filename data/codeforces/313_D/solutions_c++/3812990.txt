#include <stdio.h>
#include <stdlib.h>
#include <iostream>
#include <algorithm>

using namespace std;

#define	LL	long long int
#define	MAXN	310
#define	INF	99999999999999LL

int n, m, k;
LL cost[MAXN][MAXN], pd[MAXN][MAXN];

int main (void) {
	int i, j, l, r;
	LL c;

	while(scanf("%d %d %d", &n, &m, &k) != EOF) {
		for(i=0; i < MAXN; i++)
			for(j=0; j < MAXN; j++) cost[i][j] = pd[i][j] = INF;
		
		for(i=0; i < m; i++) {
			cin >> l >> r >> c;
			for(j=l; j <= r; j++) cost[l][j] = min(cost[l][j],c);
		}

		pd[0][0] = 0;
		for(i=1; i <= n; i++)
			for(j=0; j <= k; j++) {
				for(l=0; l < i; l++) pd[i][j] = min(pd[i][j], pd[l][j]); /* do nothing */
				for(l=0; l <= j; l++) if(i-l>=0 && i-l-1 >= 0 && j-l-1 >= 0) pd[i][j] = min(pd[i][j], cost[i-l][i] + pd[i-l-1][j-l-1]);
			}

		if(pd[n][k] < INF) cout << pd[n][k] << endl;
		else printf("-1\n");
	}

	return 0;
}
