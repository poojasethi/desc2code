#include<cstdio>
#include<algorithm>
#include<iostream>
#include<cstring>
using namespace std;

const int N = 300+7;
int n, m, k; 
long long f[N][N], c[N][N];

long long min(long long a, long long b) { return a < b ? a : b; }

void enter() {
	scanf("%d%d%d", &n, &m, &k);
	memset(c, 0x3f, sizeof c);
	for(int i = 0; i < m; ++i) {
		int l, r, w;
		scanf("%d%d%d", &l, &r, &w);
		for(int x = l; x <= r; ++x)
			for(int y = x; y <= r; ++y)
				c[x][y] = min(c[x][y], w);
	}
}

void solve() {
	memset(f, 0x3f, sizeof f);
	f[0][0] = 0;
	for(int i = 1; i <= n; ++i) {
		f[i][0] = 0;
		for(int j = 1; j <= min(i, k); ++j) {
			f[i][j] = f[i-1][j];
			for(int t = 1; t <= j; ++t)
				f[i][j] = min(f[i][j], f[i-t][j-t] + c[i-t+1][i]);
		}
	}
	cout << (f[n][k] >= 0x3f3f3f3f3f3f3f3f ? -1 : f[n][k]) << "\n";
}

int main() {
	enter();
	solve();
	return 0;
}
