#include<iostream>
using namespace std;
#define FOR(i,a,b) for(i=a;i<b;i++)
#define FORE(i,a,b) for(i=a;i<=b;i++)
#define MAX 300000000010
#define min(a,b) a<b?a:b
typedef long long ll;
int main(){
   ll c,C[307][307],minVal=MAX, A[307][307],ans;
   int n,m,k,i,j,x,y,z,l,r;
   cin>>n>>m>>k;
   FORE(i,0,n) FORE(j,0,n+1) C[i][j]=A[i][j]=MAX;
   FOR(x,0,m){ cin>>l>>r>>c; FORE(j,l,r) C[l][j]=min(C[l][j],c);}
   A[0][0]=0;
   FORE(x,1,n) FORE(y,0,x){ A[x][y]=A[x-1][y]; FORE(z,1,y) A[x][y]=min(A[x][y],A[x-z][y-z]+C[x-z+1][x]);}
   ans=A[n][k]>=MAX?-1:A[n][k];
   cout<<ans<<endl;
return 0;
}
