#include<cstdio>
#include<iostream>
using namespace std;
#define LL __int64
int n,m,k;
LL inf=1000000000000LL;
LL cost[305][305];
LL dp[305][305];
int main()
{
	int l,r,c,i,j,o;
	scanf("%d %d %d",&n,&m,&k);
	for(i=0;i<305;i++)
		for(j=0;j<305;j++)
			cost[i][j]=dp[i][j]=inf;
	for(i=0;i<m;i++)
	{
		scanf("%d %d %d",&l,&r,&c);
		for(j=l;j<=r;j++)
			cost[j][r]=min(cost[j][r],(LL)c);
	}
	dp[0][0]=0;
	for(i=1;i<=n;i++)
	{
		for(j=0;j<=i;j++)
		{
			for(o=j;o<=i;o++)
				dp[i][j]=min(dp[i][j],dp[o][j]);
			for(o=i-j+1;o<=i;o++)
				dp[i][j]=min(dp[i][j],cost[o][i]+dp[o-1][o-i+j-1]);
		}
	}
	//for(i=1;i<=n;i++)
	//	for(j=0;j<=i;j++)
	//		cout<<i<<" "<<j<<" "<<dp[i][j]<<endl;
	if(dp[n][k]==inf)
		printf("-1\n");
	else
		cout<<dp[n][k]<<endl;
	return 0;
}
