#include<iostream>
#include<cstdio>
#include<cstring>
#define Min(a,b) a>b?b:a
using namespace std;
long long dp[400][400],cost[400][400];
void solve(){
    int n,m,k;scanf("%d%d%d",&n,&m,&k);
    for(int i=0;i<=n;i++)
        for(int j=0;j<=n;j++)
            cost[i][j]=dp[i][j]=(1LL<<45);
    for(int i=1;i<=m;i++){
        int l,r;long long x;
        scanf("%d%d",&l,&r);
        scanf("%I64d",&x);
        for(int j=l;j<=r;j++)cost[l][j]=Min(x,cost[l][j]);
    }
    long long INF=dp[0][0];
    dp[0][0]=0;
    for(int i=1;i<=n;i++){
        for(int j=0;j<=i;j++){
            dp[i][j]=dp[i-1][j];
            for(int k=1;k<=j;k++){
                dp[i][j]=Min(dp[i][j],dp[i-k][j-k]+cost[i-k+1][i]);
            }
        }
    }
    if(dp[n][k]<INF) cout<<dp[n][k]<<endl;
    else cout<<"-1"<<endl;
}
int main(){
    solve();
    return 0;
}
