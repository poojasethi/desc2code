#include<cstring>
#include<algorithm>
#include<cstdio>
using namespace std;
typedef long long LL;
const int MAX = 400;
const LL inf =1e17+10;
LL dp[MAX][MAX];
LL cost[MAX][MAX];
int main()
{
	int n,m,k; LL l,r,c;
	while(scanf("%d %d %d",&n,&m,&k)==3)
	{
		for(int i=0;i<MAX;i++) for(int j=0;j<MAX;j++)
		dp[i][j]=inf,cost[i][j]=inf;
		for(int i=0;i<m;i++)
		{
			scanf("%I64d %I64d %I64d",&l,&r,&c);
			for(int i=l;i<=r;i++)
			cost[l][i]=min(cost[l][i],c);
		}
		dp[0][0]=0;
		for(int i=1;i<=n;i++)
		{
			for(int j=0;j<=i&&j<=k;j++)
			{
				dp[i][j]=min(dp[i-1][j],dp[i][j]);
				for(int k=1;k<=j;k++)
				dp[i][j]=min(dp[i][j],dp[i-k][j-k]+cost[i-k+1][i]);
			}
		}
		if(dp[n][k]>=inf) printf("-1\n");
		else printf("%I64d\n",dp[n][k]);
	}
	return 0;
}
