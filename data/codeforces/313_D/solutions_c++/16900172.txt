#include<iostream>
#include<algorithm>
#include<stdio.h>
#define MAXN 400


using namespace std;
int n,m,k;

long long INF=400000000000LL;
long long val[MAXN][MAXN];
long long f[MAXN][MAXN];

void init(){
	for(int i=0;i<=n;i++){
		for(int j=0;j<=n;j++){
			val[i][j]=INF;
			f[i][j]=INF;
		}
		f[i][0]=0;
	}
	
}

int main(){
	cin>>n>>m>>k;
	init();
	
	int li,ri;
	long long ci;
	for(int i=0;i<m;i++){
		cin>>li>>ri>>ci;
		for(int j=li;j<=ri;j++)
			val[li][j]=min(val[li][j],ci);
	}
	
	//f[i][j] minimum cost to fix j holes from 1->i
	for(int i=1;i<=n;i++){
		for(int j=1;j<=i;j++){
			f[i][j]=f[i-1][j];
			
			for(int t=1;t<=j;t++){
				f[i][j]=min(f[i][j],f[i-t][j-t]+val[i-t+1][i]);
			}
		}
	}
	if(f[n][k]==INF){
		cout<<-1;
	}else{
		cout<<f[n][k];
	}
	return 0;
}
