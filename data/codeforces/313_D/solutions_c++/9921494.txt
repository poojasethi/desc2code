#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;
#define N 100010
#define M 310
#define INF 300000000010
typedef long long ll;

ll cost[M][M];
ll dp[M][M];

int main()
{
    int n,m,s;
    cin >> n >> m >> s;
    for(int i=0; i<=n; i++)
        for(int j=0; j<=n; j++)
            dp[i][j] = cost[i][j] = INF;
    while(m--)
    {
        int x,y;
        ll c;
        cin >> x >> y >> c;
        for(int i=x; i<=y; i++)
            cost[x][i] = min(cost[x][i],c);
    }
    dp[0][0] = 0;
    for(int i=1; i<=n; i++)
        for(int j=0; j<=i; j++)
        {
            dp[i][j] = dp[i-1][j];
            for(int k=1; k<=j; k++)
                dp[i][j] = min(dp[i][j] , dp[i-k][j-k] + cost[i-k+1][i]);
        }
    if(dp[n][s] >= INF) dp[n][s] = -1;
    cout << dp[n][s] << endl;
    return 0;
}
