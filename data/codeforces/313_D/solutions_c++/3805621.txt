#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
typedef long long LL;
const LL inf=1LL<<60;
LL v[512][512],d[512][512];
int n,m,k;
main(){
  cin>>n>>m>>k;
  memset(v,0x3f,sizeof(v)),memset(d,0x3f,sizeof(d));
  for(int i=0,a,b,c;i<m&&cin>>a>>b>>c;++i)d[a][b]=min(d[a][b],c+0LL);
  LL sol=1LL<<60,z,w;
  v[1][0]=0;
  for(int i=1;i<=n;++i){
    for(int j=n;j-->i;)d[i][j]=min(d[i][j],d[i][j+1]);
    for(int j=0;j<i;++j)if((z=v[i][j])<inf){
      v[i+1][j]=min(v[i+1][j],z);
      for(int u=i;u<=n;++u){
        int x=j+u-i+1;
        w=v[u+1][x]=min(v[u+1][x],z+d[i][u]);
        if(x>=k)sol=min(sol,w);
      }
    }
  }
  cout<<(sol<1LL<<60?sol:-1)<<endl;
}
