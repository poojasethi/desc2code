#include<iostream>
#define ll long long
using namespace std;

ll money[301][301],dp[301][301],z;

int main()
{
	int x,y,n,m,k;
	cin>>n>>m>>k;
	for(int i=1;i<=n;i++)
	   for(int j=1;j<=n;j++)
	     money[i][j]=1000000000000000;
	for(int i=1;i<=m;i++)
	{
		cin>>x>>y>>z;
		for(int j=x;j<=y;j++)
		  for(int p=j;p<=y;p++)
		     money[j][p]=min(money[j][p],z);
	}
	for(int i=0;i<=n;i++)
	   for(int j=1;j<=k;j++)
	     dp[i][j]=1000000000000000;
	dp[0][0]=0;
	for(int i=1;i<=n;i++)
	   for(int j=1;j<=min(i,k);j++){
	   		dp[i][j]=dp[i-1][j];
			for(int p=1;p<=j;p++)
			   dp[i][j]=min(dp[i][j],dp[i-p][j-p]+money[i-p+1][i]);
		}
	/*for(int i=1;i<=n;i++)
	{
		for(int j=1;j<=min(i,k);j++)
		   cout<<dp[i][j]<<" ";
		cout<<endl;
	}*/
	cout<<((dp[n][k]<1000000000000000)?(dp[n][k]):-1)<<endl;
	return 0;
}
