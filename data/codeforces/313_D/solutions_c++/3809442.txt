#include <cstdio>
#include <cstring>
#include <algorithm>
#include <iostream>
#define LL long long
using namespace std;
const int N = 333;
LL n, m, k, l, r, c, val[N][N], dp[N][N];
void chg(LL &a, LL b){
	if (a == -1 || a > b) a = b;
}
int main(){
	cin >> n >> m >> k;
	for (int i = 0; i < m; i++){
		cin >> l >> r >> c;
		for (int i = l; i <= r; i++)
			if (val[i][r] == 0 || val[i][r] >= c) val[i][r] = c;
	}
	memset(dp, -1, sizeof(dp));
	dp[0][0] = 0;
	for (int i = 1; i <= n; i++){
		for (int j = 0; j <= k; j++)
			dp[i][j] = dp[i - 1][j];
		for (int j = 1; j <= k; j++)
			for (int p = i - 1; p >= i - j && p >= 0; p--)
				if (val[p + 1][i] && dp[p][j - i + p] != -1) chg(dp[i][j], dp[p][j - i + p] + val[p + 1][i]);
			for (int j = k; j >= 1; j--)
				if (dp[i][j] != -1) chg(dp[i][j - 1], dp[i][j]);
	}
	cout << dp[n][k] << endl;
}
