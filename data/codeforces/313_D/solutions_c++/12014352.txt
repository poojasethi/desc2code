#include <iostream>
using namespace std;

long long dp[1000][1000], Cost[1000][1000], INF = 99999999999999999, l, r, c;

int main()
{
	int n, m, k;
	cin >> n >> m >> k;
	for(int i=0;i<1000;i++)
		for(int j=0;j<1000;j++)
			dp[i][j] = Cost[i][j] = INF;
			
	for(int i=0;i<m;i++)
	{
		cin >> l >> r >> c;
		for(int i=l;i<=r;i++)
			Cost[l][i] = min(Cost[l][i], c);
	}

	dp[0][0] = 0;
	for(int i=0;i<=n;i++)
		for(int j=0;j<=k;j++)
		{
			for(int y=0;y<i;y++)
				dp[i][j] = min(dp[i][j], dp[y][j]);
			
			for(int y=0;y<=j;y++)
				if(i-y-1>=0 && j-y-1>=0)
				dp[i][j] = min(dp[i][j], dp[i-y-1][j-y-1] + Cost[i-y][i]);
		}
	if(dp[n][k] == INF)
		cout << -1 << endl;
	else
		cout << dp[n][k] << endl;
}
