#include <cstdio>
#include <map>
#include <cstring>
#include <algorithm>
#define maxn 301111
#define LL long long
int a[maxn];
std::map<int,int>last;
LL sum[maxn];
int n;
int main()
{
    scanf("%d",&n);
    sum[0]=0;
    for(int i=1;i<=n;++i)
    {
        scanf("%d",a+i);
        sum[i]=sum[i-1];
        if(a[i]>0)sum[i]+=a[i];    
    }
    LL max=-0x7ffffffffffLL;
    int l,r;
    for(int i=1;i<=n;++i)
        if(last.count(a[i])==0)
            last[a[i]]=i;
        else 
        {
            int t=last[a[i]];
            if(sum[i-1]-sum[t]+a[i]+a[t]>max)
            {
                max=sum[i-1]-sum[t]+a[i]+a[t];
                l=t; r=i;
            }
        }
    int num=0;
    for(int i=l+1;i<r;++i)
        if(a[i]<0)++num;
    printf("%I64d %d\n",max,l-1+n-r+num);
    for(int i=1;i<l;++i)
        printf("%d ",i);
    for(int i=l+1;i<r;++i)
        if(a[i]<0)printf("%d ",i);
    for(int i=r+1;i<=n;++i)
        printf("%d ",i);

    puts("");
    return 0;
}
