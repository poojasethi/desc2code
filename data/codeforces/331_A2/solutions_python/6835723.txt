#include <bits/stdc++.h>
using namespace std;

void maximize(long long &a, long long b){
	if(a<b) a=b;
}

map<int, int> mp;

int main(){
	//freopen("input.inp", "r", stdin);
	int n;
	scanf("%d", &n);
	int a[n+1];
	long long f[n+1];
	f[0]=0;
	long long ans=-1e18;
	for(int i=1; i<=n; i++){
		scanf("%d", a+i);
		if(a[i]<=0) f[i]=f[i-1];
		else f[i]=f[i-1]+a[i];
	}
	for(int i=n; i>=1; i--) if(mp[a[i]]==0) mp[a[i]]=i;
	for(int i=1; i<=n; i++)
		if(mp[a[i]]>i) {
			int j=mp[a[i]];
			long long t=a[i]+a[j]+f[j-1]-f[i];
			maximize(ans, t);
		}
	cout << ans;
	for(int i=1; i<=n; i++)
		if(mp[a[i]]>i){
			int j=mp[a[i]];
			long long t=a[i]+a[j]+f[j-1]-f[i];
			int cnt=0;
			if(t==ans){
				for(int k=i+1; k<j; k++) if(a[k]<=0) cnt++;
				printf(" %d\n", cnt+i-1+n-j);
				for(int k=1; k<i; k++) printf("%d ", k);
				for(int k=i+1; k<j; k++) if(a[k]<=0) printf("%d ", k);
				for(int k=j+1; k<=n; k++) printf("%d ", k);
				break;
			}
		}
}
