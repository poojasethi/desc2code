#include<iostream>
#include<cstring>
#include<set>
#include<map>
#include<algorithm>
using namespace std;

const long long INF = 1000ll*1000ll*1000ll*1000ll*1000ll;
const int MAX = 3*100*1000+7;
map<int,int> bg;
long long plt[MAX];
int a[MAX], n;

int main() {
	cin >> n;
	for (int i=0; i<n; i++) {
		cin >> a[i];
		if (i > 0)
			plt[i] = plt[i-1];

		if (a[i] > 0)
			plt[i] += a[i];
		bg[a[i]] = i;
	}
	
	long long ans = -INF;
	int xa = 0, xb = 0;
	for (int i=0; i<n; i++) {
		long long c = bg[a[i]], tmp = 0;
		if (c != i) {
			tmp += plt[c-1] - plt[i];
			tmp += (long long)(2*a[i]);
			if (tmp > ans) {
				ans = tmp;
				xa = i;
				xb = c;
			}
		}
	}
	int k = xa + (n-xb-1);
	for (int i=xa+1; i<xb; i++)
		if (a[i] < 0)
			k++;
	cout << ans << " " << k << endl;
	for (int i=0; i<xa; i++)
		cout << i+1 << " ";
	for (int i=xa+1; i<xb; i++)
		if (a[i] < 0)
			cout << i+1 << " ";
	for (int i=xb+1; i<n; i++)
		cout << i+1 << " ";

	cout << endl;

	return 0;
}
