#include<cstdio>
#include<algorithm>
#include<vector>
#include<map>
#include<iostream>
#define x first
#define y second
#define mp make_pair
#define pb push_back
using namespace std;
typedef long long LL;
typedef pair<int,int> pii;
const int NMax=301000,MMax=200;
int N,A[NMax];
map<int,int>S;
LL sum[NMax];
vector<int> R;
int main()
{
	scanf("%d",&N);
	for(int i=1;i<=N;i++) {
		scanf("%d",A+i);
		S[A[i]]=max(S[A[i]],i);
		if(A[i]>=0) sum[i]=sum[i-1]+A[i];
		else sum[i]=sum[i-1];
	}
	LL maxx=-(~0uLL>>1);
	pii ans;
	for(int i=1;i<=N;i++) {
		int t=S[A[i]];
		if(t>i) {
			LL x=sum[t]-sum[i-1];
			if(A[i]<0) x+=(LL)A[i]*2;
			if(x>maxx) {
				maxx=x;
				ans=mp(i,t);	
			}
		}
	}
	printf("%I64d ",maxx);
	for(int i=1;i<ans.x;i++)
		R.pb(i);
	for(int i=ans.x+1;i<ans.y;i++) if(A[i]<0) R.pb(i);
	for(int i=ans.y+1;i<=N;i++) R.pb(i);
	printf("%d\n",(int)R.size());
	for(int i=0;i<R.size();i++)printf("%d ",R[i]);puts("");
	return 0;
}
