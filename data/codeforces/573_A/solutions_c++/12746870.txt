#include<cstdio>
int n, i, x, p;

int main()
{
  scanf("%d", &n);
  
  for(i = 0; i < n; i++)
  {
    scanf("%d", &x);
    while(x % 2 == 0)
    {
      x /= 2;
    }
    while(x % 3 == 0)
    {
      x /= 3;
    }
    if (i != 0 && p != x){printf("NO\n");return 0;}
    p = x;
  }
  printf("YES\n");
  
return 0;
}
