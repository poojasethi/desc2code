#include<iostream>
using namespace std;
int i; int t;

int main(){
	cin>>i;
	int ar [i];
	for(int x=0;x<i;x++)
	{	
		cin>>t;
		while(t%2==0) t/=2;
		while(t%3==0) t/=3;
		ar[x]=t;
	}
	bool b=true;
	int check=ar[0];
	for(int x=1;x<i;x++)
		if(ar[x]!=check) b=false;
	
	if(b) cout<<"Yes";
	else cout<<"No";	
}
