def div2div3(num):
    while num % 2 == 0:
        num = num / 2
    while num % 3 == 0:
        num = num / 3
    return num

n = int(raw_input())
data = map(div2div3, map(lambda x: int(x), raw_input().split(' ')))

if all([data[i] == data[i+1] for i in xrange(0, n -1)]):
    print "Yes"
else:
    print "No"
