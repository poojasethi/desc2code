#include<iostream>
#include<string>
using namespace std;
int n,i,l,r,j,ans;
string s;
main()
{
 cin>>n;
  getline(cin,s);
 getline(cin,s);
 i=0;
 while(i<s.size())
  {
   l=1; 
   r=0;
  if(s[i]==' '&&i<s.size())
    i++;
   while(l<n && i<s.size())
    {
     l++;
     i++;
     if(s[i]=='.' || s[i]=='?' || s[i]=='!')
	  {
	 r++; 
	 j=i+1;
	 }
    }
   if(r==0)
   { 
   cout<<"Impossible"<<endl; 
    return 0; 
	 }
   ans++;
   i=j;
  }
 cout<<ans<<endl; 
}
