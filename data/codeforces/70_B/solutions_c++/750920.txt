#include<cstdio>
char s[1<<20];
//by xiaoshua
int n;
void work()
{
    int i=0,j,cnt=0;
    while(1)
    {
        while(s[i]==' ') i++;
        if(s[i]==0) break;
            bool f=0;
        for(j=i+n-1;j>=i;j--)
        {
            if(s[j]=='.'||s[j]=='?'||s[j]=='!')
            {
                f=1;
                cnt++; break;
            }
        }
            if(!f)
            {
                puts("Impossible"); return;
            }
            i=j+1;
    }
    printf("%d\n",cnt);
}
int main()
{
    scanf("%d",&n);
    gets(s);
    gets(s);
    work();
    return 0;
}
