#include <iostream>
using namespace std;
int pos[10001];
int lmax,ls,a,b,ans,i,np;
string s;
bool ok=true;
main(){
 
 cin>>lmax; getline(cin,s);
 getline(cin,s); ls=s.size();
 for (i=0;i<ls;i++)
  if (s[i]=='.' || s[i]=='!' || s[i]=='?')
   pos[++np]=i;
   


 pos[0]=-2; 
 a=b=0;   
 
 while (b<np){
  
  if (pos[b+1]-pos[a]-1<=lmax) b++;
   else {
    if (a==b) {ok=false; break;}
    ans++;
    a=b; 
   }
 }
 
 if (a<b) ans++;  
 
 if (ok) cout<<ans<<endl;
    else cout<<"Impossible"<<endl;
    
// system("pause");
 
 
 
 
       
       
}
