#include<stdio.h>
#include<string.h>
#include<algorithm>
using namespace std;
const int N=10005;
int n,m;
char le[N];
int main() {
	scanf("%d",&n);
	getchar(); gets(le);
	int i,k,l,r;
	bool tmp=false;
	for(i=k=r=l=0,m=-1;le[i];i++) 
	{
		if(le[i]=='.'||le[i]=='?'||le[i]=='!') 
		{
			r=i;
			if(r-m>n)
			{
				k++;
				m=l;
			}
			if(r-m>n) {puts("Impossible"); return 0;}
			tmp=false;
		}
		else if(!tmp&&!(le[i]==' '))
		{
			l=i-1; tmp=true;
		}
	}
	printf("%d\n",k+1);
	return 0;
}
