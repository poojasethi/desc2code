#include<iostream>
#include<vector>
#include<string>
using namespace std;
int main()
{
	int n,prev = 0;
	cin>>n;
	string s;
	getline(cin,s);
	vector<string> v;
	getline(cin,s);
	for(int i = 0; i < s.length(); i++)
		if (s[i] == '.' || s[i] == '!' || s[i] == '?')
		{
			v.push_back(s.substr(prev,i + 1 - prev));
               prev = i + 2;
		}
	int ans = 0,t = 0;
	for(size_t i = 0; i < v.size(); i++)
		if (t >= v[i].length() + 1) t -= v[i].length() + 1; else
		if (v[i].length() > n)
		{
			cout<<"Impossible";
			return 0;
		} else
		{
			++ans;
			t = n;
			t -= v[i].length();
		}
	cout<<ans;
	return 0;
}
