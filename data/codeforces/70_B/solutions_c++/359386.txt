#include <iostream>
#include <string>
#include <cstdio>

using namespace std;

int main(){
  int n;
  while(scanf("%d\n",&n)==1){ 
    string s;
    getline(cin,s);
    int last = 0;
    int start = 0;
    int count = 0;
    for(size_t pos = 0; pos < s.size(); pos++){
      if(s[pos] == '.' || s[pos] == '!' || s[pos] == '?'){
	if(pos - last  >= n+2){
	  cout << "Impossible" << endl;
	  goto next;
	}
	if(pos - start >= n){
	  count ++;
	  start = last + 2;
	  last = pos;
	}else{	  
	  last = pos;
	}
      }
    }
    count ++;
    cout << count <<endl;
  next:;
  }
  return 0;
}
