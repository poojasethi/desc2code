#include <iostream>
#include <vector>

using namespace std;

int main() {
  int n;
  cin >> n;
  string s;
  getline(cin, s);
  getline(cin, s);
  vector <int> Pt;
  Pt.push_back(-2);
  int y = s.size();
  
  for (int i = 0; i < y; i++)
    if (s[i] == '.' || s[i] == '!' || s[i] == '?')
      Pt.push_back(i);
  
  int x = Pt.size();
  
  for (int i = 1; i < x; i++)
    if (Pt[i] - Pt[i - 1] - 1 > n) {
      cout << "Impossible";
      return 0;
    }
  
  int l = 0, num = 1;
  
  for (int i = 1; i < x; i++)
    if (Pt[i] - Pt[l] - 1 > n) {
      num++;
      l = i - 1;
    }
  
  cout << num;
}
