#include <iostream>
#include <cstring>
#include <cstdio>
using namespace std;

char readin[10100];

main(){
	int i, n, last, now;
	scanf ( "%d", &n );
	gets( readin );
	gets( readin );
	last = 0;
	int ret = 0;
	int text = 0;
	for ( i = 0; readin[i]; i ++ )
		if ( readin[i] == '.' || readin[i] == '?' || readin[i] == '!' ){
			now = i - last + 1;
			if ( now > n ){
				printf( "Impossible\n" );
				return 0;
			}
			if ( now + text > n ){
				ret ++;
				text = now + 1;
			}
			else
				text += now + 1;
			last = i + 2;
		}
	ret ++;
	printf( "%d\n", ret );
}
