#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n;

	scanf("%d", &n);
	getchar();

	string a;
	getline(cin, a);
	int l = a.length();

	vector <int> v;
	
	int p = -1;
	for(int i = 0; i < l; i++) {
		if(a[i] == '.' || a[i] == '!' || a[i] == '?') {
			v.push_back(i-p);
			p = i+1;
		}
	}


	for(int i = 0; i < v.size(); i++) {
		if(v[i] > n) {
			printf("Impossible\n");
			return 0;
		}
	}

	int i = 0, ans = 0;
	l = v.size();
	while(i < l) {
		p = v[i++];
		while(i < l && p+v[i]+1 <= n) {
			p += v[i]+1;
			i++;
		}
		ans++;
	}
	printf("%d\n", ans);

	return 0;
}
