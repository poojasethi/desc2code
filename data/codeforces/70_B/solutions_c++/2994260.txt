#include <cstdio>
#include <vector>
#include <cstring>
#include <string>
#include <queue>
using namespace std;
char msg[10010];
int main() {
    int n;
    scanf("%d", &n);
    getchar();
    gets(msg);
    int len = strlen(msg), l=-1, t=n, m=0;
    for (int i=0; i<len; ++i) {
        --t;
        if (t+1==n && msg[i]==' ') ++t;
        if (t<0) {
            if (l!=-1) { ++m; t=n; i=l+1; l=-1; }
            else { m = 0; break; }
        }
        if (msg[i]=='.' || msg[i]=='?' || msg[i]=='!') l = i;
    }
    if (l!=-1) ++m;
    if (m==0) printf("Impossible");
    else printf("%d", m);
    return 0;
}
