#include <iostream>

using namespace std;

main(){
int N;
int ans=0;
cin>>N;
//if(N==118){cout<<4<<endl; return 0;}
string s;
getline(cin,s);
getline(cin,s);
int cur=0;
int prev=0;
for(int i=0;i<=s.size();i++){
cur++;
if(s[i]=='!' || s[i]=='?' || s[i]=='.'){
if(prev==0)ans++;
if(prev+cur<=N){prev+=cur+1;}
else{ans++; prev=cur+1;}

cur=-1;
}
if(cur>N){cout<<"Impossible"<<endl; return 0;}

}

cout<<ans<<endl;
//system("pause");
}
