#include<bits/stdc++.h>
using namespace std;

string str;
int n,l,r,ans;

int main(){
	cin >> n;
	getline(cin,str);
	getline(cin,str);
	int len = str.size();
	while(l < len){
		r = l + n - 1;
		while(l < r){
			if(str[r] == '?' or str[r] == '.' or str[r] == '!'){
				l = r + 2;
				break;
			}
			else r--;
		}
			if(l == r){
				cout << "Impossible"<<endl;
				return 0;
			}
			else
				ans++;
		
	}
	cout << ans << endl;
}
