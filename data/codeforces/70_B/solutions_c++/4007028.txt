#include <cstdio>
int main(int argc, char const *argv[])
{
	int n;
	scanf("%d ", &n);
	int r = 0;
	int line = 0, sentence = 0;
	char c;
	while (c = getchar()) {
		if (c == '\n') break;
		if (c == ' ' && sentence == 0) continue;
		++sentence;
		if (c == '.' || c == '?' || c == '!') {
			if (sentence > n) {
				puts("Impossible");
				return 0;
			}
			if (line == 0) {
				line = sentence;
			} else {
				if (line + sentence + 1 <= n) {
					line += 1 + sentence;
				} else {
					++r;
					line = sentence;
				}
			}
			sentence = 0;
		}
	}
	if (line) ++r;
	printf("%d\n", r);
	return 0;
}
