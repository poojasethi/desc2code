#include <bits/stdc++.h>
using namespace std;

int main(){

#ifndef ONLINE_JUDGE
    freopen("read.txt","r",stdin);
//  freopen("out.txt","w",stdout);
#endif

    int n,r=0,w=0;
    string s;
    cin>>n;
    getchar();
    getline(cin,s);
    while(s.length()){
        int p=s.find_first_of(".?!");
        string t = s.substr(0,p+1);
        if(t.length()>n){
            cout<<"Impossible";
            return 0;
        }
        if(t.length()+w>n)
            w=0;
        r+=!w;
        w+=t.length()+1;
        s.erase(0,min(s.length(),t.length()+1));
    }
    cout<<r;
    return 0;
}
