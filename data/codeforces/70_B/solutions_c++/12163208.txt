#include <bits/stdc++.h>

using namespace std;

char text[12345];

int main() {
  int n;
  scanf("%d", &n);
  getchar();
  gets(text);
  int cur = 0, g = 0, f = -1, ans = 1;
  for (int i = 0; text[i]; i++) {
    if (text[i] == '.' || text[i] == '?' || text[i] == '!') {
      g = i + 1;
    }
    if (++cur == n) {
      if (g && g - f) {
        cur = cur - (g - f); ans++;
        f = g;
        if (!text[i + 1]) ans--;
      } else {
        puts("Impossible");
        return 0;
      }
    }
  }
  printf("%d", ans);
}
