#include<stdio.h>
#include<algorithm>
#include<iostream>
using namespace std;
char s[10001];
int main(void)
{
    int n,i,j,k;
    scanf("%d", &n);
    gets(s);
    gets(s);
    int ans = 0;
    for (i=0; s[i]; ans++)
    {
        for (j = i, k=0; j < i+n; j++)
            if (s[j] == '.' || s[j] == '!' || s[j] == '?')
                k = j;
        if (!k)
        {
           puts("Impossible");
           return 0;
        }
        i = k+2;
    }
    printf("%d", ans);

    return 0;
}
