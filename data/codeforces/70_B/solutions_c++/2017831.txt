#include <cstdio>
#include <cstring>

int main()
{
	bool possible = true;
	int i, n, messages = 0, last = 0;
	char st[10001];
	scanf("%d", &n);
	getchar();
	gets(st);
	int tam = strlen(st);
	while (possible)
	{
		if( n + last >= tam)
		{
			messages++;
			break;
		}
		for(i = last + n - 1; i >= last; i--)
		{
			if(st[i] == '!' || st[i] == '?' || st[i] == '.')
			{
				last = i + 2;
				messages++;
				break;
			}
			if(i == last)
				possible = false;
		}
	
	}
	if(possible)
		printf("%d", messages);
	else
		printf("Impossible");
}
