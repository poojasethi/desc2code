#include <iostream>
#include <string>
#include<cstdio>
using namespace std;

int main()
{
    int n,count=0,i,j,k=0;
    string s;
    cin>>n;
    getchar();
    getline(cin,s);
    for(i=0;i<s.length();i++)
    {
        int a=0,ch=0;
        for(int j=i;j<n+i && j<s.length();j++)
            if(s[j]=='.' || s[j]=='?' || s[j]=='!')
            {
                ch=1;
                a=j;
                //cout<<a<<endl;
            }
        if(ch==0)
        {
            cout<<"Impossible"<<endl;
            return 0;
        }
        count++;
        i=a+1;
    }
    cout<<count<<endl;
    return 0;
}
