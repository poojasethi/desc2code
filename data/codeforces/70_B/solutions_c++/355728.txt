#include <iostream>
#include <fstream>

using namespace std;

char s[1<<14];

int main()
{
    int n;
    cin >> n;
    cin.getline(s, sizeof(s));
    cin.getline(s, sizeof(s));

    int ans = 0;
    for (int i=0, j, k; s[i]; ans++)
    {
        for (j=i, k=0; j < i+n; j++)
            if (s[j] == '.' || s[j] == '!' || s[j] == '?')
                k = j;
        if (!k)
        {
            cout << "Impossible" << endl;
            return 0;
        }
        i = k+2;
    }
    cout << ans << endl;

    return 0;
}
