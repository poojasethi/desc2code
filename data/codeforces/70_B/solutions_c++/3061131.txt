#include<iostream>
#include<vector>
using namespace std;
vector<int>vec;
int main()
{
	int len,now=0,res=1;
	bool f=true;
	string tmp;
	cin>>len;
	while(cin>>tmp)
	{
		now+=tmp.size();
		if(tmp[tmp.size()-1]=='?' || tmp[tmp.size()-1]=='!' || tmp[tmp.size()-1]=='.')
		{
			vec.push_back(now);
			if(now>len)f=false;
			now=0;
		}
		else now++;
	}
	if(!f){cout<<"Impossible";//system("pause");
	return 0;}
	for(int i=0;i<vec.size();i++)
	{
		if(now+vec[i]>len)
		{
			res++;
			now=vec[i]+1;
		}
		else{
			now+=vec[i]+1;
		}
	}
	cout<<res;
	//system("pause");
	return 0;
}
