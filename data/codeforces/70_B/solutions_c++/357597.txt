#include <iostream>
#include <vector>
#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int main(){
	int n;
	int i,j,k;
	vector<int> v;
	char s[10010];
	
	cin>>n;
	getchar();
	gets(s);
	for(i=0,k=1;s[i] != '\0';i++,k++){
		if(s[i] == '.' || s[i] == '?' || s[i] == '!'){
			v.push_back(k);
			k=0; i++;
		}
	}
	
	for(i=0,k=1,j=n;i<v.size();i++){
		if(v[i] > n){
			cout<<"Impossible"<<endl;
			return 0;
		}
		if(v[i] <= j)
			j -= (v[i] + 1);
		else{
			k++;
			j = n - (v[i]+1);
		}
	}
	cout<<k<<endl;
	return 0;
}
