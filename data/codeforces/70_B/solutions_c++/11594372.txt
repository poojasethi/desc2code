#include <iostream>
#include <cstdio>

using namespace std;

int main()
{

    int n,ans=1,l=-1,sc=0,i,p=0;
    string j;

    cin >> n;

    getchar();

    getline(cin,j);

    for(i=0;i<j.length();i++)
    {
        sc++;
        if(sc>n)
        {
            if(l==-1) {cout << "Impossible" << endl; return 0;}
            ans++;
            sc = i-l-1;
            l=-1;
        }

        if(j[i]=='.' || j[i]=='?' || j[i]=='!') l=i;

    }


    cout << ans << endl;

    return 0;
}
