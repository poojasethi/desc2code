#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
#define p(x) cout<<#x<<":"<<x<<"\n"

int n,i,z,s,j;
char str[10001];

int main()
{
  scanf("%d",&n);
  gets(str);
  gets(str);
  for(i=0,z=strlen(str);i<z;)
  {
    for(j=min(i+n-1,z-1);j>=i && str[j]!='.' && str[j]!='!' && str[j]!='?';j--);
    if(j<i)
      break;
    i=j+2;
    s++;
  } 
  if(i<z)
    printf("Impossible\n");
  else
    printf("%d\n",s);
  return 0;
}
