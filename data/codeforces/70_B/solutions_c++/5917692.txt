#include <stdio.h>
#include <algorithm>
using namespace std;
char a[10005];
int dp[10005];

int main(void) {
	int n;
	scanf("%d", &n);
	gets(a);
	gets(a+1);
	const int INF = 100000000;
	for(int i=1;a[i];i++) dp[i]=INF;
	dp[0] = 0;

	int l=1;
	for(int i=1;a[i];i++,l++) {
		if(dp[i-1]==INF) continue;
		if(a[i]==' ') dp[i] = min(dp[i], dp[i-1]);
		for(int j=0;j<n && a[i+j];j++) {
			if(a[i+j]=='.' || a[i+j]=='?' || a[i+j]=='!') {
				dp[i+j] = min(dp[i+j], dp[i-1] + 1);
			}
		}
	}
	if(dp[l-1] == INF)
		printf("Impossible\n");
	else printf("%d\n", dp[l-1]);
	return 0;
}
