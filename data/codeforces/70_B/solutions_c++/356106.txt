#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

int n,m,ret,cnt,S;
char s[10005];

int main(){

	scanf("%d",&m);
	gets(s);
	gets(s);
	n = strlen(s);

	ret = 0;
	cnt = 0;
	S = -1;
	for (int i=0; i<n; i++){
		if (s[i]=='!'||s[i]=='?'||s[i]=='.'){
			cnt++;
			if (cnt>m){
				ret=-1;
				break;
			}
			
			if (S+cnt+1>m){
				S=cnt;
				ret++;
			}
			else S+=cnt+1;

			cnt=0;
		}
		else if (s[i]==' '){
			if (cnt) cnt++;
		}
		else cnt++;
	}
	
	if (ret==-1) printf("Impossible\n");
	else{
		if (S!=-1) ret++;
		printf("%d\n",ret);
	}

	return 0;
}
