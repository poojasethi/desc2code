#include <iostream>
#include <cstdio>
#include <string>
using namespace std;
int n,l,r,ans;
string s;
int main()
{
	cin>>n;
	getline(cin,s);
	getline(cin,s);
	while (l<s.size()){
	r=l+n-1;
	while (r>l){
	if (s[r]=='.' || s[r]=='?' || s[r]=='!'){
		l=r+2; 
		break; }
	else r--;
	}
	if (r==l) {cout<<"Impossible";
		return 0;}
	else ans++;
	}
	cout<<ans;
	return 0;
}
