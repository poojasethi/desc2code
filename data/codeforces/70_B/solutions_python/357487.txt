import sys;
import re;

n = int(sys.stdin.readline())
t = sys.stdin.readline().strip()

#print t

l = [len(x) for x in re.split('[.!?]', t)[:-1]]
l[0] += 1

#print l

for x in l:
  if (x > n):
   print "Impossible"
   sys.exit(0)

ln = l[0]
r = 1

for x in l[1:]:
  if (x + 1 + ln <= n):
    ln += x + 1
  else:
    ln = x
    r += 1
print r
