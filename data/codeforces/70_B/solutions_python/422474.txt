import re
import sys

n = int(sys.stdin.readline())
line = sys.stdin.readline()

sents = map(lambda x: x.strip(),  re.findall(r'[A-Za-z\ ]+[\.\!\?]', line))

res = 0
i = 0
while i < len(sents):
    cur = 0
    if len(sents[i]) > n:
        print 'Impossible'
        exit(0)
    cur = cur + len(sents[i])
    i = i + 1
    while i < len(sents) and (cur + 1 + len(sents[i]) <= n):
        cur = cur + 1 + len(sents[i])
        i = i + 1
    res = res + 1
        
print res
