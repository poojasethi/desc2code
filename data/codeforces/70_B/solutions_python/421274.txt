#!/usr/bin/env python
# -*- coding: utf-8 -*-

import re

def main():
    n = int(raw_input())
    t = raw_input()
    r = re.findall("(\w[a-zA-Z.?! ]{0,%d}[.?!]{1})" % (n-2), t)
    if t.replace(" ", "") != "".join(r).replace(" ", ""):
        print "Impossible"
    else:
        print len(r)
    return 0

if __name__ == '__main__':
    main()
