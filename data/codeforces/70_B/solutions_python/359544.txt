import re

n = int(raw_input())
line = raw_input()

impossible = False
count = 1
running = 0
for a in line.split('.'):
  for b in a.split('?'):
    for c in b.split('!'):
      if c:
        l = len(c.lstrip().rstrip()) + 1
        if not running:
          if l > n:
            impossible = True
          running = l
        else:
          if running + l + 1 > n:
            count += 1
            if l > n:
              impossible = True
            running = l
          else:
            running += l + 1

if impossible:
  print 'Impossible'
else:
  print count
