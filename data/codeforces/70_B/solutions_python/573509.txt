n = int(raw_input())
a = map(len, (" " + raw_input()).replace('?', '.').replace('!', '.')[:-1].split('.'))
res = 0
t = -1
for i in a:
    if i > n:
        print "Impossible"
        exit()
    if t + i + 1 <= n:
        t += i + 1
    else:
        res += 1
        t = i
if t != 0:
    res += 1
print res
