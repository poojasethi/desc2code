#!/usr/bin/python

import re

n = int(raw_input())
text = raw_input() + " "
sents = re.split('[!?\.] ', text)
L = map(len, sents)[:-1]
ans = 0
cur = 0
ok = True
for s in L:
	if cur + s + 2 > n:
		if cur + s + 1 > n:
			ans += 1
			cur = s + 2
		else:
			ans += 1
			cur = 0
	else:
		cur += s + 2
	if cur - 1 > n:
		ok = False
if cur > 0:
	ans += 1
if ok:
	print ans
else:
	print "Impossible"
