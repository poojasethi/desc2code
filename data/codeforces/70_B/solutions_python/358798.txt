from re import *
n = input ()
text = split (r'[.?!]{1}[ ]?', raw_input ())[:-1]
cur = 0
cnt = 1
for sentence in text:
	if len (sentence) + 1 > n:
		print 'Impossible'
		exit ()
	if cur + len (sentence) + (1 if cur == 0 else 2) <= n:
		cur += len (sentence) + (1 if cur == 0 else 2)
	else:
		cnt += 1
		cur = len (sentence) + 1
print cnt
