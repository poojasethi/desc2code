import sys
import re

n = int(sys.stdin.next())
text = sys.stdin.next()

res = 1
cur = 0

for s in re.split("[.?!]", text):
  s = s.strip()
  if not s:
    continue
  s += "."
#  print s, len(s)
  if len(s) > n:
    print "Impossible"
    sys.exit(0)

  next_len = cur + len(s) + (cur > 0)
#  print next_len
  if next_len > n:
    res += 1
    cur = len(s)
  else:
    cur = next_len

print res
