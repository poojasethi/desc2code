# -*- coding: utf-8 -*-

import sys

n = int(raw_input())
a_list = []
b_list = []
a_cnt = 0
b_cnt = 0
for i in xrange(n):
    a,b = map(int, raw_input().split())
    a_list.append(a)
    b_list.append(b)
c = 0
while c < n:
    if a_list[a_cnt] < b_list[b_cnt] or b_cnt >= n:
        a_cnt += 1
    else:
        b_cnt += 1
    c += 1

a_r = n - max(n/2,a_cnt)
b_r = n - max(n/2,b_cnt)
print '1' * max(n/2,a_cnt) + '0' * a_r
print '1' * max(n/2,b_cnt) + '0' * b_r
