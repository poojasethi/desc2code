#!/usr/bin/env python2
import sys

n = int(sys.stdin.readline())
a, b, c = [], [], []
for i in range(n):
	l = sys.stdin.readline().split()
	a.append(int(l[0]))
	b.append(int(l[1]))
c.extend(a)
c.extend(b)
c.sort()
mp = {}
x = 0
for it in c:
	mp[it] = x
	x += 1
s1, s2 = '1'*(n/2), '1'*(n/2)
f1, f2 = True, True
for i in range(n/2, n):
	if f1 and mp[a[i]] < n:
		s1 += '1'
	else:
		s1 += '0'
		f1 = False
print s1
for i in range(n/2, n):
	if f2 and mp[b[i]] < n:
		s2 += '1'
	else:
		s2 += '0'
		f2 = False
print s2
