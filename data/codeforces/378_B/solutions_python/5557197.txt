import sys
n = int(raw_input())
a, b = [0]*n, [0]*n
for i in range(n):
    x, y = map(int, raw_input().split())
    a[i] = x
    b[i] = y
k = n / 2
a1 = [1] * k + [0] * (n - k)
a2 = [1] * k + [0] * (n - k)
l, r = 0, 0
for j in range(n):
    if a[l] < b[r]:
        a1[l] = 1
        l += 1
    else:
        a2[r] = 1
        r += 1
as1 = ''
as2 = ''
for i in range(n):
    as1 += str(a1[i])
    as2 += str(a2[i])
print as1
print as2
