la=[]
lb=[]
n=int(input())
for i in range(1,n+1):
    li=raw_input().split(' ')
    la.append(int(li[0]))
    lb.append(int(li[1]))
ans1=""
ans2=""
for i in range(0,n):
    if i*2+2<=n:
	ans1=ans1+'1'
	ans2=ans2+'1'
    else:
	if lb[n-i-1]>la[i]:
	    ans1=ans1+'1'
	else:
	    ans1+='0'
	if la[n-i-1]>lb[i]:
	    ans2+='1'
	else:
	    ans2+='0'
print ans1
print ans2
