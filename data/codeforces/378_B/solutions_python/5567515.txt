# a person can go to final 
# iff he is either top n amoung all or top n/2 in its semifinal
import sys


n = eval(sys.stdin.readline())

a = []
b = []
for i in range(n):
    inp = map(int, sys.stdin.readline().split())
    a.append(inp[0])
    b.append(inp[1])

ansForA = []
ansForB = []
for i in range(n):
    if i < n / 2:
        ansForA.append(1)
        ansForB.append(1)
    else:
        ansForA.append(0)
        ansForB.append(0)

indA = 0
indB = 0

for i in range(n):
    if a[indA] < b[indB]:
        ansForA[indA] = 1
        indA += 1
    else:
        ansForB[indB] = 1
        indB += 1

ans = ''
for i in ansForA:
    ans += str(i)
print ans
ans = ''
for i in ansForB:
    ans += str(i)
print ans
