import math
n=input()
l1=[]
l3=[]
l2=[]
ind={}
for i in xrange(n):
	l=map(int,raw_input().split())
	l1.append(l[0])
	l2.append(l[1])
	l3.append(l[0])
	l3.append(l[1])
	ind[l[0]],ind[l[1]]=0,0
l3=sorted(l3)
for i in xrange(n/2):
	ind[l1[i]]=1
	ind[l2[i]]=1
for i in xrange(n):
	ind[l3[i]]=1
s=""
t=""
for i in l1:
	s+=str(ind[i])
for i in l2:
	t+=str(ind[i])
print s
print t
