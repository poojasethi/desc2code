#!/usr/bin/env python


def read():
    ret = [[] for i in range(2)]
    for i in range(int(raw_input())):
        a, b = map(int, raw_input().split())
        ret[0].append((a, 0, i))
        ret[1].append((b, 1, i))
    return ret


def work(score):
    vList = []
    vList.extend(score[0])
    vList.extend(score[1])
    vList.sort()
    
    selected = [[0 for j in range(len(score[0]))] for i in range(2)]
    
    for i in range(len(score[0])):
        selected[vList[i][1]][vList[i][2]] = 1
    
    for i in range(len(score[0]) / 2):
        selected[0][i] = 1
        selected[1][i] = 1
    
    print "".join(map(str, selected[0]))
    print "".join(map(str, selected[1]))


if __name__ == "__main__":
    work(read())
