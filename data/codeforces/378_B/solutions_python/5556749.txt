num = int(raw_input().strip())
parts = []
for i in range(num):
    parts.append([int(x) for x in raw_input().split()])
one = list('0'*num)
two = list('0'*num)
for k in [0, num/2]:
    for i in range(k):
        one[i] = '1'
        two[i] = '1'
    l = k
    r = k
    for i in range(num - 2*k):
        if parts[l][0] < parts[r][1]:
            one[l] = '1'
            l += 1
        else:
            two[r] = '1'
            r += 1
print ''.join(one)
print ''.join(two)
