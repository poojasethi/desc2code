n = int(raw_input())
a, b = [0] * n, [0] * n
for i in range(n):
    a[i], b[i] = map(int, raw_input().split())
k = n / 2
i = j = k - 1
if 2 * k < n:
    if a[k] < b[k]: i = k
    else: j = k
if a[n - 1] < b[0]: x = n
else:
    x, y = i + 1, j
    while a[x] < b[y]:
        x += 1
        y -= 1
if a[0] > b[n - 1]: v = n
else:
    u, v = i, j + 1
    while a[u] > b[v]:
        u -= 1
        v += 1
i, j = max(i, x), max(j, v)
print '1' * i + '0' * (n - i)
print '1' * j + '0' * (n - j)
