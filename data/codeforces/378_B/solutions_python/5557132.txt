sf1,sf2=[],[]
q1,q2=[],[]
n=input()
for i in xrange(n):
	a,b=[int(i) for i in raw_input().split()]
	sf1.append(a)
	q1.append('0')
	sf2.append(b)
	q2.append('0')
i,j,k=0,0,1
while(k<=n):
	if(sf1[i]<sf2[j]):
		i+=1
	else:
		j+=1
	k+=1
	
I=max(n/2,i)
J=max(n/2,j)
i,j=0,0
while j<J:
	q2[j]='1'
	j+=1
while i<I:
	q1[i]='1'
	i+=1

print ''.join(q1)
print ''.join(q2)
