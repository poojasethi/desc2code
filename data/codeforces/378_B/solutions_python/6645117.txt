#!/usr/bin/env python

if __name__ == '__main__':
    N = int(raw_input())
    semi1, semi2 = [], []
    for i in range(N):
        semi1i, semi2i = map(int, raw_input().split())
        semi1.append(semi1i)
        semi2.append(semi2i)
    
    not_eliminated = {}
    semis = sorted(semi1 + semi2)
    for i in range(N):
        not_eliminated[semis[i]] = True
    
    for i in range(N / 2):
        not_eliminated[semi1[i]] = True
        not_eliminated[semi2[i]] = True

    res = ''
    for s in semi1:
        if s in not_eliminated:
            res += '1' 
        else:
            res += '0'
    print res

    res = ''
    for s in semi2:
        if s in not_eliminated:
            res += '1' 
        else:
            res += '0'
    print res
