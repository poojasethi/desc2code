import sys

all = map(int, sys.stdin.read().split())
n = all[0]
del all[0]
k = n/2
semi1 = [1]*k + [0]*(n-k)
semi2 = [1]*k + [0]*(n-k)

a = all[::2]
b = all[1::2]

chk_a = 0
chk_b = 0

for i in range(n):
    if a[chk_a] <= b[chk_b]:
        semi1[chk_a] = 1
        chk_a += 1
    else:
        semi2[chk_b] = 1
        chk_b += 1

print "".join(map(str, semi1))
print "".join(map(str, semi2))
