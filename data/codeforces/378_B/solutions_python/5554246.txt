n = int(raw_input())

vs = []
for k in xrange(n):
	a, b = map(int, raw_input().split())
	vs.append((1, k, a))
	vs.append((2, k, b))

vs.sort(key=lambda (ab, k, v): v)

ans_a = [True] * (n // 2) + [False] * (n - n // 2)
ans_b = [True] * (n // 2) + [False] * (n - n // 2)

for ab, k, v in vs[:n]:
	if ab == 1:
		ans_a[k] = True
	else:
		ans_b[k] = True

print ''.join('1' if v else '0' for v in ans_a)
print ''.join('1' if v else '0' for v in ans_b)
