from sys import stdin
n=input()
a=[]
b=[]
dic={}
for line in stdin:
	x,y=map(int,str(line).split())
	a.append(x)
	b.append(y)
	dic[x]=0
	dic[y]=0
for i in xrange(n/2):
	dic[a[i]]=1
	dic[b[i]]=1
x,y=0,0
while(x+y<n):
	if a[x]<b[y]:
		dic[a[x]]=1
		x+=1
	else:
		dic[b[y]]=1
		y+=1
s=""
t=""
for i in a:
	s+=str(dic[i])
for i in b:
	t+=str(dic[i])
print s
print t
