n = int(raw_input())
times = []
for i in range(n):
    a,b = [int(x) for x in raw_input().split()]
    times.append((a, (0,i)))
    times.append((b, (1,i)))

maxK = n/2
results = [[1]*maxK + [0]*(n-maxK), [1]*maxK + [0]*(n-maxK)]

times.sort(key = lambda x: x[0])
for i in range(n):
    i1 = times[i][1][0]
    i2 = times[i][1][1]
    results[i1][i2] = 1
    
print ''.join(str(x) for x in results[0])
print ''.join(str(x) for x in results[1])
