n = input()
l1 = [0] * n
l2 = [0] * n
l3 = []
table = {}
for i in xrange(n):
    l1[i], l2[i] = map(int, raw_input().split())
    table[l1[i]], table[l2[i]] = 0, 0
l3 = sorted(l1 + l2)
for i in xrange(n/2):
    table[l1[i]], table[l2[i]] = 1, 1
for i in xrange(n):
    table[l3[i]] = 1
s1 = ""
s2 = ""
for i in l1:
    s1 += str(table[i])
for i in l2:
    s2 += str(table[i])
print s1
print s2
