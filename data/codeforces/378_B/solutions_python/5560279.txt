n = int(raw_input())
L = []
for i in xrange(n):
    a, b = map(int, raw_input().split())
    L += [(a,0), (b,1)]
L.sort()
count_a = 0; count_b = 0
half = n//2
for i in xrange(n):
    if L[i][1] == 0:
        count_a += 1
    else:
        count_b += 1
a_1 = max(half, count_a); b_1 = max(half, count_b)
print "1"*a_1 + "0"*(n-a_1)
print "1"*b_1 + "0"*(n-b_1)
