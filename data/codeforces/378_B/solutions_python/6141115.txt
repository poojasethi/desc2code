import sys

n=input()
een,twee=[],[]

for i in range(n):
    hulp=map(int,raw_input().split())
    een.append(hulp[0])
    twee.append(hulp[1])
    
u,v,w=0,0,0

while w<n:
    if v>=n or een[v]>twee[u]:
        u+=1
    else:
        v+=1
    w+=1

u,v=max(u,n/2),max(v,n/2)

print '1'*v+'0'*(n-v)
print '1'*u+'0'*(n-u)
