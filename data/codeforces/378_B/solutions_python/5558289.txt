n = input()
b = []
c = []
d = []
e = []
for x in range(n):
    r = raw_input().split()
    b.append(int(r[0]))
    c.append(int(r[1]))
    d.append(0)
    e.append(0)
k = n // 2
for x in range(k):
    d[x] = 1
    e[x] = 1
j = 0
k= 0
for x in range(n):
    #print j,k
    if b[j] < c[k]:
        j += 1
        d[j-1] = 1
    else:
        k += 1
        e[k-1] = 1
f = ''
g = ''
for x in range(n):
     f += str(d[x])
     g += str(e[x])
print f
