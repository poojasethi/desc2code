n=int(input())
a=[0]*(n+1)
b=[0]*(n+1)
a.append(1000000001)
b.append(1000000001)
c=[(0,0)]*(n+1)
for i in range(1,n+1):
	a[i],b[i]=map(int,raw_input().split())

i=1
j=1
k=1
while(k<=n):
	if(a[i]<=b[j]):
		c[i]=(1,0)
#		print c[i][0]
		k+=1
		i+=1
	else:
		c[j]=(0,1)
		k+=1
		j+=1

for i in range(1,n/2+1):
	c[i]=(1,1)

s=""
p=""

	s+=str(x[0])
	p+=str(x[1])
