#include<stdio.h>
__int64 a[100010],b[100010];
int main()
{
    int n,i;
    while(scanf("%d",&n)!=EOF)
	{
		for(i=1;i<=n;i++)
			scanf("%I64d %I64d",&a[i],&b[i]);
		for(i=1;i<=n;i++)
		{
			if(i*2<=n||a[i]<b[n+1-i])
				printf("1");
			else
				printf("0");
		}
		printf("\n");
		for(i=1;i<=n;i++)
		{
			if(i*2<=n||b[i]<a[n+1-i])
				printf("1");
			else
				printf("0");
		}
		printf("\n");
	}
	return 0;
}
