#include<iostream>
using namespace std;
int n;
int a[100010],b[100010];
char x[100010],y[100010];
int main(){
    cin>>n;
    for(int i=0;i<n;++i){
        cin>>a[i]>>b[i];
        x[i]=48;
        y[i]=48;}
    for (int i=0;i<n/2;++i){
        x[i]=49;
        y[i]=49;}
    int k=0,m=0;
    for (int i=0;i<n;++i){
        if (a[k]<b[m]){
            x[k++]=49;
        }
        else{
            y[m++]=49;
        }
    }
    cout<<x<<endl<<y;
    return 0;
}
