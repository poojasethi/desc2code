#include <iostream>
using namespace std;

int main(){

	int n, arr[100005][2], l = 0, r = 0;
	bool ok[100006][2] = {};
	cin >> n;
	for(int j = 0; j<n; j++) for(int i = 0; i<2; i++) cin >> arr[j][i];
	for(int i = 0; i<n/2; i++) ok[i][0] = ok[i][1] = 1;
	for(int i = 0; i<n; i++){
		if(arr[l][0]<arr[r][1]){
			ok[l][0] = 1;
			l++;
		} else {
			ok[r][1] = 1;
			r++;
		}
	}
	for(int i = 0; i<2; i++){
		for(int j = 0; j<n; j++) cout << ok[j][i];
		cout << '\n';
	}

}
