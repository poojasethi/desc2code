int a[100005],b[100005],n;
int main() {
 //   freopen("input.txt","r",stdin);
    scanf("%d",&n);
    for(int i=1; i<=n; ++i)
        scanf("%d%d",&a[i],&b[i]);
    for(int i=1; i<=n; ++i)
        if(2*i<=n||a[i]<b[n+1-i])
            printf("1");
        else printf("0");
    puts("");
