#include<iostream>
#include<algorithm> 
#include<vector>
#include<string>
#include<set>
using namespace std;
int n,i,j,a[100005],b[100005],c[100005],d[100005];
int main(){
	cin>>n;
	for(i=1;i<=n;i++){
		cin>>a[i]>>b[i];
	}
	for(i=1;i<=n/2;i++){
		c[i]=1;
		d[i]=1;
	}
	for(i=1;i<=n;i++){
		if(i==n/2 || b[i]<a[n-i+1])
		   d[i]=1;
		if(i==n/2 || a[i]<b[n-i+1])
		   c[i]=1;   
	}
	for(i=1;i<=n;i++){
		cout<<c[i];
	}
	cout<<"\n";
	for(i=1;i<=n;i++){
		cout<<d[i];
	}
	return 0;
}
