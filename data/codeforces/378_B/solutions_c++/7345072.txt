#include <iostream>
using namespace std;

int n, a[100000], b[100000], c, d, i, k;

int main()
{
	cin>>n;
	for (i=0; i<n; i++) cin>>a[i]>>b[i];
	for (i=0; i<n; i++) {
		if (a[c]<=b[d]) c++; 
		else d++;
	}
	c=max(c, n/2); 
	d=max(d, n/2);
	for (i=0; i<c; i++) cout<<1;
	for (i=c; i<n; i++) cout<<0; 
	cout<<endl;
	for (i=0; i<d; i++) cout<<1;
	for (i=d; i<n; i++) cout<<0; 
	cout<<endl;
	return 0;
}
