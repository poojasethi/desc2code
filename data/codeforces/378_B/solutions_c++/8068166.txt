#include<cstdio>
using namespace std;
int n,i,j,k,a[100005],b[100005];
inline void P(int k)
{
  for (int i=1;i<=k;i++) putchar('1');
  for (int i=k+1;i<=n;i++) putchar('0');
  puts("");
}
int main()
{
  scanf("%d",&n);
  for (i=1;i<=n;i++)
    scanf("%d%d",&a[i],&b[i]);
  i=0;j=0;
  for (k=1;k<=n;k++)
    if (a[i+1]<b[j+1]) i++;else j++;
  if (n/2>i) i=n/2;P(i);
  if (n/2>j) j=n/2;P(j);
}
