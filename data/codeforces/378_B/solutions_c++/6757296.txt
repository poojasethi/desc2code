#include<bits/stdc++.h>

using namespace std;

int a,k,telsat,teldu,bts,satu[100005],dua[100005];
string ans1,ans2;

int main(){
	scanf("%d",&a);
	for(int x=0;x<a;x++){
		scanf("%d %d",&satu[x],&dua[x]);
		ans1+='0';
		ans2+='0';
	}
	k=a/2;
	for(int x=0;x<k;x++){
		ans1[x]='1';
		ans2[x]='1';
	}
	telsat=teldu=bts=0;
	while(bts<a){
		if(satu[telsat]<dua[teldu]) ans1[telsat++]='1'; else ans2[teldu++]='1';
		bts++;
	}
	cout<<ans1<<endl;
	cout<<ans2<<endl;
}
