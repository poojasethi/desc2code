#include <iostream>

using namespace std;

int main()
{
    int n;
    cin>>n;
    int k = n/2;
    
    int a[(int)1e5+5];
    int b[(int)1e5+5];
    for (int i=0 ; i<n ; i++) cin>>a[i]>>b[i];
    
    int i=0 , j=0;
    while ((i+j)<n)
    {
        if (a[i]<b[j]) i++;
        else j++;
    }
    
    int d = max(k,i);
    for (int q=0 ; q<n ; q++) cout<<(q<d);
    cout<<"\n";
    d = max(k,j);
    for (int q=0 ; q<n ; q++) cout<<(q<d);
    
    return 0;
}
