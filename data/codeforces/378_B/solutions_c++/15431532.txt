
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

int main()
{
LL n,i,p,q;
cin>>n;

vector <int> u(n),v(n);

for(i=0;i<n;i++)
	cin>>v[i]>>u[i];
p=0;q=0;

while(p+q<(n))
	{
	if(v[p]<u[q])
		p++;
	else
		q++;
	}

for(i=0;i<n;i++)
	if(i<p || i<n/2)
		cout<<1;
	else
		cout<<0;
cout<<endl;
for(i=0;i<n;i++)
	if(i<q || i<n/2)
		cout<<1;
	else
		cout<<0;

cout<<endl;
}
