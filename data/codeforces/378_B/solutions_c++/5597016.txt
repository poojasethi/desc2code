#include <iostream>
using namespace std;
int m[100000][2];
bool u[100000][2]={0};
int n,u1=0,u2=0;
void change()
{
 if (m[u1][0]<m[u2][1]) {u[u1][0]=1;u1++;return;}
 u[u2][1]=1;u2++;return;
}
int main()
{
cin >> n;
for (int i=0;i<n;i++)
{
 cin >> m[i][0] ;cin >> m[i][1]; change();
}

for (int i=0;i<n/2;i++)
for (int z=0;z<2;z++)
u[i][z]=1;
for (int z=0;z<2;z++)
{
for (int i=0;i<n;i++)
cout << u[i][z];
cout << endl;
}
}
