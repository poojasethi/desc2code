#include <iostream>
#define mx 1000009

using namespace std;

long n,i,a[mx],b[mx],aa[mx],ab[mx],l,r;

int main()
{
	cin >> n;
	for (i=1;i<=n;i++) cin >> a[i] >> b[i];
	for (i=1;i<=n/2;i++) {
		aa[i]=1;
		ab[i]=1;
	}
	l=1;r=1;
	for (i=1;i<=n;i++){
		if (a[l]<b[r]) {
			aa[l]=1;
			l++;
		}
		else
		{
			ab[r]=1;
			r++;
		}
	}
	for (i=1;i<=n;i++) cout << aa[i];
	cout << endl;
	for (i=1;i<=n;i++) cout << ab[i];
	return 0;
}
