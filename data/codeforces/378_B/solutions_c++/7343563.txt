#include<stdio.h>
int main()
{
	int x[100050],y[100050];
	int i,j,sum,p,q,k,n;
	scanf("%d",&n);
	for(i=0;i<n;i++)
	{
		scanf("%d%d",&x[i],&y[i]);
	}
	p=q=0;
	for(i=0;i<n;i++)
	{
		if(x[p]<y[q])p++;
		else q++;
	}
	k=n/2;
	for(i=0;i<n;i++)
	{
		if(i<p||i<k)printf("1");
		else printf("0");
	}
	printf("\n");
	for(i=0;i<n;i++)
	{
		if(i<q||i<k)printf("1");
		else printf("0");
	}
	printf("\n");
}
