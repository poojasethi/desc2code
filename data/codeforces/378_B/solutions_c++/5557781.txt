#include<iostream>
#include<cstdio>

using namespace std;

int main()
{
	long int n,count1=0,count2=0,k,i,j;
	long int a[100005],b[100005];
	char sa[100005],sb[100005];
	cin>>n;
	k=n/2;
	for(i=0;i<n;i++)
	{
		cin>>a[i]>>b[i];
		if(i<k)
		{
			sa[i]='1';
			sb[i]='1';
		}
		else
		{
			sa[i]='0';
			sb[i]='0';
		}
	}
	k=0;
	for(i=0,j=0;i<n,j<n,k<n;k++)
	{
		if(a[i]<b[j]){ sa[i]='1'; i++;}
		else{ sb[j]='1'; j++;}
	}
	sa[n+1]='\0';
	sb[n+1]='\0';
	printf("%s\n%s\n",sa,sb);
	return 0;
}
