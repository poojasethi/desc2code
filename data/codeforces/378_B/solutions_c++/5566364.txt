#include <cstdio>
int a[100200], b[100200];
int i, j, k, l, m, n;
int main(){
    scanf("%d\n", &n);
    for (i = 1; i <= n; i++) scanf("%d %d\n", &a[i], &b[i]);
    j = k = 1;
    for (l = 1; l <= n; l++){
        if (a[j] < b[k]) j++; else k++;
    }
    j--; k--;
    for (i = 1; i + i <= n || i <= j; i++) putchar('1');
    for (; i <= n; i++) putchar('0');
    putchar('\n');
    for (i = 1; i + i <= n || i <= k; i++) putchar('1');
    for (; i <= n; i++) putchar('0');
    return 0;
}
