#include<iostream>
using namespace std;
int main()
{
	int a[100005], b[100005], i,n;
	cin >> n;
	for (i=1; i<=n; i++)
	{
		cin >> a[i] >> b[i];
	}
	for (i=1; i<=n; i++)
	{
		if (i<=n/2 || a[i]<b[n-i+1]) cout << 1;
		else cout << 0;
	}
	cout << endl;
	for (i=1; i<=n; i++)
	{
		if (i<=n/2 || b[i]<a[n-i+1]) cout << 1;
		else cout << 0;
	}
	return 0;
}
