#include <cstdio>

char C[100001], D[100001];
int a, b, k, n, A[100000], B[100000];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; i++) {
		scanf("%d %d", A + i, B + i);
		C[i] = D[i] = '0';
	}
	C[n] = D[n] = 0;
	while (k < n) {
		if (A[a] < B[b]) C[a++] = '1';
		else D[b++] = '1';
		k++;
	}
	for (int i = 0; i < (n >> 1); i++) C[i] = D[i] = '1';
	puts(C);
	puts(D);
	return 0;
}
