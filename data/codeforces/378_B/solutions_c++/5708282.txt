#include<stdio.h>
# define N 100007
int main()
{	
	int a[N],b[N],i;
	int n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
		scanf("%d%d",&a[i],&b[i]);
	for(i=1;i<=n;i++)
	{
		if(i<=n/2||a[i]<=b[n-i+1])
			printf("1");
		else
			printf("0");
	}
	printf("\n");
	for(i=1;i<=n;i++)
	{
			if(i<=n/2||b[i]<=a[n-i+1])
				printf("1");
			else
				printf("0");
	}
	printf("\n");
	return 0;
}
