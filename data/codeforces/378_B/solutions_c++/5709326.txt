#include<stdio.h>
unsigned int str1[100001],str2[100001];
int main()
{
	int n,i;
	while(scanf("%d",&n)!=EOF)
	{	
		for(i=1;i<=n;i++)
		{
			scanf("%u %u",&str1[i],&str2[i]);
		}
		for(i=1;i<=n;i++)
		{
			if(i*2<=n||str1[i]<str2[n+1-i])
				printf("1");
			else
				printf("0");
		}
		printf("\n");
		for(i=1;i<=n;i++)
		{
			if(i*2<=n||str2[i]<str1[n+1-i])
				printf("1");
			else
				printf("0");
		}
		printf("\n");
	}
	return 0;
}
