#include<stdio.h>
#define maxn 100005
int a[maxn],b[maxn],c[maxn]={0},d[maxn]={0};
int main()
{
	int k,n,i;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d",&a[i]);
		scanf("%d",&b[i]);
	}
	for(k=0;k<=n/2;k++)
	{
		c[k]=1;
		d[k]=1;
	}
		for(i=1;i<=n;i++)
		{
			if(a[i]<b[n+1-i])
				c[i]=1;
			if(b[i]<a[n+1-i])
				d[i]=1;
		}
		for(i=1;i<=n;i++)
			printf("%d",c[i]);
		printf("\n");
		for(i=1;i<=n;i++)
			printf("%d",d[i]);
	return 0;
}
