#include<iostream>
#include <string.h>
using namespace std;
int main()
{
	int a[50001*2],b[50001*2];
	int f[2][50001*2];
	int n;
	int i,j;
	cin>>n;
	for(i=1;i<=n;i++)
		cin>>a[i]>>b[i];
	memset(f,0,sizeof(f));
	for(i=1;i<=n/2;i++)
		f[0][i]=f[1][i]=1;
	for(i=n/2+1;i<=n;i++)
 	{
        if(a[i]<b[n-i+1])
            f[0][i]=1;
        if(b[i]<a[n-i+1])
            f[1][i]=1;
    }
	for(i=0;i<2;i++)
	{
		for(j=1;j<=n;j++)
			cout<<f[i][j];
		cout<<endl;
	}
	return 0;
}
