#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
#define MAX 100010
int n, A[MAX], B[MAX];

int main() {
	scanf("%d", &n);
	for (int i = 1; i <= n; i++)
	{
		scanf("%d %d", &A[i], &B[i]);
	}
	
	for (int i = 1; i <= n; i++)
	{
		if (i <= n/2 || A[i] < B[n - i + 1])
			printf("1");
		else
			printf("0");
	}
	printf("\n");
	for (int i = 1; i <= n; i++)
	{
		if (i <= n/2 || B[i] < A[n - i + 1])
			printf("1");
		else
			printf("0");
	}
	return 0;
}
