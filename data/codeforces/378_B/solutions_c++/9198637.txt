#include <string>
#include <iostream>
using namespace std;
int main(){
    string s, t;
    int n, arr[100000], arr2[100000], po1 = 0, po2 = 0;
    cin >> n;
    for (int i = 0; i < n; ++i)cin >> arr[i] >> arr2[i];
    s.resize(n, '0');
    t.resize(n, '0');
    for (int i = 0; i < n; ++i)
    if (arr[po1] < arr2[po2])s[po1++] = '1';
    else t[po2++] = '1';
    for (int i = n / 2 - 1; i >= 0; --i)s[i] = t[i] = '1';
    cout << s << endl << t << endl;
    return 0;
}
