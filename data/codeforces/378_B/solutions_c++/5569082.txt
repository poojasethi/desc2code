#include <cstdio>

const int MAXN = 100010;

int n,a[MAXN],b[MAXN];
char A[MAXN],B[MAXN];

void init(){
    scanf("%d",&n);
    for (int i=0;i<n;i++)
        scanf("%d %d",&a[i],&b[i]);

    for (int i=0;i<n/2;i++) A[i] = B[i] = '1';
    for (int i=n/2;i<n;i++) A[i] = B[i] = '0';


    int p1 = 0,p2 = 0;
    for (int i=0;i<n;i++){
        if (a[p1]<b[p2]) A[p1] = '1',p1++;
        else B[p2] = '1',p2++;
    }
    printf("%s\n%s",A,B);   
}

int main(){
    init();
    return 0;
}
