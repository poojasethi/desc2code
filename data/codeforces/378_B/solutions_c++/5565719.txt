#include <iostream>
#include <math.h>
#include <cmath>
using namespace std;
int main ()
{
	int a[2][100000],b[2][100000]={0},i,j,n,m,l=0,r=0;
	cin>>n;
	for (i=0;i<n;i++)
	{
		cin>>a[0][i]>>a[1][i];
	}
	for (i=0;i<n;i++)
	{
		if (i<n/2)
		{
			b[0][i]=1;
			b[1][i]=1;
		}
		if (a[0][l]<a[1][r])
		{
			b[0][l++]=1;
		}
		else
		{
			b[1][r++]=1;
		}
	}
	for (i=0;i<n;i++)
	{
		cout<<b[0][i];
	}cout<<endl;
	for (i=0;i<n;i++)
	{
		cout<<b[1][i];
	}
	return 0;
}
