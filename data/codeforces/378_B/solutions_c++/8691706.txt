#include <iostream>

using namespace std;

int n;
int a[1000000],b[1000000];
int ra[1000000],rb[1000000];

int main()
{
  cin>>n;
  for (int i=0;i<n;i++)
    cin>>a[i]>>b[i];
  for (int i=0;i<n/2;i++)
    ra[i]=rb[i]=1;
  int ia=0;
  int ib=0;
  while (ia+ib<n) {
    if (a[ia]<b[ib]) {
      ra[ia++]=1;
    } else {
      rb[ib++]=1;
    }
  }
  for (int i=0;i<n;i++)
    cout<<ra[i];
  cout<<endl;
  for (int i=0;i<n;i++)
    cout<<rb[i];
  cout<<endl;
}
