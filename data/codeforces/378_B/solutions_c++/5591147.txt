#include <iostream>
#define mx 1000009

using namespace std;

long n,i,a[mx],b[mx],aa[mx],ab[mx],lt,rt;

int main()
{
	cin >> n;
	for (i=1;i<=n;i++) cin >> a[i] >> b[i];
	for (i=1;i<=n/2;i++) {
		aa[i]=1;
		ab[i]=1;
	}
	lt=1;rt=1;
	for (i=1;i<=n;i++){
		if (a[lt]<b[rt]) {
			aa[lt]=1;
			lt++;
		}
		else
		{
			ab[rt]=1;
			rt++;
		}
	}
	for (i=1;i<=n;i++) cout << aa[i];
	cout << endl;
	for (i=1;i<=n;i++) cout << ab[i];
	return 0;
}
