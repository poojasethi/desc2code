#include<stdio.h>
#define NN 100010
int a[NN],b[NN];
int main()
{
	int i,t,n;
	scanf("%d",&n);
	for(i=1;i<=n;i++)
	{
		scanf("%d%d",&a[i],&b[i]);
	}
	for(i=1;i<=n;i++)
	{
		if(i+i<=n||a[i]<b[n-i+1])
			t=1;
		else
			t=0;
        printf("%d",t);
	}
	printf("\n");
	for(i=1;i<=n;i++)
	{
		if(i+i<=n||b[i]<a[n-i+1])
			t=1;
		else
			t=0;
		printf("%d",t);
	}
    printf("\n");
	return 0;
}
