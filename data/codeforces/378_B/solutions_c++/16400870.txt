#include<cstdio>
#include<algorithm>
using namespace std;

int n;
int a[111111], b[111111];
int ans[2][111111];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%d%d", &a[i], &b[i]);
	for(int i=0;i+i<n-1;i++)
		ans[0][i] = ans[1][i] = 1;
	
	for(int x=0, y=0, i=0; i<n;i++)
		if(a[x]<b[y]) ans[0][x++] = 1;
		else ans[1][y++] = 1;
	
	for(int i=0;i<2;i++) {
		for(int j=0;j<n;j++)
			printf("%d", ans[i][j]);
		printf("\n");
	}
	
	return 0;
}
