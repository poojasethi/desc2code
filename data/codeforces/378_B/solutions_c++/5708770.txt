#include<stdio.h>
int main()
{
int a[100001],b[100001];
int n,i;
while((scanf("%d",&n))!=EOF)
{
for(i=1;i<=n;i++)
scanf("%d%d",&a[i],&b[i]);
for(i=1;i<=n;i++)
{
if(2*i<=n||a[i]<b[n+1-i])
printf("1");
else printf("0");
}
printf("\n");
for(i=1;i<=n;i++)
{
if(2*i<=n||b[i]<a[n+1-i])
printf("1");
else printf("0");
}
printf("\n");
}
return 0;
}
