#include<cstring>
#include<cstdlib>
#include<cstdio>
#include<algorithm>
#include<set>
#define N 120000
using namespace std;
set<int> S[N];
int n,m,s,e,a[N],x,i,j,f[N][310],ans;
int main(){
	scanf("%d%d%d%d",&n,&m,&s,&e);
	for(i=1;i<=n;++i)scanf("%d",&a[i]);
	for(i=1;i<=m;++i){
		scanf("%d",&x);
		S[x].insert(i);
	}
	memset(f,3,sizeof(f));
	f[0][0]=0;
	for(i=1;i<=n;++i){
		for(j=0;j<=300;++j)if(f[i-1][j]<1<<20){
			f[i][j]=min(f[i][j],f[i-1][j]);
			set<int> ::iterator v=S[a[i]].lower_bound(f[i-1][j]);
			if(v==S[a[i]].end())continue;
			f[i][j+1]=min(f[i][j+1],(*v)+1);
		}
	}
	for(i=1;i<=n;++i)
		for(j=0;j<=300;++j)if(j*e+i+f[i][j]-1<=s)ans=max(ans,j);
	printf("%d\n",ans);
}
