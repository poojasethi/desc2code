#include<set>
#include<cstring>
#include<algorithm>
#include<iostream>
#include<cstdio>
#include<vector>
using namespace std;
#define N 200000
int n,m,s,e,a[N],b[N];
int G[305];
vector<int>V[N];
int main(){
	scanf("%d%d%d%d",&n,&m,&s,&e);
	for(int i=1;i<=n;i++)scanf("%d",&a[i]);
	for(int i=1;i<=m;i++)scanf("%d",&b[i]),V[b[i]].push_back(i);
	int limit=s/e,ans=0;
	for(int i=1;i<=limit;i++)G[i]=m+1;
	for(int i=1;i<=n;i++){
		for(int j=min(i,limit);j>=1;j--){
			int tmp=G[j-1];
			if(tmp<=m){
				vector<int>::iterator it=upper_bound(V[a[i]].begin(),V[a[i]].end(),tmp);
				if(it!=V[a[i]].end()){
					G[j]=min(G[j],*it);
					if(s-e*j-(*it+i)>=0)
						ans=max(ans,j);
				}
			}
		}
	}
	printf("%d\n",ans);
}
