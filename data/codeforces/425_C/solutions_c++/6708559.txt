#include <cstdio>
#include <climits>
#include <algorithm>
#include <vector>

using namespace std;

int main() {
  int n,m,s,e;
  scanf("%d %d %d %d",&n,&m,&s,&e);
  vector<int> a(n),b(m);
  vector<vector<int> > bindexes(100*1000+1);
  for(int i=0;i<n;i++)scanf("%d",&a[i]);
  for(int i=0;i<m;i++){
    scanf("%d",&b[i]);
    bindexes[b[i]].push_back(i);
  }
  vector<int> earliest(302,INT_MAX);
  earliest[0]=-1;
  int ans=0;
  for(int i=0;i<n;i++){
    for(int j=earliest.size()-1;j>=1;j--) {
      int limit = earliest[j-1];
      vector<int>::iterator it = upper_bound(bindexes[a[i]].begin(),bindexes[a[i]].end(),limit);
      if (it != bindexes[a[i]].end()) {
        earliest[j] = min(earliest[j], *it);
      }
    }
    for(int j=0;j<earliest.size();j++) {
      if (earliest[j]!=INT_MAX) {
        int total = j*e + i+1 + earliest[j]+1;
        if(total<=s)ans=max(ans,j);
      }
    }
  }
  printf("%d\n",ans);
  return 0;
}
