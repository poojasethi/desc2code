#include <algorithm>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <map>
#include <set>
#include <vector>
using namespace std;

int n, m, s, e;
int a[100005], b[100005];
set<int> idb[100005];

int dp[333];

int best;

int main()
{
	scanf("%d%d%d%d", &n, &m, &s, &e);
	for(int i=0; i<n; i++) scanf("%d", &a[i]);
	for(int i=0; i<m; i++) {
		scanf("%d", &b[i]);
		idb[b[i]].insert(i);
	}
	
	best=0;

	dp[0]=-1;
	for(int i=1; i<333; i++) dp[i]=100005;

	for(int i=0; i<n; i++) {
		for(int d=305; d>=0; d--) {
			set<int>::iterator it=idb[a[i]].upper_bound(dp[d]);
			if(it!=idb[a[i]].end() && (*it)<dp[d+1]) {
				dp[d+1]=(*it);
				if((d+1)*e+(i+(*it)+2)<=s) {
					best=max(best, d+1);
				}
			}
		}
	}

	printf("%d\n", best);

	return 0;
}
