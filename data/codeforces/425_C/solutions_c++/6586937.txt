#include<cstdio>
#include<iostream>
#include<cstring>
#include<cstdlib>
#include<vector>
#include<algorithm>
using namespace std;
vector<int> Place[100002];
int n,m,s,e,i,j,ans;
int a[100002],b[100002],Dp[100002];
int main()
{
	scanf("%d%d%d%d",&n,&m,&s,&e);
	for (i=1;i<=n;++i)
	scanf("%d",&a[i]);
	for (i=1;i<=m;++i)
	{
		scanf("%d",&b[i]);
		Place[b[i]].push_back(i);
	}
	for (i=1;i<=100002;++i)
	Dp[i]=100000000;
	Dp[0]=0;
	for (i=1;i<=n;++i)
	for (j=s/e;j>=0;j--)
	{
		vector<int>::iterator it=upper_bound(Place[a[i]].begin(),Place[a[i]].end(),Dp[j]);
		if (it==Place[a[i]].end()) continue;
		Dp[j+1]=min(Dp[j+1],*it);
		if (Dp[j+1]+i+(j+1)*e<=s)
		ans=max(ans,j+1);
	}
	printf("%d\n",ans);
}
