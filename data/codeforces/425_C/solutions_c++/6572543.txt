#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>

using namespace std;

const int INF = 10000000;

int n,m,s,e;
int b[100010],a[100010];
vector<int> ind[100010];
int f[100010][310];

int main()
{
	scanf("%d%d%d%d",&n,&m,&s,&e);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]),ind[a[i]].push_back(i);
	for(int i=1;i<=m;i++) scanf("%d",&b[i]);
	int T=s/e,ans=0;
	for(int i=1;i<=T;i++) f[0][i]=INF;
	f[0][0]=0;
	for(int j=1;j<=m;j++){
		for(int t=0;t<=T;t++) f[j][t]=f[j-1][t];
		for(int t=0;t<T;t++){
			if(f[j-1][t]>n) break;
			vector<int> :: iterator p=lower_bound(ind[b[j]].begin(),ind[b[j]].end(),f[j-1][t]+1);
			if(p!=ind[b[j]].end()) f[j][t+1]=min(f[j][t+1],*p);
		}
		for(int t=1;t<=T;t++){
			int i=f[j][t];if(i>n) break;
			ans=max(ans,min((s-(i+j))/e,t));
		}
	}
	printf("%d\n",ans);
	return 0;
}
