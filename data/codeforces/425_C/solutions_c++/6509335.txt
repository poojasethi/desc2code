#include<stdio.h>
#include<algorithm>
#include<string.h>
#include<vector>
using namespace std;
	
const int N=100005,inf=1<<30;
int dp[N][305],a[N],b[N];
vector<int> v[N];
vector<int>::iterator it;
int main(){
	int ans,n,i,j,k,t,l,m,e,s,x;
	scanf("%d%d%d%d",&n,&m,&s,&e);
	for(i=1;i<=n;i++) scanf("%d",a+i);
	for(i=1;i<=m;i++) scanf("%d",b+i),v[b[i]].push_back(i);
	for(k=s/e,i=0;i<=n;i++)
		for(dp[i][0]=0,j=1;j<=k;j++) dp[i][j]=inf;
	for(ans=dp[0][0]=0,i=1;i<=n;i++){
		for(j=1;j<=k;j++){
			dp[i][j]=dp[i-1][j];
			it=lower_bound(v[a[i]].begin(),v[a[i]].end(),dp[i-1][j-1]+1);
			if(it!=v[a[i]].end()&&*it<dp[i][j]) dp[i][j]=*it;
			if(j>ans&&dp[i][j]+i+e*j<=s) ans=j;
		}
	}
	printf("%d\n",ans);
	return 0;
}
