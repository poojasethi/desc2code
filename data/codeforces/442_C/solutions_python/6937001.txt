from sys import stdin
def main():
    n = int(stdin.readline())
    a = map(int, stdin.readline().split())
    b = []
    ans = 0
    for x in a:
        b.append(x)
        while len(b) > 2 and b[-2] <= b[-1] and b[-2] <= b[-3]:
            ans += min(b[-1], b[-3])
            b[-2] = b[-1]
            b.pop()
    for x in xrange(1, len(b) - 1):
        ans += min(b[x-1], b[x+1])
    print ans
main()
