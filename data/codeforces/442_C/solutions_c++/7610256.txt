#include<cstdio>
#include<cstring>
#include<algorithm>
#include<stack>
using namespace std;
long long p[600000],top;
int main()
{
    long long i,x,n,ans;
    top=-1;ans=0;
    scanf("%I64d",&n);
    while(n--)
    {
        scanf("%I64d",&x);
        while(top>=1&&p[top]<=p[top-1]&&p[top]<=x)
        {
            ans+=min(p[top-1],x);
            top--;
        }
        p[++top]=x;
    }
    sort(p,p+(top+1));
    for(i=0;i<top-1;i++) ans+=p[i];
    printf("%I64d\n",ans);
    return 0;
}
