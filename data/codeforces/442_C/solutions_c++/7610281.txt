#include<cstdio>
#include<cstring>
#include<algorithm>
#include<stack>
using namespace std;

long long p[600000],t=-1;
int main()
{
    long long i,x,n,ans=0;
    scanf("%I64d",&n);
    while(n--)
    {
        scanf("%I64d",&x);
        while(t>=1&&p[t]<=p[t-1]&&p[t]<=x)
        {
            ans+=min(p[t-1],x);
            t--;
        }
        p[++t]=x;
    }
    sort(p,p+(t+1));
    for(i=0;i<t-1;i++)ans+=p[i] ;
    printf("%I64d\n",ans);
    return 0;
}
