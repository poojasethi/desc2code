#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

int q[500010];

void solve()
{
    int n; cin >> n; 
    int top = 0;
    long long ans = 0;
    for (int i=0; i<n; i++) {
        int a; cin >> a;
        while (top > 1 && q[top-1] <= q[top-2] && q[top-1] <= a) {
            ans += min(a, q[top-2]);
            top --;
        }
        q[top++] = a;
    }
    sort(q, q+top);
    for (int i=0; i<top-2; i++) ans += q[i];
    cout << ans << endl;
}

int main()
{
    solve(); return 0;
}
