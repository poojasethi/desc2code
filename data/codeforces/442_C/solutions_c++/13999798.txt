#include<cstdio>
#include<algorithm>
using namespace std;
int a[500500];
int main()
{
	int n,i,p=0,x;
	long long ans=0;
	scanf("%d",&n);
	for(i=0;i<n;++i){
		scanf("%d",&x);
		while(p>=2&&a[p-2]>=a[p-1]&&x>=a[p-1]){
			if(a[p-2]<x)ans+=a[p-2];
			else ans+=x;
			--p;
		}
		a[p++]=x;
	}
	sort(a,a+p);
	for(i=p-3;i>=0;--i)ans+=a[i];
	printf("%I64d\n",ans);
	return 0;
}
