#include <bits/stdc++.h>

#define INF 1234567890

using namespace std;
typedef pair<int, int> ii;
const int LIM = 500005;

int main() {
	ios_base::sync_with_stdio(0);
	int n;
	int a[LIM], b[LIM], p;
	cin >> n;
	long long ans = 0;
	int pos = 0;
	for(int i = 0; i < n; i++) {
		cin >> a[i];
		while(p >=  2 && b[p-1] <= a[i] && b[p-2] >= b[p-1]) {
			ans += min(b[p-2], a[i]);
			p--;
		}
		b[p++] = a[i];
	}
	sort(b, b + p);
	for(int i = 0; i < p - 2; i++)
		ans += b[i];
	cout << ans << "\n";

	return 0;
}
