#include<cmath>
#include<string>
#include<iostream>
#include<algorithm>
#include<cstdio>
using namespace std;

const int N=500005;

int n,c=-1;
long long stack[N];

int main(){
	long long x,ans=0;
	scanf("%d",&n);
	for (int i=0;i<n;i++){
		scanf("%I64d",&x);

		while (c>0&&stack[c-1]>=stack[c]&&stack[c]<=x){
			ans+=min(stack[c-1],x);
			c--;
		}
		stack[++c]=x;
	}
	sort(stack,stack+c+1);

	for (int i=0;i<=c-2;i++)	ans+=stack[i];
		printf("%I64d\n",ans);
	return 0;
}
