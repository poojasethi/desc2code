#include <cstdio>

typedef long long LL;

const int maxn = 500010;

LL res;
int n;

int main() {
//	freopen("input.txt", "r", stdin);
	
	scanf("%d", &n);
	int lst = 0, mx = 0, mmx = 0;
	for (int i = 1; i <= n; ++ i) {
		int x; scanf("%d", &x);
		res += 2*x - (lst < x ? lst : x);
		if (x > mx) {
			mmx = mx;
			mx = x;
		} else if (x > mmx)
			mmx = x;
		lst = x;
	}
	
	printf("%I64d\n", res - mx * 2 - mmx);
	
//	for(;;);
	
	return 0;
}
