#include <bits/stdc++.h>

using namespace std;

int main(){
	#ifndef ONLINE_JUDGE
	freopen("a.in", "r", stdin);
	#endif
    ios_base::sync_with_stdio(false);
    int n;
    cin >> n;
    int last = 0, m1 = 0, m2 = 0;
    long long res = 0;
    int x;
    for(int i = 0 ; i < n; i++){
    	cin >> x;
    	res += x*2 - (x < last? x : last);
    	if (x > m1){
    		m2 = m1;
    		m1 = x;
    	}
    	else {
    		if (x > m2){
    			m2 = x;
    		}
    	}
    	last = x;
    }
    cout << res - m1*2 -m2;
	return 0;
}
