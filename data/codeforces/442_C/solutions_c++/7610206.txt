#include<iostream>
#include<cstdio>
#define ll long long
using namespace std;

int a[100005],n,tot;
ll ans=0;

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		int x;
		scanf("%d",&x);
		while(tot>=2&&a[tot]<=a[tot-1]&&a[tot]<=x){
			ans+=min(a[tot-1],x);tot--;
		}
		a[++tot]=x;
	}
	
	ll tmp1=0,tmp2=0;
	for(int i=1;i<=tot;i++){
		ans+=a[i];
		if(a[i]>tmp1) tmp2=tmp1,tmp1=a[i];
		else if(a[i]>tmp2) tmp2=a[i];
	}
	ans-=tmp1+tmp2;
	printf("%I64d\n",ans);
	return 0;
}
