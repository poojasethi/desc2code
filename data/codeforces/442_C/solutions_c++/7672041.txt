#include<stdio.h>
#include<queue>
#include<algorithm>
using namespace std;
const int maxn=100050;
int n,A[maxn];
long long ans=0;
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	scanf("%d",&n);
	for(i=j=0;i<n;i++)
	{
		scanf("%d",&A[j++]);
		while(j>=3&&A[j-2]<=min(A[j-3],A[j-1]))
			j--,A[j-1]=A[j],ans+=min(A[j-2],A[j]);
	}
	for(i=1;i<j-1;i++)
		ans+=min(A[i-1],A[i+1]);
	printf("%I64d\n",ans);
	return 0;
}
