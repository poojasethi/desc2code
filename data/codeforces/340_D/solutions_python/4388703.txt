n = input()
a = [n+1] * (n+1)

def find(a, r, v):
    l = m = 0
    while l < r:
        m = (l+r) >> 1
        if a[m] < v:
            l = m + 1
        else:
            r = m
    return l

for i, x in enumerate(map(int, raw_input().split())):
    t = find(a, i, x)
    a[t] = x

print find(a, n, n+1)
