def lis(a, n):
	f = [0] * (n + 1);
	l = 0;
	for i in range(n):
		low = 1;
		high = l;
		while (low <= high):
			mid = (low + high) / 2;
			if (a[i] > f[mid]): low = mid + 1;
			else: high = mid - 1;
		f[low] = a[i];
		if (low > l): l = low;
	return l;
	
n = input();
a = map(int, raw_input().split());
print lis(a, n);
