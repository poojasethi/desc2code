#include<cstdio>
#include<algorithm>
using namespace std;
int DP[100000 + 10],n,k,sol;
int main(){
    scanf("%d",&n);
    for(int i = 0 ; i < n ; i++){
        scanf("%d",&k);
        DP[lower_bound(DP,DP+sol,k)-DP]= k;
        sol = max(sol,lower_bound(DP,DP+sol,k)-DP+1);
    }
    printf("%d\n",sol);
}
