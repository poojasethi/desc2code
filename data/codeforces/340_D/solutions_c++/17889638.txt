#include <bits/stdc++.h>

using namespace std;

const int N = 100001;
int a[N];

main() {
	int n;
	cin >> n;
	int ans = 0;
	a[0] = 0;
	for (int i = 0; i < n; i++) {
		int x;
		cin >> x;
		int it = upper_bound(a, a + ans + 1, x) - a;
		if (it == ans + 1)
			a[++ans] = x;
		else a[it] = min(x, a[it]);
	}
	cout << ans << endl;
}
