#include<bits/stdc++.h>
using namespace std;
int a,n,i;
main()
{
    scanf("%d",&n);
    set<int> st;
    for (i=0;i<n;i++)
    {
        scanf("%d",&a);
        st.insert(a);
        set<int>::iterator it=st.find(a);
        it++;
        if (it!=st.end())
            st.erase(it);
    }
    cout<<st.size();
}
