#include <bits/stdc++.h>
#define ll long long
using namespace std;
int main()
{
	int n;
	cin>>n;
	int el;
	set<int> s;
    set<int>::iterator k;
	for(int i=0;i<n;i++)
	{
		cin>>el;
		if (s.insert(el).second)
        {
           k = s.find(el);
           k++;
           if (k!=s.end())
              s.erase(k);
        }
	}
	cout<<s.size()<<endl;
}
