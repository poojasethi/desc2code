#include <iostream>
#include <set>
#include <cstdio>
using namespace std;

set<int> st;
int n,tmp;
int main() {
	scanf("%d",&n);
	for(int i = 0 ; i < n ; i++){
		scanf("%d",&tmp);
		set<int>::iterator it = st.lower_bound(tmp);
		if(it != st.end())
			st.erase(it);
		st.insert(tmp);
	}
	printf("%d\n",st.size());
	return 0;
}
