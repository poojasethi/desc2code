#include <stdio.h>
#include <algorithm>
using namespace std;
#define N 100010
#define INF 0x7fffffff
int dp[N],d[N];
int main()
{
	int i,k,n,x,max;
	scanf("%d",&n);
	for(i=1;i<=n;i++)d[i]=INF;
	for(max=0,i=1;i<=n;i++)
	{
		scanf("%d",&x);
		k=lower_bound(d+1,d+n+1,x)-d;
		dp[i]=k;
		d[k]=x;
		if(k>max)max=k;
	}
	printf("%d\n",max);
	return 0;
}
