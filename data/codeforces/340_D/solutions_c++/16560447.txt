#include <bits/stdc++.h>
#define MAX_N 100100 

using namespace std; 

int tick[MAX_N]; 
int x,n,res; 

int main ()
 	{
 		cin >> n; 
 		for (int i = 0; i < n; i++) 
 			{
 				cin >> x; 
 				tick[i] = n+1;
 				*lower_bound(tick,tick+i+1,x) = x;
 			}
 			
 		for (res = 0; res < n && tick[res] <= n; res++); 
 		cout << res; 
 		return 0; 
 	}
