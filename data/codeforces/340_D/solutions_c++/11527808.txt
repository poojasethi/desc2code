#include <bits/stdc++.h>
using namespace std;
int main()
{
 int n; 
 cin >> n;
 vector<int> a(n+1),d(n+1,10000000);
 for(int i=0;i<n;++i)
    cin >> a[i];
 for(int i=0;i<n;++i)
 {
  *lower_bound(d.begin(),d.end(),a[i]) = a[i];  
 }
 for(int i=0;i<=n;++i)
 {
    if(d[i] == 10000000)
    {
        cout << i << endl;
        break;
    }
 }
}
