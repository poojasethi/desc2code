#include<iostream>
#include<algorithm>
using namespace std;
int a[100100],b[100100];
int main()
{
    int n,ans=0;
    cin>>n;
    for(int i=0;i<n;i++)
    {
        cin>>a[i];
        if(ans==0 || a[i]>b[ans-1])
            b[ans++]=a[i];
        else
            b[lower_bound(b,b+ans,a[i])-b]=a[i];
    }
    cout<<ans<<endl;
    return 0;
}
