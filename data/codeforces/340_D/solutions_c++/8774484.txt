#include<bits/stdc++.h>
using namespace std;
int main()
{
    long n,i,a;
    set<long> st;
    cin>>n;
    for(i=0;i<n;i++){
        cin>>a;
        st.insert(a);
        set<long>::iterator it=st.find(a);
        it++;
        if(it!=st.end()){
            st.erase(it);
        }
    }
    cout<<st.size();
    return 0;
}
