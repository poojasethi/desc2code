#include <bits/stdc++.h>
using namespace std;

int n,a[100005];
int len , lis[100005];
int main() {
	//freopen("input.txt","r",stdin);
	scanf("%d",&n); for(int i=1;i<=n;++i) scanf("%d",a+i);
	for(int i=1;i<=n;++i){
		int x=lower_bound(lis,lis+len,a[i])-lis;
		if(x==len)lis[x]=a[i],++len;
		else
			lis[x]=a[i];
	}

	printf("%d",len);
	return 0;
}
