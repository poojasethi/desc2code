#include<cstdio>
#include<algorithm>
using namespace std;

int n, v, tot, p;
int f[111111];

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++) {
		scanf("%d", &v);
		if (!tot || v > f[tot-1]) f[tot++] = v;
		else {
			p = lower_bound(f, f+tot, v) - f;
			f[p] = v;
		}
	}
	printf("%d\n", tot);
	
	return 0;
}
