#include <iostream>
#include <algorithm>
using namespace std;
int ans[1000001],c;
int main()
{
	int n;
	cin>>n;
	int tmp;
	c=1;
	cin>>ans[0];
	for(int i=1;i<n;++i)
	{
		cin>>tmp;
		int tm=lower_bound(ans, ans+c, tmp) -ans;
		//cout<<tm<<"*\n";
		ans[tm]=tmp;
		if(tm==c)
			c++;
	}
	cout<<c<<endl;
	return 0;
}
