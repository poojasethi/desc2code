#include<bits/stdc++.h>
using namespace std;
int a[100005];
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
		cin>>a[i];
	vector<int> ans;
	ans.push_back(a[0]);
	vector<int>::iterator p;
	for(int i=1;i<n;i++){
		p=upper_bound(ans.begin(),ans.end(),a[i]);
		if(p==ans.end())
			ans.push_back(a[i]);
		else
			*p=a[i];
	}
	cout<<ans.size()<<endl;
}
