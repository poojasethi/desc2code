#include <iostream>
#include <vector>
int n,m,l,i,b,e;
std::vector<int>a(2,-1);
int main(){
  std::cin>>n;a[1]=100001;
  for(i;i<n;i++){
   std::cin>>m;
   b=l=0;e=a.size()-1;
   while(b<e){
    l=(b+e)/2;
    if(a[l]>m)e=l;else b=l+1;
   }
   a[e]=m;
   if(e+2>a.size())a.push_back(100001);
  }
  std::cout<<a.size()-2;return 0;
}
