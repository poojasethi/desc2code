#include<cstdio>
#include<algorithm>
using namespace std;
int n,a[100001],b[100001],n_c;
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&a[i]);
        int x = upper_bound(b,b+n_c,a[i]) - b;
        if(x == n_c) b[n_c++] = a[i];
        else b[x] = a[i];
    }
    printf("%d\n",n_c);
    return 0;
}
