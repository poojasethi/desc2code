#include<stdio.h>
#include<algorithm>
#include<set>
using namespace std;

int main(){
    int i,n,h;
    set<int> s;
    scanf("%d",&n);
    for(i=1;i<=n;i++){
        scanf("%d",&h);
        if(s.lower_bound(h)!=s.end())
            s.erase(s.lower_bound(h));
        s.insert(h);
    }
    printf("%d",s.size());
    return 0;
}
