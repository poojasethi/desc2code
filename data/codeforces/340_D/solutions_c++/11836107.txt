#include<iostream>
#include<cstdio>
#include<algorithm>
#define maxn 100000
int dp[maxn];
using namespace std;
int main()
{
	int n,ans = 0,x;
	scanf("%d",&n);
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&x);
		dp[i] = n+1;
		int u = lower_bound(dp+1,dp+i+1,x) -dp;
		dp[u] = x;
		ans = max(u,ans);
	}
	printf("%d\n",ans);
	return 0;
}
