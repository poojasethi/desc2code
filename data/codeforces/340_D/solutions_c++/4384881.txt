#include <stdio.h>
#include <algorithm>
using namespace std;
int oo=(1<<31)-1;
int arr[100010];
int main(){
	int N,lis,v,p;
	scanf("%d",&N);
	arr[0]=oo;
	lis=1;
	for(int i=0; i<N; i++){
		scanf("%d",&v);
		p=lower_bound(arr,arr+lis,v)-arr;
		arr[p]=v;
		if(p==lis) lis++;
	}
	printf("%d\n",lis);		
	
	return 0;
}
