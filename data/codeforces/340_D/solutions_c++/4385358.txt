#include<cstdio>
#include<algorithm>
using namespace std;
int a[100009];
int main()
{
    int n,k,x,cnt=0;
    scanf("%d",&n);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&x);
        int k=upper_bound(a,a+cnt,x)-a;
        if(k==cnt)
            cnt++;
        a[k]=x;
    }
    printf("%d\n",cnt);
    return 0;
}
