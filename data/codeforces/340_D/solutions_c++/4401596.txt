#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int a[100006];
int n, tmp, cnt, x;
int main()
{
    cin >> n;
    for (int i = 1; i <= n; i++)
    {
        cin >> x;
        tmp = upper_bound(a, a+cnt, x) - a;
        if(tmp == cnt) cnt++;
        a[tmp] = x;
    }
    cout << cnt <<endl;
    return 0;
}
