#include <iostream>
#include<algorithm>

using namespace std;

int n,a[100001],b[100001],x,q;

int main(){
   cin>>n;
   for(int k=1;k<=n;k++){
   cin>>a[k];        
        x=upper_bound(b,b+q,a[k])-b;
        if(x==q){
        b[q]=a[k];
        q++;        
        }else{
        b[x]=a[k];
        }
   }
   cout<<q<<endl;
}
