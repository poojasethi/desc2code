import sys
readline = sys.stdin.readline

n, m = map(int, readline().split(" "))
P = [[p, i, 0] for i, p in enumerate(map(int, readline().split(" ")))]
P.sort()
for i in xrange(n):
    P[i][2] = i % 2
def cmp(a, b):
    return a[1] - b[1]
P.sort(cmp=cmp)

print " ".join(map(lambda s: str(s[2]), P))
