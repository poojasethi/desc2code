import sys

def Ni(): return tuple(map(int, sys.stdin.readline().split()))

Ni()
points = Ni()
indexed = sorted((point, i) for i, point in enumerate(points))
colored = sorted((i, j % 2) for j, (point, i) in enumerate(indexed))
print " ".join(str(c) for i, c in colored)
