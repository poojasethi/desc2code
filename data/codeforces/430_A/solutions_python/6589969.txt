import sys

[n,m] = map(int, sys.stdin.readline().split())
points = map(int, sys.stdin.readline().split())

l = ['0'] * n
pointss = []
for i in points:
    pointss.append(i)
pointss.sort()
# print pointss

for i in range(0,n):
    index = points.index(pointss[i])
    if (i % 2 == 1):
        l[index] = '1'
print ' '.join(l)
