#include <iostream>

using namespace std;
int ma[101],f,g,n,m;
int main(int argc, char** argv) {
    cin>>n>>m;
    for (int i=0;i<n;i++) cin>>ma[i];
    for (int i=0;i<m;i++) cin>>f>>g;
    for (int i=0;i<n;i++) 
    {
        f=0;
      for (int j=0;j<n;j++) 
        if (ma[i]<ma[j]) f++;
       cout<<f%2<<" ";
    }
    cout<<"\n";
    return 0;
}
