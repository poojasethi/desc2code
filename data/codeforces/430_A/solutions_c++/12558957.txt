#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
  int n, m;
  cin >> n >> m;
  int x[101], y[101], z[101];
  for (int i = 0; i < n; ++i) {
    cin >> x[i];
    y[i] = x[i];
  }
  sort(y, y + n);
  for (int i = 0; i < n; ++i) {
    z[y[i]] = i % 2;
  }
  for (int i = 0; i < n; ++i) {
    cout << z[x[i]] << " ";
  }
  cout << "\n";
  return 0;
}
