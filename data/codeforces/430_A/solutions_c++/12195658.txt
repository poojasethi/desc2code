#include<iostream>
#include<algorithm>
using namespace std;
int n,m,b[10005];
pair<int,int>a[10005];
main(){
    cin>>n>>m;
    for(int i=1;i<=n;i++)cin>>a[i].first,a[i].second=i;
    sort(a+1,a+n+1);
    for(int i=1;i<=n;i++)b[a[i].second]=i%2;
    for(int i=1;i<=n;i++)cout<<b[i]<<" ";cout<<endl;      
}
