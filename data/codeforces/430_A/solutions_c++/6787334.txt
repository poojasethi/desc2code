#include<cstdio>
#include<algorithm>
int n, m, a[110], b[110], ans[110];
int main() {
  scanf("%d%d", &n, &m);
  for(int i = 0; i < n; ++i) scanf("%d", a+i), b[i] = a[i];
  std::sort(a, a+n);
  for(int i = 0; i < n; ++i) ans[a[i]] = i&1;
  for(int i = 0; i < n; ++i) printf("%d ", ans[b[i]]);
}
