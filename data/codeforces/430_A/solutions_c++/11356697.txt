#include "stdio.h"
#include "algorithm"
using namespace std;

int b[105];
pair<int,int> a[105];
int n,m;

int main(){
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++){
		scanf("%d",&a[i].first);
		a[i].second = i;
	}
	sort(a,a+n);
	for(int i=0;i<n;i++){
		if(i%2==1) b[a[i].second] = 1;
		else b[a[i].second] = 0;
	}
	for(int i=0;i<n;i++){
		printf("%d ",b[i]);
	}
}
