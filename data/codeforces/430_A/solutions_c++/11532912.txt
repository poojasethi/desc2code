#include<cstdio>
#include<algorithm>

using namespace std;

int m,n,x,y,a[105],b[105],c[105];

int main()
{
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
        b[i]=a[i];
    }
    for(int i=0;i<m;i++)    scanf("%d%d",&x,&y);
    sort(b,b+n);
    for(int i=0;i<n;i++)    c[b[i]]=(i%2);
    for(int i=0;i<n;i++)    printf("%d ",c[a[i]]);
    return 0;
}
