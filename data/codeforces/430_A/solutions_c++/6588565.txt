#include <cstdio>
int arr[100],res[100];
int main(){
	int n,m,x,y;
	scanf("%d%d",&n,&m);
	for(int i=0;i<n;i++)scanf("%d",&arr[i]);
	for(int i=0;i<m;i++)scanf("%d%d",&x,&y);
	bool flag=0;
	for(x=0;x<=100;x++){
		for(int i=0;i<n;i++){
			if(arr[i]==x){res[i]=flag;flag=!flag;}
		}
	}
	for(int i=0;i<n;i++){
		printf("%d",res[i]);
		if(i==n-1)printf("\n");
		else printf(" ");
	}
	return 0;
}
