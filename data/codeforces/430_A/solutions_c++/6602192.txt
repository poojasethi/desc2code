#include <iostream>
#include <algorithm>

using namespace std;

int n, m, res[111];
pair<int, int> a[111];

int main() {
    cin >> n >> m;
    for (int i = 0; i < n; i++) {
        int x;
        cin >> x;
        a[i] = make_pair(x, i);
    }
    sort(a, a + n);
    for (int i = 0; i < n; i++)
        res[a[i].second] = i % 2;
    for (int i = 0; i < n; i++) cout << res[i] << " ";
}
