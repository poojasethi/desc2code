#include<cstdio>
#include<algorithm>
using namespace std;

const int MAXN=100;
int c[MAXN];
int main()
{
	int n,m;
	scanf("%d%d",&n,&m);
	pair<int,int> node[MAXN];
	for(int i=0;i<n;i++)
	{
		int pos;
		scanf("%d",&pos);
		node[i]=make_pair(pos,i);
	}
	sort(node,node+n);
	for(int i=0;i<n;i++)
	{
		c[node[i].second]=i&1;
	}
	for(int i=0;i<n;i++)
	{
		printf("%d ",c[i]);
	}
	return 0;
}
