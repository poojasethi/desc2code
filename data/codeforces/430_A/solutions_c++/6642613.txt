#include<iostream>
#include<algorithm>

using namespace std;
int n, m, a[1000], b[1000];
int main()
{
    cin >> n >> m;
    for (int i = 0; i < n; i++)
    {
        cin >> a[i];
        b[i] = a[i];
    }
    sort(b, b + n);

    for (int i = 0; i < n; i++)
    {
        int index = 0;
        while (b[index] != a[i])
            index ++;

        cout << (index & 1) << " ";
    }
    cout << endl;
}
