#include <bits/stdc++.h>
using namespace std;
typedef pair<int,int> ii;
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	ii a[n];
	for(int i=0;i<n;i++){
		scanf("%d",&a[i].first);
		a[i].second=i;
	}
	sort(a,a+n);
	int b[n];
	for(int i=0;i<n;i++) b[a[i].second]=i%2;
	for(int i=0;i<n;i++) printf("%d ",b[i]);
}
