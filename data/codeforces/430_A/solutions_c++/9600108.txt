#include <iostream>
#include <algorithm>
using namespace std;
int d[105];
int p[105];
int color[105];
int n,m;
int main() {
	cin>>n>>m;
	for(int i=0;i<n;i++) cin>>d[i],p[i]=d[i];
	sort(p,p+n);
	for(int i=0;i<n;i++) color[p[i]]=i%2;
	int x,y;
	for(int i=0;i<m;i++) cin>>x>>y;
	for(int i=0;i<n;i++) cout<<color[d[i]]<<' ';
	cout<<endl;
	return 0;
}
