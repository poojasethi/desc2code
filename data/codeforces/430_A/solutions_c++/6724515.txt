#include<cstdio>
#include<algorithm>
struct node{
  int x,id;
}f[105];
bool cmp(node u,node v){return u.x<v.x;}
int ans[105];
int main(){
  int n,m;scanf("%d%d",&n,&m);
  for(int i=1;i<=n;i++)
    scanf("%d",&f[i].x),f[i].id=i;
  std::sort(f+1,f+n+1,cmp);
  for(int i=1;i<=n;i++)
    ans[f[i].id]=i&1;
  for(int i=1;i<=n;i++)
    printf("%d ",ans[i]);    
}
