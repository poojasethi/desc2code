#include<bits/stdc++.h>
using namespace std;
const int l = 101;
int arr[l],pos[l],ans[l];
int main()
{
    int n,m,i,j,k;
    for(cin>>n>>m,i=0;i<n;cin>>arr[i],pos[i]=arr[i],++i);
    for(i=0;i<m;cin>>j>>k,++i);
    sort(arr,arr+n);
    k=0;
    for(i=0;i<n;ans[arr[i]]=k,k^=1,++i);
    for(i=0;i<n;cout<<ans[pos[i]]<<' ',++i);

}
