#include <stdio.h>
#include<stdlib.h>
int main(void)
{
	int n,m,i,j,flag,l,r;
	int a[100];
	scanf("%d %d",&n,&m);
	for(i=0;i<n;i++)
	{
		scanf("%d",&a[i]);
	}
	for(i=0;i<m;i++)
		scanf("%d %d",&l,&r);
	for(i=0;i<n;i++)
	{
		flag=0;
		for(j=0;j<n;j++)
			if(a[i]<a[j])
				flag++;
		printf("%d ",flag%2);
	}
	return 0;
}
