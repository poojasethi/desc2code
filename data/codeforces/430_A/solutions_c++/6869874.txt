#include <iostream>
#include <algorithm>

using namespace std;

typedef pair<int, int> pii;
const int MAXN = 101;
int ans[MAXN];
pii all[MAXN];

int main() {
	int n, m;
	cin >> n >> m;
	for (int i = 0; i < n; i++)
		cin >> all[i].first, all[i].second = i;
	sort(all, all + n);
	for (int i = 0; i < n; i++)
		ans[all[i].second] = i % 2;
	for (int i = 0; i < n; i++)
		cout << ans[i] << " ";
	return 0;
}
