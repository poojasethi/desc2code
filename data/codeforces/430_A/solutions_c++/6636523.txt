#include<iostream>
using namespace std;
int main(){
    int n, m;
    int a[101];
    cin >> n >> m;
    for(int i=0; i<n; i++){
	cin >> a[i];
    }
    for(int i=0; i<m; i++){
	int tmp;
	cin >> tmp >> tmp;
    }
    for(int i=0; i<n; i++){
	int num = 0;

	}

    }
    cout << endl;
    return 0;
}
