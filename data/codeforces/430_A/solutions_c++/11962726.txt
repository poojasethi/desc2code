#include <bits/stdc++.h>

using namespace std;

int main()
{
    int n,m,x=1; cin>>n>>m; int a[n],a2[n],ans[101];

    for(int i=0;i<n;i++)
    {
        cin>>a[i]; a2[i]=a[i];
    }
    sort(a,a+n);
    for(int i=0;i<n;i++ )
    {
        ans[a[i]]=x; x=1-x;
    }
    for(int y=0;y<n;y++)
        cout<<ans[a2[y]]<<" ";

    return 0;
}
