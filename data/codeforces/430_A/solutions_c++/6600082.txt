
#include <iostream>
#include <algorithm>
#include <utility>

using namespace std;

pair<int,int> pr[110];
int ans[110];

int main()
{
	int n,i;
	cin>>n>>i;
	for(i=0;i<n;++i){
		cin>>(pr[i].first);
		pr[i].second = i;
	}

	sort(pr,pr+n);
	for(int i=0;i<n;++i)ans[ pr[i].second ] = i%2;
	for(int i=0;i<n;++i)cout<<ans[i]<<" ";

	return 0;
}
