#include <iostream>
#include <vector>
#include <algorithm>


int main(){
  int n, m;
  std::cin >> n >> m;
  std::pair<int, int> l[100];
  for (int i=0; i<n; i++){ std::cin >> l[i].first; l[i].second=i; }
  std::sort(l, l+n);
  int a, b;
  for (int i=0; i<m; i++) std::cin >> a >> b;
  int ans[100];
  for (int i=0; i<n; i++) ans[l[i].second]=i%2;
  for (int i=0; i<n; i++) std::cout << ans[i] << ' ';
}
