#include<cstdio>
#include<algorithm>

using namespace std;


int x[200],xx[200],xxx[200];

int main()
{
    int n,m;
    scanf("%d%d",&n,&m);
    for(int i=0;i<n;i++)
    {
        scanf("%d",&xx[i]);
        xxx[i]=xx[i];
    }
    sort(xx,xx+n);
    for(int i=0;i<n;i++)
    {
        x[xx[i]]=i%2;
    }
    for(int i=0;i<n;i++)
    {
        printf("%d ",x[xxx[i]]);
    }
}
