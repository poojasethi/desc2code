#include <iostream>
#include <algorithm>
using namespace std;
pair <int, int> a[150];
int ans[150],n,i,m;
int main()
{
    cin>>n>>m;
    for (i=0; i<n; i++)
    {
        cin>>a[i].first;
        a[i].second=i;
    }
    sort(a,a+n);
    for (i=0; i<n; i++)
        ans[a[i].second]=i%2;
    for (i=0; i<n; i++)
        cout<<ans[i]<<" ";
    cout<<endl;
    return 0;
}
