#include <iostream>
#include <algorithm>
using namespace std;

int main() {
	int n, m;
	cin >> n >> m;
	int x[101], inv[101];
	for (int i = 0; i < n; i++) {
		cin >> x[i];
		inv[x[i]] = i;
	}
	sort(x, x+n);
	int ans[101];
	for (int i = 0; i < n; i++) {
		ans[inv[x[i]]] = i%2;
	}
	for (int i = 0; i < n; i++) {
		cout << ans[i] << ' ';
	}
	return 0;
}
