#include<cstdio>
#include<algorithm>
#include<cstring>
using namespace std;
struct N{
	int x,y,id;
	bool operator < (const N&now) const{
		return x<now.x;
	}
}a[1000];
int ans[1000];
int main(){
	int n,m;
	scanf("%d%d",&n,&m);
	for (int i=0;i<n;i++){
		scanf("%d",&a[i].x);
		a[i].id=i;
	}
	sort(a,a+n);
	for (int i=0;i<n;i++)
		ans[a[i].id]=i&1;
	for (int i=0;i<n;i++)
		printf("%d ",ans[i]);
	return 0;
}
