#include <iostream>
#include <algorithm>a
using namespace std;

int n,m,a[101],b[101],ans[101];

int main(){
    cin>>n>>m;
    int i;
    for(i=0;i<n;++i)cin>>a[i],b[i]=a[i];
    sort(a,a+n);
    for(i=0;i<n;++i)ans[a[i]]=i%2;
    for(i=0;i<n;++i)cout<<ans[b[i]]<<" ";
    cout<<endl;
    return 0;
}
