#include <bits/stdc++.h>
using namespace std;
#define REP(i, n) for(__typeof(n) i = 0; i < (n); ++i)
bool check[100];
int main(){
    int n, m; cin >> n >> m;
    vector<pair<int,int> > point(n);
    REP(i, n) cin >> point[i].first, point[i].second = i;
    sort(point.begin(), point.end());

    REP(i, n) check[point[i].second] = i&1;
    REP(i, n) cout << check[i] << ' ';
}
