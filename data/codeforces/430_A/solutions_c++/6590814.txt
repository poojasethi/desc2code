#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

int n;
int a[105],b[105];

int main()
{
	int i,j,k,l;
	cin>>n>>l;
	for (i=1;i<=n;i++)
	{
		cin>>a[i];
		b[i]=a[i];
	}
	sort(a+1,a+n+1);
	for (i=1;i<=n;i++)
	{
		for (j=1;j<=n;j++)
			if (b[i]==a[j])
				break;
		cout<<j%2<<" ";
	}
	cout<<endl;
	return 0;
}
