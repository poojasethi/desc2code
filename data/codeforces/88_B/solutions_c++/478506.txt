#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
char s[35][55];
int dis[26];
char str[500010];
int main()
{
	int n,m,x,i,j,k,p;
	int visit[26];
	scanf("%d%d%d",&n,&m,&x);
	for(i=1;i<=n;i++)
		scanf("%s",s[i]+1);
	memset(visit,0,sizeof(visit));
	for(i=0;i<26;i++)
		dis[i]=2100000000;
	for(i=1;i<=n;i++){
		for(j=1;j<=m;j++){
			if(s[i][j]=='S') continue;
			visit[s[i][j]-'a']=1;
			int d=2100000000;
			for(k=1;k<=n;k++){
				for(p=1;p<=m;p++)
					if(s[k][p]=='S'){
						if(d>(i-k)*(i-k)+(j-p)*(j-p))
							d=(i-k)*(i-k)+(j-p)*(j-p);
					}
			}
			if(d<dis[s[i][j]-'a']) dis[s[i][j]-'a']=d;
		}
	}
	scanf("%d",&n);
	scanf("%s",str);
	x*=x;
	int ans=0;
	for(i=0;i<n;i++){
		if(str[i]>='a'){
			if(!visit[str[i]-'a']) 
				break;
		}
		else{
			if(dis[str[i]-'A']==2100000000) break;
			if(!visit[str[i]-'A']) break;
			else if(dis[str[i]-'A']>x) ans++;
		}
	}
	if(i<n) printf("-1\n");
	else printf("%d\n",ans);
	return 0;
}
