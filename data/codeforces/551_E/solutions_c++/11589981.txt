#include<bits/stdc++.h>
#define pb push_back
#define mp make_pair
#define ff first
#define ss second
#define mod 1000000007

using namespace std;
typedef long long LL;
typedef vector<int> VI;
typedef vector< vector<int> > VVI;
typedef pair< LL, LL > PII;

const int N=500010;
LL n,a[N],sz,cnt,inc[1000];
vector<PII> X[1000];

int main()
{
	ios_base::sync_with_stdio(false);
	cin.tie(NULL);

	int q; cin>>n>>q;
	for(int i=0; i<n; i++) cin>>a[i];
	sz = 1;
	while(sz*sz < n) sz++;
	cnt = (n+sz-1)/sz;
	for(int i=0; i<n; i++) X[i/sz].pb(mp(a[i],i));
	for(int i=0; i<cnt; i++) sort(X[i].begin(),X[i].end());

	LL type,l,r,x,y,tmp,mn,mx;
	while(q--)
	{
		cin>>type;
		if(type==1)
		{
			cin>>l>>r>>x; l--; r--;
			for(int i=(l/sz)+1; i<(r/sz); i++) inc[i] += x;
			tmp = l/sz;
			for(int i=0; i<X[tmp].size(); i++)
				if(X[tmp][i].ss >= l && X[tmp][i].ss <= r)
					X[tmp][i].ff += x;
			sort(X[tmp].begin(),X[tmp].end());
			if(tmp == (r/sz)) continue;
			tmp = r/sz;
			for(int i=0; i<X[tmp].size(); i++)
				if(X[tmp][i].ss >= l && X[tmp][i].ss <= r)
					X[tmp][i].ff += x;
			sort(X[tmp].begin(),X[tmp].end());
		}
		else
		{
			mn = N; mx = -1; cin>>y;
			for(int i=0; i<cnt; i++)
			{
				tmp = lower_bound(X[i].begin(),X[i].end(),mp(y-inc[i],-1LL)) - X[i].begin();
				if(X[i][tmp].ff == y-inc[i])
					mn = min(mn,X[i][tmp].ss);
				tmp = upper_bound(X[i].begin(),X[i].end(),mp(y-inc[i],(LL)N)) - X[i].begin();
				if(tmp>0 && X[i][tmp-1].ff == y-inc[i])
					mx = max(mx,X[i][tmp-1].ss);
			}
			if(mx==-1) cout<<"-1\n";
			else cout<<mx-mn<<"\n";
		}
	}
	return 0;
}
