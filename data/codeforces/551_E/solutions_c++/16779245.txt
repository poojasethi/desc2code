#include<bits/stdc++.h>
#define st first
#define nd second
#define mp make_pair
#define pb push_back
#define endl '\n'
#define endll puts("")
#define mod 1000000007
#define mp3(x,y,z) mp(x,mp(y,z))
#define pii pair< ll , int > 
#define orta ((bas+son)/2)
#define sol (root+root)
#define sag (root+root+1)
using namespace std;
typedef long long ll;
const int N = 5e5+3;
int n,m,kn,i,j,k,l,s,t,d,u,ii,jj,x,y,z,h[N];
ll hh[N];
pii A[N];
int main()
{
	cin>>n>>m;
	k=sqrt(n);
	s=n/k;
	if(s*k<n)s++;
	for(i=1;i<=n;i++)
	{
		scanf("%d",&x);
		A[i]=mp(x,i);
	}
	for(i=0;i<s;i++)
		sort(A+i*k+1,A+min(n,i*k+k)+1);

	while(m--)
	{
		scanf("%d",&x);
		if(x==1)
		{
			scanf("%d%d%d",&x,&y,&z);
			for(i=0;i<s;i++)
			{
				if(x<=i*k+1 and y>=min(n,i*k+k))
					hh[i]+=z;
				else if(!(y<i*k+1 or x>min(n,i*k+k)))
				{
					for(j=i*k+1;j<=min(n,i*k+k);j++)
						if(A[j].nd>=x and A[j].nd<=y)
							A[j].st+=z;
					sort(A+i*k+1,A+min(n,i*k+k)+1);
				}
			}
		}
		else
		{
			scanf("%d",&z);
			x=y=0;
			for(i=0;i<s;i++)
			{
				j=lower_bound(A+i*k+1,A+min(n,i*k+k)+1,mp(z-hh[i],0))-A;
				if(j<=min(n,i*k+k) and A[j].st==z-hh[i] and !x)
					x=A[j].nd;
				j=lower_bound(A+i*k+1,A+min(n,i*k+k)+1,mp(z-hh[i]+1,0))-A;
				if(j>i*k+1 and A[j-1].st==z-hh[i] and y<A[j-1].nd)
					y=A[j-1].nd;
			}
		//	cout<<x<<' '<<y<<endl;
			if(!x)puts("-1");
			else printf("%d\n",y-x);
		}
	}
}
