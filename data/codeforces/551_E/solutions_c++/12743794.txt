#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<map>
#define P 500
#define L 1000
#define N 500000
#define T 1000000000
#define LL long long
using namespace std;
int n,m;
LL s[N+5],o[P+5];
map<int,int> M1[P+5];
map<int,int> M2[P+5];
void fix(int pt){
  M1[pt].clear();
  M2[pt].clear();
  if(o[pt]>T) return;
  for(int i=pt*L;i<pt*L+L;i++){
    if(M1[pt].count(s[i])==0){
      M1[pt][s[i]]=i;
    }
    M2[pt][s[i]]=i;
  }
  return;
}
int main(){
  scanf("%d %d",&n,&m);
  memset(s,0,sizeof(s));
  for(int i=0;i<n;i++) scanf("%d",&s[i]);
  n=N;
  memset(o,0,sizeof(o));
  for(int i=0;i<P;i++) fix(i);
  for(int i=1;i<=m;i++){
    int a,b,c,d;
    scanf("%d",&a);
    if(a==1){
      scanf("%d %d %d",&b,&c,&d);
      b--,c--;
      int pb=b/L,pc=c/L;
      if(pb==pc){
        for(int j=b;j<=c;j++) s[j]+=d;
        fix(pb);
        continue;
      }
      if(b%L!=0){
        for(int j=b;j<pb*L+L;j++) s[j]+=d;
        fix(pb++);
      }
      if(c%L!=L-1){
        for(int j=pc*L;j<=c;j++) s[j]+=d;
        fix(pc--);
      }
      for(int j=pb;j<=pc;j++) o[j]+=d;
    }else{
      scanf("%d",&b);
      int lt=P,rt=-1;
      for(int j=0;j<P;j++){
        if(o[j]>T) continue;
        if(M1[j].count(b-o[j])){
          if(lt==P) lt=j;
          rt=j;
        }
      }
      if(lt==P) printf("-1\n");
      else printf("%d\n",M2[rt][b-o[rt]]-M1[lt][b-o[lt]]);
    }
  }
  return 0;
}
