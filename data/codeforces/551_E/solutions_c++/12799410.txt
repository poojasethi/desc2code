#include <cstdio>
#include <algorithm>
#include <cmath>
#define N 500005
using namespace std;
struct node{long long data;int id;}b[N];
long long a[N],d,add[N];
int c,e,f,g,h,i,j,k,l,m,n,r,mid;
int L[N],R[N],from[N];
inline bool cmp(node a,node b){return (a.data<b.data)||(a.data==b.data&&a.id<b.id);}
int main(){
	scanf("%d%d",&n,&m);
	for(i=1;i<=n;i++)scanf("%I64d",&a[i]);
	int X=max((int)sqrt(n),100);int Y=(n-1)/X+1;
	for(j=0;j<Y;j++){
		L[j]=j*X+1;
		R[j]=min(j*X+X,n);
		for(i=L[j];i<=R[j];i++){b[i].data=a[i];b[i].id=i;from[i]=j;}
		sort(b+L[j],b+R[j]+1,cmp);
	}
	for(;m;m--){
		scanf("%d",&c);
		if(c&1){
			scanf("%d%d",&l,&r);
			int ll=from[l],rr=from[r];
			scanf("%I64d",&d);
			if(ll==rr){
				for(i=l;i<=r;i++)a[i]+=d;
				for(i=L[ll];i<=R[ll];i++)
				{b[i].data=a[i];b[i].id=i;}
				sort(b+L[ll],b+R[ll]+1,cmp);
				continue;
			}
			for(i=l;i<=R[ll];i++)a[i]+=d;
			for(i=L[ll];i<=R[ll];i++)
			{b[i].data=a[i];b[i].id=i;}
			sort(b+L[ll],b+R[ll]+1,cmp);
			for(i=L[rr];i<=r;i++)a[i]+=d;
			for(i=L[rr];i<=R[rr];i++)
			{b[i].data=a[i];b[i].id=i;}
			sort(b+L[rr],b+R[rr]+1,cmp);
			for(i=ll+1;i<rr;i++)add[i]+=d;
		}
		else{
			scanf("%I64d",&d);e=g=0;
			for(i=0;i<Y;i++){
				l=L[i];r=R[i];mid=(l+r+1)>>1;
				long long D=d-add[i];
				for(;l<r;mid=(l+r+1)>>1){
					if(b[mid].data>=D)r=mid-1;
					else l=mid;
				}
				if(b[l].data!=D)l++;
				if(b[l].data==D){e=b[l].id;break;}
			}
			if(!e){printf("-1\n");continue;}
			for(i=Y-1;i>=0;i--){
				l=L[i];r=R[i];mid=(l+r)>>1;
				long long D=d-add[i];
				for(;l<r;mid=(l+r)>>1){
					if(b[mid].data<=D)l=mid+1;
					else r=mid;
				}
				if(b[l].data!=D)l--;
				if(b[l].data==D){g=b[l].id;break;}
			}
			printf("%d\n",g-e);
		}
	}
	return 0;
}
