#include<bits/stdc++.h>

#define mp make_pair
#define ss second
#define ff first
#define FOR(i,j,k) for(int i = j; i < (int)k; i++)

using namespace std;

typedef long long ll;
typedef pair<ll,ll> pii;

const ll INF = 1e18;
const int maxn = 5e5+11111;
const int sq = 750;

ll num[maxn], n, q, diff[maxn];
pii st[2*maxn/sq][2*sq];

int get(ll val)
{
    int a = -1, b = -1;
    for(int i = 0; i <= (n-1)/sq; i++)
    {
	val -= diff[i];
	int it = lower_bound(st[i],st[i]+sq,mp(val,-INF)) - st[i];
	
	if(st[i][it].ff == val && a == -1)
	    a = st[i][it].ss;

	it = upper_bound(st[i],st[i]+sq,mp(val,INF)) - st[i];
	it--;
	
	if(st[i][it].ff == val)
	    b = st[i][it].ss;

	val += diff[i];
    }

    return (a == -1? -1 :  b-a);
}

void modify(int l, int r, ll val)
{
    for(int i = l; i <= r; i++)
	diff[i] += val;
}

void modify(int id, int l, int r, ll val)
{
    for(int i = id*sq; i < (id+1)*sq; i++)
    {
	if(l <= i && i <= r)
	    num[i] += val;
	st[i/sq][i%sq] = mp(num[i],i);
    }
    sort(st[id],st[id]+sq);
}
int main()
{
    FOR(i,0,maxn)
	num[i] = -INF;

    cin >> n >> q;
    FOR(i,0,n)
	cin >> num[i];
    
  /*  FOR(i,0,10)
	cout << num[i] << " ";
    cout << endl;*/

    FOR(i,0,maxn)
	st[i/sq][i%sq] = mp(num[i],i);
    
   /* FOR(i,0,5)
	cout << st[0][i].ff << " ";
    cout << endl;*/

    FOR(i,0,maxn/sq)
	sort(st[i],st[i]+sq);
    
  /*  FOR(i,sq-5,sq)
	cout << st[0][i].ff << " ";
    cout << endl;*/

    FOR(i,0,q)
    {

/*	FOR(j,sq-5,sq)
	    cout << st[0][j].ff << " ";
	cout << endl;*/

	ll t,l,r,val;
	cin >> t;
	if(t == 1)
	{
	    cin >> l >> r >> val;
	    l--;r--;
	    int a = l/sq, b = r/sq;

	    if(a == b)
		modify(a,l,r,val);
	    else
	    {
		modify(a,l,r,val);
		modify(a+1,b-1,val);
		modify(b,l,r,val);
	    }
	}
	else
	{
	    cin >> val;
	    cout << get(val) << "\n";
	}

    }
}
