#include <bits/stdc++.h>
#define fi first
#define se second
#define type(x) __typeof(x.begin());
using namespace std;

typedef pair < int , int > ii;
const int maxn = 5e5+5 , sqrtn = 710 , MAX = 1e9;

int n,m,add[sqrtn];
vector < ii > srt[sqrtn];
int main(){
	
	scanf("%d%d",&n,&m);n--;
	for( int i=0,x ; i <= n ; i++ ){
		scanf("%d",&x);
		srt[i / sqrtn].push_back(ii(x , i));
	}
	
	int cntB = n / sqrtn;
	for( int i=0 ; i <= cntB ; i++ )
		sort(srt[i].begin() , srt[i].end());
	
	for( int qq=0,type,l,r,x ; qq < m ; qq++ ){
		scanf("%d",&type);
		if( type == 1 ){
			scanf("%d%d%d",&l,&r,&x);l--;r--;
			if( !x )	continue;
			int leftB = l / sqrtn;
			int righB = r / sqrtn;
			if( leftB == righB ){
				for( int i=0 ; i < (int)srt[leftB].size() ; i++ )
					if( srt[leftB][i].se >= l and srt[leftB][i].se <= r and srt[leftB][i].fi <= MAX )
						srt[leftB][i].fi += x;
				sort(srt[leftB].begin() , srt[leftB].end());
				continue;
			}
			for( int i=0 ; i < (int)srt[leftB].size() ; i++ )
				if( srt[leftB][i].se >= l and srt[leftB][i].fi <= MAX )
					srt[leftB][i].fi += x;
			sort(srt[leftB].begin() , srt[leftB].end());
			leftB++;
			
			for( int i=0 ; i < (int)srt[righB].size() ; i++ )
				if( srt[righB][i].se <= r and srt[righB][i].fi <= MAX )
					srt[righB][i].fi += x;
			sort(srt[righB].begin() , srt[righB].end());
			righB--;
			
			for( int i=leftB ; i <= righB ; i++ )
				if( add[i] <= MAX )
					add[i] += x;
			
		} else {
			scanf("%d",&x);
			int mn=-1 , mx = -2;
			vector < ii > :: iterator it;
			for( int i=0 ; i <= cntB ; i++ )
				if( add[i] < x ){
					it = lower_bound(srt[i].begin() , srt[i].end() , ii(x-add[i] , 0));
					if( it -> fi == x - add[i] ){
						mn = it -> se;break;}
				}
			for( int i=cntB ; i >= 0 ; i-- )
				if( add[i] < x ){
					it = --upper_bound(srt[i].begin() , srt[i].end() , ii(x-add[i] , n+1));
					if( it -> fi == x - add[i] ){
						mx = it -> se;break;}
				}
			printf("%d\n",mx-mn);
		}
	}
	
	return 0;
}
