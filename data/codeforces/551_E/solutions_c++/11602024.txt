#include<bits/stdc++.h>
#define ii pair<int,int>
#define val first
#define id second
using namespace std;
const int MAX=1e9;
pair<int,int> a[500005];
int n,q,GR,NGR,b[1005],bb[1005];
void up(int l,int r,int x)
{
    int i,G1=(l-1)/GR+1,G2=(r-1)/GR+1;
    if (G1==G2)
    {
        for(i=b[G1];i<=b[G1+1];i++) if (l<=a[i].id && a[i].id<=r) a[i].val=min(MAX+1,a[i].val+x);
        sort(a+b[G1],a+b[G1+1]);
    }
    else
    {
        for(i=1; i<=NGR; i++) if (l<=b[i] && b[i+1]-1<=r) bb[i]=min(MAX+1,bb[i]+x);
        if (b[G1]<l) {for(i=b[G1];i<b[G1+1];i++) if (l<=a[i].id) a[i].val=min(MAX+1,a[i].val+x);sort(a+b[G1],a+b[G1+1]);}
        if (r<b[G2+1]-1) {for(i=b[G2];i<b[G2+1];i++) if (a[i].id<=r) a[i].val=min(MAX+1,a[i].val+x);sort(a+b[G2],a+b[G2+1]);}
    }
}
int BS(int i,int x,int type)
{
    int l=b[i],r=b[i+1]-1,m,kq=-1;
    while(l<=r)
    {
        m=(l+r)/2;
        if (a[m].val+bb[i]<x) l=m+1;
        if (a[m].val+bb[i]==x)
        {
            kq=a[m].id;
            if (type==1) r=m-1;
            else l=m+1;
        }
        if (a[m].val+bb[i]>x) r=m-1;
    }
    return kq;
}
int get(int y)
{
    int i,j;
    for(i=1; i<=NGR; i++)
        if (BS(i,y,1)!=-1)
            for(j=NGR; j>=i; j--) if (BS(j,y,2)!=-1) return BS(j,y,2)-BS(i,y,1);
    return -1;
}
main()
{
    cin >> n >> q;
    int i,x,y,l,r,g,key;
    GR=int(sqrt(n));// SL 1 nhom (nhom cuoi co the it hon)
    NGR=(n-1)/GR+1;
    for(i=1; i<=n; i++) cin >> x, a[i]=ii(x,i);

    b[1]=1;
    b[NGR+1]=n+1;

    for(i=2; i<=NGR; i++)
    {
        bb[i-1]=0;
        b[i]=b[i-1]+GR;
        sort(a+b[i-1],a+b[i]);
    }
    sort(a+b[NGR],a+n+1);
    for(i=1; i<=q; i++)
    {
        cin >> key;
        if (key==1)
        {
            cin >> l >> r >> x;
            up(l,r,x);
        }
        else
        {
            cin >> y;
            cout<<get(y)<<endl;
        }
    }
}
