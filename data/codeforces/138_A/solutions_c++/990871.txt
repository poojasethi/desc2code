#include <cstdio>
#include <cstring>
int k;
bool f[4][4];
char s[4][10000];
const char Str[5][5]={"aaaa","aabb","abab","abba","NO"};

bool match(char *s1,char *s2)
{
	int kk=0;
	for (int i=strlen(s1)-1, j=strlen(s2)-1; i>=0 && j>=0 && s1[i]==s2[j] && kk<k; --i,--j)
	    if (s1[i]=='a' || s1[i]=='e' || s1[i]=='i' || s1[i]=='o' || s1[i]=='u') ++kk;
 	return kk==k;
}

int main()
{
	int n,i,j,p,pp;
	scanf("%d%d",&n,&k);
	p=-1;
	bool flag;
	while (n--)
	{
		scanf("%s%s%s%s",s,s+1,s+2,s+3);
		flag=true;
		for (i=0; i<3; ++i)
		    for (j=i+1; j<4; ++j)
			{
			    f[i][j]=match(s[i],s[j]);
			    flag=flag && f[i][j];
			}
  		if (flag) pp=0; else
  		if (f[0][1] && f[2][3]) pp=1; else
  		if (f[0][2] && f[1][3]) pp=2; else
  		if (f[0][3] && f[1][2]) pp=3; else pp=-1;
  		if (pp<0 || p>=0 && pp!=p && pp && p) {p=4; break;}
  		if (p<0) p=pp; else
  		if (!p) p=pp;
	}
	printf("%s\n",Str[p]);
	return 0;
}
