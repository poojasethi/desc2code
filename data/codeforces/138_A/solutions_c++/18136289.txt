
#include<bits/stdc++.h>

typedef long long LL;
using namespace std;
const int M = 1000000007;

LL k;

bool isvowel(char c)
{
return (c == 'a' || c == 'e' || c == 'o' || c =='u' || c =='i' ) ;
}

bool check(string &a,string &b)
{
LL i,t=k,j;
i = a.size()-1;
j = b.size()-1;

for(;i>=0 && j>=0;i--,j--)
	
	{if(a[i]==b[j])
		{
		if(isvowel(a[i])) t--;	
		}
	else
		return false;

	if(t==0)
		return true;
	}

return false;

}


int main()
{
LL n,i;
string p="start";
cin>>n>>k;

while(n--)
{
string a,b,c,d;
cin>>a>>b>>c>>d;

string s;

if(check(a,b) && check(c,d))
		{if(check(b,c))  s = "aaaa";
		else	
				 s = "aabb";
		}
else if(check(a,c) && check(b,d))	
		 s = "abab";
else if(check(a,d) && check(b,c))
		 s = "abba";
else
		 s = "NO";


if(s=="NO")
		{cout<<"NO"<<endl; return 0;}

if(p == "start")
	p = s;
else if(p == "aaaa")
	{if(s=="aaaa") p = "aaaa";
		else p = s;
	}
else if(p==s || s == "aaaa")
	continue;
else
	{cout<<"NO"<<endl; return 0;}

}

cout<<p<<endl;


}
