#include <iostream>
#include <string>
#include <algorithm>
#include <cstdio>
using namespace std;

string a[2500*4+1];

int main(){
	int n, k;
	cin>>n>>k;
	for(int i=0; i<4*n; i++){
		cin>>a[i];
		int z=a[i].size()-1;
		int numvow = 0;
		while(z>=0){
			if(a[i][z]=='a' || a[i][z]=='e' || a[i][z]=='i' || a[i][z]=='o' || a[i][z]=='u'){
				numvow++;
				if(numvow==k){
					a[i] = a[i].substr(z);
					break;
				}
			}
			z--;
		}
		if(z<0) a[i]="";
	}
	int type = 4;
	for(int i=0; i<4*n && type>0; i+=4){
		if(a[i]=="" || a[i+1]=="" || a[i+2]=="" || a[i+3]=="") type=0;
		else if(a[i]==a[i+1] && a[i]==a[i+2] && a[i]==a[i+3]){
		}else if(a[i]==a[i+1] && a[i+2]==a[i+3]){
			if(type!=4 && type!=1) type=0; else type=1;
		}else if(a[i]==a[i+2] && a[i+1]==a[i+3]){
			if(type!=4 && type!=2) type=0; else type=2;
		}else if(a[i]==a[i+3] && a[i+1]==a[i+2]){
			if(type!=4 && type!=3) type=0; else type=3;
		}else type=0;
	}
	switch(type){
		case 0:
			cout<<"NO"<<endl;
			break;
		case 1:
			cout<<"aabb"<<endl;
			break;
		case 2:
			cout<<"abab"<<endl;
			break;
		case 3:
			cout<<"abba"<<endl;
			break;
		case 4:
			cout<<"aaaa"<<endl;
			break;
	}
}
