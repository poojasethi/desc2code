#include <cstdio>
#include <iostream>
using namespace std;

int main()
{
    int n,k;
    scanf("%d%d",&n,&k);
    string last = "aaaa";
    for (int i=0;i<n;++i)
    {
        string v[4];
        string p[4];
        for (int l=0;l<4;++l)
        {
            cin >> v[l];
            int t=0,r;
            for (r=v[l].length()-1;r>=0;--r)
                if (v[l][r]=='a'||v[l][r]=='e'||v[l][r]=='i'||v[l][r]=='o'||v[l][r]=='u')
                    if (++t==k)
                        break;
            if (t!=k)
            {
                printf("NO\n");
                return 0;
            }
            p[l]=v[l].substr(r);
        }
        string cur="NO";
        if (p[0]==p[1]&&p[1]==p[2]&&p[2]==p[3])
            cur = "aaaa";
        else if (p[0]==p[1]&&p[2]==p[3])
            cur = "aabb";
        else if (p[0]==p[2]&&p[1]==p[3])
            cur = "abab";
        else if (p[0]==p[3]&&p[1]==p[2])
            cur = "abba";
        if (last=="aaaa")
            last = cur;
        else if (cur=="aaaa")
            last = last;
        else if (last!=cur)
            last = "NO";
    }
    cout << last << endl;
    return 0;
}
