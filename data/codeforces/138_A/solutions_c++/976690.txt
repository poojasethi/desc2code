#define CC(x,i) do{memset(x,i,sizeof(x));}while(0)
#define INFI 2000000000
#define INFF 1e100

#define MP make_pair
#define X first

typedef unsigned long long  LL;
const LL p=2357;
char s[12345];
int main ( int argc, char *argv[] )
{
	int n,k;scanf("%d%d",&n,&k);
	int AA=7;
	for(int i=0;i<n;i++){
		LL hs[4];int aa=0;
		for(int j=0;j<4;j++)
		{
			scanf("%s",s);
			int l=strlen(s),q,cv=0;hs[j]=0;
			for(q=l-1;q>=0;q--){
				hs[j]=hs[j]*p+s[q];
				if ( s[q]=='a'||
					s[q]=='e'||
					s[q]=='i'||
					s[q]=='o'||s[q]=='u')cv++;
				if(cv>=k)break;
			}
			if(cv<k){puts("NO");return 0;}
		}
		if(hs[0]==hs[3]&&hs[1]==hs[2])aa+=4;
		if(hs[0]==hs[2]&&hs[3]==hs[1])aa+=2;
		if(hs[0]==hs[1]&&hs[3]==hs[2])aa+=1;
		AA&=aa;
		if(AA==0){puts("NO");return 0;}
	}
	if(AA==7)puts("aaaa");
	else if(AA==4)puts("abba");
	else if(AA==2)puts("abab");
	else if(AA==1)puts("aabb");
	return EXIT_SUCCESS;
}
