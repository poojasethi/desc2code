#include<cstdio>
#include<iostream>
#include<string>
#include<algorithm>

using namespace std;

bool is_vowel(char c) { return c=='a'||c=='e'||c=='i'||c=='o'||c=='u'; }

int main(void) {
	int n, k;
	cin >> n >> k;
	string rhyme[4];
	string s;
	string ans="aaaa";
	
	for(int i=0; i<n; i++) {
		for(int j=0; j<4; j++) {
			cin >> s;
			if(ans == "NO") continue;
			int q=(int)s.size()-1, count=0;
			while(q>=0) {
				if(is_vowel(s[q])) count++;
				if(count == k) break;
				q--;
			}
			// cout << q << endl;
			if(q<0) { 
				ans="NO";
				continue;
			}
			rhyme[j] = s.substr(q);
			// cout << rhyme[j] << endl;
		}
		if(ans == "NO") continue;
		else if(rhyme[0] == rhyme[1] && rhyme[0] == rhyme[2] && rhyme[0] == rhyme[3]) continue;
		else if(rhyme[0] == rhyme[1] && rhyme[2] == rhyme[3]) {
			if(ans == "aaaa") ans = "aabb";
			else if(ans != "aabb") ans = "NO";
		} else if(rhyme[0] == rhyme[2] && rhyme[1] == rhyme[3]) {
			if(ans == "aaaa") ans = "abab";
			else if(ans != "abab") ans = "NO";
		} else if(rhyme[0] == rhyme[3] && rhyme[1] == rhyme[2]) {
			if(ans == "aaaa") ans = "abba";
			else if(ans != "abba") ans = "NO";
		} else ans = "NO";
		// cout << ans << endl;
	}
	
	cout << ans << endl;	
}
