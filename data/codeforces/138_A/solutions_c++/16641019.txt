#include <bits/stdc++.h>
using namespace std;

int n,k;
int c[10]; // no:0 , aabb:1 , abab:2 , abba:3, aaaa:4
string x[10];
char vow[] = {'a','e','i','o','u'};

void solve () {

    int ic;
    for (int i=1; i<=4; i++) {

        ic = 0;
        for (int j=x[i].size()-1; j>=0; j-- ) {
            for (int l=0; l<5; l++)
                if (x[i][j] == vow[l])
                    ic++;
            if (ic == k) {
                x[i] = x[i].substr(j);
                break;
            }
        }

        if (ic != k) {
            c[0]++;
        }
    }

    if (x[1] == x[2] && x[3] == x[4]) c[1]++;
    if (x[1] == x[3] && x[2] == x[4]) c[2]++;
    if (x[1] == x[4] && x[2] == x[3]) c[3]++;
    if (x[1] == x[2] && x[2] == x[3] && x[3] == x[4]) c[4]++;
}

int main () {

    cin >> n >> k;

    for (int i=0; i<n; i++) {
        for (int j=1; j<=4; j++)
            cin >> x[j];
        solve();
    }

    string ans = "NO";
    if (c[0]) ans="NO";
    else if (c[4] == n) ans = "aaaa";
    else if (c[1] == n) ans = "aabb";
    else if (c[2] == n) ans = "abab";
    else if (c[3] == n) ans = "abba";

    cout << ans << endl;
}
