#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<algorithm>

using namespace std;

string Get(int k)
{
  char c=getchar();
  while(c<'a'||c>'z')c=getchar();
  int N=1;char S[10000];S[0]=c;
  while(c=getchar(),c>='a'&&c<='z')S[N++]=c;
  int Count=0,i;
  for(i=N-1;i>=0&&Count<k;i--)if(S[i]=='a'||S[i]=='e'||S[i]=='i'||S[i]=='o'||S[i]=='u')Count++;
  if(Count<k)return "";
  string Ans="";
  i++;while(i<N)Ans+=S[i++];
  return Ans;
}

int main()
{
  int n,k,Ans=0;scanf("%d%d",&n,&k);
  for(int i=0;i<n;i++)
    {
      string S1=Get(k),S2=Get(k),S3=Get(k),S4=Get(k);
      if(S1==""||S2==""||S3==""||S4=="")
        Ans=-1;
      if(S1==S2&&S3==S4&&S1!=S3)
        {if(Ans==0||Ans==1)Ans=1;else Ans=-1;}
      else if(S1==S3&&S2==S4&&S1!=S2)
        {if(Ans==0||Ans==2)Ans=2;else Ans=-1;}
      else if(S1==S4&&S2==S3&&S1!=S2)
        {if(Ans==0||Ans==3)Ans=3;else Ans=-1;}
      else if(S1!=S2||S1!=S3||S1!=S4)
        Ans=-1;
    }
  if(Ans==-1)printf("NO\n");
  if(Ans==0)printf("aaaa\n");
  if(Ans==1)printf("aabb\n");
  if(Ans==2)printf("abab\n");
  if(Ans==3)printf("abba\n");
  return 0;
}
