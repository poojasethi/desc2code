#include<string>
#include<cstring>
#include<iostream>

#define rep(i,n) for(int i=0;i<(n);i++)

using namespace std;

bool isvowel(char c){ return strchr("aiueo",c); }

int main(){
	int n,k; cin>>n>>k;
	string s[2500][4];
	rep(i,n) rep(j,4) cin>>s[i][j];

	rep(i,n) rep(j,4) {
		int kk=k,l;
		for(l=(int)s[i][j].length()-1;l>=0;l--) if(isvowel(s[i][j][l])) {
			kk--;
			if(kk==0) break;
		}
		if(l<0){
			cout<<"NO"<<endl;
			return 0;
		}
		s[i][j]=s[i][j].substr(l);
	}

	int Ans=7;
	rep(i,n){
		int S=0;
		if(s[i][0]==s[i][1] && s[i][2]==s[i][3]) S|=1<<0;
		if(s[i][0]==s[i][2] && s[i][1]==s[i][3]) S|=1<<1;
		if(s[i][0]==s[i][3] && s[i][1]==s[i][2]) S|=1<<2;
		Ans&=S;
	}

	if     (Ans==1) cout<<"aabb"<<endl;
	else if(Ans==2) cout<<"abab"<<endl;
	else if(Ans==4) cout<<"abba"<<endl;
	else if(Ans==7) cout<<"aaaa"<<endl;
	else            cout<<"NO"<<endl;

	return 0;
}
