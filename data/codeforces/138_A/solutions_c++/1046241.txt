#include <iostream>
#include <string>
#include <vector>

using namespace std;

vector<int> result;

int solve(vector<string>& d, int k) {
	for (int i = 0; i < 4; ++i) {
		string tmp = "";
		int cnt = 0;
		for (int j = d[i].size() - 1; j >= 0; --j) {
			tmp += d[i][j];
			if (d[i][j] == 'a' || d[i][j] == 'o' || d[i][j] == 'u' || d[i][j] == 'e' || d[i][j] == 'i') {
				cnt++;
				if (cnt == k) {
					cnt = -1;
					break;
				}
			}
		}
		if (cnt != -1) return 0;
		d[i] = tmp;
	}
	if (d[0] == d[1] && d[1] == d[2] && d[2] == d[3]) return 1;
	if (d[0] == d[1] && d[2] == d[3]) return 2;
	if (d[0] == d[3] && d[1] == d[2]) return 3;
	if (d[0] == d[2] && d[1] == d[3]) return 4;
	return 0;
}

int main() {
	int n, k;
	cin >> n >> k;
	while (n--) {
		vector<string> d(4);
		cin >> d[0] >> d[1] >> d[2] >> d[3];
		result.push_back(solve(d, k));
	}
	int p = 0;
	for (int i = 0; i < (int)result.size(); ++i) {
		if (result[i] == 0) {
			cout << "NO\n";
			return 0;
		}
		if (result[i] == 1) continue;
		if (result[i] != p && p != 0) {
			cout << "NO\n";
			return 0;
		}
		p = result[i];
	}
	if (p == 0) cout << "aaaa\n";
	if (p == 2) cout << "aabb\n";
	if (p == 3) cout << "abba\n";
	if (p == 4) cout << "abab\n";
	return 0;
}
