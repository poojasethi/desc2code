#include <algorithm>
#include <iostream>
#include <cstdlib>
using namespace std;
int N,K;
const int MOD=1000000009;
bool is_vowel(char c){
	return c=='a'||c=='e'||c=='u'||c=='i'||c=='o';
}
int input(){
	string s;
	cin>>s;
	int cnt=0;
	long long ans=0;
	for(int i=s.size()-1;i>=0;i--){
		if(is_vowel(s[i])){
			cnt++;
		}
		ans*=307;
		ans+=s[i];
		ans%=MOD;
		if(cnt>=K){
			return ans;
		}
	}
	cout<<"NO"<<endl;
	exit(0);
}
int in[2500][4];
string answer[4]={"aaaa","aabb","abab","abba"};
bool check(int a[4],int t){
	switch(t){
	case 0:
		return a[0]==a[1]&&a[1]==a[2]&&a[2]==a[3];
	case 1:
		return a[0]==a[1]&&a[2]==a[3];
	case 2:
		return a[0]==a[2]&&a[1]==a[3];
	case 3:
		return a[0]==a[3]&&a[1]==a[2];
	}
	return false;
}
int main() {
	cin>>N>>K;
	for(int i=0;i<N;i++){
		for(int j=0;j<4;j++){
			in[i][j]=input();
		}
	}
	for(int i=0;i<4;i++){
		bool ok=true;
		for(int j=0;j<N;j++){
			if(!check(in[j],i)){
				ok=false;break;
			}
		}
		if(ok){
			cout<<answer[i]<<endl;
			return 0;
		}
	}
	cout<<"NO"<<endl;
	return 0;
}
