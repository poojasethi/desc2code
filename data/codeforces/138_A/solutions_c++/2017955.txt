#include <iostream>
#include <string>
using namespace std;

int k;
int isVowel(char c) { return c=='a' || c=='u' || c=='o' || c=='e' || c=='i'; }

string last(string s) {
	int cnt = 0;
	for(int i = s.size()-1; i >= 0; i--) {
		if(isVowel(s[i])) {
			cnt++;
			if(cnt == k) {
				return s.substr(i);
			}
		}
	}
	return "BAD";
}

int main() {
	int n;
	string res;
	int r = 0, tmp; // 0->aaaa 1->aabb 2->abba 3->abab
	cin >> n >> k;
	string a[4], f[4];
	for(int i = 0; i < n; i++) {
		for(int j=0; j<4; j++) cin >> a[j];

		for(int j=0; j<4; j++) {f[j] = last(a[j]); if(f[j] == "BAD") goto impos;}

		if(f[0] == f[1] && f[1]==f[2] && f[2]==f[3]) tmp = 0;
		else if(f[0]==f[1] && f[2]==f[3]) tmp = 1;
		else if(f[0]==f[2] && f[1]==f[3]) tmp = 3;
		else if(f[0]==f[3] && f[1]==f[2]) tmp = 2;
		else goto impos;
		
		if(tmp == 0) ;
		else if(r==0) r = tmp;
		else if(r == tmp) r = tmp;
		else goto impos;
	}


pos:
	switch(r) {
	case 0:res = "aaaa";break;
	case 1:res = "aabb";break;
	case 2:res = "abba";break;
	case 3:res = "abab";break;
	}
	cout << res << endl;
	return 0;

impos:
	cout << "NO" << endl;
	return 0;
}
