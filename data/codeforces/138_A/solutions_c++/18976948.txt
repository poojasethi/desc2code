#include<bits/stdc++.h>
using namespace std;
int x,y,cnt;
bool flag;
string s,d[100001],r,ans="?";
int main(){
    cin>>x>>y;
    for(int i=0;i<(4*x);i++){
        cin>>s;
        cnt=0;
        if(s.size()<y)flag=true;
        for(int j=s.size()-1;j>=0;j--){
            d[i]+=s[j];
            if(s[j]=='a'||s[j]=='e'||s[j]=='i'||s[j]=='o'||s[j]=='u'){
                cnt++;
            }
            if(cnt==y)break;
        }
    }
    if(flag){
        cout<<"NO"<<endl;
        exit(0);
    }
    for(int i=0;i<(4*x);i+=4){
        if(d[i]==d[i+1]&&d[i+1]==d[i+2]&&d[i+2]==d[i+3])r="aaaa";
        else if(d[i]==d[i+1]&&d[i+2]==d[i+3])r="aabb";
        else if(d[i]==d[i+2]&&d[i+1]==d[i+3])r="abab";
        else if(d[i]==d[i+3]&&d[i+1]==d[i+2])r="abba";
        else{
            cout<<"NO"<<endl;
            exit(0);
        }
        if(ans=="?"&&r!="aaaa")ans=r;
        if(r=="aaaa")continue;
        else{
            if(r!=ans){
                cout<<"NO"<<endl;
                exit(0);
            }
        }
    }
    if(ans=="?")ans="aaaa";
    cout<<ans<<endl;
}
