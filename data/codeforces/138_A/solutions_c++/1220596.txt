#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
#include <iostream>
using namespace std;

bool isvowel(char x) {
	return x=='a'||x=='e'||x=='i'||x=='o'||x=='u';
}

string getsuffix(string s, int k) {
	int i=0;
	for(i=(int)s.size()-1;i>=0;--i) {
		if(isvowel(s[i]) && --k==0) break;
	}
	if(i<0) return "";
	return s.substr(i);
}

int main(void) {
	int n, k;
	cin >> n >> k;
	int t[4]={};
	for(int i=0;i<n;i++) {
		string s[4];
		for(int j=0;j<4;j++) cin >> s[j];
		for(int j=0;j<4;j++) {
			s[j]=getsuffix(s[j], k);
			if(s[j]=="") { i=n; break; }
		}
		if(s[0]==s[1] && s[2]==s[3]) ++t[0];
		if(s[0]==s[2] && s[1]==s[3]) ++t[1];
		if(s[0]==s[3] && s[1]==s[2]) ++t[2];
		if(s[0]==s[1] && s[1]==s[2] && s[2]==s[3]) ++t[3];
	}
	if(t[3]==n) puts("aaaa");
	else if(t[0]==n) puts("aabb");
	else if(t[1]==n) puts("abab");
	else if(t[2]==n) puts("abba");
	else puts("NO");
	return 0;
}
