#include <bits/stdc++.h>

using namespace std;

int isVowel(char c)
{
     if(c=='a' || c=='e' || c=='i' || c=='o' || c=='u')
        return 1;
    return 0;
}
int main()
{
    int n,k,i,j,l,c;

   // freopen("input.txt","r",stdin);
    cin>>n>>k;
    int aabb=1,abab=1,abba=1;
    for(i=0;i<n;i++)
    {
        vector < string > v(4);
        cin>>v[0]>>v[1]>>v[2]>>v[3];

        for(j=0;j<4;j++)
        {for(c=0,l=v[j].size()-1;l>=0 && c<k;l--)
            {if(isVowel(v[j][l]))
                c++;}
        if(c!=k)
        {
            cout<<"NO"<<endl; return 0;
        }
        else v[j]=v[j].substr(l+1);}
        //cout<<v[0]<<v[1]<<v[2]<<v[3]<<endl;;
        if(v[0]!=v[1] || v[2]!=v[3]) aabb=0;
        if(v[0]!=v[3] || v[1]!=v[2]) abba=0;
        if(v[0]!=v[2] || v[1]!=v[3]) abab=0;
        //cout<<"s"<<endl;
    }
    if(abab && abba) cout<<"aaaa"<<endl;
    else if(abba) cout<<"abba"<<endl;
    else if(abab) cout<<"abab"<<endl;
    else if(aabb) cout<<"aabb"<<endl;
    else cout<<"NO"<<endl;

}
