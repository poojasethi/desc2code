#include <iostream>
#include <vector>
#include <string>
using namespace std;
int main() {
    int n, k;
    cin >> n >> k;
    vector<string> a;
    for (int i = 0; i < 4*n; ++ i) {
        string buf;
        cin >> buf;
        int c = 0;
        for (int i = buf.length()-1; i >= 0; -- i) {
            if (buf[i] == 'a' || buf[i] == 'i' || buf[i] == 'u' || buf[i] == 'e' || buf[i] == 'o') {
                if (++ c == k) {
                    a.push_back(buf.substr(i));
                    break;
                }
            }
        }
        if (c < k) {
            cout << "NO" << endl;
            return 0;
        }
    }
    int x = 7;
    for (int i = 0; i < n; ++ i) {
        if (a[4*i+0] != a[4*i+1] || a[4*i+2] != a[4*i+3]) x &= ~1;
        if (a[4*i+0] != a[4*i+2] || a[4*i+1] != a[4*i+3]) x &= ~2;
        if (a[4*i+0] != a[4*i+3] || a[4*i+1] != a[4*i+2]) x &= ~4;
    }
    if (x == 0) {
        cout << "NO" << endl;
    } else if (x == 1) {
        cout << "aabb" << endl;
    } else if (x == 2) {
        cout << "abab" << endl;
    } else if (x == 4) {
        cout << "abba" << endl;
    } else {
        cout << "aaaa" << endl;
    }
}
