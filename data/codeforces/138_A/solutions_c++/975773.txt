#include <cstdio>
#include <iostream>
#include <cstring>

using namespace std;

char str1[20005],str2[20005],str3[20005],str4[20005];
long n,k;

bool check(char str1[],char str2[])
{
 long l1,l2;
 long i,j;
 long t=k;
 
 l1=strlen(str1);
 l2=strlen(str2);
 
 for(i=l1-1;t&&i>=0;i--)
   if(str1[i]=='a'||str1[i]=='e'||str1[i]=='i'||str1[i]=='o'||str1[i]=='u')
    t--;
 if(t)
  return false;
 i++;
 j=l2-(l1-i);
 if(j<0)
  return false;
 for(;str1[i]==str2[j]&&str1[i];i++,j++);
 if(str1[i]||str2[j])
  return false;
 return true;
}

int main()
{
 long a[5]={0};
 
 scanf("%ld%ld",&n,&k);
 
 for(long i=1;i<=n;i++)
   {
    scanf("%s%s%s%s",str1,str2,str3,str4);
    if(check(str1,str2)&&check(str3,str4))
     {
      if(check(str2,str3))
       continue;
      a[1]=1;
      continue;
     }
    else if(check(str1,str3)&&check(str2,str4))
     a[2]=1;
    else if(check(str1,str4)&&check(str2,str3))
     a[3]=1;
    else
     {
      printf("NO\n");
      return 0;
     }
   }
 if(a[1]+a[2]+a[3]==0)
  printf("aaaa\n");
 else if(a[1]+a[2]+a[3]==1)
  printf("%s\n",a[1]?"aabb":a[2]?"abab":"abba");
 else 
  printf("NO\n");
 
 return 0;
}
