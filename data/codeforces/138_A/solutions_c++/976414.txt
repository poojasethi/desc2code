#include<iostream>
#include<stdio.h>
#include<string.h>
#include<algorithm>
#include<cmath>
#include<string>
using namespace std;

#define F(i,n) for(i=0;i<n;i++)
#define FF(i,n) for(i=1;i<=n;i++)
#define fit( x ) ( (x)=='a'||(x)=='e'||(x)=='i'||(x)=='o'||(x)=='u' )

string s[2600*4];
int n, k;
// aaaa: 0 
// aabb: 1
// abab: 2
// abba: 4

int gao(){
	int i, j, p, ans = 0;
	for(i=0;i<4*n;i++){
		for(p=0,j = s[i].length()-1;j>=0;j--){
			if( fit( s[i][j] ) ) p++;
			if( p==k ) break;
		}
		if( p!=k ) return 3;
		s[i].erase(0, j);
	}
	F(i,n){
		int pos = i*4, tmp = 0;
		if( s[pos+0] == s[pos+1] && s[pos+2] == s[pos+3] ) tmp |= 1; 
		if( s[pos+0] == s[pos+2] && s[pos+1] == s[pos+3] ) tmp |= 2; 
		if( s[pos+0] == s[pos+3] && s[pos+1] == s[pos+2] ) tmp |= 4; 
		if( tmp == 0 ) return 3;
		if( tmp == 7 ) tmp = 0;
		ans |= tmp;
	}
	return ans;
}

int main(){
     int i, j;
     while(cin>>n>>k){
		F(i,n*4) cin>>s[i];
		i = gao();
		if( i== 0) cout<< "aaaa"<<endl;
		else if(i==1) cout<<"aabb"<<endl;
		else if(i==2) cout<<"abab"<<endl;
		else if(i==4) cout<<"abba"<<endl;
		else cout<<"NO"<<endl;
	}
     return 0;
}
