#include <iostream>
#include <string>

using namespace std;

int N,K;
string S[4];

string suffix(string s)
{
	int cnt=0;
	for (int i=s.size()-1;i>=0;i--)
	{
		if (s[i]=='a'||s[i]=='e'||s[i]=='i'||s[i]=='o'||s[i]=='u')
			cnt++;
		if (cnt==K)
			return s.substr(i,s.size()-i);
	}
	return "";
}

string scheme()
{
	string suf[4];
	for (int i=0;i<4;i++)
	{
		suf[i]=suffix(S[i]);
		if (suf[i].empty())
			return "NO";
	}
	if (suf[0]==suf[1]&&suf[0]==suf[2]&&suf[0]==suf[3])
		return "aaaa";
	if (suf[0]==suf[1]&&suf[2]==suf[3])
		return "aabb";
	if (suf[0]==suf[2]&&suf[1]==suf[3])
		return "abab";
	if (suf[0]==suf[3]&&suf[1]==suf[2])
		return "abba";
	return "NO";
}

int main()
{
	string prev;
	int check=0;
	cin >> N >> K;
	for (int i=0;i<N;i++)
	{
		for (int j=0;j<4;j++)
			cin >> S[j];
		if (!check)
		{
			string s=scheme();
			if (s=="NO")
				check=1;
			else
			{
				if (prev.empty()||(prev=="aaaa"&&s!="aaaa"))
					prev=s;
				else if (s!=prev&&s!="aaaa")
					check=1;
			}
		}
	}
	if (check)
		cout << "NO" << endl;
	else
		cout << prev << endl;
	return 0;
}
