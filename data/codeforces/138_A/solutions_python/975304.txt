def X():
	print"NO"
	exit()
n,k=map(int,raw_input().split())
q=0
for i in [0]*n:
	t=[]
	for j in [0]*4:
		c=raw_input()
		v=0
		for p,l in enumerate(c[::-1]):
			if l in "aeoiu":
				v+=1
				if v==k:break
		if v<k:X()
		t.append(c[-p-1:])
	if t[0]==t[1]==t[2]==t[3]:
		continue
	elif t[0]==t[1]and t[2]==t[3]:
		if q>1:X()
		q=1
	elif t[0]==t[2]and t[1]==t[3]:
		if q&1:X()
		q=2
	elif t[0]==t[3]and t[1]==t[2]:
		if q%3:X()
		q=3
	else:X()
print"aaaaaabbabababba"[q::4]
