
#include <iostream>
#include <vector>
#define MAXN 100000

using namespace std;

int a[MAXN];
int cnt[MAXN + 1];
int n, m;

int l[MAXN], r[MAXN], ans[MAXN];
int pcnt[MAXN + 1];

int main() {
	cin >> n >> m;
	for (int i = 0; i < n; i++) {
		cin >> a[i];
		if (a[i] <= n)
			cnt[a[i]]++;
	}
	for (int i = 0; i < m; i++)
		cin >> l[i] >> r[i];

	for (int i = 1; i <= n; i++) {
		if (cnt[i] < i)
			continue;

		pcnt[0] = 0;
		for (int j = 0; j < n; j++)
			pcnt[j + 1] = pcnt[j] + (a[j] == i ? 1 : 0);

		for (int j = 0; j < m; j++)
			if (pcnt[r[j]] - pcnt[l[j] - 1] == i)
				ans[j]++;
	}

	for (int i = 0; i < m; i++)
		cout << ans[i] << endl;
}
