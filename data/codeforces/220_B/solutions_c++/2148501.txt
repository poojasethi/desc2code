#include<iostream>
#include<algorithm>
#include<cstdio>
const int MAX =101*1000;
int l[MAX],r[MAX],out[MAX],p[MAX];
int cnt[MAX];
using namespace std;
int main()
{
	int n,m;
	scanf("%d %d",&n,&m);
	int x[n];
	for(int i=1;i<=n;i++)
	{
		scanf("%d",&x[i]);
		if(x[i]<MAX)
		{
			cnt[x[i]]++;
		}
	}
	for(int i=0;i<m;i++)
		scanf("%d %d",&l[i],&r[i]);
	for(int num=1;num<MAX;num++)
	{
		if(cnt[num]>=num)
		{
			p[0]=0;
			for(int i=1;i<=n;i++)
			{
				p[i]=p[i-1]+(x[i]==num);
			}
			for(int i=0;i<m;i++)
			{
				out[i]+=p[r[i]]-p[l[i]-1]==num;
			}
		}
	}
	for(int i=0;i<m;i++)
		printf("%d\n",out[i]);
	return 0;

}
