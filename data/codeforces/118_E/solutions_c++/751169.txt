#include<iostream>
#include<cstdio>

using namespace std;

const int maxn=100010;
int n, m, x, y, ans[maxn*3][2];
int px[maxn*6], next[maxn*6], P[maxn], cnt;
int d[maxn], low[maxn];
bool flag;

void dfs(int x, int fa, int dep){
	d[x]=low[x]=dep;
	for (int p=P[x]; p!=0; p=next[p])
		if (px[p]!=fa){
			if (low[px[p]]==0){
				dfs(px[p], x, dep+1);
				if (low[px[p]]>dep) flag=true;
				low[x]=min(low[x], low[px[p]]);
				ans[cnt][0]=x; ans[cnt++][1]=px[p];
			}else if (d[px[p]]<d[x]){
				low[x]=min(low[x], d[px[p]]);
				ans[cnt][0]=x; ans[cnt++][1]=px[p];
			}
		}
}

int main(){
	scanf("%d%d", &n, &m);
	for (int i=0; i<m; i++){
		scanf("%d%d", &x, &y);
		px[++cnt]=y; next[cnt]=P[x]; P[x]=cnt;
		px[++cnt]=x; next[cnt]=P[y]; P[y]=cnt;
	}
	cnt=0; dfs(1, 0, 1);
	if (flag) printf("0\n");
	else for (int i=0; i<m; i++) printf("%d %d\n", ans[i][0], ans[i][1]);
}
