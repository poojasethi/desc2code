#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <vector>

#define fi first
#define se second
#define pb push_back

using namespace std;

typedef pair<int,int> ii;

const int MAXN=200020;

int N,M,cnt;
int id[MAXN];
int mn[MAXN];

bool used[MAXN];

vector<ii> way;
vector<int> g[MAXN];

void dfs( int nd , int pre=0 ){
	
	used[nd]=1;
	id[nd]=mn[nd]=++cnt;
	
	for( vector<int>::iterator it=g[nd].begin() ; it!=g[nd].end() ; it++ ){
		
		if( *it==pre )
			continue;
		
		if( used[*it] ){
			mn[nd]=min(mn[nd],id[*it]);
			if( id[nd]>id[*it] )
				way.pb(ii(nd,*it));
		}
		else{
			way.pb(ii(nd,*it));
			dfs(*it,nd);
			mn[nd]=min(mn[nd],mn[*it]);
			if( mn[*it]>id[nd] ){
				printf("0\n");
				exit(0);
			}
		}
	}
}

int main(){
	
	scanf(" %d %d",&N,&M);
	
	for( int a,b,i=0 ; i<M ; i++ ){
		scanf(" %d %d",&a,&b);
		g[a].pb(b);
		g[b].pb(a);
	}
	
	dfs(1);
	
	for( vector<ii>::iterator it=way.begin() ; it!=way.end() ; it++ )
		printf("%d %d\n",it->fi,it->se);
	
	return 0;
}
