#include <bits/stdc++.h>
#define F first
#define S second
using namespace std;
typedef pair<int,int> pii;
const int maxn=1e5+8;

vector <int> a[maxn];
vector <pii> ans;
int flag[maxn],par[maxn];
bool Ok;
inline int dfs(int x)
{
    int ret=flag[x];
    for(int i=0;i<a[x].size();i++)
    {
        int G=a[x][i];
        if(flag[G]==0)
        {
            flag[G]=flag[x]+1;
            ans.push_back(pii(x,G));
            par[G]=x;
            ret=min(ret,dfs(G));
        }
        else if(par[x]!=G)
            ret=min(ret,flag[G]);
    }
    for(int i=0;i<a[x].size();i++)
    {
        int G=a[x][i];
        if(flag[G]>flag[x]&&par[G]!=x)
            ans.push_back(pii(G,x));

    }
    if(ret>=flag[x]&&x!=1)
        Ok=false;
    return ret;

}

int main()
{
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=m;i++)
    {
        int x,y;
        cin>>x>>y;
        a[x].push_back(y);
        a[y].push_back(x);
    }
    flag[1]=1;
    Ok=true;
    dfs(1);
    if(Ok==false)
        cout<<0;
    else
        for(int i=0;i<ans.size();i++)
            cout<<ans[i].F<<" "<<ans[i].S<<"\n";
    return 0;
}
