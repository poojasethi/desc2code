#include <bits/stdc++.h>

using namespace std;

typedef long long ll;
typedef pair<int, int> pii;

const int MAXN = 1e5 + 100, MAXE = 3e5 + 100;

vector<int> G[MAXN];
bool vis[MAXN];
pii ans[MAXE]; int cnt;
int h[MAXN];

int dfs(int u, int par){
//	cerr << " at " << u << endl;
	h[u] = (par == -1 ? 0 : h[par] + 1);
	vis[u] = true;

	int minh = -1u/4, temp = -1u/4;
	bool child = false;

	for (int i = 0; i < G[u].size(); i++){
		int v = G[u][i];
		if (vis[v] && v != par && h[v] < h[u]){
			ans[cnt++] = pii(u + 1, v + 1);
			minh = min(minh, h[v]);
		}
		else if (!vis[v]) {
			ans[cnt++] = pii(u + 1, v + 1);
			temp = dfs(v, u);
			if (temp > h[u]){
				cout << 0;
				exit(0);
			}

			minh = min(minh, temp);
			child = true;
		}
	}

//	cerr << u << " is " << h[u] << " : " << minh << " & " << temp << endl;

//	cerr << "up\n";
	return minh;
}

int main(){
	ios::sync_with_stdio( false );
	cout.tie(0); cin.tie(0);

	int n, m; cin >> n >> m;
	for (int i = 0; i < m; i++){
		int u, v; cin >> u >> v; u--; v--;
		G[u].push_back(v);
		G[v].push_back(u);
	}

	dfs(0, -1);

	for (int i = 0; i < cnt; i++)
		cout << ans[i].first << " " << ans[i].second << "\n";


}
