#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<string>
#include<cstring>
using namespace std;
int n,m;
int Fs[100010],Nx[600010],Y[600010],E=1,Dir[600010];

int Dfn[100010],Low[100010],DFS;

void Ins(int x,int y)
{
	Nx[++E]=Fs[x];Fs[x]=E;Y[E]=y;
	Nx[++E]=Fs[y];Fs[y]=E;Y[E]=x;
}

bool Dfs(int X)
{
	Dfn[X]=Low[X]=++DFS;
	for(int v=Fs[X];v;v=Nx[v])if(!Dir[v])
	{
		Dir[v]=1;Dir[v^1]=-1;
		if(!Dfn[Y[v]])
		{
			if(!Dfs(Y[v]))return 0;
			Low[X]=min(Low[X],Low[Y[v]]);
		}
		else Low[X]=min(Low[X],Dfn[Y[v]]);
	}
	if(X!=1&&Dfn[X]==Low[X])return 0;
	return 1;
}

int main()
{
	//freopen("input.txt","rb",stdin);
	//freopen("output.txt","wb",stdout);
	scanf("%d%d",&n,&m);
	for(int i=0,x,y;i<m;++i)
	{
		scanf("%d%d",&x,&y);
		Ins(x,y);
	}
	if(!Dfs(1))printf("0\n");else 
	for(int i=1;i<=n;++i)
		for(int v=Fs[i];v;v=Nx[v])if(Dir[v]==1)cout<<i<<' '<<Y[v]<<endl;
	return 0;
}
