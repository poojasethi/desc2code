#include <bits/stdc++.h>

#define ll long long
#define ld long double

using namespace std;

const int MAXN=100*100*10*3+10;
vector<pair<int,int> >a[MAXN];
vector<int>b[2][MAXN];
int ptr[MAXN],cnt[2],n,m;
bool mark[MAXN],vis[2][MAXN];

void dfs2(int v,int k){
	cnt[k]++;
	vis[k][v]=true;
	for(int i=0;i<b[k][v].size();i++){
		int u=b[k][v][i];
		if(!vis[k][u])
			dfs2(u,k);
	}
}

void dfs(int v){
	while(ptr[v]<a[v].size()){
		int u = a[v][ptr[v]].first , id = a[v][ptr[v]].second;
		if(!mark[id]){
			mark[id]++;
			dfs(u);
			b[0][u].push_back(v);
			b[1][v].push_back(u);
		}
		ptr[v]++;
	}
}

int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);cout.tie(0);
	int x,y,t=0;
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>x>>y;
		a[x].push_back(make_pair(y,t));
		a[y].push_back(make_pair(x,t++));
	}
	dfs(1);
	dfs2(1,0);
	dfs2(1,1);
	if(cnt[0]!=n or cnt[1]!=n)
		return cout<<0,0;
	for(int i=1;i<n+1;i++)
		for(int j=0;j<b[0][i].size();j++)
		cout<<i<<' '<<b[0][i][j]<<endl;
	return 0;
}
