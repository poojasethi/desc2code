#include<bits/stdc++.h>

using namespace std;

vector < int > V[100005];

int D[100005],Pre[100005],n,w,i,j,s;

bool h;

void f(int x){

	int i;

	D[x] = ++s;

	for(i=0;i<V[x].size();i++){
        
		if(!D[ V[x][i] ]){
			Pre[ V[x][i] ] = x;
			f(V[x][i]);
		}

		if(!h && (Pre[ V[x][i] ] == x || (D[x] > D[ V[x][i] ] && V[x][i] != Pre[x])))
			V[x].erase(V[x].begin() + i--);
	}
}
int main(){

	cin >> n >> w;

	while(w--){
		scanf("%d%d",&i,&j);
		V[i].push_back(j);
		V[j].push_back(i);
	}

	f(1);
    memset(D , 0  , sizeof D); h=1;
	f(1);

	if(s != 2*n) cout << 0;

	else
		for(i=1;i<=n;i++)
			for(j=0;j<V[i].size();j++)
				printf("%d %d\n",i,V[i][j]);

	return 0;
}
