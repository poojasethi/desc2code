#include <bits/stdc++.h>
using namespace std;
int n,m,Time = 0;
vector<int> num,low;
vector<pair<int,int> > res;
vector<vector<int> > graph;
void input(){
    scanf("%d%d",&n,&m);
    graph.resize(n);
    for(int i = 0;i < m;++i){
        int u,v;
        scanf("%d%d",&u,&v);
        u--;v--;
        graph[u].push_back(v);
        graph[v].push_back(u);
    }
    num = low = vector<int> (n,-1);
}
void dfs(int u,int dad){
    num[u] = low[u] = Time++;
    for(vector<int> :: iterator it = graph[u].begin();it != graph[u].end();++it){
        int v = *it;
        if(v == dad) continue;
        if(num[v] == -1){
            res.push_back(make_pair(u,v));
            dfs(v,u);
            low[u] = min(low[u],low[v]);
            if(low[v] >= num[v]) {
                puts("0");exit(0);
            }
        }
        else{
            low[u] = min(low[u],num[v]);
            if(num[v] < num[u]) res.push_back(make_pair(u,v));
        }
    }
}
int main(){
    //freopen("input.txt","r",stdin);
    input();
    dfs(0,-1);
    for(unsigned i = 0;i < res.size();++i)
        printf("%d %d\n",res[i].first+1,res[i].second+1);
return 0;
}
