#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
int n,m,i,a,b,T[100001],F[100001],R,l;
vector <int> g[100001];
bool U[100001];
pair<int,int> O[300000];
void S(int v,int p=-1){
	U[v]=true;
	T[v]=F[v]=R++;
	for(int i=0;i<g[v].size();i++){
		int C=g[v][i];
		if(C==p)
			continue;
		if(U[C]){
			F[v]=min(F[v],T[C]);
			if(T[C]<T[v])
				O[l++]=make_pair(v,C);
		}
		else{
			O[l++]=make_pair(v,C);
			S(C,v);
			F[v]=min(F[v],F[C]);
			if(F[C]>T[v]){
				printf("0");
				exit(0);
			}
		}
	}
}
main(){
	scanf("%d%d",&n,&m);
	for(i=0;i<m;i++){
		scanf("%d%d",&a,&b);
		g[a].push_back(b);
		g[b].push_back(a);
	}
	S(1);
	for(i=0;i<l;i++)
		printf("%d %d\n",O[i].first,O[i].second);
}
