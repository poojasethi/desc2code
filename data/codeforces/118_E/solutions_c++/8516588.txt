#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
#include <cstring>
using namespace std;
class sp{
  public:
  int q,w;
  sp *e;
  sp(int q1,int w1,sp *e1){
   q=q1;w=w1;e=e1;}
};
sp *a[100000];
int s[300000],d[300000],f[100000],g[100000];
bool h[300000];
int k(int q){
  if(g[g[q]]!=g[q])
   g[q]=k(g[q]);
  return g[q];
}
void l(int q,int e){
  sp *c;
  int r=q,t;
  f[q]=e;
  for(c=a[q];c;c=c->e)
   if(h[c->w]){
    h[c->w]=0;
    if(f[c->q]==0)
     l(c->q,e+1);
    if(f[k(c->q)]>0){
     if(s[c->w]!=q){
      t=s[c->w];s[c->w]=d[c->w];d[c->w]=t;}
     if(f[k(c->q)]<f[r])
      r=k(c->q);}}
  g[q]=k(r);
  f[q]=-1;
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t;
  scanf("%d%d",&q,&w);
  for(e=0;e<q;e++){
   a[e]=0;
   f[e]=0;
   g[e]=e;}
  for(e=0;e<w;e++){
   scanf("%d%d",&s[e],&d[e]);
   s[e]--;d[e]--;
   a[s[e]]=new sp(d[e],e,a[s[e]]);
   a[d[e]]=new sp(s[e],e,a[d[e]]);
   h[e]=1;}
  l(0,1);
  for(e=1;e<q;e++)
   if(k(e)!=k(0)){
    cout<<0;
    return 0;}
  for(e=0;e<w;e++)
   printf("%d %d\n",s[e]+1,d[e]+1);
  return 0;}
