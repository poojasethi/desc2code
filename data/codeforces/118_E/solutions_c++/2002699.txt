#include <cstdio>
#include <cstring>
#include <vector>

using namespace std;

vector<int> e[100001],no[100001],f[100001];
bool v[100001],u[300001];
int n,m,t;

void dfs(int x)
{
  v[x]=true;
  for(int z=0;z<(int)e[x].size();++z)
  {
    int y=e[x][z],w=no[x][z];
    if(!u[w])
    {
      u[w]=true;
      f[y].push_back(x);
      if(!v[y])
        dfs(y);
    }
  }
}

void check(int x)
{
  v[x]=true;
  ++t;
  for(int z=0;z<(int)f[x].size();++z)
  {
    int y=f[x][z];
    if(!v[y])
      check(y);
  }
}

int main(void)
{
  scanf("%d %d",&n,&m);
  for(int i=1;i<=m;++i)
  {
    int j,k;
    scanf("%d %d",&j,&k);
    e[j].push_back(k);
    no[j].push_back(i);
    e[k].push_back(j);
    no[k].push_back(i);
  }
  dfs(1);
  memset(v,0,sizeof(v));
  check(1);
  if(t<n)
    printf("0\n");
  else
    for(int i=1;i<=n;++i)
      for(int j=0;j<(int)f[i].size();++j)
        printf("%d %d\n",f[i][j],i);
  return 0;
}
