# include <bits/stdc++.h>

using namespace std;

# define L first
# define R second

const int maxn = 2e6  + 100;

pair <int,int> a[maxn];

int main(){
	ios_base::sync_with_stdio(false);
	int n, k;
	cin >> n >> k;
	for (int i = 0; i < n; i++){
		cin >> a[2 * i + 0].L, a[2 * i + 0].R = 0;
		cin >> a[2 * i + 1].L, a[2 * i + 1].R = 1;
	}
	sort(a, a + 2 * n);
	
	int stck = 0, l;
	vector <pair<int,int> > ans;
	for (int i = 0; i < 2 * n; i++)
		if(a[i].R == 0){
			stck++;
			if(stck == k)
				l = a[i].L;
		}
		else{
			if(stck == k)
				ans.push_back(make_pair(l, a[i].L));
			stck--;
		}
	
	cout << ans.size() << '\n';
	for (int i = 0; i < ans.size(); i++)
		cout << ans[i].L << " " << ans[i].R << '\n';
	return 0;
}
