#include <stdio.h>
#include <algorithm>

using namespace std;

int n, k;
int beg[1000010], end[1000010], ans[1000010][2], a = 0;

int main()
{
	int i;
	scanf("%d%d", &n, &k);
	for(i = 0; i < n; i++)
		scanf("%d%d", &beg[i], &end[i]);
	sort(beg, beg + n);
	sort(end, end + n);
	int cur = 0, e = 0;				
	ans[a][0] = -1000000001;
	for(i = 0; i < n; i++)
	{
		while(end[e] < beg[i])
		{
			e++, cur--;
			if(cur == k - 1)
			{
				ans[a++][1] = end[e - 1];
				ans[a][0] = -1000000001;
			}
		}
		cur++;
		if(cur == k)
			ans[a][0] = beg[i];
	}
	while(cur > k)
		cur--, e++;
	if(ans[a][0] != -1000000001)
		ans[a++][1] = end[e];
	printf("%d\n", a);
	for(i = 0; i < a; i++)
		printf("%d %d\n", ans[i][0], ans[i][1]);
	return 0;
}
