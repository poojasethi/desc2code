#include <cstdio>
#include <algorithm>

#define f first
#define s second

using namespace std;

pair <int, int> v[2000010], rez[1000010];

int main ()
{
    int n, k;
    scanf ("%d %d", &n, &k);

    int p = 0;
    for (int i = 1; i <= n; ++i)
    {
        scanf ("%d %d", &v[p + 1].f, &v[p + 2].f);
        v[++p].s = -1;
        v[++p].s = 1;
    }

    sort (v + 1, v + p + 1);

    int x = 0, w = 0;
    for (int i = 1; i <= p; ++i)
    {
        x -= v[i].s;
        if (x == k && v[i].s == -1) rez[++w].f = v[i].f;
        if (x == k - 1 && v[i].s == 1) rez[w].s = v[i].f;
    }

    printf ("%d\n", w);

    for (int i = 1; i <= w; ++i)
        printf ("%d %d\n", rez[i].f, rez[i].s);

    return 0;
}
