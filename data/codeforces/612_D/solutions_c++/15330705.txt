#include<iostream>
#include<stdio.h>
#include<algorithm>
using namespace std;
struct node
{
    int num,id;
    friend bool operator<(node x,node y)
    {
        if(x.num==y.num) return x.id>y.id;
        return x.num<y.num;
    }
}a[2000005],ans[1000005];
int n,k;
int main()
{
    cin>>n>>k;
    for(int i=1;i<=2*n;i+=2)
    {
        scanf("%d%d",&a[i].num,&a[i+1].num);
        a[i].id=1,a[i+1].id=-1;
    }
    sort(a+1,a+1+2*n);
    int l=-2139062143,tot=0,sum=0;
    for(int i=1;i<=2*n;i++)
    {
        tot+=a[i].id;
        if(tot>=k&&l==-2139062143) l=a[i].num;
        if(tot<k&&l!=-2139062143) ans[++sum]=(node){l,a[i].num},l=-2139062143;
    }
    cout<<sum<<endl;
    for(int i=1;i<=sum;i++)
        printf("%d %d\n",ans[i].num,ans[i].id);
}
