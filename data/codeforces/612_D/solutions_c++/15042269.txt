#include <iostream>
#include <vector>
#include <algorithm>

#define z std
#define q z::cin
#define w z::cout
#define mp(a, b) z::make_pair(a, b)

int main(){
	z::ios::sync_with_stdio(false);
	q.tie(0); w.tie(0);
	int x, y, n, k, b = 0, lb = 0, l = 0;
	z::vector<z::pair<int, int> > ev, ans;
	q >> n >> k;
	for (int i = 0; i < n; i++){
		q >> x >> y;
		ev.push_back(mp(x, -1));
		ev.push_back(mp(y, 1));
	}
	z::sort(ev.begin(), ev.end());
	for (int i = 0; i < ev.size(); i++){
		b -= ev[i].second;
		if ((b == k) && (lb == k - 1))l = ev[i].first;
		if ((b == k - 1) && (lb == k)){
			ans.push_back(mp(l, ev[i].first));
		}
		lb = b;
	}
	w << ans.size() << z::endl;
	for (int i = 0; i < ans.size(); i++){
		w << ans[i].first << ' ' << ans[i].second << z::endl;
	}
	return 0;
}
