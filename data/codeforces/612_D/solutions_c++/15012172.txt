#include <bits/stdc++.h>
using namespace std;

#define mp(x, y) make_pair((x), (y))

typedef long long ll;

int n, k;
struct event {
	int x;
	int t;
	bool operator<(const event &e) const {
		return x<e.x || (x==e.x && t<e.t);
	}
} e[2000005];
int cur;
int le;
vector<pair<int,int> > ans;

int main()
{
	scanf("%d%d", &n, &k);
	for(int i=0; i<n; i++) {
		int l, r;
		scanf("%d%d", &l, &r);
		e[2*i].x=l;
		e[2*i].t=-1;
		e[2*i+1].x=r;
		e[2*i+1].t=1;
	}
	sort(e, e+2*n);
	for(int i=0; i<2*n; i++) {
		if(e[i].t==-1) {
			cur++;
			if(cur==k) {
				le=e[i].x;
			}
		} else {
			cur--;
			if(cur==k-1) {
				ans.push_back(mp(le, e[i].x));
			}
		}
	}

	printf("%lu\n", ans.size());
	for(size_t i=0; i<ans.size(); i++) {
		printf("%d %d\n", ans[i].first, ans[i].second);
	}

	return 0;
}
