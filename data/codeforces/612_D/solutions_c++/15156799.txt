#include<bits/stdc++.h>
using namespace std;
struct event
{
	int point;
	int type;
}arr[2000012];
bool comp(event a,event b)
{
	return ((a.point <b.point) ||(a.point==b.point && a.type<b.type));
}
vector<int> ans;
int main()
{
	int n,k,bal=0,t1,t2;
	cin>>n>>k;
	for(int i=0;i<2*n;i+=2)
	{
		scanf("%d%d",&t1,&t2);
		arr[i].point=t1;	arr[i].type=-1;
		arr[i+1].point=t2;	arr[i+1].type=1;
	}
	sort(arr,arr+2*n,comp);
	for(int i=0;i<2*n;i++)
	{
		bal-=arr[i].type;
		if(bal==k && bal+arr[i].type==k-1)
				ans.push_back(arr[i].point);
		else if(bal==k-1 && bal+arr[i].type==k)
			ans.push_back(arr[i].point);
	}
	cout<<ans.size()/2<<endl;
	for(int i=0;i<ans.size();i+=2)
		printf("%d %d\n",ans[i],ans[i+1]);
	return 0;
}
