#include <cstdio>
#include <cstring>
#include <algorithm>
#include <map>
#define MP make_pair
#define F first
#define S second
using namespace std;

typedef pair<int, int> PII;

const int maxn = 1000000 + 10;

PII event[maxn * 2];
int ans[maxn * 2];

int main()
{
	int n, k; scanf("%d%d", &n, &k);
	for(int i = 0; i < n; i++) {
		int l, r; scanf("%d%d", &l, &r);
		event[i<<1] = MP(l, -1);
		event[i<<1|1] = MP(r, 1);
	}
	sort(event, event + n * 2);
	int cnt = 0, num = 0;
	for(int i = 0; i < n * 2; i++) {
		if(event[i].S == -1) {
			num++;
			if(num == k) ans[cnt++] = event[i].F;
		} else {
			if(num == k) ans[cnt++] = event[i].F;
			num--;
		}
	}

	printf("%d\n", cnt / 2);
	for(int i = 0; i < cnt; i += 2)
		printf("%d %d\n", ans[i], ans[i+1]);

	return 0;
}
