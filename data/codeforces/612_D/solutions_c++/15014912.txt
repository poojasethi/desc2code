#include<bits/stdc++.h>
using namespace std;

struct event{
int t,ty;
bool operator <(event a) const{
if(t!=a.t) return t<a.t;
else return ty<a.ty;
}
};event ev[2000005];

int n,k,a,b,re;
int main(){
scanf("%d%d",&n,&k);
for(int i=0;i<n;i++){
    scanf("%d%d",&a,&b);
    ev[2*i]=(event){a,0};
    ev[2*i+1]=(event){b,1};
}
sort(ev,ev+2*n);vector<int> v1,v2;re=0;
for(int i=0;i<2*n;i++){
    if(ev[i].ty==0){
        if(re>=k) re++;
        else{
            re++;
            if(re>=k) v1.push_back(ev[i].t);
        }
    }
    else{
        if(re<k) re--;
        else{
            re--;
            if(re<k) v2.push_back(ev[i].t);
        }
    }
}
printf("%d\n",v1.size());
for(int i=0;i<v1.size();i++){
    printf("%d %d\n",v1[i],v2[i]);
}
return 0;
}
