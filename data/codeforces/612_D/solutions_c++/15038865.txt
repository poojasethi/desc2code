#include <bits/stdc++.h>
using namespace std;

vector<pair<int, int> > ans;
vector<pair<int, int> > val;

int main() {
  //freopen("case.in", "r", stdin);
  int n, k;
  cin >> n >> k;
  for (int i = 1; i <= n; i++) {
    int l, r;
    scanf("%d%d", &l, &r);
    val.push_back(make_pair(l, 1));
    val.push_back(make_pair(r + 1, -1));
  }
  sort(val.begin(), val.end());
  int now = 0, l;
  for (int i = 0; i < 2 * n; i++) {
    if (val[i].second == -1) {
      if (now == k) ans.push_back(make_pair(l, val[i].first - 1));
      now--;
    } else {
      now++;
      if (now == k) l = val[i].first;
    }
  }
  printf("%d\n", ans.size());
  for (int i = 0; i < (int)ans.size(); i++)
    printf("%d %d\n", ans[i].first, ans[i].second);
  return 0;
}
