#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;
typedef pair<int,int> pii;

const int maxn = 1e6+10;
#define mk make_pair
#define X first
#define Y second
vector<pii> v;

vector<int> ret;

int main() {

	int n,k;
	//freopen("in","r",stdin);
	scanf("%d%d",&n,&k);
	for(int i = 0; i < n; i++) {
		int s,e;
		scanf("%d%d",&s,&e);
		v.push_back(mk(s,0));
		v.push_back(mk(e,1));
	}
	sort(v.begin(),v.end());
	int cnt = 0;
	for(int i = 0; i < v.size(); i++) {
		if(v[i].Y) {
			if(cnt == k) {
				ret.push_back(v[i].X);
			}
			cnt--;
		}else{
			cnt++;
			if(cnt==k) {
				ret.push_back(v[i].X);
			}
		}
	}
	cout<<int(ret.size())/2<<endl;
	for(int i = 0; i < ret.size()/2; i++) {
		cout<<ret[i*2]<<" "<<ret[i*2+1]<<endl;
	}
	


	return 0;
}
