#include<bits/stdc++.h>
using namespace std;
vector < pair < int , int > > v;
int main()
{
    ios::sync_with_stdio(0);
    cin.tie(0);
    int n , k , i , j , l , r ;
    cin>>n>>k;
    for(i=0;i<n;++i)
    {
        cin>>l>>r;
        v.push_back(make_pair(l,-1));
        v.push_back(make_pair(r,+1));
    }
    sort(v.begin(),v.end());
    int sum = 0 , cur_l , cur_r ;
    vector < pair < int  , int > > ans;
    for(i=0;i<2*n;++i)
    {
        if(k-1==sum&&v[i].second==-1)
            cur_l = v[i].first;
        else if(sum == k && v[i].second==1 )
            ans.push_back(make_pair(cur_l,v[i].first));
        sum-=v[i].second;
    }
    cout<<ans.size()<<endl;
    for(i=0;i<(int)ans.size();++i)
        cout<<ans[i].first<<" "<<ans[i].second<<endl;
    return 0;
}
