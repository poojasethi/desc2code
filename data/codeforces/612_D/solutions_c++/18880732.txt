#include <bits/stdc++.h>
using namespace std;

#define N 2000006
#define pb push_back
#define mp make_pair
#define pi pair<int, int>
#define F first
#define S second
#define ll long long
vector<pi> v;
vector<pi> res;

int main()
{
	int n, k;
	cin >> n >> k;

	int x, y;
	for(int i = 0; i<n; i++)
	{
		scanf("%d%d", &x, &y);
		v.pb(mp(x, 0));
		v.pb(mp(y, 1));
	}

	sort(v.begin(), v.end());

	int sz = v.size();
	int ct = 0;
	int left , right;
	for(int i = 0; i<sz ; i++)
	{
		// cout << v[i].F << " " << v[i].S << endl;
		if(v[i].S==0)
		{
			if(ct<k)
				left = v[i].F;
			ct++;
		}
		if(v[i].S==1)
		{
			if(ct==k)
			{
				right = v[i].F;
				res.pb(mp(left, right));
			}
			ct--;
		}
	}

	sz = res.size();
	cout << sz << endl;
	for(int i = 0; i<sz; i++)
	{
		printf("%d %d\n", res[i].F, res[i].S);
	} 
}
