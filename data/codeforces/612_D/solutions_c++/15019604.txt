#include <bits/stdc++.h>
using namespace std;
#define M 1000005

int n, k;
struct p { int v, t; } s[2*M], a[M];
bool cmp (p a, p b) { return a.v == b.v ? a.t > b.t : a.v < b.v; }

int main () {
	scanf("%d%d", &n, &k);
	for (int i = 0; i < n; i++) {
		scanf("%d%d", &s[i].v, &s[i+n].v);
		s[i].t = 1;
		s[i+n].t = -1;
	}
	sort(s, s+2*n, cmp);
	int m = 0, h = 0, o = 0;
	for (int i = 0; i < 2*n; i++) {
		m += s[i].t;
		if (m >= k) {
			if (!o) {
				o = 1;
				a[h].v = s[i].v;
			}
		} else if (o) {
			o = 0;
			a[h++].t = s[i].v;
		}
	}
	printf("%d\n", h);
	for (int i = 0; i < h; i++) {
		printf("%d %d\n", a[i].v, a[i].t);
	}
	return 0;
}
