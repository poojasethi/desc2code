// 问哪些段被覆盖了至少k次，这个设一下端点处先进后出的扫描即可。
#include<cstdio>
#include<algorithm>
using namespace std;

int n,m;
int tot,x,y,h,c,last;
pair<int,int> v[2222222];
int ans[1111111];

int main() {
	scanf("%d%d", &n, &m);
	tot=0;
	for(int i=0;i<n;i++) {
		scanf("%d%d", &x, &y);
		v[tot++] = make_pair(x, 0);
		v[tot++] = make_pair(y, 1);
	}
	sort(v,v+tot);
	
	h=c=last=0;
	for(int i=0;i<tot;i++) {
		if(v[i].second==0) h++;
		else h--;
		if(last>=m&&h<m) ans[c++] = v[i].first;
		if(last<m&&h>=m) ans[c++] = v[i].first;
		last=h;
	}
	printf("%d\n", c/2);
	for(int i=0;i*2<c;i++)
		printf("%d %d\n", ans[i*2], ans[i*2+1]);
	
	return 0;
}
