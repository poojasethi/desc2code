#include<bits/stdc++.h>
using namespace std;
int main()
{
   int n,k,i;
    cin>>n>>k;
    int x,y,ans=0,prev=0;
   vector<pair<int,int> > vp;
     vector<int> vp1;
    for(i=0;i<n;i++)
    {
      scanf("%d %d",&x,&y);
      vp.push_back(make_pair(x,-1));
      vp.push_back(make_pair(y,1));
    }
    sort(vp.begin(),vp.end());
      for(i=0;i<2*n;i++)
      {
        if(vp[i].second==-1)
         ans++;
        else
          ans--; 
         if((ans==k-1 && prev==k) || (ans==k  && prev==k-1))
           vp1.push_back(vp[i].first);
        prev=ans;      
      }
      cout<<vp1.size()/2<<endl;
      for(i=0;i<vp1.size();i+=2)
        printf("%d %d\n",vp1[i],vp1[i+1]);
 return 0;
}
