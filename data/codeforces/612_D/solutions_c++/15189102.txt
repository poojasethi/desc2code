#include <bits/stdc++.h>
using namespace std;
pair<int,int> ar[(int)2e6];
vector<pair<int,int> >ret;
int main(){
    int a,b,n,i,k,cc=0,s=0;cin>>n>>k;
    for(i=0;i<n;++i)
        scanf("%d%d",&a,&b),ar[2*i]=make_pair(a,0),
            ar[2*i+1]=make_pair(b,1);
    sort(ar,ar+2*n);
    for(i=0;i<2*n;++i){
        if(ar[i].second==1){
            if( (cc--)==k)
                ret.push_back(make_pair(s,ar[i].first));
        }
        else{
            ++cc;
            if(cc==k)
            s=ar[i].first;
        }
    }
    printf("%d\n",(int)ret.size());
    for(i=0;i<ret.size();++i)
        printf("%d %d\n",ret[i].first,ret[i].second);
}
