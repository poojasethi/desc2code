#include <bits/stdc++.h>
using namespace std;

typedef pair<int,int> pr;


int main() {
  int i,n,k,o,t1,t2;

  scanf("%d %d",&n,&k);

  if (n==0) {cout<<"0"; return 0;}
  vector<pr> s;
  vector<int> r;

  for (i=0;i<n;++i) {
    scanf("%d %d",&t1,&t2);
    s.push_back(pr(t1,0));
    s.push_back(pr(t2,1));
  }
  sort(s.begin(),s.end());

  o=0;
  for (i=0;i<s.size();++i) {
    if (s[i].second==0) {
      o++;
      if (o==k) r.push_back(s[i].first);
    } else {
      if (o==k) r.push_back(s[i].first);
      o--;
    }
  }

  cout<<r.size()/2<<endl;

  for (i=0;i<r.size();i+=2) cout<< r[i]<<" "<<r[i+1]<<endl;
  return 0;
}
