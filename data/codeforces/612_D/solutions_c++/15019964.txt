#include <bits/stdc++.h>

using namespace std;

#define X first
#define T second

pair<int,int> p[2001000];
vector < pair<int,int> >ans;
int n  , K , a , b , all=0 ;


int main()
{
scanf("%d %d",&n,&K);
while(n--){
scanf("%d %d",&a,&b);
p[all].X=a;
p[all].T=-1;
all++;
//-------
p[all].X=b;
p[all].T=1;
all++;
}




sort(p,p+all);

int c=0;
int from , to;
for(int i=0;i<all;i++){
 c-=p[i].T;

 if( p[i].T==-1 && c==K ){from=p[i].X;}

 if( p[i].T==1&&c==K-1){to=p[i].X;
 ans.push_back(make_pair(from,to));
 }




}


cout<<ans.size()<<endl;
for(unsigned int i=0;i<ans.size();i++)
printf("%d %d\n",ans[i].first,ans[i].second);
    return 0;
}
