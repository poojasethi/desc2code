#include <bits/stdc++.h>
using namespace std;
int n,k,s,e,cnt=0;
typedef pair<int,int> ii;
vector <ii> v;
vector <int> a;
int main(){
	scanf("%d %d",&n,&k);
	for(int x=0;x<n;x++){
		scanf("%d %d",&s,&e);
		v.push_back(ii(s,0));
		v.push_back(ii(e,1));
	}
	sort(v.begin(),v.end());
	for(int x=0;x<v.size();x++){
		if(v[x].second){
			if(cnt==k) a.push_back(v[x].first);
			cnt--;
		}
		else{
			cnt++;
			if(cnt==k) a.push_back(v[x].first);
		}
	}
	printf("%d\n",a.size()/2);
	for(int x=0;x<a.size()/2;x++) printf("%d %d\n",a[x*2],a[x*2+1]);
	return 0;
}
