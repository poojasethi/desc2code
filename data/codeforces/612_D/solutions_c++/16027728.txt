#include <bits/stdc++.h>
using namespace std;
#define M 2001001
typedef pair<int, int> pii;

int n, m, k;
pii p[M];
int A[M], B[M];

int main() {
	//freopen("Edu4_D2.in", "r", stdin);
	scanf("%d %d", &m, &k);
	for (int x, y; m--; ) {
		scanf("%d %d", &x, &y);
		p[++n] = pii(x, 0);
		p[++n] = pii(y, 1);
	}
	sort(p + 1, p + n + 1);

	m = 0;
	int cross = 0;
	for (int i = 1; i <= n; i++) {
		if (p[i].second == 0) {
			cross++;
			if (cross == k) A[++m] = p[i].first;
		} else {
			if (cross == k) B[m] = p[i].first;
			cross--;
		}
	}

	printf("%d\n", m);
	for (int i = 1; i <= m; i++) printf("%d %d\n", A[i], B[i]);
	return 0;
}
