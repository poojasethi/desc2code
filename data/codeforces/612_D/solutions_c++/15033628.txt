#include <bits/stdc++.h>
using namespace std;
#define ll long long int
#define pb push_back
typedef pair<int,int> ii;
int main(){
	vector<ii> v;
	vector<int> ans;
	int n,k,s,e,curr=0;
	scanf("%d %d",&n,&k);
	for( int i=0; i<n;i++){
		scanf("%d %d",&s,&e);
		v.pb(ii(s,0));
		v.pb(ii(e,1));
	}
	sort(v.begin(),v.end());
	for( int i=0; i<v.size();i++){
		if(v[i].second){
			if(curr==k)ans.pb(v[i].first);
			curr--;
		}
		else{
			curr++;
			if(curr==k)ans.pb(v[i].first);
		}
	}
	printf("%d\n",ans.size()/2);
	for( int i=0; i<ans.size()/2;i++){
		printf("%d %d\n",ans[i*2],ans[i*2+1] );
	}
}
