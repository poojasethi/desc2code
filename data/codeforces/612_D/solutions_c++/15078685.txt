#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#define F first
#define S second
using namespace std;

int n,k,c;
vector<pair<int, int> > a;
vector<int> b;
int main(){
    int i,x,y;
    cin>>n>>k;
    for(i=0; i<n; i++) {
        scanf("%d%d",&x,&y);
        a.push_back(make_pair(x,0));
        a.push_back(make_pair(y,1));
    }
    sort(a.begin(),a.end());
    for(i=0; i<2*n; i++) {
        if(!a[i].S){
            c++;
            if(c==k) b.push_back(a[i].F);
        } else {
            c--;
            if(c==k-1) b.push_back(a[i].F);
        }
    }
    cout<<b.size()/2<<endl;
    for(i=0; i<b.size(); i+=2){
        printf("%d %d\n",b[i],b[i+1]);
    }
    return 0;
}
