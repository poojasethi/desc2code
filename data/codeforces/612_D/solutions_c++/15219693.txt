#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#define F first
#define S second
using namespace std;

int n, k, s;
pair<int, int> a[2000005];
vector<int> ans;
int main()
{
	int i, t;
//	freopen ("input.txt", "r", stdin);
	cin >> n >> k;
	for (i = 0; i < n * 2; i++) {
		scanf ("%d", &t);
		a[i].F = t;
		a[i].S = i % 2;
	}
	sort(a, a + n * 2);
	for (i = 0; i < n * 2; i++) {
		if (a[i].S == 0) {
			s++;
			if (s == k) ans.push_back(a[i].F);
		} else {
			if (s == k) ans.push_back(a[i].F);
			s--;
		}
	}
    cout << ans.size() / 2 << endl;
    for (i = 0; i < ans.size(); i+=2) {
		printf("%d %d\n", ans[i], ans[i+1]);
    }
    return 0;
}
