#include <cstdio>
#include <algorithm>
#include <vector>
#include <utility>

std::pair<int, bool>c[2000020];

int main()
{
    int n, i, r, k, x, y, m;
    scanf("%d %d", &n, &k);
    for (i=0; i<n; i++) {
        scanf ("%d %d", &x, &y);
        c[2*i]=std::make_pair(x, false);
        c[2*i+1]=std::make_pair(y, true);
    }
    m=2*n;
    std::sort(c, c+m);
    int f, a=0;
    std::vector <std::pair<int, int> > z;
    for (i=0; i<m; i++) {
        if (c[i].second) {
            if (a==k) z.push_back (std::make_pair(f, c[i].first));
            a--;
        } else {
            a++;
            if (a==k) f=c[i].first;
        }
    }
    r=z.size();
    printf ("%d\n", r);
    for (i=0; i<r; i++) printf ("%d %d\n", z[i].first, z[i].second);
    return 0;
}
