#include <bits/stdc++.h>
#define FOR(i,a,b) for (int i=a;i<=b;++i)
#define FORD(i,a,b) for (int i=a;i>=b;--i)
#define fi first
#define se second
using namespace std;
typedef long long ll;
typedef pair<int,int> ii;
const int N = 2000010;
int n, k, m;
ii a[N], b[N];

int main()
{
    scanf("%d%d",&n,&k);
    for (int i=1,x,y;i<=n;++i) scanf("%d%d",&x,&y), a[i] = ii(x,-1), a[i+n] = ii(y,1);
    n *= 2;
    sort(a+1,a+1+n);
    int f = 0, t, l, r;
    FOR(i,1,n) {
        t = f;
        f -= a[i].se;
        if (t == k-1 && f == k) {
            l = a[i].fi;
        }
        if (t == k && f == k-1) {
            r = a[i].fi;
            b[++m] = ii(l,r);
        }
    }
    printf("%d\n",m);
    FOR(i,1,m) printf("%d %d\n",b[i].fi,b[i].se);
    return 0;
}
