#include <bits/stdc++.h>
#define fore(i,a,b) for(int i=a,to=b;i<to;i++)
#define pb push_back
#define mp make_pair
#define fst first
#define snd second

using namespace std;

typedef long long ll;
typedef pair<int,int> ii;
int const N = 2000000;
multiset<ii> s;
ii seg[N];
int main(){
		int n,k,cur = 0, h = 0;
		scanf("%d%d",&n,&k);
		fore(i,0,n) {
			int u,v;
			scanf("%d%d",&u,&v);
			s.insert(ii(u,0));
			s.insert(ii(v,1));
		}
		for(set<ii>::iterator it = s.begin(); it != s.end(); it++) {
			ii p = *it;
			if(p.snd == 0) {
				cur++;
				if(cur == k) {
					seg[h].fst = p.fst;
				}
			} else {
				cur--;
				if(cur == k-1)
					seg[h++].snd = p.fst;
			}
		}
		printf("%d\n",h);
		fore(i,0,h) printf("%d %d\n",seg[i].fst,seg[i].snd);
}
