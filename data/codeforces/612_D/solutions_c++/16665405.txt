#include <bits/stdc++.h>
using namespace std;
typedef pair <int , int> P;
vector <P> G;
vector <int> ans;
int main()
{
	int k , n , u , v;
	scanf("%d %d" , &k , &n);
	for(int i = 1 ; i <= k ; i++) {
		scanf("%d %d" , &u , &v);
		G.push_back(P(u , 1));
		G.push_back(P(v + 1 , -1));
	}
	sort(G.begin() , G.end());
	int sum = 0 , temp = 0;
	for(int i = 0 ; i < G.size() ; i++) {
		sum += G[i].second;
		if(sum == n && !temp) {
			ans.push_back(G[i].first);
			temp = 1;
		}
		if(sum < n && temp) {
			temp = 0;
			ans.push_back(G[i].first - 1);
		}
	}
	printf("%d\n" , ans.size() / 2);
	for(int i = 0 ; i < ans.size() ; i += 2) {
		printf("%d %d\n" , ans[i] , ans[i + 1]);
	}
}
