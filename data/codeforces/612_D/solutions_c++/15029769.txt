#include <bits/stdc++.h>

using namespace std;

#define ll long long int

int main()
{
	// ios::sync_with_stdio(false);
	ll n,i,k,a,b,cur=0;
	scanf("%lld%lld",&n,&k);
	vector< pair<ll,ll> > A,V;
	for(i=0;i<n;i++)
	{
		scanf("%lld%lld",&a,&b);
		A.push_back(make_pair(a,-1));
		A.push_back(make_pair(b,1));
	}
	sort(A.begin(),A.end());
	n=A.size();
	cur=0;
	for(i=0;i<n;i++)
	{	
		if(cur==k-1 && A[i].second==-1)
		{
			a=A[i].first;
			cur-=A[i].second;
		}
		else if(cur==k && A[i].second==1)
		{
			V.push_back(make_pair(a,A[i].first));
			cur=k-1;
		}
		else cur=max(0LL,cur-A[i].second);
	}
	n=V.size();
	printf("%lld\n",n);
	for(i=0;i<n;i++)
		printf("%lld %lld\n",V[i].first,V[i].second); 
	return 0;
}
