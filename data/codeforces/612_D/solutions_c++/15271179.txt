#include<bits/stdc++.h>
using namespace std;
const int N=1e6+5;
struct dot{
    int x;
    int tp;
    bool operator<(const dot &b)const
    {
        return x<b.x||x==b.x&&tp>b.tp;
    }
}a[2*N];
int l[2*N],r[2*N];
int n,k,i,x,cnt,tot;
int main()
{
    scanf("%d%d",&n,&k);
    for (i=1;i<=n;i++)
    {
        scanf("%d",&x); a[i*2-1].x=x;a[i*2-1].tp=1;
        scanf("%d",&x); a[i*2].x=x;a[i*2].tp=-1;
    }
    sort(a+1,a+n*2+1);
    for (i=1;i<=2*n;i++)
    {
        cnt+=a[i].tp;
        if (a[i].tp==1&&cnt==k) l[++tot]=a[i].x;
        if (a[i].tp==-1&&cnt==k-1) r[tot]=a[i].x;
    }
    printf("%d\n",tot);
    for (i=1;i<=tot;i++) printf("%d %d\n",l[i],r[i]);
}
