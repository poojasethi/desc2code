#include <stdio.h>
#include <string.h>
#include <algorithm>
using namespace std;
int l[1000010],r[1000010],a[2000010];
int main()
{
	int n,k;
	scanf("%d %d",&n,&k);
	for (int i=0;i<n;i++)
	{
		scanf("%d %d",&l[i],&r[i]);
	}
	sort(l,l+n);
	sort(r,r+n);
	int f=0,l0=0,r0=0;
	int t=0,ans=0;
	
	while (f!=2*n)
	{
		f++;
		if (l[l0]<=r[r0] && l0!=n)
		{
			t++;
			if (t==k)
			{
				a[2*ans]=l[l0];
				ans++;
			
			}
			l0++;
		}
		else
		{
			if (t==k)
			{
				a[2*ans-1]=r[r0];
				
			}
			t--;
			r0++;
		}
		
	}
	
	
	printf("%d\n",ans);
	for (int i=0;i<ans;i++)
	{
		printf("%d %d\n",a[2*i],a[2*i+1]);
	}
	
}
