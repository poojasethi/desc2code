#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<vector>
using namespace std;

pair<int,int> p[2000010];
vector<pair<int,int> > d;
int main()
{
    int n,k,x,y;
    cin>>n>>k;
    int tot=0;
    for(int i=1;i<=n;i++){
        scanf("%d%d",&x,&y);
        p[tot++]={x,-1};
        p[tot++]={y,1};
    }
    sort(p,p+tot);
    int cur=0,L;
    for(int i=0;i<tot;i++){
        if(cur==k-1&&p[i].second==-1){
            L=p[i].first;
        }
        if(cur==k&&p[i].second==1){
            d.push_back(make_pair(L,p[i].first));
        }
        cur-=p[i].second;
    }
    cout<<d.size()<<endl;
    for(int i=0;i<d.size();i++)
        printf("%d %d\n",d[i].first,d[i].second);
    return 0;
}
