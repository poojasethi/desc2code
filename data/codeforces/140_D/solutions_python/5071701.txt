from bisect import bisect_right
n, t = int(raw_input()), sorted(list(map(int, raw_input().split())))
if t[0] > 710: print 0, 0
else:
    for i in range(1, n):
        t[i] += t[i - 1]
        if t[i] > 710: t = t[: i]; break
    i, j = bisect_right(t, 350), len(t)
    print j, sum(t[i: ]) - 350 * (j - i)
