n = 3
problems = [30, 330, 720]

problems = list(sorted(problems))

solved = 0
penalty = 0

LIM = 60 * 6
LIM2 = 60 * 12

time = 10
for problem in problems:
  if time + problem <= LIM2:
    solved += 1
    time += problem
    if time >= LIM:
      penalty += time - LIM

print solved, penalty
