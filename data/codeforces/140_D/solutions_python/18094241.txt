def f(n, data):
    num_solved = 0
    current_time = 10
    penalty = 0
    data.sort()
    for i in data:
        current_time += i
        if current_time > 720:
            return num_solved, penalty
        num_solved += 1
        if current_time > 360:
            penalty += current_time - 360
    return num_solved, penalty


n = raw_input()
data = map(int, raw_input().split())
tup = f(n, data)
print tup[0], tup[1]
