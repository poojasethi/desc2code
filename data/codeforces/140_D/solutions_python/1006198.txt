n = input()
a = map(int, raw_input().split())
a.sort()
T = 710
T2 = 350
sum = 0
for i, x in enumerate(a):
	sum += x
	if sum <= T: continue
	a = a[:i]
	break
n = len(a)
if n == 0:
	print "0 0"
	exit(0)
res = 1 << 30
for i in xrange(n):
	t = -T2
	sum = 0
	for j in xrange(i, n):
		t += a[j]
		if t < 0: continue
		sum += t
		break
	if t < 0: t = 0
	for k in xrange(i):
		t += a[k]
		sum += t
	for k in xrange(j + 1, n):
		t += a[k]
		sum += t
	res = min(res, sum)
print n, res
