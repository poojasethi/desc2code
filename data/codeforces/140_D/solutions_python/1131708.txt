import sys


n = int(sys.stdin.readline())
ints = [int(s) for s in sys.stdin.readline().split(' ')]

ints.sort()
contest_length = 12 * 60
time_spent = 10
points = 0
penalty = 0

for pr in ints:
    time_spent += pr
    if time_spent <= contest_length:
        points += 1
        penalty += max(0, time_spent - 6 * 60)
    else:
        break

print points, penalty
