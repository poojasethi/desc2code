#include <algorithm>
#include <stdio.h>

int a[111];
int n;

int main() {
	int i, s = 10, t = 0;
	scanf("%d", &n);
	for (i = 0; i < n; ++i) scanf("%d", a + i);

	std::sort(a, a + n);
	for (i = 0; i < n; ++i) {
		s += a[i];
		if (s > 720) break;
		if (s > 360) t += s - 360;
	}

	printf("%d %d", i, t);
	return 0;
}
