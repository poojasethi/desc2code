#include<iostream>
#include<algorithm>
using namespace std;
int t=-350;
int n,a[100500],s,d;
int main(){
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    for(int i=0;i<n;i++){
        t+=a[i];
       
        if(t>360)break; d++;
        if(t>0)s+=t;
        }
    cout<<d<< " "<<s;
    }
