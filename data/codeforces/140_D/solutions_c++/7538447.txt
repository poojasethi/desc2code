#include<cstdio>
#include<algorithm>
using namespace std;

int n, a[110];
int main() {
	scanf("%d", &n);
	for (int i=0; i<n; i++) scanf("%d", &a[i]);
	sort(a, a+n);
	int jml=0, j=0, sum=0;
	while (jml+a[j]<=710 && j<n) {
		jml += a[j];
		if (jml>350) sum += abs(jml-350); 
		j++;
	}
	printf("%d %d\n", j, sum);
}
