#include <iostream>
#include <algorithm>

using namespace std;

int N,A[110],cnt=10,penalty,problem;

int main()
{
	cin >> N;
	for (int i=0;i<N;i++)
		cin >> A[i];
	sort(A,A+N);
	for (int i=0;i<N;i++)
	{
		cnt+=A[i];
		if (cnt>720)
			break;
		if (cnt>360)
			penalty+=cnt-360;
		problem++;
	}
	cout << problem << " " << penalty << endl;
	return 0;
}
