#include <stdio.h>
#include <string.h>
#include <algorithm>
#include <iostream>

using namespace std;

int n,a[110];

int main()
{
	scanf("%d",&n);
	for(int i=0;i<n;i++)
		scanf("%d",a+i);
	sort(a,a+n);
	int ret=0,p=0,t=10;
	for(int i=0;i<n;i++)
	{
		if(t+a[i]<=360) ret++,t+=a[i];
		else if(t+a[i]<=720) ret++,t+=a[i],p+=t-360;
	}
	cout<<ret<<" "<<p<<endl;
	return 0;
}
