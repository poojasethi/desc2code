#include<iostream>
#include<algorithm>
using namespace std;
int a[101];
int aa,bb;
int time=10;
int main()
{
int n;
cin>>n;
for(int i=1;i<=n;++i)
cin>>a[i];
sort(a+1,a+n+1);
for(int i=1;i<=n;++i)
{
if(time+a[i]<=720)
{
time+=a[i],aa++;
if(time>360)
bb+=(time-360);
}
else break;
}
cout<<aa<<' '<<bb;
}
