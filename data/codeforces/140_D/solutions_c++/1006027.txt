#include <iostream>
#include <algorithm>
using namespace std;

int n;
const int nmax = 100;
int a[nmax];

int main() {
	int i, t = 10, T = 720, o = 0, p = 0;

	cin >> n;
	for (i = 0; i <= n; i++) cin >> a[i];

	sort(a, a + n);
	for (i = 0; i < n; i++) {
		t += a[i];
		if (t <= T) {
			o++;
			p += max(t - T / 2, 0);
		}
	}

	cout << o << ' ' << p;
	
	return 0;
}
