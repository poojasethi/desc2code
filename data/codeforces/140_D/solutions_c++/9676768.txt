#include<cstdio>
#include<algorithm>
using namespace std;
int t[101];
int main(){
    int n,l=0,sum=10,ans=0;scanf("%d",&n);
    for(int i=0;i<n;i++)scanf("%d",&t[i]);
    sort(t,t+n);
    for(int i=0;i<n&&sum+t[i]<=720;i++,l++,ans+=max(0,sum-360))sum+=t[i];
    printf("%d %d",l,ans);
    return 0;
}
