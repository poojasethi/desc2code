#include<cstdio>
#include<iostream>
#include<algorithm>

using namespace std;

int t,p,n,prob[1000],ans,sl,i;

int main()
{
	t=710; p=360;
	scanf("%d", &n);
	for (i=0; i<n; i++) scanf("%d", &prob[i]);
	sort(prob,prob+n);
	for (i=0; i<n; i++)
	{
		if (t-prob[i]<0) break;
		t-=prob[i]; ++sl;
		if (t<p) ans+=p-t;
	}
	printf("%d %d\n", sl, ans);
}
