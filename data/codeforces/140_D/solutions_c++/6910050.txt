#include<stdio.h>
#include<algorithm>
using namespace std;
int main()
{
	int n,i,j,num,sum,a[105];
	while(scanf("%d",&n) != EOF)
	{
		for(i=0;i<n;i++)
			scanf("%d",&a[i]);
		sort(a,a+n);
		sum=10;num=0;
		for(i=0,num=0;i<n;i++)
		{
			if(sum+a[i]>720)
				break;
			sum+=a[i];
			if(sum>360)
				num+=sum-360;
		}
		printf("%d %d\n",i,num);
	}
	return 0;
}
