#include <cstdio>
#include <vector>
#include <algorithm>
using namespace std;

int n, a[105];

int main() {
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", a+i);
	}

	sort(a, a+n);
	int sum = 10, m = 0, p = 0;
	for (int i = 0; i < n && sum + a[i] <= 720; m=++i) {
		sum += a[i];
		p += max(0, sum - 360);
	}
	printf("%d %d\n", m, p);
}
