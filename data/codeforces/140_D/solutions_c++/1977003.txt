#include <iostream>
#include <algorithm>

using namespace std;

int main() {
  int n;
  cin >> n;
  int a[n];

  for (int i = 0; i < n; i++)
    cin >> a[i];
  
  sort(a, a + n);
  
  int x = 0, y = 0;
  for (int i = 0, nt = -350; i < n && nt + a[i] <= 360; i++) {
    x++;
    nt += a[i];
    y += max(0, nt);
  }
  
  cout << x << " " << y;
  return 0;
}
