#include <stdio.h>
#include <algorithm>
using namespace std;
int n,t=710,p,r;
int a[105];
int main(){
    int i;
    scanf("%d",&n);
    for(i=1;i<=n;i++)scanf("%d",&a[i]);
    a[++n]=9999999;
    sort(a+1,a+n+1);
    for(i=1;i<=n;i++){
        r+=a[i];
        if(r>t)break;
        if(r>350)p+=(r-350);
    }printf("%d %d",i-1,p);
}
