#include<cstdio>
#include<algorithm>
int main(){
	int a[111],n;
	scanf("%d",&n);
	for(int i=0;i<n;i++)	scanf("%d",a+i);
	std::sort(a,a+n);
	int cnt=0,time=10,penalty=0;
	for(int i=0;i<n;i++){
		time+=a[i];
		if(time>720)	break;
		cnt++;
		if(time>360)	penalty+=(time-360);
	}
	printf("%d %d\n",cnt,penalty);
	return 0;
}
