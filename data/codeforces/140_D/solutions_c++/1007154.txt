#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using  namespace std;
const int N = 305;
int a[N];
int main()
{
	int n,i;
	scanf("%d",&n);
	for(i=0;i<n;i++)scanf("%d",a+i);
	sort(a,a+n);
	int t=10,r=0;
	for(i=0;i<n;i++)
	{
		t+=a[i];
		if(t>720)break;
		if(t>360)r+=t-360;
	}
	printf("%d %d\n",i,r);
	return 0;
}
