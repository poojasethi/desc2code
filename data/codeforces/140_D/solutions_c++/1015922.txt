#include<iostream>
#include<algorithm>
using namespace std;
int main(){
    int a[111],n,sum=10,p=0,pen=0;
    cin>>n;for(int i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    while(sum<=360)
        sum+=a[p++];
    sum-=360;
    while(sum<=360){
        pen+=sum;
        sum+=a[p++];
    }
    cout<<p-1<<" "<<pen<<endl;
}
