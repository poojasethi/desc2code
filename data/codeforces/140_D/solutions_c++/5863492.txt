#include<cstdio>
#include<algorithm>
using namespace std;

int n;
int v[111];
int cur,ans,tot;

int main() {
	scanf("%d", &n);
	for(int i=0;i<n;i++)
		scanf("%d", &v[i]);
	sort(v,v+n);
	cur=10;ans=tot=0;
	for(int i=0;i<n;i++){
		cur+=v[i];
		if(cur>720)break;
		tot++;
		ans+=max(0,cur-360);
	}
	printf("%d %d\n", tot,ans);

	return 0;
}
