#include <cstdio>
#include <algorithm>
using namespace std;

int N,A[101],t=10,s,i;
int main () {
    scanf("%d",&N);
    for (;i<N;++i) scanf("%d",&A[i]);
    sort(A,A+N);
    for (i=0;i<N;++i) {
        t+=A[i];
        if (t>720) break;
        if (t>360) s+=t-360;
    }
    printf("%d %d",i,s);
    return 0;
}
