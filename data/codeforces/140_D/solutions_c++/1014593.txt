#include <iostream>
#include <algorithm>
using namespace std;
int a[110], n, ans, cnt;
int main()
{
  cin >> n;
  for(int i = 0; i < n; i++)
    cin >> a[i];
  sort(a, a + n);
  int s = 10;
  for(int i = 0; i < n; i++)
  {
    s += a[i];
    if (s <= 360) cnt++;
    else if (s <= 720) cnt++, ans += s - 360;
  }
  cout << cnt << " " << ans;
  return 0;
}
