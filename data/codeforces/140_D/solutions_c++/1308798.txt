#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
 int n,i,t=10,p=0,s=0,a[100],f=0;
 cin>>n;
 for(i=0;i<n;++i)
  cin>>a[i];
 sort(a,a+n);
 for(i=0;i<n;++i)
 {
  t+=a[i];
  if(t>720)
   break;
  else if(t>360)
   p+=t-360;
  if(t<=720)
   ++s; 
 }
 cout<<s<<' '<<p;
 return 0; 
}
