#include <bits/stdc++.h>
using namespace std;
int arr[105];
int main(){
    int n;
    cin>>n;
    for(int i=0;i<n;i++)cin>>arr[i];
    sort(arr,arr+n);

    int mn=10;
    int i=0;
    int cnt=0,pen=0;
    while(i<n && mn+arr[i]<=720){
         mn+=arr[i];
         cnt++;
         i++;
         pen+=max(mn-360,0);
    }

    cout<<cnt<<" "<<pen<<"\n";
}
