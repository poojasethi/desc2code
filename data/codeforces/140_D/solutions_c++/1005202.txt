#include <stdio.h>
#include <algorithm>

int a[100];

int main()
{
	int n,i,ti=10,tot=0,cnt=0;
	scanf("%d",&n);
	for (i=0;i<n;i++)
		scanf("%d",&a[i]);
	std::sort(a,a+n);
	for (i=0;i<n&&ti<720;i++)
	{
		ti+=a[i];
		if (ti<=720&&ti>360)tot+=(ti-360);
		if (ti<=720)cnt++;
	}
	printf("%d %d\n",cnt,tot);
}
