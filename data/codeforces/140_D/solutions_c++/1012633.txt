#include<iostream>
#include<algorithm>
using namespace std;
int main()
{   
    int n,a[111],i;
    cin>>n;
    for(i=0;i<n;i++)cin>>a[i];
    sort(a,a+n);
    int solv=0,num=710,p=0;
    for(i=0;i<n;i++)
    {
        if(num-a[i]<0)break;
        num-=a[i];
        solv++;
        if(num<360)p+=710-num-350;
    }
    cout<<solv<<" "<<p;
}
