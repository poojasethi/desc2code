#include<iostream>
#include<algorithm>
using namespace std;
long a[10004];
int main(){
long n;
cin>>n;
for(long i=1;i<=n;i++)cin>>a[i];sort(a+1,a+n+1);
long s=10;long l=0,p=0;
for(long i=1;i<=n;i++){
    s=s+a[i];
    if(s<=360){l++;}
    else{
        if(s<=720){l++;p=p+s-360;}
    }
}
cout<<l<<" "<<p<<endl;


}
