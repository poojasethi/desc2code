#include<cstdio>
#include<cstring>
#include<string>
#include<iostream>
using namespace std;
const int N=10000;
int next[N][26];
int main()
{
	int i,j,n,p,sn,tn,ans;
	string s,t;
	cin>>s>>t;
	sn=s.size();
	tn=t.size();
	memset(next,-1,sizeof(next));
	for(i=0;i<sn;i++)
	{
		j=i;
		do
		{
			j=(j+sn-1)%sn;
			next[j][s[i]-'a']=i;
		}while(s[j]!=s[i]);
	}
	p=sn-1;
	ans=0;
	for(i=0;i<tn;i++)
	{
		n=next[p][t[i]-'a'];
		if(n==-1)
		{
			printf("-1\n");
			return 0;
		}
		if(n<=p)ans++;
		p=n;
	}
	printf("%d\n",ans);
	return 0;
}
