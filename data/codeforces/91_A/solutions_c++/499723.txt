#include<iostream>
#include<cstring>
using namespace std;
const int MN=110000;
int a[MN][26],mark[MN];
int main(){
	string s,p;
	cin>>s>>p;
	memset(a,-1,sizeof(a));
	for(int i=s.size()-1;i>=0;i--){
		for(int j=0;j<26;j++){
			a[i][j]=a[i+1][j];
		}
		a[i][s[i]-'a']=i;
		mark[s[i]-'a']=1;
	}
	int state=0;
	int con=1;
	for(int i=0;i<p.size();i++){
		if(!mark[p[i]-'a']){
			cout<<"-1"<<endl;
			return 0;
		}
		if(a[state][p[i]-'a']==-1){
			con++;
			state=0;
		}
		state=a[state][p[i]-'a']+1;
	}
	cout<<con<<endl;
}
