#include<cstdio>
#include<cstring>

char from[11111], to[1111111];
int nxt[11111][27], cur[55];
int len, p, ans;

int main(){
	scanf("%s%s", from, to);
	
	memset(cur, 0, sizeof cur);
	len = strlen(from);
	for(int i=len-1;i>=0;i--) {
		for(int j=0;j<26;j++)
			nxt[i+1][j] = cur[j];
		cur[from[i] - 'a'] = i+1;
	}
	for(int i=0;i<26;i++)
		nxt[0][i] = cur[i];
	
	p = 0;ans=1;
	for(int i=0;to[i];i++){
		int id = to[i] - 'a';
		if (!nxt[p][id]){
			p=0;ans++;
		}
		p = nxt[p][id];
		if (!p){
			printf("-1\n");
			return 0;
		}
	}
	printf("%d\n", ans);

	return 0;
}
