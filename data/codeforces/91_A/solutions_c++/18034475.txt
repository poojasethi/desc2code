#include<bits/stdc++.h>
using namespace std;
int main()
{
	string a,b;
	cin>>a>>b;
	vector <int> v[100];
	int pos=-1,ans=1;
	for(int i=0;i<a.size();i++)
	v[a[i]-'a'].push_back(i);
	for(int i=0;i<b.size();i++)
	{
		if(v[b[i]-'a'].size()==0)
		{cout<<-1;return 0;}
	}
	for(int i=0;i<b.size();i++)
	{
		if(upper_bound(v[b[i]-'a'].begin(),v[b[i]-'a'].end(),pos)==v[b[i]-'a'].end())
		{
			ans++;
			pos=-1;
		}
		pos=*(upper_bound(v[b[i]-'a'].begin(),v[b[i]-'a'].end(),pos));
	}
	cout<<ans;
	return 0;
}
