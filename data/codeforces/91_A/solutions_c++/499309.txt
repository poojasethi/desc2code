#include <stdio.h>
#include <string.h>

char dat1[10003],dat2[1000003];
int lp[10003][26];
int main(){
	scanf("%s%s",dat1,dat2);
	memset(lp,-1,sizeof(lp));
	for(int i = strlen(dat1) - 1; i >= 0;i --){
		for(int j = 0;j < 26;j++){
			lp[i][j] = lp[i+1][j];
		}
		lp[i][dat1[i]-'a'] = i;
	}
	int ans = 1;
	int j = 0;
	for(int i = 0; dat2[i]; i++) {
		j = lp[j][dat2[i]-'a'];
		if(j == -1) {
			j = 0;
			j = lp[j][dat2[i]-'a'];
			ans ++;
			if(j == -1) {
				ans = -1;
				break;
			}
		}
		j ++;
	}
	printf("%d\n",ans);
	return 0;
}
