#include <bits/stdc++.h>
#define sz(a) ((int)(a).size())

using namespace std;

set<int>indices[200];
int main(){
    string s,d;
    cin>>s>>d;

    for(int i=0;i<sz(s);i++){
        indices[s[i]].insert(i);
    }


    set<int>::iterator it;
    int idx=0;
    int res=1;
    for(int i=0;i<sz(d);i++){


        if(indices[d[i]].empty()){cout<<"-1\n"; return 0;}

        it=indices[d[i]].lower_bound(idx);
        if(it==indices[d[i]].end()){
            idx=*(indices[d[i]].begin())+1;
            res++;
        }
        else idx= (*it)+1;

       // cout<<idx-1<<"***"<<endl;

    }

    cout<<res<<"\n";
}
