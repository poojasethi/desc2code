#include<bits/stdc++.h>
using namespace std;
const int N=1e6+9,M=1e4+9,K=26;
set<int > s[K];
char s1[M],s2[N];
int p=-1,ans=1,i;
int main()
{
	ios::sync_with_stdio(0);cin.tie(0);
	cin>> s1 >> s2;
	for( i=0 ; s1[i] ; i++ )
		s[s1[i]-'a'].insert(i);
	for( i=0 ; s2[i] ; i++)
	{
		if( s[s2[i]-'a'].empty() )
			{
				cout<<-1;
				return 0;
			}
		if(s[s2[i]-'a'].upper_bound(p)==s[s2[i]-'a'].end())
             p = -1, ans++;
        p = *s[s2[i]-'a'].upper_bound(p);
	}
	cout << ans;
	return 0;
}
