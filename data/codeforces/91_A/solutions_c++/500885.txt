#include <iostream>
#include <string>
#include <map>
using namespace std;

int main(){
	string s1, s2;
	cin>>s1>>s2;
	int count=0;
	bool can=true;
	map<int, int> pos[26];
	for (int i=0; i<26; i++){
		for (int j=0; j<s1.length(); j++)
			if (s1[j]=='a'+i){
				pos[i][j]=0;
			}
	}
	int prev=s2.length();
	for (int i=0; i<s2.length(); i++){
		if (pos[s2[i]-'a'].size()==0){
			can=false;	
			break;
		}		
		map<int, int>::iterator iter=pos[s2[i]-'a'].upper_bound(prev);
		if (iter==pos[s2[i]-'a'].end()){
			count++;
			iter=pos[s2[i]-'a'].begin();
		}
		prev=iter->first;
	}
	if (can)
		cout<<count;
	else 
		cout<<-1;
}
