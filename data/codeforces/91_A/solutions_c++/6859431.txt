#include <cstdio>
#include <cstdlib>
char s1[10020], s2[1002000];
int next[10020][28];
int i, j, k = 1, n;
int main(){
  scanf("%s\n%s", s1 + 1, s2 + 1);
  for (n = 1; s1[n]; n++); n--;
  for (j = 0; j < 26; j++){
    next[n][j] = n + 1;
    for (i = n - 1; i >= 0; i--){
      if (s1[i + 1] == j + 'a') next[i][j] = i + 1;
      else next[i][j] = next[i + 1][j];
    }
  }
  j = 0;
  for (i = 1; s2[i]; i++){
    if (next[j][s2[i] - 'a'] == 0){printf("-1"); return 0;}
    j = next[j][s2[i] - 'a'];
    if (j == n + 1) j = next[0][s2[i] - 'a'], k++;
  }
  printf("%d", k);
  return 0;
}
