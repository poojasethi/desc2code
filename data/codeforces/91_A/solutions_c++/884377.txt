#include <iostream>
#include <map>
#include <set>
#include <string>

using namespace std;

int main()
{
  string s1,s2;
  cin>>s1>>s2;
  map<char,set<int> > m;
  for (int i=0;i<int(s1.size());i++)
    m[s1[i]].insert(i);
  bool error=false;
  for (int i=0;i<int(s2.size()) and not error;i++)
    error=(m.find(s2[i])==m.end());
  if (error)
    cout<<-1<<endl;
  else {
    int count=1;
    int index=0;
    for (int i=0;i<int(s2.size());i++) {
      set<int> &s=m[s2[i]];
      set<int>::iterator it=s.lower_bound(index);
      if (it==s.end()) {
	count++;
	index=0;
	it=s.lower_bound(index);
      }
      index=(*it)+1;
    }
    cout<<count<<endl;
  }
}
