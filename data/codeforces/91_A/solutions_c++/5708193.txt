#include <stdio.h>
#include <iostream>
#include <string>
using namespace std;
int p=0,n,m,a[15000][30];
string s,t;
int main( ){
	cin>>s>>t;
	n=s.length();
	m=t.length();
	for (int i=0;i<26;i++) a[n][i]=n;
	for (int i=n-1;i>=0;i--){
		for (int j=0;j<26;j++)
			a[i][j]=a[i+1][j];
		a[i][s[i]-97]=i;
	}
	for (int i=0,j=0;i<m;i++,j++){
		j=a[j][t[i]-97];
		if (j==n){
			j=a[0][t[i]-97];
			p++;
			if (j==n){
				cout<<-1<<endl;
				return 0;
			}
		}
	}
	cout<<p+1<<endl;
	return 0;
}
