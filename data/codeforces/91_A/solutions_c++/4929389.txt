#include <cstdio>
#include <set>
#include <cstring>
#include <algorithm>

using namespace std;
char s1[10001],s2[1000001];
set<int> h[26];
int main(){
	// freopen("A.in","r",stdin);
	scanf("%s%s",s1,s2);
	int n = strlen(s1),m = strlen(s2);
	for (int i = 0; i < n; i++)
		h[s1[i]-'a'].insert(i);
	int cur = 0, cnt = 1;
	for (int i = 0; i < m; i++){
		if (h[s2[i]-'a'].size() == 0) {
			puts("-1");
			return 0;
		}
		set<int>::iterator it = h[s2[i]-'a'].lower_bound(cur);
		if (it == h[s2[i]-'a'].end()) {
			cnt++;
			it = h[s2[i]-'a'].begin();
		}
		cur = *it + 1;
	}
	printf("%d\n",cnt);
}
