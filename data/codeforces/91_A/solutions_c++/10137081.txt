#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;

string s1,s2 ;
vector < vector <int> > v(26);
int l=-1,ans=1;

int main()
{
	cin>>s1>>s2;
	for(int j=0 ; j<int(s1.length()) ; j++)
		v[(s1[j]-'a')] . push_back(j);
	for(int i=0 ; i<int(s2.length()) ; i++){
		if(v[(s2[i]-'a')] .empty())
		{	cout<<-1<<endl;   return 0; }
		if ( upper_bound (v[(s2[i]-'a')].begin() , v[(s2[i]-'a')].end() , l) == v[(s2[i]-'a')].end() )
		{	ans++;  l=-1;	}
		l = *upper_bound (v[(s2[i]-'a')].begin() , v[(s2[i]-'a')].end() , l) ;
	}
	cout<<ans<<endl;
	return 0;
}
