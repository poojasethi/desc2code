#include<cstdio>
#include<cstring>
#include<set>
using namespace std;
char s1[10005],s2[1000005];
set<int> aa[26];
int main(){
    int flag=1,ans=1,n,m,cur=0;
    scanf("%s%s",s1,s2);
    n=strlen(s1);m=strlen(s2);
    for(int i=0;i<n;i++){
    	aa[s1[i]-'a'].insert(i);
    }
	for(int i=0;i<m;i++){
		if(aa[s2[i]-'a'].size()==0){
			flag=0;break;
		}
		else{
			set<int>::iterator it=aa[s2[i]-'a'].lower_bound(cur);
			if(it==aa[s2[i]-'a'].end()){
				ans++;it=aa[s2[i]-'a'].begin();
			}cur=*it+1;
		}
	}

    if(!flag)printf("-1\n");
    else printf("\n%d\n",ans);
    return 0;
}
