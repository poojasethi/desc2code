#include <stdio.h>
#include <string.h>

int a[10005][26];
int d[26];
char s1[10005], s2[1000005];

int main(void) {
	scanf("%s", s1+1);
	for(int i=0;i<26;i++) d[i]=-1;
	int n = strlen(s1+1);
	for(int i=n;i>0;--i)
		d[s1[i]-'a'] = i;
	for(int i=n;i>=0;--i) {
		for(int j=0;j<26;j++)
			a[i][j] = d[j];
		if(i) d[s1[i]-'a'] = i;
	}
	scanf("%s", s2);
	int t=1, now = 0, flag=0;
	for(int i=0;s2[i];i++) {
		if(a[now][s2[i]-'a'] == -1) flag=1;
		else {
			if(now >= a[now][s2[i]-'a']) ++t;
			now = a[now][s2[i]-'a'];
		}
	}
	printf("%d\n", flag?-1:t);
	return 0;
}
