#include<stdio.h>
char s[10010],c[1000100];
struct ee{int l,c[10010];}d[30];
int FIND(int y,int x){
	int l=-1,r=d[x].l-1,m;
	if(r==-1)return -1;
	if(d[x].c[r]<y)return d[x].c[0];
	while(l+1<r){
		m=(l+r)/2;
		if(d[x].c[m]>=y)r=m;
		else l=m;
	}
	return d[x].c[r];
}
int main(){
	int i,ans=1,y,x;
	scanf("%s%s",s,c);
	for(i=0;i<30;i++)d[i].l=0;
	for(i=0;s[i]!=0;i++){
		d[s[i]-'a'].c[d[s[i]-'a'].l++]=i;
	}
	for(i=0,y=0;c[i]!=0;i++){
		x=FIND(y,c[i]-'a');
		if(x==-1){ans=-1;break;}
		if(x<y)ans++;
		y=x+1;
	}
	printf("%d\n",ans);
	scanf(" ");
}
