#include <cstdio>
#include <cstring>
using namespace std;

char s[10002],t[1000002];
int i,j,n,m,nxt[10002][26],used[256],cnt=1;
int main ()
{
    scanf("%s%s",s,t);
    n=strlen(s);
    m=strlen(t);
    for (i=0;i<n;++i) used[s[i]]=1;
    for (i=0;i<m;++i) if (!used[t[i]]) { printf("-1"); return 0; }
    for (i=0;i<26;++i) nxt[n][i]=-1;
    for (i=n-2;i>=-1;--i) {
        for (j=0;j<26;++j) nxt[i+1][j]=nxt[i+2][j];
        nxt[i+1][s[i+1]-'a']=i+1;
    }
    j=0;
    for (i=0;i<m;++i) {
        if (nxt[j][t[i]-'a']==-1) ++cnt,j=0;
        j=nxt[j][t[i]-'a']+1;
    }
    printf("%d",cnt);
    return 0;
}
