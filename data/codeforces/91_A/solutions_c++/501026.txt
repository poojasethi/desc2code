#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std;

char buf1[10100], buf2[1000100];
int a[26][10100];
int sz[26];

int main(){
  gets(buf1);
  gets(buf2);
  for (int i=0; buf1[i]; ++i)
    a[buf1[i]-'a'][sz[buf1[i]-'a']++]=i;
  int ans=1, pos=0;
  for (int i=0; buf2[i]; ++i){
    if (sz[buf2[i]-'a']==0){
      cout<<-1;
      return 0;
    }
    int j=lower_bound(a[buf2[i]-'a'], a[buf2[i]-'a']+sz[buf2[i]-'a'], pos)-a[buf2[i]-'a'];
    if (j==sz[buf2[i]-'a']){
      ans++;
      j=0;
    }
    pos=a[buf2[i]-'a'][j]+1;
  }
  cout<<ans;
}
