#include<stdio.h>
#include<vector>
#include<algorithm>
using namespace std;
char a[10010];
char b[1000010];
vector<int> e[26];
int main(){
	scanf("%s%s", a, b);
	for (int i = 0; a[i]; i++)
		e[a[i] - 'a'].push_back(i);
	int ans = 1, now = 0;
	for (int j = 0; b[j]; j++){
		int t = b[j] - 'a';
		if (e[t].empty()){
			puts("-1");
			return 0;
		}
		vector<int>::iterator it = lower_bound(e[t].begin(), e[t].end(), now);
		if (it == e[t].end()){
			ans++;
			now = *e[t].begin() + 1;
		}
		else
			now = *it + 1;
	}
	printf("%d\n", ans);
	return 0;
}
