#include <cstdio>
#include <cstring>
using namespace std;
char s1[10010],s2[1001000];
int ps[10010][30]={};
int main()
{
    scanf("%s%s",s1,s2);
    memset(ps,-1,sizeof(ps));
    for (int i=0;s1[i];++i)
        for (int j=i;j>=0&&ps[j][s1[i]-'a']==-1;--j)
            ps[j][s1[i]-'a']=i;
    int ans = 1,pos=0;
    for (int i=0;s2[i];++i)
    {
        pos = ps[pos][s2[i]-'a'];
        if (pos==-1)
        {
            if (ps[0][s2[i]-'a']==-1)
            {
                ans = -1;
                break;
            }
            ans++;
            i--;
        }
        pos++;
    }
    printf("%d\n",ans);
    return 0;
}
