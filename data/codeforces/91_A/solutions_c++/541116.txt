#include <vector>
#include <iostream>
#include <set>
using namespace std;

string s1,s2;
set<int> places[128+10];

int main()
{
  cin>>s1>>s2;
  for(int i=0;i<s1.size();++i)
    places[s1[i]].insert(i);
  int used=1;
  int index=-1;
  for(int i=0;i<s2.size();++i)
    {
      char x=s2[i];
      if(places[x].empty())
	{
	  cout<<-1<<endl;
	  return 0;
	}
      set<int>::iterator next=places[x].upper_bound(index);
      if(next==places[x].end())
	{
	  used++;
	  index=-1;
	  next=places[x].upper_bound(index);
	}
      index=(*next);
    }
  cout<<used<<endl;
  return 0;
}
