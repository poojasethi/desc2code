#include<iostream>
#include<vector>
#include<algorithm>
#include<string>
using namespace std;
string s1,s2;
const int H=30;
vector<int> a[H];
int main(){
	cin>>s1;
	cin>>s2;
	for(int i=0;i<s1.length();i++){
		a[s1[i]-'a'].push_back(i);
	}
	int out=0;
	int now=0;
	for (int j=0;j<s2.length();j++){
		int x=s2[j]-'a';
		int p=lower_bound(a[x].begin(),a[x].end(),now)-a[x].begin();
	//	cerr<<p<<endl;
		if(p==a[x].size()){
			out++;
			now=0;
			if(a[x].size()==0){
				cout<<-1<<endl;
				return 0;
			}
			j--;
		}else{
		//	cerr<<"----->"<<a[x][p]+1<<endl;
			now=a[x][p]+1;
		}
	}
	cout<<out+1<<endl;
	return 0;
}
