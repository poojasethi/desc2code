#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 10010
#define M 1000010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,an;
int a[N][26];
char p[N],s[M];
int main() {
	scanf("%s%s",p+1,s+1);
	n=strlen(p+1);
	m=strlen(s+1);
	for (i=n;i;i--) {
		For(j,0,25) a[i][j]=a[i+1][j];
		a[i][p[i]-'a']=i;
	}
	j=1;
	for (an=1;;an++) {
		int now=1;
		for (;j<=m;) {
			int A=s[j]-'a';
			if (!a[1][A]) {
				printf("-1\n");
				return 0;
			}
			if (!a[now][A]) break;
			now=a[now][A]+1;
			j++;
		}
		if (j>m) {
			printf("%d\n",an);
			return 0;
		}
	}
	return 0;
}
