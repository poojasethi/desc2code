# include <stdio.h>
# include <string.h>

int ind[10050][30];

int ans;

char a[10050];
char b[1000050];

int main()
{
	scanf("%s\n%s",a,b);
	
	int s=strlen(a);
	int s1=strlen(b);
	
	for (int i=0;i<26;i++) ind[s][i]=-1;
	
	for (int i=s-1;i>=0;i--)
	{
		for (int j=0;j<26;j++) ind[i][j]=ind[i+1][j];
		
		ind[i][a[i]-97]=i;
	}
	
	int u=0;
	
	for (int i=0;i<s1;i++)
	{
		if(ind[u][b[i]-97] != -1) u=ind[u][b[i]-97]+1;
		else
		{
			if(ind[0][b[i]-97] == -1) {printf("-1");return 0;}
			
			u=ind[0][b[i]-97]+1;ans++;
		}
	}
	
	printf("%d",ans+1);
	
	getchar();getchar();
}
