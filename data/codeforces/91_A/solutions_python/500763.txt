s1 = raw_input()
s2 = raw_input()
l = len(s1)
v = [[] for i in range(0,l + 10)]
cur = [-1] * 26
for i in range(0, l):
    v[i] = cur[:]
    cur[ord(s1[i]) - ord('a')] = i
v[l] = cur[:]
c = l
tot = 1
for i in range(len(s2) - 1, -1, -1):
    c = v[c][ord(s2[i]) - ord('a')]
    if (c == -1):
        c = v[l][ord(s2[i]) - ord('a')]
        tot += 1
    if (c == -1):
       tot = -1
       break
print(tot)
