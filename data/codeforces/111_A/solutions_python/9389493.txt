n, a, b = map(int, raw_input().split())

ans = []
k = b - n + 1
for i in range(n):
    nex = 0
    if i < n-1:
        nex = 1
    else:
        nex = k
    a -= nex**2
    ans.append(nex)
    
if a > 0 or k <= 0 :
    print -1
    exit()

for i in ans:
    print i
