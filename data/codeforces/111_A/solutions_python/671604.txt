L = map(int, raw_input().strip().split())
n = L[0]
x = L[1]
y = L[2]

sumx = 0
ans = []
for i in range(0, n - 1):
    ans.append(1)
    sumx = sumx + (1)*(1)
for i in range(0, 1):
    ans.append(y - n + 1)
    sumx = sumx + (y - n + 1) * (y - n + 1)

if sumx < x or y < n:
    print -1
else:
    for i in ans:
        print i
