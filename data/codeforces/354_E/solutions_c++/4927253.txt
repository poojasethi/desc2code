#include<iostream>
#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<cmath>
#include<vector>
#include<algorithm>
using namespace std;
typedef long long LL;
LL n,b[7];
int g[25][25][3],bit[25];
bool f[25][25];
void work()
{
	int k=0;
	memset(g,0,sizeof(g));
	memset(f,false,sizeof(f));
	memset(b,0,sizeof(b));
	while(n>0)
	{
		bit[++k]=n%10;
		n/=10;
	}
	f[0][0]=true;
	for(int i=1;i<=k;i++)
	{
		for(int j=0;j<25;j++)
			if(f[i-1][j])
			{
				for(int ki=0;ki<=6;ki++)
					for(int kj=0;kj+ki<=6;kj++)
					{
						if((j+ki*4+kj*7)%10==bit[i])
						{
							f[i][(j+ki*4+kj*7)/10]=true;
							g[i][(j+ki*4+kj*7)/10][0]=j;
							g[i][(j+ki*4+kj*7)/10][1]=ki;
							g[i][(j+ki*4+kj*7)/10][2]=kj;
						}
					}
			}
	}
	if(!f[k][0])
	{
		cout<<"-1"<<endl;
		return;
	}
	int j=0,ki;
	for(int i=k;i>0;i--)
	{
		int k=g[i][j][0];
		int x=g[i][j][1];
		int y=g[i][j][2];
		for(ki=0;ki<x;ki++)b[ki]=b[ki]*10+4;
		for(int kj=0;kj<y;kj++)
		{
			b[ki]=b[ki]*10+7;
			ki++;
		}
		for(;ki<6;ki++)
			b[ki]*=10;
		j=k;
	}
	for(int i=0;i<6;i++)
		cout<<b[i]<<" ";
	cout<<endl;
}
int main()
{
	int t;
	scanf("%d",&t);
	while(t--)
	{
		cin>>n;
		work();
	}
	return 0;
}
