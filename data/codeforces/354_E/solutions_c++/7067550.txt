#include <cstdio>
#include <cstring>
#include <iostream>
using namespace std;
#define dep(i,a,b) for (int i=a;i>=b;--i)
#define rep(i,a,b) for (int i=a;i<=b;++i)
#define d(x) cout<<x<<endl
typedef long long ll;

int a[30],dp[30][50],prex[30][50],prey[30][50],ans[30][50];

void work() {
	memset(dp,0,sizeof(dp));
	memset(ans,0,sizeof(ans));
	memset(prex,0,sizeof(prex));
	memset(prey,0,sizeof(prey));
	memset(a,0,sizeof(a));
	ll x; cin>>x;
	int i=1;
	while (x) {
		a[i]=x%10;
		x/=10;
		i++;
	}
	memset(dp,0,sizeof(dp));
	memset(ans,0,sizeof(ans));
	memset(prex,0,sizeof(prex));
	memset(prey,0,sizeof(prey));
	dp[0][0]=1;
	rep(i,1,20)
	rep(j,0,5) {
		rep(x,0,6) {
			if (dp[i][j]) break;
			rep(y,0,6-x) {
				rep(k,0,5)
					if (dp[i-1][k] && k+4*x+7*y-j*10 == a[i]) {
						dp[i][j] = 1;
						prex[i][j] = x;
						prey[i][j] = y;
					}
			}
		}
	}
	
	if (!dp[20][0]) { puts("-1");return; }
	
	int j = 0;
	for (int i=20;i;--i) {
		int x = prex[i][j], y = prey[i][j];
		rep(k,1,x) ans[k][i]=4;
		rep(k,x+1,x+y) ans[k][i]=7;
		j = j*10 + a[i] - 4*x - 7*y;
	}
	
	rep(i,1,6) {
		ll x = 0;
		dep(j,20,1) x = x*10+ans[i][j];
		cout<<x<<' ';
	}
	puts("");
}
int main() {
	int te; scanf("%d",&te);
	while (te--) work();
	return 0;
}
