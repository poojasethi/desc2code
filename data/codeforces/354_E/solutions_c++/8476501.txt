#include<cstdio>
#include<cstring>
using namespace std;
long long pow[20],x,ans[10];
int a[20],i,last,four,seven,Test,cnt,get,k,j;
int f[20][10],F[20][10],S[20][10],L[20][10];
int main()
{
  scanf("%d",&Test);
  pow[1]=1;for (i=2;i<=19;i++) pow[i]=pow[i-1]*10ll;
  while (Test--)
  {  
    scanf("%I64d",&x);
    memset(a,0,sizeof(a));
    for (cnt=0;x;a[++cnt]=x%10,x/=10);
    memset(f,0,sizeof(f));
    memset(F,-1,sizeof(F));
    memset(S,-1,sizeof(S));
    memset(L,-1,sizeof(L));
    f[0][0]=1;
    for (i=0;i<=18;i++)
      for (last=0;last<5;last++)
        if (f[i][last])
          for (four=0;four<=6;four++)
            for (seven=0;seven+four<=6;seven++)
            {
              get=four*4+seven*7+last;
              if (get%10!=a[i+1]) continue;
              f[i+1][get/10]=1;
              F[i+1][get/10]=four;
              S[i+1][get/10]=seven;
              L[i+1][get/10]=last;
            }
     if (!f[19][0]) {puts("-1");continue;}
     for (i=1;i<=6;i++) ans[i]=0;k=0;
     for (i=19;i;i--)
     {
       four=F[i][k];seven=S[i][k];
       k=L[i][k];
       for (j=1;j<=four;j++) ans[j]+=pow[i]*4ll;
       for (j=1;j<=seven;j++) ans[7-j]+=pow[i]*7ll;
     }
     for (i=1;i<=6;i++)
       printf("%I64d ",ans[i]);
     puts("");
  }
  return 0;
}
