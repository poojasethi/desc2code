#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define int64 long long
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
struct ww {
	int a,b,i,j;
} f[20][5];
int i,j,n,m;
int64 k,an[7];
void dfs(int x,int y,int64 z) {
	if (x==1) return;
	int i;
	For(i,1,f[x][y].i) an[i]+=z*7;
	For(i,1,f[x][y].j) an[7-i]+=z*4;
	dfs(x-1,f[x][y].b,z/10);
}
inline void work(int64 x) {
	int i,j,g[20],n,u,r;
	for (n=0;x;x/=10) g[++n]=x%10;
	int64 k=1;
	For(i,1,n-1) k*=10;
	memset(f,0,sizeof(f));
	f[1][0].a=1;
	For(i,1,n)For(j,0,4) if (f[i][j].a) {
		int A=0,B=g[i];
		B-=j;
		if (B<0) A++,B+=10;
		For(u,0,6)For(r,0,6-u) if ((u*7+4*r)%10==B) {
			int C=A+(u*7+4*r)/10;
			f[i+1][C].a=1;
			f[i+1][C].b=j;
			f[i+1][C].i=u;
			f[i+1][C].j=r;
		}
	}
	if (!f[n+1][0].a) {
		printf("-1\n"); return;
	}
	memset(an,0,sizeof(an));
	dfs(n+1,0,k);
	For(i,1,6) printf("%I64d ",an[i]);
	printf("\n");
}
int main() {
	scanf("%d",&n);
	For(i,1,n) scanf("%I64d",&k),work(k);
	//for(;;);
	return 0;
}
