#include<cstdio>
#include<cstring>
#define ll long long

struct node
{
	int num[6];
	bool ise;
}sta[5][10];
ll ans[6];

void prework()
{
	for(int i=0;i<5;i++)
	  for(int j=0;j<10;j++)
		sta[i][j].ise=false;

	for(int i=0;i<=6;i++)
	  for(int j=0;j<=6;j++)
		if(i+j<=6)
		{
			int n1=(i*4+j*7)/10,n2=(i*4+j*7)%10;
			sta[n1][n2].ise=true;
			for(int k=0;k<i;k++)
			  sta[n1][n2].num[k]=4;
			for(int k=0;k<j;k++)
			  sta[n1][n2].num[k+i]=7;
			for(int k=i+j;k<6;k++)
			  sta[n1][n2].num[k]=0;
		}
}

int t;
ll n;

bool dfs(ll n,int l)
{
	ll p=1;
	ll tmp[6];
	for(int i=0;i<l;i++)
      p*=10;
	for(int i=0;i<5;i++)
	 if(n/10-i>=0&&sta[i][n%10].ise)
	 {
		for(int k=0;k<6;k++)
		  ans[k]+=sta[i][n%10].num[k]*p;
		if(n/10-i==0)
		  return true;
		if(dfs(n/10-i,l+1))
		  return true;
		for(int k=0;k<6;k++)
		  ans[k]-=sta[i][n%10].num[k]*p;
	 }
	return false;
}
int main()
{
	scanf("%d",&t);
	prework();
	while(t--)
	{
		for(int i=0;i<6;i++)
		  ans[i]=0;
		scanf("%I64d",&n);
		if(dfs(n,0)==false)
		  puts("-1");
		else
		{
		  for(int i=0;i<6;i++)
		  {
			printf("%I64d ",ans[i]);
		  }
		puts("");
		}
	}
	return 0;
}
