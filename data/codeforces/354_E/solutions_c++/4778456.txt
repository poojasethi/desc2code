#include<iostream>
#include<string.h>

using namespace std;

int main(){
	long long int p[19]; long long int r[6]; long long int x; int n; int t;
	int a[3]; int v[243]; int w[100000];
	p[0] = 1;
	for(int i = 0; i < 18; i++)
		p[i+1] = 10*p[i];
	a[0] = 0; a[1] = 4; a[2] = 7;
	n = 0;
	for(int j=0; j<3; j++)
	for(int k=0; k<3; k++)
	for(int l=0; l<3; l++)
	for(int m=0; m<3; m++){
		v[n] = 1000*a[j]+100*a[k]+10*a[l]+a[m];
		n++;
	}
	memset(w, 0, 100000*sizeof(int));
	w[0] = 1;
	for(int i = 0; i < 6; i++)
		for(int j = 45000; j >= 0; j--)
			if(w[j])
				for(int k = 1; k < 81; k++)
					if(w[j+v[k]] == 0)
						w[j+v[k]]=v[k];
	cin >> t;
	while(t){
		cin >> x;
		for(int i = 0; i < 6; i++)
			r[i] = 0;
		n = 0;
		while(4*p[n] + p[n]/2 <= x)
			n++;
		n--;
		while(n > 3){
			int i = 0;
			while(x >= 8*p[n]){
				x -= 7*p[n];
				r[i] += 7*p[n];
				i++;
			}
			if(2*x >= 9*p[n]){
				x -= 4*p[n];
				r[i] += 4*p[n];
			}
			n--;
		}
		if(w[x]){
			int j = 0;
			while(x){
				r[j] += w[x];
				x -= w[x];
				j++;
			}
			for(int i = 0; i < 5; i++)
				cout << r[i] << ' ';
			cout << r[5] << endl;
		}
		else
			cout << -1 << endl;
		t--;
	}
	return 0;
}
