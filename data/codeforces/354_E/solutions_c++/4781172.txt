
#include <iostream>

using namespace std;

long long sol[20][2]; // sol[i][0] is 4s, sol[i][1] is 7s

bool solve(long long n, int k) {
    if (n == 0) {
        long long a[6];
        for (int i = 0; i < 6; i++)
            a[i] = 0;

        long long p = 1;
        for (int i = 0; i < k; i++, p *= 10) {
            int j = 0;
            while (sol[i][0]--)
                a[j++] += p * 4;
            while (sol[i][1]--)
                a[j++] += p * 7;
        }
        for (int i = 0; i < 6; i++)
            cout << a[i] << " ";
        cout << endl;
        return true;
    }

    for (int i = 0; i <= 6; i++) {
        for (int j = 0; i + j <= 6; j++) {
            int w = 4 * i + 7 * j;
            if (n >= w && (n - w) % 10 == 0) {
                sol[k][0] = i;
                sol[k][1] = j;
                if (solve((n - w) / 10, k + 1))
                    return true;
            }
        }
    }
    return false;
}

int main() {
    int T;
    cin >> T;
    while (T--) {
        long long num;
        cin >> num;
        if (!solve(num, 0))
            cout << -1 << endl;
    }
}
