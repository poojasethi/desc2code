#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>
using namespace std;

typedef int arr32[40][110];

int n;
char s[200];
arr32 g, f, xt, yt;
long long a[10];

int main()
{
	scanf("%d\n", &n);
	for (int i = 1; i <= n; ++i)  {
		scanf("%s", s + 1);

		int len = strlen(s + 1);

		memset(g, 0, sizeof(g));

		g[0][0] = 1;
		for (int i = 1; i <= len; ++i)  {
			for (int j = 0; j <= 4; ++j)
				if (g[i - 1][j])  {
					for (int x = 0; x <= 6; ++x)
						for (int y = 0; y <= 6 - x; ++y)  {
							int h = j * 10 + s[i] - '0' - x * 7 - y * 4;
							if (h < 0)  continue;
							g[i][h] = 1;  f[i][h] = j;  xt[i][h] = x, yt[i][h] = y;
						}
				}
		}

		if (g[len][0] == 0)  printf("%d\n", -1);
		else  {
			for (int i = 1; i <= 6; ++i)  a[i] = 0;
			int x = len, h = 0;  long long R = 1;
			while (x)  {
				for (int i = 1; i <= xt[x][h]; ++i)  a[i] += 7 * R;
				for (int i = 1; i <= yt[x][h]; ++i)  a[i + xt[x][h]] += 4 * R;

				h = f[x][h];  --x;  R *= 10;
			}
			for (int i = 1; i <= 6; ++i)  printf("%I64d ", a[i]);
			printf("\n");
		}
	}
}
