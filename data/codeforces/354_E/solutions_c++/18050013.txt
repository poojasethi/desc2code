#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

typedef long long int64;
int task,a[25],c4[25][10],c7[25][10],pre[25][10];
int64 n,ans[7];
bool f[25][10];
int main() {
	scanf("%d",&task);
	while (task--) {
		scanf("%I64d",&n);
		memset(a,0,sizeof(a));
		for (int i=1;i<25;i++) {
			a[i]=n%10;
			n/=10;
		}
		memset(f,false,sizeof(f));
		memset(pre,0,sizeof(pre));
		memset(c4,0,sizeof(c4));
		memset(c7,0,sizeof(c7));
		f[0][0]=true;
		for (int i=0;i<24;i++)
			for (int j=0;j<=9;j++) if (f[i][j])
				for (int num4=0;num4<=6;num4++)
					for (int num7=0;num7<=6-num4;num7++) if ((num4*4+num7*7+j)%10==a[i+1]) {
						int next=(num4*4+num7*7+j)/10;
						if (next>9) puts("WA");
						f[i+1][next]=true; pre[i+1][next]=j; c4[i+1][next]=num4; c7[i+1][next]=num7;
					}
		if (!f[24][0]) {
			puts("-1");
			continue;
		}
		memset(ans,0,sizeof(ans));
		for (int i=24,j=0;i;i--) {
			for (int k=1;k<=6;k++) ans[k]*=10;
			for (int k=1;k<=c4[i][j];k++) ans[k]+=4;
			for (int k=c4[i][j]+1;k<=c4[i][j]+c7[i][j];k++) ans[k]+=7;
			j=pre[i][j];
		}
		for (int i=1;i<=6;i++) printf("%I64d%c",ans[i],i==6?'\n':' ');
	}
	return 0;
}
