#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
int a[100][6],a1[100],s[6];
void l(int q,int t){
  int w;
  if(a1[q]==0){
   a1[q]=1;
   for(w=0;w<t;w++)
    a[q][w]=s[w];
   for(w=t;w<6;w++)
    a[q][w]=0;}
  else
   return;
  if(t>5)
   return;
  s[t]=4;
  l(q+4,t+1);
  s[t]=7;
  l(q+7,t+1);
}
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int w,e,r,t,o,c,v,s[10][10],s1[10],d[20][10],f[20];
  long long q,g[6];
  for(w=1;w<100;w++)
   a1[w]=0;
  l(0,0);
  for(w=0;w<10;w++)
   s1[w]=0;
  for(w=0;w<100;w++)
   if(a1[w]){
    s[w%10][s1[w%10]]=w;
    s1[w%10]++;}
  for(scanf("%d",&o);o;o--){
   cin>>q;
   for(w=0;w<20;w++)
    for(e=0;e<10;e++)
     d[w][e]=-1;
   d[0][0]=0;
   for(w=0;w<19;w++){
    f[w]=c=q%10;
    for(e=0;e<10;e++){
     if(d[w][e]>=0)
      for(r=0;r<s1[c];r++)
       d[w+1][(s[c][r]+e)/10]=s[c][r];
     c--;
     if(c<0)
      c+=10;}
    q/=10;}
   if(d[19][0]<0){
    printf("-1\n");
    continue;}
   for(w=0;w<6;w++)
    g[w]=0;
   c=0;
   for(w=19;w>0;w--){
    for(e=0;e<6;e++){
     g[e]=g[e]*10+a[d[w][c]][e];}
    c=c*10+f[w-1]-d[w][c];}
   for(w=0;w<6;w++)
    printf("%I64d ",g[w]);
   printf("\n");}
  return 0;}
