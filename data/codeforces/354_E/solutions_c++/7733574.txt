#include<cstdio>
#include<cstdlib>
#include<algorithm>
#define N 4
#define LL long long
#define rep(i,n) for(int i=0;i<=n;i++)
#define Rep(i,n) for(int i=1;i<=n;i++)
using namespace std;
int T,now,pre;
LL n;
struct rec{LL ans[8];bool success;}New,f[2][5];
int main(){
    scanf("%d",&T);
    while (T--){
        scanf("%I64d",&n);
        LL ans=0;
        LL sum=(LL)1e18;
        for (;sum>n;sum/=10);
        rep(i,N) f[0][i]=f[1][i]=New;
        f[now][0].success=1;
        while (sum){
            int val=n/sum;
            pre=now; now^=1;
            rep(i,N) f[now][i]=New;
            rep(i,6) rep(j,6)
            if(i+j<=6){
                int dat=i*4+j*7;
                rep(k,4)
                if(f[pre][k].success){
                    int x=k*10+val-dat;
                    if(0<=x&&x<=4){
                        f[now][x]=f[pre][k];
                        Rep(d,i) f[now][x].ans[d]+=sum*4;
                        Rep(d,j) f[now][x].ans[d+i]+=sum*7;
                    }
                }
            }
            n%=sum; sum/=10;
        }
        if(!f[now][0].success) printf("-1");
        else Rep(d,6) printf("%I64d ",f[now][0].ans[d]); printf("\n");
    }
}
