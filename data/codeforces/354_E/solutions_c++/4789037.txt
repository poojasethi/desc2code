#include<iostream>
using namespace std;
typedef long long LL;
LL f[1000];
LL t[7];
LL s[1000];
LL g[100];
LL z = 0;
bool flag;
LL a[7];
LL e[3] = {0, 4, 7};
void print()
{
	for(int i = 1; i <= 6; i++)
		a[i] = 0;
	for(int i = z; i >= 1; i--)
	{
		for(int j = 1; j <= 7; j++)
		{
			a[j] = a[j] * 10 + e[s[i]%3];
			s[i] /= 3;
		}
	}
	for(int i = 1; i <= 6; i++)
		cout << a[i] << " ";
	cout << endl;
}
void solve(LL n)
{
	if(flag)
		return ;
	if(n == 0)
	{
		print();
		flag = true;
		return ;
	}
	for(LL i = 0; i <= 42; i++)
		if(n >= i && g[i] != -1 && n % 10 == i % 10)
		{
			z++;
			s[z] = g[i];
			solve((n-i)/10);
			z--;
		}
	return ;
}
int main()
{
	t[0] = 1;
	for(int i = 1; i <= 6; i++)
		t[i] = t[i-1] * 3;
	for(int i = 0; i < t[6]; i++)
	{
		int x = i;
		for(int j = 5; j >= 0; j--)
		{
			if(x >= t[j])
			{
				if(x >= t[j]*2)
					f[i] += 7;
				else
					f[i] += 4;
				x %= t[j];
			}
		}
	}
	for(int i = 0; i < 50; i++)
		g[i] = -1;
	for(int i = 0; i < t[6]; i++)
		g[f[i]] = i;
	int T;
	cin >> T;
	for(; T > 0; T--)
	{
		LL n;
		flag = false;
		cin >> n;
		solve(n);
		if(!flag)
			cout << -1 << endl;
	}
	return 0;
}
