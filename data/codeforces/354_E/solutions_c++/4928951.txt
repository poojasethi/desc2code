#include <iostream>
#include <cstdio>
#include <vector>

#define pii pair<int, int>
#define mk make_pair
#define pb push_back
#define F first
#define S second

using namespace std;

long long n, res[6];
vector<pii> digit[10];

int init()
{
  for (int i = 0; i <= 6; i++)
    for (int j = 0; j <= 6 - i; j++) 
      digit[(4 * i + 7 * j) % 10].pb(mk(i, j));
}

int TRY(long long n, long long pow)
{
  if (n == 0) return 1;
  if (n < 0) return 0;
  int d = n % 10;
  
  for (int i = 0; i < digit[d].size(); i++)
    if (TRY((n - 4 * digit[d][i].F - 7 * digit[d][i].S) / 10, pow * 10)) 
    {
      for (int j = 0; j < digit[d][i].F; j++) res[j] += pow * 4;
      for (int j = 0; j < digit[d][i].S; j++) res[5 - j] += pow * 7;
      return 1;
    }
  return 0;
}

main()
{
  init();
  int test;
  cin >> test;
  while (test--)
  {
    cin >> n;
    for (int i = 0; i < 6; i++) res[i] = 0;
    if (TRY(n, 1)) 
      for (int i = 0; i < 6; i++) cout << res[i] << " ";
    else
      cout << -1;
    cout << "\n";
  }
//  system("pause");
}
