#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<climits>
using namespace std;
struct d{
	int x,y,d;
}f[28];
int a[20],len;
int l[20][10],m[20][10];
long long ans[20];
inline bool operator<(d _a , d _b){
	return  _a.d<_b.d;
}
int t;
int main(){
	for (int i=0;i<=6;i++)
		for (int j=0;j+i<=6;j++){
			f[t].x=i;f[t].y=j;f[t].d=i*4+j*7;
			t++;
		}
	sort(f,f+28);
	/*for (int i=0;i<28;i++)
		printf("%d %d %d\n",f[i].x,f[i].y,f[i].d);*/
	int t;scanf("%d",&t);while (t--){
		long long P;
		cin>>P;
		len=0;
		while (P){
			len++;a[len]=P%10;
			P/=10;
		}
		memset(m,-1,sizeof m);
		m[0][0]=0;
		for (int i=0;i<len;i++)
			for (int j=0;j<10;j++)
				if (m[i][j]!=-1)
					for (int t=0;t<28;t++)
					if ((j+f[t].d)%10==a[i+1]){
						m[i+1][(j+f[t].d)/10]=t;
						l[i+1][(j+f[t].d)/10]=j;
					}
		int flg=-1;
		
		if (m[len][0]!=-1)
			flg=0;
		if (flg==-1)puts("-1");
		else {
			int pf=len,p=flg;
			for (int t=1;t<=6;t++)ans[t]=0;
			while (pf){
				for (int t=1;t<=6;t++)ans[t]*=10;
				for (int t=1;t<=f[m[pf][p] ].x;t++)ans[t]+=4;
				for (int t=f[m[pf][p] ].x+1;t<=f[m[pf][p] ].x+f[m[pf][p] ].y;t++)ans[t]+=7;
				p=l[pf][p];
				pf--;
			}
			cout<<ans[1]<<" "<<ans[2]<<" "<<ans[3]<<" "<<ans[4]<<" "<<ans[5]<<" "<<ans[6]<<endl;
		}
}}
