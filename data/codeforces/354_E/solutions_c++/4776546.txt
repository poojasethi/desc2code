#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define LL __int64
int ans[20][4],flag;
int dfs(LL n,int dep)
{
    if(n==0)
    {
        flag=1;
        LL num[6],j=1;
        memset(num,0,sizeof(num));
        for(int i=0;i<dep;++i,j*=10)
        {
            int w=0;
            while(ans[i][0])num[w++]+=4*j,--ans[i][0];
            while(ans[i][1])num[w++]+=7*j,--ans[i][1];
        }
        for(int i=0;i<6;++i)printf("%I64d ",num[i]);
        cout<<endl;
        return 1;
    }
    LL k=n%10;
    for(int i=0;i<=6;++i)
        for(int j=0;i+j<=6;++j)
        {
            LL w=i*4+j*7;
            if(w%10==k&&n>=w)
            {
                ans[dep][0]=i;
                ans[dep][1]=j;
                ans[dep][2]=6-i-j;
                dfs((n-w)/10,dep+1);
                if(flag)return 1;
            }
        }
    return 0;
}
int main()
{
    int t;
    scanf("%d",&t);
    while(t--)
    {
        LL n;
        flag=0;
        scanf("%I64d",&n);
        if(!dfs(n,0))printf("-1\n");
    }
    return 0;
}
