#include<cstdio>
#include<string>
#include<cstring>
#include<sstream>
#include<iostream>
#include<algorithm>
using namespace std;
const int N = 5e5 + 9;
struct Node {
	double st, ed, dir;
} p[N];
struct flag {
	double num, id;
} q[N];
bool cmp(flag a, flag b) {
	return a.num < b.num;
}
int main() {
	int n, m, i, j, k;

	while(~scanf("%d", &n)) {
		for(i = 0; i < n; ++i) {
			scanf("%lf%lf", &p[i].st, &p[i].dir);
		}

		double L = 0, R = 1e9 + 10, mid, ans = -1;

		for(j=0; j<100; ++j) {
			mid = (L + R) / 2;
			int cnt = 0;
			bool ok = false;
			double mx = -1e100, cur;

			for(i = 0; i < n; ++i) {
				cur = p[i].st + mid * p[i].dir;

				if(p[i].dir > 0) {
					mx = max(mx, cur);
				} else if(mx >= cur) {
					ok = true;
					break;
				}
			}

			if(ok) {
				R = ans = mid;
			} else {
				L = mid;
			}
		}

		printf("%.20f\n", ans);
	}

	return 0;
}
