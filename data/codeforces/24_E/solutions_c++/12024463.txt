#include <cstdio>
#include <algorithm>
#include <cmath>
using namespace std;
int n, x[500005], v[500005];

int ok(double t){
    double now = -1e20;
    for (int i = 0; i < n; ++ i){
        if (v[i] > 0){
            now = max(now, x[i] + t * v[i]);
        } else {
            if (x[i] + t * v[i] <= now) return 1;
        }
    }
    return 0;
}

int main(){
    scanf("%d", &n);
    for (int i = 0; i < n; ++ i){
        scanf("%d%d", &x[i], &v[i]);
    }
    double l = 0, r = 1e10;
    int t = 0;
    while (r - l > 1e-10 && t < 100){
        t++;
        double mid = (l + r) / 2;
        if (ok(mid)) r = mid; else l = mid;
    }
    if (r >= 1e10){
        puts("-1");
    } else {
        printf("%.10f\n", l);
    }
    return 0;
}
