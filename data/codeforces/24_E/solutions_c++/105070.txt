#include<cstdio>
#include<iostream>
#define LD long double
#define _min(a,b) ((a)<(b)?(a):(b))
using namespace std;
#define N 555555

int f=0;
LD s[N],l,r,c;
int x[N],v[N],n,ps;
int main(){
/*
	freopen(".in","r",stdin);
	freopen(".out","w",stdout);
//*/
	scanf("%d",&n);
	for (int i=0;i<n;i++) scanf("%d%d",&x[i],&v[i]);
	l = 0,r = 1e+9 + 10;
	while (r - l > 1e-9){
		ps = - 1,c = (l + r) / 2,f = 0;
		for (int i=0;i<n;i++){
			s[i] = c * v[i] + x[i];
			if (v[i] > 0){
				if (ps == -1 || s[i] > s[ps]) ps = i;
			}else if (ps > -1){
				r = _min(r,(LD) (x[i] - x[ps]) / (LD) (v[ps] - v[i]));
				if (s[i] < s[ps]) f = 1;
			}
		}
		if (!f) l = c;
	}
	cout.precision(13);
	if (r > 1e+9 + 5) puts("-1");else cout<< fixed<< r<< endl;
	return 0;
}
