#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
typedef long long ll;
const int N = 500005;
const ll INF = (ll)(1e31) - 1;
const double eps = 1e-9;
ll a[N], v[N];
int n;
bool check(double t) {
	double ma = -INF, mi = INF;
	double nw = -INF;
	for(int i = 0; i < n; i ++) {
		if(v[i] > 0) {
			nw = max(nw, a[i] + v[i] * t);
		} else {
			if( nw >= a[i] + v[i] * t) {
				return true;
			}
		}
	}
	return false;
}
int main() {
	scanf("%d", &n);
	ll ma = -INF, mi = INF;
	for(int i = 0; i < n; i ++) {
		scanf("%I64d%I64d", &a[i], &v[i]);
		if(v[i] < 0) ma = max(ma, a[i]);
		else mi = min(mi, a[i]);
	}
	if(ma < mi) {
		puts("-1"); return 0;
	}
	if(ma == mi) {
		puts("0"); return 0;
	}
	
	double l = 0, r = 1e9;
	while((r-l) >= eps * max(1.0, r)) {
		double m = (l + r) / 2;
		if(check(m)) r = m;
		else l = m;
	}
	printf("%.10f\n", (l + r) / 2);
	return 0;
}
