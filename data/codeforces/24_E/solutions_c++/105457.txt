#include <iostream>
#include <vector>
#include <cstdio>
#include <iomanip>

using namespace std;

int n;
vector<int> x, v;

double get (double l = -1e9, double r = 1e9, int depth = 0)
{
	double t = (l + r) / 2;
	
	if(depth == 61) return t;
	
	int i = 0;
	while(v[i] < 0) i++;
	double lim = x[i] + v[i] * t;
	for(++i; i < n; i++)
	{
		double tmp = x[i] + v[i] * t;
		if(v[i] > 0) {
			if(tmp > lim) lim = tmp;	}
		else if(tmp <= lim)
			return get(l, t, depth + 1);
	}

	return get(t, r, depth + 1);
}

int main ()
{
	scanf("%d", &n);; x.resize(n); v.resize(n);
	bool phew = true;
	for(int i = 0; i < n; i++)
	{
		scanf("%d %d", &x[i], &v[i]);
		if(i > 0 && v[i - 1] > 0 && v[i] < 0) phew = false;
	}

	if(phew)
		cout << -1 << endl;
	else
	{
		cout << fixed;
		cout << setprecision(20) << get() << endl;
	}
}
