#include <cstdio>
#include <cstring>
#include <iostream>
#include <cctype>
using namespace std;
double x[510000], v[510000];
int n, m;
int main()
{
    int i, j;
    scanf("%d", &n);
    for (i = 1; i <= n; ++ i)
    {
        scanf("%lf%lf", &x[i], &v[i]);
    }
    double l = 0, r = 2e9 + 2;
    while (r - l >= 1e-10)
    {
        double mid = (l + r) / 2, ma = -1e20;
        int f = 0;
        for (i = 1; i <= n; ++ i)
        if (v[i] > 0)  ma = max(ma, x[i] + v[i] * mid);
        else if (v[i] * mid + x[i] <= ma) {f = 1; break;}
        if (f) r = mid;
            else l = mid;
    }
    if (l > 1e9)  puts("-1");
    else printf("%.9lf\n", l);
    return 0;
}
