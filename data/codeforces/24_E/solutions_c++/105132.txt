#include <stdio.h>
const int MaxN = 500000;
int x[MaxN], v[MaxN];
int n;
bool bigbang;

int main()
{
    scanf("%i",&n);
    double lt=0, rt=2000000000, mgood;
    for (int i=0; i<n; i++) scanf("%i %i", &x[i], &v[i]);
    for (int k=0; k<400; k++)
    {
        bool was = false;
        double t = (lt+rt)/2;
        double max = -1e50;
        int i=0;
        while (v[i]<0) i++;
        while (i<n)
        {
            double w = t*v[i]+x[i];
            if (v[i] < 0)
            {
                if (w < max)
                    was = true;
            }
            else
                if (w > max)
                    max = w;
            i++;
        }
        if (was)
        {
            rt = t;
            mgood = rt;
        }
        else
            lt = t;
    }
    if (lt>1000000000)
        printf("-1\n");
    else
        printf("%0.10lf\n",mgood);
}
