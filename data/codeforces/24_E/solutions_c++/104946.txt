#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#define oo 500005
#define Max(x,y) ((x)>(y) ? (x):(y))
#define Min(x,y) ((x)<(y) ? (x):(y))

int N,opp=0;
int x[oo],y[oo];
double Ans,l=0,r=1e100,arr,mid,now;

bool Check(double t)
{
	arr=-1e100;
	for (int i=1;i<=N;++i){
		now=x[i]+y[i]*t;
		if (y[i]>0) arr=Max(arr,now);
		else if (now<=arr) return 1;
	}
	return 0;
}
int main()
{
//	freopen("E.in","r",stdin);
//	freopen("E.out","w",stdout);
	scanf("%d",&N);
	bool flag=0;
	for (int i=1;i<=N;++i){
		scanf("%d%d",&x[i],&y[i]);
		if (y[i]>0) opp=i;
		if (y[i]<0 && opp){
			flag=1;
			now=double(x[i]-x[opp])/double(y[opp]-y[i]);
			r=Min(r,now);
		}
	}
	if (!flag){
		printf("-1\n");
		exit(0);
	}
	Ans=r;
	for (int i=1;i<=200;++i){
		mid=(l+r)/2;
		flag=Check(mid);
		if (flag){
			Ans=Min(Ans,mid);
			r=mid;
		}else l=mid;
	}
	printf("%.10lf\n",Ans);
	return 0;
}
