#include <iostream>
#include <cstdio>

using namespace std;

const int N = 5e5;
const int K = 200;
const double INF = 1e9 + 1;

int n;
double x[N], v[N];

bool check(double t) {
    double mx = -INF * INF * INF;
    for (int i = 0; i < n; i++) {
        double now = x[i] + v[i] * t;
        if (v[i] < 0) {
            if (now <= mx)
                return 1;
        }
        else
            mx = max(mx, now);
    }
    return 0;
}

int main() {
    // freopen("input.txt", "r", stdin);
    scanf("%d", &n);
    for (int i = 0; i < n; i++)
        scanf("%lf %lf", x + i, v + i);
    double l = 0, r = INF;
    for (int i = 0; i < K; i++) {
        double m = (l + r) / 2;
        if (check(m))
            r = m;
        else
            l = m;
    }
    printf("%.11lf\n", check(r) ? r : -1);
    return 0;
}
