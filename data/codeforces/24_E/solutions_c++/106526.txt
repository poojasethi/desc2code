
#include <cstdio>
#include <algorithm>
using namespace std;

int n, s, f;
int po[500111], ve[500111];

bool cross(double t) {
	int bi=-1;
	for (int i=s; i<f; i++) {
		if (ve[i]>0) {
			if (bi<0 || po[bi]-po[i]<t*(ve[i]-ve[bi])) bi=i;
		} else {
			if (bi>-1 && po[i]-po[bi]<t*(ve[bi]-ve[i])) return true;
		}
	}
	return false;
}

int main() {
	scanf("%d", &n);
	for (int i=0; i<n; i++) {
		scanf("%d%d", po+i, ve+i);
	}

	s=0;	while (s<n && ve[s]<0) ++s;
	f=n;	while (f && ve[f-1]>0) --f;
	if (s==f) {
		printf("-1");
		return 0;
	}
	
	double a=0, b=1e+10, c;
	for (int i=0; i<200; i++) {
		c=(a+b)/2;
		if (c==a || c==b) break;
		if (cross(c)) b=c; else a=c;
	}
	printf("%.22f", b);
	return 0;
}
