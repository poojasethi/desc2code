#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define MAXN 500005
int n,x[MAXN],v[MAXN];
bool ck(double mid)
{
    double now=-1e99;
    for(int i=0;i<n;i++)
    {
        if(v[i]>0) now=max(now,x[i]+mid*v[i]);
        else if(x[i]+v[i]*mid<=now) return true;
    }
    return false;
}
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++) scanf("%d%d",x+i,v+i);
    double low=0,high=1e12,mid;
    for(int i=0;i<100;i++)
    {
        mid=(low+high)*0.5;
        if(ck(mid)) high=mid;
        else low=mid;
    }
    if(high<1e10) printf("%.12f\n",high);
    else printf("-1\n");
    return 0;
}
