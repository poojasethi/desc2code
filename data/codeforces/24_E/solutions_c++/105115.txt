#include <stdio.h>

using namespace std;

int x[500005], v[500005];

int main(){
    //freopen("test.txt", "r", stdin);
    int n;
    scanf("%d\n", &n);
    for(int i = 0; i < n; i++){
        scanf("%d %d\n", &x[i], &v[i]);
    }
    double high = x[n - 1] - (double)x[0] + 10, low = 0, mid;
    while((high - low) * 2 / (high + low) >= 1e-11){
        mid = low + (high - low) / 2;
        double R = -1e60;
        bool ok = false;
        for(int i = 0; !ok && i < n; i++){
            double dis = x[i] + v[i] * mid;
            if(v[i] > 0){
                if(dis > R) R = dis;
            }else if(v[i] < 0){
                if(dis < R) ok = true;
            }
        }
        if(ok) high = mid;
        else low = mid;
    }
    if(mid >= x[n - 1] - (double)x[0]){
        printf("-1\n");
    }else{
        printf("%.131f\n", mid);
    }
    return 0;
}
