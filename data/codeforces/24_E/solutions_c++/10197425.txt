#include <bits/stdc++.h>

#define mp make_pair
#define st first
#define nd second
#define pb push_back

using namespace std;

typedef long long ll;
typedef pair< ll,ll > pll;
typedef pair< int,int > pii;

const int inf = 1000000000, N = 500000;

int n;
pii p[N+5];

bool check(double t) {
	double mx = -1e19;

	for (int i = 1; i <= n; i++) {
		if (p[i].nd < 0) {
			if (mx > p[i].st + p[i].nd * t)
				return true;
		}
		else
			mx = max(mx, p[i].st + p[i].nd * t);
	}

	return false;
}

int main()
{
	scanf("%d", &n);

	for (int i = 1; i <= n; i++) {
		scanf("%d %d", &p[i].st, &p[i].nd);
	}

	sort(p+1, p+n+1);

	double le = 0, ri = 1e9 + 5;

	for (int i = 1; i <= 100; i++) {
		double mi = (le + ri) / 2;

		if (check(mi))
			ri = mi;
		else
			le = mi;
	}
	
	if (check(ri))
		printf("%.20lf\n", le);
	else
		printf("-1\n");
	return 0;
}
