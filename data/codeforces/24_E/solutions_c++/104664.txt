#include <cstdio>
#include <cmath>
#include <algorithm>
#define FOR(i,s,e) for(int i=s;i<e;i++)
using namespace std;

const int N = 2000000;
int a[N], v[N];

int main() {
    int n;
    scanf("%d", &n);
    FOR(i,0,n)
        scanf("%d%d", a+i, v+i);
    double lo = 0, hi = 1e10, md;
    FOR(iter,0,100) {
        int nn = 0;
        md = (hi + lo) * .5;
        int c[3] = {0};
        double R = -1e60;
        bool ok = 0;
        FOR(i,0,n) {
            if (v[i] > 0)   R = max(a[i]+v[i]*md, R);
            else if (a[i]+v[i]*md<R)    ok = 1;
            if (ok) break;
        }
        (ok ? hi : lo) = md;
    }
    if (hi > 5e9)   puts("-1");
    else    printf("%.20f\n", md);
    return 0;
}
