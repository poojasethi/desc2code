#include<bits/stdc++.h>
using namespace std;
const int maxn = 500050;
int a[maxn][2];
int n;
bool check(double m)
{
    double cur=-1e20;
    for(int i=0;i<n;i++)
    {
        if(a[i][1]>0)
            cur=max(cur,a[i][0]+a[i][1]*m);
        else
            if(a[i][0]+a[i][1]*m<=cur)
                return true;
    }
    return false;
}
int main()
{
    scanf("%d",&n);
    for(int i=0;i<n;i++)
        scanf("%d%d",&a[i][0],&a[i][1]);
    double l=0,r=1e10;
    int t=0;
    while(r-l>=1e-12)
    {
        t++;
        double m=(l+r)/2;
        if(check(m))
            r=m;
        else
            l=m;
        if(t==100) break;
    }
    if(r==1e10)
    {
        puts("-1");
        return 0;
    }
    printf("%.14f\n",r);
    return 0;
}
