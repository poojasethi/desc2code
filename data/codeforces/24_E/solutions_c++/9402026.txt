#include <stdio.h>
#include <string.h>
#include <iostream>
#include <algorithm>
using namespace std;
const int maxn=500005;
int n,a[maxn][2];

bool judge(double t)
{
    double d=-1e20;
    for(int i=0;i<n;i++)
    {
         if(a[i][1]>0)
            d=max(d,a[i][0]+t*a[i][1]);
         else if(a[i][0]+t*a[i][1]<=d)return true;
    }
    return false;
}
int main()
{
    while(~scanf("%d",&n))
    {
        for(int i=0;i<n;i++)
            scanf("%d%d",&a[i][0],&a[i][1]);
        double l=0.0,r=1e10;
        int t=0;
        while(r-l>1e-12&&t<100)
        {
            t++;
            double mid=(l+r)/2;
           // printf("%.20lf\n",mid);
            if(judge(mid))
                r=mid;
            else
                l=mid;
        }
        if(r==1e10)
            puts("-1");
        else
            printf("%.14lf\n",r);
    }
    return 0;
}
