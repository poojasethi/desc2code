#include<cstdio>
int x[500005],v[500005],a[500005];
double h=0,t=1e21,m;
int N;
int main(){
	scanf("%d",&N);
	for (int i=1;i<=N;++i){
		scanf("%d%d",&x[i],&v[i]);
	}
	int mx = x[N]-x[1] + 10;
	t = x[N]-x[1] + 10;
	
	while ((t-h)*2/(h+t)>=1e-11){
		//printf("%lf %lf\n",h,t);
		m = (h+t)/2;
		double R = -1e60;
		bool ok = false;
		
		for (int i=1;!ok && i<=N;++i){
			double t = x[i]+v[i]*m;
		
			if (v[i]>0){
				if (t > R){
					R = t ;
				}
			}else if (v[i]<0){
				if (t < R){
					ok = true;
				}
			}
		}
		
		if (ok){
			t = m;
		}else{
			h = m;
		}
	}
	
	//	printf("%lf %lf\n",h,t);
	if (m>=x[N]-x[1]){
		printf("-1\n");
	}else{
		printf("%.13lf\n",m);
	}

	return 0;
}
