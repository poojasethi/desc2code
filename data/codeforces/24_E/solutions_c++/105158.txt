#define INF 2000000000

using namespace std;

struct t1{
	double x,v;
}mas[500005];
double up=INF-1,down,mid,maxx,alt;
int n,neg,poz;
bool flag2=1;

bool parb(double mid){
	maxx=-1e22;
	for (int i=0; i<n; ++i){
		alt=mas[i].x+mas[i].v*mid;
		if (mas[i].v>0){
			if (alt>maxx) {maxx=alt;}
		} else {
			if (alt<=maxx) return 1;
		}
	}
	return 0;
}

int main(){
	//freopen("e.in","r",stdin);
	//freopen("e.out","w",stdout);
	scanf("%d",&n);
	for (int i=0; i<n; ++i){
		scanf("%lf%lf",&mas[i].x,&mas[i].v);
		if (mas[i].v>0) ++poz;
		else
			if (poz>0) flag2=0;
	}
	if (!flag2){
		for (int i=1; i<=70; ++i){
			mid=(up+down)/2;
			if (parb(mid)) up=mid; else down=mid;
		}
		printf("%.10lf",mid);
	}
	else printf("-1");
	return 0;
}
