#include <stdio.h>
#include <math.h>
#include <iostream>

int const N = 555555;
int n;
double x[N], v[N];
int ans1 = -1, ans2 = -1;

bool check(double t)
{
  double right = -2e9;
  int who = -1;
  for (int i = 0; i < n; i++)
  {
    if (v[i] > 0)
    {
      if (x[i] + t*v[i] > right)
      {
        right = x[i] + t*v[i];
        who = i;
      }
    }
    else
    {
      if (right + 1e-10 > x[i] + t*v[i])
      {
        ans1 = who, ans2 = i;
        return 1;
      }
    }
  }

  return 0;
}

int main()
{
  scanf("%d", &n);
  for (int i = 0; i < n; i++)
    scanf("%lf%lf", &x[i], &v[i]);

  double l = 0, r = 2e9;
  for (int i = 0; i < 100; i++)
  {
    double m = (r + l) / 2;
    if (check(m))
      r = m;
    else
      l = m;
  }
  check(r);

  if (ans1 != -1)
    printf("%.16lf\n", fabs((x[ans1] - x[ans2])/(v[ans1] - v[ans2])));
  else
    puts("-1\n");
  return 0;
}
