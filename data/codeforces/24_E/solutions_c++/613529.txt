#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;

int main(void) {
	int n;
	scanf("%d", &n);
	vector<pair<int, int> > a;
	for(int i=0;i<n;i++) {
		int x, v;
		scanf("%d%d", &x, &v);
		a.push_back(make_pair(x, v));
	}
	double L = 0, R = 1e10, ans = -1, M;
	for(int u=0;u<=70;u++){
		M = (L+R)/2;
		int i;
		double lastPos = -1e100;
		for(i=0;i<n;i++) {
			if(a[i].second > 0)
				lastPos = max(lastPos, a[i].first + a[i].second * M);
			else if(a[i].first + a[i].second * M < lastPos)
				break;
		}
		if(i==n) L = M;
		else {ans = M; R = M;}
	}
	if(ans<0) puts("-1");
	else
	printf("%.10lf\n", ans);
	return 0;
}
