import string
import itertools
import sys

def bullcow(num1, num2):
    num1 = list(num1)
    num2 = list(num2)
    bull = 0
    cow = 0
    for idx1, dg1 in enumerate(num1):
        for idx2, dg2 in enumerate(num2):
            if dg2 == dg1:
                if idx2 == idx1:
                    bull += 1
                else:
                    cow += 1
    return bull, cow

n = int(raw_input())
guesses = []
bullscow = []
for _ in range(n):
    g, b, c = raw_input().split()
    guesses.append(g)
    bullscow.append((int(b), int(c)))

candidates = set(itertools.permutations(string.digits, r=4))

for i, guess in enumerate(guesses):
    new_candidates = set()
    b, c = bullscow[i]
    for s in candidates:
        b2, c2 = bullcow(s, guess)
        if b2==b and c2==c:
            new_candidates.add(s)
    candidates = new_candidates
    if len(candidates) == 0:
        print 'Incorrect data'
        sys.exit(0)

if len(candidates)>1:
    print 'Need more data'
    sys.exit(0)

print ''.join(candidates.pop())
