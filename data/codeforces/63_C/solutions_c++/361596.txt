#include<iostream>
#include<cstdio>
using namespace std;

int f[11][3];

int main()
{
    int n;cin>>n;
    for (int i=0; i<n; i++) cin>>f[i][0]>>f[i][1]>>f[i][2];
    int ans=0,t;
    for (int i=123;i<10000;i++){
        int a[4];a[0]=i/1000;a[1]=(i/100)%10;a[2]=(i/10)%10;a[3]=i%10;
        bool p=true;
        for (int j=0; j<3; j++){
            for (int k=j+1; k<4; k++) if (a[j]==a[k]) {p=false; break;}
            if (!p) break;
        }
        if (!p) continue;
        for (int j=0; j<n; j++){
            int x=0,y=0;
            int b[4];b[0]=f[j][0]/1000;b[1]=(f[j][0]/100)%10;b[2]=(f[j][0]/10)%10;b[3]=f[j][0]%10;
            for (int k=0;k<4;k++) for (int l=0;l<4;l++) if (a[k]==b[l]) if (k==l) x++; else y++;
            if((x!=f[j][1])||(y!=f[j][2])) {p=false;break;}
        }
        if (p) {ans++;t=i;if (ans>1) break;}
    }
    if(ans==0) cout<<"Incorrect data"<<endl; else if (ans==2) cout<<"Need more data"<<endl; else {
        if (t<1000) cout<<0;
        cout<<t<<endl;
    }
    return 0;
}
