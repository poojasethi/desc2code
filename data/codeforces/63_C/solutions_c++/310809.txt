const int INF=0x3f3f3f3f;

struct GUASS {
	char op[6];
	int a, b;
}g[20];

int cnt, key, t[4], n;
bool flag[10];
char op[13][5];

void chk() {
	for(int i=0; i<n; ++i) {
		int a=0, b=0;
		for(int j=0; j<4; ++j) {
			if( g[i].op[j]-'0'==t[j] )
				++a;
			else if( flag[ g[i].op[j]-'0' ] )
				++b;
		}

		if(g[i].a!=a || g[i].b!=b)
			return ;
	}

	++cnt;
	key=t[0]*1000+t[1]*100+t[2]*10+t[3];
}

void dfs(int dep) {
	if(dep==4)
		return chk();
	for(int i=0; i<10; ++i) {
		if(!flag[i]) {
			flag[i]=true;
			t[dep]=i;
			dfs(dep+1);
			flag[i]=false;
		}
	}
}

int main() {
#ifndef ONLINE_JUDGE
  //  freopen("in","r",stdin);
#endif

    scanf("%d", &n);
    for(int i=0; i<n; ++i) {
    	scanf("%s %d %d", g[i].op, &g[i].a, &g[i].b);
    }
    cnt=0;
    memset(flag, 0, sizeof(flag));
    dfs(0);

    if(cnt>1)
    	puts("Need more data");
    else if(!cnt)
    	puts("Incorrect data");
    else
    	printf("%04d\n", key);

    return 0;
}
