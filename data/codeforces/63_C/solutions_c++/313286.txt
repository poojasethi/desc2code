#include <iostream>
#include <cstdio>
#include <cstring>
using namespace std;

int N,t1,t2,tot,B[10],C[10];
char tmp[5],A[10][5],ans[5];
bool can,cur,M1[4],M2[4];

int main(){
	scanf("%d",&N);
	for(int i=0;i<N;++i) scanf(" %s %d %d",A[i],&B[i],&C[i]);
	for(char a='0';a<='9';++a)
		for(char b='0';b<='9';++b)
			for(char c='0';c<='9';++c)
				for(char d='0';d<='9';++d){
					if(a == b || a == c || a == d || b == c || b == d || c == d) continue;
					tmp[0] = a;
					tmp[1] = b;
					tmp[2] = c;
					tmp[3] = d;
					can = 1;
					for(int i=0;i<N;++i){
						t1 = 0;
						t2 = 0;
						for(int j=0;j<4;++j)
							for(int k=0;k<4;++k)
								if(tmp[j] == A[i][k]){
									if(j == k) ++t1;
									else ++t2;
								}
						if(t1 != B[i] || t2 != C[i]){
							can = 0;
							break;
						}
					}
					if(can == 1){
						++tot;
						for(int i=0;i<4;++i) ans[i] = tmp[i];
					}
				}
	if(tot == 0) printf("Incorrect data\n");
	else if(tot == 1) printf("%s\n",ans);
	else printf("Need more data\n");
}
