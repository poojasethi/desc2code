#include<iostream>
#include<cstdio>
#include<cstdlib>
#include<algorithm>

using namespace std;
void dig(int k,int a[4])
{
	for(int i=0;i<4;i++)
		{
			a[i]=k%10;
			k/=10;
		}
}
int main()
{
	int count=0,i,n,ss[15][3],t[4],sd[15][4],j,k,ok=1,bull,cow,l,val;
	cin>>n;
	for(i=0;i<n;i++)
	{
		cin>>ss[i][0]>>ss[i][1]>>ss[i][2];
		dig(ss[i][0],sd[i]);
	}
	for(i=100;i<9999;i++)
	{
		dig(i,t);
		ok=1;
		for(j=0;j<4;j++)
			for(k=j+1;k<4;k++)
				if(t[j]==t[k])
					ok=0;
		if(!ok)
			continue;
		for(j=0;j<n;j++)
		{
			bull=cow=0;
			for(k=0;k<4;k++)
				if(t[k]==sd[j][k])
					bull++;
			for(k=0;k<4;k++)
			{
				for(l=0;l<4;l++)
					if(t[k]==sd[j][l])
						cow++;
			}
			cow-=bull;
			if((cow!=ss[j][2])||(bull!=ss[j][1]))
			{
				ok=0;
				break;
			}
		}
		if(ok)
		{
			val=i;
			count++;
		}
		//if(i==123) cout<<count<<' '<<ok<<endl;
	}
	if(count==0) cout<<"Incorrect data\n";
	else if(count==1) {
		dig(val,t);
		for(i=3;i>=0;i--) cout<<t[i];
		cout<<endl;
	}
	else cout<<"Need more data\n";
	return 0;
}
