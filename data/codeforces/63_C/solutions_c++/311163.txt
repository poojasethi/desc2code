#define pb push_back

using namespace std;

struct game
{
	int b,k;
	string s;
};

int u[11],n;
vector<string>   rez;
game a[11];

bool check(string s,string t,int x)
{
	int k=0,i,j;

	for(i=0;i<4;i++)
		if(s[i]==t[i])  k++;

    if(k!=a[x].b) return 0;

    k=0;
    for(i=0;i<4;i++)
    	for(j=0;j<4;j++)
    		if((i!=j)&&(s[i]==t[j]))  k++;

    if(k==a[x].k)   return 1;

    return 0;
}

void f(int v,string s)
{
	int i,j,ok;

	if(v>4)
	{
		ok=1;
		for(j=0;j<n;j++)
			if(check(s,a[j].s,j)==0)			
			{
				ok=0;
				break;
			}

		if(ok) rez.pb(s);   
	}
	else
		for(i=0;i<=9;i++)
			if(u[i]==0)
			{
				u[i]=1;
				f(v+1,s+(char)(i+48));
				u[i]=0;
			}
}

int main()
{	
	int i;

	cin>>n;
	for(i=0;i<n;i++)
		cin>>a[i].s>>a[i].b>>a[i].k;

	f(1,"");
	
	if(rez.size()==0)  cout<<"Incorrect data";
	else
		if(rez.size()==1)   cout<<rez[0];
		else cout<<"Need more data";

	return 0;
}
