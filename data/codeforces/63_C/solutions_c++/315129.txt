#include<stdio.h>
#include <stdlib.h>

int a[20] ,b[20] , c[20] ,n;

int check(int k)
{ int i ,bb ,cc;
  int q ,j, g  ,aa ,kk,h, w[10]={0};
  q=k; i=4;
  
  while (i>0)
  { j=q%10;
    q/=10;
    if (w[j]==0 ) w[j]=1;
    else return 0;
    i--;
  }
   
  for (i=1 ; i<=n ;  i++)
  { aa=a[i]; j=4; bb=0 ; cc=0;
      while (j>0)
      { g=4; h=aa%10; aa/=10; kk=k;
        while (g>0)
         { q=kk%10;
           kk/=10;
           if (q==h && g==j) bb++;
           else if (q==h) cc++;
           g--; 
         } j--;
      }
    if (bb!=b[i] || cc!=c[i])  return 0;
  }
  
return 1;
}

int main ()
{  int i ,p,ans=0;
    scanf("%d",&n);
    
    for (i=1 ; i<=n ; i++)
    scanf("%d%d%d",&a[i],&b[i],&c[i]); p=0;

for (i=0; i<=9999 ; i++)
  if (check(i)) { p++; ans=i; }

if (p>1)    printf("Need more data");
else if (p==0) printf("Incorrect data");
else 
{ 
if (ans/1000==0) printf("0");
printf("%d",ans);
}
// system("pause");
}
