#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int vis[20];
    int a[20],b[20];
    string str[20];
int ans=0;
string tt;
int n ;
void dfs(int deep,string s){
    if(deep == 4){
        
          
        for(int i=0;i<n;i++){
            int cnt1=0,cnt=0;
            for(int j=0;j<4;j++){
                if(s[j]==str[i][j])cnt++;
                else if(s.find(str[i][j])!=-1)cnt1++;
            }
            if(cnt!=a[i] || cnt1!=b[i])return;
        }
       // cout<<s<<endl;
        ans++;
       
        tt=s;
            
        
    }
    for(int i=0;i<10;i++){
        if(vis[i])continue;
        else {
            vis[i]=1;
            string t;
            t=s+char(i+'0');
            dfs(deep+1,t);
            vis[i]=0;
        }
    }
}
int main(){
 
    cin>>n;

    for(int i=0;i<n;i++){
        cin>>str[i]>>a[i]>>b[i];
    }
    memset(vis,0,sizeof(vis));
    ans = 0;
    dfs(0,"");
    
    if(ans == 0){
        cout<<"Incorrect data" <<endl;
    }
    else if(ans ==1){
        cout<<tt<<endl;
    }
    else cout << "Need more data" <<endl;
    //cin>>n;
}
