#include <stdio.h>
#include <string.h>
#include <algorithm>

using namespace std;
int getdig(int num, int p){
	int mod = 1;
	while( p-- ) mod *= 10;
	return num/mod%10;
}

int digapr[10];
int mch[10][2], tnum[4], tocmp[10][10];
int n;
int navl, nwavl;
int chk_num(int num){
	for( int i=0; i<10; i++ ) digapr[i] = 0;
	for( int i=0; i<4; i++ ){
		if( digapr[getdig(num, i)] )
			return 0;
		digapr[getdig(num, i)] = i+1;
	}
	return 1;
}
int avail(int p){
	int c, b;
	c = b = 0;
	for( int i=0; i<10; i++ )
		if( tocmp[p][i]==digapr[i]&&tocmp[p][i] )
			c++;
		else if( tocmp[p][i]&&digapr[i] )
			b++;
	return c==mch[p][0] && b==mch[p][1];
}
int main(){
	int tmp, flg;
	scanf("%d", &n);
	for( int i=0; i<n; i++ ){
		scanf("%d %d %d", &tmp, &mch[i][0], &mch[i][1]);
		for( int j=0; j<4; j++ )
			tocmp[i][getdig(tmp, j)] = j+1;
	}
	for( int i=0; i<10000; i++ ){
		if( chk_num(i) ){
			flg = 1;
			for( int j=0; j<n; j++ )
				if( !avail(j) ){
					flg = 0;
					break;
				}
			if( flg )
				navl++, nwavl=i;
		}
	}
	if( navl==1 )
		printf("%04d\n", nwavl);
	else if( navl==0 )
		puts("Incorrect data");
	else
		puts("Need more data");
	return 0;
}
