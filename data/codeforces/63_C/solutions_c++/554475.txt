#include <cstdio>
#include <iostream>
#include <cstdlib>
#include <algorithm>
#include <cmath>
#include <cstring>
using namespace std;
int dp[10005];
int check(int m,char s[],int k1,int k2)
{
	char r[6];
	sprintf(r,"%04d",m);
	int i,j;
	for(i=0;i<4;i++)
		for(j=0;j<4;j++)
			if(s[i]==r[j])
			{
				if(i==j)
					k1--;
				else
					k2--;
			}
	if(k1==0&&k2==0)
		return 1;
	else
		return 0;
}
int main()
{
	int n;
	while(scanf("%d",&n)==1)
	{
		memset(dp,0,sizeof(dp));
		int i,j,k,l;
		for(i=0;i<10;i++)	
			for(j=0;j<10;j++)
				for(k=0;k<10;k++)
					for(l=0;l<10;l++)
						if(i==j||i==k||i==l||j==k||j==l||k==l)
							dp[i*1000+j*100+k*10+l]=1;
		for(i=0;i<n;i++)
		{
			char s[6];
			int b,c;
			scanf("%s%d%d",&s,&b,&c);
			int j;
			for(j=0;j<10000;j++)
				if(!check(j,s,b,c))
					dp[j]=1;
		}
		int ans,flag=0;
		for(i=0;i<10000;i++)
			if(!dp[i])
			{
				flag++;
				ans=i;
			}
		if(!flag) 
			printf("Incorrect data\n");
		else if(flag==1)
		{
			if(ans<1000)
				printf("0");
			printf("%d\n",ans);
		}
		else
			printf("Need more data\n");
	}
	return 0;
}
