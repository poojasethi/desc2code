
#include<cstdio>
int b[10],c[10],i,j,k,m,n,A,B,u,v,w,cnt;
char a[10][5],ans[5];
int main(){
next:	while(scanf("%d",&n)==1){
		for(i=0;i<n;++i){
			scanf("%s%d%d",a[i],&b[i],&c[i]);
			a[i][0]-='0';
			a[i][1]-='0';
			a[i][2]-='0';
			a[i][3]-='0';
		}
		cnt=0;
		for(i=0;i<10;++i){
			for(j=0;j<10;++j){
				if(j!=i)
				for(k=0;k<10;++k){
					if(k!=i&&k!=j)
					for(m=0;m<10;++m){
						if(m==i||m==j||m==k)continue;
						if(i==2&&j==1&&k==3&&m==4){
							i=2; 
						}
						for(u=0;u<n;++u){
							A=B=0;
							if(a[u][0]==i)++A;
							if(a[u][1]==j)++A;
							if(a[u][2]==k)++A;
							if(a[u][3]==m)++A;
							if(A!=b[u])break;
							if(i==a[u][1]||i==a[u][2]||i==a[u][3])++B;
							if(j==a[u][0]||j==a[u][2]||j==a[u][3])++B;
							if(k==a[u][1]||k==a[u][0]||k==a[u][3])++B;
							if(m==a[u][1]||m==a[u][2]||m==a[u][0])++B;
							if(B!=c[u])break;
						}
						if(u==n){
							++cnt;
							if(cnt>1){
								puts("Need more data");
								goto next;
							}
							ans[0]=i+'0';
							ans[1]=j+'0';
							ans[2]=k+'0';
							ans[3]=m+'0';
						}
					}
				}
			}
		}
		if(cnt==0)puts("Incorrect data");
		else puts(ans);
	}
	return 0;
}
