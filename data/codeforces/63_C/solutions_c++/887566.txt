#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;

int r[10005];

bool cal(int D, int X, int a, int b) {
	int t[10]={};
	b += a;
	for(int i=0, v=1;i<4;i++, v*=10) {
		if(D/v%10 == X/v%10) --a;
		else { t[D/v%10]++; t[X/v%10]--; }
	}
	int s=0;
	for(int i=0;i<10;i++)
		if(t[i]>0) ++s;
	b -= (4-s);
	return a==0 && b==0;
}

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<10000;i++)
		for(int j=1;j<=1000;j*=10)
			for(int k=j*10;k<=1000;k*=10)
				if(i/j%10==i/k%10) r[i]=1;
	for(int i=0;i<n;i++) {
		int D, a, b;
		scanf("%d%d%d", &D, &a, &b);
		for(int j=0;j<10000;j++)
			if(cal(D, j, a, b)==false)
				r[j]=1;
	}
	int c=0, p=0;
	for(int i=0;i<10000;i++)
		if(r[i]==0) { ++c; p=i; }
	if(c==0) puts("Incorrect data");
	else if(c>1) puts("Need more data");
	else printf("%04d\n", p);
	return 0;
}
