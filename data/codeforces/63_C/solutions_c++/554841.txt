#include <stdio.h>
int a[11][5],b[11],c[11],d[5],n;
int main()
{ int x,i,j,k1,k2,bull,cow,ans,y;
  bool fg,boo;
  scanf("%d",&n);boo=false;
  for (i=1;i<=n;i++)
  { scanf("%d%d%d",&x,&b[i],&c[i]);
    a[i][1]=x%10;
    x=x/10;
    a[i][2]=x%10;
    x=x/10;
    a[i][3]=x%10;
    x=x/10;
    a[i][4]=x;
  }
  for (y=102;y<=9876;y++)
  { x=y;
    d[1]=x%10;
    x=x/10;
    d[2]=x%10;
    x=x/10;
    d[3]=x%10;
    x=x/10;
    d[4]=x;   
    if ((d[1]==d[2])||(d[1]==d[3])||(d[1]==d[4])||(d[2]==d[3])||(d[2]==d[4])||(d[3]==d[4]))
    continue;
    fg=true;
    for (j=1;j<=n;j++)
     {  bull=0;cow=0;
        for (k1=1;k1<=4;k1++)
        for (k2=1;k2<=4;k2++)
         {if (d[k1]==a[j][k2])  {if (k1==k2) bull++;
                                 else cow++;
                                }
         }
       if ((bull!=b[j]) || (cow!=c[j])) {fg=false;break;}
     }
   if (fg) {if (boo) {printf("Need more data\n");return 0;} boo=true;ans=y;}
  }
  if (boo) {if (ans<1000) printf("0");printf("%d\n",ans);} else printf("Incorrect data\n");
  return 0;
}
