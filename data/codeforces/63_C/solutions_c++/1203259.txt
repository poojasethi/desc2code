#include <stdio.h>
#include <string.h>
int mark[10];
char ss[10];
char anslist[10];
int ans;
struct CCmd
{
	char pe[10];
	int same,app;
}cmd[100];
int n;
void DFS(int cur) {
	int i,j;
	if(cur == 4) {
		int same,app;
		for(i=0;i<n;i++) {
			same = app = 0;
			for(j=0;j<4;j++) {
				if(cmd[i].pe[j] == ss[j])same ++;
			}
			for(j=0;j<4;j++) {
				int k;
				for(k=0;k<4;k++) {
					if(j==k)continue;
					if(cmd[i].pe[j] == ss[k]) {
						app++;
						break;
					}
				}
			}
			if(!(same == cmd[i].same && app == cmd[i].app)) {
				return;
			}
		}
		ans++;
		for(i=0;i<4;i++)anslist[i] = ss[i];
		return;
	}
	for(i=0;i<10;i++) {
		if(mark[i])continue;
		mark[i] = 1;
		ss[cur] = i+'0';
		DFS(cur+1);
		mark[i] = 0;
	}
	return;
}
int main()
{
	int i,j;
	
	while(scanf("%d",&n)!=EOF) {
		for(i=0;i<n;i++) {
			scanf("%s%d%d",cmd[i].pe,&cmd[i].same,&cmd[i].app);
		}
		ans = 0;
		for(i=0;i<10;i++) {
			mark[i] = 0;
		}
		DFS(0);
		if(ans == 0) {
			printf("Incorrect data\n");
		}
		else if(ans == 1) {
			for(i=0;i<4;i++)printf("%c",anslist[i]);printf("\n");
		}
		else {
			printf("Need more data\n");
		}
	}
	return 0;
}
