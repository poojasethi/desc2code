#include<cstdio>
#include<cstring>
int b[15];
int c[15];
char g[15][6];
bool used[10]={0};
int m;
char sol[10000][5];
bool ok[10000];
char a[6];
bool cumple(int x,int y){
	int B = 0;
	int C = 0;
	for (int i=0; i<4; ++i){
		if (g[x][i] == sol[y][i]) B++;
		else if (g[x][0] == sol[y][i] || g[x][1] == sol[y][i] || g[x][2] == sol[y][i] || g[x][3] == sol[y][i])
				C++;
	}
	return B==b[x] && C==c[x];
}

void generate_seed(int p){
	if (p == 4){
		a[p] ='\0';
		strcpy(sol[m],a);
		ok[m] = true;
		m++;
		return;
	}
	for (int i=0; i<10; ++i)
		if (!used[i]){
			a[p] = (char)('0' + i);
			used[i] = true;
			generate_seed(p+1);
			used[i] = false;
		}
}
int main(){
	int n;
	int w;
	m = 0;
	scanf("%d",&n);
	bool incorrect = false;
	for (int i=0; i < n; ++i){
		scanf("%s %d %d",g[i],b+i,c+i);
	}
	generate_seed(0);
	int tmp = m;
	for (int i=0; i < n && tmp > 0; ++i){
		for (int j=0; j < m; ++j){
			if (ok[j]){
				if (!cumple(i,j)){ ok[j] = false; tmp--; }
			}
		}
	}
	if (tmp==0) puts("Incorrect data");
	else if (tmp > 1) puts("Need more data");
	else {
		for (int i=0; i < m; ++i) if (ok[i]){ puts(sol[i]); break; }
	}
	return 0;
}
