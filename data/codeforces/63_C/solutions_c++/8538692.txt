#include <iostream>
#include <cstdlib>
#include <cstdio>
#include <stdio.h>
using namespace std;
int main(){
  //freopen("input.txt","r",stdin);
  //freopen("output.txt","w",stdout);
  int q,w,e,r,t=0,c,v,a[10],s[10],d[4],f[10],h[4];
  char g[10][5];
  cin>>q;
  for(w=0;w<q;w++){
   cin>>g[w]>>a[w]>>s[w];
   for(e=0;e<4;e++)
    g[w][e]-=48;}
  for(w=123;w<10000;w++){
   r=w;
   for(e=0;e<10;e++)
    f[e]=0;
   for(e=0;e<4;e++){
    d[e]=r%10;
    f[d[e]]++;
    r/=10;}
   for(e=0;e<10;e++)
    if(f[e]>1)
     goto br;
   for(e=0;e<q;e++){
    c=v=0;
    for(r=0;r<4;r++)
     if(d[r]==g[e][r])
      c++;
     else
      if(f[g[e][r]])
       v++;
    if((a[e]!=c)||(s[e]!=v))
     goto br;}
   t++;
   for(e=0;e<4;e++)
    h[e]=d[e];
   br:;}
  if(t>1)
   cout<<"Need more data";
  if(t==1)
   for(e=0;e<4;e++)
    cout<<h[e];
  if(t==0)
   cout<<"Incorrect data";
  return 0;}
