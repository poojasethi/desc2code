#include<cstdio>
int n,t[11][3],p[4],q[4];
//                 Last Change:  2011-07-13 20:41:54
void fenjie(int x,int *r)
{
	r[3]=x%10;
	x/=10;
	r[2]=x%10;
	x/=10;
	r[1]=x%10;
	x/=10;
	r[0]=x;
}
int calc(int a,int b,int &x,int &y)
{
	fenjie(a,p);
	fenjie(b,q);
	x=y=0;
	for(int i=0;i<4;++i)
		if(p[i]==q[i])
			++x;
	for(int i=0;i<4;++i)
		for(int j=0;j<4;++j)
			if(i!=j&&p[i]==q[j])
				++y;
}
bool check(int u)
{
	if(u/1000==u/100%10)return false;
	if(u/1000==u/10%10)return false;
	if(u/1000==u%10)return false;
	if(u/100 %10==u/10%10)return false;
	if(u/100 %10==u%10)return false;
	if(u/10%10==u%10)return false;
	for(int i=1;i<=n;++i)
	{
		int x,y;
		calc(u,t[i][0],x,y);
		if(x!=t[i][1]||y!=t[i][2])
			return false;
	}
	return true;
}
int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;++i)
		scanf("%d%d%d",t[i],t[i]+1,t[i]+2);
	int ans=0,s1;
	for(int i=0;i<10000;++i)
		if(check(i))
			++ans,s1=i;
	if(ans==1)printf("%04d\n",s1);
	else if(ans>1)puts("Need more data");
	else puts("Incorrect data");
	return 0;
}
