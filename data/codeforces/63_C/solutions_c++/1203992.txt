#include<stdio.h>
#include<string.h>
struct node{
	char c[6];
	int a,b;
}dd[40];
int main()
{
	char ans[5],ddd[5];
    int n,i,j,k,t,m,h,mark,s,aa,bb,hhhh;
    while(scanf("%d",&n)!=EOF)
    {
 		for(i=0;i<n;i++)
 		scanf("%s%d%d",dd[i].c,&dd[i].a,&dd[i].b);mark=0;
 		for(i=0;i<10;i++)
 		{
		 	ans[0]=i+'0';
		 	for(j=0;j<10;j++)
		 	{
	 			ans[1]=j+'0';if(ans[0]==ans[1])continue;
	 			for(k=0;k<10;k++)
	 			{
			 		ans[2]=k+'0';if(ans[0]==ans[2]||ans[2]==ans[1])continue;
			 		for(t=0;t<10;t++)
			 		{
		 				ans[3]=t+'0';if(ans[0]==ans[3]||ans[3]==ans[1]||ans[3]==ans[2])continue;hhhh=1;
		 				for(m=0;m<n;m++)
		 				{
		 					aa=0;bb=0;
				 			for(s=0;s<4;s++)
				 			{
			 					for(h=0;h<4;h++)
			 					{
					 				if(s==h&&dd[m].c[s]==ans[h])aa++;
					 				else if(s!=h&&dd[m].c[s]==ans[h])bb++;
					 			}
			 				}
			 				if(!(aa==dd[m].a&&bb==dd[m].b)){hhhh=0;break;}
				 		}
				 		if(hhhh){
		 					mark++;ans[4]='\0';strcpy(ddd,ans);
		 				}
		 			}
			 	}
	 		}
		 }
 		
		 if(mark==0)printf("Incorrect data\n");
		 else if(mark==1)printf("%s\n",ddd);
		 else printf("Need more data\n");
 	}
}
