#include<iostream>
using namespace std;
int arr[4]={0,1,2,3};
int data[6][10];
int n;
bool plusch(void);
void one(void);
bool check(void);
int main(){
	int counter=0;
	int ans[4];
	cin>>n;
	for(int i=0;i<n;i++){
		string a;
		cin>>a;
		for(int j=0;j<4;j++)
			data[i][j]=int(a[j]-'0');
		cin>>data[i][4]>>data[i][5];
	}
	while(arr[0]!=9||arr[1]!=8||arr[2]!=7||arr[3]!=6){
		if(check()){
			counter++;
			for(int i=0;i<4;i++)
				ans[i]=arr[i];
		}
		one();
	}
	if(counter==0)
		cout<<"Incorrect data"<<endl;
	else if(counter==1)
		for(int i=0;i<4;i++)
			cout<<ans[i];
	else
		cout<<"Need more data"<<endl;
}
bool check(void){
	for(int i=0;i<n;i++){
		int x=0,y=0;
		for(int j=0;j<4;j++){
			if(arr[j]==data[i][j])
				x++;
		}
		for(int j=0;j<4;j++)
			for(int p=0;p<4;p++)
				if(arr[j]==data[i][p]&&j!=p)
					y++;
		if(x!=data[i][4]||y!=data[i][5])
			return false;
	}
	return true;
}
bool rptsch(void){
	for(int i=0;i<3;i++)
		for(int j=i+1;j<4;j++)
			if(arr[i]==arr[j])
				return true;
	return false;
}
void one(void){
	do {
		int w=3;
		while(arr[w]==9){
			arr[w]=0;
			w--;
		}
		arr[w]++;
	}while(rptsch());
}
