#include<stdio.h>
#include<string.h>
struct node{
	char a[10];
	int x,y;
}s[100];
int num,cout,n;
char zhp[100];
int flage[20];
char ans[10];
int fit(int k){
	int i,j;
	int x=0,y=0;
	for(i=0;i<4;i++){
		if(zhp[i]==s[k].a[i]){
			x++;
		}
	}
	for(i=0;i<4;i++){
		for(j=0;j<4;j++){
			if(zhp[i]==s[k].a[j])y++;
		}
	}
	y=y-x;
	if(s[k].x==x&&s[k].y==y)return 1;
	return 0;
}
int check(){
	int i;
	for(i=0;i<n;i++){
		if(fit(i)){
			continue;
		}
		break;
	}
	if(i<n)return 0;
	return 1;
}
int dfs(int step){
	cout++;
	
	int i;
	if(step==4){
		if(check()){
			num++;
			for(i=0;i<4;i++)
			ans[i]=zhp[i];
			ans[i]='\0';
		}
		return 0;
	}
	for(i=0;i<=9;i++){
		 if(flage[i]==0){
		 	zhp[step]=i+'0';
		 	flage[i]=1;
		 	dfs(step+1);
		 	flage[i]=0;	
		 }
		 else continue;
	}
	return 0;
}
int main(){
	int i,j;
	char str[10],xx,yy;
	while(scanf("%d",&n)!=EOF){
		num=0,cout=0;
		memset(flage,0,sizeof(flage));
		for(i=0;i<n;i++){
			scanf("%s%d%d",s[i].a,&s[i].x,&s[i].y);
		}
		dfs(0);
		if(num==0)printf("Incorrect data\n");
		else if(num==1) printf("%s\n",ans);
		else printf("Need more data\n");
	}
	return 0;
}
