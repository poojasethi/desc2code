#include<stdio.h>
#include<memory.h>

int N;
int how;
int Ans[4];
int inp[10][4];
int inq[10][2];

FILE *in = stdin;
FILE *out = stdout;

int main(void){
	int i,j,k;
	int a,b,c,d,p[4];
	char imsi[6];
	fscanf(in,"%d\n",&N);
	for(i=0;i<N;i++){
		fscanf(in,"%s %d %d\n",&imsi,&inq[i][0],&inq[i][1]);
		for(j=0;j<4;j++){
			inp[i][j]=imsi[j]-'0';
		}
	}
	for(a=0;a<10;a++){
		for(b=0;b<10;b++){
			if(a==b)continue;
			for(c=0;c<10;c++){
				if(a==c||b==c)continue;
				for(d=0;d<10;d++){
					if(a==d||b==d||c==d)continue;
					p[0]=a,p[1]=b,p[2]=c,p[3]=d;
					for(i=0;i<N;i++){
						for(j=0,k=0;j<4;j++){
							if(p[j]==inp[i][j])k++;
						}
						if(k!=inq[i][0])break;
						for(j=0,k=0;j<4;j++){
							if(p[j]!=inp[i][j]&&(inp[i][j]==a||inp[i][j]==b||inp[i][j]==c||inp[i][j]==d))k++;
						}
						if(k!=inq[i][1])break;
					}
					if(i==N){
						how++;
						if(how==2){
							fprintf(out,"Need more data\n");
							return 0;
						}
						memcpy(Ans,p,sizeof(Ans));
					}
				}
			}
		}
	}
	if(how==0){
		fprintf(out,"Incorrect data\n");
		return 0;
	}for(i=0;i<4;i++){
		fprintf(out,"%d",Ans[i]);
	}fprintf(out,"\n");
	return 0;
}
