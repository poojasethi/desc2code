#include <iostream>
#include <string>
#include <utility>

using namespace std;

pair<int, int> cnt(const string &s, const string &q)	{
	int c = 0, b = 0;
	for (int i = 0; i < 4; i++)
		for (int j = 0; j < 4; j++)
			if (s[i] == q[j])
				if (i == j)
					b++;
				else
					c++;
	return make_pair(b, c);
}

int main() {
	int n;
	cin >> n;
	string query[10], s, ans;
	s.resize(4);
	int cows[10];
	int bulls[10];
	for (int i = 0; i < n; i++)
		cin >> query[i] >> bulls[i] >> cows[i];
	int count = 0;
	for (int a = 0; a < 10; a++)
		for (int b = 0; b < 10; b++)
			if (a != b)
			for (int c = 0; c < 10; c++)
				if (a != c && b != c)
				for (int d = 0; d < 10; d++)
					if (a != d && b != d && c != d)	{
						s[0] = a+'0';
						s[1] = b+'0';
						s[2] = c+'0';
						s[3] = d+'0';
						bool fail = false;
						for (int i = 0; !fail && i < n; i++) {
							pair<int, int> p = cnt(s, query[i]);
							if (p.first != bulls[i] || p.second != cows[i])
								fail = true;
						}
						if (!fail) {
							count++;
							ans = s;
						}
					}
	if (count == 0)
		cout << "Incorrect data" << endl;
	else if (count == 1)
		cout << ans << endl;
	else
		cout << "Need more data" << endl;
	return 0;
}
