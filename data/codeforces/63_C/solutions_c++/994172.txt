#include <iostream>
#define rep(i,n) for(int i=0;i<n;i++)
#define pr(i) cout<<i<<endl
using namespace std;

int n,b,c,t,mb,mc,t1,t2,t3,a[10000],r,cn,d[4],d0[4];

int main(){
  
  cin>>n;
  rep(k,10000){
    int b=1;t=k;
    rep(i,4){d[i]=t%10;t/=10;}
    rep(i,4)for(int j=i+1;j<4;j++)if(d[i]==d[j])b=0;
    a[k]=b;
  }

  rep(i,n){
    cin>>t1>>b>>c;
    rep(j,4){d0[j]=t1%10;t1/=10;}
    rep(j,10000){
      if(a[j]==0)continue;
      t=j;mb=0;mc=0;
      rep(k,4){d[k]=t%10;t/=10;}
      rep(k,4)if(d[k]==d0[k])mb++;
      rep(k,4)rep(l,4)if(d[k]==d0[l])mc++;
      mc-=mb;
      if(mb!=b||mc!=c)a[j]=0;
    }
  }
  rep(i,10000)if(a[i]==1){cn++;r=i;}
  if(!cn)pr("Incorrect data");
  else if(cn==1){if(r<1000)cout<<"0";pr(r);}
  else pr("Need more data");

}
