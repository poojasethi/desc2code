#include<stdio.h>
#include<algorithm>

using std::sort;
using std::lower_bound;
using std::upper_bound;

const int MAXN=100000+9;
const long long MOD=1000000007;

int pow(long long a,int b)
{
	long long ans=1;
	for(;b;b>>=1)
	{
		if(b&1)
			ans=ans*a%MOD;
		a=a*a%MOD;
	}
	return ans;
}

int a[MAXN],d[MAXN];
int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	int n,m=0,i,j;
	scanf("%d",&n);
	for(i=0;i<n;++i)
	{
		scanf("%d",a+i);
		if(a[i]>m)
			m=a[i];
	}
	sort(a,a+n);
	int ans=0;
	for(i=1;i<=m;++i)
	{
		int nd=0;
		for(j=1;j*j<i;++j)
			if(!(i%j))
				d[++nd]=j;
		if(j*j==i)
			d[++nd]=j;
		for(--j;j;--j)
			if(!(i%j))
				d[++nd]=i/j;
		long long t=1;
		for(j=1;j<nd;++j)
			t=t*pow(j,lower_bound(a,a+n,d[j+1])-lower_bound(a,a+n,d[j]))%MOD;
		int k=a+n-lower_bound(a,a+n,i);
		t=t*(pow(j,k)-pow(j-1,k)+MOD)%MOD;
		ans=(ans+t)%MOD;
	}
	printf("%d\n",ans);
	return 0;
}
