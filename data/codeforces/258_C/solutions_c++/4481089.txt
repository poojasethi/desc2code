#include<cstdio>
#include<cstdlib>
#include<vector>
#include<cstring>
#include<algorithm>
#define int64 long long
using namespace std;
const int mo=1000000007;
int res,p[120000],x,a[120000],n,i,j,ans,m;
vector <int> v[120000];
int64 pow(int64 x,int64 y){
	int64 res=1;
	while(y){
		if(y&1)res=res*x%mo;
		x=x*x%mo;
		y>>=1;
	}
	return res;
}
int main(){
	scanf("%d",&n);
	for(i=1;i<=n;++i)scanf("%d",&a[i]),m=max(m,a[i]);
	for(i=1;i<=m;++i)
		for(j=i;j<=m;j+=i)v[j].push_back(i);
	sort(a+1,a+n+1);
	for(i=1;i<=m;++i){
		res=1;
		for(j=0;j<v[i].size();++j)p[j]=lower_bound(a+1,a+n+1,v[i][j])-a;
		p[j]=n+1;
		for(j--;j>=0;--j){
			x=pow(j+1,p[j+1]-p[j]);
			if(v[i][j]==i)x-=pow(j,p[j+1]-p[j]);
			res=(int64)res*x%mo;
		}
		ans=(ans+res)%mo;
	}
	printf("%d\n",(ans+mo)%mo);
}
