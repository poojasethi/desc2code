#include<stdio.h>
#include<stdlib.h>
#include<iostream>
#include<algorithm>
#define N 210000
#define MOD 1000000007
#define int64 long long
#define foru(i,a,b) for(int i=a;i<=b;++i)

using namespace std;
int a[N],num[N],now[N],n,m,l;
int64 ans,f[N];
int64 sqr(int64 a)
{
	return (a*a)%MOD;
}
int64 exp(int64 a,int64 b)
{
	if(!b) return 1;
	return (sqr(exp(a,b>>1))*(b&1?a:1))%MOD;
}
int main()
{
	//freopen("input.txt","r",stdin);
	//freopen("output.txt","w",stdout);
	cin>>n;
	foru(i,1,n) cin>>a[i];
	sort(a+1,a+n+1);
	l=1; m=a[n];
	foru(i,1,m) f[i]=1,num[i]=1,now[i]=1;
	ans=1;
	foru(i,2,m)
	{
		while(a[l]<i) ++l;
		for(int j=1;j*i<=m;++j)
		  f[j*i]*=exp(num[j*i],l-now[j*i]),
		  f[j*i]%=MOD,++num[j*i],now[j*i]=l;
		f[i]*=exp(num[i],n-l+1)-exp(num[i]-1,n-l+1);
		f[i]=(f[i]%MOD+MOD)%MOD; ans+=f[i]; ans%=MOD;
		//cout<<i<<' '<<f[i]<<' '<<num[i]<<endl;
	}
	cout<<ans<<endl;
	return 0;
}
