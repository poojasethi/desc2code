a, b, c, l = map(int, raw_input().split())

def f(a, b, c, l):
	half = (a + b + c + l) / 2
	rest = half - b - c
	if rest < 0: return 0
	if rest > l: rest = l
	return (rest + 2) * (rest + 1) / 2

ans = 0
for i in xrange(l + 1):
	ans += (i + 2) * (i + 1) / 2
	ans -= f(a, b, c, i)
	ans -= f(b, a, c, i)
	ans -= f(c, a, b, i)
print ans
