def f(n):
  if n < 0:
    return 0
  return (n + 1) * (n + 2) / 2


def calc(a, abc, d):
  res = 0
  bc = abc - a
  for i in range(0, d + 1):
    diff = a + i - bc
    if diff >= 0:
      res += f(min(diff, d - i))
  return res


numbers = map(int, raw_input().split())
a = numbers[0]
b = numbers[1]
c = numbers[2]
d = numbers[3]

total = 0
for i in range(0, d + 1):
  total += f(d - i)

s = a + b + c
print total - calc(a, s, d) - calc(b, s, d) - calc(c, s, d)
