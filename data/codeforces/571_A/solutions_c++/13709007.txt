#include<bits/stdc++.h>
using namespace std;
typedef long long int64;
int a,b,c,L;

int64 f(int a,int b,int c){
	int64 ans=0;
	for(int i=0;i<=L;++i){
		int t=min(L-i,c+i-a-b);
		if(t>=0) ans+=1LL*(t+1)*(t+2)/2;
	}
	return ans;
}

int main(){
	cin>>a>>b>>c>>L;
	cout<<1LL*(L+1)*(L+2)*(L+3)/6-f(a,b,c)-f(b,c,a)-f(c,a,b)<<endl;
	return 0;
}
