#include <iostream>
#include <cstdio>

using namespace std;

long long func(long long x)
{
    if(x<0)
        return 0;
    return ((x+2)*(x+1))/2;
}

int main()
{
    long long a,b,c,l,i;
    cin>>a>>b>>c>>l;
    long long ans,temp=0;
    ans=(l+3)*(l+2)*(l+1);
    ans/=6;
    for(i=0;i<=l;i++){
        temp+=func(min(a-b-c+i,l-i));
        temp+=func(min(b-a-c+i,l-i));
        temp+=func(min(c-a-b+i,l-i));
    }
    cout<<ans-temp<<endl;
    return 0;
}
