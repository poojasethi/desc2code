#include <bits/stdc++.h>
using namespace std;
long long int no(long long int x,long long int y,long long int z,long long int l) {
	long long int c = (x+y+z+l)/2;
	long long int b = c-y-z;
	if(b<0) return 0;
	if(b>l) b = l;
	return ((b+2)*(b+1))/2;
}
int main() {
	long long int i,a,b,c,n=0,l;
	cin>>a>>b>>c>>l;
	for(i=1;i<=l+1;i++) {
		n += (i*(i+1))/2;
		n -= no(a,b,c,i-1);
		n -= no(b,c,a,i-1);
		n -= no(c,a,b,i-1);
	}
	cout<<n<<endl;
	return 0;
}
