#include <bits/stdc++.h>
using namespace std;
long long sol(int a,int b, int c,int l)
 { long long res=0;
   for(int i=max(0,b+c-a);i<=l;++i)
    { long long m=min(l-i,a+i-b-c);
      res+=(m+1)*(m+2)/2;
    }
     return res;
 }
int main()
 { long long a,b,c,l;
   cin >> a >> b >> c >> l;
   long long k=(l+1)*(l+2)*(l+3)/6;
   cout << k-sol(a,b,c,l)-sol(b,c,a,l)-sol(c,a,b,l);
    return 0;
 }
