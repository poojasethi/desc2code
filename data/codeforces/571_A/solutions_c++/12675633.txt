#include<iostream>
using namespace std;

int main(){
	long long u,ans,a,b,c,l;
	cin >> a >> b >> c >> l;
	ans=(l+3)*(l+2)*(l+1)/6;
	for(int x=0; x<=l; x++){
		u=min(l-x,a-b-c+x);
		if(a+x<b+c)u=-1;
		ans-=(u+2)*(u+1)/2;
		u=min(l-x,b-a-c+x);
		if(b+x<a+c)u=-1;
		ans-=(u+2)*(u+1)/2;
		u=min(l-x,c-b-a+x);
		if(c+x<b+a)u=-1;
		ans-=(u+2)*(u+1)/2;
		
	}
	cout << ans << endl;
	return 0;
}
