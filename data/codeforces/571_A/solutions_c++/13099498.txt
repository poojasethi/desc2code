#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll dem1( int a, int b, int c, int l)
{
     ll x=max(0,b+c-a);
     ll ans=0;
     ll m;
    for(int i=x;i<=l;i++)
    {
     m=min(l-i,a+i-b-c);

     ans=ans+ll(m+2)*(m+1)/2;
    }
    return ans;
}
int main()
{
    int a,b,c,l;
    cin>>a>>b>>c>>l;
    ll k=ll(l+2)*(l+1)*(l+3)/6;
    ll x=dem1(a,b,c,l);
    ll y=dem1(b,c,a,l);
    ll z=dem1(c,a,b,l);

    cout<<k-x-y-z;

}
