#include <bits/stdc++.h>
using namespace std;
long long  sol(int a, int  b, int  c, int  l) {
	long long res=0;
	for(int i=max(0,b+c-a);i<=l;++i)
	{
	long long m=min(l-i,a+i-b-c);

		res+= (m+1)*(m+2)/2;
	}
	return res;
	}
int main() {
	long long int i,a,b,c,n=0,l;
	cin>>a>>b>>c>>l;
	long long int e=(l+1)*(l+2)*(l+3)/6;
	long long int m1=sol(a,b,c,l);
	long long int m2=sol(b,a,c,l);
	long long int m3=sol(c,a,b,l);
	cout<<e-m1-m2-m3<<endl;
	return 0;
}
