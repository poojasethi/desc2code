#include<bits/stdc++.h>
using namespace std;

typedef long long ll;

ll bad(ll a,ll b,ll c,ll rem){
    if(a < b + c) return 0;
    ll cur = min(rem,a - b - c);
    return (cur+1)*(cur+2)/2;
}

int main(){
    ll a,b,c,d = 0,l; cin >> a >> b >> c >> l;
    d = (l+1)*(l+2)*(l+3)/6;
    for(int i=0;i<=l;i++){
        d -= bad(a+i,b,c,l-i) + bad(b+i,c,a,l-i) + bad(c+i,a,b,l-i);
    }
    cout << d << '\n';
    return 0;
}
