#include<cstdio>
#include<cstring>
#include<algorithm>
typedef long long ll;
using namespace std;
int a,b,c,l;ll ans;

ll f(int a,int b,int c){
	ll res=0;
	for (int i=0;i<=l;i++){
		int t=min(l-i,c+i-a-b);
		if (t>=0) res+=1ll*(t+2)*(t+1)/2;
	}
	return res;
}

int main(){
	scanf("%d%d%d%d",&a,&b,&c,&l);
	for (int i=0;i<=l;i++) ans+=1ll*(i+2)*(i+1)/2;
	ans-=f(c,b,a),ans-=f(b,a,c),ans-=f(c,a,b);
	printf("%I64d\n",ans);
	return 0;
}
