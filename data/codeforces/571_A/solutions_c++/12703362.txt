#include<iostream>
#include<algorithm>
using namespace std;
long long solve(long long a,long long b,long long c,long long l)
{
	long long temp=min(a-b-c,l);
	if (temp<0) return 0;
	return (temp+1)*(temp+2)/2;
}
int main()
{
	long long a,b,c,l;
	cin>>a>>b>>c>>l;
	long long ans=(l+1)*(l+2)*(l+3)/6;
	for (long long i=0;i<=l;i++)
	{
		ans-=solve(a+i,b,c,l-i);
		ans-=solve(b+i,c,a,l-i);
		ans-=solve(c+i,a,b,l-i);
	}
	cout<<ans<<endl;
	return 0;
}
