#include<bits/stdc++.h>
using namespace std;
typedef long long LL;
LL a,b,c,l;
LL cal(LL x,LL y,LL z,LL l){
    LL ans=0;
    for(LL i=x;i<=x+l;i++){
        if(y+z>i)continue;
        LL tmp=min(i,x+y+z+l-i)-y-z;
        ans+=(tmp+1)*(tmp+2)/2;
    }
    return ans;
}
int main(){
    cin>>a>>b>>c>>l;
    LL ans=(l+3)*(l+2)*(l+1)/6;
    ans-=cal(a,b,c,l);
    ans-=cal(c,b,a,l);
    ans-=cal(b,a,c,l);
    cout<<ans<<endl;
    return 0;
}
