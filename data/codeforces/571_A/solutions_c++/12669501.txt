#include<stdio.h>
#include<algorithm>
using namespace std;
int a,b,c,l;
long long ans=0;
inline long long calc(int x,int t)
{
	if(t<0) return 0LL;
	int s=min(l-x,t);
	return 1LL*(s+1)*(s+2)/2;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("data.in","r",stdin);
#endif
	int i,j;
	scanf("%d%d%d%d",&a,&b,&c,&l);
	ans=1LL*(l+1)*(l+2)*(l+3)/6;
	for(i=0;i<=l;i++)
		ans-=calc(i,a+i-b-c)+calc(i,b+i-a-c)+calc(i,c+i-a-b);
	printf("%I64d\n",ans);
	return 0;
}
