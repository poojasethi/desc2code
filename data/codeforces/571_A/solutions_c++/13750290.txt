#include<cstdio>
#include<cstdlib>
#include<algorithm>
using namespace std;
#define LL long long

int a,b,c,L;
inline LL f(int a,int b,int c)
{
	LL ret=0;
	for(int i=0,tmp;i<=L;i++) {
		tmp=min(c+i-a-b,L-i);
		if(tmp>=0) 
		ret+=1LL*(tmp+1)*(tmp+2)/2;
	}
	return ret;
}
int main()
{
	scanf("%d%d%d%d",&a,&b,&c,&L);
	printf("%I64d\n",1LL*(L+1)*(L+2)*(L+3)/6-f(a,b,c)-f(a,c,b)-f(b,c,a));
	return 0;
}
