#include <iostream>

using namespace std;

int a, b, c, l;

long long g(int a, int b, int c) {
    long long ret = 0;
    for(int i = 0; i <= l; i++) {
        int r = min(l - i, a - c - b + i);
        if(r >= 0) ret += (long long) (r + 1) * (r + 2) / 2;
    }
    return ret;
}

int main() {
    cin >> a >> b >> c >> l;
    cout << (long long) (l + 1) * (l + 2) * (l + 3) / 6 - g(a, b, c) - g(b, c, a) - g(c, a, b) << endl;
}
