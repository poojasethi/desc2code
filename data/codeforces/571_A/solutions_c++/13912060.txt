#include<bits/stdc++.h>
using namespace std; 
#define ll long long
ll a,b,c,l;
ll calc(ll a,ll b,ll c)
{
    ll ret=0,t;
    for(int i=0;i<=l;i++)
    {
        t=min(l-i,a+i-b-c);
        if(t>=0) ret+=(t+1)*(t+2)/2;
    }
    return ret;
}
int main()
{
    cin>>a>>b>>c>>l;
    cout<<(l+1)*(l+2)/2*(l+3)/3-calc(a,b,c)-calc(b,c,a)-calc(c,b,a)<<endl;
}
