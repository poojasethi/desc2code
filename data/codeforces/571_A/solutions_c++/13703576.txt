#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define ll long long 
using namespace std;
ll a,b,c,res,l,i,j,k;
ll f(ll x,ll y,ll z,ll t){
   x=min(t,x-y-z);
   if(x<0)
   return 0;
   return (x+2)*(x+1)/2;
}
int main(){
	scanf("%lld %lld %lld %lld",&a,&b,&c,&l);
	res=(l+3)*(l+2)*(l+1)/6;
	for(i=0;i<=l;i++)
	res-=f(a+i,b,c,l-i)+f(b+i,a,c,l-i)+f(c+i,a,b,l-i);
	printf("%lld\n",res);
	return 0;
}
