#include <iostream>
#include <cmath>

using namespace std;

typedef long long int TInt;

int main() {
	TInt ar[3];
	TInt l;
	cin >> ar[0] >> ar[1] >> ar[2] >> l;
	TInt ans = ((l + 1) * (l + 2) * (l + 3)) / 6;
	for (TInt j = 0; j < 3; ++j) {
		TInt a = ar[j];
		TInt b = ar[(j + 1) % 3];
		TInt c = ar[(j + 2) % 3];
		for (TInt i = 0; i <= l; ++i) {
			TInt r = min(a - b - c + i, l - i);
			if (r >= 0) {
				ans -= ((r + 1) * (r + 2)) / 2;
			}
		}
	}
	cout << ans << endl;
}
