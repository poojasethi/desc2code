#include <iostream>
#include <cmath>

using namespace std;

typedef long long int ll;

ll maneras(ll m,ll s,ll l)
{
  if (s>m) return 0;
  ll add=min(m-s,l);
  return (add+1)*(add+2)/2;
}

int main()
{
  ll a,b,c,l;
  cin>>a>>b>>c>>l;
  ll com=0;
  for (int i=0;i<=l;i++)
    com+=maneras(a+i,b+c,l-i)+maneras(b+i,a+c,l-i)+maneras(c+i,a+b,l-i);
  cout<<(l+3)*(l+2)*(l+1)/6-com<<endl;
}
