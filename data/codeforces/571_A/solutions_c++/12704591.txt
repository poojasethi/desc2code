#include <stdio.h>
#include <iostream>

using namespace std;

int a,b,c,l;
long long answ;

long long Solve(int a,int b,int c,int l){
    long long answ=0;
    for (int la=0;la<=l;la++){
        long long x=min(a-b-c+la,l-la);
        if (x>=0)
            answ+=((x+1)*(x+2))/2;
    }
    return answ;
}

int main(){

    scanf("%d %d %d %d",&a,&b,&c,&l);
    answ=((l+1)*(long long)(l+2)*(l+3))/6;
    answ-=(Solve(a,b,c,l))+Solve(b,a,c,l)+Solve(c,a,b,l);
    printf("%I64d",answ);
}
