#include <iostream>
#include <algorithm>
using namespace std;
const int MAXN = 2 * 100 * 1000 + 1000;
struct box{
	int a, o, v;
}f[MAXN];
bool cmp(box x, box y){
	return x.a < y.a;
}
int main(){
	ios::sync_with_stdio(false);
	int t, n;
	cin >> t;
	while(t--)
	{
		cin >> n;
		n = 2 * n - 1;
		for(int i=0; i<n; i++)
		{
			cin >> f[i].a >> f[i].o;
			f[i].v = i + 1;
		}
		sort(f, f+n, cmp);
		long long s1 = 0, s2 = 0;
		for(int i=0; i<n-1; i++)
			if(i % 2)
				s2 += f[i].o;
			else
				s1 += f[i].o;
		cout << "YES\n";
		if(s1 > s2)
			for(int i=0; i<n-1; i+=2)
				cout << f[i].v <<' ';
		else
			for(int i=1; i<n-1; i+=2)
				cout << f[i].v <<' ';
		cout << f[n - 1].v <<endl;
	}
}
