#include <iostream>
#include <algorithm>
#include <vector>
#include <set>
#include <map>

using namespace std;

#define pie pair <int, int>
#define ff first
#define ss second

const int maxN = 200 * 1000 + 1000;

int n;
pair <pie, int> a[maxN];

int main()
{
	ios::sync_with_stdio (false);
	int t; cin >> t;
	while (t --> 0)
	{
		cin >> n;
		for (int i = 0; i < 2 * n - 1; i++) 
			cin >> a[i].ff.ff >> a[i].ff.ss, a[i].ss = i + 1;
		sort (a, a + 2 * n - 1);
		
		cout << "YES" << endl;
		for (int i = 0; i < n - 1; i++)
			if (a[2 * i].ff.ss > a[2 * i + 1].ff.ss)
				cout << a[2 * i].ss << ' ';
			else
				cout << a[2 * i + 1].ss << ' ';
		cout << a[2 * n - 2].ss << endl;		
	}
	return 0;
}
