{

	int t;

	{
		scanf("%d",&n);
	  for (i=1;i<n+n;i++) scanf("%d%d",&a[i].x,&a[i].y),a[i].z=i,b[i]=0;
	  sort(a+1,a+n+n,cmp),b[a[1].z]=1;
		long long s=0;

		printf("YES\n");

	}

}
