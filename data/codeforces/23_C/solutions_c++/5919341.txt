#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;


int main() {
	int t, n;
	cin >> t;
	while(t--) {
		cin >> n;

		vector< pair< pair<int, int>, int > > p;
		for(int i = 0; i < 2*n-1; i++) {
			p.push_back(make_pair(make_pair(0, 0), 0));
			cin >> p[i].first.first >> p[i].first.second;
			p[i].second = (i+1);
		}
		sort(p.begin(), p.end());
		cout << "YES" << endl;

		for(int i = 0; i < p.size()-1; i+=2) {
			if(p[i].first.second < p[i+1].first.second) 
			{
				cout << p[i+1].second << " ";
			}
			else 	cout << p[ i ].second << " ";
		}
		cout << p[2*n-2].second << endl;
	}

	return 0;
}
