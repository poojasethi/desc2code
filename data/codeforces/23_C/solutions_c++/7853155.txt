#include <cstdio>
#include <algorithm>
using namespace std;
int t,n;
struct P
{
	long long o,a;
        int ID;
	inline void Read(int x)
	{
		scanf("%I64d%I64d",&o,&a);
		ID=x;
	}
}a[200005];
inline bool Cmp(const P& A,const P& B)
{
	return A.o<B.o;
}
int main()
{
	scanf("%d",&t);
	while (t--)
	{
		scanf("%d",&n);
		for (int i=1;i<=2*n-1;i++) a[i].Read(i);
		sort(a+1,a+2*n,Cmp);
		long long A=0,B=A;
		for (int i=1;i<2*n-1;i++)
			(i&1)?A+=a[i].a:B+=a[i].a;
		puts("YES");
		for (int i=A>B?1:2;i<2*n-1;i+=2) printf("%d ",a[i].ID);
		printf("%d\n",a[2*n-1].ID);
	}
	return 0;
}
