#include<iostream>
#include<algorithm>
#include<utility>
#include<cmath>
#include<vector>
using namespace std;
pair<pair<int,int> , int> p[200000+10];
int main(){
    int t;
    cin>>t;
    while(t!=0)
    {
        t--;
        int n;cin>>n;
        int r=(2*n)-1;
        for(int i=0;i<r;i++)
        {
            cin>>p[i].first.first>>p[i].first.second;
            p[i].second=i+1;
        }
        sort(p,p+r);
        cout<<"YES"<<endl;
        cout<<p[r-1].second<<" ";
        for(int i=0;i<r-1;i+=2) {
            if(p[i].first.second>p[i+1].first.second)
                cout<<p[i].second<<" ";
            else
                cout<<p[i+1].second<<" ";
        }
        cout<<endl;
    }
}
