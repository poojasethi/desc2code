#include<iostream>
#include<cstdio>
#include<cstring>
#include<cmath>
#include<algorithm>
using namespace std;

struct box
{
	int num;
	int o;
	int a;
}bx[200010];
bool cmp(box x,box y)
{
	return x.a<y.a;
}
int main()
{
	int n,cnt,o,k;
	scanf("%d",&o);
	while(o--)
	{
		scanf("%d",&n);
		k=2*n-1;
		for(int i=0;i<k;i++)
		{
			bx[i].num=i+1;
			scanf("%d%d",&bx[i].a,&bx[i].o);
		}
		sort(bx,bx+k,cmp);
		printf("YES\n");	
		for(int i=0;i<k-1;i=i+2)
		{
			if(bx[i].o>bx[i+1].o)
				printf("%d ",bx[i].num);
			else
				printf("%d ",bx[i+1].num);
		}	
		printf("%d\n",bx[k-1]);

	}
	return 0;
}
