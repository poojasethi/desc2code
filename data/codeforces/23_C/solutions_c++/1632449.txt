#include <cstdio>
#include <algorithm>

using namespace std;

struct box
{
  int a,b,no;
  bool operator<(const box& x) const
  {
    return a<x.a;
  }
};

box f[200001];

int main(void)
{
  int t;
  scanf("%d",&t);
  for(int count=1;count<=t;++count)
  {
    int n;
    scanf("%d",&n);
    for(int i=1;i<2*n;++i)
    {
      f[i].no=i;
      scanf("%d %d",&f[i].a,&f[i].b);
    }
    sort(f+1,f+n+n);
    printf("YES\n");
    for(int i=2;i<2*n;i+=2)
    {
      if(f[i].b>f[i-1].b)
        printf("%d ",f[i].no);
      else
        printf("%d ",f[i-1].no);
    }
    printf("%d\n",f[2*n-1].no);
  }
  return 0;
}
