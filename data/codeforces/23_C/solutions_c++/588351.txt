#include <stdio.h>
#include <vector>
#include <algorithm>
using namespace std;

void solve(void) {
	int n, i;
	scanf("%d", &n);
	vector<pair<pair<int, int>, int> > a, b;
	
	for(i=1;i<=2*n-1;i++) {
		int x, y;
		scanf("%d%d", &x, &y);
		a.push_back(make_pair(make_pair(x, y), i));
	}
	sort(a.begin(), a.end());
	printf("YES\n");
	for(i=0;i<(int)a.size();i++){
		if(b.size()>0 && (a[i].first.second >= b.back().first.second)) {
			printf("%d ", a[i].second);
			b.pop_back();
		} else
			b.push_back(a[i]);
	}
	for(i=0;i<(int)b.size();i+=2)
		printf("%d ", b[i].second);
	puts("");
}

int main(void) {
	int T;
	scanf("%d", &T);
	while(T--) solve();
	return 0;
}
