#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#include <vector>
#include <map>

using namespace std ;

const int MAXN = 2*(int)1e5+10 ;

int a[MAXN] , b[MAXN] ;
int n ;

int p[MAXN] ;

bool cmp(const int &x,const int &y)
{
    return a[x] > a[y];
}

int main()
{
    int _;
    scanf("%d",&_);
    while (_--)
    {
        scanf("%d",&n);
        for (int i = 0 ; i < 2*n-1 ; i++) scanf("%d%d",&a[i],&b[i]) , p[i]=i ;
        sort(p,p+2*n-1,cmp) ;
        printf("YES\n%d ",p[0]+1) ;
        for (int i = 1 ; i < 2*n-1 ; i+=2)
        {
            int x=p[i],y=p[i+1];
            if (b[x]<b[y]) swap(x,y);
            printf("%d ",x+1);
        }
        printf("\n");
    }
    return 0 ;
}
