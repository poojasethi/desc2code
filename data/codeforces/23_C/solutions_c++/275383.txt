#include<iostream>
#include<algorithm>
using namespace std;

struct PII{
	int a, o, n;
	PII(int aa = 0, int oo = 0, int nn = 0){
		a = aa;
		o = oo;
		n = nn;
	}
} p[210100];

bool cmp(PII a, PII b){
	return a.a > b.a;
}

int main(){
	int t;
	cin >> t;
	for(int i = 0; i < t; i++){
		int n;
		cin >> n;
		n = 2*n-1;
		for(int j = 0; j < n; j++){
			int a, o;
			cin >> a >> o;
			p[j] = PII(a, o, j+1);
		}
		sort(p, p+n, cmp);
		cout << "YES" << endl << p[0].n ;
		for(int i = 1; i < n; i+= 2)
			if(p[i].o >= p[i+1].o)
				cout << " " << p[i].n;
			else 
				cout << " " << p[i+1].n;
		cout << endl;
	}
	return 0;
}
