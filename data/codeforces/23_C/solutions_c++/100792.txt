#include <stdio.h>
#include <algorithm>
#include <vector>
#define MAX 1000000
using namespace std;

int na[MAX], no[MAX], ni[MAX];

bool cmp(int i, int j){
	return na[i] < na[j];
}

int main(){
	int t, o, a, n;
	scanf("%d\n", &t);
	for(int test = 0; test < t; test++){
		scanf("%d\n", &n);
		for(int i = 0; i < 2 * n - 1; i++){
			scanf("%d %d\n", &a, &o);
			na[i] = a; no[i] = o; ni[i] = i;
		}
		sort(ni, ni + 2 * n - 1, cmp);
		printf("YES\n");
		for(int i = 0; i < 2 * n - 2; i += 2){
			if(no[ni[i]] > no[ni[i + 1]]) printf("%d ", ni[i] + 1);
			else printf("%d ", ni[i + 1] + 1);
		}
		printf("%d\n", ni[2 * n - 2] + 1);
	}
	return 0;
}
