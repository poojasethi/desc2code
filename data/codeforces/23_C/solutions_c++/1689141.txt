#include<cstdio>
#include<iostream>
#include<algorithm>
using namespace std;
struct box
{
    int a,o,no;
}xx[211111];
bool cmp(box i,box j)
{
    return i.a<j.a;
}
int main()
{
    int t,n;
    cin>>t;
    while(t--)
    {
        cin>>n;
        int i,j;
        for(i=0;i<2*n-1;i++)
        {
            cin>>xx[i].a>>xx[i].o;
            xx[i].no=i+1;
        }
        sort(xx,xx+(2*n-1),cmp);
        printf("YES\n");
        for(i=1;i<2*n-1;i+=2)
        {
            if(xx[i].o>xx[i-1].o)cout<<xx[i].no<<' ';
            else cout<<xx[i-1].no<<' ';
        }
        cout<<xx[2*n-2].no<<endl;
    }
}
