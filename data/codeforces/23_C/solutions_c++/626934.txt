#include<iostream>
#include<cstdio>
#include<algorithm>
using namespace std;

#define MAXN 210000
struct box{
       long a, o, id;
       bool operator<(box b)const {return o>b.o;}
};

box a[MAXN];
long i, j, n;

int main(){
    int _;
    scanf("%d\n",&_);
    while(_--){
      scanf("%d\n",&n);
      for(i = 1; i < n*2;i++) scanf("%d%d\n",&a[i].a,&a[i].o);
      for(i=1;i<2*n;i++)a[i].id=i;
      sort(a + 1, a + 2*n);
      printf("YES\n");
      for(i = 2;i<2*n;i+=2) printf("%d ",a[a[i].a>a[i + 1].a?i:i+1].id);
      printf("%d\n",a[1].id);
    }    
}
