#include <cstdio>
#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;
struct node
{
    int a, o, i;
}abox[300005];
bool cmpa(node a, node b)
{
    return a.a < b.a;
}
int main()
{
    int t;
    scanf("%d", &t);
    while(t--)
    {
        int n;
        scanf("%d", &n);
        for(int i = 0; i < 2*n-1; i++)
        {
            scanf("%d%d", &abox[i].a, &abox[i].o);
            abox[i].i = i + 1;
        }
        sort(abox, abox + 2*n-1, cmpa);
        puts("YES");
        for(int i = 1; i < 2*n-1; i += 2)
        {
            printf("%d ", abox[i-1].o > abox[i].o? abox[i-1].i : abox[i].i);
        }
        printf("%d\n", abox[2*n-2].i);
    }
    return 0;
}
