#include <algorithm>
#include <iostream>
#include <utility>
#include <vector>

using namespace std;

typedef long long ll;
typedef pair<ll, ll> pii;

int main() {
	int T; cin >> T;
	for (int t = 1; t <= T; ++t) {
		ll n; cin >> n;
		vector<pair<pii, ll> > v(2*n - 1);
		ll total = 0;
		for (int i = 0; i < v.size(); ++i) {
			cin >> v[i].first.first >> v[i].first.second;
			v[i].second = i + 1;
			total += v[i].first.second;
		}
		sort(v.begin(), v.end());
		ll evens = 0;
		for (int i = 0; i < v.size(); i+=2)
			evens += v[i].first.second;
		int oset = !(2*evens >= total);
		cout << "YES" << endl;
		for (int i = 0; i + 1 < n; ++i)
			cout << v[2*i + oset].second << " ";
		cout << v.back().second << endl;
	}
}
