#include<iostream>
#include<algorithm>
using namespace std;

#define PII pair<int,int>
#define PPI pair<PII,int>
#define F first 
#define S second
#define A F.F
#define O F.S
#define Ind S

const int MAX = 300*1000;
PPI box[MAX+10];
int n;

int main()
{
	int t;
	cin >> t;
	for(int tt=0; tt<t; tt++)
	{
		cin >> n;
		for(int i=0; i<2*n-1; i++)
			cin >> box[i].A >> box[i].O , box[i].Ind = i;
		sort( box , box + 2*n-1 );
		cout << "YES" << endl;
		for(int i=0; i<n-1; i++)
			if( box[2*i].O > box[2*i+1].O )
				cout << box[2*i].Ind + 1 << " ";
			else
				cout << box[2*i+1].Ind + 1 << " ";
		cout << box[2*n-2].Ind + 1 << endl;
	}
	return 0;
}
