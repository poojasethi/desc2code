#include<iostream>  
#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
struct node
{
	int a;
	int b;
	int index;
}box[2*100000+5];
int k,t,n;
int cmp(node x,node y)
{
	return x.a<y.a;
}

int main()
{
	scanf("%d",&t);
	while(t--)
	{
		scanf("%d",&n);
		k=2*n-1;
		for(int i=1;i<=k;i++)
		{
			scanf("%d%d",&box[i].a,&box[i].b);
			box[i].index=i;
		}
		sort(box+1,box+1+k,cmp);
		printf("YES\n");
		for(int i=2;i<=k-1;i=i+2)
			if(box[i].b>box[i-1].b)	
				printf("%d ",box[i].index);
			else	
				printf("%d ",box[i-1].index);
		printf("%d\n",box[k].index);
	}
	return 0;
}
