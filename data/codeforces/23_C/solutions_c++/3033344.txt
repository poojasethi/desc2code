#include <iostream>
#include <algorithm>

using namespace std;

struct box
{
    int a,o,num;
}s[400015];

int cmp(const box &a,const box &b)
{
    return a.a<b.a;
}


int main()
{
    int n,t,i,l;
    cin >> t;
    while(t--)
    {
        cin >> n;
        l = 2*n-1;
        for(i=0;i<l;i++)
        {
            cin >> s[i].a >> s[i].o;
            s[i].num = i+1;
        }
        sort(s,s+l,cmp);
        cout << "YES" << endl;
        for(i=1;i<l;i+=2)
        if(s[i].o > s[i-1].o) cout << s[i].num << " ";
        else cout << s[i-1].num << " ";
        cout << s[l-1].num << endl;
    }
    return 0;
}
