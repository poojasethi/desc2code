#include <iostream>
#include <stdio.h>
#include <cstdlib>
#include <algorithm>

using namespace std;

struct fruit {
	int a,o,in;
} p[222222];

bool cmp ( fruit a,fruit b )
{
	return (a.o < b.o) || ( a.o == b.o && a.in < b.in);
}

int T,n;

int main()
{
	scanf("%d",&T);
	for ( int t = 0; t < T; t++ ) {
		scanf("%d",&n);
		long long app = 0,oran = 0;
		for ( int i = 1; i < 2 * n; i++ ) {
			scanf("%d%d",&p[i].a,&p[i].o);
			p[i].in = i;
		}
		sort(p+1,p+2*n,cmp);
		puts("YES");
		cout << p[2*n-1].in << " ";
		for ( int i = 2 * n - 2; i > 1; i-=2 )
			if ( p[i].a > p[i-1].a ) cout << p[i].in << " ";
				else cout << p[i-1].in << " ";
	}
}
