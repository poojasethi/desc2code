#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;
const int N = 200001;
struct node
{
    int x, y, id;
    bool operator < (const node &T) const
    {return x < T.x;}
}p[N];
int main ()
{
    int t, i, n; scanf ("%d", &t);
    while(t --)
    {
        scanf ("%d", &n);
        for (i = 1; i < n << 1; i ++) scanf ("%d%d", &p[i].x, &p[i].y), p[i].id = i;
        sort (p + 1, p + n + n);
        puts ("YES");
        for (i = 2; i < n << 1; i += 2)
        printf ("%d ", p[i].y >= p[i - 1].y ? p[i].id: p[i - 1].id);
        printf ("%d\n", p[n + n - 1].id);
    }
    return 0;
}
