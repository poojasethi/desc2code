#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
int main()
{
	int t,n,a,o;
	vector< pair< pair<int, int>, int> > v;
	cin>>t;
	while(t--)
	{
		cin>>n;
		n = 2 * n - 1;
		v.resize(n);
		long long full = 0;
		for(int i = 0; i < n; i++)
		{
			cin>>a>>o;
			full += o;
	          v[i] = make_pair(make_pair(a,o),i + 1);
		}
		sort(v.begin(),v.end());
		cout<<"YES"<<endl;
		long long odd = 0;
		for(int i = 0; i < n; i += 2)
			odd += v[i].first.second;
		if (odd >= full / 2 + full % 2)
		{
			for(int i = 0; i < n; i += 2)
				cout<<v[i].second<<" ";
		} else
		{
			for(int i = 1; i < n; i += 2)
				cout<<v[i].second<<" ";
			cout<<v[n - 1].second;
		}
		cout<<endl;
	}
	return 0;
}
