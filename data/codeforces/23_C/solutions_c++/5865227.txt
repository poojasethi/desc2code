#include <iostream>
#include <algorithm>

using namespace std;

struct my {
    int a, b, ii;
};

bool cmp ( my a, my b ){
    if ( a.a < b.a ) return true;
    return false;
}

my res [1111111];

int main()
{
    int n, t;
    cin >> t;
    for (int u = 0 ; u <t ; u++){
        cin >> n;
        n*=2; n--;
        for (int i = 0 ; i < n; i++){
            cin >> res[i].a >> res[i].b;
            res[i].ii = i;
        }
        sort ( res, res + n, cmp );
        cout << "YES\n";
       // cout << "ii;"
        for (int i = 1 ; i < n; i+=2)
            if ( res[i].b > res[i-1].b ) cout << res[i].ii + 1 << " ";
            else cout << res[i-1].ii + 1 << " ";
        cout << res[n-1].ii+1 << "\n";
    }
    return 0;
}
