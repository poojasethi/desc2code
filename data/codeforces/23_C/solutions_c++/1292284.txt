#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
typedef long long ll;
typedef pair<ll,ll> P;
typedef pair<P,ll> PP;
int tests,n;
ll s[2],x,y;
vector <PP> V;
int main()
{
	cin >> tests;
	while(tests--)
	{
		cin >> n;
		V.clear();
		for(int i=0;i<2*n-1;i++)
		{
			cin >> x >> y;
			V.push_back(PP(P(x,y),i+1));
		}
		sort(V.begin(),V.end());
		reverse(V.begin(),V.end());
		s[0]=s[1]=0;
		for(int j=1;j<2*n-1;j++)
			s[j%2]+=V[j].first.second;
		cout << "YES" << endl;
		cout << V[0].second << " ";
		if(s[0]>=s[1])
			for(int j=2;j<2*n-1;j+=2)
				cout << V[j].second << " ";
		else
			for(int j=1;j<2*n-1;j+=2)
				cout << V[j].second << " ";
		cout << endl;
		}
	return 0;
}
