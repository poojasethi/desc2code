#include <cstdio>
#include <cstring>
#include <algorithm>
using namespace std;
#define N 200008
int n;
int app[N], org[N], id[N], visit[N] = { 0 };

inline int cmp(int x, int y)
{
	return app[x] > app[y];
}

int main()
{
	int ca, i, j, t = 0;
	scanf("%d", &ca);
	while (ca--)
	{
		t++;
		scanf("%d", &n);
		for (i = 0; i < 2*n-1; i++)
		{
			scanf("%d%d", app+i, org+i);
			id[i] = i;
		}
		sort(id, id+(2*n-1), cmp);
		visit[id[0]] = t;
		for (i = 1; i < 2*n-1; i += 2)
		{
			if (org[id[i]] > org[id[i+1]])
				visit[id[i]] = t;
			else
				visit[id[i+1]] = t;
		}
		puts("YES");
		for (i = j = 0; i < 2*n-1; i++)
		{
			if (visit[i] == t)
			{
				if (j) printf(" ");
				j = 1;
				printf("%d", i+1);
			}
		}
		puts("");
	}
}
