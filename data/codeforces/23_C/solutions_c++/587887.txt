#include <iostream>
#include <cstdio>
#include <string>
#include <string.h>
#include <algorithm>
using namespace std;

struct node{
    int x,y,z;
}a[265555];

bool cmp(node a,node b){
    return a.x<b.x;
}
int n;
int main(){
    int T;
    scanf("%d",&T);
    while(T--) {
        scanf("%d",&n);
        for(int i=1;i<n+n;i++){
          scanf("%d%d",&a[i].x,&a[i].y);
          a[i].z=i;
        }
        sort(a+1,a+n+n,cmp);
        puts("YES");
        for(int i=2;i<n+n;i+=2)
         printf("%d ",a[a[i-1].y>a[i].y?i-1:i].z);
        printf("%d\n",a[n+n-1].z);
    }
    return 0;
}
