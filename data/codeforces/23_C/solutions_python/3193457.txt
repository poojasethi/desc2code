#!/usr/bin/python
from operator import attrgetter

class Box:
	def __init__(self, a, o, i):
		self.a = a
		self.o = o
		self.i = i

def singleTest():
	n = int(raw_input())
	boxes = []
	totalApple = totalOrange = 0
	for i in range(2*n-1):
		a, o = map(lambda x: int(x), raw_input().split())
		boxes.append(Box(a, o, i+1))
		totalApple += a
		totalOrange += o
	appleSort = sorted(boxes, key=attrgetter('a'), reverse=True)
	chosen = appleSort[0:2*n-1:2]
	oranges = sum(map(attrgetter('o'), chosen))
	print "YES"
	result = [appleSort[0].i]
	if 2*oranges < totalOrange:
		for i in range(1, 2*n-1, 2):
			result.append(appleSort[i].i)
	else:
		for i in range(2, 2*n-1, 2):
			result.append(appleSort[i].i)
	result.sort()
	for item in result:
		print item,
	print

t = int(raw_input())
for i in range(t):
	singleTest()
