#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
using namespace std;

const int MAXN = 10;
const int MAXS = 1024;

int n, m, p, cstate, ans;
int t[MAXN];
int f[MAXS][MAXS];

inline int lowbit(int i) {
	return (i & (-i));
}

int main() {
	scanf("%d %d %d", &n, &m, &p);
	for (int i = 0; i < m; i++) {
		int a, b;
		scanf("%d %d", &a, &b);
		a--, b--;
		f[(1 << a) | (1 << b)][(1 << a) | (1 << b)]++;
		t[a] |= 1 << b, t[b] |= 1 << a;
	}
	
	cstate = 1 << n;
	for (int i = 0; i < cstate; i++)
		for (int j = 0; j < cstate; j++) if (f[i][j]) {
			for (int k = 0; k < n; k++) if (!(i & (1 << k))) {
				for (int s = i & t[k], p = lowbit(s); s; s -= p, p = lowbit(s)) {
					int s1 = i | (1 << k), s2 = ((j | p) ^ p) | (1 << k);
					if ((1 << (k + 1)) > s2) f[s1][s2] += f[i][j];
				}
			}
		}

	for (int i = 0; i < cstate; i++) {
		int cnt = 0;
		for (int j = 0; j < n; j++) if (i & (1 << j)) cnt++;
		if (cnt == p) ans += f[cstate - 1][i];
	}
	
	printf("%d\n", ans);

	return 0;
}
