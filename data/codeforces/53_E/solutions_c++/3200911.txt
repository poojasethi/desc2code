#include <iostream>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <string>
#include <cmath>
#include <algorithm>
using namespace std;
typedef long long LL;

int f[1050][1050],n,m,K,link[15][15];

int get()
{
	int f=0,v=0; char ch;
	while (!isdigit(ch=getchar()))if (ch=='-')break;
	if (ch=='-')f=1; else v=ch-48;
	while (isdigit(ch=getchar()))v=v*10+ch-48;
	if (f==1)return -v ;return v;
}

int main()
{
	n=get(),m=get(),K=get();
	for(int i=1;i<=m;i++)
	{
		int x=get(),y=get();x--;y--;
		link[x][y]=link[y][x]=f[1<<x|1<<y][1<<x|1<<y]=1;
	}
	for(int i=1;i<(1<<n);i++)
		for(int j=i;j;j--,j&=i)if(f[i][j])
		{
			for(int x=0;x<n;x++)if((1<<x)&i)
			{
				for(int y=0;y<n;y++)if(!((1<<y)&i))
				{
					if(!link[x][y])continue;
					int s=(j&(~(1<<x)))|(1<<y);
					if(!(s>>y+1))f[i|(1<<y)][s]+=f[i][j];
				}
			}
		}
	int ans=0;
	for(int i=1;i<(1<<n);i++)
		if(__builtin_popcount(i)==K)ans+=f[(1<<n)-1][i];
	printf("%d\n",ans);
	return 0;
}
