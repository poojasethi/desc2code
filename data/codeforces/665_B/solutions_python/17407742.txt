from sys import stdin

l2i = lambda l: [int(x) for x in l.strip().split()]

n, m, k = l2i(stdin.readline())
shop = l2i(stdin.readline())

def buy(i):
    sw = i
    pos = 0
    while shop[0] != sw or pos == 0:
        sw, shop[pos] = shop[pos], sw
        pos += 1

    return pos

time = 0
while n != 0:
    n -= 1
    for commodity in l2i(stdin.readline()):
        time += buy(commodity)
    
print time
