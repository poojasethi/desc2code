import math

n,m,k = map(int, raw_input().strip().split())
p = map(int, raw_input().strip().split())
a = [[]]*0
for i in xrange(n):
	a.append( map(int, raw_input().strip().split()) )

ans = 0
for i in xrange(n):
	for j in xrange(m):
		pos = p.index(a[i][j])
		ans += pos+1
		p.pop(pos)
		p.insert(0, a[i][j])
print ans
