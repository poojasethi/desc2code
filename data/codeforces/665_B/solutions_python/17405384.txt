n,m,k = map(int,raw_input().split())
p = map(int,raw_input().split())
ans = 0
for i in xrange(n):
    ui = map(int,raw_input().split())
    for it in ui:
        ans += p.index(it) + 1
        del p[p.index(it)]
        finl = [it] + p
        p = finl[:]
print ans
