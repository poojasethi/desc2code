def solve(positions, orders):
    steps = 0
    for order in orders:
        for item in order:
            item_idx = positions.index(item)
            steps += item_idx + 1
            positions.pop(item_idx)
            positions.insert(0, item)
    return steps


n, m, k = map(int, raw_input().split())
positions = map(int, raw_input().split())
orders = [map(int, raw_input().split()) for _ in xrange(n)]


print solve(positions, orders)
