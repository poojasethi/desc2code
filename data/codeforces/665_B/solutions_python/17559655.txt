def add(pos, v):
    for _ in range(len(pos)):
        if pos[_] < v:
            pos[_] += 1


n, m, k = map(int, raw_input().split())

p = map(int, raw_input().split())

pos = [0 for i in range(k + 1)]

for i in range(k):
    pos[p[i]] = i + 1

cost = 0

for i in range(n):
    a = map(int, raw_input().split())
    for e in a:
        cost += pos[e]
        add(pos, pos[e])
        pos[e] = 1

print cost
