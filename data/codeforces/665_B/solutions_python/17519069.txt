n,m,k = map(int,raw_input().split())
k_lst = map(int,raw_input().split())
sum = 0
#print "k_lst",k_lst
for i in xrange(n):
        c_lst = map(int,raw_input().split())
        for order in c_lst:
                ord_index = k_lst.index(order)
                sum += ord_index+1
                prev = k_lst[ord_index]
                k_lst.remove(prev)
                k_lst = [prev] + k_lst
                #print "k_lst",k_lst
        #print "sum",sum

print sum
