n=[int(i) for i in raw_input().split()]
k=[int(i) for i in raw_input().split()]
i=0
j=0
sum=0
while(i<n[0]):  
    a=[int(p) for p in raw_input().split()]
    j=0
    while(j<len(a)):
        sum+=k.index(a[j])+1
        k.remove(a[j])
        k.insert(0,a[j])
        j+=1
    i+=1
print(sum)
