#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;

int n, m, k;
int a[111], t[111];
int ans, x;

int main() {
	scanf("%d%d%d", &n, &m, &k);
	for(int i=0;i<k;i++) scanf("%d", &a[i]);
	ans=n*m;
	for(int i=0;i<n;i++) {
		memcpy(t, a, sizeof a);
		for(int j=0;j<m;j++) {
			scanf("%d", &x);
			int p=0;
			for(;a[p]!=x;p++);
			for(;p>0;swap(a[p], a[p-1]), p--, ans++);
		}
	}
	printf("%d\n", ans);
	
	return 0;
}
