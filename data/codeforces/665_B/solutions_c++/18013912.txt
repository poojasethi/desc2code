#include "stdio.h"
#include "iostream"

using namespace std;

int n,m,k;
int point[103];

int main()
{
	int total = 0;
	cin >> n >> m >>k;
	for(int t = 1;t <= k;t ++)
	{
		int c;
		cin >> c;
		point[c] = t;
	}
	int z = n * m;
	while(z --)
	{
		int v;
		cin >> v;
		total += point[v];
		int r = point[v];
		point[v] = 0;
		for(int t = 1;t <= k;t ++)
		{
			if(point[t] < r)
				point[t] ++;
		}
		


	}
	cout << total;



	return 0;
}
