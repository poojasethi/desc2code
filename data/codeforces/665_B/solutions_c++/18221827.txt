#include<iostream>
using namespace std;
int a,b,c,d,ans,i,j,k,x,t,w[100];
main()
{
	cin>>a>>b>>c;
	i=0;while(i<c)cin>>w[i++];
	i=0;while(i<a)
	{
		j=0;while(j<b)
		{
			cin>>d;
			x=0;while(w[x]!=d)x++;
			ans+=x+1;
			t=w[x],k=x;while(k>0){w[k]=w[k-1];k--;}
			w[0]=t;
			j++;
		}
		i++;
	}
	cout<<ans;
}
