#include <iostream>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	int K[k],M[n][m],otvet=0;
	for(int i=0;i<k;i++)
		cin>>K[i];
	for(int i=0;i<n;i++)
	{	
		for(int j=0;j<m;j++)
		{
			cin>>M[i][j];
			for(int l=0;l<k;l++)
			{
				if(K[l]==M[i][j])
				{
					otvet+=l+1;
					for(int b=0;b<l;b++)
						swap(K[b],K[l]);
					break;
				}
			}
		}
	}
	cout<<otvet<<endl;
	return 0;
}
