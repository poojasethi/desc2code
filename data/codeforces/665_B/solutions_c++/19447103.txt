#include <bits/stdc++.h>
using namespace std;

int arr[102];
int pos[102];
int main()
{
	int n, m , k;
	cin >> n >> m >> k;

	for(int i = 0; i<k; i++)
	{
		cin >> arr[i];
		pos[arr[i]] = i+1;
	}

	int ans = 0;
	int x;
	for(int i = 0; i<n; i++)
	{
		for(int j = 0; j<m; j++)
		{
			cin >> x;
			ans+=pos[x];
			for(int l = 1;l<=k; l++)
			{
				if(pos[l]<pos[x])
					pos[l]++;
			}
			pos[x] = 1;
			
		}
		// cout << endl;
	}

	cout << ans << endl;
}
