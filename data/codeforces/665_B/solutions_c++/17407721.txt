#include <iostream>

using namespace std;

int modif(int a[],int b,int d){
    int i;
    for(i=b;i>=2;i--)
        a[i]=a[i-1];
    a[1]=d;
}

int main()
{
int n,m,k;
int pos[105];
int a,i,j,b;
int time=0;
cin>>n>>m>>k;
for(i=1;i<=k;i++)
    cin>>pos[i];
for(i=1;i<=n;i++)
    for(j=1;j<=m;j++){
        cin>>a;
        b=1;
        while(pos[b]!=a)
            b++;
        time+=b;
        modif(pos,b,a);
    }
cout<<time;
}
