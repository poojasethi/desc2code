#include <cstdio>
int n,m,k;
int shu[110];
int pos[110];
int ans=0;
int main() {
	scanf("%d%d%d",&n,&m,&k);
	for (int i=1;i<=k;i++) {
		scanf("%d",&shu[i]);
		pos[shu[i]]=i;
	}
	int a;
	for (int i=1;i<=n;i++) {
		for (int j=1;j<=m;j++) {
			scanf("%d",&a);
			ans+=pos[a];
			for (int t=1;t<=k;t++) {
				if (pos[t]<pos[a]) pos[t]++;
			}
			pos[a]=1;
		}
	}
	printf("%d\n",ans);
	return 0;
}
