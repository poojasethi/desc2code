#include<cstdio>

int a[110],n,m,k;

int main(){
	scanf("%d%d%d",&n,&m,&k);
	for(int i = 1;i <= k;i++)
		scanf("%d",&a[i]);
	
	int ans = 0;
	for(int i = 0;i < n;i++)
	{
		int x;
		for(int j = 1;j <= m;j++)
		{
			scanf("%d",&x);
			int l;
			for(l = 1;l <= k;l++)
				if(a[l]==x)break;
			ans += l;
			while(l>1)
			{
				a[l]=a[l-1];
				l--;
			}
			a[1]=x;
		}
	}
	printf("%d\n",ans);
	return 0;
}
