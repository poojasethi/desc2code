#include<stdio.h>
using namespace std;

int pos[1000];
int main()
{
    int n,m,k;
    scanf("%d%d%d",&n,&m,&k);
    for(int i=1;i<=k;i++) scanf("%d",&pos[i]);
    int rez=0;
    for(int i=1;i<=n;i++)
        for(int j=1;j<=m;j++)
        {
            int p,z;
            scanf("%d",&z);
            for(p=1;pos[p]!=z;p++);
            rez+=p;
            for(int i=p;i>1;i--) pos[i]=pos[i-1];
            pos[1]=z;
        }
    printf("%d\n",rez);
}
