#include<iostream>
#include<cstdio>
using namespace std;
int a[105];
int sum;
int p;
int main(){
	int n,m,k;
	scanf("%d%d%d",&n,&m,&k);
	for(int i=1;i<=k;i++)scanf("%d",&a[i]); 
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			scanf("%d",&p);
			for(int o=1;o<=k;o++){
				if(a[o]==p){
					sum+=o;
					for(int q=o;q>=2;q--){
						a[q]=a[q-1];
					} 
					a[1]=p;
					break;
				}
			} 
		}
	}
	printf("%d\n",sum);
	return 0;
}
