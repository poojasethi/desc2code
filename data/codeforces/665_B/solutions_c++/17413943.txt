#include "cstdio"
using namespace std;
int n, m, k, ans, a[101];
int main()
{
    int x, i;
    scanf("%d%d%d", &n, &m, &k);
    for (i=1; i<=k; i++)
    {
        scanf("%d", &x);
        a[x]=i;
    }
    for (i=0; i<n*m; i++)
    {
        scanf("%d", &x);
        ans+=a[x];
        for (int j=1; j<=k; j++)
            if (a[j]<a[x])
                a[j]++;
        a[x]=1;
    }
    printf("%d", ans);
    return 0;
}
