#include<bits/stdc++.h>
using namespace std;
int main(){
	int n,m,k,x;
	int a[100001];
	scanf("%d %d %d",&n,&m,&k);
	for(int i=0;i<k;i++){
		scanf("%d",&x);
		a[x]=i+1;
	}
	int ans=0;
	for(int i=0;i<n;i++){
		for(int j=0;j<m;j++){
			scanf("%d",&x);
			ans+=a[x];
			//printf("%d %d\n",x,a[x]);
			for(int l=1;l<=k;l++){
				if(a[l]<a[x])
					a[l]++;//printf("l=%d\n",l);
			}
			a[x]=1;
		}
	}
	printf("%d\n",ans);
	return 0;
}
