#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	int pos[105],idx[105];
	int a;
	long int sum=0;
	for(int i=0;i<k;i++)
	{
	cin>>a;
	pos[a]=i+1;
	idx[i+1]=a;
	}
	for(int i=0;i<m*n;i++)
	{
		cin>>a;
		//cout<<i<<" "<<pos[a]<<"\n";
		sum+=pos[a];
		for(int j=pos[a]-1;j>=1;j--)
		{
			pos[idx[j]]+=1;
			idx[j+1]=idx[j];
		}
		pos[a]=1;
		idx[1]=a;
	}
	cout<<sum<<"\n";
	return 0;
}
