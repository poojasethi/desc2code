#include <bits/stdc++.h>
using namespace std;

int main()
{
	int n,m,k,i,a,j,count=0;
	cin>>n>>m>>k;
	list<int> v;
	for(i=0;i<k;i++)
	{
		cin>>a;
		v.push_back(a);
	}
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			cin>>a;
			for(list<int>::const_iterator i=v.begin();i!=v.end();i++)
			{
				if(*i==a)
				{
					v.remove(a);
					break;
				}
				else
					count++;
			}
			count++;
			//cout<<count<<endl;
			v.push_front(a);
		}
	}
	cout<<count<<endl;
	return 0;
}
