#include <iostream>
using namespace std;
int a[100006],k,n,m,sum,b;
int main ()
{
	cin>>n>>m>>k;

	for (int i=0; i<k; i++) cin>>a[i];

		for (int i=0; i<n*m; i++)
		{
			cin>>b;
			for (int j=0; j<k; j++)
			{
				if (b==a[j])
				{
					 sum=sum+j+1;
					 for (int x=j; x>0; x--)
						a[x]=a[x-1];
						a[0]=b;
				          break;
				}
			}
		}
			cout<<sum<<endl;
}
