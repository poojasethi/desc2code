#include <iostream>

using namespace std;

int main () {
  int n, m, k, r = 0, t, t2;
  int a[100];
  cin >> n;
  cin >> m;
  cin >> k;
  for (int i = 1; i <= k; i++) {
    cin >> t;
    a[t - 1] = i;
  }
  for (int i = 0; i < n; i++) {
    for (int j = 0; j < m; j++) {
      cin >> t2;
      t = a[t2 - 1];
      r += t;
      for (int l = 0; l < k; l++) {
	if (a[l] < t) a[l]++;
      }
      a[t2 - 1] = 1;
    }
  }
  cout << r;
  return 0;
}
