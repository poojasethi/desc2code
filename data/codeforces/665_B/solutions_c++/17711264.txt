#include<bits/stdc++.h>
using namespace std;
int n,m,k,i,j,t=0,x,a[102],p;
int main()
{
    cin>>n>>m>>k;
    for(i=1; i<=k; i++)
    {
        cin>>x;
        a[x]=i;
    }
    for(i=0; i<n; i++)
    {
        for(j=0; j<m; j++)
        {
            cin>>x;
            t+=a[x];
            for(p=1; p<=k; p++)
                if(a[x]>a[p])
                    a[p]++;
            a[x]=1;
        }
    }
    cout<<t;
}
