#include<bits/stdc++.h>
using namespace std;
int n,m,k;
const int N=111;
int a[N];
int main()
{
	cin>>n>>m>>k;
	for (int i=1;i<=k;i++) cin>>a[i];
	int tot=0,tem,p;
	for (int o=1;o<=n*m;o++)
	{
		cin>>tem;
		for (int i=1;i<=k;i++)
		{
			if (a[i]==tem)
			{
				tot+=i;
				p=i;
				break;
			}
		}
		
		for (int i=p-1;i>=1;i--)
		{
			a[i+1]=a[i];
		}
		a[1]=tem;
	}
	cout<<tot;
}
