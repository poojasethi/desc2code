#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
int main()
{
    ll a[101];
	ll n,m,k,i,j,l,y,x;
	cin>>n>>m>>k;
	for(i=0;i<k;i++)cin>>a[i];
    
	ll sum=0;
	for(i=0;i<n;i++)for(j=0;j<m;j++){
	    cin>>y;
	for(l=0;l<k;l++)if(a[l]==y)x=l;
    sum+=x+1;
	for(ll d=x;d>=1;d--)a[d]=a[d-1];
	a[0]=y;
	//for(i=0;i<k;i++)cout<<a[i]<<" ";
	}
	cout<<sum<<endl;

	return 0;
}
