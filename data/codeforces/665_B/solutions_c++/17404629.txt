#include <cstdlib>
#include <iostream>
#include <string>

using namespace std;

int n,m,k,x,s,a[200];

int main()
{    
    cin >> n >> m >> k;
    for(int i = 1; i <= k; ++i) {
        cin >> x;
        a[x] = i;
    }
    for(int i = 0; i < n; ++i)
    for(int j = 0; j < m; ++j) {
        cin >> x;
        s += a[x];
        for(int l = 1; l <= k; ++l)
        if(a[l] <= a[x] && l != x) a[l]++;
        a[x] = 1;    
    }
    cout << s;
}
