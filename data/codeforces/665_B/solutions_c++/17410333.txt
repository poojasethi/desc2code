#include <bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,k,i,j,p[105],a[105][105],l,o,temp,t=0;
	cin>>n>>m>>k;
	for(i=1;i<=k;i++)
		cin>>p[i];
	for(i=0;i<n;i++)
		for(j=0;j<m;j++)
			cin>>a[i][j];
	for(i=0;i<n;i++)
	{
		for(j=0;j<m;j++)
		{
			for(l=1;l<=k;l++)
				if(a[i][j]==p[l])
					o=l;
			for(l=o;l>1;l--)
			{
				temp=p[l];
				p[l]=p[l-1];
				p[l-1]=temp;
			}
			t+=o;
		}
	}
	cout<<t<<endl;
	return 0;
}
