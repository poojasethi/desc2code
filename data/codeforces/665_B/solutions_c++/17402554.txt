#include <cstdio>
int ans=0,n,m,k;
int a[105];
int t;
int main()
{
    int i,j;
    scanf("%d%d%d",&n,&m,&k);
    for(i=1;i<=k;++i)
        scanf("%d",a+i);
    m*=n;
    for(j=1;j<=m;++j)
    {
        scanf("%d",&t);
        for(i=1;i<=k;++i)
            if(a[i]==t)
                break;
        ans+=i;
        for(;i>1;--i)
            a[i]=a[i-1];
        a[1]=t;
    }
    printf("%d\n",ans);
    return 0;
}
