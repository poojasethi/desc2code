#include<bits/stdc++.h>
using namespace std;
int n,m,k,ar[100000],pos[100000],ans,a;
int main(){
	scanf("%d %d %d",&n,&m,&k);
	for(int i=1 ; i<=k ; i++){
		scanf("%d",&ar[i]);
		pos[ar[i]]=i;
	}
	for(int i=1 ; i<=n ; i++){
		for(int j=1 ; j<=m ; j++){
			 scanf("%d",&a);
			 ans+=pos[a];
			 for(int i=pos[a]-1 ; i>0 ; i--){
			 	swap(ar[i],ar[i+1]);
				swap(pos[ar[i]],pos[ar[i+1]]);
			 }
		}
	}
	printf("%d\n",ans);
	return 0;
}
