#include<iostream>
using namespace std;
int main()
{
	int n,m,k;
	cin>>n>>m>>k;
	int a[k];
	for(int i=0;i<k;i++)
		cin>>a[i];
	int i,x,count=0;
	while(n--)
	{
		i=0;
		while(i<m)
		{
			cin>>x;
			for(int j=0;j<k;j++)
			{
				if(a[j]==x)
				{		
					count+=j+1;
					for(int s=j;s>0;s--)
					{
						a[s]=a[s-1];
					}
					a[0]=x;
					break;
				}
			}
			i++;
		}
	}
	cout<<count<<endl;
}
