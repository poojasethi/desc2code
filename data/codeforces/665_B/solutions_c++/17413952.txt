#include <bits/stdc++.h>
using namespace std;

int n, m, k, a[105], c;

int main()
{
    scanf("%d%d%d", &n, &m, &k);
    for (int i = 0; i < k; i++) {
        scanf("%d", &a[i]);
    }
    int cnt = 0;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            scanf("%d", &c);
            int idx = find(a,a+k,c)-a;
            cnt += (idx+1);
            rotate(a,a+idx,a+idx+1);
        }
    }
    printf("%d\n", cnt);
}
