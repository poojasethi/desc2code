#include <cstdio>

using namespace std;
const int maxn = 1e2 + 5;
int n, m, k, t, a, pos[maxn];

int main(){
	scanf("%d%d%d", &n, &m, &k);
	for(int i = 1; i <= k; i++) scanf("%d", &pos[i]);
	
	for(int i = 0; i < m*n; i++){
		scanf("%d", &a);
		for(int l = 1; l <= k; l++){
			if(pos[l] == a){
				t += l;
				for(int j = l; j > 1; j--) pos[j] = pos[j-1];
				pos[1] = a;
			}
		}
	}
		
	printf("%d\n", t);
	
	return 0;
}
