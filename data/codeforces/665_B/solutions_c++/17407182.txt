#include<iostream>
using namespace std;
int main()
{
	int n,m,k,ma[101],temp,i,j,l,h,ans=0,tar;
	cin>>n>>m>>k;
	for(i=1;i<=k;i++)
	{
		cin>>ma[i];
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			cin>>tar;
			for(l=1;l<=k;l++)
			{
				if(ma[l]==tar)
				{
					ans+=l;
					temp=ma[l];
					for(h=l;h>=2;h--)
					{
						ma[h]=ma[h-1];
					}
					ma[1]=temp;
					break;
				}
			}
		}
	}
	cout<<ans;
	return 0;
}
