#include <bits/stdc++.h>

using namespace std;

void rotate(int a[], int p)
{
	int i = p, temp = a[p];
	while(i > 0)
		a[i] = a[i-1], i--;
	a[0] = temp;
}

int main()
{
	int n, m, k, v, s = 0;
	cin >> n >> m >> k;
	int a[k];
	for(int i = 0; i < k; i++) cin >> a[i];
	for(int i = 0; i < n; i++)
		for(int j = 0; j < m; j++)
		{
			cin >> v;
			int x = 0;
			while(a[x] != v) x++;
			s += x+1;
			rotate(a, x);
		}
	cout << s << endl;
}
