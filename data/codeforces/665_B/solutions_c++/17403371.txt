#include <iostream>
#include <cstdio>
using namespace std;
int n,m,k,A[110],pos[110],Ans = 0,a;
int main() {
	cin>>n>>m>>k;
	for(int i = 1;i <= k;i ++)
		scanf("%d",&A[i]),pos[A[i]] = i;
	for(int i = 1;i <= n;i ++)
		for(int j = 1;j <= m;j ++) 
		{
			scanf("%d",&a);
			Ans += pos[a];
			for(int w = 1;w < pos[a];w ++)
				pos[A[w]] ++;
			pos[a] = 1;
			for(int w = 1;w <= k;w ++)
				A[pos[w]] = w;
		}
	cout<<Ans;
}
