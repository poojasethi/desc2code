int n,m,k;
int p[200];
int main(){
	scanf("%d%d%d",&n,&m,&k);int sol=0,t;
	for(int i=1;i<=k;i++){
		scanf("%d",&t);
		p[t]=i;
	}
	for(int i=1;i<=n;i++)
		for(int j=1;j<=m;j++){
			scanf("%d",&t);
			sol+=p[t];
			for(int tt=1;tt<=k;tt++)
				if(p[tt]<p[t])p[tt]++;
			p[t]=1;
		}
	printf("%d",sol);
	return 0;
}
