#include<bits/stdc++.h>
using namespace std;
int main()
{
	long long n,m,k,i,j,w,c=0,x;
	cin>>n>>m>>k;
	long long a[k+1],b[10001]={0};
	for(i=1;i<=k;i++)
	{
		cin>>a[i];
		b[a[i]]=i;
	}
	for(i=1;i<=n;i++)
	{
		for(j=1;j<=m;j++)
		{
			cin>>x;
		c=c+b[x];
		for(w=b[x]-1;w>=1;w--)
		{    a[w+1]=a[w];
	}
	
		a[1]=x;
		for(int i=1;i<=k;i++)
		{
			b[a[i]]=i;
		}
		
	}
	
}
cout<<c;
}
