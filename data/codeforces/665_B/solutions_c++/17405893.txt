#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,k;
	int a[105],b[105];
	cin>>n>>m>>k;
	for(int i=1;i<=k;i++)
	cin>>a[i];
	int ans=0;
	for(int i=1;i<=n*m;i++)
	{
		int x;
		cin>>x;
		int j;
		for(j=1;j<=k;j++)
		{
			if(a[j]==x)
			break;
		}
		//cout<<j<<endl;
		ans+=j;
		for(int l=j;l>=2;l--)
		{
			a[l]=a[l-1];
		}
		a[1]=x;
	}
	cout<<ans<<endl;
}
