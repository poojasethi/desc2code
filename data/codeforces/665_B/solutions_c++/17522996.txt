#include <iostream>

using namespace std;

int n,m,k;
int elem[200];

int main()
{
  cin>>n>>m>>k;
  for (int i=1;i<=k;i++)
    cin>>elem[i];
  int sol=0;
  for (int paso=0;paso<n*m;paso++) {
    int x;
    cin>>x;
    int i=1;
    while (elem[i]!=x) i++;
    sol+=i;
    for (;1<i;i--) swap(elem[i-1],elem[i]);
  }
  cout<<sol<<endl;
}
