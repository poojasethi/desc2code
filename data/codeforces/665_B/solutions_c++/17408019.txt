#include<bits/stdc++.h>
using namespace std;
int ara[100000];
int n,m,k,s,sum=0;

void update(int p){
    int x;
    for(int i=0;i<k;i++){
        if(ara[i]==p) {x=i;sum+=x+1;break;}
    }
    for(int i=x;i>=1;i--) ara[i]=ara[i-1];
    ara[0]=p;
}

int main(){
    scanf("%d %d %d",&n,&m,&k);
    for(int i=0;i<k;i++) {scanf("%d",&ara[i]);}
    for(int i=0;i<n*m;i++){scanf("%d",&s); update(s);}
    printf("%d",sum);
}
