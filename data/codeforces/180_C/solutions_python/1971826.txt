mess = raw_input()
before = [ [0], [0] ]
for i in mess:
    if i.islower():
        before[1].append(before[1][-1] + 1)
        before[0].append(before[0][-1])
    else:
        before[0].append(before[0][-1] + 1)
        before[1].append(before[1][-1])
        
ans = min(before[0][-1], before[1][-1])
for i in xrange(len(mess)):
    ans = min(ans, before[1][i] + before[0][-1] - (before[0][i] if i != 0 else 0))
print ans
