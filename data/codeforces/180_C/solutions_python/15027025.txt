# -*- coding: utf-8 -*-

def letter_problem(word):
    start = -1
    end = 1
    list_of_ones = []
    for letter in word:
        if letter == letter.upper():
            list_of_ones.append(start)
        else:
            list_of_ones.append(end)
    amount_moves = list_of_ones.count(start)
    minimum = amount_moves
    for one in list_of_ones:
        amount_moves += one
        minimum = min(minimum, amount_moves)
    print minimum
    
    
if __name__ == "__main__":
    word_problem = raw_input()
    letter_problem(word_problem)
