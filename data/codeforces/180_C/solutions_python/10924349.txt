
S = raw_input() 

def dp(S, l, r): 
    n = len(S)
    D = [0] * (n+1)
    for i, c in zip(range(1, n+1), S):
        if not (l <= c <= r):
            D[i] = D[i-1] + 1
        else:
            D[i] = D[i-1]

    return D

D = dp(S, 'A', 'Z')
R = dp(S[::-1], 'a', 'z')

n = len(S)
min_op = n

for l in range(n+1): 
    if D[l] + R[n-l] < min_op:
        min_op = D[l] + R[n-l]

print min_op
