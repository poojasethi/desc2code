def is_upper(x):
    if x >= 'A' and x <= 'Z':
        return True
    else:
        return False


s = raw_input()
length = len(s)
left_upper = [0] * length
if is_upper(s[0]):
    left_upper[0] = 1
for i in xrange(1, length):
    left_upper[i] = left_upper[i - 1]
    if is_upper(s[i]):
        left_upper[i] += 1

total = left_upper[length - 1]
ret = total

for i in xrange(length):
    a = i + 1 - left_upper[i] + total - left_upper[i]
    if a < ret:
        ret = a
print ret
