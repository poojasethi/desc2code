s=raw_input()
fend=0
while fend<len(s) and 'A'<=s[fend]<='Z':
    fend+=1
s=s[fend:]
if len(s)>0:
    numcap=[0 for i in range(len(s))]
    for i in range(len(s)):
        numcap[i]=numcap[i-1]+('A'<=s[i]<='Z')
    fend=numcap[-1]
    for i in range(len(s)):
        fend=min(fend,i+1-2*numcap[i]+numcap[-1])
    print fend
else:
    print 0
