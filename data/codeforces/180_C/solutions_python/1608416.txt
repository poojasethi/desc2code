import sys

s = sys.stdin.readline().rstrip('\n')

to_up = []
for c in s:
    l = to_up[-1] if to_up else 0
    to_up.append(l if c.isupper() else l + 1)

to_low = []
for c in s[::-1]:
    l = to_low[-1] if to_low else 0
    to_low.append(l if c.islower() else l + 1)
to_low.reverse()

z = zip([0] + to_up, to_low + [0])
print min(a+b for a,b in z)
