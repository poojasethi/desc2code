import sys
#sys.stdin = open("input.txt", "r")


s = raw_input()



if __name__ == '__main__':
    ans = 0
    nUppers = 0
    l = len(s)
    for i in range(l):
        if 'A' <= s[i] <= 'Z':
            nUppers += 1
    ans = min(nUppers, l - nUppers)
    cnt = 0
    for i in range(l):
        if 'A' <= s[i] <= 'Z':
            cnt += 1
        ans = min(ans, i + 1 - cnt + nUppers - cnt)
    print ans
