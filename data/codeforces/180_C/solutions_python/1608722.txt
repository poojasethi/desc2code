cad = raw_input().strip()
N = len(cad)

minusto = [0]*N
minusto[0] = int(cad[0].islower())
for i in xrange(1,N):
	minusto[i] = minusto[i-1] + int(cad[i].islower())


mayusfrom = [0]*N
mayusfrom[N-1] = int(cad[N-1].isupper())
for i in xrange(N-2,-1,-1):
	mayusfrom[i]=mayusfrom[i+1] + int(cad[i].isupper())


	
best = min(mayusfrom[0],minusto[N-1])
for i in xrange(N-1):
	best = min(best, mayusfrom[i+1]+minusto[i])

print best
