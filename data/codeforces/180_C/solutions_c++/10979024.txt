#include <bits/stdc++.h>
using namespace std;

int S[100000][2];
char s[100100];

int main(){
	scanf("%s",s);
	int L = strlen(s);
	for(int i = 1;i <= L;i++){
		S[i][0] = S[i - 1][0];
		S[i][1] = S[i - 1][1];
		S[i][s[i - 1] >= 'A' && s[i - 1] <= 'Z']++;
	}
	int ans = 1 << 30;
	for(int i = 0;i <= L;i++) 
		ans = min(ans,S[i][0] + S[L][1] - S[i][1]);
	printf("%d\n",ans);
	return 0;
}
