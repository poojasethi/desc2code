#include<bits/stdc++.h>

using namespace std;

const int Maxn=1e5+7;

int dp[Maxn];

main() 
{
  ios_base::sync_with_stdio(0);cin.tie(0);
  string s;
  cin>>s;
  for(int ted=0,i=0;i<s.size();i++)
    {
      if('a'<=s[i] && s[i]<='z')
	dp[i+1]=dp[i];
      else
	{
	  dp[i+1]=min(dp[i]+1,i-ted);
	  ted++;
	}
    }
  cout<<dp[s.size()]<<endl;
  return 0;
}
