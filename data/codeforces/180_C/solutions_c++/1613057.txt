#include <iostream>
#include <vector>
#include <string>
#include <math.h>
#include <stdio.h>
#include <string>
using namespace std;

int main() {
	string k;
	int l,i,s=0,s1=0,p=0,p1=0,cur;

	cin >> k;
	l=k.size();
	for (i=0;i<l;i++){
		if (isupper(k[i])) p++;
		else s++;
	}
	cur=p;
	for (i=0;i<l;i++){
		if (isupper(k[i])) p1++;
			else s1++;
		cur=min(cur,p-p1+s1);
	}
	cout << cur;

	return 0;
}
