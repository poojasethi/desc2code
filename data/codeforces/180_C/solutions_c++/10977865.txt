#include<bits/stdc++.h>
using namespace std;
int main() 
{
    string s; int ans = 0, count = 0; cin >> s;
    for(int i = 0; s[i]; ++i) {
        if(islower(s[i]))   
            ++count; 
        if(isupper(s[i]) && count) 
        {
            ++ans; --count;
        } 
    }
    cout << ans;
    return 0;
}
