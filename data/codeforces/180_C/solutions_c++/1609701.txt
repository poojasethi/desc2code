#include <iostream>
#include <string>
#include <string.h>
using namespace std;
int main()
{
	string s;
	int f[100010],n,i,j,k,l,m;
	cin>>s;
	n=s.length();
	f[0]=0;
	for (i=0;i<n;i++) {f[i+1]=f[i]+(s[i]>='A'&&s[i]<='Z');}
	m=10000000;
	for (i=0;i<=n;i++)
	{
		j=i-f[i]+f[n]-f[i];
		if (j<m) {m=j;}
	}
	cout<<m;
}
