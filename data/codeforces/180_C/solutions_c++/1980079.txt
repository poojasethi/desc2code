#include <iostream>
#include <cstdio>

using namespace std;

string s;
int t, ans = ( 1<< 30), i, n, d[200000], l;

int main(){
	cin >> s;	
	n = s.size();

	for ( i = 0; i < n; i++ )
	if ( s[i] >= 'a' ) d[i+1] = d[i] + 1; else d[i+1] = d[i];

	for (i = 0; i <= n; i++)
	{
		l = n - i;
		t = d[i] + (l - (d[n] - d[i]));
		ans = min(t, ans);
	}

	cout << ans;
 	return 0;
}
