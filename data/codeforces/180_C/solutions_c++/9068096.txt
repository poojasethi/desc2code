#include <bits/stdc++.h>
using namespace std;
int caps[100010]={0};

int main()
{
	string s;
	cin>>s;
	int n = s.size();
	s = " " + s;
	for(int i=1;i<=n ;i++)
	{
		if(s[i]>='A' && s[i] <='Z')
			caps[i]=caps[i-1]+1;
		else caps[i]=caps[i-1];
	}
	int res = n+2;
	for(int i=0;i<=n;i++)
	{
		res = min ( res, i-caps[i]+caps[n]-caps[i]);
	}
	cout<<res<<endl;
}
