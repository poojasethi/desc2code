#include <iostream>
#include <vector>

using namespace std;

int main(){
	string s;
	cin >> s;
	int n=s.size();
	vector <int>v(n,0);
	if(s[0]>='A' and s[0]<='Z')++v[0];
	for(int i=1;i<n;++i){
		v[i]=v[i-1];
		if(s[i]>='A' and s[i]<='Z')++v[i];
	}
	int ans=v[n-1];
	for(int i=0;i<n;++i){
		ans=min(ans,v[n-1]-2*v[i]+i+1);
	}	
	cout << ans << endl;
}
