#include<iostream>

using namespace std;

int main(){
	int n, g, f= 0, mine = 10000000;
	string s;
	cin >> s;
	for(int i = 0; i < s.size(); i++)
		if( (int) s[i] < 96)
		f++;
	g = f;
	mine = g;
	for(int i = 0; i < s.size(); i++)
		{
			if( (int) s[i] > 96)
				g++;
			else
				g--;
			mine = min(mine , g);
		}
	cout << mine;
}
