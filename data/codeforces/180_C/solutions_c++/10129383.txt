#include<bits/stdc++.h>
int main(){
    int ans=0,count=0;
    char str[100005];
    scanf("%s",str);
    for(int i=0;i<strlen(str);i++){
        if(str[i]>='a'&&str[i]<='z')
            count++;
        else if(count>0)
        {
            count--;
            ans++;
        }
    }
    printf("%d\n",ans);
    return 0;
}
