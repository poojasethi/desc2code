#include<stdio.h>
#include<string.h>
#include<ctype.h>
int i,s,n,H[1000000],min;
char arr[1000000];
int main(){
	scanf("%s",arr);
	n=strlen(arr);
	for(i=0;i<n;i++)
		if(isupper(arr[i]))
			s++;

	min = s;
	for(i=0;i<n;i++){
		if(isupper(arr[i]))
			H[i]=--s;
		else
			H[i]=++s;
	}
	for(i=0;i<n;i++)
		if(H[i]<min)
			min = H[i];
	printf("%d",min);
	return 0;
}
