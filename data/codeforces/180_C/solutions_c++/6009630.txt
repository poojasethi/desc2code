#include <iostream>
#include <string>

using namespace std;

int main()
{
	int ans = 1000000 , u = 0 , l = 0;
	string s;
	cin>>s;
	for(int i = 0 ; i < int(s.size()) ; i++)
		if(s[i] < 'a')
			u++;
	ans = u;
	for(int i = 0 ; i < int(s.size()) ; i++)
	{
		if(s[i] >= 'a')
			l++;
		else
			u--;
		ans = min(ans , l + u);
	}
	cout<<ans<<endl;
	return 0;
}
