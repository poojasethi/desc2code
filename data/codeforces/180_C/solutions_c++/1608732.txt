#include <cstdio>
#include <cstring>
#define fMIN(a,b) (a<b?a:b)
char s[100005];
int t[100005];
int main(){
    scanf("%s",s);
    int l=strlen(s),c=0;
    for (int i=0;i<l;++i){
        if (s[i]>='A' && s[i]<='Z')
            ++c;
        t[i]=c;
    }
    int min=fMIN(c,l-c);
    for (int i=0;i<l;++i){
        min=fMIN(min,i+1-t[i]+(c-t[i]));
    }
    printf("%d",min);
}
