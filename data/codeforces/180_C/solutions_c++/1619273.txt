#include <stdio.h>
#include <string.h>
#define N 100005
#define min(a,b) (a<b?a:b)
int main()
{
	int i,low=0,cap=0;
	char ch;
	while(scanf("%c",&ch)&&ch!='\n')
	{
		if(ch>='a'&&ch<='z')
			low=min(low,cap),cap=cap+1;
		else
			cap=cap,low=min(low,cap)+1;
	}
	printf("%d\n",min(low,cap));
	return 0;
}
