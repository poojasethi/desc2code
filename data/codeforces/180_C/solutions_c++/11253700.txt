#include "stdio.h"
#include "string.h"
#include "algorithm"
using namespace std;

char str[100005];
int dp[100005];
int n=0;
int len;

int main(){
	scanf("%s",str);
	len=strlen(str);
	for(int i=1;i<=len;i++){
		int ii=i-1;
		if(str[ii]>='A'&&str[ii]<='Z'){
			dp[i] = min(dp[i-1]+1,n);
		} else{
			dp[i] = dp[i-1];
			n++;
		}
	}
	printf("%d\n",dp[len]);
	return 0;
}
