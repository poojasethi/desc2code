#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

string S;
int d[100010],result=100000;

int main()
{
	cin >> S;
	for (int i=0;i<S.size();i++)
		d[i+1]=d[i]+isupper(S[i]);
	for (int i=0;i<=S.size();i++)
		result=min(result,i+d[S.size()]-2*d[i]);
	cout << result << endl;
	return 0;
}
