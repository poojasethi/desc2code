#include <iostream>
using namespace std;

string s;
long D[100001],P[100001],b,i,j,k,ans;


main()
{
      
 cin>>s;
 k=s.size();
 
 for (i=0;i<k;i++)
 {
   if  (s[i]>'Z') P[i]+=1; else D[i]+=1;
   P[i]+=P[i-1];
   D[i]+=D[i-1];
} 

   ans=100000;
   
   for (i=0;i<=k;i++)
      if (P[i]+D[k-1]-D[i]<ans)   ans=P[i]+D[k-1]-D[i];
cout<<ans<<endl;
      }
