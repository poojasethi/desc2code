# include <bits/stdc++.h>

using namespace std;

string a;
int t,s,ans;

int main(){
	cin >> a;

	for(int i=0 ; i<a.size() ; i++)
		if('A'<=a[i] && a[i]<='Z')
			t++;
	
	ans = t;
	for(int i=0 ; i<a.size() ; i++){
		if('a'<=a[i] && a[i]<='z')
			s++;
		ans = min(ans , s + (t - (i+1-s) ));
	}

	cout << ans << endl;
	
	return 0;
}
