#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

int main(){
  string a;
  cin>>a;
  int l[a.size()];
  int hl=0;
  for(int i=0;i<a.size();i++){
    l[i]=hl;
    if(a[i]>='a') hl++;
  }
  int ans=hl;
  for(int i=0;i<a.size();i++){
    int u=a.size()-i-(hl-l[i]);
    ans=min(ans,u+l[i]);
  }
  cout<<ans<<endl;
  return 0;
}
