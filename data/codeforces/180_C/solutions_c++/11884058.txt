#include <iostream>
#include <algorithm>

using namespace std;

int main()
{ int cLow=0,res=0;
    string s;
    cin>>s;
    for(int i=0;i<s.size();i++)
    {
        if(s[i]==tolower(s[i]))
            cLow++;

        else if(s[i]==toupper(s[i]) && cLow>0)
        {   res++;
            cLow--; }
    }
cout<<res;

    return 0;
}
