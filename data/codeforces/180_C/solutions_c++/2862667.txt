#include<cstdio>
#include<cstring>
using namespace std;

char s[100000];

int main(){
    int i,j,k,un,tn,ans;
    gets(s);
    un=0;
    for(i=0;i<strlen(s);i++){
        if(s[i]<97)un++;
    }
    ans=un;
    tn=0;
    for(i=0;i<strlen(s);i++){
        if(s[i]<97)tn++;
        if((i+1-tn+un-tn)<ans){ans=i+1-tn+un-tn;}
    }
    printf("%d",ans);
    return(0);
}
