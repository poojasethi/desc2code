#include <bits/stdc++.h>

using namespace std;

int a[100021];
string s;

int main() {
	cin >> s;
	for(int i = 0; i < s.size(); i++) {
		a[i + 1] = a[i] + isupper(s[i]);
	}
	int ans = 1e5;
	for(int i = 1; i <= s.size(); i++) {
		ans = min(i + a[s.size()] - 2 * a[i], ans);
	}
	ans = min(ans, a[s.size()]);
	cout << ans << endl;
	return 0;
}
