#include<iostream>
using namespace std;
int main()
{
	string s;
	cin>>s;
	int n= s.length();
	int f[n+2];
	int b[n+2];
	f[0]=0;
	b[n+1]=0;
	for(int i=1;i<=n;i++)
	{
		f[i] = f[i-1]+(s[i-1]>='A' && s[i-1]<='Z');
		b[n-i+1] = b[n-i+2] + (s[n-i]>='A' && s[n-i]<='Z'); 
	}
	int res= 1000000;
	for(int i=0;i<=n;i++)
	{
		res = min(res, (i-f[i]) + b[i+1]);
	}
	cout<<res;
}
