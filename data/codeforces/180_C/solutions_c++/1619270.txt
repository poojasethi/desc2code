#include <stdio.h>
#include <string.h>
#define N 100005
#define min(a,b) (a<b?a:b)
int main()
{
	int i,low=0,cap=0;
	char ch[N];
	scanf("%s",ch);
	int len=strlen(ch);
	for(i=0;i<len;i++)
	{
		if(ch[i]>='a'&&ch[i]<='z')
			low=min(low,cap),cap=cap+1;
		else
			cap=cap,low=min(low,cap)+1;
	}
	printf("%d\n",min(low,cap));
	return 0;
}
