
#include <stdio.h>
#include <string.h>
int main()
{
	int a, i, ans;
	char s[100005];
	while(scanf("%s", s) == 1){
		for(i = a = ans = 0;i < strlen(s);i ++){
			if(s[i] <= 'z' && s[i] >= 'a')
				a ++;
			if(s[i] <= 'Z' && s[i] >= 'A' && a > 0)
				a --, ans ++;
		}
		printf("%d\n", ans);
	}
	return 0;
}
