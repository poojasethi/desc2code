n, k = map(int, raw_input().split())
s = list(raw_input())
if k > 2:
	c, r = set(chr(ord('A') + i) for i in range(26)), 0
	for i in range(1, len(s)):
		if s[i - 1] == s[i]:
			r += 1
			s[i] = list(c - set([s[i - 1], s[i + 1] if i + 1 < len(s) else '']))[0]
	print r
	print ''.join(s)
else:
	f = lambda x, y : sum(x[i] != y[i] for i in range(len(x)))
	x, y = 'AB' * n, 'BA' * n
	z = [x, y][f(s, x) > f(s, y)]
	print f(s, z)
	print z[ : n]
