def diff(a,b):
    return sum([1 for x in xrange(len(a)) if a[x]!=b[x]])
n,k = map(int,raw_input().split())
s = list(raw_input()) + ['']
if k == 2:
    a ='AB'*(n/2)+'A'*(n%2)
    b ='BA'*(n/2)+'B'*(n%2)
    print min(diff(a,s),diff(b,s))
    print [a,b][diff(b,s) <= diff(a,s)]
else:
    c,p = set('ABC'),0
    for i in xrange(1,n):
        if s[i-1] == s[i]:
            p += 1
            s[i] = list(c-set(s[i:i+2]))[0]
    print p
    print ''.join(s)
