n,k=map(int, raw_input().split())
r=list(raw_input())
if k==2:
    s='AB'*n
    m=lambda x,y:sum(x[i] != y[i] for i in range(n))
    if m(s[1:n+1],r) < m(s[:n],r):
        print m(s[1:n+1],r)
        print s[1:n+1]
    else:
        print m(s[:n],r)
        print s[:n]
    exit()
c=0
s=set(map(lambda x:chr(x+ord('A')), range(k)))
for i in range(1,n):
    if r[i-1] == r[i]:
        c+=1
        r[i]=list(s-set([r[i-1]] +([] if i==n-1 else [r[i+1]])))[0]
print  c
print ''.join(r)
