#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
int dp[600010][27],prev[600010][27],a[600010];
char str[600010];

int main()
{
int n,k,i,j,l,q,min=600010,c,x,y;

scanf("%d%d",&n,&k);
scanf("%s",&str);
memset(dp,0,(n+1)*(k+1)*sizeof(int));

for(i=1;i<=n;i++)
{
for(j=1;j<=k;j++)
{
q=600010;
for(l=1;l<=k;l++)
{
if(j!=l && dp[i-1][l]<q)
{
q=dp[i-1][l];
c=l;
}
}

if(j==str[i-1]-64)
dp[i][j]=q;
else
dp[i][j]=q+1;

prev[i][j]=c;

if(i==n && dp[i][j]<min)
{
min=dp[i][j];
y=j;
}
}
}

printf("%d\n",min);
x=n;

while(x)
{
a[x]=y;
y=prev[x][y];
x--;
}

for(i=1;i<=n;i++)
{
printf("%c",a[i]+64);
}
printf("\n");
return 0;
}
