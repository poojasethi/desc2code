#include<bits/stdc++.h>
using namespace std;
int n,k;
string s;
string a,b;
void init(){
	for(int i=0;i<n;i++){
		if(i&1)a+="A",b+="B";
		else a+="B",b+="A";
	}
}
int get(int x,int y){
	for(int i=0;i<k;i++)
		if(i!=x&&i!=y)return i;
}
int main(){
	while(cin>>n>>k){
		cin>>s;
		if(k==2){
			init();
			int cnt1,cnt2;
			cnt1=cnt2=0;
			for(int i=0;i<n;i++){
				if(s[i]!=a[i])++cnt1;
				if(s[i]!=b[i])++cnt2;
			}
			if(cnt1>cnt2)
				cout<<cnt2<<endl<<b<<endl;
			else
				cout<<cnt1<<endl<<a<<endl;
			continue;
		}
		int ans=0;
		for(int i=0;i<n;i++){
			if(s[i]==s[i+1]){
				++ans;
				int j=0;
				while(j==s[i]-'A'||j==s[i+2]-'A')++j;
				s[i+1]=j+'A';
			}
		}
		cout<<ans<<endl<<s<<endl;
	}
	return 0;
}
