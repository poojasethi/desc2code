#include<stdio.h>
char s[500005];
int main()
{
	int i, j, n, k, ans = 0, d = 0;
	scanf("%d%d%s", &n, &k, s);
	if(k == 2)
	{
		for(i = 0; i < n; i++)
			if(s[i] == 'A' + (i&1) ) d++;
		ans = n - d < d ? n - d : d;
		for(i = 0; i < n; i++)
			s[i] = 'A' +( ( (d == ans) + i )&1 );
	}
	else
	{
		for(i = 0; i < n; i++)
		{
			if(s[i] == s[i+1])
			{
				ans++;
				char c = 'A';
				while(c == s[i] || c == s[i+2])
					c++;
				s[i+1] = c;
			}
		}
	}
	printf("%d\n", ans);
	puts(s);
	return 0;
}
