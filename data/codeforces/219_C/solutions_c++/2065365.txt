#include<cstdio>
#include<algorithm>
#define N 500010
using namespace std;
char a[N], ans[N];
int pr[N][26], dp[N][26];
int main(){
	int n, c;
	scanf("%d%d%s", &n, &c, a);
	for(int i = 0; i < c; i++) dp[0][i] = (a[0]-'A'!=i), pr[0][i] = -1;
	for(int i = 1; i < n; i++){
		for(int j = 0; j < c; j++){
			dp[i][j] = 999999999;
			for(int k = 0; k < c; k++){
				if(j == k) continue;
				if(dp[i][j] > dp[i-1][k])
					dp[i][j] = dp[i-1][k], pr[i][j] = k;
			}
			if(j != a[i]-'A') dp[i][j]++;
		}
	}
	int x = 0, bst = 999999999;
	for(int i = 0; i < c; i++) if(dp[n-1][i] < bst) bst = dp[n-1][i], x = i;
	printf("%d\n", bst);
	for(int i = x, j = n-1; j >= 0; i = pr[j][i], j-- ) ans[j] = i+'A'; ans[n] = 0;
	printf("%s\n",ans);
}
