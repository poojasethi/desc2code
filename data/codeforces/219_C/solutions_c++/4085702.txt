#include<iostream>
#include<cstdio>
#include<string>

using namespace std;

int n;
string s;

int main(){
	cin>>n>>n>>s;
	if(n==2){
		string s1,s2;
		int n1,n2;
		for(int i=0; i<s.size(); i++){
			s1+=(i&1)?'A':'B';
			s2+=(i&1)?'B':'A';
			if(s1[i]!=s[i]) n1++;
			if(s2[i]!=s[i]) n2++;
		}
		cout<<min(n1,n2)<<endl;
		cout<<((n1<n2)?s1:s2)<<endl;
	}else{
		int con=0;
		for(int i=1; i<s.size(); i++){
			if(s[i]==s[i-1]){ 
				con++;
				while(s[i]==s[i-1] or s[i]==s[i+1]){
					s[i]++;
					if(s[i]-'A'>=n) s[i]='A';
				}
			}
		}
		cout<<con<<endl;
		cout<<s<<endl;
	}
}
