#include <iostream>
#include <string>

using namespace std;

int main(int argc, char *argv[])
{
    int n, k, r = 0;
    string s;
    cin>>n>>k>>s;
    if (k == 2)
    {
	int p1 = 0, p2 = 0;
	for (int i = 0; i < n; i++)
	{
	    if (s[i] - 'A' == i % 2) p1++;
	    else p2++;
	}
	if (p1 > p2)
	    for (int i = 0; i < n; i++)
		s[i] = i % 2 + 'A';
	else
	    for (int i = 0; i < n; i++)
		s[i] = 1 - i % 2 + 'A';
	cout<<min(p1, p2)<<'\n'<<s<<endl;
    }
    else
    {
	for (int i = 1; i < n; i++)
	{
	    if (s[i] == s[i - 1])
	    {
		char c = 'A';
		while (c == s[i] || c == s[i + 1])
		    c++;
		s[i] = c;
		r++;
	    }
	}
	cout<<r<<'\n'<<s<<endl;
    }
}
