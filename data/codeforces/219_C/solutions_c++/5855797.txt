#include<cstdio>
#include<algorithm>
using namespace std;

char s[1111111];
int k,l,x,y;

int main() {
	scanf("%d%d", &l,&k);
	scanf("%s",s);
	if(k==2){
		x=y=0;
		for(int i=0;i<l;i++)
			if(s[i]=='A'+(i&1)) x++;
			else y++;
		printf("%d\n",l-max(x,y));
		for(int i=0;i<l;i++)
			printf("%c", 'A'+(((x>y?0:1)+i)&1));
		printf("\n");
	} else {
		x=0;
		for(int i=0;i<l;i++)
			if (i && s[i]==s[i-1]) {
				x++;
				s[i]='A';
				while(s[i]==s[i-1] || (i<l-1 && s[i]==s[i+1]))
					s[i]++;
			}
		printf("%d\n%s\n",x,s);
	}

	return 0;
}
