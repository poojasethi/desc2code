#include <bits/stdc++.h>
using namespace std;
int main() {
	int i,j,k,l,n,m,a1=0,a2=0;
	string c;
	char v;
	cin>>n>>k>>c;
	if(k==2) {
		for(i=0;i<n;i++) {
			if(c[i]!=i%2+'A')
				a1++;
			if(c[i]!=(i+1)%2+'A')
				a2++;
		}
		cout<<min(a1,a2)<<endl;
		if(a1<a2)
			for(i=0;i<n;i++) {
				v = i%2+'A';
				cout<<v;
			}
		else
			for(i=1;i<=n;i++) {
				v = i%2+'A';
				cout<<v;
			}
		cout<<endl;

	}
	else {
		for(i=1;i<n;i++)
			if(c[i]==c[i-1])
				for(j=0;j<k;j++) 
					if(c[i-1]!=j+'A' && c[i+1]!=j+'A') {
						c[i] = j+'A';
						a1++;
						break;
					}
		cout<<a1<<endl<<c<<endl;
	}
	return 0;
}
