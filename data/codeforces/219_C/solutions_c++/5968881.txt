#include<iostream>
#include<string>
#include<algorithm>
using namespace std;
int main() {
    int n, k, d=0, a=0;
    string s;
    cin >> n >> k >> s;
    if (k == 2){
        for (int i = 0; i < n; i++){
            if (s[i] == 'A' + (i & 1)) d++;
        }
        a = min(n - d, d);
        for (int i = 0; i<n; i++)
            s[i] = 'A' + (((d == a) + i) & 1);
    }
    else
        for (int i = 0; i<n; i++)
            if (s[i] == s[i + 1])
            {
                a++;
                char c = 'A';
                while (c == s[i] || c == s[i + 2])
                    c++;
                s[i + 1] = c;
            }
    cout << a << '\n' << s;
}
