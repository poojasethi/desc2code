#include<stdio.h>
#include<algorithm>
using namespace std;
char c[1000000];
int fuck(int n,int k,char c[])
{
	int i,j,s=0;
	for(i=1;i<n;i++)
	{
		if(c[i]==c[i-1])
		{
			s++;
			for(j=0;j<k;j++)
			{
				c[i]='A'+j;
				if(k!=2&&c[i]!=c[i-1]&&c[i]!=c[i+1])
                    break;
				else if(k==2&&c[i]!=c[i-1])
                    break;
			}
		}
	}
	return s;
}
int main()
{
	int n,k;
	while(scanf("%d%d",&n,&k)!=EOF)
	{
		getchar();
		scanf("%s",c);
		int sum,i;
		sum=fuck(n,k,c);
		if(k==2&&sum>n/2)
		{
			sum=n-sum;
			for(i=0; i<n; i++)
			{
				if(c[i]=='A')
                    c[i]='B';
				else c[i]='A';
			}
		}

}
