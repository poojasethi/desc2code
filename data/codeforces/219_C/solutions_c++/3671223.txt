#include<stdio.h>
#include<string.h>
#define N 500010
char s[N],l[N],r[N];
int al,ar;
int main(){
	int n,k;
	while(scanf("%d%d",&n,&k)!=EOF){
		scanf("%s",s);
		int len=strlen(s),i,j;
		if(k>2){
			int res=0;
			for(i=1;i<len;i++){
				if(s[i-1]==s[i]){
					for(j=0;j<3;j++){
						char x=j+'A';
						if(x!=s[i-1]&&x!=s[i+1]){
							s[i]=x;
							}
						}
					res++; 
					}
				}
			printf("%d\n%s\n",res,s);
			continue;
			}
		for(i=0;i<len;i++)l[i]= (i&1) ? 'A':'B';
		for(i=0;i<len;i++)r[i]= (i&1) ? 'B':'A';
		al=0;ar=0;
		for(i=0;i<len;i++)if(l[i]!=s[i])al++;
		for(i=0;i<len;i++)if(r[i]!=s[i])ar++;
		if(ar>al)printf("%d\n%s\n",al,l);
		else printf("%d\n%s\n",ar,r);
		}
	return 0;
	}
