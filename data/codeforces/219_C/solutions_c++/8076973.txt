#include<iostream>
#include<string>
#include<stdio.h>
#include<cstdio>
#include<map>
#include<cmath>
#include<algorithm>
using namespace std;

int main(){
int n,k;cin>>n>>k;
int d=0,ans=0;
char a[5000005];cin>>a;
    if(k==2){
        for(int i=0;i<n;i++){
            if(a[i]=='A'+(i&1))d++;
        }
        if(n-d<d)ans=n-d;
        else ans=d;
        for(int i=0;i<n;i++){
            a[i]='A'+(((d==ans)+i)&1);
        }
    }
    else{
        for(int i=0;i<n;i++){
            if(a[i]==a[i+1]){
                ans++;
                char c='A';
                while(c==a[i]||c==a[i+2]){
                    c++;
                }
                a[i+1]=c;
            }
        }
    }
    cout<<ans<<endl<<a;
}
