#include<cstdio>
#include<string>
#include<iostream>
using namespace std;
int main(){
  string s, str1, str2;
  int n, k, cnt = 0;
  cin>>n>>k>>s;
  int len = s.size();
  if(k == 2){
    int cnt1 = 0, cnt2 = 0;
    for(int i = 0; i < len; i++){
      char t = 'A'+(i%2);
      str1+=t;
      if(t!=s[i]) cnt1++;
      t = 'A'+((i+1)%2);
      str2+=t;
      if(t!=s[i]) cnt2++;
    }
    if(cnt1 < cnt2 ) cout<<cnt1<<endl<<str1<<endl;
    else cout<<cnt2<<endl<<str2<<endl;
    return 0;
  }
  for(int i = 1; i < n; i++){
    if(s[i] == s[i-1]){
      cnt++;
      for(int j = 0; j < k; j++){
        if('A'+j != s[i-1] && 'A'+j != s[i+1]){
          s[i] = 'A'+j;
        }
      }
    }
  }
  cout<<cnt<<endl<<s<<endl;
  return 0;
}
