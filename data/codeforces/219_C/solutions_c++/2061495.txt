#include <iostream>
#include <string>
using namespace std;
int main()
{
	int n,k;
	while (cin>>n>>k) {
		string s; cin>>s;
		if (k==2) {
			string t,u;
			for(int i=0;i<s.size();++i) { t+=char('A'+i%2); u+=char('A'+1-i%2); }
			int c0=0,c1=0;
			for(int i=0;i<s.size();++i) {
				c0+=(s[i]!=t[i]);
				c1+=(s[i]!=u[i]);
			}
			if (c0<c1) { cout<<c0<<'\n'<<t<<endl; }
			else { cout<<c1<<'\n'<<u<<endl; }
			continue;
		}
		s+='\n';
		int cnt=0;
		for (int i=1;i<s.size();++i) if (s[i-1]==s[i])
			for(int j=0;j<k;++j) if ('A'+j!=s[i-1] && 'A'+j!=s[i+1]) {
				s[i]='A'+j;
				cnt++;
				break;
			}

		cout<<cnt<<'\n'<<s;
	}
	return 0;
}
