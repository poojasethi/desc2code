#include<iostream>
#include<algorithm>
using namespace std;
main(){
int n,k;
cin>>n>>k;
string s;
cin>>s;
int dp[n][k];
int pop[n][k];
for(int i=0;i<k;i++)dp[0][i]=1;
dp[0][s[0]-'A']=0;
for(int i=1;i<n;i++){
for(int j=0;j<k;j++){
dp[i][j]=1000000000;
for(int q=0;q<k;q++){
if(j==q)continue;
if(dp[i-1][q]+(s[i]!=char(j+'A'))<dp[i][j]){dp[i][j]=dp[i-1][q]+(s[i]!=char(j+'A'));pop[i][j]=q;}
}
}
}
int w=1000000000;
int l;
for(int i=0;i<k;i++){if(w>dp[n-1][i]){w=dp[n-1][i];l=i;}}
cout<<w<<"\n";
string s1;
for(int i=n-1;i>=0;i--){
//cout<<"l "<<l<<"\n";
s1+=char('A'+l);
if(i==0)continue;
l=pop[i][l];
}
for(int i=n-1;i>=0;i--)cout<<s1[i];
cout<<"\n";
}
