int n, k;

int main(void) {
	scanf("%d%d", &n, &k);
	scanf("%s", s);
	int ans=0;
	if(k==2) {
		int d=0;
		for(int i=0;i<n;i++) d+=(((s[i]=='A')^(i%2))?1:-1);
		for(int i=0;i<n;i++) { char v=((d<=0)^(i%2))+'A'; ans+=(s[i]!=v); s[i]=v;}
	} else {
		for(int i=0;i+1<n;i++)
			if(s[i]==s[i+1]) {
				char w='A';
				if(s[i]==w || s[i+2]==w) ++w;
				if(s[i]==w || s[i+2]==w) ++w;
				++ans;
				s[i+1]=w;
			}
		
	}
	printf("%d\n", ans);

}
