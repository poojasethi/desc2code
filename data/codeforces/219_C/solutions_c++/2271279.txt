#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <cstdlib>
#define N 500009
using namespace std;
int f[N][30];
int p[N][30];
int a[N];
main(){
	int n,k,i,j,l,ans,t;
	//memset(f,1,sizeof(f));
	
	  cin>>n>>k;
	getchar();
	for (i=1;i<=n;i++) {char ch=getchar();a[i]=ch-'A';}
	for (i=0;i<k;i++) {f[n][i]=(a[n]!=i);}
	for (i=n-1;i>=1;i--)
	for (l=0;l<k;l++){
		ans=N;
		for (t=0;t<k;t++)
		if (l!=t){
			if (ans>f[i+1][t]){
				ans=f[i+1][t];
				p[i][l]=t;
			}
		}
		f[i][l]=ans+(l!=a[i]);
		//cout<<i<<" "<<l<<" "<<f[i][l]<<endl;
	}
	ans=0;
	for (i=0;i<k;i++)
		if (f[1][i]<f[1][ans]) ans=i;
	cout<<f[1][ans]<<endl;
	for (i=1;i<=n;i++){
		cout<<(char)(ans+'A');
		ans=p[i][ans];
	}  
	
}
