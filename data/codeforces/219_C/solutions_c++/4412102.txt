#include <bits/stdc++.h>
#define F(i,b,e) for(int i=(b), _e=(e); i<_e; i++)
using namespace std;

int main(){

    int n, k;
    string ln;

    cin>>n>>k>>ln;

    int cnt=0;

    if(k==2){
        int ta=0, tb=0;
        F(i, 0, n){
            if(ln[i] == 'A'+(i&1)) ta++;
            if(ln[i] == 'B'-(i&1)) tb++;
        }
        cnt = min(ta, tb);
        F(i, 0, n)
            ln[i]=(ta>tb?'A'+(i&1):'B'-(i&1));
    }else{
        F(i,0,n) if(ln[i]==ln[i+1]){
            cnt++;
            F(j,0,k) if(ln[i]!='A'+j && ln[i+2]!='A'+j){
                ln[i+1]='A'+j;
                break;
            }
        }
    }

    cout<<cnt<<endl<<ln<<endl;

    return 0;
}
