#include <iostream>
#include <string>
#include <cstdio>

using namespace std;

int n,k;
string s;

int main(void)
{
	cin>>n>>k;
	cin>>s;
	int A=0,B=0;
	if(k==2)
	{
		for(int i=0;i<n;i+=2) if(s[i]=='B') A++; else B++;
        for(int i=1;i<n;i+=2) if(s[i]=='A') A++; else B++;
        printf("%d\n",min(A,B));
        for(int i=0;i<n;i++) putchar((A<B?i&1:~i&1)+'A');
	}
    else
    {
    	int cnt=0;
    	for(int i=0;i<n;i++)
    	{
    		if(i && s[i]==s[i-1])
    		{
    			for(s[i]='A';s[i]==s[i-1]||s[i]==s[i+1];s[i]++);
    			cnt++;
    		}
    	}
    	cout<<cnt<<endl;
    	for(int i=0;i<n;i++)
    		cout<<s[i];
    }
    return 0;
}
