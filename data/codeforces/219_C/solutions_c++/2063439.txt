#include<iostream>
#include<algorithm>
#include<string>

using namespace std;

char pick_color(char a, char b, int k)
{
    for(int i=0;i<k;i++)
	if('A'+i!=a && 'A'+i!=b)
	    return 'A'+i;
    return b;
}

int main()
{
    int n,k;
    string s;
    cin>>n>>k>>s;
    int sz=s.size();
    int i,j;
    if(k==2){
	int cnt1,cnt2;
	cnt1=0;
	cnt2=0;
	for(i=0;i<sz;i++)
	    if(s[i]!='A'+i%2)
		cnt1++;
	    else
		cnt2++;
	cout<< (cnt1<cnt2 ? cnt1 : cnt2)<<endl;
	for(i=0;i<sz;i++)
	    cout<< (char)(cnt1<cnt2? 'A'+i%2 : 'B'-i%2);
	cout<<endl;
    }
    else{
	int cnt=0;
	for(i=1;i<sz;i++)
	    if(s[i]==s[i-1]){
		cnt++;
		s[i]=pick_color(s[i],s[i+1],k);
	    }
	cout<<cnt<<endl;
	for(i=0;i<sz;i++)
	    cout<<s[i];
	cout<<endl;
    }
    return 0;
}
