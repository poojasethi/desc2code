#include <stdio.h>
#include <string.h>
#define MAX 500010
#define INF 1000000

char ch[MAX], print[MAX];
int len, K, min;
int dy[MAX][30], via[MAX][30];

int main(void){
	int i, j, k, T;
	scanf("%d %d", &len, &K);
	scanf("%s", &ch[1]);
	for (i=1;i<=len;i++){
		for (j=1;j<=K;j++){
			dy[i][j]=INF;
			for (k=1;k<=K;k++){
				if (j==k) continue;
				T=(j+'A'-1==ch[i]) ? 0 : 1;
				if (dy[i][j]>dy[i-1][k] + T){
					dy[i][j]=dy[i-1][k] + T;
					via[i][j]=k;
				}
			}
		}
	}
	min=INF;
	for (i=1;i<=K;i++){
		if (min>dy[len][i]){
			min=dy[len][i];
			k=i;
		}
	}
	printf("%d\n", min);
	i=len;
	for (;;){
		print[i]=k+'A'-1;
		k=via[i][k];
		i--;
		if (i==0) break;
	}
	printf("%s\n", &print[1]);
	return false;
}
