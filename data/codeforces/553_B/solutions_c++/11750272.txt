#include <cstdio>
typedef long long ll;
const int N = 51;
ll f[N], k;
int n, m;

int main() {
	f[0] = f[1] = 1;
	for (int i = 2; i < N; i++) f[i] = f[i-1]+f[i-2];
	scanf("%d%I64d", &n, &k);
	while (n > m) {
		if (f[n-m-1] < k) {
			k -= f[n-m-1];
			printf("%d %d ", m+2, m+1);
			m += 2;
		} else {
			printf("%d ",  m+1);
			m++;
		}
	}
	return 0;
}
