#include <bits/stdc++.h>
using namespace std;

typedef long long ll;
#define fr(i,a,b)   for(int i=a;i<b;i++)

ll fib[52];
void _fib()
{
    fib[0] = 0; fib[1]=1;
    fr(i,2,52)  fib[i] = fib[i-1]+fib[i-2];
}

int main()
{
    _fib();
    int n;  cin>>n;

    ll k; cin>>k;
    fr(i,0,n)
    {
        if(fib[n-i]<k)  {cout<<++i+1<<' '<<i<<' ';  k-=fib[n-i+1];}
        else cout<<i+1<<' ';
    }
}
