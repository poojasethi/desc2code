#include <iostream>
#include <cstdio>
 using namespace std;

int main()
{
		long long llim=10e9;
		llim*=2*llim;
		
		long long pave[60];
		pave[0]=1;
		pave[1]=1;
		for(int i = 2; i < 60; i++) 
			pave[i]=pave[i-1]+pave[i-2];

		long long n, k;
		cin>>n>>k;

		for(int i = 1; i <= n; i++) {
			if(k <= pave[n-i])
				printf("%d ", i);
			else{
				printf("%d %d ", i+1, i);
				k -= pave[n-i];
				i++;
			}
		}
		printf("\n");
		return 0;
}
