#include<bits/stdc++.h>
using namespace std;

int main() {
	long long len,n,k,i,j,a,b,c;
	scanf("%lld%lld",&n,&k);
	long long A[n];
	long long fib[n];
	fib[0] = fib[1] = 1;
	for(i=2;i<n;i++)
		fib[i] = fib[i-1]+fib[i-2];
	len=0;
	i--;
	while(len<n) {
		if(k>fib[i]) {
			k-=fib[i];
			i-=2;
			A[len] = len+2;
			A[len+1] = len+1;
			len+=2;
		}
		else {
			i--;
			A[len] = len+1;
			len++;
		}
	}
	for(i=0;i<n;i++)
		printf("%lld ",A[i]);
	printf("\n");
	return 0;
}
