#include<bits/stdc++.h>
using namespace std;

typedef long long int ll;

ll tf[59];
int twyn[59];


int main()
{
	int n,pocz=1;
	ll k,sf=0;
	scanf("%d%lld",&n,&k);
	tf[0]=1;
	tf[1]=1;
	for(int i=2;i<=n;++i) tf[i]=tf[i-1]+tf[i-2];
	while(pocz<=n)
	{
		if(sf+tf[n-pocz]>=k)
		{
			twyn[pocz]=pocz;
			++pocz;
		}
		else
		{
			twyn[pocz]=pocz+1;
			twyn[pocz+1]=pocz;
			sf+=tf[n-pocz];
			pocz+=2;
		}
	}
	for(int i=1;i<=n;++i) printf("%d ",twyn[i]);
	return 0;
}
