#include <iostream>

using namespace std;

const int MAXN = 60;

long long fib[MAXN];

int main()
{
	int n;
	long long k;
	cin >> n >> k;
	fib[0] = 1;
	fib[1] = 1;
	for (int i = 2; i <= n; i++)
		fib[i] = fib[i - 1] + fib[i - 2];
	int i = 0;
	while (i < n)
	{
		if (fib[n - i - 1] < k)
		{
			k -= fib[n - i - 1];
			cout << i + 2 << " " << i + 1 << " ";
			i += 2;
		}
		else
		{
			cout << i + 1 << " ";
			i++;
		}
	}
	cout << endl;


}
