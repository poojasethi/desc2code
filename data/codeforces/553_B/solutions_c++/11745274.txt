#include <bits/stdc++.h>
using namespace std;
long long fib[61];
int main() {
  fib[1] = 1;
  fib[2] = 1;
  for (int i = 3; i <= 60; i++) {
    fib[i] = fib[i - 1] + fib[i - 2];
  }
  int n;
  long long k;
  cin >> n >> k;
  for (int i = n, j = 1; i >= 1; i--, j++) {
    if (k > fib[i]) {
      cout << j + 1 << " " << j << " ";
      k -= fib[i];
      i--;
      j++;
    } else {
      cout << j << " ";
    }
  }
  return 0;
}
