#include<bits/stdc++.h>
using namespace std;
typedef long long int ll;
int N;
ll K;
ll fib[55];
int ans[55];
int main()
{
	scanf("%d%lld",&N,&K);
	fib[0]=1;fib[1]=2;
	for(int i=2;i<55;i++){
		fib[i]=fib[i-1]+fib[i-2];
	}
	for(int i=0;i<N;i++)ans[i]=i+1;
	for(int i=0;i<N-1;i++){
		if(K>fib[N-i-2]){
			K-=fib[N-i-2];
			swap(ans[i],ans[i+1]);
			i++;
		}
	}
	printf("%d",ans[0]);
	for(int i=1;i<N;i++)printf(" %d",ans[i]);
	printf("\n");
}
