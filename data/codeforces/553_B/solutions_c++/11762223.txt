#include<cstdio>
using namespace std;
int n;
long long f[111],K;
int main()
{
	//freopen("1.in","r",stdin);freopen("1.out","w",stdout);
	scanf("%d%I64d",&n,&K);
	f[0]=1,f[1]=1;
	int i;
	for (i=2;i<=n;i++) f[i]=f[i-1]+f[i-2];
	for (i=1;i<=n;)
		if (K<=f[n-i])
		{
			printf("%d ",i);
			i++;
		}
		else
		{
			K-=f[n-i];
			printf("%d %d ",i+1,i);
			i+=2;
		}
	return 0;
}
