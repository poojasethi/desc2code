#include<cstdio>

int n;
long long k, f[55];

int main() {
	scanf("%d%I64d", &n, &k);
	f[0] = f[1] = 1LL;
	for(int i=2;i<=n;i++)
		f[i] = f[i-1]+f[i-2];
	
	for(int i=1;i<=n;) {
		if(k<=f[n-i]) {
			printf("%d", i);
			i++;
		} else {
			k -= f[n-i];
			printf("%d %d", i+1, i);
			i+=2;
		}
		printf("%c", i>n ? '\n' : ' ');
	}
	
	return 0;
}
