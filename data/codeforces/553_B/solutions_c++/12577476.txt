#include <bits/stdc++.h>
#define N 55
using namespace std;
typedef long long ll;
ll fib[N]={1,1};
int main()
{
    int n,i;
    ll k;
    scanf("%d %I64d",&n,&k);
    for(i=2;i<=n;i++)
        fib[i]=fib[i-1]+fib[i-2];
    i=1;
    while(n>0){
        if(k>fib[n-1]){
            printf("%d %d ",i+1,i);
            k-=fib[n-1]; i+=2; n-=2;
        }
        else{
            printf("%d ",i++);
            n--;
        }
    }
    return 0;
}
