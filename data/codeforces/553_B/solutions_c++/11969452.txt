#include<iostream>

#define rep(i,a) for(i=0;i<a;++i)
#define f(i,a,b) for(i=a;i<b;++i)
#define ll long long

using namespace std;

ll N,k,f[100];

void ans(ll n, ll k)
{
	if(n==1)
	{
		cout<<N;
		return;
	}
	if(n==0)
		return;
	if(k<=f[n-1])
	{
		cout<<N-n+1<<" ";
		ans(n-1,k);
	}
	else
	{
		cout<<N-n+2<<" "<<N-n+1<<" ";
		ans(n-2,k-f[n-1]);
	}
}

int main()
{
	cin>>N>>k;
	ll i,j;

	f[1] = 1;
	f[2] = 2;
	f(i,3,N+1)
		f[i] = f[i-1] + f[i-2];

	ans(N,k);
}
