#include <cstdio>
#include <algorithm>
using namespace std;
int a,b,c,d,e,g,h,i,j,l,m,n;
long long f[55],k;
int main(){
	scanf("%d%I64d",&n,&k);
	f[0]=f[1]=1;
	for(i=2;i<=n;i++)f[i]=f[i-1]+f[i-2];
	for(i=1;i<=n;i++){
		if(f[n-i]<k){
			k-=f[n-i];
			printf("%d %d ",i+1,i);
			i++;
		}
		else printf("%d ",i);
	}
	return 0;
}
