#include<cstdio>
#include<cstdlib>
#define N 100
#include<iostream>
using namespace std;

int n;
long long m;
long long f[N];
int ans[N];
int main()
{
	scanf("%d%I64d",&n,&m);
	m--;
	f[1]=0;
	for (int i=2;i<=n;i++)
		f[i]=f[i-1]+f[i-2]+1;
	int now=0;
	for (int i=1;i<=n;i++) ans[i]=n-i+1;
	for (int i=n;i>=2;i--)
	{
		if (f[i-1]<m) {swap(ans[i],ans[i-1]);m-=f[i-1];if (m==0) break;else m--;}
	}
	for (int i=n;i>=1;i--) printf("%d ",ans[i]); 
}
