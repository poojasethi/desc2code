#include <bits/stdc++.h>

using namespace std;

typedef pair<int, int>  ii;

const int N = 2e5 + 10 ;

long long fib[55], n, k;

int ans[55];

int main ()
{
	fib[0] = fib[1] = 1;
	for(int i = 2; i<55; i++) {
		fib[i] = fib[i-1]+fib[i-2];
	}
	cin >> n >> k;
	for(int i = 0; i<n; i++)
		ans[i] = n-i;
	k--;
	for(int i = n-1; i>0; i--) {
		if(k>=fib[i]) {
			k-= fib[i];
			swap(ans[i], ans[i-1]);
		}
	}
	for(int i = n-1; i>=0; i--)
		printf("%d ", ans[i]);
  	return 0;
}
