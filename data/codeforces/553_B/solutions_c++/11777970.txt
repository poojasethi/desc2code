#include<bits/stdc++.h>
using namespace std;

typedef long long ll;
ll n, k, fib[1000];
int main(){
	scanf("%lld%lld", &n, &k);
	fib[0] = fib[1] = 1;
	for (int i=2; i<=n+2; i++){
		fib[i] = fib[i-1] + fib[i-2];
	}
	for (int i=1; i<=n; i++){
		if (k > fib[n-i]){
			k -= fib[n-i];
			printf("%d %d ", i+1, i);
			i++;
		}
		else printf("%d ", i);
	}
	puts("");
	return 0;
}
