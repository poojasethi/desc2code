#include<bits/stdc++.h>
using namespace std;
typedef long long ll;
ll fib[55];
int main()
{
	fib[0] = 1;
	fib[1] = 1;
	fib[2] = 2;
	int i;
	for(i=3;i<=53;i++)
	fib[i] = fib[i-1] + fib[i-2];
	int n;
	ll k;
	cin>>n;
	cin>>k;
	for(i=1;i<=n;)
	{
	int rem;
	rem = n+1-i;
	if(k<=fib[rem-1])
	{
	cout << i << " ";
	i++;
	}
	else {
	cout << i+1 << " " << i<<" ";
	i+=2;
	k-=fib[rem-1];
	}
	}
	cout<<endl;
}
