#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;
#define LL long long
int main(){
	LL n,k;
	scanf("%I64d%I64d",&n,&k);
	LL f[60]={1,1};
	for(int i=2;i<=50;i++)f[i]=f[i-1]+f[i-2];
	int out=1;
	for(int i=n;i>=1;i--){
		//cout<<k<<" "<<f[i-1]<<endl;
		if(k>f[i-1]){
			k-=f[i-1];
			cout<<out+1<<" "<<out<<" ";
			out+=2;i--;
		}else cout<<out++<<" ";
	}
	return 0;	
}
