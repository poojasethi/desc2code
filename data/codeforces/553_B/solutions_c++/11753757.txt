#include<cstdio>
#include<cstdlib>
#include<cmath>
#include<cstring>
#include<iostream>
#include<algorithm>
using namespace std;

int n;
long long f[55],k;
int main() {
	scanf("%d%I64d",&n,&k);
	f[n]=f[n-1]=1;
	for (int i=n-2;i>=0;i--) f[i]=f[i+1]+f[i+2];
	for (int i=0;i<n;i++) {
		if (f[i+1]>=k) printf("%d",i+1); else {
			printf("%d %d",i+2,i+1); k-=f[i+1]; i++;
		}
		if (k==0) printf("\n"); else printf(" ");
	}
	return 0;
}
