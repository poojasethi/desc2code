#include <cstdio>
#include <algorithm>
#include <cstring>
#include <iostream>
using namespace std;
typedef long long LL;
int main()
{
	int n;
	LL k;
	cin >> n >> k;
	LL a[55];
	a[0] = a[1] = 1;
	for(int i = 2; i <= n; ++i)
		a[i] = a[i-1] + a[i-2];
	for(int i = 1; i <= n; ++i){
		if(a[n-i] >= k)
			printf("%d ",i); 
		else {
			k -= a[n-i];
			printf("%d %d ",i + 1,i);
			++i;
		}
	}
	return 0;
}
