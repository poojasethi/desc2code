#include <cstdio>
#include <cstring>
#include <iostream>
#include <algorithm>
#define M 55
using namespace std;
long long n,k;
long long f[M];
int ans[M];
int main()
{
	int i;
	cin>>n>>k;
	f[n+1]=1;
	for(i=n;i;i--)
		f[i]=f[i+1]+f[i+2];
	for(i=1;i<=n;i++)
	{
		if( k<=f[i+1] )
			ans[i]=i;
		else
		{
			k-=f[i+1];
			ans[i]=i+1;
			ans[i+1]=i;
			++i;
		}
	}
	for(i=1;i<=n;i++)
		printf("%d%c",ans[i],i==n?'\n':' ');
	return 0;
}
