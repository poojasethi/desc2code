fib = [1, 1]
for i in xrange(2, 51):
    fib.append(fib[i - 1] + fib[i - 2])

def get(off, n, k):
    if n == 0:
        return
    if k < fib[n - 1]:
        print off + 1,
        get(off + 1, n - 1, k)
    else:
        print off + 2, off + 1,
        get(off + 2, n - 2, k - fib[n - 1])
n, k = map(int, raw_input().split())
get(0, n, k - 1)
