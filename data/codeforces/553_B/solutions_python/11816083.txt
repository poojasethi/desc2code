def fib(n):
    if n < 2:
        return n
    a,b = 0,1
    for _ in xrange(n):
        a,b = b,a+b
    return a

n,k = map(int, raw_input().split())

# fib(n+1) total
perm = []
i = 1

while len(perm) < n:
    nu = n - len(perm)
    if k <= fib(nu):
        # use 1
        perm.append(i)
        i += 1
    else:
        perm.append(i+1)
        perm.append(i)
        k -= fib(nu)
        i += 2

print ' '.join(str(t) for t in perm)
