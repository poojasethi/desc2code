dp = [0 for x in xrange(55)]
dp[0] = 1
dp[1] = 1

for i in xrange(2, 55):
    dp[i] = dp[i - 1] + dp[i - 2]

n, k = map(int, raw_input().split())

k -= 1

res = [0 for x in xrange(55)]

flag = 0

for i in xrange(n):
    if flag == 1:
        flag = 0
        continue
    if k >= dp[n - i - 1]:
        k -= dp[n - i - 1]
        res[i] = i + 2
        res[i + 1] = i + 1
        flag = 1
    else:
        res[i] = i + 1
