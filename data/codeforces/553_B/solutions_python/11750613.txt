fib = [0]*55
ar = []

def solve (st, lft, k, n):
	if st == n + 1:
		return
	if k<= fib[lft-1]:
		ar.append(st)
		solve(st+1, lft-1, k, n)
	else:
		ar.append(st+1)
		ar.append(st)
		solve(st+2,lft-2,k-fib[lft-1],n)

n, k = map(int, raw_input().split())

for i in xrange(2,n+1):
