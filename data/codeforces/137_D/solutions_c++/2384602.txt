#include <iostream>
#include <algorithm>
#include <cstring>
#include <cstdio>
using namespace std;
char s[512];
int p[512][512],v[512][512];
void pal(char* p,char* q){
  while(p<q){if(*p!=*(q-1))*(q-1)=*p;++p,--q;}
}
void go(int n,int k,int t){
  if(k>1){
    for(int i=k-1;i<n;++i)if(v[i][k-1]+p[i][n-i]==t){
      go(i,k-1,t-p[i][n-i]);
      pal(s+i,s+n),printf("+%.*s",n-i,s+i);
      break;
    }
  }
  else pal(s,s+n),printf("%.*s",n,s);
}
int main(void) {
  int k;
  cin>>s>>k;
  memset(p,0,sizeof(p));
  int n=strlen(s),t=n;
  for(int i=0;i+2<=n;++i)p[i][2]=(s[i]==s[i+1])?0:1;
  for(int i=3;i<=n;++i)for(int j=0;j+i<=n;++j)p[j][i]=(s[j]==s[j+i-1]?0:1)+p[j+1][i-2];
  memset(v,0x3f,sizeof(v));
  for(int i=1;i<=n;++i)v[i][1]=p[0][i];
  for(int i=1;i<=k;++i){
    t=min(t,v[n][i]);
    for(int j=1;j<n;++j)if(v[j][i]<t)
      for(int u=j+1;u<=n;++u)v[u][i+1]=min(v[u][i+1],v[j][i]+p[j][u-j]);
  }
  printf("%d\n",t);
  for(int i=1;i<=k;++i)if(v[n][i]==t){go(n,i,t);break;}
  puts("");
  return 0;
}
