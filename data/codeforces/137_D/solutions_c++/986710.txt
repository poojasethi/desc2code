#include<cstdio>
#include<cstring>

char s[510];
int lm,n,i,j,k,ans;
int f[510][510],g[510][510],c[510][510];
int stack[510],top=-1;

int main()
{
	scanf("%s%d",&s,&lm);
	n=strlen(s);
	for(i=0;i<2*n-1;++i)
	{
		k=0;
		for(j=i/2;j>=0 && i-j<n;--j)
		{
			if (s[j]!=s[i-j])++k;
			c[j][i-j]=k;
		}
	}
	for(i=0;i<n;++i)
	{
		f[i][1]=c[0][i];
		g[i][1]=-1;
		for(j=2;j<=i+1;++j)
		{
			f[i][j]=10000;
			for(k=j-1;k<=i;++k)
				if (f[i][j]>f[k-1][j-1]+c[k][i])
				{
					f[i][j]=f[k-1][j-1]+c[k][i];
					g[i][j]=k-1;
				}
		}
	}
	ans=1;
	for(i=2;i<=lm;++i)
		if (f[n-1][i]<f[n-1][ans])
			ans=i;
	i=n-1;j=ans;
	while(i!=-1)
	{
        for(k=i;k>g[i][j];--k)
            if (s[k]!=s[g[i][j]+1+i-k])
                s[k]=s[g[i][j]+1+i-k];
        stack[++top]=g[i][j];
        i=g[i][j];
        --j;
    }
    --top;
	printf("%d\n",f[n-1][ans]);
	for(i=0;i<n;++i)
	{
        printf("%c",s[i]);
        if (top>=0 && i==stack[top])
        {
            printf("+");
            --top;
        }
    }
    printf("\n");
    return 0;
}
