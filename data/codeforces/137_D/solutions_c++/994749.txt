using namespace std;

typedef long long LL;
typedef pair<int,int> PII;

char s[505];
int dp[505][505];
int pt[505][505];
int xy[505][505];

void output(int n, int m){
    if(!m) return;
    output(pt[n][m],m-1);
    if(pt[n][m]) putchar('+');
    int x=pt[n][m]+1,y=n;
    for(int i=x;i<=y;i++) putchar(i<=y-i+x?s[i]:s[y-i+x]);
}

int main(){
    int n=strlen(gets(s+1)),m;
    scanf("%d",&m);
    memset(dp,63,sizeof(dp));
    dp[0][0]=0;
    for(int i=1;i<=n;i++) for(int j=i;j<=n;j++){
        for(int k=i;k<=j;k++) if(s[k]!=s[j-k+i]) xy[i][j]++;
        xy[i][j]/=2;
    }
    for(int i=1;i<=n;i++) for(int j=1;j<=m;j++){
        for(int k=0;k<i;k++){
            int tmp=dp[k][j-1]+xy[k+1][i];
            if(tmp<dp[i][j]){
                dp[i][j]=tmp;
                pt[i][j]=k;
            }
        }
    }
    m=min_element(dp[n],dp[n]+m+1)-dp[n];
    printf("%d\n",dp[n][m]);
    output(n,m);puts("");
}
