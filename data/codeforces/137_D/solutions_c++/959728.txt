#include <stdio.h>
#include <string.h>

int f[1000][1000], g[1000][1000], h[1000][1000];
char a[1000];

void print(int n, int m)
{
    if (n==0) return;
    print(h[n][m], m-1);
    if (h[n][m]!=0) printf("+");
    int l=h[n][m], r=n-1;
    while (l<r)
    {
        if (a[l]!=a[r]) a[r]=a[l];
        l++; r--;
    }
    for (int i=h[n][m]; i<n; i++) printf("%c", a[i]);
}

int main()
{
    int n, m, i, j, k, ans;
    scanf("%s", a);
    n=strlen(a);
    scanf("%d", &m);
    memset(f, 0, sizeof(f));
    for (i=n-1; i>=0; i--)
        for (j=i+1; j<n; j++)
        {
            if (a[i]==a[j]) f[i][j]=f[i+1][j-1];
            else f[i][j]=f[i+1][j-1]+1;
        }
    memset(g, 63, sizeof(g));
    g[0][0]=0;
    for (i=0; i<n; i++)
        for (j=0; j<m; j++) 
        {
            for (k=i; k<n; k++) if (g[i][j]+f[i][k]<g[k+1][j+1])
            {
                g[k+1][j+1]=g[i][j]+f[i][k];
                h[k+1][j+1]=i;
            }
        }
    ans=2147483647; 
    for (i=0; i<=m; i++) if (g[n][i]<ans)
    {
        ans=g[n][i];
        k=i;
    }
    printf("%d\n", ans);
    print(n, k);
    return 0;
}
