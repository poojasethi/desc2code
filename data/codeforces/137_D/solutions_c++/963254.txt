#include <cstdio>
#include <cstring>
#include <cctype>
#include <vector>
#include <algorithm>
using namespace std;

char s[505];
int n, K, c[505][505], dp[505][505], pa[505][505];

void output(int x, int y) {
	if (!y) return ;
	int k = pa[x][y];
	output(k, y-1);
	if (y > 1) putchar('+');
	int m = (k+1+x)/2;
	for (int i = k+1; i <= m; ++i)
		putchar(s[i]);
	if ((x-k)%2==0) putchar(s[m]);
	for (int i = m-1; i >= k+1; --i)
		putchar(s[i]);
}

int main() {
	scanf("%s%d", s+1, &K);
	n = strlen(s+1);
	for (int i = 1; i <= n; ++i) {
		for (int j = i; j <= n; ++j) {
			for (int l=i, r=j; l<=r; l++,r--) {
				c[i][j] += s[l]!=s[r];
			}
		}
	}
	for (int i = 1; i <= n; ++i) {
		dp[i][1] = c[1][i];
		for (int j = 2; j <= K; ++j) {
			dp[i][j] = 0x7f7f7f7f;
			for (int k = 0; k < i; ++k) {
				int t = dp[k][j-1]+c[k+1][i];
				if (t < dp[i][j]) {
					dp[i][j] = t;
					pa[i][j] = k;
				}
			}
		}
	}
	int ans = 0x7f7f7f7f;
	for (int j = 1; j <= K; ++j) {
		ans = min(ans, dp[n][j]);
	}
	for (int j = 1; j <= K; ++j) {
		if (dp[n][j] == ans) {
			printf("%d\n", ans);
			output(n,j);
			puts("");
			break;
		}
	}
}
