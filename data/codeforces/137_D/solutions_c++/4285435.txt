#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#define inf 100000000

using namespace std;

string S;
int change[505][505],dp[505][505];
int N,K,yol[505][505];

int solve(int x,int k)
{
	if(k < 0) return inf;
	if(x == N) return 0;
	if(dp[x][k] != -1) return dp[x][k];

	int i,ret=inf,ans;
	for(i=x;i<N;i++)
	{
		ans = solve(i+1,k-1) + change[x][i];
		if( ans < ret )  
		{
			ret = ans;
			yol[x][k] = i+1;
		}
	}

	return dp[x][k] = ret;
}

int main()
{
	int i,j,k,t;
	string str;

	cin >> S >> K;
	N = S.size();

	for(k=1;k<=N;k++)
		for(i=0;i<N;i++)
			for(j=0;j<k/2;j++)
				if(S[i+j] != S[i+k-j-1]) 
					change[i][i+k-1]++;

	memset(dp,-1,sizeof(dp));
	cout << solve(0,K) << endl;

	i=0;
	k=K;
	while(i != N)
	{
		t=yol[i][k];
		str="";
	
		for(j=i;j<=(t+i-1)/2;j++)
		{
			printf("%c",S[j]);
			if((t-i) % 2 && j == (t+i-1)/2) continue;
			str+=S[j];
		}
		reverse(str.begin(),str.end());
		cout << str;

		i = t;
		k--;
		if(i != N) printf("+");
	}
	cout << endl;

	return 0;
}
