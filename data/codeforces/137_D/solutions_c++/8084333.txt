#include <cstdio>
#include <cstring>

const int MAX_N = 507;
const int INF = 0x3fffffff;

int n, k;
int dp[MAX_N][MAX_N], f[MAX_N][MAX_N], pre[MAX_N][MAX_N];
char s[MAX_N];

void dfs(int x, int y) {
	if(!x) return;
	dfs(x - 1, pre[x][y]);
	if(x != 1) printf("+");
	int l = pre[x][y] + 1, r = y;
	int p = (r-l+1)/2;
	for(int i = 0; i < p; i++)
		printf("%c", s[i+l]);
	if((r-l+1)&1)
		printf("%c", s[p+l]);
	for(int i = p-1; i >= 0; i--)
		printf("%c", s[i+l]);
}

int main() {
	scanf("%s%d", s+1, &k);
	n = strlen(s+1);
	for(int i = 1; i <= n; i++) 
		for(int j = 1; i+j-1 <= n; j++)
			for(int u = 0; u<(j/2); u++)
				if(s[i+u] != s[i+j-1-u]) f[i][i+j-1]++;
	for(int i = 0; i <= n; i++)
		for(int j = 0; j <= n; j++)
			dp[i][j] = INF;
	dp[0][0] = 0;
	for(int i = 0; i < k; i++)
		for(int j = i; j < n; j++) if(dp[i][j] != INF) {
			for(int u = 1; u + j <= n; u++) {
				int val = dp[i][j] +  f[j+1][j+u];
				if(dp[i+1][j+u] > val) {
					dp[i+1][j+u] = val;
					pre[i+1][j+u] = j;
				}
			}
		}
	int ans = INF, pos = -1;
	for (int i = 1; i <= k; i++)
		if(ans > dp[i][n]) {
			ans = dp[i][n];
			pos = i;
		}
	printf("%d\n", ans);
	dfs(pos, n);
	return 0;
}
