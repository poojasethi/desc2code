#include <cstdio>
#include <cstring>
char s[500];
int c[500][500],f[500][500],g[500][500];

void print(int i,int j)
{
    int k=g[i][j];
    if (i>1)
    {
        print(i-1,k);
        putchar('+');
    }
    int mid=(k+1+j)>>1;
    for (int l=k+1; l<=mid; ++l) putchar(s[l]);
    for (int l=mid+1; l<=j; ++l) putchar(s[k+1+j-l]);
}

int main()
{
    int m;
    scanf("%s%d",s,&m);
    int i,j,l,r,k,x, n=strlen(s);
    for (i=0; i<n; ++i)
        for (j=i; j<n; ++j)
            for (l=i, r=j; l<r; ++l,--r)
                if (s[l]!=s[r]) ++c[i][j];
    memcpy(f[1],c[0],sizeof c[0]);
    memset(g[1],-1,sizeof g[1]);
    for (i=2; i<=m; ++i)
        for (j=i-1; j<n; ++j)
        {
            f[i][j]=c[0][j]; g[i][j]=-1;
            for (k=i-2; k<j; ++k)
            {
                x=f[i-1][k]+c[k+1][j];
                if (x<f[i][j]) f[i][j]=x, g[i][j]=k;
            }
        }
    int ans=n;
    for (i=1; i<=m; ++i)
        if (f[i][n-1]<ans) ans=f[i][n-1], k=i;
    printf("%d\n",ans);
    print(k,n-1);
    return 0;
}
