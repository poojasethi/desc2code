#include<bits/stdc++.h>

using namespace std;

#define N 502

int H[N][N],M[N][N],M2[N][N],n,mx,i,j,k,l;

string s;

int f(int i, int k){

    if(k < 0)
       return N;

    if(i == n)
        return 0;

    int j, &r = M[i][k];

    if(r != -1)
        return r;

    r = N;

    for(j=i; j<n; j++)
        if( r > (l = f(j+1,k-1) + M2[i][j]) ){
            H[i][k] = j+1;
            r = l;
        }

    return r;
}

int main(){

    cin >> s >> mx;

    n = s.size();

    for(i=0;i<n;i++)
        for(j=i+1;j<n;j++)
           for(k=i,l=i+j>>1; k<=l ;k++)
               M2[i][j] += s[k] != s[j - k + i];

    memset(M , -1 , sizeof M);

    cout << f(0,mx) << endl;

    for(i=0,k=mx; i<n ;i=H[i][k],k--){
        if(i)
            cout << '+';
        for(j = i,l=i+H[i][k]>>1;j < l;j++)
            cout << s[j];
        if((H[i][k] - i) & 1)
            cout << s[l];
        for(j = l-1;j >= i;j--)
            cout << s[j];
    }

	return 0;
}
