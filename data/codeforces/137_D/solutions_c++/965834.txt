#include<cstdio>
#include<cstring>
#include<cmath>
#include<vector>
#include<algorithm>
#include<set>
#include<iostream>
using namespace std;
string str[510][510];
int f[510][510],g[510][510];
int n,m,tmp,ans=1000000,pre[510][510];
char s[510];
void out(int now,int ti){
    if(pre[now][ti])out(pre[now][ti],ti-1);
    cout<<str[pre[now][ti]+1][now];
    if(now!=n)printf("+");
} 
int main(){
    gets(s+1);s[0]='a';n=strlen(s)-1;
    scanf("%d",&m);
    for(int i=1;i<=n;++i)str[i][i]=s[i];
    for(int len=1;len<n;++len)
        for(int i=1,j;i<=n-len;++i){
            j=i+len;
            g[i][j]=g[i+1][j-1]+(s[i]!=s[j]);
            str[i][j]=s[i]+str[i+1][j-1]+s[i];
        }
    memset(f,63,sizeof(f));
    f[0][0]=0;
    for(int i=1;i<=n;++i)
        for(int k=0;k<i;++k)
            for(int j=1;j<=m;++j){
                if(f[k][j-1]+g[k+1][i]<f[i][j]){
                    f[i][j]=f[k][j-1]+g[k+1][i];
                    pre[i][j]=k;
                }
            }
    for(int j=1;j<=m;++j)
        if(f[n][j]<ans){
            ans=f[n][j];tmp=j;
        }
    printf("%d\n",ans);
    out(n,tmp);
    return 0;
}
