typedef long long LL;
using namespace std ;

char s[510];
int a[510][510],f[510][510],pre[510][510];

int print(int r,int i,int M){
    int l=pre[r][i];
    if (l) print(l,i-1,M);
    ++l;
    int j;
    for (j=l;j<=r;j++) putchar(j<=(l+r)/2?s[j]:s[l+r-j]);
    printf("%c",i<M?'+':'\n');
}

int main(){
    int M;
    cin >> s+1 >> M;
    int N=strlen(s+1),i,j,k;
    for (i=1;i<=N;i++)
        for (j=i;j<=N;j++)
            for (k=i;k<=(i+j)/2;k++)
                a[i][j]+=s[k]!=s[i+j-k];
    memset(f,0x7f,sizeof(f));
    f[0][0]=0;
    for (i=1;i<=N;i++)
        for (j=0;j<i;j++)
            for (k=1;k<=M;k++){
                int t=f[j][k-1]+a[j+1][i];
                if (t<f[i][k]){
                    f[i][k]=t; pre[i][k]=j;
                }
            }
    int p=1;
    for (i=2;i<=M;i++)
        if (f[N][i]<f[N][p]) p=i;
    cout << f[N][p] << endl;
    print(N,p,p);
}
