#include<iostream>
#include<cstdlib>
#include<cstdio>
#include<vector>
using namespace std;

const int INF = 1 << 30;
string s;
int d[501], p[501][501], n, k;
int dif[501][501];
vector<string> ans;
int main()
{
	cin >> s >> k;
	n = s.size();
	for(int i = 0; i < n; ++i)
		for(int j = i; j < n; ++j)
			for(int ii = 0; ii < (j - i + 1)/2; ++ii)
				if(s[i + ii] != s[j - ii]) 
					dif[i][j]++;
	for(int i = 1; i < 501; ++i)
		d[i] = INF;        	
	for(int i = 0; i < k; ++i)	
		for(int j = n; j >= 1; --j)
			for(int jj = 1; jj <= j; ++jj)
				if(d[j - jj] < INF && d[j] > d[j - jj] + dif[j - jj][j - 1])
					p[i][j] = jj, d[j] = d[j - jj] + dif[j - jj][j - 1];
	cout << d[n] << endl;
	int t = n, tt = n;
	while(t > 0)
	{
		while(p[tt][t] == 0) 	
			tt--;
		string tmp = s.substr(t - p[tt][t], p[tt][t]);
		for(int i = 0; i < tmp.size()/2; ++i)
			if(tmp[i] != tmp[tmp.size() - i - 1])
				tmp[i] = tmp[tmp.size() - i - 1];
		ans.push_back(tmp);
		t = t - p[tt--][t];
	}
	for(int i = ans.size() - 1; i >= 0; --i)
	{
		cout << ans[i];
		if(i > 0) cout << '+';
	}
	return 0;
}
