#include<iostream>
#include<string>
using namespace std;
////////////////////////////////////////////////////////////
#define go(i,n) for(int i=0;i<(n);++i)
//////////////////////////////////////////////////////////////////////////
////////////////////////////////////////////////////////////////////////////////
string s;
int n,y[512][512],l,f[512][512],g[512][512],x;
int gao(int k,int x){
    if(k>1){gao(k-1,g[k][x]);cout<<"+";}
    int l=x-g[k][x];
    go(i,l)cout<<((i*2<l)?s[g[k][x]+i]:s[g[k][x]+l-i-1]);
}
int main(){
    cin>>s>>n;l=s.length();
    go(i,l)go(j,l-i)go(k,j+1>>1)
        if(s[i+k]-s[i+j-k])++y[i][j];
    go(k,n+1)go(i,l+1)f[k][i]=999;f[0][0]=0;
    go(k,n)go(i,l)go(j,l-i)if(f[k][i]+y[i][j]<f[k+1][i+j+1]){
            f[k+1][i+j+1]=f[k][i]+y[i][j];
            g[k+1][i+j+1]=i;
    }
    x=0;
    go(i,n)if(f[i+1][l]<f[x][l])x=i+1;
    cout<<f[x][l]<<endl;
    gao(x,l);cout<<endl;
    return 0;
}
