#include <iostream>
#include <string>
#include <cstring>
using namespace std;
#define rep(i,n) for(int i=0;i<n;i++)
#define repp(i,a,b) for(int i=a;i<b;i++)
#define pr(i) cout<<i<<endl
#define INF 100000000

int k,n,r,a,b,sz;
string s,t;
int dp[510][510],cost[510][510],path[510][510];
int p[510];

int main(){

  cin>>s>>k;n=s.size();
  rep(i,n)repp(j,i,n)rep(l,(j-i+2)/2)if(s[i+l]!=s[j-l])cost[i][j]++;
  rep(i,n)dp[i][0]=cost[0][i];

  rep(i,n)repp(j,1,i+1){
    int m=INF,up=j-1;
    repp(l,j-1,i){
      if(m>dp[l][j-1]+cost[l+1][i]){
    m=dp[l][j-1]+cost[l+1][i];up=l;}
    }
    dp[i][j]=m;
    path[i][j]=up;
  }
  r=INF;
  rep(i,k)if(r>dp[n-1][i]){r=dp[n-1][i];a=i;}
  b=path[n-1][a];
  rep(i,a){p[a-i]=b+1;b=path[b][a-1-i];}
  p[a+1]=n;
  
  pr(r);
  rep(i,a+1){
    t=s.substr(p[i],p[i+1]-p[i]);
    sz=t.size();
    rep(j,sz)cout<<(j<(sz+1)/2?t[j]:t[sz-1-j]);
    if(i==a)pr("");
    else cout<<"+";
  }

  return 0;
}
