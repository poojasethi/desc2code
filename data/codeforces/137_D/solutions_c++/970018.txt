#include <cstdio>
#include <cstring>

int lm,n;
char s[501];
int c[500][500];
int f[500][501],g[500][501];
int stack[500],top=-1;

int main()
{
	int i,j,k,ans;
	scanf("%s%d",s,&lm);
	n = strlen(s);
	for(i=0;i<2*n-1;++i)
	{
		k=0;
		for(j=i/2;j>=0&&i-j<n;--j)
		{
			if(s[j]!=s[i-j])k++;
			c[j][i-j]=k;
		}
	}
	for(i=0;i<n;++i)
	{
		f[i][1]=c[0][i];
		g[i][1]=-1;
		//printf("%d(-1)",f[i][1]);
		for(j=2;j<=i+1;++j)
		{
			f[i][j]=1000;
			for(k=i;k>=j-1;--k)
			{
				if(f[i][j]>c[k][i]+f[k-1][j-1])
				{
					f[i][j]=c[k][i]+f[k-1][j-1];
					g[i][j]=k-1;
				}
			}
			//printf(" %d(%d)",f[i][j],g[i][j]);
		}
		//printf("\n");
	}
	for(ans=1,i=2;i<=lm;++i)
	{
		if(f[n-1][ans]>f[n-1][i])
		{
			ans=i;
		}
	}
	i=n-1,j=ans;
	while(i!=-1)
	{
		for(k=i;k>g[i][j];k--)
		{
			if(s[k]!=s[g[i][j]+1+i-k])s[k]=s[g[i][j]+1+i-k];
		}
		stack[++top]=g[i][j];
		//printf("fuck:%d %d %d %d\n",i,j,g[i][j],top);
		i=g[i][j];
		j--;
	}
	//printf("top%d\n",top);
	top--;
	//printf("top%d\n",top);
	printf("%d\n",f[n-1][ans]);
	for(i=0;i<n;++i)
	{
		printf("%c",s[i]);
		if(top>=0&&i==stack[top])
		{
			printf("+");
			top--;
		}
	}
	printf("\n");
	return 0;
}
