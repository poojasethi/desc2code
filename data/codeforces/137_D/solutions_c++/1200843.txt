# include <cstdio>
# include <cstring>
# include <cstdlib>
# include <iostream>
# define N 550
# define oo 20000000
using namespace std;
int dp[N][N],f[N][N],val[N][N];
char s[N];
void print(int n,int k)
{
	if (k>1)
	{
		print(f[n][k],k-1);
		cout<<"+";
	}
	int ss=f[n][k]+1;
	for (int i=0;i<=(n-ss)/2;i++)
		if (s[ss+i]!=s[n-i])
			s[ss+i]=s[n-i];
	for (int i=ss;i<=n;i++)
		cout<<s[i];
}
int main (void)
{
	int n;
	scanf("%s",s+1);
	cin>>n;
	int len=strlen(s+1);
	for (int i=0;i<=len;i++)
		for (int j=0;j<=n;j++)
			dp[i][j]=oo;
	dp[0][0]=0;
	for (int i=1;i<=len;i++)
		for (int j=i;j<=len;j++)
			for (int k=0;k<=(j-i)/2;k++)
				if (s[i+k]!=s[j-k])
					val[i][j]++;
	for (int i=1;i<=len;i++)
		for (int j=1;j<=n&&j<=i;j++)
			for (int k=0;k<i;k++)
				if (dp[i][j]>dp[k][j-1]+val[k+1][i])
				{
					dp[i][j]=dp[k][j-1]+val[k+1][i];
					f[i][j]=k;
				}
	int ans=dp[len][1],ansk=1;
	for (int i=2;i<=n;i++)
		if (dp[len][i]<ans)
		{
			ans=dp[len][i];
			ansk=i;
		}
	cout<<ans<<endl;
	print(len,ansk);
	cout<<endl;
	return 0;
}
