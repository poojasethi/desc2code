def go(f):
    [m, k], d, s = f(), f(), f()
    t, up, r = sum(d), 0, [0]
    for i,j in zip(s, d): r += [r[-1]+i-j]
    for i,v in (j for j in enumerate(r) if j[1]+up < 0):
        bs = max(s[:i])
        q = abs((v+up)/bs)
        t += q*k; up += q*bs
    print t

go(lambda:map(int, raw_input().split()))
