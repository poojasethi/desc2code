import sys
caonima = True
if caonima:
    f = sys.stdin
else:
    f = open('1.in')
m,k = map(int,f.readline().split())
d = map(int,f.readline().split())
s = map(int,f.readline().split())
#s = s[::-1]
#d = d[::-1]
ans = 0
gas = 0
for i in range(m):
    gas += s[i]
    ans += d[i]
    while gas < d[i]:
        gas += max(s[:i+1])
        ans += k
    gas -= d[i]
if not caonima:
    f.close()
    with open( "1.out", "w") as f1:
        f1.write(ans)
else:
    print ans
