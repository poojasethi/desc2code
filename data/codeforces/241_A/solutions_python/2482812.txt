import sys
sys.setrecursionlimit(10000)

m,k = map(int,raw_input().split())
d = map(int,raw_input().split())
s = map(int,raw_input().split())
ms = s[:]
n=m+1
for i in range(1,m):
  ms[i] = max(ms[i-1],s[i])

res=0
f=0
for i in range(m):
  f+=s[i]
  while f<d[i]:
    f+=ms[i]
    res+=k
  f-=d[i]
  res+=d[i]
print res
