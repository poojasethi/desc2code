def start():
    import math
    m, k = map(int, raw_input().split())
    distance = map(int, raw_input().split())
    fuel = map(int, raw_input().split())
    currentFuel = 0
    time = 0
    fuelMax = fuel[0]
    for currentCity in xrange(m):
        currentFuel += fuel[currentCity]
        fuelMax = max(fuel[currentCity], fuelMax)
        fuelFillNum = max(0, int(math.ceil((distance[currentCity] - currentFuel)*1.0/fuelMax)))
        time += k*fuelFillNum + distance[currentCity]
        currentFuel = currentFuel + fuelFillNum*fuelMax - distance[currentCity]
    print time

start()
