m,k = map(int,raw_input().split())
dist = map(int,raw_input().split())
supply = map(int,raw_input().split())
liters = 0
hours = 0
max_supply = 0
for i in range(m):
    liters = liters + supply[i]
    hours = hours + dist[i]
    max_supply = max(supply[i],max_supply)
    if liters > dist[i]:
        liters = liters - dist[i]
    else:
        while(liters<dist[i]):
            liters = liters + max_supply
            hours = hours + k
        liters = liters - dist[i]
print hours
