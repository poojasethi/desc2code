m, K = map(int, raw_input().split())
dist = map(int, raw_input().split())
fuel = map(int, raw_input().split())

ans = sum(dist)
cur = 0
maxCharge = 0

for i in xrange(m):
	cur += fuel[i]
	maxCharge = max(maxCharge, fuel[i])
	while cur < dist[i]:
		ans += K
		cur += maxCharge
	cur -= dist[i]

print ans
