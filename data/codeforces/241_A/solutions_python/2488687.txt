m, k = map(int, raw_input().split())

d = map(int, raw_input().split())
s = map(int, raw_input().split())

maxSoFarIndex = 0
maxSoFarValue = s[0]
currentFuel = 0
currentTime = 0

for i in range(m):
    if s[i] > maxSoFarValue:
        maxSoFarIndex = i
        maxSoFarValue = s[i]

    currentFuel += s[i]
    extraTime = 0
    if d[i] > currentFuel:
        remaining = d[i] - currentFuel
        extraTime = (remaining + maxSoFarValue - 1) / maxSoFarValue

    currentFuel += extraTime * maxSoFarValue - d[i]
    currentTime += d[i] + extraTime * k

print currentTime
