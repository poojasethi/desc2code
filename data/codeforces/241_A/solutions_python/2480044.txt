m, k = map(int, raw_input().split())
d = map(int, raw_input().split())
s = map(int, raw_input().split())
i = 0
time = 0
fuel = sup = 0
while i < m:
    sup = max(sup, s[i])
    fuel += s[i]
    while d[i] > fuel:
        time += k
        fuel += sup
    fuel -= d[i]
    time += d[i]
    i += 1
print time
