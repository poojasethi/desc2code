m, k = map(int, raw_input().split())
d = map(int, raw_input().split())
s = map(int, raw_input().split())

S = 0
t = 0
i = 0
p = -1
while i < m:
    p = max(p, s[i])
    S += s[i]
    if S >= d[i]:
        t += d[i]
        S -= d[i]
    else:
        S += p - s[i]
        i -= 1
        t += k
    i += 1
    
print t
