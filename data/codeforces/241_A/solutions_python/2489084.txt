#!/usr/bin/python
import math

m, k = [int(x) for x in raw_input().split()]
d = [int(x) for x in raw_input().split()]
s = [int(x) for x in raw_input().split()]

travelTime = 0
dist = 0
gas  = 0
smax = s[0]
for i in range(m):
	gas += s[i]
	dist += d[i]
	smax = max(smax, s[i])
	if dist > gas:
		nStay = int(math.ceil((dist - gas)/float(smax)))
		travelTime += nStay * k
		gas += nStay * smax
	travelTime += d[i]

print travelTime
