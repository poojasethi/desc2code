m, k = map(int, raw_input().strip().split())
d = map(int, raw_input().strip().split())
s = map(int, raw_input().strip().split())

t = 0
benz = 0
maxf = 0

for i in range(m):
    benz = benz + s[i]
    t = t + d[i]
    maxf = max(maxf, s[i])
    benz = benz - d[i]
    while benz < 0:
        benz = benz + maxf
        t = t+k

print t
