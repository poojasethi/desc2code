m,k = map(int,raw_input().split())
dist = map(int,raw_input().split())
supply = map(int,raw_input().split())
liters = 0
hours = 0
max_fuel = 0
for i in range(m):
    liters = liters + supply[i]
    hours = hours + dist[i]
    max_fuel = max(max_fuel, supply[i])
    if liters > dist[i]:
        liters = liters - dist[i]
    else:
        while(liters<dist[i]):
            liters = liters + max_fuel
            hours = hours + k
        liters = liters - dist[i]
print hours
