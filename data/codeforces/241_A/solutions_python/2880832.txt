m,k = map(int, raw_input().split())
d = map(int, raw_input().split())
s = map(int, raw_input().split())
a = 0
ans = 0
for i in xrange(m):
    if a+s[i]-d[i] < 0:
        a += s[i]
        while a < d[i]:
            a += max(s[:i+1])
            ans += k
        a -= d[i]
    else:
        a += s[i]-d[i]
    ans += d[i]
print ans
