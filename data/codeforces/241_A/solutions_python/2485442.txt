import sys

(n, m) = [int(x) for x in sys.stdin.readline().strip().split()]
ans = 0
d = [0]
s = [0]
d.extend([int(x) for x in sys.stdin.readline().strip().split()])
s.extend([int(x) for x in sys.stdin.readline().strip().split()])

fue = 0
mx = 0

for i in range(1, n+1):
    mx = max(mx, s[i])
    fue += s[i]
    ans += d[i]
    while fue < d[i]:
        fue += mx
        ans += m
    fue -= d[i]

print(ans)
