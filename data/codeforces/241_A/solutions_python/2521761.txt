m , k = map(int , raw_input().split())
d = list(map(int , raw_input().split()))
s = list(map(int , raw_input().split()))
max = ans = cur_fuel = 0
for i in range(m):
    if s[i] > max:
        max = s[i]
    cur_fuel += s[i]
    if cur_fuel >= d[i]:
        cur_fuel -= d[i]
        ans += d[i]
    else:
        n = int((d[i] - cur_fuel) / max)
        if (d[i] - cur_fuel) % max != 0:
            n += 1
        ans += k * n + d[i]
        cur_fuel = cur_fuel + max * n - d[i]
print ans
