def main():
    
    (m,k) = tuple(map(int,raw_input().split()))
    d = map(int,raw_input().split())
    s = map(int,raw_input().split())
    n = m + 1
    
    f = 0
    t = 0
    
    for i in range(m):
        f += s[i]
        t += d[i]
        while f < d[i] :
            f += max(s[:i+1])
            t += k
        f -= d[i]
            
    print t
        
    
    return 0

if __name__ == '__main__':
    main()
