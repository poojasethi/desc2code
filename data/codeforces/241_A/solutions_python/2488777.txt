
def main():
    cin = lambda: map(int, raw_input().split())
    [m, k] = cin()
    d = cin()
    s = cin()
    fuel = t = maxn = 0
    for i in xrange(m):
        fuel += s[i]
        maxn = max(maxn, s[i])
        while fuel < d[i]:
            fuel += maxn
            t += k
        if fuel >= d[i]:
            fuel -= d[i]
            t += d[i]
    print t
    
if __name__ == '__main__':
    main()
