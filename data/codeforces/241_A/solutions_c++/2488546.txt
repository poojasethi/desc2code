#include <iostream>
using namespace std;
#define forn(i,n) for (int i = 0; i < n; i++)

const int MAXN = 1000 + 100;

int m, k, d[MAXN], s[MAXN];


int main() {
	cin >> m >> k;
	forn(i,m) cin >> d[i];
	forn(i,m) cin >> s[i];

	int res = 0, fuel = 0, maxs = 0;
	forn(i,m) {
		maxs = max(maxs, s[i]);
		fuel += s[i] - d[i];
		res += d[i];
		while (fuel < 0) {
			res += k;
			fuel += maxs;
		}
	}

	cout << res << endl;

}
