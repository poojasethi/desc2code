#include <cstdio>
#include <iostream>

using namespace std;

int D[1010],S[1010];

int main(){
	
	int m,k;
	
	scanf("%d %d",&m,&k);
	
	for(int i=1;i<=m;i++)
		scanf(" %d",D+i);
	for(int i=1;i<=m;i++)
		scanf(" %d",S+i);
	
	int fuel=0;
	int M=0;
	int res=0;
	
	for(int i=1;i<=m;i++){
		fuel+=S[i];
		M=max(M,S[i]);
		while(fuel<D[i])
			fuel+=M,res+=k;
		fuel-=D[i];
		res+=D[i];
	}
	cout << res << endl;
	
	return 0;
	
}
