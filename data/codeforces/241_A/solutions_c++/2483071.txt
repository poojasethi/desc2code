#include<stdio.h>
int main()
{
	int m=0,k=0,i=0;
	int d[1000],s[1000];
	scanf("%d%d",&m,&k);
	for(i=0;i<m;i++)
		scanf("%d",&d[i]);
	for(i=0;i<m;i++)
		scanf("%d",&s[i]);
	int pt=0,dis=0,tm=0,mx=0;
	for(i=0;i<m;i++)
	{
		pt+=s[i];
		dis+=d[i];
		if(s[i] > mx)
			mx=s[i];
		while(pt < dis)
		{
			tm+=k;
			pt+=mx;
		}
		tm+=d[i];
		pt-=d[i];
		dis-=d[i];
	}
	printf("%d\n",tm);
	return 0;
}
