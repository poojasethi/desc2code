#include<iostream>
#include<algorithm>

using namespace std;

int main()
{
	int m,k,n,i,j,d[1001],s[1001],fuel,maxi,tim=0;
	cin>>n;
	cin>>k;
	for(i=0;i<n;i++)
	{
		cin>>d[i];
	}
	for(i=0;i<n;i++)
	{
		cin>>s[i];
	}
	fuel=s[0];
	maxi=s[0];
	tim=d[0];
	for(i=0;i<n;i++)
	{
		if(i>0){tim+=d[i];fuel+=s[i];
		fuel-=d[i-1];}
		maxi=max(maxi,s[i]);
		while(d[i]>fuel)
		{
			fuel+=maxi;
			tim+=k;
		}
	}
	cout<<tim<<endl;
	return 0;
}
