#include <cstdio>
int d[1001],s[1001];

int main()
{
	int m,k,i;
	scanf("%d%d",&m,&k);
	for (i=1; i<=m; ++i) scanf("%d",d+i);
	for (i=1; i<=m; ++i) scanf("%d",s+i);
	int t, sum=0, cost=0, max=0;
	for (i=1; i<=m; ++i)
	{
		sum+=s[i];
		if (s[i]>max) max=s[i];
		if (d[i]>sum)
		{
			t=(d[i]-sum-1)/max+1;
			sum+=t*max, cost+=k*t;
		}
		sum-=d[i], cost+=d[i];
	}
	printf("%d\n",cost);
	return 0;
}
