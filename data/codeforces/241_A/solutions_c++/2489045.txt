#include <iostream>
#include <stdio.h>
#include <string.h>
#include <algorithm>

using namespace std;

int n,k;
int d[1010],s[1010];

int main()
{
	scanf("%d%d",&n,&k);
	for(int i=1;i<=n;i++)
		scanf("%d",d+i);
	for(int j=1;j<=n;j++)
		scanf("%d",s+j);
	int p=0,ret=0,mx=s[1];
	for(int i=1;i<=n;i++)
	{
		p+=s[i];mx=max(mx,s[i]);
		while(d[i]>p) p+=mx,ret+=k;
		ret+=d[i];
		p-=d[i];
	}
	printf("%d\n",ret);
	return 0;
}
