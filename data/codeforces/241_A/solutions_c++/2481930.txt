#include <iostream>
using namespace std;
int main()
{
int m, k, supl, i, sum, bak, t;
cin>>m>>k;
int d[m];
int s[m];
for (i=0; i<m; i++)
cin>>d[i];
for (i=0; i<m; i++)
cin>>s[i];
supl=s[0];
bak=0;
t=0;
for (i=0; i<m; i++)
{
	bak+=s[i];
	t+=d[i];
	if (supl<s[i])
		supl=s[i];
	while (bak<d[i])
	{
		bak+=supl;
		t+=k;
	}	
	bak-=d[i];
}

}
