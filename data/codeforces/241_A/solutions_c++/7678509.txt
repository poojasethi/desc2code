#include <bits/stdc++.h> 
using namespace std;

int n, k, mx, s[1003], d[1003], tt, ss;

int main (){
ios_base::sync_with_stdio(0);
cin >> n >> k;
for(int i=0;i<n;i++) cin >> d[i];
for(int i=0;i<n;i++) cin >> s[i];
for(int i=0;i<n;i++) {
ss += s[i] - d[i];
mx = max(mx, s[i]);
tt += d[i];
while (ss < 0) { ss += mx; tt += k; }
}
cout << tt;
return 0;
}
