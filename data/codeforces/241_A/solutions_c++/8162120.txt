#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define N 1010
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,K,an,ma,you;
int d[N];
int main() {
	scanf("%d%d",&n,&K);
	For(i,1,n) scanf("%d",&d[i]),d[i]+=d[i-1];
	For(i,1,n) {
		scanf("%d",&k);
		ma=max(ma,k);
		you+=k;
		for (;you<d[i];you+=ma) an+=K;
	}
	printf("%d\n",an+d[n]);
	return 0;
}
