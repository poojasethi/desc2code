#include <iostream>

using namespace std;

const int maxn=1000+10;
int m,k;
int dis[maxn],s[maxn];
int mmax;
int tot,tim;

int main()
{
	cin>>m>>k;
	for (int i=1;i<=m;i++) cin>>dis[i];
	for (int i=1;i<=m;i++) cin>>s[i];
	mmax=0;tot=0;tim=0;
	for (int i=1;i<=m;i++) 
	{
		tot+=s[i];
		if (mmax<s[i]) mmax=s[i];
		while (tot<dis[i]) {tot+=mmax;tim+=k;}
		tim+=dis[i];tot-=dis[i];
	}
	cout<<tim<<endl;
	return 0;
}
