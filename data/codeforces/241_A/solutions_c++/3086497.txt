#include <iostream>
#include <algorithm>
using namespace std;
int main(){
    int m,k,t,fuel=0,max=0,sum=0;
        cin>>m>>k;
        int a[m];
    for(int s=0;s<m;s++){
        cin>>a[s];
    }
    for(int i=0;i<m;i++){
    cin>>t;
    if(t>max){
    max=t;
    }
    fuel=fuel+t;
    while(fuel<a[i]){sum+=k;fuel+=max;}
        sum=sum+a[i];
        fuel=fuel-a[i];
    }
    cout<<sum;
return 0;
}
