#include<iostream>
using namespace std;
const int N = 1005;
int s[N],d[N];
int main(){
	int n,k;
	cin>>n>>k;
	for(int i = 0; i<n;i++)cin>>d[i];
	for(int i = 0; i<n;i++)cin>>s[i];
	int mx = 0,sum = 0,ans = 0;
	for(int i = 0; i < n; i++){
		mx = max(s[i],mx);
		sum += s[i];
		while(sum < d[i]) {
			sum += mx;
			ans += k;
		}
		sum -= d[i];
		ans += d[i];
	}
	cout << ans << endl;
}
