
#include <iostream>

using namespace std;

int m, k;
int d[1000], s[1000];

int main() {
	cin >> m >> k;
	for (int i = 0; i < m; i++)
		cin >> d[i];
	for (int i = 0; i < m; i++)
		cin >> s[i];

	int f = 0, t = 0;
	int most = 0;
	for (int i = 0; i < m; i++) {
		if (most < s[i])
			most = s[i];

		f += s[i];
		while (f < d[i])
			f += most, t += k;
		f -= d[i], t += d[i];
	}
	cout << t << endl;
}
