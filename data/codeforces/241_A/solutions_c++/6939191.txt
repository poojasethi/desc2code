//In the name of God

#include <iostream>
#include <algorithm>
using namespace std;
int a[1100],b[1100];
int main()
{
  int m,k;
  cin>>m>>k;
  for(int i=1;i<=m;i++)
    cin>>a[i];
  int f=0,ans=0,M=0;
  for(int i=1;i<=m;i++)
    {
      cin>>b[i];
      M=max(M,b[i]);
      f+=b[i];
      while(f<a[i])
	{
	  f+=M;
	  ans+=k;
	}
      ans+=a[i];
      f-=a[i];
    }
  cout<<ans<<endl;
  return 0;
}
