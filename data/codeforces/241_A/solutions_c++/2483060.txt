#include<stdio.h>
int main()
{
int ans,max,i,m,k,dd,ss,d[1005],s[1005];
while(scanf("%d%d",&m,&k)==2)
{
dd=0;ss=0;ans=0;  
for(i=1;i<=m;i++){scanf("%d",&d[i]);ans+=d[i];}
for(i=1;i<=m;i++)
{
scanf("%d",&s[i]); 
max=max>s[i]?max:s[i]; 
dd+=d[i];
ss+=s[i];
if(dd>ss)
{
if((dd-ss)%max){ans+=k*((dd-ss)/max+1);ss+=max*((dd-ss)/max+1);}
else{ans+=k*((dd-ss)/max);ss=dd;}
}
}
printf("%d\n",ans);
}
return 0;
}
