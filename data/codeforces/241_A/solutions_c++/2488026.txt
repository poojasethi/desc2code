#include<cstdio>
using namespace std;
int main()
{
  int i,n,k,tot,max,t;
  scanf("%d%d",&n,&k);
  int d[n],s[n];
  for (i=0;i<n;i++)
    scanf("%d",&d[i]);
  for (i=0;i<n;i++)
    scanf("%d",&s[i]);
  tot=0;
  max=0;
  t=0;
  for (i=0;i<n;i++)
  {
    tot+=s[i];
    if (s[i]>max) max=s[i];
    while(tot<d[i])
    {
      tot+=max;
      t+=k;
    }
    tot-=d[i];
    t+=d[i];
  }
  printf("%d\n",t);
}
