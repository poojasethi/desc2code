#include<stdio.h>
#include<stdlib.h>
int m,k,i,oil,d[1050],s[1050],ss,ans;
int main()
{
	scanf("%d%d",&m,&k);
	for (i=1;i<=m;i++)
		scanf("%d",&d[i]);
	for (i=1;i<=m;i++)
		scanf("%d",&s[i]);
	
	oil=0;
	for (i=1;i<=m;i++)
		{
		oil+=s[i];
		if (ss<s[i])
			ss=s[i];
		while (oil<d[i])
			{
			oil+=ss;
			ans+=k;
			}
		oil-=d[i];
		ans+=d[i];
		}
	printf("%d\n",ans);

	return 0;
}
