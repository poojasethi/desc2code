#include <stdio.h>
#define N 1010
int d[N],s[N];
int main()
{
	int m,k,i,t,sum,max;
	scanf("%d%d",&m,&k);
	for(i=1;i<=m;i++)
		scanf("%d",&d[i]);
	for(i=0;i<m;i++)
		scanf("%d",&s[i]);
	for(i=1,sum=s[0],t=0,max=s[0];i<=m;i++)
	{
		t+=d[i];
		while(d[i]>sum)
		{
			t+=k;
			sum+=max;
		}
		sum-=d[i];
		if(s[i]>max)
		{	
			max=s[i];
		
		}
		sum+=s[i];
	}
	printf("%d\n",t);
	return 0;
}
