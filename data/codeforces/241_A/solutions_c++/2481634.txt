#include<iostream>
using namespace std;
int main()
{
int m,k,i;
cin>>m>>k;
int d[m],s[m];
int c=0;
for(i=0;i<m;i++)
cin>>d[i];
for(i=0;i<m;i++)
cin>>s[i];
c=0;
int hr=0;
int mx=0;
for(i=0;i<m;i++)
{
c+=s[i];
if(s[i]>mx)
mx=s[i];
//cout<<c<<endl;
if(c<d[i])
{
while(c<d[i])
{
hr+=k;
c+=mx;
}
}


hr+=d[i];
c=c-d[i];

}
cout<<hr;
return 0;
}
