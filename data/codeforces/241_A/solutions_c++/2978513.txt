#include<iostream>
using namespace std;
int main()
{
	int n,k,i,d[1005],s[1005],x,p=0,r=0;
	cin>>n>>k;
	for(i=1;i<=n;i++)
		cin>>d[i];
	for(i=1;i<=n;i++)
		cin>>s[i];
	for(i=1;i<=n;i++)
	{
		r+=s[i]-d[i];
		p+=d[i];
		if(i==1 || x<s[i])
			x=s[i];
		while(r<0)
			p+=k,r+=x;
	}
	cout<<p;
	return 0;
}
