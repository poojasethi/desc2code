#include<stdio.h>

int main(){
	int m,k,i,j;
	scanf("%d %d",&m,&k);
	int d[m],fuel[m];
	for(i=0;i<m;i++)
		scanf("%d",&d[i]);
	for(i=0;i<m;i++)
		scanf("%d",&fuel[i]);
	int f,maxf,t;
	f=0;
	maxf=0;
	t=0;

	for(i=0;i<m;i++){
		f=f+fuel[i];
		if(maxf<fuel[i])
			maxf=fuel[i];

			while(f<d[i]){
				f=f+maxf;
				t=t+k;
			}
			f=f-d[i];
			t=t+d[i];
	}
		printf("%d\n",t);
	return 0;
}
