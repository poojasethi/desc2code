#include<cstdio>
#define go(i,n) for(int i=0;i<n;++i)
int d[1024],s[1024],a[1024],m,k;
int main(){
	scanf("%d%d",&m,&k);
	go(i,m)scanf("%d",d+i);
	go(i,m)scanf("%d",s+i);
	int n=0,t=0,f=0,j=0;
	go(i,m)if(s[i]>t){a[n++]=i;t=s[i];}
	a[n]=999999;t=0;
	go(i,m){
		f+=s[i];
		if(f<d[i]){
			while(a[j+1]<=i)++j;
			int w=(d[i]-f-1)/s[a[j]]+1;
			f+=w*s[a[j]];
			t+=w*k;
		}
		f-=d[i];
		t+=d[i];
	}
	printf("%d\n",t);
}
