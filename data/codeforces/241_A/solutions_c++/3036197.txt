#include <cstdio>

int d[1000];
int s[1000];
int m,k;
int main() {
	scanf("%d%d",&m,&k);
	for(int i=0;i<m;i++) scanf("%d",&d[i]);
	for(int i=0;i<m;i++) scanf("%d",&s[i]);
	int sol=0;
	int smax=0;
	int tank=0;
	for(int i=0;i<m;i++) {
		tank+=s[i];
		if(smax<s[i]) smax=s[i];
		while(tank<d[i]) {
			tank+=smax;
			sol+=k;
		}
		sol+=d[i];
		tank-=d[i];
	}
	printf("%d\n",sol);
	return 0;
}
