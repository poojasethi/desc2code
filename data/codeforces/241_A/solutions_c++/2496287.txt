#include <iostream>
#include <algorithm>
using namespace std;
int main(){
	int m,k,d[1000],s[1000],maxs=0,sumd=0,sums=0,time=0;
	cin >> m >> k;
	for(int i=0;i<m;i++) cin >> d[i];
	for(int i=0;i<m;i++) cin >> s[i];
	for(int i=0;i<m;i++){
		sumd+=d[i];
		sums+=s[i];
		maxs=max(maxs,s[i]);
		while(sumd>sums){
			time++;
			sums+=maxs;
		}
	}
	cout << sumd+time*k;
	return 0;
}
