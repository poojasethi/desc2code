#include<iostream>
using namespace std;
int main(){
	int m,k;
	cin>>m>>k;
	int i;
	int fuel=0;
	int a[m];
	int t;
	int sum=0;
	int max=0;
	for(i=0;i<m;i++)cin>>a[i];
	for(i=0;i<m;i++){cin>>t;
	if(t>max)max=t;
	fuel+=t;
	while(fuel<a[i]){sum+=k;fuel+=max;}
	sum+=a[i];
	fuel-=a[i];
	}
	cout<<sum<<"\n";
	return 0;
}
