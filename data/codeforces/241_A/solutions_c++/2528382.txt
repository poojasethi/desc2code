#include<iostream>
using namespace std;
main(){
    int ans=0,tank=0,i,m,k,s,mx;
    cin>>m>>k;
    int d[m];
    for(i=0;i<m;i++)cin>>d[i],ans+=d[i];

    for(i=0;i<m;i++){
        cin>>s;
        tank+=s-d[i];
        if(i==0 || mx<s) mx=s;
        while(tank<0)tank+=mx,ans+=k;
    }
    cout<<ans;
}
