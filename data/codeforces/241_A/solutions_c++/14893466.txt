#include <iostream>
#include <cstdio>

using namespace std;

int main(){
	int m,k;
	cin>>m>>k;
	int d[m],s[m];
	for(int i=0;i<m;i++){
		cin>>d[i];
	}
	long long time=0;
	long long td=0,ts=0;
	int ms;
	for(int i=0;i<m;i++){
		cin>>s[i];
		td+=d[i];
		ts+=s[i];
		time+=d[i];
		if(i==0 || ms<s[i]){
			ms=s[i];
		}
		while(td>ts){
			time+=k;
			ts+=ms;
		}
	}
	cout<<time;
	return 0;
}
