#include<cstdio>
#include<cstring>
#include<algorithm>
#include<iostream>
using namespace std;
int n,k,sum,ans,x,m;
int d[1005],s[1005];
int main()
{
	scanf("%d%d",&n,&k);
	for (int i=1;i<=n;i++) scanf("%d",&d[i]),ans+=d[i];
	for (int i=1;i<=n;i++) scanf("%d",&s[i]);
	for (int i=1;i<=n;i++){
		sum+=s[i]-d[i]; if (s[i]>m) m=s[i];
		if (sum<0) x=(-sum-1)/m+1,ans+=x*k,sum+=x*m;
		}
	printf("%d\n",ans); return 0;
}
