#include <iostream>

using namespace std;

const int MAX = 1005;
int M,K, D[MAX], S[MAX];

int main(){
    cin >> M >> K;
    for(int i=0; i<M; i++){
	cin >> D[i];
    }
    for(int i=0; i<M; i++){
	cin >> S[i];
    }
    int hi = 0, cur = 0, res = 0;
    for(int i=0; i<M; i++){
	res += D[i];
	cur = cur + S[i] - D[i];
	hi = max (hi, S[i]);
	while (cur < 0){
	    cur += hi;
	    res += K;
	}
    }
    cout << res << endl;
}
