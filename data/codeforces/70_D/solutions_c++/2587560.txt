#include<cstdio>
#include<map>
#include<iostream>

using namespace std;

map <int,int> Up,Down;

int Q,T,x,y;

long long XJ(pair<int,int> A,pair<int,int> B,pair<int,int> C)
{
	return 1ll*(B.first-A.first)*(C.second-A.second)-1ll*(C.first-A.first)*(B.second-A.second);
}

bool Check(map<int,int>&Map,int x,int y)
{
	map <int,int> ::iterator L,R;
	if(Map.count(x))
		return Map[x]<y;
	L=R=Map.upper_bound(x);
	if(R==Map.begin()||R==Map.end())return 1;
	L--;
	if(XJ(*L,make_pair(x,y),*R)<0)
		return 1;
	return 0;
}

void Add(map<int,int>&Map,int x,int y)
{
	map <int,int> ::iterator L,_L,R,_R;
	Map[x]=y;
	L=Map.lower_bound(x);
	if(L!=Map.begin())
	{
		L--;
		for(;L!=Map.begin();)
		{
			_L=L;
			_L--;
			if(XJ(*_L,*L,make_pair(x,y))<0)break;
			Map.erase(L);
			L=_L;
		}
	}
	R=Map.upper_bound(x);
	if(R!=Map.end())
	{
		for(;R!=Map.end();)
		{
			_R=R;
			_R++;
			if(_R==Map.end())break;
			if(XJ(make_pair(x,y),*R,*_R)<0)break;
			Map.erase(R);
			R=_R;
		}
	}
}

int main()
{
	for(scanf("%d",&Q);Q--;)
	{
		scanf("%d%d%d",&T,&x,&y);
		if(T==2)
		{
			if(Check(Up,x,y)||Check(Down,x,-y))
				printf("NO\n");
			else
				printf("YES\n");
		}
		else
		{
			if(Check(Up,x,y))
Add(Up,x,y);
			if(Check(Down,x,-y))
Add(Down,x,-y);
		}
	}
	return 0;
}
