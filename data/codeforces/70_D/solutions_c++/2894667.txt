#include<cstdio>
#include<map>
#include<cmath>
using namespace std;
#define pr pair<int,int>
#define x first
#define y second
#define ll long long
typedef map<double,pr > :: iterator It;
map<double,pr > hull;
pr A,B,C,D,O;
void read(pr &A){scanf("%d%d",&A.x,&A.y),A.x=A.x*3-O.x,A.y=A.y*3-O.y;}
ll xot(ll x1,ll y1,ll x2,ll y2){return x1*y2-x2*y1;}
ll det(pr A,pr B,pr C){return xot(B.x-A.x,B.y-A.y,C.x-A.x,C.y-A.y);}
It pre(It now){if (now==hull.begin()) return --hull.end(); return --now;}
It suc(It now){if (++now==hull.end()) return hull.begin(); return now;}
bool check(pr A,double a)
{
	It now=hull.upper_bound(a); now=pre(now);
	return det(suc(now)->y,now->y,A)<=0;
}
void add(pr A,double a,bool flag)
{
	if (flag && check(A,a)) return; It now=hull.find(a);
	if (now!=hull.end()) hull.erase(now); now=hull.insert(make_pair(a,A)).x;
	while (det(suc(suc(now))->y,suc(now)->y,now->y)>0) hull.erase(suc(now));
	while (det(now->y,pre(now)->y,pre(pre(now))->y)>0) hull.erase(pre(now));
}
void ins(pr A){A.x-=O.x,A.y-=O.y; double a=atan2(A.y,A.x); add(A,a,0);}
int main()
{
	int n,type; scanf("%d",&n),n-=3;
	scanf("%*d"),read(A),scanf("%*d"),read(B),scanf("%*d"),read(C);
	O=make_pair((A.x+B.x+C.x)/3,(A.y+B.y+C.y)/3); ins(A),ins(B),ins(C);
	while (n--){
		scanf("%d",&type),read(A); double a=atan2(A.y,A.x);
		if (type==1) add(A,a,1); else puts(check(A,a)?"YES":"NO");
		}

	return 0;
}
