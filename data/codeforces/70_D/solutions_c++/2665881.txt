#include <map>
#include <cstdio>

using namespace std;

typedef map <int, int> MII;
typedef long long LL;

MII low, upp;
int Q;

void init()
{
	scanf("%d", &Q);
	low.clear(); upp.clear();
}

inline LL Cross(LL x0, LL y0, LL x1, LL y1, LL x2, LL y2){
	return (x1 - x0) * (y2 - y0) - (x2 - x0) * (y1 - y0);
}

bool check(MII &H, int x, int y)
{
	if (H.empty()) return 0;
	if (x < H.begin()->first || x > H.rbegin()->first) return 0;
	if (H.find(x) != H.end()) return y <= H[x];
	MII::iterator r = H.lower_bound(x), l = r; l --;
	return Cross(l->first, l->second, x, y, r->first, r->second) >= 0;
}

void add(MII &H, int x, int y)
{
	H[x] = y;
	MII::iterator it = H.find(x), i, j;
	if (x > H.begin()->first)
		for (i = it, i --, j = i, j --; i != H.begin(); i = j, j = i, j --)
			if (Cross(j->first, j->second, i->first, i->second, x, y) >= 0) H.erase(i);
				else break;
	if (x < H.rbegin()->first)
		for (i = it, i ++, j = i, j ++; j != H.end(); i = j, j = i, j ++)
			if (Cross(x, y, i->first, i->second, j->first, j->second) >= 0) H.erase(i);
				else break;
}

void print()
{
	int type, x, y;
	for (int t = 0; t < Q; t ++){
		scanf("%d%d%d", &type, &x, &y);
		if (type == 1){
			if (!check(low, x, y)) add(low, x, y);
			if (!check(upp, x, -y)) add(upp, x, -y);
		} else {
			if (check(low, x, y) && check(upp, x, -y)) puts("YES");
				else puts("NO");
		}
	}
}

int main()
{
	init();
	print();
	return 0;
}
