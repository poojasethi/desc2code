#include<cstdio>
#include<map>
using namespace std;
#define fs first
#define sc second
typedef map<int,int> mi;
typedef mi::iterator mii;
typedef long long ll;
ll cross(int xx,int xy,int yx,int yy){
	return xx*ll(yy)-xy*ll(yx);
}

mi upp,low;
bool ck(mi &a,int x,int y){
	mii it1,it2;
	it1=a.lower_bound(x);
	if (it1==a.end()) return false;
	if (it1->first==x)
		return y<=it1->second;
	if (it1==a.begin()) return false;
	it2=it1--;
	return cross(x-it1->fs,y-it1->sc,it2->fs-x,it2->sc-y)>=0;
}
void add(mi &a,int x,int y){
	mii it1,it2,it3,itf;
	if (ck(a,x,y)) return;
	a[x]=y;
	itf=a.find(x);
	for (;;a.erase(it2)){
		it1=it2=it3=itf;
		it1++;
		if (it1==a.end()) break;
		it1++;it2++;
		if (it1==a.end()) break;
		if (cross(it2->fs-it3->fs,it2->sc-it3->sc,it1->fs-it2->fs,it1->sc-it2->sc)<0) break;
	}
	for (;;a.erase(it2)){
		it1=it2=it3=itf;
		if (it1==a.begin()) break;
		it1--;
		if (it1==a.begin()) break;
		it1--;it2--;
		if (cross(it2->fs-it3->fs,it2->sc-it3->sc,it1->fs-it2->fs,it1->sc-it2->sc)>0) break;
	}
}
int main(){
	int n,i,t,x,y;
	scanf("%d",&n);
	for (i=0;i<n;i++){
		scanf("%d%d%d",&t,&x,&y);
		if (t==1) add(upp,x,y),add(low,x,-y); else
			puts(ck(upp,x,y) && ck(low,x,-y)?"YES":"NO");
	}
	return 0;
}
