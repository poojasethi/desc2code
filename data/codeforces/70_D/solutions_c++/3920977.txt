#include<cstdio>
#include<iostream>
#include<map>
#include<algorithm>
using namespace std;
typedef long long LL;
typedef pair<int, int> PII;
#define MP make_pair
map<int, int>sta[2];//sta[0]下凸包,sta[1]上凸包
LL cross(PII a, PII b, PII c){
	return LL(b.first-a.first)*LL(c.second-a.second)-LL(c.first-a.first)*LL(b.second-a.second);
}
bool check(map<int, int>&sta, int x, int y){
	if(sta.empty())return false;
	if(sta.find(x)!=sta.end())return y >= sta[x];
	if(x < sta.begin()->first || (--sta.end())->first < x)return false;
	map<int, int>::iterator t1 = sta.upper_bound(x), t2=t1;t2--;
	return cross(MP(x, y), *t2, *t1)>=0;
}
int insert(map<int, int>&sta, int x, int y){
	if(check(sta, x, y))return 0;sta[x]=y;
	map<int, int>::iterator t1 = sta.upper_bound(x), t2=t1;t2++;
	if(t1 != sta.end()){
		while(t1 != sta.end() && t2 != sta.end() && cross(MP(x, y), *t2, *t1)>=0){
			sta.erase(t1);t1=sta.upper_bound(x);t2=t1;t2++;
		}
	}
	if(sta.find(x) == sta.begin())return 0;
	t1=sta.find(x);t1--;t2=t1;t2--;
	while(t1 != sta.begin() && t2 != sta.end() && cross(MP(x, y), *t1, *t2)>=0){
		sta.erase(t1);t1=sta.find(x);t1--;t2=t1;t2--;
	}
	return 0;
}
int n, i;
int main(){
	scanf("%d", &n);
	for(i=1;i<=n;i++){
		int x, y, t;
		scanf("%d%d%d", &t, &x, &y);
		if(t == 1)insert(sta[0], x, y), insert(sta[1], x, -y);else{
			if(check(sta[0], x, y)&&check(sta[1], x, -y))printf("YES\n");else printf("NO\n");
		}
	}
}
