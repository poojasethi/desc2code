const int MAXN=200*1000 + 5;
//const int MOD=1000*1000*1000 + 7;
//const lli INF=1000000000000000000ll;
pll b[MAXN];
lli dp[MAXN];//chand sanie tool mikeshe
lli rem[MAXN];//cheghad mimoone

inline pll cst(int x,int y)//x>y
{
    lli tt=((b[x].FF-rem[y]+b[y].SS-1)/b[y].SS);
    return MP(tt+dp[y],rem[y]+tt*b[y].SS-b[x].FF);
}
bool cmp(const pll& x,const pll& y){if(x.FF!=y.FF)return x.FF<y.FF;return x.SS>y.SS;}
inline bool isBet(int i,int j,int k){return !cmp(cst(i,k),cst(i,j));}//i az roo j  behtar az i az roo k
int main()
{
	ios::sync_with_stdio(false);
	int n;
	lli s;
	cin>>n>>s;
	for(int i=0;i<n;i++)
        cin>>b[i].SS>>b[i].FF;
    b[n].FF=s;
    sort(b,b+n,cmp);
    assert(!b[0].FF);
    deque<pii> cands;
    cands.push_back(MP(0,1));
    lli ans=cst(n,0).FF;
    lli lst=b[0].SS;
    for(int i=1;i<n;i++)
    {
        if(b[i].SS<=lst)continue;
        while(cands.size()>1 && isBet(i,cands[1].FF,cands[0].FF))
            cands.pop_front();
        pll tmp=cst(i,cands[0].FF);
        dp[i]=tmp.FF;
        rem[i]=tmp.SS;
        ans=min(ans,cst(n,i).FF);
        while(cands.size() && isBet(cands.back().SS,i,cands.back().FF))
            cands.pop_back();
        if(cands.size())
        {
            int d=cands.back().SS,u=n-1;
            while(d<u-1)
            {
                int mid=(d+u)/2;
                if(isBet(mid,i,cands.back().FF))
                    u=mid;
                else
                    d=mid;
            }
            if(isBet(u,i,cands.back().FF))
                cands.push_back(MP(i,u));
        }
        lst=b[i].SS;
    }
    cout<<ans<<endl;
    return 0;
}
