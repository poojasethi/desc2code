#include<cstdio>
#include<algorithm>
using namespace std;
typedef long long ll;
#define rep(i,n) for (int i=0;i<n;++i)
const int N=200005;
int n,h,t,p[N],a[N]; ll s,d,V,v[N],c[N],f[N],ans=1LL<<60;
bool cmp(const int i,const int j){return c[i]<c[j];}
ll day(ll s,int i){return max(0LL,(s-f[i]+v[i]-1)/v[i]);}
bool check(int i,int j,ll s)
{
	ll Di=day(s,i),Dj=day(s,j);
	if (Di!=Dj) return Di>Dj;
	return f[i]+Di*v[i]<=f[j]+Dj*v[j];
}
ll T(int i,int j){return (f[i]-f[j])/(v[j]-v[i]);}
int main()
{
	scanf("%d%I64d",&n,&s),h=1,t=0;
	rep(i,n) scanf("%I64d%I64d",v+i,c+i),p[i]=i; sort(p,p+n,cmp);
	rep(k,n){
		int i=p[k],j; if (v[i]<=V) continue; V=v[i];
		while (h<t && check(a[h],a[h+1],c[i])) ++h;
		if (k) j=a[h],f[i]=f[j]-c[i]-day(c[i],j)*(v[i]-v[j]);
		ans=min(ans,day(s,i));
		while (h<t && T(a[t-1],a[t])>=T(a[t],i)) --t; a[++t]=i;
	}
	printf("%I64d\n",ans); return 0;
}
