#include<cstdio>
#include<algorithm>
#define LL long long
using namespace std;
struct Pair{
	LL x,y;
	Pair(){}
	Pair(LL x,LL y):
		x(x),y(y){}
}d[400005],r[400005],P[400005];
LL N,n,mm;
Pair work(LL x,LL y)
{
	Pair tmp;
	tmp.x=y;
	LL l=1,r=N,m;
	while(l<r)
	{
		m=l+r+1>>1;
		if(P[m].y<=x)
			l=m;
		else
			r=m-1;
	}
	LL dx=(x-d[l].y)/d[l].x;
	if(dx*d[l].x+d[l].y<x)
		dx++;
	if(l<N&&dx>=P[l+1].x)
	{
		l++;
		dx=P[l].x;
	}
	LL dy=dx*d[l].x+d[l].y-x;
	tmp.y=dy-dx*y;
	return tmp;
}
Pair Calc(Pair x,Pair y)
{
	LL dx=(y.y-x.y)/(x.x-y.x);
	if(dx*x.x+x.y<dx*y.x+y.y)
		dx++;
	return Pair(dx,dx*x.x+x.y);
}
bool cmp(const Pair &x,const Pair &y)
{
	return x.x-y.x?x.x<y.x:x.y>y.y;
}
int main()
{
#ifndef ONLINE_JUDGE
	freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);
#endif
	scanf("%I64d%I64d",&n,&mm);
	for(LL i=1;i<=n;i++)
		scanf("%I64d%I64d",&d[i].y,&d[i].x);
	sort(d+1,d+1+n,cmp);
	for(LL i=1;i<=n;i++)
		if(d[i].y>d[N].y)
			d[++N]=d[i];
	n=N;N=1;
	d[1]=Pair(d[1].y,0);
	P[1]=Pair(0,0);
	for(LL i=2;i<=n;i++)
	{
		Pair tmp=work(d[i].x,d[i].y);
		while(P[N].x*tmp.x+tmp.y>=P[N].y)
			N--;
		P[N+1]=Calc(tmp,d[N]);
		d[++N]=tmp;
	}
	LL l=1,r=N,m;
	while(l<r)
	{
		m=l+r+1>>1;
		if(P[m].y<=mm)
			l=m;
		else
			r=m-1;
	}
	LL ans=(mm-d[l].y)/d[l].x;
	if(ans*d[l].x+d[l].y<mm)
		ans++;
	if(l<N&&ans>=P[l+1].x)
	{
		l++;
		ans=P[l].x;
	}
	printf("%I64d\n",ans);
	return 0;
}
