#include <cstdio>

const int MAX_N = 1005;

typedef long long int64;

int n, m;
int64 f[4][MAX_N][MAX_N];

inline void add(int x, int y, int64 z) {
  int t = (x&1)<<1 | (y&1);
  for(; x <= n; x += x&-x) {
    for(int j = y; j <= n; j += j&-j) {
      f[t][x][j] ^= z;
    }
  }
}

inline int64 ask(int x, int y) {
  int64 ret = 0;
  int t = (x&1)<<1 | (y&1);
  for(; x; x -= x&-x) {
    for(int j = y; j; j -= j&-j) {
      ret ^= f[t][x][j];
    }
  }
  return ret;
}

int main() {
  scanf("%d %d", &n, &m);
  while(m --> 0) {
    int c, xa, xb, ya, yb;
    scanf("%d %d %d %d %d", &c, &xa, &ya, &xb, &yb);
    if(c == 1) {
      printf("%lld\n", ask(xb, yb) ^ ask(xa-1, yb) ^ ask(xb, ya-1) ^ ask(xa-1, ya-1));
    } else {
      int64 z;
      scanf("%I64d\n", &z);
      add(xa, ya, z);
      add(xa, yb+1, z);
      add(xb+1, ya, z);
      add(xb+1, yb+1, z);
    }
  }
}
