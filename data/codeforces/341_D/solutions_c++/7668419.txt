#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn=1e3+10;
int fen[5][maxn][maxn],n,q;
void add(int x,int y,long long val){
	int tmp=(x%2)*2+(y%2);
	for(int i=x ; i<maxn ; i+=(i&-i))
		for(int j=y ; j<maxn ; j+=(j&-j))
			fen[tmp][i][j] ^= val;
}
ll get(int x,int y){
	int tmp=(x%2)*2+(y%2);
	ll ans=0;
	for(int i=x ; i>0 ; i-=(i&-i))
		for(int j=y ; j>0 ; j-=(j&-j))
			ans ^= fen[tmp][i][j];
	return ans;
}
int main(){
	ios_base::sync_with_stdio(false);
	cin>>n>>q;
	while(q--){
		ll t,x1,y1,x2,y2,val;
		cin>>t>>x1>>y1>>x2>>y2;
		x1++;x2++;
		y1++;y2++;
		if(t==2){
			x2++;y2++;
			cin>>val;
			add(x1,y1,val);add(x1,y2,val);
			add(x2,y1,val);add(x2,y2,val);
		}
		else{
			x1--;y1--;
			ll ans=get(x1,y1)^ get(x1,y2)^ get(x2,y1)^ get(x2,y2);
			cout<<ans<<endl;
		}
	}
	return 0;
}
