#include<cstdio>
#include<cstdlib>
#include<cstring>

using namespace std;

#define lb(x) ((x)&(-(x)))

const int maxn=1010;

int n,m,z[2][2][maxn][maxn];

void modify(int x,int y,long long v)
{
	for (int a=x;a<=n;a+=lb(a))
		for (int b=y;b<=n;b+=lb(b))
			z[x&1][y&1][a][b]^=v;
}

long long query(int x,int y)
{
	long long ans=0;
	for (int a=x;a;a-=lb(a))
		for (int b=y;b;b-=lb(b))
			ans^=z[x&1][y&1][a][b];
	return ans;
}

int main()
{
	scanf("%d%d",&n,&m);
	for (int a=1;a<=m;a++)
	{
		int opt,x1,y1,x2,y2;
		scanf("%d%d%d%d%d",&opt,&x1,&y1,&x2,&y2);
		if (opt==1) printf("%I64d\n",query(x1-1,y1-1)^query(x2,y1-1)^query(x1-1,y2)^query(x2,y2));
		else
		{
			long long v;
			scanf("%I64d",&v);
			modify(x1,y1,v);
			modify(x2+1,y1,v);
			modify(x1,y2+1,v);
			modify(x2+1,y2+1,v);
		}
	}
	return 0;
}
