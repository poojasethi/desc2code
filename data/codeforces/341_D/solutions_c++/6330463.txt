#include <iostream>
using namespace std;
const int N = 1100;

int n, m;
long long f[4][N][N];

void add(long long val, int x, int y) {
	int l = (x % 2) * 2 + (y % 2);
	for (int i = x; i < N; i += i & -i)
		for (int j = y; j < N; j += j & -j)
			f[l][i][j] ^= val;
}
long long query(int x, int y) {
	int l = (x % 2) * 2 + (y % 2);
	long long res = 0;
	for (int i = x; i; i -= i & -i)
		for (int j = y; j; j -= j & -j)
			res ^= f[l][i][j];
	return res;
}
	
int main() {
	ios_base::sync_with_stdio(false);
	cin >> n >> m;
	while (m--) {
		int t, x1, x2, y1, y2;
		cin >> t >> x1 >> y1 >> x2 >> y2;
		x1++, y1++, x2++, y2++;
		if (t == 2) {
			long long val; cin >> val;
			x2++, y2++;
			add(val, x1, y1), add(val, x1, y2), add(val, x2, y1), add(val, x2, y2);
		}
		else {
			x1--, y1--;
			cout << (query(x2, y2) ^ query(x2, y1) ^ query(x1, y2) ^ query(x1, y1)) << '\n';
		}
	}
	return 0;
}
