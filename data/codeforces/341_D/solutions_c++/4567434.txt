#include <cstdio>
#define ll long long
using namespace std;

int N,M;
ll s[4][1005][1005];

void add(int x,int y,ll z)
{
    int t=(x&1)*2+(y&1);
    for (int i=x; i<=N; i+=i&-i)
        for (int j=y; j<=N; j+=j&-j) s[t][i][j]^=z;
}

ll sum(int x,int y)
{
    int t=(x&1)*2+(y&1); ll z=0;
    for (int i=x; i; i-=i&-i)
        for (int j=y; j; j-=j&-j) z^=s[t][i][j];
    return z;
}

void doit()
{
    scanf("%d%d",&N,&M); ll z;
    for (int i=1,t,x1,x2,y1,y2; i<=M; i++)
    {
        scanf("%d%d%d%d%d",&t,&x1,&y1,&x2,&y2);
        if (t==1) x1--,y1--,printf("%I64d\n",sum(x1,y1)^sum(x1,y2)^sum(x2,y1)^sum(x2,y2)); else scanf("%I64d",&z),x2++,y2++,add(x1,y1,z),add(x1,y2,z),add(x2,y1,z),add(x2,y2,z);
    }
}

int main()
{
    doit();
    return 0;
}
