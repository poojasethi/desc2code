#include<cstdio>
#include<cstdlib>
#define N 4005
#define rep(i,l,r) for(int i=l;i<=r;i++)
#define lowbit(x) x&(-x)
int n,m,ll,rr,ch,x0,x1,y0,y1;
long long ans,val,f[3][3][1005][1005];
void ins(int x,int y){
	int xx=x&1,yy=y&1;
	for (int i=x;i<=n;i+=lowbit(i))
		for (int j=y;j<=n;j+=lowbit(j)) f[xx][yy][i][j]^=val;
}
long long query(int x,int y){
	int xx=x&1,yy=y&1;
	long long res=0;
	for (int i=x;i;i-=lowbit(i))
		for (int j=y;j;j-=lowbit(j)) res^=f[xx][yy][i][j];
	return res;
}
void work(){
	scanf("%d%d",&n,&m);
	while (m--){
		scanf("%d%d%d%d%d",&ch,&x0,&y0,&x1,&y1);
		if(ch==1){
			printf("%I64d\n",query(x1,y1)^query(x0-1,y0-1)^query(x0-1,y1)^query(x1,y0-1));
		}
		if(ch==2){
			scanf("%I64d",&val);
			ins(x1+1,y1+1); ins(x0,y0);
			ins(x0,y1+1); ins(x1+1,y0);
		}
	}
}
int main(){
	work();
}
