#include <bits/stdc++.h>

using namespace std;

#define LL long long
#define ULL unsigned long long
#define pii pair<int, int>
#define MP make_pair
#define ls i << 1
#define rs ls | 1
#define md (ll + rr >> 1)
#define lson ll, md, ls
#define rson md + 1, rr, rs
#define mod 1000000007
#define inf 0x3f3f3f3f
#define eps 1e-12
#define N 1010
#define M 200020
#define Pi acos(-1.0)

LL dp[2][2][N][N];
int n, m;
LL query(int x, int y){
	int ret = 0;
	for(int i = x; i > 0; i -= i & -i)
		for(int j = y; j > 0; j -= j & -j)
			ret ^= dp[x&1][y&1][i][j];
	return ret;
}
void update(int x, int y, LL v){
	for(int i = x; i < N; i += i & -i)
		for(int j = y; j < N; j += j & -j)
			dp[(x+1)&1][(y+1)&1][i][j] ^= v;
}
int main(){
	scanf("%d%d", &n, &m);
	while(m--){
		int op, x, y, xx, yy;
		scanf("%d%d%d%d%d", &op, &x, &y, &xx, &yy);
		xx++, yy++;
		if(op == 1){
			LL ans = query(x, y) ^ query(x, yy) ^ query(xx, y) ^ query(xx, yy);
			printf("%lld\n", ans);
		}
		else{
			LL v;
			scanf("%lld", &v);
			update(x, y, v);
			update(x, yy, v);
			update(xx, y, v);
			update(xx, yy, v);
		}	
	}
	return 0;
}
