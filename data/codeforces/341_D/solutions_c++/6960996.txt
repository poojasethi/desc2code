#include <bits/stdc++.h>
using namespace std;
const int MAX = 1e3+10;
int C[2][2][MAX][MAX];
int n,m;
inline int lowbit(int x) {return x&-x;}
void modify(int x,int y,int val) {
    for(int i=x;i<=n;i+=lowbit(i)) {
        for(int j=y;j<=n;j+=lowbit(j)) {
            C[x&1][y&1][i][j]^=val;
        }
    }
}
int sum(int x,int y) {
    int res=0;
    for(int i=x;i;i-=lowbit(i)) {
        for(int j=y;j;j-=lowbit(j)) {
            res^=C[x&1][y&1][i][j];
        }
    }
    return res;
}
int main() {
    int x0,y0,x1,y1,c,v;
    while(scanf("%d %d",&n,&m)==2) {
        for(int i=0;i<m;i++) {
            scanf("%d",&c);
            if(c==1) {
                scanf("%d %d %d %d",&x0,&y0,&x1,&y1);
                int ans=sum(x1,y1)^sum(x0-1,y0-1)^sum(x0-1,y1)^sum(x1,y0-1);
                printf("%d\n",ans);
            }
            else {
                scanf("%d %d %d %d %d",&x0,&y0,&x1,&y1,&v);
                modify(x0, y0, v);
                modify(x0, y1 + 1, v);
                modify(x1 + 1, y0, v);
                modify(x1 + 1, y1 + 1, v);

            }
        }
    }
}
