#include <bits/stdc++.h>
typedef long long LL;
#define rep(i,n) for(int i = 0;i < n;i ++)

const int N = 1024;
LL sum[2][2][N][N];
int n;
void update(int x, int y, LL val) {
	for(int i = x; i <= n; i += i&-i)
		for(int j = y; j <= n; j += j&-j)
			sum[x&1][y&1][i][j] ^= val;
}
LL query(int x, int y) {
	LL ret = 0;
	for(int i = x; i > 0; i -= i&-i)
		for(int j = y; j > 0; j -= j&-j)
			ret ^= sum[x&1][y&1][i][j];
	return ret;
}

	int op, x1, y1, x2, y2, m;
	LL val;
	scanf("%d%d", &n, &m);
	rep(i, m) {
		scanf("%d%d%d%d%d", &op, &x1, &y1, &x2, &y2);
		if(op == 1) {
			LL ret = 0;
			ret ^= query(x1 - 1, y1 - 1);
			ret ^= query(x1 - 1, y2);
			ret ^= query(x2, y1 - 1);
			ret ^= query(x2, y2);
			printf("%I64d\n", ret);
		} else {
			scanf("%I64d", &val);
			update(x1, y1, val);
			update(x2 + 1, y1, val);
			update(x1, y2 + 1, val);
			update(x2 + 1, y2 + 1, val);
		}
	}
	return 0;
}
