#include <iostream>
#include <cstdio>
#include <vector>
#include <map>
#include <algorithm>
#include <string.h>
using namespace std;
int N,M;
long long c[2][2][1010][1010];

int lowbit(int x){
    return x&(-x);
}

void add(int x,int y,long long v){
    for (int i=x;i<=N;i+=lowbit(i))
        for (int j=y;j<=N;j+=lowbit(j))
            c[x&1][y&1][i][j]^=v;
}

long long get(int x,int y){
    long long ans=0;
    for (int a=x;a;a-=lowbit(a))
        for (int b=y;b;b-=lowbit(b))
            ans^=c[x&1][y&1][a][b];
    return ans;
}


int main(){
    //freopen("input.txt","r",stdin);
    scanf("%d%d",&N,&M);
    for (int i=1;i<=M;++i){
        int opt,x1,x2,y1,y2;
        scanf("%d%d%d%d%d",&opt,&x1,&y1,&x2,&y2);
        if (opt==1){
            cout<<(get(x1-1,y1-1)^get(x2,y1-1)^get(x1-1,y2)^get(x2,y2))<<endl;
        }
        else{
            long long v;
            cin>>v;
            add(x1,y1,v);
            add(x2+1,y1,v);
            add(x1,y2+1,v);
            add(x2+1,y2+1,v);
        }
    }
    return 0;
}
