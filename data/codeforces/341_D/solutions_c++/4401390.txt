#include<cstdio>
#include<cstring>
#define ll long long
const int Maxn=1005;

ll sum[4][Maxn][Maxn];
int x0,y0,x1,y1,n,q,op;
ll v,ans;

int lowbit(int x) { return x&-x;}

int parity(int x,int y)
{
	int ans=0;
	if(x%2)
	  ans+=1;
	if(y%2)
	  ans+=2;
	return ans;
}

void update(int x,int y,ll val)
{
	int p=parity(x,y);
	for(int i=x;i<=n;i+=lowbit(i))
	  for(int j=y;j<=n;j+=lowbit(j))
		sum[p][i][j]^=val;
}

ll query(int x,int y)
{
	ll ans=0;
	int p=parity(x,y);
	for(int i=x;i>0;i-=lowbit(i))
	  for(int j=y;j>0;j-=lowbit(j))
		ans^=sum[p][i][j];
	return ans;
}

int main()
{
	memset(sum,0,sizeof(sum));
	scanf("%d%d",&n,&q);
	for(int i=0;i<q;i++)
	{
		scanf("%d%d%d%d%d",&op,&x0,&y0,&x1,&y1);
		if(op==1)
		{
			ans=query(x1,y1);
			ans^=query(x0-1,y0-1);
			ans^=query(x0-1,y1);
			ans^=query(x1,y0-1);
			printf("%I64d\n",ans);
		}
		else
		{
			scanf("%I64d",&v);
			update(x0,y0,v);
			update(x0,y1+1,v);
			update(x1+1,y0,v);
			update(x1+1,y1+1,v);
		}
	}
	return 0;
}
