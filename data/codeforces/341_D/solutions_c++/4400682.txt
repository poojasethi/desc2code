#include<iostream>
using namespace std;
long long tree[2][2][1010][1010];
int n, m;
void add(int x, int y, int v)
{
	for (int i=x; i<=n; i+=i & -i)
		for (int j=y; j<=n; j+=j & -j)
			tree[x &1][y &1][i][j]^=v;
}

long long find(int x, int y)
{
	long long ans=0;
	for (int i=x; i>0; i-=i & -i)
		for (int j=y; j>0; j-=j & -j)
			ans^=tree[x &1][y &1][i][j];
	return ans;
}
int main()
{
	int c, x1, y1, x2, y2;
	long long v;
	cin>>n>>m;
	n++;
	for (int i=0; i<m; i++)
	{
		cin>>c>>x1>>y1>>x2>>y2;
		x1++;y1++;x2++;y2++;
		if (c==2)
		{
			cin>>v;
			add(x1, y1, v);
			add(x1, y2+1, v);
			add(x2+1, y1, v);
			add(x2+1, y2+1, v);
		}
		else
		{
			long long ans=find(x1-1, y1-1)^find(x1-1, y2)^find(x2, y1-1)^find(x2, y2);
			cout<<ans<<endl;
		}
	}
}
