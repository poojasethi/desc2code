#include <cstdio>
using namespace std;

typedef long long ll;
const int N = 1010;

int n, m, x1, x2, y1, y2, d;
ll c[2][2][N][N], v;

int lowbit(int x) {
    return x & -x;
}

void modify(int x, int y, ll v) {
    for (int i = x; i <= n; i += lowbit(i))
        for (int j = y; j <= n; j += lowbit(j))
            c[x & 1][y & 1][i][j] ^= v;
}

ll xorsum(int x, int y) {
    ll ret = 0;
    for (int i = x; i > 0; i -= lowbit(i))
        for (int j = y; j > 0; j -= lowbit(j))
            ret ^= c[x & 1][y & 1][i][j];
    return ret;
}

int main() {
//    freopen("in", "r", stdin);
//    freopen("out", "w", stdout);
    scanf("%d%d", &n, &m);
    while (m--) {
        scanf("%d%d%d%d%d", &d, &x1, &y1, &x2, &y2);
        if (d == 2) {
            scanf("%I64d", &v);
            modify(x1, y1, v);
            modify(x1, y2 + 1, v);
            modify(x2 + 1, y1, v);
            modify(x2 + 1, y2 + 1, v);
        }
        else
            printf("%I64d\n", xorsum(x2, y2) ^ xorsum(x1 - 1, y1 - 1) ^ xorsum(x1 - 1, y2) ^ xorsum(x2, y1 - 1));
    }

    return 0;
}
