#include<iostream>
#include<cstring>
using namespace std;
const int N=1010;
typedef long long ll;
ll sum[2][2][N][N];
int n;
void update(int x,int y,ll val){
	for(int i=x;i<=n;i+=i&-i)
		for(int j=y;j<=n;j+=j&-j)
			sum[x&1][y&1][i][j]^=val;
}
ll query(int x,int y){
	ll ans=0;
	for(int i=x;i>0;i-=i&-i)
		for(int j=y;j>0;j-=j&-j)
		ans^=sum[x&1][y&1][i][j];
	return ans;
}
int main(){
	int m;
	while(cin>>n>>m){
		for(int i=1;i<=m;i++){
			int op,x0,y0,x1,y1;
			ll val;
			cin>>op;
			if(op==1){
				cin>>x0>>y0>>x1>>y1;
				cout<<(query(x1,y1)^query(x1,y0-1)^query(x0-1,y1)^query(x0-1,y0-1))<<endl;
			}else{
				cin>>x0>>y0>>x1>>y1>>val;
				update(x1+1,y1+1,val);
				update(x0,y0,val);
				update(x0,y1+1,val);
				update(x1+1,y0,val);
			}
		}
	}
	return 0;
}
