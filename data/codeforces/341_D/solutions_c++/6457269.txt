#include <cstdio>
#include <algorithm>

const int MAXN = 1005;

typedef long long llint;

llint loga[2][2][MAXN][MAXN];
int n, m;

void update (int x, int y, llint v){
  for (int i = x; i <= n; i += i&-i)
    for (int j = y; j <= n; j += j&-j)
      loga[x&1][y&1][i][j] ^= v;
}

llint query (int x, int y){
  llint ret = 0;
  for (int i = x; i > 0; i -= i&-i)
    for (int j = y; j > 0; j -= j&-j)
      ret ^= loga[x&1][y&1][i][j];

  return ret;
}

void update (int x1, int y1, int x2, int y2, int v){
  update (x1, y1, v);
  update (x2+1, y1, v);
  update (x1, y2+1, v);
  update (x2+1, y2+1, v);
}

llint query(int x1, int y1, int x2, int y2){
  return query(x2, y2) ^ query(x1-1, y2) ^ query(x2, y1-1) ^ query(x1-1, y1-1);
}

int main (void){
  scanf("%d%d", &n, &m);
  for (int i = 0; i < m; ++i){
    int t, x1, y1, x2, y2;
    llint v;
    scanf ("%d%d%d%d%d", &t, &x1, &y1, &x2, &y2);
    if (t == 2){
      scanf("%lld", &v);
      update(x1, y1, x2, y2, v);
    }
    else printf("%lld\n",query(x1, y1, x2, y2));
  }
  return 0;
}
