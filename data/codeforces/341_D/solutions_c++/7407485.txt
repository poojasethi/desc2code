#include<cstdio>
#include<cstring>
#define LL long long
LL c[2][2][1010][1010],v;
int n,m;
int lowbit(int x)
{
    return x&(-x);
}
void update(int x,int y)
{
    for(int i=x;i<=n;i+=lowbit(i))
        for(int j=y;j<=n;j+=lowbit(j))
        c[x&1][y&1][i][j]^=v;
}
LL getsum(int x,int y)
{
    LL res=0;
    for(int i=x;i>0;i-=lowbit(i))
        for(int j=y;j>0;j-=lowbit(j))
        res^=c[x&1][y&1][i][j];
        return res;
}
int main()
{
    while(scanf("%d%d",&n,&m)>0)
    {
        memset(c,0,sizeof(c));
        while(m--)
        {
            int op,x1,y1,x2,y2;
            scanf("%d%d%d%d%d",&op,&x1,&y1,&x2,&y2);
            if(op==1)
            {
                LL ans=getsum(x2,y2)^getsum(x2,y1-1)^getsum(x1-1,y2)^getsum(x1-1,y1-1);
                printf("%I64d\n",ans);
            }
            else
            {
                scanf("%I64d",&v);
                update(x1,y1);update(x1,y2+1);
                update(x2+1,y1);update(x2+1,y2+1);
            }
        }
    }
}
