#include<cstdio>
#include<cstdlib>
#include<algorithm>
#include<cstring>
using namespace std;
int i,j,type,x1,y1,x0,y0,n,m;
long long f[4][1100][1100],v;
void renew(int x,int y,long long v){
    int d=(x&1)*2+(y&1);
    for(int i=x;i<=n;i+=i&(-i))
        for(int j=y;j<=n;j+=j&(-j))f[d][i][j]^=v;
}
long long query(int x,int y){
    int d=(x&1)*2+(y&1);
    long long res=0;
    for(int i=x;i;i-=i&(-i))
        for(int j=y;j;j-=j&(-j))res^=f[d][i][j];
    return res;
}
int main(){
    scanf("%d%d",&n,&m);
    for(i=1;i<=m;++i){
        scanf("%d%d%d%d%d",&type,&x0,&y0,&x1,&y1);
        if(type==1)
            printf("%I64d\n",query(x1,y1)^query(x1,y0-1)^query(x0-1,y1)^query(x0-1,y0-1));
        else{
            scanf("%I64d",&v);
            renew(x0,y0,v);  renew(x0,y1+1,v);  renew(x1+1,y0,v);  renew(x1+1,y1+1,v);
        }
    }
}
