#include <iostream>
using namespace std;
long long d [2][2][1010][1010], ans, val;
void update (int x, int y) {for (int i = x; i < 1010; i += (i & (-i)))for (int j = y; j < 1010; j += (j & (-j)))d [x&1][y&1][i][j] ^= val;}
void query (int x, int y) {for (int i = x; i; i -= (i & (-i)))for (int j = y; j; j -= (j & (-j)))ans ^= d [x&1][y&1][i][j];}
int main () {
	int n, m, x1, x2, y1, y2, c;cin >> n >> m;
	for (int i = 0; i < m; i++) {
		cin >> c >> x1 >> y1 >> x2 >> y2, x1++, x2++, y1++, y2++;
		if (c == 2) {cin >> val;update (x1, y1), update (x2 + 1, y2 + 1), update (x1, y2 + 1), update (x2 + 1, y1);}
		else {ans = 0, query (x1 - 1, y1 - 1), query (x2, y2), query (x1 - 1, y2), query (x2, y1 - 1);cout << ans << endl;}
	}
}
