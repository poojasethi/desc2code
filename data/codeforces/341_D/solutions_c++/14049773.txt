#include <cstdio>
#include <algorithm>
using namespace std;
typedef long long ll;
const int N=1005;
int n,m,x0,y0,x1,y1;
ll v,t[2][2][N][N];
ll query(int x,int y){
	ll ans=0;
	for(int i=x;i;i-=i&-i)
		for(int j=y;j;j-=j&-j)
			ans^=t[x&1][y&1][i][j];
	return ans;
}
void modify(int x,int y){
	for(int i=x+1;i<=n;i+=i&-i)
		for(int j=y+1;j<=n;j+=j&-j)
			t[(x+1)&1][(y+1)&1][i][j]^=v;
}
int main(){
	scanf("%d%d",&n,&m);
	while(m--){
		int type; scanf("%d%d%d%d%d",&type,&x0,&y0,&x1,&y1);
		x0--; y0--;
		if(type==1)
			printf("%I64d\n",query(x0,y0)^query(x0,y1)^query(x1,y0)^query(x1,y1));
		else{
			scanf("%I64d",&v);
			modify(x0,y0);
			modify(x0,y1);
			modify(x1,y0);
			modify(x1,y1);
		}
	}
}
