#include<stdio.h>
#include<algorithm>
#include<math.h>

#include<string>

#include<set>
using namespace std;
typedef long long ll;
int n;
ll c[2][2][1003][1003];

ll query(int x,int y)
{
 ll ret=0;
 for (int i=x;i>0;i-=i&-i)
 for (int j=y;j>0;j-=j&-j)
 ret^=c[x&1][y&1][i][j];
 return ret;
}

void update(int x,int y,ll v)
{
   for (int i=x;i<=n;i+=i&-i)
   for (int j=y;j<=n;j+=j&-j)
   c[x&1][y&1][i][j]^=v;
}

int main()
{
   int m,x,y,xx,yy,op;
   ll v;
   scanf("%d%d",&n,&m);
   while (m--)
   {
     scanf("%d%d%d%d%d",&op,&x,&y,&xx,&yy);
     if (op==1)
     {
       x--;
       y--;
       printf("%I64d\n",query(x,y)^query(x,yy)^query(xx,y)^query(xx,yy));
     }
     else
     {
        scanf("%I64d",&v);
        xx++;
        yy++;
        update(x,y,v);
        update(xx,yy,v);
        update(x,yy,v);
        update(xx,y,v);
     }
   }
   return 0;
}
