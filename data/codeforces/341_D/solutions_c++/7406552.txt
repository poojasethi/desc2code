#include <cstdio>
int n,m;
long long f[4][1024][1024];
using namespace std;
int c(int x,int y){
    int res=0;
    if(x&1)res+=1;
    if(y&1)res+=2;
    return res;
}
long long getsum(int x,int y){
    long long res=0;
    int d=c(x,y);
    for(int i=x;i;i-=i&-i)
    for(int j=y;j;j-=j&-j)res^=f[d][i][j];
    return res;
}
void update(int x,int y,long long val){
    int d=c(x,y);
    for(int i=x;i<=n;i+=i&-i)
    for(int j=y;j<=n;j+=j&-j)f[d][i][j]^=val;
}
int main()
{
    scanf("%d%d",&n,&m);
    while(m--){
        int op,x0,y0,x1,y1;
        long long val;
        scanf("%d%d%d%d%d",&op,&x0,&y0,&x1,&y1);
        if(op==1){
            long long res=getsum(x1,y1);
            res^=getsum(x0-1,y1);
            res^=getsum(x1,y0-1);
            res^=getsum(x0-1,y0-1);
            printf("%I64d\n",res);
        }
        else {
            scanf("%I64d",&val);
            update(x0,y0,val);
            update(x0,y1+1,val);
            update(x1+1,y0,val);
            update(x1+1,y1+1,val);
        }
    }
    return 0;
}
