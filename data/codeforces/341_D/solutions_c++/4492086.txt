#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
#define rep(i,n) for (int i=0;i<n;++i)
typedef long long ll;
int n,m,x0,y0,x1,y1,z;ll v,a[4][1005][1005];
ll get(int t,int x,int y){ll res=0;for (int i=x;i;i-=i&-i)for (int j=y;j;j-=j&-j)res^=a[t][i][j];return res;}
void set(int t,int x,int y){for (int i=x;i<=n;i+=i&-i)for (int j=y;j<=n;j+=j&-j)a[t][i][j]^=v;}
ll query(int x,int y){return get((x&1)<<1|y&1,x,y);}
void modify(int x,int y){set((x&1)<<1|y&1,x,y);}
int main()
{
	scanf("%d%d",&n,&m);
	while (m--){
		scanf("%d%d%d%d%d",&z,&x0,&y0,&x1,&y1);
		if (z==1) x0--,y0--,printf("%I64d\n",query(x0,y0)^query(x0,y1)^query(x1,y0)^query(x1,y1));
		else scanf("%I64d",&v),x1++,y1++,modify(x0,y0),modify(x0,y1),modify(x1,y0),modify(x1,y1);
		}
	return 0;
}
