#include<cstdio>
#define count(arr) for(i=x;i;i-=i&-i)\
					for(j=y;j;j-=j&-j)\
						res^=arr[i][j]
#define modify(arr) for(i=x;i<=n;i+=i&-i)\
					for(j=y;j<=n;j+=j&-j)\
						arr[i][j]^=v
int i,j,n,m,x0,y0,x1,y1;
__int64 v,res,a[1001][1001],b[1001][1001],c[1001][1001],d[1001][1001];
inline void update(int x,int y)
{
	modify(a);
	if(y&1)modify(b);
	if(x&1)modify(c);
	if(x&1&&y&1)modify(d);
} 
inline __int64 query(int x,int y)
{
	res=0;
	if((x&1)==0&&(y&1)==0)count(a);
	if((x&1)==0)count(b);
	if((y&1)==0)count(c);
	count(d);
	return res;
}
int main()
{
	scanf("%d%d",&n,&m);
	while(m--){
		scanf("%d",&i);
		if(i==1){
			scanf("%d%d%d%d",&x0,&y0,&x1,&y1);
			printf("%I64d\n",query(x1,y1)^query(x1,y0-1)^query(x0-1,y1)^query(x0-1,y0-1));
		}
		else{
			scanf("%d%d%d%d%I64d",&x0,&y0,&x1,&y1,&v);
			if(x1<n&&y1<n)update(x1+1,y1+1);
			if(x1<n)update(x1+1,y0);
			if(y1<n)update(x0,y1+1);
			update(x0,y0);
		}
	}
	return 0;
}
