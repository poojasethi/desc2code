#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cstdlib>
#include<cstring>
#include<cmath>
#define lowbit(x) ((x)&(-(x)))
#define N 1005
using namespace std;
typedef long long LL;
LL a,b,c,d,e,f,g,h,i,j,k;
LL F[2][2][N][N];
LL x[2],y[2];
LL an;

LL Query(LL x,LL y)
{
    LL ii,jj,kk;
    LL R=0;
    for(ii=x;ii;ii-=lowbit(ii))
        for(jj=y;jj;jj-=lowbit(jj))
            R^=F[x&1][y&1][ii][jj];
    return R;
}
void Change(LL x,LL y,LL z)
{
    LL ii,jj,kk;
    for(ii=x;ii<=a;ii+=lowbit(ii))
        for(jj=y;jj<=a;jj+=lowbit(jj))
            F[x&1][y&1][ii][jj]^=z;
}

int main()
{
    cin>>a>>b;
    for(i=1;i<=b;i++)
    {
        cin>>c>>x[0]>>y[0]>>x[1]>>y[1];
        if(c==1)
        {
            an=Query(x[1],y[1])^Query(x[0]-1,y[1])^Query(x[1],y[0]-1)^Query(x[0]-1,y[0]-1);
            cout<< an <<endl;
        }
        else
        {
            cin>>d;
            Change(x[0],y[0],d);
            Change(x[1]+1,y[0],d);
            Change(x[0],y[1]+1,d);
            Change(x[1]+1,y[1]+1,d);
        }
    }   
    return 0;
}
