#!/usr/bin/env python

from __future__ import print_function
import sys

DEBUG = '-d' in sys.argv

def debug(*args, **kwargs):
    if DEBUG:
        print(*args, file=sys.stderr, **kwargs)

    return None


import math
# find max k with k**3 <= m < (k+1)**3
def find_k(a):
    k = 0
    while (k+1)**3 <= a:
        k += 1
    return k

class recursive(object):
    def __init__(self):
        self.best = (0, 0)

    def rec(self, m, steps, substracted):
        if m == 0:
            self.best = max(self.best, (steps, substracted))
        else:
            k = find_k(m)
            self.rec(m - k**3, steps+1, substracted + k**3)

            if m >= 1:
                self.rec(k**3 - 1 - (k-1)**3, steps+1, substracted + (k-1)**3)

result = recursive()
m = long(raw_input())
result.rec(m, 0, 0)
print(" ".join(map(str, result.best)))
