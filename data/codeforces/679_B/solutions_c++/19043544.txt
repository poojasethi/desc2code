#include <cstdio>
#include <algorithm>
#include <utility> 
using namespace std;
long long n;
pair<int,long long> dfs(long long n,int num,long long sum){
    if(n==0){return make_pair(num,sum);}
    long long t;for(t=1;t*t*t<=n;t++);t--;
    return max(dfs(n-t*t*t,num+1,sum+t*t*t),dfs(t*t*t-(t-1)*(t-1)*(t-1)-1,num+1,sum+(t-1)*(t-1)*(t-1)));
}
int main(){
    scanf("%I64d",&n);
	pair<int,long long>ans=dfs(n,0,0);
	printf("%d %I64d\n",ans.first,ans.second);
	return 0;
}
