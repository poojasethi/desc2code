#include<bits/stdc++.h>
#define ll long long 
#define mp make_pair
using namespace std;
pair<ll,ll> best;
ll cube(ll x){return x*x*x;}
void find(ll m,ll b,ll c)
{
	if(m==0){best=max(best,mp(b,c));return ;}
	ll x=1;
	while(cube(x+1)<=m)++x;
    find(m-cube(x),b+1,c+cube(x));
    find(cube(x)-1-cube(x-1),b+1,c+cube(x-1));
}
int main()
{
	ios::sync_with_stdio(false);
    ll m;
    cin>>m;
    find(m,0,0);
    cout<<best.first<<" "<<best.second;
	return 0;
}
