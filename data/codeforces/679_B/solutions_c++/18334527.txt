#include<bits/stdc++.h>
typedef long long LL;
#define TP(x) (x)*(x)*(x)

LL ans,res;

void wk(LL left,LL cnt,LL cur){
	if(left==0){
		if(cnt>ans){
			ans=cnt;
			res=cur;
		}
		return ;
	}
	LL x=1;
	while(TP(x+1)<=left){
		x++;
	}
	wk(left-TP(x),cnt+1,cur+TP(x));
	if(x>0)
	wk(TP(x)-1-TP(x-1),cnt+1,cur+TP(x-1));
}
int main(void) {
	LL n;
	scanf("%I64d",&n);
	ans=0;
	wk(n,0,0);
	printf("%I64d %I64d\n",ans,res);
}
