#include<bits/stdc++.h>
#define ll long long
using namespace std;
ll ans1,ans2;
void func(ll m, ll x, ll s)
{
    if(s>=ans2)
    {
        if(s>ans2)
            ans1 = x;
        else
            ans1 = max(x,ans1);
        ans2 = s;
    }
    if(m==0)
        return;
    ll a = cbrt(m);
    func(m-a*a*a,x+a*a*a,s+1);
    func(a*a*a-1-(a-1)*(a-1)*(a-1),x+(a-1)*(a-1)*(a-1),s+1);
}
int main()
{
    ll m;
    cin>>m;
    func(m,0,0);
    cout<<ans2<<" "<<ans1;
    return 0;
}
