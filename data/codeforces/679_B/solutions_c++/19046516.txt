#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
#include<cmath>
#include<queue>
using namespace std;
#define MAXN 100100
#define LL long long
LL ans,x,m;
void dfs(LL ret,int num,LL sum){
	if(ret==0){
		if(num>ans){
			ans=num;
			x=sum;
		}
		return;
	}
	LL i;
	for(i=1;(LL)i*i*i<=ret;++i);
	i--;
	dfs(ret-i*i*i,num+1,sum+i*i*i);
	dfs(i*i*i-1-(i-1)*(i-1)*(i-1),num+1,sum+(i-1)*(i-1)*(i-1));
}
int main(){
	while(cin>>m){
		ans=x=0;
		dfs(m,0,0);
		cout<<ans<<" "<<x<<endl;
	}
}
