#include<bits/stdc++.h>
using namespace std;
void work(long long now,int& d,long long& X)
{
	if(now<8ll)
	{
		d=X=now;
		return;
	}
	long long x=1;
	while(x*x*x<=now)
		++x;
	--x;
	int d1=0,d2=0;
	long long X1=0,X2=0;
	work(now-x*x*x,d1,X1);
	work(x*x*x-(x-1)*(x-1)*(x-1)-1,d2,X2);
	if(d1>=d2)
		d=d1+1,X=X1+x*x*x;
	else
		d=d2+1,X=(x-1)*(x-1)*(x-1)+X2;
}
int main()
{
	long long m;
	cin>>m;
	int maxd=0;
	long long X=0; 
	work(m,maxd,X);
	cout<<maxd<<' '<<X<<endl;
}
