#include <iostream>
using namespace std;
#define ll long long

ll m;
pair<ll,ll> best;

inline ll vol(int x) { return 1LL*x*x*x; }

void doit(ll m,ll bloc,ll h)
{
    if(m==0)
    {
        best=max(best,make_pair(bloc,h));
        return;
    }
    int x=1;
    while(vol(x+1)<=m) x++;
    doit(m-vol(x),bloc+1,h+vol(x));
    if(x-1>=0)
        doit(vol(x)-vol(x-1)-1,bloc+1,h+vol(x-1));
}

int main()
{
    cin>>m;
    doit(m,0,0);
    cout<<best.first<<' '<<best.second<<'\n';
    return 0;
}
