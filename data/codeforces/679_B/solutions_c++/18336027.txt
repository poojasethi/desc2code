#include<iostream>
#include<math.h>
using namespace std;
struct sajal{long long int v,n;};
sajal ans(long long int i){//cout<<i<<endl;
    if(i<=7){sajal r;r.v=i;r.n=i;return r;};

long long int h=cbrt(i);
long long int u=h-1;
if(u!=0){
    long long int p=h*h*h;
   sajal ans1=ans(i-p);ans1.n=ans1.n+1;ans1.v=ans1.v+p;
    long long int r=u*u*u;
  sajal ans2=ans(p-1-r);ans2.n=ans2.n+1;ans2.v=ans2.v+r;
    if(ans1.n>=ans2.n)return ans1;
    else return ans2;
}



}
int main(){
long long int m;cin>>m;

sajal k=ans(m);
cout<<k.n<<" "<<k.v<<endl;
}
