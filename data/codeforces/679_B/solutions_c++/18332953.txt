#include<stdio.h>
#include<iostream>
using namespace std;
typedef long long LL;
const int maxn=1E5+77;

LL V[maxn],M;

pair<int,LL> solve(LL v){
	if(v<=6) return make_pair(v,v);
	int p=lower_bound(V,V+maxn,v)-V;
	pair<int,LL> a,b;
	if(V[p]!=v) --p;
	a=solve(v-V[p]);
	++a.first;a.second+=V[p];
	b=solve(V[p]-1-V[p-1]);
	++b.first;b.second+=V[p-1];
	return max(a,b);
}

int main(){
	for(int i=1;i<maxn;++i) V[i]=1ll*i*i*i;
	cin>>M;
	pair<int,LL> ans=solve(M);
	cout<<ans.first<<" "<<ans.second<<endl;
}
