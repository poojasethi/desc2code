#include <iostream>
#include <stdio.h>
#define LL long long

using namespace std;

LL jpow(LL x)   
{
	return x*x*x;
}
pair<LL,LL> ans;    
void dfs(LL m,LL num, LL x)
{
	if(m==0){            
		ans = max(ans,make_pair(num,x));
		return ;
	}
	LL tpx = 1;
	while(jpow(tpx+1)<=m)  
		tpx++;
	dfs(m - jpow(tpx),num+1,x+jpow(tpx));  
	if(tpx - 1 >= 0)                
		dfs(jpow(tpx)-1-jpow(tpx - 1),num+1,x+jpow(tpx-1));
}
int main()
{
	LL m;
	scanf("%I64d",&m);
	dfs(m,0,0);
	printf("%I64d %I64d\n",ans.first,ans.second);
	return 0;
}
