#include<bits/stdc++.h>
#define ll long long
using namespace std;

pair<ll,ll>  ans;
ll c(ll x){
	return x*x*x;
}
void solve(ll m, ll size, ll vol){
	if(m == 0){
		ans = max(ans, make_pair(size,vol));
		return ;
	}
	ll x;
	for(x=100000; x && c(x)>m; x--){}
	solve(m-c(x), size+1, vol+c(x));
	solve(c(x)-1-c(x-1), size+1, vol+c(x-1));
}
int main(){
	ll m; cin>>m;
	solve(m, 0, 0);		
	cout<<ans.first<<" "<<ans.second<<endl;
	return 0;
}
