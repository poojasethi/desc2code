#include<bits/stdc++.h>
using namespace std;
typedef long long int des;
des po(des x)
{	return (x*x*x);
}
pair<des,des> best;
void go(des m,des steps,des su)
{	if(m==0)
	{	best=max(best,make_pair(steps,su));
		return;
	}
	des x=1;
	while(po(x+1)<=m)x++;
	go(m-po(x),steps+1,su+po(x));
	if(x-1>=0)
		go(po(x)-1-po(x-1),steps+1,su+po(x-1));
}
int main()
{	des m;
	scanf("%lld",&m);
	go(m,0,0);
	printf("%lld %lld\n",best.first,best.second);
	return 0;
}
