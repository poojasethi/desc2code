#include<bits/stdc++.h>

using namespace std;

typedef long long LL;
typedef pair <LL, LL> par;

par sol;

LL pow3(LL n)
{
    return n * n * n;
}

void rec(LL M, LL C, LL X)
{
	if(M == 0) {
		sol = max(sol, par(C, X));
		return;
	}
	LL p = 1;
	while(pow3(p + 1) <= M) p++;

	rec(M - pow3(p), C + 1, X + pow3(p));
	if(p - 1 >= 0)
		rec(pow3(p) - 1 - pow3(p - 1), C + 1, X + pow3(p - 1));
}

int main()
{
	LL N;
	cin >> N;

	rec(N, 0, 0);

	cout << sol.first << " " << sol.second << "\n";
	return 0;
}
