#include<cstdio>
#include<algorithm>
#define ll long long
#define V(x) ((x)*(x)*(x))
using namespace std;

ll n,m,MMH,mmh;
inline void dg(ll n,ll ans,ll m,ll v){
	if (n==0){
		if (ans>mmh||(ans==mmh&&v>MMH)) mmh=ans,MMH=v;
		return;
	}
	while(n<m*m*m) m--;
	dg(n-V(m),ans+1,m,v+V(m));
	if (m>1) dg(V(m)-V(m-1)-1,ans+1,m,v+V(m-1));
}
int main(){
	scanf("%I64d",&n);
	dg(n,0,1e5,0);
	printf("%I64d %I64d\n",mmh,MMH);
}
