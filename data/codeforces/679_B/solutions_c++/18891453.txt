#include <bits/stdc++.h>
using namespace std;

#define TRI_POW(x) (x)*(x)*(x) 
typedef long long LL;

LL ans = 0, res = 0;

void dfs(LL left, LL cnt, LL cur)
{
	if(left == 0)
	{
		if(cnt > ans)
		{
			ans = cnt;
			res = cur;
		}
		return;
	}
	LL x = 1;
	while(TRI_POW(x + 1) <= left)
		x++;
		
	dfs(left - TRI_POW(x), cnt + 1, cur + TRI_POW(x));
	
	dfs(TRI_POW(x) - 1 - TRI_POW(x-1), cnt + 1, cur + TRI_POW(x-1));
}
int main()
{
	LL a;
	cin>>a;
	dfs(a, 0, 0);
	cout<<ans<<" "<<res<<endl;
	return 0;
}
