#include<cstdio>
#include<iostream>
using namespace std;

long long m, ans = 0, ansv = 0;;

void work(long long x, long long cnt, long long cnv){
  if(x == 0){
    if(cnt > ans){
      ans = cnt;
      ansv = cnv;
    }
    return;
  }
  long long a = 0;
  while((a+1)*(a+1)*(a+1) <= x)a++;
  work(x-a*a*a, cnt+1, cnv + a*a*a);
  if(a > 1)work(a*a*a-1-(a-1)*(a-1)*(a-1), cnt+1, cnv + (a-1)*(a-1)*(a-1));
}

int main()
{
    cin>>m;
    work(m, 0, 0);
    cout<<ans<<" "<<ansv;
    return 0;
}
