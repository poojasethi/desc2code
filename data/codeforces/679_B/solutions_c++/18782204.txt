#include<cstdio>
#include<cstring>
#include<cstdlib>
#include<algorithm>

#define ll long long
#define inf 1e9
#define md
#define pr(x,y) make_pair(x,y)
using namespace std;
pair<int,ll> ans;
void dfs(ll x,int tot,ll sum)
{
	if (x<8) { tot+=x; sum+=x; ans=max(ans,pr(tot,sum)); return;}
	ll p=1; while (p*p*p<=x) p++; p--;
	dfs(x-p*p*p,tot+1,sum+p*p*p);
	dfs(p*p*p-1-(p-1)*(p-1)*(p-1),tot+1,sum+(p-1)*(p-1)*(p-1));
}
int main()
{
	ll m;
	scanf("%I64d",&m);
	ans=make_pair(0,1);
	dfs(m,0,0);
	printf("%d% I64d\n",ans.first,ans.second);
	return 0;
}
