# include <cstring>
# include <cstdio>
# include <cstdio>
# include <iostream>
# include <algorithm>
using namespace std;
# define N 250
int a[N];
int main (void)
{
    int n;
    cin>>n;
    for (int i=1;i<=n;i++)
        cin>>a[i];
    sort(a+1,a+1+n);
    int cnt1=0,cnt0=0;
    for (int i=1;i<=n;i++)
        if (a[i]>0) cout<<a[i]<<" ",cnt1++;
        else if (a[i]==0) cnt0++;
    int cnt=0;
    for (int i=1;i<=n;i++)
        if (a[i]<0) cnt++;
    for (int i=1;i<=cnt-(cnt&1);i++)
        cout<<a[i]<<" ";
    if (cnt0==n)
        cout<<0;
    if (cnt1==0 && cnt==1){
        if (cnt0)
            cout<<0;
        else cout<<a[1];
    }
    cout<<endl;
    return 0;
}
