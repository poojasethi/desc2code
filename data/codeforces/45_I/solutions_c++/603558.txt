#include<cstdio>
#include<algorithm>
int flag=0;
void out(int v){
    if(flag)putchar(' ');
    flag=1;
    printf("%d",v);
}
int main(){
    int pos[100],neg[100],zf,pc,nc,n,i,v;
    zf=0;pc=0;nc=0;
    scanf("%d",&n);
    for(i=0;i<n;i++){
        scanf("%d",&v);
        if(v==0)zf=1;
        else if(v>0)pos[pc++]=v;
        else if(v<0)neg[nc++]=v;
    }
    for(i=0;i<pc;i++)out(pos[i]);
    std::sort(neg,neg+nc);
    for(i=0;i<(nc-nc%2);i++)out(neg[i]);
    if(flag==0){
        if(zf)out(0);
        else out(neg[0]);
    }
    putchar('\n');
    return 0;
}
