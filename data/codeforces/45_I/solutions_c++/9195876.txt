#include <iostream>
#include <algorithm>
#define maxn 100
using namespace std;

int n, i, a[maxn], cnt, cnt_out, first;

int main()
{
	cin >> n;
	for(i = 0; i < n; i++) {
		cin >> a[i];
		cnt += a[i] < 0 ? 1 : 0;
	}
	if(n == 1) {
		cout << a[0] << endl;
		return 0;
	}
	sort(a, a + n);
	cnt_out = 0;
	first = 1;
	for(i = n - 1; i >= 0; i--) {
		if(a[i] < 0) break;
		if(a[i] == 0) continue;
		if(first) first = 0;
		else cout << " ";
		cout << a[i];
		cnt_out++;
	}
	if(cnt & 1) i--;
	for(; i>=0; i--) {
		if(first) first = 0;
		else cout << " ";
		cout << a[i];
		cnt_out++;
	}
	if(cnt_out == 0) cout << "0\n";
	else cout << "\n";
	return 0;
}
