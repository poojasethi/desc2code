#include<iostream>
#include<algorithm>
#include<vector>
using namespace std;
vector<int> v;
main()
{
	int n , c = 0 , a = -1;
	cin>>n;
	int cost[n];
	for(int i=0;i<n;i++)
		cin>>cost[i];
	sort(cost,cost+n);
	for(int i=0;i<n;i++)
	{
		if(cost [i] < 0)
		{
			c++;
			a=i;
		}
	}
	if(c%2==0){
		for(int i=0;i<n;i++)
			if(cost[i]!=0){
				v.push_back(cost[i]);
			}
	}
	else	
		for(int i=0;i<n;i++)
			if(i!=a && cost[i] != 0)
				v.push_back(cost[i]);
	if(v.size() == 0)
		cout<<cost[n-1];
	else
		for(int i = 0 ; i < v.size() ; i ++ )
			cout<<v[i]<<" ";
	return 0;
}
