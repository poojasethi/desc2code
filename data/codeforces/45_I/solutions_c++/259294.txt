#include <iostream>
#include <algorithm>
using namespace std;
int main()
{
  int a[105],n;
  cin>>n;
  for(int i=0;i<n;i++)cin>>a[i];
  if (n==1)
  {
    cout<<a[0];
    return 0;
  }
  sort(a,a+n);
  bool out=false;
  for(int i=0;i<n;i++)
    if (a[i]<0 && a[i+1]<0)
    {
      cout<<a[i]<<" "<<a[i+1]<<" ";
      i++;
      out=true;
    } else if (a[i]>0)
    {
      cout<<a[i]<<" ";
      out=true;
    }
  if (!out)cout<<0;
  return 0;
}
