#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

bool cmp(int a, int b){
  return a > b;
}

int main(){
  int n, x, fl = 0, in0 = 0;
  vector <int> v;
  cin >> n;
  for(int i = 0; i < n; i++){
    cin >> x;
    if(x > 0){
      fl = 1;
      cout << x << ' ';
    }else if(x < 0){
      v.push_back(x);
    }else{
      in0++;
    }
  }
  sort(v.begin(), v.end(), cmp);
  while(v.size() >= 2){
    cout << v.back() << ' ';
    v.pop_back();
    cout << v.back() << ' ';
    v.pop_back();
    fl = 1;
  }
  if(fl == 0){
    if(in0 > 0){
      cout << 0;
    }else{
      cout << v[0];
    }
  }
  cout << endl;
}
