#include <cstdio>
#include <cstring>
#include <vector>
#include <algorithm>
using namespace std;

int a[105];
vector<int> v, vp, vz;

int main(void) {
	int n;
	scanf("%d", &n);
	for(int i=0;i<n;i++) scanf("%d", &a[i]);
	for(int i=0;i<n;i++) if(a[i]==0) vz.push_back(0);
	else if(a[i]>0) v.push_back(a[i]);
	else if(a[i]<0) vp.push_back(a[i]);
	for(int i=0;i<(int)v.size();i++)
		printf("%d ", v[i]);
	sort(vp.begin(), vp.end());
	for(int i=1;i<(int)vp.size();i+=2)
		printf("%d %d ", vp[i-1], vp[i]);
	if(v.size()==0 && vp.size()<2 && vz.size()>0) printf("0");
	else if(v.size()==0 && vz.size()==0 && vp.size()<2) printf("%d", *max_element(vp.begin(), vp.end()));
	puts("");
	return 0;
}
