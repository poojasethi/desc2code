#include <iostream>
#include <cstdio>
#include <vector>
#include <algorithm>
#include <climits>
using namespace std;

int N,C[105],hi;
bool use[105];
vector <int> S;

bool cmp(int a,int b){return C[a] < C[b];}

int main(){
	scanf("%d",&N);
	for(int i=0;i<N;++i){
		scanf("%d",&C[i]);
		if(C[i] > 0) use[i] = 1;
		if(C[i] < 0) S.push_back(i); 
	}
	sort(S.begin(),S.end(),cmp);
	for(int i=0;i<S.size()/2*2;++i) use[S[i]] = 1;
	for(int i=0;i<N;++i) hi += use[i];
	if(hi == 0){
		hi = INT_MIN;
		for(int i=0;i<N;++i) hi = max(hi,C[i]);
		printf("%d\n",hi);
	}
	else{
		for(int i=0;i<N;++i) if(use[i]) printf("%d ",C[i]);
		printf("\n");
	}
//	system("pause");
}
