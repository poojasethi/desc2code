#include <vector>
#include <iostream>
using namespace std;

int main() {
	int c[105], col,n ;
	vector<int> ans;
	cin >> n;
	for (int i=0;i<n;++i)
	{
		cin >> c[i];
		if (c[i] > 0)
			ans.push_back(c[i]);
		if (c[i] < 0)
			++col;
	}

	if (n == 1)
	{
		cout << c[0];
		return 0;
	}

	int pos = -1;
	if (col % 2 == 1)
	{
		int min = -1000;
		for (int i = 0; i < n; ++i)
			if (c[i] < 0 && c[i] > min)
				min = c[i], pos = i;
	}

	for (int i=0;i<n;++i)
	{
		if (c[i] < 0 && i != pos)
			ans.push_back(c[i]);
	}
	if (ans.size() == 0)
		for (int i=0;i<n;++i)
			if (c[i] == 0)
			{
				ans.push_back(c[i]);
				break;
			}
//	cout << ans.size() << endl;
	for (int i=0;i<ans.size();++i)
		cout << ans[i] << " ";

	return 0;
}
