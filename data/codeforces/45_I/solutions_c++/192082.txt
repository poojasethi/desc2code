#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int main()
{
	vector<int> mosbat,manfi;
	int sefr=0;
	int n;
	cin>>n;
	for(int i=0;i<n;i++)
	{
		int x;
		cin>>x;
		if(x>0)
			mosbat.push_back(x);
		if(x<0)
			manfi.push_back(x);
		else
			sefr++;
	}
	vector<int> prn;
	for(int i=0;i<mosbat.size();i++)
		prn.push_back(mosbat[i]);
	sort(&manfi[0],&manfi[0]+manfi.size());
	for(int i=0;i<manfi.size();i++)
		prn.push_back(manfi[i]);
	if(manfi.size()%2==1)
		prn.pop_back();
	if(prn.size()!=0)
	{
		for(int i=0;i<prn.size()-1;i++)
			cout<<prn[i]<<" ";
		cout<<prn[prn.size()-1]<<endl;
	}
	else if (sefr!=0)
	{
		cout<<"0"<<endl;
	}
	else
	{
		cout<<manfi[0]<<endl;
	}
	return 0;
}
