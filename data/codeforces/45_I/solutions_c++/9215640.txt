#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <algorithm>

using namespace std;

int vis[110],a[110];

int main()
{
    int n,flag=0;
    scanf("%d",&n);
    for(int i=1;i<=n;i++)
        scanf("%d",&a[i]);
    sort(a+1,a+1+n);
    for(int i=n;i>=1;i--)
    {
        if(a[i]<=0)
            break;
        vis[i]=1;
        flag=1;
    }
    for(int i=1;i+1<=n;i+=2)
    {
        if(a[i+1]<0)
        {
            vis[i]=1;
            vis[i+1]=1;
            flag=1;
        }
    }
    if(flag==0)
        vis[n]=1;
    for(int i=1;i<=n;i++)
        if(vis[i]==1)
            printf("%d ",a[i]);
    printf("\n");
    return 0;
}
