#include <iostream>
#include <algorithm>
using namespace std;
const int sz=105;
int a[sz];
int main()
{
	int n,ot=0,pol=0;
	cin>>n;
	bool nol=false;
	for(int i=0;i<n;i++)
	{
		cin>>a[i];
		if (a[i]==0)
			nol=true;
		if (a[i]>0)pol++;
		else if (a[i]<0)
			ot++;
	}
	sort(a,a+n);
	for(int i=0;i<ot/2*2;i++)
		cout<<a[i]<<" ";
	if (ot==1 && pol==0 && !nol)cout<<a[0];
	else if (ot<2 && pol==0 && nol)cout<<0;
	else 
	  for(int i=n-pol;i<n;i++)
		  cout<<a[i]<<" ";
	return 0;
}
