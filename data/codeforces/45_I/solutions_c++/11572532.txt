#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

int main()
{
    int i,n,x,p=0,k=0;
    vector <int> neg;
    cin >> n;
    for(i=0;i<n;i++)
    {
        cin >> x;
        if(x>0) {p=1; cout << x << " ";}
        else if (x<0) neg.push_back(x);
        else k=1;
    }
    //if(neg.size()==0) return 0;
    sort(neg.begin(),neg.end());
    if(p==0 && neg.size()<2 && k==1) cout << 0;
    else if(p==0 && neg.size()==1) cout << neg[0];
    else if (p==0 && neg.size()==0) cout << 0;
    if(neg.size()%2==1) neg.pop_back();

    for(i=0;i<neg.size();i++) cout << neg[i] << " ";

    return 0;
}
