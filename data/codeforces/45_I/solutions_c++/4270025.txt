#include <iostream>
#include <cstdio>
#include <algorithm>
using namespace std;
int n,i,a[101],j,vr;
int main()
{
cin>>n;
for (i=0;i<n;i++)
cin>>a[i];
sort(a,a+n);
for (i=0;i<n;i++)
{
if (a[i]<0)
	{
	if (a[i]*a[i+1]>=1)
		{cout<<a[i]<<" "<<a[i+1]<<" ";
		i++;
		vr=1;
		continue;}

	}
else {
    if (a[i]>=1)
    	{
    	cout<<a[i]<<" ";
    	vr=1;
    	continue;
    	}
	}
}
if (vr==0)
cout<<a[n-1];



return 0;
}
