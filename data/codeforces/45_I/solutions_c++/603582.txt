#include <cstdio>
#include <algorithm>
using namespace std;

int alst[110],blst[110],a=0,b=0,c=0;
int lst[110],s=0;
int main()
{
int n;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
    {
    int t;
    scanf("%d",&t);
    if(t>0)
        alst[a++]=t;
    else if(t<0)
        blst[b++]=t;
    else ++c;
    }
    sort(blst,blst+b);
    for(int i=0;i+1<b;i+=2)
    {
        alst[a++]=blst[i];
        alst[a++]=blst[i+1];
    }
    if(a==0 && c>0)
        puts("0");
    else if(a==0 && c==0 && b==1)
        printf("%d\n",blst[0]);
    else
    {
        for(int i=0;i<a;++i)
            printf("%d%c",alst[i],i==a-1?'\n':' ');
    }
}
