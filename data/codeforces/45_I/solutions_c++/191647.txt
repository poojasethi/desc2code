#include <cstdio>
#include <algorithm>
using namespace std;

int a[111], b[111], n, m;

int main()
{
	n = m = 0;
	int t; 	scanf("%d", &t);
	bool zero = false, out = false;
	for (int i = 0; i < t; ++i){
		int x; scanf("%d", &x);
		if (x > 0) a[n++] = x; else
		if (x < 0) b[m++] = -x; else
			zero = true;
	}
	sort(a, a + n); 
	sort(b, b+m);
	if (m % 2 == 1)
		for (int i = 1; i < m; i += 2){
			printf("-%d -%d ", b[i], b[i + 1]);
			out = true;
			}
	else 
		for (int i = 0; i < m; i += 2){
			printf("-%d -%d ", b[i], b[i + 1]);
		out = true;
		}
	for (int i = 0; i < n; ++i){ printf("%d ", a[i]); out = true;}
	if (!out){
		if (zero) puts("0"); else
			printf("-%d\n", b[0]);
		} else 
	puts("");
}
