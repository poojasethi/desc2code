#include <iostream>
#include <algorithm>
#include <cstdio>
using namespace std ;

int main() {
    int N ;
    int num[1000] ;
    int i ;
    int usd = 0 ;
    
    scanf("%d",&N ) ;
    for( i=0; i<N; i++ )
        scanf("%d",&num[i] ) ;
    
    sort( num, num+N ) ;
    
    i = 0 ;
    while( num[i]<0 && i<N ) {
        if( i+1>=N || num[i+1]>=0 ) {
            i ++ ;
            break ;
        }
        usd = 1 ;
        printf("%d %d ",num[i], num[i+1] ) ;
        i += 2 ;
    }
    while( i<N && num[i]==0 )  {
        i ++ ;
    }
    while( i<N ) {
        printf("%d ",num[i] ) ;
        usd = 1 ;
        i ++ ;
    }
    
    if( !usd ) 
        printf("%d",num[N-1] ) ;
    
    cout << endl ;
    
    return 0 ;
}
