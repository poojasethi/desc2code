#include <iostream>
#include <algorithm>

using namespace std;

int main()
{
    int n;
    int a[100];
    cin>>n;
    for(int i=0;i<n;i++)
        cin>>a[i];
    sort(a,a+n);
    bool q=false;
    for(int i=n-1;i>=0;i--)
        if(a[i]<=0)break;
        else{
        cout<<a[i]<<' ';
        q=true;
        }
    for(int i=0;i+1<n;i+=2)
    {
        if(a[i]<0&&a[i+1]<0)
            {
                cout<<a[i]<<' '<<a[i+1]<<' ';
                q=true;
            }
        else
            break;
    }
    if(!q)
        cout<<a[n-1];
    return 0;
}
