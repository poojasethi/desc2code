#include <cstdio>
#include <algorithm>
using namespace std;

int n;
int a[105],aa,b[105],bb,c,z;

int main()
{
	while (scanf("%d",&n)==1)
	{
		aa=bb=z=0;
		for (int i=0;i<n;i++)
		{
			scanf("%d",&c);
			if (c>0) a[aa++]=c;
			else if (c<0) b[bb++]=-c;
			else z++;
		}
		sort(b,b+bb);
		if (aa==0 && bb<=1)
		{
			if (z) printf("0");
			else printf("%d",-b[0]);
		}
		if (bb&1)
		{
			for (int i=0;i<aa;i++) printf("%d ",a[i]);
			for (int i=1;i<bb;i++) printf("%d ",-b[i]);
		}
		else
		{
			for (int i=0;i<aa;i++) printf("%d ",a[i]);
			for (int i=0;i<bb;i++) printf("%d ",-b[i]);
		}
		printf("\n");
	}
	return 0;
}
