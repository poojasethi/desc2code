#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

#define PB push_back

int main () {
	vector<int> pos,neg;
	int n,j;
	cin >> n;

	bool have_zero = false;
	for(int i = 0; i < n; i++) {
		cin >> j;
		if(j>0) pos.PB(j);
		else if(j<0) neg.PB(j);
		else have_zero = true;
	}
	vector<int> r;
	for(int i = 0; i < pos.size(); i++) r.PB(pos[i]);
	sort(neg.begin(),neg.end());
	for(int i = 0; i+1<neg.size(); i+=2) {
		r.PB(neg[i]);
		r.PB(neg[i+1]);
	}
	if(r.size()==0 && have_zero) {cout << 0 << endl; return 0; }
	if(r.size()==0 && !have_zero) {cout << neg[0] << endl; return 0; }
	for(int i = 0; i < r.size(); i++) {
		if(i) cout << " ";
		cout << r[i];
	}
	cout << endl;
	return 0;
}
