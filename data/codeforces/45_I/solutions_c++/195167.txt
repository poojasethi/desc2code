#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

vector<int> lo,hi;
int z;

void input()
{
	int N,x;
	cin >> N;
	while(N--)
	{
		cin >> x;
		if(x > 0)
			hi.push_back(x);
		else if(x < 0)
			lo.push_back(x);
		else
			z++;
	}
	sort(lo.begin(), lo.end());
}


int main()
{
	input();
	if(hi.size() == 0 && z == 0 && lo.size() == 1)
		cout << lo[0] << endl;
	else if(hi.size() == 0 && z > 0 && lo.size() <= 1)
		cout << 0 << endl;
	else
	{
		for(int i = 0; i < hi.size(); i++)
			cout << hi[i] << " ";
		for(int i = 0; i + 1 < lo.size(); i += 2)
			cout << lo[i] << " " << lo[i+1] << " ";
		cout << endl;
	}
}
