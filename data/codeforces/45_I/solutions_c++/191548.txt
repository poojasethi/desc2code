#include <algorithm>
#include <iostream>
#include <cstdio>
using namespace std;
int n, a[1000], kol = 0;
int main()
{
    cin >> n;
    for(int i = 0; i < n; i++)
        cin >> a[i];
    sort(a, a + n);
    for(int i = 0; i < n && a[i] < 0; i++)
        if(a[i+1] < 0) {
            kol = 1;
            cout << a[i] << ' ';
            i++;
            cout << a[i] << ' ';
        }
    for(int i = n - 1; i >= 0 && a[i] > 1; i--) {
        cout << a[i] << ' ';
        kol = 1;
    }
    if(!kol)cout << a[n-1];
    return 0;
}
