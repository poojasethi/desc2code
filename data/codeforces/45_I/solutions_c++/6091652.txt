#include <cstdio>
#include <iostream>
using namespace std;
int n, i, j, k, l;
int c[100200];
int main(){
  cin >> n; k = 0;
  for (i = 1; i <= n; i++){
    cin >> c[i];
    if (c[i] < 0) k++;
  }
  l = 0;
  if (k & 1){
    j = 0; c[0] = -10000;
    for (i = 1; i <= n; i++) if (c[i] < 0 && c[i] > c[j]) j = i;
    for (i = 1; i <= n; i++) if (i != j && c[i] != 0){
    printf("%d ", c[i]);
    l = 1;
      }
  } else{
    for (i = 1; i <= n; i++) if (c[i] != 0){
    printf("%d ", c[i]);
    l = 1;
      }
  }
  if (l == 0){
    for (i = 1; i <= n; i++) if (c[i] > c[1]) c[1] = c[i];
    printf("%d", c[1]);
  }
  return 0;
}
