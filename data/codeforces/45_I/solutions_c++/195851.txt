#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;

int main() {
    int n,t;
    cin>>n;
    vector<int> v(n);
    vector<int> ans,neg;
    bool isz = false;
    for(int i=0; i<n; ++i) {
        cin>>t;
        if(t > 0) ans.push_back(t);
        else if(t == 0) isz = true;
        else neg.push_back(t);
    }

    sort(neg.begin(), neg.end());
    for(int i=0; i+1<neg.size(); i+=2) {
        ans.push_back(neg[i]);
        ans.push_back(neg[i+1]);
    }

    if(ans.size() == 0) {
        if(isz) cout<<0<<endl;
        else cout<<neg[neg.size()-1]<<endl;
    }else{
        for(int i=0; i<ans.size()-1; ++i) {
            cout<<ans[i]<<" ";
        }
        cout<<ans[ans.size()-1]<<endl;
    }
}
