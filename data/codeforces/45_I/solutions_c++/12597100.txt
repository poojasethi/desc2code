#include<bits/stdc++.h>
#include <cstdio>
#include <algorithm>
#include <iostream>
using namespace std;
const int N = 105;
int a[N], b[N], ans[N];
int main() {
	int n, as = 0, bs = 0, size = 0, zero = 0;
	bool f = 0;
	scanf("%d", &n);
	for(int i = 0, x; i < n; i ++) 
        {
		scanf("%d", &x);
		if(x > 0) a[as++] = x;
		else if(x < 0) b[bs++] = x;
		else zero = 1;
	}
	sort(a, a+as);
	sort(b, b+bs);
	for(int i = 0; i < as; i ++) ans[size ++] = a[i];
	if(size == 0 && bs <= 1 && zero) ans[size++] = 0;
	
	if(bs == 1 && size == 0) ans[size++] = b[0];
	if(bs & 1) bs --;
	for(int i = 0; i < bs; i ++) ans[size++] = b[i];

	for(int i = 0; i < size; i ++) printf("%d ", ans[i]);
	puts("");
	return 0;
}
