#include <stdio.h>

int p[200], ans[200], t = 0;

int main()
{
	int n, z = 0, cnt = 0, min = 1000;
	scanf("%d", &n);
	for (int i = 0; i < n; i++)
	{
		scanf("%d", &p[i]);
		if (p[i] == 0) z++;
		if (p[i] < 0) 
		{
			cnt++;
			if ( (-p[i]) < min) min = -p[i];
		}
	}
	if (z == n)
	{
		printf ("0");
		return 0;
	}
	if (cnt % 2)
	{
		if (n == 1) 
		{
			printf ("%d", p[0]);
			return 0;
		}
		if (cnt == 1 && cnt + z == n) 
		{
			printf ("0");
			return 0;
		}
		for (int i = 0; i < n; i++)
		{
			if ((cnt % 2) && p[i] == -min)
			{
				cnt--;
			}
			else if (p[i]) printf ("%d ", p[i]);
		}
	}
	else 
	{
		for (int i = 0; i < n; i++)
		{
			if (p[i]) printf ("%d ", p[i]);
		}
	}
	return 0;
}
