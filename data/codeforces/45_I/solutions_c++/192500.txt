#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
	int n,a[100];
	cin>>n;
	for(int i = 0; i < n; i++)
		cin>>a[i];
	sort(a,a + n);
	int zero = 0,otr = 0,pol = 0;
	for(int i = 0; i < n; i++)
		if (a[i] == 0)
			zero++;
		else
		if (a[i] > 0)
			pol++;
		else
		if (a[i] < 0)
			otr++;
	if (pol == 0 && otr == 0)
	{
		cout<<0;
		return 0;
	}
	if (pol == 0 && otr == 1 &&zero > 0)
	{
		cout<<0;
		return 0;
	}
	if (pol == 0 && otr == 1 && zero == 0)
	{
		cout<<a[0];
		return 0;
	}
	for(int i = 0; i < n - 1; i += 2)
		if (a[i] < 0 && a[i + 1] < 0)
			cout<<a[i]<<" "<<a[i + 1]<<" ";
     for(int i = 0; i < n; i++)
     	if (a[i] > 0) cout<<a[i]<<" ";		
	return 0;
}
