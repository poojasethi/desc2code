#include <iostream>
#include <fstream>
#include <algorithm>
#include <vector>

using namespace std;

vector <int> res;

int a[1000];
int n;

int main()
{
	cin >> n;
	int cnt = 0;
	for (int i = 0; i < n; i ++)
	{
		cin >> a[i];
		if (a[i] > 0)
		{
			cout << a[i] << " ";
			cnt ++;
		}
		if (a[i] < 0)
			res.push_back(a[i]);
	}
	sort(a,a + n);
	sort(res.begin(),res.end());
	for (int i = 0; i < res.size() - (res.size() % 2 == 1); i ++)
	{
		cout << res[i] << " ";
		cnt ++;
	}
	if (cnt == 0)
		cout << a[n - 1] << " ";
	cout << endl;
	return 0;
}
