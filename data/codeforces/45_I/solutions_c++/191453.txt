#include <stdio.h>

int n, nr, v[300];

int main ()
{
	#ifndef ONLINE_JUDGE
	freopen ("i.in", "r", stdin);
	freopen ("i.out", "w", stdout);
	#endif
	
	scanf ("%d", &n);
	
	int i, max = -300, nr0 = 0;
	
	for (i = 1; i <= n; i ++)
	{
		scanf ("%d", &v[i]);
		if (v[i] < 0)
		{
			nr ++;
			if (v[i] > max)
				max = v[i];
		}
		if (v[i] == 0)
			nr0 ++;
	}
	if (n == 1)
	{
		printf ("%d\n", v[1]);
		return 0;
	}
	if (nr0 == n || (nr0 == n - 1 && nr == 1))
	{
		printf ("0\n");
		return 0;
	}
	for (i = 1; i <= n; i ++)
	{
		if (v[i] == 0)
			continue;
		if (nr % 2 && v[i] == max)
		{
			nr --;
			continue;
		}
		printf ("%d ", v[i]);
	}
	return 0;
}
