#define MAXN 10000

#include <iostream>
#include <cstdio>
#include <cstring>
#include <algorithm>

using namespace std;

int n, a[MAXN];

int main(){
	/*freopen("input.txt","r",stdin);
	freopen("output.txt","w",stdout);*/
	scanf("%d", &n);
	for (int i=0; i<n; ++i)
		scanf("%d", &a[i]);
	sort(a, a+n);
	int k=0;
	for (int i=n-1; i>-1 && a[i]>0; i--, k++)
		printf("%d ", a[i]);
	
	for (int i=0; i<n-1 && a[i]<0 && a[i+1]<0; i+=2, k++)
		printf("%d %d ", a[i], a[i+1]);
	if (k==0) printf("%d", a[n-1]);
	return 0;
}
