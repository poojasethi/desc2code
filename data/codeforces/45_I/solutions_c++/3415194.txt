#include <cstdio>
#include <cstring>
#include <cstdlib>
#include <vector>
#include <algorithm>
using namespace std;

const int maxn	=	105;

int n,a[maxn];

int main()
{
	scanf("%d",&n);
	int zero=0;
	vector<int> pos,neg;
	for (int i=0;i<n;++i){
		scanf("%d",&a[i]);
		if (a[i]>0) pos.push_back(a[i]);
		else if (a[i]==0) ++zero;
		else if (a[i]<0) neg.push_back(a[i]);
	}
	sort(neg.begin(),neg.end());
	if (pos.size()>0 || neg.size()>=2){
		for (int i=0;i<n;++i){
			if (a[i]>0) printf("%d ",a[i]);
		}
		for (int i=0;i+1<neg.size();i+=2){
			printf("%d %d ",neg[i],neg[i+1]);
		}
		puts("");
		return 0;
	}
	if (zero){
		puts("0");
		return 0;
	}
	printf("%d\n",neg[0]);
	
	return 0;
}
