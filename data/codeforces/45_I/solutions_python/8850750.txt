c=map(int,raw_input().split())
c.sort()
neg=0
out=[]
nul=False
for i in c:
    if i<0:
        neg+=1
if n>neg:
    neg=neg-neg%2
for i in c:
    if i>0:
        out.append(i)
    elif i<0:
        if neg>0:
            out.append(i)
            neg-=1
    elif i==0:
        if n==1:
            out.append(i)
        else:
            nul=True
if len(out)==0 and nul:
