costs = map(int, raw_input().split())
costs.sort()
num_negs = len([x for x in costs if x < 0])
if num_negs % 2 == 1 and len(costs) > 1:
	for i in xrange(len(costs)):
		if costs[i] >= 0:
			costs.pop(i-1)
			break
	else:
		costs.pop(len(costs)-1)
for n in xrange(costs.count(0)):
	if len(costs) > 1:
		costs.remove(0)
print " ".join(map(str, costs))
