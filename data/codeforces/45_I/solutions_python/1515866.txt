n = input()
c = map(int, raw_input().split())
minus = []
for i in range(n):
    if c[i] < 0:
        minus.append(c[i])
res = []
minus = sorted(minus)
if len(minus) % 2 == 0:
    res += minus
else:
    res += minus[0:len(minus)-1]
for i in range(n):
    if c[i] > 0:
        res.append(c[i])
if len(res) == 0:
    res.append(max(c))
for i in range(len(res)):
    print str(res[i]) + "",
