#!/usr/bin/env python
# -*- coding: utf-8 -*-

def main():
    raw_input()
    p, n, z = list(), list(), False
    for i in map(int, raw_input().split()):
        if i > 0:
            p.append(i)
        elif i < 0:
            n.append(i)
        else:
            z = True
    n.sort()
    if len(p) == 0 and z is True and len(n) < 2:
        print 0
    else:
        if len(p) != 0:
            print " ".join(map(str, p)), 
        print " ".join(map(str, n if len(n)%2 == 0 or (len(n) == 1 and len(p) == 0) else n[:-1]))
    return 0

if __name__ == '__main__':
    main()
