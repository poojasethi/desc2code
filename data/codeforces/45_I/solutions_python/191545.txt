import sys
n = int (sys.stdin.readline ())
a = [int (x) for x in sys.stdin.readline ().split ()]
assert len (a) == n

res = a[-1]
cur = 1
b = []
while len (a) >= 1:
	temp = cur * a[-1]
	if cur >= temp:
		break
	cur = temp
	b += a[-1:]
	a = a[:-1]
while len (a) >= 2:
	temp = cur * a[0] * a[1]
	if cur > temp:
		break
	cur = temp
	b += a[:2]
	a = a[2:]

	print res
