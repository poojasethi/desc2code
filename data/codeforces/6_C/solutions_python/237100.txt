#!/usr/bin/env python
import string

n = int(raw_input())
times = map(int, string.split(raw_input()))

sum = 0
for time in times:
	sum += time

half = sum / 2.0

aliceCounter = 0
aliceTime = 0.0
for time in times:
	if (aliceTime + time) <= half:
		aliceCounter += 1
		aliceTime += time
	else:
		break

bobCounter = 0
bobTime = 0.0
for i in range(len(times)):
	time = times[n - i - 1]
	if (bobTime + time) <= half:
		bobCounter += 1
		bobTime += time
	else:
		break

if (aliceCounter + bobCounter) < n:
	if aliceTime == bobTime:
		aliceCounter += 1
	elif aliceTime > bobTime:
		bobCounter += 1
	elif aliceTime < bobTime:
		aliceCounter += 1

print str(aliceCounter) + ' ' + str(bobCounter)
