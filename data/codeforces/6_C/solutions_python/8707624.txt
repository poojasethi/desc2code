n = input()
a = list(map(int, raw_input().split()))

i = 0
j = n-1
m = a[i]
k = a[j]


if n == 1:
    print 1, 0
elif n == 2:
    print 1, 1
else:
    while j-i != 1:
        if (m <= k):
            i += 1
            m += a[i]
        elif (m > k):
            j -= 1
            k += a[j]
    print i+1, n - j
