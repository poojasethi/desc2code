n = int(raw_input())
v = map(int, raw_input().split())

a, b, t, s = [0 for i in range(n)], [0 for i in range(n)], 0, 0

for i in range(n):
   a[i] = t = t + v[i]
   b[i] = s = s + v[n - i - 1]
   
i, j = 0, 0

while(i + j < n):
   if(a[i] < b[j]):
      i += 1
   elif(b[j] < a[i]):
      j += 1
   elif(b[j] == a[i] and i + j == n - 1):
      i += 1
   else:
      i += 1
      j += 1
print i, j
