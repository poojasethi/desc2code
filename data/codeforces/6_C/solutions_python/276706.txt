n, t = input (), map (int, raw_input ().split ())
rt, lt = sum (t), 0
for lc, i in enumerate (t):
    rt -= i
    if abs (rt - lt) <= i:
        if rt < lt:
            print lc, n - lc
        else:
            print lc + 1, n - lc - 1
        break
    else:
        lt += i
