import sys
from itertools import product

file = sys.stdin
#file = open("test", "r")
cnt = int(file.readline().rstrip())
p = map(int, file.readline().rstrip().split())
left = 0
right = 0
li = -1
ri = cnt

while ri - li > 1:
    if left <= right:
        li += 1
        left += p[li]
    else:
        ri -= 1
        right += p[ri]
    
if ri != 0:
    print str(li + 1), str(cnt - ri)
else:
    print "1", "0"
