N=int(raw_input().strip())
nums=map(int,raw_input().strip().split())
pt2=N-1
cnt1=0
pt1=0
cnt2=0
alice=True
left=0
while pt1 <= pt2:
    if alice:
        v=nums[pt1]
        pt1+=1
        cnt1+=1
    else:
        v=nums[pt2]
        pt2-=1
        cnt2+=1

    if left > v:
        left=left-v
    elif left == v and alice:
        left=0
    else:
        alice=not alice
        left=v-left
print cnt1,cnt2
