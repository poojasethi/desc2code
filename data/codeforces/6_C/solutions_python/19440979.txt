Bob,Alice=int(raw_input().strip())-1,0
arr=list(map(int,raw_input().strip().split()))
CB,CA,sumB,sumA=0,0,0,0
while Alice<Bob:
	if sumA<sumB:
		sumA+=arr[Alice]
		Alice+=1
		CA+=1
	elif sumA>sumB:
		sumB+=arr[Bob]
		Bob-=1
		CB+=1
	else:
		sumA+=arr[Alice]
		sumB+=arr[Bob]
		CA+=1
		CB+=1
		Alice+=1
		Bob-=1
if Alice==Bob:
	if sumB<sumA:
		CB+=1
	else:
		CA+=1
print CA,CB
