#!/usr/bin/env python

from sys import stdin

N = int(stdin.readline().strip())

T = map(int, stdin.readline().split())

alice_t, alice_idx = 0, 0
bob_t, bob_idx = 0, N-1

while alice_idx <= bob_idx:
   if alice_t == bob_t:
      if alice_idx == bob_idx:
         alice_t += T[alice_idx]
         alice_idx += 1
      else:
         alice_t += T[alice_idx]
         alice_idx += 1
         bob_t += T[bob_idx]
         bob_idx -= 1
      continue

   if alice_t < bob_t:
      alice_t += T[alice_idx]
      alice_idx += 1
   else:
      bob_t += T[bob_idx]
      bob_idx -= 1

print alice_idx, N - bob_idx - 1
