import sys

def main():
	n=int(raw_input())
	ar=map(int,raw_input().split(" ",n))
	i=0;j=n-1;
	sumr=ar[j];suml=ar[0];
	nosl=1
	nosr=1
	if n==1:
		print "1 0";
	else:
		while((i+1)!=j):
			if sumr<suml:
				j-=1;
				sumr+=ar[j]
				nosr+=1
			else:
				i+=1
				suml+=ar[i]
				nosl+=1
		print nosl,nosr;
main()
