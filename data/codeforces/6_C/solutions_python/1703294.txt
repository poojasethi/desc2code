#!/usr/bin/python
import sys

tyo = 0
tyo_t = 0
ali = 0
ali_t = 0

for c,s in enumerate(sys.stdin):
    if c == 0:
        ln = int(s)
    if c == 1:
        po = s.split()
        while ali + tyo < ln:
            if ali_t > tyo_t:
                tyo_t += int(po[-(1+tyo)])
                tyo += 1
            else:
                ali_t += int(po[ali])
                ali += 1

print str(ali) +' '+ str(tyo)
