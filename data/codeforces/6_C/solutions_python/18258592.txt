n=int(raw_input())
Bars=map(int, raw_input().split())
half=sum(Bars)//2
alice=0
count=0
while (alice+Bars[count]<=half):
    alice=alice+Bars[count]
    count=count+1
bob=sum(Bars[count+1::])
if (bob>=alice):
    count=count+1
out=[count,n-count]
out=[str(o) for o in out]
out=" ".join(out)
print(out)
