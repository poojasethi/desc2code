n = input()
arr = map(int, raw_input().split())

sl = [0] * n
sr = [0] * n
s = 0
for i in xrange(n):
    s += arr[i]
    sl[i] = s

s = 0
for i in xrange(n-1, -1, -1):
    s += arr[i]
    sr[i] = s

l, r = 0, n-1

ansl, ansr = 0, 0
while l+1 < r:
    t = min(sl[l], sr[r])

    if t == sl[l]:
        l += 1
    else:
        r -= 1

print l + 1, n - l - 1
