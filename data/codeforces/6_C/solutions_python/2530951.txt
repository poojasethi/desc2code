n = int(raw_input())
bars = map(int, raw_input().split())

alice = 0
bob = n-1
atime = 0
btime = 0

while alice <= bob:
        if atime <= btime:
                atime += bars[alice]
                alice += 1
        else:
                btime += bars[bob]
                bob -= 1

print alice, (n - 1 - bob)
