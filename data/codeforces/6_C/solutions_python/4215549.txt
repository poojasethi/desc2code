
def main():
	from sys import stdin
	lst = map(int,stdin.readlines()[1][:-1].split(' '))
	l = 0
	r = len(lst)-1
	a = 0
	b = 0
	flag = [False for i in xrange(len(lst))]
	while l <= r:
		if l == r:
			if not flag[l]:
				a += 1
			break
		if not flag[l]:
			flag[l] = True
			a += 1
		if not flag[r]:
			flag[r] = True
			b += 1
		c = min(lst[l],lst[r])
		lst[l] -= c
		lst[r] -= c
		if lst[l] == 0:
			l += 1
		if lst[r] == 0:
			r -= 1
	
	print a,b



if __name__ == '__main__':
	main()
