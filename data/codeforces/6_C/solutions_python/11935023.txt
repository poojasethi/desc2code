#!/usr/bin/env python
n = input()
a = map(int, raw_input().split())
i = 0
j = n-1
while i < j:
    if a[i] == a[j]:
        if i == j-1:
            break;
        elif i == j-2:
            i += 1
            break
        else:
            i += 1
            j -= 1
    elif a[i] < a[j]:
        a[j] -= a[i]
        if i+1 == j:
            break
        else:
            i += 1
    else:
        a[i] -= a[j]
        if i == j-1:
            break
        else:
            j -= 1
if n == 1:
    print 1, 0
else:
    print i+1, n-j
