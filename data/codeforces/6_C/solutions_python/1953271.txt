import sys                                                                      

n = int(sys.stdin.readline())
t = []
t = sys.stdin.readline().split(' ')
b, b_ch = 0, 0
a_ch = int(t[0])
a = 1

while a + b < n:
  if a_ch <= b_ch:
    a_ch += int(t[a])
    a += 1
  else:
    b_ch += int(t[n-b-1])
    b += 1

print a, b
