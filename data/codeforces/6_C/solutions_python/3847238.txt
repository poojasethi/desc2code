def main():
  m = int(raw_input())
  line = [int(x) for x in raw_input().split()]
  i, j = 0, m - 1
  itime, jtime = line[i], line[j]
  while i != j:
	if itime > jtime:
	  j -= 1
	  if j >= 0:
		jtime += line[j]
	elif i + 1 == j:
	  j -= 1
	  break
	elif i + 1 :
	  i += 1
	  if i < m:
		itime += line[i]
  print i + 1, m - j - 1

if __name__ == '__main__':
  main()
