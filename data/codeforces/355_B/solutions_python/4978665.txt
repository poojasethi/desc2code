def main():
  R = lambda:map(int,raw_input().split())

  c = R()
  n,m =R()
  a = R()
  b = R()
  
  bus = [-1]*n
  trolley = [-1]*m
  
  for i in range(n):
    bus[i] = min(a[i]*c[0],c[1])    
  for i in range(m):
    trolley[i] = min(b[i]*c[0],c[1])
    
  print min(min(sum(bus),c[2])+min(sum(trolley),c[2]),c[3]) 
  
main()
