c1,c2,c3,c4=map(int,raw_input().split(' '))
n,m=map(int,raw_input().split(' '))
a=map(int,raw_input().split(' '))
b=map(int,raw_input().split(' '))
def solve(d):
	sum=0
	for i in d:
		if i*c1 <c2:
			sum+=i*c1
		else:
			sum+=c2
	if sum<c3:
		return sum
	else:
		return c3
ans=solve(a)+solve(b)
print min(ans,c4)
