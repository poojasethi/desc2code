import sys

c1, c2, c3, c4 = map(int, sys.stdin.readline().split())
n, m = map(int, sys.stdin.readline().split())


a = map(int, sys.stdin.readline().split())
b = map(int, sys.stdin.readline().split())

bus = 0
trolley = 0

for i in a:
    if i > 0:
        bus += min(i * c1, c2)
for i in b:
    if i > 0:
        trolley += min(i * c1, c2)

ans = bus + trolley
ans = min(ans, bus + c3)
ans = min(ans, trolley + c3)
ans = min(ans, c3 + c3)
ans = min(ans, c4)
print ans
