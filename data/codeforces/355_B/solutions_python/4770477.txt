c1,c2,c3,c4 = map(int,raw_input().split())
n,m = map(int,raw_input().split())
a = map(int,raw_input().split())
b = map(int,raw_input().split())

pa = sum(a)
pb = sum(b)

a1 = ( pa + pb ) * c1

a2,a21,a22 = 0,0,0
for i in range(n):
	a21 += min(a[i]*c1,c2)
for i in range(m):
	a22 += min(b[i]*c1,c2)
a2 = a21 + a22

a3 = min(2*c3 , c3 + a21,c3 + a22)

a4 = c4

print min(a1,a2,a3,a4)
