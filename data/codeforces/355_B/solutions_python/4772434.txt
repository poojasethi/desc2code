c=map(int,raw_input().split(" "))
ans1=min(2*c[2],c[3])
n,m=map(int,raw_input().split(" "))
arb=map(int,raw_input().split(" "))
art=map(int,raw_input().split(" "))
ans=0
for i in arb:
	if(c[0]*i)<c[1]:
		ans+=c[0]*i
	else:
		ans+=c[1]
if ans>c[2]:
	ans=c[2]
ans2=0
for i in art:
	if(c[0]*i)<c[1]:
		ans2+=c[0]*i
	else:
		ans2+=c[1]
if(ans2>c[2]):
	ans2=c[2]
print min(ans+ans2,ans1)
