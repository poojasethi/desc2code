import sys

c1,c2,c3,c4=map(int,raw_input().split())
n,m=(map(int,raw_input().split()))
a=(map(int,raw_input().split()))
b=(map(int,raw_input().split()))

def f(x, k):
    res = 0
    for i in range (k):
        res += min(c2, x[i] * c1)
    return min(c3, res);

ans = min(c4, f(a, n) + f(b, m));
        
print ans
