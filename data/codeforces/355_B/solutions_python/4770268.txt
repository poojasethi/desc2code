c1,c2,c3,c4 = map(int,raw_input().split())
n,m =  map(int,raw_input().split())
a =  map(int,raw_input().split())
b =  map(int,raw_input().split())

auto = 0
for i in xrange(n):
	auto += min(a[i]*c1, c2)
auto = min(auto, c3)

trol= 0
for i in xrange(m):
	trol += min(b[i]*c1, c2)
trol = min(trol, c3)

print min(auto+trol,c4)
