def inp():
	b = raw_input()
	b = b.split(' ')
	a =[]
	for i in b:

def cost(x,y):
	tot =0
	for i in x:
		tot += i*y[0] if i*y[0]< y[1] else y[1]
	return tot if tot < y[2] else y[2]
