import sys

c1, c2, c3, c4 = (int(x) for x in sys.stdin.readline().split())
sys.stdin.readline()
b_count = [int(x) for x in sys.stdin.readline().split()]
t_count = [int(x) for x in sys.stdin.readline().split()]

b_sum = sum([min(c2, c1 * n) for n in b_count])
t_sum = sum([min(c2, c1 * n) for n in t_count])

print min(c4, min(c3, b_sum, c3) + min(c3, t_sum))
