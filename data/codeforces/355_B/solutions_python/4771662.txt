#!/usr/bin/env python2

def main():
	c1, c2, c3, c4 = map(int, raw_input().split())
	n, m = map(int, raw_input().split())
	a = map(int, raw_input().split())
	b = map(int, raw_input().split())
	a_sum = 0
	for x in a:
		if c1 * x < c2:
			a_sum += c1 * x
		else:
			a_sum += c2
	if a_sum > c3:
		a_sum = c3
	b_sum = 0
	for y in b:
		if c1 * y < c2:
			b_sum += c1 * y
		else:
			b_sum += c2
	if b_sum > c3:
		b_sum = c3
	if a_sum + b_sum > c4:
		print c4
	else:
		print a_sum + b_sum
		
if __name__ == '__main__':
	main()
