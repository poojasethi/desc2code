if __name__=="__main__":
    c1, c2, c3, c4 = map(int,raw_input().split())
    n, m = map(int,raw_input().split())
    a = map(int, raw_input().split())
    b = map(int, raw_input().split())

    suma = 0
    for i in range(n) :
      suma += min(a[i]*c1, c2)

    sumb = 0
    for j in range(m) :
      sumb += min(b[j]*c1, c2)

    suma = min(suma, c3)

    sumb = min(sumb, c3)

    print min(suma+sumb, c4)
