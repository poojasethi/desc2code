def main():
	c1, c2, c3, c4 = map(int, raw_input().split())
	n, m = map(int, raw_input().split())
	a = map(int, raw_input().split())
	b = map(int, raw_input().split())
	ans1, ans2, bus, tro = 0, 0, 0, 0
	for x in a: 
		if x != 0:
			bus = 1
			ans1 += min(c1 * x, c2)
	for x in b: 
		if x != 0:
			tro = 1
			ans2 += min(c1 * x, c2)			
	ans = min(ans1, bus * c3) + min(ans2, tro * c3)
	ans = min(ans, c4)
	print ans
	
if __name__ == "__main__": main()
