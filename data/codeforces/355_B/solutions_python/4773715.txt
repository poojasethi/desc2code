c1, c2, c3, c4 = map(int, raw_input().split())
n, m = map(int, raw_input().split())
b = map(int, raw_input().split())
t = map(int, raw_input().split())
a = 0
for i in range(n):
    if c1 * b[i] < c2:
        a += c1 * b[i]
    else:
        a += c2
if a > c3:
    a = c3
a1 = 0
for i in range(m):
    if c1 * t[i] < c2:
        a1 += c1 * t[i]
    else:
        a1 += c2
if a1 > c3:
    a1 = c3
if a1 + a < c4:
    print a1 + a
else:
    print c4
