def main():
    c1, c2, c3, c4 = map(int, raw_input().split())
    n, m = map(int, raw_input().split())
    As = map(int, raw_input().split())
    Bs = map(int, raw_input().split())

    sa = 0
    for i in range(n):
        sa += min(As[i] * c1, c2)
    sb = 0
    for j in range(m):
        sb += min(Bs[j] * c1, c2)

    sa = min(sa, c3)
    sb = min(sb, c3)

    s = min(sa + sb, c4)

    print s


if __name__ == "__main__":
    main()
