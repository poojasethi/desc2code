c = [int(_) for _ in raw_input().split()]

n, m = [int(_) for _ in raw_input().split()]

a = [int(_) for _ in raw_input().split()]
b = [int(_) for _ in raw_input().split()]


def f(x):
    res = 0
    for xi in x:
        res = res + min(c[1], xi*c[0])

    return min(c[2], res)
