#!/usr/bin/env python

bus_or_tram, unlim_bus_or_tram, unlim_all_buses_or_trams, unlim = map(int, raw_input().split())
n, m = map(int, raw_input().split())
buses = map(int, raw_input().split())
trams = map(int, raw_input().split())

ans = unlim
# Buses
ans_buses = min(unlim_all_buses_or_trams,
                sum(min(unlim_bus_or_tram, bus * bus_or_tram) for bus in buses))
# Trams
ans_trams = min(unlim_all_buses_or_trams,
                sum(min(unlim_bus_or_tram, tram * bus_or_tram) for tram in trams))

if ans_buses + ans_trams < ans:
    ans = ans_buses + ans_trams
print ans
