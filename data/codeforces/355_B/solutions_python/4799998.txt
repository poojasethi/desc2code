
import sys

lines = sys.stdin.readlines()
(c1, c2, c3, c4) = [int(i) for i in lines[0].split()]
(n, m) = [int(i) for i in lines[1].split()]
a = [int(i) for i in lines[2].split()]
assert len(a) == n
b = [int(i) for i in lines[3].split()]
assert len(b) == m

min_cost_a = min(c3, sum([min(c2,i*c1) for i in a]))
min_cost_b = min(c3, sum([min(c2,i*c1) for i in b]))
min_cost = min(c4, min_cost_a+min_cost_b)
print min_cost
