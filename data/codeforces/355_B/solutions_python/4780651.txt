c1, c2, c3, c4 = map(int, raw_input().split())
raw_input()
buses = map(int, raw_input().split())
trolleys = map(int, raw_input().split())

def minPerBusOrTrolley(rides):
    res = 0
    for x in rides:
        res += min(c2, c1*x)
    return res


res = min(c4, min(c3, minPerBusOrTrolley(buses)) + min(c3, minPerBusOrTrolley(trolleys)))

print res
