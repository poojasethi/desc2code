import sys

c1,c2,c3,c4 = map(int, raw_input().split())
n, m = map(int, raw_input().split())
buses = map(int, raw_input().split())
trolleys = map(int, raw_input().split())

bus_total = 0
trolley_total = 0

for b in buses:
  b_v = min(b*c1, c2)
  bus_total += b_v

for t in trolleys:
  t_v = min(t*c1, c2)
  trolley_total += t_v

if bus_total > c3:
  bus_total = c3
if trolley_total > c3:
  trolley_total = c3

if c4 <= bus_total + trolley_total:
  print c4
else:
  print bus_total + trolley_total
