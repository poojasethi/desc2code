#include<iostream>
#include<algorithm>
#include<string>
using namespace std;
int main(){
	int a,b,c,d,s=0,t=0,p=0;
	cin>>a>>b>>c>>d;
	int n,m;
	cin>>n>>m;
	int x[1001],y[1001];
	for(int i=0;i<n;++i){
		cin>>x[i];
		s+=min(x[i]*a,b);}
	s=min(s,c);
	for(int i=0;i<m;++i){
		cin>>y[i];
		t+=min(y[i]*a,b);}
	t=min(t,c);
	cout<<min(s+t,d);
	return 0;
}
