#include<bits/stdc++.h>
using namespace std;
int main(){
    int c1,c2,c3,c4,n,m,vn[1003],vm[1003];
    cin>>c1>>c2>>c3>>c4>>n>>m;
    int sn=0,sm=0;
    for(int i=0;i<n;i++){
      cin>>vn[i];
        sn+=min(vn[i]*c1,c2);
    }
    sn=min(sn,c3);
     for(int i=0;i<m;i++){
     cin>>vm[i];
        sm+=min(vm[i]*c1,c2);
    }
    sm=min(sm,c3);
    cout<<min(sn+sm,c4)<<endl;
   return 0;
}
