#include <iostream>
#include <cstdio>
#include <vector>
#include <utility>
#include <algorithm>

using namespace std;

int main()
{
	int c1,c2,c3,c4,m,n,x,sb=0,st=0,i;
	cin>>c1>>c2>>c3>>c4>>m>>n;
	for(i=0;i<m;i++)
	{
		cin>>x;
		sb+=min(c2,x*c1);
	}
	for(i=0;i<n;i++)
	{
		cin>>x;
		st+=min(x*c1,c2);
	}
	cout<<min(min(min(c4,2*c3),min(sb,st)+c3),sb+st)<<endl;
	return 0;

}
