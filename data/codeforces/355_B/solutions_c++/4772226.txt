#include <iostream>
#include <vector>
using namespace std;

int main() {
	int c1,c2,c3,c4,n,m,i,ans;
	cin>>c1>>c2>>c3>>c4>>n>>m;
	vector<int> b(n);
	vector<int> t(m);
	for(i=0;i<n;i++)
		cin>>b[i];
	for(i=0;i<m;i++)
		cin>>t[i];
	int t_b=0,t_t=0;
	for(i=0;i<n;i++) 
		t_b+=min(c2,b[i]*c1);
	for(i=0;i<m;i++)
		t_t+=min(c2,t[i]*c1);
	t_b=min(t_b,c3);
	t_t=min(t_t,c3);
	ans=min(t_b+t_t,c4);
	cout<<ans<<endl;
	return 0;
}
