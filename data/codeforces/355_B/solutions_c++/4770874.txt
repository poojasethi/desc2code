#include<iostream>
using namespace std;

int main(){
    int e,a,An,x,Q,minnA=0,minnT=0,A,T;
    cin>>a>>An>>x>>Q;
    cin>>A>>T;
    for(int i=0;i<A;++i){
        cin>>e;
        minnA+=min(e*a,An);
    }
    minnA=min(minnA,x);
    for(int i=0;i<T;++i){
        cin>>e;
        minnT+=min(e*a,An);
    } 
    minnT=min(minnT,x); 
    cout<<min(Q,minnA+minnT);
}
