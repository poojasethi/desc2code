#include <iostream>
#include<algorithm>
using namespace std;
int c1,c2,c3,c4,n,m,i,x,a,t,sol;
int main()
{
    cin>>c1>>c2>>c3>>c4>>n>>m;
    for(i=1;i<=n;i++)
    {
        cin>>x;
        a+=min(c2,x*c1);
    }
    for(i=1;i<=m;i++)
    {
        cin>>x;
        t+=min(c2,x*c1);
    }
    sol=min(min(min(c4,2*c3),a+c3),min(t+c3,a+t));
    cout<<sol;
    return 0;
}
