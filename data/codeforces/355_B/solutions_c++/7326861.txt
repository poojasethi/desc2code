#include<iostream>
using namespace std;
int main()
{
	int a,b,c,d,n,m,ansn=0,ansm=0;
	cin>>a>>b>>c>>d>>n>>m;
	for(int i=0;i<n;i++)
	{
		int sum;
		cin>>sum;
		ansn+=min(sum*a,b);
	}
	for(int i=0;i<m;i++)
	{
		int sum;
		cin>>sum;
		ansm+=min(sum*a,b);
	}
	ansn=min(ansn,c),ansm=min(ansm,c);
	cout<<min(ansn+ansm,d);
	return 0;
}
