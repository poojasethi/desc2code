#include <cstdio>
#include <algorithm>
using namespace std;
int N,M,A,B,C,D,s1,s2;
int main()
{
	scanf("%d%d%d%d%d%d",&A,&B,&C,&D,&N,&M);
	for (int i=1,x; i<=N; i++) scanf("%d",&x),s1+=min(A*x,B);
	for (int i=1,x; i<=M; i++) scanf("%d",&x),s2+=min(A*x,B);
	s1=min(s1,C),s2=min(s2,C),printf("%d\n",min(s1+s2,D));
}
