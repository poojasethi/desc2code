#include <iostream>
using namespace std;

int main(){
    int c1, c2, c3, c4, x = 0, y = 0, m, n;
    cin >> c1 >> c2 >> c3 >> c4;
    cin >> m >> n;
    for(int i = 0; i<m; i++){
        int j;
        cin >> j;
        x+=min(j*c1, c2);
    }
    for(int i = 0; i<n; i++){
        int j;
        cin >> j;
        y+=min(j*c1, c2);
    }
    cout << min(min(x,c3)+min(y,c3), c4);
}
