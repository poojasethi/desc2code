#include <iostream>
#include <stdlib.h>
#include <stdio.h>
using namespace std;
int c1,c2,c3,c4;
int n,m,a,b,r;
int main() {
	cin >> c1 >> c2 >> c3 >> c4;
	cin >> n >> m;
	for (int i=0; i<n; i++) {
		cin >> r;
		a+=min(r*c1,c2);
	}
	a=min(a,c3);
	for (int i=0; i<m; i++) {
		cin >> r;
		b+=min(r*c1,c2);	
	}
	b=min(b,c3);
	cout << min(a+b,c4);
	return 0;
}
