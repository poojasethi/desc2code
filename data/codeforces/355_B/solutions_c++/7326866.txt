#include <iostream>
using namespace std;

int c1, c2, c3, c4, n, m, a, b, A, B, i;
int main()
{
	cin>>c1>>c2>>c3>>c4>>n>>m;
	for (i=0; i<n; i++) { cin>>a; a=min(c2, a*c1); A+=a; } A=min(A, c3);
	for (i=0; i<m; i++) { cin>>b; b=min(c2, b*c1); B+=b; } B=min(B, c3);
	cout<<min(c4, A+B)<<endl;
	return 0;
}
