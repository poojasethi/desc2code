#include<algorithm>
#include<iostream>
#include<cmath>
using namespace std;

int main()
{
int c1,c2,c3,c4,t,b,*arr1,*arr2,sum1=0,sum2=0,ans;
cin>>c1>>c2>>c3>>c4>>b>>t;
arr1=new int [b]; arr2=new int[t];
for(int i=0;i<b;i++)
{cin>>arr1[i]; sum1+=min(arr1[i]*c1,c2);
}sum1=min(sum1,c3);
for(int i=0;i<t;i++)
{cin>>arr2[i];sum2+=min(arr2[i]*c1,c2);}
sum2=min(sum2,c3);
ans=min(sum1+sum2,c4);
cout<<ans;
}
