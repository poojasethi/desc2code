#include <iostream>
#include <string>
using namespace std;
int c1,c2,c3,c4,t1,t2,t3,n,m,a[1001],b[1001];
string s;
int main()
{
	cin>>c1>>c2>>c3>>c4>>n>>m;
	for (int i=0;i<n;i++)
		cin>>a[i];
	for (int i=0;i<m;i++)
		cin>>b[i];
	for (int i=0;i<n;i++)
		t1+=min(a[i]*c1,c2);
	t1=min(t1,c3);
	for (int i=0;i<m;i++)
		t2+=min(b[i]*c1,c2);
	t2=min(t2,c3);
	t3=min(t1+t2,c4);
	cout<<t3<<endl;
}
