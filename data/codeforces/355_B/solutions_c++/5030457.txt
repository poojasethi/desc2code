#include <iostream>

using namespace std;

int main()
{
	int c1, c2, c3, c4, n, m;
	int a=0, b=0, ans=0;
	cin>>c1>>c2>>c3>>c4>>n>>m;
	for(int i=1; i<=n; i++)
	{
		int x; cin>>x;
		if(c1*x<c2) a+=c1*x;
		else a+=c2;
	}
	for(int i=1; i<=m; i++)
	{
		int x; cin>>x;
		if(c1*x<c2) b+=c1*x;
		else b+=c2;
	}
	if(a>c3) a=c3;
	if(b>c3) b=c3;
	if(a+b<c4) ans=a+b;
	else ans=c4;
	cout<<ans<<endl;
}
