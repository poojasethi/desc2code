#include <iostream>

using namespace std;
int bu[1500],t[1500];
int main()
{
    int a,b,c,d;
    cin>>a>>b>>c>>d;
    int n,m;
    int sum=0,sum1=0;
    cin>>n>>m;
    for(int i = 0 ; i<n ; i++) {cin>>bu[i]; sum+=min(a*bu[i] ,b);}
    for(int i = 0 ; i<m ; i++) {cin>>t[i]; sum1+=min(a*t[i],b);}

    sum = min(sum,c);
    sum1=min(sum1,c);

    cout<<min(sum+sum1 , d)<<endl;



    return 0;
}
