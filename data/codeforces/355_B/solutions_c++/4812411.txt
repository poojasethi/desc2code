#include <iostream>
#include <cmath>
using namespace std;
int main()
{
	int c1,c2,c3,c4,n,m,ans1=0,ans2=0,a,b;
	cin>>c1>>c2>>c3>>c4>>n>>m;
	for(int i=1;i<=n;i++)
	{
		cin>>a;
		ans1+=min(a*c1,c2);
	}
	for(int i=1;i<=m;i++)
	{
		cin>>b;
		ans2+=min(b*c1,c2);
	}
	ans1=min(ans1,c3),ans2=min(ans2,c3);
	cout<<min(ans1+ans2,c4)<<endl;
	return 0;
}
