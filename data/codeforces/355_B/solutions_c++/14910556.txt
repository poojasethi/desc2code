#include <iostream>
#include <cmath>
using namespace std;

int main()
{
	int c1, c2, c3, c4, n, m, ans1 = 0, ans2 = 0;
	cin>> c1 >> c2 >> c3 >> c4;

	cin>> n >> m;
	for(int i = 0; i < n; i ++)
	{
		int a; cin>> a;
		ans1 += min(c1 * a, c2);
	}
	for(int i = 0; i < m; i ++)
	{
		int a; cin >> a;
		ans2 += min(c1 * a, c2);
	}

	ans1 = min(ans1, c3);
	ans2 = min(ans2, c3);

	cout<< min(c4, ans1 + ans2) << endl;
}
