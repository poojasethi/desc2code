#include <iostream>
using namespace std;
int main()
{
	int c1,c2,c3,c4,bs,tr,a;
	cin>>c1>>c2>>c3>>c4>>bs>>tr;
	int bus=0,troll=0,sum;
	for(int i=0;i!=bs;i++)
	{
		cin>>a;
		bus+=min(c1*a,c2);
	}
	bus=min(bus,c3);
	for(int i=0;i!=tr;i++)
	{
		cin>>a;
		troll=troll+min(c1*a,c2);
	}
	troll=min(troll,c3);
	sum=bus+troll;
	sum=min(sum,c4);
	cout<<sum<<endl;
	return 0;
}
