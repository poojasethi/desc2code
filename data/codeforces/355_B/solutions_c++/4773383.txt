#include <iostream>
using namespace std;
int main()
{
	int c1,c2,c3,c4,bs,tr,a;
	cin>>c1>>c2>>c3>>c4>>bs>>tr;
	int bssum=0,trsum=0,sum;
	for(int i=0;i!=bs;i++)
	{
		cin>>a;
		bssum=bssum+min(c1*a,c2);
	}
	bssum=min(bssum,c3);
	for(int i=0;i!=tr;i++)
	{
		cin>>a;
		trsum=trsum+min(c1*a,c2);
	}
	trsum=min(trsum,c3);
	sum=bssum+trsum;
	sum=min(sum,c4);
	cout<<sum<<endl;
}
