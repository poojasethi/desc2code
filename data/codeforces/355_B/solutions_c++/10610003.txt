#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int main(){
	int c1,c2,c3,c4,a,b;
	int n,m,bus=0,trolley=0;
	cin>>c1>>c2>>c3>>c4>>n>>m;
	for(int i=0;i<n;i++)
	{   
	    cin>>a;
		bus+=min(c2,c1*a);
	}
	bus=min(c3,bus);
	for(int i=0;i<m;i++)
	{   
	    cin>>b;
		trolley+=min(c2,c1*b);
	}
	trolley=min(c3,trolley);
	cout<<min(c4,bus+trolley);
	return 	0;
}
