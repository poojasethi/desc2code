#include <iostream>

using namespace std;

int main()
{
	int n, m, c1, c2, c3, c4, s1 = 0, s2 = 0, a;
	cin >> c1 >> c2 >> c3 >> c4 >> n >> m;
	for (int i = 0; i < n + m; i++)
	{
		cin >> a;
		a = min(a * c1, c2);
		if (i < n)
			s1 += a;
		else
			s2 += a;
	}
	cout << min(min(min((min(s1, s2) + c3), s1 + s2), 2 * c3), c4) << endl;
	return 0;
}
