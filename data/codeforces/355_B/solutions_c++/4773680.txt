#include<iostream>
#include<algorithm>
using namespace std;
int main()
{
    int c1,c2,c3,c4,n,m,x,tmp=0,ans=0;
    cin>>c1>>c2>>c3>>c4>>n>>m;
    for(int i=0;i<n;++i)
    {
        cin>>x;
        tmp+=min(c1*x,c2);
    }
    ans+=min(tmp,c3);
    tmp=0;
    for(int i=0;i<m;++i)
    {
        cin>>x;
        tmp+=min(c1*x,c2);
    }
    ans+=min(tmp,c3);
    ans=min(ans,c4);
    cout<<ans;
}
