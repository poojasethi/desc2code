#include<iostream>
using namespace std;
int main()
{
	int c[4],a=0,b=0,x,n,m;
	for(int i=0;i<4;i++)
		cin>>c[i];
  	cin>>n>>m;
  	for(int i=1;i<=n;i++)
  	{
  		cin>>x;
  		a+=min(c[0]*x,c[1]);
  	}
  	a=min(a,c[2]);
  	for(int i=1;i<=m;i++)
  	{
  		cin>>x;
  		b+=min(c[0]*x,c[1]);
  	}
  	b=min(b,c[2]);
  	cout<<min(a+b,c[3]);
  	return 0;
}
