#include <iostream>
using namespace std;
const int M=1e3+5;
int b[M],t[M],c1,c2,c3,c4,n,m,sumb,sumt;
int main()
{
    cin>>c1>>c2>>c3>>c4>>n>>m;
    for(int i=0;i<n;i++)
    {
        cin>>b[i];
        sumb+=min(c1*b[i],c2);
    }
    for(int i=0;i<m;i++)
    {
        cin>>t[i];
        sumt+=min(c1*t[i],c2);
    }
    cout<<min(c4,min(c3,sumb)+min(c3,sumt));
}
