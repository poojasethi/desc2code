#include<bits/stdc++.h>
using namespace std;
int main()
{
	int n,m,a[1010],b[1010],i,c1,c2,c3,c4,sum1=0,sum2=0;
	cin>>c1>>c2>>c3>>c4;
	cin>>n>>m;
	for(i=0;i<n;i++)
		cin>>a[i];
	for(i=0;i<m;i++)
		cin>>b[i];
	for(i=0;i<n;i++)
		sum1+=min(a[i]*c1,c2);
	sum1=min(sum1,c3);
	for(i=0;i<m;i++)
		sum2+=min(b[i]*c1,c2);
	sum2=min(sum2,c3);
	cout<<min(sum1+sum2,c4)<<endl;
	return 0;
}
