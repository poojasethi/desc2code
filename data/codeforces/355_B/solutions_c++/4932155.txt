#include <math.h>
#include <stdio.h>
#include <algorithm>
using namespace std;
int c1,c2,c3,c4,n,m,x,i,s1,s2;
int main(){
  scanf("%d %d %d %d",&c1,&c2,&c3,&c4);
  scanf("%d %d",&n,&m);
  for(i=1;i<=n;i++){
   scanf("%d",&x);
   s1+=min(c1*x,c2);
  }
  s1=min(s1,c3);
  for(i=1;i<=m;i++){
   scanf("%d",&x);
   s2+=min(c1*x,c2);
  }
  s2=min(s2,c3);
  printf("%d",min(c4,s1+s2));
 return 0;
}
