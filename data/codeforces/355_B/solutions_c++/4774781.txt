#include<iostream>
using namespace std;
#define rep(i,n) for(int i=0;i<(n);++i)
int main(){
  int c[4];
  rep(i,4)cin>>c[i];
  int n,m;
  cin>>n>>m;
  int a[n],b[m];
  int sumA=0,sumB=0,cost=0;
  rep(i,n){
    cin>>a[i];
    sumA+=min(a[i]*c[0],c[1]);
  }
  cost+=min(sumA,c[2]);
  rep(i,m){
    cin>>b[i];
    sumB+=min(b[i]*c[0],c[1]);
  }
  cost+=min(sumB,c[2]);
  cout<<min(cost,c[3])<<endl;
  return 0;
}
