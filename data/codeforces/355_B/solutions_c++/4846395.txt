#include <iostream>
#include <cmath>

using namespace std;

int n,m;
int c1,c2,c3,c4;

int main()
{
  cin>>c1>>c2>>c3>>c4>>n>>m;
  c4=min(c4,c3+c3);
  int s1=0,s2=0;
  for (int i=0;i<n;i++) {
    int a;
    cin>>a;
    s1+=min(a*c1,c2);
  }
  for (int i=0;i<m;i++) {
    int a;
    cin>>a;
    s2+=min(a*c1,c2);
  }
  cout<<min(min(s1+s2,c3+s2),min(s1+c3,c4))<<endl;
}
