	scanf("%d%d",&n,&m);
	for(m2=i=0;i<n;i++){
		scanf("%d",&x);
		m2+=min(x*c1,c2);

	for(m2=j=0;j<m;j++){
		scanf("%d",&x);
		m2+=min(x*c1,c2);
	}

}
