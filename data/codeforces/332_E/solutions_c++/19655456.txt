#include<bits/stdc++.h>
using namespace std;


vector<string> solve1(string s, int len)
{
    vector<string> ret(len, "");
    for(int i = 0; i < s.size(); i++)
        ret[i % len] += s[i];
    return ret;
}

string solve2(vector<string> K, vector<string> a, int k)
{
    string ans(k, '0');
    int j = (int)a.size() - 1;
    for(int i = k - 1; i >= 0; i--)
    {
        if(K[i] == a[j])
        {
            ans[i] = '1';
            j--;
        }
        if(j == -1) break;
    }
    if(j == -1) return ans;
    return "";
}

int main ()
{
    
    string p, s;
    int k;
    getline(cin, p);
    getline(cin, s);
    cin >> k;
    
    vector<string> K = solve1(p, k);
    string ans(k + 1, '1');
    for(int i = 1; i <= s.size(); i++)
    {
        vector<string> a = solve1(s, i);
        string ret = solve2(K, a, k);
        if(ret != "") ans = min(ans, ret);
    }
    if((int)ans.size() == k) cout << ans << endl;
    else cout << "0" << endl;
}
