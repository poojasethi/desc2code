#include <bits/stdc++.h>
#define maxn 1000005
using namespace std;

int N,M,K;
char A[maxn],B[205],p[205];
bool b[2005];

void init()
{
	gets(A+1),N=strlen(A+1);
	gets(B+1),M=strlen(B+1);
	scanf("%d",&K);
}

bool cmp(int l,int p,int q)
{
	for (int i=0;; i++)
	{
		int x=i*K+p,y=i*l+q;
		if (x>N&&y>M) return 1; 
		if (x>N||y>M||A[x]!=B[y]) return 0;
	}
}

bool check(int l)
{
	memset(b,0,sizeof(b));
	for (int i=l,j=K; i; i--,j--)
	{
		for (;j&&!cmp(l,j,i); j--);
		if (!j) return 0;
		b[j]=1;
	}
	return 1;
}

void doit()
{
	string ans="2";
	for (int i=1; i<=M&&i<=K; i++) if (check(i))
	{
		string s="";
		for (int j=1; j<=K; j++) s+=b[j]?"1":"0";
		ans=min(ans,s);
	}
	if (ans=="2") puts("0"); else puts(ans.c_str());
}

int main()
{
	init();
	doit();
	return 0;
}
