#include<iostream>
#include<cstdio>
#include<cstring>
#include<string>
using namespace std;
int n, m, t;
char a[1000100], b[1000];
string ans = "9";
int main()
{
	fgets(a+1,1001000,stdin);
	n = strlen(a+1);
	fgets(b+1,1000,stdin);
	m = strlen(b+1);
	if(a[n]=='\n')
		n--;
	if(b[m]=='\n')
		m--;
	scanf("%d", &t);
	for(int i = 1; i <= t && i <= m; i++)
	{
		string s = "0";
		for(int j = 1; j <= t; j++)
			s += '0';
		int cnt = i;
		for(int j = t; j >= 1 && cnt > 0; j--)
		{
			int t1 = j, t2 = cnt;
			bool flag = false;
			while(true)
			{
				if(a[t1] != b[t2])
					break;
				t1 += t;
				t2 += i;
				if(t1 > n && t2 > m)
				{
					flag = true;
					break;
				}
				if(t1 > n || t2 > m)
					break;
			}
			if(flag)
			{
				cnt--;
				s[j] = '1';
			}
		}
		if(cnt == 0)
			ans = min(ans, s);
	}
	if(ans[0] != '9')
	{
		for(int i = 1; i <= t; i++)
			printf("%c", ans[i]);
		printf("\n");
	}
	else
		printf("0\n");
	return 0;
}
