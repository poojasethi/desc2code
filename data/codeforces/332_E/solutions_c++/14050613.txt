#include<cstring>
#include<cstdlib>
#include<algorithm>
#include<cstdio>
using namespace std;
int i,n,j,m,la,lb,k,bo,v[2200],ans[2200],x,l;
char a[1200000],b[1200];
bool check(){
	for(int i=1;i<=k;++i){
		if(v[i]<ans[i])return 1;
		if(v[i]>ans[i])return 0;
	}
	return 0;
}
int main(){
	gets(a+1);
	gets(b+1);
	scanf("%d",&k);
	la=strlen(a+1);
	lb=strlen(b+1);
	m=la/k;
	for(i=0;i<=lb;++i)if(i*m<=lb && i*(m+1)>=lb){
		x=i;
		memset(v,0,sizeof(v));
		for(j=k;j>=1 && x;--j){
			for(l=0;l<=m;++l){
				if(l*k+j>la && l*i+x>lb)continue;
				if(a[l*k+j]!=b[l*i+x])break;
			}
			if(l<=m)continue;
			x--;
			v[j]=1;
		}
		if(!x){
			if(!bo || check()){
				memcpy(ans,v,sizeof(v));
				bo=1;
			}
		}
	}
	if(!bo)printf("0");
	else for(i=1;i<=k;++i)printf("%d",ans[i]);
}
