#include <bits/stdc++.h>
using namespace std;

const int N = 1000010;
char s[N],str[N];
int ans[2010];
int re[2010];
void update(int k)
{
    for(int i = 0; i < k; i ++) 
        if(re[i] > ans[i]) return; 
        else if(re[i] < ans[i]) break;
    for(int i = 0; i < k; i ++) ans[i] = re[i];
}

    
int main()
{
    int n,m,k;
    gets(s);
    gets(str);
    cin >> k;
    n = strlen(s);
    m = strlen(str);
    fill(ans,ans + k + 1,0x7fffffff);
    int x = n / k;
    for(int i = min(m,k); i >= 1; i --) {
        if(!(i * x <= m && i * (x + 1) >= m)) continue;
        //cout << "enter : " << i << endl;
        memset(re,0,sizeof(re));
        int one = i;
        for(int j = k - 1; j >= 0 && one; j --) {
            bool flag = true;
            for(int r = 0; r <= x; r ++) {
                if(r * k + j >= n && r * i + one - 1 >= m) continue;
                if(s[r * k + j] != str[r * i + one - 1]) {
                    flag = false;
                    break;
                }
            }
            if(!flag) continue;
            one --;
            re[j] = 1;
        }
        if(one) continue;
        update(k);
    }
    if(ans[0] > 1) cout << 0 << endl;
    else {
        for(int i = 0; i < k; i ++) cout << ans[i];
        cout << endl;
    }
    return 0;
}
