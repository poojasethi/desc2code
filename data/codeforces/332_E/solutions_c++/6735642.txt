#include <iostream>
#include <fstream>
#include <vector>
#include <string>
#include <cstring>
using namespace std;

#define F cin
#define G cout

const int N = 1000005;
const int K = 2005;

char p[N],s[N];
int ss,sp,k;
vector<char> a[K],b[K];
string ans = "$";

int main()
{
    #ifndef ONLINE_JUDGE
        ifstream F("p.in");
    #endif

    F.getline(p,N), sp = strlen(p);
    F.getline(s,N), ss = strlen(s);
    F>>k;
    for (int i=0;i<k;++i)
        for (int j=i;j<sp;j+=k)
            a[i].push_back(p[j]);
    for (int i=1;i<=k;++i)
    {
        for (int j=0;j<k;++j) b[j].clear();
        for (int j=0;j<ss;++j)
            b[j%i].push_back(s[j]);
        string now(k,'0');
        int wh = i-1;
        for (int j=k-1;j>=0 && wh>=0;--j)
            if ( a[j] == b[wh] )
            {
                now[j] = '1';
                wh--;
            }
        if ( wh < 0 && ( ans == "$" || now < ans ) )
            ans = now;
    }
    if ( ans == "$" )
        G<<"0\n";
    else
        G<<ans<<'\n';
}
