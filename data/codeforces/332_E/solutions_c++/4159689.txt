#include <cstdio>
#include <cstring>
#include <string>
using std::string;

char buf[1000010];
char s[210];
char key[2010];
char temp[2010];
int n,R,x,lenb,lens;

string div[210];

bool solve(int ones)
{
    for(int i=0;i<ones;++i){
        div[i]="";
        for(int j=i;j<lens;j+=ones) div[i] += s[j];
    }

    int i = n;
    int j = ones-1;
    while(i--){
        string t = "";
        for(int k=i;k<lenb;k+=n) t+=buf[k];

        if(j>=0 && t==div[j]){
            temp[i]='1';
            j--;
        }
        else temp[i]='0';
    }

    return j<0;
}

int main()
{
    gets(buf);
    gets(s);
    scanf("%d",&n);

    lenb = strlen(buf);
    lens = strlen(s);

    R = lenb / n; 
    x = lenb % n;

    bool ok=false;
    for(int i=1;i<=lens;++i)for(int j=0;j<=i && j<=x;++j)
        if( R*i + j == lens  && solve(i)){
            if(ok==false){ok=true; strcpy(key,temp);}
            else if(strcmp(temp,key)<0) strcpy(key,temp);
        }
    
    puts( ok ? key : "0");

    return 0;
}
