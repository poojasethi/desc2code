#include<stdio.h>
#include<string.h>
#include<algorithm>
#define N 1100000
using namespace std;
typedef long long ll;
char p[N],s[N];
int n,m,k,l,ans[N],now[N];
bool check(int x,int y)
{
	if(!y)	return 0;
	int i,j;
	for(i=x,j=y-1;i<n;i+=k,j+=l)
	{
		if(p[i]!=s[j])
			return 0;
	}
	if(j<m)
		return 0;
	return 1;
}
bool cmp()
{
	for(int i=0;i<k;i++)
	{
		if(now[i]!=ans[i])
		{
			return ans[i]>now[i];
		}
	}
	return 0;
}
int main()
{
	#ifndef ONLINE_JUDGE
	freopen("tt.in","r",stdin);
//	freopen("mine.out","w",stdout);
	#endif
	int i,j;
	bool flag;
	gets(p);
	gets(s);
	scanf("%d",&k);
	n=strlen(p),m=strlen(s);
	flag=0;
	for(l=1;l<=k;l++)
	{
		for(i=k-1,j=l;i>=0;i--)
		{
			if(check(i,j))
				now[i]=1,j--;
			else
				now[i]=0;
		}
		if(j==0)
		{
			if(!flag||cmp())
			{
				for(i=0;i<k;i++)
					ans[i]=now[i];
			}
			flag=1;
		}
	}
	if(!flag)
		puts("0");
	else
	{
		for(i=0;i<k;i++)
			printf("%d",ans[i]);
	}
	return 0;
}
