#include<iostream>
#include<algorithm>
#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
#define N 1200000
int n=0,m=0,k,ans[N],q[N];
char p[N],s[N];
bool pf(int t,int z,int i,int k){
     if(t>m && i>n)return 1;
     if(t>m || i>n)return 0;
     if(s[t]!=p[i])return 0;
     return pf(t+z,z,i+k,k);
}
bool check(int z){
     for(int i=1;i<=k;i++)q[i]=0;
     int t=z;
     for(int i=k;i;i--)
       if(pf(t,z,i,k)){
         t--;
         q[i]=1;
         if(t==0)return 1;
         }
     return 0;
}
bool work(){
     for(int i=1;i<=k;i++){
       if(q[i]<ans[i])return 1;
       if(q[i]>ans[i])return 0;
       }
     return 0;
}
int main(){
    char ch=getchar();
    while(ch!='\n')p[++n]=ch,ch=getchar();
    ch=getchar();
    while(ch!='\n')s[++m]=ch,ch=getchar();
    cin>>k;
    bool pd=0;
    for(int i=1;i<=m;i++)
        if(check(i))
          if(!pd || work()){
            for(int j=1;j<=k;j++)ans[j]=q[j];
            pd=1;
          }
    if(pd){
      for(int i=1;i<=k;i++)printf("%d",ans[i]);
    }
    else printf("0");
    return 0;
}
