#include<iostream>
#include<cstdio>
#include<cstring>
#include<vector>
#include<algorithm>
#include<string>
using namespace std;
char p[1000005],s[205];
int n,m,k,l1,l2;
int main() {
    gets(p);
    gets(s);
    scanf("%d",&k);
    n=strlen(p);
    m=strlen(s);
    string ans;
    for (int i=0; i<=k; i++) ans+="1";
    int rboud=min(m,k), cntIn=n/k, R=n%k;
    for (int d=1; d<=rboud && cntIn*d<=m; d++) {
        string cans;
        for (int i=0; i<k; i++) cans+="0";
        int i,r;
        for (i=min(n-1,k-1),r=d-1; i>=0 && r>=0; i--) {
            l1=m/d+(r<m%d);
            l2=cntIn+(i<R);
            if (l1==l2) {
                int e;
                for (e=0; e<l1 && s[r+e*d]==p[i+e*k]; e++);
                if (e==l1) cans[i]='1',r--;
            }
        }
        if (r==-1) ans=min(ans,cans);
    }
    if ((int)ans.size()>k) cout<<0<<endl;
    else cout<<ans<<endl;
    return 0;
}
