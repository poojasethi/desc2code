char p[1010000],s[210],v[2010];
int cnt[2010],ct[2010],l,blk,lp,ls;

int main() {
	gets(p);
	gets(s);
	scanf("%d",&l);
	lp=strlen(p); ls=strlen(s);
	blk=lp/l;
	ret="0";
	rep(i,0,l) cnt[i]=0;
	rep(i,0,lp) cnt[i%l]++;
	rep(i,0,ls+1) if (blk*i<=ls&&(blk+1)*i>=ls) {
		int cr=l;
		bool fg=1;
		rep(k,0,l) v[k]='0';
		rep(k,0,i) ct[k]=0;
		rep(k,0,ls) ct[k%i]++;
		per(j,0,i) {
			bool gf=0;
			per(k,0,cr) {
				if (ct[j]!=cnt[k]) continue;
				bool ff=1;
				for (int p1=j,p2=k;p1<ls&&p2<lp;p1+=i,p2+=l) if (s[p1]!=p[p2]) {
					ff=0; break;
				}
				if (ff) {
					gf=1; v[k]='1';
					cr=k; break;
				}
			}
			if (!gf) { fg=0; break;}
		}
		if (fg) {
			string tmp(v,v+l);
			if (ret=="0"||tmp<ret) ret=tmp;
		}
