using namespace std;
typedef long long LL;
typedef double ld;
typedef pair<ld,ld> pii;
const int NMax=1001000,MMax=500;
char A[NMax],B[MMax];
int K,N,M;
int main()	
{
	fgets(A,NMax,stdin);
	N=strlen(A);
	fgets(B,NMax,stdin);
	M=strlen(B);
	if(A[N-1]=='\n') N--;
	if(B[M-1]=='\n') M--;
	scanf("%d",&K);
	string ret="9";
	for(int i=1;i<=K;i++) {
		int p1=i-1;
		string s="";
		for(int j=0;j<K;j++) s.pb('0');
		for(int j=K;j>=1&&p1>=0;j--) {
			int p=j-1,q=p1,flag=1;
			while(1) {
				if(B[q]!=A[p]) {
					flag=0;
					break;
				}
				p+=K;q+=i;
				if((q>=M)+(p>=N)==1) {
					flag=0;
					break;
				}else if((q>=M)+(p>=N)==2) break;
			}
			if(flag) {
				s[j-1]='1';
				p1--;
			}
		}
		if(p1<0)ret=min(ret,s);
	}
	if(ret=="9") puts("0");
	else cout <<ret<<endl;
	return 0;
}
