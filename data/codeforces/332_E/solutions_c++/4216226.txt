#include <iostream>
#include <cstdio>
#include <algorithm>
#include <cstring>
#include <string>

using namespace std;

const int MAXN = int(1e6)+100;
char  s[MAXN], t[MAXN];
int   K;
string  ss[2100], tt[2100];

int main()
{
    while (gets(s) && gets(t) && ~scanf("%d", &K))
    {
        for (int i = 0; i < K; ++i) ss[i] = "";
        for (int i = 0; s[i]; ++i) ss[i%K] += s[i];
        
        string ans = "2";
        for (int l = 1, i, j; l <= K; ++l)
        {
            string tans = string(K, '0');
            for (i = 0; i < l; ++i) tt[i] = "";
            for (i = 0; t[i]; ++i) tt[i%l] += t[i];

            for (i = K-1, j = l-1; i>=0 && j >= 0; --i, --j)
            {
                for (; i >= 0 && tt[j] != ss[i]; --i);
                if (i == -1) break;
                tans[i] = '1';
            }
            
            if (j >= 0) continue;
            if (ans > tans) ans = tans;
        }
        
        printf("%s\n", ans=="2" ? "0" : ans.c_str());
    }
    return 0;
}
