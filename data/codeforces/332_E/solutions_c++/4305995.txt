#include <algorithm>
#include <cstring>
#include <cstdlib>
#include <cstdio>
#define For(i,x,y) for (i=x;i<=y;i++)
using namespace std;
int i,j,k,n,m,len,kuai;
int an[2010],re[2010],ff[2010];
char p[1000010],s[210];
inline void work(int x) {
    int i,j,u,last=k;
    For(i,1,k) re[i]=0;
    for (i=x;i;i--) {
        For(j,1,last) ff[j]=0;
        int sum=0;
        for (j=i;j<=m;j+=x) {
            sum++;
            int l=(sum-1)*k+1,r=min(sum*k,min(l+last-1,n));
            For(u,l,r) if (p[u]==s[j]) ff[u-l+1]++;
        }
        int L=sum>kuai?0:n%k,ans=0;
        for (j=last;j>L;j--) if (ff[j]==sum) {
            ans=j; break;
        }
        if (!ans) return;
        re[ans]=1; last=ans-1;
    }
    For(i,0,k) {
        if (an[i]>re[i]) break;
        if (an[i]<re[i]) return;
    }
    For(i,0,k) an[i]=re[i];
}
int main() {
    gets(p+1); gets(s+1); scanf("%d",&len);
    n=strlen(p+1),m=strlen(s+1); k=min(len,n);
    an[0]=1; kuai=n/k;
    For(i,0,m) if ((m-i)%kuai==0) {
        if (n%k==0&&i) break;
        int A=(m-i)/kuai;
        if (A<i) break;
        work(A);
    }
    if (an[0]) printf("0");
    else For(i,1,len) printf("%d",an[i]);
    printf("\n"); return 0;
}
