I=lambda:map(int, raw_input().split())
x,y=set(),set()
z = set()
for _ in xrange(8):
    xi, yi = I()
    x.add(xi)
    y.add(yi)
    z.add((xi,yi))
    
if len(x) != 3 or len(y) != 3 or len(z) != 8: print 'ugly'; exit()
a, b = sorted(list(x))[1], sorted(list(y))[1]

if (a,b) in z: print 'ugly'; exit()
print 'respectable'
