#!/usr/bin/python

xs = []
ys = []
pts = []

for i in range(8):
	pts.append(map(int,raw_input().split()))

pts.sort()
for i in range(len(pts)-1):
	while i < len(pts)-1 and pts[i] == pts[i+1] :
		del pts[i+1]

for [x,y] in pts:
	xs.append(x)
	ys.append(y)

ys.sort()
xs.sort()
#print(xs)
#print(ys)

if ys.count(ys[0]) != 3 or \
		ys.count(ys[3]) != 2 or \
		ys.count(ys[5]) != 3:
			print("ugly")
			exit(0)

if xs.count(xs[0]) != 3 or \
		xs.count(xs[3]) != 2 or \
		xs.count(xs[5]) != 3:
			print("ugly")
			exit(0)

print("respectable")
