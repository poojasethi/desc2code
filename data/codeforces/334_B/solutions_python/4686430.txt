r=lambda:map(int, raw_input().split())
p=[r() for i in [0]*8]
z=zip(*p)
s=lambda x:sorted(list(set(x)))
if len(s(z[0])) != 3 or len(s(z[1])) != 3:
    print "ugly"
    exit() 
p.append([s(z[0])[1],s(z[1])[1]])
p.sort()
z=zip(*p)
if list(z[0])==[z[0][0]]*3+[z[0][3]]*3+[z[0][6]]*3 and list(z[1]) == list(z[1][:3])*3:
    print "respectable"
else:
    print "ugly"
